#!/usr/bin/python

import fileinput

def main():
    points = {}
    xs = {}
    ys = {}
    for line in fileinput.input():
        t = line.split()
        x, y = int(t[0]), int(t[1])
        xs[x] = 0
        ys[y] = 0
        points[(x, y)] = 0
    if len(xs) != 3 or len(ys) != 3:
        print "ugly"
        return
    xs = sorted(xs.keys())
    ys = sorted(ys.keys())
    num = 3
    for i in xrange(num):
        for j in xrange(num):
            if i == 1 and j == 1:
                continue
            if not points.has_key((xs[i], ys[j])):
                print "ugly"
                return
    print "respectable"

if __name__ == '__main__':
    main()
