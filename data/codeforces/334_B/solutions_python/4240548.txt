#!/usr/bin/env 

import sys

a = set()
xs = set()
ys = set()
for i in xrange(0, 8):
	x, y = map(long, sys.stdin.readline().split(" "))
	a.add((x,y))
	xs.add(x)
	ys.add(y)
xs = sorted(list(xs))
ys = sorted(list(ys))

def resp(a, xs, ys):
	if len(xs) == 3 and len(ys) == 3:
		b = set([(xs[i],ys[j]) for i in range(0,3) for j in range(0,3) if (i,j)!=(1,1)])
		if a==b:
			return 'respectable'
	return 'ugly'

print resp(a,xs,ys)
