import sys
x = []
y = []
points = []
for i in range(8):
	xi,yi = map(int,raw_input().split())
	x.append(xi)
	y.append(yi)
	points.append((xi,yi))
if len(set(x)) != 3 or len(set(y)) != 3:
	print "ugly"
	sys.exit()
else:
	newx = sorted(set(x))
	newy = sorted(set(y))
	for i in range(3):
		for j in range(3):
			if (newx[i],newy[j]) not in points:
				if not (i == 1 and j == 1):
					print "ugly"
					sys.exit()
	print "respectable"
