points, rows, cols = set(), set(), set()
for _ in range(8):
  a, b = map(int, raw_input().split())
  points.add((a, b))
  rows.add(a)
  cols.add(b)
if len(points) != 8 or len(rows) != 3 or len(cols) != 3:
  print 'ugly'
  exit()
rows, cols = sorted(rows), sorted(cols)
for r in rows:
  for c in cols:
    if r == rows[1] and c == cols[1]: continue
    if not (r, c) in points:
      print 'ugly'
      exit()
print 'respectable'
