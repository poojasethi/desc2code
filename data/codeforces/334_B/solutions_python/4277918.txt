#!/usr/bin/python
R = lambda: map(int, raw_input().split())
n = 8
x, y = [0]*n, [0]*n
for i in xrange(n):
	x[i], y[i] = R()

if len(set(x)) != 3:
	print "ugly"
	exit(0)
if len(set(y)) != 3:
	print "ugly"
	exit(0)

dct = {}
for i in xrange(n):
	if not x[i] in dct:
		dct[x[i]] = []
	dct[x[i]].append(y[i])

for i, l in dct.iteritems():
	l = sorted(l)
	dct[i] = l

x = sorted(set(x))

if len(set(dct[x[0]])) != 3:
	print "ugly"
	exit(0)

if len(set(dct[x[1]])) != 2:
	print "ugly"
	exit(0)

if len(set(dct[x[2]])) != 3:
	print "ugly"
	exit(0)

up = dct[x[0]]
dw = dct[x[2]]
for i, j in zip(up, dw):
	if i != j:
		print "ugly"
		exit(0)

md = dct[x[1]]
if up[0] != md[0] or up[2] != md[1]:
	print "ugly"
	exit(0)

print "respectable"
