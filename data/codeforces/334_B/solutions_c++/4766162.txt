#include<cstdio>
#include<set>
#include<utility>
using namespace std;

int main()
{
    set<int>  setx,sety;
    set<pair<int,int> >  myset;

    for(int i=0; i<8; i++)
    {
        int a,b;
        scanf("%d %d",&a,&b);
        setx.insert(a);
        sety.insert(b);
        myset.insert(make_pair(a,b));
    }
    set<int> ::iterator itx = setx.begin(), ity = sety.begin();
    if(setx.size() == 3 && sety.size() == 3 && myset.size() == 8 && myset.find(make_pair(*(++itx),*(++ity))) == myset.end() )
        printf("respectable\n");
    else
        printf("ugly\n");
}
