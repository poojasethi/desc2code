#include<iostream>
#include<set>
using namespace std;
int main(){
    set<int>sx,sy;
    int x[8],y[8],f=1,mx,my;
    for(int i=0;i<8;i++){
        cin>>x[i]>>y[i];
        for(int j=0;j<i;j++)
            f&=x[i]!=x[j]||y[i]!=y[j];
        sx.insert(x[i]);
        sy.insert(y[i]);
    }
    f&=sx.size()==3&&sy.size()==3;
    mx=*(++sx.begin());
    my=*(++sy.begin());
    for(int i=0;i<8;i++)
        f&=x[i]!=mx||y[i]!=my;
    cout<<(f?"respectable":"ugly");
}
