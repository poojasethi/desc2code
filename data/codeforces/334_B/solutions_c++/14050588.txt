#include <cstdio>
#include <set>
using namespace std;
set<int>A,B;
set<pair<int,int> >P;
int main()
{
    for(int i=0;i<8;i++)
        {int x,y; scanf("%d%d",&x,&y); A.insert(x),B.insert(y),P.insert(make_pair(x,y));}
    set<int>::iterator it=A.begin(),jt=B.begin();
    it++,jt++;
    printf(A.size()==3&&B.size()==3&&P.size()==8&&P.count(make_pair(*it,*jt))==0?"respectable":"ugly");
}
