#include<iostream>
#include<set>
#include<algorithm>
using namespace std;
int main()
{
	set<int> l,c;
	set<pair<int,int> > m;
	for(int i=0;i<8;i++)
	{
		int x,y;
		cin>>x>>y;
		l.insert(x);
		c.insert(y);
		m.insert(make_pair(x,y));
	}
	if(c.size()==3 && l.size()==3 && m.size()==8)
	{
		int x=*(++l.begin()),y=*(++c.begin());
		if(m.find(make_pair(x,y))==m.end())
		{
			cout<<"respectable"<<endl;
			return 0;
		}
	}
	cout<<"ugly"<<endl;
	return 0;
}
