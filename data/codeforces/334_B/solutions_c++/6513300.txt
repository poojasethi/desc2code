#include <iostream>
#include <utility>
#include <cstdio>
#include <set>

using namespace std;

int main()
{
    set<int> X,Y;
    set< pair<int,int> > ans;
    int x,y;
    for(int i=0;i<8;i++)
    {
        scanf("%d %d",&x,&y);
        X.insert(x);
        Y.insert(y);
        ans.insert(make_pair(x,y));
    }
    set<int>::iterator it=X.begin(),is=Y.begin();
    it++,is++;
    printf(X.size()==3&&Y.size()==3&&ans.size()==8&&ans.count(make_pair(*it,*is))==0?"respectable":"ugly");
    return 0;
}
