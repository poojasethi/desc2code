#include<iostream>
#include<set>
#include<utility>
using namespace std;
int main()
{
    set<int> stx,sty;
    set<pair<int,int> > st;
    for(int i=0;i<8;i++)
       {int a,b;
        cin>>a>>b;
        stx.insert(a);
        sty.insert(b);
        st.insert(make_pair(a,b));}
    if(stx.size()==3 and sty.size()==3 and st.size()==8)
      {int r=*(++stx.begin()),c=*(++sty.begin());
       if(st.find(make_pair(r,c))==st.end())cout<<"respectable";
       else cout<<"ugly";}
    else cout<<"ugly";
    return 0;
}
