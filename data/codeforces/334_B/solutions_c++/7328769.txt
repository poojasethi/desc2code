			}
		}

	if(a[1]==a[0]&&a[2]==a[1]&&a[4]==a[3]&&a[7]==a[6]&&a[5]==
	a[7]&&a[3]>a[0]&&a[6]>a[3]&&b[3]>b[2]&&b[5]>b[4]&&b[4]==b[3]&&b[0]==b[1]&&b[1]==b[2]&&b[6]==b[7]&&b[5]==b[6]&&b[3]!=b[1])
		cout<<"respectable"<<endl;

}
