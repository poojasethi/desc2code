#include <bits/stdc++.h>

using namespace std;

int main()
{
    set<int> x, y;
    set<pair<int, int> > p;
    for(int i=1; i<=8; i++) {int a, b;
     cin>>a>>b;
     x.insert(a);
     y.insert(b);
     p.insert(make_pair(a,b));
     }
     if(x.size()==3 and y.size()==3 and p.size()==8)
     {
         int px = *(++x.begin());
         int py = *(++y.begin());
         if(p.find(make_pair(px,py))==p.end()) {cout<<"respectable"<<endl; return 0;}
         else {cout<<"ugly"<<endl; return 0;}
     }
     cout<<"ugly"<<endl;
    return 0;
}
