#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <string>
#include <set>
using namespace std;
set<int>stx;
set<int>sty;
set<pair<int,int> > stp;
int main(){
	for (int i=0;i<8;i++){
		int x,y;
		cin>>x>>y;
		stx.insert(x);
		sty.insert(y);
		stp.insert(make_pair(x,y));
	}
	if (stx.size()==3 && sty.size()==3 && stp.size()==8){
		int mx=*(++stx.begin());
		int my=*(++sty.begin());
		if (stp.find(make_pair(mx,my))==stp.end()){
			cout<<"respectable";
			return 0;
		}
		else{
			cout<<"ugly";
			return 0;
		}
	}
	cout<<"ugly";
	return 0;
	
}
