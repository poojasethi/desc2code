#include <bits/stdc++.h>
using namespace std;


int main() {
    int x, y;
    set <int> px, py;
    set <pair <int, int> > p;
	for (int i = 0; i < 8; i++)
	{
		cin >> x >> y;
		px.insert(x);
		py.insert(y);
		p.insert(make_pair(x, y));
	}
	x = *++px.begin();
	y = *++py.begin();
	if (px.size() == 3 && py.size() == 3 && p.size() == 8 && p.find(make_pair(x, y)) == p.end())
		cout << "respectable";
	else
		cout << "ugly";
	return 0;
}
