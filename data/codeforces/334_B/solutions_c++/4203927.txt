#include <iostream>
#include <algorithm>
#include <set>

using namespace std;

set <int> x, y;
set <pair <int, int> > p;

int main()
{
    for (int i = 0 ; i < 8; i++){
        int q, w; cin >> q >> w;
        x.insert ( q );
        y.insert ( w);
        p.insert ( make_pair (q, w) );
    }
    if ( x.size () == 3 && y.size () == 3 && p.size () == 8){
        int cx = *(++x.begin()), cy = *(++y.begin());
        if ( p.find ( make_pair ( cx, cy) ) == p.end () ) cout << "respectable";
        else cout << "ugly";
    }
    else cout << "ugly";
    return 0;
}
