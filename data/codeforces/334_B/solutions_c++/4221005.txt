#include<iostream>
#include<set>
#include<string>
using namespace std;
int main()
{
    set<int>tx,ty;
    set<pair<int,int> >tp;
    int x,y;
    for(int i=0;i<8;i++)
    {
            cin>>x>>y;
            tx.insert(x);
            ty.insert(y);
            tp.insert(make_pair(x,y));
            }
     if(tx.size()==3 && ty.size()==3 && tp.size()==8)
     {
     int mx=*(++tx.begin()),my=*(++ty.begin());
     if(tp.find(make_pair(mx,my))==tp.end())
     cout<<"respectable";
     else
     cout<<"ugly";
     }
     else
     cout<<"ugly";
     return 0;}
