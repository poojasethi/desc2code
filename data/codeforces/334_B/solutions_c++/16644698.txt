#include <iostream>
#include <set>
using namespace std;
int x, y;
set <int> sx, sy;
set <pair <int, int> > sp;

int main() {
	for (int i = 0; i < 8; i++)
	{
		cin >> x >> y;
		sx.insert(x);
		sy.insert(y);
		sp.insert(make_pair(x, y));
	}
	x = *++sx.begin();
	y = *++sy.begin();
	if (sx.size() == 3 && sy.size() == 3 && sp.size() == 8 && sp.find(make_pair(x, y)) == sp.end())
		cout << "respectable";
	else
		cout << "ugly";
	return 0;
}
