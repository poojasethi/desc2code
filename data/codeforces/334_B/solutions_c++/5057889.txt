#include<iostream>
#include<set>
using namespace std;
set<int> sx,sy;
set<pair<int,int> > stp;
int main()
{
    for(int i=0;i<8;i++)
    {
        int a,b;
        cin>>a>>b;
        sx.insert(a);
        sy.insert(b);
        stp.insert(make_pair(a,b));
    }
    if (sx.size()==3 && sy.size()==3 && stp.size()==8)
    {
        int x=*(++sx.begin());
        int y=*(++sy.begin());
        if (stp.find(make_pair(x,y))==stp.end())
        {
            cout<<"respectable";
            return 0;
        }
    }
    cout<<"ugly";
    return 0;
}
