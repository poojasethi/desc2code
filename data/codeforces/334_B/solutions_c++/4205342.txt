#include<iostream>
#include"algorithm"
using namespace std;
int a[8],b[8];
int main(){
       for(int i=0;i<8;i++)
        cin>>a[i]>>b[i];
       for(int i=0;i<7;i++)
        for(int j=i+1;j<8;j++)
        if(a[i]==a[j]&&b[i]==b[j]){cout<<"ugly";return 0;} 
       sort(a,a+8);
       sort(b,b+8);
       if(a[0]==a[1]&&a[1]==a[2]&&a[2]<a[3]&&a[3]==a[4]&&a[4]<a[5]&&a[5]==a[6]&&a[6]==a[7]&&
       b[0]==b[1]&&b[1]==b[2]&&b[2]<b[3]&&b[3]==b[4]&&b[4]<b[5]&&b[5]==b[6]&&b[6]==b[7])
        cout<<"respectable";
       else cout<<"ugly";
       return 0;
     //  system("pause");  
       }
