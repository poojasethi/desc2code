#include<iostream>
#include<cstdio>
#include<set>
using namespace std;
typedef pair<int,int> P;
set<P> a;
set<int> b,c;
set<int>:: iterator ite,jte;
int  main()
{
    int x,y;
    for(int i=0;i<8;i++){
       scanf("%d%d",&x,&y);
       a.insert(P(x,y));
       b.insert(x);
       c.insert(y);
    }
    ite=b.begin();jte=c.begin();
    ite++;jte++;
    if(b.size()==3&&c.size()==3&&a.size()==8&&a.find(P(*ite,*jte))==a.end()){
           printf("respectable\n");
    }else printf("ugly\n");
    return 0;
}
