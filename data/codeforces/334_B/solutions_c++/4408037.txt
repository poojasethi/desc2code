#include <iostream>
#include <cstdio>
#include <set>
using namespace std;
set<int>a,b;
set<pair<int,int > >p;
int main ()
{
  for (int i=0;i<8;i++)
  {
    int x,y;
    scanf("%d %d",&x,&y);
    a.insert(x),b.insert(y),p.insert(make_pair(x,y));
  }  
  set<int> ::iterator it=a.begin(),jt=b.begin();
  it++,jt++;
  printf(a.size()==3&&b.size()==3&&p.size()==8&&p.find(make_pair(*it,*jt))==p.end()?"respectable":"ugly" );
  return 0;
}
