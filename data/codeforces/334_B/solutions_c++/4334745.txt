#include<iostream>
#include<set>
using namespace std;
int main()
{
	set<int> sx,sy;
	set<pair<int,int> > sp;
	for(int i=0;i<8;i++)
	{
		int x,y;
		cin>>x>>y;
		sx.insert(x);
		sy.insert(y);
		sp.insert(make_pair(x,y));
	}
	if(sx.size()==3 && sy.size()==3 && sp.size()==8)
	{
		int mx=*(++sx.begin()),my=*(++sy.begin());
		if(sp.find(make_pair(mx,my))==sp.end())
		{
			cout<<"respectable"<<endl;
		}
		else
		{
			cout<<"ugly"<<endl;
		}
	}
	else
	{
		cout<<"ugly";
	}
	return 0;
}
