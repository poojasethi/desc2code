#include<iostream>
using namespace std;
int main()
{
int n,m,i,j,s=0,k,flag=0,count=0;
cin>>n>>m;
char a[n][m];
for(i=0;i<n;i++)
for(j=0;j<m;j++)
 cin>>a[i][j];
for(i=0;i<n;i++)
{
  s=0;
 for(j=0;j<m;j++)
  {
      flag=1;
  for(k=0;k<n;k++)
  if(a[k][j]>a[i][j])
   flag=0;
   if(flag==1)
    s=1;
  }
  count+=s;
}
cout<<count;
}
