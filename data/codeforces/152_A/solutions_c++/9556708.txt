#include<stdio.h>
int a[105];
char s[105][105];
int main()
{
int n,m;
scanf("%d%d",&n,&m);
int i,j;

for(i=0;i<n;i++)
scanf("%s",&s[i]);

for(i=0;i<m;i++)
for(j=0;j<n;j++)
if(s[j][i] > a[i])
a[i] = s[j][i];

int ans = 0;
for(i=0;i<n;i++)
{
int found = 0;
for(j=0;j<m;j++)
if(s[i][j] == a[j])found = 1;
ans+=found;
}

printf("%d\n",ans);


return 0;
}
