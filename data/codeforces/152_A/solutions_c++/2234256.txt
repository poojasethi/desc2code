#include<iostream>
using namespace std;
int main()
{
int n,m,i,j,k,san=0,maxi;
cin>>n>>m;
char A[n][m];
for (i=0; i<n; i++)
for (j=0; j<m; j++)
cin>>A[i][j];
for (k=0; k<n; k++)
{  for (i=0; i<m; i++)
{  maxi=0; 
for (j=0; j<n; j++)
if (maxi<(A[j][i]-48)) maxi=A[j][i]-48;
if (san == n) break;
if (A[k][i]-48 == maxi) 
{ san++;
  goto m1; }  }
m1:;  }
cout<<san;
return 0;
}
