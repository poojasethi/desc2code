#include<cstdio>
const int N=101;
int ans,a[N][N],best[N];
int main()
{
	int i,j,n,m,s;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			scanf("%1d",&a[i][j]);
			if(a[i][j]>best[j])
				best[j]=a[i][j];
		}
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
			if(a[i][j]==best[j])
			{
				ans++;
				break;
			}
	}
	printf("%d\n",ans);
	return 0;
}
