#include <cstdio>

char O[100][101], N[100];
int m, n, s;

int main() {
	scanf("%d %d", &n, &m);
	for (int i = 0; i < n; i++) {
		scanf("%s", O[i]);
		for (int j = 0; j < m; j++) {
			if (O[i][j] > N[j]) N[j] = O[i][j];
		}
	}
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			if (O[i][j] == N[j]) {
				s++;
				break;
			}
		}
	}
	printf("%d\n", s);
	return 0;
}
