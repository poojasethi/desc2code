
#include <iostream>
using namespace std;
int n,m,i,j,k;
int main()
{
	char a[101][101],b[101];
	cin>>n>>m;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
		{
			cin>>a[i][j];
			if(a[i][j]>b[j])
			  b[j]=a[i][j];
		}
		for(i=0;i<n;i++)
			for(j=0;j<m;j++)
			if(a[i][j]==b[j])
			{
				k++;
				break;
			}
			cout<<k<<endl;
			return 0;
}
