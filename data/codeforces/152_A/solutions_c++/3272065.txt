#include<cstdio>
char s[100][100],a[100];
bool v[100];
int main(){
	int n,m,i,j,ans;
	while(~scanf("%d%d",&n,&m)){
		for(i=0;i<n;i++)v[i]=0,scanf("%s",s[i]);
		for(i=0;i<m;i++){
			a[i]='0';
			for(j=0;j<n;j++)if(s[j][i]>a[i])a[i]=s[j][i];
		}
		ans=0;
		for(i=0;i<n;i++){
			for(j=0;j<m;j++)if(s[i][j]==a[j]){ans++;break;}
		}
		printf("%d\n",ans);
	}
	return 0;
}
