#include<iostream>
#include<cstdio>
using namespace std;
int main(){
	char a[100][101],x[101];
	int i,n,m,j,s;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++){
		scanf("%s",a[i]);
	}
	for(i=0;i<m;i++){
		x[i]='0';
		for(j=0;j<n;j++){
			if(x[i]<a[j][i])
				x[i]=a[j][i];
		}
	}
	s=0;
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			if(x[j]==a[i][j]){
				s=s+1;
				break;
			}
		}
	}
	cout<<s<<"\n";
	return(0);
}
