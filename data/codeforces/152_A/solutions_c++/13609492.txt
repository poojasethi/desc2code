#include <string>
#include <iostream>
#include <algorithm>
using namespace std;
int main(){
	int n,m;cin>>n>>m;
	string ar[100];
	int maxi[100]={0,};
	for(int i=0;i<n;++i){
		cin>>ar[i];
		for(int j=0;j<m;++j)
			maxi[j]= max(maxi[j],ar[i][j]-'0');
	}
	int ret= 0;
	for(int i=0;i<n;++i)
		for(int j=0;j<m;++j)
			if(maxi[j]==ar[i][j]-'0'){++ret;break;}
	cout<<ret<<endl;
}
