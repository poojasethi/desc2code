#include<bits/stdc++.h>
using namespace std;
int main(){
int x,y,i,j,k,c=0;
cin>>x>>y;
string arr[x];
for(i=0;i<x;i++){
    cin>>arr[i];
}
for(i=0;i<x;i++){
    for(j=0;j<y;j++){
        for(k=0;k<x;k++){
            if(arr[i][j]<arr[k][j])
                break;
        }
        if(k==x){
            c++;
            break;
        }
    }
}
cout<<c;
return 0;
}
