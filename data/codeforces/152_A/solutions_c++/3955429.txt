#include<stdio.h>
int main(){
int k=0,f=0,x=0,n,m,l=0,o=0,c=0;
char j[100][101];
char max[100];
scanf("%d%d",&n,&m);
for(f=0;f<n;f++){
scanf("%s",j[f]);
}
for(x=0;x<n;x++){
for(k=0;k<m;k++){
if(j[x][k]-'0'>max[k]-'0')
max[k]=j[x][k];
}
}
for(l=0;l<n;l++){
for(o=0;o<m;o++){
if(j[l][o]-'0'==max[o]-'0'){
c++;
break;
}
}
}
printf("%d",c);
return 0;
}
