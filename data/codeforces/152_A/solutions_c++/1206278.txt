#include <iostream>
#include <set>
#include <string>
using namespace std;

int main()
{
	int n, m;
	cin >> n >> m;
	string s[n];
	for (int i=0; i<n; i++)
		cin >> s[i];
	set<int> be;
	for (int x=0; x<m; x++)
	{
		int MAX = s[0][x]-'0';
		for (int i=1; i<n; i++)
			MAX = max(MAX, s[i][x]-'0');
		for (int i=0; i<n; i++)
			if (MAX == s[i][x]-'0')
				be.insert(i);
	}

	cout << be.size() << endl;
}
