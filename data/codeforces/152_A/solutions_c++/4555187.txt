#include<iostream>
using namespace std;
int main()
{ int i,j,k,n,m,max,c=0;
  char a[100][101];
  cin>>n>>m;
  for(i=0;i<n;i++)
    cin>>a[i];
  for(k=0;k<n;k++)
     for(j=0;j<m;j++)
       { max=a[0][j];
         for(i=0;i<n;i++)
            if(a[i][j]>max)
               max=a[i][j];
          if(a[k][j]==max)
            {c++;break;}
       }

    cout<<c;
  return 0;
}
