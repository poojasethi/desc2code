#include<cstdio>
char s[105][105];
int main()
{
	int n,m,ans=0;
	scanf("%d%d",&n,&m); gets(s[0]);
	for (int i=0;i<n;i++) gets(s[i]);
	for (int i=0;i<n;i++){
		bool flag=0;
		for (int j=0;j<m;j++){
			bool bo=0;
			for (int k=0;k<n;k++) if (i!=k)
				if (s[k][j]>s[i][j]) bo=1;
			if (!bo) flag=1;
			}
		ans+=flag;
		}
	printf("%d\n",ans); return 0;
}
