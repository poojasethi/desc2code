#include<iostream>
using namespace std;
int main()
{
	int res  = 0, n , m;
	bool k =0;
	string a[100];
	cin>>n>>m;
	for(int y = 0 ; y < n ; y++) cin>>a[y];
	for(int y = 0 ; y < n ; y++)
	{

		k = 0;
		for(int i = 0 ; i < m ; i++)
			if(!k)
			for(int r = 0 ; r < n ; r++)
			{
				if(a[y][i] < a[r][i])break;
				else if(r==n-1){k =1; res++;}
			}
			else break;
	}
	cout<<res;
}
