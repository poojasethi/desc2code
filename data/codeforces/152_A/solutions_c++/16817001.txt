#include <iostream>
using namespace std;
int n,m,ans,i,j;
char a[111][111],b[111];
int main()
{
	cin>>n>>m;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
		{
			cin>>a[i][j];
			if (a[i][j]>b[j])
				b[j]=a[i][j];
		}
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			if (a[i][j]==b[j])
			{
				ans++;
				break;
			}
	cout<<ans<<endl;
	return 0;
}
