n, m = map(int, raw_input().split())
st = [raw_input() for i in range(n)]
best = [0 for i in range(n)]
subj = [[int(st[i][j]) for i in range(n)] for j in range(m)]
for i in range(n):
    for j in range(m):
       if subj[j][i] == max(subj[j]):
          best[i] = 1
          break
print sum(best)
