n, m = map(int, raw_input().split())
maxScore = ['0' for i in range(m)]
score = []
for i in range(n):
	score.append(raw_input())
	for j in range(m):
		maxScore[j] = max(maxScore[j], score[i][j])
ans = 0
for i in range(n):
	for j in range(m):
		if maxScore[j] == score[i][j]:
			ans += 1
			break
print ans
