n, m = map(int, raw_input().split())
col = ['0']*m
r = []
for i in xrange(n):
    s = raw_input()
    r.append(s)
    for j in xrange(m):
        col[j] = max(col[j], s[j])

ans = 0
for i in xrange(n):
    for j in xrange(m):
        if r[i][j] == col[j]: 
            ans += 1
            break
print ans
