n, m = map(int, raw_input().split())
best = [0] * m
a = [0] * n
total = 0
for i in xrange(n):
    a[i] = raw_input()
    for j in xrange(m):
	if int(a[i][j]) >= best[j]:
	    best[j] = int(a[i][j])
for i in xrange(n):
    for j in xrange(m):
	if int(a[i][j]) == best[j]:
	    total += 1
	    break
print total
