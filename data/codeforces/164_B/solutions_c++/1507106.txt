#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int MaxN = 1000005;
int a[MaxN*2],b[MaxN*2],cb[MaxN];
int l[MaxN*2];

int check(int i, int j, int k)
{
    if(i==j)return 1;
    if(i==k)return 0;
    if(i<j)return j<k || k<i;
    else return j<k && k<i;
}

int main()
{
    int n,m;scanf("%d%d",&n,&m);
    for(int i = 0; i < n; i++)
    {
        scanf("%d",&a[i]);
        a[i+n] = a[i];
    }
    for(int i = 0; i < m; i++)
    {
        scanf("%d",&b[i]);
        cb[b[i]] = i+1;
    }

    int ret = 0;
    for(int i = 0; i < 2*n; i++)
    {
        if(!cb[a[i]])l[i] = 0;
        else if(i && l[i-1])
        {
            int f = i-l[i-1];
            while(!check(cb[a[f]],cb[a[i-1]],cb[a[i]]))f++;
            l[i] = i-f+1;
        }
        else l[i]=1;
        ret = max(ret,l[i]);
    }
    printf("%d\n",min(ret,n));


    return 0;
}
