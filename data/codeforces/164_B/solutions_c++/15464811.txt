#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
#include<vector>
#include<cmath>
#include<map>
using namespace std;
typedef long long ll;
const int N=2e6+10;
int a[N],b[N],c[N];
int main()
{
    int i,j,n,m,sum;
    cin>>n>>m;memset(b,0,sizeof b);
    for(i=0;i<n;i++)scanf("%d",a+i);
    for(i=0;i<n;i++)a[i+n]=a[i];
    for(i=1;i<=m;i++){int x;scanf("%d",&x);b[x]=i;}
    for(i=0;i<2*n-1;i++)
    if(b[a[i]]&&b[a[i+1]])c[i+1]=(b[a[i+1]]-b[a[i]]+m)%m;
    else c[i+1]=m;
    int f=1;
    for(i=0;i<2*n;i++)
    if(b[a[i]]){f=0;break;}
    if(f){printf("0");return 0;}
    int ans=0;
    for(i=1,j=sum=0;i<=n;i++)
    {
        while(j<2*n&&sum+c[j+1]<m)sum+=c[++j];
        ans=max(j-i+2,ans);
        sum-=c[i];
    }
    printf("%d",min(ans,min(n,m)));
    return 0;
}
