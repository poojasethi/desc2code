#include <cstdio>
#include <iostream>
#include <vector>
#include <string>
#include <cstring>
#include <algorithm>
#include <set>

using namespace std;

long long d[3000000];
int a[3000000];
int b[3000000];
int p[3000000];

int main() {
//	freopen ("input.txt", "r", stdin);
//	freopen ("output.txt", "w", stdout);
    int n,m;
    scanf("%d%d", &n, &m);
    for (int i=0; i<n; i++) {
        scanf("%d", a+i);
        a[n+i] = a[i];
    }
    memset(p, -1, sizeof(p));
    for (int i=0; i<m; i++) {
        int x;
        scanf("%d", &x);
        p[x] = i;
    }
    int c = 0;
    long long q = 0;
    int ans = 0;
    for (int i=0; i<2*n; i++) {
        if (p[a[i]]==-1)
            c = i+1;
        else {
            if (q%m < p[a[i]])
                q = (q/m)*m + p[a[i]];
            else
                q = (q/m+1)*m + p[a[i]];
        }
        d[i] = q;
        while (c<i && d[i]-d[c]>=m)
            c++;
        if (i-c+1 > ans)
            ans = i-c+1;
    }
    printf("%d\n", ans);
    return 0;
}
