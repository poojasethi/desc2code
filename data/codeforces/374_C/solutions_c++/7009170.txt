#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <vector>
using namespace std;
const int INF=1e9;
int n,m,f[1111][1111],ans;
char s[1111][1111],next[11]="DIMA";
int dfs(int x,int y,int z){
    int t=0;
    if(f[x][y]!=-1) return f[x][y];
    f[x][y]=INF;
    z=(z+1)%4;
    if(s[x+1][y]==next[z]) t=max(t,dfs(x+1,y,z));
    if(x>0&&s[x-1][y]==next[z]) t=max(t,dfs(x-1,y,z));
    if(s[x][y+1]==next[z]) t=max(t,dfs(x,y+1,z));
    if(y>0&&s[x][y-1]==next[z]) t=max(t,dfs(x,y-1,z));
    f[x][y]=t+1;
    return f[x][y];
}
int main(){
    scanf("%d%d\n",&n,&m);
    for(int i=0;i<n;i++) scanf("%s",s[i]);
    memset(f,-1,sizeof(f));
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
            if(s[i][j]=='D'){
                ans=max(ans,dfs(i,j,0));
                if(ans==INF) break;
            }
    if(ans>=INF) printf("Poor Inna!");else if(ans<4) printf("Poor Dima!");else printf("%d\n",ans/4);

}
