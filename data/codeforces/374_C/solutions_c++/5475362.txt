#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int INF=0x3f3f3f3f;
char map[1005][1005],s[]="DIMA";
int dp[1005][1005],dx[]= {-1,1,0,0},dy[]= {0,0,-1,1};
int n,m,ans=0;
int dfs(int p,int q,int l)
{
    if(dp[p][q]!=-1)
        return dp[p][q];
    dp[p][q]=INF;
    l=(l+1)%4;
    int tmp=0;
    for(int i=0; i<4; ++i)
    {
        int tp=p+dx[i],tq=q+dy[i];
        if(tp>=0&&tp<n&&tq>=0&&tq<m&&map[tp][tq]==s[l])
            tmp=max(tmp,dfs(tp,tq,l));
    }
    return dp[p][q]=min(INF,tmp+1);
}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=0; i<n; ++i)
        scanf("%s",map[i]);
    memset(dp,-1,sizeof(dp));
    for(int i=0; ans!=INF&&i<n; ++i)
        for(int j=0; ans!=INF&&j<m; ++j)
            if(map[i][j]=='D')
                ans=max(ans,dfs(i,j,0));
    if(ans==INF)
        printf("Poor Inna!");
    else if(ans<4)
        printf("Poor Dima!");
    else
        printf("%d",ans/4);
}
