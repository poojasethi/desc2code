#include<bits/stdc++.h>
using namespace std;
#define ll long long

const int N=1005, inf = (1<<30)-1, dx[] = {0,0,1,-1}, dy[] = {-1,1,0,0};
int n, m;
char a[N][N], next[400];
bool vis[N][N];
int dp[N][N];
int dfs(int x, int y) {
	if(dp[x][y])
		return dp[x][y];
	if(vis[x][y])
		return inf;
	vis[x][y]=1;
	int tx, ty, res=0;
	for(int i=0;i<4;i++) {
		tx = x+ dx[i], ty= y+ dy[i];
		if(tx <1 || tx >n || ty < 1 || ty>m)
			continue;
		if(a[tx][ty]==next[a[x][y]])
			res = max(res, dfs(tx,ty));
	}
	vis[x][y]=0;
	dp[x][y] = res+1;
	return res+1;
}

int main() {
	cin >> n >> m;
	for(int i=1;i<=n;i++) {
		scanf("\n%s",a[i]+1);
	}
	next['D']= 'I', next['I'] = 'M' , next['M'] = 'A', next['A']='D';
	int ans=0;
	for(int i=1;i<=n;i++) {
		for(int j=1;j<=m;j++) {
			if(a[i][j]=='D')
				ans = max(ans, dfs(i,j));
		}
	}
	if(ans>=inf)
		cout << "Poor Inna!" << endl;
	else if(ans/4==0)
		cout << "Poor Dima!" << endl;
	else 
		cout << ans/4 << endl;
	return 0;
}
