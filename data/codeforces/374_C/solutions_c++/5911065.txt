#include <iostream>
#include <cstring>

using namespace std;

string s[10000];
int inf = 1000*1000;
int dp[1009][1009];
string c = "DIMA";
int ans, n, m;

int dfs(int i, int j, int pos=0) {
    int tmp = 0;
    if (dp[i][j] != 0) return dp[i][j];
    dp[i][j] = inf;

    if (j < m && s[i][j+1] == c[(pos+1)%4]) {
        tmp = max(tmp, dfs(i, j+1, (pos+1)%4));
    }
    if (i < n && s[i+1][j] == c[(pos+1)%4]) {
        tmp = max(tmp, dfs(i + 1, j, (pos+1)%4));
    }
    if (i > 0 && s[i-1][j] == c[(pos+1)%4]) {
        tmp = max(tmp, dfs(i-1, j, (pos+1)%4));
    }
    if (j > 0 && s[i][j-1] == c[(pos+1)%4]) {
        tmp = max(tmp, dfs(i, j-1, (pos+1)%4));
    }

    return dp[i][j] = min(tmp+1, inf);

}

int main() {

    cin >> n >> m;
    for (int i = 0; i < n; i++) {
        cin >> s[i];
    }

    for (int i = 0; i < n; i++){
        for (int j = 0; j < m; j++) {
            if (s[i][j] == 'D')
                ans = max(ans, dfs(i, j));
        }
    }

    if (ans < 4) cout << "Poor Dima!\n";
    else if (ans == inf) cout << "Poor Inna!\n";
    else cout << ans / 4 << endl;

    return 0;
}
