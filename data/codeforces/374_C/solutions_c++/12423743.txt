#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

char st[] = "DIMA";

const int INF = 1e9;

int maze[1005][1005];
int dp[1005][1005];
int n, m;

int dx[] = {1, 0, -1, 0};
int dy[] = {0, 1, 0, -1};

int dfs(int x, int y, int pos)
{
    if (dp[x][y] != -1) return dp[x][y];
    dp[x][y] = INF;
    int res = 0;
    pos = (pos + 1) % 4;
    for (int i = 0; i < 4; ++i) {
        int nx = x + dx[i], ny = y + dy[i];
        if (maze[nx][ny] == st[pos])
            res = max(res, dfs(nx, ny, pos));
    }
    dp[x][y] = ++res;
    return res;
}

int main()
{
    memset(dp, -1, sizeof(dp));
    scanf("%d%d\n", &n, &m);
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= m; ++j)
            maze[i][j] = getchar();
        getchar();
    }
    int ans = 0;
    for (int i = 1; i <= n; ++i)
    for (int j = 1; j <= m; ++j) {
        if (maze[i][j] == 'D')
            ans = max(ans, dfs(i, j, 0));
    }
    if (ans >= INF)
        printf("Poor Inna!\n");
    else if (ans / 4)
        printf("%d\n", ans / 4);
    else
        printf("Poor Dima!\n");

    return 0;
}
