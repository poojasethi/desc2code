#include<cstdio>
#include<cstring>
int max(int a,int b){
    return a>b?a:b;
}
const int X=1010;
const int inf=1e9;
char s[X][X],st[10]="DIMA";
int dp[X][X];
int dfs(int x,int y,int pos){
    if(dp[x][y]!=-1)return dp[x][y];
    dp[x][y]=inf;
    int tmp=0;
    pos=(pos+1)%4;
    if(s[x+1][y]==st[pos])
        tmp=max(tmp,dfs(x+1,y,pos));
    if(s[x-1][y]==st[pos])
        tmp=max(tmp,dfs(x-1,y,pos));
    if(s[x][y+1]==st[pos])
        tmp=max(tmp,dfs(x,y+1,pos));
    if(s[x][y-1]==st[pos])
        tmp=max(tmp,dfs(x,y-1,pos));
    dp[x][y]=++tmp;
    return tmp;
}
int main(){
    int i,j,n,m,k,as=0;
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;i++)
        scanf("%s",s[i]+1);
    memset(dp,-1,sizeof(dp));
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
            if(s[i][j]=='D'){
                if(as==inf)break;
                as=max(as,dfs(i,j,0));
            }
    if(as>=inf)puts("Poor Inna!");
    else if(as<4)puts("Poor Dima!");
    else printf("%d\n",as/4);
    return 0;
}
