#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<cmath>
#include<algorithm>
#define INF 0x3f3f3f3f
using namespace std;
typedef long long ll;
const int maxn=1e3+5;
int dp[maxn][maxn];
int n,m,max1;
char ch[maxn][maxn];
char str[]="DIMA";
int jx[]={0,0,1,-1};
int jy[]={1,-1,0,0};
int dfs(int x,int y,int temp)
{
	if(dp[x][y]!=-1)
	return dp[x][y];
	dp[x][y]=INF;
	temp=(temp+1)%4;
	int t=0;
	for(int i=0;i<4;i++)
	{
		int a=x+jx[i];
		int b=y+jy[i];
		if(a>=1&&a<=n&&b>=1&&b<=m&&ch[a][b]==str[temp])
		{
			t=max(t,dfs(a,b,temp));
		}
	}
	dp[x][y]=++t;
	return t;
}
int main()
{
	scanf("%d%d\n",&n,&m);
	for(int i=1;i<=n;i++)
	{
		for(int k=1;k<=m;k++)
		{
			scanf("%c",&ch[i][k]);
		}
		getchar();
	}	
	memset(dp,-1,sizeof(dp));
	
	for(int i=1;i<=n;i++)
	{
		for(int k=1;k<=m;k++)
		{
			if(ch[i][k]=='D')
			{
				max1=max(max1,dfs(i,k,0));
			}
		}
	}
	if(max1/4==0)
	{
		puts("Poor Dima!\n");
	}
	else if(max1>INF)
	{
		puts("Poor Inna!\n");
	}
	else
	{
		printf("%d\n",max1/4);
	}
	return 0;
 }
