#include <iostream>
#include<bits/stdc++.h>

using namespace std;
int vis[1005][1005],dp[1005][1005],n,m;
char a[1005][1005];
string s="DIMA";

bool isValid(int x, int y){
    return x>=0&&y>=0&&x<n&&y<m;
}
int dfs(int i,int j,int p)
{
	if(vis[i][j]==1)
	{
		cout<<"Poor Inna!";
        exit(0);
	}
	
	if(vis[i][j]==2)
	{
		return dp[i][j];
	}
	vis[i][j]=1;
	dp[i][j]=1;
	int x[] = {1, 0, -1, 0}, y[] = {0, 1, 0, -1};
	p=(p+1)%4;
	int ans=0;
	for(int k=0;k<4;k++)
	{
		int ii=i+x[k];
		int jj=j+y[k];
		if(isValid(ii,jj) && a[ii][jj]==s[p])
		{
			dp[i][j]=max(dp[i][j],dfs(ii,jj,p)+1);
		}
	}
//	dp[i][j]=ans;
	vis[i][j]=2;
	return dp[i][j];
	
}
int main() 
{
	int i,j;
	cin>>n>>m;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			cin>>a[i][j];
		}
	}
	memset(dp,-1,sizeof(dp));
	int res=0;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			if(a[i][j]=='D' && vis[i][j]==0)
			{
			
				res=max(res,dfs(i,j,0)/4);
			//	cout<<res<<endl;
			}
		}
	}
	
	if(res==0)
	{
		cout<<"Poor Dima!\n";
	}
	else
	{
		cout<<res<<endl;
	}
	return 0;
}
