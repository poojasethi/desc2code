#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
char ooo[10]="DIMA";
char s[1111][1111];
int dp[1111][1111];
const int INF=1e9;
int i,j,n,m,k,ans=0;
int dfs(int x,int y,int pos)
{
    if(dp[x][y]!=-1)return dp[x][y];
    dp[x][y]=INF;
    int tmp=0;
    pos=(pos+1)%4;
    if(s[x+1][y]==ooo[pos])
        tmp=max(tmp,dfs(x+1,y,pos));
    if(s[x-1][y]==ooo[pos])
        tmp=max(tmp,dfs(x-1,y,pos));
    if(s[x][y+1]==ooo[pos])
        tmp=max(tmp,dfs(x,y+1,pos));
    if(s[x][y-1]==ooo[pos])
        tmp=max(tmp,dfs(x,y-1,pos));
    dp[x][y]=++tmp;
    return tmp;
}

int main()
{
    
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;i++)
        scanf("%s",s[i]+1);
    memset(dp,-1,sizeof(dp));
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
            if(s[i][j]=='D')
            {
                if(ans==INF)break;
                ans=max(ans,dfs(i,j,0));
            }
    if(ans>=INF) puts("Poor Inna!");
    else if(ans<4) puts("Poor Dima!");
    else printf("%d\n",ans/4);
    return 0;
}
