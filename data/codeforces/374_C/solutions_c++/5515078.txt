#include<cstdio>
#include<cstring>
int lim = 1e9;
int dp[1005][1005];
char s[1005][1005];
char dim[8]="DIMA";
int max(int a , int b)
{
	return a>b?a:b;
}
int dfs(int x,int y , int p)
{
	
	if(dp[x][y]!=-1)
	return dp[x][y];
	int k=0;
	dp[x][y] = lim;
	p = (p+1)%4;
	
	if(s[x+1][y]==dim[p])
	k = max(k,dfs(x+1,y,p));
	if(s[x-1][y]==dim[p])
	k = max(k,dfs(x-1,y,p));
	if(s[x][y+1]==dim[p])
	k = max(k,dfs(x,y+1,p));
	if(s[x][y-1]==dim[p])
	k = max(k,dfs(x,y-1,p));
	//k=+1;
	dp[x][y] = ++k;
	return k;
	

	
}
