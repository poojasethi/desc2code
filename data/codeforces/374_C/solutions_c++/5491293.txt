#include <algorithm>
#include <iostream>
using namespace std;

const int N = 1005;

int n,m,i,j,ans=0;
int dp[N][N];
int nx,ny;
char a[N][N];
char next[5];
int used[N][N];
bool flag;
int kx[]={1,0,-1,0};
int ky[]={0,1,0,-1};

bool ok(int x,int y){return (x>0 && y>0 && x<=n && y<=m);}

int dfs(int x,int y){
	if(used[x][y]==2){
		flag = 1;
		return 0;
	}
	if(dp[x][y])return dp[x][y];
	dp[x][y] = 1;
	used[x][y] = 2;
	for(int i=0;i<4;i++){
		nx = x + kx[i] ;
		ny = y + ky[i] ;
		if(ok(nx,ny) && next[a[x][y]] == a[nx][ny])
			dp[x][y] = max(dp[x][y],dfs(nx,ny)+1);
	}
	used[x][y] = 1;
	return dp[x][y];
}
  
int main () {
	cin >> n >> m ;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			cin >> a[i][j] ;
	next['D'] = 'I' ;
	next['I'] = 'M' ;
	next['M'] = 'A' ;
	next['A'] = 'D' ;
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			if(a[i][j]!='D' || used[i][j])continue;
			ans = max(ans,dfs(i,j));
			if(flag)break;
		}    
		if(flag)break;
	}
	if(flag)cout<<"Poor Inna!";
	else if(ans/4<1)cout<<"Poor Dima!";
	else cout<<ans/4;
        return 0;
}
