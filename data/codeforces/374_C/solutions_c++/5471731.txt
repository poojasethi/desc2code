#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
#define N 1000
using namespace std;

int memo[N+5][N+5][4], m, n;
bool done[N+5][N+5][4], process[N+5][N+5][4], ciclo=0;
char M[N+5][N+5];
string s="DIMA";

int dx[]={0,0,1,-1};
int dy[]={1,-1,0,0};

int f(int x,int y, int letra){
	int &ret=memo[x][y][letra];
	if(done[x][y][letra])return ret;
	if(process[x][y][letra]){
		ciclo=1;
		return 0;
	}
	process[x][y][letra]=1;
	ret=0;
	for(int i=0;i<4;++i){
		int nx=x+dx[i], ny =y+dy[i];
		if(nx>=0 && nx<n && ny>=0 && ny<m && M[nx][ny]==s[(letra+1)%4] ){
			if(M[nx][ny]=='A'){
				ret=max(ret, 1+ f( nx, ny, (letra+1)%4) );
			}
			else ret=max(ret, f( nx, ny, (letra+1)%4) );
		}
	}
	done[x][y][letra]=1;
	return ret;
}


int main(){
	cin>>n>>m;
	for(int i=0;i<n;++i){
		for(int j=0;j<m;++j){
			cin>>M[i][j];
		}
	}
	memset(memo,-1,sizeof(memo));
	int maxi=0;
	for(int i=0;i<n;++i){
		for(int j=0;j<m;++j){
			if(M[i][j]=='D'){
				maxi=max(maxi, f(i,j,0) );
			}
		}
	}
	if(ciclo)puts("Poor Inna!");
	else{
		if(maxi==0)puts("Poor Dima!");
		else cout<<maxi<<endl;
	}
}
