#include <cstdio>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstring>

using namespace std;

const int INF=19950920;

int n,m;
char s[1005][1005];
int map[1005][1005];
int ans;
char str[]="DIMA";
int dx[]={0,0,1,-1};
int dy[]={1,-1,0,0};
int dfs(int x,int y,int dq)
{
	int now;
	if (map[x][y]!=-1) return map[x][y];
	map[x][y]=INF;
	dq=(dq+1)%4;
	now=0;
	for (int i=0;i<4;++i)
	{
		int dqx,dqy;
		dqx=x+dx[i];
		dqy=y+dy[i];
		if (dqx>=1&&dqx<=n&&dqy>=1&&dqy<=m&&s[dqx][dqy]==str[dq])
			now=max(now,dfs(dqx,dqy,dq));
	}
	map[x][y]=min(INF,now+1);
	return map[x][y];
}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i)
		scanf("%s",s[i]+1);
	for (int i=1;i<=n;++i)
		for (int j=1;j<=m;++j) map[i][j]=-1;
	ans=0;
	for (int i=1;i<=n;++i)
		for (int j=1;j<=m;++j)
		{
			if (s[i][j]!='D') continue;
			ans=max(ans,dfs(i,j,0));
			if (ans==INF) break;
		}
	if (ans==INF)
	{
		printf("Poor Inna!\n");
		return 0;
	}
	ans=ans/4;
	if (ans==0)
	{
		printf("Poor Dima!\n");
		return 0;
	}
	printf("%d\n",ans);
	return 0;
}
