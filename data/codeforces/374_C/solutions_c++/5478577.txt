#include <iostream>
#include <cstdlib>
#include <map>
using namespace std;

map<char,int> dima;
int n, m, arr[1005][1005], dp[1005][1005], vis[1005][1005] = {};

bool is_valid(int x, int y){
    return x>=0&&y>=0&&x<n&&y<m;
}

int check(int x, int y){
    if(vis[x][y]==1){
        cout << "Poor Inna!";
        exit(0);
    }
    if(vis[x][y]==2) return dp[x][y];
    dp[x][y] = 1;
    vis[x][y] = 1;
    int dx[] = {1, 0, -1, 0}, dy[] = {0, 1, 0, -1};
    for(int i = 0; i<4; i++){
        if(is_valid(x+dx[i], y+dy[i])&&arr[x+dx[i]][y+dy[i]]==(arr[x][y]+1)%4){
            dp[x][y] = max(dp[x][y], check(x+dx[i], y+dy[i])+1);
        }
    }
    vis[x][y] = 2;
    return dp[x][y];
}

int main(){

    dima['D'] = 0;
    dima['I'] = 1;
    dima['M'] = 2;
    dima['A'] = 3;
    cin >> n >> m;
    for(int i = 0; i<n; i++){
        string s;
        cin >> s;
        for(int j = 0; j<m; j++) arr[i][j] = dima[s[j]];
    }
    int res = 0;
    for(int i = 0; i<n; i++) for(int j = 0; j<m; j++) if(!arr[i][j]) res = max(res, check(i,j)/4);
    if(!res) cout << "Poor Dima!";
    else cout << res;

}
