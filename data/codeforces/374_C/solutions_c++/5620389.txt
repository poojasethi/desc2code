#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;

const int MAX=1002,inf=1e9;
char map[MAX][MAX],ch[5]="DIMA";
int v[4][2]={1,0,-1,0,0,1,0,-1};
int dp[MAX][MAX];//此点开始搜的最大连续长度

int dfs(int x,int y,int pos){//在(i,j)字母为pos的最大长度
	if(dp[x][y]!=-1) return dp[x][y];
	dp[x][y]=inf;
	int rst=1;
	pos=(pos+1)%4;
	for(int i=0;i<4;i++){
		int nx=x+v[i][0],ny=y+v[i][1];
		if(map[nx][ny]==ch[pos])
		    rst=max(rst,1+dfs(nx,ny,pos));
	}
	dp[x][y]=rst;
	return rst;
}

int main(){
	int n,m;
	while(cin>>n>>m){
		for(int i=1;i<=n;i++)
				scanf("%s",map[i]+1);
		memset(dp,-1,sizeof dp);

		int ans=0,flag=1;
		for(int i=1;i<=n&&flag;i++)
			for(int j=1;j<=m&&flag;j++)
				if(map[i][j]=='D'){
				    ans=max(ans,dfs(i,j,0));
					if(ans>=inf) flag=0;
				}

		if(ans<4)
			printf("Poor Dima!\n");
		else if(ans>=inf)
			printf("Poor Inna!\n");
		else
			printf("%d\n",ans/4);
	}
	return 0;
}
