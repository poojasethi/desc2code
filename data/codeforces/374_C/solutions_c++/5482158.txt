#include <iostream>
#include <cmath>

using namespace std;

long m,n,i,j,w[1010][1010],ans;
char a[1010][1010];
bool cycle,u[1010][1010];

long dfs (long x, long y)
{
	char nex;
	if (cycle || u[x][y]) {
		cycle=1;
		return 0;
	}
	if (w[x][y]>0) return w[x][y];
	u[x][y]=1;
	if (a[x][y]=='D') nex='I';
	if (a[x][y]=='I') nex='M';
	if (a[x][y]=='M') nex='A';
	if (a[x][y]=='A') nex='D';
	if (a[x-1][y]==nex) w[x][y]=dfs(x-1,y)+1;
	if (a[x+1][y]==nex) w[x][y]=max(dfs(x+1,y)+1,w[x][y]);
 	if (a[x][y-1]==nex) w[x][y]=max(dfs(x,y-1)+1,w[x][y]);
 	if (a[x][y+1]==nex) w[x][y]=max(dfs(x,y+1)+1,w[x][y]);
 	w[x][y]=max(w[x][y],long(1));
 	u[x][y]=0;
 	return w[x][y];
}

int main()
{
	cin >> n >> m;
	for (i=1;i<=n;i++) 
		for (j=1;j<=m;j++) cin >> a[i][j];
	for	(i=1;i<=n;i++)
		for (j=1;j<=m;j++)
			if (a[i][j]=='D') dfs(i,j);
	ans=0;		
	if (cycle) cout << "Poor Inna!";
	else {
		for (i=1;i<=n;i++)
			 for (j=1;j<=m;j++) 
			 	ans=max(ans,w[i][j]/4);
		if (ans==0) cout << "Poor Dima!";
		else cout << ans;	 	
	}
	return 0;
}
