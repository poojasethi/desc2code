#include<cstdio>
#include<cstdlib>
int n,m,i,j,a[1010][1010],ans,now,f[1010][1010];bool vis[1010][1010];
char c;int dx[4]={0,0,1,-1},dy[4]={1,-1,0,0};
int max(int a,int b){return a>b?a:b;}
int dfs(int i,int j)
{
	if(f[i][j])return f[i][j]+(a[i][j]==3);
	for(int k=0;k<4;k++)
	{
		int xx=i+dx[k],yy=j+dy[k];
		if((a[xx][yy]-a[i][j]+4)%4==1)
		{
			if(vis[xx][yy]){printf("Poor Inna!");exit(0);}
			vis[xx][yy]=true;
			f[i][j]=max(dfs(xx,yy),f[i][j]);
			ans=max(f[i][j],ans);
			vis[xx][yy]=false;
		}
	}
	return f[i][j]+(a[i][j]==3);
}
int main()
{
	scanf("%d%d",&n,&m);getchar();
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			c=getchar();if(c=='D')a[i][j]=0;
			else if(c=='I')a[i][j]=1;
			else if(c=='M')a[i][j]=2;
			else if(c=='A')a[i][j]=3;
		}
		getchar();
	}
	for(i=1;i<=n;i++)
	for(j=1;j<=m;j++)
	if(a[i][j]==0)
	{vis[i][j]=true;dfs(i,j);vis[i][j]=false;}
	if(ans==0)printf("Poor Dima!");
	else printf("%d\n",ans);
