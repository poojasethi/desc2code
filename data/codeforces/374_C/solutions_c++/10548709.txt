#include <bits/stdc++.h>
using namespace std;
#define INF 10000000100
long long n,m,dp[1010][1010];
char str[8]="DIMA",mat[1010][1010];
long long dfs(long long x,long long y,long long pos){
    if(dp[x][y]!=-1) return dp[x][y];
    dp[x][y]=INF;
    long long ans=0;
    pos=(pos+1)%4;
    if(mat[x+1][y]==str[pos]) ans=max(ans,dfs(x+1,y,pos));
    if(mat[x-1][y]==str[pos]) ans=max(ans,dfs(x-1,y,pos));
    if(mat[x][y+1]==str[pos]) ans=max(ans,dfs(x,y+1,pos));
    if(mat[x][y-1]==str[pos]) ans=max(ans,dfs(x,y-1,pos));
    dp[x][y]=++ans;
    return ans;
}
int main(){
    char a;
    memset(dp,-1,sizeof(dp));
    scanf("%lld%lld",&n,&m);
    long long ans=0;
    for(int i=1;i<=n;i++)
        scanf("%s",mat[i]+1);
    for(long long i=1;i<=n;i++){
        for(long long j=1;j<=m;j++){
            if(mat[i][j]=='D'){
                if(ans==INF) break;
                ans=max(ans,dfs(i,j,0));
            }
        }
    }
    if(ans>=INF) printf("Poor Inna!\n");
    else if(ans<4) printf("Poor Dima!\n");
    else printf("%lld\n",ans/4);    
    return 0;
}
