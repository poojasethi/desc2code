#include <iostream>
#include <cstdio>
#include <string>
#include <algorithm>
#include <cstring>
#define N 1009
using namespace std;
int n,m,ans;
bool loop;
char x[N][N];
int has[N][N];
int ax[]={-1,1,0,0};
int ay[]={0,0,-1,1};
int INF;
char next[30];
int dfs(int curx,int cury)
{
    if(has[curx][cury]!=-1)return has[curx][cury];
    has[curx][cury]=INF;
    int mx=0;
    for(int i=0,nx,ny;i<4;i++)
    {
        nx=curx+ax[i];
        ny=cury+ay[i];
        if(nx>=0&&ny>=0&&nx<n&&ny<m&&next[x[curx][cury]-'A']==x[nx][ny]){
            mx=max(mx,dfs(nx,ny));
        }
    }
    return has[curx][cury]=min(INF,mx+1);
}
int main()
{
    cin>>n>>m;
    for(int i=0;i<n;i++)
        scanf("%s",x[i]);
    memset(has,-1,sizeof(has));
    INF=n*m+1;
    next['D'-'A']='I';
    next['I'-'A']='M';
    next['M'-'A']='A';
    next['A'-'A']='D';
    for(int i=0;i<n&&ans!=INF;i++)
        for(int j=0;j<m&&ans!=INF;j++)
            if(x[i][j]=='D')
                ans=max(ans,dfs(i,j));
    if(ans==INF)cout<<"Poor Inna!"<<endl;
    else if(ans<4)cout<<"Poor Dima!"<<endl;
    else cout<<ans/4<<endl;
    return 0;
}
