#include <bits/stdc++.h>  
using namespace std;
const int maxn = 1010;  
const int inf = 1e9;  
char s[maxn][maxn], st[]="DIMA";  
int dp[maxn][maxn];
int n, m, ans;
int dfs(int x, int y, int pos)
{  
	if (dp[x][y] != -1)
	  return dp[x][y];
	dp[x][y] = inf;  
	int tmp=0; 
	pos = (pos+1) % 4;  
	if (s[x+1][y] == st[pos])
	  tmp = max(tmp, dfs(x+1, y, pos));
	if (s[x-1][y] == st[pos])
	  tmp = max(tmp, dfs(x-1, y, pos));
	if(s[x][y+1]==st[pos])
	  tmp=max(tmp,dfs(x,y+1,pos));
	if(s[x][y-1]==st[pos])
	  tmp=max(tmp,dfs(x,y-1,pos));
	//��tmp=0�������x,y��ȥ���·�����ȣ����dp[x][y]=tmp+1
	dp[x][y]=++tmp;
	return tmp;
}
int main()
{
	scanf("%d%d", &n, &m); 
	for (int i = 1; i <= n; i++)  
	  scanf("%s", s[i]+1);
	memset(dp, -1, sizeof(dp));  
	ans = 0;
	for (int i = 1; i <= n; i++)  
	  for (int j = 1; j <= m; j++)  
		if (s[i][j] == 'D')
		  ans = max(ans, dfs(i, j,0));  
	if (ans >= inf)
	  puts("Poor Inna!");  
	else if (ans<4)
	  puts("Poor Dima!");  
	else
	  printf("%d\n",ans/4);
	return 0;  
}
