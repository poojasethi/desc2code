#include <iostream>

#include <vector>
using namespace std;
int i,j,x,y,z,n,m,len,k,tp;
int a[1111][1111],b[1111][1111];
int ans=0;
char c[1111][1111];
string DIMA="DIMAD";
int di[4]={1,0,-1,0};
int dj[4]={0,1,0,-1};

void dfs(int i,int j,int k)
{


 b[i][j]=1;
 a[i][j]=1;

 for (int t=0;t<4;t++)
 {
  int x=i+di[t];
  int y=j+dj[t];
  if (c[x][y]==DIMA[k+1])
    {
     if (b[x][y]==1)  {tp=1;} else
         {
           if (b[x][y]==0) dfs(x,y,(k+1)%4);
           a[i][j]=max(a[i][j],a[x][y]+1);
         }
    }
 }
 //if (k==3) a[i][j]++;
 if (k==0) ans=max(ans,a[i][j]/4);
 b[i][j]=2;
}


int main()
{
 cin>>n>>m;
 for (i=1;i<=n;i++)
  for (j=1;j<=m;j++)
   cin>>c[i][j];


 for (i=1;i<=n;i++)
  for (j=1;j<=m;j++)
   if (c[i][j]=='D' && b[i][j]==0) dfs(i,j,0);

//for (i=1;i<=n;i++)
//{
// for (j=1;j<=m;j++) cout<<a[i][j]<<' ';
// cout<<endl;}

 if (tp) cout<<"Poor Inna!"; else
  if (ans==0) cout<<"Poor Dima!"; else cout<<ans;


 }
