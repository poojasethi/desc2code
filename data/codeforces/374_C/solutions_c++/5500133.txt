#include<iostream>
#include<stdio.h>
using namespace std;
const int INF=0x3f3f3f3f;
string map[1005],s="DIMA";
int dp[1005][1005]={0};
int n,m,ans=0;

int dfs(int i,int j,int l)
{if(dp[i][j]!=0) return dp[i][j];
 dp[i][j]=INF;
 l=(l+1)%4;
 int tmp=0;
 if (i>0&&map[i-1][j]==s[l])  tmp=max(tmp,dfs(i-1,j,l));
 if (i<n-1&&map[i+1][j]==s[l])tmp=max(tmp,dfs(i+1,j,l));
 if (j>0&&map[i][j-1]==s[l])  tmp=max(tmp,dfs(i,j-1,l));
 if (j<m-1&&map[i][j+1]==s[l])tmp=max(tmp,dfs(i,j+1,l));
return dp[i][j]=min(INF,tmp+1);
}
int main()
{//freopen("i.txt","r",stdin);
cin>>n>>m;
for(int i=0;i<n;i++) cin>>map[i];
for(int i=0; ans!=INF&&i<n; i++)
   for(int j=0; ans!=INF&&j<m;j++)
     if(map[i][j]=='D')   ans=max(ans,dfs(i,j,0));
if(ans==INF) cout<<"Poor Inna!";
 else if(ans<4) cout<<"Poor Dima!";
    else  cout<<ans/4;
return 0;
}
