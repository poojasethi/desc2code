#include<iostream>
#include<cstdio>   
#include<cstring> 
using namespace std;

int max(int i,int j)
{
	return i>j?i:j;
}

int n,m;
char map[1000][1000];
int dp[1000][1000];
int infinite=1e9;

char word[10]="DIMA";

int dfs(int i,int j,int x)
{

	if(dp[i][j]!=-1) return dp[i][j];
    dp[i][j]=infinite;
	int tmp=0;
	x=(x+1)%4;
	if(i-1>=0&&map[i-1][j]==word[x]) tmp=max(tmp,dfs(i-1,j,x));
	if(i+1<n&&map[i+1][j]==word[x]) tmp=max(tmp,dfs(i+1,j,x));
	if(j-1>=0&&map[i][j-1]==word[x]) tmp=max(tmp,dfs(i,j-1,x));
	if(j+1<m&&map[i][j+1]==word[x]) tmp=max(tmp,dfs(i,j+1,x));
	dp[i][j]=++tmp;
	return tmp;
}

int main()
{
	int count=0;
	int i,j;
	cin>>n>>m;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			cin>>map[i][j];

	memset(dp,-1,sizeof(dp));
	
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			
			if(map[i][j]=='D')
			{
				if(dp[i][j]==infinite)
				break;
				count=max(count,dfs(i,j,0));
			}
		}
	}
	if(count>=infinite)
		cout<<"Poor Inna!"<<endl;
	else if(count<4)
		cout<<"Poor Dima!"<<endl;
	else
		cout<<count/4<<endl;
   
    return 0;
 }
