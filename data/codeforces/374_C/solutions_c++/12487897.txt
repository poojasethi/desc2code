#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std ;

int n,m ;
char a[1002][1002];
int  visit[1002][1002];
bool mark[1002][1002];
char s[5]="DIMA";
const int INF = 1000005;
int dx[4]={0,0,-1,1};
int dy[4]={1,-1,0,0};

int dfs(int cnt,int x ,int y )
{
	if(mark[x][y])   return INF ;
	
	int & ans=visit[x][y];
	if(visit[x][y]!=-1)
		return   ans ;
	ans=0;
	int px,py,index=(cnt+1)%4 ;
	mark[x][y]=true ;
	for(int i=0;i<4;i++)
	{
		px=x+dx[i];
		py=y+dy[i];
		if(px>=0&&px<n&&py>=0&&py<m&&s[index]==a[px][py])
		{
			ans=max(ans,dfs(index,px,py));
		}
	}
	ans++;
	mark[x][y]=false ;
	return ans ;
}



int main()
{
	cin>>n>>m;
	for(int i=0;i<n;i++)
	scanf("%s",a[i]);
	int max=0;
	memset(visit,-1,sizeof(visit));
	memset(mark,false ,sizeof(mark));
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if(a[i][j]=='D')
			{
				int num=dfs(0,i,j);
				if(num>max)
				max=num;
				if(max>=INF)
				break ;
			}
		}
		if(max>=INF)
		break ;
	}
	if(max/4==0)
	printf("Poor Dima!\n");
	else 	if(max>=INF)
	printf("Poor Inna!\n");
	else  
	printf("%d\n",max/4);
	return 0 ;
}
