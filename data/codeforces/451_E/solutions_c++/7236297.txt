#include <iostream>
using namespace std;

const long long p = 1000000007;
long long inv[25];
int n;

long long c(long long m)
{
	long long f = 1;
	for (int i = 1; i < n; ++i)
	{
		f = (f * ((((m + i) % p) * inv[i]) % p)) % p;	
	}
	return f;
}

int main()
{
	long long s;
	long long f[25];

	cin >> n >> s;

	inv[1] = 1;
	for (int i = 2; i <= n; ++i)
	{
		long long j = p + 1;
		while (j % i != 0)
			j += p;
		inv[i] = j / i;
	}

	for (int i = 0; i < n; ++i)
	{
			cin >> f[i];
			f[i]++;
	}

	long long res = 0;
	for (int i = 0; i < (1 << n); ++i)
	{
		long long d = 0;
		long long pos = 1;
		for (int j = 0; j < n; ++j)
		{
			if ((i & (1 << j)) > 0)
			{
				pos *= -1;
				d += f[j];
			}
		}
//		cout << i << " " << d << " " << pos * c(s-d) << endl;
		if (s >= d)
			res = (res + p + pos * c(s-d)) % p;
	}

	cout << res << endl;

	return 0;
}
