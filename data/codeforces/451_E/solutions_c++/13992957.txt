#include <stdio.h>
#include <iostream>
using namespace std;
#define ll long long
const int MOD=1e9+7;
ll p,s,f[25],v=0,inv[25];
int n,c;
ll get(ll n,ll r){
	ll t=1;
	for (ll i=0;i<r;i++)
		t=t*((n-i)%MOD)%MOD*inv[i+1]%MOD;
	return t;
}
int main( ){
	scanf("%d %I64d",&n,&s);
	for (int i=0;i<n;i++)
		scanf("%I64d",&f[i]);
	inv[1]=1;
	for (int i=2;i<=20;i++)
		inv[i]=(MOD-MOD/i*inv[MOD%i]%MOD)%MOD;
	for (int j=0;j<(1<<n);j++){
		p=0,c=1;
		for (int k=0;k<n;k++)
			if (j&(1<<k))
				p+=f[k]+1,c=-c;
		if (s-p>=0) v=((v+c*get(n+s-p-1,n-1))%MOD+MOD)%MOD;
	}
	printf("%I64d\n",v);
	return 0;
}
