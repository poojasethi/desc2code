#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
const ll MOD=1e9+7;
#define N 22
ll a[N],inv[N];
ll s,ans;
int n;

ll C(ll n,ll m){
  ll c=1;
  for (ll i=1;i<=m;++i){
    c=(c*((n-i+1)%MOD))%MOD;
    c=(c*(inv[i]%MOD))%MOD;
  }
  return c;
}

int main(){
  int i,j;
  // freopen("1.txt","r",stdin);
  inv[1]=1;
  for (i=2;i<N;++i) inv[i]=(MOD-MOD/i)*inv[MOD%i]%MOD;
  scanf("%d%I64d",&n,&s);
  for (i=0;i<n;++i) scanf("%I64d",&a[i]);
  ans=0;
  int nn=1<<n;
  for (i=0;i<nn;++i){
    int m=0;
    ll t=s;
    for (j=0;j<n;++j)
      if ((i>>j)&1){
        ++m;
        t-=a[j]+1;
      }
    if (t<0) t=0;
    else t=C(t+n-1,n-1);
    if (m&1) ans=(MOD-t+ans)%MOD;
    else ans=(t+ans)%MOD;
  }
  printf("%I64d\n",ans);
  return 0;
}
