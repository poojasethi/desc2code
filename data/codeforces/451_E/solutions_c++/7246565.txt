#include<bits/stdc++.h>
using namespace std;

const int MOD=1000000007;
#define ll long long

ll inv[25];

ll comb(ll n, ll r)
{
    ll res = 1;
    for(ll i = 0; i < r; i++)
        res = res * ((n - i) % MOD) % MOD * inv[i + 1] % MOD;
    return res;
}

int main()
{
    ll sum, s, f[25], ans = 0;
    int i, j, k, n, c;
    cin >> n >> s;
    for(i = 0; i < n; i++)
        cin >> f[i];
    inv[1] = 1;
    for(i = 2; i <= 20; i++)
        inv[i] = (MOD - MOD / i * inv [MOD % i] % MOD) % MOD;
    for(j = 0; j < (1<<n); j++)
    {
        sum = 0;
        c = 1;
        for(k = 0; k < n; k++)
        {
            if(j & (1<<k))
            {
                sum += f[k] + 1;
                c = -c;
            }
        }
        if(s - sum >= 0)
            ans = ((ans + c*comb(n + s - sum - 1,n - 1))%MOD + MOD) % MOD;
    }
    cout << ans << endl;
    return 0;
}
