#include <iostream>
#include <cstdio>

using namespace std;
typedef long long ll;

ll N; ll S;
ll F[30];
ll Ans = 0;
ll MOD = 1000000007;
ll re[30];

int main()
{
    cin>>N>>S;

    for(ll Ni = 0; Ni < N; Ni++) cin>>F[Ni];

    re[1] = 1;

    for(int i = 2; i < 30; i++)
        re[i] = re[MOD%i]*(MOD-MOD/i)%MOD;

    ll n = (1<<N);

    for(ll i = 0; i < n; i++)
    {
        ll cnt = 0; ll f = 1;

        for(ll bi = 0, di = 1; bi < N; bi++, di*=2)
        {
            if( i&di ) cnt += F[bi]+1, f*=-1;
        }

        if( cnt > S ) continue;
        cnt = S-cnt;

        ll o = N+cnt-1; o %= MOD;
        ll p = N-1;

        ll t = 1;

        for(ll pi = 1; pi <= p; pi++)
        {
            t *= o-pi+1; t %= MOD;
            t *= re[pi]; t %= MOD;
        }

        Ans += f*t; Ans %= MOD;
        Ans += MOD; Ans %= MOD;
    }

    cout<<Ans<<endl;
}
