#include<stdlib.h>

using namespace std;
#define LL long long
LL qPow(LL a, LL k, LL p){LL ans=1;while(k){if(k&1)ans=(ans*a)%p;a=(a*a)%p;k/=2;}return ans;}
LL C(LL a,LL b,LL p)
{
    if(a<b)return 0;b=min(b,a-b);LL up=1,down=1;
    for(LL i=0;i<b;i++)up=up*(a-i)%p,down=down*(i+1)%p;
    return up*qPow(down,p-2,p)%p;
}
LL lucas(LL a,LL b,LL p){return b?C(a%p, b%p, p)*lucas(a/p, b/p, p)%p:1;}
const int maxn =25;const LL mod =1e9+7;int n;LL s, f[maxn];
LL solve()
{
    LL ans=0;
    for (int i=0;i<(1<<n);i++)
      {
      LL sign=1,sum=s;for(int j=0;j<n;j++)if(i&(1<<j))sum-=(f[j]+1),sign*=-1;
      if(sum<0)continue;ans+=sign*lucas(sum+n-1,n-1,mod);ans%=mod;
      }
    return (ans+mod)%mod;
}
int main(){scanf("%d%I64d",&n,&s);for(int i=0;i<n;i++)scanf("%I64d",&f[i]);printf("%I64d\n",solve());return 0;}
