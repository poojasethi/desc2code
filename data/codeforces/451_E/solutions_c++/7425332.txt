#include <iostream>
#include <cstdio>
using namespace std;
int n;
#define LL long long
#define mo 1000000007
LL f[25],G,ans,s;

void dfs(int k,LL d,LL cur){
    if (k == n){
        if (n + s - 1 - cur < n - 1) return;
        LL tmp = G;
        for (LL i = n + s - 1 - cur;i >= s - cur + 1;i--)
            tmp = (tmp * (i % mo)) % mo;
        ans = (ans + d * tmp) % mo;
        return;
    }
    dfs(k+1,d,cur);
    dfs(k+1,-d,cur+f[k]+1);
}

LL mulpow(LL x,LL k){
    LL a = x % mo, c = 1;
    while (k){
        if (k&1){
            c = (c * a) % mo;
        }
        a = (a * a) % mo;
        k >>= 1LL;
    }
    return c;
}

int main(){
    //freopen("in","r",stdin);
    cin>>n>>s;
    for (int i=0;i<n;i++){
        cin>>f[i];
    }
    G = 1;
    for (int i=2;i<n;i++)
        G = (G * mulpow(i,mo-2)) % mo;
    dfs(0,1,0);
    cout<<(ans+mo)%mo;
    return 0;
}
