#include <iostream>
#include <vector>
#define pb push_back
#define mp make_pair
#define ii pair<int,int>
#define I64 long long
#define I32 int
#define MOD 1000000007
using namespace std;
I64 N,S;
I64 arr[20],ans=0;
I64 pow(I64 a,I64 b,I64 mod){
	if(b==1)
		return a;
	I64 ans=pow(a,b/2,mod);
	return (ans*((ans*(b%2==1?a:1))%mod))%mod;
}
I64 dinv[30];
I64 inv(I64 a,I64 b){
	if(dinv[a]!=0)
		return dinv[a];
	return dinv[a]=pow(a,b-2,b);
}
I64 comb(I64 a,I64 b){
	if(a<0||b>a)
		return 0;
	if(a<=1)
		return 1;
	b=max(a-b,b);
	I64 ans=1;
	for(I64 i=b+1,j=1;i<=a;i++,j++){
		ans=((ans*(i%MOD))%MOD*inv(j,MOD))%MOD;
	}
	return ans;
}
int main(){
	cin >> N >> S;
	for(int i=0;i<N;i++)
		cin >> arr[i];
	for(int i=0,cnt;i<1<<N;i++){
		cnt=0;
		I64 sum=0;
		for(int j=0;j<N;j++)
			if((1<<j)&i)
				cnt++,sum+=arr[j]+1;
		ans+=(cnt%2==0?1:-1)*comb(N+S-1-sum,N-1);
		ans%=MOD;
	}
	ans=(ans+MOD)%MOD;
	cout << ans << endl;//316418090
	return 0;
}
