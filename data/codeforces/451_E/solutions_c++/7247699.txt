#include <cstdio>
#include <algorithm>
using namespace std;

typedef long long ll;
const int N=20,mo=1e9+7;

int n;
ll s,a[N];

inline ll power(ll a,ll b){
	ll ans=1;
	for(;b;b>>=1,a=a*a%mo)
		if(b&1) ans=ans*a%mo;
	return ans;
}

inline ll C(ll n,ll m){
	if(m<0||m>n) return 0;
	if(n-m<m) m=n-m;
	ll ans=1;
	for(int i=1;i<=m;i++) ans=ans*i%mo;
	ans=power(ans,mo-2);
	for(int i=1;i<=m;i++) ans=ans*((n-i+1)%mo)%mo;
	return ans;
}

int main(){
	scanf("%d%I64d",&n,&s);
	for(int i=0;i<n;i++) scanf("%I64d",a+i);
	ll ans=0;
	for(int t=0;t<1<<n;t++){
		ll sum=0,flag=1;
		for(int i=0;i<n;i++) if(t>>i&1){
			sum+=a[i]+1;
			flag*=-1;
		}
		ans=(ans+flag*C(n+(s-sum)-1,n-1))%mo;
	}
	printf("%I64d\n",(ans+mo)%mo);
}
