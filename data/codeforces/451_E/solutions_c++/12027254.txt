#include<cstdio>
#include<cstring>
#define LL long long
#define MIN(a,b) ((a)<(b)?(a):(b))
#define MAX(a,b) ((a)>(b)?(a):(b))
#define MOD 1000000007
LL n,s,ans,f[22];
LL ie[22];
LL pow(LL a,LL b,LL mod){
	LL ans=1;
	while(b){
		if(b&1) ans=(ans*a)%mod;
		a=(a*a)%mod;
		b>>=1;
	}
	return ans;
}
LL C(LL m,LL n){
	LL i,ans=1;
	m=MIN(m,n-m);
	for(i=0;i<m;++i){
		ans=(n-i)%MOD*ans%MOD;
		ans=ans*ie[i+1]%MOD;
	}
	return ans;
}
void dfs(LL t,LL sum,int cnt){
	if(sum>s){
		return ;
	}
	if(t==n){
		// odd
		if(cnt&1){
			ans=(ans-C(n-1,s-sum+n-1)+MOD)%MOD;
		}
		else{
			ans=(ans+C(n-1,s-sum+n-1))%MOD;
		}
		return ;
	}
	dfs(t+1,sum,cnt);
	dfs(t+1,sum+f[t]+1,cnt+1);
}
int main(){
	LL i;
	scanf("%I64d %I64d",&n,&s);
	for(i=0;i<n;++i){
		scanf("%I64d",&f[i]);
		ie[i+1]=pow(i+1,MOD-2,MOD);
	}
	ans=0;
	dfs(0,0,0);
	printf("%I64d",ans);
	return 0;
}
