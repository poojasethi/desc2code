#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
const int mod = 1e9+7;
int n;
long long s,f[20];
long long inverse(long long a){
	return a==1?1:(long long)(mod-mod/a)*inverse(mod%a)%mod;
}
long long choose(long long s, long long n){
	long long ans = 1;
	for(int i = 0; i < n; ++i)
		ans=((ans*((s-i)%mod))%mod*inverse(i+1))%mod;
	return ans;
}
long long solve(long long s, int cur){
	if(s<0) return 0;
	if(cur==n){
		return choose(s+n-1,n-1);
	}
	return (solve(s,cur+1)-solve(s-f[cur],cur+1))%mod;
}
int main(){
	cin >> n >> s;
	for(int i = 0; i < n; ++i){
		cin >> f[i];
		f[i]++;
	}
	cout << (solve(s,0)+mod)%mod << '\n';
}
