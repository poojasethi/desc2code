#include<iostream>
using namespace std;
#define mod 1000000007
typedef long long ll;
ll n,s,a[22],ans;
ll bin(ll a, ll n)
{
    ll ans=1;
    while(n)
    {
        if(n&1)
            ans*=a,ans%=mod;
        a*=a,a%=mod;
        n/=2;
    }
    return ans;
}
void f(ll pos, ll s, ll cnt)
{
    if(pos>n)
    {
        ll calc=1,con=1;
        if(s<n)
            return;
        for(ll i=s-n+1;i<=s-1;i++)
            calc*=(i%mod),calc%=mod;
        for(ll i=2;i<=n-1;i++)
            con*=i,con%=mod;
        calc*=bin(con,mod-2),calc%=mod;
        if(cnt&1)
            ans-=calc;
        else
            ans+=calc;
        ans=(ans+mod)%mod;
        return;
    }
    f(pos+1,s,cnt);
    f(pos+1,s-a[pos],cnt+1);
}
int main()
{
    ios_base::sync_with_stdio(0);

    cin>>n>>s;
    s+=n;
    for(ll i=1;i<=n;i++)
        cin>>a[i],a[i]++;
    f(1,s,0);
    cout<<ans;
}
