#include<bits/stdc++.h>
#include<cstdio>
using namespace std;
typedef long long LL;
const LL  Eps=1e9+7 ;
LL f[30];
LL C[300];
LL calc(LL n,LL m)
{
    LL ans=1;
    for(LL i=n;i>n-m;i--)
        ans=ans*(i%Eps)%Eps;
    for(LL i=1;i<=m;i++)
        ans=ans*C[i]%Eps;
    return ans;
}
main()
{
    C[1]=1;
    for(int i=2;i<100;i++)
        C[i]=Eps/i*(Eps-C[Eps%i])%Eps;
    LL n,s;
    cin>>n>>s;
    for(int i=0;i<n;i++)cin>>f[i];
    LL ans=0;
    for(int i=0;i<(1<<n);i++)
    {
        int cnt=0;
        LL tmp=s;
        for(int j=0;j<n;j++)
        {
            if(i&(1<<j))
            {
                cnt++;
                tmp-=f[j]+1;
            }
        }
        LL ret;
        if(tmp<0)ret=0;
        else ret=calc(tmp+n-1,n-1);
        if(cnt&1)ans-=ret;
        else ans+=ret;
        ans=(ans%Eps+Eps)%Eps;
    }
    cout<<ans;
    return 0;
}
