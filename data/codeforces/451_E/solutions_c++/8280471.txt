#include<cstdio>
#include<cstdlib>
#define LL long long
#define MOD 1000000007

LL n,fac_inv[25];
LL a[25];

inline LL C(LL x,LL y){
	LL ret=1;
	for(LL i=x;i>x-y;i--)ret=(ret*(i%MOD))%MOD;
	ret=(ret*fac_inv[y])%MOD;
	return ret;
}
LL solve(LL x,LL s)
{
	if(s<0)return 0;
	if(x==(n+1))return C(s+n-1,n-1);
	return (solve(x+1,s)-solve(x+1,s-a[x]-1)+MOD)%MOD;
}
inline LL power(LL x,LL y){
	LL ret=1;
	for(;y;y>>=1,x=(x*x)%MOD)
	if(y&1)ret=(ret*x)%MOD;
	return ret;
}
int main()
{
	LL s;LL value=1;
	for(LL i=1;i<=20;i++)value=value*i%MOD;
	fac_inv[20]=power(value,MOD-2);
	for(LL i=19;i+1;i--)fac_inv[i]=(fac_inv[i+1]*(i+1))%MOD;
	
	scanf("%I64d%I64d",&n,&s);
	for(LL i=1;i<=n;i++)scanf("%I64d",&a[i]);
	printf("%I64d\n",solve(1,s));
	return 0;
}
