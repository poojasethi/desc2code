#include <cstdio>
#include <cstdlib>
#define P 1000000007
using namespace std;

int n;
long long s,f[21],Ans,ny[21];

long long calc(long long a,long long b)
{
	long long ans=1;
	while (b)
	{
		if (b&1) ans=(ans*a)%P;
		a=(a*a)%P;
		b>>=1;
	}
	return ans;
}

long long get(long long s)
{
	long long ans=1;
	for (long long i=n+s-1;i>=s+1;i--) ans=(ans*(i%P))%P;
	for (int i=1;i<=n-1;i++) ans=(ans*ny[i])%P;	
	return ans;
}

void dfs(int t,long long sum,long long xs)
{
	if (sum>s) return;
	if (t>n)
	{
		Ans=(Ans+xs*get(s-sum))%P;
		return;
	}
	dfs(t+1,sum,xs);
	dfs(t+1,sum+f[t],-xs);
}

int main()
{
//	freopen("451E.in","r",stdin);
//	freopen("451E.out","w",stdout);
	scanf("%d%I64d",&n,&s);
	for (int i=1;i<=n;i++) ny[i]=calc(i,P-2);
	for (int i=1;i<=n;i++) scanf("%I64d",&f[i]),f[i]++;
	dfs(1,0,1);
	// pi ( 1 - x^f(i) )
	printf("%I64d\n",(Ans+P)%P);
}
