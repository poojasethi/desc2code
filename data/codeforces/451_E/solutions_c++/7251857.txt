#define MOD 1000000007
long long int a[25];
long long int n, s;
int inv(int a) {
	return a == 1 ? 1 : (long long)(MOD - MOD / a) * inv(MOD % a) % MOD;
}

long long choose(long long s, long long n) {
	long long ret = 1;
	for (int i = 0; i < n; ++i) {
		ret = (long long)ret * ((s - i) % MOD) % MOD * inv(i + 1) % MOD;
	}
	return ret;
}
long long int solve(long long int i, long long int s)
{
	if (s < 0)
		return 0;
	if (i == n)
		return choose(s + n - 1, n - 1);
	else
		return (solve(i+1,s)-solve(i+1,s-a[i]))%MOD;
}
int main()
{
	scanf("%lld%lld", &n, &s);
 	for (int i = 0; i < n; i++)
	{
		scanf("%lld", a + i);
		a[i]++;
	}
	printf("%lld", (solve(0, s)+MOD)%MOD);
	return 0;
}
