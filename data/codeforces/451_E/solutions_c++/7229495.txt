#include <stdio.h>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;
#define ll long long
const ll inf=0x7fffffff/11.27;
const ll Eps=1e9+7 ;
ll f[30];
ll C[300];
ll i,j,t,n,m,l,r,k,z,y,x,s,ans,cnt,tmp,ret;
ll calc(ll n,ll m)
{
    ll i,j,t,ans=1;
    for(i=n;i>n-m;i--) ans=ans*(i%Eps)%Eps;
    for(i=1;i<=m;i++) ans=ans*C[i]%Eps;
    return ans;
}
main()
{
    C[1]=1;
    for(i=2;i<100;i++) C[i]=Eps/i*(Eps-C[Eps%i])%Eps;
    cin>>n>>s;
    for(i=0;i<n;i++) cin>>f[i];
    ans=0;
    for(i=0;i<(1<<n);i++)
    {
        cnt=0;tmp=s;
        for(j=0;j<n;j++)
        {
            if(i&(1<<j))
            {
                cnt++;
                tmp-=f[j]+1;
            }
        }
        if (tmp<0) ret=0;
        else ret=calc(tmp+n-1,n-1);
        if (cnt&1) ans-=ret;
        else ans+=ret;
        ans=(ans%Eps+Eps)%Eps;
    }
    cout<<ans<<endl;
}
