#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

typedef long long LL;
LL s;
int n;
LL a[100];
int ans;
const int P = 1e9 + 7;
int b[100];
int c[100];

int calc(LL n, LL m){
	int ans = 1;
	for (int i = 1; i <= m; ++i) {
		ans = (LL)ans * ((n - i + 1) % P) % P;
		ans = (LL)ans * c[i] % P;
	}
	return ans;
}

void dfs(int x, int sum, LL s){
	if (s < 0) return ;
	if (x > n) {
		ans += (sum * 2 - 1) * calc(s + n - 1, n - 1);
		if (ans < 0) ans += P;
		ans %= P;
		return ;
	}
	dfs(x + 1, sum, s);
	dfs(x + 1, sum ^ 1, s - a[x] - 1);
}

int main(){
	scanf("%d%I64d", &n, &s);
	for (int i = 1; i <= n; ++i)
		scanf("%I64d", &a[i]);
	ans = 0;
	c[1] = 1;
	for (int i = 2; i <= n; ++i)
		c[i] = (LL)(P - P / i) * c[P % i] % P;
	dfs(1, 1, s);
        
	printf("%d\n", ans);
}
