#include<cstdio>
#define N 21
using namespace std;
typedef long long LL;
const LL P=1000000007ll;
LL f[N],sum,all,sign,ans;
int n,i,j;
inline LL ni(LL a,LL b)
{
  LL res=1;
  for (;b;b>>=1,a=a*a%P) if (b&1) res=res*a%P;
  return res%P;
}
inline LL C(LL n,LL m)
{
  LL res=1,mo=1;
  for (int i=1;i<=m;i++)
  {
    res=res*1ll*(n-m+i)%P;
    mo=mo*1ll*i%P;
  }
  return res*ni(mo,P-2)%P;
}
inline LL Lucas(LL n,LL m)
{
  if (!m) return 1ll;
  return Lucas(n/P,m/P)*C(n%P,m%P)%P;
}
int main()
{
  scanf("%d%I64d",&n,&sum);
  for (i=1;i<=n;i++) scanf("%I64d",&f[i]);
  int cnt=0;
  for (i=0;i<(1<<n);i++)
  {
    sign=1;all=sum;cnt++;
    for (j=1;j<=n;j++)
      if (i&(1<<(j-1))) sign*=-1,all-=f[j]+1;
    if (all<0) continue;
    LL temp=sign*Lucas(n+all-1,n-1);
    ans+=temp;if (ans<0) ans=(ans%P+P)%P;else ans%=P;
  }
  printf("%I64d",ans);
  return 0;
}
