#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
const int mod = 1e9+7;
int n;
long long s,f[20];
long long extended_euclid(long long a, long long x, long long b, long long y){
	if(b==1)
		return y;
	return extended_euclid(b,y,a-(a/b)*b,x-(a/b)*y);
}
long long inverse(long long a){
	return extended_euclid(mod,0,a,1);
}
long long choose(long long s, long long n){
	long long ans = 1;
	for(int i = 0; i < n; ++i)
		ans=((ans*((s-i)%mod))%mod*inverse(i+1))%mod;
	return ans;
}
long long solve(long long s, int cur){
	if(s<0) return 0;
	if(cur==n){
		return choose(s+n-1,n-1);
	}
	return (solve(s,cur+1)-solve(s-f[cur],cur+1))%mod;
}
int main(){
	cin >> n >> s;
	for(int i = 0; i < n; ++i){
		cin >> f[i];
		f[i]++;
	}
	cout << (solve(s,0)+mod)%mod << '\n';
}
