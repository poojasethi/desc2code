#include<stdio.h>
#include<string.h>
#define MOD 1000000007
#define maxn 1000
__int64 num[maxn], s;
__int64 mul(__int64 a,__int64 b)
{
	__int64 tem = 1;
	while(b)
	{
		if(b&1)
		{
			tem = tem*a%MOD;
		}
		a = a*a%MOD;
		b>>=1;
	}
	return tem;
}
__int64 C(__int64 n,__int64 m)
{
	if(n<m) return 0;
	__int64 a = 1, b = 1, i;
	for(i=1;i<=m;i++)
	{
		a = a*i%MOD;
		b = b*((n-i+1)%MOD)%MOD;
	}
	return b*mul(a,MOD-2)%MOD;
}
int main()
{
	int n, i, j;
	scanf("%d %I64d",&n,&s);
		//printf("%I64d\n",C(100000000000000ll,20));
	for(i=1;i<=n;i++) scanf("%I64d",&num[i]);

	__int64 ans = 0;
	for(i=0;i<(1<<n);i++)
	{
		__int64 sum = s, flag = 1;
		for(j=0;j<=20;j++)
		{
			if(i&(1<<j)) sum-=num[j+1]+1,flag*=-1;
		}
		if(sum<0) continue;
	//	printf("%d %lld\n",i,C(sum+n-1,n-1));
		ans = ans+flag*C(sum+n-1,n-1);
		ans=ans+2LL*MOD;
		ans%=MOD;
	}
	printf("%I64d\n",ans);
	return 0;
}
