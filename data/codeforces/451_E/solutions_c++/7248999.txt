#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const ll MOD = 1000000007;
const int N = 20;
ll n, s, f[N], sgn[2] = {1, -1}, res = 0;

ll inv(ll a){
  return a == 1 ? 1 : (MOD - MOD / a) * inv(MOD % a) % MOD;
}

ll binom(ll a, ll b){
  if(a < b) return 0;
  ll res = 1;
  for(int i = 0; i < b; i++){
    res = (a - i) % MOD * inv(b - i) % MOD * res % MOD;
  }
  return res;
}

void dfs(int d, ll t, int p){
  if(d == n) res = (res + sgn[p % 2] * binom(s - t + n - 1, n - 1) + MOD) % MOD;
  else {
    dfs(d + 1, t, p);
    dfs(d + 1, t + f[d] + 1, p + 1);
  }
}

int main(){
  cin >> n >> s;
  for(int i = 0; i < n; i++) cin >> f[i];
  dfs(0, 0, 0);
  cout << res << '\n';
}
