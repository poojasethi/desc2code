#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll  MM = 1e9 + 7 ;
ll f[30];
ll I[300];
ll C(ll n,ll m)
{
    ll ans = 1;
    for(ll i = n; i>n-m; i--)
    {
        ans = ans*(i%MM)%MM;
    }
    for(ll i = 1; i<=m; i++)
    {
        ans = ans*I[i]%MM;
    }
    return ans;
}
int main()
{
    I[1] = 1;
    for(int i = 2; i<100; i++)I[i] = MM/i*(MM-I[MM%i])%MM;
    ll n,s;
    cin>>n>>s;
    for(int i = 0; i<n; i++) cin>>f[i];
    ll ans = 0;
    for(int i = 0; i < (1<<n) ; i++)
    {
        int cnt = 0;
        ll tmp = s;
        for(int j = 0; j<n; j++)
        {
            if(i&(1<<j))
            {
                cnt++;
                tmp-=f[j]+1;
            }
        }
        ll ret;
        if(tmp<0) ret = 0;
        else ret = C(tmp+n-1,n-1);
        if(cnt&1) ans-=ret;
        else ans+=ret;
        ans = (ans%MM+MM)%MM;
    }
    cout<<ans<<endl;
}
