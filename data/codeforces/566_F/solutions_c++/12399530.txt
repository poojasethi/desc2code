#include<bits/stdc++.h>
using namespace std;
int arr[1000002];
int a[1000002];
int main(){
    int n;
    cin>>n;
    memset(arr,0,sizeof arr);
    int ans=0;
    for(int i=1; i<=n; i++) scanf("%d",&a[i]);
    for(int i=n; i>0; i--){
        arr[a[i]]=1;
        for(int j=2; a[i]*j<=a[n]; j++){
            arr[a[i]]=max(arr[a[i]],arr[a[i]*j]+1);
        }
        ans=max(ans,arr[a[i]]);
    }
    cout<<ans<<endl;
}
