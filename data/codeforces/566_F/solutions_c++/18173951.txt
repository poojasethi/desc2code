#include<bits/stdc++.h>
using namespace std;
int h[1000001];
int main()
{
    int n,m,i,j,k,l=0;
    scanf("%d",&n);
    for(i=0;i<n;++i)
    {
        scanf("%d",&m);
        ++h[m];
        for(j=2*m;j<=1000000;j+=m)
            h[j]=max(h[j],h[m]);
        l=max(l,h[m]);
    }
    printf("%d",l);
    return 0;
}
