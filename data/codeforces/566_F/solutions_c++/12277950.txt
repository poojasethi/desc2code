#include <cstdio>
#include <algorithm>
#define repu(i,x,y) for (int i=x; i<=y; ++i)
using namespace std;

int n,x,s[1000100],f[1000100],ans;

int main()
{
    for (scanf("%d",&n); n--;)
        scanf("%d",&x),++s[x];
    repu(i,1,1000000)
    {
        ans=max(ans,f[i]+=s[i]);
        for (int j=2; i*j<=1000000; ++j)
            f[i*j]=max(f[i*j],f[i]);
    }
    printf("%d\n",ans);
    return 0;
}
