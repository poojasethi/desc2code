#include <cstdio>
#include <algorithm>
using namespace std;

int n, a[1000001], b[1000001], c[1000001];

int main() {
  scanf("%d", &n);
  for (int i = 1; i <= n; i++) {
    scanf("%d", a + i);
    b[a[i]] = i;
  }
  for (int i = 1; i < n; i++)
    for (int j = 2 * a[i]; j <= a[n]; j += a[i])
      if (b[j])
        c[b[j]] = max(c[b[j]], c[i] + 1);
  printf("%d\n", 1 + *max_element(c, c + n + 1));
}
