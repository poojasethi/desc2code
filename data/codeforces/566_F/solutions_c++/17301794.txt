#include<iostream>
#include<cstring>
#include<cstdio>
#include<cstdlib>
#include<algorithm>

using namespace std;

int dp[1000100];
int n,m,x;
int main()
{
    while(~scanf("%d",&n))
	{
		int Max=0;
		memset(dp,0,sizeof(dp));
		for(int i=0;i<n;i++)
		{
			scanf("%d",&x);
			dp[x]++;
				Max=max(Max,dp[x]);
			for(int j=2;j*x<=1000001;j++)
			{
				dp[j*x]=max(dp[j*x],dp[x]);
			}

		}
		printf("%d\n",Max);
	}
}
