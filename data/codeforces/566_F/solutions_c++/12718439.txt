#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
int n,u,i,j,k;
int d[1000004];
int ans;
int main(){
	scanf("%d", &n);
	for(i = 0; i <n ; i++){
		scanf("%d",&k);
		d[k]++;
		ans = max(d[k], ans);
		for(j = k*2; j <= 1000003; j+=k){
			d[j] = max(d[j], d[k]);
		}
	}
	printf("%d\n", ans);
	return 0;
}
