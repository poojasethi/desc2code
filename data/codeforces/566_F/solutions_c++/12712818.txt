#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int n,arr[1000009];
int dp[1000009];
int main(){
    cin>>n;
    int ans=0;
    for(int i=0;i<n;i++){
        int x;
        scanf("%d",&x);
        dp[x]++;
        ans=max(ans,dp[x]);
        for(int j=x*2;j<=1000000;j+=x){
            dp[j]=max(dp[j],dp[x]);
        }
    }
    cout<<ans<<endl;
}
