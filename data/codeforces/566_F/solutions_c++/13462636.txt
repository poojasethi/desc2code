#include <stdio.h>
#include <algorithm>
using namespace std;
int n,vet[1000010],m[1000010],maxi;
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&vet[i]);
    }
    for(int i=0;i<n;i++){
        m[vet[i]]++;
        maxi=max(maxi,m[vet[i]]);
        for(int j=2*vet[i];j<=1000000;j+=vet[i]){
            m[j]=max(m[j],m[vet[i]]);
        }
    }
    printf("%d\n",maxi);
    return 0;
}
