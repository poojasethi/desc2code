#include <algorithm>
#include <cstdio>
int a[1000010],f[1000010],i,j,n,x,mx,ans;
int main()
{
	scanf("%d\n",&n);
	for (i=1;i<=n;++i)
	{
		scanf("%d",&x);
		a[x]++;
		mx=std::max(mx,x);
	}
	for (i=1;i<=mx;++i)
		if (a[i])
			for (f[i]+=a[i],j=2;i*j<=mx;++j)
					f[i*j]=std::max(f[i*j],f[i]);
	for (i=1;i<=mx;++i)
		if (a[i]) ans=std::max(ans,f[i]);
	printf("%d\n",ans);
}
