#define N (1<<20)
#include <bits/stdc++.h>
using namespace std;

bool f[N];
int n,x,dp[N],res;

int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
		scanf("%d",&x),f[x]=1;

	for(int i=1;i<N;i++)
		if(f[i])
		{
			dp[i]++;
			for(int j=i+i;j<N;j+=i)
				dp[j]=max(dp[j],dp[i]);
			res=max(res,dp[i]);
		}

	cout<<res<<endl;
	return 0;
}
