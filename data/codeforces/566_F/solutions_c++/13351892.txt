#include<bits/stdc++.h>

using namespace std;

int const MAXN = 1e6+10;

int n, v[MAXN], dp[MAXN];

int main(){
    cin.sync_with_stdio(0);
    cin >> n;
    int ans = 0;
    for(int i = 1; i <= n; i++){
        cin >> v[i];
        dp[v[i]]++;
        ans = max(ans,dp[v[i]]);
        for(int j = 2*v[i]; j <= MAXN; j+=v[i]) 
                    dp[j] = max(dp[j],dp[v[i]]);
    }
    cout << ans << '\n';
    return 0;
}
