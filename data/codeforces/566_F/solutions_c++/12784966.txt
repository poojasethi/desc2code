#include <bits/stdc++.h>
using namespace std;

#define ll int

ll n;
ll dp[1000005];

int main() {
    scanf("%d",&n);
    ll mx = 1;
    for (ll i = 0; i < n; i++) {
        ll num;
        scanf("%d", &num);
        dp[num] = max(dp[num],1);
        for (ll j = num+num; j < 1000005; j+= num)
            dp[j] = max(dp[j],dp[num]+1);
        mx = max(mx,dp[num]);
    }
    printf("%d\n",mx);
    return 0;
}
