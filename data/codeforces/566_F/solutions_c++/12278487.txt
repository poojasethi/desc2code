#include <iostream>
#include <cstdio>
using namespace std;

int a;
int M[1000009]={0};
int count[1000009]={0}; 

int main() {

    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {scanf("%d",&a);count[a]++;}
    int L=1;
    for(int i=1000000;i>=1;i--)
    {
    for(int j=i;j<=1000000;j+=i)
    M[i]=max(M[i],M[j]+count[i]);
    L=max(M[i],L);
    }
    cout<<L<<"\n";
    return 0;
}
