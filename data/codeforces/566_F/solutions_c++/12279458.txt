#include <cstdio>

int main(){
	int n=0;
	int*R = new int[1000001];
	scanf("%d", &n);
	int*A = new int[n];
	for(int i=0;i<n;i++){
		scanf("%d", &(A[i]));
	}
	int res =1;
	for(int i=n-1;i>=0;i--){
		int x = A[i];
		R[x]=1;
		for(int j=2;x*j<1000001;j++){
			if(R[x*j]>=R[x])R[x]= 1+R[x*j];
		}
		if(R[x]>res)res = R[x];
	}
	printf("%d\n", res);
}
