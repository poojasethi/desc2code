#include<algorithm>
#include<cstdio>
using namespace std;

const int maxn=1000010;
int num[maxn];
int n;

int main(){
	scanf("%d",&n);
	int res=0;
	for(int i=1;i<=n;++i){
		int x;scanf("%d",&x);res=max(res,++num[x]);
		for(int j=2*x;j<=1000000;j+=x)num[j]=max(num[j],num[x]);
	}
	printf("%d\n",res);
	return 0;
}
