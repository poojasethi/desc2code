#include<bits/stdc++.h>
using namespace std;

int a[1000006],dp[1000006];

int main()
{
    int n,i,j,mx,ans=-1;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    sort(a,a+n);
    for(i=n-1;i>=0;i--)
    {
        for(j=1;j*a[i]<=1000000;j++)
            dp[ a[i] ]= max(dp[a[i]], dp[a[i]*j]+1);
    }
    for(i=1;i<=1000000;i++)
        ans=max(ans,dp[i]);
    printf("%d\n",ans);
}
