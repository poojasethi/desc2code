#include <bits/stdc++.h>
using namespace std;

int n;
int a[1000000+5];
int dp[1000000+5];

int main () {
	int n;
	scanf ("%d", &n);
	for (int i=1; i<=n; i++) {
		scanf ("%d", &a[i]);
		dp[a[i]] = 1;
	}
	int ans = 0;
	for (int i=1; i<=n; i++) {

		ans = max(ans, dp[a[i]]);
		int s = 2;
		while(a[i]*s <= a[n]) {
			dp[a[i]*s] = max(dp[a[i]*s], dp[a[i]] + 1);
			s++;
		}
	}

	printf("%d\n", ans);

	return 0;
}
