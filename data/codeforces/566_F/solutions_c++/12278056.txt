#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,s[1000005],num,ans=0;
int main(){
	scanf("%d",&n);
	memset(s,0,sizeof(s));
	for(int i=1;i<=n;i++){
		scanf("%d",&num);
		ans=max(ans,s[num]+1);
		for(int j=num+num;j<=1000000;j+=num) s[j]=max(s[j],s[num]+1);
	}
	printf("%d\n",ans);
	return 0;
}
