#include <cstdio>
#include <algorithm>
using namespace std;
int a,k[1111112],m=1,n;
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&a); k[a]=1;
    }
    int M=a;
    for(int i=1;i<=M;i++) if (k[i])
        for(int a=2*i;a<=M;a+=i) if (k[a])
            k[a]=max(k[a],k[i]+1),m=max(m,k[a]);
    printf("%d",m);
}
