
#include <bits/stdc++.h>

using namespace std;

int s[1000005];
int dp[1000005];
int main(void)
{
	int n;
	scanf("%d",&n);
	int i,j;
	for(i=0;i<n;i++)
	  scanf("%d",&s[i]);
	for(i=0;i<=1000000;i++)
	  dp[i]=1;
	int ans=0;
	for(i=0;i<n;i++)
	{
		ans=max(ans,dp[s[i]]);
		for(j=2*s[i];j<=1000000;j+=s[i])
		  dp[j]=max(dp[j],dp[s[i]]+1);
	}
	printf("%d\n",ans);
	return 0;
}
