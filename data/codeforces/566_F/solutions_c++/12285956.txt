#include<cstdio>
#include<algorithm>
#define N 1000010
using namespace std;
int n,i,j,ans,f[N],a[N];
int main()
{
    scanf("%d",&n);
    for (i=1;i<=n;i++)
        scanf("%d",&a[i]);
    sort(a+1,a+1+n);
    for (i=1;i<=1000000;i++)
    f[i]=1;
    for (i=1;i<=n;i++)
    {
    if (f[a[i]]>ans) ans=f[a[i]];
    for (j=2;j<=1000000/a[i];j++)
    f[j*a[i]]=max(f[j*a[i]],f[a[i]]+1);
    }
    printf("%d\n",ans);
}
