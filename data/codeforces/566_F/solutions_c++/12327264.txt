#include<bits/stdc++.h>
using namespace std;
int dp[1000111],a[1000111];
int main()
{
    int i,j,n,k;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    sort(a,a+n);
    for(i=n-1;i>=0;i--)
    {
        for(j=1;(a[i]*j)<=1000000;j++)
            dp[a[i]]=max(dp[a[i]],dp[a[i]*j]+1);
    }
    int ans=-1;
    for(i=1;i<=1000000;i++)
        ans=max(ans,dp[i]);
    printf("%d\n",ans);
    return 0;
}
