#include "iostream"
using namespace std;
long long n,wynik,maks,tab[1000005],w[1000005];
int main()
{
	ios_base::sync_with_stdio(0);
	cin>>n;
	for (int i=0; i<n; i++)
	{
		cin>>tab[i];
		if (tab[i]>maks) maks=tab[i];
	}
	for (int i=0; i<n; i++)
	{
		w[tab[i]]++;
		for (int j=tab[i]; j<=maks; j+=tab[i]) if (w[tab[i]]>w[j]) w[j]=w[tab[i]];
	}
	for (int i=0; i<=maks; i++) if (w[i]>wynik) wynik=w[i];
	cout<<wynik<<"\n";
	
	return 0;
}
