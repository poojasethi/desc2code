#include <bits/stdc++.h>
using namespace std;

const int maxN=1e6+10;

int f[maxN];
int n;

int main() {
    int res=0;
    cin>>n;
    for(int i=0;i<n;i++) {
        int x;
        scanf("%i",&x);
        ++f[x];
        for(int j=1;j<=1000000/x;j++) f[j*x]=max(f[j*x],f[x]);
        res=max(res,f[x]);
    }
    cout<<res;
}
