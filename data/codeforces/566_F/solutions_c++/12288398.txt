#include<cstdio>
#include<algorithm>
using namespace std;
int n,a[1000005],x,ac;
int main(){
	scanf("%d",&n);
	for (int i=1; i<=1000000; i++) a[i]=0;
	ac=0;
	for (int i=1 ;i<=n; i++)
		scanf("%d",&x),++a[x];
	for (int i=1; i<=1000000; i++){
		if (a[i]){
			ac=max(ac,a[i]);
			for (int j=i*2; j<=1000000; j+=i)
				if (a[j]) a[j]=max(a[j],a[i]+1);
		}
	}
	printf("%d\n",ac);
	return 0;
}
