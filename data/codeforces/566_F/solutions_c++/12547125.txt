#include<bits/stdc++.h>
using namespace std;
int dp[1000000],a;
int main()
{
    int n,maxn;
    scanf("%d",&n);
    maxn =0;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a);
        dp[a]++;
        maxn = max(maxn,dp[a]);
        for(int j=a*2;j<=1000000;j+=a)
            dp[j]=max(dp[j],dp[a]);
    }
    printf("%d\n",maxn);

    return 0;
}
