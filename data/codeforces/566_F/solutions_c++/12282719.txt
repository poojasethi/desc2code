#include <iostream>

using namespace std;

const int maxa = 1e6 + 7;

int cnt[maxa];
int dp[maxa];

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	int n;
	cin >> n;
	for (int i = 0; i < n; ++i) {
		int x;
		cin >> x;
		cnt[x] ++;
	}
	int ans = 1;
	for (int i = maxa - 1; i >= 1; --i) {
		if (cnt[i])
			for (int j = i; j < maxa; j += i)
				dp[i] = max(dp[i], cnt[i] + dp[j]);
		ans = max(ans, dp[i]);
	}
	cout << ans << '\n';
	return 0;
}
