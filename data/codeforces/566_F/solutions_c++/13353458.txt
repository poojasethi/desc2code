#include<bits/stdc++.h>
#define MAX 1000001
using namespace std;
int n,a[1000005],h[1000005],mx;
int main ()
{
	int i;
	cin>>n;
	for(i=0; i<n; i++)
		scanf("%d",a+i);
	for(i=0; i<n; i++)
	{
		int x=a[i];
		h[x]++;
		for(int j=x; j<MAX; j+=x)
			h[j]=max(h[x],h[j]);
	}
	for(i=0; i<MAX; i++)
	  mx=max(h[i],mx);
	cout<<mx;
}
