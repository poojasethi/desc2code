#include<cstdio>
#include<algorithm>
using namespace std;
#define maxn 1000010
int a[maxn],dp[maxn],n;

int main()
{
	scanf("%d",&n);
	for(int i=1; i<=n; i++)
		scanf("%d",&a[i]);
	int ans=0;
	for(int i=1; i<=n; i++)
	{
		dp[a[i]]=max(dp[a[i]],1);
		if(dp[a[i]]>ans) ans=dp[a[i]];
		for(int j=2; j*a[i]<=a[n]; j++)
			dp[j*a[i]]=max(dp[j*a[i]],dp[a[i]]+1);
	}
	printf("%d\n",ans);
	return 0;
}
