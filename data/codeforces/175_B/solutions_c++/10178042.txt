#include <bits/stdc++.h>

using namespace std;

map<string, int> m;

int main(){
	int n, a, temp1, temp2, x, y;
	cin>>n;

	string s;
	for(int i=0 ; i<n ; i++){
		cin>>s>>a;
		if(m.find(s) == m.end()){
			m[s] = a;
		} else {
			m[s] = max(m[s], a);
		}
	}

	int count = 0;
	pair<int, string> p[n];
	map<string, int>::iterator i;
	for(i = m.begin() ; i != m.end() ; i++){
		p[count++] = make_pair(i->second, i->first);
	}

	sort(p, p+count);
	cout<<count<<endl;
	cout<<p[count-1].second<<" pro"<<endl;

	x = count, y = 0; 
	for(int i=count-2 ; i>=0 ; i--){
		if(p[i].first < p[i+1].first){
			x--;
			x -= y;
			y = 0;
		} else {
			y++;
		}
		temp1 = (x * 100) / count;
		cout<<p[i].second<<" ";
		if(temp1>=99){
			cout<<"pro"<<endl;
		} else if(temp1>=90){
			cout<<"hardcore"<<endl;
		} else if(temp1>=80){
			cout<<"average"<<endl;
		} else if(temp1>=50){
			cout<<"random"<<endl;
		} else {
			cout<<"noob"<<endl;
		}
	}

	return 0;
}
