#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char b[5][10]={"noob","random","average","hardcore","pro"};
struct rec{char s[15]; int t,p;}a[1010];
int v[1010],m,n,i,j,k;

bool cmp(rec a,rec b)
{
	int k=strcmp(a.s,b.s);
	return k<0||k==0&&a.t>b.t;
}

int main()
{
	cin>>n;
	for(i=1;i<=n;i++) scanf("%s%d",a[i].s,&a[i].t);
	sort(a+1,a+n+1,cmp);
	for(i=1;i<=n;i++)
		if(i==1||strcmp(a[i].s,a[i-1].s)) v[i]=1,m++;
	for(i=1;i<=n;i++)
		if(v[i])
		{
			k=0;
			for(j=1;j<=n;j++)
				if(v[j]&&a[j].t>a[i].t) k++;
			if(k*2>m) a[i].p=0;
			else if(k*5>m) a[i].p=1;
			else if(k*10>m) a[i].p=2;
			else if(k*100>m) a[i].p=3;
			else a[i].p=4;
		}
	cout<<m<<endl;
	for(i=1;i<=n;i++) 
		if(v[i]) printf("%s %s\n",a[i].s,b[a[i].p]);
	return 0;
}
