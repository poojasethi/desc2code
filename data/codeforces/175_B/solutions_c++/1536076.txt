#include <iostream>
#include <map>
#include <string>
#include <vector>
#include <cmath>

using namespace std;

int main()
{
  int n;
  cin>>n;
  map<string,int> m;
  for (int i=0;i<n;i++) {
    string name;
    int val;
    cin>>name>>val;
    m[name]=max(m[name],val);
  }
  vector<string> nombre;
  vector<int> valor;
  for (map<string,int>::iterator it=m.begin();it!=m.end();it++) {
    nombre.push_back(it->first);
    valor.push_back(it->second);
  }
  int total=int(valor.size());
  cout<<total<<endl;
  for (int i=0;i<int(valor.size());i++) {
    int val=valor[i];
    int q=0;
    for (int j=0;j<int(valor.size());j++)
      if (valor[j]>val) q++;
    string cat;
    if (100*q>total*50) cat="noob";
    else if (100*q>total*20) cat="random";
    else if (100*q>total*10) cat="average";
    else if (100*q>total*1) cat="hardcore";
    else cat="pro";
    cout<<nombre[i]<<" "<<cat<<endl;
  }
}
