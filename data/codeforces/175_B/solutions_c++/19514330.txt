#include<iostream>
#include<cstdio>
#include<map>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
	int n;
	map<string,int>mp;
	scanf("%d",&n);
	int i;
	for(i=1;i<=n;i++)
	{
		string a;
		int d;
		cin>>a;
		scanf("%d",&d);
		mp[a]=max(mp[a],d);
	}
	map<string,int>::iterator ii,jj;
	int ans=0;
	for(ii=mp.begin();ii!=mp.end();ii++) ans++;
	printf("%d\n",ans);
	if(ans==0) return 0;
	for(ii=mp.begin();ii!=mp.end();ii++)
	{
		
		int a=0;
		int b=0;
		for(jj=mp.begin();jj!=mp.end();jj++)
		{
			if(jj->second > ii->second) a++;
			else if(jj->second<ii->second) b++;
			
		}
		double f=a;
		f=f/ans;
		cout<<ii->first <<" ";
		if(f>0.500) puts("noob");
		else if(f>0.200) puts("random");
		else if(f>0.1000) puts("average");
		else if(f>0.0100) puts("hardcore");
		else puts("pro");
	}
	return 0;
}
