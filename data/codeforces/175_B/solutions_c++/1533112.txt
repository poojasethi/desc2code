#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
int n,x,flag,now=0;
int score[1005];
char Name[1005][15],s[15];
double p[4];
inline int max(int a,int b){return a>b?a:b;}
void prt(int x)
{
	if (x==0) puts("noob");
	if (x==1) puts("random");
	if (x==2) puts("average");
	if (x==3) puts("hardcore");
	if (x==4) puts("pro");
}
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++){
		scanf("%s%d",s,&x); flag=1;
		for (int j=1;j<=now;j++)
			if (strcmp(Name[j],s)==0) {score[j]=max(score[j],x); flag=0; break;}
		if (flag) now++,memcpy(Name[now],s,sizeof(Name[now])),score[now]=x;
		}
	n=now; p[0]=n*0.5; p[1]=n*0.8; p[2]=n*0.9; p[3]=n*0.99;
	printf("%d\n",n);
	for (int i=1;i<=n;i++){
		int sum=0,k;
		for (int j=1;j<=n;j++)
			if (score[j]<=score[i]) sum++;
		for (k=0;k<4;k++) if (sum<p[k]) break;
		cout<<Name[i]<<" "; prt(k);
		}
	return 0;
}
