#include <iostream>
#include <map>
using namespace std;

int n, id = 0;
map<string, int> ID;
map<int, string> Name;
string name[1000];
int score[1000];
int cat[1000];

string Switch(int r)
{
	if (2 * r > id) return "noob";
	if (5 * r > id) return "random";
	if (10 * r > id) return "average";
	if (100 * r > id) return "hardcore";
	return "pro";
}

int main()
{
	cin >> n;
	string s;
	int a;
	for (int i = 0; i < n; i++)
	{
		cin >> s >> a;
		name[i] = s;
		if (ID.find(s) == ID.end())
		{
			ID[s] = id;
			Name[id] = s;
			score[id] = a;
			id++;
		}
		else
		{
			score[ID[s]] = max(score[ID[s]], a);
		}
	}
	cout << id << endl;
	for (int i = 0; i < id; i++)
	{
		int r = 0;
		for (int j = 0; j < id; j++)
		{
			if (score[i] < score[j]) r++;
		}
		cout << Name[i] << " " << Switch(r) << endl;
		//cout << Name[i] << " " << score[i] << " " << r << endl;
	}
	return 0;
}
