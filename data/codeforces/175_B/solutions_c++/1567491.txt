#include <map>
#include <stdio.h>
#include <iostream>
#include <algorithm>

using namespace std;

map <string, int> a;

int main()
{int n, x;
string s;
//freopen ("in.txt","r",stdin);
cin >> n;
for (int i=0; i<n; i++)
{
    cin >> s >> x;
    a[s]=max(x,a[s]);
}
cout << a.size() << endl;
map <string, int>::iterator it1=a.begin();
for (; it1!=a.end(); it1++)
{
    int k=0;
    map <string, int>::iterator it2=a.begin();
    for (; it2!=a.end(); it2++)
    if (it1->second>=it2->second) k++;
    cout << it1->first << " ";
    k=k*100/a.size();
    if (k<50) {cout << "noob" << endl;} else
    if (k<80) {cout << "random" << endl;} else
    if (k<90) {cout << "average" << endl;} else
    if (k<99) {cout << "hardcore" << endl;} else {cout << "pro" << endl;}
}
return 0;
}
