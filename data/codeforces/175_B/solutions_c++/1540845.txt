#include <iostream>
#include <map>
#include <stdio.h>
using namespace std;

int main()
{
//freopen("i.txt","r",stdin);
map<string,int> a;
int n;
cin>>n;
string st;int x;
for(int i=0;i<n;i++)
 {
  cin>>st>>x;
  if (x>a[st]) a[st]=x;
 }
cout<<a.size()<<endl;
map<string,int>::iterator it=a.begin();
for(;it!=a.end();it++)
{ int kol=0;
  map<string,int>::iterator it1=a.begin();
  for(;it1!=a.end();it1++) if (it->second>=it1->second) kol++;
  cout<<it->first<<" ";
  int k=kol*100/a.size();
  if (k<50) cout<<"noob"<<endl;
   else if (k<80) cout<<"random"<<endl;
   else if (k<90) cout<<"average"<<endl;
   else if (k<99) cout<<"hardcore"<<endl;
   else  cout<<"pro"<<endl;
}
return 0;
}
