#include <iostream>
#include <map>

using namespace std;

int main() {
  int n;
  cin >> n;
  map <string, int> player;
  
  for (int i = 0 ; i < n; i++) {
    string s;
    int a;
    cin >> s >> a;
    player[s] = max(player[s], a);
  }
  
  map <string, int>::iterator it, cur;
  
  int num = player.size();
  cout << num;
  for (it = player.begin(); it != player.end(); it++) {
    int x = 0;
    for (cur = player.begin(); cur != player.end(); cur++)
      if ((*it).second >= (*cur).second)
        x++;
    cout << endl << (*it).first;
    if (num > (x * 2))
      cout << " noob";
    else
      if (num * 4 > (x * 5))
        cout << " random";
      else
        if (num * 9 > x * 10)
          cout << "  average";
        else
          if (num * 99 > x * 100)
            cout << " hardcore";
          else
            cout << " pro";
  }
  
  return 0;
}
