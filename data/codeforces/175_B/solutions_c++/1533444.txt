#include <iostream>
#include <string>
#include <map>
using namespace std;
map<string, int> mp;
int n;

int main()
{
	cin >> n;
	mp.clear();
	for (int i = 1; i <= n; i++)
	{
		int k;
		string s;
		cin >> s >> k;
		if (mp[s] < k)
			mp[s] = k;
	}
	map<string, int>::iterator mapi, mapj;
	int z = mp.size();
	cout << z << endl;
	for (mapi = mp.begin(); mapi != mp.end(); mapi++)
	{
		int x = 0;
		for (mapj = mp.begin(); mapj != mp.end(); mapj++)
			if (mapi->second < mapj->second)
				x++;
		if (x * 100 > 50 * z)
			cout << mapi->first << " noob" << endl;
		else if (x * 100 > 20 * z)
			cout << mapi->first << " random" << endl;
		else if (x * 100 > 10 * z)
			cout << mapi->first << " average" << endl;
		else if (x * 100 > z)
			cout << mapi->first << " hardcore" << endl;
		else cout << mapi->first << " pro" << endl;
	}
	return 0;
}
