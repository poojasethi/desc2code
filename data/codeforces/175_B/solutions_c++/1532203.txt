#include<cstdio>
#include<cstring>
#include<algorithm>
#include<string>
#include<map>
#include<iostream>

using namespace std;
int n,score;
string name;
map<string,int>M;
map<string,int>::iterator it1,it2;
int main(){
    cin>>n;
    for(int i=1;i<=n;++i){
        cin>>name>>score;
        M[name]=max(M[name],score);
    }
    int sum=M.size();
    cout<<sum<<endl;
    for(it1=M.begin();it1!=M.end();++it1){
        int cnt=0;
        for(it2=M.begin();it2!=M.end();++it2){
            if(it1->second>=it2->second)++cnt;
        }
        cout<<it1->first<<" ";
        if(cnt*100>=sum*99){
            cout<<"pro"<<endl;
        }else if(cnt*100>=sum*90){
            cout<<"hardcore"<<endl;
        }else if(cnt*100>=sum*80){
            cout<<"average"<<endl;
        }else if(cnt*100>=sum*50){
            cout<<"random"<<endl;
        }else cout<<"noob"<<endl;
    }
    return 0;
}
