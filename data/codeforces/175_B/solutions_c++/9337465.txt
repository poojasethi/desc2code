#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <map>
#define fi first
#define se second
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,sum;
map<string,int> Map;
typedef map<string,int>::iterator w1;
inline void work(int x) {
	if (x*2>sum) printf(" noob\n");
	else if (x*5>sum) printf(" random\n");
	else if (x*10>sum) printf(" average\n");
	else if (x*100>sum) printf(" hardcore\n");
	else printf(" pro\n");
}
int main() {
	scanf("%d",&n);
	For(i,1,n) {
		string s;
		cin>>s;
		scanf("%d",&k);
		Map[s]=max(Map[s],k);
	}
	sum=Map.size();
	printf("%d\n",sum);
	w1 A=Map.begin();
	for (;A!=Map.end();A++) {
		w1 B=Map.begin();
		int s=0;
		for (;B!=Map.end();B++) if (B->se>A->se) s++;
		cout<<A->fi;
		work(s);
	}
	return 0;
}
