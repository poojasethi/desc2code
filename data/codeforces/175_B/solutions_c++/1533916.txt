#include<cstdio>
#include<cstring>
#include<algorithm>
#define E 10e-8
using namespace std;
struct score{ char nme[20]; int sc; } a[1010];
bool comp( score x, score y ){ return strcmp( x.nme, y.nme ) < 0; }
int main(){
	int cnt = 0, n; 
	scanf( "%d", &n );
	for( int i = 0 ; i < n ; i++ ){
		char na[20]; int s, f = 0; scanf( "%s%d", na, &s );
		for( int j = 0 ; j < cnt ; j++ ) if( !strcmp( na, a[j].nme ) ){ f = 1; a[j].sc = max( a[j].sc, s ); }
		if( !f ){ strcpy( a[cnt].nme, na ); a[cnt++].sc = s; }
	}
	sort( a, a+cnt, comp );
	printf("%d\n",cnt);
	for( int i = 0 ; i < cnt ; i++ ){
		printf( "%s ", a[i].nme );
		int c = 0;
		for( int j = 0 ; j < cnt ; j++ ) if( a[i].sc >= a[j].sc ) c++;
		double p = 100.0*c/cnt;
		if( p < 50 ) puts("noob");
		if( p >= 50 && p < 80 ) puts("random");
		if( p >= 80 && p < 90 ) puts("average");
		if( p >= 90 && p < 99 ) puts("hardcore");
		if( p >= 99  ) puts("pro");
	}
	return 0;
}
