#include<stdio.h>
#include<string.h>
#include<string>
#include<iostream>
#include<map>

using namespace std;

int main()
{
//	freopen("1.txt","r",stdin);
	int i,j;
	int n;
	map<string,int> M;
	string name[1000],tname;
	int pnum=0,tmp;
	double con;

	cin>>n;
	while (n-->0)
	{
		cin>>tname>>tmp;
		if (M[tname]==0)
		{
			name[pnum++]=tname;
			if (tmp==0)
				M[tname]=-1;
			else
				M[tname]=tmp;
		}
		else if (tmp>M[tname] && tmp>0)
		{
			M[tname]=tmp;
		}
	}
	printf("%d\n",pnum);
	for (i=0;i<pnum;i++)
	{
		con=0;
		cout<<name[i]<<" ";
		for (j=0;j<pnum;j++)
		{
			if (M[name[j]]>M[name[i]])
				con++;
		}
		if (con/pnum<=0.01)
			printf("pro");
		else if (con/pnum<=0.1)
			printf("hardcore");
		else if (con/pnum<=0.2)
			printf("average");
		else if (con/pnum<=0.5)
			printf("random");
		else
			printf("noob");
		printf("\n");
	}
	return 0;
}
