#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#define N 1000
using namespace std;
struct Player
{
	int score;
	char name[15];
	bool operator < (const Player &x) const
	{ return score>x.score;}
}p[N+1];
int m,cs,i,j,tot=0,x;
char cur[15];
bool Flag;

int main()
{
	scanf("%d",&m);
	for(i=1;i<=m;++i)
	{
		scanf("%s%d",cur,&cs);
		for(Flag=0,j=1;j<=tot;++j)
			if(!strcmp(cur,p[j].name))
			{
				Flag=true;
				break;
			}
		if(Flag)
			p[j].score=max(p[j].score,cs);
		else
			p[++tot].score=cs,strcpy(p[tot].name,cur);
	}
	sort(p+1,p+tot+1);
	printf("%d\n",tot);
	for(i=1,x=1;i<=tot;++i)
	{
		printf("%s ",p[i].name,cur);
		if(p[x].score>p[i].score)x=i;
		j=(tot-x+1)*100/tot;
		if(j<50)puts("noob");
		else if(j<80)puts("random");
		else if(j<90)puts("average");
		else if(j<99)puts("hardcore");
		else puts("pro");
	}
	return 0;
}
