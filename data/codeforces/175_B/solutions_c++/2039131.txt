#include <iostream>
#include <vector>
#include <algorithm>
#include <map>
using namespace std;

int main() {
    int n; cin >> n;
    map<string,int> res;
    for (int i = 0; i < n; i++) {
        string s; int r;
        cin >> s >> r;
        res[s] = max(res[s], r);
    }

    n = res.size();
    cout << n << "\n";
    for (map<string,int>::iterator it = res.begin(); it != res.end(); it++) {
        int mu = 0;
        for (map<string,int>::iterator it2 = res.begin(); it2 != res.end(); it2++) {
            if (it->second >= it2->second) mu++;
        }
        double rapp = (double)mu/n * 100;
        cout << it->first << " ";
        if (rapp >= 99) cout << "pro";
        else if (rapp >= 90) cout << "hardcore";
        else if (rapp >= 80) cout << "average";
        else if (rapp >= 50) cout << "random";
        else cout << "noob";
        cout << "\n";
    }
}
