#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <map>
#include <string>

using namespace std;

int n;
double x;
int a[1000];
map<string,int> hash;

int main() {
	//freopen("b.in","r",stdin);
	scanf("%d",&n);
	for (int i=0;i<n;++i) {
		string s; int pp;
		cin >> s >> pp;
		hash[s]=max(hash[s],pp);
	}
	n=hash.size();
	x=n;
	cout << n << endl;
	for (map<string,int>::iterator i=hash.begin();i!=hash.end();++i) {
		double cnt=0,cnt1=0;
		for (map<string,int>::iterator j=hash.begin();j!=hash.end();++j)
			if (j->second<=i->second) cnt++;
			else cnt1++;
		cout << i->first << " ";
		if (cnt1>x*0.5) cout << "noob\n";
		else {
			if (cnt1>=x*0.2) cout << "random\n";
			else if (cnt1>=x*0.1) cout << "average\n";
			else if (cnt1>=x*0.01) cout << "hardcore\n";
			else cout << "pro\n";
		}
	}

	return 0;
}
