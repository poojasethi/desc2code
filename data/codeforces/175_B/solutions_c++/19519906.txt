#include<cstdio>
#include<map>
#include<string>
#include<iostream>
using namespace std;
int main()
{
	int point,n,mark[1005]={0},num,t,i,sum;
	char s[15];
	map<string,int>m;
	map<string,int>::iterator iter;
	scanf("%d",&n);
	while(n--)
	{
		scanf("%s%d",s,&point);
		if(m.find(s)==m.end())
		{
			m[s]=point;
			mark[point]++;
		}
		else if(point>m[s])
		{
			mark[m[s]]--;
			mark[point]++;
			m[s]=point;
		}
	}
	num=m.size();
	printf("%d\n",num); 
	for(iter=m.begin();iter!=m.end();iter++)
	{
		cout<<iter->first<<" ";
		t=mark[iter->second];
		for(i=iter->second+1,sum=0;i<=1000;i++)
			sum+=mark[i];
		if(sum*1.0<=0.01*num)
		{
			puts("pro");
		}
		else if(sum*1.0<=0.10*num)
		{
			puts("hardcore");
		}
		else if(sum*1.0<=0.20*num)
		{
			puts("average");
		}
		else if(sum*1.0<=0.50*num)
		{
			puts("random");
		}
		else
		{
			puts("noob");
		}
	}
	return 0;
}
