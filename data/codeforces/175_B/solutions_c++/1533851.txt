#include<cstdio>
#include<iostream>
#include<map>
#include<string>
using namespace std;
#define MAXN 1000010

map<string,int> m;
int N;
int cnt[MAXN];

int main()
{
    cin>>N;
    string tmp;
    int score;
    for(int i=0;i<N;i++)
    {
        cin>>tmp>>score;
        if(m.count(tmp)) m[tmp]=max(m[tmp],score);
        else m[tmp]=score;
    }
    map<string,int>::iterator it;
    for(it=m.begin();it!=m.end();it++) cnt[it->second]++;
    for(int i=1;i<MAXN;i++) cnt[i]+=cnt[i-1];

    int num=m.size();
    cout<<num<<endl;
    for(it=m.begin();it!=m.end();it++)
    {
        cout<<it->first<<" ";
        int rank=cnt[it->second];
        rank=num-rank;
        if(rank>0.5*num) cout<<"noob"<<endl;
        else if(rank>0.2*num) cout<<"random"<<endl;
        else if(rank>0.1*num) cout<<"average"<<endl;
        else if(rank>0.01*num) cout<<"hardcore"<<endl;
        else cout<<"pro"<<endl;
    }
    return 0;
}
