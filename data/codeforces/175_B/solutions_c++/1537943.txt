#include<map>
#include<string>
#include<iostream>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	map<string,int> f;
	int n; cin>>n;
	rep(i,n){
		string name;
		int score; cin>>name>>score;
		f[name]=max(f[name],score);
	}

	int m=f.size();
	cout<<m<<endl;
	map<string,int>::iterator it,jt;
	for(it=f.begin();it!=f.end();++it){
		int cnt=0;
		for(jt=f.begin();jt!=f.end();++jt) if(jt->second<=it->second) cnt++;

		string grade;
		if     (100*cnt<50*m) grade="noob";
		else if(100*cnt<80*m) grade="random";
		else if(100*cnt<90*m) grade="average";
		else if(100*cnt<99*m) grade="hardcore";
		else                  grade="pro";

		cout<<it->first<<' '<<grade<<endl;
	}

	return 0;
}
