#include <string>
#include <cstdio>
#include <map>
#include <algorithm>
using namespace std;

int main() {
    int n;
    scanf("%d",&n);
    map<string,int> m;
    while(n--) {
        char buf[60];
        scanf("%s",buf);
        int pts;
        scanf("%d",&pts);
        string s(buf);
        m[s] = max(m[s], pts);
    }
    n = m.size();
    printf("%d\n",n);
    for(map<string,int>::iterator i=m.begin();i!=m.end();++i) {
        int worse=0,better=0;
        for(map<string,int>::iterator j=m.begin();j!=m.end();++j) {
            if (j->second < i->second) worse++;
            if (j->second > i->second) better++;
        }
        string s = i->first;
        string rank;
        if (better * 2 > n) rank = "noob";
        else if (better * 5 > n) rank = "random";
        else if (better * 10 > n) rank = "average";
        else if (better * 100 > n) rank = "hardcore";
        else rank = "pro";
        
        printf("%s %s\n",s.c_str(), rank.c_str());
    }
    return 0;
}
