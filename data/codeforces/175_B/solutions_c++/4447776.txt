#include <iostream>
#include <cstdlib>
#include <string>

using namespace std;

int main(){
	string name[1000],profile[1000];
	int n,i,score[1000],j,s;
	cin >> n;
	for(i=0;i<n;i++){
		cin >> name[i] >> score[i];
		for(j=0;j<i;j++)
			if(name[j] == name[i]){
				if(score[i] > score[j])
					score[j] = score[i];
				i--;
				n--;
				break;
			}
	}
	cout << n << endl;
	for(i=0;i<n;i++){
		s=0;
		for(j=0;j<n;j++)
			if(score[i] >= score[j])
				s++;
		cout << name[i] << " ";
		if(s*100/n >= 99)
			cout << "pro";
		else if(s*100/n >= 90)
			cout << "hardcore";
		else if(s*100/n >= 80)
			cout << "average";
		else if(s*100/n >= 50)
			cout << "random"; 
		else
			cout << "noob";
		cout << endl;
	}
	return 0;
}
