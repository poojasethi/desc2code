#include<cstdio>
#include<iostream>
#include<map>
#include<vector>

using namespace std;

int n;
vector<string> player;
map<string,int> score;

int main()
{
	int i, j, k, a, t;
	string s;
	scanf("%d\n", &n);
	for (i = 0; i < n; i++)
	{
		cin >> s >> a;
		if (!score.count(s)) 
		{ 
			score[s] = a; 
			player.push_back(s);
		}
		else score[s] = max(score[s],a);
	}
	t = player.size();
	printf("%d\n", t);
	for ( i = 0; i < t; i++)
	{
		k = 0;
		for ( j = 0; j < t; j++)
		{
			if (score[player[i]] >= score[player[j]]) ++k;
		}
		cout << player[i] << " ";
		if ( (k*100)/t >= 99 ) cout << "pro" << endl;
		else if ( (k*100)/t >= 90) cout << "hardcore" << endl;
		else if ( (k*100)/t >= 80) cout << "average" << endl;
		else if ( (k*100)/t >= 50) cout << "random" << endl;
		else cout << "noob" << endl;
	}
}
