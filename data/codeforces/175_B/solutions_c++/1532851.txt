#include<algorithm>
#include<iostream>
#include<map>
#include<string>
#include<vector>

using namespace std;

string Level(int r, int S) {
  if(100*r >= 99*S)
    return "pro";
  if(100*r >= 90*S)
    return "hardcore";
  if(100*r >= 80*S)
    return "average";
  if(100*r >= 50*S)
    return "random";
  return "noob";
}

int main() {
  map<string, int> m;
  int N;
  cin >> N;
  for(int n = 0; n < N; ++n)
    {
      string name;
      int score;
      cin >> name >> score;
      m[name] = max(m[name], score);
    }

  vector<int> scores;
  for(map<string, int>::const_iterator i = m.begin(); i != m.end(); ++i)
    scores.push_back(i->second);
  sort(scores.begin(), scores.end());
  
  const int S = m.size();
  cout << S << endl;
  for(map<string, int>::const_iterator i = m.begin(); i != m.end(); ++i)
    {
      const int r = upper_bound(scores.begin(), scores.end(), i->second) - scores.begin();
      cout << i->first << " " << Level(r, S) << endl;
    }
}
