#include<iostream>
#include<algorithm>
#include<vector>
#include<map>
#include<string>

using namespace std;
int main()
{
    map<string,int> f;
    int n; cin>>n;
    string name;
    int score;
    
    for(int i=0;i<n;i++)
    {
        cin>>name>>score;
        f[name]=max(f[name],score);
    }
    int m=f.size();
    cout<<m<<endl;

    map<string,int>::iterator i,j;
    int sum=0;

    for ( i=f.begin() ; i != f.end(); i++ )
    {
        int cnt=0;
        for ( j=f.begin() ; j != f.end(); j++ )
         if(j->second<=i->second) cnt++;
        string grade;
        if     (100*cnt<50*m) grade="noob";
        else if(100*cnt<80*m) grade="random";
        else if(100*cnt<90*m) grade="average";
        else if(100*cnt<99*m) grade="hardcore";
        else                  grade="pro";

        cout<<i->first<<' '<<grade<<endl;
    }
    

    return 0;
}
