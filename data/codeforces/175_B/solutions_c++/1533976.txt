#include <cstdio>
#include <iostream>
#include <string>
#include <map>
#include <algorithm>
#include <cmath>

using namespace std;

map<string,int> score;
map<string,string> cate;
string na[1001];

int main(void)
{
  int n=0,m=0;
  scanf("%d",&n);
  for(int i=1;i<=n;++i)
  {
    string s;
    int k;
    cin>>s;
    scanf("%d",&k);
    if(score.find(s)==score.end())
    {
      na[++m]=s;
      score[s]=k;
    }
    else
      score[s]=max(score[s],k);
  }
  for(int i=1;i<=m;++i)
  {
    int k=0;
    for(int j=1;j<=m;++j)
      if(score[na[i]]>=score[na[j]])
        ++k;
    int p=(int)floor(k*100.0/m);
    if(p>=99)
      cate[na[i]]="pro";
    else if(p>=90)
      cate[na[i]]="hardcore";
    else if(p>=80)
      cate[na[i]]="average";
    else if(p>=50)
      cate[na[i]]="random";
    else
      cate[na[i]]="noob";
  }
  printf("%d\n",m);
  for(int i=1;i<=m;++i)
    cout<<na[i]<<" "<<cate[na[i]]<<endl;
}
