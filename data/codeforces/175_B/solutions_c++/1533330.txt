#include <map>
#include <string>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;

const int MAX_N = 1000 + 10;
int N, tot;
string n[MAX_N];
map<string, int> M;
int sc[MAX_N];

const double eps = 1e-8;

string ali(int n)
{
	int cnt = 0;
	for(int i = 0; i < N; ++ i)
		cnt += sc[i] <= sc[n];
	double x = double(cnt) / double(N);
	if (x < 0.5 - eps) return "noob";
	if (x < 0.8 - eps) return "random";
	if (x < 0.9 - eps) return "average";
	if (x < 0.99 - eps) return "hardcore";
	return "pro";
}

void solve()
{
	int x, p;
	string cur;
	cin >> N;
	for(int i = 0; i < N; ++ i) {
		cin >> cur >> p;
		if (M.find(cur) != M.end()) {
			x = M[cur];
			sc[x] = max(sc[x], p);
		} else {
			n[tot] = cur;
			M[cur] = tot;
			sc[tot ++] = p;
		}
	}
	N = tot;
	cout << N << endl;
	for(int i = 0; i < N; ++ i) 
		cout << n[i] << ' ' << ali(i) << endl;
}

int main()
{
	solve();
	return 0;
}
