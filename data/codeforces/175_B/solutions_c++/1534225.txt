#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <map>
using namespace std;
#define P pair <pair<int, int>, int>
#define x first
#define y second
typedef map <string, int> PLY;
PLY mp;
int main(){
	int n, p;
	cin >> n;
	string s;
	
	for (int i = 0;i < n;++i){
		cin >> s >> p;
		if (mp.find(s) == mp.end()) mp[s] = p;
		else mp[s] = max(mp[s], p);
	}
	int num = mp.size();
	cout << num << endl;
	for (PLY :: iterator it = mp.begin();it != mp.end();++it){
		int bet = 0;
		for (PLY :: iterator sf = mp.begin();sf != mp.end();++sf)
			if (sf -> y > it -> y)
				++bet;
		string kind;
		if (bet * 2 > num) kind = "noob";
		else if (bet * 5 > num) kind = "random";
		else if (bet * 10 > num) kind = "average";
		else if (bet * 100 > num) kind = "hardcore";
		else kind = "pro";
		cout << it -> first << " " << kind << endl;
	}
	return 0;
}
