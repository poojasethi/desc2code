#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <map>
#include <string>
using namespace std;

int n;
string s,buff;
int x;
map<string,int> H;
int m,A,B;
int c[10005];

int main()
{
 //	freopen("input.txt","r",stdin);
	
	scanf("%d",&n);
	for (int i=1;i<=n;++i){
		cin >> s >> x;
		if (H.find(s)==H.end()) H[s]=x;
		else H[s]=max(H[s],x);
	}
	
	for (map<string,int>::iterator it=H.begin();it!=H.end();++it)
		c[it->second]++;
	
	
	for (int i=1;i<=10000;++i) c[i]+=c[i-1];
	
	m=H.size();
	cout << m << endl;
	for (map<string,int>::iterator it=H.begin();it!=H.end();++it){
		A=c[it->second];
		B=m-A;
		if (2*B>m) buff="noob";
		else if (2*A>=m && 5*B>m) buff="random";
		else if (5*A>=4*m && 10*B>m) buff="average";
		else if (10*A>=9*m && 100*B>m) buff="hardcore";
		else buff="pro";
		
		cout << it->first << " " << buff << endl;
	}
	
	return 0;
}
