#include<iostream>
#include<string>
#include<cstdio>
#include<algorithm>
using namespace std;
int cnt;
string s[1005];
int c[1005];
int t[1005];
int n;
int main()
{
    cin>>n;

    string a;
    int score;

    cnt=0;

    for(int i=0;i<n;i++)
    {
        cin>>a>>score;

        int j;

        for(j=0;j<cnt;j++)
            if(s[j]==a)
            {
                if(c[j]<score)
                c[j]=score;
                break;
            }
            if(j>=cnt) s[cnt++]=a,c[cnt-1]=score;
    }

    int y;

    cout<<cnt<<endl;

    for(int i=0;i<cnt;i++)
    {
        y=0;
        for(int j=0;j<cnt;j++)
        {
            if(c[j]>c[i]) y++;
        }
        cout<<s[i]<<" ";

        if(y*2>cnt) puts("noob");
        else if(y*5>cnt) puts("random");
        else
        if(y*10>cnt) puts("average");
        else if(y*100>cnt)
        puts("hardcore");
        else
        puts("pro");

    }
}
