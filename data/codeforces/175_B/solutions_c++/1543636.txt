#include <iostream>
#include <vector>
#include <string>

using namespace std;

int main() {
int n;
cin >> n;
string name[n];
int pts[n];
vector<string> str;
vector<int> result;
for(int i = 0; i < n; i++) {
cin >> name[i] >> pts[i];
}
for(int i = 0; i < n; i++) {
int max = pts[i];
if(name[i] == "")
continue;
else
for(int j = i+1; j < n; j++) {
if(name[i] == name[j]) {
name[j] = "";
if(pts[j] > max)
max = pts[j];
}
}
str.push_back(name[i]);
result.push_back(max);
}
int size = str.size();
cout << size << endl;
for(int i = 0; i < size; i++) {
int br = 0;
for(int j = 0; j < size; j++) {
if(result[i] >= result[j])
br++;
}
if(br*100/size < 50)
cout << str[i] << " noob" << endl;
else if(br*100/size < 80)
cout << str[i] << " random" << endl;
else if(br*100/size < 90)
cout << str[i] << " average" << endl;
else if(br*100/size < 99)
cout << str[i] << " hardcore" << endl;
else
cout << str[i] << " pro" << endl;
}
return 0;
}
