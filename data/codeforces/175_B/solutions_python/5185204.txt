n=input()
from collections import defaultdict as D
C=D(int)
for i in range(n):
    N,c=raw_input().split()
    c=int(c)
    C[N] = max(c,C[N])
from bisect import bisect_right as B
s=sorted(C.values())
N=len(s)
print N
for (n,c) in C.items():
    b=B(s,c)
    print n,
    if 2*b < N:
        print "noob"
    elif 5*b < 4*N:
        print "random"
    elif 10*b < 9*N:
        print "average"
    elif 100*b < 99*N:
        print "hardcore"
    else:
        print "pro"
