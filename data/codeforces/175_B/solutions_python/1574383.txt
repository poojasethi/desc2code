#!/usr/bin/python -O
import sys
import collections
import operator
from collections import defaultdict


n = int(sys.stdin.readline())
val = defaultdict(int)
for i in range(n):
	name, value = sys.stdin.readline().split()
	val[name] = max(val[name], int(value))

print len(val)
names = defaultdict(set)
for name, value in val.iteritems():
	names[value].add(name)
i = 0
for (value, names) in sorted(names.iteritems()):
	i += len(names)
	percent = float(i) / len(val)
	for name in names:
		if percent < .5:
			print name, "noob"
		elif percent < .8:
			print name, "random"
		elif percent < .9:
			print name, "average"
		elif percent < .99:
			print name, "hardcore"
		else:
			print name, "pro"
