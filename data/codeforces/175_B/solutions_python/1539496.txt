n=int(raw_input())
dd={}
for i in xrange(n):
    nm,sc=raw_input().split()
    dd[nm]=max(dd.get(nm,0),int(sc))
n=len(dd)
print n    
for i in dd:
    xx=0
    for j in dd:    
        if (dd[j]>dd[i]):  
            xx+=1
    if (xx*2>n):
        rs="noob"
    elif (xx*5>n):
        rs="random"
    elif (xx*10>n):
        rs="average"
    elif (xx*100>n):
        rs="hardcore"
    else:
        rs="pro"
    print i,rs
