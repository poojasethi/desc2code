import itertools as it
from sys import stdin

for line in stdin:
    n, m, c0, d0 = map(int, line.split())
    
    xss = [map(int, next(stdin).split()) for _ in xrange(m)]
    xss.insert(0, [0]*4)
    a, b, c, d = zip(*xss)
    ## print a, b, c, d
    
    dp = [[0]*(m+1) for n in xrange(n+1)]
    for i in xrange(n+1):
        for j in xrange(1, m+1):
            dp[i][j] = max(dp[i-c[j]*k][j-1] + d[j]*k \
                            for k in xrange(a[j]//b[j]+1) if i-c[j]*k >= 0)
    print max(dp[k][m] + (n-k)//c0 * d0 for k in xrange(n+1))
