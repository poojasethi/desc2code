l = []
n, m, c, d = map(int, raw_input().split())
l.append([1050, c, d])
for _ in xrange(m):
    s = map(int, raw_input().split())
    l.append([s[0] / s[1], s[2], s[3]])

f = [0] * 1001
s = [[x[0] for x in l]]

for i in range(1, n + 1):
    t = s[0]
    for j in range(m + 1):
        if i >= l[j][1] and s[i - l[j][1]][j] > 0 and f[i] < f[i - l[j][1]] + l[j][2]:
            f[i] = f[i - l[j][1]] + l[j][2]
            t = s[i - l[j][1]][:]
            t[j] -= 1
    s.append(t)
print max(f)
