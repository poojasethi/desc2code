r=lambda:map(int,raw_input().split())
[n,m,c0,d0]=r()
cnt=[1000]
c=[c0]
d=[d0]
for i in range(m):
	[a,b,c1,d1]=r()
	cnt+=[a/b]
	c+=[c1]
	d+=[d1]
l=[-1]*(n+1)
l[0]=0
for i in range(m+1):
	for k in xrange(n,-1,-1):
		for j in xrange(cnt[i]+1):
			ind=k+j*c[i]
			if l[k]!=-1 and ind<=n:
				l[ind] = max(l[ind],l[k]+j*d[i])
print max(l)
