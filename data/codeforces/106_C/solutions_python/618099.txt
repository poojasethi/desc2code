from sys import stdin
from math import floor

buf = []
for line in stdin:
    buf.append(line.strip())
buf.reverse()

def getints():
    return tuple([int(e) for e in buf.pop().split()])

n, m, c, d = getints()
dat = [0 for i in range(n+1)]
for k in range(n/c+1):
    dat[k*c] = k*d
for i in range(m):
    a, b, c, d = getints()
    for j in reversed(range(n)):
        for k in range(min((n-j)/c, a/b)+1):
            q = j+k*c
            dat[q] = max(dat[q], dat[j]+k*d)

print max(dat)
