#include<iostream>
using namespace std;
int a[11],b[11],c[11],d[11],dp[1001][11];
int main(){
	int n,m,c0,d0;
	cin>>n>>m>>c0>>d0;
	for(int i=1;i<=m;i++){
		cin>>a[i]>>b[i]>>c[i]>>d[i];
	}
	for(int i=1;i<=n;i++){
		dp[i][0]=i/c0*d0;
	}
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			int z=min(a[j]/b[j],i/c[j]);
			for(int k=0;k<=z;k++){
				dp[i][j]=max(dp[i-k*c[j]][j-1]+k*d[j],dp[i][j]);
			}
		}
	}
	cout<<dp[n][m]<<endl;
	return 0;
}
