# include <stdio.h>
int f[2000];
int main (void)
{
    int m,n,a,d,c,b;
    int i,j,p;
    scanf("%d%d%d%d",&m,&n,&c,&d);
    for (i=c;i<=m;i++)
        if (f[i-c]+d>f[i])
            f[i]=f[i-c]+d;
    for (i=1;i<=n;i++)
    {
        scanf("%d%d%d%d",&a,&b,&c,&d);
        for (p=1;p<=a/b;p++)
            for (j=m;j>=c;j--)
                if (f[j-c]+d>f[j])
                    f[j]=f[j-c]+d;
    }
    printf("%d\n",f[m]);
    return 0 ;
}
