#include <iostream>
using namespace std;
int dp[1001][11];

int main()
{
  int n, m, c0, d0;
  int a[11], b[11], c[11], d[11];
  cin>>n>>m>>c0>>d0;
  for (int i = 1; i <= m; i++)
    cin>>a[i]>>b[i]>>c[i]>>d[i];
  for (int i = 1; i <= n; i++) dp[i][0] = (i/c0)*d0;
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) {
      dp[i][j] = max(dp[i][j], dp[i][j-1]);
      for (int k = 1; k <= a[j]/b[j]; k++) if (k*c[j] <= i) {
	  dp[i][j] = max(dp[i][j], dp[i - k*c[j]][j-1] + k*d[j]);
	}
    }
  }
  cout<<dp[n][m]<<endl;
  return 0;
}
