#include <iostream>
#include <stdio.h>
using namespace std;
#define N 1001
int main()
{ int n,m,c0,d0;
  int a[N],b[N],c[N],d[N],f[N][101];
// freopen("i.txt","r",stdin);
cin>>n>>m>>c0>>d0;
for (int i=1;i<=m;i++) cin>>a[i]>>b[i]>>c[i]>>d[i];

for (int i=0;i<=n;i++) f[i][0]=i/c0*d0;

for (int j=1; j<=m;j++)
 for (int i=0;i<=n;i++)
 {int s=0;

  for (int k=0;k<=i;k++)
   {int x=min(k/c[j],a[j]/b[j]);
    int s1=x*d[j]+f[i-k][j-1];
    if (s1>s) s=s1;
   }

  f[i][j]=s;
 }

cout<<f[n][m];
return 0;
}
