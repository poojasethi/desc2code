#include <bits/stdc++.h>
using namespace std;
int dp[1010];
 int n,m,c0,d0,a,b,c,d,ans; 
int main(){
	cin >> n >> m >> c0 >> d0; 
    for(int i=0;i<=n;i++) dp[i]=i/c0*d0;
    for(int i=0;i<m;i++){  
		cin >> a >> b >> c >> d; 
        for(int k=1;k<=a/b;k++)
        {   for(int j=n;j>=c;j--)
               if(dp[j]<dp[j-c]+d) dp[j]=dp[j-c]+d;
        }
    }
    for(int i=0;i<=n;i++)
    if(ans<dp[i]) ans=dp[i];
	cout << ans << endl;
	return 0 ;
}
