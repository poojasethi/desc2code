#include <iostream>
using namespace std;
int main()
{
	int n, m, a, b, c, d, mas[1101]={};
	cin >> n >> m >> c >> d;
	for(int i = c; i <= n; i++) 
		mas[i]=mas[i-c]+d;
	for(int q = 0; q < m; q++)
	{
		cin >> a >> b >> c >> d;
		for(int i = 0; i < a/b; i++)
			for(int j = n; j >= c; j--) 
				mas[j] = max(mas[j], mas[j-c]+d);
	}
	cout << mas[n];
	return 0;
}
