#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
int m,n,left,need,test,cost,c0,d0;
cin >> n >> m >> c0 >> d0;
vector <int> d(n+1);
for(int i = 0;i <= n;i++) {
	d[i] = (i / c0) * d0;
}
for(int k = 0;k < m;k++) {
	cin >> left >> need >> test >> cost;
	vector <int> new_d = d;
	for(int i = 0;i <= n;i++) {
		for(int j = 0;j * need <= left;j++) {
			if(j * test <= i) {
				new_d[i] = max(new_d[i], d[i - j * test] + j * cost);
			}
		}
	}
	d = new_d;
}
cout << d[n] << endl;
}
