#include <iostream>
#define inf 1000000000
using namespace std;
int a[11],b[11],c[11],d[11];
int n,m,dp[1001];

int main()
{

    cin>>n>>m;
    a[0]=inf; b[0]=1;
    cin>>c[0]>>d[0];
    for(int i=1;i<=m;i++)
        cin>>a[i]>>b[i]>>c[i]>>d[i];

    for(int j=0;j<=m;j++){
        int lim=a[j]/b[j];
        for(int i=n;i>=1;i--){
            for(int k=1;k<=lim&&(i-k*c[j]>=0);k++)
                dp[i]=max(dp[i],dp[i-k*c[j]]+k*d[j]);
        }
    }

    cout<<dp[n]<<endl;
    return 0;
}
