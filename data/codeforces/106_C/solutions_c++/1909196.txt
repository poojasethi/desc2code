#include <iostream>
#include <cstring>
#include <string>
#include <cstdio>

using namespace std;

int n, m, c0, d0, a, b, c, d, cost[1001];

int main()
{
    cin >> n >> m >> c0 >> d0;
    for (int i = c0; i <= n; i++) cost[i] = cost[i - c0] + d0;
    for (int i = 1; i <= m; i++)
    {
        cin >> a >> b >> c >> d;
        for (int j = 1; j <= a / b; j++)
        {
            for (int k = n; k >= c; k--) cost[k] = max(cost[k], cost[k - c] + d);
        }
    }
    cout << cost[n];
    return 0;
}
