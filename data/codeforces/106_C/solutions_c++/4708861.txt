#include<iostream>
#include<math.h>
using namespace std;


int dp[2000][2000];
int main()
{
    int n,m,i,j,k;
    int a[500],b[500],c[500],d[500];

    cin>>n>>m>>c[0]>>d[0];
    for(i=1;i<=m;i++)
        cin>>a[i]>>b[i]>>c[i]>>d[i];
    for(i=0;i<=n;i++)
        dp[i][0]=int(i/c[0])*d[0];
    for(i=1;i<=m;i++)
         for(j=0;j<=n;j++)
          for(k=0;k*c[i]<=j && k*b[i]<=a[i];k++)
        dp[j][i]=max(dp[j][i],k*d[i]+dp[j-k*c[i]][i-1]);
        cout<<dp[n][m]<<endl;

    return 0;
}
