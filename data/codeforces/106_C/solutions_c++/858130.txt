#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(int argc, char **argv)
{
  int n, m, c0, d0, a, b, c, d;
  cin >> n >> m >> c0 >> d0;
  vector<int> v(n+1);
    for(int i = c0; i <= n; i+=c0)
      v[i] = v[i-c0] + d0;
    for(int i = 0; i < m; ++i)
    {
      cin >> a >> b >> c >> d;
        for(int j = 0; j < a/b; ++j)
          for(int k = n; k >= c; --k)
              v[k] = max(v[k], v[k-c] + d);
    }
  sort(v.rbegin(),v.rend());  
  cout << v[0];
  //system("pause");
  return 0;
}
