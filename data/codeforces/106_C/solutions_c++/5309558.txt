#include <iostream>
#include <algorithm>
using namespace std;
int weight[10001];
int main(){
    int n,m,c0,d0,a,b,c,d,time;
    cin >>n>>m>>c0>>d0;
    for(int i=1;i<=n;i++)
        weight[i]=d0*(i/c0);
    for(int i=1;i<=m;i++){
            cin >>a>>b>>c>>d;
            time=a/b;
        for(int k=0;k<time;k++)
        for(int j=n;j>=c;j--)
             weight[j]=max(weight[j],weight[j-c]+d);
    }
    cout<< weight[n];
    return 0;
}
