#include <iostream>

using namespace std;

 
int n, m;
int a[12], b[12], c[12], d[12];
int tug[1010][12];

int main(){

	cin >> n >> m >> c[0] >> d[0];
	
	for(int i = 1; i <= m; i++)
		cin >> a[i] >> b[i] >> c[i] >> d[i];

	for(int i = 0; i <= n; i++)
		tug[i][0] = int(i / c[0]) * d[0];

	for(int p = 0; p <= n; p++)
		for(int i = 1; i <= m; i++)
			for(int k = 0; k * c[i] <= p and k * b[i] <= a[i]; k++)
				tug[p][i] = max(tug[p][i], k * d[i] + tug[p - k * c[i]][i - 1]);
	
	cout << tug[n][m];
}
