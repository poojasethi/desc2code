{

		F[i] = F[i - c0] + d0;
	while (m--)
	{
		scanf("%d %d %d %d", &a, &b, &c, &d);
		for(int i = 1; i <= a/b; i++)
			for(int j = n; j >= c; j--)
			{
				int tmp = F[j - c] + d;
				if (F[j] < tmp)
					F[j] = tmp;
			}
	}
	printf("%d", F[n]);
	return 0;
}
