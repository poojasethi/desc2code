#include <iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int dp[1001]={0};
int main()
{   int n,m,c0,d0,a,b,c,d;
    scanf("%d %d %d %d",&n,&m,&c0,&d0);
    for(int i=0;i<=n;i++) dp[i]=i/c0*d0;
    for(int i=0;i<m;i++)
    {   scanf("%d %d %d %d",&a,&b,&c,&d);
        for(int k=1;k<=a/b;k++)
        {   for(int j=n;j>=c;j--)
               if(dp[j]<dp[j-c]+d) dp[j]=dp[j-c]+d;
        }
    }
    int ans=0;
    for(int i=0;i<=n;i++)
    if(ans<dp[i]) ans=dp[i];
    printf("%d\n",ans);
}
