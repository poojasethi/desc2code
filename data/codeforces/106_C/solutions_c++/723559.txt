#include <iostream>

using namespace std;

int memo[1001];

int main(){

	int n, m, c0, d0, a, b, c, d;
	cin >> n >> m >> c0 >> d0;
	for (int i = 0 ; i <= n ; i++) memo[i] = 0;
	for (int i = c0 ; i <= n; i++) memo[i] = memo[i-c0] + d0;
	
	for (int i = 0 ; i < m; i++) {
		cin >> a >> b >> c >> d;
		for (int j = 0 ; j < a / b ; j++){
			for (int k = n; k >= c ; k--){
				memo[k]=max(memo[k],memo[k-c]+d);
			}
		}
	}
	
	cout << memo[n] << endl;
	
	return 0;
}
