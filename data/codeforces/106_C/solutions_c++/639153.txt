#include<cstdio>
#include<iostream>

using namespace std;

int n,m,c0,d0;
int dp[1024];

int main() {

    int i,j,k;
    int a,b,c,d;

    scanf("%d%d%d%d",&n,&m,&c0,&d0);

    for(i=c0;i<=n;i++) {
        dp[i]=dp[i-c0]+d0;
    }

    for(i=0;i<m;i++) {
        scanf("%d%d%d%d",&a,&b,&c,&d);
        for(j=0;j<a/b;j++)
            for(k=n;k>=c;k--) {
                dp[k]=max(dp[k],dp[k-c]+d);
            }
    }

    printf("%d\n",dp[n]);

    return 0;

}
