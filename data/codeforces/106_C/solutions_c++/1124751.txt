#include<iostream>
#include<cstdio>
using namespace std;

int dp[1001];
int main()
{
	int n,m,c0,d0,i,j,k,a,b,c,d;  cin>>n>>m>>c0>>d0;

	for(i=1;i<=n;i++)
	   dp[i]=d0*(i/c0);


	for(i=1;i<=m;i++){  
		cin>>a>>b>>c>>d;

		for(k=1;k<=a/b;k++)
		   for(j=n;j>=1;j--)
		   {
			   if(j>=c)
			   {
				   if(dp[j-c]+d>dp[j])
					   dp[j]=dp[j-c]+d;
			   }
		   }
	}

	printf("%d\n",dp[n]);

	return 0;
}
