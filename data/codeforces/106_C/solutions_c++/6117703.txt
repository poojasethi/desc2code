#include <iostream>
#include <vector>
#include <map>
using namespace std;
int arr[5000];
int max(int u, int v)
{
	if(u>v)  return u;
	else return v;
}

int main()
{
	int n,m,c0,d0;
	cin>>n>>m>>c0>>d0;
	for(int i=c0;i<=n;++i) 
	{
		arr[i]=arr[i-c0]+d0;
	}
	for(int x=0;x<m;++x)
	{
		int a,b,c,d,r;
		cin>>a>>b>>c>>d;
		r=a/b;
		for(int i=0;i<r;++i)
		{
		for(int j=n;j>=c;--j) 
		arr[j]=max(arr[j-c]+d,arr[j]);
		}
	}
	cout<<arr[n]<<"\n";
}
