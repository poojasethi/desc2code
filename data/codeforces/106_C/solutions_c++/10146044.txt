#include<iostream>
using namespace std;
int c[15],d[15];
int dp[15][1005];
int lim[15];

int main(){
int n,m;
cin >> n >> m >> c[0] >> d[0];
for(int i=1;i<=m;i++){
int a, b;
cin >> a >> b>> c[i] >> d[i];
lim[i]=a/b;
}
lim[0]=-1;
for(int i=0;i<=m;i++){
    for(int j=0;(j<=lim[i] || lim[i]==-1) && j<=n  ;j++){
        for(int k=j*c[i];k<=n;k++){
       if(i) dp[i][k] = max(dp[i][k],dp[i-1][k-j*c[i]]+j*d[i]);
        else
        dp[i][k] = max(dp[i][k],j*d[i]);
        }
    
    }


}
cout << dp[m][n];

return 0;
}
