#include <bits/stdc++.h>

using namespace std;

int b[101],c0,d0,c[101],d[101],m,a[101];
int dp[1001][11];

int f(int n,int i)
{
    if(i==m)
        return (n/c0)*d0;
    if(dp[n][i]!=-1)
        return dp[n][i];
    int mx=0,j=0;
    while(j*b[i]<=a[i]&&j*c[i]<=n)
        mx=max(mx,j*d[i]+f(n-j*c[i],i+1)),j++;
    return dp[n][i]=mx;
}

int main()
{
    int n;
    memset(dp,-1,sizeof(dp));
    cin>>n>>m>>c0>>d0;
    for(int i=0;i<m;i++)
        cin>>a[i]>>b[i]>>c[i]>>d[i];
    cout<<f(n,0)<<endl;
    return 0;
}
