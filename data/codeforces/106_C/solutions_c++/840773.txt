#include<iostream>
#include<string.h>
using namespace std;
int dp[10000][110],m,c[3000],a[3000],b[3000],d[3000];
int rec(int n,int ind)
{
    if(ind>m)return 0;
    int &ret=dp[n][ind];
    if(ret!=-1)return ret;
    for(int i=0;i*b[ind]<=a[ind]&&c[ind]*i<=n;i++)
    {
        ret=max(ret,d[ind]*i+rec(n-c[ind]*i,ind+1));
    }
    return ret;
}
int main()
{
    int n,i;
    memset(dp,-1,sizeof(dp));
    cin>>n>>m>>c[0]>>d[0];
    for(i=1;i<=m;i++)cin>>a[i]>>b[i]>>c[i]>>d[i];
    int ans=rec(n,0);
    cout<<ans<<endl;
}
