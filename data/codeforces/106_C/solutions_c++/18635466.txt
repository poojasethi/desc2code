#include<bits/stdc++.h>
using namespace std;
int n,m,c0,d0,i,j,k,a[15],b[15],c[15],d[15],dp[1005][15],ans;
int main()
{
    ios::sync_with_stdio(false);
    cin>>n>>m>>c0>>d0;
    for(i=1;i<=m;i++)cin>>a[i]>>b[i]>>c[i]>>d[i];
    for(i=0;i<=n;i++)
      for(j=1;j<=m;j++)
      	for(k=0;k<=a[j]/b[j];k++)
      		if(i>=k*c[j])
          dp[i][j]=max(dp[i][j],dp[i-k*c[j]][j-1]+k*d[j]);
    for(i=0;i<=n;i++)
        ans=max(ans,dp[i][m]+(n-i)/c0*d0);
    cout<<ans;
	return 0;
}
