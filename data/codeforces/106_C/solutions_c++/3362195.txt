#include <iostream>
using namespace std;

int coast[1001], n, m, a, b, c, d, temp = 0;
int main()
{
    cin >> n >> m >> c >> d;
	
	for (int i = 0; i <= n; i++)
		coast[i] = 0;

	for (int i = c; i <= n; i++)
		coast[i] = coast[i - c] + d;

	for (int k = 0; k < m; k++)
	{	
		cin >> a >> b >> c >> d;
		for (int i = 0; i < a/b; i++)
			for (int j = n; j >= c; j--)
					coast[j] = max(coast[j], coast[j - c] + d);
	}
	cout << coast[n];
	return 0;
}
