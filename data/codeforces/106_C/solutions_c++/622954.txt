#include<iostream>
using namespace std;

long f[1100], c0, d0, n, m, a, b, c, d, i, j, k;
int main(){
    cin>>n>>m>>c0>>d0;
    for (i = c0; i <= n; i++) f[i] = f[i - c0] + d0;
    for (i = 0; i < m; i++){
       cin>>a>>b>>c>>d;
       for (j = 0; j < a/b; j++)
         for (k = n; k >= c; k--) f[k] = max(f[k], f[k - c] + d);
    }
    cout<<f[n];
    //system("pause");
}
