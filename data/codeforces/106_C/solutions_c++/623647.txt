#include<iostream>
#include<algorithm>

using namespace std;

int n,m;
int dp[1001][101];
int a[102],b[102],c[102],d[102],c0,d0;

int main()
{
    cin>>n>>m>>c0>>d0;
    for (int i=0;i<m;i++)
        cin>>a[i]>>b[i]>>c[i]>>d[i];
    a[m] = 0x7fffffff;
    b[m] = 1;
    c[m] = c0;
    d[m] = d0;
    for (int i=0;i<=n;i++)
        for (int j=m;j>=0;j--)
            for (int k=0;k*b[j]<=a[j]&&k*c[j]<=i;k++)
                dp[i][j] = max(dp[i][j],dp[i-k*c[j]][j+1] + k*d[j]);
    
    cout<<dp[n][0]<<endl;
    return 0;
}
