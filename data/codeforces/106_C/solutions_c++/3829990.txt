#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int main()
{
	int n,m,c0,d0;
	scanf("%d%d%d%d",&n,&m,&c0,&d0);
	int dp[1010];
	memset(dp,0,sizeof(dp));
	for(int i = c0; i <= n; i ++)
		dp[i] = max(dp[i - 1],dp[i - c0] + d0);
	while(m --) {
		int ai,bi,ci,di;
		scanf("%d%d%d%d",&ai,&bi,&ci,&di);
		for(int i = n; i >= 0; i --) {
			for(int j = 1; j * bi <= ai && j * ci <= i; j ++)
				dp[i] = max(dp[i],dp[i - j * ci] + j * di);
		}
	}
	printf("%d\n",dp[n]);
	return 0;
}
