#include <iostream>
#include <cstring>
using namespace std;
int c[1100];
int main()
{
        int n,m,c0,d0;
        cin>>n>>m>>c0>>d0;
        for(int i=c0; i<=n; i++)c[i]=c[i-c0]+d0;
        for(int i=0; i<m; i++) {
                int a,b,c_,d;
                cin>>a>>b>>c_>>d;
                for(int j=0; j<a/b; j++) {
                        for(int k=n; k>=c_; k--)
                                c[k]=max(c[k-c_]+d,c[k]);
                }
        }
        cout<<c[n]<<endl;
        return 0;
}
