#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int dp[10000];
int main()
{
    int n, m, c0, d0;
    cin>>n>>m>>c0>>d0;
    memset(dp,0,sizeof(dp));
    for(int i = c0; i <= n; i++)
        dp[i] = i / c0 * d0;
    for(int i = 0; i < m; i++)
    {
        int a, b, c, d;
        cin>>a>>b>>c>>d;
        for(int j = 1; j <= a / b; j++)
            for(int k = n; k >= c; k--)
                dp[k] = max(dp[k], dp[k-c] + d);
    }
    cout<<dp[n]<<endl;
    return 0;
}
