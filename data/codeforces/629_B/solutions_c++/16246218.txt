#include <bits/stdc++.h>
using namespace std;

int m[370],f[370];

int main() {
  int n,i,j,l,r;
  char g;
  cin>>n;
  for(i=0;i<n;i++)
  {
    cin>>g>>l>>r;
    if(g=='M')
    {
      for(j=l;j<=r;j++)
        m[j]++;
    }
    else
    {
      for(j=l;j<=r;j++)
        f[j]++;
    }
  }
  int mx=0;
  for(i=0;i<370;i++)
      mx=max(mx,2*min(m[i],f[i]));
  cout<<mx<<endl;
  return 0;
}
