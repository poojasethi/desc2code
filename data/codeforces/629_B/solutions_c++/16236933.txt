#include<bits/stdc++.h>
using namespace std;
int a[370];
int b[370];
int main(){
	int n,i,j,x,y,m=0;
	char c;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>c>>x>>y;
		if(c=='M'){
			for(j=x;j<=y;j++){
				a[j]+=1;
			}	
		}
		if(c=='F'){
			for(j=x;j<=y;j++){
				b[j]+=1;
			}	
		}
	}
	for(i=1;i<=366;i++){
		m=max(m,min(a[i],b[i]));	
	}
	cout<<m*2<<endl;
}
