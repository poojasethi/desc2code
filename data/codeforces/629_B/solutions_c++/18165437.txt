#include<bits/stdc++.h>
using namespace std;
# define l long long int
# define rep(i,a,b) for(l i=a;i<b;i++)
int main()
{
	int n;
	cin>>n;
	int f[367]={0};
	int m[367]={0};
	rep(i,0,n)
	{
		char ch;
		cin>>ch;
		int a,b;
		cin>>a>>b;
		if(ch=='F')
		rep(j,a,b+1) f[j]++;
		else rep(j,a,b+1) m[j]++;
	}
	int ans=0;
	rep(i,1,367)
		{
			if(ans<min(m[i],f[i])) ans=min(m[i],f[i]);
		}
	cout<<2*ans<<"\n";
	return 0;
}
