#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>

using namespace std;

int F[367],M[367];

int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		char ch;
		int x,y;
		scanf("\n%c",&ch);
		scanf("%d%d",&x,&y);
		for(int j=x;j<=y;j++)if(ch=='F')F[j]++;else M[j]++;
	}
	int mx=0;
	for(int j=1;j<=366;j++)
		mx=max(mx,min(F[j],M[j]));
	printf("%d",mx*2);
	return 0;
}
