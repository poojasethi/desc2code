#include<bits/stdc++.h>
using namespace std;
int main(){
	int f[370]={0},m[370]={0},ans=0;
	int n,a,b;
	char c[2];
	scanf("%d",&n);
	for(int i = 0; i <n; ++i){
		scanf("%s%d%d",&c,&a,&b);
		if(c[0]=='M') m[a]++,m[b+1]--;
		else f[a]++,f[b+1]--;
	}
	for(int i = 1; i <= 366; ++i) 
			m[i]+=m[i-1],f[i]+=f[i-1],ans = max(ans,min(m[i],f[i]));
	printf("%d",2*ans);
	return 0;
	}
