#include <bits/stdc++.h>
using namespace std;
int a[500]={0},b[500]={0};
int main(){
    int n,i,l,r,j;
    char c;
    cin>>n;
    for (i=0;i<n;i++){
        cin>>c>>l>>r;
        for (j=l;j<=r;j++){
            if(c=='M')a[j]++;
            else b[j]++;
        }
    }
    int sum=0;
    for (i=1;i<=366;i++){
        sum=max(sum,min(a[i],b[i])*2);
    }
    cout<<sum<<endl;
}
