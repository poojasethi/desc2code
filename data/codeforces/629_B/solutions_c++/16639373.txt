#include <iostream>

using namespace std;

int main()
{
    int n,from,to,days[367][2]={0,0},m = 0;
    char c;
    cin >> n;
    for(int i=0;i<n;i++)
    {
        cin >> c >> from >> to;
        for(int j=from;j<=to;j++) days[j][!(c=='M')]++;
    }
    for(int i=1;i<=366;i++)
    {
        int mi = 2*min(days[i][0],days[i][1]);
        if(mi > m) m = mi;
    }
    cout << m;
    return 0;
}
