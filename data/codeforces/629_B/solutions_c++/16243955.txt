#include<iostream>
using namespace std;
int w[5001],m[5001],n,l,r,maxx,sum;
char x;
int main()
{
	cin>>n;
	for (int i=1;i<=n;++i)
	{
		cin>>x>>l>>r;
		if (x=='M')
		{
		for (int j=l;j<=r;++j)
		m[j]++;
		
			
		}
		else
	{
		for (int j=l;j<=r;++j)
		w[j]++;
	
	
	}
		
		
		
	}
	
	
	for (int i=1;i<=500;++i)
	{
		sum=min(w[i],m[i]);
		maxx=max(maxx,sum*2);
		
		
	}
	cout<<maxx;
	
	
}
