#include <bits/stdc++.h>
using namespace std;
int main(){
	int n,m[400]={0},f[400]={0},x,y;
	char c;
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>c>>x>>y;
		if(c=='M'){
			for(int j=x;j<=y;j++)m[j]++;
		}
		else for(int j=x;j<=y;j++)f[j]++;
	}
	for(int i=1;i<=366;i++){
		if(min(m[i],f[i])>m[0])m[0]=min(m[i],f[i]);
	}
	cout<<2*m[0]<<endl;
}
