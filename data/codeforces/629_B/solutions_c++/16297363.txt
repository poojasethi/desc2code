#include<bits/stdc++.h>
using namespace std;
int M[400], K[400];
int main() {
	int n;
	scanf("%d\n", &n);
	char c;
	int a, b;
	for(int i = 0; i < n; i++) {
		cin >> c >> a >> b;
		if(c=='M') for(int j = a; j <= b; j++) M[j]++;
		else for(int j = a; j <= b; j++) K[j]++;
	}
	int wynik = 0;
	for(int i = 0; i <= 400; i++) wynik = max(wynik, min(M[i], K[i])*2);
	printf("%d\n", wynik);
}
