#include <iostream>

using namespace std;
long long n,B,F,S;
long a[5000][2];
char c[5000];
int main()
{   cin>>n;
S=0;
for (int i=0; i<n; i++){
  cin>>c[i]>>a[i][1]>>a[i][2];
}
for (int j=1; j<367; j++){
B=0; F=0;
    for (int i=0; i<n; i++){
        if (a[i][1]<=j&&a[i][2]>=j)  {if (c[i]=='M') B++; else F++;}

    }
    B=min(B,F);
    if (S<2*B) S=2*B;

}

    cout << S << endl;
    return 0;
}
