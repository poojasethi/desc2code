#include <iostream>
#include <cmath>
using namespace std;


const int MAXN = 5 * 1000 + 10, MAXD = 400;
int f[MAXD], m[MAXD];

int main()
{
	int n;
	cin>> n;

	for(int i = 0; i < n; i ++)
	{
		int a, b;
		char c;
		cin>> c >> a >> b;
		for(int j = a; j <= b; j ++)
			if(c == 'F')
				f[j] ++;
			else
				m[j] ++;
	}

	int ans = 0;
	for(int i = 0; i <= 366; i ++)
		ans = max(ans, min(f[i], m[i]) * 2);

	cout<< ans << endl;

}
