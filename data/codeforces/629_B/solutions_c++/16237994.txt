#include <cstdio>
#include <algorithm>
using namespace std;

int n,f,x,y,ans,a[2][400];
char s[5];

int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%s %d %d",s,&x,&y);
		if(s[0]=='M') f=0;
		else f=1;
		for(int j=x;j<=y;j++) a[f][j]++;
	}
	for(int i=1;i<=366;i++)
		ans=max(ans,min(a[0][i],a[1][i]));
	printf("%d\n",ans*2);
	return 0;
}
