#include <iostream>
#include <cmath>

using namespace std;

int n;
int numf[500],numm[500];

int main()
{
  cin>>n;
  for (int i=0;i<n;i++) {
    char c;
    int a,b;
    cin>>c>>a>>b;
    for (int j=a;j<=b;j++) {
      numf[j]+=c=='F';
      numm[j]+=c=='M';
    }
  }
  int sol=0;
  for (int i=0;i<500;i++)
    sol=max(sol,2*min(numf[i],numm[i]));
  cout<<sol<<endl;
}
