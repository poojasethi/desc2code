#include<bits/stdc++.h>
using namespace std;


int y[2][370];
//int y[365];


int main()
{
	int n;
	cin >> n;
	char g;
	int a, b;
	
	for(int i = 0; i < n; i++){
		cin >> g >> a >> b;
		for( int j = a; j <= b; j++){
			y[g=='F'][j]++;
		}
	}
	
	
	
	int maxres = 0;
	for(int i = 0 ; i < 370; i++){
		maxres = max(maxres,min(y[0][i], y[1][i]) );
	}
	
	cout << 2 * maxres ;
		
	
	
	
	
	
	return 0;
}
