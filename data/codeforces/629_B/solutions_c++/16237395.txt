#include <bits/stdc++.h>
using namespace std;
vector<int>A,B;
int main(){
	int n;
	while(cin>>n){
		A.assign(400,0);
		B.assign(400,0);
		int a,b;
		string gen;
		for(int i=0;i<n;i++){
			cin>>gen>>a>>b;
			if(gen[0]=='M')
				for(int j=a;j<=b;j++)A[j]++;
			else
				for(int j=a;j<=b;j++)B[j]++;		
		}
		int r=0;
		for(int i=0;i<400;i++)
			r=max(r,2*min(A[i],B[i]));
		cout<<r<<endl;
	}
	return 0;
}
