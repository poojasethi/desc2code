#include <bits/stdc++.h>

using namespace std;

int n,m[367],f[367],i,j,a,b,ans;
char g;
int main()
{
    scanf("%d",&n);
    for(i=0;i<n;++i)
    {
        scanf(" %c %d %d",&g,&a,&b);
        int* v=(g=='M'?m:f);
        for(j=a;j<=b;++j) ++v[j];
    }
    for(i=1;i<=366;++i)
    {
        ans=max(ans,2*min(m[i],f[i]));
    }
    printf("%d\n",ans);
    return 0;
}
