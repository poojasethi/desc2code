#include<bits/stdc++.h>
using namespace std;
long long f[10001],m[10001],l,r,n,ans;
char k;
int main(){

cin>>n;

for(int i=1; i<=n; i++)
{
cin>>k>>l>>r;	

if(k=='F') for(int i=l; i<=r; i++) m[i]++;
else for(int i=l; i<=r; i++) f[i]++;
}

for(int i=1; i<=366; i++)	
ans=max(ans,min(m[i],f[i])*2);

cout<<ans<<endl;

return 0;
}
