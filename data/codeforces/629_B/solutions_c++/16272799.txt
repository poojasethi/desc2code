#include<cstdio>
#include<algorithm>
using namespace std;

int n, a, b, ans;
int s[2][555];
char op[3];

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%s%d%d", &op, &a, &b);
		int id = (op[0] == 'F');
		for(int j=a;j<=b;j++)
			s[id][j]++;
	}
	ans=0;
	for(int i=1;i<=366;i++)
		ans=max(ans, min(s[0][i], s[1][i])*2);
	printf("%d\n", ans);
	
	return 0;
}
