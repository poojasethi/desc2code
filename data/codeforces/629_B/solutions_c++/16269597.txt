#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

using namespace std;

int mc[369],fc[369],n,ans,a,b;
char x;

int main(){
  scanf("%d",&n);
  FI(i,1,n){
    scanf("%c%c %d %d",&x,&x,&a,&b);
    if(x=='M') FI(j,a,b) mc[j]++;
    if(x=='F') FI(j,a,b) fc[j]++;
  }
  FI(i,1,366) ans=max(ans,2*min(mc[i],fc[i]));
  printf("%d\n",ans);
  return 0;
}
