#include <bits/stdc++.h>
#define ri(n) scanf("%d\n",&n)
#define rcii(n,a,b) scanf("%c %d %d\n",&n,&a,&b)
#define FOR(i,l,r) for(int i=l;i<r;i++)
using namespace std;

char S;
int N, a, b, A[368], B[368], ans=0;

int main() {
	ri(N); FOR(i,0,N) {
		rcii(S,a,b); b++;
		if(S=='F') {A[a]++;A[b]--;}
		if(S=='M') {B[a]++;B[b]--;}
	}
	a=b=0;
	FOR(i,1,367) {
		a+=A[i];b+=B[i];
		ans=max(ans,2*min(a,b));
	}
	printf("%d\n",ans);
}
