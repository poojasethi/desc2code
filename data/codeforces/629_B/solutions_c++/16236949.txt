#include <iostream>
#include <algorithm>
using namespace std;

int F[400],B[400],N,rs = 0;
char c;
int main(){
	cin >> N;
	for(int i = 0,x,y;i<N;i++){
		cin >> c >> x >> y;
		if(c == 'M'){
			for(int i = x;i<=y;i++)B[i]++;
		}else
			for(int i = x;i<=y;i++)F[i]++;
	}
	for(int i = 0;i<=370;i++)
		if(rs < 2*min(F[i],B[i])) rs = 2*min(F[i],B[i]);
	cout << rs;
	return 0;
}
