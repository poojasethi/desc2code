#include <bits/stdc++.h>

using namespace std;

int main()
{
    int k,s,f,i,n,ml[400]={0},fl[400]={0},ans=0;
    char j;
    cin>>n;
    for(i=1;i<=n;i++){
        cin>>j>>s>>f;
    for(k=s;k<=f;k++){
        if(j=='M')
            ml[k]++;
        else
            fl[k]++;
    }
}
for(i=1;i<=366;i++){
    if(2*min(ml[i],fl[i])>ans)
        ans=2*min(ml[i],fl[i]);
}
cout<<ans;
}
