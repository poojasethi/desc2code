int j,n,t,i,a[400],b[400],m,p;
int main()
{
    scanf("%d",&t);
    while(t--)
	{
        scanf("%s%d%d",&p,&n,&m);
             for(j=n;j<=m;++j)
               if (p=='M') ++a[j];
               else ++b[j];
    }
    for (i=j=0;i<=370;++i)
    {
        if (a[i]>b[i]) a[i]=b[i];
         if (j<a[i]) j=a[i];
    }
