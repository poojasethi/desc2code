#include<bits/stdc++.h>
using namespace std;
int main()
{
int n;
cin>>n;
int f[400]={0};
int m[400]={0};
while(n--)
{
	string s;
	int a,b;
	cin>>s>>a>>b;
	if(s[0]=='F')
	{
		for(int i=a;i<=b;i++)
		  f[i]++;
		
	}
	else
     	for(int i=a;i<=b;i++)
		 m[i]++;	
}
int max=0;
for(int i=1;i<=366;i++)
  {
    m[i]=m[i]<f[i]?m[i]:f[i];
    if(max<m[i])max=m[i];
 }

    cout<<2*max<<endl;;
}
