#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long int i,j,k,m[500]={0},f[500]={0},max=-1,a[5005],b[5005],n,ans=1000000000;
	char s[5005];
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>s[i]>>a[i]>>b[i];
		for(j=a[i];j<=b[i];j++)
		{
			if(s[i]=='F')
				f[j]++;
			else
				m[j]++;
		}		
	}
	for(i=1;i<=366;i++)
	{
		ans=min(f[i],m[i]);
		if(ans>max)
			max=ans;
	}
	//cout<<s<<endl;
	cout<<max*2<<endl;
	return 0;
}
