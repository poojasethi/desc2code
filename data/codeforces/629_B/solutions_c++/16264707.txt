#include <iostream>
using namespace std;
int n,s1[370],s2[370],l,r,ans;
char t;

int main()
{
    cin>>n;
    for (int i=1;i<=n;++i)
    {
        cin>>t>>l>>r;
        for (int j=l;j<=r;++j) s1[j]+=bool(t=='F'),s2[j]+=bool(t=='M');
    }
    for (int i=1;i<=366;++i)
        ans=max(ans,min(s1[i],s2[i])*2);
    cout<<ans;
    return 0;
}
