#include<bits/stdc++.h>
using namespace std;
int main(){
	int f[370]={0},m[370]={0},ans=0;
	int n,a,b;
	char c[2];
	scanf("%d",&n);
	for(int i = 0; i <n; ++i){
		scanf("%s%d%d",&c,&a,&b);
		for(int j = 1; j <= 366; ++j) {
			if(a <= j && b >= j) {
				if(c[0]=='M') m[j]++;
				else f[j]++;
			}
		}
	}

	for(int i = 1; i <= 366; ++i) 
			ans = max(ans,min(m[i],f[i]));
	printf("%d",2*ans);
	return 0;
	}
