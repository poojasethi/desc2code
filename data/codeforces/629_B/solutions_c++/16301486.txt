#include <iostream>
#include <cstdio>
int a[400],b[400],i,n,l,r,x,y,ans=0;
char ch;
int main()
{
		scanf("%d\n",&n);
		for (i=1;i<=n;++i)
		{
			scanf("%c %d%d\n",&ch,&l,&r);
			if (ch=='M') ++a[l],--a[r+1];
				else ++b[l],--b[r+1];
		}
		x=y=0;
		for (i=1;i<=366;++i)
		{
			x+=a[i];y+=b[i];
			ans=std::max(ans,std::min(x,y)*2);
		}
		printf("%d\n",ans);
}
