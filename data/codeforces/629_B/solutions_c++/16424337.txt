#include<bits/stdc++.h>
using namespace std;
int ara[400][2];
int main(){
    int n,l,h=2;
    char ch;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>ch>>l>>h;
        if(ch=='M') { for(int j=l;j<=h;j++) ara[j][0]++;}
        else { for(int j=l;j<=h;j++) ara[j][1]++;}
    }

    int mx=0;
    for(int i=1;i<=366;i++){
        mx=max(mx,min(ara[i][0],ara[i][1]));}
    cout<<2*mx;
}
