#include<iostream>
using namespace std;
int n,m,a,b,i,j,boys[367],girls[367];
char g;
int main()
 {
 cin>>n;
 for(i=0;i<n;i++)
  {
  cin>>g>>a>>b;
  for(j=a;j<=b;j++)
    {
    if(g=='M')
    boys[j]=boys[j]+1;
    else
    girls[j]=girls[j]+1;
    }
  }
  for(i=0;i<=366;i++)
   m=max(m,min(girls[i],boys[i]));
   cout<<2*m;
   return 0;
 }
