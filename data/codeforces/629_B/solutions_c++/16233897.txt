#include <bits/stdc++.h>
using namespace std;
int cntm[400], cntf[400];
int main(){
	int n;
	cin>>n;
	char x;
	int y, z;
	for (int i = 0; i < n; i++) {
		cin>>x>>y>>z;
		if(x == 'M') for (int j = y; j <= z; j++) cntm[j]++; 
		else for (int j = y; j <= z; j++) cntf[j]++; 
	}
	int maxi = 0;
	for (int i = 1; i <= 366; i++) {
		maxi = max(maxi, min(cntm[i], cntf[i]));
	}
	cout<<2 * maxi<<endl;
}
