#include<iostream>
#include<cmath>
using namespace std;
int x,i,j,df[400],dm[400],a,b,o=0;
char c;
int main(){
    cin>>x;
    for(j=0;j<x;j++){
		cin>>c>>a>>b;
		for(i=a;i<=b;i++){
			if(c=='F')df[i]++;
			else dm[i]++;
			if(min(dm[i],df[i])*2>o)o=2*(min(dm[i],df[i]));
		}
	}
	cout<<o;
	return 0;
}
