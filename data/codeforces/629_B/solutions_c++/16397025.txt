#include<bits/stdc++.h>
using namespace std;
int n;
char s[5005][2];
int a[5005];
int b[5005];
int ma;
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;++i){
		scanf("%s%d%d",s[i],&a[i],&b[i]);
	}
	for(int i=1;i<=366;++i){
		int mf[2]={};
		for(int j=0;j<n;++j){
			if(a[j]<=i&&i<=b[j]){
				++mf[s[j][0]=='M'];
			}
		}
		ma=max(ma,min(mf[0],mf[1]));
	}
	printf("%d\n",ma*2);
	return 0;
}
