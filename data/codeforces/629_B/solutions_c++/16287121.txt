#include<cstring>
#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
#define rep(i,l,r) for(int i=l;i<=r;i++)
int a[370],b[370];
int n,l,r;
char opt[3];
int main(){
	cin>>n;
	rep(i,1,n){
		scanf("%s",opt);
		scanf("%d%d",&l,&r);
		if(opt[0]=='M') 
			rep(j,l,r) a[j]++;
		else rep(j,l,r) b[j]++;
	}
	int ans=0;
	rep(i,1,366)
		ans=max(ans,min(a[i],b[i]));
	cout<<ans*2<<endl;
}
