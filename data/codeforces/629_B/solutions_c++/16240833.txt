#include<bits/stdc++.h>
using namespace std;

int main() {
	int N,i,j,a,b,F[370],M[370],Max=-1;
	memset(F,0,sizeof F);
	memset(M,0,sizeof M);
	char C;
	scanf("%d",&N);
	for(i=0;i<N;++i) {
		cin >> C >> a >> b;
		if(C=='F') {
			for(j=a;j<=b;++j) ++F[j];
		}
		else if(C=='M') {
			for(j=a;j<=b;++j) ++M[j];
		}
	}
	for(i=0;i<370;++i) {
		if(min(F[i],M[i])>Max) Max=min(F[i],M[i]);
	}
	cout << 2*Max << endl;
}
