#include<bits/stdc++.h>
using namespace std;
int f[555],m[555],a[555],b[555];
int main()
{
int n;
scanf("%d",&n);
char s[5];
int i,x,y;
for(i=0;i<n;i++)
{
scanf("%s%d%d",&s,&x,&y);
if(s[0]=='M'){a[x]++;a[y+1]--;}
else {b[x]++; b[y+1]--;}
}
int fir=0;
int u,v;
u=v=0;
for(i=1;i<=500;i++)
{
u+=a[i];
v+=b[i];
fir = max(fir,min(u,v));
}

printf("%d\n",2*fir);
return 0;
}
