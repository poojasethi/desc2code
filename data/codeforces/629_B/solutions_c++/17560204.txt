#include<cstdio>
#include<algorithm>
using namespace std;
const int N=5005,M=366;
int a[400],b[400],n,c,l,r,ans;
int main(){
	scanf("%d",&n);
	while (n--){
		scanf(" %c%d%d",&c,&l,&r);
		if (c=='M') for (int i=l;i<=r;i++) a[i]++;
		else for (int i=l;i<=r;i++) b[i]++;
	}
	for (int i=1;i<=M;i++) ans=max(ans,min(a[i],b[i]));
	printf("%d",ans*2);
	return 0;
}
