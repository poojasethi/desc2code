#include<iostream>
#define N 367
using namespace std;
int Erkek[N],Ayal[N];
int min(int a, int b)
{
	if(a>b) return b; return a;
}
int main()
{
	int n,i,j,k,l;
	cin>>n; char c; int a, b,max=0;
	for(i=1; i<=n; i++)
	{
		cin>>c>>a>>b;
		for(j=a; j<=b; j++)
		{
			if(c=='M') Erkek[j]++;
			else Ayal[j]++;
			if(min(Erkek[j],Ayal[j])*2>max) max=min(Erkek[j],Ayal[j])*2;
		}
	}
	cout<<max;
	return 0;
}
