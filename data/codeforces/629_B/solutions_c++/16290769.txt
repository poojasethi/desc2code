#include <algorithm>
#include <iostream>
#include <string>
#include <cstdio>
using namespace std;
int a[400],b[400];
int main()
{
	int n,i,x,y,j,ans=0;
	char c;
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	{
		scanf(" %c",&c);
		scanf("%d%d",&x,&y);
		for (j=x;j<=y;j++) if (c=='M') a[j]++; else b[j]++;
	}
	for (i=1;i<=366;i++) ans=max(ans,min(a[i],b[i]));
	printf("%d\n",ans*2);
	return 0;
}
