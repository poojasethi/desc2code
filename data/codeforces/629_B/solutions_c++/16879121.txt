#include<bits/stdc++.h>
using namespace std;

pair<int,int> day[400];

int main()
{
	int n,a,b,mx=0;
	char g;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>g>>a>>b;
		
		for(int i=a;i<=b;i++)
		{
			if(g=='F')
			day[i].first++;
			else
			day[i].second++;
		}
		for(int i=0;i<400;i++)
		mx=max(mx,min(day[i].first,day[i].second)*2);
	}
	cout<<mx<<endl;
	end:
    return 0;
}
