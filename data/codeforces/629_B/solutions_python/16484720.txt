male=[0]*367
female=[0]*367
n = int(raw_input())
for x in range(n):
	s = raw_input().split()
	a = int(s[1])
	b = int(s[2])
	for i in range(a,b+1):
		if s[0] == 'M':
			male[i]+=1
		else:
			female[i]+=1
ans = -1
for i in range(1,367):
	ans = max(ans, 2*min(female[i], male[i]))
print ans
