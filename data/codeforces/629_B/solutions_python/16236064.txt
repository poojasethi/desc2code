
n = input()
males = [0]*367
females = [0]*367
for i in range(n):
        g,s,e = [i for i in raw_input().split(' ')]
        
        s=int(s)
        e=int(e)
        if g=='F':
                for i in range(s,e+1):
                        females[i]+=1
        else:
                for i in range(s,e+1):
                        males[i]+=1
        best=0
for i in range(367):
        best=max(best,min(males[i],females[i]))

print best*2
