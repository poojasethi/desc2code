n = int(raw_input())
m = [0] * 400
f = [0] * 400

for _ in xrange(n):
    ss = raw_input().split()
    sd, ed = int(ss[1]), int(ss[2])
    if ss[0] == 'M':
        for d in xrange(sd, ed+1):
            m[d] += 1
    else:
        for d in xrange(sd, ed+1):
            f[d] += 1

ans = 0
for d in xrange(1, 367):
    ans = max(ans, min(m[d], f[d]))

print ans * 2
