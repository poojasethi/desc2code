n = int(raw_input())
valasz = 0

f,m = [0]*368,[0]*368

for i in xrange(n):
    sor = raw_input().split()
    nem,kezd,veg = list(str(sor[0])) + map(int,sor[1:])
    if nem == 'F':
        f[kezd] += 1
        f[veg+1] -= 1
    else:
        m[kezd] += 1
        m[veg+1] -= 1


aktf,aktm = 0,0

for i in xrange(1,367):
    aktf += f[i]
    aktm += m[i]
    valasz = max(valasz,min(aktm,aktf))

print valasz*2
