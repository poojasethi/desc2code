
n = input()

DM = [0] * 400
DF = [0] * 400

for _ in xrange(n):
    sex, start, end = raw_input().split()
    start = int(start)
    end = int(end)
    for i in xrange(start, end+1):
        if sex == 'M':
            DM[i] += 1
        else:
            DF[i] += 1

ma = 0
for i in xrange(400):
    ma = max(ma, min(DF[i], DM[i]))
print 2 * ma
