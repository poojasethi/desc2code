def fun():
    n = int(raw_input())
    g = []
    for i in range(0,n):
        g.append(raw_input().split())
    ans = 0
    #print g
    for i in range(1,367):
        m = 0
        f = 0
        for j in range(0,n):
            if g[j][0] == 'M' and int(g[j][1])<= i and int(g[j][2])>=i:
                m += 1
            if g[j][0] == 'F' and int(g[j][1])<= i and int(g[j][2])>= i:
                f += 1
        ans = max(ans , 2*min(m,f))
    print ans

fun()
