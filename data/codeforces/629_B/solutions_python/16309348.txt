import os
M = [0]*400
F = [0]*400

n = int(raw_input())
for _ in range(n) :
	x = raw_input().split()
	l, r = map(int, x[1:])
	for i in range(l, r+1):
		if x[0] == 'M' :
			M[i] += 1
		else :
			F[i] += 1
ans = 0
for i in range(1, 367):
	ans = max(ans, 2*min(M[i], F[i]))
print ans
