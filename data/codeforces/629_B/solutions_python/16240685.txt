n = int(raw_input())

M = [0]*400
F = [0]*400

for i in range(n):
    t, a, b = raw_input().split()
    a = int(a)
    b = int(b)
    if t == "M":
        for j in range(a, b + 1):
            M[j] += 1
    else:
        for j in range(a, b + 1):
            F[j] += 1

best = 0

for i in range(400):
    best = max(best, min(M[i], F[i]))

print best * 2
