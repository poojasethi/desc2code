f = [[0 for i in range(367)] for j in range(2)]

n = int(raw_input())
for i in range(n):
    g, s, e = raw_input().split()
    s, e = int(s), int(e)
    if g == 'F':
        for q in range(s, e+1):
            f[1][q] += 1
    else:
        for q in range(s, e+1):
            f[0][q] += 1
mins = [min(q, w) for q, w in zip(f[0], f[1])]
print max(mins)*2
