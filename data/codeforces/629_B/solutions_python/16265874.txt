n = input()
l=[0]*366
p = []
for i in range(n):
    a = raw_input().split()    
    a[1] = int(a[1])   
    a[2] = int(a[2])
    p.append(a)


for i in range(1, 367):
    males = 0
    females = 0
    for j in p:
   #     print j
        if j[1] <= i and i <= j[2]:
            if j[0] == 'M':
                males+=1
            else:
                females+=1
    if males > females :
        l[i-1] = females*2
    else :
        l[i-1] = males*2
print max(l)
