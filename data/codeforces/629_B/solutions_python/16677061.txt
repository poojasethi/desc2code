n = int(raw_input())

daysM = [0]*400
daysF = [0]*400
store = []


for _ in range(n):
    q = raw_input().split()
    if q[0] == 'M':
        for i in range(int(q[1]),int(q[2])+1):
            daysM[i]+=1

    else:
        for i in range(int(q[1]),int(q[2])+1):
            daysF[i]+=1

            

    

ans = max(min(x,y) for x,y in zip(daysM,daysF))

    


print ans*2
