n = int( raw_input() )

MAXN = 5000
MAXD = 366

gender = ['-']*MAXN
st = [0]*MAXN
en = [0]*MAXN

for i in range( 0, n ):
	x = raw_input().split( ' ' )
	gender[i] = x[0]
	st[i] = int( x[1] )
	en[i] = int( x[2] )

ans = 0

for day in range( 1, MAXD+1 ):

	totM = 0
	totF = 0

	for i in range( 0, n ):
		if day >= st[i] and day <= en[i]:
			if gender[i] == 'M':
				totM += 1
			else:
				totF += 1
				
	ans = max( ans, min( totM, totF ) )
	
print "%d" % (ans+ans)
