n = int(raw_input())
f = [[0] * 370, [0] * 370]
for _ in xrange(n):
    line = raw_input().split()
    s, t = int(line[1]), int(line[2])
    f[line[0] == 'M'][s] += 1
    f[line[0] == 'M'][t+1] -= 1
ans = 0
for i in xrange(1, 370):
    f[0][i] += f[0][i-1]
    f[1][i] += f[1][i-1]
    ans = max(ans, min(f[0][i], f[1][i]))
print ans * 2
