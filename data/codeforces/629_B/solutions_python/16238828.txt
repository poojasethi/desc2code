n = int(raw_input())

M_num = [0 for i in range(0, 366)]
F_num = [0 for i in range(0, 366)]

for _ in range(0, n):
    l = raw_input().split()
    gender, fday, sday = l[0], int(l[1]), int(l[2])
    cl = None
    if gender == "M":
        cl = M_num
    else:
        cl = F_num
    for i in range(fday-1, sday):
        cl[i] += 1
    
max_possible = 0
    
for i in range(0, 366):
    max_possible = max(max_possible, min(M_num[i], F_num[i]))
        
print max_possible * 2
