n = int(raw_input())
guest, day = [], []
for p in range(n):
    x,y,z = raw_input().split()
    guest.append([x, int(y), int(z)])  
    day.append(int(guest[p][1]))
    day.append(int(guest[p][2]))
day = sorted(set(day))
res = 0
for i in day:
    m,f = 0,0
    for j in guest:
        if i >= j[1] and i <= j[2]:
            if j[0] == 'M':
                m += 1
            else:
                f += 1
    res = max(res,min(m,f))

print res*2
