if(__name__ == "__main__"):
	n = int(raw_input())
	male, female = [0] * 367, [0]*367
	for i in range(n):
		sex, start, end = raw_input().split(' ')
		start, end = int(start), int(end)
		if(sex == "M"):
			for j in range(start, end + 1):
				male[j] += 1
		else:
			for j in range(start, end + 1):
				female[j] += 1

	ans = 0
	for i in range(367):
		ans = max(ans, min(male[i], female[i]))

	print ans*2
