t = input()
m,n = [],[]
for i in range(t):
	a,b,c = raw_input().split()
	b,c = int(b),int(c)
	if a=='M': m.append((b,c))
	else: n.append((b,c))
p,q = [0]*370,[0]*370
for i in range(1,367):
	for (x,y) in m:
		if x<=i<=y: p[i]+=1
	for (x,y) in n:
		if x<=i<=y: q[i]+=1
for i in range(1,367):
	p[i]=min(p[i],q[i])
print max(p)*2
