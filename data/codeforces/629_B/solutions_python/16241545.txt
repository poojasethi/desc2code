n = input()
vals = []
disc = [False] * n
days = [[0,0] for _ in xrange(368)]  
for i in xrange(n):
    row = raw_input().split(' ')
    if row[0] == 'M':
        g = 0
    else:
        g = 1
    d1,d2 = int(row[1]), int(row[2])
    for j in xrange(d1,d2+1):
        days[j][g] += 1 

val = 0
for i in xrange(len(days)):
    val  = max(val, min(days[i][0],days[i][1]))
print val*2
