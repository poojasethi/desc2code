n=int(raw_input())
arr=[]
for i in range(0,n):
    arr.append(raw_input().split())
    arr[i][1]=int(arr[i][1])
    arr[i][2]=int(arr[i][2])
mx=float("-inf")
for day in range(1,367):
    males=0
    females=0
    for i in range(0,n):
        if day>=arr[i][1] and day<=arr[i][2]:
            if arr[i][0]=='M':
                males+=1
            else:
                females+=1
    mx=max(mx,2*min(males,females))
print mx
