m = [0 for x in xrange(368)]
f = [0 for x in xrange(368)]
for _ in xrange(input()):
    l = raw_input().split()
    if l[0] == 'M':
        m[int(l[1])] += 1
        m[int(l[2]) + 1] -= 1
    else:
        f[int(l[1])] += 1
        f[int(l[2]) + 1] -= 1
for i in xrange(1, 368):
    m[i] += m[i-1]
    f[i] += f[i-1]
a = [2*min(m[x], f[x]) for x in xrange(368)]
print max(a)
