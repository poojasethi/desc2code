n = int(raw_input())

daysM = [0]*400
daysF = [0]*400
store = []


for _ in range(n):
    q = raw_input().split()
    if q[0] == 'M':
        daysM[int(q[1])]+=1
        daysM[int(q[2])+1]-=1

    else:
        daysF[int(q[1])]+=1
        daysF[int(q[2])+1]-=1

ans = 0
            
M = 0
F = 0

for _ in range(367):
    M+= daysM[_]
    F+= daysF[_]
    ans = max(ans,2*min(M,F))

 


print ans
