import fileinput

# input
source = fileinput.input()

# source = open('data.txt', 'r').readlines()

data = []
for line in source:
	data.append(line.strip().split())

data = data[1:]
fs, ms = [0] * 367, [0] * 367

for f in data:
	if f[0] == 'M':
		for i in range(int(f[1]), int(f[2]) + 1):
			ms[i] += 1
	else:
		for i in range(int(f[1]), int(f[2]) + 1):
			fs[i] += 1

mx = 0
for i in range(0, 367):
	if mx < min(ms[i], fs[i]):
		mx = min(ms[i], fs[i])

print mx * 2
