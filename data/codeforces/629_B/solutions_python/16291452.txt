import collections

n = input()
l = []
Guest = collections.namedtuple('Guest', ['gender', 'begin', 'end'])
for i in range(0, n):
	s = raw_input().split()
	l.append(Guest._make([s[0], int(s[1]), int(s[2])]))

ans = 0
for i in range(1, 367):
	def counter(gender):
		return sum(1 for g in l if g.gender == gender and g.begin <= i and i <= g.end)
	ans = max(ans, min(counter('M'), counter('F')))

print ans * 2
