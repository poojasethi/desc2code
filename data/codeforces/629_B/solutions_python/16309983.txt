n = int(raw_input())

mday = [0]*500
fday = [0]*500


for i in range(n):
    inp = raw_input().split()
    a = int(inp[1])
    b = int(inp[2])
    for j in range(a,b+1):
        if inp[0] is 'M': mday[j] = mday[j] + 1
        if inp[0] is 'F': fday[j] = fday[j] + 1

comp = [min(mday[i], fday[i])*2 for i in range(500)]

print max(comp)
