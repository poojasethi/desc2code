import sys

n = int(sys.stdin.readline())

FN=[0 for _ in xrange(367)]
MN=[0 for _ in xrange(367)]
for _ in xrange(n):
  g, a, b = sys.stdin.readline().split()
  a = int(a)
  b = int(b)
  if g == 'F':
    FN[a:(b+1)] = map(lambda e: e+1, FN[a:(b+1)])
  else:
    MN[a:(b+1)] = map(lambda e: e+1, MN[a:(b+1)])

r = 0
for i in xrange(1,367):
  r = max(r, min(MN[i], FN[i]) * 2)

print r
