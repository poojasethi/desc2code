n = int(raw_input())

dh = [0]*366
dm = [0]*366

for i in xrange(n):
	a,b,c = raw_input().split()
	
	if a=='M': 
		for n in xrange(int(b), int(c)+1): dh[n-1] += 1
	if a=='F': 
		for n in xrange(int(b), int(c)+1): dm[n-1] += 1

u = 0
for i in xrange(366):
	k = min(dh[i], dm[i])
	if k > u: u = k
	
print u*2
