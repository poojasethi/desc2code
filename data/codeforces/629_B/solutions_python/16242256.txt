n = int(raw_input().rstrip())
males = []
females = []
for i in range(n):
	s = raw_input().split()
	x = int(s[1])
	y = int(s[2])
	if s[0] == 'M':
		males.append([x, y])
	else:
		females.append([x, y])

ans = 0
for i in range(1, 367):
	m = f = 0
	for male in males:
		if male[1] >= i and male[1] - i <= male[1] - male[0]:
			m += 1
	for female in females:
		if female[1] >= i and female[1] - i <= female[1] - female[0]:
			f += 1
	cnt = min(m, f)
	cnt *= 2
	ans = max(ans, cnt)

print ans
