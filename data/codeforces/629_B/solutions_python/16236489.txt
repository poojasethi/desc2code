n = int(raw_input())
v = []
for i in range(0,n):
    v.append(raw_input().split())


cc = 0

for i in range(1,367):
    f = 0
    m = 0
    for j in v:

        if int(j[1]) <= i and int(j[2]) >= i:
            if j[0] =='M':
                m+=1
            else:
                f+=1

    cc = max(cc , min(m,f)*2)
print cc
