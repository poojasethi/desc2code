#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

def ig():
    l = raw_input()
    l = l.split()
    return l[0], int(l[1]), int(l[2])

n, D = ir(), []
for i in range(n):
    D.append(ig())

def count(d):
    nc = nm = nf = 0
    for g, a, b in D:
        if d<a or d>b: continue
        nf += 1 if g=='F' else 0
        nm += 1 if g=='M' else 0
    return nf, nm

ma = 0
for d in range(1, 366+1):
    nf, nm = count(d)
    nc = 2*min(nf, nm)
    if  nc>ma: ma=nc

print ma
