n = input()
male = [0]*368
female = [0]*368
for i in xrange(n):
	stats = raw_input().split()
	if stats[0] == 'M':
		for j in xrange(int(stats[1]), int(stats[2])+1):
			male[j] += 1
	else:
		for j in xrange(int(stats[1]), int(stats[2])+1):
			female[j] += 1

result = -1
for i in xrange(1, 367):
	result = max(result, min(male[i], female[i]))
print 2*result
