import sys

f = sys.stdin
n = int(f.readline())
d = []
for x in xrange(367):
    d += [[0, 0]]
for i in xrange(n):
    a, b, c = f.readline().strip().split(' ')
    b = int(b)
    c = int(c)
    for j in xrange(b, c + 1):
        if a == 'M':
            d[j][0] += 1
        else:
            d[j][1] += 1
ans = 0
for x in xrange(367):
    if min(d[x][0], d[x][1]) > ans:
        ans = min(d[x][0], d[x][1])
print ans * 2
