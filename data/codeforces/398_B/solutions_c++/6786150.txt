#include <cstdio>

using namespace std;

double f[2055][2055];
bool H[2055],Z[2055];
int A,B,n,m,h,z;

int main(){
    for(scanf("%d%d",&n,&m);m--;){
        scanf("%d%d",&h,&z);
        if(!H[h]){H[h]=true;A++;}
        if(!Z[z]){Z[z]=true;B++;}
    }
    for(int i=n;i>=A;--i)
    for(int j=n;j>=B;--j)
    if(i<n || j<n)
    f[i][j]=(n*n+(n-i)*j*f[i+1][j]+(n-j)*i*f[i][j+1]+(n-i)*(n-j)*f[i+1][j+1])/(n*n-i*j);
    printf("%.9lf\n",f[A][B]);
}
