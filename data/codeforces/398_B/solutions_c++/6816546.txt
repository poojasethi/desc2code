#include <cstdio>
int n;
double s;
bool row[2001],col[2001];
double Table[2001][2001];

int main(){
	int m,r,c;
	scanf("%d%d",&n,&m);
	s=(double)n*n;
	for(int i=0;i<m;i++){
		scanf("%d%d",&r,&c);
		row[r]=1;  col[c]=1; 
	}
	int R=n,C=n; 
	for(int i=1;i<=n;i++){
		if(row[i])R--;
		if(col[i])C--;
	}


	double N=n;
	for(int i=1;i<=n;i++)Table[i][0]=(N/i)+Table[i-1][0];
	for(int i=1;i<=n;i++)Table[0][i]=(N/i)+Table[0][i-1];
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++){
					
			Table[i][j]=i*(n-j)*Table[i-1][j] + j*(n-i)*Table[i][j-1] + i*j*Table[i-1][j-1]+N*N;
			Table[i][j]/=N*(i+j)-i*j;
	
		}
	
	printf("%lf\n",Table[R][C]);
}
