#include <iostream>
#include <cmath>
#include <cstdio>
#define N 2005
using namespace std;

int main()
{
    int x=0,y=0,n,m,r,c,nr[N]={0},nc[N]={0};
    double f[N][N]={0};
    cin >> n >> m;
    while (m--)
    {
        cin >> r >> c;
        if (nr[r]++==0)x++;
        if (nc[c]++==0)y++;
    }
    //cout << x << ' ' << y << endl;
    for (int i=n;i>=x;i--)
        for (int j=n;j>=y;j--)
        {
            if (i*j==n*n) continue;
            f[i][j]=1+(f[i+1][j]*(n-i)*j+f[i][j+1]*i*(n-j)+f[i+1][j+1]*(n-i)*(n-j))/(n*n-i*j);
            double y= (double)i*j/(n*n);
            f[i][j]+=y/(1-y);
        }
    printf("%.6f\n",f[x][y]);
    return 0;
}
