#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

double a[3005][3005];
int b[3005], c[3005];
int main() {
    int n, m, N = 0, M = 0, i, x, y, j;

    scanf("%d%d",&n,&m);
    for (i = 0; i < m; i++) {
        scanf("%d%d",&x,&y);
        if (!b[x]) N++;
        if (!c[y]) M++;
        b[x] = c[y] = 1;
    }

    for (i = n; i >= N; i --)
        for (j = n; j >= M; j --)
            if (i != n || j != n) {
                a[i][j] = (n * n + a[i + 1][j + 1] * (n - i) * (n - j) + a[i + 1][j] * (n - i) * j + a[i][j + 1] * i * (n - j)) / (n * n - i * j);
            }

    printf("%lf\n", a[N][M]);

    return 0;
}
