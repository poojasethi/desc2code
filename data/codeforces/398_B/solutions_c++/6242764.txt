#include<cstdio>
#define eps 1e-7
using namespace std;
int vis[2005][2005];
double f[2005][2005];
int k1[2005],k2[2005],n,m;
int x,y,a,b,i;
double dfs(int a,int b)
{
	double p1,p2,p3,p4;
	if ((!a && !b) || (a<0 || b<0))
		return 0;
	if (vis[a][b]) return f[a][b];
	p1=1.*(n-a)/n*(n-b)/n;
	p2=1.*(n-a)/n*b/n;
	p3=1.*a/n*(n-b)/n;
	p4=1.*a/n*b/n;
	f[a][b]=(p2*dfs(a,b-1)+p3*dfs(a-1,b)+p4*dfs(a-1,b-1)+1)/(1.-p1);
	vis[a][b]=1;
	return f[a][b];
} 
int main()
{
	scanf("%d %d",&n,&m);
	for(i=1;i<=m;i++)
		{
			scanf("%d %d",&x,&y);
			k1[x]=1; k2[y]=1;
		}
	for(i=1;i<=n;i++)
		a+=(!k1[i]),b+=(!k2[i]);
	printf("%.9lf",dfs(a,b));
	return 0;
}
