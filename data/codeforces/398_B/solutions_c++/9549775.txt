#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#define N 3205
#define For(i,l,r) for(int i=l;i<=r;i++)
#define Dor(i,r,l) for(int i=r;i>=l;i--)
using namespace std;
bool Px[N],Py[N];
double f[N][N];
int n,m,nx,ny;
int main(){
    scanf("%d%d",&n,&m);
    For(i,1,m){
        int x,y; scanf("%d%d",&x,&y);
        Px[x]=1;
        Py[y]=1;
    }
    For(i,1,n) nx+=Px[i];
    For(i,1,n) ny+=Py[i];
    
    Dor(i,n,0) Dor(j,n,0) if(i!=n||j!=n){
        f[i][j]=f[i+1][j+1]*(n-i)*(n-j)+f[i+1][j]*(n-i)*j+f[i][j+1]*i*(n-j)+n*n;
        f[i][j]/=n*n-i*j;
    }
    
    printf("%.8lf\n",f[nx][ny]);
}
