// 0x7f
const int N=100100;

int x[3000],y[3000];
double f[3000][3000]={0.0};

int main()
{	
		int n,m,A=0,B=0;
		cin>>n>>m;
		for (int i=0,a,b; i<m; i++) cin>>a>>b, x[a]=y[b]=1;
		for (int i=1; i<=n; i++) A+=x[i],B+=y[i];
		for (int i=n; i>=A; i--)
				for (int j=n; j>=B; j--) {
						if (i==n && j==n) continue;
						f[i][j]=(f[i+1][j+1]*(n-i)*(n-j)+f[i+1][j]*(n-i)*j+f[i][j+1]*i*(n-j)+n*n)/(n*n-i*j);
				}
		cout<<f[A][B]<<endl;
		//system("pause");
		return 0;
}
