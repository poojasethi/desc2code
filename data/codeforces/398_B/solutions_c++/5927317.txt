#include <iostream>
#include <cstdio>
using namespace std;

int row[2005], col[2005];
double dp[2005][2005];

int main()
{
	int N, M;
	cin >> N >> M;

	for (int i = 0; i < M; i++)
	{
		int r, c;
		cin >> r >> c;
		row[r] = 1;
		col[c] = 1;
	}

	int pr = N, pc = N;
	for (int i = 1; i <= N; i++)
	{
		pr -= row[i];
		pc -= col[i];
	}

	for (int i = 1; i <= N; i++)
	{
		dp[i][0] = dp[i-1][0] + (double)N/i;
		dp[0][i] = dp[0][i-1] + (double)N/i;
	}

	for (int i = 1; i <= N; i++)
		for (int j = 1; j <= N; j++)
		{
			dp[i][j] = N*N;
			dp[i][j] += dp[i-1][j-1]*i*j;
			dp[i][j] += dp[i-1][j]*i*(N-j);
			dp[i][j] += dp[i][j-1]*j*(N-i);
			dp[i][j] /= (N*(i+j)-i*j);
		}

	printf("%.10f\n", dp[pr][pc]);

	return 0;
}
