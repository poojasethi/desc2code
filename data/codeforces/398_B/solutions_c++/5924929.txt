#include <iostream>
#include <stdio.h>
#include <string>
#include <string.h>
#include <vector>
#include <algorithm>
using namespace std;
double f[2005][2005];
bool a[2005],b[2005];
bool v[2005][2005];
int n;
double calc(int x,int y)
{
	if (x<0 || y<0 || x>n || y>n) return 0;
	if (v[x][y]) return f[x][y];
	v[x][y]=1;
	return f[x][y]=((1.0*(n-x)/n*y/n)*calc(x+1,y)+(1.0*x/n*(n-y)/n)*calc(x,y+1)+(1.0*(n-x)/n*(n-y)/n)*calc(x+1,y+1)+1.0)/(1.0-1.0*x/n*y/n);
}
int main()
{
	int i,j,k,l,p=0,q=0,m;
	cin>>n>>m;
	memset(f,0,sizeof(f));
	memset(a,0,sizeof(a));
	memset(b,0,sizeof(b));
	for (i=0;i<m;i++)
	{
		cin>>j>>k;
		if (!a[j]) p++;
		if (!b[k]) q++;
		a[j]=b[k]=1;
	}
	memset(v,0,sizeof(v));
	v[n][n]=1;
	printf("%.6lf\n",calc(p,q));
}
