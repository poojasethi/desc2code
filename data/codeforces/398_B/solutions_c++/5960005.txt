#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;

int N,M,A,B;
bool X[2005],Y[2005];
double f[2005][2005];

void init()
{
	scanf("%d%d",&N,&M);
	for (int i=1,x,y; i<=M; i++) scanf("%d%d",&x,&y),X[x]=Y[y]=1;
	for (int i=1; i<=N; i++) A+=!X[i],B+=!Y[i];
}

void doit()
{
	for (int i=0; i<=A; i++)
		for (int j=0; j<=B; j++) if (i||j) f[i+1][j+1]=(f[i][j]*i*j+f[i+1][j]*(N-i)*j+f[i][j+1]*(N-j)*i+N*N)/(i*j+(N-i)*j+(N-j)*i);
	printf("%.9lf\n",f[A+1][B+1]);
}

int main()
{
	init();
	doit();
	return 0;
}
