#include<cstdio>
#include<cstring>

const int MAX_N = 2000;

long double f[MAX_N + 10][MAX_N + 10];
int n, m;
bool col[MAX_N + 10], row[MAX_N + 10];

int main(){
	scanf("%d%d", &n, &m);
	for(int i = 0; i < m; i ++){
		int x, y;
		scanf("%d%d", &x, &y);
		row[x] = col[y] = true;
	}
	int rc = 0, cc = 0;
	for(int i = 1; i <= n; i ++){
		rc += row[i];
		cc += col[i];
	}
	for(int i = n; i >= rc; i --){
		for(int j = n; j >= cc; j --){
			if(i == n && j == n)
				continue;
			f[i][j] = ((f[i + 1][j] + 1.0) * (n - i) * j + (f[i][j + 1] + 1) * i * (n - j) + (f[i + 1][j + 1] + 1) * (n - i) * (n - j) + i * j) / n / n / (1.0 - (long double)i * j / n / n);
		}
	}
	printf("%.10f\n", (double)f[rc][cc]);

	return 0;
}
