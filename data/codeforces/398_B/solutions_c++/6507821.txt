#include<iostream>
#include<stdio.h>
using namespace std;
double wall[2001][2001];
int main()
{
	int n,m,i,j;
	cin>>n>>m;
	int row[n+1],col[n+1],r,c,x=0,y=0;
	for(i=0;i<=n;i++)
		row[i]=col[i]=0;
	for(i=0;i<m;i++)
	{
		cin>>r>>c;
		if(row[r]==0)
			x+=row[r]=1;
		if(col[c]==0)
			y+=col[c]=1;
	}
	wall[0][0]=0.00;
	for(i=1;i<=n;i++)
		wall[i][0]=wall[0][i]=(double)n/i+wall[i-1][0];
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
	{
		wall[i][j]=((n*n+i*j*wall[i-1][j-1]+i*(n-j)*wall[i-1][j]+(n-i)*j*wall[i][j-1]))/(n*n-((n-i)*(n-j)));
	}
	printf("%.7lf",wall[n-x][n-y]);
}
