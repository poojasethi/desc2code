#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>

using namespace std;
int n,m;
int flag1[2500],flag2[2500],l,r,x,y;
double dp[2500][2500],dq;
int main()
{
	scanf("%d%d",&n,&m);
	memset(flag1,0,sizeof(flag1));
	memset(flag2,0,sizeof(flag2));
	l=r=n;
	for (int i=1;i<=m;++i)
	{
		scanf("%d%d",&x,&y);
		if (flag1[x]==0) l--;
		if (flag2[y]==0) r--;
		flag1[x]++;flag2[y]++; 
	}
	memset(dp,0,sizeof(dp));
	for (int i=0;i<=l;++i)
		for (int j=0;j<=r;++j)
		{
			if (i==j&&i==0) continue;
			dq=n*n;
			if (i!=0&&j!=0) dq+=dp[i-1][j-1]*i*j;
			if (i!=0) dq+=dp[i-1][j]*(i)*(n-j);
			if (j!=0) dq+=dp[i][j-1]*(j)*(n-i);
			dq/=(n*i+n*j-i*j);
			dp[i][j]=dq;
		}
	printf("%.10lf",dp[l][r]);
	return 0;
}
