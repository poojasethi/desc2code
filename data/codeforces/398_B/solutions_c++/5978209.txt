#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
#define N 2100
using namespace std;
double f[N][N];
bool bo[N][N];
int A[N],B[N],x,y,n,m,i;
double work(int i,int j){
	if(i>n || j>n)return 0;
	if(i==n && j==n)return 0;
	if(bo[i][j])return f[i][j];
	bo[i][j]=1;
	f[i][j]=((work(i+1,j)*(n-i)*j+work(i,j+1)*i*(n-j)+work(i+1,j+1)*(n-i)*(n-j))/n/n +1)/(1-1.0*i*j/n/n);
	return f[i][j];
}
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;++i){
		scanf("%d%d",&x,&y);
		A[x]=1;
		B[y]=1;
	}
	x=y=0;
	for(i=1;i<=n;++i)if(A[i])x++;
	for(i=1;i<=n;++i)if(B[i])y++;
	printf("%lf\n",work(x,y));
}
