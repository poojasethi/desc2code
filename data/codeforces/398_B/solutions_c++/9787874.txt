#include<cstdio>
#include<cstring>
#include<queue>
#include<vector>
#include<algorithm>
#define rep(i,s,t) for(int i=s;i<=t;++i)
#define drep(i,t,s) for(int i=t;i>=s;--i)
#define inf 0x3f3f3f3f
using namespace std;
int n,m;
double dp[2001][2001];
bool mx[2001],my[2001];
int main(){
  scanf("%d%d",&n,&m);
  int n1=n,n2=n,x,y;
  rep(i,1,m){
    scanf("%d%d",&x,&y);
    if (!mx[x]) mx[x]=1,--n1;
    if (!my[y]) my[y]=1,--n2;
  }
  dp[0][0]=0;
  rep(i,1,n){
      dp[i][0]=dp[i-1][0]+(double)n/i;
      dp[0][i]=dp[0][i-1]+(double)n/i;
  }
  rep(i,1,n1)
    rep(j,1,n2){
      double s=n*n+dp[i-1][j]*i*(n-j)+dp[i][j-1]*(n-i)*j+dp[i-1][j-1]*i*j;
      dp[i][j]=s/(double)(n*(j+i)-i*j);
  }
  printf("%.10lf\n",dp[n1][n2]);
  return 0;
}
