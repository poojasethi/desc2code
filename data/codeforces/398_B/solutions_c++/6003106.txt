#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
double f[2001][2001];
int vx[2001]={0},vy[2001]={0},n,m,r,c;
int main(){
	scanf("%d%d",&n,&m);
	r=n,c=n;
	int x,y;
	for(int i=1;i<=m;i++){
		scanf("%d%d",&x,&y);
		if(!vx[x])	vx[x]=1,r--;
		if(!vy[y])	vy[y]=1,c--;
	}
	f[0][0]=0;
	for(int i=1;i<=n;i++)	f[i][0]=f[i-1][0]+1.0*n/(1.0*i);
	for(int i=1;i<=n;i++)	f[0][i]=f[0][i-1]+1.0*n/(1.0*i);

	for(int i=1;i<=r;i++)
		for(int j=1;j<=c;j++){
			f[i][j]=n*n+1.0*i*j*f[i-1][j-1]+1.0*i*(n-j)*f[i-1][j]+1.0*j*(n-i)*f[i][j-1];
			f[i][j]/=1.0*n*n-1.0*(n-i)*(n-j);
		}
	printf("%.10lf\n",f[r][c]);
	return 0;
}
