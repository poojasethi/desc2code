#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
using namespace std;
double f[2100][2100];
#define dd double
int n,m;
int s1[2100],s2[2100];
int a1,a2;
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;i++){
      int a,b;
      scanf("%d%d",&a,&b);  
      s1[a]++;
      s2[b]++;
    }
    for(int i=1;i<=n;i++){
      if(s1[i])a1++;
      if(s2[i])a2++;    
    }
    for(int i=n;i>=0;i--)for(int j=n;j>=0;j--){
      if(i==n&&j==n)continue;   
        f[i][j]=f[i+1][j]*dd((n-i)*j)/dd(n*n)+f[i][j+1]*dd(i*(n-j))/dd(n*n)+f[i+1][j+1]*dd((n-j)*(n-i))/dd(n*n)+1;
        f[i][j]=f[i][j]*dd(n*n)/dd(n*n-i*j);
    }
    printf("%.6f",f[a1][a2]);
}
