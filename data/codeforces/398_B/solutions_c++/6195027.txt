#include <stdio.h>
#include <memory.h>
#include <algorithm>
using namespace std;
#define M 2005
bool r[M], c[M];
double p[M][M];
int n;
int main(){
	int i, j, m;
	scanf("%d %d", &n, &m);
	int x=n, y=n;
	while(m--){
		scanf("%d %d", &i, &j);
		if(!r[i])r[i]=1, x--;
		if(!c[j])c[j]=1, y--;
	}
	double a, b;
	for(i=0; i<=x; i++)for(j=0; j<=y; j++)if(i||j){
		b=1.*n*(i+j)-i*j;
		a = 1.*n*n/b;
		if(i)a+=(i*(n-j)/b)*p[i-1][j];
		if(j)a+=((n-i)*j/b)*p[i][j-1];
		if(i&&j)a+=(i*j/b)*p[i-1][j-1];
		p[i][j] = a;
	}
	printf("%.10lf\n", p[x][y]);
	return 0;
}
