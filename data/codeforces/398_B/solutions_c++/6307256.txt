#include<cstdio>
#include<cstdlib>
#define MAXN 2005
using namespace std;

int N, M, A, B ;
double dp[MAXN][MAXN] ;
bool row[MAXN], col[MAXN] ;

int main()
{
	int i, j, x, y ;
	scanf("%d %d", &N, &M) ;
	for(i = 1; i <= M; i ++)
		scanf("%d %d", &x, &y), row[x] = 1, col[y] = 1 ;
	for(i = 1; i <= N; i ++) A += row[i] ;
	for(i = 1; i <= N; i ++) B += col[i] ;
	for(i = N; i >= A; i --)
		for(j = N-(i==N); j >= B; j --)
			dp[i][j] = (N*N+dp[i+1][j+1]*(N-i)*(N-j)+dp[i][j+1]*i*(N-j)+dp[i+1][j]*j*(N-i))/(N*N-i*j) ;
	printf("%.10lf\n", dp[A][B]) ;
	//system("pause") ;
	return 0 ;
}
