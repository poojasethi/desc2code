#include <cstdio>
double f[2005][2005];
bool a[2005], b[2005];
int n, m, A, B, x, y;
int main()
{
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= m; ++i) scanf("%d%d", &x, &y), a[x] = b[y] = 1;
    for (int i = 1; i <= n; ++i) A += a[i], B += b[i];
    for (int i = n; i >= A; --i)
	for (int j = n; j >= B; --j) 
	    if (i < n || j < n)
		f[i][j] = ((n - i) * j * f[i + 1][j] +
			   i * (n - j) * f[i][j + 1] + 
			   (n - i) * (n - j) * f[i + 1][j + 1] +
			   n * n) / (n * n - i * j);
    printf("%.9lf\n", f[A][B]); return 0;
}
