#include <cstdio>
#include <iostream>
#define N 2002
using namespace std;
int n,m,R,C,t1[N],t2[N];
double f[N][N];
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1,r,c;i<=m;++i){
		scanf("%d%d",&r,&c);
		if(!t1[r])t1[r]=1,++R;
		if(!t2[c])t2[c]=1,++C;
    }
	for(int i=0;i<=n;++i)
	    for(int j=0;j<=n;++j)if(i+j){
	    	f[i][j]=n*n;
	    	if(i && j)f[i][j]+=f[i-1][j-1]*i*j;
	    	if(i)f[i][j]+=f[i-1][j]*i*(n-j);
	    	if(j)f[i][j]+=f[i][j-1]*(n-i)*j;
	    	f[i][j]/=(i+j)*n-i*j;
	    }
	printf("%.10lf",f[n-R][n-C]);
}
