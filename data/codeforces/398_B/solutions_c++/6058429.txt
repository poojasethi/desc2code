#include<stdio.h>
#include<string.h>

int a,b,c,d;
bool rw[2005],cl[2005];
double dp[2005][2005];

int main()
{
    scanf("%d%d",&a,&b);
    int row=a,col=a;
    for (int i=0;i<b;i++)
    {
        scanf("%d%d",&c,&d);
        if (!rw[c])rw[c]=1,row--;
        if (!cl[d])cl[d]=1,col--;
    }
    dp[0][0]=0.0;
    for (int i=1;i<=row;i++)dp[i][0]=dp[i-1][0]+(double)a/i;
    for (int i=1;i<=col;i++)dp[0][i]=dp[0][i-1]+(double)a/i;
    for (int i=1;i<=row;i++)
        for (int j=1;j<=col;j++)
        {
            dp[i][j]=((double)a*a+dp[i-1][j-1]*i*j+dp[i-1][j]*i*(a-j)+dp[i][j-1]*(a-i)*j)/(a*i+a*j-i*j);
        }
    printf("%.12lf\n",dp[row][col]);
    return 0;
}
