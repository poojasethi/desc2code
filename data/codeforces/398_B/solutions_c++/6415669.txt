#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <algorithm>
using namespace std;

int n, m, R, C;
bool r[2005], c[2005];
double f[2005][2005];

int main() {
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= m; ++i) {
		int x, y;
		scanf("%d%d", &x, &y);
		r[x] = true; c[y] = true;
	}
	for (int i = 1; i <= n; ++i) R += r[i];
	for (int i = 1; i <= n; ++i) C += c[i];
	f[n][n] = 0;
	for (int i = n; i >= 0; --i)
		for (int j = n; j >= 0; --j) if (i != n || j != n) {
			double I = i, J = j, N = n;
			f[i][j] = I * (N - J) * f[i][j + 1] + J * (N - I) * f[i + 1][j] + (N - I) * (N - J) * f[i + 1][j + 1];
			f[i][j] = (f[i][j] / N / N + 1.0) / (1.0 - I * J / N / N);
		}
	printf("%.8lf\n", f[R][C]);
	return 0;
}
