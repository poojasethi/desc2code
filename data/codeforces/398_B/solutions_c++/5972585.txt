#include <iostream>

using namespace std;

const int MAXN = 2008;

double dp[MAXN][MAXN]; 
bool a[MAXN], b[MAXN]; 
long long n, m, tota, totb;

int main() {
	cin >> n >> m;
	for (int i=0; i<m; i++) {
		int r,c; cin >> r >> c;
		a[r] = true;
		b[c] = true;
	}
	for (int i=0; i<=n; i++) {
		tota += a[i];
		totb += b[i];
	}
	for (int i=n; i>=tota; i--)
		for (int j=n; j>=totb; j--) {
			if ( i == n && j == n) continue;
			dp[i][j] = ((double)n*n + dp[i][j+1]*i*(n-j) + dp[i+1][j]*j*(n-i) + dp[i+1][j+1]*(n-i)*(n-j) )/(n*n- i*j);
		}
	cout << dp[tota][totb] << '\n';
}
