#include<cstdio>
using namespace std;
const int maxn=2001;
bool posx[maxn]={0};
bool posy[maxn]={0};
double dp[maxn][maxn]={0};
int n,m;
double find(int x,int y){
if(!x&&!y)return 0.0;
if(dp[x][y])return dp[x][y];
double k=0;
if(x&&y)k+=(double)x*(double)y*find(x-1,y-1);
if(x)k+=(double)x*(double)(n-y)*find(x-1,y);
if(y)k+=(double)(n-x)*(double)y*find(x,y-1);
k/=(double)n*(double)n;
k+=1.0;
k/=1.0-(double)(n-x)*(double)(n-y)/(double)n/(double)n;
dp[x][y]=k;
return k;
}
void first(){
int b,c;
scanf("%d%d",&n,&m);
int restx=n,resty=n;
while(m--){
scanf("%d%d",&b,&c);
if(!posx[b]){restx--;posx[b]=1;}
if(!posy[c]){resty--;posy[c]=1;}
}
printf("%.10lf\n",find(restx,resty));
}
int main(){
first();




}
