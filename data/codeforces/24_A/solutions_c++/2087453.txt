#include <stdio.h>

int main() {
	int n,a,b,c[101][101],ans=0,g[101][2]={},p=1,ans2=0;
	scanf("%d",&n);
	for(int i=0;i<n;++i){
		scanf("%d%d",&a,&b);
		scanf("%d",&c[a][b]);
		if(g[a][0]==0)
			g[a][0]=b;
		else
			g[a][1]=b;
		if(g[b][0]==0)
			g[b][0]=a;
		else
			g[b][1]=a;
		c[b][a]=-c[a][b];
	}
	int at=g[1][0];
	for(int i=0;i<n;++i){
		if(c[p][at]>0)
			ans+=c[p][at];
		else
			ans2+=c[at][p];
		if(g[at][0]!=p){
			p=at;
			at=g[at][0];
		}
		else{
			p=at;
			at=g[at][1];
		}
	}
	printf("%d",ans<ans2?ans:ans2);
	return 0;
}
