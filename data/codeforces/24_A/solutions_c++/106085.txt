#include <cstdio>
#include <algorithm>

using namespace std;

const int N = 102;

bool adj[N][N];
int cost[N][N];

int main(){
	int n; scanf("%d", &n);
	for (int i = 0; i < n; i++){
		int u, v, c; scanf("%d %d %d", &u, &v, &c);
		u--, v--;
		adj[u][v] = adj[v][u] = true;
		cost[v][u] = c;
	}

	int ans1 = 0, ans2 = 0;
	int last = -1;
	int u = 0;
	do{
		for (int v = 0; v < n; v++) if (adj[u][v] && v != last){
			ans1 += cost[u][v];
			ans2 += cost[v][u];
			last = u;
			u = v;
			break;
		}
	} while (u != 0);

	printf("%d\n", min(ans1, ans2));

	return 0;
}
