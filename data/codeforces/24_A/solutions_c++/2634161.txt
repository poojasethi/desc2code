#include <cstdio>
#include <algorithm>
using namespace std;
int w[105][105],e[105],vis[105];
int main(){
    int n,a,b,c,ans=0,sum=0;
    scanf("%d",&n);
    for(int i=0; i<n; ++i){
        scanf("%d%d%d",&a,&b,&c);
        sum+=c;
        w[b][a]=c;
        e[a]^=b;
        e[b]^=a;
    }
    e[b]^=a;
    for(int i=b; e[i]; i=e[i]){
        ans+=w[i][e[i]];
        e[e[i]]^=i;
    }
    printf("%d\n",min(ans,sum-ans));
    return 0;
}
