#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
int n,i,a,b,t1,t2,j,c,co[101][101],u[101],k,x;
vector <int> g[101];
int main()
{
cin>>n;
for (i=1;i<=n;i++)
{
cin>>a>>b>>c;
g[a].push_back(b);
g[b].push_back(a);
co[a][b]=c;
}
k=1;
u[k]=1;
for (i=1;i<=n;i++)
{
    x=1;
    for (j=0;j<g[k].size();j++)
    {
    if (!u[g[k][j]])
        {
        x=g[k][j];
        break;
        }
    }
    t1+=co[k][x];
    t2+=co[x][k];
    k=x;
    u[k]=1;
    
}
cout<<min(t1,t2);



return 0;
}
