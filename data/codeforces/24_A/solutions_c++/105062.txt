#include <cstdio>

int q[101], g[101][101], v, u, w, n, i, l, c1, c2;
bool was[101];

void dfs(int v)
{
	was[v] = 1;
	q[l++] = v;
	for (int u = 1; u <= n; u++)
	if (!was[u] && (g[v][u] || g[u][v])) dfs(u);
}

int main()
{
	scanf("%d", &n);

	for (i = 0; i < n; i++)
	{
		scanf("%d%d%d", &v, &u, &w);
		g[u][v] = w;
	}

	dfs(1);
	q[l] = 1;

	for (i = 0; i < l; i++) c1 += g[q[i]][q[i+1]];
	for (i = l; i > 0; i--) c2 += g[q[i]][q[i-1]];

	printf("%d", c1 < c2 ? c1 : c2);
}
