#include <iostream>
using namespace std;
int main()
{
	int n, i, p, x, ans1=0, ans2=0;
	cin>>n;
	int a[n], b[n], c[n];
	for (i=0; i<n; i++) cin>>a[i]>>b[i]>>c[i];
	p=1;
	do {
		for (i=0; i<n; i++) {
			if (a[i]==p) {
				ans1+=c[i];
				p=b[i];
				a[i]=b[i]=0;
				break;
			}
			else if (b[i]==p) {
				ans2+=c[i];
				p=a[i];
				a[i]=b[i]=0;
				break;
			}
		}
	} while (p!=1);
	cout<<min(ans1, ans2)<<endl;
	return 0;
}
