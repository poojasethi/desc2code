#include <bits/stdc++.h>
using namespace std;

struct bum{
	int n,x,d;
};
vector<bum> v[105];
int n,a,b,c,to,from;
bool ok[105];

void dfs(int xx, int ct){
	ok[xx]=true;
	for (int i=0;i<=1;i++){
		if (!ok[v[xx][i].n]||
		ct==n&&v[xx][i].n==1){
			if (v[xx][i].d==xx) from+=v[xx][i].x;
			else to+=v[xx][i].x;
			dfs(v[xx][i].n,ct+1);
		}
	}
}

int main(){
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> n;
	for (int i=1;i<=n;i++){
		cin >> a >> b >> c;
		v[a].push_back((bum){b,c,b});
		v[b].push_back((bum){a,c,b});
	}
	dfs(1,1);
	cout << min(to,from) << "\n";
}
