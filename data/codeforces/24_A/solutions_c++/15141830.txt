#include <stdio.h>
#include <vector>
using namespace std;
vector<pair<int,int> > g[128];
int DFS(int src,int parent){
	if(src==1 && parent!=-1)
		return 0;
	int cost=1<<30;
	for(int i=0;i<g[src].size();++i)
		if(g[src][i].first!=parent)
			cost=min(cost,g[src][i].second+DFS(g[src][i].first,src));
	return cost;
}
int main()
{
	int n,a,b,c;
	scanf("%d",&n);
	for(int i=0;i<n;++i){
		scanf("%d%d%d",&a,&b,&c);
		g[a].push_back(make_pair(b,0));
		g[b].push_back(make_pair(a,c));
	}
	printf("%d",DFS(1,-1));
	return 0;
}
