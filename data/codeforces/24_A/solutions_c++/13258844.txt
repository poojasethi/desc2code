#include<bits/stdc++.h>
using namespace std;
int n, a, b, c, ans, tot;
bool U[105],V[105];
int main(){
	scanf("%d",&n);
	for (int i = 0; i < n; i++){
		scanf("%d %d %d",&a,&b,&c);
		if (V[a] || U[b]) ans += c, U[a] = V[b] = true;
		else V[a] = U[b] = true;
		tot += c;
	}
	if (ans < tot - ans) printf("%d\n", ans);
	else printf("%d\n", tot - ans);
}
