#include <stdio.h>

int e[105][3], u[105];

int main(void) {
	int n, i, j;
	scanf("%d", &n);
	for(i=0;i<n;i++){
		scanf("%d%d%d", &e[i][0], &e[i][1], &e[i][2]);
	}
	int start = 1, total = 0, sum = 0;
	for(i=1;i<=n;i++){
		for(j=0;j<n;j++)
			if(!u[j] && (e[j][0]==start||e[j][1]==start)) {
				u[j] = 1;
				if(e[j][0]==start) {total += e[j][2], start = e[j][1];}
				else {total -= e[j][2]; start = e[j][0];}
			}
		sum += e[i-1][2];
	}
	if(total > 0) total = - total;
	printf("%d\n", (sum + total)/2);
	return 0;
}
