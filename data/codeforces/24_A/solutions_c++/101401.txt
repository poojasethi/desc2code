#include <cstdio>
#include <vector>
#include <algorithm>
#define MAXN 101
using namespace std;
vector<int> v[MAXN];
int C[MAXN][MAXN];
int w[2];
void policz(int u, int p, int k) {
	w[k]+=C[u][p];
	if(p!=1)
		for(int i=0;i<2;i++)
			if(v[p][i]!=u)
				policz(p,v[p][i],k);
}
int main() {
	int n,a,b,c;
	scanf("%d",&n);
	for(int i=1;i<=n;i++) {
		scanf("%d%d%d",&a,&b,&c);
		v[a].push_back(b);
		v[b].push_back(a);
		C[b][a]=c;
	}
	policz(1,v[1][0],0);
	policz(1,v[1][1],1);
	printf("%d\n",min(w[0],w[1]));
	return 0;
}
