#include <iostream>
#include <set>
using namespace std;

int main() {
	int n;
	cin>>n;
	int a, b, c;
	set<int> fro, to;
	int tot=0, cost=0;
	for (int i=0; i<n; i++) {
		cin>>a>>b>>c;
		tot+=c;
		if (fro.find(a)!=fro.end() || to.find(b)!=to.end()) {
			cost+=c;		
			fro.insert(b);
			to.insert(a);
		}
		else {
			fro.insert(a);
			to.insert(b);
		}
	}
	cout<<min(tot-cost, cost)<<'\n';
	return 0;
}
