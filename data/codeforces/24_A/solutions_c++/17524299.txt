#include <algorithm>
#include <iostream>
using namespace std;

int main()
{
	int n,sum=0,p=1,an=0;
    int a[100],b[100],c[100];

	cin>>n;
	for (int i=0;i<n;i++)	{
		cin>>a[i]>>b[i]>>c[i];
		sum+=c[i];
	}
	for (int i=1;i<=n;i++)
	 for (int j=0;j<n;j++){
	  if (a[j]==p || b[j]==p){
			if (b[j]==p)
                        an+=c[j];
		        if (a[j]==p) 
                        p=b[j];
		        else p=a[j];
		       a[j]=0;
                       b[j]=0;
				break;
			}
        }
	cout<<min(an,sum-an)<<endl;
	return 0;
}
