#include <iostream>
#include <vector>
using namespace std;

typedef pair<int, int> pii;
#define F first
#define S second
const int MAXN = 100 + 10;
vector<pii> nxt[MAXN];

int main()
{
	int n, s = 0;
	cin >> n;
	for(int i=0; i<n; i++)
	{
		int f, t, c;
		cin >> f >> t >> c;
		s += c;
		nxt[f].push_back(pii(t, 0));
		nxt[t].push_back(pii(f, c));
	}
	int pr = 0, hr = 1, cs = 0;
	for(int i=0; i<n; i++)
	{
		if(nxt[hr][0].F != pr)
		{
			cs += nxt[hr][0].S;
			pr = hr;
			hr = nxt[hr][0].F;
		}
		else
		{
			cs += nxt[hr][1].S;
			pr = hr;
			hr = nxt[hr][1].F;
		}
	}
	cout << min(cs, s - cs) << endl;
	return 0;
}
