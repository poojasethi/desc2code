#include<iostream>
#include<cstdio>
#include<algorithm>
#include<string>
#include<cstring>
#include<vector>
#include<cmath>
using namespace std;
int n;
int W[100][100];
int start = 0;
int x = 0, y = 0;

void dfs(int u, int par = -1){	
	if(u == 0 && par != -1) return;
	for(int i = 0; i < n; i++) if(W[u][i] && i != par){
		if(W[u][i] < 0) y -= W[u][i];
		else x += W[u][i];
		dfs(i, u);
		break;
	}
}
int main(){
	cin >> n;
	memset(W, 0, sizeof W);
	for(int i = 0; i < n; i++){
		int a, b, c; cin >> a >> b >> c;
		a--; b--;
		W[a][b] = c;
		W[b][a] = -c;
	}
	dfs(0);
	cout << min(x, y) << endl;
}
