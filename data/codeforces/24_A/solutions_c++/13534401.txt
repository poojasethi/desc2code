#include<bits/stdc++.h>

using namespace std;

int A[102][102],n,i,u,v,w,t,mn = 100005;

bool H[102][102],Q[102];

void f(int x){

    int i;
	bool h=1;

    Q[x] = 1;

	for(i=1;i<=n;i++)
		if(!Q[i] && A[x][i]){h=0;
			t += A[x][i] * H[x][i];
			f(i);
			t -= A[x][i] * H[x][i];
		}

	Q[x] = 0;

	if(h)
		mn = min(mn , t + A[x][1] * H[x][1]);

}
int main(){

	cin >> n;

	for(i=1;i<=n;i++){
	
		cin >> u >> v >> w;

		A[u][v] = w;
		A[v][u] = w; H[v][u]= 1;
	}
  
    f(1);

	cout << mn;
	return 0;
}
