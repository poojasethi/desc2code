#include<cstdio>
#include<math.h>
#include<algorithm>
using namespace std;
int main(){
	int n;
	while (scanf("%d", &n) == 1){
		bool in[100] = { false };
		bool out[100] = { false };
		int ans = 0, total = 0;
		for (int i = 0; i < n; i++){
			int a, b, c;
			scanf("%d%d%d", &a, &b, &c);
			a--, b--;
			if (out[a] || in[b]){
				ans += c;
				in[a] = out[b] = true;
			}
			else
				out[a] = in[b] = true;
			total += c;
		}
		if (ans < total - ans)
			printf("%d\n", ans);
		else
			printf("%d\n", total - ans);
	}
	return 0;
}
