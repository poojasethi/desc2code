#include <iostream>
#include <stdio.h>
#include <algorithm>

using namespace std;

int n,g[111][111],t,res,was[111];

void dfs(int v,int p)
{
	
	for ( int i = 1; i <= n; i++ ) {
		if (  !was[i] && i!=p && g[v][i]) { was[i] = 1; res+=(g[v][i]-1); dfs(i,v); }
    }
}

int main()
{
	int u,v,c;
	scanf("%d",&n);
	for ( int i = 1; i <= n; i++ ) {
		scanf("%d%d%d",&u,&v,&c);
		g[u][v] = 1;
		g[v][u] = c + 1;
		t+=c;
	}
	dfs(1,0);
	printf("%d",min(res,t-res));
}
