#include <iostream> 
#include <stdio.h>
#include <string.h>
using namespace std;
int main()
{
	int a[110],b[110],p[110];
	int i,j,sum,add,n,f,h;
	while(cin>>n)
	{
		for(i=0;i<n;i++)
		{	cin>>a[i]>>b[i]>>p[i];	}
		f=b[0];sum=p[0];add=0;h=0;
		for(i=0;i<n-1;i++)
		{
			for(j=0;j<n;j++)
			{
				if(a[j]==f&&j!=h)
				{
					f=b[j];sum+=p[j];h=j;break;
				}
				else if(b[j]==f&&j!=h)
				{
					f=a[j];add+=p[j];h=j;break;
				}
			}
		}
		printf("%d\n",add>=sum?sum:add);
	}
}
