#include<iostream>

using namespace std;
const int maxn = 110;
int w[maxn][maxn];
int d[maxn][maxn];
int c[maxn];
int cind=1;
int check[maxn];

int n;
void dfs(int x)
{
	check[x] = 1;
	c[cind++] = x;
	for(int i=1;i<=n;i++)
		if(!check[i] &&  d[x][i])
			dfs(i);
}


int main()
{
	cin >> n;
	for(int i=0;i<n;i++)
	{
		int a,b,wei;
		cin >> a >> b >> wei;
		d[a][b]= 1;
		d[b][a] = -1;
		w[a][b] = 0;
		w[b][a] = wei;
	}

	dfs(1);
	int x=0,y=0;
	for(int i=1;i<=n;i++)
		x+= w[c[i]][(i!=n)?c[i+1]:c[1]];
	for(int i=n;i>0;i--)
		y+= w[c[i]][(i!=1)?c[i-1]:c[n]];
	cout << min(x,y) << endl;
	return 0;
}
