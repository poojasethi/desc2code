#include <iostream>
#include <algorithm>
using namespace std;

int v[101][101] = {0};

int main()
{
	int n,a,b,r;
	cin >> n;
	int m = -1;
	int x, y, tmp = 0;
	for(int i = 0; i < n ; i++) {
		cin >> a >> b >> r;
		tmp += r;
		v[a][b] = r;
		if(m < r) {m = r; x = a; y = b; }
	}


	int sum = 0;
	v[x][y] = 0;
	while(y != x) {
		for(int j = 1; j <= n; j++) {
			if(v[j][y] != 0) {
				sum += v[j][y];
				v[j][y] = 0;
				y = j;
				break;
			}
			else if(v[y][j] != 0) {
				v[y][j] = 0;
				y = j;
				break;
			}
		}
	}

	cout << min(sum, (tmp- sum)) << endl;
	return 0;
}
