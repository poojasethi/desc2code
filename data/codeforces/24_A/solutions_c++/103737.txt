#include <cstdio>
#include <algorithm>
using namespace std;

int n;
int a[105][105] = {0};
bool v[105] = {0};
int d[105]; 
int t = 0;

void dfs(int x, int k) {
	v[x] = 1;
	d[k] = x;
	for (int i = 0; i < n; ++i) {
		if (a[x][i] != 0) {
			if (!v[i]) dfs(i, k + 1); 
		}
	}
}
int main() {
	scanf("%d", &n);
	int cnt = 0;
	for (int i = 0; i < n; ++i) {
		int x, y, w; scanf("%d%d%d", &x, &y, &w); --x; --y; a[x][y] = w; a[y][x] = -w;  cnt += w;
	}
	dfs(0, 0); d[n] = d[0];
	for (int i = 0; i < n; ++i) {
		int tmp =  a[d[i]][d[i+1]];
		if (tmp > 0) t += tmp;
	}
	printf("%d\n", min(t, cnt - t));
	return 0;
}
