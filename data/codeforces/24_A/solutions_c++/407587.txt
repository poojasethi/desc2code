#include<stdlib.h>
#include<stdio.h>
#include<iostream>
#include<memory.h>
#include<algorithm>
using namespace std;
int ms[200][200],n,i,j,q,_min,x,y,z;
bool f[1000];
void dfs(int x,int s,int s1){
	int i;
	f[x]=true;
	if (s1==n) _min=min(_min,s+ms[x][1]);
	for (i=1;i<=n;i++) if (ms[x][i]<100000&&f[i]==false) dfs(i,s+ms[x][i],s1+1);
	f[x]=false;
}

int main(){
	scanf("%d",&n);
	memset(ms,63,sizeof(ms));
	_min=100000000;
	for (i=1;i<=n;i++) ms[i][i]=0;
	for (i=0;i<n;i++){
		scanf("%d%d%d",&x,&y,&z); 
		ms[x][y]=0;
		ms[y][x]=z;
	}
	dfs(1,0,1);
	printf("%d\n",_min);
	return 0;
}
