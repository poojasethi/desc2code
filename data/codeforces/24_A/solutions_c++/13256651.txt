#include<bits/stdc++.h>
using namespace std;

int main(){
	int n,total=0,htotal=0,amasuk,akeluar,jawab;
	vector<int>masuk,keluar;
	cin>>n;
	while(n--){
		int dari,ke,harga;
		cin>>dari>>ke>>harga;
		total+=harga;
		amasuk=count(masuk.begin(),masuk.end(),dari);
		akeluar=count(keluar.begin(),keluar.end(),ke);
		if(amasuk>0||akeluar>0){
			htotal+=harga;
			masuk.push_back(ke);
			keluar.push_back(dari);
		}
		else if(amasuk==0&&akeluar==0){
			masuk.push_back(dari);
			keluar.push_back(ke);			
		}
	}
	cout<<min(total-htotal,htotal)<<endl;
}
