#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int main(){
	int a[105],b[105],v[105];
	int i,j,suml,sumr,n,flag,id;
	scanf("%d",&n);
	for(i = 0; i < n;i++)
	scanf("%d%d%d",&a[i],&b[i],&v[i]);
	flag = b[0];
	sumr = v[0];
	suml = 0;
	id = 0;
	for(i = 0; i < n - 1;i++){
		for(j = 0; j < n;j++){
			if(a[j] == flag && id != j){
				flag = b[j];
				sumr += v[j];
				id = j;
				break;
			}
			else if(b[j] == flag && j != id){
				flag = a[j];
				suml += v[j];
				id = j;
				break;
			}
		}
	}
	printf("%d\n",suml < sumr? suml : sumr);
	return 0;
}
