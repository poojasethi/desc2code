#include <algorithm>
#include <iostream>
using namespace std;

int main(void)
{
	int n, a[100], b[100], c[100], sum=0, pos=1, answer=0;

	cin >> n;
	for (int i=0; i<n; ++i)	{
		cin >> a[i] >> b[i] >> c[i];
		sum+=c[i];
	}
	for (int i=1; i<=n; ++i)
		for (int j=0; j<n; ++j)
			if (a[j]==pos || b[j]==pos)	{
				if (b[j] == pos)	answer+=c[j];
				if (a[j] == pos)	pos=b[j];
				else	pos=a[j];
				a[j]=b[j]=0;
				break;
			}
	cout << min(answer,sum-answer) << endl;
}
