#include<bits/stdc++.h>
using namespace std;
int visit[101];
int a[101][101];
int main()
{
	int n,i,x,y,z,s=0,j;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>x>>y>>z;
		a[x][y]=z;
		s+=z;
	}
	x=1;
	y=0;
	for(i=1;i<=n;i++)
	{
		visit[x]=1;
		for(j=1;j<=n;j++)
		{
			if(a[x][j]!=0&&!visit[j])
			{
				y+=a[x][j];
				x=j;
				break;
			}
			if(a[j][x]!=0&&!visit[j])
			{
				x=j;
				break;
			}	
		}
		if(i==n-1)
			visit[1]=0;
		//cout<<x<<" "<<y<<endl;
	}
	cout<<min(y,s-y)<<endl;
	return 0;
}
