#include<cstdio>
#include<cstring>
const int maxn=110;
int map[maxn][maxn];
int e[maxn];

int Min(int a,int b){
    return a<b?a:b;
}

int main(){
    int u,v,w,n;
    int i;
    scanf("%d",&n);
    int sum=0;
    for(i=0;i<n;i++){
        scanf("%d%d%d",&u,&v,&w);
        map[u][v]=0; map[v][u]=w;
        e[u]^=v; e[v]^=u;
        sum+=w;
    }
    e[v]^=u;
    int ans=0;
    for(i=v;e[i];i=e[i]){
        ans+=map[i][e[i]];
        e[e[i]]^=i;
    }
    printf("%d\n",Min(ans,sum-ans));
    return 0;
}
