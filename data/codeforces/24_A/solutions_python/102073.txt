#!/usr/bin/python
import sys
MaxC = 0x3F3F3F3F
NA = -1
n = int (sys.stdin.readline ())
a = [[MaxC for i in range (n)] for i in range (n)]
for i in range (n):
        u, v, w = [int (x) for x in sys.stdin.readline ().split ()]
        u -= 1
        v -= 1
	a[u][v] = 0
	a[v][u] = w
s, t = 0, 0
v, p = 0, NA
for i in range (n):
	for w in range (n):
		if a[v][w] < MaxC and w != p:
			s += a[v][w]
			t += a[w][v]
			v, p = w, v
			break
print min (s, t)
