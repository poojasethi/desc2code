n=input()
d={}
s=0
for i in xrange(n):
    a,b,c=map(int,raw_input().split())
    a-=1; b-=1
    d[(a,b)]=c
    d[(b,a)]=0
    s+=c
cur=0
ans=0
vis=[False]*n
vis[0]=True
for ct in xrange(n):
    if ct==n-1: vis[0]=False
    for j in xrange(n):
        if not vis[j] and (cur,j) in d:
            ans+=d[(cur,j)]
            cur=j
            vis[cur]=True
            break
print min(ans,s-ans)
