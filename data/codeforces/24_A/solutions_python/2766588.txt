n = input()
e = [[] for _ in range(n + 1)]
d = {}
f = lambda p : sum(d[(p[i], p[i + 1])] for i in range(n))
for _ in range(n):
	a, b, c = map(int, raw_input().split())
	e[a].append(b)
	e[b].append(a)
	d[(a, b)] = 0
	d[(b, a)] = c
p = [1, e[1][0]]
for _ in range(2, n):
	for v in e[p[-1]]:
		if v not in p:
			p += [v]
p += [1]
print min(f(p), f(p[ : : -1]))
