def find(x, y):
    global t
    t -= 1
    if t < 0:
        return 0
    if a[x][1][0] == y:
        return a[x][0][1] + find(a[x][0][0], x)
    else:
        return a[x][1][1] + find(a[x][1][0], x)

n = int(raw_input())
a = [[] for _ in xrange(n)]
for i in xrange(n):
    x, y, z = map(int, raw_input().split())
    x -= 1
    y -= 1
    a[x].append([y, 0])
    a[y].append([x, z])
t = n
res = find(0, a[0][1][0])
t = n
res = min(res, find(0, a[0][0][0]))
print res
