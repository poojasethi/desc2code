n=input()
r=[ [] for i in range(n)]

for i in range(n):
    a,b,c=map(int,raw_input().split())
    a-=1
    b-=1
    r[a].append((b,0,c))
    r[b].append((a,c,0))
R=[]

i=0
p=-1
l=[]
while True:

    for v in r[i]:
        if v[0]!=p:
            l.append(v)
            p = i
            i = v[0]
            break
    
    if not i:break
print min(map(sum,zip(*l))[1:])
