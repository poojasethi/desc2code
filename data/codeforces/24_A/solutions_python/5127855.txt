n = int(raw_input())
p = [tuple(map(int, raw_input().split())) for i in range(n)]
t, s = [[0, 0] for i in range(n + 1)], 0
for a, b, c in p:
    t[a][t[a][0] != 0], t[b][t[b][0] != 0] = b, a
    s += c
t[1], x = t[1][0], 1
for i in range(1, n):
    y = t[x]
    t[y], x = t[y][t[y][0] == x], y
t[x] = 1
d = sum(c for a, b, c in p if t[a] == b)
print min(d, s - d)
