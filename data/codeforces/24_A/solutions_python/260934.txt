from copy import * 
n = int(raw_input())
r = range(0,n)
l = [[] for i in r]; next = deepcopy(l); w = deepcopy(l)
sm = 0
for i in r:
    a,b,c = map(int,raw_input().split())
    a-=1; b-=1
    next[a] += [b]; w[a] += [c]
    next[b] += [a]; w[b] += [-c]
    sm += c
    
j = 0
sm2 = 0
pred = -1

for i in r:
    k = 0 if next[j][0] != pred else 1
    if (w[j][k] > 0):
        sm2 += w[j][k]
    pred,j = j,next[j][k]

print min(sm2, sm-sm2)
