n = input()
s = []
for i in range(n):
	a,b,c = map(int, raw_input().split())
	s.append((a,b,c))
st = 1
a = b = 0
for x in range(n):
	for i in range(n):
		if s[i][0] == st:
			st = s[i][1]
			a += s[i][2]
			s[i] = (0,0)
			break
		if s[i][1] == st:
			st = s[i][0]
			b += s[i][2]
			s[i] = (0,0)
			break
print min(a,b)
