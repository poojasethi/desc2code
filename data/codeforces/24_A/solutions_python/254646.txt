from itertools import *
import sys
n = input ()
g = [list () for _ in range (n + 1)]

def add (u, v, c):
    g[u].append ([v, 0])
    g[v].append ([u, c])
    
pairs = [add (*map (int, raw_input ().split ())) for _ in range (n)]

def dfs (u, p, money=0):
    if u == 1 and p != -1:
        return money
    res = 2**30
    for dest in g[u]:
        if dest[0] != p:
            res = min (res, dfs (dest[0], u, money + dest[1]))
    return res
