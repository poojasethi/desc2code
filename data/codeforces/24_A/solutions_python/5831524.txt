n = input()
way = []
for i in range(n):
    a,b,c = map(int, raw_input().split())
    way.append([a,b,c])

#print way
st = 1
a = b = 0
for j in range(n):
    for i in range(n):
        if way[i][0] == st:
            st = way[i][1]
            a = a + way[i][2]
            way[i] = [0,0]
            break
        if way[i][1] == st:
            st = way[i][0]
            b = b + way[i][2]
            way[i] = [0,0]
            break

print min(a,b)
