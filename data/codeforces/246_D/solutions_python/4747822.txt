def main():
    r=lambda:map(int,raw_input().split())
    n,m=r()
    c=r()
    C={}
    for e in c:
        C[e]=[]
    for i in range(m):
        x,y=r()
        if c[x-1] == c[y-1]: continue
        C[c[x-1]].append(c[y-1])
        C[c[y-1]].append(c[x-1])
    print max(C.items(),key=lambda x:(len(set(x[1])),-x[0]))[0]
main()
