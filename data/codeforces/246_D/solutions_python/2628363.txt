q = [set() for i in xrange(100010)]
n, m = map(int, raw_input().split())
c = map(int, raw_input().split())
for i in xrange(m) :
    a, b = map(int, raw_input().split())
    a -= 1
    b -= 1
    if c[a] != c[b] :
        q[c[a]].add(c[b])
        q[c[b]].add(c[a])
ans = 0
cnt = 0
for i in xrange(1, 100001) :
    if len(q[i]) > cnt :
        cnt = len(q[i])
        ans = i
if ans == 0 :
    c.sort()
    ans = c[0]
print(ans)
