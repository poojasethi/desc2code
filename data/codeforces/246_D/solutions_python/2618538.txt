from sys import stdin
def main():
    n, m = map(int, stdin.readline().split())
    c = map(int, stdin.readline().split())
    s = [set() for i in xrange(10 ** 5 + 10)]
    for x in c:
        s[x].add(x)
    for line in stdin:
        a, b = map(int, line.split())
        if c[a-1] != c[b-1]:
            s[c[a-1]].add(c[b-1])
            s[c[b-1]].add(c[a-1])
    x = ans = 0
    for i, t in enumerate(s):
        if len(t) > x:
            x = len(t)
            ans = i
    print ans
main()
