from sets import Set
n,m=map(int,raw_input().split())
col=map(int,raw_input().split())
ans=[]
for i in range(0,100002):
    ans.append(Set([]))
for i in range (0,m):
    u,v=map(int,raw_input().split())
    u=u-1
    v=v-1
    if col[u]!=col[v]:
        ans[col[u]].add(col[v])
        ans[col[v]].add(col[u])
mx=min(col)
for i in range(mx,100001):
    if len(ans[mx])<len(ans[i]):
        mx=i
print mx
