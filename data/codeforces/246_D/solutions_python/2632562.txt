r = lambda: map(int, raw_input().split())
n, m = r()
c = r()
cnt = dict((x, set()) for x in list(set(c)))
e = dict((i, []) for i in xrange(n))
for i in xrange(m):
    a, b = r()
    e[a - 1].append(b - 1)
    e[b - 1].append(a - 1)
for i in xrange(n):
    for item in list(set(e[i])):
        if c[i] != c[item]:
            cnt[c[i]].add(c[item])
print -max((len(v), -k) for k, v in cnt.iteritems())[1]
