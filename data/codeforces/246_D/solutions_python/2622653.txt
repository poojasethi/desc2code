n, m = map(int, raw_input().split())
c = map(int, raw_input().split())
Q = {}
for i in range(m):
    ca, cb = (c[int(n)-1] for n in raw_input().split())
    if ca != cb:
        Q.setdefault(ca, set()).add(cb)
        Q.setdefault(cb, set()).add(ca)
print(max(c, key=lambda x: (len(Q.get(x, [])), -x)))
