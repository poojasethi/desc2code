n, m = map(int, raw_input().split())
c = map(int, raw_input().split())
e, d = [set() for _ in range(n)], {}
for a, b in [map(int, raw_input().split()) for _ in range(m)]:
	a, b = a - 1, b - 1
	if c[a] != c[b]:
		e[a].add(c[b])
		e[b].add(c[a])
for i, v in enumerate(e):
	d[c[i]] = d.get(c[i], set()) | v
v = max(len(s) for s in d.values())
print min(s for s in d.keys() if len(d[s]) == v)
