n,m = map(int,raw_input().split())
c = [int(x)-1 for x in raw_input().split()]
g = [set() for _ in xrange(100000)]
for _ in xrange(m):
    x,y = map(int,raw_input().split())
    x -= 1
    y -= 1
    if c[x] != c[y]:
        g[c[x]].add(c[y])
        g[c[y]].add(c[x])
ans = min(c)
for i in xrange(100000):
    if len(g[i])>len(g[ans]):
        ans = i
print ans+1
