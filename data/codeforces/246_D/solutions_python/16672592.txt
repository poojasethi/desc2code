n, m = map(int, raw_input().split())
c = map(int, raw_input().split())
s = [set() for i in xrange(10 ** 5 + 1)]
for x in c: s[x].add(x)
for i in xrange(m):
	u, v = map(int, raw_input().split())
	u -= 1
	v -= 1
	if c[u] != c[v]:
		s[c[u]].add(c[v])
		s[c[v]].add(c[u])
ans = 0
maxCard = 0
for i, t in enumerate(s):
	if len(t) > maxCard:
		maxCard = len(t)
		ans = i
print ans
