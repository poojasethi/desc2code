n,m = map(int,raw_input().split())
c = map(int,raw_input().split())
cols = {}
for cc in c:
    cols[cc] = {}
for i in xrange(m):
    x,y = map(int,raw_input().split())
    if c[x-1]!=c[y-1]:
        cols[c[x-1]][c[y-1]]=1
        cols[c[y-1]][c[x-1]]=1
mm = -1
mc = -1
for cc in c:
    if len(cols[cc])>mm :
        mm = len(cols[cc])
        mc = cc
    elif len(cols[cc])==mm and mc>cc:
        mc = cc
print mc
