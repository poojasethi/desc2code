#! /usr/bin/env python2

n, m = map(int, raw_input().strip().split(' '))

c = map(int, raw_input().strip().split(' '))

d = {}

for ci in c:
    d.setdefault(ci, set())

for i in xrange(m):
    a, b = map(int, raw_input().strip().split(' '))
    a -= 1
    b -= 1
    d[c[a]].add(c[b]);
    d[c[b]].add(c[a]);

ans = -1
maxi = -1

for k in sorted(d.keys()):
    l = len(d[k])
    if k in d[k]:
        l -= 1
    if l > maxi:
        maxi = l
        ans = k

print ans
