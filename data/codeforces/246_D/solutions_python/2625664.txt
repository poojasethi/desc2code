n, m = map(int, raw_input().split())
c = map(int, raw_input().split())

C = {}
for color in c:
    C[color] = set()
for i in range(m):
    a, b = map(int, raw_input().split())
    ca = c[a-1]
    cb = c[b-1]
    if ca != cb:
        C[ca].add(cb)
        C[cb].add(ca)

max_n = -1
for color in C:
    if len(C[color]) > max_n:
        max_n = len(C[color])
        max_c = color
    elif len(C[color]) == max_n and color < max_c:
        max_c = color
print max_c
