#!/usr/local/bin/python
n, m = map(int, raw_input().split())
color = map(int, raw_input().split())
g = [[] for i in xrange(n)]

for i in xrange(m):
    a, b = map(int, raw_input().split())
    a-=1
    b-=1
    if color[a] != color[b]:
        g[a].append(b)
        g[b].append(a)

map = {}
for i in color:
    map[i] = set()
for i in xrange(n):
    s = map[color[i]]
    for j in g[i]:
        if color[i] != color[j]:
            s.add(color[j])
    map[color[i]] = s

max, ans = 0, -1
for c, p in map.iteritems():
    if max < len(p):
        max = len(p)
        ans = c
    elif max == len(p) and (ans == -1 or ans > c):
        ans = c

print ans
