import sys
import collections

n, m = map(lambda x: int(x), sys.stdin.readline().split())
c = map(lambda x: int(x), sys.stdin.readline().split())
d = collections.defaultdict(set)
for i in xrange(m):
    a, b = map(lambda x: int(x), sys.stdin.readline().split())
    if c[a - 1] != c[b - 1]:
        d[c[a - 1]].add(c[b - 1])
        d[c[b - 1]].add(c[a - 1])
if not d:
    print min(c)
else:
    print max(d.iteritems(), key=lambda q: len(q[1]))[0]
