#include <iostream>
#include <set>
#define N 100005
using namespace std;
int c[N];
set<int> s[N];
int main( void )
{
  int n,m,a,b;
  int mincolor = N;
  cin >> n >> m;
  for( int i=1; i<=n; i++ ){
    cin >> c[i];
    if( mincolor > c[i] )
      mincolor = c[i];
  }
  for( int i=1; i<=m; i++ ){
    cin >> a >> b;
    if( c[a] != c[b] ){
      s[c[a]].insert( c[b] );
      s[c[b]].insert( c[a] );
    }
  }
  int ret = mincolor;
  for( int i=2; i<N; i++ ){
    if( s[ret].size() < s[i].size() )
      ret = i;
  }
  cout << ret << endl;
  return 0;
}
