#include<iostream>
#include<set>
#define maxi 100001
using namespace std;
set<int>g[maxi];
int c[maxi];
int main(){
	int v,e,i,max=-1,min=100001;
	cin>>v>>e;
	for(i=0;i<v;i++){
		cin>>c[i];
		min = (--c[i]<min)?c[i]:min;
		}
	for(i=0;i<e;i++){
		int n1,n2;
		cin>>n1>>n2;
		n1--;
		n2--;
		if(c[n1]!=c[n2]){
			g[c[n1]].insert(c[n2]);
			g[c[n2]].insert(c[n1]);
		}
	}
	for(i=0;i<maxi;i++)
		if(g[i].size()>g[min].size()){
			min = i;
		}
	cout<<++min<<endl;
	return 0;
}
