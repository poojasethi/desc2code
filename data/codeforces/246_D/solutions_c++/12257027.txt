#include <bits/stdc++.h>

using namespace std;

const long long linf = 1e18 + 9;
const int inf = 1e9 + 9;
const int N = 5e5 + 9;

set < int > gr[N];

int n, m, mn, u, v, c[N];

int main() {

	cin >> n >> m;

	mn = inf;

	for (int i = 1; i <= n; i++) {
		cin >> c[i];
		mn = min(c[i], mn);
	}

	for (int i = 1; i <= m; i++) {
		cin >> u >> v;
		if (c[u] != c[v]) {
			gr[c[u]].insert(c[v]);
			gr[c[v]].insert(c[u]);
		}
	}

	for (int i = 1; i < N; i++)
		if (gr[i].size() > gr[mn].size())
			mn = i;

	cout << mn << endl;

	return 0;
}
