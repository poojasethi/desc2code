#include <bits/stdc++.h>
using namespace std;
set <int> adj[100010];
int c[100010]={} , i1 , i2 , n , m , ans = 1e9;
int main(){
	ios::sync_with_stdio(false); cin.tie();
	cin >> n >> m;
	for (int i = 1; i <= n; i++){
		cin >> c[i];
		ans = min(ans, c[i]);
	}
	for(int i = 1 ; i <= m ; i++){
		int i1 , i2;
		cin >> i1 >> i2;
		if (c[i1] != c[i2]){
			adj[c[i1]].insert(c[i2]);
			adj[c[i2]].insert(c[i1]);
		}
	}
	for(int i = 0 ; i < 100010 ; i++)
		if(adj[ans].size() < adj[i].size())
			ans = i;
	cout << ans;
}
