#include<bits/stdc++.h>
using namespace std;
#define ll long long 
set<int>g[100001];
ll c[100001],v,e,i,x=-1,z=100001,a,b;
int main(){
  cin>>v>>e;
  for(i=0;i<v;i++){
    cin>>c[i];
    if(c[i]<z)
      z=c[i];
  }
  
  for(i=0;i<e;i++){
    cin>>a>>b;
    a--;
    b--;
     if(c[a]!=c[b]){
      g[c[a]].insert(c[b]);
      g[c[b]].insert(c[a]);
    }
  }
  for(i=0;i<100001;i++)
    if(g[i].size()>g[z].size()){
      z=i;
    }
  cout<<z;
  return 0;
}
