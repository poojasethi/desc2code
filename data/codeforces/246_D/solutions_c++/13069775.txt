//In the Name of God

#include<iostream>
#include<set>

using namespace std;

const int maxn=100100;
int n,m,M,ans=1000000;
int c[maxn];
set<int> s[maxn];

int main()
{
  cin>>n>>m;
  for(int i=1;i<=n;i++)
    cin>>c[i];
  for(int v,u;m--;)
    {
      cin>>v>>u;
      if(c[u]!=c[v])
	{
	  s[c[u]].insert(c[v]);
	  s[c[v]].insert(c[u]);
	}
    }
  for(int i=1;i<=n;i++)
    M=max(M,(int)s[c[i]].size());
  for(int i=1;i<=n;i++)
    if(s[c[i]].size()==M && c[i]<ans)
      ans=c[i];
  cout<<ans<<endl;
  return 0;
}
