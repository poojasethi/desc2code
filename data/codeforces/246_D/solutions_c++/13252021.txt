#include<iostream>
#include<vector>
#include<set>
using namespace std;
set<int>c[100005];
int main()
{
	int n,m;
	cin>>n>>m;
	int a[n];
	int q=100006;
	for(int i=0;i<n;i++)
	cin>>a[i],q=min(q,a[i]);
	int x,y;

	for(int i=0;i<m;i++)
	{
		cin>>x>>y;
		x--;
		y--;
		if(a[x]!=a[y])
		{
			c[a[x]].insert(a[y]);
			c[a[y]].insert(a[x]);
		}
	}
	int j=0,k=q;
	for(int i=0;i<=100000;i++)
	{
		if(c[i].size()>j)
		{
			j=c[i].size();
			k=i;
		}
	}
	cout<<k;
}
