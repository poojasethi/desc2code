#include <iostream>
#include <set>

using namespace std;

int c[100010];
set < int > Q[100010];

int main()
{
	int n, m, a, b, mx = 100010;
	cin >> n >> m;
	for( int i = 1 ; i <= n ; i ++ )
		cin >> c[i], mx = min( mx , c[i] );
	for( int i = 0 ; i < m ; i ++ )
	{
		cin >> a >> b;
		if( c[a] != c[b] )
		{
			Q[ c[a] ].insert( c[b] );
			Q[ c[b] ].insert( c[a] );
		}
	}
	for( int i = 1 ; i <= n ; i ++ )
		if( Q[i].size() > Q[mx].size() )
			mx = i;
	cout << mx << endl;
	return 0;
}
