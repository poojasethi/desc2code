#include <bits/stdc++.h>

using namespace std;

#define ll long long int

vector< set<int> > Color(100010);

int main()
{
	int n,m,i,j,k,c;
	cin>>n>>m;
	int C[n+1];
	for(i=1;i<=n;i++)
		cin>>C[i];
	for(i=0;i<m;i++)
	{
		cin>>j>>k;
		c=C[j];
		if(C[j]!=C[k])
		{
			Color[C[j]].insert(C[k]);
			Color[C[k]].insert(C[j]);
		}
	}
	int ans=0;
	for(i=1;i<=n;i++)
	{
		int s=Color[C[i]].size();
		if(s>ans)
		{
			ans=s;
			c=C[i];
		}
		if(ans==s)
			c=min(c,C[i]);
	}
	cout<<c<<endl;
	return 0;
}
