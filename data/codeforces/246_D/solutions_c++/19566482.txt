#include<bits/stdc++.h>
using namespace std;
#define mx 100001
int main(){
	int n,m,res=0,index=1000000;
	cin >> n >> m;
	int col[n+1];
	for(int i=1;i<=n;i++){
		cin >> col[i];
		index=min(index,col[i]);
	}
	set<int>st[mx];
	while(m--){
		int x,y;
		cin >> x >> y;
		if(col[x]!=col[y]){
			st[col[x]].insert(col[y]);
			st[col[y]].insert(col[x]);
		}
	}
	for(int i=1;i<mx;i++){
		if(st[i].size()>res){
			res=st[i].size();
			index=i;
		}
	}
	cout << index << endl;
	return 0;
}
