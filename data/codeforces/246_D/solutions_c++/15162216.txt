#include<bits/stdc++.h>
using namespace std;
bool mark[100010]={false};
set<int>ss[100010];
int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	int c[n];
	for(int i=0;i<n;i++)
	{
		scanf("%d",&c[i]);
		mark[c[i]]=true;
	}
	while(m--)
	{
		int x,y;
		scanf("%d %d",&x,&y);
		x--;y--;
		if(c[x]==c[y])
			continue;
		ss[c[x]].insert(c[y]);
		ss[c[y]].insert(c[x]);
	}
	int ans=0;
	while(!mark[ans])ans++;
	for(int i=1;i<100010;i++)
	{
		if(ss[i].size()>ss[ans].size())
			ans=i;
	}
	printf("%d\n",ans);
	return 0;
}
