#include <iostream>
#include <set>

using namespace std;

const int MAXN = 1e5 + 10;
int c[MAXN];
set<int> ng[MAXN];

int main() {
	int n, m;
	cin >> n >> m;
	int ans = 1e9;
	for (int i = 0; i < n; i++)
		cin >> c[i], ans = min(ans, c[i]);
	for (int i = 0; i < m; i++) {
		int x, y;
		cin >> x >> y;
		x--; y--;
		if (c[x] != c[y])
			ng[c[x]].insert(c[y]), ng[c[y]].insert(c[x]);
	}
	for (int i = 1; i < MAXN; i++)
		if (ng[i].size() > ng[ans].size())
			ans = i;
	cout << ans << endl;
	return 0;
}
