# include <set>
# include <stdio.h>
# include <algorithm>

using namespace std;

int n,m,a,b,mn = 100009;

int c[100009];
set < int > v[100009];

int main()
{
	scanf("%d %d",&n,&m);
	
	for( int i=1;i<=n;i++)
	{scanf("%d",&c[i]); mn=min(mn,c[i]);}
	
	for( int i = 0 ; i < m ; i ++ )
	{
		scanf("%d %d",&a,&b);
		if(c[a]!=c[b])
		{
			v[c[a]].insert(c[b]);
			v[c[b]].insert(c[a]);
		}
	}
	for(int i = 1;i<=n;i++)
	if(v[i].size()>v[mn].size())
	mn=i;
	
	printf("%d",mn);
	
	return 0;
}
