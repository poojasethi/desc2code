#include<bits/stdc++.h>
using namespace std;
int const maxn = 1e5+10;
int n, m, cor[maxn];
set<int> C[maxn];
main(){
	scanf("%d%d",&n,&m);
	for(int i = 1; i <= n; i++) scanf("%d",&cor[i]), C[cor[i]].insert(0);
	for(int i = 1; i <= m; i++){
		int u, v;
		scanf("%d%d",&u,&v);
		C[cor[u]].insert(cor[v]);
		C[cor[v]].insert(cor[u]);
	}
	int id = 0, maior = 0;
	for(int i = 1; i <= 100000; i++){
		C[i].erase(i);
		if(C[i].size() > maior) maior = C[i].size(), id = i;
	}
	printf("%d\n",id);
}
