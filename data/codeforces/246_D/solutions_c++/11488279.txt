#include<bits/stdc++.h>
using namespace std;

int n,m,i,c[100005],a,b,ma=0,mark=0,ans;
set< int >s[100005];

int main()
{
    
cin>>n>>m;
for(i=1;i<=n;i++)
{
    cin>>c[i];
    
}
for(i=0;i<m;i++)
{
    cin>>a>>b;
    if(c[a]!=c[b])
    {
    s[c[a]].insert(c[b]);
    s[c[b]].insert(c[a]);
    }
    
}
ans=100000000;
for(i=1;i<=n;i++)
{
    ma = s[c[i]].size();
    if(mark<ma)
        {mark=ma;ans=c[i];}
    else if(mark==ma)ans = min(ans,c[i]);   
        
    
}
    cout<<ans;
    
    
return 0;   
}
