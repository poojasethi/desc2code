#include <bits/stdc++.h>

using namespace std;

const int MN = 1e5 + 100;

int n ,m;
int c[MN];
set<int>S[MN];

void input()
{
	cin >> n >> m;
	for(int i=0;i<n;++i){
		cin >> c[i];
		S[c[i]].insert(c[i]);
	}
	int a,b; 
	for(int i=0;i<m;++i){
		cin >> a >> b;
		--a, --b;
		S[c[a]].insert(c[b]);
		S[c[b]].insert(c[a]);
	}
}

void solve()
{
	sort(c, c+n);
	int mn = 0;
	for(int i=0;i<n;++i){
		if(S[mn].size() < S[c[i]].size())
			mn = c[i];
	}
	cout << mn << endl;
}

int main()
{
	input();
	solve();
}
