#include<bits/stdc++.h>
using namespace std;
vector <int> g[100005];
set <int> a[100005];
int main()
{
	int n,m,c[100005]={0};
	cin >> n >> m;
	for(int i=1;i<=n;i++)cin >> c[i];
	for(int i=0;i<m;i++)
	{
		int u,v;
		cin >> u >> v;
		g[u].push_back(v);
		g[v].push_back(u);
		if(c[u]!=c[v])
		{
			a[c[u]].insert(c[v]);
			a[c[v]].insert(c[u]);
		}
	}
	int mx=0,ans=0;
	for(int i=0;i<=100000;i++)
		if(mx<a[i].size()){mx=a[i].size();ans=i;}
	if(mx==0 && ans==0)
		ans=*min_element(c+1,c+n+1);
	cout << ans << endl;
	return 0;
}
