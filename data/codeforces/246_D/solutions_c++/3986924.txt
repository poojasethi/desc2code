#include<iostream>
#include<set>
using namespace std;

int main(){

int n,m,p,q;
int input_array[100010];
set<int> set_array[100010];
cin>>n>>m;
for(int i=0;i<n;i++)
{
		cin>>input_array[i];
		set_array[input_array[i]].insert(input_array[i]);

}
for(int i=0;i<m;i++){
	cin>>p>>q;
	p = input_array[p-1];
	q = input_array[q-1];
	set_array[p].insert(q);
	set_array[q].insert(p);
}
int max=0,color=0;
for(int i=0;i<100004;i++){
	if(set_array[i].size()>max){
		max = set_array[i].size();
		color=i;
	}
}
cout<<color;

return 0;
}
