#include<set>
#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n,m; scanf("%d%d",&n,&m);
	static int color[100000];
	static bool b[100000];
	rep(u,n) scanf("%d",color+u), color[u]--, b[color[u]]=true;

	static set<int> C[100000];
	rep(_,m){
		int u,v; scanf("%d%d",&u,&v); u--; v--;
		if(color[u]!=color[v]){
			C[color[u]].insert(color[v]);
			C[color[v]].insert(color[u]);
		}
	}

	int ans=-1;
	rep(c,100000) if(b[c]) if(ans==-1 || C[c].size()>C[ans].size()) ans=c;
	printf("%d\n",ans+1);

	return 0;
}
