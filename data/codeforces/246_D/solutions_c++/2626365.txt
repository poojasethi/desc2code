#include <iostream>
#include <map>
#include <algorithm>
using namespace std;

int c[100001];
map<int,bool> Q[100001];

main () {
  int n, m;
  cin >> n >> m;
  for (int i=1; i<=n; i++)
    cin >> c[i];
  for (int i=0; i<m; i++) {
    int a, b;
    cin >> a >> b;
    int ca = c[a], cb = c[b];
    if (ca != cb)
      Q[ca][cb] = Q[cb][ca] = 1;
  }
  int best = *min_element(c+1, c+n+1);
  for (int i=1; i<=n; i++)
    if (!best || Q[i].size() > Q[best].size())
      best = i;
  cout << best << endl;
}
