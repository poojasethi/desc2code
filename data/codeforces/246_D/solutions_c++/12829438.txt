#include <bits/stdc++.h>
using namespace std;

int c[100001];
set<int> s[100001];
bool cc[100001];

int main(){
	int n,m,a,b,cnt=-1,ret;
	cin >> n >> m;
	for (int i = 0; i < n; i++){
		cin >> c[i+1];
		cc[c[i+1]]=true;
	}
	for (int i = 0; i < m; i++){
		cin >> a >> b;
		if (c[a]==c[b]) continue;
		s[c[a]].insert(c[b]);
		s[c[b]].insert(c[a]);
	}
	for (int i = 1; i <= 100000; i++){
		if (!cc[i]) continue;
		if (cnt<(int)s[i].size()){
			cnt=(int)s[i].size();
			ret=i;
		}
	}
	cout << ret << endl;
	return 0;
}
