#include <cstdio>
#include <set>

using namespace std;

set<int> skup[100001];
int n,m,c[100001],a,b;

int main(void) {
	scanf("%d %d",&n,&m);
	for(int i=1;i<=n;++i){
		scanf("%d",&c[i]);
		skup[c[i]].insert(c[i]);
	}
	for(int i=0;i<m;++i) {
		scanf("%d %d",&a,&b);
		if(c[a]!=c[b]) {
			skup[c[a]].insert(c[b]);
			skup[c[b]].insert(c[a]);
		}
	}
	
	int sol=0,ind=0;
	for(int i=1;i<=100000;++i)
		if(skup[i].size()>sol) {
			sol=skup[i].size();
			ind=i;
		}
	printf("%d",ind);
	return 0;
}
