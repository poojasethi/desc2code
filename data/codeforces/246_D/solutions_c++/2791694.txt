# include <iostream>
# include <set>
using namespace std;
const int MAX_N=100*1000+10;

int N,M,C[MAX_N],MX=MAX_N;
set<int> S[MAX_N];

int main()
{
	ios::sync_with_stdio(false);
	cin >> N >> M;
	for(int i=1;i<=N;i++)
			cin >> C[i],
			MX=min(MX,C[i]);
	for(int i=0;i<M;i++)
	{
		int a,b;
		cin >> a >> b;
		if(C[a]!=C[b])
				S[C[a]].insert(C[b]),
				S[C[b]].insert(C[a]);
	}
	for(int i=1;i<MAX_N;i++)
			if(MX==-1 || S[MX].size()<S[i].size())
				MX=i;
	cout<<MX<<endl;
	return 0;
}
