#include<iostream>
#include<set>
using namespace std;
set<int>g[100001];
int c[100001],v,e,i,x=-1,z=100001,a,b;
main(){
    cin>>v>>e;
    for(i=0;i<v;i++){
        cin>>c[i];
        z=(--c[i]<z)?c[i]:z;
    }
    for(i=0;i<e;i++){
        cin>>a>>b;
        a--;
        b--;
        if(c[a]!=c[b]){
            g[c[a]].insert(c[b]);
            g[c[b]].insert(c[a]);
        }
    }
    for(i=0;i<100001;i++)
        if(g[i].size()>g[z].size()){
            z=i;
        }
    cout<<z+1;
}
