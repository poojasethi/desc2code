#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n, m, start, end, i,index = 1000000;
	int color[100001];
	cin >> n >> m;
	for(i=1;i <= n;i++)
		scanf("%d", &color[i]), index = min(index, color[i]);
	set<int>colorset[100001];
	int ans = 0;
	while(m--)
	{
		scanf("%d %d", &start, &end);
		if(color[start]!=color[end])
			colorset[color[start]].insert(color[end]), colorset[color[end]].insert(color[start]);
	}
	for(i=1;i<=100000;i++)
	{
		if(colorset[i].size()>ans)
			ans = colorset[i].size(), index = i;
	}
	printf("%d\n",index);
	return 0;
}
