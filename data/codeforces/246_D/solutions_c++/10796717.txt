#include<iostream>
#include<algorithm>
#include<set>
#define MAX 100001
using namespace std;
int C[MAX];
set<int>l[MAX];
int main()
{
	int i,n,m,ans=100001,max=0,a,b;
	cin>>n>>m;
	for(i=1;i<=n;i++){cin>>C[i];ans=min(ans,C[i]);}
	for(i=1;i<=m;i++){
		cin>>a>>b;
		if(C[a]!=C[b]){
			l[C[a]].insert(C[b]);
			l[C[b]].insert(C[a]);
		}
	}
	for(i=1;i<=100001;i++){
		if(l[i].size()>max){
			max=l[i].size();
			ans=i;
		}
	}
	cout<<ans<<endl;
}
