#include <bits/stdc++.h>
#define MAX 111111

using namespace std;

int C[MAX];
set<int> S[MAX];
bool used[MAX];

int main()
{
	int N, M;
	cin >> N >> M;
	for (int i = 1; i <= N; i++)
	{
		cin >> C[i];
		used[C[i]] = true;
	}
		
		
	int u, v;
	while (M--)
	{
		cin >> u >> v;
		u = C[u]; v = C[v];
		if (u == v) continue;
		S[u].insert(v);
		S[v].insert(u);
	}
	
	int ans = -1;
	for (int i = 1; i < MAX; i++)
		if (used[i] && (ans == -1 || S[i].size() > S[ans].size()))
			ans = i;
	cout << ans;
} /* 3 1 13 13 4 1 2*/
