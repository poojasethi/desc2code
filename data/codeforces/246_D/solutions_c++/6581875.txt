#include <bits/stdc++.h>
using namespace std;
#define REP(i,n) for(int i = 0; i < (n); ++i)
int color[(int)1e5+5];
set<int> s[(int)1e5+5];
int N, M, a, b;
int main(){
	ios::sync_with_stdio(0);
	cin >> N >> M;
	REP(i, N) cin >> color[i];
    REP(i, M){
		cin >> a >> b;
		if (color[a - 1] != color[b - 1]){
			s[color[a - 1]].insert(color[b - 1]);
			s[color[b - 1]].insert(color[a - 1]);
		}
	}
	int res = INT_MAX, t = 0;
	REP(i, N) res = min(res,color[i]);
	REP(i, (int)1e5+1) if(s[i].size() > t) res = i, t = s[i].size();
	cout << res;
}
