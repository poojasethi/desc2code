#include<iostream>
#include<set>
#include<algorithm>
using namespace std;

const int oo = 100010;
int a[oo], mi = oo;
set<int> ran[oo];

int main(){
  int n, m, x, y;
  cin >> n >> m;
  for(int i = 1; i <= n; i++){
    cin >> a[i];
    mi = min(mi, a[i]);
  }
  for(int i = 0; i < m; i++){
    cin >> x >> y;
    if(a[x] != a[y]){
      ran[a[x]].insert(a[y]);
      ran[a[y]].insert(a[x]);
    }
  }
  for(int i = 1; i < oo; i++){
    if(ran[mi].size() < ran[i].size()){
      mi = i;
    }
  }
  cout << mi;
  return 0;
}
