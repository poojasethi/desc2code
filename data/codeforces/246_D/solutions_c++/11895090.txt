#include<bits/stdc++.h>
using namespace std;
#define INF 1000000000
vector <int> g[100005];
set <int> a[100005];
int main()
{
	int n,m,c[100005]={0},mx=0,ans=INF;
	cin >> n >> m;
	for(int i=1;i<=n;i++){cin >> c[i];ans=min(ans,c[i]);}
	for(int i=0;i<m;i++)
	{
		int u,v;
		cin >> u >> v;
		g[u].push_back(v);
		g[v].push_back(u);
		if(c[u]!=c[v])
		{
			a[c[u]].insert(c[v]);
			a[c[v]].insert(c[u]);
		}
	}
	for(int i=0;i<=100000;i++)
		if(mx<a[i].size()){mx=a[i].size();ans=i;}
	cout << ans << endl;
	return 0;
}
