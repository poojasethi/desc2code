#include <cstdio>
#include <cmath>

int n;

bool check(int v){
	for(int i=2;i<=(int)sqrt(v);++i)
		if(v%i==0)return false;
	return true;
}

int main(){
	scanf("%d",&n);
	int cnt=0;
	int zz=13;
	while(cnt!=n){
		int tot=0;
		int gzz=zz;
		while(gzz!=0){
			tot=tot*10+gzz%10;
			gzz/=10;
		}
		if(tot!=zz&&check(tot)&&check(zz))cnt++;
		if(cnt==n)break;
		zz++;
	}
	printf("%d\n",zz);
}
