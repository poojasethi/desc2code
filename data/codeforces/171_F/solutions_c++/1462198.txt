#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <iostream>

using namespace std;

long long inv(long long a)
{
    long long r;
    for(r=0;a>0;a/=10)
     r = r*10+(a%10);
    return r;
}

bool simp(long long n)
{
    long long i;
    for(i=2;i<=sqrt(n);i++)
     if(n%i==0)
      return 0;
    return 1;
}

int main()
{
    long long n, i, j;
    cin >> n;
    for(i=10, j=0;j<n;i++)
     if(simp(i) && simp(inv(i)) && i!=inv(i))
      j++;
    cout << i-1;
    return 0;
}
