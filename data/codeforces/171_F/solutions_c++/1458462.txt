#include<cstdio>


	int n;
int gct=0;
bool err=false;

bool pr(int a){
	for(int i=2;i*i<=a;i++)
		if(a%i==0) return false;
	return true;
}

void zerk(int a){
	int a1=a,b=0;
	while(a1){
		b=10*b+a1%10;
		a1/=10;
	}
	if(b!=a)
		if(pr(b)){
			gct++;
			if(gct==n) {
				printf("%d\n",a);
				err=true;
			}
		}
}


int main(){
	scanf("%d",&n);
	int i=12;
	while(err==false){
		i++;
		if(pr(i)) zerk(i);
	}
	return 0;
}
