#include<stdio.h>
#include<math.h>
int n,stk[11185],top;
inline bool check(int a)
{
       int i,max=sqrt(a);
       for(i=2;i<=max;i++)
         if(a%i==0)return 0;
       return 1;
}
inline int change(int a)
{
       int b=0;
       while(a)
       {
         b*=10;
         b+=a%10;
         a/=10;
       }
       return b;
}
int main()
{
    int i,j;
    scanf("%d",&n);
    for(i=13;top<n;i+=2)
      if(check(i)&&check(change(i))&&change(i)!=i)stk[++top]=i;
    printf("%d\n",stk[n]);
    return 0;
}
