#include<cstdio>
#include<cmath>
int res[11184];
int rev(int n){
	int ret=0;
	while(n){
		ret*=10;
		ret+=n%10;
		n/=10;
	}
	return ret;
}
int prime(int n){
	int max=sqrt((double)n);
	int i;
	if(n%2==0)return n==2;
	for(i=3;i<=max;i+=2)if(n%i==0)return 0;
	return 1;
}
int main(){
	int cur=0;
	int i;
	int r;
	for(i=13;cur<11184;++i){
		r=rev(i);
		if(i!=r&&prime(i)&&prime(r))res[cur++]=i;
	}
	int n;
	while(scanf("%d",&n)!=EOF)printf("%d\n",res[n-1]);
}
