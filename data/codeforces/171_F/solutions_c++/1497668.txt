#include <cstdio>

inline bool prime(int n)
{
  for (int i = 2; i * i <= n; i++)
    if (n % i == 0)
      return 0;
  return 1;
}

inline int mirror(int n)
{
  int res = 0;
  while (n)
    res *= 10, res += n % 10, n /= 10;
  return res;
}

int main()
{
  int n;
  scanf("%d", &n);
  int now = 0;
  int now_num = 10;
  while (now < n)
  {
    if (prime(now_num) && prime(mirror(now_num)) && now_num != mirror(now_num))
      ++now;
    if (now < n)
      ++now_num;
  }
  printf("%d\n", now_num);
}
