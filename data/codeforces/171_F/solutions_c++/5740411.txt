#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
#include <utility>

using namespace std;

char isPrime (int N)
{
	int i;
	for (i = 2; (i*i) <= N; i++)
	{
		if ((N % i) == 0) return 0;
	}
	return 1;
}

int reverse (int B)
{
	int C = 0;
	while (B)
	{
		C = C * 10 + (B % 10);
		B /= 10;
	}
	return C;
}

int main ()
{
	int N;
	scanf("%d",&N);
	int cnt = 0;
	int res = 1;
	while (cnt < N)
	{
		res++;
		if ((isPrime(res)) && (res != reverse(res)) && (isPrime(reverse(res)))) cnt++;
	}
	printf("%d\n",res);
	return 0;
}
