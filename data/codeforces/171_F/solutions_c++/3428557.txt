//In the Name of God
#include <iostream>
#include <string>

using namespace std;

const int maxn = 1000000 + 10;
int n , cnt = 0;
bool mark[maxn];

int f(int a){
  int ret = 0;
  while(a > 0){
    ret = ret * 10 + a % 10;
    a /= 10;
  }
  return ret;
}

int main(){
  cin >> n;
  for(int i = 2 ; i < maxn ; i++)
    for(int j = i * 2 ; j < maxn ; j += i)
      mark[j] = true;
  for(int i = 2 ; i < maxn ; i++)
    if(!mark[i] && !mark[f(i)] && f(i) != i){
      cnt++;
      if(cnt == n){
	cout << i << endl;
	return 0;
      }
    }
  return 0;
}
