#include<iostream>
#include<cmath>
using namespace std;
long long  ters(long long s){
	long long cvp=0; 
	while(s){
		cvp=(cvp*10)+s%10;
		s/=10;
	}
   return cvp;
}
int isprime(long long n){
	if(n==2 || n==3)
		return 1;
	if(n%2==0 || n%3==0)
		return 0;
	for(long long i=2;i<=sqrt(n);i++){
		if(n%i==0)
			return 0;
	}
	return 1;
}
int main(){
	long long n,say=0,i;
	cin>>n;
	for(i=10;1;i++){
		if(ters(i)!=i) 
			if(isprime(i) && isprime(ters(i))){
			say++;
			if(say==n){
				cout<<i;
				return 0;
			}
		}
	}
	return 0;	
}
