#include <stdio.h>
main() {
  bool a[999999];
  int c=999999,d=0,k=0,n,p,rev(int);
  for (int i=2; i<c; i++)
    a[i]=1;
  for (int i=2; i<c; i++)
    if (a[i])
      for (int j=2*i; j<c; j+=i)
        a[j]=0;
  scanf("%d",&n);
  while (k<n) {
    d++;
    if (a[d]) {
      p=rev(d);
      if (d!=p && a[p])
        k++;
      }
    }
  printf("%d",d);
  return 0;
  }

int rev(int a) {
  int r=0;
  while (a) {
    r=10*r+a%10;
    a/=10;
    }
  return r;
  }
