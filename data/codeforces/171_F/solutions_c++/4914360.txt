#include<iostream>
using namespace std;
int rev(int x)
{
    int y=0;
    while(x)
    {
        y=y*10+x%10;
        x/=10;
    }
    return y;
}
bool isprime(int x)
{
    for(int i=2;i*i<=x;i++)
    {
        if(x%i==0) return 0;
    }
    return 1;
}
int main()
{
    int a,b=10;
    cin>>a;
    while(a)
    {
        b++;
        if(b!=rev(b) && isprime(b) && isprime(rev(b))) a--;
    }
    cout<<b;//the ath one
    return 0;
}
