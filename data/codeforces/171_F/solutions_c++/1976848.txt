#include <iostream>
#include <cmath>
using namespace std;
bool prime(int a)
{
    for(int i=2; i*i<=a; i++)
    {
        if (a%i==0)return 0;
    }
    return 1;
}
int mirrow(int n) 
    {    int c=0;
        while (n) {
        c = c * 10 + n % 10;
        n /= 10;}
        return c;
      }
int main()
{ int n,num=0;
cin>>n;

  for(int i=13;i<=10000000;i++)
      {if(prime(i)){
          if(prime(mirrow(i))&&i!=mirrow(i)){
              num++;
              if (num == n) {cout<<i;return 0;}}
   }
   }
}
