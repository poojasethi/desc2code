#include<stdio.h>
#include<math.h>

int a[11185];

int prime(int n)
{
	for(int i=2;i*i<=n;i++)
	{
		if(n%i==0)
			return 0;
	}
	return 1;
}

int mainn(int x)
{
	int sum=0;
	while(x)
	{
		sum=sum*10+x%10;
		x=x/10;
	}
	return sum;
}

int main()
{
	int i,k=1;
	for(i=12;i<=999998;i++)
	{
		if(prime(i)==1&&prime(mainn(i))==1&&i!=mainn(i))
		a[k++]=i;
	}
	int n;
	while(~scanf("%d",&n))
	{
		printf("%d\n",a[n]);
	}
	return 0;
}
