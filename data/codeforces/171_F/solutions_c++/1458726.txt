#include<cstdio>
#include<iostream>

using namespace std;

int i,n,j,prime[1000005],k,a;

int reverse(int i)
{
	int ret=0;
	while (i)
	{
		ret=ret*10+i%10;
		i/=10;
	}
	return ret;
}

int main()
{
	scanf("%d", &n);
	for (i=2; i<=1000000; i++)
		if (!prime[i])
			for (j=i+i; j<=1000000; j+=i) prime[j]=true;
	for (i=2; i<=1000000; i++) 
	{
		k=reverse(i);
		if (!prime[k] && !prime[i] && reverse(i)!=i) { --n; }
		if (!n) { printf("%d\n", i); return 0; }
	}
}
