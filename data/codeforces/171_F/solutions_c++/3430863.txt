# include <stdio.h>
# include <math.h>
# include <algorithm>

using namespace std;

int a,z,x,y;

bool v[1000010];

int main()
{
	scanf("%d",&a);
	
	for(int i=2;i<=1000009;i++)
	for(int j=i*2;j<=1000009;j+=i)
	v[j]=1;
	
	while(a>0)
	{
		z++;
		x=z;
		y=0;
		
		while(x>0)
		{
			y=y*10+x%10;
			x/=10;
		}
		
		if(v[z]==0 && v[y]==0 && z!=y)
		a--;
	}
	
	printf("%d",z);
	
	getchar();
	getchar();
}
