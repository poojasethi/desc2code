#include<iostream>
using namespace std;
inline int calc(int x)
{
    int ret = 0;
    while(x)
        ret = ret * 10 + x % 10, x /= 10;
    return ret;
}
inline bool isprime(const int &x)
{
    for(int i = 2; i * i <= x; i++)
        if(x % i == 0)
			return false;
    return true;
}
int main()
{
    int n, ans = 10;
    cin >> n;
    while (n)
    {
        ans++;
        if (ans != calc(ans) && isprime(ans) && isprime(calc(ans)))
			n--;
    }
    cout << ans << endl;
    return 0;
}
