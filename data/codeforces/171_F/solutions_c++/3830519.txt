#include <iostream>
#include <cmath>

using namespace std;

#define ll int

ll reverse(ll n)
{
	ll ans = 0;
	while (n != 0)
	{
		ans = ans * 10 + n % 10;
		n /= 10;
	}
	return ans;
}

bool prime(ll n)
{
	ll s = sqrt(n);
	for (ll i = 2; i <= s; i++)
		if (n % i == 0)
			return false;
	return true;
};

int main()
{
	int d;
	cin >> d;
	ll a = 2, index = 0;
	while (1)
	{
		ll r = reverse(a);
		if (prime(a) && a != r && prime(r))
			index++;
		if (index == d)
		{
			cout << a << endl;
			return 0;
		}
		a++;
	}
}
