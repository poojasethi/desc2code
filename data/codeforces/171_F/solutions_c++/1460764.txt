#include <iostream>
using namespace std;
bool is[1000000];
long long f(long long a)
{
    long long t=0;
    while(a)
    {
        t=10*t+a%10;
        a/=10;
    }
    return t;
}
long long num[11190],p;
int main()
{
    for(int i=2;i<=1000000/2;++i)
        if(!is[i])
        for(int j=i+i;j<=1000000;j+=i)
            is[j]=true;
    p=0;
    for(long long i=13;i<=1000000;++i)
        if(!is[i]&&!is[f(i)]&&i!=f(i))
            num[++p]=i;
    int a;
    cin>>a;
    cout<<num[a]<<endl;
    return 0;
}
