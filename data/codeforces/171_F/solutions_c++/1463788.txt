#include<stdio.h>
#include<cmath>
int asal(int x)
{
	int i,n=(int)sqrt(x);
	for(i=2;i<=n;i++)
	{
		if(x%i==0)return 0;
	}
	return 1;
}
int ters(long long k)
{
	int n=0;
	while(k)
	{
		n*=10;
		n+=k%10;
		k/=10;
	}
	return n;
}

	{
		l=ters(i);
