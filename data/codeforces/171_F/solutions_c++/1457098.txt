#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
bool asal(int t)
{
	int i,j;
	if(t==2 || t==3)return 1;
	if(t%2==0) return 0;

	j=sqrt(t)+2;
	for(i=3;i<=j;i+=2)
		if(t%i==0)return 0;
	return 1;	
}

int main()
{
	int N;
	cin>>N;
	int i=11,s=0;
	int b,d;
	s=0,i;
	for(i=13;i;i+=2)
	{
		b=i;
		d=0;
		while(b)
		{
			d=(d*10)+b%10;
		    b/=10;
		}
		if(d!=i && asal(i) && asal(d))s++;
		if(s==N){cout<<i<<endl; break;}
	}
	return 0;
}
