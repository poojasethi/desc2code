#include<stdio.h>
bool isprime(int x){
	for(int i=2; i*i<=x; i++)
		if(x%i==0)
			return false;
	return true;
}
int rev(int x){
	int r=0;
	while(x)r=r*10+x%10,x/=10;
	return r;
}
int main(){
	int i;
	scanf("%d",&i);
	for(int x=13;;x++)
		if(isprime(x) && rev(x)!=x && isprime(rev(x))){
			--i;
			if(i==0){
				printf("%d\n",x);
				return 0;
			}
		}
}
