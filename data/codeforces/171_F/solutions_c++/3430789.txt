#include <stdio.h>
int say,n;
int asal(int x)
{
	int i;
	for (i=2;i*i<=x;i++)
		if (!(x%i)) return 0;
	return 1;
}
int ters(int x)
{
	int y=0,as;
	as=x;
	while (x)
	{
		y=y*10+(x%10);
		x=x/10;
	}
	if (y==as) return 0;
	return (asal(y));
}
int main()
{
	scanf("%d",&n);
	for (int i=12;;i++)
	{
		if (asal(i) && ters(i))
			say++;
		if (say==n)
		{
			printf("%d\n",i);
			scanf(" ");
			return 0;
		}
	}	
}
