#include <iostream>
#include <cmath>

using namespace std;

bool F(int x)
{
	int n,i,n1;
	int t=0;
	int y=x;
	while(y)
	{
		t*=10;
		t+=y%10;
		y/=10;
	}
	if(x == t) return false;


	n=sqrt(x);
	for(i=2;i<=n;i++)if(!(x%i))  return false;
	
	n1=sqrt(t);
	for(i=2;i<=n1;i++)if(!(t%i))  return false;
	return true;
}

int main()
{
	int x=0,i,n;
	cin >> n;
	for(i=10;i;i++)
	{
		if(F(i))
		{
			x++;
			if(x == n)
			{ 
				cout << i << endl;
				return 0;
			}
		}
	}
	return 0;
}
