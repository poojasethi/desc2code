#include<stdio.h>
#include<math.h>

int a[11185];

int prime(int n)
{
	for(int i=2;i*i<=n;i++)
	{
		if(n%i==0)
			return 0;
	}
	return 1;
}

void mainn()
{
	int k=1;
	for(int i=12;i<=999998;i++)
	{
		if(prime(i))
		{
			int m=i;
			int sum=0;
			while(m)
			{
				sum=sum*10+m%10;
				m=m/10;
			}
			if(prime(sum)&&sum!=i)
				a[k++]=i;
		}
	}
}

int main()
{
	int n;
	scanf("%d",&n);
	mainn();
	printf("%d\n",a[n]);
}
