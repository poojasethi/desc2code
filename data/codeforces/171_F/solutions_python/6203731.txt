from math import sqrt
V = int(1e6)
isPrime = [True for i in xrange(V + 1)]
isPrime[0] = isPrime[1] = False
for i in xrange(2, int(sqrt(V)) + 1):
	if isPrime[i]:
		for j in xrange(2 * i, V + 1, i):
			isPrime[j] = False
d = input()
found = 0
num = 0
while found < d:
	num += 1
	if isPrime[num]:
		revNum = int(str(num)[::-1])
		if num != revNum and isPrime[revNum]:
			found += 1
print num
