primes = {}
emirps = []


sieve = range(1000000)
for i in range(2,1000000):
    if sieve[i]==0:
        continue
    primes[i]=0
    #print "append ",i
    for j in range(2*i,1000000,i):
        sieve[j]=0
for prime in primes:
    #print prime
    if prime<10: continue
    rev=0
    p=prime
    while p>0:
        rev*=10
        rev+=p%10
        p/=10
    if rev==prime: continue
    if rev in primes:
        emirps.append(prime)

emirps.sort()

a = int(raw_input())
print emirps[a-1]
