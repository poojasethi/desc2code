#include<stdio.h>
int a[1005];
int weight[1005]; 
int vis[1005];
int vex[1005];
int main()
{
	int i,n,p;
	int tmp1,tmp2,tmp3;
	scanf("%d%d",&n,&p);
	for(i=0;i<p;i++)
	{
		scanf("%d%d%d",&tmp1,&tmp2,&tmp3);
		vis[tmp1]--;
		vis[tmp2]++;
		a[tmp1]=tmp2;
		weight[tmp1]=tmp3;
	}
	int num=0,q=0;
	for(i=1;i<=n;i++)
		if(vis[i]==-1)
		{
			num++;
			vex[q++]=i;
		}
	for(i=1;i<=n;i++)
		if(vis[i]==1)
		{
			weight[i]=1000000;
		}
	if(!num)
		printf("0\n");
	else
	{
		printf("%d\n",num);

		for(i=0;i<q;i++)
		{
			int k=vex[i];
			int min=weight[k];
			while(a[k])
			{
				k=a[k];
				if(weight[k]<min)
					min=weight[k];
			}
			printf("%d %d %d\n",vex[i],k,min);
		}
	}
	return 0;
	
}
