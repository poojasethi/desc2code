#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
int n,m,d[1010],next[1010],w[1010];

int main()
{
    scanf("%d%d",&n,&m);
    for (int i=1;i<=m;i++)
    {
        int a,b,c; scanf("%d%d%d",&a,&b,&c);
        d[b]++;
        next[a] = b;
        w[a] = c;
    }
    int total = 0;
    for (int i=1;i<=n;i++)
        if ( d[i]==0 && next[i]!=0 ) total++;
    printf("%d\n",total);
    for (int i=1;i<=n;i++)
        if ( d[i]==0 )
        {
            int x = i;
            int c = 10000000;
            while ( next[x] )
                c = min( c , w[x] ) , x = next[x];
            if (i!=x) printf("%d %d %d\n",i,x,c);
        }
    return 0;
}
