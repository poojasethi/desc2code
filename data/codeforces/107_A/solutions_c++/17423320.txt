#include<bits/stdc++.h>
#include<vector>

using namespace std;

int n,p;
 pair <int, int>  v[1050];
bool b[1050];
int mn,ans;
bool xx[1050];

void dfs(int x){
   // cout<<v[x].first<<endl;
    if(v[x].first!=0){
   		ans = v[x].first;
   		 mn = min(mn, v[x].second);
    	dfs(v[x].first);
    		
	}
}

int main(){
	int x,y,d;
	cin>>n>>p;
	for(int i=0;i<p;i++){
		cin>>x>>y>>d;
		v[x]=make_pair(y,d);
		b[y]++;
		xx[x]++;
	}
	int cnt=0;
	for(int i=1;i<=n;i++){
		if(xx[i]==1 && b[i]==0) {
			b[i]=1;
			cnt++;
		}
		else b[i]=0;
	}
	cout<<cnt<<endl;
	for(int i=1;i<=n;i++){
		mn = 1e7 ;
		if(b[i]!=0){
			dfs(i);
			cout<<i<<" "<<ans<<" "<<mn<<endl;
		}
	}
}
