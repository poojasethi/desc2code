#include<bits/stdc++.h>
using namespace std;
int main() {
	int n,e;
	scanf("%d %d",&n,&e);
	int cap[n+1][n+1];
	int in[n+1], out[n+1];
	memset(cap,0,sizeof(cap));
	memset(in,0,sizeof(in));
	memset(out,0,sizeof(out));
	for(int i=0;i<e;i++) {
		int a,b,c;
		cin >> a >> b >> c;
		cap[a][b]=c;
		in[b]=a;
		out[a]=b;
	}
	int cnt=0;
	for(int i=1;i<=n;i++) {
		if(in[i]==0) {
			if(out[i]!=0) {
				++cnt;
			}
		}
	}
	cout << cnt << endl;
	for(int i=1;i<=n;i++) {
		if(in[i]==0 && out[i]!=0) {
			int mid = INT_MAX;
			int u=i;
			while(true) {
				int v = out[u];
				if(!v)
					break;
				mid = min(mid, cap[u][v]);
				u= v;
			}
			cout << i << " " << u << " " << mid << endl;
		}
	}
	return 0;
}
