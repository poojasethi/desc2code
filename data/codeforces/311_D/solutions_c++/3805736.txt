#include<cstdio>    
#define lson l,m,rt<<1
#define rson m+1,r,rt<<1|1
const int maxn=100011;
const int mod=95542721;
int sum[maxn<<2][48];
int exp[maxn<<2];
    
void build(int l,int r,int rt){
    exp[rt]=0;
    if(l==r){
        scanf("%d",&sum[rt][0]);
        sum[rt][0]%=mod;
        for(int i=0;i<47;i++){
            sum[rt][i+1]=(1LL*sum[rt][i]*sum[rt][i])%mod;
            sum[rt][i+1]=(1LL*sum[rt][i+1]*sum[rt][i])%mod;
        }
        return ;
    }
    int m=(l+r)>>1;
    build(lson);
    build(rson);
    for(int i=0;i<48;i++)
        sum[rt][i]=(sum[rt<<1][i]+sum[rt<<1|1][i])%mod;
}   
    
void update(int L,int R,int l,int r,int rt){    
    if(L<=l&&r<=R){
        if(++exp[rt]==48) exp[rt]=0;
        return;
    }
    int m=(l+r)>>1;
    if(L<=m) update(L,R,lson);
    if(m<R) update(L,R,rson);
    for(int i=0;i<48;i++)
        sum[rt][i]=(sum[rt<<1][(i+exp[rt<<1])%48]+sum[rt<<1|1][(i+exp[rt<<1|1])%48])%mod;
}   
    
int query(int L,int R,int l,int r,int rt,int ex)
{   
    ex=(ex+exp[rt])%48;
    if(L<=l&&r<=R) return sum[rt][ex];
    int m=(l+r)>>1,ret=0;
    if(L<=m) ret=(ret+query(L,R,lson,ex))%mod;
    if(m<R) ret=(ret+query(L,R,rson,ex))%mod;
    return ret;
}   
    
int main()
{   
    int n,q;
    scanf("%d",&n);
    build(1,n,1);
    scanf("%d",&q);
    while(q--)
    {
        int l,r,t;
        scanf("%d%d%d",&t,&l,&r);
        if(t==1)
            printf("%d\n",query(l,r,1,n,1,0));
        else
            update(l,r,1,n,1);
    }
    return 0;
}
