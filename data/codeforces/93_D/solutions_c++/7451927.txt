#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
ll p,l,r;
inline ll power(ll a,ll b){
	ll ret=1;
	while(b){
		if(b&1)
			ret=ret*a%p;
		a=a*a%p;
		b>>=1;
	}
	return ret;
}
inline ll f(ll n){
	if(n==1)
		return 4;
	else if(n==0)
		return 0;
	else if(n&1)
		return (11*power(3,n/2)-7)%p;
	else
		return (19*power(3,n/2-1)-7)%p;
}
inline ll g(ll x){
    return ((f(x)+f((x+1)/2))*(p+1)/2)%p;
}
