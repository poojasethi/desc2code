#include<stdio.h>
#define ll long long

const int MOD = 1000000007;

ll Pow(ll x, int k) {
    ll z = 1;
    for (; k; k >>= 1, (x *= x) %= MOD)
        if (k & 1) (z *= x) %= MOD;
    return z;
}

ll B(ll x) {
    if(!x) return 0;
    else if(x == 1) return 4;
    else if(x & 1) return (11 * Pow(3, x / 2) - 7) % MOD;
    else return (19 * Pow(3, x / 2 - 1) - 7) % MOD;
}

ll A(ll x) {
    return (B(x) + B((x + 1) / 2)) * Pow(2, MOD - 2) % MOD;
}

int main() {
    ll l, r;
    scanf("%I64d %I64d", &l, &r);
    printf("%I64d\n",(A(r) - A(l - 1) + MOD) % MOD);
    return 0;
}
