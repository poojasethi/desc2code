#include <cstdio>
#define P 1000000007
const int f[5] = {0, 4, 4, 11, 12};
int l, r, ans;
int calc(int n) {
    n++;
    int t = 3, s = 0, u = 1;
    for (; n; n>>=1, u = (u+1LL*t*u)%P, t = 1LL*t*t%P)
        if (n&1) s = (u+1LL*t*s)%P;
    return s;
}
int query(int l, int r) {
    if (l > r) return 0;
    int ret = calc(r);
    if (l > 0) ret -= calc(l-1);
    if (ret < 0) ret += P;
    return ret;
}
int main() {
    scanf("%d%d", &l, &r);
    if (r <= 4) {
        for (int i = l; i <= r; i++) ans += f[i];
        printf("%d\n", ans);
        return 0;
    } else if (l <= 4) {
        for (int i = l; i <= 4; i++) ans += f[i];
        l = 5;
    }
    ans = (ans + 4LL * query((l+1)/2-1, r/2-1)) % P;
    ans = (ans + 1LL * query(l/2, (r-1)/2)) % P;
    ans = (ans + 4LL * query(l/2-1, (r-1)/2-1)) % P;
    ans = (ans + 4LL * query((l+2)/4-1, (r-1)/4-1)) % P;
    ans = (ans + 1LL * query((l+2)/4, (r-1)/4)) % P;
    ans = (ans + 4LL * query(l/4, (r-3)/4)) % P;
    printf("%d\n", ans);
    return 0;
}
