#include <iostream>

using namespace std;

typedef long long int ll;

ll modulo=1e9+7;

ll eleva(ll b,ll e)
{
  if (e==0) return 1;
  if (e%2==0) return eleva(b*b%modulo,e/2);
  return b*eleva(b*b%modulo,e/2)%modulo;
}

ll inv2=5*1e8+4;

ll valoracion(ll n)
{
  if (n==0) return 1;
  if (n==1) return 4;
  if (n==2) return 8;
  if (n%2==0)
    return 8*eleva(3,(n-2)/2)%modulo;
  return 14*eleva(3,(n-3)/2)%modulo;
}

ll sumahastabis(ll n)
{
  if (n==0) return 1;
  if (n==1) return 1+4;
  if (n==2) return 1+4+8;
  ll s;
  if (n%2==0)
    s=1+4+22*(eleva(3,(n-2)/2)-1+modulo)%modulo*inv2%modulo+valoracion(n);
  else
    s=1+4+22*(eleva(3,(n-1)/2)-1+modulo)%modulo*inv2;
  return s%modulo;
}

ll sumahasta(ll n)
{
  if (n==0) return 1;
  if (n==1) return 1+4;
  if (n==2) return 1+4+4;
  return (sumahastabis(n)+sumahastabis((n+1)/2))*inv2%modulo;
}
    
int main()
{
  ll l,r;
  cin>>l>>r;
  cout<<(sumahasta(r)-sumahasta(l-1)+modulo)%modulo<<endl;
}
