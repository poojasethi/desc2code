#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <string>

using namespace std;

#define fin cin
#define fout cout

long long m=1000000007;

long long l,r;

long long pw(long long x,long long k)
{
	long long z=1;
    for (; k; k >>= 1, (x *= x) %= m)
        if (k & 1) (z *= x) %= m;
    return z;
}

long long s(long long x)
{
	if (!x)
		return 0;
	if (x==1)
		return 4;
	if (x&1)
		return (11*pw(3,x/2)-7)%m;
	else
		return (19*pw(3,x/2-1)-7)%m;
}

long long g(long long x)
{
	return (s(x)+s(x+1>>1))*pw(2,m-2)%m;
}

int main()
{
	fin>>l>>r;
	fout<<(g(r)-g(l-1)+m)%m<<endl;
	return 0;
}
