#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
using namespace std;
typedef long long LL;
const int P=1e9+7;
int L,R;

int pow(LL x,int k){
    LL Fc=1;
    for(;k;k>>=1,x=x*x%P)k&1?Fc=Fc*x%P:0;
    return Fc;
}

namespace Ninit{
    void init(){
        scanf("%d%d",&L,&R);
    }
}

namespace Nsolve{
    LL S(int x){
        return x<2?4*x:
            x&1?(11LL*pow(3,x/2)-7)%P
            :(19LL*pow(3,x/2-1)-7)%P;
    }
    LL F(int x){
        return (S(x)+S((x+1)/2))*pow(2,P-2)%P;
    }
    void solve(){
        cout<<(F(R)-F(L-1)+P)%P<<endl;
    }
}

int main(){
    Ninit::init();
    Nsolve::solve();
    return 0;
}
