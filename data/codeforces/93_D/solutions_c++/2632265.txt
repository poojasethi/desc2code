#include<cstdio>
#include<cstring>
const int mo=1000000007;
typedef long long ll;
int f[9][9],g[9][9],h[9][9],a[9],b[9];
int getres(int t){
	if (t==1) return 4;
	int i,j,k;
	memcpy(g,f,sizeof g);
	for (i=0;i<8;i++) a[i]=1;
	a[8]=8;
	for (t-=2;t;t>>=1){
		if (t&1){
			memset(b,0,sizeof b);
			for (i=0;i<9;i++)
				for (j=0;j<9;j++)
					b[j]=(b[j]+a[i]*ll(g[i][j]))%mo;
			memcpy(a,b,sizeof a);
		}
		memset(h,0,sizeof h);
		for (i=0;i<9;i++)
			for (j=0;j<9;j++)
				for (k=0;k<9;k++)
					h[i][k]=(h[i][k]+g[i][j]*ll(g[j][k]))%mo;
		memcpy(g,h,sizeof g);
	}
	return a[8]+4;
}
int dv2(int a){
	a%=mo;
	if (a&1) return (a+mo)/2; else return a/2;
}
int getans(int t){
	return dv2(getres(t)+getres((t+1)/2));
}
int main(){
//	freopen("93d.in","r",stdin);
//	freopen("93d.out","w",stdout);
	int l,r,i,j,b1,b2,b3,b4;
	scanf("%d%d",&l,&r);
	memset(f,0,sizeof f);
	for (i=0;i<8;i++){
		b1=i/2,b2=(i<4?2:0)+(i&1);
		for (j=0;j<8;j++){
			b3=j/2,b4=(j<4?2:0)+(j&1);
			f[i][8]+=(f[i][j]=(b2==b3 && (b2!=0 || b1+b4!=5)));
		}
	}
	f[8][8]=1;
	if (l!=1) printf("%d\n",(getans(r)-getans(l-1)+mo)%mo); else printf("%d\n",getans(r));
	return 0;
}
