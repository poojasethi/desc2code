#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

typedef long long LL;
const int P = 1000000007;

int pow(int a, int n) {
    int ans = 1;
    for (; n; n >>= 1, a = (LL)a * a % P)
        if (n & 1) ans = (LL)ans * a % P;
    return ans;
}

int get(int n) {
    if (n < 2) return n ? 4 : 0;
    return n & 1 ? (11LL * pow(3, n / 2) + P - 7) % P
                 : (19LL * pow(3, n / 2 - 1) + P - 7) % P;
}

int calc(int n) {
    return (LL)(get(n) + get((n + 1) / 2)) * 500000004 % P;
}

int main() {
    int L, R;
    scanf("%d%d", &L, &R);
    printf("%d\n", (((calc(R) - calc(L - 1)) % P) + P) % P);
    return 0;
}
