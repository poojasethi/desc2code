ll pow3(int n){
    ll res=1;
    ll a=3;
    while(n>0){
        if(n%2==1) res=(res*a)%m;
        a=(a*a)%m;
        n/=2;
    }
    return res;
}

ll g(int n){
    int l1=n/2;
    int l2=(n-1)/2;
    return 4*(pow3(l1)-1)+7*(pow3(l2)-1);
}

ll f(int n){
    if(n==0) return 0;
    int l1=n/2;
    int l2=(n-1)/2;
    ll num=(7*(pow3(l2)-1)+g(l2+1))%m;
    ll num2;
    if(num%2==0) num2=num/2;
    else num2=(num+m)/2;
    return 4+2*(pow3(l1)-1)+num2;
}
