#include <cstdio>
#include <cstdlib>
#define maxn 1000000007

using namespace std;

long long l,r,ans;

long long ksm(long long j,long long x)
{
	long long q=j,an=1;
	while (x>0)
	{
		if (x%2==1) an=(an*q)%maxn;
		q=(q*q)%maxn;x=x/2;
	}
	if (an==0) an+=maxn;
	return an;
}

long long  doitji(long long x)
{
	long long a1=7;
	long long xx;
	xx=(a1*(ksm(3,x)-1)) %maxn;
	return (xx)%maxn;
}

long long doitou(long long x)
{
	long long a1=4;
	long long xx;
	xx=(a1*(ksm(3,x)-1))%maxn;
	return (xx)%maxn;
}

long long mul(long long x)
{
	long long xx=0;
	if (x==1) return xx;
	if (x==0) return 0;
	long long ji,ou;
	if (x%2==0)
		ou=x/2,ji=ou-1;
	else 
		ou=ji=x/2;
	xx+=(doitji(ji)+doitou(ou))%maxn;
	xx=(xx*ksm(2,maxn-2))%maxn;
	//if (xx%2==1) xx+=maxn;
	return (xx)%maxn;
}

long long cal(long long r)
{
    long long xx;
    xx=(mul(r)+mul((r+1)/2))%maxn;
	return xx;
}

int main()
{
	/*freopen("color.in","r",stdin);
	freopen("color.out","w",stdout);*/
	scanf("%d%d",&l,&r);
	if (l==1&&r==1) {printf("%d",4);return 0;}
    ans=cal(r)-cal(l-1);
    if (l==1) ans+=4;
    if (ans<0) ans+=maxn;
	printf("%d",ans);
	return 0;
}
