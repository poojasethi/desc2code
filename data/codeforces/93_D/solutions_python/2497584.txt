modu = 10 ** 9 + 7

def S(x):
	if x == 0:
		return 0
	if x & 1:
		return 11 * pow(3, x / 2, modu) - 7
	else:
		return 19 * pow(3, x / 2 - 1, modu) - 7

def T(x):
	return (S(x) + S((x + 1) / 2)) * pow(2, modu - 2, modu) % modu

[l, r] = [int(x) for x in raw_input().split(' ')]
print (T(r) - T(l - 1) + modu) % modu
