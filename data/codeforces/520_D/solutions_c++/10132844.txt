#include<cstdio>
#include<iostream>
#include<map>
#include<set>
#define MAXN 100000
const int mod=1000000009;
using namespace std;
map<pair<int,int>,int>A;
set<int>s;
int ans[MAXN+10];
int x[MAXN+10],y[MAXN+10];
long long jinzhi(int n)
{
	long long sum=0;
	for(int i=0;i<n;i++)
		sum=(sum*n%mod+ans[i])%mod;
	return sum;
}
bool check(int i)
{
	if(y[i]+1==0) return true;
	if(A.count(make_pair(x[i]-1,y[i]+1)))
	{
		if(!A.count(make_pair(x[i]-2,y[i]))&&!A.count(make_pair(x[i]-1,y[i])))
			return false;
	}
	if(A.count(make_pair(x[i],y[i]+1)))
	{
		if(!A.count(make_pair(x[i]-1,y[i]))&&!A.count(make_pair(x[i]+1,y[i])))
			return false;
	}
	if(A.count(make_pair(x[i]+1,y[i]+1)))
	{
		if(!A.count(make_pair(x[i]+1,y[i]))&&!A.count(make_pair(x[i]+2,y[i])))
			return false;
	}
    return true;
}
void update(int x,int y)
{
	if(A.count(make_pair(x,y)))
	{
		if(check(A[make_pair(x,y)]))
			s.insert(A[make_pair(x,y)]);
	}
}
int main()
{
	int n,i,k;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&x[i],&y[i]);
		A[make_pair(x[i],y[i])]=i;
	}
	for(i=0;i<n;i++)
		if(check(i))
			s.insert(i);
	for(i=0;i<n;i++)
	{
		if(i%2==0) 
		{
			do
			{
				k=*s.rbegin();
			    s.erase(k);	
			}while(!check(k));
		}
		else 
		{
			do
			{
				k=*s.begin();
			    s.erase(k);	
			}while(!check(k));
		}
		ans[i]=k;
		A.erase(make_pair(x[k],y[k]));
		update(x[k]-1,y[k]-1);
		update(x[k],y[k]-1);
		update(x[k]+1,y[k]-1);
	}
	cout<<jinzhi(n)<<endl;
}
