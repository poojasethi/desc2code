#include <bits/stdc++.h>
#define MOD 1000000009
#define N 234567

using namespace std;

int x[N], y[N], dd[N];
set<int> s;
map<pair<int, int>, int> m;

bool candel(int t)
{
	for(int i = 0; i < 3; i++)
	{
		int h = m[make_pair(x[t] + i-1, y[t] + 1)] - 1;
		if(h >= 0 && !dd[h])
		{
			int ret = 0;
			for(int j = 0; j < 3; j++)
			{
				int h = m[make_pair(x[t] + i-1 + j-1, y[t])] - 1;
				if(h >= 0 && !dd[h])ret++;
			}
			if(ret == 1)return false;
		}
	}
	return true;
}

int main()
{
	ios_base::sync_with_stdio(false);
	//freopen("d.in","r",stdin);
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
	{
		cin >> x[i] >> y[i];
		m[make_pair(x[i], y[i])] = i + 1;
	}
	for(int i = 0; i < n; i++) if(candel(i)) s.insert(i);
	long long res = 0, trn = 0;
	while(s.size())
	{
		int t;
		if(trn % 2 == 0)
			t = *s.rbegin();
		else
			t = *s.begin();
		s.erase(t);
		if(!candel(t)) continue;
		trn++;
		dd[t] = true;
		res = (res*n + t) % MOD;
		for(int i = 0; i < 3; i++)
		{
			int h = m[make_pair(x[t] + i-1, y[t] + -1)] - 1;
			if(h >= 0 && candel(h) && !dd[h])
				s.insert(h);
		}
	}
	cout << res;
	return 0;
}
