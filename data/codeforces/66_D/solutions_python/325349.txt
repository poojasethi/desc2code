import sys

def isPrime(n) :
    for d in range(2, n):
        if n % d == 0 :
            return False;
    return True;


n = int(raw_input())

if n == 2 :
    print -1
    sys.exit(0)

i = 2
primes = []
while len(primes) < n:
    if isPrime(i) :
        primes.append(i)
    i += 1

mul = 1;
for i in primes :
    mul *= i

for i in range(n - 1, -1, -1):
    print mul / primes[i]
