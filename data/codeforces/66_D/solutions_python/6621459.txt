MAXN = 250
Primes = []
P = [0 for i in range(MAXN)]
def sieve():
	for i in xrange(2,MAXN):
		if(i*i>MAXN):
			break
		if P[i] == 1:
			continue
		for j in xrange(i*i,MAXN,i):
			P[j] = 1
	for i in xrange(2,MAXN):
		if P[i]==0:
			Primes.append(i)

sieve()
N = input()
Ans = 1
if(N>2):
	for i in range(N):
		Ans = Ans * Primes[i]
	for i in range(N):
		print Ans/Primes[i]
else: print -1
