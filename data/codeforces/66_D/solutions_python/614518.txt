import sys

primes = [2]
for i in range(3, 542, 2):
  if not any([ (i % p) == 0 for p in primes ]):
    primes.append(i)

n = int(raw_input())

if n == 2:
  print -1
  sys.exit()

res = primes[-n:]
for i in range(0, n):
  for j in range(0, n):
    if i != j:
      res[j] *= primes[i]

for i in range(0, n):
  print res[i]
