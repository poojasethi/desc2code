import math
def judge_prime(a):
	if(a == 2):
		return True
	for i in range(2, int(math.sqrt(a)) + 1):
		if a % i == 0:
			return False
	return True
n = input()
if n == 2:
	print -1
else:
	a = [i for i in xrange(2, 100000) if judge_prime(i)][:n]
	mult = 1
	for i in a:
		mult *= i;
	for i in a:
		print mult / i
