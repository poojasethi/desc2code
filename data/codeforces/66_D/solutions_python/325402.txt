#!/usr/bin/env python

n=int(raw_input())
if n==2:
	print -1
	exit(0)
a=[2]
k=3
while len(a)<n:
	found=True
	for i in a:
		if k%i==0:
			found=False
			break
	if found:
		a.append(k)
	k+=1

ans=[1]*n
for i in range(n):
	for j in range(n):
		if j==i:
			continue
		ans[j]*=a[i]

for i in ans:
	print i
