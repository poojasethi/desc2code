#!/usr/bin/python

import sys

def prime(x):
  for i in range(2, x):
    if x % i == 0: return False
  return True

n = int(raw_input())
if n == 2:
  print(-1)
  sys.exit(0)  
p = []  
for i in range(2, 1000):
  if prime(i) : p.append(i)
#print(p)  

print(p[0]*p[n])
x = p[0]
for i in range(1, n-1):
  x *= p[i]
  print(x)
print(x * p[n] / p[0])
