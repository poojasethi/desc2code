


def main():

	upDays = int(raw_input())
	downDays = int(raw_input())

	print solve(upDays,downDays)


def solve(up,down):
	
	heights = range(1,up+down+2)

	increasing = heights[0:up+1]

	del heights[0:up+1]

	decreasing = heights[::-1]

	return " ".join(map(str,decreasing + increasing))
	
main()
