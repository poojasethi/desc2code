def hill(A, B):
    """
    >>> hill(0, 1)
    2 1

    >>> hill(2, 2)
    3 4 5 2 1

    """
    up = []
    down = []
    N = A + B + 1

    for i in range(A):
	    up.append(N - A + i)

    for i in range(B):
	    down.append(B - i)

    output = up + [N] + down
    print ' '.join(str(x) for x in output)

if __name__ == "__main__":
    strA = raw_input()
    strB = raw_input()
    hill(int(strA), int(strB))
