#include <iostream>
#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
int main()
{
    long long n,x[1000],y[1000],i,ans;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>x[i]>>y[i];

    }
    sort(x,x+n);
    sort(y,y+n);
     ans=max((x[n-1]-x[0]),(y[n-1]-y[0]));
    cout<<ans*ans;
}
