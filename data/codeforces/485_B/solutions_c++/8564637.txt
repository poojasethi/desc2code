#include <iostream>

using namespace std;

int main()
{
	long long n,x,y;
	long long sx, dx, up, dw;
	cin>>n >> sx >> dw;
	up=dw;
	dx=sx;
	for(int i=1; i<n; i++)
	{
		cin >> x >> y;
		sx = min(sx,x);
		dx = max(dx,x);
		up = max(up,y);
		dw = min(dw,y);
	}
	cout << max(dx-sx,up-dw)*max(dx-sx,up-dw);
	return 0;
}
