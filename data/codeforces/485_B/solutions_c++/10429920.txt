#include <bits/stdc++.h>
using namespace std;
int n,x,y,r,l,u,d;
bool DEBUG=false;
int main(){
	cin>>n>>x>>y;
	l=r=x;
	u=d=y;
	for(int q=1;q<n;q++){
		cin>>x>>y;
		l=min(l,x);
		r=max(r,x);
		u=max(u,y);
		d=min(d,y);
	}

	long long longest=max((r-l),(u-d));
	cout<<longest*longest<<endl;

	return 0;
}
