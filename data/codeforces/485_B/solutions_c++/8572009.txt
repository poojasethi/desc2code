#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n; cin >> n;
    long long int x[n],y[n],i;
    for(i=0;i<n;i++)
        cin >> x[i] >> y[i];
    sort(x,x+n);
    sort(y,y+n);
    long long int ans = max(x[n-1]-x[0],y[n-1]-y[0]);
    cout << ans*ans << endl;
    return 0;
}
