#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
#define mn 1000000001
int main()
{
	int n,a=mn,b=-mn,c=mn,d=-mn,e,f;
	scanf("%d",&n);
	while(n--)
	{
		scanf("%d%d",&e,&f);
		if(e<a)a=e;if(e>b)b=e;
		if(f<c)c=f;if(f>d)d=f;
	}
	a=max(b-a,d-c);
	printf("%I64d\n",(ll)a*a);
}
