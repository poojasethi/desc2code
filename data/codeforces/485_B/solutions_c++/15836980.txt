#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int n;
	long long a[1010],b[1010];
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%lld %lld",&a[i],&b[i]);
	sort(a,a+n);
	sort(b,b+n);
	a[0]=max(a[n-1]-a[0],b[n-1]-b[0]);
	printf("%lld\n",a[0]*a[0]);
	return 0;
}
