#include<bits/stdc++.h> 
#define ll long long
using namespace std; 

int main(){
    int n; cin>>n;
    ll a[n],b[n];
    for(int i=0 ; i<n ; i++){
        cin>>a[i]>>b[i];
    }
    sort(a,a+n);
    sort(b,b+n);
    ll maxx = max(abs(a[0]-a[n-1]),abs(b[0]-b[n-1]));
    cout<<maxx*maxx;
    return 0;
}
