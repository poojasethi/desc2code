#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll x[1005];
ll y[1005];
int main(){
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>x[i]>>y[i];
    sort(x,x+n);
    sort(y,y+n);
    ll nx=x[n-1]-x[0];
    ll ny=y[n-1]-y[0];
    ll nc=max(nx,ny);
    cout<<nc*nc<<endl;
}
