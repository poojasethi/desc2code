#include <bits/stdc++.h>

using namespace std;

typedef long long lld;
typedef string str;

lld n,a[2000],b[2000],q3;

int main()
{
	cin >> n;
	
	for (int i=0; i<n; i++)
	   cin >> a[i] >> b[i];
	
	sort (a,a+n);
	sort (b,b+n);
	
	q3 = max((a[n-1]-a[0]),(b[n-1]-b[0]));
	
	cout << q3*q3;
	  
	return 0;
}
