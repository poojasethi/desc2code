#include <stdio.h>
#include <memory.h>
using namespace std;
#define Min(a,b) (a<b?a:b)

int n;
int a[5];

void Solve() {
	int i,j,k,s=0;
	scanf("%d",&n);
	for(i=0;i<n;i++) {
		scanf("%d",&k);
		a[k]++;
		s+=k;
	}
	if(s<3 || s==5) {puts("-1");return;}
	int ans=0;
	k=Min(a[1],a[2]);ans+=k;a[1]-=k;a[2]-=k;
	if(a[1]) {
		k=a[1]/3;
		a[3]+=k;ans+=2*k;a[1]%=3;
		if(a[1]) {
			if(a[3]>=a[1]) ans+=a[1]; else ans+=2;
		}
	}
	if(a[2]) {
		k=a[2]/3;
		a[3]+=k*2;ans+=k*2;a[2]%=3;
		if(a[2]) {
			if(a[4]>=a[2]) ans+=a[2]; else ans+=2;
		}
	}
	printf("%d\n",ans);
}

int main() {
//	freopen("c1.in","r",stdin);
	Solve();
	return 0;
}
