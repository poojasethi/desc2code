#include<cstdio>
#include<cstring>
#include<algorithm>	
using namespace std;
#define rep(i,n) for (int i=1;i<=n;i++)
int n,x,s,ans,c[5];
int main()
{
	scanf("%d",&n);
	rep(i,n) scanf("%d",&x),s+=x,c[x]++;
	if (s==1 || s==2 || s==5) return puts("-1"),0;
	ans+=x=min(c[1],c[2]),c[0]+=x,c[1]-=x,c[2]-=x,c[3]+=x;
	if (c[1]){
		ans+=2*(x=c[1]/3),c[1]-=x*3,c[3]+=x;
		if (c[1]) if (c[1]==1 && c[3]) ++ans; else ans+=2;
	}else{
		ans+=2*(x=c[2]/3),c[2]-=x*3,c[3]+=2*x;
		if (c[2]) if (c[2]==1 && c[4]) ++ans; else ans+=2;
	}
	printf("%d\n",ans);
	return 0;
}
