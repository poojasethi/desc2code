#include <cstdio>
#include <algorithm>
using namespace std;

int N;
int cs[8];
int as[1000009];

int main()
{
    scanf("%d", &N);
    
    int sum = 0;
    for(int i = 1; i <= N; i++)
    {
	int v;
	scanf("%d", &v);
	as[i] = v;
	cs[v]++;
	sum += v;
    }

    sort(as + 1, as + N + 1);
    for(int i = 1; i <= N; i++)
	as[i] += as[i - 1];

    int ans = -1;
    for(int i = 0; i <= N; i++)
	if(3 * i <= sum && (sum - 3 * i) % 4 == 0 && i + (sum - 3 * i) / 4 <= N)
	{
	    int j = (sum - 3 * i) / 4;
            
	    int t = max(0, cs[4] - j) + as[N - i - j];
	    if(ans == -1 || ans > t)
		ans = t;
	}
    printf("%d\n", ans);


    

    return 0;
}
