#include<iostream>
#include<algorithm>
using namespace std;
int n, nstd, narr[5], p, ntre, nfr, minch=1000000000, cn;
int main() {
	cin >> n;
	for(int i=0;i<n;i++) {
		cin >> p;
		nstd += p;
		narr[p]++;
	}
	if(nstd > 0 && (nstd < 3 || nstd == 5) )
		cout << -1 << '\n';
	else {
		for(int i=0;i<4;i++)
			if((nstd - 3*i) % 4 == 0) {
				ntre = i;
				break;
			}
		nfr = (nstd - 3*ntre) / 4;
		while(nfr >= 0) {
			cn = max(narr[4] - nfr, 0);
			for(int i=0, j=n - nfr - ntre; i<=4 && j>0;j-= narr[i++])
				cn += i * min(j, narr[i]);
			minch = min(cn, minch);
			nfr -= 3;
			ntre += 4;
		}
		cout << minch << '\n';
	}
	return 0;
}
