#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

int n,sum,tmp,ans;
int s[5];

int main()
{
	scanf("%d",&n);
	for (int i=0;i<n;i++)
	{
		scanf("%d",&tmp);
		s[tmp]++;
		sum+=tmp;
	}
	if (sum==1 || sum==2 || sum==5) puts("-1");
	else
	{
		int nima=min(s[1],s[2]);
		ans+=nima;
		s[1]-=nima;
		s[2]-=nima;
		s[3]+=nima;
		if (s[1])
		{
			ans+=s[1]-s[1]/3;
			if (s[1]==1 && s[3]==0) ans++;
		}
		else if (s[2])
		{
			ans+=s[2]-s[2]/3;
			if (s[2]%3==1 && !s[4]) ans++;
		}
		printf("%d\n",ans);
	}
	return 0;
}
