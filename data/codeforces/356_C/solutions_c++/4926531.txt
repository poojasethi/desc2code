#include <cstdio>
using namespace std;

#define min(x, y) ((x) < (y) ? (x) : (y))
int n, i, x, sum, s[5], s1[5], ss, ans, p[5];

int main() {
	scanf("%d", &n);
	for (i = 1; i <= n; i++) {
		scanf("%d", &x);
		s[x]++;
		sum += x;
	}
	ans = 1000000000;
	for (s1[4] = 0; 4 * s1[4] <= sum; s1[4]++)
		if ((sum - 4 * s1[4]) % 3 == 0) {
			s1[3] = (sum - 4 * s1[4]) / 3;
			s1[0] = n - s1[3] - s1[4];
			if (s1[0] < 0)	continue;
			for (ss = 0, i = 1; i <= 4; i++)	p[i] = s1[i - 1] - s[i - 1] + p[i - 1];
			for (ss = 0, i = 1; i <= 4; i++)	if (p[i] > 0)	ss += p[i];
			ans = min(ans, ss);
		}
	if (ans == 1000000000)	printf("-1\n");
	else	printf("%d\n", ans);
}
