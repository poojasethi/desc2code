#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>

using namespace std;

int cnt[5], sum=0, n, ans=0;

int main()
{
    scanf("%d",&n);
    for(int i=1; i<=n; i++)
    {
        int x;
        scanf("%d",&x);
        sum += x;
        cnt[x]++;
    }

    if( sum == 1 || sum == 2 || sum == 5 ) {printf("-1\n"); return 0; }
    ans = min(cnt[1],cnt[2]);
    cnt[1] -= ans;
    cnt[2] -= ans;
    if( cnt[1] > 0)
    {
        ans += cnt[1] - (cnt[1]/3);
        if( cnt[1] == 1 && cnt[3] == 0 ) ans++;
    }
    if( cnt[2] >0 )
    {
        ans += cnt[2] - (cnt[2]/3);
        if( cnt[2] % 3 == 1 && cnt[4] == 0 ) ans++;
    }

    printf("%d\n",ans);

    return 0;
}
