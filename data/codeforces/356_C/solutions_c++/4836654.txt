#include <cstdio>
#include <cstdlib>
#include <algorithm>

using namespace std;

int sum,n;
int S[5]={0};

int main()
{
  scanf("%d",&n);
  sum=0;
  for (int i=1;i<=n;i++)
  {
    int a; scanf("%d",&a);
    sum+=a; S[a]++;
  }
  if (sum==1||sum==2||sum==5)
  {
    printf("-1\n");
    return 0;
  }
  int ans=0;
  ans=min(S[1],S[2]);
  if (S[1]>S[2])
  {
    int tmp=S[1]-S[2];
    ans+=2*(tmp/3); 
    S[3]+=2*(tmp/3); tmp%=3;
    if (tmp==2) ans+=2;
    if (tmp==1) if (S[3]) ans+=1; else ans+=2;
  }
  if (S[2]>S[1])
  {
    int tmp=S[2]-S[1];
    ans+=2*(tmp/3);
    S[3]+=2*(tmp/3); tmp%=3;
    if (tmp==2) ans+=2;
    if (tmp==1) if (S[4]) ans+=1; else ans+=2;
  }  
  printf("%d\n",ans);
  return 0;
}
