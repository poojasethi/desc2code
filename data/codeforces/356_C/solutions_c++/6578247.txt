#include <cstdio>
#include <cstring>

int n;
int d[6];

int main() {
	scanf("%d", &n); 
	int tot = 0, x = 0;
	for (int i = 1; i <= n; ++i) {
		scanf("%d", &x);
		tot += x;
		++d[x]; 
	}
	if (tot < 6  &&  tot != 4  &&  tot != 3)  {
		puts("-1");
		return 0;
	}
	int k = 0;
	if (d[2] > d[1])  k = 2;  
	else  k = 1;
	for (int i = 0; i < 5; ++i) {
	//	printf("%d ", d[i]);
	} 
	int res = 0, z = d[3 - k];
	res += z, d[k] -= z;  d[3] += z;
	res += d[k] / 3 * 2, d[3] += d[k] / 3 * k, d[k] %= 3;
	if (d[k] == 1  &&  d[k + 2])  ++ res;
	else  if (d[k])  res += 2;
	printf("%d\n", res);
	return 0;
}
