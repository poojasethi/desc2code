#include <iostream>

using namespace std;

int n;
int a[5] = {0};
int count;

int main() {
    cin >> n;
    for (int i = 0; i < n; i++) {
        int t;
        cin >> t;
        a[t]++;
    }
    if (a[1] == 0 && a[2] == 0) {
        cout << "0";
        return 0;
    }
    int tot = 0;
    for (int i = 0; i < 5; i++) {
        tot += i * a[i];
    }
    if (tot < 3 || tot == 5) {
        cout << "-1";
        return 0;
    }
    int s3, s4;
    if (a[4] > 0) s3 = (tot - a[4] + 1) / 3;
    else s3 = tot / 3;
    s4 = tot - s3 * 3;
    if (s4 > s3) {
        s4 -= 3;
        s3++;
    }
    //cout << s4 << " " << s3 << "\n";
    cout << max(0, s4 - a[4]) + s3 - a[3] - a[4] + max(s3 - a[3] - a[4] - a[2], 0);
    return 0;
}
