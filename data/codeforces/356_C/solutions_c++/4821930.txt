#include <stdio.h>

int main (void) {
	int n, tmp, i;
	long long cnt[5] = {0,0,0,0,0}, ans = 0, g, r, sum;
	scanf ("%d", &n);
	for (i = 0; i < n; i++) {
		scanf ("%d", &tmp);
		cnt[tmp]++;
	}
	sum = cnt[1]+2*cnt[2]+3*cnt[3]+4*cnt[4];
	if (sum == 1 || sum == 2 || sum == 5)	ans = -1;
	else {
		if (cnt[1] <= cnt[2]) {
			ans += cnt[1];
			cnt[3] += cnt[1];
			cnt[2] -= cnt[1];
			cnt[1] = 0;
		} else {
			ans += cnt[2];
			cnt[3] += cnt[2];
			cnt[1] -= cnt[2];
			cnt[2] = 0;
		}
	
		
		g = (cnt[1]+2*cnt[2])/3;
		r = (cnt[1]+2*cnt[2])%3;
		cnt[3] += g;
		if (cnt[1]) {
			if (r == 1 && !cnt[3])	r = 2;
			ans += g*2+r;
		} else if (cnt[2]) {
			if (r == 2 && cnt[4])	r = 1;
			ans += g+r;
		}
	}
	printf ("%Ld\n", ans);
	return 0;
}
