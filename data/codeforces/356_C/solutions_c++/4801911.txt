#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
using namespace std;
int n;
int cnt[5];
int sum;
int main(){
   scanf("%d",&n);
   for(int i = 0 ; i < n ; ++ i ) {
      int k;
      scanf("%d",&k);
      cnt[k]++;
   }
   for(int i  = 0 ; i < 5 ;  ++ i ) sum = sum + i*cnt[i];
   int ans=-1;
   for(int i = 0 ; i <= n-cnt[0] ; ++ i ){
      int mn=3*i;
      int mx=4*i;
      if(mn<=sum && sum <=mx ){
         int c4=sum-mn;
         int x = max(0,cnt[4]-c4);
         int need=n-cnt[0]-i;
         x+=min(cnt[1],need);
         x+=2* (need-min(cnt[1],need) );
         if(ans==-1 || x < ans ) ans=x;
      }
   }
   printf("%d\n",ans);
}
