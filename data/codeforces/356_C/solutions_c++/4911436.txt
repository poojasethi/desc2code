#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int MaxN = 1000010;
const int inf = 1000000000;

int N, a[MaxN], numf, fl, sol, place, extra;

int main (void) {
  sol = inf;
  scanf("%d",&N);
  for (int i = 0; i < N; ++i) {
    scanf("%d",&a[i]);
    if (a[i] == 4)
      ++numf;
    if (a[i] <= 2)
      fl += 3 - a[i];
    place += a[i];
  }
  sort(a, a + N);  

  int take = 0;

  for (int i = 0; i < N && place + take <= (N - i) * 4; ++i) {
    if (take + numf >= fl) 
      sol = min(sol, max(take, fl));      
    
    if (a[i] == 4)
      --numf;
    if (a[i] <= 2)
      fl -= 3 - a[i];
    place -= a[i];
    take += a[i];
  }

  printf("%d\n",sol < inf ? sol : -1);

  return 0;
}
