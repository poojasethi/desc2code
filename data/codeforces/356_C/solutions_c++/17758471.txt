#include <bits/stdc++.h>
using namespace std;

int n,x,a[10],ans;

int main(){
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> n;
	while (n--){
		cin >> x; a[x]++;
	}
	ans=min(a[1],a[2]);
	if (a[1]<=a[2]){
		a[2]-=a[1]; a[3]+=a[1]; a[1]=0;
		a[3]+=2*(a[2]/3); ans+=2*(a[2]/3); a[2]%=3;
		if (a[2]==1){
			if (a[4]) ans++;
			else if (a[3]>1) ans+=2;
			else ans=-1;
		}else if (a[2]==2) ans+=2;
	}else{
		a[1]-=a[2]; a[3]+=a[2]; a[2]=0;
		a[3]+=a[1]/3; ans+=2*(a[1]/3); a[1]%=3;
		if (a[1]==1){
			if (a[3]) ans++;
			else if (a[4]>1) ans+=2;
			else ans=-1;
		}else if (a[1]==2){
			if (a[4]||a[3]>1) ans+=2;
			else ans=-1;
		}
	}
	cout << ans << "\n";
}
