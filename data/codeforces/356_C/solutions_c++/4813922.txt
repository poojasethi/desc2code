#include <cstdio>
#include <algorithm>
using namespace std;
int N,cnt[5],S,ans;
int main(){
	scanf("%d", &N);
	for (int i = 1;i <= N;i++){
		int k;
		scanf("%d", &k);
		S+=k;
		cnt[k]++;
	}
	if (S<3||S==5){
		printf("-1");
		return 0;
	}
	if (cnt[1]==cnt[2]){
		printf("%d",cnt[1]);
		return 0;
	}
	ans = 0;
	int k = min(cnt[1], cnt[2]);
	cnt[1] -= k;
	cnt[2] -= k;
	cnt[3] += k;
	ans += k;
	if (cnt[1]){
		ans += cnt[1] / 3 * 2;
		cnt[3] += cnt[1] / 3;
		cnt[1] %= 3;
		if (cnt[1])
		if (cnt[1] == 1&&cnt[3] > 0)ans++;
		else						ans+=2;
	}
	else{
		ans+=cnt[2] * 2 / 3;
		cnt[3] += cnt[2] * 2 / 3;
		k = cnt[2] * 2 % 3;
		if (k)
		if ((k == 2&&cnt[4] > 0)||(k==1&&cnt[3] > 0))ans++;
		else					ans+=2;
	}
	printf("%d",ans);
	return 0;
}
