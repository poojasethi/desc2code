#include <iostream>
using namespace std;
int main(){
	int n;
	cin >> n;
	int a[5]={};
	for(int i=0;i<n;i++){
		int hoge;
		cin >> hoge;
		a[hoge]++;
	}
	int sum=a[1]+2*a[2]+3*a[3]+4*a[4];
	int ans=10000000;
	if(sum<3 || sum==5){
		cout << -1;
		return 0;
	}
	for(int i=a[3]+a[4];3*i<=sum && i<=n-a[0];i++){
		if(4*i<sum) continue;
		int ta=0,ama=0;
		ta+=i-(a[3]+a[4]);
		ta+=max( 0 , i-(a[2]+a[3]+a[4]) );
		ama=sum-(4*a[4]+3*a[3]+2*(i-a[3]-a[4])-max(0,i-a[4]-a[3]-a[2]));
		int poyo=max(ta,ama);
		ans=min(ans,poyo);
	}
	cout << ans;
	return 0;
}
