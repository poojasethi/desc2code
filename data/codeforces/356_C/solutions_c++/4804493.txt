#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
#include <string.h>
#include <math.h>
using namespace std;
#define MXX 6
int n,cnt[MXX],tot,total;
void Init()
{
	int x;
	scanf("%d",&n);
	for(int i=1;i<=n;i++) {scanf("%d",&x);cnt[x]++;tot+=x;}//
}
void solve()
{
	if(tot<6&&tot!=3&&tot!=4) {printf("-1\n");return ;}
	int K;
	cnt[1]>cnt[2]?K=1:K=2;
	total+=cnt[3-K];cnt[K]-=cnt[3-K];cnt[3]+=cnt[3-K];
	total+=cnt[K]/3*2;cnt[3]+=cnt[K]/3*K;cnt[K]%=3;
	if(K==2)
	  {
		  	if(cnt[2]) {if(cnt[4]>=cnt[2]) total+=cnt[2];else total+=2;}
	  }
    else
      {
      	  	if(cnt[1]==1 &&cnt[3]) total++;
      	  	else if(cnt[1]) total+=2;
	  }
	printf("%d\n",total);
}
int main()
{
	Init();
	solve();
    return 0;
}
