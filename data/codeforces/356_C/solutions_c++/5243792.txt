#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cstdlib>
using namespace std;
int n,x,a[1200000],ans,g,i;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i)scanf("%d",&x),a[x]++;
	//union 1 + 2
	g=min(a[1],a[2]);
	a[1]-=g; a[2]-=g; a[3]+=g;
	ans=g;
	if(a[1]){
		//union 1 + 1 + 1
		g=a[1]/3;
		ans+=g*2;
		a[1]%=3;
		a[3]+=g;
		//union 1 + 1 + 4
		if(a[4] && a[1]==2)ans+=2;
		else if(a[3]>=a[1])ans+=min(a[1],a[3]);
		else if(a[4]>=2)ans+=2;
		else return printf("-1\n"),0;
	}
	if(a[2]){
		//union 2 + 2 + 2
		g=a[2]/3;
		ans+=g*2;
		a[3]+=g*2;
		a[2]%=3;
		if(a[2]==2)ans+=2;
		else if(a[2]==1){
			if(a[4])ans++;
			else if(a[3]>=2)ans+=2;
			else return printf("-1\n"),0;
		}
	}
	printf("%d\n",ans);
}
