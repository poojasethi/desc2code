int main() 
{
  srand(time(NULL));
  #define TASK "text"
  #ifdef HOME
  assert(freopen(TASK ".in", "rt", stdin));
  assert(freopen(TASK ".out", "wt", stdout));
  #endif

  int n, m;
  while (scanf("%d%d", &n, &m) == 2) {
    vvi es = vvi(n);
    for (int i = 0; i < m; i++) {
      int s, t;
      scanf("%d%d", &s, &t);
      --s, --t;
      if (s == t)
        continue;
      es[s].pb(t), es[t].pb(s);
    }

    for (int i = 0; i < n; i++) {
      sort(es[i].begin(), es[i].end());
      es[i].resize(unique(es[i].begin(), es[i].end()) - es[i].begin());
    }
    
    for (int i = 0; i < n; i++) {
      vector<pii> ps(sz(es[i]));
      for (int j = 0; j < sz(es[i]); j++)
        ps[j] = mp(sz(es[es[i][j]]), es[i][j]);
      sort(ps.begin(), ps.end());
      for (int j = 0; j < sz(es[i]); j++)
        es[i][j] = ps[j].second;
    }
    bool found = 0;
    for (int start = 0; start < n; start++) {
      vi used(n, 0);
      bool ok = 1;
      for (int v = start, iter = 0; iter < n - 1; iter++) {
        used[v] = 1;
        ok = 0;
        for (int it = 0; it < sz(es[v]); it++) {
          int u = es[v][it];
          if (used[u])
            continue;
          v = u;
          ok = 1;
          break;
        }
        if (!ok)
          break;
      }
      if (ok) {
        found = 1;
        break;
      }
    }
    printf(found ? "Yes\n" : "No\n");
    //break;
  }
  return 0;
}
