#include <iostream>
#include <fstream>
#include <cstring>
#include <string>
#include <cstdlib>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <bitset>
#include <ctime>
#include <map>
#include <queue>
#include <set>
using namespace std;

const int maxn=20+15;
int n,m,x,y;
int dd[maxn];
bool edge[maxn][maxn];
bool boo[maxn];
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;i++) 
	{
		scanf("%d%d",&x,&y);
	    if (x==y) continue;
	    if (edge[x][y]) continue;
	    dd[x]++;dd[y]++;
	    edge[x][y]=edge[y][x]=true;
	}
	for (int i=1;i<=n;i++)
	{
		int st=i,j;
		memset(boo,false,sizeof(boo));
		boo[st]=true;
		for (j=1;j<n;j++)
		{
			int minx=0,de=0;
			for (int k=1;k<=n;k++)
			 if (edge[st][k] && !boo[k])
			  if (de==0 || minx>dd[k])
			  {
			  	de=k;
			  	minx=dd[k];
			  }
			if (de==0) break;
			st=de;
			boo[st]=true;
		}
		if (j==n) 
		{
			printf("Yes\n");
			return 0;
		}
	}
	printf("No\n");
	return 0;
}
