#include <iostream>
#include <fstream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <cstring>
#include <string>
#include <cctype>
#include <climits>
#include <algorithm>
#define MAXN 25
using namespace std;

int n,m,d[MAXN], cnt;
bool map[MAXN][MAXN], flag[MAXN], ans;

void dfs(int nowv)
{
    int i, v, dmin=LONG_MAX;
    flag[nowv]=true;
    cnt++;
    for(i=1;i<=n;i++)
        if(map[nowv][i] && !flag[i] && ( d[i]<dmin || ( d[i]==dmin && i<v ) ))   dmin=d[i],  v=i;
    if(dmin!=LONG_MAX) dfs(v);
    return;
}

int main()
{
    int i, a,b;
    /*freopen("d:\\1.in","r",stdin);
    freopen("d:\\1.out","w",stdout);*/
    scanf("%d%d",&n,&m);
    memset(d, 0, sizeof(d));
    memset(map, 0, sizeof(map));
    while(m--)
    {
        scanf("%d%d",&a,&b);
        if(a==b)continue;
        if(!map[a][b])
        {
            map[a][b]=map[b][a]=true;
            d[a]++; d[b]++;
        }
    }
    ans=false;
    for(i=1;i<=n;i++)
    {
        memset(flag, 0, sizeof(flag));
        cnt=0;
        dfs(i);
        if(cnt==n)
        {
            ans=true;
            break;
        }
    }
    if(ans) printf("Yes\n");
        else printf("No\n");
    return 0;
}
