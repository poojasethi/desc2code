import sys
n,m,mmin,mmax = [int(x) for x in raw_input().split()]
flagMin = 0
flagMax = 0

a = [int(x) for x in raw_input().split()]
for x in a:
    if (x < mmin) or (x>mmax):
        print "Incorrect"
        sys.exit()
    else:
        if x == mmin: flagMin = 1
        else:
            if x == mmax: flagMax = 1

if (n-m)>=(2-(flagMin+flagMax)): print "Correct"
else:
    print "Incorrect"
