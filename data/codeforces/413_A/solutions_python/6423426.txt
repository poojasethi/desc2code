"""
http://codeforces.com/contest/413/problem/0
"""
import sys

N,M,Min,Max = map(int,raw_input().split())
arr = map(int,raw_input().split())
correct = "Correct"
incorrect = "Incorrect"
ans = correct
for t in arr :
	if not Min <= t <= Max :
		ans = incorrect
if ans == incorrect :
	print ans
	sys.exit(0)
need = 0 
if not Min in arr : 
	need+=1
if not Max in arr : 
	need+=1
if N - M < need : 
	ans = incorrect
print ans
