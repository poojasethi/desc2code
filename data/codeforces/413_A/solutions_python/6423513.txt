import sys
file = sys.stdin

n, m, tmin, tmax = file.readline().split(' ')

n = int(n)
m = int(m)
tmin = int(tmin)
tmax = int(tmax)

t = [int(val) for val in file.readline().split(' ')]


def f():
    ncs = 2
    for tv in t:
        if tv == tmin or tv == tmax:
            ncs -= 1
        if tv < tmin or tv > tmax:
            return False
            break

    if len(t) + ncs <= n:
        return True
    else:
        return False

if f():
    print("Correct")
else:
    print("Incorrect")
