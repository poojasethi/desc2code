n,m,mini,maxi=map(int,raw_input().split())
t=map(int,raw_input().split())
t=sorted(t)
needed=n-m
if t[0]>=mini and t[-1]<=maxi:
    if needed>=2:
        print "Correct"
    elif needed==1:
        if  mini==t[0] or maxi==t[-1]:
            print "Correct"
        else:
            print "Incorrect"
    elif n-m==0 and t[0]==mini and t[-1]==maxi:
        print "Correct"
    else:
        print "Incorrect"
        
else:
    print "Incorrect"
