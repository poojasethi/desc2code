maxi = int(line[3])
line = raw_input().split()
count = 0
flag = True
for i in line:
    x = int(i)
    if x == maxi or x == mini:
        count += 1
    flag &= x <= maxi and x >= mini
if flag and n-m >= 2-count:
    print "Correct"
else:
    print "Incorrect"
