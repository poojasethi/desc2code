n, m, mintemp, maxtemp = map(int, raw_input().split())
mList = map(int, raw_input().split())
newList = list(mList)
if mintemp not in newList:
    newList.append(mintemp)
if maxtemp not in newList:
    newList.append(maxtemp)
if len(newList) <= n:
    if min(newList) == mintemp and max(newList) == maxtemp:
        print "Correct"
    else:
        print "Incorrect"
else:
    print "Incorrect"
