n, m, mi, ma = map(int, raw_input().split())
a = map(int, raw_input().split())
_ma = max(a)
_mi = min(a)
result = "Correct"
if ((_ma > ma) or (_mi < mi)):
    result = "Incorrect"
else:
    t = 2
    if (_ma == ma):
        t -= 1
    if (_mi == mi):
        t -= 1
    if (t > n - m):
        result = "Incorrect"
print result
