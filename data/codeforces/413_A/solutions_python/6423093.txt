n, m, min, max = map(int, raw_input().split())
t = map(int, raw_input().split())

cmin = 10000
cmax = -1

for ti in t:
    if (ti < cmin):
        cmin = ti
    if (ti > cmax):
        cmax = ti
        
if (cmax > max):
    print 'Incorrect'
elif (cmin < min):
    print 'Incorrect'
elif (n - m == 1):
    if (cmax == max) or (cmin == min):
        print 'Correct'
    else:
        print 'Incorrect'
else:
    print 'Correct'
