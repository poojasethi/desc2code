import sys

(n,m,minV,maxV) = sys.stdin.readline().split()
n = int(n)
m = int(m)
minV = int(minV)
maxV = int(maxV)

R = sys.stdin.readline().split()
for i in range(m):
	R[i] = int(R[i])
	
sp = 0
done = 1
if min(R) < minV or max(R) > maxV:
	print "Incorrect"
		
else:
	if minV not in R:
		sp += 1
	if maxV not in R:
		sp += 1
	if n-m < sp:
		print "Incorrect"
	else:
		print "Correct"
