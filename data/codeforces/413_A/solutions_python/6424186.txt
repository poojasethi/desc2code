import sys

line = sys.stdin.readline()
a = line.split()
n = int(a[0])
m = int(a[1])
Min = int(a[2])
Max = int(a[3])

mins = 100
maxs = 1
Incorrect = "Incorrect"
Correct = "Correct"
line = sys.stdin.readline()
a = line.split()
for i in range(0,m):
    t = int(a[i])
    if t<mins:
        mins = t
    if t>maxs:
        maxs = t

if mins < Min or maxs > Max:
        print Incorrect
        
elif n == 1:
    if Min != Max:
        print Incorrect
    elif Min != mins:
        print Incorrect
    else:
        print Correct
elif n - m == 1:
    if Min != mins and Max != maxs:
        print Incorrect
    else:
        print Correct
else:
    print Correct
