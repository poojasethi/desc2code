n, m, m1, m2 = map(int, raw_input().split())
a = map(int, raw_input().split())
s = "Correct"
k1 = k2 = 0
if n==2:
    if (a[0]!=m1 and a[0]!=m2) : s = "Incorrect"
for i in a:
    if (i<m1) or (i>m2): s = "Incorrect"
    if i==m1: k1 = 1
    if i==m2: k2 = 1
if (m+2-(k1+k2)>n): s = "Incorrect"
print s
