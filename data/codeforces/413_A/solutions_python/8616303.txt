n,m,i,a=map(int,raw_input().split())
t=map(int,raw_input().split())
r=[i,a]
ans=['Incorrect','Correct']
chk=1
while 1:
    if i in t:
        t.remove(i)
        m-=1
    elif a in t:
        t.remove(a)
        m-=1
    else:
        break
n-=2
if n<m:
    print ans[0]
    exit()
for x in t:
    if x not in r:
        if n>0 and i<=x<=a:
            r.append(x)
        else:
            chk=0
    n-=1
print ans[chk]
