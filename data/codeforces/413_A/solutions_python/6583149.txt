n, m, mn, mx = map(int, raw_input().split())
ms, mnf, mxf = map(int, raw_input().split()), 0, 0
for x in ms:
	if x < mn or x > mx:
		print 'Incorrect'
		break
	if x == mn:
		mnf += 1
	if x == mx:
		mxf += 1
else:
	f = 2
	if mnf > 0:
		f -= 1
	if mxf > 0:
		f -= 1
	if f > n - m:
		print 'Incorrect'
	else:
		print 'Correct'
