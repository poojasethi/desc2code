import sys,math

if __name__ == '__main__':
	[n,m,mintemp,maxtemp] = map(int,raw_input().split())
	temp = map(int,raw_input().split())
	minf,maxf = min(temp),max(temp)
	remain = n - m
	ans =  1 if minf != mintemp else 0
	ans += 1 if maxf != maxtemp else 0
	remain = -1 if maxf > maxtemp or minf < mintemp else remain
	
	print 'Correct' if remain >= ans else 'Incorrect'
