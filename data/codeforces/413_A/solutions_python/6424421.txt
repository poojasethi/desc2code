n, m, mi, ma =  map(int, raw_input().split())
s = map(int, raw_input().split())
h1 = h2 = False
if mi not in s:
    h1 = True
if ma not in s:
    h2 = True
if n - m == 1 and (h1 and h2):
    print "Incorrect"
    exit()
for x in s:
    if x < mi or x > ma:
        print "Incorrect"
        exit()
print "Correct"
