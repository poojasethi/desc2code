[n, m, minn, maxx] = [int(x) for x in raw_input().strip().split(" ")]
nums = [int(x) for x in raw_input().strip().split(" ")]
real_minn = min(nums)
real_maxx = max(nums)
need = 0
if real_minn < minn or real_maxx > maxx:
    print "Incorrect"
    exit()
if real_minn != minn:
    need += 1
if real_maxx != maxx:
    need += 1
if need <= n - m:
    print "Correct"
else:
    print "Incorrect"
