#include<stdio.h>
main(){
	int n,m,min,max,a[100],f=0;
	scanf("%d%d%d%d",&n,&m,&min,&max);
	int s=m+2;
	for(int i=1;i<=m;i++){
		scanf("%d",&a[i]);
		if(a[i]<min||a[i]>max)
		f=1;
		else if(a[i]==min)
		s--;
		else if(a[i]==max)
		s--;
	}
	if(f!=0||s>n)
		printf("Incorrect\n");
	else
		printf("Correct\n");
		
}
