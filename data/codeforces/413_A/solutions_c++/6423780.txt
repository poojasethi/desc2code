#include <iostream>
using namespace std;

main()
{
	int n,m,min,max;
	cin >> n >> m>> min >> max;
	int a=0,b=0;
	for(int i=0; i<m; i++)
	{
		int c;
		cin >> c;
		if(min == c) a=1;
		if(max == c) b=1;
		if(c < min || c > max) {cout <<"Incorrect"; return 0;}
	}
	a = 2 - a - b;
	if(a > n-m) cout << "Incorrect";
	else cout << "Correct";
}
