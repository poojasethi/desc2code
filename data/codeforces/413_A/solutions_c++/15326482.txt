#include <stdio.h>
int main()
{
	int n,m,min,max,a,i,flag,x;
	scanf("%d%d%d%d",&n,&m,&min,&max);
	flag=0;x=m+2;
	for(a=0;m>a;a++){
		scanf("%d",&i);
		if(i>max||i<min){
			flag=1;
		}
		if(i==max){
			x--;
		}
		if(i==min){
			x--;
		}
	}
	if(flag==1||x>n){
		printf("Incorrect\n");
	}
	else{
		printf("Correct\n");
	}
	return 0;
}
