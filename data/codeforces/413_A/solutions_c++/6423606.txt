#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int n, m, a, b, x, i, c, d;
	cin>>n>>m>>a>>b;
	c=1; d=1;
	for (i=0; i<m; i++) {
		cin>>x;
		if (x<a || x>b) {
			cout<<"Incorrect"<<endl;
			return 0;
		}
		else if (x==a) c=0;
		else if (x==b) d=0;
	}
	if (n-m>=c+d) cout<<"Correct"<<endl;
	else cout<<"Incorrect"<<endl;
	return 0;
}
