#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <ctype.h>
#include <math.h>

int main(){ 
	
	int n,m,mn,mx;
	scanf("%d%d%d%d",&n,&m,&mn,&mx);
	for(int i=0,b=0;i<m;i++){
		scanf("%d",&b);
		if(b<mn||b>mx){
			n=-1;
			break;
		}
		if(mn<b&&mx>b)n--;
	}
	if(n>=2)printf("Correct\n");
	else printf("Incorrect\n");
	return 0;
}
