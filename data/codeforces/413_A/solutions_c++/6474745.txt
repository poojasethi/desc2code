#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,m,mn,mx;
    cin>>n>>m>>mn>>mx;int a[m];
    for(int i=0;i<m;++i) cin>>a[i];
    sort(a,a+m);
    if(mn>a[0] || mx<a[m-1]) {cout<<"Incorrect\n";return 0;}
    int change=0;
    if(a[0]!=mn) change++;
    if(a[m-1]!=mx) change++;
    if(change>(n-m)){cout<<"Incorrect\n";return 0;}
    cout<<"Correct\n";
    return 0;
}
