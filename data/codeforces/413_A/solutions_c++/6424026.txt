#include <cstdio>
using namespace std;

int n,m,x,y,a;
bool mini,maxi;

int main(){
 scanf("%d %d %d %d",&n,&m,&x,&y);
 for(int i=0;i<m;i++){
  scanf("%d",&a);
  if(a<x||a>y){
   puts("Incorrect");
   return 0;
  }
  if(a==x) mini=1;
  if(a==y) maxi=1;
 }
 int hm=2-mini-maxi;
 if(n-m>=hm)
  puts("Correct");
 else
  puts("Incorrect");
 return 0;
}
