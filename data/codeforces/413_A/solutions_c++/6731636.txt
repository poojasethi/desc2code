#include<cstdio>

int main(){
  int n,m,max,min;
  scanf("%d%d%d%d",&n,&m,&min,&max);
  int t1=1,t2=1,t3=0;
  for(int i=1;i<=m;i++){
    int x;scanf("%d",&x);
    if(x<min) t3=1;
    if(x>max) t3=1;
    if(x==min) t1=0;
    if(x==max) t2=0;
  }
  if(t3) printf("Incorrect");
  else if(m+t1+t2>n) printf("Incorrect");
  else printf("Correct");
}
