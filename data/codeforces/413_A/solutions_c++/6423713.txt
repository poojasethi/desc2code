#include<iostream>
using namespace std;
int main() {
	bool xmin=false,xmax=false;
	int n,m,min,max,x;
	cin>>n>>m>>min>>max;
	for(int i=0;i<m;i++) {
		cin>>x;
		if(x==min)xmin=true;
		if(x==max)xmax=true;
		if(x<min||x>max) {
			cout<<"Incorrect";
			return 0;
		}
	}
	if(!xmin)m++;
	if(!xmax)m++;
	if(m>n)cout<<"Incorrect";
	else cout<<"Correct";
	return 0;
}
