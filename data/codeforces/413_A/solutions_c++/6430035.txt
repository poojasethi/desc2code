#include <stdio.h>
int main(){
	int count,n,m,min,max,i,temp;
	count=0;
	scanf("%d%d%d%d",&n,&m,&min,&max);
	for(i=1;i<=m;i++){
		scanf("%d",&temp);
		if(temp<min||temp>max){
			printf("Incorrect\n");
			return 0;
		}
		if(temp==min){
			count++;
		}
		if(temp==max){
			count++;
		}
	}
	if(n-m<2-count){
		printf("Incorrect\n");
	}
	else{
		printf("Correct\n");
	}
	return 0;
}
