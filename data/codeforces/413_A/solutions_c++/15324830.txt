#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
	int n,m,minn,maxx,t,fmin=0,fmax=0,flag=0;
	cin>>n>>m>>minn>>maxx;
	for(int i=0;i<m;i++)
	{
		cin>>t;
		if(t==minn)
			fmin=1;
		if(t==maxx)
			fmax=1;
		if(t<minn||t>maxx)
			flag=1;
	}
	if(!fmin&&!fmax&&n-m<2)
		flag=1;
	if(flag)
		cout<<"Incorrect"<<endl;
	else
		cout<<"Correct"<<endl;
	return 0;
}
