#include<stdio.h>
int main()
{
	int i,j,f=0,n,min,max,m,a[1000],freq[10000];
	scanf("%d %d %d %d",&n,&m,&min,&max);
	
	for(i=0;i<m;i++)
	{
		scanf("%d",&a[i]);
		if(a[i]>=min&&a[i]<=max){;}
		else f=1;
		freq[a[i]]++;
	}
	if(f){printf("Incorrect\n");return 0;}
	j=n-m;
	if(!freq[min]){freq[min]++;j--;}
	if(!freq[max]){freq[max]++;j--;}
	if(j<0){printf("Incorrect\n");return 0;}
	else printf("Correct\n");
	return 0;
}
