#include<stdio.h>

int x,arr[1000],i,j,k,tmin,tmax,sayac,n,m,max,min;
int main(){
	scanf("%d %d %d %d",&n,&m,&min,&max);
	int t=0;
	for(i=1;i<=m;i++){
		scanf("%d",&x);
		if(x>max||x<min){
			t=1;
		}
		if(x==max&&!tmax)
			tmax++;
		else if(x==min&&!tmin)
			tmin++;

		else sayac++;
	}
	sayac+=!tmax+!tmin;
	if(t||sayac>n)
		printf("Incorrect");
	
	else
		printf("Correct");

	return 0;
}
