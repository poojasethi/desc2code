#include <bits/stdc++.h>
using namespace std;

int main() {
	//freopen("a.in","r",stdin);
	int n, m, a, b, dif, f = 2;
	cin >> n >> m >> a >> b;
	dif = n - m;
	for (int i = 0; i < m; ++i) {
		int num;
		cin >> num;
		if (num > b || num < a) {
			puts("Incorrect");
			return 0;
		}
		if (num == a || num == b)
			--f;
	}
	if(dif < f)puts("Incorrect");
	else puts("Correct");
	return 0;
}
