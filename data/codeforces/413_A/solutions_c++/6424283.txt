#include<iostream>
#include<algorithm>
using namespace std;
int m,n,minn,maxx;
int a[200];
int main()
{
	cin>>n>>m>>minn>>maxx;
	for(int i=0;i<m;i++)
	{
		cin>>a[i];
	}
	sort(a,a+m);
	if(a[0]<minn||a[m-1]>maxx)
		cout<<"Incorrect"<<endl;
	else
		if(n-m>=2)
			cout<<"Correct"<<endl;
		else
				if((a[0]==minn||a[m-1]==maxx)&&n-m==1)
					cout<<"Correct"<<endl;
			else
				cout<<"Incorrect"<<endl;
}
