#include<bits/stdc++.h>

using namespace std;

const int N = 1e5 + 7;

int a[N];
int n, minl, maxl, m;

int main(){
    scanf("%d%d%d%d", &n, &m, &minl, &maxl);
    for (int i = 1; i <= m; i++)
        scanf("%d", &a[i]);

    sort(a + 1, a + m + 1);
    puts(a[1] >= minl && a[m] <= maxl && (n - m) >= ((a[1] != minl) + (a[m] != maxl)) ? "Correct" : "Incorrect");
    return 0;
}
