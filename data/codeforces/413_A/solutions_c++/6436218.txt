#include <iostream>
#include <algorithm>

using namespace std;

int main() {
	int n, m, mn, mx, x, tmn = 1000, tmx = -1;
	cin >> n >> m >> mn >> mx;
	for (int i = 0; i < m; i++) {
		cin >> x;
		tmn = min(tmn, x);
		tmx = max(tmx, x);
	}
	if (tmn >= mn && tmx <= mx && ((int)(tmn != mn) + (tmx != mx) <= n - m))
		cout << "Correct" << endl;
	else
		cout << "Incorrect" << endl;
	return 0;
}
