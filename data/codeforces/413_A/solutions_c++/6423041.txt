#include <stdio.h>
#include <stdlib.h>

int n,m,mint,maxt,minn,maxn,i,x;

int main()
{
	scanf("%d%d%d%D",&n,&m,&mint,&maxt);
	minn=100000000;
	maxn=-1;
	for (i=1; i<=m; i++)
	{
		scanf("%d",&x);
		if (x>maxn)
			maxn=x;

		if (x<minn)
			minn=x;
	}
	
	if (minn>=mint && maxn<=maxt && (n-2>=m || minn==mint || maxn==maxt))
		printf("Correct\n");
	else
		printf("Incorrect\n");

	return 0;
}
