#include <iostream>
using namespace std;
int n,m,i,a,b,flag1,flag2,x;
int main() 
{
	cin>>n>>m>>a>>b;
	for (i=0; i<m; i++)
	{
		cin>>x;
		if (x<a)
		{
			cout<<"Incorrect"<<endl;
			return 0;
		}
		if (x==a) flag1=1;
		if (x==b) flag2=1;
		if (x>b)
		{
			cout<<"Incorrect"<<endl;
			return 0;
		}
	}
	if (m+1-flag1+1-flag2<=n) cout<<"Correct"<<endl;
	else cout<<"Incorrect"<<endl;
	return 0;
}
