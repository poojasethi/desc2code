#include <stdio.h>
int main ( ) {
	int n,m,min,max;
	while (scanf("%d%d%d%d",&n,&m,&min,&max)!=EOF) {
		int flag=1,s=2,val;
		for (int i=0;i<m;i++) {
			scanf("%d",&val);
			if (val<min||val>max)
				flag=0;
			if (val==min)
				s--;
			if (val==max)
				s--;
		}
		if (!flag||n-m<s) {
			printf("Incorrect\n");
			continue;
		}
		printf("Correct\n");
		
	}
	return 0;
}
