#include<iostream>
using namespace std;

int main()
{ int n,m,min,max,i,k,flag,a[100];
  cin>>n>>m>>min>>max;
  for(i=1;i<=m;i++) cin>>a[i];
  if(min==max) k=1;
  else k=2;
  flag=1;
  for(i=1;i<=m;i++)
  { if(a[i]>=min && a[i]<=max)
    { if(a[i]>min && a[i]<max) k++; }
    else {flag=0; break;}
  }
  if(k>n) flag=0;
  if(flag) cout<<"Correct";
  else cout<<"Incorrect";
  return 0;
}
