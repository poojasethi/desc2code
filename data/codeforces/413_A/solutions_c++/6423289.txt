#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
int i,n,m,min,max,a[109];
cin>>n;
cin>>m;
cin>>min;
cin>>max;
for(i=0;i<m;i++)cin>>a[i];
int flag=1;
sort(a,a+m);
if(a[0]<min)flag=0;
if(a[m-1]>max)flag=0;
if((n==m) && (a[0]>min || a[m-1]<max))flag=0;
if((n==(m+1)) && (a[0]>min && a[m-1]<max))flag=0;
if(flag)cout<<"Correct"<<endl;
else cout<<"Incorrect"<<endl;
return 0;
}
