#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int c=0,n,m,mn,mx;
    cin >> n >> m >> mn >> mx;
    int s[m];

    for(int i=0 ; i<m; i++)
        cin >> s[i];

    sort(s,s+m);

    if(s[0]<mn || s[m-1]>mx)
        cout << "Incorrect";
    else if(s[0]==mn || s[m-1]==mx || n-m>=2)
        cout << "Correct";
    else cout << "Incorrect";

}
