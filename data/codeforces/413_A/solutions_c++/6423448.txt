#include <bits/stdc++.h>
using namespace std;
int a,b,n,m,aa=0,bb=101,nn,t;
int main() {
    cin>>n>>m>>b>>a;
    nn=m;
    for(int i=0;i<m;i++) {
        scanf("%d",&t);
        aa=max(aa,t);
        bb=min(bb,t);
    }
    if (bb>b) nn++;
    if (aa<a) nn++;
    if (bb<b || aa>a || nn>n) puts("Incorrect");
    else puts("Correct");
    return 0;
}
