#include<cstdio>
int tmp[101]={0};
int main () {
	int a,b,c,d;
	int res=0;
	scanf("%d%d%d%d",&a,&b,&c,&d);
	for(int i=0;i<b;i++) {
		int n;
		scanf("%d",&n);
		if(n<c||n>d) {
			printf("Incorrect");
			return 0;
		}
		if(tmp[n]==0) res++;
		tmp[n]++;
	}
	if(tmp[c]==0) { b++; res++; }
	if(tmp[d]==0) { b++; res++; }
	if(b>a) printf("Incorrect");
	else printf("Correct");
	return 0;
}
