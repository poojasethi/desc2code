#include<cstdio>
using namespace std;
int n,m,min,max,i,t,put;
int main()
{
  scanf("%d%d%d%d",&n,&m,&min,&max);
  for (i=1;i<=m;i++)
  {
    scanf("%d",&t);
    if (t<min||t>max) {printf("Incorrect");return 0;}
    if (t==min) put++;
    if (t==max) put++;
  }
  if (n-m>=2) {printf("Correct");return 0;}
  if (n-m<2-put) printf("Incorrect");else printf("Correct");
  return 0;
}
