#include <bits/stdc++.h>

using namespace std;
int main() {

    int n,m,mn,mx ;
    cin >> n >> m >> mn >> mx ;
    bool bn=0 , bx=0 ;
    int tmp ;
    while (cin >> tmp)
    {
        bn|=tmp==mn ;
        bx|=tmp==mx ;
        if (tmp<mn or tmp>mx)
            goto r;
    }
    if (n-m>=!bn+!bx)
        cout << "Correct" ;
    else r:cout << "Incorrect" ;

}
