#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n,m,mi,ma,x,xx=2;
int main()
{
	scanf("%d%d%d%d",&n,&m,&mi,&ma);
	for(int i=1;i<=m;i++){scanf("%d",&x);if(x<mi || x>ma)
	{printf("Incorrect\n");return 0;}if(x==mi || x==ma)xx--;}
	printf("%s\n",xx>n-m?"Incorrect":"Correct");return 0;
}
