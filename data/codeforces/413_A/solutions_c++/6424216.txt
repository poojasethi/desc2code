#include <cstdio>
int main()
{
    int m,n,min,max;
    scanf("%d %d %d %d",&n,&m,&min,&max);
    int a[m];
    int b=2;
    int fl=0;
    for(int i=0; i<m; i++)
    {
        scanf("%d",&a[i]);
        if(a[i]<min || a[i]>max) fl=1;
        if(a[i]==min)b--;
        else if(a[i]==max) b--;
    }
    if(b>n-m || fl) printf("Incorrect");
    else printf("Correct");
    return 0;
}
