#include<stdio.h>
int n,m,i,ma,mi,mt,sg=1,av1=0,av2=0;
int main()
{
	scanf("%d %d %d %d",&n,&m,&mi,&ma);
	for(i=0;i<m;i++)
	{
		scanf("%d",&mt);
		if(mt>ma||mt<mi) {sg=0;continue;}
		if(mt==ma) av2=1;
		if(mt==mi) av1=1;
	}
	if(sg&&2-av1-av2<=n-m) printf("Correct\n");
	else printf("Incorrect\n");
	return 0;
}
