#include<stdio.h>
int n,m,mn,mx,i,d1,d2,x;
int main() {

	scanf("%d %d %d %d",&n,&m,&mn,&mx);

	for(i = 1 ; i<=m ; i++){
		scanf("%d",&x);
		if(x < mn || x > mx){
			puts("Incorrect");
			return 0;
		}
		if(x == mn) d1=1;
		if(x == mx) d2=1;
	}
	if(n-m >=2-d1-d2)
		puts("Correct");
	else
		printf("Incorrect");
	return 0;
}
