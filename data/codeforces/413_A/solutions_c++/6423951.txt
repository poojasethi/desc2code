#include <iostream>
using namespace std;

int main() 
{
    int i,m,n,x,y,up=0,low=101,a;
    cin>>n>>m>>x>>y;
    for(i=1;i<=m;i++)
    {
        cin>>a;
        up=max(up,a);
        low=min(low,a);
    }
    if(up>y||low<x)
    cout<<"Incorrect";
    else
    {
        if(up==y||low==x) m++;
        else m+=2;
        if(m>n) cout<<"Incorrect";
        else cout<<"Correct";
    }
}
