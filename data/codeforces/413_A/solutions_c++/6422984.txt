#include<iostream>
#include<algorithm>
using namespace std;
//A
const int oo=110;
int n,m,ma,mi,a[oo],p;

int main(){
  cin>>n>>m>>mi>>ma;
  for(int i=0;i<m;i++)cin>>a[i];
  sort(a,a+m);
  if(a[0]<mi || a[m-1]>ma){
    cout<<"Incorrect";
    return 0;
  }
  if(a[0]!=mi)
    p++;
  if(a[m-1]!=ma)
    p++;
  if(p+m>n){
    cout<<"Incorrect";
    return 0;
  }
  cout<<"Correct";
  return 0;
}
