#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
int n,m,min,max;
scanf("%d %d %d %d",&n,&m,&min,&max);
int a[m];
for(int i=0;i<m;i++)
scanf("%d",a+i);
int size=m;
sort(a,a+m);
if(min!=a[0])
size++;
if(max!=a[m-1])
size++;
if(m>n||min>max||min>a[0]||max<a[m-1]||n<size)
printf("Incorrect");
else
printf("Correct");		
return 0;
}
