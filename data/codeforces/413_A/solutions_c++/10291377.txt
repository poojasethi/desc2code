#include <iostream>
#include <cmath>

using namespace std;

int main()
{
  int n,m,mini,maxi;
  cin>>n>>m>>mini>>maxi;
  int minimo=1000;
  int maximo=0;
  for (int i=0;i<m;i++) {
    int x;
    cin>>x;
    minimo=min(minimo,x);
    maximo=max(maximo,x);
  }
  if (minimo<mini or maximo>maxi or (minimo!=mini and maximo!=maxi and n==m+1))
    cout<<"Incorrect"<<endl;
  else
    cout<<"Correct"<<endl;
}
