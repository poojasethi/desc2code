#include <iostream>

using namespace std;



int main()
{
int n,m,min,max,k=0;
cin>>n>>m>>min>>max;
int x[m];
for (int i=0;i<m;i++){
    cin>>x[i];
if (x[i]>max){cout<<"Incorrect";return 0;}
if (x[i]<min){cout<<"Incorrect";return 0;}

}
k=m;
for (int i=0;i<m;i++){
  if (x[i]==min){k--;}
  if (x[i]==max){k--;}

}

if (k+2<=n){cout<<"Correct";}
else cout<<"Incorrect";


}
