#include <stdio.h>
int main()
{
	int n,m,min,max,ti;
	int i,l,h,res;
	scanf("%d%d%d%d",&n,&m,&min,&max);
	l = 111;
	h = -1;
	for(i = 1;i <= m;++i)
	{
		scanf("%d",&ti);
		if(ti < l) l = ti;
		if(ti > h) h = ti;
	}
	if(l < min || h > max)
	{
		puts("Incorrect");
		return 0;
	}
	res = (l > min) + (h < max);
	if(n - m >= res) puts("Correct");
	else puts("Incorrect");
	return 0;
}
