#include <cstdio>
#include <algorithm>
#include <queue>
#include <vector>

using namespace std;

#define maxn (300000)
#define PB push_back
#define MP make_pair
#define A first
#define B second

typedef pair<int, int> PII;
typedef long long LL;

vector<PII> g[maxn];

priority_queue< PII, vector<PII>, greater<PII> > f;
int a[maxn], n, m;

int main() {
    scanf("%d%d", &n, &m); LL ans = 0ll;
    for (int i = 1; i <= n; ++i) scanf("%d", &a[i]);
    for (int i = 1; i <= m; ++i) {
        int x, y, v; scanf("%d%d%d", &x, &y, &v);
        g[x].PB(MP(y, v)), ans += v;
    }

    for (int i = 1; i <= n; ++i) {
        for (int j = 0; j < g[i].size(); ++j) f.push(g[i][j]);
        while (!f.empty() && f.top().A < i) f.pop();

        while (!f.empty() && a[i]) {
            PII now = f.top(); f.pop();
            int v = min(a[i], now.B);
            now.B -= v, a[i] -= v, ans -= v;
            if (now.B) f.push(now);
        }
    }

    printf("%I64d", ans);

    return 0;
}
