#include <iostream>
#include <map>
#include <cmath>
#include <algorithm>
#include <cstring>
#include <iomanip>
#include <queue>
#include <vector>
#include<cstdio>
#define LL long long
const int maxn=200008;
#define REP(i,a,n) for(int i=a;i<=n;i++)
using namespace std;
struct node
{
    int r;
    int p;
    bool operator<(const node &b)const
    {
        return b.r<r;
    }
}k;
priority_queue<node>q;
vector<node>v[maxn];
int a[maxn];
int main()
{
   int n,m;
   LL ans;
   while(cin>>n>>m)
   {
       while(!q.empty())q.pop();
       memset(v,0,sizeof(v));
       ans=0;
       int s;
       REP(i,1,n)cin>>a[i];
       REP(i,1,m)
       {
           cin>>s>>k.r>>k.p;
           v[s].push_back(k);
           ans+=k.p;
       }
       REP(i,1,n)
       {
          for(int j=0;j<v[i].size();j++)
          q.push(v[i][j]);
          while(!q.empty()&&q.top().r<i)q.pop();
          while(!q.empty()&&a[i])
          {
              node t=q.top();
              q.pop();
              int d=min(t.p,a[i]);
              t.p-=d;
              ans-=d;
              a[i]-=d;
              if(t.p)q.push(t);

          }
       }
       cout<<ans<<endl;
   }
   return 0;
   }
