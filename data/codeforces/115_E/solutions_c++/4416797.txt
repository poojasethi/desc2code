#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
#include<vector>
#include<iostream>
#define ll long long

using namespace std;

const int maxn = 200001;

int data[maxn];

struct node
{
    int r,val;
    bool operator<(const node &a) const
    {
        return r > a.r;
    }
}ji;

vector<node> v[maxn];
priority_queue<node> q;

int main()
{
    int n,m,a;
    scanf("%d%d",&n,&m);
    ll res = 0;
    for(int i=1;i<=n;i++)
    {
        scanf("%d",data+i);
    }
    for(int i=1;i<=m;i++)
    {
        scanf("%d%d%d",&a,&ji.r,&ji.val);
        res += ji.val;
        v[a].push_back(ji);
    }
    for(int i=1;i<=n;i++)
    {
        for(int j=0;j<v[i].size();j++)
        {
            q.push(v[i][j]);
        }
        while(!q.empty()&&q.top().r<i)
            q.pop();
        while(!q.empty()&&data[i])
        {
            ji = q.top();
            q.pop();
            int st = min(data[i],ji.val);
            res -= st;
            ji.val -= st;
            data[i] -= st;
            if(ji.val)
                q.push(ji);
        }
    }
    cout << res <<endl;
}
