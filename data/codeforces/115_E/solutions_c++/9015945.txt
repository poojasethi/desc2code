#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 200010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int x,y,z;
} a[N];
int i,j,k,n,m;
int b[N];
int64 tr[N*4],cov[N*4];
inline bool cc1(const ww &a,const ww &b) {
	return a.y<b.y;
}
inline void update(int x) {
	int y=x*2,z=x*2+1;
	tr[x]=max(tr[y],tr[z]);
}
inline void Inc(int x,int64 y) {
	tr[x]+=y;
	cov[x]+=y;
}
inline void renew(int x) {
	if (cov[x]) Inc(x*2,cov[x]),Inc(x*2+1,cov[x]);
	cov[x]=0;
}
void inc(int q,int x,int y,int k,int t) {
	if (y<=k) {
		Inc(q,t);
		return;
	}
	renew(q);
	int mid=(x+y)/2;
	inc(q*2,x,mid,k,t);
	if (mid<k) inc(q*2+1,mid+1,y,k,t);
	update(q);
}
void gai(int q,int x,int y,int k,int64 t) {
	if (x==y) {
		tr[q]=max(tr[q],t);
		return;
	}
	renew(q);
	int mid=(x+y)/2;
	if (k<=mid) gai(q*2,x,mid,k,t);
	else gai(q*2+1,mid+1,y,k,t);
	update(q);
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n) scanf("%d",&b[i]);
	For(i,1,m) scanf("%d%d%d",&a[i].x,&a[i].y,&a[i].z);
	sort(a+1,a+m+1,cc1);
	j=1; n++;
	For(i,1,n-1) {
		gai(1,1,n,i,tr[1]);
		inc(1,1,n,i,-b[i]);
		for (;j<=m&&a[j].y==i;j++) inc(1,1,n,a[j].x,a[j].z);
		gai(1,1,n,n,tr[1]);
	}
	printf("%I64d\n",tr[1]);
	return 0;
}
