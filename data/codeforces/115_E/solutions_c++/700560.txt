#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
#define N 200005
using namespace std;

struct seg{
	int x, y, s;
	bool operator < (seg const &T) const{return x < T.x;}
};

struct dat{
	int x, s;
	dat(){}
	dat(int x, int s):x(x), s(s){}
	bool operator < (dat const &T) const{return x > T.x;}
};

int n, m, x, r[N];
seg c[N];
priority_queue<dat> q;
long long ret;
dat tmp;

int main(){
	scanf("%d%d", &n, &m);
	for (int i=1; i<=n; i++) scanf("%d", &r[i]);
	for (int i=0; i<m; i++){
		scanf("%d%d%d", &c[i].x, &c[i].y, &c[i].s);
		ret += c[i].s;
	}
	sort(c, c+m);
	
	for (int i=1, j=0; i<=n; i++){
		while (j<m && c[j].x == i) q.push(dat(c[j].y, c[j].s)), j++;
		while (!q.empty() && q.top().x<i) q.pop();
		
		while (!q.empty() && r[i]>0){
			tmp = q.top();
			q.pop();
			x = min(tmp.s, r[i]);
			tmp.s -= x;
			r[i] -= x;
			ret -= x;
			if (tmp.s > 0) q.push(dat(tmp.x, tmp.s));
		}
	}

	printf("%I64d\n", ret);
	return 0;
}
