#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<vector>
#include<queue>
using namespace std;
const int N=200000+10;
typedef pair<int,int >PII;
vector<PII> adj[N];
int val[N];
priority_queue<PII,vector<PII>,greater<PII> >Q;
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)scanf("%d",&val[i]);
    long long sum=0;
    for(int i=1;i<=m;i++)
    {
        int x,y,z;
        scanf("%d%d%d",&x,&y,&z);
        adj[x].push_back(make_pair(y,z));
        sum+=z;
    }
    for(int i=1;i<=n;i++)
    {
        while(!Q.empty()&&Q.top().first<i)Q.pop();
        for(int j=0;j<adj[i].size();j++)Q.push(adj[i][j]);
        while(!Q.empty()&&val[i])
        {
            PII T=Q.top();
            Q.pop();
            int tmp=min(T.second,val[i]);
            T.second-=tmp,val[i]-=tmp,sum-=tmp;
            if(T.second)Q.push(T);
        }
    }
    cout<<sum<<endl;
    return 0;
}
