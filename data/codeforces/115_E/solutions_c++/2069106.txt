#include<cstdio>
#include<cstdlib>
#include<vector>
#include<set>
#include<algorithm>
using namespace std;
int n,m,a1,a2;
long long a[300001],a3,ans = 0;
multiset < pair<int,long long> > S1;
vector < pair<int,long long> > X[300001];
pair<int ,long long> P1;
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++) scanf("%I64d",&a[i]);
    for(int i=0;i<m;i++) scanf("%d%d%I64d",&a1,&a2,&a3), X[a1].push_back( make_pair(a2,a3) ),ans += a3;
    for(int i=1;i<=n;i++){
        for(int j=0;j<X[i].size();j++) S1.insert( X[i][j] );
        while(!S1.empty() && S1.begin()->first < i) S1.erase(S1.begin());        
        while(!S1.empty() && a[i]){
             P1 = *S1.begin(); S1.erase(S1.begin());
             if(P1.second > a[i]){
                   P1.second -= a[i]; ans -= a[i];  a[i] = 0; S1.insert(P1);       
             }       
             else a[i] -= P1.second, ans -= P1.second;
        }    
    }
    printf("%I64d\n",ans);
    //system("pause");
    return 0;
}
