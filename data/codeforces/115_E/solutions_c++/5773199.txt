#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<vector>
#include<cmath>

#define PII pair<ll,ll>
#define F first
#define S second
#define MP make_pair
#define PB push_back

using namespace std;

typedef long long ll;

const ll maxn=2*1000*100+100;

ll n,m,a[maxn],dif[4*maxn],mx[4*maxn],dp[maxn];

vector<PII>v[maxn];

void diff(ll x){
	mx[2*x]+=dif[x];
	mx[2*x+1]+=dif[x];
	dif[2*x]+=dif[x];
	dif[2*x+1]+=dif[x];
	dif[x]=0;
}

void update(ll x,ll y,ll w,ll l=0,ll r=n+1,ll cnt=1){
	if(y<l || r<x)
		return;
	if(x<=l && r<=y){
		dif[cnt]+=w;
		mx[cnt]+=w;
		return;
	}
	diff(cnt);
	ll med=(l+r)/2;
	update(x,y,w,l,med,2*cnt);
	update(x,y,w,med+1,r,2*cnt+1);
	mx[cnt]=max(mx[2*cnt],mx[2*cnt+1]);
}

ll maxx(ll x,ll l=0,ll r=n+1,ll cnt=1){
	if(x<l)
		return 0;
	if(r<=x)
		return mx[cnt];
	ll med=(l+r)/2;
	diff(cnt);
	return max(maxx(x,l,med,2*cnt),maxx(x,med+1,r,2*cnt+1));
}

int main(){
	ios::sync_with_stdio(false);
	cin>>n>>m;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	for(int i=1;i<=m;i++){
		ll l,r,w;
		cin>>l>>r>>w;
		v[r].PB(MP(l,w));
	}
	for(int i=1;i<=n+1;i++){
		dp[i]=max(maxx(i-1),0ll);
		update(i,i,dp[i]);
		for(int j=0;j<v[i].size();j++){
			ll l=v[i][j].F,w=v[i][j].S;
			update(0,l-1,w);
		}
		update(0,i-1,-a[i]);
	}
	cout<<dp[n+1]<<endl;
	return 0;
}
