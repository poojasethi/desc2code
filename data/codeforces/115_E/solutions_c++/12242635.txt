#define N (1<<18)
#define PII pair<int,int>
#include <bits/stdc++.h>
using namespace std;

vector<PII> g[N];
int n,m,x,y,z,a[N];
long long t[N<<2],lz[N<<2];

void relax(int i)
{
	lz[i<<1]+=lz[i];
	lz[(i<<1)+1]+=lz[i];
	t[i<<1]+=lz[i],t[(i<<1)+1]+=lz[i];
	lz[i]=0;
}

void update(int x,long long v,int l,int r,int i)
{
	if(l==r)
	{
		t[i]=v;
		lz[i]=0;
	}
	else
	{
		relax(i);
		int mid=(l+r)>>1;
		if(x<=mid)
			update(x,v,l,mid,i<<1);
		else
			update(x,v,mid+1,r,(i<<1)+1);
		t[i]=max(t[i<<1],t[(i<<1)+1]);
	}
}

void update(int tl,int tr,int l,int r,int i,int v)
{
    if(tl>r || tr<l) return;
    if(tl<=l && r<=tr)
	{
		t[i]+=v;
		lz[i]+=v;
	}
	else
	{
		relax(i);
		int mid=(l+r)>>1;
		update(tl,tr,l,mid,i<<1,v);
		update(tl,tr,mid+1,r,(i<<1)+1,v);
		t[i]=max(t[i<<1],t[(i<<1)+1]);
	}
}

int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++)
		scanf("%d",a+i);
	for(int i=1;i<=m;i++)
		scanf("%d%d%d",&x,&y,&z),g[y].push_back(PII(x,z));

	for(int i=1;i<=n;i++)
	{
		update(0,t[1],0,n,1);
		update(i,t[1],0,n,1);
		update(1,i,0,n,1,-a[i]);
		for(int j=0;j<g[i].size();j++)
			update(1,g[i][j].first,0,n,1,g[i][j].second);
	}

	cout<<t[1]<<endl;
}
