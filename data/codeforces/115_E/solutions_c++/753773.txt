#include <cstdio>
#include <algorithm>
#include <vector>
#include <queue>
using namespace std;

typedef pair<int, int> Pair_i_i;

int n, m;
int c[200010];
vector<Pair_i_i> g[200010];
priority_queue<Pair_i_i, vector<Pair_i_i>, greater<Pair_i_i> > q;

int main() {
    scanf("%d%d", &n, &m);
    for (int i=1; i<=n; ++i)
        scanf("%d", &c[i]);
    long long ans=0;
    for (int i=0, x, y, v; i<m; ++i) {
        scanf("%d%d%d", &x, &y, &v);
        g[x].push_back(make_pair(y, v));
        ans+=v;
    }
    
    for (int i=1; i<=n; ++i) {
        for (int j=0; j<g[i].size(); ++j)
            q.push(g[i][j]);
        while (!q.empty() && q.top().first<i)
            q.pop();
        while (!q.empty() && c[i]) {
            Pair_i_i tmp=q.top(); q.pop();
            int v=min(c[i], tmp.second);
            c[i]-=v, tmp.second-=v, ans-=v;
            if (tmp.second) q.push(tmp);
        }
    }
    
    printf("%I64d\n", ans);
    
    return 0;
}
