n=int(raw_input())
ans=0
nowq=set()
a=[0]*200
for i in range(1,n+1):
    ch,id=raw_input().split()
    id=int(id)
    a[i]=a[i-1]
    if ch=='+':
        nowq.add(id)
        a[i]+=1
    elif id in nowq:
        nowq.remove(id)
        a[i]-=1
    else:
        for j in range(0,i):
            a[j]+=1
for i in range(0,n+1):
    ans=max(ans,a[i])
print ans
