mcapacity = 0
c = 0
a = []
for i in range(input()):
     sign, val = raw_input().split()
     val = int(val)
     if val in a:
          c-=1
          if c<0: c = 0
          a.remove(val)
          mcapacity = max(c,mcapacity)
          continue
     if sign=='+':
          c+=1
          a.append(val)
          mcapacity = max(c,mcapacity)
     else:
          mcapacity +=1
print max(mcapacity,c)
