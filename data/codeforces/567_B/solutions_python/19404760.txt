n = input()

sensor = set()
count = 0

for i in range(n):

    action, e_id = map(str, raw_input().split())

    if action == '+':
        sensor.add(e_id)
        count = max(count, len(sensor))
    else:

        if e_id in sensor:
            sensor.remove(e_id)
        else:
            count += 1
print count
