n=input()
visitors=set([])
mv=0
for i in range(n):
    s=raw_input().strip().split()
    #print s[0],s[1]
    if s[0]=='+':
        visitors.add(s[1])
    else:
        if s[1] in visitors:
            visitors.remove(s[1])
        else:
            mv+=1
    mv=max(mv,len(visitors))
        
print mv
