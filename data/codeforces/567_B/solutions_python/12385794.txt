num = int(input())
people = {}
amount = 0
for i in range(num):
    action, p_id = raw_input().split(' ')
    if action == '-':
        if people.has_key(p_id):
            del people[p_id]
        amount += 1
    elif action == '+':
        people[p_id] = '1'
        if amount > 0:
            amount -= 1

print len(people) + amount
