n = int(raw_input())
a = []
b = []
current = 0
ma = 0
for i in range(n):
    x = raw_input().split()
    if x[0] == '+':
        a.append(x[1])
        current += 1
    else:
        if x[1] not in a:
            ma += 1
        else:
            a.remove(x[1])
            current -= 1
    if current > ma:
        ma = current
print ma
