n = int(raw_input())

arr = set([])
now ,capa  = 0,0
for i in range(n):
    line = raw_input().split(' ')
    sign, num  = line[0], int(line[1])
    if( sign == '+'):
        now +=1
        arr.add(num)
    elif(sign == '-'):
        if( num in arr):
            now -=1
        else:
            capa +=1
    capa = max( now,capa)
        
    
print capa
