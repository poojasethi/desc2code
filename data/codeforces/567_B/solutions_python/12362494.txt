if __name__ == '__main__':
	n = int(raw_input())
	logs = []
	registered = set()
	inside = [0]*(n+1)
	for i in xrange(1, n+1):
		entry = raw_input().split()
		if entry[0] == '+':
			inside[i] = inside[i-1] + 1
			registered.add(entry[1])
		else:
			if entry[1] not in registered:
				for j in xrange(i):
					inside[j] += 1
			inside[i] = inside[i-1] - 1

	print max(inside)
