n = int(input())
enter = {}
left = 0
res = 0
flag = True

for i in xrange(n):
	log = raw_input().strip().split()
	if log[0] == '-':
		if enter.has_key(log[1]):
			del enter[log[1]]
		left += 1

	else:
		enter[log[1]] = True
		if left > 0:
			left -= 1

	if len(enter) + left > res:
		res = len(enter) + left

print res
