import sys
inp = raw_input
n = int(inp())
mapp = [-1 for x in xrange(10**6)]
maxm = 0
curr = 0
for i in xrange(n):
    a,b = inp().split()
    b = int(b)
    if a == '+':
        curr += 1
        if curr > maxm:
            maxm = curr
        mapp[b-1] = 1
    else:
        if mapp[b-1] == -1:
            maxm += 1
        elif mapp[b-1] == 1:
            curr -= 1
        mapp[b-1] = 0
print maxm
