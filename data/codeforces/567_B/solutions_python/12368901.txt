n = int(raw_input())

min_cap = 0

people_in = set()

for i in range(n):
	r = raw_input().split()
	r[1] = int(r[1])
	if(r[0] == '+'):
		people_in.add(r[1])
		min_cap = max(min_cap, len(people_in))
	else:
		if r[1] in people_in:
			sgt = set()
			sgt.add(r[1])			
			people_in -= sgt
		else:
			min_cap += 1

print min_cap
