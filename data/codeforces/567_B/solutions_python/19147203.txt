#!/usr/bin/python

def ir():
    return int(raw_input())

n = ir()
h = []
for i in range(n):
    s = raw_input()
    s = s.split()
    p, k = s; k = int(k)
    h.append((p, k))

s = set(); m0 = 0
for p, k in h:
    if      p=='-' and not k in s: m0 += 1
    else                         : s.add(k)

ma = m0    
for p, k in h:
    if    p=='+': m0 += 1
    else        : m0 -= 1
    ma = max(m0, ma)

print ma
