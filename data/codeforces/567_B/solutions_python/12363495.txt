n = int(raw_input())
c = []
d = []
e = []
cnt = 0
for i in xrange(n):
    a, b = raw_input().split()
    a = str(a)
    b = int(b)
    d.append((a, b))
    if a == '-':
        if b not in c:
            cnt+=1
    else:
        c.append(b)
e.append(cnt)
for i in xrange(n):
    if d[i][0] == '-':
        cnt-=1
        e.append(cnt)
    else:
        cnt+=1
        e.append(cnt)
print max(e)
