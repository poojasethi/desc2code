n = int(input())
maxp = 0
curr = 0
ar = []

for i in range(n):
    inp = raw_input().split()
    if inp[0] == '+':
        curr += 1
        ar.append(int(inp[1]))
    if inp[0] == '-' and int(inp[1]) not in ar:
        maxp += 1
    if inp[0] == '-' and int(inp[1]) in ar:
        curr -= 1
        ar.remove(int(inp[1]))
    if curr>maxp:
        maxp = curr

print maxp
