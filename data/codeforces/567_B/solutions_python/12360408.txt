def sc():
    return raw_input().split()

n, = sc()
n = int(n)

p = [0] * 1000002
maxC = 0
c = 0
for i in range(n):
    t, r = sc()
    r = int(r)
    if t == '+':
        c += 1
        p[r] = 1
    else:
        if p[r] == 1: 
            c -= 1 
            p[r] = 0
        else:
            maxC += 1
    if maxC < c: maxC = c

print maxC
