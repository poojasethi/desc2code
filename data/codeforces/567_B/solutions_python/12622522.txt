from sys import stdin
n = int(stdin.readline())
d = {}
cap = mincap = 0
for i in xrange(n):
    rec = stdin.readline().split()
    if rec[0] == "+":
        cap += 1
        d[rec[1]] = 1
    elif rec[0] == "-":
        if rec[1] in d:
            cap -= 1
            d.pop(rec[1])
        else:
            mincap += 1
    mincap = max(cap, mincap)
print mincap
