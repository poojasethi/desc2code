n = input()
val = 0
acc = 0
arr = []
for i in range(n):
    a,b = raw_input().split()
    if b in arr:
        acc-=1
        if acc<0:
            acc=0
        arr.remove(b)
        val = max(acc,val)
        continue
    if a == '+':
        acc+=1
        arr.append(b)
        val = max(acc,val)
    else:
        val+=1
        
print max(val,acc)
