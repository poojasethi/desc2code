count = int(raw_input())

lt = []
min = 0

for i in range(count):
	t = raw_input().split(" ")
	
	if t[0] == "+":
		lt.append(t[1])
		if min < len(lt):
			min = len(lt)
	else:
		if t[1] in lt:
			del lt[lt.index(t[1])]
		else:
			min += 1

print min
