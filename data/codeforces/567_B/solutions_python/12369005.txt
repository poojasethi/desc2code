a=[0]*1000010

n= int(raw_input())

ans=0
cur=0

for i in range(n):
    s = raw_input()

    j=2
    x=0
    while(j<len(s)):
       x= x*10 + int(s[j])
       j+=1

    if(s[0]=='+'):
        cur+=1
        a[x]=1
        ans = max(ans,cur)
    else:
        if(a[x]==0):
            ans+=1
        else:
            cur-=1
            a[x]=0


print ans
