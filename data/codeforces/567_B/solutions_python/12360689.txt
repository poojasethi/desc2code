t=input()
l=[]
maxi=0
for _ in xrange(t):
    a,b=raw_input().split()
    if a=='+':
        if not(b in l):
            l.append(b)
        if len(l)>maxi:
            maxi=len(l) 
    elif a=='-':
        if b in l:
            if len(l)>maxi:
                maxi=len(l)
            del l[l.index(b)]
        else:
            maxi+=1
print maxi
