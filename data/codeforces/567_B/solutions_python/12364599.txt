import sys

n = sys.stdin.readline()

solution = 0
atm = 0
inTheRoom = []

for line in sys.stdin.readlines():
  sign, reg = line.rstrip().split()

  if sign == '+':
    inTheRoom.append(reg)
    atm += 1
    if atm > solution:
      solution = atm
    continue

  if reg in inTheRoom:
    inTheRoom.remove(reg)
    atm -= 1
  else:
    solution += 1

print solution
