import sys
def f(i):
    return (i[0],int(i[1]))



n= int(raw_input())
ls= [f(raw_input().split(' ')) for i in range(0,n)]
k=set()
maxi=0
size=0
for i in range(0,n):
    (a,b)=ls[i]
    if a == '+':
        k.add(b)
        size=size+1
        if(maxi <size):
            maxi=size
    if a == '-':
        if b not in k:
            maxi=maxi+1
        else:
            k.remove(b)
            size=size-1
