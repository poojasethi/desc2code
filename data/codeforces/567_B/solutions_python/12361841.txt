n = int(raw_input())

li = []
maxi = 0;

for i in xrange(n):
    c,num = raw_input().split()
    num = int(num)
    
    if c == '+':
        li.append(num);
        maxi = max(maxi, len(li))
    if c == '-':
        if num not in li:
            maxi += 1
        else:
            li.remove(num)
    
print maxi
