n = raw_input()
a = []
maxi, cur = 0, 0

for i in xrange(int(n)):
	temp = raw_input()
	temp = list(temp)
	temp.remove(' ')
	temp = ''.join(temp)
	ttemp = int(temp)

	a.append(int(temp))
	if ttemp > 0:
		cur += 1
		maxi = max(maxi, cur)
	elif ttemp < 0 and abs(ttemp) not in a:
		maxi += 1
	elif ttemp < 0 and abs(ttemp) in a:
		cur -= 1
		a.remove(abs(ttemp))
print maxi
