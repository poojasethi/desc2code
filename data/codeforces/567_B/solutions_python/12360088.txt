n = int(raw_input())

ev = []
r = 0

c = []
maxes = [0]

for i in xrange(n):
	e = raw_input().split()
	d = int(e[1])

	if e[0] == '+':
		c.append(d)
		maxes.append(maxes[i] + 1)
	
	if e[0] == '-':
		if d in c:
			c.remove(d)
			maxes.append(maxes[i] - 1)
		else:
			for j in xrange(0, i + 1):
				maxes[j] += 1
			maxes.append(maxes[i] - 1)
r = 0
for i in maxes:
	r = max(i, r)
		
print r
