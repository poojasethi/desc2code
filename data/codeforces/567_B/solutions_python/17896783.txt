n = input()
whoisinside=set()
currentcapacity=0
maxcapacity=0
for i in xrange(n):
    a=raw_input().split()
    if a[0]=='+':
        whoisinside.add(a[1])
        currentcapacity+=1
        maxcapacity=max(maxcapacity,currentcapacity)
    else:
        if a[1] in whoisinside:
            whoisinside.discard(a[1])
            currentcapacity-=1
        else:
            #currentcapacity-=1
            maxcapacity+=1
print maxcapacity
