n = int(raw_input())
v = set([])
max = 0
current = 0
before = 0
for _ in range(n):
    sign, num = raw_input().split()
    if sign == "+":
        v.add(num)
        current += 1
        if max < current: max = current
        continue
    current -= 1
    if num not in v:
        before += 1
print max + before
