#coding: utf-8

n = int(raw_input())

d = [0]*1000001

max = 0
total = 0
plus = 0

for i in xrange(n):

	log = raw_input().split()

	if log[0] == '+':
		d[int(log[1])] += 1
		total += 1

	else:
		if d[int(log[1])] > 0:
			d[int(log[1])] -= 1

		else:
			plus += 1

		total -= 1


	if total > max:
		max = total

print max + plus
