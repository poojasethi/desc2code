#include<stdio.h>
#include<string.h>

int n,arr[2000000],i,j,kisi,y,maxkisi;
char x;

int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		scanf(" %c%d",&x,&y);

		if(x=='+'){
			kisi++;
			arr[y]=1;
		}
		else if(arr[y]==1){
			kisi--;
			arr[y]=0;
		}
		else{
			maxkisi++;
		}
		if(kisi>maxkisi)
			maxkisi=kisi;
	}
	printf("%d",maxkisi);
}
