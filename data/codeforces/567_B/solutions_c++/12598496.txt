#include<iostream>
#include<cstdio>

using namespace std;

int n, i, w, x;
bool t[1000001];
char znak;

int main()
{
	ios_base::sync_with_stdio(0);
	cin>>n;
	for( int a = 1; a <= n; a++ )
	{
		cin>>znak>>x;
		if( znak == '+' )
		{
			t[x] = 1;
			i++;
			w = max( w, i );
		}
		else if( t[x] )
		{
			i--;
		}
		else w++;
	}
	cout<<w;
	return 0;
}
