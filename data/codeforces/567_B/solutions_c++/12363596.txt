#include <iostream>
using namespace std;

bool A[1000001]{};

int main()
{
	int n, m, k = 0;
	char c;
	int max = 0;

	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> c >> m;
		if (c == '+') { k++; A[m] = 1; }
		else { if (A[m] == 0) max++; else { A[m] = 0; k--; } }

		if (k > max) max = k;
	}
	
	cout << max;	
	
	//system("pause");
	return 0;
}
