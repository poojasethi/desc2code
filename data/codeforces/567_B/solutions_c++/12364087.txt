#include <stdio.h>
const int maxn = 1000001;
bool in[maxn];
int res;
int main(){
	int n, q, w = 0;
	char c;
	scanf("%d",&n);
	while(n --){
		getchar();
		c = getchar();
		scanf("%d",&q);
		if(c == '+'){
			w ++;
			in[q] = true;
		}else{
			if(in[q]) in[q] = false, w --;
			else res ++;
		}
		if(res < w) res = w;
	}
	printf("%d", res);
	return 0;
}
