#include<stdio.h>
int n,m,i,j;
int a[1000500];
char k[5];
int room,man;
main(){
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%s %d",k,&m);
		if(k[0]=='-'){
			if(a[m]>0){
				a[m]--;
				man--;
			}
			else{
				room++;
			}
		}
		else{
			a[m]++;
			if(room<=man){
				room++;
			}
			man++;
		}
	}
	printf("%d",room);
	return 0;
}
