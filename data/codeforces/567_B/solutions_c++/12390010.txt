#define N (1<<20)
#include <bits/stdc++.h>
using namespace std;

int n,x;
char s[5];
bool vis[N];

int main()
{
	cin>>n;
	int mx=0,cnt=0,dt=0;
	for(int i=0;i<n;i++)
	{
		scanf("%s%d",s,&x);
		if(s[0]=='+')
			vis[x]=1,cnt++,mx=max(mx,cnt);
		else if(!vis[x])
			dt++,cnt--;
		else
			vis[x]=0,cnt--;
	}
	cout<<mx+dt<<endl;
	return 0;
}
