#include <iostream>
using namespace std;

int a[1000001],n,id,nr,c;
char ch;


int main() {

	cin >> n;
	for (int i = 0; i<n; i++) {
		cin >> ch >> id;
		if (ch == '+') {
			a[id] = 1;
			nr++;
			if (nr>c) {
				c = nr;
			}
		} else {
			if (a[id] == 0) {
				c++;
			} else {
				a[id] = 0;
				nr--;
			}
		}
	}

	cout << c;


	return 0;
}
