#include <bits/stdc++.h>
using namespace std;
int vis[1000005];
int main() {
	int n;
	scanf("%d", &n);
	char s[5];
	int now=0, mx=0;
	while(n--) {
		int x;
		scanf("%s%d", s, &x);
		if(s[0]=='-') {
			if(vis[x]) {
				vis[x]=0;
				--now;
			}
			else {
				mx++;
			}
		}
		else {
			++now;
			if(now>mx) {
				mx=now;
			}
			vis[x]=1;
		}
	}
	printf("%d\n", mx);
	return 0;
}
