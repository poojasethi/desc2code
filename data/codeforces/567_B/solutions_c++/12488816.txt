#include<iostream>
using namespace std;
bool s[1000001];
int main()
{
	int n,i,w,sum=0,ans=0;
	char r;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>r>>w;
		if(r == '+')
		{
			s[w]=true;
			sum++;
			if(sum>ans)ans=sum;
		}
		else 
		{
			if(s[w]){s[w]=false;sum--;}
			else 	ans++;
		}
	}
	cout<<ans<<endl;
	//system("pause");
	return 0;
}
