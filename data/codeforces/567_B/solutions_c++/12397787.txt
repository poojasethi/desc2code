#include <bits/stdc++.h>
using namespace std;
int n,i,s,ma,kay;
char b[10];
int a[1000008];
int main()
{
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%s %d",b,&kay);
		if(b[0]=='+')
		{
			a[kay]++;
			s++;
		}
		else
		{
			if(a[kay]==0)
			{
				ma++;
			}
			else
			{
				a[kay]--;
				s--;
			}
		}
		ma=ma>s?ma:s;
	}
	printf("%d\n",ma);
	return 0;
}
