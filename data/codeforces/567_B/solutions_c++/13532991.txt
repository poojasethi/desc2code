#include <cstdio>
const int MAXN = 1e6 + 10;
bool v[MAXN];
int resp=0;
int main(){
	int cur = 0,n;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		int a; char c;
		scanf(" %c %d",&c,&a);
		if(c == '+'){
			cur++;
			if(cur > resp) resp = cur;
			v[a] = true;
		}
		else{
			if(v[a]){
				v[a] = false;
				cur--;
			}
			else resp++;
		}
	}
	printf("%d",resp);
}
