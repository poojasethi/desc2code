#include<bits/stdc++.h>
using namespace std;

int main() {
	int n,q,w=0, ret=0;
	cin>>n;
	bool state[1000001];
	for (int i=0; i<1000001; i++) state[i]=false;

	while(n--) {
		char ch; cin>>ch>>q;
		if(ch=='+') {
			w++;
			state[q]=true;
		} else {
			if(state[q]) state[q]=false, w--;
			else ret++;
		}
		ret=max(ret,w);
	}
	cout << ret << endl;
	return 0;
}
