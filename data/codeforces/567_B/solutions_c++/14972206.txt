#include<iostream>
using namespace std;
int n,a[1000005]={0},ans,in,b;
int main()
{
	char c;
	cin>>n;
	while(n--)
	{
		cin>>c>>b;
		if(c=='+')
		{
			a[b]=1;
			in++;
			in>ans?ans++:ans;
		}
		else
		{
			if(!a[b])
				ans++;
			else
				in--,a[b]=0;
		}
	}
	cout<<ans<<endl;
}
