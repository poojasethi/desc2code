#include<bits/stdc++.h>
#define f(i,a,b) for(i=a;i<b;i++)
using namespace std;
int main()
{
    int n,i,j,visitor=0,mx=0;char ch;
    cin>>n;map<int,int>m;
    f(i,0,n){
        cin>>ch>>j;
        if(!m[j]){
                if(ch=='+')m[j]=1,mx=max(mx,++visitor);
                else mx++;
        }
        else visitor--,m[j]=0;
    }
    cout<<mx;
}
