#include<iostream>
#include<map>
using namespace std;

main(){
	int n, index, d = 0, tmp = 0;
	char ch;
	map<int, int> mp;
	cin>>n;
	for (int i=0; i<n; i++){
		cin>>ch>>index;
		if (ch=='+'){
			tmp++;
			d = max(tmp, d);
			mp[index]++;
		}
		else if (mp[index] == 0){
			d++;
		}
		else{
			mp[index]--;
			tmp--;
		}
	}
	cout<<d;
}
