#include<iostream>
using namespace std;
bool arr[1000001];
int main(){
	int n,cnt=0,mx=0,nop=0;
	cin>>n;
	for(int i=0;i<n;i++){
		char a;
		int b;
		cin>>a>>b;
		if(a=='+'){
			arr[b]=1;
			nop++;
			if(mx<nop) mx=nop;
		}
		else if(arr[b]==0&&a=='-'){
			mx++;
		}
		else if(arr[b]==1&&a=='-'){
			arr[b]=0;
			nop--;
		}		
	}
	cout<<mx;
	return 0;
}
