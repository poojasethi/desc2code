#include<bits/stdc++.h>
using namespace std;
int n, mx, q;
map<int, bool> was;
int main()
	{
	cin >> n;
	for(int i = 1; i <= n; i++)
		{
		char c;
		int x;
		cin >> c >> x;
		if(c == '-' && !was[x])
			mx++;
		else if(c == '+')
			{
			q++;
			was[x] = true;
			}
		else
			q--;	
		mx = max(mx, q);
		}
	cout << mx;	
	}
