#include<cstdio>
#include<algorithm>
using namespace std;
bool flag[1000001]={0};
int n,x,cnt=0,ans=0;
char ch;
int main(){
	scanf("%d",&n);
	for (int i=0;i<n;i++){
		scanf("\n%c %d",&ch,&x);
		if (ch == '+') flag[x]=1,cnt++;
		else if (ch == '-' )
			if (flag[x])
				flag[x]=0,cnt--;
			else ans++;
		ans=max(ans,cnt);
	}
	printf("%d\n",ans);
	return 0;
}
