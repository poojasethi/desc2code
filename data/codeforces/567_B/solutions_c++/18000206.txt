#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,k,state=0,ans=0;
	cin>>n;
	char ch;
	bool a[1000005];
	memset(a,false,sizeof(a));
	while(n--){
		cin>>ch>>k;
		if(ch=='+'){
			a[k]=true;
			state++;
			ans=max(ans,state);
		}
		else{
			if(a[k])
			state--;
			else
			ans++;
			a[k]=false;
		}
	}
	cout<<ans;
	return 0;
}
