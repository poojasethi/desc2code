#include <bits/stdc++.h>
using namespace std;
map<int, int> m;
int n;
char op[5];

int main(){
	scanf("%d", &n);
	int t, ans=0, cnt=0;
	while (n--){
		scanf("%s%d", op, &t);
		if (op[0] == '+'){
			++cnt, ++m[t];
			ans = max(ans, cnt);
		}
		else{
			if (m[t] == 0)
				++ans;
			else
				--cnt, --m[t];
		}
	}
	printf("%d\n", ans);
	return 0;
}
