#include <bits/stdc++.h>

using namespace std;

#define mp make_pair

int main()
{
	int n;
	cin>>n;
	char a;
	int act=0, sol=0, v;
	set <int> l;
	for(int i=0;i<n;i++)
	{
		cin>>a>>v;
		if(a=='+')
		{
			act++;
			sol=max(sol, act);
			l.insert(v);
		}
		else
		{
			if(l.count(v))
				act--;
			else
				sol++;
		}
	}
	cout<<sol<<endl;
}
