#include <iostream>
using namespace std;
int n, ans, p, q;
int ai[3000000];
int main()
{
	cin >> n;
	for (int i = 1; i <= n; ++ i)
	{
		char ch; int b;
		cin >> ch >> b;
		if (ch == '+')
		{
			p ++;
			ai[b] = 1;
			if (p > ans) ans = p;
		}
		else if (ai[b]) ai[b] = 0, p --;
		else q ++, p --;
	}
	cout << q + ans;
}
