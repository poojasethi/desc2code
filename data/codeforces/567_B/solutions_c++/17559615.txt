#include<iostream>
#include<cmath>
using namespace std;
int x[1000005],i,j,ans,n,y,k;
char c;

int main(){
	cin>>n;
	for(i=0; i<n; i++){
		cin>>c>>y;
		if(c=='+'){
			x[y]++;
			k++;
		}
		else{
			if(x[y]==0){
			ans++;

		    }		
		    else{
			    x[y] = 0;
		        k--;
		    }
		}
		ans = max(ans,k);
	}
	cout<<ans;
}
