#include<bits/stdc++.h>
using namespace std;
int main(){
	char c;
	int n,j;
	cin>>n;
	int a[1000000]={0};
	int curr=0;
	int cap=0;
	for(int i=0;i<n;i++){
		cin>>c;
		cin>>j;
		if(c=='+'){
			curr++;
			if(curr>cap){
				cap=curr;
			}
			a[j] = 1;
		}
		else{
			if(a[j]==1){
				curr--;
			}
			else{
				cap++;
			}
		}
	}
	cout<<cap<<endl;
	return 0;
}
