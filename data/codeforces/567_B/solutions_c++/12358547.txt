#include <iostream>
using namespace std;
int main() {
	int i,j,k,l,max=0,n,m,a[1000005]={0},sum=0;
	char c;
	cin>>n;
	for(i=0;i<n;i++) {
		cin>>c>>j;
		if(c == '+') {
			a[j] = 1;
			sum++;
			if(sum>max)
				max = sum;
		}
		else {
			if(a[j]==1) {
				sum--;
				a[j] = 0;
			}
			else
				max++;
		}
	}
	cout<<max<<endl;
	return 0;
}
