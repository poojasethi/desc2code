#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int n,t=0,j,i,k=0,m=0;	char c;
	cin>>n; 	
	int a[n];
	for(i=0;i<n;i++){
		cin>>c>>a[k];
		if(c=='+') t++;
		else{
		for(j=0;j<k;j++) 
	{		if(a[j]==a[k]){ t--; break;}}
		if(j==k) m++;
		}
		m=max(m,t);
		k++;
		if(c=='-') k--;
		}
		cout<<m;
	return 0;
}
