#include<stdio.h>
int n,r[200],i,p,q,m,k,s,x,j;
char t,b;
int main(){
	scanf("%d",&n);
	getchar();
	for(i=0;i<n;i++){
		scanf("%c%c%d",&t,&b,&q);
		getchar();
		if(t=='+'){
			r[p++]=q;
			k++;
			if(m<k)m=k;
		}
		else {
			s=0;
			for(j=0;j<n;j++)
				if(r[j]==q){s=1;x=j;}
			if(s==1){r[x]=0;k--;}
			else m++;
		}
	}
	printf("%d",m);
}
