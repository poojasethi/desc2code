# include<iostream>
# include<map>
using namespace std;
int main(){
map<int,bool>ahmed;

int visitor,capacity=0,inn=0,n;cin>>n;

   if(ch=='+'){ahmed[visitor]=true,inn++;
   if(inn>capacity)capacity++;}
   else {if(ahmed[visitor]==true)inn--;
       if(ahmed[visitor]==false)capacity++;
   }}cout<<capacity<<endl;}
