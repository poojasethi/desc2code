#include <bits/stdc++.h>

using namespace std;

int n,q,w=0, ret=0;
bool s[1000001];

int main()
{
	cin>>n;
	memset(s,0, sizeof(s));
	while(n--)
  {
		char cha;
		cin>>cha>>q;
		if(cha=='+') {
			w++;
			s[q]=true;
		} else
		{
			if(s[q])
			{
			  s[q]=false; w--;
			}
			else ret++;
		}
		ret=max(ret,w);
	}
	cout << ret << endl;
	return 0;
}
