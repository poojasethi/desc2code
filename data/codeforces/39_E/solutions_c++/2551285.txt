#include<cstdio>
#include<map>

using namespace std;

int a,b,n;

long long Now;

map <pair<int,int>,int> Map;

int Min(int a,int b){return a<b?a:b;}

int Dfs(int a,int b)
{
	if(Map.count(make_pair(a,b)))
		return Map[make_pair(a,b)];
	if(b==1&&a*a>=n)
		return 2*((n-a+1)&1);
	if(a==1)
	{
		Now=1;
		for(int i=0;i<b;i++)Now*=a+1;
		if(Now>=n)return Map[make_pair(a,b)]=1;
	}
	Now=1;
	for(int i=0;i<b;i++)Now*=a;
	if(Now>=n)
	{
	//	printf("%d %d %d\n",a,b,1);
		return Map[make_pair(a,b)]=2;
	}
	//printf("%d %d %d\n",a,b,1- Min(Dfs(a+1,b),Dfs(a,b+1)));
	return Map[make_pair(a,b)]=2-Min(Dfs(a+1,b),Dfs(a,b+1));
}

int main()
{
	scanf("%d%d%d",&a,&b,&n);
	n=Dfs(a,b);
	if(n==2)printf("Masha\n");
	else if(n==1)printf("Missing\n");
	else printf("Stas\n");
	return 0;
}
