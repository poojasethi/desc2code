#include <cstdio>

int a, b, n;
bool sg[35002][52], v[35002][52];

double power(double x, int e){
	if (e == 0) return 1;
	if (e == 1) return x;
	double ret = power(x, e >> 1);
	ret *= ret;
	if (e & 1) return ret * x; else return ret;
}

bool calc(int a, int b){
	if (power(a, b) >= n) return true;
	if ((long long)a * a >= n) return !((n - a) & 1);
	if (v[a][b]) return sg[a][b];
	v[a][b] = true;
	if (calc(a + 1, b) && calc(a, b + 1)) sg[a][b] = false;
	else sg[a][b] = true;
	return sg[a][b];
}

int main(){
	scanf("%d%d%d", &a, &b, &n);
	if (a == 1 && power(2, b) >= n) printf("Missing");
	else if (calc(a, b)) printf("Masha"); else printf("Stas");
}
