#include<cstdio>
typedef long long ll;

ll power(ll a, ll b){
	if(b == 0) return 1;
	ll temp = power(a, b/2);
	if(b % 2 == 0)
		return temp * temp;
	else
		return temp * temp * a;

}

int dfs(ll a, ll b, ll n) {
	if(a == 1 && power(2,b) >= n) return 2;
	if(power(a, b) >= n) return 0;
	int state1 = dfs(a, b + 1, n);
	if(state1 == 1) return 0;

	int state2 = dfs(a + 1, b, n);
	if(state2 == 1) return 0;

	if(state1 == 2 || state2 == 2)
		return 2;

	return 1;
}

int main(){
	ll a, b, n;
	scanf("%I64d%I64d%I64d", &a, &b, &n);
	
	int f = dfs(a,b,n);
	if(f == 0) puts("Masha");
	else if(f == 1) puts("Stas");
	else puts("Missing");
	
	return 0;
}
