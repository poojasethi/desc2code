#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<time.h>
#include<math.h>
#include<iostream>
#include<algorithm>
using namespace std;
#define N 100005
char str[3][100]={"Stas","Missing","Masha"};
int f[N][32],a,b,n;
int DP(int a,int b){
	if(f[a][b]!=-1) return f[a][b];
	if(a==1&&1<<b>=n) return 1;
	if(b==1&&a*a>=n) return n-a&1?0:2;
	if(pow(a,b)+1e-6>n) return f[a][b]=2;
	return f[a][b]=2-min(DP(a+1,b),DP(a,b+1));
}
int main()
{
	//freopen("1.in","r",stdin);
	//freopen("1.out","w",stdout);
	scanf("%d%d%d",&a,&b,&n);
	memset(f,-1,sizeof(f));
	puts(str[DP(a,b)]);
	//system("pause");
	return 0;
}
