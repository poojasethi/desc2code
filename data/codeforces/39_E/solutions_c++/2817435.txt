#include<cstdio>
#include<cstring>
int i,j,X,Y,N,res;
int A[33335][35],f[33335][35];
int dfs(int x,int y)
{
	int f1,f2;
	if(f[x][y]) return f[x][y];
	if(x==1&&y>30)
	{
		f[x][y]=3;
		return 3;
	}
	if(!A[x][2]&&y==1)
	{
		f[x][y]=2-((N-x)&1);
		return f[x][y];
	}
	if(x>1&&y>1&&!A[x][y])
	{
		f[x][y]=1;
		return 1;
	}
	f1=dfs(x+1,y),f2=dfs(x,y+1);
	if(f1==2||f2==2)
	{
		f[x][y]=1;
		return 1;
	}
	if(f1==3||f2==3)
	{
		f[x][y]=3;
		return 3;
	}
	f[x][y]=2;
	return 2;
}
int main()
{
	scanf("%d%d%d",&X,&Y,&N);
	--N;
	for(i=1;i*i<=N;i++)
		for(A[i][0]=1,j=1;A[i][j-1]<=N/i&&j<=30;++j)
			A[i][j]=A[i][j-1]*i;
	res=dfs(X,Y);
	if(res==1) printf("Masha\n");
	else if(res==2) printf("Stas\n");
		else printf("Missing\n");
	return 0;
}
