#include <bits/stdc++.h>
using namespace std;

int f[40000][40];
int N,A,B,t;
bool quick(int x,int y)
{
    long long s=1,t=x;
    for (;y;y/=2)
    {
        if (y&1)s*=t;
        t*=t;
    }
    if (s>=N)return true;else return false;
}
int play(int A,int B)
{
    if (f[A][B]>=0)return f[A][B];
    if (A==1&&1<<B>=N)return f[A][B]=1;
    if (B==1&&A*A>=N)
    {
        if ((N-A)&1)return f[A][B]=0;
        else    return f[A][B]=2;
    }
    if (quick(A,B))return f[A][B]=2;
    return f[A][B]=2-min(play(A+1,B),play(A,B+1));
}
int main()
{
    scanf("%d%d%d",&A,&B,&N);
    memset(f,128,sizeof(f));
    t=play(A,B);
    if (t==0)printf("Stas");
    else    if (t==1)printf("Missing");
    else    printf("Masha");
}
