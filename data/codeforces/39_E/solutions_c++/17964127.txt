#include<stdio.h>
int n;
bool POW(int a, int b) {
    int ret = 1;
    for(int i = 1; i <= b; i ++) {
        if(1ll * ret * a >= 1ll * n) return 0;
        ret *= a;
    }
    return 1;
}
int dfs(int a, int b) {
    if(a == 1) {
        if(!POW(a + 1, b)) return -1;
        int c = dfs(a + 1, b);
        if(!c) return 1;
        return 1 - dfs(a, b + 1);
    }
    if(POW(a, b + 1) && !dfs(a, b + 1)) return 1;
    if(POW(a + 1, b) && !dfs(a + 1, b)) return 1;
    return 0;
}
int main(){
    int a, b;
    scanf("%d%d%d", &a, &b, &n);
    int ans = dfs(a, b);
    if(ans == 1)puts("Masha");
   if(!ans) puts("Stas");
   if(ans == -1) puts("Missing");
    return 0;
}
