#include <cstdio>

using namespace std;

int n;

long long power(int a, int b) {
	if (b == 0) {
		return 1;
	}
	long long ret = power(a, b / 2);
	if (b & 1) {
		return ret * ret * a;
	} else {
		return ret * ret;
	}
}

int f(int a, int b) {
	if (a == 1 && 1 << b >= n) return 2;
	if (power(a, b) >= n) return 0;

	int ansO2 = f(a, b+1);
	if (ansO2 == 1) return 0;

	int ansO1 = f(a+1, b);
	if (ansO1 == 1) return 0;

	if (ansO1 == 2) return 2;
	if (ansO2 == 2) return 2;
	return 1;
}


int main() {
	int a, b;
	scanf("%d%d%d", &a, &b, &n);
		int t = f(a, b);
		if (t == 0) {
			puts("Masha");
		}
		else if (t == 2) {
			puts("Missing");
		}
		else {
			puts("Stas");
		}
	return 0;
}
