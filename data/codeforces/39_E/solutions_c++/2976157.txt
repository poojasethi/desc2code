#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstdlib>
using namespace std;
int n,a,b;
int f[40000][40],g[40000][40];
int dfs(int a,int b)
{
	long long ans=1;
	for (int i=1;i<=b;i++)
		{
			ans*=a;
			if (ans>=n) return 1;
		}
	if (g[a][b]) return f[a][b];
	if (a==1 && b<log(n)/log(2))
		if (dfs(a+1,b) && dfs(a,b+1)) return 0; else return 1;
	else if (a==1) {return 2;}
	if ((b==1) && (a*a>=n))
		if ((n-a)%2==1) return 0; else return 1;
	f[a][b]=max(dfs(a+1,b) xor 1,dfs(a,b+1) xor 1);
	g[a][b]=1;
	return f[a][b];
}
int main()
{
	scanf("%d%d%d",&a,&b,&n);
  int	k=dfs(a,b);
	if (k==1) {printf("Masha"); } else
		if (k==0) printf("Stas"); else printf("Missing");
	return 0;
}
