#include <cmath>
#include <cstdio>
#include <cstring>

int A,B,N,f[100005][35];

int dfs(int a,int b)
{
	if (f[a][b]!=-1) return f[a][b];
	int&x=f[a][b],u,v;
	if (b==1&&a*a>=N) return x=((N-a)&1?1:0);
	if (a==1&&(1ll<<b)>=N) return x=2;
	if (pow(a,b)-N>-1e-9) return x=0;
	u=dfs(a+1,b),v=dfs(a,b+1);
	if (u==1||v==1) x=0; else if (u==2||v==2) x=2; else x=1;
	return x;
}

int main()
{
	scanf("%d%d%d",&A,&B,&N),memset(f,-1,sizeof(f));
	int t=dfs(A,B);
	puts(!t?"Masha":t==1?"Stas":"Missing");
}
