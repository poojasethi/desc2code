#include<cstdio>
#include<cstring>
int a[100005][35],b[100005][35],x,y,n;
int dfs(int x,int y)
{
	if (b[x][y]) return b[x][y]; int &res=b[x][y];
	if (x*x>n && y==1) return res=2-((n-x)&1);
	if (x==1 && y>30) return res=3;
	if (x-1 && y-1 && !a[x][y]) return res=1;
	int t1=dfs(x+1,y),t2=dfs(x,y+1);
	if (t1==2 || t2==2) return res=1;
	if (t1==3 || t2==3) return res=3;
	return res=2;
}
int main()
{
	scanf("%d%d%d",&x,&y,&n),n--;
	for (int i=2,j;i*i<=n;i++)
		for (j=1,a[i][1]=i;a[i][j]<=n/i;j++) a[i][j+1]=a[i][j]*i;
	int res=dfs(x,y);
	if (res==1) printf("Masha\n");
	else if (res==2) printf("Stas\n");
		else printf("Missing\n");
	return 0;
}
