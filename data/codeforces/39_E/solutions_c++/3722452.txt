#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define LL long long
using namespace std;
LL n;
int a,b,f[32000][40],tmp;
LL ksm(int x,int y){LL b;return y?(b=ksm(x,y/2))*b*(y&1?x:1):1;}
int dfs(int x,int y)
{
    if(f[x][y]!=-1)return f[x][y];
    if(x==1&&(1<<y)>=n)return f[x][y]=1;
    if(ksm(x,y)>=n)return f[x][y]=2;
    if (x*x>=n&&y==1)return 2-2*((n-x)&1);
    return f[x][y]=2-min(dfs(x+1,y),dfs(x,y+1));
}
int main()
{
    scanf("%d%d%I64d",&a,&b,&n);
    memset(f,-1,sizeof(f));
    switch(tmp=dfs(a,b))
    {
        case 0:{printf("Stas");break;}
        case 1:{printf("Missing");break;}
        case 2:{printf("Masha");break;}
    }
    return 0;
}
