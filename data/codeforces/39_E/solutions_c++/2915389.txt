#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;

typedef long long LL;
const int maxn = 32005, maxm = 35;
const string res[3] = {"Stas", "Missing", "Masha"};
int f[maxn][maxm], n, a, b;

LL pow(int a, int n) {
    LL ans = 1;
    for (int i = 1; i <= n; ++i)
        ans *= a;
    return ans;
}

int calc(int a, int b) {
    if (f[a][b] != -1) return f[a][b];
    if (a == 1 && 1 << b >= n) return 1;
    if (b == 1 && a * a >= n) return (n - a) & 1 ? 0 : 2;
    if (pow(a, b) >= n) return f[a][b] = 2;
    return f[a][b] = 2 - min(calc(a + 1, b), calc(a, b + 1));
}

int main() {
    memset(f, 255, sizeof f);
    cin >> a >> b >> n;
    cout << res[calc(a, b)] << endl;
    return 0;
}
