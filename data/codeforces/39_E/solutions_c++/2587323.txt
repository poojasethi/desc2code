#include<iostream>
#include<cstring>
#include<cmath>
#include<cstdio>
using namespace std;
int f[400000][32],a,b,c,n;
int get(int a,int b){
	if(f[a][b]!=-1)return f[a][b];
	if(a==1&&1<<b>=n)return f[a][b]=2;
	if(b==1&&a*a>=n)return f[a][b]=(n-a)%2;
	if(pow(1.0*a,1.0*b)+1e-8>n)return f[a][b]=0;
	int x=get(a+1,b),y=get(a,b+1);
	if(x==1||y==1)return f[a][b]=0;
	return f[a][b]=1;
}
int main(){
	memset(f,-1,sizeof(f));
	cin>>a>>b>>n;
	int c=get(a,b);
	if(c==0)puts("Masha");
	if(c==1)puts("Stas");
	if(c==2)puts("Missing");
	//system("pause");
	return 0;
}
