#include <cstdio>

int n;

long long pow(int a, int b) {
	long long p = 1;

	for (int i = 1; i <= b; i++) {
		p *= a;
	}

	return p;
}

int f(int a, int b) {
	//hack?
	if (a == 1 && n <= 1 << b) return 2;
	if (pow(a, b) >= n) return 0;

	int ansO2 = f(a, b+1);
	if (ansO2 == 1) return 0;
	int ansO1 = f(a+1, b);
	if (ansO1 == 1) return 0;

	if (ansO1 == 2) return 2;
	if (ansO2 == 2) return 2;
	return 1;
}

int main() {
	int a, b;
	scanf("%d %d %d", &a, &b, &n);
	int ans = f(a, b);
	if (ans == 0) printf("Masha\n");
	if (ans == 1) printf("Stas\n");
	if (ans == 2) printf("Missing\n");
	return 0;
}
