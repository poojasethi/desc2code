#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
int f[40000][40],n,a,b;
int dp(int a,int b)
{
    if(f[a][b]!=-1)return f[a][b];
    if(pow(double(a),b)+0.1>n)return f[a][b]=2;
    if(b>30)return f[a][b]=1;
    if(b==1&&a*a>n)return f[a][b]=2-2*((n-a)%2);
    return f[a][b]=2-min(dp(a+1,b),dp(a,b+1));
}
int main()
{
    scanf("%d%d%d",&a,&b,&n);
    memset(f,255,sizeof f);
    dp(a,b);
    if(f[a][b]==2)puts("Masha");
    else if(f[a][b]==1)puts("Missing");
    else puts("Stas");
    return 0;
}
