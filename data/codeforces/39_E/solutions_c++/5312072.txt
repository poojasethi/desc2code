#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
int f[100100][41],a,b,n;

bool pd(int a,int b){
  int tmp=1;
  for (int i=1;i<=b;i++){
    if ((n-1)/tmp<a) return 1;
    tmp=tmp*a;
  }
  return 0;
}

int dp(int a,int b){
  if (f[a][b]>0) return f[a][b];
  if (a==1 && (1<<b)>=n) return f[a][b]=1;
  if (b==1 && a*a>=n) return f[a][b]=2-2*((n-a)&1);
  if (pd(a,b)) return f[a][b]=2;
  return f[a][b]=2-min(dp(a+1,b),dp(a,b+1));
  
}

int main(){
  //freopen("39E.in","r",stdin);
  //freopen("39E.out","w",stdout);
  scanf("%d%d%d",&a,&b,&n);
  memset(f,-1,sizeof(f));
  dp(a,b);
  if (f[a][b]==2) puts("Masha");
    else if (f[a][b]==1) puts("Missing");
      else puts("Stas");
  return 0;
}
