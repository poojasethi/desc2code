#include<cstring>
#include<cstdio>
#include<cmath>
#define maxn 50000
#define maxm 50
#define max(a,b) (a>b?a:b)
using namespace std;
int f[maxn+10][maxm+10],a,b,n,s,j,i;
int pow1(int x,int y){long long t=1;while(y--){t*=x;if(t>n)return n+1;}return t;}
int main()
{
	scanf("%d%d%d",&a,&b,&n);n--;
	memset(f,-1,sizeof(f));f[1][maxm+1]=0;
	for(j=maxm;j>=b;j--)
	{
		s=int(pow(n,1/(double)j));while(pow1(s+1,j)<=n)s++;while(pow1(s,j)>n)s--;
		if(s>maxn)s=maxn,f[maxn+1][j]=(n-maxn)&1?1:-1;
		for(i=s;i>=a;i--)f[i][j]=-max(f[i+1][j],f[i][j+1]);
	}if(f[a][b]<0)printf("Masha\n");else
	if(f[a][b]>0)printf("Stas\n");else printf("Missing\n");
}
