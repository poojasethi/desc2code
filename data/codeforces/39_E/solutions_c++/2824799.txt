#include <cstring>
#include <cstdio>
using namespace std;

int visit[40000][50], f[40000][50];
int n,a,b,ans;

long long power(long long x, long long p)
{
	long long res=1, res2=x;
	for (;p;p>>=1)
	{
		if (p&1) res*=res2;
		res2*=res2;
	}
	return res;
}

// 0 win 1 lose 2 draw
int dp(int a, int b)
{
	if (visit[a][b]) return f[a][b];
	visit[a][b]=1;
	
	if (power(a,b)>=n) return f[a][b]=0;
	if (a*a>n && b==1) return f[a][b]=(n-a)%2;
	if (a==1 && power(a+1,b)>=n) return f[a][b]=2;
	
	int res=1,p;
	p=dp(a+1,b);
	if (p==1) res=0; if (p==2 && res==1) res=2;
	p=dp(a,b+1);
	if (p==1) res=0; if (p==2 && res==1) res=2;
	return f[a][b]=res;
}

int main()
{
	scanf("%d%d%d",&a,&b,&n);
	ans=dp(a,b);
	if (ans==0) printf("Masha\n");
	if (ans==1) printf("Stas\n");
	if (ans==2) printf("Missing\n");
	//for (;;);
}
