#include <iostream>
#include <algorithm>
#include <cstring>
#include <cmath>
using namespace std;
string c[3]={"Stas","Missing","Masha"};
int f[35000][35],a,b,n;
int dp(int a,int b) {
    if (f[a][b]!=-1) return f[a][b];
    if (a==1&&(1<<b)>=n) return 1;
    if (a*a>=n&&b==1) return 2-2*((n-a)&1);
    if (pow(a,b)+1e-6>=n) return f[a][b]=2;
    return f[a][b]=2-min(dp(a+1,b),dp(a,b+1));
}
int main() {
    memset(f,-1,sizeof(f));
    cin>>a>>b>>n;
    cout<<c[dp(a,b)];
}
