#include <iostream>
#include <cmath>
#include <cstring>
using namespace std;

int f[32000][40],n;

int solve(int a,int b){
        if (f[a][b]!=-1) return f[a][b];
        if (pow(double(a),b)+0.5>=n)  return 2;
        if (b>31) return 1;
        if (b==1&&a*a>=n) return f[a][b]=2-2*((n-a)%2);
        return f[a][b]=2-min(solve(a+1,b),solve(a,b+1));
}

int main(){
        int a,b;
        cin>>a>>b>>n;
        memset(f,-1,sizeof f);
        int ans=solve(a,b);
        if (ans==2) cout<<"Masha";else if (ans==1) cout<<"Missing";else cout<<"Stas";
}
