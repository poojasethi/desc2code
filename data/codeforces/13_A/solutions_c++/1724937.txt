#include<cstdio>
int n,i,j,ans1,ans2;
int gcd(int a,int b){return b?gcd(b,a%b):a;}
int main()
{
    scanf("%d",&n);
    for (i=2;i<n;i++)
    {
        ans2++;
        for (j=n;j;j/=i)ans1+=j%i;
    }
    i=gcd(ans1,ans2);
    printf("%d/%d\n",ans1/i,ans2/i);
    return 0;
}
