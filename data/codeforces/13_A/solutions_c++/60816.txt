#include <cstdio>
#include <algorithm>
using namespace std;
int A;
int main()
{
	scanf("%d", &A);
	int num = 0, den = A-2;
	for (int b = 2; b < A; b++)
	{
		int x = A;
		while (x)
		{
			num += x % b;
			x /= b;
		}
	}
	int gcd = __gcd(num, den);
	num /= gcd;
	den /= gcd;
	printf("%d/%d\n", num, den);
}
