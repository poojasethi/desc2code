#include <cstdio>

int gcd(int x, int y) {
	return y ? gcd(y, x % y) : x; }

int main() {
	int a, sum = 0, i, tmp;
	scanf("%d", &a);
	for(i = 2; i < a; ++i) {
		tmp = a;
		while(tmp) {
			sum += tmp % i;
			tmp /= i; } }
	tmp = gcd(a - 2, sum);
	printf("%d/%d", sum / tmp, (a - 2) / tmp);
	return 0; }
