#include <iostream>

using namespace std;

int top,N;

void Base(int g)
{
	int x;
	x=N;
	while(x)
	{
		top+=x%g;
		x/=g;
	}
}

int gcd(int a,int b)
{
	if(!b) return a;
	return gcd(b,a%b);
}

int main()
{
	int x;
	cin >> N;
	for(int i=2;i<N;i++) Base(i);
	x=gcd(top,N-2);
	cout<<top/x<<"/"<<(N-2)/x<<endl;
}
