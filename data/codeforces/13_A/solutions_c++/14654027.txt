#include <iostream>
#include <algorithm>

using namespace std;

int a,sorat,makhraj,bmm,k;
int main(){
  cin>>a;
  makhraj=a-2;
  for(int i=2;i<a;i++){
    k=a;
    while(k){
      sorat+=k%i;
      k/=i;
    }
  }
  bmm=__gcd(sorat,makhraj);
  sorat/=bmm;
  makhraj/=bmm;
  cout<<sorat<<"/"<<makhraj<<endl;
}
