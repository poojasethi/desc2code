#include <stdio.h>

int gcd(int a,int b)
{ int c;
  while((c=a%b))
  {
    a=b;
    b=c;
  }
  return b;
}
int main() {
int A;
scanf("%d",&A);
int sum=0;
for(int i=2;i<A;i++)
  for(int k=A;k;sum+=k%i,k=k/i)
    ;
  int dem=A-2;
  int g=gcd(sum,dem);
sum/=g;
dem/=g;
printf("%d/%d",sum,dem);

  return 0;
}
