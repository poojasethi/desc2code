
#include<stdio.h> 
int main()
{
long long int i,n,h,p,k,kk=0,a[100002]={0};
scanf("%lld %lld %lld",&n,&h,&k);
for(i=0;i<n;i++)
scanf("%lld",&a[i]);
i=0;
p=0;
while(i<n)
{
	while(p+a[i]<=h&&i<n)
	{
		p+=a[i++];
	}
kk+=p/k;
p=p%k;
if(p+a[i]>h)
	{
	kk++;
	p=0;
	}
}
if(p>0)kk++;
printf("%lld\n",kk);
return 0;
}
