#include <cstdio>
using namespace std;
int main(){
	long long n,h,k,a;
	long long cur = 0,ans = 0,pre = 0;
	scanf("%lld%lld%lld",&n,&h,&k);
	for(int i = 1;i <= n;i++){
		scanf("%lld",&a);
		cur  = pre + a;
		if(cur > h){
			ans++;
			cur = a;
		}
		ans = ans + (cur / k );
		pre = cur % k;
	}
	if(pre > 0) ans ++;
	printf("%lld\n",ans);
}
