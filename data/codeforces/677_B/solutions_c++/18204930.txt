#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,t=0,h,k,i,j,a[100005],z=0,p,q;
	cin>>n>>h>>k;
	for(i=0;i<n;i++)
		cin>>a[i];
	a[n]=0;
	j=i=0;
	while(i<n)
	{
		if(j+a[i]<=h)
			j+=a[i++];
		else
		{
			t+=j/k;
			j=j%k;
			if(j+a[i]>h)
			{
				t++;
				j=0;
			}
		}
	}
	t+=(j+k-1)/k;
	cout<<t<<endl;
	return 0;
}
