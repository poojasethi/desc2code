#include <iostream>
using namespace std;

int main() {
	// your code goes here
	long long int n,k,h,i,a[100005],s=0,x;
	cin>>n>>h>>k;
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n-1;i++)
	{
		s+=a[i]/k;
		x=a[i]%k;
		if(a[i+1]+x<=h)
		a[i+1]+=x;
		else
		s+=1;
	}
	
	s+=a[n-1]/k;
	if(a[n-1]%k!=0)
	s+=1;
	cout<<s;
	return 0;
}
