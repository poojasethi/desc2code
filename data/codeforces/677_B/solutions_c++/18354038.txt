#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,h,k,i,cur_h=0, sec=0,temp;
	cin >> n >> h >> k;
	for(i=0;i<n;i++)
	{
		cin >> temp;
		if(temp+cur_h<=h)
			cur_h+=temp;
		else
		{
			sec++;
			cur_h=temp;
		}
		sec+=cur_h/k;
		cur_h%=k;
	}
	sec+=cur_h/k;
	cur_h%=k;
	sec+=(cur_h>0);
	cout << sec << endl;
	return 0;

}
