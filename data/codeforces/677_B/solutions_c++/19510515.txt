#include <iostream>
#include <queue>
using namespace std;

int main() {
	long long n,h,k,second=0,sum=0,t;
	cin>>n>>h>>k;	
	for(long long i=0;i<n;i++){		
		cin>>t;
		if(sum+t<=h){
			sum+=t;
		}else{
			sum=t;
			second++;
		}
		second+=sum/k;
		sum%=k;
	}
	second+=(sum+k-1)/k;
	cout<<second;
	return 0;
}
