#include <bits/stdc++.h>
#define ll long long

using namespace std;

int main()
{
	ll int n, a, d=0, h, k, ans=0;
	cin >> n >> h >> k;
	
	for (int i=0; i<n; i++)
	{
		cin >> a;
		
		if (d+a <= h)
			d += a;
		else 
			ans++, d = a;
		ans += d/k;
		d %= k;
	}
	ans += d/k;
	d %= k;
	ans += (d>0);
	cout << ans;
	
	return 0;
}
