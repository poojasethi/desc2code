#include <cstdio>
long long n, h, k, a;
long long tmp = 0, time = 0;
int main (){
	scanf("%I64d %I64d %I64d", &n, &h, &k);
	for (int i = 0; i < n; i++){
		scanf("%I64d", &a);
		if (tmp + a > h){
			tmp = a;
			time++;
		} else 
			tmp += a;
		time += tmp / k;
		tmp %= k;
	}
	if (tmp > 0)
		time++;
	printf("%I64d\n", time);
	return 0;
}
