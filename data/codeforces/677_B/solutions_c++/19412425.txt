__int64 a[100009];
int main()
{
	int n,h,k;
	scanf("%d%d%d",&n,&h,&k);
	__int64 sum=0,ans=0;
	for(int i=0;i<n;i++)
	{
		scanf("%I64d",&a[i]);
		if(sum+a[i]<=h) sum+=a[i];
		else
		{
			sum=a[i];
			ans++;
		}
		__int64 tt=sum/k;
		ans+=tt;
		sum%=k;
	}
	if(sum)
	ans++;
	printf("%I64d",ans);
	return 0;
}
