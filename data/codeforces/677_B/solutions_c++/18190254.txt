#include<iostream>

using namespace std;

int main()
{
	int n,h,k;
	cin>>n>>h>>k;

	long long c=0;
	int a[100001];
	for(int i=0;i<n;i++)
		cin>>a[i];
	
	for(int i=0; i<n-1; i++)
	{
		c += a[i]/k;

		if(a[i]%k + a[i+1] <= h)
				a[i+1] += a[i]%k;
		else
				c+=1;
					
	}

	c+=a[n-1]/k;
	if(a[n-1]%k != 0)
			c++;

	cout<<c<<endl;
	

	return 0;
}
