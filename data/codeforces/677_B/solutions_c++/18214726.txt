#include <iostream>
using namespace std;

int main(){
	long long n,h,k,smash=0,time = 0;
	cin>>n>>h>>k;
	while(n--){
		long i;
		cin>>i;
		if (smash+i <= h)
			smash+=i;
		else{
			time++;
			smash = i;
		}
		time += smash/k;
		smash = smash % k;
	}
	time += smash/k;
	if(smash%k != 0) time++;
	cout<<time;
}
