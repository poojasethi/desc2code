#include <cstdio>
int n,h,m,rest;
long long ans;
int main()
{
	scanf("%d%d%d",&n,&h,&m);
	for(int i=1,x,y;i<=n;i++)
	{
		scanf("%d",&x);
		if(rest+x>h)
		{
			y=(rest+x-h)/m+((rest+x-h)%m>0);
			rest-=m*y;if(rest<0) rest=0;
			ans+=y;
		}
		rest+=x;
	}
	ans+=rest/m+(rest%m>0);
	printf("%I64d\n",ans);
}
