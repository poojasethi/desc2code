#include <iostream>

typedef long long ll;

using namespace std;
int main()
{
    int n;
    ll h,k,a,hp=0,sum=0;

    cin>>n>>h>>k;

    while(n--){
        cin>>a;
        while(hp+a>h){if(hp<k){hp=0;sum++;}sum+=hp/k;hp=hp%k; }
        hp+=a;
    }
     while(hp>0){if(hp<k){hp=0;sum++;}sum+=hp/k;hp=hp%k; }
    cout<<sum;
	return 0;
}
