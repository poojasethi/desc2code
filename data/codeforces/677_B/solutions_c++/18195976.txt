#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
    ll int n,a,h,k,tp=0,tt=0;cin>>n>>h>>k;
    for(int i=1;i<=n;i++)
    {
        cin>>a;
        if(a+tp>h){tt++;tt+=a/k;tp=a%k;continue;}
        a+=tp;
        tt+=a/k;tp=a%k;
    }
    if(tp)tt++;
    cout<<tt;
    return 0;
}
