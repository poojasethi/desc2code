#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main() {
	long long n,h,k,x,ch,ans,i;
	cin>>n>>h>>k;
	ch=0;
	ans=0;
	for(i=0;i<n;i++){
	   cin>>x;
	   if((ch+x)<=h)
	   ch=ch+x;
	   else{
	   ans++;
	   ch=x;
	   }
	   ans+=ch/k;
	   ch=ch%k;
	}
if(ch!=0)
ans++;
	cout<<ans;
	return 0;
}
