#include<iostream>
#include<string>
#include<set>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{
long long n,h,k,i,j;
cin>>n>>h>>k;
long long a[n],x=0,time=0;
for(i=0;i<n;i++)
{
cin>>a[i];
if(x+a[i]<=h)
x=x+a[i];
else
{
time++;
x=a[i];
}
time+=x/k;
x=x%k;
}
time+=x/k;
x=x%k;
if(x>0)
time=time+1;

cout<<time<<endl;
}
