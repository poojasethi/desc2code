#include<bits/stdc++.h>

using namespace std;

typedef long long int ll;

int main()
{
	ll n,h,k;
	scanf(" %lld %lld %lld",&n,&h,&k);
	ll r = 0,at = 0;
	for(int i=0; i<n; i++)
	{
		ll a;
		scanf(" %lld",&a);
		if(at+a>h) { r+=(at/k); at-=k * (at/k); }
		if(at+a>h) { r++; at = 0; }
		at += a;
	}
	printf("%lld\n",r + at/k + int(at%k!=0));
	return 0;
}
