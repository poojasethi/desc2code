#include <bits/stdc++.h>

using namespace std;
#define ll long long
int main()
{
    ll n, h, k, i, c=0, ret=0, x;
    cin >> n >> h >> k;
    for (i=0; i<n; ++i)
    {
       cin >> x;
       if (c+x>h)c=0, ++ret;
       c+=x;
       ret+=(c/k);
       c%=k;
    }
    ret+=(c>0);
    cout << ret;
    return 0;
}
