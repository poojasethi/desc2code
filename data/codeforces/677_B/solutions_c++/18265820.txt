#include<iostream>
using namespace std;
int main()
{
	int n,i,j,arr[100001];
	long long int h,k,rem,ans;
	cin >> n >> h >> k;
	for(i=0;i<n;i++)
		cin >> arr[i];
	rem=ans=0;
	for(i=0;i<n;i++)
	{
		if(rem+arr[i]>h)
		{
			ans+=1;
			rem=0;
		}
		rem+=arr[i];
		ans+=rem/k;
		rem=rem%k;
	}
	if(rem)
		ans+=1;
	cout << ans << endl;
	return 0;
}
