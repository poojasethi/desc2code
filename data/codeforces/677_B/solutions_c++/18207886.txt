#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
	int n,h,k,x;
	scanf("%d%d%d",&n,&h,&k);
	ll sum=0,ans=0;
	while(n--){
		scanf("%d",&x);
		if(sum+x>h){
			ans++;
			sum=0;
		}
		if(sum+x<=h){
			sum+=x;
			ans+=sum/k;
			sum%=k;
		}
	}
	if(sum>0)ans++;
	cout<<ans<<endl;
	return 0;
}
