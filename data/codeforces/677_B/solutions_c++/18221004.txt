#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int n,h,k,i=0,ch=0,ans=0,x;
	cin>>n>>h>>k;
	for(i=0;i<n;i++)
		{
			cin>>x;
			if (ch + x <= h)
		   	ch += x;
			else
			{
				ans++;
				ch = x;
			}
			ans += ch/k;
			ch %= k;
	}
	ans += ch/k;
	ch %= k;
	ans += (ch>0);
	cout << ans << endl;
	return 0;
}
