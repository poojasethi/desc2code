first_line = raw_input().split()
processor_height = int(first_line[1])
potato_smashed_per_sec = int(first_line[2])

potato_heights = [int(p) for p in raw_input().split()]

max_index = len(potato_heights)

amount_in_processor = 0
time = 0
index = 0
while True:
    if max_index == index:
        print( time + (amount_in_processor +  (potato_smashed_per_sec - 1) )/ potato_smashed_per_sec)
        break
    forward_time = 1

    while max_index != index:
        if (potato_heights[index] + amount_in_processor) <= processor_height:
            amount_in_processor += potato_heights[index]
            index += 1
        else:
            forward_time = (amount_in_processor  + potato_heights[index] - processor_height + (potato_smashed_per_sec - 1)) / potato_smashed_per_sec
            break

    amount_in_processor = max((amount_in_processor - forward_time * potato_smashed_per_sec), 0)
    time += forward_time
