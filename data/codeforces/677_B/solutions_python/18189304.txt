#!/usr/bin/python

import sys

def Ni(): return map(int, sys.stdin.readline().split())
n, h, k = Ni()
a = Ni()

fill = 0
t = 0
i = 0
while True:
    while i < n and fill + a[i] <= h:
        fill += a[i]
        i += 1

    if i < n:
        needemptyto = h - a[i]
        needsmashed = fill - needemptyto
        ss = (needsmashed + k - 1) / k
    else:
        ss = (fill + k - 1) / k

    t += ss
    fill = max(0, fill - ss * k)

    if i == n and fill == 0:
        break

print t
