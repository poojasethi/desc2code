n,h,k=map(int,raw_input().split())
l=map(int,raw_input().split())
s=0
ans=chk=0

for i in l:
    if chk+i<k:
        chk+=i
        continue
    elif chk+i<=h:
        tmp=chk+i
        ans+=tmp/k
        chk=tmp%k
    else:
        ans+=1
        chk=i
        if chk>=k:
            ans+=chk/k
            chk%=k


print ans+1 if chk else ans
