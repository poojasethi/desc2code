n,h,k = map(int, raw_input().split())
A = map(int, raw_input().split())
X = 0
ans = 0
i = 0
while True:
    new = False
    while X+A[i]<=h: 
        new = True
        X+=A[i]
        i+=1
        if i>=n: break
    if not new:
        ans +=1
        X=0
    ans += X/k
    X = X%k
    if i>=n: break

if X>0: ans+=1
print ans
