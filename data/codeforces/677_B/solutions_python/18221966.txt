from sys import stdin, stdout

input = stdin.readline

n, h, k = map(int, input().split())

a = map(int, input().split())

ans = 0

cur = 0

for i in xrange(n):

  if a[i] + cur > h:
    ans += cur / k

    if cur % k + a[i] > h:
      ans += 1
      cur = 0

    ans += (cur + a[i]) / k
    cur = (cur + a[i]) % k

  else:
    ans += (cur + a[i]) / k
    cur = (cur + a[i]) % k

stdout.write(str(ans + cur / k + bool(cur % k)))
