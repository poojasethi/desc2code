n,h,k = map(int,raw_input().split())
ht = [int(i) for i in raw_input().split()]
curr = 0
ind = 0
ans = 0
while(1):
    if ind == n:
        break
    while((ind!=n) and (curr+ht[ind]<=h)):
        curr += ht[ind]
        ind+=1
    if(ind==n):
        ans += (curr+k-1)/k
        break
    else:
        free = h - curr
        afree = ht[ind]-free
        time = (afree+k-1)/k
        curr -= time*k
        curr = max(0,curr)
        ans += time
print ans
