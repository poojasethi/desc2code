n,h,k = map(int,raw_input().split(' '))

a = map(int,raw_input().split(' '))

s = 0
i = 0
r = 0
while True:
    while i < len(a) and r+a[i]<=h:
        r = r+a[i]
        i += 1
    d = r/k
    r = r-d*k
    s += d
    if r > 0:
        while i < len(a) and r+a[i] <= h:
            r = r+a[i]
            i += 1
        r = max(0,r-k)
        s += 1
    if r == 0 and i >= len(a):
        break

print s
