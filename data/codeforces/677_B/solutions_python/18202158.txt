n,k,h = raw_input().split(" ")
n = int(n)
k = int(k)
h = int(h)
A = raw_input().split(" ")
for i in xrange(n):
	A[i] = int(A[i])
sum = 0
i = 1
buffer = A[0]
while i<n:
	if buffer+A[i]<=k:
		buffer = buffer + A[i]
	else:
		carry = buffer%h
		sum += buffer/h
		buffer = carry 
		if buffer+A[i] > k:
			sum += 1
			buffer = 0
		buffer = buffer + A[i]
	i += 1
carry = buffer%h
if carry!=0:
	sum += 1
sum += buffer/h
print sum
