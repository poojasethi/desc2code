import sys

(n,h,k) = map(int, sys.stdin.next().rstrip().split(' '))

arr = map(int, sys.stdin.next().rstrip().split(' '))


state = 0
ind = 0
time = 0

while state > 0 or ind < len(arr):
    while ind < len(arr) and h - state >= arr[ind]:
        state += arr[ind]
        ind += 1
    if ind == len(arr):
        nxt = h
    else:
        nxt = arr[ind]
    cnt =  max(1, 1+ (nxt + state -h-1 )/k)
    state = max(0, state-cnt*k)
    time += cnt




print time
