n, h, k = raw_input().split()
l = raw_input().split()
n = int(n)
h = int(h)
k = int(k)
for i in range (n):
	l[i] = int(l[i])

i = 0
filled = 0
time = 0
while (True):
	while (filled < h) and (i < n):
		if (filled + l[i] > h):
			break
		else:
			filled = filled + l[i]
			i = i + 1
	if (i == n) and (filled == 0):
		break
	if (filled < k):
		filled = 0
		time = time + 1
	else:
		time = time + filled // k
		filled = filled % k
print(time)
