import math
N, H, K = (int(i) for i in raw_input().split())

# N is number of potatos, H height of food-processor, K is smashed potato every second

pts = [int(i) for i in raw_input().split()]
seconds = 0
ch = 0		# Current height

for item in pts:
	if ch+item <= H:
		ch += item
		continue
	elif ch+item > H:
		cut = int(math.ceil(float(ch - H + item)/K))
		seconds += cut
		ch -= cut*K
		if ch<0:
			ch=0 
		ch = ch+item

if ch!=0:
	seconds += math.ceil(float(ch)/K)

print int(seconds)
