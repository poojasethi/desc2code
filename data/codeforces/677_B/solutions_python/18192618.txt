n,h,k = map(int,raw_input().split())
l = list(map(int,raw_input().split()))

c = 0
h1 = 0
for i in l:
    if i + h1 > h:
        p = (h1 + i) - h
        if p % k == 0:
            c += p / k
            h1 -= k * (p/k)
        else:
            c += p / k + 1
            h1 -= k * (p/k + 1)
        if h1 < 0:
            h1 = 0
    #print c,h1
    h1 += i
    
#print "c = ",c
if h1 > 0:
    if h1 % k == 0: c += h1/k
    else:   c += h1/k + 1
    
print c
