def solve():
    n,h,k=map(int,raw_input().split())
    a=map(int,raw_input().split())
    count=0
    result=0
    for x in xrange(n):
        count+=a[x]
        if count>=k:
            if count>h:
                count=a[x]%k
                result+=1+a[x]/k
            else:
                result+=count/k
                count%=k
    if count>0:
        result+=1
    print result
solve()
