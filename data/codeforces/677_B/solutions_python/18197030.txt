#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

def hh(a, b):
    m = a / b
    r = a % b
    if r>0: m+=1
    return m

n, h, k = ia()
a = ia()

q, i, t  = 0, 0, 0
while True:
    while i<n and q+a[i]<=h: q+=a[i]; i+=1
    if i==n and q==0: break
    if i==n and q!=0: t+=hh(q, k); break

    need = q + a[i] - h
    st   = hh(need, k)
    q -= k*st
    q = max(0, q)
    t += st

print t
