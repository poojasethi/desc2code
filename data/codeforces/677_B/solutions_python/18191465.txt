n, h, k = [int(i) for i in raw_input().split()]
height_arr = [int(i) for i in raw_input().split()]

i = 0
buff = 0
timer = 0
while (i < len(height_arr)) or (buff != 0):
    while (i < len(height_arr)) and (buff + height_arr[i] <= h): 
        buff += height_arr[i]
        i += 1

    if buff>=k: 
        timer += buff/k
        buff = buff%k
    else:
        timer += 1
        buff = 0

print timer
