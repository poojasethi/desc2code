n,h,k=map(int,raw_input().split())
a=map(int,raw_input().split())
ans = 0
filled = 0
i=0
s=0
while 1:
	if i==n and filled==0:
		break
	add_more = 1
	while add_more:
		if i<n and filled<h and a[i]<=h-filled:
			filled += a[i]
			i+=1
		else:
			add_more=0
	ff=filled/k
	if ff>0:
		s+=ff
		filled-=k*ff
	else:
		s+=1
		filled-=k
	if filled<0:
		filled=0
print s
