n,h,k=map(int,raw_input().split())
l=map(int,raw_input().split())

curr_h=0

i,tim=0,0
while 1:
	
	if i>=n:
		break


	if 	curr_h+l[i]<=h:
		curr_h+=l[i]
		i+=1

	else:
		want=l[i]-(h-curr_h)
		rr=(want//k)
		if want%k!=0:
			rr+=1
		curr_h-=(k*rr)
		if curr_h<0:
			curr_h=0
		tim+=rr
		# i+=1

# print curr_h	
tim+=((curr_h//k)+(curr_h%k>0))
print tim
