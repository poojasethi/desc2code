def readval(typ=int):
    return typ( raw_input() )

def readvals(typ=int):
    return map( typ, raw_input().split() )


if __name__=='__main__':
    n,h,k = readvals()
    a = readvals()
    i, t, l = 0, 0, 0
    from math import ceil
    while 1: 
        while i<n and l+a[i]<=h: 
            l += a[i]; i+=1
        if i<n: 
            dt = ceil( (a[i]-h+l)*1.0/k )
        else: dt = ceil( l*1.0/k )
        t += int(dt)
        l = max(l-dt*k, 0)
        if l==0 and i>=n: break
    print t
