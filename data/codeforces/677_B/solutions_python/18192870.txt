n,h,k=map(int,raw_input().split())
a=map(int,raw_input().split())
ans=0
i=1
t=a[0]
while t>0:
    if i<n:
        while(t+a[i]<=h):
            t+=a[i]
            i+=1
            if i>=n:
                break
    if t>k:
        ans+=t/k
        t=t%k
    else:
        t=0
        ans+=1
    if i<n and t==0:
        t=a[i]
        i+=1
print ans
