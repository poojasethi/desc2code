#!/usr/bin/env python

import math

def main():
    n, h, k = map(int, raw_input().split())
    potatos = map(int, raw_input().split())
    ans, pro = 0, 0
    for i in potatos:
        if pro + i > h:
            more = pro + i - h
            times = more / k + (1 if more % k else 0);
            ans += times
            decr = times * k
            pro = 0 if decr >= pro else pro - decr
        pro += i
    else:
        if pro:
            ans += pro / k + (1 if pro % k else 0)
    print ans

if __name__ == '__main__':
    main()
