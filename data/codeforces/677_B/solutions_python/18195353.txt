import math
n,h,k = map(int, raw_input().split())
a = map(int, raw_input().split())
a += [0]
pot = 0
sec = 0
for i in xrange(len(a)):
    if pot + a[i] <= h:
        pot += a[i]
    else:
        des = int(math.ceil((pot + a[i] - h)/float(k)))
#        print pot, des, 'xx'
        sec += des
        pot = max(pot-des*k, 0)
        pot += a[i]
#    print sec
if pot != 0:
    sec += int(math.ceil(pot/float(k)))
print sec
