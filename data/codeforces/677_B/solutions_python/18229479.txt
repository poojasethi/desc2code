
import sys

firstLine = sys.stdin.readline().split()
numPotato = int(firstLine[0])
heightProc = int(firstLine[1])
potatoSmash = int(firstLine[2])

heights = sys.stdin.readline().split()

filled = 0
duration = 0
pieceNo = 0

while numPotato > 0 or filled > 0:
    while numPotato > 0 and int(heights[pieceNo]) + filled <= heightProc:
        filled += int(heights[pieceNo])
        pieceNo+= 1
        numPotato -= 1
    if filled >= potatoSmash:
        mult = filled / potatoSmash
        duration += mult
        filled -= potatoSmash * mult
    else:
        filled = 0
        duration += 1

print duration
