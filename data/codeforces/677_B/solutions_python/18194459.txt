[n,h,k]=[int(x) for x in raw_input().split()]
vlist=[int(x) for x in raw_input().split()]
s=0
re=0
for x in vlist:
    if re+x <=h:
        if re+x >=k:
            re=re+x-k
            s+=1
        else:
            re+=x
    else:
        s+=re/k
        re=re-re/k*k
        if re+x>h:
            re=x
            s+=1
        else:
            re+=x
print s+int(math.ceil(float(re)/k))
