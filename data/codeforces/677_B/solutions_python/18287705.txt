


def main():
	n,h,k=map(int,raw_input().split())
	heights=map(int,raw_input().split())
	seconds=0
	currHeight=heights[0]
	done=False
	idx=1
	while not done:
		while(idx<n and currHeight+heights[idx]<=h):
			currHeight+=heights[idx]
			idx+=1
		if(currHeight/k==0):
			secondsToSmash=1
			currHeight=0
		else:
			secondsToSmash= currHeight/k
			currHeight%=k
		seconds+=secondsToSmash
		if(idx==n and currHeight==0):
			done=True
	print seconds



if __name__=="__main__":
	main()
