h = raw_input().split(' ')
a = raw_input().split(' ')

total = 0
k = int(h[2])
food_processor = 0
height = int(h[1])

for i in a:
	x = int(i)
	if x + food_processor > height: 
		total += 1
		food_processor = x
	else:
		food_processor += x
	total += food_processor/k
	food_processor %= k

if food_processor > 0:
	total += 1

print total
