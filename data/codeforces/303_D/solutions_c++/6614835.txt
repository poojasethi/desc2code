#include<cstdio>
#include<algorithm>
#include<iostream>
#include<vector>
using namespace std;
int n,x,a[1000000],tot;
long long modpow(long long x,long long y,int p){
	long long res=1;
	for(;y;y>>=1,(x*=x)%=p)
		if(y&1)(res*=x)%=p;
	return res;
}
bool check(int b,int p){
	for(int i=1;i*i<p;i++)
		if((p-1)%i==0){
			a[tot++]=i;
			if(i*i!=p-1)
				a[tot++]=(p-1)/i;
		}
	sort(a,a+tot);
	for(int i=0;i<tot;i++)
		if(a[i]!=1&&modpow(b,(p-1)/a[i],p)==1)return 0;
	return 1;
}
int main(){
	cin>>n>>x;
	n++;
	if(x==2)return puts("-1"),0;
	for(int i=2;i<n;i++)
		if(n%i==0)return puts("-1"),0;
	for(int i=x-1;i>=1;i--)
		if(check(i%n,n))
			return printf("%d\n",i),0;
	puts("-1");
}
