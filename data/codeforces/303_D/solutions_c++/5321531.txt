#include<cstdio>
#define REP(i,l,r) for((i)=(l);(i)<=(r);++(i))
using namespace std;

typedef long long LL;

int n,x;

LL power(LL a,LL b,LL Mod)
{
    LL k=1;
    for(;b;b/=2,a=a*a%Mod)
        if(b%2==1)
            k=k*a%Mod;
    return k;
}

int main()
{
    int i;
    scanf("%d%d",&n,&x);
    REP(i,2,n)
        if((n+1)%i==0)
        {
            printf("-1\n");
            return 0;
        }
    for(i=x-1;i>1;--i)
    {
        int flag=0;
        for(int j=2;j*j<=n && !flag;++j)
            if(n%j==0)
                flag|=power(i,j,n+1)==1 || power(i,n/j,n+1)==1;
        if(!flag)
        {
            printf("%d\n",i);
            return 0;
        }
    }
    printf("-1\n");
    return 0;
}
