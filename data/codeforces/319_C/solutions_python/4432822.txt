n=input()
a,b=[map(int,raw_input().split()) for _ in (1,2)]
c=[0]*n
d=[0]
for i in range(1,n):
  while len(d)>1 and (c[d[1]]-c[d[0]])<=a[i]*(b[d[0]]-b[d[1]]):d.pop(0)
  c[i]=c[d[0]]+b[d[0]]*a[i]
  while len(d)>1 and (c[d[-1]]-c[d[-2]])*(b[d[-1]]-b[i])>(c[i]-c[d[-1]])*(b[d[-2]]-b[d[-1]]):d.pop()
  d+=[i]
print c[-1]
