#include<stdio.h>
__int64 f[100005],a[100005],b[100005],l[100005];
__int64 K1(int x,int i){
	return f[x]+b[x]*a[i];
}
double K2(int x,int y){
	return (f[y]-f[x])/(0.+b[x]-b[y]);
}
__int64 n,i,head,tail;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%I64d",&a[i]);
	for(i=1;i<=n;i++)scanf("%I64d",&b[i]);
	head=1;tail=0;
	for(i=1;i<=n;i++){
		while(head<tail&&K1(l[head],i)>=K1(l[head+1],i))head++;
		f[i]=f[l[head]]+b[l[head]]*a[i];
		while(head<tail&&K2(i,l[tail])<=K2(l[tail],l[tail-1]))tail--;
		l[++tail]=i;				
	}
	printf("%I64d\n",f[n]);
}
