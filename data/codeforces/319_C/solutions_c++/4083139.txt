#include <stdio.h>
#include <string.h>
#define MM 100005
__int64 a[MM],b[MM],dp[MM],fa[MM],tou,wei;
int main()
{
	int N;
	scanf("%d",&N);
	for(int i=1;i<=N;i++)
	scanf("%I64d",&a[i]);
	for(int i=1;i<=N;i++)
	scanf("%I64d",&b[i]);
	memset(dp,0,sizeof(dp));
	tou=0;wei=1;fa[0]=1;
	for(int i=2;i<=N;i++)
	{
		while(tou<wei-1
		&&dp[fa[tou]]+a[i]*b[fa[tou]]
		>=dp[fa[tou+1]]+a[i]*b[fa[tou+1]])
		tou++; 
		dp[i]=dp[fa[tou]]+a[i]*b[fa[tou]];
		while(tou<wei-1
		&&(dp[i]-dp[fa[wei-1]])*1.0/(b[i]-b[fa[wei-1]])
		>=(dp[fa[wei-1]]-dp[fa[wei-2]])*1.0/(b[fa[wei-1]]-b[fa[wei-2]]))
		wei--;
		fa[wei]=i;
		wei++;
	}
	printf("%I64d\n",dp[N]);
}
