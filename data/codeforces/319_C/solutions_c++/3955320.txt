#include <cstdio>
typedef long long LL;
const int Maxn=100001;
LL a[Maxn],b[Maxn],f[Maxn];
int o[Maxn];

int main()
{
    int n,i;
    scanf("%d",&n);
    for (i=1; i<=n; ++i) scanf("%I64d",a+i);
    for (i=1; i<=n; ++i) scanf("%I64d",b+i);
    f[1]=0; o[1]=1;
    int s=1, t=1;
    for (i=2; i<=n; ++i)
    {
        while (s<t && f[o[s]]-f[o[s+1]]>=a[i]*(b[o[s+1]]-b[o[s]])) ++s;
        int j=o[s];
        f[i]=f[j]+a[i]*b[j];
        while (s<t && (double)(f[o[t-1]]-f[o[t]])*(b[i]-b[o[t]])>=(double)(f[o[t]]-f[i])*(b[o[t]]-b[o[t-1]])) --t;
        o[++t]=i;
    }
    printf("%I64d\n",f[n]);
    return 0;
}
