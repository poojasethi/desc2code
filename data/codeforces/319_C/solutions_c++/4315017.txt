#include <stdio.h>
#define maxn (200000)
long long a[maxn],b[maxn];
int dl[maxn],p,q,n;
long long f[maxn];

double xx(int x,int y){
	return (double)(f[y]-f[x])/(double)(b[x]-b[y]);
}

int main(){
	int i;
	scanf("%d",&n);
	for (i=1;i<=n;++i) scanf("%I64d",&a[i]);
	for (i=1;i<=n;++i) scanf("%I64d",&b[i]);
	f[1]=0;
	dl[1]=q=p=1;
	for (i=2;i<=n;++i){
		while (p<q &&   xx(dl[p],dl[p+1])<(double)a[i]) ++p;
		f[i]=f[dl[p]]+(long long)b[dl[p]]*a[i];
		while (p<q &&  xx(dl[q-1],dl[q])>=xx(dl[q],i)) --q;
		dl[++q]=i;
	}
	printf("%I64d",f[n]);
	return 0;
}
