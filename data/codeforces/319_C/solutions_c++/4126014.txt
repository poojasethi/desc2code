#include <stdio.h>
#include <string.h>
typedef __int64 ll;
ll q[100001],a[100001],b[100001],dp[100001],n;
ll slope(ll j,ll k)
{
	return (dp[k]-dp[j])/(b[j]-b[k]);
}
int main()
{
	while(scanf("%I64d",&n)!=EOF)
	{
		for(ll i=1;i<=n;i++)
		scanf("%I64d",&a[i]);
		for(ll i=1;i<=n;i++)
		scanf("%I64d",&b[i]);
		ll head=0,tail=0;
		for(ll i=1;i<=n;i++)
		{
			while(head<tail&&slope(q[head],q[head+1])<a[i])
			head++;
			dp[i]=dp[q[head]]+a[i]*b[q[head]];
			while(head<tail&&slope(q[tail-1],q[tail])>slope(q[tail],i))
			tail--;
			tail++;
			q[tail]=i;
		}
		printf("%I64d\n",dp[n]);
	}
}
