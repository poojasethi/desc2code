#include<iostream>
using namespace std;
int n;
long long a[110000];
long long b[110000];
int q[110000];
int l,r;
const double LIT=1e-10;
long long F[110000];
bool D(int p1,int p2,int p3){
	double ret=double(F[p2]-F[p1])/(b[p1]-b[p2])-double(F[p3]-F[p2])/(b[p2]-b[p3]);
	if(ret>=-LIT)return true;
	else return false;
}
int main(void){
	cin>>n;
	for(int i=1;i<=n;i++)cin>>a[i];
	for(int i=1;i<=n;i++)cin>>b[i];
	F[1]=0;
	l=r=1;
	q[1]=1;
	for(int i=2;i<=n;i++){
		while(l<r&&double(F[q[l+1]]-F[q[l]])/(b[q[l]]-b[q[l+1]])-a[i]<=LIT)l++;
		F[i]=F[q[l]]+b[q[l]]*a[i];
		while(l<r&&D(q[r-1],q[r],i))r--;
		q[++r]=i;
	}
	cout<<F[n]<<endl;
}
