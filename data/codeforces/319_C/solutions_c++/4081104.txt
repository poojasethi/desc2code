#include<cstdio>
#include<cstring>
#include<algorithm>
#define rep(i,n) for (int i=1;i<=n;i++)
#define F(x) f[x]+b[x]*a[i]
#define G(x,y) (f[y]-f[x])/(0.+b[x]-b[y])
#define N 100005
long long a[N],b[N],f[N];
int n,h,t,q[N];
int main()
{
	scanf("%d",&n);
	rep(i,n) scanf("%I64d",a+i);
	rep(i,n) scanf("%I64d",b+i);
	int r=0;
	int l=1;
	rep(i,n){
		while(l<r&&F(q[l])>=F(q[l+1]))l++;
		f[i]=f[q[l]]+b[q[l]]*a[i];
		while (l<r && G(i,q[r])<=G(q[r],q[r-1])) r--;
		q[++r]=i;
		}
	printf("%I64d\n",f[n]); return 0;
}
