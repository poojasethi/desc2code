#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;

const int N = 100005;
int a[N], b[N], l[N], n;
ll f[N];
double slope(int j, int k) {
	return -((double)(f[j] - f[k])) / (b[j] - b[k]);
}
int main() {
	scanf("%d", &n);
	for (int i=1; i<=n; ++i) scanf("%d", &a[i]);
	for (int i=1; i<=n; ++i) scanf("%d", &b[i]);
	f[1] = 0; l[1] = 1;
	int ls = 1, le = 1;
	for (int i=2; i<=n; ++i) {
		for (; ls<le && slope(l[ls], l[ls+1]) <= a[i]; ++ls);
		f[i] = f[l[ls]] + (ll)b[l[ls]] * a[i];
		for (; ls<le && slope(l[le], i) <= slope(l[le-1], l[le]); --le);
		l[++le] = i;
	}
	printf("%I64d\n", f[n]);
	return 0;
}
