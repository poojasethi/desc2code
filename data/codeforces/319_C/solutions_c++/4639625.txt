#include <cstdio>
#include <cstring>
#include <cmath>
using namespace std;
#define N 100200
typedef long long ll;
int q[N], n;
ll a[N], b[N], f[N];
double g(int j, int k) {
    return (double)(f[j]-f[k])/(b[k]-b[j]);
}
int main() {
    scanf("%d", &n);
    for (int i=1; i<=n; i++) scanf("%I64d", &a[i]);
    for (int i=1; i<=n; i++) scanf("%I64d", &b[i]);

    int h = 0, t = 0;
    q[t] = 1;
    f[1] = 0;
    for (int i=2; i<=n; i++) {
        while (h < t && g(q[h], q[h+1]) < a[i]) h++;
        f[i] = f[q[h]] + b[q[h]]*a[i];
        while (h < t && g(q[t], i) <= g(q[t-1], q[t])) --t;
        q[++t] = i;
    }

    printf("%I64d\n", f[n]);

    return 0;
}
