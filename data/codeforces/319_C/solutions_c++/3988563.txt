#include <cstdio>
using namespace std;
const int NMax=110000;
int N;
long long A[NMax],B[NMax],dp[NMax];
int Q[NMax],L,R;
int main()
{
    scanf("%d",&N);
    for(int i=1;i<=N;i++) scanf("%I64d",A+i);
    for(int i=1;i<=N;i++) scanf("%I64d",B+i);
    L=0;R=0;Q[L]=1;
    for(int i=2;i<=N;i++) {
        //int leftmost=Q[L],rightmost=Q[R];
        while(R>L&&(double)(dp[Q[L+1]]-dp[Q[L]])/(B[Q[L]]-B[Q[L+1]])<(double)A[i]) L++;
        dp[i]=dp[Q[L]]+B[Q[L]]*A[i];
        while(R>L&&(double)(dp[i]-dp[Q[R]])/(B[Q[R]]-B[i])<(double)(dp[Q[R]]-dp[Q[R-1]])/(B[Q[R-1]]-B[Q[R]])) R--;
        Q[++R]=i;
    }
    printf("%I64d\n",dp[N]);
    getchar();getchar();
    return 0;
}
