#include<stdio.h>
#include<cstring>
#include<iostream>
#define XL(p,q) ((double)(dp[(p)]-dp[(q)])/(double)(b[(q)]-b[(p)]))
using namespace std;
#define MAXN 100010
#define Min(a,b) ((a)>(b))?(b):(a)
long long N,a[MAXN],b[MAXN],dp[MAXN],q[MAXN],h,t;
int main()
{
	int i,j;
	cin >> N;
	for (i=1;i<=N;i++) cin >> a[i];
	for (i=1;i<=N;i++) cin >> b[i];
	memset(dp,0x7f,sizeof dp);
	dp[1]=0; q[1]=h=t=1;
	for (i=2;i<=N;i++)
	{
		while (h<t && XL(q[h+1],q[h])<a[i]) h++;
		dp[i]=dp[q[h]]+a[i]*b[q[h]];
		while (h<t && XL(q[t-1],q[t])>=XL(q[t],i)) t--;
		q[++t]=i;
	}
	cout << dp[N];
	return 0;		
}
