#include <bits/stdc++.h>
#define big long long
#define rep(i) for(int i=1;i<=N;++i)
#define val(j) (f[j]+b[j]*a[i])
#define sec(i, j) ((f[i]-f[j])/(0.+b[j]-b[i]))
using namespace std;

const int MAXN = 2e5;
big a[MAXN], b[MAXN], f[MAXN], q[MAXN], p, Q, N;

int main() {
	cin>>N;
	rep(i) cin>>a[i];
	rep(i) cin>>b[i];
	rep(i) {
		while(p<Q-1&&val(q[p+1])<val(q[p])) ++p;
		f[i] = val(q[p]);
		while(Q>1&&sec(q[Q-2], i)<= sec(q[Q-2], q[Q-1])) --Q;
		q[Q++]=i;
	}
	cout<<f[N]<<'\n';
}
