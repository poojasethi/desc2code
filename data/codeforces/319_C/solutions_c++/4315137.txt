#include<stdio.h>
#define calc(j,k) (f[j]-f[k])/(b[k]-b[j])
const int maxn=100050;
long long a[maxn],b[maxn],ans;
long double f[maxn];
int n,qu[maxn]={0},st=1,top=0;
int main()
{
    //freopen("data.in","r",stdin);
    int i;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
        scanf("%I64d",&a[i]);
    for(i=1;i<=n;i++)
        scanf("%I64d",&b[i]);
    for(i=1;i<=n;i++)
        {
        while(st<top&&calc(qu[st],qu[st+1])<=a[i]) st++;
        f[i]=f[qu[st]]+a[i]*b[qu[st]];  
        while(top>st&&calc(qu[top-1],qu[top])>calc(qu[top],i)) top--;
        qu[++top]=i;
        }
    printf("%I64d\n",(ans=f[n]));
    return 0;
}
