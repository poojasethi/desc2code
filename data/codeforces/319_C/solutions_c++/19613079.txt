#include <bits/stdc++.h>
#define big long long
#define rep(i) for(int i=1;i<=N;++i)
#define val(j) (f[j]+b[j]*a[i])
#define sec(i, j) ((f[i]-f[j])/(0.+b[j]-b[i]))
using namespace std;

const int MAXN = 2e5;
big a[MAXN], b[MAXN], f[MAXN], q[MAXN], p, Q, N;

int main() {
	cin>>N;
	rep(i) cin>>a[i];
	rep(i) cin>>b[i];
	rep(i) {
		while(p<Q-1&&val(q[p+1])<val(q[p])) ++p;
		f[i] = val(q[p]);
		while(Q>1&&((f[q[Q-2]]-f[i])/(0.+b[i]-b[q[Q-2]]))<((f[q[Q-2]]-f[q[Q-1]])/(0.+b[q[Q-1]]-b[q[Q-2]]))) --Q;
		q[Q++]=i;
	}
	cout<<f[N]<<'\n';
}
