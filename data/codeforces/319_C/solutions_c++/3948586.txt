#include<cstdio>

#define MAXX 100111
#define eps 1e-8

long long a[MAXX],b[MAXX];
long long dp[MAXX];
long long q[MAXX],qf,qt;

int n,i,j,k;

int main()
{
    scanf("%d",&n);
    for(i=1;i<=n;++i)
        scanf("%I64d",a+i);
    for(i=1;i<=n;++i)
        scanf("%I64d",b+i);
    q[++qf]=qt=1;
    for(i=2;i<=n;++i)
    {
        while(qf<qt && dp[q[qf+1]]-dp[q[qf]]<=a[i]*(b[q[qf]]-b[q[qf+1]]))
            ++qf;
        dp[i]=a[i]*b[q[qf]]+dp[q[qf]];

        while(qf<qt && (double)(dp[q[qt]]-dp[q[qt-1]])*(b[q[qt]]-b[i])-(double)(dp[i]-dp[q[qt]])*(b[q[qt-1]]-b[q[qt]])>-eps)
            --qt;
        q[++qt]=i;
    }
    printf("%I64d\n",dp[n]);
    return 0;
}
