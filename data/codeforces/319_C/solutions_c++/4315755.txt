#include<stdio.h>
#include<string.h>
#define xx(p,q) ((double)(f[(p)]-f[(q)])/(double)(b[(q)]-b[(p)]))
long long f[100005],a[100005],b[100005],q[100005],ed,st,n;
int main()
{
	long long i;
	scanf("%I64d",&n);
	for(i=1;i<=n;i++)scanf("%I64d",&a[i]);
	for(i=1;i<=n;i++)scanf("%I64d",&b[i]);
	memset(f,0x7f,sizeof(f));
	f[1]=0;
	q[1]=ed=st=1;
	for(i=2;i<=n;i++){
	  while(ed>st&&(xx(q[st+1],q[st]))<a[i])st++;
	  f[i]=f[q[st]]+a[i]*b[q[st]];
	  while(ed>st&&xx(q[ed-1],q[ed])>=xx(q[ed],i))ed--;
	  q[++ed]=i;
	}
	printf("%I64d\n",f[n]);
	return 0;
}
