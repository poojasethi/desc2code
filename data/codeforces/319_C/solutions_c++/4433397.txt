#include <cstdio>
using namespace std;
const int maxn=100010;
long long dp[maxn],a[maxn],b[maxn],p[maxn];
double f(long long x,long long y){
    return (double)(dp[x]-dp[y])/(double)(b[y]-b[x]);
}
int main(){
    int n,i,c,n1;
    scanf("%d",&n);
    for (i=1;i<=n;i++) scanf("%d",&a[i]);
    for (i=1;i<=n;i++) scanf("%d",&b[i]);
    dp[1]=0;
    p[1]=1;
    c=1;n1=1;
    for (i=2;i<=n;i++){
        while (c<n1&&a[i]>=f(p[c],p[c+1])) c++;
        dp[i]=dp[p[c]]+a[i]*b[p[c]];
        while (n1>c&&f(p[n1],i)<=f(p[n1-1],p[n1])) n1--;
        n1++;
        p[n1]=i;
    }
    printf("%I64d\n",dp[n]);
    return 0;
}
