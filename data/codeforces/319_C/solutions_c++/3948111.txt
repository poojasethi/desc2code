#include<cstdio>
#include<cstring>
#include<algorithm>
#define rep(i,n) for (int i=1;i<=n;i++)
#define F(x) f[x]+b[x]*a[i]
#define G(x,y) (f[y]-f[x])/(0.+b[x]-b[y])
#define N 100005
long long a[N],b[N],f[N];
int n,h,t,q[N];
int main()
{
	scanf("%d",&n),h=1;
	rep(i,n) scanf("%I64d",a+i);
	rep(i,n) scanf("%I64d",b+i);
	rep(i,n){
		while (h<t && F(q[h])>=F(q[h+1])) h++;
		f[i]=f[q[h]]+b[q[h]]*a[i];
		while (h<t && G(i,q[t])<=G(q[t],q[t-1])) t--;
		q[++t]=i;
		}
	printf("%I64d\n",f[n]); return 0;
}
