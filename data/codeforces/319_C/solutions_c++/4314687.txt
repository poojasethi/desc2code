#include<stdio.h>
#include<string.h>
#include<stdlib.h>
const int maxn=100001;
int n,q[maxn],h,l=1;
long long a[maxn],b[maxn],f[maxn];
double xx(int x,int y)
{
	return double(f[x]-f[y])/double(b[y]-b[x]);
}
int main()
{
	int i,j,k;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	scanf("%I64d",&a[i]);
	for(i=1;i<=n;i++)
	scanf("%I64d",&b[i]);
	f[1]=0;
	q[++h]=1;
	for(i=2;i<=n;i++)
	{
		while(h>l&&xx(q[l+1],q[l])<double(a[i]))
		l++;
		f[i]=f[q[l]]+b[q[l]]*a[i];
		q[++h]=i;
		for(j=h-1;j>l;j--)
		{
			if(xx(q[j-1],q[j])>=xx(q[j],q[j+1]))
			q[j]=q[h--];
			else
			break;
		}
	}
	printf("%I64d\n",f[n]);
	return 0;
}
