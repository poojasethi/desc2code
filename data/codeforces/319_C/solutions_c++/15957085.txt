#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
typedef long long LL;
const int maxn=1e5+10;
LL a[maxn],b[maxn],f[maxn];
int q[maxn];
bool better(int j,int k,LL x)
{
	return (f[j]+b[j]*x)<(f[k]+b[k]*x);
}
double getx(int i,int j)
{
	return (f[i]-f[j])/(b[i]-b[j]);
}
int main()
{
	int n,i,j,l,r;
	while (cin>>n)
	{
		for (i=1;i<=n;++i) cin>>a[i];
		for (i=1;i<=n;++i) cin>>b[i];
		f[1]=0;
		l=r=1;
		q[1]=1;
		for (i=2;i<=n;++i)
		{
			while (l<r&&!better(q[l],q[l+1],a[i])) ++l;
			j=q[l];
			f[i]=f[j]+b[j]*a[i];
			while (l<r&&getx(i,q[r])>getx(q[r],q[r-1])) --r;
			q[++r]=i;
		}
		cout<<f[n]<<endl;
	}
	return 0;
}
