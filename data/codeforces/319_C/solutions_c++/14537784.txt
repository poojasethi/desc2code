#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll dp[100003];
ll a[100001],b[100001],n;
ll q[100002];
int main(){
	cin>>n;int i;
	for(i=1;i<=n;++i)cin>>a[i];
	for(i=1;i<=n;++i)cin>>b[i];
	q[0]=b[0];
	int l=1,t=0;
#define f(x) (dp[x]+b[x]*a[i])
#define limit(i,t) ((dp[i]-dp[t] )/(0.+b[t]-b[i]))
	for(i=1;i<=n;++i){
		while(l<t && f(q[l])>=f(q[l+1]) )++l;
		dp[i]= dp[q[l] ]+b[q[l]]*a[i];
		while(l<t && limit(i,q[t])<=limit(q[t],q[t-1]) )
			--t;
		q[++t]=i;
	}
	cout<<dp[n]<<endl;
}
