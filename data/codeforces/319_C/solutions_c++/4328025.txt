#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
#define maxn 100005
#define LL long long
LL a[maxn],b[maxn],f[maxn];
LL q[maxn],h,t;
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;++i)cin>>a[i];
    for(int i=0;i<n;++i)cin>>b[i];
    h=t=0;
    q[t++]=0;
    for(int i=1;i<n;++i)
    {
        while(t-h>1&&f[q[h+1]]-f[q[h]]<=a[i]*(b[q[h]]-b[q[h+1]]))++h;
        f[i]=f[q[h]]+a[i]*b[q[h]];
        while(t-h>1&&1.0*(f[i]-f[q[t-1]])/(b[i]-b[q[t-1]])>=1.0*(f[q[t-1]]-f[q[t-2]])/(b[q[t-1]]-b[q[t-2]]))--t;
        q[t++]=i;
    }
    cout<<f[n-1]<<endl;
}
