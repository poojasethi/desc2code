#include <iostream>
using namespace std;
#define N 222222

int S[N],t=0;
long long dp[N],a[N],b[N];
double get(int x,int y) {
    return double(dp[y]-dp[x])/(b[x]-b[y]);
}
int main() {
    int n,k=0,i=0;
    for(cin>>n;i<n;i++) cin>>a[i];
    for(i=0;i<n;i++) cin>>b[i];
    for(i=0;i<n;i++) {
        while(k+1 < t && b[S[k+1]]*a[i]+dp[S[k+1]] < b[S[k]]*a[i]+dp[S[k]]) k++;
        if(i==0) dp[i] = 0;
        else dp[i] = b[S[k]]*a[i] + dp[S[k]];
        S[t++] = i;
        while(t>2 && (get(S[t-3],S[t-1])<=get(S[t-3],S[t-2])))
            S[t-2] = S[t-1],t--;
    }
    cout << dp[n-1];
    return 0;
}
