#include <cstdio>
#include <algorithm>
#define oo 999999999987654321LL
int n;
long long a[101010], b[101010], dt[101010], q[101010], h, t;

double f(int x, int y)
{
    return (double)(dt[x]-dt[y])/(b[y]-b[x]);
}

int main()
{
    scanf("%d", &n);
    for(int i = 0 ; i < n ; i++ ) scanf("%d", a+i);
    for(int i = 0 ; i < n ; i++ ) scanf("%d", b+i);
    q[t++] = 0;
    for(int i = 1 ; i < n ; i++ )
    {
        while( t-h > 1 && f(q[h], q[h+1]) < (double)a[i] ) h++;
        dt[i] = dt[q[h]] + a[i]*b[q[h]];
        while( t-h > 1 && f(q[t-1], i) <= f(q[t-2], q[t-1]) ) t--;
        q[t++] = i;
    }
    printf("%lld", dt[n-1]);
}
