#include<cstdio>
#include<cstring>
#include<algorithm>
#define rep(i,n) for (int i=1;i<=n;i++)
#define F(x) f[x]+b[x]*a[i]
#define G(x,y) (f[y]-f[x])/(b[x]-b[y])
#define N 100005
long long a[N],b[N],f[N];
int n,h,t,q[N];
int main()
{
	scanf("%d",&n),h=1;
	rep(i,n) scanf("%I64d",a+i);
	rep(i,n) scanf("%I64d",b+i);
	rep(i,n){
		while (h<t && (f[q[h+1]]-f[q[h]]<=(b[q[h]]-b[q[h+1]])*a[i])) h++;
		f[i]=f[q[h]]+b[q[h]]*a[i];
		while (h<t && ((f[q[t-1]]-f[q[t]])/(b[q[t]]-b[q[t-1]])>=(f[q[t]]-f[i])/(b[i]-b[q[t]]))) t--;
	  //  while (h<t && ((f[q[t-1]]-f[q[t]])/(b[t]-b[q[t-1]])>=(f[q[t]]-f[i])/(b[q[i]]-b[q[t]]))) t--;
		q[++t]=i;
		}
	printf("%I64d\n",f[n]); 
    return 0;
}
