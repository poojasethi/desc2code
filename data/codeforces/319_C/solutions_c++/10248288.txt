#include<stdio.h>
#include<string.h>
    __int64 dp[111111];
    __int64 a[111111];
    __int64 b[111111];
    int q[111111];
double g(int j,int k)
{
    return (dp[j]-dp[k])*1.0/(b[k]-b[j]);
}
int main()
{

    memset(dp,0,sizeof(dp));
    int h=0,t=0,i;
    int n;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    scanf("%I64d",&a[i]);
    for(i=1;i<=n;i++)
    scanf("%I64d",&b[i]);
    q[++t]=1;
    for(i=2;i<=n;i++)
    {
        while(h+1<t&&g(q[h+1],q[h+2])<a[i])
        ++h;
        dp[i]=b[q[h+1]]*a[i]+dp[q[h+1]];
        while(h+1<t&&g(q[t-1],q[t])>=g(q[t],i))
        --t;
        q[++t]=i;
        
    }
    printf("%I64d\n",dp[n]);
}
