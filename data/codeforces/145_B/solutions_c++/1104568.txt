#include<iostream>

using namespace std;

int main()
{
    int c4,c7,c47,c74;
    cin>>c4>>c7>>c47>>c74;
    string A = "";
    for(int i = 0; i < c47; ++i)
        A += "47" , c74-- , c7-- , c4--;
    c74++;
    if(c74 <= 2 && c74 >= 0 && c7 >= 0 && c4 >= 0)
    {
        if(c74==1)
        {
            if(c4>0)A = A+"4" , c4--;
            else A = "7"+A , c7--;
        }
        else if(c74==2)A = A.substr(1)+"474", c7-- , c74-- , c4--;
        if(c7<0 || c4 < 0)
            A = "-1";
        else
        {
            string B4 = string(c4 , '4');
            string B7 = string(c7 , '7');
            int p4 = A.find("4");
            A = A.substr(0,p4) + B4 + A.substr(p4);
            int p7 = A.rfind("7");
            A = A.substr(0,p7) + B7 + A.substr(p7);
        }
    }
    else
        A = "-1";
    cout << A << endl;
}
