#include<bits/stdc++.h>

using namespace std;

#define LL long long
#define mod 1000000007

int main()
{
    int mask,a1,a2,a3,a4,i;
    cin>>a1>>a2>>a3>>a4;

    a1-=a3;
    a2-=a3;
    a4-=a3-1;

    mask = 0;
    if(a4==2)
    {
        mask|=3;
        a2--;
        a1--;
    }
    else if(a4==1)
    {
        if(a1)
        {
            a1--;
            mask|=1;
        }
        else
        {
            a2--;
            mask|=2;
        }
    }

    if(a1<0 || a2<0 || a4>2 || a4<0)
        cout<<"-1";
    else
    {
        if(mask&2)  cout<<"7";
        for(i=0;i<a1;i++)   cout<<"4";
        for(i=0;i<a3;i++)   cout<<"47";
        for(i=0;i<a2;i++)   cout<<"7";
        if(mask&1)  cout<<"4";
    }
    return 0;
}
