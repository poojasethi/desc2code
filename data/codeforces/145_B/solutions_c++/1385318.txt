using namespace std;

string f(int a1, int a2, int a3, int a4)
{
	if(a1 >= a3 && a1 >= a4 && a2 >= a3 && a2 >= a4)
	{
		string s(a1-a3, '4');
		for(int i=0; i<a3; i++)
			s += "47";
		s += string(a2-a3, '7');
		
		if(a3 - 1 > a4) return "-1";
		else if(a3 - 1 == a4) return s;
		else if(a3 == a4)
		{
			if(a1-a3>0) return s.substr(1) + "4";
			else if(a2-a3>0) return "7" + s.substr(0, s.size()-1);
			else return "-1";
		}
		else if(a3 == a4 - 1)
		{
			if(a2-a3>0 && a1-a3>0)
			{
				swap(s[0], s[s.size()-1]);
				
				return s;		
			} 
			else return "-1";
		}
		else return "-1";
	}
	else return "-1";
}

int main()
{
	int a1, a2, a3, a4;
	while(cin>>a1>>a2>>a3>>a4)
	{
		cout<<f(a1, a2, a3, a4)<<endl;
	}
	
	return 0;
}
