#include<stdio.h>
#include<iostream>
#include<string>
using namespace std;
int main(){
	int a,b,c,d,i;
	bool have;
	string ans,ans1,ans3;
	while(scanf("%d%d%d%d",&a,&b,&c,&d)!=EOF){
		if(c>a||c>b||d>a||d>b){
			printf("-1\n");
			continue;
		}
		if(c!=d&&c!=d+1&&d!=c+1){
			printf("-1\n");
			continue;
		}
		have=1;
		ans="";
		ans1="";
		ans3="";
		if(d>c){
			b--;
			d--;
			ans3="7";
		}
		for(i=0;i<c;i++){
			ans+="47";
		}
		a-=c;
		b-=c;
		d-=(c-1);
		if(d==1){
			if(a>=1){
				for(i=0;i<b;i++){
					ans+='7';
				}
				ans+='4';
				a--;
			}
			else if(b>=1){
				b--;
				for(i=0;i<b;i++){
					ans+='7';
				}
				ans3="7";
			}
			else{
				have=0;
			}
		}
		else{
			for(i=0;i<b;i++){
				ans+='7';
			}
		}
		if(!have){
			printf("-1\n");
			continue;
		}
		for(i=0;i<a;i++){
			ans1+='4';
		}
		ans=ans1+ans;
		ans=ans3+ans;
		if(have){
			cout<<ans<<endl;
		}
		else{
			printf("-1\n");
		}
	}
	return 0;
}
