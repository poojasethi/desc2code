#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n, a[4];
char s[2000005];

int chk(int d){
	if (d && s[d-1] == '7') a[3]--;
	int ok = 0;

	if (a[2] == a[3]){
		if (a[0] >= a[2] && a[1] >= a[2]) ok = 1;
	}
	if (a[2] - 1 == a[3]){
		if (a[0] + 1 >= a[2] && a[1] >= a[2]) ok = 1;
	}
	if (a[3] < 0) ok = 0;
	if (a[1] && !a[2]) ok = 0;
	
 	if (d && s[d-1] == '7') a[3]++;
 	return ok;
}

int main(){
	for (int i=0; i<4; i++) scanf("%d", &a[i]);
	n = a[0] + a[1];
	
	for (int i=0; i<n; i++){
		if (!a[0]) s[i] = '7', a[1]--;
		else if (!a[1]) s[i] = '4', a[0]--;
		else{
			s[i] = '4', a[0]--;
			if (!chk(i)){
				a[0]++;
				s[i] = '7';
				a[1]--;
			}
		}
		
		if (i){
			if (s[i-1] == '4' && s[i] == '7') a[2]--;
			if (s[i-1] == '7' && s[i] == '4') a[3]--;
		}
	}
	
	s[n] = 0;
	if (a[2] || a[3]) puts("-1");
	else puts(s);
	return 0;
}
