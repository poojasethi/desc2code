#include <iostream>
#include <cmath>
using namespace std;

int a,b,c,d;
bool av(int a,int b,int c,int d,int last)
{
    if (a<0 || b<0 || c<0 || d<0) return 0;
    if (last==4 && b>0 && c==0) return 0;
    if (last==7 && a>0 && d==0) return 0;
    if ((a+(last==4))<c || a<d) return 0;
    if (b<c || (b+(last==7))<d) return 0;
    if (abs(c-d)>1) return 0;
    if (last==4 && d>c) return 0;
    if (last==7 && c>d) return 0;
    return 1;
}
int main()
{
    cin>>a>>b>>c>>d;
    int last=-1;
    int e=a+b;
    for (int i=0;i<e;i++)
    {
        if (av(a-1,b,c,d-(last==7),4))
        {
            a--;
            d-=(last==7);
            last=4;
            cout<<4;
        }
        else if (av(a,b-1,c-(last==4),d,7))
        {
            b--;
            c-=(last==4);
            last=7;
            cout<<7;
        }
        else
        {
            cout<<-1<<endl;
            return 0;
        }
    }
    cout<<endl;
    return 0;
}
