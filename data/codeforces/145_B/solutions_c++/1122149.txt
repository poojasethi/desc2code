    a1 -= a3;
    a2 -= a3;
    int p = 0;
    if (a4 == 2) {
        a1--;
        a2--;
        p = 3;
    } else if (a4 == 1) {
        if (a1) {
            a1--;
            p = 1;
        } else {
            a2--;
            p = 2;
        }
    }
    if (a1 < 0 || a2 < 0 || a4 > 2 || a4 < 0) {
        cout << -1 << endl;
    } else {
        if (p & 2) cout << 7;

}
