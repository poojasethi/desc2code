#include <iostream>
#include <stdio.h>
using namespace std;

int main() {
  int a1, a2, a3, a4; cin>>a1>>a2>>a3>>a4;
  if (a3 == a4 + 1 && a1 >= a3 && a2 >= a3) {
    for (int i = 0; i < a1 - a3; ++i) printf("4");
    for (int i = 0; i < a3; ++i) printf("47");
    for (int i = 0; i < a2 - a3; ++i) printf("7");
  } else if (a4 == a3 + 1 && a1 >= a4 && a2 >= a4) {
    printf("74");
    for (int i = 0; i < a1 - a4; ++i) printf("4");
    for (int i = 0; i < a4 - 2; ++i) printf("74");
    for (int i = 0; i < a2 - a4 + 1; ++i) printf("7");
    printf("4");
  } else if (a4 == a3 && a1 >= a3 + 1 && a2 >= a3) {
    for (int i = 0; i < a1 - a3 - 1; ++i) printf("4");
    for (int i = 0; i < a3; ++i) printf("47");
    for (int i = 0; i < a2 - a3; ++i) printf("7");
    printf("4");
  } else if (a4 == a3 && a2 >= a3 + 1 && a1 == a3) {
    for (int i = 0; i < a3; ++i) printf("74");
    for (int i = 0; i < a2 - a3; ++i) printf("7");
  } else printf("-1");
  printf("\n");
}
