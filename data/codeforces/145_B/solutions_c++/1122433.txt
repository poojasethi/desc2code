#include <iostream>
#include <algorithm>
#define rep(i,n) for(int i=0;i<n;i++)
#define ab(i) ((i)>0?(i):(-(i)))
using namespace std;

int a1,a2,a3,a4;

int main(){

  cin>>a1>>a2>>a3>>a4;
  if(max(a3,a4)>min(a1,a2)||ab(a3-a4)>1){
    cout<<-1<<endl;return 0;
  }
  if(a3==a4){
    if(a1>=a3+1){
      rep(i,a1-a3-1)cout<<4;
      rep(i,a3-1)cout<<47;cout<<4;      
      rep(i,a2-a4+1)cout<<7;cout<<4<<endl;
    }
    else if(a2>=a4+1){
      cout<<7;rep(i,a1-a3+1)cout<<4;
      rep(i,a3-1)cout<<74;rep(i,a2-a3)cout<<7;cout<<endl;
    }
    else  cout<<-1<<endl;return 0;
  }
  else if(a3==a4+1){
    rep(i,a1-a3)cout<<4;
    rep(i,a4)cout<<47;cout<<4;
    rep(i,a2-a4)cout<<7;cout<<endl;
  }
  else {
    cout<<7;rep(i,a1-a3)cout<<4;
    rep(i,a3-1)cout<<74;rep(i,a2-a3)cout<<7;cout<<4<<endl;
  }

  return 0;

}
