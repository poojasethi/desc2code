#include<stdio.h>
#include<math.h>
#include<iostream>
#include<stdlib.h>
using namespace std;
int main()
{
	int a,b,c,d,i,j;
	while(scanf("%d%d%d%d",&a,&b,&c,&d)!=EOF)
	{
		if(fabs(c-d)>1||max(c,d)>min(a,b)||c+d>=a+b){printf("-1\n");continue;}
		if(d>c){printf("7");d--;b--;}
		if(c==d)
		{
			if(a>c)
			{
				for(i=1;i<=a-c-1;i++)
				printf("4");
				for(i=1;i<=c;i++)
				printf("47");
				for(i=1;i<=b-c;i++)
				printf("7");
				printf("4");
				printf("\n");
			}
			else if(a==c)
			{
				printf("7");b--;
				for(i=1;i<=c;i++)
				printf("47");
				for(i=1;i<=b-c;i++)
				printf("7");
				printf("\n");
			}
			
		}
		if(c>d)
		{
			for(i=1;i<=a-c;i++)printf("4");
			for(i=1;i<=c;i++)printf("47");
			for(i=1;i<=b-c;i++)printf("7");
			printf("\n");
		}
	}
}
