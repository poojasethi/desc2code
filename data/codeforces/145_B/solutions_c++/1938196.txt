#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<cmath>

using namespace std;

int main()
{
    int m,n,c,d,i,f=0,k;
    scanf("%d%d%d%d",&m,&n,&c,&d);
    if (c>d+1 || d>c+1) {printf("-1\n");return 0;}
    vector<int> s;
    for (i=1; i<=c; i++)
     {
         s.push_back(4); m--;
         s.push_back(7); n--;
     }
    if (d==c) {if (m>0) {s.push_back(4); m--;} else {f=1; n--;}}
    if (d>c)  {s.push_back(4); m--;f=1; n--;}
    if (m<0 || n<0) {printf("-1\n"); return 0;}
    for (k=s.size()-1; k>=0; k--) if (s[k]==7) break;
    if (f)
    {
        printf("7");
        if (k<0) while (n>0) {printf("7"); n--;}
    }
    for (int it=0; it<s.size(); it++)
    {
        printf("%d",s[it]);
        if (s[it]==4)
         while (m>0) {printf("4"); m--;}
        if (it==k)
         while (n>0) {printf("7"); n--;}
    }
    printf("\n");
}
