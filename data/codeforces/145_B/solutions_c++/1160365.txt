#include <cstdio>

int a, b, c, d, f, s, n;

int main(){
    scanf("%d%d%d%d", &a, &b, &c, &d);
    if(c == d){
    	if(a > c) {f = c + 1; s = c; n = 4;}
    	else {f = c; s = c + 1; n = 7;}
    }
    else if(c > d) {f = s = c; n = 4;}
    else {f = s = d; n = 7;}
    if(c-d>1 || d-c>1 || f>a || s>b)
    	printf("-1");
    else while(f || s)
    	if(n == 4){
    		while(a > f) {printf("4"); a--;}
    		a = --f;
    		printf("4");
    		n = 7;
    	}
    	else{
    		b--;
    		s--;
    		printf("7");
    		if(!s) while(b--) printf("7");
    		n = 4;
    	}
    puts("");
    return 0;
}
