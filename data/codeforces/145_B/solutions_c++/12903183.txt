#include<bits/stdc++.h>
using namespace std;

int main()
{
    int a1,a2,a3,a4,p=0;
    while(cin>>a1>>a2>>a3>>a4)
    {
        a4-= a3-1;
        a1-= a3;
        a2-= a3;

        if(a4==2){
            a1--,a2--;
            p=3;
        }
        if(a4==1)
        {
            if(a1){
                a1--;
                p=1;
            }
            else{
                a2--;
                p=2;
            }
        }
        if(a1<0||a2<0||a4>2||a4<0)
        {
            puts("-1");
        }
        else{
            if(p&2)cout<<"7";
            for(int i=0;i<a1;i++) cout<<"4";
            for(int i=0;i<a3;i++) cout<<"47";
            for(int i=0;i<a2;i++) cout<<"7";
            if(p&1) cout<<"4";
            cout<<endl;
        }

    }
}
