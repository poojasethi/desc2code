#!/usr/bin/env python

if __name__ == "__main__":
    a1, a2, a3, a4 = map(int, raw_input().strip().split())
    m = max(a3, a4)
    a1, a2 = a1 - m, a2 - m
    if a3 == a4 and a4 != 0:
        if a1 > 0 and a2 >= 0:
            print "4" * a1 + "7" + "47" * (a3 - 1) + "7" * a2 + "4"
        elif a1 >= 0 and a2 > 0:
            print "74" * a3 + "4" * a1 + "7" * a2
        else:
         print -1
    elif abs(a3 - a4) == 1:
        if a3 > a4 and a1 >= 0 and a2 >= 0:
            print "4" * a1 + "47" * a3 + "7" * a2 # ok
        elif a3 < a4 and a1 >= 0 and a2 >= 0:
            if a4 > 1:
                print "74" + "4" * a1 + "74" * (a4 - 2) + "7" * (a2 + 1) + "4" # ok
            else:
                print "7" * a2 + "74" * a4 + "4" * a1 # ok
        else:
            print -1
    else:
        print -1
