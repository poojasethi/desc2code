a, b, c, d = map(int, raw_input().split())
if abs(c - d) > 1 or min(a, b) < max(c, d) or a == b == c == d :
	z = '-1'
elif c == d:
	if a > c:
		z = '4' * (a - c - 1) + '47' * c + '7' * (b - c) + '4'
	else:
		z = '74' * d + '7' * (b - d)
elif c < d: # c = d - 1
	z = '7' + '4' * (a - d + 1) + '74' * (d - 2) + '7' * (b - d + 1) + '4'
else: # d = c - 1
	z = '4' * (a - d) + '74' * d + '7' * (b - d)
print z
