import sys

inp = sys.stdin
out = sys.stdout

(a, b, c, d) = map(int, inp.readline().split())

# a:4, b:7, c:47, d:74

if abs(c-d) > 1 or max(c, d) > min(a, b) or (a == b and b == c and c == d):
	s = -1
else:
	if c == d:
		s = '74'*c
		if a > c:
			s = '4'*(a-c) + s[:-1] + '7'*(b-c) + '4'
		else:
			s += '7'*(b-c)
	elif c < d:
		s = '7' + '4'*(a-c) + '74'*(c-1) + '7'*(b-c) + '4'
	else:
		s = '4'*(a-d) + '74'*d + '7'*(b-d)

print s
