a, b, c, d = map(int, raw_input().split())

if abs(c - d) > 1 or min(a, b) < max(c, d) or a == b == c == d:
    print -1
    exit()

if c == d:
    if a == c:
        ans = '74' * c + '7' * (b - c)
    else:
        ans = '4' * (a - c - 1) + '47' * c + '7' * (b - c) + '4'
elif c > d:
    ans = '4' * (a - c) + '47' * c + '7' * (b - c)
else:
    ans = '7' + '4' * (a - c) + '74' * (c-1) + '7' * (b - c) + '4'

print ans
