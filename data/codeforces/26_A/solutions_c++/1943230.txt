#include <cstdio>
#include <vector>
using namespace std;

int main () {
	vector<int> a(4000,0);
	int n,num=0;
	scanf("%d",&n);
	for (int i=2;i<=n;i++) {
		for (int j=2;j<i;j++)
			if (a[j]==0 && i%j==0) a[i]++;
		if (a[i]==2) num++;
	}
	printf("%d",num);
}
