#include <stdlib.h>
#include <stdio.h>

int main()
{

  int ans=0, j, i, n;
  int m[3010]={0};
  scanf("%d",&n); 
   
  for(i=2; i<=n; i++)
  {
    if(m[i]!=0)
     continue;
    for(j=i; j<=n; j+=i)
     m[j]++;
  }

  for(i=6; i<=n; i++)
   if(m[i]==2)
    ans++;
	
  printf("%d\n",ans);
  return 0;
  
}
