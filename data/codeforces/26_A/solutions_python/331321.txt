n = int(raw_input())
import math
def is_prime(x):
    for i in xrange(2,int(math.sqrt(x)+1)):
        if (x % i == 0): return False
    return True

lst = []
for i in xrange(2,n+1):
    if (is_prime(i)): lst.append(i)
#print lst
ans = 0
for i in xrange(1,n+1):
    c = 0
    for j in lst:
        if (i%j == 0): c+=1
    if c == 2: ans+=1

print ans
