n = input()
prime = [2]
isprime = [False] * (n + 1)
ans = 0
for i in range(3, n+1):
    flag = True
    factors = set()
    for p in prime:
        if i % p == 0:
            flag = False
            while (i % p == 0):
                factors.add(p)
                i /= p
    if flag:
        prime.append(i)
        isprime[i] = True
    elif len(factors) == 2:
        ans += 1
print ans
