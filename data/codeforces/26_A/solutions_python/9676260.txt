a=[0]*3001
b=[]
for i in xrange(2,3001):
    a[i]=1
for i in xrange(2,3001):
    j=i
    while i*j<3001:
        a[i*j]=0
        j+=1
for i in xrange(2,3001):
    if a[i]==1:
        b.append(i)
ctr=0
res=0
n=input()
for i in xrange(1,n+1):
    for j in xrange(0,len(b)):
        if i%b[j]==0:
            ctr+=1
    if ctr==2:
        res+=1
    ctr=0
print res
