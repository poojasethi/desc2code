from itertools import *
from collections import defaultdict

def isprime(n):
    for x in range(2, int(n**0.5)+1):
        if n % x == 0:
            return False
    return True

a = input()
b = [0]*3001
for i in xrange(2, a+1):
    for j in xrange(2, i):
        if i % j == 0:
            if isprime(j):
                b[i] += 1
                
print len([ i for i in b if i == 2 ])
