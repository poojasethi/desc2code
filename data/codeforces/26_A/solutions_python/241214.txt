from math import *

def almostEasy (x):
    r = int (floor (sqrt (x)))
    count = int ()
    for divider in range (2, r + 1):
        if x % divider == 0:
            count += 1
        while x % divider == 0:
            x //= divider
    return count + (x != 1)
    
ans = int ()
for i in range (1, input () + 1):
    if almostEasy (i) == 2:
        ans += 1
print ans
