def get_prime_factors(number):
   
    if number == 1:
        return []

    for i in xrange(2, number):
        
        rd, qt = divmod(number, i)
        if not qt:
            return [i] + get_prime_factors(rd)

    return [number]

n = int(raw_input())
if n < 6:
    print 0
else:
    ans = 0
    for i in range(6, n+1):
        if len(set(get_prime_factors(i))) == 2:
            ans += 1
    print ans
