def f(a):
    for i in range(2,a):
        if a%i==0:
            return False
    return True
a=input()
t=0
for i in range(6,a+1):
    if f(i):
        continue
    c=0
    for j in range(2,i):
        if i%j==0 and f(j):
            c+=1
            if c>2:
                break
    if c==2 :
        t+=1
print t
