from sys import stdin
from math import sqrt, floor
  
def ok(n):
  d = 2
  s = set()
  while d*d <= n:
    if n % d == 0:
      s.add(d)
      n /= d
    else:
      d += 1
  
  if n > 1:
    s.add(n)
  return (len(s) == 2)
  
rv = 0
for n in range(1, int(stdin.readline())+1):
  if ok(n):
    rv += 1

print rv
