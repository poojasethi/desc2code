from math import sqrt
n = input()

p = range(2,n)
for i in xrange(2,int(sqrt(n))+1):
	p = filter(lambda x: x == i or x % i, p)

def isp2(k):
	j = x = 0
	while k>1 and j<len(p):
		if k%p[j]==0 :
			x += 1
			if x > 2 : return 0
			while k>1 and k%p[j]==0 : k/=p[j]
		j+=1
		
	return 1 if x==2 else 0		

print sum(map(isp2, range(6,n+1)))
