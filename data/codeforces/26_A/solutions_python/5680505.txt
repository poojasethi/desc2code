import time
import sys, io
import re, math
start = time.clock()
n=int(raw_input())
prime=[]
ans=0
for i in range(2,n+1):
    if all(i%p!=0 for p in prime):
        prime+=[i]
for j in range(2,n+1):
    chk=0
    for k in range(2,j):
        if j%k==0 and (k in prime):
            chk+=1
    if chk==2:
        ans+=1
print ans
