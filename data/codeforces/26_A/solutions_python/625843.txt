def pp(x, p):
    col = 0
    for i in range(2, x / 2 + 1):
        if(x % i == 0) and (i in p): col += 1
    return (col == 2)

n = input()
a = [1 for i in range(n+1)]
a[0] = 0; a[1] = 0;
p = []
for i in range(2, n + 1):
    if(a[i]):
        p.append(i)
        j = i
        while(j + i <= n):
            j += i
            a[j] = 0
col = 0
for i in range(1, n + 1):
    if(not a[i]) and pp(i, p): col += 1
print(col)
