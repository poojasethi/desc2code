n = input()
i,aP = 0,0
def isPrime(n):
    if n == 1: return False
    for x in xrange(2,int(n**0.5)+1):
        if n % x == 0:
            return False
    return True
while i < n+1:
    N,c,div = i,2,[]
    while N > 1:
        if N % c == 0:
            div.append(c)
            N /= c
        else:
            c += 1
    i += 1
    aP += (len(set(div)) == 2)
print aP
