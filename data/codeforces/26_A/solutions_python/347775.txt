n = int(raw_input().strip())
ans = 0
for i in range (1,n+1):
    ntmp = i
    j = 2
    cnt = 0
    while j*j <= ntmp:
        if ntmp%j == 0:
            while ntmp%j == 0:
                ntmp = ntmp/j
            cnt = cnt + 1
        j = j + 1
    if ntmp != 1:
        cnt = cnt + 1
    if cnt == 2:
        ans = ans + 1
print ans
