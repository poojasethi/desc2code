n = int(raw_input())

p = []
for i in range(2, n + 1):
    for j in range(2, i):
        if i % j == 0:
            break
    else:
        p += [i]

ap = 0
for i in range(2, n + 1):
    t = 0
    for j in p:
        if i % j == 0:
            while i % j == 0:
                i /= j
            t += 1
    if t == 2:
        ap += 1

print ap
