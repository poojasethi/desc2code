#!/usr/bin/env python

n = int(raw_input())

def chk(n):
    i = 2
    j = 0
    k = 1
    while n > 1:
        u = 0
        while (n > 1) and (n % i == 0):
            n /= i
            u = 1
        if i == 3:
            k = 2
        i += k
        if u > 0:
            j += 1
    return j
        
s = 0
for i in xrange(6, n + 1):
    if chk(i) == 2:
        s += 1

print s
