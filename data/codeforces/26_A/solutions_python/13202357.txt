def is_almost(n):
    s = set()
    while n > 1:
        for i in range(2, n+1):
            if n % i == 0:
                s.add(i)
                n /= i
                break
    if len(s) == 2:
        return True
    return False

n = int(raw_input())
c = 0
for i in range(1, n+1):
    c += is_almost(i)
print c
