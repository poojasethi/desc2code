def isAlmostPrime(x):
	cnt = 0
	i = 2
	while x > 1:
		if 0 == x % i:
			cnt += 1
			if cnt > 2: return False
			while 0 == x % i: x /= i
		i += 1
	return 2 == cnt

def main():
	n = input()
	ans = 0
	for i in xrange(6, n + 1):
		if isAlmostPrime(i): ans += 1
	print ans

if __name__ == "__main__": main()
