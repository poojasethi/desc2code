p=[]
b=[True]*3000
for i in range(2,3000):
    if not b[i]:
        continue
    p.append(i)
    j=2*i
    while j<3000:
        b[j]=False
        j+=i
n=input()
result=0
for i in range(4,n+1):
    counter=0
    t=i
    for j in p:
        if j>t:
            break
        if t%j>0:
            continue
        counter+=1
        while t%j>1:
            t//=j
    if counter==2:
        result+=1
print(result)
