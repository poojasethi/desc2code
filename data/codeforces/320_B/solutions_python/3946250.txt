import sys

n = int(sys.stdin.readline())

intervals = []

def is_edge(a, b):
    return (a[0] < b[1] and a[0] > b[0]) or (a[1] < b[1] and a[1] > b[0])

def is_path(intervals, intervals_covered, a, b):
    if is_edge(a, b):
        return True
    else:
        for i in xrange(len(intervals)):
            if is_edge(a, intervals[i]):
                if i not in intervals_covered:
                    intervals_covered[i] = True
                    if is_path(intervals, intervals_covered, intervals[i], b):
                        return True
        return False

for i in xrange(n):
    t, x, y = map(int, sys.stdin.readline().split(" "))
    if t == 1:
        intervals.append([x, y])
    else:
        a = intervals[x - 1]
        b = intervals[y - 1]

        intervals_covered = {}
        intervals_covered[x - 1] = True

        if is_path(intervals, intervals_covered, a, b):
            print "YES"

            print "NO"
