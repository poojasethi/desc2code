    if p[x] != -1: return p[x]
    p[x] = 0
    for i in range(len(s)):
        if s[i][0]<s[x][0]<s[i][1] or s[i][0]<s[x][1]<s[i][1]:
            p[x] = p[x] or d(i)
    return p[x]
