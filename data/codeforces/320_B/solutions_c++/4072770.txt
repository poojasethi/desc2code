#include <stdio.h>

int main(){
	int xx[105],yy[105],s,go[105];
	int line[105],l,r;
	int n,c,x,y;
	s=0;
	scanf("%d",&n);
	while(n--){
		scanf("%d%d%d",&c,&x,&y);
		if(c==1){
			s++;
			xx[s]=x;yy[s]=y;
		}
		else {
			for(int i=1;i<=s;i++)go[i]=0;
			line[1]=x;
			l=1;r=1;
			go[x]=1;
			
			while(l<=r){
				int now=line[l];
				for(int i=1;i<=s;i++){
					if(go[i]==1)continue;
					
					if((xx[i]<xx[now]&&xx[now]<yy[i])||(xx[i]<yy[now]&&yy[now]<yy[i]))
					{
						go[i]=1;
						r++;
						line[r]=i;
					}
				}
				
				l++;
			}
			if(go[y]==1)printf("YES\n");
			else printf("NO\n");
		}
	}
	
	return 0;
}
