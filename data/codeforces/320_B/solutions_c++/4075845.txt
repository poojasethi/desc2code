#include<stdio.h>
#include<string.h>
int mark[105][2],map[105][105];
int main()
{
	int n,i,j,k,t,a,x,y,s;
	scanf("%d",&n);
	memset(map,0,sizeof(map));
	t=0;
	while(n--)
	{
		scanf("%d%d%d",&a,&x,&y);
		if(a==1)
		{
			mark[++t][0]=x;
			mark[t][1]=y;
			map[t][t]=1;
			for(s=1;s<=t;s++)
			{
				if(mark[s][0]>x&&mark[s][0]<y||mark[s][1]>x&&mark[s][1]<y)map[s][t]=1;
				if(mark[s][0]<x&&mark[s][1]>x||mark[s][0]<y&&mark[s][1]>y)map[t][s]=1;
			}
			for(i=1;i<=t;i++)
			for(j=1;j<=t;j++)
			for(k=1;k<=t;k++)
			if(map[k][i]&&map[i][j])map[k][j]=1;
		}
		if(a==2)
		{
			if(map[x][y])printf("YES\n");
			else printf("NO\n");
		}
	}
}
