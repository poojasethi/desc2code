#include <stdio.h>
#include <string.h>
#include <iostream>
using namespace std;
int l[200],r[200];
bool vis[200];
int n=0;
bool dfs(int x,int y){
	int i;
	vis[x]=1;
	if(x==y)return 1;
	for(i=1;i<=n;i++)
		if(!vis[i]&& (l[i] < l[x] && l[x] < r[i] || l[i] < r[x] && r[x] < r[i]))
			if(dfs(i,y)) return 1;
	return 0;
}
int main(){
	int t,a,b,k,i;
	scanf("%d",&t);
	for(i=1;i<=t;i++){
		scanf("%d%d%d",&k,&a,&b);
		if(k==1){
			l[++n]=a;
			r[n]=b;
		}
		else{
			memset(vis,0,sizeof vis);
			if(dfs(a,b))printf("YES\n");
			else printf("NO\n");
		}
	}
	return 0;
}
