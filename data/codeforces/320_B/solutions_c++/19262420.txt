#include<iostream>
#include<string.h>
using namespace std;
long long int a[108],b[108],foo[108],n=0;
void fun(long long int j)
{foo[j]=1;

	for(long long int g=1;g<=n;g++)
	{
		if(foo[g])continue;
		if((a[j]>a[g] && a[j]<b[g])||(b[j]>a[g] && b[j]<b[g]))
		{
			fun(g);
		}
	}
}
int main()
{
    long long int x,y,m,i,k;
	cin>>k;
	memset(a,0,sizeof(a));memset(b,0,sizeof(b)),memset(foo,0,sizeof(foo));
	for(i=1;i<=k;i++)
	{
		cin>>m;
		if(m==1)
		{   n++;
			cin>>a[n]>>b[n];
		}
		if(m==2)
		{cin>>x>>y;
		memset(foo,0,sizeof(foo));
		fun(x);	
		if(foo[y]==1)cout<<"YES\n";
		else cout<<"NO\n";	
		}
	}
	return 0;
}
