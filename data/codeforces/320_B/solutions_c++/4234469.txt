#include<cstdio>
#include<iostream>
#include <string.h>
#include<algorithm>
using namespace std;

int n, a[105], b[105], m, x, y, r;

bool visited[105];

void dfs(int p)
{
	visited[p] = true;
	for(int i = 1; i <= n; i++)
	{
		if(visited[i])
			continue;
		if(a[p] > a[i] && a[p] < b[i])
			dfs(i);
		else if(b[p] > a[i] && b[p] < b[i])
			dfs(i);
	}
}

int main()
{
	scanf("%d", &n);
	for(int j = 1; j <= n; j++)
	{
		scanf("%d", &m);
		if(m == 1)
			r++, scanf("%d %d", &a[r], &b[r]);
		else
		{
			scanf("%d %d", &x, &y);
			
			memset(visited, 0, 104);
			dfs(x);
			visited[y] ? printf("YES\n") : printf("NO\n");
		}
		
	}
		

	return 0;
}
