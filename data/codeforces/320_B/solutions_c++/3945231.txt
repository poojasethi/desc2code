#include <cstdio>
#include <cstring>
int n,o[101];
bool u[101];
struct Interval {int x,y;} a[101];

bool bfs(int s,int t)
{
	int q1=0, q2=0, p,i;
	memset(u,0,sizeof u);
	o[0]=s; u[s]=true;
	while (q1<=q2)
	{
		p=o[q1++];
		for (i=1; i<=n; ++i)
			if (!u[i] && (a[i].x<a[p].x && a[p].x<a[i].y || a[i].x<a[p].y && a[p].y<a[i].y))
			{
				if (i==t) return true;
				u[i]=true;
				o[++q2]=i;
			}
	}
	return false;
}

int main()
{
	//freopen("b.in","r",stdin);
	int m,flag,x,y;
	scanf("%d",&m);
	while (m--)
	{
		scanf("%d%d%d",&flag,&x,&y);
		if (flag==1) ++n, a[n].x=x, a[n].y=y;
		else if (bfs(x,y)) puts("YES"); else puts("NO");
	}
	return 0;
}
