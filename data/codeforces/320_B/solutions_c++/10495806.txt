#include <cstdio>
#include <cstdlib>

int l[111111], r[111111];
int vis[111111];
int n;

void dfs(int x){
  if (vis[x]) return;
  vis[x] = 1;
  for (int i = 1; i <= n; i++){
    if (l[i] < l[x] && l[x] < r[i]) dfs(i);
    if (l[i] < r[x] && r[x] < r[i]) dfs(i);
  }
}

void ask(int x, int y){
  for (int i = 1; i <= n; i++) vis[i] = 0;
  dfs(x); if (vis[y]) printf("YES\n"); else printf("NO\n");
}

int main(){
  int t, x, y, N;
  scanf("%d", &N);
  while (N--){
    scanf("%d %d %d", &t, &x, &y);
    if (t == 1){n++; l[n] = x; r[n] = y;}
    else ask(x, y);
  }
  return 0;
}
