#include<bits/stdc++.h>
using namespace std;
vector< pair<int ,int > > gr[103];
int x,a,b,n,t,s,d,m,h[103];
void f(int a)
{
	pair < int ,int > k;
	int i;
	k=gr[a][0];
	if(h[a] or d)return;
	if(a==b){d=1;cout<<"YES"<<endl;return;}
	h[a]=1;
	for(i=1;i<=s;i++)
	{
		pair<int ,int>p;
		p=gr[i][0];
		if((p.first<k.first and k.first<p.second) or (p.first<k.second and k.second<p.second))f(i);
	}
	//h[a]=0;
}
int main()
{
	cin>>n;
	while(n--)
	{
		int i;
		cin>>x>>a>>b;
		if(x==1){gr[++s].push_back(make_pair(a,b));}
		else
		{
			d=0;
			f(a);
			if(!d)cout<<"NO"<<endl;
		}
		for(i=1;i<=s;i++)h[i]=0;
	}
	return 0;
}
