#include<cstdio>
#include<cstring>

const int MAXN=100+9;

bool h[MAXN];
int x[MAXN],y[MAXN],q[MAXN];
int main()
{
#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
#endif
		int n,m=0,i,j,t,a,b;
		scanf("%d",&n);
		for(i=0;i<n;++i)
		{
			scanf("%d%d%d",&t,&a,&b);
			if(t==1)
			{
				x[++m]=a, y[m]=b;
			}
			else
			{
				int head=0, tail=1;
				memset(h,0,sizeof h);
				for(h[q[1]=a]=1;head<tail;)
				{
					int u=q[++head];
					for(j=1;j<=m;++j)
						if(((x[j]<x[u] && x[u]<y[j]) || (x[j]<y[u] && y[u]<y[j])) && !h[j])
							h[q[++tail]=j]=1;
				}
				puts(h[b]?"YES":"NO");
			}
		}
		return 0;
}
