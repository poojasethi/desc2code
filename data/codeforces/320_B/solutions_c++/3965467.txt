#include <iostream>
using namespace std;
int X[100],Y[100];
int visit[100];
int total = 0;

bool pquery(int st, int end) {
	if (visit[st] == 1) return false;
	if (st == end) return true;
	visit[st] = 1;
	bool ret =false;
	for (int i=0; i<total; i++) {
		if (i!=st) {
			if ((X[st]>X[i]&&X[st]<Y[i]) || (Y[st]>X[i]&&Y[st]<Y[i])) {
				ret = ret | pquery(i,end);
			}
		}
	}
	return ret;
}

int main () {
int n;
cin >> n;
for (int i=0; i<n; i++) {
	int qtyp; cin >> qtyp;
	for (int j=0; j<100; j++) visit[j] = 0;
	if (qtyp == 2) { int st,end; cin >> st >> end; if (pquery(st-1,end-1)) cout << "YES" << endl; else cout << "NO" << endl; }
	else {
		cin >> X[total] >> Y[total]; total++;
	}
}
return 0;
}
