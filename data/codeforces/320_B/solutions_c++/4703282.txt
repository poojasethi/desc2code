#include <cstring>
#include <cstdio>
const int N=200;
int n,m,k,x,y,i,j,l[N],r[N];
bool v[N];
int check(int i,int j) {
	if ((l[j]<l[i])&&(l[i]<r[j])) return true;
	if ((l[j]<r[i])&&(r[i]<r[j])) return true;
	return false;
}
bool dfs(int x,int y) {
	if (v[x]) return false;
	v[x]=true;
	for (int i=1;i<=n;i++)
		if ((i!=x)&&(check(x,i))) {
			if (i==y) return true;
			if (dfs(i,y)) return true;
		}
	return false;
}
int main() {
	scanf("%d",&m);
	for (i=1;i<=m;i++) {
		scanf("%d%d%d",&k,&x,&y);
		if (k==1) {
			l[++n]=x;r[n]=y;
		} else {
			memset(v,false,sizeof(v));
			if (dfs(x,y)) printf("YES\n");
			else printf("NO\n");
		}
	}
	return 0;
}
