#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int n , vis[101] , u=0;

int A[101] , B[101];
void dfs(int a, int tar){

	vis[a] = 1;
	for(int i=0;i<u;i++){
		if(((A[i] < A[a] && B[i] > A[a]) || (A[i] < B[a] && B[i] > B[a])) && !vis[i])
			 dfs(i , tar);
	}

}


int main(){
	scanf("%d" , &n);
	int t , l , r;
	for(int i=0;i<n;i++){

		scanf("%d%d%d" , &t , &l , &r );
		if(t == 1){
			  A[u] = l , B[u] = r, u++;
		}
		else{
			memset(vis , 0 , sizeof(vis));
			dfs(l-1, r-1);
			vis[r-1]? puts("YES") : puts("NO");
		}
	}


	return 0;
}
