#include<stdio.h>
#include<string.h>
struct num
{
	int a,b;
}n[105];
int j,flag[105];
int judge(int x,int y)
{
	int i;
	flag[x]=1;
	if(x==y)
		return 1;
	for(i=1;i<j;i++)
		if(!flag[i]&&((n[i].a<n[x].a&&n[x].a<n[i].b)||(n[i].a<n[x].b&&n[x].b<n[i].b)))
			if(judge(i,y))
				return 1;
	return 0;
}
int main()
{
	int N,m,x,y,i;
	while(scanf("%d",&N) != EOF)
	{
		for(i=0,j=1;i<N;i++)
		{
			scanf("%d%d%d",&m,&x,&y);
			if(m==1)
			{
				n[j].a=x;
				n[j++].b=y;
			}
			else
			{
				memset(flag,0,sizeof(flag));
				if(judge(x,y))
					printf("YES\n");
				else
					printf("NO\n");
			}
		}
	}
	return 0;
}
