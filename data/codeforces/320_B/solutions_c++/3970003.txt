#include<iostream>
#include<vector>
#include<cstring>
using namespace std;

struct pp{
	int s,t;
};
vector<pp> g;
bool vis[101];
void dfs(int x)
{
	vis[x]=true;
	for(int i=0;i<g.size();i++)
	   if(!vis[i] && ((g[i].s<g[x].s && g[x].s<g[i].t) || (g[i].s<g[x].t && g[x].t<g[i].t))) dfs(i);
}

int main()
{
	int n;
	cin>>n;g.clear();
	for(int i=1;i<=n;i++)
	{
		int a,b,c;
		cin>>a>>b>>c;
		if(a==1) 
		{
			pp ans;
			ans.s=b;ans.t=c;
			g.push_back(ans);
		}
		else{
			memset(vis,0,sizeof(vis));
			dfs(--b);
			if(vis[--c]) cout<<"YES"<<endl; else cout<<"NO"<<endl;
		}
	}
	return 0;
}
