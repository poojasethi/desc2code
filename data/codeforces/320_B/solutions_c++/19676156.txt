#include <bits/stdc++.h>

using namespace std;
int a[110],b[110],j=0;
bool v[110];

void dfs(int i){
        v[i]=true;
        for(int k=1;k<=j;k++){
            if(v[k])continue;
            if((a[i]>a[k]&&a[i]<b[k])||(b[i]>a[k]&&b[i]<b[k]))dfs(k);
        }

}

int main()
{
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    int n , i  ,x,y,k;
    cin >> n;
    while(n--){
        memset(v,false,sizeof(v));
        cin >> k>>x>>y;
        if(k==1){j++; a[j]=x; b[j]=y; continue;}
        else{

            dfs(x);
            if(v[y])cout<<"YES"<<endl;
            else cout<<"NO"<<endl;
        }


    }





    return 0;
}
