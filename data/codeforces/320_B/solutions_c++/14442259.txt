#include<iostream>
#include<vector>
using namespace std;
vector <pair<int,int> > a;
bool mark[110][110];
int n,x,c,d,t;
string dfs(int v)
{
    if(v==d)
        return "YES";
    mark[t][v]=true;
    for(int i=0; i<a.size(); i++)
        if(mark[t][i]==false && ((a[i].first < a[v].first && a[v].first < a[i].second) || (a[i].first < a[v].second && a[v].second < a[i].second)) && dfs(i)=="YES")
            return "YES";
    return "NO";
}
int main()
{
    cin>>n;
    for(t=0; t<n; t++)
    {
        cin>>x>>c>>d;
        if(x==1)
            a.push_back(make_pair(c,d));
        else{
            d--;
            cout<<dfs(c-1)<<'\n';
        }
    }
    return 0;
}
