#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

const int N=121;

int a[N],b[N];
bool vis[N];

int main()
{
	ios::sync_with_stdio(false);
	int Q; cin>>Q;
	int n=0;
	while(Q--)
	{
		int type; cin>>type;
		if(type==1)
		{
			cin>>a[n];
			cin>>b[n];
			n++;
		}
		else
		{
			int x,y; cin>>x>>y;
			x--; y--;
			memset(vis,0,sizeof(vis));
			vis[x]=true;
			for(int i=0;i<n;i++)
				for(int j=0;j<n;j++)
					if(vis[j])
					{
						for(int k=0;k<n;k++)
							if((a[j]>a[k]&&a[j]<b[k])||(b[j]>a[k]&&b[j]<b[k]))
								vis[k]=true;
					}
			if(vis[y]) cout<<"YES\n";
			else cout<<"NO\n";
		}
	}
	return 0;
}
