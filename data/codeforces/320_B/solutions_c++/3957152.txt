#include<iostream>
#include<cstring>
#include<string>
using namespace std;
int a[1002],b[1002];
int op,f,sec,count=1;
int vis[1002];
void dfs(int x)
{
	vis[x] = 1;
	for(int i = 1;i<count;i++)
	{
		if(!vis[i] && ((a[i] < a[x] && a[x]< b[i]) || (a[i] < b[x] && b[x] < b[i])))
		{
			dfs(i);
		}
	}
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		cin>>op>>f>>sec;
		if(op == 1)
		{
			a[count] = f;
			b[count] = sec;
			count ++;
		}
		else
		{
			memset(vis,0,sizeof(vis));
			dfs(f);
			if(vis[sec])
				cout<<"YES"<<endl;
			else
				cout<<"NO\n";
		}
	}
	return 0;
}
