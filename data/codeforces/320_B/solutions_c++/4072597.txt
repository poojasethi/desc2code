#include<stdio.h>
#include<string.h>
int x[101]={0};
int y[101]={0};
int visit[101]={0};
int q[101]={0};
int n=0;
int ans=0;
void solve(int s,int e){
ans=0;
int i,j,h=1,t=1;
memset(visit,0,sizeof(visit));
q[1]=s;
visit[s]=1;
while(h<=t){
i=q[h];
for(j=1;j<=n;j++)if(visit[j]==0){
if((x[j]<x[i]&&x[i]<y[j])||(x[j]<y[i]&&y[i]<y[j])){
visit[j]=1;
q[++t]=j;
}
}
h++;
}
if(visit[e]==1)printf("YES\n");
else printf("NO\n");
}
main(){
int m;
int b,c,d;
scanf("%d",&m);
while(m--){
scanf("%d%d%d",&b,&c,&d);
if(b==1){n++;x[n]=c;y[n]=d;}
else solve(c,d);
}
}
