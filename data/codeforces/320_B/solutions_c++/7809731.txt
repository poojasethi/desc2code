#include <iostream>
#include <cstring>
#include <string>
using namespace std;
int n,t;
int x[101],y[101];
int cc=0;
bool f;
bool m[101];
string s;
void dfs(int a,int b)
{
	m[a]=true;
	for (int i=1;i<=cc;i++)
	{
		if (!m[i]&&(x[i]<x[a]&&x[a]<y[i]||x[i]<y[a]&&y[a]<y[i]))
		{
			if (i==b)
			{
				f=true;
				return;
			}
			else
				dfs(i,b);
		}
	}
}
int main()
{
	int a,b;
	cin>>n;
	for (int i=1;i<=n;i++)
	{
		cin>>t;
		if (t==1)
		{
			cc++;
			cin>>x[cc]>>y[cc];
		}
		else
		{
			cin>>a>>b;
			f=false;
			memset(m,false,sizeof(m));
			dfs(a,b);
			s+=f?"YES\n":"NO\n";
		}
	}
	cout<<s;
	return 0;
}
