#include<iostream>
#include<cstring>

using namespace std;

int vis[1010],xi[1010],yi[1010];
int i;

void dfs(int x) {
    if(vis[x] == 1)
        return;
    vis[x] = 1;
    for(int j=1;j<i;j++) {
        if((xi[j] < xi[x] && xi[x] < yi[j]) || (xi[j] < yi[x] && yi[x] < yi[j]))
            dfs(j);
    }
    return;
}

int main()
{
    int n,t,a,b;
    cin >> n;
    i = 1;
    while(n--) {
        cin >> t;
        if(t == 1) {
            cin >> xi[i] >> yi[i];
            i++;
        } else {
            cin >> a >> b;
            memset(vis,0,sizeof(vis));
            dfs(a);
            cout << ((vis[b] == 1) ? "YES" : "NO") << endl;
        }
    }
}
