#include <cstdio>
#include <cstring>

bool visit[ 101 ];
int n;
int x[ 101 ], y[ 101 ];

void dfs( int p )
{
	int i;
	visit[ p ] = true;
	for ( i = 1; i <= n; i++ )
		if ( !visit[ i ] && ( x[ p ] > x[ i ] && x[ p ] < y[ i ] || y[ p ] > x[ i ] && y[ p ] < y[ i ] ) )
			dfs( i );
}

int main( )
{
	int m, cmd, a, b;
	scanf("%d", &m);
	n = 1;
	while ( m-- )
	{
		scanf("%d%d%d", &cmd, &a, &b);
		if ( cmd == 1 )
		{
			x[ n ] = a;
			y[ n++ ] = b;
		}
		else
		{
			memset( visit, 0, sizeof( visit ) );
			dfs( a );
			if ( visit[ b ] )
				puts("YES");
			else
				puts("NO");
		}
	}
	return 0;
}
