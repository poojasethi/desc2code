#include <stdio.h>
#include <string.h>
int a[101], b[101], v[101];
int dfs(int cnt, int cur, int dst) {
	int i;
	if (cur == dst) return 1;
	v[cur] = 1;
	for (i = 1; i < cnt; i++) {
		if (v[i]) continue;
		if (((a[cur] > a[i] && a[cur] < b[i]) || (b[cur] > a[i] && b[cur] < b[i])) && dfs(cnt, i, dst))
			return 1;
	}
	return 0;
}
int main() {
	int n, i, j, t, x, y;
	scanf("%d", &n);
	for (i = 0, j = 1; i < n; i++) {
		scanf("%d%d%d", &t, &x, &y);
		if (1 == t) a[j] = x, b[j] = y, j++;
		else {
			memset(v, 0, sizeof(v));
			printf("%s\n", dfs(j, x, y) ? "YES" : "NO");
		}
	}
	return 0;
}
