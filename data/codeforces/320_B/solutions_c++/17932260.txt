#include <bits/stdc++.h>

using namespace std;

const int MN = 1e3 + 2*85;

bool mk[MN];
vector<int>x , y;

void dfs(int a)
{
	mk[a] = true;
	for(int i=0;i<x.size();++i){
		if(!mk[i])
			if((x[i] < x[a] && x[a] < y[i]) || (x[i] < y[a] && y[a] < y[i]))
				dfs(i);
	}
}

void input()
{
	int q;
	cin >> q;
	int mod1, x1 , y1; 
	for(int i=0;i<q;++i){
		cin >> mod1 >> x1 >> y1;
		if(mod1 == 2){
			fill_n(mk , MN , false);
			dfs(x1-1);
			if(mk[y1-1])
				cout << "YES" << endl;	
			else
				cout << "NO" << endl;
		}else{
			x.push_back(x1);
			y.push_back(y1);
		}
	}
}

int main()
{
	input();
	return 0;
}
