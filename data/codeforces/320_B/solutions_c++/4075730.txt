#include <stdio.h>
#include <string.h>
int x,y,from[200],to[200],num=1,flag;
int vis[200];
void dfs(int cur)
{
	if(cur==y) 
	{
		flag=1;
		return;
	}
	for(int i=1;i<num;i++)
	{
		if(!vis[i]&&(from[i]<from[cur]&&from[cur]<to[i]||from[i]<to[cur]&&to[cur]<to[i]))
		{
			vis[i]=1;
			dfs(i);
		}
	}
}
int main()
{
	int n,opp;
	scanf("%d",&n);
	memset(vis,0,sizeof(vis));
	while(n--)
	{
		scanf("%d%d%d",&opp,&x,&y);
		if(opp==1)
		{
			from[num]=x;
			to[num]=y;
			num++;
		}
		else 
		{
			memset(vis,0,sizeof(vis));
			flag=0;
			if(x<num&&y<num)
			{
				vis[x]=1;
				dfs(x);
			}
			if(flag)  printf("YES\n");
			else printf("NO\n");
		}
	}
}
