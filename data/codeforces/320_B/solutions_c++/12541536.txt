#include <iostream>
using namespace std;
int q,N,v[100];
struct interval{
	int s,t;
}a[100]; 
void dfs(int n)
{
	int i;
	v[n]=1;
	for (i=0;i<N;i++){
		if (((a[i].s<a[n].s && a[n].s<a[i].t)||(a[i].s<a[n].t && a[n].t<a[i].t)) && !v[i])
	dfs(i);}
}
int main(){
	cin >>q;
	int i,j,aa,b,c;
	for (i=0;i<q;i++)
	{
		cin >>c>>aa>>b;
		if (c==1)
		{
			a[N].s=aa,a[N].t=b,N++;
		}
		else{
			--aa,--b;
			for (j=0;j<N;j++)v[j]=0;
			dfs(aa);
			if (v[b]){
				cout <<"YES"<<endl;
			}
			else
			{
				cout <<"NO"<<endl;
			}
		}
	}
	return 0;
}
