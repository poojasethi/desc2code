#include <iostream>
#include <cstdlib>
using namespace std;
int n,x,y,z,a[3][105],len;
bool vis[105];
void dfs (int s)
{
	 vis[s]=true;
	 for (int i=1;i<=len;i++)
		 if (!vis[i] && ((a[1][i]<a[1][s] && a[1][s]<a[2][i]) || (a[1][i]<a[2][s] && a[2][s]<a[2][i]))) dfs(i);
}
	 
int main ()
{
	cin>>n;
	for (int i=1;i<=n;i++)
	{
		cin>>x>>y>>z;
		if (x==1)
		{
		   len++;
		   a[1][len]=y;
		   a[2][len]=z;
        }
        else
        {
			//memset (vis,false,sizeof(vis));
			for (int j=1;j<=105;j++)
				vis[j]=false;
        	dfs(y);
        	if (vis[z]) cout<<"YES"; else cout<<"NO";
        	cout<<endl;
        }
    }
    return 0;
}
