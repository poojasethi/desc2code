#include<iostream>
#include<string>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
int m;
int a[105],b[105];
bool r[105];
int n;
int yy;
void dfs(int i)
{
	r[i]=true;
	for(int j=0;j<n;j++)
	{
		if(r[j]==false && ((a[j]<a[i] && a[i]<b[j])||(a[j]<b[i] && b[i]<b[j]))){dfs(j);}
	}
}
bool did(int x,int y)
{
	memset(r,false,sizeof(r));
	dfs(x-1);
	if(r[y-1])return true;
	return false;
}
int main()
{
	cin>>m;
	for(int i=0;i<m;i++)
	{
		int t,x,y;
		cin>>t>>x>>y;
		if(t==1)
		{
			a[n]=x;
			b[n++]=y;
		}
		else
		{
			if(did(x,y))cout<<"YES"<<endl;else cout<<"NO"<<endl;
		}
	}
	//system("pause");
	return 0;
}
