#include<iostream>
#include<memory>
using namespace std;
int l[200],r[200],n;
bool v[200];
bool dfs(int x,int y)
{
	v[x]=1;
	if(x==y)return 1;
	for(int i=1;i<=n;i++)
	{
		if(!v[i]&& (l[i]<l[x] && l[x]<r[i] || l[i]<r[x] && r[x]<r[i]))
		{
			if(dfs(i,y))return 1;
		}
	}
	return 0;
}
int main()
{
	int t,a,b,k;
	cin>>t;
	for(int i=1;i<=t;i++)
	{
		cin>>k>>a>>b;
		if(k==1)
		{
			l[++n]=a;
			r[n]=b;
		}
		else
		{
			for(int i=0;i<200;i++)v[i]=0;
			if(dfs(a,b))cout<<"YES"<<endl;else cout<<"NO"<<endl;
		}
	}
	return 0;
}
