#include<iostream>
#include<cstring>
using namespace std;
int t,n,pd,s1[1000],s2[1000],x,y,visit[1000];
void f(int a,int b)
{
if(a==x&&b==y)
{
pd=1;return;
}
if(pd==1)return;
for(int i=1;i<=n;i++)
{
if(pd==1)return;
if(visit[i]==0)
{
if(s1[i]<a&&s2[i]>a)
{
visit[i]=1;
f(s1[i],s2[i]);
}
if(s1[i]<b&&s2[i]>b)
{
visit[i]=1;
f(s1[i],s2[i]);
}
}
}
}
int main()
{
int T,a,b,c;
cin>>T;
memset(s1,0,sizeof(s1));
memset(s2,0,sizeof(s2));
while(T--)
{
cin>>c>>a>>b;
if(c==1)
{
s1[++n]=a;
s2[n]=b;
}
else
{
memset(visit,0,sizeof(visit));
pd=0;
x=s1[b];y=s2[b];
f(s1[a],s2[a]);
if(pd==1)cout<<"YES"<<endl;
else cout<<"NO"<<endl;
}
}
return 0;
}
