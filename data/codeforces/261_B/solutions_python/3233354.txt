from math import *

n = input()
a = map(int, raw_input().split())
p = input()
f = [factorial(i) for i in range(n + 1)]
if sum(a) <= p:
	print n
else:
	r = 0
	for i in range(n):
		d = [[0] * (p + 1) for _ in range(n + 1)]
		d[0][0] = 1.
		for j in range(n):
			if i != j:
				for c in range(j, -1, -1):
					for v in range(p - a[j] + 1):
						d[c + 1][v + a[j]] += d[c][v]
		r += sum(d[c][v] * f[c] * f[n - 1 - c] * c for c in range(n) for v in range(max(0, p - a[i] + 1), p + 1))
	print r / f[n]
