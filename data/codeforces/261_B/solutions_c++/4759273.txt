#include <cstdio>
double c[51][51];
int a[50];
long long d[51][51];
int main()
{
    int n,p,i,j,k;
    double ans=0;
    scanf("%d",&n);
    for(i=0;i<=n;i++)
    {
        c[i][0]=c[i][i]=1;
        for(j=1;j<i;j++)
            c[i][j]=c[i-1][j-1]+c[i-1][j];
    }
    for(i=0;i<n;i++)
        scanf("%d",a+i);
    scanf("%d",&p);
    d[0][0]=1;
    for(i=0;i<n;i++)
        for(j=i+1;j>0;j--)
            for(k=p;k>=a[i];k--)
                d[j][k]+=d[j-1][k-a[i]];
    for(i=1;i<=n;i++)
        for(j=1;j<=p;j++)
            ans+=d[i][j]/c[n][i];
    printf("%lf",ans);
}
