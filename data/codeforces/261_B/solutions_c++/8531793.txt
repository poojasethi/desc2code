#include <cstdio>
#include <algorithm>
using namespace std;
const int N=55;
int n,p,a[N];
long long f[N][N][N],C[N][N];
double ans;
int main(){
	scanf("%d",&n);
	for(int i=0;i<=n;i++)
		for(int j=0;j<=i;j++)
			C[i][j]=(j?C[i-1][j]+C[i-1][j-1]:1);
	for(int i=1;i<=n;i++) scanf("%d",a+i);
	scanf("%d",&p);
	f[0][0][0]=1;
	for(int i=1;i<=n;i++)
		for(int j=0;j<=i;j++)
			for(int k=0;k<=p;k++)
				f[i][j][k]=f[i-1][j][k]+(j&&k>=a[i]?f[i-1][j-1][k-a[i]]:0);
	for(int i=1;i<=n;i++){
		long long now=0;
		for(int k=0;k<=p;k++)
			now+=f[n][i][k];
		ans+=now/double(C[n][i]);
	}
	printf("%.12lf\n",ans);
	scanf("\n");
}
