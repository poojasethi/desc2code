#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int n,a[51],p,sum;
long double ans;
long double dp[53][53];
long double fact[53];
int main(){
	scanf("%d",&n);
	fact[0]=1;
	for(int i=1;i<=n;++i) fact[i]=fact[i-1]*i;
	for(int i=0;i<n;++i) scanf("%d",a+i),sum+=a[i];
	scanf("%d",&p);
	if(sum<=p){
		printf("%d\n",n);
		return 0;
	}
	for(int i=0;i<n;++i){
		memset(dp,0,sizeof(dp));
		dp[0][0]=1;
		for(int j=0;j<n;++j) if(j!=i){
			for(int k=n;k>0;--k) for(int l=p;l>=a[j];--l){
				dp[k][l]+=dp[k-1][l-a[j]];
			}
		}
		for(int j=0;j<n;++j) for(int k=max(0,p-a[i]+1);k<=p;++k){
			ans+=dp[j][k]*j*fact[j]*fact[n-j-1];
		}
	}
	printf("%.10f\n",(double)(ans/fact[n]));
	return 0;
}
