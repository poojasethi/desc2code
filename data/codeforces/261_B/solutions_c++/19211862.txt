#include<iostream>
#include<cstdio>
using namespace std;
typedef long double ll;
int n,a[51],p;
ll jc[51];
ll f[51][51][201];
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)cin>>a[i];
	cin>>p;
	jc[1]=1.0;jc[0]=1.0;
	for(int i=2;i<=n;i++)jc[i]=jc[i-1]*(ll)i;
	f[1][0][0]=1.0;f[1][1][a[1]]=1.0;
	for(int i=1;i<n;i++)
	    for(int j=0;j<=i;j++)
	        for(int k=0;k<=p;k++)
	            {
	            	f[i+1][j][k]+=f[i][j][k];
	            	f[i+1][j+1][k+a[i+1]]+=f[i][j][k];
				}
	ll ans=0;
	for(int j=1;j<=n;j++)
	    for(int k=0;k<=p;k++)
	        ans+=f[n][j][k]*jc[n-j]*jc[j];
	ans/=jc[n];
	cout<<ans;
	return 0;	        
}
