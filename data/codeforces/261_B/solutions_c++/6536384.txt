#include<stdio.h>
#include<algorithm>
using namespace std;
#include<string.h>
double loli[55];
double dp[55][55];
int main()
{
   int n,m,i,j,k,t,x[55];
   int p;
    loli[0]=1.0;
	for(i=1;i<=51;i++)loli[i]=i*loli[i-1];
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&x[i]);
	}
	dp[0][0]=1;
	scanf("%d",&p);
	for(i=1;i<=n;i++)
	{
		for(j=n;j>=1;j--)
		{
			for(t=p;t>=x[i];t--)
			  dp[j][t]=dp[j][t]+dp[j-1][t-x[i]];
		}
	}
	double sum=0;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=p;j++)
		{
			sum=sum+dp[i][j]*loli[i]*loli[n-i];
		}
	}
	//printf("%lf\n",sum);
	printf("%.10f\n",sum/loli[n]);
}
