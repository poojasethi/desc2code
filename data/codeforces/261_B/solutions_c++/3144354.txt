#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <set>
#include <vector>
#include <cmath>
#include <queue>
#include <iostream>
using namespace std;
typedef long long LL;
const int inf=1000000000;//1e9
double ans,dp[60][60];
int a[60];
int main(){
	int n,p;
	cin>>n;
	for (int i=0;i<n;i++) cin>>a[i];
	cin>>p;
	int s=0;
	for (int i=0;i<n;i++) s+=a[i];
	if (s<=p)
	{
		cout<<n;
		return 0;
	}
	dp[0][0]=1.0;
	for (int i=0;i<n;i++)
		for (int k=n-1;k>=0;k--)
			for (int j=0;j<p-a[i]+1;j++)
				dp[k+1][j+a[i]]+=dp[k][j]/(n-k)*(k+1);
	for (int i=1;i<=n;i++)
		for (int j=0;j<p+1;j++)
			ans+=dp[i][j];
	printf("%.9lf",ans);
}
