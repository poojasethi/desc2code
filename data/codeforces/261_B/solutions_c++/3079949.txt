#include<cstdio>
using namespace std;
#define FOR(i,a,b) for(int i=(int)a;i<=(int)b;++i)
#define FORD(i,a,b) for(int i=(int)a;i>=(int)b;--i)
#define REP(i,n) FOR(i,0,n-1)
#define REPT(i,n) FOR(i,1,n)
int n,a[60],p,total;
double res,dp[60][60];
main(){
	scanf("%d",&n);
	REP(i,n)
		scanf("%d",a+i),
		total+=a[i];
	scanf("%d",&p);
	if (total<=p) {
		printf("%d",n);
		return 0;
	}
	dp[0][0]=1.0;
	REP(i,n)FORD(k,n-1,0)REP(j,p-a[i]+1)
		dp[k+1][j+a[i]]+=dp[k][j]/(n-k)*(k+1);
	REPT(i,n)REP(j,p+1)
		res+=dp[i][j];
	printf("%.9lf",res);
}
