#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iomanip>
using namespace std;
__float128 p[51],ans;
__int64 f[51][51];
int a[51],n,m,i,j,k,l,sum; 
int main()
{
	cin>>n;
	for(i=1;i<=n;i++) scanf("%d",&a[i]),sum+=a[i];
	cin>>m;
	for(p[0]=i=1;i<=n;i++) p[i]=p[i-1]*i;
	if(sum<=m) {cout<<n<<".000000\n"; return 0;}
	for(i=1;i<=n;i++)
	{
		memset(f,0,sizeof(f));
		f[0][0]=1;
		for(j=1;j<=n;j++) if(i!=j)
			for(k=m-a[j];k>=0;k--)
				for(l=0;l<j;l++)
					f[k+a[j]][l+1]+=f[k][l];
		for(j=max(m-a[i]+1,0);j<=m;j++)
			for(k=0;k<n;k++)
				ans+=k*f[j][k]*p[k]/p[n]*p[n-1-k];
	}
	cout<<setprecision(10)<<fixed<<(long double)ans<<endl;
	return 0;
}
