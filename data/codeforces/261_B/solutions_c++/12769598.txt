#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <iostream>
using namespace std;

long double p[55],dp[55][55];
int a[55];
int main()
{
 //   freopen("aa.txt","r",stdin);
    int i,j,k,n,l;long double ans=0;
    cin>>n;
    for(i=1;i<=n;i++) cin>>a[i];cin>>l;
    for(i=0;i<=l;i++) dp[0][i]=1;
    for(i=1;i<=n;i++) for(j=n;j>=1;j--)
        for(k=l;k>=a[i];k--) dp[j][k]+=dp[j-1][k-a[i]];
    p[0]=1;
    for(i=1;i<=n;i++) p[i]=p[i-1]*i;
    for(i=1;i<=n;i++) ans+=dp[i][l]*p[i]*p[n-i];
/*  for(i=1;i<=n;i++) for(j=1;j<=l;j++) {
        ans+=dp[i][j]*p[i]*p[n-i];
    }*/
    cout<<ans/p[n]<<endl;
}
