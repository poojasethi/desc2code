#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
using namespace std;
int i,n,m,j,a[120],k;
double p[120],ans;
long long f[120][120];
int main(){
    scanf("%d",&n);
    for(i=1;i<=n;++i)scanf("%d",&a[i]);
    scanf("%d",&m);
    f[0][0]=1;
    for(i=1;i<=n;++i)
        for(j=n;j>=0;--j)
            for(k=m;k>=a[i];--k)
                f[j][k]+=f[j-1][k-a[i]];
    p[0]=1;
    for(i=1;i<=n;++i)p[i]=p[i-1]*i;
    for(i=0;i<=n;++i)
        for(j=1;j<=m;++j)ans+=f[i][j]*p[i]*p[n-i];
    printf("%.10lf\n",ans/p[n]);
}
