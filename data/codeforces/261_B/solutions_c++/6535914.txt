#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int a[100];
double f[100];
double dp[100][100];

int main()
{
	int n,p;
	int i,j,k;
	double ans=0.0;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	scanf("%d",&a[i]);
	f[0]=f[1]=1.0;
	for(i=2;i<=n;i++) f[i]=f[i-1]*i;
	scanf("%d",&p);
	dp[0][0]=1;
	for(i=1;i<=n;i++)
	{
		for(j=n;j>=1;j--)
		{
			for(k=p;k>=a[i];k--)
			    dp[j][k]+=dp[j-1][k-a[i]];
		}
	}
	
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=p;j++)
		{
			ans+=dp[i][j]*f[i]*f[n-i];
		}
	}
	ans=ans/f[n];
	printf("%.6f\n",ans);
	return 0;
}
