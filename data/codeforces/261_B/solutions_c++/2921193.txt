#include<cstdio>
#include<cstring>
int n,m,x; long long f[88][88],c[88];
double ans,p[88];
int main()
{
	scanf("%d",&n),f[0][0]=1;
	for (int i=1;i<=n;i++){
		scanf("%d",&x);
		for (int j=i;j;j--)
			for (int k=60;k>=x;k--)
				f[j][k]+=f[j-1][k-x];		
		}
	scanf("%d",&m);
	for (int i=1;i<=n;i++) for (int j=1;j<=m;j++) c[i+i>n?i:n-i]+=f[i][j];
	p[0]=1; for (int i=1;i<=n;i++) p[i]=p[i-1]*i;
	for (int i=1;i<=n;i++) ans/=i,ans+=c[i]*p[n-i];
	printf("%.9lf\n",ans); return 0;
}
