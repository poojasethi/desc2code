#include"stdio.h"
double jc[55],dp[55][55];
int a[55];
int main()
{
	int i,j,n,x,k;
	scanf("%d",&n);
	jc[0]=1;
	for(i=1;i<=n;i++)
	{
		jc[i]=jc[i-1]*i;
		scanf("%d",&a[i]);
	}
	scanf("%d",&x);
	dp[0][0]=1;
	for(i=1;i<=n;i++)
	{
		for(j=n;j>=1;j--)
		{
			for(k=x;k>=a[i];k--)
			{
				dp[j][k]+=dp[j-1][k-a[i]];
			}
		}
	}
	double ans=0;
	for(i=1;i<=n;i++)
	{
		for(k=1;k<=x;k++)
		{
			ans+=dp[i][k]*jc[i]*jc[n-i];
		}
	}
	printf("%.7lf\n",ans/jc[n]);
}
