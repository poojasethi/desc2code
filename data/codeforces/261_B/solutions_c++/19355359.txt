#include<cstdio>
#define N 51
int n,a[N],p;
double fac[N]= {1},f[N][N],tol,ans;
int main()
{
    scanf("%d",&n);
    for(int i=1; i<=n; i++)
    {
        scanf("%d",a+i);
        fac[i]=fac[i-1]*i;
        tol+=a[i];
    }
    scanf("%d",&p);
    if(tol<=p)
    {
        printf("%d",n);
        return 0;
    }
    f[0][0]=1;
    for(int i=1; i<=n; i++)
        for(int j=n-1; j>=0; j--)
            for(int k=0; k+a[i]<=p; k++)
                f[j+1][k+a[i]]+=f[j][k];
    for(int j=1; j<=n; j++)
        for(int k=0; k<=p; k++)
            ans+=f[j][k]*fac[j]*fac[n-j];
    ans/=fac[n];
    printf("%lf",ans);
}
