#include<stdio.h>
double g[100],f[100][100],ans;
int a[100],i,j,p,n,k;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	scanf("%d",&p);
	for(i=0;i<=n;i++)
		for(j=0;j<=p;j++)
			f[i][j]=0;
	ans=0;
	g[0]=1.0;
	for(i=1;i<=n;i++)g[i]=g[i-1]*i;
	f[0][0]=1.0;
	for(k=1;k<=n;k++)
		for(i=k-1;i>=0;i--)
			for(j=0;j<=p;j++){
				if(f[i][j]==0)continue;
				if(j+a[k]>p)continue;
				f[i+1][j+a[k]]+=f[i][j];
			}
	for(i=1;i<=n;i++)
		for(j=1;j<=p;j++)
			ans+=f[i][j]*g[i]*g[n-i];
	printf("%.10f\n",ans/g[n]);			
}
