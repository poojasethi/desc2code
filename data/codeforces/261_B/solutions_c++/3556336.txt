#include<stdio.h>
#include<algorithm>
#include<iostream>

using namespace std;

int i,j,k,n;
double ans[51][51],res,f[51],b[51],p;

main()
 {
  cin>>n;
  f[0]=1;
  for (i=1;i<=n;i++)
   f[i]=f[i-1]*i;
  for (i=1;i<=n;i++)
   cin>>b[i];
  cin>>p;
  ans[0][0]=1;
  for (i=1;i<=n;i++)
   for (j=n;j>=1;j--)
    for (k=(int)b[i];k<=p;k++)
     ans[j][k]+=ans[j-1][k-(int)b[i]];
  for (i=1;i<=n;i++)
   for (j=1;j<=p;j++)
    res+=ans[i][j]*f[i]*f[n-i];
  res/=f[n];
  cout<<res<<endl;
 }
