#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
#define N 55
double dp[55][55][55];
double f[N];
int n, a[N], p;
int main() {
	f[0] = 1; for(int i = 1; i < N; ++i) f[i] = f[i - 1] * i;
	scanf("%d", &n);
	for(int i = 1; i <= n; ++i) scanf("%d", &a[i]);
	scanf("%d", &p);
	dp[1][0][0] = 1;
	for(int i = 1; i <= n; ++i) {
		for(int j = 0; j < i; ++j) {
			for(int k = 0; k <= p; ++k) {
				if(k + a[i] <= p)
					dp[i+1][j+1][k+a[i]] += dp[i][j][k];
				dp[i+1][j][k] += dp[i][j][k];
			}
		}
	}
	double ans = 0;
	for(int i = 1; i <= n; ++i) {
		for(int j = 0; j <= p; ++j) {
			ans += dp[n+1][i][j] * f[n-i] * f[i];
		}
	}
	ans /= f[n];
	printf("%.12lf\n", ans);
	return 0;
}
