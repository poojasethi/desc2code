#include <iostream>
using namespace std;

int a[55];
long long s[55][55][55];
long double f[55];

int main() {
  f[0]=1;
  for (int ni=1;ni<=50;ni++)
    f[ni]=f[ni-1]*ni;
  int n,p;
  cin>>n;
  for (int ni=0;ni<n;ni++)
    cin>>a[ni];
  cin>>p;
  for (int ni=0;ni<=n;ni++)
    s[ni][0][0]=1;
  for (int ni=0;ni<n;ni++)
    for (int nj=1;nj<=ni+1;nj++)
      for (int nk=0;nk<=p;nk++) {
        s[ni+1][nj][nk] += s[ni][nj][nk];
        if (nk+a[ni]<=p)
          s[ni+1][nj][nk+a[ni]] += s[ni][nj-1][nk];
      }
  long double answer=0;
  for (int ni=1;ni<=n;ni++)
    for (int pi=1;pi<=p;pi++)
      answer+=s[n][ni][pi]*f[ni]*f[n-ni]/f[n];
  cout<<answer<<endl;
  return 0;
}
