#include<iostream>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<cstdio>
#include<iomanip>
using namespace std;
double jc[55];
double f[55][55][105]={0};
int n,p;
int a[55];
void solve()
{
	int i,j,k;
	jc[0]=jc[1]=1;
	for(i=2;i<=n;i++)
		jc[i]=jc[i-1]*i;
	f[1][0][0]=1;
	f[1][1][a[1]]=1;
	for(i=1;i<=n-1;i++)
		for(j=0;j<=i;j++)
			for(k=0;k<=p;k++)
			{
				f[i+1][j][k]+=f[i][j][k];
				f[i+1][j+1][k+a[i+1]]+=f[i][j][k];
			}
	double ans=0;
	for(i=1;i<=n;i++)
		for(j=0;j<=p;j++)
			ans+=f[n][i][j]*jc[i]*jc[n-i];
	ans/=jc[n];
	cout<<fixed<<setprecision(5)<<ans<<endl;
}
int main()
{
	int i;
	cin>>n;
	for(i=1;i<=n;i++)
		cin>>a[i];
	cin>>p;
	solve();
	return 0;
}
