#include<stdio.h>
double dp[60][60];
double fac[60];
int len,n;
int a[60];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    fac[0]=1.0;
    for(int i=1;i<=n;i++)
        fac[i]=fac[i-1]*i;
    scanf("%d",&len);
    dp[0][0]=1;
    for(int i=1;i<=n;i++)
    for(int j=n;j>=0;j--)
    for(int k=0;k<=len;k++)
    if(j>0&&k>=a[i])
    dp[j][k]+=dp[j-1][k-a[i]];
    double sum=0;
    for(int i=1;i<=n;i++)
    for(int k=1;k<=len;k++)
    sum+=fac[i]*fac[n-i]*dp[i][k];
    printf("%f\n",sum/fac[n]);
}
