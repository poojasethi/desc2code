#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <iostream>
#include <algorithm>
using namespace std;
double f[100][100];
int a[100];
int n,p;
int main()
{
	scanf("%d",&n);
	for(int i = 1; i <= n; i++) scanf("%d",&a[i]);
	scanf("%d",&p);
	f[0][0] = 1; 
	for(int k = 1; k <= n; k++)
		for(int i = n-1; i >= 0; i--)
			for(int j = 0;j <= p; j++)
			if(a[k] + j <= p)
				f[i+1][a[k]+j]+=(i+1)*f[i][j]*(1.0/(n-i));
	double ans(0);
	double k(0);
	for(int i = 1; i <= n+1; i++)
	{
		double tmp(0);
		for(int j = 0; j <= p; j++)
			tmp += f[i][j];
		ans += (i-1)*(1-k-tmp);
		k += (1-k-tmp);
	}
	printf("%lf\n",ans);
	return 0;
}
