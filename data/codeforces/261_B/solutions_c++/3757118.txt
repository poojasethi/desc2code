#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int maxn=52;
double dp[maxn][maxn];
double f[maxn];
int a[maxn];
int main()
{
    f[0]=1;
    for(int i=1;i<=50;i++)
    f[i]=f[i-1]*i;
    int n,p;
    while(scanf("%d",&n)!=EOF)
    {
        for(int i=0;i<n;i++)
        scanf("%d",&a[i]);
        scanf("%d",&p);
        double ans=0;   
        dp[0][0]=1;
        for(int i=0;i<n;i++)
        for(int k=p;k>=a[i];k--)
        for(int j=1;j<=i+1;j++)
        {
            dp[j][k]+=dp[j-1][k-a[i]];
        }
        for(int i=1;i<=n;i++)
        for(int j=1;j<=p;j++)
        ans+=f[i]*f[n-i]*dp[i][j];
        printf("%.10lf\n",ans/f[n]);
    }
}
