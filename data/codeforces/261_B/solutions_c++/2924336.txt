#include <stdio.h>
double dp[51][51],fact[51],res;
int arr[51],n,p;
int main(){
    fact[0]=1;
    for (int i=1; i<51; i++) fact[i]=fact[i-1]*i;
    scanf("%d",&n);
    for (int i=0; i<n; i++) scanf("%d",&arr[i]);
    scanf("%d",&p);
    dp[0][0]=1;
    for (int i=0; i<n; i++)
        for (int j=n-1; j>=0; j--)
            for (int l=0; l<p; l++)
                if (l+arr[i]<=p&&dp[j][l]!=0) dp[j+1][l+arr[i]]+=dp[j][l];
    for (int i=1; i<=n; i++)
        for (int j=1; j<=p; j++) res+=fact[i]*fact[n-i]*dp[i][j];
    printf("%.9lf",res/fact[n]);
}
