#include <bits/stdc++.h>
using namespace std;
int sa[255],ta[255];
int main(){
	string s,t;cin>>s>>t;
	int i,diff= 'a'-'A';
	for(i=0;i<s.size();++i)
		sa[s[i]]++;
	for(i=0;i<t.size();++i)
		ta[t[i]]++;
	int a=0,b=0,m;
	for(i=0;i<255;++i)
		if(ta[i]>0){
			m = min(ta[i],sa[i]);
			a+=m;
			ta[i]-=m,sa[i]-=m;
		}
	for(i=0;i<255;++i){
		if(ta[i]>0){
			m = sa[i+diff];
			if(i>='a')m=sa[i-diff];
			m = min(ta[i],m);
			ta[i]-=m;b+=m;
		}
	}
	cout<<a<<" "<<b<<endl;
}
