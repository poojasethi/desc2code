#include<cstdio>
#include<string>
#include<cstring>
using namespace std;
const int N=2e5+1;
int a,b,n,m,i;
char s[N],t[N];
int hs[130],ht[130];
int main(){
    scanf("%s %s",s,t);
    n=strlen(s);m=strlen(t);
    for(i=0;i<n;++i) hs[s[i]]++;
    for(i=0;i<m;++i) ht[t[i]]++;
    for(char ch='A';ch<='Z';++ch){
        int t=min(hs[ch],ht[ch])+min(hs[ch+32],ht[ch+32]);
        a+=t;
        b+=min(hs[ch]+hs[ch+32],ht[ch]+ht[ch+32])-t;
    }
    printf("%d %d\n",a,b);
    return 0;
}
