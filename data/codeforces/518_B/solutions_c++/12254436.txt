#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s,t;
	int Y=0,sum=0;
	int a,b,A,B;
	cin>>s>>t;
	for(int i=0;i<26;i++)
	{
		a=count(s.begin(),s.end(),i+'a');
		b=count(t.begin(),t.end(),i+'a');
		Y+=(a<=b)? a:b;
		A=count(s.begin(),s.end(),i+'A');
		B=count(t.begin(),t.end(),i+'A');
		Y+=(A<=B)? A:B;
		sum+=min(a+A,b+B);
	}
	cout<<Y<<" "<<sum-Y<<"\n";
	return 0;
}
