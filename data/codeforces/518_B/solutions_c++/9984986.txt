#include<bits\stdc++.h>
using namespace std;
int main(){
	string s, t;
	int ans1 = 0, ans2 = 0;
	int a[500] = { 0 }, b[500] = { 0 };
	cin >> s >> t;
	for (int i = 0; i < s.length(); i++)
		a[s[i]]++;
	for (int i = 0; i < t.length(); i++)
		b[t[i]]++;
	for (int i = 0; i < 500; i++){
		int tmp = min(a[i], b[i]);
		ans1 += tmp;
		a[i] -= tmp, b[i] -= tmp;
	}
	for (int i = 65; i < 97; i++){
		int tmp = min(a[i] + a[i + 32], b[i] + b[i + 32]);
		ans2 += tmp;
	}
	printf("%d %d\n", ans1, ans2);
	return 0;
}
