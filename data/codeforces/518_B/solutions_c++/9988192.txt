#include <iostream>
#include <cstring>
using namespace std;

int N,M,cnt[200]; char s[300000],t[300000];

int inv(char c){
    if (c<='Z') return 'a'+c-'A';
    return 'A'+c-'a';
}


int main(){
    cin >> s;
    cin >> t;
    N=strlen(s);
    M=strlen(t);

    int i,yay=0,oops=0;
    for (i=0; i<M; i++) cnt[t[i]]++;

    for (i=0; i<N; i++) if (cnt[s[i]])  yay++,cnt[s[i]]--,s[i]='1';
    for (i=0; i<N; i++) if (s[i]!='1') if (cnt[inv(s[i])])  oops++,cnt[inv(s[i])]--;

    cout << yay << " " << oops;
    return 0;
}
