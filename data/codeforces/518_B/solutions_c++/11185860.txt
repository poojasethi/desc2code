#include <iostream>

using namespace std;

int main()
{
	char ch;
    int a[200020]={0};
    bool v[200020]={0};
    string s,t;
    int x,y;
    cin>>s>>t;
    for(int i=0;i<t.length();++i)
    {
    a[t[i]]++;
    }
    for (int i=0;i<s.length();i++)
    {
    if(a[s[i]])
    {
    a[s[i]]--;
    x++;
    v[i]=1;
    }
    }

    for(int i=0;i<s.size();i++)
    {
        if(v[i])continue;
        if(isupper(s[i])) ch=tolower(s[i]);else ch=toupper(s[i]);
        if(a[ch]) a[ch]--,y++;
    }
    cout<<x<<" "<<y;
}
