#include <bits/stdc++.h>
using namespace std;
typedef long long  ll;
int main()
{
	ios_base::sync_with_stdio(false);
	string s,t;cin>>s>>t;
	int cc[200];memset(cc,0,sizeof cc);
	for(int i=0;i<(int)t.size();i++)
		cc[(int)t[i]]++;
	int yay=0,whoops=0;
	for(int i=0;i<(int)s.size();i++)
		if(cc[(int)s[i]]>0)yay++,cc[(int)s[i]]--,s[i]='!';
	for(int i=0;i<(int)s.size();i++)
	{
		char c=(s[i]==tolower(s[i]))?(toupper(s[i])):(tolower(s[i]));
		if(cc[(int)c]>0)whoops++,cc[(int)c]--;
	}
	cout<<yay<<' '<<whoops<<endl;
	return 0;
}
