#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;

int n,m,k,l,res,i,j,res1,p,b,a;
int d[10005],g[1005];
string s,s1;
int main(){

cin>>s>>s1;
n=s.size();
m=s1.size();
for(i=0;i<n;i++){
    d[s[i]]++;
}
for(i=0;i<m;i++){
    g[s1[i]]++;
}

for(i=0;i<150;i++){
    a=min(d[i],g[i]);
    res+=a;
    d[i]-=a;
    g[i]-=a;
//    if(a>0) cout<<i;
}
for(i=0;i<95;i++){
    res1+=min(d[i],g[i+32]);
    res1+=min(g[i],d[i+32]);

}

cout<<res<<" "<<res1;
}
