#include<bits/stdc++.h>
using namespace std;

int a1[200];
int b1[200];
int main()
{
    string a,b;

    cin >> a >> b;
    int sum=0,sum1=0,x,y;
    for(int i=0; i<a.size(); i++)
        a1[a[i]]++;
    for(int i=0; i<b.size(); i++)
        b1[b[i]]++;
    for(int i='A'; i<='Z'; i++)
    {
        x=min(a1[i],b1[i]);
        y=min(a1[i+32],b1[i+32]);
        sum+=x+y;
        sum1+=min(a1[i]+a1[i+32],b1[i]+b1[i+32]);
    }
    cout << sum << " " << abs(sum1-sum) << endl;

}
