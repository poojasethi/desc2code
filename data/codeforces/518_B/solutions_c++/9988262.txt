#include<bits/stdc++.h>

using namespace std;

string S,A;
int i,w,y,r;

int C[3500];
bool H[500005];

int main(){

	cin >> S >> A;
	
	r  = 'a' - 'A';

	for(i=0 ; i<A.size() ; i++)
		C[A[i]]++;
	
	for(i=0 ; i<S.size() ; i++)
		if(C[S[i]])
		{
			C[S[i]]--;
			y++;
			H[i] = 1;
		}
		
	for(i=0 ; i<S.size() ; i++)
		if(!H[i])
		{
			if(S[i] <= 'Z' && C[S[i]+r])
			{
				C[S[i]+r]--;
				w++;
			}

			else if(S[i] >= 'a' && C[S[i]-r])
			{
				C[S[i]-r]--;
				w++;
			}
		}

	cout << y << ' ' << w ;

	return 0;
}
