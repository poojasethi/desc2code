#include <cstdio>

const int N = 200200;

int c[999], i, x, y;
bool b[N];
char s[N], t[N];

int main()
{
	scanf("%s%s", s, t);

	for (i = 0; s[i]; i++)
		++c[s[i]];

	for (i = 0; t[i]; i++)
		if (c[t[i]])
			++x, b[i] = true, --c[t[i]];

	for (i = 0; t[i]; i++)
		if (!b[i])
		{
			char p = t[i];
			if (p >= 'a' && p <= 'z')
				p -= 32;
			else
				p += 32;
			if (c[p])
				++y, --c[p];
		}

	printf("%d %d\n", x, y);

	return 0;
}
