#include<iostream>
using namespace std;

const unsigned char dif='A'-'a';
int len,a1,a2,a[500];
unsigned char c,s[200010],t[200010];

char cr(char c)
{
	return (c>='a'&&c<='z')?(c+dif):(c-dif);
}

int main()
{
	ios_base::sync_with_stdio(0);
	
	cin>>s>>t;
	
	for(int i=0;t[i];i++) a[t[i]]++;
		
	for(int i=0;s[i];i++)
		if(a[s[i]])
		{
			a1++;
			a[s[i]]--;
			s[i]=1;
		}
		
	for(int i=0;s[i];i++)
		if(s[i]!=1&&a[cr(s[i])])
		{
			a2++;
			a[cr(s[i])]--;
		}
		
	cout<<a1<<" "<<a2;
	
	return 0;
}
