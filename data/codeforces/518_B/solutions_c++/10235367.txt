#include<stdio.h>
#include<string.h>
char str[200050],s[200050];
int f[256],g[200050];
int len,t1,t2;
int oth(int x){
	if(x>='a'&&x<='z')return x-'a'+'A';
	return x-'A'+'a';
}
int main(){
	scanf("%s",s);
	scanf("%s",str);
	len=strlen(str);
	for(int i=0;i<len;i++){
		f[str[i]]++;
	}
	len=strlen(s);
	for(int i=0;i<len;i++){
		if(f[s[i]]>0){
			g[i]=1;f[s[i]]--;t1++;
		}
	}
	for(int i=0;i<len;i++){
		if(!g[i]){
			if(f[oth(s[i])]>0){
				f[oth(s[i])]--;t2++;
			}
		}
	}
	printf("%d %d\n",t1,t2);
}
