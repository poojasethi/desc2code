#include <bits/stdc++.h>

using namespace std;

string s,t;

int cont[500],cont2[500];

int main(){
	cin >> s >> t;

	for(int i=0;i<t.size();i++){
		cont[t[i]]++;		
	}
	int r1=0;
	int r2=0;
	for(int i=0;i<s.size();i++){
		cont2[s[i]]++;	
	}

	for(int i=0;i<500;i++){
		int m=min(cont[i],cont2[i]);		
		r1+=m;
		cont[i]-=m;
		cont2[i]-=m;
	}

	for(char i='a';i<='z';i++){
		r2+= min(cont[i],cont2[i-'a'+'A'])+min(cont2[i],cont[i-'a'+'A']);	
	}
	printf("%d %d\n",r1,r2);
}
