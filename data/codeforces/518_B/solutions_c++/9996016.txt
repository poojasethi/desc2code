#include<bits/stdc++.h>
using namespace std;
int x,y,da[200001];
char ch;
map<char,int>m;
string s,t;
main()
{
    cin>>s>>t;
    for(int i=0;i<t.size();i++)m[t[i]]++;
    for(int i=0;i<s.size();i++)
        if(m[s[i]])
        {
            m[s[i]]--;
            x++;
            da[i]=1;
        }
    for(int i=0;i<s.size();i++)
    {
        if(da[i])continue;
        if(isupper(s[i])) ch=tolower(s[i]);else ch=toupper(s[i]);
        if(m[ch]) m[ch]--,y++;
    }
    cout<<x<<' '<<y;
}
