#include<bits/stdc++.h>

int main()
{
	int i,j,k,n,p,u,y=0,w=0;
	char s[200001],t[200001];
	scanf("%s%s",s,t);
	n=strlen(s);
	u=strlen(t);
	int a[126]={0};
	int b[126]={0};
	for(i=0;i<n;++i)
	{
		a[s[i]]++;
	}
	for(i=0;i<u;++i)
	{
		b[t[i]]++;
		if(a[t[i]])
		{
			a[t[i]]--;
			y++;
			b[t[i]]--;
		}
	}
	for(i=64;i<92;++i)
	{
		if(a[i]&&b[i+32])
		{
			w+=a[i]<b[i+32]?a[i]:b[i+32];
		}
	}
	for(i=96;i<124;++i)
	{
		if(a[i]&&b[i-32])
		{				                        
			w+=a[i]<b[i-32]?a[i]:b[i-32];
		}
	}
	printf("%d %d\n",y,w);
	return 0;
}
