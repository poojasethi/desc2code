#include<iostream>
#include<algorithm>
#include<string>
#include<map>
using namespace std;
map<char,int>c,d;
int main(){
	string a,b;
	cin>>a>>b;
	for(int i = 0; i<a.size(); i++) c[a[i]]++;
	for(int i = 0; i<b.size(); i++) d[b[i]]++;
	int ans1 = 0,ans2 = 0;
	for(char i = 'A'; i<='Z';i++) ans1 += min(c[i],d[i]);
	for(char i = 'a'; i<='z';i++) ans1 += min(c[i],d[i]);
	for(char i = 'A'; i<='Z';i++) ans2 += min(c[i]+c[i+32],d[i]+d[i+32]);
	cout<<ans1<<" "<<ans2-ans1<<endl;
	return 0;
}
