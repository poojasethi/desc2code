#include<bits/stdc++.h>
#define fr(i,n) for(i=0;i<n;i++)
#define ifr(i,n) for(i=n-1,i>=0;i--)
#define reset(x,y) memset(x,y,sizeof(x))
using namespace std;



string s,t;
int a[10000]={0},b[10000]={0};

int main(){
	cin>>s>>t;
	int y=0,w=0;
	int i,l=s.length();
	fr(i,l)
	a[s[i]-'A']++;
	l=t.length();
	fr(i,l)
	b[t[i]-'A']++;
	
	fr(i,60){
	int bx = min(a[i],b[i]);
	y+=bx;
	a[i]-=bx;
	b[i]-=bx;
	}
	for(i=32;i<60;i++)
	a[i-32]+=a[i],b[i-32]+=b[i];
	
	
	fr(i,26)
	w+= min(a[i],b[i]);
	
	cout<<y<<" "<<w<<endl;
	

return 0;
}
