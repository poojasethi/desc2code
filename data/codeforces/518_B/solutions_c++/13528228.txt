#include<bits/stdc++.h>
using namespace std;
string s,t;
int cnt[300],yay,whoops,h[200005];
int main ()
{
	cin>>s>>t;
	for(int i=0; i<t.length(); i++)
		cnt[t[i]]++;
	for(int i=0; i<s.length(); i++)
	{
		if(cnt[s[i]]>0)
			yay++,
				cnt[s[i]]--,
				h[i]=1;
	}
	for(int i=0; i<s.length(); i++)
	{
		char c=(s[i]>96)?s[i]-32:s[i]+32;
		if(cnt[c]>0 && !h[i])
			whoops++,
				cnt[c]--;
	}
		cout<<yay<<' '<<whoops;

}
