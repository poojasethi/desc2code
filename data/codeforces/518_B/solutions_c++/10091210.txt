#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main(){
	string s, t;
	int A[300] = {};
	cin >> s >> t;
	for(int i = 0; i < t.size(); i++)
		A[t[i] - 'A']++;
	int con = 0;
	for(int i = 0; i < s.size(); i++)
		if(A[s[i] - 'A'] > 0){con++; A[s[i] - 'A']--; s[i] = 254;}
	int con2 = 0;
	for(int i = 0; i < s.size(); i++){
		if(s[i] < 'a') s[i] += 'a' - 'A';
		else s[i] -= 'a' - 'A';
		if(A[s[i] - 'A'] > 0){con2++; A[s[i] - 'A']--;}
	}
	cout << con << " " << con2;
	return 0;
}
