#include<cstdio>
#include<algorithm>
using namespace std;
int a[58],ans1,ans2,c[58];
char b;
int main(){
    while(scanf("%c",&b),b!='\n'){
        a[b-'A']++;
    }
    while(scanf("%c",&b),b!='\n'){
        if(a[b-'A'])a[b-'A']--,ans1++;
        else c[b-'A']++;
    }
    for(int i=0;i<26;i++){
        if(a[i])ans2+=min(a[i],c[i+32]);
    }
    for(int i=32;i<58;i++)
        if(a[i])ans2+=min(a[i],c[i-32]);
    printf("%d %d\n",ans1,ans2);
    return 0;
}
