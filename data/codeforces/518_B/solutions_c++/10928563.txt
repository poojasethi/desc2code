#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int MM = 2 * 100000 +10 ;
char s[MM],t[MM];
int sa[333],ta[333];
int main(){
	scanf("%s",s);
	scanf("%s",t);
	int len1=strlen(s),len2=strlen(t);
	for(int i=0;i<len1;i++)sa[s[i]]++;
	for(int i=0;i<len2;i++)ta[t[i]]++;
	int ans=0,ans2=0;
	for(char i='A';i<='Z';i++){
		int pd=min(sa[i],ta[i])+min(sa[i+32],ta[i+32]);
		ans+=pd;
		ans2+=min(sa[i]+sa[i+32],ta[i]+ta[i+32])-pd;
	}
	printf("%d %d\n",ans,ans2);
	return 0;
}
