#include <bits/stdc++.h>
typedef long long LL;
#define rep(i,n) for(int i = 0;i < n;i ++)

const int N = 200010;
char s[N], t[N];
int cnt[128];
int main() {
	scanf("%s%s", s, t);
	int n = strlen(s), m = strlen(t);
	int r1 = 0, r2 = 0;
	rep(i, n)
		cnt[s[i]] ++;
	rep(i, m) {
		if(cnt[t[i]] > 0)
			r1 ++;
		cnt[t[i]] --;
	}
	for(char c = 'a'; c <= 'z'; c ++) {
		if(1LL * cnt[c] * cnt[c - 32] >= 0)	continue;
		r2 += std::min(abs(cnt[c]), abs(cnt[c - 32]));
	}
	printf("%d %d\n", r1, r2);
	return 0;
}
