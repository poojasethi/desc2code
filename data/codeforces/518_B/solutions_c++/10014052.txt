#include<iostream>
#include<string>
#include<ctype.h>
using namespace std;

const int maxn=200100;
string s,t;
int cnt[210]={};
int vis[maxn]={};

int main(){
	cin>>s>>t;
	for (int i=0;i<t.length();i++){
		cnt[t[i]]++;
	}
	int y=0,n=0;
	for (int i=0;i<s.length();i++){
		if (cnt[s[i]]){
			y++;
			cnt[s[i]]--;
			vis[i]=1;
		}
	}
	for (int i=0;i<s.size();i++){
		if (vis[i]) continue;
		if (isupper(s[i])&&cnt[s[i]+32]){
			cnt[s[i]+32]--;n++;
		}
		if (islower(s[i])&&cnt[s[i]-32]){
			cnt[s[i]-32]--;n++;
		}
	}
	cout<<y<<" "<<n<<endl;
	return 0;
}
