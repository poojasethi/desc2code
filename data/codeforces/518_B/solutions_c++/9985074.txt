#include <bits/stdc++.h>

using namespace std;

const int MAXN = 210000;

char s[MAXN];
int c[256],d[256];
int A,B;

int main()
{
	scanf("%s",&s);
	memset(c,0,sizeof(c));
	memset(d,0,sizeof(d));
	A = B = 0;
	for(int i = 0;s[i];i++) c[s[i]]++;

	scanf("%s",s);
	for(int i = 0;s[i];i++) if (c[s[i]])
	{
		c[s[i]]--;
		A++;
	} else d[s[i]]++;

	for(int i = 'a';i <= 'z';i++) B += min(d[i],c[i-'a'+'A']);
	for(int i = 'A';i <= 'Z';i++) B += min(d[i],c[i-'A'+'a']);

	printf("%d %d\n", A,B);

	return 0;
}
