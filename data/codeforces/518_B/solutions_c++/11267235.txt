#include <bits/stdc++.h>
using namespace std;

int y, w, a[127], b[127];
string s, t;
int main() {
    int i, k;
    cin >> s >> t;
    for (i = 0; i < s.size(); i++) a[s[i]]++;
    for (i = 0; i < t.size(); i++) b[t[i]]++;
    for (i = 'A'; i <= 'Z'; i++) {
        k = min(a[i], b[i]);
        a[i] -= k, b[i] -= k; y += k;
        k = min(a[i + 32], b[i + 32]);
        a[i + 32] -= k, b[i + 32] -= k; y += k;
        w += min(a[i], b[i + 32]);
        w += min(a[i + 32], b[i]);
    }
    cout << y << ' ' << w;
}
