#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

const int maxn = 2*100000+10;

char s[maxn],t[maxn];
int num[200];
int main(){
    scanf("%s%s",s,t);
    int m = strlen(t), n = strlen(s);
    for(int i=0;i<m;i++) num[t[i]]++;
    int ans1 = 0, ans2 = 0;

    for(int i=0;i<n;i++)
        if(num[s[i]]){num[s[i]]--; ans1++; s[i]=0;}
    for(int i=0;i<n;i++) if(s[i]){
        if(num[s[i]^32]){num[s[i]^32]--; ans2++; }
    }
    printf("%d %d\n",ans1,ans2);
    return 0;

}
