#include <bits/stdc++.h>
#define rep(I, N) for (int I = 0; I < (N); ++I)
using namespace std;
map<char,int> m1,m2;
int x,y;
int main()
{
    string a,b;
    cin>>a>>b;
    rep(i,a.size()) m1[a[i]]++;
    rep(i,b.size()) if(m1[b[i]]) m1[b[i]]--,x++;else m2[b[i]]++;
    for(char i='a';i<='z';i++) y+=min(m1[i],m2[toupper(i)]),y+=min(m1[toupper(i)],m2[i]);
    cout<<x<<" "<<y;
}
