#include<iostream>
#include<string>
using namespace std;
int a[150],b[150],m,i,ans1,ans2;
string s1,s2;
int main()
{
cin>>s1>>s2;
for(i=0;i<s1.size();i++)
a[s1[i]]++;
for(i=0;i<s2.size();i++)
b[s2[i]]++;
for(i='A';i<='Z';i++)
{
m=min(a[i],b[i]);
a[i]-=m;
b[i]-=m;
ans1+=m;
m=min(a[i+32],b[i+32]);
a[i+32]-=m;
b[i+32]-=m;
ans1+=m;
ans2+=min(a[i],b[i+32]);
ans2+=min(a[i+32],b[i]);

}
cout<<ans1<<" "<<ans2;
return 0;
}
