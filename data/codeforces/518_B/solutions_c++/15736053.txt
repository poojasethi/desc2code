#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int main()
{
	int c,d,e[123]={0},f[123]={0},i,j,s=0,g,h=0;
	char a[200005],b[200005];
	scanf("%s %s",a,b);
	c=strlen(a);
	d=strlen(b);
	for(i=0;i<c;i++) e[a[i]]++;
	for(i=0;i<d;i++) f[b[i]]++;
	for(i='A';i<='z';i++){
		g=min(e[i],f[i]);
		s+=g;
		e[i]-=g;
		f[i]-=g;
	}
	for(i='A',j='a';i<='Z';i++,j++){
		h+=min(e[i],f[j]);
		h+=min(e[j],f[i]);
	}
	printf("%d %d",s,h);
	return 0;
}
