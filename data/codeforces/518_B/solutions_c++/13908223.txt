#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int a1[2][2][26];
char s[200001];
int main()
{
	int a=0,b=0,c;
	for(int ii=0;ii<2;++ii)
	{
		scanf("%s",s);
		for(int i=0;s[i];++i)
		if(s[i]<97)++a1[ii][1][s[i]-65];
		else ++a1[ii][0][s[i]-97];
	}
	for(int i=0;i<26;++i)
	{
		c=min(a1[0][0][i],a1[1][0][i])+min(a1[0][1][i],a1[1][1][i]);
		a+=c;
		b+=min(a1[0][0][i]+a1[0][1][i],a1[1][0][i]+a1[1][1][i])-c;
	}
	printf("%d %d\n",a,b);
}
