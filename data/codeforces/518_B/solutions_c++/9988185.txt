#include <bits/stdc++.h>
using namespace std;
int main(){
	string s1,s2;
	cin>>s1>>s2;
	int carry=1;
	vector<int> c1(256,0);
	vector<int> c2(256,0);
	for(int i=0;i<s1.length();++i)
		++c1[s1[i]];
	for(int i=0;i<s2.length();++i)
		++c2[s2[i]];
	int yay=0,whoops=0;
	for(int i=0;i<256;++i){
		int m=min(c1[i],c2[i]);
		c1[i]-=m;
		c2[i]-=m;
		yay+=m;
	}
	for(int i=0;i<26;++i){
		char minu=i+'a';
		char mayu=i+'A';
		whoops+=min(c1[minu]+c1[mayu],c2[minu]+c2[mayu]);
	}
	cout<<yay<<" "<<whoops;
}
