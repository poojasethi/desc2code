#include<bits/stdc++.h>

using namespace std;

string str,ans;
int H[10000],H2[200006];
int i,y,w;

int main(){
	cin >> str >> ans;

	for(i=0 ; i<ans.size() ; i++){
		H[ans[i]]++;
	}
	for(i=0 ; i<str.size() ; i++){
		if(H[str[i]]){
			H[str[i]]--;
			y++;
			H2[i]=1;
		}
	}

	for(i=0 ; i<str.size() ; i++){
		if(!H2[i]){
			if(str[i] <= 'Z'){
				if(H[str[i]+'a'-'A']){
					H[str[i]+'a'-'A']--;
					w++;
				}
			}
			else{
				if(H[str[i]-'a'+'A']){
					H[str[i]-'a'+'A']--;
					w++;
				}
			}
				
		}
	}

	cout << y << ' ' << w << endl;
}
