#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
char s[200001], t[200001];
int a[200], b[200], x, y;
int main () {
	scanf("%s %s", s, t);
	int lens=strlen(s);
	int lent=strlen(t);
	for (int i=0; i<lens; i++) {
		a[s[i]]++;
	}
	for (int i=0; i<lent; i++) {
		b[t[i]]++;
	}
	for (int i=0; i<26; i++) {
		char t1 = i+'A', t2 = i+'a';
		int temp=min(a[t1], b[t1])+min(a[t2], b[t2]);
		x+=temp;
		y+=min(a[t1]+a[t2], b[t1]+b[t2])-temp;
	}
	printf("%d %d\n", x, y);
	return 0;
}
