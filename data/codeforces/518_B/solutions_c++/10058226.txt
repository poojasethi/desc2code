#include <bits/stdc++.h>
using namespace std;
int main ()
{
string str,s;
char p;
int arr[200001];
cin>>str>>s;
int c=0,x=0;
map<char,int>m;
for(int i=0;i<s.size();i++)
{
    m[s[i]]++;
}
for(int i=str.size()-1;i>=0;i--)
{
   if(m[str[i]])
   {
       m[str[i]]--;
       c++;
    str.erase(i,1);
   }

}
for(int i=0;i<str.size();i++)
{

      if(isupper (str[i]))
        p=tolower (str[i]);
        else p=toupper (str[i]);
      if(m[p])
      {
          m[p]--;
          x++;
      }

}
cout<<c<<" "<<x;
}
