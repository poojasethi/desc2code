#include <bits/stdc++.h>
#define rep(I, N) for (int I = 0; I < (N); ++I)
using namespace std;
map<char,int> m1,m2;
int ans,rans;
int main()
{
    string a,b;
    cin >> a >> b ;
    rep(i,a.size()) m1[a[i]]++;
    rep(i,b.size()) if(m1[b[i]]) m1[b[i]]--,ans++;else m2[ b[i] ]++;
    for(char i='a';i<='z';i++)
    {
        rans+=min(m1[i],m2[toupper(i)]);
        rans+=min(m1[toupper(i)],m2[i]);
    }

    cout << ans << " " << rans ;

}
