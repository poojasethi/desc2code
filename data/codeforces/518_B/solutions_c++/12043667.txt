#include <iostream>
using namespace std;
int S[160], T[160], yay, whoops;
string s, t;
int main() {
    cin >> s >> t;
    for (int i = 0; i < s.length(); i++) S[s[i]]++;
    for (int i = 0; i < t.length(); i++) T[t[i]]++;
    for (int i = 'A'; i <= 'z'; i++) {
        int x = min(S[i], T[i]);
        yay += x;
        S[i] -= x, T[i] -= x;
    }
    for (int i = 'A'; i <= 'Z'; i++) {
        int x = min(S[i], T[i | 32]),
            y = min(S[i | 32], T[i]);
        whoops += x+y;
    }
    cout << yay << " " << whoops;
}
