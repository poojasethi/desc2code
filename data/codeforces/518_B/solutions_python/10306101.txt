y,w,d = 0,0,{}
s,t = raw_input(),raw_input()
used = [0] * len(s)
for x in t: d[x] = d.get(x,0) + 1
for x in xrange(len(s)): 
    if s[x] in d and d[s[x]] > 0:
        d[s[x]] -= 1; y += 1; used[x] += 1
for x in xrange(len(s)):
    if not used[x]:
        if s[x].lower() in d and d[s[x].lower()] > 0:
            d[s[x].lower()] -= 1; w += 1
        elif s[x].upper() in d and d[s[x].upper()] > 0:
            d[s[x].upper()] -= 1; w += 1
print y,w
