a={}
for x in raw_input():
    a[x]=1 if x not in a else a[x]+1
b={}
for x in raw_input():
    b[x]=1 if x not in b else b[x]+1
ans1=ans2=0;ch='a'
for i in a:
    if i not in b: b[i]=0
    minn=min([a[i],b[i]])
    a[i]-=minn;b[i]-=minn
    ans1+=minn
for i in a:
    if a[i]!=0:
        if i.isupper(): ch=i.lower()
        else: ch=i.upper()
        if ch not in b: b[ch]=0
        ans2+=min([a[i],b[ch]])
print ans1,ans2
