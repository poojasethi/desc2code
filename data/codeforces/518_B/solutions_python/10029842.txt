a=[0]*200
b=[0]*200
sa=raw_input()
sb=raw_input()
for x in sa: a[ord(x)]+=1
for x in sb: b[ord(x)]+=1
ch=ans1=ans2=0
for i in set(sa):
    minn=min([a[ord(i)],b[ord(i)]])
    a[ord(i)]-=minn;b[ord(i)]-=minn
    ans1+=minn
for i in set(sa):
    if a[ord(i)]:
        if i.isupper(): ch=ord(i.lower())
        else: ch=ord(i.upper())
        ans2+=min([a[ord(i)],b[ch]])
