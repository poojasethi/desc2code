import sys
from collections import defaultdict

s = sys.stdin.readline().strip()
t = sys.stdin.readline().strip()

d = defaultdict(int)
for c in t:
	d[c] += 1

ura = 0
opa = 0

found = set()
for i, c in enumerate(s):
	if d[c] > 0:
		d[c] -= 1
		ura += 1
		found.add(i)

for i, c in enumerate(s):
	if i in found:
		continue
	if c.islower():
		c = c.upper()
	else:
		c = c.lower()
	if d[c] > 0:
		d[c] -= 1
		opa += 1

print ura, opa
