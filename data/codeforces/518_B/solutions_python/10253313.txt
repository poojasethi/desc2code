s = raw_input()
t = raw_input()

ans1 = ans2 = 0

cnt = [0] * 256
for c in t:
    cnt[ord(c)] += 1

used = [False] * len(s)
for i in range(len(s)):
    if cnt[ord(s[i])] > 0:
        ans1 += 1
        cnt[ord(s[i])] -= 1
        used[i] = True

for i in range(len(s)):
    uc = chr(ord(s[i]) ^ 0x20)
    if (not used[i]) and cnt[ord(uc)] > 0:
        cnt[ord(uc)] -= 1
        ans2 += 1

print ans1, ans2
