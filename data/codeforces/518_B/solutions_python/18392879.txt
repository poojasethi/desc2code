#!/usr/bin/env python

msg = raw_input()
text = raw_input()

dict = {}
for i in text:
	if i not in dict.keys():
		dict[i] = text.count(i)

yay = 0
whoops = 0

for i in msg:
	# if dict.get(i) == None:
	# 	dict[i] = -1
	# else:
	if dict.get(i) != None:
		if (dict[i] <= 0):
			# whoops+=1
			dict[i] -= 1
		else:
			yay +=1
			dict[i] -= 1
	else:
		dict[i] = -1

for i in msg:
	if dict.get(i) != None and dict[i] < 0:
		temp = i
		if (temp.isupper()):
			temp = temp.lower()
		else:
			temp = temp.upper()
		if (dict.get(temp) != None and dict[temp] > 0):
			whoops+=1
			dict[temp]-=1 
			dict[i] += 1

print yay , " " , whoops
