def f(a,b):
    aL,aU = a
    bL,bU = b
    mL = min(aL,bL)
    mU = min(aU,bU)
    yay = mL + mU
    whoops = min( aL+aU, bL+bU ) - mL -mU
    return yay,whoops

x = raw_input()
y = raw_input()

A1 = [ x.count(i) for i in 'abcdefghijklmnopqrstuvwxyz' ]
A2 = [ x.count(i) for i in 'ABCDEFGHIJKLMNOPQRSTUVWXYZ' ]
B1 = [ y.count(i) for i in 'abcdefghijklmnopqrstuvwxyz' ]
B2 = [ y.count(i) for i in 'ABCDEFGHIJKLMNOPQRSTUVWXYZ' ]
A = zip(A1,A2)
B = zip(B1,B2)
#lower, then upper
# A is the intended
# B is the source
yay = 0
whoops = 0
for i in xrange(26):
    y0,w0 = f(A[i],B[i])
    yay += y0
    whoops += w0
print yay,whoops
