s = raw_input()
t = raw_input()
times = [0] * 256
for c in t: times[ord(c)] += 1
vis = [False] * len(s)
ans1 = 0
ans2 = 0
for i in xrange(len(s)):
	if times[ord(s[i])] > 0:
		times[ord(s[i])] -= 1
		vis[i] = True
		ans1 += 1
for i in xrange(len(s)):
	if not vis[i]:
		index = ord(s[i]) + ord('A') - ord('a') if 'a' <= s[i] <= 'z' else ord(s[i]) + ord('a') - ord('A')
		if times[index] > 0:
			times[index] -= 1
			ans2 += 1
print ans1, ans2
