from collections import Counter as C
a=C(raw_input())
b=C(raw_input())

r=0
R=0
for (k, v) in a.items():
    if k in b: 
        r += min(b[k],v)
        a[k] -= min(b[k],v)
        b[k] -= min(b[k],v)
for (k, v) in a.items():
    for e in [k.lower(), k.upper()]:
        if e in b:
            R += min(b[e],v)
            b[e] -= min(b[e],v)
print r,R
