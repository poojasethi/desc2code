from sys import stdin
import sys
from fractions import gcd
import string

from collections import defaultdict
d = defaultdict(int)

s = stdin.readline().rstrip()
t = stdin.readline().rstrip()

for i in t:
	d[i] += 1


yays = 0
whoops = 0

asd = [0]*len(s)

for idx, i in enumerate(s):
	if d[i] > 0:
		d[i] -= 1
		yays += 1
		asd[idx] = 1

for idx, i in enumerate(s):
	if asd[idx] == 0:
		if i.islower():
			newi = i.upper()
		else:
			newi = i.lower()
		if d[newi] > 0:
			d[newi] -= 1
			whoops += 1

print yays, whoops
