cnt = [0]*125
 
a = raw_input()
b = raw_input()
 
y, w = (0,)*2

for c in b:
    cnt[ord(c)] += 1

aux = ''

for c in a:
    if(cnt[ord(c)] > 0):
        y += 1
        cnt[ord(c)] -= 1
    else:
        aux += c

for c in aux:
    if(c >= 'a'):
        if(cnt[ord(c) - 32] > 0):
            w += 1
            cnt[ord(c) - 32] -= 1
    else:
        if(cnt[ord(c) + 32] > 0):
            w += 1
            cnt[ord(c) + 32] -= 1

print y, w
