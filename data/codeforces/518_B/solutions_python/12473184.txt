from collections import Counter

def solve(s, t):
    t_counter = Counter(t)
    yay = whoops = 0
    potential_whoops_inds = []
    
    for i, c in enumerate(s):
        if t_counter[c] > 0:
            t_counter[c] -= 1
            yay += 1
        else:
            potential_whoops_inds.append(i)
    
    for i in potential_whoops_inds:
        c = s[i]
        opp = opposite_case(c)
        
        if t_counter[opp] > 0:
            whoops += 1
            t_counter[opp] -= 1
    
    return yay, whoops


def opposite_case(c):
    if c.islower():
        return c.upper()
    else:
        return c.lower()
    
s = raw_input()
t = raw_input()

x, y = solve(s, t)

print '{} {}'.format(x, y)
