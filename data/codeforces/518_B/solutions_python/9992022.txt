s = raw_input().strip()
t = raw_input().strip()
n = len(s)

tmap = {}
for e in t:
	if e not in tmap: tmap[e] = 1
	else: tmap[e] += 1

yay = 0
whoops = 0
news = ""
for e in s:
	if e in tmap and tmap[e] > 0:
		tmap[e] -= 1
		yay += 1
	else: news += e

for e in news:
	if e.isupper():
		e2 = e.lower()
		if e2 in tmap and tmap[e2] > 0:
			whoops += 1
			tmap[e2] -= 1
	else:
		e2 = e.upper()
		if e2 in tmap and tmap[e2] > 0:
			whoops += 1
			tmap[e2] -= 1

print yay, whoops
