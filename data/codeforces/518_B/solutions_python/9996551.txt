import string

s = raw_input()
t = raw_input()

table = {x: 0 for x in string.lowercase + string.uppercase}

for x in t:
    table[x] += 1
u, o = 0, 0
r = ''
for x in s:
    k = table[x]
    if k > 0:
        u += 1
        table[x] -= 1
        r += x
    else:
        r += '*'
for i in xrange(len(s)):
    if r[i] != '*':
        continue
    x = s[i]
    y = chr(ord(x) ^ 32)
    k = table[y]
    if k > 0:
        o += 1
        table[y] -= 1
print u, o
