# coding: utf-8

s = raw_input()
t = raw_input()

d = {}
l = [False] * len(s)

# yay and whoops
y = 0
w = 0

for e in t:
	try:
		d[e] += 1
	except KeyError:
		d[e] = 1

# yay loop
for i in xrange(len(s)):
	if d.has_key(s[i]):
		if d[s[i]] > 0:
			y += 1
			d[s[i]] -= 1
			l[i] = True

# whoops loop
for i in xrange(len(s)):
	case = s[i].lower() if s[i].isupper() else s[i].upper()
	if d.has_key(case) and not l[i]:
		if d[case] > 0:
			w += 1
			d[case] -= 1

print y, w
