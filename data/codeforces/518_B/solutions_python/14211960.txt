Y = W = 0
TT = 256 * [0]
F = 200001 * [0]
s = raw_input().strip()
t = map(ord, raw_input().strip())
for i in t:
    TT[i] += 1
for i in xrange(len(s)):
    x = ord(s[i])
    if TT[x]:
        Y += 1
        TT[x] -= 1
        F[i] = 1
s = s.swapcase()        
for i in xrange(len(s)):
    if F[i]:
        continue
    x = ord(s[i])
    if TT[x]:
        W += 1
        TT[x] -= 1
print Y, W
