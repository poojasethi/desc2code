#!/usr/bin/python2

a = list(raw_input())
b = raw_input()
l = len(a)
m = {}
ura = 0
ups = 0

for y in b:
    if y in m:
        m[y] += 1
    else:
        m[y] = 1

for x in range(len(a)):
    if a[x] in m and m[a[x]] > 0:
        m[a[x]] -= 1
        ura += 1
        a[x] = ' '

for z in range(len(a)):
    if a[z] != ' ':
        if a[z] == a[z].upper():
            a[z] = a[z].lower()
        else:
            a[z] = a[z].upper()

        if a[z] in m and m[a[z]] > 0:
            m[a[z]] -= 1
            ups += 1

print ura, ups
