#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
#define xx first
#define yy second
using namespace std;

typedef pair<int,int> pii;

pii pt[222222];
int bh[222222];

int n;
int i,u;
int ans,e,ne;
int ass[222222];
int ggg[222222];

inline bool cmp(const int &a,const int &b)
{ return pt[a]<pt[b]; }

int main()
{
//	freopen("306B.in","r",stdin);
//	freopen("306B.out","w",stdout);
	
	scanf("%d",&n);
	scanf("%d",&n);
	
	for (i=0;i<n;i++)
	{
		scanf("%d%d",&pt[i].xx,&pt[i].yy);
		bh[i]=i;
		pt[i].yy+=pt[i].xx;
	}
	sort(bh,bh+n,cmp);
	
	ans=0;
	e=0;
	ne=0;
	for (i=0;i<n;i++)
	{
		u=bh[i];
		if (pt[u].yy<=ne)
			continue;
		if (pt[u].xx>e)
		{
			ans++;
			e=max(ne,pt[u].xx);
		}
		ne=pt[u].yy;
		ass[ans]=u;
	}
	
	for (i=1;i<=ans;i++)
		ggg[ass[i]]=1;
	printf("%d\n",n-ans);
	for (i=0;i<n;i++)
		if (!ggg[i])
			printf("%d ",i+1);
	
	printf("\n");
	
	return 0;
}
