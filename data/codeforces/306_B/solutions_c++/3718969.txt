#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

const int maxn = 2111111;

struct Node
{
    int l,r,d,i;
    bool flag;
}node[maxn],ans[maxn];
int tot;
bool cmp(Node a,Node b)
{
    return a.l < b.l || a.l == b.l && a.d > b.d;
}

int main()
{
    int n,m,b;
    scanf("%d%d",&n,&m);
    for(int i = 0;i<m;i++)
    {scanf("%d%d",&node[i].l,&node[i].d);node[i].r = node[i].l + node[i].d - 1;node[i].i = i+1;node[i].flag = false;}
    sort(node,node+m,cmp);
    tot = 0;
    int r = node[0].r;
    node[0].flag = true;
    for(int i = 1;i<m;)
    {
        int k = i;
        while(node[i].l <= r + 1 && i < m)
        {
            if(node[i].r > node[k].r) k = i;i++;
        }
        if( node[k].r > r)
        {
            node[k].flag = true;
            r = node[k].r;
        }
    }
    for(int i = 0;i<m;i++)
    if(!node[i].flag) ans[tot++] = node[i];
    printf("%d\n",tot);
    for(int i = 0;i<tot;i++)
    printf("%d ",ans[i].i);
    printf("\n");
}
