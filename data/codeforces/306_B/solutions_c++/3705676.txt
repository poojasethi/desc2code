#include <iostream>
#include <algorithm>
#include <stack>
#include <cstdio>

using namespace std;

struct mt{
	int l,r,d,i;
	bool b;
};

mt a[200001];
int n,m;

bool cmp(mt i, mt j)
{
	return i.l<j.l || i.l==j.l && i.d>j.d;
}

int main(){
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	cin>>n>>m;
	for (int i=1; i<=m; i++)
	{
		cin>>a[i].l>>a[i].d;
		a[i].r=a[i].l+a[i].d-1;
		a[i].i=i;
		}
	sort(a,a+m+1,cmp);
	//for (int i=1; i<=m; i++) cout<<a[i].i<<' '<<a[i].l<<' '<<a[i].r<<endl;
	a[1].b=1;
	int r=a[1].r;
	int i=2;
	while (i<=m)
	{
		int k=i;
		while (a[i].l<=r+1 && i<=m)
		{
			if (a[i].r>a[k].r) k=i;
			i++;
			}
		//	cout<<k<<endl;
		if (a[k].r>r){
		a[k].b=1;
		r=a[k].r;
	}	
		}
//	cout<<endl;
	stack <int> st;
	for (int i=1; i<=m; i++) if (!a[i].b) st.push(a[i].i);
	cout<<st.size()<<endl;
	while (!st.empty())
	{
		cout<<st.top()<<' ';
		st.pop();
		}
	
	}
