#include<cstdio>
#include<algorithm>
#include<iostream>
using namespace std;

struct node
{
    int x, y, r;
    bool operator < (node const &T) const
    {
        return x == T.x ? y > T.y : x < T.x;
    }
} a[200005];
int n, m, mx, y, mp;
int v[200005];

int main()
{
    scanf("%d%d", &m, &n);
    for(int i=0;i<n;i++)
    {
        scanf("%d%d", &a[i].x, &a[i].y);
        a[i].y += a[i].x - 1;//right
        a[i].r = i + 1;//indexing
    }
    sort(a, a + n);

    for (int i=0, j=0; i<n;)
    {
        mx = y = a[i].y;
        mp = i;
        v[a[i].r] = 1;
        //cout<<i<<" "<<j<<endl;
        while (j < n)
        {
            if (a[j].x > y + 1) break;
            if (a[j].y > mx) mx = a[j].y, mp = j;
            ++j;
        }
        if (mp == i) i = j;
        else i = mp;
    }
    int ret = 0;
    for(int i=1;i<=n;i++) ret += 1 - v[i];
    printf("%d\n", ret);
    for(int i=1;i<=n;i++) if (!v[i]) printf("%d ", i);
    return 0;
}
