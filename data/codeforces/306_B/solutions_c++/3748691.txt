#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <vector>
using namespace std;
struct ot
{int l,r,i;
 bool f;
} a[200005];
bool cmp(ot a, ot b)
{if (a.l==b.l) return a.r>b.r;
  else return a.l<b.l;
}
vector <int> ans;
int main()
{//freopen("i.txt","r",stdin);
int n,m; cin>>n>>m;
for (int i=0; i<m;i++)
  {int x;
   cin>>a[i].l>>a[i].r;
   a[i].r+=a[i].l;a[i].i=i;a[i].f=0;
  }
sort(a,a+m,cmp);
a[0].f=1;
int r=a[0].r;
for (int i=1;i<m;)
{int k=i;
  while (a[i].l<=r&&i<m)
   {if (a[i].r>a[k].r) k=i;
     i++;
   }
  if (a[k].r>r) {a[k].f=1; r=a[k].r;}
}
for (int i=0; i<m;i++) if (!a[i].f) ans.push_back(a[i].i);
cout<<ans.size()<<endl; for (int i=0; i<ans.size();i++) cout<<ans[i]+1<<" ";
return 0;
}
