#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

const int maxm=200005;

struct data
{
	int st, en, num;
};

bool operator<(const data &a, const data &b)
{
	return a.st<b.st || a.st==b.st && a.en>b.en;
}

int N, M;
data query[maxm];
bool choose[maxm];

int main()
{
	scanf("%d%d",&N,&M);
	for (int i=1;i<=M;i++)
	{
		scanf("%d%d",&query[i].st,&query[i].en);
		query[i].en+=query[i].st-1;
		query[i].num=i;
	}
	sort(query+1,query+M+1);
	int right(0), select(0), cnt(M);
	for (int i=1;i<=M;i++)
	{
		if (query[i].st>right+1 && select && query[select].en>right)
		{
			choose[query[select].num]=true;
			cnt--;
			right=query[select].en;
		}
		if (query[i].st>right+1)
		{
			choose[query[i].num]=true;
			cnt--;
			right=query[i].en;
		} else
		if (!select || query[i].en>query[select].en)
			select=i;
	}
	if (select && query[select].en>right)
	{
		choose[query[select].num]=true;
		cnt--;
	}
	printf("%d\n",cnt);
	bool been(false);
	for (int i=1;i<=M;i++) if (!choose[i])
	{
		if (been) putchar(' ');
		been=true;
		printf("%d",i);
	}
	putchar('\n');
	return 0;
}
