#include <iostream>
#include <algorithm>
using namespace std;

struct S {
  int l, r, i;
  bool operator<(const S &x) const {
    return l < x.l || l == x.l && r > x.r;
  }
} a[200000];
int b[200000], c;

int main() {
  int n, m;
  cin >> n >> m;
  for (int i(0); i < m; ++i) {
    cin >> a[i].l >> a[i].r;
    a[i].r += a[i].l;
    a[i].i = i + 1;
  }
  sort(a, a + m);
  for (int i(1), j(0), k; i < m; ) {
    while (i < m && a[i].l >= a[j].r)
      j = i++;
    int x(j);
    for (k = i; k < m && a[k].l <= a[j].r; ++k)
      if (a[k].r > a[x].r)
        x = k;
    for (k = i; k < m && a[k].l <= a[j].r; ++k)
      if (k != x)
        b[c++] = a[k].i;
    j = x;
    i = k;
  }
  cout << c << endl;
  for (int i(0); i < c - 1; ++i)
    cout << b[i] << ' ';
  if (c)
    cout << b[c - 1] << endl;
}
