#include <cstdio>
#include <algorithm>
using namespace std;
const int N=200004;
struct rec {int num,l,r;} p[N];
int n,m,s[N],tot,ans[N];
bool del[N];
bool operator <(rec a,rec b) {return (a.l<b.l)||((a.l==b.l)&&(a.r>b.r));}
int main()
{
	int i; scanf("%d%d",&n,&m);
	for (i=1;i<=m;i++)
	{
		scanf("%d%d",&p[i].l,&p[i].r);
		p[i].r+=p[i].l-1;
		p[i].num=i;
	}
	sort(p+1,p+m+1);
	for (i=1;i<=m;i++) del[i]=false;
	int top=0;
	for (i=1;i<=m;i++)
	{
		if ((i>1)&&(p[i].l==p[i-1].l)) {del[p[i].num]=true; continue;}
		if (p[s[top]].r>=p[i].r) {del[p[i].num]=true; continue;}
		while ((top>1)&&(p[s[top-1]].r>=p[i].l-1)) del[p[s[top]].num]=true,top--;
		s[++top]=i;
	}
	tot=0;
	for (i=1;i<=m;i++) if (del[i]) ans[++tot]=i;
	printf("%d\n",tot);
	if (tot)
	{
		for (i=1;i<tot;i++) printf("%d ",ans[i]);
		printf("%d\n",ans[tot]);
	}
	return 0;
}
