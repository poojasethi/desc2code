#include<iostream>
#include<algorithm>
using namespace std;
int a,s,d,f,g,h,j,k,l,i,n,m;
typedef struct{
        int st;
        int end;
        int ind;
        int chck;
        } seg;
seg q[200003];
bool sorting(seg v1,seg v2){
     if(v1.st==v2.st) return v1.end>v2.end;
     return v1.st<v2.st;
     }
main(){
cin>>n>>m;
j=m;
for(i=0;i<m;i++){
cin>>q[i].st>>q[i].end;
q[i].end+=q[i].st;
q[i].ind=i;
}
sort(q,q+m,sorting);
for(i=0;i<m;){
a=q[i].end;
k=a;
l=-1;
q[i].chck=1;
j--;
for(i=i+1;i<m && q[i].st<=a;i++){
if(q[i].end>k) {k=q[i].end;l=i;}
}
if(l!=-1) i=l;
}
cout<<j<<endl;
for(i=0;i<m;i++){
if(q[i].chck==0) cout<<q[i].ind+1<<" ";
}
}
