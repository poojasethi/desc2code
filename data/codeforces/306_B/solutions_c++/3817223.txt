#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 200010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
    int a,b,id;
} a[N];
int i,j,k,n,m,ff[N],cnt,l;
inline bool cc1(const ww &a,const ww &b) {
    return a.a<b.a||a.a==b.a&&a.b>b.b;
}
int main() {
    scanf("%d%d",&n,&m);
    For(i,1,m) scanf("%d%d",&a[i].a,&a[i].b),a[i].id=i,a[i].b+=a[i].a-1;
    sort(a+1,a+m+1,cc1); l=k=1; ff[a[1].id]=1;
    For(i,2,m)
    if (a[i].a-1<=a[k].b) {
        if (a[i].b>a[l].b) l=i;
    } else {
        ff[a[l].id]=1; k=l;
        if (a[i].b>a[l].b) l=i;
    }
    ff[a[l].id]=1;
    k=0;
    For(i,1,m) if (!ff[i]) k++;
    printf("%d\n",k);
    For(i,1,m) if (!ff[i]) printf("%d ",i);
    return 0;
}
