#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
#define xx first
#define yy second
using namespace std;

typedef pair<int,int> pii;
typedef pair<pii,int> ppi;

ppi pt[222222];

int n;
int i;
int ans,e,ne;
int ass[222222];
int ggg[222222];

int main()
{
//	freopen("306B.in","r",stdin);
//	freopen("306B.out","w",stdout);
	
	scanf("%d",&n);
	scanf("%d",&n);
	
	for (i=0;i<n;i++)
	{
		scanf("%d%d",&pt[i].xx.xx,&pt[i].xx.yy);
		pt[i].yy=i;
		pt[i].xx.yy+=pt[i].xx.xx;
	}
	sort(pt,pt+n);
	
	ans=0;
	e=0;
	ne=0;
	for (i=0;i<n;i++)
	{
		if (pt[i].xx.yy<=ne)
			continue;
		if (pt[i].xx.xx>e)
		{
			ans++;
			e=max(ne,pt[i].xx.xx);
		}
		ne=pt[i].xx.yy;
		ass[ans]=pt[i].yy;
	}
	
	for (i=1;i<=ans;i++)
		ggg[ass[i]]=1;
	printf("%d\n",n-ans);
	for (i=0;i<n;i++)
		if (!ggg[i])
			printf("%d ",i+1);
	
	printf("\n");
	
	return 0;
}
