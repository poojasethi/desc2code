#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
struct ppp{
	int x,y,id;
} a[2200000];
int i,j,k,l,n,m,use[2200000],x,y,res;
bool cmp(const ppp&a,const ppp&b){
	return a.x<b.x || a.x==b.x && a.y>b.y;
}
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;++i){
		scanf("%d%d",&x,&y);
		y+=x-1;
		a[i].x=x; a[i].y=y; a[i].id=i;
	}
	sort(a+1,a+m+1,cmp);
	l=-1; k=0; a[0].y=-1;
	for(i=1;i<=m;++i){
		if(a[i].x<=l+1){
			if(a[i].y>a[k].y)k=i;
		}else{
			use[a[k].id]=1;
			if(a[i].x<=a[k].y+1){
				l=a[k].y;
				if(a[i].y>a[k].y)k=i;
			}else{
				l=a[i].y;
				use[a[i].id]=1;
				k=i;
			}
		}
	}
	use[a[k].id]=1;
	for(i=1;i<=m;++i)if(!use[i])res++;
	printf("%d\n",res);
	for(i=1;i<=m;++i)if(!use[i])printf("%d ",i);
}
