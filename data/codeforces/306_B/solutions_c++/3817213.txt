#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 200005
#define rep(i,n) for (int i=0;i<n;i++)
int x[N],y[N],p[N],b[N],ans,n,m,L,k;
bool cmp(const int i,const int j){return x[i]<x[j];}
int main()
{
	scanf("%d%d",&n,&m);
	rep(i,m) scanf("%d%d",x+i,&L),y[i]=x[i]+L,p[i]=i;
	sort(p,p+m,cmp),L=0;
	while (k<m){
		int opt=-1,j=-1;
		while (x[p[k]]<=L && k<m){if (y[p[k]]>opt) j=p[k],opt=y[j]; k++;}
		if (j<0) L=x[p[k]];
		while (x[p[k]]<=L && k<m){if (y[p[k]]>opt) j=p[k],opt=y[j]; k++;}		
		if (opt>L) b[j]=1,L=opt;
//		printf("%d %d\n",j,opt);
		}
	rep(i,m) ans+=!b[i]; printf("%d\n",ans);
	rep(i,m) if (!b[i]) printf("%d ",i+1); puts("");
	return 0;
}
