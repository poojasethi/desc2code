#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;	
struct master{
int x,y;
int num;
}cd[200001]={0};
bool cmp(master a,master b){
if(a.x==b.x)return a.y<b.y;
return a.x<b.x;
}
bool bj[200001]={0};
bool in[200001]={0};
int d=0;
int m,n;
main(){
int a,b,c,i,j;
int z;
int pos,max;
int ans=0;
scanf("%d%d",&m,&n);
z=n;
for(a=1;a<=n;a++){
scanf("%d%d",&b,&c);
cd[a].x=b;
cd[a].y=b+c-1;
cd[a].num=a;
}
sort(cd+1,cd+n+1,cmp);
for(i=1;i<=n;i++)if(cd[i].x!=cd[i+1].x)cd[++d]=cd[i];
n=d;
for(i=1;i<=n;i++)bj[i]=1;
for(i=1;i<=n;i++)if(bj[i]==1)
{
max=0;
pos=0;	
j=i+1;
while(j<=n){
if(cd[j].x>cd[i].y+1)break;
if(cd[j].y>max){max=cd[j].y;pos=j;}
j++;
}
if(pos!=0){
if(max<=cd[i].y)bj[pos]=0;
for(a=i+1;a<j;a++)if(a!=pos)bj[a]=0;
}
}
for(a=1;a<=n;a++)if(bj[a]==1)in[cd[a].num]=1;
for(a=1;a<=z;a++)
if(in[a]==0)ans++;
printf("%d\n",ans);
d=ans;
for(a=1;a<=z;a++)
if(in[a]==0){
if(d==ans){printf("%d",a);d--;}
else printf(" %d",a);
}
if(ans!=0)printf("\n");
}
