#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
#define maxm 2000010
#define maxn 200010
using namespace std;
int n,m;
pair<int,int> R[maxm];
int maxv[maxm],sum[maxm];
void read()
{
	scanf("%d %d",&m,&n);
	for(int i=1,l,r;i<=n;++i)
	{
		scanf("%d %d",&l,&r);
		r+=l-1,R[l]=max(R[l],make_pair(r,i));
		++sum[l],--sum[r+1];
	}
	for(int i=1;i<=m+1;++i)
		sum[i]+=sum[i-1],maxv[i]=R[maxv[i-1]]>R[i]?maxv[i-1]:i;
}
void work()
{
	static bool use[maxn];
	int cnt=0,now=0;
	for(int i=1;i<=m;++i)
		if(i>now&&sum[i])
		{
			use[R[i].second]=true;
			now=R[i].first;
			++cnt;
			while(R[maxv[now+1]].first>now)
			{
				use[R[maxv[now+1]].second]=true;
				++cnt;
				now=R[maxv[now+1]].first;
			}
		}
	cnt=n-cnt;
	printf("%d\n",cnt);
	for(int i=1;i<=n;++i)
		if(!use[i])
			printf("%d%c",i,--cnt?' ':'\n');
}
int main()
{
	read();
	work();
	return 0;
}
