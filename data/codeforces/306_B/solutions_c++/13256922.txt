#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
#define N 200050
struct Point {int x,y,z;} li[N];
bool operator< (const Point &x,const Point &y)
 {return x.x<y.x||(x.x==y.x&&x.y>y.y);}
int n,m,ans,ma,Ans[N],cnt;bool b[N];
inline int Read()
 {
    int x=0;char y;
    do y=getchar(); while (y<'0'||y>'9');
    do x=x*10+y-'0',y=getchar(); while (y>='0'&&y<='9');
    return x;
 }
int main()
 {
    m=Read();n=Read();
    for (int i=1;i<=n;i++)
      li[i].x=Read(),li[i].y=li[i].x+Read()-1,
      li[i].z=i;
    sort(li+1,li+n+1);ma=-1;li[0].y=-1;
    for (int i=1;i<=n;)
     {
        int k=i,l=0;
        while (k<=n&&li[k].x<=ma+1)
         {
            if (li[k].y>li[l].y&&li[k].y>ma) l=k;
            k++;
         }
        ans+=k-i;
        if (l)
          b[li[l].z]=1,ans--,i=k,ma=li[l].y; else
        if (k==n+1) break; else
          ma=li[k].y,i=k+1,b[li[k].z]=1;
     }
    cout <<ans<<endl;
    if (!ans) return 0;
    sort(Ans+1,Ans+ans+1);
    for (int i=1;i<=n;i++)
     if (!b[i]) printf("%d ",i);
    puts("");
    return 0;
 }
