#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 200010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int a,b,id;
} a[N];
int i,j,k,n,m,r,num,rr;
int f[N];
inline bool cc1(const ww &a,const ww &b) {
	return a.a<b.a||a.a==b.a&&a.b>b.b;
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,m) scanf("%d%d",&a[i].a,&k),a[i].b=a[i].a+k-1,a[i].id=i;
	sort(a+1,a+m+1,cc1);
	For(i,1,m) if (a[i].a>r+1) {
		if (rr>r) f[num]=1;
		r=rr; num=0;
		if (a[i].b>rr) rr=a[i].b,num=a[i].id;
	} else if (a[i].b>rr) rr=a[i].b,num=a[i].id;
	if (rr>r) f[num]=1;
	k=0;
	For(i,1,m) if (!f[i]) k++;
	printf("%d\n",k);
	For(i,1,m) if (!f[i]) printf("%d ",i);
	printf("\n");
	//for(;;);
	return 0;
}
