#include <cstdio>
#include <algorithm>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
#define N 2222222

int l[N]={0},till=0;
int p[N];
int ans[N],ptr=0;
int main() {
    int n,m,a,b;
    scanf("%d%d",&n,&m);
    rep(i,m) {
        scanf("%d%d",&a,&b);
        if(l[a]<b) {
            if(l[a]!=0) ans[ptr++]=p[a];
            l[a]=b; p[a]=i+1;
        } else ans[ptr++] = i+1;
    }
    rep(i,n+10) if(l[i]) {
        if(i>till) {
            till = i+l[i]-1;
        }
        else if(i+l[i]-1<=till) {
            ans[ptr++] = p[i];
        }
        else {
            a=till+1;
            b=(i+l[i]-1)-till;
            if(l[a]<b) {
                if(l[a]!=0) ans[ptr++]=p[a];
                l[a]=b; p[a]=p[i];
            } else ans[ptr++] = p[i];
        }
    }
    printf("%d\n",ptr);
    if(ptr) {
    sort(ans,ans+ptr);
    rep(i,ptr) printf("%d ",ans[i]); }
}
