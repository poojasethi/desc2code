}
int Use[200005];

    for(i=1;i<=m;i++){

        A[i].L=j;
        A[i].R=j+k-1;
        A[i].x=i;
    }
    sort(A+1,A+1+m);
    int R=0;
//    j=1;
    /*for(i=1;i<=m&&A[i].L==A[1].L;i++)

    for(i=1;i<=m;){

            if(A[j].R>A[k].R)k=j;
            j++;
        }
        i=j;
        if(A[k].R>R){
            Use[A[k].x]=1;
            R=A[k].R;
        }
