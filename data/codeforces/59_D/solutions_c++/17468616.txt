#include<iostream>
#include<cstdio>
#include<map>
using namespace std;
int num[300010];
int sign[300010];
bool sign2[300010];
map<int,int>mp;
int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=3*n;i++)
    {
        int temp;
        scanf("%d",&temp);
        num[temp]=i;
    }
    for(int i=1;i<=n;i++)
    {
        int temp1,temp2,temp3;
        scanf("%d%d%d",&temp1,&temp2,&temp3);
        sign[temp1]=i;
        sign[temp2]=i;
        sign[temp3]=i;
        if(num[temp1]<num[temp2]&&num[temp1]<num[temp3])
            mp[temp1]=i;
        else if(num[temp2]<num[temp3])
            mp[temp2]=i;
        else mp[temp3]=i;
    }
    int m;
    cin>>m;
    if(mp.find(m)==mp.end())
    {
        for(int i=1;i<=3*n;i++)
        {
            if(i==m)
                continue;
            printf("%d ",i);
        }
        cout<<endl;
    }
    else
    {
        int tot=0;
        for(int i=1;i<=3*n;i++)
        {
            if(i==m)
                continue;
            if(sign[i]<=sign[m])
            {
                sign2[i]=true;
                printf("%d ",i);
                if(sign[i]==sign[m])
                    tot++;
                if(tot==2)
                    break;
            }
        }
        for(int i=1;i<=3*n;i++)
        {
            if(i==m)
                continue;
            if(!sign2[i])
                printf("%d ",i);
        }
        cout<<endl;
    }
    return 0;
}
