#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int a[300005];
int b[300005];
int r[100005][3];
int g[300005];
int ok[300005];

bool cmp(int x, int y) {
	return b[x] < b[y];
}

int main(void) {
	int n, k;
	scanf("%d", &n);
	for(int i=1;i<=3*n;i++) {
		scanf("%d", &a[i]);
		b[a[i]]=i;
	}
	for(int i=1;i<=n;i++) {
		scanf("%d%d%d", &r[i][0], &r[i][1], &r[i][2]);
		sort(r[i], r[i]+3, cmp);
		g[r[i][0]]=g[r[i][1]]=g[r[i][2]]=i;
	}
	scanf("%d", &k);
	for(int i=1;i<g[k];i++)
		ok[r[i][0]]=ok[r[i][1]]=ok[r[i][2]]=1;
	if(r[g[k]][0] != k) {
		for(int i=1;i<=3*n;i++)
			if(i!=k) printf("%d ", i);
		puts("");
	} else {
		int L=r[g[k]][1], R=r[g[k]][2];
		if(L>R) swap(L, R);
		for(int i=1;i<L;i++) if(ok[i]) printf("%d ", i);
		ok[L]=ok[R]=1;
		printf("%d ", L);
		for(int i=L+1;i<R;i++) if(ok[i]) printf("%d ", i);
		printf("%d ", R);
		ok[k]=1;
		for(int i=1;i<=R;i++) if(!ok[i]) printf("%d ", i);
		for(int i=R+1;i<=3*n;i++) if(k!=i) printf("%d ", i);
		puts("");
	}
	return 0;
}
