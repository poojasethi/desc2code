#include<iostream>
#include<cstdio>
#include<string.h>
#include<string>
#include<math.h>
#include<algorithm>
using namespace std;

int a[300009],b[300009];
bool flag[300009];
int n,t;
int k;

int main(){
    scanf("%d",&n);
    for(int i=0;i<3*n;i++) scanf("%d",&b[i]);
    for(int i=0;i<3*n;i++) scanf("%d",&a[i]);
    scanf("%d",&k);
    int pos=0,g,pp=0;
    while(a[pos]!=k) pos++;
    g=pos/3+1;
    for(int i=0;i<3*(g-1);i++) flag[a[i]]=true;
    for(pp=0;flag[b[pp]];pp++);
    if(a[pos]!=b[pp]) sort(a,a+3*n);
    else{
        int temp=-1;
        for(int i=3*(g-1);i<3*g;i++)
            if(i!=pos) temp=max(temp,a[i]);
        sort(a,a+3*g);
        int l;
        for(l=0;a[l]!=temp;l++);
        sort(a+l+1,a+3*n);
    }
    int ii=0;
    for(int i=0;i<3*n;i++){
        if(a[i]!=k){
             ii++;
             if(ii!=3*n-1) printf("%d ",a[i]);
             else printf("%d\n",a[i]);
        }
    }
    return 0;
}
