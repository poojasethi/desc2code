#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<string>
#include<queue>

using namespace std;
const int maxn = 100001;
char sel[maxn*3];
int p[maxn*3];
int n;
int d[3*maxn];


int main()
{
	while(scanf("%d",&n)==1)
	{
		int n3 = n*3;
		int i,j,k;
		for(i=1;i<=n3;i++)
		{
			scanf("%d",p+i);
			sel[i]=0;
		}
		
		for(i=1;i<=n3;i++)	scanf("%d",d+i);
		scanf("%d",&k);
		int ki=-1;
		for(i=1;i<=n3;i+=3)
		{
			if(d[i]==k)	ki = i;
			if(d[i+1]==k)	ki = i+1;
			if(d[i+2]==k)	ki = i + 2;
			if(ki!=-1)	break;
			sel[d[i]] = sel[d[i+1]] = sel[d[i+2]] = 1;
		}
		
		for(j=1;j<=n3;j++)	if(sel[p[j]]==0)	break;
		if(p[j]!=k)
		{
			char f = 0;
			for(i=1;i<=n3;i++)
				if(i!=k)
				{
					if(f)	putchar(' ');f = 1;
					printf("%d",i);
				}
			putchar('\n');
			continue;
		}
		
		if(ki%3==1)	d[ki] = d[ki+2],ki+=2;
		else	
		if(ki%3==2)	d[ki] = d[ki+1],ki++;
		d[ki] = k;
		
		int tp = max(d[ki-2],d[ki-1]);
		sort(d+1,d+1+ki);
		for(i=1;tp!=d[i];i++)
		;
		
		sort(d+i+1,d+1+n3);
		char f = 0;
		for(i=1;i<=n3;i++)
		if(d[i]!=k)
		{
			if(f)	putchar(' ');f = 1;
			printf("%d",d[i]);
		}
		putchar('\n');
	}
	return 0;
}
