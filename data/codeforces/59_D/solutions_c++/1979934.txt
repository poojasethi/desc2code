#include <stdio.h>
#include <algorithm>
#include <vector>
using namespace std;

vector<int> v;
struct data{int i,v;};
struct team{int a[5];}t[100005];
data a[500000];
bool ck[500005];
bool ck2[500005];
int n,k;
bool sf(data a,data b){ return a.i<b.i;}
bool sf2(int x,int y){
    return a[x].v<a[y].v;
}
int main(){
    scanf("%d",&n);
    int i,j,l;
    for(i=1;i<=3*n;i++)scanf("%d",&a[i].i),a[i].v=i;
    sort(a+1,a+3*n+1,sf);
    for(i=1;i<=n;i++)for(j=1;j<=3;j++)scanf("%d",&t[i].a[j]);
    for(i=1;i<=n;i++){
        sort(t[i].a+1,t[i].a+4,sf2);
    }
    scanf("%d",&k);
    for(i=1;i<=n;i++)if(t[i].a[1]==k){
        for(j=1;j<=i;j++){
            for(l=1;l<=3;l++){
                ck[t[j].a[l]]=1;
            }
        }
        int cnt=0;
        for(j=1;j<=3*n;j++){
            if(j==k)continue;
            if(j==t[i].a[2] || j==t[i].a[3]){
                cnt++;
            }
            if(ck[j])printf("%d ",j);
            if(cnt==2){
                for(l=1;l<=j;l++){
                    ck2[l]=ck[l];
                }
                break;

            }
        }for(j=1;j<=3*n;j++){
            if(j==k)continue;
            if(!ck2[j])printf("%d ",j);
        }
        return 0;
    }
    for(i=1;i<=3*n;i++){
        if(i==k)continue;
        printf("%d ",i);
    }
}
