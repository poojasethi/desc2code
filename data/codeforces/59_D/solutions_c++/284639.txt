#include <cstdio>
#include <algorithm>

using namespace std;

#define MAXN 500000

int n, x, k, t, i;
int rating[MAXN], team[MAXN], ans[MAXN], used[MAXN];
int a[MAXN][5];

bool cmp (int a, int b) {return rating[a] < rating[b];}

int main ()
{
 	scanf ("%d", &n);

 	for (i = 0; i < 3*n; i++)
 	{
 	 	scanf ("%d", &x);
 	 	rating[x] = i;
 	}

 	for (i = 0; i < n; i++)
 	{
 		for (int j = 0; j < 3; j++)
 			scanf ("%d", &a[i][j]);
 		sort (a[i], a[i]+3, cmp);
 		team[a[i][0]] = i+1;
 	}

 	scanf ("%d", &k);

// 		printf ("team[%d] = %d\n", k, team[k]);

 	if (!team[k])
 	{
		for (i = 0; i < 3*n; i++)
			ans[i] = (i+1) + ((i+1)>=k);
		t = 3*n-1;
 	}
 	else
 	{
		for (i = 0; i < team[k]; i++)
			for (int j = 0; j < 3; j++)
				used[a[i][j]] = 1;
		used[k] = 0;
		for (i = 1; i <= max (a[team[k]-1][1], a[team[k]-1][2]); i++)
			if (used[i]) ans[t++] = i, used[i] = 2;
		used[k] = 2;
		for (i = 1; i <= 3*n; i++)
			if (used[i] != 2)
				ans[t++] = i;
 	}
	for (i = 0; i < t; i++)
		printf ("%d%c", ans[i], i==t-1?'\n':' ');
}
