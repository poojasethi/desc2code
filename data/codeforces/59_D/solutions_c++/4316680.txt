#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
#include <queue>
//#include <>
using namespace std;

const int MN=3*1e5+40;
#define pb push_back
int p[MN],ep[MN],g[MN],eg[MN];
vector<int> m,w;

int main(){
    int n,k;
    cin >>n;
    for(int i=1;i<=3*n;i++){
        cin >>p[i];
        ep[p[i]]=i;
    }
    for(int i=0;i<3*n;i++){
        cin >>g[i];
        eg[g[i]]=i;
    }
    cin >>k;
    int t=eg[k]/3;
    if( ep[g[t*3]]<ep[k] || ep[g[t*3+1]]<ep[k] || ep[g[t*3+2]]<ep[k] ){
        for(int i=1;i<=3*n;i++) if(i!=k) cout <<i<<" ";
        cout <<endl;
        return 0;
    }
    for(int i=0;i<t*3;i++) m.pb(g[i]);
    for(int i=t*3;i<3*n;i++) if(g[i]!=k) w.pb(g[i]);
    sort(w.begin(),w.begin()+2);
    sort(w.begin()+2,w.end());
    sort(m.begin(),m.end());
    m.pb(n*4);
    w.pb(n*4);
    for(int pw=0,pm=0;(m[pm]!=n*4)||(w[pw]!=n*4);){
        cout <<min(m[pm],w[pw])<<" ";
        if(m[pm]<w[pw]) pm++; else pw++;
    }
    cout <<endl;
    return 0;
}
