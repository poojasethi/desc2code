#include <stdio.h>
#define MAX 100010

struct data{
	int idx, plc, team;
	bool c;
};

data sibal[MAX*3];
int n, baki;
int answer[MAX*3], ac;
bool cap, check[MAX*3];

int main(void){
	int i, j, k;
	int max, cc;
	scanf("%d", &n);

	for (i=1;i<=n*3;i++){
		scanf("%d", &k);
		sibal[k].plc=i;
	}
	for (i=1;i<=n;i++){
		max=1000000;
		for (j=1;j<=3;j++){
			scanf("%d", &k);
			sibal[k].team=i;
			if (max>sibal[k].plc){
				max=sibal[k].plc;
				cc=k;
			}
		}
		sibal[cc].c=1;
	}

	scanf("%d", &baki);
	cap=sibal[baki].c;
	n*=3;
	cc=0;
	if (cap){
		for (i=1;i<=n;i++){
			if (baki==i) continue;
			if (sibal[baki].team>=sibal[i].team){
				check[i]=true;
				answer[++ac]=i;
			}
			if (sibal[baki].team==sibal[i].team){
				cc++;
			}
			if (cc==2) break;
		}
		for (i=1;i<=n;i++){
			if (baki==i || check[i]) continue;
			answer[++ac]=i;
		}
	}else{
		for (i=1;i<=n;i++){
			if (baki==i) continue;
			answer[++ac]=i;
		}
	}
	for (i=1;i<n;i++){
		printf("%d ", answer[i]);
	}
	return false;
}
