#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int a[300300],p[300300];
bool done[300300];
int main()
{
    int n,t;
    int k;
    scanf("%d",&n);
    for(int i=0;i<3*n;i++) scanf("%d",p+i);
    for(int i=0;i<3*n;i++) scanf("%d",a+i);
    scanf("%d",&k);
    int pos=0,group,pp=0;
    while(a[pos]!=k) pos++;
    group=pos/3+1;
    for(int i=0;i<3*(group-1);i++) done[a[i]]=true;
    for(pp=0;pp<3*n;pp++) if(!done[p[pp]])
        break;
   // cout<<group<<" "<<pos<<endl;
    if(a[pos]!=p[pp]) sort(a,a+3*n);
    else
    {
        int larger=-1;
        for(int i=3*(group-1);i<3*group;i++) if(i!=pos)
            larger=max(larger,a[i]);
        sort(a,a+3*group);
        int tpos;
        for(tpos=0;a[tpos]!=larger;tpos++);
       // cout<<larger<<" "<<tpos<<endl;
        sort(a+tpos+1,a+3*n);
    }
    for(int i=0;i<3*n;i++) if(a[i]!=k)
        printf("%d ",a[i]);
    printf("\n");
    return 0;
}
