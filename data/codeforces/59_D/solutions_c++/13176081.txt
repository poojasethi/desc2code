#include <cstdio>

const int N = 100000 + 1;

int n;
int a[N * 3];
int rank[N * 3];
int b[N][3], top[N];
int id[N * 3];
int k;
bool del[N * 3];

int main(){
	scanf("%d", &n);
	for(int i = 1; i <= 3 * n; i++){
		scanf("%d", &a[i]);
		rank[a[i]] = i;
	}
	for(int i = 1; i <= n; i++){
		top[i] = a[3 * n];
		for(int j = 0; j < 3; j++){
			scanf("%d", &b[i][j]);
			id[b[i][j]] = i;
			if (rank[b[i][j]] < rank[top[i]]) top[i] = b[i][j];
		}
	}
	scanf("%d", &k);
	if (top[id[k]] != k){
		for(int i = 1; i <= 3 * n; i++){
			if (i == k) continue;
			printf("%d ", i);
		}
		puts("");
		return 0;
	}
	int up = 0;
	for(int i = 1; i <= 3 * n; i++){
		if (i == k) continue;
		if (id[i] != id[k]) continue;
		if (i > up) up = i;
	}
	for(int i = 1; i <= up; i++){
		if (i == k) continue;
		if (rank[top[id[i]]] <= rank[top[id[k]]]){
			del[i] = true;
			printf("%d ", i);
		}
	}
	for(int i = 1; i <= 3 * n; i++){
		if (i == k) continue;
		if (del[i]) continue;
		printf("%d ", i);
	}
	puts("");
}
