#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int a[100010][3];
int b[300010];
int c[300010];
int mp[300010];
int
main() {
	int n;
	scanf("%d", &n);
	int i, j;
	for(i = 0; i < 3*n; i++) {
		int tmp;
		scanf("%d", &tmp);
		mp[tmp] = i;
	}
	int pos;
	for(i = 0; i < n; i++)
		scanf("%d %d %d", &a[i][0], &a[i][1], &a[i][2]);
	int k;
	scanf("%d", &k);
	int now = mp[k];
	for(i = 0; i < n; i++) {
		for(j = 0; j < 3; j++) {
			if(a[i][j] == k) {
				pos = i;
				break;
			}
		}
		if(j < 3)
			break;
	}
	if(mp[k] > min(min(mp[a[pos][0]], mp[a[pos][1]]), mp[a[pos][2]])) {
		for(i = 1; i <= 3*n; i++) {
			if(i == k)
				continue;
			printf("%d ", i);
		}
		printf("\n");
		return 0;
	}
	int r = 0;
	for(i = 0; i < 3; i++)
		if(a[pos][i] != k)
			b[r++] = a[pos][i];
	int mn = max(b[0], b[1]);
	int m = 0;
	for(i = 0; i < pos; i++) for(j = 0; j < 3; j++) {
		if(a[i][j] < mn)
			b[r++] = a[i][j];
		else
			c[m++] = a[i][j];
	}
	for(i = pos + 1; i < n; i++) for(j = 0; j < 3; j++)
		c[m++] = a[i][j];
	sort(b, b+r);
	sort(c, c+m);
	for(i = 0; i < r; i++)
		printf("%d ", b[i]);
	for(i = 0; i < m; i++)
		printf("%d ", c[i]);
	printf("\n");
	return 0;
}
