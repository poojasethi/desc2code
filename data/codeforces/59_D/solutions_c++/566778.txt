#include<iostream>
#include<cstdio>
#include<fstream>
#include<algorithm>
#include<cmath>
#include<vector>
#include<map>
#include<cstring>
using namespace std;
int a[300005],ans[300005];
int b[100005][5];
int v[300005]={0};

int main()
{
    int n,m,i,j,x,y,z,l,k,zz,yy,r;
    scanf("%d",&n);
    m=3*n;
    for (i=1; i<=m; i++)
     {
         scanf("%d",&x);
         a[x]=i;
     }
    for (i=1; i<=n; i++)
     for (j=1; j<=3; j++)
      scanf("%d",&b[i][j]);
    scanf("%d",&k);
    l=0;
    for (i=1; i<=n; i++)
     {
         x=b[i][1]; y=b[i][2]; z=b[i][3];
         if (a[y]<a[x]) swap(x,y);
         if (a[z]<a[x]) swap(x,z);
         l++; ans[l]=y; v[y]=1;
         l++; ans[l]=z; v[z]=1;
         if (x==k) { yy=y; zz=z; break;}
         l++; ans[l]=x; v[x]=1;
     }
    if (i>n)
    {
        sort(ans+1,ans+l+1);
        for (i=1; i<=l; i++)
         if (i!=k)
          printf("%d ",ans[i]);
        printf("\n");
    }
    else
    {
        sort(ans+1,ans+l+1);
        for (i=1; i<=l; i++)
         if (ans[i]==yy || ans[i]==zz) r=i;
        for (i=1; i<=m; i++)
         if (v[i]==0 && i!=k)
          { l++; ans[l]=i; }
        sort(ans+r+1,ans+l+1);
        for (i=1; i<=l; i++)
         printf("%d ",ans[i]);
        printf("\n");
    }
}
