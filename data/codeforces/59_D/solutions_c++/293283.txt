#include<cstdio>
#include<algorithm>
//          Last Change:  2011-02-21 16:58:37
using namespace std;
#define MX 300003
int n;
int rank[MX],a[MX];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=3*n;++i)
	{
		int x;
		scanf("%d",&x);
		rank[x]=i;
	}
	for(int i=1;i<=3*n;++i)
		scanf("%d",a+i);
	int k;
	scanf("%d",&k);
	for(int i=1;i<=3*n;++i)
		if(a[i]==k)
		{
			int x,y;
			if(i%3==0)
				x=i-2,y=i-1;
			else if(i%3==1)
				x=i+1,y=i+2;
			else
				x=i-1,y=i+1;
			if(rank[k]<rank[a[x]]&&rank[k]<rank[a[y]])
			{
				int t=max(a[x],a[y]);
				if(y%3==0)
				{
					a[i]=a[y],a[y]=k;
					sort(a+1,a+y);
					sort(a+y+1,a+3*n+1);
				}
				else
				{
					sort(a+1,a+i);
					sort(a+i+1,a+3*n+1);
				}
				for(int j=1;j<=3*n;++j)
					if(a[j]==t)
					{
						sort(a+j+1,a+3*n+1);
						break;
					}
				for(int j=1;j<=3*n;++j)
					if(a[j]!=k)
						printf("%d ",a[j]);
				puts("");
			}
			else
			{
				for(int j=1;j<=3*n;++j)
					if(j!=k)
						printf("%d ",j);
			}
			return 0;
		}
}
