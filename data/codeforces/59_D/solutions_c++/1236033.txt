#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<vector>
#include<set>
using namespace std;

bool used[300010];

int dost[300010];
int ch[100010][3];
int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);

	int n, tmp, k;
	cin >> n;

	for(int i = 0; i < 3*n; ++i)
	{
		cin >> tmp;
		dost[tmp] = i;
	}

	for(int i = 0; i < n; ++i)
		for(int j = 0; j < 3; ++j)
			cin >> ch[i][j];
	cin >> k;
	bool found = false;
	vector<int> ans;
	ans.reserve(3*n);
	for(int i = 0; i < n && !found; ++i)
	{
		if(dost[ch[i][0]] > dost[ch[i][1]])
			swap(ch[i][0], ch[i][1]);
		if(dost[ch[i][0]] > dost[ch[i][2]])
			swap(ch[i][0], ch[i][2]);
		if(ch[i][1] > ch[i][2])
			swap(ch[i][1], ch[i][2]);

		if(ch[i][0] == k)
		{                             	
			found = true;
			for(int j = 1; j < 3; ++j)
				used[ch[i][j]] = 1;
			for(int j = 1; j <= ch[i][2]; ++j)
				if(used[j])
					ans.push_back(j);
			used[ch[i][0]] = 1;
			for(int j = 1; j <= ch[i][2]; ++j)
				if(!used[j])
					ans.push_back(j);
			for(int j = ch[i][2] + 1; j <= 3*n; ++j)
				if(j != k) ans.push_back(j);
		}
		else
			for(int j = 0; j < 3; ++j)
				used[ch[i][j]] = true;
	}
	if(found)
		for(int i = 0; i < ans.size(); ++i)
			cout << ans[i] << ' ';
	else
	{
		for(int i = 1; i <= 3*n; ++i)
			if(i != k) cout << i << ' ';
	}
	return 0;
}
