#include<cstdio>
#include<vector>
#include<algorithm>

#define	pb			push_back
#define	rep(i,n)	for(int i=0;i<n;i++)

using namespace std;

typedef	vector<int>	vi;

int main(){
	int n;	scanf("%d",&n);
	static int ord[300000];
	rep(i,3*n)	scanf("%d",ord+i),ord[i]--;
	static int team[100000][3],teamid[300000];
	rep(i,n)rep(j,3){
		scanf("%d",&team[i][j]);
		team[i][j]--;
		teamid[team[i][j]]=i;
	}
	int k;	scanf("%d",&k),k--;

	bool iscaptain=false;
	static bool chosen[300000]={};
	rep(i,3*n){
		int id=ord[i];
		if(id==k && !chosen[id])	iscaptain=true;
		if(!chosen[id]){	// i-th place person is a captain
			chosen[id]=true;
			rep(j,3)	chosen[team[teamid[id]][j]]=true;
		}
		if(id==k)	break;
	}

	if(!iscaptain){
		bool b=false;
		rep(i,3*n)if(i!=k)	printf("%s%d",b?" ":"",i+1),b=true;
	}
	else{
		vi former,latter;
		int mateidmax=0;
		rep(j,3){
			int mateid=team[teamid[k]][j];
			if(mateid!=k)	mateidmax=max(mateidmax,mateid);
		}
		rep(i,3*n)if(i!=k){
			if(chosen[i] && i<=mateidmax)	former.pb(i);
			else							latter.pb(i);
		}
		sort(former.begin(),former.end());
		sort(latter.begin(),latter.end());

		bool b=false;
		rep(i,former.size())	printf("%s%d",b?" ":"",former[i]+1),b=true;
		rep(i,latter.size())	printf("%s%d",b?" ":"",latter[i]+1),b=true;
	}

	return 0;
}
