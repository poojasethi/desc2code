#include <iostream>
#include <memory.h>
using namespace std;
int arr[100000][3],place[300001];
bool mark[300001]={false};
int main()
{
    int n,k,a,b,c;
    cin>>n;
    for (int i=0;i<3*n;i++){
        cin>>a;
        place[a]=i;
    }
    for (int i=0;i<n;++i){
        cin>>a>>b>>c;
        if (place[b]<place[a] && place[b]<place[c]) swap(a,b);
        if (place[c]<place[a] && place[c]<place[b]) swap(a,c);
        arr[i][0]=a;
        arr[i][1]=b;
        arr[i][2]=c;
    }
    cin>>k;
    int x,y,find=true;
    for (x=0;x<n && find;++x)
        for (y=0;y<3 && find;++y)
            if (arr[x][y]==k)
                find=false;
    x--;
    y--;
    //cout<<x<<" " <<y<<endl;
    if (y==0){
        for (int i=0;i<=x;++i)
            for (int j=0;j<3;++j)
                mark[arr[i][j]]=true;
        for (int i=1;i<=3*n;++i)
            if (i!=k && mark[i] && i<=max(arr[x][1],arr[x][2])){
                cout<<i<<" ";
                mark[i]=false;
            }
        
        for (int i=x+1;i<n;++i)
            for (int j=0;j<3;++j)
                mark[arr[i][j]]=true;
        for (int i=1;i<=3*n;++i)
            if (i!=k && mark[i])
                cout<<i<<" ";
    }
    else {
        for (int i=1;i<=3*n;++i){
            if (i!=k)
                cout<<i<<" ";
        }
    }
    cout<<endl;
    return 0;
}
