#include <iostream>

using namespace std;

int a[300010], team[100010][4];
bool marked[300010];
int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < 3 * n; i++)
		cin >> a[i];
	for (int i = 0; i < n; i++)
		for (int j = 0; j < 3; j++)
			cin >> team[i][j];
	int k;
	cin >> k;
	int ptr = 0;
	for (int i = 0; i < n; i++)
	{
		bool captain = false, found = false;
		for (int j = 0; j < 3; j++)
		{
			if (team[i][j] == k)
				if (a[ptr] == k)
					captain = true;
				else
					found = true;
			marked[team[i][j]] = true;
		}
		if (captain)
		{
			int goTo = 0;
			for (int j = 0; j < 3; j++)
				if (team[i][j] != k)
					goTo = max(goTo, team[i][j]);
			for (int i = 1; i <= goTo; i++)
				if (i != k && marked[i])
					cout << i << " ";
			for (int i = 1; i <= 3 * n; i++)
				if (i <= goTo)
				{
					if (i != k && !marked[i])
						cout << i << " ";
				}
				else if (i != k)
					cout << i << " ";
			cout << endl;
			return 0;
		}
		else if (found)
		{
			for (int i = 1; i <= 3 * n; i++)
				if (i != k)
					cout << i << " ";
			cout << endl;
			return 0;
		}
		while (ptr < 3 * n && marked[a[ptr]])
			ptr++;
	}
}
