#include<bits/stdc++.h>
using namespace std;

int n[1000005];
int main()
{
    int p, x, s, c; bool flag1=true, flag2=false;
    cin>>p>>x; s=x-1;
    while((flag1)&&(s<10)){
        s++; n[0]=s; c=0;
        for(int i=1;i<p;i++){ n[i]=(((n[i-1]*x)+c)%10); c=(((n[i-1]*x)+c)/10); }
        if(((n[p-1]*x)+c)==n[0]) { flag1=false; flag2=true; }
    }

    if(flag2) for(int i=p-1;i>=0;i--) printf("%i", n[i]);
    else printf("Impossible");
    return 0;
}
