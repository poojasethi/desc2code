int main() {
	int n, x;
	cin >> n >> x;
	for(int i=1; i<=9; i++) {
		vector<int> num(n);
		num[n-1]=i;
		int carry=0;
		for(int j=n-2; j>=0; j--) {
			num[j]=(num[j+1]*x+carry)%10;
			carry=(num[j+1]*x+carry)/10;
		}
		if(num[0]!=0 && x*num[0]+carry==i) {
			for(int i=0; i<n; i++)
				cout << num[i];
			cout << endl;
			return 0;
		}
	}
	cout << "Impossible\n";
}
