#include<stdio.h>
#include<string.h>
int a[1000000];
int main(){
	int p,x,i,j,f,m,flag=0;
	scanf("%d%d",&p,&x);
	if(p<=1&&x!=1){
		printf("Impossible");
		return 0;
	}
	if(p==1&&x==1){
		printf("1");
		return 0;
	}
	for(i=1;i<=9;i++){
		memset(a,0,sizeof(a));
		f=i;
		a[0]=f*x%10;
		m=f*x/10;
		for(j=1;j<p-1;j++){
			a[j]=(a[j-1]*x+m)%10;
			m=(a[j-1]*x+m)/10;
		}
		if((a[p-2]*x+m)%10==f&&a[p-2]!=0&&(a[p-2]*x+m)/10==0){
			flag=1;
			break; 
		}
	}
	if(flag){
		for(i=p-2;i>=0;i--){
			printf("%d",a[i]);
		}
			printf("%d\n",f);
	}
	else printf("Impossible\n");
}
