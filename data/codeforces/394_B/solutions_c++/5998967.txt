#include<iostream>
#include<cmath>
using namespace std;
   int a[1000000];
int main()
{
    int p,x;
    cin>>p>>x;
 
    
    for (int i=1; i<=9; i++)
    {
        int carry=0;
        a[0]=i;
        for (int j=0; j<p; j++)
        {
            a[j+1]=(x*a[j]+carry)%10;
            carry=(x*a[j]+carry)/10;
        }
        if(a[p]!=i || carry!=0)
        continue;
        if (a[p-1]==0)
        continue;
        for(int j=p-1; j>=0; j--)
        cout<<a[j];
        cout<<endl;
        return 0;
    }
    cout<<"Impossible"<<endl;
   
    return 0;
}
