#include<stdio.h>
#define MAXN 1000005
int dp[MAXN],s1[MAXN],s2[MAXN];
int main()
{
    int p,x,flag=0;
    scanf("%d%d",&p,&x);
    for(int i=1;i<=9;i++)
    {
        int s=p,t=p-1;
        int add=0,a=0;
        dp[s]=i;
        while(s)
        {
             dp[t]=(dp[s]*x+add)%10;
             add=(dp[s]*x+add)/10;
             s1[a]=dp[s];
             s2[a++]=dp[t];
             s--;
             t--;
        }
        if(add)
            continue;
        if(s1[0]==s2[p-1]&&dp[1])
        {
            flag=1;
            break;
        }
    }
    if(flag)
        for(int i=1;i<=p;i++)
            printf("%d",dp[i]);
    else printf("Impossible");
}
