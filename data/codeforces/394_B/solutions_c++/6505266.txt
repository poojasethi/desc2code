#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
char s[1000005];
int main()
{
    int p,x;
    cin>>p>>x;
    int sign=0;
    for(int i=x;i<=9;i++)
    {
     int last=i;
     int jin=0;
     s[0]=i+'0';
    for(int j=1;j<p;j++)
    {
        last=last*x+jin;
        jin=last/10;
        last=last%10;
        s[j]=last+'0';
    }
    if(x*(s[p-1]-'0')+jin==i)
     {
            s[p]='\0';
            sign=1;
            break;
     }
    }
    if(sign==1)
    {
        int len=strlen(s);
        for(int i=len-1;i>=0;i--)   cout<<s[i];
    }
    else cout<<"Impossible";
}
