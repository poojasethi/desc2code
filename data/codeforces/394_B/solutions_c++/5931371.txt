#include <iostream>
#include <cstring>

using namespace std;

int s[1000000];

int main()
{
	cin.sync_with_stdio(false);
	int p, x;
	cin >> p >> x;
	bool ok = false;
	for (int d = 1; d < 10 && !ok; ++d)
	{
		int c = 0;
		int dig = d;
		for (int i = p-1; i > -1; --i)
		{
			s[i] = dig;
			int val = dig*x+c;
			c = val/10;
			dig = val%10;
		}
		if (dig == d && s[0] != 0 && c == 0)
			ok = true;
	}
	if (!ok)
		cout << "Impossible\n";
	else
	{
		for (int i = 0; i < p; ++i)
			cout << s[i];
		cout << "\n";
	}
	return 0;
}
