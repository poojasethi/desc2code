#include<iostream>
using namespace std;

int main(){

    int p,x;
    cin >> p >> x; 

    for(int d1 = 1; d1 < 10; d1++){
        int number[p];
        number[p-1] = d1;
        int r = 0;
        for(int i = p-2; i >= 0; i--){ //alert
            number[i] = (number[i+1]*x + r) % 10;
            r = (number[i+1]*x + r) / 10;
        }
        if(x*number[0] + r == d1 && number[0] > 0){
            for(int j = 0; j < p; j++) cout << number[j];
            cout << endl;
            return 0;
        }
    }

    cout << "Impossible" << endl;
    return 0;
}
