#include <stdio.h>

int N, M;
int ans[1000010];

int main() {
	//why are you stalking me?
	scanf("%d %d", &N, &M);
	for (int i = M; i <= 9; i++) {
		ans[0] = i;
		int r = 0;
		for (int j = 1; j < N; j++) {
			r += ans[j - 1] * M;
			ans[j] = r % 10;
			r /= 10;
		}
		if (ans[N - 1] > 0 && ans[N - 1] * M + r == ans[0]) {
			for (int j = N - 1; j >= 0; j--) {
				printf("%d", ans[j]);
			}
			return 0;
		}
	}
	puts("Impossible");
}
