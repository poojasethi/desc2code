char a[1500000];

int main()
{
	int p, x;
	while (scanf("%d%d",&p,&x)!=EOF)
	{
		int flag = 0;
		a[p] = 0;
		for (int j = 1; j<10; j++)
		{
			int flag1 = 0;
			a[p - 1] = j + 48;
			int c = 0, z=0;
			for (int i = p - 1; i>0; i--)
			{
				z = (a[i] - 48)*x + c;
				a[i - 1] = z % 10 + 48;
				c = z / 10;
			}
			if (((a[0] -48)*x + c) == j&&a[0] - 48 != 0) flag1 = 1;
			if (flag1)
			{
				flag = 1;
				break;
			}
		}
		if (flag)printf("%s\n", a);
		else printf("Impossible\n");
	}
	return 0;
}
