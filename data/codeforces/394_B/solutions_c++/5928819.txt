#include<stdio.h>
#include<string.h>
int s[1000005];
int main()
{
    int n,m,k,f=0,i,j;
    scanf("%d%d",&n,&m);
    for(i=m;i<=9;i++)
    {
        memset(s,0,sizeof(s));
       s[0]=i;
       for(j=0;j<n;j++)
       {
           k=s[j]*m;
           s[j+1]+=k%10;
           s[j+2]+=k/10+s[j+1]/10;
           s[j+1]%=10;
       }
       if(s[n]==i&&s[n+1]==0)
       {
           f=1;
           break;
       }

    }
    if(f==0)
    printf("Impossible\n");
    else
    {
        for(i=n-1;i>=0;i--)
        printf("%d",s[i]);
        printf("\n");
    }
    return 0;
}
