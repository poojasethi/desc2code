#include <iostream>
#include <string>
using namespace std;

int p, x;
int num[1000005];

bool test(int last)
{
	num[p - 1] = last;
	int carry = 0;
	for (int i = p - 2; i >= 0; i--)
	{
		carry += num[i+1]*x;
		num[i] = carry % 10;
		carry /= 10;
	}

	return (num[0] > 0 && carry + num[0]*x == last);
}

int main()
{
	cin >> p >> x;

	for (int last = x; last <= 9; last++)
		if (test(last))
		{
			for (int i = 0; i < p; i++)
				cout << num[i];
			cout << "\n";
			return 0;
		}

	cout << "Impossible\n";

	return 0;
}
