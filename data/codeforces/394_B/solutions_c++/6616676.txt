#include <bits/stdc++.h>

using namespace std;

const int maxN = 1000*1000 + 321;
int ans[maxN];
int main()
{
  int n, m;
  cin >> n >> m;
  for(int k = m; k <= 9; k ++)
    {
      ans[0] = k;
      int rem = 0;
      for(int i = 1; i < n ; i ++)
	{
	  rem += ans[i - 1] * m;
	  ans[i] = rem % 10;
	  rem /= 10;
	}
      if(ans[n - 1] > 0 && ans[n - 1] * m + rem == ans[0])
	{
	  for(int i = n - 1; i >= 0; i --) cout << ans[i];
	  cout << endl;
	  return 0;
	} 
    }
  cout << "Impossible" << endl;
  return 0;
}
