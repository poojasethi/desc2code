#include<stdio.h>
int a[1000000];
int main()
{
	int p,x;
	scanf("%d%d",&p,&x);
		int flag=1;
		for(int i=1;i<=9;i++)
		{
			a[0]=i;
			int mod=0;
			for(int j=1;j<p;j++)
			{
			   int t=a[j-1]*x+mod;
			   a[j]=t%10;
			   mod=t/10;
			}
			if(a[p-1]*x+mod==i&&a[p-1]!=0)
			{
				flag=0;break;
			}
		}
		if(flag)
		printf("Impossible\n");
		else
		{
		for(int i=p-1;i>=0;i--)
		printf("%d",a[i]);
		printf("\n");	
		}
		
	return 0;
}
