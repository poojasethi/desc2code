#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<math.h>


using namespace std;

#define maxn 1000000
int a[maxn];

int main()
{
	int p, x;
	int i, j;
	bool flag = 0;
	scanf("%d%d", &p, &x);
	for (i = 1; i <= 9&&!flag; i ++)
	{
		int r = 0, b = i;
		a[p] = i;
		for (j = 1; j < p; j ++)
		{
			a[j] = (r * 10 + b) / x;
			if (a[1] == 0)
			{
				b = -1;
				break;
			}
			r = (r * 10 + b) % x;
			b = a[j];
		}
		if (b == (i * x) % 10)
		{
			flag = 1;
			break;
		}
	}
	if (flag) {for (i = 1; i <= p; i ++) printf("%d", a[i]);printf("\n");}
	else
		printf("Impossible\n");
return 0;
}
