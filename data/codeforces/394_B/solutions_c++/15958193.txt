
#include<stdio.h>
int a[1000005],b[1000005];
int main()
{

    int p,x,i,j;
    int wei;
    int flag=0;
    scanf("%d%d",&p,&x);
    for(i=x;i<10;i++)
    {
        wei=0;
        a[p]=i;b[1]=i;
        for(j=0;j<p;j++)
        {
            a[j+1]=(b[j+1]+wei*10)/x;
            wei=(wei*10+b[j+1])%x;
            b[j+2]=a[j+1];

        }
        int h=(i*x)%10;

        if(h==b[p])
        {
            for(j=1;j<=p;j++)
                printf("%d",a[j]);
            printf("\n");
            flag=1;
            break;
        }
    }
    if(!flag) printf("Impossible");
    return 0;
}
