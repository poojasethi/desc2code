#include<bits/stdc++.h>
using namespace std;
int n,p,a[1000005],i;
bool fills(int n,int p,int a[])
{
    int c=0;
    for(int i=0;i<n;i++)
    {
        a[i+1]=(a[i]*p+c)%10;
        c=(a[i]*p+c)/10;
    }
    if(c==0&&a[n]==a[0]&&a[n-1]!=0)
    return true;
    return false;
}
int main()
{
    scanf("%d %d",&n,&p);
    for(a[0]=0;a[0]<=9;a[0]++)
    {
        if(fills(n,p,a))
        break;
    }
    if(a[0]==10)
    printf("Impossible");
    else
    for(i=n-1;i>=0;i--)
        printf("%d",a[i]);
    printf("\n");
    return 0;
}
