#include<stdio.h>
#include<string.h>
int a[1000005];
int main()
{
	int i,j,n,m,d,fuck,shit;
	scanf("%d%d",&n,&m);
	//memset(a,0,sizeof(a));
	d=0;
	for(i=1;i<=9;i++)
	{
		a[1]=i;shit=0;
		for(j=1;j<=n;j++)
		{
			fuck=a[j]*m;
			a[j+1]=(fuck+shit)%10;
			shit=(fuck+shit)/10;
		}
		if(shit)
			continue;
		if(a[1]==a[n+1]&&a[n])
		{
			d=1;
			break;
		}
	}
	if(d)
		for(i=n;i>=1;i--)
			printf("%d",a[i]);
	else
		printf("Impossible");
	printf("\n");
	return 0;
}
