#include <iostream>
#include <cstdlib>
#include <cstring>
#include <string>
#include <vector>
#include <utility>
#include <algorithm>
#include <cmath>

using namespace std;

int p,x;
int a[1000010];

int main() {
	cin >> p >> x;
	if (p == 1) {
		cout << (x==1 ? "1" : "Impossible");
		return 0;
	}
	for (a[1] = 1; a[1] < 10; a[1]++) {
		for (int i = 1; i <= p; i++) {
			a[i+1] = x*(a[i]%10) + a[i]/10;
		}
		if (a[p+1] == a[1] && a[p]%10 != 0) {
			for (int i = p; i > 0; i--) cout << a[i]%10;
			return 0;
		}
	}
	cout << "Impossible";
}
