#include<iostream>

using namespace std;
const int MAXN = (int)1e6 + 100;
int a[MAXN];
int p , x;

bool check (int ap)
{
  fill (a , a+MAXN , 0);
  a[p] = ap;
  int bef = 0;
  for (int i = p-1;i >= 0; i--)
    {
      bef /= 10;
      bef += (x * a[i+1]);
      a[i] = bef % 10;
    }
  bef /= 10;
  if (a[1] != 0 && a[0] == ap && bef == 0)
    return 1;
  return false;
}

int main ()
{
  cin >> p >> x;
  for (int i = 1;i <= 9; i++)
    {
      if (check (i))
	{
	  for (int i = 1;i <= p; i++)
	    cout << a[i];
	  cout << endl;
	  return 0;
	}
    }
  cout << "Impossible" << endl;
  return 0;
}
