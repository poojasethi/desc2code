#include <bits/stdc++.h>

using namespace std;

int p, x, a[1000009];

int main()
{
	cin >> p >> x;
	
	for (a[p] = 1; a[p] <= 9; a[p]++) {
		int  san = 0;
		for (int i = p - 1; i > 0; i--) {
			a[i] = (a[i + 1] * x) + san;
			san = a[i] / 10;
			a[i] = a[i] % 10;
		}
		
		a[0] = (a[1] * x) + san;
		
		if (a[0] == a[p] && a[1] != 0) {
			for (int i = 1; i <= p; i++)
				cout << a[i];
			return 0;
		}
	}
	cout << "Impossible";
	return 0;
}
