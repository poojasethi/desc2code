#include <iostream>
using namespace std;
int a[1000002],b[1000002];
int main()
{
    int p,x,ans,f=0;
    cin>>p>>x;
    for(int i=x;i<=9;i++)
    {
        a[p]=i;
        b[1]=i;
        ans=0;
        for(int j=1;j<=p;j++)
        {
            a[j]=(ans*10+b[j])/x;
            ans=(ans*10+b[j])%x;
            b[j+1]=a[j];
        }
        if(b[p]==(i*x)%10)
        {
            for(int k=1;k<=p;k++)
            cout<<a[k];
            cout<<endl;
            f=1;
            break;
        }
    }
    if(f==0)
    cout<<"Impossible"<<endl;
    return 0;
}
