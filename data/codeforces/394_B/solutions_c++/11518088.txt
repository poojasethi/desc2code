#include <cstdio>
int n, t, a[1001000];
inline int Judge(int LR){
	a[1]=LR*t; int x=a[1]/10; a[1]%=10;
	for(int i=2; i<=n; i++)
	{
		a[i]=a[i-1]*t+x;
		x=a[i]/10;
		a[i]%=10;
	}
	if(x || !a[n] || !a[n-1] || a[n]^LR) return 0;
	return 1;
}
inline void Print(){
	for(int i=n-1; i; i--)
		printf("%d", a[i]);
	printf("%d\n", a[n]);
}
int main(){
	scanf("%d%d", &n, &t); if(n==1 && t==1) return puts("1"), 0;
	for(int i=1; i<=9; i++)
		if(Judge(i)) return Print(), 0;
	puts("Impossible");
}
