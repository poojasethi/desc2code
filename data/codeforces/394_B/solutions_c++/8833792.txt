int main(){

	int c[n];
	bool flag=true;
	for (int i = 1; i < 10; i++) {
		c[n-1]=i;
		int curr=c[n-1],carry=0;
		for (int j = n-2; j >=0; j--) {
			c[j]=(x*curr+carry)%10;
			carry=(x*curr+carry)/10;
			curr=c[j];
		}
		if(x*curr+carry==i && c[0]!=0){
			for (int j = 0; j < n; j++) cout<<c[j];
				cout<<endl;
			flag=false;
			break; 
		}
	}
	if(flag) cout<<"Impossible\n";
}
