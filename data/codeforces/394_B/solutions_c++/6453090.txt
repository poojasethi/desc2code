#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cstring>

const int Len = 1001000;

using namespace std;

int d[Len];

int main(void){
	int n, m;
	scanf("%d%d", &n, &m);
	if (n == 1){
		if (m == 1)
			printf("1\n");
		else
			printf("Impossible\n");
		return 0;
	}
	int b;
	for (d[n] = 1; d[n] <= 9; d[n]++){
			b = 0;
			for (int i = n ; i > 1; i--){
				d[i - 1] = (d[i] * m + b) % 10;
				b = (d[i] * m + b) / 10;
			}
			if ((d[1] * m + b) % 10 != d[n] || d[1] == 0 || (d[1] * m + b) / 10 > 0)
				continue;
			for (int i = 1; i <= n; i++)
				printf("%d", d[i]);
			putchar('\n');
			return 0;
	}
	printf("Impossible\n");
	
	return 0;
}
