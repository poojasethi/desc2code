#include <iostream>
using namespace std;
int a[1000000],b[1000000];
int main()
{
	int n,k,s,f=1;
	cin>>n>>k;
	for(int i=1;i<=9;i++){
		a[n]=i;
		int g=0;
		for(int j=n-1;j>=1;j--){
			s=b[j+1]+a[j+1]*k;
			g=s/10;
			b[j+1]=s%10;
			a[j]=b[j+1];
			b[j]=g;
		}
		b[1]=b[1]+(a[1]*k);
		if (b[1]==a[n]&&(a[1]!=0)){
			f=0;
			for(int j=1;j<=n;j++){
				cout<<a[j];
			}
			return 0;
		}
		for(int j=1;j<=n;j++){
			a[j]=0;
			b[j]=0;
		}
	}
	if (f==1) cout<<"Impossible";
	return 0;
}
