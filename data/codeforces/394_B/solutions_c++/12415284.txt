#include <bits/stdc++.h>

using namespace std;


int main(){
	int p,x;
	cin>>p>>x;
	vector<int> v(p+1);
	int temp,carry=0;
	bool found=false;
	for(int j=1;j<=9;j++){
		carry=0;
		v[p] = j;
		for (int i = p; i>=1; i--){
			temp = v[i]*x + carry;
			v[i-1] = temp%10;
			carry=temp/10;
		}
		if(carry == 0 && v[0] == v[p] && v[1] !=0){
			found=true;
			break;
		}
		
	}
	if(found){
		for(int k = 1; k<p+1;k++){
			cout<<v[k];
		}
		cout<<endl;
	}
	else{
		cout<<"Impossible"<<endl;
	}


}
