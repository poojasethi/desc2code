#include <iostream>
using namespace std;
#define N 1000001
int p, x, n[N];
bool fill() {
	int c = 0;
	for (int i = 0; i < p; i++) {
		n[i+1] = (n[i]*x+c)%10;
		c = (n[i]*x+c)/10;
	}
	return c == 0 && n[p] == n[0] && n[p-1] != 0;
}
int main() {
	cin >> p >> x;
	for (n[0] = 1; n[0] < 10; n[0]++)
		if (fill())
			break;
	if (n[0] == 10) {
		cout << "Impossible";
	} else {
		for (int i = p-1; i >= 0; i--)
			cout << (char)('0' + n[i]);
	}
}
