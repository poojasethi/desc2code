int pow(int a,int b){
    int ans=1;
    while(b){
        if(b&1)ans=1LL*ans*a%mod;
        b>>=1;a=1LL*a*a%mod;
    }
    return ans;
}

int fac[N],inv[N];
ll C(int n,int m){
    if(n<m)return 0;
    if(m==0||n==m)return 1;
    return 1LL*fac[n]*inv[m]%mod*inv[n-m]%mod;
}
const int M=1000005;
int phi[M],cnt[M];
vi d[M];
int main(){
    //freopen("2.txt","r",stdin);
    int n,m,i,j,k,K;
    for(i=1;i<M;i++)phi[i]=i,d[i].pb(i);
    for(i=1;i<M;i++){
        for(j=i+i;j<M;j+=i)d[j].pb(i),phi[j]-=phi[i];
    }
    fac[0]=1;
    for(i=1;i<N;i++)fac[i]=fac[i-1]*(ll)i%mod;
    inv[N-1]=pow(fac[N-1],mod-2);
    for(i=N-2;i>=0;i--)inv[i]=inv[i+1]*(ll)(i+1)%mod;
    scanf("%d%d%d",&n,&K,&m);
    ll ans=0;
    for(i=0;i<n+m;i++){
        scanf("%d",&k);
        FOREACH(it,d[k]){
            ans+=C(cnt[*it],K-1)*phi[*it];
            ans%=mod;
            ++cnt[*it];
        }
        if(i>=n){
            if(ans<0)ans+=mod;
            printf("%d\n",(int)ans);
        }
    }
    return 0;
}
