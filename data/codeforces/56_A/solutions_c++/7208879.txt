#include <iostream>
using namespace std;
int main()
{
	int n, i, x, ans=0;
	string s, wine[11]={"ABSINTH", "BEER", "BRANDY", "CHAMPAGNE", "GIN", "RUM", "SAKE", "TEQUILA", "VODKA", "WHISKEY", "WINE"};
	cin>>n;
	while (n--) {
		cin>>s;
		for (x=i=0; i<s.size(); i++) {
			if (s[i]>'9') break;
			else x=x*10+s[i]-'0';
		}
		if (i==s.size() && x<18) {
			ans++; continue;
		} 
		for (i=0; i<11; i++) if (wine[i]==s) {
			ans++; break;
		}
	}
	cout<<ans<<endl;
	return 0;
}
