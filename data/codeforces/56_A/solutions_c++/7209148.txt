#include<iostream>
#include<cstdio>
using namespace std;
int main(){
	int n, i, b=0, j, s;
	string x;
	cin >> n;
	for(i=0; i<n; i++){
		cin >> x;
		if((x=="ABSINTH")||(x=="BEER")||(x=="BRANDY")||(x=="CHAMPAGNE")||(x=="GIN")||(x=="RUM")||(x=="SAKE")||(x=="TEQUILA")||(x=="VODKA")||(x=="WHISKEY")||(x=="WINE")){
			b++;
		}
		else if(x[0]<='9'){
			for(s=0,j=0; j<x.size();j++){
				s=s*10+x[j]-'0';
			}
			if(s<18){
				b++;
			}
		}
	}
	cout << b << endl;
	return 0;
}
