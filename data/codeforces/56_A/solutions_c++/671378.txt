#include <cstdio>
#include <cstring>
char a[][10] = {"ABSINTH", "BEER", "BRANDY", "CHAMPAGNE", "GIN", "RUM", "SAKE", "TEQUILA", "VODKA", "WHISKEY", "WINE"};
char s[111];
int main() {
    int n, r = 0, c; scanf("%d", &n);
    while (n--) {
        scanf("%s", s);
        if ('0' <= s[0] && s[0] <= '9') { sscanf(s, "%d", &c); if (c < 18) r++; }
        else { for (int i = c = 0;i < 11 && !c;i++) if (strcmp(a[i], s) == 0) c = 1; r += c; }
    }
    printf("%d\n", r);
}
