#include <iostream>
#include<string>
using namespace std;

int main()
{int n,su=0;cin>>n;
for(int i=0;i<n;i++)
{string s;
    cin>>s;
    if(isdigit(s[0]))
    {int j=0,r=0;
    while(j<s.size())
    {
    r=r*10+(s[j]-'0');j++;
    }
    if(r<18) {su++;}}
    else
    {
     if(s=="ABSINTH"||s=="BEER"||s=="BRANDY"||s=="CHAMPAGNE"||s=="GIN"||s=="RUM"||s=="SAKE"||s=="TEQUILA"||s=="VODKA"||s=="WHISKEY"||s=="WINE") su++;
    }

}cout<<su;
    return 0;
}
