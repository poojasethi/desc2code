#include <iostream>
using namespace std;
int main()
{
string s;
int i,n,br=0;
cin>>n;
 for(i=0;i<n;i++)
 {
  cin>>s;
   if(s=="ABSINTH" || s=="BEER" || s=="BRANDY" || s=="CHAMPAGNE" || s=="GIN" || s=="RUM" || s=="SAKE" || s=="TEQUILA" || s=="VODKA" || s=="WHISKEY" || s=="WINE")
    br++;
   if(s.length()==1 && s[0]>='0' && s[0]<='9')
    br++;
   if(s.length()==2 && s[0]=='1' && s[1]>='0' && s[1]<'8')
    br++;
 }
cout<<br<<endl;
}
