#include <iostream>
#include <string>
#include <cstdlib>

using namespace std;
main(){
	int n,i,j,res=0,t;
	string s,drinks [11] ={ "ABSINTH", "BEER", "BRANDY", "CHAMPAGNE", "GIN", "RUM", "SAKE", "TEQUILA", "VODKA", "WHISKEY", "WINE"};
	cin >> n;
	for(i=0;i<n;i++){
		cin >> s;
		t = 0;
		for(j=0;j<11;j++)
		 if(s==drinks[j])res++,t=1;
		if((atoi(s.data())!=0 || s=="0") &&!t && atoi(s.data()) < 18)res++;
	 }

	 cout << res << endl;
	//system("pause");
}
