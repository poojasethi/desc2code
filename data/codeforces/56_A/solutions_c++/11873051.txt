#include <iostream>
#include<cstdlib>
using namespace std;

int main()
{
    int n,sum=0; string k; cin>>n;

    for(int i=0;i<n;i++)
    {
        cin>>k;
        if(k=="ABSINTH" ||k=="BEER" ||k=="BRANDY" ||k=="CHAMPAGNE" ||k=="GIN" ||k=="RUM" ||
           k=="SAKE" ||k=="TEQUILA" ||k=="VODKA" ||k=="WHISKEY" ||k=="WINE")
            sum++;
            else if(isdigit(k[0])) if(atoi(k.c_str())<18) sum++;
    }

    cout<<sum;
        return 0;
}
