#include<stdio.h>
#include<iostream>
using namespace std;
int n,o;
string s;
int main(){
	scanf("%d\n",&n);
	o=0;
	for(int i=0;i<n;++i){
		cin >> s;
		int t=-1;
		sscanf(s.c_str(),"%d",&t);
		if(t!=-1){
			if(t<18) o++;
		}
		else if(	s=="ABSINTH" ||
			s=="BEER" ||
			s=="BRANDY" ||
			s=="CHAMPAGNE" ||
			s=="GIN" ||
			s=="RUM" ||
			s=="SAKE" ||
			s=="TEQUILA" ||
			s=="VODKA" ||
			s=="WHISKEY" ||
			s=="WINE")
			++o;
	}
	printf("%d\n",o);
	return 0;
}
