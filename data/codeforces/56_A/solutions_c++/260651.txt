#include <iostream>
#include <string>
#include <algorithm>
#include <map>

using namespace std;


string alco[12] = {"ABSINTH", "BEER", "BRANDY", "CHAMPAGNE", "GIN", "RUM", "SAKE", "TEQUILA", "VODKA", "WHISKEY", "WINE"};
int main () {
	int n;
	cin >> n;
	int total = 0;
	while(n--) {
		string s;
		cin >> s;
		if(binary_search(alco,alco+11,s) or (isdigit(s[0]) and atoi(s.c_str()) < 18)) total++;
	}
	cout << total << endl;
	return 0;
}
