#include <bits/stdc++.h>
using namespace std;

int main()
{
  
  int t;
  cin>>t;
  
  int c=0;
  
  while(t--){
   string x;
   
   cin>>x;
   
   if(isdigit(x[0])){
    int e=atoi(x.c_str());
    if(e<18)c++;
   }else{
       
       if(x=="WINE"||x=="WHISKEY"||x=="VODKA"||x=="TEQUILA"||x=="SAKE"||x=="RUM"||x=="GIN"||x=="CHAMPAGNE"||x=="BRANDY"||x=="BEER"||x=="ABSINTH")
        c++;
    
    
   }
   
      
  }
  
  cout<<c<<endl;
}
