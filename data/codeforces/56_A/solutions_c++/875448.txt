#include<iostream>
using namespace std;
int main()
{
    int n,c=0;
    string s;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    cin>>s;
    if(s=="ABSINTH"||s=="BEER"||s=="BRANDY"||s=="CHAMPAGNE"||s=="GIN"||s=="RUM"||s=="SAKE"||s=="TEQUILA"||s=="VODKA"||s=="WHISKEY"||s=="WINE")
    c++;
    else if(s.length()==1&&'0'<=s[0]&&s[0]<='9')
    c++;
    else if(s.length()==2&&s[0]=='1'&&'0'<=s[1]&&s[1]<='7')
    c++;
}
    cout<<c;
    return 0;
}
