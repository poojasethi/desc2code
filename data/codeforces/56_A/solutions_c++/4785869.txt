#include <iostream>

using namespace std;

string a[11] = {"ABSINTH", "BEER", "BRANDY", "CHAMPAGNE", "GIN", "RUM", "SAKE", "TEQUILA", "VODKA", "WHISKEY", "WINE"}, s;

int main()
{
	int n, cnt = 0;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> s;
		if (s[0] - '0' < 10)
		{
			if (s.size() == 1 || (s.size() == 2 && s[0] == '1' && s[1] < '8'))
				cnt++;
		}
		else
		{
			for (int j = 0; j < 11; j++)
				if (a[j] == s)
					cnt++;
		}
	}
	cout << cnt << endl;
	return 0;
}
