#include<iostream>
using namespace std;
int main()
{
int n,i,j,k=0,p;
string s,a[20]={"ABSINTH", "BEER", "BRANDY", "CHAMPAGNE", "GIN", "RUM", "SAKE", "TEQUILA", "VODKA", "WHISKEY", "WINE"};
cin>>n;
for(i=1;i<=n;i++)
{
cin>>s;
if(s[0]>='0' && s[0]<='9')
{
p=0;
for(j=0;j<s.length();j++)
{
p*=10;
p+=s[j]-'0';
}
if(p<18)
{
k++;
}
}
else
{
for(j=0;j<11;j++)
{
if(s==a[j])
{
k++;
break;
}
}
}
}
cout<<k;
return 0;
}
