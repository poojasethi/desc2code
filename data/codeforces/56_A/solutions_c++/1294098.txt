#include <cstdio>
#include <cstring>

using namespace std;

char s[110];
char alc[][12]={"ABSINTH", "BEER", "BRANDY", "CHAMPAGNE", "GIN", "RUM", "SAKE", "TEQUILA", "VODKA", "WHISKEY", "WINE"};

int main() {
	int n;
	scanf("%d\n",&n);
	int r=0;
	for(int i=0;i<n;i++) {
		scanf("%s",s);
		if (s[0]>'9') {
			for(int j=0;j<11;j++) if (strcmp(s,alc[j])==0) { r++; break; }
		} else {
			int t;
			sscanf(s,"%d",&t);
			if (t < 18) r++;
		}
	}
	printf("%d\n",r);
	return 0;
}
