#include <iostream>
#include <string>
#include <cstdlib>

using namespace std;
string A[12] ={"ABSINTH", "BEER", "BRANDY", "CHAMPAGNE", "GIN", "RUM", "SAKE", "TEQUILA", "VODKA", "WHISKEY", "WINE"};
int main()
{
	int n;
	string s;
	cin >> n;
	int t = 0;
	int num;
	while(n--){
		cin >> s;
		if(s[0]>57){
			for(int i=0;i<11;i++){
				if(A[i] == s)
					t++;
			}
		}
		else{
			num = atoi(s.c_str());
			if(num < 18)
				t++;
		}
	}
	cout << t <<"\n";
	return 0;
}
