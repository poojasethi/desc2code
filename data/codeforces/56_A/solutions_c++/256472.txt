#include<iostream>
#include<sstream>
#include<string>
using namespace std;
int main()
{
	int n,x,k = 0;
	cin>>n;
	string s,a[] = {"ABSINTH", "BEER", "BRANDY", "CHAMPAGNE", "GIN", "RUM", "SAKE", "TEQUILA", "VODKA", "WHISKEY", "WINE"};
	for(int i = 0; i < n; i++)
	{
		cin>>s;
		stringstream ss(s);
		if (ss>>x)
		{
			if (x < 18) k++;
		} else
		{
			for(int i = 0; i < 11; i++)
				if (s == a[i])
				{
					k++;
					break;
				}
		}
	}
	cout<<k;
	return 0;
}
