#include <iostream>
#include <stdlib.h>
using namespace std;

int main()
{
    int x,t,c=0;
    string s;
    cin >> x;

    for (int i=0; i<x; i++){
        cin >> s;
        if (s=="ABSINTH"|| s=="BEER"|| s=="BRANDY"|| s=="CHAMPAGNE"|| s=="GIN"|| s=="RUM"|| s=="SAKE"|| s=="TEQUILA"|| s=="VODKA"|| s=="WHISKEY" || s=="WINE")
            c++;
        if(int(s[0]<65 || s[0]>90)){
                t=atoi(s.c_str());
                if(t<18)c++;}
    }
    cout << c;
}
