#include<iostream>
#include <cmath>
using namespace std;
int main(){
int n, c=0; cin>>n;
string s;
for (int i=0;i<n;i++) {cin>>s;
	if (s[0]<'A')
		if (s.size()==2&&s[1]<'8'&&s[0]=='1'){ c++; }
		else if (s.size()==1) {c++;}
	if (s=="ABSINTH"||s=="BEER"||s=="BRANDY"||s=="CHAMPAGNE"||s=="GIN"||s=="RUM"||s=="SAKE"||s=="TEQUILA"||s=="VODKA"||s=="WHISKEY"||s=="WINE") {c++;}
	}

	cout<<c;}
