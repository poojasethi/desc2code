#include <iostream>

using namespace std;

int main()
{int n,ans=0;
string s;
cin>>n;
for(int i=0;i<n;i++)
{
cin>>s;
if (s=="ABSINTH" || s=="BEER" || s=="BRANDY" || s=="CHAMPAGNE" || s=="GIN" || s=="RUM" || s=="SAKE" || s=="TEQUILA" || s=="VODKA" || s=="WHISKEY" || s=="WINE")
ans++;
if(s[0]<='9' && s[0]>='0')
{
    int sum=0;
    for(int i=0; i<s.size(); i++)
{
    sum=(sum*10)+(s[i]-'0');
}
    if(sum<18)
    ans++;
}
}
cout<<ans;
}
