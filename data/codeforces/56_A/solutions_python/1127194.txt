n = input()
L = ['ABSINTH', 'BEER', 'BRANDY', 'CHAMPAGNE', 'GIN', 'RUM', 'SAKE', 'TEQUILA', 'VODKA', 'WHISKEY', 'WINE']
ans = 0
for i in xrange(n):
    s = raw_input()
    if s.isdigit():
        s = int(s)
        if s < 18:
            ans += 1
    else:
        if s in L:
            ans += 1
print ans
