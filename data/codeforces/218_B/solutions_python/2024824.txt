n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
b = a[:]
maxr = 0
minr = 0
for i in range(n):
    maxi = 0
    for j in range(0, m):
	if b[j] > 0:
	    mini = j
	    break
    for j in range(0, m):
	if a[j] > a[maxi]:
	    maxi = j
	if b[j] > 0 and b[j] < b[mini]:
	    mini = j
    maxr += a[maxi]
    minr += b[mini]
    a[maxi] -= 1
    b[mini] -= 1
print maxr, minr
