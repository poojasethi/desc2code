n, m = map(int, raw_input().split())
a = map(int, raw_input().split())

a.sort()
b=[ _ for _ in a]
 
min_ , max_ = 0, 0

index = 0 
for i in range(n):
    min_ += a[index]
    a[index] -= 1
    if a[index] == 0:
        index += 1
for i in range(n):
    b.sort()
    b.reverse()
    max_ += b[0]
    b[0] -= 1

print max_, min_
