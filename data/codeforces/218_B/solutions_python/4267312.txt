def main():
	n,p=map(int,raw_input().split(" "))
	ar=map(int,raw_input().split(" "))
	maxp=0
	minp=0
	armax=[x for x in ar]
	armin=[x for x in ar]
	for i in range(n):
		mv=max(armax)
		maxp+=mv
		ind=armax.index(mv)
		del armax[ind:ind+1]
		armax.append(mv-1)
	for i in range(n):
		mv=min(armin)
		minp+=mv
		ind=armin.index(mv)
		del armin[ind:ind+1]
		if mv-1>0:
			armin.append(mv-1)
	print str(maxp)+" "+str(minp)
main()
