import heapq
n,m=map(int,raw_input().split())
h1=map(int,raw_input().split())
h2=[-v for v in h1]
heapq.heapify(h1)
heapq.heapify(h2)
minRes=0
for i in xrange(n):
    a=heapq.heappop(h1)
    minRes+=a
    a-=1
    if a>0:
        heapq.heappush(h1,a)
maxRes=0
for i in xrange(n):
    a=heapq.heappop(h2)
    maxRes-=a
    a+=1
    if a<0:
        heapq.heappush(h2,a)
print(str(maxRes)+' '+str(minRes))
