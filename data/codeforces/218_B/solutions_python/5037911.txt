n, m = map(int, raw_input().split())
n, m = int(n), int(m)

l = map(int, ((raw_input().split())))
l.sort()
k = list(l)
k.reverse()

maxim, minim = 0, 0

p = 0
for i in xrange(n):
    if p != m - 1 and k[p] < k[p+1]:
        p += 1
    elif p > 0 and k[p] > k[p-1]:
        pass
    else:
        p = 0
    maxim += k[p]
    k[p] -= 1
        
p = 0
for i in xrange(n):
    if l[p] == 0:
        p += 1
    minim += l[p]
    l[p] -= 1
        
print maxim, minim
