import copy, math
n, m = map(int, raw_input().split())
z = map(int, raw_input().split())

zei = copy.deepcopy(z)
z.sort(reverse=True)
zei.sort()
ans1, ans2 = 0, 0
for x in xrange(n):
    ans1 += z[0]
    z[0] -= 1
    if zei[0]>0:
        ans2 += zei[0]
    zei[0] -= 1
    if zei[0] == 0:
        zei.remove(0)
    zei.sort()
    z.sort(reverse=True)
print ans1, ans2
