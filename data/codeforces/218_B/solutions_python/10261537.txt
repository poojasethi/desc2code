from heapq import *


read = lambda: map(int, raw_input().split())

n, m = read()
a = read()

min_heap = [x for x in a]
max_heap = [-x for x in a]
heapify(min_heap)
heapify(max_heap)

min_sum, max_sum = 0, 0
for _ in xrange(n):
    min_val = heappop(min_heap)
    max_val = -heappop(max_heap)
    min_sum += min_val
    max_sum += max_val
    if min_val - 1 > 0: heappush(min_heap, min_val - 1)
    if max_val - 1 > 0: heappush(max_heap, -max_val + 1)

print max_sum, min_sum
