n, m = map(int, raw_input().split())
a = map(int, raw_input().split())

min_ans, max_ans = 0, 0
b = a[:]
for i in range(n):
    min_idx = min(xrange(m), key = lambda x, b=b: b[x] if b[x] != 0 else float('inf'))
    min_ans += b[min_idx]
    b[min_idx] -= 1

b = a[:]
for i in range(n):
    max_idx = max(xrange(m), key = b.__getitem__)
    max_ans += b[max_idx]
    b[max_idx] -= 1

print("%i %i" % (max_ans, min_ans))
