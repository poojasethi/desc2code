ans1, ans2 = 0, 0

def Max(x):
	global ans1
	ri = 0
	for i in xrange(len(x)):
		if x[i] > x[ri]: ri = i
	ans1 += x[ri]
	x[ri] -= 1
	
def Min(x):
	global ans2
	ri = -1
	for i in xrange(len(x)):
		if (ri == -1 or x[i] < x[ri]) and x[i] > 0: ri = i
	ans2 += x[ri]
	x[ri] -= 1

n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
b = a[:]

for i in xrange(n):
	Max(a)
	Min(b)

print ans1, ans2
