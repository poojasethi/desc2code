#!/usr/bin/env python
import Queue;

n,m = map(int, raw_input().strip().split())
a = map(int, raw_input().strip().split())

t = n
mincost=0;
a.sort()
q = Queue.PriorityQueue()
for x in a:
    q.put(x)
for i in xrange(n):
    num = q.get()
    mincost += num
    if num>1:
        q.put(num-1)

maxcost = 0;

q = Queue.PriorityQueue()
for x in a:
    q.put(-x)
for i in xrange(n):
    num = -q.get()
    maxcost += num
    if num>1:
        q.put(-num+1)
print maxcost, mincost
