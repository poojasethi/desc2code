def main():
	n,m = map(int,raw_input().split())
	a = map(int,raw_input().split())
	b = [a[i] for i in xrange(m)]
	b.append(1001)
	mxm,mim=0,0
	for i in range(n):
		cmax = 0
		for j in range(m):
			if a[j] > a[cmax]:
				cmax = j
		mxm+=a[cmax]
		a[cmax]-=1
	for i in range(n):
		cmin = m
		for j in range(m):
			if b[j]> 0 and b[j]< b[cmin]:
				cmin = j
		mim+=b[cmin]
		b[cmin]-=1 
	print mxm,mim
	return 0

if __name__ == '__main__':
	main()
