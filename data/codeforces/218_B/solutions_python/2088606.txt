n, m = map(int, raw_input().split())
b = map(int, raw_input().split())

a = list(b)
_max = 0
for i in xrange(n,0,-1):
    _max += max(a)
    a[a.index(max(a))] -= 1
    
_min = 0
a = b
for i in xrange(n,0,-1):
    _min += min(a)
    c=a.index(min(a))
    a[c] -= 1
    if a[c]==0 : del a[c]

print "%i" % (_max),
print "%i" % (_min)
