n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
tmp = a[:]
smax = 0
for _ in range(n):
  amax = max(tmp)
  smax += amax
  idx = tmp.index(amax)
  tmp[idx] -= 1
  if tmp[idx] == 0:
    del tmp[idx]
smin = 0
for _ in range(n):
  amin = min(a)
  smin += amin
  idx = a.index(amin)
  a[idx] -= 1
  if a[idx] == 0:
    del a[idx]
print smax, smin
