n,m = map(int,raw_input().split())

ls = map(int,raw_input().split())

ls.sort()

mn = 0
mx = 0
count=0
for i in range(m):
    t = ls[i]
    while t>0 and count<n:
        mn += t
        count+=1
        t-=1

ls.sort(cmp=None, key=None, reverse=True)
count=0
while count<n:
    j = max(ls)
    i = ls.index(j)
    if j>0:
        mx += j
        count+=1
    ls[i]-=1

print mx,mn
