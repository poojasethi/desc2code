import copy

n , m = map(int , raw_input().split())
empty = list(map(int , raw_input().split()))
max = 0
min = 0
n1 = n
empty.sort()

for i in empty:
    if i < n1:
        min += (i + 1) * i / 2
        n1 -= i
    else:
        min += (2 * i - n1 + 1) * n1 / 2
        break
    
min = int(min)
n1 = n

while n1 > 0:
    max += empty[m - 1]
    empty[m - 1] -= 1
    empty.sort()
    n1 -= 1

print max , min
