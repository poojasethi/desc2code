n, m = map(int, raw_input().split())
s_1 = map(int, raw_input().split())
s_2 = s_1[::]
mini = 0
maxi = 0
for i in xrange(n):
    s_1.sort()
    s_2.sort()
    mini += s_1[0]
    maxi += s_2[-1]
    s_1[0] -= 1
    s_2[-1] -= 1
    if s_1[0] == 0:
        s_1.pop(0)
    if s_2[-1] == 0:
        s_2.pop(-1)
print maxi, mini
