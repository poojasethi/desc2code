import sys
import heapq

n, m = map(int, raw_input().split())

min_heap = map(int, raw_input().split())
max_heap = [-x for x in min_heap]
heapq.heapify(min_heap)
heapq.heapify(max_heap)

min_cost = 0
max_cost = 0


for i in range(n):
	cur_min = heapq.heappop(min_heap)
	min_cost += cur_min
	if cur_min-1>0:
		heapq.heappush(min_heap, cur_min-1)
	
	cur_max = heapq.heappop(max_heap)
	max_cost += -cur_max
	if cur_max+1<0:
		heapq.heappush(max_heap, cur_max+1)

print max_cost, min_cost
