#!/usr/bin/env python
from heapq import *
N, K = map(int, raw_input().split())
seats = map(int, raw_input().split())
heap = [i for i in seats]
heap2 = [-i for i in seats]
heapify(heap)
heapify(heap2)
mmin, mmax = 0, 0
for i in xrange(N):
    if heap:
        x = heappop(heap)
        if x > 1:
            heappush(heap, x - 1)
        mmin += x
    if heap2:
        x = heappop(heap2)
        if x < -1:
            heappush(heap2, x + 1)
        mmax += -x
print("%s %s" % (mmax, mmin))
