n, m = map(int, raw_input().strip().split())
a = map(int, raw_input().strip().split())
aa = sorted(a, reverse = True)


maxa = 0
mina = 0
for i in range(n):
    maxa = maxa + max(a)
    a[a.index(max(a))] = a[a.index(max(a))]-1

for i in range(n):
    if min(aa) > 0:
        mina = mina + min(aa)
    else:
        aa.remove(0)
        mina = mina + min(aa)
    aa[aa.index(min(aa))] = aa[aa.index(min(aa))]-1
