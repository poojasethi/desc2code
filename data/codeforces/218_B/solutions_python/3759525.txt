n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
b = list(a)
max_num = 0
min_num = 0
for i in xrange(n):
    a.sort(reverse = True)
    max_num += a[0]
    a[0] -= 1
for i in xrange(n):
    b.sort()
    if b[0] == 0:
        b.remove(0)
    min_num += b[0]
    b[0] -= 1
print max_num, min_num
