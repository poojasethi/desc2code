import heapq
n,m = map(int,raw_input().split())
h1 = map(int,raw_input().split())
h2 = [-v for v in h1]
heapq.heapify(h1)
heapq.heapify(h2)
amin = 0
amax = 0
for i in range(n):
    a = heapq.heappop(h1)
    amin+=a
    a-=1
    if a>0:
        heapq.heappush(h1,a)
for i in range(n):
    a = heapq.heappop(h2)
    amax-=a
    a+=1
    if a<0:
        heapq.heappush(h2,a)
print amax,amin
