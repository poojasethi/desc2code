n,m=map(int,raw_input().split())
a=[int(i) for i in raw_input().split()]
a=sorted(a)
b=[a[i] for i in reversed(range(m))]
s,l=0,0
i,j=0,0
while i<n:
    s+=a[j]
    a[j]-=1
    if a[j]==0:
        j+=1
    i+=1
i=0
while i<n:
    pos=0
    for j in range(m):
        if b[j]>b[pos]:
            pos=j
    l+=b[pos]
    b[pos]-=1
    i+=1
print l,s
