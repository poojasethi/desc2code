

import heapq


def main():
	n,m = [int(i) for i in raw_input().split()]
	h = [int(i) for i in raw_input().split()]
	h2 = [-i for i in h]
	mincount = 0
	maxcount = 0
	heapq.heapify(h)
	heapq.heapify(h2)
	for i in xrange(n):
		x = heapq.heappop(h)
		mincount+=x
		if (x!=1):
			heapq.heappush(h,x-1)
	for i in xrange(n):
		x = heapq.heappop(h2)
		maxcount-=x
		heapq.heappush(h2,x+1)
	print maxcount,mincount



main()
