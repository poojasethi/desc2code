input = lambda: map(int, raw_input().split())
n, m = input()
free_places = input()
free_places.sort()
copy = free_places[::]
maximum = minimum = 0
while n > 0:
    if free_places[0] == 0:
        free_places.pop(0)

    minimum += free_places[0]
    maximum += copy[-1]

    free_places[0] -= 1
    copy[-1] -= 1

    copy.sort()
    n -= 1
print('{0} {1}'.format(maximum, minimum))
