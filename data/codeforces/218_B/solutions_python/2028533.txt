import copy
a = [int(i) for i in raw_input().split(' ')]
d = [int(i) for i in raw_input().split(' ')]
data = copy.deepcopy(d)
mi = 0
ma = 0
for i in range(a[0]):
    index = data.index(max(data))
    ma += data[index]
    data[index] -=1
    
data = copy.deepcopy(d)
for i in range(a[0]):
    index = data.index(min(data))
    mi += data[index]
    if data[index] >=1 :
        data[index] -=1
    if 0 in data:
        data.remove(0)
    
res = str(ma)+" "+ str(mi)
