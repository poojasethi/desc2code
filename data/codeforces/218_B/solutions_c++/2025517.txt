#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int i,n,m,a[1010],Min,Max,b[1010];
int main()
{
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;i++) scanf("%d",&a[i]);
    memcpy(b,a,sizeof(b));
    Min=Max=0;
    for(i=1;i<=n;i++)
    {
        sort(a+1,a+m+1);sort(b+1,b+m+1);
        Min+=a[1];a[1]--;
        if (!a[1]) a[1]=1010;
        Max+=b[m];b[m]--;
        if (!b[m]) b[m]=-1;
    }
    printf("%d %d\n",Max,Min);
    return 0;
}
