#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
	int n,i,j,k,max,min,b[1001],a[1001];
	scanf("%d%d",&n,&k);
	for(i=0;i<k;i++)
	{
		scanf("%d",&a[i]);
		b[i]=a[i];
	}
	min=0;
	max=0;
	for(i=0;i<n;i++)
	{
		sort(a,a+k);
		max=max+a[k-1];
		a[k-1]--;
	}
	for(i=0;i<n;i++)
	{
		sort(b,b+k);
		for(j=0;j<k;j++)
		{
			if(b[j]>0)
			{
				min=min+b[j];
				b[j]--;
				break;
			}
		}
	}
	printf("%d %d\n",max,min);
}
