#include <iostream>
#include <algorithm>
#include <cstdio>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;
int n, m, a[1234], f = 0, s = 0;
main()
{
	cin >> n >> m;
	rep(i,m) cin >> a[i];
	sort(a, a + m);
	int k = n;
	rep(i, m) 
		rep(j, a[i]) 
			if(k > 0) s += a[i] - j, k--;
	rep(i, n) 
	{
		int mx = max_element(a, a + m) - a;
		f += a[mx];
		a[mx]--;
	}
	cout << f << " " << s;
}
