#include<stdio.h>
#include<algorithm>
using namespace std;
int a[1000];
int main(){
	int n,m,i,j; 
	scanf("%d%d",&n,&m);
	for(i=0;i<m;i++) scanf("%d",a+i);
	sort(a,a+m);

	int ans_min=0;
	int k=n;
	for(i=0;i<n;i++){
		for(j=0;j<a[i];j++){
			if(k>0){
				ans_min+=a[i]-j;
				k--;
			}
		}
	}

	int ans_max=0;
	for(i=0;i<n;i++){
		int j=max_element(a,a+m)-a;
		ans_max+=a[j];
		a[j]--;
	}
	printf("%d %d\n",ans_max,ans_min);

	return 0;
}
