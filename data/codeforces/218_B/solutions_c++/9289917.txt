#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int main(){
    int n,m,a,b,i,j,s=0,t=0;
    int x[1001],y[1001];
    cin>>n>>m;
    for(i=0;i<m;++i)
        cin>>x[i];
    copy(x,x+m,y);
    i=0;
    while(n--){
        sort(x,x+m);
        sort(y,y+m);
        s+=x[m-1];
        while(y[i]<=0)
            i++;
        t+=y[i];
        x[m-1]--;
        y[i]--;
    }
    cout<<s<<" "<<t;
    return 0;
}
