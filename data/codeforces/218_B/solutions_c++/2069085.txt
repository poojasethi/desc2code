#include<cstdio>
#include<queue>
#include<algorithm>
using namespace std;
priority_queue<int> mx, mn;
int main(){
	int n, m, x; scanf("%d%d", &n, &m);
	for(int i = 0; i < m; i++) scanf("%d", &x), mx.push(x), mn.push(-x);
	int sum = 0;
	for(int i = 0; i < n; i++){
		sum+=(x=mx.top()); mx.pop(); if(x>1) mx.push(x-1);
	}printf("%d ", sum);
	sum = 0;
	for(int i = 0; i < n; i++){
		sum+=(x=mn.top()); mn.pop(); if(x<-1) mn.push(x+1);
	}printf("%d\n", -sum);
}
