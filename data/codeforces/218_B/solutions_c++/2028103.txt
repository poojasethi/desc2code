#include <iostream>
#include <algorithm>
using namespace std;
#define MAXN 1001
int main()
{
	int n,m,a[MAXN],b[MAXN],min_e=0,max_e=0;
	cin >> n >> m;
	for(int i=0;i<m;i++)
	{
		cin >> a[i];
		b[i] = a[i];
	}
	for(int i=0;i<n;i++)
	{
		max_e += (*max_element(a,a+m))--;
		int *p = min_element(b,b+m);
		min_e += (*p) --;
		if (0 == *p)
		{
			*p = MAXN; 
		}
	}
	cout<<max_e<<' '<<min_e<<endl;
	return 0;
}
