#include <bits/stdc++.h>
using namespace std;

int main() {
	//freopen("a.in", "r", stdin);
	int n, m;
	cin >> n >> m;
	priority_queue<int> a, b;
	for (int i = 0; i < m; ++i) {
		int x;
		cin >> x;
		a.push(x);
		b.push(-x);
	}
	int s[2] = { };
	while (n--) {
		s[0] += a.top();
		int x = a.top();
		a.pop();
		a.push(x - 1);
		s[1] = s[1] - b.top();
		x = b.top() + 1;
		b.pop();
		if(x)
		b.push(x);
	}
	printf("%d %d\n", s[0], s[1]);
}
