#include<bits/stdc++.h>
using namespace std;
int arr[10000],arr2[10000];
int main()
{
	int g,a,b,t=0,f=0,j=0;
	cin>>a>>b;
	for(int i=0;i<b;i++)
	{
		cin>>arr[i];
		arr2[i]=arr[i];
	}
	g=a;
	sort(arr,arr+b);
	while(a>0)
	{
		sort(arr,arr+b);
		a--;
		t+=arr[b-1];
		arr[b-1]--;
	}
	a=g;
	while(a>0)
	{
		sort(arr2,arr2+b);
		f=0;
		a--;
		while(arr2[f]==0)
		f++;
		j+=arr2[f];
		arr2[f]--;
	}
	cout<<t<<" "<<j<<endl;




return 0;
}
