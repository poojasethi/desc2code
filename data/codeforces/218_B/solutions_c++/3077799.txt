#include<stdio.h>
#include<algorithm>
using namespace std;
int a[1010],n,m;
int main(){
	int i,j,k,as,mas;
	scanf("%d%d",&m,&n);
	for(i=0;i<n;i++)scanf("%d",&a[i]);
	sort(a,a+n);
	k=m;as=mas=0;
	for(i=0;i<n&&k-a[i]>0;i++){
		mas+=(a[i]+1)*a[i]/2;
		k-=a[i];
	}
	mas+=(a[i]+a[i]-k+1)*k/2;
	for(i=0;i<n/2;i++)swap(a[i],a[n-1-i]);
	while(m){
		k=a[0];
		for(i=0;i<n&&a[i]==k&&m;i++){
			as+=a[i]--;
			m--;
		}
	}
	printf("%d %d\n",as,mas);
	return 0;
}
