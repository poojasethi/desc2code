#include <cstdio>
#include <queue>
using namespace std;
priority_queue<int,vector<int>,greater<int> > mq;
priority_queue<int> Mq;
int n,m,a;
int main()
{
scanf("%d%d",&n,&m);
for(int i=1; i<=m; ++i)
scanf("%d",&a),mq.push(a),Mq.push(a);
int msum=0,Msum=0;
for(int i=1; i<=n; ++i)
{
int mtop=mq.top(),Mtop=Mq.top();
msum+=mtop,Msum+=Mtop;
mq.pop(),Mq.pop();
if (mtop>1) mq.push(mtop-1);
if (Mtop>1) Mq.push(Mtop-1);
}
printf("%d %d\n",Msum,msum);
}
