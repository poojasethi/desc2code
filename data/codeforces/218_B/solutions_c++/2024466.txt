#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
using namespace std;
int n,m,a[1005];
priority_queue<int> q;
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=0;i<m;i++)
		scanf("%d",&a[i]),q.push(a[i]);
	sort(a,a+m);
	int pos=0,amin=0,amax=0;
	for (int i=0;i<n;i++)
	{
		amin+=a[pos];
		a[pos]--;
		if (a[pos]==0)pos++;
	}
	for (int i=0;i<n;i++)
	{
		int x=q.top();q.pop();
		amax+=x;
		if (x-1!=0)q.push(x-1);
	}
	printf("%d %d",amax,amin);
	return 0;
}
