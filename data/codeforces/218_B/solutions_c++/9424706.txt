#include<iostream>
#include<algorithm>

using namespace std;
int main(){
	int n,m,min=0,max=0;
	cin>>n>>m;
	int *a = new int[m];
	int *b = new int[m];
	for(int i=0 ; i<m ; i++){
		cin>>a[i];
		b[i] = a[i];
	}

	sort(a,a+m);
	sort(b,b+m);
	int j=0;
	int i=0;
	while(i<n){
		min+=a[j];
		i++;
		a[j]--;
		if(a[j]==0)
			j++;
	}
	j=m-1;
	i=0;
	while(i<n){
		max+=b[j];
		i++;
		b[j]--;
		sort(b,b+m);
	}
	cout<<max<<" "<<min<<endl;
	return 0;
}
