#include <iostream>
#include <queue>

using namespace std;

int main()
{
priority_queue <int> m1,m2;
int n,m,a,min=0,max=0;
cin>>n>>m;
for  (int i=1; i<=m;i++)
 {
  cin>>a; m1.push(a); m2.push(-a);
 }

for  (int i=1; i<=n;i++)
 { int q=m1.top(); max+=q; m1.pop();
   if (q-1) m1.push(q-1);

   q=-m2.top(); min+=q; m2.pop();
   if (1-q) m2.push(1-q);
 }

cout<<max << " " << min<<endl;
return 0;
}
