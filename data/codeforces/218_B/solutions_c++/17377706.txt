#include <bits/stdc++.h>
using namespace std;
#define maxn 1000

int a[maxn], b[maxn];

int main()
{
	int n, m, i, ret1 = 0, ret2 = 0;
	scanf("%d %d", &n, &m);
	for(i = 0; i < m; i++)
	{
		scanf("%d", a + i);
		b[i] = a[i];
	}
	while(n--)
	{
		sort(a, a + m);
		ret1 += a[m - 1];
		a[m - 1]--;
		sort(b, b + m);
		i = upper_bound(b, b + m, 0) - b;
		ret2 += b[i];
		b[i]--;
	}
	printf("%d %d", ret1, ret2);
	return 0;
}
