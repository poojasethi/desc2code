#include<iostream>
#include<algorithm>
using namespace std;
int a[10001];
int b[10001];
int main()
{
	int n,m;
	cin>>n>>m;
	int p=m;
	for(int i=0;i<m;i++)
	{
		cin>>a[i];
		b[i]=a[i];
	}
	sort(a,a+m);
	int maxx=0;
	int minx=0;
	for(int i=0;i<n;i++)
	{
		maxx+=a[m-1];
		a[m-1]--;
		if(a[m-1]==0)
			m--;
		sort(a,a+m);
	}
	sort(b,b+p);
	int l=0;
	for(int i=0;i<n;i++)
	{
		minx+=b[l];
		b[l]--;
		if(b[l]==0)
			l++;
		sort(b,b+p);
	}
	cout<<maxx<<" "<<minx<<endl;
}
