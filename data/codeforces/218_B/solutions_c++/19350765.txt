#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int n, m, a[1000], b[1000], max = 0, min = 0, c = 0;
	cin >> n >> m;
	for (int i = 0; i < m; i++)cin >> a[i];
	for (int i = 0; i < m; i++)b[i] = a[i];
	while(c!=n){
		sort(b, b + m);
		max += b[m - 1];
		b[m - 1]--;
		c++;
	}
	c = 0;
	int i = 0;
	sort(a, a + m);

	while (c != n){
		if (a[i] == 0)i++;
		min += a[i];
		a[i]--;
		c++;
	}

	cout << max << " " << min;
	return 0;
}
