#include<cstdio>
#include<iostream>
#include<queue>
using namespace std;
int main()
{
int n,m,a,i,max=0,min=0,h,l;
priority_queue<int> maxheap,minheap;
scanf("%d %d",&n,&m);
for(i=0;i<m;i++)
{
scanf("%d",&a);
maxheap.push(a);
minheap.push(-1*a);
}
for(i=0;i<n;i++)
{
h=maxheap.top();
maxheap.pop();
max+=h;
maxheap.push(h-1);
l=-1*minheap.top();
minheap.pop();
min+=l;
if(l-1)
minheap.push(-1*(l-1));
}
printf("%d %d\n",max,min);
return 0;
}
