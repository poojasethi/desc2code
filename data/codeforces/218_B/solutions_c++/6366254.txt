#include<cstdio>
#include<algorithm>
using namespace std;
int a1[1001];
int main()
{
	int k,m,n,x,o1=0,o2=0,a;
	scanf("%d%d",&k,&n);
	m=k;
	while(n--)
	{
		scanf("%d",&x);
		++a1[x];
	}
	for(int i=1;m&&i<1001;++i)
	for(int j=0;j<a1[i];++j)
	{
		a=min(i,m);
		o1+=a*(2*i-a+1)/2;
		m-=a;
	}
	m=k;
	for(int i=1000;m&&i;--i)
	{
		a=min(a1[i],m);
		a1[i-1]+=a;
		o2+=i*a;
		m-=a;
	}
	printf("%d %d\n",o2,o1);
}
