#include "iostream"
#include"algorithm"
using namespace std;


int main()
{
int n,m,arr[1000],arr2[1000],max=0,min=0;
cin>>n>>m;
for(int i=0;i<m;i++)
{
	cin>>arr[i];
arr2[i]=arr[i];
}
for(int i=0;i<n;i++)
{
	sort(arr2,arr2+m);
	max=arr2[m-1]+max;
	arr2[m-1]--;
}
int t=0;
for(int i=0;i<n;i++)
{
	sort(arr,arr+m);
	if(arr[t]==0)
		t++;
	min=arr[t]+min;
	arr[t]--;
}
cout<<max<<" "<<min;

	

	return 0;
}
