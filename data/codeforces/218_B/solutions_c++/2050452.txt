#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main()
{
	int i,n,m,t,mi=0,ma=0;
	cin>>n>>m;
	vector<int> a;
	for (i=0;i<m;i++)
		cin>>t, a.push_back(t);
	sort(a.begin(),a.end());
	t=0;
	for (i=0;i<m;i++)
	{
		mi+=((a[i]+1)*a[i])/2;
		t+=a[i];
		if (t>n)
		{
			t=t-n;
			mi-=((t+1)*t)/2;
			break;
		}
	}
	while (n>0)
	{
		sort(a.rbegin(),a.rend());
		ma+=a[0];
		a[0]--;
		n--;
	}
	cout<<ma<<' '<<mi<<endl;
	return 0;
}
