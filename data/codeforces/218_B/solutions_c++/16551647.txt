#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,m1,m,j=0,mx=0,mn=0,a[1001],b[1001];
    cin>>n>>m;
    m1=m;
    for(i=0;i<m;i++) {cin>>a[i];b[i]=a[i];}
    for(i=0;i<n;i++){
        sort(a,a+m);
        mx+=a[m-1];
        a[m-1]--;
        if(a[m-1]==0) m--;
    }
    for(i=0;i<n;i++){
        sort(b,b+m1);
        mn+=b[j];
        b[j]--;
        if(b[j]==0) j++;
    }
    cout<<mx<<' '<<mn;
    return 0;
}
