#include<iostream>
#include<algorithm>
#include<cstdio>
#include<vector>
using namespace std;

vector<int> a,b;

int main()
{
	int n,m,res1=0,res2=0,x;
	cin >> n >> m;
	for(int i=0;i<m;i++) cin >> x,a.push_back(x),b.push_back(x);
	for(int i=0;i<n;i++){
		sort(a.begin(),a.end());
		res1+=a[0]--;
		if(!a[0]) a.erase(a.begin());
		sort(b.rbegin(),b.rend());
		res2+=b[0]--;
		if(!b[0]) b.erase(b.begin());
	}
	cout << res2 << " " << res1 ;
	return 0;
}
