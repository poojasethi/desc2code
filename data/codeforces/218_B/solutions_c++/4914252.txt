#include<iostream>
#include<algorithm>
using namespace std;

const int oo=1010;
int a[oo],b[oo];

int main(){
  int n,m,Max=0,Min=0,p=0;
  cin>>n>>m;
  for(int i=0;i<m;i++){
    cin>>a[i];
    b[i]=a[i];
  }
  for(int i=0;i<n;i++){
    sort(a,a+m);
    Max+=a[m-1];
    a[m-1]--;
  }
  for(int i=0;i<n;i++){
    sort(b,b+m);
    while(!b[p]){
    p++;}
      Min+=b[p];
      b[p]--;
  }
  cout<<Max<<' '<<Min;
  return 0;
}
