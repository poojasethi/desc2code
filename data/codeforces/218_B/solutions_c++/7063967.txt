#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,b[1010],i,j,maxa=0,mina=0,cnt;
	vector<int> a;
	cin>>n>>m;
	for(i=0;i<m;i++)
	{
		cin>>b[i];
		for(j=1;j<=b[i];j++)
			a.push_back(j);
	}
	sort(a.begin(),a.end());
	reverse(a.begin(),a.end());
	for(i=0;i<n;i++)
		maxa+=a[i];
	sort(b,b+m);
	for(i=0,cnt=0;cnt<n;i++)
	{
		while(cnt<n && b[i])
		{
			mina+=b[i]--;
			cnt++;
		}
	}
	cout<<maxa<<' '<<mina<<endl;
	return 0;
}
