#include<cstdio>
int n,m,a[1100],ans,b[1100];
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;++i) scanf("%d",&a[i]),b[i]=a[i];
	for (int i=1;i<=n;++i)
	{
		int c=-1,k;
		for (int j=1;j<=m;++j)
		if (a[j]&&a[j]>c) c=a[j],k=j;
		ans+=a[k],--a[k];
	}
	printf("%d ",ans);
	ans=0;
	for (int i=1;i<=n;++i)
	{
		int c=100000000,k;
		for (int j=1;j<=m;++j)
		if (b[j]&&b[j]<c) c=b[j],k=j;
		ans+=b[k],--b[k];
	}
	printf("%d\n",ans);
	return 0;
}
