#include<iostream>
#include<queue>
using namespace std;
main(){
priority_queue<int> q1,q2;
int n,m,a,min=0,max=0;
cin>>n>>m;
for(int i=0;i<m;i++)
    cin>>a,q1.push(a),q2.push(-a);

for(int i=1;i<=n;i++){
 int q=q1.top(); max+=q; q1.pop();
   if (q-1)   q1.push(q-1);

   q=-q2.top(); min+=q; q2.pop();
   if (1-q) q2.push(1-q);
}


cout <<max<<" "<<min<<endl;

}
