int i, j, n, m, s1, s2;

int main(){
	cin >> n >> m;
	while( i<m ){
		cin >> a[i];
		b[i]=a[i];
		i++;
	}
	sort(a,a+m);
	for(i=0;i<n;i++){
		if (a[j]==0) j++;
		s1+=a[j];

}
