#include<bits/stdc++.h>
using namespace std;
typedef unsigned long long ll;
#define s(n) scanf("%llu",&n)
#define p(n) printf("%llu ",n)
ll a[1002],b[1002];
int main()
{ ll n,m,i,j,min=0,max=0,x; s(n);s(m);
  for(i=0;i<m;++i) s(x), a[x]++,b[x]++;
  for(i=0;i<n;++i)
  { for(j=1;j<=1000;++j)
     if(a[j]!=0){ min+=j*1;++a[j-1];--a[j];break;}
     for(j=1000;j>=1;--j)
     if(b[j]!=0){max+=j*1;++b[j-1];--b[j];break;}
  }
   cout<<max<<" "<<min;
}
