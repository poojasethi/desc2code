#include<iostream>
#include<algorithm>
using namespace std;
long long q,n,a[3001],b[1001],m,o,p,f;
string s;
int main(){
	cin>>n>>m;
	for(int i=1;i<=m;i++){
		cin>>a[i];
		b[i]=a[i];
	}
	for(int i=1;i<=n;i++){
		sort(a+1,a+1+m);
		q=1;
		while(a[q]<1){
			q++;
		}
		o+=a[q];
		a[q]--;
		
	}

		sort(b+1,b+1+m);

		while(b[q]<1){

	}
