#include<bits/stdc++.h>
using namespace std;

int n,nn,m,i,j,a[100000],b[100000],maks=0,minn=0;

int main(){
	cin >> n >> m ;
	for(i=0;i<m;i++){
		cin >> a[i];
		b[i]=a[i];
	}
	sort(a,a+m);
	i=0;nn=n;
	while(n>0){
		while(a[i]>0 && n>0){
			minn+=a[i];
			a[i]--;
			n--;
		}
		i++;
	}
	sort(b,b+m);
	while(nn>0){
		maks+=b[m-1];
		b[m-1]--;
		nn--;
		sort(b,b+m);
	}
	cout << maks <<" "<< minn << endl;
}
