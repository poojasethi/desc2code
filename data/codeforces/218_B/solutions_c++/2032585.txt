#include <iostream>
#include <algorithm>
using namespace std;
int a[1010],b[1010];
int main()
{
	int n,m;
	while (cin>>n>>m) {
		for (int i=0;i<m;++i) { cin>>a[i]; b[i]=a[i]; }
		int maxi=0,mini=0;
		for (int p=0;p<n;++p) {
			int k=max_element(a, a+m)-a;
			maxi+=a[k];
			a[k]--;
			k=-1;
			for (int i=0;i<m;++i) if (b[i]>0 && (k<0 || b[k]>b[i])) k=i;
			mini+=b[k];
			b[k]--;
		}
		cout<<maxi<<' '<<mini<<endl;
	}
	return 0;
}
