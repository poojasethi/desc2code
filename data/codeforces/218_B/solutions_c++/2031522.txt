#include<cstdio>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n,m; scanf("%d%d",&n,&m);
	int a[1000];
	rep(i,m) scanf("%d",a+i);
	sort(a,a+m);

	int ans_min=0;
	int k=n;
	rep(i,m) rep(j,a[i]) if(k>0) ans_min+=a[i]-j, k--;

	int ans_max=0;
	rep(i,n){
		int j=max_element(a,a+m)-a;
		ans_max+=a[j];
		a[j]--;
	}

	printf("%d %d\n",ans_max,ans_min);

	return 0;
}
