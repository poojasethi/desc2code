n,m,k=map(int,raw_input().split())
rt=[x for x in range(n)]
def root_of(x):
  if x!=rt[x]:
    rt[x]=root_of(rt[x])
  return rt[x]
if n>=k:
  for i in xrange(k,1,-2):
    for j in range(n-k+1):
      rt[root_of(j+i-1)]=root_of(j)
t=1
for i in range(n):
  if i==root_of(i):
    t=t*m%1000000007
print t
