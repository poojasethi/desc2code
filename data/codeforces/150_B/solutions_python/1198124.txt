def __main__():
    #import sys.stdin as inp
    __inp = raw_input().split(' ')
    n = int(__inp[0])
    m = int(__inp[1])
    k = int(__inp[2])

    res = 0
    if k > n or k == 1:
        res = m**n
    elif k == n:
        res = m**((n + 1)/2)
    else:
        res = m
        if k%2 == 1:
            res += m*(m - 1)
    print res%(10**9 + 7)
    
__main__()
