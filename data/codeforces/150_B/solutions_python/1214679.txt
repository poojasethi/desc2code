import sys

MOD = 1000000007
n, m, k = [int(t) for t in sys.stdin.readline().strip().split(' ')]

r = 1

if k == n:
    for i in xrange((n + 1) / 2):
        r = (r * m) % MOD
elif k == 1 or k > n:
    for i in xrange(n):
        r = (r * m) % MOD
elif k % 2 == 0:
    r = m
else:
    r = m * m

print r
