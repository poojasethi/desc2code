#include<cstdio>
#define ll __int64
const int mod=1000000007;
int main()
{
	int i,n,m,k;
	ll ans;
	scanf("%d%d%d",&n,&m,&k);
	if(k==1||k>n)
	{
		for(ans=i=1;i<=n;i++)
			ans=ans*m%mod;
	}
	else
	{
		if(n>k)
			ans=k&1?m*m:m;
		else
			for(ans=i=1;i<=(n+1)/2;i++)
				ans=(ans*m)%mod;
	}
	printf("%I64d\n",ans);
	return 0;
}
