#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

int main()
{
    int64_t q = 1;
    int n, m, k;

    cin >> n >> m >> k;

    if ( (n <= k) || (k == 1) )
    {
        if (n == k) n = (n + 1)/2;
        for (k = 0; k < n; k++) q = (q * m)%1000000007;
    }
    else
    {
            if (k % 2) q = m * m;
            else q = m;
    }

    cout << q;

    return 0;
}
