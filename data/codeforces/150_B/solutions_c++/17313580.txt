#include <cstdio>
#include <algorithm>
using namespace std;
#define MOD 1000000007
int main(){
	int n, m, k;
	long long ans = 0;
	scanf("%d %d %d", &n, &m, &k);
	if (n <= k || k==1){
		int Min;
		if(n<k||k==1) Min=n;
		else Min=(n+1)/2;
		ans = 1;
		for (int i = 0; i < Min; i++)
			ans = (ans*m) % MOD;
	}
	else{
		ans += m;
		if (k % 2 == 1) ans = (ans + m*(m - 1)) % MOD;
	}
	printf("%I64d", ans);
	return 0;
}
