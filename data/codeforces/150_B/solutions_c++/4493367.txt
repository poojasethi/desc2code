#include <cstdio>
#define MOD 1000000007
int n,m,k;
int main()
{
	scanf("%d %d %d",&n,&m,&k);
	if (k>n || k==1)
	{
		long long int son=1;
		for (int i=1;i<=n;i++)
		{	
			son*=m;
			son=son%MOD;
		}
		printf("%I64d",son);
		return 0;
	}
	if (k==n)
	{
		long long int son=1;
		for (int i=1;i<=(n+1)/2;i++)
		{	
			son*=m;
			son=son%MOD;
		}
		printf("%I64d",son);
		return 0;
	}
	if (k%2)
		printf("%d",m*m);
	else 
		printf("%d",m);
}
