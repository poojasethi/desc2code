#include<iostream>
using namespace std;
int main(){
	int n,m,k,i;
	unsigned long long ans=1;
	cin>>n>>m>>k;
	if(k==1||k>n){
		for(i=0;i<n;i++){
			ans*=m;
			ans%=1000000007;
		}
	}
	else if(k==n){
		for(i=0;i<(n+1)/2;i++){
			ans*=m;
			ans%=1000000007;
		}
	}
	else if(k%2==0){
		ans=m;
	}
	else{
		ans=m*m;
	}
	cout<<ans;
	return 0;
}
