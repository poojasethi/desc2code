#include<iostream>
using namespace std;
int main()
{
	int n,m,k;
	cin>>n>>m>>k;
	long long ans=1;
	if(k>n||k==1)
		for(int i=1;i<=n;++i)
			ans=ans*m%1000000007;
	else if(k==n)
		for(int i=1;i<=((n+1)>>1);++i)
			ans=ans*m%1000000007;
	else if(k&1)
		ans=m*m;
	else
		ans=m;
	cout<<ans<<endl;
	return 0;
}
