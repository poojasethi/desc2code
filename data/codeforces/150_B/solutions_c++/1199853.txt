#include<iostream>
using namespace std;

int n,m,k,i;
long long ans=1;

int main()
{
    cin >> n >> m >> k;
    if (k>n || k==1)
    {
        for(i=1;i<=n;++i)
            ans=ans*m % 1000000007;
    }
    else if (k == n)
    {
        for(i=1;i<=(k+1)/2;++i)
            ans=ans*m % 1000000007;
    }
    else if (k % 2 == 1)
    {
        ans=m*m;
    }
    else
    {
        ans=m;
    }
    cout << ans;
    return 0;
}
