#include <stdio.h>
#include <bitset>
#include <algorithm>
#include <cmath>
#include <vector>
#include <utility>
#include <set>//set for distinct
using namespace std;
#define f(i,n) for (int i=0;i<n;++i)
#define FOR(i,a,b) for (int i=a;i<=b;i++)
#define DFOR(i,a,b) for (int i=a;i>=b;i--)
int k,m;
set<int> res;
int head;
void generate(int v,int rem)
{
    if (rem == 0)
    {
        int dif=abs(abs(v)-k);
        if (dif < 10000)
        {
            if (res.size() < m) res.insert(dif+head*10000);
            if (res.size() < m) res.insert(dif*10000+head);
        }
        return;
    }
    generate(v + rem%10,rem/10);
    generate(v - rem%10,rem/10);//前是0-9999
    generate(v * (rem%10),rem/10);
}
int main()
{
    scanf("%d%d",&k,&m);
    FOR(i,0,9999)
    {
        head=i;//用head的循環gen loop
        generate(i%10,i/10);//recur 取每位..a.
    }
    for(set<int>::iterator i=res.begin();i!=res.end();++i) printf("%08d\n",*i);
    return 0;
}
