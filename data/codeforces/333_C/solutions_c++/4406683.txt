#include <iostream>
#include <cstdio>
#include <cmath>
#include <set>
#include <limits>
#include <map>
#include <vector>
#include <algorithm>
#include <iomanip>
#include <sstream>
using namespace std;
typedef long long ll;
int m;
size_t number_of_tickets;
set<ll> results;
void solve(int value, int str, int remaining) {
    if (results.size() >= number_of_tickets)
        return;
    if (remaining == 0) {
        int diff = abs(abs(value) - m);
        if (diff < 10000) {
            results.insert(str * 10000 + diff);
            results.insert(str + diff * 10000);
        }
        return;
    }
    solve(value + (remaining % 10), str, remaining / 10);
    solve(value - (remaining % 10), str, remaining / 10);
    solve(value * (remaining % 10), str, remaining / 10);
}
int main() {
    ios_base::sync_with_stdio(false);
    cin >> m >> number_of_tickets;
    for (int i = 0; i < 10000; i++)
        solve(i % 10, i, i / 10);
    size_t i = 0;
//  cout << "Matches: " << results.size() << endl;
    for (set<ll>::iterator it = results.begin(); it != results.end(); it++) {
        if (i == number_of_tickets)
            return 0;
        cout << setfill('0') << setw(8) << *it << endl;
        i++;
    }
    return 0;
}
