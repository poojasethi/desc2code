#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
using namespace std;
int k,m,b[10000];
void dfs(int i,int s,int x,int n){
	if(i>3){
		if(i==4){
			s=abs(s-k);
			if(s<1e4&&b[s]!=n){
				printf("%04d%04d\n",n,s);
				b[s]=n;
				if(!--m) exit(0);
			}
		}
		return;
	}
	dfs(i+1,s+x%10,x/10,n);
	dfs(i+1,s-x%10,x/10,n);
	dfs(i+1,x%10*s,x/10,n);
	dfs(i+2,s+x%100,x/100,n);
	dfs(i+2,s-x%100,x/100,n);
	dfs(i+2,x%100*s,x/100,n);
}
int main(){
	scanf("%d%d",&k,&m);
	memset(b,-1,sizeof(b));
	for(int i=0;i<1e4;i++){
		dfs(1,i%10,i/10,i);
		dfs(2,i%100,i/100,i);
	}
}
