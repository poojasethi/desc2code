#include <cstdio>
#include <vector>
#include <cassert>
#include <set>
using namespace std;
int k, m;
int flen(int x){int r=0;while(x>0)x/=10,r++;return r;}
vector<int> f(int a, int len){
  vector<int> res;
  res.push_back(a);
  res.push_back(-a);
  for(int i = 1, j = 10; i < len; i++, j *= 10){
    vector<int> L = f(a % j, i);
    vector<int> R = f(a / j, len - i);
    for(int k1 = 0; k1 < (int)L.size(); k1++)
      for(int k2 = 0; k2 < (int)R.size(); k2++)
	res.push_back(L[k1] + R[k2]), res.push_back(L[k1] * R[k2]); 
  }
  return res;
}
int main(){
  scanf("%d%d", &k, &m);
  set<int> S;
  for(int i = 1; i < 10000 && S.size() < m; i++){
    vector<int> V = f(i, flen(i));
    for(int j = 0; j < (int)V.size() && S.size() < m; j++){
      if(k - V[j] >= 0 && k - V[j] < 10000){
	S.insert((k-V[j])*10000+i);
      }
    }
  }

  assert(S.size() == m);
  for(set<int>::iterator it = S.begin(); it != S.end(); it++)
    printf("%08d\n", *it);
  return 0;
}
