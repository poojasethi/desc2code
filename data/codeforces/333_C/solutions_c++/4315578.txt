#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <set>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
set<int>s;
typedef set<int>::iterator w1;
void work(int x,int y) {
    if (!y) {
        int A=abs(abs(x)-n);
        if (A<10000) {
            if (s.size()<m) s.insert(k*10000+A);
            if (s.size()<m) s.insert(k+A*10000);
        } return;
    }
    work(x+(y%10),y/10);
    work(x-(y%10),y/10);
    work(x*(y%10),y/10);
}
int main() {
    scanf("%d%d",&n,&m);
    For(i,0,9999) {
        k=i; work(i%10,i/10);
    }
    w1 A=s.begin();
    for (;A!=s.end();A++) printf("%08d\n",*A);
    return 0;
}
