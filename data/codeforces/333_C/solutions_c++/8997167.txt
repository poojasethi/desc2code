#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
typedef vector<int> V;
V a[10001];int n,k;
void work(V&A,V&B,V&C)
{
	for(V::iterator b=B.begin();b!=B.end();++b)
		for(V::iterator c=C.begin();c!=C.end();++c)
			A.push_back(*b+*c),A.push_back(*b* *c);
}
int main()
{
	scanf("%d%d",&n,&k);
	for(int i=1;i<10000;i++)
	{
		a[i].push_back(i);
		a[i].push_back(-i);
		if(i>9)work(a[i],a[i/10],a[i%10]);
		if(i>99)work(a[i],a[i/100],a[i%100]);
		if(i>999)work(a[i],a[i/1000],a[i%1000]);
		sort(a[i].begin(),a[i].end());
		a[i].resize(unique(a[i].begin(),a[i].end())-a[i].begin());
	}
	for(int i=0;i<10000&&k;i++)
		for(V::iterator it=a[i].begin();it!=a[i].end()&&k;++it)
			if(*it+n>0&&*it+n<10000)
				--k,printf("%04d%04d\n",*it+n,i);
}
