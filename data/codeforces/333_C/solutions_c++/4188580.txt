#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
vector <int> f(int i, int k)
{
    vector <int> v;
    v.push_back(i);
    v.push_back(-i);
    if (k>1)
    {
        for (int l=1,a=10; l<k; l++,a*=10)
        {
            vector <int> v1=f(i/a,k-l),v2=f(i%a,l);
            for (int j=0; j<v1.size(); j++)
            {
                for (int h=0; h<v2.size(); h++)
                {
                    v.push_back(v1[j]+v2[h]);
                    v.push_back(v1[j]*v2[h]);
                }
            }
        }
    }
    sort(v.begin(),v.end());
    vector <int> ans;
    for (int j=0; j<v.size(); j++)
    {
        if (j==0 || v[j-1]<v[j]) ans.push_back(v[j]);
    }
    return ans;
}
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for (int i=0; i<10000; i++)
    {
        vector <int> v=f(i,4);
        for (int j=0; j<v.size(); j++)
        {
            int k=n-v[j];
            if (0<=k && k<10000) {printf("%04d%04d\n",k,i); m--;}
            if (m==0) return 0;
        }
    }
    return 0;
}
