const int mod=1e9+7;
const int N=1e4+10;
int n,k,vis[N];
void dfs(int dep,int s,int x,int num){
    if(!n) return;
    if(dep>3){
        int ans=abs(s-k);
        if(dep==4&&(ans<1e4&&vis[ans]!=num)){
            printf("%04d%04d\n",num,ans); n--; vis[ans]=num;
        }
        return;
    }
    dfs(dep+1,x%10+s,x/10,num);
    dfs(dep+1,x%10*s,x/10,num);
    dfs(dep+1,s-(x%10),x/10,num);
    dfs(dep+2,s+(x%100),x/100,num);
    dfs(dep+2,x%100*s,x/100,num);
    dfs(dep+2,s-(x%100),x/100,num);
}
