#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<set>
using namespace std;
set<int>a;
set<int>::iterator it;
int n,k,xx;
void dfs(int x,int z){
     if(a.size()>=n)return;
     if(!x){
       int y=abs(abs(z)-k);
       if(y<10000){
         a.insert(xx*10000+y);
         a.insert(y*10000+xx);
       }
       return;
     }
     dfs(x/10,z+(x%10));
     dfs(x/10,z-(x%10));
     dfs(x/10,z*(x%10));
}
int main(){
    cin>>k>>n;
    for(int i=0;i<=9999;++i){
      xx=i;
      dfs(i/10,i%10);
    }
    for(it=a.begin();n>0;++it,--n)printf("%08d\n",*it);
    return 0;
}
