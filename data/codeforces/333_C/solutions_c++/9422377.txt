#include<cstdio>
#include<map>
#include<cstdlib>
using namespace std;
map <int,bool> x;
int k,m;
int p10[20]={0,1,10,100,1000,10000,100000,1000000,10000000,100000000,1000000000};
void dfs(int cal,int val,int pla,int pla2,int ex,int w)
{
	if(k-cal>=p10[pla-pla2+2]||k<cal)
		return;
	val+=(k-cal)*p10[pla2];
		//printf("%d %d %d %d %d\n",cal,val-(k-cal)*p10[pla2],pla,pla2,ex);
	if(!x[val])
	{
		x[val]=1;
		printf("%08d\n",val);
		--m;
		if(!m)
			exit(0);
	}
	val-=(k-cal)*p10[pla2];
	if(pla>=pla2)
		for(int i=0;i<10;++i)
		{
			dfs(cal+i,val+p10[pla]*i,pla-1,pla2,ex,w);
			dfs(cal-i,val+p10[pla]*i,pla-1,pla2,ex,w);
			dfs(cal+i,val+p10[pla2]*i,pla,pla2+1,1,w);
			dfs(cal-i,val+p10[pla2]*i,pla,pla2+1,1,w);
			dfs(cal+w*i*p10[ex],val+p10[pla2]*i,pla,pla2+1,ex+1,w);
		}
}
int main()
{
	scanf("%d%d",&k,&m);
	dfs(0,0,8,1,1,1);
	printf("s%d\n",m);
	return 0;
}
