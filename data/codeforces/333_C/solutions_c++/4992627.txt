set<int> s;
int m, k, idx;

void Pro(int x, int num)
{
	if ((int)s.size() >= m) return ;
	if (!num)
	{
		int dif = abs(abs(x) - k);
		if (dif < 10000)
			s.insert(idx * 10000 + dif),
			s.insert(dif * 10000 + idx);
		return ;
	}
	Pro(x + num % 10, num / 10);
	Pro(x - num % 10, num / 10);
	Pro(x * (num % 10), num / 10);
}
