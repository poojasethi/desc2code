#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
#define N 10000
int a[N],n,m;
void DFS(int x,int y,int z,int o)
 {
    if (x>3)
     {
        y=abs(n-y);
        if (x!=4||y>=N||a[y]==o) return;
        printf("%04d%04d\n",o,y);a[y]=o;
        if (!(--m)) exit(0);
        return;
     }
    DFS(x+1,y+z%10,z/10,o);
    DFS(x+1,y-z%10,z/10,o);
    DFS(x+1,y*(z%10),z/10,o);
    DFS(x+2,y+z%100,z/100,o);
    DFS(x+2,y-z%100,z/100,o);
    DFS(x+2,y*(z%100),z/100,o);
 }
int main()
 {
    cin >>n>>m;
    for (int i=0;i<N;i++) a[i]=N;
    for (int i=0;i<N;i++)
      DFS(1,i%10,i/10,i),DFS(2,i%100,i/100,i);
    return 0;
 }
