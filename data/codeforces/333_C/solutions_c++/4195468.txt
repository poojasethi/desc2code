#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>

#include <map>

#include <cassert>
#include <sstream>
#include <fstream>

using namespace std;

const int WINDOW = 10000;

set <int> st;
int n, m, xValue;

void solve(int value, int remain)
{
	if(remain == 0)
	{
		int diff = abs(abs(value) - n);
		if(diff < WINDOW)
		{
			if(st.size() < m)
				st.insert(diff + xValue * WINDOW);
			if(st.size() < m)
				st.insert(xValue + diff * WINDOW);
		}
		return;
	}
	solve(value + remain % 10, remain / 10);
	solve(value - remain % 10, remain / 10);
	solve(value * (remain % 10), remain / 10);
}

int main()
{
//	freopen("in.txt", "r", stdin);
	while(scanf("%d %d", &n, &m) == 2)
	{
		st.clear();
		for(int i = 0; i < WINDOW; i++)
		{
			xValue = i;
			solve(i % 10, i / 10);
		}
		for(set <int> :: iterator it = st.begin(); it != st.end(); it++)
			printf("%08d\n", *it);
	}
	return 0;
}
