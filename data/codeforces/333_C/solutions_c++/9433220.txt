#include <cstdio>
#include <cstdlib>
#include <set>
std::set<int>s;
int b,K,m,lim;
void dfs(int k,int sum,int x)
{
	if (k > lim) {
		int y = abs(K - sum);
		if (y < b) s.insert(x * b + y),s.insert(y * b + x);
		y = abs(K + sum);
		if (y < b) s.insert(x * b + y),s.insert(y * b + x);
		if (s.size() >= m) {
			for (std::set<int>::iterator it = s.begin(); it != s.end() && m--; ++it)
				printf("%08d\n",*it);
			exit(0);
		}
		return;
	}
	for (int i = 0; i < 10; ++i) {
		dfs(k + 1,sum + i,x * 10 + i);
		dfs(k + 1,sum - i,x * 10 + i);
		if (k == 1) dfs(k + 1,i,x * 10 + i);
		else dfs(k + 1,sum * i,x * 10 + i);
	}
}
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("333C.in","r",stdin);
	freopen("333C.out","w",stdout);
	#endif
	scanf("%d%d",&K,&m);
	for (b = 10000,lim = 4; b <= 10000; b *= 10,--lim) dfs(1,0,0);
}
