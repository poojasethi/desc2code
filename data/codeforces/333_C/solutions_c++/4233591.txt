#include <iostream>
#include <cstdio>
#include <cmath>
#include <set>

using namespace std;

const int WINDOW = 10000;

set <int> st;
int n, xValue;
unsigned m;

void solve(int value, int remain)
{
    if(remain == 0) {
        int diff = abs(abs(value) - n);
        if(diff < WINDOW) {
            if(st.size() < m)
                st.insert(diff + xValue * WINDOW);
            if(st.size() < m)
                st.insert(xValue + diff * WINDOW);
        }
        return;
    }
    solve(value + remain % 10, remain / 10);
    solve(value - remain % 10, remain / 10);
    solve(value * (remain % 10), remain / 10);
}

int main()
{
    while(scanf("%d%u", &n, &m) == 2) {
        st.clear();
        for(int i = 0; i < WINDOW; i++) {
            xValue = i;
            solve(i % 10, i / 10);
        }
        for(set <int> :: iterator it = st.begin(); it != st.end(); it++)
            printf("%08d\n", *it);
    }
    return 0;
}
