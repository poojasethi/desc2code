#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
using namespace std;
int i,n,m,a1,a3,a2,a4,bo[12000];
void renew(int x){
	x=abs(n-x);
	if(x<10000 && bo[x]!=i){
		bo[x]=i;
		printf("%04d%04d\n",x,i);
		m--;
		if(!m)exit(0);
	}
}
void work(int a,int b){
	renew(a*b);
	renew(-a*b);
	renew(a+b);
	renew(a-b);
	renew(-a+b);
	renew(-a-b);
}
int main(){
	scanf("%d%d",&n,&m);
	memset(bo,3,sizeof(bo));
	for(i=0;i<=9999;++i){
		a1=i/1000;
		a2=i/100%10;
		a3=i/10%10;
		a4=i%10;
		//two number
		work(a1,a2*100+a3*10+a4);
		work(a1*10+a2,a3*10+a4);
		work(a1*100+a2*10+a3,a4);
		//three number
		work(a1*10+a2,a3+a4);
		work(a1*10+a2,a3*a4);
		work(a1*10+a2,a3-a4);
		work(a1+a2,a3*10+a4);
		work(a1-a2,a3*10+a4);
		work(a1*a2,a3*10+a4);
		work(a1-(a2*10+a3),a4);
		work(a1+(a2*10+a3),a4);
		work(a1*(a2*10+a3),a4);
		//four number
		work(a1+a2,a3+a4);
		work(a1-a2,a3+a4);
		work(a1*a2,a3+a4);
		work(a1+a2,a3-a4);
		work(a1-a2,a3-a4);
		work(a1*a2,a3-a4);
		work(a1+a2,a3*a4);
		work(a1-a2,a3*a4);
		work(a1*a2,a3*a4);
	}
}
