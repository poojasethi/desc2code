#include<stdio.h>
int main()
{
	int n,m;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		int i,j;
		if(n==1&&m==2||(n==2&&m==1))
		{
			printf("0\n");
			printf("1 1\n");
			printf("%d %d\n",n,m);
			printf("1 1\n");
		}
		else if(n==1&&m>2)
		{
			printf("1\n");
			printf("%d %d 1 1\n",n,m);
			for(i=1;i<=m;i++)
				printf("1 %d\n",i);
			printf("1 1\n");
		}
		else if(m==1&&n>2)
		{
			printf("1\n");
			printf("%d %d 1 1\n",n,m);
			for(i=1;i<=n;i++)
				printf("%d 1\n",i);
			printf("1 1\n");
		}
		else
		{
		if(n%2==0)
		{
			printf("0\n");
			printf("1 1\n");
			for(i=1;i<=n;i++)
			{
				if(i%2!=0)
				{
					for(j=2;j<=m;j++)
						printf("%d %d\n",i,j);
				}
				else
				{
					for(j=m;j>1;j--)
						printf("%d %d\n",i,j);
				}
			}
			for(i=n;i>=1;i--)
				printf("%d 1\n",i);
		}
		else if(m%2==0)
		{
			printf("0\n");
			printf("1 1\n");
			for(i=1;i<=m;i++)
			{
				if(i%2!=0)
				{
					for(j=2;j<=n;j++)
						printf("%d %d\n",j,i);
				}
				else
				{
					for(j=n;j>1;j--)
						printf("%d %d\n",j,i);
				}
			}
			for(i=m;i>=1;i--)
				printf("1 %d\n",i);
		}
		else
		{
			printf("1\n");
			printf("%d %d 1 1\n",n,m);
			for(i=1;i<=n;i++)
			{
				if(i%2!=0)
				{
					for(j=1;j<=m;j++)
						printf("%d %d\n",i,j);
				}
				else
				{
					for(j=m;j>=1;j--)
						printf("%d %d\n",i,j);
				}
			}
			printf("1 1\n");
		}
		}
	}
	return 0;
}
