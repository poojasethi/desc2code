#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;

const int N = 105;

int x[N * N],y[N * N];
int n,m,tot;

void spe(){
    printf("1\n");
    printf("%d %d %d %d\n",n,m,1,1);
    for(int i = 1;i <= n;i ++){
        if(i & 1){
            for(int j = 1;j <= m;j ++)
                printf("%d %d\n",i,j);
        }
        else{
            for(int j = m;j >= 1;j --)
                printf("%d %d\n",i,j);
        }
    }
    printf("1 1\n");
}

void add(int a,int b){
    x[tot] = a;
    y[tot ++] = b;
}

void cal(){
    add(1,1);
    for(int i = 1;i <= n;i ++){
        // printf("i:%d\n",i);
        if(i & 1){
            for(int j = 2;j <= m;j ++)
                add(i,j);
        }
        else{
            for(int j = m;j >= 2;j --)
                add(i,j);
        }
    }
    for(int i = n;i >= 1;i --)
        add(i,1);
}

int main(){
    // freopen("input.txt","r",stdin);
    scanf("%d%d",&n,&m);
    bool flag = false;
    if(((n & 1) && (m & 1)) || (n == 1 && m != 2) || (m == 1 && n != 2)) {
        spe();
        return 0;
    }
    else if(n & 1) swap(n,m) , flag = true;
    cal();
    printf("0\n");
    for(int i = 0;i < tot;i ++)
        if(flag) printf("%d %d\n",y[i],x[i]);
        else printf("%d %d\n",x[i],y[i]);
    return 0;
}
