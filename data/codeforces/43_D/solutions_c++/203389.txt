#include <cstdlib>
#include <iostream>


using namespace std;

int main(int argc, char *argv[])
{
    int n, m;
    cin>>n>>m;
    if(n==1 && m==2){
        cout<<0<<endl;
        cout<<"1 1\n1 2\n1 1\n";
    }else if(n==2 && m==1){
        cout<<0<<endl;
        cout<<"1 1\n2 1\n1 1\n";
    }else if(n%2==0 && m>1){
        cout<<0<<endl;
        cout<<"1 1\n";
        for(int i=1;i<=n;i++){
            if(i%2==1){
                for(int j=2;j<=m;j++) cout<<i<<" "<<j<<endl;
            }else{
                for(int j=m;j>=2;j--) cout<<i<<" "<<j<<endl;
            }
        }
        for(int i=n;i>=1;i--) cout<<i<<" 1"<<endl;
    }else if(m%2==0 && n>1){
        cout<<0<<endl;
        cout<<"1 1\n";
        for(int i=1;i<=m;i++){
            if(i%2==1){
                for(int j=2;j<=n;j++) cout<<j<<" "<<i<<endl;
            }else{
                for(int j=n;j>=2;j--) cout<<j<<" "<<i<<endl;
            }
        }
        for(int i=m;i>=1;i--) cout<<"1 "<<i<<endl;
    }else{
        cout<<1<<endl;
        cout<<n<<" "<<m<<" 1 1"<<endl;
        for(int i=1;i<=m;i++){
            if(i%2==1){
                for(int j=1;j<=n;j++) cout<<j<<" "<<i<<endl;
            }else{
                for(int j=n;j>=1;j--) cout<<j<<" "<<i<<endl;
            }
        }
        cout<<"1 1\n";
    }

    return EXIT_SUCCESS;
}
