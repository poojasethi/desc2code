#include <iostream>
#include <set>
#include <map>
#include <vector>
#include <cmath>
#include <algorithm>

using namespace std;


int main(){
	int m,n,a,b;
	bool s = false;
	cin >> n >> m;
	if((n!=1 && m!=1) || n*m==2){
		if(n%2==1 && m%2==1){
			cout << 1 << endl << n << " " << m << " 1 1" << endl;
			for(int i=1;i<=n;i++){
				for(int j=1;j<=m;j++){
					cout << i << " " << (i%2 ? j : (m+1 - j));
					cout << endl;

				}
			}
			cout << "1 1" << endl;
		}
		else {
			cout << 0 << endl;
			if(n%2==1){
				swap(m,n);
				s=true;
			}
			cout << "1 1" << endl;
			for(int i=1;i<=n;i++){
				for(int j=1;j<=m-1;j++){
					a=i;
					b = 1 + (i%2?j:(m-j));
					if(s)swap(a,b);
					cout << a << " " << b << endl;
				}
			}
			for(int i=n;i>=1;i--){
				a=i;
				b=1;
				if(s)swap(a,b);
				cout << a << " " << b << endl;
			}
		}
	}
	else {
		cout << 1 << endl;
		cout << n<<" " <<m << " 1 1" << endl;
		if(m!=1){
			swap(n,m);
			s=true;
		}
		
		for(int i=1;i<=n;i++){
			a=i;
			b=1;
			if(s)swap(a,b);
			cout << a << " " << b<<endl;
		}
		cout <<"1 1" << endl;
	}
	return 0;
}
