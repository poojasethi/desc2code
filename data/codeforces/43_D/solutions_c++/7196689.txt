#include <bits/stdc++.h>
using namespace std;
int n,m;
int main(){
    //freopen("a.inp","r",stdin);
    scanf("%d%d",&n,&m);
    if( (n&1) && (m&1) || ((n == 1 || m == 1) && n*m > 2) ){
        printf("1\n");
        printf("%d %d 1 1\n",n,m);
        for(int i = 1;i <= n;++i){
            if(i&1) for(int j = 1;j <= m;++j) printf("%d %d\n",i,j);
            else for(int j = m;j > 0;--j) printf("%d %d\n",i,j);
        }
        printf("1 1\n");
    }
    else{
        printf("0\n");
        if(m%2 == 0){
            for(int i = 1;i <= m;++i) printf("1 %d\n",i);
            for(int i = 2;i <= n;++i) printf("%d %d\n",i,m);
            int cnt = 0;
            for(int i = m-1;i > 0;--i){
                cnt++;
                if(cnt&1) for(int j = n;j >= 2;--j) printf("%d %d\n",j,i);
                else for(int j = 2;j <= n;++j) printf("%d %d\n",j,i);
            }
            printf("1 1\n");
        }
        else{
            for(int i = 1;i <= n;++i) printf("%d 1\n",i);
            for(int i = 2;i <= m;++i) printf("%d %d\n",n,i);
            int cnt = 0;
            for(int i = n-1;i > 0;--i){
                cnt++;
                if(cnt&1) for(int j = m;j >= 2;--j) printf("%d %d\n",i,j);
                else for(int j = 2;j <= m;++j) printf("%d %d\n",i,j);
            }
            printf("1 1\n");
        }
    }
return 0;
}
