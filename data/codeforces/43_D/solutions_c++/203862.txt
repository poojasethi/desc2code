#include <iostream>
using namespace std;

int main()
{
	int n,m;
	cin >> n >> m;
	if (n%2==1&&m%2==1&&(m!=1)&&(n!=1))
	{
		cout << 1 << endl << "1 " << m << " "<< "1 1" << endl;
	}
	else 
	if (n>2&&m==1)
	{
		cout << 1 << endl;
		cout << n << " 1" << " 1 1" << endl;
		for (int i=0;i<n;i++)
			cout << i+1 << " 1" << endl;
		cout << "1 1" << endl;
		return 0;
	}
	else
	if (m>2&&n==1)
	{
		cout << 1 << endl;
		cout << "1 " << m << " 1 1" << endl;
		for (int i=0;i<m;i++)
			cout << "1 " << i+1 << endl;
		cout << "1 1" << endl;
		return 0;
	}
	else 
		cout << 0 << endl;
	for (int i=0;i<n;i++)
		cout << i+1  << " 1" << endl;
	for (int j=1;j<m;j++)
		cout << n << " "<< j+1 << endl;
	
	if (n%2==0||(m%2==1&&n%2==1))
	for (int i=n-2;i>=0;i--)
		for (int j=m-1;j>0;j--)	
			cout << i+1 << " " << ((n-i)%2==0?j+1:m-j+1)<< endl;
	else
		for (int j=m-1;j>0;j--)
			for (int i=n-2;i>=0;i--)	
				cout << ((m-j)%2==1?i+1:n-i-1) << " " << j+1<< endl;
	
	cout << "1 1" << endl;
	return 0;
}
