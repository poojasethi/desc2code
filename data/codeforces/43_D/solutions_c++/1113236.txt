#include <cstdio>

using namespace std;

int main() {
	int n, m;
	scanf("%i%i", &n, &m);
	if(n * m % 2 || n == 1 && m > 2 || m == 1 && n > 2) {
		printf("1\n");
		printf("%i %i 1 1\n", n, m);
		for(int i = 0; i < n; i++) {
			if(i % 2) for(int j = m - 1; j >= 0; j--) printf("%i %i\n", i + 1, j + 1);
			else for(int j = 0; j < m; j++) printf("%i %i\n", i + 1, j + 1);
		}
		printf("1 1\n");
	} else if(m % 2) {
		printf("0\n");
		printf("1 1\n");
		for(int i = 0; i < n; i++) {
			if(i % 2) for(int j = m - 1; j >= 1; j--) printf("%i %i\n", i + 1, j + 1);
			else for(int j = 1; j < m; j++) printf("%i %i\n", i + 1, j + 1);
		}
		for(int i = n - 1; i >= 0; i--) printf("%i 1\n", i + 1);
	} else {
		printf("0\n");
		printf("1 1\n");
		for(int j = 0; j < m; j++) {
			if(j % 2) for(int i = n - 1; i >= 1; i--) printf("%i %i\n", i + 1, j + 1);
			else for(int i = 1; i < n; i++) printf("%i %i\n", i + 1, j + 1);
		}
		for(int j = m - 1; j >= 0; j--) printf("1 %i\n", j + 1);
	}
}
