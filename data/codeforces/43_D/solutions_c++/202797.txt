#include <iostream>
#include <fstream>

using namespace std;

bool inv = false;

int n,m;

void write(int i, int j)
{
	i ++;
	j ++;
	if (inv)
		printf("%d %d\n",j,i);
	else
		printf("%d %d\n",i,j);
}

int main()
{
	cin >> n >> m;
	if (((n % 2 == 0 || m % 2 == 0) && n != 1 && m != 1) || (n + m == 3))
	{
		printf("0\n");
		if (n % 2 == 1)
		{
			swap(n,m);
			inv = true;
		}
		int i = 0;
		int j = 0;
		for (int I = 0; I < n; I ++)
		{
			if (I % 2 == 0)
			{
				while (j < m) 
				{
					write(i,j);
					j ++;
				}
				j --;
				i ++;
			}
			if (I % 2 == 1)
			{
				while (j)
				{
					write(i,j);
					j --;
				}
				j ++;
				i ++;
			}
			if (i == n) i --;
		}		
		j --;
		while (i >= 0)
		{
			write(i,j);
			i --;
		}
	}
	else
	{
		printf("1\n");
		if (m == 1)
		{
			printf("%d %d %d %d\n",n,m,m,1);
			swap(n,m);
			inv = true;
		}
		else
		{
			printf("%d %d %d %d\n",n,m,n,1);
		}
		int i = 0;
		int j = 0;
		for (int I = 0; I < n; I ++)
		{
			if (I % 2 == 0)
			{
				while (j < m) 
				{
					write(i,j);
					j ++;
				}
				j --;
				i ++;
			}
			if (I % 2 == 1)
			{
				while (j)
				{
					write(i,j);
					j --;
				}
				j ++;
				i ++;
			}
			if (i == n) i --;
		}		
		j = 0;
		while (i >= 0)
		{
			write(i,j);
			i --;
		}

	}	
	return 0;
}
