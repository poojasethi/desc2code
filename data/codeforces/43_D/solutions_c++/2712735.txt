#include<stdio.h>
#include<string.h>
int main()
{
	int n,m;
	while(scanf("%d%d",&n,&m)!=-1)
	{
		int i,j;
		
		if(n==1&&m!=2)
		{
			printf("1\n1 %d 1 1\n",m);
			for(i=1;i<=m;i++)
				printf("1 %d\n",i);
			printf("1 1\n");
			continue;
		}
		if(m==1&&n!=2)
		{
			printf("1\n%d 1 1 1\n",n);
			for(i=1;i<=n;i++)
				printf("%d 1\n",i);
			printf("1 1\n");
			continue;
		}
		if(n%2==0)
		{
			printf("0\n");
			for(i=1;i<=n;i++)
				printf("%d 1\n",i);
			for(i=2;i<=m;i++)
				printf("%d %d\n",n,i);
			for(i=n-1;i>0;i--)
			{
				if(i%2==1)
				for(j=m;j>1;j--)
					printf("%d %d\n",i,j);
				else
				for(j=2;j<=m;j++)
					printf("%d %d\n",i,j);
			}
			printf("1 1\n");
		}
		else if(m%2==0)
		{
			printf("0\n");
			for(j=1;j<=m;j++)
				printf("1 %d\n",j);
			for(j=2;j<=n;j++)
				printf("%d %d\n",j,m);
			for(j=m-1;j>0;j--)
			{
				if(j%2==1)
				for(i=n;i>1;i--)
					printf("%d %d\n",i,j);
				else
				for(i=2;i<=n;i++)
					printf("%d %d\n",i,j);
			}
			printf("1 1\n");
		}
		else
		{
			printf("1\n");
			printf("%d %d 1 1\n",n,m);
			for(i=1;i<=n;i++)
			{
				if(i%2==1)
					for(j=1;j<=m;j++)
						printf("%d %d\n",i,j);
					else
						for(j=m;j>=1;j--)
							printf("%d %d\n",i,j);
			}
			printf("1 1\n");
		}
	}
	return 0;
}
