#include<iostream>
using namespace std;



void fx(int x1, int x2 , int y , int f)
{
    if(f==0){ for(int i = x1 ; i<=x2 ; i++)cout<<i<<" "<<y<<endl; }

    else if(f==1){ for(int i = x2 ; i>=x1 ; i--)cout<<i<<" "<<y<<endl; }
}


void fy(int y1, int y2 , int x , int f)
{
    if(f==0){   for(int i = y1 ; i<=y2 ; i++)cout<<x<<" "<<i<<endl; }

    else if(f==1){   for(int i = y2 ; i>=y1 ; i--)cout<<x<<" "<<i<<endl; }
}


int main()
{
    int m , n;cin>>n>>m;
    if( (n%2==0 && m>1) || (n==2 && m==1) )
    {
        cout<<0<<endl<<"1 1"<<endl;
        int f=0;
        for(int i = 1 ; i<=n ; i++)
        {
            fy(2,m,i,f);
            f=(f==0?1:0);
        }
        fx(1,n,1,1);
    }
    else if((m%2==0 && n>1) || (m==2 && n==1))
    {
        cout<<0<<endl<<"1 1"<<endl;
        int f=0;
        for(int i = 1 ; i<=m ; i++)
        {
            fx(2,n,i,f);
            f=(f==0?1:0);
