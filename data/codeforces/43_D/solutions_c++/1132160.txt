#include<iostream>
#include<stdio.h>
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	if(n==1&&m==1)
	{
		printf("0\n");
		printf("1 1\n");
		printf("1 1\n");
	}
	else if(n==1&&m>2)
	{
		printf("1\n");
		printf("%d %d 1 1\n",n,m);
		for(int i=1;i<=m;i++)
			printf("1 %d\n",i);
		printf("1 1\n");
	}
	else if(m==1&&n>2)
	{
		printf("1\n");
		printf("%d %d 1 1\n",n,m);
		for(int i=1;i<=n;i++)
			printf("%d 1\n",i);
		printf("1 1\n");
	}
	else if((n&1)&&(m&1))
	{
		printf("1\n");
		printf("%d %d 1 1\n",n,m);
		for(int i=1;i<=n;i++)
		{
			if(i&1)
			{
				for(int j=1;j<=m;j++)
				{
					printf("%d %d\n",i,j);
				}
			}
			else
			{
				for(int j=m;j>0;j--)
				{
					printf("%d %d\n",i,j);
				}
			}
		}
		printf("1 1\n");
	}
	else
	{
		printf("0\n");
		if(n&1)
		{
			for(int i=1;i<=m;i++)
				printf("%d %d\n",1,i);
			for(int j=0;j<m;j++)
			{
				if(j&1)
				{
					for(int i=n;i>1;i--)
						printf("%d %d\n",i,m-j);
				}
				else
				{
					for(int i=2;i<=n;i++)
						printf("%d %d\n",i,m-j);
				}
			}
			printf("1 1\n");
		}
		else
		{
			for(int i=1;i<=n;i++)
				printf("%d %d\n",i,1);
			for(int j=0;j<n;j++)
			{
				if(j&1)
				{
					for(int i=m;i>1;i--)
						printf("%d %d\n",n-j,i);
				}
				else
				{
					for(int i=2;i<=m;i++)
						printf("%d %d\n",n-j,i);
				}
			}
			printf("1 1\n");
		}
	}
}
