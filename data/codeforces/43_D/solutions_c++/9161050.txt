#include <cstdio>
using namespace std;

typedef long long li;	

int main() {
	int n, m, k;
	
	scanf("%d %d", &n, &m);
	if ((n % 2 == 0) && (m >= 2)) {
		printf("0\n");
		printf("1 1\n");
		for (int i = 1; i <= n;) {
			for (int j = 2; j <= m; j++) {
				printf("%d %d\n", i, j);
			}
			i++;
			for (int j = m; j >= 2; j--) {
				printf("%d %d\n", i, j);
			}
			i++;
		}
		for (int i = n; i >= 1; i--)
			printf("%d 1\n", i);
	}
	else if ((m % 2 == 0) && (n >= 2)) {
		printf("0\n");
		printf("1 1\n");
		for (int j = 1; j <= m;) {
			for (int i = 2; i <= n; i++) {
				printf("%d %d\n", i, j);
			}
			j++;
			for (int i = n; i >= 2; i--) {
				printf("%d %d\n", i, j);
			}
			j++;
		}
		for (int j = m; j >= 1; j--)
			printf("1 %d\n", j);
	}
	else {
		if ((n == 1) && (m == 1)) printf("0\n");
		else if (((n == 1) && (m == 2)) || ((n == 2) && (m == 1))) printf("0\n");
		else {
			printf("1\n");
			printf("%d %d 1 1\n", n, m);
		}
		for (int i = 1; i <= n;) {
			for (int j = 1; j <= m; j++)
				printf("%d %d\n", i, j);
			i++;
			if (i > n) break;
			for (int j = m; j >= 1; j--)
				printf("%d %d\n", i, j);
			i++;
		}
		printf("1 1");
	}
	return 0;
}
