#include <bits/stdc++.h>
using namespace std;

int main() {

    int n, m;

    // freopen("D:\\1.in", "r", stdin);
    // freopen("D:\\1.out", "w", stdout);

    scanf("%d%d", &n, &m);
    if(n == 1 || m == 1)
    {
        bool rf = true;
        if(n != 1) rf = false, swap(n, m);
        if(m == 2) printf("0\n"); else printf("1\n%d %d %d %d\n", rf ? n : m, rf ? m : n, 1, 1);
        for(int i = 1; i <= m; i ++) printf("%d %d\n", rf ? n : i, rf ? i : n);
        printf("1 1\n");
    }
    else if(n % 2 == 1 && m % 2 == 1)
    {
        printf("1\n%d %d %d %d\n", n, m, 1, 1);
        for(int i = 1; i <= n; i ++)
        if(i % 2 == 1) for(int j = 1; j <= m; j ++) printf("%d %d\n", i, j);
        else for(int j = m; j >= 1; j --) printf("%d %d\n", i, j);
        printf("1 1\n");
    }
    else
    {
        bool rf = true;
        if(n % 2 == 1) rf = false, swap(n, m);
        printf("0\n1 1\n");
        for(int i = 1; i <= n; i ++)
        {
            for(int j = 2; j <= m; j ++) printf("%d %d\n", rf ? i : j, rf ? j : i);
            i ++;
            for(int j = m; j >= 2; j --) printf("%d %d\n", rf ? i : j, rf ? j : i);
        }
        for(int i = n; i >= 1; i --) printf("%d %d\n", rf ? i : 1, rf ? 1 : i);
    }

	return 0;
}
