#include<iostream>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	if(n%2&&m%2||n==1&&m!=2||m==1&&n!=2)
	{
		cout<<1<<endl<<n<<" "<<m<<" "<<1<<" "<<1<<endl;
		for(int i=1;i<=n;i++)
		{
			if(i%2)for(int j=1;j<=m;j++)cout<<i<<" "<<j<<endl;
			else for(int j=m;j>0;j--)cout<<i<<" "<<j<<endl;
		}
		cout<<1<<" "<<1<<endl;
	}
	else if(n%2==0)
	{
		cout<<0<<endl<<1<<" "<<1<<endl;
		for(int i=1;i<=n;i++)
		{
			if(i%2)for(int j=2;j<=m;j++)cout<<i<<" "<<j<<endl;
			else for(int j=m;j>1;j--)cout<<i<<" "<<j<<endl;
		}
		for(int i=n;i>=1;i--)cout<<i<<" "<<1<<endl;
	}
	else
	{
		cout<<0<<endl<<1<<" "<<1<<endl;
		for(int i=1;i<=m;i++)
		{
			if(i%2)for(int j=2;j<=n;j++)cout<<j<<" "<<i<<endl;
			else for(int j=n;j>1;j--)cout<<j<<" "<<i<<endl;
		}
		for(int i=m;i>=1;i--)cout<<1<<" "<<i<<endl;
	}
}
