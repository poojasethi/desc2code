#include <cstdio>
#include <algorithm>
using namespace std;
bool swapped=false;
void out(int x,int y)
{
	if(swapped) printf("%d %d",y,x); else printf("%d %d",x,y);
}
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	if(n==1&&m==1)
	{
		puts("0");
		puts("1 1");
		puts("1 1");
		return 0;
	}
	if(m==1)
	{
		swap(n,m);
		swapped=true;
	}
	if(n==1)
	{
		if(m==2)
		{
			puts("0");
		}
		else
		{
			puts("1");
			out(1,m);
			puts(" 1 1");
		}
		for(int i=1;i<=m;i++)
		{
			out(1,i);
			putchar('\n');
		}
		puts("1 1");
		return 0;
	}
	if((m&1)==0)
	{
		swap(n,m);
		swapped=true;
	}
	if((n&1)==0)
	{
		puts("0");
		for(int i=1;i<=n;i++)
		{
			out(i,1);
			putchar('\n');
		}
		for(int i=n;i>0;i-=2)
		{
			for(int j=2;j<=m;j++)
			{
				out(i,j);
				putchar('\n');
			}
			for(int j=m;j>=2;j--)
			{
				out(i-1,j);
				putchar('\n');
			}
		}
		puts("1 1");
		return 0;
	}
	puts("1");
	out(n,m);
	puts(" 1 1");
	for(int i=1;i<=n;i++)
	{
		if(i&1)
		{
			for(int j=1;j<=m;j++)
			{
				out(i,j);
				putchar('\n');
			}
		}
		else
		{
			for(int j=m;j>0;j--)
			{
				out(i,j);
				putchar('\n');
			}
		}
	}
	puts("1 1");
	return 0;
}
