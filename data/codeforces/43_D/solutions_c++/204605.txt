#include <iostream>
#include <cstdio>

using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    if((m == 1 || n == 1) && (n*m > 2)) {
        cout << "1\n" << n << " " << m << " 1 1\n";
        for(int i=0; i<max(n, m); ++i)
            cout << ((n == 1)?(1):(i+1)) << " " << ((m == 1)?(1):(i+1)) << endl;
        cout << "1 1\n";
    }
    else if((m == 1 || n == 1) && (n*m == 2)){
        cout << "0\n1 1\n" << n << " " << m << "\n1 1\n";
    }
    else if(n%2 && m%2) {
        cout << "1\n" << n << " " << m << " 1 1\n";
        for(int i=0; i<n; ++i)
            for(int j=0; j<m; ++j)
                cout << i+1 << " " << ((i%2)?(m-j):(j+1)) << endl;
        cout << "1 1\n";
    }
    else if(n%2) {
        cout << "0\n1 1\n";
        for(int j=0; j<m; ++j)
            for(int i=1; i<n; ++i)
                cout << ((j%2)?(n-i+1):(i+1)) << " " << j+1 << endl;
        for(int j=0; j<m; ++j)
            cout << "1 " << m-j << endl;
    }
    else {
        cout << "0\n1 1\n";
        for(int i=0; i<n; ++i)
            for(int j=1; j<m; ++j)
                cout << i+1 << " " << ((i%2)?(m-j+1):(j+1)) << endl;
        for(int i=0; i<n; ++i)
            cout << n-i << " 1\n";
    }
    
    return 0;
}
