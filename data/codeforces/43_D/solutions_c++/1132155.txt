#include<stdio.h>

int main()
{
	int n,m;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		if((n%2&&m%2)||(n==1&&m!=2)||(n!=2&&m==1))
		{
			printf("1\n");
			printf("%d %d 1 1\n",n,m);
			for(int i=1;i<=n;i++)
				if(i%2)
					for(int j=1;j<=m;j++) printf("%d %d\n",i,j);
				else
					for(int j=m;j>=1;j--) printf("%d %d\n",i,j);
			printf("1 1\n");
		}
		else if(n%2==0)
		{
			printf("0\n");
			for(int i=1;i<=m;i++) printf("1 %d\n",i);
			for(int j=2;j<=n;j++) printf("%d %d\n",j,m);
			for(int i=n;i>=2;i--)
				if(i%2)
					for(int j=1;j<m;j++) printf("%d %d\n",i,j);
				else
					for(int j=m-1;j>=1;j--) printf("%d %d\n",i,j);	
			printf("1 1\n");
		}
		else
		{
			printf("0\n");
			for(int i=1;i<=n;i++) printf("%d 1\n",i);
			for(int j=2;j<=m;j++) printf("%d %d\n",n,j);
			for(int i=m;i>=2;i--)
				if(i%2)
					for(int j=1;j<n;j++) printf("%d %d\n",j,i);
				else
					for(int j=n-1;j>=1;j--) printf("%d %d\n",j,i);
			printf("1 1\n");
		}
	}	
	return 0;
}
