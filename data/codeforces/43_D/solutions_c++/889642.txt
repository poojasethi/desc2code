#include <stdio.h>

int main(void) {
	int n, m;
	scanf("%d%d", &n, &m);
	if(n==1 && m==2) {
		printf("0\n1 1\n1 2\n1 1\n");
	} else if(n==2 && m==1) {
		printf("0\n1 1\n2 1\n1 1\n");
	} else if(n==1 || m==1 || n*m%2) {
		printf("1\n");
		if(n%2) printf("%d %d 1 1\n", n, m);
		else printf("%d 1 1 1\n", n);
		for(int i=1;i<=n;i++)
			if(i%2)
				for(int j=1;j<=m;j++)
					printf("%d %d\n", i, j);
			else
				for(int j=m;j>=1;j--)
					printf("%d %d\n", i, j);
		printf("1 1\n");
	} else if(n%2==0) {
		printf("0\n");
		for(int j=1;j<=m;j++) printf("%d %d\n", 1, j);
		for(int i=2;i<=n;i++) printf("%d %d\n", i, m);
		for(int i=n;i>=2;--i)
			if(i%2)
				for(int j=1;j<=m-1;j++) printf("%d %d\n", i, j);
			else
				for(int j=m-1;j>=1;j--) printf("%d %d\n", i, j);
		printf("1 1\n");
	} else {
		printf("0\n");
		for(int i=1;i<=n;i++) printf("%d %d\n", i, 1);
		for(int j=2;j<=m;j++) printf("%d %d\n", n, j);
		for(int j=m;j>=2;j--)
			if(j%2)
				for(int i=1;i<=n-1;i++) printf("%d %d\n", i, j);
			else
				for(int i=n-1;i>=1;i--) printf("%d %d\n", i, j);
		printf("1 1\n");
	}
	return 0;
}
