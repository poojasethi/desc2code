# include <iostream>
# include <string>
# include <map>
using namespace std;
int m,n;
int presx,presy;
bool swp;
void out(int x,int y){
	if(swp) cout<<y<<" "<<x<<endl;	
	else cout<<x<<" "<<y<<endl;	
}
void normal(){
	presx=1,presy=2;
	out(1,1);
	while(1){
		while(presy!=n){
			out(presx,presy);
			presy++;
		}
		out(presx,presy);
		if(presx==m){
			break;
		}
		
		presx++;
		while(presy!=2){
			out(presx,presy);
			presy--;
		}
		out(presx,presy);		
		if(presx==m){
			break;
		}
		
		presx++;
	}
	for (int i = m; i >= 1; i -= 1)
	{
		out(i,1);
	}
}
 
int main(){
	
	cin>>m>>n;
	swp = false;
	if(m==1){
		if(n==2){
			cout<<"0\n";
			cout<<"1 1\n"<<"1 2\n"<<"1 1\n";
		}
		else{
			cout<<"1\n"<<1<<" "<<n<<" "<<1<<" "<<"1\n";
			for (int i = 1; i <= n; i += 1)
			{
				cout<<"1 "<<i<<"\n";				
			}
			cout<<"1 1\n";
		}
	}
	else if(n==1){
		if(m==2){
			cout<<"0\n";
			cout<<"1 1\n"<<"2 1\n"<<"1 1\n";
		}
		else{
			cout<<"1\n"<<m<<" "<<1<<" "<<1<<" "<<"1\n";
			for (int i = 1; i <= m; i += 1)
			{
				cout<<i<<" 1\n";				
			}
			cout<<"1 1\n";
		}
	}
	else if(m%2==0){
		cout<<"0\n";
		normal();
	}
	else if(n%2==0){
		cout<<"0\n";
		swp = true;	
		int tmp = m;
		m = n;
		n = tmp;
		normal();
	}
	else{
		cout<<"1\n";
		cout<<m<<" "<<n<<" "<<m<<" 1\n";		
		normal();
	}	
}
