#include <bits/stdc++.h>
using namespace std;

int n, m;
int main(){
	scanf("%d%d", &n, &m);
	if (n == 1 && m == 2){
		printf("0\n1 1\n1 2\n1 1\n");
		return 0;
	}
	if (m == 1 && n == 2){
		printf("0\n1 1\n2 1\n1 1\n");
		return 0;
	}
	if (n == 1){
		printf("1\n1 %d 1 1\n", m);
		for (int i = 1; i <= m; ++i)
			printf("%d %d\n", 1, i);
		printf("1 1\n");
		return 0;
	}
	if (m == 1){
		printf("1\n%d 1 1 1\n", n);
		for (int i = 1; i <= n; ++i)
			printf("%d %d\n", i, 1);
		printf("1 1\n");
		return 0;
	}
	if (n % 2 == 0){
		printf("0\n");
		for (int i = 1; i <= n; ++i) printf("%d 1\n", i);
		for (int i = n; i >= 1; --i){
			if (i % 2 == 0)
				for (int j = 2; j <= m; ++j) printf("%d %d\n", i, j);
			else for (int j = m; j >= 2; --j) printf("%d %d\n", i, j);
		}
		printf("1 1\n");
		return 0;
	}
	if (m % 2 == 0){
		printf("0\n");
		for (int i = 1; i <= m; ++i) printf("1 %d\n", i);
		for (int i = m; i >= 1; --i){
			if (i % 2 == 0)
				for (int j = 2; j <= n; ++j) printf("%d %d\n", j, i);
			else for (int j = n; j >= 2; --j) printf("%d %d\n", j, i);
		}
		printf("1 1\n");
		return 0;
	}
	printf("1\n%d %d %d %d\n", n, m, 1, 1);
	for (int i = 1; i <= n; ++i){
		if (i % 2 == 1)
			for (int j = 1; j <= m; ++j) printf("%d %d\n", i, j);
		else for (int j = m; j >= 1; --j) printf("%d %d\n", i, j);
	}
	printf("1 1\n");
	return 0;
}
