#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int n, m, s, d, t, cur;
vector<int> a, b, h, pr, ans;
int main()
{
	cin >> n >> m >> s >> d;
	if (d==1)
	{
		cout<<"IMPOSSIBLE\n";
		return 0;
	}
	a.resize(n);
	for(int i=0;i<n;++i)
		cin>>a[i];
	sort(a.begin(), a.end());
	b.resize(n+1, 0);
	pr.resize(n+1, 0);
	h.resize(n+1, 0);
	if (a[0]>s)
	{
		b[0]=1;
		pr[0]=-1;
	}
	for(int i=0; i<n; ++i)
		if (b[i])
		{
			cur = a[i]-1;
			while(t<n && cur+d>a[t])
			{
				if (t==n-1 || a[t+1]-a[t]>s+1)
				{
					b[t+1]=1;
					h[t+1]=a[t]+1-cur;
					pr[t+1] = i;
				}
				++t;
			}
		}
	if (b[n])
	{
		t=n;
		cur = m;
		while(t>0)
		{
			if (cur-a[t-1]-1>0)
				ans.push_back(cur-a[t-1]-1);
			ans.push_back(h[t]);
			t = pr[t];
			cur = a[t]-1;
		}
		ans.push_back(cur);
		reverse(ans.begin(), ans.end());
		for(int i = 0; i<ans.size(); ++i)
			if (i&1)
				cout << "JUMP " << ans[i] << endl;
			else cout << "RUN " << ans[i] << endl;
	}
	else cout<<"IMPOSSIBLE\n";
	return 0;
}
