#include <bits/stdc++.h>
using namespace std;
const int N=200005;
int a[N], c[N*5], b[N*5];
int main() {
	int n, m, s, d;
	scanf("%d%d%d%d", &n, &m, &s, &d);
	for(int i=1; i<=n; ++i) {
		scanf("%d", &a[i]);
	}
	sort(a+1, a+1+n);
	int now=0, tot=0, i=1;
	for(i=1; i<=n; ++i) {
		if(a[i]-1-now<s) {
			break;
		}
		c[++tot]=1;
		b[tot]=a[i]-1-now;
		now=a[i]-1;
		int j;
		for(j=i; j<=n; ++j) {
			if(a[j]+1-now<=d && (j==n || (j<n && a[j]+1+s<a[j+1]))) {
				break;
			}
			// printf("no :%d %d %d, a[j]:%d, s:%d, a[j+1]:%d\n", j, a[j]+1-now<=d, a[j]+1+s<a[j+1], a[j], s, a[j+1]);
		}
		if(j>n) {
			break;
		}
		// printf("i:%d j:%d, now:%d\n", i, j, now);
		c[++tot]=0;
		b[tot]=a[j]+1-now;
		now=a[j]+1;
		i=j;
	}
	if(i!=n+1 || now>m) {
		puts("IMPOSSIBLE");
	}
	else {
		for(int i=1; i<=tot; ++i) {
			printf("%s ", c[i]?"RUN":"JUMP");
			printf("%d\n", b[i]);
		}
		if(m!=now) {
			printf("RUN %d\n", m-now);
		}
	}
	return 0;
}
