#include <bits/stdc++.h>
using namespace std;

const int N = (int)3e5;

int main( void )
{
    int n,m,s,d;
    int x[N];
    int curx;
    vector< pair<string,int> > ans;

    cin >> n >> m >> s >> d;
    for( int i = 0; i < n; i++ )
        cin >> x[i];
    sort(x,x+n);
    curx = x[0]-1;
    if( curx < s ) {
        cout << "IMPOSSIBLE";
        return 0;
    }
    ans.push_back(make_pair("RUN ",curx));
    for( int i = 1; i < n; i++ ) {
        if( curx+d <= x[i-1] ) {
            cout << "IMPOSSIBLE";
            return 0;
        }
        if( x[i]-x[i-1]-1>s ) {
            ans.push_back(make_pair("JUMP ",(x[i-1]-curx+1)));
            curx = x[i]-1;
            ans.push_back(make_pair("RUN ",(curx-x[i-1]-1)));
        }
    }
    if( curx+d <= x[n-1] ) {
        cout << "IMPOSSIBLE";
        return 0;
    }
    ans.push_back(make_pair("JUMP ",x[n-1]-curx+1));
    if( m-x[n-1]-1 )
        ans.push_back(make_pair("RUN ",(m-x[n-1]-1)));
    for( unsigned int i = 0; i < ans.size(); i++ )
        cout << ans[i].first << ans[i].second << endl;

    return 0;
}
