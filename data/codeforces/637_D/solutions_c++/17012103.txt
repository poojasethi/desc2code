#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;
int main() {
	int n,m,s,d;
	scanf("%d%d%d%d",&n,&m,&s,&d);
	vector <int> a(n+1,-1);
	vector <int> b(0);
	for (int i=0;i<n;i++) scanf("%d",&a[i]);
	sort(a.begin(),a.end());
 
	bool f;
	if (a[1] <= s) f = false;
	else f = true;
	for (int i=0;i<n;i++) {
		if (a[i+1]-a[i] >= s+2) {
			b.push_back(a[i]+1);
			b.push_back(a[i+1]-1);
		}
	}
	b.push_back(a[n]+1);
	for (int i=0;i<(int)b.size()/2;i++) {
		if (b[2*i+2]-b[2*i+1] > d) f = false;
	}
	if (f) {
		for (int i=0;i<(int)b.size()/2;i++) {
			printf("RUN %d\n",b[2*i+1]-b[2*i]);
			printf("JUMP %d\n",b[2*i+2]-b[2*i+1]);
		}
		if (a[n]+1 < m) printf("RUN %d\n",m - (a[n]+1));
	} else printf("IMPOSSIBLE\n");
	return 0;
}
