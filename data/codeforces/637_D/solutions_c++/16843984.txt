#include <bits/stdc++.h>
using namespace std;
#define N 200009
int n,m,s,d,cur;
int a[N];
vector<int> v,run,jump;
int main() 
{
	ios_base::sync_with_stdio(false);cin.tie(NULL);
	cin>>n>>m>>s>>d;
	for(int i=1;i<=n;i++)cin>>a[i];
	sort(a+1,a+n+1);
	a[0]=-1;
	for(int i=1;i<=n;i++)
	{
		if(a[i]-a[i-1]-2>=s)v.push_back(i-1);
	}
	if(m-a[n]-1>=s)v.push_back(n);//can end with a run
	else v.push_back(-1);// can NOT end with a run
	//for(int i=0;i<v.size();i++)cout<<v[i]<<" ";cout<<endl;
	if(v[0]!=0){cout<<"IMPOSSIBLE";return 0;}
	for(int i=0;i<v.size();i++)
	{
		int idx=v[i];
		if(i==v.size()-1 && v[i]!=-1){run.push_back(m-(a[v[i]]+1));break;}
		else if(i==v.size()-1){break;}
		cur=a[idx]+1;
		run.push_back(a[idx+1]-1-cur);
		if(v[i+1]==-1 && m-(a[idx+1]-1)<=d)jump.push_back(m-(a[idx+1]-1));
		else if(v[i+1]!=-1 && a[v[i+1]]+1-(a[idx+1]-1)<=d)jump.push_back(a[v[i+1]]+1-(a[idx+1]-1));
		else {cout<<"IMPOSSIBLE";return 0;}
	}
	int i=0;
	//cout<<run.size()<<" "<<jump.size()<<endl;
	while(i<jump.size())
	{
		cout<<"RUN "<<run[i]<<endl;
		cout<<"JUMP "<<jump[i++]<<endl;
	}
	if(i<run.size())cout<<"RUN "<<run[i]<<endl;
	return 0;
}
