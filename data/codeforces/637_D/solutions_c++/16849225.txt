#include "iostream"
#include "algorithm"
#include "sstream"
#define NMAX 200001
using namespace std;
int n, m, s, d, a[NMAX];
int main()
{
    ios_base::sync_with_stdio(false);
    int i, j=0;
    ostringstream oss;
    cin>> n >> m >> s >> d;
    for (i=0; i<n; i++)
        cin>> a[i];
    if (d==1)
        cout<< "IMPOSSIBLE\n";
    else
    {
        sort(a, a+n);
        i=0;
        while (i<n)
        {
            if (a[i]-1-j<s)
            {
                cout<< "IMPOSSIBLE\n";
                return 0;
            }
            else
            {
                oss<< "RUN " << a[i]-1-j << "\n";
                j=a[i]-1;
                while ((a[i+1]+1-j<=d)&&(i<n-1)&&(a[i+1]-a[i]-2<s))
                    i++;
                oss<< "JUMP " << a[i]+1-j << "\n";
                j=a[i]+1;
                ++i;
            }
            
        }
        if (j<m)
            oss<< "RUN " << m-j << "\n";
        cout<< oss.str();
    }
    return 0;
}
