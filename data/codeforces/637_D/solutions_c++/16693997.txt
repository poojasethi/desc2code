#include<cstdio>
#include<algorithm>
#include<cstring>
#include<map>
#include<iostream>
#include<string>
using namespace std;


int n,m,s,d;
int a[200005];
int dp[200005];

void dfs(int u){
	if(!u)return;

	dfs(dp[u]);
	int ra=a[dp[u]+1]-1-(a[dp[u]]+1),ja=a[u]+1-(a[dp[u]+1]-1);
	if(ra)printf("RUN %d\n",ra);
	if(ja)printf("JUMP %d\n",ja);
}
int main()
{
	scanf("%d%d%d%d",&n,&m,&s,&d);
	for (int i=1;i<=n;i++)scanf("%d",&a[i]);
	sort(a+1,a+1+n);
	a[0]=-1;

	int far=0;
	for (int i=0;i<=n;i++){
		if(far<i){
			printf("IMPOSSIBLE\n");
			return 0;
		}
		if(i==n)break;
		int si=a[i+1]-1-(a[i]+1);
		if(si>=s){
			int tar=a[i+1]-1+d;
			while(far<n && tar>a[far+1]){
				dp[far+1]=i;
				far++;
			}
		}
	}
	dfs(n);
	int go=m-a[n]-1;
	if(go)printf("RUN %d\n",go);
	return 0;

}
