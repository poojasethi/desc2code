#include<bits/stdc++.h>
using namespace std;
int jilu[200050];
int ans[200050];
int main()
{
    int n,m,s,d;
    int num=0;
    scanf("%d%d%d%d",&n,&m,&s,&d);
    for(int i=0;i<n;i++){
        scanf("%d",&jilu[i]);
    }
    if(d==1){puts("IMPOSSIBLE");return 0;}
    sort(jilu,jilu+n);
    if(jilu[0]-1<s){puts("IMPOSSIBLE");return 0;}
    ans[num++]=jilu[0]-1;
    for(int i=1;i<n;i++){
        if(jilu[i]-jilu[i-1]-2>=s){
            ans[num++]=jilu[i-1]+1;
            ans[num++]=jilu[i]-1;
        }
    }
    ans[num++]=jilu[n-1]+1;
    for(int i=1;i<num;i+=2){
        if(ans[i]-ans[i-1]>d){
            puts("IMPOSSIBLE");return 0;
        }
    }
    printf("RUN %d\n",ans[0]);
    for(int i=1;i<num;i++){
        if(i&1)printf("JUMP ");
        else printf("RUN ");
        printf("%d\n",ans[i]-ans[i-1]);
    }
    if(ans[num-1]<m){
        printf("RUN %d\n",m-ans[num-1]);
    }
}
