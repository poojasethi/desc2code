#include "stdio.h"
#include "vector"
#include "algorithm"
using namespace std;
int n,m,s,d;
vector<bool> check;
vector<int> v;
bool path(vector<int> x);
void update(bool x,int y);
int main(){
    scanf("%d %d %d %d",&n,&m,&s,&d);
    vector<int> x(n);
    for(int i=0;i<n;i++) scanf("%d",&x[i]);
    sort(x.begin(),x.end());
    if(!path(x)) printf("IMPOSSIBLE\n");
}
bool path(vector<int> x){
    if(x[0]<=s) return false;
    int now=x[0]-1;
    update(false,now);
    for(int i=0;i<n-1;i++){
        if(x[i+1]-(x[i]+1)<=s) continue;
        if(x[i]-now+1>d) return false;
        update(true,x[i]-now+1);
        update(false,x[i+1]-x[i]-2);
        now=x[i+1]-1;
    }
    if(x[n-1]-now+1>d) return false;
    update(true,x[n-1]-now+1);
    if(m-(x[n-1]+1)!=0) update(false,m-(x[n-1]+1));
    for(int i=0;i<check.size();i++){
        if(check[i]) printf("JUMP %d\n",v[i]);
        else printf("RUN %d\n",v[i]);
    }
    return true;
}
void update(bool x,int y){
    check.push_back(x);
    v.push_back(y);
}
