#include <bits/stdc++.h>

using namespace std;

int main() {
  int n, m, s, d;
  cin >> n >> m >> s >> d;
  int a[n];
  for (int i = 0; i < n; i++) scanf("%d", &a[i]);
  sort(a, a + n);

  vector < pair < int, int > > bl;

  int cur = a[0];

  for (int i = 1; i < n; i++)
    if (a[i] - a[i - 1] - 2 < s) {
      continue;
    }
    else {
      bl.push_back(make_pair(cur, a[i - 1]));
      cur = a[i];
    }
	bl.push_back(make_pair(cur, a[n - 1]));

  int rast = 0;
  //
  vector < int > ans;
  //
  for (int i = 0; i < bl.size(); i++) {
    pair < int, int > pr = bl[i];
    //
    if (pr.first - rast - 1 < s) {
      printf("IMPOSSIBLE");
      return 0;
    }
    //
    if (pr.second - pr.first + 2 > d) {
      printf("IMPOSSIBLE");
      return 0;
    }
    //
    ans.push_back(pr.first - rast - 1);
    ans.push_back(pr.second - pr.first + 2);
    //
    rast = pr.second + 1;
  }

  if (m - rast > 0)
    ans.push_back(m - rast);

  for (int i = 0; i < ans.size(); i++)
    if (i % 2 == 0)
      printf("RUN %d\n", ans[i]);
    else
      printf("JUMP %d\n", ans[i]);

  return 0;
}
