#include <bits/stdc++.h>
using namespace std;
#define fi first
#define se second
#define mp make_pair

int a[200002];
pair <int, int> run[200002];
int main()
{
	cin.tie(NULL);
	ios_base::sync_with_stdio(false);
	int n, m, s, d;
	cin >> n >> m >> s >> d;
	for(int i = 0; i < n; ++i)
	{
		cin >> a[i];
	}
	sort(a, a + n);
	int t = 0;
	run[t].fi = a[0] - 1;
	if(run[t].fi < s)
	{
		cout << "IMPOSSIBLE";
		return 0;
	}
	
	for(int i = 1; i < n; ++i)
	{
		if(a[i] - a [i - 1] - 2 >= s)
		{
			run[t].se = a[i - 1] + 1;
			if(run[t].se - run[t].fi > d)
			{
				cout << "IMPOSSIBLE";
				return 0;
			}
			++t;
			run[t].fi = a[i] - 1;
		}
	}
	
	if(run[t].fi + d <= a[n - 1])
	{
		cout << "IMPOSSIBLE";
		return 0;
	}
	
	cout << "RUN " << run[0].fi << "\n";
	for(int i = 1; i <= t; ++i)
	{
		cout << "JUMP " << run[i - 1].se - run[i - 1].fi << "\n";
		cout << "RUN " << run[i].fi - run[i - 1].se<< "\n";
	}
	cout << "JUMP " << a[n - 1] + 1 - run[t].fi << "\n";
	if(m - a[n - 1] - 1)
		cout << "RUN " << m  - a[n - 1] - 1<< "\n";
	return 0;
}
