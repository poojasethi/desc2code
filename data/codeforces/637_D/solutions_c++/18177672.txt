#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<iostream>
#include<algorithm>
#define maxn 200010

using namespace std;

int a[maxn],s,d,m;
int n,f[maxn],q[maxn],fr[maxn],st[maxn];

int main()
{
	scanf("%d%d%d%d",&n,&m,&s,&d);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	a[++n]=-1;
	sort(a+1,a+n+1);
	int l=1,r=0;
	if (a[2]-a[1]-2>=s) q[++r]=1;
	for (int i=2;i<=n;i++)
	{
		while (l<=r && a[i]-a[q[l]+1]>d-2) l++;
		if (l<=r) 
		{
			fr[i]=q[l];
			if (a[i+1]-a[i]-2>=s) q[++r]=i;
		}
	}
	if (!fr[n]) {printf("IMPOSSIBLE\n");return 0;}
	int top=0,x=n;
	while (x!=1) st[++top]=x,x=fr[x];
	st[++top]=1;
	while (top>1)
	{
		printf("RUN %d\n",a[st[top]+1]-a[st[top]]-2);
		printf("JUMP %d\n",a[st[top-1]]+1-a[st[top]+1]+1);
		top--;
	}
	if (m!=a[n]+1) printf("RUN %d\n",m-(a[n]+1));
	return 0;
}
