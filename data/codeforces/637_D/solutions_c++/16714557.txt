#include <cstdio>
#include <sstream>
#include <algorithm>
using namespace std;

const int MAXN = 200000 + 23;

int a[MAXN];

int main()
{
	int n, m, s, d;
	scanf("%d%d%d%d", &n, &m, &s, &d);

	for (int i = 0; i < n; ++i)
	{
		scanf("%d", &a[i]);
	}

	sort(a, a + n);

	if (a[0] <= s)
	{
		printf("IMPOSSIBLE\n");
		return 0;
	}

	stringstream os;
	os << "RUN " << a[0] - 1 << endl;
	int segment_begin = a[0] - 1;
	for (int i = 1; i < n; ++i)
	{
		if (a[i] - a[i-1] - 2 >= s)
		{
			if (a[i - 1] + 1 - segment_begin > d)
			{
				printf("IMPOSSIBLE\n");
				return 0;
			}
			os << "JUMP " << a[i - 1] + 1 - segment_begin << endl;
			os << "RUN " << a[i] - a[i - 1] - 2 << endl;
			segment_begin = a[i] - 1;
		}
	}

	if (a[n - 1] + 1 - segment_begin > d)
	{
		printf("IMPOSSIBLE\n");
		return 0;
	}
	os << "JUMP " << a[n - 1] + 1 - segment_begin << endl;
	if (m - a[n - 1] - 1)
	{
		os << "RUN " << m - a[n - 1] - 1 << endl;
	}

	printf("%s", os.str().c_str());
	
	return 0;
}
