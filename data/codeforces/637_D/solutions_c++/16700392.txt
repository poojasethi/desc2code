#include <bits/stdc++.h>

#define lli long long int
#define sc scanf
#define pr printf
#define pb push_back
#define p_b pop_back
#define mp make_pair
#define fi first
#define se second

using namespace std;

long long int n, i, j, h, finish, s, d, a, b;
vector < int > g;
vector < pair < string, int > > ans;
int start;

int main()
{
    g.pb(start);
    cin >> n >> finish >> s >> d;
    for( i = 1; i <= n; i++ ){
        cin >> a;
        g.pb(a + 1);
    }
    sort( g.begin(), g.end() );
    if( d == 1 ){
        cout << "IMPOSSIBLE" << endl;
        return 0;
    }
    if( g[1] - g[0] - 2 < s ){
        cout << "IMPOSSIBLE" << endl;
        return 0;
    }
    for( i = 0; i < n; i++ ){
        int gg = g[i + 1] - 2;
        ans.pb(mp("RUN", gg - g[i]));
        j = i + 1;
        while( j < n && g[j + 1] - g[j] - 2 < s )j++;
        if( g[j] - gg > d ){
            cout << "IMPOSSIBLE" << endl;
            return 0;
        }
        ans.pb(mp("JUMP", g[j] - gg));
        i = j - 1;
    }
    if( g[n] != finish )ans.pb(mp("RUN", finish - g[n]));
    for( i = 0; i < ans.size(); i++ )cout << ans[i].fi << " " << ans[i].se << endl;
}
