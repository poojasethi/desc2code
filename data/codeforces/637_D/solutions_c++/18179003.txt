#include <bits/stdc++.h>
using namespace std;

int n,m,s,d,top=0,a[200005],b[200005],Free[200005];
queue<int> Queue;

int main()
{
    //freopen("637D.INP","r",stdin);
    cin>>n>>m>>s>>d;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    a[++n]=-1;
    sort(a+1,a+n+1);
    if(a[2]-a[1]-2>=s)
        Queue.push(1);
    for(int i=2;i<=n;i++)
    {
        while(Queue.size()>0&&a[i]-a[Queue.front()+1]+2>d)
            Queue.pop();
        if(Queue.size()>0)
        {
            Free[i]=Queue.front();
            if(a[i+1]-a[i]-2>=s)
                Queue.push(i);
        }
    }
    if(Free[n]==0)
    {
        cout<<"IMPOSSIBLE";
        return 0;
    }
    int x=n;
    while(x!=1)
    {
        b[++top]=x;
        x=Free[x];
    }
    b[++top]=1;
    for(int i=top;i>1;i--)
    {
        cout<<"RUN "<<a[b[i]+1]-a[b[i]]-2<<"\n";
        cout<<"JUMP "<<a[b[i-1]]-a[b[i]+1]+2<<"\n";
    }
    if(m!=a[n]+1)
        cout<<"RUN "<<m-a[n]-1;
}
