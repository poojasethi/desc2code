#include <bits/stdc++.h>
#define ff(i,a,b) for (int i = a; i <= b; i++)
#define gg(i,a,b) for (int i = a; i >= b; i--)
#define pb push_back
#define mp make_pair

using namespace std;
const int maxN = round(2E5) + 2;
int n, m, s, d;
vector < pair<int,int> > suzy;
int a[maxN];
int last_jump;
int x;

int main(){
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  cin >> n >> m >> s >> d;
  ff(i,1,n) cin >> a[i];
  sort(a+1,a+n+1);
  if (a[1]-1-0 < s) return cout << "IMPOSSIBLE", 0;
  x = a[1] + 1;
  suzy.pb(mp(1,a[1]-1));
  last_jump = 2;
  ff(i,2,n){
    if (last_jump > d) return cout << "IMPOSSIBLE", 0;
    if (a[i]-1-x >= s){
      suzy.pb(mp(2,last_jump)); // vuot qua chuong ngai vat thu i-1
      suzy.pb(mp(1,a[i]-x-1));
      last_jump = 2;
      x = a[i] + 1;
    } else{
      last_jump += a[i] + 1 - x;
      x = a[i] + 1;
    }
  }
  if (last_jump > d) return cout << "IMPOSSIBLE", 0;
  suzy.pb(mp(2,last_jump));
  if (x != m) suzy.pb(mp(1,m-x));
  ff(i,0,suzy.size()-1){
    if (suzy[i].first == 1) cout << "RUN " << suzy[i].second << "\n";
    if (suzy[i].first == 2) cout << "JUMP " << suzy[i].second << "\n";
  }
  return 0;
}
