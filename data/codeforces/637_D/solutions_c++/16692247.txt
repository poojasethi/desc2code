#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int N=401000;
int a[N],dis[N],ord[N],tot,n,m,s,d;
bool solve() {
	if (a[0]<=s) return 0;
	int pre=a[0]-1;
	ord[tot]=0; dis[tot]=pre; tot++;
	for (int i=0;i<n-1;i++) {
		if (a[i+1]-a[i]<=s+1) continue;
		if (a[i]+1-pre>d) return 0;
		ord[tot]=1; dis[tot]=a[i]+1-pre; tot++;
		ord[tot]=0; dis[tot]=a[i+1]-a[i]-2; tot++;
		pre=a[i+1]-1;
	}
	if (a[n-1]+1-pre>d) return 0;
	ord[tot]=1; dis[tot]=a[n-1]+1-pre; tot++;
	if (a[n-1]+1!=m) {
		ord[tot]=0; dis[tot]=m-a[n-1]-1; tot++;
	}
	return 1;
}
int main() {
	scanf("%d%d%d%d",&n,&m,&s,&d);
	for (int i=0;i<n;i++) scanf("%d",a+i);
	sort(a,a+n);
	if (!solve()) {
		puts("IMPOSSIBLE");
		return 0;
	}
	for (int i=0;i<tot;i++) {
		printf("%s ",ord[i]?"JUMP":"RUN");
		printf("%d\n",dis[i]);
	}
}
