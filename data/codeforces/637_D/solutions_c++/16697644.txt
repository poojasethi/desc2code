#include <iostream>
#include <set>
#include <algorithm>
#include <map>
#include <cstdio>

using namespace std;

int a[200500];
int b[200500];
int c[200500];

int main()
{
    int n,m,s,d;
    cin >> n >> m >> s >>d;
    for(int i=0;i<n;i++)
        scanf("%d",a+i);
    sort(a,a+n);
    if(a[0]<=s){cout <<"IMPOSSIBLE"<<endl; return 0;}
    int prev = 0;
    b[prev] = a[0];
    c[prev] = a[0];
    for(int i=1;i<n;i++)
    {
        if(a[i] - c[prev]>=s+2)
        {
            prev++;
            b[prev]=c[prev] = a[i];
        }else
            c[prev] = a[i];
    }
    for(int i=0;i<=prev;i++)
        if(c[i] - b[i] + 2 > d) {cout <<"IMPOSSIBLE"<<endl; return 0;}
    int pos = 0;
    for(int i=0;i<=prev;i++)
    {
        if(pos!=b[i]-1) {printf("RUN %d\n",b[i]-1-pos);}
        printf("JUMP %d\n",c[i]-b[i]+2);
        pos = c[i]+1;
    }
    if(pos!=m){printf("RUN %d\n",m-pos);}
    return 0;
}
