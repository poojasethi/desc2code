#define maxn 400005
#define LL long long
int cas=1,T;
int a[maxn],flag[maxn],dist[maxn];
int n,m,s,d,t;
bool solve()
{
	if (a[1]<=s)return false;
	int now=a[1]-1;
	flag[t]=0,dist[t]=now,t++;
	for (int i = 1;i<n;i++)
	{
		if (a[i+1]-a[i]<=s+1)continue;
		if (a[i]+1-now>d)return false;
		flag[t]=1,dist[t]=a[i]+1-now,t++;
		flag[t]=0,dist[t]=a[i+1]-a[i]-2,t++;
		now=a[i+1]-1;
	}
	if (a[n]+1-now>d)
		return false;
	flag[t]=1,dist[t]=a[n]+1-now,t++;
	if (a[n]+1!=m)
		flag[t]=0,dist[t]=m-a[n]-1,t++;
	for (int i = 0;i<t;i++)
	{
		if (flag[i])
			printf("JUMP ");
		else
			printf("RUN ");
		printf("%d\n",dist[i]);
	}
	return true;
}
int main()
{
	while (scanf("%d%d%d%d",&n,&m,&s,&d)!=EOF)
	{
		for (int i = 1;i<=n;i++)
			scanf("%d",&a[i]);
		sort(a+1,a+1+n);
		if (!solve())
			puts("IMPOSSIBLE");
	}
	//freopen("in","r",stdin);
	//scanf("%d",&T);
	//printf("time=%.3lf",(double)clock()/CLOCKS_PER_SEC);
	return 0;
}
