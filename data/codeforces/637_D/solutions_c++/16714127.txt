#include <bits/stdc++.h>
using namespace std;

vector <int> v;
vector <int> gr[200010];
vector <int> tmp;

int main() {
	ios::sync_with_stdio(false);
	int n, m, s, d;
	cin >> n >> m >> s >> d;
	for (int i=0; i<n; i++) {
		int x;
		cin >> x;
		v.push_back(x);
	}
	sort(v.begin(), v.end());
	int cnt=0;
	if (v[0]-1 < s) {
		cout << "IMPOSSIBLE\n";
		return 0;
	}
	tmp.push_back(v[0]);
	for (int i=1; i<n; i++) {
		if (v[i] - tmp[tmp.size()-1] - 2 < s) {
			tmp.push_back(v[i]);
		}
		else {
			gr[cnt++] = tmp;
			tmp.clear();
			tmp.push_back(v[i]);
		}
	}
	gr[cnt++] = tmp;
	tmp.clear();
	for (int i=0; i<cnt; i++) {
		if (gr[i][gr[i].size()-1]-gr[i][0] + 2 > d) {
			cout << "IMPOSSIBLE\n";
			return 0;
		}
	}
	int last = 0;
	for (int i=0; i<cnt; i++) {
		cout << "RUN " << gr[i][0]-1-last << "\n";
		cout << "JUMP " << gr[i][gr[i].size()-1]-gr[i][0] + 2 << "\n";
		last = gr[i][gr[i].size()-1]+1;
	}
	if (last!=m) cout << "RUN " << m-last << "\n";
	return 0;
}
