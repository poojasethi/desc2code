#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
using namespace std;

struct dao
{
    int l,r;
};

vector<dao>DAO;
int sb[200005];

void print()
{
    int sz=DAO.size();
    for(int i=0;i<sz-1;i++)
        printf("RUN %d\nJUMP %d\n",DAO[i].r-DAO[i].l,DAO[i+1].l-DAO[i].r);
    if(DAO[sz-1].l!=DAO[sz-1].r)
        printf("RUN %d\n",DAO[sz-1].r-DAO[sz-1].l);
}

int main()
{
    int n,m,s,d;
    cin>>n>>m>>s>>d;
    int last=-1;
    int now;
    struct dao temp;
    for(int i=0;i<n;i++)
        scanf("%d",&sb[i]);
    sort(sb,sb+n);
    for(int i=0;i<n;i++)
    {
        now=sb[i];
        if(now-last-2<s){last=now;continue;}
        temp.l=last+1;
        temp.r=now-1;
        DAO.push_back(temp);
        last=now;
    }
    temp.l=last+1;
    temp.r=m;
    DAO.push_back(temp);
    bool ok=true;
    int sz=DAO.size();
    if(DAO[0].l!=0){printf("IMPOSSIBLE\n");return 0;}
    for(int i=1;i<sz;i++)
    {
        if(DAO[i-1].r+d<DAO[i].l)
        {
            ok=false;
            break;
        }
    }
    if(ok)
        print();
    else
        printf("IMPOSSIBLE\n");
    return 0;
}
