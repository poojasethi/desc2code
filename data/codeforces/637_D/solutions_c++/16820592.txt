#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std;

const int maxn=200020;

int a[maxn],ans[maxn],dp[maxn];

int main(){
    int n, m ,s, d;
    scanf("%d%d%d%d",&n,&m,&s,&d);
    for(int i=1;i<=n;i++) 
        scanf("%d",&a[i]);
    sort(a+1,a+n+1);
    memset(dp,0,sizeof(dp));
    if(a[1]<=s) dp[1]=1;
    for(int i=2;i<=n;i++) 
    	if(a[i]-a[i-1]-2<s) dp[i]=1;
    bool ok=1;
    int res=0,st=0;
        for(int i=1;i<=n&&ok;i++){
            int l=a[i]-st;
            if(l<=s) ok=0;
            else{
                ans[res++]=a[i]-1-st;
                st=a[i]-1;
                while(i<n&&dp[i+1]) i++;
                if(a[i]+1-st>d) ok=0;
                ans[res++]=a[i]+1-st;
                st=a[i]+1;
            }
        }
        if(!ok) printf("IMPOSSIBLE\n");
        else{
            for(int i=0;i<res;i++){
                if(i%2) printf("JUMP %d\n",ans[i]);
                else printf("RUN %d\n",ans[i]);
            }
            if(m-st) printf("RUN %d\n",m-st);
        }
}
