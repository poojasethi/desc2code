#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
	int n,m,s,d;
	cin >> n >> m >> s >> d;
	
	vector <int> ob (n+1);
	ob[0] = -1;
	for (int i = 1; i <= n; i++) scanf("%d",&ob[i]);
	sort(ob.begin(),ob.end());
	vector <int> tr;
	for (int i = 1; i <= n; i++) tr.push_back(ob[i]-ob[i-1]-2);
	tr.push_back(s+1);
	vector <int> anw;
	anw.push_back(tr[0]);
	if (anw[0] < s)
	{
		cout << "IMPOSSIBLE" << endl;
		return 0;
	}
	for (int i = 1; i < tr.size(); i++)
	{
		int x = 2;
		for (int j = i; j < tr.size(); j++)
		{
			if (x > d){
				cout << "IMPOSSIBLE" << endl;
				return 0;
			}
			if (tr[j] >= s)
			{
				anw.push_back(x);
				anw.push_back(tr[j]);
				break;
			}
			else x += tr[j]+2;
			i++;
		}
	}
	anw[anw.size()-1] = m-ob[n]-1;
	if (anw[anw.size()-1] == 0) anw.pop_back();
	for (int i = 0; i < anw.size(); i++) printf("%s %d\n",(i%2 ? "JUMP" : "RUN"),anw[i]);
}
