#include <bits/stdc++.h>

#define db 	cout << "*****" << endl;
#define Max 200009
#define pb push_back
#define pp pop_back
#define ff first
#define ss second
#define mk make_pair
#define MOD 1000000007

using namespace std;

int n,m,s,d;
int arr[Max];
int nxt[Max];

vector < int > v(0);
  
int main(){
	
	cin >> n >> m >> s >> d;
	for( int i=1;i<=n;i++)scanf("%d" , arr + i );	
	
	sort( arr , arr + n + 1 );
	
	arr[0]=-1;
	
	int last=n;
	for( int i=n-1;i>=0;i--){
		if( (arr[i+1]-1) - (arr[i]+1) >= s  ){
			nxt[i] = last;
			last=i;
		}
	}
	
	int nw=0;
	while(nw<n){
		v.pb(arr[nw+1] - 1 - (arr[nw]+1));
		if( arr[nw+1]-1 - arr[nw] - 1 < s  ){
			cout << "IMPOSSIBLE\n";
			return 0;	
		}
		if( (arr[nxt[nw]]+1 - ( arr[nw+1] - 1 )) > d ){
			cout << "IMPOSSIBLE\n";		
			return 0;	
		}
		v.pb(arr[nxt[nw]]+1 - ( arr[nw+1] - 1 ));
		nw=nxt[nw];
	}
	if( arr[nw]+1 != m )
		v.pb(m-arr[nw]-1);
		
	for( int i=0;i<int(v.size());i++){
		if( i % 2 == 0 )printf("RUN %d\n" ,v[i]); 	
		else 			printf("JUMP %d\n" ,v[i]); 	
	}
	return 0;
}
