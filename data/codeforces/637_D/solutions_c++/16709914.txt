int n,m,q,x,y,z,cur,s,d,f;
int a[1000005];
string ans[1000005];
int ans2[1000005];
int main(){
    read(n);read(m);read(s);read(d);
    for(int i=1;i<=n;i++){
        read(a[i]);
    }a[0]=-1;
    sort(a,a+n+1);
    if(a[1]<=s){cout<<"IMPOSSIBLE"<<endl;return 0;}
    for(int i=0;i<=n;i++){f=0;
        q++;//dbg(cur);
        ans[q]="RUN";ans2[q]=(int)(a[i+1]-cur-1);cur=a[i+1]-1;
        if(a[n]+1-cur<=d){q++;ans[q]="JUMP";ans2[q]=(int)(a[n]+1-cur);cur=a[n]+1;break;}
        for(int j=i+1;j<=n;j++){
            if(a[j+1]-a[j]-2>=s&&a[j]-cur+1<=d){q++;ans[q]="JUMP";ans2[q]=(int)(a[j]-cur+1);cur=a[j]+1;i=j-1;f=1;break;}
        }
        if(f==0){cout<<"IMPOSSIBLE"<<endl;return 0;}
    }
    for(int i=1;i<=q;i++)cout<<ans[i]<<" "<<ans2[i]<<endl;
    if(cur!=m){cout<<"RUN "<<m-cur<<endl;}
    return 0;
}
