#include <cstdio>
#include <utility>
#include <vector>
#include <algorithm>

int a[200010];

int main()
{
    int n, i, m, s, d;
    scanf("%d %d %d %d", &n, &m, &s, &d);
    for (i=0; i<n; i++) scanf ("%d", &a[i]);
    std::sort (a, a+n);
    std::vector<int> z;
    z.push_back(0);
    int now=0, nowz=1;
    bool f=true;
    if (a[0]-1<s) f=false;
    while ((now<n-1)&&(f)) {
        if (a[now+1]-a[now]-2<s) now++;
        else {
            z.push_back(now);
            now++;
            z.push_back(now);
            if (a[z[nowz]]-a[z[nowz-1]]+2>d) f=false;
            nowz+=2;
        }
    }
    z.push_back(n-1);
    if (a[z[nowz]]-a[z[nowz-1]]+2>d) f=false;
    if (!f) printf ("IMPOSSIBLE");
    else {
        now=a[0]-1;
        printf ("RUN %d\n", now);
        for (i=1; i<z.size()-2; i+=2) {
            printf ("JUMP %d\n", a[z[i]]-a[z[i-1]]+2);
            printf ("RUN %d\n", a[z[i+1]]-a[z[i]]-2);
        }
        printf ("JUMP %d", a[z[i]]-a[z[i-1]]+2);
        if (m-a[z[i]]-1>0) printf ("\nRUN %d", m-a[z[i]]-1);
    }
    return 0;
}
