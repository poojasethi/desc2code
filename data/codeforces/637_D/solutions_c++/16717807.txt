#include <iostream>
#include <cstdio>
#include <algorithm>

#define For(i,a,b) for(int i = a; i <= b; i ++)
#define Dfu(i,a,b) for(int i = a; i >= b; i --)
#define nm 200009

using namespace std;

int n,m,s,d,pos;
int a[nm], res[4][2*nm], sl;

int main() {
	cin >> n >> m >> s >> d;
	For(i,1,n) cin >> a[i];
	sort(a + 1, a + 1 + n);
	a[n + 1] = 2 * a[n];
	sl = 0;
	pos = 0;
	int vt = 1;
	if (d < 2) { cout << "IMPOSSIBLE"; return 0;}
	while (vt <= n){
		if (a[vt] - pos <= s) { cout << "IMPOSSIBLE"; return 0;}
		int next;
		next = vt;
		while (next < n && a[next + 1] - a[vt] + 1 < d && a[next + 1] - a[next] - 2 < s) next ++;
		sl ++;
		res[1][sl] = 1; res[2][sl] = a[vt] - 1 - pos;
		sl ++;
		pos =a[vt] - 1;
		res[1][sl] = 2; res[2][sl] = a[next] + 1 - pos;
		pos = a[next] + 1;
		vt = next + 1;
	}
	if (pos < m){
		sl ++;
		res[1][sl] = 1;
		res[2][sl] = m - pos;
	}
	For(i,1,sl){
		if (res[1][i] == 1) cout << "RUN" << " " << res[2][i] << endl;
		else cout << "JUMP" << " "<< res[2][i]<<endl;
	}
	return 0;
}
