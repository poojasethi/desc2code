#include<cstdio>
#include<algorithm>
int a[200005];
int ans[400005][2];
int main()
{
	int n,m,s,d;
	scanf("%d%d%d%d",&n,&m,&s,&d);
	for(int i=0;i<n;i++)scanf("%d",&a[i]);
	std::sort(a,a+n);
	if(a[0]-1<s||d<2)
	{
		printf("IMPOSSIBLE\n");
		return 0;
	}
	int countt=1;
	ans[0][0]=0;
	ans[0][1]=a[0]-1;
	int jp=2;
	for(int i=1;i<n;i++)
	{
		int len=a[i]-a[i-1]-2;
		if(len<s)jp+=(a[i]-a[i-1]);
		else
		{
			ans[countt][0]=1;
			ans[countt][1]=jp;
			jp=2;
			countt++;
			ans[countt][0]=0;
			ans[countt][1]=len;
			countt++;
		}
		if(jp>d)
		{
			printf("IMPOSSIBLE\n");
			return 0;
		}
	}
	ans[countt][0]=1;
	ans[countt][1]=jp;
	countt++;
	jp=m-a[n-1]-1;
	if(jp!=0)
	{
		ans[countt][0]=0;
		ans[countt][1]=jp;
		countt++;
	}
	for(int i=0;i<countt;i++)
	{
		if(ans[i][0])printf("JUMP ");
		else printf("RUN ");
		printf("%d\n",ans[i][1]);
	}
}
