#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define ull unsigned long long
#define mp make_pair
#define pb push_back
#define RI(x) scanf("%d", &x)
#define RL(x) scanf("%I64d", &x)
#define RD(x) scanf("%f", &x)

int a[200007];
vector<pair<string, int> > v;

int main()
{
    int n, m, s, d;
    cin >> n >> m >> s >> d;
    for (int i=0; i<n; ++i)
        cin >> a[i];
    sort(a, a+n);
    a[n] = INT_MAX;

    int pos = 0;
    if (a[0] <= s)
    {
        cout << "IMPOSSIBLE";
        return 0;
    }
    for (int i=0; i<n; ++i)
    {
        v.pb(mp("RUN", a[i] - pos - 1));
        pos = a[i]-1;
        while (i < n && a[i+1] - a[i] - 1 <= s)
            ++i;
        if (a[i] + 1 - pos > d)
        {
            cout << "IMPOSSIBLE";
            return 0;
        }
        v.pb(mp("JUMP", a[i] + 1 - pos));
        pos = a[i] + 1;
    }
    if (m - pos > 0)
        v.pb(mp("RUN", m - pos));
    for (int j=0; j<v.size(); ++j)
        cout << v[j].first << ' ' << v[j].second << "\n";



    return 0;
}
