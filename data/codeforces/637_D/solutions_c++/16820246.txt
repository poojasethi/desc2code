#include<bits/stdc++.h>

using namespace std;

#define sgmn int v, int tl, int tr
#define tm  (tl + tr >> 1)
#define sol v*2, tl, tm
#define sag v*2+1, tm+1, tr
#define mp make_pair
#define st first
#define nd second
#define N 200005

pair < int , int > B[N];

int A[N],b,C[N],c,n,m,s,d,i,j,last;

int main(){

	scanf("%d%d%d%d",&n,&m,&s,&d);

	for(i=1;i<=n;i++)
		scanf("%d",&A[i]);

	sort(A+1 , A+n+1);

	if(A[1] < s + 1)
	{  puts("IMPOSSIBLE");  return 0;  }

	for(i=1;i<n;i++)
		if(A[i] < A[i+1] - s - 1)
			B[++b] = mp( A[i] , i );

	B[++b] = mp( A[n] , n );

	for(i=1;i<=n;i++){

		C[++c] = A[i] - 1 - last;

		last = A[i] - 1;

		j = upper_bound(B+1 , B+b+1 , mp( last + d , 0 ) ) - B - 1;

		if(last > B[j].first)
		{  puts("IMPOSSIBLE");  return 0;  }

		C[++c] = B[j].first + 1 - last;

		last = B[j].first + 1;

		i = B[j].second;

	}

	if(last < m)
		C[++c] = m - last;

	for(i=1;i<=c;i++){
		printf(i & 1 ? "RUN " : "JUMP ");
		printf("%d\n",C[i]);
	}

	return 0;
}
