def fail():
    print 'NO'
    exit()

n = input()
a = sorted(map(int, raw_input().split()))
b = [1]
for i in xrange(1, n):
    if a[i] > a[i-1] + 1: fail()
    elif a[i] == a[i-1]: b[-1] += 1
    else: b.append(1)
for i in xrange(len(b)-1):
    if b[i] and b[i+1]>=b[i]: b[i+1] -= b[i]
    else: fail()
print b[-1]==0 and 'YES' or 'NO'
