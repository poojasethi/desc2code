#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

int arr[100000];
int n;
vector <int> cnt;

int main(void)
{
	scanf("%d", &n);
	for(int i=0;i<n;i++) scanf("%d", arr+i);
	sort(arr, arr+n);

	bool isAble=true;
	for(int i=0;i<n;)
	{
		if(i && arr[i]-1!=arr[i-1]) { isAble=false; break; }
		int j;
		for(j=i;j<n && arr[j]==arr[i];j++);
		cnt.push_back(j-i);
		i=j;
	}

	for(int i=0;i<cnt.size()-1;i++)
	{
		if(cnt[i]<=0) { isAble=false; break; }
		cnt[i+1]-=cnt[i];
		cnt[i]=0;
	}

	if(cnt[cnt.size()-1]!=0) isAble=false;

	if(isAble) printf("YES\n");
	else printf("NO\n");

	return 0;
}
