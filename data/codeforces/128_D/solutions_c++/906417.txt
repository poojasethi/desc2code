#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,a[100010],b[100010];
bool check(){
    if(n%2)return 0;
    for(int i=1;i<n;++i)
        if(a[i]+1<a[i+1])
            return 0;
    for(int i=1,j;i<=n;i=j){
        j=i;while(a[i]==a[j])++j;
        b[++m]=j-i;
    }
    for(int i=1;i<m;++i){
        if(b[i+1]<b[i]||b[i]==0)return 0;
        b[i+1]-=b[i];b[i]=0;
    }
    return b[m]==0;
}
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;++i)scanf("%d",&a[i]);
    sort(a+1,a+n+1);
    printf("%s\n",check()?"YES":"NO");
    return 0;
}
