#include <cstdio>
#include <algorithm>

using namespace std;

const int N = 100000 + 10;

int n;
int a[N];
int cnt[N];

int main(){
	scanf("%d", &n);
	for(int i = 0; i < n; i++){
		scanf("%d", &a[i]);
	}
	sort(a, a + n);
	for(int i = 0; i < n - 1; i++){
		if (a[i] < a[i + 1] - 1){
			puts("NO");
			return 0;
		}
	}
	int nn = 1;
	cnt[0] = 1;
	for(int i = 1; i < n; i++){
		if (a[i] == a[i - 1]){
			cnt[nn - 1]++;
		}
		else {
			a[nn++] = a[i];
			cnt[nn - 1]++;
		}
	}
	bool ok = true;
	for(int i = 0; i < nn - 1; i++){
		if (cnt[i] == 0 || cnt[i] > cnt[i + 1]){
			puts("NO");
			return 0;
		}
		cnt[i + 1] -= cnt[i];
	}
	if (cnt[nn - 1]) puts("NO");
	else puts("YES");
}
