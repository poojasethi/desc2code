#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

const int MAX=1e5+10;

int a[MAX];
vector <int> v;

int main()
{
	int n;
	cin>>n;
	
	for(int i=0;i<n;i++)
		cin>>a[i];
	
	sort(a,a+n);
	int t=1;
	for(int i=1;i<n;i++)
	{
		if(a[i]!=a[i-1])
		{
			if(a[i]-a[i-1]>1)
			{
				cout<<"NO"<<endl;
				return 0;
			}

			v.push_back(t-1);
			t=1;
		}
		else
			t++;
	}

	v[0]++;
	v.push_back(t);
	
	for(int i=v.size()-2;i>=0;i--)
	{
		if(v[i]<v[i+1] || v[i]==0)
		{
			cout<<"NO"<<endl;
			return 0;
		}

		else
		{
			v[i]-=v[i+1];
			v[i]++;
		}
	}

	if(v[0]!=1)
	{
		cout<<"NO"<<endl;
		return 0; 
	}

	cout<<"YES"<<endl;

}
