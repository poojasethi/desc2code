#include<iostream>
#include<algorithm>

using namespace std;

int a[102400], n, i, j, p, t, s;

int main() {
	cin >> n;
	for (i = 0; i < n; ++i)
		cin >> a[i];
	sort(a, a + n);
	p = *a;
	t = 0;
	for (i = 0; i < n; i = j) {
		for (j = i + 1; a[j] == a[i]; ++j);
		if (a[i] - p>1) {
			cout << "NO";
			return 0;
		}
		s = j-i;
		if (s>t) {
			p = a[i];
			t = s - t;
		}
		else {
			t -= s;
		}
	}
	cout << (t ? "NO" : "YES");
	return 0;
}
