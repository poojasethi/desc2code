#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int N = 100005;
int a[N],k,cnt[N];
int main()
{
	int n,i;
	scanf("%d",&n);
	for(i=0;i<n;i++)scanf("%d",a+i);
	sort(a,a+n);
	cnt[0]=1;
	bool flag=0;
	for(k=i=1;i<n;i++)
	{
		if(a[i]!=a[i-1])
		{
			cnt[k]=1;
			k++;
			if(a[i]!=a[i-1]+1)flag=1;
		}
		else
		{
			cnt[k-1]++;
		}
	}
	if(flag){puts("NO");return 0;}
	for(i=0;i<k-1;i++)
	{
		if(!cnt[i]||cnt[i+1]<cnt[i])flag=1;
		cnt[i+1]-=cnt[i];
	}
	if(cnt[k-1])flag=1;
	if(flag)puts("NO");
	else puts("YES");
	return 0;
}
