#include <cstdio>
#include <iostream>
#include <map>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int minn=1e9+1;
    map<int,int> mp;
    for(int i=0;i<n;i++)
    {
        int t;
        scanf("%d",&t);
        mp[t]++;
        minn=min(minn,t);
    }
    int cnt=n;
    int now=0;
    while(cnt)
    {
        cnt-=mp[minn];
        mp[minn]-=now;
        now=mp[minn];
        if(now<=0&&cnt!=0)
        {
            cout<<"NO"<<endl;return 0;
        }
        minn++;
    }
    if(now==0)
    cout<<"YES"<<endl;
    else
        cout<<"NO"<<endl;

}
