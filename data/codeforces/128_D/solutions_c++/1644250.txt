#include<cstdio>
#include <algorithm>
using namespace std;
bool cmp(int a,int b)
{
return a<b;  
}
int a[111111],n,i,j,p,t,s;
int main()
{
	scanf("%d",&n);
	for(i=0;i<n;++i)
		scanf("%d",&a[i]);
	sort(a,a+n,cmp);
	p=a[0];t=0;
	for(i=0;i<n;i=j)
	{
		for(j=i+1;a[j]==a[i];j++);
			if(a[i]-p>1)
			{
				printf("NO\n");
				return 0;
			}
		s=j-i;
		if(s>t)
		{
			p=a[i];
			t=s-t;
		}
			
		else t-=s;
	}
	if(t) printf("NO\n");
	else printf("YES\n");
	return 0;
}
