#include <cstdio>
#include <algorithm>

const int N = 100000 + 10;

int n, a[N];
int num[N], cnt[N], m;
int ans[N];

int main()
{
	scanf("%d", &n);
	for(int i = 1; i <= n; i++) scanf("%d", &a[i]);
	std::sort(a+1, a+1+n);
	for(int i = 1; i <= n; i++)
	{
		if(a[i] != a[i-1]) num[++m] = a[i];
		cnt[m]++;
	}
	for(int i = 2; i <= m; i++)
		if(num[i] - num[i-1] > 1)
		{
			puts("NO");
			return 0;
		}
	bool flag = 1;
	for(int i = 1; i < m; i++)
	{
		if(cnt[i] && cnt[i] <= cnt[i+1])
			cnt[i+1] -= cnt[i];
		else flag = 0;
	}
	if(flag && cnt[m] == 0) puts("YES");
	else puts("NO");
	return 0;
}
