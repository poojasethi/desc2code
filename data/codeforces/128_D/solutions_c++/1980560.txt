const int N=220001;

using namespace std;

map<int,int> mp;
set<int> st;

int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
	int a;
	scanf("%d",&a);
	st.insert(a);
	mp[a]=mp[a]+1;
    }
    while(st.size())
    {
	set<int>::iterator it=st.begin();
	int a=*it;
	if(mp[a]==0)
	{
	    st.erase(a);
	    continue;
	}
	if(mp[a+1]<2-(n==2))
	{
	    puts("NO");
	    return 0;
	}
	mp[a]--;mp[a+1]--;
	n-=2;
    }
    puts("YES");
    return 0;
}
