#include<cstdio>
#include<algorithm>
using namespace std;

int arr[100000];
int main()
{
    int n,i,j,p,t,count;
    scanf("%d",&n);
    for(i=0;i<n;++i)
        scanf("%d",&arr[i]);
    sort(arr,arr+n);
    p=arr[0];
    t=0;
    for(i=0;i<n;i=j)
    {
        //tim counto khac arr[i]
        for(j=i+1;arr[j]==arr[i];++j);
        
        if(arr[i]-p>1)
        {
            printf("NO");
            return 0;
        }
        
        count=j-i;
        
        if( count > t)
        {
            p=arr[i];
            t=count-t;
        }
        else 
            t-=count;
    }
    puts(t?"NO":"YES");
    return 0;
}
