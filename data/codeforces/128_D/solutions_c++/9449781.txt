#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std;

int N,a[100100],b[100100];

void work()
{
	memset(b,0,sizeof(b));

	sort(a+1,a+1+N);
	int base=a[1]-1; a[1]=1; b[1]++;
	for(int i=2;i<=N;i++)
	{
		a[i]-=base;
		if(a[i]-a[i-1]>1) {puts("NO");return;}
		b[a[i]]++;
	}

//	for(int i=1;i<a[N];i++) printf("%d%c",b[i],i==a[N]?' ':'\n');
	bool flag=1;
	for(int i=1;i<a[N];i++)
		if(b[i] && b[i]<=b[i+1]) b[i+1]-=b[i];
		else flag=0;
	if(flag && b[a[N]]==0) puts("YES");
	else puts("NO");
}

int main()
{
	while(scanf("%d",&N)==1)
	{
		for(int i=1;i<=N;i++) scanf("%d",&a[i]);
		work();
	}

	return 0;
}
