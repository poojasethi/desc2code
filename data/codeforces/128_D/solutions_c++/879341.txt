#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#include <map>
using namespace std;
int n;
vector<int> U;
map<int,int> BR;
int main(){
  scanf("%d",&n); U.resize(n);
  for(int i=0;i<n;++i)scanf("%d",&U[i]) , BR[U[i]]+=2;
  int L=*min_element(U.begin(), U.end()),R=*max_element(U.begin(), U.end());
  bool OK=true; int last=BR[L];
  for(int i=L+1;i<=R;++i)
   if(!BR[i]){OK=false; break;}
   else{
    last=BR[i]-last;
    if(last<0 || last==0&&i!=R){OK=false; break;}
       }
  if(!OK || last!=0)printf("NO\n");
  else printf("YES\n");
  //system("pause");
  return 0;
}
