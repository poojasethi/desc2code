#include <algorithm>
#include <stdio.h>
#include <map>
#define  fi        first
#define  se        second
using    namespace std;

int n;
map<int,int>hash;

int main()
{
  map<int,int>::iterator it;
  scanf("%d",&n);
  for(int i=0,x ; i<n ; i++)
  {
    scanf("%d",&x);
    hash[x]++;
  }
  if(n%2) { printf("NO\n");  exit(0); }
  while(!hash.empty())
  {
    it=hash.begin();
    if(it->se==0)
    {
      hash.erase(it);
      continue;
    }
    if((hash[it->fi+1]<2 && n>2) || (hash[it->fi+1]<1 && n==2))
    {
      printf("NO\n");
      exit(0);
    }
    hash[it->fi]--;
    hash[it->fi+1]--;
    n-=2;
  }
  printf("YES\n");
  return 0;
}
