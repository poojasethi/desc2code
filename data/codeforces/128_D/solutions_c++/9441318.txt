#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cmath>
#define maxn 2000005
using namespace std;
int v[maxn],num[maxn],a[maxn];

int main()
{
	int n,tot=0;
	memset(v,0,sizeof(v));
	scanf("%d",&n);	
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	a[0]=-1;
	for (int i=1;i<=n;i++) 
	{
		if (a[i]!=a[i-1]) num[++tot]=a[i];
		v[tot]++;
	}
	int k=1,ans=1;
	v[1]--;
	while ( (v[k-1]>0 && num[k]==num[k-1]+1) || (v[k+1]>0 && num[k+1]==num[k]+1) )
	{
		if (v[k+1]>0 && num[k+1]==num[k]+1) v[k+1]--,k+=1;
		else v[k-1]--,k-=1;
		ans++;
	}
	if (ans==n&&k==2) printf("YES\n");
	else printf("NO\n");
	return 0;
}
