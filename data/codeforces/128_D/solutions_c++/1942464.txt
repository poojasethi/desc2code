#include <iostream>
#include <algorithm>

using namespace std;

const int MAXN = 1000*100+10;

int a[MAXN], cnt[MAXN], n;

int main()
{
	ios::sync_with_stdio(0);
	cin >> n;
	for (int i = 1 ; i <= n ; i++)
		cin >> a[i];
	sort(a+1, a+1+n);
	for (int i = 2 ; i <= n ; i++)
		if (a[i] - a[i-1] >= 2)
		{
			cout << "NO" << endl;
			return 0;
		}
	for (int i = 1 ; i <= n ; i++)
		cnt[a[i]-a[1]+1]++;
	for (int i = 1 ; i < a[n]-a[1]+1 ; i++)
		if (cnt[i] > cnt[i+1] || cnt[i] == 0)
		{
			cout << "NO" << endl;
			return 0;
		}
		else
			cnt[i+1] -= cnt[i];
	if (cnt[a[n]-a[1]+1] == 0)
		cout << "YES" << endl;
	else
		cout << "NO" << endl;
	return 0;
}
