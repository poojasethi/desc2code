#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n, m, a[100005], b[100005], ok;

int main(){
	scanf("%d", &n);
	for (int i=0; i<n; i++) scanf("%d", &a[i]);
	sort(a, a+n);
	
	ok = 1;
	for (int i=1; i<n; i++)
	if (a[i] - a[i-1] > 1) ok = 0;
	
	if (ok){
		for (int i=0; i<n; i++) b[a[i]-a[0]]++;
		m = a[n-1] - a[0];
		for (int i=0; i<m; i++){
			if (!b[i]) ok = 0;
			if (b[i] > b[i+1]) ok = 0;
			b[i+1] -= b[i];
		}
		if (b[m] != 0) ok = 0;
	}
	
	printf("%s\n", ok ? "YES" : "NO");
	return 0;
}
