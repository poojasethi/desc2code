#include <map>
#include <iostream>
#include <string>

using namespace std;

int main(){
  int n; 
  cin >> n;
  map<int, int> h;
  int v;
  for(int i = 0; i < n; i++){
    cin >> v;
    h[v]++;
  }
  string res="YES";
  if(n%2==1)
    res="NO";
  else
    for(int i = 0; i < n/2; i++){
      int l=h.begin()->first;
      h[l]--;
      if(h[l]==0) h.erase(l);
      h[l+1]--;
      if((i<n-1 and h[l+1]<0) or (i==n-1 and h[l+1]!=0)){
        res="NO";
        break;
      }
    }
  cout << res << endl;
  return 0;
}
