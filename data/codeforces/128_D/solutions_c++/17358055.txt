#include <bits/stdc++.h>
using namespace std;
int n,a[100005],b[100005],c[100005];
int main(){
	cin >> n;
	for(int i=0;i<n;i++) cin >> a[i];
	sort(a,a+n);
	for(int i=1;i<n;i++){
		if(a[i]-a[i-1] > 1){
			puts("NO"); return 0;
		}
	}
	for(int i=0;i<n;i++){
		b[a[i]-a[0]]++;
	}
	c[0] = b[0]*2; // i & i+1
	for(int i=1;i<a[n-1]-a[0];i++){
		c[i] = b[i]*2-c[i-1];
	}
	for(int i=0;i<=a[n-1]-a[0];i++){
		if(c[i]<0){
			puts("NO"); return 0;
		}
		if(i>=1&&i<=a[n-1]-a[0]-2&&c[i]<=0){
			puts("NO"); return 0;
		}
		int x = 0;
		x += c[i];
		if(i) x+=c[i-1];
		if(x != b[i]*2){
			puts("NO"); return 0;
		}
	}
	puts("YES");
}
