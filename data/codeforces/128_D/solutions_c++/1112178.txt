#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;
const int N=100005;
int a[N],b[N];

int main()
{
    int i,n,m,l,d,t;
    scanf("%d",&n);
    for (i=1; i<=n; i++) scanf("%d",&a[i]);
    sort(a+1,a+n+1);
    m=0;
    for (i=1; i<=n; i++)
     if (i==1 || a[i]!=a[i-1])
     {a[++m]=a[i]; b[m]=1; }
     else b[m]++;
   for (i=1; i<m; i++)
    if (a[i+1]-a[i]!=1) break;
  if (i<m) {printf("NO\n"); return 0;}
  for (i=1; i<m; i++)
   if (b[i]>0 && b[i+1]>=b[i]) b[i+1]-=b[i];
  else break;
  if (i<m) {printf("NO\n"); return 0;}
  if (b[m]==0) printf("YES\n"); else printf("NO\n");
}
