#include<iostream>
#include<algorithm>
using namespace std;
const int N=110000;
static int a[N];
int main()
{
    ios::sync_with_stdio(false);
	int n;
	cin>>n;
	int i;
	for (i=0;i<n;i++) cin>>a[i];
	sort(a,a+n);
	int gap=0,need=1,last=a[0];
	for (i=1;i<n;i++)
	{
		gap|=a[i]>a[i-1]+1;
		if (a[i]!=a[i-1]) 
		{
			gap|=need<=0;
			need=-need;
		}
		need++;
	}
	cout<<(gap||need?"NO":"YES")<<endl;
    return 0;
}
