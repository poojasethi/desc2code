#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    scanf("%d", &n);
    map<int, int> mp;
    int mx = 1e9 + 1;
    int mn = 0;
    for(int i = 0; i < n; i ++) {
        int t;
        scanf("%d", &t);
        mp[t] ++;
        mx = min(mx, t);
        mn = max(mn, t);
    }
    for(int i = mx; i < mn; i ++) {
        if(mp[i] == 0 || mp[i+1] - mp[i] < 0) {
            puts("NO");
            return 0;
        }
        mp[i+1] -= mp[i];
    }
    if(mp[mn] != 0)
        puts("NO");
    else 
        puts("YES");
    return 0;

}
