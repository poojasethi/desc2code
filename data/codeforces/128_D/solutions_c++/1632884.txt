#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxn = 100000 + 2;
int a[maxn], n;
int main()
{
//    freopen("D.in", "r", stdin);
//  freopen("D.out", "w", stdout);
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) scanf("%d", a + i);
    int lt = 1, x = 1;
    sort(a + 1, a + n + 1);
    for (int i = 2; i <= n; i++)
    {
        if (a[i] - a[lt] > 1) return printf("NO\n"), 0;
        if (a[lt] == a[i]) x++;
        else x--;
        if (x == -1) lt = i, x = 1;
    }
    printf(x < 1 ? "YES\n" : "NO\n");
    return 0;
}
