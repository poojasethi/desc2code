#include <stdio.h>
#include <string.h>
#include <algorithm>
#define maxn 100010

using namespace std;

int a[maxn];
int b[maxn],m;
int n;
int check(){
    for (int i=1;i<n;i++)
        if (a[i]>a[i-1]+1) return 0;
    m=0;
    for (int i=0,j;i<n;i=j+1){
        for (j=i;j<n-1 && a[j]==a[j+1];j++);
        b[m++]=j-i+1;
    }
    for (int i=0;i<m-1;i++){
        if (b[i+1]<b[i] || b[i]==0) return 0;
        b[i+1]-=b[i];
        b[i]=0;
    }
    return b[m-1]==0;
}
int main(){
    scanf("%d",&n);
    for (int i=0;i<n;i++) scanf("%d",a+i);
    sort(a,a+n);
    puts(check()?"YES":"NO");
    return 0;
}
