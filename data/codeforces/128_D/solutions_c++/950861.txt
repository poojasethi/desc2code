#include<iostream>
using namespace std;
const int maxtn=100001;
int n,mint,maxt,a[maxtn],v[maxtn];

void print(int x)
{
     if (x) cout<<"YES";
     else cout<<"NO";
}

int main(){
    cin>>n;
    mint=1<<30;maxt=0;
    for (int i=0;i<n;i++) {
        cin>>a[i];
        if (a[i]<mint) mint=a[i];
        if (a[i]>maxt) maxt=a[i];
    }
    if ((maxt-mint)*2>n) {print(0);return(0);}
    for (int i=0;i<n;i++) v[a[i]-mint]++;
    for (int i=0;i<maxt-mint;i++) 
        if (v[i]==0||v[i]>v[i+1]) {print(0);return(0);}
        else v[i+1]-=v[i];
    print(!v[maxt-mint]);
   // system("pause");
    return(0);
}
