#include<cstdio>
int a[100000];
int main()
{
	int n,n0=0,n1=0,i,min=1000000001,max=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		if(a[i]>max) max=a[i];
		if(a[i]<min) min=a[i];
		if(a[i]%2) n1++;
		else n0++;
	}
	int k1=max-min;
	if(n0!=n1||k1>n/2+1) puts("NO");
	else 
	{
		int k[50010]={0};
		for(i=0;i<n;i++)
			k[a[i]-min]++;
		for(i=0;i<k1;i++)
		{
			if(k[i]&&k[i+1]>=k[i]) k[i+1]-=k[i];
			else break;
		}
		if(i==k1&&!k[k1]) puts("YES");
		else puts("NO");
	}
	return 0;
}
