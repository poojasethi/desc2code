
n,m = map(int, raw_input().split())
books = map(int, raw_input().split())

cnt= {}
for book in books:
	if book not in cnt: cnt[book]=0
	cnt[book] += 1
#print cnt
ans = 0
keys = cnt.values()
for i in range(0, len(cnt)-1):
	for j in range(i+1, len(cnt)):
		#print i, j
		ans += keys[i]*keys[j]
	#print ans
print ans
