n, m = map(int, raw_input().split())
array = map(int, raw_input().split())
genres = [0] * (m + 1)
for i in array:
    genres[i] += 1
for i in xrange(len(genres)-2, 0, -1):
    genres[i] += genres[i + 1]
ans = 0
for i in xrange(2, len(genres)):
    ans += (genres[i-1] - genres[i]) * genres[i]
print ans
