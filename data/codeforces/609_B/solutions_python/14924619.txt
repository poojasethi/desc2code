def read():
    line = raw_input()
    return map(int, line.split())

if __name__ == '__main__':
    n, m = read()
    count = [0] * m
    sum = 0
    for x in read():
        count[x - 1] += 1
        sum += 1
    ans = 0
    for i in xrange(m):
        sum -= count[i]
        ans += count[i] * sum
    print ans
