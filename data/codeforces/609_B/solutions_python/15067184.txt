n, m = map(int, raw_input().split())
arr = map(int, raw_input().split())

genres = [0] * m

for book in arr:
    genres[book - 1] += 1

acc = 0
suffix_sum = [0] * m
for i in xrange(len(genres) - 1, -1, -1):
    suffix_sum[i] = acc
    acc += genres[i]

print sum(map(lambda (x, y) : x * y, zip(genres, suffix_sum)))
