inputs = [int(i) for i in raw_input().split()]
n = inputs[0]
m = inputs[1]
books = [0 for i in range(m)]
counts = [int(i) for i in raw_input().split()]
for book in counts:
    books[book - 1] += 1
books_sum = sum(books)
result = 0
for book in books:
    result += book * (books_sum - book)
print result/2
