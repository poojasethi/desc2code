n,m = map(int,raw_input().split())
a= map(int,raw_input().split())
res = [0 for _ in range(m+1)]
cnt = (n*(n-1)/2)

for i in range(1,m+1):
    for j in range(n):
        if i == a[j]: 
            res[i] += 1
            
for i in range(1,m+1):
    if res[i] > 1 :
        cnt -= (res[i]*(res[i]-1))/2
print cnt
