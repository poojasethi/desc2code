
#include<bits/stdc++.h>
using namespace std;
int a[3005];

int main(){
	int n, s;
	scanf("%d%d", &n, &s);
	for( int i=0 ; i<s ; i++ ) scanf("%d", &a[i]);
	sort( a, a+s );
	if( a[0]==1||a[s-1]==n ){
		printf("NO\n");
		return 0;
	}
	for( int i=0 ; i<s-2 ; i++ ){
		if( a[i]==a[i+1]-1&&a[i]==a[i+2]-2 ){
			printf("NO\n");
			return 0;
		}
	}
	printf("YES\n");
	return 0;
}
