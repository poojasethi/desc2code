#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int k,m,a[3005],i;
	cin>>k>>m;
	for(i=0;i<m;i++)
		cin>>a[i];
	sort(a,a+m);
	if(a[m-1]==k||a[0]==1){
		cout<<"NO"<<endl;
		return 0;
	}
	for(i=2;i<m;i++){
		if(a[i]-a[i-2]==2){
			cout<<"NO"<<endl;
			return 0;
		}
	}
	cout<<"YES"<<endl;
	return 0;
}
