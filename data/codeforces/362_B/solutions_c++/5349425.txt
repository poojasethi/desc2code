#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int i,m,n;
	int a[10003];
	cin>>n>>m;
	for(i=0;i<m;i++)
	{
		cin>>a[i];
		if(a[i]==1||a[i]==n)
		{
			cout<<"NO";
			return 0;	
		}
	}
	sort(a,a+m);
	
	for(i=2;i<m;i++)
	if(a[i-2]==(a[i]-2))
	{
		cout<<"NO";
		return 0;	
	}
	cout<<"YES";
	return 0;
}
