#include <cstdio>
#include <algorithm>

int n, l[5000], m;
bool b;

int main(){
	scanf("%d%d", &n, &m);
	for(int i = 0; i < m; i++) scanf("%d", &l[i]);
	std::sort(l, l + m);
	for(int i = 0; i < m-2; i++) if(l[i] == l[i+1] - 1 && l[i+2]-2 == l[i])
		b = 1;
	if(b || l[0] == 1 || l[m-1] == n) printf("NO\n");
	else printf("YES\n");
	
	scanf("\n");
}
