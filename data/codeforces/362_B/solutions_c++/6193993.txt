#include<cstdio>
#include<algorithm>
using namespace std;
int a[3010];
int main(){
	int n,m;scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++)scanf("%d",&a[i]);
	sort(a+1,a+m+1);if(a[m]==n||a[1]==1){
		printf("NO");return 0;
	}
	for(int i=3;i<=m;i++){
		if(a[i-2]+1==a[i-1]&&a[i-1]+1==a[i]){
			printf("NO");return 0;
		}
	}printf("YES");
}
