#include <cstdio>
#include <algorithm>
int all[3005];
int main() {
	int n, k;
	scanf("%d%d", &n, &k);
	for(int i=0; i<k; ++i) scanf("%d", &all[i]);
	std::sort(all, all+k);
	bool ans=true;
	if( all[0]==1 || all[k-1]==n ) ans=false;
	for(int i=0; ans&&i<k-2; ++i)
		if( all[i+1]-all[i]==1 && all[i+2]-all[i+1]==1 ) ans=false;
	puts(ans?"YES":"NO");
	return 0;
}
