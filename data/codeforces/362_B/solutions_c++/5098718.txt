#include<cstdio>
#include<algorithm>
using std::sort;

int v[3005];
int main(void){
	int x, n;
	scanf("%d%d", &x, &n);

	for(int i=0;i<n;i++)
		scanf("%d", v+i);

	sort(v,v+n);
	if(v[0] == 1 || v[n-1] == x){
		puts("NO");
		return 0;
	}

	for(int i=0;i<n-2;i++)
		if(v[i+1] == v[i] + 1 && v[i+2] == v[i] + 2){
			puts("NO");
			return 0;
		}

	puts("YES");
	return 0;
}
