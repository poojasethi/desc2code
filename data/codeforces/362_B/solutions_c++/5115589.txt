#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n,m;
	cin >> n >> m;
	int a[m];
	for (int i=0;i<m;i++) cin >> a[i];
	sort(a,a+m);
	string ans="YES";
	if (a[0]==1 || a[m-1]==n) ans="NO";
	for (int i=0;i<m-2;i++)
	{
		//cout << i << " " <<a[i] << endl;
		if (a[i]==a[i+1]-1 && a[i]==a[i+2]-2) ans="NO";
	}
	cout << ans << endl;
}
