#include <iostream>
#include <algorithm>
using namespace std;
int n,m,d[3030];
int main()
{
	cin >> n >> m;
	for(int i = 0; i < m; i++)
		cin >> d[i];
	sort(d,d+m);
	bool yes = true;
        yes = (d[0]!=1 && d[m-1]!=n);
	for(int i = 2; i < m && yes; i++) if(d[i]<n && d[i]-d[i-2]==2) {yes = false;}        
	if(yes) cout << "YES";
	else cout << "NO";
}
