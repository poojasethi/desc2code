#include<bits/stdc++.h>
using namespace std;
#define MAX 1<<25

int main()
{
    int n,m,a[MAX],i;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;++i)scanf("%d",&a[i]);
    sort(a+1,a+m+1);
    if(a[1]==1 || a[m]==n)return puts("NO"),0;
    for(i=1;i+2<=m;++i)if(a[i]==a[i+1]-1 && a[i]==a[i+2]-2)return puts("NO"),0;
    puts("YES");
}
