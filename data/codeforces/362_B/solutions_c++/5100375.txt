#include<cstdio>
#include<algorithm>
using namespace std;
int num[3003];
int main () {
	int n,m,i;
	scanf("%d%d",&n,&m);
	for(i=0;i<m;i++) scanf("%d",&num[i]);
	sort(num,num+m);
	for(i=2;i<m;i++) 
		if(num[i]-num[i-1]==1&&num[i-1]-num[i-2]==1) {printf("NO");return 0;}
	if(num[0]!=1&&num[m-1]!=n) printf("YES");
	else printf("NO");
	return 0;
}
