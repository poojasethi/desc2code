#include <bits/stdc++.h>
using namespace std;
int A[3001];
int main()
{
	int n,m;
	cin>>n>>m;
	int flag=0;
	for(int i=0;i<m;++i)
	{
		cin>>A[i];
		if(A[i]==1 || A[i]==n)
		flag=1;
	}
	sort(A,A+m);
	for(int i=0;i<m-2;++i)
	{
		if(A[i+1]-A[i]==1 && A[i+2]-A[i+1]==1)
			flag=1;
	}
	if(flag )
		cout<<"NO"<<endl;
	else
		cout<<"YES"<<endl;
}
