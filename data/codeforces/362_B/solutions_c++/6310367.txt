#include<cstdio>
#include<algorithm>
using namespace std;
int a1[3000];
int main()
{
	int n,a;
	scanf("%d%d",&a,&n);
	for(int i=0;i<n;++i)
		scanf("%d",a1+i);
	sort(a1,a1+n);
	if(a1[0]==1||a1[n-1]==a)
	{puts("NO");return 0;}
	for(int i=2;i<n;++i)
	if(a1[i]==a1[i-1]+1&&a1[i]==a1[i-2]+2)
	{puts("NO");return 0;}
	puts("YES");
}
