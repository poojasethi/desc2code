#include <iostream>
#include <algorithm>
using namespace std;
int main()
{	
	int n,m,d[3000],i;
	cin>>n>>m;
	for (i=0;i<m;i++) cin>>d[i];
	sort(d,d+m);
	for (i=1;i<m-1;i++) if (d[i+1]-d[i]==d[i]-d[i-1]&&d[i]-d[i-1]==1) break;
	if (i<m-1) cout<<"NO"<<endl;
	else if (d[0]==1||d[m-1]==n) cout<<"NO"<<endl;
	else cout<<"YES"<<endl;
	return 0;
}
