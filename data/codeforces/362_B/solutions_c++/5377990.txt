#include <iostream>
#include <algorithm>
using namespace std;
int n,m,i,a[3001];
int main()
{
    cin>>n>>m;
    for(i=1;i<=m;i++)
        cin>>a[i];
    sort(a+1,a+m+1);
    if(a[1]==1||a[m]==n){cout<<"NO";return 0;}
    for(i=3;i<=m;i++)
        if(a[i]==a[i-2]+2)
            {cout<<"NO";return 0;}
    cout<<"YES";
    return 0;
}
