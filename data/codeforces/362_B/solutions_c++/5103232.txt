#include <cstdio>
#include <algorithm>
using namespace std;
int N,M,ar[3005];
int main ()
{
	scanf("%d %d",&N,&M);
	for (int i = 0; i < M; i++)
		scanf("%d",&ar[i]);
	sort(ar,ar+M);
	if(ar[0]==1 || ar[M-1]==N)
	{
		puts("NO");
		return 0;
	}
	for (int i = 1; i <M-1; i++)
	if(ar[i]==ar[i-1]+1 && ar[i]==ar[i+1]-1)
	{
		puts("NO");
		return 0;
	}
	puts("YES");
	
}
