#include <bits/stdc++.h>

using namespace std;
	int m,n;
	int a[10003];
int main(){
	cin>>n>>m;
	for(int i=0;i<m;i++){
		cin>>a[i];
		if(a[i]==1||a[i]==n){
			cout<<"NO";
			return 0;	
		}
	}
	sort(a,a+m);
	
	for(int i=2;i<m;i++)
	if(a[i-2]==(a[i]-2)){
		cout<<"NO";
		return 0;	
	}
	cout<<"YES";
	return 0;
}
