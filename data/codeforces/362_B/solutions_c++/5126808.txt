#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,m,a[3001];
    cin>>n>>m;
  
  for(int i=0;i<m;i++)
  cin>>a[i];
  sort(a,a+m);
  if(a[0]==1||a[m-1]==n)
  {
    cout<<"NO";
    return 0;
  }
   for(int i=0;i<m-2;i++)
    if(a[i]==a[i+1]-1&&a[i]==a[i+2]-2)
    {
    cout<<"NO";
    return 0;
    }
    cout<<"YES";
}
