#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
int n,m,i,f=0;
cin>>n>>m;
int a[m];
for(i=0;i<m;i++)
{
   cin>>a[i];
}
sort(a,a+m);
if(a[0]==1||a[m-1]==n)
 {cout<<"NO";
 return 0;}
 for(i=1;i<m-1;i++)
 {
     if(a[i]==a[i-1]+1&&a[i+1]==a[i]+1)
     {cout<<"NO";
      return 0;
     }
 }
 cout<<"YES";
return 0;
}
