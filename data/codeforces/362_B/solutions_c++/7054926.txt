#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,a[3010],i;
	cin>>n>>m;
	for(i=0;i<m;i++)
		cin>>a[i];
	sort(a,a+m);
	for(i=0;i<m;i++)
	{
		if(a[i]==1 || a[i]==n || (i+2<m && a[i+1]==a[i]+1 && a[i+2]==a[i+1]+1))
		{
			cout<<"NO\n";
			return 0;
		}
	}
	cout<<"YES\n";
	return 0;
}
