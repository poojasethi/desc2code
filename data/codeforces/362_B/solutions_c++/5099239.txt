#include <iostream>
#include <cmath>
#include <algorithm>

using namespace std;
int main()
{long long n,r; cin>>n>>r;
long long a[3005];
for (int i=0;i<r;i++) cin>>a[i];
sort(a,a+r);
if (a[r-1]==n||a[0]==1) {cout<<"NO"; return 0;}
for (int i=2;i<r;i++) if (a[i]==a[i-1]+1&&a[i-1]==a[i-2]+1) {cout<<"NO"; return 0;}
cout<<"YES";
return 0;
}
