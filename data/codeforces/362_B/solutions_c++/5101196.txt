#include <bits/stdc++.h>
#define maxn 3100

using  namespace std;

int N,M,ar[maxn],a;

int main()
{
	scanf(" %d %d",&N,&M);
	for(int i=1;i<=M;i++)
		scanf(" %d",&ar[i]);
	sort(ar+1,ar+1+M);
	if(ar[M]==N || ar[1]==1){
		printf("NO\n");
		return 0;
	}
	for(int i=1;i<=M-2;i++)
		if(ar[i+2]-ar[i]==2){
			printf("NO\n");
			return 0;
		}
	printf("YES\n");
	return 0;
}
