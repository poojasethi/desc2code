#include <algorithm>
#include <iostream>
using namespace std;

int n,m,a[3333],i;

int main(){
	cin >> n >> m ;
	for(i=1;i<=m;i++){
		cin >> a[i];
		if(a[i]==n || a[i]==1){cout<<"NO";return 0;}
	}
	sort(a+1,a+m+1);
	for(i=3;i<=m;i++){
		if(a[i]-a[i-1]==1 && a[i-1]-a[i-2]==1){
			cout<<"NO";
			return 0;
		}
	}
	cout<<"YES";
}
