#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int n,m,d[5000],i,err=0;
    scanf("%d %d",&n,&m);
    for(i=0;i<m;i++)
    {
        scanf("%d",d+i);
    }
    sort(d,d+m);
    if(d[0]==1||d[m-1]==n)err=1;
    for(i=2;i<m;i++)
    {
        if(d[i-2]==d[i]-2&&d[i-1]==d[i]-1)err=1;
    }
    printf("%s\n",err?"NO":"YES");
}
