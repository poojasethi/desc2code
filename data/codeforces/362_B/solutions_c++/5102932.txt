#include <iostream>
#include <algorithm>
using namespace std;

int n,m,v[3009];

int main()
{
	cin >> n >> m;
	
	for (int i = 0; i < m; i++)
		cin >> v[i];
	
	sort(v,v+m);
	
	if (v[0] == 1 || v[m-1] == n) {cout << "NO";return 0;}
	
	for (int i = 1; i < m; i++)
		if (i+1 < m && v[i]-v[i-1] == 1 && v[i+1]-v[i] == 1) {cout << "NO";return 0;}
	
	cout << "YES";
}
