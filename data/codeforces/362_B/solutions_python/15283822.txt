n,m=map(int,raw_input().split())
if(m!=0):
    li=map(int,raw_input().split())
    li.sort()
    f=0
    if(li[0]!=1):
        for i in xrange(m-2):
            if(li[i+2]-li[i]==2):
                f=1
                break
        if(li[m-1]==n):
            f=1
        if(f==0):
            print "YES"
        else:
            print "NO"
    else:
        print "NO"
else:
    print "YES"
