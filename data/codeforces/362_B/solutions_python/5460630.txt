n,m=map(int,raw_input().split())
flag = True
if m!=0:
    steps = map(int,raw_input().split())
    steps.sort()
if m==0:
    flag = True
elif steps[0]==1 or steps[m-1]==n:
    flag = False
else:
    for i in range(len(steps)-2):
        if steps[i+1]==steps[i]+1 and steps[i+2]==steps[i+1]+1:
            flag = False   
if flag==True:
    print "YES"
else:
    print "NO"
