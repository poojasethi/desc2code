n, m = map(int,raw_input().split())
if m == 0:
    print "YES"
else:
    d = sorted(map(int,raw_input().split()))
    if d[0] == 1 or d[-1] == n:
        print "NO"
        exit(0)
    for x in xrange(m-2):
        if d[x] + 2 == d[x+1] + 1 == d[x+2]:
            print "NO"
            exit(0)
    print "YES"
