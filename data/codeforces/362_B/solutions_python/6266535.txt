n,m=[int(x) for x in raw_input().split()]

if(m==0):print "YES"
else:

    d.sort()
    if d[0]==1 or d[m-1]==n:print "NO"
    else:
        for i in xrange(1,m-1):
            if d[i+1]-d[i]==1 and d[i]-d[i-1]==1:
                print "NO"
                break
