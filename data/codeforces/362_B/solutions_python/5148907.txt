n,m = map(int, raw_input().split())

if m==0:
    print "YES"
else:
    a = map(int, raw_input().split())
    a.sort()
    if a[0]==1 or a[m-1]==n:
        print "NO"
        exit()

    for k in xrange(m-2):
        if a[k+2]==a[k+1]+1 and a[k+1]==a[k]+1:
            print "NO"
            exit()
    print "YES"
