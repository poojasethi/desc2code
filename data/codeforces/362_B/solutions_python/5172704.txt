n,m = map(int,raw_input().split(' '))
if m == 0:
    print "YES"
else:
    s = map(int,raw_input().split(' '))
    if s.count(1) > 0 or s.count(n) > 0:
        print "NO"
    else:
        s.sort()
        for i in range(2,m):
            if s[i] == s[i-1] + 1 and s[i-1] == s[i-2] + 1:
                print "NO"
                exit(0)
        print "YES"
