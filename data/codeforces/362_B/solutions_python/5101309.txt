n,m=map(int,raw_input().split())
if m==0:
    print "YES"
    exit()
a=map(int,raw_input().split())
a.sort()
ans=1;
if a[0]==1 or a[-1]==n: ans=0;
for x in xrange(len(a)):
    if (x>=2 and a[x-2]+1==a[x-1] and a[x-1]+1==a[x]): ans=0;
print "YES" if ans==1 else "NO"
