def main():
	n, m = map(int, raw_input().split())
	if 0 == m:
		print "YES"
		return
	d = map(int, raw_input().split())
	d.sort()
	if 1 == d[0] or n == d[m - 1]:
		print "NO"
		return
	for i in xrange(2, m):
		if 1 == d[i] - d[i - 1] and 1 == d[i - 1] - d[i - 2]:
			print "NO"
			return
	print "YES"

if __name__ == "__main__": main()
