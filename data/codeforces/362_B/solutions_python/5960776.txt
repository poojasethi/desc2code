n, m = map(int, raw_input().split())
res = True

if (m > 0):
	d = map(int, raw_input().split())
	d.sort()
	res = not ((d[0] == 1) or (d[m-1] == n))
	i = 2
	while (i < m):
		if (d[i] == d[i - 1] + 1) and (d[i] == d[i - 2] + 2):
			res = False
			break
		else:
			i +=1
		
if (res):
	print "YES"
else:
	print "NO"
