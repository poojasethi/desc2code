r=lambda:map(int,raw_input().split())
n,m=r()
if m>0:
    a=sorted(r())
    if a[0]==1 or a[-1]==n:
        print"NO"
        exit(0)
    else:
        t=0
        for i in range(m-1):
            if a[i]+1==a[i+1]:
                t+=1
            else:
                t=0
            if t>1:
                print"NO"
                exit(0)
print"YES"
