def solve(n, o):
    if not o: return True
    o.sort()
    if o[0] == 1: return False
    if o[-1] == n: return False
    for i in range(2, len(o)):
        if o[i]-o[i-1]==1 and o[i-1]-o[i-2]==1: return False
    return True

n, m = map(int, raw_input().split())
if m > 0:
    o = map(int, raw_input().split())
else:
    o = []
print 'YES' if solve(n, o) else 'NO'
