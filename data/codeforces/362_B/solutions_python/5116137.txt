f = lambda: map(int, raw_input().split())
n, m = f()
if m:
  a = sorted(f())
  if a[0] == 1 or a[m - 1] == n:
    print "NO"
  else:
    c, d = 1, 1
    for i in range(1, m):
      if a[i] - a[i - 1] == 1:
        c += 1
        if c >= 3: d = 0
      else:
        c = 1
    print ["NO", "YES"][d]
else:
  print "YES"
