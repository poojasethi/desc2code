entrada = raw_input().split() 
n = int(entrada[0]) 
m = int(entrada[1]) 
sujas = []
if m >0:
    sujas = map(int,sorted(raw_input().split())) 
flag = 0 
if m >2: 
    for i in range(m-2): 
        if sujas[i] == sujas[i+1] - 1 == sujas[i+2] -2 : 
            flag = 1 
if flag == 1 or(1 in sujas) or n in sujas : 
    print "NO" 
elif flag ==0: 
    print "YES"
