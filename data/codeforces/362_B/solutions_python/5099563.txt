n,m=map(int,raw_input().split())
if m>0:
    l=map(int,raw_input().split())
else:
    l=[]
dic={}
f=0
for i in xrange(m):
    dic[l[i]]=True

    if l[i]+1 in dic and l[i]-1 in dic:
        f=1
    if l[i]+1 in dic and l[i]+2 in dic:
        f=1
    if l[i]-1 in dic and l[i]-2 in dic:
        f=1

if 1 in dic:
    f=1
if n in dic:
    f=1

if f==0:
    print "YES"
else:
    print "NO"
