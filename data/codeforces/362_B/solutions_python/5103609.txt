n, m = map(int, raw_input().split())

if m == 0:
	print 'YES'
else:
	a = sorted(map(int, raw_input().split()))
	if a[0] == 1 or a[-1] == n:
		print 'NO'
	else:
		c = 0
		p = a[0]
		for i in range(1, m):
			if a[i] == p + 1:
				c += 1
				if c == 2:
					c = -1
					break
			else:
				c = 0
			p = a[i]

		if c == -1:
			print 'NO'
		else:
			print 'YES'
