def main():
    n, m = map(int, raw_input().split())
    if m == 0:
        print "YES"
        return
    a = map(int, raw_input().split())
    a.sort()
    if a[0] == 1 or a[-1] == n:
        print "NO"
        return
    for i in xrange(2, m):
        if a[i-1] == a[i] - 1 and a[i-2] == a[i] - 2:
            print "NO"
            return
    print "YES"
main()
