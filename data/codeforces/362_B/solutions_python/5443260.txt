n,m=map(int,raw_input().split())
if m==0:
    print 'YES'
else:
    d=sorted(map(int,raw_input().split()))

    if d[0]==1 or d[-1]==n:
        print 'NO'
    else:
        for i in range(2,m):
            if d[i-2]+2==d[i-1]+1==d[i]:
                print 'NO'
                break
        else:
            print 'YES'
