import sys

n, m = map(int, raw_input().split())

if m == 0:
    print 'YES'
    sys.exit(0)

d = list(sorted(map(int, raw_input().split())))

if d[0] == 1 or d[-1] == n:
    print 'NO'
    sys.exit(0)

for i in xrange(len(d)-2):
    if d[i+1] == d[i]+1 and d[i+2] == d[i+1]+1:
        print 'NO'
        sys.exit(0)

print 'YES'
