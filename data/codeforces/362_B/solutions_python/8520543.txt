n,m = [int(i) for i in raw_input().split()]

if m != 0:
    l = [int(i) for i in raw_input().split()]
    l.sort()

else:
    print "YES"
    exit(0)

if l[0] == 1 or l[m-1] == n:
    print "NO"
    exit(0)

idx = 0

while idx < m-2:
    if l[idx] + 1 == l[idx+1] and l[idx+1] + 1== l[idx+2]:
        print "NO"
        exit(0)
    idx += 1

print "YES"
