n,m=map(int,raw_input().split())
if m!=0:
    ar=map(int,raw_input().split())
    ar=sorted(ar)
    if (1 in ar) or (n in ar):
        print "NO"
    else:
        flag=True
        for i in range(0,m-2):
            if ar[i]+1==ar[i+1] and ar[i+1]+1==ar[i+2]:
                print "NO"
                flag=False
                break
        if flag:
            print "YES"
else:
    print "YES"
