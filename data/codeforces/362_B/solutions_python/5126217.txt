I=lambda:map(int, raw_input().split())
n, m = I()
if m == 0: print 'YES'; exit()
a = sorted(I())
cnt, prev = 0, -1
for x in a:
    if x == 1 or x == n: print 'NO'; exit()
    if prev != -1 and prev + 1 == x: 
        cnt += 1
    else: cnt = 1
    if cnt == 3: print 'NO'; exit()
    prev = x
print 'YES'
