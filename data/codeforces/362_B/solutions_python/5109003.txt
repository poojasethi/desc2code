try:
    a = raw_input()
    m = int(a.split()[0])
    b = [int(x) for x in raw_input().split()]
    def ans(b):
        for i in xrange(len(b) - 2):
            if  (b[i+2]-b[i+1] == 1) and (b[i+1] - b[i] == 1):
                return True
        return False
    if (1 in b) or (m in b):
        print "NO"
    else:
        print "NO" if ans(sorted(b)) else "YES"
except:
    print "YES"
