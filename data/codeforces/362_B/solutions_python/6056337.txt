n, m = map(int,raw_input().split())

if m != 0:
    s = map(int,raw_input().split())
else:
    print 'YES'
    exit()

ss = sorted(s)

if 1 in s or n in s:
    print 'NO'
    exit()

k = 0
for i in range(m-1):
    if ss[i+1] - ss[i] == 1:
        k = k+1
        if k >= 2:
            print "NO"
            exit()
    else:
        k = 0

print "YES"
