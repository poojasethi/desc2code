#include <iostream>
#include <cstring>

using namespace std;

int dp[5005][5005];
bool p[5005][5005];

int main() {
    string s;
    cin >> s;

    int size = s.size();
    for (int len=1; len <= size; ++len)
        for (int i=1; i+len <= size+1; ++i) {
            int j = i+len-1;
            p[i][j] = s[i-1] == s[j-1];
            if (len > 2) p[i][j] = p[i][j] && p[i+1][j-1];
            dp[i][j] = p[i][j] + dp[i+1][j] + dp[i][j-1] - dp[i+1][j-1];
        }
    int q;
    cin >> q;
    while (q--) {
        int a, b;
        cin >> a >> b;
        cout << dp[a][b] << endl;
    }
    return 0;
}
