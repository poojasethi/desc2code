#include<bits/stdc++.h>
using namespace std;
bool ispal[5001][5001];
char s[5001];
int dp[5002][5002];
int main()
{
	cin>>s;
	int n=strlen(s);
	for(int i=0;i<n;i++)ispal[i][i]=1;
	for(int i=n-1;i>=0;i--)
	{
		for(int j=i+1;j<n;j++)
		{
			ispal[i][j]=((ispal[i+1][j-1] || i+1>j-1) & (s[i]==s[j]));
		}
	}
	for(int i=n-1;i>=0;i--)
	{
		dp[i][i]=1;
		for(int j=i+1;j<n;j++)
		{
			dp[i][j]=(ispal[i][j]+dp[i][j-1]+dp[i+1][j]-dp[i+1][j-1]);
		}
	}
	int q;
	cin>>q;
	int a,b;
	for(int i=0;i<q;i++)
	{
		scanf("%d%d",&a,&b);
		a--;
		b--;
		printf("%d\n",dp[a][b]);
	}
	return 0;
}
