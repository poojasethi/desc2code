#include<cstring>
#include<cstdio>
#include<iostream>
#include<string>
using namespace std;
const int mm=5005;
int a[mm][mm];
bool b[mm][mm];
int main(){
    string s;
    cin>>s;
    int i,j,k,w,d,q,l,r,n=s.length();
    for(w=0;w<=1;w++)
        for(i=0;i<n-w;i++){
            j=i+w,k=i;
            while(j>=0&&k<n&&s[j]==s[k])
                b[k++][j--]=1;
        }
    a[0][0]=1;
    for(i=1;i<n;i++){
        d=0;
        for(j=i;j>=0;j--)
            d+=b[i][j],a[j][i]=a[j][i-1]+d;
    }
    cin>>q;
    while(q--)
        cin>>l>>r,cout<<a[l-1][r-1]<<' ';
}
