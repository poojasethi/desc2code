#include <cstring>
#include <cstdio>
int f[5010][5010],i,j,n,m,x,y;
char s[5010];
int main()
{
	scanf("%s\n",s+1);n=strlen(s+1);
	for (i=1;i<=n;++i)
	{
		f[i][i]=1;
		if (i<n && s[i]==s[i+1]) f[i][i+1]=1;
	}
	if (i>2)
	for (i=n-2;i;--i)
		for (j=i+2;j<=n;++j)
			f[i][j]|=s[i]==s[j]&&f[i+1][j-1];
	for (i=n;i;i--)
		for (j=i;j<=n;++j)
			f[i][j]+=f[i+1][j]+f[i][j-1]-f[i+1][j-1];
	for (scanf("%d\n",&m);m--;)
	{
		scanf("%d%d\n",&x,&y);
		printf("%d\n",f[x][y]);
	}
}
