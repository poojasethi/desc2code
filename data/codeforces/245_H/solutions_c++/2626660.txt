#include<stdio.h>
#include<string.h>
char s[5005];
int dp[5005][5005];
int main()
{
	int i, j, k, x, y;
	int len = strlen(gets(s+1));
	for(i = 1; i <= len; i++)
	{
		for(j = k = i; j >= 1 && k <= len && s[j] == s[k]; j--, k++)
			dp[j][k]++;
		for(j = i, k = i+1; j >= 1 && k <= len && s[j] == s[k]; j--, k++)
			dp[j][k]++;
	}
	for(i = 1; i <= len; i++)
		for(j = i+1; j <= len; j++)
			dp[i][j] += dp[i][j-1];
	for(i = 1; i <= len; i++)
    	for(j = i-1; j >=  1; j--)
			dp[j][i] += dp[j+1][i];
	int Q;
	scanf("%d", &Q);
	while(Q--)
	{
		scanf("%d%d", &x, &y);
		printf("%d\n", dp[x][y]);
	}
	return 0;
}
