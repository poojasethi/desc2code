#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char s[5005];
int dp[5005][5005],y[5005][5005];
int main()
{
    scanf("%s",s);
    int n=strlen(s);
    for(int i=0;i<n;++i)y[i][i]=dp[i][i]=1;
    for(int k=1;k<n;++k)
        for(int i=0;i+k<n;++i)
        {
            int j=i+k;
            dp[i][j]=dp[i+1][j]+dp[i][j-1]-dp[i+1][j-1];
            if(s[i]==s[j]&&(y[i+1][j-1]||i+1>j-1))++dp[i][j],y[i][j]=1;
        }
    int q;
    scanf("%d",&q);
    while(q--)
    {
        int l,r;
        scanf("%d%d",&l,&r);
        --l,--r;
        printf("%d\n",dp[l][r]);
    }
}
