#include<iostream>
#include<string>
#include<string.h>
using namespace std;
#define maxn 5010
char str[maxn];
int cnt[maxn][maxn];
int len;

void check(int i,int j)
{
	while(i>=0&&j<len&&str[i]==str[j])
	{
		cnt[i][j]++;
		i--;
		j++;
	}
}

int main()
{
	int n,a,b;
	cin>>str;
	cin>>n;

	len=strlen(str);
	check(0,0);
	
	for(int i=1;i<len;i++)
	{
		check(i,i);
		check(i-1,i);
	}

	for(int i=1;i<len;i++)
		for(int j=0;j+i<len;j++)
		{
			cnt[j][j+i]+=cnt[j][j+i-1]+cnt[j+1][j+i]-cnt[j+1][j+i-1];
		}

	while(n--)
	{
		cin>>a>>b;
		cout<<cnt[a-1][b-1]<<endl;
	}
	
	return 0;
}
