#include<stdio.h>
#include<string.h>

int dp[5005][5005];
int is[5005][5005];
char s[10000];

int main()
{
	int i,j,k;
	scanf("%s",s+1);
	int n=strlen(s+1);
	for(i=1;i<=n;i++) dp[i][i]=1,is[i][i]=1;
	for(k=1;k<n;k++)
	{
		for(i=1;i+k<=n;i++)
		{
			int j=i+k;
			dp[i][j]+=dp[i][j-1]+dp[i+1][j];
			if(i+1<=j-1) dp[i][j]-=dp[i+1][j-1];
			if(s[i]==s[j])
			{
				if(i+1>j-1||is[i+1][j-1])
				{
					dp[i][j]++;
					is[i][j]=1;
				}
			}
		}
	}
	int q;
	scanf("%d",&q);
	while(q--)
	{
		int l,r;
		scanf("%d%d",&l,&r);
		printf("%d\n",dp[l][r]);
	}
	return 0;
}
