#include <bits/stdc++.h>
using namespace std;
int n;
struct node
{
	int a, l;
}sp[105];
bool cmp(node u, node v){return u.a < v.a;}
int f[105][105][2], ans = 0;
int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) scanf("%d%d", &sp[i].a, &sp[i].l);
	sort(sp + 1, sp + n + 1, cmp);
	sp[0].a = -200000000;
	for (int i = 0; i <= n; i++)
		for (int j = 0; j <= i; j++)
			for (int k = 0; k <= 1; k++)
			{
				int nowr = sp[j].a + sp[j].l * k;
				ans = max(ans, f[i][j][k]);
				int maxi = -0x7f7f7f7f, lastj = 0, lastk = 0;
				for (int mjy = i + 1; mjy <= n; mjy++)
					for (int xjy = 0; xjy <= 1; xjy++)
					{
						int nextr = sp[mjy].a + sp[mjy].l * xjy;
						if (nextr > maxi) maxi = nextr, lastj = mjy, lastk = xjy;
						f[mjy][lastj][lastk] = max(f[mjy][lastj][lastk], f[i][j][k] + min(sp[mjy].l, nextr - nowr) + maxi - nextr);
					}
			}
	printf("%d\n", ans);
	return 0;
}
