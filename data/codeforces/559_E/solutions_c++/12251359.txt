#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
int a[101],s[101],s1[102],s2[102],f[202],f1[202],d[102][102][204];
void qs(int q,int w){
  int e=q,r=w,t=a[q+(rand()%(w-q+1))],y;
  do{
  while(a[e]<t)e++;
  while(a[r]>t)r--;
  if(e<=r){
   y=a[e];a[e]=a[r];a[r]=y;
   y=s[e];s[e]=s[r];s[r]=y;
   e++;r--;}
  }while(e<=r);
  if(q<r)qs(q,r);
  if(e<w)qs(e,w);
}
void qs1(int q,int w){
  int e=q,r=w,t=f[q+(rand()%(w-q+1))],y;
  do{
  while(f[e]<t)e++;
  while(f[r]>t)r--;
  if(e<=r){
   y=f[e];f[e]=f[r];f[r]=y;
   y=f1[e];f1[e]=f1[r];f1[r]=y;
   e++;r--;}
  }while(e<=r);
  if(q<r)qs1(q,r);
  if(e<w)qs1(e,w);
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t,c,v;
  scanf("%d",&q);
  for(w=0;w<q;w++)
   scanf("%d%d",&a[w],&s[w]);
  a[q]=-100000001;
  s[q]=0;
  q++;
  qs(0,q-1);
  for(w=0;w<q;w++){
   f[w]=a[w]+s[w];
   f1[w]=w;
   f[w+q]=a[w];
   f1[w+q]=q+w;}
  qs1(0,q*2-1);
  for(w=0;w<q*2;w++)
   if(f1[w]<q)
    s1[f1[w]]=w;
   else
    s2[f1[w]-q]=w;
  for(w=0;w<=q;w++)
   for(e=0;e<=q;e++)
    for(r=0;r<q<<1;r++)
     d[w][e][r]=-1;
  d[0][q][0]=0;
  for(w=0;w<q;w++){
   for(r=0;r<q<<1;r++)
    if(d[w][q][r]>=0){
     for(e=w+1;e<q;e++){
      c=d[w][q][r];
      v=r;
      if(max(s2[w],r)<s2[e]){
       c+=min(s[e],f[s2[e]]-f[v]);
       v=s2[e];
       d[w][e][v]=max(d[w][e][v],c);}}
     c=d[w][q][r];
     v=r;
     if(f[r]<a[w]){
      c+=min(s[w],a[w]-f[r]);
      v=s2[w];
      d[w+1][q][v]=max(d[w+1][q][v],c);}
     c=d[w][q][r];
     v=r;
     if(v<s1[w]){
      c+=min(s[w],f[s1[w]]-f[v]);
      v=s1[w];}
     d[w+1][q][v]=max(d[w+1][q][v],c);}
   for(e=0;e<q;e++)
    for(r=0;r<q<<1;r++)
     if(d[w][e][r]>=0){
      c=d[w][e][r];
      if(e==w){
       d[w+1][q][r]=max(d[w+1][q][r],c);
       continue;}
      v=max(s2[w],r);
      if(v<s1[w]){
       c+=f[s1[w]]-f[v];
       v=s1[w];}
      d[w+1][e][v]=max(d[w+1][e][v],c);}}
  t=0;
  for(r=0;r<q<<1;r++)
   t=max(t,d[q][q][r]);
  cout<<t;
  return 0;}
