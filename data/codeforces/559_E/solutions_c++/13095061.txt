#include <cstdio>
#include <algorithm>
using namespace std;

#define n   105
#define For(i,a,b)  for(int i=a;i<=b;i++)

int     N,Ans,F[n][n][2];
struct  Nod{int x,l;}A[n];

bool    Cmp(Nod a,Nod b)    {return a.x<b.x;}

int     main(){
        scanf("%d",&N);
        For(i,1,N)  scanf("%d%d",&A[i].x,&A[i].l);
        sort(A+1,A+N+1,Cmp);    A[0].x=-(1<<30);

        For(i,0,N)  For(j,0,i)  For(k,0,1)  {
            int Pr=A[j].x+k*A[j].l,Max=-1,a,b;
            Ans=max(Ans,F[i][j][k]);

            For(t,i+1,N)    For(p,0,1)  {
                int Nr=A[t].x+p*A[t].l;
                if  (Nr>Max)Max=Nr,a=t,b=p;
                F[t][a][b]=max(F[t][a][b],F[i][j][k]+min(A[t].l,Nr-Pr)+Max-Nr);
            }
        }
        printf("%d\n",Ans);
}
