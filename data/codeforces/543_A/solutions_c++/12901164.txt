#include<bits/stdc++.h>
using namespace std;

int dp[505][505];
int arr[505];
int n,m,b,mod;
int main()
{
	scanf("%d%d%d%d",&n,&m,&b,&mod);
	for(int i=1;i<=n;i++)
		cin>>arr[i];
	for(int i=0;i<=b;i++)
	{
		dp[0][i]=1;
	}
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			for(int k=arr[i];k<=b;k++)
			{
				dp[j][k]=(dp[j][k]+dp[j-1][k-arr[i]])%mod;
			}
		}
	}
	cout<<dp[m][b]<<endl;
	return 0;
}
