#include<bits/stdc++.h>
using namespace std;
int i,j,k,dp[505][1005],a[505],n,m,b,mod;
main()
{
    scanf("%d %d %d %d",&n,&m,&b,&mod);
    for (i=0;i<n;i++) scanf("%d",&a[i]);
    for (i=0;i<502;i++) dp[0][i]=1;
    for (i=0;i<n;i++)
        for (j=0;j<501;j++)
            for (k=0;k<501;k++)
                (dp[j+1][k+a[i]]+=dp[j][k])%=mod;
    printf("%d",dp[m][b]);
}
