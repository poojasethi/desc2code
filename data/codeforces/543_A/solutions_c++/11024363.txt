#include<bits/stdc++.h>
using namespace std;

int n,m,b,a[505],k,i,j;
long long mod,res;
long long ans[505][505];

main() {
 cin>>n>>m>>b>>mod;
 for (i=1;i<=n;i++) cin>>a[i];
 ans[0][0]=1;
 for (k=1;k<=n;k++)
  {
   for (i=1;i<=m;i++)
    for (j=a[k];j<=b;j++)
     ans[i][j]=(ans[i][j]+ans[i-1][j-a[k]])%mod;
  }
 for (j=0;j<=b;j++)
  res=(res+ans[m][j])%mod;
  cout<<res<<endl;
}
