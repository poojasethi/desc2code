#include <bits/stdc++.h>
using namespace std;
int n, m, b, l, a, x[510][510];
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin >> n >> m >> b >> l;
    for(int i=0; i<=b; i++)x[0][i]=1;
    for(int i=1; i<=n; i++){
        cin >> a;
        for(int j=1; j<=m; j++){
            for(int k=a; k<=b; k++){
                x[j][k]=(x[j][k]+x[j-1][k-a])%l;
            }
        }
    }
    cout << x[m][b];
}
