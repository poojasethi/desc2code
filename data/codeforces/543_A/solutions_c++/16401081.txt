#include <iostream>

using namespace std;

int a[510];

int d[510][510];

int main()
{
	int n, m, b;

	long long int mod;

	cin >> n >> m >> b >> mod;

	d[0][0] = 1;

	for(int i=0;i<n;i++)
		cin >> a[i];

	for(int i =0;i<n;i++)
		for(int k=0;k <= (b-a[i]);k++)
			for(int j=0;j < m;j++)
				d[j+1][k+a[i]] = (d[j+1][k+a[i]]+d[j][k])%mod;
	int ans =0;

	for(int i=0;i<= b;i++)
		ans =( ans+ d[m][i])%mod;
	cout << ans << endl;
}
