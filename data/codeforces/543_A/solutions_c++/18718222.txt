#include <cstdio>
int f[510][510],i,j,k,n,m,b,mo,x,ans;
int main()
{
	scanf("%d%d%d%d\n",&n,&m,&b,&mo);
	f[0][0]=1;
	for (i=1;i<=n;++i)
	{
		scanf("%d",&x);
		for (j=1;j<=m;++j)
			for (k=x;k<=b;++k) f[j][k]=(f[j][k]+f[j-1][k-x])%mo;
	}
	ans=0;
	for (i=0;i<=b;++i) ans=(ans+f[m][i])%mo;
	printf("%d\n",ans);
}
