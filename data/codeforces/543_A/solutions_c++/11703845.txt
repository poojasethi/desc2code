#include<cstdio>
#include<cstring>
using namespace std;
int n,m,b,mod,c[510];
int f[510][510];
int main()
{
	scanf("%d%d%d%d",&n,&m,&b,&mod);
	for(int i=1; i<=n; i++)
		scanf("%d",&c[i]);
	f[0][0]=1;
	for(int i=1; i<=n; i++)
		for(int j=1; j<=m; j++)
			for(int k=c[i]; k<=b; k++)
				f[j][k]=(f[j][k]+f[j-1][k-c[i]])%mod;
	int ans=0;
	for(int i=0; i<=b; i++)
		ans=(ans+f[m][i])%mod;
	printf("%d\n",ans);
	return 0;
}
