#include  <cstdio>

int mod;
int n,m,b,a[509];
int f[509][509];

int main()
{
	scanf("%d%d%d%d",&n,&m,&b,&mod);
	for (int i=1;i<=n;++i) scanf("%d",a+i);
	f[0][0]=1;
	for (int k=1;k<=n;++k)
	{
		for (int i=1;i<=m;++i)
		{
			for (int j=a[k];j<=b;++j)
			{
				f[i][j]+=f[i-1][j-a[k]];
				if (f[i][j]>=mod) f[i][j]-=mod;
			}
		}
	}
	int ans=0;
	for (int j=0;j<=b;++j) if ((ans+=f[m][j])>=mod) ans-=mod;
	printf("%d",ans);
	return 0;
}
