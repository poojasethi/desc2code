#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<cmath>
using namespace std;
int n,m,k,mo,i,j,x,ans,f[501][501];
int main()
{
		scanf("%d%d%d%d\n",&n,&m,&k,&mo);
		f[0][0]=1;
		for (;n;n--)
		{
			scanf("%d",&x);
			for (i=1;i<=m;i++)
				for (j=k;j>=x;j--)
					f[i][j]=(f[i][j]+f[i-1][j-x])%mo;	
		}
		for (i=0;i<=k;i++) ans=(ans+f[m][i])%mo;
		printf("%d",ans);
}
