#include <stdio.h>
int n,m,b,M,w[512][1024];
int main(){
  scanf("%d%d%d%d",&n,&m,&b,&M);
  w[0][0]=1;
  for(int i=0,a;i<n&&1==scanf("%d",&a);++i){
    for(int j=0;j<m;++j)for(int k=0;k<=b;++k){
      (w[j+1][k+a]+=w[j][k])%=M;
    }
  }
  int t=0;
  for(int i=0;i<=b;++i)(t+=w[m][i])%=M;
  printf("%d\n",t);
}
