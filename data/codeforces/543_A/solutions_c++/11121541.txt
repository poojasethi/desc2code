#include<cstdio>
using namespace std;
int dp[502][502];
int n,m,b,p;
int a[502],ans;
int main()
{
	scanf("%d%d%d%d",&n,&m,&b,&p);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	dp[0][0]=1;
	int cur=0;
	for(int i=1;i<=n;i++)
	{
		dp[0][0]=1;
		for(int j=0;j<=m;j++) for(int k=0;k<=b-a[i];k++) if(dp[j][k]) dp[j+1][k+a[i]]=(dp[j+1][k+a[i]]+dp[j][k])%p;
	}
	for(int j=0;j<=b;j++) ans=(ans+dp[m][j])%p;
	printf("%d\n",ans);
	return 0;
}
