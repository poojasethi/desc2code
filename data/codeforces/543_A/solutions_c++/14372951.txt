#include<bits/stdc++.h>
using namespace std;
long long mod,respuesta;
long long ans[510][510];
int n,m,b,a[510];


main() {

 cin>>n>>m>>b>>mod;
 for (int i=1;i<=n;i++) {cin>>a[i];}

 ans[0][0]=1;

for (int k=1;k<=n;k++){
for (int i=1;i<=m;i++){
for (int j=a[k];j<=b;j++){
     ans[i][j]=(ans[i][j]+ans[i-1][j-a[k]])%mod;}}}



 for (int j=0;j<=b;j++){respuesta=(respuesta+ans[m][j])%mod;}


  cout<<respuesta<<endl;
}
