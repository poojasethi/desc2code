#include<bits/stdc++.h>
using namespace std;
#define pb(x) push_back(x)
int dp[502][502];
int main()
{
    int m, n, b, a, mod;
    cin>>m>>n>>b>>mod;
    for(int i = 0; i<=b; i++)
        dp[0][i] = 1;
    while(m--) {
        cin>>a;
        for(int i = 1; i<=n; i++) {
            for(int j = a; j<=b; j++) {
                dp[i][j] = (dp[i][j] + dp[i-1][j-a])%mod;
            }
        }
    }
    cout<<dp[n][b]<<endl;
}
