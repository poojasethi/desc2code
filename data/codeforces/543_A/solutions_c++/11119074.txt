#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll n,m,b,mod;

ll a;
ll dp[510][510];

int main()
{
	cin >> n >> m >> b >> mod;


   for(int i=0;i<=505;i++)
   {
   	 dp[0][i]=1;
   }

    for(int i=1;i<=n;i++)
    {
    	cin >> a;

    	for(int j=1;j<505;j++)
    	{
    		for(int k=a;k<505;k++)
    		{
    			dp[j][k]=(dp[j][k]+dp[j-1][k-a])%mod;
    		}
    	}
    }
    
    
    cout<<dp[m][b]<<endl;

	return 0;
}
