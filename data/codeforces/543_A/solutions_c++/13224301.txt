 #include<bits/stdc++.h>
using namespace std;
typedef long long LL;
LL f[510][510],mod,ans;
int n,m,b,a[510];
int main()
{
	scanf("%d%d%d",&n,&m,&b);cin>>mod;
	for (int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	f[0][0]=1;
	for (int i=1;i<=n;i++)
		for (int j=1;j<=m;j++)
			for (int k=a[i];k<=b;k++)
				f[j][k]=(f[j][k]+f[j-1][k-a[i]])%mod;
	for (int i=0;i<=b;i++)
		ans=(ans+f[m][i])%mod;
	cout<<ans<<endl;
	return 0;
}
