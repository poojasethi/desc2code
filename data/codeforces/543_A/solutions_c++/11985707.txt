#include<stdio.h>
#include<string.h>
using namespace std;
int dp[501][501];
int main(){
	int n,m,b,mod;
	while(scanf("%d%d%d%d",&n,&m,&b,&mod)==4){
		memset(dp,0,sizeof(dp));
		for(int i=0;i<=b;i++)
			dp[0][i]=1;
		for(int i=0;i<n;i++){
			int d;
			scanf("%d",&d);
			for(int j=0;j<m;j++){
				for(int k=0;k+d<=b;k++){
					dp[j+1][k+d]=(dp[j+1][k+d]+dp[j][k])%mod;
				}
			}
		}
		printf("%d\n",dp[m][b]);
	}
}
