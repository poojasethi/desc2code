from heapq import *
input()
d = {}
for r in map(int, raw_input().split()):
	d[r] = d.get(r, 0) - 1
h = [(v, k) for k, v in d.items()]
heapify(h)
r = []
while len(h) > 2:
	l = [(v + 1, k) for v, k in [heappop(h) for _ in range(3)]]
	r.append(sorted(k for v, k in l)[ : : -1])
	[heappush(h, x) for x in l if x[0]]
print len(r)
print '\n'.join(' '.join(map(str, x)) for x in r)
