from heapq import *
n = input()
a = sorted(map(int, raw_input().split()))
pre = -1
ball = []
for i in a:
	if i == pre:
		ball[-1][0] -= 1
	else:
		ball.append([-1, i])
	pre = i
heapify(ball)
b = []
ans = 0
while len(ball) > 2:
	ans += 1
	c = [heappop(ball), heappop(ball), heappop(ball)]
	b.append([i[1] for i in c])
	for i in c:
		if i[0] < -1:
			heappush(ball, [i[0] + 1, i[1]])
print ans
for i in b:
	i.sort(reverse = True)
	print ' '.join(map(str, i))
