from heapq import *
from collections import defaultdict as d
n=input()
s=map(int,raw_input().split())
c=d(int)
for e in s:
    c[e]-=1
c=map(lambda (x,y):(y,x),c.items())
heapify(c)
r=[]
while len(c) > 2:
    l = map(heappop, [c]*3)
    r.append(map(lambda x:x[1], l))
    [heappush(c, (x+1,y)) for x,y in l if x < -1]
print len(r)
print '\n'.join(map(lambda x:' '.join(map(str,sorted(x,reverse=True))), r))
