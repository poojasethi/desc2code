def get(k, ns):
    ns1 = []
    prev = -1
    for n in ns:
        if n != prev:
            tmp = 1
            prev = n
        else:
            tmp += 1
        
        if tmp <= k:
            ns1.append(n)
    
    return ns1

def check(k, ns):
    if len(get(k, ns)) >= 3*k:
        return True
    else:
        return False
