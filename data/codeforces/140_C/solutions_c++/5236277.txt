#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
#include<set>
#define N 120000
using namespace std;
int f[N][3],a[N],l,r,mid,n,i;
bool check(int mid){
	int i,v,j;
	for(i=1;i<=mid;++i)f[i][0]=a[i];
	j=1; v=0;
	for(i=mid+1;i<=n;++i){
		if(a[i]>f[j][v]){
			f[j][v+1]=a[i];
			j++;
			if(j>mid)j=1,v++;
			if(v==2)return 1;
		}
	}
	return 0;
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i)scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	l=1; r=n/3;
	while(l<=r){
		mid=(l+r)/2;
		if(check(mid))l=mid+1;
		else r=mid-1;
	}
	check(l-1);
	printf("%d\n",l-1);
	for(i=1;i<=l-1;++i)printf("%d %d %d\n",f[i][2],f[i][1],f[i][0]);
}
