#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
#include <cstdlib>
#include <algorithm>
#include <vector>


using namespace std;
const int M = 1e5 + 5;

int a[M],b[M],c[M],num[M];
int n;
int chack(int m){
	int i = 0,t = 0;
	while(i < m){
		c[i++] = num[t++];
	}
	i = 0;
	while(i < m){
		while(c[i] == num[t]) t++;
		if(t >= n) return 0;
		b[i++] = num[t++];
	}
	i = 0;
	while(i < m){
		while(b[i] == num[t]) t++;
		if(t >= n) return 0;
		a[i++] = num[t++];
	}
	return 1;
}
int main()
{
	cin>>n;
	for(int i = 0;i < n;++i){
		cin>>num[i];
	}
	sort(num,num + n);
	int L = 0,R = n/3;
	while(L < R){
		int m = (L + R +  1)/2;
		if(chack(m)) L = m;
		else R = m-1;
	}
	chack(L);
	cout<<L<<endl;
	for(int i = 0;i < L;++i){
		cout<<a[i]<<" "<<b[i]<<" "<<c[i]<<endl;
	}
	return 0;
}
