#include <iostream>
#include <map>
#include <utility>
#include <algorithm>
#include <queue>

using namespace std;

int N,snowman[50000][3],result;
map<int,int> snowball;
priority_queue<pair<int,int> > ball;

int main()
{
	cin >> N;
	for (int i=0;i<N;i++)
	{
		int j;
		cin >> j;
		snowball[j]++;
	}
	for (map<int,int>::iterator it=snowball.begin();it!=snowball.end();it++)
		ball.push(make_pair(it->second,it->first));
	while (ball.size()>=3)
	{
		pair<int,int> b[3];
		for (int i=0;i<3;i++)
		{
			b[i]=ball.top();
			ball.pop();
		}
		for (int i=0;i<3;i++)
		{
			snowman[result][i]=b[i].second;
			if (--b[i].first)
				ball.push(b[i]);
		}
		sort(snowman[result],snowman[result]+3);
		result++;
	}
	cout << result << endl;
	for (int i=0;i<result;i++)
		cout << snowman[i][2] << " " << snowman[i][1] << " " << snowman[i][0] << endl;
	return 0;
}
