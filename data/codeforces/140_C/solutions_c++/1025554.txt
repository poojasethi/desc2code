#include<stdio.h>
#include<queue>
#include<algorithm>
using namespace std;
typedef pair<int,int> pii;
int n,i,j,k,aa[100000],b[33333][3];
int main(){
	scanf("%d",&n);
	for(i=0;i<n;i++)scanf("%d",&aa[i]);
	sort(aa,aa+n);
	priority_queue<pii>a;
	pii x,y,z;
	k=1;j=aa[0];
	for(i=1;i<n;i++)
	if(aa[i]==j)k++;else{
		a.push(make_pair(k,j));
		k=1;j=aa[i];
	}
	a.push(make_pair(k,j));
	i=0;
	for(;a.size()>2;){
		x=a.top();a.pop();b[i][0]=x.second;//	printf("%d\t%d\n",x.first,x.second);
		y=a.top();a.pop();b[i][1]=y.second;//	printf("%d\t%d\n",y.first,y.second);
		z=a.top();a.pop();b[i][2]=z.second;//	printf("%d\t%d\n",z.first,z.second);
		sort(b[i],b[i]+3);
		if(x.first-- >1)a.push(x);
		if(y.first-- >1)a.push(y);
		if(z.first-- >1)a.push(z);
		i++;
	}
	printf("%d\n",i);
	for(j=0;j<i;j++)printf("%d %d %d\n",b[j][2],b[j][1],b[j][0]);
	return 0;
}
