#include<bits/stdc++.h>
using namespace std;

int a[100005],b[100005];
	vector<pair<int , int> >vec;

int check(int x)
{
		int sum = 0,i;
	for (i = 0; i < vec.size(); ++ i)
		sum += min (vec[i].first , x);
	if (sum >= 3 * x)
		return 1;
	return 0;
}
int main()
{
	int n,i,l,r,j,mid;
	cin>>n;
	for(i=0;i<n;i++)
	cin>>a[i];
	sort(a,a+n);

	for(i=0;i<n;i++)
	{
		if(i==0||a[i]!=a[i-1])
		{
			vec.push_back(make_pair(0,a[i]));
		}
		vec.back().first++;
	}
	l=0,r=n/3+1;
	while(l<=r)
	{
		 mid=(l+r)/2;
		if(check(mid)==0)
		r=mid-1;
		else
		l=mid+1;
	}
	if(check(mid)==0)
	mid--;
	int p=0;
	cout<<mid<<"\n";
	for(i=0;i<vec.size();i++)
	{
		for(j=0;j<min(mid,vec[i].first);j++)
		{
			b[p++]=vec[i].second;
		}
	}
	for(i=0;i<mid;i++)
	{
		cout<<b[i+2*mid]<<" "<<b[i+mid]<<" "<<b[i]<<"\n";
	}
	return 0;
}
