#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int n,a[100001],x[40000],y[40000],z[40000];
bool kt(int m){
    int i,t=0;
    for(i=0;i<m;x[i++]=a[t++]);
    for(i=0;i<m;y[i++]=a[t++]){
        while(x[i]==a[t]) t++;
        if(t==n) return false;
    }
    for(int i=0;i<m;z[i++]=a[t++]){
        while(y[i]==a[t]) t++;
        if(t==n) return false;
    }
    return true;
}
int main(){
    cin>>n;
    for(int i=0;i<n;i++) cin>>a[i];
    sort(a,a+n);
    int lo=0,hi=n/3;
    while(lo<hi){
        int m=(lo+hi+1)/2;
        if(kt(m)) lo=m; else hi=m-1;
    }
    kt(lo);
    cout<<lo<<endl;
    int i;
    for(i=0;i<lo;i++) cout<<z[i]<<" "<<y[i]<<" "<<x[i]<<endl;
}
