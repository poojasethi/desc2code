#include <cstdio>
#include <cstring>
#include <algorithm>
#define N 100005
using namespace std;

int n, m, x, v[N], a[N], s0[N], s1[N], s2[N], ret;

int doit(int num){
	memset(v, 0, sizeof(v));
	for (int i=0; i<num; i++) s0[i] = a[i];
	x = num;
	for (int i=0; i<num; i++){
		while (x < n && (a[x] <= s0[i] || v[x])) x++;
		if (x == n) return 0;
		s1[i] = a[x];
		v[x] = 1;
	}
	x = num;
	for (int i=0; i<num; i++){
		while (x < n && (a[x] <= s1[i] || v[x])) x++;
		if (x == n) return 0;
		s2[i] = a[x];
		v[x] = 1;
	}
	return 1;
}

int main(){
	scanf("%d", &n);
	for (int i=0; i<n; i++) scanf("%d", &a[i]);
	sort(a, a+n);

	for (int i=0, j=n/3, k; j>=i;){
		k = (i+j) >> 1;
		if (doit(k)) i = k+1, ret = k;
		else j = k-1;
	}

	printf("%d\n", ret);
	doit(ret);
	for (int i=0; i<ret; i++) printf("%d %d %d\n", s2[i], s1[i], s0[i]);
	return 0;
}
