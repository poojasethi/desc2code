#include<bits/stdc++.h>
using namespace std;

typedef pair<int,int> ii;

int main() {
	int n, r; cin>>n;
	map<int,int> mp;
	for (int i=0; i<n; i++)
		cin>>r,mp[r]++;
	priority_queue<ii> que;
	for(map<int,int>::iterator it=mp.begin(); it!=mp.end(); it++)
		que.push(make_pair((*it).second, (*it).first));

	int ret[33333][3], k=0;
	while(que.size() >= 3) {
		ii x=que.top(); que.pop(); ret[k][0]=x.second;
		ii y=que.top(); que.pop(); ret[k][1]=y.second;
		ii z=que.top(); que.pop(); ret[k][2]=z.second;
		sort(ret[k], ret[k]+3);
		if(x.first-->1) que.push(x);
		if(y.first-->1) que.push(y);
		if(z.first-->1) que.push(z);
		k++;
	}
	cout << k << endl;
	for (int i=0; i<k; i++)
		cout << ret[i][2] << " " << ret[i][1] << " " << ret[i][0] << endl;
	return 0;
}
