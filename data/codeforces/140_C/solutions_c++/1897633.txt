#include <cstdio>
#include <algorithm>

using namespace std;

const int maxn=100005;
int n,a[maxn],list[maxn],zm[maxn];

bool cmp(int x,int y)
{
	if (x>y) return true;
	else return false;
}

bool can(int k)
{
	int sum=0;
	for (int i=1;i<=list[0];++i) sum+=min(list[i],k);
	if (k*3<=sum) return true;
	else return false;
}

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;++i) scanf("%d",&a[i]);
	sort(a+1,a+n+1,cmp);
	for (int i=1;i<=n;++i)
		if ((i==1)||(a[i]!=a[i-1]))
		{
			int r=i;
			while ((r<n)&&(a[r+1]==a[r])) ++r;
			list[++list[0]]=r-i+1;
			zm[++zm[0]]=a[r];
		}
	int l=0,r=n/3+1;
	while (l!=r)
	{
		int mid=(l+r)>>1;
		if (can(mid)) l=mid+1;
		else r=mid;
	}
	
	for (int i=1;i<=list[0];++i)
		for (int j=1;j<=min(list[i],l-1);++j) a[++a[0]]=zm[i];
	printf("%d\n",l-1);
	for (int i=1;i<=a[0]/3;++i)
		printf("%d %d %d\n",a[i],a[i+a[0]/3],a[i+a[0]/3+a[0]/3]);
	return 0;
}
