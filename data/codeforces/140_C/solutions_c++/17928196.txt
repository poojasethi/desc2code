#include <bits/stdc++.h>

using namespace std;

set <pair <int, int> > s;
map <int, int> m;
vector <int> ans[40000];

void update(int x) {
	if (m[x])
			s.erase(s.find(make_pair(m[x], x)));
	m[x]++;
	if (m[x])
		s.insert(make_pair(m[x], x));
}

main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++) {
		int x;
		scanf("%d", &x);
		if (m[x])
			s.erase(s.find(make_pair(m[x], x)));
		s.insert(make_pair(m[x] - 1, x));
		m[x]--;
	}
	int k = 0;
	while (s.size() >= 3) {
		set <pair<int, int> > :: iterator it = s.begin();
		for (int i = 0; i < 3; i++) {
			ans[k].push_back(it->second);
			it++;
		}
		for (int i = 0; i < 3; i++)
			update(ans[k][i]);
		k++;
	}
	cout << k << endl;
	for (int i = 0; i < k; i++) {
		sort(ans[i].begin(), ans[i].end());
		for (int j = 2; j > -1; j--) {
			cout << ans[i][j] << " \n" [j == 0];
		}
	}
}
