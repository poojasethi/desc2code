#include<stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;
#define maxn 100005
int n;
int a[maxn],b[maxn],c[maxn],num[maxn];
int check(int m){
	int i=0,t=0;
	while(i<m){
		c[i++]=num[t++];
	}
	i=0;
	while(i<m){
		while(c[i]==num[t])
		t++;
		if(t>=n)
			return 0;
		b[i++]=num[t++];
	}
	i=0;
	while(i<m){
		while(b[i]==num[t])
		t++;
		if(t>=n)
			return 0;
		a[i++]=num[t++];
	}
	return 1;
}
int main(){
	int i,L,M;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&num[i]);
	}
	sort(num,num+n);
	L=0;M=n/3;
	while(L<M){
		int m=(L+M+1)/2;
		if(check(m))
		L=m;
		else{
			M=m-1;
		}
	}
	check(L);
	printf("%d\n",L);
	for(i=0;i<L;i++){
		printf("%d %d %d\n",a[i],b[i],c[i]);
	}
	return 0;
}
