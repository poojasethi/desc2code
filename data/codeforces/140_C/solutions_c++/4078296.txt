#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;

const int maxn=100001;
int n,a[maxn],b[maxn];

int cmp(const void *b,const void *a)
{
	return *(int *)a-*(int *)b;
}

bool check(int num)
{
	int i,j=0,g=0;
	for (i=1;i<=n;i++)
	{
		if (a[i]!=a[i-1]) g=i-1;
		if (i-g<=num)
		{
			j++;
			b[j]=a[i];
		}
	}
	if (j>=num*3)
		return true;
	else return false;
}

int main()
{
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	int i,j,k;
	cin >> n;
	for (i=1;i<=n;i++)
		cin >> a[i];
	qsort(a+1,n,4,cmp);
	
	int mid,head=0,last=n/3+1;
	while (head!=last && head!=last-1)
	{
		mid=(head+last)/2;
		if (check(mid))//mid can be ans
		{
			head = mid;
		}
		else //mid can't be ans
		{
			last = mid;
		}
	}
	cout << head << endl;
	check(head);
	for (i=1;i<=head;i++)
	{
		cout << b[i] <<" "<< b[i+head] << " " << b[i+head+head]<< endl;
	}
	return 0;
}
