#include <cstdio>
#include <algorithm>
using namespace std;

int n,a[100001],x[40000],y[40000],z[40000];

bool gao(int m){
    int t=0,i;
    for(i=0;i<m;x[i++]=a[t++]);

    for(i=0;i<m;y[i++]=a[t++]){
        while(x[i]==a[t]) t++;
        if(t==n) return false;
    }
    for(i=0;i<m;z[i++]=a[t++]){
        while(y[i]==a[t]) t++;
        if(t==n) return false;
    }
    return true;
}

int main(){	
	int i;
    scanf("%d",&n);
    for( i=0;i<n;i++) scanf("%d",a+i);
    sort(a,a+n);


    int lo=0,hi=n/3;


    while(lo<hi){
        int m=(lo+hi+1)/2;
        if(gao(m))
			lo=m;
		else
			hi=m-1;
    }
    gao(lo);
    printf("%d\n",lo);
    for(i=0;i<lo;i++) printf("%d %d %d\n",z[i],y[i],x[i]);

	return 0;
}
