#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

int n,num[100010];
int x[100010][3];
bool solve(int m)
{
    memset(x,0,sizeof(x));
    int lev = 0,pos = 0;
    for(int i = n - 1;i >= 0 && lev <=2;i--)
    {
        if(lev == 0)x[pos++][lev] = num[i];
        else
        {
            if(x[pos][lev - 1] > num[i]){x[pos][lev] = num[i],pos++;}
        }
        if(pos == m)lev++,pos = 0;
    }
    return lev == 3;
}


int main()
{
    scanf("%d",&n);
    for(int i = 0;i < n;i++)scanf("%d",num+i);
    int l = 0,r = n/3+2;
    sort(num,num + n);
    while(r-l>1)
    {
        int m = (l + r)>>1;
        if(solve(m))l = m;
        else r = m;
    }
    solve(l);
    printf("%d\n",l);
    for(int i = 0;i <l;i++)printf("%d %d %d\n",x[i][0],x[i][1],x[i][2]);
    return 0;
}
