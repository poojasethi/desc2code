#include <bits/stdc++.h>

using namespace std;

const int maxl = 1000 * 100 + 5;

int a [ maxl ];
int n;

int ans [ maxl ];

inline bool chk ( int k ){
	int cnt = 0, lst = -1, x = 3 * k;
	for ( int i = 0 ; i < n && x ; i++ ){
		if ( lst != a [ i ] ){
			cnt = 0;
			lst = a [ i ];
		}

		if ( cnt < k ){
			ans [ x - 1 ] = a [ i ];
			cnt++;
			x--;
		}
	}

	if ( x )
		return false;
	return true;
}

int main (){
	cin >> n;
	for ( int i = 0 ; i < n ; i++ )
		cin >> a [ i ];

	sort ( a, a + n );
	reverse ( a, a + n );

	int lo = 0, hi = n / 3 + 1;
	while ( hi - lo > 1 ){
		int mid = (lo + hi) >> 1;

		if ( chk ( mid ) )
			lo = mid;
		else
			hi = mid;
	}

	chk ( lo );
	cout << lo << endl;
	for ( int i = 0 ; i < lo ; i++ )
		cout << ans [ 2 * lo + i ] << " " << ans [ lo + i ] << " " << ans [ i ] << '\n';
}
