#include<cstdio>
#include<cstring>
#define mem(a,b) memset(a,b,sizeof(a))
#include<algorithm>
using namespace std;

const int N = 101111;

int a[N], n;
int x[N], y[N], z[N];

bool can(int m) {
	int p=0;
	for(int i=0;i<m;i++) x[i]=a[p++];
	for(int i=0;i<m;i++) {
		while(p<n&&a[p]==x[i]) p++;
		if(p>=n) return false;
		y[i]=a[p++];
	}
	for(int i=0;i<m;i++) {
		while(p<n&&a[p]==y[i]) p++;
		if(p>=n) return false;
		z[i]=a[p++];
	}
	return true;
}

int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++) scanf("%d",&a[i]);
	sort(a,a+n);

	int l=0,r=n/3,ans;
	while(l<=r) {
		int m=(l+r)/2;
		if(can(m)) {
			l=m+1;
			ans=m;
		} else r=m-1;
	}
	can(ans);
	printf("%d\n",ans);
	for(int i=0;i<ans;i++) {
		printf("%d %d %d\n",z[i],y[i],x[i]);
	}

	return 0;
}
