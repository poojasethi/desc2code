#include <bits/stdc++.h>

using namespace std;

const int MAXN = 100005;
int n, a[MAXN], b[MAXN];

int Check(int x)
{
    int i = 0, k = x + 1, j = n - x;
    while(k <= n - x)
    {
        if (a[i + 1] < a[k] && a[k] < a[j + i + 1]) b[++i] = a[k];
        if ( i == x) break;
        k++;
    }
    return (i == x);

}
int main()
{
    ios_base::sync_with_stdio(0);
    
    cin >> n;
    for(int i = 1; i <= n; i++) cin >> a[i];
    sort( a + 1, a + n + 1);

    int l = 1;
    int r = n/3;
    int m, res = 0;
    while(l <= r)
    {
        m = ( l + r ) / 2;
        if (Check(m)) res = m, l = m + 1;
        else r = m - 1;
    }
    cout << res <<endl;
    if (!res) return 0;
    Check(res);
    for(int i = 1; i <= res; i++)
        cout << a[i + n - res] << " " << b[i] <<" " << a[i] << endl;
    return 0;
}
