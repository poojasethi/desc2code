#include<cstdio>
#include<cstring>
#include<cmath>
#include<map>
#include<queue>
#include<set>
#include<vector> 
#include<algorithm>
#include<utility>
using namespace std;
const int qq = 1e5+10;
int num[qq];
int main()
{
	int n;scanf("%d",&n);
	priority_queue< pair<int,int> >PQ;
	for(int i=0; i<n; ++i)	scanf("%d",&num[i]);
	sort(num,num+n);
	for(int i=0,cnt=1; i<n; ++i){
		if(num[i]==num[i+1])	++cnt;
		else{
			PQ.push(make_pair(cnt, num[i]));
			cnt=1;
		}
	}
	pair<int,int>tmp[3];
	int ans[qq][3];
	int nans=0;
	while(PQ.size()>=3){
		for(int i=0; i<3; ++i){
			tmp[i]=PQ.top();
			ans[nans][i]=tmp[i].second;
			PQ.pop();
		}
		for(int i=0; i<3; ++i)
			if(--tmp[i].first)
				PQ.push(tmp[i]);
		sort(ans[nans], ans[nans]+3);
		++nans;
	}
	printf("%d\n",nans);
	for(int i=0; i<nans; ++i)
		printf("%d %d %d\n",ans[i][2],ans[i][1],ans[i][0]);
	return 0;
}
