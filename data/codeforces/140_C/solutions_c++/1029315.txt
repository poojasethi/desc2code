#include <iostream>
#include <stdio.h>
#include <map>

using namespace std;

map<int, int> m;
int a[100000][3];

int main()
{ int n;
 cin>>n;
for (int i=0; i<n;i++) {int j; cin>>j; m[j]++;}

map<int,int>::iterator it;

// for (it=m.begin(); it!=m.end();it++)  {cout<<it->first<<" "<<it->second<<endl;}

int L=0, R=n/3;
int ans;
while (L<R)
{
  ans=(L+R+1)/2;
  int s=0;
  for (it=m.begin(); it!=m.end();it++){int k=min(ans,it->second); s+=k;}
  if (s>=ans*3) L=ans; else R=ans-1;

}

int n1=L;
cout<<n1<<endl;
int i=0, j=0;
for (it=m.begin(); it!=m.end();it++)
  {int k=min(n1,(it->second));
   while (k)
   {
    a[i][j]=it->first;
    i++; if (i==n1) {j++; i=0;}
    k--;
    if (j==3) break;
   }
  if (j==3) break;
 }

for (int i=0; i<n1;i++) cout<<a[i][2]<<" "<<a[i][1]<<" "<<a[i][0]<<endl;

return 0;
}
