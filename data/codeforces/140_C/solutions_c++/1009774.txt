#include <cstdio>
#include <vector>
#include <queue>
#include <map>
#include <utility>
#include <algorithm>
using namespace std;

typedef pair<int,int> PII;
#define X first
#define Y second

int n, r;
map<int,int> mp;
priority_queue<PII> q;
vector<int> a, b, c;

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &r);
		mp[r]++;
	}
	map<int,int>::iterator it;
	for (it=mp.begin(); it!=mp.end(); ++it) {
		q.push(PII(it->Y, it->X));
	}
	while (q.size() >= 3) {
		PII v[3];
		int r[3];
		for (int i = 0; i < 3; ++i) {
			v[i] = q.top();
			r[i] = v[i].Y;
			q.pop();
		}
		sort(r, r+3);
		a.push_back(r[2]);
		b.push_back(r[1]);
		c.push_back(r[0]);
		for (int i = 0; i < 3; ++i) {
			if (--v[i].X) {
				q.push(v[i]);
			}
		}
	}
	printf("%d\n", a.size());
	for (int i = 0; i < a.size(); ++i) {
		printf("%d %d %d\n", a[i], b[i], c[i]);
	}
}
