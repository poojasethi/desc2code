#include<cstdio>
#include<algorithm>
using namespace std;
int n,a[100001],x[40000],y[40000],z[40000],i,t;
bool G(int m){
    t=0;
    for(i=0;i<m;x[i++]=a[t++]);
    for(i=0;i<m;y[i++]=a[t++]){
        while(x[i]==a[t])
			t++;
        if(t==n)
			return 0;
    }
    for(i=0;i<m;z[i++]=a[t++]){
        while(y[i]==a[t])
			t++;
        if(t==n)
			return 0;
    }
    return 1;
}
main(){
    scanf("%d",&n);
    for(i=0;i<n;i++)
		scanf("%d",a+i);
    sort(a,a+n);
    int L=0,H=n/3;
    while(L<H){
        int m=(L+H+1)/2;
        if(G(m))L=m;
		else H=m-1;
    }
    G(L);
    printf("%d\n",L);
    for(i=0;i<L;i++)
		printf("%d %d %d\n",z[i],y[i],x[i]);
}
