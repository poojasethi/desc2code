# include <cstdio>
# include <iostream>
# include <algorithm>
using namespace std;
int s[100001],a[35000],b[35000],c[35000],n;
bool change(int m)
{
	int t,i;
	for(t=0,i=0;i<m;)
		a[i++]=s[t++];
	for(i=0;i<m;){
		while(s[t]==a[i] && t<n)
			t++;
		if(t>=n)
			return false;
		b[i++]=s[t++];
	}
	for(i=0;i<m;){
			while(s[t]==b[i] && t<n)
				t++;
			if(t>=n)
				return false;
			c[i++]=s[t++];
		}
	return true;
}


int main()
{
	int i,sum,flag,m;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&s[i]);
	sort(s,s+n);
	sum=0;
	flag=n/3;
	while(sum<flag)
	{
		m=(sum+flag+1)/2;
		if(change(m))
			sum=m;
		else
			flag=m-1;
	}
	change(sum);
	printf("%d\n",sum);
	for(i=0;i<sum;i++)
		printf("%d %d %d\n",c[i],b[i],a[i]);

}
