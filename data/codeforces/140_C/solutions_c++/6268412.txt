#include <cstdio>
#include <algorithm>
using namespace std;

int arr[100010],a[100010],b[100010],n,i,j,k,x,y,l,r,ans[35000][10];


int main()
{
	scanf("%d",&n);
	for (i=1;i<=n;++i) scanf("%d",arr+i);
	sort(arr+1,arr+n+1);
	for (i=1,j=1;i<=n;++i)
		if (arr[i]==arr[i-1])
			++b[j-1];
		else
		{
			a[j++]=arr[i];
			++b[j-1];
		}
	l=0; r=n/3;
	n=j-1;
	while (l<r)
	{
		k=(l+r+1)>>1;
		for (x=0,i=1;i<=n;++i)
			x+=min(b[i],k);
		if (k*3<=x) l=k; else r=k-1;
	}
	k=l;
	printf("%d\n",k);
	for (x=1,y=0,i=1;i<=n&&y<3;++i)
		for (j=1;j<=b[i]&&j<=k&&y<3;++j)
		{
			ans[x++][y]=a[i];
			if (x>k) {x=1; ++y;}
		}
	for (i=1;i<=k;++i)
		printf("%d %d %d\n",ans[i][2],ans[i][1],ans[i][0]);
	return 0;
}
