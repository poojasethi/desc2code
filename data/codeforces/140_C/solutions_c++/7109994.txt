#include<stdio.h>
#include<map>
using namespace std;
int main()
{
	__int64 ball[34000][3];
	int n,i,j,k,l,r,mid,num,sum;
	while(scanf("%d",&n) != EOF)
	{
		map<int,int> flag;
		map<int,int>::iterator p;
		for(i=0;i<n;i++)
		{
			scanf("%d",&r);
			flag[r]++;
		}
		l=0;r=n/3;
		while(l<r)
		{
			mid=(l+r+1)/2;
			sum=0;
			for(p=flag.begin();p!=flag.end();p++)
				sum+=p->second<mid?p->second:mid;
			if(sum>=3*mid)
				l=mid;
			else
				r=mid-1;
		}
		num=l;i=0;j=0;
		for(p=flag.begin();p!=flag.end();p++)
		{
			for(k=p->second<num?p->second:num;k>0;k--)
			{
				ball[i++][j]=p->first;
				if(i==num)
				{
					i=0;
					j++;
				}
				if(j==3)
					break;
			}
			if(j==3)
				break;
		}
		printf("%d\n",num);
		for(i=0;i<num;i++)
			printf("%I64d %I64d %I64d\n",ball[i][2],ball[i][1],ball[i][0]);
	}
	return 0;
}
