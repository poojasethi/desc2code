#include <cstdio>
#include <algorithm>
#include <queue>
#include <map>
using namespace std;
int n, m; 
int ans[33333][3];
map<int, int> cnt;
priority_queue<pair<int, int> > Q;
int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		int x;
		scanf("%d", &x);
		++cnt[x];
	}
	for (map<int, int>::iterator e = cnt.begin(); e != cnt.end(); ++e)
		Q.push(make_pair(e->second, e->first));
	while (Q.size() > 2) {
		pair<int, int> A = Q.top(); Q.pop();
		pair<int, int> B = Q.top(); Q.pop();
		pair<int, int> C = Q.top(); Q.pop();
		ans[m][0] = A.second;
		ans[m][1] = B.second;
		ans[m][2] = C.second;
		sort(ans[m], ans[m] + 3);
		m++;
		if (--A.first)
			Q.push(A);
		if (--B.first)
			Q.push(B);
		if (--C.first)
			Q.push(C);
	}
	printf("%d\n", m);
	for (int i = 0; i < m; ++i)
		printf("%d %d %d\n", ans[i][2], ans[i][1], ans[i][0]);
}
