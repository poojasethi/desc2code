#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define N 1000000007
#define MAX 100001
ll i,j,k,l,m,n,t;
map<ll,ll>a;
priority_queue<pair<ll,ll> >q;
vector<vector<ll> > msd;
int main()
{
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>t;
		a[t]++;
	}
	for(map<ll,ll> :: iterator it=a.begin();it!=a.end();++it)
	{
		q.push(make_pair(it->second, it->first) );
	}
	while(q.size() >= 3)
	{
		vector<ll>res;
		for(i=0;i<3;i++)
		{
			res.push_back(q.top().second);
			q.pop();
		}
		msd.push_back(res);
		for(i=0;i<3;i++)
		{
			if(--a[res[i]])
				q.push(make_pair(a[res[i]],res[i]));
		}
	}
	cout<<msd.size()<<endl;
	for(i=0;i<msd.size();i++)
	{
		sort(msd[i].begin(), msd[i].end());
		cout<<msd[i][2]<<" "<<msd[i][1]<<" "<<msd[i][0]<<endl;
	}
	return 0;
}
