#include<iostream>
#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
int a[100000+10],ans[100000+10][3];
struct node
{
	int x,y;
	node(){}
	node(int m,int n)
	{
		x=m;y=n;
	}
	bool operator <(node a)const
	{
		return y<a.y;
	}
};
priority_queue <node> q;
int main()
{
	int n;
	scanf("%d",&n);
	int i,j,val,m;
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	sort(a,a+n);
	int flag=0;
	i=0;
	for(;i<n;)
	{
		 val=a[i];m=1;
		for(j=i+1;j<n;j++)
		{
			if(val==a[j])
			m++;
			else
				break;
		}
		q.push(node(val,m));
		i=j;
	}
	while(q.size()>2)
	{
		node k[3];
		for(i=0;i<3;i++)
		{
			k[i]=q.top();
			ans[flag][i]=-k[i].x;
			q.pop();
			k[i].y--;
		}
		sort(ans[flag],ans[flag]+3);
		for(i=0;i<3;i++)
		{
			if(k[i].y>0)
				q.push(k[i]);
		}
		flag++;
	}
	printf("%d\n",flag);
	for(i=0;i<flag;i++)
		printf("%d %d %d\n",-ans[i][0],-ans[i][1],-ans[i][2]);
	return 0;
}
