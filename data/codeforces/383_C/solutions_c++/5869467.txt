#include <cstdio>
#define maxn 400005

int N,M,tot,a[maxn],lk[maxn],next[maxn],head[maxn],d[maxn],L[maxn],R[maxn],s[maxn];

void add(int u,int v,int t) {lk[t]=v,next[t]=head[u],head[u]=t;}

void init()
{
	scanf("%d%d",&N,&M);
	for (int i=1; i<=N; i++) scanf("%d",&a[i]);
	for (int i=1,u,v; i<N; i++) scanf("%d%d",&u,&v),add(u,v,i),add(v,u,i+N-1);
}

void dfs(int u)
{
	L[u]=++tot;
	for (int i=head[u],v; v=lk[i],i; i=next[i]) if (!d[v]) d[v]=d[u]+1,dfs(v);
	R[u]=++tot;
}

void add(int x,int v) {for (;x<=tot; x+=x&-x) s[x]+=v;}
int sum(int x) {int v=0; for (;x; x-=x&-x) v+=s[x]; return v;}

void doit()
{
	d[1]=1,dfs(1);
	for (int i=1,t,u,v; i<=M; i++)
	{	
		scanf("%d%d",&t,&u);
		if (t==1) scanf("%d",&v),v=d[u]&1?v:-v,add(L[u],v),add(R[u],-v);
		else printf("%d\n",a[u]+(d[u]&1?1:-1)*sum(L[u]));
	}
}

int main()
{
	init();
	doit();
}
