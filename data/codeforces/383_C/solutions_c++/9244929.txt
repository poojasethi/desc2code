#include <stdio.h>
#include <vector>
#define MAX 200010

int n, m, cnt;
int a[MAX], in[MAX], out[MAX], l[MAX];
int bit[MAX*2];
std::vector<int> vec[MAX];

void dfs(int par,int now, int sw){
	int i;
	l[now]=sw;
	in[now]=++cnt;
	for (i=0;i<vec[now].size();i++){
		if (vec[now][i]!=par) dfs(now, vec[now][i], 1-sw);
	}
	out[now]=++cnt;
}

int main(void){
	int i, j, p, q;
	int op, x, val;
	scanf("%d %d", &n, &m);
	for (i=1;i<=n;i++) scanf("%d", &a[i]);
	for (i=1;i<n;i++){
		scanf("%d %d", &p, &q);
		vec[p].push_back(q);
		vec[q].push_back(p);
	}
	dfs(0, 1, 0);
	int sum;
	for(;m>=1;m--){
		scanf("%d %d", &op, &x);
		if (op==1){
			scanf("%d", &val);
			if (l[x]) val*=(-1);
			for (i=in[x];i<=cnt;i+= i&-i) bit[i]+=val;
			for (i=out[x];i<=cnt;i+=i&-i) bit[i]-=val;
		}else{
			sum=0;
			for (i=in[x];i;i-=i&-i) sum+=bit[i];
			if (l[x]) sum*=(-1);
			printf("%d\n", a[x]+sum);
		}
	}
	return false;
}
