#include <cstdio>

#define N 200000
#define M 1000

int a[N], b[N], bb[N], bc[N], c[2][2 * N / M + 1], d[N], e[N], f[N << 1], h[N << 1], n, m, i, x, y, t, k;
bool g[N];

void go(int v = 0, int p = 0)
{
	d[v] = k;
	g[v] = p;
	h[k] = v;
	++k;
	for (int i = b[v]; i; i = bb[i])
		go(bc[i], p ^ 1);
	e[v] = k;
	h[k] = v;
	++k;
}

int main()
{
	scanf("%d%d", &n, &m);

	for (i = 0; i < n; i ++)
		scanf("%d", &a[i]);

	for (i = 1; i < n; i ++)
	{
		scanf("%d%d", &x, &y);
		--x, --y;
		if (y < x)
			k = x, x = y, y = k;
		bc[i] = y;
		bb[i] = b[x];
		b[x] = i;
	}

	go();

	while (m--)
	{
		scanf("%d", &t);
		if (t == 1)
		{
			scanf("%d%d", &x, &y);
			--x;
			if (g[x])
				y = -y;
			for (i = d[x]; i <= e[x]; i ++)
				if (i % M == 0 && e[x] >= i + M)
				{
					c[0][i / M] += y;
					c[1][i / M] -= y;
					i += M - 1;
				}
				else
					if (i == d[h[i]])
						f[i] += !g[h[i]] ? y : -y;
		}
		else
		{
			scanf("%d", &x);
			--x;
			printf("%d\n", a[x] + c[g[x]][d[x] / M] + f[d[x]]);
		}
	}

	return 0;
}
