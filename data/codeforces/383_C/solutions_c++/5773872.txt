#include <cstdio>
#include <vector>

#define N 200000
#define M 475

int a[N], c[2][2 * N / M + 1], d[N], e[N], f[N << 1], h[N << 1], n, m, i, x, y, t, k;
bool g[N];
std::vector <int> b[N];

void go(int v = 0, int p = 0)
{
	d[v] = k;
	g[v] = p;
	h[k] = v;
	++k;
	for (int i = 0; i < b[v].size(); i ++)
		go(b[v][i], p ^ 1);
	e[v] = k;
	h[k] = v;
	++k;
}

int main()
{
	scanf("%d%d", &n, &m);

	for (i = 0; i < n; i ++)
		scanf("%d", &a[i]);

	for (i = 1; i < n; i ++)
	{
		scanf("%d%d", &x, &y);
		--x, --y;
		if (y < x)
			k = x, x = y, y = k;
		b[x].push_back(y);
	}

	go();

	while (m--)
	{
		scanf("%d", &t);
		if (t == 1)
		{
			scanf("%d%d", &x, &y);
			--x;
			k = g[x] ? -y : y;
			for (i = d[x]; i <= e[x]; i ++)
				if (i % M == 0 && e[x] >= i + M)
				{
					c[0][i / M] += k;
					c[1][i / M] -= k;
					i += M - 1;
				}
				else
					f[i] += g[h[i]] == g[x] ? y : -y;
		}
		else
		{
			scanf("%d", &x);
			--x;
			printf("%d\n", a[x] + c[g[x]][d[x] / M] + f[d[x]]);
		}
	}

	return 0;
}
