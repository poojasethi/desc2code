#include <cstdio>
#include <vector>
using namespace std;

typedef int T[200001];
T a, l, r, g, t;
int n, m, k = 1;
vector<int> e[200001];

void dfs(int u, int v, int f) {
	g[v] = f;
	l[v] = k++;
	for (int i = 0; i < e[v].size(); ++i)
		if (e[v][i] != u)
			dfs(v, e[v][i], !f);
	r[v] = k;
}

int main() {
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; ++i)
		scanf("%d", a + i);
	for (int i = 1; i < n; ++i) {
		int v, u;
		scanf("%d%d", &v, &u);
		e[v].push_back(u);
		e[u].push_back(v);
	}
	dfs(0, 1, 0);
	while (m--) {
		int y, x, val;
		scanf("%d%d", &y, &x);
		if (y == 1) {
			scanf("%d", &val);
			if (g[x])
				val = -val;
			for (int i = l[x]; i <= n; i += i & -i)
				t[i] += val;
			for (int i = r[x]; i <= n; i += i & -i)
				t[i] -= val;
		} else {
			int ans = 0;
			for (int i = l[x]; i; i ^= i & -i)
				ans += t[i];
			if (g[x])
				ans = -ans;
			printf("%d\n", a[x] + ans);
		}
	}
}
