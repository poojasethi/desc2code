//In the Name of God
#include<iostream>
#include<vector>
#define P push_back
#define Z int
using namespace std;
const Z M=200200,N=262144;
Z n,q,k=1,T,X[M],Y[M],a[M],D[M],s[N*2][2];
bool W[M];
vector<Z> g[M];
inline Z B(Z v){
  W[v]=1;
  X[v]=Y[v]=k++;
  for(Z i=0;i<g[v].size();i++)
    if(!W[g[v][i]])
      D[g[v][i]]=D[v]+1,Y[v]=B(g[v][i]);
  return Y[v];
}
inline void A(Z S,Z F,Z L,Z f,Z l,Z V,Z t){
  if(F==f && L==l){
    s[S][t]+=V;
    return;
  }
  Z m=(F+L)/2;
  if(l<=m)
    A(S*2,F,m,f,l,V,t);
  else if(m<f)
    A(S*2+1,m+1,L,f,l,V,t);
  else
    A(S*2,F,m,f,m,V,t),A(S*2+1,m+1,L,m+1,l,V,t);
}
Z main(){
  ios_base::sync_with_stdio(0);
  cin>>n>>q;
  for(Z i=1;i<=n;i++)
    cin>>a[i];
  for(Z v,u,i=1;i<n;i++)
    cin>>v>>u,g[v].P(u),g[u].P(v);
  B(1);
  for(Z t,x,V;q--;){
    cin>>t;
    if(t==1){
      cin>>x>>V,A(1,1,N,X[x],Y[x],V,D[x]&1);
    }
    else{
      cin>>x;
      V=0;
      for(Z I=N+X[x]-1,i=D[x]&1;I;V+=s[I][i]-s[I][!i],I/=2);
      cout<<V+a[x]<<endl;
    }
  }
  return 0;
}
