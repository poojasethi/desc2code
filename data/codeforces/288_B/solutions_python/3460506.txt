MOD = 1000000007

def exp(x, n):
	if n == 0:
		return 1
	if n & 1 == 0: # n is even
		return (exp(x, n >> 1) ** 2) % MOD
	else: # n is odd
		return (x * exp(x, n - 1)) % MOD

n, k = map(int, raw_input().strip().split())
R = [1, 2, 9, 64, 625, 7776, 117649, 2097152]
print (R[k-1] * exp(n - k, n - k)) % MOD
