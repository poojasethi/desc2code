#coding=utf-8



n = int(raw_input())

words = [ raw_input() for i in range(n)]

s = raw_input()

t = "<3" + "<3".join(words) + "<3"

#print t
s, t = t, s
ti = 0
si = 0
for ch in s:
  while ti < len(t):
    if t[ti] != ch:
      ti += 1
    else:
      break
  if ti < len(t) and t[ti] == ch:
    ti += 1
    si += 1
  else:
    break

if si == len(s):
  print "yes"
else:
  print "no"
