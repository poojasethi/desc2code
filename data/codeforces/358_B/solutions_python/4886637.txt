#/usr/bin/env python2

import sys

n = int(raw_input())
words = []
for x in xrange(n):
	words.append(raw_input())
letter = raw_input()

heart = "<3"
answer=heart+(heart.join(words))+heart

def lazyanswer(words):
	for w in words:
		yield "<"
		yield "3"
		for c in w:
			yield c
	yield "<"
	yield "3"

i = 0
for c in lazyanswer(words):
	while (i<len(letter)) and (c != letter[i]):
		i+=1
	if(i==len(letter)):
		print("no")
		sys.exit()
	i+=1
print("yes")
