r=lambda:map(str,raw_input().split())
n=int(raw_input())
str1=['<'+'3']
for i in range(n):
    a=str(r()).strip('[\'').strip('\']')
    str1.append(a)
    str1.append('<')
    str1.append('3')
str2=str(r()).strip('[\'').strip('\']')
s=''.join(str1)
i=0
j=0
while j<len(str2):
    if i==len(s):
        break
    if s[i]==str2[j]:
        i=i+1
    j=j+1
print 'yes' if i==len(s) else 'no'
