import sys

#sys.stdin = open("2.txt", "r")

n = input()
a = []
for i in xrange(n):
    s = raw_input()
    a.append(s)
a[n-1] = a[n-1] + "<3"
a[0] = "<3" + a[0]

s = "<3".join(a)
t = raw_input()
i, j = 0, 0
n, m = len(s), len(t)

cnt = 0
while i < n and j < m:
    while j < m and s[i] != t[j]:
        j += 1
    if j < m and s[i] == t[j]:
        cnt += 1
        i += 1
        j += 1
if cnt == n:
    print "yes"
else:
    print "no"
