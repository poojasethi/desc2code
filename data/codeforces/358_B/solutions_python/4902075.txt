def main():
    import sys
    n = input()
    msg = "<3"
    for i in xrange(n):
    	msg += raw_input() + "<3"
    check = raw_input()
    if len(msg)>len(check):
    	print "no"
    	return
    indx = 0
    l = len(msg)
    for i in check:
    	if indx==l: break
    	if i==msg[indx]: indx += 1
    if indx == l:
    	print "yes"
    else:
    	print "no"
main()
