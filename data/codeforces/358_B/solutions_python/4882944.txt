def check(s,l):
    i,j=0,0
    n,m = len(s),len(l)
    while i < n and j < m: 
        if(s[i] == l[j]):
            i,j = i+1,j+1
        else : 
            j = j + 1
    return i == n

n = int(raw_input())
s = ""
for _ in xrange(n):
    s += "<3" + raw_input()
s += "<3"
l = raw_input()
if check(s,l): print "yes"
else : print "no"
