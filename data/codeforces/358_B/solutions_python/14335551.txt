w = int(raw_input())
s = ""

for i in xrange(w):
    w1 = raw_input()
    s = s + "<3" + w1
    
s += "<3"
msg = raw_input()
index = 0

for i in xrange(len(msg)):
    if (msg[i] == s[index]):
        index += 1
    if (index >= len(s)):
        break

if (index == len(s)):
    print "yes"
else:
    print "no"
