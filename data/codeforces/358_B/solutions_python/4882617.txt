from sys import stdin
def main():
    n = int(stdin.readline())
    s = '<3'.join([''] + [stdin.readline().strip() for _ in xrange(n)] + [''])
    t = stdin.readline().strip()
    i, L = 0, len(t)
    for c in s:
        while i < L and t[i] != c:
            i = i + 1
        if i == L:
            print "no"
            return
        i = i + 1
    print "yes"
main()
