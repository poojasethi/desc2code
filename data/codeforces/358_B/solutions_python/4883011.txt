#!/usr/bin/env python

n = int(raw_input())
words = ['']
for i in xrange(n):
    w = raw_input().strip()
    words.append(w)
words.append('')

msg = '<3'.join(words)
enc_msg = raw_input().strip()


i = 0
for c in enc_msg:
    if i == len(msg):
        break
    if msg[i] == c:
        i += 1
if i == len(msg):
    print "yes"
else:
    print "no"
