def solver():

    word = "<3"
    for _ in xrange(n):
        word += raw_input() + "<3"

    line = raw_input()
    
    for c in word:
        x = line.find(c)
        if x == -1:
            print "no"
            return

        line = line[x+1:]

    print "yes"
