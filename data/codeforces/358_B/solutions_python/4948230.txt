n = int(raw_input())
word = []

for i in range(n):
	word.append(raw_input())

msg = raw_input()


flag = [False, False]
iword = 0

for c in msg:
	if not flag[0] or not flag[1]:
		if c == '<': flag[0] = True
		if c == '3': flag[1] = flag[0]
	else:
		if iword == n: continue
		if c == word[iword][0]:
			word[iword] = word[iword][1: ]
		if len(word[iword]) == 0:
			iword += 1
			flag[0] = flag[1] = False

if iword == n and len(word[-1]) == 0 and flag[0] and flag[1]:
	print "yes"
else:
	print "no"
