words = int(raw_input())
text = ""
for i in xrange(words):
    word = raw_input()
    text += "<3%s" % word

text += "<3"
message = raw_input()
index = 0

for i in xrange(len(message)):
    if message[i] == text[index]:
        index += 1
    if index >= len(text):
        break

if index == len(text):
    print "yes"
else:
    print "no"
