n = input()
txt = '<3'
for _ in range(n):
    txt += raw_input() + '<3'
msg = raw_input()
ind = 0
count = -1
k = len(msg)
for i in range(len(txt)):
    count+=1
    while(count < k and msg[count]!=txt[i]):
        count+=1
    if count == k :
        ind = 1
        break
if ind == 0:
    print 'yes'
else:
    print 'no'
