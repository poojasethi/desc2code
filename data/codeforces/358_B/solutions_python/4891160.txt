n = int(raw_input())
word = []
for x in xrange(n):
    word.append(raw_input())
correct = "<3"+"<3".join(word)+"<3"
line = raw_input()
run = 0
for x in xrange(len(line)):
    if run == len(correct):
        break
    if correct[run] == line[x]:
        run+=1
print "yes" if run == len(correct) else "no"
