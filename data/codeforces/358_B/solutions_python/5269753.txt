def is_possible():
    count = int(raw_input(''))
    nums = [0] * count
    for i in xrange(count):
        nums[i] = raw_input('')
    target = '<3' + '<3'.join(nums) + '<3'
    matched_so_far = 0
    sample = raw_input('')
    for char in sample:
        if char == target[matched_so_far]:
            matched_so_far += 1
        if matched_so_far == len(target):
            return True
    return False

if is_possible():
    print 'yes'
else:
    print 'no'
