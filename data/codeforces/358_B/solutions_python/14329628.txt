n = int(raw_input())
words = []

for i in xrange(n): words.append(raw_input())
encoded = raw_input()
correct = "<3" + "<3".join(words) + "<3"

i = 0
possible = False
for c in encoded:
    if c == correct[i]:
        i += 1
        if i == len(correct):
            possible = True
            break

if possible: print "yes"
else: print "no"
