
import sys

n = int(raw_input())
result = "<3"
for i in xrange(n):
    result += raw_input().strip() + "<3"

input = raw_input()

completed = 0
for chr in input:
    if chr == result[completed]:
        completed += 1
    if completed == len(result):
        print "yes"
        sys.exit(0)

print "no"
