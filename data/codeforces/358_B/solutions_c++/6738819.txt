#include<iostream>
using namespace std;
int main ()
{
  int n;  cin>>n;
  string s,c,ans;
  for(int i=0;i<n;i++)
  { ans+="<3"; cin>>s; ans+=s; }
  ans+="<3"; cin>>c;
  for(int i=0;i<c.size();i++)
  { if(c[i]!=ans[i])
     {c.erase(i,1); i--;}
  }
  if(c==ans)cout<<"yes";
  else cout<<"no";
 return 0;
}
