#include<iostream>
#include<string>
using namespace std;

int n,sz;
string s,temp;

int main ()
{
	cin>>n;
	s="<3";
	while (n--)
	{
		cin>>temp;
		s+=temp;
		s+="<3";
	}
	cin>>temp;
	for (int i=0;temp[i];i++)
	{
		if (sz<s.size() && s[sz]==temp[i])
			sz++;
	}
	if (sz==s.size())
		cout<<"yes"<<endl;
	else
		cout<<"no"<<endl;
	return 0;
}
