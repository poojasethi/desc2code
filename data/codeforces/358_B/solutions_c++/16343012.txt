#include <bits/stdc++.h>

using namespace std;

int main(){
	string real="<3",s,s1,l="<3";
	int n;
	cin>>n;
	
	for(int i=1;i<=n;i++){
		cin>>s1;
		real+=s1;
		real+=l;
	}
	
	cin>>s;
	int cnt=0;
	for(int i=0;i<s.size();i++){
		if(s[i]==real[cnt])cnt++;
	}
	if(cnt==real.size())cout<<"yes";
	else cout<<"no";
	
	return 0;
}
