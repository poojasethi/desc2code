#include <bits/stdc++.h>
using namespace std;

int main(){
  int n;
  string all = "<3", foo, bar;
  cin >> n;
  for (int i = 0; i < n; i++)
    cin >> foo, all += foo + "<3";
  cin >> bar;
  int i = 0, j = 0;
  for (; i < bar.size(); i++){
    if (bar[i] == all[j] && j < all.size())
      j++;
  }
  (j == all.size())? puts("yes"): puts("no");
}
