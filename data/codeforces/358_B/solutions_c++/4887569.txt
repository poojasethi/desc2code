#include<bits/stdc++.h>

using namespace std;
int n,u,ou,bas;
string str,orj,old;
int main()
{
	int i,j;
	cin>>n;
	orj="<3";
	for(i=0; i<n; i++)
	{
		cin>>str;
		orj+=str;
		orj+="<3";
	}
	cin>>old;
	ou=old.size();
	u=orj.size();
		for(j=0; j<ou; j++)
		{
			if(orj[bas]==old[j])
			{
				bas++;
				if(bas==u) { cout<<"yes"; return 0; }
			}
		}
	cout<<"no";


	return 0;
}
