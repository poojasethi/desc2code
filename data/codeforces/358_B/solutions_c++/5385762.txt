#include <bits/stdc++.h>

using namespace std;

int main ()
{
 string str, s, wait = "";
 int n;
 cin >> n;
 for (int i = 0; i < n; i++) cin >> str, wait += "<3" + str; wait += "<3";
 cin >> s;
 int j = 0;
 for (int i = 0; i < s.size(); i++)
  if (s[i] == wait [j])
   j++;
 if (j == wait.size()) cout << "yes"; else cout << "no";
 return 0;
}
