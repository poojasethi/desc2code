#include <bits/stdc++.h>
using namespace std;
int main()
{
	string ans;
	ans += "<3";
	int i,n;
	scanf("%d",&n);
	string x;
	while(n--)
	{
		cin >> x;
		ans+=x;
		ans+="<3";
	}
	int j=0;
	string  y;
	cin >> y;
	int l2=strlen(y.c_str());
	for(i=0;i<l2;i++)
		if(y[i]==ans[j]) j++;
	if(j==strlen(ans.c_str())) cout<<"yes"<<endl;
	else cout<<"no"<<endl;
	return 0;
}
