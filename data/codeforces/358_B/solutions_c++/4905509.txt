#include <string>
#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    int n;
    cin >> n;
    string s="<3",t;
    for(int i=1;i<=n;i++){
        cin>>t;
        s=s+t+"<3";
    }
    cin>>t;
    int j=0;
    for(int i=0;t[i];i++)
        if(t[i]==s[j]) j++;
    if(j==s.size()) printf("yes\n");
    else printf("no\n");
    return 0;
}
