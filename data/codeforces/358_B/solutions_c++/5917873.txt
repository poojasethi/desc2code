#include<bits/stdc++.h>
using namespace std;

int n,i;
string s,a="<3";

int main()
{
	scanf("%d",&n);
	
	for(i=0;i<n;i++)
	{
		cin >> s;
		a+=s+"<3";
	}
	
	cin >>s;
	
	int m=0;
	
	for(i=0;i<s.size();i++)
	{
		m+=(s[i]==a[m]);
	}
	
	if(m==a.size())
	{
		printf("yes\n");
	}
	else
	{
		printf("no\n");
	}
	
	return 0;
}
