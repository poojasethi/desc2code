#include <iostream>
using namespace std;

int main()
{
	int n;
	string a, s = "<3", w;
	cin >> n;
	for (int i = 0; i < n; i++){
		cin >> a;
		s += a;
		s += "<3";
	}
	cin >> w;
	int i = 0, j = 0;
	while (i < w.size()){
		if (w[i] == s[j])
			j++;
		i++;
	}
	if (j == s.size())
		cout << "yes";
	else
		cout << "no";
	cout << endl;
	//cout << j << s[j - 1] << endl;
}
