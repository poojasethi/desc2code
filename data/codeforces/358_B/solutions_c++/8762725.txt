#include<bits/stdc++.h>
using namespace std;

int n;
string s, s1, x;

main() {
    scanf("%d", &n);
    for(int i=0; i<n; i++){
        cin>>x;
        s+="<3"+x;
    }s+="<3";
    cin>>s1;
    int j=0;
    for(int i=0; i<s1.length()&&j<s.length(); i++){
        if(s1[i]==s[j]) j++;
    }
    (j==s.length()) ? puts("yes") : puts("no");
}
