#include<cstdio>
#include<string>
#include<iostream>
using namespace std;
string ch1,ch2,s;
int main() {
  int n;
  scanf("%d",&n);
  ch1+="<3";
  for(int i=0;i<n;i++) {
    cin>>s;
    ch1+=(s+"<3");
  }
  cin>>ch2;
  //cout<<ch1<<endl<<ch2<<endl;
  int N=ch2.size(),c=0;
  for(int i=0;i<N;i++) if(ch2[i]==ch1[c]) c++;
  if(c==ch1.size()) printf("yes");
  else printf("no");

}
