#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s,a,m="<3";
	int n,i,j,l;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>s;
		m+=s;
		m+="<3";
	}
	cin>>a;
	l=a.length();
	int l1=m.length();
	{
		
		for(i=0,j=0;i<l && j<l1;i++)
		{
			if(m[j]==a[i])
				j++;
		}
		if(j==l1)
			printf("yes\n");
		else
			printf("no\n");
	}
	return 0;
}
