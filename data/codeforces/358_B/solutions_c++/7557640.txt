#include <iostream>
using namespace std;
int main()
{
	string s1="<3", s2, t;
	int n, i, j;
	cin>>n;
	while (n--) {
		cin>>t;
		s1+=t+"<3";
	}
	cin>>s2;
	for (i=0,j=0; i<s1.size()&&j<s2.size(); i++,j++) {
		while (s1[i]!=s2[j]&&j<s2.size()) j++;
	}
	if (i==s1.size()) cout<<"yes"<<endl;
	else cout<<"no"<<endl;
	return 0;
}
