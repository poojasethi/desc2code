#include<iostream>
using namespace std;

string s,s2,a[100010];
int n;
int main(){
	cin>>n;
	s2="<3";
	for (int i=0;i<n;i++){
		cin>>a[i];
		s2+=a[i]+"<3";
	}

	cin>>s;
	int m=s.length();
	int j=0,found=0;
	for (int i=0;i<s2.length();i++){
		if (s.find(s2[i],j)>m){
			found=1;
			break;
		}
		j=s.find(s2[i],j)+1;
	}
	if (!found) cout<<"yes"<<endl;
	else cout<<"no"<<endl;
	return 0;
}
