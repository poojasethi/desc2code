#include<iostream>
#include<cstdio>
using namespace std;

int main(){
	int n,len,j;
	string st,s,ch;
	scanf( "%d",&n);
	st=st+"<3";
	for(int i=0;i<n;i++){
		cin>>ch;
		st+=ch;
		st+="<3";
	}
	cin>>s;
	len=s.size();
	j=0;
	for(int i=0;i<len;i++){
		if(s[i]==st[j]) j++;
	}
	if(j==st.size()) puts("yes");
	else puts("no");
	return 0;
}
