#include <iostream>
#include <stdio.h>
using namespace std;
string st,st1,s;

int main()
{//freopen("i.txt","r",stdin);
int n; cin>>n;
st1="<3";
for (int i=1;i<=n;i++) {cin>>s; st1+=s+"<3";}
cin>>st;
int j=0;
for (int i=0;i<st.size();i++)
{if (st[i]==st1[j]) j++;
 if (j==st1.size()) {cout<<"yes"; return 0;}    
} 
cout<<"no";
return 0;
}
