#include<bits/stdc++.h>
using namespace std;

long long n,i,y;
string s,k,m;

int main(){
	 cin>>n;
	 k="<3";
	 for(i=0;i<n;i++){
	 	cin>>s;
	 	k+=s+"<3";
	 }
	 cin>>m;y=0;
	 for(i=0;i<m.length() && y<k.length();i++){
	 	if(m[i]==k[y])y++;
	 }
	 if(y==k.length())cout<<"yes"<<endl; else
	 cout<<"no"<<endl;
	 	
	return 0;
}
