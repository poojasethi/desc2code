#include <iostream>

using namespace std;

int main(){
  int n, k;
  string a, b, s;
  a = "<3";
  cin >> n;
  for(int i=0; i<n; i++)
  {
    cin >> s;
    a += s + "<3";
  }
  cin >> b;
  k = 0;
  for(int i=0; i<b.size(); i++)
  {
    if(a[k] == b[i])
      k++;
  }
  if(k == a.size())
    cout << "yes" << endl;
  else
    cout << "no" << endl;
}
