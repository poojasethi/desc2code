#include<iostream>
using namespace std;
int main()
{
	int n ;
	string s="<3";
	cin>>n;
	for(int i=0;i<n;i++)
	{
		string t;
		cin>>t;
		s.append(t);
		s.append("<3");
	}
	string check;
	cin>>check;
	int ptr=0;

	for(int i=0;i<check.size();i++)
	{
		if(ptr>=s.size())
			break;
		if(check[i]==s[ptr])
			ptr++;
	}
	if(ptr>=s.size())
		cout<<"yes"<<endl;
	else
		cout<<"no"<<endl;
}
