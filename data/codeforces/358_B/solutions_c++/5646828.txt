#include <iostream>
#include <string>
using namespace std;
int main() {
    int n,j=0;
    string words[100000],s,a;
    cin>>n;
    for(int i=0;i<n;i++) cin>>words[i];
    for(int i=0;i<n;i++){a+="<3"; a+=words[i];} a+="<3";
    cin>>s;
    for(int i=0;i<s.size() && j<a.size();i++)if(a[j]==s[i]) j++;
    cout<< ((j==a.size())? "yes" : "no");
    return 0;
}
