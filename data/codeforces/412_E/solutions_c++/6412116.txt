#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	#define int long long int
	string s;
	cin>>s;
	vector<int> v;
	int sum=0;
	for(int i=0;i<s.size();i++)
	{
		if(s[i]=='@')
			v.push_back(i);
	}
	for(int i=0;i<v.size();i++)
	{
		int pos=v[i];
		int llim=0,ulim=0,start=0,end=0;
		if(i==0) llim=-1; else llim=v[i-1];
		pos--;
		while(pos>llim && (isalnum(s[pos]) || s[pos]=='_'))
		{
			if(isalpha(s[pos]))
				start++;
			pos--;
		}
		pos=v[i]+1;
		int done=0;
		if(i==(v.size()-1))
			ulim=s.size();
		else ulim=v[i+1];
		while(pos<ulim && isalnum(s[pos]))
			{pos++;done=1;}
		if(s[pos]=='.')
			{
				pos++;
				while(pos<ulim && isalpha(s[pos]))
					{end++;pos++;}
			}
		sum+=start*end*done;
	}
	cout<<sum;
	return 0;
}
