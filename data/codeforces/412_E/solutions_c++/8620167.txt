#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<string>
#include <iostream>
#include<vector>
#define MAXN 1000050
#define lson i<<1
#define rson i<<1|1
#define LL long long
using namespace std;
LL d[MAXN][5];
char s[MAXN];
int main() {
	int n;
	LL ans = 0;
	scanf(" %s", s);
	n = strlen(s);
	for (int i = 1; i <= n; ++i) {
		if (s[i - 1] == '@') {
			d[i][1] = d[i - 1][0];
		} else if (s[i - 1] == '.') {
			d[i][3] = d[i - 1][2];
		} else if (s[i - 1] == '_') {
			d[i][0] = d[i - 1][0];
		} else if (s[i - 1] >= '0' && s[i - 1] <= '9') {
			d[i][0] = d[i - 1][0];
			d[i][2] = d[i - 1][1] + d[i - 1][2];

		} else {
			d[i][0] = d[i - 1][0]+1;
			d[i][2] = d[i - 1][1] + d[i - 1][2];
			d[i][4] = d[i - 1][3] + d[i - 1][4];
		}
//		for(int j=0;j<5;++j)
//			printf("%I64d ",d[i][j]);
//		puts("");
		ans += d[i][4];
	}
	printf("%I64d\n", ans);
}
