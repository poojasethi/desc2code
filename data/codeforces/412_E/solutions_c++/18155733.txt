#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;
char s[1000000+5];
__int64 sum;
int main(void){
	scanf("%s",s);
	int l=strlen(s);
	for(int i=0;i<l;i++){
		if(s[i]=='@'){
			__int64 s1=0;
			for(int j=i-1;j>=0;j--){
				if(!((s[j]<='9'&&s[j]>='0')||(s[j]>='a'&&s[j]<='z')||(s[j]=='_'))) break;
				if(s[j]>='a'&&s[j]<='z') s1++;
			}
			__int64 s2=0;
			int k;
			int flag=0;
			for(int j=i+1;j<l;j++){
				if(s[j]=='.'&&j!=i+1){
				   k=j;
				   flag=1;
				   break;	
				}
				if(!((s[j]<='9'&&s[j]>='0')||(s[j]>='a'&&s[j]<='z'))) break;
			}
			if(flag==0) continue;
			for(int j=k+1;j<l;j++){
				if(!(s[j]>='a'&&s[j]<='z')) break;
				s2++;
			}
			sum+=s1*s2;
		}
	}
	printf("%I64d\n",sum); 
	return 0;
}
