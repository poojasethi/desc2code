#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
	string s;
int main() {
	cin >> s;
	long long int size = s.size(), sum = 0, vaild = 0, tsum = 0;
	for (int i = 0; i < size; i++){
		if (s[i] == '@'){
			int j = i - 1;
			while (j>-1 && s[j] != '@' && s[j] != '.'){
				if (s[j] >= 'a' && s[j] <= 'z')
					tsum++;
				j--;
			}
			j = i + 1;
			while (s[j] != '_' && s[j] != '@' && s[j] != '.' && j < size)j++;
			if (j-i>1 && s[j] == '.'){
				int y = j+1;
				while (s[y] >= 'a' && s[y] <= 'z' && y < size){
					y++;
					vaild++;
				}
			}
			sum += tsum * vaild;
			tsum = 0;
			vaild = 0;
		}
	}
	cout << sum << endl;
}
