#include <iostream>
#include <string>
#include <cstring>
#include <vector>
#include <fstream>

using namespace std;

int main() {
//	freopen("txt", "r", stdin);
	string str;
	getline(cin, str);
//	cout << str << endl;
	int n = str.length();
	long long ans = 0;
	for (int i = 0; i < n - 1; ++i) {
		if (str[i] != '@')
			continue;
		long long num_pref = 0;
		for (int j = i - 1; j >= 0 && (str[j] == '_' || isalpha(str[j]) || isalnum(str[j])); --j)
			num_pref += (bool)isalpha(str[j]);

		bool next_can = isalpha(str[i + 1]) || isalnum(str[i + 1]);
		int j;
		for (j = i + 1; j < n && (isalpha(str[j]) || isalnum(str[j])); ++j)
			;
		next_can &= j != n && str[j] == '.';
		long long num_pos = 0;
		for (++j; j < n && isalpha(str[j]); ++j)
			num_pos++;
		if (next_can)
			ans += num_pref * num_pos;
	}
	cout << ans << endl;
	return 0;
}
