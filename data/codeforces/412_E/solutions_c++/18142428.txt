#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
char a[1000005];
typedef long long ll;
int main(){
	ll n,i,j,m,x1,x2,x3,d,k;
	scanf("%s",a+1);
	ll sum1=0,sum2=0,sum3=0,flag1=0,flag2=0,flag3=0,num=0;
	for(k=strlen(a+1);k>=1;k--){
		if(a[k]=='.'){
			x3=k;
			flag3=1;
		}
		else if(a[k]=='@'){
			x2=k;
			flag2=1;
		}
		if(flag2&&flag3&&x2+2<=x3){
			flag2=flag3=0;
			flag1=0;
			for(i=x2+1;i<x3;i++){
				if(!((a[i]>='a'&&a[i]<='z')||(a[i]>='0'&&a[i]<='9'))){
					flag1=1;
					break;
				}
			}
			if(!flag1){
				ll sum=0;
				for(i=x3+1;i<=strlen(a+1);i++){
					if((a[i]>='a'&&a[i]<='z')){
						sum++;
					}
					else break;
				}
				for(i=x2-1;i>=1;i--){
					if(a[i]=='.'||a[i]=='@')break;
					else if(a[i]>='a'&&a[i]<='z')num+=sum;
				}
			}
		}
	}
	printf("%lld\n",num);
	return 0;
}
