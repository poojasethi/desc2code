#include <cstdio>
#include <cstring>
using namespace std;
int main()
{
  long long sum=0;
  char str[1000003];
  scanf("%s",str);
  long len,i;
  len=strlen(str);
  for(i=1;i<len-3;i++)
	if(str[i]=='@')
	{
	  long long t1=0,t2=0,i1,i2;
	  bool flag=true;
	  i1=i-1;
	  while(i1>=0&&str[i1]!='.'&&str[i1]!='@')
	  {
	    if(str[i1]>='a'&&str[i1]<='z') t1++;
		i1--;
	  }
	  i2=i+1;
	  bool fff=false;
	  while(str[i2]!='.'&&flag)
	  {
	    if(str[i2]=='_'||str[i2]=='@') flag=false;
		fff=true;
		i2++;
		if(i2>=len-1) flag=false;
	  }
	  if(!fff)flag=false;
	  if(flag&&i2<len)
	  {
	    i2++;
	    while(i2<len&&str[i2]>='a'&&str[i2]<='z')
		{
		  t2++;
		  i2++;
		}
		sum+=t1*t2;
	  }
	}
  printf("%I64d",sum);
  
  //while(1);
  return 0;
}
