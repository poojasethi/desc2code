#include <cstdio>
#include <cstring>
const int maxn = 1000005;
typedef long long ll;
char s[maxn];
ll ans = 0;
bool check(int x)
{
	if('a' <= s[x] && s[x] <= 'z') return true;
	if('0' <= s[x] && s[x] <= '9') return true;
	return false;
}
void solve()
{
	scanf("%s",s);
	int len = strlen(s);
	for(int i = 0; i < len; ++i)
		if(s[i] == '@'){
			int tmp = -1,res = 0;
			for(int j = i + 1; j < len; ++j){
				if(check(j)){
					++res;
					continue;
				}
				if(s[j] == '.')
					tmp = j;
				break;
			}
			if(!res) continue;
			if(tmp == -1) continue;
			int l = 0;
			for(int j = i - 1; j >= 0; --j){
				if('a' <= s[j] && s[j] <= 'z'){
					++l;
				}
				if(check(j) || s[j] == '_') continue;
				break;
			}
			if(!l) continue;
			int r = 0;
			for(int j = tmp + 1; j < len; ++j){
				if('a' <= s[j] && s[j] <= 'z'){
					++r;
					continue;
				}
				break;
			}
			ans += (ll)l * r;
		}
	printf("%I64d",ans);
}
int main()
{
	solve();
	return 0;
}
