#include <cstdio>
#include <iostream>
#include <cstring>
#include <cmath>
#include <algorithm>
typedef long long LL;
using namespace std;
char s[1000005];
LL ans,x,y;

bool pd(char ch)
{
	if (ch>='a'&&ch<='z') return true;
	if (ch>='A'&&ch<='Z') return true;
	return false;
}
int main()
{
	scanf("%s",s);
	ans=0;
	int n=strlen(s);
	for (int i=0;i<n;++i)
	{
		if (s[i]=='@')
		{
			x=0;y=0;
			for (int j=i-1;j>=0;--j)
			{
				if (s[j]=='@'||s[j]=='.') break;
				if (pd(s[j])) x++;
			}
			for (int j=i+1;j<n;++j)
			{
				if (s[j]=='@'||s[j]=='_')
				{
					i=j-1;
					break;
				}
				if (s[j]=='.')
				{
					if (j==i+1) break;
					int k;
					for (k=j+1;k<n;++k)
					{
						if (!pd(s[k])) break;
						++y;
					}
					i=k-1;
					break;
				}
			}
			ans+=x*y;
		}
	}
	printf("%I64d\n",ans);
	return 0;
}
