#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
char str[1000005];
long long cal(int s,int e) {
    if(e-s==1) return 0;
    long long t1=0,t2=0;
    s--;e++;
    while(s>=0&&str[s]!='@'&&str[s]!='.') {
        if(str[s]>='a'&&str[s]<='z') t1++;
        s--;
    }
    while(str[e]>='a'&&str[e]<='z') {
        t2++;
        e++;
    }
    return t1*t2;
}
int main() {
    long long ans=0;
    long long t1=0,t2=0;
    scanf("%s",str);
    int len=strlen(str);
    for(int i=0;i<len;i++) {
        if(str[i]=='@') {
            int j=i+1;
            while(j<len&&((str[j]>='a'&&str[j]<='z')||(str[j]>='0'&&str[j]<='9'))) {
                j++;
            }
            if(str[j]=='.') ans+=cal(i,j);
            i=j-1;
        }
    }
    printf("%I64d\n",ans);
}
