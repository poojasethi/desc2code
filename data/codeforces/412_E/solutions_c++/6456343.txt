#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
#define x first
#define y second
ll g(ll a,ll b){if(b==0)return a;return g(b,a%b);}
int main(){
	string s;
	cin>>s;
	ll ans=0;
	for(ll i=0;i<s.length();i++){
		if(s[i]!='@')continue;
		for(ll j=i+1;j<s.length();j++){
			ll cnt=0;
			if(s[j]=='_' || s[j]=='@')break;
			if(s[j]=='.'){
				if(j==i+1)break;
				ll k=j+1;
				while(k<s.length() && s[k]>96 && s[k]<123){cnt++;k++;}
				if(cnt==0)break;
			}
			if(cnt==0)continue;
			ll k=i-1;
			while(k>=0){
				if(s[k]=='@' || s[k]=='.')break;
				if(s[k]>96 && s[k]<123)ans+=cnt;
				k--;
			}
			break;
		}
	}
	cout<<ans;
}
