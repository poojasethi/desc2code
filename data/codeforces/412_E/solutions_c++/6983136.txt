#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <set>
#include <map>
#include <vector>
#include <cctype>
using namespace std;

const int maxn = 1000000 + 123;

char str[maxn];

int main()
{
	scanf("%s", str);
	long long ans = 0;
	for (int i = 0; str[i]; ++ i)
		if (str[i] == '@' && str[i + 1] != '.')
		{
			bool flag = false;
			long long countl = 0, countr = 0;
			for (int j = i - 1; j >= 0; -- j)
			{
				if (!isalnum(str[j]) && str[j] != '_') break;
				if (isalpha(str[j])) countl ++;
			}
			for (int j = i + 1; str[j]; ++ j)
				if (flag)
				{
					if (isalpha(str[j])) countr ++;
					else break;
				}
				else
				{
					if (str[j] == '.')
					{
						flag = true;
						continue;
					}
					if (!isalnum(str[j])) break;
				}
			ans += countl * countr;
		}
	cout << ans << endl;
   	return 0;
}
