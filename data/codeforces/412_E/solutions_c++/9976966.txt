#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int n,m,k;
string s;
long long ln[1000010],rn[1000010],abf[1000010],la[1000010];
long long ans;
int main()
{
	cin>>s;
	n=s.length();
	for(int i=n;i>0;i--) s[i]=s[i-1];
	s[0]='#';
	for(int i=2;i<=n;i++)
	{
		if(s[i-1]!='.'&&s[i-1]!='@')
		{
			ln[i]=ln[i-1]+1ll;
			if(s[i-1]!='_') la[i]=la[i-1]+1ll;
		}
	}
	for(int i=n-1;i>0;i--)
		if(s[i+1]>='a'&&s[i+1]<='z') rn[i]=rn[i+1]+1ll;
	//for(int i=1;i<=n;i++) if(s[i]=='.'||s[i]=='@')cout<<ln[i]<<" "<<rn[i]<<endl;
	for(int i=1;i<=n;i++)
	{
		abf[i]=abf[i-1];
		if(s[i]>='a'&&s[i]<='z') abf[i]=abf[i]+1ll;
	}
	for(int i=1;i<=n;i++)
		if(s[i]=='.'&&s[i-la[i]-1]=='@'&&la[i]>0)
		{
			int sa=i-la[i]-1;
			ans=ans+(abf[sa]-abf[sa-ln[sa]-1])*rn[i];
		}
	cout<<ans<<endl;
	return 0;
}
