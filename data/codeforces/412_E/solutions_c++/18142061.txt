#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
#define ll long long
using namespace std;
char str[1100000];
int main()
{
    scanf("%s",str);
    int len=strlen(str) ;
    ll cnt=0,a=0,ans=0 ;
    for(int i=0;i<len;i++)
    {
        if((str[i]>='a'&&str[i]<='z')) 
			cnt++;
        else if(str[i]=='.') 
			cnt=0;
        else if(str[i]=='@')
        {
            int j=++i;
            while(i<len&&((str[i]>='a'&&str[i]<= 'z')||(str[i]>='0'&&str[i]<= '9')))
                ++i;
            if(i!=j&&str[i]=='.')
            {
                j=++i;
                a=0;
                while(i<len&&(str[i]>='a'&&str[i]<='z'))
                {
                    i++;
                    a++;
                }
                ans+=a*cnt ;
            }
            i=j-1;
            cnt=0 ;
        }
    }
    printf("%I64d\n",ans) ;
    return 0 ;
}
