#include <iostream>
#include <string>

using namespace std;

typedef long long int ll;

int main()
{
  ios::sync_with_stdio(false);
  string s;
  cin>>s;
  ll n1=0,n2=0,n3=0,n4=0,n5=0;
  ll sol=0;
  for (int i=0;i<int(s.size());i++) {
    ll nn1=0,nn2=0,nn3=0,nn4=0,nn5=0;
    if (s[i]>='a' and s[i]<='z') {
      nn1=n1+1;
      if (n3)
	nn3=n3;
      else
	nn3=n2;
      if (n5)
	nn5=n5;
      else
	nn5=n4;
    } else if (s[i]>='0' and s[i]<='9') {
      nn1=n1;
      if (n3)
	nn3=n3;
      else
	nn3=n2;
    } else if (s[i]=='.') {
      nn4=n3;
    } else if (s[i]=='@') {
      nn2=n1;
    } else if (s[i]=='_') {
      nn1=n1;
    }
    n1=nn1;n2=nn2;n3=nn3;n4=nn4;n5=nn5;
    //cout<<n1<<" "<<n2<<" "<<n3<<" "<<n4<<" "<<n5<<endl;
    sol+=n5;
  }
  cout<<sol<<endl;
}
