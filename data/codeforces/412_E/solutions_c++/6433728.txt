#include<bits/stdc++.h>

using namespace std;

#define FOR(ii,aa,bb) for(lli ii=aa;ii<=bb;ii++)
#define TFOR(ii,aa,bb) for(lli ii=aa;ii>=bb;ii--)
#define N (lli)1e6+5LL
#define pb push_back
#define lli long long int

string s;
lli n,ha[N],et[N],sa[N],al[N],no[N],ans;
lli hak,etk,sak,alk,nok;
vector<lli> v1,v2;

int main () {

	cin >> s;
	n=s.size();

	hak=etk=sak=alk=nok=n;

	TFOR(i,n,0)
	{
		if(s[i]>='a' and s[i]<='z') hak=i;
		if(s[i]=='@') etk=i;
		if(s[i]>='0' and s[i]<='9') sak=i;
		if(s[i]=='_') alk=i;
		if(s[i]=='.') nok=i;

		ha[i]=hak;
		et[i]=etk;
		sa[i]=sak;
		al[i]=alk;
		no[i]=nok;
	}

	FOR(i,0,n-1)
	{
		if(s[i]>='a' and s[i]<='z' and et[i]!=n and no[i]!=n and et[i]<no[i] and no[i]<al[et[i]+1])
			if(et[et[i]+1]>no[i] and ha[no[i]+1]!=n and (ha[et[i]+1]<no[i] or sa[et[i]+1]<no[i]))
				ans+=min(min(et[no[i]+1],sa[no[i]+1]),min(al[no[i]+1],no[no[i]+1]))-no[i]-1LL;
	}

	cout << ans << endl;

	return 0;

}
