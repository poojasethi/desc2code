#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
#define LL long long
char s[1000010];
LL before[1000010],next[1000010],link[1000010];
LL p[1000010][3];
LL len,res,ans;
int main(){
	scanf("%s",s+1);
	len=strlen(s+1);
	for (LL i=1;i<=len;i++){
		p[i][0]=p[i-1][0];p[i][1]=p[i-1][1];p[i][2]=p[i-1][2];
		before[i]=before[i-1];
		if (s[i-1]=='@'||s[i-1]=='.')before[i]=i-1;
		if (s[i]>='a'&&s[i]<='z')p[i][0]++;
		if (s[i]>='0'&&s[i]<='9')p[i][1]++;
		if (s[i]=='_')p[i][2]++;
	}
	for (LL i=len;i;i--){
		if (s[i]>='a'&&s[i]<='z')link[i]=link[i+1]+1;else link[i]=0;
		next[i]=next[i+1];
		if (s[i+1]=='@'||s[i+1]=='.')next[i]=i+1;
	}
	for (LL i=1;i<=len;i++)
	if (s[i]=='@'&&s[next[i]]=='.'&&next[i]>i+1){
		if (p[next[i]][2]-p[i][2]>0)continue;
		res=p[i][0]-p[before[i]][0];
		res=res*link[next[i]+1];
		ans=ans+res;
	}
	printf("%I64d",ans);
}
