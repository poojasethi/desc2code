#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<math.h>
#include<algorithm>
using namespace std;
int n;
long long p,q;
char a[1000001];
int main()
{
    int i,j,k,l;
    scanf("%s",&a);
    n=strlen(a);
    for(i=0;i<n;i++)
      if(a[i]=='@')
        {
         j=i;
         while(j>0 && a[j-1]!='.' && a[j-1]!='@')
           j--;
         k=i;
         while(k<n-1 && a[k+1]!='.' && a[k+1]!='_' && a[k+1]!='@')
           k++;
         if(a[k+1]!='.' || i==k)
           continue;
         k++;
         l=k;
         while(l<n-1 && a[l+1]>='a' && a[l+1]<='z')
           l++;
         q=0;
         while(j<i)
           {
            if(a[j]>='a' && a[j]<='z')
              q++;
            j++;
           }
         q*=l-k;
         p+=q;
        }
    cout<<p<<"\n";
    return 0;
}
