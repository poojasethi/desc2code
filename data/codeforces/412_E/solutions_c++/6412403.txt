#include <cstring>
#include <iostream>
using namespace std;

string s; 
long long a, aa, b, bb, c, res;
int main() {
	cin >> s; int n = s.length();
	for (int i = 0; i < n; i ++) {
		long long ta = 0, taa = 0, tbb = 0, tb = 0, tc = 0;
		if (s[i] == '.') tbb += b;
		else if (s[i] == '@') taa += a;
		else if (s[i] == '_') ta += a;
		else if (s[i] >= '0' && s[i] <= '9'){
			ta += a;
			tb += aa + b;
		}
		else {
			ta += a + 1;
			tb += aa + b;
			tc += bb + c;
		}
		res += tc;
		a = ta; b = tb; aa = taa; bb = tbb; c = tc;
	}
	cout << res << endl;
	return 0;
}
