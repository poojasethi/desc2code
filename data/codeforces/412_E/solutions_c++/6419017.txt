#include <iostream>
#include <string.h>
#include <string>
#include <stdio.h>
using namespace std;
int main()
{
	int i,j,k,l,n;
	long long ans=0;
	string s;
	cin>>s;
	j=0;
	for (i=0;i+3<s.length();i++) if (s[i]=='@' && j>0)
	{
		l=0;
		int o=0;
		for (k=i+1;k<s.length();k++)
		{
			if (s[k]=='_'||s[k]=='@') {j=l; break;}
			if (s[k]>='a'&&s[k]<='z') l++;
			if (s[k]=='.')
			{
				if (k==i+1) {break;}
				l=0;
				for (k++;k<s.length();k++)
				{
					if (s[k]<'a' || s[k]>'z') break;
					l++;
				}
				ans+=1LL*j*l;
				j=l;
				break;
			}
		}
		i=k-1;
	}
	else
	{
		if (s[i]=='@' || s[i]=='.') j=0; else if (s[i]>='a'&&s[i]<='z') j++;
	}
	cout<<ans<<endl;
}
