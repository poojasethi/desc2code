#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;

char s[1000010];
int len;

long long findl (int i)
{
    long long num=0;
    for(i=i-1;i>=0;i--)
    {
        if (s[i]>='a'&&s[i]<='z')
            num++;
        else if (s[i]!='_'&&(s[i]<'0'||s[i]>'9'))
            break;
    }
    return num;
}

long long findr (int i)
{
	i=i+1;
    long long num=0;
    int j=i;
    for(i;i<len;i++)
        if (!((s[i]>='a'&&s[i]<='z')||(s[i]>='0'&&s[i]<='9')))
            break;
    if (i>=len||s[i]!='.'||j==i)
        return 0;
    for(i=i+1;i<len;i++)
        if (s[i]>='a'&&s[i]<='z')
            num++;
        else
            break;
    return num;
}

int main()
{	
    gets(s);
    len=strlen(s);
    long long ans=0;
    for(int i=0;i<len;i++)
        if(s[i]=='@')
		    ans+=findl(i)*findr(i);
    printf("%I64d\n",ans);
    return 0;
}
