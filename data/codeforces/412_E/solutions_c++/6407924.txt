#include <iostream>
#include <cstdio>
using namespace std;

inline bool isletter(char ch)
{
	return ch >= 'a' && ch <= 'z' || ch >= 'A' && ch <= 'Z';
}

inline bool isnumber(char ch)
{
	return ch >= '0' && ch <= '9';
}

int main(void)
{
	char str[1000000 + 5];
	int i, j;
	long long cnt = 0;
	gets(str);
	for (i = 0; str[i]; i++) {
		if (str[i] == '@') {
			long long cnt1 = 0, cnt2 = 0;
			for (j = i - 1; j >= 0; j--) {
				if (isletter(str[j]) || isnumber(str[j]) || str[j] == '_') {
					if (isletter(str[j])) {
						cnt1++;
					}
				}
				else {
					break;
				}
			}
			
			if (isletter(str[i + 1]) || isnumber(str[i + 1])) {
				j = i + 2;
				while (isletter(str[j]) || isnumber(str[j])) {
					j++;
				}
				if (str[j] == '.') {
					for (j++; isletter(str[j]); j++) {
						cnt2++;
					}
				}
			}
			
			cnt += cnt1 * cnt2;
		}
	}
	
	cout << cnt << endl;
	return 0;
}
