#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;
typedef long long LL;
const int maxn = 1000005;
char s[maxn];int n;LL ans=0;
void init_work_put()
{
	int i,j,sum1,sum2;
	scanf("%s",s+1);n=strlen(s+1);
	for (i=1;i<=n;i++)
	if (s[i]=='@')
	{
		j=i-1;sum1=0;
		while (j>=1&&s[j]!='.'&&s[j]!='@')
		{
			if (s[j]>='a'&&s[j]<='z') sum1++;
			j--;
		}
		j=i+1;sum2=0;
		if (i>=n||s[i+1]=='.') continue;
		while (j<=n&&s[j]!='.'&&s[j]!='@'&&s[j]!='_') j++;
		if (s[j]!='.') continue;
		j++;
		while (j<=n&&s[j]>='a'&&s[j]<='z') j++,sum2++;
		ans+=(LL)sum1*sum2;
	}
	cout<<ans;
}
int main()
{
	init_work_put();
	return 0;
}
