#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxl=1000050;
inline bool is_let(char x) { return ('a'<=x&&x<='z')||('A'<=x&&x<='Z'); }
inline bool is_num(char x) { return '0'<=x&&x<='9'; }
int l;
long long ans=0;
char s[maxl];
long long calc(int x)
{
	int i,s1=0,s2=0;
	for(i=x-1;i>=0&&(is_let(s[i])||is_num(s[i])||s[i]=='_');i--)
		s1+=is_let(s[i]);
	for(i=x+1;i<l&&(is_let(s[i])||is_num(s[i]));i++);
	if(i==x+1||i==l||s[i]!='.') return 0;
	for(i++;i<l&&is_let(s[i]);i++)
		s2++;
	return (long long)s1*s2;
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j;
	scanf("%s",s);
	l=strlen(s);
	for(i=0;i<l;i++)
		if(s[i]=='@')
			ans+=calc(i);
	printf("%I64d\n",ans);
	return 0;
}
