#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<algorithm>
#include<string.h>
using namespace std;
int n;
char z[1000006];
int l,s=0,r[1000006];
long long ans;
long long L(int s)
{long long a=s;
 return a;
}
int main()
{scanf("%s",&z[1]);z[0]='a';n=strlen(z);z[0]=0;n--;
 for(int i=n;i>=1;i--)
    {if(z[i]>='a'&&z[i]<='z')
       r[i]=r[i+1]+1;
     else
       r[i]=0;
    }
 for(int i=1;i<=n;i++)
    {if(z[i]>='a'&&z[i]<='z')
       s++;
     if(z[i]=='.')
       {s=0;
        if(l&&(z[i-1]!='@'))
          ans+=L(l)*L(r[i+1]);
        l=0;
       }
     if(z[i]=='@')l=s,s=0;
     if(z[i]=='_')l=0;
    }
 cout<<ans<<endl;
 return 0;
}
