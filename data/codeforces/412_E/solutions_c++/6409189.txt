#include <iostream>
#include <cstdio>
#include <string>
#include <algorithm>
#include <vector>

using namespace std;

const int MAXN = 1000001;
string s;
long long a[MAXN];
long long int b[MAXN];
char tmps[MAXN];
int main()
{
	scanf("%s", &tmps);
	int n = 0;
	while (tmps[n] != 0) s.push_back(tmps[n++]);
	for (int i = 0; i < n; i++)
	{
		if (i == 0) a[i] = 0;
		else a[i] = a[i - 1];
		if ((s[i] >= 'a') && (s[i] <= 'z'))  a[i]++;
		else if ((s[i] == '@') || (s[i] == '.')) a[i] = 0;
		else {};
	}
	b[n] = 0;
	for (int i = n - 1; i >= 0; i--)
	{
		b[i] = b[i + 1];
		if ((s[i] >= 'a') && (s[i] <= 'z')) b[i]++;
		else b[i] = 0;
	}
	int p = -1;
	long long r= 0;
	for (int i = 1; i < n; i++)
	{
		if (s[i] == '.')
		{
			if ((p != -1) && (p != i - 1)) r += (a[p - 1] * b[i + 1]);
			p = -1;
		}
		else if (s[i] == '_') p = -1;
		else if (s[i] == '@') p = i;
		else {};
	}
	cout << r << endl;
	return 0;
}
