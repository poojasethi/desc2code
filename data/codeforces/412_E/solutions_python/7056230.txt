s = list(raw_input().strip())
f = [ 0 , 0 , 0 , 0 ]
ans = 0
for c in s:
	if c.isdigit():
		f[2] += f[1]
		f[1] = 0
		f[3] = 0
	elif c.isalpha():
		f[0] += 1
		f[2] += f[1]
		f[1] = 0
		ans += f[3]
	elif c == '_':
		f[1] = f[2] = f[3] = 0
	elif c == '.':
		f[3] = f[2]
		f[0] = f[1] = f[2] = 0
	elif c == '@':
		f[1] = f[0]
		f[0] = f[2] = f[3] = 0
print ans
