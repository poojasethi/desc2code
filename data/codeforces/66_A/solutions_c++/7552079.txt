#include <iostream>

using namespace std;

int main()
{
  unsigned long long x;
   cin>>x;
   if(x>=0&&x<128) cout<<"byte";
   else if(x>=128&&x<32768) cout<<"short";
   else if(x>=32768&&x<2147483648) cout<<"int";
   else if(x>=2147483648&&x<9223372036854775808) cout<<"long";
   else cout<<"BigInteger";

}
