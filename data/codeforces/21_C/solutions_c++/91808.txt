#include <stdio.h>
#include <map>
using namespace std;

int n, i, a[188888], sum, su, dp[199999];
map<long long, long long> pos, ppos;
long long can[188888], res;
pair<long long, long long> q;

int main(){
	scanf("%d\n", &n);
	for(i = 1; i<=n; i++){
		scanf("%d", &a[i]);
		sum += a[i];
		dp[i] = ppos[sum/2];
		++ppos[sum];
	}
	su = 0;
	for(i = 1; i<n; i++){
		su += a[i];
		if (sum-su == su/2 && su % 2 == 0) res += dp[i];
	}
	printf("%I64d\n", res);
	return 0;
}
