#include <iostream>
#include <map>

using namespace std;
typedef long long LL;

const int maxN = 1e5 + 7;

int a[maxN], p[maxN];

map <int,int> cnt;

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	int n;
	cin >> n;
	for (int i = 1; i <= n; ++i) {
		cin >> a[i];
		p[i] = p[i - 1] + a[i];
	}
	if (p[n] % 3) {
		cout << "0\n";
		return 0;
	}
	LL ans = 0;
	for (int i = n; i >= 1; --i) {
		if (p[i] == p[n] / 3)
			ans += cnt[2 * p[n] / 3];
		if (i != n)
			cnt[p[i]] ++;
	}
	cout << ans << "\n";
	return 0;
}
