# include <iostream>

using namespace std;

int sum [ 100009 ];

int main(){
	
	int n, i;
	
	cin >> n;
	
	cin >> sum [ 0 ];
	
	for ( i = 1; i < n; ++ i ){
		cin >> sum [ i ];
		sum [ i ] += sum [ i - 1 ];
	}
	
	if ( sum [ n - 1 ] % 3 ){
		cout << "0\n";
		return 0;
	}
	
	long long c1, c2, s;
	c1 = c2 = 0;	s = sum [ n - 1 ];
	
	for ( i = 0; i < n - 1; ++ i ){
		if ( sum [ i ] == s * 2 / 3 )
			c2 += c1;
		if ( sum [ i ] == s / 3 )
			++ c1;
	}
	
	cout << c2 << '\n';
	
	return 0;
	
}
