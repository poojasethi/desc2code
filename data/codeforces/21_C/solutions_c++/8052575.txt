#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<iostream>
using namespace std;
int n; 
long long a[100010];
long long ans,s,ave;
int main() 
{
	cin>>n;
	int x; 
	for (int i=1;i<=n;i++) 
	{
		scanf("%d", &x);
		a[i]=a[i-1]+x;
	}
	if (a[n]%3!=0) {printf("0\n"); return 0;}
	ave=a[n]/3;
	for (int i=1;i<n;i++) 
	{
		if(a[i]==ave*2) ans+=s;
		if(a[i]==ave) s++;
	}
	cout<<ans<<endl;
	return 0;
}
