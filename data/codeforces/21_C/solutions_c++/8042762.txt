#include <cstdio>
#include <algorithm>
#define MAXN 100000
using namespace std;
int sum[MAXN+5];
int main() {
	int n;
	scanf("%d", &n);
	long long ans = 0, leap = 0; sum[0] = 0;
	for (int i = 1; i <= n; i++) {
		scanf("%d", &sum[i]);
		sum[i] += sum[i-1];
	}
	if (sum[n]%3) {
		printf("0");
		return 0;
	}
	int suc = sum[n]/3;
	for (int i = 2; i < n; i++) {
		if (sum[i-1] == suc) leap++;
		if (sum[i] == suc*2) ans += leap;
	}
	printf("%I64d", ans);
	return 0;
}
