#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;

long long sum[100010], t, d[100010], ans =0;
int n;

int main(){
	cin >> n >> sum[0];
	for(int i = 1; i < n; i++){
		int a;
		scanf("%d" , &a);
		sum[i] = sum[i-1] + a;
	}
	if(sum[n-1] % 3){
		cout <<"0" << endl;
		return 0;
	}
	t = sum[n-1]/3;
	if(sum[0] == t)
		d[0] = 1;
	
	
	for(int i = 1; i < n-1; i++){
		d[i] = d[i-1];
		if(sum[i] == 2*t)
			ans += d[i];
		if(sum[i] == t)
			d[i]++;
	}
	cout << ans << endl;
	return 0;
}
