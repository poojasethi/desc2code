#include <iostream>
#include <cstdlib>
#include <map>
#include <cstdio>
using namespace std;
map<int,int> a;
int n,sm=0,sm1=0;
long long res=0;
int elems[100000];
int main()
{
    scanf("%d",&n);
    for (int i=0; i<n; i++) { scanf("%d",&elems[i]); sm+=elems[i]; };
    for (int i=0; i<n-1; i++) 
    {
        sm1+=elems[i];
        if ((sm1%2==0) && ((sm-sm1)*2==sm1) ) res+=(long long)a[sm1/2];
        a[sm1]++;
    };
    cout<<res;
    return 0;
};
