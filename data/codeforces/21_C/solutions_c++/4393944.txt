#include <iostream>
#include<vector>

using namespace std;

int main() {
    int n,s=0,l=0,t=0;
    long long r=0;
    cin>>n;
    vector<int> v(n);
    for(int i=0;i<n;i++)
    {
            cin>>v[i];
            s+=v[i];
    }
    if(s%3!=0)
    {
              cout<<0<<endl;
              return 0;
    }
    s/=3;
    for(int i=0;i<n-1;i++)
    {
            t+=v[i];
            if(t==2*s)r+=l;
            if(t==s)l++;
    }
    cout<<r<<endl;
    //system("pause");
    return 0;
}
