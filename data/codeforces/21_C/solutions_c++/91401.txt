#include <iostream>
using namespace std;

int main()
{
	int n;
	long long A[100000];
	long long S[100000];
	cin >> n;
	long long sum = 0;
	for (int i = 0; i < n; i++)
	{
		cin >> A[i];
		if (i)
			S[i] = S[i - 1] + A[i];
		else
			S[i] = A[i];
		sum += A[i];
	}
	if (sum % 3)
	{
		cout << 0;
		return 0;
	}
	
	long long ans = 0;
	
	int p1 = 0, p2 = 0;
	for (int i = 0; i < n - 1; i++)
	{
		if (S[i] == sum * 2 / 3)
		{
			ans += p1;
			p2++;
		}
		if (S[i] == sum / 3)
			p1++;
	}
	cout << ans;
}
