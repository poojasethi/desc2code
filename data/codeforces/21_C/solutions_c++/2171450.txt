#include<stdio.h>
__int64 a[120000];
int main()
{
	__int64 n;
	__int64 x;
	__int64 i,j;
	__int64 ans;
	while(scanf("%I64d",&n)!=EOF)
	{
		ans=0;
		a[0]=0;
		for(i=1;i<=n;i++){
			scanf("%I64d",&x);
			a[i]=a[i-1]+x;
		}
		__int64 sum=a[n];
		if(sum%3!=0){
			printf("0\n");
			continue;
		}
		sum/=3;
		__int64 s1,s2;
		s1=s2=0;
		for(i=1;i<n;i++)
		{
			if(a[i]==sum){
				if(sum==0)ans+=s1;
				s1++;
			}
			else if(a[i]==sum*2){
				ans+=s1;
			}
		}
		printf("%I64d\n",ans);
	}
}
