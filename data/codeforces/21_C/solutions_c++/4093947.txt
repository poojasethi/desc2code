#include <iostream>
#include <numeric>
#define MAXN 100005

using namespace std;

int A[MAXN];

int main() {
	ios::sync_with_stdio(0);
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> A[i];
	int sum = accumulate(A, A+n, 0), ac, fr = 0;
	long long ans = 0;
	if (sum % 3 == 0) {
		ac = 0;
		for (int i = 0; i < n - 1; i++) {
			ac += A[i];
			//cout << ac << endl;
			if (ac == sum / 3 * 2)
				ans += fr;
			if (ac == sum / 3)
				fr ++;
		}
	}
	cout << ans << endl;
}
