#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <map>
using namespace std;

typedef long long LL;
int n,sum,all,f[100001];
long long ans;
map <int,int> M;
int main() {
    cin>>n;
    for(int i=0;i<n;i++)
        {cin>>f[i];
            sum+=f[i];
            f[i]+=f[i-1];
            if(i!=n-1){M[f[i]]++;}
        }
    for(int i=0;i<n-1;i++){
        M[f[i]]--;
        if(sum==3*f[i])ans+=1LL*M[2*f[i]];
    }
    cout<<ans<<endl;
}
