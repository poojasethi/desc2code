#include <iostream>
#include <vector>

using namespace std;

int main()
{
  int n;
  cin>>n;
  vector<int> v(n);
  int suma=0;
  for (int i=0;i<n;i++) {
    cin>>v[i];
    suma+=v[i];
  }
  if (suma%3==0) {
    int suma1=suma/3;
    int suma2=2*suma1;
    long long int num1=0;
    long long int amount=0;
    suma=0;
    for (int i=0;i<n-1;i++) {
      suma+=v[i];
      if (suma==suma2) {
	amount+=num1;
      }
      if (suma==suma1) num1++;
    }
    cout<<amount<<endl;
  } else {
    cout<<0<<endl;
  }
}
