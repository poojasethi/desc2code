#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<iostream>
using namespace std;

long long a[100010];
int main() {
	int n; scanf("%d", &n);
	memset(a, 0, sizeof 0);
	for (int i=1; i<=n; ++i) {
		int x; scanf("%d", &x);
		a[i] = a[i-1] + x;
	}
	if (a[n] % 3 != 0) {printf("0\n"); return 0;}
	long long ans = 0, s = 0, pp = a[n]/3;
	for (int i=1; i<n; ++i) {
		if (a[i] == pp * 2) ans += s;
		if (a[i] == pp) s++;
	}
	printf("%I64d", ans);
}
