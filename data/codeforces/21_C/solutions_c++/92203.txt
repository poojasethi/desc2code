#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <map>
using namespace std;

map <int, int> H;

int s[100005];

int main()
{
	int n;
	long long ans=0;
	scanf("%d",&n);
	for (int i=0; i<n; i++)
	{
		scanf("%d",&s[i]);
		if (i>0) s[i]+=s[i-1];
	}

	if (n>2)
	{
		H.clear();
		H[s[0]]++;
		for (int i=1; i<n-1; i++)
		{
			if (2*s[n-1]-2*s[i]==s[i])
			{
				int t1=s[n-1]-s[i];
				ans=ans+H[t1];
			}
			H[s[i]]++;
		}
	}

	printf("%I64d\n",ans);

	return 0;
}
