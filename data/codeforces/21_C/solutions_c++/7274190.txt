#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;
int n,sum,all,f[100001];
long long ans;
map <int,int> M;
int main(){
  scanf("%d",&n);
  for(int i=1;i<=n;i++){ 
    scanf("%d",&f[i]);
    sum += f[i];
    f[i] += f[i-1]; if(i != n) M[f[i]]++;
  }
  for(int i=1;i<n;i++){
    M[f[i]]--;
    if(sum == 3*f[i]) ans+= 1ll * M[2*f[i]];
  }
  printf("%I64d\n",ans);
  return 0;
}
