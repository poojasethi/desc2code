#include <iostream>
#include <vector>
#include <map>
using namespace std;

int main(){
    int k = 0;
    int n;
    cin >> n;
    vector<int> v(n);
    for(int i = 0 ; i < n ; i++) cin >> v[i] , k += v[i];
    if(k%3){
        cout << 0 << endl;
        return 0;
    }
    k /= 3;
    map<int,int> p;
    int sum = 0;
    long long ans = 0;
    for(int i = 0 ; i < n-1 ; i++){
        sum += v[i];
        if(k*2==sum) ans += p[sum-k];
        
        p[sum]++;
    }
    cout << ans << endl;
}
