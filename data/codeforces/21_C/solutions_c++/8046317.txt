#include<cstdio>

using namespace std;

int N;
long long prefix_sum[100100];

int main()
{
	scanf("%d",&N);
	for(int i=1;i<=N;i++)
	{
		int a; scanf("%d",&a);
		prefix_sum[i]=a+prefix_sum[i-1];
	}
	if(prefix_sum[N] % 3 != 0) {printf("0"); return 0;}

	long long ans=0,sum=0,piece=prefix_sum[N]/3;
	for(int i=1;i<N;i++)
	{
		if(prefix_sum[i]==piece*2) ans+=sum;
		if(prefix_sum[i]==piece) sum++;
	}

	printf("%I64d",ans);
	return 0;
}
