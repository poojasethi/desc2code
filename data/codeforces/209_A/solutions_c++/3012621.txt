#include <cstdio>

long long const MOD = (long long) 1e9 + 7;
long long fib[1000005];
long long total = 0;
int n;

int main()
{
    scanf("%d", &n);
    fib[0] = 1;
    fib[1] = 2;
    for (int i = 0; i < n; i++)
    {
        total += fib[i];
        total %= MOD;
        fib[i + 2] = fib[i] + fib[i + 1];
        fib[i + 2] %= MOD;
    }
    printf("%I64d\n", total);
}
