#include <iostream>
#include <stdio.h>
using namespace std;
const int N = 1000000007;
__int64 ans[1000000];
int main()
{
	ans[0]=1;
	ans[1]=3;
	int i;
	for(i = 2; i < 1000000; i++)
		ans[i] = (2 + ans[i-1] + ans[i-2])%N;
	for(int n;scanf("%I64d",&n)!=EOF;)
		printf("%I64d\n",ans[n-1]);
	return 0;
}
