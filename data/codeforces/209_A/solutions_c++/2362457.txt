#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;

#define LL long long
#define mod 1000000007

int dp[1000006];
int main() {
    LL n;
    int i,j,k,l;
    dp[1] = 1;
    dp[2] = 3;
    for(int i = 3 ; i <= 1000000 ; i ++) {
        dp[i] = (dp[i - 1] + dp[i - 2] + 2) % mod;
    }
    while(~scanf("%d",&n)) {
        cout<<dp[n]<<endl;
    }
}
