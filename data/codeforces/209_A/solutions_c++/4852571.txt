#include<iostream>
#include<vector>
using namespace std;

long long inf=1000000007;

int main()
{
    vector<long long>a(1000*1000);
  long long ans=0;
  long long n;
  cin>>n;
  a[0]=1;
  a[1]=2;
for(int i=2;i<n;i++)
    a[i]=(a[i-1]+a[i-2])%inf;
for(int i=0;i<n;i++)
    ans=(ans+a[i])%inf;
cout<<ans<<endl;
}
