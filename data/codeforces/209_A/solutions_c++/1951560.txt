#include<iostream>
#include<cstring>
#include<stdio.h>
using namespace std;
long long a[1000001]={0,1,3};
int main()
{	
	int i=3;
	for(;i<=1000000;i++)
	{
		a[i]=(a[i-1]%1000000007+a[i-2]%1000000007+2)%1000000007;
	}
	int n;
	while(cin>>n)
	{
		cout<<a[n]<<endl;
	}
	return 0;
}
