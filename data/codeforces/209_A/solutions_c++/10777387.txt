#include <stdio.h>
#define MM 1000000007
int a[1000005][3];
int main(){
	a[1][1]=1;
	a[1][0]=0;
	for(int i=2;i<=1000000;i++){
		int t=i%2;
		a[i][1]=a[i-1][1];
		a[i][0]=a[i-1][0];
		a[i][t]+=a[i-1][t^1]+1;
		a[i][0]%=MM;
		a[i][1]%=MM;
	}
	int n;
while(scanf("%d",&n)!=EOF){
	printf("%d\n",(a[n][0]+a[n][1])%MM);	
}

}
