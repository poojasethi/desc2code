#include<iostream>
#include<cstdio>

using namespace std;

const int mod = 1000000007;

int n;

int main() {
    scanf( "%d" , &n );
    
    int b = 0 , r = 0;
    for ( int i = 1 ; i <= n ; i++ )
        if ( i & 1 ) b = ( b + r + 1 ) % mod;
            else r = ( r + b + 1 ) % mod;
    
    int ans = ( b + r ) % mod;
    printf( "%d\n" , ans );
    
    return 0;
}
