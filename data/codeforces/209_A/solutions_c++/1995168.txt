#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int M=1000000007;
int dp[1000010][2];
int main()
{
    dp[1][0]=0;
    dp[1][1]=1;
    for(int i=2;i<=1000000;i++)
    {
        dp[i][0]=dp[i-1][1];
        dp[i][1]=(dp[i-1][1]+dp[i-1][0]+1)%M;
    }
    int n;
    while(~scanf("%d",&n))
    printf("%d\n",(dp[n][0]+dp[n][1])%M);
}
