#include<stdio.h>
#include<iostream>
#define M 1000000007
using namespace std;
int main()
{
   long long a[4];
   int i,j,n;
   while(scanf("%d",&n)!=EOF){
	     a[0]=0,a[1]=1,a[2]=3;
		 if(n<3) printf("%lld\n",a[n]);
		 else{
		for(i=3;i<=n;i++)
			a[i%4]=(2*a[(i-1)%4]-a[(i-3)%4]+M)%M;
		printf("%lld\n",a[(i-1)%4]);
		 }
   }
  return 0;
}
