#include<bits/stdc++.h>
using namespace std;
long long dp[1000006];

int main()
{
     dp[1]=1;
     dp[2]=3;
     int i,N;
     scanf("%d",&N);
     for(i=3; i<=N; i++)
     {
          dp[i]=(dp[i-1]+dp[i-2]+2)%1000000007;
          //cout<<dp[i]<<endl;
     }
     cout<<dp[N]<<endl;
     return 0;
}
