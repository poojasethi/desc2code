#include<iostream>
using namespace std;
typedef long long int64;
int64 dp[1000005];
int64 mod=1000000007;
int  main(){
	dp[1]=1;
	dp[2]=3;
	dp[3]=6;
	for(int64 i=4;i<=1000000;i++){
		
	
		
			 dp[i]=(  (2+dp[i-1])%mod+dp[i-2])%mod;



		}




	
	int64 n;
	while(cin>>n)
		cout<<dp[n]%mod<<endl;
	
}
