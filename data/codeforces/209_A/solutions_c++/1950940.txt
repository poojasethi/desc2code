#include <cstdio>
#include <cstring>

const int maxn = 1000005;
const int mod = 1000000007;

int f[maxn], s[maxn], n;

int main() {
    f[1] = s[1] = 1;
    f[2] = s[2] = 2;
    scanf("%d", &n);
    for (int i = 3; i <= n; ++ i) {
        f[i] = (s[i - 1] + 1) % mod;
        s[i] = (s[i - 2] + f[i]) % mod; 
    }
    printf("%d\n", (s[n] + s[n - 1]) % mod); 
    return 0;
}
