#include<cstdio>
#include<iostream>
#define  P 1000000007
using namespace std;

long long opt[1000005];
int main()
{
  long long ans=0;
  int n;
  cin>>n;
  int i;
  long long f_1=1,f_2=1,f;
  opt[1]=1;
  for(i=2;i<=n;i++)
  {
    f=f_1+f_2;
    f=f%P;
    f_1=f_2;
    f_2=f;
    opt[i]=opt[i-1]+f;
    opt[i]%=P;
  }
  cout<<opt[n]<<endl;
  return 0;
}
