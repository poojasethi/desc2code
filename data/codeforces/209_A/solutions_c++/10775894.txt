#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define mod 1000000007

__int64 a[1000010];

int main()
{
	int i,n;
	scanf("%d",&n);
	a[1]=1;
	a[2]=3;
	for(i=3;i<=n;i++) a[i]=(a[i-1]+a[i-2]+2)%mod;
	printf("%I64d\n",a[n]);
	return 0;
}
