#include<iostream>
#include<vector>
using namespace std;
int main()
{
    vector<long long> f (1000 * 1000);
    long long inf = 1000 *1000*1000+7;
    long long n,i,ans =0;
    cin>>n;
    f[0]=1;f[1]=2;
    for (i=2;i<n;i++)
    {
        f[i]=(f[i-1]+f[i-2])%inf;
    }
    for(i=0;i<n;i++)
    {
        ans+=f[i];
        ans%=inf;
    }
    cout<<ans;
}
