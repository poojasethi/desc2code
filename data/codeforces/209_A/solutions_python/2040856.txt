# n - length of zebroid
def cnt_subzebroes(n):
        cnt = 1      
        pred, pred_pred = 0, 0
        last = n - 1
        while last > 0:
                cur = pred + pred_pred + 1
                pred_pred = pred % 1000000007	
                pred = cur % 1000000007	
                cnt = cnt + cur + 1
                last -= 1
	return cnt % 1000000007	

print cnt_subzebroes(int(input()))
