#!/usr/bin/env python

n = int(raw_input())

numbers = []

def print_containers():
    max_nums = sorted(numbers, reverse=True)[:3]
    max_nums.extend([None] * 3)
    m1, m2, m3 = max_nums[:3]
    s = q = d = False
    for n in numbers:
        if n == m1 and not s:
            print "pushStack"
            s = True
        elif n == m2 and not q:
            print "pushQueue"
            q = True
        elif n == m3 and not d:
            print "pushFront"
            d = True
        else:
            print "pushBack"
    m = []
    m.append(str(sum([s, q, d])))
    if s:
        m.append("popStack")
    if q:
        m.append("popQueue")
    if d:
        m.append("popFront")
    print ' '.join(m)


for _ in xrange(n):
    op = int(raw_input())
    if op == 0:
        print_containers()
        numbers = []
    else:
        numbers.append(op)
for _ in xrange(len(numbers)):
    print "pushStack"
