#include <cstdio>
#define maxn 100005
int a[maxn],b[maxn],s[maxn],N;
int main()
{
	scanf("%d",&N);
	for (int i=1,p=0; i<=N; i++)
	{
		scanf("%d",&a[i]);
		if (!a[i])
		{
			int x=0,y=0,z=0;
			for (int j=p+1; j<i; j++) if (a[j]>=a[x]) z=y,y=x,x=j; else if (a[j]>=a[y]) z=y,y=j; else if (a[j]>a[z]) z=j;
			b[x]=1,b[y]=2,b[z]=3,s[i]=i-p-1<3?i-p-1:3,p=i;
		}
	}
	for (int i=1; i<=N; i++) if (a[i]) puts(!b[i]?"pushFront":b[i]==1?"pushStack":b[i]==2?"pushQueue":"pushBack"); else
	{
		printf("%d",s[i]);
		if (s[i]>=1) printf(" popStack");
		if (s[i]>=2) printf(" popQueue");
		if (s[i]>=3) printf(" popBack");
		puts("");
	}
}
