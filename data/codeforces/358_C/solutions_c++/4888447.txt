#include<cstdio>
#include<algorithm>
using namespace std;
int x[100005];
int main() {
  int n;
  scanf("%d",&n);
  for(int i=1;i<=n;i++) scanf("%d",&x[i]);
  int mx=1,mx1=0,mx2=0,mx3=0,mn=1,hv=0;
  while(mx<=n) {
    for(mn=mx;mx<=n;mx++) {
      if(x[mx]==0) break;
      else {
	if(x[mx1]<x[mx]) {
	  swap(mx3,mx2);
	  swap(mx2,mx1);
	  mx1=mx;
	}
	else if(x[mx2]<x[mx]) {
	  swap(mx3,mx2);
	  mx2=mx;
	}
	else if(x[mx3]<x[mx]) {
	  mx3=mx;
	}
      }
    }
    for(int i=mn;i<mx;i++) {
      if(i==mx1) printf("pushStack\n");
      else if(i==mx2) printf("pushQueue\n");
      else if(i==mx3) printf("pushFront\n");
      else printf("pushBack\n");
    }

    if(mx1>0) hv++;
    if(mx2>0) hv++;
    if(mx3>0) hv++;

    if(x[mx]==0&&mx<=n) {
      printf("%d",hv);
      switch(hv) {
      case 3:printf(" popFront");
      case 2:printf(" popQueue");
      case 1:printf(" popStack");
      default:printf("\n");
      }
      hv=0;
      mx1=0;
      mx2=0;
      mx3=0;
    }
    mx++;
  }
}
