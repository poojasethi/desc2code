#include<iostream>
#include<queue>
#include<vector>
using namespace std;
priority_queue<pair <int,int> >  pq;
string output[100010];
int num[100010];
string action[]={"pushBack","pushStack","pushQueue","pushFront"};
string pop[]={"popStack","popQueue","popFront"};
int main()
{
	int n,i;
	cin>>n;
	int a[100100];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		if(!a[i])
		{
			for(int j=1;j<=3 && !pq.empty();j++)
			{
				output[pq.top().second]+=action[j];
				num[i]=j;
				pq.pop();	
			}
			while(!pq.empty())
				pq.pop();
		}
		else
		{
			pq.push(make_pair(a[i],i));
		}
	}
	for(i=0;i<n;i++)
	{
		if(!a[i])
		{
			if(num[i]!=0)
				cout<<num[i]<<" ";
			else
				cout<<num[i];
			for(int j=0;j<num[i];j++)
				if(j!=num[i]-1)
					cout<<pop[j]<<" ";
				else
					cout<<pop[j];

			cout<<endl;
		}
		else
		{
			if(output[i].size()!=0)
				cout<<output[i];
			else
				cout<<action[0];
			cout<<endl;
		}
	}
	return 0;
}
