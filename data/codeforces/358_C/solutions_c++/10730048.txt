int main()
{

  for(int i=1,j;i<=n;i=j+1)
  {
     for(j=i;j<=n && arr[j]!=0;j++);
        int a=0,b=0,c=0;
      for(int k=i;k<j;k++)
      {
        if(arr[k]>arr[c])c=k;
         if(arr[b]<arr[c])swap(b,c);
         if(arr[b]>arr[a])swap(b,a);
      }
     for(int k=i;k<j;k++)
     {
        if(k==a)puts("pushStack");else
         if(k==b)puts("pushQueue");else
         if(k==c)puts("pushFront");
        else puts("pushBack");

     }

     }

}
