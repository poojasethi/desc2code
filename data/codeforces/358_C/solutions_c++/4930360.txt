#include <cstdio>
using namespace std;
const int maxn=100010;
int a[maxn];
int main(){
    int n,i,j,mp1,mp2,mp3,p,m;
    scanf("%d",&n);
    for (i=1;i<=n;i++) scanf("%d",&a[i]);
    p=0;a[n+1]=-1;
    m=n;
    if (a[n]!=0) m++;
    for (i=1;i<=m;i++) if (i==m||a[i]==0){
        mp1=mp2=mp3=n+1;
        for (j=p+1;j<i;j++) if (a[mp1]<a[j]) mp1=j;
        for (j=p+1;j<i;j++) if (j!=mp1&&a[mp2]<a[j]) mp2=j;
        for (j=p+1;j<i;j++) if (j!=mp1&&j!=mp2&&a[mp3]<a[j]) mp3=j;
        for (j=p+1;j<i;j++) if (j!=mp1&&j!=mp2&&j!=mp3) printf("pushBack\n"); else
        if (j==mp1) printf("pushStack\n"); else
        if (j==mp2) printf("pushQueue\n"); else
        printf("pushFront\n");
        if (i<=n){
            if (mp1==n+1) printf("0\n"); else
            if (mp2==n+1) printf("1 popStack\n"); else
            if (mp3==n+1) printf("2 popStack popQueue\n"); else
            printf("3 popStack popQueue popFront\n");
        }
        p=i;
    }
    return 0;
}
