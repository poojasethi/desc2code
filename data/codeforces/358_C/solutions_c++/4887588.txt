#include<bits/stdc++.h>

#define st
#define FP( ii,aa,bb ) for( int ii=aa;ii<=bb;ii++ )

using namespace std;

int n,q,s,d;
vector<int> v;
map<int,int> a;
void solve(){
	a.clear();
	vector<int> V=v;
	if( v.size() )
		sort(v.begin(),v.end(),greater<int>());
	
	if( v.size()>=1 ) a[v[0]]++;
	if( v.size()>=2 ) a[v[1]]++;
	if( v.size()>=3 ) a[v[2]]++;
	
	int len=v.size();
	FP( i,0,len-1 ){
		if( a[V[i]] and !d ){ puts("pushBack"); d=1; a[V[i]]--; }
		else if( a[V[i]] and !s ){ puts("pushStack"); s=1; a[V[i]]--; }
		else if( a[V[i]] and !q ){ puts("pushQueue"); q=1; a[V[i]]--; }
		else					 { puts("pushFront"); }
	}

	cout << q+s+d;
	if( d ) cout << " ";
	if( d ) cout << "popBack";
	if( s ) cout << " ";
	if( s ) cout << "popStack";
	if( q ) cout << " ";
	if( q ) cout << "popQueue";
	cout << endl;

	d=s=q=0;
	v.clear();

}

int main(){
	
	int x;
	cin >> n;

	FP( i,1,n ){
		cin >> x;
		if( !x )
			solve();
		else
			v.push_back(x);
	}

	FP( i,0,(int)v.size()-1 ) cout << "pushStack" << endl;

}
