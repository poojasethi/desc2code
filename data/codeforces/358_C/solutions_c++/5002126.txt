#include<bits/stdc++.h>
using namespace std;

int a[111111],b[111111];
int n;
int main(){
	cin>>n;
	int cnt = 0;
	for(int i = 0; i < n; i++){
		int j = 0;
		bool flag = 0;
		while(cin>>a[j]){
			if(a[j] == 0){
				flag = 1;
				break;
			}
			j++;
		}
		if(j==0)printf("0\n");
		else if(j==1){
			puts("pushStack");
			if(flag) puts("1 popStack");
		}else if(j==2){
			printf("%s","pushStack\npushQueue\n");
			if(flag) printf("%s","2 popStack popQueue\n");
		}else {
			for(int k = 0;k < j;k++)b[k] = a[k];
			sort(b,b+j);
			bool ok[4] = {0};
			for(int k = 0;k < j;k++){
				if(a[k] == b[j-1] && !ok[1])puts("pushStack"),ok[1] = 1;
				else if(a[k] == b[j-2] && !ok[2]) puts("pushQueue"),ok[2] = 1;
				else if(a[k] == b[j-3] && !ok[3])puts("pushFront"),ok[3] = 1;
				else puts("pushBack");
			}
			if(flag)
			printf("%s","3 popStack popQueue popFront\n");
		}
		i += j;
	}
	return 0;
}
