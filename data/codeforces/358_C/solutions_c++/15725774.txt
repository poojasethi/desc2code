#include<iostream>
using namespace std;

int a[100010],b[100010],s[100010],n,last;

void doing ()
{
	cin>>n;
	for (int i=1;i<=n;i++)
	{
		cin>>a[i];
		if (!a[i])
		{
			int first,second,third;
			first=second=third=0;
			for (int j=last+1;j<i;j++)
				if (a[j]>=a[first]){
					third=second;second=first;first=j;
				}
				else
					if (a[j]>=a[second]){
						third=second;second=j;
					}
					else if (a[j]>a[third]) third=j;
			b[first]=1;
			b[second]=2;
			b[third]=3;
			if (i-last-1<3) s[i]=i-last-1;
			else s[i]=3;
			last=i;
		}
	}

}

void myprint ()
{
	for (int i=1;i<=n;i++)
		if (a[i])
		{
			if (!b[i]) cout<<"pushFront"<<endl;
			if (b[i]==1) cout<<"pushStack"<<endl;
			if (b[i]==2) cout<<"pushQueue"<<endl;
			if (b[i]==3) cout<<"pushBack"<<endl;
		}
		else
	    {
  		    cout<<s[i];
  		    if (s[i]>=1) cout<<" popStack";
  		    if (s[i]>=2) cout<<" popQueue";
  		    if (s[i]>=3) cout<<" popBack";
  		    cout<<endl;
	    }
}

int main()
{
	doing ();
	myprint ();
	return 0;
}
