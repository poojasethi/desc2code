#include <iostream>
#include <queue>
#include <vector>
using namespace std;

typedef pair<int,int> pii;

int n, type[100005], rq[100005] = {};
priority_queue<pii> pq;
string dat[100005];
string arr[] = {"popStack","popQueue", "popFront"};
string darn[] = {"pushStack", "pushQueue", "pushFront", "pushBack"};

int main(){

	cin >> n;
	for(int i = 0; i<n; i++){
		cin >> type[i];
		if(!type[i]){
			for(int j = 0; j<3; j++){
				if(pq.empty()) break;
				dat[i]+=' ';
				dat[i]+=arr[j];
				dat[pq.top().second] = darn[j];
				pq.pop();
				rq[i] = j+1;
			}
			while(pq.size()) pq.pop();
		} else pq.push(pii(type[i],i));
	}
	for(int i = 0; i<n; i++){
		if(type[i]){
			if(dat[i].size()) cout << dat[i] << '\n';
			else cout << darn[3] << '\n';
		} else cout << rq[i] << dat[i] << '\n';
	}

}
