#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>

using namespace std;

int a[100000+3],F[100000+3],n,pre;

int main()
{
scanf("%d",&n);
for (int i=1;i<=n;i++) scanf("%d",&a[i]);
for (int i=1;i<=n;i++)
  if (!a[i])
    {
	int max1=0,max2=0,max3=0;
	for (int j=pre+1;j<i;j++)
	  {
	  if (a[j]>=a[max1]) max3=max2,max2=max1,max1=j;
	    else if (a[j]>=a[max2]) max3=max2,max2=j;
		       else if (a[j]>=a[max3]) max3=j;
	  }
	F[max1]=1;F[max2]=2;F[max3]=3;
	F[i]=min(i-pre-1,3);
	pre=i;
	}
for (int i=1;i<=n;i++)
  if (a[i])
    {
	if (F[i]==1) puts("pushStack");
	if (F[i]==2) puts("pushQueue");
	if (F[i]==3) puts("pushFront");
	if (!F[i]) puts("pushBack");
	}
	else {
	     printf("%d",F[i]);
		 if (F[i]>0) printf(" popStack");
		 if (F[i]>1) printf(" popQueue");
		 if (F[i]>2) printf(" popFront");
		 puts("");
		 }
}
