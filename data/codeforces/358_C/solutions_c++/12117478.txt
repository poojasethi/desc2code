#include <bits/stdc++.h>
using namespace std;

int main(){
  int n;
  string push[4] = {"pushStack", "pushQueue", "pushFront", "pushBack"};
  string pop[4] = {"popStack", "popQueue", "popFront", "popBack"};
  cin >> n;
  vector<pair<int, int> > vec;
  for(int k = 0; k < n; k++){
    int foo, bar;
    cin >> foo;
    if (foo){
      vec.push_back(make_pair(foo, vec.size()));
      continue;
    }
    vector<string> shit(vec.size(), "");
    sort(vec.rbegin(), vec.rend());
    int sz = min(3, (int) vec.size());
    for (int i = 0; i < sz; i++)
      shit[vec[i].second] = push[i]; 
    for (int i = sz; i < vec.size(); i++)
      shit[vec[i].second] = push[3];
    for (int i = 0; i < shit.size(); i++)
      puts(shit[i].c_str());
    printf("%d", sz);
    for (int i = 0; i < sz; i++)
      putchar(' '), printf("%s", pop[i].c_str());
    putchar('\n');
    vec.clear();
  }
  for (int i = 0; i < vec.size(); i++)
    puts(push[0].c_str()); 
}
