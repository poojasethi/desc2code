#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int f[N],a[N];
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%d",&a[i]);
	j=1;
	For(i,1,n) if (!a[i]) {
		int ma1,ma2,ma3;
		ma1=ma2=ma3=0;
		for (;j<=i;j++) if (a[ma1]<a[j]) ma3=ma2,ma2=ma1,ma1=j;
		else if (a[ma2]<a[j]) ma3=ma2,ma2=j;
		else if (a[ma3]<a[j]) ma3=j;
		f[i]=(ma1>0)+(ma2>0)+(ma3>0);
		f[ma1]=1; f[ma2]=2; f[ma3]=3;
	}
	For(i,1,n) {
		if (!a[i]) {
			printf("%d",f[i]);
			if (f[i]>0) printf(" popFront");
			if (f[i]>1) printf(" popQueue");
			if (f[i]>2) printf(" popStack");
			printf("\n");
		} else {
			if (f[i]==0) printf("pushBack\n");
			if (f[i]==1) printf("pushFront\n");
			if (f[i]==2) printf("pushQueue\n");
			if (f[i]==3) printf("pushStack\n");
		}
	}
	//for(;;);
	return 0;
}
