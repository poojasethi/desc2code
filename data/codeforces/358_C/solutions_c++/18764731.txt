#include<bits/stdc++.h>
using namespace std;
typedef pair<int,int> ii;
const int N=1e5+9;
ii ar[N];
int n,cnt,take,i,num,j;
int main()
{
	ios::sync_with_stdio(false);cin.tie(0);
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>num;
    	ar[cnt]=ii(-num,cnt);
    	if(!num)
    	{
           sort(ar,ar+cnt);
           for(j=0;j<cnt;j++)
           { 
             if (j == ar[0].second) cout<<"pushStack\n";
             else if (j == ar[1].second) cout<<"pushQueue\n";
             else if (j == ar[2].second)cout<<"pushFront\n";
             else cout<<"pushBack\n";
           }
           take=3;
           if(cnt<take)take=cnt;
           cout<<take;
           if (take >= 1) cout<<" popStack";
           if (take >= 2) cout<<" popQueue";
           if (take >= 3) cout<<" popFront";
           cout<<"\n";
           cnt=0;
    	}
    	else cnt++;
    }
    for(i=0;i<cnt;i++)
    	cout<<"pushStack\n";
	return 0;
}
