#include <stdio.h>
#include<algorithm>
int main() {
int n;
scanf("%d",&n);
int k,a[100002],c[100002],b[100022],i,j=0;
b[j]=0;j++;
for(i=1;i<=n;i++)
{
	scanf("%d",&a[i]);
	c[i]=a[i];
	if(a[i]==0)
	{
		b[j]=i;j++;
	}
}

for(i=1;i<j;i++)
{//printf("j=%d ",b[i]);
	if(-b[i-1]+b[i]==0||-b[i-1]+b[i]==1)
	{printf("0\n");}
	if(-b[i-1]+b[i]==2)
	{printf("pushStack\n1 popStack\n");
	}
	if(-b[i-1]+b[i]==3)
	{printf("pushStack\npushQueue\n2 popStack popQueue\n");}
	if(-b[i-1]+b[i]>3)
	{
		std::sort(a+b[i-1],a+b[i]);
		for(k=b[i-1]+1;k<b[i];k++)
		{
			if(c[k]==a[b[i]-1])
			{printf("pushStack\n");a[b[i]-1]=-100;}
			else if(c[k]==a[b[i]-2])
			{printf("pushQueue\n");a[b[i]-2]=-100;}
			else if(c[k]==a[b[i]-3])
			{printf("pushFront\n");a[b[i]-3]=-100;}
			else			printf("pushBack\n");
		}
	printf("3 popStack popQueue popFront\n");
	}
	
}//printf("%d",b[j-1]);
if(b[j-1]!=n)
	{//printf("ef");
		for(k=b[j-1];k<n;k++)
		printf("pushBack\n");
	}
	return 0;
}
