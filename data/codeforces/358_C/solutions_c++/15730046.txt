#include <cstdio>
#include <algorithm>
using namespace std;

int add[100005], buffer[100005];

bool cmp(int x, int y) { return x > y; }

int main() {
	int n, a, num = 0, stasz = 0, quesz = 0, dqfsz = 0;
	scanf("%d", &n);
	while (n--) {
		scanf("%d", &a);
		if (a) { add[num] = buffer[num] = a; ++num; }
		else { // extract
			sort(buffer, buffer + num, cmp);
			if (num == 0) printf("0\n");
			else if (num == 1) printf("pushStack\n1 popStack\n");
			else if (num == 2) printf("pushStack\npushQueue\n2 popStack popQueue\n");
			else {
				for (int i = 0; i < num; ++i)
					if (add[i] == buffer[0] && !stasz) { printf("pushStack\n"); stasz = 1; }
					else if (add[i] == buffer[1] && !quesz) { printf("pushQueue\n"); quesz = 1; }
					else if (add[i] == buffer[2] && !dqfsz) { printf("pushFront\n"); dqfsz = 1; }
					else printf("pushBack\n");
				printf("3 popStack popQueue popFront\n");
			}
			num = stasz = quesz = dqfsz = 0;
		}
	}
	while (num--) printf("pushStack\n");
	return 0;
}
