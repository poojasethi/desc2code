#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
#include<string>
#include<vector>
using namespace std;

priority_queue<pair<int,int>,vector<pair<int,int> >,less<pair<int,int> > >que;
string Push[]={"pushBack","pushStack","pushQueue","pushFront"};
string Pop[]={"popBack","popStack","popQueue","popFront"};
int index[100100],num[100100],n;
string ans[100100];

int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)scanf("%d",&num[i]);
	for(int i=0;i<n;i++){
		if(num[i]){
			que.push(make_pair(num[i],i));
		}else {
			for(int j=1;j<=3&&!que.empty();j++){
				ans[que.top().second]+=Push[j];
				index[i]=j;
				que.pop();
			}
			while(!que.empty())que.pop();
		}
	}
	for(int i=0;i<n;i++){
		if(num[i]){
			if(ans[i]!="")cout<<ans[i]<<endl;
			else cout<<Push[0]<<endl;
		}else {
			cout<<index[i];
			for(int j=1;j<=index[i];j++){
				cout<<" "<<Pop[j];
			}
			cout<<endl;
		}
	}
	return 0;
}
