#include <iostream>

using namespace std;

int main()
{
	int n;
	cin >> n;
	int st,qf,df;
	int si,qi,di;
	int temp;
	int le = 0;
	st = qf = df = -1;
	si = qi = di = -1;
	for(int index = 1;index <= n;index++){
		cin >> temp;
		if(temp > 0){
			if(temp > st){
				df = qf; di = qi;
				qf = st; qi = si;
				st = temp;
				si = index;
			}
			else if(temp > qf){
				df = qf; di = qi;
				qf = temp;
				qi = index;
			}
			else if(temp > df) {df = temp; di = index;}
		}
		
		else{
			for(int index1 = le + 1;index1 < index;index1++){
				if(index1 == si) cout << "pushStack\n";
				else if(index1 == qi) cout << "pushQueue\n";
				else if(index1 == di) cout << "pushFront\n";
				else cout << "pushBack\n";
			}
			le = index;
			int k;
			cout << (k = (si > 0) + (qi > 0) + (di > 0));	
			if(k >= 1) cout << " popStack";
			if(k >= 2) cout << " popQueue";
			if(k == 3) cout << " popFront";
			cout << "\n";
		
			st = qf = df = -1;
			si = qi = di = -1;
		}
	}
	for(int index1 = le + 1;index1 <= n;index1++) cout << "pushBack\n";
	return 0;
}
