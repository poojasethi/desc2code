#include<bits/stdc++.h>
using namespace std;
long long int top;
int n,ar[1000];
int main(){
	scanf("%d",&n);
	for(int i=1 ; i<=n ; i++){
		scanf("%d",&ar[i]);
	}
	sort(ar+1,ar+n+1);
	top=ar[n];
	for(int i=n-1 ; i>0 ; i--){
		while(ar[i]>=ar[i+1] && ar[i]!=0){
			ar[i]--;
		}
		top+=ar[i];
	}
	printf("%lld\n",top);
	return 0;
}
