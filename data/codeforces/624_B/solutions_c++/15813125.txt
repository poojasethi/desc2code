#include <bits/stdc++.h>
using namespace std;
long long n,A[30],sum = 0, lim;
int main(){
	cin >> n;
	for(int i = 0; i<n; i++){
		cin >> A[i];
	}
	sort(A,A+n,greater<int>());
	lim = A[0]+1;
	for(int i = 0;i<n && lim > 0; i++){
		if(A[i] < lim) sum+=A[i],lim = A[i];
		else sum+=lim-1,lim=lim-1;
	}
	printf("%I64d\n", sum);
}
