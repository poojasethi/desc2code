#include <bits/stdc++.h>

using namespace std;

long long N, a[30], sum = 0; 

int main(){

	cin >> N;

	for (int i=0; i<N; i++) cin >> a[i];

	sort(a,a+N);

	sum += a[N-1];
	for (int i=N-2; i>=0; i--){
		if (a[i] >= a[i+1]) a[i] = a[i+1] - 1;

		if (a[i] > 0) sum += a[i];
	}

	cout << sum << endl;

	return 0;
}
