#include<iostream>
using namespace std;
long long n,i,j,ans,a[27];
main (){
  cin>>n;
  for (i=1; i<=n; i++)
    cin>>a[i];
  for (i=1; i<=n; i++){
    for (j=1; j<=n; j++){
      if (a[i]==0) break;
      if (a[i]==a[j] && i!=j) {
        a[i]--;
        j=0;
      }
    } 
  ans=ans+a[i];
  }
  cout<<ans<<endl;
}
