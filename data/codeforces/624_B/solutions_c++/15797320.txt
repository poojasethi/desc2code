#include<bits/stdc++.h>
using namespace std;
int main()
{
	
	long long n,sum=0;
	cin>>n;
	vector<long long > v(n);
	for(int i=0;i<n;cin>>v[i],i++);
	
	sort(v.begin(),v.end());
	long long ans=v[n-1];
	for(int i=n-2;i>=0;i--)
	{
		if(v[i]>=v[i+1]) v[i]=v[i+1]-1;
		if(v[i]==0) break;
		ans+=v[i];
	}
	cout<<ans<<endl;
}
