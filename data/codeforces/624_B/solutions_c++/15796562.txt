#include <bits/stdc++.h>

using namespace std;

int n;
long long a[30], res, tmp;

int main() {
	cin >> n;
	for(int i = 0; i < n; i ++) cin >> a[i];
	sort(a, a+n);
	tmp = 1e12;
	for(int i = n-1; i >= 0; i --) {
		tmp = min(tmp, a[i]);
		if(tmp) {
			res += tmp;
			tmp --;
		}
	}
	cout << res << endl;
	return 0;
}
