#include<iostream>
#include<algorithm>
#define int long long
using namespace std;

main() {
  int n, a[26];
  cin>>n;
  for (int i=0;i<n;i++)
    cin>>a[i];
  sort(a, a + n);
  int ans = 0, last = 1000000001;
  for (int i=n-1;i>=0;i--) {
    last = max(0ll, min(last - 1, a[i]));
    ans += last;
  }
  cout<<ans<<endl;
  return 0;
}
