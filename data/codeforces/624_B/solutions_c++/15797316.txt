#include <bits/stdc++.h>

using namespace std;


int main()
{
	set<int> conj;
	int n,aux;
	long long ans=0;
	scanf("%d",&n);
	for(int i=0; i<n; i++){
		scanf("%d",&aux);
		while(aux){
			if(conj.count(aux)==0){
				conj.insert(aux);
				ans+=aux;
				break;
			}
			aux--;
		}
	}
	cout << ans << endl;
	return 0;
}
