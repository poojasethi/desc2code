#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int n,a[26],l,i;
    long long c=0;
    cin >> n;
    for(i=0;i<n;i++) cin>>a[i];
    sort(a,a+n);
    for(i=n-1,l=a[i];i>=0&&l>0;--i)
        if(a[i]>=l) c+=l--;
        else c+=a[i],l=a[i]-1;
    cout << endl << c;
    return 0;
}
