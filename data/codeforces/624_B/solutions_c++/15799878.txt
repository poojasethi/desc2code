#include <bits/stdc++.h>
#define ll long long int
using namespace std;
map<int,int>m;
int main(){
	ll res=0,n,a[30]={0},aux;
	cin>>n;
	for(int i=0;i<n;i++)cin>>a[i];
	sort(a,a+n);
	for(int i=n-1;i>=0;i--){
		aux=a[i];
		while(m[aux]&&aux){
			aux--;
		}
		res+=aux;
		m[aux]=1;
	}
	cout<<res<<endl;
}
