#include<bits/stdc++.h>
using namespace std;
int ar[27];
int main()
{
	int n;
	long long ans=0;
	cin>>n;
	for(int i=0;i<n;i++) 
		cin>>ar[i];
	for(int q=0;q<n;q++)
	{
		sort(ar,ar+n);
		reverse(ar,ar+n);
		int s=ar[0];
		if(s>=0)
		{
		ans+=s;
		ar[0]=0;
		for(int i=0;i<n;i++) if(ar[i]==s) ar[i]--;
		}
	}
	cout<<ans<<endl;
	return 0;
}
