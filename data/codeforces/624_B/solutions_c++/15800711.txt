#include <bits/stdc++.h>
using namespace std;
const int maxn=30;
int a[maxn],n;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	sort(a+1,a+1+n);
	for(int i=n-1;i;i--)
		if(a[i]>=a[i+1])a[i]=a[i+1]-1;
	long long ans=0;
	for(int i=1;i<=n;i++)
		if(a[i]>0)ans+=a[i];
	cout<<ans<<endl;
	return 0;
}
