#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int i, n, a[26];
    cin >> n;
    for(i = 0; i < n; i++) cin >> a[i];
    sort(a, a + n);
    long long c = a[n - 1];
    for(i = n - 2; i >= 0; i--) {
        while(a[i] && a[i] >= a[i + 1]) a[i]--;
        c += a[i];
    }
    cout << c;
}
