#include<iostream>
#include<algorithm>
using namespace std;
long long n, a[1000],i,ans;
int main(){
	cin>>n;
	for(i=0;i<n;i++){
		cin>>a[i];
}
sort(a,a+n);
reverse(a,a+n);
i=0;
ans=a[0];
int t=a[0];
for(i=1;i<n;i++){
	if (a[i]>=t) {ans=ans+t-1; if (t>1) t--;}
	else {ans+=a[i]; t=a[i];}
	
}cout<<ans;
}
