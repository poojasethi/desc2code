#include<bits/stdc++.h>
using namespace std;
int main() {
    long long int ara[26];
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) cin>>ara[i];

    long long int sum=0,mn=LLONG_MAX;
    sort(ara,ara+n);
    for(int i=n-1;i>=0;i--){
        mn=min(mn-1,ara[i]);
        if(mn>0) sum=sum+mn;
    }
    cout<<sum;
}
