#include<cstdio>

int n, a, b[30];
long long ans;

bool check(int x, int y) {
	for(int i = 0; i < y; i++) if (x == b[i]) return false;
	return true;
}

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &a);
		while (a && !check(a, i)) a--;
		ans += b[i] = a;
	}
	printf("%I64d\n", ans);
}
