#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long n,ans=1e18,a[27],c=0;
    cin>>n;
    for(int i=0 ; i < n ; i++)cin>>a[i];
    sort(a,a+n);
    for(int i=n-1 ; i >= 0 ; i--){
         ans=min(ans,a[i]);
         if(ans>=0)c+=ans;
         ans--;
    }cout<<c<<endl;
    return 0;
}
