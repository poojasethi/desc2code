#include <bits/stdc++.h>
using namespace std;
#define ll long long
map<int,int> mp;
int main()
{
    int T,m;
    ll sum=0;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d",&m);
        while(mp[m]==1&&m!=0) m--;
        mp[m]++;
        sum+=m;
    }
    printf("%I64d\n",sum);
    return 0;
}
