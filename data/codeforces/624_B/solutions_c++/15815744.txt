#include <cstdio>
#include <cstring>
#include <algorithm>
#include <set>
using namespace std;

set<int> S;

int main()
{
	int n; scanf("%d", &n);
	long long ans = 0;
	while(n--) {
		int x; scanf("%d", &x);
		while(x && S.count(x)) x--;
		ans += x;
		S.insert(x);
	}

	printf("%lld\n", ans);

	return 0;
}
