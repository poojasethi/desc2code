alphabet = int(raw_input())
limitations = [int(tmp) for tmp in raw_input().split(' ')]

limitations.sort(reverse = True)

# print(limitations)
ans, last = 0, 10**9 + 5

for limit in limitations:
	if(limit >= last):
		ans += last - 1
		last = last - 1
	else:
		ans += limit
		last = limit
	if(last == 0):
		break
print(ans)
