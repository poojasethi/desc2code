__author__ = 'aste'


def main():
    n = int(raw_input())
    a = map(int, raw_input().split())

    a.sort()

    res = a[n - 1]
    for i in range(n - 2, -1, -1):
        if a[i] >= a[i + 1]:
            a[i] = a[i + 1] - 1
        if a[i] == 0:
            break
        res += a[i]

    print res

main()
