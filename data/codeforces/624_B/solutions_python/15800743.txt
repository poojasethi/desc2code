lengthl=0
dicts={}
n=input()

arr=map(int,raw_input().split())
arr.sort(reverse=True)

for i in range(n):
    if arr[i] not in dicts:
        dicts[arr[i]]=1
        lengthl+=arr[i]
    else:
        t=arr[i]
        while t in dicts:
            t-=1
        if(t>=0):
            dicts[t]=1
            lengthl+=t
print lengthl
