n = input()
vals = map(int,raw_input().split())
vals = sorted(vals, reverse=True)
curr = 10000000000000 
sm = 0
for a in vals:
    if a == curr and a != 1:
        curr = a-1
    elif a < curr:
        curr = a
    else:
        curr = curr - 1 
    sm += curr
    if curr == 0:
        break
print sm
