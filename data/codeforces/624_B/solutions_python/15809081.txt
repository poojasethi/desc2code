n = int( raw_input() )
L  = sorted( map( int, raw_input().split() ) )

ans = L[n-1]


took = {}
#print ans

took[ L[n-1] ] = 1
n -= 1
#print L, n-1
while n > 0:

	if L[n-1] >= L[n]:
		L[n-1] = max( 0, L[n]-1 )
	#print L	
	if L[n-1] not in took:
		ans += L[n-1]
	#print ans
	took[ L[n-1] ] = 1
	
	n -= 1
	
print "%d" % (ans)
