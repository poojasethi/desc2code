n=input()
counts=map(int,raw_input().split())
counts.sort()
ans=0
mapi={}
i=n-1
while i>=0:
    if counts[i] not in mapi:
        mapi[counts[i]]=1
        ans+=counts[i]
    else:
        temp=counts[i]
        while temp>0 and temp in mapi:
            temp-=1
        mapi[temp]=1
        ans+=temp
    i-=1
print ans
