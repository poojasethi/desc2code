#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
a = ia()

a.sort()
a.reverse()

ans, s = 0, set()
for e in a:
    while e in s: e-=1
    if e>0      : s.add(e)
    ans += e

print ans
