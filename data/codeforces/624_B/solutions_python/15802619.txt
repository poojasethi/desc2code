n = int(raw_input())
s = raw_input()
a = s.split(' ')
for i in range(len(a)):
    a[i] = int(a[i])

a.sort()
i = len(a) - 1
uniques = []
while i >= 0:
    poss = True
    while a[i] in uniques:
        a[i] -= 1
        if a[i] < 0:
            poss = False
            break
    if poss:
        uniques.append(a[i])
    i -= 1

print sum(uniques)
