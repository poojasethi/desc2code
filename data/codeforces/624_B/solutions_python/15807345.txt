n=input()
x=[]
x=map(int,raw_input().split());
x=sorted(x,reverse=True);
s=0;
c=0;
#print x;
for i in range(n):
    if i==0:
        c=x[i];
        s=s+x[i];
    else:
        if c<=x[i] and s+c-1>=s:
            s=s+c-1;
            c=c-1;
        elif c>x[i]:
            s=s+x[i];
            c=x[i];
print s;
