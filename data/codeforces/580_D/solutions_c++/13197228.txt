#include <iostream>
#include <string.h>
#include <stdio.h>

using namespace std;

long long cc[1<<20][20];
int c[20][20];
int a[20];
int n,m,k,f,b,d;

long long go(int mk,int pr,int sz)
{
	if(sz == m) return 0;
	long long &tp = cc[mk][pr];
	if(tp != -1) return tp;
	tp = 0;
	for(int i=1;i<=n;i++)
	{
		if(mk&(1<<i)) continue;
		tp = max(tp,c[pr][i]+a[i]+go(mk|(1<<i),i,sz+1));
	}
	return tp;
}
int main()
{
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif
	
	cin>>n>>m>>k;
	for(int i=1;i<=n;i++) cin>>a[i];
	for(int i=1;i<=k;i++)
	{
		cin>>f>>b>>d;
		c[f][b]=d;
	}
	memset(cc,255,sizeof(cc));
	cout<<go(0,0,0)<<endl;
	return 0;
}
