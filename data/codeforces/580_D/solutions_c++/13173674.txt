#include<bits/stdc++.h>
#define ll long long int
using namespace std;

ll dp[19][1<<19],arr[20][20],a[20],n,m,k;

ll solve(ll last , ll mask, ll co){
    if(co == m) return 0;
    if(dp[last][mask]!=-1)  return dp[last][mask];
    ll ans = 0;
    for(int i=0;i<n;i++)
        if((mask&(1<<i))==0)
            ans = max(ans, solve(i,mask|(1<<i),co+1) + arr[last][i] + a[i]);
    return dp[last][mask]=ans;
}

int main(){
    memset(dp,-1,sizeof(dp));
    cin>>n>>m>>k;
    for(int i = 0; i < n; i++)  cin>>a[i];
    for(int i=0;i<k;i++){
        int x,y,z;
        cin>>x>>y>>z;
        arr[x-1][y-1] = z;
    }
    cout<<solve(n,0,0);
    return 0;
}
