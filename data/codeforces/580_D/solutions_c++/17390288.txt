#include <bits/stdc++.h>

using namespace std;

int extra[19][19],cost[19];
long long ans,dp[19][1<<18];

int main(){
	int n,m,k,x,y,t;
	scanf("%d%d%d",&n,&m,&k);
	for (int i=0;i<n;++i)
		scanf("%d",&cost[i]);
	while (k--){
		scanf("%d%d",&x,&y);
		scanf("%d",&extra[x-1][y-1]);
	}
	for (int i=0;i<n;++i)
		dp[i][1<<i]=cost[i];
	int S=1<<n;
	for(int s=0;s<S;s++){
		int cnt = __builtin_popcount(s);
		for(int i=0;i<n;i++)
			if( (s >> i) & 1){
				if(cnt==m)
					ans = max(ans,dp[i][s]);
				for(int j=0;j<n;j++)
					if (((s>>j) & 1)==0){
						t = s | (1<<j);
						dp[j][t]=max(dp[j][t],dp[i][s]+cost[j]+extra[i][j]);
					}
			}
	}
	cout << ans << endl;
	return 0;
}
