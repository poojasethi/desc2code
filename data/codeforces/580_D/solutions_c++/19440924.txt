#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

int map[20][20],a[20];
long long dp[300000][20];
int n,m,k;

int main (){
	int x,y,z;
	memset (dp,0,sizeof(dp));
	scanf ("%d%d%d",&n,&m,&k);
	for (int i=0;i<n;i++)
		scanf ("%d",&a[i]);
	for (int i=0;i<k;i++){
		scanf ("%d%d%d",&x,&y,&z);
		map[--x][--y]=z;
	}
	for (int i=0;i<n;i++)
		dp[1<<i][i]=a[i];
	for (int i=1;i<(1<<n);i++)
		for (int j=0;j<n;j++)
			if ((1<<j)&i)
				for (int k=0;k<n;k++)
					if (!((1<<k)&i))
						dp[(1<<k)|i][k]=max(dp[(1<<k)|i][k],dp[i][j]+map[j][k]+a[k]);
	long long ans=0;
	for (int i=1;i<1<<n;i++){
		int num=0;
		for (int j=0;j<n;j++)
			if ((1<<j)&i) num++;
		if (num!=m) continue ;
		for (int j=0;j<n;j++)
			ans=max(ans,dp[i][j]);
	}
	printf ("%lld\n",ans);
}
