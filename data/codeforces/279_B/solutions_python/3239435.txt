#!/usr/bin/env/ python
import bisect

n,t = map(int,raw_input().split())
a = map(int,raw_input().split())

l = len(a)
b = []
for i in range(l):
    if i!=0:
        b.append(a[i]+b[i-1])
    else:
        b.append(a[i])

maxn = 0
for i in range(l):
    x = bisect.bisect(b,b[i]+t-a[i])
    if x-i+1>maxn:
        maxn = x-i

print maxn
