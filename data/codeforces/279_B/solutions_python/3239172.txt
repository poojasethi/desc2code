from sys import stdin

def line():
	return stdin.readline().strip()

n, t = map(int, line().split())
A = map(int, line().split())
x = y = 0
sum = 0
range = 0
while x < n:
	if sum < t and y < n:
		sum += A[y]
		y += 1
	else:
		sum -= A[x]
		x += 1
	if sum <= t:
		range = max(range, y - x)
print range
