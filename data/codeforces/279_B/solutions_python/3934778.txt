#!/usr/bin

def B2P ( n, t, books ):
    tempsum = maxIndex = a = b = 0
    for b in xrange(n):
        tempsum = tempsum + books[b]
        if tempsum <= t:
            maxIndex = max(-~(b-a), maxIndex)
        else:
            tempsum = tempsum - books[a]
            a = a + 1
    return maxIndex

n, t = map( int, raw_input().split() )
print B2P ( n, t, map(int, raw_input().split()) )
