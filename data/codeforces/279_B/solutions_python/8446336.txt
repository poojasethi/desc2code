def solve(n, a, t):
    i, j = 0, 0
    res, c = 0, 0 
    while i < n:
        while j < n and c+a[j] <= t:
            c += a[j]
            j += 1
        res = max(res, j-i)
        if j == n: break
        c -= a[i]
        i += 1
    return res

n, t = map(int, raw_input().split())
a = map(int, raw_input().split())
print solve(n, a, t)
