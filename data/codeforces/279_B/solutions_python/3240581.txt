from sys import stdin
def main():
    n, l = map(int, stdin.readline().split())
    a = map(int, stdin.readline().split())
    e = t = c = ans = 0
    for s in xrange(n):
        while e < n and t + a[e] <= l:
            t += a[e]
            c += 1
            e += 1
        ans = max(ans, c)
        t -= a[s]
        c -= 1
    print ans
main()
