n,t=map(int,raw_input().split())
z=map(int,raw_input().split())
if min(z)>t:
    print 0
else:
    st=0
    ma=0
    end=0
    tot=z[st]
    while(1):
        if(tot<=t):
            ma=max(ma,end-st)
            if(end+1<n):
                end=end+1
                tot=tot+z[end]
            else:
                break
        else:
            tot=tot-z[st]
            st=st+1
    print ma+1
