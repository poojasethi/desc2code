N, T = map(int, raw_input().split())
tiempo = map(int, raw_input().split())

lenbest = 0
izq=N
der=N-1
suma=0
while izq>=0:
	while suma<=T:
		lenbest = max(lenbest, der-izq+1)
		izq-=1
		if izq==-1:
			print lenbest
			exit(0)
		suma += tiempo[izq]
	while suma>T:
		suma -= tiempo[der]
		der -= 1
