n,t = map(int,raw_input().split())

ls = map(int,raw_input().split())
sum=0
count=0
ans=-1
j=0
for i in range(n):
    sum += ls[i]
    count+=1
    if sum>t:
         sum -= ls[j]
         j+=1
         count-=1
         if ans<count:
             ans = count

if count>ans:
    ans = count

print ans
