import sys,math



hulp=[int(x) for x in sys.stdin.readline().split()]

n,t=hulp[0],hulp[1]

a=[int(x) for x in sys.stdin.readline().split()]

som,begin,lengte=0,0,0

for x in range(n):
    som+=a[x]
    while som>t:
        som-=a[begin]
        begin+=1
    lengte=max(lengte,x-begin+1)
    
print lengte
