#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using namespace std;

int s[200100], z[200100][26], to;
char m[200100];

void ins(void)
{
	int i = 0;
	char *p = m;
	while (*p)
	{
		int id = *p - 'a';
		if (z[i][id] == 0)
		{
			memset(z[to], 0, sizeof(z[to]));
			z[i][id] = to++;
		}
		i = z[i][id];
		p++;
	}
}

int main()
{
	int n, i, j, x, k;
	to = 1;
	memset(z[0], 0, sizeof(z[0]));
	scanf("%d%d", &n, &k);
	while (n--)
	{
		scanf(" %s", m);
		ins();
	}
	for (i = to - 1; i >= 0; --i)
	{
		s[i] = 0;
		int fl = 0;
		for (j = 0; j < 26; ++j)
		  if ((x = z[i][j]))
		  {
			  fl = 1;
			  s[i] |= s[x] ^ 3;
		  }
		if (!fl) s[i] = 1;
	}
	if (s[0] == 3) puts("First");
	else if (s[0] <= 1) puts("Second");
	else if (k & 1) puts("First");
	else puts("Second");
	return 0;
}
