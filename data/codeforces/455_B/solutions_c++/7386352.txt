#include <bits/stdc++.h>
using namespace std;

const int N = 100005;

int n, k, e[N][26], sz;

void insert(const string& s) {
	int n = s.size();
	int x = 0;
	for (int i = 0; i < n; ++i) {
		int c = s[i] - 'a';
		if (!e[x][c]) e[x][c] = sz++;
		x = e[x][c];
	}
}

int wantToWin(int x) {
	int ret = 0;
	for (int i = 0; i < 26; ++i) if (e[x][i]) {
		ret |= !wantToWin(e[x][i]);
	}
	return ret;
}

int wantToLose(int x) {
	int ret = 0;
	int leaf = 1;
	for (int i = 0; i < 26; ++i) if (e[x][i]) {
		leaf = 0;
		ret |= !wantToLose(e[x][i]);
	}
	if (leaf) return 1;
	return ret;
}

int main() {
	sz = 1;
	cin >> n >> k;
	for (int i = 0; i < n; ++i) {
		string s;
		cin >> s;
		insert(s);
	}
	int w = wantToWin(0);
	int l = wantToLose(0);
	if (w) {
		if (l) puts("First");
		else puts(k % 2 ? "First" : "Second");
	} else {
		puts("Second");
	}
}
