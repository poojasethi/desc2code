#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxn=2000050;
int ns=1,son[maxn][30]={0};
bool sg[maxn][2]={0};
void insert(char *str,int l)
{
	int i,now=1;
	for(i=0;i<l;i++)
	{
		int t=str[i]-'a';
		if(!son[now][t])
			son[now][t]=++ns;
		now=son[now][t];
	}
}
void get_sg(int x)
{
	int i;
	bool z=0;
	for(i=0;i<26;i++)
		if(son[x][i])
		{
			get_sg(son[x][i]);
			z=1;
			sg[x][0]|=!sg[son[x][i]][0];
			sg[x][1]|=!sg[son[x][i]][1];
		}
	if(!z) sg[x][1]=1;
}
int n,K;
char ans[2][10]={"Second","First"},s0[maxn];
