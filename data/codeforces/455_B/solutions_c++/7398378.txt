#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
typedef long long LL;
#define rep(i,n) for (int i=0;i<n;++i)
const int N=110005;
int n,m,k,T,c[N][26],f[N]; char s[N];
int main()
{
	scanf("%d%d",&n,&k),T=1;
	rep(i,n){
		scanf("%s",s),m=strlen(s); int x=1;
		rep(j,m){
			int y=s[j]-97;
			if (!c[x][y]) c[x][y]=++T;
			x=c[x][y];
		}
	}
	for (int i=T,x;i;--i){
		f[i]=0; bool flag=0;
		rep(j,26) if (x=c[i][j])
			flag=1,f[i]|=f[x]^3;
		if (!flag) f[i]=1;
	}
	//printf("%d\n",f[1]);
	if (f[1]==3) puts("First");
	else if (f[1]<=1) puts("Second");
	else if (k&1) puts("First"); else puts("Second");
	return 0;
}
