#include <iostream>  
#include<string>
#include<cstring>
#include<cstdio>
#define maxn 100005
using namespace std;

int next[maxn][26];
int vis[maxn];
int top, root;
char s[maxn];

void insert(void)
{
	int len = strlen(s);
	int now = root, tmp;
	for(int i = 0; i < len; i++) 
	{
		tmp = s[i] - 'a';
		if(!next[now][tmp]) next[now][tmp] = ++top;
		now = next[now][tmp];
	}
}
int dfs(int now)
{
	int ok = 0, ans = 0;
	for(int i = 0; i < 26; i++)
		if(next[now][i]) 
		{
			ok = 1;
			ans |= dfs(next[now][i])^3;
		}
	if(!ok) ans = 1;
	return ans;
}
int main(void)
{
	int n, k;
	top = root = 1;
	scanf("%d%d", &n, &k);
	for(int i = 1; i <= n; i++)
	{
		scanf("%s", s);
		insert();
	}
	int tmp = dfs(root);
	if(tmp == 3) printf("First\n");
	else if(k&1 && tmp == 2) printf("First\n");
	else printf("Second\n");
	return 0;
}
