#include <iostream>

using namespace std;

int n,k,co,f[2000000][30],w[2000000],l[2000000];
string s;

void add()
{
    int c=0;
    for (int i=0;i<s.length();i++)
    {
        int t=s[i]-'a';
        if (f[c][t]) c=f[c][t];
        else f[c][t]=++co,c=co;
    }
}

void dfs(int u)
{
    int c=0;
    for (int i=0;i<26;i++)
    {
        if (!f[u][i]) continue;
        c++;
        dfs(f[u][i]);
        if (!w[f[u][i]]) w[u]=1;
        if (!l[f[u][i]]) l[u]=1;
    }
    if (!c) l[u]=1;
}

int main()
{
    cin >> n >> k;
    for (int i=1;i<=n;i++)
        cin >> s,add();
    dfs(0);
    if (w[0] and l[0])
        cout << "First";
    else if (l[0])
        cout << "Second";
    else if (w[0])
    {
        if (k%2)
            cout << "First";
        else cout << "Second";
    }
    else cout << "Second";
}
