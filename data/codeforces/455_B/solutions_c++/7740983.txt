#include<stdio.h>
#include<string.h>
char s[111111];
int tr[111111][26];
int si=1;
int intsert()
{
	int i;
	int t=strlen(s);
	int u=0;
	for(i=0;i<t;i++)
	{
		if(!tr[u][s[i]-'a'])
		{
		tr[u][s[i]-'a']=si;
		si++;	
		}
		u=tr[u][s[i]-'a'];
		
	}
}
int dfs1(int t)
{
	int i;
	for(i=0;i<26;i++)
	if(tr[t][i]!=0)
	{
		if(!dfs1(tr[t][i]))
		return 1;
	}
	return 0;
	
}
int dfs2(int t)
{
	int i;
	int f=1;
   for(i=0;i<26;i++)
	if(tr[t][i]!=0)
	{
		f=0;
		if(!dfs2(tr[t][i]))
		return 1;
	}
	if(f)return 1;
	return 0;
}
int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	memset(tr,0,sizeof(tr));
	for(int i=0;i<n;i++)
	{
	scanf("%s",s);
	intsert();
	}

	int a=dfs1(0);
	int b=dfs2(0);
	if(!a)
    printf("Second\n");
    else if(a&&b)printf("First\n");
    else 
    {
    	if(k&1)printf("First\n");
    	else printf("Second\n");
    }
	
}
