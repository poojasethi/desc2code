#include <cstdio>
#include <algorithm>
using namespace std;
const int maxn = 1e5 + 10;
int n,K,tot,root,Flag;
int ch[maxn][26],d[maxn];

void insert(int &t,char *s){
	if(!t) t = ++tot;
	if(s[0] != '\0') insert(ch[t][s[0]-'a'],s + 1);
}

int dfs(int x){
	for(int i = 0; i < 26; i++)
		if(ch[x][i] && !dfs(ch[x][i])) return 1;
	return 0;
}

int Dfs(int x){
	for(int i = 0; i < 26; i++)
		if(ch[x][i]) d[x]++;
	if(!d[x]) return 1;
	for(int i = 0; i < 26; i++)
		if(ch[x][i] && !Dfs(ch[x][i])) return 1;
	return 0;
}

int main(){
#ifndef ONLINE_JUDGE
	freopen("B.in","r",stdin);
	freopen("B.out","w",stdout);
#endif
	scanf("%d %d",&n,&K);
	for(char s[maxn]; n--; )
		scanf("\n%s",s),insert(root,s);
	if(!dfs(root)) puts("Second");
	else if(Dfs(root)) puts("First");
	else K & 1?puts("First"):puts("Second");
	return 0;
}
