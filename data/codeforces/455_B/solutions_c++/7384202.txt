#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,t,ty,K;
int a[N][26],sg[N];
char p[N];
int dfs(int x) {
	int i;
	int f[40]={},k=0;
	For(i,0,25) if (a[x][i]) f[dfs(a[x][i])]=1,k=1;
	sg[x]=0;
	for (;f[sg[x]];sg[x]++);
	if (!k) sg[x]=ty;
	return sg[x];
}
int main() {
	scanf("%d%d",&n,&K);
	For(i,1,n) {
		int v=0;
		scanf("%s",p+1);
		m=strlen(p+1);
		For(j,1,m) {
			int A=p[j]-'a';
			if (!a[v][A]) a[v][A]=++t;
			v=a[v][A];
		}
	}
	int A=dfs(0);
	ty=1;
	int B=dfs(0);
	if (A&&B) printf("First\n");
	else if (A&&(K&1)) printf("First\n");
	else printf("Second\n");
	return 0;
}
