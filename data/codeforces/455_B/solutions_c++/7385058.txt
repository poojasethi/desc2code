#include<cstdio>
#include <map>
using namespace std;

struct Node
{
    map<char, Node*> ch;
};


Node root, buf[123456];
int total;

void build(char* str)
{
    Node* node = &root;
    while(*str)
    {
        if(!node->ch[*str])
            node->ch[*str] = &buf[total++];
        
        node = node->ch[*str];
        str++;
    }
}

int go(Node* node, int ret)
{
    if(!node->ch.size())
        return ret;

    for(map<char, Node*>::iterator it = node->ch.begin(); it != node->ch.end(); it++)
        if(!go(it->second, ret))
            return 1;
    return 0;
}

char input[123456];

int main()
{
    int N, K;
    scanf("%d%d", &N, &K);

    while(N--)
    {
        scanf("%s", input);
        build(input);
    }


    if(!go(&root, 0))
        puts("Second");
    else if(!go(&root, 1))
        puts((K & 1)? "First": "Second");
    else
        puts("First");


    


    return 0;
}
