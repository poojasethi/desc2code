#include <bits/stdc++.h>

#define MAX_N 100010
#define SZ 26
int ch[MAX_N][SZ];
char buf[MAX_N];
bool dfs(int u, int ifNone) {
	bool haveSon = false;
	for(int i = 0; i < SZ; i ++) {
		if(ch[u][i]) {
			haveSon = true;
			if(!dfs(ch[u][i], ifNone))
				return true;
		}
	}
	if(haveSon == false)
		return ifNone;
	return false;
}
int main() {
	int N, K, sz = 0;
	scanf("%d%d", &N, &K);
	for(int i = 0; i < N; i ++) {
		scanf("%s", buf);
		for(int j = 0, u=0; buf[j]; j ++) {
			int v = buf[j] - 'a';
			if(ch[u][v] == 0) 
				ch[u][v] = ++ sz;
			u = ch[u][v];
		}
	}
	bool canWin = dfs(0, 0), canLose = dfs(0, 1);
	if(canWin && canLose || canWin && K % 2)
		puts("First");
	else
		puts("Second");
	return 0;
}
