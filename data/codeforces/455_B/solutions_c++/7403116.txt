using namespace std;
typedef long long ll;

const int N=1e5+5;

struct trie{
	trie *go[26];
	bool dp(){
		for(int i=0;i<26;i++) if(go[i])
			if(!go[i]->dp()) return 1;
		return 0;
	}
	bool dp2(){
		bool mark=0;
		for(int i=0;i<26;i++) if(go[i]){
			mark=1;
			if(!go[i]->dp2()) return 1;
		}
		if(!mark) return 1;
		return 0;
	}
		
} root[N],*newtrie=root+1;

char s[N];
int n,k;

int main(){
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++){
		scanf("%s",s);
		trie *now=root;
		for(char *j=s;*j;j++){
			if(!now->go[*j-'a'])
				now->go[*j-'a']=newtrie++;
			now=now->go[*j-'a'];
		}
	}
	if(root->dp())
		if(root->dp2())
			puts("First");
		else
			puts(k&1?"First":"Second");
	else
		puts("Second");
}
