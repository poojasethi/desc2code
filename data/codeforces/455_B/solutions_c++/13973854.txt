#include <cstring>
#include <cstdio>
const int N = 101000,M = 26;
int ch[N][M],sz,n,k;
char s[N];
void insert(const char* s) {
	int u = 0,l = strlen(s);
	for (int i = 0;i < l;u = ch[u][s[i++]-'a'])
		if (!ch[u][s[i]-'a'])
			ch[u][s[i]-'a'] = ++sz;
}
int dfs(int x) {
	int flag = 1,ans = 0;
	for (int i = 0;i < M;i++)
		if (ch[x][i]) {
			flag = 0;
			ans |= dfs(ch[x][i])^3;
		}
	if (flag) ans = 1;
	return ans;
}
int main() {
	scanf("%d%d",&n,&k);
	for (int i = 0;i < n;i++) {
		scanf("%s",s);
		insert(s);
	}
	int t = dfs(0);
	printf("%s\n",((t == 3)||((t > 1)&&(k&1)))?"First":"Second");
	return 0;
}
