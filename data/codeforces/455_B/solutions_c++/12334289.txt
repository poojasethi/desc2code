#include<algorithm>
#include<iostream>
#include<cstring>
#include<cstdio>
#include<vector>
#include<map>
using namespace std;
#define N 100005
int n,k,cnt;
int win[N],lose[N],son[N][26];
char s[N];
void insert(int len)
{
    int x=0;
    for(int i=1;i<=len;i++)
    {
        if(!son[x][s[i]-'a']) son[x][s[i]-'a']=++cnt;
        x=son[x][s[i]-'a'];
    }
}
void get_sg(int u)
{
    bool flag=1;
    for(int i=0;i<26;i++)
    if(son[u][i])
    {
        get_sg(son[u][i]);
        if(!win[son[u][i]]) win[u]=1;
        if(!lose[son[u][i]]) lose[u]=1;
        flag=0;
    }
    if(flag) lose[u]=1;
}
int main()
{
    cin>>n>>k;
    for(int i=1;i<=n;i++) scanf("%s",s+1),insert(strlen(s+1));
    get_sg(0);
    if(win[0]&&(lose[0]||k&1)) puts("First");
    else puts("Second");
}
