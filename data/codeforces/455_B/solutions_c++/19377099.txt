#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
// 2能必胜
// 1能必胜 k&1 
// 1能必胜必输
int tot,n,k,trie[100005][26];

void insert(string s){
	int x=0,i=0,y;
	for(;i<s.size();i++){
		y=s[i]-'a';
		if(!trie[x][y])
			trie[x][y]=++tot;
		x=trie[x][y];
	}
}

bool dfs1(int x){//先手必胜 
	for(int i=0;i<26;i++)
		if(trie[x][i]!=0)
			if(!dfs1(trie[x][i]))
				return 1;
	return 0;
}
bool dfs2(int x){//先手必输 
	int flag=1;
	for(int i=0;i<26;i++){
		if(trie[x][i]!=0){
			flag=0;
			if(!dfs2(trie[x][i]))
				return 1;
		}
	}
	return flag;
}

int main(){
	scanf("%d%d",&n,&k);
	string s;
	for(int i=0;i<n;i++)
		cin>>s,insert(s);
	int a=dfs1(0),b=dfs2(0);
	if(!a)puts("Second");
	else if(a==1&&b==1)puts("First");
	else if(a==1&&b==0)
		puts(k&1?"First":"Second");
}
