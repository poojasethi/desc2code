#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
#define f(i,n) for(int i=1;i<=n;i++)
#define rep(i,n) for(int i=0;i<n;i++)
#define fd(i,n) for(int i=n;i;i--)
#define F (100000007)
#define MAXN (200000+10)
int a[MAXN][26+1],n=1;
char s[MAXN];
bool can_win[MAXN]={0},can_lose[MAXN]={0};
int main()
{
	int m,k;
	scanf("%d%d",&m,&k);
	f(i,m)
	{
		scanf("%s",s);
		int len=strlen(s);
		int t=1;
		rep(j,len)
		{
			int p=s[j]-'a';
			if(!a[t][p]) a[t][p]=++n;
			t=a[t][p];
		} 
	}
	fd(i,n)
	{
		bool b=0;
		rep(j,26)if(a[i][j]){b=1;break;}
		if(!b)can_lose[i]=1;
		rep(j,26)if(a[i][j])
		{
			if(!can_win[a[i][j]])  can_win[i]=1;
			if(!can_lose[a[i][j]])can_lose[i]=1;
		}
	}
	bool b=0;
	if(!can_win[1]) b=0;
	else if(can_lose[1]) b=1;
	else if(k%2) b=1;
	else b=0;
	if(b)cout<<"First"<<endl;
	else cout<<"Second"<<endl;
	return 0;
}
