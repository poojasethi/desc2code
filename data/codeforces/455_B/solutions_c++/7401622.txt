#include<cstdio>
#include<cstring>
using namespace std;
int a[100005][26],f[100005];
char s[100005];
int node,i,n,K;
inline void Insert()
{
  scanf("%s",s+1);
  int L=strlen(s+1),p=1;
  for (int i=1;i<=L;i++)
  {
    int now=s[i]-'a';
    if (!a[p][now]) a[p][now]=++node,p=node;
    else p=a[p][now];
  }
}
//1:win
//2:lose
//3:win&lose
//4:no
int Dfs(int k)
{
  int son=0,go,sum[5]={0};
  for (int i=0;i<26;i++)
    if (go=a[k][i])
      son++,sum[Dfs(go)]++;
  if (!son) return f[k]=1;
  if (son==1) return f[k]=((sum[1]||sum[2])?(sum[1]>0)+1:(sum[3]>0)+3);
  if (sum[4]||sum[1]&&sum[2]) return f[k]=3;
  if (sum[1]||sum[2]) return f[k]=(sum[1]>0)+1;
  return f[k]=4;
}

int main()
{
  scanf("%d%d",&n,&K);
  node=1;
  for (i=1;i<=n;i++)
    Insert();
  Dfs(1);
  if (f[1]==3||f[1]==2&&(K&1)) puts("First");
  else puts("Second");
  return 0;
}
