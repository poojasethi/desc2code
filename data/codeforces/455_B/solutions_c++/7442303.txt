#include<cstdio>
#include<cstring>
int N,K,Cnt,l;
char S[100005];
int son[100005][30];
int f[100005],g[100005];
int main()
{
	scanf("%d%d",&N,&K);
	for(int i=1;i<=N;++i)
	{
		scanf("%s",S);
		l=strlen(S);
		for(int p=0,j=0;j<l;++j)
		{
			if(!son[p][S[j]-'a']) son[p][S[j]-'a']=++Cnt;
			p=son[p][S[j]-'a'];
		}
	}
	for(int i=Cnt;i>=0;--i)
	{
		for(int j=0;j<26&&!f[i];++j)
			if(son[i][j]&&!f[son[i][j]])
					f[i]=1;
	}
	for(int i=Cnt;i>=0;--i)
	{
		bool f1=0,f2=0;
		for(int j=0;j<26;++j)
			if(son[i][j])
			{
				if(g[son[i][j]]) f1=1;
				else f2=1;
			}
		if(!f1&&f2) g[i]=1;
		else g[i]=0;
	}
	if(!f[0]) printf("Second\n");
	else
	{
		if(g[0])
		{
			if(K&1) printf("First\n");
			else printf("Second\n");
		}
		else printf("First\n");
	}
	return 0;
}
