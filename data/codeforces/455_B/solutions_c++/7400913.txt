#include<cstdio>
#include<algorithm>
#include<string>
#include<iostream>
using namespace std;
string s1[100000];
int f(int a,int b,int i)
{

    if(a+1==b)return (s1[a].length()-i&1)+1;
    while(a<b&&s1[a].length()==i)++a;
    if(a==b)return 1;
    int c=a,d,x;bool bl1=0,bl2=0;
    while(1)
    {
        for(d=c+1;d<b&&s1[c][i]==s1[d][i];++d);
        x=f(c,d,i+1);c=d;
        bl1|=!(x>1);bl2|=!(x&1);
        if(c==b)break;
    }
    return bl1*2+bl2;
}
int main()
{
    int n,k,x;
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;++i)
        cin>>s1[i];
    sort(s1,s1+n);
    x=f(0,n,0);
    puts(x==3||x>1&&(k&1)?"First":"Second");
}
