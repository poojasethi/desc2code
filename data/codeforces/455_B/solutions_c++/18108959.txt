#include <bits/stdc++.h>

using namespace std;

const int MN = 1e5 + 5;

int tri[MN][26];
int dp[MN];
int sz = 1,n ,k;

void dfs(int s)
{
	int counter = 0 ,w=0 ;
	for(int i=0;i<26;++i){
		if(tri[s][i]){
			dfs(tri[s][i]);
			++counter;
			w |= (!(dp[tri[s][i]]&2))*2;
			w |= (!(dp[tri[s][i]]&1));
		}		
	}
	if(counter)
		dp[s] = w;
	else
		dp[s] = 1;
}

void insertTrie(string s)
{
	int p = 0;
	for(int i=0;i<s.size();++i){
		if(!tri[p][s[i]-'a']) tri[p][s[i]-'a'] = sz++;
		p = tri[p][s[i]-'a'];
	}
}

void input()
{
	cin >> n >> k;
	string s;
	for(int i=0;i<n;++i){
		cin >> s;
		insertTrie(s);
	}
}

int main()
{
	input();
	dfs(0);
	if(dp[0] == 3)
		cout << "First" << endl;
	else if(dp[0] == 2){
		if(k%2)
			cout << "First" << endl;
		else
			cout << "Second" << endl;
	}else
		cout << "Second" << endl;
	return 0;
}
