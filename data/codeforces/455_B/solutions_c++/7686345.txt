#include <cstdio>
#include <algorithm>
#include <cstring>

using namespace std;

const int D = 100010;

int n, k;
char s[D];

int dn;
int c[D][26];

bool f[D];
bool g[D];

int main() {	
	scanf("%d %d", &n, &k);
	int root = ++dn;
	for (int i = 1; i <= n; ++i) {
		scanf("%s", s);
		int len = strlen(s);
		int o = root;
		for (int j = 0; j < len; ++j) {
			if (!c[o][s[j] - 'a'])
				c[o][s[j] - 'a'] = ++dn;
			o = c[o][s[j] - 'a'];
		}
	}
	for (int i = dn; i >= 1; --i) {
		int sons = 0;
		for (int x = 0; x < 26; ++x)
			if (c[i][x]) {
				++sons;
				int j = c[i][x];
				if (!f[j]) f[i] = 1;
				if (!g[j]) g[i] = 1;
			}
		if (!sons) {
			f[i] = 0;
			g[i] = 1;
		}
	}
	if (!f[root]) {
		printf("Second\n");
	} else {
		if (g[root])
			printf("First\n");
		else
			printf("%s\n", (k & 1)? "First": "Second");
	}
}
