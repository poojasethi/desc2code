#include <cstring>
#include <cstdio>
int tr[100010][26],f[100010],g[100010],i,n,k,ans,tot=0;
char s[100010];
void add()
{
	scanf("%s\n",s);
	int now=0,i,j,len=strlen(s);
	for (i=0;i<len;++i)
	{
		j=s[i]-'a';
		if (!tr[now][j]) tr[now][j]=++tot;
		now=tr[now][j];
	}
}
void dfs(int x)
{
	f[x]=0;g[x]=1;int bo=0;
	for (int i=0;i<26;++i)
	if (tr[x][i])
	{
		dfs(tr[x][i]);bo=1;
		f[x]|=!f[tr[x][i]];
		g[x]&=g[tr[x][i]];
	}
	if (bo) g[x]^=1;
}
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%d%d\n",&n,&k);
	for (i=1;i<=n;++i) add();
	dfs(0);
	if (f[0] && g[0]) ans=1;
		else if (f[0]) ans=k&1;
			else ans=0;
	if (ans) printf("First\n");
		else printf("Second\n");
}
