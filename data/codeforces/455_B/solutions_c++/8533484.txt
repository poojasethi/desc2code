#include <cstdio>
#include <cmath>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <string>
using namespace std;
int ch[200000][30];
int num,root,ans,n,k;
string s;
void insert(string s)
{
	int l=s.size();
	int dq=root;
	for (int i=0;i<l;++i)
	{
		if (!ch[dq][s[i]-'a'])
		{
			++num;
			ch[dq][s[i]-'a']=num;
		}
		dq=ch[dq][s[i]-'a'];
	}
}
int dfs(int x)
{
	int flag=1,ans=0;
	for (int i=0;i<26;++i)
		if (ch[x][i])
		{
			flag=0;
			ans|=dfs(ch[x][i])^3;
		}
	if (flag) ans=1;
	return ans;
}
int main()
{
	num=root=0;
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;++i)
	{
		cin>>s;
		insert(s);
	}
	int ans=dfs(0);
	if (ans==3)
	{
		printf("First");
		return 0;
	}
	if (ans==2)
	{
		if (k&1) printf("First");else printf("Second");
		return 0;
	}
	printf("Second");
	return 0;
}
