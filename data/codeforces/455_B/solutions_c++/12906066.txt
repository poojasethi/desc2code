#include <bits/stdc++.h>
#define ll long long
#define mst(a,x) memset(a,x,sizeof(a))
using namespace std;

const int N = 200005;
int n, k, nxt[N][35], tot, ans, dp[N];
char s[N];
bool Lose[N];

int dfs(int x)
{
	int cnt = 0;

	dp[x] = 0;
	Lose[x] = 0;
	for(int i = 0; i < 26; i++)
	{
		if(nxt[x][i] == -1) continue;
		cnt++;
		if(!dfs(nxt[x][i])) dp[x] = 1;
		if(!Lose[nxt[x][i]]) Lose[x] = 1;
	}
	if(!cnt) Lose[x] = 1;
	return dp[x];
}

int main()
{
	int j;

	scanf("%d%d", &n, &k);
	mst(nxt, -1);
	tot = 1;
	for(int t = 0; t < n; t++)
	{
		scanf("%s", s);
		j = 0;
		for(int i = 0; i < strlen(s); i++)
		{
			if(nxt[j][s[i] - 'a'] == -1) 
				nxt[j][s[i] - 'a'] = tot++;
			j = nxt[j][s[i] - 'a'];
		}
	}

	ans = dfs(0);
	if(ans == 1)
	{
		if(Lose[0] || (k & 1)) puts("First");
		else puts("Second");
	}
	else puts("Second");
	return 0;
}
