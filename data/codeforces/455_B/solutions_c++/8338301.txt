#include <bits/stdc++.h>
using namespace std;
int n,k;
int d[1000005][30],c[1000005];
int wn[1000005],ls[1000005],say=1;
int main()
{
	scanf("%d %d",&n,&k);
	for (int i = 0; i < n; i++)
	{
		string a;
		cin >> a;
		int tmp=0;
		for (int j = 0; j <(int) a.size(); j++)
		{
			if(d[tmp][a[j]-'a'])
				tmp=d[tmp][a[j]-'a'];
			else
			{
				c[tmp]=1;
				d[tmp][a[j]-'a'] = say;
				tmp=say;
				say++;
			}
		}
	}
	for (int i = say; i >=0 ; i--)
	{
		if(c[i] == 0)
		{
			wn[i]=0;
			ls[i]=1;
			continue;
		}
		for (int j = 0; j <= 26; j++)
		{
			if(d[i][j])
			{
				wn[i]|=!wn[d[i][j]];
				ls[i]|=!ls[d[i][j]];
			}
		}
	}
	int w=wn[0],l=ls[0];	
	if(!w) puts("Second");
	else if(w & l) puts("First");
	else if(k & 1) puts("First");
	else puts("Second");
}
