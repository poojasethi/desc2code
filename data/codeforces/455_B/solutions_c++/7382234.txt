#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
#include<iostream>
using namespace std;
struct node{
	int first;
	bool dp[2];// first win first lost
}N[2700100];
int ln=27;
char s[101000];
void ins(){
	int now=0;
	for(int i=0;s[i];++i){
		now=N[now].first+s[i]-'a';
		if(!N[now].first) N[now].first=ln,ln+=26;
	}
}
int n,m;
int main(){
	scanf("%d%d",&n,&m);
	N[0].first=1;
	for(int i=0;i<n;++i){
		scanf("%s",s);
		ins();
	}
	for(int i=ln;i>=0;--i){
		bool flag=false;
		for(int j=0;j<26;++j) if(N[N[i].first+j].first){
			flag=true;
			N[i].dp[0]|=!N[N[i].first+j].dp[0];
			N[i].dp[1]|=!N[N[i].first+j].dp[1];
		}
		if(!flag) N[i].dp[0]=0,N[i].dp[1]=1;
	}
	if(!N[0].dp[0]) printf("Second\n");
	else{
		if(N[0].dp[1]) printf("First\n");
		else printf("%s\n",m&1?"First":"Second");
	}
	return 0;
}
