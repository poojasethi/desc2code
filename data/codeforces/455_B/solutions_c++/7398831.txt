#include <bits/stdc++.h>
using namespace std;
int k,n;
string str;
int ch[200005][26];
int cnt=2;
int D(int N){
    int ans=0,flag=0;
    for(int i=0;i<26;i++)ch[N][i] && (flag=1,ans|=D(ch[N][i])^3);
    flag || (ans=1);
    return ans;
}
int main(){
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++){
        cin>>str;
        int N=1;
        for(int i=0;i<str.length();i++){
            if (!ch[N][str[i]-97]){
                ch[N][str[i]-97]=cnt++;
            }
            N=ch[N][str[i]-97];
        }
    }
    int h = D(1);
    if (h==3) puts("First");
    else if (h<=1) puts("Second");
    else puts(k&1?"First":"Second");
    return 0;
}
