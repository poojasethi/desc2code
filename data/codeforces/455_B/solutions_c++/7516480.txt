#include <cstdio>
#include <cstring>

using namespace std;

const int MaxN = 100010;

struct Node{
	int win,lose;
	Node *nxt[26];
}pool[MaxN << 1],*tail=pool,*rt;

int n,k;
char buf[MaxN];

void insert(char *buf,Node *(&rt)){
	if(!rt) rt=tail++;
	if(*buf) insert(buf+1,rt->nxt[*buf-'a']);
}

void calc(Node *rt){
	int cnt=0;
	for(int i=0;i<26;i++) if(rt->nxt[i]){
		cnt=1;calc(rt->nxt[i]);
		rt->win|=!rt->nxt[i]->win;
		rt->lose|=!rt->nxt[i]->lose;
	}
	if(!cnt) rt->lose=1;
}

int main()
{
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++) scanf("%s",buf),insert(buf,rt);
	calc(rt);
	if(rt->win && rt->lose) puts("First");
	else if(rt->lose) puts("Second");
	else if(rt->win) puts(k&1 ? "First" : "Second");
	else puts("Second");
	return 0;
}
