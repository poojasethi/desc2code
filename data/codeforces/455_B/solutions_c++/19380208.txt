#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define N 110000

int a[N][30],fw[N],fl[N];
int n,k,sum=1;
char ch;

int main(){
	scanf("%d %d",&n,&k);
	for (int i=1;i<=n;i++){
		int now = 1 ;
		char ch;
		for (ch=getchar();ch < 'a' || ch > 'z';ch=getchar());
		for (;ch >='a' && ch<='z';ch=getchar()) {
			int x = ch	- 'a' + 1;
			if (! a[now][x]) a[now][x] = ++sum;
			now = a[now][x];
		}
	}

	for (int i=sum;i>=1;i--){
		int son = 1;
		for (int j=1;j<=26;j++)
			if (a[i][j]) son = 0;
		if (son) {
			fl[i] = 1;
			fw[i] = 0;
		}
		else {
			for (int j=1;j<=26;j++) if (a[i][j]) {
				if (!fw[a[i][j]]) fw[i] = 1;
				if (!fl[a[i][j]]) fl[i] = 1;
			}	
		}
	}
 	int b;
  	if (!fw[1]) b=0; else 
  	if (fl[1]) b=1; else 
  	if (k%2) b=1; else b=0;

  	if (b) printf("First\n") ; else printf("Second\n");
  
	return 0;
}
