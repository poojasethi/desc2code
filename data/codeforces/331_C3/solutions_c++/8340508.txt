#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define N 21
using namespace std;
long long n,ans,f[N][N][N];
int g[N][N][N],d[N];
void work(int x,int y,long long &a,int &b){
	if(!x&&!y) return;
	if(!y){a=1,b=0; return;}
	int ret=0;
	while(x>=0){x-=max(x,y);ret++;}		
	a=ret;b=10+x;
}
int main(){
	scanf("%I64d",&n);
	for(int j=0;j<=9;j++)
	for(int k=0;k<=9;k++) work(k,j,f[0][j][k],g[0][j][k]);
	for(int i=1;i<=18;i++)
	for(int j=0;j<=9;j++)
	for(int k=0;k<=9;k++){
		int now=k;
		for(int l=9;l>=0;l--) f[i][j][k]+=f[i-1][max(j,l)][now],now=g[i-1][max(j,l)][now];
		g[i][j][k]=now;
	}
	for(int i=0;i<=18;i++) d[i]=n%10,n/=10;
	int now=d[0],cnt=0;
	for(int j=18;j;j--) cnt=max(cnt,d[j]);
	ans=f[0][cnt][now];
	now=g[0][cnt][now];
	for(int i=1;i<=18&&d[i]!=-1;i++){
		cnt=0;
		for(int j=18;j>i;j--) cnt=max(cnt,d[j]);
		for(int j=d[i]-1;j>=0;j--){
			ans+=f[i-1][max(j,cnt)][now];
			now=g[i-1][max(j,cnt)][now];
		}
	}
	printf("%I64d\n",ans);
}
