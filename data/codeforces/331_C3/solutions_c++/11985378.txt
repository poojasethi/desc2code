#include<cstdio>
#include<map>
typedef long long ll;
struct P{
	ll x;int y;
	bool operator<(const P&a)const{return x<a.x||x==a.x&&y<a.y;}
};
std::map<P,P>f;
ll pw[20]={1},n;
#define max(a,b) (a>b?a:b)
P calc(int k,P a){
	if(k==1)return a.x>=a.y?(P){1,0}:(P){0,a.x};
	if(f.count(a))return f[a];k--;
	P b=(P){0,0},c;ll ret=a.x%pw[k];
	for(int i=a.x/pw[k];i;i--){
		c=calc(k,(P){ret,max(i,a.y)});
		b.x+=c.x+1;ret=pw[k]+c.y-max(i,a.y);
	}
	return c=calc(k,(P){ret,a.y}),b.x+=c.x,b.y=c.y,f[a]=b;
}
int main(){
	scanf("%I64d",&n);
	for(int i=1;i<19;i++)pw[i]=pw[i-1]*10;
	int len=0;for(long long x=n;x;x/=10)len++;
	printf("%I64d\n",n==0?0:calc(len,(P){n,0}).x);
}
