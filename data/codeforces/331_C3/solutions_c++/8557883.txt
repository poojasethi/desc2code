#define ll long long
using namespace std;

struct node{
	ll sm,num;
	node(ll s=0,ll n=0):sm(s),num(n) {}
	inline int operator <(const node &x)const{
		if(sm!=x.sm)return sm<x.sm;
		return num<x.num;
	}
};

map<node,node>mp;
ll n;

node dfs(node x){
	if(x.num<10) return node(x.sm||x.num,x.num-max(x.sm,x.num));
	if(mp.count(x))return mp[x];
	ll p=1,an=0;while(p<=x.num/10)p*=10;
	node C=dfs(node(max(x.sm,x.num/p),x.num%p));
	an+=C.sm;
	node Ano=node(x.sm,x.num/p*p+C.num);
	C=dfs(Ano);
	return mp[x]=node(an+C.sm,C.num);
}
