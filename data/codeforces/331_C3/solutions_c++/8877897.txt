#include<cstdio>
#include<set>
using namespace std;
struct ty{int c1;long long c2,c3,ans;};
bool operator < (ty s1,ty s2)
{
    if(s1.c1==s2.c1)
        return s1.c2<s2.c2;
    return s1.c1<s2.c1;
}
int max(int x,int y)
{
    if(x<y)
        return y;
    return x;
}
multiset <ty> k;
ty dfs(int maxx,long long val)
{
    ty ret;
    ret.c1=maxx;
    ret.c2=val;
    ret.ans=0;
    if(val==0)
    {
        ret.c3=maxx;
        ret.ans=1;
        return ret;
    }
    if(k.find(ret)!=k.end())
        return *k.find(ret);
    long long s;
    ty tmp;
    for(s=1;(double)s*10<=val;s*=10);
    while(val>0)
    {
        tmp=dfs(max(maxx,val/s),val%s);
        val=val-tmp.c3;
        ret.ans+=tmp.ans;
    }
    ret.c3=ret.c2-val;
    k.insert(ret);
    return ret;
}
int main()
{
    long long n;
    scanf("%I64d",&n);
    if(n)
        printf("%I64d\n",dfs(0,n).ans);
    else
        puts("0");
    return 0;
}
