#include <cstdio>
#include <algorithm>
#include <map>
#include <cstring>
using namespace std;
#define rep(i,a,n) for (int i=a;i<n;i++)
#define mp make_pair
#define fi first
#define se second
typedef long long ll;
typedef pair<ll,int> pll;
map<pll,pll> dp;
ll n;
pll solve(ll n,int mx) {
	if (n==0) return mp(0,0); else if (n<10) return mp(1,max(mx-n,0ll));
	pll s=mp(n,mx),res=mp(0,0);
	if (dp.count(s)) return dp[s];
	ll pw=1;
	while (n/10>=pw) pw*=10;
	while (1) {
		int d=n/pw,nx=max(mx,d);
		pll re=solve(n%pw,nx);
		n-=n%pw;res.fi+=re.fi;
		if (re.se) n-=re.se; else n-=nx,res.fi+=!!nx;
		if (!d) break;
	}
	res.se=-n;
	return dp[s]=res;
}
int main() {
	scanf("%I64d",&n);
	printf("%I64d\n",solve(n,0).first);
}
