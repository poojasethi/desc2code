#include<map>
#include<cstdio>
#include<algorithm>
#define LL long long
using namespace std;
typedef pair<LL,int> Pair;
LL d[19];
map<LL,Pair> S;
Pair Dfs(LL n,int x,int z)
{
	LL t=(n*18+x)*10+z;
	if(!x)
		return n?Pair(1,n>z?0:n-z):Pair(0,0);
	if(S.count(t))
		return S[t];
	LL ans=0;
	while(n>0)
	{
		int w=n/d[x];
		Pair t1=Dfs(n%d[x],x-1,max(w,z)); 
		ans+=t1.first;
		if(w&&!t1.second)
			n=w*d[x]-max(z,w),ans++;
		else
			n=w*d[x]+t1.second;
	}
	return S[t]=Pair(ans,n);
}
int main()
{
	LL n;
	scanf("%I64d",&n);
	d[0]=1;
	for(int i=1;i<19;i++)
		d[i]=d[i-1]*10;
	printf("%I64d\n",Dfs(n,18,0).first);
}
