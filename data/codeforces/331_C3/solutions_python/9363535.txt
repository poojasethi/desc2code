#include <cstdio>
#include <algorithm>
#include <map>
#define x first
#define y second
using namespace std;
typedef long long ll;
typedef pair<ll,int> pli;
typedef map<pli,pli> F;
F f;
pli dp(ll x,int y)
{
	if (!x) return pli(0,0);
	if (x < 10) return pli(1LL,(int)max(y - x,0ll));
	pli now = pli(x,y);
	if (f.count(now)) return f[now];
	ll tmp = 1;
	pli res(0,0);
	for (; tmp <= x / 10; tmp *= 10);
	bool z = ((10 - x % 10) + (unsigned long long)x == tmp * 10ull);
	while (x > 0) {
		int bit = max((ll)y,x / tmp);
		pli g = dp(x % tmp,bit);
		res.x += g.x;
		x -= x % tmp + g.y;
		if(!g.y && x) x -= bit,res.x++;
	}
	res.y = -x;
	if (z) f[now] = res;
	return res;
}
