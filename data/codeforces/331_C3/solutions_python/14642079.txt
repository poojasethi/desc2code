#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<ll, int> pr;
bool vis[65][10][10];
pr f[65][10][10];
int c[65], top;
pr dfs(int d, int mx, int rem, bool line) {
	if(d==0) {
		if(mx==0) {
			return pr(0, rem);
		}
		if(rem>0) {
			return pr(0, rem-1);
		}
		return pr(1, mx-1);
	}
	if(!line && vis[d][mx][rem]) {
		return f[d][mx][rem];
	}
	pr ans(0, rem);
	for(int g=line?c[d]:9, i=g; i>=0; --i) {
		pr t=dfs(d-1, max(mx, i), ans.second, line && i==g);
		ans.first+=t.first;
		ans.second=t.second;
	}
	if(!line) {
		vis[d][mx][rem]=1;
		f[d][mx][rem]=ans;
	}
	return ans;
}
ll cal(ll n) {
	for(top=0; n; c[++top]=n%10, n/=10);
	return dfs(top, 0, 0, 1).first;
}
int main() {
	ll n;
	cin >> n;
	cout << cal(n);
	return 0;
}
