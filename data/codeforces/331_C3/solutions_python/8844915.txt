#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
#include<vector>
#include<string>
#include<set>
#include<map>
#define LL long long
#define fi first
#define se second
#define MP make_pair
#define PLI pair<LL,int>
using namespace std;
LL n;
map<PLI,PLI> A;
PLI Solve(LL x,int y)
{
	PLI now=MP(x,y),ans;
	if (A.count(now)) return A[now];
	if (x<10) return A[now]=MP(x||y,x-max((int)x,y));
	LL tmp=x/10,t=1;for(;tmp;tmp/=10,t*=10);
	PLI t1=Solve(x%t,max((LL)y,x/t));
	PLI t2=Solve(x-x%t+t1.se,y);
	return A[now]=MP(t1.fi+t2.fi,t2.se);
}
int main()
{
	//freopen("a.txt","r",stdin);
	scanf("%I64d",&n);
	printf("%I64d\n",Solve(n,0).fi);
	return 0;
}
