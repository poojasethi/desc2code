#include <bits/stdc++.h>
#define LL long long
using namespace std;
LL n;
map <LL, LL> f[10], g[10];
LL mx(LL a)
{
	LL mt = 0;
	while (a)
	{
		mt = max(a % 10, mt);
		a /= 10;
	}
	return mt;
}
void dfs(LL a, LL b)
{
	if (f[a][b] != 0) return;
	LL bn = b;
	while (bn > 0)
	{
		LL p = 1;
		while ((p * 10 <= bn) && (p < 1000000000000000000LL)) p *= 10;
		LL ap = max(a, bn / p), bp = bn % p;
		dfs(ap, bp);
		bn = bn - bp + g[ap][bp];
		f[a][b] += f[ap][bp];
	}
	g[a][b] = bn;
}
int main()
{
	for (LL i = 0; i <= 9; ++ i)
		for (LL j = 0; j <= 9; ++ j)
			f[i][j] = 1, g[i][j] = j - max(i, j);
	f[0][0] = 0;
	cin >> n;
	dfs(0, n);
	cout << f[0][n];
}
