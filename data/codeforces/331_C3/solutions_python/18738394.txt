#include <cstdio>
#include <cstring>
#include <algorithm>
#include <map>
#define fe first
#define se second

using namespace std;

typedef long long LL;
typedef pair<LL,LL> P;

map<P,P> Hash;

P Calc(LL lar,LL n)
{
	if (n < 10) return P(lar || n,n - max(lar,n));
	if (Hash[P(lar,n)].fe) return Hash[P(lar,n)];
	LL l = 1;
	for(;l <= n / 10;l *= 10);
	P bk = Calc(max(lar,n / l),n - (n / l) * l);
	P tmp = Calc(lar,(n / l) * l + bk.se);
	return Hash[P(lar,n)] = P(bk.fe + tmp.fe,tmp.se);	
}

int main()
{
	LL n;
	scanf("%I64d", &n);
	printf("%I64d\n", Calc(0,n).fe);
	return 0;
}
