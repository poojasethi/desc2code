#include <map>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
#define MP make_pair
#define x first
#define y second
typedef long long LL;
typedef pair<LL,int> PR;
const int Maxn = 10000005;
LL n;
map <PR,PR> v;

PR dp(LL x,int y){
  PR cur = MP(x, y);
  if (v.count(cur)) return v[cur];
  if (x<10) return v[cur] = MP(x||y, x-max((int)x, y) );
  LL t=1; for (;t<=x/10;t*=10);
  PR res1 = dp(x%t, max((int)(x/t), y) );
  PR res2 = dp(x-x%t+res1.y, y);
  return v[cur] = MP(res1.x+res2.x, res2.y);
}

int main(){
  scanf("%I64d",&n);
  printf("%I64d\n",dp(n,0).x);
  return 0;
}
