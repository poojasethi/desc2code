#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <map>
#define fi first
#define se second
#define int64 long long
#define mp(x,y) make_pair(x,y)
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
typedef pair<int64,int> Mp;
int i,j,k,m;
int64 n;
map<Mp,Mp> M;
Mp work(Mp x) {
	Mp an=mp(0,0);
	int64 k=1,z=x.fi,y=x.se;
	if (!z) return an;
	if (z<10) {
		an.fi=1;
		if (y>z) an.se=y-z;
		return an;
	}
	if (M.count(x)) return M[x];
	for (;k<=z/10;k*=10);
	for (;z>0;) {
		int A=max(z/k,(int64)y);
		Mp re=work(mp(z%k,A));
		an.fi+=re.fi;
		z-=re.se+z%k;
		if (!re.se) {
			z-=A;
			if (A) an.fi++;
		}
	}
	an.se=-z;
	M[x]=an;
	return an;
}
int main() {
	scanf("%I64d",&n);
	Mp A=work(mp(n,0));
	printf("%I64d\n",A.fi);
	return 0;
}
