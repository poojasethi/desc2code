using namespace std;
typedef long long ll;
const int N = 1e2+3;
int n,m,i,j,k,l,s,t,d,u=-1,ii,jj,x,y,z,h[N][N],hh[N][N][12],dp[N][N][12];
char c;
void f(int i,int j,int t,int d)
{
	if(j>m or j<1 or i>n)return;
	t+=h[i][j];
	if(dp[i][j][t%k]>=t)return ;
	dp[i][j][t%k]=t;
	hh[i][j][t%k]=d;
	f(i+1,j+1,t,1);
	f(i+1,j-1,t,0);
}
int main()
{
	cin>>n>>m>>k;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{
			cin>>c;
			h[i][j]=c-'0';
		}
	k++;
	memset(dp,-1,sizeof dp);
	for(i=1;i<=m;i++)
		f(1,i,0,-1);
	for(i=1;i<=m;i++)
		if(dp[n][i][0]>u)
		{u=dp[n][i][0];x=i;}
	cout<<u<<endl;
	if(u<0)return 0;
	cout<<x<<endl;
	i=x;j=n;t=0;
	while(j>1)
	{
		if(hh[j][i][t]){cout<<'L';t-=h[j][i];t%=k;t+=k;t%=k;i--;j--;}
		else {cout<<"R";t-=h[j][i];t%=k;t+=k;t%=k;i++;j--;}
	}
	endll;
}
