#include <bits/stdc++.h>
#define ll long long

using namespace std;

const int MAXN=100+10,MAXM=10+3;
int dp[MAXN][MAXN][MAXM];
char s[MAXN][MAXN][MAXM];
int a[MAXN][MAXN];

int main(){
	int n,m,k;
	cin>>n>>m>>k;
	k++;
	char c;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			cin>>c,a[i][j]=c-'0';
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			for(int l=0;l<k;l++)
				dp[i][j][l]=-1;
	for(int i=0;i<m;i++)
		dp[n-1][i][a[n-1][i]%k]=a[n-1][i];
	for(int i=n-2;i>-1;i--)
		for(int j=0;j<m;j++)
			for(int l=0;l<k;l++){
				int h=((l-a[i][j])+20*k)%k;
				if(j-1>-1 and dp[i+1][j-1][h]>-1 and dp[i+1][j-1][h]+a[i][j]>dp[i][j][l])
					dp[i][j][l]=dp[i+1][j-1][h]+a[i][j],s[i][j][l]='R';
				if(j+1<m and dp[i+1][j+1][h]>-1 and dp[i+1][j+1][h]+a[i][j]>dp[i][j][l])
					dp[i][j][l]=dp[i+1][j+1][h]+a[i][j],s[i][j][l]='L';
			}
	int ans=-1,ansi;
	for(int i=0;i<m;i++)
		if(dp[0][i][0]>ans)
			ans=dp[0][i][0],ansi=i;
	int x,y=ansi,h=0;
	string u;
	cout<<ans<<endl;
	if(ans!=-1){
		while(x<n-1){
            int f=y;
            u=s[x][y][h]+u;
            if(s[x][y][h]=='L')f++;
            else f--;
            h-=a[x][y];
            h=(h+20*k)%k;
            y=f;
            x++;
        }
	cout<<y+1<<endl<<u;
	}
	
}
