#include<iostream>
#include<cstring>
#include<string>
#include<cstdio>
#include<algorithm>

using namespace std;

const int maxn = 110;
int n,m,K;
char dp[maxn][maxn][9*100+10];
char prec[maxn][maxn][9*100+10];
char s[maxn][maxn];
string res;

int main()
{
	while(scanf("%d%d%d",&n,&m,&K)==3)
	{
		int i,j,k;
		for(i=0;i<n;i++)	scanf("%s",s[i]);
		memset(dp,0,sizeof(dp));
		
		for(i=0;i<m;i++) dp[n-1][i][s[n-1][i] - '0'] = 1;
		
		for(i=n-2;i>=0;i--)
		for(j=0;j<m;j++)
		{
			int tp = s[i][j] - '0';
			int limit = 9 * (n - i);
			for(k = tp;k<=limit;k++)
			//if(dp[i][j][k]==0)
			{
				if(j>0&&dp[i+1][j-1][k - tp])
				{
					dp[i][j][k] = 1;
					prec[i][j][k] ='R';
				}
				else
				if(j+1<m&&dp[i+1][j+1][k-tp])
				{
					dp[i][j][k] = 1;
					prec[i][j][k] ='L';
				}
			}
		}
		
		int ans=-1,limit = n*9/(K+1)*(K+1);
		int jj;
		for(j=0;j<m;j++)
		for(k=limit;k>=0;k-=K+1)
		if(dp[0][j][k])	
		{
			if(ans<k)
				jj=j,ans = k;
			break;
		}
		
		if(ans==-1)
		{
			puts("-1");
			continue;
		}
		
		printf("%d\n",ans);
		i = 0,j=jj,k=ans;
		res ="";
		while(i<n-1)
		{
			char c = prec[i][j][k];
			res += c;
			k -= s[i][j] - '0';
			i++;
			if(c=='L') j++;
			else	j--;	
		}
		
		printf("%d\n",j+1);
		for(i=res.size()-1;i>=0;i--)
		putchar(res[i]);
		putchar('\n');
	}
	return 0;
}
