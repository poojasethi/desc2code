#include<bits/stdc++.h>
using namespace std;
int a[110][110],ans[110][110][15];
pair<int,int> ans2[110][110][15];
int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n,m,k;
    cin>>n>>m>>k;
    k++;
    for(int i=1; i<=n; i++)
        for(int j=1; j<=m; j++)
        {
            char s;
            cin>>s;
            a[i][j]=int(s)-int('0')+k;
        }
    for(int i=1; i<=m; i++)
        ans[1][i][a[1][i]%k]=a[1][i];
    for(int i=2; i<=n; i++)
        for(int j=1; j<=m; j++)
            for(int l=0; l<k; l++)
                if(ans[i-1][j-1][l]+ans[i-1][j+1][l]>0)
                {
                    ans[i][j][(l+a[i][j])%k]=max(ans[i-1][j-1][l],ans[i-1][j+1][l])+a[i][j];
                    if(ans[i-1][j-1][l]>ans[i-1][j+1][l])
                        ans2[i][j][(l+a[i][j])%k]=make_pair(l,j-1);
                    else
                        ans2[i][j][(l+a[i][j])%k]=make_pair(l,j+1);
                }
    int mx=-1,j,l=0;
    for(int i=1; i<=m; i++)
        if(ans[n][i][0]!=0 && ans[n][i][0]>mx)
            mx=ans[n][i][0],j=i;
    if(mx==-1)
    {
        cout<<-1<<'\n';
        return 0;
    }
    cout<<mx-(n*k)<<'\n'<<j<<'\n';
    for(int i=n; i>1; i--)
    {
        int l2=ans2[i][j][l].first,j2=ans2[i][j][l].second;
        if(j>j2)
            cout<<'L';
        else
            cout<<'R';
        l=l2,j=j2;
    }
    cout<<'\n';
    return 0;
}
