#include<bits/stdc++.h>

using namespace std;

#define N 102

int M[N][N][N],m,n,i,j,md,mx,t;

bool H[N][N][N];

char A[N][N];

int f(int x, int y, int k){

    if(x && (!y || y == n + 1))
        return -1000000009;

    if(!x){
        if(k)
            return -1000000009;
        else
            return 0;
    }

    k = (k + A[x][y]) % md;

    int &r = M[x][y][k];

    if(r != -1)
        return r;

    if( f(x-1,y+1,k) > f(x-1,y-1,k) ){
        H[x][y][k] = 1;
        r = f(x-1,y+1,k);
    }
    else
        r = f(x-1,y-1,k);

    return r += A[x][y];
}
int main(){

    cin >> m >> n >> md; md++;

    for(i=1;i<=m;i++)
        for(j=1;j<=n;j++){
            cin >> A[i][j];
            A[i][j] -= '0';
        }

    memset(M , -1 , sizeof M);

    for(j=0,i=1;i<=n;i++)
        if(mx <= f(m,i,0)){
            j = i;
            mx = f(m,i,0);
        }

    if(j){
        cout << mx << endl << j << endl;
        for(i=m;i>1;i--){
            t = (t + A[i][j]) % md;
            if(H[i][j][t]){
                cout << 'R';
                j++;
            }
            else{
                cout << 'L';
                j--;
            }
        }
    }
    else
        cout << -1;
    return 0;
}
