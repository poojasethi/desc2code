#include <cstdio>
#include <cstring>

int ch[110][110]={0};
int dp[110][110][11];
int kk[110][110][11]={0};
char dir[110][110][11]={0};
int main()
{
	memset(dp,-1,sizeof(dp));
	int n,m,k;char cc;
	scanf("%d%d%d",&n,&m,&k);
	++k;
	for(int i=1;i<=n;++i)
	{
		getchar();
		for(int j=1;j<=m;++j)
		{
			scanf("%c",&cc);
			ch[i][j]=cc-'0';
		}
	}
	for(int j=1;j<=m;++j)
		dp[1][j][ch[1][j]%k]=ch[1][j];
	for(int i=2;i<=n;++i)
		for(int j=1;j<=m;++j)
			for(int t=0;t<k;++t)
			{
				int u=(t-ch[i][j]+11*k)%k;
				if(j!=1 && dp[i-1][j-1][u]>=0)
					if(dp[i][j][t] < ch[i][j]+dp[i-1][j-1][u])
					{
						dp[i][j][t]=ch[i][j]+dp[i-1][j-1][u];
						dir[i][j][t]='L';
						kk[i][j][t]=u;
					}
				if(j!=m && dp[i-1][j+1][u]>=0)
					if(dp[i][j][t] < ch[i][j]+dp[i-1][j+1][u])
					{
						dp[i][j][t]=ch[i][j]+dp[i-1][j+1][u];
						dir[i][j][t]='R';
						kk[i][j][t]=u;
					}
			}
	int ans=-1,beg=0;
	for(int j=1;j<=m;++j)
		if(ans<dp[n][j][0])
			ans=dp[n][j][0],beg=j;
	if(!beg)
	{
		puts("-1");
		return 0;
	}
	printf("%d\n%d\n",ans,beg);
	k=0;
	for(int i=n;i>1;--i)
	{
		printf("%c",dir[i][beg][k]);
		int tt=kk[i][beg][k];
		if(dir[i][beg][k]=='L')beg--;
		else beg++;
		k=tt;
	}
	puts("");
	return 0;
}
