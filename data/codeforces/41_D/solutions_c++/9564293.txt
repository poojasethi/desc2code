#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
char a[100][101];
int q,w,y,*s[100][100],*d[100][100];
void l(int e,int r,int t,int i){
  int c;
  r+=i;
  if((0<=r)&&(r<w)){
   c=(t+a[e][r])%y;
   if(s[e][r][c]<t+a[e][r]){
    s[e][r][c]=t+a[e][r];
    d[e][r][c]=i;}}
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int e,r,t,c,v;
  char f[100];
  cin>>q>>w>>y;
  y++;
  for(e=0;e<q;e++){
   cin>>a[e];
   for(r=0;r<w;r++){
    a[e][r]-=48;
    s[e][r]=new int[y];
    d[e][r]=new int[y];
    for(c=0;c<y;c++){
     s[e][r][c]=-1;
     d[e][r][c]=0;}}}
  for(r=0;r<w;r++)
   l(q-1,r,0,0);
  for(e=q-1;e>0;e--)
   for(r=0;r<w;r++)
    for(c=0;c<y;c++)
     if(s[e][r][c]>=0){
      l(e-1,r,s[e][r][c],1);
      l(e-1,r,s[e][r][c],-1);}
  c=0;
  for(r=0;r<w;r++)
   if(s[0][r][0]>s[0][c][0])
    c=r;
  v=s[0][c][0];
  if(v<0){
   cout<<-1;
   return 0;}
  cout<<v<<"\n";
  for(e=0;e<q-1;e++){
   r=v-a[e][c];
   if(d[e][c][v%y]==1)
    f[e]='R';
   else
    f[e]='L';
   c-=d[e][c][v%y];
   v=r;}
  cout<<c+1<<"\n";
  for(e=q-2;e>=0;e--)
   cout<<f[e];
  return 0;}
