#include <iostream>
using namespace std;


bool sum[105][105][905]={false};
int cur[105][105],u;
char way[105][105][905],q,res[105];

int main()
{
	int n,m,k,i,j,p,r,t,max;

	cin>>n>>m>>k;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			cin>>q;
			cur[i][j]=q-'0';
		}
	}
	
	for(i=1;i<=m;i++)
		sum[n][i][cur[n][i]]=true;

	for(i=n-1;i>=1;i--)
	{
		for(j=1;j<=m;j++)
		{
			u=cur[i][j];
			for(p=u;p<=900;p++)
			{
				if(sum[i+1][j-1][p-u])
				{
					sum[i][j][p]=true;
					way[i][j][p]='R';
				}
				else
					if(sum[i+1][j+1][p-u])
					{
						sum[i][j][p]=true;
						way[i][j][p]='L';
					}
			}
		}
	}

	max=-1;
	j=-1;
	for(r=1;r<=m;r++)
	{
		for(t=0;t<=900;t++)
		{
			if(sum[1][r][t] && t%(k+1) == 0 && t > max)
			{
				max=t;
				j=r;
			}
		}
	}

	if(max == -1)
	{
		cout<<max;
		return 0;
	}

	cout<<max<<endl;
	i=1;
	p=max;

	while(i < n)
	{
		res[n-i]=way[i][j][p];

		if(way[i][j][p] == 'L')
		{
			p-=cur[i][j];
			j++;	
		}
		else
		{
			p-=cur[i][j];
			j--;
		}

		i++;
	}

	cout<<j<<endl;

	for(i=1;i<n;i++)
		cout<<res[i];
	return 0;
}
