#include <iostream>
#include <cstring>
#include <cstdio>
#include <cctype>
#include <string>

using namespace std;
int dm[101][101][12];
int p[101][101][12];
string a[102]; 
int main() {
	int n, m, k;
	cin >> n >> m >> k;
	k++;
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}
	memset(dm, -1, sizeof(dm));
	for (int i = 0; i < m; i++) {
		dm[0][i][a[0][i] % k] = a[0][i] - '0';
	}
	for (int i = 0; i + 1 < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			for (int r = 0; r < k; r++)
			{
				if (dm[i][j][r] < 0) continue;
				if (j+1<m) 
				{
					int c = dm[i][j][r] + a[i + 1][j + 1] - '0';
					if (dm[i + 1][j + 1][c % k] < c) {
						dm[i + 1][j + 1][c % k] = c;
						p[i + 1][j + 1][c % k] = 1;
					}
				}
				if (j - 1 >= 0)
				{
					int c = dm[i][j][r] + a[i + 1][j - 1] - '0';
					if (dm[i + 1][j - 1][c % k] < c) 
					{
						dm[i + 1][j - 1][c % k] = c;
						p[i + 1][j - 1][c % k] = 0;
					}
				}
			}
		}
	}
	int res = -1, mj = -1;
	for (int j = 0; j < m; j++) {
		if (dm[n - 1][j][0] > res) {
			res = dm[n - 1][j][0];
			mj = j;
		}
	}
	if (res == -1) {
		cout << res;
		return 0;
	}
	cout << res << endl << mj + 1 << endl;
	for (int i = n - 1, j = mj, r = 0; i; i--) {
		if (p[i][j][r]) {
			r = (r  - (a[i][j] - '0') % k + k) % k;
			printf("L");
			j--;
		} else {
			r = (r  - (a[i][j] - '0') % k + k) % k;
			printf("R");
			j++;
		}
	}
	return 0;
}
