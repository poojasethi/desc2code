#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int dp[105][105][11];
int from[105][105][11];
char a[105][105];

const int INF = 123456789;
int main(void) {
	int n, m, k;
	scanf("%d%d%d", &n, &m, &k);
	for(int i=0;i<=n;i++)
		for(int j=0;j<m;j++)
			for(int w=0;w<=k;w++)
				if(i<n || w>0)
					dp[i][j][w] = -INF;
	for(int i=0;i<n;i++)
		scanf("%s", a[i]);
	++k;
	for(int i=n-1;i>=0;i--)
		for(int j=0;j<m;j++) {
			for(int w=0;w<k;w++) {
				int v = a[i][j]-'0';
				if(j>0 && dp[i][j][(w+v)%k] < dp[i+1][j-1][w]+v) {
					dp[i][j][(w+v)%k] = dp[i+1][j-1][w]+v;
					from[i][j][(w+v)%k] = 1;
				}
				if(j<m-1 && dp[i][j][(w+v)%k] < dp[i+1][j+1][w]+v) {
					dp[i][j][(w+v)%k] = dp[i+1][j+1][w]+v;
					from[i][j][(w+v)%k] = 2;
				}
			}
		}
	/*
	for(int i=0;i<n;i++,printf("\n"))
		for(int j=0;j<n;j++)
			printf("%d ", dp[i][j][0]);*/
	int mx=0, mxbg=0;
	char st[105]={};
	int stn=0;
	for(int i=0;i<m;i++)
		if(dp[0][i][0] > dp[0][mxbg][0])
			mxbg = i;
	mx = dp[0][mxbg][0];
	if(mx>=0) {
		printf("%d\n", mx);
		int j=mxbg, jw=0;
		for(int i=0;i<n-1;i++) {
			int rw=jw;
			jw = ((jw - (a[i][j]-'0')) % k +k)%k;
			if(from[i][j][rw] == 1) { st[stn++] = 'R'; --j; }
			else if(from[i][j][rw] == 2){ st[stn++] = 'L'; ++j; }
		}
		printf("%d\n", j+1);
		for(int i=stn-1;i>=0;--i) printf("%c", st[i]);
		puts("");
	} else
		puts("-1");
	return 0;
}
