#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<cmath>
#include<algorithm>
using namespace std;
int f[105][105][11];
char route[105][105][11];
int a[105][105];
int main()
{
	int n,m,p;
	while(cin>>n>>m>>p)
	{
		int i,j,k;
		p++;
		string s;
		memset(f,-1,sizeof(f));
		for(i=0;i<n;i++)
		{
			cin>>s;
			for(j=0;j<m;j++)
				a[i][j]=s[j]-'0';
		}
		for(i=0;i<m;i++)
			f[n-1][i][a[i][j]%p]=a[n-1][i];
		for(i=n-1;i>=0;i--)
			for(j=0;j<m;j++)
				for(k=0;k<p;k++)
				{
					int tmp;
					if(j>0&&f[i+1][j-1][k]!=-1)
					{
						tmp=f[i+1][j-1][k]+a[i][j];
						if(tmp>f[i][j][tmp%p])
						{
							f[i][j][tmp%p]=tmp;
							route[i][j][tmp%p]='R';
						}
					}
					if(j<m&&f[i+1][j+1][k]!=-1)
					{
						tmp=f[i+1][j+1][k]+a[i][j];
						if(tmp>f[i][j][tmp%p])
						{
							f[i][j][tmp%p]=tmp;
							route[i][j][tmp%p]='L';
						}
					}
				}
		string l="";
		int ans=-1;int pos;
		for(i=0;i<m;i++)
			if(f[0][i][0]>ans)
			{
				ans=f[0][i][0];
				pos=i;
			}
		if(ans>=0)
		{
			
			k=0;
			for(i=0;i<n-1;i++)
			{
				l=route[i][pos][k]+l;int tmp=(k-a[i][pos]%p+p)%p;
				if(route[i][pos][k]=='L')
					pos++;
				else
					pos--;
				k=tmp;
			}
		}
		printf("%d\n",ans);
		if(ans>=0)cout<<pos+1<<endl<<l<<endl;
	}

}
