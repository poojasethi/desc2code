#include<stdio.h>
#include<string.h>
int dp[105][105][12],p[105][105][12];
char g[105][105];
int max(int x,int y)
{
	return x>y?x:y;
}
int main()
{
	int n,m,k,i,j,l,v,pp;
	scanf("%d%d%d",&m,&n,&k);
	k++;
	for(i=0;i<m;i++)scanf("%s",g[i]);
	memset(dp,-1,sizeof(dp));
	for(i=0;i<n;i++)
	{
		v=g[0][i]-'0';
		dp[0][i][v%k]=v;
	}
	for(i=1;i<m;i++)
	{
		for(j=0;j<n;j++)
		{
			v=g[i][j]-'0';
			for(l=0;l<k;l++)
			{
				if(j>0&&dp[i-1][j-1][(l+k-v%k)%k]!=-1&&dp[i][j][l]<dp[i-1][j-1][(l+k-v%k)%k]+v)
				{
					dp[i][j][l]=dp[i-1][j-1][(l+k-v%k)%k]+v;
					p[i][j][l]=0;
				}
				if(dp[i-1][j+1][(l+k-v%k)%k]!=-1&&dp[i][j][l]<dp[i-1][j+1][(l+k-v%k)%k]+v)
				{
					dp[i][j][l]=dp[i-1][j+1][(l+k-v%k)%k]+v;
					p[i][j][l]=1;
				}
			}
		}
	}
	int r=-1,pos;
	for(i=0;i<n;i++)
	{
			if(r<dp[m-1][i][0])
			{
				r=dp[m-1][i][0];
				pos=i;
			}
		
	}
	if(r==-1)
	{
		puts("-1");
		return 0;
	}
	printf("%d\n",r);
	printf("%d\n",pos+1);
	for(pp=0,l=pos,i=m-1;i>0;i--)
	{
		v=g[i][l]-'0';
		
		if(p[i][l][pp])
		{
			l++;
			putchar('R');
		}
		else 
		{
			l--;
			putchar('L');
		}
		pp=(pp+k-v%k)%k;
	}
	return 0;
}
