#include <stdio.h>
#include <string.h>
const int N=128;
int n,m,k,v[N][N][12],r[N][N][12],t=-1,a;
char g[N][N];
main(){
  scanf("%d%d%d",&n,&m,&k),++k;
  for(int i=1;i<=n;++i){scanf("%s",g[i]+1);for(int j=1;j<=m;++j)g[i][j]-='0';}
  memset(v,0xff,sizeof(v));
  for(int i=1;i<=m;++i)v[1][i][g[1][i]%k]=g[1][i];
  for(;;){
    bool f=1;
    for(int i=1;i<n;++i)for(int j=1;j<=m;++j)for(int d=0,u;d<k;++d)if((u=v[i][j][d])>=0){
      int x=u+g[i+1][j-1],y=x%k;
      if(v[i+1][j-1][y]<x)v[i+1][j-1][y]=x,r[i+1][j-1][y]=1,f=0;
      x=u+g[i+1][j+1],y=x%k;
      if(v[i+1][j+1][y]<x)v[i+1][j+1][y]=x,r[i+1][j+1][y]=-1,f=0;
    }
    if(f)break;
  }
  for(int i=1;i<=m;++i)if(t<v[n][i][0])t=v[n][a=i][0];
  printf("%d\n",t);
  if(t>=0){
    printf("%d\n",a);
    for(int i=n+1,j=0,x;i-->2;)putchar(r[i][a][j]>0?'R':'L'),x=((j-g[i][a])%k+k)%k,a+=r[i][a][j],j=x;
    puts("");
  }
}
