#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,m;
    cin>>n>>m;
    string a;
    for(int i=0;i<n;i++)
        cin>>a;
    int last = -1,ans = 0;
    for(int i=0;i<m;i++)
    {
        if(a[i] == 'B')
        {
            if(i == 0)
                ans++;
            else
            {
                if(last != i-1)
                    ans++;
            }
            last = i;
        }
    }
    cout<<ans;
}
