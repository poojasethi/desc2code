#include<bits/stdc++.h>
using namespace std;

int n,m,ans;
char a[505][505];

int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            cin>>a[i][j];
    a[n][0]='.';
    for(int i=1;i<=m;i++)
        if(a[n][i]=='B' && a[n][i-1]=='.')
            ++ans;
    cout<<ans<<"\n";
}
