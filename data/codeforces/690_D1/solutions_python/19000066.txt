#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>

#define MAXN 105
using namespace std;

char wall[MAXN];
int r, c;

int main()
{
    scanf("%d%d", &r, &c);
    for(int i=0;i<r;i++){
        scanf("%s", wall);
    }
    bool b=false;  int cnt=0;
    for(int i=0;i<c;i++){
        if(wall[i]=='.') b=false;
        else if(wall[i]!='.'&&!b){
            cnt++;
            b=true;
        }
    }
    printf("%d\n", cnt);
    return 0;
}
