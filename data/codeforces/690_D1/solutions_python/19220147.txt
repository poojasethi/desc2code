#include <bits/stdc++.h>
using namespace std;
int R, C, ans;
char g[100][100];

int main()
{
	cin >> R >> C;
	
	for(int i = 0; i < R; i++) cin >> g[i];
	
	int i = 0;
	
	while(g[R-1][i] == '.') i++;
	while(i < C)
	{
		while(g[R-1][i] == 'B') i++;
		if(i < C)
		ans++;
		while(g[R-1][i] == '.') i++;
	}

	if(g[R-1][C-1] == 'B') ans++;
	cout << ans;
}
