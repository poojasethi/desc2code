#include <bits/stdc++.h>
#define ll long long
#define ln "\n"
using namespace std;

int m,n;
char a[111][111];

int main() {
	#ifdef jfdw
		freopen("data.inp","r",stdin);
		//freopen("data.out","w",stdout);
	#endif
	//-------------------------------------------
	cin >> m >> n;
	for (int i=1; i<=m; i++)
		for (int j=1; j<=n; j++) cin >> a[i][j];

	int ans=0;
	a[m][0]='.';
	for (int i=1; i<=n; i++)
		if (a[m][i]=='B' && a[m][i-1]=='.') ans++;
	cout << ans << ln;
}
