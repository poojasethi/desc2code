#include<bits/stdc++.h>
#define read_fast ios_base::sync_with_stdio( 0 ) , cin.tie( 0 )
using namespace std ;
int main()
{
     read_fast ;
     int R , C ;
     cin >> R >> C ;
     string s ;
     for ( int i = 0 ; i < R ; i ++ ) {
          cin >> s ;
     }
     int ans = 0 ;
     for ( int i = 0 ; i < C ; i ++ )
          if ( s [ i ] == 'B' and ( s [ i - 1 ] == '.' or ! i ) )
               ans ++ ;
     cout << ans ;
     return 0 ;
}
