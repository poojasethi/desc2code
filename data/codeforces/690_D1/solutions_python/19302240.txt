# include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int r,c;
char a[102][102];
bool b[102];

int main()
{
    cin>>r>>c;
    for (int i=0;i<r;i++)
        for (int j=0;j<c;j++)
            cin>>a[i][j] , b[j] |= (a[i][j] == 'B');
    
    int res = 0;
    for (int i=0;i<c;i++)
        if (!b[i])
            if (i > 0 && b[i-1])
                res++;
        
    if (b[c-1]) res++;
    cout<<res<<endl;
    return 0;
}
