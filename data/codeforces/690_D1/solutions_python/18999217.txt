#include <iostream>
using namespace std;
int main ()
{	
int r, c;
cin >> r;
cin >> c;
char temp;

for (int i = 0; i < r - 1; i++)
	for (int j = 0; j < c; j++)
		cin >> temp;

char *arr = new char[c];
int count = 0;
int streak = 0;
for (int i = 0; i < c; i++){
	cin >> arr[i];
	if (arr[i] == 'B') {
		if (streak != 1){
			count++;
			streak = 1;
		}	
	}
	else
		streak = 0;
}
cout << count;
return 0; 
}
