#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
	int r, c, i, count = 0;
	bool inwall = false;
	scanf("%d%d", &r, &c);
	cin.ignore();
	string wall[r];
	for (i=0;i<r;i++) getline(cin, wall[i]);
	for (i=0;i<c;i++)
	{
		if (wall[r-1][i]=='B')
		{
			if (!inwall) count++;
			inwall = true;
		} else inwall = false;
	}
	printf("%d\n", count);
}
