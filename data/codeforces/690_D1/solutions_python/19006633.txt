
#include <iostream>
using namespace std;

int main() {
int n,m,count=0;
	cin>>n;
	cin>>m;
	char str[105][105];
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			cin>>str[i][j];
		}
	}
	for (int i = 0; i < m; i++)
	{
		if (str[n-1][i]=='B')
		{
			

			while(str[n-1][i]=='B')
			{
				i++;		
			}
			count++;
			
		}
	}
cout<<count;
}
