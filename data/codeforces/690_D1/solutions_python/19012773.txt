#include <bits/stdc++.h>

main()
{
    int n,m;
    scanf("%d%d\n",&n,&m);
    char a[n][m];
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++) scanf("%c",&a[i][j]);
        scanf("\n");
    }
    bool wall = 0;
    int c = 0;
    for (int i = 0; i < m; i++)
    {
        if (a[n-1][i] == 'B' && !wall)
        {
            wall = 1;
            c++;
        }
        if (a[n-1][i] == '.') wall = 0;
    }
    printf("%d",c);
}
