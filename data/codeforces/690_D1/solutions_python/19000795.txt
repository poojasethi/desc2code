#include <iostream>
#include <cstdio>
using namespace std;
typedef long long LL;
const int N = 120;
char mymap[N][N];

int main() {
    int R, C;
    cin >> R >> C;
    for (int i = 0; i < R; i++) {
        scanf("%s", mymap[i]);
    }
    int cnt = 0;
    for(int i = 0; i < C; i++) {
        if (mymap[R-1][i] == 'B' && (i == 0 || mymap[R-1][i-1] == '.')) cnt++;
    }
    cout << cnt << endl;
    return 0;
}
