#include <bits/stdc++.h>

using namespace std;

int main()
{
    int R,C;
    char a[110][110];
    cin >>R>>C;
    for (int r=0;r<R;r++){
		for (int c=0;c<C;c++){
			cin >>a[r][c];
		}
    }
    a[R-1][C]='.';
    int m=0;
    for (int i=0;i<C;i++){
		if (a[R-1][i]=='B'){
			while (a[R-1][i]=='B'){
				i++;
			}
			m++;
		}
    }
    cout <<m<<endl;
    return 0;
}
