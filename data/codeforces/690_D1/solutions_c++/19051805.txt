#include<bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    int r,c,i,count=0;
    cin>>r>>c;
    while(r--)
        cin>>s;
    for(i=0;i<c;i++)
    {
        if(i==0)
        {
            if(s[i]=='B')
                count++;
        }
        else {
            if(s[i]=='B' and s[i-1]=='.')
                count++;
        }
    }
    cout<<count;
}
