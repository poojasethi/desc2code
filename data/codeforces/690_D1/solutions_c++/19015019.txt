#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main()
{
  int   c, nw, i, nb, r ; 
  string s;
//  ifstream cin("input.txt"); 
  cin >> r >> c;
  for(i=0; i < r; i++)
    cin >> s;

  nb = 0;
  nw = 1;
  for(i=0; i < c; i++)
  {
    if(s[i] == '.')
      nw = 1;
    else if ( nw==1)
	{
	   nw = 0;
	   nb++;
    }   	    
  }
  cout << nb << endl;
    
//  system("pause");
  return 0;                               
}
