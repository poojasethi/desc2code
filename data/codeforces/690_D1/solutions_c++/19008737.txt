#include<bits/stdc++.h>
using namespace std;
long long A[105][105];
int main()
{
	int n,m,i,j,k=0,t=0;
	cin>>n>>m;
	for(i=1; i<=n; i++)
	{
		for(j=1; j<=m; j++)
		{
			char a; cin>>a; if(a=='.') A[i][j]=0;
			else A[i][j]=1;
		}
	}int san=0;
	for(i=1; i<=max(n,m); i++)
	{
		int t=0;
		for(j=1; j<=max(n,m); j++)
		{
			if(A[j][i]==1) {t=1; break;}
		}
		if(k==1 && t==0) san++; k=t;
	}if(k==1&&t==0) san++;
	cout<<san;
	return 0;
}
