#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<string.h>

using namespace std;

const int V=12345;
char a[123][123];

int main()
{
    int n,m;
    cin>>n>>m;
    for(int i=0;i<n;i++) scanf("%s",a[i]);

    int ans=a[n-1][0]=='B';

    for(int i=1;i<m;i++){
        if(a[n-1][i-1]=='B' || a[n-1][i]!='B') continue;
        ans++;
    }
    cout<<ans<<endl;
    return 0;
}
