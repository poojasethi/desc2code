#include <bits/stdc++.h>

using namespace std;
string a[1001];
int r,h;
int main()
{
    cin>>r>>h;
    getline(cin, a[0]);
    for(int i = 1; i <= r; ++i)
        getline(cin, a[i]);
    bool f = false;
    int cnt = 0;
    for(int i = 0; i < h; ++i)
        if(a[r][i] == 'B' && f == false){
            cnt++;
            f = true;
        }
        else
            if(a[r][i] != 'B')
                f = false;
    cout<<cnt<<endl;
    return 0;
}
