#include <iostream>

using namespace std;

int main()
{
    int r,c,i,j;
    char a[1000][1000];
    cin >> r >> c;
    for (i=1;i<=r;i++)
        for (j=1;j<=c;j++)
        cin >> a[i][j];
    a[r][0]=='.';
    int k=0;
    for (i=1;i<=c;i++)
    {
        if (a[r][i]=='B' && a[r][i-1]!='B')
            k++;
    }
    cout << k << endl;
}
