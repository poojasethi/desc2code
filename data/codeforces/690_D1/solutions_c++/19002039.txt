#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;


int main() {
ll r,c,i,j,count=0,p=0;
char a[105][105];
   cin>>r>>c;
   for(i=0;i<r;i++)
   {
       for(j=0;j<c;j++)
       {
           cin>>a[i][j];
       }
   }
   
   for(i=0;i<c;i++)
   {
       if(a[r-1][i]=='B'&&p==0)
       {
           count++;
           p=1;
       }
       else if(a[r-1][i]!='B')
       p=0;
   }
   cout<<count;
	return 0;
}
