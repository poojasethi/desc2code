#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
char s[110][110];
int main()
{
	int n, m;
	cin >> n >> m;
	for(int i = 0; i < n; i++)
	{
        scanf("%s", s[i]);
	}
	int ans = 0, st = 0;
	for(int i = 0; i < m; i++)
	{
		if(s[n-1][i] == 'B' && i != m-1)
		   st = 1;
		else if(s[n-1][i] == 'B' && i == m-1)
		   ans++;
		else if(s[n-1][i] == '.')
	    {
	    	if(st == 1)
	    	   ans++;
	        st = 0;
	    }
	}
	printf("%d\n", ans);
	return 0;
}
