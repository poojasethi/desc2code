#include <bits/stdc++.h>

using namespace std;

int main(){
    int w,i,j,r,c;
    cin>>r>>c;
    bool b[c];
    memset(b,0,sizeof(b));
    vector<string> v(r);
    for(i=0;i<r;i++)
        cin>>v[i];
    for(i=0;i<r;i++){
        for(j=0;j<c;j++)
            if(v[i][j] == 'B')
                b[j]=1;
    }
    w = 0;
    if(b[0] == 1)
        w++;
    for(i=1;i<c;i++)
        if(b[i]==1 && b[i-1]==0)
            w++;
    cout<<w;
return 0;
}
