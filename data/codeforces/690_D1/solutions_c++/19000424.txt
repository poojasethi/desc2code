#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,m; scanf("%d %d", &n, &m);
	char t[n+2][m+2];
	for(int i=0;i<n;i++) 
		for(int j=0;j<m;j++)
			scanf(" %c", &t[i][j]);
	int ans=0;	
	for(int i=0;i<m;i++){
		if(t[n-1][i] == 'B'){
			ans++;
			while(i<m && t[n-1][i] != '.') i++;
		}
	}
	printf("%d", ans);
	
}
