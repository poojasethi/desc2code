#include <iostream>
#include<cstring>
using namespace std;

int main() {
	int i,j,r,c;
	cin>>r>>c;
	char b[r][c];
	for(i=0;i<r;i++){
		for(j=0;j<c;j++){
		cin>>b[i][j];
	}
}
	int p=0;
	if(b[r-1][0]=='B')
	p=1;
	for(j=1;j<c;j++){
		
		if(b[r-1][j]=='B'&&b[r-1][j-1]=='.')
		p++;
	}
	cout<<p<<"\n";

	return 0;
}
