#include <iostream>
using namespace std;
char ch;
bool a[101];
int main()
{
    int c,r,ans=0,i,j;
    cin>>r>>c;
    cin.get(ch);
    for(i=1;i<=r;i++)
    {
     for(j=1;j<=c;j++)
     {
      cin>>ch;
      if(ch=='B')
        a[j]=true;
     }
     cin.get(ch);
    }
    for(j=1;j<=c;j++)
    {
     i=j;
     if(a[i])
        ans++;
     while(a[i] && j<=c)
      i++;
     j=i;
    }
    cout<<ans<<endl;
     return 0;
}
