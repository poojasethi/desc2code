#include<bits/stdc++.h>
#define For(s,i,n) for(int i=s;i<n;i++)
#define rof(n,i,s) for(int i=n;i>=s;i--)
#define ll long long
#define print(i) setprecision(i)<<fixed<<
using namespace std;

int main()
{
	int n,m;
	cin>>m>>n;
	char a[m][n];
	For(0,i,m)
		For(0,j,n)
			cin>>a[i][j];
	int c=0,i=0;
	while(i<n)
	{
		if(a[m-1][i]=='B')
		{
			c++;
			i++;
			while(a[m-1][i]=='B'){
				i++;
			}
		}
		else
		i++;
	}
	cout<<c;	
   
   
}
