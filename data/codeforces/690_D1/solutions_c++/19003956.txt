#include <iostream>
#include <string>

using namespace std;

int r;
int c = 7;

int countWords(char* str)
{
	int num = 0;
	for(int i = 0; i < c; i++)
		if((str[i] == 'B' && i == 0) || (i != 0 && str[i - 1] != 'B' && str[i] == 'B'))
			num++;
	return num;
}

int main()
{
	//cout << countWords(".BB.B..");

	cin >> r;
	cin >> c;

	char matrix[r][c];
	for(int i = 0; i < r; i++)
		for(int j = 0; j < c; j++)
			cin >> matrix[i][j];
	
	cout << countWords(matrix[r - 1]);

	return 0;
}
