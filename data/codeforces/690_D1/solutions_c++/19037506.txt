#include <bits/stdc++.h>
using namespace std;

int r,c,total;
char wall[100];

int main()
{
    cin>>r>>c;
    for(int i=0;i<r-1;i++)
        for(int j=0;j<c;j++)
        cin>>wall[0];
    cin>>wall[0];
    if(wall[0]=='B')
        total++;
    for(int i=1;i<c;i++)
    {
        cin>>wall[i];
        if(wall[i-1]=='.'&&wall[i]=='B')
            total++;
    }
    cout<<total<<endl;
}
