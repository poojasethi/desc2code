#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

const int N = 110;
char s[N];

int n, m;

int main(){
	while(~scanf("%d %d", &n, &m)){
		for(int i=0; i<n; i++)	scanf("%s", s);
        int ans = 0;
        bool flag = 0;
        for(int i=0; i<m; i++){
			if(s[i]=='B'){
                if(!flag){
					flag = 1;
					ans++;
                }
			} else {
				flag = 0;
			}
        }
        printf("%d\n", ans);
	}
	return 0;
}
