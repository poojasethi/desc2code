#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

vector<int> r[27*27*27*27];
int YX[205], u[27*27*27*27], ucs;

int go(int x) {
	u[x]=ucs;
	for(int i=0;i<(int)r[x].size();i++)
		if(YX[r[x][i]] == -1) {
			YX[r[x][i]] = x;
			return 1;
		}
	for(int i=0;i<(int)r[x].size();i++)
		if(u[YX[r[x][i]]]!=ucs && go(YX[r[x][i]])) {
			YX[r[x][i]] = x;
			return 1;
		}
	return 0;
}

char s[205][15];
int main(void) {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%s", s[i]);
		int L = strlen(s[i]);
		for(int j=0;j<L;j++) {
			int v=s[i][j]-'a'+1;
			r[v].push_back(i);
			for(int k=j+1;k<L;k++) {
				int v2=v+27*(s[i][k]-'a'+1);
				r[v2].push_back(i);
				for(int l=k+1;l<L;l++) {
					int v3=v2+27*27*(s[i][l]-'a'+1);
					r[v3].push_back(i);
					for(int z=l+1;z<L;z++) {
						int v4=v3+27*27*27*(s[i][z]-'a'+1);
						r[v4].push_back(i);
					}
				}
			}
		}
	}
	for(int i=0;i<n;i++) YX[i]=-1;
	for(int i=0;i<27*27*27*27;i++) {
		++ucs;
		go(i);
	}
	for(int i=0;i<n;i++)
		if(YX[i]==-1) {
			puts("-1");
			return 0;
		}
	for(int i=0;i<n;i++) {
		int v=YX[i];
		char w[10];
		int	wn=0;
		while(v>0) {
			w[wn++]=v%27-1+'a';
			v/=27;
		}
		w[wn]=0;
		printf("%s\n", w);
	}
	return 0;
}
