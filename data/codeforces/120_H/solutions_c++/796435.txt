#include<iostream>
#include<cstdio>
#include<string>
#include<cstring>
#include<map>
using namespace std;
int n,n_,ans,I;
string S[300],ANS[300];
map<string,int> Gy;
int Fs[300],Nx[540000],Y[540000],E,Mat[540000],K[540000];

int fd[540000];

void Ins(int x,int y)
{
	Nx[++E]=Fs[x];Fs[x]=E;Y[E]=y;
}

bool find(int X)
{
	for(int v=Fs[X];v;v=Nx[v])if(fd[Y[v]]!=I)
	{
		fd[Y[v]]=I;
		if(Mat[Y[v]]==-1||find(Mat[Y[v]]))
		{
			Mat[Y[v]]=X;
			return 1;
		}
	}
	return 0;
}


int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	cin>>n;
	for(int i=0;i<n;++i)cin>>S[i],Gy[S[i]]=-i-1;
	memset(K,-1,sizeof K);
	for(int i=0,h;i<n;++i)
	{
		int len=S[i].length();
		for(int j=0;j<len;++j)
		{
			if(K[h=S[i][j]-'a'+1]!=i)
			{
				K[h]=i;
				Ins(i,h);
			}
			for(int k=j+1;k<len;++k)
			{
				if(K[h=(S[i][j]-'a'+1)*27+S[i][k]-'a'+1]!=i)
				{
					K[h]=i;
					Ins(i,h);
				}
				for(int l=k+1;l<len;++l)
				{
					if(K[h=((S[i][j]-'a'+1)*27+S[i][k]-'a'+1)*27+S[i][l]-'a'+1]!=i)
					{
						K[h]=i;
						Ins(i,h);
					}
					for(int m=l+1;m<len;++m)
					if(K[h=(((S[i][j]-'a'+1)*27+S[i][k]-'a'+1)*27+S[i][l]-'a'+1)*27+S[i][m]-'a'+1]!=i)
					{
						K[h]=i;
						Ins(i,h);
					}
				}
			}
		}
	}
	memset(Mat,-1,sizeof Mat);
	memset(fd,-1,sizeof fd);
	for(I=0;I<n;++I)
	if(find(I))ans++;
	if(ans!=n)
	{
		cout<<-1<<endl;
		return 0;
	}
	for(int i=27*27*27*27-1;i;--i)
		if(Mat[i]!=-1)
			for(int j=i;j;j/=27)ANS[Mat[i]]=(char)(j%27+'a'-1)+ANS[Mat[i]];
	for(int i=0;i<n;++i)cout<<ANS[i]<<endl;
	return 0;
}
