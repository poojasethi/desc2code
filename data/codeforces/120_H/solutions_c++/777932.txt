#include <iostream>
#include <cstdio>
#include <vector>
#include <cstring>
#include <map>
using namespace std;

#define ll long long
#define pii pair <int, int>
#define pb push_back
#define mp make_pair
#define f first
#define s second
const int N = 205;

map <string, int> ed;
int lm[N], ind, n, use[205], st;
string ss[N];
vector <int> g[N], rm;
vector <string> pp;

bool kun(int v)
{
  use[v] = st;
  for (int i = 0; i < g[v].size(); i++)
  {
    int to = g[v][i];
    if (rm[to] == -1 || use[rm[to]] != st && kun(rm[to]))
    {
      rm[to] = v;
      lm[v] = to;
      return true;
    }
  }
  return false;
}

void get(int l, int p, string reb)
{
  if (l == 0)
  {
    int t = ed[reb];
    if (t == 0)
    {
      t = ed[reb] = pp.size();
      rm.pb(-1);
      pp.pb(reb);
    }
    g[ind].pb(t);
    return;
  }
  if (l > ss[ind].size() - p || p >= ss[ind].size()) return;
  get(l, p + 1, reb);
  get(l - 1, p + 1, reb + ss[ind][p]);
}
int main()
{
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
  cin >> n;
  rm.pb(-1);
  pp.pb("");
  for (int i = 0;  i < n; i++)
  {
    cin >> ss[i];
    ind = i;
    for (int j = 1; j <= min((int)ss[i].size(), 4); j++)
      get(j, 0, "");
  }
  st = 1;
  for (int i = 0; i < n; st++, i++)
    if (!kun(i))
    {
      cout << -1;
      return 0;
    }
  for (int i = 0; i < n; i++) cout << pp[lm[i]] << endl;
  return 0;
}
