#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <string>
#include <map>
using namespace std;

int n;
string s[200];

map<string,int> m;
vector<string> w;
vector<int> next[200];

void go(int id, int len, int last, string r) {
	if (len==4) return;
	for (int i=last+1;i<s[id].size();i++) {
		string r2=r+s[id][i];
		if (m.count(r2)==0) {
			m[r2]=w.size();
			w.push_back(r2);
		}
		next[id].push_back(m[r2]);
		go(id,len+1,i,r2);
	}
}

int prev[100000];
int vis[200];

int match(int id) {
	if (vis[id]) return 0;
	vis[id]=1;
	for (int i=0;i<next[id].size();i++) {
		int id2=next[id][i];
		if (prev[id2]==-1 || match(prev[id2])) {
			prev[id2]=id;
			return 1;
		}
	}
	return 0;
}

int repr[200];

int main() {
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	cin >> n;
	for (int i=0;i<n;i++) {
		cin >> s[i];
		go(i,0,-1,"");
	}
	int st=0;
	memset(prev,-1,sizeof(prev));
	for (int i=0;i<n;i++) {
		memset(vis,0,sizeof(vis));
		st+=match(i);
	}
	if (st<n) printf("-1\n");
	else {
		for (int i=0;i<w.size();i++) if (prev[i]!=-1) {
			repr[prev[i]]=i;
		}
		for (int i=0;i<n;i++) {
			cout << w[repr[i]] << endl;
		}
	}
    return 0;
}
