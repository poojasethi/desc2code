#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
#include<vector>
#include<map>
using namespace std;

#define LEN 4

char s[205][15];
bool b[1000005];
int link[1000005];
int n;
map<string,int> M;
int cnt,CNT;
int end[1000005];
int next[1000005];
int g[1000005];
int ans;
int ANS[1000005];
string MM[1000005];

void addedge(int x,int y)
{
	end[++CNT]=y;
	next[CNT]=g[x];
	g[x]=CNT;
	end[++CNT]=x;
	next[CNT]=g[y];
	g[y]=CNT;
}

void dfs(int x,string S,int now,int rem)
{
	if(rem==-1)
		return;
	if(rem!=LEN)
	{
		int t;
		if((t=M[S])==0)
		{
			M[S]=t=++cnt;
			MM[cnt]=S;
		}
		addedge(x+1,n+t);
	}
	if(now==strlen(s[x]))
		return;
	dfs(x,S+s[x][now],now+1,rem-1);
	dfs(x,S,now+1,rem);
}

bool find(int x)
{
	for(int i=g[x];i;i=next[i])
	{
		int p=end[i];
		if(!b[p])
		{
			b[p]=true;
			if(link[p]==-1||find(link[p]))
			{
				link[p]=x;
				ANS[x]=p;
				return true;
			}
		}
	}
	return false;
}

void match()
{
	memset(link,-1,sizeof(link));
	for(int i=1;i<=n;i++)
	{
		memset(b,0,sizeof(b));
		if(find(i))
			ans++;
	}
}

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%s",s[i]);
	for(int i=0;i<n;i++)
	{
		if(strlen(s[i])>LEN)
			dfs(i,"",0,LEN);
		else
		{
			int t;
			if((t=M[s[i]])==0)
			{
				M[s[i]]=t=++cnt;
				MM[cnt]=s[i];
			}
			addedge(i+1,n+t);
		}
	}
	match();
	if(ans!=n)
		puts("-1");
	else
	{
		for(int i=1;i<=n;i++)
			puts(MM[ANS[i]-n].c_str());
	}
	return 0;
}
