const int MAXN=20000+10;
int n;
string s[MAXN];
int sum=1;
map <string,int> m;
map <int,string> sina;
vector <int> v[MAXN];
int ta[1000000*2+100];
bool mark[MAXN];
int math[MAXN];

bool dfs(int x){
	mark[x]=true;
	for(int i=0;i<v[x].size();i++)
		if(ta[v[x][i]]==0 || (!mark[ta[v[x][i]]] && dfs(ta[v[x][i]]))){
			math[x]=v[x][i];
			ta[v[x][i]]=x;
			return true;
		}
	return false;

}

int main(){
	cin >> n;
	for(int i=1;i<=n;i++)
	       cin >> s[i];
	for(int i=1;i<=n;i++){
		if(s[i].length()<=4){
			if(m[s[i]]==0){
				m[s[i]]=sum;
				sina[sum++]=s[i];
			}
			v[i].push_back(m[s[i]]);
			continue;
		}
		int mm=s[i].length();
		for(int j=0;j<(1<<mm);j++){
			int cnt=0;
			for(int k=0;k<mm;k++)
				if((j>>k)&1)
					cnt++;
			if(cnt > 4 || cnt == 0)
				continue;
			stringstream ss;
			for(int k=0;k<mm;k++){
				if((j>>k)&1)
					ss <<s[i][k];
			}
			string t=ss.str();
		//	cerr << t << endl;

			if(m[t]==0){
				m[t]=sum;
				sina[sum++]=t;
			}
			//				cerr << t << " " << m[t] << " " << sina[sum-1]<< endl;
			v[i].push_back(m[t]);
		}
	}
	for(int i=1;i<=n;i++)
		if(math[i]==0){
			memset(mark,false,sizeof(mark));
			if(!dfs(i)){
				cout << -1 << endl;
				return 0;
			}
		}
	for(int i=1;i<=n;i++)
		cout << sina[math[i]] << endl;
}
