#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;

int n, m, ed, a[205], l[205], L[205], mat[205], qd[200000], qn[200000], v[200000], f[200000], ret;
char p[10], x, s[205][15], S[200000][5];

void dfs(int x){
	if (x > 0){
		p[x] = 0;
		int ok = 0;
		for (int i=1; i<=n; i++){
			a[i] = 0;
			int y = 0;
			for (int j=0; j<l[i] && y<x; j++){
				if (s[i][j] == p[y]) y++;
			}
			if (y == x){
				ok = 1;
				if (l[i]>4 || x>=l[i]) a[i] = 1;
			}
		}
		if (!ok) return;
		m++;
		strcpy(S[m], p);
		for (int i=1; i<=n; i++){
			if (!a[i]) continue;
			qd[++ed] = m;
			qn[ed] = L[i];
			L[i] = ed;
		}
	}
	if (x <= 3){
		for (char c='a'; c<='z'; c++) p[x] = c, dfs(x+1);
	}
}

int dfs2(int x){
	for (int i=L[x]; i; i=qn[i])
		if (!v[qd[i]]){
			v[qd[i]] = 1;
			if (!f[qd[i]] || dfs2(f[qd[i]])){
				f[qd[i]] = x;
				mat[x] = qd[i];
				return 1;
			}
		}
	return 0;
}


int main(){
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

	scanf("%d", &n);
	for (int i=1; i<=n; i++) scanf("%s", s[i]), l[i] = strlen(s[i]);
	dfs(0);

	for (int i=1; i<=m; i++) f[i] = 0;
	ret = 0;
	for (int i=1; i<=n; i++){
		for (int j=1; j<=m; j++) v[j] = 0;
		ret += dfs2(i);
	}
	
	if (ret == n){
		for (int i=1; i<=n; i++) printf("%s\n", S[mat[i]]);
	}
	else printf("-1\n");
	return 0;
}
