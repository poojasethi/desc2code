	scanf("%d",&n);
	if(n <= 10){
		printf("%d\n",ans[n]);
		return 0;
	}
	int k = 1;
	while(n > (1 << (k + 1)) - 2) k++;
	while(k){
		int t = n - (1 << k) + 2;
		if(t % 2 == 0){
			t /= 2;
			printf("%d\n",t - 1);
			return 0;
		}
		k--;
	}
	int ans = n / 2 - 1;
	if((ans + 1) * 2 < n) ans++;
	printf("%d\n",ans);
	return 0;
}
