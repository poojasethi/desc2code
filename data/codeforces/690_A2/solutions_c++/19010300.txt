int f(int x){
	if(x%2) return x/2;
	return f(x/2);
}

int main(){
	cin>>n;
	if(n%2==0) {
		if(n==2) {
		    cout<<0<<endl;
		    return 0;
		}
		int p =1;
		while((1<<p)-2<n) p++;
		p--;
		n-=((1<<p)-2);
		cout<<max((n/2)-1,0)<<endl;
	}else cout<<n/2<<endl;
	return 0;
}
