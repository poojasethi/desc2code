ll pow(ll base, ll exponent)
{
	ll result = 1;
	while(exponent > 0)
	{
		if (exponent % 2 == 1)
			result *= base;
		exponent = exponent >> 1;
		base *= base;
	}
	return result;
}

ll powerWithMod(ll base, ll exponent, ll modulus)
{
	ll result = 1;
	base %= modulus;
	while(exponent > 0)
	{
		if (exponent % 2 == 1)
			result = (result * base) % modulus;
		exponent = exponent >> 1;
		base = (base * base) % modulus;
	}
	return result;
}

bool isPrime(int n)
{
	if(n == 2)
		return true;
	if(n%2 == 0)
		return false;
	int limit = ceil(sqrt(n)) + 1;
	for(int d=3; d <= limit; d += 2)
	{
		if(n%d == 0)
			return false;
	}
	return true;
}

int gc(string a, string b)
{
	int i = 0, j = 0;
	for(i = 0; i < a.size(); i++)
	{
		if(a[i] == b[j] && j == b.size() - 1)
			return i;
		if(a[i] == b[j])
			j++;
	}
	return -1;
}

ll gcdExtended(ll a, ll b, ll *x, ll *y)
{
	if (a == 0)
	{
		*x = 0, *y = 1;
		return b;
	}
	ll x1, y1;
	ll gcd = gcdExtended(b%a, a, &x1, &y1);
	*x = y1 - (b/a) * x1;
	*y = x1;
 
	return gcd;
}

ll modInverse(ll a, ll m)

    return (int)pow(2, floor(log(n) / log(2)));
