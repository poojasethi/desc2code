#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<iostream>
#include<algorithm>

using namespace std;

int n;

int main()
{
	scanf("%d",&n);
	if (n&1) printf("%d\n",n/2);
	else
	{
		int x=(int)floor(log((double)n)/log(2.0));
		printf("%d\n",(n-(1<<x))/2);
	}
	return 0;
}
