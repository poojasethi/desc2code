#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
using namespace std;
int main () 
{
	int n;
	scanf ("%d", &n);
	if(n%2!= 0) printf ("%d\n", n/2);
	else
	{
		for (int i = 1;n >= pow(2,i) ; i++)
		{
			if(n == pow(2,i)) printf("0\n");
			else if(n < pow(2,i+1) && n > pow(2,i))
			{
				int x = n - pow(2,i);
				printf("%d\n", x/2);
			}
		}
	}
}
