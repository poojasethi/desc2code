#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int main(){
	int n;
	while(~scanf("%d", &n)){
		if(n & 1){
			printf("%d\n", n/2);
		} else {
            int p = 1;
            while(p<=n){
				p<<=1;
            }
            n -= (p>>1);
            printf("%d\n", n>>1);
		}
	}
	return 0;
}
