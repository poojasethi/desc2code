_ = int(raw_input())

all_str = [raw_input() for x in range(_)]

str = ''
for x in range(len(all_str[0])):

    all_column = [all_str[y][x] for y in range(_) if all_str[y][x] !='?']

    str += all_column[0] if len(set(all_column))==1 else '?' if len(set(all_column))>1 else 'a'

print str
