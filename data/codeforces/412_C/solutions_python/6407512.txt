#!/usr/bin/env python

import sys
import logging

n = input()

res = []
flag = []

    
for _ in range(n): 
    p = raw_input()
    if not res: 
        res = list(p)
        flag = [False] * len(res)
    else: 
        for i in range(len(p)): 
            if p[i] != '?': 
                if res[i] == '?': 
                    if not flag[i]: 
                        res[i] = p[i]
                    else: 
                        pass
                elif res[i] == p[i]: 
                    pass
                else: 
                    flag[i] = True
                    res[i] = '?'
            else: 
                pass

rtn = list(res)
for i in range(len(res)): 
    if flag[i]: 
        rtn[i] = '?'
    else: 
        rtn[i] = 'x' if res[i] == '?' else res[i]

print ''.join(rtn)
