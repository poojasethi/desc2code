from sys import stdin
def main():
    n = int(raw_input())
    s = [raw_input().strip() for _ in xrange(n)]
    ans = []
    for x in zip(*s):
        t = set(x)
        if '?' in t:
            t.remove('?')
        if len(t) > 1:
            ans.append('?')
        elif len(t) == 1:
            ans.append(t.pop())
        else:
            ans.append('a')
    print ''.join(ans)
main()
