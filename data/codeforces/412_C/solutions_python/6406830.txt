n = input()
x = list(raw_input())
l = len(x)
for i in range(n - 1):
    y = list(raw_input())
    for j in range(l):
        if x[j] == "?" and y[j] != "?":
            x[j] = y[j]
        if x[j] != y[j] and y[j] != "?":
            x[j] = "-"                
print ''.join(x).replace("?", "a").replace("-", "?")
