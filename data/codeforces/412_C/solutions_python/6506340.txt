from sys import stdout

n = int(raw_input())
patterns = [raw_input() for _ in xrange(n)]
for i in xrange(len(patterns[0])):
	character = '_'
	for j in xrange(len(patterns)):
		c = patterns[j][i]
		if c != '?':
			if character != '_' and c != character:
				stdout.write('?')
				break
			character = c
	else:
		if character != '_':
			stdout.write(character)
		else:
			stdout.write('x')		# arbitrary
