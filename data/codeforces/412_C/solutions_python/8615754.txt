n=int(raw_input())
l=[]
x=n
ans=''
while x:
    x-=1
    l.append(raw_input())
chk=cnt=len(l[0])
for i in range(cnt):
    m=''
    for j in range(n):
        if l[j][i]!='?':
            if m=='':
                m=l[j][i]
            elif l[j][i]!=m:
                m+=l[j][i]
    if len(m)==1:
        ans+=m
    elif m=='':
        ans+='a'
    else:
        ans+='?'
print ans
