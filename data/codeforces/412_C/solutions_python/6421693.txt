n=int(raw_input())
l=[]
for i in range(n):
    l.append(raw_input())
pal=""
for i in range(len(l[0])):
    s=set()
    for j in range(n):
        s.add(l[j][i])
    if len(s)==1 or (len(s)==2 and '?' in s):
        ch=s.pop()
        if ch=='?' and s: ch=s.pop()
        if ch=='?':ch='x'
        pal+=ch
    else: pal+='?'
print pal
