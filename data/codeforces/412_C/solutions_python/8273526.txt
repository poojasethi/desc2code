n = int(raw_input())
strings = []

for i in range(n):
    strings.append(str(raw_input()).rstrip())

m = len(strings[0])

result = ['-' for i in range(m)]

for string in strings:
    for i in range(m):
        if string[i] == '?':
            pass
        elif result[i] == '-':
            result[i] = string[i]
        elif string[i] != result[i]:
            result[i] = '?'

for i in range(m):
    if result[i] == '-':
        result[i] = 'x'

print ''.join(result)
