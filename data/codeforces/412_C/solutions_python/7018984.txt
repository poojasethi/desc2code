def list_to_dict(lst):
    dct = {}
    for elt in lst:
        if elt not in dct:
            dct[elt] = 0
        dct[elt] += 1
    return dct

def compute(lines):
    length = len(lines[0])
    characters = [set() for _ in range(length)]
    for line in lines:
        for i in range(length):
            if line[i] == '?': continue
            characters[i].add(line[i])

    out_string = ''
    for char_set in characters:
        if len(char_set) == 0:
            out_string += 'x'
        elif len(char_set) == 1:
            out_string += char_set.pop()
        else:
            out_string += '?'
    return out_string
