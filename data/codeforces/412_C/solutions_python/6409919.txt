n = int(raw_input())
a = list()
k = str()
s = str()
for i in xrange(n):
    a.append(str(raw_input()))
for j in xrange(len(a[0])):
    for i in xrange(len(a)):
        if (i == 0) or ((k == '?') and (a[i][j] not in '?!')):
            k = a[i][j]
        elif ((k not in '?!') and (a[i][j] not in '?!') and (a[i][j] != k)):
            k = '!'
    if k == '?':
        s += 'x'
    elif k == '!':
        s += '?'
    elif k not in '?!':
        s += k
    k = str()
print s
