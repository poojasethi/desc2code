import sys

n = int(raw_input())

p = []

for i in xrange(n):
	newp = [c for c in raw_input()]
	if not p:
		p = newp
	else:
		for i, c in enumerate(newp):
			if p[i] == c or c == '?':
				continue
			elif p[i] == '?':
				p[i] = c
			else:
				p[i] = ' '

for i, c in enumerate(p):
	if c == '?':
		p[i] = 'x'	
	elif c == ' ':
		p[i] = '?'

print "".join(p)
