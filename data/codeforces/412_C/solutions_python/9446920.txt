n = input()
p = []
for i in xrange(n):
	p.append(raw_input())
l = len(p[0])
res = ''
for i in xrange(l):
	r = p[0][i]
	can = True
	for j in xrange(1,n):
		if p[j][i] == '?':
			continue
		elif r == '?':
			r = p[j][i]
		elif r != p[j][i]:
			can = False
			break
	if(can):
		res += r if r != '?' else 'a'
	else:
		res += '?'
print res
