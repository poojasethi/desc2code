n = int(raw_input())
result = list(raw_input())
marked = [False] * len(result)
for _ in range(n-1):
    pattern = raw_input()
    for i, c in enumerate(pattern):
        if result[i] == '?' and not marked[i]:
            result[i] = c
        elif c != '?' and result[i] != c:
            result[i] = '?'
            marked[i] = True
for i,c in enumerate(result):
    if c == '?' and not marked[i]:
        result[i] = 'a'
print ''.join(result)
