n = int(raw_input())

words = []
for _ in range(n):
    words.append(list(raw_input()))
    
ans = []
for i in range(len(words[0])):
    col = []
    for j in range(n):
        col.append(words[j][i])

    chars = set([x for x in col if x != '?'])
    c = None
    if chars:
        if len(chars) == 1:
            c = chars.pop()
        elif len(chars) > 1:
            c = '?'
        else:
            c = 'x'
    else:
        c = 'x'

    ans.append(c)

print ''.join(ans)
