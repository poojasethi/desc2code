#include <iostream>

using namespace std;
char a[100100];
char b[100100];
int main()
{
    int n;
    while(cin>>n)
    {
     for(int i=0;i<=n-1;i++)
     {
      cin>>a;
      for(int j=0;a[j];j++)
      {
       if(a[j]=='?')
        continue;
       if(!b[j])
        b[j]=a[j];
       else if(b[j]!=a[j])
       {
        b[j]='?';
       }
      }
     }
     for(int i=0;a[i];i++)
     {
      if(!b[i])
        b[i]='a';
     }
     cout<<b<<endl;
    }
    return 0;
}
