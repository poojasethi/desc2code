#include <iostream>
#include <vector>
#include <set>
using namespace std;

int main() {
	int n; cin >> n;
	vector<string> s (n);
	for(int i = 0; i < n; ++i) cin >> s[i];
	int m = s[0].size();
	string res (m, '?');
	for(int j = 0; j < m; ++j) {
		set<char> ss;
		for(int i = 0; i < n; ++i)
			if(isalpha(s[i][j])) ss.insert(s[i][j]);
		res[j] = ss.empty() ? 'a' : ss.size() == 1 ? *ss.begin() : '?';
	}
	cout << res;
	return 0;
}
