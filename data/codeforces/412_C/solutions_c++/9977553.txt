#include<iostream>
#include<string>
using namespace std;
int n;
string s[100010];
char ans[100010];
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++) cin>>s[i];
	int l=s[1].length();
	for(int i=0;i<l;i++)
	{
		char tmp='#';
		for(int j=1;j<=n;j++)
			if(s[j][i]!='?')
			{
				if(tmp=='#') tmp=s[j][i];
				else if(s[j][i]!=tmp) tmp='?';
			}
		if(tmp=='#') tmp='a';
		ans[i]=tmp;
	}
	for(int i=0;i<l;i++) cout<<ans[i];
	cout<<endl;
	return 0;
}
