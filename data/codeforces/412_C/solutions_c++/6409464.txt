#include<cstdio>
#include<cstdlib>
#include<string>
#include<iostream>
using namespace std;
int n;
string s[100100];
char c;
int main()
{
	int i,j,len;
	scanf("%d\n",&n);
	for(i=1;i<=n;i++) cin>>s[i]; len=s[1].length();
	for(i=0;i<len;i++)
	{
		c='0';
		for(j=1;j<=n;j++) if(s[j][i]>='a'&&s[j][i]<='z')
		{
			if(c=='0') c=s[j][i]; else if(c!=s[j][i])c='?';
		}
		if(c=='0') c='a'; 
		printf("%c",c);
	}
	return 0;
}
