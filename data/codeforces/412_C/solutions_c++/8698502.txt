#include <iostream>
using namespace std;
int main (){
    int a;
    string s,z;
    cin>>a>>s;
    for(int i =1 ; i < a && cin>>z; i++)
    for(int j =0 ; j < s.length(); j++)
    if(s[j]=='?')s[j]=z[j];
    else
    if(z[j]!='?' && s[j]!=z[j])s[j]='*';
    for (int i=0; i<s.length(); i++)
    if (s[i]=='?') cout<<"x";
    else if (s[i]=='*') cout<<"?";else cout<<s[i];
    return 0;}
