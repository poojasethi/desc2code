#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <set>
using namespace std;
string s[100005];
int main()
{//freopen("i.txt","r",stdin);
int n; cin>>n;
for (int i=1;i<=n;i++) cin>>s[i];
for (int j=0;j<s[1].size();j++)
 {set<char> se;
  for (int i=1;i<=n;i++)
     if (s[i][j]!='?') se.insert(s[i][j]);
  if (se.size()==0) cout<<'x';
  if (se.size()==1) cout<<(*se.begin());
  if (se.size()>1)  cout<<'?';
 }


return 0;
}
