#include <bits/stdc++.h>

using namespace std;

const int maxn = 100000+10;

int main()
{
	int n;
	cin >> n;

	string s;
	int len;

	char ans[maxn];
	memset(ans, '*', sizeof(ans));

	for (int i = 0; i < n; ++i)
	{
		cin >> s;
		len = s.length();
		for (int j = 0; j < len; ++j)
		{
			if (s[j] != '?')
			{
				if (ans[j] == '*')
					ans[j] = s[j];
				else if (ans[j] != s[j])
					ans[j] = '?';
			}
		}
	}

	for (int i = 0; i < len; ++i)
		cout << (ans[i] == '*' ? 'a' : ans[i]);
	cout << endl;

	return 0;
}
