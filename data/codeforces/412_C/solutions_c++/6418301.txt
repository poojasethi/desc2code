#include <iostream>

using namespace std;

int main()
{
    int x,num=0;
    char c='S';
    string res;
    cin>>x;
    string n[x];
    for(int i=0;i<x;i++)cin>>n[i];

    for(int i=0;i<n[0].size();i++){
    c='S';num=0;
        for(int j=0;j<x;j++){
            if(n[j][i]!=c&&n[j][i]!='?'){num++; c=n[j][i];
        }
    }

    if(!num)res+="x";
    else if(num==1)res+=c;
    else res+='?';
    }
    cout<<res;
    return 0;
}
