#include<bits/stdc++.h>
using namespace std;

int n,i,j,k;
char arr[100005],c[100005];
int main(){
	scanf("%d",&n);
	for(i=0;i<100005;i++)
		c[i]='?';
	for(i=0;i<n;i++){
		scanf("%s",arr);
		k=strlen(arr);
		for(j=0;j<k;j++){
			if(arr[j]=='?')
				continue;
			if(c[j]=='?')
				c[j]=arr[j];
			else if(c[j]!=arr[j]&&arr[j]!='?')	
				c[j]='.';
		}
	}
	for(i=0;i<k;i++){
		if(c[i]=='?')
			c[i]='x';
		if(c[i]=='.')
			c[i]='?';
	}
	for(i=0;i<k;i++)
		printf("%c",c[i]);
	return 0;
}
