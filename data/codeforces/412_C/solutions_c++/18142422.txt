#include<stdio.h>
#include<string.h>
char b[100055];
int c[100055];
int main()
{
	int n;
	scanf("%d",&n);
	char a[100555];
	int i,j;
	scanf("%s",a);
	int m=strlen(a);
	for(i=1;i<n;i++)
	{
		scanf("%s",b);
		for(j=0;j<m;j++)
		{
			if(c[j]==1)continue;
			else if(b[j]!=a[j]&&b[j]!='?'&&a[j]!='?')c[j]=1;
			else if(a[j]=='?')a[j]=b[j];
		}
	}
	for(i=0;i<m;i++)
	{
		if(a[i]=='?')a[i]='a';
		else if(c[i]==1)a[i]='?';
		printf("%c",a[i]);
	}
	printf("\n");
}
