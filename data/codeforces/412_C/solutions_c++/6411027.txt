#include <stdio.h>
#include <string>
#include <vector>
#include <iostream>

using namespace std;

int main (void) {
	int n, i, j;
	vector<string> vs;
	string s;
	cin >> n;
	for (i = 0; i < n; i++) {
		cin >> s;
		vs.push_back(s);
	}
	for (j = 0; j < vs[0].length(); j++) {
		char c = (vs[0][j] == '?' ? '!' : vs[0][j]);
		for (i = 1; i < n; i++) {
			if (vs[i][j] == '?')	continue;
			if (c == '!')	c = vs[i][j];
			else if (c != vs[i][j])	c = '?';
		}
		s[j] = (c == '!'?'a':c);
	}
	cout << s << endl;
	return 0;
}
