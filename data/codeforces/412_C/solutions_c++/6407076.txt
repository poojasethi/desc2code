#include <bits/stdc++.h>
#define maxn 100020

using namespace std;

int a;
string ar[maxn];

int main()
{
	scanf("%d",&a);
	for(int i=1;i<=a;i++)
		cin >> ar[i];
	
	int k = ar[1].size();
	
	for(int i=0;i<k;i++) {
		char ch = 0;
		int t = 0;
		for(int j=1;j<=a;j++)
			if( ar[j][i] != '?' ) {
				if( ch && ar[j][i] != ch ) { t = 1; break; }
				else ch = ar[j][i];
			}
		if( t ) {
			printf("?");
		}
		else if( !ch ) printf("a");
		else printf("%c",ch);
	} 
	printf("\n");
	
	
	return 0;
}
