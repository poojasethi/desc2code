#include<iostream>
#include<cstdio>
#include<string>
#include<cstring>
#include<algorithm>
using namespace std;
const int N = 1234567;
char a[N],b[N];
int main()
{
	int n,len;
	cin>>n;
	memset(a, '*', sizeof(a));
	for(int i = 0 ;i < n ;i++){
		cin>>b;
	len = strlen(b);
	for(int j = 0;b[j];j++){
		if(b[j] != '?'){
			if(a[j] == '*')  a[j] = b[j];
			else if(a[j] != b[j])  a[j] = '?';
		}
	}
}
	for (int j = 0; j < len; j++) putchar(a[j] == '*' ? 'a' : a[j]);
	return 0;
}
