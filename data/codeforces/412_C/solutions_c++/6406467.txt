#include<iostream>
using namespace std;
int a[100010];
int main()
{
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	int i,j,n,m;
	string s;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>s;
		m=s.size();
		for(j=0;j<m;j++)
		{
			if(s[j]=='?')continue;
			if(a[j]==0)a[j]=s[j]-'a'+1;
			else if(a[j]!=int(s[j]-'a')+1)a[j]=-1;
		}
	}
	for(i=0;i<m;i++)
	{
		if(a[i]==0)a[i]++;
		if(a[i]==-1)cout<<'?';
		else cout<<char(a[i]+'a'-1);
	}
	cout<<'\n';
}
