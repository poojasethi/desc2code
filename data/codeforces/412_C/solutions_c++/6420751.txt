#include <bits/stdc++.h>
using namespace std;
char ar[1000001];
int main()
{
    int n, f=0, len;
    scanf("%d",&n);
    string s;
    for(int i=0; i<n; i++)
    {
        cin>>s;
        len=s.size();
        for(int i=0; i<s.size(); i++)
        {
            if(s[i]!='?'&&s[i]!=ar[i]&&ar[i]!='\0') ar[i]='?';
            else if(s[i]!='?') ar[i]=s[i], f++;
        }
    }
    for(int i=0; i<len; i++) if(ar[i]=='\0') ar[i]='x';
    ar[len]='\0';
    puts(ar);
    return 0;
}
