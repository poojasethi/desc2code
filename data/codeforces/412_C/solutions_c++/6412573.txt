#include <iostream>

using namespace std;

const int MAXN = 1e5;
string s[MAXN];

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> s[i];
	for (int j = 0; j < s[0].size(); j++) {
		int cnt = 0; char out = 'z';
		for (char c = 'a'; c <= 'z'; c++)
			for (int i = 0; i < n; i++)
				if (s[i][j] == c) {
					cnt++, out = c;
					break;
				}
		if (cnt <= 1)
			cout << out;
		else
			cout << "?";
	}
	cout << endl;
	return 0;
}
