#include <iostream>
#include<algorithm>
#include<string.h>
using namespace std;

int main() 
{  long int n;
char s[100002],t[100002];
cin>>n;
cin>>t;
strcpy(s,t);
for(long int i=1;i<n;i++)
{cin>>t;
for(long int j=0;t[j];j++)
{if(t[j]!='?')
{if(s[j]=='?')
s[j]=t[j];
else if(s[j]!=t[j])
{s[j]='.';
}
}
}
}
for(long int j=0;s[j];j++)
{if(s[j]=='?')
s[j]='x';
if(s[j]=='.')
s[j]='?';
}
cout<<s;

      return 0;
}
