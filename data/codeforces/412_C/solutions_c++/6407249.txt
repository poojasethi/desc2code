#include <iostream>

using namespace std;
int a[100009];
int main(){
    int n;
	cin>>n;
	string s;
	int mx=0;
	for(int i=0;i<100005;i++) a[i]=0;
	for(int i=0;i<n;i++)
	{
		 cin>>s;
		 mx=s.size();
		 for(int j=0;j<mx;j++)
		   if(s[j]!='?') {
		   	  if(a[j]==0 || a[j]==(int)s[j] ) a[j]=s[j];
		   	  else a[j]=-1;
		   } 
	}   
	for(int i=0;i<mx;i++)
	   if(a[i]==0 ) cout<<"a";
	   else if(a[i]==-1) cout<<"?";
	   else cout<<(char)a[i];
	return 0;
}
