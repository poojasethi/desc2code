#include <iostream>
using namespace std;
int main()
{
	int n, i, j, k;
	string s, t;
	cin>>n;
	cin>>s;
	for (i=1; i<n; i++) {
		cin>>t;
		for (j=0; j<s.size(); j++) {
			if (s[j]=='?') s[j]=t[j];
			else if (t[j]!='?' && s[j]!=t[j]) s[j]='_';
		}
	}
	for (i=0; i<s.size(); i++) if (s[i]=='?') s[i]='x';
	else if (s[i]=='_') s[i]='?';
	cout<<s<<endl; 
	return 0;
}
