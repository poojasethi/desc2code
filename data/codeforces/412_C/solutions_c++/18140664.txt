#include<stdio.h>
#include<string.h>
char s[100005],ans[100005];
int n;
int main()
{
	scanf("%d",&n);
	n--;
	scanf("%s",ans);
	int l=strlen(ans);
	while(n--)
	{
		scanf("%s",s);
		for(int i=0;i<l;i++)
		{
			if(s[i]=='?')	continue;
			if(ans[i]=='?')	ans[i]=s[i];
			else if(ans[i]!=s[i])	ans[i]='.';
		}
	}
	for(int i=0;i<l;i++)
		if(ans[i]=='?')	ans[i]='a';
	for(int i=0;i<l;i++)
		if(ans[i]=='.')	ans[i]='?';
	printf("%s",ans);
	return 0;
}
