
#include <iostream>
#include <algorithm>

using namespace std;

char a[100005];

int main()
{
	ios_base::sync_with_stdio(0);

	int n;
	cin>>n;
	string s;
	for (int i=0 ; i<n ; i++)
	{
		cin>>s;
		for (int j=0 ; j<s.size() ; j++)
		{
			if (s[j]!='?')
			{
				if (a[j]==0)
					a[j] = s[j];
				else if (a[j]!=s[j])
					a[j] = '?';
			}
		}
	}
	
	for (int i=0 ; i<s.size() ; i++)
		cout<<(a[i]==0 ? 'x' : a[i]);
	
	return 0;
}
