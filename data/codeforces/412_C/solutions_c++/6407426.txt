#include <bits/stdc++.h>
#define maxn 110000

using namespace std;

char ar[maxn],tmp[maxn];
int N,uz,used[maxn];

int main(){
	scanf(" %d",&N);
	for(int i=1;i<=N;i++){
		scanf(" %s",ar+1);
		uz=strlen(ar+1);
		for(int j=1;j<=uz;j++)
			if(ar[j]=='?')
				continue;
			else if(!tmp[j] || tmp[j]==ar[j])
				tmp[j]=ar[j];
			else
				used[j]=1;
	}
	for(int i=1;i<=uz;i++)
		if(used[i])
			printf("?");
		else if(!tmp[i])
			printf("a");
		else
			printf("%c",tmp[i]);
	printf("\n");
	return 0;
}
