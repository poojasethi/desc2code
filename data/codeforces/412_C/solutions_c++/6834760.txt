#include<iostream>
using namespace std;
int main()
{
	int n,i,j;
	string s,x;
	cin>>n;
	for (i=1; i<=n; i++)
	{
		cin>>s;
		if (i==1) {x=s;continue;}
		for (j=0; j<s.size(); j++)
		{
			if (x[j]=='?' && s[j]!='?') {x[j]=s[j];continue;}
			if (x[j]!='?' && s[j]=='?') continue;
			if (x[j]!=s[j]) x[j]='*';
		}
	}
	for (i=0; i<s.size(); i++)
	{
		if (x[i]=='?') x[i]='a';
		if (x[i]=='*') x[i]='?';
	}
	cout<<x;
return 0;
}
