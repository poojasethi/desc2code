# include <iostream>
# include <cstring>
# define fi cin
# define fo cout
using namespace std;
string S[int(1e5)+5];
int main(void)
{
    int n;
    fi>>n;
    for (int i=0;i<n;++i) fi>>S[i];
    int N=S[0].size();
    for (int i=0;i<N;++i)
    {
        int chr=-1;
        for (int j=0;j<n && chr != -2;++j)
        if (S[j][i]!='?')
        if (chr!=-1 && chr!=S[j][i]) chr=-2;else chr=S[j][i];
        if (chr==-1) fo<<'a';else if (chr==-2) fo<<'?'; else fo<<char(chr);
    }
    return 0;
}
