#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
  string ans="";
  for(int i=1;i<=100011;i++)
    ans+='0';
  int n;
  cin>>n;
  string s;
  for(int i=1;i<=n;i++)
    {
      cin>>s;
      for(int j=0;j<s.length();j++)
	if(s[j]!='?' && ans[j]=='0')
	  ans[j]=s[j];
	else if(s[j]!=ans[j] && s[j]!='?')
	  ans[j]='?';
    }
  for(int i=0;i<s.length();i++)
    if(ans[i]!='0')
      cout<<ans[i];
    else
      cout<<"x";
  cout<<endl;
  return 0;
}
