#include<iostream>
using namespace std;
char a[100001],b[100001];
int main() {
    int n;
    cin>>n>>a;
    for(int i=1;i<n;i++) {
        cin>>b;
        for(int j=0;a[j]!='\0';j++) {
            if(a[j]=='?')a[j]=b[j];
            else if(b[j]=='?');
            else if(a[j]!=b[j])a[j]=-1;
        }
    }
    for(int i=0;a[i]!='\0';i++) {
        if(a[i]=='?')a[i]='a';
        if(a[i]==-1)a[i]='?';
    }
    cout<<a;
    return 0;
}
