#include<stdio.h>
#include<string.h>

int n;
char str[100005],a[100005];

int main()
{
	int i,j,ls;
	scanf("%d",&n);
	memset(a,'\0',sizeof(a));
	for(j=0;j<n;j++){
		scanf("%s",str);
		ls=strlen(str);
		for(i=0;i<ls;i++){
			if((!a[i])&&(str[i]!='?'))a[i]=str[i];
			else if((a[i])&&(str[i]!=a[i])&&(str[i]!='?'))a[i]='?';
		}
	}
	for(i=0;i<ls;i++){
		if(a[i])printf("%c",a[i]);
		else printf("a");
	}
	printf("\n");
	return 0;
}
