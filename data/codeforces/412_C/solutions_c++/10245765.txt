#include <bits/stdc++.h>
using namespace std;
 
 
int main() {
 
 int t;
 vector<string>v;
 cin>>t;
 while(t--)
 {
 
    string s;
    cin>>s;
    v.push_back(s);
 }
 
 for(int i=0;i<v[0].size();i++)
 {
    set<char>st;
     for(int j=0;j<v.size();j++)
     {
        if(v[j][i]!='?')
        st.insert(v[j][i]);
     }
     if(st.size()==1)
     cout<<*st.begin();
     else
     if(st.size()>1)
     cout<<"?";
     else
     cout<<"x";
 }
    cout<<endl;
    return 0;
}
