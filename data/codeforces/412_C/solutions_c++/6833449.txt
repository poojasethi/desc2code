#include<iostream>
#include<string.h>
#include<vector>
using namespace std;
int main()
{
 vector<string> vs;
 string s;
 int n,i,j;
 cin>>n;
 for(i=0;i<n;i++)
 {
  cin>>s;
  vs.push_back(s);
 }
 for(j=0;j<vs[0].length();j++)
 {
  char c = (vs[0][j]=='?' ? '!': vs[0][j]);
  for(i=1;i<n;i++)
  {
   if(vs[i][j]=='?') continue;
   if(c=='!')  c=vs[i][j];
   else if(c!=vs[i][j])  c='?';
  }
  s[j]=(c=='!' ? 'a':c);
 }
 cout<<s;
// system("pause");
 return 0;
}
