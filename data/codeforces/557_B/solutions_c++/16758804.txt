#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int main()
{
    int w,n,a[200005];
    cin>>n>>w;
    for(int i=0;i<n*2;i++)
        cin>>a[i];
    sort(a,a+2*n);
    double s=3/2.0*n*(a[0]*2<a[n]?a[0]*2:a[n]);
    printf("%.7lf",s<w?s:w);
}
