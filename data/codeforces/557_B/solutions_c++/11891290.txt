#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
  int N, arr[200000];
  double W, x;
  scanf("%d %lf", &N, &W);
  for(int i=0; i<N<<1; i++) scanf("%d", &arr[i]);
  sort(arr, arr+(N<<1));
  x = min(min((double)arr[0], (double)arr[N]/2), W/3/N);
  printf("%lf\n", 3*N*x);
  return 0;
}
