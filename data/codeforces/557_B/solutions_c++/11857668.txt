#include <cstdio>
#include <algorithm>

using namespace std;

int N,W, A[200005];

int main() {
	scanf("%d %d", &N, &W);
	double x = W/(3*(double)N);
	for(int i = 0; i < 2*N; ++i)
		scanf("%d", A+i);
	sort(A, A+2*N);
	double sol = min(min(x,(double)A[0]),(double)A[N]/2);
	printf("%lf", sol*3*N);
	return 0;
}
