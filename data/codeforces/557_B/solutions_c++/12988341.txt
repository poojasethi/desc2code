#include <bits/stdc++.h>
using namespace std;

const int MAX_N = 2e5 +7;

#define long long long

int n, w, a[MAX_N];

int main() {
  scanf("%d %d", &n, &w);
  for (int i = 0; i < 2*n; i++) scanf("%d", a+i);
  sort(a, a+2*n);
  double x = min(a[0]/1.0, a[n]/2.0);
  printf("%lf\n", min(x*3*n, w/1.0));
}
