#include<iostream>
#include<algorithm>
#include<cstdio>

using namespace std;

int main(void) {
	double w;
	int n;
	scanf("%d %lf",&n,&w);
	double cups[2*n];
	for(int i=0;i<2*n;i++)
		scanf("%lf",&cups[i]);

	sort(cups,cups+2*n);
	
	printf("%.6lf\n",min(min(2*cups[0],cups[n])*1.5*n,w));
	return 0;
}
