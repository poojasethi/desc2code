#include<iostream>
#include<algorithm>
#include<vector>
#include<string>
#include<set>
using namespace std;
int total, me,has[100005];
set<int>s;
int main()
{
	cin >> total >> me;
	int first=0;
	bool flag = 0;
	for (int i = 1; i <= me;i++)
	{
		char a;
		int b;
		cin >> a >> b;
		if (a == '+')
		{    
			if (flag&&b!=first)
				{
					set<int>::iterator it = s.find(first);
					if (it == s.end())
						has[first] = 1;
				}
			if (!s.empty()||flag&&first!=b)
			{
				has[b] = 1;
				
			}
			else if (!flag)
			{
				first = b; flag = 1;
			}

			s.insert(b);
		}
		else
		{
			set<int>::iterator it = s.find(b);
		
			if (it == s.end())
			{
				has[first] = 1;
				first = b;
				flag = 1;
			}
			
				s.erase(b);
				if (!s.empty())
					has[b] = 1;
			
			
		}
	}
	int ans = 0;
	for (int i = 1; i <= total; i++)
		if (!has[i])
			ans++;
	cout << ans<<endl;
	for (int i = 1; i <= total; i++)
		if (!has[i])
			cout << i << ' ';
	int u = 0;
	return 0;
}
