#include <bits/stdc++.h>
using namespace std;
#define maxn 100005
bool ans[maxn], vis[maxn], online[maxn];
char t[maxn];
int id[maxn], a[maxn];
int main() {
	int n, m, on = 0;
	scanf("%d %d\n", &n, &m);
	for(int i = 1; i <= m; i++) {
		scanf("%c %d\n", &t[i], &id[i]);
	}
	for(int i = 1; i <= m; i++) {
		if(!vis[id[i]] && t[i] == '-') {
			online[id[i]] = 1;
			on++;
		}
		vis[id[i]] = 1;
	}
	for(int i = 1; i <= m; i++) {
		if(t[i] == '+') {
			if(on) ans[id[i]] = 1;
			if(a[id[i]] < i - 1) ans[id[i]] = 1;
			online[id[i]] = 1; on++;
		} else {
			on--;
			a[id[i]] = i; online[id[i]] = 0;
			if(on) ans[id[i]] = 1;
		}
	}
	for(int i = 1; i <= n; i++) if(vis[i] && !online[i] && a[i] < m) ans[i] = 1;
	int cnt = 0;
	for(int i = 1; i <= n; i++) if(!ans[i]) cnt++;
	printf("%d\n", cnt);
	for(int i = 1; i <= n; i++) if(!ans[i]) printf("%d ", i); printf("\n");
	return 0;
}
