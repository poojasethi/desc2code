#include<iostream>
#include<algorithm>
#include<math.h>
#include<cstring>
#include<string>
#include<cstdio>
using namespace std;
const int N=100005;
int n,m,x[N],q,ans,num;
bool f[N],p[N],b[N];
char c[N];
int main()
{
	memset(b,1,sizeof(b));
	cin>>n>>m;
	for(int i=1;i<=m;i++){
		scanf(" %c %d",&c[i],&x[i]);
	}
	for(int i=1;i<=m;i++){
		if(c[i]=='+'){
			f[x[i]]=1;num++;
			if(i!=1 && x[i-1]!=x[i])b[x[i]]=0;
			else q=x[i];
		}
		if(c[i]=='-'){
			if(!f[x[i]])b[q]=0;
			else num--;
			if(num>0)b[x[i]]=0;
			//if(i==m && num>1 || (!f[x[i]] && num))b[x[i]]=0;
			f[x[i]]=0;
			if(x[i+1]!=x[i] && i!=m)b[x[i]]=0;
		}
	}
	for(int i=1;i<=n;i++)
		if(b[i])ans++;
	cout<<ans<<endl;
	for(int i=1;i<=n;i++){
		if(b[i])printf("%d ",i);
	}
	if(ans)puts("");

    //system("pause");
    return 0;
}
