#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>

using namespace std;

struct event
{
	int type,num;
}e[100010];

int n,m,sum,first[100010];
bool ok[100010],online[100010];

int main()
{
	memset(ok,true,sizeof(ok));
	memset(online,true,sizeof(online));
	scanf("%d%d",&n,&m);
	char s[2]; 
	for (int i=1;i<=m;i++)
	{
		scanf("%s",s);
		if (s[0]=='+') e[i].type=1;
		else e[i].type=2;
		scanf("%d",&e[i].num);
	}
	sum=0;
	for (int i=1;i<=m;i++)
	{
		if (!first[e[i].num])
		{
			first[e[i].num]=i;
			if (e[i].type==2) sum++; 
		}
	}
	for (int i=1;i<=m;i++)
	{
		if (e[i].type==1)
		{
			if (i>1&&e[i-1].num!=e[i].num) ok[e[i].num]=false;
			if (sum>0) ok[e[i].num]=false;
			sum++;
		}
		else
		{
			if (i<m&&e[i+1].num!=e[i].num) ok[e[i].num]=false;
			if (--sum>0) ok[e[i].num]=false;
		}
	}
	int ans=0;
	for (int i=1;i<=n;i++)if (ok[i]) ans++;
	printf("%d\n",ans);
	for (int i=1;i<=n;i++)if (ok[i]) printf("%d ",i);
	//system("pause");
}
