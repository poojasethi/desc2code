#include<stdio.h>
char t[2]={0};
int n,m,a[100005],v[100005]={0},ov[100005]={0},i,rt;
int re[100005]={0},ren=0;
int main()
{
    scanf("%d %d",&n,&m);
    for(i=0,rt=0;i<m;i++)
    {
        scanf("%s %d",t,&a[i]);
        if(t[0]=='-') a[i]=-a[i];
        if(a[i]>0) ov[a[i]]++;
        else if(!v[-a[i]]&&!ov[-a[i]]) {v[-a[i]]=1;rt++;}
    }
    for(i=1;i<=n;i++) if(v[i]||(!v[i]&&!ov[i])) re[i]=1;
    for(i=0;i<m;i++)
    {
        if(rt>1&&a[i]<0) re[-a[i]]=-1;
        else if(rt>0&&a[i]>0) re[a[i]]=-1;
        else if(rt==1&&a[i]<0&&i!=m-1&&a[i+1]!=-a[i]) re[-a[i]]=-1;
        else if(rt==0&&a[i]>0&&i&&a[i-1]!=-a[i]) re[a[i]]=-1;
        if(a[i]<0) {v[-a[i]]--;rt--;}
        else if(a[i]>0) {v[a[i]]++;rt++;}
    }
    for(i=1;i<=n;i++) if(re[i]>=0) ren++;
    printf("%d\n",ren);
    if(ren)
    {
        for(i=1;i<=n&&re[i]<0;i++);printf("%d",i);
        for(i++;i<=n;i++) if(re[i]>=0) printf(" %d",i);
        printf("\n");
    }
    //scanf("%d",&i);
    return 0;
}
