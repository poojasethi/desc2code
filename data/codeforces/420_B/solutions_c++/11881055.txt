#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define n   100005
#define For(i,a,b)  for(int i=a;i<=b;i++)
#define Rep(i,a,b)  for(int i=a;i>=b;i--)

int     N,M,C,G[n],F[n],V[n],D[n],P[n],Ans[n];
char    s[5];

bool    check(int x){
        For(i,1,N)  if  (P[i]==-1)  D[i]=1,C++;
        For(i,1,M)  {
            D[F[i]]+=G[i];  C+=(D[F[i]]?1:-1);
            if  (C&&!D[x])  return  0;
        }   return  1;
}

int     main(){
        scanf("%d%d",&N,&M);
        For(i,1,M)  scanf("%s%d",s,&F[i]),G[i]=(s[0]=='+'?1:-1),V[F[i]]=1;
        For(i,1,N)  if  (!V[i]) Ans[++*Ans]=i;

        int Gt=0,YY=0;
        For(i,1,M)  if  (!P[F[i]])  {
            P[F[i]]=G[i];
            if  (G[i]==-1)  Gt=F[i];
            if  (G[i]==1&&!YY)  YY=F[i];
        }
        if  (!Gt)   Gt=YY;

        if  (check(Gt)) Ans[++*Ans]=Gt;
        sort(Ans+1,Ans+*Ans+1); printf("%d\n",*Ans);
        For(i,1,*Ans)   printf("%d ",Ans[i]);
}
