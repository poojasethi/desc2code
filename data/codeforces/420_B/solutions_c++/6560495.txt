#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,sum,K;
int f[N],a[N],s[N],F[N],an[N],kai[N],qi[N];
char p[N][2];
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,m) {
		scanf("%s%d",p[i],&a[i]);
		if (p[i][0]=='+') s[a[i]]++;
		else if (s[a[i]]) s[a[i]]--;
		else F[a[i]]=1;
		an[a[i]]=1;
	}
	For(i,1,n) s[i]=F[i],sum+=s[i];
	For(i,1,m) if (p[i][0]=='+') {
		if (sum) an[a[i]]=2;
		qi[a[i]]=K;
		sum++;
		s[a[i]]++;
	} else {
		K++;
		if (sum>1) an[a[i]]=2;
		kai[a[i]]+=K-qi[a[i]];
		sum--;
		s[a[i]]--;
	}
	if (sum) K++;
	For(i,1,n) if (s[i]) kai[i]+=K-qi[i];
	int res=0;
	For(i,1,n) if (!an[i]||an[i]==1&&kai[i]==K) res++;
	printf("%d\n",res);
	For(i,1,n) if (!an[i]||an[i]==1&&kai[i]==K) printf("%d ",i);
	printf("\n");
	//for(;;);
	return 0;
}
