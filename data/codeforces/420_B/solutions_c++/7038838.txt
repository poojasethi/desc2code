#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<vector>
using namespace std;

int ans[111111],v[111111];
int n,m,a,sum,leader,num;
char s[1];

int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++) {ans[i]=1;v[i]=0;}
    sum=0;leader=0;
    for(int i=1;i<=m;i++){
        scanf("%s%d",s,&a);
        if(*s=='+'){
            if(!leader||(leader==a)) leader=a; else ans[a]=0;
            v[a]=1;
            if(a!=leader&&!v[leader]) ans[leader]=0;
            sum++;
        }else{
            if(!leader) leader=a;
            if(a!=leader&&v[a]==0) {ans[leader]=0;leader=a;}
            if(v[a]) {v[a]=0;sum--;}
            if(sum)ans[a]=0;
        }
    }
    for(int i=1;i<=n;i++)if(ans[i]) num++;
    printf("%d\n",num);
    bool p=true;
    for(int i=1;i<=n;i++)if(ans[i]){if(!p) printf(" %d",i);else {p=false;printf("%d",i);}}
    printf("\n");

    return 0;
}
