#include<cstdio>
#include<cstring>
#include<algorithm>
#include<set>
using namespace std;
set<int> s;
set<int>::iterator it;
char ch;
bool v[110001];
int i,j,k,n,m,a[110001],b[110001],cot;
int main(){
	scanf("%d%d",&n,&m);memset(v,0,sizeof(v));memset(a,0,sizeof(a));
	for (i=1,cot=0;i<=m;i++){
		for (ch=getchar();ch!='+'&&ch!='-';ch=getchar());
		scanf("%d",&b[i]);
		if (!a[b[i]]){
			if (ch=='+') a[b[i]]=-1;else a[b[i]]=1,cot++;
		}
		if (ch=='-') b[i]*=-1;
	}
	if (cot){
		for (i=1;i<=n;i++)
		 if (a[i]==-1) v[i]=1;
	}
	for (i=1;i<=n;i++)
	 if (!v[i]&&a[i]==-1) s.insert(i);
	for (i=1;i<=m;i++){
		if (b[i]>0){
			if (!v[b[i]]) s.erase(b[i]);
			if (++cot==1){
				for (it=s.begin();it!=s.end();it++) v[*it]=1;
				s.clear();
			}
		}else{
			if (--cot) v[-b[i]]=1;
			else if (!v[-b[i]]) s.insert(-b[i]);
		}
	}
	for (i=1,k=0;i<=n;i++)
	 if (!v[i]) k++;
	printf("%d\n",k);
	for (i=1;i<=n;i++)
	 if (!v[i]) printf("%d ",i);
	puts("");
	return 0;
}
