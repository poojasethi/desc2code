#include <iostream>
#include <cstring>
using namespace std;
const int MAX = 100005;
char c[MAX];
int a[MAX];
bool seen[MAX], ans[MAX];
int main()
{
	ios::sync_with_stdio(false);
	memset(ans, true, sizeof(ans));
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < m; i++)
	{
		cin >> c[i] >> a[i];
		a[i]--;
	}
	int cnt = 0;
	for (int i = 0; i < m; i++)
	{
		if (c[i] == '-' && !seen[a[i]])
			cnt++;
		seen[a[i]] = true;
	}
	for (int i = 0; i < m; i++)
		if (c[i] == '+')
		{
			if ((i > 0 && a[i - 1] != a[i]) || cnt)
				ans[a[i]] = false;
			cnt++;
		}
		else
		{
			cnt--;
			if ((i < m - 1 && a[i + 1] != a[i]) || cnt)
				ans[a[i]] = false;
		}
	int size = 0;
	for (int i = 0; i < n; i++)
		if (ans[i])
			size++;
	cout << size << endl;
	for (int i = 0; i < n; i++)
		if (ans[i])
			cout << i + 1 << " ";
	cout << endl;
	return 0;
}
