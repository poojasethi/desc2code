#include<iostream>
#include<cstring>
using namespace std;
#define N 100005
int a[N],n,m;
void slove()
{
    int online=0,sum=0,first,first_flag,i;
    char c;
    for (i=0;i<m;i++)
    {
        int k;
        cin>>c>>k;
        if (i==0){
            first=k;
            first_flag=1;
        }
        if (c=='+'){
            if (k!=first && a[first]==-1) first_flag=0;
            a[k]=1;
            online++;
        }
        else{
            if (a[k]==1) online--;
            if (k!=first && a[k]==0){
                first=k;
                first_flag=1;
            }
            if (k==first && online>0) first_flag=0;
            a[k]=-1;
        }
    }
    if (first_flag) a[first]=0;
    for (i=1;i<=n;i++) if (a[i]==0) sum++;
    cout<<sum<<endl;
    for (i=1;i<=n;i++) if (a[i]==0)
        cout<<i<<" ";
    cout<<endl;
}
int main()
{
    while(cin>>n>>m)
    {
        slove();
        memset(a,0,sizeof(a));
    }
}
