#include <cstdio>

#define N 100000

int n, m, i, k, d, p = -1;
char c;
bool b[N], z[N], f;

int main()
{
    scanf("%d%d", &n, &m);

    while (m--)
    {
        scanf("\n%c %d", &c, &d);
        --d;
        z[d] = 1;
        if (c == '+')
        {
            if (!f)
                p = d;
            else
                if (!k && p != d)
                    p = -1;
            ++k;
            b[d] = 1;
        }
        else
        {
            if (!b[d])
                p = k ? -1 : d;
            else
            {
                --k;
                if (p == d && k)
                    p = -1;
            }
            b[d] = 0;
        }
        f = 1;
    }

    k = ~p ? 1 : 0;

    for (i = 0; i < n; i++)
        if (!z[i])
            ++k;

    printf("%d\n", k);

    for (i = 0; i < n; i++)
        if (!z[i] || i == p)
            printf("%d ", i + 1);

    return 0;
}
