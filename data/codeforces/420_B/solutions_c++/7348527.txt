#include<iostream>
using namespace std;

const int oo = 1e5 + 10;
int n, m, a[oo], cnt, p;
char c[oo];
bool seen[oo], ans[oo];

int main()
{
	cin >> n >> m;
	for (int i = 0; i < m; i++)
		cin >> c[i] >> a[i];
	for (int i = 0; i < m; i++)
	{
		if (c[i] == '-' && !seen[a[i]])
			cnt++;
		seen[a[i]] = true;
	}
	for (int i = 0; i < m; i++)
		if (c[i] == '+')
		{
			if ((i && a[i - 1] != a[i]) || cnt)
				ans[a[i]] = true;
			cnt++;
		}
		else
		{
			cnt--;
			if ((i < m - 1 && a[i + 1] != a[i]) || cnt)
				ans[a[i]] = true;
		}
	for (int i = 1; i <= n; i++)
		if (!ans[i])
			p++;
	cout << p << endl;
	for (int i = 1; i <= n; i++)
		if (!ans[i])
			cout << i << ' ';
	return 0;
}
