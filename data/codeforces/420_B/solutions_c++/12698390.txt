#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 9;

int n, m, id[N], beg[N], end[N], pre[N], sz, hash[N], hash2[N], ans[N];

char ch[N];

int main() {

	cin >> n >> m;


	for (int i = 1; i <= m; i++) {
		cin >> ch[i] >> id[i];
		if (ch[i] == '-' and hash[id[i]] == false) sz++;
		else hash[id[i]] = true;
	}
	memset(hash, false, sizeof hash);
	for (int i = 1; i <= m; i++) {
		if (ch[i] == '+') {
			if (i > 1 and hash[id[i]] == false) ans[id[i]] = true;
			else if (sz > 0) ans[id[i]] = true;
			sz++;
		}
		else {
			sz--;
			if (sz > 0) ans[id[i]] = true;
			else if (i < m and id[i+1] != id[i]) ans[id[i]] = true;
			hash[id[i]] = true;		
		}
	}
	sz = 0;
	for (int i = 1; i <= n; i++)
		if (ans[i] == false)
			sz++;

	cout << sz << endl;
	for (int i = 1; i <= n; i++)
		if (ans[i] == false)
			cout << i << " ";

	return 0;
}
