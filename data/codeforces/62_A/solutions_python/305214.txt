from sys import stdin

in_lines = ""
in_lines = stdin.readlines()

def int_values(line) :
    return map( int, line.split(' ') )

lines = map( int_values, map( lambda x:x.strip(), in_lines  ) )

#print lines
al, ar = lines[0]
bl, br = lines[1]
def good( a, b ) :
    return ( b >= a - 1 ) and ( a >= (b-1) / 2 )

if good( al, br ) or good( ar, bl ) :
    print "YES"
else :
    print "NO"
