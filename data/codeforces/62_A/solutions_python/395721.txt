import sys
import math


lines = [i.rstrip() for i in sys.stdin.readlines()]
girls = [int(i) for i in lines[0].split(" ")]
boys = [int(i) for i in lines[1].split(" ")]

if ((girls[0] - boys[1] < 2) and (boys[1] <= (2 * (girls[0] + 1)))) or ((girls[1] - boys[0] < 2) and (boys[0] <= (2 * (girls[1] + 1)))):
    print "YES"
else:
    print "NO"
