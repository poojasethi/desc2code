#!/usr/bin/env python
# -*- coding: utf-8 -*-

def test(a, b):
    return True if b >= (a-1) and b <= (a+1)*2 else False

def main():
    al, ar = map(int, raw_input().split())
    bl, br = map(int, raw_input().split())
    if (test(al, br) or test(ar, bl)):
        print "YES"
    else:
        print "NO"
    return 0

if __name__ == '__main__':
    main()
