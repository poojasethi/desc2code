def main():
    a = raw_input()
    b = raw_input()
    a = [int(i) for i in a.strip().split()]
    b = [int(i) for i in b.strip().split()]
    
    
    if b[0]+1>=a[1] and 2*a[1]+2>=b[0] or b[1]+1>=a[0] and 2*a[0]+2>=b[1]:
        print 'YES'
    else:
        print 'NO'

if __name__ == '__main__':
    main()
