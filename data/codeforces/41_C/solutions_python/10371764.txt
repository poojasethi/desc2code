inp = [i for i in raw_input()]
n = len(inp)
a = True
i = 1
while(i <n-2):
    if(i!=n-3 and inp[i]=='d' and inp[i+1]=='o' and inp[i+2]=='t'):
        inp = inp[:i]+['.']+inp[i+3:]
        n-=2
    elif(inp[i]=='a' and inp[i+1]=='t'):
        if(a==True):
            inp = inp[:i]+['@']+inp[i+2:]
            a = False
            n-=1
    i+=1
print "".join(inp)
