from math import *
from Queue import *
from sys import *
from random import *






s = raw_input()
n = len(s)
i = 1
red = s[0]
while i < n-3:
    if s[i] == 'd' and s[i+1] == 'o' and s[i+2] == 't':
        red += '.'
        i += 3
    else:
        red += s[i]
        i += 1
red += s[i:]
cand = []
for i in range(1, len(red)-2):
    if red[i] == 'a' and red[i+1] == 't':
        cand.append(red[:i] + '@' + red[i+2:])
cand.sort()
print(cand[0])
