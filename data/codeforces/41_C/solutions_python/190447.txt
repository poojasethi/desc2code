s=raw_input()
e=[]
at=True
for i in range(0,len(s)):
    x=s[i]
    if s[i-2:i+1]=='dot' and i!=len(s)-1 and i!=2:
        e.pop()
        e.pop()
        e.append('.')
    elif s[i-1:i+1]=='at' and at and i!=1 and i!=len(s)-1:
        e.pop()
        e.append('@')
        at=False
    else:
        e.append(x)
z=''
for i in range(0,len(e)):
    z=z+e[i]
print z
