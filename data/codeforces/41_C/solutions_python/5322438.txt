s = raw_input()
if (len(s)) == 0: print '';exit()
ans = s[0]
flag = False
i, n = 1, len(s)
while i < n:
    if i+1 < n-1 and i+2 < n-1 and s[i] == 'd' and s[i+1] == 'o' and s[i+2] == 't':
        ans += '.'; i += 3
    elif i+1 < n and s[i] == 'a' and s[i+1] == 't' and not flag:
        ans += '@'; i += 2; flag=True
    else:
        ans += s[i]; i += 1
print ans
