rs = raw_input()
s = rs[1:-1]
l = len(s)
res = rs[0]
i = 0
findAt = False
while(i < l):
	if(i + 3 <= l):
		if(s[i:i+3] == "dot"):
			res += "."
			i += 3
			continue
	if(i + 2 <= l and (not findAt)):
		if(s[i:i+2] == "at"):
			res += "@"
			findAt = True
			i += 2
			continue
	res += s[i]
	i += 1
res += rs[-1]
print res
