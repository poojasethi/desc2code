import sys
import re

r = re.compile('at')

input = sys.stdin

pre = ''
body = input.readline().strip()
suf = ''

if body.startswith('dot'):
  pre = 'dot'
  body = body[3:]
elif body.startswith('at'):
  pre = 'at'
  body = body[2:]

if body.endswith('dot'):
  suf = 'dot'
  body = body[:len(body)-3]
elif body.endswith('at'):
  suf = 'at'
  body = body[:len(body)-2]

print pre + r.sub('@', body.replace('dot','.'), 1) + suf
