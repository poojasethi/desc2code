s = raw_input()
res = ""
at = False
i = 0
while i<len(s):
    if i!=0 and i+3<len(s) and s[i:i+3]=="dot":
        res += '.'
        i += 2
    elif i!=0 and i+2<len(s) and at == False and s[i:i+2]=="at":
        res += '@'
        at = True
        i += 1
    else:
        res += s[i]
    i += 1
print res
