#include<iostream>

using namespace std;

int main(){
	string s,ans;
	cin >> s;
	ans+=s[0];
	bool ok=0;
	for(int i=1;i<s.length();i++)
		if(i+3 < s.length() && s.substr(i,3) == "dot"){
			ans+=".";
			i+=2;
		}else if(i+2 < s.length() && s.substr(i,2) == "at" && !ok){
			ans+="@";
			i++;
			ok=1;
		}else
			ans+=s[i];
	cout << ans << endl;
	return 0;
}
