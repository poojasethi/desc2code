#include <iostream>
#include <string>

using namespace std;

int main(){
  string s;
  cin >> s;
  bool b = 0;
  for(int i=1;i<s.size()-2;i++){
    if(s[i] == 'a' && s[i+1] == 't' && b==0){
      s.replace(i,2,"@");
      b = 1;
    }
    else{
      if(s[i]=='d' && s[i+1]=='o' && s[i+2]=='t' && (i+3<s.size())) 
      s.replace(i,3,".");
    }
  } 
  cout << s;
  //for(;;);
  
  return 0;
}
