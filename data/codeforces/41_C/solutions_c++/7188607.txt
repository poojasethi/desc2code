#include <iostream>
using namespace std;

int main()
{
  string s,ans;
  cin>>s;
  bool y=false;
  for(int i=0;i<s.length();i++)
    {
      if(i>0 && s[i]=='a' && s[i+1]=='t' && i<s.length()-2 && y==false)
	ans+='@',i++,y=true;
      else if(i>0 && s[i]=='d' && s[i+1]=='o' && s[i+2]=='t' && i<s.length()-3)
	ans+='.',i+=2;
      else
	ans+=s[i];
    }
  cout<<ans<<endl;
  return 0;
}
