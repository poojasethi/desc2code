#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
using namespace std;

string org;
int main(){
	cin>>org;
	int pos;
	while((pos = org.find("dot",1))!=string::npos){
		if(pos!=org.size() - 3){
			org.replace(pos,3,".");
		}else break;
	}
	pos = org.find("at",1);
	org.replace(pos,2,"@");
	cout<<org<<endl;
	return 0;
}
