#include <iostream>
#include <string>
using namespace std;
string s;
int n;
int main()
{
    cin >> s; 
    s.replace((int)s.find("at", 1), 2, "@");
    for(int i = 1; i < (int)s.size() - 3; i++)
        if(s[i] == 'd' && s[i + 1] == 'o' && s[i + 2] == 't')
            s.replace(i, 3, ".");
    cout << s;
    return 0;
}
