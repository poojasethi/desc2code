#include<iostream>
#include<string>
using namespace std;
int main()
{
    string a;
    cin>>a;
    int i=a.find("at",1);
    a[i]='@';
    a.erase(i+1,1);
    i=a.find("dot",1);
    while(i<a.size()-3)
    {
        a[i]='.';
        a.erase(i+1,2);
        i=a.find("dot",1);
    }
    cout<<a;
    return 0;
}
