#include<stdio.h>
#include<string.h>
int a,i;
char d[100005];
int main(){
	scanf("%s",d);

	a=strlen(d);

	for(i=1 ; i<a-2 ; i++)
		if( d[i]=='a' && d[i+1] == 't'){
			d[i]='0';
			d[i+1]='@';
			break;
		}
	for(i=1 ; i<a-3 ; i++)
		if(d[i]=='d' && d[i+1]=='o' && d[i+2] == 't'){
			d[i]=d[i+1]='0';
			d[i+2]='.';
		}

	for(i=0 ; i<a ; i++)
		if(d[i] != '0')
			printf("%c",d[i]);
	return 0;
}
