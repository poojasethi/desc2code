#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    while(cin>>s)
    {
        s.replace(s.find("at",1),2,"@");
        int  t=s.find("dot",1);
        while(t!=s.npos&&t!=s.size()-3)
        {
            s.replace(t,3,".");
            t=s.find("dot",1);
        }
        cout<<s<<endl;
    }
}
