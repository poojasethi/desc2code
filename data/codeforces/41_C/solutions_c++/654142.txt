#include <cstdio>
#include <string>
#include <iostream>

using namespace std;

int main() {

    string S;
    getline(cin, S);
    int p = S.find("at");
    if (p == 0) p = S.find("at", p+1);
    S.replace(p,2,"@");
    for (int p = 0; (p = S.find("dot", p)) != -1; p++) {
        if (p == 0 || p+3 == S.size()) continue;
        S.replace(p,3,".");
    }
    cout << S;

    return 0;
}
