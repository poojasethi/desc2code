#include <iostream>
using namespace std;

int main()
{
	string s;
	cin >> s;
	string t = "";
	bool b = false;
	t += s[0];
	for(int i = 1; i < s.size() ; i++)
	{
		if(s.substr(i , 2) == "at" && ! b)
		{
			t += "@";
			b = true;
			i += 1;
		}
		else if(s.substr(i , 3) == "dot" && i + 3 != s.size())
		{
			t += ".";
			i += 2;
		}
		else
		{
			t += s[i];
		}
	}
	cout << t;
}
