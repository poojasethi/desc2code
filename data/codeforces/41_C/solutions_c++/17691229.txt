#include<bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int n=s.size();
    for(int i=1;i<n-2;i++) if(s[i]=='a'&&s[i+1]=='t'){s[i]='@'; s.erase(i+1,1); n--; break;} 
    for(int i=1;i<n-3;i++)
        if(s[i]=='d'&&s[i+1]=='o'&&s[i+2]=='t') 
        {
            s[i]='.';
            s.erase(i+1,2);
            n-=2;
        }
   cout<<s<<endl;
    return 0;
}
