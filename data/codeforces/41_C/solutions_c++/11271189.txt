#include <iostream>

using namespace std;

int main()
{
    string j;
    int i,a=0;
    cin >> j;
    for(i=0;i<j.length();i++)
    {
        if(!a && i && j.substr(i,2)=="at") {cout << "@"; a=1; i++;}
        else if (i&&i<j.length()-3 && j.substr(i,3)=="dot") {cout << "."; i+=2;}
        else cout << j[i];
    }

    return 0;
}
