#include<iostream>
using namespace std;
int main(){
	string a;
	cin>>a;
	for(int i=1;i<a.length()-2;i++)
		if(a[i]=='a'&&a[i+1]=='t'){
			a[i]='@';
			a[i+1]='@';
			break;
		}
	for(int i=1;i<a.length()-3;i++)
		if(a[i]=='d'&&a[i+1]=='o'&&a[i+2]=='t')
			a[i]=a[i+1]=a[i+2]='.';
	for(int i=0;i<a.length();i++){
		cout<<a[i];
		if(a[i]=='@')
			i++;
		else if(a[i]=='.')
			i+=2;
	}
}
