#include <bits/stdc++.h>
using namespace std;
int i, k;
string S;
int main () {
	cin>>S;
	k = S.find("at", 1);
	if (k<S.length()-2 && k!=-1)
		S.replace(k, 2, "@");
	i = 1;	
	do {
		k = S.find("dot", i);
		if (k!=-1) {
			S.replace(k, 3, ".");
			i++;
		}
	}
	while (k!=-1);
	if (S[S.length()-1]=='.')
		S.replace(S.length()-1, 3, "dot");
	
	cout<<S<<endl;
	return 0;
}
