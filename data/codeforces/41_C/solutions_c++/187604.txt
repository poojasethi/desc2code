#include<iostream>
#include<map>
#include<string.h>
#include<cmath>
using namespace std;
int main(){
	string s,t="";
	cin>>s; int n=s.length(),flag=0;
	t+=s[0];
	for(int i=1;i<n;i++){
		if(i<n-3 && s.substr(i,3)=="dot") { t+='.'; i+=2; }
		else if(i<n-2 && s.substr(i,2)=="at" && flag==0) { t+='@';  flag=1; i+=1; }
		else { t+=s[i]; }
	}
	cout<<t<<endl;
	return 0;
}
