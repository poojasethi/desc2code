#include <stdio.h>

int main(void) {
	char a[105];
	scanf("%s", a);
	printf("%c", a[0]);
	int i, flag=0;
	for(i=1;a[i+1];i++) {
		if(a[i+3] && a[i]=='d' && a[i+1]=='o' && a[i+2]=='t') {
			printf("."); i+=2;
		} else if(a[i+2] && a[i]=='a' && a[i+1]=='t' && flag==0) {
			flag=1;
			printf("@");
			++i;
		} else 
			printf("%c", a[i]);
	}
	printf("%c\n", a[i]);
	return 0;
}
