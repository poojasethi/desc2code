#include<iostream>
using namespace std;
int main()
{
	string a;
	cin>>a;
	bool w=true;
	for(int i=1;i<a.size()-2;i++)
	{
		if(a[i]=='a'&&a[i+1]=='t'&&w)
		{
			w=false;
			a.erase(a.begin()+i);
			a[i]='@';
		}
		if(a[i]=='d'&&a[i+1]=='o'&&a[i+2]=='t'&&i+3!=a.size())
		{
			a.erase(a.begin()+i);
			a.erase(a.begin()+i);
			a[i]='.';
		}
	}
	cout<<a<<endl;
	return 0;
}
