#include <bits/stdc++.h>

using namespace std;

int main() {

  string s;
  cin >> s;

  int idx = s.find("at", 1);
  s = s.substr(0, idx) + "@" + s.substr(idx + 2);
  while((idx = s.find("dot", 1)) != string::npos && idx + 3 < (int)s.size()) {
    s = s.substr(0, idx) + "." + s.substr(idx + 3);
  }
  cout << s << endl;

  return 0;
}
