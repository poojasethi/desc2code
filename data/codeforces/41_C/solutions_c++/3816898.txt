#include <cstdio>
#include <iostream>

using namespace std;

int main(void) {
string s, ff;
cin>>s;
int f2=-1;
for(int i=0;i<s.size();i++){
 if(i!=0&&i!=s.size()-3&&s[i]=='d'&& s[i+1]=='o'&&s[i+2]=='t'&&i+2<s.size()){
   i+=2;
   ff+=".";
 }

 else if(i!=0&&s[i]=='a'&&s[i+1]=='t'&&i+1<s.size()&&f2==-1){
   i+=1;
   ff+="@";
   f2=0;
 }
 else
 ff+=s[i];
}
cout<<ff;
return 0;

}
