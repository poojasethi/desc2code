#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
int i,flag=0,l;char a[101];
scanf("%s",&a);l=strlen(a);
cout<<a[0];
for(i=1;i<l;i++)
{
    if(a[i]=='d'&&a[i+1]=='o'&a[i+2]=='t'&&(i+2)!=(l-1))
    {
        i+=2;cout<<".";
    }
    else if(a[i]=='a'&&a[i+1]=='t'&&flag==0)
    {
        cout<<"@";i++;flag=1;
    }
    else cout<<a[i];    
}
return 0;
}
