#include<cstdio>
#include<cstring>
int main()
{
	char a[1111];
	while(scanf("%s",a)!=EOF){
		int i,len=strlen(a);
		int at=0,t=0;
		for(i=0;i<len;i++){
			if(a[i]=='a'&&a[i+1]=='t'&&at==0&&t){printf("@");i+=1;at=1;t=1;}
			else if(a[i]=='d'&&a[i+1]=='o'&&a[i+2]=='t'&&t&&i+3<=len-1){printf(".");i+=2;}
			else {printf("%c",a[i]);t=1;}
		}
		putchar('\n');
	}
	return 0;
}
