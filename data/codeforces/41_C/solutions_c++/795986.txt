#include<cstdio>
char s[105];
int main()
{
    int i;
    bool at=0;
    gets(s);
    printf("%c",s[0]);
    for(i=1;s[i];)
    {
        if(s[i]=='a'&&s[i+1]=='t'&&!at)
        {
            printf("@"); at=1;i+=2;
        }
        else if(s[i]=='d'&&s[i+1]=='o'&&s[i+2]=='t'&&s[i+3]!=0)
        {
            printf("."); i+=3;
        }
        else printf("%c",s[i++]);
    }
}
