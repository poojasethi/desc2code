#include <iostream>
using namespace std;
int main()
{
	string s, t;
	int i, n, k;
	cin>>s;
	n=s.size();
	t=s.substr(0,1);
	for (k=0,i=1; i<n; i++) {
		if (i+3<n && s.substr(i,3)=="dot") {
			t+="."; i+=2;
		}
		else if (k==0 && i+2<n && s.substr(i,2)=="at") {
			t+="@"; i++; k=1;
		}
		else t+=s[i]; 
	}
	cout<<t<<endl;
	return 0;
}
