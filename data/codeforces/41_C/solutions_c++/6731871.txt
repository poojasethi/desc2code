#include<iostream>
#include<algorithm>
using namespace std;
int k,x;
int main()
{
    string a;
    while(cin>>a)
    {
      x=1;
      k=a.find("at",x);
      a.replace(k,2,"@");
      k=a.find("dot",x);
      while(k!=a.npos&&k!=a.size()-3)
      {
        //cout<<a.size()<<endl;
        a.replace(k,3,".");
        k=a.find("dot",x);
    }
    cout<<a<<endl;
    }
    return 0;
}
