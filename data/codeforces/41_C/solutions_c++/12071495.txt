#include <iostream>
#include <algorithm>
using namespace std;
int main()
{string s;cin>>s;string x=s.substr(0,1);bool check=false;
for(int i=1; i<s.size();)
{if(s.substr(i,3)=="dot"&&i+3!=s.size())
{x=x+'.';
i+=3;}
else if(s.substr(i,2)=="at"&&i+2!=s.size()&&check==false)
{x=x+'@';
i+=2;
check=true;}
else
{x=x+s.substr(i,1);
i++;}}
cout<<x;
return 0;}
