#include <iostream>
#include <algorithm>
#include <vector>
#include <sstream>

using namespace std;

int main() {
  string s;
  cin >> s;
  size_t pos;
  while ((pos = s.find("dot", 1)), pos != string::npos && pos < s.size() - 3)
    s.replace(pos, 3, ".");
  s.replace(s.find("at", 1), 2, "@");
  cout << s << endl;
  return 0;
}
