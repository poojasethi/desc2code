#include <iostream>

using namespace std;

int main() {
	string s;
	cin >> s;
	while (s.find("dot", 1) < s.size() - 3) {
		int k = s.find("dot", 1);
		s = s.substr(0, k) + '.' + s.substr(k + 3);
	}
	int k = s.find("at", 1);
	if (k < s.size() - 2)
		s = s.substr(0, k) + '@' + s.substr(k + 2);
	cout << s;
	return 0;
}
