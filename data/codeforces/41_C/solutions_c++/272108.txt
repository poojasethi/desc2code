#include <iostream>
#include <string>

using namespace std;

int main(){
  string s;cin >> s;
  char first = s[0];
  s = s.substr(1,s.size()-1);
 
  while(s.find("dot")!=string::npos && s.find("dot")!=s.size()-3){
    size_t pos = s.find("dot");
    s.replace(pos, 3,".");
  } 
  if(s.find("at")!=string::npos){
    size_t pos = s.find("at");
    s.replace(pos, 2,"@");
  } 
  
  cout << first+s << endl;

}
