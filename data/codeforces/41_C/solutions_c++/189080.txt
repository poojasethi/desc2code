#include <iostream>

using namespace std;

int main() {
    string s;
    cin >> s;
    char e = s[s.size() - 1];
    s.erase(s.size()-1, 1);
    s.replace(s.find("at", 1), 2, "@");
    int pos = 1;
    while( (pos = s.find("dot", pos)) != string::npos )
        s.replace(pos, 3, ".");

    cout << s + e;
    return 0;
}
