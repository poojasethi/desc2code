#include<iostream>
#include<string>
#include<algorithm>

using namespace std;

int main() {
    string s;
    cin>>s;
    while(1) {
              int f=s.find("dot",1);
              if(f==-1 || f==s.size()-3)break;
              s=s.replace(f,3,".");
    }
    if(s.find("at",1)!=-1)s=s.replace(s.find("at",1),2,"@");
    cout<<s<<endl;
    //system("pause");
    return 0;
}
