#include <iostream>

using namespace std;

int main(int argc, char* argv[])
{
	string s;
	cin >> s;
	
	s.replace(s.find("at", 1), 2, "@");
	
	int p = 0;
	
	while ((p = s.find("dot", p)) != string::npos)
	{
		if ((p != 0) && (p != s.size() - 3))
			s.replace(p, 3, ".");
		p++;
	}
	
	cout << s << endl;
	
	return 0;
}
