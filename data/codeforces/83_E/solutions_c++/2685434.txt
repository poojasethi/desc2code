#include<iostream>
#include<cstring>
using namespace std;
int a[200010],f[21][1100000],n,m,i,j,k,l,ans;
string s;
int main()
{
	ios::sync_with_stdio(false);
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>s;
		for(j=0;j<s.length();j++) a[i]=a[i]*2+s[j]-'0';
	}
	ans=m=s.length();
	memset(f,0x3f,sizeof(f));
	f[0][0]=0;
	for(i=2;i<=n;i++)
	{
		for(j=m;j;j--)
			if((a[i-1]&(1<<j)-1)==(a[i]>>m-j)) break;
		ans+=(k=m-j),l=1<<30;
		for(j=0;j<=m;j++)
			l=min(l,f[m-j][a[i]>>j]+j);
		for(j=0;j<=m;j++)
			f[j][a[i-1]&(1<<j)-1]=min(f[j][a[i-1]&(1<<j)-1],l-k);
	}
	for(l=1<<30,i=0;i<=m;i++)
		for(j=0;j<1<<i;j++) l=min(l,f[i][j]);
	cout<<ans+l<<endl;
	return 0;
}
