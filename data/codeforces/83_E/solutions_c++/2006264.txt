#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
int n,sum = 0,l,prev,now,ss,sv;
int dp[21][(1<<20)],LIM = 50000001;
char s[25];
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%s",s); now = 0;
        if(!i) l = strlen(s);
        for(int j=0;j<l;j++)
            now |= ((s[j]-'0')<<(l-j-1));
        if(!i){
            sum = l;
            for(int j=0;j<=l;j++)
               for(int k=0;k<(1<<j);k++)
                  dp[j][k] = LIM;
            dp[0][0] = 0;
        }
        else{ 
            for(ss=l;ss>=0;ss--)
               if(prev%(1<<ss) == (now>>(l-ss)))
                  break;
            sv = LIM;
            for(int j=0;j<=l;j++) sv = min(sv,dp[j][now>>(l-j)] + l-j);
            for(int j=0;j<=l;j++) dp[j][prev%(1<<j)] = min(dp[j][prev%(1<<j)],sv-(l-ss));
            sum += l-ss;
        }    
        prev = now;
    }
    sv = LIM;
    for(int i=0;i<=l;i++)
      for(int j=0;j<(1<<i);j++) 
          sv = min(sv,dp[i][j]);
    printf("%d\n",sum+sv);
    //system("pause");
    return 0;
}
