#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;
int n,m,ans;
int a[200100];
int f[21][1<<20],e[30];
char s[100];
int ask(const int x,const int y)
{
    for(int i=m;i>0;--i)
        if ( (x&e[i])==(y>>(m-i)) ) return m-i;
    return m;
}
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;++i)
    {
        scanf("%s",s+1);
        m=strlen(s+1);
        a[i]=0;
        for(int j=1;j<=m;++j) a[i]=(a[i]<<1)+s[j]-'0';
    }
    e[0]=0;
    for(int i=1;i<=m;++i) e[i]=e[i-1]*2+1;
    memset(f,1,sizeof(f));
    f[0][0]=0;
    ans=m;int p,q;
    for(int i=2;i<=n;++i)
    {
        p=ask(a[i-1],a[i]),q=1<<20;
        ans+=p;
        for(int j=0;j<=m;++j)
            if ( f[m-j][a[i]>>j]+j<q ) q=f[m-j][a[i]>>j]+j;
        for(int j=0;j<=m;++j)
            if ( f[j][a[i-1]&e[j]]>q-p ) f[j][a[i-1]&e[j]]=q-p;
    }
    q=1<<20;
    for(int i=0;i<=m;++i)
        for(int j=0;j<=e[m];++j)
            if ( f[i][j]<q ) q=f[i][j];
    printf("%d\n",ans+q);
    return 0;
}
