#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define fo(i,a,b) for(i=a;i<=b;++i)
#define fd(i,a,b) for(i=a;i>=b;--i)
using namespace std;
const int mn=210000,mm=1100000,inf=1<<29;
int n,m,i,j,k,v,ans;
int bi[30],a[mn],f[30][mm];
char st[30];
int com(int x,int y)
{
    int i;
    fd(i,m,0)
        if((x&(bi[i]-1))==y>>(m-i))return i;
}
int main()
{
    scanf("%d",&n);
    fo(i,0,25)bi[i]=1<<i;
    fo(i,1,n)
    {
        scanf("%s",st);
        m=strlen(st);
        fo(j,0,m-1)a[i]=a[i]*2+st[j]-48;
    }
    fill(&f[0][0],&f[25][mm-1],inf);
    f[0][0]=0,ans=m;
    fo(i,2,n)
    {
        k=m-com(a[i-1],a[i]),v=inf;
        ans+=k;
        fo(j,0,m)v=min(v,f[j][a[i]>>(m-j)]+m-j);
        fo(j,0,m)
            f[j][a[i-1]&(bi[j]-1)]=min(f[j][a[i-1]&(bi[j]-1)],v-k);
    }
    v=inf;
    fo(i,0,m)
        fo(j,0,bi[m]-1)
            v=min(v,f[i][j]);
    printf("%d\n",ans+v);
    return 0;
}
