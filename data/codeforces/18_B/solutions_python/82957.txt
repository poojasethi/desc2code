s = raw_input()
n,d,m,l = s.split(' ')
n = int(n)
d = int(d)
m = int(m)
l = int(l)
if m - l == 1:
    print d * ( ( ( n-1 ) * m + l ) //d  + 1 )
    exit()
k = 0
xMax = (n-1)*m + l
while True:
    x = (k+1)*m
    x1 = k*m + l
    num = d * (x1//d + 1)
#    print "x = ",x, " x1 = ", x1, " num = ",num, " k = ",k
    if (num > x1) and (num < x):
        print num
        exit()
    if num>xMax:
        print num
        exit()        
    k = k + 1
