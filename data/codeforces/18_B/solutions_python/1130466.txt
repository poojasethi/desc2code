n, d, m, l = map(int, raw_input().split())

now = 0
idx = 0
limit = n * m

if m == l+1:
    print d * (((n-1)*m + l)/d + 1) 
    exit()

while 1:
    step = (l - idx) / d

    now += step * d
    idx += step * d

    now += d
    idx = (idx + d)%m
    if now >= limit or idx > l:
        print now
        break
