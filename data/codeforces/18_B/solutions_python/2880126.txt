import sys
#sys.stdin = open("in.in", "r")

n, d, m, l = map(int, raw_input().split())
b = 0
ans = 0
for k in xrange(n):
    L = k * m
    R = k * m + l
    if b > R:
        continue
    if L <= b and b <= R:
        j = (R - b) / d
        b += (j + 1) * d
    else:
        ans = b
        break

if ans != 0:
    print ans
else:
    print b
