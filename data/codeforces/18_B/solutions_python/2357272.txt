#!/usr/bin/python
import sys

(n, d, m, l) = map(lambda(x): int(x), sys.stdin.readline().split())

r = m - l
if d % r == 0 and l % d == 0:
	k = ((n-1)*m+l) // d
	x = k * d + d
elif d == m:
	x = n * m
else:
	x = 0
	for k in range(m):
		if k * d % m > l:
			x = k * d
			break
	if x > (n-1)*m + l or x == 0:
		x = ( ((n-1)*m+l) // d + 1) * d

print x
