from collections import defaultdict

def main():
    n, d, m, l = map(int, raw_input().split())

    for i in xrange(1, n + 1):
        x = (i - 1) * m + l
        step = x / d

        if i == n:
            print (step + 1) * d
            return

        if (step + 1) * d < i * m:
            print (step + 1) * d
            return

if __name__ == '__main__':
    main()
