#include<iostream>
using namespace std;
int main(){
int n,i,j,flag=0;
cin>>n;
char a[300][300],b,c;
for(i=0;i<n;i++)
for(j=0;j<n;j++)
cin>>a[i][j];
b=a[0][0];
c=a[0][1];
if (b==c) flag=1;
for(i=0;i<n;i++){
for(j=0;j<n;j++){
if(i==j || i+j==n-1){
if(a[i][j]!=b) {flag=1;break;}}
else
if(a[i][j]!=c){flag=1;break;}}}
if(!flag) cout<<"YES";
else cout<<"NO";

}
