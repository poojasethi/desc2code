#include<cstdio>
using namespace std;
char s1[300][300];
int main()
{
	int n;char a,b;
	scanf("%d",&n);
	for(int i=0;i<n;++i)
		scanf("%s",s1[i]);
	a=s1[0][0];b=s1[0][1];
	if(a==b)
	{puts("NO");return 0;}
	for(int i=0;i<n;++i)
	for(int j=0;j<n;++j)
	{
		if((i==j||i+j+1==n))
		{
			if(s1[i][j]!=a)
			{puts("NO");return 0;}
		}
		else if(s1[i][j]!=b)
		{puts("NO");return 0;}
	}
	puts("YES");
}
