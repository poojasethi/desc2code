#include<iostream>
using namespace std;
int main()
{
int n=0;
cin>>n;
char x=' ',y=' ';
cin>>x>>y;
if(x==y)
{
cout<<"NO";
return 0;
}
for(int i=0;i<n;i++)
for(int j=(i==0)?2:0;j<n;j++)
{
char t=' ';
cin>>t;
if(i==j||i+j==n-1)
{
if(t!=x)
{
cout<<"NO";
return 0;
}
}
else{
if(t!=y){
cout<<"NO";
return 0;
}
}
}
cout<<"YES";

return 0;
}
