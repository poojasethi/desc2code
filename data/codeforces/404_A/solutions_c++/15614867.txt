#include<stdio.h>
int main()
{
int n,i,j,count=0,other=0,other1=0;
scanf("%d",&n);
char a[n][n],x,p;
for(i=0;i<n;i++)
scanf("%s",a[i]);
x=a[0][0];
p=a[0][1];
for(i=0;i<n;i++)
for(j=0;j<n;j++)
{
if(i==j&&a[i][j]==x)
count++;
if(i+j==n-1&&a[i][j]==x)
count++;
if(i!=j&&i+j!=n-1&&a[i][j]==p)
other++;
}
if(count==(2*n)&&other==(n*n)-(2*n)+1&&x!=p)
printf("YES");
else
printf("NO");
}
