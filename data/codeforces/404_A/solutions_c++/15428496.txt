#include<stdio.h>
int main()
{
	int n,i,f,j;
	char a,c,d;
	
	scanf("%d",&n);
	f=1;
	getchar();
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			if(i==1&&j==1)c=getchar();
			else if(i==1&&j==2)d=getchar();
			else 
			{
				a=getchar();
				if(((i==j||i+j==n+1)&&a-c)||(i!=j&&i+j!=n+1&&a-d))f=0;
			}
		}
		getchar();
	}
	if(c-d&&f)printf("YES\n");
	else printf("NO\n");
	return 0;
}
