#include<bits/stdc++.h>

int main(){
	int n,  f = 1, i, j;
	char x,y,c;
	scanf("%d\n", &n);
	scanf("%c %c", &x, &y);
	f &= x != y;
	for (i = 0; f && i < n; ++i)
	for (j = (i == 0) ? 2 : 0; j < n; ++j)
	{ 
		scanf(" %c", &c);
		f &= (i == j || j == n - i - 1) ? c == x : c == y;
	}
	puts((f) ? "YES" : "NO");
}
