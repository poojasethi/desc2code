#include<iostream>
using namespace std;
int main()
{
	std::ios::sync_with_stdio(false);
	int f,i,n,j;
	char s[400][400],a,b;
	cin>>n;
	for(i=0;i<n;i++)
		for(j=0;j<n;j++)
			cin>>s[i][j];
	a=s[0][0];
	b=s[0][1];
	for(f=1,i=0;i<n;i++)
		for(j=0;j<n;j++)
		{
			if((i==j||i+j==n-1)&&s[i][j]!=a)
				f=0;
			else if(i!=j&&i+j!=n-1&&s[i][j]!=b)
					f=0;
		}
		f&&a!=b?cout<<"YES\n":cout<<"NO\n";
}
