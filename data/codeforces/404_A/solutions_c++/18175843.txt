#include <iostream>

using namespace std;

int main()
{
	char s[305][305], a[2];
	int n;
	cin >> n;
	for(int i = 0; i < n; i++) cin >> s[i];
	if((a[0] = s[0][1]) == (a[1] = s[0][0]))
	{
		cout << "NO\n";
		return 0;
	}	
	for(int i = 0; i < n; i++)
		for(int j = 0; j < n; j++)
			if(s[i][j] != a[i == j || i+j == n-1])
			{
				cout << "NO\n";
				return 0;
			}
	cout << "YES\n";
}
