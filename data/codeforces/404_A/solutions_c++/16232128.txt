#include <bits/stdc++.h>
using namespace std;
char a[333][333];
int n;
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)cin>>a[i];
	char c=a[0][0];
	char d=a[0][1];
	bool ok = true;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			if(i==j || i+j==n-1){
				if(a[i][j]!=c)ok=false;
			}
			else{
				if(a[i][j]!=d)ok=false;
			}
		}
	}
	if(ok && c!=d)cout<<"YES";
	else cout<<"NO";
	return 0;
}
