#include<cstdio>
#define NO {puts("NO");return 0;}
using namespace std;
char a[1005][1005],s,ss;
int n,i,j;
int main()
{
  scanf("%d",&n);
  for (i=1;i<=n;i++)
    scanf("%s",a[i]+1);
  s=a[1][1];ss=a[1][2];
  if (s==ss) NO;
  for (i=1;i<=n;i++)
    for (j=1;j<=n;j++)
    {
      if (j==i||j==n-i+1)
        {if (a[i][j]!=s) NO;}
      else if (a[i][j]!=ss) NO;
    }
  puts("YES");
  return 0;
}
