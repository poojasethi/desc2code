#include <iostream>
using namespace std;
int n;
char d,o,x;
int main(){
	cin>>n>>d>>o;
	if(d==o){
			cout<<"NO"<<endl;
			return 0;
		}
	for(int i=2;i<n*n;i++){
		cin>>x;
		if((i%n==i/n || n-(i%n)-1==i/n) && x!=d){
			cout<<"NO"<<endl;
			return 0;
		}
		else if(!(i%n==i/n || n-(i%n)-1==i/n) && x!=o){
			cout<<"NO"<<endl;
			return 0;
		}
	}
	cout<<"YES"<<endl;
	return 0;
}
