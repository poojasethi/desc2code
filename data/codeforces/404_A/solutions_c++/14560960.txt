#include<iostream>
#include <cmath>
#include <map>
using namespace std;


int main(){
	char a,x=0,nx=0;
	int n;
	bool N=0;
	cin >> n;
	for(int i = 0; i < n; i++){
		for(int j = 0; j < n; j++){
			cin >> a;
			if(x==0) x = a;
			if(nx==0 and i+1==j) nx=a;
			if((i==j or i==n-j-1) and !N) N=x!=a;
			else  if(!N) N=nx!=a;
		}
	}
	if(!N and x!=nx) cout << "YES";
	else cout << "NO";
	return 0;
}
