#include<bits/stdc++.h>

using namespace std;


int main (){

	int n;
	char c;
	set <char> di;
	set <char> all;

	cin >> n;

	for (int i = 0; i < n; i++){
		for (int j = 0; j < n; j++){
			cin >> c;
			if (i == j || j == n - i - 1)
				di.insert (c);
			else all.insert (c);
		}
	}

	if (di.size() != 1 || all.size() != 1 || di == all)
		cout << "NO";
	else cout << "YES";

}
