#include<stdio.h>
int main ()
{
int n,i,j,f=0;
char x,a,b;
scanf("%d",&n);
for(i=0; i<n; i++)
	for(j=0; j<n; j++)
	{	
		scanf(" %c",&x);
		if (!i && !j)
			a=x;
		if (!i && j==1)
			b=x;
		if ((i+j == n-1 || i==j) && (x != a || x==b))
			f=1;
		if (!(i+j == n-1 || i==j) &&(x != b || x==a))
			f=1;
	}
puts((f)?"NO":"YES");
}
