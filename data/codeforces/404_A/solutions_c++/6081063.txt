#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
 int n,t=0,ch=1;
 char c1,c2;
 cin>>n;
 for(int i=0;i<n;i++)
  for(int j=0;j<n;j++)
  {
   char c;
   cin>>c;
   if(t==0)
    {c1=c;t++;}
    else if(t==1)
    {c2=c;t++;}
   if(i==j||i+j==n-1){if(c!=c1)ch=0;}
   else 
   if(c1==c2||c!=c2)
   ch=0;
   }
  if(ch)
  cout<<"YES";
  else
  cout<<"NO";
 return 0;
}
