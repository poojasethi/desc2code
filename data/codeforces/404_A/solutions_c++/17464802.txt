#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll n,a;
int main()
{

	cin>>n;
	string s[n+1];
	for(ll i=0;i<n;i++)cin>>s[i];
	char c=s[0][0],b=s[0][1];
	for(ll i=0;i<n;i++){
		for(ll j=0;j<n;j++){
			if(i==j||i==n-j-1||n-i-1==j){
				if(s[i][j]!=c) a=1;
			}
			else if(s[i][j]==c||s[i][j]!=b) a=1;
		}
	}
	if(a) cout<<"NO";
	else cout<<"YES";
	return 0;
}
