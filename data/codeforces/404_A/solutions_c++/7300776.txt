#include<stdio.h>
#include<math.h>
#include<cstring>
#include<iostream>

int n;char a[301][301];bool x;

{
	scanf("%d",&n);for(int i=0;i<n;i++)scanf("%s",&a[i]);
	for(int i=0;i<n;i++)for(int j=1;j<=n;j++)if(a[i][j-1]!=a[(i!=j-1 && i+j!=n)][0])x=true;
