#include<iostream>
using namespace std;
int main(){
     char p[2],s[320][320];
     int i,j,n;
     bool flag;
     while(cin>>n)
     {
         for(i=0;i<n;i++) cin>>s[i];
         p[0]=s[0][1],p[1]=s[0][0];
         flag=!(p[0]==p[1]);
         for(i=0;i<n;i++)
         for(j=0;j<n;j++)
         if(s[i][j]!=p[(i==j)||(i+j==n-1)])
         flag=0;
         cout<<(flag?"YES":"NO")<<endl;
     }
}
