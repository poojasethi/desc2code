#include <iostream>

using namespace std;

int main(){
  int n;   cin >>n;
  char a,je,nie;
  for(int i=0; i<n; i++)
    for (int j=0; j<n; j++){
      cin >>a;
      if(i==0&&j==0){je=a; }
      if(i==0&&j==1){nie=a;}
      if (j>1&&je==nie){cout<<"NO\n";return 0;}
      if(i==j || i==n-j-1 ){if (a!=je){cout<<"NO\n"; return 0;}}
      else{ if(a!=nie){cout<<"NO\n";return 0;}}
    }
  cout<<"YES\n";
}
