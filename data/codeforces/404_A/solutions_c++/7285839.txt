#include <cstdio>
#include <algorithm>
using namespace std;
char a,b,s[505];
int n,i,j;
int main(){
	scanf("%d",&n);if (n==1){printf("YES");return 0;}
	for(i=1;i<=n;i++){
		scanf("%s",s+1);if(i==1){a=s[1];b=s[2];if (a==b){printf("NO");return 0;}}
		for(j=1;j<=n;j++)if (((j==i||j==n-i+1)&&(s[j]!=a))||((j!=i&&j!=n-i+1)&&(s[j]!=b))){printf("NO");return 0;}
	}
	printf("YES");
	return 0;
}
