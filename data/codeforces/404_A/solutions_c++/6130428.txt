#include<bits/stdc++.h>
using namespace std;

int main(){
	int n, ans, f = 1, i, j;
	char grid[305][305];
	cin >> n;
	for (i = 0; i < n; cin >> grid[i], ++i);
	f &= grid[0][0] != grid[0][1];
	for (i = 0;f &&  i < n; ++i){
		for (j = 0; j < n; ++j)if (i == j || n - i - 1 == j)f &= grid[i][j] == grid[0][0];
		else f &= grid[0][1] == grid[i][j];
	}
	puts((f) ? "YES" : "NO");
}
