#include <iostream>
#include <set>
using namespace std;
set<char> a, s; int n; char w;
int main()
{ cin >> n;
    for (int i = 0; i < n;i++)
    {
    for (int j = 0; j < n; j++)
    {
        cin >> w;
        if (j == i||j==n-1-i)
        a.insert(w);
        else
         s.insert(w);
    }
    }
    if (a.size() == 1 && s.size() == 1 && a != s)
    cout << "YES";
    else cout << "NO";
}
