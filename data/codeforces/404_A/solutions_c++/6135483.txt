#include <iostream>
#include <algorithm>
using namespace std;
int main()
{int n; cin>>n; char a[n][n]; bool res=true;
for(int i=0;i<n;i++) for(int j=0;j<n;j++) cin>>a[i][j];
char tr=a[0][0]; char ex=a[0][1];
for(int i=0;i<n;i++)
{for(int j=0;j<n;j++)
{
{if(i==j||i+j==n-1)
{if(a[i][j]!=tr) res=false;}
else 
if(a[i][j]!=ex) res=false;
}
}}
if(res&&ex!=tr) cout<<"YES"; else cout<<"NO";
return 0;
}
