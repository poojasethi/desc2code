#include<iostream>
using namespace std;
int main() { int n,z;
char di,check,x;
bool f=true;
cin>>n>>di>>check;
z=n-1; if(di==check) f=false;  
for(int i=0;i<n;i++)
for(int j=0;j<n;j++)
{ cin>>x;
    if((i==0)&&(j==0))
j=2;
if((i==j)||((i+j)==z))
{if(x!=di)
f=false;
}
else { if(x!=check) f=false;  }
}   if(f)
    cout<<"YES"<<endl;
else
    cout<<"NO"<<endl;
return 0; }
