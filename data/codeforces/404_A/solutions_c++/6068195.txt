#include<bits/stdc++.h>
using namespace std;

int main() {
	int n; cin >> n;
	set<int> dia, notdia;
	for(int i = 0; i < n; ++i) {
		string s; cin >> s;
		for(int j = 0; j < n; ++j) {
			if(i == j || j == n - i - 1) dia.insert(s[j]);
			else notdia.insert(s[j]);
		}
	}
	puts(dia.size() == 1 && notdia.size() == 1 && *dia.begin() != *notdia.begin() ? "YES" : "NO");
}
