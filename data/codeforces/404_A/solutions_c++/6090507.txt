#include <cstdio>

int n, i, j;
bool f;
char s[303], c, x;

int main()
{
	scanf("%d%s", &n, s);

	x = s[0], c = s[1];

	f = c != x;



	for (i = 0; f && i < n; i++)
	{
		for (j = 0; j < n; j++)
			f &= j == i || j == n - i - 1 ? s[j] == x : s[j] == c;
		scanf("%s", s);
	}

	puts(f ? "YES" : "NO");
}
