#!/bin/python2

from sys import stdin

n=int(stdin.readline().strip())
ans="YES"
c=''
d=''
flag=1
for i in range(n):
    line=stdin.readline().strip()
    if i==0:
        c=line[0]
        d=line[1]
    if c==d:
        ans="NO"
        break
    if line[i]!=c or line[n-i-1]!=c:
        ans="NO"
        break
    for j in range(n):
        if j==i or j==n-i-1:
            continue
        if line[j]!=d:
            ans="NO"
            break
print ans
