n = int(raw_input())
x = []
for i in range(0,n):
    x.append(list(raw_input()))
d = x[0][0]
s = x[0][1]
if d == s :
    print "NO"
    exit()

for i in range(0, n):
    if x[i][i] == d and x[i][n-i-1]==d :
        x[i][i] = s
        x[i][n-i-1]=s
    else:
        print "NO"
        exit()


for i in range(0,n):
    for j in range(0,n):
        if x[i][j] !=s:
            print "NO"
            exit()
print "YES"
