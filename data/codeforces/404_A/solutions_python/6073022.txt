import sys

def main():
  n = int(sys.stdin.readline())
  c = 0
  ok = True
  for i in xrange(n):
    line = sys.stdin.readline()
    t = 0
    if i == 0:
      XLetter = line[0]
      VLetter = line[1]
      if XLetter == VLetter:
        ok = False
    for letter in line:
      if t == i or t == n-i-1:
        if letter != XLetter:
          ok = False
      elif t < n:
        if letter != VLetter:
          ok = False
      t += 1
  if ok:
    print "YES"
  else:
    print "NO"

main()
