[n] = map(int,raw_input().split())
a = []
ans = "YES"
for i in range(0,n):
    a.append(raw_input())
for i in range(0,n):
    for j in range(0,n):
        if i == j :
            if a[i][j] <> a[0][0] :
                ans = "NO"
        elif i+j == n-1 :
            if a[i][j] <> a[0][0] :
                ans = "NO"
        else:
            if a[i][j]<>a[0][1] :
                ans = "NO"
if a[0][0] == a[1][0] :
    ans = "NO"
print ans
