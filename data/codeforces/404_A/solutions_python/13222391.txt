from sets import Set

n = int(raw_input())
mat = []
for i in range(n):
    l = raw_input()
    mat.append(l)

d = Set([])
nd = Set([])
for i in range(n):
    for j in range(n):
        if i == j or n-j-1 == i:
            d.add(mat[i][j])
        else:
            nd.add(mat[i][j])

if len(d) == 1 and len(nd) == 1 and d.pop() != nd.pop():
    print "YES"
else:
    print "NO"
