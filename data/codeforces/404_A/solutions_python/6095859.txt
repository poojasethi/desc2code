r = lambda: map(int, raw_input().split())

n = input()
s = [raw_input() for _ in xrange(n)]
u, v = [], []
for i in xrange(n):
    for j in xrange(n):
        if j == i or i + j == n - 1:
            u.append(s[i][j])
        else:
            v.append(s[i][j])
u.sort()
v.sort()
if u[0] == u[-1] and v[0] == v[-1] and u[0] != v[0]:
    print "YES"
else:
    print "NO"
