n = input()
b = [raw_input() for x in xrange(n)]
d_char,s_char = b[0][0],b[0][1]
for x in xrange(n):
    for y in xrange(n):
        if x == y or x + y == n -1:
            if b[x][y] != d_char:
                print "NO"
                exit(0)
        elif b[x][y] != s_char:
            print "NO"
            exit(0)
print "YES" if d_char != s_char else "NO"
