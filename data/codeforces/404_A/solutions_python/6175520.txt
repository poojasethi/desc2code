n = int(raw_input())
s = []
for i in range(n):
    s.append(raw_input())

X = s[0][0]
Y = s[0][1]
if X == Y:
    print "NO"
    exit()
for i in range(n):
    for j in range(n):
        let = s[i][j]
        if i == j or i == n - j - 1:
            if let is not X:
                print "NO"
                exit()
        else:
            if let is not Y:
                print "NO"
                exit()
print "YES"
