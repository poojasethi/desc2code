n = int(raw_input())
table = []
for i in xrange(0, n):
	table.append(raw_input().strip())
diag = []
not_diag = []
for i in xrange(0, n):
	for j in xrange(0, n):
		if ( i == j) or (i + j == n - 1):
			diag.append(table[i][j])
		else:
			not_diag.append(table[i][j])

if (len(set(diag)) == 1) and (len(set(not_diag)) == 1) and (not (diag[0] == not_diag[0])):
	print "YES"
else:
	print "NO"
