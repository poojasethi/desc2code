n = input()
A= map(lambda x: raw_input() , xrange(n))
xletter=A[0][0]
nonxletter=A[0][1]
exitflag=False
for i in xrange(n):
    for j in xrange(n):
        if (A[i][j] not in [xletter,nonxletter]) or (A[i][j]!= xletter and (j==i or j==n-1-i)) or (A[i][j]!= nonxletter and (j!=i and j!=n-1-i)):
            print "NO"
            exitflag=True
            break
    if exitflag:
        break
if not exitflag:
    if len(set([xletter,nonxletter]))==1:
        print "NO"
    else:
        print "YES"
