def solve(grid):
  n = len(grid)
  s1, s2 = set(), set()
  for i in range(n):
    for j in range(n):
      if i == j or n-1-i == j:
        s1.add(grid[i][j])
      else:
        s2.add(grid[i][j])
  return len(s1) == 1 and len(s2) == 1 and list(s1)[0] != list(s2)[0]

n = int(raw_input())
grid = [[c for c in raw_input()] for i in range(n)]
print 'YES' if solve(grid) else 'NO'
