def judge(maze,n):
    c = maze[0][0]
    for i in range(0,n):
        if maze[i][i] != c or maze[i][n - i - 1] != c:
            return "NO"
    if maze[0][0] == maze[0][1]:
        return "NO"
    c = maze[0][1]
    for i in range(0,n):
        for j in range(0,n):
            if i != j and j != n - i - 1 and maze[i][j] != c:
                return "NO"
    return "YES"

n = (int)(raw_input())
maze = [None] * n;
for i in range(0,n):
    maze[i] = raw_input()
print judge(maze,n)
