n = input()
s = []
for i in range(n):
	s.append(raw_input())
#print s
cros = s[0][0]
othe = s[0][1]
#print cros, othe
if cros == othe:
	print "NO"
else:
	can = True
	for i in range(n):
		for j in range(n):
			if i == j or i+j == n-1: 
				if s[i][j] != cros:
					can = False
			elif s[i][j] != othe:
				can = False
	if can:
		print "YES"
	else:
		print "NO";
