T=input()
L=[]
Z=[0]*26
for _ in xrange(T):
	L.append(raw_input())
	for x in L[-1]:
		Z[ord(x)-97]+=1
count=0
N=[]
for x in Z:
	if(x>0):
		count+=1
		N.append(x)
if(count!=2):
	print "NO"
else:
	y=L[0][0]
	flag=True
	for x in xrange(T):
		if(L[x][x]==y and L[x][T-x-1]==y):
			continue
		flag=False
		break
	if(flag):
		N.sort()
		if(N[0]!=2*T-1 and N[1]!=2*T-1):
			flag=False
	if(flag):
		print "YES"
	else:
		print "NO"
