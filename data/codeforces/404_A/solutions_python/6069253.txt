import sys
n = int (sys.stdin.readline())
a = sys.stdin.readlines()
res = "YES"
if a[0][0] == a[0][1]:
  res = "NO"
for i in range (n):
  for j in range (n):
    if i == j or i + j == n - 1:
      if a[i][j] != a[0][0]:
        res = "NO"
    else:
      if a[i][j] != a[0][1]:
        res = "NO"

print res
