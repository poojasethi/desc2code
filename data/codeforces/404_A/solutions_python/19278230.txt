N=int(raw_input().strip())
m=[list(raw_input().strip()) for _ in xrange(N)]
if N==1:
	print 'YES'
else:
	d,x,flag=m[0][0],m[0][1],False
	for i in xrange(N):
		for j in xrange(N):
			if ((i==j or i==N-j-1)and d!=m[i][j]) or (i!=j and i!=N-j-1 and x!=m[i][j]) or d==x:
				print 'NO'
				flag=True
				break
		if flag:
			break
	else:
		print 'YES'
