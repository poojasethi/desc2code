n = int(raw_input())
a = set([])
x = 0
ans = "YES"
first = 1
for i in range(n):
    r = raw_input().strip()
    a.update(r)
    if first:
        b = r[0]
        c = b * 2
        first = 0
    x += r.count(b)    
    if r[i] + r[n-i-1] != c or len(a) != 2:
        ans = "NO"
        break
if x != n * 2 - 1: ans = "NO"    
print ans
