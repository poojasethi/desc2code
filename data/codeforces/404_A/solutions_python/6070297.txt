n = input()
field = []
for i in range(n):
    field.append(raw_input().strip())

dia = set()
other = set()
for i in range(n):
    for j in range(n):
        if i == j or i + j == n - 1:
            dia.add(field[i][j])
        else:
            other.add(field[i][j])
if len(dia) == 1 and len(other) == 1 and len(dia | other) == 2:
    print "YES"
else:
    print "NO"
