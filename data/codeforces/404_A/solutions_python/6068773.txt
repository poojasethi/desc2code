n = int(raw_input())
f = True

for i in range(n):
  line = raw_input()
  if i == 0:
    diag_char = line[0]
    other_char = line[1]
    if diag_char == other_char:
      f = False
      break
  if line[i] != diag_char or line[n-1-i] != diag_char:
    f = False
    break
  for j in range(n):
    if j == i or j == n-1-i:
      continue
    if line[j] != other_char:
      f = False
      break
  if not f:
    break

if f:
  print 'YES'
else:
  print 'NO'
