n=int(raw_input())
b=[]
flag=0
for i in range(n):
    a=list(raw_input())
    b.append(a)
m=b[0][0]
p=b[0][1]
if m==p:
    print "NO"
else:
    for i in range(n):
        for j in range(n):
            if i==j or j==n-1-i:
                if b[i][j]!=m:
                    flag=1
                    break
            else:
                if b[i][j]!=p:
                    flag=1
                    break
    if flag==1:
        print "NO"
    else:
        print "YES"
