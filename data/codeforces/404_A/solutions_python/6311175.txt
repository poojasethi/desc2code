#!/usr/bin/env python
n = input()
lines = map(lambda x: raw_input(), range(n))

dc = lines[0][0]
ndc = lines[0][1]

center = (n-1) / 2
valid = dc != ndc

for i in range(n):
	line = lines[i]

	def isOk(pos, c):
		if abs(center - i) == abs(center - pos):
			return c == dc
		else:
			return c == ndc
	status = map(lambda (x, y): isOk(x, y), enumerate(line))
	
	if not all(status):
		valid = False
		break

if valid:
	print 'YES'
else:
	print 'NO'
