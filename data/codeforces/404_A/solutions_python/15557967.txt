n = input()

a = raw_input()
x, x1, f = a[0], a[-1], 1
if x != x1:
    f = 0

for i in xrange(1, n - 2):
    if a[i] != a[i + 1]:
        f = 0
        break
o = a[1]

if o == x:
    f = 0

for i in xrange(1, n):
    a = raw_input()
    x_a, x1_a = a[i], a[n - i - 1]
    if x_a != x or x1_a != x:
        f = 0
    for j in xrange(n):
        if j == i or j == (n - i - 1): continue
        if a[j] != o:
            f = 0
            break
if f:
    print 'YES'
else:
    print 'NO'
