I=lambda:map(int, raw_input().split())
n = input()
a, b = '', ''
for i in xrange(n):
    s = raw_input()
    if i == 0: a = s[0]
    for j in xrange(n):
        if i == j or i + j == n-1:
            if a != s[j]: print 'NO';exit()
        else:
            if a == s[j] or (b != '' and b != s[j]): print 'NO';exit()
            else: b = s[j]
print 'YES'
