n = int(raw_input())

l = [raw_input() for _ in xrange(n)]


d = set(l[i][i] for i in range(n)) | set(l[n-i-1][i] for i in range(n))

o = set(l[i][j] for i in range(n) for j in range(n) if i != j and n-j-1 != i)


if len(d) == 1 and len(o) == 1 and list(d)[0] != list(o)[0]:
    print 'YES'
else:
    print 'NO'
