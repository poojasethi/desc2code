#include<cstdio>
#include<algorithm>
using namespace std;
int a[1001],b[1001],len,out[1<<20];
int main()
{
    int i,j,n;
    scanf("%d",&n);
    for(i=0;i<n;i++) scanf("%d",a+i);
    for(i=0;i<n;i++) scanf("%d",b+i);
    for(i=0;i<n;i++)
    {
        for(j=i;j<n;j++)
        {
            if(a[i]==b[j]) break;
        }
        if(i==j) continue;
        for(;j!=i;j--)
        {
            swap(b[j-1],b[j]);
            out[len++]=j;
        }
    }
    printf("%d\n",len);
    for(i=0;i<len;i++) printf("%d %d\n",out[i],out[i]+1);
    return 0;
}
