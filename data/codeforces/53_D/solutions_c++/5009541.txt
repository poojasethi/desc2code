#include <iostream>
#include <cmath>
#include <string>

using namespace std;

int p[1000005][2];
int a[305];
int b[305];

int main()
{
	int ans = 0;
	int n;
	cin >> n;
	for (int i = 0; i<n; i++)
		cin >> a[i];
	for (int i = 0; i<n; i++)
		cin >> b[i];
	for (int i = 0; i<n; i++)
	{
		int j = i;
		for (; b[j] != a[i]; j++);
		for (int k = j-1; k >= i; k--)
		{
			p[ans][0] = k;
			p[ans][1] = k+1;
			ans++;
			swap(b[k], b[k+1]);
		}
	}
	cout << ans << endl;
	for (int i = 0; i<ans; i++)
		cout << p[i][0] + 1 << ' ' << p[i][1] + 1 << endl;
	return 0;
}
