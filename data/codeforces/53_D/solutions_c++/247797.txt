#include <iostream>
#include <vector>

using namespace std;

int main()
{
  int n;
  cin>>n;
  vector<int> a(n);
  for (int i=0;i<n;i++) cin>>a[i];
  vector<int> b(n);
  for (int i=0;i<n;i++) cin>>b[i];
  vector<pair<int,int> > sol;
  for (int i=0;i<n;i++) {
    int buscat=a[i];
    int pos=-1;
    for (int j=i;pos==-1;j++)
      if (b[j]==buscat) pos=j;
    for (;pos>i;pos--) {
      sol.push_back(pair<int,int> (pos-1,pos));
      swap(b[pos-1],b[pos]);
    }
  }
  cout<<int(sol.size())<<endl;
  for (int i=0;i<int(sol.size());i++)
    cout<<sol[i].first+1<<" "<<sol[i].second+1<<endl;
}
