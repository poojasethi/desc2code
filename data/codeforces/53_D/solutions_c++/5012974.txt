#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

int answer[1000000], ans = 0, a[1000], b[1000];

int main ()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	for (int i = 0; i < n; i++)
		cin >> b[i];
	for (int i = n - 1; i >= 0; i--)
		for (int j = i - 1; j >= 0; j--)
			if (b[j] == a[i])
			{
				for (int k = j; k < i; k++)
				{
					answer[ans] = k;
					ans++;
					swap (b[k], b[k + 1]);
				}
				break;
			}
	cout << ans << endl;
	for (int i = 0; i < ans; i++)
		cout << answer[i] + 1 << " " << answer[i] + 2 << endl;
    return 0;
}
