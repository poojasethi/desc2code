#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;

int n, a[300], b[300];
vector <int> x, y;
int main()
{
	cin >> n;
	for(int i = 0; i < n; i++)	cin >> a[i];
	for(int i = 0; i < n; i++)	cin >> b[i];
	for(int i = 0; i < n; i++)
		if(b[i] != a[i])
		{
			int j;
			for(j = i + 1; b[j] != a[i]; j++);
			for(int k = j - 1; k >= i; k--)
			{
				swap(b[k], b[k + 1]);
				x.push_back(k + 1);
				y.push_back(k + 2);
			}
		}
	cout << x.size() << endl;
	for(int i = 0; i < x.size(); i++)
		cout << x[i] << ' ' << y[i] << endl;
	return 0;
}
