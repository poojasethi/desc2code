#include <iostream>
#include <algorithm>
#include <vector>

#define rep(i,n) for(int i=0;i<n;i++)
#define pr(i) cout<<i<<endl

using namespace std;

int n,a[321],b[321],c,p;
vector<int> s;

int main(){

  cin>>n;
  rep(i,n)cin>>a[i];
  rep(i,n)cin>>b[i];
  rep(i,n){

    p=find(b+i,b+n,a[i])-b;

    while(i<p){
      s.push_back(p-1);
      swap(b[p-1],b[p]);
      p--;
    }
  }
  
  pr(s.size());
  rep(i,s.size())pr(s[i]+1<<" "<<s[i]+2);

  return 0;

}
