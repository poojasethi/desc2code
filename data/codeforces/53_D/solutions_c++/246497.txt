#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
  vector<pair<int,int> > sw;
  int n;
  cin>>n;
  int current[n],must[n];
  for(int i=0;i<n;i++)
    cin>>must[i];
  for(int j=0;j<n;++j)
    cin>>current[j];

  for(int i=0;i<n;i++)
    for(int j=i;j<n;++j)
      if(current[j]==must[i])
	{
	  for(int prn=j;prn>i;prn--)
	    {
	      sw.push_back(pair<int,int> (prn,prn+1));
	      swap(current[prn],current[prn-1]);
	    }
	  
	  break;
	}

  cout<<sw.size()<<endl;
  for(int i=0;i<sw.size();++i)
    cout<<sw[i].first<<" "<<sw[i].second<<endl;

  return 0;
}
