#include<iostream>
#include<vector>
using namespace std;

int a[311],b[311];
vector<int> x;
int n, i, j,t;
int main(){
    cin>>n;
    for(i=1;i<=n;i++)cin>>a[i];
    for(i=1;i<=n;i++)cin>>b[i];
    for(i=1;i<n;i++)
      for(j = i; j<= n;j++) if(a[i] == b[j])
        {
            for(t =j-1; t>=i; t--) {
                  x.push_back(t);
                  int tmp = b[t];
                  b[t] = b[t+1]; b[t+1] = tmp;
            }
            break;
        }
    cout<<x.size()<<endl;
    for(i=0;i<x.size();i++)cout<<x[i]<<" "<<x[i]+1<<endl;
    //system("pause");
}
