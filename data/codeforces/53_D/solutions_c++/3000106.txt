#include<iostream>
#include<cstdio>

using namespace std;

int a[330],b[330];
inline void iexchange(int &a,int &b){
	if(a==b)return;
	a^=b;b^=a;a^=b;	
}
int c[1123456];
int d[1123456];
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;++i)
		cin>>a[i];
	for(int i=1;i<=n;++i)
		cin>>b[i];
		
	int k=0;	
	for(int i=1;i<=n;++i)
	{
		int j=i;
		for(;j<=n;++j)
		{
			if(a[i]==b[j])
				break;
		}
		while(i<j)
		{
			iexchange(b[j],b[j-1]);
			c[k]=j-1;
			d[k]=j;
			--j;
			++k;
		}
	}
	cout<<k<<endl;
	for(int i=0;i<k;++i)
	{
		cout<<c[i]<<' '<<d[i]<<endl;
	}
	return 0;
}
