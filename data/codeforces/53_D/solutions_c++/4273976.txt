#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
int n,i,a[301],b[301],j;
vector <pair<int,int> > path;
int main()
{
cin>>n;
for (i=0;i<n;i++)
cin>>a[i];
for (i=0;i<n;i++)
cin>>b[i];
for (i=0;i<n;i++)
{
	for (j=i;j<n;j++)
	{
		if (b[j]==a[i])
			break;
	}
	while (j!=i)
		{
			swap(b[j],b[j-1]);
			path.push_back(make_pair(j,j+1));
			j--;
		}
}
cout<<path.size()<<endl;
for (i=0;i<path.size();i++)
cout<<path[i].first<<" "<<path[i].second<<endl;



return 0;
}
