#include<math.h>
#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int n,m,o=0;
long long a[500],b[500];
struct q{
	int x,y;
}q[1000001];
void swap(long long &x,long long &y)
{
	long long t=x;x=y;y=t;
}
int main()
{
	int i,j;
	scanf("%d",&n);
	for(i=1;i<=n;i++)cin>>a[i];
	for(i=1;i<=n;i++)cin>>b[i];
	for(i=1;i<=n;i++){
		for(j=i;j<=n;j++){
			if(a[i]==b[j])break;
		}
		while(j>i){
			swap(b[j-1],b[j]);
			o++;q[o].x=j-1;q[o].y=j;
			j--;
		}
	}
	printf("%d\n",o);
	for(i=1;i<=o;i++)printf("%d %d\n",q[i].x,q[i].y);
	//while(1);
}
