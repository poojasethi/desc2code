#include<stdio.h>
int main()
{
	long int a[510],b[510],x[100001],y[100001],i,j,k,n,l,temp;
	scanf("%ld",&n);
	for(i=1;i<=n;i++) scanf("%ld",&a[i]);
	for(i=1;i<=n;i++) scanf("%ld",&b[i]);
	k=0;
	for(i=1;i<=n;i++)
	{
		for(j=i;j<=n;j++)
			if (b[j]==a[i]) break;
		for(l=j;l>i;l--) 
		{
			k++;
			x[k]=l-1;y[k]=l;
			temp=b[l];b[l]=b[l-1];b[l-1]=temp;
		}
	}
	printf("%ld\n",k);
	for(i=1;i<=k;i++) printf("%ld %ld\n",x[i],y[i]);
	return 0;
}
