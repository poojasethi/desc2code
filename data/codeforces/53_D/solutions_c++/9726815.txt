#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	scanf("%d",&n);
	int A[n],B[n];
	for(int i=0;i<n;++i)
		scanf("%d",A+i);
	for(int i=0;i<n;++i)
		scanf("%d",B+i);
	vector<int> moves;
	for(int i=0;i<n;++i)
	{
		int j=i;
		for(;j<n;++j)
			if(B[j]==A[i])
				break;
		while(j!=i)
		{
			moves.push_back(j-1);
			swap(B[j-1],B[j]);
			--j;
		}
	}
	cout << moves.size() << endl;
	for(int i=0;i<moves.size();++i)
		cout << moves[i]+1 << " " << moves[i]+2 << endl;
	return 0;
}
