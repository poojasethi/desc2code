#include<iostream>
#include<stdio.h>
#include<vector>
using namespace std;
int arr[305],brr[305];
typedef pair<int,int> p;
vector<p> vec;
int main()
{
int i,j,k,n,tar;
cin>>n;
for(i=1;i<=n;i++)
cin>>arr[i];
for(i=1;i<=n;i++)
cin>>brr[i];
for(i=1;i<=n;i++)
{
tar=arr[i];
j=i;
while(j<=n && brr[j]!=tar)
j++;
while(j!=i)
{
vec.push_back(make_pair(j-1,j));
k=brr[j-1];
brr[j-1]=brr[j];
brr[j]=k;
j--;
}
}
vector<p>::iterator it;
cout<<vec.size()<<endl;
for(it=vec.begin();it!=vec.end();it++)
cout<<it->first<<" "<<it->second<<endl;
return 0;
}
