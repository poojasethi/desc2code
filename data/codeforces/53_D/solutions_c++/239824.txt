#include<iostream>
#include<vector>
#include<algorithm>
#define pb push_back
using namespace std;
int main()
{   
    int n,c=0;
    cin>>n;
    vector<int> a(n),b(n),q;
    for(int i=0;i<n;i++)
    cin>>a[i];
    for(int i=0;i<n;i++)
    cin>>b[i];
    for(int i=0;i<n;i++)
    if(a[i]!=b[i])
    for(int j=i+1;j<n;j++)
    if(a[i]==b[j]){
    int k=j;
    j=n;
    while(a[i]!=b[i]){
    c++;
    swap(b[k],b[k-1]);
    q.pb(k+1);
    k--;}}
    cout<<c<<endl;
    for(int i=0;i<c;i++)
    cout<<q[i]-1<<" "<<q[i]<<endl;
    return 0;
}
