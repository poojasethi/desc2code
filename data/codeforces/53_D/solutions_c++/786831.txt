#include <stdio.h>
#include <string.h>
int a[310],b[310],c[310];
int swa[1000010];
int main()
{
	int n,i,j,t,sum;
	while(scanf("%d",&n)!=EOF)
	{
		for(i=0;i<n;i++)
			scanf("%d",a+i);
		for(i=0;i<n;i++)
			scanf("%d",b+i);
		for(i=0;i<n;i++)
			for(j=0;j<n;j++)
				if(a[i]==b[j])  {c[j]=i;b[j]=-1;break;}
		sum=0;
		for(i=1;i<n;i++)
			for(j=i;j>0;j--)
				if(c[j]<c[j-1])
				{
					t=c[j];
					c[j]=c[j-1];
					c[j-1]=t;
					swa[sum++]=j;
				}
		printf("%d\n",sum);
		for(i=0;i<sum;i++)
			printf("%d %d\n",swa[i],swa[i]+1);
	}
	return 0;
}
