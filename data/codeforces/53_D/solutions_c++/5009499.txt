#include<iostream>
#include<vector>
using namespace std;
int a[400];
int b[400];
vector< pair<int,int> > c;
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>b[i];
	for(int i=0;i<n;i++)
		cin>>a[i];

	for(int i=0;i<n;i++)
	{
		for(int j=i;j<n;j++)
		{
			if(a[j]==b[i])
			{
				cerr<<i<<" "<<j<<endl;
				int p = j-1;
				while(p >= i){ 
				 	swap(a[p],a[p+1]);
					c.push_back(make_pair(p,p+1));
					p--;
				 }	 
				 break;
			}
		}
	}
	cout<<c.size()<<endl;
	for(int i=0;i<c.size();i++)
	cout<<c[i].first+1<<" "<<c[i].second+1<<endl;
	
}
