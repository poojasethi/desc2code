#include<cmath>
#include<cstdio>
#include<string>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;
int n,l,a[310],b[310],ans[100100];
int main()
{
	while (~scanf("%d",&n))
	{
		for(int i=0;i<n;i++)scanf("%d",b+i);
		for(int i=0;i<n;i++)scanf("%d",a+i);
		l=0;
		for(int i=0;i<n;i++)if(a[i]!=b[i]){
			int k;
			for(k=i+1;k<n;k++)if(a[k]==b[i])break;
			for(int j=k;j>i;j--){
				ans[l++]=j;
				swap(a[j],a[j-1]);
			}
		}
		printf("%d\n",l);
		for(int i=0;i<l;i++)printf("%d %d\n",ans[i],ans[i]+1);
	}
	return 0;
}
