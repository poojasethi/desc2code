#include<iostream>
#include<vector>
using namespace std;
int main()
{
int n,a[305],b[305],i,j,k=0;
vector<int> x,y;
cin>>n;
for(i=1;i<=n;i++)
{
cin>>a[i];
}
for(i=1;i<=n;i++)
{
cin>>b[i];
}
for(i=1;i<=n;i++)
{
for(j=i;j<=n;j++)
{
if(a[i]==b[j])
{
break;
}
}
while(j>i)
{
k++;
swap(b[j-1],b[j]);
x.push_back(j-1);
y.push_back(j);
j--;
}
}
cout<<k<<endl;
for(i=0;i<x.size();i++)
{
cout<<x[i]<<" "<<y[i]<<endl;
}
return 0;
}
