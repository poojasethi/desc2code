#include <iostream>
using namespace std;
int main(){
 long a[300],i,j,m,n,b[1000000][2],l[300],p=0,k,u=0;
 cin>>n;
 for(i=0;i<n;i++)cin>>a[i];
 for(i=0;i<n;i++)cin>>l[i];
 for(i=0;i<n;i++)
  if(a[i]!=l[i])
   for(j=i+1;j<n;j++)
    if(a[i]==l[j]){
     p+=j-i;
     for(k=1;k<=j-i;k++){
      m=l[j-k];l[j-k]=l[j-k+1];l[j-k+1]=m;
      b[u][1]=j-k;b[u][0]=j-k+1;u++;
     }break;

    }
 cout<<u<<endl;
 for(i=0;i<u;i++)
  cout<<b[i][1]+1<<' '<<b[i][0]+1<<endl;
}
