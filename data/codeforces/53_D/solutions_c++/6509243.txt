#include <cstdio>
#include <algorithm>
int a[301],b[301],ans[1000001][2],j,k;
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0; i<n; i++) scanf("%d",&a[i]);
    for(int i=0; i<n; i++) scanf("%d",&b[i]);
    for(int i=0; i<n; i++){
        for(j=i;a[i]!=b[j];j++);
        while(j>i){
            std::swap(b[j-1],b[j]);
            ans[k][0]=j;
            ans[k++][1]=j+1;
            j--;
        }
    }
    printf("%d\n",k);
    for(int i=0; i<k; i++)
        printf("%d %d\n",ans[i][0],ans[i][1]);
    return 0;
}
