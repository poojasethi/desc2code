#include <iostream>
#include <cstdio>

#define FOR(i,x,y) for(int i = x; i < (y); i++)
using namespace std;

int ans[1000*1000+100];
int len;

int n;
int tar[400];
int ini[400];

int main(){
	cin >> n;
	FOR(i,0,n)
		cin >> tar[i];
	FOR(i,0,n)
		cin >> ini[i];

	int cur;
	FOR(i,0,n-1){
		cur = i;
		while(ini[cur] != tar[i])
			cur++;
		while(cur != i){
			cur--;
			swap(ini[cur], ini[cur+1]);
			ans[len] = cur+1;
			len++;
		}
	}

	printf("%d\n", len);
	FOR(i,0,len)
		printf("%d %d\n", ans[i], ans[i]+1);

	return 0;
}
