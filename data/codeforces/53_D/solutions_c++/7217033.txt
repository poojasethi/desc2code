#include<iostream>
#include<vector>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n],b[n];
	vector<int> ans;
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<n;i++)cin>>b[i];
	for(int i=0;i<n;i++)
	{
		if(a[i]==b[i])continue;
		for(int j=i+1;j<n;j++)
		{
			if(a[i]==b[j])
			{
				for(int k=j-1;k>=i;k--)
				{
					ans.push_back(k);
					int tmp=b[k];
					b[k]=b[k+1];
					b[k+1]=tmp;
				}
				break;
			}
		}
	}
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++)cout<<ans[i]+1<<" "<<ans[i]+2<<endl;
}
