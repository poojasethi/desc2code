#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
int main(){
	int n, arr[5][10005];
	queue<int> q;
	scanf("%d", &n);
	for(int i=0; i<2; i++){
		for(int j=0; j<n; j++){
			scanf("%d", &arr[i][j]);
		}
	}
	for(int i=0; i<n; i++){
		if(arr[0][i]!=arr[1][i]){
			for(int j=n-1; j>i; j--){
				if(arr[0][i]==arr[1][j]){
					swap(arr[1][j-1], arr[1][j]);
					q.push(j);
				}
			}
		}
	}
	printf("%d\n", q.size());
	while(!q.empty()){
		printf("%d %d\n", q.front(), q.front()+1);
		q.pop();
	}
}
