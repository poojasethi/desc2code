#include <iostream>
#include <vector>
using namespace std;
vector <int> ans;
int main() {
	int n, a[301], b[301], i, j, k, t, m=0;
	cin>>n;
	for (int i=1; i<=n; i++) cin>>a[i];
	for (int i=1; i<=n; i++) cin>>b[i];
	for (int i=1; i<=n; i++) {
		for (int j=i; j<=n; j++) {
			if (a[i]==b[j]) {
				for (int k=j; k>i; k--) {
					ans.push_back(k-1);
					ans.push_back(k);
					b[k]=b[k-1];
					m++;
				}
				break;
			}
		}
	}
	cout<<m<<endl;
	for (int i=0; i<m; i++) cout<<ans[2*i]<<" "<<ans[2*i+1]<<endl;
	return 0;
}
