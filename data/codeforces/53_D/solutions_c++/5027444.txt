#include <iostream>
#include <vector>
#include <cstdlib>

using namespace std;

int a[305], b[305];
vector <int> vec;

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	for (int i = 0; i < n; i++)
		cin >> b[i];
	for (int i = 0; i < n; i++)
	{
		int j = i;
		for (; j < n; j++)
			if (a[i] == b[j])
				break;
		for (int k = j; k > i; k--)
		{
			vec.push_back(k);
			swap(b[k], b[k - 1]);
		}
	}
	cout << vec.size() << endl;
	for (int i = 0; i < vec.size(); i++)
		cout << vec[i] << " " << vec[i] + 1 << endl;
}
