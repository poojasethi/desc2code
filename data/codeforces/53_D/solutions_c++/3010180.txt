#include<stdio.h>
long n,i,ans,j,k,a[500],b[500],x[200000],y[200000];
int main(void)
{
  scanf("%d",&n);
  for (i=1;i<=n;i++)
    scanf("%d",&a[i]);
  for (i=1;i<=n;i++)
    scanf("%d",&b[i]);
  ans=0;
  for (i=1;i<=n-1;i++)
    if (a[i]!=b[i])
    {
      for (j=i+1;j<=n;j++)
        if (b[j]==a[i]) break;
      for (k=j;k>=i+1;k--)
        b[k]=b[k-1];
      b[i]=a[i];
      for (k=j-1;k>=i;k--)
      {
        ans++;
        x[ans]=k;
        y[ans]=k+1;
      }
    }
  printf("%d\n",ans);
  for (i=1;i<=ans;i++)
    printf("%d %d\n",x[i],y[i]);
  return 0;
}
