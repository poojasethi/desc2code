#include <cstdio>

int main (int argc, char const* argv[]) {
	
	int n, a[310], b[310], mvs[90010][2], i, j, ptr, t;
	
	ptr = 0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",a+i);
	for(i=0;i<n;i++)
		scanf("%d",b+i);
	for(i=0;i<n;i++) {
		for(j=i;j<n;j++)
			if(b[j]==a[i])
				break;
		for(;j>i;j--) {
			mvs[ptr][1] = j+1;
			mvs[ptr++][0] = j;
			t = b[j-1];
			b[j-1] = b[j];
			b[j] = t;
		}
	}
	
	printf("%d\n",ptr);
	for(i=0;i<ptr;i++) {
		printf("%d %d\n",mvs[i][0],mvs[i][1]);
	}
	
	return 0;
}
