#include<bits/stdc++.h>
using namespace std;
int main(){
    int i,j,k,x=0,t1[90000],t2[90000],n,a[300],b[300];
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++){
        scanf("%d",&b[i]);
    }
    for(i=0;i<n;i++){
        for(j=n-1;j>i;j--){
            if(a[i]==b[j]&&j!=i){
                t1[x]=j;
                t2[x]=j+1;
                x++;
                swap(b[j],b[j-1]);
            }
        }
    }
    printf("%d\n",x);
    for(i=0;i<x;i++){
        printf("%d %d\n",t1[i],t2[i]);
    }
}
