#include<iostream>
#include<algorithm>
using namespace std;

const int maxn = 321;
int a[maxn], b[maxn], c[1000000], k = 0, n;

int main(){
	cin >> n;
	for(int i = 1; i <= n; i++)
		cin >> a[i];
	for(int i = 1; i <= n; i++)
		cin >> b[i];
	
	for(int i = n; i > 1; i--){
		int j = i;
		while(b[j] != a[i])
			j--;
		for( ; j < i; j++){
			//cerr << "!" << j << " " << j+1 << endl;
			c[k++] = j;
			swap(b[j], b[j+1]);
		}
	}
	cout << k << endl;
	for(int i = 0; i < k; i++)
		cout << c[i] << " " << c[i]+1 << endl;
	return 0;
}
