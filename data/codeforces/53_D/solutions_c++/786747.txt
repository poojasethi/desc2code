#include<cstdio>
#include<algorithm>
using namespace std;
int a[305],b[305],ans[1000005];
int main()
{
    int n,i,j,l=0;
    scanf("%d",&n);
    for(i=1;i<=n;i++)scanf("%d",a+i);
    for(i=1;i<=n;i++)scanf("%d",b+i);
    for(i=1;i<=n;i++)
    {
        for(j=i;j<=n;j++)if(a[i]==b[j])break;
        for(j--; j>=i; j--)
        {
            swap(b[j],b[j+1]);
            ans[l++]=j;
        }
    }
    printf("%d\n",l);
    for(i=0;i<l;i++)printf("%d %d\n",ans[i],ans[i]+1);
}
