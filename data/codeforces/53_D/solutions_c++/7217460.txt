#include <iostream>
#include <vector> 
using namespace std;
int main()
{
	int n, a[301], b[301], i, j, k, t, m=0;
	vector <int> ans;
	cin>>n;
	for (i=1; i<=n; i++) cin>>a[i];
	for (i=1; i<=n; i++) cin>>b[i];
	for (i=1; i<=n; i++) for (j=i; j<=n; j++)
	if (a[i]==b[j]) {
		for (k=j; k>i; k--) {
			ans.push_back(k-1); ans.push_back(k);
			b[k]=b[k-1]; m++; 
		}
		break;
	}
	cout<<m<<endl;
	for (i=0; i<m; i++) cout<<ans[2*i]<<" "<<ans[2*i+1]<<endl;
	return 0;
}
