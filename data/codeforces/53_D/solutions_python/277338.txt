[n], a, b = [map (int, raw_input ().split ()) for _ in xrange (3)]
ans = list ()
for i in xrange (n):
    j = i
    while a[i] != b[j]:
        j += 1
    for k in xrange (j, i, -1):
        b[k], b[k - 1] = b[k - 1], b[k]
        ans.append ([k, k + 1])        
print len (ans)
for i, j in ans:
    print i, j
