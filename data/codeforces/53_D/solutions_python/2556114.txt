r = lambda: map(int, raw_input().split())
n = input()
b = r()
a = r()
ans = []
for i in xrange(n):
    if a[i] == b[i]: continue
    for j in xrange(n - 1, i, -1):
        if a[j] == b[i]:
            for k in xrange(j, i, -1):
                ans.append((k, k + 1))
                a[k - 1], a[k] = a[k], a[k - 1]
            break
print len(ans)
for x, y in ans:
    print x, y
