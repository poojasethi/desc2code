#!/usr/bin/python
import sys
n = int (sys.stdin.readline ())
a = [int (x) for x in sys.stdin.readline ().split ()]
assert len (a) == n
b = [int (x) for x in sys.stdin.readline ().split ()]
assert len (b) == n
ans = []
for i in range (n):
	j = i
	while b[j] != a[i]:
		j += 1
	while j > i:
		ans += [(j, j + 1)]
		b[j - 1], b[j] = b[j], b[j - 1]
		j -= 1
print len (ans)
for p, q in ans:
	print p, q
