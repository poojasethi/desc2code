res = ""
count = 0

def moveto(i, j):
    global res, count
    for k in range(i, j):
        count += 1
        res += "%d %d\n" % (k+1, k+2)

n = int(raw_input())
A = map(int, raw_input().split())
B = map(int, raw_input().split())

for i in range(n-1, -1, -1):
    if A[i] == B[i]: continue
    for j in range(i, -1, -1):
        if A[i] == B[j]:
            moveto(j, i)
            B.pop(j)
            break

print count
print res
