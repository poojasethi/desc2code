#include <iostream>
#include <cstdio> 
using namespace std;
int main()
{
	long long n,a,b=0;
	while(cin>>n)
	{
	long long ans=0,k=0;
		while(n--)
		{
			cin>>a;
			if(a==b) k++;
			else k=1;
			ans+=k;
			b=a;
			
			
		}
		
		cout<<ans<<endl;
		
		
		
		
	}
	
	
	
	return 0;
}
