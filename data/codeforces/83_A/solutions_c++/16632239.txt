
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,i;
cin>>n;

vector <LL> v(n);

for(i=0;i<n;i++)
		cin>>v[i];

LL p=v[0],s=0,c=1;
for(i=1;i<n;i++)
{if(v[i]!=p)
	{s+=c*(c+1)/2;
		c=1;}
else
	c++;
	

p=v[i];	
}

s+=c*(c+1)/2;

cout<<s<<endl;

}
