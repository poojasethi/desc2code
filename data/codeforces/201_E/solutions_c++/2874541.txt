#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int t,n,m,l,r,mid;

bool calc(int k)
{
	__int64 a=n,b=1,c=(__int64)m*k,d;
	for(int i=0;i<=k&&c>=0&&a;i++)
	{
		d=min(a,b);
		a-=d,c-=d*i;
		b=b*(k-i)/(i+1);
	}
	return c>=0&&!a;
}

int main()
{
	cin>>t;
	while(t--)
	{
		scanf("%d%d",&n,&m);
		m=min(m,n/2); 
		l=0,r=n;
		while(l<r)
		{
			mid=(l+r)>>1;
			if(calc(mid)) r=mid; else l=mid+1;
		}
		printf("%d\n",l);
	}
}
