#include <cstdio>
#include <algorithm>
using namespace std;

typedef long long LL;

bool calc(LL m, LL k, LL n)
{
    LL c = 1, t = m*k;
    LL bc = 1;
    for(int i = 1; i <= k && t >= i && c < n; ++i) {
        bc *= k-i+1;
        bc /= i;
        LL x = min(t/i, bc);
        c += x;
        t -= i*x;
    }
    return c >= n;
}

int main()
{
    int tc;
    scanf("%d", &tc);
    while(tc--) {
        LL n, m;
        scanf("%lld%lld", &n, &m);
        LL lo = 0, hi = n-1;
        while(lo < hi) {
            LL k = (lo + hi) / 2;
            if(!calc(m, k, n))
                lo = k+1;
            else hi = k; 
        }
        printf("%lld\n", lo);
    }
    return 0;
}
