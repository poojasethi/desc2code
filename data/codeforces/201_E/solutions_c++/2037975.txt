#include <stdio.h>
#include <algorithm>

using namespace std;

#define ll long long
ll t,n,m;

bool can(int x){
	ll k=n,s=x*m,p,num=1;
	for(ll i=0;i<=x;i++){
		p=min(num,k);
		s-=p*i;
		if(s<0)return 0;
		k-=p;
		if(k==0)return 1;
		num*=(x-i);
		num/=(i+1);
		
	}return 0;
}
int main(){
	scanf("%I64d",&t);while(t--){
		scanf("%I64d%I64d",&n,&m);
		ll s=0,e=n,m,r;
		while(s<=e){
			m=(s+e)/2;
			if(can(m))r=m,e=m-1;
			else s=m+1;
		}printf("%I64d\n",r);
	}
}
