#include <iostream>
#include <cmath>

using namespace std;

typedef long long int ll;

bool puede(ll n,ll m,ll k)
{
  ll num1=m*k;
  ll c=1;
  n-=c;
  if (n<=0) return true;
  for (int i=1;i<=k;i++) {
    if (num1/(k-(i-1))<c) {
      n-=num1/i;
      return n<=0;
    } else {
      c=c*(k-(i-1))/i;
      num1-=c*i;
      n-=c;
      if (n<=0) return true;
    }
  }
  return false;
}

int main()
{
  int casos;
  cin>>casos;
  for (int cas=0;cas<casos;cas++) {
    ll n,m;
    cin>>n>>m;
    ll inf=-1;
    ll sup=n;
    while (inf+1<sup) {
      ll med=(inf+sup)/2;
      if (puede(n,m,med))
	sup=med;
      else
	inf=med;
    }
    cout<<sup<<endl;
  }
}
