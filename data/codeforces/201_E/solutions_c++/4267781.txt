#include <algorithm>
#define LL long long
using namespace std;
LL n,m;
bool check(LL k)
{
	LL ret=1;
	LL T=1,tot=k*m;
	for(int i=1;i<=k;i++)
	{
		T=T*(k-i+1)/i;
		if(T*i<=tot)
		{
			tot-=T*i;
			ret+=T;
			if(ret>=n)
				return 1;
		}
		else
		{
			ret+=tot/i;
			break;
		}
	}
	return ret>=n;
}
int main()
{
	int T;
	scanf("%d",&T);
	for(;T--;)
	{
		scanf("%I64d%I64d",&n,&m);
		LL l=0,r=n;
		for(;l<r;)
		{
			LL mid=l+r>>1;
			if(check(mid))
				r=mid;
			else
				l=mid+1;
		}
		printf("%I64d\n",r);
	}
	return 0;
}
