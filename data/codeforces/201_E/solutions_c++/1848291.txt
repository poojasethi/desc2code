#include<iostream>
using namespace std;
typedef long long ll; 
int n,m; 

int main()
{
	#ifndef ONLINE_JUDGE 
		freopen("A.in","r",stdin);
		freopen("A.out","w",stdout); 
	#endif 
	int t;
	cin>>t;
	for(;t--;)
	{
		cin>>n>>m; 
		if(n==1)
		{
			cout<<0<<endl; 
			continue; 
		} 
		int ansl=0,ansr=n,ans=(ansl+ansr)/2;
		for(;ansl<ansr;ans=(ansl+ansr)/2)
		{
			int tmp=n;
			ll C=1,all=0; 
			for(int i=0;i<=ans;++i) 
			{
				all+=(ll)min((ll)tmp,C)*i; 
				tmp-=min((ll)tmp,C); 
				if(!tmp)break;
				C*=ans-i;
				C/=i+1; 
			}
			if(!tmp)
				if(all>(ll)m*ans) ansl=ans+1;
				else ansr=ans; 
			else ansl=ans+1; 	  
		} 
		cout<<ans<<endl; 
	} 
	return 0; 
}
