#include<bits/stdc++.h>
using namespace std;

int nq,n,m,ans;
long long inf=7e18;

long long check(int k)
{int i;
 long long x,ret=0,s=(long long)m*k;

 x=1;
 for(i=0;i<=k;i++)
 {	if(i) x=x*(k-i+1)/i;
 	if(x<inf && (long long)x*i<=s)
 	{	ret+=x;
 		s-=(long long)x*i;
 	}
 	else
 	{	ret+=s/i;
 		return ret;
 	}
 	if(ret>n) return ret;
 }
 return ret;
}

int main()
{int i,j,q,l,r,mid;
 
 scanf("%d",&nq);
 for(q=1;q<=nq;q++)
 {	scanf("%d%d",&n,&m);
 	l=0;
 	r=n;
 	ans=n+1;
 	while(l<=r)
 	{	mid=(l+r)/2;
 		if(check(mid)>=n)
 		{	ans=min(ans,mid);
 			r=mid-1;
		}
		else
			l=mid+1;
	}
	printf("%d\n",ans);
 }
 return 0;
}
