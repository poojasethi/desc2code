#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

long long n,m;

bool check(long long mid)
{
	long long tmp = 1,tot = mid * m,sum = 1;
	for(long long i = 1; i <= mid && tot && sum < n; i ++) {
		tmp = tmp * (mid - i + 1) / i;
		if(tot / i > tmp) {
			sum += tmp;
			tot -= tmp * i;
		}
		else {
			sum += tot / i;
			tot = 0;
		}
	}
	return sum >= n;
}

int main()
{
	int t;
	scanf("%d",&t);
	while(t --) {
		scanf("%I64d%I64d",&n,&m);
		long long lt = 0,rt = n,mid,ans = 0;
		while(lt <= rt) {
			mid = (lt + rt) / 2;
			if(check(mid)) ans = mid,rt = mid - 1;
			else lt = mid + 1;
		}
		cout << ans << endl;
	}
	return 0;
}
