#include<cstdio>
#include<algorithm>
#define LL long long
using namespace std;
LL n,m,l,r,mid,T,ss;
bool calc(LL k,LL n)
{
    if(--n<=0)return 1;
    LL now=1,L=k*m,l;
    for(LL i=1;i<=k;++i)
    {
        now=now*(k-i+1)/i;
        l=min(now,L/i);
        n-=l,L-=l*i;
        if(now>l)break;
        if(n<=0)return 1;
    }
    return n<=0;
}
int main()
{
    scanf("%I64d",&T);
    while(T--)
    {
        scanf("%I64d%I64d",&n,&m);
        l=0,r=n-1;
        while(l<r)
        {
            mid=(l+r)/2;
            if(calc(mid,n))r=mid;
            else l=mid+1;
        }
        printf("%I64d\n",l);
    }
    return 0;
}
