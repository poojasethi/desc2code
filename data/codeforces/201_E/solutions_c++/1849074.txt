#include <iostream>
#include <cstdio>
using namespace std;
typedef long long LL;

LL c,sum,cnt,m,Left,Right,mid,ans;
int T,n;

bool can(LL x)
{
	if (x==0) return 1>=n;
	c=1;sum=0;cnt=0;int i;
	for (i=0;cnt<=x*m && i<=x;i++)
	{
		sum+=c;if (sum>=n) return true;
		c=c*(x-i)/(i+1);
		cnt+=(i+1)*c;
	}
	cnt-=i*c;
	return (sum+(x*m-cnt)/i>=n);
}

LL work()
{
	Left=0;Right=n;ans=n;
	while (Left<=Right)
	{
		mid=(Left+Right)/2;
		if (can(mid)) Right=mid-1,ans=mid;
		else Left=mid+1;
	}
	return ans;
}

int main()
{
//	freopen("e.in","r",stdin);freopen("e.out","w",stdout);
	
	for (scanf("%d",&T);T;T--)
	{
		scanf("%d%I64d",&n,&m);m=min(m,LL(n/2));
		printf("%I64d\n",work());
	}
	
	return 0;
}
