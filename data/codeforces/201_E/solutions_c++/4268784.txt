#include<cstdio>
using namespace std;
int n,m;
long long check(long long k)
{	long long res,c,s;
	int i;
	res=c=1,s=k*m;
	for (i=1;i<=k;++i)
	{	c=c*(k-i+1)/i;
		if (c*i<=s)
		{	s-=c*i,res+=c;
			if (res>=n)
				return 1;
		}
		else
		{	res+=s/i;
			break ;
		}
	}
	return res>=n;
}
int work()
{	long long l,r,mid;
	l=0,r=n;
	while (l<r)
	{	mid=(l+r)/2;
		if (check(mid))
			r=mid;
		else l=mid+1;
	}
	return l;
}
int main()
{	int t;
	scanf("%d",&t);
	while (t--)
	{	scanf("%d%d",&n,&m);
		printf("%d\n",work());
	}
	return 0;
}
