#include<cstdio>
#define ll long long
ll min(ll a,ll b){return a<b?a:b;}
bool check(int n,int m,int mid)
{
    ll res=(ll)m*mid,z=1;
    for (int i=0;i<=mid && n;i++){
        ll k=min(z,n); z=z*(mid-i)/(i+1);
        res-=k*i; n-=k;
        }
    return res>=0 && !n;
}
int cal(int n, int m)
{
    int l=0,r=n-1,mid; m=min(m,n/2);
    while (l<r){
        mid=(l+r)/2;
        if (check(n,m,mid)) r=mid; else l=mid+1;
        }
    return r;
}
int main()
{
    int t,n,m; scanf("%d",&t);
    while (t--) scanf("%d%d",&n,&m),printf("%d\n",cal(n,m));
    return 0;
}
