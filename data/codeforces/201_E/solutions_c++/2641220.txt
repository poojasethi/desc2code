#include<stdio.h>
#include<algorithm>
using namespace std;
typedef long long ll;
int n,m;
ll calc(int k)
{
	ll t=(ll)m*k;
	ll ret=1,last=1;
	for(int i=1;i<=k;i++)
	{
		ll now=last*(k-i+1)/i,x=min(t/i,now<0?t:now);
		ret+=x;
		t-=x*i;
		last=now;
		if(ret>=n||ret<0)return n;
		if(!t)break;
		if(!x)break;
	}
	return ret;
}
int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d%d",&n,&m);
		int l=-1,r=n;
		while(l+1!=r)
		{
			int mid=l+((r-l)>>1);
			if(calc(mid)>=n)r=mid;
			else l=mid;
		}
		printf("%d\n",r);
	}
	return 0;
}
