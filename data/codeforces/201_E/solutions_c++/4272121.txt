#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
typedef long long LL;
LL n,m;
bool check(LL k)
{
	LL ans=1,C=1,s=k*m;
	for(int i=1;i<=k;++i)
	{
		C=C*(k-i+1)/i;
		if(C*i<=s)
		{
			s=s-C*i,ans+=C;
			if(ans>=n)
				return true;
		}
		else
		{
			ans=ans+s/i;
			break;
		}
	}
	return ans>=n;
}
LL binary(LL l,LL r)
{
	while(l<r)
	{
		LL mid=(l+r)>>1;
		check(mid)==false?l=mid+1:r=mid;
	}
	return l;
}
int main()
{
	LL test;
	for(cin>>test;test;--test)
	{
		cin>>n>>m;
		cout<<binary(0,n)<<endl;
	}
	return 0;
}
