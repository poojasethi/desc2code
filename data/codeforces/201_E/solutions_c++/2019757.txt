#include <iostream>
#include <cstdio>
#include <cstring>
#define FOR(i, n) for (int i = 0; i < (n); ++i)
using namespace std;

int t, n, m;

bool can(const int &n, const int &m, const int &k) {
	if (k <= 30 && (1LL << k) < n) return false;
	
	long long total = 0, choose = 1, ones = 0;
	for (int i = 0; i <= k && total < n; ++i) {
		long long add = min(n - total, choose);
		total += add; ones += add * i;
		choose = choose * (long long) (k - i) / (i + 1);
	}
	if (total < n) return false;
	return ones <= (long long) m * k;
}

int main() {
	for (scanf("%d", &t); t--; ) {
		scanf("%d%d", &n, &m);
		int s = 0, e = n;
		while (s < e) {
			int k = (s + e) / 2;
			if (can(n, m, k)) e = k;
			else s = k + 1;
		}
		printf("%d\n", e);
	}
	return 0;
}
