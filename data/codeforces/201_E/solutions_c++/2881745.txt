#include<cstdio>
#define ll long long

int TestCase,n,m,l,r;

ll Check(int K)
{
	ll T(1ll*m*K),Now(1),Cki(1);
	for(int i=1;i<=K;i++)
	{
		if(Cki>(T/i)*i/(K-i+1))return Now+T/i;
		Cki=Cki*(K-i+1)/i;
		if(Cki>(T/i))return Now+T/i;
		T-=Cki*i;
		Now+=Cki;
	}
	return Now;
}

int main()
{
	for(scanf("%d",&TestCase);TestCase--;)
	{
		scanf("%d%d",&n,&m);
		if(m>n/2)m=n/2;
		for(l=-1,r=n;l+1<r;)
			if(Check((l+r)/2)>=n)r=(l+r)/2;
				else l=(l+r)/2;
		printf("%d\n",r);
	}
	return 0;
}
