#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define inf 1000000000
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,T;
inline int64 cal(int k) {
	int i,mm=m;
	int64 an=0,C=1;
	For(i,0,k-1) {
		if (C<=mm) {
			mm-=C;
			an+=C*k/(i+1);
			if (an>inf) return an;
		} else return an+1ll*mm*k/(i+1);
		C=C*(k-1-i)/(i+1);
	}
	return an;
}
int main() {
	scanf("%d",&T);
	for (;T;T--) {
		scanf("%d%d",&n,&m); n--;
		int l=0,r=n,mid;
		for (;l<=r;) {
			mid=(l+r)/2;
			if (cal(mid)>=n) r=mid-1;
			else l=mid+1;
		}
		printf("%d\n",r+1);
	}
	return 0;
}
