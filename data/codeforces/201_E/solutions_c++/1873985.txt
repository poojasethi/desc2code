#include<cstdio>
#include<cstring>
#include<iostream>

using namespace std;
typedef long long LL; 

int		n,m;

bool	pan(int k ) 
{
	LL	C = 1, num = 1 , t = (LL)k*m;
	for (int i = 1; i<=k && t; i++ ) 
	{
		C = C*(k-i+1)/i;
		if ( t/i > C ) {
			num += C;
			t -= C*i;
		}
		else num += t/i , t = 0; 
//		printf("C %lld num %lld t %lld\n" , C , num , t ); 
		if ( num >= n ) return true; 
	}
	return num >= n; 
}
int		main()
{
	int	cas;
	scanf("%d", &cas );
	while ( cas -- )
	{
		scanf("%d%d", &n, &m );
		int g = 0 , h = n , ans = 0; 
		while ( g <= h ) 
		{
			int mid = (g+h)/2;
//			printf("%d %d %d\n" , g , h,  mid );
			if ( pan(mid) ) {
				ans = mid; 
				h = mid-1; 
			}
			else g = mid+1;
		}
		printf("%d\n" , ans );
	}
	return 0;
}
