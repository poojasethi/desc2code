#include<cstdio>
int T,n,m;

bool check(long long k)
{
    long long tot=m*k,res=1,x=1;
    for (int i=1; i<=k; i++)
    {
        x=x*(k-i+1)/i;
        if (i*x<=tot)
            tot-=i*x,res+=x;
        else
        {
            res+=tot/i; break;
        }
        if (res>=n)
            return true;
    }
    return res>=n;
}

int main()
{
    scanf("%d",&T);
    for (int i=1; i<=T; i++)
    {
        scanf("%d%d",&n,&m);
        long long l=0,r=n-1,mid;
        while (l<r-1)
        {
            mid=(l+r)/2;
            if (check(mid))
                r=mid;
            else
                l=mid;
        }
        printf("%I64d\n",r);
    }
    return 0;
}
