#include<cstdio>
int T,N,M,i,l,r,mid;
bool Check(int mid)
{
	int i;
	long long res=0,Z=1,size=(long long)mid*(long long)M;
	for(i=0;i<=mid;++i)
	{
		if(Z>N-res) Z=N-res;
		res+=Z;
		size-=i*Z;
		if(res==N) break;
		Z=Z*(mid-i)/(i+1);
	}
	if(res==N&&size>=0) return 1;
	return 0;
}
int main()
{
	for(scanf("%d",&T);T--;)
	{
		scanf("%d%d",&N,&M);
		for(l=0,r=N;l<r;)
		{
			mid=(l+r)/2;
			if(Check(mid)) r=mid;
			else l=mid+1;
		}
		mid=(l+r)/2;
		printf("%d\n",mid);
	}
	return 0;
}
