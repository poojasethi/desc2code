
#include<cstdio>
#include<algorithm>
#define For(i,n) for(int i=1;i<=n;i++)
using namespace std;
typedef long long ll;
ll inf=ll(1e18),cnt,s,p;
double tmp;
int T,n,m,l,mid,r;
ll C(int n,int k)
{
	tmp=1;
	For(i,k) tmp*=n-i+1,tmp/=i;
	if (tmp>inf) return inf;
            else return ll(tmp);
}
bool ok(int k)
{
	s=ll(k)*m;cnt=0;
	For(i,k)
	{
		if (s<i) break;
		cnt+=p=min(s/i,C(k,i));
		s-=p*i;
	}
	return cnt+1>=n;
}
