#include<cstdio>
typedef long long ll;
int n,m,l,r;
bool check(ll a){
	ll c=1,i,t=a*m,d=n;
	for (i=1;i<=a;i++){
		if (i<=t/(c=c*(a-i+1)/i)) t-=i*c,d-=c; else{
			d-=t/i;
			break;
		}
	}
	return d<=1;
}
int main(){
	int t;
	for (scanf("%d",&t);t;t--){
		for (l=0,r=(scanf("%d%d",&n,&m),n);l<r;)
			check(l+r>>1)?r=l+r>>1:l=l+r+2>>1;
		printf("%d\n",l);
	}
	return 0;
}
