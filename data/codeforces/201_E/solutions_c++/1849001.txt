#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
#define fo(i,a,b) for(int i=a;i<=b;i++)
using namespace std;
typedef long long LL;
int cas,l,r,mid,n,m;
long long calc(LL n,LL m){
	LL tmp=1;
	fo(i,1,m){
		tmp=tmp*(n-i+1)/i;
	}
	return tmp;
}
bool ch(int k,int n,int m){
	LL t=(long long)k*m;
	LL ans=1;
	fo(i,1,k){
		LL cc=calc(k,i);
		if (t/i>=cc){
			ans=ans+cc;
			t=t-cc*i;
		}else {
			ans=ans+t/i;
			break;
		}
		if (ans>=n) return 1;
	}
	if (ans>=n) return 1;
	else return 0;
}
int main(){
	//freopen("a.in","r",stdin);
	cin>>cas;
	fo(i,1,cas) {
		scanf("%d%d",&n,&m);
		l=0;
		r=n;
		do{
			mid=(l+r)/2;
			if (ch(mid,n,m)) r=mid;
			else l=mid+1;
		}while (l<r);
		cout<<l<<endl;
	}
}
