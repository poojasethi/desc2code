#include<cstdio>
#include<iostream>
using namespace std;
typedef long long LL;
LL T,n,m,C,t,ans;
bool check(LL k)
{
	C=1,t=k*m,ans=1;
	for(int i=1;i<=k;i++)
	{
		C=C*(k-i+1)/i;
		if(C*i<=t)
		{
			t=t-C*i,ans=ans+C;
			if(ans>=n)
				return true;
		}
		else
		{
			ans=ans+t/i;
			break;
		}
	}
	return ans>=n;
}
int main()
{
	cin>>T;
	while(T--)
	{
		cin>>n>>m;	
		LL l=0,r=n,mid;
		while(l<r)
		{
			mid=(l+r)>>1;
			if(check(mid)) r=mid;
			else l=mid+1;	
		}
		cout<<l<<endl;
	}
	return 0;
}
