#include <bits/stdc++.h>
using namespace std;

const int M = 300001;

int a[M], b[M];
char s[M];

int main() {
    //freopen("E.in", "r", stdin);
    int n, k;
    scanf("%d %d", &n, &k);
	  int cnt = 0, nxt = 0;
    for(int i = 0; i <= n; i ++) {
        scanf("%s", s);
        if(s[0] == '?') b[i] = 1, cnt ++;
        else a[i] = atoi(s), nxt = 1 - nxt;
    }
    if(k == 0) {
      if(b[0] == 1) {
          if(!nxt) return puts("No");
          return puts("Yes");
      }
      if(a[0] == 0) return puts("Yes");
      return puts("No");
    }
    if(!cnt) {
        int x = 0;
        for(int i = 0; i <= n; i ++) {
                x += a[i];
                if(x % k) return puts("No");
                x /= k;
        }
        if(x) return puts("No");
        return puts("Yes");
    }
    if((cnt + nxt) % 2) return puts("No");
    return puts("Yes");
}
