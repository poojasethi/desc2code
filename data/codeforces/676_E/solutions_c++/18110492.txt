#include <bits/stdc++.h>

using namespace std;

const int N = 100010;
int a[N], isset[N];
typedef long long ll;
const ll MOD = 999999733;

ll fpow(ll x, ll y)
{
	if (y == 0)
		return 1LL;

	if (y%2)
		return ((x%MOD)*(fpow(x, y-1)%MOD))%MOD;
	
	ll tmp = fpow(x, y/2)%MOD;
	return (tmp*tmp)%MOD;
}

int main()
{
	int n, k, gone = 0;
	scanf("%d %d", &n, &k);

	n++;
	for (int i = 0; i < n; i++)
	{
		char cur[10];
		scanf("%s", cur);
		if (cur[0] != '?')
		{
			a[i] = atoi(cur);
			gone++;
			isset[i] = 1;
		}
	}

	bool ans;

	if (gone == n)
	{
		ll pk = 0;
		for (int i = 0; i < n; i++)
		{
			ll cur = (MOD + (a[i]*fpow(k, (ll)i))%MOD)%MOD;
			pk = (pk + cur)%MOD;
		}
		if (pk == 0)
			ans = true;
		else
			ans = false;
	}
	else if (k == 0)
	{
		if (isset[0])
		{
			if (a[0] == 0)
				ans = true;
			else
				ans = false;
		}
		else
		{
			if (gone%2 == 1)
				ans = true;
			else
				ans = false;
		}
	}
	else
	{
		if (n%2 == 0)
			ans = true;
		else
			ans = false;
	}
	printf("%s\n", ans ? "Yes" : "No");
	return 0;
}
