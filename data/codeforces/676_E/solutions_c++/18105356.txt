#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<queue>
#include<vector>
using namespace std;

const int p[10]={11497667,11686547,29310161,46560473,46309231,86173201,99457471,84031069,18773669,53421283};
char s[100010][10];
int a[100010],n,k,cnt;

void solve(){
	bool flag=true;
	for(int i=0;i<=n;i++) sscanf(s[i],"%d",a+i);
	for(int i=0;i<10;i++){
		int x=a[n];
		for(int j=n-1;~j;j--) x=(1ll*k*x+a[j])%p[i];
		if(x!=0) flag=false;
	}
	if(flag) puts("Yes");
	else puts("No");
}
int main(){
	scanf("%d%d",&n,&k);
	for(int i=0;i<=n;i++) scanf("%s",s[i]);
	for(int i=0;i<=n;i++) cnt+=s[i][0]=='?';
	if(k==0){
		if(s[0][0]=='?'){
			if((n+1-cnt)&1) puts("Yes");
			else puts("No");
			return 0;
		}
		if(s[0][0]=='0') puts("Yes");
		else puts("No");
		return 0;
	}
	if(cnt==0) {solve(); return 0;}
	if(n&1) puts("Yes");
	else puts("No");
	return 0;
}
