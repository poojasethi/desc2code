#include <bits/stdc++.h>

using namespace std;

typedef long long LL;
#define FOR(i,x,y)  for(int i = x;i < y;++ i)
#define IFOR(i,x,y) for(int i = x;i > y;-- i)

const int maxn = 100010;
const int inf = 1e9+7;
const int maxm = 10;
int a[maxn],mark[maxn],n,k,sum;
char str[maxn];

bool work(){
    sum = 0;
    memset(mark,1,sizeof(mark));
    FOR(i,0,n+1){
        scanf("%s",str);
        if(strcmp(str,"?") == 0)    mark[i] = 0,sum ++;
        else    a[i] = atoi(str);
    }
    if(k == 0){
        if(mark[0]) return a[0] == 0;
        return (n+1-sum)%2;
    }
    else if(sum == 0){
        LL ans = a[n];
        IFOR(i,n-1,-1){
            ans = ans*k+a[i];
            if(abs(ans) > inf)  return false;    
        }
        return ans == 0;
    }
    return (n+1)%2 == 0;
}

int main(){
    while(~scanf("%d%d",&n,&k)){
        if(work())  printf("Yes\n");
        else    printf("No\n");
    }
    return 0;
}
