#include <bits/stdc++.h>

using namespace std;
#define maxn 100005
typedef long long ll;
int n, k;
char s[7];
int polin[maxn];
bool b[maxn];

int main() {
	cin >> n >> k;

	int cantQuestMark = 0;

	for (int i = 0; i <= n; ++i) {
		cin >> s;
		if (s[0] != '?') {
			b[i] = true;
			polin[i] = atoi(s);
		} else
			cantQuestMark++;
	}

	if (k == 0) {
		if (b[0])
			cout << (polin[0] ? "No" : "Yes") << endl;
		else
			cout << ((n + 1 - cantQuestMark) % 2 ? "Yes" : "No") << endl;
	} else {
		if (cantQuestMark)
			cout << (n % 2 ? "Yes" : "No") << endl;
		else {
			ll sum = 0;
			for (int i = 0; i <= n; ++i) {
				sum += polin[i];
				if (sum % k != 0) {
					cout << "No" << endl;
					return 0;
				}
				sum /= k;
			}
			if(!sum)
			    cout << "Yes" << endl;
			else
			     cout << "No" << endl;
		}
	}

	return 0;
}
