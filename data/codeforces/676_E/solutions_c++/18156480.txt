#include<bits/stdc++.h>
#define FOR(i,l,r) for(int i=l;i<=r;i++)
#define ll long long
#define M 100010
#define priy printf("Yes\n")
#define prin printf("No\n")
using namespace std;
ll n,k,tmp,f[5][M],a[M];
ll const p[5]={0,998244353,100000007,10657601,299999813};
ll read(){
	char ch=getchar(); ll x=0,f=1;
	while(ch<'0'||ch>'9'){
		if(ch=='-')f=-1;
		if(ch=='?'){tmp++; return p[1];}
		ch=getchar();
	}
	while(ch>='0'&&ch<='9'){x=x*10+ch-'0'; ch=getchar();}
	return x*f; 
}
int main(){
	n=read(); k=read();
	FOR(i,0,n)a[i]=read(); int fl=1;
	if(!k){if(!a[0]||(a[0]==p[1]&&tmp&&(n+1-tmp)&1))priy; else prin; return 0;}
	if(tmp){if(n&1)priy; else prin; return 0;}
	FOR(i,1,4){
		f[i][0]=1; ll kk=(k<0)?p[i]+k:k;
		FOR(j,1,n)f[i][j]=f[i][j-1]*kk%p[i];
	}
	FOR(i,1,4){
		ll t=0; 
		FOR(j,0,n){
			ll kk=(a[j]<0)?p[i]+a[j]:a[j];
			t=(t+kk*f[i][j])%p[i];
		}
		if(t)fl=0;
	}
	if(fl)priy; else prin;
}
