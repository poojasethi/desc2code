#include<cmath>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>

#ifdef WIN32
#define orz "%I64d"
#else
#define orz "%lld"
#endif
#define N 100010
#define mod ((479<<21)-1)

using namespace std;
typedef long long LL;
int a[N],n,k,cnt;
bool ok=true;
int main()
{
	scanf("%d%d",&n,&k);
	for (int i=0;i<=n;i++)
	{
		static char s[10];
		scanf("%s",s+1);
		if (s[1]=='?')
		{
			cnt++;
			if (i==0)
				ok=false;
		}
		else
		{
			int now=1,x=1,l=strlen(s+1);
			if (s[1]=='-')
				x=-1,now++;
			for (int j=now;j<=l;j++)
				a[i]=a[i]*10+s[j]-'0';
			a[i]=a[i]*x;
		}
	}
	if (k==0)
	{
		if (ok)
			return printf(a[0]==0?"Yes":"No"),0;
		else
		{
			int t=n+1-cnt;
			return printf(t&1?"Yes":"No"),0;
		}
	}
	if (cnt==0)
	{
		LL res=0;
		for (int i=n;i>=0;i--)
			res=((res*k)%mod+a[i])%mod;
		if (res==0)
			printf("Yes\n");
		else
			printf("No\n");
		return 0;
	}
	printf((n&1)?"Yes":"No");
	return 0;
}
