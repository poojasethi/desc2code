#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;

const int INF=0x3f3f3f3f;
const int maxn=100010;

int n,k;
int a[maxn];

bool checkrt(int* a,int n,int k){
	long long s=0;
	for(int i=0;i<n;++i){
		s=s/k+a[i];
		if(s%k!=0)return 0;
	}
	return s==0;
}

void solve(){
	scanf("%d%d",&n,&k);++n;
	int t=0;
	for(int i=0;i<n;++i){
		char s[10];scanf("%s",s);
		if(s[0]=='?')a[i]=INF;else a[i]=atoi(s),++t;
	}
	if(k==0){
		if(a[0]==INF)puts(((t+1)&1) ? "No" : "Yes");else puts(a[0]==0 ? "Yes" : "No");
	}else{
		if(t==n)puts(checkrt(a,n,k) ? "Yes" : "No");else puts((n&1) ? "No" : "Yes");
	}
}

int main(){
//	freopen("in.txt","r",stdin);
	solve();
	return 0;
}
