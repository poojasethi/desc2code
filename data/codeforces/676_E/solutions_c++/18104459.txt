#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#include<vector>
#include<cmath>
#define ll long long
#define ld long double
using namespace std;
const ll maxn= 5e5,oo= 1e9,mo= 999999929;
char s[100];
ll a[maxn];
ll n,k,i,j,cnt,bl;
ll num(char *s){
	ll i,x= 0;
	if (s[1]=='-'){
		for (i= 2;s[i];i++)
			x= x*10+s[i]-'0';
		return -x;
	}else {
		for (i= 1;s[i];i++)
			x= x*10+s[i]-'0';
		return x;
	}
}
int main()
{
#ifndef ONLINE_JUDGE	
	freopen("1.in","r",stdin);
	freopen("1.out","w",stdout);
#endif
	scanf("%I64d%I64d",&n, &k);
	for (i= 0;i<=n;i++){
		scanf("%s",s+1);
		if (s[1]=='?') a[i]= oo;
		else {
			a[i]= num(s);
			cnt++;
		}
	}
	if (k==0){
		if (a[0]==oo){
			bl= cnt%2;
		}else {
			bl= !a[0];
		}
	}else {
		if (cnt==n+1){
			ll x= 0;
			for (i= n;i>=0;i--)
				x= (x*k+a[i])%mo;
			if (x==0) bl= 1;
			else bl= 0;
		}else {
			n++;
			if (n%2) bl= 0;
			else bl= 1;
		}
	}
	if (bl) printf("Yes");
	else printf("No");
	return 0;
}
