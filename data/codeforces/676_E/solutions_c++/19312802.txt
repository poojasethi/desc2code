#include <algorithm>
#include <cmath>
#include <cstring>
#include <cassert>
#include <cstdio>
#include <iostream>

using namespace std;

int a[100005];

int main(){
	int w=-1,n,k,s=0;
	scanf("%d%d",&n,&k);getchar();
	for(int i=0;i<n+1;++i){
		char c=getchar();
		if(c=='?'){
			getchar();
			s++;
		}
		else {
			int t=1;
			if(c=='-')t=-1,c=getchar();
			while(c!='\n'){
				a[i]*=10;a[i]+=c-'0';
				c=getchar();
			}
			a[i]*=t;
			if(!i)w=a[i]!=0;
		}
	}
	if (w==0 && k == 0) {
		puts("Yes");
		return 0;
	}
	
	if(!s){
		if(!k)puts(w?"No":"Yes");
		else {
			int b=0;bool r=true;
			for(int i=0;r&&i<n;++i){
				r=(b-a[i])%k==0;
				b=(b-a[i])/k;
			}
			r=r&&b==a[n];
			puts(r?"Yes":"No");
		}
	}
	else if(!k&&(w>0))puts("No");
	else if (k) puts((n+1)%2?"No":"Yes");
	else puts((n+1-s)%2?"Yes":"No");
	return 0;
}
