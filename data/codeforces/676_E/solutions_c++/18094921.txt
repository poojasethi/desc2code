#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <vector>
#include <queue>
using namespace std;
typedef long long ll;
const int N=1E5+5;
const ll inf=1E14;
int n,k,tot;
ll b[N],a[N];

inline ll read()
{
    int x=0,f=1;char ch=getchar();
    while(ch>'9'||ch<'0')
	{
		if(ch=='-') f=-1;
		if (ch=='?')
		{
			tot++;
			return inf;
		}
		ch=getchar();
	}
    while(ch>='0'&&ch<='9'){x=x*10+ch-'0';ch=getchar();}
    return x*f;
}

void print(bool o)
{
	if (!o) cout<<"No";else cout<<"Yes";
	exit(0);
}

int main()
{
	cin>>n>>k;
	for (int i=0;i<=n;i++) 
		a[i]=read();
	if (!k)
	{
		if (!a[0]) print(1); else if (a[0]==inf)
		{
			print(((n-tot)%2==0));
		} else print(0);		
	}
	if (tot)
	{
		if (n%2==0) cout<<"No"; else cout<<"Yes";
		return 0;
	}
	b[n-1]=a[n];
	for (int i=n-1;i;i--)
	{
		b[i-1]=b[i]*k+a[i];
		if (abs(b[i-1])>inf) print(0);
	}
	b[0]*=k;
	if (b[0]!=-a[0]) cout<<"No"; else cout<<"Yes";
}
