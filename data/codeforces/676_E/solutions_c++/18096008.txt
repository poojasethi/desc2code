#include <bits/stdc++.h>
#define FOR(i, l, r) for (int i = int(l); i < int(r); i++)
#define FST first
#define SCD second
#define modp 1000000007
#define MAXN 100113
using namespace std;

typedef long long LL;
typedef unsigned int UI;

int N, k, a[MAXN], t, f, l;
bool solve()
{
    if (k == 0) {
        if (a[0] == 0 || a[0] == 12345 && f == 1) {
            return true;
        }
        return false;
    }
    if (t != 0) {
        return l == 1;
    }
    FOR(i, 0, N) {
        if (a[i] % k != 0) {
            return false;
        }
        a[i] /= k;
        a[i + 1] -= a[i];
    }
    return a[N] == 0;
}
int main()
{
    scanf("%d%d", &N, &k);
    k = -k;
    FOR(i, 0, N + 1) {
        int x;
        if (scanf("%d", &x) != 1) {
            scanf(" ?");
            x = 12345;
            t++;
        }
        a[i] = x;
    }
    f = (N - t + 1) % 2;
    l = N % 2;
    if (solve()) {
        printf("Yes\n");
    }
    else {
        printf("No\n");
    }
    return 0;
}
