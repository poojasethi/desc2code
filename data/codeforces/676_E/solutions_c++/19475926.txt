#include <iostream>
#include <stdio.h>
#define mod1 100003619
#define mod2 19991105
#define MAXN 100010

using namespace std;

int n , k , vis[ MAXN ] , v[ MAXN ] , cnt;
char s[10];
long long sum1 , sum2;

int main()
{
	cin >> n >> k;
	for( register int i = 0 ; i <= n ; i++ )
	{
		scanf( "%s" , s );
		if( s[0] != '?' ) cnt += ( vis[i] = 1 ) , sscanf( s , "%d" , &v[i] );
	}
	for( register int i = n ; ~i ; i-- )
		sum1 = ( sum1 * k + v[i] ) % mod1 + mod1;
	sum1 %= mod1;
	for( register int i = n ; ~i ; i-- )
		sum2 = ( sum2 * k + v[i] ) % mod2 + mod2;
	sum2 %= mod2;
	if( !k )
		if( !vis[0] )
			if( cnt & 1 ) puts( "Yes" );
			else puts( "No" );
		else
			if( v[0] ) puts( "No" );
			else puts( "Yes" );
	else
		if( cnt != n + 1 )
			if( n & 1 ) puts( "Yes" );
			else puts( "No" );
		else
			if( !sum1 && !sum2 ) puts( "Yes" );
			else puts( "No" );
	return 0;
}
