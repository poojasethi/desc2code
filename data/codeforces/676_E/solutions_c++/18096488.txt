#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <ctime>
#include <cstring>
#include <cmath>
using namespace std;
typedef long long ll;
const int N = 101000;
int read(){
	int x = 0,f = 1;char ch = getchar();
	while (ch < '0' || '9' < ch){
		if (ch == '-') f = -1;
		if (ch == '?') return 20000;
		ch = getchar();
	}
	while ('0' <= ch && ch <= '9') x = x * 10 + ch - '0',ch = getchar();
	return x*f;
};
int n,k,a[N],cnt;
bool solve();
int main(){
	srand(time(NULL));
	scanf("%d%d",&n,&k);
	for (int i = 0;i <= n;i++) {
		a[i] = read();
		if (a[i] == 20000 ) cnt++;
	}
	if (solve())printf("Yes\n");else printf("No\n");
	
	return 0;
}
bool solve(){
	if (k == 0){
		if (a[0] == 20000) return (n+1-cnt)&1;
		return a[0] == 0;
	}
	
	if (cnt) return (n)&1;
	double ans = 0;
	for (int i = n;i >= 0;i--)
		 ans = ans * k + a[i];
	if (fabs(ans) < 1e-6) return 1;else return 0;	
}
