#include<bits/stdc++.h>
using namespace std;
int n,k;
int a[100005];
bool vis[100005];
typedef long long ll;
int main()
{
    cin>>n>>k;
    char s[55];
    int num=0,num1=0;
    memset(vis,0,sizeof vis);
    int last=0;
    for(int i=0;i<=n;++i){
        scanf("%s",s);
        if(s[0]=='?'){vis[i]=1;++num1;}
        else{
            stringstream ss(s);
            ss>>a[i];
            ++num;
        }
    }
    if(k==0){
        if((!vis[0]&&a[0]==0)||(num%2&&vis[0]))cout<<"Yes\n";
        else cout<<"No\n";
        return 0;
    }
    if(((n+1)&1)&&num1){
        cout<<"No\n";
        return 0;
    }
    if(num1){
        cout<<"Yes\n";
        return 0;
    }
    long double now=0;
    for(int i=n;i>-1;--i){
        now*=k;
        now+=a[i];
    }
    if(fabs(now)<0.5)cout<<"Yes\n";
    else cout<<"No\n";
}
