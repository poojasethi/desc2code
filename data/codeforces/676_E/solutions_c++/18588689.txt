#include <iostream>
#include <stdlib.h>
using namespace std;

int INF = 0x3f3f3f3f;

int main() {
	int n,k;
	cin>>n>>k;
	int a[n+1];
	char str[11];
	int unknown = 0;
	//input
	for (int i=0;i<=n;i++){
		cin>>str;
		if (str[0]=='?'){
			a[i] = INF;
			unknown++;
		}
		else{
			a[i] = atoi(str);
		}
	}

	//case k==0
	if (k==0){
		if (a[0]==0 || ((n-unknown)%2==0 && a[0]==INF)){
			cout<<"Yes";
		}
		else{
			cout<<"No";
		}
		return 0;
	}

	if (unknown == 0){
		//case if all are known
		bool isYes = true;
		for (int i=0;i<=n;i++){
			if (a[i]%k){
				isYes = false;
				break;
			}
			else{
				if(i!=n){
					a[i+1] += (a[i]/k);
				}
				else{
					isYes = (a[n]==0);
				}
			}
		}
		if (isYes){
			cout<<"Yes";
		}
		else{
			cout<<"No";
		}

	}
	else{
		//case exist unknown
		if (n%2==1){
			cout<<"Yes";
		}
		else{
			cout<<"No";
		}
	}
	return 0;
}
