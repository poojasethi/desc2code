#include<cstdio>
#include<cstring>
#include<string>
#include<sstream>
using namespace std;
double a[100005];
char re[20];
int main()
{
	int n,k,p=0;
	scanf("%d%d",&n,&k);
	bool flag=true,tflag=false;
	for(int i=0;i<=n;i++)
	{
		scanf("%s",re);
		if(re[0]=='?')
		{
			if(i==0)tflag=true;
			flag=false;
			p++;
			continue;
		}
		string s=re;
		stringstream ss;
		ss<<s;
		ss>>a[i];
	}
	int np=(n+1)-p;
	if(k==0)
	{
		if(tflag)
		{
			if(np&1)printf("Yes\n");
			else printf("No\n");
			return 0;
		}
		if(a[0]==0)printf("Yes\n");
		else printf("No\n");
		return 0;
	}
	if(!flag)
	{
		if(np&1)
		{
			if(p&1)printf("Yes\n");
			else printf("No\n");
		}
		else
		{
			if(p&1)printf("No\n");
			else printf("Yes\n");
		}
		return 0;
	}
	double ans=-a[0]/k;
	for(int i=1;i<n;i++)
	{
		ans-=a[i];
		ans/=k;
	}
	if(ans==a[n])printf("Yes\n");
	else printf("No\n");
}
