#include<cstdio>
#include<cstring>
#include<cstdlib>

using namespace std;
typedef long long LL;
const int maxn = 1e5+5;
const LL MOD = 100003619;

int n,m,a[maxn],b[maxn];
char s[maxn];

LL mod_pow(LL a,LL b){
    LL res = 1;
    while(b){
        if(b&1) res = (res*a)%MOD;
        a = (a*a)%MOD;
        b >>= 1;
    }
    return res;
}

int main(){
    int i,j;
    scanf("%d%d",&n,&m);
    int cnt = 0;
    LL val = 0;
    for(i = 0;i <= n;i++){
        scanf("%s",s);
        if(s[0]=='?'){
            cnt++;
            b[i] = 0;
        }else{
            b[i] = 1;
            a[i] = atoi(s);
            val = (val+a[i]*mod_pow(m,i))%MOD;
        }
    }
    bool ans = false;
    if(!m) ans = b[0]?(a[0]==0):((n+1-cnt)&1);
    else if(!cnt) ans = val==0;
    else ans = (n&1);
    printf("%s\n",ans?"Yes":"No");

    return 0;
}
