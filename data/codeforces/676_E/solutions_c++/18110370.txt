#include<stdio.h>
#include<stdlib.h>
typedef long long ll;
// Short code hahahahaha
ll M1=1000000007,M2=999999937;
int n,k,know[100010]={0},a[100010];
char s[20];
int main()
{
	scanf("%d%d",&n,&k);
	int i,j,ch=0;
	for (i=0;i<n+1;i++)
	{
		scanf("%s",s);
		if (s[0]!='?')
		{
			know[i]=1;
			a[i]=atoi(s);
		}
		else ch++;
	}
	if (k==0)
	{
		if (know[0])
		{	
			(a[0]?printf("No\n"):printf("Yes\n"));
			return 0;
		}
		((n+1-ch)%2==0)?printf("No\n"):printf("Yes\n");
		return 0;
	}
	if (ch!=0)
	{
		(n)%2==0?printf("No\n"):printf("Yes\n");
		return 0;
	}
	ll x1=0,ex1=1,x2=0,ex2=1;
	for (i=0;i<=n;i++)
	{
		x1=(x1+((a[i])%M1*ex1)%M1)%M1;
		x2=(x2+((a[i]%M2)*ex2)%M2)%M2;
		ex1=(ex1*(k%M1))%M1;
		ex2=(ex2*(k%M2))%M2;
	}
	(x1==0&&x2==0)?printf("Yes\n"):printf("No\n");
	return 0;
}
