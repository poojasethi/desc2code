#include <cstdio>
#include <cstring>
#include <cmath>
#include <queue>
#include <iostream>
#include <algorithm>
using namespace std;
#define INF 0x3f3f3f3f
#define eps 1e-8
typedef long long LL;

int n, k, cnt = 0;
LL tmp = 0;
char s[20];
int a[100010];
bool flag = 1;

int main()
{
	scanf("%d%d", &n, &k);
	for (int i = 0; i <= n; i++)
	{
		scanf("%s", s);
		if (k == 0 && i == 0)
		{
			if (s[0] == '0')
			{
				printf("Yes\n");
				return 0;
			}
			if (s[0] != '?')
			{
				printf("No\n");
				return 0;
			}
		}
		if (s[0] == '?') cnt++;
		else sscanf(s, "%d", &a[i]);
	}
	for (int i = n; i >= 0; i--)
	{
		tmp = tmp*k + a[i];
		if (abs(tmp) > abs(k*10000))
		{
			flag = 0;
			break;	
		}
	}
	if (tmp) flag = 0;
	if (k == 0)
	{
		if ((n+1-cnt) & 1) printf("Yes\n");
		else printf("No\n");
	}
	else if (cnt == 0)
	{
		 if (flag) printf("Yes\n");
		 else printf("No\n");
	}
	else if (n & 1) printf("Yes\n");
	else printf("No\n");
	return 0;
}
