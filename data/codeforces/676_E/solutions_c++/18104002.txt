#include <iostream>
#include<math.h>
#include<stdlib.h>
#include<string>
#include<stdio.h>
#include<limits.h>
#define ll long long

using namespace std;

int coeff[100001];


int main()
{
  //freopen("input.txt","r",stdin);
  int n,k,ctrp=0,ctrr=0;
  char str[11];
  cin>>n>>k;
  for(int i=0; i<=n; i++)
  {
     cin>>str;
     if(str[0]=='?')
     {
         ctrr++;
         coeff[i]=INT_MAX;
     }
     else { ctrp++;
        coeff[i]=atoi(str); }
  }
  long double sum = 0;
  for(int i=n; i>=0; i--)
  {
     sum*=k;
     sum+=(long double)coeff[i];
  }
  ll total = ctrp + ctrr - 1;
  if(ctrr==0) { if(sum==0) cout<<"Yes\n";  else  cout<<"No\n"; return 0; }
  if(k==0)
  {
      if(coeff[0]==0 || (ctrp%2 && coeff[0]==INT_MAX) ) cout<<"Yes\n"; else cout<<"No\n";
      return 0;
  }
  //cout<<total<<' '<<ctrp<<' '<<ctrr<<endl;
  if(total%2) cout<<"Yes\n"; else cout<<"No\n";

return 0;
}
