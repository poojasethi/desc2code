#include<stdio.h>
#include<iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
#define ll long long
const int N=100001;
const ll M1=1000000007ll;
const ll M2=2333333333ll;
char str[20];
int a[N];
int main(void)
{
	//freopen("sduoj.in","r",stdin);
	int n,m,i,j,k,p,flag;
	ll t1,t2,x1,x2;
	scanf("%d%d",&n,&m);p=0;
	for(i=0;i<=n;i++)
	{
		memset(str,0,sizeof(str));
		scanf("%s",str+1);
		if(str[1]=='?'){a[i]=N;p++;}
		else
		{
			j=1;
			if(str[j]=='-'){flag=-1;j++;}else flag=1;
			k=str[j]-48;for(j++;str[j];j++)k=k*10+str[j]-48;
			a[i]=k*flag;
		}
	}
	if(m==0)
	{
		if(a[0]==N&&((n+1-p)%2))printf("Yes");
		else if(a[0]==0)printf("Yes");
		else printf("No");
	}
	else if(p==0)
	{
		t1=t2=0;x1=x2=m;
		for(i=0;i<=n;i++)
		{
			t1=(t1+x1*a[i])%M1;x1=x1*m%M1;
			t2=(t2+x2*a[i])%M2;x2=x2*m%M2;
		}
		if(t1==0&&t2==0)printf("Yes");
		else printf("No");
	}
	else if((n+1)%2==0)printf("Yes");
	else printf("No");
	return 0;
}
