#include<bits/stdc++.h>
using namespace std;
#define  ll long long
ll  MOD=999999937 ;
int n,k,know[100007]={0},a[100007];
char s[15];
int main()
{
	scanf("%d%d",&n,&k);
	int i,j,ques=0;
	for (i=0;i<n+1;i++)
	{
		scanf("%s",s);
		if (s[0]!='?')
		{
			know[i]=1;
			a[i]=atoi(s);
		}
		else ques++;
	}
	if (k==0)
	{
		if (know[0])
		{	
			(a[0]?printf("No\n"):printf("Yes\n"));
			return 0;
		}
		((n+1-ques)%2==0)?printf("No\n"):printf("Yes\n");
		return 0;
	}
	if (ques!=0)
	{
		(n)%2==0?printf("No\n"):printf("Yes\n");
		return 0;
	}
	ll x1=0,ex1=1;
	for (i=0;i<=n;i++)
	{
		x1=(x1+((a[i])%MOD*ex1)%MOD)%MOD;
		ex1=(ex1*(k%MOD))%MOD;
	//	cout<<"---------------"<<endl;
	//	cout<<x1<<"  "<<ex1; 
	}
	(x1==0)?printf("Yes\n"):printf("No\n");
	return 0;
}
