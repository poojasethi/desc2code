#include <iostream>
#include <string>
#include <sstream>
#include <cstdlib>

using namespace std;

typedef long long int ll;

ll absolut(ll x)
{
  if (x<0) return -x;
  return x;
}

const int limite=1000000;

int n,k;
int x[limite];
int def[limite];
int numdef=0;

int main()
{
  ios::sync_with_stdio(false);
  cin>>n>>k;
  for (int i=0;i<=n;i++) {
    string s;
    cin>>s;
    if (s[0]!='?') {
      def[i]=1;
      istringstream mycin(s);
      mycin>>x[i];
      numdef++;
    }
  }
  if (k==0) {
    if (def[0]) {
      if (x[0]==0) cout<<"Yes"<<endl;
      else cout<<"No"<<endl;
    } else if (numdef%2==0) cout<<"No"<<endl;
    else cout<<"Yes"<<endl;
  } else if (numdef<n+1) {
    if ((n+1)%2==0) cout<<"Yes"<<endl;
    else cout<<"No"<<endl;
  } else {
    ll suma=0;
    for (int i=n;i>=0;i--) {
      suma=suma*k+x[i];
      if (absolut(suma)>1000000000000LL) {
	cout<<"No"<<endl;
	exit(0);
      }
    }
    if (suma==0) cout<<"Yes"<<endl;
    else cout<<"No"<<endl;
  }
}
