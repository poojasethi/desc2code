#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int maxn = 100000 + 10;
int a[maxn];
bool known[maxn];
char s[100];

void output(bool x) {
    printf("%s\n", x ? "Yes" : "No");
}

int main()
{
    int n, k, cnt = 0;
    scanf("%d%d", &n, &k);
    for(int i = 0; i <= n; i++) {
        scanf("%s", s);
        if(s[0] == '?') continue;
        known[i] = true;
        cnt++;
        sscanf(s, "%d", a + i);
    }

    if(k == 0) {
        if(known[0]) output(a[0] == 0);
        else output(cnt & 1);
    } else {
        if(cnt < n + 1) output(n & 1);
        else {
            int t = 0;
            for(int i = 0; i < n; i++) {
                t += a[i];
                if(t % k != 0) { output(false); return 0; }
                t /= k;
            }
            output(a[n] + t == 0);
        }
    }

    return 0;
}
