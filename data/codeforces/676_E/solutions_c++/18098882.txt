#include <iostream>
#include <cstdio>
using namespace std;
const int inf=0x3f3f3f3f;

int n, k, cnt, a[100005];
char w[105];

int main()
{
    scanf("%d%d",&n,&k);
    for(int i=0;i<=n;++i)
    {
        scanf("%s",w);
        if(w[0]=='?')a[i]=inf, ++cnt;
        else sscanf(w,"%d",&a[i]);
    }
    if(!k)
    {
        if(a[0]==inf)puts((n+1-cnt)%2?"Yes":"No");
        else puts(!a[0]?"Yes":"No");
    }
    else if(!cnt)
    {
        bool ans=1;
        for(int i=0;i<=n;++i)
            if(a[i]%k){ans=0;break;}
            else a[i+1]+=a[i]/k;
        if(a[n])ans=0;
        puts(ans?"Yes":"No");
    }
    else puts(n%2?"Yes":"No");
    return 0;
}
