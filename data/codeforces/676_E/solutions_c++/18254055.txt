#include <stdio.h>

int n;
int k;
int a[100100];
bool und[100100];

bool check_final()
{
	int result = 0;
	for (int i = 0; i <= n; ++i) {
		if (a[i] % k != 0)
			return false;
		a[i + 1] += a[i] / k;
	}
	return (a[n + 1] == 0);
}

bool solve()
{
	if (k == 0) {
		if (!und[0])
			return a[0] == 0;
	}
	int count = 0;
	for (int i = 0; i <= n; ++i)
		if (und[i]) count++;
	if (count == 0)
		return check_final();

	if (k == 0)
		return ((n + 1 - count) & 1) != 0;
	if ((n & 1) == 0)
		return false;
	// even round in total
	return true;
}

int main()
{
	scanf("%d %d", &n, &k);
	for (int i = 0; i <= n; ++i) {
		if (scanf("%d", &a[i]) < 1) {
			char qmark;
			scanf(" %c", &qmark);
			und[i] = true;
		} else und[i] = false;
	}
	printf("%s\n", solve() ? "YES" : "NO");

	return 0;
}
