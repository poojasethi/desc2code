#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#define N 100005
using namespace std;
int n,k,cnt,a[N]; char st[15];
void work()
{
	int i;
	for(i=0;i<=n;i++){
		if(a[i]!=0)
			a[i+1]+=a[i]/k,a[i]%=k;
		if(a[i]!=0){
			printf("No");
			exit(0);
			}
	  }
	if(a[n+1]!=0){
		printf("No");
		exit(0);
	  }
	printf("Yes");
	exit(0);
}
int main()
{
	int i,j,op;
	scanf("%d %d\n",&n,&k);
	for(i=0;i<=n;i++){
		scanf("%s",st+1);
		if(st[1]=='?') cnt++,a[i]=(1<<30);
		else{
			int l=strlen(st+1);
			op=1;
			for(j=1;j<=l;j++){
				if(st[j]=='-') op=-1;
				else a[i]=a[i]*10+st[j]-'0';
			  }
			a[i]*=op;
		  }
	  }
	if(k==0){
		if(a[0]==(1<<30)){
			if((n+1-cnt)&1) printf("Yes"); else printf("No");
		  }
		else{
			if(a[0]==0) printf("Yes"); else printf("No");
		  }
		return 0;
	  }
	if(!cnt) work();
	if(n&1) printf("Yes"); else printf("No");
	return 0;
}
