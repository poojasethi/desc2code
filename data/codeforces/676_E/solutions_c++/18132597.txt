#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const int maxn = 1e5 + 10, mod = 100003619;
int a[maxn], t[maxn], n, k;
char s[110];

int main() {
//  freopen("case.in", "r", stdin);
  cin >> n >> k;
  int ok = 0, c = 0;
  for (int i = 0; i <= n; i++) {
    scanf("%s", s);
    if (s[0] == '?') ok = 1;
    else {
      t[i] = 1;
      sscanf(s, "%d", a + i);
      c ^= 1;
    }
  }
  if (k == 0) {
    if (t[0]) puts(a[0] == 0 ? "Yes" : "No");
    else puts(c & 1 ? "Yes" : "No");
  } else {
    if (ok) puts(n & 1 ? "Yes" : "No");
    else {
      ll res = 0;
      for (int i = n; i >= 0; i--) {
        res = res * k + a[i];
        res %= mod;
      }
      puts(res == 0 ? "Yes" : "No");
    }
  }
  return 0;
}
