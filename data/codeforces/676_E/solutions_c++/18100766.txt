#include <bits/stdc++.h>
using namespace std;
void py()
{
	puts("YES");
	exit(0);
}
void pn()
{
	puts("NO");
	exit(0);
}
int main() {
	int n,k;
	cin>>n>>k;
	string s;
	if(k == 0)
	{
		cin>>s;
		if(s == "0")
			py();
		else if(s == "?")
		{
			int cnt = 1;
			for(int i=0;i<n;i++)
			{
				cin>>s;
				if(s == "?")cnt++;
			}
			if( (n+1-cnt)%2 == 1)py();
			else pn();
		}
		else pn();
	}
	else
	{
		long long temp = 0;
		int fail = 0;
		for(int i=0;i<=n;i++)
		{
			cin>>s;
			if(s == "?")
			{
				if((n+1)%2 == 0)py();
				else pn();
			}
			if(temp%abs(k))fail = 1;
			temp /= k;
			temp += atoi(s.c_str());
		}
		if(temp || fail)pn();
		else py();
	}
	return 0;
}
