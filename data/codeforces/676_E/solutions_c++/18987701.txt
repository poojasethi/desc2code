#include<bits/stdc++.h>
#include<cstring>
typedef long long LL;
const LL MO = 100003619;
bool val[100010];
int p[100010];
char s[20];
int num, k, n;

int main(){
	scanf("%d%d", &n, &k);
	for(int i = 0; i <= n; i++) {
		scanf("%s", &s);
		if (strcmp(s,"?") == 0) num++, val[i] = false;
		else {
			val[i] = true;
			sscanf(s, "%d", &p[i]);
		}
	}
	LL x = 1, r = 0;
	for(int i = 0; i <= n; i++) {
		r = (r+x*p[i]%MO)%MO;
		x = (x*k)%MO;
	}
	bool pan;
	if (k == 0)
		if ((val[0] && (!p[0])) || ((!val[0]) && ((n-num+1)%2 != 0))) pan = true;
		else pan = false;
	else
		if (!num) pan = (r == 0);
		else pan = n%2;
	printf("%s\n", pan ? "Yes":"No");
}
