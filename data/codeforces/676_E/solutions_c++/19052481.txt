#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;

const int N = 100010;
typedef long long LL;
const LL Lim = 1LL << 40;
int n, k, a[N];
char s[110];
int main(){
	scanf("%d%d",&n,&k);
	int ques = 0;
	bool sign = 0;
	for(int i = 0; i <= n; i ++){
		scanf("%s", s);
		if (s[0] == '?') ques ++;
		else sscanf(s,"%d",&a[i]);
		if (i == 0 && s[0] == '?') sign = 1;
	}
	if (k == 0){
		if (sign){
			if ((n - ques) & 1) puts("No");
			else puts("Yes");
		}else{
			if (a[0] == 0) puts("Yes");
			else puts("No");
		}
		return 0;
	}
	//solve k != 0
	if (ques){
		if (n & 1) puts("Yes");
		else puts("No");
	}else{
		LL now = a[n];
		for(int i = n - 1; i >= 0; i --){
			now = now * k + a[i];
			if (abs(now) > Lim){
				puts("No");
				return 0;
			}
		}
		if (now == 0) puts("Yes");
		else puts("No");
	}
	return 0;
}
