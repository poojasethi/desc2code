#include<cstdio>
#include<cstdlib>

using namespace std;
typedef long long LL;
int n,k,cnt;
char s[100];
const int inf=0x3f3f3f3f;
const LL maxa=1e14;
int a[100010];

int main()
{
	scanf("%d%d",&n,&k);
	for(int i=0;i<=n;i++)
	{
		while(gets(s),s[0]==0);
		if(s[0]=='?')
		{
			cnt++;
			a[i]=inf;
		}
		else
		{
			a[i]=atoi(s);
		}
	}
	if(k==0)
	{
		if(a[0]==inf)
		{
			
			puts(((n+1-cnt)&1)?"Yes":"No");
		}
		else
		{
			puts(a[0]==0?"Yes":"No");
		}
	}
	else
	{
		if(cnt)
		{
			puts((n&1)?"Yes":"No");
		}
		else
		{
			LL tmp=0;
			bool flag=0;
			for(int i=n;i>=0;i--)
			{
				tmp=tmp*k+a[i];
				if(abs(tmp)>maxa)
				{
					flag=1;
					break;
				}
			}
			if(flag)puts("No");
			else puts(tmp==0?"Yes":"No");
		}
	}
	
	return 0;
}
