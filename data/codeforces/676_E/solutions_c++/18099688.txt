#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#define nMax 100010
#define Mod 998244353
using namespace std;
int n,k,b[nMax];
long long c[nMax];
char a[nMax];
int main()
{
	scanf("%d%d",&n,&k);
	int cnt=0;
	for(int i=0;i<=n;i++)
	{
		scanf("%s",a);
		if(a[0]=='?') {cnt++; b[i]=-nMax; continue;}
		b[i]=atoi(a);
	}
	if(k==0)
	{
		if(cnt==0)
		{
			if(b[0]==0) printf("Yes\n"); else printf("No\n");
			return 0;
		}
		if(b[0]==0) printf("Yes\n"); else
		if(b[0]!=-nMax) printf("No\n"); else
		if((n+1-cnt)%2) printf("Yes\n"); else printf("No\n");
		return 0;
	}
	if(cnt)
	{
		if((n+1)%2) printf("No\n"); else printf("Yes\n");
		return 0;
	}
	for(int i=n;i>0;i--)
	{
		c[i-1]=c[i]*k+b[i];
		c[i-1]%=Mod;
	}
	if(((-c[0]*k)%Mod+Mod)%Mod==((b[0]%Mod)+Mod)%Mod) printf("Yes\n"); else printf("No\n");
	return 0;
}
