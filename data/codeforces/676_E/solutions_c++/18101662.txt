#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

int main()
{
	int n, k, count = 0, i;
	double A[100001];
	double sum = 0;
	char str[100];

	cin>>n>>k;

	for(i = 1; i <= n + 1; i++)
	{
		cin>>str;
		if(str[0] == '?')
		{
			count++;
		}
		else
		{
			A[i - 1] = atof(str);
			if(k == 0 && i == 1 && A[i - 1] == 0)
			{
				cout<<"Yes";
				return 0;
			}
			else if(k == 0 && i == 1)
			{
				cout<<"No";
				return 0;
			}
		}
	}

	if(k == 0 && (n + 1 - count) % 2 == 0)
	{
		cout<<"No";
		return 0;
	}
	if(k == 0)
	{
		cout<<"Yes";
		return 0;
	}

	if(n % 2 == 0 && count > 0)
	{
		cout<<"No";
		return 0;
	}

	if(n % 2 != 0 && count > 0)
	{
		cout<<"Yes";
		return 0;
	}

	for(i = n; i >= 0; i--)
	{
		sum = k * sum + A[i];
	}

	if(sum == 0)
	{
		cout<<"Yes";
	}
	else
	{
		cout<<"No";
	}
	return 0;
}
