#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm> 
using namespace std;
const int N = 100005;
int n, k, a[N], quest;
bool a0;
char buf[10];
int main() {
	scanf("%d%d", &n, &k);
	for (int i = 0; i <= n; ++i) {
		scanf("%s", buf);
		if (*buf == '?') {
			quest++;
			if (i == 0) a0 = true;
		} else {
			sscanf(buf, "%d", a + i);
		}
	}
	if (k == 0) {
		if (a0) printf((n - quest) & 1 ? "No\n" : "Yes\n");
		else printf(a[0] == 0 ? "Yes\n" : "No\n");
	} else if (quest > 0) {
		printf(n & 1 ? "Yes\n" : "No\n");
	} else {
		long long y = a[n];
		for (int i = n - 1; i >= 0; --i) {
			y = y * k + a[i];
			if (y > (1LL << 32) || y < -(1LL << 32)) {
				printf("No\n");
				return 0;
			}
		}
		printf(y == 0 ? "Yes\n" : "No\n");
	}
}
