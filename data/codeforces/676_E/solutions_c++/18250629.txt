#include<cstdio>
#include<iostream>
using namespace std;
long long n,k;
long long sum;
long long mi;
char buf[200];
int ukn;
const long long mod=19991105;
long long cst;
int main(){
	scanf("%I64d%I64d",&n,&k);
	n++;mi=1;k+=mod;k%=mod;
	for(int i=1;i<=n;i++){
		scanf("%s",buf);
		if(buf[0]=='?'){
			ukn++;if(i==1) cst=-mod;
		}else{
			long long bb;
			sscanf(buf,"%I64d",&bb);
			bb+=mod;bb%=mod;
			sum+=(bb*mi)%mod;sum%=mod;
			if(i==1) cst=bb;
		}
		mi*=k;mi%=mod;
	}
	if(ukn==0){
		if(sum==0){
			puts("Yes");
		}else{
			puts("No");
		}
	}else if(k!=0){
		if(n&1){
			puts("No");
		}else puts("Yes");
	}else if(k==0){
		if(cst==-mod){
			if((n-ukn)&1==1){
				puts("Yes");
			}else{
				puts("No");
			}
		}else{
			if(cst) puts("No");
			else puts("Yes");
		}
	}
	return 0;
}
