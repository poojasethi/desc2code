#include<cstdio>
#include<cstring>
#include<cmath>
#define maxn 100010
int n,k;
int a[maxn];
int main()
{
	scanf("%d%d",&n,&k);
	if(k==0)
	{
		int qcnt=0;
		char s[10]; scanf("%s",s);
		for(int i=1; i<=n; i++)
		{
			char s1[10]; scanf("%s",s1);
			if(s1[0]!='?') qcnt++;
		}
		if(qcnt&1)
		{
			if(s[0]=='?' || s[0]=='0') puts("Yes");
			else puts("No");
		}
		else
		{
			if(s[0]=='0') puts("Yes");
			else puts("No");
		}
		return 0;
	}
	for(int i=0; i<=n; i++)
	{
		char s[10]; scanf("%s",s);
		if(s[0]=='?')
		{
			puts((n+1)&1 ? "No" : "Yes");
			return 0;
		}
		sscanf(s,"%d",&a[i]);
	}
	long long tot=a[n];
	for(int i=n-1; i>=0; i--)
	{
		tot=1ll*tot*k+a[i];
		if(tot>10000 || tot<-10000) {puts("No"); return 0;}
	}
	if(tot==0) puts("Yes");
	else puts("No");
	return 0;
}
