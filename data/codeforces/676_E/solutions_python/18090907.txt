n, k = map(int, raw_input().split(' '))
a = []
cnt = 0
for i in xrange(n + 1):
	tmp = raw_input()
	if tmp == '?':
		a.append(tmp)
		cnt += 1
	else:
		a.append(int(tmp))
if k == 0:
	if a[0] == 0 or (a[0] == '?' and (n + 1 - cnt) % 2 == 1):
		print 'Yes'
	else:
		print 'No'
elif cnt:
	if n % 2 == 0:
		print 'No'
	else:
		print 'Yes'
else:
	now = 0
	flag = 0
	for i in a[:-1]:
		now -= i
		if now % k:
			flag = 1
			break
		now /= k
	if flag or now != a[-1]:
		print 'No'
	else:
		print 'Yes'
