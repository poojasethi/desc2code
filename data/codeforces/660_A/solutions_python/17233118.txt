def gcd(a, b):
	while b:
		a, b = b, a % b
	return a

n = input()
a = map(int, raw_input().split())
new_a = []

k = 0
for i in xrange(n - 1):
	new_a.append(a[i])
	nod = gcd(a[i], a[i+1])
	if nod == 1:
		continue
	else:
		new_a.append(1)
		k += 1
new_a.append(a[n-1])

print k
print ' '.join(map(str, new_a))
