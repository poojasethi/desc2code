n = int(raw_input())
a = map(int, raw_input().split())


def gcd(a, b):
    while a != 0 and b != 0:
        if a > b: a = a%b
        else: b = b%a
    return a if b ==0 else b

count = 0
res = list()
for i in range(n - 1):
    res.append(a[i])
    if gcd(a[i], a[i + 1]) > 1:
        count += 1
        res.append(1)
if n > 0:
    res.append(a[n - 1])

print count
print ' '.join(map(str, res))
