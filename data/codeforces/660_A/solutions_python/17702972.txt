
from fractions import gcd

N = int(raw_input())
vs = map(int, raw_input().split())

count = 0
updated = True
while updated:
   updated = False
   for i in range(1, len(vs)):
      if gcd(vs[i-1], vs[i]) != 1:
         vs.insert(i, 1)
         updated = True
         count += 1

print count
print ' '.join(map(str, vs))
