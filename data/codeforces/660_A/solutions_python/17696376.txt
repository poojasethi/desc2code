n = int(raw_input())
s = [int(x) for x in raw_input().split()]
def gcd(a,b):
	while(b!=0):
		t = b
		b = a % b
		a = t
	return a
l = list()
counter = 0
tmp = int(s[0])
prev = tmp
l.append(tmp)
for i in range(1,n):
	tmp = int(s[i])
	if (gcd(prev, tmp) != 1):
		counter = counter + 1
		l.append(1)
	prev = tmp
	l.append(tmp)
