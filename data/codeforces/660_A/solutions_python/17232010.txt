from fractions import gcd

import sys
stdin = sys.stdin

N =        int( stdin.readline().strip() )
aa = map(int, stdin.readline().strip().split())

rr = [aa[0]]
for el in aa[1:]:
    if gcd(rr[-1], el) != 1:
        rr.append(1)
    rr.append(el)

print len(rr)-len(aa)
print ' '.join(map(str, rr))
