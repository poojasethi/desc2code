n=int(raw_input())
def calc(a,b):
	while(b!=0):
		t=b
		b=a%t
		a=t
	if a==1:
		return 1
	else:
		return 0

i=0
count=0
l = list(map(int,raw_input().split()))

while i<(n+count-1):
	if l[i]>=l[i+1] :
		if calc(l[i],l[i+1])==0 and l[i+1]!=1:	
			l.insert(i+1,1)
			count+=1
	else:
		if calc(l[i+1],l[i])==0 and l[i]!=1:
			l.insert(i+1,1)
			count+=1
	i+=1
print count
print ' '.join(map(str, l))
