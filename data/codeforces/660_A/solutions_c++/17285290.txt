#include <bits/stdc++.h>
using namespace std;
int main() 
{
	int num,prev=1;
	cin>>num;
	vector<int>arr;
	for(int i=0;i<num;i++)
	{
		int temp;
		cin>>temp;
		if(__gcd(prev,temp)!=1)
		arr.push_back(1);
		arr.push_back(temp);
		prev=temp;
	}
	cout<<arr.size()-num<<endl;
	for(int i=0;i<arr.size();i++)
	cout<<arr[i]<<" ";
	cout<<endl;
}
