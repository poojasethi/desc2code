#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int x,a[2000],num=0,b[2000],i;
int gsd(int x,int y)
{
	while(x*y)
    {
		swap(x,y);
		x%=y;
	}
	return x+y;
}
int main()
{
    cin>>x;
	for(i=0;i<x;i++)
    {
		cin>>a[i];
		if(i!=0 && gsd(a[i-1],a[i])!=1)
			b[num++]=1;
		b[num++]=a[i];
	}
	cout<<num-x<<endl;
	for(i=0;i<num;i++)
		cout<<b[i]<<" ";
    return 0;
}
