#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b){
	return b==0?a:gcd(b,a%b);
}
int main(){
	int n,t=0,a[2010],b[2010];
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		cin>>a[i];
		if(i&&gcd(a[i],a[i-1])!=1)
		b[t++]=1;
		b[t++]=a[i];
	}
	cout<<t-n<<endl;
	cout<<b[0];
	for(int i=1;i<t;i++)
	cout<<" "<<b[i];
	cout<<endl;
	return 0;
	
	return 0;
}
