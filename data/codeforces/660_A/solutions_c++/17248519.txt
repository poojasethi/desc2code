#include <bits/stdc++.h>
using namespace std;
int a[1005];
int main()
{
   int n;
   cin>>n;
   for(int i=0;  i<n; i++) cin>>a[i];
   vector <int> v; int cnt=0;
   v.push_back(a[0]);
   for(int i=1; i<n; i++){
      if(__gcd(a[i-1], a[i])==1) v.push_back(a[i]);
      else{
         cnt++;
         v.push_back(1); v.push_back(a[i]);
      }
   }
   cout<<cnt<<endl;
   for(int i=0; i<v.size(); i++) cout<<v[i]<<' ';
}
