#include <cstdio>
#include <iostream>
using namespace std;
int n,tot,f[2005],a[2005];
int gcd(int x,int y)
{
	if(x % y == 0) return y;
	return gcd(y,x % y);
}
int main()
{
	cin.sync_with_stdio(false);
	cin>>n;
	int pre = 0;
	for(int i = 1;i <= n;i++) 
	{
		cin>>a[i];
		if(pre && gcd(a[i],pre) != 1) f[++tot] = 1;
		f[++tot] = pre = a[i];
	}
	cout<<tot-n<<endl;
	for(int i = 1;i <= tot;i++) cout<<f[i]<<" ";
 }
