#include <cstdio>

int gcd(int a, int b) {
	return !b ? a : gcd(b, a % b);
}
int main() {
	int n, m = 0, a[1000];
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &a[i]);
		if (i > 0 && gcd(a[i - 1], a[i]) > 1)
			++m;
	}
	printf("%d\n", m);
	for (int i = 0; i < n; ++i) {
		printf("%d ", a[i]);
		if (i != n - 1 && gcd(a[i], a[i + 1]) > 1)
			printf("1 ");
	}
	return 0;
}
