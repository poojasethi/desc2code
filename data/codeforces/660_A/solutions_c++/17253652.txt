#include <bits/stdc++.h>

using namespace std;

const int N=1010;
int a[N],n,ans;
int main()
{
    scanf("%d",&n);
    for (int i=1; i<=n; i++)
        scanf("%d",&a[i]);
    ans=0;
    for (int i=1; i<n; i++)
        if (__gcd(a[i],a[i+1])!=1) ans++;
    printf("%d\n%d ",ans,a[1]);
    for (int i=2; i<=n; i++)
        if (__gcd(a[i],a[i-1])!=1)
            printf("1 %d ",a[i]);
        else printf("%d ",a[i]);
}
