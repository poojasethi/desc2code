#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)return a;
	a%=b;
	return gcd(b,a);
}
int main()
{
	int i,n,k=0;
	cin>>n;
	int a[n];
	vector<int>ans;
	for(i=0;i<n;i++)cin>>a[i];
	for(i=1;i<n;i++)
	{
		if(gcd(a[i],a[i-1])==1)
		{
			ans.push_back(a[i-1]);
		}
		else
		{	
			ans.push_back(a[i-1]);
			ans.push_back(1);
			k++;
		}
	}
	ans.push_back(a[n-1]);
	cout<<k<<endl;
	for(i=0;i<n+k;i++)cout<<ans[i]<<' ';
}
