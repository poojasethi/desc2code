#include <iostream>
#include <algorithm>

using namespace std;

int n,ans;
int v[1002];

int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>v[i];
    for(int i=1;i<n;i++)
        if(__gcd(v[i],v[i+1])>1)
            ans++;
    cout<<ans<<'\n';
    for(int i=1;i<=n;i++)
    {
        cout<<v[i]<<" ";
        if(i<n && __gcd(v[i],v[i+1])>1)
            cout<<"1 ";
    }
    return 0;
}
