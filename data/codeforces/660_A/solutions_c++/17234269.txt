#include <bits/stdc++.h>
using namespace std;

const int kMaxN=2000;
int a[kMaxN];
vector<int> ans;

int main(){
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d",a+i);
	}
	int k=0;
	for(int i=1;i<n;i++){
		ans.push_back(a[i]);
		if(__gcd(a[i],a[i+1])!=1){
			k++;
			ans.push_back(1);
		}
	}
	ans.push_back(a[n]);
	printf("%d\n",k);
	for(int i=0;i<ans.size();i++){
		printf("%d ",ans[i]);
	}
	return 0;
}
