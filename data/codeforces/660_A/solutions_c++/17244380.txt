#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

int n,x,y,k;
vector<int> v;

int main(){
  cin>>n>>x;
  v.push_back(x);
  for(int i=1;i<n;i++){
    cin>>y;
    if(__gcd(x,y)!=1)
      k++,v.push_back(1);
    v.push_back(y);
    x=y;
  }
  cout<<k<<endl;
  for(int i=0;i<v.size();i++)
    cout<<v[i]<<' ';
  return 0;
}
