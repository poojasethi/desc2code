#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
vector<ll>v;
ll m,n,i,d,a[100005];
int main()
{
   cin>>n;
   for(i=0;i<n;i++)
   cin>>a[i];

   for(i=0;i<(n-1);i++)
   {
       v.pb(a[i]);
       if(__gcd(a[i],a[i+1])!=1){
        v.pb(1);
        d++;}
   }
   v.pb(a[n-1]);
   i=v.size();
   cout<<d<<endl;
   for(m=0;m<i;m++)
    cout<<v[m]<<" ";
}
