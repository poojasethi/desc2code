#include <iostream>

using namespace std;

int mcd(int a,int b)
{
    if(!b) return a;
    return mcd(b,a%b);
}

int main()
{
    int n,i,a[1001],b[2002],v=0,c=0;
    cin>>n;
    for(i=0;i<n;++i) cin>>a[i];
    for(i=0;i<n-1;++i)
    {
        b[v++]=a[i];
        if(mcd(a[i],a[i+1])-1) b[v++]=1,++c;
    }
    b[v]=a[n-1];
    cout << c << endl;
    for(i=0;i<n+c;i++) cout << b[i] << " ";
    return 0;
}
