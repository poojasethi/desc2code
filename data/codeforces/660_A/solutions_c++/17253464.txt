#include <bits/stdc++.h>
using namespace std;
int a[1000000]={1},i,n,l;
vector<int>v;

int main()
{
    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
        if(__gcd(a[i],a[i-1])>1)v.push_back(1);
        v.push_back(a[i]);
    }
    l=v.size();
    cout<<l-n<<endl;
    for(i=0;i<l;i++)cout<<v[i]<< " ";
}
