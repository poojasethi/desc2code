#include<bits/stdc++.h>
using namespace std;
int a[2020];
int main()
{
	int n;cin>>n;
	for(int i=0 ; i<(2*n) ; i+=2)
	cin>>a[i];
	
	int ans=0;
	for(int i=0 ; i<2*(n-1) ; i+=2)
	{
		if(__gcd (a[i],a[i+2]) != 1 )
		{
			a[i+1]= 1;
			ans++;
		}
	}
	
	cout<<ans<<endl;
	for(int i=0  ; i< 2*n ; i++ )
	a[i] !=0 ? cout<<a[i]<<" " : "" ;
}
