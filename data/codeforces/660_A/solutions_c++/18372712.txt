#include<bits/stdc++.h>
using namespace std;

int a[1111];
int gcd(int a, int b) {
    return !a? b : gcd(b%a,a);
}

int main() {
    int n;
    scanf("%d",&n);
    int cnt = 0;
    a[0] = 1;
    for(int i = 1; i <= n; ++i) {
        int x;
        scanf("%d",&x);
        if(gcd(x,a[cnt])!=1) a[++cnt] = 1;
        a[++cnt] = x;
    }
    printf("%d\n",cnt - n);
    for(int i = 1; i <= cnt; ++i) printf("%d ",a[i]);
}
