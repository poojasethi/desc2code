#include <cstdio>
int N,M,a[1010];
int gcd(int a,int b){
    if(b==0) return a;
    return gcd(b,a%b);
}
int main(){
    scanf("%d", &N); M=0;
    scanf("%d", &a[1]); 
    for(int i=2; i<=N; i++){
        scanf("%d", &a[i]);
        if(gcd(a[i-1],a[i])>1) M++;
    }
    printf("%d\n%d", M,a[1]);
    for(int i=2; i<=N; i++){
        if(gcd(a[i-1],a[i])>1) printf(" 1");
        printf(" %d", a[i]);
    }
}
