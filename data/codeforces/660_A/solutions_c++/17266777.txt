#include <bits/stdc++.h>

using namespace std;
int a[3000],b[3000],n,t=0;


int main()
{
    cin>>n;
    for (int i = 0 ; i < n ; i++)
    {
        cin>>a[i];
        if (i &&__gcd(a[i],a[i-1])!=1)
        {
            b[t++]=1;
        }
        b[t++]=a[i];

    }
    cout<<t-n<<endl;
    for(int i=0 ; i < t ; i++)cout<<b[i]<<" ";


    return 0;
}
