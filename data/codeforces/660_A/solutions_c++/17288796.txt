#include <bits/stdc++.h>

using namespace std;


int main(){
	
	int n, k = 0;
	cin >> n;
	int a[n + 1];
	for(int i = 1; i <= n; i++)cin >> a[i];
	for(int i = 1; i <= n - 1; i++){
		if(__gcd(a[i] , a[i + 1]) != 1){
		k++;	
		}	
	}
	cout << k << endl;
	for(int i = 1; i <= n - 1; i++){
		cout << a[i] << " ";
		if(__gcd(a[i] , a[i + 1]) != 1){
			cout << 1 << " ";
		}
	}
	cout << a[n];
	return 0;
}
