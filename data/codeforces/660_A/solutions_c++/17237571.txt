#include <bits/stdc++.h>
using namespace std;

int n, otv, a[200020];

int main()
{
    cin >> n;
    for(int i = 0; i < n; ++i)
        cin >> a[i];
    for(int i = 0; i < n - 1; ++i)
        if(__gcd(a[i], a[i+1]) > 1) otv++;
    cout << otv << "\n";
    for(int i = 0; i < n - 1; ++i)
    {
        cout << a[i] << ' ';
        if(__gcd(a[i], a[i+1]) > 1)
            cout << "1 ";
    }
    cout << a[n - 1];
}
