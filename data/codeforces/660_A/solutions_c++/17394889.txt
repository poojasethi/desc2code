#include <bits/stdc++.h>
using namespace std;

int main() 
{
	int n,a[1009]={1},t;
	cin>>n;
	vector<int> v;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		if((t=__gcd(a[i-1],a[i]))>1)
			v.push_back(1);
		v.push_back(a[i]);
	}
	cout<<v.size()-n<<endl;
	for(int i=0;i<v.size();i++)cout<<v[i]<<" ";
	return 0;
}
