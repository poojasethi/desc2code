#include <bits/stdc++.h>
using namespace std;

int a[1050],b[2050],k,n;
int main(){
	cin >> n;
	for( int i=0 ; i<n ; i++ ) cin >> a[i];
	for( int i=0 ; i<n-1 ; i++ ){
		b[i+k]=a[i];
		if( __gcd(a[i],a[i+1]) != 1 ) k++, b[i+k]=1;
	}
	b[n+k-1]=a[n-1];
	cout << k << endl;
	for( int i=0 ; i<n+k ; i++ ) printf("%d ",b[i]);
	return 0;
}
