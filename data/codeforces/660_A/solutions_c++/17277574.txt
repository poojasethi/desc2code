#include <bits/stdc++.h>

using namespace std;



int main()
{
	int n,a,g,v,c=0;
	static int A[1005];

	scanf("%d",&a);

	scanf("%d",&A[1]);
	for(int i=2;i<=a;++i)
	{
		scanf("%d",&v);
		g=__gcd(v,A[i-1]);
		//printf("%d",g);
		if(g==1){
			A[i]=v;
		}
		else
		{
			A[i]=1;
			A[i+1]=v;
			++i;
			++a;
			c++;
		}

	}
	printf ("%d\n",c);
	for (int i=1;i<=a;++i){
		printf("%d ",A[i]);
	}
	
	return 0;

	
}
