#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,a[100005]={1},t;vector<int> vec;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		if((t=__gcd(a[i-1],a[i]))>1)
			vec.push_back(1);
		vec.push_back(a[i]);
	}
	cout<<vec.size()-n<<endl;
	for(int i=0;i<vec.size();i++)cout<<vec[i]<<" ";
	return 0;
}
