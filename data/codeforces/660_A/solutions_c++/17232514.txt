#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
int n;
int ans;
int a[111111];
int main()
{
	cin >> n;
	for( int i = 1; i <= n; i++)
		cin >> a[i];
	for( int i = 1; i < n; i++)
		if(__gcd(a[i], a[i + 1]) != 1)
			ans++;
	cout << ans << "\n";
	for( int i = 1; i < n; i++)
	{
		cout << a[i] << " ";
		if(__gcd(a[i], a[i + 1]) != 1)
			cout << 1 << " ";
	}
	cout << a[n];
	return 0;
}
