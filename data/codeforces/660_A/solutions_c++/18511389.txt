#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	vector<int> v;
	cin >> n;
	int a[n];
	for(int i = 0; i < n; i++) cin >> a[i];
	v.push_back(a[0]);
	for(int i = 1; i < n; i++)
	{
		if(__gcd(a[i-1], a[i]) > 1)
			v.push_back(1);
		v.push_back(a[i]);
	}
	cout << v.size()-n << endl;
	for(int i = 0; i < v.size(); i++) cout << v[i] << ' ';
	cout << endl;
}
