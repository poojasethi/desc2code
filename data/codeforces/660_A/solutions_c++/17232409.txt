#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int a[1003];
int v[1003]={0};
int main() {
	int n,k=0;
	scanf("%d",&n);
	for(int i=0;i<n;i++) {
		scanf("%d",&a[i]);
	}
	for(int i=0;i<(n-1);i++) {
		if (__gcd(a[i],a[i+1])>1) {
			v[i+1]=1;
			k++;
		}
	}
	printf("%d\n",k);
	for(int i=0; i<n;i++) {
		if (v[i]==1) {
			printf("%d ",1);
		}
		printf("%d ",a[i]);
	}
}
