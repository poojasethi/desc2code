#include <bits/stdc++.h>
using namespace std;
int a[1010];

int main(int argc, char const *argv[])
{
	int ans = 0;
	int n;
	cin >> n;
	for(int i = 1; i <= n; i++) cin >> a[i];
	for(int i = 1; i < n; i++) ans += (__gcd(a[i], a[i + 1]) != 1); 	
	cout << ans << endl;
	for(int i = 1; i < n; i++) {
		cout << a[i] << " ";

		if(__gcd(a[i], a[i + 1]) != 1) {
			cout << "1 ";
		}
	}
	cout << a[n] << endl;
	return 0;
}
