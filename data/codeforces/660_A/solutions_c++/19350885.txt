#include<bits/stdc++.h>

using namespace std;
int a[1000000];
int gcd(int a,int b)
{
  if(b==0)
    return a;
  return gcd(b,a%b);
}
int main()
{
  int n;
  cin>>n;
  for(int i=1;i<=n;i++)
    cin>>a[i];
  int cnt=0;
  for(int i=1;i<n;i++)
    if(gcd(a[i],a[i+1])>1)
      cnt++;
  cout<<cnt<<endl;
  for(int i=1;i<n;i++){
    cout<<a[i]<<" ";
    if(gcd(a[i],a[i+1])>1)
      cout<<"1 ";}
  cout<<a[n];
}
