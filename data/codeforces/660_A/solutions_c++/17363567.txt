
#include <iostream>
#include <cmath>
using namespace std;
int x,a[2000],o=0,b[2000],i;
int gsd(int x,int y){
	while(x*y){
		swap(x,y);
		x%=y;
	}
	return x+y;
}
int main(){
    cin>>x;
	for(i=0;i<x;i++){
		cin>>a[i];
		if(i!=0 && gsd(a[i-1],a[i])!=1){
			b[o++]=1;
        }
		b[o++]=a[i];
	}
	cout<<o-x<<endl;
	for(i=0;i<o;i++){
		cout<<b[i]<<" ";
	}
    return 0;
}
