#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	if (a==0) return b;
	else return gcd(b%a,a);
}
int main()
{
	int n;
	int a[1011];
	cin>>n;
	for (int i=1;i<=n;i++) cin>>a[i];
	int k=0;
	for (int i=2;i<=n;i++)
	{
		if (gcd(a[i-1],a[i])!=1) k++;
	}
	cout<<k<<endl;
	cout<<a[1]<<" ";
	for (int i=2;i<=n;i++)
	{
		if (gcd(a[i-1],a[i])!=1) cout<<"1 ";
		cout<<a[i]<<" ";
	}
}
