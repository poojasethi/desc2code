#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <iostream>
#define LL long long
#define N 1010
using namespace std;
LL rx[N], ry[N], bx[N], by[N];
int g[N][N];
int n, m;

int main()
{
	int i, j, k;
	scanf("%d%d", &n, &m);
	for (i = 1; i <= n; ++ i)
		scanf("%I64d%I64d", &rx[i], &ry[i]);
	for (i = 1; i <= m; ++ i)
		scanf("%I64d%I64d", &bx[i], &by[i]);
	for (i = 1; i <= n; i++)
	 for (j = 1; j <= n; j++)
	 if (rx[i] < rx[j])
     {
	 	for (k = 1; k <= m; k++)
	 	{
	 		LL a = rx[j] - rx[i];
	 		LL b = ry[j] - ry[i];
	 		LL c = bx[k] - rx[i];
	 		LL d = by[k] - ry[i];
	 		if (bx[k] >= rx[i] && bx[k] < rx[j] && b * c - a * d > 0LL)
	 			g[i][j]++;
	 	}
	 	g[j][i] = -g[i][j];
     }
	 	int ans = 0;
	 	for (i = 1; i <= n; i++)
	 		for (j = i + 1; j <= n; j++)
	 			for (k = j + 1; k <= n; k++)
	 			if (g[i][j] + g[j][k] + g[k][i] == 0) ans++;
	 	printf("%d\n", ans);
	 	return 0;
}
