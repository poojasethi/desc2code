using namespace std;
#define MAXN 510
int n,m,A[MAXN][MAXN];
int S;
long long x[MAXN],y[MAXN],X[MAXN],Y[MAXN];
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;scanf("%I64d%I64d",x+i,y+i),i++);
    for(int i=1;i<=m;scanf("%I64d%I64d",X+i,Y+i),i++);
    for(int i=1;i<=n;i++)
        for(int j=1;j<=n;j++)if(x[i]<x[j]){
                long long a=y[j]-y[i],b=x[i]-x[j],c=-x[i]*a-y[i]*b;
                for(int k=1;k<=m;A[i][j]+=X[k]>x[i]&&X[k]<=x[j]&&a*X[k]+b*Y[k]+c>0,k++);
                A[j][i]=-A[i][j];
            }
    for(int i=1;i<=n;i++)
        for(int j=i+1;j<=n;j++)
            for(int k=j+1;k<=n;k++)
                S+=A[i][j]+A[j][k]+A[k][i]==0;
    printf("%d\n",S);   
}
