#include <bits/stdc++.h>

#define	st first
#define	nd second
#define	mp make_pair
#define	pb push_back
#define	lli long long int
#define	all( gg )	gg.begin(),gg.end()
#define	foreach( gg,itit )	for( typeof(gg.begin()) itit=gg.begin();itit!=gg.end();itit++ )
#define	FP( ii,aa,bb ) for( lli ii=aa;ii<=bb;ii++ )
#define	FM( ii,aa,bb ) for( lli ii=aa;ii>=bb;ii-- )
#define	debug(ccc)	cout << #ccc << " = " << ccc << endl;

#define	mod	1000000007LL

#define	point	pair<lli,lli>

using namespace std;

int	n,m;

point	red[505],blue[505];

int	val[505][505];

lli	ccw( point a,point b,point c ){
	return	(b.st-a.st)*(c.nd-a.nd)-(b.nd-a.nd)*(c.st-a.st);
}

int main(){
	cin >> n >> m;
	FP( i,1,n )
		cin >> red[i].st >> red[i].nd;
	FP( i,1,m )
		cin >> blue[i].st>> blue[i].nd;
	FP( i,1,n )
		FP( j,1,n ){
			if( i==j )	continue;
			point	a=red[i],b=red[j];
			if( a.nd==b.nd )	continue;
			if( a.nd>b.nd )	swap( a,b );
			FP( k,1,m )
				if( a.nd<blue[k].nd and blue[k].nd<=b.nd and ccw( a,b,blue[k] )<0 )	val[i][j]++;
			if( red[i].nd<red[j].nd )	val[i][j]*=-1;
		}
	int	res=0;
	FP( i,1,n )
		FP( j,i+1,n )
			FP( k,j+1,n )
				if( !(val[i][j]+val[j][k]+val[k][i]) )	res++;
	cout << res << endl;
}
