#include <iostream> 
using namespace std;  

#define FOR(i,a,b) for(int i=(a); i<(int)(b); ++i) 

const int N = 500;

int rx[N], ry[N];
int bx[N], by[N];
int bal[N][N];
double rbx[N][N];
double rby[N][N];

int main()
{
	int n, m;

	cin >> n >> m;

	FOR(i, 0, n) cin >> rx[i] >> ry[i];
	FOR(i, 0, m) cin >> bx[i] >> by[i];

	FOR(i, 0, n) FOR(j, 0, m) {
		rbx[i][j] = rx[i] - bx[j];
		rby[i][j] = ry[i] - by[j];
	}

	FOR(i, 0, n) FOR(j, 0, n) {
		int x1 = min(rx[i], rx[j]);
		int x2 = max(rx[i], rx[j]);
		FOR(k, 0, m) if (x1 <= bx[k] && bx[k] < x2) {
			bal[i][j] += rbx[j][k] * rby[i][k] > rby[j][k] * rbx[i][k] ? 1 : -1;
		}
	}

	int res = 0;
	FOR(i, 0, n) FOR(j, i + 1, n) FOR(k, j + 1, n)
		res += bal[i][j] + bal[j][k] + bal[k][i] == 0;
	cout << res << endl;

	return 0;
}
