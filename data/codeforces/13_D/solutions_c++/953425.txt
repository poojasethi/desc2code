#include<cstdio>
#include<algorithm>
#define FOR(i,a,b) for(int i=a;i<=b;i++)
using namespace std;

int n,m,i,j;
struct dot{
  double x,y;
};

dot r[600],b[600];
int ct[600][600];

double cj(dot a,dot b,dot c,dot d){
  return( (b.x-a.x)*(d.y-c.y)-(d.x-c.x)*(b.y-a.y) );
}

int main(){
  scanf("%d%d",&n,&m);
  FOR(i,1,n)scanf("%lf%lf",&r[i].x,&r[i].y);
  FOR(i,1,n){r[i].x+=181.83833;r[i].y+=303.3303;}
  FOR(i,1,m)scanf("%lf%lf",&b[i].x,&b[i].y);
  FOR(i,1,m){b[i].y+=303.3303;b[i].x+=181.83833;}

  FOR(i,1,n)
    FOR(j,1,n) ct[i][j]=-1;
  FOR(i,1,n)
    FOR(j,i+1,n){
      int ii,jj;
      if((r[i].x*r[j].y-r[j].x*r[i].y)<0){ii=j;jj=i;}
      else {ii=i;jj=j;}
      ct[ii][jj]=0;
      FOR(k,1,m)
        if((r[ii].x*b[k].y-b[k].x*r[ii].y)>0)
          if((r[jj].x*b[k].y-b[k].x*r[jj].y)<0)
            if(cj(r[ii],r[jj],r[ii],b[k])>0)ct[ii][jj]++;
    }
  FOR(i,1,n)
    FOR(j,1,n)
      if(ct[i][j]==-1)ct[i][j]=-ct[j][i];

  int ans=0;
  FOR(i,1,n)
    FOR(j,i+1,n)
      FOR(k,j+1,n)
        if((ct[i][j]+ct[j][k]+ct[k][i])==0)ans++;
  printf("%d",ans);

  //system("pause");
}
