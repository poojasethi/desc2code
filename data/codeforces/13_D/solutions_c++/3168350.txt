#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long LL;
#define N 510
struct point {
	LL x,y;
	friend LL cross(point a,point b,point c)
	{
		return (b.x-a.x)*(c.y-a.y)-(b.y-a.y)*(c.x-a.x);
	}
	friend bool operator < (point a, point b)
	{
		return a.x==b.x?a.y<b.y:a.x<b.x;
	}
	void init()
	{
		scanf("%I64d%I64d",&x,&y);
	}
} red[N], blue[N];
LL f[N][N];
LL n,m,ans;

int main()
{
	scanf("%I64d%I64d",&n,&m);
	for (LL i=0; i<n; i++) red[i].init();
	for (LL i=0; i<m; i++) blue[i].init();
	sort(red,red+n);
	for (LL i=0; i<n; i++) 
		for (LL j=i+1; j<n; j++)
			if (red[i].x<red[j].x) 
				for (LL k=0; k<m; k++)
					if (blue[k].x>=red[i].x && blue[k].x<red[j].x)
						f[i][j]+=(cross(red[i],red[j],blue[k])>0LL);
	for (LL i=0; i<n; i++)
		for (LL j=i+1; j<n; j++)
			for (LL k=j+1; k<n; k++)
				if (f[i][j]+f[j][k]==f[i][k]) ans++;
	printf("%I64d\n",ans);
	return 0;
}
