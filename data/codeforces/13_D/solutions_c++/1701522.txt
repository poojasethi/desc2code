#include <iostream>
#include <cstdio>
#define MAXN 505
using namespace std;

__int64 rx[MAXN],ry[MAXN],bx[MAXN],by[MAXN];
int n,m;
int f[MAXN][MAXN];

int main()
{
    cin>>n>>m;
    for (int i=1;i<=n;i++) cin>>rx[i]>>ry[i];
    for (int i=1;i<=m;i++) cin>>bx[i]>>by[i];
    for (int i=1;i<=n;i++)
        for (int j=1;j<=n;j++)
            if (rx[i]<rx[j])
            {
                for (int k=1;k<=m;k++)
                    if (rx[i]<bx[k] && bx[k]<=rx[j] &&
                        (rx[j]-rx[i])*(by[k]-ry[i])-(bx[k]-rx[i])*(ry[j]-ry[i])>0 )
                        f[i][j]++;
                f[j][i]=-f[i][j];
            }
    int ans=0;
    for (int i=1;i<=n;i++)
        for (int j=i+1;j<=n;j++)
            for (int k=j+1;k<=n;k++)
                if (f[i][j]+f[j][k]==f[i][k])
                    ans++;
    cout<<ans<<endl;
}
