#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define N 505
using namespace std;
int n,m;
struct point{
    long long x,y;
}a[N],b[N];
int num[N][N];
bool cmp(point a,point b)
{   if(a.x!=b.x)
    return a.x<b.x;
    return a.y<b.y;
}
long long xmulti(point a,point b,point c)
{
    //ab*ac;
    return (b.x-a.x)*(c.y-a.y)-(b.y-a.y)*(c.x-a.x);
}
int main()
{
    int i,j,k;
    cin>>n>>m;
    for(i=0;i<n;i++)
    {
        cin>>a[i].x>>a[i].y;
    }
    sort(a,a+n,cmp);
    for(i=0;i<m;i++)
    {
        cin>>b[i].x>>b[i].y;
    }
    for(i=0;i<n;i++)
    for(j=i+1;j<n;j++)
    {
        if(a[i].x==a[j].x) continue;
        for(k=0;k<m;k++)
        {
            if(b[k].x>a[i].x&&b[k].x<=a[j].x&&xmulti(a[i],b[k],a[j])<0) num[i][j]++;
        }
    }
    int ans=0;
    for(i=0;i<n;i++)
    for(j=i+1;j<n;j++)
    for(k=j+1;k<n;k++)
    {
        if(num[i][j]+num[j][k]==num[i][k]) ans++;
    }
    cout<<ans<<endl;
  // while(1);
}
