#include <bits/stdc++.h>

using namespace std;

typedef long long LL;
typedef pair<LL,LL> ii;
typedef vector<ii> vii;

LL cross(LL x1, LL y1, LL x2, LL y2){
	return (x1*y2-x2*y1);
}

int main(int argc, char const *argv[])
{
	int r,b;scanf("%d%d",&r,&b);
	int mat[r+1][r+1];memset(mat,0,sizeof mat);
	vii red;vii blue;
	for(int i=0;i<r;i++){
		LL x,y;scanf("%I64d%I64d",&x,&y);
		red.push_back(make_pair(x,y));
	}

	for(int i=0;i<b;i++){
		LL x,y;scanf("%I64d%I64d",&x,&y);
		blue.push_back(make_pair(x,y));
	}
	sort(red.begin(),red.end());
	sort(blue.begin(),blue.end());
	for(int i=0;i<r;i++){
		ii fr = red[i];
		for(int j=i+1;j<r;j++){
			ii sr = red[j];
			for(int k=0;k<b;k++){
				ii bp = blue[k];
				int bx,by,rx,ry;
				bx = bp.first;
				by = bp.second;
				rx = sr.first-fr.first;
				ry = sr.second-fr.second;
				bx-=fr.first;
				by-=fr.second;
				if(cross(rx,ry,bx,by)<0 && (bx>=0 && bx<rx))mat[i][j]++;
			}
		}
	}
	int cnt=0;
	for(int i=0;i<r;i++){
		for(int j=i+1;j<r;j++){
			for(int k=j+1;k<r;k++){
				cnt+=(mat[i][j]+mat[j][k]-mat[i][k]==0);
			}
		}
	}
	printf("%d\n",cnt);
	return 0;
}
