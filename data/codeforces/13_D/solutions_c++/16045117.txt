#include<iostream>
using namespace std;
#define ff first
#define ss second

typedef pair < long long , long long > ii;

long long f[501][501];
ii red[501], xanh[501];

int main()
{
	//freopen("test.inp", "r", stdin);
	int n, m, ans = 0;
	cin >> n >> m;
	for(int i=1;i<=n;i++)
	cin >> red[i].ff >> red[i].ss;
	for(int i=1;i<=m;i++)
	cin >> xanh[i].ff >> xanh[i].ss;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		{
			if(red[i].ff < red[j].ff)
			{
				for(int k=1;k<=m;k++)
					if(red[i].ff < xanh[k].ff && xanh[k].ff <= red[j].ff && (red[j].ff - red[i].ff) * (xanh[k].ss - red[i].ss) > (xanh[k].ff - red[i].ff) * (red[j].ss - red[i].ss))
						f[i][j]++;
				f[j][i] = -f[i][j];
			}	
		}	
	}
	for (int i=1;i<=n;i++)
        for (int j=i+1;j<=n;j++)
            for (int k=j+1;k<=n;k++)	
            	if (f[i][j] + f[j][k] == f[i][k])
            		ans++;
    cout << ans;
}
