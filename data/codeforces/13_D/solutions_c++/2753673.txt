#include<cstdio>
#include<cstring>
#include<cmath>
#define maxn 505
#define sqr(x) ((x)*(x))
#define eps 1e-6
using namespace std;
typedef long long LL;

int     sum[maxn][maxn];
int     n,m;
struct  Point{
    int x,y;
    void init(){
        scanf("%d%d", &x, &y );
    }
}data[2][maxn];

int     cross( Point a , Point b , Point c){
    return (((LL)a.x-c.x)*(b.y-c.y)-((LL)b.x-c.x)*(a.y-c.y))>0?1:-1;
}
int     main(){
    scanf("%d%d", &n, &m );
    for (int i = 0; i<n; i++) data[0][i].init();
    for (int i = 0; i<m; i++) data[1][i].init();
    for (int i = 0; i<n; i++) 
        for (int j = 0; j<n; j++) if ( data[0][i].x < data[0][j].x ){
            for (int k = 0; k<m; k++) 
                    if ( data[1][k].x >= data[0][i].x && data[1][k].x < data[0][j].x )
                    sum[i][j] += cross(data[0][i],data[0][j],data[1][k])>0?1:-1;
            sum[j][i] = -sum[i][j];
        }
    int ans = 0; 
    for (int i = 0; i<n; i++) 
        for (int j = i+1; j<n; j++)
            for (int k = j+1; k<n; k++) 
                if ( sum[i][j]+sum[j][k]+sum[k][i] == 0 )
                    ans++;
    printf("%d\n" , ans );
    return 0;
}
