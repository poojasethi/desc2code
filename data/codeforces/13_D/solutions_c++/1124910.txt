#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long LL;

struct cor{int x,y;} a[550],b[550];
int n,m,ans,cnt[550][550];

bool operator <(cor a,cor b) { return (a.x<b.x || (a.x==b.x && a.y<b.y)); }
cor operator -(cor a,cor b) { cor tt; tt.x=a.x-b.x; tt.y=a.y-b.y; return tt; }
LL operator *(cor a,cor b) { return (LL)(a.x)*b.y-(LL)(a.y)*b.x; }

int main()
{
  //  freopen("13D.txt","r",stdin);

    cin>>n>>m;
    for (int i=1;i<=n;i++) cin>>a[i].x>>a[i].y;
    for (int i=1;i<=m;i++) cin>>b[i].x>>b[i].y;
    sort(&a[1],&a[n+1]);
    sort(&b[1],&b[m+1]);

    int r=0,l=1;
    for (int i=1;i<=n;i++)
    {
        while (l<=m && b[l].x<a[i].x) l++;
        r=l;
        for (int j=i+1;j<=n;j++)
        {
            while (r<=m && b[r].x<a[j].x) r++;
            for (int k=l;k<r;k++) if ((b[k]-a[i])*(a[j]-a[i])>0) cnt[i][j]++;
        }
    }
    for (int i=1;i<=n;i++)
        for (int j=i+1;j<=n;j++)
            for (int k=j+1;k<=n;k++)
        if (cnt[i][j]+cnt[j][k]==cnt[i][k]) ans++;

    cout<<ans<<endl;
	return 0;
}
