#include <iostream>
using namespace std;

const int MAXN = 500;
long long red[MAXN][2], blue[MAXN][2], cnt[MAXN][MAXN];

long long crossproduct(long long x1, long long y1, long long x2, long long y2)
{
    return x1 * y2 - x2 * y1;
}

int main()
{
    int n, m;
    long long ans;

    cin >> n >> m;

    for(int i = 0; i < n; i ++)
        cin >> red[i][0] >> red[i][1];
    for(int i = 0; i < m; i ++)
        cin >> blue[i][0] >> blue[i][1];

    for(int i = 0; i < n; i ++)
    for(int j = 0; j < n; j ++)
    if(i != j && red[i][0] < red[j][0])
    {
        for(int k = 0; k < m; k ++)
        if(blue[k][0] > red[i][0] && blue[k][0] <= red[j][0] &&
           crossproduct(red[j][0] - red[i][0], red[j][1] - red[i][1], blue[k][0] - red[i][0], blue[k][1] - red[i][1]) > 0)
            cnt[i][j] ++;

        cnt[j][i] = -cnt[i][j];
    }

    ans = 0;
    for(int i = 0; i < n; i ++)
    for(int j = i + 1; j < n; j ++)
    for(int k = j + 1; k < n; k ++)
    if(cnt[i][j] + cnt[j][k] + cnt[k][i] == 0)
        ans ++;

    cout << ans << endl;

    return 0;
}
