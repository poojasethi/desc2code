#include <cstdio>

using namespace std;

int xr[501],yr[501],xb[501],yb[501],s[501][501];

int main(void)
{
  int n,m;
  scanf("%d %d",&n,&m);
  for(int i=1;i<=n;++i)
    scanf("%d %d",&xr[i],&yr[i]);
  for(int i=1;i<=m;++i)
    scanf("%d %d",&xb[i],&yb[i]);
  for(int i=1;i<=n;++i)
  {
    for(int j=1;j<=n;++j)
    {
      if(xr[i]>=xr[j])
        continue;
      long long a=yr[j]-yr[i],b=xr[i]-xr[j];
      long long c=-a*xr[i]-b*yr[i];
      for(int k=1;k<=m;++k)
      {
        if((xr[i]<=xb[k])&&(xb[k]<xr[j])&&(a*xb[k]+b*yb[k]+c>0))
          ++s[i][j];
      }
      s[j][i]=-s[i][j];
    }
  }
  int ans=0;
  for(int i=1;i<n-1;++i)
    for(int j=i+1;j<n;++j)
      for(int k=j+1;k<=n;++k)
        if(s[i][j]+s[j][k]+s[k][i]==0)
          ++ans;
  printf("%d\n",ans);
  return 0;
}
