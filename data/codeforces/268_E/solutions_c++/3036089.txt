#include <stdio.h>
#include <algorithm>
#include <string.h>

using namespace std;

#define N 50010
struct note {
	int l, p;
	bool operator < (const note & o) const {
		return l*p*(100-o.p) > o.l*o.p*(100-p);
	}
} a[N];
int n;

int main() {
	while (scanf("%d", &n) == 1) {
		for ( int i=0; i<n; i++) {
			scanf("%d%d", &a[i].l, &a[i].p);
		}
		sort(a, a+n);
		double ans = 0, x = 0;
		for ( int i=0; i<n; i++) {
			ans += x*(100.0-a[i].p)/100.0 + a[i].l;
			x += a[i].l*a[i].p / 100.0;
		}
		printf("%.9lf\n", ans);
	}
	return 0;
}
