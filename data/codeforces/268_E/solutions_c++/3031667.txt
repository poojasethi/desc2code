#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;

struct pnt
{
	int a,b;
}p[50005];

int cmp(pnt x,pnt y)
{
	return x.a*x.b*(100-y.b)>y.a*y.b*(100-x.b);
}

int main()
{
	int n,i;
	scanf("%d",&n);
	for(i=0;i<n;i++) scanf("%d%d",&p[i].a,&p[i].b);
	sort(p,p+n,cmp);
	double ans=0,avg=0;
	for(i=0;i<n;i++)
	{
		double pp=p[i].b/100.0;
		ans+=p[i].a;
		ans+=(1-pp)*avg;
		avg=(avg+p[i].a)*pp+avg*(1-pp);
	}
	printf("%.10lf\n",ans);
	return 0;
}
