#include <cstdio>
#include <vector>
#include <algorithm>
#define N 50010
using namespace std;

struct song{ double l, p; }a[N];
bool comp(song x, song y){
	return x.p * x.l * (1-y.p) > y.p * y.l * (1-x.p);
}

int n;
double ans = 0, sum = 0;
int main (){
	scanf("%d", &n);
	for(int i = 0; i < n; i++){
		scanf("%lf%lf", &a[i].l, &a[i].p);
		ans += a[i].l; a[i].p /= 100;
	}
	sort(a, a+n, comp);
	for(int i = n-1; i >= 0; i--){
		ans += a[i].p * a[i].l * sum;
		sum += (1 - a[i].p);
	}
	printf("%.10lf\n", ans);
}
