#include<cstdio>
#include<algorithm>
using namespace std;

struct str{
    int p,l;
}w[50000];

bool cmp(str a,str b)
{
    return a.l*a.p*(100-b.p)>b.l*b.p*(100-a.p);
}

int main()
{
    int n,i;
    double s=0,t=0;

    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%d%d",&w[i].l,&w[i].p);

    sort(w,w+n,cmp);

    for(i=0;i<n;i++)
    {
        s+=w[i].l+(1-(double)w[i].p/100)*t;
        t+=(double)w[i].p/100*w[i].l;
    }

    printf("%.9f\n",s);

    return 0;
}
