#include <vector>
#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;

const int MAXN = 50100;
int N, M;

struct Node {
	int p, l;
	bool operator<(const Node& rhs) const {
		return rhs.p * rhs.l * (100 - p) < p * l * (100 - rhs.p);
	}
} a[MAXN];

int main() {
	scanf("%d", &N);
	for (int i = 0; i < N; ++i) {
		scanf("%d%d", &a[i].l, &a[i].p);
	}
	sort(a, a + N);
	double ans = 0, s = 0;
	for (int i = 0; i < N; ++i) {
		ans += a[i].l + (100 - a[i].p) / 100.0 * s;
		s += a[i].p * a[i].l / 100.0;
	}
	printf("%.10lf\n", ans);
	return 0;
}
