#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

struct tt{ int l,p; } a[50100];
int n;
double ans,expl;

bool cmp(tt a,tt b) { return a.l*a.p*(100-b.p)>b.l*b.p*(100-a.p); }

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d%d",&a[i].l,&a[i].p);
	sort(&a[1],&a[n+1],cmp);
	ans=expl=0;
	for (int i=1;i<=n;i++)
	{
		ans+=a[i].l+(100-a[i].p)*expl*0.01;
	//	printf("%.12f\n",ans);
		expl+=double(a[i].p)*a[i].l/100;
	}
	printf("%.12f\n",ans);

	return 0;
}
