#include<cstdlib>
#include<cstdio>
#include<algorithm>
#include<iostream>
#define zz pair< double,pair<double,double> >
using namespace std;
double l,p,ans,sum;
int n;
zz a[50010];
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;++i)
	{
		scanf("%lf%lf\n",&l,&p);
		a[i].second.first=l;
		a[i].second.second= p/=100;
		a[i].first= p==1?1e15:l*p/(1-p);
	}
	sort(a,a+n,greater< zz >());
	for(int i=0;i<n;++i)
	{
		l=a[i].second.first;
		p=a[i].second.second;
		ans+=l+(1-p)*sum;
		sum+=l*p;
	}
	printf("%.9lf",ans);
}
