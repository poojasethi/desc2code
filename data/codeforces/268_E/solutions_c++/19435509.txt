#include<cstdio>
#include<algorithm>
#define eps 1e-9
using namespace std;

const int MAX=50005;

struct Node{
	double l,p;
}a[MAX];

bool cmp(Node a,Node b){
	return a.l*a.p*(1-b.p)-b.l*b.p*(1-a.p)>eps;
}

int main(){
	int n;
	scanf("%d",&n);
	double ans=0;
	for(int i=0;i<n;i++){
		scanf("%lf%lf",&a[i].l,&a[i].p);
		a[i].p/=100.0;
		ans+=a[i].l;
	}
	sort(a,a+n,cmp);
	double tmp=0;
	for(int i=0;i<n;i++){
		ans+=tmp*(1-a[i].p);
		tmp+=a[i].l*a[i].p;
	}
	printf("%.9lf\n",ans);
	return 0;
}
