#include<cstdio>
#include<algorithm>
using namespace std;
int n;
double expl,ans;
struct node
{
       double l,p,g;
}a[50001];
__inline bool cmp(node a,node b)
{
         return a.g*(100-b.p)>b.g*(100-a.p);
}
int main()
{
    int i;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
      scanf("%lf%lf",&a[i].l,&a[i].p);
      a[i].g=a[i].l*a[i].p/100;
    }
    sort(&a[1],&a[n]+1,cmp);
    for(i=1;i<=n;i++)
    {
      ans+=a[i].l+expl*((100-a[i].p)/100.0);
      expl+=a[i].g;
    }
    printf("%.10lf\n",ans);
    return 0;
}
