#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
int t[50005],p[50005],ind[50005],a[50005],b[50005];
bool cmp(int i,int j)
{
    return a[j]*b[i]>a[i]*b[j];
}
int main()
{
    int n,i,j;
    cin>>n;
    for (i=1;i<=n;i++) {cin>>t[i]>>p[i]; ind[i]=i; a[i]=100-p[i]; b[i]=p[i]*t[i];}
    sort(ind+1,ind+1+n,cmp);
    double s=0,l=0;
    for (j=1;j<=n;j++)
    {
        i=ind[j];
        s=s+t[i]+1.0*(100-p[i])/100*l;
        l+=1.0*p[i]/100*t[i];
    }
    printf("%.10lf\n",s);
}
