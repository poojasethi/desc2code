#include <cstdio>
#include <algorithm>

#define MXN 50005
using namespace std;
struct node{
	int t;
	double p;
}d[MXN];
bool operator <(const node &a,const node &b)
{
	return a.t*a.p*(1-b.p)>b.t*b.p*(1-a.p);
}
int n;
double ans=0,len=0;
int main ()
{
	scanf("%d",&n);
	for (int i = 0; i < n; i++)
	{
		scanf("%d %lf",&d[i].t,&d[i].p);
		d[i].p/=100.0;
	}
	sort(d,d+n);
	for(int i=0;i<n;i++)
	{
		ans+=d[i].t;
		ans+=len*(1.0-d[i].p);
		len+=d[i].t*d[i].p;
	}
		
	printf("%lf\n",ans);
		
}
