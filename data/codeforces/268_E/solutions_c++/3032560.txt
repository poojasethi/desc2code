#include<algorithm>
#include<iomanip>
#include<iostream>
using namespace std;
struct z{
    int l,p;
}s[50000];
int f(z a,z b){
    return a.l*a.p*(100-b.p)>b.l*b.p*(100-a.p);
}
int main(){
    int n;
    double d=0,a=0;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>s[i].l>>s[i].p;
    sort(s,s+n,f);
    for(int i=0;i<n;i++)
        a+=s[i].l+d*(1-s[i].p/100.0),d+=s[i].l*s[i].p/100.0;
    cout<<setprecision(15)<<a;
}
