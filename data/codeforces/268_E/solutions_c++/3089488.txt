#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=50005;
struct Node {double x,p;}a[maxn];
double sum,ans;int n;
int cmp(Node a,Node b)
{
	return (1-a.p)*b.p*b.x<(1-b.p)*a.p*a.x;
}
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
		scanf("%lf%lf",&a[i].x,&a[i].p),a[i].p/=100;
	sort(a+1,a+n+1,cmp);
	for(int i=1;i<=n;i++)
		ans+=sum*(1-a[i].p)+a[i].x,sum+=a[i].x*a[i].p;
	printf("%.9lf",ans);
	return 0;
}
