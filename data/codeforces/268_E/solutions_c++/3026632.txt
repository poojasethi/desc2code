#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;
int n;
double f, g, bai = 100.0;
struct song{
	int l, p;
}a[100000];
bool cpr(song x, song y) { return x.l*x.p*(100-y.p) > y.l*y.p*(100-x.p); }
int main()
{
	scanf("%d", &n);
	f = g = 0;
	for (int i = 1; i <= n; ++i) { scanf("%d%d", &a[i].l, &a[i].p); f += a[i].l; }
	sort(a+1, a+n+1, cpr);
	for (int i = 1; i <= n; ++i)
	{
		f += g*(100-a[i].p)/bai;
		g += a[i].l*a[i].p/bai;
	}
	printf("%.9lf\n", f);
	return 0;
}
