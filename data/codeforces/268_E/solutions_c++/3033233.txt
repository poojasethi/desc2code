#include<stdio.h>
#include<algorithm>
using namespace std;
struct point{
	double x,y;
}p[50010];
bool cmp(point a,point b){
	return a.y*a.x*(1-b.y)>b.y*b.x*(1-a.y);
}
double sum[50010];
int main(){
	int n,i;
	double as=0;
	scanf("%d",&n);
	for(i=0;i<n;i++){
	    scanf("%lf%lf",&p[i].x,&p[i].y);
		as+=p[i].x;
		p[i].y/=100;
	}
	sort(p,p+n,cmp);
	for(i=n-1;i>=0;i--)
	    sum[i]=1-p[i].y+sum[i+1];
	for(i=0;i<n;i++){
//		printf("%lf %lf\n",p[i].x,p[i].y);
	    as+=p[i].x*p[i].y*sum[i+1];
	}
	printf("%.10lf\n",as);
	return 0;
}
