#include <cstdio>
#include <algorithm>
using namespace std;

int n;
struct Song{ double l, p; } s[50005];

bool cmp(Song x, Song y) { return x.p * (1 - y.p) * x.l > y.p * (1 - x.p) * y.l; }

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%lf %lf", &s[i].l, &s[i].p);
		s[i].p /= 100;
	}
	sort(s, s + n, cmp);
	double ans = 0, prep = 0;
	for (int i = n - 1; i >= 0; --i) {
		ans += s[i].l;
		if (i < n - 1) ans += s[i].p * s[i].l * prep;
		prep += 1 - s[i].p;
	}
	printf("%.9f\n", ans);
	return 0;
}
