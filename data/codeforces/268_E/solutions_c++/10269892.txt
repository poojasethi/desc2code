#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int n;
struct song{
    int l,p;
    bool operator<(const song & k) const{
        return p*(100-k.p)*l > k.p*(100-p)*k.l;
    }
};
song s[50005];
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%d %d",&s[i].l,&s[i].p);
    sort(s+1,s+n+1);
    double a=0,b=0;
    for(int i=1;i<=n;i++){
        b+=(100-s[i].p)*a+100*s[i].l;
        a+=(s[i].p+.0)*s[i].l/100;
    }
    printf("%.10lf\n",b/100);
    return 0;
}
