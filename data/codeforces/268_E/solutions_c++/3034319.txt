#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;

struct song
{
	int l;
	double p;
};
bool compare(const song&a,const song&b)
{
	return a.l*a.p/(1-a.p) > b.l*b.p/(1-b.p);
}
song s[50001];
int main()
{
	int n;
	double ans = 0,e = 0;;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> s[i].l >> s[i].p; 
		s[i].p /= 100;
		ans += s[i].l; 
	}
	sort(s,s + n,compare);
	for (int i = 0; i < n; i++)
	{
		ans += (1 - s[i].p) * e;
		e += s[i].l * s[i].p;
	}
	cout << endl;
	printf("%.9f\n",ans);
}
