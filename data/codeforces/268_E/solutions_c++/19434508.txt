#include<cstdio>
#include<algorithm>
struct d
{
	int len,p;	
}data[50005];
int cmp(d a,d b)
{
	return a.len*a.p*(100-b.p)>b.len*b.p*(100-a.p);
}
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)scanf("%d%d",&data[i].len,&data[i].p);
	std::sort(data,data+n,cmp);
	double ans=0,zt=0;
	for(int i=0;i<n;i++)
	{
		ans+=data[i].len+zt/100.0*(100-data[i].p);
		zt+=data[i].len/100.0*data[i].p;
	}
	printf("%.10lf",ans);
	return 0;
}
