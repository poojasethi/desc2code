#include<stdio.h>
#include<algorithm>
using namespace std;
struct MUSIC
{
	double p,t;
}music[50005];
bool cmp(MUSIC a, MUSIC b)
{
	return a.p*a.t*(100-b.p)>b.p*b.t*(100-a.p);
}
int main()
{
	int n;
	scanf("%d",&n);
	double ans=0,b=0;
	for(int i=0;i<n;i++)
	{
		scanf("%lf %lf",&music[i].t,&music[i].p);
	}
	sort(music,music+n,cmp);
	for(int i=0;i<n;i++)
	{
		ans+=10000.0*music[i].t+b*(100.0-music[i].p);
		b+=(music[i].p*music[i].t);
	//	printf("%lf\n",ans);
	}
	printf("%.9lf\n",ans/10000.0);
	return 0;
}
