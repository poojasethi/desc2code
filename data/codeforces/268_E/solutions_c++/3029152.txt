#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
int n,m,i,j,k,o,p;
struct  data
{ int x,y;
	double z;
} a[50100];
double ans=0,tol=0,tot=0;
bool cmp(data a,data b)
{
	return a.z*(100-b.y)>b.z*(100-a.y);
}
int main()
{
	cin >> n;
	for(i=0;i<n;i++) scanf("%d%d",&a[i].x,&a[i].y),a[i].z=(double)a[i].y/100*a[i].x;
	sort(a,a+n,cmp);
	tot=1;
	for(i=0;i<n;i++)
	{
		ans+=a[i].x;
		ans+=(1-(double)a[i].y/100)*tol;
		tol+=(double)a[i].y/100*a[i].x;
	}
	printf("%0.9lf\n",ans);
	return 0;
}
