#include<cstdio>
#include<algorithm>
using namespace std;
struct S
{
	double l,p;
};
S ss[50050];
inline bool cmp(const S& a,const S& b)
{
	return a.p*(1.0-b.p)*a.l>b.p*(1.0-a.p)*b.l;
}
int main()
{
	int i,n;
	double ans,e;
	scanf("%d",&n);
	for(i=0;i<n;++i)
	{
		scanf("%lf%lf",&ss[i].l,&ss[i].p);
		ss[i].p/=100.0;
	}
	sort(ss,ss+n,cmp);
	ans=e=0.0;
	for(i=n-1;i>=0;--i)
	{
		ans+=ss[i].l*(ss[i].p*e+1.0);
		e+=1.0-ss[i].p;
	}
	printf("%.12lf\n",ans);
	return 0;
}
