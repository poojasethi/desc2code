#include<cstdio>
#include<algorithm>
using namespace std;
struct Node {
	int x,y;
}a[50020];
bool cmp(Node a,Node b) {
	return (100-a.y)*b.x*b.y>(100-b.y)*a.x*a.y;
}
int main() {
	int n;
	while(~scanf("%d",&n)) {
		for(int i=0;i<n;++i) scanf("%d%d",&a[i].x,&a[i].y);
		sort(a,a+n,cmp);
		double sum=0,ans=0;
		
//		for(int i=0;i<n;++i) printf("%d %d\n",a[i].x,a[i].y);
		
		for(int i=0;i<n;++i) {
			sum+=a[i].x*(1+a[i].y/100.0*ans);
			ans+=(100-a[i].y)/100.0;
		}
		
		printf("%.9lf\n",sum);
	}
	return 0;
}
