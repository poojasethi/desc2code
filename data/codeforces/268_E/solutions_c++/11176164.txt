#include<cstdio>
#include<algorithm>
using namespace std;

double ans,lve;
struct node{
	int l;
	double p;
	bool operator <(const node &o)const{
		return l*p/(1-p)>o.l*o.p/(1-o.p);
	}
}a[100000];
int main(){
	int n;scanf("%d",&n);
	ans=0;lve=0;
	for(int i=0;i<n;i++){
		scanf("%d%lf",&a[i].l,&a[i].p);a[i].p/=100;
	}
	sort(a,a+n);
	for(int i=0;i<n;i++){
		ans+=a[i].l;
		ans+=lve*(1-a[i].p);
		lve+=a[i].l*a[i].p;
	}
	printf("%.20lf\n",ans);
	return 0;
}
