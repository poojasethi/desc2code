#include<cstdio>
#include<algorithm>
using namespace std;

typedef struct
{
int x,y;
}ss;

ss a[50005];

bool cmp(const ss& a,const ss& b)
{
	return a.x*b.y<a.y*b.x;
}

int main()
{
	int n,l,p;
	long long ans,sum,s;
	while(~scanf("%d",&n))
	{
		s=0;
		for(int i=0;i<n;i++)
		{
			scanf("%d%d",&l,&p);
			a[i].x=l*p;
			a[i].y=100-p;
			s+=l;
		}
		sort(a,a+n,cmp);
		ans=0;
		sum=0;
		for(int i=0;i<n;i++)
		{
			ans+=a[i].x*sum;
			sum+=a[i].y;
		}
		printf("%.12f\n",(double)ans/10000+s);
	}
}
