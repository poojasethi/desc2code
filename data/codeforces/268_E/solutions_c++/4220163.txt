#include<cstdio>
#include<algorithm>
#define X first
#define Y second
using namespace std;
pair<int,int> a[50000];
int n,i;
double ans,w;
bool cmp(const pair<int,int> &a,const pair<int,int> &b)
{
    return a.X*a.Y*(100-b.Y)>b.X*b.Y*(100-a.Y);
}
int main()
{
    scanf("%d",&n);
    for(i=0;i<n;++i)scanf("%d%d",&a[i].X,&a[i].Y);
    sort(a,a+n,cmp);
    for(i=0;i<n;++i)ans+=w*(1-a[i].Y/100.0)+a[i].X,w+=a[i].X*a[i].Y/100.0;
    printf("%.10f",ans);
    return 0;
}
