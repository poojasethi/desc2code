#include<iostream>
#include<stdio.h>
#include <algorithm>
struct music
{
int sec;
int pro; 
}playlist[50000];
int cmp(music a,music b)
{
return a.sec*a.pro/100.0*(1-b.pro/100.0)>b.sec*b.pro/100.0*(1-a.pro/100.0);

}
int main()
{

int n=0,i=0;
std::cin>>n;
for(i=0;i<n;i++)
{
std::cin>>playlist[i].sec;std::cin>>playlist[i].pro;

}
std::sort(playlist,playlist+n,cmp);
double x=0.0,y=0.0;
for(i=0;i<n;i++)
{
x=x+playlist[i].sec+(1-playlist[i].pro/100.0)*y;
y=y+playlist[i].sec*playlist[i].pro/100.0;
}
printf("%llf",x);
}
