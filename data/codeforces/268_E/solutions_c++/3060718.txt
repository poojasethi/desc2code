#include <cstdio>
#include <algorithm>
using namespace std;

struct song {
	double l, p;
	bool operator<(const song &x) const {
		return l*p*(1-x.p) < x.l*x.p*(1-p);
	}
} s[50005];

int n;
int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		int l, p;
		scanf("%d %d", &l, &p);
		s[i].l = (double)l, s[i].p = (double)p/100.0;
	}
	sort(s, s+n);

	double ans = 0.0, cump = 0.0;
	for (int i = 0; i < n; i++) {
		ans += s[i].l * (1 + s[i].p*cump);
		cump += 1 - s[i].p;
	}

	printf("%.9f\n", ans);
	return 0;
}
