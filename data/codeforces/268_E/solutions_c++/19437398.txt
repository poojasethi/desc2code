#include<cstdio>
#include<algorithm>
#define N 100010
using namespace std;
struct g{
	int a,b;
}f[N];
int n,i;
double ans,tmp;
bool cmp(g x,g y)
{
	return (x.a*x.b)*(100-y.b)	>(y.a*y.b)*(100-x.b);
} 
int main()
{
	scanf("%d",&n);
	for ( int i=1;i<=n;i++)
	scanf("%d%d",&f[i].a,&f[i].b);
	sort(f+1,f+1+n,cmp);
	for (i=1;i<=n;i++)
	{
		ans+=(100-f[i].b)*1.0/100*(f[i].a+tmp)+f[i].b*1.0/100*f[i].a;
		tmp=tmp+f[i].b*1.0/100*f[i].a;
	}
	printf("%.10lf\n",ans);
}
