#include<algorithm>
#include<iomanip>
#include<iostream>
using namespace std;
struct ss{
    int l;
    double p,k;
}s[50005];
int func(ss a,ss b){
    return a.k>b.k;
}
int main(){
    int n;
    double d=0,ans=0;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>s[i].l>>s[i].p,s[i].p/=100,s[i].k=s[i].l*s[i].p/(1-s[i].p);
    sort(s,s+n,func);
    for(int i=0;i<n;i++)
        ans+=s[i].l+d*(1-s[i].p),d+=s[i].l*s[i].p;
    cout<<setprecision(15)<<ans;
}
