#include<cstdio>
#include<algorithm>
using namespace std;
double ans,l[60000],p[60000],s[60000];
int n,a[60000];
inline bool cmp(int x,int y)
{
	return l[x]*p[x]*(1.0-p[y])>l[y]*p[y]*(1.0-p[x]);
}
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%lf %lf",&l[i],&p[i]);
		p[i]/=100.0;
		a[i]=i;
	}
	sort(a+1,a+n+1,cmp);
	for (int i=1;i<=n;i++)
	{
		s[i]=s[i-1]+l[a[i]]*p[a[i]];
		ans+=s[i-1]*(1.00-p[a[i]]);
		ans+=l[a[i]];
	}printf("%.10lf\n",ans);
	return 0;
}
