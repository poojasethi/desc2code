#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int N=50005;
struct node
{
	double l,p;
	void get(){cin>>l>>p;p/=100.;}
	bool operator<(const node &b)const
	{
		return l*p*(1-b.p)>b.l*b.p*(1-p);
	}
}op[N];
int main()
{
	int n;
	while(scanf("%d",&n)!=EOF)
	{
		for(int i=0;i<n;i++) op[i].get();
		sort(op,op+n);

		double res=0,sum=0;
		for(int i=0;i<n;i++)
		{
			res+=op[i].l;
			res+=sum*(1-op[i].p);
			sum+=op[i].p*op[i].l;
		}
		printf("%.9lf\n",res);
	}
	return 0;
}
