#include<cstdio>
#include<iostream>
#include<algorithm>
#define N 110009
using namespace std;
typedef long long ll;
struct g{
    int p,t;
}a[N];
bool cmp(g a,g b){
    return (a.p*a.t)*(100-b.p)>(b.p*b.t)*(100-a.p);
}
int i,j,k,n,m,q,r,x,y,d,l;
int main(){
    scanf("%d",&n);
    for(i=1;i<=n;i++)
        scanf("%d%d",&a[i].t,&a[i].p);
    sort(a+1,a+n+1,cmp);
    ll tp=0,tm=0;
    for(i=1;i<=n;i++){
        tp+=tm*(100-a[i].p);
        tm+=a[i].t*a[i].p;
        tp+=a[i].t*10000;
    }
    printf("%.12lf\n",tp/10000.0);
    return 0;
}
