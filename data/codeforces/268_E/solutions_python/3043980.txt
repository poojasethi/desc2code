n = int(raw_input())
a = []
for _ in xrange(n):
    x, y = map(int, raw_input().split());
    a.append((x, y))
a.sort(cmp = lambda x, y: (100 - x[1]) * y[0] * y[1] - (100 - y[1]) * x[0] * x[1])
ans = 0.0 
tmp = 0.0
for l, p in a:
    p = p / 100.0
    ans += l + (1 - p) * tmp
    tmp += p * l
print '%.10lf' % ans
