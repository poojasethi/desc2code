def comp((l1, p1), (l2, p2)):
	if l2 * p2 * (1 - p1) > l1 * p1 * (1 - p2):
		return 1
	if l2 * p2 * (1 - p1) < l1 * p1 * (1 - p2):
		return -1
	return 0

n = int(raw_input())
a = 0
lall, pall = 0, 0.0
q = []
for i in xrange(n):
	l, p = map(int, raw_input().split())
	lall += l
	pall += (100.0 - p) / 100
	q.append((l, p * 1.0 / 100))
ans = lall
for l, p in sorted(q, cmp = comp):
	pall -= 1 - p
	ans += p * pall * l
print "%.10f" % ans
