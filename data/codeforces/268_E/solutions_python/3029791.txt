from sys import stdin
def main():
    n = int(stdin.readline())
    d = []
    ans = 0
    q = 0
    for i in xrange(n):
        l, p = map(int, stdin.readline().split())
        ans += l
        d.append((l * p, 100 - p))
        q += 100 - p
    d.sort(cmp = lambda x, y: x[1] * y[0] - x[0] * y[1])
    for i in xrange(n):
        q -= d[i][1]
        ans += d[i][0] * q / 10000.0
    print "%.12f" % ans
main()
