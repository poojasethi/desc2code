n = int(raw_input())

lArr = []
pArr = []

for i in range(n):
	inpCols = raw_input().split()
	lArr += [float(inpCols[0])]
	pArr += [float(inpCols[1])]

sortIndArr = []

for i in range(n):
	if (pArr[i] == 100.0):
		sortIndArr += [lArr[i] * 10000.0]
	else:
		sortIndArr += [lArr[i] * pArr[i] / (100.0 - pArr[i])]

sortedIndArr = sorted(enumerate(sortIndArr), key=lambda x:x[1], reverse=True)

# print sortedIndArr

lRealArr = []
pRealArr = []
pSumArr = []

for j in range(n):
	lRealArr += [lArr[sortedIndArr[j][0]]];
	pRealArr += [pArr[sortedIndArr[j][0]]];
	if (j == 0):
		pSumArr += [pRealArr[0]]
	else:
		pSumArr += [pSumArr[j-1] + pRealArr[j]]

totalSum = float(0)

for j in range(n):
	totalSum += lRealArr[j]
	totalSum += lRealArr[j] * pRealArr[j]/100.0 * (float(n) - pSumArr[n-1]/100.0 - float(j+1) + pSumArr[j]/100.0)

print totalSum
