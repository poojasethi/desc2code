#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 50010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int a,fr;
} f[N][110];
int i,j,k,n,m,s;
int g[200][N];
char p[N],pp[N];
void work(int i,int j) {
	if (!i) return;
	work(i-1,f[i][j].fr);
	if (f[i][j].fr<j) pp[++s]=p[i];
}
int main() {
	scanf("%s",p+1); n=strlen(p+1);
	For(i,1,n) {
		For(j,1,190) g[j][i]=g[j][i-1];
		g[p[i]][i]=i;
	}
	f[0][0].a=n+1;
	For(i,0,n-1) {
		For(j,0,99) if (f[i][j].a>i) {
			f[i+1][j].a=f[i][j].a; f[i+1][j].fr=j;
		}
		For(j,0,99)
		if (f[i][j].a>i+1) {
			int A=g[p[i+1]][f[i][j].a-1],B=A==i+1?j+1:j+2;
			if (A&&f[i+1][B].a<A) {
				f[i+1][B].a=A;
				f[i+1][B].fr=j;
			}
		}
	}
	for (i=100;i;i--)For(j,1,n) if (f[j][i].a>=j) {
		work(j,i);
		For(k,1,s) printf("%c",pp[k]);
		for (k=i&1?s-1:s;k;k--) printf("%c",pp[k]);
		return 0;
	}
}
