using namespace std;

int main()
{
	string s;
	cin >> s;
	int n = s.size();

	vector<int> a(26, -1);
	vector<vector<int> > prev(n+1);
	prev[0] = a;
	for(int i=0; i<n; ++i){
		a[s[i]-'a'] = i;
		prev[i+1] = a;
	}

	vector<string> ret(102);
	vector<int> right(102, -1);
	right[0] = n;
	for(int i=0; i<n; ++i){
		for(int j=98; j>=0; j-=2){
			if(right[j] == -1)
				continue;

			int k = prev[right[j]][s[i]-'a'];
			if(i < k && right[j+2] < k){
				ret[j+2] = ret[j] + s[i];
				right[j+2] = k;
				if(i < k - 1)
					ret[j+3] = ret[j+2] + s[k-1];
			}
		}
	}

	for(int i=100; i>0; --i){
		if(ret[i].size() != 0){
			cout << ret[i];
			if(i % 2 == 1)
				ret[i].resize(ret[i].size() - 1);
			reverse(ret[i].begin(), ret[i].end());
			cout << ret[i] << endl;
			return 0;
		}
	}

	cout << s[0] << endl;
	return 0;
}
