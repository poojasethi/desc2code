#include <bits/stdc++.h>
using namespace std;
const int N = 50050;
char ret[105];
string s;
int maxoccur[N][27],lastoccur[27],dp[N][105];
int main(){
	cin>>s;
	int n = s.size(),i,j,k,u,v,c,x;
	s= '#'+s;
	for(i=1;i<=n;++i){
		c=s[i]-'a';
		for(j=0;j<26;++j){
			maxoccur[i][j]=lastoccur[j];
			lastoccur[j]= j==c ? i : lastoccur[j];
		}
	}
	for(i=1;i<=n;++i){
		c = s[i]-'a';dp[i][1]= i;
		dp[i][2]=max(maxoccur[i][c],dp[i-1][2]);
		for(j=1;j<=100;++j){
			u= dp[i-1][j+2], v = maxoccur[dp[i-1][j]][c];
			dp[i][j+2]=max(u,v);
		}
	}
	x=100;
	while(x>0&&!dp[n][x])--x;
	for(i=0,j=x,k=n;i<j;x-=2){
		while(s[k]!=s[dp[k][x]])--k;
		ret[i++]=ret[--j]=s[k--];
	}
	puts(ret);
}
