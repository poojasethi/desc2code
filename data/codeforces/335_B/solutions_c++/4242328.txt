#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <string>
using namespace std;
#define M 3000

string fir,ans1,ans2;
int dp[M][M],l[M][M],r[M][M];

int main()
{
	int i,j,k,len,anslen;
	cin>>fir;
	len=fir.size();
	len=min(len,2600);
	for(i=0;i<len;i++) dp[i][i]=1;
	for(i=1;i<len;i++)
	{
		for(j=0;j+i<len;j++)
		{
			k=j+i;
			if(dp[j+1][k]>dp[j][k])
			{
				dp[j][k]=dp[j+1][k];
				l[j][k]=j+1;
				r[j][k]=k;
			}
			if(dp[j][k-1]>dp[j][k])
			{
				dp[j][k]=dp[j][k-1];
				l[j][k]=j;
				r[j][k]=k-1;
			}
			if(fir[j]==fir[k]&&dp[j][k]<dp[j+1][k-1]+2)
			{
				dp[j][k]=dp[j+1][k-1]+2;
				l[j][k]=j;
				r[j][k]=k;
			}
		}
	}
	anslen=min(dp[0][len-1],100);
	//cout<<anslen<<endl;
	ans1=ans2="";
	int sl=0,sr=len-1,nowl,nowr;
	while(sl<=sr)
	{
		if(anslen<=1)
		{
			if(anslen==1) ans2+=fir[sl];
			break;
		}
		if(fir[sl]==fir[sr])
		{
			anslen-=2;
			ans1+=fir[sl];
			sl++;
			sr--;
			continue;
		}
		nowl=sl;
		nowr=sr;
		sl=l[nowl][nowr];
		sr=r[nowl][nowr];
	}
	cout<<ans1<<ans2;
	for(i=ans1.size()-1;i>=0;i--) cout<<ans1[i];
	cout<<endl;
	return 0;
}
