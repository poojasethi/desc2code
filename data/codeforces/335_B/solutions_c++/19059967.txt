#include <cstdio>
#include <cstring>
int f[2601][2601],g[2601][2601],ans[2601],num[256],n;
char s[50004];
void Print(int l,int r){
	int i=l,j=r;
	while (l<=r){
		if (g[l][r]==1) l++;
		if (g[l][r]==2) r--;
		if (g[l][r]==3) ans[++ans[0]]=s[l],l++,r--;
		if (g[l][r]==4) {ans[++ans[0]]=s[l];break;}
	}
	if (f[i][j]==101){
		for (int i=1;i<=ans[0]-1;i++) printf("%c",ans[i]);
		for (int i=ans[0]-1;i>=1;i--) printf("%c",ans[i]);		
	}
	else{
		for (int i=1;i<=ans[0];i++) printf("%c",ans[i]);
		for (int i=ans[0]-f[1][n]%2;i>=1;i--) printf("%c",ans[i]);		
	}
}
int main(){
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	scanf("%s",s+1); n=strlen(s+1);
	if (n>=2600){
		for (int i=1;i<=n;i++){
			num[s[i]]++;
			if (num[s[i]]==100){
				for (int j=1;j<=100;j++) printf("%c",s[i]);
				return 0;
			}
		}
	}
	for (int i=n;i>=1;i--){
		f[i][i]=1; g[i][i]=4;
		for (int j=i+1;j<=n;j++){
			if (f[i+1][j]>=f[i][j]) f[i][j]=f[i+1][j],g[i][j]=1;
			if (f[i][j-1]>=f[i][j]) f[i][j]=f[i][j-1],g[i][j]=2;
			if (s[i]==s[j]){
				f[i][j]=f[i+1][j-1]+2;
				g[i][j]=3;
			}
			if (f[i][j]>=100){
				Print(i,j);
				return 0;
			}
		}
	}
	Print(1,n);
}
