ll read()
{
    ll x=0,f=1;char ch=getchar();
    while(ch<'0'||ch>'9'){if(ch=='-')f=-1;ch=getchar();}
    while(ch>='0'&&ch<='9'){x=x*10+ch-'0';ch=getchar();}
    return x*f;
}
char c[50005];
int n,K,f[50005][105],p[50005][26];
string a;
int main()
{
    scanf("%s",c+1);
    n=strlen(c+1);K=100;
    for(int i=1;i<=n;i++)
    {
        for(int j=0;j<26;j++)p[i][j]=p[i-1][j];
        p[i][c[i]-'a']=i;
    }
    for(int i=1;i<=n;i++)
        f[i][0]=i+1,f[i][1]=i;
    for(int i=2;i<=n;i++)
        for(int j=2;j<=K;j++)
        {
            f[i][j]=f[i-1][j];
            if(f[i-1][j-2])
                f[i][j]=max(f[i-1][j],p[f[i-1][j-2]-1][c[i]-'a']);
        }
    int len=K,t=n;
    while(!f[n][len])len--;
    while(len>1)
    {
        a.push_back(c[f[t][len]]);
        t=p[t][c[f[t][len]]-'a']-1;
        len-=2;
    }
    cout<<a;
    if(len)cout<<c[f[t][len]];
    reverse(a.begin(),a.end());
    cout<<a;
    return 0;
}
