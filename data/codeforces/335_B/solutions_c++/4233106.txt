#include<cstdio>
#include<string>
#include<iostream>
using namespace std;
const int X=2610;
string dp[X][X],s;
void dfs(int a,int b){
    string &as=dp[a][b];
    if(as.size()!=0)return ;
    if(a==b-1){
        if(s[a]==s[b])
            as=as+s[a]+s[b];
        else as=s[a];
        return ;
    }
    dfs(a+1,b);dfs(a,b-1);
    if(s[a]==s[b]&&dp[a+1][b-1].size()+2<=101){
        as=s[a]+dp[a+1][b-1]+s[b];
        return ;
    }
    if(dp[a+1][b].size()>dp[a][b-1].size())
        as=dp[a+1][b];
    else
        as=dp[a][b-1];
//    cout<<"a="<<a<<" b="<<b<<" dp="<<dp[a][b]<<endl;
}
int cnt[30];
int main(){
    cin>>s;
    int i,j,n=s.size();
    for(i=0;i<n;i++)
        cnt[s[i]-'a']++;

    for(i=0;i<26;i++)if(cnt[i]>=100){
        for(j=0;j<100;j++)
            putchar(i+'a');
        return 0;
    }
    for(i=0;i<n;i++)dp[i][i]=s[i];
    dfs(0,n-1);
    s=dp[0][n-1];
    if(s.size()==101)s.erase(s.begin()+50);
    cout<<s<<endl;
    return 0;
}
