#include <bits/stdc++.h>
using namespace std;

const int N = 5e4+8, K = 100;
char s[N];
int n, f[N][K+1], p[N][26];
string a;

int main(){
	//freopen("input.in", "r", stdin);
	gets(s+1); n = strlen(s+1);
	memset(p, 0, sizeof p);
	for(int i=1; i<=n; i++){
		for(int j=0; j<26; j++) p[i][j] = p[i-1][j];
		p[i][s[i]-'a'] = i;
	}
	for(int i=1; i<=n; i++) {
		f[i][0] = i+1;
		f[i][1] = i;
	}
	for(int i=2; i<=n; i++)
		for(int j=2; j<=K; j++){
			f[i][j] = f[i-1][j];
			if(f[i-1][j-2]) f[i][j] = max(f[i-1][j], p[f[i-1][j-2]-1] [s[i]-'a']);
		}
	int l;
	for(int i=K; i>=1; i--) if(f[n][i]){
		l = i;
		break;
	}
	int i = n;
	while(l>1){
		a+=char(s[f[i][l]]);
		i = p[i][s[f[i][l]]-'a']-1;
		l-=2;
	}
	cout << a;
	if(l) cout << s[f[i][l]];
	reverse(a.begin(), a.end());
	cout << a;
}
