//In the name of God
#include <iostream>
#include <algorithm>
using namespace std;

int cnt[500];
string palin[5000][5000];

int main()
{
  string s;
  cin>>s;
  for(int i=0;i<s.length();++i)
    cnt[s[i]]++;
  for(int i=0;i<500;++i)
    if(cnt[i]>=100)
      {
	for(int j=0;j<100;++j) cout<<char(i);
	cout<<endl;
	return 0;
      }
  string ans=s.substr(0,1);
  for(int i=0;i<s.length();++i)
    palin[i][i]=s.substr(i,1);
  for(int len=2;len<=s.length();++len)
    for(int i=0;i<s.length();++i)
      {
	int j=i+len-1;
	if(j>=s.length()) continue;
	if(s[i]!=s[j])
	  {
	    string s1=palin[i+1][j];
	    string s2=palin[i][j-1];
	    if(s1.length()>s2.length())
	      palin[i][j]=s1;
	    else
	      palin[i][j]=s2;
	}
	else
	  palin[i][j]=s.substr(i,1)+palin[i+1][j-1]+s.substr(j,1);
	if(palin[i][j].length()>=100)
	  {
	    while(palin[i][j].length()>101)
	      palin[i][j]=palin[i][j].substr(1,palin[i][j].length()-2);
	    for(int A=0;A<50;++A)
	      cout<<palin[i][j][A];
	    for(int A=0;A<50;++A)
	      cout<<palin[i][j][49-A];
	    cout<<endl;
	    return 0;
	  }
	if(palin[i][j].length()>ans.length())
	  ans=palin[i][j];
      }
  cout<<ans<<endl;
  return 0;
}
