#include <iostream>
#include <string>
#include <cstdio>
using namespace std;
#define rep(i,n) for(int _n=n,i=0;i<_n;++i)
#define forn(i,a,b) for(int _b=(b),i=(a);i<=_b;++i)
template<class T> inline void checkmax(T& a,T b) { if(b > a) a = b; }

int d[2600][2600], n;
string s;

void print(int l, int r, bool printmid) {
  if (l >= r) {
    if (l == r && printmid) 
      putchar(s[l]);
    return;
  }
  if (d[l][r] == d[l][r - 1])
    print(l, r - 1, printmid);
  else if (d[l][r] == d[l + 1][r])
    print(l + 1, r, printmid);
  else {
    putchar(s[l]);
    print(l + 1, r - 1, printmid);
    putchar(s[r]);
  }
}

int main() {
  cin >> s;
  n = min(2600, (int)s.length());
  int l, r;
  rep(i, n) d[i][i] = 1;
  forn(len, 2, n)
    forn(base, 0, n - len) {
      l = base, r = base + len - 1;
      d[l][r] = max(d[l + 1][r], d[l][r - 1]);
      if (s[l] == s[r]) checkmax(d[l][r], d[l + 1][r - 1] + 2);
      if (d[l][r] >= 100) {
        print(l, r, d[l][r] == 100);
        cout << endl;
        return 0;
      }
    }
  print(0, n - 1, true);
  cout << endl;
  return 0;
}
