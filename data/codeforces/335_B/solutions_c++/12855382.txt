#include<bits/stdc++.h>
#define MX 50005

using namespace std;

int ne[30][MX];
int dp[110][MX];
char outp[110];


int main()
{
    int n;
    string s;
  //freopen("a.txt","r",stdin);
  cin>>s;
  n=s.size();
  for(int i=0;i<=n;i++)
  {
      dp[1][i]=i;
      dp[0][i]=i-1;
     // cout<<i<<endl;
  }
  for(int i='a';i<='z';i++)
  {
      ne[i-'a'][n]=n;
      int w=n;
      for(int j=n-1;j>=0;j--)
      {
          if(s[j]==i)w=j;
          ne[i-'a'][j]=w;
      }
  }
  int mx=1;

  for(int d=2;d<=100;d++)
  {
      dp[d][n]=n;

      for(int i=n-1;i>=0;i--)
      {
          dp[d][i]=dp[d][i+1];

          if(dp[d-2][i+1]<n)
          {

              dp[d][i]=min(dp[d][i],ne[s[i]-'a'][dp[d-2][i+1]+1]);
          }


      }
      if(dp[d][0]<n)mx=d;
  }
  int rem=mx;
  int cnt=0;
  int x=0;
  while(rem>0)
  {
      while(dp[rem][x]==dp[rem][x+1])x++;
      outp[cnt]=outp[mx-cnt-1]=s[x];
      cnt++;
      rem-=2;
      x++;
  }
  outp[mx]=0;
  printf("%s\n",outp);
    return 0;
}
