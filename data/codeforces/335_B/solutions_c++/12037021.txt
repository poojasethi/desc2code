#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;

struct Info {
	int v, l, r;
	Info() {}
	Info(int a, int b, int c) : v(a), l(b), r(c) {}
} f[3333][3333];

char s[55555];
int len;
int tot[333];
int pl, pr;
vector<char> ansl, ansr;

int main() {
	scanf("%s", s);
	len = strlen(s);
	
	for(int i=0;i<len;i++)
		tot[s[i]] ++;
	for(char i = 'a'; i<='z'; i++)
		if(tot[i]>=100) {
			for(int j=0;j<100;j++)
				printf("%c", i);
			return 0;
		}
	
	for(int i=1;i<=len;i++)
		for(int j=0;j+i-1<len;j++) {
			int l = j, r = j+i-1;
			if(l==r) f[l][r] = Info(1, l, r);
			else {
				f[l][r] = f[l+1][r];
				if(f[l][r-1].v > f[l][r].v)
					f[l][r] = f[l][r-1];
				if(s[l] == s[r] && f[l][r].v < f[l+1][r-1].v + 2)
					f[l][r] = Info(f[l+1][r-1].v+2, l, r);
			}
		}
	
	pl = 0, pr = len-1;
	while (pl <= pr) {
		pl = f[pl][pr].l; pr = f[pl][pr].r;
		ansl.push_back(s[pl]);
		if(pl<pr) ansr.push_back(s[pr]);
		pl++;pr--;
	}
	
	if(ansr.size() >= 50) {
		for(int i=0;i<50;i++) printf("%c", ansl[i]);
		for(int i=49;i>=0;i--) printf("%c",ansr[i]);
	} else {
		for(int i=0;i<ansl.size();i++) printf("%c", ansl[i]);
		for(int i=ansr.size()-1;i>=0;i--) printf("%c", ansr[i]);
	}
	
	return 0;
}
