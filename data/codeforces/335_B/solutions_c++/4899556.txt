#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int MAXN = 50010;
const int MAXK = 110;
const int MAXA = 30;

int next[MAXN][MAXA];
int dp[MAXN][MAXK];
char s[MAXN];
int n, ans;

void init() {
    for(int i = 0; i < 26; ++i) {
        int mx = n;
        for(int j = n - 1; j >= 0; --j) {
            if(s[j] == i + 'a') mx = j;
            next[j][i] = mx;
        }
    }
}

void solve() {
    for(int i = 0; i <= n; ++i) {
        dp[i][0] = i - 1;
        dp[i][1] = i;
    }
    ans = 1;
    for(int k = 2; k <= 100; ++k) {
        dp[n][k] = n;
        for(int i = n - 1; i >= 0; --i) {
            dp[i][k] = dp[i + 1][k];
            if(dp[i + 1][k - 2] + 1 < n) dp[i][k] = min(dp[i][k], next[dp[i + 1][k - 2] + 1][s[i] - 'a']);
        }
        if(dp[0][k] < n) ans = k;
    }
}

void print(int pos, int len) {
    if(len <= 0) return ;
    while(dp[pos][len] == dp[pos + 1][len]) ++pos;
    putchar(s[pos]);
    print(pos + 1, len - 2);
    if(len > 1) putchar(s[pos]);
}

int main() {
    scanf("%s", s);
    n = strlen(s);
    init();
    solve();
    print(0, ans);
}
