#include <cstdio>
#include <cstring>
#include <iostream>
#define renew(ii,jj,i,j,v) if (f[ii][jj]+v>f[i][j]) f[i][j]=f[ii][jj]+v,g[i][j][0]=ii,g[i][j][1]=jj
using namespace std;
int N,f[2700][2700],g[2700][2700][2],b[30];
char s[500005];

void print(int i,int j,int t)
{
	if (!i||!j||t==100) return;
	if (i==j) {putchar(s[i]); return;}
	int ii=g[i][j][0],jj=g[i][j][1];
	if (f[i][j]==f[ii][jj]+2) putchar(s[i]),t+=2;
	print(ii,jj,t);
	if (f[i][j]==f[ii][jj]+2) putchar(s[i]);
}

void doit()
{
	scanf("%s",s+1),N=strlen(s+1);
	if (N>=2600)
	{
		for (int i=1; i<=N; i++) b[s[i]-97]++;
		for (int i=0; i<26; i++) if (b[i]>=100)
		{
			for (int j=1; j<=100; j++) putchar(i+97);
			puts("");
			return;
		}
	}
	for (int i=1; i<=N; i++) f[i][i]=1;
	for (int l=2; l<=N; l++)
		for (int i=1,j; (j=i+l-1)<=N; i++)
		{
			renew(i+1,j,i,j,0);renew(i,j-1,i,j,0);
			if (s[i]==s[j]) renew(i+1,j-1,i,j,2);
		}
	print(1,N,0),puts("");
}

int main()
{
	doit();
	return 0;
}
