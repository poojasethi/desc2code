#include <algorithm>
#include <cstring>
#include <cstdio>
int f[3000][3000],a[30],i,j,n,m,x,y;
char s[50010],b[50010];
int main()
{
	scanf("%s\n",s+1);n=strlen(s+1);
	if (n>=2600)
	{
		for (i=1;i<=n;++i) ++a[s[i]-'a'];
		for (i=0;i<26 && a[i]<100;++i);
		for (j=1;j<=100;++j) printf("%c",i+'a');
	}
	else
	{
		for (i=2;i<=n+1;++i) f[i][i-2]=-1;
		for (i=n;i;--i)
			for (j=i;j<=n;++j)
			if (s[i]==s[j]) f[i][j]=f[i+1][j-1]+2;
				else f[i][j]=std::max(f[i+1][j],f[i][j-1]);
		m=f[1][n];
		x=1;y=n;
		for (i=1,j=m;j>0;++i,j-=2)
		{
			for (;f[x+1][y]==j;++x);
			for (;f[x][y-1]==j;--y);
			b[i]=b[m-i+1]=s[y];
			++x;--y;
		}
		if (m<=100) printf("%s",b+1);
		else
		{
			for (i=1;i<=50;++i) printf("%c",b[i]);
			for (i=50;i;--i) printf("%c",b[i]);
		}
	}
}
