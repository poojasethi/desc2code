#include <bits/stdc++.h>
#define fi first
#define se second

using namespace std;

typedef pair<int,int> ii;

const int MAXN = 60000;

ii w[3000][3000];
int N,cnt[30],dn[3000][3000];
char ar[MAXN],pr[MAXN],prr[MAXN];

int main(){
	
	scanf(" %s",ar+1);
	N=strlen(ar+1);
	
	if(N>2574){
		for(int i=1;i<=N;i++)
			cnt[ar[i]-'a']++;
		for(int i=0;i<=25;i++)
			if(cnt[i]>=100){
				for(int j=1;j<=100;j++)
					printf("%c",'a'+i);
				return 0;
			}
	}
	
	for(int i=1;i<=N;i++){
		dn[i][i]=1;
		w[i][i]=ii(i,i);
	}
	for(int i=2;i<=N;i++)
		for(int j=1;j<=N-i+1;j++){
			int k=j+i-1;
			if(ar[j]==ar[k]){
				dn[j][k]=dn[j+1][k-1]+2;
				w[j][k]=ii(j,k);
			}
			else if(dn[j][k-1]>dn[j+1][k]){
				dn[j][k]=dn[j][k-1];
				w[j][k]=w[j][k-1];
			}
			else{
				dn[j][k]=dn[j+1][k];
				w[j][k]=w[j+1][k];
			}
		}
	
	int b=1,e=N,ans=dn[1][N];
	for(int i=1;i<=(ans+1)/2;i++){
		pr[i]=ar[w[b][e].fi];
		pr[ans-i+1]=ar[w[b][e].se];
		int tb=w[b][e].fi+1;
		int te=w[b][e].se-1;
		b=tb; e=te;
	}
	
	if(ans>100){
		ans=100;
		for(int i=1;i<=50;i++)
			prr[i]=prr[100-i+1]=pr[i];
	}
	else
		for(int i=1;i<=ans;i++)
			prr[i]=pr[i];
	
	for(int i=1;i<=ans;i++)
		printf("%c",prr[i]);
	printf("\n");
	
	return 0;
}
