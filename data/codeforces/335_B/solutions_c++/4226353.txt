#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int dp[2700][2700];
char str[100010];
int solve(int l,int r){
    int &ret = dp[l][r];
    if(ret != -1) return ret;
    if(l == r) return ret = 1;
    if(l > r) return ret = 0;
    ret = max(solve(l+1,r),solve(l,r-1));
    if(str[l] == str[r]){
        ret = max(ret,solve(l+1,r-1)+2);
    }
    return ret;
}
void out(int l,int r,int num){
    if(num <= 0) return;
    if(l == r){
        printf("%c",str[l]);
        return;
    }
    if(l > r) return;
    if(dp[l][r] == dp[l+1][r]) {out(l+1,r,num);return;}
    if(dp[l][r] == dp[l][r-1]) {out(l,r-1,num);return;}
    printf("%c",str[l]);
    out(l+1,r-1,num-2);
    printf("%c",str[r]);

}
int main(){
    scanf("%s",str);
    memset(dp,-1,sizeof(dp));
    solve(0,min((int)strlen(str)-1,2600));
    out(0,min((int)strlen(str)-1,2600),100);
    puts("");
    return 0;
}
