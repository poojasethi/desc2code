#!/usr/bin/env python

from itertools import combinations

s = raw_input().strip()
m = int(raw_input())

counters = []
cur_count = [0, 0, 0]
counters.append(cur_count[:])
for c in s:
    cur_count[ord(c) - ord('x')] += 1
    counters.append(cur_count[:])


def check_alg(l, r):
    if r - l < 2:
        return 'YES'
    count_l = counters[l - 1]
    count_r = counters[r]
    count = [c2 - c1 for c1, c2 in zip(count_l, count_r)]
    for c1, c2 in combinations(count, 2):
        if abs(c1 - c2) > 1:
            return 'NO'
    return 'YES'


for _ in xrange(m):
    l, r = map(int, raw_input().split())
    print check_alg(l, r)
