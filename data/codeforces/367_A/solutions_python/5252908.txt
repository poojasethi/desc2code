s = raw_input()
pref = [[0, 0, 0]]
t = {'x': 0, 'y': 1, 'z': 2}
for i in range(len(s)):
    pref.append(pref[-1][:])
    pref[-1][t[s[i]]] += 1
n = input()
for i in range(n):
    l, r = map(int, raw_input().split())
    if r-l+1 < 3:
        print 'YES'
        continue
    c = []
    for j in range(3):
        c.append(pref[r][j] - pref[l-1][j])
    c.sort()
    if c[2] > c[0]+1:
        print 'NO'
    else:
        print 'YES'
