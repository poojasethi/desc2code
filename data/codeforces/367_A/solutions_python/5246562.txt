s = raw_input()
N = len(s)

sums = [[0] for i in range(3)]
for i in range(N):
  for (k, c) in enumerate('xyz'):
    if s[i] == c:
      sums[k].append(sums[k][-1]+1)
    else:
      sums[k].append(sums[k][-1])

m = int(raw_input())

for i in range(m):
  l, r = map(int, raw_input().split())
  l -= 1
  nb = []
  for n in sums:
    nb.append(n[r] - n[l])
  if r-l > 2 and max(nb) - min(nb) > 1:
    print 'NO'
  else:
    print 'YES'
