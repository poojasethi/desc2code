s=raw_input()
m=input()
l=len(s)
sup=[]
sup.append([0,0,0])
x,y,z=0,0,0
for i in s:
	if i=='x':
		x+=1
	if i=='y':
		y+=1
	if i=='z':
		z+=1
	sup.append([x,y,z])
    #print sup
for i in xrange(m):
	a,b=map(int,raw_input().split())
	if b-a<2:
		print "YES"
	else:
		x=sup[b][0]-sup[a-1][0]
		y=sup[b][1]-sup[a-1][1]
		z=sup[b][2]-sup[a-1][2]
		p=[x,y,z]
		p=sorted(p)
		if p[2]-p[0]<=1:
			print "YES"
		else:
			print "NO"
