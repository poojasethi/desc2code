s = raw_input()
n = len(s) 
partials = {'x': [0] * (n + 1), 'y': [0] * (n + 1), 'z': [0] * (n + 1)}
for i in range(1, n + 1):
	for c in 'xyz':
		partials[c][i] = partials[c][i - 1] + (s[i - 1] == c)
m = int(raw_input())
for _ in range(m):
	l, r = map(int, raw_input().split())
	counts = [partials[c][r] - partials[c][l - 1] for c in 'xyz']
	print 'YES' if (sum(counts) < 3) or (max(counts) - min(counts) < 2) else 'NO'
