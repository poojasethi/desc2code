#include <iostream>
#include <cstdlib>
using namespace std;

int x[100001],y[100001],z[100001],Q;
int a,b,t1,t2,t3;

int main(){
	string s;
	cin >> s;
	x[0]=y[0]=z[0]=0;
	for (int i = 0; i < (int)s.size(); i++){
		x[i+1]=x[i]+(s[i]=='x');
		y[i+1]=y[i]+(s[i]=='y');
		z[i+1]=z[i]+(s[i]=='z');
	}
	cin >> Q;
	while(Q--){
		cin >> a >> b;
		t1 = x[b]-x[a-1];
		t2 = y[b]-y[a-1];
		t3 = z[b]-z[a-1];
		if (b-a+1 < 3 || (abs(t1-t2)<=1 && abs(t2-t3)<=1 && abs(t1-t3)<=1)) cout << "YES" << endl;
		else cout << "NO" << endl;
	}
}
