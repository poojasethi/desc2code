#include<iostream>
#include<string.h>
#include<stdlib.h>
using namespace std;
int main()
{
	char c[100001];
	cin>>c;
	int n=strlen(c),m,p,q;
	cin>>m;
	int x[n+1],y[n+1],z[n+1];
	x[0]=0,y[0]=0,z[0]=0;
	for(int i=0;i<n;i++)x[i+1]=x[i]+(c[i]=='x'),y[i+1]=y[i]+(c[i]=='y'),z[i+1]=z[i]+(c[i]=='z');
	while(m--)
	{
		cin>>p>>q;
		int X=x[q]-x[p-1],Y=y[q]-y[p-1],Z=z[q]-z[p-1];
		if(!(X!=Y and Y!=Z and X!=Z) and abs(X-Y)<=1 and abs(X-Z)<=1)cout<<"YES"<<endl;
		else if(q-p<=1) cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
}
