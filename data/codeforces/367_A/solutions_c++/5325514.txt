#include <iostream>
#include <string>
#include <algorithm>
using namespace std;


int main() {
	string read;
	cin >> read;
	int tests;
	int x[100005], y[1000005], z[100005];
	x[0] = y[0] = z[0] = 0;
	for(int i = 1; i <= read.length(); i++) {
		x[i] = x[i-1] + (read[i-1] == 'x');
		y[i] = y[i-1] + (read[i-1] == 'y');
		z[i] = z[i-1] + (read[i-1] == 'z');
	}

	cin >> tests;
	int i, j;
	while (tests--) {
		cin >> i >> j;
		if (max(x[j]-x[i-1],max(y[j]-y[i-1],z[j]-z[i-1]))-min(x[j]-x[i-1],min(y[j]-y[i-1],z[j]-z[i-1])) > 1 && j-i>=2) cout << "NO" << endl;
		else cout << "YES" << endl;
	}
}
