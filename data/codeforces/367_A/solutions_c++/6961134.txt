#include<cstdio>

char a[100002];
int x[100001];
int y[100001];
int z[100001];

int main()
{
	int s,e;
	int i,n;
	int X,Y,Z;
	scanf("%s",a+1);
	for(i=1;a[i];i++)
	{
		x[i]=x[i-1];
		y[i]=y[i-1];
		z[i]=z[i-1];
		if(a[i]=='x')x[i]++;
		if(a[i]=='y')y[i]++;
		if(a[i]=='z')z[i]++;
	}
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&s,&e);
		if(e-s+1<3)puts("YES");
		else
		{
			X=x[e]-x[s-1];
			Y=y[e]-y[s-1];
			Z=z[e]-z[s-1];
			if(Y>=X-1&&Y<=X+1&&Z>=X-1&&Z<=X+1&&X>=Y-1&&X<=Y+1&&Z>=Y-1&&Z<=Y+1&&X>=Z-1&&X<=Z+1&&Y>=Z-1&&Y<=Z+1)puts("YES");
			else puts("NO");
		}
	}
	return 0;
}
