#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int ma,mi,m,l,r,Z,X,Y,i,f[100010][3];
char a[100010];
int main()
{
 scanf("%s",a+1);
 l=strlen(a+1);
 for (i=1;i<=l;i++){f[i][0]=f[i-1][0];
   f[i][1]=f[i-1][1];f[i][2]=f[i-1][2];
   f[i][a[i]-'x']++;}
   
 scanf("%d",&m);
 while (m--){
 scanf("%d%d",&l,&r);
 if (r-l+1<=2){printf("YES\n");
  continue;
}
 Z=f[r][2]-f[l-1][2];
 Y=f[r][1]-f[l-1][1];
 X=f[r][0]-f[l-1][0];
 ma=max(Z,max(Y,X));
 mi=min(Z,min(Y,X));
 if (ma-mi<=1)printf("YES\n");
 else printf("NO\n");
            }
 return 0;
}
