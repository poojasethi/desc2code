#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

char a[100005];
int c[100005][3];

int main()
{
    int n,m,l,r,x,y,z,i,t;

    scanf("%s",a);
    for(n=strlen(a),i=0;i<n;i++)
    {
        c[i+1][0]=c[i][0];
        c[i+1][1]=c[i][1];
        c[i+1][2]=c[i][2];
        c[i+1][a[i]-'x']++;
    }

    for(scanf("%d",&m),i=0;i<m;i++)
    {
        scanf("%d%d",&l,&r);l--;
        x=c[r][0]-c[l][0];
        y=c[r][1]-c[l][1];
        z=c[r][2]-c[l][2];
        t=min(min(x,y),z);
        puts(r-l<=2||(x-t<=1&&y-t<=1&&z-t<=1)?"YES":"NO");
    }

    return 0;
}
