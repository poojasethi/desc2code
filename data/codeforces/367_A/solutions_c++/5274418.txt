#include<stdio.h>
#include<stdlib.h>
#include<math.h>

char s[100010];
int rem[100010][3];

int main()
{
    int l, r, i, j, t[3], m;
    scanf("%s", s+1);
    for(i=1; s[i]; i++)
    {
        for(j=0; j<3; j++)
            rem[i][j] = rem[i-1][j];
        rem[i][s[i]-'x']++;
    }

    scanf("%d", &m);
    while(m--)
    {
        scanf("%d%d", &l, &r);
        for(j=0; j<3; j++)
            t[j] = rem[r][j]-rem[l-1][j];

        if(r-l+1<=2||(abs(t[0]-t[1])<=1&&abs(t[0]-t[2])<=1&&abs(t[1]-t[2])<=1))
            printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}
