#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#define MAXN 100005
using namespace std ;
int N, M ;
int a[5] ;
int sum[MAXN][5] ;
char ss[MAXN] ;

int main()
{
int i, j ;
int lt, rt ;
scanf("%s", ss) ;
N = strlen(ss) ;
for(i = 1; i <= N; i ++)
for(j = 0; j < 3; j ++)
sum[i][j] = sum[i-1][j]+(ss[i-1]-'x'==j) ;
scanf("%d", &M) ;
for(i = 1; i <= M; i ++)
{
scanf("%d %d", &lt, &rt) ;
for(j = 0; j < 3; j ++)
a[j] = sum[rt][j]-sum[lt-1][j] ;
sort(a, a+3) ;
if(rt-lt+1<=2 || a[2]-a[0]<=1) printf("YES\n") ;
else printf("NO\n") ;
}
return 0 ;
}
