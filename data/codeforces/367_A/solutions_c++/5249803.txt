#include<cstdio>
#include<algorithm>
using namespace std;
int dp[3][100002],n,m,l,r,a[3];
char s[300001];
int main(){
    scanf("%s",s+1);
    for(int i=1;s[i];i++)
        for(int j=0;j<3;j++)
            dp[j][i] = dp[j][i-1] + (s[i] == j+'x' ? 1 : 0);
    scanf("%d",&m);
    for(int i=1;i<=m;i++){
        scanf("%d%d",&l,&r);
        for(int j=0;j<3;j++)
            a[j] = dp[j][r]-dp[j][l-1];
        sort(a,a+3);
        if(r-l < 2 || a[2]-a[0]  <= 1) printf("YES\n");
        else printf("NO\n");
    }
        
    return 0;
}
