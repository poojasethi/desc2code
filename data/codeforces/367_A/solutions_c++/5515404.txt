#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

string S;
int M,X[100010],Y[100010],Z[100010];

string solve(int l,int r)
{
	int x=X[r]-X[l-1],y=Y[r]-Y[l-1],z=Z[r]-Z[l-1];
	if ((abs(x-y)<=1&&abs(y-z)<=1&&abs(z-x)<=1)||(r-l<=1)) return "YES";
	return "NO";
}

int main()
{
	cin >> S >> M;
	for (int i=0;i<S.size();i++)
	{
		X[i+1]=S[i]=='x'?X[i]+1:X[i];
		Y[i+1]=S[i]=='y'?Y[i]+1:Y[i];
		Z[i+1]=S[i]=='z'?Z[i]+1:Z[i];
	}
	for (int i=0;i<M;i++)
	{
		int l,r;
		cin >> l >> r;
		cout << solve(l,r) << endl;
	}
	return 0;
}
