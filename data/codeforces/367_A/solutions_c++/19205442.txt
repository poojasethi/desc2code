#include<bits/stdc++.h>
using namespace std;
int A[(int)1e5+10][3];
int main(){
    string s;cin>>s;
for(int i=0;i<s.size();i++){
    for(int j=0;j<3;j++)
        A[i][j]=A[i-1][j];
    A[i][s[i]-'x']++;
}
int q;cin>>q;
int cnt=0;
while(q--){
        cnt++;
    int l,r;cin>>l>>r;
    l--,r--;
    if(r-l+1<3){
        cout<<"YES\n";continue;
    }
    int x=A[r][0]-A[l-1][0];
    int y=A[r][1]-A[l-1][1];
    int z=A[r][2]-A[l-1][2];
    int k=max(abs(x-y),max(abs(y-z),abs(z-x)));
    if(k>=2){
        cout<<"NO\n";
        continue;
    }
    cout<<"YES\n";

}
return 0;}
