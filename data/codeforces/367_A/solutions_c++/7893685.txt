#include <cstdio>
const int N=1e5+5;
char s[N];
int sum[3][N];
int main(){
	scanf("%s",s+1);
	for(int i=1;s[i];i++){
		for(int j=0;j<3;j++) sum[j][i]=sum[j][i-1];
		sum['z'-s[i]][i]++;
	}
	int m; scanf("%d",&m);
	while(m--){
		int l,r; scanf("%d%d",&l,&r); l--;
		bool ok=1;
		for(int j=0;j<3;j++){
			int t=sum[j][r]-sum[j][l]-(r-l)/3;
			if(t!=0&&t!=1)
				ok=0;
		}
		if(r-l<=2) ok=1;
		puts(ok?"YES":"NO");
	}
	scanf("\n");
}
