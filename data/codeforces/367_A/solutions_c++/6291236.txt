#include <algorithm>
#include <iostream>
using namespace std;
const int N = 1e5 + 5;

int m, sum[N][3];
string s;

int main() {
	ios_base::sync_with_stdio(false);
	cin >> s >> m;
	for (int i = 1; i <= s.size(); i++)
		for (int j = 0; j < 3; j++)
			sum[i][j] = sum[i - 1][j] + (s[i - 1] == j + 'x'? 1: 0);
	while (m--) {
		int l, r, a[3];
		cin >> l >> r;
		for (int i = 0; i < 3; i++)
			a[i] = sum[r][i] - sum[l - 1][i];
		cout << (*min_element(a, a + 3) + 1 >= *max_element(a, a + 3) || r - l + 1 < 3 ? "YES": "NO") << '\n';
	}
	return 0;
}
