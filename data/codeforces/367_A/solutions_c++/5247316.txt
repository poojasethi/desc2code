#include <iostream>
#include <string>
#include <cstdio>
#include <algorithm>
using namespace std;

int m;
int x[100001],y[100001],z[100001];
string s;
int p[100];
int main () {
	cin >> s;
	for (int i=0;i<s.size();++i) {
		x[i+1]=x[i]+(s[i]=='x');
		y[i+1]=y[i]+(s[i]=='y');
		z[i+1]=z[i]+(s[i]=='z');
	}
	int a,b,c;
	cin >> m;
	for (int i=1;i<=m;++i) {
		cin >> a >> b;
		if (b-a<=1) {
			printf("YES\n");
			continue;
		}
		p[0] = x[b]-x[a-1];
		p[1] = y[b]-y[a-1];
		p[2] = z[b]-z[a-1];
		sort(p,p+3);
		if (p[2]-p[0]<=1) printf("YES\n");
		else printf("NO\n");
	}
	return 0;
}
