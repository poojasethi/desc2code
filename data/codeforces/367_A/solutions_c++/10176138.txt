#include <bits/stdc++.h>
using namespace std;
char s[100005];
int n, sum[100005][3];
int main() {
	scanf("%s", s+1);
	int n=strlen(s+1);
	for(int i=1; i<=n; ++i) {
		int k=s[i]-'x';
		sum[i][0]+=sum[i-1][0]+(k==0);
		sum[i][1]+=sum[i-1][1]+(k==1);
		sum[i][2]+=sum[i-1][2]+(k==2);
	}
	int m; scanf("%d", &m);
	for(int i=1; i<=m; ++i) {
		int l, r;
		scanf("%d%d", &l, &r); if(r-l<2) { puts("YES"); continue; }
		int a=sum[r][0]-sum[l-1][0], b=sum[r][1]-sum[l-1][1], c=sum[r][2]-sum[l-1][2];
		if(abs(a-b)<2 && abs(a-c)<2 && abs(b-c)<2) puts("YES");
		else puts("NO");
	}
	return 0;
}
