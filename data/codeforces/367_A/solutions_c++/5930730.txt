#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
typedef long long ll;
#define rep(i,n) for (int i=0;i<n;++i)
int n,m,l,r,c[100005][4]; char s[100005];
int main()
{
	scanf("%s",s),n=strlen(s);
	rep(i,n){rep(j,3) c[i+1][j]=c[i][j]; ++c[i+1][s[i]-'x'];}
	scanf("%d",&m);
	rep(i,m){
		scanf("%d%d",&l,&r),--l;
		int x=c[r][0]-c[l][0],y=c[r][1]-c[l][1],z=c[r][2]-c[l][2];
		int c=min(min(x,y),z);
		if (r-l<3 || abs(c-x)<2 && abs(c-y)<2 && abs(c-z)<2) puts("YES");
		else puts("NO");
	}
	return 0;
}
