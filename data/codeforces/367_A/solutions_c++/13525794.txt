#include<bits/stdc++.h>
using namespace std;
int main()
{
    int m,x,y,a[3],dp[100003][5]={0};
    string s;cin >> s;
    for(int i =1;i<=s.size();i++)
    {
        for(int j = 0;j <3;j++)
            dp[i][j]=dp[i-1][j];
        dp[i][s[i-1]-'x']++;
    }
    cin >> m;while(m--)
    {
        cin >>x >> y;x--;
        a[0]= dp[y][0]-dp[x][0];
        a[1]= dp[y][1]-dp[x][1];
        a[2]= dp[y][2]-dp[x][2];
        int mn = min(min(a[0],a[1]),a[2]);
        if(y-x<=2||abs(mn-a[0])<2&&abs(mn-a[1])<2&&abs(mn-a[2])<2)puts("YES");
        else puts("NO");
    }
}
