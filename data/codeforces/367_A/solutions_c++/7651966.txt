#include<cstdio>
#include<cstring>
#define N 100005
using namespace std;
char a[N];
int sumx[N],sumy[N],sumz[N];
int n,i,X,Y,Z,T,L,R;
int main()
{
  scanf("%s",a+1);n=strlen(a+1);
  for (i=1;i<=n;i++)
  {
    sumx[i]=sumx[i-1]+(a[i]=='x');
    sumy[i]=sumy[i-1]+(a[i]=='y');
    sumz[i]=sumz[i-1]+(a[i]=='z');
  }
  scanf("%d",&T);
  while (T--)
  {
    scanf("%d%d",&L,&R);
    if (R-L<2) {puts("YES");continue;}
    X=sumx[R]-sumx[L-1];
    Y=sumy[R]-sumy[L-1];
    Z=sumz[R]-sumz[L-1];
    if (X>Y+1||X>Z+1||Y>X+1||Y>Z+1||Z>X+1||Z>Y+1) {puts("NO");continue;}
    puts("YES");
  }
  return 0;
}
