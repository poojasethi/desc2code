#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 200000
char st[N],*buf=st;
int a[N],b[N],c[N],x;
int main(){
	scanf("%s",st);
	for(x=0;*buf;buf++,x++)
		a[x]=((*buf)=='x'),b[x]=((*buf)=='y'),c[x]=((*buf)=='z');
	for(int i=1;i<x;i++)a[i]+=a[i-1],b[i]+=b[i-1],c[i]+=c[i-1];
	scanf("%d",&x);
	while(x--)
	{
		int l,r;
		scanf("%d%d",&l,&r);
		if(r-l+1<=2)printf("YES\n");
		else{
			r--;l-=2;
			int A=a[r],B=b[r],C=c[r];
			if(l>=0)A-=a[l],B-=b[l],C-=c[l];
			printf("%s\n",abs(A-B)<=1&&abs(B-C)<=1&&abs(A-C)<=1?"YES":"NO");		}
	}
}
