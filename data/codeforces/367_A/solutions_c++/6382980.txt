#include <algorithm>
#include <iostream>
#include <string>

using namespace std;

int c[100005][3], d[3];

int main()
{
	string s;
	int m;
	cin >> s >> m;
	for(int i=0;i<s.length();i++)
		for(int j=0;j<3;j++)
			c[i+1][j] = c[i][j] + (s[i] == 'x' + j);
	for(int i=0;i<m;i++)
	{
		int a, b;
		cin >> a >> b;
		a--;
		for(int j=0;j<3;j++)
			d[j] = c[b][j] - c[a][j];
		sort(d, d + 3);
		cout << (b - a <3 || d[2] - d[0] <= 1 ? "YES" : "NO") << endl;
	}
	return 0;
}
