#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

char s[111111];
int n, len, l, r;
int tot[111111][3];
int a[3];

int main() {
	scanf("%s%d", s, &n);
	len=strlen(s);
	for(int i=1;i<=len;i++)
		tot[i][s[i-1]-'x'] = 1;
	for(int i=1;i<=len;i++)
		for(int j=0;j<3;j++)
			tot[i][j] += tot[i-1][j];
	for(int i=0;i<n;i++) {
		scanf("%d%d", &l, &r);
		for(int j=0;j<3;j++)
			a[j] = tot[r][j] - tot[l-1][j];
		sort(a,a+3);
		printf("%s\n", (r-l<2 || (a[2]-a[0])<=1) ? "YES" : "NO");
	}
	
	return 0;
}
