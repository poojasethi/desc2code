#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int N = 222222;
char s[N];
int n;
int  c[3], cc[N][3];

int main() {
  scanf("%s", s+1);
  int n = strlen(s+1);
  for(int i = 1; i <= n; i++) memcpy(cc[i], cc[i-1],sizeof(cc[i])), ++cc[i][s[i]-'x'];
  int m;
  scanf("%d",&m);
 for(int j = 0; j < m; j++){
    int l,r;
    scanf("%d%d",&l,&r);
    for(int i = 0; i < 3; i++) c[i] = cc[r][i]-cc[l-1][i];
    int mx = *max_element(c, c+3);
    int mn = *min_element(c, c+3);
    puts(r-l+1<3 || mx-mn<=1? "YES": "NO");
  }
  return 0;
}
