#include <stdio.h>
#include <string.h>
#include <math.h>
#define M 100010
char s[M];int nx[M],ny[M],nz[M],X,Y,Z;
int main(){
	int i,q,st,en,len;
	scanf("%s",s);
	len=strlen(s);
	for(i=1;i<=len;i++){
		nx[i]=nx[i-1];ny[i]=ny[i-1];nz[i]=nz[i-1];
		if(s[i-1]=='x') nx[i]++;
		if(s[i-1]=='y') ny[i]++;
		if(s[i-1]=='z') nz[i]++;
	}
	scanf("%d",&q);
	while(q--){
		scanf("%d %d",&st,&en);
		if(en-st+1<3) {puts("YES");continue;}
		X=nx[en]-nx[st-1];
		Y=ny[en]-ny[st-1];
		Z=nz[en]-nz[st-1];
		if(fabs(X-Y)<2 && fabs(X-Z)<2 && fabs(Y-Z)<2) puts("YES");
		else puts("NO");
	}
	return 0;
}
