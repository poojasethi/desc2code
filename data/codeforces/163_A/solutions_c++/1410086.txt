#include<cstdio>
#include<cstring>
#define MOD 1000000007
#define maxn 5005
using namespace std; 

char	s[maxn],t[maxn]; 
int		dp[maxn][maxn],ans; 
int 	main()
{
	scanf("%s%s", s , t );
	for (int i = 1; s[i-1]; i++)
		for (int j = 1; t[j-1]; j++)
		{
			if ( s[i-1] == t[j-1] )
			{
				dp[i][j] = (dp[i-1][j-1] + 1)%MOD;  
				ans = (ans + dp[i][j])%MOD; 
			}
			dp[i][j] = (dp[i][j] + dp[i][j-1])%MOD;
		}
	printf("%d\n" , ans );
	return 0; 
}
