#include <iostream>

using namespace std;

#define MOD 1000000007
#define maxn 5005

	int dp[maxn];

int main() {
	string s,t; 
	cin >> s >> t;
	int n = s.size(), m = t.size();
	for (int i = 0; i < m; ++i) {
		for (int j = n - 1; j > -1; --j) {
			if (s[j] == t[i]) {
				dp[j + 1] = (dp[j + 1] + dp[j] + 1) % MOD;
			}
		}
	}
	int ret = 0;
	for (int j = 0; j <= n; ++j) {
		ret = (ret + dp[j]) % MOD;
	}
	cout << ret << endl;
	return 0;
}
