#include<stdio.h>
#include<string.h>
const int bigp=1000000007;
char a[10001],b[10001];
int f[5006][5006];

int main()
{
	int n,m,i,j;
	scanf("%s%s",a+1,b+1);
	n=strlen(a+1);
	m=strlen(b+1);
	for(i=1;i<=n;++i)
		for(j=1;j<=m;++j)
		{
			f[i][j]=f[i][j-1];
			if(a[i]==b[j])
				f[i][j]=(f[i][j]+f[i-1][j-1]+1)%bigp;
		}
	int ans=0;
	for(i=1;i<=n;++i)
		ans=(ans+f[i][m])%bigp;
	printf("%d\n",ans);
	return 0;
}
