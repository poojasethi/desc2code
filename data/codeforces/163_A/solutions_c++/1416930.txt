#include <iostream>
using namespace std;
const int mod = 1000 * 1000 * 1000 + 7;
int sd[5001][5001];
main() {
	string s, t; cin >> s >> t;
	long long res = 0;
	for (int i = s.size() - 1; i >= 0; i--) for (int j = t.size() - 1; j >= 0; j--)
		sd[i][j] = ((long long)sd[i][j + 1] + (s[i] == t[j] ? 1 + sd[i + 1][j + 1] : 0) % mod) % mod;
	for (int i = 0; i < s.size(); i++) res = (res + sd[i][0]) % mod;
	cout << res;
}
