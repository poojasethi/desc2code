#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;
#define modd 1000000007
char a[5010], b[5010];
int f[5010], n, m, ans;
int main()
{
	scanf("%s", a+1); scanf("%s", b+1);
	n = strlen(a+1); m = strlen(b+1);
	ans = 0;
	for (int i = 1; i <= m; ++i)
	{
		for (int j = n; j >= 1; --j) if (a[j] == b[i]) { ans = (ans+f[j-1]+1)%modd; f[j] = (f[j]+f[j-1]+1)%modd; }
	}
	printf("%d\n", ans);
	return 0;
}
