#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int dp[5001][5001];
char s[5001],t[5001];
int main()
{
	cin>>s>>t;
	int n=strlen(s);
	int m=strlen(t);
	for(int i=n-1;i>=0;i--)
		for(int j=m-1;j>=0;j--)
			dp[i][j]+=(dp[i][j+1] + (s[i]==t[j])*(1+dp[i+1][j+1]))%mod;
			
	long long ans=0;
	for(int i=0;i<n;i++)
	{
		ans+=dp[i][0];
		ans%=mod;
	}
	cout<<ans;
	return 0;
}
