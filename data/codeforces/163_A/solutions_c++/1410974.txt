#include <cstdio>
#include <cstring>

#define MAXL 5004
#define MOD 1000000007

char s[MAXL], t[MAXL];

int c[MAXL][MAXL];

int main() {
	scanf("%s %s", s, t);
	int n = strlen(s), m = strlen(t);
	int result = 0;
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j) {
			int e = 0;
			if (s[i-1] == t[j-1])
				e = c[i-1][j-1] + 1;
			result = (result + e) % MOD;
			c[i][j] = (e + c[i][j-1]) % MOD;
		}
	printf("%d\n", result);
	return 0;
}
