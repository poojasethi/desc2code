#include <bits/stdc++.h>
using namespace std;
long long mod=1000000007;
long long dp[5010];
int main()
{
	long long ans=0;
	char a[5010],b[5010];
	scanf("%s%s",a+1,b+1);
	int n=strlen(a+1);
	int m=strlen(b+1);
	for(int i=1;i<=m;i++)
	{
		for(int j=n;j>=1;j--)
		{
			if(a[j]==b[i])
			{
				ans=(ans+dp[j-1]+1)%mod;
				dp[j]=(dp[j]+dp[j-1]+1)%mod;
			}
		}
	}
	printf("%lld\n",ans);
	return 0;
}
