#include<iostream>
using namespace std;
long long mod=1000000007,sd[5001][5001],res;
main() {
	string s,t; cin>>s>>t;
	for(int i=s.size()-1;i>=0;i--) for(int j=t.size()-1;j>=0;j--)
		sd[i][j]=(sd[i][j+1]+(s[i]==t[j]?1+sd[i + 1][j + 1]:0))%mod;
	for(int i=0;i<s.size();i++) res=(res+sd[i][0])%mod;
	cout<<res;
}
