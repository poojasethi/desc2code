#include <cstdio>
#include <cstring>
using namespace std;

const int INF = 1000000007;

char s[5005],t[5005];
int dp[5005][5005],ans;

int main(){
    int n=strlen(gets(s+1));
    int m=strlen(gets(t+1));
    for(int i=1;i<=n;i++) for(int j=1;j<=m;j++){
        if(s[i]==t[j]){
            dp[i][j]=dp[i-1][j-1]+1;
            ans=(ans+dp[i][j])%INF;
        }
        dp[i][j]=(dp[i][j]+dp[i][j-1])%INF;
    }
    printf("%d\n",ans);
}
