#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;
int dp[5001][5001];
int main()
{
    string a,b;
    int res=0,i,j;
    cin>>a>>b;
    for(i=1;i<=a.length();i++)
    {
        for(j=1;j<=b.length();j++)
            dp[i][j]=(((a[i-1]==b[j-1])?dp[i-1][j-1]+1:0)+dp[i][j-1])%MOD;
        res=(res+dp[i][b.length()])%MOD;
    }
    cout<<res;
    return 0;
}
