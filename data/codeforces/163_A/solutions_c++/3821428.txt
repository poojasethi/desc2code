#include<cstdio>
#include<cstring>
using namespace std;
const int oo=1000000007;
const int maxn=5001;
char a[maxn],b[maxn];
int f[maxn][maxn],n,m,s;

int main(){
  int i,j;
  gets(a);
  gets(b);
  n=strlen(a);
  m=strlen(b);
  for (i=0;i<n;i++)
    for (j=0;j<m;j++)
      f[i+1][j+1]=(f[i+1][j]+(a[i]==b[j])*(f[i][j]+1))%oo;
  for (i=0;i<n;i++) s=(s+f[i+1][m])%oo;
  printf("%d\n",s);
  return 0;
}
