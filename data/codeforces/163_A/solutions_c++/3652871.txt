#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
const int mo=1000000007;
int ans,i,j,k,l,n,m,f[6100],la,lb;
char a[6100],b[6100];
int main(){
	scanf("%s%s",a+1,b+1);
	la=strlen(a+1);
	lb=strlen(b+1);
	for(i=1;i<=lb;++i)
		for(j=la;j>=1;--j)if(a[j]==b[i])
			f[j]=(f[j]+f[j-1]+1)%mo;
	for(i=1;i<=la;++i)ans=(ans+f[i])%mo;
	printf("%d\n",ans);
}
