#include <cstdio>
const int N = 5005;
int f[N][N];
char s[N], t[N];
int main ()
{
    int i, j, ans, MOD = 1000000007;
    while (gets (s + 1))
    {
        gets (t + 1);
        for (ans = 0, i = 1; s[i]; ans = (ans + f[i ++][j - 1]) % MOD)
        for (j = 1; t[j]; j ++) f[i][j] = (f[i][j - 1] + (s[i] == t[j] ? f[i - 1][j - 1] + 1 : 0)) % MOD;
        printf ("%d\n", ans);
    }
    return 0;
}
