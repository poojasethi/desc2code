#include <bits/stdc++.h>
using namespace std;
const int MAX=5007,N=5,MOD=(1e9)+7;
int d[MAX][MAX],sum[MAX][MAX];
unsigned long long int ans;
string x,s;
int main()
{
	cin>>s>>x;
	for(int i=1;i<=s.size();i++)
	{
		for(int j=1;j<=x.size();j++)
		{
			if(s[i-1]==x[j-1])
			{
				d[i][j]+=sum[i-1][j-1];
				d[i][j]++;
			}
			sum[i][j]=(sum[i][j-1]+d[i][j])%MOD;
			ans+=d[i][j];
		}
	}
	cout<<ans%MOD<<"\n";
}
