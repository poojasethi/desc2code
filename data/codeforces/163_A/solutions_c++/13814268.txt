#include <cstdio>

char s[5005], t[5005];

int dp[5005][5005], i, j, ans;

const int mod = 1000 * 1000 * 1000 + 7;

int main() {
	scanf("%s%s", s, t);
	
	for (i = 0; s[i]; i++){
		for(j = 0; t[j]; j++){
			dp[i + 1][j + 1] = dp[i + 1][j] % mod;
			if(t[j] == s[i])
				dp[i + 1][j + 1] = (dp[i + 1][j + 1] + dp[i][j] + 1) % mod;
		}
	}

  	for (i = 0; s[i]; i++)
    	ans = (ans + dp[i + 1][j]) % mod;
  
	printf("%d\n", ans);
}
