#include <iostream>
#include <vector>

using namespace std;

const int maxN=5005,mod=1000000000+7;
long long dp[maxN][maxN];
string s,t;
int main(){
	cin >> s >> t;
	for(int i=s.size()-1 ; i>=0 ; i--)
		for(int j=t.size()-1 ; j>=0 ; j--)
			if(s[i]==t[j]) dp[i][j]=(dp[i][j+1]+dp[i+1][j+1]+1)%mod;
			else dp[i][j]=dp[i][j+1];
	long long ans=0;
	for(int i=0 ; i<s.size() ; i++)	ans=(ans+dp[i][0])%mod;
	cout << ans << endl;
	return 0;
}
