#include <bits/stdc++.h>
using namespace std;
const int mod = 1e9+7;
int dp[5001][5001];
int main(){
	string s,t;cin>>s>>t;
	for(int i=s.size()-1;i>=0;--i)
		for(int j=t.size()-1;j>=0;--j)
			dp[i][j]= (dp[i][j+1] + (s[i]==t[j] ? 1+dp[i+1][j+1] : 0) )%mod;
	int ret=0;
	for(int i=0;i<s.size();++i)
		ret = (ret+dp[i][0])%mod;
	cout<<ret<<endl;
}
