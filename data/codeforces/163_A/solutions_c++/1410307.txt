#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int f[5010][5010],n,m,i,j,ans,mod=1000000007;
char a[5010],b[5010];

int main()
{
	scanf("%s",a+1);
	scanf("%s",b+1);
	n=strlen(a+1);
	m=strlen(b+1);
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			f[i][j]=f[i][j-1];
			if(a[i]==b[j]) f[i][j]=(f[i][j]+f[i-1][j-1]+1)%mod;
		}
		ans=(ans+f[i][m])%mod;
	}
	cout<<ans<<endl;
	return 0;
}
