#include <cstdio>

using namespace std;

long f[5005][5005]={0};

char str1[5005]={0};
char str2[5005]={0};

const long mod=1000000007;

int main()
{
 long s=0;
 long i,j;
 long ans=0;
 
 scanf("%s%s",str1+1,str2+1);
 for(i=1;str1[i];i++)
   for(j=1,s=0;str2[j];j++)
     {
      if(str1[i]==str2[j])
       f[i][j]=(s+1)%mod;
      s+=f[i-1][j];
      s%=mod;
      ans+=f[i][j];
      ans%=mod;
     }
 printf("%ld\n",ans);
 
 return 0;
}
