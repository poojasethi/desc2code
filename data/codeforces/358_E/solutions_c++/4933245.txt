using namespace std;
int tot,n,m;
int a[1005][1005];
int x[1000005],y[1000005];
int vis[1005][1005];
bool CC(int x,int y,int dx,int dy)
{
	for(int i=min(0,dx);i<=max(0,dx);i++)
		for(int j=min(0,dy);j<=max(0,dy);j++)
			if(!a[x+i][y+j])
				return 0;
	return 1;
}
bool check(int d)
{
	int top=1,sum=2,ttt=0;
	for(int i=1;i<=top;i++)
	{
		int ss=0;
		if(x[i]<=n-d&&CC(x[i],y[i],d,0))
		{
			sum+=d-1;
			if(vis[x[i]+d][y[i]]<d)
				vis[x[i]+d][y[i]]=d,
				x[++top]=x[i]+d,
				y[top]=y[i],
				sum+=2;
			ss++;
		}
		if(x[i]>d&&CC(x[i],y[i],-d,0))
		{
			sum+=d-1;
			if(vis[x[i]-d][y[i]]<d)
				vis[x[i]-d][y[i]]=d,
				x[++top]=x[i]-d,
				y[top]=y[i],
				sum+=2;
			ss++;
		}
		if(y[i]<=m-d&&CC(x[i],y[i],0,d))
		{
			sum+=d-1;
			if(vis[x[i]][y[i]+d]<d)
				vis[x[i]][y[i]+d]=d,
				x[++top]=x[i],
				y[top]=y[i]+d,
				sum+=2;
			ss++;
		}
		if(y[i]>d&&CC(x[i],y[i],0,-d))
		{
			sum+=d-1;
			if(vis[x[i]][y[i]-d]<d)
				vis[x[i]][y[i]-d]=d,
				x[++top]=x[i],
				y[top]=y[i]-d,
				sum+=2;
			ss++;
		}
		ttt+=ss&1;
	}
	return (sum/2==tot)&&(ttt<3);
}

	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
		{
			scanf("%d",&a[i][j]);
			if(a[i][j])x[1]=i,y[1]=j;
			tot+=a[i][j];
		}
	if(tot<2)
	{
		printf("-1\n");
		return 0;
	}
	vis[x[1]][y[1]]=0x7fffffff;
	bool f=0;
	for(int i=2;i<=n;i++)
		if(check(i))
			printf("%d ",i),
			f=1;
	if(f)puts("");
	else puts("-1");
	return 0;
}
