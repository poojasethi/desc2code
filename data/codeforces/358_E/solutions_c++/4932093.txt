const int maxn=3010;

int r[maxn][maxn],c[maxn][maxn],vis[maxn][maxn],n,m,x,y,t,cnt,k;
int d[4][2]={0,1,1,0,-1,0,0,-1};

int sm(int x,int y,int nx,int ny)
{
	if(x>nx) swap(x,nx);
	if(y>ny) swap(y,ny);
	if(x==nx) return r[x][ny]-r[x][y-1];
	if(y==ny) return c[nx][y]-c[x-1][y];
}

int dfs(int x,int y)
{
	int ret=2,nx,ny,num=0;
	vis[x][y]=k;
	for(int i=0;i<4;i++)
	{
		nx=x+k*d[i][0]; ny=y+k*d[i][1];
		if(nx>=1&&nx<=n&&ny<=m&&ny>=1&&sm(x,y,nx,ny)==k+1)
		{
			num++;
			ret+=k-1;
			if(vis[nx][ny]!=k) ret+=dfs(nx,ny);
		}
	}
	t+=num&1;
	return ret;
}
