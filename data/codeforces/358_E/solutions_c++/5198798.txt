#include<stdio.h>
const int maxn=1050;
int n,m,ma[maxn][maxn],odd=0,ans=0;
int f[maxn*maxn],vis[maxn*maxn]={0};
int dx[4]={-1,0,1,0},dy[4]={0,1,0,-1};
int gcd(int u,int v)
{
	if(!v) return u;
	else return gcd(v,u%v);
}
int find(int x)
{
	if(x!=f[x])
		f[x]=find(f[x]);
	return f[x];
}
int judg(int x,int y)
{
	int i,s=0,s0=0;
	if(!ma[x][y]) return 0;
	for(i=0;i<4;i++)
		if(0<x+dx[i]&&x+dx[i]<=n&&0<y+dy[i]&&y+dy[i]<=m)
			s+=ma[x+dx[i]][y+dy[i]]<<i,s0+=ma[x+dx[i]][y+dy[i]];
	if(!s) ans=-1;
	if(s0&1) odd++;
	return s!=5&&s!=10;
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j,cr,cc[maxn]={0},z=0;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			f[i*m+j]=i*m+j,scanf("%d",&ma[i][j]);
	for(i=1;i<=n;i++)
	{
		cr=0;
		for(j=1;j<=m;j++)
			if(judg(i,j)&&ans!=-1)
			{
				if(cr) f[find(i*m+cr)]=find(i*m+j),ans=gcd(j-cr,ans);
				cr=j;
				if(cc[j]) f[find(cc[j]*m+j)]=find(i*m+j),ans=gcd(i-cc[j],ans);
				cc[j]=i;
			}
	}
	//if(n==400&&m==398) printf("%d...",ans);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			vis[find(i*m+j)]++;
	//for(i=1;i<=n;i++) {for(j=1;j<=m;j++) printf("%d ",f[i*m+j]); printf("\n");}
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{
			if(vis[i*m+j]>1) z++;
			//if(n==400&&m==398&&vis[i*m+j]>1) printf("%d %d  ",i,j);
		}
	//if(n==400&&m==398) printf("/n%d/n",z);
	if(z>1||odd>2) ans=-1;
	if(ans<=1) {printf("-1\n");return 0;}
	for(i=2;i<ans;i++)
		if(!(ans%i)) printf("%d ",i);
	printf("%d\n",ans);
	return 0;
}
