#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;

long long n,a[100010],k,sp,ep,rv,ra,tmp;

int main(){
  
  cin>>n>>k;
  rep(i,n)cin>>a[i];
  sort(a,a+n);

  while(ep<n){
    if(ep)tmp+=(ep-sp)*(a[ep]-a[ep-1]);
    while(tmp>k){
      tmp-=a[ep]-a[sp];
      sp++;
    }
    if(rv<ep-sp+1){
      rv=ep-sp+1;
      ra=a[ep];
    }
    ep++;
  }

  cout<<rv<<" "<<ra<<endl;

}
