#include<bits/stdc++.h>
using namespace std;
int a[100005];
int main()
{
    int n,k,i,j=0,maxa=0,maxv=0;
    long long temp=0;
    cin>>n>>k;
    for(i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    for(i=0;i<n;i++)
    {
        if(i)
            temp+=(long long int)(a[i]-a[i-1])*(i-j);
        while(j<i&&temp>k)
            temp-=a[i]-a[j++];
        if(i-j+1>maxa)
        {
            maxa=i-j+1;
            maxv=a[i];
        }
    }
    cout<<maxa<<" "<<maxv<<endl;
    return 0;
}
