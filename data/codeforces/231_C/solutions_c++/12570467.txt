#include <iostream>
#include <algorithm>
using namespace std;
main()
{
    int n,k;
    cin>>n>>k;
    int a[n];
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    int l=n-1,r=n-1,maxd=0,v=k,maxi=0,t=0;
    sort(a,a+n);
    for(int i=n-1;i>=0;i--){
        while(v-a[i]+a[l]>=0 && l>=0){
            v-=a[i]-a[l];
            l--;
        }
        if(i-l>=maxd){
            maxi=a[i];
            maxd=i-l;
        }
        v+=(a[i]-a[i-1])*(i-l-1);
    }
    if(t==0)cout<<maxd<<" "<<maxi;
}
