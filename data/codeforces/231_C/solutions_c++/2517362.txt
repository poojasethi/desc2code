#include <cstdio>
#include <algorithm>
long long int n,k,max,ans,dizi[100005],sum[100005],p;
int main()
{
    scanf("%I64d %I64d",&n,&k);
    
    for(int i=1;i<=n;i++)
        scanf("%I64d",&dizi[i]);
    std::sort(dizi+1,dizi+n+1);
    for(int i=1;i<=n;i++)
        sum[i]=sum[i-1]+dizi[i];
    for(int i=1;i<=n;i++)
    {
        while(dizi[i]*(i-p)-(sum[i]-sum[p])>k && p<i) p++;
        if(i-p>max){
            max=i-p;
            ans=dizi[i];
        }
    }
    printf("%I64d %I64d\n",max,ans);
}
