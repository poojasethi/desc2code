#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define FOR(i,a,b) for(int i= int(a); i <= int(b);i++)
int n;
int k;
long long sum;
int l;
int res;
int resnum;
int arr[100100];
int main()
{
	cin >> n >> k;
	FOR(i,1,n) scanf("%d",&arr[i]);
	sort(arr+1,arr+n+1);
	l=1;
	FOR(i,1ll,n)
	{
		sum+=arr[i];
		while(k < (i-l+1ll)*arr[i]-sum)
		{
		 sum-=arr[l];
		 l++;
		}
		if(res<i-l+1)
			{
				res=i-l+1;
				resnum = arr[i];
			}
	}
	cout << res << " " << resnum << endl;
	return 0;
}
