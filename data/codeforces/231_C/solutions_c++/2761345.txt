#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
	int n,k,a[100001]={},prev, i , ans = -1 , ans_num ;
	long long sum = 0 ;
	cin>>n>>k;

	for(int i = 0 ; i  < n ; ++i )
	{
		cin>>a[i];
	}

	sort(a,a+n,less<int>());
	

	prev = 0 ;
	for(i = 0 ; i < n  ; ++i )
	{
		sum += a[i];	

		while( sum +k < ((long long)a[i])*(i-prev+1))
		{
			sum -= a[prev];
			++prev;
			
		}

		if(ans < i-prev+1)
		{
			ans = i - prev+1;
			ans_num = a[i];
		}
		

	}

	cout<<ans<<' '<<ans_num<<endl;



}
