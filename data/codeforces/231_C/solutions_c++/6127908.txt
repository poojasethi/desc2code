#include <cstdio>
#include <algorithm>
using namespace std;

long long N, K, a[100001];

int main(){

	scanf("%I64d%I64d",&N,&K);
	for(int i=0; i<N; ++i){
		scanf("%I64d",&a[i]);
	}
	sort(a,a+N);

	long long c=0; long long res=0, resi=0;
	for(int i=0, j=0; i<N; ++i){
		c+=a[i];
		while( (i-j+1)*a[i] - c > K ){
			c-=a[j++];
		}
		if(i-j+1 > res){
			res = i-j+1;
			resi = i;
		}
	}

	printf("%I64d %I64d\n",res,a[resi]);
	return 0;
}
