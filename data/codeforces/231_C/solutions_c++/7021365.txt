//In the name of God

#include <iostream>
#include <algorithm>
using namespace std;
const int MAXN=100100;
long long a[MAXN];
int main()
{
  long long n,k;
  cin>>n>>k;
  for(int i=0;i<n;i++)
    cin>>a[i];
  sort(a,a+n);
  long long f=0,sum=0;
  long long best=1,val=a[0];
  for(int i=1;i<n;i++)
    {
      sum+=(a[i]-a[i-1])*(i-f);
      while(sum>k)
	{
	  sum-=(a[i]-a[f]);
	  f++;
	}
      int ans=i-f+1;
      if(ans>best)
	best=ans,val=a[i];
    }
  cout<<best<<" "<<val<<endl;
  return 0;
}
