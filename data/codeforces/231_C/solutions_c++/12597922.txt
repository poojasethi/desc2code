#include<bits/stdc++.h>	
using namespace std;
long long int n,k,a[100100],jam[100100],j=0;
int main(){
	ios::sync_with_stdio(0);cin.tie(0);
	cin>>n>>k;
	for(long long int i=0;i<n;i++){
		cin>>a[i];
	}
	sort(a,a+n);
	for(long long int i=0;i<n;i++){
		jam[i]=jam[i-1]+a[i];
	}
	long long int ans1=0,ans2=0;
	for(long long int i=0;i<n;i++){
		jam[i]=jam[i-1]+a[i];
		while(((i-j)*a[i])-(jam[i-1]-jam[j-1])>k)j++;
		if(ans1<i-j+1){
			ans1=i-j+1;
			ans2=a[i];
		}
	}
	cout<<ans1<<" "<<ans2;
}
