#include <cstdio>
#include <algorithm>
using namespace std;

typedef pair < int, int >PII;

int n, m, a[100005];

int main()
{
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; i++)
		scanf("%d", a + i);
	sort(a, a + n);
	long long sum = 0;
	PII ans(0, 0);
	for (int i = 0, j = 0; i < n; i++) {
		sum += a[i];
		while ((i - j + 1ll) * a[i] - sum > m)
			sum -= a[j++];
		ans = max(ans, PII(i - j + 1, -a[i]));
	}
	printf("%d %d\n", ans.first, -ans.second);
}
