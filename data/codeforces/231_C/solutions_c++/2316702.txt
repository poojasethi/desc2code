#include <cstdio>
#include <algorithm>
using namespace std;
#define N 100010
#define LL long long
LL n,m,a[N];
int main() {
  register int i,j;
  scanf("%I64d%I64d",&n,&m);
  for (i=0;i<n;i++) {
    scanf("%I64d",a+i);
  }
  sort(a,a+n);

  LL used=0,cnt=0,num;
  for (i=j=0;i<n;i++) {
    used+=(i-j)*(i?a[i]-a[i-1]:0LL);
    while (used>m) {
      used-=a[i]-a[j];
      j++;
    }
    if (cnt<i-j+1) {
      cnt=i-j+1,
      num=a[i];
    }
  }
  printf("%I64d %I64d\n",cnt,num);
  return 0;
}
