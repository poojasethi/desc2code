#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long int ll;

int main()
{
  int n,k;
  cin>>n>>k;
  vector<ll> v(n);
  for (int i=0;i<n;i++)
    cin>>v[i];
  sort(v.begin(),v.end());
  int maximo=1;
  int sol=v[0];
  ll suma=0;
  int ind=0;
  for (int i=1;i<n;i++) {
    suma+=(i-ind)*(v[i]-v[i-1]);
    while (suma>k) {
      suma-=v[i]-v[ind];
      ind++;
    }
    if (i-ind+1>maximo) {
      maximo=i-ind+1;
      sol=v[i];
    }
  }
  cout<<maximo<<" "<<sol<<endl;
}
