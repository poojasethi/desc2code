#include <bits/stdc++.h>
using namespace std;
int a[100001];
int main(){
    long long i,j,s;
    int n,k;
    cin>>n>>k;
    for(i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    int ans1=0,ans2=0;
    for(i=0,j=0,s=0;i<n;i++)
    {
        s+=a[i];
        while((i-j+1LL)*a[i]-s>k)s-=a[j++];
        if(ans1<i-j+1LL)
        {ans1=i-j+1LL,ans2=a[i];}
    }
    printf("%d %d\n",ans1,ans2);
    return 0;
}
