#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
long i,j,k,l,n,m;
long ans,ans2;
long a[200000];
long long b[200000];
int main(){
	scanf("%ld%ld",&n,&m);
	for(i=1;i<=n;++i){
		scanf("%ld",&a[i]);
	}
	std::sort(a+1,a+n+1);
	ans=1;ans2=a[1];
	for(i=1;i<=n;++i){
		b[i]=a[i];
		b[i]+=b[i-1];
	}
	for(i=1,j=0;i<=n;++i){
		while((i-j)*(long long)(a[i])>m+(long long)(b[i]-b[j]))++j;
		if(i-j>ans){
			ans=i-j;
			ans2=a[i];
		}
	}
	printf("%ld %ld\n",ans,ans2);
	//system("pause");
	return 0;
}
