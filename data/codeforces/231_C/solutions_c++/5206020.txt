#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<cstring>
using namespace std;
int n,i,j,ans=0,ans2=0;
long long sum=0,k,a[100005],geshu;
int main(){
	cin>>n>>k;
	for(i=1;i<=n;i++)
		cin>>a[i];
	sort(a+1,a+1+n);
	sum=0;
	ans=geshu=1;
	ans2=a[1];
	for(i=2;i<=n;i++){
		long long ne=(a[i]-a[i-1])*geshu;
		while(ne+sum>k){
			sum-=a[i]-a[i-geshu];
			geshu--;
		}
		sum+=ne;
		geshu++;
		if (geshu>ans){
			ans=geshu;
			ans2=a[i];
		}
	}
	cout<<ans<<" "<<ans2;
    return 0;
}
