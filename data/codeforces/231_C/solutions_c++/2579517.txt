
#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int main(){
	long long n,k;
	long long s[100001];
	cin >> n >> k;
	for(int i=0;i<n;i++) cin >> s[i];
	sort(s,s+n);
	long long cur=0,put=0,sto=0,lim=0,l=0;
	for(int i=0;i<n;i++){
		lim+=(s[i]-cur)*(i-l);
		while(lim>k) lim-=(s[i]-s[l++]);
		cur=s[i];
		while(i+1<n && s[i+1]==cur) i++;
		if(i-l+1>put){
			sto=cur;
			put=i-l+1;
		}
	}
	cout << put << " " << sto << endl;
	return 0;
}
