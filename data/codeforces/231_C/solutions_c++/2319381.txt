#include<stdio.h>
#include<iostream>
#include<algorithm>
int main(){
	int n,k,a[100001],s[100001];
	long long min,ans=1,sum=0,cnt=1,asum=0;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;++i)
		scanf("%d",&a[i]);
	std::sort(a,a+n);
	min=a[0];
	for(int i=1;i<n;++i)
		s[i]=a[i]-a[i-1];
	for(int i=1;i<n;++i){
		while(sum+s[i]*cnt>k){
			sum-=asum;
			--cnt;
			asum-=s[i-cnt];
		}
		sum+=s[i]*cnt;
		asum+=s[i];
		if(++cnt>ans){
			ans=cnt;
			min=a[i];
		}
	}
	std::cout << ans << " " << min;
	return 0;
}
