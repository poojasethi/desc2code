#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll inp[100003]={0};
int main()
{
    ll n,k;cin >> n >> k;
    for(int i=1;i<=n;i++)cin >> inp[i];
    ll cur = 0,count =0,ans = 0,val=inp[0];
    sort(inp+1,inp+n+1);
    for(int i=1;i<=n;i++)
    {
        while(cur+count*(inp[i]-inp[i-1])>k)cur-=inp[i-1]-inp[i-count],count--;
        cur+=count*(inp[i]-inp[i-1]);count++;
        if(count>ans)ans=count,val=inp[i];
    }
    cout << ans <<" "<< val<<endl;
}
