#include<cstdio>
#include<algorithm>
#include<cstdlib>

typedef long long ll;

const int maxn=100010;
int N,K;
int a[maxn];

int main(){
  int l,r,A=0,B,i;
  ll s;
  scanf("%d%d",&N,&K);
  for(i=0;i<N;++i)scanf("%d",a+i);
  if(N==1)return printf("1 %d",a[0]),0;
  std::sort(a,a+N);
  for(l=r=s=0;r<N-1;){
    s+=(ll)(r-l+1)*(a[r+1]-a[r]),++r;
    while(s>K)s-=a[r]-a[l++];
    if(r-l+1>A)A=r-l+1,B=a[r];
  }
  printf("%d %d\n",A,B);
  return 0;
}
