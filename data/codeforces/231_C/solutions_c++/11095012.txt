#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,i,l,r,c,m,k,x;
	scanf("%lld%lld",&n,&k);
	long long int a[n];
	for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
	sort(a,a+n);
	l=0;
	r=1;
	c=1;
	m=a[0];
	x=0;
	while(r<n)
	{
		x+=(a[r]-a[r-1])*(r-l);
		while(x>k)
		{
			x-=(a[r]-a[l]);			
			l++;
			//cout<<r<<" "<<l<<endl;
		}
		if((r-l+1)>c)
		{
			c=r-l+1;
			m=a[r];
		}
		r++;
	}	
	printf("%lld %lld\n",c,m);
	return 0;
}
