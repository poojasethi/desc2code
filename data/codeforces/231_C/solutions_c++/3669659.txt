#include <iostream>
#include <algorithm>

#define ll long long

using namespace std;

int a[100005];

int main()
{
	ll n, k;
	cin >> n >> k;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	sort(a, a + n);
	ll occ = 1, MaxOcc = 1, ans = a[0];
	int j = 0;
	for (int i = 1; i < n; i++)
	{
		k -= (a[i] - a[i - 1]) * occ;
		occ++;
		while (k < 0)
		{
			k += a[i] - a[j];
			occ--;
			j++;
		}
		if (occ > MaxOcc)
		{
			MaxOcc = occ;
			ans = a[i];
		}
	}
	cout << MaxOcc << " " << ans << endl;
}
