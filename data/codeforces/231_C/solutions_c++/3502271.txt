# include <vector>
# include <stdio.h>
# include <algorithm>

using namespace std;

int a,b,an,mx,x,s;

int c[100009];

int main()
{
	scanf("%d %d",&a,&b);
	
	for(int i=0;i<a;i++)
	scanf("%d",&c[i]);
	
	sort(c,c+a);
	
	x=a-1;
	
	for(int i=a-1;i>=0;i--)
	{
		if(i!=a-1)
		s=s-(c[i+1]-c[i])*(i-x);
		
		while(x>=0)
		{
			if(c[i]-c[x]+s<=b)
			{s+=c[i]-c[x];x--;}
			
			else
			break;
		}
		
		if(i-x>=mx)
		{mx=i-x;an=c[i];}
	}
	
	printf("%d %d",mx,an);
	getchar();
	getchar();
}
