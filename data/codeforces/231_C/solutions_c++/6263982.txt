#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long n,k,i,j,ans=0,num=0,sum=0;
    long long data[100001];
    cin>>n>>k;
    for(i=0;i<n;i++) cin>>data[i];
    sort(data,data+n);
    for(i=0,j=0;i<n;i++)
    {
        sum+=data[i];
        while((i-j+1)*data[i]-sum>k)
        {
            sum-=data[j];j++;
        }
        if(num<i-j+1)
        {
            num=i-j+1;
            ans=i;
        }
    }
    cout<<num<<" "<<data[ans]<<endl;
    return 0;
}
