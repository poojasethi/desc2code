#include <cstdio>
#include <algorithm>
using namespace std;
int a[100001];
int main(){
    long long i,j,s;
    int n,k;
    pair<int,int> ans;
    scanf("%d%d",&n,&k);
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    sort(a,a+n);
    ans.first=0;ans.second=0;
    for(i=0,j=0,s=0;i<n;i++)
    {
        s+=a[i];
        while((i-j+1LL)*a[i]-s>k)s-=a[j++];
        if(ans.second<i-j+1LL)
        ans=make_pair(a[i],i-j+1LL);
    }
    printf("%d %d\n",ans.second,ans.first);
    return 0;
}
