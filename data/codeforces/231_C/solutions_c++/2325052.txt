#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std;

long long a[100010],sum[100010];

int main(){
#ifndef ONLINE_JUDGE
    freopen("a.in","r",stdin);
#endif
    int n,k,max=0,now=0,num;
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++)
        scanf("%I64d",a+i);
    sort(a,a+n);
    for(int i=0;i<n;i++){
        sum[i+1]=sum[i]+a[i];
        while((i-now+1)*a[i]-(sum[i+1]-sum[now])>k) now++;
        if(max<i-now+1) max=i-now+1,num=a[i];
    }
    printf("%d %d\n",max,num);
    return 0;
}
