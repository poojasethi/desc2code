#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	__int64 n,k,num,ind,i,j,a[100005],save[100005];
	while(scanf("%I64d%I64d",&n,&k) != EOF)
	{
		for(i=1;i<=n;i++)
			scanf("%I64d",&a[i]);
		sort(a+1,a+n+1);
		save[0]=0;
		for(i=1;i<=n;i++)
			save[i]=save[i-1]+a[i];
		for(i=1,j=0,num=0;i<=n;i++)
		{
			for(;;j++)
				if(j>i||(a[i]*(i-j)-(save[i]-save[j]))<=k)
					break;
			if(i-j>num)
			{
				num=i-j;
				ind=i;
			}
		}
		printf("%I64d %I64d\n",num,a[ind]);
	}
	return 0;
}
