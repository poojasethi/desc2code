#include <iostream>
#include <algorithm>
using namespace std;
int a[100010];
int main()
{
	ios_base::sync_with_stdio(0);
	int n,k;
	cin>>n>>k;
	for(int i=0;i<n;++i) cin>>a[i];
	sort(a,a+n);
	int resN=0, resA=0;
	long long s=0;
	int L=0;
	for(int i=0;i<n;++i) {
		s+=a[i];
		while (s+k < 1LL*a[i]*(i-L+1)) { s-=a[L]; L++; }
		if (resN < i-L+1) { resN=i-L+1; resA=a[i]; }
	}
	cout<<resN<<' '<<resA<<endl;
	return 0;
}
