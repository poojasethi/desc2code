#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int n,k;
	cin>>n>>k;
	int a[n+5];
	for(int i=0;i<n;i++) cin>>a[i];
	if(n==1){cout<<"1 "<<a[0];return 0;}
	sort(a,a+n);
	int ml=0,nl=0,i=n-1,j=n-2;
	while(i)
	{
		while(j>=0 && a[i]-a[j]<=k)
		{
			k-=a[i]-a[j];
			j--;
		}
		if(nl<=i-j)
		{
			nl=i-j;
			ml=a[i];
		}
		k+=(i-j-1)*(a[i]-a[i-1]);
		i--;
	}
	cout<<nl<<" "<<ml<<"\n";
	return 0;
}
