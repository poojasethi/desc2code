#include <cstdio>
#include <iostream>
#include <algorithm>
#define N 100010
#define LL long long
using namespace std;

LL n, k, a[N];

int main (){
	cin >> n >> k;
	for(LL i = 0; i < n; i++)
		scanf("%I64d", &a[i]);
	sort(a, a + n);
	LL j = 0, bst = 1, val = a[0];
	for(LL i = 0; i < n; i++){
		while(j < i && k < 0) k += a[i] - a[j], j++;
		if(i - j + 1 > bst) bst = i - j + 1, val = a[i];
		k -= (i - j + 1) * (a[i+1] - a[i]);
	}
	cout << bst << " " << val << endl;
}
