#include<iostream>
#include<algorithm>
main(){
	int n,k,i,w=-1,e;
	long long r=0,a[100005],p=0;
	std::cin>>n>>k;
	for(i=1;i<=n;i++)
		std::cin>>a[i];
	std::sort(a+1,a+n+1);
	for(i=1;i<=n;i++){
		while(r+p*(a[i]-a[i-1])>k) r-=a[i-1]-a[i-p],p--;
		r+=p*(a[i]-a[i-1]);
		p++;
		if(p>w)
			w=p,e=a[i];
	}
	std::cout<<w<<" "<<e;
}
