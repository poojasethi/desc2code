# include<iostream>
# include<cstdio>
# include<cstring>
# include<algorithm>
using namespace std;
long long a[111111];
int main()
{
	long long n,k,l=0,r=0;
   cin>>n>>k;
   for(long long i=0;i<n;i++)
      cin>>a[i];
   sort(a,a+n);
   long long maxn=0,t,s=0;
   while(r<n)
   {
      s+=(r-l)*(a[r]-a[r-1]);
      while(s>k)
      s-=a[r]-a[l++];
      if(maxn<(r-l+1))
      {
         maxn=r-l+1;
         t=a[r];
      }
      r++;
   }
   cout<<maxn<<" "<<t<<endl;
   return 0;
}
