#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define LL long long
LL a[100005];
int main()
{
    LL n,k;
    cin>>n>>k;
    for(LL i=1;i<=n;++i)
        cin>>a[i];
    sort(a+1,a+n+1);
    LL ans=1,l=a[1],sum=a[1],c=1;
    for(LL i=2;i<=n;++i)
    {
        while((i-c)*a[i]-sum>k)sum-=a[c++];
        sum+=a[i];
        if(ans<i-c+1)ans=i-c+1,l=a[i];
    }
    cout<<ans<<' '<<l<<endl;
}
