#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,k;
	cin >> n >> k ;
	int a[n];
	for(int i=0;i<n;i++) cin >> a[i];
	if(n==1){
		cout << "1 " << a[0] << endl;
		return 0;
	}
	sort(a,a+n);
	int j=n-2,len=0,m;
	for(int i=n-1;i>=0;){
		while(j>=0&&a[i]-a[j]<=k){
			k-=a[i]-a[j];
			j--;
		}
		if(len<=i-j){
			len=i-j;
			m=a[i];
		}
		k+=(i-j-1)*(a[i]-a[i-1]);
		i--;
	}
	cout << len << " " << m << endl;
	return 0;
}
