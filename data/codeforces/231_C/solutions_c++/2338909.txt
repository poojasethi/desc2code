#include <cstdio>
#include <algorithm>

using namespace std;

int a[100001];

int main(void)
{
  int n,m;
  scanf("%d %d",&n,&m);
  for(int i=1;i<=n;++i)
    scanf("%d",&a[i]);
  sort(a+1,a+n+1);
  int ans=0,num=0;
  long long c=0;
  for(int i=1,j=1;i<=n;++i)
  {
    c+=1ll*(i-j)*(a[i]-a[i-1]);
    while((j<i)&&(c>m))
    {
      c-=a[i]-a[j];
      ++j;
    }
    if(i-j+1>ans)
    {
      ans=i-j+1;
      num=a[i];
    }
  }
  printf("%d %d\n",ans,num);
  return 0;
}
