#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll a[100005];
int main()
{
    ll n,k;
    scanf("%I64d %I64d",&n,&k);
    for(int i=0;i<n;i++)
        scanf("%I64d",&a[i]);
    sort(a,a+n);
    ll f=0,ans,sum=0;
    for(ll i=0,j=0;i<n&&j<n;i++)
    {
        sum+=a[i];
        while(a[i]*(i-j+1)-sum>k)
            sum-=a[j++];
        if(i-j+1>f)
        {
            f=i-j+1;
            ans=a[i];
        }
    }
    printf("%I64d %I64d",f,ans);
    return 0;
}
