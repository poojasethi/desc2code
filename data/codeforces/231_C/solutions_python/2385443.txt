#!/usr/bin/env python

import sys

n, k = map(int, next(sys.stdin).split())
a = list(map(int, next(sys.stdin).split()))

a.sort()

psum = [0]
for v in a:
    psum.append(psum[-1] + v)

maxlen = 0
maxval = -1
i = 0
j = 1
while j <= n:
    while (j - i) * a[j - 1] - (psum[j] - psum[i]) > k:
        i += 1
    if j - i > maxlen:
        maxlen = j - i
        maxval = a[j - 1]
    j += 1

print maxlen, maxval
