n,k = tuple(map(int, raw_input().split()))
data = list(map(int, raw_input().split()))

data.sort()
data = [data[0]] + data
left = 1
ans = data[0]
num = 1
white = 0
for r in xrange(1,n+1):
    white += (data[r]-data[r-1])*(r-left)
    while white>k:
        white -= (data[r]-data[left])
        left += 1
    if r-left+1>num:
        ans = data[r]
        num = r-left+1
    elif r-left+1 == num and ans > data[r]:
        ans = data[r]
print num, ans
