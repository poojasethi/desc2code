n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
a.sort()
maxLen, ans = 1, a[0]
j, sum = 0, 0
for i in xrange(1, n):
    sum += (a[i] - a[i - 1]) * (i - j)
    while sum > k:
        sum -= a[i] - a[j];
        j += 1
    if i - j + 1 > maxLen:
        maxLen = i - j + 1
        ans = a[i]
print maxLen, ans
