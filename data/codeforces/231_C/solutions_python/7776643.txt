#!/usr/bin/python3

n, k = list(map(int, raw_input().split()))
arr = sorted(list(map(int, raw_input().split())))
sums = [0]
sum = 0
for x in arr:
	sum += x
	sums.append(sum)
best_count, best_value = 1, arr[0]
pos = 0
for i in range(1, n):
	while True:
		count = i-pos+1
		need = count*arr[i] - sums[i+1] + sums[pos]
		if need <= k:
			if count > best_count:
				best_count = count
				best_value = arr[i]
			break
		pos += 1
print('%d %d' % (best_count, best_value))
