R = lambda: map(int, raw_input().split())
n, k = R()
array = R()
array.sort ()

s = [0] * (n + 1)
for i in xrange(n):
    s[i + 1] = s[i] + array[i]


i, j, ansMax, ansNum = 0, 1, 1, array[0]
while j < n:
    while ((j - i) * array[j] - s[j] + s[i]> k): i += 1
    if j - i + 1 > ansMax:
        ansMax, ansNum = j - i + 1, array[j]
    j += 1

print ansMax, ansNum
