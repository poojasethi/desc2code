r=lambda:map(int,raw_input().split())
n,k=r()
a=sorted(r())
s=0
p=[0]*(n+1)
for i in range(n):
    s += a[i]
    p[i+1] = s
l=0
f=lambda i,x: a[i-1]*(i-x) - p[i]+p[x]-k


M=(0,0)
for i in range(n+1):
   

    while f(i,l)>0: l+=1

    e=(i-l,-a[i-1])
    if M < e:
        M = e
    #print M
        
print M[0],-M[1]
