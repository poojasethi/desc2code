import sys
input = sys.stdin


N, K = map(int, input.readline().split())
A = map(int, input.readline().split())
A.sort()
C = [0] * N
for i in xrange(1, N):
  C[i] = C[i - 1] + A[i - 1]

best = None
j = 0
for i in xrange(0, N):
  if i < N - 1 and A[i] == A[i + 1]:
    continue
  while A[i] * (i - j) - (C[i] - C[j]) > K:
    j += 1
  cur = (i - j + 1, -A[i])
  if best == None or best < cur:
    best = cur

print best[0], -best[1]
