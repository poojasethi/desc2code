n, k = map(int, raw_input().split())
arr = [0] + sorted(map(int, raw_input().split()))
s = arr[:]
for i in xrange(1, len(s)):
    s[i] += s[i - 1]

ans, length = arr[1], 1

h, t = 1, 1
while t <= n:
    if arr[t] * (t - h) - s[t - 1] + s[h - 1] > k:
        h += 1
    else:
        if t - h + 1 > length:
            length = t - h + 1
            ans = arr[t]
        t += 1

print length, ans
