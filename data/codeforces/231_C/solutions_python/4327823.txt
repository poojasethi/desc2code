n,k=[int(s) for s in raw_input().split()]
a=[]
xx=raw_input().split()
for i in range(len(xx)):
    a.append(int(xx[i]))

a.sort()
ansnum=a[0]
anscnt=1
pt=0
cost=0
for i in range(1, n):
    cost += (a[i]-a[i-1])*(i-pt)
    while cost>k:
        cost -= (a[i]-a[pt])
        pt += 1
    if i-pt+1 > anscnt:
        anscnt = i-pt+1
        ansnum = a[i]
print anscnt,ansnum
