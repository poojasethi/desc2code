 #! /usr/bin/env python2

n, k = map(int, raw_input().strip().split(' '))

a = map(int, raw_input().strip().split(' '))

a.sort()

s = [0]
for ai in a:
    s.append(s[-1] + ai)

ans = []

for i in xrange(n):
    l, r = 0, i
    while l < r:
        mid = (l + r) // 2
        if a[i] * (i + 1 - mid) - (s[i + 1] - s[mid]) <= k:
            r = mid
        else:
            l = mid + 1
    if len(ans) == 0 or (i + 1 - l) > ans[0]:
        ans = (i + 1 - l, a[i])

print ans[0], ans[1]
