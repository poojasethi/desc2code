
I=lambda:map(int,raw_input().split())
n,k=I()
a=I()
a.sort()
s=[0]*(n+1)
for i in xrange(1,n+1):
    s[i]=s[i-1]+a[i-1]

def is_ok(cnt):
    global k
    for i in xrange(cnt,n+1):
        if a[i-1]*cnt-(s[i]-s[i-cnt])<=k:
            return i
    return -1

l,r=1,n
while l<=r:
    mid=l+(r-l)/2
    idx=is_ok(mid)
    if idx!=-1:
        num=a[idx-1]
        ans=mid
        l=mid+1
    else: r=mid-1
print ans,num
