n = int(raw_input())
p = map(int, raw_input().split())

#I am going to prepare a vector f such that it compute what is the number of elements that is consecutive and smaller than it

f = [0]*(n+1)
 #to compute this I am going to check if it has been computed if it has not, it will be 0.

for i in xrange(n):
    f[p[i]]=f[p[i]-1]+1

print n-max(f)
