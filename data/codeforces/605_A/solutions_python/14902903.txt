
n=int(raw_input())

cars=map(int, raw_input().split())

max_consecutive=0;
already_seen=[-1]*(n+1)

for car_id in cars:
    prec_number=already_seen[car_id-1]
    if(prec_number>=0):
        already_seen[car_id]=prec_number+1
        max_consecutive=max(max_consecutive, prec_number+1)
    else:
        already_seen[car_id]=0
 
print n-1-max_consecutive
