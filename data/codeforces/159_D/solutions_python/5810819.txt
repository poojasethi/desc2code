#!/user/bin/python

s = raw_input()
n = len(s)

left = [0 for i in range(n)]
right = [0 for i in range(n)]

for i in range(n) : 
    iterList = [(i, i)]
    if i + 1 != n and \
        s[i] == s[i+1] : 
        iterList.append((i, i+1))

    for l, r in iterList : 
        while l >= 0 and \
            r < n and \
            s[l] == s[r] : 
            left[l] += 1
            right[r] += 1
            l -= 1
            r += 1

cnt = 0
for i in range(n-1) : 
    cnt += right[i] * (sum(left[i+1:]))

print cnt
