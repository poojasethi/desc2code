s = raw_input()
l = len(s)

start = [0] * l
end = [0] * l

for i in xrange(l):
    p, q = i, i
    while p>-1 and q<l and s[p]==s[q]:
        start[p] += 1
        end[q] += 1
        p -= 1
        q += 1
    p, q = i, i+1
    while p>-1 and q<l and s[p]==s[q]:
        start[p] += 1
        end[q] += 1
        p -= 1
        q += 1

for i in xrange(1, l):
    end[i] += end[i-1]

ans = 0
for i in xrange(0, l-1):
    ans += end[i] * start[i+1]
print ans
