s=raw_input()
l=len(s)

d=[0]*l
r=[0]*l
for i in xrange(l):
    for j in [0,1]:
        k=i
        m=i+j
        while k >= 0 and m < l and s[k] == s[m]:
            d[m] += 1
            r[k] += 1                        
            m+=1
            k-=1     

s=0
c=0
for i in xrange(l-1,0,-1):
    s += r[i]
    c += s*d[i-1]
    
print c
