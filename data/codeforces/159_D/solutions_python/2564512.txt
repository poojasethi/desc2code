s = raw_input()
n = len(s)
l, r, p = [0] * n, [0] * n, 0
for x, y in [(i, i) for i in range(n)] + [(i - 1, i) for i in range(1, n) if s[i - 1] == s[i]]:
	while x >= 0 and y < n and s[x] == s[y]:
		r[x], l[y] = r[x] + 1, l[y] + 1
		x, y = x - 1, y + 1
for i in range(1, n):
	l[i] += l[i - 1]
for i in range(1, n):
	p += l[i - 1] * r[i]
print p
