#include<cstdio>
#include<iostream>
#include<algorithm>

using namespace std;

long long sz,i,j,l,r,p1[2005],p2[2005],temp,ans;
string s;

int main()
{
	cin >> s; sz=s.size();
	for (i=0; i<sz; i++)
	{
		l=r=i;
		while (l>=0 &&  r<sz && s[l]==s[r])
		{
			++p1[l]; ++p2[r]; 
			--l; ++r;
		}
		l=i; r=i+1;
		while (l>=0 && r<sz && s[l]==s[r])
		{
			++p1[l]; ++p2[r];
			--l; ++r;
		}
	}
	for (i=0; i<sz; i++)
	{
		temp=0;
		for (j=i+1; j<sz; j++) temp+=p1[j];
		ans=(ans+(temp*p2[i]));
	}
	printf("%I64d\n", ans);
}
