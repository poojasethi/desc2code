#include<iostream>
#include<stdio.h>
#include<cstring>
#include<string>
using namespace std;
char s[2050];
int i,j,p[2050],q[2050],l,a,b;
long long r;
int main()
{
    l=strlen(gets(s+1));
    for(i=1;i<=l;i++)
    {
        for(a=i,b=i+1;a&&b<=l&&s[a]==s[b];a--,b++) p[b]++,q[a]++;
        for(a=b=i;a&&b<=l&&s[a]==s[b];a--,b++) p[b]++,q[a]++;
    }

    for(i=l;i;i--) q[i]+=q[i+1];
    for(i=1;i<l;i++) r+=(long long) p[i]*q[i+1];
    cout<<r;
}
