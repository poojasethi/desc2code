#include <iostream>
#include<string>
using namespace std;
long long ans=0,front[2001]={0},back[2001]={0};
int main()
{
    string s;
    cin>>s;
   int  len=s.length();
    int i,x,y;
    for(i=0;i<len;i++)
    {
        for(x=i,y=i;x>=0&&y<len&&s[x]==s[y];x--,y++) front[x]++,back[y]++;
        for(x=i,y=i+1;x>=0&&y<len&&s[x]==s[y];x--,y++) front[x]++,back[y]++;
    }
    for(i=len-2;i>=0;i--) front[i]+=front[i+1];
    for(i=0;i<len;i++) ans+=back[i]*front[i+1];
     cout<<ans<<endl;

    return 0;
}
