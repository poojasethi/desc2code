#include <cstdio>
#include <iostream>
#include <string>
#define MAXN 2000
using namespace std;
string str;
int N;
long long A[MAXN+1][MAXN+1];
long long res;
void f(int x,int y)
{
	while(x>=0 && y<N && str[x]==str[y])
		A[(x--)+1][(y++)+1]=1;
}
int main()
{
	int i,j;
	cin >> str;
	N=str.size();
	for(i=0;i<N;i++)
	{
		A[i+1][i+1]=1;
		f(i,i+1);
		f(i-1,i+1);
	}
	for(i=1;i<=N;i++)
		for(j=2;j<=N;j++)
			A[i][j]+=A[i][j-1];
	for(i=1;i<=N;i++)
		for(j=i+1;j<=N;j++)
			res+=A[i][j-1]*A[j][N];
	cout << res << endl;
	return 0;
}
