#include <cstring>
#include <cstdio>
int f[2010][2010],i,j,n;
long long ans=0;
char s[2010];
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%s",s+1);n=strlen(s+1);
	for (i=1;i<=n;++i) f[i][i-1]=f[i+1][i-1]=1;
	for (i=n;i;--i)
		for (j=i;j<=n;++j)
		if (f[i+1][j-1] && s[i]==s[j]) f[i][j]=1;
	for (i=1;i<=n;++i)
		for (j=i+1;j<=n;++j) f[i][j]+=f[i][j-1];
	for (i=1;i<n;++i)
		for (j=i+1;j<=n;++j) ans+=f[i][j-1]*f[j][n];
	printf("%I64d\n",ans);
}
