#include <cstdio>
#include <cstring>

#define N 2000

char s[N+1];
int n;
long long d[N][N],cnt;

int main(){
    int i,j,k;
    scanf("%s",s);
    n=strlen(s);
    for(j=0;j<n;j++){
        for(i=j;i>=0;i--){
            d[i][j]=(i==j?0:d[i+1][j]);
            for(k=0;i+k<=j;k++)
                if(s[i+k]!=s[j-k])
                    break;
            if(i+k>j)
                d[i][j]++;
        }
    }
    for(i=0;i<n;i++)
        for(j=i+1;j<n;j++)
            cnt+=d[0][i]*d[i+1][j];
    printf("%I64d\n",cnt);
    return 0;
}
