#include<iostream>
using namespace std;
long long l[2003],r[2003],ans=0;
int main()
{string s; cin>>s;
int len=s.size();
for(int i=0;i<len;i++)
 {for(int j=i, k=i;j>=0&&k<len&&s[j]==s[k];j--,k++) {l[j]++;r[k]++;}
  for(int j=i,k=i+1;j>=0&&k<len&&s[j]==s[k];j--,k++) {l[j]++;r[k]++;}
 }
for(int i=1;i<len;i++) {r[i]+=r[i-1]; ans+=r[i-1]*l[i];}
cout<<ans;
return 0;
}
