#include<cstring>
#include<cstdio>
#include <iostream>
using namespace std;
char s[2048];
int p[2048],q[2048],l,a,b,i;
long long r;
int main(){
    l=strlen(gets(s+1));
    for(i=1;i<=l;++i){
        for(a=i,b=i+1;a&&b<=l&&s[a]==s[b];--a,++b)++p[b],++q[a];
        for(a=b=i;a&&b<=l&&s[a]==s[b];--a,++b)++p[b],++q[a];
    }
    long long int w=0;
    long long int il=0;
    for(int i =0;i<=l;i++)
         il+=q[i];
    for(int i =1;i<=l;i++)
    {
         il-=q[i];    
         r+=p[i]*il;
    }
    cout<<r<<"\n";
    return 0;
}
