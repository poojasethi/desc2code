#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char s[2005];
int bg[2005],ed[2005];
int main()
{
    scanf("%s",s);
    int i,j,k;
    int len=strlen(s);
    for(i=0;i<len;i++)
    {
        for(j=i,k=i+1;j>=0&&k<len&&s[j]==s[k];j--,k++)bg[j]++,ed[k]++;
        for(j=k=i;j>=0&&k<len&&s[j]==s[k];j--,k++)bg[j]++,ed[k]++;
    }
    __int64 ans=0;
    for(i=0;i<len;i++)
    {
        for(j=i+1;j<len;j++)ans+=(__int64)ed[i]*bg[j];
    }
    printf("%I64d\n",ans);
    return 0;
}
