#include <iostream>
#include <cstring>
#include <cstdio>

using namespace std;

typedef long long LL;

LL A[2006], B[2006];
char s[2006];

int main()
{
	scanf("%s", s);
	for(int i=0; s[i]; i++)
	{
		for(int x=i, y=i; x>=0 && s[y] && s[x]==s[y]; x--, y++) A[y]++, B[x]++;
		for(int x=i, y=i+1; x>=0 && s[y] && s[x]==s[y]; x--, y++) A[y]++, B[x]++;
	}
	for(int i=1; s[i]; i++) A[i]+=A[i-1];
	LL ans=0;
	for(int i=0; s[i+1]; i++) ans+=A[i]*B[i+1];
	printf("%I64d\n", ans);
	return 0;
}
