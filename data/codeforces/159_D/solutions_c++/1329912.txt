#include <iostream>
#include <algorithm>

#define MAX 2013
using namespace std;

string S;
int N,A[MAX],B[MAX];
unsigned long long res;

int main()
{
	int h,e,i;
	cin >> S;
	N=S.size();
	for(i=0;i<N;i++)
	{
		e=h=i;
		while(e<N && h>-1 && S[e] == S[h])
		{
			B[e+1]++;
			A[h+1]++;
			h--;
			e++;
		}
		h=i;
		e=i+1;
		while(e<N && h>-1 && S[e] == S[h])
		{
			B[e+1]++;
			A[h+1]++;
			h--;
			e++;
		}
	}
	for(i=1;i<=N;i++)
	{
		A[i]+=A[i-1];
		B[i]+=B[i-1];
	}
	for(i=1;i<=N;i++) res+=(B[i]-B[i-1])*(A[N]-A[i]);
	cout << res << endl;
	return 0;
}
