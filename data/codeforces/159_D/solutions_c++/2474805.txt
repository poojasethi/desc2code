#include<stdio.h>
#include<string>
#include <iostream>

using namespace std;

int n,f[2500],g[2500],i,j;
long long ans;
string str;

void count(int x,int y)
{
    for(;0<=x&&y<str.size();x--,y++)
        if(str[x]==str[y])
        {
            ++g[x];
            ++f[y];
        }
        else break;
}

main()
{
    cin>>str;
    for(i=0;i<str.size();i++) count(i,i);
    for(i=1;i<str.size();i++) count(i-1,i);
    for(i=0;i<str.size();i++) for(j=i+1;j<str.size();j++) ans+=f[i]*g[j];
    printf("%I64d\n",ans);
}
