#include<stdio.h>
#include<string.h>
char s[2003];
#define lld __int64
lld l[2003], r[2003];
int main()
{
	int i, j, k;
	int len = strlen(gets(s));
	for(i = 0; i < len; i++)
	{
		for(j = k = i; j >= 0 && k < len && s[j] == s[k]; j--, k++)
			l[j]++, r[k]++; 
		for(j = i, k = i+1; j >= 0 && k < len && s[j] == s[k]; j--, k++)
			l[j]++, r[k]++;
	}
	lld ans = 0;
	for(i = 1; i < len; i++)
		r[i] += r[i-1];
	for(i = 1; i < len; i++)
		ans += r[i-1] * l[i];
	printf("%I64d\n", ans);
	return 0;
}
