#include<iostream>

using namespace std;

string s;
int n;
long long ini[2001],fin[2001];

void findpal(int i,int j=-1)
{
	if(j==-1)
		j=i;
	while(i>=0 and j<n and s[i]==s[j])
	{
		ini[i]++;
		fin[j]++;
		i--,j++;
	}
}

void imparmode(int i)
{
	int j=i;
}

void countpalis()
{
	for(int c=0;c<n;c++)
	{
		if(c!=n-1)
			findpal(c,c+1);
		findpal(c);
	}
}

int main()
{
	cin>>s;
	n=s.size();
	for(int i=0;i<n;i++)
		ini[i]=fin[i]=0;
	countpalis();
	long long ans=0;
	for(int i=0;i<n-1;i++)
		for(int j=i+1;j<n;j++)
			ans+=fin[i]*ini[j];
	cout<<ans<<"\n";
}
