#include <stdio.h>
#include <string.h>
typedef long long ll;
char s[2000+5];
int p[2000+5],q[2000+5];
int main()
{
    ll ans=0;
    scanf("%s",s+1);
    int len = strlen(s+1);
    int a,b;
    for(int i=1;i<=len;i++)
    {
        for(a=i,b=i+1;a>=1&&b<=len&&s[a]==s[b];a--,b++) p[b]++,q[a]++;
        for(a=b=i;a>=1&&b<=len&&s[a]==s[b];a--,b++) p[b]++,q[a]++;
    }
    for(int i=len-1;i>=1;i--) q[i]+=q[i+1];
    for(int i=1;i<=len-1;i++) ans+=(ll)p[i]*q[i+1];
    printf("%I64d\n",ans);
    return 0;
}
