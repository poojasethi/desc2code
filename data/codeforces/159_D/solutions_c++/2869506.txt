#include <cstdio>
#include <iostream>
using namespace std;

string s;
long long ans,cnt[2020];
int i,j,l,r,n;

int main()
{
	cin >> s;
	n = s.length();
	for (i=0;i<n;i++)
	{
		for (l=i,r=i;0<=l&&r<n&&s[l]==s[r];l--,r++)
			cnt[r]++;
		for (l=i-1,r=i;0<=l&&r<n&&s[l]==s[r];l--,r++)
			cnt[r]++;
	}
	for (i = 1; i < n; i++)
		cnt[i] += cnt[i-1];
	for (i=0;i<n;i++)
	{
		for (l=i,r=i;0<=l&&r<n&&s[l]==s[r];l--,r++)
			ans += l?cnt[l-1]:0;
		for (l=i-1,r=i;0<=l&&r<n&&s[l]==s[r];l--,r++)
			ans += l?cnt[l-1]:0;
	}
	cout << ans;
}
