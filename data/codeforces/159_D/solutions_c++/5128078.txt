#include <iostream>
using namespace std;

int pref[2010];

int main() {
	string s;
	cin >> s;
	int n = s.size();
	s = "~" + s + "*";
	long long ans = 0;
	for (int i = 1; i <= n; i++) {
		{
			int l = i, r = i;
			while (s[l] == s[r]) {
				ans += pref[l];
				pref[r + 1]++;
				l--;
				r++;
			}
		}
		{
			int l = i, r = i + 1;
			while (s[l] == s[r]) {
				ans += pref[l];
				pref[r + 1]++;
				l--;
				r++;
			}
		}
		pref[i + 1] += pref[i];
	}
	cout << ans << endl;
}
