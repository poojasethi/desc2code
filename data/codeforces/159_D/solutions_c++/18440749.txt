#include<bits/stdc++.h>
using namespace std;
#define ll long long 
ll from[2000007],to[2000007];
int main(){
 char s[2000007];
 scanf("%s",s);
 int n=strlen(s);
 bool ok;
 for(int i=0;i<n;i++)
  {
  for(int j=i;j<n;j++)
    {
    for(int k=i;k<=j;k++)
      {
      ok=true;
      if(s[k]!=s[i+j-k]){
       ok=false;
       break;
      }
      }
if(ok){
  from[i]++;
  to[j]++;
  }
 }
}
ll ans=0;
 for(int i=0;i<n;i++)
  for(int j=i+1;j<n;j++)
    ans+= to[i]*from[j];
 cout<<ans<<endl;
 return 0;
}
