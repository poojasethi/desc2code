#include <iostream>
#include <string>

using namespace std;

string S;
int isPalin[2010][2010];
long long result,cnt;

int main()
{
	cin >> S;
	for (int i=0;i<S.size();i++)
		for (int j=i;j>=0;j--)
			isPalin[j][i]=(i==j)||(i-j==1&&S[i]==S[j])||(i-j>=2&&S[i]==S[j]&&isPalin[j+1][i-1]);
	for (int i=0;i<S.size()-1;i++)
	{
		for (int j=0;j<=i;j++)
			cnt+=isPalin[j][i];
		for (int j=i+1;j<S.size();j++)
			result+=isPalin[i+1][j]*cnt;
	}
	cout << result << endl;
	return 0;
}
