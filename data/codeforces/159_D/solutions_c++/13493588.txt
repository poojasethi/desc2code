#include<bits/stdc++.h>

using namespace std;

#define N 2003

int M[N],T[N],n,i,j,k;

long long t;

bool H[N][N];

string s;

int main(){

	cin >> s;

	n = s.size();

	for(i=0;i<n;i++){

		for(j=k=i; j >= 0 && k < n && s[j] == s[k];j--,k++)
			H[j][k] = 1;

	    for(j=i,k=i+1; j >= 0 && k < n && s[j] == s[k];j--,k++)
			H[j][k] = 1;
	}

	for(i=0;i<n;i++)
		for(j=0;j<=i;j++)
		    M[i] += H[j][i];

    for(i=n-1;i>=0;i--)
		for(T[i] = T[i+1],j=i;j<n;j++)
			T[i] += H[i][j];    

	for(i=0;i<n;i++)
		t += M[i] * T[i+1];
		
	cout << t;
	return 0;
}
