# include <stdio.h>
# include <string.h>

# define MAXN 2009

typedef long long lld;

int n;
char s[MAXN];
lld dp[MAXN], end[MAXN], ans;

int main(){
	scanf("%s",s);
	
	n = strlen(s);
	
	for(int i=0; i<n; i++){
		
		int x = i, y = i;
		
		while(s[x] == s[y]  &&  x > -1  &&  y < n){
			end[y++]++;
			dp[x--]++;
		}
		
		x = i-1, y = i;
		
		while(s[x] == s[y]  &&  x > -1  &&  y < n){
			end[y++]++;
			dp[x--]++;
		}
	}
	
	for(int i=n-1; i>=0; i--)
		dp[i] += dp[i+1];
	
	for(int i=0; i<n; i++)
		ans += dp[i+1] * end[i];
	
	printf("%I64d",ans);
}
