#include <cstdio>
#include <cstring>
#define FOR(x,y,z) for(int x=(y);x<=(z);++x)
int n;
long long cntl[2005],cntr[2005],ans;
char s[2005];
int main(){
	scanf("%s",s);
	int sl=strlen(s);
	FOR(i,0,sl-1) cntl[i]=cntr[i]=0;
	FOR(i,0,sl-1){
		int l=i,r=i;
		while(l>=0&&r<=sl-1&&s[l]==s[r]){
			++cntl[l],++cntr[r];
			--l,++r;
		}
	}
	FOR(i,0,sl-2){
		int l=i,r=i+1;
		while(l>=0&&r<=sl-1&&s[l]==s[r]){
			++cntl[l],++cntr[r];
			--l,++r;
		}
	}
	ans=0;
	FOR(i,0,sl-2) FOR(j,i+1,sl-1) ans+=(cntr[i]*cntl[j]);
	printf("%I64d\n",ans);
	return 0;
}
