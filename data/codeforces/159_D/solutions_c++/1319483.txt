#include <iostream>
#include <string>
using namespace std;

typedef long long LL;
string s;
int n;
LL dp[2005][2005];
LL ans = 0;

int main() {
	cin >> s;
	n = s.size();
	for (int i = 0; i < 2*n-1; ++i) {
		for (int j=i%2; (i-j)/2>=0 && s[(i-j)/2]==s[(i+j)/2]; j+=2) {
			dp[(i-j)/2][(i+j)/2] = 1;
		}
	}
	for (int i = 0; i < n; ++i) {
		for (int j = i+1; j < n; ++j) {
			dp[i][j] += dp[i][j-1];
		}
	}
	for (int i = 0; i < n; ++i) {
		for (int j = i+1; j < n; ++j) {
			ans += dp[i][j-1] * dp[j][n-1];
		}
	}
	cout << ans << endl;
}
