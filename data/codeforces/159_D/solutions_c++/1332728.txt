#include <cstdio>
#include <cstring>

char s[2005];
int dp[2005][2005];

int main(){
	scanf("%s",s);
	int len=strlen(s);
	for(int i=0;i<2*len-1;++i)
		for(int j=i%2;(i-j)/2>=0&&s[(i-j)/2]==s[(i+j)/2];j+=2)
			dp[(i-j)/2][(i+j)/2]=1;
	for(int i=0;i<len;++i)
		for(int j=i+1;j<len;++j)
			dp[i][j]+=dp[i][j-1];
	long long ans=0;
	for(int i=0;i<len;++i)
		for(int j=i+1;j<len;++j){
			long long cnt;
			cnt=dp[i][j-1]*dp[j][len-1];
			ans+=cnt;
		}
	printf("%I64d\n",ans);
}
