#include<cstdio>
#include<cstring>
using namespace std;
int L;
long long F[2010],G[2222],ans;
char s[2222];
int main()
{
    int i,j,k,l;
    scanf("%s",s+1);
    L = strlen(s+1);
    s[0] = '^'; s[L+1] = '$';
    for (k=1;k<=L;k++)
    {
        for (l=1; s[k-l+1]==s[k+l-1]; l++) F[k-l+1]++,G[k+l-1]++;
        for (l=1; s[k-l+1]==s[k+l]; l++) F[k-l+1]++,G[k+l]++;
    }
    ans = 0;
    for (i=1;i<L;i++)
        for (j=i+1;j<=L;j++)
            ans += G[i]*F[j];
    printf("%I64d\n",ans);
    return 0;
}
