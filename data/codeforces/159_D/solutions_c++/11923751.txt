#include<iostream>
#include<algorithm>
#include<map>
#include<iomanip>
using namespace std;

long long int z=0,td[1000006],p[1000006];
string s;

bool pld(int l,int r)
{
for(int i=l;i<=(r+l)/2;i++)
if(s[i]!=s[r+l-i])
return 0;
return 1;
}

int main()
{
ios_base::sync_with_stdio(0);cin.tie();cout.tie();
cin >> s;
for(int i=0;i<s.size();i++)
{
for(int j=0;j<=i;j++)
if(pld(j,i))
{
//cout<<i<<" "<<j<<endl;
p[i]++;
z+=p[j-1];
}
p[i]+=p[i-1];
}
cout<<z;
return 0;
}
