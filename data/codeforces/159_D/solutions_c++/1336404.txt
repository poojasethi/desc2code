#include<cstdio>
#include<cstring>
using namespace std;
int b1[2000],b2[2000];
char i[2001];
int main()
{
	int a,s,n;
	long long sm;
	scanf("%s",i);
	n=strlen(i);
	for(a=0;a<=2*(n-1);a++)
	{
		for(s=a/2;s>=0 && a-s<n;s--)
		{
			if( i[s]!=i[a-s] ) break;
			b1[s]++;
			b2[a-s]++;
		}
	}
	sm=0;
	for(a=1;a<n;a++)
	{
		b2[a]+=b2[a-1];
		sm+=(long long)b1[a]*b2[a-1];
	}
	printf("%I64d\n",sm);
//for(;;);
	return 0;
}
