#include<bits/stdc++.h>
using namespace std;
const int N=2006;
char s[N];int p[N],q[N],i,a,b,l;
long long ans;
int main()
{

    scanf("%s",s+1);
    l=strlen(s+1);
    for(i=1;i<=l;i++)
    {
        for(a=i,b=i+1;a&&b<=l&&s[a]==s[b];a--,b++)++p[a],++q[b];
        for(a=b=i;a&&b<=l&&s[a]==s[b];a--,b++)++p[a],++q[b];
    }
    for(i=l;i>=1;i--)p[i]+=p[i+1];
    for(i=1;i<l;i++)
       ans+=(long long )q[i]*p[i+1];
    printf("%lld",ans); 
	return 0;
}
