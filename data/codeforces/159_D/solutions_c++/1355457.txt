#include<iostream>
using namespace std;
main(){
    string a;
    int i,j,l,x,y;
    long long ans=0,front[2001]={},back[2001]={};
    cin>>a;
	l=a.size();
    for(i=0;i<l;i++){
		for(x=i,y=i;x>=0&&y<l&&a[x]==a[y];x--,y++)front[x]++,back[y]++;
		for(x=i,y=i+1;x>=0&&y<l&&a[x]==a[y];x--,y++)front[x]++,back[y]++;
	}
	for(i=l-2;i>=0;i--)front[i]+=front[i+1];
	for(i=0;i<l;i++)ans+=back[i]*front[i+1];
	cout<<ans<<endl;
}
