#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
#define mxn 2010
#define ll long long

ll A[mxn], B[mxn];
char s[mxn];

int main()
{
	scanf("%s", s);
	for(int i=0; s[i]; i++)
	{
		for(int x=i, y=i; x>=0 && s[y] && s[x]==s[y]; x--, y++) A[y]++, B[x]++;
		for(int x=i, y=i+1; x>=0 && s[y] && s[x]==s[y]; x--, y++) A[y]++, B[x]++;
	}
	for(int i=1; s[i]; i++) A[i]+=A[i-1];
	ll ans=0;
	for(int i=0; s[i+1]; i++) ans+=A[i]*B[i+1];
	printf("%I64d\n", ans);
	return 0;
}
