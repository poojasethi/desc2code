#include<stdio.h>
#include<string.h>

int main()
{
	char s[2001];
	gets(s);
	int i,j,a,b,left[2001]={0},right[2001]={0},k=strlen(s);
	for(i=0;i<k;i++)
	{
		for(a=i,b=i;a>=0 && b<k && s[a] == s[b];a--,b++)
		{
			left[a]++;
			right[b]++;
		}
		for(a=i,b=i+1;a>=0&&b<k&&s[a]==s[b];a--,b++)
		{
			left[a]++;
			right[b]++;
		}
	}
	for(i=k-1;i>=1;i--)
		left[i-1]+=left[i];
	__int64 res=0;
	for(i=0;i<k-1;i++)
	{
		res+=(right[i]*left[i+1]);
	}
	printf("%I64d\n",res);
	return 0;
}
