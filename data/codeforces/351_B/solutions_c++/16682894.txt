#include <iostream>
using namespace std;

int f(int k)
{
  if(k==0) return 0;
  if(k==1) return 1;
  return 4+f(k-2);
}

int main()
{
  int n, k=0;
  cin>>n;
  int t[n];
  for(int i=0; i<n; i++) cin>>t[i];
  for(int i=0; i<n; i++) for(int j=i+1; j<n; j++) if(t[j]<t[i]) k++;
  cout<<f(k);
  return 0;
}
