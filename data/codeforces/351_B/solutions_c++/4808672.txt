#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main ( )
{
	ll ans = 0;
	int n;
	cin >> n;
	vector<int> v ( n );
	for ( int i = 0; i < n; ++i ) {
		cin >> v[i];
		for ( int j = 0; j < i; ++j )
			if ( v[j] > v[i] )
				ans++;
	}

	cout << (ans%2) + 4LL*(ans/2) << endl;
	return 0;
}
