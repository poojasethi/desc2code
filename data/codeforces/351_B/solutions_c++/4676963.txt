#include<iostream>
#include<cstdio>

using namespace std;
#define FOR( i, a , n) for( int (i)=(a);(i)<(int)n;++(i))
int a[3030],n,cnt;
int main()
{
  scanf("%d",&n);
  FOR(i,0,n) scanf("%d",a+i);
  FOR(i,0,n) FOR(j,i+1,n) if(a[j]<a[i]) cnt++;
  printf("%d.000000\n",cnt*2-(cnt%2));
    
  return 0;
}
