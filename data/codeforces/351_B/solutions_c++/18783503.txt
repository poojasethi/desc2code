#include <cstdio>
int i,j,n,a[3010],f[9000000],m;
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%d\n",&n);
	for (i=1;i<=n;++i) scanf("%d",&a[i]);
	for (i=1;i<=n;++i)
		for (j=i+1;j<=n;++j)
		if (a[i]>a[j]) ++m;
	f[0]=0;f[1]=1;
	for (i=2;i<=m;++i) f[i]=f[i-2]+4;
	printf("%.6f\n",(double)f[m]);
}
