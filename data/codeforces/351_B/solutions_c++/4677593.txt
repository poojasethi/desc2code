#include<cstdio>
#include<cstdlib>
int n,a[3001],x;
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
        for(int j=0;j<i;j++)
            if(a[j] > a[i])
                x++; 
    }
    if(x%2) printf("%d\n",2*x-1);
    else printf("%d\n",2*x);
    return 0;
}
