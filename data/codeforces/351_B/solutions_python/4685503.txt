N=input()
a,ans,s,i=map(int,raw_input().split()),0,[0]*(N+1),N
def add(x):
    while x<=N:
        s[x]+=1
        x+=x&-x
def sum(x):
    k=0
    while x:
        k+=s[x]
        x-=x&-x
    return k
while i:
    i=i-1
    ans=ans+sum(a[i])
    add(a[i])
if ans&1:
    print 2*ans-1
else:print 2*ans
