#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
long long f[2001][2001];
long long g[2001][2001];
int n, m;
long long mod = (int) 1e9+7;
void add(long long& x, long long y)
{
	x += y;
	if(x > mod)
		x -= mod;
	return ;
}
int main()
{
	scanf("%d%d", &n, &m);
	for(int i = 1; i <= n; i++)
	{
		long long x = 0;
		for(int j = 2; j <= m; j++)
		{
			add(f[i][j], g[i-1][j] + 1);
			add(x, f[i][j]);
			add(g[i][j], g[i][j-1] + x);
		}
	}
	long long ans = 0;
	for(int i = 1; i <= n; i++)
	{
		for(int j = 2; j <= m; j++)
			ans += ((f[i][j] * (g[n-i][j] - f[n-i][j] + 1))%mod)*(m-j+1)%mod;
		ans %= mod;
	}
	int ia;
	ia = ans % mod;
	printf("%d\n", ia);
	return 0;
}
