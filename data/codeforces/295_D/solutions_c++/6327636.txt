#include<iostream>
#include<algorithm>
#include<vector>
#include<cstring>

using namespace std;

typedef long long ll;

const ll maxn=2010,mod=1000*1000*1000+7;

ll ans,dp[maxn][maxn],sum[maxn],n,m,t[maxn];

int main(){
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		dp[2][i]=i;
		sum[i+1]+=i;
		t[i+1]+=i;
	}
	for(int i=3;i<=m;i++){
		for(int j=1;j<=n;j++){
			sum[j]+=dp[i][j-1];
			sum[j]%=mod;
			t[j]+=sum[j];
			t[j]%=mod;
			dp[i][j]=(t[j]+1)%mod;
		}
	}
	for(int i=1;i<=n;i++){
		for(int j=2;j<=m;j++){
			ll x=(dp[j][i]*(dp[j][n-i+1]-dp[j][n-i]+mod))%mod;
			x=(x*(m+1-j)) % mod;
			ans+=x;
			ans%=mod;
		}
	}
	cout<<ans<<endl;
	return 0;
}
