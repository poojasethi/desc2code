#include <iostream>
#define L first
#define R second
#define MP make_pair
#define LLI long long int
using namespace std;

const LLI MX = 2100 , MOD = 10000*100000 + 7;
long long ans , dp[MX][MX], sum[MX][MX];

LLI M(LLI x){
	return ((x % MOD) + MOD) % MOD;
}

int main(){
	LLI n, m;
	cin >> n >> m;
	for(LLI i=2;i<=m;i++)
		dp[1][i] = i * (i - 1) / 2, sum[1][i] = dp[1][i];
	for(LLI i=2;i<=n;i++)
		for(LLI j=2;j<=m;j++){
			dp[i][j] = dp[i-1][j] + dp[i][j-1] + dp[i][j-1] - dp[i][j-2];
			dp[i][j] = M(dp[i][j]);
			sum[i][j] = sum[i-1][j] + dp[i][j];
			sum[i][j] = M(sum[i][j]);
		}
	for(LLI t=1;t<=n;t++){
		for(LLI x=2;x<=m;x++){
			ans += (m-x+1) * M((1 + sum[t-1][x]) * M(1 + sum[n-t][x-1] + sum[n-t][x-1] - sum[n-t][x-2]));
			ans = M(ans);
		}
	}
	cout << ans << endl;
}
