#include <numeric>
using namespace std;
typedef long long LL;
const LL Mod = 1000000007;
LL Num[2005][2005],Sum[2005][2005],N,M;
int main(){
    scanf("%d%d",&N,&M);
    for(int j=2;j<=M;++j){
        Sum[1][j]=Num[1][j]=1;
        for(int i=2;i<=N;++i){
            Num[i][j]=(2*Num[i][j-1]+Num[i-1][j]-Num[i][j-2])%Mod;
            Sum[i][j]=(Sum[i-1][j]+Num[i][j])%Mod;
        }
    }
    LL Ans=0;
    for(int i=1;i<=N;++i)
        for(int j=2;j<=M;++j)
            Ans+=(M-j+1)*Sum[N-i+1][j]%Mod*Num[i][j]%Mod;
    cout<<(Ans%Mod+Mod)%Mod<<endl;
    return 0;
}
