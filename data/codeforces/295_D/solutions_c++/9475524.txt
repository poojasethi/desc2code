#include <cstdio>
const int N = 2009,MOD = 1000000007;
int n,m,ans,f[N][N];
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("295D.in","r",stdin);
	freopen("295D.out","w",stdout);
	#endif
	scanf("%d%d",&n,&m);
	if (m == 1) return puts("0"),0;
	for (int i = 1; i <= n; ++i) {
		int x = 1,y = 0;
		for (int j = 1; j < m; ++j) {
			y = (y + f[i - 1][j]) % MOD;
			x = (x + y) % MOD;
			f[i][j] = x;
		}
	}
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j < m; ++j)
			ans = (ans + 1ll * f[i][j] * (m - j) % MOD * (f[n - i + 1][j] - f[n - i][j])) % MOD;
	if (ans < 0) ans += MOD;
	printf("%d\n",ans);
}
