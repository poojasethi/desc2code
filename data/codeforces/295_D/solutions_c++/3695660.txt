#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define int64 long long
#define modnum 1000000007

int64 f[2100][2100],s[2100][2100],sum,ans;
int n,m;
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i)
	{
		sum=0;
		for (int j=2;j<=m;++j) f[i][j]=(s[i-1][j]+1)%modnum,sum=(sum+f[i][j])%modnum,s[i][j]=(s[i][j-1]+sum)%modnum;
	}
	for (int i=1;i<=n;++i)
	  for (int j=2;j<=m;++j) ans+=(f[i][j]*(s[n-i][j]-f[n-i][j]+1)%modnum)*(m-j+1)%modnum,ans%=modnum;
	printf("%I64d\n",ans);    
	return 0;
}
