#include <iostream>
#include <stdio.h>
using namespace std;
const int N=2002;
const int MD=1000000007;
typedef long long LL;
LL n, m, ans, f[N][N], g[N][N]; 
int main()
{

    cin >> n >> m; 
    for (LL i=2; i<=m; ++i)
        g[1][i]=f[1][i]=1;
    LL s1, s2; 
    for (LL i=2; i<=n; ++i)
    {
        s1=s2=0;
        for (int j=2; j<=m; ++j)
        {
            f[i][j]=( (s1*(j+1) % MD-s2) % MD+MD ) % MD;
            f[i][j]=(f[i][j]+1) % MD;
            g[i][j]=(g[i-1][j]+f[i][j]) % MD;
            s1=(s1+g[i-1][j]) % MD;
            s2=(s2+g[i-1][j]*j) % MD;
        }
    }
    for (LL i=1; i<=n; ++i)
        for (LL j=2; j<=m; ++j)
        ans=(ans+(m-j+1)*f[i][j] % MD * g[n-i+1][j] % MD) % MD;
    cout << ans << endl;
}
