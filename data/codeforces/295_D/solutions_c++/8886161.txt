#include <cstdio>
#define M 1000000007
using namespace std;

int n,m,ans,dp[2010][2010];
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		int s=0;
		dp[i][1]=1;
		for(int j=2;j<=m;j++){
			s+=dp[i-1][j];s%=M;
			dp[i][j]=(dp[i][j-1]+s)%M;
		}
	}
	for(int i=1;i<=n;i++)
		for(int j=2;j<=m;j++)
			ans=(ans+(m-j+1LL)*(dp[i][j]-dp[i-1][j]+M)%M*dp[n-i+1][j])%M;
	printf("%d\n",ans);
	return 0;
}
