#include<cstdio>
#include<cstring>

#define ran 2222
#define mo 1000000007

int n,m;
int f[ran][ran],s[ran][ran];

int add(int a,int b){ return a+b>=mo?a+b-mo:a+b; }
int mul(int a,int b){ return 1LL*a*b%mo; }

int main(){
	scanf("%d%d",&n,&m);
	memset(f,0,sizeof f);
	memset(s,0,sizeof s);
	for(int i=1;i<=n;i++)
		for(int j=2;j<=m;j++){
			f[i][j]=(j==2)?1:0;
			f[i][j]=add(f[i][j],f[i][j-1]);
			f[i][j]=add(f[i][j],s[i-1][j]);
			s[i][j]=add(s[i][j-1],f[i][j]);
		}
	int res=0;
	for(int i=1;i<=n;i++)
		for(int j=2;j<=m;j++){
			int t=m-j+1;
			t=mul(t,f[i][j]);
			t=mul(t,add(f[n-i+1][j], mo-f[n-i][j]));
			res=add(res,t);
		}
	printf("%d\n",res);

	return 0;
}
