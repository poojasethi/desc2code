#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <cstring>
#include <string>
#include <cctype>
#include <algorithm>
#define mode 1000000007
#define maxn 2005

using namespace std;
typedef long long LL;
int n,m;LL f[maxn][maxn],g[maxn][maxn],ans;

void init()
 {
  scanf("%d%d",&n,&m);
 }
 
void work()
 {
  int i,j,k;LL sum;
  for (i=1;i<=n;i++)
   {
    sum=0;
    for (j=2;j<=m;j++)
     {
      f[i][j]=(g[i-1][j]+1)%mode;
      sum=(sum+f[i][j])%mode;
      g[i][j]=(g[i][j-1]+sum)%mode;
     }
   }
  for (i=1;i<=n;i++)
   for (j=2;j<=m;j++)ans=(ans+(f[i][j]*(g[n-i][j]-f[n-i][j]+mode+1)%mode*(m-j+1))%mode)%mode;
  cout<<ans<<endl;
 }
 
int main()
 {
  init();
  work();
  return 0;
 }
