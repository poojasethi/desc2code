#include <iostream>
#define MAXN 2005
#define MOD 1000000007

using namespace std;

int n, m;
long long res, c[MAXN][MAXN];

main()
{
  cin >> n >> m;
  
  long long p, q, s;
  for (int i = 1; i <= n; i++)
  {
    p = 1, q = 0;
    for (int j = 2; j <= m; j++)
    {
      q = (q + c[i - 1][j]) % MOD;
      p = (p + q) % MOD;
      c[i][j] = p;
    }
  }
    
  res = 0;
  for (int i = 1; i < n; i++)
  {
    p = 1; q = 0;
    for (int j = 2; j <= m; j++)
    {
      s = m - j + 1;
      p = (p + q) % MOD;
      q = (q + c[n - i][j]) % MOD;
      res = (res + s * (c[i][j] * p % MOD)) % MOD;
      p = (p + c[n - i][j]) % MOD;
    }
  }
  for (int j = 2; j <= m; j++) 
  {
    s = m - j + 1;
    res = (res + s * c[n][j]) % MOD;
  }
  
  cout << res;
  
//  system("pause");
}
