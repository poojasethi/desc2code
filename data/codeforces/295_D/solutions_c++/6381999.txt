#include <cstdio>
#include <cstring>

using namespace std;

#define ll long long
#define MOD 1000000007

int n,m;
int dp[2005][2005];
int C[2005][2005];
int sum[2005];

int main() {
	for(int i=0;i<=2000;++i) C[i][0] = C[i][i] = 1;
	for(int i=1;i<=2000;++i) for(int j=1;j<=2000;++j)
		C[i][j] = (C[i-1][j] + C[i-1][j-1]) % MOD;

	scanf("%d%d",&n,&m);

	for(int i=1;i<=n;++i) {
		dp[i][0] = 1;
		dp[i][1] = 1;
		for(int j=2;j<=m;++j) {
			dp[i][j] = ((ll)dp[i][j-1] + sum[j]) % MOD;
		}
		for(int j=2;j<=m;++j) sum[j] = (sum[j-1] + dp[i][j]) % MOD;
	}

	int ans = 0;
	for(int i=1;i<=n;++i) for(int j=2;j<=m;++j) {
		ll add = m-j+1;
		ll up = (dp[i][j] - dp[i-1][j] + MOD) % MOD;
		ll dw = dp[n-i+1][j];
		ans = (ans + add * up % MOD * dw) % MOD;
	}

	printf("%d",ans);
}
