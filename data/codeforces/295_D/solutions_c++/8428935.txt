#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <ctime>
#include <algorithm>

using namespace std;

const int oo=1000000007;

int N,M;
int F[2005][2005],G[2005][2005];

int main()
{
	//freopen("295D.in","r",stdin);
	//freopen("295D.out","w",stdout);
	scanf("%d%d",&N,&M);
	memset(F,0,sizeof(F));
	memset(G,0,sizeof(G));
	for (int i=1; i<=N; i++)
	{
		int sum=0;
		for (int j=2; j<=M; j++)
		{
			F[i][j]=(F[i][j]+G[i-1][j]+1) % oo;
			sum=(sum+F[i][j]) % oo;
			G[i][j]=(G[i][j]+G[i][j-1]+sum) % oo;
		}
	}
	int Ans=0;
	for (int i=1; i<=N; i++)
		for (int j=2; j<=M; j++)
		{
			Ans+=(long long)F[i][j]*(G[N-i][j]-F[N-i][j]+1) % oo * (M-j+1) % oo; 
			Ans%=oo;
		}
	cout << Ans << endl;
	return 0;
}
