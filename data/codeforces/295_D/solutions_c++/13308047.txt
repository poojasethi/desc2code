#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
#define N 2050
#define P 1000000007
#define ll long long
ll ans,A[N][N],s[N][N],cnt[N][N],B[N][N];int n,m;
int main()
 {
 	cin >>n>>m;
 	for (int i=1;i<=n;i++)
 	 for (int j=2;j<=m;j++)
 	   B[i][j]=(1+s[i-1][j-1]+j*cnt[i-1][j-1])%P,
 	   A[i][j]=(B[i][j]+A[i-1][j])%P,
 	   s[i][j]=(s[i][j-1]+A[i][j]*(1-j)%P+P)%P,
 	   cnt[i][j]=(cnt[i][j-1]+A[i][j])%P;
 	for (int i=1;i<=n;i++)
 	 for (int j=2;j<=m;j++)
 	   ans=(ans+B[i][j]*A[n-i+1][j]%P*(m-j+1))%P;
 	cout <<ans<<endl;
 	return 0;
 }
