#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
typedef long long lld;
#define maxn 2010
#define mod 1000000007
lld dp1[maxn][maxn];
lld dp2[maxn][maxn];
lld sum[maxn];
int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	memset(dp1,0,sizeof(dp1));
	for(int i=1;i<=n;i++)
	{
		memset(sum,0,sizeof(sum));
		lld now=0;
		for(int j=2;j<=m;j++)
		{
			sum[j]=(sum[j-1]+dp1[i-1][j])%mod;
			now=(now+sum[j])%mod;
			dp1[i][j]=(now+1)%mod;
			dp2[i][j]=(dp1[i][j]-dp1[i-1][j]+mod)%mod;
		}
	}
	lld ans=0;
	for(int i=1;i<=n;i++)
		for(int j=2;j<=m;j++)
		{
			lld now=(m-j+1);
			now=(now*dp1[i][j])%mod;
			now=(now*dp2[n-i+1][j])%mod;
			ans=(ans+now)%mod;
		}
	cout << ans << endl;
	return 0;
}
