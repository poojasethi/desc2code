#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#define N 2105
#define mo 1000000007
#define Long long long
#define For(i,l,r) for(int i=l;i<=r;i++)
using namespace std;
int n,m,Ans;
int S[N][N],F[N][N],G[N][N];
int main(){
    scanf("%d%d",&n,&m);
    For(i,1,n) For(j,1,m){
        F[i][j]=(S[i-1][j]+F[i][j-1]+(j==2))%mo;
        S[i][j]=(S[i][j-1]+F[i][j])%mo;
    }
    For(i,1,n) For(j,1,m) G[i][j]=(G[i][j-1]+S[i][j])%mo;
    For(i,1,n) For(j,1,m) Ans=(Ans+(Long)F[i][j]*(G[n-i][j]-F[n-i][j]+1+mo)%mo*(m-j+1))%mo;
    printf("%d\n",Ans);
}
