#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 2010
#define mo 1000000007
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int64 f[N][N],h[N][N],C[N][N],an,a[N][N];
int main() {
	scanf("%d%d",&n,&m);
	For(i,2,m) f[1][i]=h[1][i]=1;
	For(i,2,n) {
		int64 s=0,sum=0;
		For(j,2,m) {
			h[i][j]=(sum+s)%mo;
			s=(s+f[i-1][j])%mo;
			sum=(sum+s)%mo;
			f[i][j]=sum;
		}
	}
	For(j,2,m)For(i,1,n) f[i][j]=(f[i][j]+f[i-1][j])%mo,h[i][j]=(h[i][j]+h[i-1][j])%mo;
	For(j,2,m) {
		int64 s=0;
		For(i,1,n) s=(s+h[i][j]*f[n-i+1][j])%mo;
		an=(an+s*(m-j+1))%mo;
	}
	printf("%I64d\n",an);
	return 0;
}
