#include<iostream>
using namespace std;

#define ll	long long
#define N	2010
#define mod	1000000007

ll dp[N][N];
ll sum[N][N];
void prep()
{
	for(int e=2;e<N;e++){
		dp[e][1] = 1;	
	}	
	for(int t=2;t<N;t++){
		dp[2][t]=1;
		ll add = 1;
		for(int e=3;e<N;e++){
			add = (dp[e][t-1]+add)%mod;
			dp[e][t]=(dp[e-1][t]+add)%mod;
		}	
	}
	
	for(int l=2;l<N;l++)sum[l][1] = dp[l][1];
	for(int r = 2;r<N;r++){
		for(int l=2;l<N;l++)
			sum[r][l] = (sum[r][l-1]+dp[r][l])%mod;		
	}
	
}
int n,m;
int main()
{
	prep();
	while(cin>>n>>m){
		ll ans = 0ll;
		for(int r = 1; r<=n; r++){
			for(int l = 2; l<=m; l++){
				//cout<<l<<' '<<r<<' '<<sum[l][r]<<' '<<sum[l][n+1-r]<<endl;
				ans = (ans + ( (sum[l][r]*(sum[l][n+1-r]-sum[l][n-r]))%mod ) * (m+1-l) )%mod;
			}	
		}
		cout<<ans<<endl;
	}
}
