#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <string>
using namespace std;
typedef long long LL;
const int Maxn=2005,mod=1000000007;
int n,m,f[Maxn][Maxn];
int ans=0;

inline int add(int a,int b) {
	return (a+=b)>=mod?a-mod:a;
}

int main() {
	cin>>n>>m;
	for (int i=1,t;i<=n;i++) {
		f[i][2]=add(t=f[i-1][2],1);
		for (int j=3;j<=m;j++)
			t=add(t,f[i-1][j]),
			f[i][j]=add(t,f[i][j-1]);
	}
	for (int i=1;i<=n;i++)
		for (int j=2;j<=m;j++)
			ans=(ans+f[i][j]*LL(f[n-i+1][j]-f[n-i][j])%mod*(m-j+1))%mod;
	cout<<add(ans,mod)<<endl;
	return 0;
}
