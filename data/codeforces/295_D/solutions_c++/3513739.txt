#include <stdio.h>
#include <stdlib.h>
#include <iostream>
#include <algorithm>
#define mo 1000000007

using namespace std;

long long n, m, i, j, now1, now2, now, Ans;
long long f[2005][2005], f2[2005][2005];

int main()
{
	scanf("%I64d %I64d", &n, &m);
	f[0][0] = 1;
	for (i = 1; i <= n; i++)
	{
		f[i][0] = 1;
		now1 = 0;
		now2 = 1;
		for (j = 2; j <= m; j++)
		{
			now1 += f[i - 1][j], now1 %= mo;
			now2 += now1, now2 %= mo;
			f[i][j] = now2;
			f2[i][j] = (f[i][j] - f[i - 1][j]) % mo;
		}
	}
	
	Ans = 0;
	for (i = 1; i <= n; i++)
		for (j = 2; j <= m; j++)
		{
			now = f2[i][j];
			now *= f[n - i + 1][j];
			now %= mo;
			now *= (m - j + 1);
			now %= mo;
			Ans += now;
			Ans %= mo;
		}
	
	while (Ans < 0)
		Ans += mo;
	printf("%I64d\n", Ans);
	return 0;
}
