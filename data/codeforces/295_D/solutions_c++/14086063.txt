#include<cstdio>
const int N=2005,p=1000000007;
#define rep(i,a,b) for (int i=a;i<=b;i++)
int f[N][N],n,m,s,t,ans;
int main()
{
	scanf("%d%d",&n,&m);
	rep(i,1,n){t=0,s=1; rep(j,2,m) f[i][j]=s=(s+(t=(t+f[i-1][j])%p))%p;}
	rep(i,1,n) rep(j,2,m) ans=(1LL*(m-j+1)*(f[i][j]-f[i-1][j])%p*f[n-i+1][j]+ans)%p;
	printf("%d\n",(ans+p)%p); return 0;
}
