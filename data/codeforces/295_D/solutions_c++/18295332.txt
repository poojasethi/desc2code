#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#define LL long long
using namespace std;
const int P=1000000007;
int n,m,dp[2003][2004];
int main(){
	scanf("%d%d",&n,&m);
	for(int j=2;j<=m;j++) dp[1][j]=1;
	for(int i=2;i<=n;i++){
		LL s1=0,s2=0;
		for(int j=2;j<=m;j++){
			s1+=dp[i-1][j];
			if(s1>=P) s1-=P;
			s2+=s1;
			if(s2>=P) s2-=P;
			dp[i][j]=s2;	
		}
	}
	
	for(int i=2;i<=n;i++){
		for(int j=2;j<=m;j++){
			dp[i][j]+=dp[i-1][j];
			if(dp[i][j]>=P) dp[i][j]-=P;
		}
	}
	
	
	LL ans=0;
	for(int i=1;i<=n;i++){
		for(int j=2;j<=m;j++){
			ans=(ans+(LL)(m-j+1)*(dp[i][j]-dp[i-1][j])%P*dp[n-i+1][j])%P;
//			if(j==2) cout<<i<<" "<<(m-j+1)*(dp[i][j]-dp[i-1][j])*dp[n-i+1][j]<<endl;
		}
		
	}
	cout<<(ans+P)%P<<endl;
	return 0;
}
