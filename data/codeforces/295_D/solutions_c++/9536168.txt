#include<cstdio>
#define LL long long
using namespace std;
const int mod=1000000007;
int n,m,ans,f[2005][2005];
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
		for(int j=f[i][0]=1,s=0;j<m;j++)
			s=(s+f[i-1][j])%mod,
			f[i][j]=(f[i][j-1]+s)%mod;
	for(int i=1;i<=n;i++)
		for(int j=1,t=1,s=0;j<m;j++)
			s=(s+f[n-i][j])%mod,
			t=(t+s)%mod,
			ans=(ans+(LL)(m-j)*f[i][j]%mod*(t-f[n-i][j]))%mod;
	printf("%d\n",ans);
}
