import sys

def num_type(s):
    lets = [c for c in s if c != '-']
    if all(c == lets[0] for c in lets):
        return "taxi"
    elif sorted(set(lets), reverse=True) == lets:
        return "pizza"
    else:
        return "girl"

n = int(sys.stdin.readline())
maxtaxi, maxpizza, maxgirl = 0, 0, 0

people = []

for i in xrange(n):
    p, name = sys.stdin.readline().rstrip('\n').split(' ')
    p = int(p)
    taxi, pizza, girl = 0, 0, 0
    for j in xrange(p):
        num = sys.stdin.readline().rstrip('\n')
        nt = num_type(num)
        if nt == 'taxi':
            taxi += 1
        elif nt == 'pizza':
            pizza += 1
        else:
            girl += 1
    people.append((name, taxi, pizza, girl))
    maxtaxi = max(maxtaxi, taxi)
    maxpizza = max(maxpizza, pizza)
    maxgirl = max(maxgirl, girl)

print "If you want to call a taxi, you should call: %s." % ', '.join(n for n, t, p, g in people if t == maxtaxi)
print "If you want to order a pizza, you should call: %s." % ', '.join(n for n, t, p, g in people if p == maxpizza)
print "If you want to go to a cafe with a wonderful girl, you should call: %s." % ', '.join(n for n, t, p, g in people if g == maxgirl)
