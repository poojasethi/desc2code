import string
import re

n = int(raw_input())

def readFriend(fs):
    name, n = readNameAndCount()
    phoneCounter = [0,0,0]
    for i in xrange(n):
        readPhone(phoneCounter)
    fs.append((name, phoneCounter))

def readNameAndCount():
    s = raw_input()
    i = s.index(' ')
    return s[i+1:], int(s[:i])

def isTaxi(number):
    for i in xrange(len(number)):
        if number[0] != number[i]:
            return False
    return True

def isPizza(number):
    for i in xrange(len(number) - 1):
        if int(number[i]) <= int(number[i + 1]):
            return False
    return True

def readPhone(phoneCounter):
    number = re.compile('-').sub('', raw_input())
    if isTaxi(number): phoneCounter[0] += 1
    elif isPizza(number): phoneCounter[1] += 1
    else: phoneCounter[2] += 1

def printResult(fs, message, i):
    m = 0
    for f in fs:
        if f[1][i] > m:
            m = f[1][i]
    out = []
    for f in fs:
        if f[1][i] == m:
            out.append(f[0])
    print 'If you want to ' + message + ', you should call: ' + string.join(out, ', ') + '.'

fs = []
for i in xrange(n):
    readFriend(fs)

#print fs

printResult(fs, 'call a taxi', 0)
printResult(fs, 'order a pizza', 1)
printResult(fs, 'go to a cafe with a wonderful girl', 2)
