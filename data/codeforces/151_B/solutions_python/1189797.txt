#!/usr/bin/env python
from sys import stdin as cin

def count_types(k):
	p, t = 0, 0
	for i in range(k):
		l = next(cin).strip().replace('-', '')
		if len(set(l)) == 1:
			t += 1
		elif l[0] > l[1] > l[2] > l[3] > l[4] > l[5]:
			p += 1
	return p, t, k - t - p

def main():
	P, T, G = 0, 0, 0
	Np, Nt, Ng = [], [], []
	n = int(next(cin))
	for i in range(n):
		k, name = next(cin).split()
		p, t, g = count_types(int(k))

		if p > P:
			P = p
			Np = [name]
		elif p == P:
			Np.append(name)

		if t > T:
			T = t
			Nt = [name]
		elif t == T:
			Nt.append(name)

		if g > G:
			G = g
			Ng = [name]
		elif g == G:
			Ng.append(name)

	print "If you want to call a taxi, you should call: " + ', '.join(Nt) + '.'
	print "If you want to order a pizza, you should call: " + ', '.join(Np) + '.'
	print "If you want to go to a cafe with a wonderful girl, you should call: " + ', '.join(Ng) + '.'


main()
