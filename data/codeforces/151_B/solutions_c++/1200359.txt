#include <cstdio>
#include <cstring>
const int N=110;
char s[N][100];
int t[N][3];
int n,ans[3]={};
int max(int a,int b){return a>b?a:b;}
void out(int a)
{
    bool f=false;
    for(int i=0;i<n;i++)
        if(ans[a]==t[i][a])
        {
            if(f)printf(", ");
            printf("%s",s[i]);
            f=true;
        }
    puts(".");
}
int main()
{
    
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        int a,x,y,z;
        scanf("%d%s",&a,s[i]);
        while(a--)
        {
            scanf("%d-%d-%d",&x,&y,&z);
            if(x==y&&y==z&&x%10==x/10)t[i][0]++;
            else if(x/10 > x%10 && x%10>y/10 && y/10>y%10 &&y%10>z/10&&z/10>z%10)
                t[i][1]++;
            else t[i][2]++;
        }
        for(int j=0;j<3;j++)
            ans[j]=max(ans[j],t[i][j]);
    }
    printf("If you want to call a taxi, you should call: ");
    out(0);
    printf("If you want to order a pizza, you should call: ");
    out(1);
    printf("If you want to go to a cafe with a wonderful girl, you should call: ");
    out(2);
    scanf(" ");
}
