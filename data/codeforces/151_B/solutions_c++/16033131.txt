#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long

using namespace std;

int n,a[105],b[105],c[105];
char s[105][29],t[10];
int ma,mb,mc;

vector<int> va,vb,vc;

int main(){
  scanf("%d",&n);
  FI(i,1,n){
    int x;
    scanf("%d %s",&x,s[i]);
    while(x--){
      scanf("\n%s",t);
      int len=strlen(t);
      if(t[0]==t[1] && t[1]==t[3] && t[3]==t[4] && t[4]==t[6] && t[6]==t[7]) a[i]++;
      else if(t[0]>t[1] && t[1]>t[3] && t[3]>t[4] && t[4]>t[6] && t[6]>t[7]) b[i]++;
      else c[i]++;
    }
    ma=max(ma,a[i]);
    mb=max(mb,b[i]);
    mc=max(mc,c[i]);
  }
  FI(i,1,n){
    if(a[i]==ma) va.push_back(i);
    if(b[i]==mb) vb.push_back(i);
    if(c[i]==mc) vc.push_back(i);
  }
  printf("If you want to call a taxi, you should call:");
  FI(i,0,(int)va.size()-1) printf(" %s%s",s[va[i]],i==(int)va.size()-1?".\n":",");
  printf("If you want to order a pizza, you should call:");
  FI(i,0,(int)vb.size()-1) printf(" %s%s",s[vb[i]],i==(int)vb.size()-1?".\n":",");
  printf("If you want to go to a cafe with a wonderful girl, you should call:");
  FI(i,0,(int)vc.size()-1) printf(" %s%s",s[vc[i]],i==(int)vc.size()-1?".\n":",");
  return 0;
}
