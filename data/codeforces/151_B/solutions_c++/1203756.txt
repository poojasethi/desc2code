#define L(s) (int)((s).size())
int k;
vector<string> best[3];
int vals[3];
int main() {
	cin >> k;
	while(k--) {
		int n; string name;
		cin >> n; cin >> name;
		int cnt[3]; memset(cnt, 0, sizeof(cnt));
		while(n--) {
			string s;
			cin >> s;
			if (s[0] == s[1] && s[1] == s[3] && s[3] == s[4] && s[4] == s[6] && s[6] == s[7]) ++cnt[0];
			else
				if (s[0] > s[1] && s[1] > s[3] && s[3] > s[4] && s[4] > s[6] && s[6] > s[7]) ++cnt[1];
				else
					++cnt[2];

			if (cnt[i] > vals[i]) {vals[i] = cnt[i]; best[i].clear(); }

	}
	cout << "If you want to call a taxi, you should call:";
	for(int i = 0; i < L(best[0]); ++i) {
		if (i) cout << ","; cout << " " <<  best[0][i];
	} cout << "." << endl;
