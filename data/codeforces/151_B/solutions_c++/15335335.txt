#include<bits/stdc++.h>
using namespace std;
char state[3][101]={"If you want to call a taxi, you should call: ","If you want to order a pizza, you should call: ",
"If you want to go to a cafe with a wonderful girl, you should call: "};
int main()
{
    int n,k,mx[3]={0};
    string name[101],s;
    int v[101][3]={0};
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>k>>name[i];
        for(int j=0;j<k;j++){
            cin>>s;
            if(s[0]==s[1]&&s[3]==s[1]&&s[3]==s[4]&&s[4]==s[6]&&s[6]==s[7])v[i][0]++;
            else if(s[0]>s[1]&&s[1]>s[3]&&s[3]>s[4]&&s[4]>s[6]&&s[6]>s[7])v[i][1]++;
            else v[i][2]++;
        }
        for(int j=0;j<3;j++){
            if(mx[j]<v[i][j]){
                mx[j] = v[i][j];
            }
        }
    }

    for(int i=0;i<3;i++){
//            cout<<mx[i]<<endl;
        cout<<state[i];
        bool comma = false;
        for(int j=0;j<n;j++){
            if(v[j][i]==mx[i]){
                if(comma==true){
                    cout<<", ";

                }
                cout<<name[j];
                comma=1;
            }
        }
        cout<<".\n";
    }
}
