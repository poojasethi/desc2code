#include <cstdio>

char par[][110]={"If you want to call a taxi, you should call: ","If you want to order a pizza, you should call: ",
    "If you want to go to a cafe with a wonderful girl, you should call: "};

int res[110][3];
int mx[3];

int main() {
    int n;
    scanf("%d",&n);
    char nm[110][24];
    for(int i=0;i<n;i++) {
        int k;
        scanf("%d%s",&k,nm[i]);
        for(int j=0;j<k;j++) {
            char a[10];
            scanf("%s",a);
            if (a[0]==a[1] && a[0]==a[3] && a[0]==a[4] && a[0]==a[6] && a[0]==a[7]) res[i][0]++;
            else if (a[0]>a[1] && a[1]>a[3] && a[3]>a[4] && a[4]>a[6] && a[6]>a[7]) res[i][1]++;
            else res[i][2]++;
        }
        for(int j=0;j<3;j++) if (res[i][j]>mx[j]) mx[j]=res[i][j];
    }
    for(int i=0;i<3;i++) {
        printf(par[i]);
        bool comma=false;
        for(int j=0;j<n;j++)
            if (res[j][i]==mx[i]) {
                if (comma) printf(", ");
                comma=true;
                printf(nm[j]);
            }
        printf(".\n");
    }
    return 0;
}
