#include <string.h>
#include <stdio.h>
#include <stdlib.h>

int n,a[101][101],k,t;
char s[100][101],s1[101];

int f(){
	int t1=1,t2=1;
	char c=s1[0];
	for(int i=1; i<strlen(s1); i++) {
	  if(s1[i]=='-') continue;
	  if(s1[i]!=c) t1=0;
	  if(s1[i]>=c) t2=0;
	  c=s1[i]; 
	}
	if(t1) return 0;
	if(t2) return 1;
	return 2;
}

int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  scanf("%i",&n);
  for(int i=0; i<101; i++) for(int j=0; j<3; j++) a[i][j]=0;
  for(int i=0; i<n; i++){
		scanf("%i%s",&k,&s[i][0]);	
		for(int j=0; j<k; j++) {
	  	scanf("%s",&s1);
	  	int f1=f();
	  	a[i][f1]++;
		}
	  for(int j=0; j<3; j++) if(a[100][j]<a[i][j]) a[100][j]=a[i][j];
  } 
	for(int i=0; i<3; i++){ 
  	t=0;
  	if(i==0) printf("If you want to call a taxi, you should call: ");
  	if(i==1) printf("If you want to order a pizza, you should call: ");
  	if(i==2) printf("If you want to go to a cafe with a wonderful girl, you should call: ");
  	for(int j=0; j<n; j++) if (a[j][i]==a[100][i]){
			if(t) printf(", ");
			printf("%s",&s[j][0]);
			t=1;
		}
  	printf(".");
  	if(i<2) printf("\n");
	}
}
