#include<cstdio>
#include<cstring>

int N;
int s[100];
char nm[100][30];
int ct[100][3];
char *tpg[3] = {"call a taxi", "order a pizza", "go to a cafe with a wonderful girl"};

int main(){
	scanf("%d", &N);
	for(int i = 0; i < N; i++){
		ct[i][0] = ct[i][1] = ct[i][2] = 0;
		
		scanf("%d %s", s + i, nm[i]);
		for(int j = 0; j < s[i]; j++){
			char numb[30];
			bool fg[10];
			for(int k = 0; k <= 9; k++)
				fg[k] = false;
			scanf("%s", numb);
			if		(	numb[0] == numb[1] &&
						numb[1] == numb[3] &&
						numb[3] == numb[4] &&
						numb[4] == numb[6] &&
						numb[6] == numb[7])		ct[i][0]++;
			else if	(	numb[0]  > numb[1] &&
						numb[1]  > numb[3] &&
						numb[3]  > numb[4] &&
						numb[4]  > numb[6] &&
						numb[6]  > numb[7])		ct[i][1]++;
			else								ct[i][2]++;
		}
	}
	for(int i = 0; i < 3; i++){
		int m = -1;
		for(int j = 0; j < N; j++)
			if(m < ct[j][i])
				m = ct[j][i];
		printf("If you want to %s, you should call:", tpg[i]);
		char tk[3] = {' ', 0, 0};
		for(int j = 0; j < N; j++)
			if(m == ct[j][i]){
				printf("%s%s", tk, nm[j]);
				tk[0] = ',';
				tk[1] = ' ';
			}
		printf(".\n");
	}
	
	return 0;
}
