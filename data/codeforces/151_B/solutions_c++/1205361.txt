#include <iostream>
using namespace std;
struct person {
	string name;
	int cnt[3];
	void add() {
		string t;
		for (int i = 0; i < 6; i++) {
			char c;
			while (true) {
				cin >> c;
				if (c >= '0' && c <= '9') {
					break;
				}
			}
			t += c;
		}
		if (t[0] == t[1] && t[1] == t[2] && t[2] == t[3] && t[3] == t[4] && t[4] == t[5]) {
			cnt[0]++;
		} else if (t[0] > t[1] && t[1] > t[2] && t[2] > t[3] && t[3] > t[4] && t[4] > t[5]) {
			cnt[1]++;
		} else {
			cnt[2]++;
		}
	}
	void init() {
		int tn;
		cin >> tn >> name;
		for (int i = 0; i < tn; i++) {
			add();
		}
	}
};

int n;
person arr[150];
void print(int type, string os) {
	cout << "If you want to " << os << ", you should call: ";
	int mx = 0;
	for (int i = 0; i < n; i++) {
		if (arr[i].cnt[type] > mx) {
			mx = arr[i].cnt[type];
		}
	}
	bool f = false;
	for (int i = 0; i < n; i++) {
		if (arr[i].cnt[type] == mx) {
			if (f) {
				cout << ", ";
			}
			f = true;
			cout << arr[i].name;
		}
	}
	cout << ".\n";
}
int main() {
	cin >> n;
	for (int i = 0; i < n; i++) {
		arr[i].init();
	}
	print(0, "call a taxi");
	print(1, "order a pizza");
	print(2, "go to a cafe with a wonderful girl");
}
