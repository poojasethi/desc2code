#include <stdio.h>
#define MAXN 105
#define MAXA 105

char srv[3][100]={"call a taxi","order a pizza","go to a cafe with a wonderful girl"};

long jud(char w[]){
	long i;
	w[2]=w[3];w[3]=w[4];
	w[4]=w[6];w[5]=w[7];
	for(i=1;i<6;i++){if(w[i]!=w[0])break;}
	if(i==6)return 0;
	for(i=1;i<6;i++){if(w[i-1]<=w[i])break;}
	if(i==6)return 1;
	return 2;
}

int main(void){
	long n,i,j,k,l; long c[MAXN]; long d[MAXN][3]; char a[MAXN][MAXA]; char w[20];
	long ltop; long iscomma;
	scanf("%ld",&n);
	for(k=0;k<n;k++){
		scanf("%ld%s",c+k,a[k]);
		d[k][0]=d[k][1]=d[k][2]=0;
		for(l=0;l<c[k];l++){
			scanf("%s",w);
			d[k][jud(w)]++;
		}
	}
	for(i=0;i<3;i++){
		printf("If you want to %s, you should call:",srv[i]);
		ltop=-1;
		for(k=0;k<n;k++){if(d[k][i]>ltop)ltop=d[k][i];}
		iscomma=0;
		for(k=0;k<n;k++){
			if(d[k][i]==ltop){
				printf("%s %s",iscomma?",":"",a[k]);
				iscomma=1;
			}
		}
		printf(".\n");
	}
	return 0;
}
