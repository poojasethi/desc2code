#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int num;
	int c[3] = {0};
	int n[3][100] = {{0}, {0}, {0}};
	char name[100][64];
	
	scanf("%d", &num);
	
	for(int i = 0; i < num; ++i) {
		
		int s = 0; 
		int t[3] = {0};
		
		scanf("%d%s", &s, name[i]);
		
		for(int j = 0; j < s; ++j) {
			
			char str[64];
			static int idx[6] = {0,1,3,4,6,7};
			bool b1 = true, b2 = true;
			
			scanf("%s", str);
			
			for(int k = 1; k < 6; ++k) {
				char ch1 = str[idx[k]], ch2 = str[idx[k - 1]];
				if(ch1 != ch2)
					b1 = false;
				if(ch1 >= ch2)
					b2 = false;
			}
			
			if(b1)
				++t[0];
			if(b2)
				++t[1];
			if(!(b1 || b2))
				++t[2];
		}
		
		for(int j = 0; j < 3; ++j) {
			n[j][i] = t[j];
			c[j] = max(c[j], t[j]);
		}
	}
	
	char pre[3][256] = {"If you want to call a taxi, you should call: ",
		"If you want to order a pizza, you should call: ",
		"If you want to go to a cafe with a wonderful girl, you should call: "};
	
	for(int i = 0; i < 3; ++i) {
		
		bool x = false;
		printf("%s", pre[i]);
		
		for(int j = 0; j < num; ++j) {
			if(n[i][j] == c[i]) {
				if(x)
					printf(", ");
				x = true;
				printf("%s", name[j]);
			}
		}
		
		printf(".\n");
	}
	
	return 0;
}
