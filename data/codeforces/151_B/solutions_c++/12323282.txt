//In the Name of God

#include<iostream>
#include<vector>

using namespace std;

int n,si,g,p,t,k,mg,mt,mp;
bool G,P,T;
string s,S;
vector<string> MG,MP,MT;

int main()
{
  cin>>n;
  for(int i=0;i<n;i++)
    {
      g=p=t=0;
      char c;
      cin>>si;
      cin>>S;
      for(int j=0;j<si;j++)
	{
	  cin>>s;
	  c=s[0];
	  T=G=P=true;
	  if(s[0]==s[1] && s[0]==s[3] && s[0]==s[4] && s[0]==s[6] && s[0]==s[7])
	    t++;
	  else if(s[0]>s[1] && s[1]>s[3] && s[3]>s[4] && s[4]>s[6] && s[6]>s[7])
	    p++;
	  else
	    g++;
	}
      if(g==mg)
	MG.push_back(S);
      if(g>mg)
	{
	  mg=g;
	  MG.clear();
	  MG.push_back(S);
	}
      if(t==mt)
	MT.push_back(S);
      if(t>mt)
	{
	  mt=t;
	  MT.clear();
	  MT.push_back(S);
	}
      if(p==mp)
	MP.push_back(S);
      if(p>mp)
	{
	  mp=p;
	  MP.clear();
	  MP.push_back(S);
	}
    }
  cout<<"If you want to call a taxi, you should call: ";
  for(int i=0;i<MT.size()-1;i++)
    cout<<MT[i]<<", ";
  cout<<MT.back()<<"."<<endl;
  cout<<"If you want to order a pizza, you should call: ";
  for(int i=0;i<MP.size()-1;i++)
    cout<<MP[i]<<", ";
  cout<<MP.back()<<"."<<endl;
  cout<<"If you want to go to a cafe with a wonderful girl, you should call: ";
  for(int i=0;i<MG.size()-1;i++)
    cout<<MG[i]<<", ";
  cout<<MG.back()<<"."<<endl;
  return 0;
}
