#include <iostream>;
#include <string>
using namespace std;

int main(){
	int n,i,s,j,maxa=-1,maxb=-1,maxc=-1,na=0,nb=0,nc=0;
	char ch;
	string t[100],S;
	int a[100]={0} ,b[100]={0},c[100]={0};

	cin >> n;
	for (i=0; i<n; i++){
		cin >> s;
		cin >> t[i];
		for (j=0; j<s; j++){
			cin >> S;
			if (S[0]==S[1] && S[0]==S[3]
			&& S[0]==S[4] && S[0]==S[6]
			&& S[0]==S[7])
				a[i]++;
			else
				if (S[0]>S[1] && S[1]>S[3]
				&& S[3]>S[4] && S[4]>S[6]
                && S[6]>S[7])
					b[i]++;
				else
					c[i]++;
		}
		if (a[i]>=maxa){
			maxa=a[i];
			na=i;
		}
		if (b[i]>=maxb){
			maxb=b[i];
			nb=i;
		}
		if (c[i]>=maxc){
			maxc=c[i];
			nc=i;
		}
	}
	cout << "If you want to call a taxi, you should call: ";
	for (i=0; i<na; i++)
		if (a[i]==maxa)
			cout << t[i] << ", ";
	cout << t[na] << '.' << endl;

	cout << "If you want to order a pizza, you should call: ";
	for (i=0; i<nb; i++)
		if (b[i]==maxb)
			cout << t[i] << ", ";
	cout << t[nb] << '.' << endl;

	cout << "If you want to go to a cafe with a wonderful girl, you should call: ";
	for (i=0; i<nc; i++)
		if (c[i]==maxc)
			cout << t[i] << ", ";
	cout << t[nc] << '.' << endl;

	cin >> ch;
	return 0;
}
