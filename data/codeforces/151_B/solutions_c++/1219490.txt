#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
#include <string>
#include <iostream>
using namespace std;

vector<string> s[3];
int b[3];

int typ() {
	string str;
	cin >> str;
	int w[6]={0,1,3,4,6,7};
	int f1=0, f2=0;
	for(int i=1;i<6;i++) {
		if(str[w[i]] != str[w[i-1]]) f1=1;
		if(str[w[i]] >= str[w[i-1]]) f2=1;
	}
	return !f1?0: !f2?1: 2;
}

int main(void) {
	int n;
	cin >> n;
	for(int i=0;i<n;i++) {
		string name;
		int k, r[3]={};
		cin >> k >> name;
		for(int j=0;j<k;j++) r[typ()]++;
		for(int j=0;j<3;j++) {
			if(r[j]>b[j]) {
				b[j]=r[j];
				s[j].clear();
			}
			if(r[j]==b[j]) s[j].push_back(name);
		}
	}
	cout << "If you want to call a taxi, you should call:";
	for(int i=0;i<(int)s[0].size();i++)
		cout << " " << s[0][i] << (i==(int)s[0].size()-1?'.':',');
	cout << endl;

	cout << "If you want to order a pizza, you should call:";
	for(int i=0;i<(int)s[1].size();i++)
		cout << " " << s[1][i] << (i==(int)s[1].size()-1?'.':',');
	cout << endl;

	cout << "If you want to go to a cafe with a wonderful girl, you should call:";
	for(int i=0;i<(int)s[2].size();i++)
		cout << " " << s[2][i] << (i==(int)s[2].size()-1?'.':',');
	cout << endl;
	return 0;
}
