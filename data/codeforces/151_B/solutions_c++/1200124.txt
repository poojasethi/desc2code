#include<stdlib.h>
#include<iostream>
#include<string.h>
using namespace std;
int main()
{ int n,i,j,a[110]={0},b[110]={0},c[111]={0},m,p=0,q=0,w=0;
   string s[110],t[110];
   cin>>n;
   for(i=1;i<=n;i++)
   {  cin>>m>>s[i];
   for(j=1;j<=m;j++){cin>>t[j];
                    if(t[j][0]==t[j][1]&&t[j][3]==t[j][4]&&t[j][6]==t[j][7]&&t[j][0]==t[j][3]&&t[j][3]==t[j][6])a[i]++;
                   else{ if(t[j][0]>t[j][1]&&t[j][1]>t[j][3]&&t[j][3]>t[j][4]&&t[j][4]>t[j][6]&&t[j][6]>t[j][7])b[i]++;
                   else c[i]++;
                   }}
   }
   for(i=1;i<=n;i++)
   {if(a[i]>p)p=a[i];
   if(b[i]>q)q=b[i];
   if(c[i]>w)w=c[i];
   }
   cout<<"If you want to call a taxi, you should call:";
   for(i=1;i<=n;i++)if(p==a[i]){cout<<" "<<s[i];break;}
   for(j=i+1;j<=n;j++)if(p==a[j])cout<<", "<<s[j];
   cout<<"."<<endl;
   cout<<"If you want to order a pizza, you should call:";
   for(i=1;i<=n;i++)if(q==b[i]){cout<<" "<<s[i];break;}
   for(j=i+1;j<=n;j++)if(q==b[j])cout<<", "<<s[j];
   cout<<"."<<endl;
   cout<<"If you want to go to a cafe with a wonderful girl, you should call:";
   for(i=1;i<=n;i++)if(w==c[i]){cout<<" "<<s[i];break;}
   for(j=i+1;j<=n;j++)if(w==c[j])cout<<", "<<s[j];
   cout<<"."<<endl;
  
   return 0;
}
