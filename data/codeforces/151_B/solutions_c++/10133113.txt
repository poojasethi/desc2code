#include<iostream>
#include<map>
using namespace std;

int func(string s) {
	if(s[0]==s[1] && s[1]==s[3] && s[3]==s[4] && s[4] == s[6] && s[6]==s[7]) return 1;
	else if(s[0] > s[1] && s[1] > s[3] && s[3] > s[4] && s[4] > s[6] && s[6] > s[7]) return 2;
	else return 3;
}

int main() {

	string l[] = {"If you want to call a taxi, you should call: ", "If you want to order a pizza, you should call: ", "If you want to go to a cafe with a wonderful girl, you should call: "}; 
	
	int n,k=0,t;
	
	string s,s1;
	
	cin >> n;
	
	map<int,int> m[n];
	map<int, string> ma;
	
	while(n--) {
		cin >> t >> s;
		
		ma.insert(make_pair(k, s));
		
		while(t--) {
			cin >> s1;
			int d = func(s1);
			
			m[k][d]++;
		}
		
		k++;
	}
	
	for(int j=1;j<=3;j++) {
		int max = -1;
		for(int i=0;i<k;i++) {
			if(max < m[i][j]) {
				max = m[i][j];
			}
		}
		
		cout<<l[j-1];
		int f = 0;
		for(int i=0;i<k;i++) {
			if(max==m[i][j]) {
				if(f!=0) cout<<", ";
				cout<<ma[i];
				f++;
			}
		}
		cout<<"."<<endl;	
	}
	
	return 0;
}
