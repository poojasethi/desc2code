#include<stdio.h>
#include<string.h>
int main()
{
	
	int n,i,s,a[101][3],b[10],j,k,ok,max,l,q;
	char c[101][21],x[21],p[6];
	for(i=0;i<101;i++)
		for(j=0;j<3;j++)
			a[i][j]=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{		
		scanf("%d",&s);
		scanf("%s",c[i]);
		for(j=0;j<s;j++)
		{
			scanf("%s",x);
			l=strlen(x);
			for(k=0,q=0;k<l;k++)
			{
				if(x[k]>='0' && x[k]<='9')
				{p[q]=x[k];q++;}
			}
			for(k=0;k<5;k++)
			{
				if(p[k]<=p[k+1])
					break;
			}
			if(k==5) {a[i][1]++; continue;}
			memset(b,0,sizeof(b));
			for(k=0;k<l;k++)
			{
				if(x[k]>='0' && x[k]<='9')
					b[x[k]-'0']++;
			}
			for(k=0;k<=9;k++)
			{
				if(b[k]==6)
				{a[i][0]++;break;}
			}
			if(k==10)
				a[i][2]++;
		}
	}
	for(j=0;j<3;j++)
	{
	
		
		if(j==0)
			printf("If you want to call a taxi, you should call: ");
		else if(j==1)
			printf("If you want to order a pizza, you should call: ");
		else
			printf("If you want to go to a cafe with a wonderful girl, you should call: ");
        max=0;
		for(k=1;k<n;k++)
		{
			if(a[k][j]>a[max][j]) max=k;
		}
		for(k=0,ok=0;k<n;k++)
		{
			if(a[k][j]==a[max][j]) 
			{
				if(ok==1) printf(", ");
				printf("%s",c[k]);ok=1;
			}	
		}
		printf(".\n");
	}
	return 0;
}
