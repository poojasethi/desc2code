#include<stdio.h>
#include<string.h>

char name[100][100];
int res[100][3],max[3];
int n;

void output(int type){
	int flag=0,i;
	for(i=0;i<n;i++){
		if(res[i][type]==max[type]){
			if(flag){
				printf(",");
			}
			printf(" %s",name[i]);
			flag=1;
		}
	}
	printf(".\n");
	return;
}

int main(){
	int t,i,j,k;
	int a,b,c;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		for(j=0;j<3;j++){
			res[i][j]=0;
		}
		scanf("%d%s",&t,&name[i]);
		for(j=0;j<t;j++){
			scanf("%d-%d-%d",&a,&b,&c);
			if(a%10==a/10){
				if(a==b&&a==c) res[i][0]++;
				else res[i][2]++;
			}
			else{
				a=a*10000+b*100+c;
				for(k=0;k<5;k++){
					if(a%10>=(a/10)%10){
						res[i][2]++;
						a=-1;
						break;
					}
					a/=10;
				}
				if(a!=-1)res[i][1]++;
			}
		}
		if(max[0]<res[i][0])max[0]=res[i][0];
		if(max[1]<res[i][1])max[1]=res[i][1];
		if(max[2]<res[i][2])max[2]=res[i][2];
	}
	printf("If you want to call a taxi, you should call:");	
	output(0);
	printf("If you want to order a pizza, you should call:");	
	output(1);
	printf("If you want to go to a cafe with a wonderful girl, you should call:");	
	output(2);
	
	return 0;
}
