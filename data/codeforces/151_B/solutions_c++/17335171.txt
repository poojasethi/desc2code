#include <iostream>
#include <set>
using namespace std;
int main()
{
	int n,g[3] = {0};
	cin>>n;
	string ans[3] = {""};
	

	while(n--)
	{
		set<char>s;
		string name,ph;
		int sz,gi[3] = {0};
		cin>>sz>>name;

		while(sz--)
		{
			s.clear();
			cin>>ph;

			for(int i = 0;i<8;++i)
				s.insert(ph[i]);

			//cout<<s.size()<<endl;

			if(s.size() == 2)
				++gi[2];
			else if(s.size()<7)
				++gi[0];
			else{
				int c = 0;
				if(ph[1] < ph[0])
					++c;

				for(int i = 4;i<8;i+=3){
					if(ph[i] < ph[i-1] && ph[i-1] < ph[i-3])
						++c;
				}

				if(c == 3)
					++gi[1];
				else
					++gi[0];
			}
		}

		//cout<<gi[0]<<" "<<gi[1]<<" "<<gi[2]<<endl;

		for(int i = 0;i<3;++i){
			if(gi[i] > g[i]){
				g[i] =gi[i];
				ans[i] = name;
			}

			else if(gi[i] == g[i]){
				if(ans[i]!="")
					ans[i]+=", ";
				ans[i]+= name;
			}

			else
				continue;
		}

		//cout<<g[0]<<" "<<g[1]<<" "<<g[2]<<endl;
	}

	cout<<"If you want to call a taxi, you should call: "<<ans[2]<<"."<<endl;
	cout<<"If you want to order a pizza, you should call: "<<ans[1]<<"."<<endl;
	cout<<"If you want to go to a cafe with a wonderful girl, you should call: "<<ans[0]<<"."<<endl;
}
