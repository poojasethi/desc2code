#include <cstdio>
#include <algorithm>
#define FI(a, b, c) for(int a = b; a <= c; a++)
#define FD(a, b, c) for(int a = b; a >= c; a--)
using namespace std;

int n, T, P, G;
struct book{
	int t, p, g;
	char s[22];
} r[105];

int main(){
	scanf("%d", &n);
	FI(i, 1, n){
		int a;
		char q[10];
		scanf("%d %s", &a, r[i].s);
		
		while(a--){
			scanf("%s", q);
			bool pf = 1, tf = 1;
			char pre = q[0];
			
			FI(j, 1, 7) if(q[j] != '-'){
				if(q[j] != pre) tf = 0;
				if(q[j] >= pre) pf = 0;
				
				pre = q[j];
			}
			
			if(tf) r[i].t++;
			else if(pf) r[i].p++;
			else r[i].g++;
		}
		
		T = max(T, r[i].t);
		P = max(P, r[i].p);
		G = max(G, r[i].g);
	}
	
	printf("If you want to call a taxi, you should call:");
	bool f = 1;
	FI(i, 1, n) if(r[i].t == T){
		if(!f) printf(",");
		printf(" %s", r[i].s);
		f = 0;
	}
	puts(".");
	
	printf("If you want to order a pizza, you should call:");
	f = 1;
	FI(i, 1, n) if(r[i].p == P){
		if(!f) printf(",");
		printf(" %s", r[i].s);
		f = 0;
	}
	puts(".");
	
	printf("If you want to go to a cafe with a wonderful girl, you should call:");
	f = 1;
	FI(i, 1, n) if(r[i].g == G){
		if(!f) printf(",");
		printf(" %s", r[i].s);
		f = 0;
	}
	puts(".");
}
