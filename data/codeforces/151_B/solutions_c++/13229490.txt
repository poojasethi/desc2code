#include<bits/stdc++.h>
using namespace std;

char names[105][24], x[7];
int n, s[105], T[3][105], mx[3] = {-1,-1,-1};
char ss[3][110] = {
		"If you want to call a taxi, you should call: ",
		"If you want to order a pizza, you should call: ",
		"If you want to go to a cafe with a wonderful girl, you should call: "
};

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d", &n);
	for(int i=0 ; i<n ; ++i) {
		scanf("%d %s", &s[i], names[i]);
		for(int j=0 ; j<s[i] ; ++j){
			for(int k=0 ; k<6 ; ++k) {
				scanf(" %c", &x[k]);
				if(x[k] == '-') --k;
			}
			bool t = 1, p = 1;
			for(int k=0 ; k<5 ; ++k) if(x[k] != x[k+1]) t = 0;
			for(int k=0 ; k<5 ; ++k) if(x[k] <= x[k+1]) p = 0;
			if(t) ++T[0][i];
			else if(p) ++T[1][i];
			else ++T[2][i];
		}
	}
	for(int i=0 ; i<3 ; ++i) {
		for(int j=0 ; j<n ; ++j) mx[i] = max(mx[i], T[i][j]);
		printf(ss[i]);
		bool ok = false;
		for(int j=0 ; j<n ; ++j) {
			if(T[i][j] == mx[i]) {
				if(ok) printf(", ");
				printf("%s", names[j]);
				ok = 1;
			}
		}
		puts(".");
	}
	return 0;
}
