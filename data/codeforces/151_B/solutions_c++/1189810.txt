#include <iostream>
#include <string>
#include <fstream>

int main(){
	using namespace std;
	int n;
	cin >> n;
	string a, b, c;
	int ac = -1, bc = -1, cc = -1;
	for(int i = 0; i < n; i++){
		int m;
		string cur;
		cin >> m >> cur;
		int acc = 0, bcc = 0, ccc = 0;
		for(int j = 0; j < m; j++){
			char ch;
			int a, b, c;
			cin >> a >> ch >> b >> ch >> c;
			if(a == b && b == c && a / 10 == a % 10){
				acc++;
				continue;
			}
			if(a / 10 > a % 10 && a % 10 > b / 10 && b / 10 > b % 10 && b % 10 > c / 10 && c / 10 > c % 10){
				bcc++;
				continue;
			}
			ccc++;
		}
		if(acc == ac){
			a += ", " + cur;
		}
		if(bcc == bc){
			b += ", " + cur;
		}
		if(ccc == cc){
			c += ", " + cur;
		}
		if(acc > ac){
			a = cur;
			ac = acc;
		}
		if(bcc > bc){
			b = cur;
			bc = bcc;
		}
		if(ccc > cc){
			c = cur;
			cc = ccc;
		}
	}
	cout << "If you want to call a taxi, you should call: " << a << "." << endl;
	cout << "If you want to order a pizza, you should call: " << b << "." << endl;
	cout << "If you want to go to a cafe with a wonderful girl, you should call: " << c << ".";
	return 0;
}
