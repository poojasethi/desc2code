#include<cstdio>
#include<cstring>
using namespace std;
char name[105][21];
int aa[105][3];
int diff(int a,int b,int c){
    int a1,a2,b1,b2,c1,c2;
    a1=a/10;a2=a%10;
    b1=b/10;b2=b%10;
    c1=c/10;c2=c%10;
    if(a1>a2&&a2>b1&&b1>b2&&b2>c1&&c1>c2)return 1;
    else return 0;
}
int main(){
    int n,m,a,b,c,t=0,d1=0,d2=0,d3=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d%s",&m,name[i]);d1=d2=d3=0;
        for(int j=0;j<m;j++){
            scanf("%d-%d-%d",&a,&b,&c);
            if(a==b&&b==c&&(a%10==a/10))d1++;
            else if(diff(a,b,c))d2++;
            else d3++;
        }
        if(d1>aa[0][0])aa[0][0]=d1;
        if(d2>aa[0][1])aa[0][1]=d2;
        if(d3>aa[0][2])aa[0][2]=d3;
        aa[i+1][0]=d1;aa[i+1][1]=d2;aa[i+1][2]=d3;
    }
    for(int i=0;i<3;i++){
        t=0;
        if(i==0)printf("If you want to call a taxi, you should call: ");
        else if(i==1)printf("If you want to order a pizza, you should call: ");
        else printf("If you want to go to a cafe with a wonderful girl, you should call: ");
        for(int j=1;j<=n;j++){
            if(aa[j][i]==aa[0][i]){
                if(t==1)printf(", ");else t=1;
                printf("%s",name[j-1]);
            }
        }printf(".\n");
    }
    return 0;
}
