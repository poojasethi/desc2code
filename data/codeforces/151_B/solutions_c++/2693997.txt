#include <iostream>
#include <string>
#include <cstdio>

using namespace std;

string name[200];
int a[10],b[200][3],x,i,j,N,M;

int main(){
    scanf("%d",&N);
    for (i=1;i<=N;i++){
        cin >> M >> name[i];
        while (M--){
            scanf("%1d%1d-%1d%1d-%1d%1d",&a[0],&a[1],&a[2],&a[3],&a[4],&a[5]);
            x=0;
            for (j=1;j<6;j++)
                if (a[j]!=a[j-1]) x=1;
            if (x)
                for (j=1;j<6;j++)
                    if (a[j-1]<=a[j]) x=2;
            ++b[i][x];
        }
    }
    for (int k=0;k<3;k++){
        int Ma=0;
        for (i=1;i<=N;i++) Ma=max(Ma,b[i][k]);
        if (k==0) cout << "If you want to call a taxi, you should call:";
        else if (k<2) cout << "If you want to order a pizza, you should call:";
        else cout << "If you want to go to a cafe with a wonderful girl, you should call:";
        int flag=0;
        for (i=1;i<=N;i++)
            if (b[i][k]==Ma){
                if (flag) cout << ',';
                flag=1;
                cout << ' ' << name[i];
            }
        cout << '.' << endl;
    }
}
