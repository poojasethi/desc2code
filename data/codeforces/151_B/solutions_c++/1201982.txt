#include<stdio.h>

int n;
int t[150];
int p[150];
int g[150];

char a[150][30];

int main()
{
	scanf("%d",&n);
	for (int i = 0; i < n; i++)
	{
		int x;
		scanf("%d%s",&x,a[i]);
		g[i]=x;
		for(int o=0; o<x; o++)
		{
			char c[30];
			scanf("%s",c);
			if(c[0]==c[1] && c[0]==c[3] && c[0]==c[4] && c[0]==c[6] && c[0]==c[7])
			{ t[i]++; g[i]--; continue; }
			if(c[0]>c[1] && c[1]>c[3] && c[3]>c[4] && c[4]>c[6] && c[6]>c[7])
			{ p[i]++; g[i]--; }
		}
	}

	int mxg=0,mxt=0,mxp=0;

	for(int i=0; i<n; i++)
	{ 
		if(g[i]>mxg)
		mxg=g[i];
		if(t[i]>mxt)
		mxt=t[i];
		if(p[i]>mxp)
		mxp=p[i];
	}
	
	int gel=0;
	
	printf("If you want to call a taxi, you should call: ");
	for(int i=0; i<n; i++)
	if(t[i]==mxt)
	{
		if(gel==1)	printf(", ");
		printf("%s",a[i]);
		gel=1;
	}

	printf(".\n"); gel=0;
	
	printf("If you want to order a pizza, you should call: ");
	for(int i=0; i<n; i++)
	if(p[i]==mxp)
	{
		if(gel==1) printf(", ");
		printf("%s",a[i]);
		gel=1;
	}

	printf(".\n"); gel=0;
	
	printf("If you want to go to a cafe with a wonderful girl, you should call: ");
	for(int i=0; i<n; i++)
	if(g[i]==mxg)
	{
		if(gel==1) printf(", ");
		printf("%s",a[i]);
		gel=1;
	}
	
	printf(".");	
	return 0;
}
