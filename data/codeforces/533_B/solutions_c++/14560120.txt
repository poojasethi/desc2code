#include <bits/stdc++.h>

using namespace std;

int n;
int d[2000010], p[2000010];
long long dp[2000010][2];

int main() {
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> p[i] >> d[i];
		p[i] --;
		dp[i][1] = -1000000000000000;
	}
	for (int i = n-1; i >= 0; i--) {
		dp[i][1] = max(dp[i][1], dp[i][0] + d[i]);
		if (!i) break;
		long long nv0, nv1;
		nv0 = max(dp[p[i]][0] + dp[i][0], dp[p[i]][1] + dp[i][1]);
		nv1 = max(dp[p[i]][0] + dp[i][1], dp[p[i]][1] + dp[i][0]);
		dp[p[i]][0] = nv0, dp[p[i]][1] = nv1;
	}
	cout << dp[0][1] << endl;
	return 0;
}
