#include <iostream>
#include <string>
using namespace std;

long long d[30];

int main()
{
	string s;
	cin >> s;

	int len = s.length();
	int ds = 26;
	for (int i = 1; i <= 26; i++)
	{
		int occ[26];
		for (int j = 0; j < 26; j++)
			occ[j] = 0;

		long long div = 0;
		int right = 0;
		for (int left = 0; left < len; left++)
		{
			while (right < len && div <= i)
			{
				if (occ[s[right]-'a'] == 0)
					div++;
				occ[s[right]-'a']++;
				right++;
			}

			if (div <= i)
				d[i] += right - left;
			else
				d[i] += right - left - 1;

			occ[s[left]-'a']--;
			if (occ[s[left]-'a'] == 0)
				div--;
		}

		if (d[i] == d[i-1])
		{
			ds = i - 1;
			break;
		}
	}

	cout << ds << "\n";
	for (int i = 1; i <= ds; i++)
		cout << d[i] - d[i-1] << "\n";

	return 0;
}
