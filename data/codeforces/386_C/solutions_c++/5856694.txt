#include <algorithm>
#include <cstdio>
#include <cstring>
#include <vector>
#define FOR(i,a,b) for(i=a; i<=b; i++)
#define TFOR(i,a,b) for(i=a; i>=b; i--)
#define pb push_back
#define all(x) x.begin(),x.end()
#define MAXN 300002
using namespace std;
long long res[MAXN];
int next[MAXN][26];
char A[MAXN];
int main()
{
	int N,i,j,t;
	scanf("%s" , A+1 );
	N = strlen(A+1);
	TFOR(i,N,1)
		FOR(j,0,25)
			next[i][j] = (A[i] - 'a' == j) ? i : next[i+1][j];

	vector < int > V;
	FOR(i,1,N)
	{
		V.clear();
		FOR(j,0,25)
			if(next[i][j])
				V.pb( next[i][j] );

		sort( all(V) );
		V.pb( N+1 );
		if(i == 1) t = V.size() - 1;

		for(j = 1; j < V.size(); j++)
			res[j] += V[j] - V[j-1];
	}

	printf("%d\n" , t ); 
	FOR(i,1,t)
		printf("%lld\n" , res[i] );

	return 0;
}
