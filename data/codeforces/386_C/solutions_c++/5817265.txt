#include <cstdio>
#include <cstring>
#include <algorithm>
const int MAX_N=300050;
char s[MAX_N];
int pos[MAX_N];
int sa[30];
long long ans[30];
int k=0;
int main(){
	scanf("%s",s+1);
	int len=strlen(s+1);
	for (int i=1;i<=len;++i){
		if (pos[s[i]-'a'+1]==0) k++;
		pos[s[i]-'a'+1]=i;
		for (int i=0;i<=27;++i)
		  sa[i]=pos[i];
		std::sort(sa,sa+28);
		for (int i=26;i>=26-k+1;i--)
		  ans[26-i+1]+=sa[i+1]-sa[i];
	}
	printf("%d\n",k);
	for (int i=1;i<=k;++i)
	  printf("%I64d\n",ans[i]);

}
