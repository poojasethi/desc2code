#include <cstdio>
#include <cstring>
#include <algorithm>
#define MXN 300010
using namespace std;
typedef long long LL;

LL N;
char str[MXN];
LL ans[30], dofs = -1;

LL alp[30];
LL getsol(LL bound){
	LL div = 0, ret = 0;
	fill(alp, alp + 26, 0);

	for(LL i = 0, j = 0; i < N; i++){
		while(j < N && div < bound){
			alp[str[j] - 'a'] ++;
			if(alp[str[j] - 'a'] == 1) div ++;
			j ++;
		}

		if(div >= bound) ret += N - j + 1;

		alp[str[i] - 'a'] --;
		if(alp[str[i] - 'a'] == 0) div --;
	}
	return ret;
}

int main (){
	scanf("%s", str);
	N = strlen(str);

	for(LL i = 26; i >= 1; i--){
		ans[i] = getsol(i);
		if(ans[i] != 0 && dofs == -1) dofs = i;
	}

	printf("%I64d\n", dofs);
	for(LL i = 1; i <= dofs; i++){
		printf("%I64d\n", ans[i] - ans[i + 1]);
	}
}
