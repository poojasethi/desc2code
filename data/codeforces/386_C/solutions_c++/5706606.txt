#include <iostream>
#include <map>
#include <string>
#include <vector>
#include <cstdio>
#include <queue>
#include <cctype>
#include <cstdlib>
#include <algorithm>
using namespace std;

int main()
{
	map<char,int> m;
	
	long long ar[26]={0};
	string s;
	cin>>s;
	for (int i=(int)s.length()-1; i>=0; i--)
	{
		m[s[i]]= i;
		vector<int> v;
		for (map<char,int>::iterator it = m.begin(); it!=m.end(); it++)
		{
			v.push_back(it->second);
		}
		sort(v.begin(), v.end());
		for (int i=0; i<(int)v.size(); i++)
		{
			int t = i<(int)v.size()-1 ? v[i+1] : (int)s.length();
			ar[i] += t-v[i]; 
		}
	}
	vector<long long > a;
	for (int i=0; i<26; i++)
	{
		if(ar[i])
		{
			a.push_back(ar[i]);
		}
	}
	cout << a.size()<<endl;
	for (int i=0; i<a.size(); i++)
	{
		cout <<a[i]<<endl;
	}
  return 0;
}
