#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int nextchar[300001][27], n, dv;
long long diversity[27];
string s;
int main() {
	cin >> s;
	n=s.size();
	for(int i=0;i<27;i++)
		nextchar[n][i] = n;
	for(int i=n-1;i>=0;i--)
		for(int j=0;j<27;j++) {
			if(s[i] == 'a'+j)
				nextchar[i][j]=i;
			else
				nextchar[i][j] = nextchar[i+1][j];
		}
	
	for(int i=0;i<n;i++) {
		sort(nextchar[i], nextchar[i] + 27);
		int cur=i;
		for(int j=0;j<27;j++)
			diversity[j] += nextchar[i][j] - cur, cur=nextchar[i][j];
	}
	for(int i=0;i<27;i++)
		if(diversity[i] != 0)
			dv = i;
	cout << dv << '\n';
	for(int i=1;i<=dv;i++)
		cout << diversity[i] << '\n';
	return 0;
}
