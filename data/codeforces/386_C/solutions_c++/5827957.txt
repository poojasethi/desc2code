#include<cstring>
#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int n,i,j,now[29],hh[29];
long long ans[28];
char s[300001];
int main()
{
    scanf("%s",s);
    int len=strlen(s);
    for (i=0;i<=26;i++) now[i]=-1;
    for (i=0;i<len;i++)
    {
        now[s[i]-'a']=i;
        for (j=0;j<=26;j++) hh[j]=now[j];
        sort(hh,hh+27);
        for (j=26;j>0;j--){
                ans[26-j+1]+=hh[j]-hh[j-1];
        }
        //cout<<ans[2]<<endl;
    }
    for (i=1;i<=26;i++) if (!ans[i]) break;
    printf("%d\n",i-1);
    for (i=1;i<=26;i++)
      if (ans[i]) printf("%I64d\n",ans[i]);
}
