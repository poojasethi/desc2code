#include<stdio.h>
#include<string.h>
    
typedef __int64 LL;
const int N=300001;
char s[N];
int pre[128];
LL a[30],b[30];
int main(){
    int i,j,n,m,k,t,x,y;
    gets(s);
    n=strlen(s);
    memset(pre,-1,sizeof pre);
    for(i=0;i<n;i++){
        k=pre[s[i]];
        for(k=i-k-1,t=0;k;t++,k-=b[t]);
        for(j=t+1;j;j--) b[j]+=b[j-1],b[j-1]=0;
        b[1]++;
        pre[s[i]]=i;
        for(j=1;j<27;j++) a[j]+=b[j];
    }
    for(i=26;!a[i];i--);
    printf("%d\n",i);
    for(j=1;j<=i;j++) printf("%I64d\n",a[j]);
    return 0;
}
