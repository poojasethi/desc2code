#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long LL;
const int maxn = 300002;
const int maxm = 27;
int N, now[maxm], to[maxm];
LL ans[maxm];
char s[maxn];
int main()
{
    scanf ("%s", s+1);
    N = strlen(s+1);
    for (int i=0; i<maxm; ++i) {
        now[i] = N + 1;
    }
    for (int i=N; i>0; --i) {
        now[s[i]-'a'] = i;
        for (int j=0; j<maxm; ++j) {
            to[j] = now[j];
        }
        sort(to, to+maxm);
        for (int j=0; j<maxm-1; ++j) {
            ans[j+1] += to[j+1] - to[j];
        }
    }
    int l = 26;
    for (; l>0; --l) {
        if (ans[l]) {
            break;
        }
    }
    printf ("%d\n", l);
    for (int i=1; i<=l; ++i) {
        printf ("%I64d\n", ans[i]);
    }
    return 0;
}
