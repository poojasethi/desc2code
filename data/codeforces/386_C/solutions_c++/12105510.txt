#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define n   300005
#define I64 long long
#define For(i,a,b)  for(int i=a;i<=b;i++)

I64     Ans[n];
int     N,A[n],B[n];
char    s[n];

bool    Cmp(int a,int b)    {return a>b;}

int     main(){
        scanf("%s",s+1);    N=strlen(s+1);
        For(i,1,N)  {
            B[s[i]-'a'+1]=i;
            For(j,1,26) A[j]=B[j];
            sort(A+1,A+27,Cmp);
            For(j,1,26) Ans[j]+=A[j]-A[j+1];
        }

        int ret=0;
        For(i,1,26) if  (Ans[i])    ret++;
        printf("%d\n",ret);
        For(i,1,26) if  (Ans[i])    printf("%I64d\n",Ans[i]);
}
