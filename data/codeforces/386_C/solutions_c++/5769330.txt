#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char a[345678]; int n, c1[26], c2[26], d1, d2, td;
int main() {
	scanf("%s", a); n = strlen(a);
	for (int i = 0; i < n; i++) c1[a[i]-'a']++;
	for (int i = 0; i < 26; i++) td += (c1[i]>0);
	printf("%d\n", td);
	for (int k = 1; k <= td; k++) {
		long long ans = 0; d1 = d2 = 0;
		for (int i = 0; i < 26; i++) c1[i] = c2[i] = 0;
		for (int i = 0, s = 0, e = -1; i < n; i++) {
			while (s < n && d1 < k) {
				d1 += !c1[a[s]-'a'];
				c1[a[s]-'a']++; s++;
			}
			while (e+1 < n && d2 + (!c2[a[e+1]-'a']) <= k) {
				d2 += !c2[a[e+1]-'a'];
				c2[a[e+1]-'a']++; e++;
			}
			if (d1==k && d2==k) ans += e-s+2;
			if (c1[a[i]-'a']--==1) d1--;
			if (c2[a[i]-'a']--==1) d2--;
		}
		printf("%lld\n", ans);
	}

	return 0;
}
