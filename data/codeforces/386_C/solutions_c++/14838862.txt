#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

#define int long long

vector<int> arr(26);
string s;
int ans[40];

main()
{
    cin >> s;

    for(int i = 0; i < 26; ++i) arr[i] = -1;

    for(int i = 0; i < s.size(); ++i) {
        int al = s[i] - 'a';
        arr[al] = i;

        vector<int> sarr = arr;
        sort(sarr.begin(), sarr.end());

        for(int j = 0; j < 26; ++j) {
            if(sarr[j] == -1) continue;

            int l;
            if(j == 0) l = -1;
            else l = sarr[j - 1];

            ans[26 - j] += sarr[j] - l;
        }
    }

    int td = 26;
    while(ans[td] == 0) --td;

    cout << td << endl;
    for(int i = 1; i <= td; ++i) cout << ans[i] << endl;
}
