#include <iostream>
#include <cstring>
using namespace std;
long long c[30], ans[30], f;
int main()
{
	ios::sync_with_stdio(false);
	string s;
	cin >> s;
	for (int i = 0; i < s.length(); i++)
		s[i] -= 'a';
	for (int k = 1; k <= 26; k++)
	{
		memset(c, 0, sizeof(c));
		f = 0;
		long long l = 0, r = 1;
		c[s[0]]++;
		f++;
		while (l < s.length())
		{
			while (f <= k && r < s.length())
			{
				c[s[r]]++;
				if (c[s[r]] == 1)
					f++;
				r++;
			}
			if (f > k)
			{
				c[s[--r]]--;
				f--;
			}
			ans[k] += r - l;
			c[s[l]]--;
			if (c[s[l]] == 0)
				f--;
			l++;
		}
	}
	int p = -1;
	for (int i = 26; i >= 1; i--)
	{
		ans[i] -= ans[i - 1];
		if (p == -1 && ans[i])
			p = i;
	}
	cout << p << endl;
	for (int i = 1; i <= p; i++)
		cout << ans[i] << endl;
	return 0;
}
