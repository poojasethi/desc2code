#include<cstdio>
#include<string>
#include<iostream>
#include<algorithm>
using namespace std;
int sm[30],cm[30];
long long all[30];
bool cmp(int a,int b)
{
    return a>b;
}
int main()
{
    string s;
    cin>>s;
    int n=s.size(),nw;
    s=" "+s;
    for(int i=1;i<=n;i++)
    {
        sm[s[i]-'a']=i;
        for(int j=0;j<26;j++)
            cm[j]=sm[j];
        sort(cm,cm+27,cmp);
        for(int j=1;cm[j-1]>0;j++)
            all[j]+=(long long)(cm[j-1]-cm[j]);
    }
    for(int i=1;i<=30;i++)
        if(all[i]<=0)
        {
            nw=i-1;
            break;
        }
    printf("%d\n",nw);
    for(int i=1;i<=nw;i++)
        printf("%I64d\n",all[i]);
}
