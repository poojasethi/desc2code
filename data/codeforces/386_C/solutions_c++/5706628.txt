#include <iostream>
#include <set>
using namespace std;

string s;
long long arr[27] = {}, div;
int cur[27][128] = {};
set<char> un[28];

int main(){

    ios::sync_with_stdio(false);
    cin >> s;
    for(int i = 0; i<s.size(); i++) un[27].insert(s[i]);
    div = un[27].size();
    cout << div << endl;
    int l[27] = {};
    for(int i = 0; i<s.size(); i++){
        for(int j = 0; j<=div; j++){
            cur[j][s[i]]++;
            un[j].insert(s[i]);
            while(un[j].size()>j){
                cur[j][s[l[j]]]--;
                if(cur[j][s[l[j]]]==0) un[j].erase(s[l[j]]);
                l[j]++;
            }
        }
        for(int j = 1; j<=div; j++) arr[j]+=l[j-1]-l[j];
    }
    for(int i = 1; i<=div; i++) cout << arr[i] << endl;

}
