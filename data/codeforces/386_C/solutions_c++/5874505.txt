#include <iostream>
#include <algorithm>
#include <cstring>
#include <cstdio>
#include <string>
using namespace std;

#define M 27
#define N 300005

long long sum[M];
int pos[M];
int n;
int main()
{
    string str;
    cin>>str;
    int l = str.length();
    for(int i=0;i<l;i++){
        int t=str[i]-'a';
        if(!pos[t]) n++;
        pos[t]=i+1;
        int c[M];
        for(int j=0;j<M;j++) c[j]=pos[j];
        sort(c,c+M);
        for(int j=0;j<M && c[M-j-1];j++){
            sum[j+1] += c[M-j-1]-c[M-j-2];
        }
    }
    cout<<n<<endl;
    for(int i=1;i<=n;i++){
        cout<<sum[i]<<endl;
    }


}
