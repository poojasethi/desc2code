#include <bits/stdc++.h>
using namespace std;

int N, F[300003][26], Ds;
char S[300003];

int getD(int l, int r)
{
    int d = 0;
    for (int j = 0; j < 26; j++)
        d += (F[r][j] - F[l - 1][j] > 0);
    return d;
}

main()
{
    scanf("%s", S + 1);    N = strlen(S + 1);
    for (int i = 1; i <= N; i++)
    {
        for (int j = 0; j < 26; j++) F[i][j] = F[i - 1][j];
        F[i][S[i] - 'a']++;
    }

    printf("%d\n", Ds = getD(1, N));
    for (int D = 1; D <= Ds; D++)
    {
        long long T = 0;
        for (int i = 1, l = 1, r = 1; i <= N; i++)
        {
            for (; l <= N && getD(i, l) < D; l++) ;
            for (r = max(l, r); r <= N && getD(i, r) == D; r++) ;
            T += r - l;
        }
        printf("%I64d\n", T);
    }
}
