#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 300010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int f[N][27],g[N][27];
int64 an[27];
char p[N];
int main() {
	scanf("%s",p+1); n=strlen(p+1);
	f[0][0]=1;
	For(i,1,n) {
		int A=p[i]-'a'; f[i][0]=1;
		For(j,0,26) if (f[i-1][j]) {
			if (!(g[i-1][j]&1<<A)) {
				f[i][j+1]+=f[i-1][j];
				g[i][j+1]=g[i-1][j]|1<<A;
			} else f[i][j]+=f[i-1][j],g[i][j]=g[i-1][j];
		}
		For(j,1,26) if (f[i][j]>=0) an[j]+=f[i][j];
	}
	for (i=26;i&&!an[i];i--);
	printf("%d\n",i);
	For(j,1,26) if (an[j]) printf("%I64d\n",an[j]);
	//for(;;);
	return 0;
}
