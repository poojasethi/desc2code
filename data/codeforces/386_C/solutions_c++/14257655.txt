#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

char str[300100];
int last[30];
long long num_sub[30], ans[30];

int main()
{
	int i, j, k, len;
	scanf("%s", str+1);
	len = strlen(str+1);
	int cnt_ch, seg_len;
	for (i = 1; i <= len; ++i) {
		j = last[str[i]-'a'];
		
		seg_len = i - j - 1;
		for (cnt_ch = 1; seg_len; ++cnt_ch) {
			seg_len -= num_sub[cnt_ch];
		}
		num_sub[cnt_ch] += num_sub[cnt_ch-1];
		for (k = cnt_ch-1; k >= 1; --k) {
			num_sub[k] = num_sub[k-1];
		}
		++num_sub[1];
			
		last[str[i]-'a'] = i;
		for (k = 1; k <= 26; ++k) {
			ans[k] += num_sub[k];
		}		
	}
	int tot = 26;
	while (ans[tot] == 0) {
		--tot;
	}
	printf("%d\n", tot);
	for (k = 1; k <= tot; ++k) {
		printf("%lld\n", ans[k]);
	}
}
