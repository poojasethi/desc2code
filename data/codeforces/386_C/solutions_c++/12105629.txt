#include <cstdio>
#include <algorithm>
using namespace std;

const int N=3e5+5;

int n,m;

char s[N];
int last[26];
int ord[26];
bool cmp(const int i,const int j){
	return last[i]>last[j];
}

long long ans[27];

int main(){
	scanf("%s",s+1);
	for(n=1;s[n];n++); n--;
	for(int i=0;i<26;i++) ord[i]=i;
	for(int i=1;i<=n;i++){
		last[s[i]-'a']=i;
		sort(ord,ord+26,cmp);
		int t=i;
		for(int j=0;j<26;j++){
			ans[j]+=t-last[ord[j]];
			t=last[ord[j]];
		}
		ans[26]+=t;
	}
	m=26;
	while(!ans[m]) m--;
	printf("%d\n",m);
	for(int i=1;i<=m;i++) printf("%I64d\n",ans[i]);
}
