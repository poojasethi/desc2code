#include <cstdio>
#include <vector>
#include <cstring>

char ss[400000];
long long unsigned int resp[30];
int md;

int diver(long long unsigned int* resp,char*ss,int init,int sz,int mdiv){
  int maxDivi[30]={0};
  int md=0;
  int i;
  for(i=init;i<sz;i++){
    if(0==maxDivi[ss[i]-'a']++){
      md++;
    }
    if(md < mdiv)
      resp[md]++;
    else{
       resp[md]+=sz-i;
       return 0;  
    }  
  }
  return 0;
}

long long unsigned int maxDiv[30];

int main(){
  scanf("%s",ss);
  int len=strlen(ss);
  for(int i=0;i<len;i++){
    if(0==maxDiv[ss[i]-'a']++){
      md++;
    }
  }
  diver(resp,ss,0,len,md);
  int auxmd=md;
  for(int i=1;i<len;i++){
    if(0==--maxDiv[ss[i-1]-'a'])
      md--;
    diver(resp,ss,i,len,md);
  }
  printf("%d\n",auxmd);
  for(int i=1;i<=auxmd;i++)
    printf("%I64u\n",resp[i]);
}
