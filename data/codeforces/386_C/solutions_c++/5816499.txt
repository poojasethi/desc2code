#include <cstdio>
#include <set>
using namespace std;

char s[300001];
int k, c[128], m;

int main() {
  scanf("%s", s);
  k = set<char>(s, s + 300001).size() - 1;
  printf("%d\n", k);
  for (int i = 1; i <= k; ++i) {
    long long b = 0;
    int j = 0, l = 0;
    for (int z = 0; s[z]; ++z) {
      while (m < i && s[j])
        m += !c[s[j++]]++;
      while (c[s[l]])
        ++l;
      if (m == i)
        b += l - j + 1;
      m -= !--c[s[z]];
    }
    printf("%I64d\n", b);
  }
}
