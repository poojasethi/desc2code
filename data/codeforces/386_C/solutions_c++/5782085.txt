#include <iostream>
#include <string>
#include <vector>

using namespace std;

const int MAXN = 300000 + 11;
string s;

int main()
{
	cin >> s;
	long long ans[27]  = {0}, now[27]  = {0}, total; 
	for (int i = 0;i < s.size();i++)
	{
		int bny = 0;
		total = 0;
		now[1]++;
		for (int j = i - 1;j >= 0;j--)
		{
			if (s[i] == s[j]) break;
			//cout << i << ' ' << j << ' ' << total << endl;
			if ((bny & (1 << (s[j] - 'a'))) == 0)
			{
				//cout << 123 << endl;
				bny |= 1 << (s[j] - 'a');
				total++;
			}
			now[total]--;
			now[total + 1]++;
		}
		for (int j = 1;j <= 26;j++)
			ans[j] += now[j];
	}
	int  num = 26;
	for (;num >= 1;num--)
		if (ans[num]) break;
	cout << num << endl;
	for (int i = 1;i <= num;i++)
		cout << ans[i] << endl;
	return 0; 
}
