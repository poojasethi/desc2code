#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int l, i, j, now[30], to[30];
char S[310000];
long long ans[30];

int main() {
    scanf("%s", S + 1);
    l = strlen(S + 1);
    for (i = 0; i <= 26; i++)   now[i] = l + 1;
    for (i = l; i; i--) {
        now[S[i] - 'a'] = i;
        for (j = 0; j <= 26; j++)   to[j] = now[j];
        sort(to, to + 27);
        for (j = 0; j < 26; j++)    ans[j + 1] += to[j + 1] - to[j];
    }
    for (i = 1; i <= 26; i++)   if (!ans[i])    break;
    printf("%d\n", i - 1);
    for (i = 1; i <= 26; i++)
        if (ans[i] > 0) printf("%I64d\n", ans[i]);
}
