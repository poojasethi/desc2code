#include <iostream>
#include <vector>
#include <algorithm>
#include <stdio.h>
using namespace std;
long long curr[30],res[300300]={};

int main()
{//freopen("i.txt","r",stdin);
string s; cin>>s;
int n=s.size();
for (int i=0;i<=30;i++) curr[i]=-1;
for (int i=n-1;i>=0;i--)
{int a=s[i]-'a';
 curr[a]=i;
 vector <int> pos;
 pos.push_back(n);
  for (int j=0;j<30;j++)
   if (curr[j]!=-1) pos.push_back(curr[j]);
 sort(pos.begin(),pos.end());
 for (int j=1;j<pos.size();j++) res[j]+=pos[j]-pos[j-1];
}
int ds=0;
for (int j=0;j<30;j++) if (curr[j]!=-1) ds++;
cout<<ds<<endl;
for (int i=1;i<=ds;i++) cout<<res[i]<<endl;
return 0;
}
