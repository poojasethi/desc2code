#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
char str[500000];
long long ans[30];
int a[30];
int appear[30][300010];
int main(int argc, char const *argv[])
{
	scanf("%s",str);
	int len=strlen(str);
	for (int i=0;i<26;i++){
		appear[i][len]=len;
		for (int j=len-1;j>=0;j--)
			if (str[j]-'a'==i)
				appear[i][j]=j;
			else appear[i][j]=appear[i][j+1];
	}
	for (int i=0;i<len;i++){
		for (int j=0;j<26;j++)
			a[j]=appear[j][i];
		sort(a,a+26);
		for (int j=1;j<26;j++){
			ans[j]=ans[j]+a[j]-a[j-1];
			if (a[j]>=len) break;
		}
		if (a[25]<len) ans[26]=ans[26]+len-a[25];
	}
	int mx=0;
	for (int i=1;i<=26;i++)
		if (ans[i]) mx=i;
	printf("%d\n",mx );
	for (int i=1;i<=mx;i++)
		printf("%I64d\n",ans[i] );
	return 0;
}
