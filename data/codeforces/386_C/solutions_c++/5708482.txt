#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

vector<int> plc[30];
string s;
long long ans[30];

int main()
{
	cin >> s;
	int n = s.size();
	for (int i = 0; i < n; i++)
		plc[s[i] - 'a'].push_back(i);
	int mx = 0;
	for (int i = 0; i < n; i++)
	{
		vector<int> tmp;
		for (int j = 0; j < 26; j++)
			if (lower_bound(plc[j].begin(), plc[j].end(), i) != plc[j].end())
				tmp.push_back(*lower_bound(plc[j].begin(), plc[j].end(), i));
		tmp.push_back(n);
		sort(tmp.begin(), tmp.end());
		mx = max(mx, (int)tmp.size());
		for (int j = 0; j < tmp.size() - 1; j++)
			ans[j + 1] += tmp[j + 1] - tmp[j];
	}
	cout << mx - 1 << endl;
	for (int i = 1; i < mx; i++)
		cout << ans[i] << " ";
	cout << endl;
	return 0;
}
