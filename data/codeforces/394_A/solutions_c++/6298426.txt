#include<stdio.h>
#include<stdlib.h>

using namespace std;

int main(void){
	int a,b,c;
	a = b = c = 0;
	char d;
	scanf("%c",&d);
	while ('+' != d){
		a++;
		scanf("%c",&d);
	}
	scanf("%c",&d);
	while ('=' != d){
		b++;
		scanf("%c",&d);
	}
	scanf("%c",&d);
	while ('\n' != d){
		c++;
		scanf("%c",&d);
	}
	bool sol = true;
	if ((a+b) != c){
		if (((a+b)+1) == (c-1)) {
			b++;
			c--;
		} else { if ((a+b-1) == (c+1)) {
				if (1 == a) b--;
				else a--;
				c++;
			} else sol = false;
		}
	}
	if (sol) {
		for (int i = 0; i < a; i++) printf("|");
		printf("+");
		for (int i = 0; i < b; i++) printf("|");
		printf("=");
		for (int i = 0; i < c; i++) printf("|");
		printf("\n");
	} else printf("Impossible\n");
}
