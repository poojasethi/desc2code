#include<iostream>
using namespace std;
int main()
{
	string s;
	cin>>s;
	int a=0,b=0,c=0,sw=0,i;
	for(i=0;i<s.length();i++)
	{
		if(sw==0)
		{
			if(s[i]=='+')
				sw=1;
			else
				a++;
		}
		else if(sw==1)
		{
			if(s[i]=='=')
				sw=2;
			else b++;
		}
		else
			c++;
	}
	if(a+b-c!=0&&a+b-c!=2&&a+b-c!=-2)
		{cout<<"Impossible";return 0;}
	if(a+b-c==2)
	{
		if(a>1)
			a-=1;
		else b-=1;
		c+=1;
	}
	else if(a+b-c==-2)
		{c-=1;a+=1;}
	for(i=0;i<a;i++)
		cout<<'|';
	cout<<'+';
	for(i=0;i<b;i++)
		cout<<'|';
	cout<<'=';
	for(i=0;i<c;i++)
		cout<<'|';
}
