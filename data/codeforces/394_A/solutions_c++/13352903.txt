#include <iostream>
using namespace std;
string exp;
int a, b, c;
int main()
{
	cin >> exp;
	int i = 0;
	while(exp[i++] != '+') a++;
	while(exp[i++] != '=') b++;
	c = exp.size() - i;

	
	if(a + b == c) cout << exp;
	else if(a + b < c)
	{
		if(a + 1 + b == c - 1)
		{
			a++;
			c--;
			while(a--) cout << '|';
			cout << '+';
			while(b--) cout << '|';
			cout << '=';
			while(c--) cout << '|';
		}
		else cout << "Impossible";
	}
	else
	{
		if(a - 1 + b == c + 1)
		{
			if(a == 1) b--;
			else a--;
			c++;
			while(a--) cout << '|';
			cout << '+';
			while(b--) cout << '|';
			cout << '=';
			while(c--) cout << '|';
		}
		else cout << "Impossible";
	}
	
}
