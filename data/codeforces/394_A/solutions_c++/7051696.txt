#include <iostream>
#include <cstring>
#include <cstdio>

using namespace std;

char s[500]; int sn;

int main()
{
    scanf("%s", s);
    sn = strlen(s);

    int si, a, b, c;
    si = a = b = c = 0;

    for(; s[si]!='+'; si++, a++); si++;
    for(; s[si]!='='; si++, b++); si++;
    for(; si < sn; si++, c++);

    if( c == a+b+2 ) c--, a++;
    if( c == a+b-2 )
    {
        if( a > 1 ) a--, c++;
        else b--, c++;
    }

    if( c != a+b ) puts("Impossible");
    else
    {
        while(a--) printf("|"); printf("+");
        while(b--) printf("|"); printf("=");
        while(c--) printf("|"); puts(" ");
    }
}
