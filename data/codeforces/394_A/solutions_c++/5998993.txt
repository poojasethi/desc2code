#include <iostream>
using namespace std;

string s;
char c[3] = {'+', '=', '\n'};
int a[3] = {}, k = 0;

int main(){
  cin >> s;
  for(int i = 0; i < s.size(); i++){
    if(s[i] == '|') a[k]++;
    else k++;
  }
  if(a[0] + a[1] == a[2] + 2){
    if(a[0] > 1) a[0]--;
    else a[1]--;
    a[2]++;
  } else if(a[0] + a[1] + 2 == a[2]){
    a[0]++;
    a[2]--;
  } else if(a[0] + a[1] != a[2]){
    cout << "Impossible";
    return 0;
  }
  for(int i = 0; i < 3; i++){
    for(int j = 0; j < a[i]; j++) cout << '|';
    cout << c[i];
  }
}
