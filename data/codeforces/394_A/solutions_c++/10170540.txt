#include <iostream>
using namespace std;
int main(){
	string t;
	int plus,equals;
	cin >> t;
	for(int i = 0; i < t.size(); ++i){
		if(t[i]=='+')
			plus=i;
		if(t[i]=='=')
			equals=i;
	}
	int ab = equals-1;
	int sum = t.size()-2;
	int c = sum-ab;
	if(c-ab==0){
		cout << t << '\n';
	}
	else if(c-ab==-2){
		t.push_back('|');
		if(plus>=2)
			t.erase(t.begin());
		else
			t.erase(t.begin()+plus+1);
		cout << t << '\n';
	}
	else if(c-ab==2){
		t.insert(t.begin(),'|');
		t.resize(t.size()-1);
		cout << t << '\n';
	}
	else
		cout << "Impossible\n";
}
