#include<stdio.h>
#include<string.h>
int main()
{
 char a[300];
 gets(a);
 int n=strlen(a),i=0,x=0,y=0;
 for(i=0;i<n;i++)
  if(a[i]=='+')break; else x++;
 for(i=i+1;i<n;i++)
  if(a[i]=='=')break; else y++;
 if(2*(x+y)==n-4)
  {for(i=0;i<=x;i++)
   printf("|");
   printf("+");
   for(i=0;i<y;i++)
   printf("|");
   printf("=");
   for(i=0;i<n-3-x-y;i++)
   printf("|");
   }
  else if(2*(x+y)==n)
  {
   if(x>=y)x--; else y--;
   for(i=0;i<x;i++)
   printf("|");
   printf("+");
   for(i=0;i<y;i++)
   printf("|");
   printf("=");
   for(i=0;i<n-x-y-2;i++)
   printf("|");
   }  
 else if(2*(x+y)==n-2)
 puts(a);
 else 
 printf("Impossible");
 return 0;
}
