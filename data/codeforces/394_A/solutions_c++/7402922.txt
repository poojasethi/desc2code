#include <iostream>
using namespace std;
int main()
{
	string s;
	int a, b, c, i;
	cin>>s;
	for(i=0,a=0; s[i]!='+'; i++)a++;
	for(i++,b=0; s[i]!='='; i++)b++;
	for(i++,c=0; i<s.size(); i++)c++;
	if(a+b-c==2){
		if(a>1)a--; 
		else if (b>1) b--;
		else {
			cout<<"Impossible"<<endl; return 0;
		}
		c++;
	}
	else if (c-a-b==2) {
		a++; c--;
		if (c<1) {
			cout<<"Impossible"<<endl; return 0;
		}
	}
	else if (c-a-b==0) {
		cout<<s<<endl; return 0;
	}
	else {
		cout<<"Impossible"<<endl; return 0;
	}
	for (i=0; i<a; i++) cout<<"|"; cout<<"+";
	for (i=0; i<b; i++) cout<<"|"; cout<<"=";
	for (i=0; i<c; i++) cout<<"|"; cout<<endl;
	return 0;
}
