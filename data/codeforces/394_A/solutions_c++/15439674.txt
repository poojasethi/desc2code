#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    char s[305];
    cin>>s;
    int i,A=0,B=0,C=0;
    for(i=0;s[i]!='+';i++)
        A++;
    for(i++;s[i]!='=';i++)
        B++;
    for(i++;i<strlen(s);i++)
        C++;
    if(A+B==C+2)
    {
        if(A>1)A--;
        else B--;
        C++;
    }
    if(A+B==C-2)
    {
        C--;
        A++;
    }
    if(A+B==C){
    for(int j=0;j<A;j++)
        cout<<'|';
    cout<<'+';
    for(int j=0;j<B;j++)
        cout<<'|';
    cout<<'=';
    for(int j=0;j<C;j++)
        cout<<'|';}
    else cout<<"Impossible";
}
