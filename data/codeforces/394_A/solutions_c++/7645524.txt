#include <iostream>
using namespace std;

void print(char c, int n) {
	for (int i = 0; i < n; i++)
		cout << c;
}

int main() {
	int ar[3] = {}, i = 0;
	string in;
	cin >> in;
	for (int j = 0; j < in.length(); j++) {
		if (in[j] == '|')
			ar[i]++;
		if (in[j] == '+' || in[j] == '=')
			i++;
	}
	for (int a = -1; a <= 1; a++)
		for (int b = -1; b <= 1; b++)
			for (int c = -1; c <= 1; c++)
				if ((a*a + b*b + c*c == 2 || a*a + b*b + c*c == 0) && a + b + c == 0 && ar[0]+a + ar[1]+b == ar[2]+c && ar[0]+a != 0 && ar[1]+b != 0) {
					print('|', ar[0]+a);
					cout << '+';
					print('|', ar[1]+b);
					cout << '=';
					print('|', ar[2]+c);
					return 0;
				}
	cout << "Impossible";
}
