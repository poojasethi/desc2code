#include<iostream>
#include<cmath>
#define FOR(i,x,n)  for(int i=x;i<n;i++)
#define srt insert
#define ers erase
#define SZ size()
using namespace std;
int main()
{
  string s; cin>>s; int p,e,x;
  FOR(i,0,s.SZ) 
  { if(s[i]=='+') p=i;
    if(s[i]=='=') e=i;
  }
  int sz1=p,sz2=e-1-p,sz3=s.SZ-sz1-sz2-2; 
  x=abs(sz3-sz1-sz2);
  if(x!=0&&x!=2) { cout<<"Impossible"; return 0;} 
  if(sz3>sz1+sz2)
  { FOR(i,0,1) s.srt(0,"|");
  	s.ers(s.SZ-1,1);
  }
  else 
  if(sz3<sz1+sz2)
  { if(sz1!=1) s.ers(0,1);
    else s.ers(p+1,1);
    FOR(i,0,1) s+="|"; 
  }
  cout<<s;
}
