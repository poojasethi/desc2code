#include <iostream>
#include <string>
using namespace std;
int main(){
	int a=0,b=0,c=0;
	string str;
	cin>>str;
	int i=0;
	while(str[i]!='+'){++a;++i;}
	++i;
	while(str[i]!='='){++b;++i;}
	c=str.length()-i-1;
	if(a+b==c){}
	else if(a+b==c-2) {++a;--c;}
	else if(a+b==c+2&&!(a==1&&b==1)) {
		if(a>1)--a;
		else if(b>1) --b;
		++c;
	}
	else {
		cout<<"Impossible";
		return 0;
	}
	for(int i=0;i<a;++i) cout<<'|';
	cout<<'+';
	for(int i=0;i<b;++i) cout<<'|';
	cout<<'=';
	for(int i=0;i<c;++i) cout<<'|';
}
