#include<stdio.h>
int a=0,b=0,c=0;
void print(int x,int y,int z)
{
	int i;
	for(i=0;i<x;i++)
		printf("|");
	printf("+");
	for(i=0;i<y;i++)
		printf("|");
	printf("=");
	for(i=0;i<z;i++)
		printf("|");
	printf("\n");
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	char ci;
	ci=getchar();
	while(ci!='+') a++,ci=getchar();
	ci=getchar();
	while(ci!='=') b++,ci=getchar();
	ci=getchar();
	while(ci!='\n') c++,ci=getchar();
	if(c==a+b) 
		print(a,b,c);
	else if(c==a+b+2)
		print(a,b+1,c-1);
	else if(c==a+b-2)
	{
		if(b>1) print(a,b-1,c+1);
		else print(a-1,b,c+1);
	}
	else 
		printf("Impossible\n");
	return 0;
}
