#include<cstdio>
#include<iostream>
#include<algorithm>
#include<string.h>
#include<math.h>
using namespace std;
char s[320];
int main()
{
    memset(s,'0',sizeof(s));
    cin>>s;
    int len=strlen(s);
    int i=0;
    while(s[i++]!='+');
    int a=i-1;
    while(s[i++]!='=');
    int b=i-a-2;
    int c=len-i;
    if(abs(a+b-c)!=2&&a+b!=c){cout<<"Impossible"<<endl;return 0;}
    else
    {
        if(a+b>c)
        {
            if(a>1){a-=1;c+=1;}
            else{b-=1;c+=1;}
        }
        else if(a+b<c){a+=1;c-=1;}
        for(int i=1;i<=a;i++)cout<<"|";
            cout<<"+";
            for(int i=1;i<=b;i++)cout<<"|";
            cout<<"=";
            for(int i=1;i<=c;i++)cout<<"|";
    }
}
