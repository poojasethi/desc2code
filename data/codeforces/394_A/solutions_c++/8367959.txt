#include <stdio.h>

int A[3];

char zn[333];
char sep[3]={'+','=','\n'};
int main(){
	gets(zn);
	
	int ind=0;
	for(char*p = zn;*p;++p){
		if(*p=='|')
			++A[ind];
		else if(*p==sep[ind])
			++ind;
	}

	if(A[0]+A[1]!=A[2]){

		if(A[0]+A[1]-1 == A[2]+1){
			++A[2];
			A[A[0]<=1]--;
		}
		else if(A[0]+A[1]+1 == A[2]-1){
			--A[2];
			A[0]++;
		}
		else{
			printf("Impossible\n");
			return 0;
		}
			

	}

	for(int i=0;i<3;++i){
		for(int j=0;j<A[i];++j)
			putchar('|');
		putchar(sep[i]);
	}

}
