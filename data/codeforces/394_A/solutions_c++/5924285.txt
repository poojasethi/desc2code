#include<bits/stdc++.h>
using namespace std;

int main()
{
    string s; int siz, l=0, r=0, a=0; bool flag=true; cin>>s; siz=s.size();
    int i=0; for(;i<siz;i++) { if(s[i]=='+') { flag=false; continue; } else if(s[i]=='=') { break; } l++; if(flag) a++; }
    i++; for(;i<siz;i++) r++;

    if(r==l) cout<<s;
    else if(abs(r-l)!=2) cout<<"Impossible";
    else if(l<r) cout<<'|'<<s.substr(0,siz-1);
    else if(a>1) cout<<s.substr(1)<<'|';
    else cout<<s.substr(0,2)<<s.substr(3)<<'|';
    return 0;
}
