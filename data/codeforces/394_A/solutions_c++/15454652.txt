#include<stdio.h>
int main()
{
	int a[3];
	char c;
	int i;

	a[0]=a[1]=a[2]=i=0;
	while((c=getchar())!=EOF)
	{
		if(c=='|')a[i]++;
		else if(c=='+'||c=='=')i++;
	}
	if(a[0]+a[1]-1==a[2]+1)
	{
		a[0]--;
		a[2]++;
	}
	else if(a[0]+a[1]+1==a[2]-1)
	{
		a[0]++;
		a[2]--;
	}
	else if(a[0]+a[1]==a[2]);
	else 
	{
		printf("Impossible\n");
		return 0;
	}
	if(!a[0])
	{
		a[0]++;
		a[1]--;
	}
	for(i=1;i<=a[0];i++)putchar('|');
	putchar('+');
	for(i=1;i<=a[1];i++)putchar('|');
	putchar('=');
	for(i=1;i<=a[2];i++)putchar('|');
	putchar('\n');
	return 0;
}
