#include <iostream>
#include <stdlib.h>
#include <stdio.h>
using namespace std;
int i=0;
int a,b,c;
int main () 
{
	char str[1024];
	cin >> str;
	
	a=b=c=0;
	int *inc;
	inc=&a;
	while(str[i]!='\0') {
		if(str[i]=='|') {
			*inc+=1; 
		} else if(str[i]=='+') 
			inc=&b;
		else if(str[i]=='=') 
			inc=&c;
		i++;
	}
	if(a+b == c) 
		cout << str;
	else if(abs(c-a-b)!=2)
		cout << "Impossible";
	else {
		if(c > a+b) {
			c--; a++;
		} else {
			c++; (a>1?a:b)--;
		}
		for(i=0; i<a; i++) cout << '|';
		cout << '+';
		for(i=0; i<b; i++) cout << '|';
		cout << '=';
		for(i=0; i<c; i++) cout << '|';
	}
}
