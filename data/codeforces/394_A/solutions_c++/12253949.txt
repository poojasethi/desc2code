#include<iostream>
#include<string>
using namespace std;
int main() { int a,b,c,counter=0;
string x;
bool d=true;
cin>>x;
for(int i=0;i<x.size();i++)
{   if(x[i]=='|')
        counter++;
    else if(x[i]=='+')
    { a=counter; counter=0;    }
    else if(x[i]=='=')
    { b=counter; counter=0;  }
    if((i+1)==x.size())
        c=counter;
}
if((a+b+1)==(c-1))
{ a++; c--; }
else if((a+b-1)==(c+1))
{ if(a==1)
b--;
else a--; c++; } 
else if((a+b)!=c)
    d=false;
if(d)
{ for(int i=1;i<=a;i++)
    cout<<"|";
cout<<"+";
for(int i=1;i<=b;i++)
    cout<<"|";
cout<<"=";
for(int i=1;i<=c;i++)
    cout<<"|"; }
else 
    cout<<"Impossible";
cout<<endl;
return 0; }
