#include<stdio.h>

int s[3];

int main() {
  char c;
  int a=0;
  s[0]=0; s[1]=0; s[2]=0;
  while ((c=getchar()) != EOF) {
    if (c=='|') { s[a]++; }
    else if (c=='+'||c=='=') { a++; }
    else {break;}
  }
  if (s[0]+s[1]-2==s[2] && (s[0]>1 || s[1]>1)) {
    if (s[0]>1) { s[0]--; s[2]++; }
    else        { s[1]--; s[2]++; }
  } else  if (s[0]+s[1]+2==s[2]) {
    s[0]++; s[2]--;
  } else if (s[0]+s[1]==s[2]) {
  } else {
    printf("Impossible");
    return 0;
  }
  for (int i=0; i<s[0]; i++) { putchar('|'); }
  putchar('+');
  for (int i=0; i<s[1]; i++) { putchar('|'); }
  putchar('=');
  for (int i=0; i<s[2]; i++) { putchar('|'); }



  return 0;
}
