#include<iostream>
#include<cstdio>
using namespace std;

int A, B, C;char ch;

void opt(int a,int b,int c)
{
	for(int i = 1; i <= a; ++i) putchar('|');
	putchar('+');
	for(int i = 1; i <= b; ++i) putchar('|');
	putchar('=');
	for(int i = 1; i <= c; ++i) putchar('|');
	putchar('\n');
}

int main()
{
	ch = 0;
	while(ch != EOF && ch != '+')  A += (ch = getchar()) == '|';
	ch = 0;
	while(ch != EOF && ch != '=')  B += (ch = getchar()) == '|';
	ch = 0;
	while(ch != EOF)  C += (ch = getchar()) == '|';
	
	if(A + B == C) opt(A, B, C);
	else if(A - 1 + B == C + 1)
	{
		if(A - 1) opt(A - 1, B, C + 1);
		else opt(A, B - 1, C + 1);
	}
	else if(A + 1 + B == C - 1) opt(A + 1, B, C - 1);
	else printf("Impossible\n");
	return 0;
}
