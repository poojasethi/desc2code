main()
{
    int c1=0,c2=0,c3=0,i; char s;

    while( (s=getchar()) != '+' )   c1++;
    while( (s=getchar()) != '=' )   c2++;
    while( (s=getchar()) != '\n' )  c3++;

    if(c1+c2==c3){}

	else if(c1+c2==c3-2)
    {
        ++c1;--c3;
    }

	else if(c1+c2==c3+2&&!(c1==1&&c2==1))
    {
		if(c1>1)--c1;
		else if(c2>1) --c2;
		++c3;
	}
    else
    {
        cout<<"Impossible"; return 0;
    }

    for(i=1;i<=c1;i++) cout<<"|";  cout<<"+";
    for(i=1;i<=c2;i++) cout<<"|";  cout<<"=";

}
