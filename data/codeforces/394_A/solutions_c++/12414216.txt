#include <bits/stdc++.h>
using namespace std;
//a+b=c
void myprint(int a, int b, int c){
	for(int i=0;i<a;i++){
		cout<<"|";
	}
	cout<<"+";
	for(int i=0;i<b;i++){
		cout<<"|";
	}
	cout<<"=";
	for(int i=0;i<c;i++){
		cout<<"|";
	}
	cout<<endl;
}
int main(){
	int a=0,b=0,c=0;
	int check=0;
	string t;
	cin>>t;
	for(int i=0;i<t.size();i++){
		if(check==2){
			c++;
		}
		else if(check==1){
			b++;
			if(t[i]=='='){
				check++;
				b--;
			}
		}
		else {
			a++;
			if(t[i]=='+'){
				check++;
				a--;
			}

		}
	}
	if(a+b==c) myprint(a,b,c);
	else if(a+b+1==c-1) myprint(a+1,b,c-1);
	else if(a+b-1==c+1 && a>b) myprint(a-1,b,c+1);
	else if(a+b-1==c+1) myprint(a,b-1,c+1);
	else cout<<"Impossible"<<endl;
}
