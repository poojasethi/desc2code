#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,A,B,C;
char p[310];
inline void work(int A,int B,int C) {
	int i;
	if (!A||!B||!C) return;
	if (A+B==C) {
		For(i,1,A) printf("|");
		printf("+");
		For(i,1,B) printf("|");
		printf("=");
		For(i,1,C) printf("|");
		printf("\n");
		exit(0);
	}
}
int main() {
	scanf("%s",p+1);
	n=strlen(p+1);
	For(i,1,n) if (p[i]=='+') break;
	A=i-1;
	For(i,1,n) if (p[i]=='=') break;
	B=i-A-2;
	C=n-i;
	work(A,B,C);
	For(i,-1,1)For(j,-1,1)For(k,-1,1) if (abs(i)+abs(j)+abs(k)==2&&i+j+k==0)
		work(A+i,B+j,C+k);
	printf("Impossible\n");
	return 0;
}
