#include"stdio.h"
int i, j, a, b, c;
char s[500];
int main() {
	scanf("%s", s);
	for (i = 0; s[i]; i++)
		if (s[i] == '+') a = i;
		else if (s[i] == '=') b = i - a - 1;
	c = i - a - b - 2;
	if (a + b == c) printf("%s", s);
	else if (a + b + 2 == c) {
		a++; c--;
		for (i = 0; i < a; i++) printf("|");
		printf("+");
		for (i = 0; i < b; i++) printf("|");
		printf("=");
		for (i = 0; i < c; i++) printf("|");
	}
	else if (a + b == c + 2) {
		if (a > 1) a--;
		else b--;
		c++;
		for (i = 0; i < a; i++) printf("|");
		printf("+");
		for (i = 0; i < b; i++) printf("|");
		printf("=");
		for (i = 0; i < c; i++) printf("|");
	}
	else printf("Impossible");
}
