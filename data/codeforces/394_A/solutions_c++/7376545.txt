#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	int a=0,b=0,c=0;
	while(getchar()=='|')a++;
	while(getchar()=='|')b++;
	while(getchar()=='|')c++;
	if(a+b==c)
	{
		while(a--)cout<<"|";
		cout<<"+";
		while(b--)cout<<"|";
		cout<<"=";
		while(c--)cout<<"|";
	}
	else if(a+b==c+2&&(a!=1||b!=1))
	{
		if(a>1)a--;
		else b--;
		c++;
		while(a--)cout<<"|";
		cout<<"+";
		while(b--)cout<<"|";
		cout<<"=";
		while(c--)cout<<"|";
	}
	else if(a+b+2==c)
	{
		c--;
		a++;
		while(a--)cout<<"|";
		cout<<"+";
		while(b--)cout<<"|";
		cout<<"=";
		while(c--)cout<<"|";
	}
	else
	{
		cout<<"Impossible";
	}
}
