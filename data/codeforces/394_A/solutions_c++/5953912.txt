#include <iostream>
#include <string>
#include <cstdlib>

using namespace std;

void escribe(int a)
{
  for (int i=0;i<a;i++)
    cout<<"|";
}

int main()
{
  string s;
  cin>>s;
  int a=0;
  int is=0;
  while (s[is]!='+') {
    is++;
    a++;
  }
  is++;
  int b=0;
  while (s[is]!='=') {
    is++;
    b++;
  }
  is++;
  int c=0;
  while (is<int(s.size())) {
    is++;
    c++;
  }
  if (a+b==c) {
  } else if (a+b==c+2) {
    if (a>1) a--;
    else b--;
    c++;
  } else if (c==a+b+2) {
    c--;
    a++;
  } else {
    cout<<"Impossible"<<endl;
    exit(0);
  }
  escribe(a);
  cout<<"+";
  escribe(b);
  cout<<"=";
  escribe(c);
  cout<<endl;
}
