#include<iostream>
using namespace std;
char s[303];
void print(int k){
    for(int j=0;j<k;j++)cout<<'|';
}
int main(){
    int i,a,b,c,d;
    a=b=c=0;
    cin>>s;
    for(i=0;s[i]=='|';i++)a++;
    for(++i;s[i]=='|';i++)b++;
    for(++i;s[i]=='|';i++)c++;
    d=a+b-c;
    if(d==0)cout<<s;
    else if(d==-2||d==2){
        c+=d/2;
        if(a-d/2>0)a-=d/2;
        else  b-=d/2;
        print(a);cout<<'+';print(b);cout<<'=';print(c);
    }
    else cout<<"Impossible";
}
