#include <iostream>
using namespace std;
void print(int x,int y,int z)
{
    int ii;
    for(ii=0;ii<x;ii++)cout<<"|";
    cout<<"+";
    for(ii=0;ii<y;ii++)cout<<"|";
    cout<<"=";
    for(ii=0;ii<z;ii++)cout<<"|";
}

int main()
{
    string s;
    int i,j,k,p1=0,p2=0,p3=0;
    cin>>s;
    for(i=0;s[i]!='+';i++)p1++;
    for(j=i+1;s[j]!='=';j++)p2++;
    p3=s.size()-p2-p1-2;
    if(p1+p2==p3)cout<<s;
    else if(p1+p2+1==p3-1){print(p1+1,p2,p3-1);}
    else if(p1+p2-1==p3+1){if(p1>p2)print(p1-1,p2,p3+1); else print(p1,p2-1,p3+1);}
    else cout<<"Impossible";
    return 0;
}
