#include <bits/stdc++.h>

int main()
{
	int a,b,s;
	char c;
	c=1;
	a=b=s=0;
	while(c!='+')
	{
		scanf("%c",&c);
		if(c=='|')a++;
	}
	while(c!='=')
	{
		scanf("%c",&c);
		if(c=='|')b++;
	}
	while(scanf("%c",&c)==1)
	{
		if(c=='|')s++;
	}
	if((a+b)==s+2)
	{
		if(a>b)a--;
		else b--;
		s++;
	}
	else if((a+b+2)==s)
	{
		a++;
		s--;
	}
	if((a+b)==s)
	{
		for(int i=0;i<a;i++)
		{
			printf("|");
		}
		printf("+");
		for(int i=0;i<b;i++)
		{
			printf("|");
		}
		
		printf("=");
		
		for(int i=0;i<s;i++)
		{
			printf("|");
		}
		printf("\n");
	}
	else printf("Impossible\n");
	return 0;
}
