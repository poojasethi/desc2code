#include<iostream>
using namespace std;
const int N=1000;
char s[N];
void AAA(int a,int b,int c)
{
    while(a--) cout<<"|";
    cout<<"+";
    while(b--) cout<<"|";
    cout<<"=";
    while(c--) cout<<"|";
    cout<<endl;
}
int main()
{
    while(cin>>s)
    {
        int a=0,b=0,c=0;
        int i=0;
        while(s[i++]!='+') a++;
        while(s[i++]!='=') b++;
        while(s[i++]!='\0') c++;
        if (a+b==c) cout<<s<<endl;
        else if (a+b-2==c) a>b?AAA(a-1,b,c+1):AAA(a,b-1,c+1);
        else if (a+b==c-2) AAA(a+1,b,c-1);
        else cout<<"Impossible"<<endl;
    }
}
