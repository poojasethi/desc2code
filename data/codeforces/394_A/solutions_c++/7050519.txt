
#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <set>
#include <stack>

using namespace std;

int main()
{
	string s;
	int a=0,b=0,c=0;
	
	cin>>s;
	int i=0;
	while (s[i++]=='|') a++;
	while (s[i++]=='|') b++;
	while (i<s.length() && s[i++]=='|') c++;
	
	if ( (a+b)-c == 2 )
		(a>1 ? a : b)-- , c++;
	else if ( (a+b)-c == -2 )
		c-- , a++;
	else if ( a+b != c )
	{
		cout<<"Impossible";
		return 0;
	}
		
	while (a--) cout<<'|';	cout<<'+';
	while (b--) cout<<'|';	cout<<'=';
	while (c--) cout<<'|';
	
	return 0;
}
