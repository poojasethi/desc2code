#include <cstdio>
#include <cstring>
int A, B, C;
inline void Print(int a, int b, int c){
	for(int i=1; i<=a; i++) printf("|");
	printf("+");
	for(int i=1; i<=b; i++) printf("|");
	printf("=");
	for(int i=1; i<=c; i++) printf("|");
	puts("");
}
int main(){
	char ch=getchar();
	while(ch=='|') ch=getchar(), A++;
	ch=getchar();
	while(ch=='|') ch=getchar(), B++;
	ch=getchar();
	while(ch=='|') ch=getchar(), C++;
	if(A+B==C) Print(A, B, C);
	else if(A+B==C+2) if(A==1) Print(A, B-1, C+1); else Print(A-1, B, C+1);
	else if(A+B==C-2) Print(A+1, B, C-1);
	else puts("Impossible");
}
