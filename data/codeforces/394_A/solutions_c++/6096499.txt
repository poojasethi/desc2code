#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <cstdio>
#include <iomanip>
using namespace std;
int main ()
{
	char s[1000005];
	int a=0,b=0,c=0,i,j,k;
	cin>>s;
	for(i=0;s[i]!='+';i++)
	{
		a++;
	}
	for(j=i+1;s[j]!='=';j++)
	{
		b++;
	}
	for(k=j+1;s[k]!='\0';k++)
	{
		c++;
	}
	if(a+b==c)
	{
		cout<<s<<endl;
	}
	else
	{
		if(a+b-1==c+1)
		{
			if(a==1)
			{
				b--;
			}
			else
			{
				a--;
			}
			c++;
		}
		if(a+b+1==c-1)
		{
			a++;
			c--;
		}
		if(a+b==c)
		{
			while(a--)
			cout<<"|";
			cout<<"+";
			while(b--)
			cout<<"|";
			cout<<"=";
			while(c--)
			cout<<"|";
			cout<<endl;
		}
		else
		{
			cout<<"Impossible"<<endl;
		}
	}
	return 0;
}
