#include <iostream>
using namespace std;

int main()
{
	string a;
	cin>> a;

	int x = 0, y = 0;
	bool chck1 = false;

	for(int i = 0; i < a.size(); i ++)
	{
		if(a[i] == '=')
			chck1 = true;
		else if(!chck1)
		{	
			if(a[i] != '+')
				x ++;
		}
		else
		{
			y ++;
		}
	}	

	if(x == y)
	{
		cout<< a << endl;
		return 0;
	}

	if(x + 1 == y - 1)
	{
		string b;
		b = '|';
		for(int i = 0; i < a.size() - 1; i ++)
			b += a[i];
		cout << b << endl;
		return 0;
	}

	if(x - 1 == y + 1)
	{
		string b;
		if(a[1] == '+')
		{
			b += "|+";
			for(int i = 3; i < a.size(); i ++)	
				b += a[i];
		}
		else
			for(int i = 1; i < a.size(); i ++)
				b += a[i];
		b += '|';
		cout << b << endl;
		return 0;
	}

	cout<< "Impossible\n";

}
