#include<cstdio>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
#include<queue>
#include<vector>
using namespace std;
void f(int a,int b,int c)
{
    int i;
    for(i=1;i<=a;i++)
        cout<<'|';
    cout<<'+';

    for(i=1;i<=b;i++)
        cout<<'|';
    cout<<'=';

    for(i=1;i<=c;i++)
        cout<<'|';
}

int main(){
    int a,b,c;
    a=b=c=0;
    char s;
    while(cin>>s&&s!='+')a++;
    while(cin>>s&&s!='=')b++;
    while(scanf("%c",&s)!=EOF&&s=='|')c++;
    if(a+b==c)f(a,b,c);
    else if(a+1+b==c-1)
        f(a,b+1,c-1);
    else if(a+b-1==c+1)
        if(b>1)f(a,b-1,c+1);
        else f(a-1,b,c+1);
    else printf("Impossible");
}
