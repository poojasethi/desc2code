#include<stdio.h>
#include<string.h>
int main()
{
	int i,a,b,c,len,flag;
	char s[10000];
	while(gets(s))
	{
		len=strlen(s);
		for(i=0;i<len;i++)
		{
			if(s[i]=='+')
				a=i;
			else if(s[i]=='=')
				b=i-a-1;
		}
		c=len-a-b-2;
		if(a+b==c)
			flag=1;
		else if(a-1+b==c+1&&a>1)
		{
			a--;
			c++;
			flag=1;
		}
		else if(a+1+b==c-1&&c>1)
		{
			a++;
			c--;
			flag=1;
		}
		else if(a+b-1==c+1&&b>1)
		{
			b--;
			c++;
			flag=1;
		}
		else
			flag=0;
		if(flag==1)
		{
			for(i=0;i<a;i++)
				printf("|");
			printf("+");
			for(i=0;i<b;i++)
				printf("|");
			printf("=");
			for(i=0;i<c;i++)
				printf("|");
			printf("\n");
		}
		else
			printf("Impossible\n");
	}
	return 0;
}
