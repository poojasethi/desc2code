#include <cstdio>
#include <algorithm>
#include <cstring>
char a[222],b[111],c[1100];
int main(){
	scanf("%[|]+%[|]=%[|]",a,b,c);
	int la=strlen(a),lb=strlen(b),lc=strlen(c);
	if (la+lb==lc){
		printf("%s+%s=%s\n",a,b,c);
		}else
	if (lc-la-lb==2){
		printf("|%s+%s=%s\n",a,b,c+1);
		}else
	if (la+lb-lc==2){
		if (la>=lb) printf("%s+%s=%s|",a+1,b,c);
			else	printf("%s+%s=%s|",a,b+1,c);
		}else
		printf("Impossible\n");
}
