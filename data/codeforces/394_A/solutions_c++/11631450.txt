#include<iostream>
#include<string>
using namespace std;
	
int main()
{
	string s;
	cin>>s;
	int l=s.size(),a=0,b=0,c=0;
	bool k=true,t=false;
	for (int i=0;i<l;i++)
	{
		if (s[i]=='=')
		{
			break;
		}
		a++;
	}
	for (int i=0;i<l;i++)
	{
		if (s[i]=='+')
		{
			break;
		}
		c++;
	}
	a--;
	b=l-a-2;
	if (a==b)
	{
		cout<<s<<endl;
	}
	else if (b-a==2)
	{
		s.erase(l-1,1);
		cout<<'|'<<s<<endl;
	}
	else if (b-a==-2)
	{
		if (c==1)
		{
			s.erase(2,1);
		}
		else
		{
		s.erase(0,1);
		}
		cout<<s<<'|'<<endl;
	}
	else
	{
		cout<<"Impossible"<<endl;
	}
	return 0;
}
