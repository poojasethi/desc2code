#include<iostream>
#include<algorithm>
#include<math.h>
#include<queue>
#include<stack>
#include<string>
using namespace std;

int n;

int main()
{
      string str;
      cin>>str;
      int left,right,sum;
      left=str.find('+');
      right=str.find('=')-left-1;
      sum=str.size()-str.find('=')-1;
      if(left+right==sum)
      cout<<str<<endl;
      else if(left+right-sum==2)
      {
            if(left>1)cout<<str.substr(1)<<"|"<<endl;
            else
            cout<<str.substr(0,2)<<str.substr(3)<<"|"<<endl;
      }
      else if(left+right-sum==-2)
      {
            cout<<"|"<<str.substr(0,str.size()-1)<<endl;
      }
      else
      cout<<"Impossible"<<endl;

      return 0;
}
