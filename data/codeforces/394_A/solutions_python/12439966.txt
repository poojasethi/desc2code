a = raw_input()
f = len(a.split("+")[0])
s = len(a.split("+")[1].split("=")[0])
e = len(a.split("+")[1].split("=")[1])

if f + s == e:
	print "|" * f + "+" + "|" * s + "=" + "|" * e

elif f + s + 1 == e - 1: 
	print "|" * f + "+" + "|" * (s + 1) + "=" + "|" * (e - 1)

elif f + s - 1 == e + 1:
	if s - 1 == 0:
		print "|" * (f - 1) + "+" + "|" * s + "=" + "|" * (e + 1)
	else:
		print "|" * f + "+" + "|" * (s - 1) + "=" + "|" * (e + 1)

else: print "Impossible"
