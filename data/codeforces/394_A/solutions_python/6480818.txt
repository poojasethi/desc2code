s = str(raw_input())
a = s.index('+')
j = s.index('=')
b = j-a-1
c = len(s)-j-1
if a+b == c:
    print s
else:
    if a+b-c == 2:
        if a>1: a -= 1
        else: b -= 1
        c += 1
        s2 = a*'|' + '+' + b*'|' + '=' + c*'|'
        print s2
    else:
        if c-a-b == 2:
            c -= 1
            a += 1
            s2 = a*'|' + '+' + b*'|' + '=' + c*'|'
            print s2
        else: print "Impossible"
