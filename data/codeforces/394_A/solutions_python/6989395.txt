x,y,z = [len( i ) for i in raw_input().replace('+',' ').replace('=',' ').split()]
if x + y == z:
    print '|'*x + '+' + '|'*y + '=' + '|'*z
elif x + y != z - 2 and x + y != z + 2:
    print 'Impossible'
else:
    if x + y == z - 2:
        z -= 1
        x += 1
    else: 
        z += 1
        if x != 1:
            x -= 1
        else:
            y -= 1
    print '|'*x + '+' + '|'*y + '=' + '|'*z
