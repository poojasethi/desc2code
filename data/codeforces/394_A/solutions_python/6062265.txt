s=raw_input()
a,q = s.split('+')
b,c = q.split('=')
a,b,c = map(len, [a,b,c])

if a+b not in [c-2, c, c+2]:
  res = "Impossible"
else:
  if a+b == c-2:
    a += 1
    c -= 1
  if a+b == c+2:
    if a>1:
      a -= 1
    else:
      b -= 1
    c += 1
  res = '%s+%s=%s' % ('|'*a,'|'*b,'|'*c)
print res
