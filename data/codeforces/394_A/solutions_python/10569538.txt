s=raw_input()
lc=0
rc=0
ec=0
i=0
while s[i]!='+':
    lc+=1
    i+=1
i+=1
while s[i]!='=':
    rc+=1
    i+=1
i+=1
while i<len(s):
    ec+=1
    i+=1
if lc+rc==ec:
    print s
elif lc-1+rc==ec+1 and lc>1:
    s="|"*(lc-1)+"+"+"|"*rc+"="+"|"*(ec+1)
    print s
elif lc+rc-1==ec+1 and rc>1:
    s="|"*(lc)+"+"+"|"*(rc-1)+"="+"|"*(ec+1)
    print s
elif lc+rc+1==ec-1 and ec>1:
    s="|"*(lc+1)+"+"+"|"*rc+"="+"|"*(ec-1)
    print s
else:
    print "Impossible"
