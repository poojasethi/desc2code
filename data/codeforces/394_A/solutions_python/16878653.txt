#!/usr/bin/python
#from __future__ import division
from copy import deepcopy
s = raw_input()
s_orig=deepcopy(s)
s=list(s)
group1=0
group2=0
group3=0
for i in xrange(len(s)):
    if s[i]!='+':
        group1+=1
    else:
        last_i=i
        break

for i in xrange(last_i+1,len(s)):
    if s[i]!='=':
        group2+=1
    else:
        last_i=i
        break

for i in xrange(last_i+1,len(s)):
        group3+=1

found=False
if group1+group2==group3:
    print s_orig
    found=True
elif group1+group2<group3:
    if group1+group2+1==group3-1:
        if group3 != 1:
            print "|"*group1 + '+' + "|"*(group2+1) + '=' + "|"*(group3-1)
            found=True
elif group1+group2>group3:
    if group1+group2-1==group3+1:
        if group2!=1:
            print "|"*group1+'+'+"|"*(group2-1)+'='+"|"*(group3+1)
            found=True
        elif group1!=1:
            print "|"*(group1-1)+'+'+"|"*group2+'='+"|"*(group3+1) 
            found=True

if not found:
    print "Impossible"
