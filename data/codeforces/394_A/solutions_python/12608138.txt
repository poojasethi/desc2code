A, B= raw_input().split('+')
B,C= B.split('=')
a= A.count('|')
b= B.count('|')
c= C.count('|')
if a + b ==c:
    print A + '+' + B + '=' + C
elif a+b==c-2:
    print A + '|+' + B + '=' + C[1:]
elif a+b==c+2:
    if a>1:
        print A[1:] + '+' + B + '=' + C +'|'
    elif b>1:
        print A + '+' + B[1:] + '=' + C + '|'
else:
    print "Impossible"
