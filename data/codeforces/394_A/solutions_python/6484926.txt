from itertools import permutations
def main():
    s = raw_input().strip()
    t, r = s.split('=')
    l, m = t.split('+')
    a = map(len, [l, m, r])
    if a[0] + a[1] == a[2]:
        print s
        return
    for x in permutations([-1, 0, 1]):
        b = [a[i] + x[i] for i in range(3)]
        if all(x > 0 for x in b) and b[0] + b[1] == b[2]:
            print '|' * b[0] + '+' + '|' * b[1] + '=' + '|' * b[2]
            return
    print "Impossible"
main()
