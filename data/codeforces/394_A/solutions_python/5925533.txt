s = raw_input()
left,right = s.split('=')
c = len(right)
first,second = left.split('+')
a = len(first)
b = len(second)
dif = a+b-c
if dif == 0:
	print s
elif abs(dif)==2:
	if dif>0:
		if a>1:
			right+='|'
			first = first[0:a-1]
		else:
			right+='|'
			second = second[0:b-1]
	else:
		first+='|'
		right=right[0:c-1]
	print first+'+'+second+'='+right
else:
	print 'Impossible'
