n,k=map(str,raw_input().split('='))
l=n.count('|')
r=k.count('|')
av=(l+r)/2
if l==r:
    print n+'='+k
elif (l+r)%2 or abs(l-r)>2 or l+r<4:
    print 'Impossible'
elif l<r:
    print '|'+n+'='+'|'*av
else:
    a,b=n.split('+')
    if len(a)>len(b):
        print n[1:]+'='+'|'*av
    else:
        print n[:-1]+'='+'|'*av
