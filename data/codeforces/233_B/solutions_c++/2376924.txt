# include <iostream>
# include <math.h>
# include <algorithm>

using namespace std;

long long a,b,c,sum;

int main()
{
	cin>>a;
	
	b=(long long)sqrt(a);
	
	for(long long i=b-100;i<=b;i++)
	{
		if(i<1)
		continue;
		
		c=i;
		sum=0;
		
		while(c>0)
		{
			sum=sum+c%10;
			c/=10;
		}
		
		if((sum+i)*i==a)
		{cout<<i;return 0;}
	}
	
	c=-1;
	
	cout<<c;
}
