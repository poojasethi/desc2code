#include <iostream>
#include <cmath>
#include <ctime>

using namespace std;

long long n;

int sum(long long n)
{
	int s=0;
	while(n>0)
		{s+=n%10;n/=10;}
	return s;
}

int main()
{
	cin>>n;

	for(long long i=sqrt(n);(double)clock()<(double)CLOCKS_PER_SEC-20 && i>0;--i)
		if(i*i+i*sum(i)==n)
			{cout<<i<<endl;return 0;}

	cout<<-1<<endl;
	return 0;
}
