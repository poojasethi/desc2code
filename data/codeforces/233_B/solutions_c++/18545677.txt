#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
int s(long long int x)
{
	if(x<10) return x; 
	else return s(x/10)+x%10;
}
int main()
{
	long long int n;
	cin>>n;
	for (long long int i=max(1,(int)sqrt(n)-100000); i*(i+1)<=n; i++)
	{
		if(i*i+s(i)*i==n)
		{
			cout<<i;
			return 0;
		}
	}
	cout<<-1;
}
