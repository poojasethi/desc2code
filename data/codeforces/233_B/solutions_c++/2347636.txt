#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
int s(long long n) {
	int sum=0;
	while (n>0) {
		sum+=n%10;
		n/=10;
	}
	return sum;
}
int main()
{
	long long n,l,r,i;
	cin>>n;
	l=sqrt(n)-100;
	if (l<0) l=1;
	r=sqrt(n)+1;
	for (i=l; i<r; i++) {
		if (i*i+i*s(i)==n) {
			cout<<i<<endl;
			return 0;
		}
	}
	cout<<-1<<endl;
	return 0;
}
