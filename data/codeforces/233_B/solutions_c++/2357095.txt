#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
long long s(long long i){
  return i < 10 ? i : i%10 + s(i/10);
}
int main(){
  long long n;
  cin >> n;
  int ns = sqrt(n);
  for (long long i = max(1, ns-100); i <= ns; ++i)
    if (i*(i+s(i)) == n) return cout << i << '\n', 0;
  return cout << "-1\n", 0;
}
