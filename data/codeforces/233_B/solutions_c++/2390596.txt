#include<iostream>
#include<cmath>
#include<algorithm>

using namespace std;

typedef long long ll;

ll s(ll x){
	ll ret=0;
	for(;x;x/=10) ret+=x%10;
	return ret;
}

int main()
{
	ll n,xmax,ans=-1;
	cin>>n;
	xmax=(ll)sqrt(n);
	for(ll x=max(1LL,xmax-1000); x<=xmax; ++x){
		if((x+s(x))*x==n){
			ans=x;
			break;
		}
	}
	cout<<ans<<endl;
	return 0;
}
