#include<iostream>
#include<math.h>
using namespace std;
long long int fun(long long int l)
{
int s=0;
while(l>0)
{
s+=l%10;
l=l/10;
}
return s;
}
int main()
{
long long int i,j,k,l,n;
cin>>n;
double t;
for(i=1;i<=81;i++)
{
t=(sqrt(i*i+4*n)-i)/2;
//cout<<t<<" ";

if(t-k==0.0)
{
if(i==fun(k))
{cout<<k;
return 0;}
}
}

return 0;
}
