#include <bits/stdc++.h>
using namespace std;
int d(long long x){
    int ans=0;
    while(x){
        ans+=x%10;
        x/=10;
    }
    return ans;
}
int main() {
  long long n,s,e;
  cin>>n;
  s=sqrt(n);
  e=sqrt(s);
  e*=d(e);
  for(long long i=s;i>0&&i>=s-e;i--){
    if((i*i+i*d(i))==n)
    {
        cout<<i;
        return 0;
    }
  }
  cout<<-1;
}
