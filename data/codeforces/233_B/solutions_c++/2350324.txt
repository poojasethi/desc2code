#include<iostream>
#include<cmath>
using namespace std;

int sum_up(long long tp)
{
	int t=tp%10;
	while (tp/=10) t+=tp%10;
	return t;
}
int main()
{
	long long n;
	cin>>n;
	for (long long i=static_cast<long long >(sqrt(n))+1; i<sqrt(n)+100; ++i)
	{
		if ((sum_up(n/i)+n/i)*(n/i)==n) 
		{
			cout<<n/i<<endl;
			return 0;
		}
	}
	cout<<-1<<endl; 
	return 0;
}
