#include<cstring>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<iostream>
using namespace std;
typedef long long ll;ll n;
ll s(ll a) {int ans=0;while (a) {ans+=a%10;a/=10;}return ans;}
int main() {
    cin>>n;ll a=sqrt(n);
    for (int i=max((ll)1,a-10000);i<=a;++i) if ((s(i)+i)*i==n)return cout<<i<<endl,0;
    return cout<<-1<<endl,0;
}
