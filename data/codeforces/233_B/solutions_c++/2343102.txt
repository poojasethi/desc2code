#include<cstdio>
#include<cmath>
int main()
{
	__int64 n,x,z,k,z1;
	scanf("%I64d",&n);
	double x1=sqrt((double)n);
	x=(__int64)x1;
	for(int i=0;i<200;i++)
	{
		if(i>x)break;
		z1=z=x-i;
		k=0;
		while(z1)
		{
			k+=z1%10;
			z1/=10;
		}
		if(z*(z+k)==n)
		{
			printf("%I64d",z);
			return 0;
		}
	}
	printf("-1");
	return 0;
}
