#include<iostream>
#include<cmath>
using namespace std;
int digit (long long int a)
{
	int x=0;
	while(a>0)
	{
		x+=a%10;
		a=a/10;
	}
	return x;
}
int main()
{
	long long int n,x;
	cin>>n;
	x=sqrt(n);
	while(x*x+90*x>=n)
	{
		if(x*x+digit(x)*x==n)
		{
			cout<<x;
			return 0;
		}
		x--;
	}
	
	cout<<"-1";
	return 0;
}
