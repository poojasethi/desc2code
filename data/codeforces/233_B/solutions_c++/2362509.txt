#include <iostream>
#include <cmath>

#define ll long long

using namespace std;

const double eps = 1e-9;

ll s(ll n){
	ll ans=0;
	while(n){
		ans+=n%10;
		n/=10;
	}
	return ans;
}

int main(){
	ll n,ans=-1,i,j;
	cin>>n;
	for(i=(ll)sqrt(n+eps),j=0;i>0&&j<162;i--,j++)
		if(n==i*i+s(i)*i) ans=i;
	cout<<ans<<endl;
	return 0;
}
