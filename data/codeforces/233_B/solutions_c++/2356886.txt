#include <iostream>
#include <cmath>
using namespace std;

typedef long long ll;

int s(ll x)
{
    return x < 10 ? x : x % 10 + s(x / 10);
}

int main()
{
    ll n, nn;
    cin >> n;
    nn = sqrt(n);
    for (ll x = max(nn - 100, 1LL); x < nn + 1; x++)
        if (x * x + s(x) * x - n == 0)
            return cout << x << endl, 0;
    return cout << -1 << endl, 0;
}
