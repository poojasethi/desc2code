#include<iostream>
#include<cmath>
using namespace std;


int main()
{
	long long n, a, b, w=-1;
	cin>>n;
	for(long long i=sqrt(n)+1; i-n/i<170 && i<=n; i++)
	{	
		if(n%i==0)
		{
			b=0;
			a=n/i;
			while(a)
			{
				b+=a%10;
				a/=10;
			}
			if(n/i+b==i)
			{
					w=n/i;
			}
		}
	}
	cout<<w;
	return 0;
}
