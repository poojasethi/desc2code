#include <cstdio>
#include <iostream>
#include <cmath>
using namespace std;
int main (){
	long long n,i,m,r,ans=-1;
	scanf ("%I64d",&n);
	for (i=sqrt(n)+1;i-n/i<=162&&i<=n;i++){
		if (n%i==0){
			m=n/i;
			r=0;
			while (m){
				r=r+m%10;
				m=m/10;
			}
			if (n/i+r==i){ans=n/i;break;}
		}
	}
	printf ("%I64d\n",ans);
	return 0;
}
