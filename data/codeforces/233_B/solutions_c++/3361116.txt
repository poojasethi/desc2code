#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
long long s(long long x)
{
long long sum=0;
while(x>0)
{
sum+=(x%10);
x/=10;
}
return sum;
}
int main()
{
long long int n,x,i;
cin>>n;
x=(long long) sqrt(n);
for(i=max(1LL,x-100);i<=x;i++)
{
if(i*(i+s(i))==n)
{
cout<<i;
return 0;
}

}
cout<<"-1";
return 0;
}
