#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    long long n;
    cin>>n;
    double m,d;
    long long int x=-1,s;
    for(s=1;s<=90;s++){
        m = sqrt(s*s+4*n);
        d = (-s+m)/2;
        x = d;
        if(x==d&&x>=s){
            break;
        }
        else{
            x=-1;
        }
    }
    cout<<x;

}
