#include <cstdio>
#include <cmath>
int s(long long n) {
  int sum = 0;
  while (n) {
    sum += n % 10;
    n /= 10;
  }
  return sum;
}

int main() {
  long long n, x;
  scanf("%I64d", &n);
  x = sqrt(n) - 1000;
  if (x < 0) x = 0;
  for (; x <= sqrt(n); x++)
    if ((x + s(x)) * x == n) {
      printf("%I64d", x);
      return 0;
    }
  printf("-1");
  return 0;
}
