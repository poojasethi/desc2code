#include<iostream>
#include<cmath>
using namespace std;
long long sod(long long n)
{
long long sum=0;
while(n)
{
long long r=n%10;
sum=sum+r;
n=n/10;
}
return sum;
}

int main()
{
long long n,l,r,i;
cin>>n;
l=sqrt(n)-40;
if(l<0)l=1;
r=sqrt(n)+1;

for(i=l;i<r;i++)
if(i*i+sod(i)*i==n)
{cout<<i<<endl;return 0;}
cout<<-1<<endl;
return 0;
}
