#include<iostream>
#include<math.h>
#define ll long long
using namespace std;
ll digsum(ll a){if(a==0) return 0;else return (a%10)+digsum(a/10);}
main(){
    ll n,x;cin>>n;
    int awal; (((ll)sqrt(n)-50)<2)?awal=1:awal=(ll)sqrt(n)-50;
    for(x=awal;x*x+digsum(x)*x!=n&&(x*x+x<=n);x++){}
    if(x*x+digsum(x)*x==n)cout<<x;
    else cout<<"-1";
}
