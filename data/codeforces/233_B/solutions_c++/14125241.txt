#include <iostream>
#include <cmath>
using namespace std;
int summ(long long x)
{if (x<10) return x; else return x%10+summ(x/10);
}
int main()
{long long n; cin>>n;

 for (long long i=max((long long)sqrt(n)-100,1LL);i<sqrt(n)+1;i++)
  if (i*i+summ(i)*i==n) {cout<<i; return 0;}
cout<<-1;
return 0;
}
