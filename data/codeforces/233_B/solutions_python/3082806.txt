from sys import stdin

n = int(stdin.readline().strip())

l = 0
r = n

while l + 1 < r:
    m = l + r >> 1
    if m * (m + 90) >= n:
        r = m
    else:
        l = m

ans = -1

for x in range(r, r + 90):
    if x * (x + sum([int(c) for c in str(x)])) == n:
        ans = x
        break

print(ans)
