n=input()
s=lambda x:sum(map(int, str(x)))
m=10**10
t=False
for i in range(91):
    xs = (i*i+4*n)
    if int(xs**0.5)**2 != xs:
        continue
    x=-i + int(xs**0.5)
    if x%2:
        continue
    x/=2
    if s(x) != i:
        continue
    m=min(m,x)
    t=True
if t:
    print m
else: 
    print -1
