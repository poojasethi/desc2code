n = int(raw_input())

ans = -1
for s in xrange(1,9*10+1):
    found = False
    l = 1
    r = 10**19
    while l <= r:
        m = ((r-l)/2) + l
        t = (m**2 + (s)*m)
        if ( t ==n):
            if (sum(map(int,str(m))) != s):
                break
            found = True
            ans = m
            break
        elif (t > n):
            r = m-1
        else:
            l = m+1
    if found:
        break

print ans
