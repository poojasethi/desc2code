n = int(input())

def find_x(n, s):
    low = 1
    high = 10**9
    while low <= high:
        mid = (low+high)/2
        val = mid ** 2 + (mid * s)
        if val == n:
            return mid
        elif val < n:
            low = mid + 1
        else:
            high = mid - 1
    return -1

res = float('inf')
for s in range(1, 91):
    x = find_x(n, s)
    if x != -1 and sum(map(int, str(x))) == s:
        res = min(res, x)

print(res if res != float('inf') else -1)
