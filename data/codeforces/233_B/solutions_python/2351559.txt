n = input()
ans = -1
for sx in xrange(18 * 9 + 1):
     delta = long( (sx * sx + 4 * n) ** 0.5 )
     if delta * delta != sx * sx + 4 * n:
        continue
     x = ( -sx + delta ) / 2
     if x * 2 == -sx + delta:
        tmp = x
        s = 0
        while tmp > 0:
            s += tmp % 10
            tmp /= 10
        if s == sx:
            ans = x
            break
print ans
