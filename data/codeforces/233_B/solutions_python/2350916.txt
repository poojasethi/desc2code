
def calc(n):
    ret = 0
    while n > 0:
        ret += n%10
        n /= 10
    return ret

def main():
    n = input()
    x = int(n**0.5)
    flags = False
    for i in xrange(max(1, x-1000), x+1):
        if i*i+calc(i)*i == n:
            print(i)
            flags = True
            break
    if not flags:
        print(-1)
    
if __name__ == '__main__':
    main()
