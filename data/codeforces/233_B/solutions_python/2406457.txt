# -*- coding: utf-8 -*-
from sys import exit

try:
    read = raw_input
except:
    read = input

readints = lambda: map(int, read().split())

n, = readints()

l = 1
r = 10 ** 18
f = lambda x: x * (x + sum(map(int, list(str(x)))))
while (r - l > 70000):
    m = (l + r) / 2
    y = f(m)
    if y == n:
        l = m
        break
    elif y < n:
        l = m
    else:
        r = m

for i in xrange(l, r + 1):
	if f(i) == n:
		print i
		exit(0)

print "-1"
