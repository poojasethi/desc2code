import math
def main():

    n = int(raw_input())
    for sx in range(1,9*18):

        z= (sx*sx)+(4*n)
        c= int(math.sqrt(z))
        if c*c != z:
            continue
        x= (c-sx)/2
        t=x
        sd=0
        while t>0:
            sd+=(t%10)
            t/=10
        if sd==sx and x>0:

    print "-1"
main()
