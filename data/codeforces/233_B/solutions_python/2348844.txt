import math
def num_sum(number):
    result = 0
    while number > 0:
        result += number % 10
        number = number // 10
    return result

n = int(raw_input())
x = int(math.ceil(math.sqrt(n)))
for i in xrange(x,x-1000,-1):
    if i<0:
        break;
    tmp = i * i + i * num_sum(i) - n
    if tmp==0:
        print i;
        break;

if tmp!=0:
    print "-1";
