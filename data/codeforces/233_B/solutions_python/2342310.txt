n = int(raw_input())
ans = 10 ** 20
for s in xrange(1, 100):
    d = s * s + 4.0 * n
    a = (-s + d ** 0.5) / 2.0
    for t in xrange(max(int(a-2), 1), int(a+3)):
        if n % t == 0:
            m = n / t
            tmp = min(t, m)
            if abs(t - m) == s and sum([int(c) for c in str(tmp)]) == s:
                ans = min(ans, tmp)
print -1 if ans == 10 ** 20 else ans
