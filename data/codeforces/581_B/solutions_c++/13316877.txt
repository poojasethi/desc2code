#include <bits/stdc++.h>
using namespace std;
int main() {
	int i,n,m=0,a[100005],r;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	m = a[n-1];
	a[n-1] = 0;
	for(i=n-2;i>=0;i--) {
		r = a[i];
		a[i] = max(m-a[i]+1,0);
		m = max(r,m);
	}
	for(i=0;i<n;i++)
		printf("%d%c",a[i],i==n-1?'\n':' ');
	return 0;
}
