#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int N, M, a[2][101];
void getA(int x, int N) {
    if (N==1) {
        a[x][1] = 1;
    } else if (N==2) {
        a[x][1] = 3;
        a[x][2] = 4;
    } else {
        for (int i=1; i<N; ++i) {
            a[x][i] = 2;
        }
        a[x][N] = N-2;
    }
}
int main()
{
    scanf ("%d%d", &N, &M);
    getA(0, N);
    getA(1, M);
    for (int i=1; i<=N; ++i) {
        for (int j=1; j<=M; ++j) {
            printf ("%d ", a[0][i]*a[1][j]);
        }
        printf ("\n");
    }
    return 0;
}
