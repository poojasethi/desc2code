#include <iostream>
#include <cstring>
using namespace std;

int main(void) {
	int n, m;
	cin >> n >> m;
	
	int a[n], b[m];
	
	if (n == 1) a[0] = 8;
	else if (n == 2) {
		a[0] = 7;
		a[1] = 24;
	} else {
		a[0] = n - 2;
		for (int i = 1; i < n; i++) a[i] = 2;
	}
	
	if (m == 1) b[0] = 8;
	else if (m == 2) {
		b[0] = 5;
		b[1] = 12;
	} else {
		b[0] = m - 2;
		for (int i = 1; i < m; i++) b[i] = 2;
	}
	
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) cout << a[i] * b[j] << ' ';
		cout << '\n';
	}
	
	return 0;
}
