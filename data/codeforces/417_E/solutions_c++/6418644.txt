#include<iostream>
#include<vector>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int a[100];
int b[100];
void f(int e,int * r){
    if(e==1){
        r[0]=1;
        return;
    }
    if(e==2){
        r[0]=3;      
        r[1]=4;
        return ;
    }
    r[0]=e-2;
    for(int i=1;i<e;i++)r[i]=2;
}
int main(){
    
    int x,y;
    cin>>x>>y;
    f(x,a);f(y,b);
    for(int i=0;i<x;i++){
        for(int j=0;j<y;j++){
            if(j!=0)cout<<" ";
            cout<<a[i]*b[j];
        }
        cout<<endl;
    }

}
