#include <cstdio>
using namespace std;

int A[105], B[105], M, N;

int create(int A[], int k){
	int M;
	if (k == 1) {A[0] = 1; return 0;}
	if (k == 2) {A[0] = 3; A[1] = 4; return 0;}
	if (k % 2 == 0){
		for (int i = 0; i < k - 1; i++) A[i] = 1; M = k - 1;
	}
	else {
		A[0] = 2; M = k + 2;
		for (int i = 1; i < k - 1; i++) A[i] = 1;
	}
	A[k - 1] = M / 2;
}

int main(void){
	scanf("%d%d", &N, &M);
	create(A, N); create(B, M);
	for (int i = 0; i < N; i++){
	   for (int j = 0; j < M; j++) printf("%d ", A[i]*B[j]);
	printf("\n");
    }
	return 0;
}
