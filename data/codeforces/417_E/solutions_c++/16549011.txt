#include <iostream>
#include <cstdio>
#include<memory.h>

using namespace std;

int n, m, a[100 + 10], b[100 + 10];

void get(int n,int a[])
{
	if (n==1) { a[1]=1; return ;}
	else if (n==2) { a[1]=3; a[2]=4; return ;}
	else if (n&1)
    {
		a[1]=2;
		for (int i=2; i<n; i++) a[i]=1;
		a[n]=(n+1)/2;
	}
	else
	{
		for (int i=1; i<n; i++) a[i]=1;
		a[n] = (n-2)/2;
	}
}
int main()
{
	scanf("%d%d", &n, &m);
	get(n, a);
	get(m, b);
	for (int i=1; i<=n; i++)
    {
		for (int j=1; j<=m; j++)
		{
		    printf("%d ", a[i]*b[j]);
		}
		printf("\n");
	}
	return 0;
}
