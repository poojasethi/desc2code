#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=102;
int a[2][maxn];

void solve(int n,int k){
  if (n==1){
    a[k][0]=1;
    return ;
  }
  if (n==2){
    a[k][0]=3;
    a[k][1]=4;
    return ;
  }
  for (int i=0;i<n-1;++i) a[k][i]=2;
  a[k][n-1]=n-2;
}

int main(){
  int n,m;
  int i,j;
  scanf("%d%d",&n,&m);
  solve(n,0);
  solve(m,1);
  for (i=0;i<n;++i){
    for (j=0;j<m-1;++j) printf("%d ",a[0][i]*a[1][j]);
    printf("%d\n",a[0][i]*a[1][j]);
  }
  return 0;
}
