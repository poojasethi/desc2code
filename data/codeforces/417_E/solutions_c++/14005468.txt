#include <cstdio>

int n, m, i, j;

int f(int i, int n)
{
    return i < n - 1 || n == 1 ? 1 : n == 2 ? i == 1 ? 3 : 4 : n & 1 ? i == n - 1 ? 2 : n / 2 + 1 : i == n - 1 ? 1 : n / 2 - 1;
}

int main()
{
    for (scanf("%d%d", &n, &m), i = 1; i <= n; i++, puts(""))
        for (j = 1; j <= m; j++)
            printf("%d ", f(i, n) * (f(j, m)));
}
