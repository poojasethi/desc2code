#include <stdio.h>
#include <iostream>
#include <cstring>
using namespace std;
int f[2][105],n,m;
inline void back(int x,int y){
	if (x==1){
		f[y][1]=1;
		return;
	}
	if (x==2){
		f[y][1]=3;
		f[y][2]=4;
		return;
	}
	for (int i=1;i<x;i++)
		f[y][i]=2;
	f[y][x]=x-2;
}
int main( ){
	cin>>n>>m;
	back(n,0);
	back(m,1);
	for (int i=1;i<=n;i++){
		for (int j=1;j<m;j++)
			cout<<f[0][i]*f[1][j]<<' ';
		cout<<f[0][i]*f[1][m]<<endl;
	}
	return 0;
}
