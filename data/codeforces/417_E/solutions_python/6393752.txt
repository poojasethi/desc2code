def ns(n):
    nums = [1 for i in range(n)]
    ssq = sum([x**2 for x in nums])
    root = ssq**0.5
    l = 0
    while root != int(root):
        nums[l] += 1
        l += 1
        if l == n:
            l = 0
        ssq = sum([x**2 for x in nums])
        root = ssq**0.5
    return nums
D = {}
for i in range(1, 101):
    D[i] = ns(i)
n, m = map(int, raw_input().split(" "))
N = D[n]
M = D[m]
for i in range(n):
    for j in range(m):
