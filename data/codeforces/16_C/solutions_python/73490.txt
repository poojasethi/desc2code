def GCD(a, b):
    while b > 0:
        a, b = b, a % b
    return a

a, b, x, y = [int(x) for x in raw_input().split()]
#a = max(data[0], data[1])
#b = min(data[0], data[1])
#x = max(data[2], data[3])
#y = min(data[2], data[3])
tp = GCD(x, y)
x /= tp
y /= tp

if x > a or y > b:
    print '0 0'
else:
    portion = a / x
    if portion * y <= b:
        print str(portion * x), str(portion * y)
    else:
        portion = b / y
        print str(portion * x), str(portion * y)
