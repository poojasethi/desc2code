
a, b, x, y = map(int, raw_input().split()) 

def gcd(x, y): 
    if y == 0: 
        return x 
    return gcd(y, x % y) 

d = gcd(x, y)

x, y = x / d, y / d

if a < x or b < y: 
    print 0, 0
else: 
    M = 2 * (10**9)
    l = 1 
    r = M 

    while l <= r: 
        m = (l + r) / 2
        if a >= m * x and b >= m * y:
            l = m + 1
        else: 
            r = m - 1

    print r * x, r * y
