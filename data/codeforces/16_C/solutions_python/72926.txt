def readint(): return int(raw_input())
def readfloat(): return float(raw_input())
def readarray(N, foo):
        res = []
        for _ in xrange(N):
                res.append(foo())
        return res
def readlinearray(foo): return map(foo, raw_input().split())

def gcd(a, b):
    """Return greatest common divisor using Euclid's Algorithm."""
    while b:      
        a, b = b, a % b
    return a

def simplify(x, y):
    _gcd = gcd(x, y)
    while(_gcd != 1):
        x /= _gcd
        y /= _gcd
        _gcd = gcd(x, y)
    return x, y

def get_reduction(a, b, x, y):
    x, y = simplify(x, y)
    z = min(a / x, b / y)
    return z * x, z * y

if __name__ == '__main__':
    a, b, x, y = readlinearray(int)
    print '%d %d' % get_reduction(a, b, x, y)
