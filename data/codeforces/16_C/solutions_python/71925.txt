#! /usr/bin/python
import sys
import fractions

#nacteni 2 cisel
numbers = sys.stdin.readline()[:-1].split()
a = int(numbers[0])
b = int(numbers[1])
x = int(numbers[2])
y = int(numbers[3])

divide = fractions.gcd(x,y)
x = x/divide   
y = y/divide   

if (a < x) or (b < y):
    print "0 0"
else:
    factor = min(a/x, b/y)
    print "%d %d" %(x*factor,y*factor)
