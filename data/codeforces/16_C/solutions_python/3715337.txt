import sys
def gcd(a,b):
    while(b != 0):
        temp = a
        a = b
        b = temp%a
    return a
if __name__ == '__main__':

    a,b,x,y = map(int,sys.stdin.readline().split())
    x,y = x/gcd(x,y),y/gcd(x,y)
    m = a/x
    n = b/y
    if m > n:
        print n*x,n*y
    else:
        print m*x,m*y
