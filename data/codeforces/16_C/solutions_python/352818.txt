a, b, x, y = [int(x) for x in raw_input().split()]

def gcd(a, b):
    if b == 0: return a
    return gcd(b, a % b)

g = gcd(x, y)
x /= g
y /= g

A, B = 0, 0

r = a / x
s = b / y

if r * x <= a and r * y <= b:
    if A < r * x:
        A, B = r * x, r * y

if s * x <= a and s * y <= b:
    if A < s * x:
        A, B = s * x, s * y

print A, B
