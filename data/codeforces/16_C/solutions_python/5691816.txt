a, b, x, y = map ( int, raw_input( ).split( ) )

low = 0
high = int(2e9+1)

def gcd(a,b):
    if ( not b ):
        return a
    return gcd (b, a%b)

t = gcd ( x, y )
x /= t
y /= t
while True:
    if ( low == high- 1 ):
        print "%d %d" %(low*x, low*y)
        break
    mid = (low+high)/2
    #print mid
    if ( mid * x <= a and mid * y <= b ):
        low = mid
    else:
        high = mid
