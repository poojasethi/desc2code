#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;

#define N 111111
ll a[N];
ll sum[N];
ll ans;

int main(){
	int n;
    sum[0]=0;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%I64d",&a[i]);
		sum[i]=sum[i-1]+a[i];
	}
	ans=a[2]*a[2]+1;
    for(ll i=1;i<n;i++){
		if(i*i>ans) break;
		ll t=ans;
		for(int j=i+1;j<=n;j++){
		   t=min(t,(sum[j]-sum[j-i])*(sum[j]-sum[j-i]));
        }
	    ans=min(ans,t+i*i);
		}
		cout<<ans<<endl;
		}
