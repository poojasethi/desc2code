#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
#define maxn 101000
int a[maxn];
#define sqr(x) ((x)*1ll*(x))
int main(){
	int n,x,sum=0;
	scanf("%d",&n);
	for (int i=1;i<=n;i++) {
		scanf("%d",&x);
		sum+=x;
		a[i]=sum;
	}
	long long ans=0x7ffffff*1ll*0x7fffffff;
	for (int l=1;l<=n;l++) {
		if (sqr(l)>=ans) break;
		for (int i=1;l+i<=n;i++) {
			ans=min(ans,sqr(l)+sqr(a[i+l]-a[i]));
		}
	}
	printf("%I64d\n",ans);
	return 0;
}
