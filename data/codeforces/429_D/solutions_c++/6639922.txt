#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
int n;
int s[100005],b[100005];

int main(){
    scanf("%d",&n);
    int ans=2000000000;
    for (int i=1;i<=10002;i++) b[i]=10001;
    for (int i=1;i<=n;i++){
        scanf("%d",&s[i]);
        s[i]+=s[i-1];
        for (int j=1;j<10002&&i-j>0;j++) b[j]=min(abs(s[i]-s[i-j]),b[j]);
    }
    for (int i=1;i<10002;i++)
        ans=min(ans,b[i]*b[i]+i*i);
    printf("%d\n",ans);
    return 0;
}
