#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int n;
int a[100100];
__int64 s[100100];

int main(){
    int i,j;
    cin>>n;
    for(i=1;i<=n;++i)cin>>a[i],s[i]=a[i]+s[i-1];
    __int64 ans = (__int64)0x7fffffff*0x7fffffff;
    for(j = 1 ; j <= 1010 ; ++j){
        for(i=1;i+j<=n;++i){
            __int64 t = j*j+(s[i+j]-s[i])*(s[i+j]-s[i]);
            if(ans>t)ans = t;
        }
    }
    cout<<ans<<endl;
    return 0;
}
