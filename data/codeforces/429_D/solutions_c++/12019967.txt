#include<cstdio>
#include<iostream>
#include<algorithm>

using namespace std;

long long int a[100000];

int main() {
    int n, temp;
	//freopen("D.in","r",stdin),freopen("D.out","w",stdout);
    cin >> n;
    cin >> a[0];
    for (int i = 1; i < n; i++) {
        scanf("%d", &temp);
        a[i] = a[i-1] + temp;
    }
    long long int minimum = 1LL<<62;
    for (int i = 0; i < n; i++) {
        for (int j = i - 1; j >= max(0, i - 1003); j--) {
            long long int hasil = a[i] - a[j];
            minimum = min(minimum, (i-j)*(i-j) + hasil*hasil);
        }
    }
    cout << minimum << endl;
    return 0;
}
