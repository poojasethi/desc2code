#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
typedef long double ld;
#define pb push_back
#define mp make_pair
const ll N=1e6+6;
int num[N],pref[N];
ll ans=1e14;
int main()
{
    int n; cin>>n;
    for(int i=1;i<=n;i++){
        cin>>num[i];
        pref[i]=pref[i-1]+num[i];
    }
    for(int i=1;i<=n;i++){
        for(int j=i+1;j<=min(i+1100,n);j++){
            ans=min(ans,(ll)(i-j)*(i-j)+(ll)(pref[j]-pref[i])*(pref[j]-pref[i]));
        }
    }
    cout<<ans;
    return 0;
}
