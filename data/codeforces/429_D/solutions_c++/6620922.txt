#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>

using namespace std;

int n,a[100010];

int main()
{
	int ans=200000000;
	scanf("%d",&n);
	for(int i=0;i<n;i++) scanf("%d",&a[i]);
	for(int k=1;k<n;k++){
		if(k*k>ans) break;
		int res=0,sum=0;
		for(int i=1;i<=k;i++) sum+=a[i];
		res=abs(sum);
		for(int i=k+1;i<n;i++){
			sum+=a[i];sum-=a[i-k];
			res=min(res,abs(sum));
		}
		if(res<20000) ans=min(ans,k*k+res*res);
	}
	printf("%d\n",ans);
	return 0;
}
