#include <cstdio>
#include <algorithm>

using namespace std;

typedef long long lld;

const int maxN = 100010, maxI = 1010;
lld n, minC, a[ maxN ], s[ maxN ];

lld sqr( lld x ) { return x * x; }

int main( )
{
    scanf( "%I64d", &n );
    for ( lld i = 1; i <= n; i++ )
        scanf( "%I64d", &a[ i ] ), s[ i ] = s[ i - 1 ] + a[ i ];
    minC = 1 + sqr( a[ 2 ] );
    for ( lld i = 1; i < n; i++ )
        for ( lld j = i + 1; j <= min( n, i + maxI ); j++ )
            minC = min( minC, sqr( j - i ) + sqr( s[ i ] - s[ j ] ) );
    printf( "%I64d\n", minC );
    return 0;
}
