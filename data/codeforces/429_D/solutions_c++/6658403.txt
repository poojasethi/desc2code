#include <cstdio>
#include <vector>
#include <cstring>
#include <cstdlib>
#include <iostream>
#include <algorithm>
using namespace std;
int n, a[100010];

int main()
{
	scanf("%d", &n);
	for (int i = 0; i < n; i ++) {
		scanf("%d", &a[i]);
	}
	long long ret = 2000000000;
	for (int i = 0; i < n; i ++) {
		long long s = 0;
		for (int j = 0; j < 1010 && j < i; j ++) {
			s += a[i - j];
			ret = min(ret, s * s + (j + 1) * (j + 1));
		}
	}
	printf("%I64d\n", ret);
	return 0;
}
