#include <bits/stdc++.h>

using namespace std;
long long a[100000], s[100000];
int n;
void ENTER()
{
    cin>>n;
    for(long long i=1; i<=n; i++)cin>>a[i];
    s[0]=0;
    for(long long i=1; i<=n; i++) s[i]=a[i]+s[i-1];
}
long long f(long long i, long long j)
{
    return ((i-j)*(i-j)+ (s[i]-s[j])*(s[i]-s[j]));
}
int main()
{
    ENTER();
    long long kq=f(1, 2);
    for(int i=1; i<n; i++)
        for(int j=i+1; j<=min(i+2000, n); j++)
        kq=min(kq, f(i,j));
    cout<<kq;
    return 0;
}
