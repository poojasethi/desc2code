#include<bits/stdc++.h>
#define ll long long

using namespace std;

const int maxn = 1e5 + 17;

ll sum[maxn];

int main()
{
    ios_base::sync_with_stdio(false);

    ll n;
    cin >> n;

    for(int i = 1 ; i <= n ; i++){
        ll a;
        cin >> a;
        sum[i] = sum[i-1] + a;
    }

    ll ans = 2e15;
    for(ll i = 1 ; i <= n ; i++)
        for(ll j = i + 1 ; j <= min(i + 1111 , n) ; j++)
            ans = min(ans , (j - i) * (j - i) + (sum[j] - sum[i]) * (sum[j] - sum[i]));
    cout << ans;
}
