#include <stdio.h>

int k,x,n,m;

struct data
{
	long long cnt;
	char s,e;
}d[55];

bool process()
{
	int i,j,l,I,J,tn,tm;
	for(i=0;i<4;i++)
	{
		if(i==3 && n==1) continue;
		for(j=0;j<4;j++)
		{
			if(j==3 && m==1) continue;
			for(I=0;I<=(n-(i>0)-(i==3))/2;I++)
			{
				for(J=0;J<=(m-(j>0)-(j==3))/2;J++)
				{
					d[1].cnt=I;
					d[1].s=(i&1)?'C':'B';
					d[1].e=((i>>1)&1)?'A':'B';

					d[2].cnt=J;
					d[2].s=(j&1)?'C':'B';
					d[2].e=((j>>1)&1)?'A':'B';

					for(l=3;l<=k;l++)
					{
						d[l].cnt=d[l-2].cnt+d[l-1].cnt;
						d[l].s=d[l-2].s;
						d[l].e=d[l-1].e;
						if(d[l-2].e=='A' && d[l-1].s=='C') d[l].cnt++;
						if(d[l].cnt>x) break;
					}
					if(l>k && d[k].cnt==x)
					{
						tn=n;
						tm=m;

						if(i&1)
						{
							printf("C");
							tn--;
						}
						for(l=1;l<=I;l++)
						{
							printf("AC");
							tn-=2;
						}
						for(;tn>1;tn--) printf("B");
						if((i>>1)&1) printf("A");
						else if(tn>0) printf("B");
						printf("\n");
						if(j&1)
						{
							printf("C");
							tm--;
						}
						for(l=1;l<=J;l++)
						{
							printf("AC");
							tm-=2;
						}
						for(;tm>1;tm--) printf("B");
						if((j>>1)&1) printf("A");
						else if(tm>0) printf("B");

						return true;
					}
				}
			}
		}
	}
	return false;
}

int main()
{
	scanf("%d %d %d %d",&k,&x,&n,&m);
	if(!process()) printf("Happy new year!");
	return 0;
}
