//In the name of God
#include <iostream>
using namespace std;

int n, m, k, x;
string s, t;

bool pos(int s, int e, int f, int n) {
	if (n == 1)
		return !f && (!s || !e);
	return 2 * f + s + e <= n;
}
bool acc(int s1, int e1, long long f1, int s2, int e2, long long f2) {
	for (int i = 3; i <= k; i++) {
		int st = s1, ed = e2; long long f = f1 + f2 + (e1 && s2);
		swap(f1, f2), swap(s1, s2), swap(e1, e2);
		swap(f2, f), swap(s2, st), swap(e2, ed);
	}
	return f2 == x;
}
bool make(int ac, int head, int tail, int len, string &ret) {
	ret = "";
	if (head)
		ret += 'C';
	for (int i = 0; i < ac; ++i)
		ret += "AC";
	while (ret.size() < len - tail)
		ret += 'X';
	if (tail)
		ret += 'A';
	return ret.size() == len;
}
int main() {
	cin >> k >> x >> n >> m;
	for (int s1 = 0; s1 < 2; s1++)
		for (int e1 = 0; e1 < 2; e1++)
			for (int s2 = 0; s2 < 2; s2++)
				for (int e2 = 0; e2 < 2; e2++)
					for (int f1 = 0; 2 * f1 <= n; f1++)
						for (int f2 = 0; 2 * f2 <= m; f2++)
							if (pos(s1, e1, f1, n) && pos(s2, e2, f2, m) && acc(s1, e1, f1, s2, e2, f2))
								if (make(f1, s1, e1, n, s) && make(f2, s2, e2, m, t)) {
									cout << s << '\n' << t << '\n';
									return 0;
							}
	cout << "Happy new year!\n";
	return 0;
}
