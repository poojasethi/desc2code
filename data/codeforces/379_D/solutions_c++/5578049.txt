#include<cstdio>

int x;

bool compute(int k, long long nr1, long long nr2, bool a1, bool c1, bool a2, bool c2){
	long long nr3;
	bool a3, c3;
	while(k--){
		nr3 = nr1 + nr2 + (a1 && c2);
		if(nr3 > x)
			return 0;
		c3 = c1;
		a3 = a2;

		nr1 = nr2;
		nr2 = nr3;
		c1 = c2;
		c2 = c3;
		a1 = a2;
		a2 = a3;
	}

	return nr3 == x;
}

void putstr(int n, long long nr, bool a, bool c){
	n -= 2*nr + a + c;
	if(c)
		putchar('C');
	for(int i=0;i<nr;i++)
		printf("AC");
	for(int i=0;i<n;i++)
		printf("X");
	if(a)
		putchar('A');
}

int main(void){
	int k, n, m;
	scanf("%d%d%d%d", &k, &x, &n, &m);
	k-=2;
	
	for(int a1 = 0;a1<2;a1++)
		for(int c1 = 0;c1<2;c1++)
			for(long long nr1 = 0;nr1*2 + c1 + a1 <= n;nr1++)
				for(int a2 = 0;a2<2;a2++)
					for(int c2=0;c2<2;c2++)
						for(long long nr2 = 0;nr2*2 + c2 + a2 <= m;nr2++)
							if(compute(k, nr1, nr2, a1, c1, a2, c2)){
								putstr(n, nr1, a1, c1);
								puts("");
								putstr(m, nr2, a2, c2);
								return 0;
							}

	puts("Happy new year!");
	return 0;
}
