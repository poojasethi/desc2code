#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <vector>
using namespace std;
const int N = 300005;
int n, m, k, x;
int cal(int id, int l1, int l2, int s1, int e1, int s2, int e2)
{
    if (l2 > x || id == k) return l2;
    return cal(id + 1, l2, l1 + l2 + (e1 & s2), s2, e2, s1, e2);
}
void print(int l, int s, int e, int n)
{
    int left = n - 2 * l - s - e;
    if (s) putchar('C');
    while (l --) putchar('A'), putchar('C');
    while (left --) putchar('Z');
    if (e) putchar('A');
    putchar('\n');
}
int dfs(int l1, int l2, int s1, int e1, int s2, int e2)
{
    if (l1 * 2 + s1 + e1 > n || l2 * 2 + s2 + e2 > m) return 0;
    if (x == cal(2, l1, l2, s1, e1, s2, e2))
        return print(l1, s1, e1, n), print(l2, s2, e2, m), 1;
    return 0;
}
int main()
{
    scanf("%d%d%d%d", &k,&x,&n,&m);
    for (int i = 0; i * 2 <= n; i ++)
        for (int j = 0; j * 2 <= m; j ++)
            for (int k = 0; k < 16; k ++)
                if (dfs(i, j, k & 1, !!(k & 2), !!(k & 4), !!(k & 8)))
                    return 0;

    puts("Happy new year!");
    return 0;
}
