#include <iostream>
#include <string>
using namespace std;
string s[100],ss[9]={"AA","AC","CC","CA","AB","BA","BC","CB","BB"};
int k,t,x,y,n,m,i,K,a[100];
int main() {
	ios::sync_with_stdio(false);
	cin >> k >> K >> n >> m;
	for (a[1]=0;a[1]*2<=n;a[1]++)
		for (a[2]=0;a[2]*2<=m;a[2]++)
			for (x=0;x<9;x++)
				for (y=0;y<9;y++) {
					s[1]=ss[x];s[2]=ss[y];
					if ((a[1]==0)&&(s[1]=="AC")) continue;
					if ((a[2]==0)&&(s[2]=="AC")) continue;
					if ((a[1]*2==n)&&(s[1]!="AC")) continue;
					if ((a[2]*2==m)&&(s[2]!="AC")) continue;
					if ((a[1]*2+1==n)&&(s[1][0]!=s[1][1])) continue;
					if ((a[2]*2+1==m)&&(s[2][0]!=s[2][1])) continue;
					for (i=3;i<=k;i++) {
						a[i]=a[i-1]+a[i-2];
						if (a[i]>K) {a[k]=K-1;break;}
						if ((s[i-1][0]=='C')&&(s[i-2][1]=='A')) a[i]++;
						s[i]="AA";
						s[i][0]=s[i-2][0];
						s[i][1]=s[i-1][1];
					}
					if (a[k]==K) {
						t=0;cout << s[1][0];t++;
						if ((a[1]>0)&&(s[1][0]=='A')) {cout << 'C';a[1]--;t++;}
						while (a[1]--) {cout << "AC";t+=2;}
						while (t<n-1) {cout << "B";t++;}
						if (t<n) cout << s[1][1] << endl;
						else cout << endl;
						t=0;cout << s[2][0];t++;
						if ((a[2]>0)&&(s[2][0]=='A')) {cout << 'C';a[2]--;t++;}
						while (a[2]--) {cout << "AC";t+=2;}
						while (t<m-1) {cout << "B";t++;}
						if (t<m) cout << s[2][1] << endl;
						else cout << endl;
						return 0;
					}
				}
	cout << "Happy new year!" << endl;
	return 0;
}
