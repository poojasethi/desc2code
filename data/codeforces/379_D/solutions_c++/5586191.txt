#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
#include<iostream>
using namespace std;
int k,x,n,m;
char A[1000],B[1000];
bool get(char *a,int x1,int x2,int i,int len)
{
	a[len]=0;
	if(len==1)if(x1!=x2)return 0;
	a[0]='A'-1+x1;
	a[len-1]='A'-1+x2;
	if(i*2<=len-2)
	{
		for(int j=1;j<=i;j++)
			a[j*2-1]='A',a[j*2]='C';
		for(int j=i*2+1;j<len-1;j++)a[j]='B';
	}else{
		if(i*2>len)return 0;
		if(i*2==len)
		{
			if(a[0]!='A'||a[len-1]!='C')return 0;
			for(int j=0;j<len;j+=2)a[j]='A',a[j+1]='C';
		}else 
		{
			if(a[0]=='A')
				for(int j=0;j<len-1;j+=2)a[j]='A',a[j+1]='C';
			else if(a[len-1]=='C')
				for(int j=1;j<len;j+=2)a[j]='A',a[j+1]='C';
			else if(n>1||(n==1&&i!=0))return 0;
		}
	}return 1;
}
bool work(int kth,int x1,int x2,int x3,int x4,int i,int j)
{
	if(kth==k)return (i==x);
	int nx1=x3,nx2=x4,ni=j,nx3=x1,nx4=x4,nj=i+j+(x2==1&&x3==3);
	if(i>x)return 0;
	return work(kth+1,nx1,nx2,nx3,nx4,ni,nj);
}
int main(){
	scanf("%d%d%d%d",&k,&x,&n,&m);
	for(int x1=1;x1<=3;x1++)
		for(int x2=1;x2<=3;x2++)
			for(int x3=1;x3<=3;x3++)
				for(int x4=1;x4<=3;x4++)
					for(int i=0;i<=max(n,0);i++)
						for(int j=0;j<=max(0,m);j++)
						{
							if(work(1,x1,x2,x3,x4,i,j))
							{
								if(get(A,x1,x2,i,n)&&get(B,x3,x4,j,m))
								{cout<<A<<endl<<B<<endl;
								return 0;}
							}
						}
	puts("Happy new year!");
}
