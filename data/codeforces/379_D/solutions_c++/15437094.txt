#include<bits/stdc++.h>
using namespace std ;
#define ll long long
ll k,x,n,m ;
bool flag ;
int a[109],c[109] ;
ll f[109] ;
char s[111] ;
void gg(int ind,int nn)
{
  int i= 0;
  //cout<<nn<<endl;
  for(i = 0;i<nn;i++) s[i] = 'B' ;
  //cout<<s<<endl;
  if(a[ind]) s[i-1] = 'A'  ;
  int j =0 ;
  if(c[ind]) s[j++] = 'C' ;
  for(int i = 0;i<f[ind] ;i++)
  {
    s[2*i+j] = 'A' ;
    s[2*i+j+1] = 'C' ;
 }
  s[nn] = '\0' ;
  puts(s) ;
}
void func(ll ind)
{
  if(ind == 1||ind==2)
  {  int nn ;
    if(ind==1) nn  = n;
    else nn = m ;
    for(a[ind]=0;a[ind]<2;a[ind]++) for(c[ind]=0;c[ind]<2;c[ind]++)for(f[ind]=0;2*f[ind]+a[ind]+c[ind]<=nn;f[ind]++) {
    func(ind+1); if(flag) return ;
    }
    }
  else{
      {
   for(;ind<=k;ind++)
   {
     f[ind] = f[ind-1]+f[ind-2] + (a[ind-2]&&c[ind-1]) ;
     a[ind] = a[ind-1] ;
     c[ind] = c[ind-2] ;
    }
  }
  if(ind>k &&f[k]==x) {
    flag = true ;
    gg(1,n);
    gg(2,m) ;
    return ;
    }
  }
}
int main()
{
  cin>>k>>x>>n>>m ;
  func(1) ;
  if(!flag )
  cout<<"Happy new year!" ;
}
