#include<algorithm>
#include<cstring>
#include<cstdio>
#include<vector>

using namespace std;

typedef long long ll;

int k, x, n, m;

ll t[51];
ll f(bool e1, bool e2, bool b1, bool b2, int c1, int c2)
{
    t[1] = c1;
    t[2] = c2;
    for(int i = 3; i <= k; ++i)
    {
        t[i] = t[i-1] + t[i-2];
        if(i == 3)
            t[i] += e1 && b2;
        else if(i & 1)
            t[i] += e2 && b2;
        else
            t[i] += e2 && b1;
    }
    return t[k];
}

void o(bool e, bool b, int c, int n)
{
    if(b)
        printf("C");
    for(int i = 0; i < c; ++i)
        printf("AC");
    for(int i = e + b + 2 * c; i < n; ++i)
        printf("B");
    if(e)
        printf("A");
    puts("");
}

int main()
{
    scanf("%d %d %d %d", &k, &x, &n, &m);

    for(int e1 = 0; e1 < 2; ++e1)
        for(int e2 = 0; e2 < 2; ++e2)
            for(int b1 = 0; b1 < 2; ++b1)
                for(int b2 = 0; b2 < 2; ++b2)
                    for(int c1 = 0; 2 * c1 <= n - e1 - b1; ++c1)
                        for(int c2 = 0; 2 * c2 <= m - e2 - b2; ++c2)
                            if(f(e1, e2, b1, b2, c1, c2) == x)
                            {
                                o(e1, b1, c1, n);
                                o(e2, b2, c2, m);
                                return 0;
                            }
    puts("Happy new year!");
    return 0;
}
