#include<stdio.h>
int k,n,m,x;

long long check(long long a,long long b,int* c)
{
	int i;
	long long tmp;
	for(i=3;i<=k;i++){
		tmp=a;
		if(i==3)a=b+a+c[0];
		else a=b+a+c[i%2+1];
		b=tmp;
	}
	return a;
}

bool output(int a,int b,int* c)
{
	char s1[105]={0},s2[105]={0};
	if(c[0]==1)s1[n]='A',s2[1]='C';
	if(c[1]==1){
		if(s1[1]=='C')return false;
		s2[m]='A',s1[1]='C';
	}
	if(c[2]==1){
		if(s2[m]=='C'||s2[1]=='A')return false;
		s2[m]='A',s2[1]='C';
	}
	for(int i=1;i<=a;i++){
		if(s1[1]!=0){
			if(s1[i*2]!=0||s1[i*2+1]!=0)return false;
			s1[i*2]='A',s1[i*2+1]='C';
		}
		else{
			if(s1[i*2-1]!=0||s1[i*2]!=0)return false;
			s1[i*2-1]='A',s1[i*2]='C';
		}
	}
	for(int i=1;i<=b;i++){
		if(s2[1]!=0){
			if(s2[i*2]!=0||s2[i*2+1]!=0)return false;
			s2[i*2]='A',s2[i*2+1]='C';
		}
		else{
			if(s2[i*2-1]!=0||s2[i*2]!=0)return false;
			s2[i*2-1]='A',s2[i*2]='C';
		}
	}
	if(s1[n+1]!=0||s2[m+1]!=0)return false;
	for(int i=1;i<=n;i++){
		if(s1[i]==0)s1[i]='B';
	}
	for(int i=1;i<=m;i++)if(s2[i]==0)s2[i]='B';
	printf("%s\n%s",s1+1,s2+1);
	return true;
}

int main()
{
	int i,j;
	int c[3]={0,0,0};
	scanf("%d%d%d%d",&k,&x,&n,&m);
	for(int i=0;i<=n/2;i++){
		for(int j=0;j<=m/2;j++){
			for(int l=0;l<8;l++){
				if(l==3)continue;
				c[0]=l%2;
				c[1]=(l/2)%2;
				c[2]=l/4;
				if(check(j,i,c)==x&&output(i,j,c)){
					return 0;
				}
			}
		}
	}
	printf("Happy new year!");
	return 0;
}
