#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int k, x, n, m, C[5], A[5], V[5], i, j; 
	cin>>k>>x>>n>>m;
	for (C[0]=0; C[0]<2; C[0]++) for (C[1]=0; C[1]<2; C[1]++)
	for (A[0]=0; A[0]<2; A[0]++) for (A[1]=0; A[1]<2; A[1]++)
	for (V[0]=0; n-C[0]-A[0]>=0 && V[0]<=(n-C[0]-A[0])/2; V[0]++)
	for (V[1]=0; m-C[1]-A[1]>=0 && V[1]<=(m-C[1]-A[1])/2; V[1]++) {
		C[2]=C[0]; A[2]=A[0]; V[2]=V[0];
		C[3]=C[1]; A[3]=A[1]; V[3]=V[1];
		for (i=3; i<=k; i++) {
			C[4]=C[2]; A[4]=A[3]; V[4]=V[2]+V[3]+A[2]*C[3];
			if (V[4]>x) break;
			C[2]=C[3]; A[2]=A[3]; V[2]=V[3];
			C[3]=C[4]; A[3]=A[4]; V[3]=V[4];
		}
		if (V[4]==x) {
			for (j=0; j<2; j++) {
				if (C[j]) cout<<"C";
				for (i=0; i<V[j]; i++) cout<<"AC";
				for (i=0; i<(j==1?m:n)-V[j]*2-A[j]-C[j]; i++) cout<<"Z";
				if (A[j]) cout<<"A";
				cout<<endl;
			}
			return 0;
		}
	}
	cout<<"Happy new year!"<<endl;
	return 0;
}
