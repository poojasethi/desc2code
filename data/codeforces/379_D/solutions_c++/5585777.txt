#include <cstdio>

#define LL long long

int k, x, n, m;
char A[101], B[101];

bool check(int m, int a, int n)
{
	int r = (m&1) ? 1 : 0;
	r += (m&2) ? 1 : 0;
	if ((n-r)/2 < a) return false;
	return true;
}

void monta(char* s, int m, int a, int n)
{
	s[n] = '\0';
	if (m&2) s[0] = 'C';
	if (m&1) s[n-1] = 'A';
	int i = 0;
	if (m&2) i = 1;
	while (a--) s[i] = 'A', s[i+1] = 'C', i += 2;
	while (i < n) 
	{
		if (!s[i]) s[i] = 'B';
		++i;
	}
}

bool solve(int z, LL a, LL b)
{
	if (!check(z>>2, b, m)) return false;
	if (!check(z, a, n)) return false;
	
	int p = 2, o = z, aa = a, bb = b;
	while (p < k)
	{
		++p;
		LL c = a+b;
		a = b;
		if ((z&1) && (z&8)) ++c;
		b = c;
		int zz = z>>2;
		zz |= (z&2) ? 8 : 0;
		zz |= (z&4) ? 4 : 0;
		z = zz;
	}
	
	if (b == x)
	{
		monta(A, o, aa, n);
		monta(B, o>>2, bb, m);
	}
	
	return b == x;
}

int main()
{
	scanf("%d%d%d%d", &k, &x, &n, &m);
	
	bool g = 0;
	for (int i = 0; i < 1<<4 && !g; ++i)
	{
		if (n == 1 && (i^3) == 0) continue;
		for (int a = 0; a <= n && !g; ++a)
		{
			if (m == 1 && (i>>2^3) == 0) continue;
			for (int b = 0; b <= m && !g; ++b)
			{
				g = solve(i, a, b);
			}
		}
	}
	
	if (!g) printf("Happy new year!\n");
	else printf("%s\n%s\n", A, B);
}
