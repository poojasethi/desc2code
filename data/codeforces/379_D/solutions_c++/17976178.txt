#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int k,x,m,n,s[105],e[105];
long long f[105];
void out (int s,int e,int num,int n)
{
	char temp[105];
	memset(temp,0,sizeof(temp));
	int t=(s=='A'?0:1);
	temp[0]=s;temp[n-1]=e;
	for(int i=1;i<n-1;i++)
		temp[i]='B';
	for(int i=0;i<num;i++)
	{
		temp[t+2*i]='A';temp[t+2*i+1]='C';
	}
	printf("%s\n",temp);
}
int main()
{
	scanf("%d%d%d%d",&k,&x,&n,&m);
	for(char st1='A';st1<='C';st1++)
		for(char ed1='A';ed1<='C';ed1++)
		{
			if(n==1&&st1!=ed1)continue;
			for(char st2='A';st2<='C';st2++)
				for(char ed2='A';ed2<='C';ed2++)
				{
					if(m==1&&st2!=ed2)	continue;
					int max1=(n-(st1!='A')-(ed1!='C'))/2;
					int max2=(m-(st2!='A')-(ed2!='C'))/2;
					for(int n1=0;n1<=max1;n1++)
					{
						if(n==2&&st1=='A'&&ed1=='C'&&n1==0)	continue;
						for(int n2=0;n2<=max2;n2++)
						{
							if(m==2&&st2=='A'&&ed2=='C'&&n2==0) continue;
							s[1]=st1;s[2]=st2;e[1]=ed1;e[2]=ed2;f[1]=n1;f[2]=n2;
							for(int i=3;i<=k;i++)
							{
								s[i]=s[i-2];e[i]=e[i-1];
								f[i]=f[i-1]+f[i-2]+(e[i-2]=='A'&&s[i-1]=='C');
							}
							if(f[k]==x)
							{
								out(st1,ed1,n1,n);
								out(st2,ed2,n2,m);
								return 0;
							}
							
						}
					}
				}
		}
	printf("Happy new year!\n");
	return 0;
}
