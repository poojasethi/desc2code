int x;
bool compute(int k,int nr1,int nr2,bool a1,bool c1,bool a2,bool c2)
{
	bool a3,c3;
	int nr3;
	while(k--)
	{
		nr3=nr1+nr2+(a1 & c2);
		if(nr3>x) return 0;
		a3=a2;
		c3=c1;
		
		a1=a2;
		c1=c2;
		
		a2=a3;
		c2=c3;
		
		nr1=nr2;
		nr2=nr3;
		
	}
	
	return nr3==x;
	
}

void putstr(bool a1,int nr1,bool c1,int n)
{
	n-=nr1*2+a1+c1;
	if(c1)cout<<"C";
	for(int i=0;i<nr1;++i)
	cout<<"AC";
	for(int i=0;i<n;++i)
	cout<<"X";
	if(a1)cout<<"A";
	cout<<endl;
}

int main()
{
	int k,n,m;
	
	while(cin>>k>>x>>n>>m)
	{
		k-=2;
		for(int a1=0;a1<2;++a1)
			for(int c1=0;c1<2;++c1)
				for(int a2=0;a2<2;++a2)
					for(int c2=0;c2<2;++c2)
						for(int nr1=0;nr1*2+a1+c1<=n;++nr1)
							for(int nr2=0;nr2*2+a2+c2<=m;++nr2)
								if(compute(k,nr1,nr2,a1,c1,a2,c2))
								{
									putstr(a1,nr1,c1,n);
									putstr(a2,nr2,c2,m);
									goto signal;
								}
		
		cout<<"Happy new year!"<<endl;
		signal:;
	}
	
	
	return 0;
}
