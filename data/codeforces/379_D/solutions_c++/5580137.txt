#include<iostream>
#include<string>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<ctime>
#include<algorithm>
#include<map>
using namespace std;

const int num[] = { 0, 1, 1, 2 };
#define N 110
long long val[N];
int head[N], tail[N];
int n, m, x, k;

void print() {
	int i, j;
	j=0;
	if (head[0]) { printf("C"); ++j; }
	for (i=0; i<val[0]; ++i) { printf("AC"); j+=2; }
	for (; j<n-1; ++j) printf("D");
	if (j<n) if (tail[0]) printf("A"); else printf("D");
	printf("\n");

	j=0;
	if (head[1]) { printf("C"); ++j; }
	for (i=0; i<val[1]; ++i) { printf("AC"); j+=2; }
	for (; j<m-1; ++j) printf("D");
	if (j<m) if (tail[1]) printf("A"); else printf("D");
	printf("\n");
}

int cal() {
	int i;
	for (i=2; i<k; ++i) {
		val[i]=val[i-2]+val[i-1];
		head[i]=head[i-2]; tail[i]=tail[i-1];
		if (tail[i-2]&&head[i-1]) val[i]+=1;
	}
	if (val[k-1]==x) return 1;
	return 0;
}

void conduct() {
	int i, j, k, l;
	for (i=0; i<=n; i+=2)
		for (j=0; j<=m; j+=2)
			for (k=0; k<4; ++k) for (l=0; l<4; ++l) {
				if (n-i<num[k] || m-j<num[l]) continue;
				if (k&1) head[0]=1; else head[0]=0;
				if (k&2) tail[0]=1; else tail[0]=0;
				if (l&1) head[1]=1; else head[1]=0;
				if (l&2) tail[1]=1; else tail[1]=0;
				val[0]=i/2; val[1]=j/2;
				if (cal()) { print(); return; }
			}
	printf("Happy new year!\n");
}

int main() {
	while (scanf("%d%d%d%d", &k, &x, &n, &m)!=EOF) conduct();
	return 0;
}
