using namespace std;

#include <cstdio>
#include <algorithm>

#define REP(i, n) for (int i=0; i<(n); i++)
#define FOR(i, a, b) for (int i=(a); i<=(b); i++)
#define FORD(i, a, b) for (int i=(a); i>=(b); i--)

typedef long long ll;

void outp(int l, int c, int sc, int ea)
{
    char buf[l+1];
    int  i=0;

    fill(buf, buf+l, 'X');
    if (sc) buf[i++]='C';
    if (ea) buf[l-1]='A';
    while (c--) {
        buf[i]  ='A';
        buf[i+1]='C';
        i+=2;
    }
    buf[l]=0;
    printf("%s\n", buf);
}

int main()
{
    int sc[51];
    int ea[51];
    ll  dp[51];
    int k, x, n, m;
    scanf("%d%d%d%d", &k, &x, &n, &m);
    for (sc[1]=0; sc[1]<2;    sc[1]++)
    for (ea[1]=0; ea[1]<2;    ea[1]++)
    for (dp[1]=0; dp[1]*2<=n; dp[1]++) if (sc[1]+ea[1]+dp[1]*2<=n)
    for (sc[2]=0; sc[2]<2;    sc[2]++)
    for (ea[2]=0; ea[2]<2;    ea[2]++)
    for (dp[2]=0; dp[2]*2<=m; dp[2]++) if (sc[2]+ea[2]+dp[2]*2<=m) {
        FOR(i, 3, k) {
            dp[i]=dp[i-2]+dp[i-1]+(ea[i-2]&sc[i-1]);
            sc[i]=sc[i-2];
            ea[i]=ea[i-1];
        }
        if (dp[k]==x) {
            outp(n, dp[1], sc[1], ea[1]);
            outp(m, dp[2], sc[2], ea[2]);
            return 0;
        }
    }
    printf("Happy new year!\n");
    return 0;
}
