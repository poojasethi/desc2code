#include<iostream>
using namespace std;
long long res(int ac1, int ac2, bool f1, bool f2, bool e1, bool e2, int k) {
    long long dp[50];
    bool f[50], e[50];
    dp[0] = ac1;
    dp[1] = ac2;
    f[0]=f1;
    f[1]=f2;
    e[0]=e1;
    e[1]=e2;
    for(int i=2;i<k;i++) {
        dp[i] = dp[i-2] + dp[i-1] + (e[i-2] && f[i-1]);
        f[i] = f[i-2];
        e[i] = e[i-1];
    }
    return dp[k-1];
}
void writeres(int ac, bool f, bool e, int n) {
    if(f)
        cout << "C";
    for(int k=0;k<ac;k++)
        cout << "AC";
    for(int k=f + 2*ac; k < n-e;k++)
        cout << "X";
    if(e)
        cout << "A";
    cout << '\n';
}
int main() {
    int k, n, m;
    long long x;
    cin >> k >> x >> n >> m;
    for(int ef=0;ef<16;ef++)
        for(int i=0;i*2 + ((ef&1) != 0) + ((ef&2) != 0) <= n;i++)
            for(int j=0;j*2 + ((ef&4) != 0) + ((ef&8) != 0) <= m;j++)
                if(res(i, j, ((ef&1) != 0), ((ef&4) != 0), ((ef&2) != 0), ((ef&8) != 0), k) == x) {
                    writeres(i, ((ef&1) != 0), ((ef&2) != 0), n);
                    writeres(j, ((ef&4) != 0), ((ef&8) != 0), m);
                    return 0;
                }
    cout << "Happy new year!\n";
    return 0;
}
