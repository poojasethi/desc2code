#include <iostream>

using namespace std;

int main () {
  int k,x,n,m;
  cin >> k >> x >> n >> m;
  long long S1[51], S2[51];
  S1[1]=S2[2]=1, S1[2]=S2[1]=0;
  for (int i=3; i<=k; i++) 
    S1[i]=S1[i-1]+S1[i-2], S2[i]=S2[i-1]+S2[i-2];
  long long s1=S1[k], s2=S2[k], s12=s1, s21=s1-(k&1), s22=s1+s2-1-s12-s21;
  /*
  cout << "s1=" << s1 << endl;
  cout << "s2=" << s2 << endl;
  cout << "s12=" << s12 << endl;
  cout << "s21=" << s21 << endl;
  cout << "s22=" << s22 << endl;
  */
  for (int s12ac=0; s12ac<2; s12ac++)
    for (int s21ac=0; s21ac<2; s21ac++)
      for (int s22ac=0; s22ac<2; s22ac++)
	for (int i1=0; i1<=(n-s12ac-s21ac)/2; i1++) {
	  int i2=(x-s1*i1-s12ac*s12-s21ac*s21-s22ac*s22)/s2;
	  if (0<=i2 && 2*i2+(s12ac|s22ac)+(s21ac|s22ac)<=m 
	            && i1*s1+i2*s2+s12ac*s12+s21ac*s21+s22ac*s22==x) {
	    if (s21ac) cout << "C";
	    for (int i=0; i<i1; i++) cout << "AC";
	    for (int i=2*i1+s12ac+s21ac; i<n; i++) cout << "X";
	    if (s12ac) cout << "A";
	    cout << endl;
	    if (s12ac | s22ac) cout << "C";
	    for (int i=0; i<i2; i++) cout << "AC";
	    for (int i=2*i2+(s12ac|s22ac)+(s21ac|s22ac); i<m; i++) cout << "Y";
	    if (s21ac | s22ac) cout << "A";
	    cout << endl;
	    return 0;
	  };
	};

  cout << "Happy new year!" << endl;
  return 0;
}
