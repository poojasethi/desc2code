#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
#define ll long long
using namespace std;

int N,M,X,K;

int main()
{
	scanf("%d%d%d%d",&K,&X,&N,&M);
	for (int i=0; i+i<=N; i++)
		for (int j=0; j+j<=M; j++)
			for (int k=0; k<16; k++)
			{
				int a=k&1,b=(k>>1)&1,c=(k>>2)&1,d=(k>>3)&1; ll f1=i,f2=j,x;
				if (a+b+i*2>N||c+d+j*2>M) continue;
				for (int l=3,A=a,B=b,C=c,D=d; l<=K; l++) x=f1+f2+(B&&C),B=D,swap(A,C),f1=f2,f2=x;
				if (f2==X)
				{
					if (a) putchar('C');
					for (int l=1; l<=i; l++) printf("AC");
					for (int l=1; l<=N-2*i-a-b; l++) putchar('B');
					if (b) putchar('A');
					puts("");
					if (c) putchar('C');
					for (int l=1; l<=j; l++) printf("AC");
					for (int l=1; l<=M-2*j-c-d; l++) putchar('B');
					if (d) putchar('A');
					puts("");
					return 0;
				}
			}
	puts("Happy new year!");
}
