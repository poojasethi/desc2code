bool can(ll n,ll ac,ll a,ll c){
	
	if(ac*2+a+c<=n)

	else return 0;
}

void print(int n, int a, int c, int ac)
{	
	
	char s[n+1];
	
	s[n]='\0';
	
	ll p=0;
	
	if(a)
	s[n-1]='A';
	
	if(c)
	s[p++]='C';
	
	for(ll i=1;i<=ac;i++)
	{
		s[p++]='A';
		s[p++]='C';	
	}
	
	if(a)
	while(p<n-1)
	s[p++]='Z';
	
	else 
	while(p<n)
	s[p++]='Z';
	
	cout<<s<<"\n";
    
}
