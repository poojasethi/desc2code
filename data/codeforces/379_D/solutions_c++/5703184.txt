#include <iostream>
#include <string>

using namespace std;
#define int long long

int count (int k, int ac1, int ac2, bool s1, bool e1, bool s2, bool e2)
{
    if (k == 2) return ac2;
    return count (k-1, ac2, ac1 + ac2 + e1*s2, s2, e2, s1, e2);
}

string gen (int n, int ac, bool s, bool e)
{
    string res;
    res.resize(n);
    int i = 0;
    if (s) res[i++] = 'C';
    if (e) res[--n] = 'A';
    while (ac--) res[i++] = 'A', res[i++] = 'C';
    while (i < n) res[i++] = 'Z';
    return res;
}

int k, x, n, m;

bool solve(int ac1, int ac2, bool s1, bool e1, bool s2, bool  e2)
{
    if(ac1 * 2 + s1 + e1 > n || ac2 * 2 + s2 + e2 > m) return false;
    if(count (k, ac1, ac2, s1, e1, s2, e2) == x)
    {
        return true;
    }
    return false;
}

#undef int

int main(void)
{
    cin >> k >> x >> n >> m;
    for (int i1 = 0; i1 <= n; i1++)
    {
        for (int i2 = 0; i2 <= m; i2++)
        {
            for (int a = 0; a < (1<<4); a++)
            {
                if (solve(i1,i2,1&a,1&(a>>1),1&(a>>2),1&(a>>3)))
                {
                    cout << gen (n, i1, 1&a, 1&(a>>1)) << endl
                        << gen (m, i2, 1&(a>>2), 1&(a>>3));
                    return 0;
                }
            }
        }
    }
    cout << "Happy new year!";
    return 0;
}
