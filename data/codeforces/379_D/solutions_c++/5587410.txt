#include <iostream>
#include <stdio.h>

using namespace std;

bool computer(int k, int x, int a1, int c1, int s1, int a2, int c2, int s2)
{
    int s3, a3, c3;
    while(k--)
    {
        s3 = s1+s2+(a1&&c2);
        if(s3>x) return 0;
        a3 = a2;
        c3 = c1;

        s1 = s2;
        s2 = s3;
        a1 = a2;
        c1 = c2;
        a2 = a3;
        c2 = c3;
    }
    return (x==s3);
}

void Print(int n, int s, int a, int c)
{
    int k = n-(s*2+a+c);

    if(c) putchar('C');
    while(s--) printf("AC");
    while(k--) putchar('X');
    if(a) putchar('A');

}

int main()
{
    int a1, c1, a2, c2, m, n, k, x, s1, s2;
    cin>>k>>x>>n>>m;

    k-=2;
    for(a1=0; a1<2; a1++)
        for(c1=0; c1<2; c1++)
            for(s1=0; s1*2+a1+c1<=n; s1++)
                for(a2=0; a2<2; a2++)
                    for(c2=0; c2<2; c2++)
                        for(s2=0; s2*2+a2+c2<=m;s2++)
                        {
                            if(computer(k,x,a1,c1,s1,a2,c2,s2))
                            {
                                Print(n,s1,a1,c1);
                                printf("\n");
                                Print(m,s2,a2,c2);
                                return 0;
                            }
                        }

    printf("Happy new year!\n");
    return 0;
}
