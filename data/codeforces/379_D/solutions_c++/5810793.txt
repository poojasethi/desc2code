#include <iostream>
#include <cstdlib>
using namespace std;
typedef long long ll;
ll count(int k, ll ac, ll bc, bool as, bool ae, bool bs, bool be) {
    if (k==2) return bc;
    int f=0;
    if(ae==true && bs==true)f++;
    return count(k-1, bc, ac+bc+f, bs, be, as, be);
}
int k,x,n,m;
string make(int n, int c, bool s, bool e) {
    string r;
    r.resize(n);
    int i=0;
    if (s) r[i++] = 'C';
    if (e) r[--n] = 'A';
    while(c--) r[i++]='A', r[i++]='C';
    while(i<n) r[i++]='X';
    return r;
}
void test(ll ac, ll bc, bool as, bool ae, bool bs, bool be) {
    if (2*ac + as + ae > n) return;
    if (2*bc + bs + be > m) return;
    if (count(k, ac, bc, as, ae, bs, be) == x) {
        cout<<make(n,ac,as,ae)<<'\n'<<make(m,bc,bs,be)<<'\n';
        exit(0);
    }
}

int main() {
    cin>>k>>x>>n>>m;
    for(int i=0; 2*i<=n; ++i) {
        for(int j=0; 2*j<=m; ++j) {
            for(int a=0; a<16; ++a) {
                test(i, j, a&1, 1&(a>>1), 1&(a>>2), 1&(a>>3));
            }
        }
    }
    cout<<"Happy new year!\n";
}
