#include <bits/stdc++.h>
#define int long long
using namespace std;
int n,m,k;
int x;
int F[55],sC[55],fA[55];
string S1,S2;
int find(int C1,int n1,int A1,int C2,int n2,int A2){
    F[1]=n1;
    F[2]=n2;
    sC[1]=C1;
    sC[2]=C2;
    fA[1]=A1;
    fA[2]=A2;
    for(int i=3;i<=k;i++){
        F[i]=F[i-1]+F[i-2]+(fA[i-2]&&sC[i-1]);
        fA[i]=fA[i-1];
        sC[i]=sC[i-2];
    }
    return F[k];
}
void build(int C1,int n1,int A1,string& S,int n){
    int fr=0;
    int to=n;
    if(C1) S[fr++]='C';
    if(A1) S[--to]='A';
    for(int i=fr;i<to;i++){
        if(n1){
            n1--;
            S[i++]='A';
            S[i]='C';
        }else S[i]='B';
    }
}
main(){
    ios_base::sync_with_stdio(false);
    int need;
    cin>>k>>need>>n>>m;
    S1.resize(n);
    S2.resize(m);
    for(int s1A=0;s1A<2;s1A++)
        for(int s1C=0;s1C<2;s1C++)
        for(int s2A=0;s2A<2;s2A++)
        for(int s2C=0;s2C<2;s2C++)
        for(int s1=0;s1<=((n-s1A-s1C)>>1);s1++)
        for(int s2=0;s2<=((m-s2A-s2C)>>1);s2++)
    if(find(s1C,s1,s1A,s2C,s2,s2A)==need){
        build(s1C,s1,s1A,S1,n);
        build(s2C,s2,s2A,S2,m);
        cout<<S1<<endl<<S2<<endl;
        return 0;
    }
    cout<<"Happy new year!"<<endl;
    }
