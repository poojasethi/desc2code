#include "iostream"

using namespace std;

int k, x, n, m;

bool chk(int n, int x, int y) {
	int num = (y & 1) + ((y & 2) >> 1);
	if (x * 2 + num > n) return false;
	return true;
}

bool check(int a, int b, int c, int d) {
	long long n1 = a, n2 = c, n3;
	int s1 = b, s2 = d, s3;
	for (int i = 3; i <= k; i++) {
		n1 = n1 + n2 + ((s1 & 1) && (s2 & 2));
		s1 = (s1 & 2) + (s2 & 1);
		swap(s1, s2); swap(n1, n2);
		if (n2 > x) return false;
	}
	if (n2 == x) {
		int num = 0;
		if (b & 2) { cout << "C"; num = 1; } else num = 0;
		for (int i = 1; i <= a; i++) { cout << "AC"; num += 2; }
		for (int i = num + 1; i < n; i++) cout << "B";
		if (b & 1) cout << "A\n"; else if (num < n) cout << "B\n"; else cout << endl;
 		if (d & 2) { cout << "C"; num = 1; } else num = 0;
		for (int i = 1; i <= c; i++) { cout << "AC"; num += 2; }
		for (int i = num + 1; i < m; i++) cout << "B";
		if (d & 1) cout << "A\n"; else if (num < m) cout << "B\n"; else cout << endl;
 	}
	return n2 == x;
}

int main() {
	cin >> k >> x >> n >> m;
	for (int n1 = 0; n1 <= n / 2; n1++)
		for (int s1 = 0; s1 < 4; s1++) {
			if (!chk(n, n1, s1)) continue;
			for (int n2 = 0; n2 <= m / 2; n2++)
				for (int s2 = 0; s2 < 4; s2++) {
					if (!chk(m, n2, s2)) continue;
					if (check(n1, s1, n2, s2)) return 0;
				}
		}
	cout << "Happy new year!\n";
}
