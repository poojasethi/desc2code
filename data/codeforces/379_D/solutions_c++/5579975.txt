#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<set>
using namespace std;

int k,x,n,m,yes;
int a1,a2,a3;

int f(int x) {if (!x) return 0;return 1;}

void out(int i,int j,int t)
{
	if (t&1) putchar('C');
	for (int o=0;o<i;o++)	putchar('A'),putchar('C');
	for (int o=0;o<n-2*i-f(t&1)-f(t&2);o++) putchar('B');
	if (t&2) putchar('A');
	puts("");
	if (t&4) putchar('C');
	for (int o=0;o<j;o++)	putchar('A'),putchar('C');
	for (int o=0;o<m-2*j-f(t&4)-f(t&8);o++) putchar('B');
	if (t&8) putchar('A');
	puts("");
}

int main()
{
	scanf("%d%d%d%d",&k,&x,&n,&m);
	for (int i=0;i<=n/2;i++)
	for (int j=0;j<=m/2;j++)
	for (int t=0;t<16;t++)	
	{
		if (yes) break;
		if (f(t&1)+f(t&2)+i*2>n) continue;
		if (f(t&4)+f(t&8)+j*2>m) continue;
		a1=i,a2=j;
		bool ok=1;
		for (int o=0;o<k-2;o++)
		{
			a3=a1+a2;
			if (o==0)
			{
				if ((t&2)>0 && (t&4)>0) a3++;
			}
			else if (o%2==1)
			{
				if ((t&8)>0 && (t&1)>0) a3++;
			}
			else if ((t&8)>0 && (t&4)>0) a3++;
			a1=a2,a2=a3;
			if (a2>x) {ok=0;break;}
		}
		if (a2<x) ok=0;
		if (ok) {yes=1;out(i,j,t);break;}
	}
	if (!yes) puts("Happy new year!");
	return 0;
}
