int k,x,n,m;

void print(int len,int num,int st){
	
	len-=st&1;
	if((st>>1)&1) printf("C"), len--;
	while(num--) printf("AC"), len-=2;
	while(len--) printf("G");
	if(st&1) printf("A");
	puts("");
}

void check(int a,int b,int st){

	int d[4];
	for(int i=0;i<4;i++) d[i]=(st>>i)&1;
	if(2*a+d[0]+d[1]>n) return ;
	if(2*b+d[2]+d[3]>m) return ;
	ll p=a,q=b;
	for(int i=3;i<=k;i++){
		ll tmp=p+q+d[3]*d[0];
		p=q, q=tmp;
		d[0]=d[2];
		swap(d[3],d[1]);
		if(q>x) return ;
	}
	if(q==x){
		print(n,a,st);
		print(m,b,st>>2);
		exit(0);
	}
}

int main(){
	
	scanf("%d%d%d%d",&k,&x,&n,&m);
	for(int i=0;2*i<=n;i++)
		for(int j=0;2*j<=m;j++)
			for(int st=0;st<16;st++)
				check(i,j,st);
	puts("Happy new year!");
	return 0;
}
