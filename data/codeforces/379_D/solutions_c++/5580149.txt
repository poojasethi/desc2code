#include <bits/stdc++.h>
using namespace std;
typedef long long lint;
#ifdef ONLINE_JUDGE
	#define LLD "%I64d"
#else
	#define LLD "%lld"
#endif

int k, x, n, m;
int w[] = {0, 1, 1, 2};

bool gao(int k, int s1, int s2, int p, int q) {
	if (k == 0) {
		return s2 == x;
	}
	int now = s1 + s2;
	if ((p & 1) && (q & 2)) {
		now++;
	}
	if (now > x) {
		return false;
	}
	return gao(k - 1, s2, now, q, (q & 1) | (p & 2));
}

char a[111], b[111];

void print(int s1, int s2, int p, int q) {
	a[n] = b[m] = 0;
	int beg = 0, end = n;
	if (p & 2) {
		a[0] = 'C';
		beg = 1;
	}
	if (p & 1) {
		a[n - 1] = 'A';
		end--;
	}
	for (int i = 0; i < s1; i++) {
		a[beg + i * 2] = 'A';
		a[beg + i * 2 + 1] = 'C';
	}
	for (int j = beg + s1 * 2; j < end; j++) {
		a[j] = 'B';
	}
	beg = 0, end = m;
	if (q & 1) {
		b[m - 1] = 'A';
		end--;
	}
	if (q & 2) {
		b[0] = 'C';
		beg = 1;
	}
	for (int i = 0; i < s2; i++) {
		b[beg + i * 2] = 'A';
		b[beg + i * 2 + 1] = 'C';
	}
	for (int j = beg + s2 * 2; j < end; j++) {
		b[j] = 'B';
	}
	printf("%s\n%s\n", a, b);
}

int main() {
	scanf("%d%d%d%d", &k, &x, &n, &m);
	for (int p = 0; p < 4; p++) {
		for (int q = 0; q < 4; q++) {
			int nn = n - w[p];
			int mm = m - w[q];
			for (int i = 0; i * 2 <= nn; i++) {
				for (int j = 0; j * 2 <= mm; j++) {
					if (gao(k - 2, i, j, p, q)) {
						print(i, j, p, q);
						return 0;
					}
				}
			}
		}
	}
	printf("Happy new year!\n");
	return 0;
}
