#!/usr/bin/python

import sys

k, x, n, m = map(int, raw_input().split())

# C? (AC)* F* A?
# c   ac   f  a

def count(leftc, leftac, lefta, rightc, rightac, righta):
    s = [(leftc, leftac, lefta), (rightc, rightac, righta)]
    for i in range(2, k):
        leftc, leftac, lefta = s[-2]
        rightc, rightac, righta = s[-1]
        if lefta > 0 and rightc > 0:
            s.append((leftc, leftac + rightac + 1, righta))
        else:
            s.append((leftc, leftac + rightac, righta))

    #print s
    return s[-1][1] 

def output(leftc, leftac, lefta, rightc, rightac, righta):
    leftf = n - leftc - lefta - 2 * leftac
    rightf = m - rightc - righta - 2 * rightac
    print "C" * leftc + "AC" * leftac + "F" * leftf + "A" * lefta
    print "C" * rightc + "AC" * rightac + "F" * rightf + "A" * righta

for C in range(16):
    leftc = 1 if C & 1 else 0
    lefta = 1 if C & 2 else 0
    rightc = 1 if C & 4 else 0
    righta = 1 if C & 8 else 0
    for leftac in range((n - leftc - lefta + 2)/2):
        for rightac in range((m - rightc - righta + 2)/2):
            #output(leftc, leftac, lefta, rightc, rightac, righta)
            _x = count(leftc, leftac, lefta, rightc, rightac, righta)
            #print "count ", _x
            if _x == x:
                output(leftc, leftac, lefta, rightc, rightac, righta)
                sys.exit(0)

print "Happy new year!"
