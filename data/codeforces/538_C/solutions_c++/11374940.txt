#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
    int m, n;
    cin >> n >> m;
    int a, b, a0, b0, mx;
    cin >> a >> b;
    mx = b + a - 1;
    for (int i = 1; i < m; i++){
        a0 = a;
        b0 = b;
        cin >> a >> b;
        if (a - a0 < abs(b - b0)){
            cout << "IMPOSSIBLE";
            return 0;
        }
        mx = max((a - a0 + b + b0) / 2, mx);
    }
    cout << max(mx, b + n - a);
    return 0;
}
