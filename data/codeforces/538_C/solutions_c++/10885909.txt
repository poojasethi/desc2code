#include<bits/stdc++.h>

using namespace std;

int i,n,m,h[100006];
int a[100006];
int ans;

int main(){
	cin >> n >> m;

	for(i=1 ; i<=m ; i++){
		scanf("%d %d",a+i,h+i);

		if(i==1){
			ans = h[i]+a[i]-1;
			continue;
		}

		if(abs(h[i-1]-h[i]) > a[i]-a[i-1]){
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}

		ans = max(ans , (a[i]-a[i-1]-abs(h[i]-h[i-1]))/2 + max(h[i] , h[i-1]));
	}

	ans = max(ans,h[m]+n-a[m]);

	cout << ans << endl;
}
