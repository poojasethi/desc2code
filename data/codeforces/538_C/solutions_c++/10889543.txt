#include<bits/stdc++.h>
using namespace std;
int n,x,m,ans,h,e,f;
void input(){
	cin >>n >>m >>x >>h;
	return;
}
int main(){
	input();
	ans=h+x-1;
	for(int i=1;i<m;i++){
		cin >> e >> f;
		int an = abs(f-h);
		if(an>e-x){
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}
		ans=max(ans,max(f, h)+(e-x-an)/2);
		h=f;
		x=e;
	}
	ans=max(ans, h+(n-x));
	cout << ans << endl;
}
