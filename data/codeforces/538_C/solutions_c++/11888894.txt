#include <bits/stdc++.h>

using namespace std;

int n, m;

int main() {
	scanf("%d%d", &n, &m);
	int d1, h1, d2, h2, ans;
	scanf("%d%d", &d2, &h2);
	ans = h2 + d2 -1;
	for(int i= 1; i<m; i++) {
		d1 = d2;
		h1 = h2;
		scanf("%d%d", &d2, &h2);
		ans = max(ans, h2);
		int r = d2-d1 - abs(h2 - h1);
		if(r < 0) {
			puts("IMPOSSIBLE");
			return 0;
		}
		ans = max(ans, max(h1, h2)+r/2);
	}
	ans = max(ans, h2 + n-d2);
	printf("%d\n", ans);
	return 0;
}
