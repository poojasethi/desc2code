#include <bits/stdc++.h>
using namespace std;
int d[101010],h[101014];
int main()
{
    int n,m;
    scanf("%d %d",&n,&m);
    int res=0;
    for(int i=1;i<=m;i++) 
    {
		scanf("%d %d",&d[i],&h[i]);
    }
    for(int i=2;i<=m;i++)
    {
        if(d[i]-d[i-1]<abs(h[i]-h[i-1])) 
        {
			printf("IMPOSSIBLE\n"); 
			return 0;
		}
        res=max(res,(d[i]-d[i-1]+abs(h[i]-h[i-1]))/2+min(h[i],h[i-1]));
    }
    res=max(res,max(h[1]+d[1]-1,h[m]+n-d[m]));
    printf("%d\n",res);
    return 0;
}
