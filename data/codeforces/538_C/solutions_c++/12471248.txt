#include <stdio.h>
#include <algorithm>
#define FOR(i, to, from) for (int i=to; i<from; i++)
#define rii(n, m) scanf("%d%d", &n, &m)
#define ri(n) scanf("%d", &m)
using namespace std;

int main () {
	int n, m, d1, d2, h1, h2, mp;
	rii(n, m);
	rii(d1, h1);
	mp=h1+d1-1;
	FOR(i, 1, m) {
		rii(d2, h2);		
		if (abs(h2-h1)>d2-d1) {
			printf("IMPOSSIBLE\n");
			return 0;
		}
		mp=max(mp, (h1+h2+d2-d1)/2);
		d1=d2; h1=h2;
	}
	
	mp=max(mp, h1+(n-d1));
	printf("%d\n", mp);
}
