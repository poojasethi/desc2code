#include <bits/stdc++.h>
#define ll long long int

using namespace std;

ll d[100010],h[100010];

int main()
{
	ll n,m,maxi,f=0;
	cin>>n>>m;
	for(int i=0;i<m;i++)
	{
		cin>>d[i]>>h[i];		
	}
	maxi=h[0]+d[0]-1;
	for(int i=1;i<m;i++)
	{
		if(d[i]-d[i-1]<abs(h[i]-h[i-1]))
		{
			f=1;
			break;
		}
		else
			maxi=max(max(h[i],h[i-1])+((d[i]-d[i-1])-abs(h[i]-h[i-1]))/2,maxi);
	}
	maxi=max(h[m-1]+n-d[m-1],maxi);
	if(f)
		cout<<"IMPOSSIBLE\n";
	else
		cout<<maxi<<endl;
	return 0;
}
