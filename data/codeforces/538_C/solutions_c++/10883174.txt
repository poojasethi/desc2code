#include <bits/stdc++.h>
using namespace std;

int main(){
	int D,N,d,h,dd,hh;
	cin >> D >> N;
	vector<pair<int,int> >v;
	cin >> d >> h;
	int ma=h+d-1;
	N--;
	while(N--){
		cin >> dd >> hh;
		if (dd-d < abs(hh-h)){
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}else{
			int tmp = (dd-d)-abs(hh-h);
			ma = max(ma,max(h,hh)+tmp/2);
		}
		d=dd;
		h=hh;
	}
	ma = max(ma,h+D-d);
	cout << ma << endl;
	return 0;
}
