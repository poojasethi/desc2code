#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
int n,m;
cin >> n >> m;

int mx=-1;

int a,b,c,d;
cin >> a>>b;
mx=max(mx,b+a-1);
//cout<<mx<<endl;
for(int i=1;i<m;i++)
{
cin >> c>> d;
if(abs(d-b)>c-a)
{
cout<<"IMPOSSIBLE";
return 0;
}
int k=c-a-1;
k=k-abs(b-d);
int g=max(d,b);
mx=max(mx,g+(k+1)/2);
//cout<<mx<<endl;

a=c;
b=d;
}
mx=max(mx,b+n-a);
cout<<mx;

return 0;
}
