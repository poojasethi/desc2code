#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    int d[m+5],h[m+5];
    for(int i=0;i<m;i++)   { scanf("%d%d",d+i,h+i);}
    int dd=1;int hh=h[0]+d[0]-1;
    int ans=0;
    for(int i=0;i<m;i++)
    {
        if(abs(hh-h[i])>abs(d[i]-dd)) {cout<<"IMPOSSIBLE";return 0;}
        ans=max(ans,max(hh,h[i])+(d[i]-dd-abs(h[i]-hh))/2);
        dd=d[i];hh=h[i];
    }
    ans=max(ans,h[m-1]+n-d[m-1]);
    cout<<ans;
    return 0;
}
