#include <bits/stdc++.h>

#define KILL {cout << "IMPOSSIBLE" << endl; return 0;}

using namespace std;

const int maxl = 1000 * 100 + 5;

int n, m, d, h;

int ans;

int main (){
	cin >> n >> m >> d >> h;

	ans = h + d - 1;
	for ( int i = 1 ; i < m ; i++ ){
		int a, b;
		cin >> a >> b;

		if ( (a - d) < abs ( h - b ) )
			KILL;

		ans = max ( ans, max ( h, b ) + (a - d - abs ( h - b )) / 2 );
		d = a;
		h = b;
	}
	ans = max ( ans, n - d + h );

	cout << ans << endl;
}
