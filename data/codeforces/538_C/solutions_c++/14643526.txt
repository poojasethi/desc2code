# include <iostream>

using namespace std;

long long a,c,n,m,x,y,mx,a1,c1;

int main()
{
	cin>>n>>m;
	for(int i=0;i<m;i++)
	{
		cin>>x>>y;
		if(i==0){
		a=x;a1=x;c=y;c1=y;mx=y;continue;}
		if(x-a<y-c || x-a<c-y)
		{cout<<"IMPOSSIBLE"; return 0;}
		if(((x+y-1)-a+c+1)/2>mx)
		mx=((x+y-1)-a+c+1)/2;
		a=x;c=y;
	}
	
	if(mx<n-x+y)
	mx=n-x+y;
	
	if(mx<a1+c1-1)
	mx=a1+c1-1;
	cout<<mx;
	return 0;
}
