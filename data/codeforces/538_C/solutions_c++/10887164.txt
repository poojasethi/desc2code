#include<bits/stdc++.h>

using namespace std;

int n,m,i,mx=0,qwe;
int D[100005],H[100005];

int main(){

	cin>>n>>m;

	//THISS ISSS asdafdsf

	for(i=1 ; i<=m ; i++){
		scanf("%d %d",&D[i],&H[i]);
		if(i!=1){
			if(abs(H[i]-H[i-1])>D[i]-D[i-1])
				return cout<<"IMPOSSIBLE",0;
		}
		mx=max(mx,H[i]);
	}
	
	mx=max(mx,D[1]-1+H[1]);

	mx=max(mx,n-D[m]+H[m]);

	for(i=2 ; i<=m ; i++)
		mx=max(mx,((D[i]-D[i-1])-(abs(H[i]-H[i-1])))/2+max(H[i],H[i-1]));

	//ANSWER

	qwe=mx;

	cout<<qwe;


	return 0;
}
