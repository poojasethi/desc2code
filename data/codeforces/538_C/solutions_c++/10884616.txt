#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,m,d,ans,u1,v1,u2,v2;
	cin>>n>>m>>u2>>v2;
	ans=max(0ll,u2+v2-1);
	while(m-->1)
	{
		u1=u2;
		v1=v2;
		scanf("%I64d%I64d",&u2,&v2);
		if(u2-u1<abs(v2-v1))
		{
			puts("IMPOSSIBLE");
			return 0;
		}
		else
		{
			d=u2-u1-abs(v2-v1);
			ans=max(ans,max(v1,v2)+(d/2));
		}
	}
	ans=max(ans,v2+n-u2);
	cout<<ans;
	return 0;
}
