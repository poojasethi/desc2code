#include <cstdlib>
#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
	int n, m;
	scanf("%d %d", &n, &m);
	
	int cd, ch;
	scanf("%d %d", &cd, &ch);
	int ans = ch+cd-1;
	for(int i=1; i<m; i++){
		int nd, nh;
		scanf("%d %d", &nd, &nh);
		int df = (nd-cd)-abs(nh-ch);
		if(df < 0){
			printf("IMPOSSIBLE\n");
			return 0;
		}
		ans = max(ans, max(nh, ch)+df/2);
		cd = nd;
		ch = nh;
	}
	printf("%d\n", max(ans, ch+(n-cd)));

	return 0;
}
