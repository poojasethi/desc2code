#define N (1<<17)
#include <bits/stdc++.h>

using namespace std;

int n,m,d[N],p[N];

int main()
{
	cin>>n>>m;
	for(int i=1;i<=m;i++)
		scanf("%d%d",d+i,p+i);

	int res=0;
	for(int i=2;i<=m;i++)
	{
		if(abs(p[i]-p[i-1])>d[i]-d[i-1])
		{
			puts("IMPOSSIBLE");
			return 0;
		}
		res=max(res,(d[i]-d[i-1]+p[i]-p[i-1])/2+p[i-1]);
	}
	res=max(res,max(p[1]+d[1]-1,p[m]+n-d[m]));
	cout<<res<<endl;
}
