#include<bits/stdc++.h>
using namespace std;

int main(){
	int n, m;
	cin >> n >> m;
	int res=0;
	int x, h;
	cin >> x >> h;
	res=h+x-1;
	for(int i=1;i<m;i++){
		int nx, nh;
		cin >> nx >> nh;
		if(abs(nh-h)>nx-x){
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}
		res=max(res, max(nh, h)+((nx-x)-(abs(nh-h)))/2);
		h=nh, x=nx;
	}
	res=max(res, h+(n-x));
	cout << res << endl;
}
