#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
	long long n, m, d, h;
	scanf("%I64d%I64d%I64d%I64d", &n, &m, &d, &h);
	long long ans = h + d - 1;
	long long pd = d, ph = h;
	for (int i = 1; i < m; ++i)
	{
		scanf("%I64d%I64d", &d, &h);
		if (abs(d - pd) < abs(h - ph)) {
			printf("IMPOSSIBLE\n");
			return 0;
		}
		ans = max(ans, ((d - pd) - abs(h - ph)) / 2 + max(h, ph));
		pd = d;
		ph = h;
	}
	ans = max(ans, h + n - pd);
	printf("%I64d\n", ans);
	return 0;
}
