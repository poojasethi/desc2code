#include<iostream>
#include<vector>
#include<cmath>

using namespace std;

int main(){
	int n, m, f ,g, h,d, maxe = -10000000;
	cin >> n >> m;
	cin >>h >>d;
	maxe = max(maxe , h - 1 + d);
	for(int i = 1; i < m; i++)	
		{
			cin >> f >>g ;
			if(abs(f - h) < abs(g - d))
				{
					cout << "IMPOSSIBLE";
					return 0;
				}
			int s = f - h;
			s -= abs(g - d);
			maxe = max(maxe , max(g , d) + s / 2);
			h = f; d = g;
		}
	maxe = max(maxe , n - h + d);
	cout << maxe;
}
