#include<cstdio>
#include<cstdlib>
int main(){
int n,m,x,y;
    scanf("%d %d",&n,&m);
    scanf("%d %d",&x,&y);
    int lastx=x,lasty=y;
    int ans=x+y-1;
for(int i=1;i<m;i++){
    scanf("%d %d",&x,&y);
    if(abs(y-lasty)>x-lastx){
        printf("IMPOSSIBLE\n");
        return 0;
        }
        int t=(lasty+y+x-lastx)/2;
        if(t>ans)ans=t;
        lastx=x;
        lasty=y;
}
int t =lasty+(n-lastx);
if(t>ans) ans=t;
printf("%d\n",ans);
return 0;
}
