#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,h,d,x,maxi,y;
	ios_base::sync_with_stdio(false);
	cin>>n>>m;
	m--;
	cin>>d>>h;
	maxi=h+d-1;
	x=d;
	y=h;
	while(m--)
	{
		cin>>d>>h;
		if((d-x)<abs(h-y))
		{
			cout<<"IMPOSSIBLE"<<endl;
			return 0;
		}
		if(h<=y)
			maxi=max(maxi,y+(d-x-abs(h-y))/2);
		else
			maxi=max(maxi,h+(d-x-abs(h-y))/2);	
		x=d;
		y=h;	
	}
	maxi=max(maxi,y+n-x);
	cout<<maxi<<endl;
	return 0;
}
