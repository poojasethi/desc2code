#include<bits/stdc++.h>
using namespace std;
int main(){
   int n,m,a,b,p,ph;
   cin >> n >> m;
   cin >> p >> ph;
   int g = ph + p-1;;

   while(--m){
      cin >> a >> b;
      if(abs(b-ph) > abs(p-a)){
         cout << "IMPOSSIBLE" << "\n";
         return 0;
      }
      int diff = abs(p-a) - abs(b-ph);
      g = max(g,max(ph,b) + diff/2);
      p = a,ph=b;
   }
   g = max(g,ph + n-p);
   cout << g << "\n";
}
