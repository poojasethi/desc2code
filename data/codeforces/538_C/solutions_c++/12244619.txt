#include<bits/stdc++.h>
using namespace std;
long long int n , m , ar[100010] , br[100010] , c , a , b , d;
int main()
{
     cin >> n >> m;
     for(int i = 0 ; i <m ; i++)
          cin >> ar[i] >> br[i];
     c =  br[0]+ar[0]-1;
     for(int i = 1 ; i < m ; i++)
     {
         if(abs(br[i]-br[i-1]) > ar[i]-ar[i-1])
         {cout <<"IMPOSSIBLE";return 0;}
         c = max(c , (ar[i]-ar[i-1]+br[i]+br[i-1])/2);
     }
    c = max(c ,br[m-1]+n-ar[m-1]);
    cout << c;  
}
