# include <stdio.h>
# include <algorithm>

using namespace std;

int a,b,h,d,x,y,mx;

int c[100009][2];

int main()
{
	scanf("%d %d",&a,&b);
	
	for(int i=0;i<b;i++)
	{
		scanf("%d %d",&c[i][0],&c[i][1]);
		
		if(i!=0)
		{
			d=c[i][0]-c[i-1][0];
			h=abs(c[i][1]-c[i-1][1]);
//			printf("%d %d",d,h);
			if(h>d){printf("IMPOSSIBLE");return 0;}
			d=d-h;
			mx=max(mx,d/2+max(c[i][1],c[i-1][1]));
		}
	}
	
	mx=max(mx,a-c[b-1][0]+c[b-1][1]);
	mx=max(mx,c[0][0]+c[0][1]-1);
	
	printf("%d",mx);
}
