#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
const int N = 1e5 + 5;

int n, m, ans;
int h[N], d[N];

int main() {
	int i;
	scanf("%d%d", &n, &m);
	for (i = 1; i <= m; ++i) scanf("%d%d", d + i, h + i);
	ans = max(d[1] + h[1] - 1, n - d[m] + h[m]);
	for (i = 1; i < m; ++i) {
		if (d[i + 1] - d[i] < abs(h[i + 1] - h[i])) {
			puts("IMPOSSIBLE");
			return 0;
		}
		ans = max(ans, (h[i] + h[i + 1] + d[i + 1] - d[i]) >> 1);
	}
	printf("%d\n", ans);
	return 0;
}
