#include<bits/stdc++.h>
using namespace std;

int main() {
	int n,m; cin>>n>>m;
	int a,x; cin>>a>>x;
	int A=max(A,x+a-1);
	for (int i=1; i<m; i++) {
		int b,y; cin>>b>>y;
		if(abs(a-b)<abs(x-y)) {
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}
		A=max(A, (b-a+x+y)/2);
		a=b, x=y;
	}
	A=max(A,x+n-a);
	cout << A << endl;
	return 0;
}
