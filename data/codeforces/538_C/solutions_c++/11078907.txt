#include<iostream>
#include<string>
#include<cmath>
using namespace std;
int main(){
long int n,m,m1,s,d,pd,h,ph,i,ma;

cin>>n>>m;
cin>>d>>h;
if(d)
    ma=h+d-1;
else
    ma=0;
pd=d,ph=h;
for(i=1;i<m;i++){
    cin>>d>>h;
    s=d-pd;
    s-=abs(h-ph);
    if(s<0){
        cout<<"IMPOSSIBLE";
        return 0;
    }
    m1=max(h,ph)+s/2;
    ma=max(m1,ma);
    pd=d,ph=h;
}

ma=max(ma,ph+n-pd);


cout<<ma<<endl;
return 0;
}
