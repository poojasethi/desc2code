#include<bits/stdc++.h>
using namespace std;
int n,m,a,b,i,d,h,ma=0;
main()
{
    scanf("%d%d",&n,&m);
    scanf("%d%d",&a,&b);
    ma=b+a-1;
    for (i=1;i<m;i++)
    {
        scanf("%d%d",&d,&h);
        if (d-a<abs(h-b))
            return printf("IMPOSSIBLE"),0;
        ma=max(ma,(b+h+d-a)/2);
        a=d,b=h;
    }
     ma=max(b+n-a,ma);
    printf("%d",ma);
}
