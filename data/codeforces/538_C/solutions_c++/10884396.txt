#include <cstdio>
#include <algorithm>
using namespace std;

int n,m,ans,d[100009],h[100009];
int t,s;
int l,r;
bool f;

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;++i)
	{
		scanf("%d%d",d+i,h+i);
		ans=max(ans,h[i]);
	}
	ans=max(ans,h[1]+d[1]-1);	
	ans=max(ans,h[m]+n-d[m]);
	for (int i=1;i<m;++i)
	{
		l=min(h[i],h[i+1]);
		r=max(h[i],h[i+1]);
		t=d[i+1]-d[i]-1;
		t-=r-l;
		if (t<-1) f=true;
		ans=max(ans,r+(t+1)/2);
	}
	if (f) puts("IMPOSSIBLE");
	else printf("%d\n",ans);
	return 0;
}
