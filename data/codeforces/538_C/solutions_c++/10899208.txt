#include <bits/stdc++.h>
#define ll long long

using namespace std;

ll n,ma,m,a,b,ao,bo;

int main()
{ cin>>n>>m; 
  cin>>a>>b; ma=b+(a-1);
  for(int i=1;i<m;i++){ao=a;bo=b; cin>>a>>b; if(a-ao<abs(b-bo)) {cout<<"IMPOSSIBLE"; return 0;}
                       ma=max((a-ao+b+bo)/2,ma);} 
  
  cout<<max(b+n-a,ma)<<endl;
}
