#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    int n,m,a,b,maxx,p_b,p_a;
    cin>>n>>m;
    cin>>a>>b;
    maxx=b+(a-1);
    for(int i=1;i<=m-1;i++){
        p_a=a;
        p_b=b;
        cin>>a>>b;
        if(abs(a-p_a)<abs(b-p_b)){
            cout<<"IMPOSSIBLE";
            return 0;
        }
        maxx=max(maxx,(a-p_a+b+p_b)/2);
    }
    cout<<max(maxx,n-a+b);


    return 0;
}
