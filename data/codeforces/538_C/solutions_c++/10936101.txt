#include <iostream>
#include <cmath>

using namespace std;

int main() {
	int n,m,x,y;
	
	cin >> n >> m;
	cin >> x >> y;
	
	int lastx = x,lasty = y;
	
	int ans = x + y - 1;
	
	for (int i = 1; i < m; ++i) {
		cin >> x >> y;
		
		if(abs(y - lasty) > x - lastx) {
			cout << "IMPOSSIBLE";
			return 0;
		}
		
		int t = (lasty + y + x - lastx) / 2;
		if(t > ans) ans = t;
		lastx = x;
		lasty = y;
		
	}
	
	int t = lasty + (n - lastx);
	
	if(t > ans) ans = t;
	
	cout << ans;
	
	return 0;
}
