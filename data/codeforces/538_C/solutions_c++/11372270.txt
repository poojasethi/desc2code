#include <stdio.h>
#define max(a,b) (a>b?a:b)
#define abs(a) (a<0?-(a):a)
int n,m,a,x,b,y,A=0;
main() {
	//freopen("in.txt","r",stdin);
	scanf("%d%d",&n,&m);
	scanf("%d%d",&a,&x);
	A=max(A,x+a-1);
	for (int i=1;i<m;i++) {
		scanf("%d%d",&b,&y);
		if (abs(a-b)<abs(x-y)) {
			puts("IMPOSSIBLE");
			return 0;
		}
		A=max(A,(b-a+x+y)/2);
		a=b;x=y;
	}
	A=max(A,x+n-a);
	printf("%d\n",A);
}
