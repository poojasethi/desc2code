#include <cstdio>
#include <cstdlib>
#include <algorithm>
#define REP(i,n) for (int i=1;i<=n;++i)
#define LL long long
using namespace std;

int n,m;
LL x,y,sx,sy,ans;

int main() {
	scanf("%d%d",&n,&m);
	REP(i,m) {
		scanf("%I64d%I64d",&x,&y);
		if (i==1) ans=y+x-1;
		else {
			LL tmp1=abs(x-sx);
			LL tmp2=abs(y-sy);
			if (tmp1<tmp2) {puts("IMPOSSIBLE");return 0;}
			else ans=max(ans,max(y,sy)+(tmp1-tmp2)/2);
		}
		sx=x; sy=y;
	}
	ans=max(ans,y+n-x);
	printf("%I64d\n",ans);
	return 0;
}
