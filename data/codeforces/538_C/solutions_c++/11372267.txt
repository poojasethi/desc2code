#include <stdio.h>
#include <string.h>
#include <algorithm>

using namespace std;

int main(){
    int n,m,d,h,mh,sd,sh;
    scanf("%d %d", &n, &m);
    scanf("%d %d", &d, &h);
    mh = h+d-1;
    for(int i=1; i<m; i++){
        sd = d; sh = h;
        scanf("%d %d", &d, &h);
        if(abs(h-sh) > d-sd) {puts("IMPOSSIBLE"); return 0;}
        mh = max(mh, (d-sd+h+sh)/2);
    }
    mh = max(mh, h+n-d);
    printf("%d\n", mh);
}
