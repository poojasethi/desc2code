n, m = map(int, raw_input().split())
pd, ph = map(int, raw_input().split())
ans = ph+pd-1
for i in range(m-1):
    d, h = map(int, raw_input().split())
    xd = d-pd
    xh = abs(h-ph)
    if xh > xd:
        print 'IMPOSSIBLE'
        exit()
    tmp = (d-pd+h+ph)/2
    ans = max(ans, tmp)
    pd = d
    ph = h
ans = max(ans, ph+n-pd)
print ans
