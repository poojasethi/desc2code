#!/usr/bin/env python
# -*- coding: utf-8 -*-

def work():
    n, m = map(int, raw_input().split())
    days = [map(int, raw_input().split()) for i in range(m)]
    days.sort()

    ans = days[0][0] - 1 + days[0][1]
    ans = max(ans, n - days[-1][0] + days[-1][1])
    imp = False
    for i in range(1, m):
        if days[i][0] - days[i-1][0] < abs(days[i-1][1] - days[i][1]):
            break
        ans = max(ans, int((days[i][0] - days[i-1][0] + days[i][1] + days[i-1][1])/2))
    else:
        imp = True
    print ans if imp else "IMPOSSIBLE"
    #if imp:
        #print ans
    #else:
        #print "IMPOSSIBLE"

if __name__ == '__main__':
    work()
