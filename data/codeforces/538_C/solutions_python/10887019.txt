n,m=map(int,raw_input().split())
d=[0 for i in range(m)]
h=[0 for i in range(m)]
for i in range(m):
    d[i],h[i]=map(int,raw_input().split())
hmax=max(d[0]+h[0]-1,n-d[m-1]+h[m-1])
b=True
for i in range(m-1):
    if abs(h[i+1]-h[i])>d[i+1]-d[i]:
        print "IMPOSSIBLE"
        b=False
        break
    if max(h[i+1],h[i])+(d[i+1]-d[i]-abs(h[i+1]-h[i]))/2>hmax:
        hmax=max(h[i+1],h[i])+(d[i+1]-d[i]-abs(h[i+1]-h[i]))/2
if b:
    print hmax
