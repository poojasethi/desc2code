n, m = map(int, raw_input().split())
pd = ph = -1
ans = 0
for _ in xrange(m):
    d, h = map(int, raw_input().split())
    if pd != -1:
        if abs(ph - h) > abs(pd - d):
            print "IMPOSSIBLE"
            exit(0)
        ans = max(ans, max(h, ph) + (abs(pd - d) - abs(ph - h)) / 2)
    else:
        ans = max(ans, h + d - 1)
    pd, ph = d, h
print max(ans, ph + (n - pd))
