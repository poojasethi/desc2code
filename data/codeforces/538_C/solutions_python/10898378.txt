n, m = map(int, raw_input().split())
ds = []
import math

for i in range(m):
    ds.append(map(int,raw_input().split()))

mx = max(ds[0][1] + max(ds[0][0] - 1, 0), ds[-1][1] + (n - ds[-1][0]))
f = True

for i in range(1, m):
    t = abs(ds[i-1][1] - ds[i][1])
    dt = ds[i][0] - ds[i-1][0]
    if t > dt:
        f = False
    b = dt - t
    if b % 2:
        b -= 1
    mx = max(mx, max(ds[i][1], ds[i-1][1]) + b / 2, ds[i][1])

if f:
    print int(mx)
else:
    print "IMPOSSIBLE"
