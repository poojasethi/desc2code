import sys

n, m = map(int, raw_input().split())
heights = []
for _ in range(m):
  day, h = map(int, raw_input().split())
  day -= 1
  heights.append((day, h))

ans = max(heights[0][1] + heights[0][0],
          heights[-1][1] + (n - heights[-1][0] - 1))
for (d1, h1), (d2, h2) in zip(heights, heights[1:]):
  if abs(h1 - h2) > d2-d1:
    print 'IMPOSSIBLE'
    sys.exit(0)
  else:
    lax = (d2 - d1) - abs(h1-h2)
    ans = max(ans, max(h1, h2) + lax/2)
print ans
