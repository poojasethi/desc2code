#!/usr/bin/python

import sys

def Ni(): return tuple(map(int, raw_input().split()))

n, m = Ni()
dh = [Ni() for i in range(m)]

lastDay, lastHeight = dh[0]
maxHeight = lastHeight + lastDay - 1

for d, h in dh[1:]:
    totalDays = d - lastDay
    if h > lastHeight:
        # climb up
        climbingDays = h - lastHeight
        if climbingDays > totalDays:
            print "IMPOSSIBLE"
            sys.exit(0)
        daysLeft = totalDays - climbingDays
        maxHeight = max(maxHeight, h + (daysLeft - daysLeft % 2) / 2)
    else:
        # climb down
        climbingDays = lastHeight - h
        if climbingDays > totalDays:
            print "IMPOSSIBLE"
            sys.exit(0)
        daysLeft = totalDays - climbingDays
        maxHeight = max(maxHeight, lastHeight + (daysLeft - daysLeft % 2) / 2)

    lastDay, lastHeight = d, h
       
maxHeight = max(maxHeight, lastHeight + (n - lastDay))

print maxHeight
