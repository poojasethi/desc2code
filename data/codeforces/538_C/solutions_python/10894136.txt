n,m=map(int,raw_input().split())
L=[]
for _ in xrange(m):
	L.append(map(int,raw_input().split()))
for x in xrange(m-1):
	if(abs(L[x+1][0] - L[x][0])>=abs(L[x+1][1]-L[x][1])):
		continue;
	print "IMPOSSIBLE"
	exit()
m1=L[0][0]-1 + L[0][1]
for x in xrange(m-1):
	a,b=max(L[x][1],L[x+1][1]),min(L[x][1],L[x+1][1])
	m1=max(m1,(L[x+1][0]-L[x][0] -b+a)/2 + b)
m2=n-L[m-1][0] + L[m-1][1]
print max(m1,m2)
