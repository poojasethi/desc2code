n, m  = map(int, raw_input().split())
pd, ph  = map(int, raw_input().split())
mh = ph + pd - 1
for i in range(m - 1):
	d, h  = map(int, raw_input().split())
	dh, dd = abs(ph - h), d - pd
	if dh > dd: print "IMPOSSIBLE", exit()
	mh = max(mh, (dd - dh) / 2 + max(ph, h))
	pd, ph = d, h
mh = max(mh, ph + n - pd)
print mh
