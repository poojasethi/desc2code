n, m = map(int, raw_input().split(" "))

d, h = map(int, raw_input().split(" "))

max_height = d + h - 1

prev_day, prev_height = d, h

bad = False

for i in xrange(m - 1):
	d, h = map(int, raw_input().split(" "))
	diff = (d - prev_day) - abs(h - prev_height)
	if diff < 0:
		bad = True
		break
	else:
		this_round = diff / 2 + max(prev_height, h)
		if this_round > max_height:
			max_height = this_round
		prev_day, prev_height = d, h

last = prev_height + (n - prev_day)

if (last > max_height):
	max_height = last

if bad:
	print 'IMPOSSIBLE'
else:
	print max_height
