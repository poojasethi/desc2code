R=lambda:map(int,raw_input().split())
n,k=R()
h=[R() for i in range(k)]
if any((abs(h[i][1]-h[i-1][1])>abs(h[i][0]-h[i-1][0]) for i in range(1,k))):
	print 'IMPOSSIBLE'
	exit(0)
def check(m):
	if h[0][1]>=m: 
		return 1
	for i in range(1,k):
		if h[i][1]>=m or 2*m-h[i-1][1]-h[i][1]<=h[i][0]-h[i-1][0]: 
			return 1
	return 0
l,r,m=0,2*10**8,0
while l<=r:
	m=l+r>>1
	if check(m): l=m+1
	else: r=m-1
print max(r,h[0][0]-1+h[0][1],h[-1][1]+n-h[-1][0])
