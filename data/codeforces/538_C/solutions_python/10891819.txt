from collections import namedtuple

Note = namedtuple('Note', 'day height')

[n, m] = map(int,raw_input().split(' '))
notes = []

for i in range(0,m):
	values = map(int,raw_input().split(' '))
	note = Note(values[0],values[1])
	notes.append(note)

# print notes

max_height = notes[0].height
max_start = notes[0].height + notes[0].day - 1
max_end = notes[m-1].height + n - notes[m-1].day 

# print max_start, max_end

last_note = notes[0]

for i in range(1,m):
	note = notes[i]
	if abs(note.height - last_note.height) > abs(note.day - last_note.day):
		max_height = 'IMPOSSIBLE'
		break
	else:
		max_height = max(max_height, (note.height + last_note.height + note.day - last_note.day) / 2)
		# print 'MAx', last_note, note, max_height
	last_note = note

if max_height == 'IMPOSSIBLE':
	print max_height
else:
	print max(max_height,max_start,max_end)
