import sys

n,m=map(int,raw_input().split())
hi=-1
lo=1
lo,hi=map(int,raw_input().split())
lh=hi
hi+=(lo-1)
for i in range(2,m+1):
    a,h=map(int,raw_input().split())
    if (abs(h-lh)>a-lo):
        print "IMPOSSIBLE"
        sys.exit(0)
    hmas=max(h,lh)
    lo+=(hmas-min(h,lh))
    hi=max(hi,hmas+(a-lo)/2)

    lo=a
    lh=h

if lo!=n:
    hi=max(hi,lh+n-lo)
print hi
