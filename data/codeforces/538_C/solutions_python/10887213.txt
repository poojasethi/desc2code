import sys

l1 = sys.stdin.readline().split()

n = int(l1[0])
m = int(l1[1])

s = []
for i in range(m):
  b = sys.stdin.readline().split()
  s.append((int(b[0]), int(b[1])))


max = 0
for i in range(m-1):
  a = s[i]
  b = s[i+1]

  if abs(a[1]-b[1]) > (b[0]-a[0]):
    print 'IMPOSSIBLE'
    sys.exit()

  y = (b[1]+b[0]+a[1]-a[0])/2

  if y > max:
    max = y


a = s[0]
t = a[1]+(a[0]-1)
if t > max:
  max = t

a = s[-1]
t = a[1]+(n-a[0])
if t > max:
  max = t

print max
