n, m = map(int, raw_input().split())

notes = [map(int, raw_input().split()) for _ in xrange(m)]
ind = 0
d1, h1 = notes[ind]
ind += 1

best = h1 + d1 - 1
hei = h1
day = d1
imp = False

while ind < len(notes):
    day2, next_he = notes[ind]
    ind += 1
    # see best possible increase/decrease
    if day2 - day < abs(next_he - hei):
        imp = True
        break
    # 1 inc, 1 dec, then # of days to go back down
    diff = abs(next_he - hei)
    best = max(best, max(hei, next_he) + (day2-day-diff) // 2)
    hei = next_he
    day = day2

# add last day
best = max(best, hei + n - day)

if imp:
    print 'IMPOSSIBLE'
else:
    print best
