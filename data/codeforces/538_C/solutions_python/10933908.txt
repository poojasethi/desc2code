n, m = map(int, raw_input().split() )
last = 0,0
mx = 0
for i in range(m):
    cur = map(int, raw_input().split() )
    if i == 0:
        last = cur
        mx = cur[1]+cur[0]-1
    if abs(cur[1]-last[1]) > abs(cur[0]-last[0]):
        print "IMPOSSIBLE"
        exit()
    mx = max(mx,last[1]+(cur[0]-last[0] - (last[1]-cur[1]))/2)
    mx = max(mx,cur[1])
    mx = max(mx,last[1])
    last = cur
mx = max(mx,n-last[0]+last[1])
#print last,n
print mx
