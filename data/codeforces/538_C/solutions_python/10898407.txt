
def f():
    n,m = map(int, raw_input().split())
    
    d1, h1= map(int, raw_input().split())
    d2, h2 = d1, h1
    maxh = h1+d1-1
    for i in xrange(m-1):
        d2, h2 = map(int, raw_input().split())
        if abs(h2-h1)>d2-d1: 
            print 'IMPOSSIBLE'
            return
        mh = h1 - d1 + (d1+d2+h2-h1)/2
        if maxh==None or maxh<mh:
            maxh = mh
        d1, h1 = d2, h2
    maxh =max(maxh, h2+n-d2)
    print maxh
f()
