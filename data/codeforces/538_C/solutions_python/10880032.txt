N, M = map(int, raw_input().split())
d, h = [], []
for loop in xrange(M):
    di, hi = map(int, raw_input().split())
    d.append(di); h.append(hi)

ans = max(h[0] + (d[0] - 1), h[-1] + (N - d[-1]))
for i in xrange(M - 1):
    if abs(h[i] - h[i + 1]) > d[i + 1] - d[i]:
        print "IMPOSSIBLE"
        break
    tmp = ((d[i + 1] - d[i]) - abs(h[i] - h[i + 1])) / 2 + max(h[i], h[i + 1])
    ans = max(ans, tmp)
else:
    print ans
