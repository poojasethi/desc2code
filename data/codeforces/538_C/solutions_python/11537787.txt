n, m = map(int, raw_input().split())
d1, hd1 = map(int, raw_input().split())
highest = hd1+(d1-1)
flag = 1
for i in xrange(1, m):
    d2, hd2 = map(int, raw_input().split())
    if abs(hd2-hd1)>d2-d1:
        print "IMPOSSIBLE"
        flag = 0
        break
    else:
        highest = max(highest, int((hd2+hd1+d2-d1)/2))
    d1=d2
    hd1=hd2
highest = max(highest, hd1+(n-d1))
if flag is 1:
