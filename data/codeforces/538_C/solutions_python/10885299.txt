import sys
n,m = sys.stdin.readline().split()
n,m = int(n), int(m)
h = [0] * m
t = [0] * m
for i in range (m):
  l,k = sys.stdin.readline().split()
  t[i],h[i] = int(l), int(k)
res = t[0] + h[0] - 1
f = True
for i in range (1, m):
  if abs (h[i] - h[i-1]) > t[i] - t[i-1]:
    f = False
  res = max (res, (h[i] + h[i-1] + t[i] - t[i-1])/2) 
res = max (res, h[m-1] + n - t[m-1])
if f:
  print (res)
else:
  print ('IMPOSSIBLE')
