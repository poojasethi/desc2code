
x = int(raw_input())

facts = []
import math

for i in range(1, int(math.floor(math.sqrt(x))) + 1):
    if x % i == 0:
        facts.append(i)
        if x / i != i:
            facts.append(x / i)
count = 0

for i in facts:
    for j in str(i):
        if j in str(x):
            count += 1
            break

print count
