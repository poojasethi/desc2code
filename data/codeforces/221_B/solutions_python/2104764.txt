x = int(raw_input())
t = set(list(str(x)))
f = 0
for i in range(1, int(x ** 0.5) + 1):
    if x % i == 0:
        r = set(list(str(i)))
        if len(r - t) < len(r):
            f += 1
        if i * i != x:
            r = set(list(str(x // i)))
            if len(r - t) < len(r):
                f += 1           
print(f)
