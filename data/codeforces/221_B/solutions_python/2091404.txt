#!/usr/bin/env python

X = int(raw_input())
s = set(map(int, str(X)))
def F(x):
    while x > 0:
        u = x % 10
        if u in s:
            return 1
        x /= 10
    return 0

i = 1
c = 0
while i * i <= X:
    if X % i == 0:
        c += F(i)
        if i * i < X:
            c += F(X / i)
    i += 1
print c,
