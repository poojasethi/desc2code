#coding=utf-8

from math import *

x = int(input())

ans = 0
#print sqrt(1000)

def check(x, d):
	s1 = "%d"%d
	s2 = "%d"%x
#	print d, s1, s2
	for i in s1:
		if i in s2:
			return True
	return False
for d in range(1, int(sqrt(x))+1):
	if x % d == 0:
		x_d = x / d
		if check(x, d):
			ans+= 1
		if (x_d != d):
			if check(x, x_d):
				ans+=1
print ans
