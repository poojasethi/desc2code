x = input()
divisors = []
y = [0]*10
ans = 0
for c in str(x):
    y[int(c)] = 1
for i in xrange(1,int(x**0.5)+1):
    if x % i == 0:
        divisors.append(str(x/i))
        divisors.append(str(i))
for d in list(set(divisors)):
    for d2 in d:
        if y[int(d2)]:
            ans += 1
            break
print ans
