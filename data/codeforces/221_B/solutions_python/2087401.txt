import math
x = int(raw_input())
n = int(math.sqrt(x)) + 1
ans = 0
for i in xrange(1, n):
    if x % i == 0:
        for j in str(i):
            if j in str(x):
                ans = ans + 1
                break
        if i <> x/i:
            for j in str(x/i):
                if j in str(x):
                    ans = ans + 1
                    break
print ans
