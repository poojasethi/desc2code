import sys, math

s = sys.stdin.readline()
x = int(s)
count = 0
for d in xrange(1, int(math.sqrt(x)) + 1):
    if x % d == 0:
        for c in str(d):
            if c in s:
                count += 1
                break
        if d != x / d:
            for c in str(x / d):
                if c in s:
                    count += 1
                    break
print count
