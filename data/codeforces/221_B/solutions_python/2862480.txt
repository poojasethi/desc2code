def check(a,b):
    a = str(a)
    b = str(b)
    for d in range(10):
        if a.find(str(d)) != -1 and b.find(str(d)) != -1:
            return True
    return False

x = int(raw_input())
ans = 0
for div in xrange(1,int(x**0.5)+1):
    if x%div == 0:
        if check(x,div):
            #print div
            ans += 1
        if div*div != x and check(x,x/div):
            #print x/div
            ans += 1
print ans
