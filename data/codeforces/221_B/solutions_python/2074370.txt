n = int(input())
ANS = 0

if n < 10:
  print 1
else:
  for x in range(1, int(n**0.5)+1):
    if n % x == 0:
      for q in range(10):
        if (str(q) in str(n)) and (str(q) in str(x)):
          ANS += 1
          break
      x = n / x
      if x**2 == n:
        continue
      for q in range(10):
        if (str(q) in str(n)) and (str(q) in str(x)):
          ANS += 1
          break
  print ANS
