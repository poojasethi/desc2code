def solve(x):
    s = set(str(x))
    def check(k,x):
        return 1 if not x%k and any(q in s for q in str(k)) else 0
    if x == 1:
        return 1
    c = 0
    i = 1
    while i*i <= x:
        c += check(i,x)
        if i != x/i: c+=check(x/i,x) 
        i += 1
    return c    
print solve(input())
