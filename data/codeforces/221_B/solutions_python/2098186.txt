#!/usr/bin/env python
# http://www.zhihua-lai.com/acm

n = int(raw_input())
ans = 0

def C(i, n):
    return any((ch for ch in str(i) if ch in n))
    
for i in xrange(1, n + 1):
    if i * i > n:
        break
    if n % i == 0:
        if C(i, str(n)):
            ans += 1
        if i * i != n:
            if C(n / i, str(n)):
                ans += 1
        
print ans
