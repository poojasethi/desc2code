from math import *
x = int(raw_input())
xdigits = [0] * 10
for digit in map(int, list(str(x))):
	xdigits[digit] += 1
def check(n):
	for digit in map(int, list(str(n))):
		if xdigits[digit] > 0:
			return True
	return False
d = 0
for i in xrange(1, int(sqrt(x)) + 1):
	if x % i == 0:
		if check(i):
			d += 1
		if not x / i == i:
			if check(x/i):
				d += 1
print d
