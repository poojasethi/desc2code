n, ans, i = input(), 0, 1
s = set(list(str(n)))
while i*i<n:
    if n%i!=0:
        i+=1
        continue
    if set(list(str(i))).intersection(s):
        ans+=1
    if set(list(str(n/i))).intersection(s):
        ans+=1
    
    i+=1
if i*i==n and set(list(str(i))).intersection(s):
    ans+=1
print ans
