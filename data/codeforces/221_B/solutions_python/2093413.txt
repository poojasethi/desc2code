def cal(n,li):
    while(n > 0):
        if(li[n % 10]):
            return 1
        n /= 10
    return 0

n = input()
li = [0 for i in range(20)]
a = n
sums = 0
while a > 0:
    li[a % 10] = 1
    a /= 10

i = 1
while i*i <= n:
    if n % i ==0:
        sums += cal(i,li)
        if i * i != n:
            sums += cal(n/i,li)
    i += 1
print sums
