x = input()
from math import sqrt

def solve(x):
    s = set(str(x))
    def check(k):
        return any(q in s for q in str(k))
    if x == 1:
        return 1
    c = 0
    i = 1
    while i*i <= x:
        if x%i == 0 and check(i):
            c += 1
        if x%(x/i) == 0 and check(x/i) and i != x/i: 
            c += 1
        i += 1
    return c    
    
print solve(x)
