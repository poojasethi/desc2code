import math


def check(d, _index):
    return any((ch for ch in str(d) if ch in _index))


if __name__ == '__main__':
    x = int(raw_input())
    _index = set(str(x))

    counter = 0
    to = int(math.sqrt(x) + 1)
    for d in xrange(1, to):
        if x % d == 0:
            if check(d, _index):
                counter += 1
            if check(x / d, _index) and d != x / d:
                counter += 1
    print counter
