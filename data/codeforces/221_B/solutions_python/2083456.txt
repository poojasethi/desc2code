def start():
    x = input()
    answer = 0
    y = str(x)
    for i in xrange(1,x+1):
        if i*i > x: break
        if x%i == 0:
            answer += int(True in map(y.__contains__, list(str(i))))
            if i != x/i:
                answer += int(True in map(y.__contains__, list(str(x/i))))
    print answer

start()
