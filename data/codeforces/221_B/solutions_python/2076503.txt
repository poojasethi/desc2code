def same_dig(a, b):
    s1 = str(a)
    s2 = str(b)
    for i in range(10):
        if str(i) in s1 and str(i) in s2:
            return 1
    return 0

d = int(raw_input())
i = 1
ans = 0
while i * i <= d:
    (x, y) = divmod(d, i)
    if y == 0:
       ans += same_dig(d, x)
       if x != i:
           ans += same_dig(d, i) 
    i += 1
print ans
