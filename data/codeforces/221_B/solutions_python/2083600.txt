from math import sqrt

x = input()

xd = str(x)
xds = set()
for c in xd:
    xds.add(c)

def check(i):
    for c in str(i):
        if c in xds:
            return True
    return False

count = 0 #for x itself

s = int(sqrt(x))

for i in xrange(1, s+1):
    if i*i == x:
        if check(i):
            count += 1
    elif x % i == 0:
        if check(i):
            count += 1
        if check(x/i):
            count += 1

print count
