#coding:utf-8

import math

x = int(raw_input())
a = list(0 for i in range(10))
s=list()
m = x
while x:
	a[x % 10] += 1
	x /= 10
ans = 0
for i in range(1, int(math.sqrt(m)) + 1):
	if m % i == 0:
		j = i
		while j:
			if a[j % 10]:
				ans += 1
				break
			j /= 10
		j = m / i
		if i * i != m:
			while j:
				if a[j % 10]:
					ans += 1
					break
				j /= 10
print(ans)
