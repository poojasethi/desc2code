def getDigits(N):
  ret = set()
  while N:
    ret.add(N%10)
    N /= 10
  return ret
    
def check(N, M):
  return len(getDigits(N) & getDigits(M))

def getElephantNumbers(N):
  i = 1
  ret = 0
  while True:
    if N%i == 0:
      if check(N, i): ret += 1
      if check(N, N/i) and i != N/i: ret += 1
    i += 1
    if i * i > N: break
  return ret

print getElephantNumbers(int(raw_input()))
