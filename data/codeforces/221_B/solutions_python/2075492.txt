from sys import stdin
from math import sqrt

x = int(stdin.readline())
root = sqrt(x)
sqr_flag = root.is_integer()
root = int(root)
divs =  reduce(list.__add__,[[k,x/k] for k in range(1,root) if not x%k],[]) + ([root] if sqr_flag else ([] if x%root else [root,x/root]))
base = set(str(x))
print len(filter(lambda k: set(str(k)).intersection(base), divs)),
