#include <iostream>
using namespace std;
bool check(int a[10],int j)
{
	for(j;j>0;j/=10)
		if(a[j%10]) return true;
	return false;
}
int main()
{
	int a[10]={0};
	int b;
	cin>>b;
	int count=0;
	if(b==1)
	{
		cout<<"1\n";
		return 0;
	}
	int c=b;
	for(c;c>0;c/=10) a[c%10]=1;
	for(int j=1;j*j<=b;j++)
		if(!(b%j))
		{
			if(check(a,j)) count++;
			if(j*j!=b&&check(a,b/j))count++;
		}
	cout<<count<<endl;
}
