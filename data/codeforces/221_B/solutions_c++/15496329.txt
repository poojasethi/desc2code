#include<iostream>
#include<cmath>
using namespace std;
int v[10],n,a;
int function(int x)
 {
 while(x>0)
  {
  if(v[x%10])
   return 1;
   x/=10;
  }
 return 0;
 }

int main()
 {
     int x=0;
  cin>>n;
  a=n;
  while(a>0)
  {
  v[a%10]=1;
  a/=10;
  }

 for(int i=1;i*i<=n;i++)
  if(n%i==0)
  {
      x+=function(i);
     if(i!=n/i)
        x+=function(n/i);
  }

  cout<<x;
  return 0;
 }
