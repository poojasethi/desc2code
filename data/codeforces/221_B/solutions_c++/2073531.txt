#include <cstdio>
using namespace std;

bool check(int a, int b){
	bool used[10]={};
	while(a){
		used[a%10]=1;
		a/=10;
	}
	while(b){
		if(used[b%10])return 1;
		b/=10;
	}
	return 0;
}
int main() {
	int n;
	scanf("%d", &n);
	int ans=0;
	for(int i=1;i*i<=n;++i){
		if(n%i==0){
			if(check(i,n))++ans;
			if(i*i!=n)
				if(check(n/i,n))++ans;
		}
	}
	printf("%d\n",ans);
	return 0;
}
