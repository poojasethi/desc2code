#include <iostream>
#include <vector>
using namespace std;

bool digit(const vector<bool> &v, int d) {
	for (int i = d; i; i/=10)
		if (v[i%10]) return true;
	return false;
}

int main() {
	int n;
	cin >> n;
	vector<bool> v(10);
	int t = 0;
	for (int i = n; i; i/=10) v[i%10] = true;
	for (int i = 1; i*i <= n; ++i) {
		if (n%i == 0) {			
			t += digit(v, i) + (digit(v, n/i) and i*i!=n);
		}
	}
	cout << t << endl;	
}
