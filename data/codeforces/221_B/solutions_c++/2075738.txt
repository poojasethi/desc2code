#include <iostream>
#include <math.h>
using namespace std;
int b[10];
int f(int n,int i){ int t=i;while(t!=0){ int r=t%10;t=t/10;if(b[r]>0)return 1;}return 0;}
int main() { int i,n,x,c,r,a,s;
    cin>>n;c=0;
    x=n;s=(int)sqrt(n);
    for(i=0;i<10;i++)b[i]=0;
    while(x!=0){ r=x%10;x=x/10;b[r]++;}
    for(i=1;i<=s;i++) if(n%i==0){ a=n/i;if(f(n,i))c++;
        if(a!=i && f(n,a))c++; }
    cout<<c;
    return 0;
}
