#include <cstdio>

int N,dig[10];
  
int main() {
scanf("%d", &N);
int k = N;
while(k) {
dig[k%10]=1;
k/=10;
}
int ans=0;
for(int i = 1; i * i <= N; ++i) {
if(N % i == 0) {
int t = 0;
k = i;
while(k) {
t+=dig[k%10];
k/=10;
}
if(t) ++ans;
t=0;
if(i * i == N) break;
k = N / i;
while(k) {
t+=dig[k%10];
k/=10;
}
if(t) ++ans;
}
}
printf("%d\n", ans);
}
