#include <iostream>
using namespace std;

long long x, dig[10] = {}, pow = 1, res = 0;

bool check(long long k){
	while(k/10>0) if(dig[(k/=10)%10]) return 1;
	return 0;
}

int main(){

	cin >> x;
	while(x/pow>0) dig[10*x/(pow*=10)%10] = 1;
	for(int i = 1; i*i<=x; i++){
		if(x%i) continue;
		if(i*i<x) res+=check(10*i)+check(10*x/i);
		else res+=check(10*i);
	}
	cout << res;

}
