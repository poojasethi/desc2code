#include<cstdio>
#include<cmath>
using namespace std;
bool b1[10];
bool f(int a)
{
	while(a)
	{
		if(b1[a%10])
			return 1;
		a/=10;
	}
	return 0;
}
int main()
{
	int n,a,o=0;
	scanf("%d",&n);
	a=n;
	while(a)
	{
		b1[a%10]=1;
		a/=10;
	}
	a=(int)sqrt(n);
	for(int i=1;i<=a;++i)
	if(!(n%i))
		o+=f(i)+f(n/i);
	if(a*a==n&&f(a))--o;
	printf("%d\n",o);
}
