#include <stdio.h>
int get(int n){
    int r=0;
    while(n){
        r|=1<<(n%10);
        n/=10;
    }
    return r;
}
int main()
{
    int n,N,c=0;
    scanf("%d",&n);
    N=get(n);
    for(int i=1;i*i<=n;++i)
        if(n%i==0){
            c+=!!(get(i)&N);
            c+=i*i!=n && (get(n/i)&N);
        }
    printf("%d",c);
    return 0;
}
