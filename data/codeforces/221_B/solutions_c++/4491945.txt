#include <iostream>

using namespace std;
int x;
bool a[10]={0};
bool ff(int i)
{while (i) {int y=i%10; i/=10; if (a[y]) return 1;}
 return 0;
}
int main()
{//freopen("i.txt","r",stdin);
cin>>x;
int ans=0, n=x;
while (x) {int y=x%10; x/=10; a[y]=1;}
for (int i=1; i*i<=n;i++)
 if (n%i==0)
 { ans+=ff(i);
  if (i*i<n) ans+=ff(n/i);
 }
cout<<ans;
return 0;
}
