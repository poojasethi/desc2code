#include <iostream>
using namespace std;
int n,num[10],ans;
int check(int f){
    while(f){
        if(num[f%10])
            return 1;
        f/=10;
    }
    return 0;
}
int main(){
    cin>>n;
    int j=0,k=n;
    while(k){
        num[k%10]=1;
        k/=10;
    }
    for(int i=1;i*i<=n;i++){
        if(n%i==0)
            ans+=check(i)+(i*i<n and check(n/i));
        }
    cout<<ans;
}
