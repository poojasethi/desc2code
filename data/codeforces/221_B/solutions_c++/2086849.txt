#include <iostream>
using namespace std;

int mask(int a)
{
	int m = 0;
	while (a)
	{
		m |= 1 << a % 10;
		a /= 10;
	}
	return m;
}

bool check(int a, int b)
{
	return mask(a) & mask(b);
}

int main()
{
	int x; cin >> x;
	int res = 0;
	for (int i = 1; i * i <= x; ++i)
		if (x % i == 0)
		{
			if (check(i, x))
				++res;
			if (i * i != x && check(x / i, x))
				++res;
		}
	cout << res << endl;
	return 0;
}
