#include<bits/stdc++.h>
using namespace std;
bool iscom(int m,int n)
{
	int k;
	while(m)
	{
		k=n;
		while(k)
		{
			if(m%10==k%10)
				return 1;
			k/=10;
		}
		m/=10;
	}
	return 0;
}
int main()
{
	int i,n,cnt=0;
	cin>>n;
	for(i=1;i*i<=n;i++)
	{
		if(n%i==0)
		{
			if(iscom(n,i))
				cnt++;
			if(iscom(n,n/i) && i!=n/i)
				cnt++;
		}
	}
	cout<<cnt<<endl;
	return 0;
}
