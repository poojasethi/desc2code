#include <iostream>

using namespace std;
int x,i,j,c[10],cnt;
int main()
{
    cin>>x;
    for(i=x;i;i/=10)c[i%10]=1;
    for(i=1;i*i<x;i++)
    {
        if(x%i)continue;
        for(j=i;j;j/=10)if(c[j%10]){cnt++;break;}
        for(j=x/i;j;j/=10)if(c[j%10]){cnt++;break;}
    }
    if(i*i==x)for(j=i;j;j/=10)if(c[j%10]){cnt++;break;}
    cout<<cnt;
    return 0;
}
