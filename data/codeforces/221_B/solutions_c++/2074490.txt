#include<cstdio>
int main()
{
	int n,x,i,ct=0;
	scanf("%d", &n);
	for(i=1;i*i<=n;i++)
	{
		if(n%i==0)
		{
			int x=n;
			bool vis[10]={false};
			while(x>0)
			{
				vis[x%10]=1;
				x/=10;
			}
			x=i;
			while(x>0)
			{
				if(vis[x%10]){ ct++;break;}
				x/=10;
			}
			x=n/i;
			if(x!=i)
			{
			while(x>0)
			{
				if(vis[x%10]){ ct++;break;}
				x/=10;
			}
			}
		}
	}
	printf("%d\n", ct);
	return 0;
}
