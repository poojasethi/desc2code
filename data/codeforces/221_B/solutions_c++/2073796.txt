#include <cstdio>
#include <cmath>
using namespace std;

bool boo[10];
int ans;

void WORK(int n) {
	while (n) {
		if (boo[n%10]) {++ans;return;}
		n/=10;
	}
}

int main() {
	int n;
	scanf("%d",&n);
	int m=n;
	while (n) {
		boo[n%10]=true;
		n/=10;
	}
	for (int i=1;i<=(int)sqrt(m);++i)
		if (!(m%i)) {
			WORK(i);
			if (i*i!=m) WORK(m/i);
		}
	printf("%d\n",ans);
	return 0;
}
