#include<iostream>
using namespace std;
int a[100001];
int main()
{
  __int64 n=0,i,x,s,ans=0;
  bool f[10]={0};
  cin>>x;
  for(i=1;i*i<=x;i++) if (x%i==0){a[n++]=i;if (x/i!=i)a[n++]=x/i;}
  s=x;
  while(s)
  {
    f[s%10]=1; s/=10;
  }
  for(i=0;i<n;i++)
  {
    s=a[i];
    while(s)
    {
      if (f[s%10]) {ans++; break;}
      s/=10;
    }
  }
  cout<<ans;
//  system("pause");
  return 0;
}
