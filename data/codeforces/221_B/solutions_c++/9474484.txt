#include<iostream>
using namespace std;
int v[10],x,n=0,m;
int F(int n){
    for(;n;n=n/10)
        if(v[n%10])
            return 1;
    return 0;
}
int main(){
    cin>>x;
    m=x;
    for(;m;m=m/10)
        v[m%10]=1;
    for(int i=1;i*i<=x;i++)
        if(x%i==0)
            n+=F(i)+(i*i<x&&F(x/i));
    cout<<n<<endl;
    return 0;
}
