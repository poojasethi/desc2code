#include <stdio.h>
int n, m, res;
bool a[10];
bool f(int x){
	while(x){
		if(a[x % 10]) return 1;
		x = x / 10;
	}
	return 0;
}
int main(){
	scanf("%d", &n);
	m = n;
	while(m){
		a[m % 10] ++;
		m = m / 10;
	}
	for(int i = 1; i*i <= n; ++ i){
		if(n % i == 0){
			if(f(i)) res ++;
			if(i*i != n && f(n / i)) res ++;
		}
	}
	printf("%d", res);
}
