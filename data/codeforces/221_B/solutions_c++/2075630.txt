#include <cstdio>
int N, hsh[10], ans;
int cnt(int x){
	int ret(0);
	while(x){
		ret+= hsh[x%10];
		x/= 10;
	}
	return ret!=0;
}
int main(){
	scanf("%d", &N);
	for(int i= N;i;i/= 10)
		hsh[i%10]= 1;
	for(int i= 1;i*i<= N;++ i) if(N%i == 0){
		ans+= cnt(i);
		if(i*i!=N) ans+= cnt(N/i);
	}
	printf("%d\n", ans);
	return 0;
}
