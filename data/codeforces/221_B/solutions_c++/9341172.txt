#include<bits/stdc++.h>
using namespace std;
int sa(int m,int n)
{
set<int> a;
while(m){
a.insert(m%10);
m/=10;
}
while(n)
{
int x = n%10;
if(a.find(x)!=a.end())return 1;
n/=10;
}
return 0;
}
int main()
{
int n;
scanf("%d",&n);
int ans = 0;
int x;
for(x=1;x*x<=n;x++)
{
if(n%x==0)
{
int y = n/x;
if(sa(x,n))ans++;
if(y!=x)
if(sa(y,n))ans++;
}
}
printf("%d\n",ans);
return 0;

}
