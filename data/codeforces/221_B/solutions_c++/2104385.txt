#include <iostream>

using namespace std;

int main()
{
	int n;
	cin>>n;
	bool arr[10]={0};
	int m=n;
	int cnt=0;
	for(;m;arr[m%10]=1,m/=10);
	for(int i=1;i*i<=n;i++)
		if(n%i==0)
		{
			for(int t=i;t;t/=10)
				if(arr[t%10])
				{
					cnt++;
					break;
				}
			if(i*i<n)
			{
				for(int t=n/i;t;t/=10)
					if(arr[t%10])
					{
						cnt++;
						break;
					}
			}
		}
	cout<<cnt<<endl;
	return 0;
}
