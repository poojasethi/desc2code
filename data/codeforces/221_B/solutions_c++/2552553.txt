#include<iostream>
#include<cmath>
using namespace std;
bool arr[10];
bool dig(int a){
	for(; a; a/=10)
		if(arr[a%10])	return true;
	return false;
}
int main(){
	int n, count=0, d;
	cin>>n;
	for(int i=1; n/i; i*=10)
		arr[(n%(i*10))/i]=true;
	d=sqrt(n);
	for(int i=1; i<=d; i++)
		if(!(n%i)){	count+=dig(i);
			if(i*i!=n)	count+=dig(n/i);
		}
	cout<<count<<endl;
	return 0;
}
