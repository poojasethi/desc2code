#include <iostream>
using namespace std;
int n,i,ans=0,dn,a[11];
int check(int x)
{
	while(x>0)
	{
		if (a[x%10])
			return 1;
		x/=10;
	}
	return 0;
}
int main()
{
   cin>>n; dn=n;
   while (dn>0)
   {
	   a[dn%10]=true;
	   dn/=10;
   }
   for(i=1;i*i<n;i++)
	   if (n%i==0)
		   ans+=check(i)+check(n/i);
   if (i*i==n)
	   ans+=check(i);
   cout<<ans<<endl;
}
