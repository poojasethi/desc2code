#include<cstdio>
#include<iostream>

using namespace std;

bool used[13],f;

int main(){
  int i,j,x,re=0;
  cin>>x;
  for(i=x;i;i/=10){
    used[i%10]=1;
  }
  for(i=1;i*i<=x;i++){
    if(x%i)continue;
    f=0;
    for(j=i;j;j/=10){
      if(used[j%10]){
	re++;
	f=1;
	break;
      }
    }
    if(i!=x/i)for(j=x/i;j;j/=10)
      if(used[j%10]){
	re++;
	f=1;
	break;
      }
  }
  cout<<re<<endl;
  return 0;
}
