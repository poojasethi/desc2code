#include <iostream>
#include <set>

using namespace std;

int sol=0;
int x;

set<int> s;

void check(int d)
{
  while (d>0) {
    if (s.count(d%10)) {
      sol++;
      return;
    }
    d/=10;
  }
}

int main()
{
  cin>>x;
  int xx=x;
  while (xx>0) {
    s.insert(xx%10);
    xx/=10;
  }
  for (int d=1;d<=x and d*d<=x;d++) {
    if (x%d==0) {
      check(d);
      if (x/d!=d) check(x/d);
    }
  }
  cout<<sol<<endl;
}
