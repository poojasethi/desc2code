#include<cstdio>

int n, x, ans;
int s[11];

bool chk(int v) {
	while(v) {
		if(s[v%10]) return true;
		v/=10;
	}
	return false;
}

int main() {
	scanf("%d", &n);
	for(int t=n;t;t/=10)
		s[t%10] ++;
	ans=0;
	for(int i=1;i*i<=n;i++) {
		if (n%i) continue;
		x=n/i;
		ans+=chk(i);
		if(x>i) ans+=chk(x);
	}
	printf("%d\n", ans);
	
	return 0;
}
