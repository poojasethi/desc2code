#include<bits/stdc++.h>
using namespace std;
int p[11]={0};
int ans=0;
void check(int k){
	while(k){
		if(p[k%10]==1){
			++ans;
			break;
		}
		k/=10;
	}
}
int main()
{
	int n;
	scanf("%d",&n);
	int tmp=n;
	while(tmp){
		p[tmp%10]=1;
		tmp/=10;
	}
	
	if(n==1) {
		ans=1;
	}
	else for(int i=1;i*i<=n;i++){
		if(n%i==0) {
			check(i);
			if(n/i!=i) check(n/i);	
		}
	}
	cout<<ans<<endl;
	return 0;
}
