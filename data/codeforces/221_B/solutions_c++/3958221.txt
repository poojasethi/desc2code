#include<iostream>
using namespace std;
int a[10];
int find(int n)
{
    for(;n;n=n/10)
    {
        if(a[n%10])
            return 1;
    }
    return 0;
}
int main()
{
    long long x,d=0;
    cin>>x;
    for(int i=x;i;i=i/10)
    a[i%10]=1;
    for(int j=1;j*j<=x;j++)
    {
        if(x%j==0)
            d+=find(j)+(j*j<x && find(x/j));
    }
    cout<<endl<<d;
    return 0;
}
