#include <iostream>
using namespace std;
int n,num[10],ans;
int check(int f){
    while(f){
        if(num[f%10]==1)
            return 1;
        f/=10;
    }
    return 0;
}
int main(){
    cin>>n;
    int j=0,k=n;
    while(k){
        num[k%10]=1;
        k/=10;
    }
    for(int i=1;i*i<=n;i++){
        if(n%i==0){
            ans+=check(i)+(n==i*i?0:(check(n/i)));
        }
    }
    cout<<ans;
}
