#include<iostream>
#include<stdio.h>
using namespace std;
int x;
int dx[10];
int ost(int p)
{
    for(;p;p/=10)
        if(dx[p%10])
            return 1;
    return 0;
}
int main()
{
    cin>>x;
    int d=0;
    for(int i=x;i;i/=10)
        dx[i%10]=1;
    for(int i=1;i*i<=x;i++)
    {
        if(x%i==0)
        {
            d+=ost(i)+(i*i<x&&ost(x/i));
        }
    }
    cout<<d; 
}
