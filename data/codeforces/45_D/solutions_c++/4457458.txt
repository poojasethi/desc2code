#include <cstdio>
#include <algorithm>

using namespace std;

struct event
{
	int l, r, id;
};

const int maxn = 110;
int n, ans[maxn];
event eve[maxn];

bool cmp(const event &a, const event &b)
{
	return a.r < b.r;
}

int main()
{
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
	{
		scanf("%d%d", &eve[i].l, &eve[i].r);
		eve[i].id = i;
	}
	sort(eve, eve + n, cmp);
	int last = -1;
	for (int i = 0; i < n; i++)
	{
		int j = eve[i].id;
		ans[j] = eve[i].l;
		while (true)
		{
			bool flag = true;
			for (int k = 0; k < i; k++)
			    if (ans[eve[k].id] == ans[j])
			        flag = false;
			if (flag)
			    break;
			else
			    ans[j]++;
		}
	}
	for (int i = 0; i < n; i++)
		printf("%d ", ans[i]);
		
	return 0;
}
