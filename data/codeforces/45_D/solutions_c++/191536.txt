#include <iostream>
#include <algorithm>
#include <map>
using namespace std;

const int N = 100;
struct record{
	int l, r;
	int id;
	friend bool operator <(const record &a, const record &b){
		return a.l < b.l || a.l == b.l && a.r < b.r;
	}
} A[N];
int T[N];
int n;

void init(){
	for (int i=0;i<n;i++){
		cin >> A[i].l >> A[i].r;
		A[i].id = i;
	}
}


void solve(){
	
	for (int i=0;i<n;i++){
		for (int j=i+1;j<n;j++)
			if (A[j] < A[i]) swap(A[i], A[j]);
		for (int j=i+1;j<n;j++)
			A[j].l = max(A[i].l + 1, A[j].l);
		T[A[i].id] = A[i].l;
	}
	
}




int main(){
	n = 0;
	string s;
	while (cin >> n){
		init(); solve();
		for (int i=0;i<n-1;i++)
			cout << T[i] << " ";
		cout << T[n-1] << endl;
	}
}
