#include <bits/stdc++.h>
using namespace std;

typedef pair< pair <int, int> , int > PI;
PI item[1000];
int n;
int arr[1000];
bool flag[10000008];
int main(){
    cin>>n;
    int i;
    for(i = 0; i < n; ++i){
        cin >> item[i].first.second >> item[i].first.first;
        item[i].second = i;        
    }
    sort(item, item + n);
    for(i = 0; i < n; ++i){
        int y = item[i].first.second;
        //cout<<y<<' ';
        while(flag[y]){
        	y++;
        }
        //cout<<y<<endl;
        arr[item[i].second] = y;
        flag[y] = true;
    }
    cout<<arr[0];
    for(i=1;i<n;i++){
    	cout<<' '<<arr[i];
    }
    cout<<endl;
}
