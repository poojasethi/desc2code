#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define limit 10000000
#define maxn 105

int n,f[limit+5];
struct node
{
	int l,r,id;
}a[maxn];
int ans[maxn];

inline bool cmp(const node &a,const node &b)
{
	return a.r<b.r || a.r==b.r && a.l<b.l;
}

inline int Find(int x)
{
	if (f[x]==x) return x;
	return f[x]=Find(f[x]);
}

int main()
{
	scanf("%d",&n);
	for (int i=0;i<n;++i)
	{
		scanf("%d%d",&a[i].l,&a[i].r);
		a[i].id=i;
	}
	sort(a,a+n,cmp);
	for (int i=1;i<=limit;++i)
		f[i]=i;
	for (int i=0;i<n;++i)
	{
		int y=Find(a[i].l);
		f[y]=y+1;
		ans[a[i].id]=y;
	}
	
	for (int i=0;i<n;++i)
	{
		if (i) printf(" ");
		printf("%d",ans[i]);
	}
	puts("");
	return 0;
}
