#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

struct Node{
	int l,r,id;
}a[110];

inline bool cmp(const Node &a,const Node &b){
	if(a.r==b.r) return a.l>b.l;
	return a.r<b.r;
}
int n,num[110];
bool flag[10000010];

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d%d",&a[i].l,&a[i].r),a[i].id=i;
	sort(a+1,a+n+1,cmp);
	for(int i=1;i<=n;i++){
		int last=a[i].l;
		while(flag[last]) last++;
		flag[last]=1;
		num[a[i].id]=last;
	}
	for(int i=1;i<=n;i++) printf("%d ",num[i]);puts("");
	return 0;
}
