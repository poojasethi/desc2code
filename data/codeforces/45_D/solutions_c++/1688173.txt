#include <iostream>
#include <algorithm>
using namespace std;
int n,l,r,x,y,a[111],z;
pair< pair<int,int>, int> t[111];
int main()
{
    cin>>n;
    for (int i=1;i<=n;i++){
        cin>>l>>r;
        t[i]=make_pair(make_pair(l,r-l+1),i);}
    sort(t+1,t+n+1);
    for (int i=1;i<=n;i++){
        x=t[i].first.first;
        for (int j=i+1;j<=n && t[j].first.first==x;j++){
            y=t[j].first.second;z=t[j].second;
            t[j]=make_pair(make_pair(x+1,y-1),z);}
        sort(t+1,t+n+1);
        a[t[i].second]=t[i].first.first;}
    for (int i=1;i<=n;i++)cout<<a[i]<<" ";
    return 0;
}
