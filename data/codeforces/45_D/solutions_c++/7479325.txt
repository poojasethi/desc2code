#include<cstdio>
#include<algorithm>
#include<set>
#define pii pair<int,int>
#define A first
#define B second
using namespace std;
int n,ans[101],x = 0,r;
pair <pii,int> p[101];
set <pii> S;
set <pii>::iterator it;
int main(){
    scanf("%d",&n);
    for(int i=0,a,b;i<n;i++){
        scanf("%d %d",&a,&b);
        p[i] = make_pair(pii(a,b),i);
    }
    sort(p,p+n);
    do{
        if(S.empty()) x = max(x, p[r].A.A);
        while(r < n && p[r].A.A <= x)
            S.insert(pii(p[r].A.B,p[r].B)), r++;
        if(!S.empty()){
            it = S.begin();
            ans[it->B] = x++;
            S.erase(it);
        }
    }while(r < n || !S.empty());
    for(int i=0;i<n;i++) printf("%d ",ans[i]);
    return 0;    
}
