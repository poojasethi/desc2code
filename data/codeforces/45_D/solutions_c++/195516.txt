#include <stdio.h>
#include <memory.h>
#include <vector>
#include <algorithm>
using namespace std;

int N,ans[100];

struct A
{
    int L,R,id;
}a[100];

bool cmp(A a,A b)
{
    if(a.L != b.L)
        return a.L < b.L;
    return a.R < b.R;
}

int main()
{
    int i,j;
    scanf("%d",&N);
    for(i = 0;i < N;i++)
        scanf("%d%d",&a[i].L,&a[i].R),a[i].id = i;
    int cur = 1;
    for(i = 0;i < N;i++)
    {
        sort(a,a + N,cmp);
        cur = cur > a[0].L ? cur : a[0].L;
        ans[a[0].id] = cur;
        cur ++;
        a[0].L = 0x3f3f3f3f;
        for(j = 0;j < N;j++)
            a[j].L = a[j].L > cur ? a[j].L : cur;
    }
    for(i = 0;i < N;i++)
        printf("%d ",ans[i]);
    printf("\n");
    return 0;
}
