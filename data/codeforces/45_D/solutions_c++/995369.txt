#include<iostream>
#include<set>
#include<vector>
#include<algorithm>
using namespace std;
typedef pair<int,int> ii;

set<int> used;
vector<pair<ii,int> > v;
int res[110];
int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++) {
		int a,b;
		cin>>a>>b;
		v.push_back(make_pair(ii(b,a),i));
	}
	sort(v.begin(),v.end());
	for(int i=0;i<n;i++) {
		for(int j=v[i].first.second;j<=v[i].first.first;j++) {
			if(!used.count(j)) {
				res[v[i].second]=j;
				used.insert(j);
				break;
			}
		}
	}
	for(int i=0;i<n;i++) {
		cout<<res[i]<<' ';
	}
}
