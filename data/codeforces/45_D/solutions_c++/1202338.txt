#include <iostream>
#include <algorithm>
using namespace std;

const int MAXN = 100;

int n;
struct elem
{
	int l,r,id;
	bool operator < (const elem& A)const
	{
		return l<A.l || l==A.l && r<A.r;
	}
};

elem e[MAXN];
int ans[MAXN];

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		e[i].id=i;
		cin>>e[i].l>>e[i].r;
	}

	int m=n;
	for(int i=0;i<m;i++)
	{
		sort(e,e+n);
		ans[e[0].id]=e[0].l;
		for(int j=1;j<n;j++)
			if( e[j].l==e[0].l) e[j].l++;
		swap(e[0],e[n-1]);
		n--;
	}

	for(int i=0;i<m;i++)
		cout << ans[i] << ' ';

	return 0;
}
