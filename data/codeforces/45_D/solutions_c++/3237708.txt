#include <iostream>
#include <algorithm>

using namespace std;

struct LES{
    int l;
    int r;
    int d;
};

LES a[1000];
bool u[10000009];
int n, ans[1000];



bool cmp(LES a,LES b)
{
    return a.r<b.r;
}

int main ()
{
    cin>>n;
    for (int i = 0; i < n; i++)
    {
        cin>>a[i].l>>a[i].r;
        a[i].d=i;
    }
    sort (a,a+n,cmp);
    for (int i = 0; i < n; i++)
    {
        int t = a[i].l;
        while (u[t])
            t++;
        u[t] = 1;
        ans[a[i].d] = t;
    }
    for (int i = 0; i < n; i++)
        cout<<ans[i]<<" ";
    cout<<endl;
    return 0;
}
