#include <bits/stdc++.h>
using namespace std;
struct st{
	int s,e,id;
}item[1000];
bool cmp(st a,st b){
	if(a.e==b.e){
		return a.s>b.s;
	}
	return a.e<b.e;
}
int n;
int arr[1000];
bool flag[10000008];
int main(){
	cin>>n;
	int i;
	for(i=0;i<n;i++){
		cin>>item[i].s>>item[i].e;
		item[i].id = i;
	}
	sort(item,item+n,cmp);
	for(i=0;i<n;i++){
		int y = item[i].s;
		while(flag[y]==true){
			y++;
		}
		flag[y] = true;
		arr[item[i].id] = y;
	}
	cout<<arr[0];
	for(i=1;i<n;i++){
		cout<<" "<<arr[i];
	}
	cout<<endl;
}
