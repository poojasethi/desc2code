#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int n, m;
struct interval{int l, r, n, ans;} S[110];
bool used[10000010];

inline bool operator < (interval a, interval b){return a.r < b.r || (a.r == b.r && a.l < b.l);}

int main(){
	cin >> n;
	for (int i = 0; i < n; i++) cin >> S[i].l >> S[i].r, S[i].n = i;
	sort(S, S + n);
	for (int i = 0; i < n; i++)
		for (int now = S[i].l; now <= S[i].r; now++)
			if (!used[now]){
				used[now] = true;
				S[i].ans = now;
				break;
			}
	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++)
			if (S[j].n == i){
				cout << S[j].ans << ' ';
				break;
			}
	cout << endl;
}
