#include <iostream>
#include <algorithm>
using namespace std;
bool c[10000001];
pair<pair<int, int>, int > pr[101];
int ans[101];
main(){
    int n,i,j;
    cin>>n;
    for(i=1;i<=n;i++){
        cin>>pr[i].first.second>>pr[i].first.first;
        pr[i].second=i;
    }
    sort(pr+1,pr+n+1);
    for(i=1;i<=n;i++)
        for(j=pr[i].first.second;;j++)
            if (!c[j]){
                c[j]=1;
                ans[pr[i].second]=j;
                break;
            }
    for(i=1;i<=n;i++)
        cout<<ans[i]<<" ";
}
