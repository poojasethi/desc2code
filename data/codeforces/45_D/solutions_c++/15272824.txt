#include <bits/stdc++.h>
using namespace std;
struct tseg
{
    int l,r,id;
}seg[107];
bool vis[10000007];
int ans[107];
struct cmpl
{
    bool operator()(const tseg &a,const tseg &b)
    {
        return a.r<b.r||a.r==b.r&&a.l<b.l;
    }
};
int main()
{
    int n,i;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d%d",&seg[i].l,&seg[i].r);
        seg[i].id=i;
    }
    sort(seg,seg+n,cmpl());
    for(i=0;i<n;i++)
    {
        int t=seg[i].l;
        while(vis[t])
            t++;
        vis[t]=true;
        ans[seg[i].id]=t;
    }
    for(i=0;i<n;i++)
        printf("%d ",ans[i]);
    return 0;
}
