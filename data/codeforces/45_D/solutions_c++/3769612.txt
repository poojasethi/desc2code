#include <stdio.h>
#include <queue>
#include <utility>
#include <list>
using namespace std;

int a[10000];
int b[10000];
int main(){
    int n;
    scanf("%d",&n);
    for (int i=0;i<n;i++){
        scanf("%d%d",&a[i],&b[i]);
        
    }
    bool f=true;
    while (f){
        f=false;
    for (int i=0;i<n;i++)
        for (int j=0;j<n;j++)
            if (i!=j)
        {
            if (a[i]==a[j]){
                f=true;
                if (b[i]>b[j])
                    a[i]++;
                else
                    a[j]++;
            }
        }
    }
    for (int i=0;i<n;i++)
        printf("%d ",a[i]);
    return 0;
}
