#include <cstdio>
#include <algorithm>
using namespace std;
const int N = 101;
struct node {
	int l, r, id;
};
struct node a[N];
int ans[N];
bool cmp(const node x, const node y) {
	if(x.r != y.r) return x.r < y.r;
	return x.l < y.l;
}
int main() {
	int n;
	scanf("%d", &n);
	for(int i = 0; i < n; i ++) {
		scanf("%d%d", &a[i].l, &a[i].r);
		a[i]. id = i + 1;
	}
	sort(a, a + n, cmp);
//	for(int i = 0; i < n; i ++) printf("	%d %d %d %d\n", i, a[i].id, a[i].l, a[i].r);
	int day = 1, cnt = 0;
	while(cnt < n){
		for(int i = 0; i < n; i ++) {
			if(ans[ a[i].id ] > 0) continue;
			if(day >= a[i].l) {
				ans[ a[i].id ] = day;
				cnt ++;
				break;
			}
		}
		day ++;
	}
	for(int i = 1; i <= n; i ++) printf("%d ", ans[i]);
	puts("");
	return 0;
}
