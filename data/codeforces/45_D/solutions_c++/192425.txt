#include <iostream>
#include <set>
#include <algorithm>

using namespace std;

struct point{
	int x,y,z;
};

const int MAXN=100+10;
point a[MAXN];
int n;
int ans[MAXN];
set <int> s;

bool f(point x,point y){
	if(x.x!=y.x)
		return x.x<y.x;
	else
		return x.y<y.y;
}

int main(){
	cin >> n;
	for(int i=1;i<=n;i++){
		cin >> a[i].x >> a[i].y;
		a[i].z=i;
	}
	sort(a+1,a+n+1,f);
	for(int i=n;i>=1;i--){
		for(int j=a[i].y;j>=a[i].x;j--)
			if(s.find(j)==s.end()){
				s.insert(j);
				ans[a[i].z]=j;
				break;
			}
	}
	for(int i=1;i<=n;i++)
		cout << ans[i] << " " ;
	cout << endl;
}
