#include<iostream>
#include<algorithm>

using namespace std;

pair<pair<int,int>,int> arr[101];

bool mark[10000001];

int main()
{
    int n;
    cin>>n;
    for (int i=0;i<n;i++)
    {
        cin>>arr[i].first.second>>arr[i].first.first;
        arr[i].second = i;
    }
    sort(arr,arr+n);
    int ans[101];
    for (int i=0;i<n;i++)
    {
        int c = arr[i].first.second;
        while(mark[c])
            c++;
        ans[arr[i].second] = c;
        mark[c] = 1;
    }
    for (int i=0;i<n;i++)
        cout<<ans[i]<<" ";
    return 0;
}
