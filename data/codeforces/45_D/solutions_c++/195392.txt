#include <iostream>
#include <cstdio>
#include <algorithm>
#include <climits>
using namespace std;

int N,cur = 1;
struct stuff{int ID,A,B;};
bool cmp(stuff a,stuff b){
	if(a.A != b.A) return a.A < b.A;
	return a.B < b.B;
}
stuff S[105];
int ans[105];

int main(){
	scanf("%d",&N);
	for(int i=0;i<N;++i){
		scanf("%d%d",&S[i].A,&S[i].B);
		S[i].ID = i;
	}
	for(int i=0;i<N;++i){
		sort(S,S+N,cmp);
		cur = max(cur,S[0].A);
		ans[S[0].ID] = cur;
		S[0].A = INT_MAX; 
		++cur;
		for(int j=0;j<N;++j) if(S[j].A <= cur) S[j].A = cur;
	}
	for(int i=0;i<N;++i) printf("%d ",ans[i]);
	printf("\n");
//	system("pause");
}
