#include <bits/stdc++.h>
using namespace std;

struct event {
	int l, r, ind;
};

bool comp (event a, event b){
	if (a.l!=b.l) return a.l<b.l;
	return a.r<b.r;
}

const int MAXN=100+10;
int res[MAXN];

int main() {
	vector <event> e;
	int n, a, b;
	cin >> n;
	for (int i=0;i<n;i++){
		cin >> a >> b;
		event temp = {a, b, i};
		e.push_back(temp);
	}
	sort(e.begin(), e.end(), comp);

	int pos=0;
	for (int i=0;i<n;i++){
		int l=e[0].l, ind=e[0].ind;
		res[ind]=max(pos, l);
		pos=max(pos, l)+1;
		for (int i=0;i<int(e.size());i++)
			if (e[i].l<=l) e[i].l=pos;
		e.erase(e.begin());
		sort(e.begin(), e.end(), comp);
	}

	for (int i=0;i<n;i++){
		if (i) cout << ' ';
		cout << res[i];
	}
	cout << endl;

	return 0;
}
