#include <cstdio>
#include <algorithm>

using namespace std;
struct couple
{
    int first;
    int second;
    int ind;
};

bool cmp(couple a, couple b)
{
    return a.second < b.second;
}

couple arr[1001];
int n, ans[1001];
bool u[10000001];

int main()
{
    scanf("%d", &n);
    for (int i = 0; i < n; i++)
    {
        scanf ("%d %d", &arr[i].first, &arr[i].second);
        arr[i].ind = i;
    }

    sort (arr, arr + n, cmp);
    for (int i = 0; i < n; i++)
    {
        int tmp = arr[i].first;
        while (u[tmp])
            tmp++;
        u[tmp] = 1;
        ans[arr[i].ind] = tmp;
    }
    for (int i = 0; i < n; i++)
    {
        printf("%d ", ans[i]);
    }
    return 0;
}
