#include<iostream>
#include<cstdio>
#include<algorithm>
#include<assert.h>

#define x first
#define y second

using namespace std;

typedef pair<int,int> pii;

int n;
pair<pair<int,int>,int> t[100];
bool q[10000001];
int sol[100];

int main()
{
    scanf("%d", &n);
    for(int i = 0; i < n; ++i)
    {
        scanf("%d %d", &t[i].x.y, &t[i].x.x);
        t[i].y = i;
    }
    sort(t, t+n);
    for(int i = 0; i < n; ++i)
    {
        int a = t[i].x.y;
        while(q[a])
            ++a;
        assert(a <= t[i].x.x);
        q[a] = 1;
        sol[t[i].y] = a;
    }
    for(int i = 0; i < n; ++i)
        printf("%d ", sol[i]);
    printf("\n");
    return 0;
}
