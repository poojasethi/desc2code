#include <cstdio>
#include <queue>
#include <algorithm>
using namespace std;
struct Node {int l,r,p;} a[100];
int ans[100];
priority_queue<Node> h;

bool cmp(Node a,Node b) {return a.l<b.l;}

bool operator<(Node a,Node b) {return a.r>b.r;}

int main()
{
	int n,i;
	scanf("%d",&n);
	for (i=0; i<n; ++i)
	{
		scanf("%d%d",&a[i].l,&a[i].r);
		a[i].p=i;
	}
	sort(a,a+n,cmp);
	int x=0, j=0;
	for (i=0; i<n; ++i)
	{
		if (h.empty()) x=a[j].l;
		while (j<n && a[j].l<=x) h.push(a[j++]);
		ans[h.top().p]=x++;
		h.pop();
	}
	for (i=0; i+1<n; ++i) printf("%d ",ans[i]);
	printf("%d\n",ans[i]);
	return 0;
}
