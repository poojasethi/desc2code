#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cstdlib>

using namespace std;

typedef pair< pair <int, int> , int > PI;
PI dates[102];
bool ismarked[10000001];

int main()
{
    for(int j = 0; j < 10000001; j++)ismarked[j] = false;
    int n;
    scanf("%d",&n);
    for(int i = 0; i < n; ++i){cin >> dates[i].first.second >> dates[i].first.first;dates[i].second = i;}
    sort(dates, dates + n);
    int arr[n];
    for(int i = 0; i < n; ++i)
    {
        int k = dates[i].first.second;
        while(ismarked[k])k++;
        arr[dates[i].second] = k;
        ismarked[k] = true;
    }
    for(int i = 0; i < n; ++i)cout << arr[i] << endl;   
    return 0;
}
