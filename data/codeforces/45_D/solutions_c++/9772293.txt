#include<bits/stdc++.h>

using namespace std;

bool  a[10000020];
int ans[110];

int main()
{
   //freopen("a.txt","r",stdin);
    memset(a,0,sizeof(a));
    pair< pair<int,int> ,int > p[110];
    int n,l,r;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>l>>r;
       p[i]=make_pair(make_pair(r,l),i);
    }
    sort(p,p+n);
    for(int i=0;i<n;i++)
    {
        for(int j=p[i].first.second;j<=p[i].first.first;j++)
        {
            if(a[j]==false)
            {
                a[j]=true;
                ans[p[i].second]=j;
                break;
            }
        }
    }

    for(int i=0;i<n;i++)cout<<ans[i]<<" ";


}
