#include<bits/stdc++.h>
using namespace std;
#define in intervals
#define se second
#define fi first
pair<pair<int,int>,int> intervals[101];
int hash[10000001]={0};

int main(){
    int n,results[101];
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d%d",&in[i].fi.se,&in[i].fi.fi);
        in[i].se = i;
    }
    sort(in,in+n);
    for(int i=0;i<n;i++){
//      printf("%d-%d,%d\n",in[i].se,in[i].fi.fi,in[i].fi.se);
        for(int j=in[i].fi.se;j<=in[i].fi.fi;j++){
            if(hash[j]==0){
                hash[j]++;
                results[in[i].se]=j;
                break;
            }
        }
    }
    for(int i=0;i<n;i++)
        printf("%d ",results[i]);
    return 0;
}
