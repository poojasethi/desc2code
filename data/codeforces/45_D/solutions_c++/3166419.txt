#include<iostream>
#include<algorithm>

using namespace std;

pair<pair<int,int>,int> arr[101];

int mark[10000001]={0};

int main()
{
    int n,i,j,count=0,res[101];
    cin>>n;
    for (int i=0;i<n;i++)
    {
        cin>>arr[i].first.second>>arr[i].first.first;
        arr[i].second = i;
    }

    sort(arr,arr+n);

     for (int i=0;i<n;i++)
    {
        for(j=arr[i].first.second;j<=arr[i].first.first;j++)
        {
            if(mark[j]==0)
            {
                res[arr[i].second] = j;
                mark[j] = 1;
                break;

            }

        }

    }

     for (int i=0;i<n;i++)
     {
         cout<<res[i]<<" ";
     }
    return 0;
}
