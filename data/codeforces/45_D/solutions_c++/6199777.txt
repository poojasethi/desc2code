#include <cstdio>
#include <algorithm>
#include <vector>
#include <set>
#define pii pair<int,int>
using namespace std;

int n , l , r , ans[101] ;
struct SS{
  int l,r,id ;
  bool operator<(const SS& S) const { return (r!=S.r)?(r<S.r):(l<S.l) ; }
}a[101] ;
set<pii> S ;
set<pii>::iterator it ;
 
int main() 
{
   scanf("%d",&n) ;
   for(int i=0 ; i<n && scanf("%d%d",&a[i].l,&a[i].r) ; i++) a[i].id=i ;
   sort(a,a+n) ;    
 
   int mx=a[n-1].r , ptr=n-1 ;
   for(int i=mx ; i>=1 ; i--) 
   { 
      while(ptr>=0 && a[ptr].r==i) { S.insert(make_pair(a[ptr].l,ptr)) ; ptr-- ; }
      if(!S.size()) continue ;
      it=S.end() ;  it-- ; 
      ans[a[it->second].id]=i ; S.erase(it) ; 
   }    
   for(int i=0 ; i<n ; i++) printf("%d ",ans[i]) ;

 return 0;
}
