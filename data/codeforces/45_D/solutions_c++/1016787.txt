#include <iostream>
#include <algorithm>
#include <vector>
#include <set>
using namespace std;
#define REP(i,a,b)for(i=a;i<b;i++)
#define MP make_pair

vector< pair< pair< int , int > , int > > v;
set< int > vis;
int ans[100],i,n,l,r,j;

int main(){
        cin>>n;
        REP(i,0,n) {
                cin>>l>>r;
                v.push_back(MP(MP(r,l),i));
        }
        sort(v.begin(),v.end());
        REP(i,0,n)
        REP(j,v[i].first.second,v[i].first.first+1) if (!vis.count(j)){
                vis.insert(j);ans[v[i].second]=j;
                break;
        }
        REP(i,0,n) cout<<ans[i]<<' ';
}
