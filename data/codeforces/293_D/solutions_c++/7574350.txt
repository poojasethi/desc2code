const int N=101000,D=2010000;
int n,x[N],y[N];
int ct[D],uph[D],dwh[D];
double cnt,s1,s2,ans;
double solve() {
	double s=0;
	rep(i,0,n) s+=x[i]*y[(i+1)%n]-y[i]*x[(i+1)%n];
	if (s>0) reverse(x,x+n),reverse(y,y+n);
	int mx=*max_element(x,x+n);
	rep(i,0,n) {
		ll x1=x[i],x2=x[(i+1)%n],y1=y[i],y2=y[(i+1)%n];
		if (x1<x2) rep(j,x1,x2+1) uph[j]=(y1*(x2-j)+y2*(j-x1))/(x2-x1);
		else if (x1>x2) rep(j,x2,x1+1) dwh[j]=(y2*(x1-j)+y1*(j-x2)-1)/(x1-x2);
	}
	rep(i,1,mx+1) ct[i]=uph[i]-dwh[i];
	cnt=s2=s1=0;
	rep(i,1,mx+1) cnt+=ct[i],s2+=1.*i*i*ct[i],s1+=1.*i*ct[i];
	return s2*cnt-s1*s1;
}
int main() {
	scanf("%d",&n);
	rep(i,0,n) scanf("%d%d",x+i,y+i);
	int px=*min_element(x,x+n)-1,py=*min_element(y,y+n)-1;
	rep(i,0,n) x[i]-=px,y[i]-=py;
	ans+=solve();
	rep(i,0,n) swap(x[i],y[i]);
	ans+=solve();
	printf("%.10f\n",ans/cnt/(cnt-1));
}
