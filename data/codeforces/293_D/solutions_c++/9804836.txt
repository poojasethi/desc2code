#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
const int N=(int)1e5+5,X=(int)2e6+100;
typedef long long ll;
typedef long double db;
int n,x[N],y[N],cnt[X];

db work(){
	db area=0;
	for(int i=0;i<n;i++) area+=x[i+1]*y[i]-x[i]*y[i+1];
	if(area<0){
		reverse(x,x+n);
		x[n]=x[0];
		reverse(y,y+n);
		y[n]=y[0];
	}
	ll tot=0;
	db ans1=0,ans2=0;
	for(int i=0;i<X;i++) cnt[i]=0;
	for(int i=0;i<n;i++){
		if(x[i]<x[i+1]){
			for(int j=x[i];j<x[i+1];j++)
				cnt[j]+=floor(y[i]+1.*(y[i+1]-y[i])/(x[i+1]-x[i])*(j-x[i])+1e-8);
			if(x[i+1]>=x[(i+2)%n])
				cnt[x[i+1]]+=y[i+1];
		}
		else
		if(x[i]>x[i+1]){
			for(int j=x[i];j>x[i+1];j--)
				cnt[j]-=floor(y[i]+1.*(y[i+1]-y[i])/(x[i+1]-x[i])*(j-x[i])-1e-8);
			if(x[i+1]<=x[(i+2)%n])
				cnt[x[i+1]]-=y[i+1]-1;
		}
	}
	for(int i=0;i<X;i++){
		tot+=cnt[i];
		ans1+=1.*i*cnt[i];
		ans2+=1.*i*i*cnt[i];
	}
	return (ans2*tot-2*ans1*ans1+ans2*tot)/tot/(tot-1);
}

int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++) scanf("%d%d",&x[i],&y[i]),x[i]+=X/2,y[i]+=X/2;
	x[n]=x[0];
	y[n]=y[0];
	db ans=work();
	for(int i=0;i<=n;i++) swap(x[i],y[i]);
	ans+=work();
	printf("%.12lf\n",double(ans/2));
}
