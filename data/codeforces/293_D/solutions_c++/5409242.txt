#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define db double
#define N 2000010
#define eps (1e-11)
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	db x,s,t,sumy,sumyy;
} a[N];
int i,j,k,n,m;
int x[N],y[N];
db l[N],r[N],s,t,an;
inline void work(int x,int y,int xx,int yy) {
	if (x>xx) swap(x,xx),swap(y,yy);
	if (x==xx) {
		l[x]=min(l[x],0.+y);
		r[x]=max(r[x],0.+yy);
		return;
	}
	int i;
	For(i,x,xx) {
		db k=1.*(yy-y)*(i-x)/(xx-x)+y;
		l[i]=min(l[i],k);
		r[i]=max(r[i],k);
	}
}
inline void Do() {
	int i;
	For(i,0,N-1) r[i]=0,l[i]=N;
	For(i,1,n) work(x[i],y[i],x[i+1],y[i+1]);
	m=0;
	For(i,0,N-1) if (l[i]<N) {
		int A=int(l[i]-eps)+1,B=int(r[i]+eps);
		m++; a[m].x=i; a[m].s=B-A+1;
		a[m].t=a[m].x*a[m].s;
	}
	s=t=0;
	For(i,1,m) s+=a[i].s,t+=a[i].t;
	For(i,1,m) an+=a[i].t*a[i].x*(s-a[i].s)-(a[i].t*(t-a[i].t));
}
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%d%d",&x[i],&y[i]),x[i]+=N/2,y[i]+=N/2;
	x[n+1]=x[1],y[n+1]=y[1];
	Do();
	For(i,1,n+1) swap(x[i],y[i]);
	Do();
	an/=s*(s-1);
	printf("%.10lf\n",an);
	//for(;;);
	return 0;
}
