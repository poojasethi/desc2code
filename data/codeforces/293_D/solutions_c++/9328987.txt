#include<cmath>
#include<cstdio>
#include<algorithm>
#define LL long long
using namespace std;
const double eps=1e-8;
struct Point{
	int x,y;
}d[200005];
long double ans,s2,s1,s0;
int n;
void Work()
{
	int t=1,l=1,r=n+1;
	s2=s1=s0=0;
	for(int i=1;i<=n;i++)
		if(d[i].x<d[t].x)
			t=i;
	for(int i=1;i<t;i++)
		d[i+n]=d[i];
	for(int i=1;i<=n;i++)
		d[i]=d[i+t-1];
	while(d[l+1].x==d[1].x)l++;
	while(d[r-1].x==d[1].x)r--;
	l++;r--;
	d[n+1]=d[1];
	for(int x=d[1].x;;x++)
	{
		while(d[l].x<x&&l<=r)l++;
		if(l>r)break;
		while(d[r].x<x)r--;
		double ly=((double)d[l].y-d[l-1].y)/(d[l].x-d[l-1].x)*(x-d[l-1].x)+d[l-1].y;
		double ry=((double)d[r].y-d[r+1].y)/(d[r].x-d[r+1].x)*(x-d[r+1].x)+d[r+1].y;
		if(ly>ry)swap(ly,ry);
		ans+=s2*(floor(ry+eps)-ceil(ly-eps)+1);
		s0+=floor(ry+eps)-ceil(ly-eps)+1;
		s1+=s0;
		s2+=s1*2-s0;
	}
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d%d",&d[i].x,&d[i].y);
	Work();
	for(int i=1;i<=n;i++)
		d[i].x=-d[i].x;
	Work();
	for(int i=1;i<=n;i++)
		swap(d[i].x,d[i].y);
	Work();
	for(int i=1;i<=n;i++)
		d[i].x=-d[i].x;
	Work();
	printf("%.20Lf\n",(double)(ans*0.5/(s0*s0-s0)));
}
