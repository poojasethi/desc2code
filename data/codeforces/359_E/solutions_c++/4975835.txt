#include<iostream>
#include<cstdio>
#include<cstring>
#include<set>
#include<string>
#include<algorithm>
using namespace std;

const int MAXN=505;

int N,tot,dx[4]={0,0,-1,1},dy[4]={-1,1,0,0},G[MAXN][MAXN],vis[MAXN][MAXN];

string ans;

int isok1(int x,int y)
{
    for(int i=y-1;i>0;--i)
    {
        if(vis[x][i]) return 0;
        if(G[x][i]) return 1;
    }
    return 0;
}

int isok2(int x,int y)
{
    for(int i=y+1;i<=N;++i)
    {
        if(vis[x][i]) return 0;
        if(G[x][i]) return 1;
    }
    return 0;
}

int isok3(int x,int y)
{
    for(int i=x-1;i>0;--i)
    {
        if(vis[i][y]) return 0;
        if(G[i][y]) return 1;
    }
    return 0;
}

int isok4(int x,int y)
{
    for(int i=x+1;i<=N;++i)
    {
        if(vis[i][y]) return 0;
        if(G[i][y]) return 1;
    }
    return 0;
}

void dfs(int x,int y)
{
    vis[x][y]=1;
    if(!G[x][y]) ans+='1';
    else --tot;
    if(isok1(x,y)) {ans+='L';dfs(x,y-1);ans+='R';}
    if(isok2(x,y)) {ans+='R';dfs(x,y+1);ans+='L';}
    if(isok3(x,y)) {ans+='U';dfs(x-1,y);ans+='D';}
    if(isok4(x,y)) {ans+='D';dfs(x+1,y);ans+='U';}
    ans+='2';
}

int main()
{
    int x,y;
    cin>>N>>x>>y;
    for(int i=1;i<=N;++i)
    {
        for(int j=1;j<=N;++j)
        {
            cin>>G[i][j];
            if(G[i][j]) ++tot;
        }
    }
    dfs(x,y);
    if(tot) cout<<"NO"<<endl;
    else cout<<"YES"<<endl<<ans<<endl;
    return 0;
}
