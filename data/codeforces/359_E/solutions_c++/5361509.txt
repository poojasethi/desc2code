#include <cstdio>
const char c[6]={'R','L','D','U','1','2'};
const int D[4][2]={{0,1},{0,-1},{1,0},{-1,0}};
int N,X,Y,S,T,a[505][505],q[3000005];
bool b[505][505];

bool ok(int x,int y,int d)
{
	for (x+=D[d][0],y+=D[d][1]; 1<=x&&x<=N&&1<=y&&y<=N; x+=D[d][0],y+=D[d][1]) if (a[x][y]) return !b[x][y];
	return 0;
}

void dfs(int x,int y)
{
	b[x][y]=1;
	if (!a[x][y]) a[x][y]=1,S++,q[++T]=4;
	for (int i=0; i<4; i++) if (ok(x,y,i)) q[++T]=i,dfs(x+D[i][0],y+D[i][1]),q[++T]=i^1;
	a[x][y]=0,S--,q[++T]=5;
}

int main()
{
	scanf("%d%d%d",&N,&X,&Y);
	for (int i=1; i<=N; i++)
		for (int j=1; j<=N; j++) scanf("%d",&a[i][j]),S+=a[i][j];
	dfs(X,Y);
	if (S) {puts("NO"); return 0;}
	puts("YES");
	for (int i=1; i<=T; i++) putchar(c[q[i]]);
	puts("");
}
