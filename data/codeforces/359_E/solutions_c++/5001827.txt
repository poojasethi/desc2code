#include <cstdio>
#include <cstring>
using namespace std;
int arr[555][555];
bool vis[555][555];
int dir[4][2]={0,1,0,-1,1,0,-1,0};
char path[4000000];
char forward[4]={'R','L','D','U'};
char back[4]={'L','R','U','D'};
int idx,total,n;
bool isout(int x,int y){
	return (x<1||x>n||y<1||y>n);
}
bool check(int x,int y,int d){
	do{
		if(isout(x,y)||vis[x][y]) return false;
		if(arr[x][y]) return true;
		x+=dir[d][0];y+=dir[d][1];
	}while(true);
}
void dfs(int x,int y){
	if(!arr[x][y]){
		total++;
		path[idx++]='1';
	}
	vis[x][y]=true;
	for(int i=0;i<4;i++){
		int cx=x+dir[i][0];
		int cy=y+dir[i][1];
		if(isout(cx,cy)||vis[cx][cy]||!check(cx,cy,i)) continue;
		path[idx++]=forward[i];
		dfs(cx,cy);
		path[idx++]=back[i];
	}
	path[idx++]='2';
	total--;
}
int main(){
	int xx,yy;
	scanf("%d%d%d",&n,&xx,&yy);
	idx=0;
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++){
			scanf("%d",&arr[i][j]);
			total+=arr[i][j];
		}
	}
	memset(vis,0,sizeof vis);
	dfs(xx,yy);
	if(total==0){
		puts("YES");
		puts(path);
	}
	else puts("NO");
	return 0;
}
