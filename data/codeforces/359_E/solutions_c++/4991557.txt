#include <iostream>
#include <cstring>
using namespace std;
int room[500][500];
bool visited[500][500];
int total = 0;
string ans = "";
int dx[] = {-1,0,1,0};
int dy[] = {0,1,0,-1};
char dir[] = "URDL";
int n;

bool CouldMove(int x,int y,int d) {
	x += dx[d];
	y += dy[d];
	while(x >= 0 && x < n && y >= 0 && y < n && !visited[x][y]) {
		if (room[x][y] == 1) {
			return true;
		}
		x += dx[d];
		y += dy[d];
	}
	return false;
}
void dfs(int x,int y) {
	if (x < 0 || x == n || y < 0 || y == n || visited[x][y]) {
		return ;
	}
	visited[x][y] = true;
	if (room[x][y] == 0) {
		room[x][y] = 1;
		ans += '1';
		total ++;
	}
	for (int d = 0 ; d < 4 ; d ++) {
		if (CouldMove(x , y , d)) {
			ans += dir[d];
			dfs(x + dx[d] , y + dy[d]);
			ans += dir[(d+2)%4];
		}
	}
	ans += '2';
	total --;
}
int main() {
	memset(visited , false , sizeof(visited));
	int x0 , y0;
	cin >> n >> x0 >> y0;
	for (int i = 0 ; i < n ; i ++) {
		for (int j = 0 ; j < n ; j ++) {
			cin >> room[i][j];
			total += room[i][j];
		}
	}
	dfs(x0 - 1 , y0 - 1);
	if (total == 0) {
		cout << "YES" << endl;
		cout << ans << endl;
	} else {
		cout << "NO" << endl;
	}
	return 0;
}
