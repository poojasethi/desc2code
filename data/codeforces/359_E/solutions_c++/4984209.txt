#include<cstdio>
#include<cstring>
int dx[4] = {1,0,-1,0};
int dy[4] = {0,1,0,-1};
int op[4] = {2,3,0,1};
int left,n,id;
int a[505][505];
char s[3000005];
inline char get(int x)
{
    if(x == 0)return 'D';
    if(x == 1)return 'R';
    if(x == 2)return 'U';
    return 'L';
}
bool ok(int x,int y,int d)
{
    while(true)
    {
        x+=dx[d],y+=dy[d];
        if(x<1||y<1||x>n||y>n)return false;
        if(a[x][y] == 2)return false;
        if(a[x][y] == 1)return true;
    }
}
void dfs(int x,int y)
{
    if(!a[x][y])
    {
        s[id++] = '1';
        left++;
    }
    a[x][y] = 2;
    for(int i = 0;i<4;i++)
    {
        if(ok(x,y,i))
        {
            s[id++] = get(i);
            dfs(x+dx[i],y+dy[i]);
            s[id++] = get(op[i]);
        }
    }
    s[id++] = '2';
    left--;
}
int main()
{
    int x,y;
    while(scanf("%d%d%d",&n,&x,&y)!=EOF)
    {
        left = id = 0;
        for(int i = 1;i<=n;i++)
        {
            for(int j = 1;j<=n;j++)
            {
                scanf("%d",&a[i][j]);
                left+=a[i][j];
            }
        }
        dfs(x,y);
        if(left)printf("NO\n");
        else printf("YES\n%s\n",s);
    }
    return 0;
}
