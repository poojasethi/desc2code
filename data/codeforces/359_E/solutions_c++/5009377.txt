#include <iostream>
using namespace std;
char a[3*1000005];
int tot=0;
char mp[502][502];
int p[502][502];
int dx[4]={0,1,0,-1};
int dy[4]={-1,0,1,0};
char fxx[6]="LDRU";
char fyy[6]="RULD";

int n;
int seach(int x,int y,int dx,int dy){
    if(x==0)return 0;
    if(x==n+1)return 0;
    if(y==0)return 0;
    if(y==n+1)return 0;
    if(mp[x][y]=='1')return 1;
    return seach(x+dx,y+dy,dx,dy);
}
int dfs(int x,int y) {
    if(mp[x][y]=='0'){a[tot++]=mp[x][y]='1';}
    p[x][y]=1;
    for(int i=0;i<4;i++){
        int fx=x+dx[i];
        int fy=y+dy[i];
        if(fx==0)continue;
        if(fx==n+1)continue;
        if(fy==0)continue;
        if(fy==n+1)continue;
        if(!p[fx][fy]&&seach(fx,fy,dx[i],dy[i])){

            a[tot++]=fxx[i];
            dfs(fx,fy);
            a[tot++]=fyy[i];

        }
    }
    a[tot++]=2+(mp[x][y]='0');
}
int check(){
    for(int i=1;i<=n;i++)
    for(int j=1;j<=n;j++){
        if(mp[i][j]=='1')return 0;
    }
    return 1;
}
int main(){
    int x0,y0;
    cin >> n >> x0 >> y0;
    for(int i=1;i<=n;i++)
    for(int j=1;j<=n;j++)cin >> mp[i]+j;
    dfs(x0,y0);
    if(check()){
        cout << "YES"<<endl;
        a[tot]=0;
        cout << a<<endl;
    }
    else cout << "NO";
}
