#include<bits/stdc++.h>
using namespace std;

int n,x,y;
char s[3333333];
int id = 0;
bool in(int x,int y){
	return 0 <= x && x < n && 0 <= y && y < n;
}
char c[4] = {'L','R','U','D'};
char nc[4] = {'R','L','D','U'};
int dir[4][2] = {
	0,-1,
	0,1,
	-1,0,
	1,0
};
int mp[555][555];
bool flag[555][555];
void dfs(int x,int y){
	if(mp[x][y]==0){
		mp[x][y] = 1;
		s[id++] = '1';
	}
	flag[x][y] = 1;
	for(int i = 0; i < 4; i++){//four dir
		int nx = x ,ny = y ;
		while(1){
			nx += dir[i][0];
			ny += dir[i][1];
			if(!in(nx,ny) || flag[nx][ny])break;
			if(mp[nx][ny] == 1)break;
		}
		if(!in(nx,ny) || flag[nx][ny])continue;
		if(mp[nx][ny] == 1){
			s[id++] = c[i];
			dfs(x+dir[i][0],y+dir[i][1]);
			s[id++] = nc[i];
		}
	}
	//flag[x][y] = 0;
	if(mp[x][y]==1){
		mp[x][y] = 0;
		s[id++] = '2';
	}
}
bool check(){
	for(int i = 0; i < n; i++)
		for(int j =0 ; j < n; j++)
			if(mp[i][j]==1)return false;
	return true;
}
int main(){
	while(scanf("%d%d%d",&n,&x,&y)==3){
		x--,y--;
		for(int i = 0; i < n; i++)
			for(int j = 0; j < n; j++)
				scanf("%d",&mp[i][j]);
		id = 0;
		memset(flag,0,sizeof(flag));
		dfs(x,y);
		s[id++] = '\0';
		if(check())printf("YES\n%s\n",s);
		else puts("NO");
	}	
	return 0;
}
