#include<cstdio>
using namespace std;
int N,X,Y,i,j,P;
int Dx[]={-1,0,0,1};
int Dy[]={0,-1,1,0};
char D[]={'U','L','R','D'};
bool A[505][505],viz[505][505];
char S[3000005];
bool Check(int X,int Y)
{
    return (X>=1 && X<=N && Y>=1 && Y<=N);
}
void DFS(int X,int Y)
{
    if(!A[X][Y]) {S[++P]='1'; A[X][Y]=1;}
    for(int k=0;k<4;k++)
    {
        bool ok=0;
        int NewX=X+Dx[k]; int NewY=Y+Dy[k];
        if(Check(NewX,NewY) && !viz[NewX][NewY])
        {
            for(int i=1;Check(X+i*Dx[k],Y+i*Dy[k]);i++)
                if(A[X+i*Dx[k]][Y+i*Dy[k]]) {ok=1; break;}
            if(ok)
            {
                S[++P]=D[k];
                viz[NewX][NewY]=1;
                DFS(NewX,NewY);
                S[++P]=D[3-k];
            }
        }
    }
    S[++P]='2'; A[X][Y]=0;
}
int main()
{
    //freopen("test.in","r",stdin);
    //freopen("test.out","w",stdout);
    scanf("%d%d%d",&N,&X,&Y);
    for(i=1;i<=N;i++)
        for(j=1;j<=N;j++) scanf("%d",&A[i][j]);
    viz[X][Y]=1; DFS(X,Y);
    for(i=1;i<=N;i++)
        for(j=1;j<=N;j++)
            if(A[i][j]) {printf("NO\n"); return 0;}
    printf("YES\n"); printf("%s\n",S+1);
    return 0;
}
