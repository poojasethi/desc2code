#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
#include<vector>
#include<math.h>
#include<queue>
using namespace std;
typedef long long ll;
const int inf=1<<30;
#define N 510

int dd[4][2]={{1,0},{-1,0},{0,-1},{0,1}};
int a[N][N],vis[N][N],n;
int check(int x,int y,int i){
    while(1){
        x+=dd[i][0];
        y+=dd[i][1];
        if(x<0||y<0||x>=n||y>=n) return 0;
        if(vis[x][y]) return 0;
        if(a[x][y]) return 1;
    }
}
char dir1[4]={'D','U','L','R'};
char dir2[4]={'U','D','R','L'};
int sum,size;
char ans[3000010];
void dfs(int x,int y){
    vis[x][y]=1;
    if(a[x][y]==0) sum++,ans[size++]='1';
    for(int i=0;i<4;i++){
        if(check(x,y,i)){
            ans[size++]=dir1[i];
            dfs(x+dd[i][0],y+dd[i][1]);
            ans[size++]=dir2[i];
        }
    }
    sum--,ans[size++]='2';
}
int main()
{
    int x,y;
    scanf("%d%d%d",&n,&x,&y);
    x--,y--;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++){
            scanf("%d",&a[i][j]);
            if(a[i][j]==1) sum++;
        }
    dfs(x,y);
    if(sum==0){
        ans[size]=0;
        printf("YES\n%s\n",ans);
    }else printf("NO\n");
}
