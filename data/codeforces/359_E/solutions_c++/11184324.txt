#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>

using namespace std;

int dx[]={0,1,-1,0};
int dy[]={1,0,0,-1};
char d[]={'R','D','U','L'};
int n,x,y;
int a[505][505];
int f[505][505];
int ans0;
char ans[3000005];
int sum;
bool pd(int x,int y,int i)
{
	while (true)
	{
		x+=dx[i];y+=dy[i];
		if (x<0) return false;
		if (x>n) return false;
		if (y<0) return false;
		if (y>n) return false;
		if (f[x][y]>0) return false;
		if (a[x][y]>0) return true;
	}
}
void dfs(int x,int y)
{
	if (a[x][y]==0)
	{
		++ans0;
		ans[ans0]='1';
		sum++;
	}
	f[x][y]=1;
	for (int i=0;i<4;++i)
		if (pd(x,y,i))
		{
			++ans0;ans[ans0]=d[i];
			dfs(x+dx[i],y+dy[i]);
			++ans0;ans[ans0]=d[3-i];
		}
	++ans0;ans[ans0]='2';
	sum--;
}
int main()
{
	scanf("%d%d%d",&n,&x,&y);
	for (int i=1;i<=n;++i)
		for (int j=1;j<=n;++j) scanf("%d",&a[i][j]);
	for (int i=1;i<=n;++i)
		for (int j=1;j<=n;++j) sum+=a[i][j];
	dfs(x,y);
	if (sum>0) 
	{
		printf("NO\n");
	}
	else
	{
		printf("YES\n");
		printf("%s\n",ans+1);
	}
	return 0;
}
