using namespace std;


int n, sum = 0;
bool v[505][505] = {0}, a[505][505] = {0};
    
int dx[5] = {0, 0, 1, -1};
int dy[5] = {1, -1, 0, 0};
string dir = "RLDU", ans = "";

bool ch(int x, int y, int i) 
{
    for (int l = 1; l <= n; l ++) 
    {
        int x_ = x + dx[i] * l, y_ = y + dy[i] * l;
        if (x_ < 1 || y_ < 1 || x_ > n || y_ > n || v[x_][y_]) return false;
        if (a[x_][y_]) return true;
    }
    return false;
}

void dfs(int x, int y) {
    v[x][y] = 1;
    if (a[x][y]) sum --;
    else ans += '1';
    for (int i = 0; i < 4; i ++) {
        if (ch(x, y, i))
        {
            ans += dir[i];
            dfs(x + dx[i], y + dy[i]);
            ans += dir[i ^ 1];
        }
    }
    ans += '2';
}

int main(){
  #ifndef ONLINE_JUDGE
    freopen ("input.txt", "rt", stdin);
    freopen ("output.txt", "wt", stdout);
  #endif
  int x0 = 0, y0 = 0;
  cin >> n >> x0 >> y0;
  for(int i = 1; i <= n; i ++) 
    for(int j = 1; j <= n; j ++) 
        cin >> a[i][j], sum += a[i][j];
  dfs(x0, y0);
  if (sum) {
    cout << "NO";
    return 0;
  }
  cout << "YES\n" << ans;


}
