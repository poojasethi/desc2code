const int maxn=501;
int n,vis[maxn][maxn],a[maxn][maxn],x,y,s,cnt;
char p[3000010];
int d[4][2]={0,1,1,0,-1,0,0,-1};

bool J(int x,int y,int i)
{
	while(true)
	{
		x+=d[i][0]; y+=d[i][1];
		if(!x||!y||x>n||y>n) return 0;
		if(vis[x][y]) return 0;
		if(a[x][y]) return 1;
	}
}

void dfs(int x,int y)
{
	if(!a[x][y]) p[cnt++]='1',s++;
	vis[x][y]=1;
	for(int i=0;i<4;i++)
		if(J(x,y,i))
		{
			p[cnt++]= (!i?'R':i==1?'D':i==2?'U':'L');
			dfs(x+d[i][0],y+d[i][1]);
			p[cnt++]= (!i?'L':i==1?'U':i==2?'D':'R');
		}
	p[cnt++]='2',s--;
}
