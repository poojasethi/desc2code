#include<stdio.h>
#include<string.h>
int vis[510][510];
int n,x0,y0;
int a[510][510],b[510][510];
char ans[510*510*4];
int tot;
int dx[4]={1,0,-1,0};
int dy[4]={0,1,0,-1};
char df[4]={'D','R','U','L'};
void dfs(int x,int y){
	int i,j;
	int tx,ty;
	vis[x][y]=1;
	if (b[x][y]==0) ans[tot++]='1';
	b[x][y]=1;
	for (i=0;i<4;i++){
		tx=x+dx[i];
		ty=y+dy[i];
		if (tx<=0||tx>n||ty<=0||ty>n) continue;
		if (vis[tx][ty]) continue;
		int flag=0;
		if (i==0){
			for (j=tx;j<=n;j++) if (b[j][y]==1) {flag=1;break;}
		}
		if (i==2){
			for (j=tx;j>=1;j--) if (b[j][y]==1) {flag=1;break;}
		}
		if (i==1){
			for (j=ty;j<=n;j++) if (b[x][j]==1) {flag=1;break;}
		}
		if (i==3){
			for (j=ty;j>=1;j--) if (b[x][j]==1) {flag=1;break;}
		}
		if (!flag) continue;
		ans[tot++]=df[i];
		dfs(tx,ty);						
		ans[tot++]=df[(i+2)%4];
	}
	b[x][y]=0;
	ans[tot++]='2';
}
int check(){
	int i,j;
	for (i=1;i<=n;i++)
		for (j=1;j<=n;j++)
			if (a[i][j]==1&&vis[i][j]==0) return 0;
	return 1;
}
int main(){
	//freopen("e.in","r",stdin);
	int i,j;
	scanf("%d%d%d",&n,&x0,&y0);
	for (i=1;i<=n;i++)
		for (j=1;j<=n;j++) scanf("%d",&a[i][j]),b[i][j]=a[i][j];
	dfs(x0,y0);
	if (check()){
		printf("YES\n");
		puts(ans);
	}else{
		puts("NO\n");
	}
	//while (1);
	return 0;
}
