#include <cstdio>
#include <iostream>
#include <cstring>
#include <string>

using namespace std;

int cx[] = {0, -1, 0, 1};
int cy[] = {-1, 0, 1, 0};
char c[] = {'L', 'U', 'R', 'D'};

int n, x, y;
bool on[555][555];
bool gone[555][555];
string trace;

void go(int x, int y) {
	gone[x][y] = 1;

	if (!on[x][y]) {
		trace += '1';
		on[x][y] = 1;
	}

	for(int k = 0; k < 4; k++) {
		int xx, yy;
		bool ok = 0;
		for(xx = x+cx[k], yy = y+cy[k]; !gone[xx][yy]; xx+=cx[k], yy+=cy[k])
			if (on[xx][yy]) ok = 1;
		if (ok) {
			trace += c[k];
			go(x+cx[k], y+cy[k]);
			trace += c[(k+2)%4];
		}
	}

	if (on[x][y]) {
		trace += '2';
		on[x][y] = 0;
	}
}

int main() {

	scanf("%d%d%d", &n, &x, &y);
	for(int i = 1; i <= n; i++)
		for(int j = 1; j <= n; j++) {
			int k;
			scanf("%d", &k);
			on[i][j] = k;
		}

	trace = "";
	memset(gone, 1, sizeof gone);
	for(int i = 1; i <= n; i++)
		for(int j = 1; j <= n; j++) gone[i][j] = 0;
	go(x, y);
	bool ok = 1;
	for(int i = 1; i <= n; i++)
		for(int j = 1; j <= n; j++)
			if (on[i][j]) ok = 0;
	if (ok) 
		printf("YES\n%s", trace.c_str());
	else printf("NO");
	return 0;
}
