#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 510
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
const int a1[4]={1,-1,0,0},a2[4]={0,0,1,-1};
int i,j,k,n,m,s,sum,x,y;
int a[N][N],f[N][N];
char p[3000010];
inline bool judge(int x,int y,int i) {
	for (;;) {
		x+=a1[i],y+=a2[i];
		if (!x||!y||x>n||y>n) return 0;
		if (f[x][y]) return 0;
		if (a[x][y]) return 1;
	}
}
void dfs(int x,int y) {
	int i;
	if (!a[x][y]) p[++s]='1',sum++;
	f[x][y]=1;
	For(i,0,3) if (judge(x,y,i)) {
		p[++s]=!i?'D':i==1?'U':i==2?'R':'L';
		dfs(x+a1[i],y+a2[i]);
		p[++s]=!i?'U':i==1?'D':i==2?'L':'R';
	}
	p[++s]='2'; sum--;
}
int main() {
	scanf("%d%d%d",&n,&x,&y);
	For(i,1,n)For(j,1,n) scanf("%d",&a[i][j]),sum+=a[i][j];
	dfs(x,y);
	if (sum) printf("NO\n");
	else printf("YES\n%s\n",p+1);
	//for(;;);
	return 0;
}
