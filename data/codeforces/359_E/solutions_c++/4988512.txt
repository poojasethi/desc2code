using namespace std;

#define fst first
#define snd second
#define mp make_pair
#define pb push_back
#define ll long long

int a[1111][1111],n,x,y;
bool used[1111][1111];

string ans;

void dfs(int x, int y)
{
	used[x][y]=true;
	if (!a[x][y])
	{
		ans+='1';
		a[x][y]=1;
	}
	bool ok=false;
	for(int i=x+1; i<n; i++)
	 if (a[i][y]) ok=true;
	if (ok && !used[x+1][y])
	{
		ans+='D';
		dfs(x+1,y);
		ans+='U';
	} 
	ok=false;
	for(int i=x-1; i>=0; i--)
	 if (a[i][y]) ok=true;
	if (ok && !used[x-1][y])
	{
		ans+='U';
		dfs(x-1,y);
		ans+='D';
	}
	ok=false;
	for(int i=y+1; i<n; i++)
	 if (a[x][i]) ok=true;
	if (ok && !used[x][y+1])
	{
		ans+='R';
		dfs(x,y+1);
		ans+='L';
	} 
	ok=false;
	for(int i=y-1; i>=0; i--)
	 if (a[x][i]) ok=true;
	if (ok && !used[x][y-1])
	{
		ans+='L';
		dfs(x,y-1);
		ans+='R';
	}  
	ans+='2';
	a[x][y]=0;
}

int main()
{
	cin >> n >> x >> y;
	x--; y--;
	for(int i=0; i<n; i++)
	 for(int j=0; j<n; j++) cin >> a[i][j];
	 dfs(x,y);
	 
	for(int i=0; i<n; i++)
	 for(int j=0; j<n; j++)
	  if (a[i][j]) 
	  {
			cout << "NO";
			return 0;
	  }
	  cout << "YES" << endl;
	  cout << ans << endl;
	  cin >> n;
	return 0;
}
