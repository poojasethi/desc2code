#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
	long long int a,b,c;
	while(cin>>a>>b>>c)
	{
		if(a==0&&b==0&&c==0)continue;
		long long int max=a,min=a;
		if(max<b)max=b;
		if(max<c)max=c;
		if(min>b)min=b;
		if(min>c)min=c;
		long long int t=a+b+c-max-min;
		if((t-min)%2==0)
			cout<<t<<endl;
		else
			cout<<max<<endl;
	}	
return 0;
}
