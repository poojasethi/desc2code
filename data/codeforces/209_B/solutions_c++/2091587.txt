#include<cstdio>
#include<algorithm>
long long s[3];
int main(){
	long long sum=(1LL<<35);
	scanf("%I64d%I64d%I64d",s,s+1,s+2);
	std::sort(s,s+3);
	if((s[0]&1)==(s[1]&1))
		if(sum>s[1])sum=s[1];
	if((s[2]&1)==(s[1]&1))
		if(sum>s[2])sum=s[2];
	if((s[0]&1)==(s[2]&1))
		if(sum>s[2])sum=s[2];
	printf("%I64d\n",sum);
}
