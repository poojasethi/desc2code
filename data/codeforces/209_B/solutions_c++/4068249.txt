#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long LL;
int main(void) {
LL a, b, c;
int ua, ub, uc;
scanf("%I64d%I64d%I64d", &a, &b, &c);
if(a>b) swap(a,b); if(a>c) swap(a,c); if(b>c) swap(b,c);
ua=a%2,ub=b%2,uc=c%2;
if(ua==ub&&ub==uc) printf("%I64d\n", b);
else if(ua==ub) printf("%I64d\n", max(a,b));
else if(ub==uc) printf("%I64d\n", max(b,c));
else printf("%I64d\n", max(a,c));
return 0;
}
