#include <cstdio>
#include <algorithm>
using namespace std;

int main() {
   unsigned int a[3];
   scanf("%u%u%u", a, a + 1, a + 2);
   unsigned int ans = 0xffffffff;
if ((a[0] + a[1]) % 2 == 0)
   ans = min(ans, max(a[0], a[1]));
if ((a[1] + a[2]) % 2 == 0)
   ans = min(ans, max(a[1], a[2]));
if ((a[0] + a[2]) % 2 == 0)
   ans = min(ans, max(a[0], a[2]));
printf("%u\n", ans);
}
