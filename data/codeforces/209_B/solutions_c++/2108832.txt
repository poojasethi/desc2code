#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main () {
    vector<long long> nums(3);
    for (int i=0; i<3; i++) cin >> nums[i]; 
    sort(nums.begin(), nums.end());
    if (nums[0]%2 == nums[1]%2) 
       cout << nums[1];              
    else cout << nums[2];   
    return 0;
}
