#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	__int64 a[3],s[3],ans;
	scanf("%I64d%I64d%I64d",&a[0],&a[1],&a[2]);
	sort(a,a+3);
	s[0]=a[0]+a[1];
	s[1]=a[0]+a[2];
	s[2]=a[1]+a[2];
	sort(s,s+3);
	if(s[0]%2==0)
		ans=s[0]/2+s[0]/2-a[0];
	else if(s[1]%2==0)
		ans=s[1]/2+s[1]/2-a[0];
	else 
		ans=s[2]/2+s[2]/2-a[1];
	printf("%I64d\n",ans);
}
