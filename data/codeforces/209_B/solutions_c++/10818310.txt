#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<math.h>
using namespace std;
int main()
{
   __int64 a[3],t,i,j;
   while(scanf("%I64d%I64d%I64d",&a[0],&a[1],&a[2])!=EOF)
   {
   	sort(a,a+3);
   	if((a[1]-a[0])%2==0){
	   	t=a[1];
	   }
	   else if((a[2]-a[1])%2==0)t=a[2];
	   else if((a[2]-a[0])%2==0)t=a[2];
   printf("%I64d\n",t);
   }	
   return 0;
}
