#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
ll a[3];
int main(){
	ll ans;
	while(scanf("%d%d%d",&a[0],&a[1],&a[2])!=EOF){
		sort(a,a+3);
		ans=a[2];
		if((a[0]+a[1])%2==0){
			ans=min(ans,a[1]);
		}
		if((a[1]+a[2])%2==0){
			ans=min(ans,a[2]);
		}
		if((a[0]+a[2])%2==0){
			ans=min(ans,a[2]);
		}
		cout<<ans<<endl;
	}
}
