#include<bits/stdc++.h>
using namespace std;

int main() {
    long long ans,a,b,c;
    scanf("%I64d %I64d %I64d",&a,&b,&c);
    ans=max(max(a,b),c);
    if((a&1)==(b&1)) ans=min(max(a,b),ans);
    if((b&1)==(c&1)) ans=min(max(b,c),ans);
    if((c&1)==(a&1)) ans=min(max(c,a),ans);
    printf("%I64d\n",ans);
    return 0;
}
