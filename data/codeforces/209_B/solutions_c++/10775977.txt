#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
__int64 c[5];

int main()
{
	scanf("%I64d%I64d%I64d",&c[1],&c[2],&c[3]);
	sort(c+1,c+1+3);
	if((c[1]+c[2])%2) printf("%I64d\n",c[3]);
	else printf("%I64d\n",c[2]);
	return 0;
}
