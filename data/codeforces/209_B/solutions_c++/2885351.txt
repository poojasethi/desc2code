#include <cstdio>
#include <algorithm>

using namespace std;

long long a, b, c;
long long best = (long long) 1e32;

int main()
{
    scanf("%I64d %I64d %I64d", &a, &b, &c);
    if (a % 2 == b % 2) best = min (best, max (a, b));
    if (a % 2 == c % 2) best = min (best, max (a, c));
    if (b % 2 == c % 2) best = min (best, max (b, c));
    printf("%I64d\n", best);
}
