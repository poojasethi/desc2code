#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
using namespace std;
long long ans=0,a,b,c;
void work(long long a,long long b,long long c){
	if(b<c)swap(b,c);
	b-=c;
	a+=c;
	if(b&1)ans=max(ans,1ll);
	else ans=max(ans,a);
}
int main(){
	scanf("%I64d%I64d%I64d",&a,&b,&c);
	work(a,b,c);
	work(b,a,c);
	work(c,a,b);
	printf("%I64d\n",a+b+c-ans);
}
