#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;
long long ans,a[4];
int main()
{
    cin>>a[0]>>a[1]>>a[2];
    sort(a,a+3);
    ans=1000000000000000LL;
    if((a[0]&1)==(a[1]&1))ans=a[1];
    else ans=a[2];
    cout<<ans<<endl;
    return 0;
}
