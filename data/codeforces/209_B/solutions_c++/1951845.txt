#include<iostream>
#include<string>
  using namespace std;
 // long long a[1100000]={0},b[1100000]={0};

int main()
{
	
 
    long long a,b,c;
     
	while(cin>>a>>b>>c)
	{
		if(a<b) {long long  t=a;a=b;b=t;}
		if(a<c) {long long  t=a;a=c;c=t;}
		if(b<c) {long long  t=b;b=c;c=t;}

	
	if(b%2==c%2) cout<<b<<endl;
	else
		cout<<a<<endl;

	}                   

}
