#include<stdio.h>
__int64 a,b,c,ans;
__int64 min(__int64 x,__int64 y){
	return x<y?x:y;
}
__int64 max(__int64 x,__int64 y){
	return x>y?x:y;
}
int main(){
	scanf("%I64d%I64d%I64d",&a,&b,&c);
	ans=max(a,max(b,c));
	if((a-b)%2==0)ans=max(a,b);
	if((a-c)%2==0)ans=min(max(a,c),ans);
	if((c-b)%2==0)ans=min(max(c,b),ans);
	printf("%I64d\n",ans);
}
