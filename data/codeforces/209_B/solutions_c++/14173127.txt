#include <iostream>
#include <algorithm>
#define ll long long

using namespace std;

int main(){
    ll a,b,c,ans;
	cin>>a>>b>>c;
	ans=max(a,max(b,c));
	if((a-b)%2==0)ans=max(a,b);
	if((a-c)%2==0)ans=min(max(a,c),ans);
	if((c-b)%2==0)ans=min(max(c,b),ans);
	cout<<ans<<endl;
	return 0;
}
