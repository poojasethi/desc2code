#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;

string a[910], s, e;
int n, m, k, L;

int main(){
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    cin>>n>>m>>k;
    L=n*m;
    for(int needs=0; needs<k; needs++){
        int x, y, tmp;
        cin>>tmp;
        if(tmp==-1){
            cin>>s;
            for(int i=0; i<L; i++)
                if(a[i]==s){
                    cout<<(i/m)+1<<" "<<i%m+1<<endl;
                    a[i]=e;
                    goto hell;
                }
            cout<<-1<<" "<<-1<<endl;
            hell:;
        }else{
            cin>>x>>y>>s;
            for(int i=(x-1)*m+y-1; i<L; i++)
                if(a[i]==e){
                    a[i]=s;
                    break;
                }
        }
    }
    return 0;
}
