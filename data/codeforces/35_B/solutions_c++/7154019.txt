#include<iostream>
#include<cstdio>
using namespace std;
string s[100][100]={};
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,m,k;
	cin>>n>>m>>k;
	while(k--)
	{
		int w;
		cin>>w;
		if(w==-1)
		{
			string t;
			cin>>t;
			for(int i=0;i<n;i++)for(int j=0;j<m;j++)
			{
				if(s[i][j]==t)
				{
					s[i][j]="";
					cout<<i+1<<" "<<j+1<<endl;
					w=1;
				}
			}
			if(w==-1)cout<<-1<<" "<<-1<<endl;
		}
		else
		{
			int x,y;
			string t;
			cin>>x>>y>>t;
			while(s[x-1][y-1]!="")
			{
				if(y<m)y++;
				else y=1,x++;
			}
			if(y<=m&&x<=n)s[x-1][y-1]=t;
		}
	}
	return 0;
}
