#include<stdio.h>
#include<string.h>
	
char s[51][51][21],op[5],st[21];
bool emp[51][51];
int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	memset(emp,0,sizeof(emp));
	int i,j,f,k,n,m,t,x,y;
	scanf("%d%d%d",&n,&m,&k);
	for(t=0;t<k;t++){
		scanf("%s",&op);
		if(op[0]=='+'){
			scanf("%d%d%s",&x,&y,st);
			for(f=1,i=x;i<=n&&f;i++){
				for(j=(i==x?y:1);j<=m&&f;j++){
					if(!emp[i][j]){
						emp[i][j]=1;
						strcpy(s[i][j],st);
						f=0;
						goto out1;
					}
				}
			}
out1:		;
//			printf("i=%d j=%d x=%d y=%d\n",i,j,x,y);
		}
		else{
			scanf("%s",st);
			for(i=f=1;i<=n&&f;i++){
				for(j=1;j<=m&&f;j++){
					if(emp[i][j]&&strcmp(s[i][j],st)==0){
						emp[i][j]=0;
						f=0;
						goto out2;
					}
				}
			}
out2:		if(f) printf("-1 -1\n");
			else printf("%d %d\n",i,j);
		}
	}
	return 0;
}
