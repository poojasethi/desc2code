#define N 655360
#define M 1000000007
#define ll long long
#include <bits/stdc++.h>
using namespace std;

ll f[N]={1};
int n,m; char s[N];
set<ll> st;

void init()
{
	for(int i=1;i<N;i++)
		f[i]=f[i-1]*257%M;
}

ll run(char* s)
{
	ll res=0;
	int l=strlen(s);
	for(int i=0;i<l;i++)
		res=(res*257+s[i])%M;
	return res;
}

int judge(char* s)
{
	ll h=run(s);
	int l=strlen(s);
	for(int i=0;i<l;i++)
		for(ll c='a';c<='c';c++)
		{
			if(s[i]==c) continue;
			if(st.find((((c-s[i])*f[l-i-1]+h)%M+M)%M)!=st.end()) return 1;
		}
	return 0;
}

int main()
{
	cin>>n>>m,init();
	for(int i=0;i<n;i++)
		scanf("%s",s),st.insert(run(s));
	for(int i=0;i<m;i++)
		scanf("%s",s),puts(judge(s)?"YES":"NO");
	return 0;
}
