#include <iostream>
#include <set>
using namespace std;
const long long modul=1e15+7;
long long n,m,i,p[600000],has,v,j,np,pp;
set <long long > d;
string s;
int main()
{
    cin>>n>>m;

    p[0]=1;
    for (i=1;i<=600000;i++)
    p[i]=(p[i-1]*4)%modul;

while (n--)
{
    cin>>s;
    has=0;
    for (i=0;i<s.size();i++)
        has=(has+(s[i]-'a'+1)*p[i])%modul;
    d.insert(has);
}

while (m--)
{
    cin>>s;
has=0;
    for (i=0;i<s.size();i++)
        has=(has+(s[i]-'a'+1)*p[i])%modul;
pp=0;
   for (i=0;i<s.size();i++)
   {
       v=s[i]-'a'+1;
       for (j=1;j<=3;j++)
       if (v!=j) {np=(has-(v*p[i]))%modul+modul;
                  np=(np+(j*p[i]))%modul;
                  if (d.find(np)!=d.end())
                  {pp=1;cout<<"YES"<<endl;}
                  if (pp==1) break;
                 }
              if (pp==1) break;

   }

if (pp==0) cout<<"NO"<<endl;

}

    return 0;
}
