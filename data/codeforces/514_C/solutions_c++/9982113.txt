#include<iostream>
#include<stdio.h>
#include<set>
#include<string.h>
using namespace std;
long long mod=1000000000007;
int main()
{
	set<long long> store;
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
	{
		char s[4000000];;
		scanf("%s",s);
		long long x=0;
		for(int j=0;s[j]!='\0';j++)
			x=(x*4+s[j]-'a'+1)%mod;
		store.insert(x);
	}
	for(int i=0;i<m;i++)
	{
		char s[4000000];
		scanf("%s",s);
		long long x=0;
		for(int j=0;s[j]!='\0';j++)
			x=(x*4+s[j]-'a'+1)%mod;
		long long int len=strlen(s);;
		long long int base=1;
		int state=0;
		for(int j=len-1;j>=0;j--)
		{
			s[j]=s[j]-'a'+1;
			long long t=((x-base*s[j])%mod+mod)%mod;
			for(long long int p=1;p<=3;p++)
				if(s[j]!=p)
					if(store.find((t+(p*base)%mod)%mod)!=store.end())
					{
						state=1;
						break;
					}
			if(state==1)
				break;
			base=(base*4)%mod;
		}
		if(state==1)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
