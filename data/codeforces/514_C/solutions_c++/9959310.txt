#include <bits/stdc++.h>
using namespace std;

const int ma = 600010;
int arr[ma][3];
int n,m;
char ch[ma];
int cnt;

bool f(int w,int x,bool b)
{
    //printf("%d %c %d %d\n",w,ch[w],x,b);
    if(!ch[w])
    return b;
    int c=ch[w]-'a';
    if(arr[x][c] && f(w+1,arr[x][c],b))
    return 1;
    if(b)
    return 0;
    c++;
    c%=3;
    if(arr[x][c]&&f(w+1,arr[x][c],1))
    return 1;
    c++;
    c%=3;
    if(arr[x][c]&&f(w+1,arr[x][c],1))
    return 1;
}

int main()
{
    //freopen("in.txt","r",stdin);
    cin>>n>>m;
    while(n--)
    {
    scanf("%s",ch);
    //cout<<ch<<endl;
    int x=0;
    int c;
    for(int i=0;ch[i];i++)
    {
        c=ch[i]-'a';
        if(!arr[x][c])
        arr[x][c]=++cnt;
        x=arr[x][c];
    }
    }
    while(m--)
    {
    scanf("%s",ch);
    //cout<<ch<<endl;
    puts(f(0,0,0)?"YES":"NO");
    }
    return 0;
}
