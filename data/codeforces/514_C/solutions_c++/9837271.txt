#include <cstdio>

const int N = 300300, M = 600600;

int n, m, i, j;
int lnk[M * 6][3], lastv = 2;
char s[M];
bool f;

inline void go(int t, int n, bool u)
{
	if (s[n] == 0)
		f |= u;
	if (!f && s[n] > 0)
	{
		char c = s[n] - 'a';
		for (int i = 0; i < 3; i++)
			if (lnk[t][i])
			{
				if (c == i)
					go(lnk[t][i], n + 1, u);
				else
				{
					if (u == false)
						go(lnk[t][i], n + 1, true);
				}
			}
	}
}

int main()
{
	scanf("%d%d", &n, &m);

	for (i = 0; i < n; i++)
	{
		scanf("%s", s);

		int t = 1;
		for (j = 0; s[j]; j++)
		{
			char c = s[j] - 'a';
			if (!lnk[t][c])
				lnk[t][c] = lastv++;
			t = lnk[t][c];
		}
	}

	for (i = 0; i < m; i++)
	{
		scanf("%s", s);
		f = false;
		go(1, 0, 0);
		puts(f ? "YES" : "NO");
	}

	return 0;
}
