#include <cstdio>
#include <cstring>
#define MAXN 600001
#define MAXH 2000001
using namespace std;
int n,m,t,len,c[MAXN][3];
char s[MAXN];
bool b[MAXN];
bool check(int x,int w,bool cg) {
    if (w==len+1 && b[x] && cg) return true;
    for (int i='a';i<='c';i++) {
      if (!c[x][i-'a']) continue;
      if (cg && i!=s[w]) continue;
      if (check(c[x][i-'a'],w+1,cg|(i!=s[w]))) return true;
    }
    return false;
}
int main() {
    scanf("%d%d",&n,&m);t=1;
    for (int i=1;i<=n;i++) {
      scanf("%s",s+1);
      len=strlen(s+1);
      int p=1;
      for (int j=1;j<=len;j++) {
        if (!c[p][s[j]-'a']) c[p][s[j]-'a']=++t;
        p=c[p][s[j]-'a'];
      }
      b[p]=true;
    }
    while (m--) {
      scanf("%s",s+1);
      len=strlen(s+1);
      printf("%s\n",check(1,1,0)?"YES":"NO");
    }
    return 0;
}
