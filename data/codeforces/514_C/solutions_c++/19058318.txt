#include<bits/stdc++.h>
using namespace std;

const int N = 600005;
int n, m, u[N][3], cnt = 1;
char a[N], *word[N>>1];
bool vis[N];

bool search(int x, char *s, bool rem) {
	if(*s == '\0') return !rem;
	int idx = *s - 'a';
	for(int i=0 ; i<3 ; ++i) {
		if(u[x][i]) {
			if(i == idx && search(u[x][i], s+1, rem)) return true;
			if(i != idx && rem)
				if(search(u[x][i], s+1, !rem)) return true;
		}
	}
	return false;
}

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d%d", &n, &m);
	for(int x, i=1 ; i<=n ; ++i) {
		scanf("%s", a);
		x = 1;
		for(int idx, j=0 ; a[j] ; ++j) {
			idx = a[j] - 'a';
			if(u[x][idx] == 0) u[x][idx] = ++cnt;
			x = u[x][idx];
		}
		vis[x] = 1;
	}
	while(m--) {
		scanf("%s", a);
		puts(search(1,a,1) ? "YES" : "NO");
	}
	return 0;
}
