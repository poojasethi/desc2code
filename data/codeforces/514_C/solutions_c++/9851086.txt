#include <cstdio>
#include <cstring>
#include <map>
#define N 600005
#define M 1000000000007LL
#define LL long long
#define fi(a, b, c) for(int a = (b); a < (c); a++)
#define FD(a, b, c) for(int a = (b); a >= (c); a--)
using namespace std;

int n, m;
char s[N];
map<LL, int> h[N];

int main(){
	scanf("%d %d", &n, &m);
	fi(i, 0, n){
		scanf("%s", s);
		int l = strlen(s);
		LL x = 0;
		fi(j, 0, l) x = (x * 4 + s[j] - 'a' + 1) % M;
		h[l][x] = 1;
	}
	fi(i, 0, m){
		scanf("%s", s);
		int l = strlen(s);
		LL x = 0, base = 1;
		fi(j, 0, l) x = (x * 4 + s[j] - 'a' + 1) % M;
		
		bool ok = 0;
		FD(j, l - 1, 0){
			s[j] -= 'a' - 1;
			LL tmp = ((x - base * s[j]) % M + M) % M;
			fi(k, 1, 4) if(k != s[j] && h[l][(tmp + base * k) % M]) ok = 1;
			base = (base * 4) % M;
		}
		
		puts(ok ? "YES" : "NO");
	}
	scanf("\n");
}
