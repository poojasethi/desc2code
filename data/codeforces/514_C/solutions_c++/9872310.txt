#include<bits/stdc++.h>
#define ll long long
#define mx 666669
#define mod 1000000007
using namespace std;

ll n,m,pow3[mx],v,l,prime;
char s[mx];
set<ll>p;

inline ll cal(char *q){
    ll sum=0;
    l=strlen(q);
    for(ll i=0;i<l;i++){
        sum=(sum*prime+s[i])%mod;
    }
    return sum;
}

inline bool judge(char *q){
    v=cal(q);
    l=strlen(q);
    for(ll i=0;i<l;i++){
        for(char a='a';a<='c';a++){
            if(q[i]==a)continue;
            if(p.find(((v+(a-q[i])*pow3[l-i-1])%mod+mod)%mod)!=p.end())return true;
        }
    }
    return false;
}

int main(){
    prime=257;
    pow3[0]=1;
    for(ll i=1;i<mx;i++)pow3[i]=(pow3[i-1]*prime)%mod;

    cin>>n>>m;

    for(ll i=0;i<n;i++){
        scanf("%s",&s);
        p.insert(cal(s));
    }

    for(ll i=0;i<m;i++){
        scanf("%s",&s);
        puts(judge(s)?"YES":"NO");
    }

    return 0;
}
