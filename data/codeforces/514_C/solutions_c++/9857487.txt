#include <stdio.h>
#include <iostream>
#include <string>
#include <set>

#define N 1000000000007LL

using namespace std;

long long int m_hash(const string &str)
{
	long long int h = 0;
	int len = str.length();

	for(int j = 0; j < len; j++)
		h = (h*4 + str[j] - 'a' + 1)%N;

	return h;
}

int main(void)
{
	int n,m;
	scanf("%d %d", &n, &m);

	set<long long int> mem;

	string temp;

	for(int i = 0; i < n; i++)
	{
		cin >> temp;
		mem.insert( m_hash(temp) );
	}

	for(int i = 0; i < m; i++)
	{
		bool ans = false;
		cin >> temp;

		long long int h = m_hash(temp), p4 = 1;

		for(int j = temp.length()-1; j >= 0; j--, p4 = (p4*4)%N)
		{
			for(int k = 1; k < 4; k++)
			{
				if( temp[j]-'a'+1 == k)
					continue;
				long long int x = ( (h-p4*(temp[j]-'a'+1))%N + N + p4*k )%N;
				ans |= (mem.find(x) != mem.end());
			}
		}

		if(ans)
			puts("YES");
		else
			puts("NO");
	}
	return 0;
}
