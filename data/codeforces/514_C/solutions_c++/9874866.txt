#include<bits/stdc++.h>
using namespace std;
#define M 600005
char inp[600005];
bool visit[M];
int u[M][3];
int N;
int dfs(int x,int d,int k)
{
  if(d==N && visit[x] && k==1)
    return 1;
  if(d>=N)return 0;
  int c=inp[d]-'a';
  if(u[x][c])
  {
    if(dfs(u[x][c],d+1,k))
        return 1;
  }
  if(k)return 0;
  for(int i=0;i<3;i++)
  {
    if(c==i)continue;
    if(dfs(u[x][i],d+1,k+1))
        return 1;
  }
  return 0;
}
int main()
{
  int n,m;
  scanf("%d %d",&n,&m);
  int cnt=1;
   for(int i=0;i<n;i++)
   {
       int x=1;
        scanf("%s",inp);
        for(int j=0;inp[j];j++)
        {
          int c=inp[j]-'a';
          if(!u[x][c])u[x][c]=++cnt;
           x=u[x][c];
        }
        visit[x]=1;
   }
   for(int i=0;i<m;i++)
   {
      scanf("%s",inp);
       N=strlen(inp);
      puts(dfs(1,0,0)? "YES":"NO");
   }
   return 0;
}
