#include <bits/stdc++.h>
using namespace std;
//printf("%.7f\n", double value);

typedef long long ll;
ll mod = 1000000000007;
int n, m;
set<ll> mec;

ll stol(string s) {
  ll x= 0;
  for(int i=0; i<s.size(); i++) {
    x= 4*x + s[i]-'a' + 1;
    x= x%mod;
  }
  return x;
}

bool yn(string s) {
  ll b= 1;
  ll x = stol(s);

  for(int i=s.size()-1; i>=0; i--) {
    ll t = (((x - b* (s[i]-'a'+1))%mod)  + mod)%mod;

    for(int j=1; j<4; j++) {
      if(s[i]-'a'==j -1) continue;
      if(mec.find( (t + j*b)%mod)!=mec.end()) return true;
    }
    b *= 4;
    b%= mod;
  }
  return false;
}

int main () {
  cin >> n >> m;
  for(int i=0; i<n; i++) {
    string t;
    cin >> t;
    mec.insert(stol(t));
  }


  for(int i=0; i<m; i++) {
    string s;
    cin >> s;
    if(yn(s)) cout << "YES" << endl;
    else cout << "NO" << endl;
  }
  return 0;
}
