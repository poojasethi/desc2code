#include<bits/stdc++.h>

using namespace std;

int i,j,k;

map<string,int> mp;

vector<string> A[600005];

int main(void)
{
	int n,m;
	cin>>n>>m;
	string s;
	while(n--)
	{
		cin>>s;
		A[s.length()].push_back(s);
	}
	while(m--)
	{
		cin>>s;
		if(mp[s]==1)cout<<"YES\n";
		else if(mp[s]==2)cout<<"NO\n";
		else
		{
			int counter=0;
			int len=s.length();
			int sizee=A[len].size();
			for(i=0;i<sizee;i++)
			{
counter=0;
				for(j=0;j<len;j++)
				{
					if(A[len][i][j]!=s[j])counter++;
					if(counter>1)break;
				}
				if(counter==1)
				{
					mp[s]=1;
					cout<<"YES\n";
					break;
				}
			}
			if(counter!=1)
			{
				mp[s]=2;
				cout<<"NO\n";
			}
			
		}
	}
	return 0;
}
