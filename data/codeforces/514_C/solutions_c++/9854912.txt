#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

bool f[600100];
int tr[600100][3], n, m, ntr, l;
char s[600100];

bool check(int p, int x, int k) {
	if ( p == l )
		return ((f[x]) && (k == 1));
	if (( k > 1 ) || ((p > 0) && (x == 0)))
		return 0;
	int i = s[p] - 'a';
	return (check(p+1, tr[x][i], k) 
		|| check(p+1, tr[x][(i+1)%3], k+1) 
		|| check(p+1, tr[x][(i+2)%3], k+1));
}
int main() {
	scanf("%d%d\n", &n, &m);
	for ( int i = 1; i <= n; i++ ) {
		gets(s);
		l = strlen(s);
		int x = 0, j = 0;
		while (j < l) {
			int p = s[j] - 'a';
			if ( tr[x][p] > 0 ) {
				x = tr[x][p];
			} else {
				ntr++;
				tr[x][p] = ntr;
				x = ntr;
			}
			if ( j == l-1 ) 
				f[x] = 1;
			j++;
		}
	}

	for ( int i = 1; i <= m; i++ ) {
		gets(s);
		l = strlen(s);
		if ( check(0, 0, 0) )
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
