#include<iostream>
#include<string>
#include<map>
#include<string>
using namespace std;
typedef long long int64;
int64 n,m;
int64 pw[6000010];
const int64 P=1000000007;
int64 mod=123ll;
map <int64,int> boa;
string s;

int main()
{
    cin>>n>>m;
    pw[0]=1ll;
    for(int i=1;i<=6000000;i++) pw[i]=pw[i-1]*mod%P;
	for(int i=1;i<=n;i++)
	{
		cin>>s;
		int64 sum=0ll;
		for(int j=0;j<s.length();j++) sum=(sum*mod+(s[j]-'a'))%P;
		boa[sum]=1;
	} 
	for(int i=1;i<=m;i++)
	{
		cin>>s;
		int64 h=0ll;
		int l=s.length(),flag=0;
		for(int j=0;j<l;j++) h=(h*mod+(s[j]-'a'))%P;
		for(int j=0;j<l&&!flag;j++)
		{
			int64 tmp=pw[l-1-j]*(s[j]-'a')%P;
			int64 nh=((h-tmp)%P+P)%P;
			for(int k=0;k<3;k++)
				if(s[j]-'a'!=k)
				{
					int64 tmp2=pw[l-1-j]*(k)%P;
					if(boa[(nh+tmp2)%P])
					{
						flag=1;
						break;
					}
				}
		}
		if(flag) cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
    return 0;
}
