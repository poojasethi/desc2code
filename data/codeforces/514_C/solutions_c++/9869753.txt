#include<bits/stdc++.h>
using namespace std;

const long long  N = 6e5 + 7;
const long long mod = 1e16+7;
long long p[N] = {1};
set<long long> arr;
string s;
long long hash()
{
	long long val=0,v;
	int i;
	for(i=0;i<s.size();i++)
	{
		v = s[i]-'a'+1;
		val= (val + v*p[i])%mod;
	}
	return val;
}
bool check()
{
	int i;
	long long h=hash(),v,val,j;
	for(i=0;i<s.size();i++)
	{
		v = s[i]-'a'+1;
		for(j=1;j<4;j++)
		{
			if(j!=v)
			{
				val = ((h-p[i]*v)%mod + mod)%mod;
				val = (val+(p[i]%mod*j%mod)%mod)%mod;
				if(arr.count(val))
					return true;
			}
		}
	}
	return false;
}
int main()
{
	int n,m;
	cin>>n>>m;
	int i;
	for(i=1;i<N;i++)
		p[i]= (p[i-1]*4)%mod;
	for(i=0;i<n;i++)
	{
		cin>>s;
		arr.insert(hash());
	}
	for(i=0;i<m;i++)
	{
		cin>>s;
		if(check())
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
