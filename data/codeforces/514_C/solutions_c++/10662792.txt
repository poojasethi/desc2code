#include <bits/stdc++.h>
#define MAXN 1000000

using namespace std;
typedef long long ll;
const ll MOD =1e10+7;
int n,m;
set<ll> h;

ll strhash(char * s){
	ll ret=0;
	ll base=257;
	while(*s){
		ret=ret*base+(*s);
		ret%=MOD;
		s++;		
		}
	return ret;
	}
void check(char *s){
	ll hash=strhash(s);
	int k=strlen(s);
	ll base=1;
	//cout << hash << "sdfs" << endl;
	for(int i=k-1;i>=0;--i){
		
			for(int j=0;j<3;++j){
				if(s[i]==('a'+j)) continue;
				ll t=hash-base*(s[i])+base*(j+'a');
				t+=2*MOD;
				t%=MOD;
			//	cout << 't' << ' ' << t << endl;
				if(h.count(t)) {cout << "YES" << endl;return;}
				}
			base*=257;
			base%=MOD;
			}
		cout << "NO" << endl;
}
int main(){

	cin >> n >> m;
	h.clear();
	
//	Trie T;
	char s[MAXN];
	for(int i=0;i<n;++i){
		cin >> s;
	//	cout << "Sfs" << endl;
		ll tmp=strhash(s);
		//cout << tmp << endl;
		h.insert(tmp);
	}
	for(int i=0;i<m;++i){
		cin >> s;
		
		check(s);
	}


	return 0;}
