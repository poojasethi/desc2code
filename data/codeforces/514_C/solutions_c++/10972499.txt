#include<iostream>
#include<vector>
#include<map>
using namespace std;
vector <string> memory[600001];
map < string , int > visited; 
int main()
{
	int n,m;
	cin>>n>>m;
	string a;
	for(int i=0;i<n;i++)
	{
		cin>>a;
		memory[a.size()].push_back(a);
	}
	for(int i=0;i<m;i++)
	{
		cin>>a;
		if(visited[a]==1)
		{
			cout<<"YES"<<endl;
		}
		else if(visited[a]==2)
		{
			cout<<"NO"<<endl;
		}
		else
		{
		int alphacnt=0;
		int mylen=a.length();
		int mysize=memory[mylen].size();
		for(int t=0;t<mysize;t++)
		{
			alphacnt=0;
			for(int j=0;j<mylen;j++)
			{
				if(memory[mylen][t][j]!=a[j])
				{
					alphacnt++;
				}
				if(alphacnt>1)
				{
					break;
				}
			}
			if(alphacnt==1)
			{
				cout<<"YES"<<endl;
				visited[a]=1;
				break;
			}
		}
		if(alphacnt!=1)
		{
			cout<<"NO"<<endl;
			visited[a]=2;
		}
		}
	}
	return 0;
}
