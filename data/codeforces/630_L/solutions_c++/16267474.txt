#include <stdio.h>
#define lli long long int

char s[10];

int main(void) {
    scanf(" %s", s);
    lli n = s[1] - '0' + 10 * (s[3] - '0') + 100 * (s[4] - '0') + 1000 * (s[2] - '0') + 10000 * (s[0] - '0');
    lli aux = n;
    for (int i = 0; i < 4; i++) {
        n = (n * aux) % 100000;
    }
    printf("%05lld\n", n);
    return 0;
}
