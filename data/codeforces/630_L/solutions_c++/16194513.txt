{

	cin>>s;
	t[0]=s[0],t[1]=s[2],t[2]=s[4],t[3]=s[3],t[4]=s[1];
	if(t[4]=='0')
	cout<<"00000"<<endl;

	    for(int i=1;i<=5;i++)

    }
}
