#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int d[5];
int main(){
	ll n;
	cin>>n;
	ll tmp=n;
	for(int j=0;j<5;++j){
		d[j]=tmp%10;
		tmp/=10;
	}
	ll ndig=0;
	ndig+=10000*d[4];
	ndig+=1000*d[2];
	ndig+=100*d[0];
	ndig+=10*d[1];
	ndig+=d[3];
	ll res=1;
	for(int j=0;j<5;++j){
		res*=ndig;
		res%=100000;
	}
	printf("%05d",res);
}
