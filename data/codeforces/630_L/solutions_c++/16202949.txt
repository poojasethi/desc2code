#include<stdio.h>
#include<math.h>
int main()
{
	int a,b,c,d,e;
	__int64 n,s;
	while(~scanf("%I64d",&n))
	{
		int t=5;
		s=1;
		e=n%10;
		d=n/10%10;
		c=n/100%10;
		b=n/1000%10;
		a=n/10000;
		n=a*10000+b+c*1000+d*10+e*100;
		while(t--)
		{
			s*=n;
			s%=100000;
		}
		printf("%05I64d",s);
	}
	return 0;
}
