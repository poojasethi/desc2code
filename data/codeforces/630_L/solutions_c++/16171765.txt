#include<stdio.h>
#include<math.h>

int main()
{
	__int64 n,tmp;
	scanf("%I64d",&n);
	__int64 one=n%10,thou=n/1000%10,ten=n/10%10,hund=n/100%10;
	n=n/10000*10000+hund*1000+one*100+ten*10+thou;
	tmp=n;
	n=n*n%100000;
	n=n*n%100000;
	n=n*tmp%100000;
	printf("%I64d%I64d%I64d%I64d%I64d\n",n/10000,n/1000%10,n/100%10,n/10%10,n%10);
	return 0;
}
