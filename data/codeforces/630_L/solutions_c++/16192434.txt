#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;

string a;
long long n, tmp, res = 1;

int main()
{
	cin >> a;
	n = ( a[0]-'0' ) * 10000 + (a[2]-'0') * 1000 + (a[4]-'0') * 100 + (a[3]-'0') * 10 + a[1]-'0';
	for(int i = 0 ; i < 5 ; i++){
	    res = res * n;
	    res = res % 1000000;
	}
	printf("%05d\n",  res % 100000 );
}
