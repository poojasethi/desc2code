#include <bits/stdc++.h>
using namespace std;
int main() {
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    string s;
    cin >> s;
    long long x, ans=1;
    x=(s[1]-'0')+10*(s[3]-'0')+100*(s[4]-'0')+1000*(s[2]-'0')+10000*(s[0]-'0');
    for(int i=0 ; i<5 ; i++)
    	ans=(ans*x)%100000;
    printf("%05I64d",ans);
	return 0;
}
