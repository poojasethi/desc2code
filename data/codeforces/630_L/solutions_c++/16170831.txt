#include <iostream>

using namespace std;
unsigned long long n,m;
int main()
{
    cin>>n;
    m=n/10000;
    m=m*10+(n%1000)/100;
    m=m*10+(n%10);
    m=m*10+(n%100)/10;
    m=m*10+(n/1000)%10;
    n=m;
    m=(m*m*m)%100000;
    m=(m*n)%100000;
    m=(m*n)%100000;
    if(m!=0)cout<<m;
    else cout<<"00000";
    return 0;
}
