#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef unsigned long long ull;

int main() {

    string s;
    cin >> s;
    ll n = (s[0]-48)*10000 + (s[2]-48)*1000 + (s[4]-48)*100 + (s[3]-48)*10 + (s[1]-48);
    ll ans = (n*n)%100000;
    ans = (ans*ans)%100000;
    ans = (ans*n)%100000;
    printf("%05I64d\n", ans);
    return 0;
}
