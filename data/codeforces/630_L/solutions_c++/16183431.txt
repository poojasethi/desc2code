#include <iostream>
#include <cstdio>
using namespace std;

long long n, b, s = 1, a[10];
int main()
{
	int i;
	cin >> n;
	for (i = 5; i > 0; i--) {
		a[i] = n % 10;
		n /= 10;
	}
	b = a[1]*10000 + a[3]*1000 + a[5]*100 + a[4]*10 + a[2];
	for (i = 0; i < 5; i++) {
		s *= b;
		s %= 100000;
	}
	printf("%05d", s);
    return 0;
}
