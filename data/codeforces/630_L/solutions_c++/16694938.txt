#include<stdio.h>
#include<string.h>
int main()
{
	__int64 ans,x,a[]={1,3,5,4,2},i;
	char s[12];
	while(scanf("%s",s)!=EOF)
	{
		x=0;
		for(i=0;i<5;i++)
		{
			x=x*10+(s[a[i]-1]-'0');
		}
		ans=1;
		for(i=0;i<5;i++)
		{
			ans=(ans*x)%100000;
		}
		printf("%05I64d\n",ans);
	}
	return 0;
}
