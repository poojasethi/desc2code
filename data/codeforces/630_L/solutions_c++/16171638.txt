#include <bits/stdc++.h>
#define kw(a) ((a)*(a))
using namespace std;
typedef long long LL;


int main() {
	string s;
	cin >> s;
	LL n = (s[0]-48)*10000 + (s[2]-48)*1000 + (s[4]-48)*100 + (s[3]-48)*10 + s[1]-48;
	LL n0 = n;
	
	for(int i = 0; i < 4; i++) 
		n = (n*n0) % 100000;
		
	cout << setfill('0') << setw(5) << n;
}
