#include <iostream>
#include <cstdio>
using namespace std;

long long b, c;
string a;
int main()
{
	int i;
	cin >> a;
	b = (a[0] - '0') * 10000 + (a[2] - '0') * 1000 + (a[4] - '0') * 100 + (a[3] - '0') * 10 + (a[1] - '0');
	c = b;
	for (i = 0; i < 4; i++) {
		c = (c * b) % 100000;
	}
	printf("%05d", c);
    return 0;
}
