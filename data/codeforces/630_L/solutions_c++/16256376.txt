#include<cstdio>
typedef long long ll;
int shuffle[6] = {0,10000,100,1,10,1000};
int main() {
	ll num, sf, res;
	scanf("%I64d", &num);
	sf = 0;
	for(int i=1, tot=1; i<=5; ++i, tot*=10)
	{
		sf = sf*10+num/shuffle[i]%10;
	}
	res = 1;
	for(int i=1; i<=5; ++i)
		(res*=sf)%=100000;
	printf("%05I64d\n", res);
	return 0;
}
