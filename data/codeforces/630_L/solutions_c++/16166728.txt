#include <iostream>
#include <cstdio>
using namespace std;
int main(){
    char s[10];
    unsigned long long cur,CUR;
    cin >> s;
    cur=(s[0]-48)*10000+(s[2]-48)*1000+(s[4]-48)*100+(s[3]-48)*10+(s[1]-48);
    CUR=cur;
    for (int i=1; i<5; i++){
        CUR*=cur;
        CUR%=100000;
    }
    printf("%05d",CUR);
}
