#include <iostream>
#include <cstdio>
using namespace std;

long long n, b = 1;
char a[10];
int main()
{
	int i;
//	freopen ("input.txt", "r", stdin);
	cin >> a;
	n = (a[0] - '0') * 10000 + (a[2] - '0') * 1000 + (a[4] - '0') * 100 + (a[3] - '0') * 10 + (a[1] - '0');
	for (i = 0; i < 5; i++) {
		b *= n;
		b %= 100000;
	}
    printf("%05d", b);
    return 0;
}
