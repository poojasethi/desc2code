#include<iostream>
#define ll long long
using namespace std;
int main()
{
    ll n,a,b,mod=1e9;
    cin>>n;
    
    a=(n/10000)*10000 + (n/100%10)*1000 + (n%10)*100 + (n/10%10)*10 + n/1000%10;
    if(a%10==0){cout<<"00000";return 0;}
    b=a;
    for(int i=0;i<4;i++)
    {
            b*=a;
            b%=mod;
    }
    cout<<b%100000;
}
