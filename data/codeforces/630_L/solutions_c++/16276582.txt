#include<bits/stdc++.h>

int main(){
    long long n,t,res;
    while(~scanf("%I64d",&n)){
        res = 1LL;
        t = n/10000*10000+(n/100%10)*1000+(n%10)*100+(n/10%10)*10+(n/1000%10);
        //printf(" sh:: %d",t);
        for(int i = 0;i < 5 ;i++){
            res = res * t %100000;
        }
        printf("%05I64d\n",res);
    }
    return 0;
}
