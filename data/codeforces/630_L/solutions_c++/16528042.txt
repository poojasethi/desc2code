#include <cstdio>
#include <cmath>
#include <iostream>
using namespace std;
string n;
int main()
{
  cin>>n;
  long long now=(n[0] - '0')*10000+(n[1] - '0') + (n[2] - '0')*1000 + (n[3] - '0')*10 + (n[4] - '0')*100;
  long long ans=1ll;
  for(int i=1;i <= 5;i++)
   ans=ans*now%100000;
  if(ans == 0) 
  {
  	cout<<"00000";
	return 0; 
  }
  cout<<ans<<endl; 
}
