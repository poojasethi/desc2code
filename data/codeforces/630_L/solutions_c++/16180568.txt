#include <bits/stdc++.h>
using namespace std;
int main() {
	long long int n, m, mod = 100000;
	cin >> n;
	m = n / 10000 * 10000;
	m = m + (n / 1000) % 10;
	m = m + ((n / 10) % 10) * 10;
	m = m + (n % 10) * 100;
	n = m + ((n / 100) % 10) * 1000;
	n = n % mod;
	m = (n * n) % mod;
	m = (m * m) % mod;
	n = (m * n) % mod;
	int ans = n;
	printf("%05d\n", ans);
	return 0;
}
