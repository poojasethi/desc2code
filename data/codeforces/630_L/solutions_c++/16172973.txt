#include <bits/stdc++.h>
using namespace std;

char c[10];
long long a, b[] = {10000, 1, 1000, 10, 100}, d;

int main () {
	scanf("%s", c);
	for (int i = 0; i < 5; i++) {
		a += b[i] * (c[i] - '0');
		c[i] = '0';
	}
	d = 1;
	for (int i = 0; i < 5; i++) {
		d = (d*a) % 100000;
	}
	a = 4;
	while (d) {
		c[a--] += d%10;
		d /= 10;
	}
	printf("%s\n", c);
	return 0;
}
