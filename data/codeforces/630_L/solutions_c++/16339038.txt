#include<cstdio>
#include<iostream>
#include<cmath>
using namespace std;
int main(){
    char s[10];
    scanf("%s",s);
    for(int i=0;s[i];i++) s[i]-='0';
    long long int n=s[1]+s[3]*10+s[4]*100+s[2]*1000+s[0]*10000;
    long long int ans=1;
    for(int i=1;i<=5;i++) ans=((ans%100000)*n)%100000;
    printf("%.5d",ans);
}
