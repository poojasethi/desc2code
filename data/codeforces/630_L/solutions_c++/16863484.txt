#include <iostream>
#include <cmath>
using namespace std;

int main()
{long long n,s=1,i,s1,s2,s3,s4,s5;
cin>>n;
s5=n%10;
s4=(n%100)/10;
s3=(n%1000)/100;
s2=(n%10000)/1000;
s1=(n%100000)/10000;


n=s1*10000+s3*1000+s5*100+s4*10+s2;


for(i=0;i<5;i++)
{s=s*n;

s=s%100000;

}
if(s!=0)
  cout<<s;
  else
    cout<<"00000";
    return 0;
}
