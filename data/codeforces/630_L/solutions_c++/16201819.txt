#include <cstdio>
int idx(char c){return c-'0';}
#ifdef ONLINE_JUDGE
	#define lld I64d
#endif
int main(){
	char s[6];
	scanf ("%s", s);
	long long n = idx(s[0]);
   	n = n*10+idx(s[2]);
	n = n*10+idx(s[4]);
	n = n*10+idx(s[3]);
	n = n*10+idx(s[1]);	
	long long ans = 1;
	for (int i = 1; i <= 5; ++i){
		ans = ans*n%100000;
	}
	printf ("%05lld", ans);
	return 0;
}
