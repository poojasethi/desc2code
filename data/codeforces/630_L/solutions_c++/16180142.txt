#include <stdio.h>
int main()
{
	long long n;
	scanf("%I64d",&n);
	long long ans=0;
	ans+=n%10*100;
	n/=10;
	ans+=n%10*10;
	n/=10;
	ans+=n%10*1000;
	n/=10;
	ans+=n%10;
	n/=10;
	ans+=n%10*10000;
	//printf("%I64d\n",ans);
	long long k=ans;
	for (int i=1;i<=4;i++)
	{
		ans*=k;
		ans%=100000;
	}
	printf("%05I64d",ans);
}
