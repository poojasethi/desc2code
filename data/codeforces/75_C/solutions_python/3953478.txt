from fractions import gcd
from bisect import bisect_right as br 
g = gcd(*map(int, raw_input().split()))
i = 1
r = []
while i*i <= g:
    if g % i == 0:
        r.append(i)
        r.append(g/i)
    i += 1
r = sorted(r)

for i in xrange(input()):
    l, h = map(int, raw_input().split())
    m = r[br(r, h)-1]
    print -1 if m < l else m
