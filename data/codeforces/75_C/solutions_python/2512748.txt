def find_divisors(a):
    if a == 1:
        return [1]
    divhigh = []
    divlow = []
    i,j = 1,a
    while i < j:
        j=a/i
        if a%i == 0:
            divhigh.append(j)
            divlow.insert(0,i)
        i = i + 1
    return divhigh+divlow
