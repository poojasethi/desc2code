from fractions import gcd
from bisect import bisect_right as br
a,b=map(int,raw_input().split())
g=gcd(a,b)
i=1
r = []
while i*i<=g:
    if g%i==0:
        r.append(i)
        r.append(g/i)
    i+=1
r=sorted(r,reverse=True)
for i in xrange(input()):
    l,h=map(int,raw_input().split())
    for x in r:
        if l<=x<=h:
            print x
            break
    else:
        print -1
