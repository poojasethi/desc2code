import fractions, bisect
def r(): return map(int,raw_input().split())
a,b=r()
g=fractions.gcd(a,b)
gl=[]
for i in xrange(1,g+1):
    if i*i > g: break
    if g%i==0: gl += [i,g/i]
gl = sorted(list(set(gl)))
n=input()
for i in xrange(n):
    l,h=r()
    i = bisect.bisect_right(gl,h)
    if i and gl[i-1]>=l: print gl[i-1]
    else: print -1
