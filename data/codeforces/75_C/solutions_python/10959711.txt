import sys
from fractions import gcd
from bisect import bisect_right

class CodeForces:
    def get_result(self):
        n, m = map(int, sys.stdin.readline().split(" "))
        commonDivisor = list()
        i = 1
        g = gcd(n, m)
        while i * i <= g:
            if g % i == 0:
                commonDivisor.append(i)
                commonDivisor.append(g / i)
            i += 1
        commonDivisor.sort()
        n = int(sys.stdin.readline())
        for i in range(n):
            low, high = map(int, sys.stdin.readline().split(" "))
            k = commonDivisor[bisect_right(commonDivisor, high) - 1]
            print -1 if k < low else k




def main():
    code = CodeForces()
    code.get_result()

if __name__ == "__main__":
    main()
