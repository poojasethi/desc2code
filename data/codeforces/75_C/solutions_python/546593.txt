s = raw_input().split()
a = int(s[0])
b = int(s[1])
while True:
	if b == 0:
		break
	a %= b
	if a == 0:
		break
	b %= a
gcd = a+b

i = 1
q = []
while i*i <= gcd:
	if gcd % i == 0:
		q.append(i)
		q.append(gcd/i)
	i += 1
q.sort()
q.reverse()

c = int(raw_input())
while c > 0:
	c -= 1
	s = raw_input().split()
	a = int(s[0])
	b = int(s[1])
	x = -1
	for i in q:
		if i <= b:
			if i >= a:
				x = i
			break
	print x
