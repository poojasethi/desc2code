#!/usr/bin/python

import sys
import math

def gcd((x, y)):
    if x < y:
        x,y = y,x
    return y if x % y ==0 else gcd((y, x % y))

if __name__=="__main__":
    data = sys.stdin.readlines()
    v = gcd([int(v) for v in data[0].split(' ')])
    li = []
    for i in range(1,int(math.sqrt(v))+1):
        if v % i == 0:
            if v / i == i:
                li.append(i)
            else:
                li += [i,v/i]
    li.sort()
    n = int(data[1])
    for i in range(0,n):
        low, high = [int(v) for v in data[i+2].split(' ')]
        ans = -1
        b, e = 0,len(li)-1
        while b <= e:
            m = (b + e) / 2
            if li[m] < low:
                b = m + 1
            elif li[m] > high:
                e = m - 1
            else:
                ans = max(ans, li[m])
                b = m + 1
        print ans
