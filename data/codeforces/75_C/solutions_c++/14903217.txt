//In The Name Of God;

#include<bits/stdc++.h>

using namespace std;

vector<int>V;

int f(int n)
{
  for(int i=1;i*i<=n;i++)
    {
      if(n%i==0)
	V.push_back(i);
      if(n%i==0 && i*i!=n)
	V.push_back(n/i);
    }
}

int main()
{
  int a,b,q,d;
  cin>>a>>b;
  d=__gcd(a,b);
  f(d);
  sort(V.begin(),V.end());
  cin>>q;
  for(int l,r;q--;)
    {
      cin>>l>>r;
      if(d<l)
	{
	  cout<<-1<<endl;
	  continue;
	}
      int p=lower_bound(V.begin(),V.end(),r)-V.begin();
      if(p>=V.size() || V[p]>r)
	p--;
      if(V[p]<l)
	cout<<-1<<endl;
      else
	cout<<V[p]<<endl;
    }
}
