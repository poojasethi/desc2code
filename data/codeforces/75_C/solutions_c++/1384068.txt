#include<stdio.h>
#include<string.h>
#include<math.h>
#include<iostream>
#include<algorithm>
using namespace std;
#define maxn 100000
int a,b;
int low,high;
int fact[maxn],cnt;
int main()
{
	int i,j,k;
	scanf("%d%d",&a,&b);
	int d=__gcd(a,b);
	cnt=0;
	for(i=1;i*i<=d;i++)
	{
		if(d%i==0)
		{
			fact[++cnt]=i;
			if(i*i!=d) fact[++cnt]=d/i;
		}
	}
	sort(fact+1,fact+cnt+1);
	int n;
	scanf("%d",&n);
	while(n--)
	{
		scanf("%d%d",&low,&high);
		for(i=cnt;i>0;i--)
		if(fact[i]<=high) break;	
		if(fact[i]<low)printf("-1\n");
		else printf("%d\n",fact[i]);
	}
	return 0;
}
