#include<bits/stdc++.h>
using namespace std;
int gcd(int x, int y) {
	return x? gcd(y%x,x): y;
}
vector<int> v;

int  main() {
	int q,low,high, x,y,g,i,j;
	scanf("%d %d",&x,&y);
	g = gcd(x,y);
	for(int i=1;i*i<=g;i++) {
		if(g%i==0) {
			v.push_back(i);
			if(g/i > i)
				v.push_back(g/i);
		}
	}
	sort(v.begin(),v.end());
	scanf("%d",&q);
	for(int i=0;i<q;i++){
		scanf("%d %d",&low,&high);
		int ans=-1;
		for(j=v.size()-1;j>=0;j--) {
			if(low<= v[j] && v[j]<=high)
			{
				ans=v[j];
				break;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
