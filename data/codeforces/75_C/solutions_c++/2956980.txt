#include <iostream>
#include <algorithm>
#include <set>

using namespace std;

int main()
{
	set<int> cdivs;
	int a, b, n, x, y;
	cin >> a >> b;
	for (int i=1; i*i<=b; i++) {
		if (b % i == 0 && a % i == 0)
			cdivs.insert(-i);
		if (b % (b / i) == 0 && a % (b / i) == 0)
			cdivs.insert(-b / i);
	}
	cin >> n;
	for (int i=0; i<n; i++) {
		cin >> x >> y;
		set<int> :: iterator it = cdivs.lower_bound(-y);
		if (it == cdivs.end() || -*it < x) cout << "-1\n";
		else cout << -*it << "\n";
	}
	return 0;
}
