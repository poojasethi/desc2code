#include <iostream>
#include <cstdio>
#include <vector>
#include <cmath>
#include <set>
#include <cmath>
#include <algorithm>
#define pb push_back
using namespace std;
int n,m,a,b,g,i,ok,low,l,r;
vector <int> v;
int main()
{
cin>>a>>b;
g=__gcd(a,b);
for (i=1;i*i<=g;i++)
	{
	if (g%i==0 && i*i!=g)
		{v.pb(i);v.pb(g/i);}
	if (g%i==0 && i*i==g)
		v.pb(i);
	}
sort(v.begin(),v.end());
cin>>n;
for (i=0;i<n;i++)
{
cin>>l>>r;
low=upper_bound(v.begin(),v.end(),r)-v.begin();
if (low>0) low--;
if (v[low]>=l && v[low]<=r) cout<<v[low]<<endl;
else cout<<-1<<endl;
}


return 0;
}
