#include <cstdio>
#include <algorithm>
using namespace std;

int d[100000];

int gcd(int a, int b) {
  	return b ? gcd(b, a%b) : a;
}

int main() {
	int n, a, b, g, cur=0;
	scanf("%d%d%d", &a, &b, &n);	
	g=gcd(a, b);
	for(int i=1; i*i <= g; i++)
		if(g%i == 0) {
			d[cur++]=i;
			if(i*i<g) d[cur++]=g/i; 
		}
	sort(d, d+cur);
	for(int l, h, upos, lpos; scanf("%d%d", &l, &h), n--;) {
		upos=lower_bound(d, d+cur, h)-d;
		lpos=lower_bound(d, d+cur, l)-d;
		if(d[upos] != h) upos--;
		printf("%d\n", upos<lpos?-1:d[upos]);
	}	
  	return 0;
}
