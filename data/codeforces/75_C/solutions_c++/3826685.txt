#include <cstdio>
#include <cstring>
#include <map>
#include <algorithm>

#define X first
#define Y second

using namespace std;

map<int, int> num;
int a, b, n;

int main() {
    scanf("%d%d%d", &a ,&b, &n);
    int d = __gcd(a, b);
    for (int i = 1; i * i <= d; ++ i) {
        if (d % i) continue;
        num[i] = num[d / i] = 1;
    }
    while (n --) {
        int l, r;
        scanf("%d%d", &l, &r);
        int t = (--num.upper_bound(r))->X;
        printf("%d\n", t >= l ? t : -1);
    }
    return 0;
}
