#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
int aa[1000],cc[1000];
int main()
{
    int a,b,p1=0,l1,p2=0,n,ans,l,r;
	scanf("%d%d",&a,&b);
	l1=sqrt(1.0*a)+0.5;
	for(int i=1;i<=l1;i++)
		if(a%i==0)
		{
			aa[p1++]=i;
			aa[p1++]=a/i;
		}
	for(int i=0;i<p1;i++)
		if(b%aa[i]==0)
			cc[p2++]=aa[i];
	sort(cc,cc+p2);
	scanf("%d",&n);
	while(n--)
	{
		ans=-1;
		scanf("%d%d",&l,&r);
		for(int i=0;i<p2;i++)
			if(cc[i]>=l&&cc[i]<=r)
				ans=cc[i];
		printf("%d\n",ans);
	}
	return 0;
}
