#include <stdio.h>
#include <string.h>
#include <vector>
#include <algorithm>
using namespace std;
#define REP(i,n) for(int(i)=0;(i)<(int)(n);(i)++)
int a,b,l,r,n;
vector <int> f;
int gcd(int a,int b) { return a%b?gcd(b,a%b):b;}
int main()
{
	scanf("%d%d",&a,&b);
	scanf("%d",&n);
	const int g=gcd(a,b);
	for (int i=1; i*i<=g; i++) {
		if (g%i==0) {
			f.push_back(i);
			if (i*i<g) f.push_back(g/i);
		}
	}
	sort(f.begin(),f.end());
	REP(i,n) {
		scanf("%d%d",&l,&r);
		int u=upper_bound(f.begin(),f.end(),r)-f.begin()-1;
		if (u<0||f[u]<l) printf("-1\n");
		else printf("%d\n",f[u]);
	}
	return  0;
}
