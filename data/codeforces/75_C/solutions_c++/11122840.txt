#include<stdio.h>
#include<stdlib.h>
int c[100000]; 

int main(void)
{
	int a,b,n,t,i,j,tot,ans,l,r; 

	scanf("%d%d",&a,&b);
	scanf("%d",&n);
	t=a%b; 
	while (t!=0)
	{
		a=b; b=t; 
		t=a%b; 
	}
	t=b;  tot=0; 
	for (i=1; i<=t; i++)
    {
    	if (i*i>t) break; 
    	if (t%i==0)
    	{
    	    tot++; c[tot]=i;
			tot++; c[tot]=t/i; 	
    	}
    }
    for (i=1; i<=n; i++)
    {
    	scanf("%d%d",&l,&r);
    	ans=-1; 
    	for (j=1; j<=tot; j++)
    	   if (c[j]>=l && c[j]<=r && c[j]>ans) ans=c[j]; 
    	printf("%d\n",ans); 
    }
    t=0; 
}
