#include <bits/stdc++.h>
using namespace std;
int main()
{
	int a, b;
	cin>>a>>b;
	int c=__gcd(a, b);
	vector<int>D;
	for (int i=1; i*i<=c; i++)
	{
		if (c%i==0)
		{
			D.push_back(i);
			D.push_back(c/i);
		}
	}
	sort(D.begin(), D.end());
	reverse(D.begin(), D.end());
	int m;
	cin>>m;
	for (int j=0; j<m; j++)
	{
		int a, b;
		cin>>a>>b;
		bool c=0;
		for (int i=0; i<D.size(); i++)
		{
			if (D[i]>=a && D[i]<=b)
			{
				cout<<D[i]<<"\n";
				c=1;
			}
			if (c==1)
			{
				break;
			}
		}
		if (c==0)
		{
			cout<<-1<<"\n";
		}
	}
}
