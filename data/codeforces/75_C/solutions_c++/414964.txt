#include <cstdio>
#include <set>
using namespace std;

int a, b;
int low,high;

int gcd(int a, int b)
{
	if (b==0) return a;
	else return gcd(b,a%b);
}

int main()
{
	int n, a, b, g;
	set<int> s;
	scanf("%d%d",&a,&b);
	g = gcd(a,b);
	for (int i = 1; i*i <= g; ++i) 
		if (g%i==0) {
			s.insert(i);
			s.insert(g/i);
		}

	scanf("%d",&n);
	for (int i = 1; i <= n; ++i) {
		scanf("%d%d",&a,&b);
		set<int>::iterator p = s.upper_bound(b);
		if (p==s.begin() || *(--p)<a) {
			printf("-1\n");
		} else {
			printf("%d\n",*p);
		}
	}
	return 0;
}
