#include<cstdio>
#include<cmath>
#include<algorithm>
#include<iostream>
using namespace std;
int p[100005];
int main()
{
	int a,b,i,x,y,tmp,cnt,q;
	scanf("%d%d",&a,&b);
	if(a>b)
	{
		tmp=a;
		a=b;
		b=tmp;
	}
	cnt=0;
	for(i=1;i*i<=a;i++)
	{
		if(a%i==0)
		{
			if(b%i==0)
				p[cnt++]=i;
			tmp=a/i;
			if(tmp!=i&&b%tmp==0)
				p[cnt++]=tmp;
		}
	}
	sort(p,p+cnt);
	scanf("%d",&q);
	while(q--)
	{
		scanf("%d%d",&x,&y);
		for(i=cnt-1;i>=0;i--)
		{
			if(p[i]>=x&&p[i]<=y)
			{
				printf("%d\n",p[i]);
				break;
			}
		}
		if(i==-1)
			printf("-1\n");
	}
	return 0;
}
