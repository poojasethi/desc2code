#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"

int i,j,s,mx,a,b,n,z,x,y;
int A[100000];

int main()
{
  scanf("%d%d%d",&a,&b,&n);
  i=__gcd(a,b);
  for(j=1;j*j<i;j++)
    if(!(i%j))
    {
      A[z++]=j;
      A[z++]=i/j;
    }
  if(!(i%j))
    A[z++]=j;
  sort(A,A+z,greater <int>());
  while(n--)
  {
    scanf("%d%d",&x,&y);
    for(i=0;i<z && (A[i]<x || A[i]>y);i++);
    printf("%d\n",i==z?-1:A[i]);
  }
  return 0;
}
