#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
#define MAXN 70000
int f[MAXN+10],cnt,a,b,n,l,h,i;
void prepare(){
	int i,t=sqrt(a+0.5);
	for(i=1;i<=t;i++){
		if(a%i==0){
			if(b%i==0)
				f[++cnt]=i;
			if(b%(a/i)==0){
				if(i!=a/i)
					f[++cnt]=a/i;
			}
		}
	}
}
int main()
{
	scanf("%d%d",&a,&b);
	scanf("%d",&n);
	prepare();
	sort(f+1,f+cnt+1);
	for(i=1;i<=n;i++){
		scanf("%d%d",&l,&h);
		int t=lower_bound(f+1,f+1+cnt,h)-f,t2=lower_bound(f+1,f+1+cnt,l)-f;
		if(h<f[t]||t>cnt)
			t--;
		if(t>=t2)
			printf("%d\n",f[t]);
		else
			puts("-1");
	}
}
