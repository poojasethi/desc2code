#include <iostream>
#include <cstdio>
#include <set>
#include <cmath>
#include <algorithm>

using namespace std;

int main(){
  int a, b;
  int n, l, h;
  set<int> d;
  set<int>::iterator p;
  scanf("%d %d", &a, &b);
  int g = __gcd(a,b);
  for(int x = 1; x*x<=g; x++){
    if( g%x == 0 ){
      d.insert(x);
      d.insert(g/x);
    }
  }
  scanf("%d", &n);
  while(n--){
    scanf("%d %d", &l, &h);
    p = d.upper_bound(h);
    p--;
    if(*p < l)
      puts("-1");
    else
      printf("%d\n", *p);
  }
}
