#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

vector <int> v;

int main()
{
	int a, b, n, x, y, c;
	cin >> a >> b;
	cin >> n;
	c = __gcd(a, b);
	for (int i = 1; i <= sqrt(c); i++)
		if (c % i == 0)
		{
		    v.push_back(i);
			if (c / i > i) v.push_back(c / i);
		}
	sort(v.begin(), v.end());
	
	for (int i = 1; i <= n; i++)
	{
		cin >> x >> y;
		int ans = -1;
		for (int j = v.size() - 1; j >= 0; j--)
			if (v[j] >= x && v[j] <= y)
			{
				ans = v[j];
				break;
			}
		cout << ans << endl;
	}
	
	return 0;
}
