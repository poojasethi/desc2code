#include<iostream>
#include<cstdio>
#include<algorithm>
#include<set>
using namespace std;
int a,b;
int n;
int Q;
set<int > s;
set <int > :: iterator ii;
int gcd(int x,int y)
{
	if(y==0)return x;
	return gcd(y,x%y);
}
int main()
{
	scanf("%d%d",&a,&b);
	n=gcd(a,b);
	for(int i=1;i*i<=n;i++)
	{
		if(n%i==0)
		{
			s.insert(i);
			s.insert(n/i);
		}
	}
	scanf("%d",&Q);
	while(Q--)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		ii=s.upper_bound(y);
		if(ii==s.begin() || *(--ii)<x)
		{
			printf("%d\n",-1);
			
		}
		else 
		printf("%d\n",*ii);
	}
	return 0;
}
