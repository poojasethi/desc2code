#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int gcd(int a, int b) { return b?gcd(b,a%b):a; }

int main() {
	ios::sync_with_stdio(false);
	int a, b;
	cin >> a >> b;
	int g = gcd(a, b);
	vector<int> v;
	for(int i = 1; i*i <= g; i++)
		if(g % i == 0)
			v.push_back(i),
			v.push_back(g/i);
	sort(v.begin(), v.end());
	int t;
	cin >> t;
	for(int i = 0; i < t; i++) {
		int l, r;
		cin >> l >> r;
		int j = upper_bound(v.begin(), v.end(), r)-v.begin()-1;
		if(j < 0 || v[j] < l)
			cout << -1 << endl;
		else
			cout << v[j] << endl;
	}
	return 0;
}
