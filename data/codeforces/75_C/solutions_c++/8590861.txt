#include <iostream>
#include <cstdio>
#include <math.h>
#include <vector>
#include <algorithm>
using namespace std;
vector <int> v;
int main()
{
	int a,b;
	scanf("%d%d",&a,&b);
	int GCD=__gcd(a,b);
	for (int i=1; i<=(int)sqrt(GCD); i++)
	{
		if (GCD%i==0) v.push_back(i),v.push_back(GCD/i);
	}
	sort(v.begin(),v.end());
	int n;
	scanf("%d",&n);
	while (n--)
	{
		int l,h;
		scanf("%d%d",&l,&h);
		int id=upper_bound(v.begin(),v.end(),h)-v.begin()-1;
		int ans=v[id];
		if (l>ans) printf("-1\n"); else printf("%d\n", ans);
    }
    return 0;
}
