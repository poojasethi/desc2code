#include    <cstdio>
#include    <algorithm>
#include    <set>

using   namespace   std;

set< int, greater< int > > s;
int a, b, d, q, l, r;

int main()
{
    scanf( "%d%d%d", &a, &b, &q );
    d = __gcd( a, b );
    for( int i = 1; i * i <= d; i ++ )
        if( d % i == 0 )
            s.insert( i ), s.insert( d / i );
    for( int i = 0; i < q; i ++ )
    {
        scanf( "%d%d", &l, &r );
        int a = *lower_bound( s.begin(), s.end(), r, greater< int >() );
        printf( "%d\n", ( a >= l && a <= r ) ? a : -1 );
    }
    return 0;
}
