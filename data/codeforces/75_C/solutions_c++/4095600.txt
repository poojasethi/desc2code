#include <iostream>
#include <set>
using namespace std;
int gcd(int a,int b){
	return b==0?a:gcd(b,a%b);
}
int main(){
	int a,b,ans;
	while (cin>>a>>b){
		int gc=gcd(a,b);
		int num,x,y;
		set<int>s;
		set<int>::iterator iter;
		for(int i=1;i*i<=gc;i++)
			if(!(gc%i)){
				s.insert(i);
				s.insert(gc/i);
			}
		cin>>num;
		while (num--){
			cin>>x>>y;
			iter=s.upper_bound(y);
			if(iter==s.begin()||*(--iter)<x)
				cout<<-1<<endl;
			else cout<<*iter<<endl;

		}
	}
	return 0;
}
