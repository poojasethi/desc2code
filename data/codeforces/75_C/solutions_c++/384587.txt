#include <iostream>
#include <sstream>
#include <algorithm>
#include <set>
#include <cmath>

using namespace std;

int main() {
	int a, b;
	cin >> a >> b;
	int n, t, l, h;
	cin >> n;
	int g = __gcd(a, b);
	set<int> gs;
	for (int i = (int)sqrt(g); i > 0; i--)
		if (g % i == 0) gs.insert(i), gs.insert(g/i);
	while (n--) {
		cin >> l >> h;
		set<int>::iterator it = gs.upper_bound(h); 
		it--;
		if (*it >= l and *it <= h) {
			cout << *it << endl;
			continue;
		}
		cout << -1 << endl;
	}
	return 0;
}
