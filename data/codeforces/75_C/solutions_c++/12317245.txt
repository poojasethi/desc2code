#include<bits/stdc++.h>
#define MP make_pair
#define R second
#define L first
using namespace std;
vector<int> v;

int main(){
	int a, b;
	int n;
	cin >> a >> b >> n;
	int d=__gcd(a, b);
	for(int i=1;i*i<=d;i++)
		if(d%i==0){
			v.push_back(d/i);
			v.push_back(i);
		}
	sort(v.begin(), v.end());
	while(n--){
		int lo, hi;
		cin >> lo >> hi;
		int res=*(--upper_bound(v.begin(), v.end(), hi));
		if(res>=lo)
			cout << res << endl;
		else
			cout << -1 << endl;
	}
}
