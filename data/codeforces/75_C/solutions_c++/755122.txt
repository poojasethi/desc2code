#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;

int main(void) {
	int a, b, n;
	vector<int> fac;
	scanf("%d%d%d",&a, &b, &n);
	a = __gcd(a, b);
	for(int i=1;i<=a/i;i++) {
		if(a%i==0) {
			fac.push_back(i);
			if(i<a/i) fac.push_back(a/i);
		}
	}
	
	sort(fac.begin(), fac.end());
	while(n--) {
		int L, R;
		scanf("%d%d", &L, &R);
		int t = lower_bound(fac.begin(), fac.end(), R+1) - fac.begin();
		--t;
		if(fac[t] >= L)
			printf("%d\n", fac[t]);
		else
			printf("-1\n");
	}
	return 0;
}
