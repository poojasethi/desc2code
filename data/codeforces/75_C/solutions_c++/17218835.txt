#include <bits/stdc++.h>

using namespace std;

int main()
{
	int a, b, n, gcd_val, x, y;
	vector <int> v;
	
	cin >> a >> b >> n;
	gcd_val = __gcd(a, b);
	
	for ( int i = 1; i <= (int)sqrt(gcd_val); i++ ) {
		if ( gcd_val % i == 0 ) {
			v.push_back(i);
			if ( i != gcd_val/i ) v.push_back(gcd_val/i);
		}
	}
	
	sort(v.begin(), v.end());
	
	while ( n-- ) {
		cin >> x >> y;
		int idx = upper_bound(v.begin(), v.end(), y) - v.begin();
		idx--;
		if ( idx < 0 || v[idx] < x ) puts("-1");
		else cout << v[idx] << endl;
	}
	return 0;
}
