#include <iostream>
#include <set>
#include <cmath>
#include <algorithm>

using namespace std;

int A,B,N;
set<int> divisor;

void init()
{
	for (int i=1;i<sqrt(max(A,B));i++)
		if (A%i==0||B%i==0)
		{
			if (A%i==0&&B%i==0)
				divisor.insert(i);
			if (A%i==0&&B%(A/i)==0)
				divisor.insert(A/i);
			if (B%i==0&&A%(B/i)==0)
				divisor.insert(B/i);
		}
}

int main()
{
	cin >> A >> B >> N;
	init();
	for (int i=0;i<N;i++)
	{
		int low,high;
		cin >> low >> high;
		set<int>::iterator it=divisor.upper_bound(high);
		it--;
		if (*it>=low)
			cout << *it << endl;
		else
			cout << -1 << endl;
	}
	return 0;
}
