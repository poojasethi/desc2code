#include <stdio.h>
#include <string.h>
#include <algorithm>

using namespace std;

int d[100000],dn;
int a,b,g;
int gcd(int a,int b){return b==0?a:gcd(b,a%b);}
int main(){
	int i,j,k;
	scanf("%d %d",&a,&b);
	g=gcd(a,b);
	dn=0;
	for (i=1;i*i<=g;i++)
		if (g%i==0){
			d[dn++]=i;
			if (i*i<g) d[dn++]=g/i;
		}
	sort(d,d+dn);
	int n;
	for (scanf("%d",&n);n>0;n--){
		int ans=-1;
		int low,high;
		scanf("%d %d",&low,&high);
		j=lower_bound(d,d+dn,low)-d;
		k=lower_bound(d,d+dn,high)-d;
		if (d[k]!=high) k--;
		if (j<=k) ans=d[k];
		printf("%d\n",ans);
	}
	return 0;
}
