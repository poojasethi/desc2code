#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int gcd(int a,int b)
{
	if(!b)
		return a;
	return gcd(b,a%b);
}
#define pb push_back
vector<int> g;
int a,b;
int p;
int main()
{
	cin >> a >> b;
	int t = gcd(a,b);
	for(int i=1; i*i<=t;i++)
		if(t%i==0)
		{
			g.pb(i);
			if(t/i !=i)
				g.pb(t/i);
		}
	sort(g.begin(),g.end());
	cin >> p;
	for(int i=0;i<p;i++)
	{
		int x,y;
		cin >> x >> y;
		vector<int>::iterator it = upper_bound(g.begin(),g.end(), y);
		if(it==g.begin() || *(--it)<x)
			cout << -1 << endl;
		else
			cout << *it << endl;
		
	}

	return 0;
}
