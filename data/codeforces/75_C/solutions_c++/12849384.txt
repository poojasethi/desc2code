#include<iostream>
#include<set>
#include<cmath>

using namespace std;

int a,b,l,r;
set<int> s;

int bmm(int x,int y)
{
  if(y==0)
    return x;
  return bmm(y,x%y);
}

int main()
{
  cin>>a>>b;
  a=bmm(a,b);
  for(int i=1;i*i<=a;i++)
    if(a%i==0)
      s.insert(-i),s.insert(-a/i);
  cin>>b;
  while(b--)
    {
      cin>>l>>r;
      a=-*s.lower_bound(-r);
      if(a>=l)
	cout<<a<<endl;
      else
	cout<<-1<<endl;
    }
  return 0;
}
