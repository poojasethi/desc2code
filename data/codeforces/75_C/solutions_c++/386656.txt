#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

vector<int> d;
int main() {
  int a, b, g; cin >> a >> b; g = __gcd(a, b);
  for (int i = 1; i * i <= g; i ++) {
    if (g % i == 0) {
      d.push_back(i);
      if (i * i != g) d.push_back(g / i);
    }
  }
  int nd = d.size();
  int n; cin >> n;
  while (n --) {
    int lo, hi; cin >> lo >> hi;
    int mx = -1;
    for (int i = 0; i < nd; i ++) {
      if (lo <= d[i] && d[i] <= hi) mx = max(mx, d[i]);
    }
    cout << mx << endl;
  }
}
