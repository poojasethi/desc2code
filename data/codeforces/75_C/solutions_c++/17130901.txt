#include <cstdio>
#include <set>
#include <cmath>
#include <algorithm>

using namespace std;

int main()
{
    set<int>ans;
    int a,b,n,low,high;
    scanf("%d%d%d",&a,&b,&n);
    ans.insert(0);
    int gcd=__gcd(a,b);
    int x=sqrt(gcd);
    for(int i=1;i<=x;i++){
        if(gcd%i==0){
            ans.insert(i);
            ans.insert(gcd/i);
        }
    }
    while(n--){
        scanf("%d%d",&low,&high);
        int findd=*--ans.upper_bound(high);
        if(findd<low)printf("-1\n");
        else printf("%d\n",findd);
    }
    return 0;
}
