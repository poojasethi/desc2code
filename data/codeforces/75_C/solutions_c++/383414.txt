#include <iostream>
#include <cstring>
#include <vector>
#include <cstdio>
#include <algorithm>
#include <set>
using namespace std;

int a,b,n;

set<int> s;


int main(){
	cin >> a >> b;
	cin >> n;

	int g = __gcd(a,b);

	for(int i=1;i*i<=g;i++)
		if(g%i==0){
			s.insert(i);
			s.insert(g/i);
		}
	
	for(int i=0;i<n;i++){
		int lo,hi;
		scanf("%d%d",&lo,&hi);
		set<int>::iterator x = s.lower_bound(lo);
		set<int>::iterator y = s.upper_bound(hi);
		if(x==y)
			printf("-1\n");
		else{
			y--;
			printf("%d\n",*y);
		}
	}

	return 0;
}
