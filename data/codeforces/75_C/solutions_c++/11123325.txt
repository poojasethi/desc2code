#include<stdio.h>
int gcd(int a,int b){
	return b==0?a:gcd(b,a%b);
}
int main(void){
	int x,y;
	scanf("%d%d",&x,&y);
	int t;
	int ans=gcd(x,y);
	scanf("%d",&t);
	while(t--){
		int m,n;int ok=0;
		scanf("%d%d",&m,&n);
		int p;int t1=ans/m,t2=ans/n;
		if(t1==0) {
			printf("-1\n");continue;
		} if(t2==0) t2=1;
		for(int i=t2;i<=t1;i++){
			if(ans%i==0&&ans/i>=m&&ans/i<=n){
				printf("%d\n",ans/i);
				ok=1;
				break;
			}if(ans/i<m) break;
		}if(ok==0) printf("-1\n");
	}return 0;
}
