#include<bits/stdc++.h>
#define pb push_back
using namespace std;
int a, b, c, q, x, y, i;
vector<int> f;
int gcd(int a, int b){ 	return !b ? a : gcd(b, a%b); }
int main(int argc, char **argv) {
	scanf("%d%d%d", &a, &b, &q);
	for(i=1 , c = gcd(a, b) ; i*i<c ; ++i) if(c % i == 0) f.pb(i), f.pb(c/i);
	if(i*i == c) f.pb(i);
	sort(f.begin(), f.end());
	while(q--){
		scanf("%d%d", &x, &y);
		i = upper_bound(f.begin(), f.end(), y) - f.begin() - 1;
		if(i<0 || f[i]<x) puts("-1");
		else printf("%d\n", f[i]);
	}
	return 0;
}
