#include<cstdio>
#include<set>
#include<algorithm>
using namespace std;

set<int> p;
int n,a,b;

int gcd(int a,int b) {return !b?a:gcd(b,a%b);}

int main()
{
    scanf("%d%d%d",&a,&b,&n);
    int g=gcd(a,b);
    for (int i=1;i*i<=g;i++)
    if (g%i==0) 
    {
        p.insert(i);
        if (i*i!=g) p.insert(g/i);
    }
    for (int l,r;n--;)
    {
        scanf("%d%d",&l,&r);
        set<int> :: iterator it=p.lower_bound(r+1);it--;
        if (it==p.end() || (*it)<l) printf("-1\n");
        else printf("%d\n",*it);
    }
}
