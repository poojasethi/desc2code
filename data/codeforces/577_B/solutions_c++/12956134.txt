#include<stdio.h>
int n,m,i,t,a[1000001],dp[1001],x[1001],j;

int main(){
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++){
		scanf("%d",&t);
		a[i]=t;
	}
	for(i=0;i<n;i++){
		x[a[i]%m]=1;
		for(j=0;j<m;j++){
			if(dp[j]!=0){
				x[(j+a[i])%m]=1;
			}
		}
		for(j=0;j<m;j++)
			dp[j]|=x[j];
		if(dp[0]==1)break;
	}
	printf("%s",dp[0]==1?"YES":"NO");
}
