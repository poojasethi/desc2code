#include <bits/stdc++.h>
using namespace std;
int main() {
	int i,j,k,l,n,m,a[1005],b[1005]={0};
	cin>>n>>m;
	for(i=0;i<n;i++) {
		cin>>j;
		memset(a,0,sizeof(a));
		for(k=0;k<m;k++)
			if(b[k]) {
				a[(k+j)%m] = 1;
				a[k] = 1;
			}
		a[j%m] = 1;
		for(j=0;j<m;j++)
			b[j] = a[j];
		if(b[0])
			break;
	}
	if(b[0])
		cout<<"YES\n";
	else
		cout<<"NO\n";
	return 0;
}
