#include<iostream>
#include<cstdio>
using namespace std;
int a[1000001];
int dp[2][1001];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	int i,j;
	for(i=1;i<=n;i++)
		scanf("%d",a+i),a[i]%=m;
	for(i=1;i<=n;i++)
	{
		for(j=0;j<m;j++)
		{
			dp[i%2][(j+a[i])%m]=dp[(i-1)%2][(j+a[i])%m]||dp[(i-1)%2][j];
			if(j==a[i])
				dp[i%2][j]++;
		}	
		if(dp[i%2][0]){
			cout<<"YES";
			return 0;
		}
	}
	cout<<"NO";
}
