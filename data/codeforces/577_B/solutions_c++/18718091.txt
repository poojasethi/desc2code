#include <cstdio>
int f[1010],g[1010],i,j,n,m,x,ans=0;
int main()
{
	scanf("%d%d\n",&n,&m);
	if (n>=m) ans=1;
	f[0]=1;
	for (i=1;i<=n && !ans;++i)
	{
		scanf("%d",&x);x%=m;
		for (j=0;j<m;++j) g[j]=f[j],f[j]=0;
		for (j=0;j<m && !ans;++j)
			if (g[j] && (j+x)%m==0) ans=1;
				else f[(j+x)%m]|=g[j];
		for (j=0;j<m;++j) f[j]|=g[j];
	}
	if (ans) printf("YES\n");else printf("NO\n");
}
