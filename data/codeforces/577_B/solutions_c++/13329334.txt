#include<cstdio>
#include<cstring>

using namespace std;

int main()
{
    int n, m, a;
    scanf("%d%d", &n, &m);
    bool s[2][1010] = {1};
    for (int i=0;i<n;++i) {
        scanf("%d", &a);
        if (a%m == 0 || s[i%2][m - a%m]) {
            printf("YES\n");
            return 0;
        }
        for (int j=0;j<m;++j)
            s[1-i%2][(j+a)%m] = s[i%2][(j+a)%m] || s[i%2][j];
    }
    printf("NO\n");
    return 0;
}
