#include<cstdio>
bool vis[2][1000];
int main(){
    //freopen("d://d.txt","r",stdin);
    int n,m,x,i;scanf("%d%d",&n,&m);
    bool now=0,lst=1;
    while(n--){
        scanf("%d",&x);
        for(i=1;i<m;i++)if(vis[lst][i])vis[now][i]=vis[now][(i+x)%m]=true;
        vis[now][x%m]=true;
        if(vis[now][0])break;
        now=!now;lst=!lst;
    }
    if(vis[now][0])printf("YES");
    else printf("NO");
    return 0;
}
