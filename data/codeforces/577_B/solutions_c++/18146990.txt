#include <stdio.h>
#include <vector>
using namespace std;

vector<int>v;
int mark[1005];
void find(int p){
	if(!mark[p]){
		mark[p] = 1;
		v.push_back(p);
	}
}
int main(){
	int n,m,a;
	scanf("%d %d",&n,&m);
	while(n--){
		scanf("%d",&a);
		int len = v.size();
		for(int i = 0;i < len;i++){
			find((v[i] + a) % m);
		}
		find(a % m);
		if(mark[0])
			break;
	}
	n < 0 ? printf("NO\n") : printf("YES\n"); 
}
