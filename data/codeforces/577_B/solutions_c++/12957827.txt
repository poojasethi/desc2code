#include<bits/stdc++.h>
using namespace std;
int i,j,n,m,h[2005],a[1000005],hh[2005],t;
int main()
{
	cin>>n>>m;
	for(;i<n;i++)
		scanf("%d",a+i);
	//if(n>=m){cout<<"YES"<<endl;return 0;}
	for(i=0;i<n;i++)
	{
		if(h[0])break;
		for(j=0;j<m;j++)
			if(h[j])
				hh[(j+a[i])%m]=1;
		hh[a[i]%m]=1;
		for(j=0;j<m;j++)h[j]=hh[j];
	}
	if(h[0])cout<<"YES"<<endl;
	else cout<<"NO"<<endl;
	return 0;

}
