#include<cstdio>
int n, m, a, s[2][1010];
int main() {
    std::scanf("%d%d", &n, &m);
    for (int i=0;i<n&&i<m;++i) {
        std::scanf("%d", &a);
        for (int j=0;j<m;++j)
            s[1-i%2][(j+a)%m] = s[i%2][(j+a)%m] || s[i%2][j] || !j;
    }
    std::printf("%s\n", s[0][0] || s[1][0] ? "YES" : "NO");
    return 0;
}
