#include <iostream>
#include <cstdio>
bool predp[1111],dp[1111];
int main(){
int i,j,m,n,x;
    scanf("%d%d",&n,&m);
    for (i=1; i<=n; i++){
        for (j=0; j<m; j++) predp[j]=dp[j];
        scanf("%d",&x);
        for (j=1; j<m; j++)
            if(predp[j])
            dp[(x+j)%m]=true;
        dp[x%m]=true;
        if(dp[0]){printf("YES"); return 0;}
    }
    printf("NO");

}
