#include <bits/stdc++.h>

using namespace std;

vector <int> v;
int n , m , x , s ;
bool a[2000]={0};

void f(int c){
if( a[c]==0 ){ a[c]=1; v.push_back(c); }
}

int main()
{
 cin>>n>>m;
 while(n--){

    cin>>x;

    s=v.size();
    for(int i=0;i<s;i++)
        f( (v[i]+x)%m );
    f(x%m);

    if(a[0]==1)break;

 }
 if(a[0]==1)cout<<"YES"<<endl;
 else       cout<<"NO"<<endl;
    return 0;
}
