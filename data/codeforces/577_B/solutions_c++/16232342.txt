#include<stdio.h>
int a[2000],n,mod,ban,i,j,b;
int c[2000],t;
main (){
scanf("%d %d",&n,&mod);
for(i=0;i<n;i++){
scanf("%d",&b);
if(ban==0)   {
t=0;
b%=mod;
for(j=1;j<=mod;j++)
    if(a[j]!=0)
    c[t++]=(j+b)%mod;
for(j=0;j<t;j++)
    a[c[j]]=1;
a[b]=1;

if(a[0]==1)
    ban++;
}
}

if(ban==0)
    printf("NO");
else
    printf("YES");
}
