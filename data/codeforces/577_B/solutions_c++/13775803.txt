#include<cstdio>

long long int data[1008];
int check[1008];
int m,n,s=0,in;

int main(){
	scanf("%d%d",&n,&m);
	data[0]=1;
	for(int i=0;i<n;i++){
		scanf("%d",&in);
		in %= m;
		for(int j=0;j<m;j++)
			check[j]=0;
		for(int j=0;j<m;j++){
			if(data[j]>=1)
				check[(j+in)%m] = 1;
		}
		for(int j=0;j<m;j++)
			data[j] += check[j];
		if(data[0]>1){
			printf("YES");
			return 0;
		}
	}
	
	printf("NO");
	return 0;
}
