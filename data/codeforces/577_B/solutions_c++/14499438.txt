#include <cstdio>

using namespace std;

const int maxm=1100;

int n,m,a;
int arr[maxm];
int temp[maxm];

int main (){
	
scanf("%d%d",&n,&m);	
	
if(n>=m){
	printf("YES");
	return 0;
}	
	
for(int i=0;i<n;i++){
	
	scanf("%d",&a);
	
	a%=m;
	
	for(int j=0;j<m;j++)
		if(arr[j]) temp[(j+a)%m]=1;
		
	arr[a]=1;	
	
	for(int j=0;j<m;j++)
		if(temp[j]) arr[j]=1;
		
}	
	
printf((arr[0])?"YES":"NO");	
	
return 0;
}
