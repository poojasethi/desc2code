#include<iostream>
#include<bitset>
using namespace std;
int n,m;
bitset<1000> rot(int r,const bitset<1000> &b)
{
	bitset<1000>c=b>>(m-r);
	return c|((b^(c<<(m-r)))<<r);
}
int main()
{
	cin>>n>>m;
	bitset<1000>dp(1);
	for(int i=1;i<=n;i++)
	{
		int x;
		cin>>x;
		x%=m;
		bitset<1000>t=rot(x,dp);
		if(t[0])
		{
			cout<<"YES";
			return 0;
		}
		dp|=t;
	}
	cout<<"NO";
	return 0;
}
