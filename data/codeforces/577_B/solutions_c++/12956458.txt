#include<stdio.h>
bool predp[1005],dp[1005],flag;
int n,m,x;
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		for(int j=1;j<m;j++)predp[j]=dp[j];
		scanf("%d",&x);
		for(int j=1;j<m;j++)if(predp[j])dp[(j+x)%m]=true;
		dp[x%m]=true;
		if(dp[0]){
			flag=true;
			break;
		}
	}
	if(flag)printf("YES\n");
	else printf("NO\n");
	return 0;
}
