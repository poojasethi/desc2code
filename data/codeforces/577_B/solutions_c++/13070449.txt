#include<cstdio>
#include<algorithm>
using namespace std;
bool b1[1000][1000];
int main()
{
    int n,k,a,b;
    scanf("%d%d%d",&n,&k,&a);
    if(n>k){puts("YES");return 0;}
    b1[0][a%k]=1;
    for(int i=1;i<n;++i)
    {
        scanf("%d",&a);a%=k;
        for(int j=0;j<k;++j)
            b1[i][j]=b1[i-1][j]||b1[i-1][j<a?j+k-a:j-a];
        b1[i][a]=1;
    }
    puts(b1[n-1][0]?"YES":"NO");
}
