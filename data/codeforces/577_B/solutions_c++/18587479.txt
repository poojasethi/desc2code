#include<iostream>
#include<queue>
using namespace std;
int main(){
	int n,p,a;
	cin>>n>>p;
	int czy[p];
	for(int i=0;i<p;i++)czy[i]=0;
	for(int i=0;i<n;i++){
	cin>>a;
	a%=p;
	queue <int> Q;
	for(int j=0;j<p;j++)if(czy[j])Q.push((j+a)%p);
	while(!Q.empty()){
	czy[Q.front()]=1;
	Q.pop();
}
	czy[a]=1;
	if(czy[0]==1){
		cout<<"YES";
		return 0;
	}
}
	cout<<"NO";
	return 0;
	
	
}
