#include<cstdio>
int rem[1001],c[1001];
int main(){
    int n,m,x;
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;++i){
        scanf("%d",&x);
        for(int j=0;j<m;++j)c[j]=rem[j];
        rem[x%=m]=1;
        for(int j=0;j<m;++j)
            if(c[j])  rem[(x+j)%m]=1;
        if(rem[0])break;
    }
    puts(rem[0]?"YES":"NO");
    return 0;
}
