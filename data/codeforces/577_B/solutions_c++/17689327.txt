#include<iostream>
using namespace std;
bool dp[1010][1010];
int a[1010];
int main()
{
	int n,m,i,j;
	cin>>n>>m;
	if(n>=m){
		cout<<"YES"<<endl;
		return 0;
	}
	for(i=1;i<=n;i=i+1){
		cin>>a[i];
		dp[i][a[i]%m]=1;
		for(j=0;j<m;j=j+1)
			dp[i][(j+a[i])%m]=dp[i][(j+a[i])%m]|dp[i-1][j]|dp[i-1][(j+a[i])%m];
		if(dp[i][0]==1){
			cout<<"YES"<<endl;
			return 0;
		}
	}
	cout<<"NO"<<endl;
	return 0;
}
