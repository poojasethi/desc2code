#include <bits/stdc++.h>

using namespace std;

int main(){
	int n,m,x;
	scanf("%d%d",&n,&m);
	vector <bool> found(m);
	for(int i=0; i<n; i++){
		scanf("%d",&x);
		x%=m;
		vector <bool> old = found;
		for(int j=0; j<m; j++)
			if(old[j]) found[(j+x)%m] = true;
		if(found[0]) break;
		found[x] = true;
	}
	if(found[0]) printf("YES\n");
	else printf("NO\n");
	return 0;
}
