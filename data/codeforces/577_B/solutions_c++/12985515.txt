#include<cstdio>
int n,m,i,a[1000100],f[1010],j,g[1000100];
int main()
{

	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++)
	{
			scanf("%d",&a[i]);
			a[i]=a[i]%m;
			for (j=0;j<m;j++)
			f[j]=0;
			f[a[i]]=1;
			for (j=m-1;j>=0;j--)
			if (g[j]) f[(j+a[i])%m]=1;
			for (j=0;j<m;j++)
			if (f[j]) g[j]=f[j];
			
			if (g[0])
			{
				printf("YES");
				return 0;
			}
	}
	printf("NO");
}
