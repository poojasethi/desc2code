#include <bits/stdc++.h>
using namespace std;
map<int,int>h,t;
int main(){
	int j,n,i,a,m;cin>>n>>m;
	map<int,int>::iterator it;
	for(i=0;i<n;++i){
		cin>>a;a%=m;
		t= h;
		t[a]=1;
		for(it=h.begin();it!=h.end();++it)
			t[(it->first+a)%m]=1;
		h= t;
		if(h.count(0))break;
	}
	if(h.count(0))cout<<"YES\n";
	else cout<<"NO\n";
}
