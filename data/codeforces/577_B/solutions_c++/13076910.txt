#include <iostream>
#include <vector>
using namespace std;
int n,m;
bool v[1001];

int main(){
	cin >> n >> m;
	for(int i=0;i<n;i++){
		int a; cin >> a;
		vector<int> nv;
		
		for(int j=1;j<m;j++)
			if(v[j]) nv.push_back((a+j)%m);
			
		v[a % m]=1;	
		
		for(int j=0;j<nv.size();j++){
			v[nv[j]]=1;
		}
		if(v[0])break;	
	}
	
	if(v[0]) cout << "YES";
	else cout << "NO";
	return 0;
	
}
