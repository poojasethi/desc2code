#include<iostream>
#include<bitset>
using namespace std;
typedef bitset<1000> bt;
bt ans;
int n,m;
int main(){
	cin>>n>>m;
	if (n>m) {
		cout<<"YES";
		return 0;
	}
	ans[0]=1;
	for(int i=0;i<n;i++){
		int x;
		cin>>x;
		x%=m;
		bt nw,tmp=ans>>(m-x);
		nw=tmp |((ans^(tmp<<(m-x)))<<x);
		if (nw[0]){
			cout<<"YES";
			return 0;
		}
		ans|=nw;
	}
	cout<<"NO";
	return 0;
}
