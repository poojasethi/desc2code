#include<stdio.h>
#include<string.h>
const int maxm=1000+10;
int a[maxm];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	if (n>m) {printf("YES\n");return 0;}
	memset(a,-1,sizeof(a));
	for (int i=1;i<=n;i++)
	{
		int x;
		scanf("%d",&x);
		x%=m;
		if (a[x]==-1) a[x]=i;
		for (int j=0;j<m;j++)
		if (a[j]!=-1&&a[j]!=i&&a[(j+x)%m]==-1) a[(j+x)%m]=i;
	}
	if (a[0]==-1) printf("NO\n");
	else printf("YES\n");
	return 0;
}
