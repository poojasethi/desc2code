#include <bits/stdc++.h>
using namespace std;

int n;
int v[1005];
int v2[1005];
bool flag;

int main()
{
	int n,m,x;
	scanf("%d %d",&n,&m);

	for(int i=0;i<n;i++)
	{
		scanf("%d",&x);
		x = x%m;

		if(v[0])
		{
			printf("YES\n");return 0;
		}

		for(int j=0;j<m;j++)
			if(v2[j])
				v[(x+j)%m]=true;
		v[x]=true;
		
		memcpy(v2,v,sizeof v);
	}

	if(v[0])
		printf("YES\n");
	else
		printf("NO\n");

	return 0;
}
