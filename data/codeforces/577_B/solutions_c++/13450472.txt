#include<bits/stdc++.h>
using namespace std;
long a,m,n,i,j;
int main()
{
  bool b[1010] = {0},c[1010] = {0};
  cin>>n>>m;
  for(i=0;i<n;i++) { 
     cin>>a;a %= m;
     memset(c,0,sizeof(c));
     for(j=0;j<m;j++) if(b[j]) c[(a+j)%m] = 1;
     for(j=0;j<m;j++) if(c[j]) b[j] = c[j];
     b[a%m] = 1;
     if(b[0]) break;
  }
 if(b[0]) cout<<"YES";    
 else     cout<<"NO";              
return 0;
}
