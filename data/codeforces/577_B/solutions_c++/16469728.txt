#include<bits/stdc++.h>
using namespace std;
int n , m , a[1010] , dp[1010][1010];

void solve(int i , int mod){
	if(!mod){
		cout <<"YES\n";
		exit(0);
	}
	if(i==n) return;
	if(dp[i][mod]) return;
	dp[i][mod] = 1;
	solve(i+1,(((mod+a[i])%m)+m)%m);
	solve(i+1,mod);
}
int main(){
	cin >> n >> m;
	if(n>m) return puts("YES\n");
	for(int i = 0 ; i < n ; i++) scanf("%d",&a[i]);
	solve(0,m);
	cout << "NO" << endl;
}
