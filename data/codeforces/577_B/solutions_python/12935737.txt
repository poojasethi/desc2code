n,m=map(int,raw_input().split())
a=map(int,raw_input().split())
new =[0]*m
md = [0]*m
flag=0
for i in range(0,n):
    if flag==1:
        break
    new=[0]*m
    nm=a[i]%m
    for j in range(0,m):
        if md[j]==1:
            newmod=(nm+j)%m
            new[newmod]=1

    new[nm]=1

    for ti in range(0,m):
        md[ti]=(md[ti] | new[ti])
    if md[0]==1:
        flag=1
        break

if flag==1:
    print "YES"
else:
    print "NO"
