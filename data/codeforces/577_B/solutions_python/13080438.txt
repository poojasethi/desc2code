def modulo_summable(lst, modulo):
    if not lst:
        return False
    
    mods = set([0])
    
    for i, num in enumerate(lst):

        temp = set()
        for m in mods:
            x = (m + num) % modulo
            if x == 0:
                return True
            temp.add(x)

        mods = mods | temp

    
    return False
    

_, modulo = map(int, raw_input().split())
lst = map(int, raw_input().split())


if modulo_summable(lst, modulo):
    print "YES"
else:
    print "NO"
