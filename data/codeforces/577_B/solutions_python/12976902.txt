#!/usr/bin/python
import math
import sys

n, m = map(int, raw_input().split())
an = map(int, raw_input().split())

ansmap = [False]*m
ansl = [0]

for x in an:
    x = x % m
    new_ansl = []
    for ans in ansl:
        new_ans = (ans + x) % m
        if new_ans == 0:
            print "YES"
            sys.exit(0)
        if (ansmap[new_ans]): continue
        new_ansl.append(new_ans)
        ansmap[new_ans] = True
    ansl.extend(new_ansl)

print "NO"
