[n, m] = map(int, raw_input().split())
A = map(int, raw_input().split())
def main():
    need = set()
    for a in A:
        a = a % m
        if a == 0:
            print "YES"
            return
        if a in need:
            print "YES"
            return
        newset = set(map(lambda x: (x - a) % m, need))
        need = need | newset
        need.add(m - a)
    print "NO"
main()
