def SubsetSum(c, valor):
    somas = []
    bsomas = {}
    for elem in c:
        if not elem % valor:
            return True
        else:
            somas.append(elem % valor)
            bsomas[elem % valor] = 1
            for i in range(len(somas) - 1):
                if not (elem + somas[i]) % valor:
                    return True
                else:
                    try:
                        bsomas[(elem+somas[i]) % valor] += 1
                    except:
                        somas.append(elem + somas[i] % valor)
                        bsomas[(elem+somas[i]) % valor] = 1
    return False

k = [int(x) for x in raw_input().split()]
n, m = k[0], k[1]

c = [(int(x) % m) for x in raw_input().split()]
if SubsetSum(c, m):
    print "YES"
else:
    print "NO"
