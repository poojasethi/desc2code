def q(p):
    global m
    return int(p)%m


n,m = map(int,raw_input().split())

store = map(q,raw_input().split())

#print store

useless = [False]*m


ans = False
av = set([0])

for i in store:
    if i == 0:
        ans = True
        break
    if useless[i] == True:
        continue

    used = True
    
    k = av.copy()
    
    for a in k:
        g = (i+a)%m
        if g == 0:
            ans = True
            break
        if g not in av:
            av.update([g])
            used = False
   
 
    #print av
    useless[i] = used
    if ans is True:
        break


if ans is True:
    print "YES"
else:
    print "NO"
