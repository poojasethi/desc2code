n,m=map(int,raw_input().split())
if(n>m):
    print 'YES'
    exit()
a=map(int,raw_input().split())
dp1=[0]*m
dp1[0]=1
for i in xrange(n):
    dp2=[j for j in dp1]
    for j in xrange(m):
        if(dp1[j]):
            dp2[(j+a[i]%m)%m]|=dp1[j]
            if((j+a[i]%m)%m==0 and dp2[0]):
                print 'YES'
                exit()
    dp1=[j for j in dp2]
print 'NO'
