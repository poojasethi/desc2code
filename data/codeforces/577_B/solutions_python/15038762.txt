from math import *
from Queue import *
from sys import *




n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
mod = set()
extra = set()
for i in range(n):
    new = a[i]%m
    if new == 0:
        print('YES')
        exit(0)
    extra = set()
    for old in mod:
        if (old+new)%m == 0:
            print('YES')
            exit(0)
        extra.add((old+new)%m)
    mod = mod | extra
    mod.add(new)
print('NO')
