n, m = map(int, raw_input().split())

a = map(int, raw_input().split())

dp = [[False for _ in xrange(m)] for _ in xrange(2)]

for i in xrange(n):
	a[i] %= m
	dp[(i + 1) % 2][(m - a[i]) % m] = True
	if (m - a[i]) % m == 0:
		print "YES"
		exit()
	for k in xrange(m):
		if dp[i % 2][k]:
			dp[(i + 1) % 2][k] = True
			dp[(i + 1) % 2][(m + (k - a[i])) % m] = True
			if (m + (k - a[i])) % m == 0:
				print "YES"
				exit()
print "NO"
exit()
