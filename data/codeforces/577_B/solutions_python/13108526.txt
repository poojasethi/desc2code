n, m = [int(x) for x in raw_input().split()]
numbers = [int(x)%m for x in raw_input().split()]
sums = set()
found = False
for a in numbers:
    if a == 0:
        found = True
    else:
        new_sums = set()
        for s in sums:
            new_sums.add((s+a)%m)
        sums.add(a)
        sums.update(new_sums)
        if 0 in sums:
            found = True
    if found:
        print 'YES'
        break
if not found:
    print 'NO'
