n, m = map(int, raw_input().split())
seq = list(map(int, raw_input().split()))
if n > m:
    print 'YES'
else:
    f = [[0 for i in range(m)] for j in range (n + 1)]
    for i in range(n):
        for j in range(m):
            f[i + 1][j] = f[i + 1][j] or f[i][j]
            f[i + 1][(j + seq[i]) % m] = f[i + 1][(j + seq[i]) % m] or f[i][j]
        f[i + 1][seq[i] % m] = 1
    print 'YES' if f[n][0] else 'NO'
