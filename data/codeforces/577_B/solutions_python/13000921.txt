n,m=map(int,raw_input().split())
a=map(int,raw_input().split())
if n>m:
   print 'YES'
   exit()
else:
    divisible={}
    for x in a:
        temp=set()
        for y in divisible:
            temp.add((y+x)%m)
        for mod in temp:
            divisible[mod]=1
        divisible[x%m]=1
    if 0 in divisible:
        print 'YES'
        exit()
    print 'NO'
