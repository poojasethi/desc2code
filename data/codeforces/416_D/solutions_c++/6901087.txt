#include <iostream>

using namespace std;

const int MAXN = 200100;

int N;
int ar[MAXN];

int main() {
  cin >> N;
  for (int i = 0; i < N; ++i) {
    cin >> ar[i];
  }

  int i = 0;
  int ans = 0;
  while (i < N) {
    ++ans;

    int a = i;
    while (a < N && ar[a] == -1) ++a;
    if (a == N) break;

    int b = a + 1;   
    while (b < N && ar[b] == -1) ++b;
    if (b == N) break;

    if ((ar[b] - ar[a]) % (b - a) != 0) {
      i = b;
      continue;
    }

    long long diff = (ar[b] - ar[a]) / (b - a);
    if ((long long) ar[b] - diff * (b - i) <= 0) {
      i = b;
      continue;
    }
    int c = b;
    while (c < N && (ar[c] == -1 || ar[c] == ar[a] + diff * (c - a))) {
      if ((long long) ar[a] + diff * (c - a) <= 0) break;
      ++c;
    }
    i = c;
  }
  cout << ans << endl;
  return 0;
}
