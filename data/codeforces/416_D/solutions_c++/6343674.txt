using namespace std;

long long n;
long long a[222222];
long long inf=1111111111,ans=0;

void work(long long cur){
if(cur==n)return ;
ans++;
long long s=-1,t=-1,end=cur;
for(long long i=cur;i<n;i++)
if(a[i]!=-1){
if(s==-1)s=i;
else {t=i;end=i;break;}
}
if(end==cur)return ;//cout<<2;
if(abs(a[t]-a[s])%(t-s)){
work(t);
}
else{
long long k=(a[t]-a[s])/(t-s);
if(a[t]-(t-cur)*k<=0)work(t);
else{
long long f=t+1;
while(a[f-1]+k>0&&f<n){
if(a[f]==-1)a[f]=a[f-1]+k;
else if(a[f]!=a[f-1]+k)break;
f++;
}
work(f);
}
}
}

int main(){
cin>>n;
for(long long i=0;i<n;i++)cin>>a[i];
work(0);
cout<<ans<<endl;
}
