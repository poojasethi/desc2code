#include<iostream>
using namespace std;
long long a[210000],n;
int main(){
    cin>>n;
    for(int i=0;i<n;i++)cin>>a[i];
    int k=0,ans=0;
    while(k<n){
        ans++;
        int i=k;
        while (a[i]==-1)i++;
        int j=i+1;
        while (a[j]==-1)j++;
        if (j>=n) break;
        if ((a[j]-a[i])%(j-i)) {
            k=j;continue;
        }
        long long d=(a[j]-a[i])/(j-i);
        if (a[j]-d*(j-k)<=0) {
            k=j;continue;
        }
        k=j+1;
        while (k<n&&a[j]+d*(k-j)>0&&(a[k]==-1||a[k]==a[j]+d*(k-j))) k++;
    }
    cout<<ans;
    return 0;
}
