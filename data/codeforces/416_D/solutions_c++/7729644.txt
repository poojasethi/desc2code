#include <cstdio>
#include <algorithm>
using namespace std;
long long a[1000005],b,c,d,e,f,g,h,i,j,k,l,m,n,last,gc;
inline void New(long long x){
	if(a[x]!=-1){last=x;}else {last=-1;l=1;}
	gc=(1ll<<50);
	h++;
}
int main(){
	scanf("%I64d",&n);
	for(i=1;i<=n;i++){
		scanf("%I64d",&a[i]);
	}
	last=-1;h=1;gc=(1ll<<50);l=0;
	for(i=n;i;i--){
		if (a[i]==-1){if(last==-1)l++;if(last==-1)continue;if(gc==(1ll<<50))continue;if(a[last]-gc*(last-i)<=0){l=1;New(i);}continue;}
		if (last==-1){last=i;continue;}
		if (gc==(1ll<<50)){
			if((a[last]-a[i])%(last-i)){New(i);l=0;continue;}
			gc=(a[last]-a[i])/(last-i);
			if(gc<0)if(a[last]+gc*l<=0){New(i);l=0;continue;}
			last=i;l=0;
			continue;
		}
		if(a[last]-gc*(last-i)!=a[i]){New(i);l=0;}
		else {last=i;l=0;}
	}
	printf("%I64d",h);
	return 0;
}
