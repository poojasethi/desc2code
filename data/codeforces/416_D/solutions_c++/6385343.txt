#include <cstdio>
#include <cstdlib>
#include <algorithm>

using namespace std;

int n;
const int N=200010;
int a[N];
bool check=1;

int main()
{
  scanf("%d",&n);
  for (int i=1;i<=n;i++)
    {
      scanf("%d",&a[i]);
      if (a[i]!=-1) check=0;
    }
  if (check)
    {
      printf("1\n");
      return 0;
    }
  int ans=0;  
  for (int i=1;i<=n;)
    {
      int now=i;
      while (now<=n&&a[now]==-1) now++;
      if (now>n)
	{
	  ans++;
	  break;
	}
      int nn=now+1;
      while (nn<=n&&a[nn]==-1) nn++;
      if (nn>n)
	{
	  ans++;
	  break;
	}
      if ((a[nn]-a[now])%(nn-now))
	{
	  ans++;
	  i=nn;
	  continue;
	}
      else
	{
	  int del=(a[nn]-a[now])/(nn-now);
	  if (a[i]==-1)
	    if (-(long long)del*(now-i)+a[now]<=0)
	      {
		ans++;
		i=nn;
		continue;
	      }
	  long long tmp=a[nn];
	  int ii;
	  for (ii=nn;ii<=n&&tmp>0&&(a[ii]==-1||a[ii]==tmp);ii++,tmp+=del);
	  i=ii;
	  ans++;
	}
    }
  printf("%d\n",ans);
  return 0;
}
