
#include <cstdio>

using namespace std;

#define N (1<<18)

int n;
int Ans;
long long a[N];

int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%I64d",&a[i]);
	}
	int k=0;
	while(k<n)
	{
		Ans++;
		int t1=k;
		while(a[t1]==-1ll)
		{
			t1++;
		}
		int t2=t1+1;
		while(a[t2]==-1ll)
		{
			t2++;
		}
		
		if(t2>=n)
		{
			break;
		}
		
		if((a[t2]-a[t1])%(t2-t1)!=0)
		{
			k=t2;
			continue;
		}
		if(a[t2]-(a[t2]-a[t1])/(t2-t1)*(t2-k)<=0)
		{
			k=t2;
			continue;
		}
		
		k=t2;
		while(true)
		{
			if((k>=n)||(a[t2]+(a[t2]-a[t1])/(t2-t1)*(k-t2)<=0)||(a[k]!=-1ll&&a[k]!=a[t2]+(a[t2]-a[t1])/(t2-t1)*(k-t2)))
			{
				break;
			}
			k++;
		}
	}
	
	printf("%d\n",Ans);
	return 0;
}
