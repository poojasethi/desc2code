#include<cstdio>
#define C continue
#define N 200005
using namespace std;
long long a[N],check,start1,start2,ans,i,div,n,upd,mi,cnt;
int main()
{
  scanf("%I64d",&n);ans=1;
  for (i=1;i<=n;i++)
    scanf("%I64d",&a[i]);
  for (i=1;i<=n;i++)
    if (a[i]!=-1)
    {
      if (!start1) {start1=i;C;}
      if (start1&&!start2) 
      {
        start2=i;upd=a[start2]-a[start1];mi=start2-start1;div=upd/mi;
        if (upd%mi!=0||a[start1]-div*cnt<=0) {start1=i;cnt=start2=0;ans++;C;}
      }
      if (a[i]!=a[start2]+div*(i-start2)) {ans++;start1=i;start2=0;cnt=0;C;}
    }
    else
    {
      if (start1==0) cnt++;
      if (start1&&start2&&a[start2]+div*(i-start2)<=0) {ans++;start1=start2=0;cnt=1;C;}
    }
  printf("%I64d",ans);
  return 0;
}
