#include <cstdio>
#include <iostream>
#include <cstring>
#define ll long long
using namespace std;
const int maxn=2e5+10;

ll a[maxn],n;

ll check(ll head){
    for(ll i=head;i<=n;i++)
        if(a[i]!=-1)return i;
    return (n+1);
}

ll dfs(ll head){
    ll i1=check(head);
    ll i2=check(i1+1);
    
    if(i1>n||i2>n)return n+1;
    
    if((a[i2]-a[i1])%(i2-i1)!=0)return i2;
    
    ll d=(a[i2]-a[i1])/(i2-i1);
    
    if(a[i1]-d*(i1-head)<1)return i2;
    
    ll sum=a[i2],i;
    for(i=i2+1;i<=n;i++){
        sum+=d;
        if(sum<1)return i;
        if(a[i]!=-1&&a[i]!=sum)return i;
    }
    
    return i;
 }

int main(){
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    ll ans=0,head=1;
    
    while(head<=n){
        ans++;
        head=dfs(head);
    }
    cout<<ans<<endl;
}
