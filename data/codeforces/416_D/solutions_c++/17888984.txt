#include<cstdio>
#include<algorithm>
#include<cstring>
#include<iostream>
#include<cstdlib>
using namespace std;

#define maxn 200100
typedef long long ll;
ll a[maxn];
int n;
int p[5];

int main()
{
	ios::sync_with_stdio(false);
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	
	int s=1,cnt=0,num=0;
	ll dif;
	int ans=0;
	for(int i=1;i<=n;i++)
	{
		num++;
		if (a[i]!=-1 && cnt<2)
			p[++cnt]=i;
		if (cnt==2)
		{
			if ((a[p[2]]-a[p[1]])%(p[2]-p[1])!=0 || (a[i]!=-1 && a[i]!=a[p[2]]-dif*(p[2]-i)))
			{
				ans++;
//				cout<<i<<endl;
				s=i;cnt=0;num=0;
				i--;
				continue;
			}
			dif=(a[p[2]]-a[p[1]])/(p[2]-p[1]);
			ll fval=a[p[2]]-dif*(p[2]-s);
			ll lval=a[p[2]]-dif*(p[2]-i);
			if (fval<=0 || lval<=0)
			{
				ans++;
//				cout<<i<<endl;
				s=i;cnt=0;num=0;
				i--;
				continue;
			}
		}
	}
	if (num) ans++;
	printf("%d\n",ans);
	return 0;
}
