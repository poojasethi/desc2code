#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
typedef long long ll;
struct prog
{
	int num;
	vector <ll> p,q;
	inline void init()
	{
		num=0;
		p.clear(); q.clear();
	}
	int sz() {return num;}
	bool ins(ll x)
	{
		num++;
		if (x!=-1 && p.size()<2)
		{
			p.push_back(x);
			q.push_back(num);
		}
		if (p.size()<2) return true;
		if ((p[1]-p[0])%(q[1]-q[0])) return false;
		ll d=(p[1]-p[0])/(q[1]-q[0]);
		ll fst=p[0]-(q[0]-1)*d;
		ll lst=p[0]+(num-q[0])*d;
		if (x!=-1 && x!=lst) return false;
		if (fst<1 || lst<1) return false;
		return true;
	}
};
int main()
{
	int n;
	scanf("%d",&n);
	prog cur;
	cur.init();
	int ans=0;
	for (int i=1; i<=n; i++)
	{
		int x;
		scanf("%I64d",&x);
		if (!cur.ins(x))
		{
			cur.init();
			cur.ins(x);
			ans++;
		}
	}
	if (cur.sz()>0) ans++;
	printf("%d\n",ans);
	return 0;
}
