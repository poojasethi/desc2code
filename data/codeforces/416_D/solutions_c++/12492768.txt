#include <iostream>
using namespace std;
long long a[200005];
int main(){
	int n;
	cin>>n;
	for (int i=1;i<=n;i++){
		cin>>a[i];
	}
	int k=0;
	int ans=0;
	for (k=1;k<=n;){
		ans++;
		int fir=k;
		while (a[fir]==-1){
			fir++;
		}
		int sec=fir+1;
		while (a[sec]==-1){
			sec++;
		}
		//cout<<fir<<" "<<sec<<endl;
		if (sec>n){
			break;
		}
		if ((a[sec]-a[fir])%(sec-fir)!=0){
			k=sec;
			continue;
		}
		long long d=(a[sec]-a[fir])/(sec-fir);
		//cout<<d<<endl;
		if (a[sec]-d*(sec-k)<1){
			k=sec;
			continue;
		}
		for (k=sec+1;k<=n && a[sec]+d*(k-sec)>=1 && (a[k]==-1 || a[k]==a[sec]+d*(k-sec));k++);
	}
	cout<<ans;
	return 0;
}
