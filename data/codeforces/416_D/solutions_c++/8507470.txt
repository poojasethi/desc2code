#include <cstdio>
#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdlib>
#include <string>

using namespace std;
typedef long long ll;
typedef double D;
const int N=200010;
ll n,num,a[N],i,j,k,t,ans;
int main(){
    scanf("%I64d",&n);
    for (i=1;i<=n;i++) scanf("%I64d",&a[i]);
    for (k=1;k<=n;) {
        ans++;//k为一个新等差数列的开始位置
        for (i=k;a[i]==-1;i++);
        for (j=i+1;a[j]==-1;j++);
        if (j>n) break;
        if ((a[j]-a[i])%(j-i)) {
            k=j;
            continue;
        }
        t=(a[j]-a[i])/(j-i);
        if (a[j]-t*(j-k)<=0) {
            k=j;
            continue;
        }
        for (k=j;a[j]+t*(k-j)>0&&(a[k]==-1||a[k]==a[j]+t*(k-j));k++);
    }
    printf("%d\n",ans);
    return 0;
}
