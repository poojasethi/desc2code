#include<cstdio>
#include<algorithm>
#include<cstring>
#include<climits>
using namespace std;
int n;
#define MAXN 200010
long long data[MAXN];
int main(){
    int ans=0;scanf("%d",&n);for(int i=1;i<=n;i++)scanf("%I64d",&data[i]);
    for(int k=1,i,j;k<=n;){
        ans++;
        for(i=k;i<=n&&data[i]==-1;i++);
        for(j=i+1;j<=n&&data[j]==-1;j++);
        if(j>n)break;
        if((data[j]-data[i])%(j-i)){
            k=j;continue;
        }long long d=(data[j]-data[i])/(j-i);
        if(data[j]-d*(j-k)<=0){
            k=j;continue;
        }for(k=j+1;k<=n&&(data[j]+d*(k-j)>0)&&(data[k]==-1||data[k]==data[j]+d*(k-j));k++);
    }printf("%d\n",ans);
}
