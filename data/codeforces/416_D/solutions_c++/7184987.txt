#include<stdio.h>
__int64 s[222222];
int main()
{
	int n,i;
	scanf("%d",&n);
	int ans=0;
	int j;
	for(i=0;i<n;i++)
	scanf("%I64d",&s[i]);
	i=0;
	while(i<n)
	{
		int l,r;
		__int64 d;
		int s1=0;
		while(s[i]==-1)
		{
			i++;s1++;
		}
		l=i;
		if(i<n)
		i++;
		while(s[i]==-1&&i<n)
		i++;
		if(i==n)
		{ans++;break;}
		r=i;
		if((s[r]-s[l])%(r-l)==0)
		d=(s[r]-s[l])/(r-l);
		else {
			ans++;
			continue;
		}
		if(d>0&&s1*d>(s[l]-1))
		{
			ans++;
			continue;
		}
		__int64 sum=s[r];
		for(j=r+1;j<n;j++)
		{
			sum+=d;
		
			if(sum<1)
			{
			ans++;
			i=j;	
			break;
			}
			if(s[j]!=-1&&sum!=s[j])
			{
				ans++;
				i=j;
				break;
				
			}
		}
		if(j==n)i=n,ans++;
	}
	printf("%d\n",ans);
}
