#include<iostream>
using namespace std;
long long a[200002];
int main()
{
	long long n,i,j,d,res=0;
	cin>>n;
	for(int i=0;i<n;i++) cin>>a[i];
	for(int k=0;k<n;)
	{
		res++;
		for(i=k;a[i]==-1;i++);
		for(j=i+1;a[j]==-1;j++);
		if(i>=n||j>=n) break;
		d=(a[j]-a[i])/(j-i);
		if((j-i)*d!=a[j]-a[i]) {
			k=j;
			continue;
		}
		if(a[j]<=(j-k)*d)
		{
			k=j;
			continue;
		}
		for(;a[k]==-1&&a[i]>d*(i-k)&&k<n||a[k]!=-1&&a[k]==a[i]+d*(k-i)&&k<n;k++);
	}
	cout<<res;
	return 0;
}
