#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
long long n,t,num,i,j,f=1,ans=1,ok=0;
vector<long long> v,p;
int main()
{
	cin>>n;
	num=0;
	v.clear();
	p.clear();
	for (i=1;i<=n;i=i+1)
	{
		cin>>t;
		if (t!=-1) ok++;
		num=num+1;
			//cout<<ans<<endl;
		if (t!=-1 && v.size()<2)
		{
			v.push_back(t);
			p.push_back(num-1);
		}
		if (v.size()<2) continue;
		if (((v[1]-v[0])%(p[1]-p[0]))!=0) f=0;
		long long d=(v[1]-v[0])/(p[1]-p[0]);
		long long first=v[0]-p[0]*d;
		long long last=v[0]+(num-1-p[0])*d;
		if (first<=0 || last<=0) f=0;
		if (t!=-1 && t!=last) f=0;
		if (f==0)
		{
			num=0;
			v.clear();
			p.clear();
			ans=ans+1;
			num=1;
			if (t!=-1)
			{
				v.push_back(t);
				p.push_back(num-1);
			}
			f=1;
		}
	}
	//if (v.size()>0) ans=ans+1;
	if (!ok || ok==1) cout<<1<<endl;else cout<<ans<<endl;
	return 0;
}
