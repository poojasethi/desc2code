#include <bits/stdc++.h>
using namespace std;
const int maxn = 222222;
__int64 a[maxn];
int n, col;
void solve(){
	int i = 0, j = 0, i1, i2;
	while(i < n){
		col ++;
		j = i;
		while(i < n && a[i] == -1ll) i ++;
		if(i == n) break;
		i1 = i ++;
		while(i < n && a[i] == -1ll) i ++;
		if(i == n) break;
		i2 = i;
		if((a[i2] - a[i1]) % (i2 - i1)) continue;
		__int64 d = (a[i2] - a[i1]) / (i2 - i1);
		if(a[i1] - (i1 - j) * d <= 0) continue;
		i ++;
		while((a[i] == -1 && a[i - 1] + d > 0) || (a[i] == a[i - 1]  + d && a[i - 1] + d > 0)){
			a[i] = a[i - 1] + d;
			i ++;
		}
	}
}
void input(){
	cin >> n;
	for(int i = 0; i < n; i ++) cin >> a[i];
}
void output(){
	cout << col;
}
int main(){
	input();
	solve();
	output();
}
