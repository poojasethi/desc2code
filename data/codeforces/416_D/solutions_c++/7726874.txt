#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n,ans;
long long a[200005];

int main()
{
/*
    freopen("revenge.in","r",stdin);
    freopen("revenge.out","w",stdout);
*/
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%I64d",&a[i]);
    for(int k=1;k<=n;)
    {
        ++ans;
        int i,j;
        for(i=k;a[i]==-1;i++);
        for(j=i+1;a[j]==-1;j++);
        if(j>n)break;
        if((a[j]-a[i])%(j-i)) {k=j;continue;}
        long long d=(a[j]-a[i])/(j-i);
        if(a[j]-d*(j-k)<=0){k=j;continue;}
        for(k=j+1;(a[j]+d*(k-j)>0)&&(k<=n)&&((a[k]==-1)||(a[k]==a[j]+d*(k-j)));k++);
    }
    printf("%d\n",ans);
    return 0;
}
