#include <cstdio>

#define N 200000

using namespace std;

int n,ans;
long long a[N+10];

int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;++i) scanf("%I64d",&a[i]);
	for (int i=1;i<=n;){
		ans++;
		int l=i;
		while (l<=n && a[l]==-1) l++;
		if (l>n) break;
		int r=l+1;
		while (r<=n && a[r]==-1) r++;
		if (r>n) break;
		if ((a[r]-a[l])%(r-l)){i=r;continue;}
		long long dlt=(a[r]-a[l])/(r-l);
		if (dlt>0 && a[l]-dlt*(l-i)<=0){i=r;continue;}
		int rr=r+1;
		for (;rr<=n;++rr){
			long long val=a[r]+(long long)(rr-r)*dlt;
			if (a[rr]!=-1 && a[rr]!=val || val<=0) break;
		}
		i=rr;
	}
	printf("%d\n",ans);
	return 0;
}
