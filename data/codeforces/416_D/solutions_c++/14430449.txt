#include <bits/stdc++.h>
using namespace std;

typedef long long lint;
const lint maxn = 2e5+5;

lint n,ar[maxn];

int main(){
	
	scanf("%lld",&n);
	
	for( lint i=1 ; i <= n ; i++ )
		scanf("%lld",&ar[i]);
	
	lint ans = 0;
	
	for( lint i=1 ; i <= n ; ){
		ans++;
		
		lint j = i;
		while( j <= n and ar[j] == -1 ) j++;
		
		if( j == n+1 ) break;
		
		lint k = j+1;
		while( k <= n and ar[k] == -1 ) k++;
		
		if( k == n+1 ) break;
		
		if( (ar[k] - ar[j]) % (k - j) ){
			i = k;
			continue;
		}
		
		lint step = (ar[k] - ar[j]) / (k - j);
		
		if( step > 0 and ar[j] - step * (j - i) <= 0 ){
			i = k;
			continue;
		}
		
		lint r = k+1;
		
		while( r <= n ){
			if( (ar[r] != -1 and ar[r] != ar[k] + step * (r - k)) or ar[k] + step * (r - k) <= 0 ) break;
			r++;
		}
		i = r;
	}
	
	cout << ans << endl;
	
	return 0;
}
