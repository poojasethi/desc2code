#include <cstdio>
#include <cstdlib>
using namespace std;

int x;
long long n,ans,res;
long long q[500001],id[500001],cnt,d,Flag,last;

int main()
{
	scanf("%I64d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&x);
		if (x!=-1) { q[++cnt]=x; id[cnt]=i; }
	}
	d=1; Flag=0; ans=1; last=1;
	for (int i=2;i<=cnt;i++)
	{
		long long num=q[i]-q[i-1];
		long long wei=id[i]-id[i-1];
		int flag=1;
		if ((num%wei!=0)) flag=0;
		else {
			long long td=(q[i]-q[i-1])/(id[i]-id[i-1]);
			if (Flag&&td!=d) flag=0;
		}
		if (!flag)
		{
			if (Flag&&q[i-1]+d*(id[i]-id[i-1]-1)<=0) 
			{
				ans++; Flag=0; last=id[i-1]+(q[i-1]-1)/(-d)+1;
			} else {
				ans++; Flag=0; last=id[i];
			}
		} else if (!Flag) {
			d=(q[i]-q[i-1])/(id[i]-id[i-1]);
			if (q[i]-(id[i]-last)*d>0) Flag=1;
			else { ans++; Flag=0; last=id[i]; }
		}
	}
	if (Flag) if (q[cnt]+d*(n-id[cnt])<=0) ans++;
	printf("%I64d\n",ans);
}
