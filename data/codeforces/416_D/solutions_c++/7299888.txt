#include <stdio.h>
#include <iostream>
using namespace std;
#define ll long long
int n,v=0,x,y,k=0;
ll f[200005],d;
int main( ){
	cin>>n;
	for (int i=0;i<n;i++)
		cin>>f[i];
	while (k<n){
		v++;
		x=k;
		while (x<n && f[x]==-1) x++;
		y=x+1;
		while (y<n && f[y]==-1) y++;
		if (y==n) break ;
		if ((f[y]-f[x])%(y-x)){
			k=y;
			continue ;
		}
		d=(f[y]-f[x])/(y-x);
		if (f[x]-d*(x-k)<=0){
			k=y;
			continue ;
		}
		k=y+1;
		while (k<n && (f[k]==f[y]+(k-y)*d || f[k]==-1) && f[y]+(k-y)*d>0)
			k++;
	}
	cout<<v<<endl;
	return 0;
}
