#include <cstdio>
typedef long long ll;
const int N=2e5+5;
int n,ans;
ll a[N];
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%I64d",a+i);
	for(int k=1,i,j;k<=n;){
		ans++;
		for(i=k;i<=n&&a[i]==-1;i++);
		for(j=i+1;j<=n&&a[j]==-1;j++);
		if(j>n) break;
		if((a[j]-a[i])%(j-i)){ k=j; continue; }
		ll d=(a[j]-a[i])/(j-i);
		if(a[j]-d*(j-k)<=0){ k=j; continue; }
		for(k=j+1;k<=n&&a[j]+d*(k-j)>0&&(a[k]==-1||a[k]==a[j]+d*(k-j));k++)
			a[k]=a[j]+d*(k-j);
	}
	printf("%d\n",ans);
}
