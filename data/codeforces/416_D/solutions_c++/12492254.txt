using namespace std;
int n,ans;
long long a[200005];
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	int start=1;
	while(start<=n)
	{
		ans++;
		int tmp1,tmp2;
		for(tmp1=start;a[tmp1]==-1;tmp1++);
		for(tmp2=tmp1+1;a[tmp2]==-1;tmp2++);
		if(tmp2>n) break;
		if((a[tmp2]-a[tmp1])%(tmp2-tmp1)) 
		{
			start=tmp2;
			continue;
		}
		long long dif=(a[tmp2]-a[tmp1])/(tmp2-tmp1);
		long long s=a[tmp2]-dif*(tmp2-start);
		if(s<1)
		{
			start=tmp2;
			continue;
		}
		start=tmp2+1;
		while(start<=n && a[tmp2]+dif*(start-tmp2)>0 && (a[start]==-1 || a[start]==a[tmp2]+dif*(start-tmp2)))
		{
			start++;
		}
	}
	cout<<ans;
	return 0;
}
