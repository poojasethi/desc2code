#include <stdio.h>
#include <cstring>
#include <algorithm>
using namespace std;
int n,res,npos=-1,d,pre=-1;
int a[200005];
bool first,second;
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++)scanf("%d",&a[i]);
	for(int i=0;i<n;i++){
		if(a[i]!=-1){
			if(!first)first=true;
			else if(!second){
				if((a[i]-a[pre])%(i-pre)){
					npos=-1;res++;
				}
				else{
					d=(a[i]-a[pre])/(i-pre);
					if(npos==-1 || a[i]-1LL*(i-npos)*d>0){
						second=true;
					}
					else{
						npos=-1;res++;
					}
				}
			}
			else{
				if(a[i]-a[pre]!=1LL*(i-pre)*d){
					npos=-1;res++;second=false;
				}
			}
			pre=i;
		}
		else{
			if(second && a[pre]+1LL*(i-pre)*d<=0){
				first=second=false;
				npos=i;res++;
			}
			else if(npos==-1)npos=i;
		}
	}
	printf("%d\n",res+1);
	return 0;
}
