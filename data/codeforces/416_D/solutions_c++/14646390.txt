#define LL long long
#define db double
#define EPS 1e-15
#define inf 1e10

using namespace std;

LL n,num,a[200010],i,j,k,t,ans;
int main(){
    scanf("%I64d",&n);
    for (i=1;i<=n;i++) scanf("%I64d",&a[i]);
    for (k=1;k<=n;) {
        ans++;
        for (i=k;a[i]==-1;i++);
        for (j=i+1;a[j]==-1;j++);
        if (j>n) break;
        if ((a[j]-a[i])%(j-i)) {
            k=j;
            continue;
        }
        t=(a[j]-a[i])/(j-i);
        if (a[j]-t*(j-k)<=0) {
            k=j;
            continue;
        }
        for (k=j;a[j]+t*(k-j)>0 && (a[k]==-1||a[k]==a[j]+t*(k-j));k++);
    }
    printf("%d\n",ans);
    return 0;
}
