#include<cstdio>
#include<iostream>
#include<vector>
#define ll long long
using namespace std;

vector<ll> v,p;
int num;
void init(){
     v.clear();
     p.clear();
     num=0;
}

bool add(int x){
     if(v.size()<2 && x!=-1)
     {  v.push_back(x);
        p.push_back(num);
     }
     num++;
     if(v.size()<2) return true;
     if((v[1]-v[0])%(p[1]-p[0])) return false;
     ll d=(v[1]-v[0])/(p[1]-p[0]);
     ll t=v[0]+1LL*d*(num-1-p[0]);
     ll h=v[0]-1LL*d*p[0];
     if(x!=t && x!=-1) return false;
     if(t<1 || h<1) return false;
     return true;
}
     

int main(){
    int n,a,i,ans=0;
    scanf("%d",&n);
    init();
    for(i=1;i<=n;i++)
    {   scanf("%d",&a);
        if(!add(a))
        {  init();
           int f=add(a);
           ans++;
        }
    }
    if(num>0) ans++;
    printf("%d\n",ans);
    return 0;
}
