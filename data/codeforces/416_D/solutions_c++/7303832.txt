#include <iostream>
#include <vector>
using namespace std;
vector <long long> val,pos;
long long num,dif,f,l,n,i,ans,x;
bool ad(long long v)
{
	num++;
	if (v!=-1 && val.size()<2)		//v!=-1!!!!!
	{
		val.push_back(v);
		pos.push_back(num);
	}
	if (val.size()<2) return true;
	if ((val[1]-val[0])%(pos[1]-pos[0])) return false;
	dif=(val[1]-val[0])/(pos[1]-pos[0]);
	f=val[0]-(pos[0]-1)*dif;
	l=val[0]+(num-pos[0])*dif;
	if (v!=-1 && v!=l) return false;
	if (f<1 || l<1) return false;//!!!!不用考虑>10^9 
	return true;
	
}
int main()
{
	cin>>n;
	val.clear();	//!!!!!
	pos.clear();	//!!!!!
	for (i=1;i<=n;i++)
	{
		cin>>x;
		if (!ad(x))
		{
			val.clear();
			pos.clear();
			num=0;
			ad(x);
			ans++;
		}
	}
	//if (val.size()>0 || pos.size()>0) ans++;	
	if (num>0) ans++;	//!!!!!
	cout<<ans<<endl;
	return 0;
}
