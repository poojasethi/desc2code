#include <cstdio>
#include <algorithm>
using namespace std;
int n,m;
int st,ed,i,ans=0;
long long a[200005],inc;
int main()
{
     scanf("%d",&n);
     for (i=1;i<=n;i++) scanf("%I64d",&a[i]);
     for (i=1;i<=n;)
     {
          ans++;
          for (st=i;a[st]==-1;st++);
          for (ed=st+1;a[ed]==-1;ed++);
          if (ed>n) break;
          if ((a[ed]-a[st])%(ed-st)!=0) {i=ed;continue;}
          inc=(a[ed]-a[st])/(ed-st);
          if (a[ed]-inc*(ed-i)<=0) {i=ed;continue;}
          for (i=ed+1;(i<=n) && (a[ed]+inc*(i-ed)>0) && (a[i]==-1 || a[i]==a[ed]+inc*(i-ed));i++);
     }
     printf("%d",ans);
     return 0;
}
