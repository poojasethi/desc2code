#include <cstdio>
const int MAXN = 2e5+10;

int A[MAXN],N,ans;
int main(){
	scanf("%d",&N);
	for(int i=1;i<=N;i++) scanf("%d",&A[i]);

	int one=0,two=0,pre=-1,arb=-1,d;
	for(int i=1;i<=N;i++){
		if(A[i] > 0){
			if(!one) one = true;
			else if(!two){
				if((A[i]-A[pre])%(i-pre)){
					ans++;
					arb = -1;
				}else{
					d = (A[i]-A[pre])/(i-pre);
					if(arb == -1 || A[i] - 1LL*(i-arb)*d > 0) two = true;
					else{
						ans++;
						arb = -1;
					}
				}
			}else{
				if(A[i]-A[pre] != 1LL*(i-pre)*d){
					ans++;
					two = false;
					arb = -1;
				}
			}
			pre = i;
		}else{
			if(two && A[pre]+1LL*(i-pre)*d <= 0){
				ans++;
				one = false;
				two = false;
				arb = i;
			}else if(arb == -1) arb = i;
		}
	}
	printf("%d\n",++ans);
	return 0;
}
