#include <iostream>

using namespace std;

typedef long long int ll;

int n;
ll a[1000000];

int main()
{
  cin>>n;
  for (int i=0;i<n;i++)
    cin>>a[i];
  int num=0;
  int iini=0;
  while (iini<n) {
    num++;
    int i1=iini;
    while (i1<n and a[i1]==-1) i1++;
    if (i1==n) break;
    int i2=i1+1;
    while (i2<n and a[i2]==-1) i2++;
    if (i2==n) break;
    ll a12=a[i2]-a[i1];
    ll d12=i2-i1;
    if (a12%d12!=0) {
      iini=i2;
      continue;
    }
    ll p=a12/d12;
    if (a[i1]-p*(i1-iini)<=0) {
      iini=i2;
      continue;
    }
    int i3=i2+1;
    while (i3<n and
	   ((a[i3]==-1 and a[i2]+p*(i3-i2)>=1) or (a[i3]!=-1 and a[i3]==a[i2]+p*(i3-i2))))
      i3++;
    iini=i3;
  }
  cout<<num<<endl;
}
