
def getValue(a, chainStart, chainRealStart, chainStep, i):
    if chainRealStart == None or chainStep == None:
        return None
    return a[chainRealStart] + chainStep * (i - chainRealStart)

def main():
    n = int( raw_input() )

    a = map(int, raw_input().split() )
    n = len(a)
    chainCount = 1
    chainStart = 0
    chainRealStart = None
    chainStep = None
    for i in xrange(n):
        if chainRealStart == None and a[i] != -1:
            chainRealStart = i

        if a[i] == -1:
            v = getValue(a, chainStart, chainRealStart, chainStep, i)
            if v != None and v <= 0:
                chainCount += 1
                chainStart = i
                chainRealStart = None
                chainStep = None
            continue

        idiff = i - chainRealStart

        if chainStep == None and idiff > 0:
            if (a[i] - a[chainRealStart]) % idiff == 0:
                chainStep = (a[i] - a[chainRealStart]) / idiff
                v = getValue(a, chainStart, chainRealStart, chainStep, chainStart)
                if v == None or v > 0:
                    continue
            chainCount += 1
            chainStart = i
            chainRealStart = i
            chainStep = None                
            continue

        v = getValue(a, chainStart, chainRealStart, chainStep, i)
        if v != None and v != a[i]:
            chainCount += 1
            chainStart = i
            chainRealStart = i
            chainStep = None                
            continue


    print chainCount

if __name__ == "__main__":
    main()
