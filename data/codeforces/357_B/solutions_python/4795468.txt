from collections import defaultdict
import itertools
import sys

n, m = map(int, raw_input().split())

cols = {}
for i in xrange(m):
    a = map(int, raw_input().split())
    ac = [1, 2, 3]

    for d in a:
        if d in cols:
            ac.remove(cols[d])
    for d in a:
        if not d in cols:
            cols[d] = ac.pop()

print ' '.join(str(x[1]) for x in sorted(cols.items()))
