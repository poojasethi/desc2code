def main():
	n, m = map(int, raw_input().split())
	ans = [0] * (n + 1)
	for i in xrange(m):
		a = map(int, raw_input().split())
		x = [1, 2, 3]
		for j in xrange(3):
			if ans[a[j]] != 0: del x[ans[a[j]] - 1]
		for j in xrange(3):
			if 0 == ans[a[j]]:
				ans[a[j]] = x[-1]
				del x[-1]
	print " ".join(map(str, ans[1:]))	

if __name__ == "__main__": main()
