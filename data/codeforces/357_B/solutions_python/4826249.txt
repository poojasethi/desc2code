n, m = map(int, raw_input().split())
l = []
color = [0 for i in range(n + 1)]
h = set()
for i in range(m):
    l.append(map(int, raw_input().split()))

for i in l:
    t = filter(lambda x: x not in h, i)
    f = filter(lambda x: x in h, i)
    c = range(1, 4)
    if f:
        c.remove(color[f[0]])
    for i in range(len(t)):
        h.add(t[i])
        color[t[i]] = c[i]
