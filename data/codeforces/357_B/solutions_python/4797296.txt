a = []
for i in range(m):
    a.append(map(int, raw_input().split()))
b = [0] * n
for i in a:
    l = i[:]
    l.sort()
    a1 = [1, 1, 1]
    for j in range(3):
        l[j] -= 1
        if b[l[j]] != 0:
            a1[b[l[j]] - 1] = 0
    for j in range(3):
        if b[l[j]] == 0:
            k = a1.index(1)
            b[l[j]] = k + 1
            a1[k] = 0
