def decrease(arg):
	return int(arg) - 1

txt = map(int, raw_input().split())

n = txt[0]
m = txt[1]

a = [None for i in range(m)]
s = [0 for i in range(n)]

for i in range(m):
	a[i] = map(decrease, raw_input().split())

for i in range(m):
	p = [1, 2, 3]
	for j in range(3):
		if s[a[i][j]] != 0:
			p[s[a[i][j]] - 1] = p[j]
			p[j] = 0

	for j in range(3):
		s[a[i][j]] += p[j]
