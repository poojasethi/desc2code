def print_group_nums(num_dancers, dances):
    dancer_groups = [-1] * num_dancers
    first_dance = map(int, dances[0].split())
    for i in range(3):
        dancer_groups[first_dance[i]-1] = i
    for i in range(1, len(dances)):
        dance = map(int, dances[i].split())
        colors_done = [False] * 3
        for dancer in dance:
            if dancer_groups[dancer-1] != -1:
                colors_done[dancer_groups[dancer-1]] = True
        for dancer in dance:
            if dancer_groups[dancer-1] == -1:
                for i in range(len(colors_done)):
                    if colors_done[i] == False:
                        colors_done[i] = True
                        dancer_groups[dancer-1] = i
                        break
    for group in dancer_groups:
        print group + 1,

nums = map(int, raw_input('').split())
num_dancers = nums[0]
num_dances = nums[1]
dances = [''] * num_dances
for i in xrange(num_dances):
    dances[i] = raw_input('')

print_group_nums(num_dancers, dances)
