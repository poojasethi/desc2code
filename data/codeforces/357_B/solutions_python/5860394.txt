#!/usr/bin/python
import sys 

data=sys.stdin.readlines()
data=map(lambda x:x.strip(), data)

n,m=map(int, data[0].split(' '))
dance=[]
for i in xrange(m):
  dance+=[map(int, data[i+1].split(' '))]

dancers = [None for i in xrange(n)]

for d in dance:
  cols=set([1,2,3])
  a,b,c=d
  
  used_cols=set([dancers[i-1] for i in d if dancers[i-1] is not None])
  to_set=[i for i in d if dancers[i-1] is None]
  
  available_cols = cols.difference(used_cols)
  for c,d in zip(available_cols,to_set):
    dancers[d-1]=c

print ' '.join(map(str, dancers))
