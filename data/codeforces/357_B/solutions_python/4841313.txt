n, m = map(int, raw_input().split())

colors = [-1]
colors = colors * (n + 1)

for _ in range(m) :
    nodes = map(int, raw_input().split())
    preNode = -1
    for node in nodes :
        if colors[node] != -1 :
            preNode = node
            break
    color = colors[preNode]
    for node in nodes :
        if node != preNode :
            color = (color) % 3 + 1
            colors[node] = color
        else :
            pass
resultStr = ""    
for color in colors[1:] :
    resultStr = resultStr + "%s " % color
print resultStr
