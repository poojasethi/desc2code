n, m = map(int, raw_input().split())
C = [0]*n
for i in range(m):
  a,b,c = map(lambda x:int(x)-1, raw_input().split())
  if C[a] == 0 and C[b] == 0 and C[c] == 0:
    C[a] = 1
    C[b] = 2
    C[c] = 3
  elif C[a] != 0:
    C[b] = (C[a]%3)+1
    C[c] = (C[a]+1)%3+1
  elif C[b] != 0:
    C[a] = (C[b]%3)+1
    C[c] = (C[b]+1)%3+1
  else:
    C[a] = (C[c]%3)+1
    C[b] = (C[c]+1)%3+1

print ' '.join(map(str, C))
