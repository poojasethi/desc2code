I=lambda:map(int, raw_input().split())
n, m = I()
a = [0] * (n+1)
for _ in xrange(m):
    x, y, z = I()
    p = 0
    if a[x]: p = x
    elif a[y]: p = y
    elif a[z]: p = z
    if p == 0: 
        a[x] = 1; a[y] = 2; a[z] = 3
    else:
        s = 1
        for q in [x, y, z]:
            if q != p:
                while s == a[p]: s +=1
                a[q] = s
                s += 1
