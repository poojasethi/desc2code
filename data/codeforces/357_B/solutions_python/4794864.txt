colors = {}
def solve(ers, dans):
    for i in xrange(dans):
        used = []
        dancers = map(int, raw_input().split())
        for dancer in dancers:
            if dancer in colors:
                used.append(colors[dancer])
        for dancer in dancers:
            if not dancer in colors:
                for i in [1,2,3]:
                    if not i in used:
                        colors[dancer] = i
                        used.append(i)
                        break

def main():
    ers, dans = map(int, raw_input().split())
    solve(ers, dans)
    ret = []
    for i in xrange(1, ers+1):
        ret.append(colors[i])
    print " ".join(map(str, ret))

if __name__ == "__main__":
    main()
