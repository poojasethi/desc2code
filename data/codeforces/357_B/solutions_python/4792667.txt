#!/usr/bin/env python

n, m = map(int, raw_input().split())
colors = [0] * (n + 1)
for _ in xrange(m):
    c = {1, 2, 3}
    dance = map(int, raw_input().split())
    for d in dance:
        if colors[d] != 0:
            c.remove(colors[d])
            dance.remove(d)
            break
    for d in dance:
        colors[d] = c.pop()

print ' '.join(map(str, colors[1:]))
