row = lambda:map(int,raw_input().split())
n, m = row()
color = [-1] * (n+1)
for i in range(m):
    c = row()
    fp, fvalue = 0, 0
    for j in range(len(c)):
        if color[c[j]] <> -1:
            fp = j
            fvalue = color[c[j]]
            break
    for j in range(len(c)):
        color[c[(fp + j) % len(c)]] = (fvalue + j) % 3
result = [item+1 for item in color]
print ' '.join(map(str,result[1:]))
