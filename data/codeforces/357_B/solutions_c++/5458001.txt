#include<cstdio>
#define N 100000
int col[N+1];
int main()
{
	int n,m;scanf("%d%d",&n,&m);
	for(int i=0;i<m;++i)
	{
		int v[3],q=0;
		for(int j=0;j<3;++j)
		{
			scanf("%d",v+j);
			q += col[v[j]];
		}
		for(int j=0,c=1;j<3;++j)
		{
			if(c==q) ++c;
			if(!col[v[j]])
				col[v[j]] = c++;
		}		
	}
	for(int i=1;i<=n;++i)
		printf("%d ",col[i]);
	puts("");
	return 0;
}
