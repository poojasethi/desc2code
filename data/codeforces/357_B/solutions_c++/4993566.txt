#include <iostream>
#include <set>

using namespace std;

int main()
{
	int n, m;
	int a[100005]={0};
	cin>>n>>m;
	while(m--)
	{
		int x, y, z;
		cin>>x>>y>>z;
		if(!a[x] && !a[y] && !a[z]) a[x]=1, a[y]=2, a[z]=3;
		else if(a[x]==1) a[y]=2, a[z]=3;
		else if(a[y]==1) a[x]=2, a[z]=3;
		else if(a[z]==1) a[x]=2, a[y]=3;
		else if(a[x]) a[y]=1, a[z]=5-a[x];
		else if(a[y]) a[x]=1, a[z]=5-a[y];
		else if(a[z]) a[x]=1, a[y]=5-a[z];
	}
	for(int i=1; i<=n; i++) cout<<a[i]<<" "; cout<<endl;
	return 0;
}
