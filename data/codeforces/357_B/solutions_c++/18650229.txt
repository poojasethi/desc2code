#include<bits/stdc++.h>

using namespace std;
long long int arr[100010];

int main()
{

	long long int n,m,i;
	cin>>n>>m;
	while (m--)
	{
	    int x,y,z;

		cin>>x>>y>>z;
		if (arr[x])
		{
			arr[y]=arr[x]%3+1;
			arr[z]=arr[y]%3+1;
		}
		else if (arr[y])
		{
			arr[z]=arr[y]%3+1;
			arr[x]=arr[z]%3+1;
		}
		else if (arr[z])
		{
			arr[x]=arr[z]%3+1;
			arr[y]=arr[x]%3+1;
		}
		else
		{
			arr[x]=1;
			arr[y]=2;
			arr[z]=3;
		}
	}
	for(i=1;i<=n;i++)
		cout<<arr[i]<<' ';
	return 0;
}
