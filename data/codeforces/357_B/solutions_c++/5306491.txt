#include<iostream>
using namespace std;

int f[100005]={0};
int main()
{
   int n,m,a,b,c;
   cin>>n>>m;
   for(int i=0;i<m;i++){
    cin>>a>>b>>c;
   if(f[a]!=0)
   {
       f[b]=f[a]%3+1;
       f[c]=f[b]%3+1;
   }
   else if(f[b]!=0)
   {
       f[a]=f[b]%3+1;
       f[c]=f[a]%3+1;
   }
   else if(f[c]!=0)
   {
       f[b]=f[c]%3+1;
       f[a]=f[b]%3+1;
   }
   else
   {
       f[a]=1;f[b]=2;f[c]=3;
   }

   }

   for(int i=1;i<=n;i++)
    cout<<f[i]<<" ";
   cout<<endl;
   return 0;
}
