#include <cstdio>

const int N = 100005;
int c[N];
int main(void) {
	int n, m;
	scanf("%d%d", &n, &m);
	for(int i=0;i<m;i++) {
		int u[4]={};
		int p, q, r, x=0, y=0;
		scanf("%d%d%d", &p, &q, &r);
		u[c[p]]++;
		u[c[q]]++;
		u[c[r]]++;
		if(u[1]) x=2,y=3;
		if(u[2]) x=1,y=3;
		if(u[3]) x=1,y=2;
		if(c[p]) c[q]=x,c[r]=y;
		else if(c[q]) c[p]=x, c[r]=y;
		else if(c[r]) c[p]=x, c[q]=y;
		else c[p]=1, c[q]=2, c[r]=3;
	}
	for(int i=1;i<=n;i++) printf("%d ", c[i]);
}
