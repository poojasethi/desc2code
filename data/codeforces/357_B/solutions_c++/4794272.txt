#include <iostream>
using namespace std;
int main()
{
	long int n, m, i;
	cin >> n >> m;
	int dan[n+1], tan[3], col=0, j, k;
	for(i=0; i<n+1; i++)
		dan[i]=-1;
	for(i=0; i<m; i++)
	{
		k=1;
		for(j=0; j<3; j++)
		{
			cin >> tan[j];
			if(dan[tan[j]]!=-1) col=dan[tan[j]];
		}
		for(j=0; j<3; j++)
			if(dan[tan[j]]!=col) 
			{
				dan[tan[j]]=(col+k)%3;
				k++;
			}
	}
	for(i=1; i<n+1; i++) cout << dan[i]+1 << " ";		
}
