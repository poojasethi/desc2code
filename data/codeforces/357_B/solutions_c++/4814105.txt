#include<iostream>
#include<string.h>
using namespace std;
int n,m;
int t[100001];
int main()
{
	int i,x,y,z;
	int n,m;
	int t[100001];
	memset(t,0,sizeof(t));
	cin>>n>>m;
	for(i=m;i>=1;i--)
	{
		cin>>x>>y>>z;
		if(t[x]!=0){t[y]=t[x]%3+1;t[z]=t[y]%3+1;}
		if(t[y]!=0){t[z]=t[y]%3+1;t[x]=t[z]%3+1;}
		if(t[z]!=0){t[x]=t[z]%3+1;t[y]=t[x]%3+1;}
		else{t[x]=1;t[y]=2;t[z]=3;}
	}
	for(i=1;i<=n;i++)
	{
		cout<<t[i]<<" ";
	}
		return 0;
}
