# include <iostream>
using namespace std;

int f[100005];
int main() {
	int n, m;
	cin >> n >> m;
	
	for (int i = 0; i < m; i++) {
		int a, b, c;
		cin >> a >> b >> c;
		if (f[a]) f[b] = f[a]+1, f[c] = f[a]--;
		else if (f[b]) f[a] = f[b]+1, f[c] = f[b]--;
		else if (f[c]) f[a] = f[c]+1, f[b] = f[c]--;
		else f[a] = 0, f[b] = 1, f[c] = 2;
		(f[a] %= 3)++; (f[b] %= 3)++; (f[c] %= 3)++;
	}
	
	for (int i = 1; i <= n; i++) {
		if (i != 1) cout << ' ';
		cout << f[i];
	}
	cout << endl;
	return 0;
}
