#include <iostream>
#include <algorithm>
using namespace std;
int color[100001];
int main(){
    int d1,d2,d3,n,round;
    cin >>n>>round;
    for(int i=0;i<round;i++){
        cin >>d1>>d2>>d3;
        if(color[d2]) swap(d1,d2);
        else if(color[d3]) swap(d1,d3);
        if(!color[d1]) color[d1]++;
        color[d1]--;
        color[d2]=((color[d1]+1)%3)+1;
        color[d3]=((color[d1]+2)%3)+1;
        color[d1]++;
    }
    for(int i=1;i<=n;i++)
        cout <<color[i]<<" ";
   return 0;
}
