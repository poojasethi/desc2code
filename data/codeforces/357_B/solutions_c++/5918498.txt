#include <iostream>

using namespace std;

int N,M,colour[100010];

int assign(int n[])
{
	for (int i=1;i<=3;i++)
	{
		bool flag=true;
		for (int j=0;j<3;j++)
			if (i==colour[n[j]]) flag=false;
		if (flag) return i;
	}
}

int main()
{
	cin >> N >> M;
	for (int i=0;i<M;i++)
	{
		int n[3];
		for (int j=0;j<3;j++) cin >> n[j];
		for (int j=0;j<3;j++)
			if (!colour[n[j]]) colour[n[j]]=assign(n);
	}
	for (int i=1;i<=N;i++) cout << colour[i] << " ";
	return 0;
}
