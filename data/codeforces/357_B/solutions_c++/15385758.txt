#include <cstdio>
int N,M,f[100005];
int main()
{
	scanf("%d%d",&N,&M);
	for (int i=1,a,b,c; i<=M; i++)
	{
		scanf("%d%d%d",&a,&b,&c);
		if (f[a]) f[b]=f[a]+1,f[c]=f[a]--;
		else if (f[b]) f[a]=f[b]+1,f[c]=f[b]--;
		else if (f[c]) f[a]=f[c]+1,f[b]=f[c]--;
		else f[a]=0,f[b]=1,f[c]=2;
		(f[a]%=3)++,(f[b]%=3)++,(f[c]%=3)++;
	}
	for (int i=1; i<=N; i++) printf("%d%c",f[i],i==N?'\n':' ');
}
