#include <iostream>
using namespace std;
int a[5],num[100005];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=m;i++)
	{
		for(int j=1;j<=3;j++)
			cin>>a[j];
		for(int j=1;j<=3;j++)
			if(num[a[j]])
			{
				num[a[j%3+1]]=(num[a[j]])%3+1;
				num[a[6-j-(j%3+1)]]=6-num[a[j]]-num[a[j%3+1]];
				break;
			}
		if(!num[a[1]] && !num[a[2]] && !num[a[3]])
			for(int j=1;j<=3;j++)
				num[a[j]]=j;	
	}
	for(int i=1;i<=n;i++)
		cout<<num[i]<<" ";
	cout<<endl;
	return 0;
}
