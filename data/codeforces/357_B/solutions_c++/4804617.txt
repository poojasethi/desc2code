#include <stdio.h>

#define MAXN 100100
int c[MAXN];

int main()
{
  int n,m;
  scanf("%d%d",&n,&m);
  while(m--)
    {
      int a[3];
      scanf("%d%d%d",a,a+1,a+2);
      int used[4]={};
      used[c[a[1]]]=used[c[a[2]]]=used[c[a[0]]]=1;
      int i,j;
      for(i=0;i<3;++i)
    for(j=1;j<4;++j)
      if(used[j]==0 && c[a[i]]==0)
        {
          c[a[i]]=j;
          used[j]=1;
        }
    }
  int i;
  for(i=1;i<=n;++i)
    printf("%d ",c[i]);
  printf("\n");
  return 0;
}
