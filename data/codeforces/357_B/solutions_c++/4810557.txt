#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(0);cin.tie(0);
using namespace std;

int d[100010],n,m,a,b,c;

int main() { _
  cin >> n >> m;
  for (int i = 0; i < m; ++i) {
    cin >> a >> b >> c;
    int f[4] = {};
    f[d[a]]++; f[d[b]]++; f[d[c]]++;
    for (int j = 1; j < 4; ++j) {
      if (f[j]) continue;
           if (!d[a]) d[a] = j;
      else if (!d[b]) d[b] = j;
      else if (!d[c]) d[c] = j;
    }
  }
  for (int i = 1; i <= n; ++i) {
    cout << d[i] << " ";
  }
  return 0;
}
