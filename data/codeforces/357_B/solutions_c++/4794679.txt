#include<iostream>
using namespace std;

const int Maxn=100010;
int n,m,color[Maxn],cnt[Maxn];

int main() {
	cin >> n >> m;
	for (int i=1;i<=m;i++) {
		int x[4];
		cin >> x[1] >> x[2] >> x[3];
		cnt[x[1]]++; cnt[x[2]]++; cnt[x[3]]++;
		if (cnt[x[1]]>=2) swap(x[1],x[color[x[1]]]);
		else if (cnt[x[2]]>=2) swap(x[2],x[color[x[2]]]);
		else if (cnt[x[3]]>=2) swap(x[3],x[color[x[3]]]);
		color[x[1]]=1;
		color[x[2]]=2;
		color[x[3]]=3;
	}
	for (int i=1;i<=n;i++) cout << color[i] << ' ';
	cout <<endl;
	return 0;
}
