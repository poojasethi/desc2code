#include<iostream>
#include<algorithm>

using namespace std;

bool ar[300002];
int col[300002];


int main()
{
	int n,m;
	cin>>n>>m;
	while(m--)
	{
		int a,b,c;
		cin>>a>>b>>c;
		if(ar[b])
		swap(a,b);
		if(ar[c])
		swap(a,c);
		col[b]=(col[a]+1)%3;
		col[c]=(col[a]+2)%3;
		ar[a]=ar[b]=ar[c]=1;
	}
	for(int i=1;i<=n;++i)
	cout<<col[i]+1<<" ";
	return 0;
}
