#include <iostream>
using namespace std;
int n, m, k[100001], x, y, z, i;
int main()
{
	cin>>n>>m;
	cin>>x>>y>>z;
	k[x]=1; k[y]=2; k[z]=3;
	for (i=1; i<m; i++) {
		cin>>x>>y>>z;
		if (k[x]) {
			k[y]=k[x]%3+1;
			k[z]=k[y]%3+1;
		}
		else if (k[y]) {
			k[z]=k[y]%3+1;
			k[x]=k[z]%3+1;
		}
		else if (k[z]) {
			k[x]=k[z]%3+1;
			k[y]=k[x]%3+1;
		}
		else {
			k[x]=1; k[y]=2; k[z]=3;
		}
	}
	for (i=1; i<=n; i++) cout<<k[i]<<" ";
	cout<<endl;
	return 0;
}
