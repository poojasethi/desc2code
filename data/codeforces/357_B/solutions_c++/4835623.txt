#include<iostream>
#include<vector>
#include<queue>
using namespace std;

int N,M;
vector<int> sol;

int main() {
	cin>>N>>M;
	sol.resize(N+1,-1);
	queue<int> cols;
	int d[3];
	for(int m=0;m<M;m++) {
		cin>>d[0]>>d[1]>>d[2];
		for(int i=0;i<3;i++) 
			if(sol[d[0]]!=i+1 && sol[d[1]]!=i+1 && sol[d[2]]!=i+1)	
				cols.push(i+1);
		for(int i=0;i<3;i++)
			if(sol[d[i]] == -1) { sol[d[i]] = cols.front(); cols.pop();}
	}
	for(int i=0;i<N;i++)
		cout<<sol[i+1]<<" ";
	cout<<endl;

	return 0;
}
