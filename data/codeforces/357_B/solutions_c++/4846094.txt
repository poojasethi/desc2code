#include <iostream>

using namespace std;

int n,m;
int color[1000000];

int main()
{
  ios::sync_with_stdio(false);
  cin>>n>>m;
  for (int i=0;i<m;i++) {
    int a,b,c;
    cin>>a>>b>>c;
    if (color[b]!=0) swap(a,b);
    else if (color[c]!=0) swap(a,c);
    if (color[a]==0) color[a]=1;
    color[a]--;
    color[b]=(color[a]+1)%3;
    color[c]=(color[a]+2)%3;
    color[a]++;
    color[b]++;
    color[c]++;
  }
  for (int i=1;i<=n;i++) {
    if (i>1) cout<<" ";
    cout<<color[i];
  }
  cout<<endl;
}
