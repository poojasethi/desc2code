#include <cstdio>
int N,M,d[100005],e[5],y[5];
int main()
{
	scanf("%d %d",&N,&M);
	scanf("%d %d %d",&e[0],&e[1],&e[2]);
	d[e[0]]=1,d[e[1]]=2,d[e[2]]=3;
	for (int i = 0; i < M-1 ; i++)
	{
		scanf("%d %d %d",&y[0],&y[1],&y[2]);
		{
			int kk=1,mk=-3;
			for(int l=0;l<3;l++)
				if(d[y[l]])
					mk=d[y[l]];
			for(int l=0;l<3;l++)
			{
				if(kk == mk)
					kk++;
				if(!d[y[l]])
					d[y[l]]=kk++;
			}
		}
		for(int j=0;j<3;j++)
			e[j]=y[j];
	}
	for (int i = 1; i <= N; i++)
		printf("%d ",d[i]);
	





	return 0;
}
