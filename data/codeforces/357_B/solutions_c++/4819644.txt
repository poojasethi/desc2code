#include <iostream>

using namespace std;

int color[100005];

int main()
{
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < m; i++)
	{
		int a[3];
		for (int j = 0; j < 3; j++)
			cin >> a[j];
		bool cant[4] = {};
		for (int j = 0; j < 3; j++)
			cant[color[a[j]]] = true;
		for (int j = 1; j <= 3; j++)
			if (!cant[j])
				for (int k = 0; k < 3; k++)
					if (color[a[k]] == 0)
					{
						color[a[k]] = j;
						break;
					}
	}
	for (int i = 1; i <= n; i++)
		cout << color[i] << " ";
	cout << endl;
}
