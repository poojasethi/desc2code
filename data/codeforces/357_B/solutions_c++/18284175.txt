#include<bits/stdc++.h>
using namespace std;

int a[100005];

int main(){

    int n, m;
    cin>>n>>m;
    while(m--)
    {
        int x, y, z;
        cin>>x>>y>>z;
        if(a[y])
            swap(x,y);
        if(a[z])
            swap(x,z);
        if(!a[x] || a[x]==1)
            a[x]=1, a[y]=2, a[z]=3;
        else if(a[x]==2)
            a[z]=3, a[y]=1;
        else
            a[y]=1, a[z]=2;
    }

    for(int i=1; i<=n; i++)
        cout<<a[i]<<" ";

    return 0;
}
