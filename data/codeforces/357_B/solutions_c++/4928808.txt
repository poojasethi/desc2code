#include<stdio.h>
#include<string.h>
const int maxn=100050,maxc=5;
int n,m,d[maxn][maxc],co[maxn]={0};
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j,k,vis[maxc];
	scanf("%d%d",&n,&m);
	for(i=0;i<m;i++)
	{
		memset(vis,0,sizeof(vis));
		for(j=0;j<3;j++)
			scanf("%d",&d[i][j]),vis[co[d[i][j]]]=1;
		for(j=0;j<3;j++)
			if(!co[d[i][j]])
			{
				for(k=1;k<=3;k++)
					if(!vis[k]) {co[d[i][j]]=k;vis[k]=1;break;}
			}
	}
	for(i=1;i<=n;i++)
		printf("%d ",co[i]);
	return 0;
}
