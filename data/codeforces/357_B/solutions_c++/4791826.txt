#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	int n,m;
	int a[100000];
	int x,y,z,i;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
	{
		a[i]=-1;
	}
	for(i=0;i<m;i++)
	{
		scanf("%d%d%d",&x,&y,&z);
		if(a[x]!=-1)
		{
			a[y]=a[x]%3+1;
			a[z]=a[y]%3+1;
		}
		else if(a[y]!=-1)
		{
			a[z]=a[y]%3+1;
			a[x]=a[z]%3+1;
		}
		else if(a[z]!=-1)
		{
			a[x]=a[z]%3+1;
			a[y]=a[x]%3+1;
		}
		else
		{
			a[x]=1;
			a[y]=2;
			a[z]=3;
		}
	}
	for(i=1;i<=n;i++)
	{
		printf("%d ",a[i]);
	}
	printf("\n");
	return 0;
}
