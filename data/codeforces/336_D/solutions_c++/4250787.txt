#include <cstdio>
#define LL long long
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

const LL MO=1000000007ll;
const int MAXN=200020;
int n,m,g;
LL ans;
LL quc[MAXN];
LL cuq[MAXN];

inline LL Pow(LL a,LL b) {
	LL ret=1;
	for (;b;b>>=1) {
		if (b&1) ret=ret*a%MO;
		a=a*a%MO;
	}
	return ret;
}

inline LL C(int n,int k) {
	return quc[n]*cuq[k]%MO*cuq[n-k]%MO;
}

int main() {
	scanf("%d%d%d",&n,&m,&g);
	if (!m) {
		if (n%2!=g) puts("1"); else puts("0");
		return 0;
	}
	quc[0]=1; cuq[0]=1;
	REP(i,n+m) {
		quc[i]=quc[i-1]*(LL)i%MO;
		cuq[i]=Pow(quc[i],MO-2);
	}
	REP(i,n+1) {
		if ((i==n+1)&&(m==1)) {
			if (i%2==g) ++ans;
			continue;
		}
		LL tmp=C(n+m-i,m-1);
		if (i%2!=g) ans=(ans+tmp)%MO;
	}
	printf("%I64d\n",ans);
	return 0;
}
