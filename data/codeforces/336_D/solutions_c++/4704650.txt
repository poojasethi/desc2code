#include<iostream>
#define ll long long
using namespace std;
long long fact[200006],invfact[200006],sum;
main()
{
    long long i,j,k,l,n,m,g,x,y,w,b,mod=1000000007;
    cin>>m>>n>>g;
    fact[0]=1;
    invfact[0]=1;
    fact[1]=1;
    invfact[1]=1;
    for(i=2;i<200004;i++) fact[i]=(fact[i-1]*i)%mod;
    for(i=2;i<200004;i++)
    {
     ll x=1,y=i,w=mod,b=mod-2;
     while(b>0)
     {
        if(b&1) x=(x*y)%(w);
        y=(y*y)%(w);
        b>>=1;
     }
     invfact[i]=(invfact[i-1]*x)%mod;
    }
    x=n+m;
    x-=1;
    while(x>=n-1)
    {
        sum=(sum+(((fact[x]*invfact[n-1])%mod)*invfact[x-n+1])%mod)%mod;
        x-=2;
    }
    if(n<=1){if(m%2==0) sum-=1; else sum+=1;} sum=max((ll)0,sum);
    sum+=2*mod; sum%=mod;
    if(g){sum=(((fact[n+m]*invfact[n])%mod)*invfact[m])%mod-sum; sum=sum+5*mod; sum%=mod; cout<<sum;}
    else cout<<sum;
}
