#include<cstdio>
#include<cstring>
#define LL long long
const LL p=1000000007;
LL f[1000001];
int n,m,g;
LL power(LL a,LL b)
{
   LL c;
   if (b==0) return 1;
   c=power(a,b/2);
   if (b%2==0) return (c*c)%p;
   else return (((c*c)%p)*a)%p;
}
LL C(int x,int y)
{
   return (f[x+y]*power((f[x]*f[y])%p,p-2))%p;
}
int main()
{
    LL ans;
    int X,Y,i;
    scanf("%d%d%d",&n,&m,&g);
    if (n==0) printf("%d\n",(m>1)^g);
    else if (m==0) printf("%d\n",(n%2)^g);
    else if (m==1)
    {
         X=n/2+(n%2==0);
         if (g) printf("%d\n",X);
         else printf("%d\n",n+1-X);
    }
    else
    {
        ans=0; f[0]=1; X=n-g; Y=m-1;
        for (i=1;i<=n+m;i++) f[i]=(f[i-1]*i)%p;
        while (X>=0)
        {
              ans=(ans+C(X,Y))%p; X-=2;
        }
        printf("%I64d\n",ans);
    }
    return 0;
}
