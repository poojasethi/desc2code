#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define mo 1000000007
#define int64 long long
#define N 200010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,x,y,an,g;
int64 a[N],b[N];
inline int64 pow(int64 x,int64 y) {
    int64 an=1;
    for (;y;y>>=1) {
        if (y&1) an=an*x%mo;
        x=x*x%mo;
    } return an;
}
inline int64 C(int n,int m) {
    if (n<0||m<0||n+m==0) return 0;
    n=n+m;
    return a[n]*b[m]%mo*b[n-m]%mo;
}
inline void work(int n,int m,int g) {
    for (;;) {
        if (n<0||m<0) return;
        if (n+m==1) {
            if (g&&m||!g&&n) an=(an+1)%mo;
            return;
        }
        if (g==0) an=(an+C(n,m-1))%mo,n--,g=1;
        else n--,g=0;
    }
}
int main() {
    scanf("%d%d%d",&n,&m,&g);
    a[0]=1; For(i,1,n+m) a[i]=a[i-1]*i%mo;
    For(i,0,n+m) b[i]=pow(a[i],mo-2);
    //printf("%I64d\n",C(2,2));
    work(n,m,g);
    printf("%d\n",an);
    return 0;
}
