#include <cstdio>
using namespace std;

const int SIZE = 400009;
int ftable[SIZE];
const int MOD = 1000000007;

long long int mpow(long long int a, int b)
{
    int ret = 1;
    while(b)
    {
    if(b & 1)
        ret = ret * a % MOD;
    b >>= 1;
    a = a * a % MOD;    
    }
    return ret;
}

int g(int, int);
int f(int n, int m)
{
    if(n + m == 1)
    return n;

    int ret = 0;
    if(m)
    ret = ftable[n + m - 1] * mpow(ftable[n], MOD - 2) % MOD * mpow(ftable[m - 1], MOD - 2) % MOD;
    if(n)
    ret = (ret + g(n - 1, m)) % MOD;
    return ret;
}

int g(int n, int m)
{
    if(n + m == 1)
    return m;
    else 
    return n? f(n - 1, m): 0;
}



int main()
{
    ftable[0] = ftable[1] = 1;
    for(long long int i = 2; i < SIZE; i++)
    ftable[i] = ftable[i - 1] * i % MOD;

    int N, M, G;
    scanf("%d%d%d", &N, &M, &G);
    printf("%d\n", G? g(N, M): f(N, M));

    return 0;
}
