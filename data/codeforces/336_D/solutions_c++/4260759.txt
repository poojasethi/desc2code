#include <iostream>
using namespace std;

typedef long long ll;
const int mod = 1000000007;

ll n,m,g;
ll fact[200005];
ll ans;

ll potenza(ll b, ll e){
	if(e==1) return b;
	ll pot = potenza(b, e/2);
	if(e%2 == 0) return (pot*pot)%mod;
	else return (b*((pot*pot)%mod))%mod;
}
ll inverso(ll a){
	return potenza(a, mod-2);
}
ll choose(ll a, ll k){
	return (fact[a]*inverso((fact[k]*fact[a-k])%mod)) % mod;
}

int main(){
	cin >> n >> m >> g;
	if(m==0) {if(n%2==1-g) cout << 1; else cout << 0; return 0;}
	if(n==0) {if(m>1) cout << 1-g; else cout << g; return 0;}
	
	fact[0] = 1;
	for(int i=1; i<=n+m; i++) fact[i]=(fact[i-1]*i)%mod;
	
	ans = 0;
	for(int k=m-1; k<n+m; k++){
		if(k==0) {if((n+m-1)%2 != g) ans = (ans + 1) % mod;}
		else if((n+m-k-1)%2 == g) ans = (ans + choose(k, m-1)) % mod;
	}
	
	cout << ans;
	return 0;
}
