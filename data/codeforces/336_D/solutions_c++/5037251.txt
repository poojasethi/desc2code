#include<cstdio>
#include<iostream>
#include<cstring>
#define ll long long
#define mod 1000000007
using namespace std;
ll fac[200100];
ll pow(ll x,ll y)
{
    ll ret = 1;
    while(y>0)
    {
        if(y&1)ret = x*ret%mod;
        x = x*x%mod;
        y>>=1;
    }
    return ret;
}
ll c(ll x,ll y)
{
    return fac[x]*pow((fac[y]*fac[x-y]%mod),mod-2)%mod;
}
int main()
{
    int n,m,g,i,j;
    while(scanf("%d%d%d",&n,&m,&g)!=EOF)
    {
        if(m == 0)
        {
            if(n%2 == 1-g)printf("1\n");
            else printf("0\n");
        }
        else
        {
            ll ans = 0;
            fac[0] = 1;
            for(i = 1;i<=n+m;i++)
                fac[i] = fac[i-1]*i%mod;
            for(i = 1;i<=n+1;i++)
            {
                if(i == n+1&&m == 1)ans+=(n%2) == 1-g;
                else if(i%2 == 1-g)ans=(ans+c(n+m-i,m-1))%mod;
            }
            printf("%I64d\n",ans);
        }
    }
    return 0;
}
