#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<vector>
using namespace std;
const long long mod=1000000007;
long long mi(long long x,long long y)
{
    if(y==-1)return mi(x,1000000005);
    if(y==0)return 1;
    if(y==1)return x%mod;
    if(y==2)return x*x%mod;
    if(y%2==0)return mi(mi(x,y>>1),2);
    return mi(mi(x,y>>1),2)*x%mod;
}int n,m,g;
long long fr[300000],ans;
bool t[300000];
long long C(long long x,long long y)
{
    if(y==0||y==x)return 1;
    return fr[x]*mi(fr[y]*fr[x-y]%mod,mod-2);
}
int main()
{
    cin>>n>>m>>g;fr[0]=1;
    for(int i=1;i<=n+m;i++)
    t[i]=!t[i-1];
    if(m==0)return cout<<(t[n-1]==g)<<endl,0;
    //else if(n==0)return cout<<(t[0]==g)<<endl,0;
    for(int i=1;i<=n+m;i++)fr[i]=fr[i-1]*i%mod;
    for(int i=1;i<=min(n+1,n+m-1);i++)
    if(t[i-1]==g)ans+=C(n+m-i,m-1),ans%=mod;
    if(n+1>=n+m&&t[n+1]==g)ans=(ans+1)%mod;
    cout<<ans<<endl;
}
