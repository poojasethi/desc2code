using namespace std;
typedef long long ll;

const int inf=1<<29;

char str[200010];

#define mod 1000000007

int power(ll n,int k){
	ll ans=1;
	while(k){
		if(k%2){
			ans*=n;
			ans%=mod;
		}
		n*=n;
		n%=mod;
		k/=2;
	}
	return ans%mod;
}
ll h[200010];
int C(int n,int m)
{
	ll ans=h[n];
	ll sum=h[m]*h[n-m]%mod;
	return ans*power(sum,mod-2)%mod;
}
int main() {
	int n,m,g;

	scanf("%d%d%d",&n,&m,&g);
	h[0]=1;
	for(int i=1;i<=n+m;i++)
		h[i]=h[i-1]*i%mod;
	int k=g,N=n,M=m;
	ll ans=0;
	for(int i=0;i<n+m;i++){
		if(i+1==n+m){
			if(k==0){
				if(N==1) ans++,ans%=mod;;
			}else{
				if(M==1) ans++,ans%=mod;;
			}
			continue;
		}
		if(k==0){
			if(M){
				ans+=C(N+M-1,N);
				ans%=mod;
			}
			N--;k=1;
			if(N<0) break;
		}else{
			N--;
			k=0;
			if(N<0) break;
		}
	}
	printf("%I64d\n",ans);
}
