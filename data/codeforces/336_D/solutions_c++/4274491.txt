#include<iostream>
#define M 1000000007
#define llu long long int 
using namespace std;
llu f[200002],inf[200002],n,m,g,i;
llu pow(llu a,llu b)
{
    llu ans=1,y=a;
    while(b!=0)
    {
        if(b%2)ans=(ans*y)%M;
        b=b/2;
        y=(y*y)%M;
    }
    return ans;
}
int main()
{
    f[0]=1;inf[0]=1;
    for(i=1;i<=200000;i++)
    {
        f[i]=(f[i-1]*i)%M;
        inf[i]=pow(f[i],M-2);
    }
    //for(i=0;i<10;i++)cout<<f[i]<<" "<<inf[i]<<endl;
    cin>>n>>m>>g;
    llu ans=0;
    for(i=0;i<n;i++)
    {
        llu a=0;
        if((m-1)>=0)a=(((f[n-i+m-1]*inf[m-1])%M)*inf[n-i])%M;
        //cout<<f[n-i]<<inf[m-1]<<inf[n-i-m+1]<<endl;
        if((i%2)==g)ans=(ans+a)%M;
       // cout<<a<<ans<<endl;
    }
    int g2;
    if(m>1)
    {
         if((n+1)%2)g2=0;
         else g2=1;   
    }
    else
    {
       if(n%2) g2=0;
       else g2=1;
    }
    if(g2==g) ans=(ans+1)%M;
    cout<<ans<<endl;
}
