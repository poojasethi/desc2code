#include<iostream>
#include<stdio.h>
const int mod=1e9+7;
const int N = 100050*2;
using namespace std;
typedef long long LL;
int n,m,g;
LL fac[N],ans;
LL exp_mod(LL a,LL b)
{
    LL ret=1,base=a;
    while(b)
    {
        if(b&1) ret=ret*base%mod;
        base=base*base%mod;
        b>>=1;
    }
    return ret;
}
LL C(int n,int m)
{
    return fac[n]*exp_mod(fac[n-m]*fac[m]%mod,mod-2)%mod;
}
int main()
{
//    freopen("1.in","r",stdin);
    fac[0]=1;
    for(int i=1;i<N;++i) fac[i]=fac[i-1]*i%mod;
    while(scanf("%d%d%d",&n,&m,&g)!=-1)
    {
        ans=0;
        for(int i=m;i<=n+m;++i)
        {
            int tmp=(n+m-i)&1;
            if(m==1&&i==1) tmp^=1;
            if(g!=tmp) continue;
            ans=ans+C(i-1,m-1);
        }
        ans%=mod;
        if(m==0&&g!=(n&1)) ans=1;
        printf("%d\n",(int)ans);
    }
    return 0;
}
