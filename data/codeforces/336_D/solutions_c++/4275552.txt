#include <stdio.h>
#include <string.h>
#include <math.h>
int cnt[100001];
long long fac[200001],res[100001];
#define MOD 1000000007
long long power(long long a,int b){
	if(b==1)
		return a;
	long long s=power(a,b>>1);
	if(b%2==0)
		return (s*s)%MOD;
	return (((s*s)%MOD)*a)%MOD;
}
int main(){
	int n,m,g,i;
	long long print=0;
	scanf("%d %d %d",&n,&m,&g);
	if(n%2==0 && m==0)
		print=1;
	if(m==1 && n%2==0)
		print=1;
	fac[1]=1;
	for(i=2;i<=200000;i++)
		fac[i]=(fac[i-1]*i)%MOD;
	res[0]=1;
	for(i=1;i<=100000;i++)
		res[i]=power(fac[i],MOD-2);
	for(i=1;i<=n;i+=2){
		if(n+m-1-i<0) break;
		print+=((fac[n+m-1-i]*res[m-1])%MOD)*res[n-i];
		print%=MOD;
	}
	if(g==1)
		printf("%I64d\n",print);
	else{
		print=((fac[n+m]*res[n])%MOD)*res[m]-print;
		print+=MOD;
		print%=MOD;
		printf("%I64d\n",print);
	}
	return 0;
}
