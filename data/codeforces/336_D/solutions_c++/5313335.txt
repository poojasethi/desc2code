#include <cstdio>
#define P 1000000007
#define maxn 200005
#define ll long long
using namespace std;

int N,M,G,a[maxn],f[maxn];

ll Pow(ll a,int b)
{
    ll c=1;
    for (;b; b>>=1,a=a*a%P) if (b&1) c=c*a%P;
    return c;
}

int C(int n,int m) {return n<m?0:Pow(a[m],P-2)*Pow(a[n-m],P-2)%P*a[n]%P;}

int main()
{
    scanf("%d%d%d",&N,&M,&G),a[0]=1,f[0]=M>1,f[1]=M<2?M+1:M;
    for (int i=1; i<=N+M; i++) a[i]=(ll)a[i-1]*i%P;
    for (int i=2; i<=N; i++) f[i]=(f[i-2]+C(i+M-1,i))%P;
    printf("%d\n",!G?f[N]:(C(N+M,N)-f[N]+P)%P);
    return 0;
}
