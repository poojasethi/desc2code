#include <stdio.h>
#include <iostream>
using namespace std;
#define ll long long
const int MOD=1000000007;
bool g;
int n,m,v,f[250000];
ll d,x,y,t;
void gcd(ll a,ll b,ll &d,ll &x,ll &y){
	if (!b)
		d=a,x=1,y=0;
	else{
		gcd(b,a%b,d,y,x);
		y-=a/b*x;
	}
}
int main( ){
	cin>>n>>m>>g;
	if (!m){
		if (n%2==!g) cout<<1<<endl; else cout<<0<<endl;
		return 0;
	}
	if (m==1) f[m]=1; else f[m]=0;
	f[m+1]=1-f[m];
	for (int c=1,i=m+2;i<=m+n+1;i++){
		gcd(i-m-1,MOD,d,x,y);
		if (d==1) t=(x%MOD+MOD)%MOD; else t=-1;
		c=(ll)c*(i-1)%MOD*t%MOD;
		f[i]=((c-f[i-1])%MOD+MOD)%MOD;
	}
	if (g) v=f[m+n]; else v=f[m+n+1];
	cout<<v<<endl;
	return 0;
}
