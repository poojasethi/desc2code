#include<cstdio>
#include<cstdlib>
#include<cstring>

using namespace std;

#define inc(a,b) {a+=b;if (a>=mo) a-=mo;}

const int maxn=200010;
const int mo=1000000007;

int n,m,g,fac[maxn],inv[maxn];

int mul(int a,int b)
{
	int ans=1;
	while (b)
	{
		if (b&1) ans=(long long)ans*a%mo;
		a=(long long)a*a%mo;
		b>>=1;
	}
	return ans;
}

int main()
{
	//freopen("in","r",stdin);
	scanf("%d%d%d",&n,&m,&g);
	if (n+m==1)
	{
		if ((g==0 && n) || (g==1 && m)) printf("1\n");
		else printf("0\n");
		return 0;
	}
	if (g==1) n--;
	if (n<0 || m<0)
	{
		printf("0\n");
		return 0;
	}
	fac[0]=1;
	for (int a=1;a<=n+m;a++)
		fac[a]=(long long)fac[a-1]*a%mo;
	for (int a=0;a<=n+m;a++)
		inv[a]=mul(fac[a],mo-2);
	int ans=0;
	while (true)
	{
		if (n+m>1 && m) inc(ans,(long long)fac[n+m-1]*inv[n]%mo*inv[m-1]%mo);
		n--;
		if (n<0) break;
		if (n+m==1 && m==1) ans++;
		if (n+m==0) ans++;
		n--;
		if (n<0) break;
	}
	printf("%d\n",ans);

	return 0;
}
