#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define int64 long long
#define N 220000
using namespace std;
const int mo=1000000007;
int64 fac[N],ans;
int i,n,m,t;
int64 pow(int64 x,int64 y){
	int64 res=1;
	while(y){
		if(y&1)res=res*x%mo;
		x=x*x%mo;
		y/=2;
	}
	return res;
}
int64 C(int n,int m){
	return (fac[n]*pow(fac[m]*fac[n-m]%mo,mo-2))%mo;
}

int main(){
	scanf("%d%d%d",&n,&m,&t);
	fac[0]=1;
	for(i=1;i<=n+m;++i)fac[i]=(fac[i-1]*i)%mo;
	if(m==0){
		if(n%2==1-t)printf("1");
		else printf("0");
		return 0;
	}
	for(i=0;i<=n;++i){
		if(i==n && m==1)ans+=(n%2==1-t);
		else
		if(i%2==t)
			ans=(ans+C(n+m-i-1,m-1))%mo;
	}
	printf("%I64d\n",ans%mo);
}
