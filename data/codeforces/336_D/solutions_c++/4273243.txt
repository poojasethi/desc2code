#include <cstdio>

typedef long long int ll;

const int mod=1e9+7;

ll pot(ll a,int b) { ll wyn=1LL; while(b) { if(b&1) wyn*=a,wyn%=mod; a*=a; a%=mod; b>>=1; } return wyn; }
inline ll odw(ll x) { return pot(x,mod-2); }

int main()
{
	int n,m,g;
	scanf("%d%d%d",&n,&m,&g);
	if(m==0)
	{
		if((n&1)^g) printf("1\n");
		else printf("0\n");
		return 0;
	}
	if(m==1)
	{
		int wyn=0;
		for(int i=0;i<n;i++) if((i&1)==g) wyn++;
		if((n&1)^g) wyn++;
		printf("%d\n",wyn);
		return 0;
	}
	m--;
	ll licznik=1LL;
	ll mianownik=1LL;
	int a=n+m,b=n;
	for(int i=a;i>b;i--) licznik*=i,licznik%=mod;
	for(int i=1;i<=m;i++) mianownik*=i,mianownik%=mod;
	ll wyn=0LL;
	for(int i=0;i<=n;i++)
	{
		if((i&1)==g) wyn+=licznik*odw(mianownik), wyn%=mod;
		licznik*=odw(a--); licznik%=mod;
		licznik*= b--; licznik%=mod;
	}
	printf("%d\n",(int)(wyn%mod));
	return 0;
}
