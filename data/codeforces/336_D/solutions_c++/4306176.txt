#include <iostream>

using namespace std;

typedef long long ll;

const ll mod = 1000000007;
#define MOD(x) ((x)%mod)

ll pow(int b, int e)
{
    if(e==0) return 1;
    ll k = pow(b,e/2);
    k = MOD(k*k);
    if(e%2==1) k = MOD(k*b);
    return k;
}

ll inv(ll x)
{
    return pow(x,mod-2);
}

ll fact[200100];

ll choose(int n, int k)
{
    return MOD(fact[n]*inv(MOD(fact[n-k]*fact[k])));
}
