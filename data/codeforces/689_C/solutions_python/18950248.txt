
import math
fl = math.floor
m = int(raw_input())

lo = int(4.8 * m)
hi = min(8.0 * m, 5.0*(10**15))

cubes = [x ** 3.0 for x in xrange(2, 180000)]

def ntom(n):
    global cubes
    return sum([fl(n/x) for x in cubes if x <= n])


while lo + 1 < hi:
    mid = (hi+lo)/2
    m1 = ntom(mid)
    if m1 < m:
        lo = mid
    else:
        hi = mid
if int(ntom(hi)) == m:
    print int(hi)
else:
    print "-1"
