#include <cstdio>
#include <iostream>
#include <climits>
using namespace std;

long long N;

long long f(long long m) {
	
	long long int res = 0;
	for(long long i=2;i*i*i<=m;i++)
		res += m/(i*i*i);

	return res;
}

int main() {
	
	cin >> N;

	long long s = 0, e = LLONG_MAX;

	while(s!=e) {
	
		long long int m = (s+e)/2;
		
		if(f(m)>=N)
			e = m;
		else
			s = m+1;

	}

	long long res = f(s)==N ? s : -1;
	cout << res << endl;

	return 0;
}
