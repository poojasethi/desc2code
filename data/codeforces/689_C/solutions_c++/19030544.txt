#include<stdio.h>
typedef long long ll;
long long check(long long x)
{
	ll num = 0;
	for(ll i = 2; i * i * i <= x; i++)//i是乘数 
		num += x / (i * i * i);
	return num;
}
int main()
{
	ll m;
	scanf("%I64d",&m);
	ll l=1,r=1e18,res=-1,mid;
	while(l<=r)
	{
		mid=(l+r)/2;
		ll p=check(mid);
		if(p==m)
			res=mid;
		if(p>=m)  //大于等于取较小者 
		r=mid-1;
		else
		l=mid+1; 
	}
	printf("%I64d\n",res);
}
