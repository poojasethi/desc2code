#include <iostream>

using namespace std;

typedef long long int ll;

ll calcula(ll n)
{
  ll sol=0;
  for (ll k=2;k*k*k<=n;k++)
    sol+=n/(k*k*k);
  return sol;
}

int main()
{
  ll m;
  cin>>m;
  ll inf=7;
  ll sup=10000000000000000LL;
  while (inf+1<sup) {
    ll med=(inf+sup)/2;
    if (calcula(med)<m) inf=med;
    else sup=med;
  }
  if (calcula(sup)==m) cout<<sup<<endl;
  else cout<<-1<<endl;
}
