#include<bits/stdc++.h>
using namespace std;
long long m;
long long check(long long n)
{
	long long cnt=0;
	for(long long k=2;k*k*k<=n;k++)
		cnt+=n/(k*k*k);
	return cnt;
}
int main()
{
	cin>>m;
	long long l=1ll,r=5e15;
	long long ans=-1;
	while(l<r)
	{
		long long mid=(l+r)/2ll;
		if(check(mid)<m) l=mid+1;
		else r=mid;
	}
	if(check(l)==m) ans=l;
	cout<<ans<<endl;
	return 0;
}
