#include <cstdio>

long long S(long long x)
{
    long long r=0;
    for (long long i=2; i<=200000; i++) {
        r+=x/(i*i*i);
    }
    return r;
}

int main()
{
    long long n, i, l, r, m;
    scanf("%I64d", &n);
    l=8; r=5000000000000000;
    while (l!=r) {
        m=(l+r)/2;
        if (S(m)<n) l=m+1;
        else r=m;
    }
    if (S(r)==n) printf("%I64d", r);
    else printf ("-1");
    return 0;
}
