#include<iostream>
using namespace std;
long long int func(long long int n)
{
	long long int ret=0;
	for(long long int i=2; i*i*i <= n;i++)
	{
		ret += n/(i*i*i);
	}
	return(ret);
}
int main()
{
	long long int m,l,r,mid,ans=-1;
	l = 1; r=1e16;
	mid = l + (r-l)/2;
	cin >> m;
	while(l < r)
	{
		if(func(mid) < m) l = mid+1;
		else r = mid;
		mid = l + (r-l)/2;
	}
	if(func(l) == m) ans = l;
	cout << ans << endl;
	return(0);
}
