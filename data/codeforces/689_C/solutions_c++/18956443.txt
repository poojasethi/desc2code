#include<bits/stdc++.h>
using namespace std;
#define ll  long long int
ll get(ll m)
{	ll ans=0;
	for(ll k=2;k*k*k<=m;k++)
	ans+=m/(1ll*k*k*k);
	return ans;
}
int main()
{	ll m,n=-1;
	cin>>m;

	ll l=0,r=5e15;
	while(l<r)
	{	ll mid=(l+r)/2;
		if(get(mid)<m)l=mid+1;
		else	r=mid;
	}	
	if(get(l)==m)n=l;
	cout<<n;

	return 0;
}
