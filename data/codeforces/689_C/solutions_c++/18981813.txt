#include <iostream>
using namespace std;

long long find (long long x) {
	long long ans=0;
	for (long long i=2; ;i++) {
		if (!(x/(i*i*i)))
			break;
		ans+=x/(i*i*i);
	}
	return ans;
}

int main() {
	long long m, n=-1;
	cin>>m;
	long long a=0, b=1e16, mid;
	while (a<b) {
		mid=(a+b)/2;
		if (find(mid)<m)
			a=mid+1;
		else
			b=mid;
	}
	if (find(a)==m)
		n=a;
	cout<<n<<'\n';
	return 0;
}
