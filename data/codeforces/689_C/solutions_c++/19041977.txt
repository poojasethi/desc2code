#include<cstdio>
#include<iostream>
using namespace std;
long long m,mid,i,s;
int main()
{
	long long l=7,r=1e18;
	cin>>m;
	while(l<r)
	{
		mid=(l+r)/2;
		s=0;
		for(i=2;i*i*i<=mid;i++)
		{
			s+=mid/(i*i*i);
			if(s>m)break;
		}
		if(l==mid)break;
		if(s>=m)r=mid;
		else l=mid;
	}
	s=0;
	for(i=2;i*i*i<=r;i++)
		s+=r/(i*i*i);
	if(s==m)cout<<r<<'\n';else cout<<-1<<'\n';
	return 0;
}
