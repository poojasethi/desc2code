#include <bits/stdc++.h>
using namespace std;
long long m;
long long solve (long long n) {
	long long ans=0;
	for (long long i=2;i*i*i<=n;i++) {
		ans += n/(i*i*i);
	}
	return ans;
}
int main() {
	scanf("%I64d",&m);
	long long lo=0, hi=8000000000000000;
	while (lo<=hi) {
		long long mid=(lo+hi)/2;
		if (solve(mid)>=m) hi=mid-1;
		else lo=mid+1;
	}
	if (solve(lo)!=m) printf("-1\n");
	else printf("%I64d\n", lo);
	return 0;
}
