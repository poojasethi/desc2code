#include<cstdio>
#include<iostream>
#define ll long long
using namespace std;

int m,n,k,a;

ll find(ll x)
{
   ll num=0;
   for(ll i=2;i*i*i<=x;i++)
   num=num+x/(i*i*i);
   return num;
}


int main()
{
	ll n;
	cin>>n;
	ll r=1e18,l=1,mid,ans=-1;
	while(l<=r)
	{
		mid=(l+r)>>1;
		ll p=find(mid);
		if(p==n)
		ans=mid;
		if(p>=n)
		 r=mid-1;
		 else l=mid+1;	 		
	}
	 cout<<ans<<endl; 
}
