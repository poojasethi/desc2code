#include <cstdio>

using namespace std;

int main() {
	long long l = 1ll, r = 10000000000000000ll, mid, m, i, s, t;
	
	scanf("%I64d", &m);
	while (l < r) {
		mid = (l + r) >> 1;
		for (i = 2ll, s = 0ll; (t = i * i * i) <= mid; ++i) s += mid / t;
		if (s >= m) r = mid; else l = mid + 1ll;
	}
	for (i = 2ll, s = 0ll; (t = i * i * i) <= l; ++i) s += l / t;
	if (s == m) printf("%I64d\n", l); else printf("-1\n");
	
	return 0;
}
