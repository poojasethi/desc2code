#include<cstdio>
using namespace std;
typedef long long ll;

ll n ,arr[1000009] ;
ll check(ll x)
{
	ll s = 0 ;
	for(ll i =2 ;i*i*i<=x;i++)
	{
		s += (x/(i*i*i)) ;
	}
	return s ;
}
int main()
{
	scanf("%I64d",&n);
	ll lo = 1 ,hi = 1000000000000000000ll , mid ;
	while(lo<hi)
	{
		mid = (lo+hi)/2 ;
		if(check(mid)<n) lo = mid+1 ;
		else hi = mid ;
	}
	if(check(lo)==n)printf("%I64d",lo);
	else printf("-1");
}
