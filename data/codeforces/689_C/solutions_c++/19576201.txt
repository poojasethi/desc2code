#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll M;
ll check(ll limit){
    ll cnt=0;
    for(ll i=2;i<=(ll)1e6&&limit/(i*i*i);i++){
        cnt+=(limit/(i*i*i));
    }
return cnt;}
int main(){cin>>M;
ll lo=0,hi=(ll)1e18+1;
while(lo<hi){
    ll mid=lo+(hi-lo)/2;
    if(check(mid)>=M)hi=mid;
    else lo=mid+1;
}
if(check(lo)==M)
    cout<<lo<<endl;
else
    cout<<-1<<endl;
return 0;}
