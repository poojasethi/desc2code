#include<stdio.h>
typedef long long ll;
ll check(ll n)
{
	ll ans=0;
	for(ll k=2;k*k*k<=n;k++)
	ans+=n/(k*k*k);
	return ans;
}
int main()
{
	ll m,l=1,r=1e18,ans=-1,mid,num;
	scanf("%I64d",&m);
	while(l<=r)
	{
	    mid=(l+r)/2;
		num=check(mid);
		if(num==m)
		ans=mid;
		if(num>=m)
		r=mid-1;
		else
		l=mid+1;
	}
	printf("%I64d\n",ans);
}
