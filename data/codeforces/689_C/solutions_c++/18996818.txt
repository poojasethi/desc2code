#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;

typedef long long ll;
ll m,l,r,mid;

ll get(ll x){
	ll ans=0;
	for(ll i=2;i*i*i<=x;i++)
	ans+=x/(i*i*i);
	return ans;
}
int main(){
	scanf("%I64d",&m);
	l=0; r=1e18;
	while(r-l>1){
		mid=(r+l)>>1;
		if(get(mid)>=m) r=mid;
		else l=mid;
	}
	if(get(r)!=m) puts("-1");
	else printf("%I64d\n",r);
	return 0;
}
