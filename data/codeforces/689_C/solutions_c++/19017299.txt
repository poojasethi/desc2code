#include<cstring>
#include<cstdio>
#include<cmath>
using namespace std;
const int qq=1e6+5;
typedef long long ll;
ll f(ll n)
{
	ll sum=0;
	for(ll i=2; i*i*i<=n; ++i)
		sum+=n/(i*i*i);
	return sum;
}
int main()
{
	ll n;scanf("%lld",&n);
	ll l=1,r=5e18;
	ll m;
	ll ans=-1;
	while(l<=r){
		m=(l+r)/2;
		ll sum=f(m);
		if(sum==n)	ans=m;
		if(sum>=n)	r=m-1;
		else	l=m+1;
	}
	printf("%lld\n",ans);
	return 0;
}
