#include <stdio.h>
#include <iostream>
using namespace std;
long long F(long long x)
{
	long long sum=0;
	for(long long i=2;i*i*i<=x;i++)
	sum+=x/(i*i*i);
	return sum;
}
int main()
{
	long long m,l=0,r=1e18,mid,k,ans=-1;
	cin>>m;
	while(r-l>1)
	{
		mid=(l+r)>>1;
		k=F(mid);
		if(k==m)
		ans=mid;
		if(k>=m)
		r=mid;
		else
		l=mid;
	}
	cout<<ans<<endl;
}
