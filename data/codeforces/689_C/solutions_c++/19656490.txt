#include<iostream>
using namespace std;
int main()
{
	long long int m,n,beg=0,end=1e16,mid,sum=0,i;
	cin>>m;
	while(beg<end)
	{
		mid=(beg+end)/2;sum=0;
		for(i=2;i*i*i<=mid;i++)sum+=mid/(i*i*i);
		if(sum<m)beg=mid+1;
		else end=mid;
	}sum=0;
	for(i=2;i*i*i<=beg;i++)sum+=(beg/(i*i*i));
	if(sum!=m)cout<<-1;
	else cout<<beg;
}
