#include<bits/stdc++.h>
using namespace std;
long long get(long long n)
{
	long long i;
	long long ans=0;
	for(i=2;i*i*i<=n;i++)
	{
		 ans+=n/(i*i*i);
	}
	return ans;
}
int main()
{
	long long m;
	cin>>m;
	long long temp,l,r,mid;
	r=10000000000000000;
	l=1;
	while(l<r)
	{
		mid=(l+r)/2;
		temp=get(mid);
	    if(temp<m)
		  l=mid+1;
		else
		  r=mid;
	}
	if(get(l)==m)
	  cout<<l;
	else
	  cout<<-1;
	return 0;
}
