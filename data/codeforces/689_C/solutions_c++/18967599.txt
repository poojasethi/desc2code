#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll func_m(ll n)
{
	ll s=0;
	for(ll k=2;k*k*k<=n;k++)
	{
		s+=(ll) (n/(k*k*k));
	}
	return s;
}
	
int main()
{
	ll m;
	cin>>m;
	ll l=0, r=(ll)2e60;
	while(l<r)
	{
		ll mid=(l+r)/2;
		if(func_m(mid)<m) l=mid+1;
		else r=mid;
	}
	if(func_m(l)==m) cout<<l<<endl;
	else cout<<"-1\n";
	return 0;
}
