#include<stdio.h>
typedef long long ll;
ll check(ll n)
{
	ll res=0;
	for(ll k=2; k*k*k<=n; k++)
		res+=n/(k*k*k);
	return res;
}
int main()
{
	ll m,res=-1,l=1,r=1e18,mid;
	scanf("%I64d",&m);
	while(r>=l)
	{
		mid=(l+r)/2;
		ll num= check(mid);
		if(num==m)
			res=mid;
		if(num>=m)
			r=mid-1;
		else
			l=mid+1;
	}
	printf("%I64d\n",res);
}
