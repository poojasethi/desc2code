#include<cstdio>
using namespace std;
typedef long long ll;
ll m,l,r;
ll judge(ll n)
{
	ll ans=0;
	for (ll i=2;i*i*i<=n;i++) ans+=n/(i*i*i);
	return ans;
}
int main()
{
	scanf("%I64d",&m);
	l=1;r=80000000ll*100000000ll;
	while (l<r)
	{
		ll mid=(l+r)>>1;
		if (judge(mid)>=m) r=mid;else l=mid+1;
	}
	if (judge(l)==m) printf("%I64d\n",l);else printf("-1\n");
}
