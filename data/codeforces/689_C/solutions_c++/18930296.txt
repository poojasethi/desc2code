#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main(){
	ios::sync_with_stdio(false);
	ll m;cin>>m;
	ll lo=1, hi=m*8;
	ll ans=-1;
	while(lo <= hi){
		ll mid = (lo+hi)>>1;
		ll cnt=0;
		for(ll i=2; i*i*i <= mid; i++){
			cnt+=(mid/(i*i*i));	
		}
		if(cnt >= m){
			hi = mid-1;
			if(cnt == m)ans=mid;
		}
		else
			lo = mid+1;	
	}
	cout<<ans<<endl;
	return 0;
}
