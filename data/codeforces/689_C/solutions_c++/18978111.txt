#include<cstdio>
typedef long long LL;

LL cal(LL n){
	LL sum=0;
	for(LL i=2;i<=1e6;++i){
		if(n/i/i/i==0) break;
		else sum+=n/i/i/i;
	}
	return sum;
}

int main(){
	LL m,n,ans;
	scanf("%I64d",&m);
	LL l=1,r=8*(m+1);
	while(l<=r){
		n=(l+r)>>1;
		if(cal(n)>=m){
			r=n-1;
			ans=n;
		}
		else l=n+1;
	}
	if(cal(ans)==m) printf("%I64d\n",ans);
	else puts("-1");
}
