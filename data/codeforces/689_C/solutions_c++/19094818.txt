#include<iostream>
using namespace std;
typedef long long ll;
ll f(ll m)
{	ll ans=0;
	for(ll k=2;k*k*k<=m;k++)
	ans+=m/(1ll*k*k*k);
	return ans;
}
int main()
{	ll m,n=-1;
	cin>>m;

	ll l=0,r=5e15;
	while(l<r)
	{	ll mid=(l+r)/2;
		if(f(mid)<m) l=mid+1;
		else r=mid;
	}	
	if(f(l)==m)n=l;
	cout<<n;

	return 0;
}
