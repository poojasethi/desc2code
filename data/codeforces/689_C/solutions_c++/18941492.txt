#include<cstdio>
#define LL long long
LL f(LL k)
{
	LL i,ans=0;
	for(i=2;i*i*i<=k;i++)
	{
		ans=ans+k/(i*i*i);
	}
	return ans;
}
int main()
{
	LL m;
	scanf("%I64d",&m);
	LL l=8,r=((LL)1<<62),mid;
	int flag=0;
	while(l<=r)
	{
		mid=(l+r)/2;
		LL tmp=f(mid);
		if(tmp==m)
			flag=1;
		if(tmp>=m)
			r=mid-1;
		else
			l=mid+1;
	}
	if(flag==0)
		puts("-1");
	else
		printf("%I64d\n",l);
	return 0;
}
