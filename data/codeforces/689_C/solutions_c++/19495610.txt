#include <stdio.h>
#define lli long long int

lli get(lli k)
{
	lli ans=0;
	for(lli i=2; i*i*i<=k; i++)
		ans+=k/(i*i*i*1ll);
	return ans;
}

int main()
{
	lli n=-1, m, l=1, r=5e16;
	scanf("%llid", &m);
	while(l<r){
		lli p=(l+r)/2;
		if(get(p)<m) l=p+1;
		else r=p;
	}
	if(get(l)==m) n=l;
	printf("%lld", n);
}
