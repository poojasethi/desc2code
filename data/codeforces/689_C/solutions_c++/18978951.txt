#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
#include <cstring>
#include <vector>
using namespace std;
typedef long long LL;
LL l,r,n,m,z,x,y;
LL getans(LL x){
	LL ans=0;
	for (LL i=2;i*i*i<=x;++i) ans+=x/(i*i*i);
	return ans;
}
int main(){
	scanf("%lld",&m);
	n=0;
	for (LL i=1LL<<60;i>0;i/=2)
	if (getans(n+i)<m) n+=i;
	++n;
	if (getans(n)==m) printf("%lld",n);
	else printf("-1");
}
