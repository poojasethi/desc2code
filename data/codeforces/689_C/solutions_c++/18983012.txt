#include <bits/stdc++.h>
using namespace std;
typedef long long LL;

LL pow3(LL &a) {
	return a * a * a;
}

LL Sum(const LL& c) {
	LL sum = 0;
	for (LL i = 2; i <= 1e6; ++i)
		sum += c / pow3(i);
	return sum;
}

int main() {
	LL N;
	cin >> N;
	LL l = 0, r = 1e18, mid;
	while (l <= r) {
		mid = (l + r) >> 1;
		if (Sum(mid) >= N) r = mid - 1;
		else l = mid + 1;
	}
	if (Sum(r + 1) == N) cout << r + 1 << endl;
	else puts("-1");
	return 0;
}
