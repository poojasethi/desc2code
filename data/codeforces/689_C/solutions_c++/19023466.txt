#include<iostream>
using namespace std;
long long n;
long long mul(long long x)
{
	return x*x*x;
}
long long cal(long long k)
{
	long long sum = 0;
	for(int i=2; i<=1e6; i++)
		sum+=k/mul(i);
	return sum;
}
int main()
{
	cin>>n;
	long long l = 0,r = 1e18;
	while(l<=r)
	{
		long long mid=(l+r)/2;
		if(cal(mid)>=n)r=mid-1;
		else l=mid+1;
	}
	if(cal(r+1)==n)cout<<r+1<<endl;
	else cout<<"-1"<<endl;
}
