#include<cstdio>
using namespace std;

long long m;

long long f(long long n)
{
	long long tmp=0,i=2;
	while(i*i*i<=n)
	{
		tmp+=n/i/i/i;
		if(tmp>m)return tmp;
		i++;
	}
	return tmp;
}

int main(){
	scanf("%I64d",&m);
	long long l=8,r=m<<3,mid;
	while(l!=r)
	{
		mid=(l+r)>>1;
		if(f(mid)>=m)r=mid;
		else l=mid+1;
	}
	if(f(l)==m)printf("%I64d",l);
	else printf("-1");
	return 0;
}
