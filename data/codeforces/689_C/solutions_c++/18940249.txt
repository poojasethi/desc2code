#include <stdio.h>

typedef long long int ll;

ll get(ll n){
	ll qtd = 0;
	for(ll k = 2; k *k *k <= n; k++) qtd += n / (k*k*k);
	return qtd;
}


int main(){
	ll m;

	while(scanf("%I64d ", &m)==1){

		ll l=0, r = 5e15, ans = -1;
		while(l<r){
			ll mid = (l+r)/2;
			if(get(mid) < m) l=mid+1;
			else r = mid;
		}

		if(get(l)==m) ans = l;

		printf("%I64d\n", ans);

	}

	return 0;
}
