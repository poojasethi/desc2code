#include <bits/stdc++.h>
#define oo 1e18
using namespace std;
typedef long long ll;
ll m;
ll cal(ll n)
{
    ll kq=0;
    for(ll i=2;i*i*i<=n;i++)
    kq+=n/(i*i*i);
    return kq;
}
ll res()
{
    ll d=1,c=oo,g;
    while(d<c)
    {
        g=(d+c)/2;
        if(cal(g)>=m) c=g;
        else d=g+1;
    }
    return cal(d)==m?d:-1;
}
int main()
{
    cin>>m;
    cout<<res();
}
