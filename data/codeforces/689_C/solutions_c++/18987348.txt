#include<cstdio>
typedef long long ll;
ll m,l,r,mid;
ll calc(ll n){
	ll res,ans=0;
	for (int i=2;i;i++){
		res=1;
		for (int j=1;j<4;j++) res=res*i;
		if (res>n) break;
		ans+=n/res;
	}
	return ans;
}
int main(){
	scanf("%I64d",&m);
	l=0,r=1e16;
	while (l<r){
		mid=(l+r)>>1;
		if (calc(mid)>=m) r=mid;
		else l=mid+1;
	}
	if (calc(l)==m)	printf("%I64d",l);
	else puts("-1"); 
	return 0;
}
