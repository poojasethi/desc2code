#include<iostream>
using namespace std;

long long int function(long long int n)
{
	long long int ans = 0,i;
	for(i = 2;i*i*i<=n;i++)
	{
		ans = ans + n/(i*i*i);
	}
	return ans;
}
int main()
{
	long long int m , l =1 , r = 10000000000000000,mid;
	cin>>m;
	while(l<r)
	{
		 mid =l+(r-l)/2;
		 if(function(mid)<m)
		   l = mid+1;
		 else
		   r = mid;
	}
	if(function(l) == m)
	  cout<<l<<endl;
	else
	  cout<<-1<<endl;
}
