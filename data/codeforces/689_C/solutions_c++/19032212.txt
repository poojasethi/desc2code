#include<iostream>
#include<cstdio>
#define ll long long
using namespace std;

int main()
{
	ll n,l=1,r=1e18,mid,ans=-1,i;
	scanf("%I64d",&n);
	while(l<=r)
	{
		mid=(l+r)/2;
		ll num=0;
		for(i=2;i*i*i<=mid;i++) num+=mid/(i*i*i);
		if(num==n) ans=mid;
		if(num>=n) r=mid-1;
		else l=mid+1;
	}
	printf("%I64d\n",ans);
	return 0;
}
