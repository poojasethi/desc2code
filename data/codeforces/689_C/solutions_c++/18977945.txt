#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll x;

ll f(ll x) {
	ll ans = 0;
	for (int i = 2; 1LL*i*i*i <= x; ++i) {
		ans += x / (1LL * i * i * i);
	}
	return ans;
}

int main() {
	cin >> x;
	ll L = 0, R = 2000000000000000000LL;
	while (L < R) {
		ll M = (L + R) >> 1;
		if (f(M) < x) {
			L = M + 1;
		} else {
			R = M;
		}
	}
	cout << (f(L) == x ? L : -1) << endl;
}
