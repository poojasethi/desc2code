#!/usr/bin/python

import sys
import math
import string

def feasible(N, M, T):
    twos = T/2 - T/6
    threes = T/3 - T/6
    sixes = T/6

    return twos + threes + sixes >= N + M and twos + sixes >= N and threes + sixes >= M

if __name__ == "__main__":
    N, M = map(int, sys.stdin.readline().split())
    lo = N+M
    hi = N+M
    while not feasible(N, M, hi):
        hi *= 2

    while lo < hi - 1:
        md = (lo + hi)/2
        if feasible(N, M, md):
            hi = md
        else:
            lo = md

    print hi
