def func(h, n, m):
    if 2 * n > h or 3 * m > h:
        return False
    om = (h + 3) // 6
    em = m - om
    an = h // 2
    rn = an - em
    return rn >= n

n, m = map(int, raw_input().split())

ans = None
low, high = 0, 10000000
while low <= high:
    mid = (low + high) // 2
    if func(mid, n, m):
        ans, high = mid, mid - 1
    else:
        low = mid + 1

print ans
