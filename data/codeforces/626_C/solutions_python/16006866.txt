n,m = [int(X) for X in raw_input().split(" ")]
l2 = [0]
l3 = [0]
for i in range(1,n+1):
	l2.append(2*i)
for i in range(1,m+1):
	l3.append(3*i)
i = 1
j = 1
l = []
n1 = n
m1 = m
while i <= n1 and j <= m1:
	if(l2[i] < l3[j]):
		l.append(l2[i])
		i += 1
	elif(l2[i] > l3[j]):
		l.append(l3[j])
		j += 1
	else:
		if l2[n1] > l3[m1]:
			m1 += 1
			l3.append(3*m1)
			j += 1
		else:
			n1 += 1
			l2.append(2*n1)
			i += 1
print max(l2[n1],l3[m1])
