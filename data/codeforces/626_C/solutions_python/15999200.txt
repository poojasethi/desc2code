n, m = map(int, raw_input().split())

ans = 0
if m >= n:
    print m*3
else:
    # n > m
    if 3 * m <= n:
        print 2 * n
    else:
        diff = (n-m)/2
        ans += 6*diff
        n -= 3*diff
        m -= diff
        blocks = m/2
        n -= 2*blocks
        m -= 2*blocks
        print ans + blocks * 6 + max(m*3, 2*n)
