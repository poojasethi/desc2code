#!/usr/bin/env python

n,m = map(int,raw_input().split())

maxtwos = 2*n
maxthrees = 3*m
overlap = len(list(set(range(2,maxtwos+1,2))&set(range(3,maxthrees+1,3))))

visited = [False]*4000000
for i in range(0, maxtwos+1, 2):
	visited[i] = True
for i in range(0, maxthrees+1, 3):
	visited[i] = True

if overlap:
	while overlap:
		while visited[maxtwos+2] and maxtwos <= maxthrees:
			maxtwos += 2
		while visited[maxthrees+3] and maxtwos > maxthrees:
			maxthrees += 3

		if maxtwos <= maxthrees:
			maxtwos += 2
			visited[maxtwos] = True
		else:
			maxthrees += 3
			visited[maxthrees] = True
		overlap -= 1

	print max(maxthrees, maxtwos)
else:
	print max(2*n, 3*m)
