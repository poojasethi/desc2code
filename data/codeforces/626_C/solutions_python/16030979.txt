n, m = map(int, raw_input().split())

# looking for smallest X s.t.
#     at least n multiples of 2 below X
#     at least m multiples of 3 below X
#     at least n + m multiples of 2 or 3 below X

def check(X):
    return X / 2 >= n and X / 3 >= m and X / 2 + X / 3 - X / 6 >= n + m

r = 1
while not check(r):
    r *= 2

l = r/2
while l < r:
    mid = (l + r) / 2
    if check(mid):
        r = mid - 1
    else:
        l = mid + 1

r -= 2
while not check(r):
    r += 1

print r
