n, m = map(int, raw_input().split())
h = 1
while (n > 0) or (m > 0):
    h += 1
    if (h%6 == 0) and (m > 0) and (n > 0):
        if (2*n > 3*m):
            n -= 1
            continue
        else:
            m -= 1
            continue
    if (h%3 == 0) and (m > 0):
        m -= 1
        continue
    if (h%2 == 0) and (n > 0):
        n -= 1
        continue
