from collections import deque

M    = 1000005 * 3

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n, m = ia()

for i in range(2, M):
    if i % 2 == 0 and i % 3 == 0:
        if m>=n:
            m -= 1
        else:
            n -= 1
    elif i % 2 == 0 and n>0: n -= 1
    elif i % 3 == 0 and m>0: m -= 1

    if n==0 and m==0:
        print i
        break
