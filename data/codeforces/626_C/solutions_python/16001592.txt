# -*- coding: utf-8 -*-


def main():
    n, m = map(int, raw_input().split())

    def check(target):
        z = target / 6
        x = target / 2 - z
        y = target / 3 - z

        if x + z < n:
            return False

        z = min(z, x + z - n)

        return y + z >= m

    lb = 0
    hb = 10000000
    while lb + 1 < hb:
        p = (lb + hb) / 2
        if check(p):
            hb = p
        else:
            lb = p

    print hb


if __name__ == '__main__':
    main()
