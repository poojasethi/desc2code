a, b=map(int, raw_input().split())
if a==0:
    print b*3
    exit()
if b==0:
    print a*2
    exit()
x=-3+6*b
if a%2==1:
    y=2+6*(a/2)
else:
    y=4+6*((a-1)/2)
z=6
ans=max(x, y)
while z<max(x, y):
    if x>y:
        x=x-6
    else:
        if y%6==2:  
            y=y-4
        else:
            y=y-2
    ans=min(ans, max(z, max(x, y)))
    z=z+6
print ans
exit()
