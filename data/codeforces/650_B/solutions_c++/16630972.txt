#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
#define LL long long
#define rep(i,a,b) for(int i=a;i<=b;++i)
#define erp(i,a,b) for(int i=a;i>=b;--i)
using namespace std;
const int MAXN = 500005;
void gmax(int&a, int b) { a<b?a=b:0; }
int N, a, b, T, ans;
char s[MAXN];
LL f[MAXN], g[MAXN];

int main()
{
	scanf("%d%d%d%d", &N, &a, &b, &T);
	scanf("%s", s+1);
	rep(i, 1, N) f[i] = f[i-1] + a*(i!=1) + b*(s[i]=='w') + 1;
	erp(i, N, 1) g[i] = g[i+1] + a + b*(s[i]=='w') + 1;
	int x = 1, y = 2;
	while (x<=N && f[x]<=T)
	{
		while (y<=N && f[x]+min(x-1, N-y+1)*a+g[y] > T) ++y;
		while (y<=x) ++y;
		gmax(ans, x+N-y+1); ++x;
	}
	printf("%d\n", ans);
	return 0;
}
