#include <cstdio>
#include <algorithm>
using namespace std;
const int maxn = 500000 + 10;
char photo[maxn];
int s[maxn * 2];
int n, a, b, T;
int main()
{
    scanf("%d%d%d%d", &n, &a, &b, &T);
    scanf("%s", photo);
    int cnt = 0;
    for(int i = 0;i < n;++ i){
        if(photo[i] == 'w') s[i] = s[i + n] = b + 1;
        else s[i] = s[i + n] = 1;
        cnt += s[i];
    }
    int L = 1, R = n, ans = 0;
    cnt -= s[0];
    while(L <= n && R < 2 * n){
        cnt += s[R ++];
        while(R - L > n || cnt + (R - L - 1 + min(R - n - 1, n - L)) * a > T){
            cnt -= s[L ++];
        }
        ans = max(ans, R - L);
    }
    printf("%d\n", ans);
    return 0;
}
