#include <cstdio>
#include <algorithm>
#define MAXN 1000005
int n, a, b, T, c[MAXN];
char s[MAXN];
int main() {
    scanf("%d%d%d%d", &n, &a, &b, &T);
    scanf("%s", s); int sum = 0;
    for(int i = 0; i < n; ++ i) {
        c[i] = c[n + i] = (s[i] == 'w') * b + 1;
        sum += c[i];
    }
    int l = 0, r = n, ans = 0;
    while(l <= n && r < n + n) {
        sum += c[r ++];
        for(; r-l>n || sum + (r-l-1+std::min(r-n-1, n-l))*a > T; sum -= c[l ++]);
        ans = std::max(ans, r-l);
        if(ans == n) break;
    }
    printf("%d\n", ans);
    return 0;
}
