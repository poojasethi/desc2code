#include<cstdio>
using namespace std;
#define max(a,b) (a>b?a:b)
#define min(a,b) (a<b?a:b)

int n,a,b,t,ans=0;
char s[500500];
int f[2][500500];

int main(){
	scanf("%d%d%d%d",&n,&a,&b,&t);
	scanf("%s",s);
	t--;
	if(s[0]=='w')t-=b;
	if(t<0)printf("0");
	else{
		for(int i=1;i<n;i++)f[0][i]=f[0][i-1]+(s[i]=='w'?1:0)*b;
		for(int i=1,j=n-1;i<n;i++,j--)f[1][i]=f[1][i-1]+(s[j]=='w'?1:0)*b;
		if(f[0][n-1]+(n-1)*(a+1)<=t){
			printf("%d",n);
			return 0;
		}
		int i=0,j=0;
		while(f[0][i]+f[1][j]+(i+j)+(min(i,j)+i+j)*a<=t)++j;
		--j;
		for(i=0;i<n;i++){
			while(f[0][i]+f[1][j]+(i+j)+(min(i,j)+i+j)*a>t && j>0)--j;
			if(f[0][i]+f[1][j]+(i+j)+(min(i,j)+i+j)*a>t)break;
			ans=max(ans,i+j+1);
		}
		printf("%d",ans);
	}
	return 0;
}
