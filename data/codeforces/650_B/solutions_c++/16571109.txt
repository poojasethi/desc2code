#include<bits/stdc++.h>
#define N 500005
using namespace std;
typedef long long LL;
char s[N];LL L[N],R[N];
int n,a,b,T,ret,now,i,ans,c;
inline void work(LL *L,LL *R){
	for (i=2;i<=n;i++){
		if (R[i]>T) continue;
		LL rest=T-R[i]-1ll*(i-1)*a;
		//if (L[2]>rest) continue;
		int id=upper_bound(L+2,L+n+1,rest)-(L+1);id--;
		ans=max(ans,i-1+id);
  }
}
int main(){
	scanf("%d%d%d%d",&n,&a,&b,&T);
	scanf("%s",s+1);
	T-=(s[1]=='w')*b;T--;
	if (T<0) {puts("0");return 0;}
	for (i=2;i<=n;i++)
		if ((R[i]=R[i-1]+a+(s[i]=='w')*b+1)<=T) ans=max(ans,i-1);
	for (i=2;i<=n;i++)
		if ((L[i]=L[i-1]+a+(s[n-i+2]=='w')*b+1)<=T) ans=max(ans,i-1);
	work(L,R);
	work(R,L);
	printf("%d\n",min(n,ans+1));
}
