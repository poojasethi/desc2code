using namespace std;

#define Max(a,b) a>b?a:b
#define Min(a,b) a>b?b:a

const int MAX=1000009;
const int INF=0x3f3f3f3f;

int a,b,c,d;
int cnt[1000080];
int main()
{
	char ch[700000];
	while(cin>>a>>b>>c>>d)
	{
		scanf("%s",ch);
		int now=0;
		for(int i=0;i<a;i++)
		{
			now+=cnt[i]=cnt[i+a]=(ch[i]=='w'?c+1:1);
		}
		int l=1,r=a,ans=0;
		now-=cnt[0];
		while(l<=a&&r<(a<<1))
		{
			now+=cnt[r++];
			while(r-l>a or now+(r-l-1+min(a-l,r-a-1))*b>d) now-=cnt[l++];
			if(l>a) break;
			ans=max(ans,r-l);
		}
		printf("%d\n",ans);
	}
}
