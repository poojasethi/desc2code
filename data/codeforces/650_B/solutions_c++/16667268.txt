using namespace std;
int n,a,b,T,ans;
string s;
long long sum[500010];
long long check(int l,int r)
{
	return (long long)(min(l,r)+l+r)*a+(sum[n]+sum[r+1]-sum[n-l]);
}
int main()
{
	cin>>n>>a>>b>>T;
	cin>>s;
	for(int i=0;i<n;i++)
		sum[i+1]=sum[i]+(s[i]=='w'?(b+1):1);
	ans=0;
	for(int l=0;l<n;l++)
	{
		if(check(l,0)>T)continue;
		int low=0,hig=n-l;
		while(hig-low>1)
		{
			int mid=(hig+low)/2;
			if(check(l,mid)<=T)low=mid;
			else hig=mid;
		}
		ans=max(ans,l+low+1);
	}
	cout<<ans;
	return 0;
}
