#include<iostream>
#include<stdio.h>
using namespace std;
char s;
int c[1000500];
int n,a,b,t;
int main()
{
	cin>>n>>a>>b>>t;
	int now=0;
	for(int i=0;i<n;i++)
	{
		cin>>s;
		now+=c[i]=c[i+n]=(s=='w')*b+1;
	}
	int l=1,r=n,ans=0;
	now-=c[0];
	while(l<=n&&r<2*n)
	{
		now+=c[r++];
		while(r-l>n or now+(r-l-1+min(n-l,r-n-1))*a>t) now-=c[l++];
		if(l>n) break;
		ans=max(ans,r-l);
	}
	cout<<ans<<endl;
}
