#include "cstdio"
#include "algorithm"
using namespace std;
typedef long long ll;
char s[500000+5];
ll t[1000000+5];
int main()
{
	int i,j,n,a,b,T;
	scanf("%d%d%d%d",&n,&a,&b,&T);
	scanf("%s",s);
	if(s[0]=='w') t[n]=b+1;
	else t[n]=1;
	for(i=1;i<n;i++)
	{
		t[i+n]=t[i+n-1];
		if(s[i]=='w') t[i+n]+=b+1;
		else t[i+n]+=1;
		t[n-i]=t[n-i+1];
		if(s[n-i]=='w') t[n-i]+=b+1;
		else t[n-i]+=1;
	}
	int res=0;
	i=1,j=n;
	while(1)
	{
		while(j-i+1<=n&&t[i]+t[j]-t[n]+(min(n-i,j-n)+j-i)*a<=T&&i<=n&&j<2*n) 
		{
			res=max(res,j-i+1);
			j++;
		}
		if(i>n||j==2*n) break;				
		i++;
	}
	printf("%d\n",res);
	return 0;
}
