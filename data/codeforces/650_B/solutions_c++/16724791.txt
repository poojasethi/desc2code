#include <bits/stdc++.h>

using namespace std;

const int N=500100;
char s[N];

int main()
{
    int n,a,b,t,c,r,ans;
    scanf("%d %d %d %d",&n,&a,&b,&t);
    scanf("%s",s);
    r=n-1; c=0; ans=0;
    for (int i=0; i<n; i++) if (s[i]=='w') c+=b;
    for (int l=0; l<n-1; l++)
    {
        while ((c+l+r+1+(r+l+min(l,r))*a)>t)
        {
            if (s[r]=='w') c-=b;
            r--;
        }
        if (r<0) break;
        ans=max(ans,r+l+1);
        if (s[n-l-1]=='w') c+=b;
    }
    printf("%d",min(ans,n));
}
