#include <bits/stdc++.h>
using namespace std;
#define M 500200

char s[M];
int n, a, b, T;
int f[M], g[M];
int rlt;

void calc(int *f) {
	f[0] = 0;
	for (int i = 1; i < n; i++) {
		f[i] = f[i-1] + 1 + (s[i] == 'w') * b + a;
	}
}

void chkmax(int x) { if (rlt < x) rlt = x; }

void update(int *f, int *g) {
	for (int i = 1, j = n - 1; i < n && f[i] <= T; i++) {
		chkmax(i + 1);
		while (j && f[i] + i * a + g[j] > T) j--;
		chkmax(i + 1 + j);
	}
}

int main() {
	scanf("%d %d %d %d %s", &n, &a, &b, &T, s);
	T -= b * (s[0] == 'w') + 1;
	calc(f);
	reverse(s+1, s+n);
	calc(g);

	if (T >= 0)
		rlt = 1,
		update(f, g),
		update(g, f);
	printf("%d\n", min(rlt, n));
	return 0;
}
