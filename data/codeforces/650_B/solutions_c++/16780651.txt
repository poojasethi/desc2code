#include <iostream>
#include <cstdio>
#define MAXN 1000005
#define LL long long int
using namespace std;

int n, a, b, cost[MAXN], t, ans;
LL sum;
char w[MAXN];

int main()
{
	scanf("%d%d%d%d%s",&n,&a,&b,&t,w);
	for(int i=0;i<n;++i)
	{
		if(w[i]=='w')cost[i]=cost[i+n]=b+1;
		else cost[i]=cost[n+i]=1;
		sum+=cost[i];
	}
	sum-=cost[0];
	int l=1, r=n;
	while(l<=n&&r<n+n)
	{
		sum+=cost[r++];
		while(r-l>n||sum+(r-l-1+min(r-n-1,n-l))*a>t)
			sum-=cost[l++];
		ans=max(ans,r-l);
	}
	printf("%d\n",ans);
	return 0;
}
