#include <bits/stdc++.h>
using namespace std;
#define LL long long
int N,A,B,T,ans,res;
char s[500010];
LL tt;
LL pre[500010],suc[500010];
void work(){
	pre[1]=1;
	if(s[1]=='w')pre[1]+=B;
	for (int i=2;i<=N;i++){
		pre[i]=pre[i-1]+A+1;
		if (s[i]=='w')pre[i]+=B;
	}
	suc[N+1]=0;
	for (int i=N;i;i--){
		suc[i]=suc[i+1]+A+1;
		if (s[i]=='w')suc[i]+=B;
	}
	int j=1;
	for (int i=1;i<=N;i++){
		tt=T-pre[i];
		if (tt>=0)ans=max(ans,i);
		tt-=A*(i-1);
		if (tt>0){
			while (j<=N&&suc[j]>tt)j++;
			res=i+N+1-j;
			if (res>N)res=N;
			ans=max(ans,res);
		}
	}
}
int main(){
	scanf("%d%d%d%d",&N,&A,&B,&T);
	scanf("%s",s+1);
	work();
	for (int i=2,j=N;i<j;i++,j--)
		swap(s[i],s[j]);
	work();
	printf("%d",ans);
}
