#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
const int N=500010;
int n,a,b,T,i,j,ans;
char s[N];
ll pre[N],suf[N];
inline int cost(char x){
  if(x=='w')return b+1;
  return 1;
}
void solve(){
  pre[1]=cost(s[1]);
  for(i=2;i<=n;i++)pre[i]=pre[i-1]+a+cost(s[i]);
  for(i=n;i;i--)suf[i]=suf[i+1]+a+cost(s[i]);
  for(i=j=1;i<=n;i++){
    if(pre[i]<=T)ans=max(ans,i);
    ll x=pre[i]+(i-1)*a;
    while(j<=n&&x+suf[j]>T)j++;
    if(j<=i)j=i+1;
    if(j<=n)ans=max(ans,i+n-j+1);
  }
}
int main(){
  scanf("%d%d%d%d%s",&n,&a,&b,&T,s+1);
  solve();
  for(i=2,j=n;i<j;i++,j--)swap(s[i],s[j]);
  solve();
  printf("%d",ans);
  return 0;
}
