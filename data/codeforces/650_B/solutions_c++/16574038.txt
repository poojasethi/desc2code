#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
#include<string>
#include<vector>
#include<cmath>
#include<map>
using namespace std;
typedef long long ll;
const int N=5e5+1;
ll d[N];
string s;
int main()
{
    ll n,a,b,t,i,j,ans=0;
    cin>>n>>a>>b>>t>>s;
    d[0]=1+(s[0]=='w')*b;
    for(i=1;i<n;i++)
        d[i]=d[i-1]+a+1+(s[i]=='w')*b;
    j=n-1;
    ll sum=0;
    for(i=0;i<n;i++)
    {
        while(sum+d[j]+min(i,j)*a>t&&j>0)j--;
        if(sum+d[j]+min(i,j)*a<=t)
        ans=max(ans,i+j+1);
        else break;
        if(ans>=n){ans=n;break;}
        sum+=1+(s[n-i-1]=='w')*b+a;
    }
    cout<<ans;
    return 0;
}
