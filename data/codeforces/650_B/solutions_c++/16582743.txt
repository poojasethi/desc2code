#include <cstdio>
#include <algorithm>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <iostream>
#define fr(a,b,c) for(int a = b; a<c; a++)
#define rp(a,b) fr(a,0,b)

using namespace std;

int n, a, b, T;
int r;
char g[1001000];
int v[1001000];

void proc() {
	v[0] = 0;
	for (int i = n-1; i > 0; i--) {
		v[n-i] = v[n-1-i] + a + (g[i]=='w')*b + 1;
	}
	int t = T+a;
	rp(i,n) {
		t = t-a-(g[i]=='w')*b-1;
		if (t < 0) break;
		r = max(r, i+1);
		if (t-i*a >= 0) {
			int j = upper_bound(v, v+n-i, t-i*a) - v - 1;
			r = max(r, i+1+j);
		}
	}
}

int main() {
	scanf("%d%d%d%d", &n, &a, &b, &T);
	scanf("%s", g);
	r = 0;
	proc();
	reverse(g+1,g+n);
	proc();
	printf("%d\n", r);
}
