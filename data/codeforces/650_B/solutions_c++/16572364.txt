#include<cstdio>
#include<algorithm>
using namespace std;
#define maxn 2000010
int n,mov,cir,t;
char s[maxn];
int ct[maxn];
int main()
{
	scanf("%d%d%d%d",&n,&mov,&cir,&t);
	scanf("%s",s+1);
	for(int i=1; i<=n; i++)
		ct[i]=(s[i]=='w') ? cir : 0;
	for(int i=n+1; i<=n+n; i++)
		ct[i]=ct[i-n];
	int usd=ct[n+1]+1;
	if(usd>t) {puts("0"); return 0;}
	int lm=n;
	for(; lm>1; lm--)
	{
		usd=usd+ct[lm]+mov+1;
		if(usd>t) break;
	}
	usd=usd-ct[lm]-mov-1;
	lm++;
	int ans=n+1-lm+1;
	for(int i=n+2; i<=n+n; i++)
	{
		usd=usd+ct[i]+mov+1;
		while(usd+1ll*mov*min(i-n-1,n+1-lm)>t)
		{
			usd=usd-ct[lm]-mov-1;
			lm++;
		}
		if(lm>n+1) break;
		ans=max(ans,i-lm+1);
	}
	printf("%d\n",min(ans,n));
	return 0;
}
