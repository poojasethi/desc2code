#include<cstdio>
#include<cstdlib>
#include<cstring>

#include<iostream>
#include<algorithm>
using namespace std;

const int _N=500010;

int n,m,a,b;
char _s[_N];
int s[_N<<1];

int main()
{
    scanf("%d%d%d%d%s",&n,&a,&b,&m,_s);
    static int now;
    for(int i=0;i<n;i++)
        s[i]=s[i+n]=(_s[i]=='w')*b+1,now+=s[i];
    static int L=1,R=n,re=0;
    now-=s[0];
    while(L<=n&&R<2*n)
    {
        now+=s[R++];
        while(R-L>n||now+(R-L-1+min(n-L,R-n-1))*a>m)
            now-=s[L++];
        if(L>R)
            break;
        re=max(re,R-L);
    }
    printf("%d\n",re);
    return 0;
}
