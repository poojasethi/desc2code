#include<cstdio>
#include<algorithm>
using namespace std;
int n,a,b,t;
int ltime[500001];
int rtime[500001];
char str[500005];
int ll,rr;
int main(){
	int dap=0;
	scanf("%d %d %d %d",&n,&a,&b,&t);
	scanf("%s",str);
	ltime[0]=rtime[0]=(str[0]=='w')*b+1;
	t+=ltime[0];
	for(int i=1;i<n;i++){
		ltime[i]=ltime[i-1]+(str[i]=='w')*b+1;
		rtime[i]=rtime[i-1]+(str[n-i]=='w')*b+1;
	}
	ll=0;rr=n-1;
	while(ll<n&&rr>=0){
		//printf("%d %d %d\n",ll,rr,(min(ll,rr)+ll+rr)*a+ltime[ll]+rtime[rr]);
		if((min(ll,rr)+ll+rr)*a+ltime[ll]+rtime[rr]<=t){
			dap=max(dap,ll+rr+1);
			ll++;
		}else{
			rr--;
		}
	}
	dap=min(dap,n);
	printf("%d",dap);
}
