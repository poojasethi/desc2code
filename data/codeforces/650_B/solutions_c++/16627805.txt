#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=2333333;
char a[N],b[N];
int sat[N],koi[N];

int main()
{
	int i,j,k,m,n,ans=0,A,B;
	scanf("%d%d%d%d%s",&n,&A,&B,&m,a);
	m--;
	if(a[0]=='w')m-=B;
	if(m<0)
	{
		printf("0\n");
		return 0;
	}
	sat[0]=koi[0]=0;
	for(i=1;i<n;i++)
	{
		b[i]=a[n-i];
		sat[i]=sat[i-1]+B*(a[i]=='w')+1;
		koi[i]=koi[i-1]+B*(b[i]=='w')+1;
//		printf("i=%d sat[i]=%d koi[i]=%d m=%d\n",i,sat[i],koi[i],m);
		if(sat[i]+i*A<=m || koi[i]+i*A<=m)ans=i;
	}
	j=n;
	for(i=1;i<n;i++)
	{
		for(;j && (i+j>n-1 || sat[i]+koi[j]+min(i,j)*A+i*A+j*A>m);j--);
		if(!j)break;
		ans=max(ans,i+j);
	}
	printf("%d\n",ans+1);
	return 0;
}
