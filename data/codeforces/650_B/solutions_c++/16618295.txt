#include<iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>
#define ll long long
using namespace std;
ll n,a,b,t,pri[500005];
char x[500005];
bool check(int mid)
{
	for(int i=1;i<=mid;++i)
	{
		ll sum1=pri[i]+(pri[n]-pri[n-(mid-i)]),sum2=min(2ll*(i-1)*a+(mid-i)*a,1ll*(i-1)*a+2ll*(mid-i)*a);
		if(sum1+sum2<=t)return 1;
	}
	return 0;
}
int main()
{
	scanf("%I64d%I64d%I64d%I64d%s",&n,&a,&b,&t,x+1);
	for(int i=1;i<=n;++i)pri[i]=pri[i-1]+1+(x[i]=='w'?b:0);
	int l=0,r=n,mid;
	while(l<r)
	{
		mid=(l+r+1)/2;
		if(check(mid))l=mid;
		else r=mid-1;
	}
	printf("%d\n",l);
}
