#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1000010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int f[N],in[N];
char p[N];
int main() {
	scanf("%s",p+1);
	n=strlen(p+1);
	for (i=2,j=0;i<=n;i++) {
		for (;j&&p[j+1]!=p[i];j=f[j]);
		if (p[j+1]==p[i]) j++;
		f[i]=j; in[j]++;
	}
	int A;
	if (in[f[n]]>1) A=f[n];
	else A=f[f[n]];
	if (!A) printf("Just a legend\n");
	else {
		For(i,1,A) printf("%c",p[i]);
		printf("\n");
	}
	//for(;;);
	return 0;
}
