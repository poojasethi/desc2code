#include<cstdio>
#include<cstring>
char s[1000005];
char y[1000005];

int p[1000005];

int main (void)
{
	gets (s+1);
	int n=strlen (s+1);
	p[1]=0;
	int x=0;
	for(int i=2;i<=n;i++){
		while(x>0&&s[x+1]!=s[i])
			x=p[x];
		if(s[x+1]==s[i])
			x++;
		p[i]=x;
	}

	int i=p[n];
	
	memcpy (y,s+1,i);
	while(i){
		y[i]=0;

		x=0;
		for(int j=2;j<n;j++){
			while(x&&s[j]!=y[x])
				x=p[x];
			if(s[j]==y[x])
				x++;
			if(x==i){
				puts (y);
				return 0;
			}
		}
		i=p[i];
	}

	puts ("Just a legend");
	return 0;
}
