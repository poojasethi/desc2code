#include<bits/stdc++.h>
using namespace std;
char s[1000005];
int b[1000005],i,n,j;
main()
{
    scanf("%s",s);
    n=strlen(s);
    for (i=0,j=b[0]=-1;i<n;)
    {
        while (j>=0&&s[i]!=s[j]) j=b[j];
        i++,j++;
        b[i]=j;
    }
    int mx=0;
    for (i=1;i<n;i++)
        mx=max(mx,b[i]);
    j=n;
    while (j>mx) j=b[j];
    s[j]=0;
    if (j==0)
        printf("Just a legend");
    else
        printf("%s",s);
}
