#include <iostream>
#include <algorithm>
using namespace std;

char s[1100000];
int L[1100000];
int Q[1100000];

int main(void) 
{
	int i,j;
	cin.getline(s,1100000);
	L[0]=-1;L[1]=0;
	i=1;
	fill(Q,Q+1000001,0);
	for (char *t=s+1;*t;t++,i++) {
		for (j=L[i];j>=0 && s[j]!=*t;j=L[j]);
		L[i+1]=j+1;
		if (t[1])
			Q[j+1]=1;
	}
	for (j=L[i];j>0 && Q[j]==0;j=L[j]);
	if (j>0)
	{s[j]=0;cout<<s;}
	else 
		cout<<"Just a legend";


	return 0;
}
