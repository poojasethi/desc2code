#include <bits/stdc++.h>

using namespace std;
const int maxn=1e6+100;
int f[maxn],cnt[maxn];
main(){
	string s;
	cin>>s;
	int n=s.size();
    int k=0;
    cnt[0]++;
    for(int i=1;i<n;i++){
        while(k && s[i]!=s[k])k=f[k];
        if(s[i]==s[k])k++;
        f[i+1]=k;
        if(i!=n-1)cnt[k]++;
	}
	int jav=f[n];
	while(jav){
        if(cnt[jav])break;
        jav=f[jav];
    }
    if(!jav){
        cout<<"Just a legend"<<endl;
        return 0;
    }
    for(int i=0;i<jav;i++)
        cout<<s[i];
}
