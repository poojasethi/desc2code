#include<iostream>
#include<algorithm>

using namespace std;

string s;
const int N = 1000000 + 100;
int nex[N], n;

int main (){
    cin >> s;
    n = (int)s.length();
    nex[0] = 0;
    for (int i = 1; i < n; i ++){
        int h = i - 1;
        while (h >= 0 && s[i] != s[nex[h]]) h = nex[h] - 1;
        nex[i] = (h >= 0? nex[h] + 1 : 0);
    }
    int mx = *max_element(nex, nex + n - 1);
    int h = nex[n - 1];
    while (h > mx) h = nex[h - 1];
    cout << (h == 0? "Just a legend" : s.substr(0, h)) << endl;
    return 0;
}
