#include<bits/stdc++.h>
using namespace std;

string ss;
int pf[1000005];
int mxp=0;

void prefix(string &s,int p[])
{
    int i,k=0;
    p[0]=0;
    for(i=1;i<s.size();i++)
    {
        while(k && s[i]!=s[k]) k=p[k-1];
        if(s[i]==s[k]) k++;
        p[i]=k;
        if(i!=s.size()-1) mxp=max(mxp,p[i]);
    }
}

int main()
{
    int i,j,k;
    cin>>ss;
    prefix(ss,pf);
    k=pf[ss.size()-1];
    while(k>mxp  && k) k=pf[k-1];
    if(k==0) cout<<"Just a legend\n";
    else for(i=0;i<k;i++) cout<<ss[i];
    return 0;
}
