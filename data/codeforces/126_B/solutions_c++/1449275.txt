#include <iostream>
#include <vector>
#include <cstdio>

using namespace std;

int zg, n, pi[1111111], mx;
string s;

int main() {
	cin >> s;
	n = s.size();
	s = '$' + s;
	
	for (int i=2; i<=n; i++) {
		while (zg > 0 && s[zg+1] != s[i])
			zg = pi[zg];
		if (s[zg+1] == s[i])
			zg++;
		pi[i] = zg;
		if (i < n)
			mx = max(mx, pi[i]);
	}
	
	zg = pi[n];
	while (zg > mx)
		zg = pi[zg];
	if (zg == 0)
		cout << "Just a legend\n";
	else
		cout << s.substr(1, zg) << endl;
//	system("pause");
}
