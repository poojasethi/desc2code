#include <stdio.h>
#include <string.h>

char dat[1000004];
int pi[1000004];
int ha[1000004];

int main(){
	scanf("%s",dat);
	pi[0] = 0;
	int n = strlen(dat);
	int j = 0;
	for(int i = 1; i < n; i ++) {
		while(j > 0 && dat[j] != dat[i]) j = pi[j-1];
		if(dat[j] == dat[i])  j ++;
		pi[i] = j;
	}
	for(int i = 0; i + 1 < n;i ++) {
		ha[pi[i]] ++;
	}
	j = n;
	while(j>0){
		j = pi[j-1];
		if(j == 0) break;
		if(ha[j] > 0){
			dat[j] = '\0';
			printf("%s\n",dat);
			return 0;
		}
	}
	printf("Just a legend\n");
	return 0;
}
