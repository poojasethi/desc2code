#include <cstdio>
int n, i, j;
char s[1000006];
int b[1000006];
int main() {
    scanf("%s", s);
    for(; s[n]; n++);
    for(i = 0, j = b[0] = -1; i < n; ) {
        while(j >= 0 && s[i] != s[j]) j = b[j];
        i++, j++;
        b[i] = j;
    }
    int mx = 0;
    for(int i = 1; i < n; i++) {
        mx = b[i] > mx? b[i]: mx;
    }
    int j = n;
    while(j > mx) j = b[j];
    s[j] = 0;
    printf(j == 0? "Just a legend": "%s", s);
}
