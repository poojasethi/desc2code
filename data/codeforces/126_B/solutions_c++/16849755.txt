#include <bits/stdc++.h>
#define N 1000001

using namespace std;
int f[N];
char p[N];
int main(){
    scanf ("%s", p);
    int n = strlen(p), l1 = 0, j = 0;
    for (int i = 1; i < n; i++){
        while (p[j] != p[i] && j) j = f[j - 1];
        f[i] = p[j] == p[i]? ++j : 0;
        if (i + 1 != n) l1 = max (l1, f[i]);
    }
    while (j){
        if (j <= l1){
            p[j] = '\0';
            printf ("%s", p);
            return 0;
        }
        j = f[j - 1];
    }
    printf("Just a legend");
    return 0;
}
