#include <stdio.h>
#include <string.h>

char str[1000005];
int f[1000005],a[1000005];

int main()
{
    int i,j,k,l;

    scanf("%s",str+1);

    l=strlen(str+1);

    for(i=2;i<=l;i++)
    {
        for(k=f[i-1];k>0 && str[k+1]!=str[i];k=f[k]) ;

        f[i]=k+(str[i]==str[k+1]);
    }

    for(i=2;i<l;i++) a[f[i]]=1;

    for(i=f[l];i;i=f[i]) if(a[i]) break;

    if(!i) printf("Just a legend\n");

    else
    {
        for(j=1;j<=i;j++) printf("%c",str[j]);
        printf("\n");
    }

    return 0;
}
