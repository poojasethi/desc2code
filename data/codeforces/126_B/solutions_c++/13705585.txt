#include <bits/stdc++.h>
using namespace std;
string s;
int main() {
    ios::sync_with_stdio(false);
    cin>>s;
    vector<int> b(s.size()+1);
    int maximo = 0, i = 0, k = -1;
    b[i] = k;
    while( i < s.size() ) {
        maximo = max(maximo, b[i]);
        while( k >= 0 && s[k] != s[i] ) k = b[k];
        i++; k++;
        b[i] = k;
    }
    if( k > maximo ) k = b[k];
    if( k <= 0 ) cout<<"Just a legend"<<endl;
    else cout<<s.substr(0, k)<<endl;
    return 0;
}
