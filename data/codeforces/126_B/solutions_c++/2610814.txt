#include<cstdio>
#include<cstring>
#include<algorithm>
#define NN 1000010

using namespace std;

char p[NN];
int b[NN], f[NN] ={1}, n;
int solve( ){
	int i = 0 , j = -1  , m = 0 ; 
	b[i] = j;
	while( i < n ){
		while( j >= 0 && p[i] != p[j] ) j =b[j];
		++i; ++j;
		if( i != j && i < n ) f[j] = 1;
		b[i] =j;
	}
	while( !f[j] ) j = b[j];
	return j;
}

int main(){
	scanf("%s" ,p );
	n = strlen(p);
	int sz = solve();
	if( sz == 0 )
		puts("Just a legend");
	else{
		for( int i =0 ; i < sz; ++i )printf("%c",p[i]);
			puts("");
	}	
	return 0;
}
