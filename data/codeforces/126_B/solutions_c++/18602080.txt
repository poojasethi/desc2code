#include<bits/stdc++.h>
using namespace std;
const int N=1000006;
int f[N],lps[N],k,n;
string str;
int main()
{
    cin>>str;
    n=str.length();
    for(int i=1;i<n;i++)
    {
    	while(k && str[k]!=str[i])k=lps[k-1];
    	if(str[i]==str[k])k++;
    	if(i<n-1)f[k]=1;
    	lps[i]=k;
    }
    k=lps[n-1];
    while(k && !f[k])k=lps[k-1];
    if(k)cout<<str.substr(0,k);
    else
    	cout<<"Just a legend";
	return 0;
}
