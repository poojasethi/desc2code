#include<bits/stdc++.h>
#define N 1000010
using namespace std;

int i,j,k,n,m,f[N],in[N];
char p[N];

int main(void)
{
	scanf("%s",p+1);
	n=strlen(p+1);
	for(i=2,j=0;i<=n;i++)
	{
		for(;j&&p[j+1]!=p[i];j=f[j]);
		if(p[j+1]==p[i])j++;
		f[i]=j; in[j]++;
	}
	int A;
	if(in[f[n]]>1)A=f[n];
	else A=f[f[n]];
	if(!A)cout<<"Just a legend";
	else{
		for(i=1;i<=A;i++)cout<<p[i];
	}
	return 0;
}
