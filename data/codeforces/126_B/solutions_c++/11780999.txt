#include<cstring>
#include<iostream>
#include<cstdio>
#define maxn 1000009
using namespace std;
char ch[maxn];
int next[maxn];
int main()
{
	scanf("%s",ch+1);
	next[1]=0;
	int n = strlen(ch+1);
	for(int i=2,j=0;i<=n;i++)
	{
		while(j>0 && ch[j+1] != ch[i])j = next[j];
		if(ch[j+1] == ch[i])j++;
		next[i]=j;
	}
	int u = next[n];
	while(u)
	{
		for(int i=u+1;i<n;i++)if(next[i]==u)
		{
			for(int j=1;j<=u;j++)
			{
				printf("%c",ch[j]);
			}
			puts("");
			return 0;
		}
		u=next[u];
	}
	puts("Just a legend");
	return 0;
}
