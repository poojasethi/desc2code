#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

const int N = 1000010;
char s[N];
int next[N];

int main()
{
  int n;

  gets(s);
  n = strlen(s);
  
  next[0] = -1;
  for (int i = 1, j = -1; i < n; i++)
  {
    for (; j > -1 && s[j + 1] != s[i]; j = next[j]);
    if (s[j + 1] == s[i]) j++;
    next[i] = j;
  }
  
  for (int i = next[n - 1]; i > -1; i = next[i])
    for (int j = i + 1; j < n - 1; j++)
      if (next[j] >= i) {puts(s + n - (i + 1)); return 0;}
  puts("Just a legend");
  return 0;
}
