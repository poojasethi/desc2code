#include<cstring>
#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
const int p=1000000007;
int n,m,ans,v[1100000];
int mu(int a,int b){
    return 1LL*a*b%p;
}
void up(int &a,int b){
    a=(a+b)%p;
    a=(a+p)%p;
}
int power(int a,int b){
    if(!b)return 1;
    int r=power(a,b>>1);
    r=mu(r,r);
    if(b&1)r=mu(r,a);
    return r;
}
int F(int x){
	int z=0,s=power(n,x);
	for(int i=0;i<n;i++){
		up(z,s);
        s=mu(mu(s,v[n+min(0,x-i-2)]),mu(n-i-1,n-i-1));
    }
	return z;
}
int main(){
    //freopen("AC.in","r",stdin);
    //freopen("AC.out","w",stdout);
    cin>>n>>m;
    v[1]=1;
    for(int i=2;i<=n;i++)
        v[i]=mu(v[p%i],(p-p/i));
    ans=F(m+1);
    up(ans,-F(m));
    up(ans,power(n,m));
    cout<<ans;
    return 0;
}
