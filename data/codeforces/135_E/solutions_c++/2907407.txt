#include<cstdio>

typedef long long ll;

const int maxn=1000010;
const int P=1000000007;
int k,w;
int f[maxn],g[maxn];

int pow(int x,int y){
  int e=1;
  while(y){
    if(y&1)e=(ll)e*x%P;
    x=(ll)x*x%P;
    y>>=1;
  }
  return e;
}

int main(){
#ifdef FILE_IO
  freopen("t.in","r",stdin);
  freopen("t.out","w",stdout);
#endif
  int i,d,t,s=0;
  scanf("%d%d",&k,&w);
  for(f[0]=g[0]=1,i=1;i<=k;++i)
    f[i]=(ll)f[i-1]*i%P,g[i]=pow(f[i],P-2);
  for(t=1;t<=k;++t){
    d=(ll)f[k]*g[k-t]%P*g[k-t]%P;
    if(t<=w-1)s=((ll)d*t%P*f[k]%P*pow(k,w-t-1)+s)%P;
    else s=((ll)d*w%P*f[k-t+w-1]+s)%P;
  }
  for(t=1;t<k;++t){
    d=(ll)f[k]*g[k-t]%P*g[k-t-1]%P;
    if(t+1<=w-1)s=((ll)d*t%P*f[k]%P*pow(k,w-t-2)+s)%P;
    else s=((ll)d*(w-1)%P*f[k-t+w-2]+s)%P;
  }
  printf("%d",s);
  return 0;
}
