#include<iostream>
using namespace std;
typedef long long ll;
ll jc[1000010],jv[1000010],w,k,i,ans,mod=1000000007;

ll pow(ll a,ll b)
{
	ll c=1;
	for(;b;b>>=1)
	{
		if(b&1) c=c*a%mod;
		a=a*a%mod;
	}
	return c;
}

int main()
{
	cin>>k>>w;
	for(jc[0]=jv[0]=i=1;i<=k;i++)
	{
		jc[i]=jc[i-1]*i%mod;
		jv[i]=pow(jc[i],mod-2);
	}
	for(i=1;i<=k;i++)
	{
		if(i<=w-1) ans=(ans+jc[k]*jc[k]%mod*i%mod*pow(k,w-1-i)%mod*jv[k-i]%mod*jv[k-i])%mod;
			else ans=(ans+jc[k]*jc[k-i+w-1]%mod*w%mod*jv[k-i]%mod*jv[k-i])%mod;
		if(i<k)
			if(i+1<=w-1) ans=(ans+jc[k]*jc[k]%mod*pow(k,w-i-2)%mod*i%mod*jv[k-i]%mod*jv[k-i-1])%mod;
				else ans=(ans+jc[k]*jc[k-i+w-2]%mod*(w-1)%mod*jv[k-i]%mod*jv[k-i-1])%mod;
	}
	cout<<ans<<endl;
	return 0;
}
