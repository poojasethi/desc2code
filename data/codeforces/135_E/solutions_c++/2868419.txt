const int maxp = 1000000007;
int Nextint(){
	char ch=getchar();
	while (ch<'0'||ch>'9') ch=getchar();
	int x=0;
	while (ch>='0'&&ch<='9'){
		x=x*10+ch-'0';
		ch=getchar();
	}
	return x;
}

int qck(int x,int y){
	int ans=1,t=x;
	for (int i=0;i<31;i++){
		if (y&(1<<i)) ans=(LL)ans*t%maxp;
		t=(LL)t*t%maxp;
	}
	return ans;
}

int N,M;
int Calc(int M){
	int ans=0;
	int ts=1,t1;
	for (int i=1;i<=N;i++)
	if (i<=M){
		ts=(LL)ts*(N-i+1)%maxp;t1=ts;
		ans=(ans+(LL)ts*ts%maxp*qck(N,M-i)%maxp)%maxp;
	}else{
		t1=(LL)t1*qck((N-(i-M)+1),maxp-2)%maxp*(N-i+1)%maxp;
		ts=(LL)ts*(N-i+1)%maxp;
		ans=(ans+(LL)t1*ts%maxp)%maxp;
	}
	return ans;
}
