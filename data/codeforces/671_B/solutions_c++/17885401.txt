#include <bits/stdc++.h>
#define L 500005
#define ll long long
using namespace std ;
ll A[L] ;
int main()
{
    ll sum=0 ;
    int N ; ll K ; scanf("%d %lld",&N,&K) ;
    for(int i=1;i<=N;++i)scanf("%lld",A+i),sum+=A[i] ; sort(A+1,A+N+1) ;
    ll i=1,req=0;
    while(req + (A[i]-A[i-1])*(i-1) <=K && i<=N)
    {
        req += (A[i]-A[i-1])*(i-1) ; ++i ;
    }
    ll mn = A[i-1] + (K-req)/(i-1) ;

    i=N, req=0 ;
    while(req + (A[i]-A[i-1])*(N-i+1) <=K && i>1)
                req = req + (A[i]-A[i-1])*(N-i+1) ,--i ;
    ll mx = A[i] - (K-req)/(N-i+1) ;
    // cout <<"mx == "<<mx<<" mn == "<<mn<<endl ;
    if(mn>=mx) cout<<(int)(sum%N==0 ?0:1)<<endl ;
    else cout<<mx- mn<<"\n" ;
    return 0 ;
}
