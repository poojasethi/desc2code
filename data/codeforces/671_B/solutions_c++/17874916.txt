#include <bits/stdc++.h>
#define N 500010
using namespace std;
typedef long long ll;
int n,k;
int a[N];

void work()
{
	sort(a+1,a+n+1);
	int last=a[1],temp=k;
	for(int i=2;i<=n;i++)
		if((ll)(a[i]-last)*(i-1)<=temp)temp-=(a[i]-last)*(i-1),last=a[i];
	int j=0;
	while(a[j+1]<=last&&j<n)j++;
	int s=temp/j,t=temp%j;
	for(int i=1;i<=j;i++)
		a[i]=last+s+(i<=t);
	for(int i=1;i<=n;i++)
		a[i]=-a[i];
}

int main()
{
//	freopen("tt.in","r",stdin);
	cin>>n>>k;
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	work(),work();
	sort(a+1,a+n+1);
	cout<<a[n]-a[1]<<endl;
	return 0;
}
