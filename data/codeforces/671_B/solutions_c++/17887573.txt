#include <bits/stdc++.h>

using namespace std;

#define N 500005

int a[N];

main() {
    int n, m;
    scanf("%d %d", &n, &m);
    long long sum = 0;
    for (int i = 1; i <= n; i ++) scanf("%d", a + i), sum += a[i];
    int lo = 0, hi = 1000000005;
    while (hi - lo > 1) {
        int mid = lo + hi >> 1;
        long long tot = 0;
        for (int i = 1; i <= n; i ++) tot += max(0, a[i] - mid);
        if (tot <= 1LL * m) hi = mid;
        else lo = mid;
    }
    int R = hi;
    lo = 0, hi = 1000000005;
    while (hi - lo > 1) {
        int mid = lo + hi >> 1;
        long long tot = 0;
        for (int i = 1; i <= n; i ++) tot += max(0, mid - a[i]);
        if (tot <= 1LL * m) lo = mid;
        else hi = mid;
    }
    int L = lo;
    if (L >= R) printf("%d\n", sum % n > 0);
    else printf("%d\n", R - L);
}
