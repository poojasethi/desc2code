#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int c[500005];
ll n,k;
bool check(ll x){
    ll sum=0,sum1=0;
    for(int i=0;i<n;++i)
        if(c[i]>x)sum+=c[i]-x;
        else sum1+=x-c[i];
    return sum<=k&&sum1>=sum;
}
bool check1(ll x){
    ll sum=0,sum1=0;
    for(int i=0;i<n;++i)
        if(c[i]<x)sum+=x-c[i];
        else sum1+=c[i]-x;
    return sum<=k&&sum1>=sum;
}
int main()
{
    cin>>n>>k;
    for(int i=0;i<n;++i){
        scanf("%d",c+i);
    }
    ll st=0,en=1e9+2,mid;
    while(st<en){
        mid=(st+en)/2;
        if(check(mid))en=mid;
        else st=mid+1;
    }
    ll re=en;
    st=0;en=1e9+2;
    while(st<en){
        mid=(st+en)/2;
        if(check1(mid))st=mid+1;
        else en=mid;
    }
    cout<<re-st+1<<endl;

}
