#include<iostream>
#include<algorithm>
using namespace std;
int w[500001];
int main()
{
	ios::sync_with_stdio(false);
	int n,k;cin>>n>>k;
	int q=0;
	for(int i=0;i<n;i++)
	{
		cin>>w[i];q=(q+w[i])%n;
	}
	sort(w,w+n);
	long long a=0,l,r,b=0;
	for(l=1;l<n;l++)
	{
		a+=(w[l]-w[l-1])*l;
		if(a>k)
		{
			a-=(w[l]-w[l-1])*l;
			break;
		}
	}
	l--;
	for(r=n-2;r>=0;r--)
	{
		b+=(w[r+1]-w[r])*(n-1-r);
		if(b>k)
		{
			b-=(w[r+1]-w[r])*(n-1-r);
			break;
		}
	}
	r++;
	int ul=w[l]+(k-a)/(l+1);
	int ur=w[r]-(k-b)/(n-r);
	if(ul>=ur)
	{
		cout<<1-(q==0)<<endl;
	}
	else cout<<ur-ul<<endl;
	return 0;
}
