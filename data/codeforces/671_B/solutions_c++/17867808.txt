#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#define nMax 500010
using namespace std;
int n,k,a[nMax];
long long ans;
int main()
{
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	int l=1,r=1e9;
	while(l<r)
	{
		int mid=(l+r)>>1;
		long long ans=0;
		for(int i=1;i<=n;i++) if(a[i]>mid) ans+=a[i]-mid;
		if(ans>k) l=mid+1; else r=mid;
	} 
	int ls=l;
	l=1; r=1e9;
	while(l<r)
	{
		if(l+1==r)
		{
			long long ans=0;
			for(int i=1;i<=n;i++) if(a[i]<r) ans+=r-a[i];
			if(ans>k) break; else {l=r; break;}
		}
		int mid=(l+r)>>1;
		long long ans=0;
		for(int i=1;i<=n;i++) if(a[i]<mid) ans+=mid-a[i];
		if(ans>k) r=mid-1; else l=mid;
	}
	long long sum=0;
	for(int i=1;i<=n;i++) sum+=a[i];
	printf("%d",max((int)((sum%n)>0),ls-l));
//printf("%d",max(0,ls-l));
	return 0;
}
