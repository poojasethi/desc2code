#include <iostream>
#include <cstdio>
#include <algorithm>
#define MAXN 500005
using namespace std;

char w;
inline void GET(int &t)
{
    do w=getchar();while(w<'0'||w>'9');
    do t=t*10+w-'0', w=getchar();while(w>='0'&&w<='9');
}
int n, k, num[MAXN];

int work()
{
    int day=k, t, maxv=num[n-1], minv=num[0];
    for(int i=n-2;~i;--i)
        if(maxv>num[i])t=min(maxv-num[i],k/(n-i-1)), k-=t*(n-i-1), maxv-=t;
    k=day;
    for(int i=1;i<n;++i)
        if(minv<num[i])
            t=min(num[i]-minv,k/i), k-=t*i, minv+=t;
    return maxv-minv;
}

int main()
{
    int sum=0;
    GET(n), GET(k);
    for(int i=0;i<n;++i)GET(num[i]), sum=(sum+num[i])%n;
    sort(num,num+n);
    printf("%d\n",max(sum?1:0,work()));
    return 0;
}
