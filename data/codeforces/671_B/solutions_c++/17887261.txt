#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

int n,k,a[500005],up,down;
long long s;
inline bool checkup(int x) {
	long long s=0;
	for (int i=1;i<=n;i++) if (a[i]>x) s+=a[i]-x;
	return s<=k;
}
inline bool checkdown(int x) {
	long long s=0;
	for (int i=1;i<=n;i++) if (a[i]<x) s+=x-a[i];
	return s<=k;
}
int main() {
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++) {
		scanf("%d",&a[i]);
		s+=a[i];
	}
	int l=0,r=1000000005;
	while (l<=r) {
		int mid=(l+r)>>1;
		if (checkup(mid)) up=mid,r=mid-1; else l=mid+1;
	}
	l=0; r=1000000005;
	while (l<=r) {
		int mid=(l+r)>>1;
		if (checkdown(mid)) down=mid,l=mid+1; else r=mid-1;
	}
	if (up<=down) {
		if (s%n) printf("1\n"); else printf("0\n");
	} else printf("%d\n",up-down);
	return 0;
}
