#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
#include<vector>
#include<cmath>
#include<map>
using namespace std;
typedef long long ll;
const int N=5e5+1;
ll a[N];
int main()
{
    ll n,k,i,j,sum=0,h=0,t=0;
    cin>>n>>k;
    for(i=0;i<n;i++){scanf("%I64d",a+i);sum+=a[i];}
    sort(a,a+n);
    for(i=0;i<n;i++)
        if(a[i]<sum/n+(sum%n!=0))h+=sum/n+(sum%n!=0)-a[i];
        else break;
    if(h-(i-sum%n)*(sum%n!=0)<=k)cout<<(sum%n!=0);
    else
    {
        h=0;
        for(i=1;i<n;i++)
            if(h+(a[i]-a[i-1])*i<k)h+=(a[i]-a[i-1])*i;
            else {h=a[i-1]+(k-h)/i;break;}
        for(i=n-1;i;i--)
            if(t+(a[i]-a[i-1])*(n-i)<k)t+=(a[i]-a[i-1])*(n-i);
            else {t=a[i]-(k-t)/(n-i);break;}
        printf("%I64d",t-h);
    }
    return 0;
}
