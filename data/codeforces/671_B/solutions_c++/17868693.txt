#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<iostream>
using namespace std;
const int maxn=500000+5;
long long n,k,a[maxn];
int main()
{
    scanf("%I64d%I64d",&n,&k);
    long long sum=0;
    for(int i=0;i<n;i++)
    {
        scanf("%I64d",&a[i]);
        sum+=a[i];
    }
    sort(a,a+n);
    long long x=k,y=k,mn=a[0],mx=a[n-1],l;
    for(int i=0;i<n;i++)
    {
        if(a[i]>mn)
        {
            l=min(a[i]-mn,x/i);
            mn+=l;
            x-=l*i;
        }
    }
    for(int i=n-1;i>=0;i--)
    {
        if(a[i]<mx)
        {
            l=min(mx-a[i],y/(n-i-1));
            mx-=l;
            y-=l*(n-i-1);
        }
    }
    printf("%I64d\n",max(sum%n?1LL:0LL,mx-mn));
    return 0;
}
