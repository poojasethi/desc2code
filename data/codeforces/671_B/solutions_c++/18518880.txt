#define problem "671B"
#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
const int maxN(5e5 + 10);
typedef int i_N[maxN];
typedef long long ll;
int N, K;
i_N A;

int main(){
    ios_base::sync_with_stdio(0);

    cin >> N >> K;
    ll sum(0);
    for(int i = 0; i < N; i++) cin >> A[i], sum += A[i];
    sort(A, A + N, greater<int>());
    ll S(0), P(0);
    int i, j;
    for(i = 0; i < N && S - ll(i) * A[i] <= K; i++) S += A[i];
    for(j = N - 1; j >= 0 && ll(N - j - 1) * A[j] - P <= K; j--) P += A[j];
    if(j < i - 1){
        cout << (sum % N? 1 : 0);
        return 0;
    }
    const ll high((S - K - 1) / i + 1), low((P + K) / (N - j - 1));
    if(high == low) cout << (sum % N? 1 : 0); else cout << high - low;
}
