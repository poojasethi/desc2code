#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<queue>
#include<vector>
using namespace std;

int n,k,l,r,mid,c[500010],ans;
long long s;

int main(){
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++) scanf("%d",c+i);
	l=0; r=(int)1e9+1;
	while(r-l>1){
		mid=(r+l)>>1;
		s=0;
		for(int i=1;i<=n;i++) s+=max(0,mid-c[i]);
		if(s>k) r=mid;
		else l=mid;
	}
	ans=l;
	l=0; r=(int)1e9+1;
	while(r-l>1){
		mid=(r+l)>>1;
		s=0;
		for(int i=1;i<=n;i++) s+=max(0,c[i]-mid);
		if(s>k) l=mid;
		else r=mid;
	}
	s=0;
	for(int i=1;i<=n;i++) s+=c[i];
	ans=r-ans;
	if(ans<=0) ans=s%n!=0;
	printf("%d\n",ans);
	return 0;
}
