from fractions import gcd
b,D=map(int,raw_input().split())
a=200*[0]
z=200*[0]
for d in range(2,101):
	c=0
	t=d
	while gcd(b,t)>1:
		t/=gcd(b,t)
		c+=1
	if t==1:
		a[d]=2
		z[d]=c
		continue
	if b%d==1:
		a[d]=3
		continue
	if (b+1)%d==0:
		a[d]=11
		continue
	a[d]=7
	for i in range(2,d):
		if d%i==0 and gcd(i,d/i)==1 and a[i]!=7 and a[d/i]!=7:
			a[d]=6
print "%d-type"%a[D]
if a[D]==2:
	print z[D]
