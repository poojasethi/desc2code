#include<cstdio>
#include<utility>
using namespace std;
long long b,d;
int check(int b,int d)
{	long long s=1;
	int i;
	if ((b-1)%d==0)
		return 1;
	if ((b+1)%d==0)
		return 1;
	for (i=1;i<=6;++i)
	{	s*=b;
		if (s%d==0)
			return 1;
	}
	return 0;
}
pair<int,int> work()
{	long long s=1;
	int i;
	if ((b-1)%d==0)
		return make_pair(3,0);
	if ((b+1)%d==0)
		return make_pair(11,0);
	for (i=1;i<=6;++i)
	{	s*=b;
		if (s%d==0)
			return make_pair(2,i);
	}
	for (i=2;i<=d;++i)
		if (d%i==0)
		{	s=1;
			while (d%i==0)
				d/=i,s*=i;
			if (!check(b,s))
				return make_pair(7,0);
		}
	return make_pair(6,0);
}
int main()
{	pair<int,int> ans;
	scanf("%d%d",&b,&d);
	ans=work();
	printf("%d-type\n",ans.first);
	if (ans.second)
		printf("%d\n",ans.second);
	return 0;
}
