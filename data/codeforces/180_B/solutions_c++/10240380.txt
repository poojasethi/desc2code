#include <iostream>
#include <vector>
using namespace std;
typedef long long ll;
ll dp[101];
ll pow(ll b, ll j){
	if(j==0)
		return 1;
	return b*pow(b,j-1);
}
int main(){
	ll b,d;
	cin >> b >> d;
	for(ll i = 2; i <= 100; ++i){
		dp[i]=7;
		for(ll j = 1; j < 10; ++j){
			if(pow(b,j)%i==0)
				dp[i]=2;
		}
		if(dp[i]!=7) continue;
		if(b%i==1){
			dp[i]=3;
			continue;
		}
		if(b%i==i-1 && ((i-1)*b)%i==1){
			dp[i]=11;
			continue;
		}
		vector<ll> divs;
		ll m = i;
		for(ll j = 2; j <= 100; ++j){
			ll cur = 1;
			while(m%j==0){
				cur*=j;
				m/=j;
			}
			if(cur>1)
				divs.push_back(cur);
		}
		bool ok = true;
		for(ll j = 0; j < divs.size(); ++j)
			if(dp[divs[j]]==7){
				ok=false;
			}
		if(ok)
			dp[i]=6;
		else
			dp[i]=7;
	}
	cout << dp[d] << "-type\n";
	if(dp[d]==2)
		for(ll i = 0; i < 10; ++i)
			if(pow(b,i)%d==0){
				cout << i << '\n';
				return 0;
			}
}
