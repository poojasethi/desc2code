#include <cstdio>
#include <algorithm>
using namespace std;

int n,m,s;

int main(){
    scanf("%d%d",&n,&m);
    while(__gcd(n,m)>1) m/=__gcd(n,m),s++;
    if(m==1) printf("2-type\n%d\n",s); //daca n=m clar e tipul 2 deoarece ultima cifra si kkt
    else if((n-1)*(n+1)/__gcd(n-1,n+1)%m==0){ //daca avem cazul 3 sau 11 avem solutie, altfel nu
        if(!s && (n-1)%m==0) puts("3-type"); //daca baza-1 e divizibila cu m atunci e cv smen cu suma cifrelor,meh
        else if(!s && (n+1)%m==0) puts("11-type"); //daca baza+1 e divizibila cu m e clar tipul 11 pt ca suma pe pozitii pare si impare kkt
        else puts("6-type"); //daca n si m nu sunt prime intre ele avem mai multe criterii
    }else puts("7-type"); //mda altfel nu exista criteriu
    return 0;
}
