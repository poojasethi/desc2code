#include <stdio.h>

int main()
{
    int b, n, resp, opa;
    long long int pot, aux;
    scanf("%d %d", &b, &n);
    pot=1;
    if(b%2==0) opa=b+1;
    else opa=(b+1)/2;
    for(int i=0; i<8; i++)
    {
        pot*=b;
    }
    if((b-1)%n==0) printf("3-type\n");
    else if(pot%n==0)
    {
        printf("2-type\n");
        resp=1;
        aux=b;
        while(aux%n!=0)
        {
            aux*=b;
            resp++;
        }
        printf("%d\n", resp);
    }
    else if((b+1)%n==0) printf("11-type\n");
    else if(((pot%n)*((b-1)%n)*(opa%n))%n==0) printf("6-type\n");
    else printf("7-type\n");
    return 0;
}
