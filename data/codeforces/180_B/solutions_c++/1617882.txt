#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int b,d,k = 0;

int gcd(int a, int b) { 
  if (a == 0) return b;
  else return gcd(b % a, a);
}

int main() {
  cin >> b >> d;
  while (gcd(b,d) > 1) {
      d = d / gcd(b,d); k++;
  }
  if (d == 1) {
    cout << "2-type" << endl;
    cout << k;
  }
  else if ((((b-1)*(b+1) / gcd((b-1),(b+1))) % d) == 0) {
      if (((b-1) % d == 0) && (k == 0)) cout << "3-type";
      else if (((b+1) % d == 0) && (k == 0)) cout << "11-type";
      else cout << "6-type";
  }
  else cout << "7-type";
}
