#include<cstdio>
int n,m,z=1,w,ans,f3,f11,flag=1;
int main()
{
	scanf("%d%d",&m,&n),f3=m%n==1,f11=m%n==n-1;
	for (int i=2;i<=n;i++) if (n%i==0){
		int c=0,cnt=-1,x=1,y=m; while (n%i==0) cnt++,n/=i,x*=i; while (y%i==0) c++,y/=i;
		if (c) ans=ans>cnt/c+1?ans:cnt/c+1; else {z*=x; if (m%x-1 && m%x-x+1) flag=0;}
		}
	if (ans && z==1) w=2; else if (f3) w=3; else if (f11) w=11; else if (flag) w=6; else w=7;
	printf("%d-type\n",w); if (w==2) printf("%d\n",ans); return 0;
}
