#include <cstdio>

using namespace std;

int b, d;

void init()
{
	scanf("%d%d", &b, &d);
}

int gcd(int x, int y){
	if (!y) return x;
	return gcd(y, x % y);
}

void print()
{
	int cnt = 0;
	while (gcd(b, d) > 1){
		d /= gcd(b, d); cnt ++;
	}
	if (d == 1)
		printf("2-type\n%d\n", cnt);
	else if (!cnt && b % d == 1)
		puts("3-type");
	else if (!cnt && b % d == d-1)
		puts("11-type");
	else if ((b - 1) * (b + 1) / ((b & 1) + 1) % d == 0)
		puts("6-type");
	else
		puts("7-type");
}

int main()
{
	init();
	print();
	return 0;
}
