#include<cstdio>
#define fo(i,a,b) for(i=a;i<=b;++i)
const int mn=201;
int b,d,m,i,x,la;
int ans[mn];
int gcd(int a,int b){return b?gcd(b,a%b):a;}
int main()
{
    scanf("%d%d",&b,&m);
    fo(d,2,m)
    {
        ans[d]=0,x=1;
        fo(i,1,d+2)
        {
            x=x*b%d;
            if(x==0){ans[d]=2;la=i;break;}
        }
        if(ans[d])continue;
        if(b%d==1)ans[d]=3;
        else if(b%d==d-1)ans[d]=11;
        else
        {
            fo(i,2,d-1)
                if(d%i==0&&gcd(i,d/i)==1)
                    if(ans[i]!=7&&ans[d/i]!=7){ans[d]=6;break;}
            if(!ans[d])ans[d]=7;
        }
    }
    printf("%d-type\n",ans[m]);
    if(ans[m]==2)printf("%d\n",la);
    return 0;
}
