#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

int Base;
int Type[101];

int GCD(int A,int B)
{
  return B?GCD(B,A%B):A;
}

int GetType(int D)
{
  if(Base%D==1)return 3;
  if(Base%D==D-1)return 11;
  int S=1;
  for(int i=0;i<100;i++)
    S=S*Base%D;
  if(!S)return 2;
  for(int i=2;i<D;i++)
    if(!(D%i)&&GCD(i,D/i)==1)
      if(Type[i]!=7&&Type[D/i]!=7)
        return 6;
  return 7;
}

void Work(int D)
{
  int S=1,Ans=0;
  while(S){S=S*Base%D;Ans++;}
  printf("%d\n",Ans);
}

int main()
{
  int D;scanf("%d%d",&Base,&D);
  for(int i=2;i<=D;i++)
    Type[i]=GetType(i);
  printf("%d-type\n",Type[D]);
  if(Type[D]==2)Work(D);
  return 0;
}
