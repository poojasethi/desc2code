#include<cstdio>

int n,m,M,K;

int Gcd(int a,int b){return !a?b:Gcd(b%a,a);}

int main()
{
	scanf("%d%d",&n,&m);M=m;
	for(K=0;Gcd(n,m)!=1;K++)m/=Gcd(n,m);
	if(m==1)printf("2-type\n%d\n",K);
	else if(!((n-1)%M))printf("3-type\n");
	else if(!((n+1)%M))printf("11-type\n");
	else
	{
		for(K=1;!(m%2);m/=2)K*=2;
		if(!((1ll*(n-1)*(n+1))%m)&&((!((n-1)%K))||(!((n+1)%K))))printf("6-type\n");
			else printf("7-type\n");
	}
	return 0;
}
