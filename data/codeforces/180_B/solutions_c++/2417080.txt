#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int b, d;

int gcd(int a, int b)
{
	return b ? gcd(b, a % b) : a;
}

int _;
bool p2(int x)
{
	_ = 0;
	while (gcd(x, b) != 1)
		x /= gcd(x, b), ++_;
	return x == 1;
}

bool p3(int x)
{
	return b % x == 1;
}

bool p11(int x)
{
	return b % x == x - 1;
}

bool p6(int x)
{
	int __;
	for (int i = 2; x > 1; ++i)
		if (x % i == 0)
		{
			__ = 1;
			while (x % i == 0)
				x /= i, __ *= i;;
			if (!(p2(__) || p3(__) || p11(__)))
				return 0;
		}
	return 1;
}

int main()
{
	scanf("%d%d", &b, &d);
	if (p2(d))
		printf("2-type\n%d\n", _);
	else if (p3(d))
		printf("3-type\n");
	else if (p11(d))
		printf("11-type\n");
	else if (p6(d))
		printf("6-type\n");
	else
		printf("7-type\n");
	return 0;
}
