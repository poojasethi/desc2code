#include<bits/stdc++.h>
using namespace std;

int n,m,ans;
int f[110],u[110];

int main()
{int i,j,k,t,s1,s2,flag;
 
 for(i=2;i<=100;i++)
 	for(j=2;j*j<=i;j++)
 		if(i%j==0) u[i]=1;
 
 scanf("%d%d",&n,&m);
 for(i=2;i<=m;i++)
 {	f[i]=7;
 	flag=1;
 	for(j=2;j<=i;j++)
 		if(u[j]==0 && i%j==0 && n%j!=0)
 		{	flag=0;
 			break;
 		}
 	if(flag)
		f[i]=2;
	else if(n%i==1)
		f[i]=3;
	else if(n%i==i-1)
		f[i]=11;
	else
	{	flag=1;
		for(j=2;j<=i;j++)
			if(u[j]==0 && i%j==0)
			{	t=i;
				while(t%j==0) t/=j;
				k=i/t;
				if(f[k]==7) flag=0;
			}
		if(flag) f[i]=6;		
 	}
 }
 printf("%d-type\n",f[m]);
 if(f[m]==2)
 {	for(j=2;j<=m;j++)
 		if(u[j]==0 && m%j==0)
		{	s1=0;
			t=m;
			while(t%j==0)
			{	t/=j;
				s1++;
			}
			s2=0;
			t=n;
			while(t%j==0)
			{	t/=j;
				s2++;
			}
			ans=max(ans,(s1-1)/s2+1);
		}
	printf("%d\n",ans);
 }
 return 0;
}
