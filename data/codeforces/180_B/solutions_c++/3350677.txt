#include<stdio.h>
int b,D,n,ans[110],d[110],p[110];
int gcd(int a,int b){return b?gcd(b,a%b):a;}
int work(int x)
{
    if(b%x==1)return 3;
    if(b%x==x-1)return 11;
    int i,t=x;
    for(i=1;i<=n;i++)while(t%d[i]==0)t/=d[i];
    if(t==1)return 2;
    for(i=2;i<x;i++)
        if(x%i==0&&gcd(i,x/i)==1)
            if(ans[i]!=7&&ans[x/i]!=7)return 6;
    return 7;
}
int main()
{
    scanf("%d%d",&b,&D);
    int i,t=b,ret;
    for(i=2;i<=t;i++)if(t%i==0)
    {
        d[++n]=i;
        while(t%i==0)t/=i,p[n]++;
    }
    for(i=2;i<=D;i++)ans[i]=work(i);
    printf("%d-type\n",ans[D]);
    if(ans[D]==2)
    {
        ret=0;
        for(i=1;i<=n;i++)
        {
            t=0;
            while(D%d[i]==0)D/=d[i],t++;
            t=(t+p[i]-1)/p[i];
            if(t>ret)ret=t;
        }
        printf("%d\n",ret);
    }
    return 0;
}
