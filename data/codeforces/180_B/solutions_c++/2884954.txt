#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;
int a,b,c,d;

int gcd(int a,int b){
	return !b?a:gcd(b,a%b);
}

int main(){
	cin>>b>>a;
	for(;d=gcd(a,b),d>1;a/=d,++c);
	if(a==1)cout<<"2-type\n"<<c<<endl;
	else if((b*b-1)/(b&1?2:1)%a==0){
		if(!c && (b-1)%a==0)cout<<"3-type"<<endl;
		else if(!c && (b+1)%a==0)cout<<"11-type"<<endl;
		else cout<<"6-type"<<endl;
	}
	else cout<<"7-type"<<endl;
}
