#include <cstdio>
#include <set>
#include <map>
#include <algorithm>
using namespace std;

int gcd (int x, int y) { return y ? gcd(y, x % y) : x; }

int gtype (int base, int divi, int &num)
{
	if (divi == 1) { num = 0; return 2; }
	set<int> st; st.insert(1);
	for (int i = 1, cr = 1; ; ++i)
	{
		cr = cr * base % divi;
		if (!cr) { num = i; return 2; }
		if (st.find(cr) != st.end()) break;
		st.insert(cr);
	}
	if (st.size() == 1) return 3;
	else if (st.size() == 2 && (*st.begin() + *++st.begin()) % divi == 0 &&
		*st.begin() == 1) return 11;
	for (int i = 2; i < divi; ++i)
		if (divi % i == 0 && gcd(i, divi / i) == 1)
			if (gtype(base, i, num) != 7 && gtype(base, divi / i, num) != 7)
				return 6;
	return 7;	
}

int main ()
{
	int x, y, num, t;
	scanf("%d%d", &x, &y);
	printf("%d-type\n", t = gtype(x, y, num));
	if (t == 2) printf("%d\n", num);
	return 0;
}
