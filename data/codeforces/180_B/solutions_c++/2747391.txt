#include<iostream>
#include<cstdio>
using namespace std;
int x,y,num;
int gcd(int x,int y)
{ return (y==0?x:gcd(y,x%y)); }
int work(int x,int y)
{
	for(;gcd(x,y)!=1;) y/=gcd(x,y),num++;
	if (num==0) num=1;
	return y;
}
int main()
{
//	freopen("1.in","r",stdin);
//	freopen("1.out","w",stdout);
	cin>>x>>y; num=0;
	if (work(x,y)==1) cout<<"2-type"<<endl<<num;	
	else
	if ((x-1)%y==0) cout<<"3-type";
	else
	if ((x+1)%y==0) cout<<"11-type";
	else
	{
		int xx=gcd(x+1,y);
		int yy=gcd(x-1,y);
		y/=xx*yy/gcd(xx,yy);
		if (work(x,y)==1) cout<<"6-type";
		else cout<<"7-type";	
	}
}
