#include <cstdio>
#include <iostream>
#include <algorithm>

int main() {
	int n, m, g, c = 0;
	scanf("%d%d", &n, &m);
	while( (g = std::__gcd(n,m)) > 1 ) {
		m/=g;
		c++;
	}
	
	if(m==1) printf("2-type\n%d",c);
	else if(!c && n%m == 1) puts("3-type");
	else if(!c && n%m == m-1) puts("11-type");
	else if((n*n-1)/(n%2+1)%m == 0) puts("6-type");
	else puts("7-type");
	
	return 0;
}
