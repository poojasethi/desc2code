#include <cstdio>
#include <algorithm>
using namespace std;

int n,m,s;

int main(){
    scanf("%d%d",&n,&m);
    while(__gcd(n,m)>1) m/=__gcd(n,m),s++;
    if(m==1) printf("2-type\n%d\n",s);
    else if((n-1)*(n+1)/__gcd(n-1,n+1)%m==0){
        if(!s && (n-1)%m==0) puts("3-type");
        else if(!s && (n+1)%m==0) puts("11-type");
        else puts("6-type");
    }else puts("7-type");
}
