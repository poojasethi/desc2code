#include<cstdio>
int gcd(int a,int b){
    return b?gcd(b,a%b):a;
}
int get2(int n,int m){
    int ans=0;
    for (;m!=1;ans++){
        int now=gcd(n,m);
        if (now==1) return -1;
        m/=now;
    }
    return ans;
}

    int n,m,i,j,a;

    if ((n-1)%m==0) printf("3-type\n"); else
    if ((n+1)%m==0) printf("11-type\n"); else{
        a=0;
        for (i=1;i<=n-1;i++) if ((n-1)%i==0)
            for (j=1;j<=n+1;j++) if ((n+1)%j==0 && gcd(i,j)==1)
                if (m%i==0 && m%j==0 && get2(n,m/i/j)!=-1) a=1;
        printf("%d-type\n",7-a);
    }
    return 0;
}
