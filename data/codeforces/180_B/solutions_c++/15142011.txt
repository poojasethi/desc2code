#include<cstdio>
int b,d,dp[105],kk,cnt,r[105],x,ok,div;
int main(){
	scanf("%d%d",&b,&d);
	for (int k=1; k<=d; k++){
		dp[k]=0;
		for (int i=0; i<k; i++) r[i]=0;
		cnt=b; r[b%k]=1; x=2;
		while (!r[0]){
			  cnt*=b;
			  cnt=cnt%k;
			  if (r[cnt]) break;
			  r[cnt]=x;
			  ++x;
		}
		if (r[0]) {dp[k]=2; continue;}
		if (b%k==1) {dp[k]=3; continue;}
		if (b%k==k-1){dp[k]=11; continue;}
		ok=1; kk=k; div=0;
		for (int i=2; i*i<=k; i++){
			if (kk%i==0){
			   div=1;
			   x=1;
			   while (kk%i==0) x*=i, kk/=i;
			   if (x==k||dp[x]==7) ok=0;
			}
		}
		if (kk!=1&&kk!=k&&ok) ok=(dp[kk]!=7)?1:0;
		if (ok&&div) dp[k]=6;
		if (!dp[k]) dp[k]=7;
	}
	printf("%d-type\n",dp[d]);
	if (dp[d]==2) printf("%d\n",r[0]);
	return 0;
}
