#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int b,d;
int num(int &x,int y){
	int ans=0;
	while(x%y==0)x/=y,ans++;
	return ans;
}
int type2(int x){
	int ans=0,y=b,i;
	for(i=2;i<=100;++i){
		int u=num(x,i),v=num(y,i);
		if(!u)continue;
		if(!v)return 0;
		ans=max(ans,(u-1)/v+1);
	}
	return ans;
}
bool type3(int d){
	return b%d==1;
}
bool type11(int d){
	return b%d==d-1;
}
int main(){
	scanf("%d%d",&b,&d);
	if(type2(d))printf("2-type\n%d\n",type2(d));
	else if(type3(d)) puts("3-type");
	else if(type11(d)) puts("11-type");
	else {
		for(int i=2;i<=100;++i){
			int x=1;
			for(int j=num(d,i);j>=1;--j)x*=i;
			if(!type2(x) && !type3(x) && !type11(x))return puts("7-type\n"),0;
		}
		puts("6-type");
	}
}
