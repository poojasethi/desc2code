#include <iostream>
#include <algorithm>
using namespace std;
int main ()
{
    long long b,d;
    cin>>b>>d;
    long long x,y,z;
    long long l=__gcd(b,d);
    long long mb=b,md=d;
    int k=0;
    while (__gcd(mb,md)!=1) {
        long long ml=__gcd(mb,md);
        mb/=ml;
        md/=ml;
        mb*=b;
        k++;
    }
    if(md==1){
        cout<<"2-type\n"<<k<<endl;
        return 0;
    }
    if((b-1)%d==0){
        cout<<"3-type\n";
        return 0;
    }
    if((b+1)%d==0){
        cout<<"11-type\n";
        return 0;
    }
    if(((b-1)*(b+1)/__gcd(b-1,b+1))%md==0){
        cout<<"6-type\n";
        return 0;
    }
    cout<<"7-type\n";

    return 0;
}
