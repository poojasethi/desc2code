#include<string.h>
#include<string>
#include<iostream>
#include<stdio.h>
using namespace std;


int main()
{
	int n,k;
	while(cin>>n>>k)
	{
		string ans,t,s;
		cin>>s;
		cin>>t;
		char notuse='a';
		int cnt=0;
		for(int i=0;i<s.size();i++)
		{
			if(s[i]==notuse) 
			{
				notuse++;
				cnt++;
			}
		}
		if(cnt==k) notuse=s[0];
		
		for(int i=0;i<n;i++) ans.push_back(notuse);
		
		for(int i=0;i<t.size();i++)
		{
			if(t[i]=='1')
			{
				for(int j=i,k=0;j<ans.size()&&k<s.size();j++,k++) ans[j]=s[k]; 
			}
		}
		int flag=1;
		for(int i=0;i<t.size();i++)
		{
			if(t[i]=='1')
			{
				if(ans.substr(i,s.size())!=s)
				{
					cout<<"No solution"<<endl;
					flag=0;
					break;
				}
			}
			if(t[i]=='0')
			{
				if(ans.substr(i,s.size())==s)
				{
					cout<<"No solution"<<endl;
					flag=0;
					break;
				}
			}
		}
		if(flag) cout<<ans<<endl;
	}
	return 0;
}
