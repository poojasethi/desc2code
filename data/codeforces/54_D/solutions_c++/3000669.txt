#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
char str[1010];
char p[1010];
char hash[1010];
using namespace std;
int main(){
    int n,k;
    while(cin>>n>>k){
      cin>>p;
      cin>>hash;
      int lp = strlen(p);
      int len  = strlen(hash);
      bool flag = true;
      for(int i = 0 ; i < len;i++){
         if(!flag) break;
         if(hash[i] == '1'){
             for(int j = 0;j < lp;j++){
                if(!str[j + i]) str[j + i] = p[j];
                else if(str[j + i] != p[j]) {
                  flag = false;
                //  cout<<i<<" "<<j<<endl;
                  break;
                }
             }

         }
      }
      if(!flag){
        puts("No solution");continue;
      }
      for(int i = 0;i < len;i++){
         if(hash[i] == '0'){
           int j;
           for( j = 0;j < lp; j++){
             if(str[j+i] && str[j+i] != p[j]) break;
             if(str[j+i] == 0){
               int f;
               for( f = 0; f < k;f++){
                 if(f+'a'!=p[j]) {
                     str[j+i] = f+'a';
                     break;
                 }
               }
               if(f != k ) break;
             }
           }
           if(j == lp) flag = false;
         }
      }
      if(flag){
        for(int i = 0; i < n;i++){
          if(str[i] == 0) str[i] = 'a';

        }
        puts(str);
      }
      else puts("No solution");
    }
    return 0;
}
