int main() {

	//freopen("out.txt", "w", stdout);

	int n, k;
	cin >> n >> k;
	string p, q;
	cin >> p >> q;
	
	bool flag = count(p.begin(), p.end(), p[0]) == p.size();
	char ch = (char) (p[0] - 'a' + 1) % k + 'a';
	string s(n, flag ? ch : p[0]);

	for (int i = 0; i < q.size(); i++) {
		if (q[i] == '1') {
			for (int j = 0; j < p.size(); j++) {
				s[i + j] = p[j];
			}
		}
	}
	for (int i = 0; i < q.size(); i++) {
		string t = s.substr(i, p.size());
		if (q[i] == '1' && t != p ||
			q[i] == '0' && t == p) {
				puts("No solution");
				return 0;
		}
	}
	puts(s.c_str());

	return 0;
}
