#include <iostream>
#include <cstring>
#include <cstdio>
#include <string>
using namespace std;

char name[110];
char app[110];
char out[110];
int exist[50];

main(){
	int n, i, j, K;
	char fill;
	
	scanf ( "%d %d", &n, &K );
	scanf ( "%s", name );
	scanf ( "%s", app );
	memset ( out, 0, sizeof ( out ) );
	for ( i = 0; i < strlen( app ); i ++ )
		if ( app[i] == '1' )
			for ( j = 0; j < strlen( name ); j ++ )
				if ( out[ i + j ] !=0 && out[ i + j ] != name[j] ){
					printf( "No solution\n" );
					return 0;
				}
				else
					out[ i + j ] = name[j];
	memset ( exist, 0, sizeof ( exist ) );
	for ( j = 0; j < strlen( name ); j ++ )
		exist[ name[j] - 'a' ] = 1;
	fill = name[0];
	for ( j = 0; j < K; j ++ )
		if ( exist[j] == 0 )
			fill = 'a' + j;
	for ( i = 0; i < n; i ++ )
		if ( out[i] == 0 )
			out[i] = fill;
	for ( i = 0; i < strlen( app ); i ++ )
		if ( app[i] == '0' ){
			for ( j = 0; j < strlen( name ); j ++ )
				if ( out[ i + j ] != name[j] )
					break;
			if ( j  == strlen( name ) ){
				printf( "No solution\n" );
				return 0;
			}
		}
	printf( "%s\n", out );
//	cin >> K;
	
	return 0;
}
