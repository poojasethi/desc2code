#include<stdio.h>
long n,k,len,i,flag,j,make,bj;
char ch;
long a[10000],b[10000],ans[10000];
int main(void)
{
  scanf("%d%d",&n,&k);
  scanf("%c",&ch);
  scanf("%c",&ch);
  len=0;
  while (ch!='\n')
  {
    len=len+1;
    b[len]=ch-'a'+1;
    scanf("%c",&ch);
  }
  for (i=1;i<=n+1-len;i++)
  {
    scanf("%c",&ch);
    a[i]=ch-'0';
  }
  for (i=1;i<=n;i++)
    ans[i]=0;
  flag=0;
  for (i=1;i<=n+1-len;i++)
    if (a[i]==1)
    {
      for (j=1;j<=len;j++)
      {
        if (ans[i+j-1]==0)
          ans[i+j-1]=b[j];
        else
          if (ans[i+j-1]!=b[j])
          {
            flag=1;
            break;
          }
      }
    }
  for (i=1;i<=n+1-len;i++)
    if (a[i]==0)
    {
      bj=0;
      for (j=1;j<=len;j++)
        if (ans[i+j-1]!=b[j])
        {
          bj=1;
          break;
        }
      if (bj==0)
      {
        flag=1;
        break;
      }
    }
  if (flag==1) printf("No solution\n");
  else
  {
    make=b[(len+1)/2];
    if (make>1) make=1;else make=2;
    for (i=1;i<=n;i++)
      if (ans[i]==0) ans[i]=make;
    for (i=1;i<=n;i++)
      printf("%c",'a'+ans[i]-1);
    printf("\n");
  }
}
