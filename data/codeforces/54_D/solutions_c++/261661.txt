#include <iostream>
#include <string>
using namespace std;

int main()
{
	int n,k;
	cin>>n>>k;
	string p;
	cin>>p;
	char unseen='a';
	for(unsigned i=0;i<p.length();++i)
		if(p[i]==unseen)
			unseen++;
	if(unseen=='a'+k)
		unseen=p[0];
	
	//make the string
	string tomake;
	cin>>tomake;
	string answer;
	while(answer.length()<n)
	{
		string add="9";
		add[0]=unseen;
		answer=answer+add;
	}
	
	for(int i=0;i<tomake.length();++i)
	{
		if(tomake[i]=='1')
		{
			answer=answer.substr(0,i)+p+answer.substr(i+p.length());
		}
	}
	
	//check the answer
	for(int i=0;i<tomake.length();++i)
	{
		if(tomake[i]=='1'&&answer.substr(i,p.length())!=p)
		{
			//cerr<<i<<endl;
			//cerr<<answer.substr(i,p.length())<<endl;
			//cerr<<answer<<endl;
			cout<<"No solution"<<endl;
			return 0;
		}
		else if(tomake[i]=='0'&&answer.substr(i,p.length())==p)
		{
			cout<<"No solution"<<endl;
			return 0;
		}
	}
	cout<<answer<<endl;
	return 0;
}
