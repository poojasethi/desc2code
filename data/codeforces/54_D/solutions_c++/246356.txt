#include<iostream>
#include<string>
using namespace std;
int a[100];
char pol(int k,string s,int sz)
{
     for(int i=0;i<sz;i++)
     a[int(s[i])-97]++;
     for(int i=0;i<k;i++)
     if(a[i]==0)
     return char(i+97);
     return s[sz-1];
}
int check(char a[1000],int n,string s,int sz)
{
    int c=0;
    bool t=true;
    for(int i=0;i<=n-sz;i++){
    t=true;
    for(int j=0;j<sz;j++){
    if(a[i+j]!=s[j]){
    t=false;
    j=sz;}}
    if(t)
    c++;}
    return c;
}
int main()
{
  //  ifstream in("input.txt");
 //   ofstream out("output.txt");
    char c[1000],ans[1000],l;
    int n,k,it=0,s,count=0;
    cin>>n>>k;
    string name;
    cin>>name;
    s=name.size();
    while(cin>>l){
    c[it]=l;
    ans[it]='1';
    it++;}
    while(it!=n){
    c[it]='0';
    ans[it]='1';
    it++;}
    l=pol(k,name,s);
    for(int i=0;i<n;i++){
    if(c[i]=='1'){
    count++;
    for(int j=0;j<s;j++)
    ans[j+i]=name[j];}}
    for(int i=0;i<n;i++)
    if(ans[i]=='1')
    ans[i]=l;
    if(check(ans,n,name,s)!=count){
    cout<<"No solution";}
    else
    for(int i=0;i<n;i++)
    cout<<ans[i];
    return 0;
}
