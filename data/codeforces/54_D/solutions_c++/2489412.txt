#include<cstdio>
#include<cstring>
#define maxn 110
int n,len,k;
char name[maxn],song[maxn];
bool ok[maxn];
bool check(int x){
    int i;
    bool bo=true;
    if(x>=len)
        for(i=len;i;i--){
            if(song[x-i+1]!=name[len-i+1]){
                bo=false;
                break;
            }
        }
    else return true;
    if(bo)return false;
    return true;
}
bool dfs(int x){
    if(x>n)return true;
    if(song[x]!='\0'){
        if(!ok[x]&&!check(x))return false;
        return dfs(x+1);
    }
    for(int i=1;i<=k;i++){
        song[x]=i+'a'-1;
        if(check(x)&&dfs(x))return true;
    }
    return false;
}
int main(){
    int i,j;
    char ch;
    scanf("%d%d",&n,&k);
    scanf("%s\n",name+1);len=strlen(name+1);
    for(i=1;i<=len;i++)if(name[i]-'a'+1<1||name[i]-'a'+1>k){printf("No solution\n");return 0;}
    for(i=1;i<=n-len+1;i++){
        scanf("%c",&ch);
        if(ch=='1'){
            for(j=1;j<=len;j++)
                if(song[i+j-1]!='\0'&&song[i+j-1]!=name[j]){
                    printf("No solution\n");
                    return 0;
                }else song[i+j-1]=name[j];
            ok[i+len-1]=true;
        }
    }
    if(dfs(1))printf("%s\n",song+1);
    else printf("No solution\n");
}
