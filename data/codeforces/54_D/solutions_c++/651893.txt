#include <iostream>
#include <string>
#include <vector>
#include <cmath>

using namespace std;

int n,k;
string p,s;
vector<int> obligado;

void inserta(int pos)
{
  obligado[pos]=1;
  for (int i=0;i<int(p.size());i++)
    s[pos+i]=p[i];
}

bool aparece(int pos)
{
  //if (pos>=int(s.size())-int(p.size())+1 or pos<0) return false;
  for (int i=0;i<int(p.size());i++)
    if (s[pos+i]!=p[i])
      return false;
  return true;
}

bool errorinicial()
{
  for (int i=0;i<n-int(p.size())+1;i++) {
    if (aparece(i)) {
      if (obligado[i]==0) return true;
    } else {
      if (obligado[i]==1) return true;
    }
  }
  return false;
}

bool computa(int pos)
{
  if (pos==n)
    return true;
  if (s[pos]!='$')
    return computa(pos+1);
  for (int i=0;i<k;i++) {
    s[pos]=char('a'+i);
    bool error=false;
    for (int j=max(0,pos-int(p.size())+1);
	 j<=min(pos,n-int(p.size())) and not error;j++)
      error=obligado[j]==0 and aparece(j);
    if (not error)
      if (computa(pos+1))
	return true;
  }
  s[pos]='$';
  return false;
}

int main()
{
  cin>>n>>k>>p;
  s=string(n,'$');
  obligado=vector<int> (n,0);
  for (int i=0;i<n-int(p.size())+1;i++) {
    char c;
    cin>>c;
    if (c=='1') inserta(i);
  }
  if (errorinicial() or not computa(0)) {
    cout<<"No solution"<<endl;
  } else {
    cout<<s<<endl;
  }
}
