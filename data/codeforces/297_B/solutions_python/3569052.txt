from sys import stdin
def main():
    read_ints = lambda: map(int, stdin.readline().split())
    n, m, k = read_ints()
    a = read_ints()
    b = read_ints()
    a.sort(reverse=True)
    b.sort(reverse=True)
    i = 0
    for x in a:
        if i >= m or b[i] < x:
            print "YES"
            return
        i = i + 1
    print "NO" if i >= n else "YES"
main()
