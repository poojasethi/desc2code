def has_advantage(a, b):
    ia = 0
    ib = 0
    while ia < len(a) and ib < len(b):
        ai = a[ia]
        bi = b[ib]
        if ai >= bi: ia += 1
        if bi >= ai: ib += 1
        if ia > ib: return True
    return len(a) > ib

m, n, k = map(int, raw_input().split())
alice = list(map(int, raw_input().split()))
bob = list(map(int, raw_input().split()))
alice.sort(reverse=True)
bob.sort(reverse=True)
print 'YES' if has_advantage(alice, bob) else 'NO'
