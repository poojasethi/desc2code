get_ints = lambda: map(int, raw_input().split())

raw_input()
a = sorted(get_ints())
b = sorted(get_ints())

def main():
    while len(a) > 0 and len(b) > 0:
        af = a.pop()
        bf = b.pop()
        if af > bf:
            print "YES"
            return

    if len(a) == 0:
        print "NO"
    else:
        print "YES"

main()
