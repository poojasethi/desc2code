[n,m,k] = [int(i) for i in raw_input().split()]
a = [int(i) for i in raw_input().split()]
b = [int(i) for i in raw_input().split()]
a.sort()
b.sort()
if n > m:
    print "YES"
else:
    yes = False
    for i in range(n):
        if a[i] > b[m-n+i]:
            yes = True
            break
    if yes:
        print "YES"
    else:
        print "NO"
