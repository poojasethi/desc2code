import sys
import os 

#http://codeforces.com/problemset/problem/297/B
[n,m,k] = [ int(i) for i in sys.stdin.readline().split(" ")]
alice = [int(i) for i in sys.stdin.readline().split(" ")][:n]
bob = [int(i) for i in sys.stdin.readline().split(" ")][:m]

alice.sort(reverse=True)
bob.sort(reverse=True)

def algo(alice,bob):
	if len(alice) > len(bob):
		return True
	for i in range(len(alice)):
		if bob[i] < alice[i]:
			return True
	return False
				
print (algo(alice,bob) and "YES" ) or "NO"
