#!/usr/bin/env python

l = raw_input().split()

n, m, k = int(l[0]), int(l[1]), int(l[2])

m1 = [int(s) for s in raw_input().split()]

m2 = [int(s) for s in raw_input().split()]

m1.sort()
m2.sort()

if n > m:
	print 'YES'
else:
	i, j = m - 1, n - 1
	while j >= 0 and m1[j] <= m2[i]:
		i -= 1
		j -= 1
	if j >= 0:
		print 'YES'
	else:
		print 'NO'
