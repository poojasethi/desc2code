n, m, k = raw_input().split()
n = int(n)
m = int(m)
k = int(k)

alice = raw_input().split()
bob = raw_input().split()

def str_to_int(x):
    ans = []
    for i in x:
        ans.append(int(i))
    return ans

alice = str_to_int(alice)
bob = str_to_int(bob)

alice.sort()
bob.sort()

#print alice
#print bob

flag = False
for i in xrange(1, min(m, n) + 1):
    if alice[n - i] > bob[m - i]:
        flag = True
        break


if n > m or flag:
    print 'YES'
else:
    print 'NO'
