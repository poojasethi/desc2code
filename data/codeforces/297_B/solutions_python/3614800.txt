#!/usr/bin/env python
# -*- coding: UTF-8 -*-

import sys

def ReadIn():
    data = [int(x) for x in sys.stdin.read().split()]
    offset = 0
    while offset < len(data):
        n, m, k = data[offset : offset + 3]
        offset += 3
        alice = data[offset : offset + n]
        offset += n
        bob = data[offset : offset + m]
        offset += m
        yield alice, bob

def Solve(alice, bob):
    cnt = {}
    for x in alice: cnt[x] = cnt.get(x, 0) + 1
    for x in bob: cnt[x] = cnt.get(x, 0) - 1
    s = 0
    for x in sorted(cnt, reverse=True):
        s += cnt[x]
        if s > 0:
            print 'YES'
            return
    print 'NO'

if __name__ == '__main__':
    for alice, bob in ReadIn():
        Solve(alice, bob)
