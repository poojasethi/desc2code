#include <cstdio>
#include <algorithm>
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

int n,m,k,a[100010],b[100010];

int main() {
	scanf("%d%d%d",&n,&m,&k);
	REP(i,n) scanf("%d",&a[i]);sort(a+1,a+n+1);
	REP(i,m) scanf("%d",&b[i]);sort(b+1,b+m+1);
	int now=1;
	REP(i,n) {
		while ((now<=m)&&(a[i]>b[now])) ++now;
		if (now>m) {puts("YES");return 0;}
		++now;
	}
	puts("NO");
	return 0;
}
