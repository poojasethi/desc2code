#include <iostream>
#include <cstdio>
#include <algorithm>

#define N 100100

using namespace std;

int n, m, k, a[N], b[N];

int main(){
    cin >> n >> m >> k;
    for (int i = 0; i < n; i++) cin >> a[i];
    for (int i = 0; i < m; i++) cin >> b[i];
    sort(a, a + n); sort(b, b + m);

    while (n && m && a[n - 1] <= b[m - 1]) n--, m--;

    if (n) cout << "YES"; else cout << "NO";
}
