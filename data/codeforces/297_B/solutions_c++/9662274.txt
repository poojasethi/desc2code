#include<iostream>
#include<algorithm>

using namespace std;

const int MAXN=100000+100;
int a[MAXN];
int b[MAXN];

int main()
{
	int n,m,k;
	cin>>n>>m>>k;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	for(int i=1;i<=m;i++)
		cin>>b[i];
	sort(a+1 , a+n+1);
	sort(b+1 , b+m+1);
	int cnt=m;
	for(int i=n;1<=i;i--)
	{
		if(a[i] > b[cnt])
		{
			cout<<"YES"<<endl;
			return 0;
		}
		cnt--;
	}
	cout<<"NO"<<endl;
	return 0;
}
