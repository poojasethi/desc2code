#include<cstdio>
#include<algorithm>
using namespace std;
int a[100001],b[100001];
bool cmp(int a,int b){return a>b;}
int main()
{
    int i,n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    for(i=1;i<=n;i++) scanf("%d",a+i);
    for(i=1;i<=m;i++) scanf("%d",b+i);
    sort(a+1,a+n+1,cmp);
    sort(b+1,b+m+1,cmp);
    for(i=1;i<=n;i++)
        if(a[i]>b[i]){
            printf("YES\n");
            return 0;
        }
    printf("NO\n");
    return 0;
}
