#include<iostream>
#include<algorithm>
using namespace std;

int a[100010], b[100010];


int main()
{
	int n, m, k;
	cin>>n>>m>>k;
	for(int i=0; i<n; i++)
	cin>>a[i];
	for(int i=0; i<m; i++)
	cin>>b[i];
	sort(a, a+n);
	sort(b, b+m);
	int ai=0, bi=0;
	while(ai<n && bi<m)
	{
		while(bi<m && b[bi]<a[ai])
		bi++;
		if(ai==n-1 && b[bi]>=a[ai])
		{
			cout<<"NO";
			return 0;
		}
		bi++;
		ai++;
	}
	cout<<"YES";
	return 0;
}
