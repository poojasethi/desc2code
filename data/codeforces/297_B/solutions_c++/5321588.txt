#include <iostream>
#include <algorithm>
using namespace std;
long i,j,n,m,k,a[110000]={},b[110000]={};
bool boo=false;
int main() {
    cin >> n >> m >> k;
    for (i=0;i<n;i++) cin >> a[i];
    for (i=0;i<m;i++) cin >> b[i];
    sort(a,a+n);sort(b,b+m);
    for (i=j=0;i<n;i++,j++) {
        while ((b[j]<a[i])&&(j<m)) j++;
        if (j>=m) {
            boo=true;break;
        }
    }
    cout << (boo?"YES":"NO") << endl;
    return 0;
}
