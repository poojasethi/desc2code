#include<bits/stdc++.h>
using namespace std;
int a[100003],b[100003],n,m,k,i;
main()
{
    scanf("%d%d%d",&n,&m,&k);
    for (i=1;i<=n;i++) scanf("%d",&a[i]);
    for (i=1;i<=m;i++) scanf("%d",&b[i]);
    sort(a+1,a+1+n);sort(b+1,b+1+m);
    while (n>0&&m>0)
        if (b[m]>=a[n]) m--,n--;
        else break;
    if (n>0) printf("YES");
    else printf("NO");
}
