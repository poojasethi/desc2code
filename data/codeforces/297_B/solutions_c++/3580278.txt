#include <iostream>
#include <algorithm>
using namespace std;
int a[100100],b[100100];
int n,m,k;
int main()
{
	cin >>n >>m >>k;
	for (int i=0; i<n; i++)
	{
		cin >>a[i];
    }
    for (int j=0; j<m; j++)
    {
    	cin >>b[j];
    }
    int l=max(n,m);
    sort(a,a+l);
    sort(b,b+l);
    for (int k=0; k<l; k++)
    {
    	if (a[k]>b[k]) {cout <<"YES" <<endl; return 0;}
    }
    cout <<"NO" <<endl;
    return 0;
}
