#include <iostream>
#include <algorithm>

using namespace std;

int n,m,k,a[1000001],b[1000001];

int main()
{
	cin>>n>>m>>k;
	for (int i=1; i<=n; i++) cin>>a[i];
	for (int i=1; i<=m; i++) cin>>b[i];
	
	sort(a, a+n+1); sort(b, b+m+1);

	int p1=n, p2=m;
	a[0]=1000000001; b[0]=0;
		while (a[p1]<=b[p2]) {
			p1-=1;
			p2-=1;
		}
		a[0]=0; b[0]=1;	
		if (a[p1]>b[p2]) 
			cout<<"YES"; else cout<<"NO";
	
	
}
