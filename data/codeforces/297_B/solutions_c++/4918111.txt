#include <iostream>
#include <algorithm>
using namespace std;
const int MAXN=100005;
int a[MAXN];
int b[MAXN];
int main(){
    int n,m,k;
    cin >> n >> m;
    cin >> k;
    for(int i=0;i<n;i++)cin >> a[i];
    for(int i=0;i<m;i++)cin >> b[i];
    sort(a,a+n);
    sort(b,b+m);
    for(int i=0,j=0;i<n&&j<m;){
        if(a[i]>b[j])j++;
        else i++;
        if(n-i>m-j){cout << "YES" << endl;return 0;}
    }
    cout <<"NO" << endl;
}
