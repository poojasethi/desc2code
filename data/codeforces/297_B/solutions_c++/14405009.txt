#include <bits/stdc++.h>
using namespace std;
int a[100010] , b[100010];
int main (){
    int n,m,k; cin >> n >> m >> k;
    for(int i = 0 ; i < n ; i++) cin >> a[i];
    for(int i = 0 ; i < m ; i++) cin >> b[i];
    sort(b,b+m); sort(a,a+n); reverse(b,b+m); reverse(a,a+n);
    for(int i = 0 ; i < max(n,m) ; i++){
        if(a[i]>b[i]) return puts("YES\n");
    }
    cout << "NO\n";
}
