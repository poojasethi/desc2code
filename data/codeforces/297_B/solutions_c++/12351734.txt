#include <iostream>
#include <algorithm>
using namespace std;
int n,m,k,a[100005],b[100005];

bool check(){
	while(m&&n&&b[m-1]>=a[n-1])m--,n--;
	if(n&&(!m||a[n-1]>b[m-1]))return true;
	return false;
}

int main(){
	cin>>n>>m>>k;
	for(int i=0;i<n;i++)cin>>a[i];
	sort(a,a+n);
	for(int i=0;i<m;i++)cin>>b[i];
	sort(b,b+m);
	if(check())cout<<"YES";else cout<<"NO";
	return 0;
}
