#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
int i,j,k,l,n,m,a[129999],b[120000];
int main(){
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;++i)scanf("%d",&a[i]);
	for(i=1;i<=m;++i)scanf("%d",&b[i]);
	sort(a+1,a+n+1);
	sort(b+1,b+m+1);
	l=1;
	for(i=1;i<=n;++i){
		while(l<=m && a[i]>b[l])l++;
		if(l>m)break;
		l++;
	}
	if(i==n+1)printf("NO\n");
	else printf("YES\n");
}
