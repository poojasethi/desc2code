#include <cstdio>
#include <algorithm>

using namespace std;

const int MAXN=100005;

int n, m, a[MAXN], b[MAXN];

int main()
{
	scanf("%d%d%*d", &n, &m);
	int i;
	for (i=0; i<n; i++)
		scanf("%d", a+i);
	for (i=0; i<m; i++)
		scanf("%d", b+i);
	int N=max(n,m);
	sort(a, a+N);
	sort(b, b+N);
	for (i=N-1; i>=0; i--)
		if (a[i] > b[i]) {
			puts("YES");
			return 0;
		}
	puts("NO");
	return 0;
}
