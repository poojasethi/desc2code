#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int n,m,k,a[100005],b[100005];
	cin>>n>>m>>k;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<m;i++)
		cin>>b[i];
	if (n>m){
		cout<<"YES";
		return 0;
	}
	sort(a,a+n);
	sort(b,b+m);
	reverse(a,a+n);
	reverse(b,b+m);
	for(int i=0;i<n;i++)
	{
		if (a[i]>b[i]){
			cout<<"YES";
			return 0;
		}
	}
	cout<<"NO";
	return 0;
}
