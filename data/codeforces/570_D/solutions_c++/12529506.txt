#include<bits/stdc++.h>

using namespace std;
#define FOR(i,a,b) for(i = a;i < b; ++i)
#define FORD(i, a, b) for(i = a; i >= b; --i)
#define pb push_back


int n, m;
string str;
vector<int> child[500001], query[500001];
int pre[500001], cur[500001], height[500001];

void bfs(int id, int h)
{
    int i, j, k;
    FOR(i, 0, query[id].size()) {
        j = query[id][i];
        cur[j] ^= pre[height[j]];
    }



    FOR(i, 0, child[id].size()) {
        bfs(child[id][i], h + 1);
    }

    pre[h] ^= (1 << (str[id - 1] - 'a'));
    FOR(i, 0, query[id].size()) {
        j = query[id][i];
        cur[j] ^= pre[height[j]];
    }

}

int main()
{
    scanf("%d%d", &n, &m);
    int i, j, k;
    FOR(i, 0, n - 1) {
        scanf("%d", &j);
        child[j].pb(i + 2);
    }
    cin >> str;
    FOR(i, 0, m) {
        scanf("%d%d", &j, &k);
        query[j].pb(i);

        height[i] = k;
    }
    bfs(1, 1);
    FOR(i,0, m) {
        if((cur[i] & (cur[i] - 1)))
            cout <<"No\n";
        else
            cout <<"Yes\n";

    }
    return 0;
}
