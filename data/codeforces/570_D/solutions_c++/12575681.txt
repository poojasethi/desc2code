#include<stdio.h>
#include<string.h>
#include<vector>
#include<algorithm>
#define maxn 500005
using namespace std;
vector<int> v[maxn][27];
vector<int> vv[maxn];
char s[maxn];
int l[maxn],r[maxn];
int cnt;
void dfs(int u,int dis)
{
    cnt++;
    l[u]=cnt;
    v[dis][s[u]-'a'].push_back(cnt);
    for(int i=0;i<vv[u].size();i++)
        dfs(vv[u][i],dis+1);
    r[u]=cnt;
}
int main()
{
    int n,m;
    while(scanf("%d%d",&n,&m)!=EOF)
    {
        for(int i=2;i<=n;i++)
        {
            int num;
            scanf("%d",&num);
            vv[num].push_back(i);
        }
        scanf("%s",s+1);
        cnt=0;
        dfs(1,1);
        while(m--)
        {
            int x,y;
            scanf("%d%d",&x,&y);
            int px=l[x],py=r[x];
            int ans=0;
            for(int i=0;i<26;i++)
            {
                if(v[y][i].size()==0) continue;
                int num=upper_bound(v[y][i].begin(),v[y][i].end(),py)-lower_bound(v[y][i].begin(),v[y][i].end(),px);
                if(num&1) ans++;
            }
            if(ans<2) puts("Yes");
            else puts("No");
        }
    }
    return 0;
}
