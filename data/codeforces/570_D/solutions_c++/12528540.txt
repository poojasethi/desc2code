#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define all(a) a.begin(), a.end()
#define mod 1000000007
#define endl "\n"

using namespace std;
typedef long long LL;
typedef vector<int> VI;
typedef vector< vector<int> > VVI;
typedef pair< LL, LL > PII;

const int MAXN = 500010;
int n,m;
vector<int> g[MAXN],H[MAXN],idx[MAXN];
int cnt[MAXN],val[MAXN];
string s;

void dfs(int u, int d=1, int par=-1)
{
	for(int i=0; i<H[u].size(); i++)
		cnt[idx[u][i]] = val[H[u][i]];
	val[d] ^= 1<<(s[u-1]-'a');
	for(int i=0; i<g[u].size(); i++)
		if(g[u][i] != par)
			dfs(g[u][i],d+1,u);
	for(int i=0; i<H[u].size(); i++)
		cnt[idx[u][i]] ^= val[H[u][i]];
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n>>m;
	for(int i=2; i<=n; i++)
	{
		int p; cin>>p;
		g[p].pb(i);
	}
	cin>>s;
	for(int i=0; i<m; i++)
	{
		int u,h;
		cin>>u>>h;
		H[u].pb(h); idx[u].pb(i);
	}
	dfs(1);
	for(int i=0; i<m; i++)
		if(__builtin_popcount(cnt[i]) >= 2) cout<<"No\n";
		else cout<<"Yes\n";
	return 0;
}
