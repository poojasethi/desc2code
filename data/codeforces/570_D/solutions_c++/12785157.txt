#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
typedef long long ll;
struct node
{
    int id,depth;
};
int T,t,n,m;
vector<int> e[500010];
vector<node> q[500010];
node A;
char s[500010];
int ret[500010];
int ans[500010];
void dfs(int u,int d)
{
    int i,j,k;
    for(i=0;i<q[u].size();i++)
    {
        ans[q[u][i].id]^=ret[q[u][i].depth];
    }
    for(i=0;i<e[u].size();i++)
    {
        dfs(e[u][i],d+1);
    }
    ret[d]^=1<<(s[u]-'a');
    for(i=0;i<q[u].size();i++)
    {
        ans[q[u][i].id]^=ret[q[u][i].depth];
    }
}
bool check(int k)
{
    while(k%2==0)
      k/=2;
    return k==1;
}
int main()
{
    int i,j,k,u,v;
    scanf("%d%d",&n,&m);
    for(v=2;v<=n;v++)
    {
        scanf("%d",&u);
        e[u].push_back(v);
    }
    scanf("%s",s+1);
    for(i=1;i<=m;i++)
    {
        scanf("%d%d",&u,&A.depth);
        A.id=i;
        q[u].push_back(A);
    }
    dfs(1,1);
    //for(i=1;i<=m;i++)
    //   printf("%d\n",ans[i]);
    for(i=1;i<=m;i++)
    {
        if(ans[i]==0 || check(ans[i]))
          printf("Yes\n");
        else
          printf("No\n");
    }
}
