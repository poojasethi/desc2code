#include <iostream>
#include <stdio.h>
#include <vector>
#include <algorithm>

using namespace std;

const int maxn = 500007;

vector<int> cnt[26][maxn];
vector<int> v[maxn];

char str[maxn]={0};

int n,m,foo,id,h,odd;
int lev[maxn];
int in[maxn],out[maxn];
int cntr;

void dfs(int n,int dep)
{
	in[n]=++cntr;
	cnt[str[n]-'a'][dep].push_back(cntr);
	for(int i=0;i<v[n].size();i++) dfs(v[n][i],dep+1);
	out[n]=cntr;
}
int main()
{
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif
		
	scanf("%d%d",&n,&m);
	for(int i=2;i<=n;i++)
	{
		scanf("%d",&foo);
		v[foo].push_back(i);
	}
	scanf("%s",str+1);
	dfs(1,1);
	for(int i=0;i<m;i++)
	{
		scanf("%d%d",&id,&h);
		odd=0;
		for(int j=0;j<26;j++)
		{
			int flag = lower_bound(cnt[j][h].begin(),cnt[j][h].end(),out[id]+1)-lower_bound(cnt[j][h].begin(),cnt[j][h].end(),in[id]);
			if(flag&1) odd++;
			if(odd>1) break;
		}
		if(odd>1) puts("No");
		else puts("Yes");
	}
	return 0;
}
