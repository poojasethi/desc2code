#include <cstdio>
#include <cmath>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

vector<int> q[500005],g[500005];
int ans[500005],f[500005];
int n,m,x,y;
int h[500005];
char s[500005];
void dfs(int x,int d)
{
	for (int i=0;i<q[x].size();++i)
	{
		int id=q[x][i];
		ans[id]^=f[h[id]];
	}
		f[d]^=1<<(s[x]-'a');
	for (int i=0;i<g[x].size();++i)
	{
		int y=g[x][i];
		dfs(y,d+1);
	}

	for (int i=0;i<q[x].size();++i)
	{
		int id=q[x][i];
		ans[id]^=f[h[id]];
	}
}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=2;i<=n;++i)
	{
		scanf("%d",&x);
		g[x].push_back(i);
	}
	scanf("%s",s+1);
	for (int i=1;i<=m;++i)
	{
		scanf("%d%d",&x,&y);
		q[x].push_back(i);
		h[i]=y;
	}
	dfs(1,1);
	for (int i=1;i<=m;++i)
	{
		if ((ans[i]&(ans[i]-1))!=0) printf("No\n");
		else printf("Yes\n");
	}
	return 0;
}
