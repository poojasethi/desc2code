#include<bits/stdc++.h>
using namespace std;
#define MAXH 500010
#define pb push_back
#define mp make_pair
vector<int> adj[MAXH];
vector<int> a[MAXH][27];
string st;
int ti=0;
int in[MAXH]={0};
int out[MAXH]={0};
void dfs(int idx,int h)
{
	a[h][st[idx-1]-'a'].pb(++ti);
	in[idx]=ti;
	for(int i=0;i<adj[idx].size();i++)
		dfs(adj[idx][i],h+1);
	out[idx]=ti++;
	return ;
}
int main()
{
	int n,m;scanf("%d%d",&n,&m);
	for(int i=2;i<=n;i++)
	{
		int num;scanf("%d",&num);
		adj[num].pb(i);
	}
	cin>>st;
	dfs(1,1);
	for(int i=1;i<=m;i++)
	{
		int v,h;scanf("%d%d",&v,&h);
		int odd=0;
		for(int j=0;j<26;j++)
		{
			int cnt=upper_bound(a[h][j].begin(),a[h][j].end(),out[v])-lower_bound(a[h][j].begin(),a[h][j].end(),in[v]);
			if(cnt&1)
				odd++;
			if(odd>1)
				break;
		}
		if(odd>1)
			printf("No\n");
		else
			printf("Yes\n");
	}
	return 0;
}
