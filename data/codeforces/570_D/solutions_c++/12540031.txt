#include<stdio.h>
#include<string.h>
#include<vector>
#include<iostream>
#include<algorithm>
#include<ctype.h>
#define M 500005
using namespace std;
int ans[M];
vector<int>son[M];
int pre[M];
char str[M];
struct node{
    int id,h;
};
vector<node>query[M];
void Rd(int &res){
    char c; res=0;
    while(c=getchar(),!isdigit(c));
    do{
        res=(res<<3)+(res<<1)+(c^48);
    }while(c=getchar(),isdigit(c));
} 
void dfs(int x,int d){
    for(int i=0;i<query[x].size();i++)
        ans[query[x][i].id]=pre[query[x][i].h];
    for(int i=0;i<son[x].size();i++){
        dfs(son[x][i],d+1);
    }
    pre[d]^=1<<(str[x]-'a');
    for(int i=0;i<query[x].size();i++)
        ans[query[x][i].id]^=pre[query[x][i].h];
}
int main(){
    int n,m,i,j,k,x,h;
    scanf("%d %d",&n,&m);
    for(i=2;i<=n;i++){
        scanf("%d",&k);
        son[k].push_back(i);
    }
    scanf("%s",str+1);
    node tmp;
    for(i=1;i<=m;i++){
        scanf("%d %d",&x,&h);
        tmp.id=i,tmp.h=h;
        query[x].push_back(tmp);
    }
    dfs(1,1);
    for(i=1;i<=m;i++)
        puts(ans[i]&(ans[i]-1)?"No":"Yes");
    return 0;
}
