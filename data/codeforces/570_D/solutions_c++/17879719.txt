char str[N];
int dep[N],que[N],ans[N],x,y,n,m;
vector<int>Q[N],V[N];
void dfs(int x,int der){
	foreach(it,Q[x]) ans[*it] ^= dep[que[*it]];	

	foreach(it,V[x]) dfs(*it,der+1);
	
	dep[der] ^= (1<<(int)(str[x] - 'a'));
	
	foreach(it,Q[x]) ans[*it] ^= dep[que[*it]];
}
int main(){
	cin >> 	n >> m;
	FOR(i,2,n){
		scanf("%d",&x);
		V[x].pb(i);
	}
	scanf("%s",str+1);
	FOR(i,1,m){
		scanf("%d %d",&x,&y);
		Q[x].pb(i);
		que[i] = y;
	}
	dfs(1,1);
	FOR(i,1,m)
		if(__builtin_popcount(ans[i]) <= 1)
			puts("Yes");

}
