#include <bits/stdc++.h>
#define ll long long
using namespace std;
vector <vector <int> > graph;
string sa;
int ti=0;
vector <int> hj[500003][28];
int in[500003];
int out[500003];
void dfs(int s,int h)
{
    in[s]=++ti;
    hj[h][sa[s]-'a'].push_back(ti);
    for(int i=0;i<graph[s].size();i++)
    {
        dfs(graph[s][i],(h+1));
    }
    out[s]=++ti;
}

int main() {
// your code goes here
graph.resize(500003);
    int n,m,a,v,h;
    cin>>n>>m;
    for(int i=0;i<(n-1);i++)
    {
        scanf("%d",&a);
        graph[a-1].push_back(i+1);
    }
    cin>>sa;
    int ct;
    dfs(0,1);
    for(int i=0;i<m;i++)
    {
        int ans=0;
        scanf("%d%d",&v,&h);
        v--;
        for(int j=0;j<26;j++)
        {
        ct=upper_bound(hj[h][j].begin(),hj[h][j].end(),out[v])-lower_bound(hj[h][j].begin(),hj[h][j].end(),in[v]);
        if(ct%2!=0)
        {
            ans++;
            if(ans>1)
            {
                break;
            }
        }
        }
        if(ans>1)
        {
            printf("No\n");
        }
        else
        {
            printf("Yes\n");
        }
    }
return 0;
}
