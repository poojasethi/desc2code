using namespace std;
const int MAXN = 500000 + 10;
int n, m, in[MAXN], out[MAXN], now;
char s[MAXN];
vector<int>go[MAXN];
vector<int>pos[MAXN][26];
int dfs(int u, int h)
{
    pos[h][s[u]-'a'].push_back(++now);
    in[u] = now; int sz = go[u].size();
    for(int i=0;i<sz;i++) dfs(go[u][i], h + 1);
    out[u] = now;
}
int main()
{
    scanf("%d%d", &n, &m);
    int x, v, h;
    for(int i=2;i<=n;i++)
    {
        scanf("%d", &x);
        go[x].push_back(i);
    }
    scanf("%s", s + 1);
    dfs(1, 1);
    while(m--)
    {
        scanf("%d%d", &v, &h);
        int odd = 0;
        for(int i=0;i<26;i++)
        {
            int res = upper_bound(pos[h][i].begin(), pos[h][i].end(), out[v]) -
            lower_bound(pos[h][i].begin(), pos[h][i].end(), in[v]);
            if(res & 1) odd++;
            if(odd > 1) break;
        }
        if(odd > 1) puts("No");
        else puts("Yes");
    }
    return 0;
}
