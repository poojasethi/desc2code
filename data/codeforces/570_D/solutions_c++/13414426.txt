#include<bits/stdc++.h>
using namespace std;
const int maxn = 5000000+5;
#define mp make_pair
typedef pair<int,int> pii;
vector<pii> gao[maxn];
vector<int> G[maxn];
int n,m;
int dfs_clock = 0;
int in[maxn],out[maxn];
char s[maxn];
void dfs(int u,int d)
{
    in[u] = ++dfs_clock;
    gao[d].push_back(mp(dfs_clock,gao[d].back().second^(1<<(s[u]-'a'))));
    for(int i=0;i<G[u].size();i++)
        dfs(G[u][i],d+1);
    out[u] = ++dfs_clock;
}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=2;i<=n;i++)
    {
        int t;
        scanf("%d",&t);
        G[t].push_back(i);
    }
    scanf("%s",s+1);
    for(int i=1;i<=n;i++)
        gao[i].resize(1);
    dfs(1,1);
    int a,b;
    for(int i=0;i<m;i++)
    {
        scanf("%d%d",&a,&b);
        int l = lower_bound(gao[b].begin(),gao[b].end(),mp(in[a],-1))-gao[b].begin()-1;
        int r = lower_bound(gao[b].begin(),gao[b].end(),mp(out[a],-1))-gao[b].begin()-1;
        int t = gao[b][l].second^gao[b][r].second;
        int ok = t - (t&(-t));
        if(!ok)
            puts("Yes");
        else
            puts("No");
    }
    return 0;
}
