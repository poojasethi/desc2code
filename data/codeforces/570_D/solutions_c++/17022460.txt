#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;

vector<int>MAP[500005];
vector<int>CHA[500005][26];
int n,m;
char str[500005];
int in[500005];
int out[500005];
int deep[500005];
int time;


void DFS(int pos,int dep)
{
    ++time;
    in[pos]=time;
    deep[pos]=dep;
    CHA[dep][str[pos]-'a'].push_back(time);
    int sz=MAP[pos].size();
    for(int i=0;i<sz;i++)
        DFS(MAP[pos][i],dep+1);
    out[pos]=time;
}

int main()
{
    scanf("%d %d",&n,&m);
    for(int i=2;i<=n;i++)
    {
        int temp;
        scanf("%d",&temp);
        MAP[temp].push_back(i);
    }
    getchar();
    scanf("%s",str+1);
    DFS(1,1);
    for(int i=0;i<m;i++)
    {
        int num=0;
        int x,y;
        scanf("%d %d",&x,&y);
        int l=in[x];
        int r=out[x];
        for(int j=0;j<26;j++)
        {
            int sb=upper_bound(CHA[y][j].begin(),CHA[y][j].end(),r)-lower_bound(CHA[y][j].begin(),CHA[y][j].end(),l);
            if(sb%2)
                num++;
        }
        if(num<=1)
            printf("Yes\n");
        else
            printf("No\n");
    }
    return 0;
}
