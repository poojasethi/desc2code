#include <iostream>
#include <cstdio>
#include <cstring>
#include <set>
#include <vector>
#include <algorithm>
using namespace std;
const int maxn=500000+100;
vector<int> ve[maxn];
vector<int> q[maxn];
int ds[maxn];
int h[maxn];
int ans[maxn];
char s[maxn];
void dfs(int x,int d)
{
    for(int i=0;i<q[x].size();i++)
    {
        int id=q[x][i];
        ans[id]=ans[id]^ds[h[id]];
    }
    for(int i=0;i<ve[x].size();i++)
    {
        int id=ve[x][i];
        dfs(id,d+1);
    }
    ds[d]=ds[d]^(1<<(s[x]-'a'));
    for(int i=0;i<q[x].size();i++)
    {
        int id=q[x][i];
        ans[id]=ans[id]^ds[h[id]];
    }
}
int main()
{
    int n,m,x;
    scanf("%d%d",&n,&m);
    for(int i=2;i<=n;i++)
    {
        scanf("%d",&x);
        ve[x].push_back(i);
    }
    scanf("%s",s+1);
    int V,H;
    for(int i=1;i<=m;i++)
    {
        scanf("%d%d",&V,&H);
        q[V].push_back(i);
        h[i]=H;
    }
    dfs(1,1);
    for(int i=1;i<=m;i++)
    {
        if(ans[i]&(ans[i]-1))
        {
            printf("No\n");
        }
        else
        {
            printf("Yes\n");
        }
    }
    return 0;
}
