#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;

#define sqr(x) ((x)*(x))

long long a[8][3],b[8][3],l[30];
int ans[8][3];

int ok;

void dfs(int x){
	if (ok) return;
	if (x==8){
		int t=0;
		for (int i=0;i<8;i++)
		for (int j=i+1;j<8;j++)
			l[t++]=sqr(a[i][0]-a[j][0])+sqr(a[i][1]-a[j][1])+sqr(a[i][2]-a[j][2]);
		sort(l,l+28);
		if (l[0]>0 && l[0]==l[11] && l[12]==l[23] && l[0]*2==l[12] && l[24]==l[27] && l[0]*3==l[24]){
			ok=1;
			for (int i=0;i<8;i++)
			for (int j=0;j<3;j++)
				ans[i][j]=a[i][j];
		}
		return;
	}
	a[x][0]=b[x][0]; a[x][1]=b[x][1]; a[x][2]=b[x][2];
	dfs(x+1);
	a[x][0]=b[x][0]; a[x][1]=b[x][2]; a[x][2]=b[x][1];
	dfs(x+1);
	a[x][0]=b[x][1]; a[x][1]=b[x][0]; a[x][2]=b[x][2];
	dfs(x+1);
	a[x][0]=b[x][1]; a[x][1]=b[x][2]; a[x][2]=b[x][0];
	dfs(x+1);
	a[x][0]=b[x][2]; a[x][1]=b[x][0]; a[x][2]=b[x][1];
	dfs(x+1);
	a[x][0]=b[x][2]; a[x][1]=b[x][1]; a[x][2]=b[x][0];
	dfs(x+1);
}

int main(){
	//freopen("1.in","r",stdin);
	//freopen("1.out","w",stdout);
	for (int i=0;i<8;i++)
	for (int j=0;j<3;j++){
		cin>>a[i][j];
		b[i][j]=a[i][j];
	}
	ok=0;
	dfs(1);
	if (ok){
		puts("YES");
		for (int i=0;i<8;i++) cout<<ans[i][0]<<' '<<ans[i][1]<<' '<<ans[i][2]<<endl;
	}
	else puts("NO");
}
