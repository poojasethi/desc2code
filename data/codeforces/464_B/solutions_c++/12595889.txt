#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
#include <cmath>
typedef long long ll;
using namespace std;

ll a[10][10], b[10];

ll dis(int i, int j) {
	ll x = a[i][0] - a[j][0];
	ll y = a[i][1] - a[j][1];
	ll z = a[i][2] - a[j][2];
	return x * x + y * y + z * z;
}

int check() {
	for (int i = 0; i < 8; i++) {
		for (int j = 0; j < 8; j++) {
			if (i == j)
				continue;
			if (j < i)
				b[j] = dis(i, j);
			else b[j-1] = dis(i, j);
		}
		sort(b, b+7);
		ll t1 = b[0], t2 = 2 * b[0], t3 = 3 * b[0];
		if (t1 == 0 || t1 != b[1] || t1 != b[2] || t2 != b[3] || t2 != b[4] || t2 != b[5] || t3 != b[6])
			return 0;
	}
	return 1;
}

int isOk(int cur) {
	if (cur == 8) 
		return check();

	sort(a[cur], a[cur]+3);
	do {
		if (isOk(cur+1))
			return 1;
	} while (next_permutation(a[cur], a[cur]+3));
	return 0;
}

int main() {
	for (int i = 0; i < 8; i++)
		cin >> a[i][0] >> a[i][1] >> a[i][2];

	if (!isOk(0))
		printf("NO\n");
	else {
		printf("YES\n");
		for (int i = 0; i < 8; i++)	{
			for (int j = 0; j < 3; j++)
				cout << a[i][j] << " ";
			printf("\n");
		}
	}
	return 0;
}
