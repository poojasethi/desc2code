#include <bits/stdc++.h>
#define FOR(i,a,b)  for(int i=a; i<=b; i++)
#define maxn 205
#define ll long long
#define ln 1ll * 2e18
using namespace std;
ll a[maxn][5] , b[maxn][4], d[maxn];

ll kc(int p, int q)
{
    ll hx = b[p][1] - b[q][1];
    ll hy = b[p][2] - b[q][2];
    ll hz = b[p][3] - b[q][3];
    return hx*hx + hy*hy + hz*hz;
}

bool check()
{
    ll Mi = ln;
    int sl1 , sl2, sl3, dem;
    sl1 = sl2 = sl3 = dem = 0;
    FOR(i,1,8)
        FOR(j,i+1,8)
            Mi = min( Mi , d[++dem] = kc(i , j) );
    FOR(i,1,dem)
    {
        if(d[i] == Mi) sl1++;
        if(d[i] == 2*Mi) sl2++;
        if(d[i] == 3*Mi) sl3++;
    }
    return sl1 == 12 && sl2 == 12 && sl3 == 4;
}

void inkq()
{
    cout<<"YES"<<endl;
    FOR(i,1,8)
        cout<<b[i][1]<<" "<<b[i][2]<<" "<<b[i][3]<<endl;
}

void DUYET(int x)
{
    if(x > 7)
    {
        if(check())
        {
            inkq();
            exit(0);
        }   return;
    }
    FOR(i,1,3)  FOR(j,1,3)  FOR(k,1,3)
    if(i!=j && j!=k && k!=i)
    {
        b[x][1] = a[x][i]; b[x][2] = a[x][j]; b[x][3] = a[x][k];
        DUYET(x+1);
    }
}

int main()
{
    FOR(i,1,8)  FOR(j,1,3)  cin>>a[i][j];
    b[8][1] = a[8][1]; b[8][2] = a[8][2]; b[8][3] = a[8][3];
    DUYET(1);
    cout<<"NO";
}
