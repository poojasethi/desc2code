#include<iostream>
#include<cmath>
#include<cstring>
#include<string>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define LL long long
using namespace std;
const LL oo=1000000000000000;
int i,x[9],y[9],z[9];

LL dist(LL x,LL y,LL z) {return x*x+y*y+z*z;}

void work(int k)
{
	if (k==9)
	{
		int i,j;
		LL d,l1=oo,l2=oo,l3=oo;
		for (i=1;i<8;i++)
			for (j=i+1;j<=8;j++)
			{
				if (x[j]==x[i] && y[j]==y[i] && z[j]==z[i]) return;
				d=dist(x[j]-x[i],y[j]-y[i],z[j]-z[i]);
				if (d<l1) {l3=l2; l2=l1; l1=d; continue;}
				if (d<l2) {l3=l2; l2=d; continue;}
				if (d<l3) l3=d;
			}
		if (l1!=l2 || l2!=l3 || l1!=l3) return;
		l2*=3;
		for (i=1;i<=8;i++)
		{
			l1=0;
			for (j=1;j<=8;j++)
			{
				d=dist(x[j]-x[i],y[j]-y[i],z[j]-z[i]);
				l1=max(d,l1);
			}
			if (l1!=l2) return;
		}
		printf("YES\n");
		for (i=1;i<=8;i++) printf("%d %d %d\n",x[i],y[i],z[i]);
		exit(0);
	}
	work(k+1);
	swap(y[k],z[k]);
	work(k+1);
	swap(x[k],z[k]);
	work(k+1);
	swap(y[k],z[k]);
	work(k+1);
	swap(x[k],z[k]);
	work(k+1);
	swap(y[k],z[k]);
	work(k+1);
}

int main()
{
	for (i=1;i<=8;i++) scanf("%d%d%d",&x[i],&y[i],&z[i]);
	work(2);
	printf("NO\n");
	return 0;
}
