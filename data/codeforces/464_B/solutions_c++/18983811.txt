#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
ll p[8][3];
ll dis(int i,int j)
{
    ll res=0;
    res+=(p[i][0]-p[j][0])*(p[i][0]-p[j][0]);
    res+=(p[i][1]-p[j][1])*(p[i][1]-p[j][1]);
    res+=(p[i][2]-p[j][2])*(p[i][2]-p[j][2]);
    return res;
}
void ok()
{
    printf("YES\n");
    for(int i=0;i<8;i++)
        printf("%I64d %I64d %I64d\n",p[i][0],p[i][1],p[i][2]);
    exit(0);
}
ll d[28];
void check()
{
    int tot=0;
    for(int i=0;i<8;i++)
        for(int j=i+1;j<8;j++)
            d[tot++]=dis(i,j);
    sort(d,d+28);
    if(d[0]==0)return;
    if(d[0]!=d[11])return;
    if(d[12]!=d[23])return;
    if(d[24]!=d[27])return;
    if(d[12]!=2*d[0])return;
    if(d[24]!=3*d[0])return;
    ok();
}
void dfs(int dep)
{
    if(dep==8)check();
    else
    {
        int cnt=0;
        do{
            dfs(dep+1);
            cnt+=(dep==0);
        }while(cnt<3 && next_permutation(p[dep],p[dep]+3));
    }
}
int main()
{
    for(int i=0;i<8;i++)
        for(int j=0;j<3;j++)
            scanf("%I64d",&p[i][j]);
    for(int i=0;i<8;i++)
        sort(p[i],p[i]+3);
    dfs(0);
    return 0*printf("NO");
}
