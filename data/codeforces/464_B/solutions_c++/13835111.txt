#include<algorithm>
#include<cstdio>
#include<cmath>
#include<iostream>
#include<string.h>
using namespace std;
#define eps 1e-8
#define ll __int64
ll p[10][3];
ll dis(ll x,ll y,ll z)
{
    return x*x+y*y+z*z;
}
int judge(int id)
{
    ll a[20];
    int cnt=0;
    for(int i=0;i<8;i++)
        if(i!=id) a[cnt++]=dis(p[i][0]-p[id][0],p[i][1]-p[id][1],p[i][2]-p[id][2]);
    sort(a,a+7);
    if(a[0]==0||a[0]!=a[1]||a[0]!=a[2]||a[3]!=a[4]||a[3]!=a[5]||a[3]!=2*a[0]||a[6]!=3*a[0])
        return 0;
    return 1;
}
int ok()
{
    for(int i=0;i<8;i++)
        if(!judge(i)) return 0;
    puts("YES");
    for(int i=0;i<8;i++)
        printf("%I64d %I64d %I64d\n",p[i][0],p[i][1],p[i][2]);
    return 1;
}
int dfs(int id)
{
    if(id==8) return ok();
    sort(p[id],p[id]+3);
    do{
        if(dfs(id+1)) return 1;
    }while(next_permutation(p[id],p[id]+3));
    return 0;
}
int main()
{
    for(int i=0;i<8;i++)
        for(int j=0;j<3;j++) scanf("%I64d",&p[i][j]);
    if(!dfs(0)) puts("NO");

}
