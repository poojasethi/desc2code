#include<bits/stdc++.h>
#include<climits>

using namespace std;

typedef long long ll;

ll points[8][3];

bool check()
{
    map<ll,int> cnt;
    for(int i=0;i<7;i++)
        for(int j=i+1;j<8;j++)
        {
            ll l=0;
            for(int k=0;k<3;k++)
                l+=(points[i][k]-points[j][k])*(points[i][k]-points[j][k]);
            cnt[l]++;
        }
    if(cnt.size()!=3)
        return 0;
    ll l=cnt.begin()->first;
    return (l!=0 && cnt[l]==12 && cnt[2*l]==12 && cnt[3*l]==4);
}

bool rec(int i)
{
    if(i==7)
        return check();
    do
    {
        if(rec(i+1))
            return 1;
    }while(next_permutation(points[i],points[i]+3));
    return 0;
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    for(int i=0;i<8;i++)
    {
        for(int j=0;j<3;j++)
            cin>>points[i][j];
        sort(points[i],points[i]+3);
    }
    if(rec(0))
    {
        cout<<"YES\n";
        for(int i=0;i<8;i++)
            for(int j=0;j<3;j++)
                cout<<points[i][j]<<" \n"[j==2];
    }
    else
        cout<<"NO\n";
    return 0;
}
