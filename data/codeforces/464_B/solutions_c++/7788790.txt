ll v[8][3];
ll dist(ll x,ll y,ll z ){
return x*x+y*y+z*z;
}
bool iscube(ll v[8][3]){
ll a=10000000000001;
for(int i=1;i<=7;i++){
    a=min(a,dist(v[0][0]-v[i][0],v[0][1]-v[i][1],v[0][2]-v[i][2]));
}
ll l1=0,l2=0,l3=0;
for(int i=0;i<=7;i++){
    l1=0;
    l2=0;
    l3=0;
for(int j=0;j<=7;j++){
ll r=dist(v[j][0]-v[i][0],v[j][1]-v[i][1],v[j][2]-v[i][2]); 
if(r==2*a) l2++;
if(r==a) l1++;
if(r==3*a) l3++;
}
if(!(l1==3&&l2==3&&l3==1)) return 0;
}
return 1;
}
