#include<stdio.h>
#include<algorithm>
using namespace std;
int a[8][3],ans;
bool judg()
{
	int i,j,k,cnt[4]={0};
	long long dis[28],len=1e14;
	for(i=k=0;i<8;i++)
		for(j=0;j<i;j++)
			len=min(len,dis[k++]=((long long)a[i][0]-a[j][0])*(a[i][0]-a[j][0])+((long long)a[i][1]-a[j][1])*(a[i][1]-a[j][1])+((long long)a[i][2]-a[j][2])*(a[i][2]-a[j][2]));
	if(!len) return 0;
	for(i=0;i<k;i++)
		if(len==dis[i]) cnt[1]++;
		else if(len*2==dis[i]) cnt[2]++;
		else if(len*3==dis[i]) cnt[3]++;
		else return 0;
		//if(dis[i]/len*len==dis[i]) return 0;
		//else if(dis[i]/len<4) cnt[dis[i]/len]++;
	if(cnt[1]!=12||cnt[2]!=12||cnt[3]!=4) return 0;
	return 1;
}
bool dfs(int x)
{
	if(x==8) return judg();
	if(dfs(x+1)) return 1;
	swap(a[x][1],a[x][2]);
	if(dfs(x+1)) return 1;
	swap(a[x][0],a[x][2]);
	if(dfs(x+1)) return 1;
	swap(a[x][1],a[x][2]);
	if(dfs(x+1)) return 1;
	swap(a[x][0],a[x][2]);
	if(dfs(x+1)) return 1;
	swap(a[x][1],a[x][2]);
	if(dfs(x+1)) return 1;
	swap(a[x][0],a[x][2]);
	return 0;
}
