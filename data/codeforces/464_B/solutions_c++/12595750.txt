#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
ll a[10][4];
ll b[10];
bool vis[5];
ll dis(int i,int j){
    ll x=a[i][0]-a[j][0],y=a[i][1]-a[j][1],z=a[i][2]-a[j][2];
    return x*x+y*y+z*z;
}
bool check(){
    for(int u=0;u<8;u++){
        for(int v=0;v<8;v++){
            if(v==u) continue;
            if(v<u) b[v]=dis(u,v);
            if(v>u) b[v-1]=dis(u,v);
        }
        sort(b,b+7);
        ll t=b[0];
        if(t==0||t!=b[1]||t!=b[2]||2*t!=b[3]||2*t!=b[4]||2*t!=b[5]||3*t!=b[6]) return false;
    }
    return true;
}
bool ok(int cur) {
    if (cur == 8) return check();
    sort(a[cur], a[cur]+3);
    do{
        if (ok(cur+1)) return 1;
    } while (next_permutation(a[cur], a[cur]+3));
    return 0;
}
int main(){
    for(int i=0;i<8;i++) cin>>a[i][0]>>a[i][1]>>a[i][2];
    if(ok(0)){
        puts("YES");
        for(int i=0;i<8;i++,puts(""))
            for(int j=0;j<3;j++) printf("%d ",a[i][j]);
    }
    else puts("NO");
    return 0;
}
