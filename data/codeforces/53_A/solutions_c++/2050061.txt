#include<iostream>
#include<string>
using namespace std;

int main()
{
    string s,tmp,min("z");
    cin>>s;
    int n,size=s.size();
    cin>>n;
    while( n-- ) { cin>>tmp; if( s == tmp.substr(0,size) && tmp<min) {min=tmp;}}
    if(min == "z")cout<<s<<endl;
    else
    cout<<min<<endl;
    return 0;
}
