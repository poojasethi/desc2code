#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	string w;
	cin >> w;

	int n;
	cin >> n;

	string t[100];
	for (int i = 0; i < n; i++) {
		cin >> t[i];
	}
	sort(t, t+n);

	for (int i = 0; i < n; i++) {
		if (t[i].substr(0, w.size()) == w) {
			cout << t[i] << endl;
			return 0;
		}
	}
	cout << w << endl;
	return 0;
}
