#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll i,j,m,n,p,q,r;
string s,ss;
vector<string>v;
int main()
{
    cin>>s>>n;
    for(i=1;i<=n;i++)
    {
        cin>>ss;
        if(ss.find(s)==0)
            v.push_back(ss);
    }
    sort(v.begin(),v.end());
    if(v.size()==0)
        cout<<s;
    else
        cout<<v[0];
}
