#include <iostream>
#include <string>
using namespace std;
int main()
{
    string s,str,lexi ; cin >> s ;
    int n,check=1 ; cin >> n ;
    lexi=s;
    while (n--){
        cin >> str ;
        if (check) lexi=str,check=0;
        if (str.find(s)==0 && lexi>str ) lexi=str;
    }
    cout << (lexi.find(s)==0  ? lexi : s) ;
    return 0;
}
