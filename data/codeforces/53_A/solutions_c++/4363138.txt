#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
string s,a[105];
int i,j,n;
cin>>s;
cin>>n;
for(i=0;i<n;i++)
{
cin>>a[i];
}
sort(a,a+n);
for(i=0;i<n;i++)
{
for(j=0;j<s.length();j++)
{
if(s[j]!=a[i][j])
{
break;
}
}
if(j==s.length())
{
cout<<a[i];
return 0;
}
}
cout<<s;
return 0;
}
