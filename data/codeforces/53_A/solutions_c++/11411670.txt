#include <bits/stdc++.h>
using namespace std;

string S[105];
int main(){

    string s;
    cin>>s;
    int n; cin>>n;
    for(int i=0;i<n;i++)cin>>S[i];
    sort(S,S+n);

    for(int i=0;i<n;i++){
        if(s==S[i].substr(0,(int)s.size())){
            cout<<S[i]<<"\n";
            return 0;
        }
    }

    cout<<s<<"\n";
}
