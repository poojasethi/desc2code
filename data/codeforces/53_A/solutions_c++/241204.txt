#include<iostream>
#include<string>

using namespace std;

int main()
{
	string s,sm="";
	cin>>s;
	int n;
	cin>>n;
	for(int i=0;i<n;++i)
	{
		string sx;
		cin>>sx;
		if( sx.length()>=s.length() )
		{
			if(sx.substr(0,s.length())==s)
			{
				if(sm=="" || sx<sm )
				{
					sm=sx;
				};
			};
		};
	};
	if(sm=="") sm=s;
	cout<<sm;
	return 0;
};
