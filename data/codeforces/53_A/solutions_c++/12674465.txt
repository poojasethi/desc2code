#include<bits/stdc++.h>
using namespace std;

int main()
{
    string s,a,ans="z";
    cin >> s;
    int n;
    scanf("%d",&n);
    while(n--)
    {
        cin >> a;
        if(a.substr(0,s.size()) == s && ans > a)
        {
            ans = a;
        }
    }
    if(ans == "z") cout << s;
    else cout << ans;
}
