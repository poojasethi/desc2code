#include <iostream>
#include <string>
using namespace std;
string s,s1,ans;
bool prefix(){
	return (s1.find(s) == 0);
}
int t;
int main() {
	cin >> s >> t;
	for(int i = 0; i < t;i++){
		cin >> s1;
		if(prefix() && (ans == "" || s1 < ans))
			ans = s1;
	}
	if(ans == "")cout<<s<<endl;else
	cout << ans << endl;
	return 0;
}
