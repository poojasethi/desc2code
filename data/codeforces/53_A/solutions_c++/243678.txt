#include <algorithm>
#include <iostream>
#include <string>
using namespace std;
int main() {
    string s, t, u;
    int N;
    cin >> s;
    cin >> N;
    for (int i=0; i<N; ++i) {
	cin >> u;
	if (u.size() >= s.size() && equal(s.begin(), s.end(), u.begin())
	    && (t.empty() || u < t)) t = u;
    }
    cout << (t.empty() ? s : t) << endl;
}
