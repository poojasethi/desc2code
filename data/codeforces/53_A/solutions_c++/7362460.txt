#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int main() {
	string s;
	cin >> s;
	int n;
	cin >> n;
	string a[n];
	for (int i = 0; i < n; ++i) {
		cin >> a[i];
	}
	sort(a, a + n);
	for (int i = 0; i < n; ++i) {
		if (a[i].find(s) < s.length()) {
			cout << a[i];
			return 0;
		}
	}
	cout << s;
}
