#include <bits/stdc++.h>
using namespace std;
int main()
{
    string s;
    int n;
    cin>>s>>n;
    int len=s.size();
    string ans=s;
int flag=1;
    while(n--)
{
    string a;
    cin>>a;
if(a.size()<len)    continue;
    if(a.substr(0,len)==s)
{
    if(flag)    ans=a;
    else if(ans>a)  ans=a;
    flag=0;
}
}
cout<<ans;
     return 0;
}
