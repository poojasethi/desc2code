#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <cstring>
#include <string>
using namespace std;
string a[105];
int main(){
	string s;
	cin>>s;
	int n;
	cin>>n;
	for (int i=0;i<n;i++){
		cin>>a[i];
	}
	sort(a,a+n);
	string x=*lower_bound(a,a+n,s);
	if (x.substr(0,s.size())==s)cout<<x;
	else cout<<s;
	return 0;
}
