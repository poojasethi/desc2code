#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
	string s, str[100];
	int n, i;
	cin>>s;
	cin>>n;
	for(i = 0; i < n; i++)
		cin>>str[i];
	
	sort(str, str + n);
	for(i = 0; i < n; i++)
		if(str[i].substr(0, s.size()) == s) break;
	if(i < n)
		cout<<str[i]<<endl;
	else
		cout<<s<<endl;

	return 0;
}
