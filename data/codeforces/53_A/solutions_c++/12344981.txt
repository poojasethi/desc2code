#include <iostream>
#include <string>
using namespace std;

int n;
string s, t, res;

int main(){
	ios_base::sync_with_stdio(0);
	cin >> s >> n;
	while (n--){
		cin >> t;
		if (t.length() >= s.length() && t.substr(0, s.length()) == s)
		if (res == "" || res > t)
			res = t;
	}

	if (res == "")
		cout << s;
	else
		cout << res;

	return 0;
}
