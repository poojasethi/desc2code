#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	string s; cin >> s;
	int n; cin >> n;
	
	vector<string> a(n);
	for(int i=0; i<n; i++) cin >> a[i];
	
	sort(a.begin(), a.end());
	
	for(int i=0; i<n; i++) 
		if(a[i].find(s) == 0) {
			cout << a[i];
			return 0;
		} 
	
	cout << s;
	
	return 0;
}
