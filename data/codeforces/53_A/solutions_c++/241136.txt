#include<iostream>
#include<vector>
#include<string>
#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
string s,s1,res;
int n,p,l;
vector<string> v;
cin>>s;
scanf("%d",&n);
for(int i=0;i<n;i++)
{
cin>>s1;
p=s1.find(s);
if((p+l)==l) v.push_back(s1);
}
sort(v.begin(),v.end());
if(v.size()==0) res=s;
else res=v[0];
cout<<res; 
}
