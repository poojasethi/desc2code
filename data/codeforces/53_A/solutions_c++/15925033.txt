#include <bits/stdc++.h>
#include <vector>
using namespace std;

int main(){
	vector<string>v;
	string s1,s2;
	int n,i,p;
	cin>>s1;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>s2;
		p=s2.find(s1);
		if(p==0){
			v.push_back(s2);
		}
	}
	if(v.size()==0){
		cout<<s1<<endl;
	}
	else{
		sort(v.begin(),v.end());
		cout<<v[0]<<endl;
	}

	return 0;
}
