#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	string s[100], a;
	int n, i;
	cin >> a >> n;
	for(i=0; i<n; i++){
		cin >> s[i];
	}
	sort(s,s+n);
	for (i=0; i<n; i++){
		if(s[i].substr(0,a.size())==a){
			cout<<s[i]<<endl;
			return 0;
		}
	}
	cout << a << endl;
	return 0;
}
