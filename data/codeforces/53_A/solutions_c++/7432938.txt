#include<bits/stdc++.h>
using namespace std;

string ss[100+10];

int main()
{
    string s;
    int n;
    cin>>s>>n;
    for(int i=0;i<n;i++)
        cin>>ss[i];
    sort(ss,ss+n);
     string x=*lower_bound(ss,ss+n,s);
    if(x.substr(0,s.size())==s)puts(x.c_str());
    else puts(s.c_str());

    return 0;
}
