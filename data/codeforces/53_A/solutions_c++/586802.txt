#include <iostream>
#include <string>

using namespace std;

int main(){
	string a;
	cin>>a;
	int n;
	cin>>n;
	bool check=true;
	string x[n];
	for(int j=0;j<n;j++){
		cin>>x[j];
	}
	string Mins=x[0];
	for(int i=0;i<n;i++){
		if(x[i].find(a)==0){
			if(Mins>x[i]){
				Mins=x[i];
			}
			check=false;
		}
				
	}
	if(check==false){
		cout<<Mins;
	}
	else{
		cout<<a;
	}
}
