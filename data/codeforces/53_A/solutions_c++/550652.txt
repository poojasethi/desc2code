#include <iostream>

using namespace std;

int main(int argc, char* argv[])
{
	string s, m;
	int n;
	cin >> s >> n;
	
	for (int i = 0; i < n; i++)
	{
		string p;
		cin >> p;
		
		if (p.find(s) == 0)
		{
			if (m.empty() || p < m)
				m = p;
		}
	}
	
	if (!m.empty())
		cout << m << endl;
	else
		cout << s << endl;
	
	return 0;
}
