#include <bits/stdc++.h>
using namespace std;
int main()
{
    string s , adress , ans = "z";
    int i , n;
    cin >> s >> n;
    for(i = 0 ; i < n ; i++)
    {
        cin >> adress;
        if(adress.substr(0,s.size()) == s && adress < ans) ans = adress;
    }
    if(ans == "z") cout << s;
    else cout << ans;
    return 0;
}
