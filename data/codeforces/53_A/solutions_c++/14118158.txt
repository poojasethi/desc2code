#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s,s2,s3;
	int n,k;
	cin>>s;
	k=0;
	s2=s;
	cin>>n;
	for(int j=0;j<n;j++)
	{
		cin>>s3;
		int i;
		for(i=0;i<s.length();i++)
		{
			if(s[i]!=s3[i])break;
		}
		if(i==s.length())
		{
			if(k==0||s3<s2){
			s2=s3; k=1;	
			}
		}
	}
	cout<<s2;
}
