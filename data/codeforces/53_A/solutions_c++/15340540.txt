#include<iostream>
#include<string>
#include<algorithm>
using namespace std;

string w, s[111], ans;
int n;

int main() {
	cin>>w>>n;
	for(int i=0;i<n;i++) cin>>s[i];
	sort(s,s+n);
	
	ans = w;
	for(int i=0;i<n;i++)
		if(s[i].length() >= w.length() && w==s[i].substr(0, w.length())) {
			ans = s[i];
			break;
		}
	cout<<ans<<endl;
	
	return 0;
}
