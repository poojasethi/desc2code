from sys import stdin


def next():
	return stdin.readline().strip()

def main():
	s, n = next(), int(next())
	has = False
	l = []
	
	for i in range(n):
		st = next()
		
		if st.startswith(s):
			has = True
			l.append(st)
	
	if has == False:
		print s
	else:
		l.sort()
		print l[0]

if __name__ == "__main__":
	main()
