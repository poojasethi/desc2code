s = raw_input()
n = int(raw_input())
i = 0
goodl = ''
length = len(s)
while i < n:
	#Proverim, ravna li stroka nachalu kakoi - to stroki.
	s1 = raw_input()
	if s1[:length] == s:
		if goodl != '':
			if goodl > s1:
				goodl = s1;
		else:
			goodl = s1;
	i += 1
print(goodl)
if goodl == '':
	print(s)
