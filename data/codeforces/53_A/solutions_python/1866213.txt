import sys

file = sys.stdin
#file = open("test", "r")

input = file.readline().rstrip()
k = int(file.readline().rstrip())
dict = []
for _ in range(k):
    line = file.readline().rstrip()
    if line.find(input) == 0: dict.append(line)

dict.sort()
if dict == []:
    print input
else:
    print dict[0]
