
#define MAXN 10000010UL
#include <cstdio>
#define ll long long
#define Mod 1000000007UL

using namespace std;

int u[MAXN];
bool vis[MAXN];
int pow[MAXN], pr[MAXN], num[MAXN];

void Pre(ll r) {
	u[1] = 1;
	for(ll i = 2 ; i <= r ; ++ i) {
		if(!vis[i]) {
			pr[++ pr[0]] = i;
			u[i] = -1;
		}
		for(ll j = 1 ; j <= pr[0] && i*pr[j] <= r ; ++ j) {
			vis[i*(ll)pr[j]] = true;
			if(!(i%pr[j])) {
				u[i*(ll)pr[j]] = 0;
				break;
			}
			u[i*(ll)pr[j]] = -u[i];
		}
	}
	return;
}

int main() {
	Pre(10000000);
	ll n, x, N = MAXN-10;
	scanf("%I64d", &n);
	pow[0] = 1;
	for(int i = 1 ; i <= n ; ++ i) scanf("%I64d", &x), ++ num[x];
	for(int i = 1 ; i <= N ; ++ i) pow[i] = (pow[i-1]*2ll)%Mod;
	ll ans = 0;
	for(int i = 2 ; i <= N ; ++ i) {
		if(u[i]==0) continue;
		ll cnt = 0;
		for(int j = i ; j <= N ; j += i) cnt += num[j];
		if(!cnt) continue;
		ll nw = (pow[cnt]-1)*(ll)(n-cnt)%Mod;
		if(u[i]==-1) ans = (ans+nw)%Mod;
		else  ans = (ans-nw+Mod)%Mod;
	}
	printf("%I64d", ans);
	return 0;
}
