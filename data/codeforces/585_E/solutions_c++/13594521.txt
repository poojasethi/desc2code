#include<iostream>
#include<cstdio>
#include<map>
#include<vector>
using namespace std;

#define GC getchar
#define MOD 1000000007
#define MAX 10000005
inline void getnum(int& Number){
    register char ch=GC();
    Number=0;
    while( ch>'9' || ch<'0') ch=GC();
    while( ch>='0' && ch<='9'){
        Number=(Number<<3)+(Number<<1)+ch-'0';
        ch=GC();
    }
}


long long pow( int a, int b ){
    if(!b) return 1;
    long long tmp = pow(a,b/2);
    tmp = (tmp*tmp)%MOD;
    if(b&1)
        return (tmp*a)%MOD;
    return tmp;
}

int cnt[MAX], u[MAX], n;

void solve(){
    long long ans=0;
    for( int i=2; i<MAX; i++ ){
        u[i]++;
        for( int j=2*i; j<MAX; j+=i )
            u[j] -= u[i];
    }
    for( int i=2; i<MAX; i++ ){
        if( u[i] != 0 ){
            int c=0;
            for( int j=i; j<MAX; j+=i ) c+=cnt[j];
            if( u[i] == 1 ){
                ans = (ans + ( (n-c)*( (pow(2,c)-1+MOD)%MOD ) )%MOD +MOD) %MOD;
            }
            else
                ans = (ans - ( (n-c)*( (pow(2,c)-1+MOD)%MOD ) )%MOD +MOD) %MOD;
        }
    }
    cout << ans;
}

int main(){
    int x;
    getnum(n);
    for( int i=0; i<n; i++ ){
        getnum(x);
        cnt[x]++;
    }
    solve();
    return 0;
}
