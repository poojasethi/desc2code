#include <bits/stdc++.h>
#define MAX 10000002
#define MOD 1000000007
#define N 500005

using namespace std;

int mobius[MAX],cnt[MAX],multi[N];

void calc(){
	mobius[1]=1;
	for(int i=1;i<MAX;i++)
		for(int j=i+i;j<MAX;j+=i){
			mobius[j]-=mobius[i];
			cnt[i]+=cnt[j];
		}
}

int main(){
	int n,x,ans;
	multi[0]=1;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		multi[i]=(2*multi[i-1])%MOD;
	for(int i=1;i<=n;i++){
		scanf("%d",&x);
		cnt[x]++;
	}
	ans=0;
	calc();
	for(int i=1;i<MAX;i++){
		if(cnt[i]==0 || mobius[i]==0) continue;
		ans += mobius[i]*(multi[cnt[i]]-1)*(long long)(cnt[i]-n)%MOD;
		if(ans >= MOD) ans -= MOD;
		if(ans < 0) ans += MOD;
	}	
	printf("%d\n",ans);
	return 0;
}
