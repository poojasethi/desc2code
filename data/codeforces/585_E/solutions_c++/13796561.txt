#define ll long long
#define maxn 10000010
#define modd 1000000007
#define inf 0x7fffffff
#define eps 1e-7
#define imax(x,y)(x>y)?x:y
#define imin(x,y)(x<y)?x:y

using namespace std;

int n,cnt[maxn]={0},mobius[maxn]={0},maxx=1;
ll two[maxn]={0};

int main()
{
      scanf("%d",&n);
      for (int i=0;i<n;i++)
      {
      int x;
      scanf("%d",&x);
      cnt[x]++;
      maxx=imax(maxx,x);
      }
      mobius[1]=two[0]=1;
      for (int i=1;i<=n;i++)
      {
      two[i]=two[i-1]<<1;
      if (two[i]>=modd)two[i]-=modd;
      }
      for (int i=1;i<=maxx;i++)
      for (int j=i+i;j<=maxx;j+=i)
      {
          mobius[j]-=mobius[i];
          cnt[i]+=cnt[j];
      }
      ll ans=0;
      for (int i=2;i<=maxx;i++)
      if (mobius[i]&&cnt[i])
      {
          ans+=((-mobius[i])*((two[cnt[i]]-1)*(long long)(n-cnt[i])%modd));
          if (ans>=modd)ans-=modd;
          if (ans<0)ans+=modd;
      }
      printf("%lld",ans);
      return 0;
}
