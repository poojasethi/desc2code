#include <cstdio>
#include <cstdlib>

#define N 500005
#define M 10000005

const int MOD = 1000000007;

int n;
int p[M], cp, minp[M];
int val[1 << 10], sgn[M], c[M];
int pw[N];

main() {
	for (int i = 2; i < M; i ++) {
		if (!minp[i]) p[cp ++] = i, minp[i] = i;
		for (int j = 0; j < cp && 1LL * p[j] * i < M; j ++) {
			minp[i * p[j]] = p[j];
			if (i % p[j] == 0) break;
		}
	}
	scanf("%d", &n);
	pw[0] = 1;
	for (int i = 1; i <= n; i ++) {
		pw[i] = pw[i - 1] * 2 % MOD;
        int x, cnt = 0;
		scanf("%d", &x);
		val[cnt ++] = 1, sgn[1] = -1;
		while (x > 1) {
			int pp = minp[x];
			while (x % pp == 0) x /= pp;
			for (int i = 0; i < cnt; i ++) {
				val[cnt + i] = val[i] * pp;
				sgn[val[cnt + i]] = -sgn[val[i]];
				c[val[cnt + i]] ++;
			}
			cnt += cnt;
		}
	}
	int rlt = 0;
	for (int i = 2; i < M; i ++) if (sgn[i]) {
		rlt = (rlt + 1LL * sgn[i] * (pw[c[i]] - 1) * (n - c[i])) % MOD;
	}
	if (rlt < 0) rlt += MOD;
	printf("%d\n", rlt);
}
