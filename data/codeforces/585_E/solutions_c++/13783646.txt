#include <bits/stdc++.h>
#define MAX 10000005
#define MOD 1000000007
#define N 500005

using namespace std;

int mobius[MAX],cnt[MAX],combination[N];

void calc(){
	mobius[1]=1;
	for(int i=1;i<MAX;i++)
		for(int j=i+i;j<MAX;j+=i){
			mobius[j]-=mobius[i];
			cnt[i]+=cnt[j];
		}
}

int main(){
	int n,x,ans;
	scanf("%d",&n);
	combination[0]=1;
	for(int i=1;i<=n;i++)
		combination[i]=(2*combination[i-1])%MOD;
	for(int i=1;i<=n;i++){
		scanf("%d",&x);
		cnt[x]++;
	}
	calc();
	ans=0;
	for(int i=1;i<MAX;i++){
		if(cnt[i]==0 || mobius[i]==0) continue;
		ans += (-mobius[i])*(combination[cnt[i]]-1)*(long long)(n-cnt[i])%MOD;
		if(ans >= MOD) ans -= MOD;
		if(ans < 0) ans += MOD;
	}	
	printf("%d\n",ans);
	return 0;
}
