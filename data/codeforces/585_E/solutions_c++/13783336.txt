#include <bits/stdc++.h>
#define Maxn 10000002
#define M 1000000007

using namespace std;

int miu[Maxn],cnt[Maxn];
int n,x,ans;
int xp[500020];

int main()
{
	xp[0]=1;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		xp[i]=(2*xp[i-1])%M;
	for(int i=1;i<=n;i++){
		scanf("%d",&x);
		cnt[x]++;
	}
	ans=0;
	miu[1]=1;
	for(int i=1;i<Maxn;i++){
		for(int j=i+i;j<Maxn;j+=i){
			miu[j]-=miu[i];
			cnt[i]+=cnt[j];
		}
		if(cnt[i]==0 || miu[i]==0) continue;
		ans += miu[i]*(xp[cnt[i]]-1)*(long long)(cnt[i]-n)%M;
		if(ans >= M) ans -= M;
		if(ans < 0) ans += M;
	}	
	printf("%d\n",ans);
	return 0;
}
