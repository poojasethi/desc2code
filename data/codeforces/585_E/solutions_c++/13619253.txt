template <class T,class U>inline void Max(T &a,U b){if(a<b)a=b;}
template <class T,class U>inline void Min(T &a,U b){if(a>b)a=b;}

const int maxn=(1e+7)+5;
int f[maxn],a[N],p[N],vis[maxn],pp[maxn],q[maxn],d[maxn];
vi v;
int main(){
    //ios::sync_with_stdio(0);
    int T,i,j,k,n,m=0;
    int t=0;
    for(i=2;i<maxn;i++){
        if(!pp[i])q[t++]=i,d[i]=i;
        for(j=0;j<t;j++){
            if(1LL*i*q[j]>=maxn)break;
            pp[i*q[j]]=1;
            if(i%q[j]==0){
                d[i*q[j]]=d[i];
                break;
            }
            d[i*q[j]]=q[j];
        }
    }
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
        int x=a[i];v.clear();
        while(x>1){
            int y=d[x];while(x>1&&d[x]==y)x/=y;
            v.pb(y);
        }
        //sort(all(v));v.erase(unique(all(v)),v.end());
        m=v.size();
        for(j=0;j<1<<m;j++){
            int q=1,p=0;
            for(k=0;k<m;k++)if(j>>k&1)q*=v[k],p++;
            f[q]++;vis[q]=p&1?1:-1;
        }
    }
    p[0]=1;
    for(i=1;i<=n;i++)p[i]=p[i-1]*2LL%mod;
    ll ret=0;
    for(i=1;i<maxn;i++)if(vis[i])ret+=1LL*(n-f[i])*(p[f[i]]-1)%mod*vis[i];
    ret%=mod;
    if(ret<0)ret+=mod;
    printf("%d\n",ret);
    return 0;
}
