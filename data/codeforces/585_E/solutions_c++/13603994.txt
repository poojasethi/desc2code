#include <iostream>
#include <cstdio>
#include <map>
#include <algorithm>
#include <vector>
#include <cstring>
using namespace std;

const int maxn = 10000007;

const int mod = 1000000007;

int mob[maxn], cnt[maxn], val[maxn];
bool vis[maxn];

typedef long long ll;
ll qp[maxn];

int main()
{
    int i,j;
    for(i = 2; i < maxn; i++) {
        val[i] = i;mob[i] = 1;
    }
    for(i = 2; i < maxn; i++) {
        if(vis[i]) continue;
        for(j = 1; i * j < maxn; j++) {
            vis[i * j] = true; mob[i * j] = -mob[i * j];
            val[i * j] /= i;
        }
    }
    for(i = 2; i < maxn; i++) if(val[i] != 1) mob[i] = 0;
 //   for(i = 2; i < 30; i++) cout << "mob["<<i<<"]:"<<mob[i]<<endl;
    qp[0] = 1; for(i = 1; i < maxn; i++) qp[i] = (qp[i-1] << 1) % mod;
    ll ans = 0;
    int n, x; cin >> n;
    for(i = 1; i <= n; i++) {
        scanf("%d",&x); cnt[x] ++;
    }
    for(i = 2; i < maxn; i++) {
        if(mob[i] == 0) continue;
        for(j = 2; i * j < maxn; j++)
            cnt[i] += cnt[i * j];
    }
    for(i = 2; i < maxn; i++) {
        ans = (ans + mob[i] * (qp[cnt[i]] - 1) * (cnt[i] - n))% mod;
        if(ans < 0) ans += mod;
 //       if(i < 10) cout<<"i:" <<i<<"   "<<ans<<endl;
    }
    cout << ans << endl;
}
