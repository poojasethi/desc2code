def f(s1, s2):
    c = 0
    for i in xrange(len(s1)):
        if s1[i] != s2[i]:
            c += 1
    return (c + 1) // 2

n = input()
p = [raw_input() for i in xrange(n)]

m = 7

for i in xrange(0, n - 1):
    for j in xrange(i + 1, n):
        m  = min(m, f(p[i], p[j]))
                    
print(m - 1)
