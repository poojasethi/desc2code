import os

def cmp(word1,word2):
	cnt = 0
	for i in xrange(6):
		if word2[i] != word1[i]:
			cnt+=1
	return cnt

seqList = []
n = int(raw_input())
for i in xrange(n):
	seqList.append(raw_input().strip())

ans = 6
for i in xrange(n):
	for j in xrange(i+1,n):
		word1 = seqList[i]
		word2 = seqList[j]
		diff = cmp(word1,word2)
		ans = min(ans,(diff-1)/2)
		if ans == 0:
			break

print ans
