# -*- coding: utf-8 -*-

class Coders(object):
	def __init__(self):
		self.n = input()
		self.L = []
		for i in range(self.n):
			self.L.append(raw_input())
	def cal(self):
		res = 6
		n, L = self.n, self.L
		for i in range(n):
			for j in range(i):
				cnt = 0
				for k in range(6):
					if L[i][k] != L[j][k]:
						cnt += 1
				res = min(res, (cnt-1)/2)
				if res == 0:
					return res
		return res

if __name__ == '__main__':
	coders = Coders()
	print coders.cal()
