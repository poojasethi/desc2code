#include <iostream>
#include <set>

using namespace std;
string s[1005];

int main()
{int n; cin>>n;
 if (n==1) {cout<<6; return 0;}
 
 for (int i=1;i<=n;i++) cin>>s[i];
 int ans=6;
 for (int i=1;i<=n;i++)
    for (int j=i+1;j<=n;j++)
 {int k=0;
  for (int i1=0;i1<6;i1++) k+=(s[i][i1]!=s[j][i1]);
  ans=min(ans,k);
 }
ans=(ans-1)/2;
cout<<ans;
    return 0;
}
