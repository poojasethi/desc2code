#include <iostream>
#include <set>

using namespace std;

int main()
{string d[1006];
set<string> s;
int n;cin>>n;
for(int i=1;i<=n;i++)cin>>d[i];
int ans=6;
for(int i=1;i<=n;i++)
  for(int j=i+1;j<=n;j++)
{int k=0;
 for(int i1=0;i1<6;i1++)
   k+=(d[i][i1]!=d[j][i1]);
 ans=min(ans,k);
}
if(n==1){cout<<6;return 0;}
cout<<(ans-1)/2;
 return 0;

}
