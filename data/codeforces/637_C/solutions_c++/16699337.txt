#include <bits/stdc++.h>
using namespace std;

string s[1000];

int main()
{
    int n,res=6;
    cin>>n;
    for (int i=0;i<n;i++)
        cin >> s[i];
    for (int i=0;i<n;i++)
    {
        for (int j=0;j<i;j++)
        {
            int cnt=0;
            for (int k=0;k<6;k++)
                cnt+=(s[i][k]!=s[j][k]);
            res=min(res,(cnt+1)/2-1);
        }
    }
    cout << res << "\n";
    return 0;
}
