#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int rec[1500][7];
int n;
int main()
{
	while(~scanf("%d",&n))
	{
		char str[7];
		for(int i=1;i<=n;i++){
			scanf("%s",str);
			for(int j=1;j<=6;j++)
				rec[i][j]=str[j-1]-'0';
		}
		int M=6;
		for(int i=1;i<=n;i++)
			for(int j=i+1;j<=n;j++)
			{
				int cnt=0;
				for(int k=1;k<=6;k++)
					if(rec[i][k]!=rec[j][k])cnt++;
				M=min(M,(cnt-1)/2);
			}
		printf("%d\n",M);
	}
	return 0;
}
