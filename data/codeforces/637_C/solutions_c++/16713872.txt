#include <iostream>

using namespace std;

int main() {
    int n, t, m = 14;
    cin >> n;
    string a[n];
    for (int i = 0; i < n; cin >> a[i++]);
    for (int i = 0; i < n - 1; i ++)
        for (int k = i + 1; k < n; k++) {
            t = 0;
            for (int j = 0; j < 6; j++)
                t += (a[i][j] != a[k][j]);
            m = (t < m ? t : m);
        }
    cout << (m + 1) / 2 - 1;
    return 0;
}
