#include <cstdio>
#include <cstring>
int n;
char s[1010][10];
int main() {
	scanf("%d",&n);
	for (int i=0;i<n;i++) scanf("%s",s[i]);
	if (n==1) {
		puts("6");
		return 0;
	}
	int c=6;
	for (int i=0;i<n;i++) for (int j=i+1;j<n;j++) {
		int t=0;
		for (int k=0;k<6;k++) t+=(s[i][k]!=s[j][k]);
		if (t<c) c=t;
	}
	printf("%d\n",(c-1)/2);
}
