#include <iostream>
#include <string>
using namespace std;

int cmp(string a, string b)
{
	int res = 6;
	for (int i = 0; i < 6; ++i)
		if (a[i] == b[i]) --res;
	return (res - 1) / 2;
}

int main()
{
	int n;
	cin >> n;
	string* a = new string[n];
	for (int i = 0; i < n; ++i)
		cin >> a[i];
	int res = 6;
	for (int i = 0; i < n; ++i)
	{
		for (int k = i + 1; k < n; ++k)
			res = min(res, cmp(a[i], a[k]));
			if (res == 0)
			{
				cout << 0;
				return 0;
			}
	}
	cout << res;
}
