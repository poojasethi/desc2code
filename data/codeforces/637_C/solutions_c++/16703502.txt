#include <iostream>

using namespace std;

int main(){
    int n;
    string s[1000];
    cin >> n;
    int m = 6;
    for(int i = 0; i < n; ++i){
        cin >> s[i];
    }
    if(n == 1){
        cout << 6;
        return 0;
    }
    for(int i = 0; i < n; ++i){
        for(int j = i+1; j < n; ++j){
            int k = 0;
            for(int l = 0; l < 6; ++l){
                if(s[i][l] != s[j][l])
                    k++;
            }
            m = min(m, k);
        }
    }
    cout << (m-1)/2;
}
