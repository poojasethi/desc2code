#include <bits/stdc++.h>
using namespace std;
#define mx 1001
int n, i,j,pos,k=13,t;
string codes[mx];

int main() {
  ios_base::sync_with_stdio(false);
  cin >> n;
  
  for(i=0;i<n;++i) 
  cin >> codes[i];
  
  for(i=0;i<n;i++)
  {
    for(j=i+1;j<n;j++)
    {
      t=0;
      for(pos=0;pos<6;++pos)
        if(codes[i][pos]!=codes[j][pos])
        t++;
        
      k = min(k, t);
    }
  }        
  cout << (k-1)/2<<endl;
  
  return 0;
}
