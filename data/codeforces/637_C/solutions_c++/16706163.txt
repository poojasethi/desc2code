#include <cstdio>

char a[1010][10];

int main()
{
    int j, n, i, min=6, r;
    scanf("%d", &n);
    for (int i=0; i<n; i++) scanf ("\n%c%c%c%c%c%c", &a[i][0], &a[i][1], &a[i][2], &a[i][3], &a[i][4], &a[i][5]);
    for (i=0; i<n-1; i++) {
        for (j=i+1; j<n; j++) {
            r=0;
            for (int z=0; z<6; z++) if (a[i][z]!=a[j][z]) r++;
            if (r<min) min=r;
        }
    }
    if (n==1) r=6;
    else r=(min+1)/2-1;
    printf("%d", r);
    return 0;
}
