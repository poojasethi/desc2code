#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    char s[1001][6];
    scanf("%d", &n);
    int m = 6;
    for (int i = 0; i < n; ++i)
        scanf("%s", s[i]);
    for (int i = 0; i < n; ++i)
        for (int j = i + 1; j < n; ++j) {
            int cnt = 0;
            for (int k = 0; k < 6; ++k)
                if (s[i][k] != s[j][k]) cnt++;
            m = min(m, (cnt - 1)/ 2 );
            m = max(m, 0);
        }


    printf("%d\n", m);

    return 0;
}
