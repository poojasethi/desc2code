#include<cstdio>
#include<cstring>
#include<bitset>

using namespace std;

char s[1005][10];
int n;

int main(){
    int i,j,k;
    scanf("%d",&n);
    for(i = 0;i < n;i++) scanf("%s",s[i]);
    int ans = 7,cnt;
    for(i = 0;i < n;i++){
        for(j = i+1;j < n;j++){
            for(k = cnt = 0;k < 6;k++) cnt += s[i][k]!=s[j][k];
            ans = min(ans,(cnt+1)/2);
        }
    }
    printf("%d\n",ans-1);
    return 0;
}
