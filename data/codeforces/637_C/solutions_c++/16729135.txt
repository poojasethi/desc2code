#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int main() {
    int n, ans = 6;
    char s[1005][7];
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%s", s[i]);
    }
    for (int i = 0; i < n; i++) {
        for (int j = i+1; j < n; j++) {
            int cnt = 0;
            for (int k = 0; k < 6; k++) {
                if (s[i][k] != s[j][k]) cnt++;
            }
            ans = min(ans, (cnt+1)/2-1);
        }
    }
    printf("%d\n", ans);
    return 0;
}
