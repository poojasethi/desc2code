#include<bits/stdc++.h>
using namespace std;
string s[1005];
int main()
{
   int n;
   cin>>n;
   for(int i=0; i<n; i++) cin>>s[i];
   int m=0, kk=6;
   for(int i=0; i<n-1; i++){
      for(int j=i+1; j<n; j++){
         m=0;
         for(int k=0; k<6; k++){
            if(s[i][k]!=s[j][k]) m++;
         }
         kk=min(kk, m);
      }
   }
   if(n>1) cout<<max(0, (kk+1)/2-1)<<endl;
   else cout<<kk<<endl;
}
