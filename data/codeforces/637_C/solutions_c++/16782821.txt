#include<algorithm>
#include<iostream>
#include<string>
#include<cstdio>
#include<stdlib.h>
using namespace std;
int main()
{
	int n,i,j,k,temp,ans;
	string s[1001];
	cin>>n;
	ans=6;
	for(i=0;i<n;i++)
	{
		cin>>s[i];
	}
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			temp=0;
			for(k=0;k<6;k++)
			{
				if(s[i][k]!=s[j][k])
					temp++;
			}
			ans=min(ans,(temp-1)/2);
		}
	}
	cout<<ans<<endl;
	return 0;
}
