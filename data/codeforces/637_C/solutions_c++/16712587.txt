#include <iostream>
#include <cmath>
#include <string>

using namespace std;

int main () {
	int n, m = 6, l;
	cin >> n;
	string s[n];
	
	for (int i = 0; i < n; i ++){
		cin >> s[i];
		
		for (int j = 0; j < i; j ++){
			l = 0;
			for (int k = 0; k < 6; k ++)
				if (s[j][k] != s[i][k])
					l ++;
		
			m = min ((l - 1) / 2, m);
		}
	}
	
	cout << m;
	
	return 0;
}
