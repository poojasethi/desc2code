# include <bits/stdc++.h>
  using namespace std;
  string srr[1010];
  int main()
  {
      int n, ans, i, j, k, l;
      cin >> n;
      for(i = 1; i <= n; i++) cin >> srr[i];
      ans = 6;
      for(i = 1; i <= n; i++){
        for(j = i + 1; j <= n; j++){
            l = 0;
            for(k = 0; k < 6; k++){
                l += (srr[i][k] != srr[j][k]);
            }
            ans = min(ans, (l / 3) + ((l % 3 != 0) * (l > 4)));
        }
      }
      cout << ans;
      return 0;
  }
