#include <stdio.h>
#include<iostream>
#include<string>
#include<cmath>
#include<algorithm>
using namespace std;

string s[1020];
int main(void){
	int n;
	cin >> n;
	for (int i = 0; i < n; ++i) {
		cin >> s[i];
	}
	int ans = 6;
	for (int i = 0; i < n; ++i) {

		for (int j = 0; j < n; ++j) {
			if (i == j) continue;

			int k = 0;
			for (int c = 0; c < s[i].size(); ++c) {
				if (s[i][c] != s[j][c]) k++;
			}
			ans = min(ans, (k - 1) / 2);
		}
	}
	cout << ans << endl;
	return 0;
}
