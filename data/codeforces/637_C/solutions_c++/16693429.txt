#include <bits/stdc++.h>

using namespace std;

int main() {
	int n;
	cin >> n;

	string a[n];
	for (int i = 0; i < n; i++) cin >> a[i];

	int r = 6;
	for (int i = 0; i < n; i++){
		for (int j = i + 1; j < n; j++){
			int cur = 0;
			for (int k = 0; k < 6; k++)
				if (a[i][k] != a[j][k]) cur++;
			 r = min(r, (cur - 1) / 2);
		}
	}

	cout << r;

  return 0;
}
