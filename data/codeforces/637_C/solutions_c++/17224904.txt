#include <iostream>

#include <vector>
using namespace std;

int solve(string &s1,string &s2){
	int res = 0;
	for(int i=s1.length()-1;i>=0;--i){
		if(s1[i]!=s2[i])	res++;
	}
	if(res<2)	return 2;
	return res+1;

}

int main(){
	ios_base::sync_with_stdio(0);	cin.tie();
	int n;	cin>>n;
	string arr[2001];
	for(int i=0;i<n;i++)	cin>>arr[i];
		int res = 6;
	for(int i=0;i<n;i++)	for(int j=i+1;j<n;j++)	
	res = min(res,solve(arr[i],arr[j])/2-1);
	cout<<res<<'\n';
}
