#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    int n;
    cin >> n;
    
    string arr[n];
    for(int i = 0; i < n; i++)
        cin >> arr[i];
    
    int k = 13; // (13 - 1) / 2 == 6
    
    for(int i = 0; i < n - 1; i++)
		for(int j = i + 1; j < n; j++)
		{
			int diff = 0;
			for(int digit = 0; digit < 6; digit++)
				if(arr[i][digit] != arr[j][digit]) 
					diff++;
			
			k = min(diff, k);
		}

	cout << (k - 1) / 2;

    return 0;
}
