#include <bits/stdc++.h>
using namespace std;

const int N = (int)2e3;

int dif( string a, string b )
{
    int n = 0;
    for( int i = 0; i < (int)a.length(); i++ )
        n += (a[i] != b[i]);
    return n;
}

int main( void )
{
    int n;
    string s[N];
    int ans = 6;

    cin >> n;
    for( int i = 0; i < n; i++ )
        cin >> s[i];
    for( int i = 0; i < n; i++ ) {
        for( int j = i+1; j < n; j++ ) {
            ans = min(ans,(dif(s[i],s[j])-1)/2);
        }
    }
    cout << ans;

    return 0;
}
