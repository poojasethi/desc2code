#include<cstdio>

int n;
char s[1111][11];

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++)
		scanf("%s", s[i]);
	
	if (n==1) {
		printf("6\n");
		return 0;
	}
	
	int ans=6;
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++) {
			int diff = 0;
			for(int k=0;k<6;k++)
				if(s[i][k]!=s[j][k]) diff++;
			if(diff<ans)ans=diff;
		}
	printf("%d\n", (ans-1)/2);
	
	return 0;
}
