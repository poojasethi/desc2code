#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,n,k,j,digit,diff;
    cin>>n;
    string arr[n];
    for(i=0;i<n;i++){cin>>arr[i];}
    k=6;
    for(i=0;i<n-1;i++)
    {
		for(j=i+1;j<n;j++)
		{
			diff=0;
			for(digit=0;digit<6;digit++){if(arr[i][digit] != arr[j][digit]){diff++;}}
			k=min((diff-1)/2,k);
		}
    }
	cout<<k<<endl;
    return 0;
}
