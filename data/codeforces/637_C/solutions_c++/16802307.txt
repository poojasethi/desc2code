string s[1000];
int dis(string a,string b){
	int xb,cnt=0;
	for(xb=0;xb<6;xb++) if(a[xb]!=b[xb]) cnt++;
	return cnt;
}

	int n;
	cin>>n;
	int xb,c=0,x2,min=6;
	for(xb=0;xb<n;xb++){
		string x;
		cin>>x;
		s[c++]=x;
	}
	for(xb=0;xb<n;xb++)
	for(x2=0;x2<n;x2++){
		if(xb==x2) continue;
			if(min>((dis(s[xb],s[x2])-1)/2)) min=((dis(s[xb],s[x2])-1)/2);
	}
	cout<<min;
	return 0; 
}
