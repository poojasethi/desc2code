#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

const int N = 1005;

string a[N];

int main()
{
	ios_base::sync_with_stdio(0);
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> a[i];
	}
	int res = 6;
	for (int i = 0; i < n; i++)
	{
		for (int j = i + 1; j < n; j++)
		{
			int diff = 0;
			for (int t = 0; t < 6; t++)
			{
				diff += (a[i][t] != a[j][t]);
			}
			res = min(res, (diff - 1) / 2);
		}
	}
	cout << res;
	return 0;
}
