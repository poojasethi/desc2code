#include<cstdio>
#include<algorithm>
#include<cstring>
#include<map>
#include<iostream>
#include<string>
using namespace std;

string a[1005];
int main()
{
	int n;
	scanf("%d",&n);
	for (int i=1;i<=n;i++){
		cin>>a[i];
	}
	int d=13;
	for (int i=1;i<=n;i++)
		for (int j=i+1;j<=n;j++){
			int c=0;
			for (int k=0;k<6;k++)c+=a[i][k]!=a[j][k];
			d=min(d,c);
		}
	printf("%d\n",(d-1)/2);
	return 0;

}
