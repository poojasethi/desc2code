#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
char s[1111][10];
int n;

int dist(int x, int y) {
	int ans = 0;
	for (int i = 0; i < 6; ++i) ans += (s[x][i] != s[y][i]);
	return ans;
}
int main() {
	scanf("%d\n", &n);
	for (int i = 0; i < n; ++i) gets(s[i]);
	int _max = 13;
	for (int i = 0; i < n; ++i) for (int j = i + 1; j < n; ++j) _max = min(_max, dist(i, j));
	cout << (_max - 1) / 2<< endl;
	return 0;
}
