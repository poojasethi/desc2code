#include <iostream> 
#include <set> 

using namespace std; 

int main() 
{ 
int n,max1=0; 
cin>>n; 
int a[n][6]; 
for(int i=0; i<n; i++) 
{ 
string x; 
cin>>x; 
for(int j=0; j<6; j++) 
{ 
a[i][j]=x[j]-48; 

} 
} 
for(int i=0; i<n-1; i++) 
{ 
for(int j=i+1; j<n; j++) 
{ 
int l=0; 
for(int k=0; k<6; k++) 
{ 
if(a[i][k]==a[j][k]) l++; 
} 
if(l>max1) 
{ 
max1=l; 
} 
} 
} 
if(n==1) cout<<6; else cout<<(6-max1-1)/2; 
}
