#include <bits/stdc++.h>
using namespace std;

int dist(string s1, string s2)
{
    int ans=0;
    for(int i=0;i<6;i++)
    {
        if(s1[i]!=s2[i])ans++;
    }
    return ans;
}

string s[1007];

int main()
{
    //freopen("input.txt","r",stdin);

    int n,i,j,md=13;
    cin>>n;
    for(i=0;i<n;i++)cin>>s[i];


    for(i=0;i<n;i++)
        for(j=i+1;j<n;j++)
            md = min(md, dist(s[i],s[j]));

    cout<< (md-1)/2 <<"\n";

    return 0;

}
