#include <iostream>
#include <string>
#include <vector>

using namespace std;

int n, ans = 13;
vector<string> str;

int main()
{
	cin >> n;
	str.resize(n);
	for(int i=0; i<n; ++i)
		cin >> str[i];
	for(int i=0; i<n; ++i)
		for(int j=i+1; j<n; ++j)
		{
			int count = 0;
			for(int k=0; k<6; ++k)
				count += (str[i][k] != str[j][k]);
			ans = min(ans, count);
		}
	cout << (ans-1) / 2 << endl;
	return 0;
}
