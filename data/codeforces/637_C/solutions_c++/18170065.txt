#include <bits/stdc++.h>
using namespace std;

int n,res=6;
string s[1005];

int main()
{
    //freopen("637C.INP","r",stdin);
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>s[i];
    for(int i=1;i<=n;i++)
        for(int j=i+1;j<=n;j++)
        {
            int d=0;
            for(int k=0;k<6;k++)
                d+=(s[i][k]!=s[j][k]);
            res=min(res,(d-1)/2);
        }
    cout<<res;
}
