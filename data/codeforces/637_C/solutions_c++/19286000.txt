#include<iostream>
#include<algorithm>
#include<string>
#include<vector>
#include<map>
#define ll long long int
using namespace std;

int n;
string str [5000];
int main(){
  cin>>n;
  for(int i=0;i<n;i++){
    cin>>str[i];
  }
  int mini = 6;
  for(int i=0;i<n-1;i++){
    for(int j=i+1;j<n;j++){
      int cnt = 0;
      for(int k=0;k<6;k++){
        cnt+=(str[i][k]!=str[j][k]);
      }
        mini = min(mini,(cnt-1)/2);
    }
  }
  cout<<mini<<endl;
  return 0;
}
