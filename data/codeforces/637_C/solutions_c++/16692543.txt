#include <iostream>
#include <algorithm>

using namespace std;
int res,m,n;
int lol(string s1, string s2)
{
    int k=0;
    for(int i=0;i<6;i++)
        if (s1[i]!=s2[i])
            k++;
    return k/2-(k+1)%2;
}
int main()
{
    string s[10000];
    cin >> n;
    for(int i=0;i<n;i++)
        cin >> s[i];
    sort(s,s+n);
    res=6;
    for(int i=0;i<n;i++)
        for(int j=i+1;j<n;j++)
            res=min(res,lol(s[i],s[j]));
    cout << res;

}
