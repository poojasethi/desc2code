#include <iostream>
#include <vector>
using namespace std;
int main(void) {
	vector<string> nom;
	string s;
	int n,d,mind = 6;
	cin >> n;
	if (n == 1) {
		cout << "6" << endl;
		return 0;
	}
	for (int i = 0; i< n; i++) {
		cin >> s;
		nom.push_back(s);
		for (int j = 0; j < i; j++) {
			d = 0;
			for(int k = 0; k<6; k++)
				d += (s[k]!=nom[j][k]);
			if (d<mind) mind = d;
		}
	}
	cout << (mind-1)/2 << endl;
	return 0;
}
