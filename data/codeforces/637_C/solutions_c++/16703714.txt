#include <iostream>
#include <fstream>
#include <map>
#include <vector>
#include <cstring>
using namespace std;

int n,i,j,mindif;
string s[1001];

int dif(int b, int a)
{int i,ans=0;

for(i=0;i<6;i++)
if(s[b][i]!=s[a][i])ans++;

return ans;
}
main(){
cin>>n;
for(i=1;i<=n;i++)	
cin>>s[i];

if(n==1) {
cout<<6<<endl;return 0;}
mindif=6;	

for(i=1;i<n;i++)	
for(j=i+1;j<=n;j++)	
{mindif=min(mindif,(dif(i,j)-1)/2);
}	
	cout<<mindif<<endl;
}
