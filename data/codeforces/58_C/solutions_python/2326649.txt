R=lambda:map(int,raw_input().split())
n=input()
a=R()
for i in xrange(n):
  a[i]=(i if i<n-1-i else n-1-i)+1-a[i]
a.sort()
s,d,u,z=sum(map(abs,a)),0,0,0
for x in a:
  if x<0:
    d+=1
  elif x>0:
    u+=1
  else:
    z+=1
t,v=u+d,0
for _ in xrange(100100):
  v+=1
  u+=z
  z=0
  while d>0 and v+a[d-1]>=0:d-=1;z+=1
  t=min(t,u+d)
print t
