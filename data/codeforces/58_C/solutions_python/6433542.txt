N = 100010
d = [0] * N
s = []

n = int(raw_input())
a = map(int,raw_input().split())

lp = 0
rp = n-1
index = 1
while lp <= rp:
   d[lp] = a[lp] - index
   d[rp] = a[rp] - index
   lp += 1
   rp -= 1
   index += 1
for i in range(n):
   if d[i] >= 0:
      s.append(d[i])
s.sort()

maxLen = 1
cnt = 1
size = len(s)
for i in range(1,size):
   if s[i] == s[i-1]:
      cnt += 1
   else:
      maxLen = max(maxLen,cnt)
      cnt = 1
maxLen = max(maxLen , cnt)
print n - maxLen
