n = int(raw_input())
data = map(int, raw_input().split())

d = {}
mid = n / 2
if n % 2 == 0:
	mid -= 1

mx = 0
for i, e in enumerate(data):
	if i == 0 or i == n - 1:
		x = e
	elif i <= mid:
		x = e - i
	else:
		x = e - (n - i - 1)
	if x > 0:
		if x not in d:
			d[x] = 0
		d[x] += 1
		if d[x] > mx:
			mx = d[x]
print n - mx
