#include <iostream>
#include <memory.h>
using namespace std;
int a[100001],cnt[100001];
int main()
{
    int n,res=0;
    cin>>n;
    memset(cnt,0,sizeof cnt);
    for (int i=1;i<=n;++i){
        cin>>a[i];
        a[i]-=(i<=n/2)?i:n-i+1;
        //cout<<a[i]<<endl;
        if (a[i]>=0){
           ++cnt[a[i]];
           res=max(res,cnt[a[i]]);
        }
    }
    cout<<n-res<<endl;
    cin>>n;
    return 0;
}
