#include <stdio.h>
#include <string.h>
#include <algorithm>
#define maxn 100010
using namespace std;
int tot[maxn];
int n;
int main(){
	int i,j,k;
	scanf("%d",&n);
	memset(tot,0,sizeof(tot));
	for (i=0;i<n;i++){
		scanf("%d",&k);
		k-=min(i,n-1-i);
		if (k>=1) tot[k]++;
	}
	int ans=n;
	for (i=1;i<=100000;i++)
		if (n-tot[i]<ans) ans=n-tot[i];
	printf("%d\n",ans);
	return 0;
}
