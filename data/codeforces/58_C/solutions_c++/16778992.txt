#include <bits/stdc++.h>
using namespace std;

int x[1000006];

int main () {

    int n; cin >> n;

    int dist,c,ma=0;
    for (int i=1; i<=n; i++) {
        cin >> c;
        dist = min( i, n-i+1 );
        if (c<dist) continue;
        c-=dist;
        x[c]++;
        if (x[c] > x[ma]) ma = c;
    }

    int ans = n-x[ma];
    cout << ans << endl;
}
