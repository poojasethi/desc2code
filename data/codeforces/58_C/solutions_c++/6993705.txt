#include <iostream>
using namespace std;
int main () {
	int n, a[100005], ans=0, b[100005]={0};
	cin >> n;
	for (int i = 0; i < n; ++i) {
		cin >> a[i];
		if(i<n/2) {b[a[i]-i]++;}
		else {b[a[i]+i-n+1]++;}
	}
	int max=0;
	for (int i = 1; i < 100001; ++i) {
		if(b[i]>max) {max=b[i];}
	}
	ans = n - max;
	cout << ans << endl;
}
