#include <cstdio>
#include <algorithm>
using namespace std;

int n,v[100010],tmp;

int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&tmp);
        tmp=max(tmp-i,tmp+i+1-n);
        if(tmp>0) v[tmp]++;
        //printf("tmp=%d v[tmp]=%d\n",tmp,v[tmp]);
    }
    int ans=n;
    for(int i=0;i<100010;i++) ans=min(ans,n-v[i]);
    printf("%d\n",ans);
}
