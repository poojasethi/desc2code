#include <cstdio>
using namespace std;

int cnt[100001], n;

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		int a;
		scanf("%d", &a);

		int k = i < n/2 ? i : n-1-i;
		if (a-k > 0) {
			cnt[a-k]++;
		}
	}

	int ans = 0;
	for (int k = 1; k <= 100000; k++) {
		if (ans < cnt[k]) {
			ans = cnt[k];
		}
	}

	printf("%d\n", n-ans);
	return 0;
}
