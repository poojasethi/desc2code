#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#include<iostream>
using namespace std;
int a,b[100007],m,n,i;
int main()
{
    //freopen("a.in","r",stdin);
    // freopen("a.out","w",stdout);
  cin>>n;
  for(i=0; i<n; i++)
  {
    cin>>a;
    if((i<n/2)&&(a>i)) b[a-i]++;
    else if((i>=n/2)&&(a-(n-i-1)>=1)) b[a-(n-i-1)]++;
  }
  for (i=1; i<=100000; i++)
       m=max(m,b[i]);
       
       
          cout<<n-m; 
}
