#include <iostream>
#include <vector>
#include <utility>
#include <cstring>
using namespace std;

int main()
{
    int n, a, b, c[100010];
    memset(c, false, sizeof(c));
    cin >> n;
    for (int i=0; i<n; i++)
    {
        cin >> a;
        b = min(i+1, n-i);
        if (a >= b) c[a-b]++;
    }

    int ans = n;

    for (int i=0; i<100010; i++)
        ans = min(ans, n-c[i]);

    cout << ans << endl;
    
    return 0;
}
