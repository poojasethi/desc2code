#include <iostream>
using namespace std;
int a[100001];
int main()
{
    int n, i, j, x;
    cin>>n;
    for (i=1; i<=n; i++) {
        j=min(i-1,n-i);
        cin>>x;
        x = x-j;
        if (x>0) a[x]++;
    }
    for (j=0,i=1; i<=100000; i++) j=max(j,a[i]);
    cout<<n-j<<endl; 
    return 0;
}
