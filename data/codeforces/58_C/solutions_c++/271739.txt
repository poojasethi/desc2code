#include <cstdio>
#include <algorithm>
using namespace std;
const int MN=1000*100+100;
int num[MN];
int a[MN];
int n;
int main()
{
	scanf("%d",&n);
	for (int i=0;i<n;i++)
		scanf("%d",&num[i]);
	for (int i=0;i<n;i++)
	{
		int t=min(i,n-1-i);
		if (num[i]-t>0)
			a[num[i]-t]++;
	}
	int mx=-1;
	for (int i=1;i<MN;i++)
		mx=max(mx,a[i]);
	printf("%d\n",n-mx);
	return 0;

}
