#include <bits/stdc++.h>

using namespace std;

int a[100010];
int main()
{
    int n,i,j,good=0,x;

  // freopen("input.txt","r",stdin);
    cin>>n;

    for(i=1;i<=n;i++)
    {
        cin>>x;
        j=min(i-1,n-i);
        if(x-j>0) {x=x-j;
        a[x]++;}
    }

    for(j=1;j<=100000;j++) good=max(good,a[j]);

    cout<<n-good<<endl;
}
