#include <map>
#include <iostream>
using namespace std;

int main() {
	int n, a;
	map<int, int> m;
	int b = 0;

	cin >> n;
	for (int i = 0; i < n; ++i) {
		cin >> a;
		if (i < n/2 && a - i > 0)
			++m[a - i];
		if (i >= n/2 && a - (n - 1 - i) > 0)
			++m[a - (n - 1 - i)];
	}

	for (map<int, int>::iterator i = m.begin(); i != m.end(); ++i)
		if (i->second > b)
			b = i->second;

	cout << n - b;

	return 0;
}
