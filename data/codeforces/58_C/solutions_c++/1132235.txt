#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int a[200020]={0};
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		int x;
		scanf("%d",&x);
		x-=min(i,n-i+1);
		x+=100000;
		a[x]++;
	}
	int max=-1;
	for(int i=100000;i<200020;i++)
	{
		if(a[i]>max)
		{
			max=a[i];
		}
	}
	printf("%d\n",n-max);
}
