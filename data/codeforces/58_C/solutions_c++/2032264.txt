#include <iostream>   
using namespace std; 
int  a[100001],b[100001];
int main()
{
	int n,i;
	while(cin>>n){
		for(i=0;i<100001;i++)  b[i]=0;
		for(i=0;i<n;i++)   cin>>a[i];
		for(i=0;i<(n+1)/2;i++){
			a[i]=a[i]-i;
			if((2*i+1)!=n) a[n-i-1]=a[n-i-1]-i;
		}
        for(i=0;i<n;i++)  if(a[i]>0)  b[a[i]]++;
		int max=0;
		for(i=1;i<100001;i++)  max=max>b[i]?max:b[i];
		cout<<n-max<<endl;
	}
}
