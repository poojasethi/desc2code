#include <iostream>
#include <algorithm>

using namespace std;

int d [ 100000 ];

int main(){
    int n, i, x, v, m, a ( 0 );
    cin >> n;
    m = ( n >> 1 ) + ( n & 1 );
    for ( i = 0; i < n; ++ i ){
        cin >> x;
        v = i < m ? i + 1 : n - i;
        x < v || ++ d [ x -= v ] && ( a = max ( a, d [ x ] ) );
    }
    a = n - a;
    cout << a << endl;
    return 0;
}
