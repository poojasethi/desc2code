#include <stdio.h>
#define NMAX 100010
int f[NMAX*2+30];
int n;
int main(){
	//freopen("C.in", "r", stdin);
	//freopen("C.out", "w", stdout);
	scanf("%d", &n);
	for(int i = 1; i <= n; ++i) {
		int x;
		scanf("%d", &x);
		if(i <= n/2 + n % 2) x -= i;
		else x -= (n-i+1);
		f[x+NMAX] ++;
	}
	int sol = n;
	for(int i = NMAX; i < 2*NMAX; ++i)
		if( n - f[i] < sol) sol = n-f[i];
	printf("%d\n", sol);
	return 0;
}
