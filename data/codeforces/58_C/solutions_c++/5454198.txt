#include <iostream>

using namespace std;

int n, a[100005], i, c, ans[100005];

int main()
{
    cin >> n;

    for (int i = 0; i < n; i++)
        cin >> a[i];

    for (int i = 0; i < n; i++)
    {
        int k = min(i, n - 1 - i);
        int f = a[i] - k;
        if (f >= 1)
            ans[f]++;
    }
    for (int i = 0; i < 100005; i++)
        c = max(c, ans[i]);

    cout << n - c << endl;

    return 0;
}
