#include <iostream>
#include <cmath>
#include <cstdio>
using namespace std;
int x,t[100005],maxi=0,n;
int main()
{
  cin>>n;
  for(int i=0;i<n;i++)
  {
    cin>>x;
    if(i<n/2 && x>i) t[x-i]++;
    else if(i>=n/2 && x-(n-i-1)>=1) t[x-(n-i-1)]++;
  }
  for(int i=1;i<=100000;i++) maxi=max(maxi,t[i]);
  cout<<n-maxi<<endl;
}
