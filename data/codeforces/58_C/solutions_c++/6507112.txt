#include <stdio.h>
#include <algorithm>

using namespace std;

int main() {
    int k, n, a[100001] ={ 0 };
scanf("%d",&n);
for(int i = 0; i < n/2; i++) {
scanf("%d",&k);
if(k-i > 0) a[k-i]++;
}
for(int i = n/2; i < n; i++) {
scanf("%d",&k);
if(k-n+i+1 > 0) a[k-n+i+1]++;
}
int Max = 0;
for(int i = 0; i <= 100000; i++)
    if(Max < a[i])Max = a[i];
printf("%d",n-Max);
return 0;
}
