#include <iostream>
#include <cstring>

using namespace std;

int main()
{
    int n;
    cin >> n;
    int dm[100000];
    memset(dm, 0, sizeof(dm));
    int cres = 1;
    for (int i = 0, h; i < n; i++) {
        cin >> h;
        int d = (h - 1) - (i >= n/2? n - i - 1 : i);
        if (d >= 0 && ++dm[d] > cres) cres = dm[d];
    }
    cout << n - cres << endl;
    return 0;
}
