#include<cstdio>
#include<map>
//              Last Change:  2011-02-06 20:23:39
using namespace std;
map<int,int> h;
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
	{
		int x;
		scanf("%d",&x);
		int j=i;
		if(n-j+1<j)j=n-j+1;
		x-=j-1;
		if(x>0)++h[x];
	}
	int ans=0;
	for(map<int,int>::iterator i=h.begin();i!=h.end();++i)
		if((*i).second>ans)
			ans=(*i).second;
	printf("%d\n",n-ans);
	return 0;
}
