#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    int a[220000];
    for(int i=0;i<n;i++)
    {
        int x;
        scanf("%d",&x);
        if(i+1<=n/2+n%2)
            if(x-i>0) a[x-i]++;
        if(i+1>n/2+n%2)
            if(x-(n-i-1)>0) a[x-(n-i-1)]++;
    }
    int ans=n;
    for(int i=0;i<=100000;i++) ans=min(ans,n-a[i]);
    printf("%d",ans);
    return 0;
}
