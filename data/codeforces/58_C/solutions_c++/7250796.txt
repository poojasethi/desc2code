#include<iostream>
using namespace std;
int s[200010],a[100010];
int main()
{
	int n;
	cin>>n;
	int mx=0,ans=0;
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<n/2;i++)a[i]+=n/2-i,a[n-i-1]+=n/2-i;
	for(int i=0;i<n;i++)s[a[i]]++,mx=max(mx,a[i]);
	for(int i=n/2+1;i<=mx;i++)ans=max(ans,s[i]);
	cout<<n-ans<<endl;
}
