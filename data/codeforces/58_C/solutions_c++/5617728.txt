#include <iostream>
using namespace std;
int n;
int a[100001];
int x;
int main() {
    int res=-1;
    cin >> n;
    for(int i=0;i<n;i++) {
        cin >> x;
        x=max(x-i,x-n+i+1);
        if(x>0) a[x]++;
    }
    for(int i=0;i<100001;i++) {
        res=max(res,a[i]);
    }
    cout << n-res << endl;
    return 0;
}
