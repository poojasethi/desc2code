#include <iostream>
#include <map>
using namespace std;
int main()
{
	int n;
	cin>>n;
	map<int,int> m;
	for(int i=0; i<n; ++i) {
		int a;cin>>a;
		int k = i;
		if (i>=n/2) k = n-i-1;
		int x = a-k;
//		cout<<"x "<<x<<' '<<k<<'\n';
		if (x>0) ++m[x];
	}
	int r = 0;
	for(map<int,int>::iterator i=m.begin(); i!=m.end(); ++i)
		r = max(r, i->second);
	cout<<n-r<<'\n';
}
