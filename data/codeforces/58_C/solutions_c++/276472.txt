#include <stdio.h>
#include <algorithm>

using namespace std;

int qnt[100010];
int main()
{
	int n, sol;
	scanf("%d", &n);
	sol = n;
	for(int i = 0; i < n; i++)
	{
		int v;
		scanf("%d", &v);
		if(i < n/2)
		{
			if(v - i > 0)
			{
				qnt[v-i]++;
				sol = min(sol, n-qnt[v-i]);
			}
		}
		else if(v - (n - 1 - i) > 0)
		{
			qnt[v-(n-1-i)]++;
			sol = min(sol, n-qnt[v-(n-1-i)]);
		}
	}
	printf("%d\n", sol);
	return 0;
}
