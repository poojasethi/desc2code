#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;
map<int,int> a;
int main()
{
    int n;
    scanf("%d",&n);
    for (int i=0;i<n;i++)
    {
        int x;
        scanf("%d",&x);
        a[x-min(i,n-i-1)]++;
    }
    int ans=0;
    for (map<int,int> ::iterator it=a.begin();it!=a.end();it++)
        if (it->first>0)
            ans=max(ans,it->second);
    printf("%d\n",n-ans);
    return 0;
}
