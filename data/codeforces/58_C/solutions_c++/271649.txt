#include<cstdio>
#include<algorithm>
using namespace std;
int a[300005];
int main()
{
    int n, i, k, m;
    scanf("%d", &n);
    int max = 0;
    for(i = 0; i < n; ++i)
    {
        scanf("%d", &k);
        m = 2*k + abs(2*i - n + 1);
        if ( m >= 0 )
        {
            a[m]++;
            if(m >= (n + 1) && a[m] > max)
                max = a[m];
        }
    }
    printf("%d", n - max);
    return 0;
}
