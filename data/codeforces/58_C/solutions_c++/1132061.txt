#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int M = 200005;
int ct[M];

int main()
{
	int n;scanf("%d",&n);
	for(int i = 0; i < n; i++)
	{
		int x;
		scanf("%d",&x);
		int y = x-(i<(n+1)/2?i:(n-1-i));
//		printf("%d\n",y);
		if(y>0)ct[y]++;
	}
	int mx = 0;
	for(int i = 0; i < M; i++)
		mx = max(mx,ct[i]);
	printf("%d\n",n-mx);

	return 0;
}
