#include<iostream>
#include<string>
#include<vector>
#include<map>
using namespace std;
int a,b[100010],c[100010],ans=-1,m=-1;
int main()
{   
	cin>>a;
	for(int i=0;i<a;i++)
	{
		cin>>b[i];
		if(b[i]>m)
			m=b[i];
		if(i<a/2)
		c[b[i]-i]++;
		else
			c[b[i]-(a-(i+1))]++;
	}
	for(int i=1;i<=m;i++)
	{
		if(c[i]>ans)
			ans=c[i];
	}
	cout<<a-ans<<endl;
	return 0;
}
