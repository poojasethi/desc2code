#include <bits/stdc++.h>
#define maxn 1009
using namespace std;
int n,S,dp[maxn][maxn],f[maxn];
struct node
{
  int l,r,w,s,v;
  bool operator<(const node &rhs)const
  {
    if(r==rhs.r)
      return l>rhs.l;
    else
      return r<rhs.r;
  }
  node(int _l=0,int _r=0,int _w=0,int _s=0,int _v=0)
  {
    l=_l;
    r=_r;
    w=_w;
    s=_s;
    v=_v;
  }
}a[maxn];
int main()
{
  scanf("%d%d",&n,&S);
  for(int i=0;i<n;i++)
  {
    scanf("%d%d%d%d%d",&a[i].l,&a[i].r,&a[i].w,&a[i].s,&a[i].v);
  }
  a[n++]=node(0,2*maxn,0,S,0);
  sort(a,a+n);
  for(int i=0;i<n;i++)
  {
        for(int j=a[i].w;j<=S;j++)
        {
            int rs=min(a[i].s,j-a[i].w);
            int cur=a[i].l;
            f[cur]=0;
            for(int k=0;k<i;k++)
            {
                if(a[k].l>=a[i].l)
                {
                    while(cur<a[k].r)
                    {
                        f[cur+1]=f[cur];
                        cur++;
                    }
                    f[cur]=max(f[cur],dp[k][rs]+f[a[k].l]);
                }
            }
            dp[i][j]=f[cur]+a[i].v;
        }
    }
  printf("%d\n",dp[n-1][S]);
  //system("pause");
  return 0;
}
