#include <bits/stdc++.h>
//Damned if I don't take what I want.

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

const int MAXN = 550, MAXS = 1010;

int n, S, in[MAXN], out[MAXN], w[MAXN], s[MAXN], v[MAXN], id[MAXN];
int dp[MAXN][MAXS], pre[2 * MAXN];

bool byOut(int a, int b){ return pii( out[a], -in[a] ) < pii( out[b], -in[b]); }

int main(){
	ios::sync_with_stdio( false );
	cout.tie(0); cin.tie(0);

	cin >> n >> S;

	n++;
	in[0] = -1; out[0] = -1u/8; id[0] = 0; s[0] = S; v[0] = w[0] = 0;
	for (int i = 1; i < n; i++)
		cin >> in[i] >> out[i] >> w[i] >> s[i] >> v[i], id[i] = i;
	sort(id, id + n, byOut);

	for (int i = 0; i < n; i++)
		for (int k = w[ id[i] ]; k <= S; k++){
			int x = id[i], rem = min(k - w[x], s[x]), p = in[x];
			pre[p] = 0;

			for (int j = 0; j < i; j++) if ( in[ id[j] ] >= in[x]){
				int y = id[j];
				while(p < out[y])
					pre[p] = pre[p++];
				pre[p] = max(pre[p], pre[ in[y] ] + dp[j][rem]);
			}

			dp[i][k] = pre[p] + v[x];
		}

	cout << dp[n - 1][S];
}
