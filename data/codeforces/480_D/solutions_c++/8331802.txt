#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;

struct re{
	int l,r,s,w,v;
	void read() {
		scanf("%d%d%d%d%d",&l,&r,&w,&s,&v);
		l++; r++;
		// [l,r)
	}
	bool operator () (const re &a,const re &b) {
		return a.r==b.r?a.l>b.l:a.r<b.r;
	}
}a[601];

int n,S;
int f[601][2001],g[2001];

/*
	f[i][j]: i ����� ����i ��������С�ڵ��� j �� ������ 
*/

int main()
{
	scanf("%d%d",&n,&S);
	for (int i=1;i<=n;i++) a[i].read();
	sort(a+1,a+1+n,re());
	a[n+1]=(re){1,n+n,S,0,0};
	for (int i=1;i<=n+1;i++)
	{
		for (int j=a[i].w;j<=S;j++)
		{
			for (int k=0;k<=a[i].r;k++) g[k]=0;
			int last=a[i].l; int limit=min(j-a[i].w,a[i].s);
			for (int k=1;k<i;k++)
			{
				if (a[k].l>=a[i].l&&a[k].r<=a[i].r)
				{
					for (;last<=a[k].l;last++) g[last]=g[last-1]>g[last]?g[last-1]:g[last];
					if (g[a[k].l]+f[k][limit]>g[a[k].r]) g[a[k].r]=g[a[k].l]+f[k][limit];
				}
			}
			for (;last<=a[i].r;last++) g[last]=g[last-1]>g[last]?g[last-1]:g[last];
			if (g[a[i].r]+a[i].v>f[i][j]) f[i][j]=g[a[i].r]+a[i].v;
		}
	}
//	printf("%d\n",f[n][S]);
	printf("%d\n",f[n+1][S]);
}
