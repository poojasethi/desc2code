#include <bits/stdc++.h>
using namespace std;
int IN[1005],Out[1005],w[1005],s[1005],v[1005],n,dp[1005][1005],next[1005],G[1005][1005],st[1005];
int solve(int i,int ss){
	if(ss<0)return 0;
	int &ret= dp[i][ss];
	if(ret!=-1)return ret;
	G[i][Out[i]+1]=0;
	for(int j=Out[i];j>=IN[i];--j){
		G[i][j]= G[i][j+1];
		for(int k=st[j];k;k=next[k]){
			if( k==i || Out[k]>Out[i])continue;
			int x= G[i][Out[k]];
			G[i][j]= max(G[i][j],solve(k,min(ss-w[k],s[k]))+x);
		}
	}
	return ret=G[i][IN[i]]+v[i];
}	
int main(){
	int S;
	cin>>n>>S;
	for(int i=1;i<=n;++i){
		cin>>IN[i]>>Out[i]>>w[i]>>s[i]>>v[i];
		next[i]= st[IN[i]];
		st[IN[i]]=i;
	}
	IN[0]=0,Out[0]=2*n+1;
	memset(dp,-1,sizeof(dp));
	printf("%d\n",solve(0,S));
}
