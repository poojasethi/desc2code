#include <bits/stdc++.h>
#define rep(i,j,k)	for (int i = j; i <= k; i++)
#define dep(i,j,k)	for (int i = j; i >= k; i--)
using namespace std;
typedef long long ll;
const int N = 1010;
inline void get(int &x)
{
	char ch = getchar();	x = 0;
	while (ch < '0')	ch = getchar();
	while (ch >= '0')	x = x*10+ch-'0', ch = getchar();
}
struct zcc{
	int in, out, w, v, str;
	bool operator<(const zcc x)const{return out<x.out||(out==x.out&&in>x.in);}
}p[N];
int n, s, ans, sv[N], li[N], f[N][N];
int main()
{
	cin >> n >> s;
	rep(i,1,n) scanf("%d%d%d%d%d", &p[i].in, &p[i].out, &p[i].w, &p[i].str, &p[i].v);
	p[n+1].out = 3*n, p[n+1].str = s, n++;	sort(p+1,p+n+1);
	rep(i,1,n)	dep(j,i-1,0)
		if (p[i].in >= p[j].out)
		{
			li[i] = j;
			break;
		}
	rep(i,1,n)	rep(j,p[i].w,s)
	{
		rep(k,1,i-1)
		{
			sv[k] = sv[k-1];
			if (p[i].str>=j-p[i].w && p[k].in>=p[i].in)
				sv[k] = max(sv[k],sv[li[k]]+f[k][j-p[i].w]);
		}
		f[i][j] = sv[i-1]+p[i].v;
		if (j)	f[i][j] = max(f[i][j-1],f[i][j]);
		ans = max(f[i][j],ans);
	}
	cout << ans << endl;
	return 0;
}
