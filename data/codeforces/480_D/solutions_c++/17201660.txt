int min(int a, int b){return a<b?a:b;}

#define N 510

int pp[N<<1],lp[N<<1];
void insert(int p, int i){
  pp[i] = lp[p];
  lp[p] = i;
}

int n, S, in[N],out[N],w[N],v[N],s[N];
int f[N][1010], G[N][1010];

int dp(int i,int j){
  int k, R;
  if(f[i][j]!=-1)
    return f[i][j];
  if(j<w[i])
    return 0;
  int *g=G[i];
  f[i][j] = 0;
  for(R = in[i]; R <= out[i]; R++){
    g[R] = (R!=in[i]?g[R-1]:0);
    for(k = lp[R]; ~k; k=pp[k]){
      if(i==k)continue;
      if(in[i] <= in[k] && out[k] <= out[i])
        g[R] = max(g[R], g[in[k]] + dp(k,min(j-w[i], s[i])));
    }
    f[i][j] = max(f[i][j] ,g[R]);
  }
  f[i][j] += v[i];
  return f[i][j];
}
int main(){
#ifdef QWERTIER

#endif 
  int i, j;
  scanf("%d%d",&n,&S);
  memset(f,-1,sizeof(f));
  memset(lp,-1,sizeof(lp));
  for(i = 0; i < n; i++){
    scanf("%d%d%d%d%d",&in[i],&out[i],&w[i],&s[i],&v[i]);
    insert(out[i],i);
  }
  int *g=G[500], k;
  for(i = 0; i < n*2; i++){
    g[i] = i?g[i-1]:0;
    for(k = lp[i]; ~k; k=pp[k])
      g[i] = max(g[i], g[in[k]] + dp(k, S));
  }
  printf("%d",g[n*2-1]);
  return 0;
}
