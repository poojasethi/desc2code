#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int l,r,w,s,v;
} a[N];
int i,j,k,n,m,u;
int f[510][N],h[N][N];
inline bool cc1(const ww &a,const ww &b) {
	return a.r<b.r||a.r==b.r&&a.l>b.l;
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n) scanf("%d%d%d%d%d",&a[i].l,&a[i].r,&a[i].w,&a[i].s,&a[i].v);
	a[0]=(ww){0,n*2+1,0,m,0};
	sort(a,a+n+1,cc1);
	For(i,0,n) {
		int t=min(a[i].s,m-a[i].w);
		memset(h,0,sizeof(h));
		for (j=0;a[j].r<=a[i].l;j++);
		For(k,a[i].l+1,a[i].r) {
			memcpy(h[k],h[k-1],sizeof(h[k]));
			for (;j<i&&a[j].r==k;j++) {
				int A=a[j].l;
				if (A<a[i].l) continue;
				For(u,0,t) h[k][u]=max(h[k][u],h[A][u]+f[j][u]);
			}
		}
		For(u,0,t) f[i][u+a[i].w]=h[a[i].r][u]+a[i].v;
		For(u,1,m) f[i][u]=max(f[i][u],f[i][u-1]);
	}
	printf("%d\n",f[n][m]);
	return 0;
}
