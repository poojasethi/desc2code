#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define n	1005

int		N,S,w[n],s[n],v[n],st[n],IN[n],Ot[n],Next[n],F[n][n],G[n][n];

int		Num(int i,int S){
		if	(S<0)	return	0;
		if	(~F[i][S])	return	F[i][S];
		G[i][Ot[i]+1]=0;
		for	(int j=Ot[i];j>=IN[i];j--){
			G[i][j]=G[i][j+1];
			for	(int k=st[j];k;k=Next[k]){
				if	(k==i||Ot[k]>Ot[i])	continue;
				G[i][j]=max(G[i][j],Num(k,min(S-w[k],s[k]))+G[i][Ot[k]]);
			}
		}	return	F[i][S]=G[i][IN[i]]+v[i];
}

int		main(){
		scanf("%d%d",&N,&S);
		for	(int i=1;i<=N;i++){
			scanf("%d%d%d%d%d",&IN[i],&Ot[i],&w[i],&s[i],&v[i]);
			Next[i]=st[IN[i]];	st[IN[i]]=i;
		}	IN[0]=0;Ot[0]=2*N+1;
		memset(F,-1,sizeof(F));
		printf("%d\n",Num(0,S));
}
