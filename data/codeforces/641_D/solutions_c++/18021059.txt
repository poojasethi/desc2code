#include <bits/stdc++.h>

#define size(n) ( int( n.size() ) )
#define sqr(n) ( (n) * (n) )

using namespace std;

const int N = 1000 * 100 + 5;

double s1 = 0, s2 = 0, x, y, maxx[N], minn[N], ans[N][2];

int n;

void print( int type ){
    for ( int i = 1; i <= n; i++ ){
        printf("%.9lf ",ans[i][type]);
    }
    printf("\n");
}

int main(){
    //freopen("input.txt","r",stdin);
    scanf("%d",&n);
    int i;
    for ( i = 1; i <= n; i++ ){
        scanf("%lf",&maxx[i]);
        maxx[i] = abs( maxx[i] );
    }
    for ( i = 1; i <= n; i++ ){
        scanf("%lf",&minn[i]);
        minn[i] = abs( minn[i] );
    }
    long double sum;
    for ( i = n; i >= 1; i-- ){
        sum = minn[i] + maxx[i];
        ans[i][0] = ( ( s1 - sum - s2 ) - sqrt( max( sqr( s1 - sum - s2 ) + 4 * ( s1 * sum - minn[i] ), 0.0L ) ) ) / -2.0;
        ans[i][1] = sum - ans[i][0];
        ans[i][0] = fabs(ans[i][0]);
        ans[i][1] = fabs(ans[i][1]);
        s1 += ans[i][0];
        s2 += ans[i][1];
    }
    print(0);
    print(1);
    return 0;
}
