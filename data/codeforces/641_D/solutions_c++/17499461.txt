#include <iostream>
#include <cstdio>
#include <cmath>
#include <cassert>

using namespace std;

const int N = 1e5;

double pMin[N], pMax[N];
double p1[N], p2[N];
double eps = 1e-9;

double solve(double b, double c) {
    double D = b * b - 4 * c;
    if (abs(D) < eps)
        D = 0;
    return 0.5 * (-b + sqrt(D));
}

int main() {
    // freopen("input.txt", "r", stdin);
    int n;
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> pMax[i];
    for (int i = 0; i < n; i++)
        cin >> pMin[i];
    double F = 0, PSum = 0;
    for (int i = 0; i < n; i++) {
        double P = pMax[i] + pMin[i];
        PSum += P;
        p1[i] = solve((2 * F - PSum), pMax[i] - P * F);
        p2[i] = P - p1[i];
        F += p1[i];
    }
    for (int i = 0; i < n; i++)
        printf("%.10f ", p1[i]);
    printf("\n");
    for (int i = 0; i < n; i++)
        printf("%.10f ", p2[i]);
    return 0;
}
