#include<bits/stdc++.h>
#define range(x) (x+eps>=0&&x<=1)
#define MAXN 100010
#define eps 1e-12
using namespace std;
double M[MAXN],m[MAXN],sum[2][MAXN],ans[2][MAXN];
typedef pair<double,double> P;
bool zero(double x) {return fabs(x)<eps;}
P calc(double p,double q,double x,double y)
{
	double A=1,B=q-p-x,C=y-q*x;
	double del=B*B-4*A*C;
	assert(zero(del)||del>0);
	if(zero(del)) del=0;
	else del=sqrt(del);
	double x1=(-B-del)/(2*A),x2=(-B+del)/(2*A);
	double a=0,b=0;
	if(range(x1)) a=x1;
	else if(range(x2)) a=x2;
	b=x-a;
	return P(a,b);
}

double solve(int i)
{
	double p,q,x,y;
	x=M[i]+m[i];
	p=sum[1][i-1];
	q=sum[0][i-1];
	y=M[i];
	P tmp=calc(p,q,x,y);
	ans[0][i]=tmp.first;
	ans[1][i]=tmp.second;
	for(int j=0;j<2;j++) 
		sum[j][i]=sum[j][i-1]+ans[j][i];
}



int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%lf",&M[i]);
	for(int i=1;i<=n;i++) scanf("%lf",&m[i]);
	for(int i=1;i<=n;i++) solve(i);
	for(int i=0;i<2;i++){
		for(int j=1;j<=n;j++) 
			printf("%.12f ",ans[i][j]);
		puts("");
	}
	return 0;
}
