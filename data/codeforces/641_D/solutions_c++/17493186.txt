#include <iostream>
#include <cstdio>
#include <algorithm>
#include <set>
#include <vector>
#include <map>
#include <list>
#include <cmath>

using namespace std;

double x[100500];
double y[100500];
double a[100500];
double b[100500];


int main()
{
    ios_base::sync_with_stdio(false);
    int n;
    cin >> n;
    for(int i=0;i<n;i++)
        cin >> x[i];
    for(int i=0;i<n;i++)
        cin >> y[i];
    double sa = 0;
    double sb = 0;
    for(int i=0;i<n-1;i++)
    {
        double bb = x[i] + y[i] - sb+sa;
        double c = x[i]*(1-sb) - y[i]*sb;
        double dyskr =  bb*bb - 4*c;
        if(dyskr<=0) dyskr = 0; else dyskr = sqrt(dyskr);
        b[i] = (bb - dyskr)/2;
        a[i] = x[i] + y[i] - b[i];
        sa+=a[i];
        sb+=b[i];
    }
    a[n-1] = 1-sa;
    b[n-1] = 1-sb;
    for(int i=0;i<n;i++)
        printf("%.9lf ",a[i]);
    printf("\n");
    for(int i=0;i<n;i++)
        printf("%.9lf ",b[i]);
    printf("\n");
    return 0;
}
