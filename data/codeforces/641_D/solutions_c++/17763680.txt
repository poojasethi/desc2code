#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
long double mx[120000],x1,x2,x3,ans1[120000],ans2[120000],hz[120000];
int n;
const long double eps=1e-10;
inline long double fab(long double x)
{
	if(x>0)return x;
	return -x;
}
inline long double read()
{
	double tmp;
	scanf("%lf",&tmp);
	return tmp;
}
long double sa[120000],sb[120000];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		mx[i]=read();
	}
	for (int i=1;i<=n;i++)
	{
		hz[i]=read();
	}
	long double a1pb1=0;//,samsb=mx[1];
	long double a1mb1=0;
	for (int i=1;i<=n;i++)
	{
		a1pb1+=mx[i]+hz[i];
		a1mb1+=mx[i];
		x1=a1pb1*a1pb1-4*a1mb1;
		x1=fab(x1);
		x1=sqrt(x1);
		x2=(x1+a1pb1)/2;
		x3=fab(x1-a1pb1)/2;
		if(x2>x3)swap(x2,x3);
		ans1[i]=x2;
		ans2[i]=x3;
	}
	for (int i=1;i<=n;i++)printf("%.10lf ",(double)(ans1[i]-ans1[i-1]));printf("\n"); 
	for (int i=1;i<=n;i++)printf("%.10lf ",(double)(ans2[i]-ans2[i-1]));printf("\n"); 
	return 0;
}
