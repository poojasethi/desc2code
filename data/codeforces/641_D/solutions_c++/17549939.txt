#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

double a[100010],b[100010],g[100010],h[100010];

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
	#endif
	int n;scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%lf",&a[i]);
	for (int i=1;i<=n;i++) scanf("%lf",&b[i]);
	for (int i=1;i<=n;i++) a[i]+=a[i-1];
	for (int i=n;i;i--) b[i]+=b[i+1];
	for (int i=1;i<=n;i++)
	{
		double w=a[i]-b[i+1]+1;
		double t=w*w-4*a[i];
		if (t<0) t=0;
		g[i]=(w-sqrt(t))/2;
		h[i]=w-g[i];
	}
	for (int i=1;i<=n;i++) printf("%.10lf ",g[i]-g[i-1]);puts("");
	for (int i=1;i<=n;i++) printf("%.10lf ",h[i]-h[i-1]);puts("");
	return 0;
}
