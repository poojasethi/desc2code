#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<cmath>
using namespace std;

#define maxn 100100
double mx[maxn],mi[maxn];
double p[maxn],q[maxn];
double eps=1e-12;
int n;

double calc(double a,double b,double c)
{
	double delta=b*b-4*a*c;
	if (delta<eps) delta=0;
	double x1=(-b+sqrt(delta))/a/2;
	double x2=(-b-sqrt(delta))/a/2;
	return max(x1,x2);
}

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%lf",&mx[i]);
	for(int i=1;i<=n;i++)
		scanf("%lf",&mi[i]);
	
	double P=0,Q=0;
	for(int i=1;i<=n;i++)
	{
		double sum=mx[i]+mi[i];
		q[i]=calc(-1,P-Q+sum,Q*sum-mx[i]);
		p[i]=sum-q[i];
		P+=p[i];Q+=q[i];
	}
	
	for(int i=1;i<=n;i++)
		printf("%.6lf ",p[i]);
	puts("");
	
	for(int i=1;i<=n;i++)
		printf("%.6lf ",q[i]);
	puts("");
	
	return 0;
}
