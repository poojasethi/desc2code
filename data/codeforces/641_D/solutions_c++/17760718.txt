#include<cstdio>
#include<cmath>
#define rep(i,n) for(int i=0;i<n;i++)
#define eps 1e-10
using namespace std;
double ans[2][100005],f[2][100005],maxi[100005],mini[100005];
int n;
int main(){
	scanf("%d",&n);
	rep(i,n)
		scanf("%lf",&maxi[i]);
	rep(i,n)
		scanf("%lf",&mini[i]);
	for(int i=1;i<n;i++)
		maxi[i]+=maxi[i-1];
	for(int i=n-2;i>=0;i--)
		mini[i]+=mini[i+1];
	rep(i,n-1){
		double b,c,delta;
		c=maxi[i];
		b=(-1.0)*(1.0+maxi[i]-mini[i+1]);
		delta=b*b-4.0*c;
		if(delta<eps)
			delta=0.0;
		delta=sqrt(delta);
		f[0][i]=((-1.0)*b+delta)/2.0;
		f[1][i]=((-1.0)*b-delta)/2.0;
	}
	f[0][n-1]=f[1][n-1]=1.0;
	ans[0][0]=f[0][0];
	ans[1][0]=f[1][0];
	for(int i=1;i<n;i++){
		ans[0][i]=f[0][i]-f[0][i-1];
		ans[1][i]=f[1][i]-f[1][i-1];
	}
	rep(i,n)
		printf("%.6lf%c",ans[0][i],i==n-1?'\n':' ');
	rep(i,n)
		printf("%.6lf%c",ans[1][i],i==n-1?'\n':' ');
	return 0;
}
