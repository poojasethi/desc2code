#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cstdlib>
#include<cmath>
#define eps 1e-10
using namespace std;

const int maxn=100010;
int n;
double a[maxn],b[maxn];
double k1,k2,N,M;
double P[maxn],Q[maxn];

void Get_ans(double &a,double &b){
	double B=N+k2-k1;
	double C=M-N*k1;
	C=B*B-4*C;
	if(C>eps)C=sqrt(C);
	b=(B+C)/2.0;a=N-b;
}

int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;++i)scanf("%lf",&a[i]);
	for(int i=1;i<=n;++i)scanf("%lf",&b[i]);
	k1=k2=0;
	for(int i=1;i<=n;++i){
		N=a[i]+b[i];
		M=a[i];
		Get_ans(P[i],Q[i]);
		k1+=Q[i];k2+=P[i];
	}
	for(int i=1;i<=n;++i)printf("%.10lf ",P[i]);
	printf("\n");
	for(int i=1;i<=n;++i)printf("%.10lf ",Q[i]);
	printf("\n");
	return 0;
}
