#include<iostream>
#include<cstdio>
#include<cmath>
#define maxn 100005
using namespace std;
int n;
double a[maxn],b[maxn];
double res1[maxn],res2[maxn];
double s1,s2; 
void did(double he,double ji)
{
//	printf("%.8lf %.8lf\n",he,ji);
//	cout<<he*he-4*ji<<endl;
double tmp=he*he-4*ji;
if(tmp<0)tmp=0;
	double one=he/2.0+sqrt(tmp)/2.0;
	double two=he-one;
	if(s1>s2)
	{
		s1=one;
		s2=two;
	}
	else
	{
		s1=two;
		s2=one;
	}
//	cout<<s1<<" "<<s2<<endl;
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%lf",&a[i]);
	for(int i=1;i<=n;i++)scanf("%lf",&b[i]);
	for(int i=1;i<=n;i++)
	{
		double he=b[i]-(1-s1)*(1-s2)+1+a[i]+s1*s2;
		double ji=a[i]+s1*s2;
		double t1=s1,t2=s2;
		did(he,ji);
		res1[i]=s1-t1;res2[i]=s2-t2;
	}
	for(int i=1;i<=n;i++)
	{
		printf("%.7lf ",res1[i]);
	}
	puts("");
	for(int i=1;i<=n;i++)
	{
		printf("%.7lf ",res2[i]);
	}
	puts("");
	return 0;
}
