#include<iostream>
#include<vector>
using namespace std;

int a[4000],b[4000];
main()
{
	int n,m;
	cin >> n >> m;
	for(int i=0;i<n;i++)
		cin >> a[i];
	for(int i=0;i<m;i++)
		cin >> b[i];

	int i=0,j=0;
	while(i<n && j<m)
	{
		if(a[i]<=b[j])
			i++,j++;
		else
			j++;
	}
	cout << (n-i);
	return 0;
}
