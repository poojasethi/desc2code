//In the name of God
#include <iostream>
using namespace std;

int n, m, pt, a[3003];

int main() {
	ios_base::sync_with_stdio(false);
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	for (int i = 0; i < m; i++) {
		int b; cin >> b;
		if (b >= a[pt])
			pt++;
	}
	cout << max(0, n - pt) << '\n';
	return 0;
}
