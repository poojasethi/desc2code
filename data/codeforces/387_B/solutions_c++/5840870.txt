#include <iostream>

using namespace std;

int n,m;
int a[4000],b[4000];

int main()
{
  cin>>n>>m;
  for (int i=0;i<n;i++)
    cin>>a[i];
  for (int i=0;i<m;i++)
    cin>>b[i];
  int ia=0;
  int ib=0;
  while (ia<n and ib<m) {
    if (a[ia]<=b[ib]) {
      ia++;
      ib++;
    } else ib++;
  }
  cout<<n-ia<<endl;
}
