#include<iostream>
using namespace std;
int main()
{
	int a[1000010];
	int b[1000010];
	int n,m,count=0;
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<m;i++)
		cin>>b[i];
	for(int i=0,j=0;i<n && j<m;j++)
	{
		if(b[j]>=a[i])
		{
			count++;
			i++;
		}
	}
	cout<<n-count<<endl;
	return 0;
}
