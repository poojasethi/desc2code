#include<bits/stdc++.h>
//George and Round
using namespace std;

main(){
	int n,m;
	cin>>n>>m;
	int a[n],b[m],i,j,res=n;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<m;i++)
		cin>>b[i];
	i=j=1;
	while(m-j>=0&&n-i>=0){
		if(b[m-j]>=a[n-i]){
			res--;
			i++;
			j++;
		}
		else
			i++;
	}
	cout<<res;
}
