#include<cstdio>
int main(){
    int n,m,a[3001],i,j,x,cnt=0;
    scanf("%d%d",&n,&m);
    for(i=0;i<n;++i)scanf("%d",&a[i]);
    for(i=0,j=0;j<m;++j){
        scanf("%d",&x);
        if(x>=a[i]&&i<n){
            i++;cnt++;
        }
    }
    printf("%d\n",n-cnt);
    return 0;
}
