#include<bits/stdc++.h>

using namespace std;

int nd[3005];
int gt[3005];

main(){
int n,m;
cin>>n>>m;
for(int i=0;i<n;i++) cin>>nd[i];
for(int i=0;i<m;i++) cin>>gt[i];
int cnt=n;
for(int i=0;i<n;i++){
	for(int j=0;j<m;j++){
		if(gt[j]>=nd[i]){ 
			cnt--;
			gt[j]=0;
			break;}
		}
	}
cout<<cnt<<endl;	
}
