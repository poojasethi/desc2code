#include <iostream>
using namespace std;
int main()
{
	int n, m;
	int a[3000], b[3000];

	cin >> n >> m;
	for( int i=0; i<n; i++ )
		cin >> a[i];
	for( int i=0; i<m; i++ )
		cin >> b[i];

	int i = 0, j = 0;
	while( i<n && j<m )
	{
		if ( b[j] >= a[i] ) i++;
		j++;
	}
	cout << n - i << endl;
	return 0;
}
