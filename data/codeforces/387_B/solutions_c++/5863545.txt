
#include <stdio.h>

int main()
{
	int n, m;
	int i, j;
	int A[3000], B[3000];
	scanf("%d %d", &n, &m);
	for (i = 0; i < n; i++)
		scanf("%d", &A[i]);
	for (i = 0; i < m; i++)
		scanf("%d", &B[i]);
	for (i = 0, j = 0; i < n && j < m; j++) {
		if (A[i] <= B[j])
			i++;
	}
	printf("%d\n", n - i);
	return 0;
}
