#include <iostream>
using namespace std;

int main()
{
	int n,m;
	cin >> n >> m;
	int a[5000],b[5000];
	for (int i=0;i<n;i++) cin >> a[i];
	for (int i=0;i<m;i++) cin >> b[i];

	int j=0,i=0;
	while (j<m && i<n)
	{
		if (b[j]>=a[i])
		{
			i++;
			j++;
		}
		else j++;
	}

	cout << max(0,n-i) << endl;
}
