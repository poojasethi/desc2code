#include<stdio.h>
#include<string.h>
int a[10000],b[10000];
int main()
{
	int n,m,i,j;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		for(i=0;i<m;i++)
			scanf("%d",&b[i]);
		for(i=0,j=0;i<n&&j<m;j++)
			if(b[j]>=a[i])
			i++;
		printf("%d\n",n-i);
	}
return 0;
}
