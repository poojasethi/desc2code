#include<stdio.h>
int main()
{
	int n,m,i,j,b,num,a[3005];
	while(scanf("%d%d",&n,&m) != EOF)
	{
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		num=0;
		for(i=0,j=0;i<m;i++)
		{
			scanf("%d",&b);
			if(b>=a[j]&&j<n)
			{
				j++;
				num++;
			}
		}
		printf("%d\n",n-num);
	}
	return 0;
}
