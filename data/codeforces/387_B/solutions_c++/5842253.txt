#include<cstdio>

int a[3005],b[3005];

int main()
{
    int n,m,i,j,s=0;
    scanf("%d%d",&n,&m);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    for(j=0;j<m;j++)
        scanf("%d",&b[j]);
    for(j=i=0;j<m&&i<n;j++)
        if(b[j]>=a[i])
            i++;
    printf("%d\n",n-i);
    return 0;
}
