#coding:utf-8
#   brute force      greedy      two pointers
# 字符串匹配,贪心

n, m = list(map(int, raw_input().split(' ')))
a = list(map(int, raw_input().split(' ')))
b = list(map(int, raw_input().split(' ')))

i = 0  #a的索引index
answer = 0
for k in b:
	if i < n and k >= a[i]:
		i += 1
		answer += 1
print(n - answer)
