n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
a.sort()
b.sort()
j = 0
ans = 0
for i in range(n):
    while j < m:
        if a[i] <= b[j]:
            j += 1
            ans += 1
            break
        j += 1
    if j == m:
        break
print n - ans
