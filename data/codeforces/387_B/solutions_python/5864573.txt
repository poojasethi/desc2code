import sys


n,m=map(int,raw_input().split())
a=(map(int,raw_input().split()))
b=(map(int,raw_input().split()))

new=0
ip,jp=0,0

for i in range(n):
    while jp<m and b[jp]<a[ip]:
        jp+=1
    if jp<m and b[jp]>=a[ip]:
        jp+=1
        ip+=1
    else:
        new+=1
        
print new
