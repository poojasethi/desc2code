#include<stdio.h>
int main()
{
	__int64 n,m,i;
	__int64 sum1,sum2,sum=0;
	__int64 a,b,c,d,e;
	scanf("%I64d%I64d",&n,&m);
	for(i=1;i<=n;i++)
	{
		sum1=0;
		scanf("%I64d%I64d%I64d%I64d",&a,&b,&c,&d);
		
		if(b-a>0)
		{
		sum1=((m-1)/(b-a)+1)*d;
		sum2=m*c+d;
		if(sum2<sum1)
			sum1=sum2;
		}
		else
		{
		sum1=m*c+d;
		}
		sum+=sum1;
	}
	printf("%I64d\n",sum);
	
	
	return 0;
	
}
