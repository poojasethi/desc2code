#include <iostream>
using namespace std;
typedef long long ll;
int main(){
	ios_base::sync_with_stdio(false);
	ll n,m;
	cin >> n >> m;
	ll ans = 0;
	for(ll i = 0; i < n; ++i){
		ll t,T,x,cost;
		cin >> t >> T >> x >> cost;
		ll s = T-t;
		if(s>=m){
			ans+=cost;
			continue;
		}
		if(s<=0){
			ans+=m*x+cost;
			continue;
		}
		ll b = (m+s-1)/s;
		ans+=min(b*cost,m*x+cost);
	}
	cout << ans << '\n';
}
