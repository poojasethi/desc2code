#include<iostream>
#include<cstdio>
#define rep(i,n) for(int i=0;i<n;++i)
using namespace std;
long long n,m,t,T,x,cost,ans=0;

int main()
{
	scanf("%I64d%I64d",&n,&m);
	rep(i,n)
	{
		scanf("%I64d%I64d%I64d%I64d",&t,&T,&x,&cost);
		if(t>=T)
		{
			ans+=m*x+cost;
			continue;
		}
		long long p=T-t;
		ans+=min((m+p-1)/p*cost,m*x+cost);
	}
	cout<<ans<<endl;
	return 0;
}
