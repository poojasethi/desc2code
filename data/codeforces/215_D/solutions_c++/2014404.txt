#include<iostream>
using namespace std;

int main()
{
  int n, m;
  cin>>n>>m;
  long long ans = 0;
  for (int i = 0; i < n; i++) {
    long long t, T, x, cost, k;
    cin>>t>>T>>x>>cost;
    k = T-t;
    long long best = cost + ((m>k)?(m*x):0);
    if (k>0) best = min(best, (long long)cost*((m + k - 1)/k));
    ans += best;
  }
  cout<<ans<<endl;
  return 0;
}
