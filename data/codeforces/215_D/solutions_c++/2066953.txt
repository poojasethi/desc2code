#include <iostream>
#define LL long long
using namespace std;

int main()
{
    int n;
    LL m, t, T, x, cost;
    LL ans = 0;
    cin>>n>>m;
    while(n--)
    {
        cin>>t>>T>>x>>cost;
        if(t>=T) ans += cost+x*m;
        else ans += min( cost+x*m, (m+T-t-1)/(T-t)*cost );
    }
    cout<<ans<<endl;
    //cout << "Hello world!" << endl;
    return 0;
}
