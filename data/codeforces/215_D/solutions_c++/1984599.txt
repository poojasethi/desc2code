#include<iostream>

using namespace std;

int main(){
	long long n,m;
	int i;
	long long a,b;
	long long t,T,x,c;
	long long res;
	cin >> n >> m;
	res = 0;
	for(i = 0;i<n;i++){
		cin >> t >> T >> x >> c;
		if(m+t>T)
			a = c+x*m;
		else
			a = c;
		if(T>t){
			b = m/(T-t)+(m%(T-t)!=0);
			b = b*c;
			if(a<b)
				res += a;
			else
				res += b;
		}
		else
			res += a;
	}
	cout << res;
	return 0;
}
