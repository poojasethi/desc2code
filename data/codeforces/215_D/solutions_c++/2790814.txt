#include <iostream>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n;
long long m,t,T,x,cost,res;

int main(){

  cin>>n>>m;
  rep(i,n){
    cin>>t>>T>>x>>cost; T-=t;
    if(T<=0)res+= x*m + cost;
    else if(m<=T) res+=cost;
    else res+=min(x*m+cost,(m%T==0?m/T*cost:(m/T+1)*cost));
  }

  cout<<res<<endl;


  return 0;

}
