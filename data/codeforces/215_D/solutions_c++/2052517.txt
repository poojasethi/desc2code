#include <iostream>
#include <cstdio>
#include <cmath>
  using namespace std; int main() { long long n, m; long long t, T, x, cost; long long ans = 0; cin >> n >> m; while (n--) { cin >> t >> T >> x >> cost; if (T <= t) { ans += cost + m * x; continue; } long long count = ceil(m / (double)(T - t)); long long t1 = count * cost; long long t2 = cost + m * x; ans += min(t1, t2); } cout << ans << endl; return 0; }
