#include<iostream>
#include<cstdio>
#define ll long long
using namespace std;
int main()
{
	ll n,m,t,T,x,cost;
	cin>>n>>m;
	ll ans=0;
	for(int i=0;i<n;i++)
	{
		cin>>t>>T>>x>>cost;
		int k=0;
		if(T-t!=0)
			k=m/(T-t);
		if(k*(T-t)!=m)
			k+=1;
		if(T<=t)
			ans+=x*m+cost;
		else if(x*m+cost<cost*k)
			ans+=x*m+cost;
		else
			ans+=cost*k;
	}
	cout<<ans<<endl;
	return 0;
}
