#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
	long long n,m;
	cin>>n>>m;
	long long t,T,x,c;
	long long ans=0;
	for(long long i=0;i<n;i++)
	{
		cin>>t>>T>>x>>c;
		if(t+m<T) {ans+=c;continue;}
		if(T<=t) {ans+=c+m*x;continue;}
		long long tmp=m%(T-t)==0?m/(T-t):m/(T-t)+1;
		ans+=min(min(c+m*x,tmp*c+(m-(tmp-1)*(T-t))*x),c*tmp);	
	}
	cout<<ans<<endl;
	return 0;
}
