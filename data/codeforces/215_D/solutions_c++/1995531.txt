#include <iostream>
using namespace std;

int main()
{
	int n,m;
	cin>>n>>m;
	long long res=0;
	for(int i=0,t,T,x,cost;i<n;i++)
	{
		cin>>t>>T>>x>>cost;
		long long a=cost;
		if(t+m>T)
			a=a+(long long )m*x;
		long long b=a;
		if(T>t)
		{
			long long nBus=m/(T-t);
			if(m%(T-t))nBus++;
			b=nBus*cost;
		}
		if(a<b)
			res+=a;
		else
			res+=b;
	}
	cout<<res<<endl;
	return 0;
}
