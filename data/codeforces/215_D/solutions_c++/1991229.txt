#include<stdio.h>
#include <iostream>
using namespace std;
int main(){
	long long int n,m;
	cin>>n>>m;
	long long int t,T,x,c;
	long long int i;
	long long int cost = 0;
	for(i=0;i<n;i++){
		cin>>t>>T>>x>>c;
		if(t >= T){
			cost += x*m + c;
		}
		else{
			if(m * x + c < ((m - 1)/(T-t) + 1)*c)
				cost += m*x + c;
			else
				cost += ((m-1)/(T-t) + 1)*c;
		}
	}
	cout << cost;

}
