#include<iostream>
#include<cmath>

using namespace std;

int main(){
	long long n,m,ti,k,Ti,xi,costi,costb;
	cin>>n>>m;
	long long A[4],cost=0,cc=0,csc;
	while(n--){
		for(int i=0;i<4;i++)	cin>>A[i];
		ti=A[0]+m;
		Ti=A[1];
		xi=A[2];
		costi=A[3];
		k=m;
		if(Ti-A[0]<=0)	cost+=m*xi+costi;
		else{
			long long th=(m+Ti-A[0]-1)/(Ti-A[0]);
			csc=costi*th;
			cost+=min(m*xi+costi,csc);
		}
	}
	cout<<cost<<endl;
	return 0;
}
