#include<stdio.h>
#include<iostream>
using namespace std;


int main()
{
int n;
__int64 t,T,p,cost,sum,x,tmp;
sum=0;
scanf("%d%I64d",&n,&x);
for(int i=0;i<n;i++)
{
scanf("%I64d%I64d%I64d%I64d",&t,&T,&p,&cost);
if(x<=(T-t)) 
{
sum+=cost;
continue;
}
if(T-t<=0)
{
sum+=p*x+cost;
continue;
}
if(x%(T-t)==0)
tmp=cost*(x/(T-t));
else
tmp=cost*(x/(T-t)+1);
sum+=min(x*p+cost,tmp);
}
printf("%I64d",sum);
return 0;
}
