#include<cstdio>
#include<iostream>
#define MAX 10000

typedef long long ll;
using namespace std;

int N, M;
ll t, T, x, cost, best, total=0, nb;

int main()
{
	scanf("%d %d", &N, &M);
	for(int i=0; i<N; i++)
	{
		cin >> t >> T >> x >> cost;
		best = cost;
		if(M>T-t)
			best += M*x;
		if(T > t)
		{
			nb = (M+T-t-1)/(T-t);
			if(best>nb*cost)
				best = nb*cost;
		}
		total += best;
	}
	cout << total << endl;
}
