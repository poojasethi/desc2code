#include <iostream>
using namespace std;

typedef long long ll;

int main(){
    ll n,m;
    cin>>n>>m;
    ll t,T,x,cost;
    ll res = 0;
    for( int i = 0; i < n; i++ ){
        cin>>t>>T>>x>>cost;
        int d = T - t;
        if( m<=d )
            res+=cost;
        else if( d<=0 )
            res += cost + m*x;
        else
            res += min( cost * ( (m-1)/d + 1 ), cost + m*x  );
    }
    cout<<res;
}
