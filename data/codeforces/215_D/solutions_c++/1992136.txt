#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
int n,m,t,T,x,c;
ll ans,s;

int main(){
  scanf("%d%d",&n,&m);
  ans=0;
  while (n--){
    scanf("%d%d%d%d",&t,&T,&x,&c);
    T-=t;
    s=(ll)m*x+c;
    if (T>0) s=min(s,(ll)(m+T-1)/T*c);
    ans+=s;
  }
  printf("%I64d\n",ans);
  return 0;
}
