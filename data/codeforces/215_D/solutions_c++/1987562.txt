#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;
int main()
{
    long long int i,j,n,m,t,T,x,cost,ans=0,kol;
    cin>>n>>m;
    for(i=1;i<=n;i++)
    {
        cin>>t>>T>>x>>cost;
        kol=T-t;
        if(T-t>0){
        long long buss=(m+kol-1)/kol;
        long long s=buss*cost;
        ans+=min(m*x+cost,s);
        }
        else{
            ans+=m*x+cost;
        }
    }
    cout<<ans;
return 0;
}
