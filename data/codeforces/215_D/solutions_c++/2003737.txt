#include <cstdio>
int main() {
    int n,m;
    scanf("%d%d",&n,&m);
    long long t,T,x,c;
    long long ans=0;
    for (int i=0;i<n;i++) {
        scanf("%I64d%I64d%I64d%I64d",&t,&T,&x,&c);
        if (t>=T) ans+=x*m+c;
        else
        ans+=(m+T-t-1)/(T-t)*c<x*m+c?(m+T-t-1)/(T-t)*c:x*m+c;
    }
    printf("%I64d\n",ans);
}
