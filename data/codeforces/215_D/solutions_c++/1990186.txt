#include<cstdio>
#include<algorithm>
#include<iostream>
using namespace std;

typedef long long LL;

LL n,m,ans,t,T,cost,x,a,b;

int main()
{
	//freopen("D.in","r",stdin);freopen("D.out","w",stdout);
	cin>>n>>m;
	int i;
	for(i=1;i<=n;++i)
	{
		cin>>t>>T>>x>>cost;
		a=cost+((t+m)>T)*m*x;
		if(t<T)
		{
			b=m/(T-t);
			if((T-t)*b<m)++b;
			a=min(a,cost*b);
		}
		ans+=a;
	}
	cout<<ans<<endl;
}
