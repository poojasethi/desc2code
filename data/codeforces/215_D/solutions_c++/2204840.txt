#include<iostream>
using namespace std;
int main()
{
long long int n,m,t,T,x,cost,i,j,p=0,p1=0,p2=0;
cin>>n>>m;
for(i=0;i<n;i++)
{
cin>>t>>T>>x>>cost;
if(T-t>0)
{
if(T-t>=m)
{p+=cost;}
else
{
p1=m*x+cost;
if(m%(T-t)==0)
{
p2=(m/(T-t))*cost;
}
else
{
p2=((m/(T-t))+1)*cost;
}
if(p2>=p1)
{
p=p+p1;}
else{
p=p+p2;}
}
}
else
{
    p=p+(m*x)+cost;
}
}
cout<<p;
return(0);
}
