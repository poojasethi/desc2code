#include <stdio.h>

int main() {
	long long ans;
	int a, b, c, d, n, m;
	while (scanf("%d%d", &n, &m) == 2) {
		ans = 0;
		while (n --) {
			scanf("%d%d%d%d", &a, &b, &c, &d);
			int x = b - a;
			if (x <= 0 || (long long)((m+x-1) / x - 1) * d >= (long long)m * c) {
				ans += (long long)m * (long long)c + d;
			} else {
				ans += (m+x-1) / x * (long long)d;
			}
		}
		printf("%I64d\n", ans);
	}
	return 0;
}
