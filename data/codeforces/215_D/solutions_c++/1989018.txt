#include<stdio.h>
#include<algorithm>
#include<iostream>  
  using namespace std;
main()
{
  long long n,i,j,m,a,b,c,d,x,y,s;
   
   cin>>n>>m;
  s=0;
  for(i=1;i<=n;i++)
  {
    cin>>a>>b>>c>>d;
    x=100000000000000000;
    if(b-a>0)
    {
    x=m/(b-a);
    x*=d;
    if(m%(b-a)>0)x+=d;
    }
    y=d;
    if(m+a>b)y=y+m*c;
    s+=min(y,x);
  }
  cout<<s;
}
