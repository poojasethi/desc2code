#include <iostream>
#include <cstdio>

using namespace std;
int main()
{
    int i,j,n;
    long long cost=0,now,m,t,t_,d,x,c,y,z;
    for (i=1,scanf("%d%I64d",&n,&m);i<=n;i++)
    {
        scanf("%I64d%I64d%I64d%I64d",&t,&t_,&x,&c),d=t_-t;
        now=x*m+c;
        if (d>0) {
            z=m/d+(m%d>0),y=z*c;
            if (now>y) now=y;
        }
        cost+=now;
    }
    printf("%I64d\n",cost);
    return 0;
}
