#include<iostream>
#include<cstdio>
using namespace std;
#define ll __int64
int main()
{
	ll n,m,t,T,x,C,s,p;
	while(scanf("%I64d%I64d",&n,&m)!=EOF)
	{
		s=0;
	    while(n--)
		{
		scanf("%I64d%I64d%I64d%I64d",&t,&T,&x,&C);
		p=T-t;
		if(T-t>=m)s+=C;
		else if(p<=0)s+=C+m*x;
		else if((m-1)/(T-t)*C<m*x)s+=(m-1)/(T-t)*C+C;
		else s+=m*x+C;
		}
		printf("%I64d\n",s);
	}
	return 0;
}
