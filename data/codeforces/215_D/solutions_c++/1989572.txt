#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
typedef long long LL;
LL n,m,t,T,x,cost,ans=0;

int main()
{
	cin>>n>>m;
	for (LL i=0;i<n;i++)
	{
		cin>>t>>T>>x>>cost;
		if (T>t)
		{
			LL tmp=m/(T-t);
			if (tmp*(T-t)<m) tmp++;
			ans+=min(cost*tmp,cost+m*x); 
		}else ans+=cost+m*x;
	}
	cout<<ans<<endl;
	return 0;
}
