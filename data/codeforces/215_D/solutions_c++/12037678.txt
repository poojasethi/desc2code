#include <bits/stdc++.h>
using namespace std;
int main()
{
	__int64 n, m, t, T, x, cost, ans = 0;
	scanf("%I64d%I64d", &n, &m);
	for (int i = 0; i < n; i++)
	{
		scanf("%I64d%I64d%I64d%I64d", &t, &T, &x, &cost);
		__int64 v = m*x + cost, u;
		if (T - t>0)
			u = (m / (T - t) + (m % (T - t) ? 1 : 0))*cost;
		else u = v;
		ans += min(u, v);
	}
	printf("%I64d\n", ans);
	return 0;
}
