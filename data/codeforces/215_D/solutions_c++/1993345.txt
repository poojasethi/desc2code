#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
long long n,m,s,t,T,x,c;
int main(){
    cin>>n>>m;
    for(int i=0;i<n;i++){
        cin>>t>>T>>x>>c;
        if(T<=t) s+=c+x*m;
        else s+=min(c+x*m,c*((m-1)/(T-t)+1));
    }
    cout<<s;
}
