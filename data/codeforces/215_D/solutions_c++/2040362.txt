#include<stdio.h>
long long min(long long a,long long b)
{
    return a<b?a:b;
}
int main()
{
    int i,n;
    long long s,t,x,tt,c,m;
    while(scanf("%d%I64d",&n,&m)!=EOF)
    {
        s=0;
        while(n--)
        {
            scanf("%I64d%I64d%I64d%I64d",&t,&tt,&x,&c);
            if(tt<=t)s+=m*x+c;
            else s+=min(m*x+c,c*((m-1)/(tt-t)+1));
        }
        printf("%I64d\n",s);
    }
    return 0;
}
