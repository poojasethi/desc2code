#include <iostream>
using namespace std;
int main(){
  //freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
  int i,n;
  long long m,k,t,T,x,c,s=0,d1,d2;
  cin >> n >> m;
  for(i=0;i<n;i++){
    cin >> t >> T >> x >> c;
    d1=c;if(t+m>T)d1+=m*x;
    d2=d1+1;
    if(T>t){k=1+(m-1)/(T-t);d2=k*c;}
    if(d1<d2)s+=d1;else s+=d2;
  }
  cout << s;
return 0;
}
