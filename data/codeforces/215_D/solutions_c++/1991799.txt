#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int main()
{
	long long n,m,s=0;
	scanf("%I64d %I64d",&n,&m);
	for(int i=0;i<n;i++) {
		long long t,T,x,cost,p;
		scanf("%I64d %I64d %I64d %I64d",&t,&T,&x,&cost);
		p=T-t;
		if (p<=0) s+=m*x+cost;
		else s+=min(m*x+cost,(m+p-1)/p*cost);
	}
	printf("%I64d\n",s);
}
