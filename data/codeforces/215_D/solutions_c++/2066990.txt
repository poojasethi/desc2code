#include<algorithm>
#include<cstdio>
using namespace std;
int main()
{
	__int64 n,m,s=0;
	__int64 ans=0;
	scanf("%I64d %I64d",&n,&m);
	for(int i=0;i<n;i++)
	{
		__int64 t,T,x,cost,p;
		scanf("%I64d %I64d %I64d %I64d",&t,&T,&x,&cost);
		p=T-t;
		if (p<=0) ans+=m*x+cost;
		else ans+=min(m*x+cost,(m+p-1)/p*cost);
	}
	printf("%I64d\n",ans);
}
