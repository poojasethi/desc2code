#include <cstdio>
#define lint long long

using namespace std;

int main()
{
	lint n, m, ans=0, i, tmp, t, T, c, x;
	scanf("%I64d%I64d", &n, &m);
	for (i=1; i<=n; i++) {
		scanf("%I64d%I64d%I64d%I64d", &t, &T, &x, &c);
		tmp = c + m*(t+m>T ? x : 0);
		if (t<T && (m+T-t-1)/(T-t)*c<tmp)
			tmp = (m+T-t-1)/(T-t)*c;
		ans += tmp;
	}
	printf("%I64d", ans);
	return 0;
}
