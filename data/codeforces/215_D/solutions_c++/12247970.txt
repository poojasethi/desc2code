#include<iostream>
#include<algorithm>
using namespace std;

long long n, m, t, T, x, cost, c, cars, cool, hot, ans;

int main() {
	cin>>n>>m;
	ans = 0LL;
	for(int i=0;i<n;i++) {
		cin>>t>>T>>x>>cost;
		c = T - t;
		
		if(c<=0) ans += 1LL * x * m + cost;
		else {
			cars = (m-1)/c+1;
			cool = 1LL*cars*cost;
			hot = 1LL * x * m + cost;
			ans += min(cool, hot);
		}
	}
	cout<<ans<<endl;
	
	return 0;
}
