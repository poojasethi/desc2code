#include<cstdio>
#include<iostream>
using namespace std;
int main(int argc,char**argv){
	long long n, k;
	scanf("%I64d %I64d",&n,&k);
	long long count = 0;
	for(int i=0; i<n; i++){
		long long t, T, c, C;
		scanf("%I64d %I64d %I64d %I64d",&t,&T,&c,&C);
		if(t>=T){
			count += C+k*c;
		}
		else {
			count += min(C+k*c, ((k-1)/(T-t)+1)*C);
		}
	}
	printf("%I64d\n",count);
	return 0;
}
