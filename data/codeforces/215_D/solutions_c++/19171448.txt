#include<cstdio>
#define ll __int64
ll min(ll a,ll b){return a<b?a:b;}
int main()
{
	ll x,y,ans=0;
	int a,b,c,n,m;
	scanf("%d%d",&n,&m);
	while(n--)
	{
		scanf("%d%d%I64d%I64d",&a,&b,&x,&y);
		if(b<=a)
		{
			ans+=m*x+y;
			continue;
		}
		c=m/(b-a)+(m%(b-a)!=0);
		if(c==1)ans+=y;
		else ans+=min(c*y,min(m*x+y,(m-(c-2)*(b-a))*x+(c-1)*y));
	}
	printf("%I64d\n",ans);
	return 0;
}
