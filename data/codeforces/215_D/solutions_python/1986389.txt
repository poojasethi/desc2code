import sys
from math import ceil, floor

def main():
    n, m = map(int,sys.stdin.readline().strip().split())
    total = 0

    def check(xi,Ti,ti, costi):
        if Ti <= ti:
            return costi + m*xi
        return min(m*xi + costi, ceil(float(m)/(Ti-ti))*costi)

    for i in range(n):
        ti, Ti, xi, costi = map(int,sys.stdin.readline().strip().split())
        total+=int(check(xi,Ti,ti,costi))

    return int(total)

if __name__ == "__main__":
    print main()
