n,m=map(int,raw_input().split())
ret=0
i=1
while i<=n:
    i+=1
    t1,t2,p,b=map(int,raw_input().split())
    if t2<=t1:
        ret+=b+m*p
        continue
    else:
        p1=(m+t2-t1-1)/(t2-t1)
        res=p1*b
        a1=b-(t2-t1)*p
        b1=b+m*p
        if a1<0:b1=a1*(p1-1)+b1
    if (res<b1):ret+=res
    else: ret+=b1
print ret
