read_ints = lambda : map(int, raw_input().split())

def main():
    n, m = read_ints()
    ans = 0
    for i in xrange(n):
        t, T, x, cost = read_ints()
        p1 = cost if t + m <= T else cost + m * x
        p2 = cost * ((m + T - t - 1) / (T - t)) if T > t else p1
        ans += min(p1, p2)
    print ans

if __name__ == '__main__':
    main()
