n,m=map(int,raw_input().split())
ob=[map(int,raw_input().split()) for i in xrange(n)]
rs=0
for t,T,x,c in ob:
    if t+m<=T:
        nn=c
    elif t>=T:
        nn=c+x*m  
    else:
        f=c+x*m
        ma=T-t
        s=((m+ma-1)/ma)*c        
        nn=min(f,s)
    rs+=nn
print rs
