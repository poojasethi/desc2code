n,m = map(int,raw_input().split())

sum = 0;l=0

for i in range(n):
    t,T,x,c = map(int,raw_input().split())
    if T <= t:
        s = c + m*x
    else:
        if m%(T-t)==0:
            g=m/(T-t)
        else:
            g=m/(T-t)+1;
            
        s = min(c + m*x,g*c )
        
   
    l += s

print l
