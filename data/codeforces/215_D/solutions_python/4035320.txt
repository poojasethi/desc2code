from sys import stdin

def main():
    n, m = map(int, stdin.readline().split())
    ans = 0
    for line in stdin.readlines():
        t, T, x, cost = map(int, line.split())
        p1 = cost if t + m <= T else cost + m * x
        p2 = cost * ((m + T - t - 1) / (T - t)) if T > t else p1
        ans += min(p1, p2)
    print ans

if __name__ == '__main__':
    main()
