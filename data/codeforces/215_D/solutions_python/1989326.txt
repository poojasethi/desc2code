n, m = raw_input().split(' ')
n, m = int(n), int(m)
ans = 0
for i in range(n):
	t, T, x, cost = raw_input().split(' ')
	t, T, x, cost = int(t), int(T), int(x), int(cost)
	if t >= T:
		ans += cost + x * m
	else:
		v1 = cost
		if t + m > T: v1 += x * m
		v2 = ((m - 1) / (T - t) + 1) * cost
		ans += min(v1, v2)
print ans
