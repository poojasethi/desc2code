n, m = [int(i) for i in raw_input().split()]

all = 0
for i in xrange(n):
    t, T, x, cost = [int(i) for i in raw_input().split()]
    if t+m <= T:
        all += cost
    elif T <= t:
        all += cost + x*m
    else:
        div = m / (T-t)
        mod = m % (T-t)
        p1 = cost*(div+(0 if mod == 0 else 1))
        p2 = cost*div+x*((T-t)+mod)
        p3 = cost+x*m
        all += min(p1, p2, p3)
print all
