#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const long long inf=10000000000000;
long long n,m,i,j,k,l,o,p,x,y;
long long len[14],a[2][2],ans[2][2],tmp[2][2],w[2][1000000],tot[2];
long long quick(long long x,long long y)
{
	long long ans=0;
	while(y)
	{
		if(y&1) ans=(ans+x)%inf;
		x=(x+x)%inf;
		y/=2;
	}
	return ans;
}
void mul(long long a[][2],long long b[][2])
{
    int i,j,k;
    memset(tmp,0,sizeof(tmp));
    for(i=0;i<2;i++)
        for(j=0;j<2;j++)
            for(k=0;k<2;k++) tmp[i][j]=(tmp[i][j]+quick(a[i][k],b[k][j]))%inf;
    for(i=0;i<2;i++)
        for(j=0;j<2;j++) a[i][j]=tmp[i][j];
}
long long calc(long long x)
{
    if(x<=1) return x;
    memset(a,0,sizeof(a));
    memset(ans,0,sizeof(ans));
    a[0][0]=1;
    a[0][1]=1;
    a[1][0]=1;
    ans[0][0]=1;
    ans[1][1]=1;
    x--;
    while(x)
    {
        if(x&1) mul(ans,a);
        mul(a,a);
        x/=2;
    }
    return ans[0][0];
}
int main()
{
    o=1;
    len[0]=1;
    cin >> x;
    for(i=1;i<=13;i++)
    {
        o*=10;
        j=len[i-1]+1;
        while(1)
        {
            k=calc(j)%o;
            l=calc(j-1)%o;
            if(k==1 && l==0) break;
            j+=len[i-1];
        }
        len[i]=j-1;
    }
    tot[0]=1;
    w[0][1]=0;
    p=1;
    for(i=1;i<=13;i++)
    {
        p*=10;
        o=i&1;
        tot[o]=0;
        l=0;
        while(1)
        {
            for(j=1;j<=tot[1-o];j++)
            {
                if(w[1-o][j]+l>len[i]) break;
                y=calc(w[1-o][j]+l);
                if(x==y)
                {
                    cout << w[1-o][j]+l << endl;
                    return 0;
                }
                if(x%p==y%p) w[o][++tot[o]]=w[1-o][j]+l;
            }
            if(w[1-o][j]+l>len[i]) break;
            l+=len[i-1];
        }
    }
    cout << -1 << endl;
    return 0;
}
