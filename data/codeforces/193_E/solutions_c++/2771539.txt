#include<cstdio>
#include<cstring>
#include<cmath>
#define fo(i,a,b) for(i=a;i<=b;++i)
typedef long long LL;
const int mn=1050;
LL t[2][2]={{1,1},{1,0}};
LL v,mo,f[2][2],c[2][2],x[2][2];
LL s[mn],s1[mn],h,L,ans;
int n,m,i,j,k;
LL S(LL x,LL y,const LL n)
{
    LL T=floor(sqrt(n)+0.5);
    LL t=T*T-n;
    LL a=x/T,b=x%T,c=y/T,d=y%T;
    LL e=a*c/T,f=a*c%T;
    LL v=((a*d+b*c)%n+e*t)%n;
    LL g=v/T,h=v%T;
    LL ans=(((f+g)*t%n+b*d%n)+h*T)%n;
    if(ans<0)ans+=n;
    return ans;
}
void mul(LL a[][2],LL b[][2],LL c[][2])
{
    c[0][0]=(S(a[0][0],b[0][0],mo)+S(a[0][1],b[1][0],mo))%mo;
    c[0][1]=(S(a[0][0],b[0][1],mo)+S(a[0][1],b[1][1],mo))%mo;
    c[1][0]=(S(a[1][0],b[0][0],mo)+S(a[1][1],b[1][0],mo))%mo;
    c[1][1]=(S(a[1][0],b[0][1],mo)+S(a[1][1],b[1][1],mo))%mo;
}
void dg(LL n)
{
    if(n==1)return;
    dg(n/2);
    mul(x,x,c);
    if(n%2)mul(c,t,x);
    else memcpy(x,c,sizeof c);
}
LL query(LL n)
{
    if(n==0)return 0;
    memset(f,0,sizeof f);
    f[0][0]=1,f[1][0]=0;
    memcpy(x,t,sizeof t);
    dg(n);
    mul(x,f,c);
    return c[1][0];
}
int main()
{
    scanf("%I64d",&v);
    L=60,mo=10;
    fo(i,0,59)
        if(query(i)==v%mo)s[++n]=i;
    fo(i,1,12)
    {
        mo*=10;
        m=0;
        fo(j,1,n)
            fo(k,0,9)
            {
                h=k*L+s[j];
                if(query(h)==v%mo)s1[++m]=h;
            }
        L*=10,n=m,memcpy(s,s1,sizeof s1);
    }
    if(!n){printf("-1\n");return 0;}
    ans=s[1];
    fo(i,2,n)if(s[i]<ans)ans=s[i];
    printf("%I64d\n",ans);
    return 0;
}
