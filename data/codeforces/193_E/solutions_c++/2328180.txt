#include<cstdio>
#include<cstring>
#define fo(i,a,b) for(i=a;i<=b;++i)
typedef long long LL;
const int mn=1050;
LL t[2][2]={{1,1},{1,0}};
LL v,mo,f[2][2],c[2][2],x[2][2];
LL s[mn],s1[mn],h,L,ans;
int n,m,i,j,k;
LL S(LL a,LL b)
{
    LL res=0,L=1LL<<60;
    while(L)
    {
        res=res*2%mo;
        if((b&L)!=0)res=(res+a)%mo;
        L>>=1;
    }
    return res;
}
void mul(LL a[][2],LL b[][2],LL c[][2])
{
    int i,j,k;
    fo(i,0,1)
        fo(j,0,1)
        {
            c[i][j]=0;
            fo(k,0,1)c[i][j]=(c[i][j]+S(a[i][k],b[k][j]))%mo;
        }
}
void dg(LL n)
{
    if(n==1)return;
    dg(n/2);
    mul(x,x,c);
    if(n%2)mul(c,t,x);
    else memcpy(x,c,sizeof c);
}
LL query(LL n)
{
    if(n==0)return 0;
    memset(f,0,sizeof f);
    f[0][0]=1,f[1][0]=0;
    memcpy(x,t,sizeof t);
    dg(n);
    mul(x,f,c);
    return c[1][0];
}
int main()
{
    scanf("%I64d",&v);
    L=60,mo=10;
    fo(i,0,59)
        if(query(i)==v%mo)s[++n]=i;
    fo(i,1,12)
    {
        mo*=10;
        m=0;
        fo(j,1,n)
            fo(k,0,9)
            {
                h=k*L+s[j];
                if(query(h)==v%mo)s1[++m]=h;
            }
        L*=10,n=m,memcpy(s,s1,sizeof s1);
    }
    if(!n){printf("-1\n");return 0;}
    ans=s[1];
    fo(i,2,n)if(s[i]<ans)ans=s[i];
    printf("%I64d\n",ans);
    return 0;
}
