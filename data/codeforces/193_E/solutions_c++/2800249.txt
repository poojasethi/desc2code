#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
inline int   read()
{
	int x = 0; char ch = getchar(); bool positive = 1;
	for (; ch < '0' || ch > '9'; ch = getchar())	if (ch == '-')  positive = 0;
	for (; ch >= '0' && ch <= '9'; ch = getchar())	x = x * 10 + ch - '0';
	return positive ? x : -x;
}
#include <vector>
typedef pair<LL, LL> PLL;
#define x first
#define y second

vector<LL> a[15];
LL n, ans = -1, p = 10, L = 60, key = 65535;

LL mul(LL x, LL y)
{
	LL res = 0;
	for (int i = 0; i < 3; ++ i)
		res += (x & key) * y % p, x >>= 16, y = (y << 16) % p;
	return res % p;
}

PLL cal(LL n)
{
	if (n == 1) return make_pair(0, 1);
	PLL A = cal(n & 1 ? n - 1 : n / 2);
	if (n & 1) return make_pair(A.y, (A.x + A.y) % p);
	return make_pair((mul(A.x, A.x) + mul(A.y, A.y)) % p, mul(A.x * 2 + A.y, A.y) % p);
}

int main()
{
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
	freopen("a.out", "w", stdout);
#endif
	
	cin >> n;
	for (int i = 1; i <= 60; i++) if (cal(i).x == n % p) a[0].push_back(i);
	for (int t = 0; t < 12; t++)
	{
		p *= 10;
		for (int k = 0; k < a[t].size(); ++ k)
		{
			LL i = a[t][k];
			for (LL j = 0; j < 10; j++)
				if (cal(L * j + i).x == n % p) a[t+1].push_back(L * j + i);
		}
		L *= 10;
	}
	for (int k = 0; k < a[12].size(); ++ k)
	{
		LL i = a[12][k];
		ans = (ans < 0) || (i - 1 < ans) ? i - 1 : ans;
	}
	cout << ans << endl;
	
	return 0;
}
