long long key=65535;

long long mul(long long x, long long y)
{
	long long res=0;
	for (long long i=0; i<3; ++i)
		res+=(x&key)*y%mod, x>>=16, y=(y<<16)%mod;
	return res%mod;
}

pair<long long,long long> cal(long long n)
{
	if(n==1) return make_pair(0,1);
	pair<long long,long long> a;
	if(n&1)
		a=cal(n-1);
   	else 
		a=cal(n/2);
	if(n&1)
	   	return make_pair(a.second,(a.first+a.second)%mod);
	return make_pair((mul(a.first,a.first)+mul(a.second,a.second))%mod,mul(a.first*2+a.second,a.second)%mod);
}
