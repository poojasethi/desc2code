global p
def F(x):
	if x==0:
		return 0,1
	elif x&1:
		a,b=F(x-1)
		return b,(a+b)%p
	else:
		a,b=F(x/2)
		return a*(2*b-a)%p,(a*a+b*b)%p
w=input()
p=10
l=60
z=[]
for i in range(l):
	if F(i)[0]==w%p:
		z.append(i)
for i in range(12):
	p*=10
	z=[i+j*l for i in z for j in range(10) if F(i+j*l)[0]==w%p]
	l*=10
if z==[]:
	print -1
else:
	print min(z)
