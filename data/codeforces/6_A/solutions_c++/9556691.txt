#include<bits/stdc++.h>
using namespace std;

main() {
    int a[4];
    ios_base::sync_with_stdio(0);
    for(int i=0; i<4; i++) cin>>a[i];
    sort(a, a+4);
    int m=max(a[0]+a[1]-a[2], a[1]+a[2]-a[3]);
    m>0 ? puts("TRIANGLE") : m ? puts("IMPOSSIBLE") : puts("SEGMENT");
}
