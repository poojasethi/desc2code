#include <iostream>

using namespace std;

int main() {
  int HPY, ATKY, DEFY, HPM, ATKM, DEFM, h, a, d, H, A, D;

  cin >> HPY >> ATKY >> DEFY >> HPM >> ATKM >> DEFM >> h >> a >> d;
  
  int best = 10000000;

  for (int D=0; D<=100; D++) 
    for (int A=0; A<=200; A++) {
      int DM = max(0,ATKY+A-DEFM);
      if (DM>0) {
	int DY = max(0,ATKM-DEFY-D);
	int s = (HPM+DM-1)/DM;
	int H = max(0,s*DY-HPY+1);
	best = min(best,a*A + d*D + h*H);
      };
    };

  cout << best << endl;

  return 0;
};
