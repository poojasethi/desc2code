#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	long long hm,am,dm,hy,ay,dy;
	cin>>hy>>ay>>dy>>hm>>am>>dm;
	long long h,a,d;
	cin>>h>>a>>d;
	long long e1=ay-dm,e2=am-dy;
	long long fans=1000000000000000000ll;
	for(long long t=1;t<=101;t++)
	{
		long long reqe1=(hm+t-1)/t; 
		for(long long reqe2=0;reqe2<=101;++reqe2)
		{
			long long lans=0;
			long long reqh = 1+t*reqe2;
			lans=max(reqh-hy,0ll)*h+max(reqe1-e1,0ll)*a+max(e2-reqe2,0ll)*d;
			fans=min(fans,lans);
		}
	}	
	cout<<fans;
	return 0;
}
