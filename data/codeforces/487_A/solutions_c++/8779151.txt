#include <iostream>
using namespace std;
int main() 
{
    int h1,a1,d1,h2,a2,d2,h3,a3,d3,i,ans=1000000000,h,a,d,pat;
    cin>>h1>>a1>>d1>>h2>>a2>>d2>>h3>>a3>>d3;
    for (i=1; i<=100; i++)
    {
        pat=1000000000;
        a=(h2+i-1)/i+d2;
        for (d=0; d<=100; d++)
        {
            h=max((a2-d),0)*i+1;
            pat=min(pat,max(h-h1,0)*h3+max(d-d1,0)*d3);
        }
        ans=min(ans,pat+max((a-a1),0)*a3);
    }
    cout<<ans<<endl;
    return 0;
}
