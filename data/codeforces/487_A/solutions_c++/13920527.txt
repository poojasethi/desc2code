#include<cstdio>
#include<algorithm>
using namespace std;
int f(int a,int b){return (a+b-1)/b;}
int main()
{
    int h1,a1,d1,h2,a2,d2,a,b,c,o=1000000000;
    scanf("%d%d%d%d%d%d%d%d%d",&h1,&a1,&d1,&h2,&a2,&d2,&a,&b,&c);
    for(int i=max(a1,d2+1);i<201;++i)for(int j=d1;j<101;++j)
        o=min(o,max(f(h2,i-d2)*max(a2-j,0)+1-h1,0)*a+(i-a1)*b+(j-d1)*c);
    printf("%d\n",o);
}
