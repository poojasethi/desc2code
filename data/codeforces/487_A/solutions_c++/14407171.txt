#include<bits/stdc++.h>
using namespace std;
int yh,ya,yd,mh,ma,md,h,a,d;
int main() {
	int i,j,k,r,b,m=1<<30;
	cin>>yh>>ya>>yd>>mh>>ma>>md>>h>>a>>d;
	for(i=ya; i<=200; i++) for(j=yd; j<=100; j++) {
		if(i<=md) continue;
		r= ceil((double)mh / (i-md));
		k= max((ma-j)*r+1, yh);
		m= min(m, (k-yh)*h + (i-ya)*a + (j-yd)*d);
	}
	cout<<m;
}
