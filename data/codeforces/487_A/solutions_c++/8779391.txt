#include<stdio.h>
#include<algorithm>
using namespace std;
int hy,ay,dy,hm,am,dm,h,a,d,ans=1e9;
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j;
	scanf("%d%d%d%d%d%d%d%d%d",&hy,&ay,&dy,&hm,&am,&dm,&h,&a,&d);
	for(i=max(ay,dm+1);i<=max(ay,hm+dm);i++)
	{
		int t=(hm-1)/(i-dm)+1;
		for(j=dy;j<=max(dy,am);j++)
			ans=min(ans,(i-ay)*a+(j-dy)*d+max(0,t*(am-j)-hy+1)*h);
	}
	printf("%d\n",ans);
	return 0;
}
