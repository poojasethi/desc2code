#include<bits/stdc++.h>
using namespace std;


int main()
{
	int hx,ax,bx,hy,ay,by;
	int ph,pa,pb;
	int i,j,k,ta,tb,th,price=2e9,moves;
	cin>>hx>>ax>>bx>>hy>>ay>>by>>ph>>pa>>pb;
	for(i=0;i<=200;i++)
	{
		for(j=0;j<=200;j++)
		{
			ta=ax+i;
			tb=bx+j;
			if(ta-by<=0) continue;
			moves=hy/(ta-by);
			if(hy%(ta-by)) moves++;
			if(moves*(ay-tb)-hx>-1) th=moves*(ay-tb)-hx+1; else th=0;
			price=min(price,th*ph+i*pa+j*pb);
		}
	}
	cout<<price<<endl;
}
