#include<bits/stdc++.h>
using namespace std;
int a[3][3],m=8e7,i,j,r,l;
double k;
int main()
{
	ios::sync_with_stdio(false);cin.tie(0);
	for(i=0;i<3;i++)
		for(j=0;j<3;j++)
			cin>>a[i][j];
    for(i=a[0][1];i<=200;i++)
     for(j=a[0][2];j<=100;j++)
       {
      	 if(i<=a[1][2])continue;
         k=ceil((double)a[1][0]/(i-a[1][2]));l=(int )k;
         r=max(a[0][0],(a[1][1]-j)*l+1);
         m=min(m,(r-a[0][0])*a[2][0]+(i-a[0][1])*a[2][1]+(j-a[0][2])*a[2][2]);
       }   
     cout<<m;
	return 0;
}
