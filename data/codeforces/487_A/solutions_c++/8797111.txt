#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int main()
{
    int H1,H2,A1,A2,D1,D2,h,a,d,ans=0x7fffffff;
    scanf("%d%d%d%d%d%d%d%d%d",&H1,&A1,&D1,&H2,&A2,&D2,&h,&a,&d);
    for (int A=max(A1,D2+1);A<=D2+H2||A==A1;A++)
        for (int D=D1;D<=A2||D==D1;D++)
        {
            int H=max(H1,(A2-D)*((H2-1)/(A-D2)+1)+1);
            ans=min(ans,(A-A1)*a+(D-D1)*d+(H-H1)*h);
        }
    printf("%d\n",ans);
    return 0;
}
