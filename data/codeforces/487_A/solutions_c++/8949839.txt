#include<cmath>
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int hy,ay,dy,hm,am,dm,h,a,d,ans=1e9;
int main()
{
	cin>>hy>>ay>>dy;
	cin>>hm>>am>>dm;
	cin>>h>>a>>d;
	for(int i=0;i<=300;i++)
		for(int j=0;j<=300;j++)
			if(ay+i>dm){
				int t=hm/(ay+i-dm)+int(hm%(ay+i-dm)>0);
				int ct=t*max(0,am-(dy+j)),p=0;
				if(ct>=hy)p=ct-hy+1;
				ans=min(ans,h*p+a*i+d*j);
			}
	cout<<ans;return 0;
}
