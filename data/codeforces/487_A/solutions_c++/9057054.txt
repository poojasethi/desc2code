#include <bits/stdc++.h>
using namespace std;
int yh, ya, yd, mh, ma, md, bh, ba, bd;

int main() {
//    freopen("input.txt", "r", stdin);
    int i, j, r, h, m = 987654321;
    cin >> yh >> ya >> yd >> mh >> ma >> md >> bh >> ba >> bd;
    for (i = ya; i <= 200; i++) {
        if (i <= md) continue;
		r = ceil((double)mh / (i - md));
        for (j = yd; j <= 100; j++) {
            h = max(yh, r * (ma - j) + 1);
            m = min(m, (h - yh) * bh + (i - ya) * ba + (j - yd) * bd);
        }
    }
	cout << m;
}
