#include<bits/stdc++.h>
using namespace std;
int h1,h2,a1,a2,d1,d2,ph,pa,pd,f,s,d,ans=1e9;
int main(){
    cin>>h1>>a1>>d1>>h2>>a2>>d2>>ph>>pa>>pd;
    for(int i=a1;i<201;i++){
        for(int j=d1;j<101;j++){
            if(i>d2){
            f=ceil((long double)h2/(i-d2));
            d=((a2-j)*f+1)-h1;
            d=max(0,d);
            s=(d*ph)+((i-a1)*pa)+((j-d1)*pd);
            ans=min(ans,s);
           }
        }
    }
    cout<<ans<<endl;
}
