#include<bits/stdc++.h>
using namespace std;
int main()
{
	int a,b,c,p,q,r,x,y,z,t;
	cin>>a>>b>>c;
	cin>>p>>q>>r;
	cin>>x>>y>>z;
	int i,j,m,n,mini=100000000;
	for(i=0;i<=200;i++)
	{
		for(j=0;j<=100;j++)
		{
			if(i+b<=r)
				continue;
			else
			{
				t=(p-1)/(b+i-r) +1;
				n=a-t*(q-j-c);
			}	
			if(n<=0)
				mini=min(mini,(-n+1)*x+y*i+z*j);
			else
				mini=min(mini,y*i+z*j);	
		}
	}
	cout<<mini<<endl;
	return 0;
}
