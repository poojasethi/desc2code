#include <bits/stdc++.h>
using namespace std;
struct T{
    int h,a,d;
};
int main(){
    int h,a,d,hm,am,dm,hy,ay,dy,i,j,k,ans=2000000000,x,y,z;
    cin>>hy>>ay>>dy>>hm>>am>>dm>>h>>a>>d;
    for(i=dy;i<=max(am,dy);i++){
        for(j=max(ay,dm+1);j<=200;j++){
            k=j-dm;
            x=((hm+k-1)/k);
            y=max(hy,1+x*(am-i));
            y-=hy;
            ans=min(ans,y*h+(i-dy)*d+(j-ay)*a);
        }
    }
    cout<<ans<<"\n";
    return 0;
}
