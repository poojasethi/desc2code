h1,a1,d1=map(int,raw_input().split())
h2,a2,d2=map(int,raw_input().split())
h,a,d=map(int,raw_input().split())
ans=10**9
for i in range(max(0,d2-a1+1),500):
    for j in range(500):
        if a2-d1-j<=0:
            ans=min(ans,j*d+i*a)
        else:
            n=h1-(a2-d1-j)*(1+(h2-1)/(a1+i-d2))
            if n>0:
                ans=min(ans,j*d+i*a)
            else:
                ans=min(ans,j*d+i*a+h*(1-n))
print ans
