R=lambda:map(int,raw_input().split())
hy,ay,dy=R()
hm,am,dm=R()
h,a,d=R()
t=1<<60
for da in range(210):
  for dd in range(123):
    if ay+da>dm:
      if am>dy+dd:
        r=(hm+ay+da-dm-1)/(ay+da-dm)
        y=r*(am-dy-dd)+1
        t=min(t,max(0,y-hy)*h+da*a+dd*d)
      else:
        t=min(t,da*a+dd*d)
print t
