#!/usr/bin/python
# coding: utf-8

R = lambda : map(int, raw_input().split())

class attributes:
  def __init__(self):
    self.HP, self.ATK, self.DEF = R()
Yang = attributes()
Monster = attributes()
Cost = attributes()
initCost = 0
ans = 50005
if Yang.ATK <= Monster.DEF:
  initCost = (Monster.DEF + 1 - Yang.ATK) * Cost.ATK
  Yang.ATK = Monster.DEF + 1
if Yang.ATK >= Monster.DEF + Monster.HP:
  Yang.ATK = Monster.DEF + Monster.HP
if Yang.DEF >= Monster.ATK:
  Yang.DEF = Monster.ATK
for ATK in xrange(Yang.ATK, Monster.DEF + Monster.HP + 1):
  for DEF in xrange(Yang.DEF, Monster.ATK + 1):
    res = (ATK - Yang.ATK) * Cost.ATK + (DEF - Yang.DEF) * Cost.DEF
    HP = (Monster.HP + ATK - Monster.DEF - 1) / (ATK - Monster.DEF) * (Monster.ATK - DEF) + 1
    if HP > Yang.HP:
      res += (HP - Yang.HP) * Cost.HP
    if res <= ans:
      ans = res
print ans + initCost
