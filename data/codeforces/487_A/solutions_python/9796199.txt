import sys
h,a,d=map(int,raw_input().split())
H,A,D=map(int,raw_input().split())
hm,am,dm=map(int,raw_input().split())
cost=100000000
for ATK in range(max(a,D+1),201):
    n,r= divmod(H, ATK-D)
    if(r):
        n+=1
    hurt=n*A
    for DEF in range(d,201):
        HP=0
        if(DEF*n+h>hurt):
            HP=h
        else:
            HP=hurt-DEF*n+1
        temp_cost=(DEF-d)*dm+(ATK-a)*am+(HP-h)*hm
        if(temp_cost<cost):
            cost=temp_cost
print(cost)
