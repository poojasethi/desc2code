#include<bits/stdc++.h>
using namespace std;
typedef long long Int;
bitset<2002>a[2002],bs[2002];
int base[2002];
char s[2002];
int num[2002];
int n;
int main()
{
    scanf("%d",&n);
    memset(base,-1,sizeof(base));
    for(int i=0;i<n;i++)
    {
        scanf("%s",s);
        int ls=strlen(s);
        reverse(s,s+ls);
        int len=ls/8+1;
        for(int j=0;j<len;j++)
        {
            num[j]=0;
            for(int k=min(j*8+7,ls-1);k>=j*8;k--)
                num[j]=(num[j]<<3)+(num[j]<<1)+s[k]-'0';
        }
        int tot=0;
        while(len)
        {
            int rest=0;
            for(int j=len-1;j>=0;j--)
            {
                Int t=num[j]+rest*(Int)100000000; 
                rest=t%(1<<25);
                num[j]=t/(1<<25);
            }
            while(len&&num[len-1]==0)len--;
            for(int j=0;j<25;j++)a[i][tot++]=(rest>>j)&1;
        }
        bool flag=true;
        for(int j=0;j<2000;j++)
        {
            if(a[i][j])
            {
                if(base[j]<0)
                {
                    flag=false;
                    base[j]=i;
                    bs[i][i]=1;
                    break;
                }
                a[i]^=a[base[j]];
                bs[i]^=bs[base[j]];
            }
        }
        if(!flag)puts("0");
        else
        {
            printf("%d",(int)bs[i].count());
            for(int j=0;j<i;j++)
                if(bs[i][j])
                    printf(" %d",j);
            puts("");
        }
    }
}
