#include<stdio.h>
#include<string.h>
#include<bitset>
#include<algorithm>
using namespace std;
const int maxm=2050,maxl=650,maxd=2050;
int m;
char num0[maxl];
int num[maxl];
bitset<maxd> mat[maxm],numb,fr[maxm],ans;
int ns=0;
void trans()
{
	int i,j,l=strlen(num0),rem;
	for(i=0;i<l;i++)
		num[i]=num0[l-i-1]-'0';
	for(i=0;i<maxd;i++)
	{
		for(j=l-1,rem=0;j>=0;j--)
			num[j]+=rem*10,rem=num[j]&1,num[j]>>=1;
		while(l>1&&!num[l-1]) l--;
		numb[i]=rem;
	}
}
bool judg(int u)
{
	int i,j,k;
	for(i=j=0;i<ns&&j<maxd;j++)
	{
		if(mat[i][j])
		{
			if(numb[j]) numb^=mat[i],ans^=fr[i];
			i++;
		}
		else if(numb[j]) break;
	}
	if(numb.none()) return 1;
	for(k=ns;k>i;k--)
		mat[k]=mat[k-1],fr[k]=fr[k-1];
	mat[i]=numb;
	fr[i]=ans;
	fr[i].flip(u);
	ns++;
	//for(i=0;i<ns;i++) { for(j=0;j<4;j++) printf("%d ",(int)mat[i][j]); printf("\n"); } printf("\n");
	return 0;
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
	freopen("data.out","w",stdout);
#endif
	int i,j;
	scanf("%d",&m);
	for(i=0;i<m;i++)
	{
		ans.reset();
		scanf("%s",num0);
		trans();
		if(!judg(i)) printf("0\n");
		else
		{
			printf("%d",ans.count());
			for(j=0;j<maxd;j++)
				if(ans[j])
					printf(" %d",j);
			printf("\n");
		}
	}
	return 0;
}
