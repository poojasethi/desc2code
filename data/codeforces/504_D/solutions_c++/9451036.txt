#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <bitset>
#include <algorithm>
#include <iostream>
using namespace std;
#define x first
#define y second
#define sqr(_) ((_)*(_))
#define	 REP(i,j,k) for(register int i=(j),_end_=(k);i<=_end_;++i)
#define REPD(i,j,k) for(register int i=(j),_end_=(k);i>=_end_;--i)

const int MAXN = 2010;
int pos[MAXN],S[MAXN];
bitset<MAXN> A[MAXN],ans[MAXN];
char st[MAXN];
void init(int u)
{
	scanf("%s",st);
	int l=strlen(st);
	REP(i,0,l-1)	S[i]=st[l-i-1]-'0';
	REP(i,0,MAXN-1)
	{
		int res(0);
		REPD(j,l,0)
			S[j]=S[j]+res*10,res=S[j]&1,S[j]>>=1;
		while(!S[l] && l) --l;
		A[u][i]=res;
	}
	ans[u][u]=1;
}
int main()
{
	int n;
	scanf("%d\n",&n);
	REP(i,1,n)
	{
		init(i);
		int can(1);
		REPD(j,MAXN-1,0)
			if(A[i][j])
			{
				if(!pos[j]) {can=0,pos[j]=i;break;}
				A[i]^=A[pos[j]];
				ans[i]^=ans[pos[j]];
			}
		if(can)
		{
			printf("%d",ans[i].count()-1);
			REP(j,1,i-1)
				if(ans[i][j])
					printf(" %d",j-1);
			printf("\n");
		}
		else printf("0\n");
	}
	return 0;
}
