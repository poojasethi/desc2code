#include <bits/stdc++.h>

#define	st first
#define	nd second
#define	mp make_pair
#define	pb push_back
#define	lli long long int
#define	all( gg )	gg.begin(),gg.end()
#define	FP( ii,aa,bb ) for( lli ii=aa;ii<=bb;ii++ )
#define	FM( ii,aa,bb ) for( lli ii=aa;ii>=bb;ii-- )
#define	debug(ccc)	cout << #ccc << " = " << ccc << endl;

#define	mod	1000000007LL

using namespace std;

int	n,first[2000];
bitset<2000>	arr[2002],get[2002];

int	a[2015];

void	f(bitset<2000> &res){
	string	s;
	cin >> s;
	reverse( all(s) );
	int	last = s.size();
	FP( i,1,last )	a[i] = s[i-1]-'0';
	int	rem,t;
	for( int i,bit=0;last;bit++ ){
		rem = 0;
		for( i=last;i>0;i-- ){
			a[i] += 10*rem;
			rem = (a[i]&1);
			a[i] >>= 1;
		}
		res[bit] = rem;
		while( last and !a[last] )	last--;
	}
}

int main(){
	cin >> n;
	memset( st,-1,sizeof st );
	FP( i,1,n ){
		f( arr[i] );
		int	ok = 1;
		get[i][i] = 1;
		FP( j,0,2000 ){
			if( !arr[i][j] )	continue;
			if( first[j]==-1 ){
				first[j] = i;
				ok = 0;
				break;
			}
			arr[i] ^= arr[first[j]];
			get[i] ^= get[first[j]];
		}
		if( !ok )	cout << 0 << endl;
		else{
			cout << get[i].count()-1 << " ";
			FP( j,1,i-1 )
				if( get[i][j] )
					cout << j-1 << " ";
			cout << endl;
		}
	}
}
