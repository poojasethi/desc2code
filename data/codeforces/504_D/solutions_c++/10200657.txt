#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<bitset>
#define MAXN 2005
using namespace std ;
typedef long long LL ;

int N, M, len ;
int vis[MAXN], p[MAXN] ;
bitset<MAXN> have[MAXN], by[MAXN] ;
char ss[MAXN] ;

int main()
{
	int i, j, tmp, get ;
	scanf("%d", &N) ;
	for(i = 1; i <= N; i ++)
	{
		scanf("%s", ss), len = strlen(ss) ;
		reverse(ss, ss+len) ;
		for(j = 0; j < len; j ++)
			p[j+1] = ss[j]-'0' ;
		get = 0 ;
		while(len) 
		{
			for(j = len, tmp = 0; j >= 1; j --)
				p[j] += tmp*10, tmp = p[j]&1, p[j] >>= 1 ;
			while(len && !p[len]) len -- ;
			have[i][get++] = tmp ;
		}
		by[i][i] = 1 ;
	}
	for(i = 1; i <= N; i ++)
	{
		for(j = 0; j < MAXN; j ++)
			if(have[i][j])
			{
				if(!vis[j]) {vis[j] = i; break ;}
				have[i] ^= have[vis[j]] ;
				by[i] ^= by[vis[j]] ;
			}
		if(j < MAXN) puts("0") ;
		else
		{
			printf("%d ", by[i].count()-1) ;
			for(j = 1; j <= N; j ++)
				if(by[i][j] && i != j)
					printf("%d ", j-1) ;
			puts("") ;
		}
	}
	//system("pause") ;
	return 0 ;
}
