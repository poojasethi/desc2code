using namespace std;
char ch[610];
bitset<2100> now[2010],s[2010];
int n,pos[2010],num[2010];
void init(int id)
{
	int i,j,k,l=strlen(ch);
	rep(i,0,l-1)num[l-i]=ch[i]-'0';
	rep(i,1,2010)
	{
		int res=0,lastres;
		reps(j,l,1)
		{
			lastres=res;res=res*10+num[j];
			num[j]=(lastres*10+num[j])/2;
			res%=2;
		}
		now[id][i]=res;
		while(l>=1&&num[l]==0)l--;
	}
	s[id][id]=1;
}
int main()
{
	int i,j,k;
	//freopen("in.txt","r",stdin);freopen("out.txt","w",stdout);
	scanf("%d",&n);
	mem(pos,-1);
	rep(i,1,n)
	{
		scanf("%s",ch);
		init(i);
		bool flag=true;
		rep(j,1,2010)
		if (now[i][j])
		{
			if (pos[j]==-1)
			{
				pos[j]=i;flag=false;break;
			}
			now[i]^=now[pos[j]];
			s[i]^=s[pos[j]];
		}
		if (!flag)printf("0\n");
		else
		{
			s[i][i]=0;
			printf("%d",s[i].count());
			rep(j,1,2010)if (s[i][j])printf(" %d",j-1);
			printf("\n");
		}
	}
	return 0;
}
