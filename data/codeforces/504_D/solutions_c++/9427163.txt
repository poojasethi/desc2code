#include<bits/stdc++.h>
using namespace std;
const int maxn=2000+10,maxl=600+10;
bitset<maxn> mp[maxn],fr[maxn],ans,numb;
int cnt;
char str[maxl];
int num[maxn];
int n;
void translate(){
	int l=strlen(str);
	for(int i=0;i<l;i++)
		num[i]=str[l-i-1]-'0';
	for(int i=0;i<maxn;i++){
		register int j,rem;
		for(j=l-1,rem=0;j>=0;j--)
			num[j]+=rem*10,rem=num[j]&1,num[j]>>=1;
		while(l>1 && !num[l-1])
			--l;
		numb[i]=rem;
	}
	//cout<<numb<<endl;
}
bool check(int u){
	register int i,j;
	for(i=j=0;i<cnt && j<maxn;j++){
		if(mp[i][j]){
			if(numb[j])
				numb^=mp[i],ans^=fr[i];
			++i;
		}
		else if(numb[j])
			break;
	}
	if(numb.none())
		return true;
	for(int k=cnt;k>i;k--)
		mp[k]=mp[k-1],fr[k]=fr[k-1];
	mp[i]=numb;
	fr[i]=ans;
	fr[i].flip(u);
	++cnt;
	return false;
}
int main(){
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
#endif
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		ans.reset();
		scanf("%s",str);
		translate();
		if(!check(i))
			puts("0");
		else{
			printf("%d",ans.count());
			for(int j=0;j<maxn;j++)
				if(ans[j])
					printf(" %d",j);
			puts("");
		}
	}
	return 0;
}
