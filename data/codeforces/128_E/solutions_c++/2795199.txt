#include<stdio.h>
#include<string.h>
#include<math.h>
#include<stdlib.h>
#include<time.h>
#include<iostream>
#include<algorithm>
#include<vector>
#include<queue>
#include<set>
#include<map>
using namespace std;
typedef long long lld;
#define X first
#define Y second
#define mp make_pair
#define pb push_back
#define eps 1e-8
#define inf 0xfffffff
#define pi acos(-1.0)
double x[1010];
double y[1010];
double r[1010];
pair<double,int>pp[100010];
int qq;
int main()
{
    int n;
    lld k;
    cin >> n >> k;
    for(int i=0;i<n;i++)
        scanf("%lf %lf %lf",&x[i],&y[i],&r[i]);
    lld t=1;
    for(int i=0;i<n;i++)
    {
        qq=0;
        for(int j=0;j<n;j++)
        {
            if(i == j)
                continue;
            double angle=atan2(y[j]-y[i],x[j]-x[i]);
            double d=sqrt((x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]));
            double d1=pi*0.5-asin((r[i]-r[j])/d);
            double d2=pi*0.5-asin((r[i]+r[j])/d);
            double x1=angle-d1;
            double x2=angle-d2;
            double y1=angle+d1;
            double y2=angle+d2;
            pp[qq++]=mp(x1+eps,1);
            pp[qq++]=mp(x2-eps,-1);
            pp[qq++]=mp(y2+eps,1);
            pp[qq++]=mp(y1-eps,-1);
        }
        sort(pp,pp+qq);
        for(int j=0;j<qq;j++)
            pp[j+qq]=pp[j];
        lld tmp=0;
        for(int j=0;j<qq+qq;j++)
        {
            tmp+=pp[j].Y;
            t=max(t,tmp+1);
        }
    }
    lld ans=n;
    for(lld i=0;i<k;i++)
        ans+=t+i;
    cout << ans << endl;
    return 0;
}
