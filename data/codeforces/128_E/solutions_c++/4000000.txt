const double PI=acos(-1);

ll n,k;
double x[1010],y[1010],r[1010];

#define sq(x) ((x)*(x))

double dist(int a,int b){
	return sqrt(sq(x[b]-x[a])+sq(y[b]-y[a]));
}

double angle(int a,int b){
	return atan2(y[b]-y[a],x[b]-x[a]);
}

int main(){
	cin.tie(0);
	ios_base::sync_with_stdio(0);
	
	cin>>n>>k;
	rep(i,n)cin>>x[i]>>y[i]>>r[i];
	ll ans=0;
	rep(i,n){
		vector<pair<double,int> > v;
		rep(j,n)if(j!=i){
			double d = dist(i,j);
			double t = angle(i,j);
			double a = asin((r[j]-r[i])/d);
			double b = asin((r[j]+r[i])/d);
			v.pb(mp(t-a,1));
			v.pb(mp(t+b,-1));
			v.pb(mp(t+PI-b,1));
			v.pb(mp(t+PI+a,-1));
		}
		int vs=sz(v);
		rep(i,vs)v.pb(mp(v[i].fir+2*PI,v[i].sec));
		sort(v.begin(),v.end());
		ll c=0;
		rep(i,sz(v)){
			ans=max(ans,c);
			c+=v[i].sec;
		}
	}
	cout<<n+k*(k+1)/2+ans*k;
}
