#include <cstdio>
int n,m,i,j,o;
double ans,a[301],p[301][3001],f[301][3001],c[3001];
int main()
{
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;++i) for (j=1;j<=m;++j) scanf("%lf",&p[j][i]),p[j][i]/=1000;
	for (j=1;j<=m;++j)
	{
		f[j][0]=1;
		for (i=1;i<=n;++i) f[j][i]=f[j][i-1]*(1-p[j][i]);
		a[j]=1-f[j][n];
	}
	for (j=1;j<=n;++j)
	{
		o=1;for (i=2;i<=m;++i) if (a[i]>a[o]) o=i;
		ans+=a[o];
		for (i=1;i<=n;++i) c[i]=c[i-1]*(1-p[o][i])+f[o][i-1]*p[o][i];
		for (i=0;i<=n;++i) f[o][i]=c[i];
		a[o]-=f[o][n];
	}
	printf("%.10lf\n",ans);
}
