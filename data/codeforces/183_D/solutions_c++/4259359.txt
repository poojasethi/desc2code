#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
#define maxn 3010
#define maxm 310
using namespace std;
int n,m;
double p[maxn][maxm],F[maxm][maxn],c[maxn],a[maxm];
void read()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;++i)
		for(int j=1,v;j<=m;++j)
		{
			scanf("%d",&v);
			p[i][j]=v/1000.0;
		}
}
double Dp()
{
	double ans=0;
	for(int j=1;j<=m;++j)
	{
		F[j][0]=1;
		for(int i=1;i<=n;++i)
			F[j][i]=F[j][i-1]*(1-p[i][j]);
		a[j]=1-F[j][n];
	}
	for(int j=1,k;j<=n;++j)
	{
		ans+=a[(k=max_element(a+1,a+m+1)-a)];
		for(int i=1;i<=n;++i)
			c[i]=c[i-1]*(1-p[i][k])+F[k][i-1]*p[i][k];
		a[k]-=c[n];
		for(int i=0;i<=n;++i)
			F[k][i]=c[i];
	}
	return ans;
}
int main()
{
	read();
	printf("%.12lf\n",Dp());
	return 0;
}
