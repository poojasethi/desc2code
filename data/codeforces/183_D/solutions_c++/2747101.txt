#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int maxn = 3010,maxm = 310;

int n,m;
double A[maxn][maxm],B[maxn][maxm];
double C[maxn],D[maxn];
double ans;

int main()
{
    int i,j,k;
    scanf("%d %d\n",&n,&m);
    for (i = 1; i <= m; ++i) B[1][i] = 1;
    for (i = 1; i <= n; ++i)
        for (j = 1; j <= m; ++j)
            scanf("%d",&k),A[i][j] = k/1000.0,B[i+1][j] = B[i][j]*(1-A[i][j]);
    for (i = 1; i <= m; ++i) C[i] = 1-B[n+1][i];
    for (i = 1; i <= n; ++i)
    {
        for (j = k = 1; j <= m; ++j) 
            if (C[j] > C[k]) k = j;
        ans += C[k];
        for (j = 1; j <= n; ++j) D[j+1] = D[j]*(1-A[j][k])+B[j][k]*A[j][k];
        for (j = 1; j <= n+1; ++j) B[j][k] = D[j];
        C[k] -= D[n+1];
    }
    printf("%.12lf",ans);
    return 0;
}
