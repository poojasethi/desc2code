#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

const int maxn = 3005, maxm = 305;
double f[maxm][maxn], p[maxm][maxn];
double val[maxm], tmp[maxn];
double ans;
int n, m;

int main() {
    int i, j, k;
    scanf("%d%d", &n, &m);
    for (j = 1; j <= n; ++j)
        for (i = 1; i <= m; ++i) {
            scanf("%lf", p[i] + j);
            p[i][j] *= 0.001;
        }
    for (i = 1; i <= m; ++i) {
        for (j = 1, f[i][0] = 1; j <= n; ++j)
            f[i][j] = f[i][j - 1] * (1 - p[i][j]);
        val[i] = 1 - f[i][n];
    }
    for (j = 1; j <= n; ++j) {
        for (i = 2, k = 1; i <= m; ++i)
            if (val[i] > val[k]) k = i;
        for (i = 1; i <= n; ++i)
            tmp[i] = tmp[i - 1] * (1 - p[k][i]) + f[k][i - 1] * p[k][i];
        for (i = 0; i <= n; ++i) f[k][i] = tmp[i];
        ans += val[k]; val[k] -= tmp[n];
    }
    printf("%.12lf\n", ans);
    return 0;
}
