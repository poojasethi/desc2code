#include<cstdio>
#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
using namespace std;

int n,m;
double ans;
double g[3010][310],f[310][3010],c[3010],d[310];

int main()
{int i,j,k,x;
 
 scanf("%d%d",&n,&m);
 for(i=1;i<=n;i++)
    for(j=1;j<=m;j++)
    {   scanf("%d",&x);
        g[i][j]=x*0.001;
    }   
 for(i=1;i<=m;i++)
 {  f[i][0]=1;
    for(j=1;j<=n;j++)
        f[i][j]=f[i][j-1]*(1-g[j][i]);
    d[i]=f[i][n];
 }
 for(i=1;i<=n;i++)
 {  k=0;
    for(j=1;j<=m;j++)
        if(k==0 || d[j]<d[k])
            k=j;
    ans+=1-d[k];
    c[0]=0;
    for(j=1;j<=n;j++)
        c[j]=c[j-1]*(1-g[j][k])+f[k][j-1]*g[j][k];
    for(j=0;j<=n;j++)
        f[k][j]=c[j];
    d[k]+=f[k][n];  
 }
 printf("%.12lf\n",ans);
 return 0;
}
