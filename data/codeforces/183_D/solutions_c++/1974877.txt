# include <cstring>
# include <cstdio>
# include <cstdlib>
# include <iostream>
using namespace std;
# define N 3050
# define M 350
double dp[N][M],p[N][M],E[M],tmp[N];
int main (void)
{
    int n,m;
    scanf("%d%d",&n,&m);
    for (int i=1;i<=n;i++)
        for (int j=1;j<=m;j++)
        {
            scanf("%lf",&p[i][j]);
            p[i][j]/=1000.0;
        }
    for (int i=1;i<=m;i++)
    {
        dp[0][i]=1.0;
        for (int j=1;j<=n;j++)
            dp[j][i]=dp[j-1][i]*(1-p[j][i]);
        E[i]=1.0-dp[n][i];
    }
    double ans=0;
    for (int i=1;i<=n;i++)
    {
        int maxx=1;
        for (int j=2;j<=m;j++)
            if (E[j]>E[maxx])
                maxx=j;
        ans+=E[maxx];
        for (int j=1;j<=n;j++)
            tmp[j]=tmp[j-1]*(1-p[j][maxx])+dp[j-1][maxx]*p[j][maxx];
        for (int j=0;j<=n;j++)
            dp[j][maxx]=tmp[j];
        E[maxx]-=tmp[n];
    }
    printf("%.12f\n",ans);
    return 0;
}
