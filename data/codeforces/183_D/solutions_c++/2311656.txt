#include<cstdio>
#include<algorithm>
#define fo(i,a,b) for(i=a;i<=b;++i)
#define fd(i,a,b) for(i=a;i>=b;--i)
using namespace std;
const int mn=5010,mm=310;
double p[mn][mm],x[mn][mm],ans,la,now;
int n,m,i,j,k;
int main()
{
    scanf("%d%d",&n,&m);
    fo(i,1,n)
        fo(j,1,m)
        {
            scanf("%d",&k);
            p[i][j]=(double)k*1e-3;
        }
    ans=0.0;
    fo(j,1,m)
    {
        x[0][j]=1.0;
        fo(i,1,n)x[i][j]=x[i-1][j]*(1.0-p[i][j]);
        fo(i,0,n)x[i][j]=1.0-x[i][j];
    }
    fo(i,1,n)
    {
        k=1;
        fo(j,2,m)if(x[n][j]>x[n][k])k=j;
        ans+=x[n][k];
        la=0.0;
        fo(j,1,n)
        {
            now=x[j-1][k]*(1.0-p[j][k])+la*p[j][k];
            la=x[j][k],x[j][k]=now;
        }
    }
    printf("%.10lf\n",ans);
    return 0;
}
