#include <stdio.h>
#include <string.h>
int N , M ;
const int MN = 3010 ;
const int MM = 310 ;
double p[MM][MN] ;
double f[MM][MN] ;
double g[MM] ;
double opt[MN] ;

void solve(){
    for(int i=1;i<=M;i++){
        f[i][0] = 1 ;
        for(int j=1;j<=N;j++)   f[i][j] = f[i][j-1] * ( 1.0 - p[i][j] ) ;
        for(int j=0;j<=N;j++)   f[i][j] = 1 - f[i][j] ;
        g[i] = f[i][N] ;
    }
    double ans = 0 ;
    for(int i=1;i<=N;i++){
        int t =  1 ;
        for( int j=2;j<=M;j++) if( g[t] < g[j] )    t = j ;
        ans += g[t] ;
        opt[0] = 0 ;
        for( int j=1;j<=N;j++ ){
            opt[j] = opt[j-1] * (1 - p[t][j] ) + f[t][j-1] * p[t][j] ;
        }
        for( int j=0;j<=N;j++ ) f[t][j] = opt[j] ;
        g[t] = f[t][N] ;
    }
    printf("%.10lf\n",ans) ;
}

int main(){
    int x ;
    scanf("%d %d",&N,&M);
    for(int i=1;i<=N;i++)   for(int j=1;j<=M;j++){
        scanf("%d",&x);
        p[j][i] = (double)x / double(1000);
    }
    solve();
    return 0 ;
}
