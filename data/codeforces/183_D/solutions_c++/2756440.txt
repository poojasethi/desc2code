#include<cstdio>
#include<cstdlib>
const int N = 3010, M = 310;
double f[M][N], p[M][N], e[M], temp[N];
int main()
{
    int n, m, i, j, x;
    double ans = 0.0;
    scanf("%d %d", &n, &m);
    for (i = 1; i <= n; i++)
    for (j = 1; j <= m; j++)
    {
        scanf("%d", &x);
        p[j][i] = double(x) / 1000.0;
    }
    for (i = 1; i <= m; i++)
    {
        f[i][0] = 1.0;
        for (j = 1; j <= n; j++)
        f[i][j] = f[i][j - 1] * (1 - p[i][j]);
        e[i] = 1 - f[i][n];
    }
    for (i = 1; i <= n; i++)
    {
        x = 1;
        for (j = 2; j <= m; j++)
        if (e[j] > e[x])
        x = j;
        ans += e[x];
        temp[0] = 0;
        for (j = 1; j <= n; j++)
        temp[j] = temp[j - 1] * (1 - p[x][j]) + f[x][j - 1] * p[x][j];
        e[x] -= temp[n];
        for (j = 0; j <= n; j++)
        f[x][j] = temp[j];
    }
    printf("%.12lf", ans);
    return(0);
}
