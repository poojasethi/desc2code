#include<cstdio>
#include<cstring>
using namespace std;
double f[305][3005];
double p[305][3005];
double cur[3005];
double pre[305];
int n,m;
double dp()
{	int i,j,pos;
	double ans=0;
	for (i=1;i<=m;++i)
	{	f[i][0]=1;
		for (j=1;j<=n;++j)
			f[i][j]=f[i][j-1]*(1-p[i][j]);
		pre[i]=1-f[i][n];
	}
	for (i=1;i<=n;++i)
	{	pos=1;
		for (j=2;j<=m;++j)
			if (pre[pos]<pre[j])
				pos=j;
		for (j=1;j<=n;++j)
			cur[j]=cur[j-1]*(1-p[pos][j])+f[pos][j-1]*p[pos][j];
		for (j=0;j<=n;++j)
			f[pos][j]=cur[j];
		ans+=pre[pos];
		pre[pos]-=f[pos][n];
	}
	return ans;
}
int main()
{	int i,j;
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;++i)
		for (j=1;j<=m;++j)
		{	scanf("%lf",&p[j][i]);
			p[j][i]/=1000;
		}
	printf("%.10lf\n",dp());
	return 0;
}
