#include <cstdio>

using namespace std;

double p[300][3000], f[300][3001], g[3001];

int main() {
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	
	int n, m;
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; i ++)
		for (int j = 0; j < m; j ++) {
			int x;
			scanf("%d", &x);
			p[j][i] = x / 1000.0;
		}
	
	for (int i = 0; i < m; i ++) {
		f[i][0] = 1;
		for (int j = 0; j < n; j ++)
			f[i][j + 1] = f[i][j] * (1 - p[i][j]);
		for (int j = 0; j <= n; j ++)
			f[i][j] = 1 - f[i][j];
	}
	
	double ans = 0;
	for (int i = 0; i < n; i ++) {
		int t = 0;
		for (int j = 1; j < m; j ++)
			if (f[j][n] > f[t][n]) t = j;
		ans += f[t][n];
		g[0] = 0;
		for (int j = 1; j <= n; j ++)
			g[j] = g[j - 1] * (1 - p[t][j - 1]) + f[t][j - 1] * p[t][j - 1];
		for (int j = 0; j <= n; j ++) f[t][j] = g[j];
	}
	
	printf("%.10lf\n", ans);
	
	return 0;
}
