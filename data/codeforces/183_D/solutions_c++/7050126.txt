#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 3310
#define M 310
#define db double
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,t;
int id[N],ge[N];
db a[N][M],f[N][N],b[N],an;
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n)For(j,1,m) scanf("%d",&k),a[i][j]=1.*k/1000;
	For(j,1,m) {
		id[j]=++t;
		f[t][0]=1; For(i,1,n) f[t][i]=f[t][i-1]*(1-a[i][j]);
		b[j]=1-f[t][n];
	}
	For(k,1,n) {
		int re=0;
		db ma=0;
		For(j,1,m) if (b[j]>ma) ma=b[j],re=j;
		an+=ma;
		int last=id[re],now=++t;
		id[re]=now;
		For(i,1,n) f[now][i]=f[last][i-1]*a[i][re]+f[now][i-1]*(1-a[i][re]);
		b[re]-=f[now][n];
	}
	printf("%.11lf\n",an);
	//for(;;);
	return 0;
}
