#include <bits/stdc++.h>
using namespace std;

const int N = 310;
double a[N * 10][N];
double dp[N][N * 10];
int n,m;

void update(int id)
{
    double f[N * 10];
    memcpy(f,dp[id],sizeof(dp[id]));
    dp[id][0] = 0;
    for(int i = 1; i <= n; i ++) {
        dp[id][i] = dp[id][i - 1] * (1 - a[i][id]) + f[i - 1] * a[i][id];
    }
}

int main()
{
    scanf("%d%d",&n,&m);
    for(int i = 1; i <= n; i ++) {
        for(int j = 1; j <= m; j ++) {
            scanf("%lf",&a[i][j]);
            a[i][j] /= 1000;
        }
    }
    for(int i = 1; i <= m; i ++) {
        for(int j = 0; j < N * 10; j ++)
            dp[i][j] = 1;
        update(i);
    }
    double ans = 0;
    int times = n;
    while(times --) {
        int id = 1;
        for(int i = 1; i <= m; i ++)
            if(dp[id][n] < dp[i][n]) id = i;
        ans += dp[id][n];
        update(id);
    }
    printf("%.20lf\n",ans);
    return 0;
}
