#include<iostream>
#include<algorithm>
using namespace std;
int main()
{int n;
cin>>n;
int a[n];int f[5001]={0};
for(int i=0;i<n;i++)
{cin>>a[i];
f[a[i]]++;
}
sort(a,a+n);
int b[n];
int top=-1;
for(int i=0;i<=5000;i++)if(f[i])b[++top]=i;
for(int i=5000;i>=0;i--)if(f[i]>=2)
{if(i!=a[n-1])b[++top]=i;
}
cout<<top+1<<endl;
for(int i=top;i>=0;i--)cout<<b[i]<<" ";
cout<<endl;
return 0;
}
