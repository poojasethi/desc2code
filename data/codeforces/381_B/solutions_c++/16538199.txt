#include<iostream>
using namespace std;
long b[100005];
int main(){
	long long m;long max=0;
	cin>>m;long x;long l=0;
	for(long long i=0;i<m;i++){
		cin>>x;b[x]=b[x]+1;if(b[x]<=2){l=l+1;
		}
		if(x>max)max=x;
	}
	if(b[max]>=2)l=l-1;
	cout<<l<<endl;
	for(long i=1;i<=max;i++){
		if(b[i]>0){
		cout<<i<<" ";b[i]=b[i]-1;}
	}
	for(long i=max-1;i>=1;i--){
	if(b[i]>0){
		cout<<i<<" ";b[i]=b[i]-1;}	
	}
cout<<endl;	
}
