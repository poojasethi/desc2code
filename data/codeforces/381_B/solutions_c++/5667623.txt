#include <cstdio>

#define M 5000

int a[M + 1], b[M * 2 + 4], n, i, k;

int main()
{
	scanf("%d", &n);

	for (i = 0; i < n; i ++)
	{
		scanf("%d", &k);
		--k;
		++a[k];
	}

	n = 0;

	for (i = 0; i <= M; i ++)
		if (a[i])
		{
			--a[i];
			b[n ++] = i;
			k = i;
		}

	for (i = k - 1; i >= 0; i --)
		if (a[i])
		{
			-- a[i];
			b[n ++] = i;
		}

	printf("%d\n", n);
	for (i = 0; i < n; i ++)
		printf("%d ", b[i] + 1);
}
