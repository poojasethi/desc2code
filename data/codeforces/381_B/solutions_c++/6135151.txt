#include <iostream>
#define N 5010
using namespace std;int n,x,A,B,cnt[N],a[N],b[N];
int main()
{
    cin>>n;
    for(;n;n--)
    {
        cin>>x;
        cnt[x]++;
    }
    for(n=5000;!cnt[n];n--);
    a[++A]=n--;
    for(;n;n--)
    {
        if(cnt[n]){a[++A]=n;cnt[n]--;}
        if(cnt[n])b[++B]=n;
    }
    cout<<A+B<<'\n';
    for(;B;B--)cout<<b[B]<<' ';
    for(B=1;B<=A;B++)cout<<a[B]<<' ';
    return 0;
}
