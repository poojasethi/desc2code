#include <iostream>
using namespace std;
int m, a[5002], x, i, c1, c2, s, t; 
int main()
{
	cin>>m;
	while (m--) {
		cin>>x; a[x]++; s=max(s,x);
	}
	for (i=1; i<s; i++) if (a[i]>1) t+=2; else if (a[i]==1) t++;
	cout<<t+1<<endl;
	for (i=1; i<=s; i++) if (a[i]) {
		cout<<i<<" "; a[i]--;
	}
	for (i=s-1; i>0; i--) if (a[i]) cout<<i<<" ";
	cout<<endl;
	return 0;
}
