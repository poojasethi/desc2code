#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,a[5005]={0};
    cin>>n;
    int x,mx=0;
    for(int i=0;i<n;++i){
    cin>>x;
    ++a[x];
    mx=max(mx,x);
    if(a[x]>2)--n,--i;
    }
    if(a[mx]>1)--n;
    cout<<n<<endl;
    for(int i=0;i<mx;++i)
    {
        if(a[i]){--a[i];cout<<i<<" ";}
    }
    for(int i=mx;i>=0;--i)
    {
        if(a[i]){--a[i];cout<<i<<" ";}
    }

}
