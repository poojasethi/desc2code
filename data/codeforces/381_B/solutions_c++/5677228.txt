#include <iostream>
#include <set>
#include <vector>
using namespace std;

int n, x, f[5005] = {}, l = -1;
vector<int> res;

int main(){

    cin >> n;
    for(int i = 0; i<n; i++) cin >> x, f[x]++;
    for(int i = 0; i<5005; i++) if(f[i]) f[i]--, res.push_back(i), l = i;
    for(int i = l-1; i>=0; i--) if(f[i]) res.push_back(i);
    cout << res.size() << '\n';
    for(int i = 0; i<res.size(); i++) cout << res[i] << ' ';

}
