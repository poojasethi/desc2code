#include<iostream>
using namespace std;
int main()
{
long int m,i,b[5001]={0},a,count=0,max;
cin>>m;
for(i=0; i<m; i++)
{
cin>>a;
b[a]++;
if(b[a]<3)
count++;
}
for(i=5000; b[i]==0; i--);
max=i;
if(b[max]>1)
count--;
cout<<count<<"\n";
for(i=1; i<=max; i++)
if(b[i]>0)
{
cout<<i<<" ";
b[i]--;
}
for(i=max-1; i>0; i--)
if(b[i]>0)
cout<<i<<" ";
return 0;
}
