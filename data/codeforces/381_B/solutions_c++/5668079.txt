#include <iostream>
#include <vector>

using namespace std;

long a[10000];
long i,n,x,last;
vector <long> ans;

int main()
{
	cin >> n;
	for (i=1;i<=n;i++) {
		cin >> x;
		a[x]++;
	}
	for (i=1;i<=5000;i++)
		if (a[i]>0) {
			last=i;
			ans.push_back(i);
			a[i]--;
		}
	for (i=last-1;i>=1;i--){
		if (a[i]>0){
			ans.push_back(i);
		}
	}	
	cout << ans.size() << "\n";
	for (i=0;i<ans.size();i++) cout << ans[i] << " ";
	return 0;
}
