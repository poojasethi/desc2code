#include<iostream>
using namespace std;
int A[5005]={0};
int O[100005],o=0;;
int main()
{
	int n;
	cin>>n;
	int t,max=0;
	for(int i=0;i<n;i++)
	{
		cin>>t;
		A[t]++;
		if(t>max)
			max=t;
	}
	int f=0;
	for(int i=0;i<=max;i++)
		if(A[i]>0)
		{
			O[o++]=i;
			A[i]--;
		}
	for(int i=max-1;i>=0;i--)
		if(A[i]>0)
			O[o++]=i;
	cout<<o<<"\n";
	cout<<O[0];
	for(int i=1;i<o;i++)
		cout<<' '<<O[i];
	cout<<"\n";
	return 0;
}
