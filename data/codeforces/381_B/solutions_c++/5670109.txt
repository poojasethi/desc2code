

#include <bits/stdc++.h>
using namespace std;
int a[5500];
int main(int argc, char **argv)
{
	int sum=0,n,i,k,max=0;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&k);
		if(k>max)max=k;
		a[k]++;
	}
	for(i=0;i<5010;i++)
		sum+=(a[i]>1)?2:a[i];
		if(a[max]>1)sum--;
		printf("%d\n",sum);
	for(i=0;i<=max;i++)if(a[i])printf("%d ",i);
	for(i=max-1;i>=0;i--)if(a[i]>1)printf("%d ",i);
	return 0;
}
