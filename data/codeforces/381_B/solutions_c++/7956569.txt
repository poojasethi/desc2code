#include <iostream>
#include <vector>
#include <queue>
#include<map>
using namespace std;
int arr[100001];
int ok[100001];
int main()
{
   int m,k,j=0,maxi=0,q=0;
    cin>>m;
    for(int i=0; i<m; i++){cin>>k;if(k>maxi)maxi=k;arr[k]++;}
    for(int i=0; i<5001;i++)if(arr[i]>0){ok[q]=i;q++;arr[i]--;}
    for(int i=5001; i>=0; i--)if(arr[i]>0&&i!=maxi){ok[q]=i;q++;}
    cout<<q<<endl;
    for(int i=0; i<q;i++)cout<<ok[i]<<" ";

    return 0;
}
