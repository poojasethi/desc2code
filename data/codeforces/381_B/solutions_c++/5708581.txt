#include <cstdio>
#include <algorithm>
using namespace std;
int n,adet[5005],enb,a,top;
int main()
{
	scanf("%d",&n);
	for (int i=1,a;i<=n;i++)
	{
		scanf("%d",&a);
		enb=max(a,enb);
		adet[a]++;
	}
	for (int i=1;i<enb;i++)
	{
		top+=min(2,adet[i]);
	}
	printf("%d\n",top+1);
	for (int i=1;i<enb;i++)
		if(adet[i]>0)
		{
			printf("%d ",i);
		}
	printf("%d ",enb);
	for (int i=enb-1;i>0;i--)
		if(adet[i]>1)
		{
			printf("%d ",i);
		}
}
