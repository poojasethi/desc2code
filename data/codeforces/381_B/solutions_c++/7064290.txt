#include<bits/stdc++.h>
using namespace std;
int main()
{
	int m,n,i,a[5010]={0},num=INT_MIN;
	vector<int> b;
	cin>>m;
	for(i=0;i<m;i++)
	{
		cin>>n;
		a[n]++;
		num=max(num,n);
	}
	for(i=1;i<=num;i++)
	{
		if(a[i])
		{
			b.push_back(i);
			a[i]--;
		}
	}
	for(i=num-1;i>=1;i--)
	{
		if(a[i])
		{
			b.push_back(i);
			a[i]--;
		}
	}
	cout<<b.size()<<endl;
	for(i=0;i<b.size();i++)
		cout<<b[i]<<' ';
	cout<<endl;
	return 0;
}
