#include <iostream>
#include <vector>
using namespace std;
int arr[5001];
int main() {
	// your code goes here
	int n, temp;
	vector<int> v;
	cin >> n;
	while(n--)
	{
		cin>>temp;
		arr[temp]++;
	}
	int c;
	for(int i=0; i< 5001;i++)
	{
		if (arr[i])
		{
			v.push_back(i);
			arr[i]--;
			c=i;
		}
	}
	for (int i= c-1; i; i--)
	{
		if(arr[i])
		{
			v.push_back(i);
		}
	}
	cout<<v.size()<<endl;
	for(int i=0; i< v.size();i++)
	cout<<v[i]<<" ";
	return 0;
}
