#include<bits/stdc++.h>
using namespace std;
int m;
int bs[5010];
vector<int> res;

int main(){
	cin >> m;
	memset(bs,0,sizeof bs);
	int t;
	for(int i=0;i<m;i++) {cin >> t;bs[t]++;}
	
	for(int i=0;i<5001;i++){
		if(bs[i]){
			bs[i]--;
			res.push_back(i);
		}
	}
	for(int i=5000;i>=0;i--){
		if(bs[i] && i!=res[res.size()-1])
			res.push_back(i);
	}
	cout << res.size() << endl;
	for(int i=0;i<res.size();i++) cout << res[i] << " ";
	


	return 0;
}
