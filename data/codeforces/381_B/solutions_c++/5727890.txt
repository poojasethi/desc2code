#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int n,i,j=1,max=0,x,a[100100]={},b[100100]={},ln=0,rn=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++){
	  scanf("%d",&x);
	  if(max<x) max=x;
	  a[x]++;
	}
	a[max]=1;
	for(i=1;i<=max-1;i++)
	  if(a[i]>=2){
	    b[j++]=i;
		a[i]--;
		ln++;
	  }
	for(i=max;i>=1;i--)
	  if(a[i]>=1){
		b[j++]=i;
		a[i]--;
		rn++;
	  }
	printf("%d\n",rn+ln);
	for(i=1;i<=(rn+ln);i++)
	  printf("%d ",b[i]);
	printf("\n");
	return 0;
}
