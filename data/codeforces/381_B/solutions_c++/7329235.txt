#include <iostream>
using namespace std;
int main()
{
    int n,tmp;
    cin>>n;
    int a[n],h[100000];
    int m=0;
    for(int i=0;i<n;i++){cin>>tmp;h[tmp]++;if(m<tmp)m=tmp;}
    int ans=0;
    for(int i=1;i<=m;i++)if(h[i]){ans++;a[ans]=i;h[i]--;}
    for(int i=m-1;i>0;i--)if(h[i]){ans++;a[ans]=i;}
    cout<<ans<<endl;
    for(int i=1;i<=ans;i++) cout<<a[i]<<" ";
}
