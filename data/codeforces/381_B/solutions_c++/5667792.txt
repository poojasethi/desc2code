#include <iostream>
#include <math.h>
#include <string>
using namespace std;
int main(){
    int a[5001],i,j,m,n=0,z,maxel=0;
    for (i=1; i<=5000; i++)a[i]=0;
    cin>>m;
    for (i=0; i<m; i++){
        cin>>z;
        a[z]++;
        if (maxel<z)maxel=z;
    }
    for (i=1; i<maxel; i++)n+=min(2,a[i]);
    n++;
    cout<<n<<endl;
    for (i=1; i<=maxel; i++)if (a[i]>0)cout<<i<<" ";
    for (i=maxel-1; i>0; i--)if (a[i]>1)cout<<i<<" ";
}
