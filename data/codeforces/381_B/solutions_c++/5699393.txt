#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
#define pb push_back
int m,x,c[5009]={0};
int main()
{
    vector<int>v;
    cin >> m;
    for (int i=0;i<m;i++)cin >> x,c[x]++;
    for (int i=1;i<5001;i++)if (c[i]>0)v.pb(i),c[i]--,x=i;
    for (int i=5000;i>=1;i--)if (c[i]>0 && i!=x)v.pb(i),c[i]--;
    cout << v.size() << "\n";
    for (int i=0;i<(int)v.size();i++)cout << v[i] << " ";
    return 0;
}
