#include <cstdio>
int ok[5004];
int main()
{
	int n, val, l = 0, max = 0;
	
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i)
	{
		scanf("%d", &val);
		++ok[val];
		if (ok[val] <3)
			++l;
		if (val > max)
			max = val;
	}
	if (ok[max] > 1)
		--l;
	printf("%d\n", l);
	for (int i = 1; i <= max; ++i)
		if (ok[i])
		{
			--ok[i];
			printf("%d ", i);
		}
	for (int i = max-1; i > 0; --i)
		if (ok[i])
			printf("%d ", i);
printf("\n");
return 0;
}
