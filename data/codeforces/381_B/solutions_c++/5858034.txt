#include <stdio.h>

int a[10000]={};

int main()
{
    int max =0;
int n,cnt=0;
scanf("%d",&n);
int x;
for(int i = 0;i<n;i++)
{
scanf("%d",&x);
if(max < x)
max = x;
a[x]++;
}
for(int i = 1;i<=max;i++)
{
if(a[i]>=2 and i!=max)cnt+=2;
else if(i==max)cnt+=1;
else cnt+=a[i];
}
printf("%d\n",cnt);
for(int i = 0;i<=max;i++)
{
if(a[i]>0)
{
    printf("%d ",i);
    a[i]--;
}
}
for(int i = max-1;i>0;i--)
{
if(a[i]>0)
printf("%d ",i);
}
return 0;
}
