#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
int main(){
int a,b[200000],c[10000]={0},e=0,f=0,maxe=0,h,j,i,k,l;
vector < int > d;
cin>>a;
for(i=1;i<=a;i++){
cin>>b[i];
c[b[i]]++;
if(b[i]>maxe)maxe=b[i];
}
for(i=1;i<=5000;i++){
if(c[i]>0){
d.push_back(i);
c[i]--;
f++;
}
}
c[maxe]=0;
for(i=5000;i>=1;i--){
if(c[i]>0){
d.push_back(i);
c[i]--;
f++;
}
}
cout<<f<<endl;
for(i=0;i<f;i++){
cout<<d[i]<<" ";
}
return 0;
}
