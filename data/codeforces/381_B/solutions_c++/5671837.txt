#include<iostream>
#include<algorithm>
#define maxn 100001
using namespace std;
int ar[maxn],N;
void program()
{
	int i,t=0,j=0,er[maxn];
	cin>>N;
	for(i=0;i<N;i++)
	cin>>ar[i];
	sort(ar,ar+N);
	for(i=0;i<N;i++)
		if(ar[i]!=t)
		{
			j++;
			er[j]=ar[i];
			t=ar[i];
			ar[i]=0;
		}		
	for(i=N-1;i>=0;i--)
		if(ar[i]&&ar[i]!=t)
		{
			t=ar[i];
			j++;
			er[j]=ar[i];
		}
	cout<<j<<endl;	
	for(i=1;i<=j;i++)
	cout<<er[i]<<' ';	
}
int main()
{
	program();
	return 0;
}
