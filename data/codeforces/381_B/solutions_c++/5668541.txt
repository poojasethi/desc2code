#include<cstdio>
int tmp[5555]={0};
int ans[11111];
int A=0;
int main () {
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++) {
		int a;
		scanf("%d",&a);
		tmp[a]++;
	}
	for(int i=0;i<=5000;i++) 
		if(tmp[i]>0) {
			tmp[i]--;
			ans[A]=i;
			A++;
		}
	for(int i=5000;i>=0;i--) {
		if(tmp[i]>0&&i<ans[A-1]) {
			tmp[i]--;
			ans[A]=i;
			A++;
		}
	}
	printf("%d\n",A);
	for(int i=0;i<A;i++) printf("%d ",ans[i]);
 	return 0;
}
