#include<cstdio>
#include<cstring>
using namespace std;
int n, i, b, m, f[5001], ans[100000];
int main(){
	while(~scanf("%d", &n)){
		memset(f,0,sizeof(f));
		for(i=0; i<n; i++){
			scanf("%d", &b);
			f[b]++;
		}
		m=0;
		for(i=1; i<=5000; i++){
			if(f[i]){
				ans[m++]=i;
				f[i]--;
			}
		}
		for(i=ans[m-1]-1; i>=1; i--){
			if(f[i]){
				ans[m++]=i;
			}
		}
		printf("%d\n%d", m, ans[0]);
		for(i=1; i<m; i++)	printf(" %d", ans[i]);
		puts("");
	}
	return 0;
}
