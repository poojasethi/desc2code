#include<iostream>
#include<vector>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int ar[n],x=-1;
	int hash[5001]={0};
	for(int i=0;i<n;i++)
		{
		cin>>ar[i];
		x=max(x,ar[i]);
		hash[ar[i]]++;
		}
		vector<int> res;
	for(int i=1;i<=x;i++)
	{
		if(hash[i])
			res.push_back(i),hash[i]--;
	}
	for(int i=x-1;i>0;i--)
	{
		if(hash[i])
		res.push_back(i);
	}
	int sz = res.size();
	cout<<sz<<endl;
	for(int i=0;i<sz;i++)
	cout<<res[i]<<" ";
	cout<<endl;	
	return 0;
}
