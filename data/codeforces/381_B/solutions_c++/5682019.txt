#include <cstdio>
#include <iostream>
using namespace std;
int d[5010],m,f,i,l,k;
int main()
{
    cin>>m;
    for(i=0;i<m;i++)
    {
        cin>>k;
        d[k]++;
        if(d[k]==1)f++;
        if(d[k]==2)f++;
        if(k>l)l=k;
    }
    if(d[l]>1)f--;
    cout<<f<<endl;
    for(i=1;i<=l;i++)
      if(d[i]>0){cout<<i<<" ";d[i]--;}
    for(i=l-1;i>0;i--)
      if(d[i]>0){cout<<i<<" ";d[i]--;}
    cout<<endl;
 return 0;
}
