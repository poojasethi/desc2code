#include<cstring>
#include<cstdio>
int a[5100],s[10001];
int main()
{
    memset(a,0,sizeof(a));
    int n,i,r,cnt=0;
    scanf("%d",&n);
    while(n--){
        scanf("%d",&i);
        ++a[i];
    }
    for(i=1;i<=5000;++i)if(a[i]){--a[i];r=s[cnt++]=i;}
    for(i=r-1;i>0;--i)if(a[i]){--a[i];s[cnt++]=i;}
    printf("%d\n%d",cnt,s[0]);
    for(i=1;i<cnt;++i)printf(" %d",s[i]);
    printf("\n");
    return 0;
}
