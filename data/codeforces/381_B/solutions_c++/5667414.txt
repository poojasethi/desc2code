#include <iostream>
using namespace std;
int m,F[5001],i,x;
int REZ[100001],n,maxi;;
int main()
{
	cin>>m;
	for (i=1;i<=m;i++)
	{
		cin>>x;
		F[x]++;
	}
	n=0;
	for (i=1;i<=5000;i++)
		if (F[i]>0)
		{
			n++;
			REZ[n]=i;
			F[i]--;
			maxi=i;
		}
	for (i=maxi-1;i>=1;i--)
		if (F[i]>0)
		{
			n++;
			REZ[n]=i;
		}
	cout<<n<<"\n";
	for (i=1;i<=n;i++)
		cout<<REZ[i]<<" ";
	return 0;
}
