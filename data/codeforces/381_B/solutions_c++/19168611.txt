#include <bits/stdc++.h>
using namespace std;

int n, m, x, i, ans, a[5005];

int main() 
{
	cin >> n;
	for (i = 0; i < n; ++i) 
	{cin >> x ;
	a[x]++;
	m = max(m, x);
	}
	for (i = 1; i < m; ++i)
	ans += min(a[i], 2);
	
	cout << ans + 1 << endl;
	
	for (i = 1; i < m; i++)
	{
	    if (a[i] > 0)
	    {
	cout << i << ' ';
	a[i]--;
	    }
	} 
	for (i = m; i > 0; i--)
	{
	if (a[i] > 0)
	cout << i << ' ';
	}
	return 0;
}
