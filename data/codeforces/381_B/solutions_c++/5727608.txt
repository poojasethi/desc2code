#include<stdio.h>
#include<algorithm>
int main()
{
int n=0,c[100000],s=0,i,m,ans=0,j=0,p=0,a[100000],b[100000];

for(i=0;i<m;i++)
scanf("%d",&a[i]);
std::sort(a,a+m);
for(i=0;i<m;i++)
{
if(j==a[i]&&p==0&&j!=a[m-1])
{
p=1;
b[n]=a[i];
n++;ans++;
}
else if(j!=a[i])
{p=0;c[s]=a[i];s++;ans++;j=a[i];}
else if(j==a[i]&&p==1)
{}
}

for(i=0;i<s;i++)
printf("%d ",c[i]);
for(i=n-1;i>=0;i--)

}
