#include<cstdio>
#include<algorithm>
#define N 100000
using namespace std;
int v[N],us[N];
int ans[N];

int main()
{
	int m;scanf("%d",&m);
	for(int i=0;i<m;++i)
		scanf("%d",v+i);
	sort(v,v+m);
	int p = 0;
	for(int i=0;i<m;++i)
		if(!p || v[i]!=ans[p-1])
			ans[p++] = v[i],us[i]=1;
	
	for(int i=m-1;i>=0;--i)
	{
		if(!us[i] && v[i]!=ans[p-1])
			ans[p++] = v[i];
	}
	printf("%d\n",p);
	for(int i=0;i<p;++i)
		printf("%d ",ans[i]);
	puts("");
	return 0;
}
