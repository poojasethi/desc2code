#include<iostream>
#include<vector>
using namespace std;
int d[5001];
vector<int>ans;
int main() {
	string s;
	int n,a,c=0;
	cin>>n;
	for(int i=0;i<n;i++) {
		cin>>a;
		d[a]++;
	}
	int i,m=0;
	for(i=1;i<=5000;i++)
		if(d[i]){d[i]--;c++;ans.push_back(i);m=i;}
	for(i=m-1;i>=1;i--)
		if(d[i]){d[i]--;c++;ans.push_back(i);}
	cout<<ans.size()<<endl;
	for(vector<int>::iterator iv=ans.begin();iv!=ans.end();iv++)
		cout<<*iv<<" ";
	return 0;
}
