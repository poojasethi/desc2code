#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int v[100005];
int main()
{
int n;
int i,x;
scanf("%d",&n);

for(i=0;i<n;i++)
{
scanf("%d",&v[i]);
}
sort(v,v+n);
int pr = -1;
vector<int> b;
for(i=0;i<n;i++)
{
if(v[i]!=pr){b.push_back(v[i]);pr = v[i]; v[i] = 0;}

}
for(i=n-1;i>=0;i--)
{
if(v[i]!=0 && v[i]!=pr){b.push_back(v[i]);pr = v[i];v[i] = 0;}

}
printf("%d\n",b.size());
for(i=0;i<b.size();i++)
printf("%d ",b[i]);

return 0;
}
