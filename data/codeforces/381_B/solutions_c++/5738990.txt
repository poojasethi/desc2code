#include<iostream>
#include<cstdio>
using namespace std;

int a[5001]={0};

int main()
{int mi=5002,ma=0,i,m,t,ans=0;
 cin >>m;
 for(i=0;i<m;i++)
 {   cin >>t;
  if(a[t]<2){a[t]++;ans++;}
  if(ma<t)ma=t;
  if(mi>t)mi=t;
 }
 if(a[ma]==2)ans--;
 cout <<ans<<endl;
 for(i=mi;i<ma;i++)
    if(a[i]==2){printf("%d ",i);a[i]--;}
 printf("%d",ma);
 for(i=ma-1;i>=mi;i--)
    if(a[i]==1){printf(" %d",i);a[i]--;}
return 0;
}
