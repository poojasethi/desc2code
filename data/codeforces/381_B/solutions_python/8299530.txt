n=input()
l=map(int,raw_input().split())
l.sort()
l=l+[-1]
ans1=[]
ans2=[]
i=0
while(i<n):
    if l[i]==l[i+1]:
        ans1.append(l[i])       
        ans2.append(l[i+1])
        i+=1
    else:
        ans1.append(l[i]) 
    i+=1        
ans1=ans1+ans2[::-1]
re=[-1]
ind=0
for i in range(len(ans1)):
    if str(ans1[i]) != re[ind]:
        re.append(str(ans1[i]))
        ind+=1
re.remove(-1)
print len(re)
print ' '.join(re)
