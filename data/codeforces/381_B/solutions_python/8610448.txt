n=input()
a=map(int,raw_input().split())
a.sort()
l=[]
prev=a[0]
t=0
for i in xrange(n):
	if prev!=a[i]:	
		l.append((prev,t))
		t=1
		prev=a[i]
	else:
		t+=1
l.append((prev,t))
count=0
s=""
c=""
for i in xrange(len(l)):
	s+=str(l[i][0])+" "
	count+=1
	if l[i][1] > 1 and i!=len(l)-1:
		c=str(l[i][0])+" "+c
		count+=1
