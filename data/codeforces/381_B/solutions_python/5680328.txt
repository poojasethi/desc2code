N = input()
a = map(int, raw_input().split())
b = [0]*5001
for x in a:
	if b[x] < 2:
		b[x] +=1
p = [i for i in xrange(1, len(b)) if b[i] > 0][::-1]
q = [i for i in xrange(1, len(b)) if b[i] > 1][::-1]
if len(q) and p[0] == q[0]:
	ans = p[::-1]+q[1:]
else:
	ans = p[::-1]+q
print len(ans)
print ' '.join(map(str, ans))
