m = int(raw_input())
b = map(int, raw_input().split())
b.sort()

used = [False] * m
prev = 0
result = []
for i, v in enumerate(b):
	if v == prev:
		continue
	result.append(v)
	prev = v
	used[i] = True

for j, v in enumerate(reversed(b)):
	if used[m - j - 1] or v == prev:
		continue
	result.append(v)
	prev = v

print len(result)
print ' '.join(str(v) for v in result)
