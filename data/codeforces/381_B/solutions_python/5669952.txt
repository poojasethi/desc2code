from collections import deque

if __name__ == '__main__':
    n = int(raw_input())
    cards = map(int, raw_input().split())
    sortc = sorted(cards)
    result = deque()
    for el in reversed(sortc):
        if len(result) == 0 or result[-1] != el:
            result.append(el)
        elif result[0] != el:
            result.appendleft(el)
    print len(result)
    print " ".join(map(str, result))
