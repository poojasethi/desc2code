import sys
m=input()
cards=map(int,raw_input().split())
cards=sorted(cards)
show=[]
dup=0
prev=-1
for i in xrange(m-1,-1,-1):
    a=cards[i]
    if a==prev:
        dup+=1
    else:
        dup=0
    if dup==0:
        show.append(a)
    if dup==1 and a!=cards[m-1]:
        show.insert(0,a)
    prev=a
print len(show)
for i in xrange(len(show)):
    if i!=0: sys.stdout.write(' ')
    sys.stdout.write(str(show[i]))
sys.stdout.write('\n')
