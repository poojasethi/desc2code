m = int(raw_input())
b = [int(bi) for bi in raw_input().split(' ')]
d = { i : 0 for i in range(1,5001)}
for bi in b:
  d[bi] = d[bi] + 1

stairs = []
m = max(b)
for i in range(1, m + 1):
  if d[i] > 0:
    stairs.append(i)
    d[i] = d[i] - 1


for i in range (1, m):
  if d[m-i] > 0:
    stairs.append(m-i)
    d[m-i] = d[m-i] - 1

print len(stairs)
print ' '.join([str(s) for s in stairs])
