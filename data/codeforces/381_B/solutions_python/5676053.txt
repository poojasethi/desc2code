from __future__ import print_function

m = int(raw_input())
b = map(int,raw_input().split())
b.sort()
rest = set()
prev = -1
result = []
m = 0
for i in xrange(len(b)):
    if b[i] != prev:
        result.append(b[i])
        prev = b[i]
        m = max(m, b[i])    
    else:
         rest.add(prev)

        
for k in sorted(rest, reverse=True):
    if k == m:
        continue    
    result.append(k)

print(len(result))
print(*result)
