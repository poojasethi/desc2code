C=[0]*(5000+1)

n=int(raw_input())
L=[int(x) for x in raw_input().split()]

for x in L:
	C[x]+=1

T=[[i,C[i]] for i in range(5000+1) if C[i]>0]

ST=[]

for i in range(len(T)):
	ST.append(T[i][0])
	T[i][1]-=1

T[-1][1]=0

for i in range(len(T)-1,-1,-1):
	if T[i][1]>0:
		ST.append(T[i][0])

print len(ST)
print " ".join(str(x) for x in ST)
