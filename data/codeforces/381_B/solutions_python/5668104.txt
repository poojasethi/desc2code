m = int(raw_input())
a = [int(i) for i in raw_input().split()]
a.sort()
b = [0 for i in range(m)]
b[0] = 1
c = [a[0]]
k = 0
flag = False
for i in range(1,m):
	if a[i] > c[k]:
		c.append(a[i])
		k += 1
		b[i] = 1
for i in range(m-1, 0, -1):
	if a[i] < c[k] and b[i] != 1:
		flag = True
		c.append(a[i])
		k += 1
		b[i] = 1
print len(c)
if flag:
	for i in range(len(c)):
		print c[i],
else:
	for i in range(len(c)):
		print c[len(c)-1-i],
