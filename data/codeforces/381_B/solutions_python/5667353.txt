n = int(raw_input())
a = map(int, raw_input().split(' '))
freq = [0]*5005
for x in a:
    freq[x] += 1

out = []

for i in range(1, 5005):
    if(freq[i] > 0):
        out.append(i)
        freq[i] -= 1

for i in range(5004, 0, -1):
    if freq[i] > 0 and (not i == out[-1]):
        out.append(i)

print len(out)
print ' '.join(map(str, out))
