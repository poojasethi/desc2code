import sys

trap={}

n=int(raw_input())
a=list(map(int, raw_input().split()))

for x in a:
    if not x in trap:
        trap[x]=1
    else:
        trap[x]+=1

op=sorted(trap.keys())
af=sorted(trap.keys(),reverse=True)

uit=op

for x in af:
    if x < op[len(op)-1] and trap[x]>1:
        uit.append(x)
            
print len(uit)
print ' '.join([str(x) for x in uit])
