def solve(b):
    b.sort()
    x, y = [], []
    t = True
    for i in range(len(b)):
        a = x if t else y
        if not a or a[-1] != b[i]:
            a.append(b[i])
            t = not t
    y.reverse()
    if x and y and x[-1] == y[0]:
        x = x[:-1]
    return x+y

m = int(raw_input())
b = map(int, raw_input().split())
res = solve(b)
print len(res)
print ' '.join(map(str, res))
