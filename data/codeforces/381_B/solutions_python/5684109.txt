
I=lambda:map(int, raw_input().split())
n = input()
a = I()
x = [0] * (10**5+1)
for y in a:
    x[y] += 1
ans = []
biggest = max(a)
for i in xrange(1, biggest+1):
    if x[i] > 0:
        ans += [i]
        x[i] -= 1
for i in xrange(biggest, 0, -1):
    if x[i] > 0 and i < ans[-1]: 
        ans += [i]
print len(ans)
print ' '.join(map(str,ans))
