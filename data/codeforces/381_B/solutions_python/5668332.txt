#AV Drift King
m=int(raw_input())
mar=map(int,raw_input().split(" "))
freqar=[0 for i in range(5001)]
for i in mar:
	freqar[i]+=1
anslen=0
stair=[]
j=-1
for i in range(1,5001):
	if freqar[i]>0:
		anslen+=1
		stair.append(i);
		freqar[i]-=1
		j=i
for i in range(j-1,0,-1):
	if freqar[i]>0:
		anslen+=1
		stair.append(i)
		freqar[i]-=1
print anslen
print " ".join(map(str,stair))
