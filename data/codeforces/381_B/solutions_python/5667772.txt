import sys

#f = open('input.txt','r')
f = sys.stdin

lines = [line.strip() for line in f.readlines()]

n = int(lines[0])
a = [0] * 5002
for item in map(int, lines[1].split()):
    a[item] += 1

result = []
max_i = 0
for i in xrange(len(a)):
    if a[i] > 0:
        result.append(i)
        a[i] -= 1
        max_i = i

for i in reversed(xrange(max_i)):
    if a[i] > 0:
        result.append(i)

print len(result)
print ' '.join(map(str, result))
