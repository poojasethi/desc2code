#include<stdio.h>
#include<string.h>
#include<vector>
#include<algorithm>
using namespace std;
typedef __int64 LL;
const int N=2001;
inline void read(LL &x){
    #define CH getchar()
    char ch;x=0;for(ch=CH;ch<'0'||ch>'9';ch=CH);
    for(;ch>='0'&&ch<='9';x=x*10+ch-48,ch=CH);
}   
LL d[N][N];
    
int main(){
    int T,i,j,n,m,k,p,l,r;
    int x,y,ans,t,f;
    scanf("%d",&n);
    for(f=0,i=1;i<=n;i++)
        for(j=1;j<=n;j++) read(d[i][j]);
    for(f=0,i=1;i<=n;i++){
        if(d[i][i]) f=1;
        for(j=i+1;j<=n;j++) if(d[i][j]!=d[j][i]||!d[i][j]) f=1;
    }
    if(f){printf("NO\n");return 0;}
    for(i=1;i<=n;i++){
        for(x=j=1;j<=n;j++) if(j!=i&&d[i][j]<d[i][x]) x=j;
        for(j=1;j<=n;j++) if(abs(d[i][j]-d[j][x])!=d[x][i]) f=1;    
    }
    if(f) puts("NO");
    else puts("YES");
    return 0;
}
