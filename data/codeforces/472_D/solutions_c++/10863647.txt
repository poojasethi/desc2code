#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;

int num[2060][2060];
int n;

int main()
{
    scanf("%d",&n);
    for(int i = 0;i<n;i++)
        for(int j = 0;j<n;j++)
    {
        scanf("%d",&num[i][j]);
    }
    int flag = 0;
    for(int i = 0;i<n;i++)
        for(int j = i;j<n;j++)
    {
        if(num[i][j]!=0&&i==j) flag = 1;
        if(num[i][j]!=num[j][i]) flag = 1;
        if(num[i][j]==0&&i!=j) flag = 1;
    }

    for(int i = 0;i<n;i++)
    {
        int inf = 1000000006;
        int pos = -1;
        for(int j = 0;j<n;j++)
        {
            if(i==j) continue;
            if(num[i][j]<inf)
            {
                inf = num[i][j];
                pos = j;
            }
        }

        for(int j = 0;j<n;j++)
        {
           if(j==i||j==pos) continue;
           if(abs(num[i][j]-num[pos][j])!=inf)
                flag = 1;
        }
    }
    if(flag == 1) puts("NO");
    else puts("YES");
    return 0;
}
