#include <iostream>
#include<bits/stdc++.h>

using namespace std;

int main() 
{
	int n,i,j,child,Min,ans;
	cin>>n;
	int a[n+1][n+1];
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			cin>>a[i][j];
		}
	}
	ans=1;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			if(i==j && a[i][j]!=0)ans=0;
			if(i!=j && a[i][j]==0)ans=0;
			if(a[i][j]!=a[j][i])ans=0;
		}
	}
	//cout<<ans<<endl;
	for(i=1;i<=n&&ans;i++)
	{
		int child,Min=1000*1000*1000+8;
		for(j=1;j<=n&&ans;j++)
		{
			if(i!=j && Min>a[i][j])
			{
				child=j;
				Min=a[i][j];
			}
		}
		
		for(j=1;j<=n&&ans;j++)
		{
			if(i!=j && j!=child &&abs(a[i][j]-a[child][j])!=a[i][child])
			{
			//	cout<<i<<"  "<<child<<endl;
				ans=0;
				break;
			}
		}
	}
	//cout<<ans<<endl;
	if(ans)cout<<"YES"<<endl;
	else cout<<"NO"<<endl;
	
	return 0;
}
