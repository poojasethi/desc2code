#include<iostream>
#include<vector>
#include<algorithm>

using namespace std ;

int n ;
bool f=true ;
int a[2010][2010] ;

int main()
{
	ios_base::sync_with_stdio(0) ;
	cin >> n ;
	for(int i=0 ; i<n; i++)
		for(int j=0 ; j<n ; j++)
			cin >> a[i][j] ;
	for(int i=0 ; i<n ; i++)
		for(int j=0 ; j<n; j++)
			if(a[i][j]!=a[j][i] || (i!=j && a[i][j]==0))
				f=false ;
	for(int i=0 ; i<n; i++)
		if(a[i][i]!=0)
			f=false ;
	for(int i=0 ; i<n ; i++)
	{
		int k=0 ;
		for(int j=0 ; j<n; j++)
			if(i!=j && a[i][j]<a[i][k])
				k=j ;
		for(int j=0 ; j<n ; j++)
			if(abs(a[i][j]-a[k][j])!=a[i][k])
				f=false ;
	}
	if(f)
		cout << "YES" << endl ;
	else
		cout << "NO" << endl ;
	return 0 ;
	
}
