# include <bits/stdc++.h>
# define nmax 2005
# define fi cin
# define fo cout
using namespace std;
long long S[nmax][nmax],n,ok=1;
int main(void)
{
     ios_base::sync_with_stdio(0);
     fi>>n;
     for(int i=1;i<=n;++i) for(int j=1;j<=n;++j) fi>>S[i][j];
     for(int i=1;i<=n;++i) for(int j=1;j<=n;++j) if (S[i][j]!=S[j][i] || (i!=j && !S[i][j])) ok=0;
     for(int i=1;i<=n;++i) if (S[i][i]) ok=0;
     for(int i=1;i<=n;++i)
     {
         int p=1;
         for(int j=1;j<=n;++j) if (i!=j && S[i][j]<S[i][p]) p=j;
         for(int j=1;j<=n;++j) if (abs(S[i][j]-S[p][j])!=S[i][p]) ok=0;
     }
     fo<<(ok ? "YES\n":"NO\n");
}
