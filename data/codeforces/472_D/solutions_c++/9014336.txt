#include <iostream>
#include <stdio.h>
#include <algorithm>
#define read(x) scanf("%d",&x)
using namespace std;

int n,p;
int a[2005][2005];
int main(){
   read(n);       
   for(int k=1;k<=n;k++){
    for(int k2=1;k2<=n;k2++){
           read(a[k][k2]);if(a[k][k2]==0&&k!=k2){cout<<"NO"<<endl;return 0;} 
    }        
   }
   for(int k=1;k<=n;k++)if(a[k][k]){cout<<"NO"<<endl;return 0;}
   
   for(int k=1;k<=n;k++){
    for(int k2=1;k2<=n;k2++){
     if(a[k][k2]!=a[k2][k]){cout<<"NO"<<endl;return 0;}        
    }        
   }
   for(int k=1;k<=n;k++){p=1;
    for(int k2=1;k2<=n;k2++){
        if(k!=k2&&a[k][k2]<a[k][p])p=k2;     
    }        
    for(int k2=1;k2<=n;k2++){
     if(abs(a[k][k2]-a[p][k2])!=a[k][p]){cout<<"NO"<<endl;return 0;}        
    }       
   }
   cout<<"YES"<<endl;
    return 0;   
}
