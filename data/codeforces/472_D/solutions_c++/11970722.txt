#include <bits/stdc++.h>
using namespace std;
int mat[2009][2009];
#define inf 1000000009
int main()
{
    int n;
    scanf("%d",&n);
    for (int i = 1; i<=n; i++)
        for (int j = 1; j<=n; j++)
            scanf("%d",&mat[i][j]);
        int flag = 0;
    for (int i = 1; i<=n; i++)
        for (int j = 1; j<=n; j++){
            if (mat[i][j] != mat[j][i] || (mat[i][j] == 0 && i!=j))
                flag = 1;
            if (mat[i][i] != 0)
                flag = 1;
        }
    for (int i = 1; i<=n; i++)
    {
        int num = inf,cur;
        for (int j = 1; j<=n; j++)
            if (i != j && mat[i][j] < num){
                num = mat[i][j];
                cur = j;
            }

        for (int j = 1; j<=n; j++)
            if (i!=j && abs(mat[i][j] - mat[cur][j]) != num)
                flag = 1;
    }
    if (flag == 0)
        cout<<"YES\n";
    else
        cout<<"NO\n";
    return 0;
        
}
