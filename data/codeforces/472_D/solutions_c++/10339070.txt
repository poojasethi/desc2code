#include <iostream>
#include <stdlib.h>
#include <stdio.h>
#include <algorithm>
#include <math.h>
#include <limits.h>
using namespace std;

long long t[2001][2001];
int n;
long long mini=LONG_MAX;
int pos;
int main(){

    cin>>n;
    
    for (int i=1;i<=n;i++) for(int j=1;j<=n;j++) scanf("%d",&t[i][j]);

    for (int i=1;i<=n;i++) for (int j=1;j<=n;j++) if ((i==j && t[i][j]!=0) || ((t[i][j]!=t[j][i] || t[i][j]==0) && i!=j)) {cout<<"NO"<<endl;exit(0);}

    for (int i=1;i<=n;i++){
        mini=LONG_MAX;pos=0;
        for (int j=1;j<=n;j++) if (mini>t[i][j] && i!=j) mini=t[i][j],pos=j;
        for (int j=1;j<=n;j++) if (i!=j && j!=pos && abs(t[i][j]-t[pos][j])!=t[i][pos]) {cout<<"NO"<<endl;exit(0);}
    }
    cout<<"YES"<<endl;
return 0;}
