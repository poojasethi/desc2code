#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;

const int maxn = 2005;
int d[maxn][maxn];
int n;

void solve(){
	for(int i = 1;i <= n;i++){
		for(int j = 1;j <= n;j++){
			if((i == j && d[i][j] != 0) || (i != j && d[i][j] == 0)){
				puts("NO");
				return;
			}
		}
	}
	
	int minn = (1<<30)-1,v;
	for(int i = 1;i <= n;i++){
		minn = (1<<30)-1;
		for(int j = 1;j <= n;j++){
			if(i == j) continue;
			if(d[i][j] < minn){
				minn = d[i][j];
				v = j;
			}
		}
		for(int k = 1;k <= n;k++){
			if(k == i || k == v) continue;
			if(abs(d[i][k] - d[k][v]) != minn){
				puts("NO");
				return;
			}
		}
	}
	puts("YES");
}

int main(){
	while(scanf("%d",&n) != EOF){
		memset(d,0,sizeof(d));
		for(int i = 1;i <= n;i++)
		 for(int j = 1;j <= n;j++) scanf("%d",&d[i][j]);
		 
		 solve();
	}
	return 0;
}
