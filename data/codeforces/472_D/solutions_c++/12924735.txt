#include <bits/stdc++.h>

using namespace std;

const int N = 2010;

int n;
int d[N][N];

int main() {
  scanf("%d", &n);
  for (int i = 0; i < n; ++i) {
    for (int j = 0; j < n; ++j) {
      scanf("%d", d[i] + j);
    }
  }
  int dd = 1;
  for (int i = 0; i < n; ++i) {
    if (d[i][i] != 0) {
      dd = 0;
    }
  }
  for (int i = 0; i < n; ++i) {
    for (int j = i + 1; j < n; ++j) {
      if (d[i][j] != d[j][i] || d[i][j] <= 0)
        dd = 0;
    }
  }
  for (int i = 0; i < n; ++i) {
    int pos = -1;
    int mn = 2e9;
    for (int j = 0; j < n; ++j) {
      if (i != j && mn > d[i][j]) {
        pos = j;
        mn = d[i][j];
      }
    }
    for (int j = 0; j < n; ++j) {
      if (i != j && abs(d[i][j] - d[j][pos]) != d[i][pos]) {
        dd = 0;
      }
    }
  }
  if (dd) {
    printf("YES\n");
  }
  else {
    printf("NO\n");
  }
  return 0;
}
