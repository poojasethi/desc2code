#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
#define io_speed ios_base::sync_with_stdio(0);cin.tie(0)
const int N = 2005;
int n;
int mp[N][N];
bool check(){
    for(int i=1;i<=n;++i){
        if(mp[i][i]!=0) return false;
        for(int j=i+1;j<=n;++j){
            if(mp[i][j]!=mp[j][i]) return false;
            if(mp[i][j]<=0) return false;
        }
    }
    return true;
}

bool solve(){
    for(int i=1;i<=n;++i){
        int pos=1;
        for(int j=1;j<=n;++j)
            if(i!=j&&mp[i][pos]>mp[i][j]) pos=j;
        for(int j=1;j<=n;++j)
            if(abs(mp[i][j]-mp[pos][j])!=mp[i][pos]) return false;
    }
    return true;
}

int main(){
    io_speed;
    cin>>n;
    for(int i=1;i<=n;++i)
        for(int j=1;j<=n;++j)
            cin>>mp[i][j];
    if(check()&&solve()) puts("YES");
    else puts("NO");
    return 0;
}
