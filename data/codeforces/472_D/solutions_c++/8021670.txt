#include<iostream>
#include<stdio.h>
#include<Algorithm>
using namespace std;
#include<cstring>
#include<vector>
#define INF 0x7fffFFFF
#define MAXN 4000010
int cnt=0;  
int n;
int a[2010][2010];
int main()
{

    scanf("%d",&n);
    memset(a,-1,sizeof(a));
    int flag=0;
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++)
        {
            scanf("%d",&a[i][j]);
            if(i==j&&a[i][j]!=0)flag=1;
            else if(a[i][j]!=-1&&a[j][i]!=-1&&a[i][j]!=a[j][i])flag=1;
            else if(i!=j&&a[i][j]==0)flag=1;
        }
    }
   
    for(int i=0;i<n;i++){
    
        int delta=INF,pos=-1;
        for(int j=0;j<n;j++){
            if(i==j)continue;
            if(a[i][j]<delta)delta=a[i][j],pos=j;
        }
        for(int j=0;j<n;j++){
            if (i == j || j == pos) continue;
            if(abs(a[i][j]-a[pos][j])!=delta)flag=1;
        
        }
    }
    if(flag){
        printf("NO\n");
        return 0;
    }
    printf("YES\n");
    return 0;
}
