#include <bits/stdc++.h>
using namespace std;

int main() {
    // your code goes here
    int n;
    cin>>n;
    
    int a[n][n];
    for(int i=0;i<n;i++) for(int j=0;j<n;j++)   cin>>a[i][j];
    
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            if(a[i][j]!=a[j][i]||(i!=j&&a[i][j]==0)||(i==j&&a[i][i]!=0))
            {
                cout<<"NO\n";
                return 0;
            }
        }
    }
    
    for(int i=0;i<n;i++)
    {
        int low=0;
        for(int j=0;j<n;j++)
            if(i!=j&&a[i][j]<a[i][low])
                low=j;
        for(int j=0;j<n;j++)
            if(i!=j&&abs(a[i][j]-a[low][j])!=a[i][low])
            {
                cout<<"NO\n";
                return 0;
            }
    }
    
    cout<<"YES\n";
    return 0;
}
