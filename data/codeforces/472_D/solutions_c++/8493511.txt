#include <cstdio>
#include <cmath>
#include <cstring>
#include <iostream>
#include <algorithm>
int map[2500][2500];
int n,flag;
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;++i)
		for (int j=1;j<=n;++j) scanf("%d",&map[i][j]);
	flag=1;
	for (int i=1;i<=n;++i)
		for (int j=1;j<=n;++j)
		{
			if (map[i][j]!=map[j][i]) flag=0;
			if (map[i][j]==0&&i!=j) flag=0;
		}
	for (int i=1;i<=n;++i)
	{
		int dqmin=19950920,dq=1;
		for (int j=1;j<=n;++j) if (i!=j&&map[i][j]<dqmin)
		{
			dqmin=map[i][j];
			dq=j;
		}
		for (int j=1;j<=n;++j) if (abs(map[i][j]-map[dq][j])!=map[i][dq]) flag=0;
	}
	if (flag) printf("YES");else printf("NO");
	return 0;
}
