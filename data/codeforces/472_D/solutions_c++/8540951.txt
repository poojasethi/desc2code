#include <cstdio>
#include <algorithm>
#include <climits>
#define N 2010


int n;
int d[N][N];

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= n; ++j)
			scanf("%d", &d[i][j]);

	for (int i = 0; i <= n; i++)
	{
		d[i][0] = d[0][i] = INT_MAX;
	}

	for (int i = 1; i <= n; ++i)
	{
		if (d[i][i] != 0)
		{
			printf("NO\n");
			return 0;
		}
		for (int j = 1; j <= n; ++j)
		{
			if (i != j && (d[i][j] != d[j][i] || d[i][j] == 0))
			{
				printf("NO\n");
				return 0;
			}
		}
	}

	for (int i = 1; i <= n; ++i)
	{
		int jmin = 0;
		for (int j = 1; j <= n; ++j)
		{
			if (i != j && (d[i][jmin] > d[i][j]))
			{
				jmin = j;
			}
		}

		for (int j = 1; j <= n; ++j)
		{
			if (i != j && (std::abs(d[i][j] - d[jmin][j]) != d[i][jmin]))
			{
				//printf("%d %d %d| %d %d %d\n", i, j, jmin, d[i][j], d[jmin][j], d[i][jmin]);
				printf("NO\n");
				return 0;
			}
		}
	}

	printf("YES\n");
	return 0;
}
