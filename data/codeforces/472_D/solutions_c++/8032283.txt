#include <cstdio>
using namespace std;
int mp[2002][2002];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=n;j++)
        {
            scanf("%d",&mp[i][j]);
        }
    }
    bool yes=true;
    for(int i=1;i<=n&&yes;i++)
    {
        int son,dis=(int)1e9+10086;
        for(int j=1;j<=n&&yes;j++)
        {
            if(i==j&&mp[i][j]!=0)yes=false;
            if(i!=j&&mp[i][j]==0)yes=false;
            if(mp[i][j]!=mp[j][i])yes=false;
            if(i==j)continue;
            if(dis>mp[i][j])
                dis=mp[i][j],son=j;
        }
        for(int j=1;j<=n;j++)
        {
            if(i==j)continue;
            if(mp[i][j]-mp[son][j]==mp[i][son]||mp[i][j]+mp[i][son]==mp[son][j])
                continue;
            else
                yes=false;
        }
    }
    if(yes)
        printf("YES\n");
    else
        printf("NO\n");
    return 0;
}
