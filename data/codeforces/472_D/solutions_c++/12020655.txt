#include <bits/stdc++.h>
#define FOR(i,a,b)  for(int i=a; i<=b; i++)
#define maxn 2005

using namespace std;

int d[maxn][maxn], n, w = 1;

int main()
{
    ios_base::sync_with_stdio(0),cin.tie(0),cerr<<"";
    cin>>n;
    FOR(i,1,n)  FOR(j,1,n)  cin>>d[i][j];
    FOR(i,1,n)  FOR(j,1,n)  if(d[i][j] != d[j][i] || (!d[i][j] && i!=j) || (i==j && d[i][j])) w = 0;
    FOR(i,1,n)
    {
        int r = 1;
        FOR(j,1,n)  if(i != j && d[i][j] < d[i][r]) r = j;
        FOR(k,1,n)  if(abs(d[i][k] - d[r][k]) != d[i][r]) w = 0;
    }
    if(w == 1) cout<<"YES";
    else cout<<"NO";
}
