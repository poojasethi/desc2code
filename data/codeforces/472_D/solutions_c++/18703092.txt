#include <bits/stdc++.h>
using namespace std;

int flag = 1, n, t[2001][2001], p[2001];

int main() {
	ios_base::sync_with_stdio(0), cin.tie(0), cout.tie(0), setbuf(stdout, NULL);
	cin >> n;
	for (int i = 1; i < n + 1; i++) {
		for (int j = 1; j < n + 1; j++) {
			cin >> t[i][j];
		}
	}
	for (int i = 1; i < n + 1; i++) {
		for (int j = i + 1; j < n + 1; j++) {
			if ((i == j && t[i][j]) || (t[i][j] != t[j][i])
					|| (i != j && !t[i][j]))
				flag = 0;
		}
	}

	int mid = 1;

	for (int i = 1; i <= n; i++) {
		mid = 1;
		//a b c
		//1 2 3
		for (int j = 1; j <= n; j++) {
			if (i != j && t[i][j] < t[i][mid])
				mid = j;
		}
		//2
		//1-3=2
		for (int m = 1; m <= n; m++)
			if (abs(t[i][m] - t[mid][m]) != t[i][mid])
				flag = 0;
	}

	if (!flag)
		cout << "NO";
	else
		cout << "YES";
	return 0;
}
