#include<bits/stdc++.h>
using namespace std;
int n, d[2005][2005];

bool check() {
	for(int i = 1; i <= n; ++ i)
		for(int j = 1; j <= n; ++ j)
			if(i != j && d[i][j] == 0 || d[i][j] != d[j][i]) return false; 
	
	for(int i = 1; i <= n; ++ i) {
		int p = 1, mv = 0x3f3f3f3f;
		
		for(int j = 1; j <= n; ++ j)
			if(i != j && d[i][j] < mv)
				mv = d[i][p=j];
		
		for(int j = 1; j <= n; ++ j)
			if(abs(d[i][j] - d[p][j]) != d[i][p]) return false;
	}
	return true;
}

int main() {
	scanf("%d", &n);
	for(int i = 1; i <= n; ++ i) for(int j = 1; j <= n; ++ j) scanf("%d", &d[i][j]);
	printf(check() ? "YES" : "NO");
}
