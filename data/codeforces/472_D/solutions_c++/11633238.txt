#include <bits/stdc++.h>
using namespace std;
int n,d[2001][2001],ans;
int main()
{   
 #ifndef ONLINE_JUDGE
 #endif
 ios::sync_with_stdio(0);
 cin.tie(0);
 cin >> n;
 for(int i=1;i<=n;++i)
  for(int j=1;j<=n;++j)
    {
        cin >> d[i][j];
    }
    ans = 1;
 for(int i=1;i<=n;++i)
  if(d[i][i] != 0) ans = 0;
 for(int i=1;i<=n;++i)
    for(int j=1;j<=n;++j)
        if(d[i][j] != d[j][i] || (d[i][j] == 0 && i!=j)) ans = 0;
 for(int i=1;i<=n;++i)
 {
    int min = 1;
    for(int j=1;j<=n;++j)
        if(i != j && d[i][j] < d[i][min]) min = j;
    for(int j=1;j<=n;++j)
    {
        if(abs(d[i][j]-d[min][j]) != d[min][i]) ans = 0;
    }
 }
 if(ans == 1)
    cout << "YES" << endl;
 else
    cout << "NO" << endl;
}
