#include<stdio.h>
#include<stdlib.h>

#define NO {puts("NO");return 0;}
#define YES {puts("YES");return 0;}

int i,j,min,n,mat[6000][6000];

int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			scanf("%d",&mat[i][j]);
				
	for(i=1;i<=n;i++)
		if(mat[i][i])
			NO;
			
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if((i!=j &&!mat[i][j] )|| mat[i][j] != mat[j][i])	
				NO;
				
	for(i=1;i<=n;i++){
		min = 1;
		for(j=1;j<=n;j++)
			if(i!=j && mat[i][j] < mat[i][min])
				min = j;
				
		for(j=1;j<=n;j++)
			if(abs(mat[i][j] - mat[min][j]) != mat[i][min]) 
				NO;
	}
	YES;
	
}
