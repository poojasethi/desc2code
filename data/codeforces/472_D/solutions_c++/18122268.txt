#include <iostream>
#include <stdio.h>
#include <stdlib.h>
int grid[2005][2005];
using namespace std;
int main()
{
    int li,flag=0,t=0;
    scanf("%d",&li);
    register int i,j,k;
    for(i=0; i<li; i++)
    {
        //t=0;
        for(j=0; j<li; j++)
        {
            scanf("%d",&grid[i][j]);

            if(j<i&&grid[j][i]!=grid[i][j]) flag=1;
            if(i==j && grid[i][j]!=0) flag=1;
            else if(i!=j&&!grid[i][j]) flag=1;
        }
        //if(!t&&li>1) flag=1;
    }

    if(flag)
    {
        printf("NO\n");
        return 0;
    }
    else
    {
        for(i=0 ; i<li ; i++)
        {
            k=0 ;
            for(j=0 ; j<li; j++)
                if(i!=j && grid[i][j]<grid[i][k])
                    k=j ;
            for(int j=0 ; j<li ; j++)
                if(abs(grid[i][j]-grid[k][j])!=grid[i][k])
                    {
                        printf("NO\n");
                        return 0;
                    }
        }
    }
    printf("YES\n");

//main();
    return 0;
}
