#include<cstdio>
#include<map>
#include<algorithm>
using namespace std;
int n, i, j, A[2001][2001];
map<int, int> C;
int main()
{
	scanf("%d", &n);
	for (i = 1; i <= n; i++)
		for (j = 1; j <= n; j++)
			scanf("%d", &A[i][j]);
	for (i = 1; i <= n; i++)
		for (j = i + 1; j <= n; j++)
			if (A[i][j] != A[j][i] || A[i][j] == 0)
			{
				printf("NO\n"); return 0;
			}
	for (i = 1; i <= n; i++) if (A[i][i]) {printf("NO\n"); return 0;}
	for (i = 1; i <= n; i++) C[2 * A[1][i]] = 1;
	for (i = 1; i <= n; i++)
		for (j = i + 1; j <= n; j++)
			if (abs(A[1][i] - A[1][j]) != A[i][j] && !C[A[1][i] + A[1][j] - A[i][j]])
			{
				printf("NO\n"); return 0;
			}
	printf("YES");
	return 0;
}
