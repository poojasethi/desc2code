#include <bits/stdc++.h>

using namespace std;

int n;
int a[2000][2000];

bool check()
{
	for(int i = 0; i < n; ++i) {
		if(a[i][i]) {
			return false;
		}
	}
	for(int i = 0; i < n; ++i) {
		for(int j = 0; j < n; ++j) {
			if((i != j) && ((a[i][j] == 0) || (a[i][j] != a[j][i]))) {
				return false;
			}
		}
	}

	for(int i = 0; i < n; ++i) {
		int min_id = 0;
		for(int j = 0; j < n; ++j) {
			if((i != j) && (a[i][j] < a[i][min_id])) {
				min_id = j;
			}
		}
		for(int j = 0; j < n; ++j) {
			if(abs(a[min_id][j] - a[i][j]) != a[i][min_id]) {
				return false;
			}
		}
	}

	return true;
}

int main()
{
	bool ok = true;
	cin >> n;
	for(int i = 0; i < n; ++i) {
		for(int j = 0; j < n; ++j) {
			cin >> a[i][j];
		}
	}
	if(!check()) {
		cout << "NO" << endl;
	} else {
		cout << "YES" << endl;
	}

	
	
	return 0;
}
