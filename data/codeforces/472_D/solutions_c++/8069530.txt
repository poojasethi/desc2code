#include <iostream>
#include <cstdio>

using namespace std;

const int maxn=2005;
const int inf=100000000;

int a[maxn][maxn];
int n;

int abs(int a)
{
	return a>0?a:-a;
}

int main()
{
	cin>>n;
	int i,j;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			scanf("%d",&a[i][j]);
			if(i==j && a[i][j]!=0 || i!=j && a[i][j]==0)
			{
				cout<<"NO\n";
				return 0;
			}			
		}
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			if(a[i][j]!=a[j][i])
			{
				cout<<"NO\n";
				return 0;
			}
		}
	}
	int r;
	for(i=1;i<=n;i++)
	{
		r=1;
		for(j=1;j<=n;j++)
		{
			if(a[i][r]>a[i][j] && i!=j)
			{
				r=j;
			}
		}
		for(int k=1;k<=n;k++)
		{
			if(abs(a[i][k]-a[r][k]) != a[i][r])
			{
				cout<<"NO\n";
				return 0;
			}
		}
	}
	cout<<"YES\n";
	return 0;
}
