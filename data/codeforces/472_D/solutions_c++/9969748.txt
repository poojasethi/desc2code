#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int d[2000][2000];

int main(){  
    int n;
    cin >> n;
    
    bool ans = true;
    
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            scanf("%d", &(d[i][j]));
        }
    }
    
    for(int i = 0; i < n; i++) {
        if(d[i][i]) ans = false;
    }
    
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            if(d[i][j] != d[j][i] || (i != j && d[i][j] == 0)) {
                ans = false;
            }
        }
    }
    
    for(int i = 0; i < n; i++) {
        int k = 0;
        for(int j = 0; j < n; j++) {
            if(i != j && d[i][j] < d[i][k]) k = j;
        }
        
        for(int j = 0; j < n; j++) {
            if(abs(d[i][j] - d[k][j]) != d[i][k]) {
                ans = false;
            }
        }
    }
    
    if(ans) {
        cout << "YES";
    } else {
        cout << "NO";
    }
    
    return 0;
}
