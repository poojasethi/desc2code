#include <stdio.h>
#include<map>
using namespace std;
const int MAXN = 2010;
int m[MAXN][MAXN];
int n;
map<int, int> ma;
int main()
{
    scanf("%d",&n);
    int resp = 1;
    for(int i=1;i<=n;i++) {
        for(int j=1;j<=n;j++){
            scanf("%d",&m[i][j]);
            if(i==j && m[i][j]!=0) resp=0;
            if(i==1) ma[m[i][j]]=1;
        }
    }
    for(int i=1;i<=n;i++) {
        for(int j=1;j<=n;j++){
            if(m[i][j]!=m[j][i]) resp=0;
            if(m[i][1]==0 && i!=1) resp=0;
            if(m[j][1]==0 && j!=1) resp=0;
            int d = m[i][1] + m[j][1] - m[i][j];
            if(d<0 || d%2==1) resp=0;
            d=d/2;
            if(ma.find(d) == ma.end()) resp=0;
        }
    }
    if(resp) printf("YES\n");
    else printf("NO\n");
}
