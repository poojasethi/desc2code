#include <iostream>
#include <algorithm>
#include <cstdio>
#include <queue>
#include <string>
#include <vector>
using namespace std;

int n;
long long  g[2010][2010];

bool flag = false;

void fuck(){
	if(flag){
		puts("NO");
		exit(0);
	}
}
long long ABS(long long u){
	if(u < 0)return -u;
	return u;
}
int main(){
	scanf("%d",&n);
	for(int i = 1;i <= n;i++)
		for(int j = 1;j <= n;j++)
			cin >> g[i][j];
	for(int i = 1;i <= n;i++)
		for(int j = 1;j <= n;j++)
			if(i!=j && g[i][j] == 0) flag = true;
			else if(i==j && g[i][j] != 0)flag = true;
	fuck();
	for(int i = 1;i <= n;i++)
		for(int j = i+1; j <= n;j++)
			if(g[i][j] != g[j][i])
				flag = true;
	fuck();
	for(int i = 1;i <= n && !flag;i++){
		int r = 1;
		for(int j = 1;j <= n;j++)
				if(i!=j && g[i][j] < g[i][r]) r = j;
		for(int j = 1;j <= n && !flag;j++)
			if( ABS(g[r][j] - g[i][j]) != g[i][r])
				flag = true;
	}
	fuck();
	puts("YES");
	return 0;
}
