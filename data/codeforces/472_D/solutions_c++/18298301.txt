#include <cstdio>
#include <cmath>
using namespace std;
const int maxn=2e3 ;
int dis[maxn][maxn] ;
int main()
{
    int n;
    scanf("%d",&n) ;
    for(int i=0;i<n;i++)
	for(int j=0;j<n;j++)
	    scanf("%d",&dis[i][j]) ;
    for(int i=0;i<n;i++)
	if(dis[i][i]!=0)
	{
	    printf("NO") ;
	    return 0;
	}
    for(int i=0;i<n;i++)
	for(int j=0;j<i;j++)
	    if(dis[i][j]!=dis[j][i] || dis[i][j]==0)
	    {
		printf("NO") ;	
		return 0;
	    }
    for(int i=0;i<n;i++)
    {	
	int r=0 ;
	int d=1e9+3 ;
	for(int j=0;j<n;j++)
	    if(j!=i && dis[i][j]<d)
	    {	
		r=j ;
		d=dis[i][j] ;
	    }
	for(int j=0;j<n;j++)
	    if(j!=i && j!=r && fabs(dis[i][j]-dis[r][j])!=dis[i][r])
	    {
		printf("NO") ;
		return 0;
	    }
    }
    printf("YES") ;
}
