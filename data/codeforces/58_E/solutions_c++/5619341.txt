ll pw10[18];
ll getlen(ll n){
	ll ret = 0;
	while(n)++ret,n/=10;
	return ret;
}
void dfs(ll a, ll b, ll c, ll carry, ll curra, ll currb, ll lenc, ll pos){
	ll totlen = getlen(curra) + getlen(currb) + getlen(curra+currb);
	if(lenc > 12 || totlen >= totlenans ) return;
	if(a+b+c+carry==0 && totlen < totlenans){
		totlenans = totlen;
		ansa=curra;
		ansb=currb;
		return;
	}
	if(c == 0){
		return dfs(0, 0, 0, 0, curra + a*pw10[pos], currb + b*pw10[pos], lenc+  getlen(a+b+carry), pos);
	}
	if((a%10+b%10+carry)%10 == c%10 ){
		dfs(a/10,b/10,c/10,(a%10+b%10+carry)/10,curra+(a%10)*pw10[pos],currb+(b%10)*pw10[pos],
				lenc,pos+1);
	}

	ll n = (a%10+b%10 + carry)%10;
	if(n!=c%10)
		dfs(a,b,c*10+n,carry,curra,currb,lenc+1,pos);

	n = (c%10 + 10 -a%10 - carry)%10;
	if(n!=b%10)
		dfs(a,b*10+n,c,carry,curra, currb,lenc+1,pos);

	n = (c%10 + 10 -b%10 - carry)%10;
	if(n!=a%10)
		dfs(a*10+n,b,c,carry,curra, currb,lenc+1,pos);
}
