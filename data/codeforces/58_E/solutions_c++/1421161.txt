#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int x,y,z,h[10],a[8],b[8],c[8],ans,f[15][15][15][8][8][8][2],g[15][15][15][8][8][8][2][3],s[4];

void expand(int i,int cn1,int cn2,int j1,int j2,int j3,int cc)
{
    int hj1,hj2,hj3,hcn1,hcn2,hcc,ad;
    for (int d1=0;d1<10;d1++)
    {
        for (int d2=0;d2<10;d2++)
        {
			ad=0;
            if (j1<a[0] && d1==a[j1+1]) hj1=j1+1;else hj1=j1;
            if (j2<b[0] && d2==b[j2+1]) hj2=j2+1;else hj2=j2;
            if (d1==0) hcn1=cn1+1;else { hcn1=0;ad+=cn1+1; }
            if (d2==0) hcn2=cn2+1;else { hcn2=0;ad+=cn2+1; }
			if (d1+d2+cc>9) hcc=1;else hcc=0;
			int d3=(d1+d2+cc)%10;
			if (j3<c[0] && d3==c[j3+1]) hj3=j3+1;else hj3=j3;
        //    printf("%d %d %d %d %d %d %d\n",i+1,hcn1,hcn2,hj1,hj2,hj3,hcc);
			if (f[i+1][hcn1][hcn2][hj1][hj2][hj3][hcc]<0 || f[i][cn1][cn2][j1][j2][j3][cc]+ad+1<f[i+1][hcn1][hcn2][hj1][hj2][hj3][hcc])
			{
				f[i+1][hcn1][hcn2][hj1][hj2][hj3][hcc]=f[i][cn1][cn2][j1][j2][j3][cc]+ad+1;
				g[i+1][hcn1][hcn2][hj1][hj2][hj3][hcc][0]=g[i][cn1][cn2][j1][j2][j3][cc][0]+h[i]*d1;
				g[i+1][hcn1][hcn2][hj1][hj2][hj3][hcc][1]=g[i][cn1][cn2][j1][j2][j3][cc][1]+h[i]*d2;
				g[i+1][hcn1][hcn2][hj1][hj2][hj3][hcc][2]=g[i][cn1][cn2][j1][j2][j3][cc][2]+h[i]*d3;
			}
        }
    }
}

int main()
{
  //  freopen("expression.in","r",stdin);freopen("expression.out","w",stdout);

	h[0]=1;
	for (int i=1;i<=9;i++) h[i]=h[i-1]*10;
    scanf("%d+%d=%d",&x,&y,&z);
    while (x) a[++a[0]]=x%10,x/=10;
    while (y) b[++b[0]]=y%10,y/=10;
    while (z) c[++c[0]]=z%10,z/=10;
    memset(f,255,sizeof(f));
    f[0][0][0][0][0][0][0]=0;ans=100;
    for (int i=0;i<=12;i++)
    {
        for (int cn1=0;cn1<=i;cn1++)
            for (int cn2=0;cn2<=i;cn2++)
                for (int j1=0;j1<=a[0];j1++)
                    for (int j2=0;j2<=b[0];j2++)
                        for (int j3=0;j3<=c[0];j3++)
                            for (int cc=0;cc<=1;cc++)
                            {
								if (f[i][cn1][cn2][j1][j2][j3][cc]<0) continue;
                                expand(i,cn1,cn2,j1,j2,j3,cc);
                                if (j1==a[0] && j2==b[0] && j3==c[0] && cc==0 && f[i][cn1][cn2][j1][j2][j3][cc]<ans)
                                {
                                    ans=f[i][cn1][cn2][j1][j2][j3][cc];
                                    for (int p=0;p<=2;p++) s[p]=g[i][cn1][cn2][j1][j2][j3][cc][p];
                                }
                            }
    }
	printf("%d+%d=%d\n",s[0],s[1],s[2]);
	//printf("%d",ans);

    return 0;
}
