#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,i,j=0;
	scanf("%d%d",&n,&m);
	long long a[n],b[m];
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<m;i++)
	cin>>b[i];
	long long ans=0;
	for(i=0;i<n;i++)
	{
		if(a[i]<=b[j])
		{
		long long temp;
		if(j>0)
		{
		temp=min(b[j]-a[i],a[i]-b[j-1]);
		ans=max(ans,temp);
		}		
		else ans=max(ans,b[j]-a[i]);
		
		}
		else
		{
		if(j+1<m)
		{
		i--;
		j++;
		}
		else break;
		}
	}	
	for(;i<n;i++)
	{
	ans=max(ans,abs(a[i]-b[m-1]));
	}
	cout<<ans<<endl;
	return 0;
}
