#include<iostream>
using namespace std;
int abs(int x){
	return x>0?x:-x;
}
int city[100005],tower[100005];
int main(){
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>city[i];
	for(int i=0;i<m;i++)
		cin>>tower[i];
	int rmax=0, prevtower=0,nextower=0,curcity=0;
	while(curcity<n && nextower<m){
		if(city[curcity]>tower[nextower] && nextower!= (m-1)){
			prevtower=nextower;
			nextower++;
		}
		else{
			rmax=max(rmax,min(abs(tower[prevtower]-city[curcity])
				,abs(tower[nextower]-city[curcity])));
			curcity++;
		}
	}
	cout<<rmax<<endl;
}
