#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#define MAXN 100005
using namespace std;

int n,m,ans=0;
int a[MAXN],b[MAXN];

int main() {
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++) {
		scanf("%d",a+i);
	}
	for (int i=1;i<=m;i++) {
		scanf("%d",b+i);
	}
	int tower=1;
	for (int i=1;i<=n;i++) {
		int dis=abs(a[i]-b[tower]);
		while (tower<m && dis>=abs(a[i]-b[tower+1])) {
			dis=abs(a[i]-b[++tower]);
		}
		ans=max(ans,dis);
	}
	printf("%d\n",ans);
	return 0;
}
