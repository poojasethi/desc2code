#include<iostream>
#include<stdlib.h>
#include<math.h>
using namespace std;
int main()
{
    long int n,i,m,j=0;
    long long int r=0,a[100000],b[100000];
    cin>>n>>m;
    for(i=0;i<n;i++)
        cin>>a[i];
    for(i=0;i<m;i++)
        cin>>b[i];
    for(i=0;i<n;i++)
    {
        while(j<m-1 && abs(a[i]-b[j])>=abs(a[i]-b[j+1]))
            j++;
        if(r < abs(a[i]-b[j]))
            r=abs(a[i]-b[j]);
    }
    cout<<r<<endl;
    return 0;
}
