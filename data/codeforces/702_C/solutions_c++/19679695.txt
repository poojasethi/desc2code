#include <iostream>
#include <cmath>
using namespace std;

int main() {
	int n, m;
	cin>>n>>m;
	int a[100100], b[100100];
	for (int i=0; i<n; i++) 
		cin>>a[i];
	for (int i=0; i<m; i++)
		cin>>b[i];
	int ans=-1, j=0;
	for (int i=0; i<n; i++) {
		while (j<m-1 && abs(a[i]-b[j])>=abs(a[i]-b[j+1]))
			j++;
		ans=max(ans, (int)abs(a[i]-b[j]));
	}
	cout<<ans<<'\n';
	return 0;
}
