#include<cstdio>
#define MAX 100050
#define max(a,b) ((a)>(b)?(a):(b))
#define min(a,b) ((a)<(b)?(a):(b))
#define abs(a) ((a)>=0?(a):-(a))
using namespace std;
int n,m,a[MAX],b[MAX],i,j,r,tmp;
int main()
{
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)scanf("%d",a+i);
	for(j=0;j<m;j++)scanf("%d",b+j);
	i=0,j=0;
	for(i=0;i<n;i++){
		while(j<m-1&&b[j+1]<=a[i])j++;
		tmp=abs(a[i]-b[j]);
		if(j<m-1)tmp=min(tmp,abs(a[i]-b[j+1]));
		r=max(r,tmp);
	}
	printf("%d",r);
	return 0;
}
