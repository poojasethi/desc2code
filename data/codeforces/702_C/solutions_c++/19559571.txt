#include<bits/stdc++.h>
#define maxn 100000
using namespace std;

int n,m;
int A[maxn+2],B[maxn+2];

int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)scanf("%d",A+i);
	for(int i=1;i<=m;i++)scanf("%d",B+i);
	int ans=0;
	for(int i=1,j=1;i<=n;i++){
		while(j<m&&abs(B[j]-A[i])>=abs(B[j+1]-A[i]))++j;
		//cout<<j<<endl;
		ans=max(ans,abs(B[j]-A[i]));
	}
	printf("%d\n",ans);
	return 0;
}
