#include <bits/stdc++.h>
using namespace std;

int x1[100000],x2[100000];
int p,q,r,n,m;
int main()
{
	scanf("%d %d",&n,&m);
	for(int i = 0;i < n;i++){
		scanf("%d",&x1[i]);
	}
	for(int i = 0;i < m;i++){
		scanf("%d",&x2[i]);
	}
	int i = 0,j = 0;
	while(i < n){
		while(j + 1 < m && x2[j+1] < x1[i]) j++;
		if(j + 1 == m){
			if(abs(x1[i] - x2[j]) > r) r = abs(x1[i] - x2[j]);
		}else{
			int a = abs(x1[i] - x2[j]),b = abs(x2[j + 1] - x1[i]);
			if(a > b) a = b;
			if(a > r) r = a;
		}
		i++;
	}
	printf("%d",r);
}
