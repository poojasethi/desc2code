#include<bits/stdc++.h>
using namespace std;
unsigned int r,i=0,j=0;
int n,m,a,max1=0,tmp1=0,tmp2=0;

int main(){
	cin>>n>>m;
	int an[n];
	int am[m];
	for(i=0;i<n;i++)
	{
		cin>>an[i];
	}
	
	for(i=0;i<m;i++)
	{
		cin>>am[i];
	}
	i=0;j=0;
	
	max1 = 0;
	int j=0;
	for(int i=0;i<n;i++)
	{
		while(j<m-1 && abs(an[i]-am[j])>=abs(an[i]-am[j+1])) j++;
		max1 = max(max1,abs(an[i]-am[j]));
	}
	cout<<max1;
}
