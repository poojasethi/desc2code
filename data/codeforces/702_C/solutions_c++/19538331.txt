# include <bits/stdc++.h>

using namespace std;

int b[100005];
map<int,int> a;

int main()
{
	int n,m,i,j,r;

	cin >> n >> m;
	for(i=0;i<n;i++) cin >> a[i];
	for(i=0;i<m;i++) cin >> b[i];

	r = 0;
	
	j=0;
	for(i=0;i<n;i++)
	{
		while(j<m-1 && abs(b[j]-a[i]) >= abs(b[j+1]-a[i]))
		j++;
		r = max(r,abs(b[j]-a[i]));
	}
	cout << r << "\n";
}
