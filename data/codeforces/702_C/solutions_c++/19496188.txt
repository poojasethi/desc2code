#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    ll n,m,i,j,mi=0;

    cin>>n>>m;

    vector<ll> v1(n),v2(m);

    for(i=0;i<n;i++)
        cin>>v1[i];

    for(i=0;i<m;i++)
        cin>>v2[i];

    i=j=0;

    while(i<n && j<m-1)
    {
        while(j<m-1 && abs(v2[j]-v1[i])>=abs(v2[j+1]-v1[i]))
            j++;

        mi=max(mi,abs(v2[j]-v1[i]));
        i++;
    }

    while(i<n)
    {
        mi=max(mi,abs(v2[j]-v1[i]));
        i++;
    }

    cout<<mi;

    return 0;
}
