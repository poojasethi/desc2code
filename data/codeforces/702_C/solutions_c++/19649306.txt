#include <iostream>
#include <cstdio>
#include <string>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;

int a[100010],b[100010];

int main()
{
	int n,m,i,ans,j;
	cin >> n >> m;
	for(i = 1; i<=n; i++)
		scanf("%d", &a[i]);
	for(i = 1; i<=m; i++)
		scanf("%d", &b[i]);
	ans = 0;
	for(j = i = 1; i<=n; i++)
	{
		while(j<m && abs(b[j+1]-a[i]) <= abs(b[j]-a[i]))
			j++;
		ans = max(ans, abs(b[j]-a[i]));
	}
	cout << ans << endl;
	return 0;
}
