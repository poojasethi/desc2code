#define F(i, n) for( int (i) = 0; (i)<(n); ++(i) )
#define F1(i,n) for(int i=1;i<=(n);i++)
#include<bits/stdc++.h>
using namespace std;

const int maxn = 100000 + 5;

int a[maxn], b[maxn];

int main(){
	ios::sync_with_stdio(false); 
	int n, m;
	cin >> n >> m;
	F(i, n) cin >> a[i];
	F(i, m) cin >> b[i];
	int ans = 0, cur = 0;
	F(i, n){
		while(cur + 1 < m && abs(a[i] - b[cur]) >= abs(a[i] - b[cur + 1]))
		   cur++;
		ans = max(ans, abs(a[i] - b[cur]));
	}
	cout << ans << endl;
	return 0;
}
