#include <iostream>
#include <algorithm>
using namespace std;

int a[100100], n, m, mx;

int main () {
  int i, j;
  cin >> n >> m;
  for (i = 0; i < n; i ++)
    cin >> a[i];
  int cur, prev;
  cin >> cur;
  i = 0;
  for (j = 1; j < m; j ++) {
    prev = cur;
    cin >> cur;
    while (cur >= a[i] && i < n) {
      mx = max (mx, min (abs(a[i]-cur), abs(a[i]-prev)));
      i++;
    }
  }
  for (; i < n; i++)
    mx = max (mx, abs(a[i]-cur));

  cout << mx;
  return 0;
}
