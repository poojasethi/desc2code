#include<iostream>
#include<stdio.h>
#include<string>
#include<math.h>
#include<algorithm>
using namespace std;
int n,m;
int a[101000],b[101000];
int main()
{
	int i,j;
	cin>>n>>m;
	
	for(i=1;i<=n;i++)
		cin>>a[i];
	for(i=1;i<=m;i++)
		cin>>b[i];	
	int ma=0;	
	j=1;
	for(i=1;i<=n;i++)
	{
		while(b[j]<a[i] and j<=m) j++;
		if (j==1) {ma=max(ma,b[j]-a[i]);continue;}
		if (j==m+1) {ma=max(ma,a[i]-b[j-1]);continue;}
		ma=max(ma,min(a[i]-b[j-1],b[j]-a[i]));
		
	}
	cout<<ma;
}
