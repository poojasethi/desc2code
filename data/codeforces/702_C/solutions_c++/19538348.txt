#include<bits/stdc++.h>
using namespace std;
int cit[100002],tow[100002];
inline int dist(int a,int b) { return abs(cit[a]-tow[b]); }
int main(){
	int n,m;
    cin>>n>>m;
    for(int i=0;i<n;i++) cin>>cit[i];
    for(int i=0;i<m;i++) cin>>tow[i];
    int ans=0;
    int cur=0;
    for(int i=0;i<n;i++){
    	while(cur<m-1 && dist(i,cur)>=dist(i,cur+1)) cur++;
    	ans=max(ans,dist(i,cur));
	}
    cout<<ans;   
    return 0;
}
