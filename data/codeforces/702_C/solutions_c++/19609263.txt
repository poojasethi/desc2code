#include<iostream>
using namespace std;
int main()
{
	long long int n,m,a[100000],b[100000],i,ans,lo,mid,hi,j;
	cin>>n>>m;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<m;i++)
		cin>>b[i];
	lo = 0;
	hi = 2000000000;
	while(lo<=hi)
	{
		mid = (lo + hi)/2;
		i = 0;
		j = 0;
		while(i<n&&j<m)
		{
			if(a[i]<=b[j]+mid&&a[i]>=b[j]-mid)
			{
				i++;
			}
			else
				j++;
		}
		if(i==n)
		{
			ans = mid;
			hi = mid - 1;
		}
		else
			lo = mid + 1;
	} 
	cout<<ans<<endl;
	return 0;
}
