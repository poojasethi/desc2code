#include <bits/stdc++.h>

using namespace std;

long long a[111111], b[111111];
#define INF 0x3f3f3f3f3f3f3f3fLL

int main() {
	
	int N, M, i, k;
	long long res;
	
	scanf("%d%d", &N, &M);
	for (i=0; i<N; i++) scanf("%I64d", a+i);
	for (i=0; i<M; i++) scanf("%I64d", b+i);
	b[M++]=-INF; b[M++]=INF;
	
	sort(a, a+N); sort(b, b+M);
	M=unique(b, b+M)-b;
	
	res=0;
	
	for (i=0; i<N; i++) {
		k=lower_bound(b, b+M, a[i])-b;
		res=max(res, min(abs(b[k-1]-a[i]),abs(b[k]-a[i])));
	}

	printf("%I64d\n", res);
	
	return 0;
}
