#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;

const int N = 1e5 + 10;
int n, m, a[N], b[N], pre, ans;

int main(){
	while(~scanf("%d %d", &n, &m)){
		for(int i=0; i<n; i++)	scanf("%d", a+i);
		for(int i=0; i<m; i++)	scanf("%d", b+i);
		b[m] = b[m-1];
		int j = 0;
		pre = b[0];
		ans = 0;
		for(int i=0; i<n; i++){
            while(j<m && a[i]>b[j]){
				pre = b[j];
				j++;
            }
            ans = max(ans, min(abs(a[i]-pre), abs(a[i]-b[j])));
		}
		printf("%d\n", ans);
	}
	return 0;
}
