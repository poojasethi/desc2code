#include<bits/stdc++.h>
using namespace std;

long long A[100005],B[100005], C;
int main()
{
	int N,M;
	cin >> N >> M;
	for(int i = 0 ; i < N ; i++)
		cin >> A[i];
	for(int i = 0 ; i < M; i++)
		cin >> B[i];

	int t = 0;
	for(int i = 0 ; i < N; i++)
	{
		while(t < M-1 && B[t+1] <= A[i])
			t++;
		if(t == M-1)
		C = max(C,abs(B[t]-A[i]));
		else
		C = max(C, min(abs(A[i]-B[t]),abs(B[t+1]-A[i])));

		//cout << i << " " << t << " "<< C << endl;
	}

	cout << C << endl;

}
