#include <bits/stdc++.h>

using namespace std;

int abs(int a) {
    if(a<0) return -a;
    return a;
}

int main() {
    int n, r, m, i, c;
    cin>>n>>m;
    int a[n],  b[m];
    for(int i = 0; i<n; i++) cin>>a[i];
    for(int i = 0; i<m; i++) cin>> b[i];
    i = r=0;
    for(int j =  0; j<n; j++) {
        while(i<m-1&& b[i+1]<a[j])
            i++;
        c = abs(b[i]-a[j]);
        if(i<m-1)
            c = min(abs(b[i+1]-a[j]), c);
        if(c>r)
            r = c;
    }
    cout << r << endl;
    return 0;
}
