#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<string.h>
#include<math.h>
using namespace std;

long long int a[100010];
long long int b[100010];
int main()
{
	 ios::sync_with_stdio( false );

	 int n,m,i,j;
		while(cin>>n>>m)
		{
		 for(i=1;i<=n;i++) cin>>a[i];
		  for(i=1;i<=m;i++) cin>>b[i];

		  int k=1;
		  int ans=0;
		  for(i=1;i<=n;i++)
		  {
			 while(k<m&&abs(b[k]-a[i])>=abs(b[k+1]-a[i])) k++;
			if(abs(b[k]-a[i])>ans) ans=abs(b[k]-a[i]);
		  }
		  cout<<ans<<endl;
		}
	 return 0; 
}
