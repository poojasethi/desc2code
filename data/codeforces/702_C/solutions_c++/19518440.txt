#include<iostream>
#include<climits>
using namespace std;
int main(){
	long long int n,m;
cin>>n>>m;
long long int a[n],b[m],i,j,d=0;
for(i=0;i<n;i++)
	cin>>a[i];
for(i=0;i<m;i++)
	cin>>b[i];
j=0;
for(i=0;i<n;i++){
	while(b[j]<a[i]&&j<=m)
		j++;
	if(j>0&&j<m){
		if(b[j]-a[i]<a[i]-b[j-1]){
			if(b[j]-a[i]>d)
			d=b[j]-a[i];}
		else{	
			if(a[i]-b[j-1]>d)
				d=a[i]-b[j-1];
		}
	}
	else{	if(j>=m){
			if(a[i]-b[m-1]>d)
				d=a[i]-b[m-1];	}
		else if(b[j]-a[i]>d)
		d=b[j]-a[i];
}
}
cout<<d<<endl;
return 0;
}
