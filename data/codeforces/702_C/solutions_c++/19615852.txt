# include <stdio.h>
# include <algorithm>
using namespace std;

int n, m;
int a[666666], b[666666];
int r=0, bn=1;

inline int gabs(int a) {
	return a<0?-a:a;
}

int main() {
	scanf("%d%d", &n, &m);
	for (int i=1; i<=n; ++i) scanf("%d", &a[i]);
	for (int i=1; i<=m; ++i) scanf("%d", &b[i]);
	
	for (int an=1; an<=n; ++an) {
		while(bn < m && gabs(b[bn] - a[an]) >= gabs(b[bn+1] - a[an])) ++bn;
		r=max(r, abs(b[bn]-a[an]));
	}
	printf("%d\n", r);
	
	return 0;
}
