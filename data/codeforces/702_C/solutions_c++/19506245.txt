#include <bits/stdc++.h>
using namespace std;
#define NN 1111111
#define rep(i, a, b) for(i=(a); i<(b); i++)
int a[NN], b[NN];
int main() {
	
	int pt, i, n, m;
	scanf("%d%d", &n, &m);
	rep(i, 0, n) scanf("%d", a+i);
	rep(i, 0, m) scanf("%d", b+i);
	
	int ans, res;
	pt=0; ans=0;
	rep(i, 0, n) {
		while(pt+1<m && b[pt+1]<=a[i]) pt++;
		res=abs(a[i]-b[pt]);
		if(pt+1<m) res=min(res, abs(a[i]-b[pt+1]));
		if(ans<res) ans=res;
	}
	printf("%d\n", ans);
	
	return 0;
}
