#include <iostream>
#include <cmath>
using namespace std;

long long a[100005],b[100005];

int main()
{
	long long ans =-1;
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int j=0;j<m;j++)
		cin>>b[j];
	
	int i=0,j=0;
	
	while(i<n)
	{
		while(a[i]>=b[j] && j<m-1)
			j++;
		long long temp;
		if(j>0)
			temp = min(abs(a[i]-b[j-1]),abs(a[i]-b[j]));
		else
			temp = abs(a[i]-b[j]);
		ans = max(ans,temp);
		i++;
	}
	
	cout<<ans;
	
	return 0;
}
