#include <bits/stdc++.h>

using namespace std;

const int maxn = 100010;

int n, m;
int a[maxn], b[maxn];

bool check(int x) {
	int i, j;
	for(i = 0, j = 0; i < n && j < m; i ++) {
		while(j < m && abs(a[i]-b[j]) > x) j ++;
	}
	return j < m;
}

int main() {
	scanf("%d%d", &n, &m);
	for(int i = 0; i < n; i ++) scanf("%d", &a[i]);
	for(int i = 0; i < m; i ++) scanf("%d", &b[i]);
	
	int l = 0, r = ~0u >> 1, mid;
	while(l < r) {
		mid = (0ll + l + r) >> 1;
		if(check(mid)) r = mid;
		else l = mid + 1;
	}
	printf("%d\n", l);
	return 0;
}
