#include<bits/stdc++.h>
using namespace std;
typedef long int ll;
ll mod(ll x)
{
	if(x<0) return  -1*x;
	else return x;
}
int main()
{
	ll n,m,a[100000+1],b[100000+1],i,j,ans=0;
	cin>>n>>m;
	for(i=1;i<=n;i++)cin>>a[i];
	for(i=1;i<=m;i++)cin>>b[i];
	i=1;
	j=1;
	while(i<=n)
	{
		while((mod(a[i]-b[j])>=mod(a[i]-b[j+1]))&&(j+1<=m))
		{
				j++;
		}
		ans=max(ans,mod(a[i]-b[j]));
		//		cout<<"i is "<<i<<" "<<"j is "<<j<<" "<<ans<<endl;
		i++;
	}
	cout<<ans<<endl;
	return 0;
}
