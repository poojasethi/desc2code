#include<bits/stdc++.h>
using namespace std;
long long a[1000001];long long b[1000001];
int main()
{
    int n,m,i,j;
    cin>>n>>m;
    for(i=0;i<n;i++)
        cin>>a[i];
    for(j=0;j<m;j++)
        cin>>b[j];
    j=0;
    long long ans=0;
    for(i=0;i<n;i++)
    {
        while(j+1<m&&(abs(b[j]-a[i]))>=abs(b[j+1]-a[i]))
            j++;
        if(abs(a[i]-b[j])>ans)
            ans=abs(a[i]-b[j]);
        //cout<<ans<<" "<<b[j]<<" "<<a[i]<<"\n";
    }
    cout<<ans<<"\n";
}
