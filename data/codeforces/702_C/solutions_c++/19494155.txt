#include<cstdio>
#include<algorithm>
using namespace std;

int a[100005];
int b[100005];

int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	for (int i=0;i<n;i++) scanf("%d",&a[i]);
	for (int i=0;i<m;i++) scanf("%d",&b[i]);
	sort(a,a+n);
	sort(b,b+m);
	int ma=0;
	for (int i=0;i<n;i++)
	{
		int p1=lower_bound(b,b+m,a[i])-b;
		int r;
		if (p1==m) r=a[i]-b[m-1];
		else if (p1==0) r=b[p1]-a[i];
		else r=min(b[p1]-a[i],a[i]-b[p1-1]);
		if (r>ma) ma=r;
	}
	printf("%d",ma);
	return 0;
}
