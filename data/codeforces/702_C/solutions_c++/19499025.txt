#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;

int n,m;
long long a[100001],b[100001];

int main(){
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++){
		scanf("%lld",&a[i]);
	}
	for(int i=0;i<m;i++){
		scanf("%lld",&b[i]);
	}
	b[m]=10000000010;
	
	long long ans=0;
	int x=0;
	for(int i=0;i<n;i++){
		while(a[i]>=b[x+1] && x<m-1)
			x++;
		long long t=min(abs(b[x]-a[i]),abs(b[x+1]-a[i]));
		//cout<<t<<endl;
		ans=max(ans,t);
		//cout<<x<<endl;
	}
	//cout<<"..\n";
	printf("%lld\n",ans);
}
