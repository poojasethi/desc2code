#include<iostream>
using namespace std;
int n,m;
int lst;
int ans;
int a[100001],b[100001];
int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++)cin>>a[i];
	for(int i=1;i<=m;i++)cin>>b[i];
	lst=1;
	for(int i=1;i<=n;i++)
	{
		while(lst<=m &&	b[lst]<a[i])lst++;
		if(lst>m)
		{
			ans=max(ans,a[i]-b[m]);
		}
		else if(lst==1)
		{
			ans=max(ans,b[1]-a[i]);
		}
		else
		{
			ans=max(ans,min(b[lst]-a[i],a[i]-b[lst-1]));
		}
	}
	cout<<ans;
	return 0;
}
