#include<cstdio>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
using namespace std;
int a[100010],b[100010];
int main()
{
	int n,m;cin>>n>>m;
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1;i<=m;i++)scanf("%d",&b[i]); 
	int j=1,ans=0;
	for(int i=1;i<=n;i++)
	{
		while(b[j]<a[i]&&j<m)j++;
		int buf;
		if(j==1)buf=abs(b[j]-a[i]);
		else buf=min(abs(b[j]-a[i]),abs(a[i]-b[j-1]));
//		cout<<j<<" "<<buf<<endl;
		ans=max(ans,buf);
	}
	cout<<ans<<endl;
	return 0;
}
