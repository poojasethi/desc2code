#include <iostream>
#include <vector>
#include <cmath>

using namespace std;



int main()
{
 int n,m;
 cin>>n>>m;
 vector <int> a(n),b(m);
 for (int i=0;i<n;i++)
  cin>>a[i];
 for (int i=0;i<m;i++)
  cin>>b[i];
 int l=-1,r=2e9;
 while (r-l>1) {
  int t=(l+r)/2;
  int i=0;
 for (int j=0;j<m;j++)
  {
    while (i<n && abs(b[j]-a[i])<=t) i++;
  }
 if (i==n) r=t; else l=t;

 }
 cout<<r
 ;
    return 0;
}
