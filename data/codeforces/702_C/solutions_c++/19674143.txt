#include<bits/stdc++.h>
using namespace std;
# define l long long int
# define rep(i,a,b) for(l i=a;i<b;i++)
int main(){
	int m,n;
	cin>>n>>m;
	l a[100005],b[100005];
	rep(i,0,n) cin>>a[i];
	rep(i,0,m) cin>>b[i];
	int t=0;
	l dist[100005],maxi=0;
	rep(c,0,n){
		dist[c]=fabs(b[t]-a[c]);
		if(t==m-1) {maxi=max(maxi,dist[c]);
		continue;}
		t++;
		while(fabs(b[t]-a[c])<=dist[c]){
			dist[c]=fabs(b[t]-a[c]);
			t++;
			if(t==m) break;
		}
		t--;
		//cout<<t<<" "<<maxi<<"\n";
		maxi=max(maxi,dist[c]);
	}
	cout<<maxi<<"\n";
	return 0;
}
