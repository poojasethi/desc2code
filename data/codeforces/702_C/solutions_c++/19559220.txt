#include<cstdio>
#include<cstring>
#include<algorithm>
#include<map>
#include<iostream>
using namespace std;
int n,m;
int ans;
int a[100010],b[100010];
int main()
{
	scanf("%d %d",&n,&m);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	ans=0;
	int pos=0;
	for(int i=0;i<m;i++)
	{
		scanf("%d",&b[i]);
	}
	for(int i=0;i<n;i++)
	{
		while(a[i]>=b[pos]&&pos<m-1)
			pos++;
		if(pos==0)
			ans=max(ans,abs(b[pos]-a[i]));
		else
			ans=max(ans,min(abs(a[i]-b[pos-1]),abs(a[i]-b[pos])));
	}
	printf("%d\n",ans);
}
