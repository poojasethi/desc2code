#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n, m, i, r = 0, p = 0, temp;
	scanf("%d%d", &n, &m);
	int city[n], tower[m];
	for (i=0;i<n;i++) scanf("%d", &city[i]);
	for (i=0;i<m;i++) scanf("%d", &tower[i]);
	for (i=0;i<n;i++)
	{
		while (tower[p]<city[i]&&p<m) p++;
		if (p==0) temp = abs(tower[p]-city[i]); else if (p==m) temp = abs(tower[p-1]-city[i]); else
			temp = min(abs(tower[p-1]-city[i]), abs(tower[p]-city[i]));
		r = max(r, temp);
	}
	printf("%d\n", r);
	return 0;
}
