#include <bits/stdc++.h>
using namespace std;

const int maxn = 100100;

int cit[maxn], tow[maxn];
int n, m, ans;

inline int dist(int a, int b) { return abs(cit[a]-tow[b]); }

int main() {
    cin>>n>>m;
    for (int i = 0; i < n; i++) cin>>cit[i];
    for (int j = 0; j < m; j++) cin>>tow[j];
    int cur = 0;
    for (int i = 0; i < n; i++) {
        while (cur < m-1 && dist(i, cur) >= dist(i, cur+1)) cur++;
        ans = max(ans, dist(i, cur));
    }
    cout<<ans;
}
