#include<bits/stdc++.h>
using namespace std;

int main()
{
    int i,n,m,a[100010],b[100010],t=0,t1=0,t2,r=0;
    cin>>n>>m;
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    for(i=0;i<m;i++)
    scanf("%d",&b[i]);
    for(i=0;i<n;i++)
    {
        while(abs(b[t]-a[i])>=abs(b[t+1]-a[i]) && t<m-1)
        {
            t++;
        }
        r=max(r,abs(b[t]-a[i]));
    }
    printf("%d",r);
    return 0;
}
