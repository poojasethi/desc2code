#include <iostream>
#include <cmath>
#include <map>
using namespace std;

int main()
{int n,m;
cin>>n>>m;
long long a[n],b[m],c[n+m];
for(int i=0;i<n;i++)
    {cin>>a[i];
    c[i]=a[i];
    }
for(int i=0;i<m;i++)
    {cin>>b[i];
c[n-1+i]=b[i];
    }
long long r1=0,rmax=0;
long j=0;
for(int i=0;i<n;i++)
{
    while(a[i]>b[j] &&  j!=m)
    {

      j++;
    }

r1=abs(a[i]-b[j]);
if(r1>abs(a[i]-b[j-1])  && j!=0)
    r1=abs(a[i]-b[j-1]);

rmax=max(rmax,r1);
}
cout<<rmax;
    return 0;
}
