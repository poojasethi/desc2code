#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,m,i,j,a[100005],b[100005],ans=0;
	cin>>n>>m;
	for(i=0;i<n;i++)
	{
    	cin>>a[i];
    }
	for(j=0;j<m;j++)
	{
    	cin>>b[j];
    }
    j=0;
	for(i=0;i<n;i++)
	{
		while(j+1<m&&(abs(b[j]-a[i]))>=abs(b[j+1]-a[i]))
        j++;
        if(abs(a[i]-b[j])>ans)
        ans=abs(a[i]-b[j]);
	}
	cout<<ans;
}
