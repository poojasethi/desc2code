#!/usr/bin/python

import sys

lst=map(int, raw_input().split())
nn = lst[0]
mm = lst[1]

n_list = map(int, raw_input().split())
m_list = map(int, raw_input().split())

m_ind = 0

max_dist = -1 
for n in n_list:
	dist = abs(n - m_list[0])
	while m_ind < mm and m_list[m_ind] < n:
		m_ind += 1
	if m_ind == mm:
		dist = min(dist, abs(n - m_list[m_ind - 1]))
	elif m_ind:
		dist = min(dist, abs(n - m_list[m_ind]), abs(n - m_list[m_ind -1]))
	else:
		dist = min(dist, abs(n - m_list[m_ind]))

	if max_dist == -1:
		max_dist = dist
	else:
		max_dist = max(max_dist, dist)

print max_dist
