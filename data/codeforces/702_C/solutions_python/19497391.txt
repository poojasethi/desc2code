from sys import stdin

n, m = map(int, stdin.readline().split(' '))
cities = map(int, stdin.readline().split(' '))
lights = map(int, stdin.readline().split(' '))
max_distance = 0
light_index = 0
for city in cities:
  while light_index < m - 1 and abs(city - lights[light_index]) >= abs(city - lights[light_index + 1]):
    light_index += 1
  max_distance = max(max_distance, abs(city - lights[light_index]))
print max_distance
