#include <cstdio>
#include <cstring>
#include <cstdlib>

using namespace std;

int n;
int cnt=0;

int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);
    for (int i=0;i<12;i++) if ((1<<i)&(n-1)) cnt=i;
    printf("%d\n",cnt+1);
    for (int i=0;i<=cnt;i++)
    {
        int c2=0;
        for (int j=0;j<n;j++) if ((1<<i)&(j)) c2++;
        printf("%d",c2);
        for (int j=0;j<n;j++) if ((1<<i)&(j)) printf(" %d",j+1);
        printf("\n");
    }
    return 0;
}
