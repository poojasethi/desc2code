#include <iostream>
#include <vector>
#include <fstream>
using namespace std;
vector<int> v;
ofstream fout ("output.txt");
ifstream fin ("input.txt");

int tr(int t)
{ return (t==0)?0:tr(t/2)+1; }

int main()
{
    int n;
    fin >>n;
    fout <<tr(n-1)<<endl;
    for(int s=0;(1<<s)<n;s++)
    {
        for(int i=0;i<n;i++)
            if((i>>s)%2) v.push_back(i+1);
        fout <<v.size();
        for(int i=0;i<v.size();i++)
            fout <<" "<<v[i];
        fout <<endl;
        v.clear();
    }
    return 0;
}
