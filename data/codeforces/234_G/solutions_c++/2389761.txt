#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n, m, ans, sum;

int main()
{
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	scanf("%d", &n), n --;
	for (ans = 0, m = 1; m <= n; ans ++, m <<= 1);
	printf("%d\n", ans), m >>= 1;
	for (sum = 0; ans --; sum = 0, m >>= 1)
	{
		for (int i = 0; i <= n; i ++)
			if ((i & m) != 0)  sum ++;
		printf("%d", sum);
		for (int i = 0; i <= n; i ++)
			if ((i & m) != 0)  printf(" %d", i + 1);
		printf("\n");
	}
	return 0;
}
