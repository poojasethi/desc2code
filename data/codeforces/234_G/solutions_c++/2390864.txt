#include <cstdio>

int main(void)
{
  freopen("input.txt","r",stdin);
  freopen("output.txt","w",stdout);
  int n;
  scanf("%d",&n);
  int m=n,ans=0;
  while(m>1)
  {
    ++ans;
    m=(m+1)/2;
  }
  printf("%d\n",ans);
  for(int i=0;i<ans;++i)
  {
    int s=0;
    for(int j=1;j<=n;++j)
      if(((1<<i)&j)>0)
        ++s;
    printf("%d",s);
    for(int j=1;j<=n;++j)
      if(((1<<i)&j)>0)
        printf(" %d",j);
    printf("\n");
  }
  return 0;
}
