#include<cstdio>
#include<cstring>
#define N 1005
int ans,n,L[N],a[N][N];
void deal(int l,int r,int dep)
{
	if (l==r) return; int mid=(l+r)/2; if (dep>ans) ans=dep;
	deal(l,mid,dep+1); deal(mid+1,r,dep+1);
	for (int i=l;i<=mid;i++) a[dep][L[dep]++]=i;
}
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d",&n); deal(1,n,1);
	printf("%d\n",ans);
	for (int i=1;i<=ans;i++){
		printf("%d",L[i]);
		for (int j=0;j<L[i];j++) printf(" %d",a[i][j]);
		printf("\n");		
		}
	return 0;
}
