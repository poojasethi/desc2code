#include <cstdio>
#include <iostream>
#include <vector>
#define pb push_back
using namespace std;
int n,us;
vector <int> v;
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	cin>>n;
	while((1<<us)<n)
			us++;
	cout<<us<<endl;
	for(int i=0;i<us;i++)
	{
		v.clear();
		for(int j=0;j<(1<<us);j++)
		{
			if((1<<i) & j && j<n)
				v.pb(j+1);
		}
		cout<<v.size();
		for(int j=0;j<(int)v.size();j++)
			cout<<" "<<v[j];
		cout<<endl;	
	}
}
