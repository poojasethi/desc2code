
#include <stdio.h>

int q[1010];
int main()
{
    int n,m,i,j,k;
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    scanf("%d", &n);
    for (i=0,m=1; ;i++){
        if (m>=n) break;
        m*=2;
    }
    printf("%d\n", i);
    for (j=0; j<i; j++)
    {
        int g=0;
        for (k=0; k<n; k++) if (k&(1<<j)) q[++g]=k;
        printf("%d ", g);
        for (k=1; k<g; k++) printf("%d ", q[k]+1);
        printf("%d\n", q[g]+1);
    }

    return 0;
}
