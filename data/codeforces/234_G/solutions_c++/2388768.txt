#include <iostream>
#include <fstream>
using namespace std;

int main()
{
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    int n; cin >> n;
    int dos = 0;
    while ((1 << dos) < n) dos++;
    cout << dos << endl;

    for (int bit = 1; bit < n; bit <<= 1)
    {
        int out = 0;
        for (int i=0; i<n; i++)
            if (i & bit) out++;
        cout << out << ' ';
        for (int i=0; i<n; i++)
            if (i & bit) cout << i << ' ';
        cout << endl;
    }
    return 0;
}
