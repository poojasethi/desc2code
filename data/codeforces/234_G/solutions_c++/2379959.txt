#include<iostream>
#include<cstdio>
using namespace std;
int res[1024];
int main()
{
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
  int n;
  cin>>n;
  int bits;
  for(bits=0;n > (1<<bits); bits++);
  cout<<bits<<endl;
  for (int i = 0; i < bits; i++) {
    int e = 0;
    for (int j = 1; j <= n; j++) if(j&(1<<i)) res[e++] = j;
    cout<<e<<" ";
    for (int j = 0; j < e; j++) cout<<res[j]<<" ";cout<<endl;
  }
  return 0;
}
