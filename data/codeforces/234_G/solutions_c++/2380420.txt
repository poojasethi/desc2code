#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using namespace std;

int main(){
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int n,m;
    scanf("%d",&n);
    m = 0;
    while((1 << m) < n) m++;
    printf("%d\n",m);
    for(int i = 0;i < m;i++){
        int cont = 0;
        for(int j = 0;j < n;j++) if((j&(1 << i)) != 0) cont++;
        printf("%d",cont);
        for(int j = 0;j < n;j++) if((j&(1 << i)) != 0) printf(" %d",j + 1);
        printf("\n");
    }
    return 0;
}
