#include<stdio.h>
long n,nr,i,j,nr2,sssssss;
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%ld",&n);
    nr=0;
    while ((1<<nr)<n)
        nr++;
    printf("%ld\n",nr);
    for (i=0;i<nr;i++)
    {
        nr2=0;
        for (j=0;j<n;j++)
            if (j&(1<<i))
                nr2++;

        for (j=0;j<n;j++)

                printf(" %ld",j+1);
        printf("\n");
    }

}
