#include <iostream>
#include <cstdio>

using namespace std;

int a[1011];

int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);

    int n;

    scanf("%d", &n);
    int m = 1;
    while((1<<m)<n) m += 1;
    printf("%d\n", m);
    for(int i=0; i<m; i++)
    {
        int c = 0;
        for(int j=1; j<=n; j++)
            if(j & (1<<i))
                a[c++] = j;
        printf("%d", c);
        for(int j=0; j<c; j++)
            printf(" %d", a[j]);
        printf("\n");
    }

    return 0;
}
