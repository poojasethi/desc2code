/*For Yukari*/
#include <cstdio>
using namespace std;

int main()
{
    int N,K,L,S;
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&N);
    K=N;L=0;
    while (K>1)
     {
          K=(K+1)>>1;++L;
     }
    printf("%d\n",L);
    K=1;
    for (int i=0;i<L;++i)
     {
             S=0;
             for (int i=1;i<=N;++i) if (i&K) ++S;
             printf("%d",S);
             for (int i=1;i<=N;++i) if (i&K) printf(" %d",i);
             puts("");
             K<<=1;
     }
    return 0;
}
