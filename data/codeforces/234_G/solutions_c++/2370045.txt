#include<cstdio>
#define fo(i,a,b) for(i=a;i<=b;++i)
int n,i,j,l;
void solve(int l,int n)
{
    if(l==1)return;
    int i,j,s=0;
    fo(i,0,n-1)
        if(i%l<l/2)++s;
    printf("%d",s);
    fo(i,0,n-1)
        if(i%l<l/2)printf(" %d",i+1);
    printf("\n");
    solve(l/2,n);
}
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);
    for(l=0;1<<l<n;++l);
    printf("%d\n",l);
    solve(1<<l,n);
    return 0;
}
