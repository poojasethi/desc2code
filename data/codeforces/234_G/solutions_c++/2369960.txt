#include <cstdio>
int pls[1024],plc;
int main()
{
	FILE *fi,*fo;
	fi=fopen("input.txt","r");
	fo=fopen("output.txt","w");
	int n,cs=0;
	fscanf(fi,"%d",&n);
	while((1<<cs)<n)
		cs++;
	fprintf(fo,"%d\n",cs);
	for(int xhc=0;xhc<cs;xhc++)
	{
		plc=0;
		for(int xhn=0;xhn<n;xhn++)
			if(xhn&(1<<xhc))
				pls[plc++]=xhn+1;
		fprintf(fo,"%d",plc);
		for(int xhp=0;xhp<plc;xhp++)
			fprintf(fo," %d",pls[xhp]);
		fprintf(fo,"\n");
	}
	return 0;
}
