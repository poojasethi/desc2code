#include<cstdio>
int main()
{
        freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,i,j,ans,cnt;
	while(~scanf("%d",&n))
	{
		for(ans=1;1<<ans<n;ans++);
		printf("%d\n",ans);
		for(i=0;i<ans;i++)
		{
			cnt=0;
			for(j=0;j<n;j++)if(1<<i&j)cnt++;
			printf("%d",cnt);
			for(j=0;j<n;j++)if(1<<i&j)printf(" %d",j+1);
			puts("");
		}
	}
	return 0;
}
