#include <cstdio>
#include <cmath>

int main(){

    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);

    int n,m,fi;
    int i,j;
    
    scanf("%d",&n);
    m = ceil(log2(n));
    
    printf("%d\n",m);
    
    for(i = 0;i < m;i++){
        fi = 0;
        for(j = 0;j < n;j++) if(j & (1 << i)) fi++;
        
        printf("%d",fi);
        for(j = 0;j < n;j++) if(j & (1 << i)) printf(" %d",j+1);
        printf("\n");
    }
    
    return 0;
}
