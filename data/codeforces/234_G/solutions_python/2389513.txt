#!/usr/bin/python

def practice(a):
	l = len(a)
	assert(l >= 1)
	if l <= 2:
		return [[a[0]]]
	else:
		h = l // 2
		a1 = a[:h]
		a2 = a[h:]
		result = []
		result.append(a1[:])
		g1 = practice(a1)
		g2 = practice(a2)
		assert(len(g1) == len(g2) or len(g1) == len(g2) - 1)
		for i in range(len(g1)):
			result.append(g1[i] + g2[i])
		if len(g1) == len(g2)-1:
			result.append(g1[0] + g2[-1])
		return result


f = open("input.txt")
n = int(f.readline())
a = [int(i) for i in range(1, n+1)]
p = practice(a)
o = open("output.txt", "w")
o.write(str(len(p)) + '\n')
for i in p:
	o.write(str(len(i)) + ' ')
	for j in i:
		o.write(str(j) + ' ')
	o.write('\n')
