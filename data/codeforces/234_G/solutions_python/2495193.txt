import sys
from itertools import izip_longest
sys.stdin = open("input.txt"); sys.stdout = open("output.txt", "w")

N, = map(int, sys.stdin.readline().strip().split())

def shift(it, k):
  return [tuple(a+k for a in t) for t in it]

def solve(n):
  if n < 2:
    return
  if n == 2:
    yield (0,)
    return
  
  a, b = (n+1)//2, n//2
  yield tuple(range(a))
  for solve1, solve2 in izip_longest(solve(a), shift(solve(b),a)):
    if solve2 is None: solve2 = ()
    yield solve1+solve2

res = list(solve(N))
print len(res)
for team in res:
  print len(team),
  print " ".join(str(a+1) for a in team)
