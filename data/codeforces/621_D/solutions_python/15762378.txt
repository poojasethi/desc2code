from math import log
from decimal import *
x, y, z = map(Decimal, raw_input().split())
ln = lambda x : Decimal(log(x))
ans = sorted([(-ln(x) * y ** z, "x^y^z"), 
	(-ln(x) * z ** y, "x^z^y"), 
	(-ln(x) * y * z, "(x^y)^z"), 
	(-ln(y) * x ** z, "y^x^z"), 
	(-ln(y) * z ** x, "y^z^x"), 
	(-ln(y) * x *z, "(y^x)^z"), 
	(-ln(z) * x ** y, "z^x^y"), 
	(-ln(z) * y ** x, "z^y^x"), 
	(-ln(z) * x * y, "(z^x)^y")
	], key = lambda lst : lst[0])
print ans[0][1]
