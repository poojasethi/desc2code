#include<iostream>
#include<algorithm>
#include<math.h>
#include<string>

using namespace std;

#define INF 1LL<<60;

int main(void){
    long double x,y,z,logx,logy,logz,a[9];
    int i,n=0;
    string ans[9]={"x^y^z","x^z^y","(x^y)^z","y^x^z","y^z^x","(y^x)^z","z^x^y","z^y^x","(z^x)^y"};
    cin >> x >> y >> z;
    logx = logl(x);
    logy = logl(y);
    logz = logl(z);
    a[0] = logx*powl(y,z);
    a[1] = logx*powl(z,y);
    a[2] = logx*y*z;
    a[3] = logy*powl(x,z);
    a[4] = logy*powl(z,x);
    a[5] = logy*x*z;
    a[6] = logz*powl(x,y);
    a[7] = logz*powl(y,x);
    a[8] = logz*x*y;
    for(i=1; i<9; i++){
        if(a[n]<a[i]){
            n = i;
        }
    }
    cout << ans[n] << endl;
    
    return 0;
}
