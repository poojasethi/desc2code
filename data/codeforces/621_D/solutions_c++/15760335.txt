#include<bits/stdc++.h>
using namespace std;
long double x,y,z;
long double ans[13];
long double eps=1e-14;
int main()
{
	cin>>x>>y>>z;
	ans[1]=pow(y,z)*log(x);
	ans[2]=pow(z,y)*log(x);
	ans[3]=y*z*log(x);
	ans[4]=ans[3];
	
	ans[5]=pow(x,z)*log(y);
	ans[6]=pow(z,x)*log(y);
	ans[7]=x*z*log(y);
	ans[8]=ans[7];
	
	ans[9]=pow(x,y)*log(z);
	ans[10]=pow(y,x)*log(z);
	ans[11]=y*x*log(z);
	ans[12]=ans[11];
	
	int index=1;
	for (int i=2;i<=12;i++)
	{
		if (ans[i]-ans[index]>eps)
		{
			index=i;
		}
	}
	
	switch(index)
	{
		case 1:puts("x^y^z");break;
		case 2:puts("x^z^y");break;
		case 3:puts("(x^y)^z");break;
		
		case 5:puts("y^x^z");break;
		case 6:puts("y^z^x");break;
		case 7:puts("(y^x)^z");break;
		
		case 9:puts("z^x^y");break;
		case 10:puts("z^y^x");break;
		case 11:puts("(z^x)^y");break;
	}
}
