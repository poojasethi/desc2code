#include <bits/stdc++.h>
using namespace std;
typedef long double ld;
const double EPS = 1e-18;
ld x,y,z;
string s[13] = {"",
    "x^y^z", "x^z^y", "(x^y)^z", "(x^z)^y",
	"y^x^z", "y^z^x", "(y^x)^z", "(y^z)^x",
	"z^x^y", "z^y^x", "(z^x)^y", "(z^y)^x"
};

ld a[13];
int main() {
    cin >> x >> y >> z;
    a[1] = pow(y,z) * log(x);
    a[2] = pow(z,y) * log(x);
    a[3] = y*z*log(x);
    a[4] = y*z*log(x);
    a[5] = pow(x,z) * log(y);
    a[6] = pow(z,x) * log(y);
    a[7] = x*z*log(y);
    a[8] = x*z*log(y);
    a[9] = pow(x,y) * log(z);
    a[10] = pow(y,x) * log(z);
    a[11] = x*y * log(z);
    a[12] = x*y * log(z);
    
    int idx=1;
    for(int i=1;i<=12;i++) {
        //printf("%lf\n", a[i]);
        if(a[i] - a[idx] > EPS) idx = i; 
    }
    cout << s[idx] << endl;
    
}
