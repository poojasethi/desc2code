#include <iostream>
#include <cmath>

using namespace std;

string S[] = {"x^y^z", "x^z^y", "(x^y)^z", "(x^z)^y", "y^x^z", "y^z^x", "(y^x)^z", "(y^z)^x", "z^x^y", "z^y^x", "(z^x)^y", "(z^y)^x"};
int ans;
#define ld long double
ld Max;

void cmp1(ld x, ld y, ld z, int l)
{
	ld tmp = logl(x) * pow(y, z);
	if (tmp > Max)
	{
		Max = tmp;
		ans = l;
	}
}

void cmp2(ld x, ld y, ld z, int l)
{
	ld tmp = logl(x) * y * z;
	if (tmp > Max)
	{
		Max = tmp;
		ans = l;
	}
}

int main()
{
	ld x, y, z;
	cin >> x >> y >> z;
	
	Max = logl(x) * pow(y, z);
	cmp1(x, z, y, 1);
	
	cmp2(x, y, z, 2);
	//cmp2(x, z, y, 3);	
		
	cmp1(y, x, z, 4);
	cmp1(y, z, x, 5);
		
	cmp2(y, x, z, 6);
	//cmp2(y, z, x, 7);
	
	cmp1(z, x, y, 8);
	cmp1(z, y, x, 9);
	
	cmp2(z, x, y, 10);
	//cmp2(z, y, x, 11);
		
	cout << S[ans] << endl;
	
	return 0;
}
