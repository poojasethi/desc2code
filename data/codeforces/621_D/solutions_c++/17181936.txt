#include<bits/stdc++.h>
using namespace std;
#define INF 1e-10
long double x,y,z;
string s[12]={"x^y^z","x^z^y","(x^y)^z","(x^z)^y" , "y^x^z","y^z^x","(y^x)^z","(y^z)^x" , "z^x^y","z^y^x","(z^x)^y","(z^y)^x"};
long double a[12];
int main()
{
	while(cin>>x>>y>>z)
	{
		a[0]=pow(y,z)*log(x);
		a[1]=pow(z,y)*log(x);
		a[2]=y*z*log(x);
		a[3]=y*z*log(x);
		a[4]=log(y)*pow(x,z);
		a[5]=pow(z,x)*log(y);
		a[6]=x*z*log(y);
		a[7]=x*z*log(y);
		a[8]=pow(x,y)*log(z);
		a[9]=pow(y,x)*log(z);
		a[10]=y*x*log(z);
		a[11]=y*x*log(z);
		int l=0;
		for(int i=0;i<12;++i)
		{
			if(a[i]>a[l])
			{
				l=i;
			}
		}
		cout<<s[l]<<endl;
	}
	return 0;
}
