#include <cstdio>
#include <cmath>
#include <iostream>
#include <cstring>
using namespace std;
long double a[13];
string s[13];
long double x,y,z;
int main()
{
	cin>>x>>y>>z;
	a[1]=log(x)*pow(y,z);
	a[2]=log(x)*pow(z,y);
	a[3]=y*z*log(x);
	a[4]=z*y*log(x);
	a[5]=pow(x,z)*log(y);
	a[6]=pow(z,x)*log(y);
	a[7]=x*z*log(y);
	a[8]=z*x*log(y);
	a[9]=pow(x,y)*log(z);
	a[10]=pow(y,x)*log(z);
	a[11]=x*y*log(z);
	a[12]=y*x*log(z);
	s[1]="x^y^z";
	s[2]="x^z^y";
	s[3]="(x^y)^z";
	s[4]="(x^z)^y";
	s[5]="y^x^z";
	s[6]="y^z^x";
	s[7]="(y^x)^z";
	s[8]="(y^z)^x";
	s[9]="z^x^y";
	s[10]="z^y^x";
	s[11]="(z^x)^y";
	s[12]="(z^y)^x";
	int ans=1;
	for(int i=1;i <= 12;i++)
		if(a[ans]<a[i]) ans=i;
	cout<<s[ans]<<endl;
 }
