#include <cstdio>
#include <cmath>
#include <iostream>
using namespace std;
char str[10][10] = {
    "x^y^z", "x^z^y", "(x^y)^z",
    "y^x^z", "y^z^x", "(y^x)^z",
    "z^x^y", "z^y^x", "(z^x)^y"
};
int main()
{
    long double x, y, z;
    cin >> x >> y >> z;
    long double lgx = log10(x);
    long double lgy = log10(y);
    long double lgz = log10(z);
    long double res[10];
    res[0] = pow(y, z) * lgx;
    res[1] = pow(z, y) * lgx;
    res[2] = y * z * lgx;
    res[3] = pow(x, z) * lgy;
    res[4] = pow(z, x) * lgy;
    res[5] = x * z * lgy;
    res[6] = pow(x, y) * lgz;
    res[7] = pow(y, x) * lgz;
    res[8] = x * y * lgz;
    /*
    for(int i = 0; i < 9; i++)
        printf("res[%d] = %f\n", i, res[i]);
        */
    int ans = 0;
    for(int i = 1; i < 9; i++)
        ans = (res[i] > res[ans])? i: ans;
    printf("%s\n", str[ans]);
    return 0;
}
