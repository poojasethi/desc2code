#include<cstdio>
#include<algorithm>
#include<cstring>
#include<iostream>
#include<cmath>
#include<string>
using namespace std;

string ans[12];
long double l[12];

int main()
{
	ans[0]="x^y^z";
	ans[1]="x^z^y";
	ans[2]=ans[3]="(x^y)^z";
	ans[4]="y^x^z";
	ans[5]="y^z^x";
	ans[6]=ans[7]="(y^x)^z";
	ans[8]="z^x^y";
	ans[9]="z^y^x";
	ans[10]=ans[11]="(z^x)^y";
	
	long double x,y,z;
	cin>>x>>y>>z;
	
	l[0]=log(x)*pow(y,z);
	l[1]=log(x)*pow(z,y);
	l[2]=l[3]=log(x)*(y*z);
	
	l[4]=log(y)*pow(x,z);
	l[5]=log(y)*pow(z,x);
	l[6]=l[7]=log(y)*(x*z);
	
	l[8]=log(z)*pow(x,y);
	l[9]=log(z)*pow(y,x);
	l[10]=l[11]=log(z)*(x*y);
	
	int mi=0;
	for(int i=1;i<12;i++)
		if (l[i]>l[mi]) mi=i;
	
	cout<<ans[mi]<<endl;
	return 0;
}
