#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <cmath>
#include <string.h>
using namespace std;
const string s[12]={"x^y^z","x^z^y","(x^y)^z","(x^z)^y","y^x^z","y^z^x","(y^x)^z",
 "(y^z)^x","z^x^y","z^y^x","(z^x)^y","(z^y)^x"};
long double x,y,z,a[12],mx;
int i,k;

int main(){
	cin>>x>>y>>z;
	a[0]=pow(y,z)*log(x);
	a[1]=pow(z,y)*log(x);
	a[2]=a[3]=z*y*log(x);
	a[4]=pow(x,z)*log(y);
	a[5]=pow(z,x)*log(y);
	a[6]=a[7]=x*z*log(y);
	a[8]=pow(x,y)*log(z);
	a[9]=pow(y,x)*log(z);
	a[10]=a[11]=x*y*log(z);
	mx=a[0];k=0;
	for(i=1;i<12;i++)if(a[i]>mx){
		mx=a[i];
		k=i;
	}
	cout<<s[k]<<endl;
}
