#include <iostream>
#include <vector>
#include <cmath>
#include <string>
using namespace std;



int main() {
    long double x, y, z, a[13] = {0}, maxa = -1;
    cin >> x >> y >> z;
    a[1] = pow(y, z) * log(x);
    a[2] = pow(z, y) * log(x);
    a[3] = z * y * log(x);
    a[4] = y * z * log(x);
    a[5] = pow(x, z) * log(y);
    a[6] = pow(z, x) * log(y);
    a[7] = z * x * log(y);
    a[8] = x * z * log(y);
    a[9] = pow(x, y) * log(z);
    a[10] = pow(y, x) * log(z);
    a[11] = y * x * log(z);
    a[12] = x * y * log(z);
    int maxi = 0;
    for(int i = 1; i <= 12; i++) {
        if(a[i] > maxa) {
            maxi = i;
            maxa = a[i];
        }
    }
    string s[13] = {"", "x^y^z", "x^z^y", "(x^y)^z", "(x^z)^y", "y^x^z", "y^z^x", "(y^x)^z", "(y^z)^x", "z^x^y", "z^y^x", "(z^x)^y", "(z^y)^x"};
    cout << s[maxi];
    return 0;

}
