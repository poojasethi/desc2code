#include "iostream"
using namespace std;
int main(int argc, char const *argv[])
{
	string msg;
	string s = "qwertyuiopasdfghjkl;zxcvbnm,./";
	char direction;
	int i=0;
	cin>>direction>>msg;
	while(msg[i]!='\0')
	{
		cout<<s[s.find(msg[i]) - (direction=='R') + (direction=='L')];
		i++;
	}
	return 0;
}
