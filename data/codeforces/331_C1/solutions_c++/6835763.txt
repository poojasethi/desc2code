#include <bits/stdc++.h>
using namespace std;

const int N=1e6+5;

int f[N];


int F(int i){
	if(i==0) return 0;
	if(f[i]!=0) return f[i];
	f[i]=1e9;
	int a=i;
	while(a){
		int t=F(i-a%10)+1;
		if(t<f[i]) f[i]=t;
		a/=10;
		
	}
	return f[i];
}


int main(){
	int n;
	scanf("%d", &n);
	printf("%d", F(n));
}
