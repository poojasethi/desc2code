#include<iostream>
using namespace std;
int main()
{
    int n;cin>>n;
    int ans=0;
    while (n>0)
    {
        ans++;
        int max=0;
        for(int k=n;k>0;k/=10)
            if (max<k%10)
                max=k%10;
        n-=max;
        //cout<<n<<"->";
    }
    cout<<ans<<endl;
    return 0;
}
