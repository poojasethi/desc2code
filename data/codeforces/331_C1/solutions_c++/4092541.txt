#include <iostream>

using namespace std;

int maxd(int n)
{
    int max=0;
    while (n>0)
    {
        if (n%10>max)
            max=n%10;
        n/=10;
    }
    return max;
}
int main()
{
    int n;
    cin>>n;
    int i=0;
    while (n>0)
    {
        i++;
        n-=maxd(n);
    }
    cout<<i;
}
