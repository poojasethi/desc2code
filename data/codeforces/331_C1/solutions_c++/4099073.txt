#include <iostream>
using namespace std;

int main()
{
    int n, ans=0;
    cin >> n;
    while(n)
    {
        int now=n, MX=0;
        while(now)
        {
            MX = max(MX, now%10);
            now/=10;
        }
        n-=MX;
        ans++;
    }
    cout << ans << "\n";
    cin >> n;
}
