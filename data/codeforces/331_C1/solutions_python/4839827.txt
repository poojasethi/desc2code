#include <cstdio>
#define max(a,b) ((a)>(b))?(a):(b)
int r[1000001];
int main(int argc, char const *argv[])
{
	for (int i = 1; i <= 1000000; ++i)
	{
		int d = 0;
		for (int j = i; j > 0; j /= 10) {
			d = max(d, j % 10);
		}
		r[i] = r[i - d] + 1;
	}
	int n;
	scanf("%d", &n);
	printf("%d\n", r[n]);
	return 0;
}
