#include<iostream>
using namespace std;
main(){
int n;
cin>>n;
int dp[n+1];
dp[0]=0;
int min,x;
for(int i=1;i<=n;i++){
x=i;
min=1000000000;
while(x>0){
if(x%10==0){x=x/10;continue;}
if(dp[i-x%10]<min)min=dp[i-x%10];
x=x/10;           
}
dp[i]=min+1;        
}       
cout<<dp[n]<<"\n";
//system("pause");
}
