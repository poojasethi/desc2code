#include<cstdio>

int func(long long n)
{
    int s=0;

    while(n)
    {
        s=n%10>s?n%10:s;
        n/=10;
    }

    return s;
}

int main()
{
    int s=0;
    long long n;

    scanf("%I64d",&n);
    while(n)
    {
        n-=func(n);
        s++;
    }

    printf("%d\n",s);

    return 0;
}
