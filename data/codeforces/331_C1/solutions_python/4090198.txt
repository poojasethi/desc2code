#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n, m;
    int ans = 0;

    cin >> n;

    while ( n )
    {
        int mx = 0;
        m = n;
        while ( m )
        {
            mx = max( m%10, mx);
            m /= 10;
        }
        n -= mx;
        ans ++;
    }

    cout << ans << endl;
}
