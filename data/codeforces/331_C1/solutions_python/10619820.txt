#include<stdio.h>
int n,k,max,t,z,s;
int main(){

	scanf("%d",&n);

	
	while(n){
		k=1;
		t=n;
		max=0;
		if(t < 10){
			s++;
			break;
		}
		while(k <= t){
			t/=k;
			if(t%10 > max)
				max=t%10;
			k=10;
		}
//		getchar();
		n-=max;
		s++;
//		printf("%d %d %d\n",max,n,s);
	}
	printf("%d",s);
	return 0;
}
