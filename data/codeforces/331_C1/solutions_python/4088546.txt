#include<iostream>
using namespace std; 
int dp[1000001];
int main(){
    int n;
    cin >>n;
for (int i = 1 ; i<=n ;i++){
    int ans = 1000000000;
    int x = i;
    while (x){
        if (x % 10 )
            ans = min(ans,1+dp[i-(x%10)]);
        x/=10;
    }
    dp[i] = ans;
}
cout << dp[n] << endl;
}
