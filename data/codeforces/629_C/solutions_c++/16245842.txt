#include <bits/stdc++.h>
using namespace std;     

#define MOD 1000000007

long long int DP[2005][2005],n,m,len,Min=0,sz=0,Ans=0;
char s[100005];    

int main() {
	cin >> n >> m;
	cin >> s+1;
	for(long long int i=1;i<=m;++i) {
		if(s[i]=='(') ++sz;
		else --sz;
		Min=min(sz,Min);
	}
	len=(n-m);
	DP[0][0]=1;
	for(long long int i=1;i<=len;++i)
		for(long long int j=0;j<=i;++j)
			DP[i][j]=(DP[i-1][j-1]+DP[i-1][j+1])%MOD;
	for(long long int i=0;i<=len;++i)
		for(long long int j=-Min;j<=i;++j) {
			if(j+sz<=len) {
				Ans+=(DP[i][j]*DP[len-i][j+sz])%MOD;
				Ans%=MOD;
			}
	}
	cout << Ans << endl;
	return 0;
}
