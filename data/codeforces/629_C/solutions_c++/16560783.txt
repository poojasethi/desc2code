#include <cstdio>
#define mod 1000000007ll
int n, m, l, r, len, x;
long long ans, dp[2005][2005];
char s[100005];

int main() {
	scanf(" %d %d %s", &n, &m, s);
	len = n-m;
	dp[0][0] = 1;
	for (int i = 1; i <= len; i++) {
		dp[i][0] = dp[i-1][1];
		for (int j = 1; j <= i; j++) {
			dp[i][j] = (dp[i-1][j-1] + dp[i-1][j+1]) % mod;
		}
	}
	for (int i = 0; i < m; i++) {
		if (s[i] == '(') l++;
		else if (l) l--;
		else r++;
	}
	for (int i = r; i <= len; i++) {
		for (int j = r; j <= i; j++) {
			x = j-r+l;
			if (x >= 0 && x <= len-i) {
				ans = (ans + dp[i][j] * dp[len-i][x]) % mod;
			}
		}
	}
	printf("%I64d", ans);
	return 0;
}
