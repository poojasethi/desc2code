#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
#define S 2005
#define N 100005
#define P 1000000007
using namespace std;
int sum,mi,n,m,dp[S][S];
char str[N];
void Mod(int &a,int b){
	a+=b;
	if(a>=P)a-=P;
}
int main(){
	scanf("%d%d",&n,&m);
	scanf("%s",str);
	for(int i=0;i<strlen(str);i++){
		sum+=(str[i]=='(')?1:-1;
		mi=min(mi,sum);
	}
	dp[0][0]=1;
	for(int i=1;i<=2000;i++){
		for(int j=0;j<=2000;j++){
			if(j+1<=2000)Mod(dp[i][j],dp[i-1][j+1]);
			if(j-1>=0)Mod(dp[i][j],dp[i-1][j-1]);
		}
	}
	mi=max(0,-mi);
	int ans=0;
	for(int i=0;i<=n-m;i++){
		for(int j=mi;j<=i;j++){
			if(n-m-i<j+sum)break;
			Mod(ans,1LL*dp[i][j]*dp[n-m-i][j+sum]%P);
		}
	}
	printf("%d\n",ans);
	return 0;
}
