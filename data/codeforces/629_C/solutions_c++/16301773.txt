#include <iostream>
#include <cstdio>
using namespace std;
long long a[100010],f[2010][2010],i,j,n,m,x,y,t,mina,bo,ans=0,mo=1000000007;
char ch;
int main()
{
		scanf("%I64d%I64d\n",&n,&m);
		x=y=0;bo=1;mina=0;
		for (i=1;i<=m;++i)
		{
			scanf("%c",&ch);
			if (ch=='(') a[i]=1,++x;else a[i]=-1,++y;
			a[i]+=a[i-1];
			mina=min(mina,a[i]);
		}
		if ((n&1) || x>n/2 || y>n/2) bo=0;
		if (bo)
		{
			t=n-m;
			f[0][0]=1;
			for (i=1;i<=t;++i)
				for (j=0;j<=i;++j)
				{
					if (j) f[i][j]+=f[i-1][j-1];
					if (j<i) f[i][j]+=f[i-1][j+1];
					f[i][j]%=mo;
				}
			for (i=0;i<=t;++i)
				for (j=max((long long)0,-mina);j<=i;++j)
				ans=(ans+f[i][j]*f[t-i][j+x-y])%mo;
		}
		printf("%I64d\n",ans);
}
