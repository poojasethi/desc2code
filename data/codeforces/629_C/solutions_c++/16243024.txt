#include <bits/stdc++.h>

const int N = 100000;
const int DELTA = 2000;
const int MOD = 1e9+7;

int n, m;
char s[N+9];
int sum;
int f[DELTA+9][DELTA+9];
int ans;
int min = N;

int main() {
    scanf("%d%d%s", &n, &m, s);
    for(int i = 0; i < m; ++i) {
        sum += s[i]=='('?1:-1;
        min = std::min(sum, min);
    }
    f[0][0] = 1;
    for(int i = 1; i <= n-m; ++i)
        for(int j = 0; j <= i; ++j) {
            f[i][j] += f[i-1][j+1];
            if(j) f[i][j] += f[i-1][j-1];
            f[i][j] %= MOD;
        }
    for(int i = 0; i <= n-m; ++i) {
        int pre = i, suf = n-m-i;
        for(int j = 0; j <= i; ++j)
            if(j+min >= 0 && j+sum <= suf)
                (ans += 1ll*f[pre][j]*f[suf][j+sum]%MOD) %= MOD;
    }
    std::cout << ans << std::endl;
    return 0;
}
