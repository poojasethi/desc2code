#include <stdio.h>
#include <string.h>
#include <vector>
#include <queue>
#include <stdlib.h>
#include <algorithm>
#define SQ(a) ((a)*(a))
#define pb push_back
#define mod 1000000007
typedef long long ll;
typedef double dd;
typedef std::pair<int,int> pii;
using namespace std;

ll n,m,p,q,a,ans,dp[2100][2100],x;
char s[110000];
int main(){
	scanf("%I64d%I64d",&n,&m);
	scanf("%s",s);

	dp[0][0]=1;
	dp[1][1]=1;
	for(int i=2;i<=2000;i++){
		for(int j=0;j<=i;j++){
			dp[i][j]=(dp[i-1][j+1]+(j>0?dp[i-1][j-1]:0))%mod;
		}
	}

	a=0;
	for(int i=0;i<m;i++){
		if(s[i]=='('){
			a++;
		}
		else{
			if(a>0)a--;
			else p++;
		}
	}
	q=a;
	x=n-m;
	for(int i=0;i<=x;i++){
		for(int j=0;j<=x;j++){
			if(p+j<=2010&&q+j<=2010)ans=(ans+dp[i][p+j]*dp[x-i][q+j])%mod;
		}
		
	}
	printf("%I64d",ans);
}
