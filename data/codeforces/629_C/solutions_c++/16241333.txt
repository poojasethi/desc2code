#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;
typedef long long LL;

int P[5000][5000];

char s[100001];

int main() {
	int n, m;
	cin >> n >> m;
	scanf("%s", &s);
	
	int d = 0, mn = 696969;
	for(int i = 0; i < m; i++) {
		d += (s[i] == '(' ? 1 : -1);
		mn = min(mn, d);
	} 
	mn = max(-mn, 0);
	
	if(d > 2000 || mn > 2000) {
		cout << 0;
		return 0;
	}
	
	P[0][0] = 1;
	for(int i = 1; i <= n-m; i++)
		for(int j = 0; j <= n-m; j++) {
			if(j == 0)
				P[i][j] = P[i-1][j+1];
			
			else if(j == n-m)
				P[i][j] = P[i-1][j-1];
		
			else
				P[i][j] = (P[i-1][j-1] + P[i-1][j+1]) % mod;
		} 
	
	LL w = 0;
	for(int i = 0; i <= n-m; i++) {
		for(int j = mn; j <= i; j++) {
			w = (w + LL(P[i][j]) * P[n-m-i][j+d]) % mod;
		}
	}
	cout << w;
}
