#include <bits/stdc++.h>

using namespace std;

const int maxLen = (int)1E5+5;
const int maxL = (int)2E3+3;
const long long modulo = (int)1E9+7;

char s[maxLen];
long long f[maxL][maxL];
int n, m;
int a, b;

int main() {
	scanf("%d %d", &n, &m);
	scanf("%s", s);

	a = b = 0;
	for(int i = 0; i < m; ++i) {
		a += ((s[i] == '(') ? 1 : -1);
		b = min(b, a);
	}

	memset(f, 0, sizeof(f));
	f[0][0] = 1;

	for(int i = 1; i <= n-m; ++i)
		for(int j = 0; j <= i; ++j) {
			if (j == 0)
				f[i][j] = f[i-1][j+1];
			else
				f[i][j] = f[i-1][j-1]+f[i-1][j+1];
			f[i][j] %= modulo;
		}

	long long result = 0;
	for(int i = 0; i <= n-m; ++i)
		for(int j = 0; j <= i; ++j)
			if (j+b >= 0 && j+a <= n-m-i)
				result += f[i][j]*f[n-m-i][j+a],
				result %= modulo;

	printf("%I64d\n", result);
	return 0;
}
