#include <bits/stdc++.h>

using namespace std;

typedef long long LL;

const int maxLen = (int)1E5+5;
const int maxL = (int)2E3+3;
const LL modulo = (int)1E9+7;

char s[maxLen];
LL f[maxL][maxL];
int n, m;
int a, b;

int main() {
	scanf("%d %d", &n, &m);
	scanf("%s", s);

	a = b = 0;
	for(int i = 0; i < m; ++i)
		a += ((s[i] == '(') ? 1 : -1),
		b = min(b, a);

	memset(f, 0, sizeof(f));
	f[0][0] = 1;

	for(int i = 1; i <= n-m; ++i)
		for(int j = 0; j <= i; ++j)
			f[i][j] = j ? (f[i-1][j+1]+f[i-1][j-1]) : f[i-1][1],
			f[i][j] %= modulo;

	LL result = 0;
	for(int i = 0; i <= n-m; ++i)
		for(int j = -b; j <= min(i, n-m-i-a); ++j)
				result += f[i][j]*f[n-m-i][j+a],
				result %= modulo;

	printf("%I64d\n", result);
	return 0;
}
