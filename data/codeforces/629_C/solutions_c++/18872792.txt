#include<bits/stdc++.h>
using namespace std;
const int N=5e3+4,MOD=1e9+7;
long long dp[N][N],n,m,i,j,tot,rem,mn=1e10,ans;
string  s;
int main()
{
	ios::sync_with_stdio(false);cin.tie(0);
	cin>> n >> m >> s;
	dp[0][0]=1;
	for( i = 1 ; i <N ; i++)
		for( j = 0 ; j <= i ; j++)
		{
			if(j)
				dp[i][j] = (dp[i][j] + dp[i-1][j-1])%MOD;
			dp[i][j] = (dp[i][j] + dp[i-1][j+1])%MOD;
		}
    for( i = 0 ; s[i] ; i++)
        {
        	if(s[i] == '(') tot++;
        	else
        	tot--;
          mn=min(mn,tot);
        }
        if(tot>2e3){cout<<0; return 0;}
    for( i= 0 ; i<= (n-m) ; i++)
       {
           rem = n-m-i;
           for( j = 0 ; j <= i ; j++ )
                if( j + mn >= 0 )  
            	ans = ( ans+ dp[i][j]*dp[rem][tot+j])%MOD;
       } 
       cout<<ans%MOD;
	return 0;
}
