#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll MOD = (ll)1e9+7;
ll dp[2005][2005];
char c[100005];
int main(){
	memset(dp, 0, sizeof dp);
	ll ans = 0,b=0,mb=100005;
	int n,m;
	scanf("%d%d",&n,&m);
	dp[0][0]=1;
	scanf("%s",&c);
	
	for(int i = 0; i < m; ++i)
		b+= c[i]=='('? 1: -1,mb = min(mb,b);
	
	for(int i = 1; i <=n-m; ++i)
	for(int j = 0; j <= i; ++j) 
		dp[i][j]= !j? dp[i-1][1]: dp[i-1][j-1]+dp[i-1][j+1],dp[i][j]%=MOD;
	
	
	for(int i = 0; i <= n-m; ++i)
	for(int j = 0; j <= i; ++j) 
		ans += (j+mb>=0 && j+b<=n-m)? dp[i][j]*dp[n-m-i][j+b]%MOD : 0;
	
	printf("%I64d",ans%MOD);
	return 0;
	}
