#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>
using namespace std;
const int N = 2010, M = 100010, mod = 1000000007;
#define LL long long
int n, m, cnt;
LL f[N][N], ans;
char s[M];
int main() {
	scanf("%d%d", &n, &m);
	scanf("%s", s);
	f[0][0] = 1;
	for(int i = 1; i <= n - m; i++)
		for(int j = 0; j <= n - m; j++) {
			if(j > 0) f[i][j] = (f[i][j] + f[i - 1][j - 1]) % mod;
			if(j < n - m) f[i][j] = (f[i][j] + f[i - 1][j + 1]) % mod;
		}
	int minr = 0;
	for(int i = 0; s[i]; i++) {
		if(s[i] == '(') cnt++;
		else cnt--;
		minr = min(minr, cnt);
	}
	for(int i = 0; i <= n - m; i++)
		for(int j = max(0, -minr); j <= n - m; j++)
			if(j + cnt >= 0 && j + cnt <= n - m)
				ans = (ans + f[i][j] * f[n - m - i][j + cnt]) % mod;
	cout << ans << endl;
	return 0;
}
