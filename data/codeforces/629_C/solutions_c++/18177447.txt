#include<bits/stdc++.h>
using namespace std;
# define l long long int
# define rep(i,a,b) for(l i=a;i<b;i++)
const int MOD=1e9+7; 
l dp[2005][2005];
int main()
{
	l n,m;
	char ch;
	l minb,b=0;
	cin>>n>>m;
	minb=m;
	dp[0][0]=1;
	rep(i,1,n-m+1)
		{dp[i][0]=dp[i-1][1]; rep(j,1,i+1) dp[i][j]=(dp[i-1][j-1]+dp[i-1][j+1])%MOD;}
	rep(i,0,m)
		{
			cin>>ch;
			if(ch==')') b--; else b++;
			if(minb>b) minb=b;
		}
	//cout<<minb<<" "<<b;
	l ans=0;
	rep(len,0,n-m+1)
	{
		rep(j,0,(len+1))
		{
			if(j+minb>=0 && n-m>=b+j)
				{ans=(ans+dp[len][j]*dp[n-m-len][j+b]%MOD)%MOD; }
		}
	}
	cout<<ans<<"\n";
	return 0;
}
