#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;
int n, m, f[2005][2005] = {{0}}, diffr, tot = 0, tat = 0;
long long ans = 0;
char s[100005] = {'\0'};
int main()
{
	scanf("%d%d", &n, &m);
	scanf("%s", s);
	diffr = n - m;
	f[0][0] = 1;
	for (int i = 1; i <= diffr; i++)
		for (int j = 0; j <= i; j++)
			f[i][j] = (f[i - 1][j + 1] + (j? f[i - 1][j - 1] : 0)) % MOD;
	for (int i = 0; i < m; i++)
	{
		tat += (s[i] == '('? 1 : -1);
		tot = max(tot, -tat);
	}
	for (int i = 0; i <= diffr; i++)
		for (int j = tot, _end = min(diffr, diffr - tat); j <= _end; j++)
			ans = (ans + 1LL * f[i][j] * f[diffr - i][j + tat]) % MOD;
	cout << ans << endl;
	return 0;
}
