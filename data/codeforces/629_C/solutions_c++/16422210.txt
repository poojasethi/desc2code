#include<iostream>
#include<algorithm>
using namespace std;
const int MOD =1e9+7;
#define M 2021
long long dp[M][M];
string  s;
int ans;

int main()
 {
 int n,m,mn=MOD,b=0;
 cin>>n>>m;
 cin>>s;
 dp[0][0]=1;
 for(int i=1;i<=n-m;i++)
   {
       dp[i][0]=dp[i-1][1];
       for(int j=1;j<=i;j++)
        dp[i][j]=(dp[i-1][j-1]+dp[i-1][j+1])%MOD;
   }
   for(int i=0;i<m;i++)
   {
       if(s[i]=='(')
        b++;
        else
        b--;
       if(mn>b)
        mn=b;
   }
   	for(int i=0;i<=n-m;i++)
		for(int j=0;j<=i;j++)
			if(0<=mn+j&&j+b<=n-m)
				ans=(ans+(dp[i][j]*dp[n-m-i][j+b])%MOD)%MOD;
   	cout<<ans;
 return 0;
 }
