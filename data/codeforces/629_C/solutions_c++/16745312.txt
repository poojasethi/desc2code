#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
#include <string>
#define mod 1000000007
#define rep(i, j, k) for(int i = j; i <= k; i++)

using namespace std;

int n, m, a[1000009], now, Min = 0x7fffffff;
long long f[2009][2009];
char s[1000009];
long long ans = 0;

void init ()
{
	f[0][0] = 1;
	rep (i, 1, n - m)
	{
		f[i][0] = f[i - 1][1];
		rep (j, 1, i)
			f[i][j] = f[i - 1][j + 1] + f[i - 1][j - 1], f[i][j] %= mod;
	}
}

int main ()
{
	cin >> n >> m;
	scanf ("%s", s);
	init ();
	rep (i, 0, m - 1)
		a[i] = s[i] == '(' ? 1 : -1, now += a[i], Min = min (Min, now);
	rep (i, 0, n - m)
		rep (j, 0, i)
			if (j >= -Min && j + now <= n - m && j + now >= 0)
				ans += f[i][j] * f[n - m - i][j + now] % mod, ans %= mod;
	cout << ans << endl;
	return 0;
}
