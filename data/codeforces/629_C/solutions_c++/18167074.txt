#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#include<iostream>
#define mod 1000000007

using namespace std;

int n,m;
int sum1,sum2,mn;
int f[2010][2010];
char s[100010];

int main()
{
	scanf("%d%d",&n,&m);
	f[0][0]=1;
	for (int i=1;i<=n-m;i++)
	{
		f[i][0]=f[i-1][1];
	    	for (int j=1;j<=i;j++) f[i][j]=(f[i-1][j-1]+f[i-1][j+1])%mod;
	}
	scanf("%s",s+1);
	int now=0,mn=0;
	for (int i=1;i<=m;i++) {if (s[i]=='(') sum1++,now++; else sum2++,now--;mn=min(mn,now);}
	int ans=0;
	for (int i=0;i<=n-m;i++)
	  for (int j=-mn;j<=n/2-sum1;j++)
	    ans=(ans+1ll*f[n-m-i][j+sum1-sum2]*f[i][j]%mod)%mod;
	printf("%d\n",ans);
	return 0;
}
