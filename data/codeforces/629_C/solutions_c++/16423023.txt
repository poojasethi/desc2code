#include <bits/stdc++.h>
using namespace std;

int _f[2001][2][4001];
int *f[2001][2];
char s[100001];

int main() {
	for (int i = 0; i<2001; i++)
		for (int j = 0; j<2; j++)
			f[i][j] = _f[i][j] + 2000;
	int n,m;
	scanf("%d%d",&n,&m);
	scanf("%s", s);
	int x = 0, y = 0;
	for (int i = 0; i<m; i++) {
		if (s[i]=='(') x++;
		else x--;
		y = min(x,y);
	}
	f[0][0][0] = 1;
	if (!y) f[0][1][0] = 1;
	for (int i = 1; i<=n-m; i++)
		for (int j = 0; j<2; j++)
			for (int k = m-n; k<=n-m; k++) {
				long long tmp = 0;
				if (k+j*x>0) tmp += f[i-1][j][k-1];
				if (k+j*x>=0) tmp += f[i-1][j][k+1];
				if (j>0 && k+y>=0 && k>=0) tmp += f[i][j-1][k];
				f[i][j][k] = tmp % 1000000007;
			}
	if (abs(x)<=n-m) printf("%d\n", f[n-m][1][-x]);
	else puts("0");
	return 0;
}
