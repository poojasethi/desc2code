#include <bits/stdc++.h>

using namespace std;

long long n, m, dp[2001][2001], ans;
int minb, b;
string s;

long long mod = 1e9+7;

int main(void) {

	int n, m;
	cin>>n>>m;

	cin>>s;
	dp[0][0] = 1;
	for(int i = 1; i <= n-m; i++) {
		dp[i][0] = dp[i-1][1];
		for(int j = 1; j <= i; ++j) {
			dp[i][j] = (dp[i-1][j-1] + dp[i-1][j+1])%mod;
		}
	}

	for(int i = 0; i < m; ++i) {
		if(s[i] == '(')	b++;
		else			b--;
		minb = min(minb, b);
	}

	for(int c = 0; c <= n-m; ++c) {
		for(int d = 0; d <= c; ++d) {
			if(d >= -minb) {
				if(d+b <= n-m && d+b >= 0)
					ans = (ans + (dp[c][d] * dp[n-m-c][d+b])%mod)%mod;
			}
		}
	}

	cout<<ans;

	return 0;
}
