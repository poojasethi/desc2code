#include<cstdio>
#include<algorithm>
using namespace std;
const int N=100005,M=2005;
const int mod=1e9+7;
typedef long long ll;
ll f[M][M],ans;
int n,m,x,mx;
char s[N];
int main(){
	scanf("%d%d",&n,&m);
	scanf("%s",s+1);
	n=n-m;
	f[0][0]=1;
	for (int i=0;i<n;i++)
		for (int j=0;j<=i;j++)if (f[i][j]){
			if (j) f[i+1][j-1]+=f[i][j],f[i+1][j-1]%=mod;
			f[i+1][j+1]+=f[i][j],f[i+1][j+1]%=mod;
		}
	for (int i=1;i<=m;i++) {
		if (s[i]==')') x++;else x--;
		mx=max(mx,x);
	}
	for (int i=mx;i<=n;i++) 
		for (int j=mx;j<=i;j++)
			if (f[i][j] &&j-x<=n-i) ans+=f[i][j]*f[n-i][j-x]%mod,ans%=mod;
	printf("%I64d",ans);
	return 0;
}
