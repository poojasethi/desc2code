#include <cstdio>
#include <cstring>
#define MAXN 2009
#define MAXM 200009
#define REM 1000000007

typedef long long int ll;

ll dp[MAXN][MAXN];
char str[MAXM];

int main()
{
	int n, m, bal=0, minbal=0;
	scanf("%d %d %s", &n, &m, str);
	for(int i=0; i<m; i++){
		if (str[i]=='(') bal++;
		else bal--;
		if (bal < minbal) minbal = bal;
	}
	memset(&dp, 0, sizeof dp);
	dp[0][0]=1;
	for(int i=0; i<=n-m; i++){
		for(int j=0; j<=i; j++){
			dp[i+1][j+1] += dp[i][j];
			dp[i+1][j+1] %= REM;
			if (j>0){
				dp[i+1][j-1] += dp[i][j];
				dp[i+1][j-1] %= REM;
			}
		}
	}
	ll ans=0;
	for(int c=0; c<=n-m; c++){
		for(int b=-minbal; b<=n-m; b++){
			if (b+bal >= MAXN) continue;
			ans += (dp[c][b]*dp[n-m-c][b+bal])%REM;
			ans %= REM;
		}
	}
	printf("%I64d\n", ans);
	return 0;
}
