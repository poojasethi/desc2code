#include<iostream>
using namespace std;
typedef long long ll;
const int maxn=2222;
const int mod=1e9+7;
int n,m;
string s;
ll dp[maxn][maxn];
int main()
{
	cin>>n>>m;
	cin>>s;
	dp[0][0]=1;
	for(int i=1;i<=n-m;i++){
		for(int j=0;j<=i;j++){
			if(j>0){
				dp[i][j]=(dp[i][j]+dp[i-1][j-1])%mod;
			}
			dp[i][j]=(dp[i][j]+dp[i-1][j+1])%mod;
		}
	}
	int mm=mod,cnt=0;
	for(int i=0;i<s.length();i++)
	{
		if(s[i]=='(')cnt++;
		else cnt--;
		mm=min(cnt,mm);
	}
	ll sum=0;
	for(int i=0;i<=n-m;i++)
	{
		for(int j=0;j<=i;j++)
		{
			if(j>=-mm&&n-m-i>=j+cnt)
				sum=(sum+dp[i][j]*dp[n-m-i][j+cnt])%mod;
		}
	}
		cout<<sum<<endl;
}
