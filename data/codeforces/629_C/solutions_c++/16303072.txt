#include<cstdio>
long long int  dp[2005][2005];
char s[100005];
#define mod 1000000007
main(){
	int m,n;
	scanf("%d%d",&m,&n);
	getchar();
	scanf("%s",s);
	int i,j;
	dp[0][0]=1;
	for(i=1;i<=m-n;i++){
		for(j=0;j<=i;j++){
			if(j>0){
				dp[i][j]+=dp[i-1][j-1];
				dp[i][j]=dp[i][j]%mod;
			}
			dp[i][j]+=dp[i-1][j+1];
			dp[i][j]=dp[i][j]%mod;
		}
	}
	int minq=0x3f3f3f3f,a=0;
	for(i=0;i<n;i++){
		if(s[i]=='('){
			a++;
		}
		else if(s[i]==')'){
			a--;
		}
		if(a<minq)
		minq=a;
	}
	long long int ans=0;
	for(i=0;i<=m-n;i++){
		for(j=0;j<=i;j++){
			if(j+minq>=0&&j+a<=m-n-i){
				ans+=(dp[i][j]*dp[m-n-i][j+a])%mod;
				ans=ans%mod;
			}
		}
	}
	printf("%lld",ans%mod);
}
