#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
static long long dp[2001][2001];
char a[100001];
int main(){
	int n,m;
	scanf("%d %d",&n,&m);
	scanf("%s",a);
	int br=0,mi=INT_MAX;
	for(int i=0;a[i]!='\0';i++){
		if(a[i]=='(') br++;
		else br--;
		mi=min(mi,br);
	}
	memset(dp,0,sizeof(dp[0][0])*2001*2001);
	dp[0][0]=1;
	long long ans=0;
	for(int i=1;i<=n-m;i++){
		dp[i][0]=dp[i-1][1];
		for(int j=1;j<=i;j++){
			dp[i][j]=dp[i-1][j-1]+dp[i-1][j+1];
			dp[i][j]%=mod;
		}
	}
	for(int i=0;i<=n-m;i++){
		for(int j=0;j<=i;j++){
			if(j>=-mi){
				if(j+br<=n-m&&j+br>=0)
				ans+=dp[i][j]*dp[n-m-i][j+br];
				ans%=mod;
			}
		}
	}
	printf("%lld\n",ans);
	return 0;
}
