#include <iostream>
#include <string>
using namespace std;
int n,m,ma,mi,t;
long long f[2010][2010],mo=1e9+7,ans;
string s;

int main()
{
    cin>>n>>m;
    cin>>s;
    for (int i=0;i<m;++i)
    {
        if (s[i]=='(') ++t;
        else --t;
        mi=min(mi,t);
        ma=max(ma,t);
    }
    f[0][0]=1;
    for (int i=1;i<=min(2000,n);++i)
    {
        f[i][0]=f[i-1][1]%mo;
        for (int j=1;j<=min(2000,n);++j)
        {
            f[i][j]=f[i-1][j-1]+f[i-1][j+1];
            f[i][j]%=mo;
        }
    }
    for (int i=0;i<=n-m;++i)
        for (int j=-mi;j<=2000;++j)
        if (j+t<=2000)
        {
            ans+=f[i][j]*f[n-m-i][j+t];
            ans%=mo;
        }
    cout<<ans;
    return 0;
}
