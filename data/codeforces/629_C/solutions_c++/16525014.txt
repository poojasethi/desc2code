#include <cstdio>
#include <iostream>
#include <cstring>
using namespace std;

const int mod=1000000007;
int n,m,i,minv=2147483647,v,j;
long long d[2010][2010],s;

int main()
{
    scanf("%d%d",&n,&m);
    char c[m+1];
    scanf("%s",c);
    d[0][0]=1;
    for (i=1;i<=n-m;++i)
        for (j=0;j<=i;++j)
        {
            if (j) d[i][j]=(d[i][j]+d[i-1][j-1])%mod;
            d[i][j]=(d[i][j]+d[i-1][j+1])%mod;
        }

    for (i=0;i<m;++i)
    {
        v+=c[i]=='('?1:-1;
        minv=minv<v?minv:v;
    }

    for (i=0;i<=n-m;++i)
        for (j=0;j<=i;++j)
            if (j+minv>=0 && j+v<=n-m-i)
                s=(s+d[i][j]*d[n-m-i][j+v]%mod)%mod;

    printf("%d\n",s);
    return 0;
}
