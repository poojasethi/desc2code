#include<iostream>
#include<string>
#include<cstdio>
using namespace std;
long long f[2100][2100];
const int p=1000000007;
int main()
{
	int n,m,i,j,val=0,minval=0;
	long long ans=0;
	string str;
	cin>>n>>m;
	cin>>str;
	if (n%2==1)
	{
		cout<<0<<endl;
		return 0;
	}
	f[0][0]=1;
	for (i=1;i<=n-m;i++)
	{
		for (j=0;j<=i;j++)
		{
			if (j!=0) f[i][j]=(f[i][j]+f[i-1][j-1])%p;
			f[i][j]=(f[i][j]+f[i-1][j+1])%p;
		}
	}
	for (i=0;i<m;i++)
	{
		if (str[i]=='(') val++; else val--;
		minval=min(minval,val);
	}
	for (i=0;i<=n-m;i++)
		for (j=0;j<=i;j++)
			if ( (n-m-i>=val+j) && (j+minval>=0) )
				ans=(ans+f[i][j]*f[n-m-i][j+val])%p;
	cout<<ans<<endl;
	return 0;
}
