#include <stdio.h>
#include <string.h>

long long int n, m, dp[4000][4000];
char lin[100010];

int main()
{
	memset(dp, 0, sizeof dp);
	scanf("%lld%lld", &n, &m);
	scanf(" %s", lin);
	int i, j;
	dp[0][0] = 1;
	for(i = 1; i <= n - m; i++)
	{
		dp[i][0] = dp[i - 1][1];
		for(j = 1; j <= i; j++)
		{
			dp[i][j] = dp[i - 1][j + 1] + dp[i - 1][j - 1];
			dp[i][j] %= 1000000007;
		}
	}
	long long int a = 0, b = 0, ans = 0;
	for(i = 0; i < m; i++)
	{
		if(lin[i] == '(')
			a++;
		else
			a--;
		if(a < b)
			b = a;
	}
	for(i = 0; i <= n - m; i++)
	{
		for(j = 0; j <= i; j++)
		{
			if(b * - 1 <= j)
				if(n - m >= j + a && j + a >= 0)
				{
					ans += dp[i][j] * dp[n - m - i][a + j] % 1000000007;
					ans %= 1000000007;
				}
		}
	}
	printf("%lld\n", ans);
	return 0;
}
