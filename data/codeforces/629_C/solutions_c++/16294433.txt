#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int mm = int(1e9) + 7;
int n, m, N;
int f[2004][2004];
char s[100050];

int main()
{
	scanf("%d%d", &n, &m);
	N = n - m;
	f[0][0] = 1;
	for (int i = 1; i <= N; ++i) {
		f[i][0] = f[i - 1][1];
		for (int j = 1; j <= i; ++j) {
			f[i][j] = f[i - 1][j - 1] + f[i - 1][j + 1];
			if (f[i][j] >= mm) f[i][j] -= mm;
		}
	}
	scanf("%s", s+1);
	int mx = 0, cnt = 0;
	for (int i = 1; i <= m; ++i) {
		if (s[i] == '(') --cnt;
		else ++cnt;
		if (cnt > mx) mx = cnt;
	}
	int ans = 0;
	for (int i = mx; i <= N; ++i) {
		for (int j = mx; j <= i; ++j) {
			if (N - i >= j - cnt){
				ans += 1ll * f[i][j] * f[N - i][j - cnt] % mm;
				if (ans >= mm) ans -= mm;
			}
		}
	}
	cout << ans << endl;
}
