#include <cstdio>
#include <iostream>
#include <string>
#include <cstring>
#include <climits>
using namespace std;
const int MOD=1000000007;
long long d[2016][2016];
int main ()
{
	string s;
	int n,m,i,j;
	cin>>n>>m>>s;
	memset(d,0,sizeof(d));
	d[0][0]=1;
	for (i=1;i<=n-m;i++)
		for (j=0;j<=i;j++)
		{
			if (j>0) d[i][j]=(d[i][j]+d[i-1][j-1])%MOD;
			d[i][j]=(d[i][j]+d[i-1][j+1])%MOD;
		}
	int cur=0,mincur=0;
	for (i=0;i<s.size();i++)
	{
		if (s[i]=='(') cur++; else cur--;
		mincur=min(mincur,cur); 
	}
	long long ans=0;
	for (i=0;i<=n-m;i++)
		for (j=0;j<=i;j++)
			if (j+mincur>=0 && j+cur<=n-m-i)
				ans=(ans+d[i][j]*d[n-m-i][j+cur]%MOD)%MOD;
	cout<<ans<<endl;
	return 0;	
}
