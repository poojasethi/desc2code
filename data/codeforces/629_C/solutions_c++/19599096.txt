#include <stdio.h>
#include <string.h>
#include <algorithm>
#define M 100005
#define N 2005
#define P 1000000007
using namespace std;
char str[M];
int n,m,dp[N][N];
int Mod(int &a,int b){
	a+=b;
	if(a>=P)a-=P;
}
void Init(){
	dp[0][0]=1;
	for(int i=1;i<=n-m;i++)
		for(int j=0;j<=i;j++){
			if(j>0)Mod(dp[i][j],dp[i-1][j-1]);
			if(j<n-m)Mod(dp[i][j],dp[i-1][j+1]);
		}
}
int main(){
	int i,j,ans=0;
	scanf("%d %d %s",&n,&m,str);
	Init();
	int cnt=0,mi=n;
	for(i=0;i<m;i++)cnt+=(str[i]==')')?-1:1,mi=min(mi,cnt);
	mi=max(-mi,0);
	for(i=0;i<=n-m;i++)
		for(j=mi;j<=i;j++){
			if(j+cnt>n-m-i)break;
			Mod(ans,1ll*dp[i][j]*dp[n-m-i][j+cnt]%P);
		}
	printf("%d\n",ans);
	return 0;
}
