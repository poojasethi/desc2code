#include <bits/stdc++.h>
#define mod 1000000007
#define REP(i,a,b) for(i=a;i<=b;i++)
using namespace std;
typedef long long ll;
ll dp[2010][2010];
string s;
void init(){
	ll i,j;
	dp[0][0]=1;
	REP(i,1,2009){
		dp[i][0]=dp[i-1][1];
		REP(j,1,i)
			dp[i][j]=(dp[i-1][j-1]+dp[i-1][j+1])%mod;
	}
}
int main(){
	//freopen("in.txt","r",stdin);
	ios::sync_with_stdio(false);
	init();
	ll i,j,n,m,l,ans=0,bal=0,mbal=0;
	cin>>n>>m>>s;
	l=s.length();
	REP(i,0,l-1){
		if(s[i]=='(')
			bal++;
		else
			bal--;
		mbal=max(mbal,-bal);
	}
	REP(i,0,n-m){
		REP(j,0,i)
			if(j>=mbal)
				if(j+bal<=n-m)
					ans=( ans + (dp[i][j]*dp[n-m-i][bal+j])%mod )%mod;
	}
	cout<<ans<<endl;
}
