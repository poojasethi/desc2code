#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int mod=1e9+7;
const int N=2010;
int n,m;
string s;
ll dp[N][N];
void add(ll &x, ll y){
   x+=y;
   x%=mod;
}
int main(){
   #ifndef ONLINE_JUDGE
      freopen("input.txt","r",stdin);
   #endif // ONLINE_JUDGE
   scanf("%d%d",&n,&m);
   cin>>s;
   int k=0;
   int mi=0;
   for(int i=0;i<s.size();++i)
      k+=s[i]==')'?-1:1,mi=min(mi,k);
   dp[0][0]=1;
   for(int i=1;i<=2000;++i)
      for(int j=0;j<=2000;++j){
         dp[i][j]=dp[i-1][j+1];
         if(j>0)add(dp[i][j],dp[i-1][j-1]);
      }
   ll ans=0;
   for(int i=0;i<=n-m;++i)
   for(int j=0;j<=n-m;++j){
      if(j+k<0||j+k>2000||j+mi<0)continue;
      add(ans,dp[i][j]*dp[n-m-i][j+k]);
   }
   printf("%I64d",ans);
   return 0;
}
