#include<stdio.h>
#include<string.h>

#define mod 1000000007
#define ll __int64

int dp[2001][2001];
char str[100001];

int main()
{
	int i,j;
	int n,m;
	scanf("%d%d",&n,&m);
	dp[0][0]=1;
	for(i=1;i<=n-m;i++)
	{
		dp[i][0]=dp[i-1][1];
		for(j=1;j<=i;j++)
			dp[i][j]=(dp[i-1][j-1]+dp[i-1][j+1])%mod;
	}
	getchar();
	gets(str);
	int cnt=0,minc=0;
	for(i=0;i<m;i++)
	{
		if(str[i]=='(') cnt++;
		else cnt--;
		if(cnt<minc)
			minc=cnt;
	}
	int ans=0;
	for(i=0;i<=n-m;i++)
	{
		if(i+cnt<0) continue;
		for(j=0;j<=i;j++)
		{
			if(j+minc<0 || j+cnt>n-m) continue;
			ans=(ans+(ll)dp[i][j]*dp[n-m-i][j+cnt])%mod;
		}
	}
	printf("%d\n",ans);
	return 0;
}
