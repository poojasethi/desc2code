#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
typedef pair<LL,LL> PLL;
#define pb push_back
#define mp make_pair
LL dp[2001][2001];
LL mod=1000000007,n,m,ans=0;
string str;
signed main(){
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	dp[0][0]=1;
	for(LL i=1;i<2001;i++){
		for(LL j=0;j<=i;j++){
			if(j==0){dp[i][j]=dp[i-1][1];continue;}
			dp[i][j]=(dp[i-1][j+1]+dp[i-1][j-1])%mod;
		}
	}
	cin>>n>>m>>str;
	LL pref=0,suff=0;
	for(LL i=0;i<str.length();i++){
		if(str[i]=='('){
			suff++;
		}else{
			if(suff>0){
				suff--;
			}else{
				pref++;
			}
		}
	}
	for(LL i=0;i<=n-m;i++){
		for(LL j=pref;j<=i;j++){
			if(j-pref+suff<=n-i-m){
					ans+=(dp[i][j]*dp[n-i-m][j-pref+suff])%mod;
					ans%=mod;
				}
		}
	}
	cout<<ans<<"\n";
	return 0;
}
