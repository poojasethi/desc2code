#include <bits/stdc++.h>

using namespace std;
pair<pair<int,int>,int> play[112345],act[112345];
int use[112345];
int ans[112345];
int main()
{
	int n;cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>play[i].first.first>>play[i].first.second;
		play[i].second=i;
	}
	int m;cin>>m;
	for(int i=1;i<=m;i++)
	{
		cin>>act[i].first.first>>act[i].first.second>>use[i];
		act[i].second=i;
	}
	sort(play,play+n);
	sort(act+1,act+1+m);
	int now=1;
	multiset<pair<int,int> > s;
	pair<int,int> tmp;
	for(int i=0;i<n;i++)
	{
		int a=play[i].first.first,b=play[i].first.second;
		while(now<=m&&act[now].first.first<=a)
		{
			s.insert(make_pair(act[now].first.second,act[now].second));
			now++;
		}
		multiset<pair<int,int> >::iterator it;
		tmp.first=b;tmp.second=0;
		it=s.lower_bound(tmp);
		if(it==s.end())
		{
			cout<<"NO"<<endl;return 0;
		}
		else
		{
			tmp=*it;
			ans[play[i].second]=(*it).second;
			s.erase(it);
			use[tmp.second]--;
			if(use[tmp.second]>0)
				s.insert(tmp);
		}
	}
	cout<<"YES"<<endl;
    for(int i=0;i<n;i++)
    {
        printf("%d%c",ans[i],i==n-1?'\n':' ');
    }
	return 0;
}
