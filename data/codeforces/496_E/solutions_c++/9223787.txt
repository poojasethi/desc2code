#include <bits/stdc++.h>
using namespace std;

struct Node{
	int lo, hi, type, num;
};

bool comp(Node a, Node b){
	if(a.lo != b.lo)	return a.lo < b.lo;
	return a.type < b.type;
}

int n, m;
Node arr[200005];
int cnt[200005], ans[200005];

typedef pair<int, int> ii;



int main(){

	int i, j;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>arr[i].lo>>arr[i].hi;
		arr[i].type = 1;
		arr[i].num = i;
	}
	
	cin>>m;
	for(j=0;j<m;j++){
		cin>>arr[i].lo>>arr[i].hi>>cnt[j];
		arr[i].type = 0;
		arr[i].num = j;
		i++;
	}

	sort(arr, arr+n+m, comp);
	
// for(i=0;i<n+m;i++)cout<<arr[i].lo<<" "<<arr[i].hi<<" "<<arr[i].num<<" "<<arr[i].type<<endl;	
	
	multiset<ii> actors;
	multiset<ii>::iterator it;
	for(i=0;i<n+m;i++){
		if(arr[i].type == 0){
			actors.insert(ii (arr[i].hi, arr[i].num));	continue;
		}
		
		it = actors.lower_bound(ii (arr[i].hi, -1));
		if(it == actors.end()){
			cout<<"NO";	return 0;
		}
		
		ans[arr[i].num] = (*it).second;
		
		cnt[(*it).second]--;
		if(cnt[(*it).second] == 0)	actors.erase(it);
	}
	
	cout<<"YES\n";
	for(i=0;i<n;i++)	cout<<ans[i]+1<<" ";
	
	return 0;
}
