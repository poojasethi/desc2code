#include<iostream>
#include<algorithm>
#include<vector>
#include<set>
using namespace std;

int ans[100005],k[100005];
vector< pair< pair<int,int>,pair<int,int> > >v;
set< pair<int,int> >s;
set< pair<int,int> >::iterator it;

int main()
{
    int i,j,n,m,temp,l,r,idx;
    cin>>n;
    for(i=0;i<n;i++) {
        cin>>l>>r;
        v.push_back(make_pair(make_pair(r,0),make_pair(l,i)));
    }
    cin>>m;
    for(i=0;i<m;i++) {
        cin>>l>>r>>k[i];
        v.push_back(make_pair(make_pair(r,1),make_pair(l,i)));
    }
    sort(v.begin(),v.end(),greater< pair<pair<int,int>,pair<int,int> > >());
    for(i=0;i<n+m;i++) {
        if(v[i].first.second==1) {
            s.insert(make_pair(-v[i].second.first,v[i].second.second));
        } else {
            it=s.lower_bound(make_pair(-v[i].second.first,0));
            if(it==s.end()) {
                cout<<"NO"; return 0;
            } else {
                idx=(*it).second;
                ans[v[i].second.second]=idx;
                k[idx]--;
                if(k[idx]<=0) s.erase(it);
            }
        }
    }
    cout<<"YES\n";
    for(i=0;i<n;i++) cout<<ans[i]+1<<" ";
    return 0;
}
