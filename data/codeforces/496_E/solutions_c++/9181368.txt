#include <iostream>
#include <cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<map>
#define N 400100
using namespace std;
struct data{int l,r,id;}a[N],b[N];
bool cmp(data a,data b)
{return a.r<b.r;}
int n,m,k[N],ans[N];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d%d",&a[i].l,&a[i].r);
		a[i].id=i;
	}
	sort(a+1,a+1+n,cmp);
	scanf("%d",&m);
	for(int i=1;i<=m;i++)
	{
		scanf("%d%d%d",&b[i].l,&b[i].r,&k[i]);
		b[i].id=i;
	}
	sort(b+1,b+1+m,cmp);
	
	multimap<int,int> song;
	int now=1;
	for(int i=1;i<=m;i++)
	{
		while(now<=n&&a[now].r<=b[i].r)
		 song.insert(make_pair(a[now].l,a[now].id)),now++;
		multimap<int,int>::iterator it=song.lower_bound(b[i].l);
		while(it!=song.end()&&k[b[i].id])
		{
			k[b[i].id]--;
			ans[it->second]=b[i].id;
			song.erase(it);
			it=song.lower_bound(b[i].l);
		}
	}
	if(now<=n||!song.empty())
	{
		puts("NO");
		return 0;
	}
	puts("YES");
	for(int i=1;i<=n;i++)
	 printf("%d ",ans[i]);
}
