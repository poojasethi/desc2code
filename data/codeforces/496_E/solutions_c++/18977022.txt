#include <bits/stdc++.h>

#define NN 200010

using namespace std;
typedef pair <int, int> pii;
typedef pair <pii, int> pi3;

pi3 p[NN];
set <pii> s;
set <pii> :: iterator it;
int ans[NN];
int cnt[NN];

int main (){
	
	int N, M, i, j;
	
	scanf("%d", &N);
	for(i=1; i<=N; i++){
		scanf("%d %d", &p[i].first.first, &p[i].first.second);
		p[i].second=i;
	}
	
	scanf("%d", &M);
	for(i=1; i<=M; i++){
		scanf("%d %d %d", &p[N+i].first.first, &p[N+i].first.second, cnt+i);
		p[N+i].first.second*=(-1);
		p[N+i].second=i;
	}
	
	sort(p+1, p+N+M+1);
	
	for(i=1; i<=N+M; i++){
		if(p[i].first.second<0){
			s.insert(pii(-p[i].first.second, p[i].second));
		}
		else{
			it=s.lower_bound(pii(p[i].first.second, 0));
			if(it==s.end()){
				puts("NO");
				return 0;
			}
			ans[p[i].second]=(*it).second;
			cnt[(*it).second]--;
			if(!cnt[(*it).second]) s.erase(it);
		}
	}
	
	puts("YES");
	for(i=1; i<=N; i++){
		printf("%d ", ans[i]);
	}
	puts("");
	
			
	return 0;
}
