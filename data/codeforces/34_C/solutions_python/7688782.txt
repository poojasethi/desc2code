l=map(int,raw_input().split(','))
l=sorted(l)
m=[[l[0]]]
j=""
s=set(l)
nl=sorted(list(s))
for i in nl:
    if i+1 in nl:
        m[len(m)-1].append(i+1)
    else:
        if i+1<max(nl):
            m.append([])
            m[len(m)-1].append(l[len(l)-l[::-1].index(i)])
for i in range(len(m)):
    if len(m[i])>1:
        j+=str(m[i][0])+'-'+str(m[i][-1])+','
    else:
        j+=str(m[i][0])+','
print j[:-1]
