#!/usr/bin/env python

import sys

l = map(int, sys.stdin.readline().rstrip('\n').split(','))

dict = {}
list = []

for e in l:
    if str(e) not in dict:
        list.append(e)
        dict[str(e)] = True

list.sort()
leng = len(list)

i = 0
sol = []

while i < leng:
    start = list[i]
    last = list[i]
    while i + 1 < leng and list[i + 1] == list[i] + 1:
        i += 1
        last = list[i]
    if start == last:
        sol.append(str(start))
    else:
        sol.append("%d-%d" % (start, last))
    i += 1

print ','.join(sol)
