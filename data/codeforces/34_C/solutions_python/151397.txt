array = map(int, raw_input().split(','))
array = list(set(array))
array.sort()
display = []

islist = False
for i in xrange(len(array)):
    if array[i-1] == array[i] - 1 and i != 0:
        islist=True
    else:
        if islist:
            display[len(display)-1] += "-%s" % array[i-1]
        display.append("%s" % array[i])
        islist = False

if islist:
    display[len(display)-1] += "-%s" % array[len(array)-1]

print ','.join(display)
