s=map(int, raw_input().split(','))
s.sort()
start,end=s[0],s[0]
ans=[]
for i in xrange(1,len(s)):
    if s[i]-s[i-1]>1:
        ans.append((start,end))
        start,end=s[i],s[i]
    else:end=s[i]
ans.append((start,end))
x=[]
for s,e in ans:
    if s==e:x.append(str(s))
    else:x.append(str(s)+'-'+str(e))
print ','.join(x)
