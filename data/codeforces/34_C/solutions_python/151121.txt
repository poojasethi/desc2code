a = [[i] for i in sorted(list(set(map(int, raw_input().split(',')))))]
try:
  for i in xrange(len(a) - 1):
    if a[i][-1] == a[i + 1][0] - 1:
      a[i], a[i + 1] = ('_', a[i][:-1] + [a[i + 1][0]]) \
        if len(a[i]) == 3 else ('_', [a[i][0], '-', a[i + 1][0]])
except:
  pass
print ','.join([str(''.join([str(p) for p in s])) for s in a if s != '_'])
