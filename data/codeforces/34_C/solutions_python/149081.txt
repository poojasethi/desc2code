s = raw_input()
nums = sorted( [ int( x ) for x in s.split( "," ) ] )
res = ""
i = 0
while i < len( nums ) :
	b = nums[ i ]
	j = i + 1
	k = b
	while j < len( nums ) and nums[ j ] - k <= 1 :
		k = nums[ j ]
		j += 1
	if len( res ) > 0 : res += ","
	if k - b >= 1 : res += str( b ) + "-" + str( k )
	else : res += str( b )
	i = j
print res
