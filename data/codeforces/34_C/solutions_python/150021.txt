l = list(set([int(s, 10) for s in raw_input().split(",")]))
l.sort()
ll = [[]]
s = ""
for i in l:
    if ll[-1] == [] or ll[-1][-1]+1==i:
        ll[-1].append(i)
    else:
        ll.append([i])
for i in ll:
    if len(i) >= 2:
        s += "%d-%d," %(i[0], i[-1])
    else:
        s += "%d," % i[0]

print s[:-1]
