inp = map(int, raw_input().split(","))
s = []
for page in inp:
	if(page not in s):
		s.append(page)
s.sort()
j = 0
res = [[s[0]]]
for i in xrange(1,len(s)):
	if(s[i] != s[i-1]+1):
		res.append([s[i]])
		j+=1
	else:
		res[j].append(s[i])
ans = []
for i in xrange(len(res)):
	if(len(res[i])>1):
		ans.append(str(res[i][0])+"-"+str(res[i][-1]))
	else:
		ans.append(str(res[i][0]))
print ",".join(ans)
