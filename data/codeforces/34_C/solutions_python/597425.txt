t = sorted(list(set(map(int, raw_input().split(",")))))
ans = [[t[0]]]
for i in xrange(1, len(t)):
    if t[i] == t[i-1]+1:
        ans[-1].append(t[i])
    else:
        ans.append([t[i]])
a = ""
for i in ans:
    if len(i) >= 2:
        a += str(i[0])+"-"+str(i[-1])+","
    if len(i) == 1:
        a += str(i[0])+","
print a[:-1]
