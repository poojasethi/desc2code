a = sorted(list(set(map(int,raw_input().split(',')))))
a = [[g] for g in a]
i = 1
while i<len(a):
    if a[i][0]-a[i-1][-1]==1:
        if len(a[i-1])==1:
            a[i-1].append(a[i][0])
        else:
            a[i-1][1] = a[i][0]
        a.pop(i)
    else:
        i+=1
s = ''
for b in a:
    if (len(b)==2):
        s+= str(b[0])+'-'+str(b[1])+','
    else:
        s+= str(b[0])+','
print s[:-1]
