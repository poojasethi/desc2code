z=[0]*2001
num=set([int(x) for x in raw_input().split(',')])
num=list(num)
num.sort()

ans=""
for i in range (0,len(num)):
		z[num[i]]=1

ans=""
zq=[0]*1001
for i in range (1,1001):
	if ((z[i]==1) and (zq[i]==0)):
		j=i
		endpos=j
		c=i 
		while(1):
			if (z[c]==0 or c>1000):
				break
			elif ((z[c]==1) and (zq[c]==0)):
				zq[c]=1
				endpos=c
				c+=1
		if (endpos==j):
			ans+=str(j)+","
		else:
			ans+=str(j)+"-"+str(endpos)+","

print(ans[:len(ans)-1])
