import sys
list=sys.stdin.readline().split(',')
for i in range(len(list)):
	list[i]=int(list[i])
list.sort()
beg=list[0]
first=1
s=""
for i in range(len(list)+1):
	if ( i==len(list) or (i>0 and list[i]>list[i-1]+1)):
		if (first==0):
			s+=","
		first=0
		if (list[i-1]==beg):
			s+=str(beg)
		else:
			s+=str(beg)
			s+="-"
			s+=str(list[i-1])
		if (i!=len(list)):
			beg=list[i]
