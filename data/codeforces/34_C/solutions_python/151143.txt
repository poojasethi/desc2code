#a = [[i] for i in sorted(list(set(map(int, raw_input().split(',')))))]
def fun(farr, obj):
  return fun(farr[:-1], farr[-1](obj)) if len(farr) > 0 else obj
a = [[i] for i in fun([sorted, list, set, lambda x: map(int, x), lambda x: x.split(',')], raw_input())]
for i in xrange(len(a) - 1):
  if a[i][-1] == a[i + 1][0] - 1:
    a[i], a[i + 1] = '', a[i][:-1] + a[i + 1] if len(a[i]) == 3 else [a[i][0], '-', a[i + 1][0]]
print ','.join([str(''.join(map(str, s))) for s in a if s != ''])
