a=sorted(set(map(int,raw_input().split(','))))
pre=0
for i in range(len(a)):
    a[i]=[a[i]]
    if i>0 and a[i][0]==a[pre][-1]+1:
        a[pre].append(a[i][0])
        a[i]=[]
    else:
        pre=i
for i in range(len(a)):
    if len(a[i])>1:
        a[i]=str(a[i][0])+'-'+str(a[i][-1])
    elif len(a[i])==1:
        a[i]=a[i][0]
print ','.join(map(str,filter(lambda x:x,a)))
