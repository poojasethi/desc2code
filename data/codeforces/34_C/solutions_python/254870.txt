from __future__ import print_function
from itertools import *
import sys
lst, i = sorted (list (set (map (int, raw_input ().split (','))))), 0
while i < len (lst):
    if i + 1 < len (lst) and lst[i] + 1 == lst[i + 1]:
        print (lst[i], '-', sep='', end='')
        while i + 1 < len (lst) and lst[i] + 1 == lst[i + 1]:
            i += 1
        print (lst[i], end=',' if i + 1 < len (lst) else '')
    else:
        print (lst[i], end=',' if i + 1 < len (lst) else '')
    i += 1
