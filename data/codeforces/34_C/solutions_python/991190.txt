a=sorted(map(int,raw_input().split(",")))
mn=mx=a[0];
ans=[]
for i in a:
	if mx+1 >= i:
		mx=i;
	else:
		ans.append([mn,mx])
		mn=mx=i
ans.append([mn,mx])
import sys
write=sys.stdout.write;
for j,i in enumerate(ans):
	if i[0]!=i[1]:
		write("%d-%d"%(i[0],i[1]))
	else:
		write(repr(i[0]))
	if j != len(ans) - 1: write(",")
print
