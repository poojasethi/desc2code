from sys import stdin
l_tmp = map(int, stdin.read().split(","))
l = sorted(set(l_tmp))
b = e = l[0]
s = []
for item in l:
    if e == item:
        continue
    elif e == item -1:
        e = item
    else:
        s.append((b,e))
        b = e = item
s.append((b,e))

def period(pair):
    if pair[0]==pair[1]:
        return str(pair[0])
    return "%s-%s" % pair

print ','.join(map(period,s))
