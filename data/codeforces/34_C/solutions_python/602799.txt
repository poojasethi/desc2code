import sys

d = {}
a = [ int(x) for x in raw_input().split(',') ]

for i in a:
  d[i] = True

i = 0
first = True
while True:
  if i > 1000:
    break

  if not (i in d):
    i += 1
    continue

  ll = 0
  while i in d:
    ll += 1
    i += 1

  if not first:
    sys.stdout.write(',')
  else:
    first = False

  if ll == 1:
    sys.stdout.write(str(i-1))
  else:
    sys.stdout.write(str(i-ll) + '-' + str(i-1))

sys.stdout.write("\n")
