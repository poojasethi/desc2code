buf = raw_input().split(',')
nums = []
for i in buf:
	nums.append(int(i))
nums.sort()
a = 0
buf = 0
arg = ''
nums.append(0)
for i in nums:
	if a == 0:
		a = i
	elif a == i:
		continue
	elif a + 1 == i:
		if buf == 0:
			buf = a
		a = i
	elif buf != 0 and a + 1 != i:
		arg += str(buf) + '-' + str(a) + ','
		buf = 0
		a = i
	elif buf == 0 and a + 1 != i:
		arg += str(a) + ','
		a = i
arg = arg.rstrip(',')
print arg
