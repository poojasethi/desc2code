from sys import stdin
a = stdin.readline().strip().split(',')
b = []
for i in a:
    b.append(int(i))
b = list(set(b))
b.sort()
c = []
pr = b[0] -1
tmp = []
ans = []
for i in b:
    if i==pr+1:
        tmp.append(i)
    else:
        c.append(tmp)
        tmp = [i]
    pr = i
if len(tmp)>0:
    c.append(tmp)

for u in c:
    if u[0]==u[-1]:
        ans.append(str(u[0]))
    else:
        ans.append(str(u[0])+"-" + str(u[-1]))
print ",".join(ans)
