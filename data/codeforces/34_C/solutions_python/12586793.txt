arr = raw_input().split(',')
arr = map(int,arr)
arr = set(arr)
arr = sorted(list(arr))
ans=''
prev = -99
flag = 0
for x in arr:
    if abs(prev-x)<=1:
        flag=1
        prev=x
        continue
    if abs(prev-x)>1:
        if flag==1:
            ans+='-'
            ans+=str(prev)
            flag=0
        if len(ans)!=0:
            ans+=','
        ans+=str(x)
        prev=x
if flag==1:
    ans+='-'
    ans+=str(prev)
print(ans)
