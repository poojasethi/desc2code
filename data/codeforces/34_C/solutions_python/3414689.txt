import sys

s = set(map(int, raw_input().split(',')))
s = sorted(s)
first = -1
last = -1
for i in range(len(s)):
    if first == -1:
	first = last = s[i]
    elif last + 1 == s[i]:
	last = s[i]
    else:
	if first == last:
            sys.stdout.write("%s," % (first))
	else:
            sys.stdout.write("%d-%d," % (first, last))
	first = last = s[i]

if first == last:
    print first
else:
    print "%d-%d" % (first, last)
