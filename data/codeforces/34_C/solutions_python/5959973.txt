import sys

pages = sorted(set(map(int, raw_input().split(','))))
sys.stdout.write(str(pages[0]))
seq = False;
for i in range(1,len(pages)):
	if pages[i] == pages[i - 1] + 1:
		if not seq:
			sys.stdout.write('-')
			seq = True;
	elif seq:
		seq = False;
		sys.stdout.write(str(pages[i - 1]) + ',' + str(pages[i]))
	else:
		sys.stdout.write(',' + str(pages[i]))
if seq:
	sys.stdout.write(str(pages[len(pages) - 1]))
sys.stdout.write('\n')
