def main():
	a = map(int, raw_input().split(","))
	a.sort()
	ans = []
	i = 0
	while i < len(a):
		x = a[i]
		j = i + 1
		for j in xrange(i + 1, len(a)):
			if a[j] == x + 1: x += 1
			elif a[j] > x + 1: break
		if x == a[i]: ans.append(str(x))
		else: ans.append(str(a[i]) + "-" + str(x))
		i = j if j == len(a) or a[j] != x else j + 1
	print ",".join(ans)	
			

if __name__ == "__main__": main()
