a = sorted(list(set(map(int, raw_input().split(',')))))
line = []
length = len(a)
a.append(1005)
i = 0
while i < length:
    j = i
    while a[i]+1==a[i+1]:
        i+=1
    if i-j>0:
        line.append("%d-%d" % (a[j], a[i]))
    else:
        line.append("%d" % (a[j]))
    i += 1
print ','.join(line)
