sp = map(int, raw_input().split(','))
sp.sort()
l = sp[0]
r = sp[0]
res = ""
for i in sp[1::]:
    if i - r > 1:
        if l != r:
            res += str(l) + "-" + str(r)
        else:
            res += str(l)
        res += ","
        l = i
        r = i
    else:
        r = i
if l != r:
    res += str(l) + "-" + str(r)
else:
    res += str(l)
print res
