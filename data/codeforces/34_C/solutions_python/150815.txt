seq = sorted(list(set([ int(x) for x in raw_input().split(',') ])))
new_seq = []
i = 0
while i < len(seq):
    j = i + 1
    while j < len(seq) and seq[j] == seq[j-1] + 1:
        j += 1
    if j - 1 > i:
        ele = "%i-%i" % (seq[i], seq[j-1])
        new_seq.append(ele)
        i = j
    else:
        new_seq.append(seq[i])
        i += 1

print ','.join(map(str, new_seq))
