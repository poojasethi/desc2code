def f(s,e):
    if s==e: return str(s)
    else: return str(s)+'-'+str(e)
    
s=map(int, raw_input().split(','))
s.sort()
start,end=s[0],s[0]
ans=[]
for i in xrange(1,len(s)):
    if s[i]-s[i-1]>1:
        ans.append(f(start,end))
        start,end=s[i],s[i]
    else:end=s[i]
ans.append(f(start,end))

print ','.join(ans)
