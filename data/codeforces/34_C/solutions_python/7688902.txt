l=sorted(map(int,raw_input().split(',')))
m=[[l[0]]]
j=""
nl=sorted(list(set(l)))
for i in nl:
    if i+1 in nl:
        m[len(m)-1].append(i+1)
    else:
        if i+1<max(nl):
            m.append([])
            m[len(m)-1].append(l[len(l)-l[::-1].index(i)])
for i in range(len(m)):
    j+=str(m[i][0])+'-'+str(m[i][-1])+',' if len(m[i])>1 else str(m[i][0])+','
print j[:-1]
