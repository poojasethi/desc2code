#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	bool was[1001] = {};
	int a[101], n = 0;
	int k;
	while (cin >> k) {
		if (!was[k]) {
			was[k] = true;
			a[n++] = k;
		}
		cin.ignore(1);
	}
	sort(a, a+n);

	cout << a[0];
	for (int i = 1; i < n; i++) {
		if (a[i-1] + 1 != a[i]) {
			if (i >= 2 && a[i-2] + 1 == a[i-1]) {
				cout << "-" << a[i-1];
			}
			cout << "," << a[i];
		}
	}
	if (n >= 2 && a[n-2]+1 == a[n-1]) {
		cout << "-" << a[n-1];
	}
	cout << endl;
	return 0;
}
