#include<iostream>
#include<cstdio>

using namespace std;

const int maxn = 1010;

int tub[maxn];

int main() {
	int x , first = 1;
	while ( scanf( "%d," , &x ) == 1 ) tub[x]++;
	for ( int i = 1 , s = -1 ; i <= 1000 ; i++ ) {
		if ( tub[i] > 0 && tub[i - 1] == 0 ) s = i;
		if ( tub[i] > 0 && tub[i + 1] == 0 ) {
			if ( first ) first = 0; else printf( "," );
			if ( i != s ) printf( "%d-%d" , s , i );
				else printf( "%d" , i );
		}
	}
	printf( "\n" );
	return 0;
}
