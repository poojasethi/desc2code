#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int n,m;
int a[100+10];

int main(){
	int n=0,t;
	bool h=true;
	while(true){
		if(h){
			h=false;
			t=scanf("%d",&a[n++]);
		}else{
			t=scanf(",%d",&a[n++]);
		}
		if(t<1)break;
	}
	n--;
	sort(a,a+n);
	n=unique(a,a+n)-a;
	h = true;
	for(int i=0;i<n;){
		int j;
		for(j=i+1;j<n&&a[j]==a[j-1]+1;j++);
		if(!h)printf(",");
		else h=false;
		if(j==i+1){
			printf("%d",a[i]);	
		}else
			printf("%d-%d",a[i],a[j-1]);
		i=j;
	}
	printf("\n");
	return 0;
}
