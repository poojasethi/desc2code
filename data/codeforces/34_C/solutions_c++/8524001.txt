#include <bits/stdc++.h>

using namespace std;

const int N = 1005;
int a[N];
int main() {
    int n = 0;
    while (~scanf("%d,", a + n)) n++;
    sort(a, a + n);
    n = unique(a, a + n) - a;
    for (int i = 0; i < n; i++) {
        int j = i + 1;
        for (; j < n && (a[j] == a[j - 1] + 1); j++) {
        } 
        if (j > i + 1) {
            cout << a[i] << '-' << a[j - 1];
            if (j < n) cout << ',';
        } else {
            cout << a[i];
            if (i + 1 < n) cout << ',';
        }
        i = j - 1;
    }
    return 0;
}
