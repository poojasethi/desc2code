#include <cstdio>
#include <algorithm>
using namespace std;
int a[10000];
int main()
{
	int n;
	for (int i=0;;i++) {
		int tmp=scanf("%d,",&a[i]);
		if (tmp!=1) {
			n=i;
			break;
		}
	}
	sort(a,a+n);
	n=unique(a,a+n)-a;
	for (int i=0;i<n;) {
		int j=i;
		for (;j<n && (i==j || a[j]==a[j-1]+1);j++);
        if (i==j-1)
            printf("%d%c",a[i],",\n"[j==n]);
        else
            printf("%d-%d%c",a[i],a[j-1],",\n"[j==n]);
        i=j;
    }
    return 0;
}
