#include<bits/stdc++.h>
using namespace std;
int main()
{
    int x,a[105],i=0;
    char c;
    while(cin>>x)
    {
        cin.ignore();
        a[i++]=x;
    }
    sort(a,a+i);
    int l=a[0],r=a[0];
    for(int j=1;j<i;++j)
    {
        if(a[j]==a[j-1])continue;
        else if(a[j]==a[j-1]+1)r=a[j];
        else
        {
            if(l==r)cout<<l<<',';
            else cout<<l<<'-'<<r<<',';
            l=r=a[j];
        }
    }
    if(l==r)cout<<l;
    else cout<<l<<'-'<<r;

}
