#include<cstdio>
#include<iostream>
#include<cstdlib>
#include<cstring>

using namespace std;

bool a[10000];

int main(){
	char c;
	int x, mx = 0;
	memset(a, 0, sizeof(a));
	while(1){
		scanf("%d", &x);
		a[x] = 1;
		if(mx < x)
			mx = x;
		if(scanf("%c", &c) <= 0)
			break;
	}
	for(int i = 1; i <= mx;){
		int start = i, end;
		while(!a[start])
			start++;
		end = start;
		while(a[end])
			end++;
		end--;
		if(end > start){
			cout << start << "-" << end;
		}
		else
			cout << start;
		if(end < mx)
			cout << ",";
		i = end+1;
	}

	return 0;
}
