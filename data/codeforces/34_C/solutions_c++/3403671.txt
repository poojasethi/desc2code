#include<cstdio>
#include<algorithm>
using namespace std;
int a[1002];
int main(){
	int i=0,j,n,l,r;
	while(~scanf("%d",&a[i++])){
		if(getchar()!=',')break;
	}n=i;
	sort(a,a+n);
	for(i=1,j=1;i<n;i++)if(a[i]!=a[i-1])a[j++]=a[i];
	n=j,l=r=a[0];
	int s=0;
	bool f=0;
	for(i=1;i<n;i++){
		if(a[i]==a[i-1]+1){s=1;r++;}
		else {
			if(f)printf(",");
			if(l!=r)printf("%d-%d",l,r);
			else printf("%d",l);
			l=r=a[i];s=2;f=1;
		}
	}
	if(f)printf(",");
	if(l!=r)printf("%d-%d\n",l,r);
	else printf("%d\n",l);
	return 0;
}
