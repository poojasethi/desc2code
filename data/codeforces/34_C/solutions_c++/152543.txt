#include <cstdio>
int v[2005];
int main()
{
	int x,i=1,q,first;
	for (scanf("%d,",&x);!feof(stdin);scanf("%d,",&x))
	{
		v[x]=1;
	}
	q=0;
	first=1;
	for (i=1;i<=1000;i++)
	{
		if (!v[i])
			continue;
		if (!q)
		{
			if (!first)
				printf(",");
			else
				first=0;
			printf("%d",i);
			if (v[i+1])
				q=1;
			continue;
		}
		if (v[i+1])
			continue;
		q=0;
		printf("-%d",i);
		first=0;
	}
	return 0;
}
