#include<stdio.h>
#include<algorithm>
using namespace std;
    
int a[100];
int main(){
    char s;
    int n,m,i,j,k,t,l,r,sum;
    for(n=0;scanf("%d%c",&a[n++],&s)&&s!='\n';);
    sort(a,a+n);
    for(t=l=a[0],i=1;i<n;i++){
        if(a[i]>t+1){
            r=t;
            if(r>l) printf("%d-%d,",l,r);
            else printf("%d,",l);
            l=a[i];
        }
        t=a[i];
    }
    r=t;
    if(r>l) printf("%d-%d\n",l,r);
    else printf("%d\n",l);
}
