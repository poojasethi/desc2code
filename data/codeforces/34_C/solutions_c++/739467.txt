#include<cstdio>
#include<algorithm>
#define maxn 10001
bool is[maxn];
int main()
{
	int i,a;
	char s[maxn];
	gets(s);
	i=0;
	while(1)
	{
		a=s[i++]-'0';
		while(s[i]!=',')
		{
			if(s[i]==0) break;
			a=a*10+s[i]-'0'; i++;
		}
		is[a]=1;
		if(s[i]==0) break;
		else i++;
	}
	int cnt=0;
	bool f=0;
	for(i=1;i<=1001;i++)
	{
		if(is[i])
		{
			cnt++;
		}
		else if(cnt)
		{
			if(f) printf(",");
			if(cnt==1) printf("%d",i-1);
			else if(cnt>1) printf("%d-%d",i-cnt,i-1);
			cnt=0;
			f=1;
		}
	}
	puts("");
	return 0;
}
