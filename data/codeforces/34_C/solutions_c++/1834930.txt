#include <cstdio>
int a[1005];
int main ( ) {
        int x, i, j;
        while ( scanf ( "%d%*c", &x ) != EOF ) a[x] = 1;
        bool flag = false;
        for ( i = j = 1; i < 1001; i = ++j )
                if ( a[i] ) {
                        for ( j = i; a[j]; ++j );
                        --j;
                        if ( !flag ) flag = true;
                        else printf ( "," );
                        if ( i == j ) printf ( "%d", i );
                        else printf ( "%d-%d", i, j );
                }
        printf ( "\n" );
}
