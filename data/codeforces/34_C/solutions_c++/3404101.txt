#include<stdio.h>
int main()
{
	int a[1001],b[1001],n=0,i,j;
	while(scanf("%d",&a[n])!=EOF)
	{
		n++;
		getchar();
	}
	int t;
	for(i=0;i<n;i++)
	{
		t=a[i];
		for(j=i-1;j>=0;j--)
		{
			if(a[j]>t)
				a[j+1]=a[j];
			else
				break;
		}
		a[j+1]=t;
	}
	b[0]=a[0];
	int k=0;
	for(i=1;i<n;i++)
	{
		if(a[i]>b[k])
		{
			k++;
			b[k]=a[i];
		}
	}
	int m=1;
	for(i=0;i<k;i++)
	{
		if(b[i+1]-b[i]>1)
		{
			printf("%d,",b[i]);
			m=1;
		}
		if(b[i+1]-b[i]==1&&m)
		{
			printf("%d-",b[i]);
			m=0;
		}
	}
	printf("%d\n",b[k]);
	return 0;
}
