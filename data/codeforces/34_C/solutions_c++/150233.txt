#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int a[10000];

int main()
{
	char c;
	int x;
	int max=0;
	while (1)
	{
		scanf("%d", &x);
		a[x] = 1;
		if (max < x) max = x;
		if (scanf("%c", &c) <= 0)
			break;
	}
	for (int i=1; i<=max; )
	{
		int s = i,e;
		while (!a[s]) s++;
		e = s;
		while (a[e]) e++;
		e--;
		if (e > s)
			printf("%d-%d", s, e);
		else
			printf("%d", s);
		if (e < max)
			printf(",");
		i = e+1;
	}
	return 0;
}
