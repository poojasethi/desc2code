#include <stdio.h>
#include <string.h>

int app[1005];

int main()
{
	//freopen("c.in", "r", stdin);

	int i, n;
	char ch;
	while (scanf("%d", &n) != EOF)
	{
		app[n] = 1;
		ch = getchar();
		if (ch != ',') break;		
	}
	int l = 0, r, flag = 0, fir = 1;
	for (i = 0; i <= 1001; i++)
	{
		if (flag && !app[i])
		{
			if (!fir) printf(",");
			else fir = 0;
			if (l == r) printf("%d", r);
			else printf("%d-%d", l, r);
			flag = 0;
		}
		else if (app[i])
		{
			if (flag == 0) l = r = i, flag = 1;
			else r = i;
		}
	}
	printf("\n");
	return 0;
}
