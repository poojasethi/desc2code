#include <bits/stdc++.h>
using namespace std;
int a[110];
int main()
{
    char s;
    int i;
    for(i=0;;i++)
    {
        scanf("%d%c",&a[i],&s);
        if(s=='\n') break;
    }
    sort(a,a+i+1);
    int start=a[0];
    int l=a[0],r;
    for(int j=1;j<=i;j++)
    {
        if(a[j]>start+1)
        {
            r=start;
            if(r>l) printf("%d-%d,",l,r);
            else printf("%d,",l);
            l=a[j];
        }
        start=a[j];
    }
    r=start;
    if(r>l) printf("%d-%d\n",l,r);
    else printf("%d\n",l);
    return 0;
}
