#include<stdio.h>
#include<string.h>

#define MAX 1000

int cou[MAX+1];

int main()
{
	int a,i,li,ri;
	char c;
	memset(cou,0,sizeof(cou));
	while(scanf("%d%c",&a,&c)==2)
	{
		cou[a]=1;
		if(c=='\n')
			break;
	}
	for(a=i=0; i<=MAX ; )
	{
		if(cou[i]==1)
		{
			li=i;
			if(a==1)
				printf(",");
			else
				a=1;
			printf("%d",i);
			++i;
			while(i<=MAX)
			{
				if(cou[i]==0)
					break;
				++i;
			}
			ri=i-1;
			if(ri>li)
				printf("-%d",ri);
		}
		else
			++i;
	}
	putchar('\n');
	//while(1);
	return 0;
}
