#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
#include<cmath>
#include<string>
#include<map>
#include<vector>
using namespace std;
int main()
{
	int page[1010];
	memset(page,0,sizeof(page));
	int n,maxx=-1;
	while(~scanf("%d",&n))
	{
		if(n>maxx) maxx=n;
		page[n]++;
		getchar();
	}
	for(int i=1;i<=maxx;i++)
	{
		if(page[i])
		{
			cout<<i;
			int j;
			for(j=i+1;j<=maxx&&page[j];j++);
			if(j-1!=i)
				cout<<"-"<<j-1;
			i=j;
			if(i<=maxx)
				cout<<',';
		}
	}
	cout<<endl;
}
