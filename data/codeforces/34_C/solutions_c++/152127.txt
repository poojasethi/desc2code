#include<iostream>
using namespace std;
int f[1001],i,a,x,n;
char c;
int main()
{
	cin>>x;
	f[x]=1;
	while(!cin.eof())
	{
cin>>c;
cin>>x;
f[x]=1;
	}
	for(i=1000;i>=0;i--)
		if(f[i]==1)
		{
			n=i;
			break;
		}
i=1;
	while(i<=n)
		{
		while(f[i]==0 && i<=n)
			i++;
		a=i;
		while(f[i]==1 && i<=n)
		
			i++;
		if((i-a)==1)
		{
			cout<<a;
			if(a!=n)
				cout<<',';
		}
		else
		{
			cout<<a;
			cout<<'-';
			cout<<i-1;
			if((i-1)!=n)
				cout<<',';
			
		}			
		}
}
