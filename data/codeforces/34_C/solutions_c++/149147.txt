#include<cstdio>
#define MAX 1010

int hash[MAX];
char str[MAX];
int main()
{
    int a,max=0;
    while(scanf("%d,",&a)==1)
    {
        hash[a]=1;
        if(max<a)max=a;
    }
    for(a=0;a<=max;a++)
    {
        if(hash[a])
        {
            printf("%d",a);
            if(hash[a+1])
            {
                printf("-");
                a++;
                while(hash[a])a++;
                printf("%d",a-1);       
            }
            if(a<max)
                printf(",");
        }
    }
    puts("");
    return 0;
}
