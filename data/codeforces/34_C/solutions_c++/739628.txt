#include<stdio.h>
#include<string.h>
int ans[1010];
int main() {
	char c;
	int n=-1,m=0;
	while(scanf("%c",&c),c!='\n') {
		if(c>='0'&&c<='9') 
	    	m=m*10+(c-'0');
		else {
			ans[m]++;
			m=0;
		}
	}
	ans[m]++;
	bool tmp=false;
	for(n=0;n<=1000;) {
		m=n;
		while(ans[n]) n++;
		if(m==n) n++;
		else {
			if(!tmp) {
				if(n==m+1) printf("%d",m);
				else printf("%d-%d",m,n-1);
			}
			else {
				if(n==m+1) printf(",%d",m);
				else printf(",%d-%d",m,n-1);
			}
			tmp=true;
		}
	}
	puts("");
	return 0;
}
