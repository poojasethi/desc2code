#include <stdio.h>
#include <string.h>
long i,j,k,l,n,m;
bool ans[2000];
char s[200];
long getNum(){
  long q=0;
  while(i!=n&&s[i]!=','){
    q*=10;
    q+=(s[i]-48);
    i++;
  }
  return q;
}
int main() {
//  freopen("1.1","r",stdin);
  scanf("%s",&s);
  n=strlen(s);
  i=0;
  while(i<n){
    k=getNum();
    i++;
    ans[k]=true;
  }
  for(i=0;i<=1000;i++){
    if(ans[i]){
      j=i++;
      while(ans[i]) i++;
      if (l>0)printf(",");
      else l++;
      if(i-1!=j)printf("%d-%d",j,i-1);
      else printf("%d",j);
    }
  }
  return 0;
}
