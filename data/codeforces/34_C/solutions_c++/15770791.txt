#include <iostream>

using namespace std;

const int MAXN = 1000 + 100;
bool mrk[MAXN];

int32_t main()
{
	int num;
	while (cin >> num)
	{
		char c;
		cin >> c;
		mrk[num] = 1;
	}
	bool h = 0;
	for (int i = 1; i < MAXN; i++)
	{
		cerr << mrk[i] << ' ';
		if (mrk[i])
		{
			int pt = i;
			while (mrk[pt])
				pt++;
			if (h)
				cout << ',';
			if (pt == i + 1)
				cout << i;
			else
				cout << i << '-' << pt - 1;
			i = pt - 1;
			h = 1;
		}
	}
	cout << endl;
	return 0;
}
