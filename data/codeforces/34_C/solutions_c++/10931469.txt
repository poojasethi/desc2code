#include<stdio.h>
#include<string.h>
int i,j,max,f=false,arr[10000],knt[22000],n,temp;
char s[10000];


int main(){
	n=1;
	while((scanf("%d,",&knt[n]))==1)	n++;
	for(i=1;i<=n;i++)
		arr[knt[i]]=1;
	for(i=1000;i>=1;i--)
		if(arr[i]){
			 max=i;
			break;
		}

	for(i=1;i<=1000;i++)
		if(arr[i]){
			for(j=i+1;j<=1000;j++)
				if(!arr[j])
					break;				
				else
					f=true,temp=j;

			if(f==true){
				printf("%d-%d",i,temp),i=temp;
				if(temp!=max) printf(",");
			}
				else{
				printf("%d",i);
				if(i!=max) printf(",");
			}

			f=false;
		}
	return 0;			
}
