#include <iostream>
#include <set>

using namespace std;

int main(int argc, char* argv[])
{
	int n;
	set<int> s;
	
	while (cin >> n)
	{
		s.insert(n);
		
		char c;
		cin >> c;
	}
	
	set<int>::iterator i = s.begin();
	
	while (i != s.end())
	{
		set<int>::iterator j = i;
		set<int>::iterator k = i;
		k++;
		
		while ((*k - *j == 1) && (k != s.end()))
		{
			j++;
			k++;
		}
		
		if (*i != *j)
			cout << *i << '-' << *j;
		else
			cout << *i;
		
		if (k != s.end())
			cout << ',';
		
		i = k;
	}
	
	cout << endl;
	
	return 0;
}
