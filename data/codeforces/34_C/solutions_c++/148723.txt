#include<stdio.h>
#include<cstring>
int hash[1005];
int main()
{
    int n,i,t,first=1;
    while(scanf("%d",&n)!=EOF)
    {
        getchar();
        hash[n]++;
    }
    first=1;
    for(i=1;i<=1000;i++)
    {
        if(hash[i])
        {
            if(first)
            {
                printf("%d",i);
                first=0;
            }
            else printf(",%d",i);
            if(hash[i+1])
            {
                while(hash[i+1])i++;
                printf("-%d",i);
            }
        }
    }
    printf("\n");
    return 0;
}
