#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int a[1002];
int main()
{
	int i, j, k=0;
	scanf("%d",&i);
	a[i]=1;
	while (scanf(",%d",&i)==1) a[i]=1; 
	for (i=1; i<=1000; i++) {
		if (a[i]&&a[i-1]==0) j=i;
		if (a[i]&&a[i+1]==0) {
			if (k) cout<<",";
			else k=1;
			if (i==j) cout<<i;
			else cout<<j<<"-"<<i;
		} 
	}
	cout<<endl;
	return 0;
}
