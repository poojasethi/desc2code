#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
bool in[1010];
int main() {
	int x,f(0);
	while(~scanf("%d,",&x))
		in[x] = true;
	for(int i(1);i!=1010;)
		if(in[i]) {
			int j(i+1);
			while(in[j]) ++j;
			if(j!=i+1) cout<<(f?",":(f = 1,""))<<i<<'-'<<j-1;
			else cout<<(f?",":(f = 1,""))<<i;
			i = j;
		} else ++i;
	return 0;
}
