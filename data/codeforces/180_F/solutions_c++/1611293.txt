#include <iostream>

using namespace std;

int main()
{
	int n;
	cin >> n;
	int a[n], b[n], p[n];
	for(int i = 0; i < n; i++)
		cin >> a[i];
	for(int i = 0; i < n; i++)
		cin >> b[i];
	for(int i = 0; i < n; i++)
		p[a[i] - 1] = b[i];
	for(int i = 0; i < n; i++)
		cout << p[i] << " ";
	cout << endl;
	return 0;
}
