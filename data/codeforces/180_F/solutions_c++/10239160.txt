#include <iostream>
using namespace std;
const int nmax = 1e5+5;
int pos[nmax];
int main(){
	ios_base::sync_with_stdio(false);
	int n;
	cin >> n;
	for(int i = 0; i < n; ++i){
		int a;
		cin >> a;
		pos[a]=i+1;
	}
	for(int i = 0; i < n; ++i){
		int a;
		cin >> a;
		cout << pos[a] << " \n"[i==n-1];
	}
}
