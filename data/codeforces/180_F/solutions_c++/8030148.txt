#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int hash[n+1];
	int third[n+1];
	int fourth[n+1];
	for(int i=1;i<=n;i++)
	{
		cin>>third[i];
		hash[third[i]]=i;
	}
	for(int i=1;i<=n;i++)
		cin>>fourth[i];
	for(int i=1;i<=n;i++)
	{
		cout<<hash[fourth[i]]<<" ";
	}
	return 0;
}
