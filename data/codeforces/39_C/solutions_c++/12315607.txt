#include<cstdio>
#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
using namespace std;

struct range
{int x,y,num;
};
struct range a[3010];

int n,n1,inf=2e9,flag;
int sa[20010],f[10010],g[10010],w[10010],next[2010][2010];

bool cmp(range aa,range bb)
{return aa.x<bb.x || (aa.x==bb.x && aa.y>bb.y);
}

void print(int i)
{int j;
 
 j=next[i][i];
 while(j!=0)
 {	if(flag)
 		printf(" ");
 	else
	 	flag=1;	
 	printf("%d",a[j].num);
	print(j);
 	j=next[i][j];
 }
}
 	
int main()
{int i,j,k,x,y;
 
 scanf("%d",&n);
 
 for(i=1;i<=n;i++)
 {	scanf("%d%d",&x,&y);
 	a[i].x=x-y;
 	a[i].y=x+y;
 	a[i].num=i;
 	n1++;
 	sa[n1]=a[i].x;
 	n1++;
 	sa[n1]=a[i].y;
 }
 a[n+1].x=-inf;
 a[n+1].y=inf;
 sa[n1+1]=-inf;
 sa[n1+2]=inf;
 n1+=2;
 
 sort(sa+1,sa+n1+1);
 
 for(i=1;i<=n+1;i++)
 {	a[i].x=lower_bound(sa+1,sa+n1+1,a[i].x)-sa;
 	a[i].y=lower_bound(sa+1,sa+n1+1,a[i].y)-sa;
 }
 
 sort(a+1,a+n+1+1,cmp);
 
 //for(i=1;i<=n+1;i++)
 //	cout<<a[i].x<<" "<<a[i].y<<endl;
 
 for(i=n+1;i>=1;i--)
 {	j=n+1;
 	while(a[j].x>a[i].y && j>=1) j--;
	//cout<<i<<endl;
	g[a[i].y+1]=0;
	w[a[i].y+1]=0;
	for(k=a[i].y;k>=a[i].x;k--)
 	{	g[k]=g[k+1];
 		w[k]=w[k+1];
 		while(a[j].x==k && j>=1)
 		{	if(a[j].y<=a[i].y)
 			{	if(f[j]+g[a[j].y]>g[k])
		 		{	g[k]=f[j]+g[a[j].y];
		 			w[k]=j;
		 			next[i][j]=w[a[j].y];
		 		}
		 	}
			j--;
		}
	}
	f[i]=g[a[i].x]+1;
	next[i][i]=w[a[i].x];
 }
 
 printf("%d\n",f[1]-1);
 
 print(1);
 
 printf("\n");
 
 return 0;
}
