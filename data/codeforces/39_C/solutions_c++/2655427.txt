#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
#define mp make_pair
#define pb push_back
using namespace std;
int f[4011][4011],b[4011][4011],l[2011],r[2011],n,i,j,x,y,num,ans;
pair <int,int> a[4011];
vector<int> v[4011];
int dp(int ll,int rr)
{
//	cout<<ll<<' '<<rr<<endl;
	if (f[ll][rr]!=-1) return f[ll][rr];
	if (ll==rr){
		f[ll][rr]=0;
		return f[ll][rr];
	}
	int zz=(b[ll][rr]!=0);
	//	if (rr==3) cout<<' '<<zz<<endl;
	int ans=zz+dp(ll,rr-1);
	//cout<<ll<<' '<<rr<<endl;
	for(int i=0;i<v[rr].size();i++)
	{
		int xx=l[v[rr][i]];
		if (xx<=ll) continue;
		int tt=zz+dp(ll,xx)+dp(xx,rr);
		if (tt>ans) ans=tt;
	}
	
	f[ll][rr]=ans;
	return f[ll][rr];
}
void fin(int ll,int rr)
{
//	cout<<ll<<' '<<rr<<endl;
	if (ll==rr) return;
	int zz=(b[ll][rr]!=0);
	if (b[ll][rr]>0)
	{
		zz=1;
		//an[++sw]=b[ll][rr];	
		printf("%d ",b[ll][rr]);
	}
	if (f[ll][rr]==zz+f[ll][rr-1])
	{
	fin(ll,rr-1);
	return;
	}
	for(i=0;i<v[rr].size();i++)
	{
		int xx=l[v[rr][i]];
		if (xx<=ll) continue;
		int tt=zz+f[ll][xx]+f[xx][rr];
		if (f[ll][rr]==tt){
			fin(ll,xx); fin(xx,rr);
			return;
		}
	}
}
int main()
{
//	freopen("1.in","r",stdin);
//	freopen("1.out","w",stdout);
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&x,&y);
		a[i*2-1]=mp(x-y,i*2-1);
		a[i*2]=mp(x+y,i*2);
	}
	sort(a+1,a+n*2+1);
	a[0].first=-11111;
	for(i=1;i<=n*2;i++)
	{
		if (a[i].first!=a[i-1].first)
		num++;
		if (a[i].second%2==1) l[(a[i].second+1)/2]=num;
		else r[a[i].second/2]=num;
	}
	for(i=1;i<=n;i++)
	v[r[i]].pb(i);
	for(i=1;i<=n;i++)
	{
	b[l[i]][r[i]]=i;
	}
	for(i=1;i<=n*2;i++)
		for(j=1;j<=n*2;j++)
		f[i][j]=-1;
	ans=dp(1,num);
	printf("%d\n",ans);
	fin(1,num);
}
