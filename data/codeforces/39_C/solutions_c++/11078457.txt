#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#include<vector>
#define N (2005)
#define M (2*N)
#define LL long long
#define oo (1<<30)
using namespace std;
vector<int> E[M];
int n,m,d[N],r[N],a[M];
short f[M][M],w[M][M],ex[M][M],s[M][M];
int map(int x){ return lower_bound(a+1,a+m+1,x)-a;}
void print(int l,int r)
{
    if(l>=r||!l||!r) return ;
    if(s[l][r]) printf("%d ",s[l][r]);
    if(r-l==1) return ;
    print(l,w[l][r]);
    print(w[l][r],r);
}
int main()
{
    int i,j,k,x,y; short o,u;
    scanf("%d",&n);
    for(i=1;i<=n;i++){
        scanf("%d %d",&d[i],&r[i]);
        a[++m]=d[i]-r[i];
        a[++m]=d[i]+r[i];
      }
    sort(a+1,a+m+1);
    m=unique(a+1,a+m+1)-a-1;
    for(i=1;i<=n;i++){
        x=map(d[i]-r[i]);
        y=map(d[i]+r[i]);
        ex[x][y]=i;
        E[x].push_back(y);
      } 
    for(x=1;x<=m;x++)
        for(i=1;i<=m-x;i++){
            j=i+x,f[i][j]=o=(ex[i][j]> 0);
            if(ex[i][j]) s[i][j]=ex[i][j];
            if(f[i+1][j]+o>f[i][j])
                f[i][j]=f[i+1][j]+o,w[i][j]=i+1;
            for(k=0;k<E[i].size();k++){
                if(E[i][k]>=j) continue;
                u=f[i][E[i][k]]+f[E[i][k]][j]+o;
                if(u>f[i][j]) f[i][j]=u,w[i][j]=E[i][k];
              }
          }
    printf("%d\n",f[1][m]);
    print(1,m);
    return 0;
}
