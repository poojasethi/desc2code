#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

struct st
{
	int l,r,p;
	bool operator <(const st&B) const {return r<B.r||(r==B.r&&l>B.l);}
} a[4005],b[4005];
int N,M,f[4005],g[4005],h[4005],q[2005][2005];

void print(int p)
{
	if (a[p].p) printf("%d ",a[p].p);
	for (int i=1; i<=q[p][0]; i++) print(q[p][i]);
}

int main()
{
	scanf("%d",&N);
	for (int i=1,d,r; i<=N; i++) scanf("%d%d",&d,&r),b[i].r=d-r,b[i].p=i,b[i+N].r=d+r,b[i+N].p=i+N,a[i].p=i;
	sort(b+1,b+2*N+1);
	for (int i=1; i<=2*N; i++)
	{
		if (i==1||b[i].r!=b[i-1].r) M++;
		if (b[i].p<=N) a[b[i].p].l=M; else a[b[i].p-N].r=M;
	}
	a[++N].l=1,a[N].r=M,sort(a+1,a+N+1);
	for (int i=1; i<=N; i++)
	{
		memset(g,0,sizeof(g)),memset(h,0,sizeof(h));
		for (int j=a[i].l,k=1; j<=a[i].r; j++)
		{
			g[j]=g[j-1],h[j]=h[j-1];
			for (;k<i&&a[k].r<j; k++);
			for (;k<i&&a[k].r==j; k++) if (a[k].l>=a[i].l) if (g[a[k].l]+f[k]>g[j]) g[j]=g[a[k].l]+f[k],h[j]=k;
		}
		f[i]=g[a[i].r]+1;
		for (int j=a[i].r; h[j]; j=a[h[j]].l) q[i][++q[i][0]]=h[j];
	}
	printf("%d\n",f[N]-1),print(N);
}
