# -*- coding: utf-8 -*-

if __name__ == '__main__':
    n = int(raw_input())
    aa = map(int, raw_input().split())
    if n == 1:
        if aa[0] == 1:
            result = 'YES'
        else:
            result = 'NO'
    else:
        if sum(aa) == n-1:
            result = 'YES'
        else:
            result = 'NO'
    print(result)
