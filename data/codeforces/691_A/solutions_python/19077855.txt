import sys


if __name__ == "__main__":
    n = int(sys.stdin.readline().strip())
    buttons = [int(i) for i in sys.stdin.readline().strip().split()]
    if len(buttons) == 1 and buttons[0] == 1:
        print "YES"
    elif len(buttons) != 1 and sum(buttons) == len(buttons) - 1:
        print "YES"
    else:
        print "NO"
