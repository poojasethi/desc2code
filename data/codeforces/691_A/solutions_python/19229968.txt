n=int(raw_input())
a=map(int,raw_input().split(" "))
if n==1:
    if a[0]==1:
        print "YES"
    else:
        print "NO"
else:
    count=0
    for i in a:
        if i==0:
            count+=1
        if count>1:
            break
    if count==1:
        print "YES"
    else:
        print "NO"
