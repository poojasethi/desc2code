n = int(raw_input())
s = raw_input()
s = s.split()
counter = 0
if n ==1 and s[0] == '1':
    print 'YES'
if n ==1 and s[0] == '0':
    print 'NO'
elif n > 1 :
    for x in s:
        if x == '0':
            counter += 1
        if counter > 1:
            break
    if counter > 1 or counter ==0:
        print 'NO'
    else:
        print 'YES'
