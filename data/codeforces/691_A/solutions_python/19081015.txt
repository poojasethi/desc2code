n = input()
i = map(int, raw_input().split())

not_fastened = 0
for j in range(0, n):
    if i[j] == 0:
        not_fastened += 1
    
if not_fastened == 0 and n == 1:
    print "YES"
elif not_fastened == 1 and n == 1:
    print "NO"
elif not_fastened == 1:
    print "YES"
else:
    print "NO"
