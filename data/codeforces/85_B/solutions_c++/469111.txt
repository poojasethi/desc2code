#include <iostream>
#include <cstdio>
#include <string>
#include <algorithm>
using namespace std;
#define  LLint  __int64

int n;
LLint k[3], t[3], a[3][100010], ans;

int main()
{
	LLint x, y;
	scanf("%I64d%I64d%I64d", &k[0], &k[1], &k[2]);
	scanf("%I64d%I64d%I64d", &t[0], &t[1], &t[2]);
	scanf("%d", &n);
	ans = 0;
	for(int i = 0; i < n; i ++)
	{
		scanf("%I64d",&x); y = x;
		for(int j = 0; j < 3; j ++)
			x = a[j][i%k[j]] = max(a[j][i%k[j]], x) + t[j];
		ans = max(ans, x - y);
	}
	printf("%I64d\n", ans);

	//system("pause");
	return 0;
}
