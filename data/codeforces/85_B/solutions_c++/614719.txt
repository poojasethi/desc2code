#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int max_n = (int) 1e5;
int c[max_n];
long long wait[max_n], windows[max_n];

int main() {
	int n, k[3], t[3];
	for (int i = 0; i < 3; ++i)
		scanf("%d", &k[i]);
	for (int i = 0; i < 3; ++i)
		scanf("%d", &t[i]);
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &c[i]);
		wait[i] = c[i];
	}
	sort(c, c + n);
	for (int i = 0; i < 3; ++i) {
		int s = min(n, k[i]);
		for (int j = 0; j < s; ++j)
			windows[j] = 0;
		for (int j = 0, fj = 0; j < n; ++j, fj = (fj + 1) % s)
			wait[j] = windows[fj] = max(wait[j], windows[fj]) + t[i];
	}
	long long res = 0;
	for (int i = 0; i < n; ++i)
		res = max(res, wait[i] - c[i]);
	printf("%I64d", res);
	return 0;
}
