#include <cstdio>
#include <iostream>
#include <cmath>
#include <stack>
#include <string>

using namespace std;

typedef long long i64;

int k[3], t[3], n;
i64 a[3][100010], R;

int main() 
{    
	cin >> k[0] >> k[1] >> k[2] >> t[0] >> t[1] >> t[2] >> n;
	for (int i = 0; i < n; ++i)
	{
		i64 c, f;
		cin >> c;
		f = c;
		for (int j = 0; j < 3; ++j)
			c = a[j][i%k[j]] = max(a[j][i%k[j]],  c) + t[j];
		R = max(R, c - f);
	}
	cout << R;
	return 0;
}
