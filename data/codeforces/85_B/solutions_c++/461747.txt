#include <cstdio>
#include <iostream>

using namespace std;

const int MaxN= 100000 + 10;
typedef long long ll;

ll n[3];
ll t[3];
int num;
ll timee[MaxN], start[MaxN];

int main(){
	cin >> n[0] >> n[1] >> n[2];
	cin >> t[0] >> t[1] >> t[2];
	cin >> num;
	for (int i=0 ; i<num ; i++){
		int k;
		scanf("%d", &k);
		start[i]= timee[i]= k;
	}
	for (int j=0 ; j<3 ; j++){
		for (int i=0 ; i<num ; i++){
			int back= i-n[j];
			if (back<0 || timee[back]<=timee[i])
				timee[i]= timee[i] + t[j];
			else
				timee[i]= timee[back] + t[j];
		}
	}
	ll max= 0;
	for (int i=0 ; i<num ; i++)
		if (max<timee[i]-start[i])
			max= timee[i]-start[i];
	cout << max << endl;
	return 0;
}
