#include<cstdio>
inline long long max(long long A, long long B){return A > B? A : B;}
long long now1[100001], now2[100001], now3[100001];
int K1, K2, K3;
long long T1, T2, T3;
int N;
long long C[100001], lst[100001];
int main(){
	scanf("%d%d%d", &K1, &K2, &K3);
	scanf("%I64d%I64d%I64d", &T1, &T2, &T3);
	scanf("%d", &N);
	for(int i = 0; i < N; i++)
	    scanf("%I64d", &C[i]);
	for(int i = 0, j = 0; i < N; i++, j++){
		if(j == K1)j -= K1;
		now1[j] = lst[i] = max(now1[j], C[i]) + T1;
	}
	for(int i = 0, j = 0; i < N; i++, j++){
		if(j == K2)j -= K2;
		now2[j] = lst[i] = max(now2[j], lst[i]) + T2;
	}
	for(int i = 0, j = 0; i < N; i++, j++){
		if(j == K3)j -= K3;
		now3[j] = lst[i] = max(now3[j], lst[i]) + T3;
	}
	printf("%I64d\n", lst[N - 1] - C[N - 1]);
	//scanf(" ");
}
