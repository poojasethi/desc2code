#include "iostream"
#include "string"
#include "algorithm"
#include "stdio.h"
#include "stdlib.h"
#include "string.h"
#include "vector"
using namespace std;

vector<long long> g;

void go(int k, long long t)
{
	long long time[100005]={};
	int cur=0;
	for(int i=0;i<g.size();i++)
	{
		g[i]=max(g[i],time[cur])+t;
		time[cur]=g[i];
		cur=(cur+1)%k;
	}
}

int main()
{
	int k1,k2,k3;
	cin>>k1>>k2>>k3;
	int t1,t2,t3;
	cin>>t1>>t2>>t3;
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		long long a;
		cin>>a;
		g.push_back(a);
	}
	vector<long long> gg=g;
	go(k1,t1);
	go(k2,t2);
	go(k3,t3);
	long long ans=0;
	for(int i=0;i<n;i++)
	{
		long long diff=g[i]-gg[i];
		ans=max(ans,diff);
	}
	cout<<ans<<endl;
	return 0;
}
