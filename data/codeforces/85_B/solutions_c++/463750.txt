#include <stdio.h>
#include <algorithm>
using namespace std;
int v[100005];
long long c[100005];
int main() {
  int k[3];
  int t[3];
  int n;
  scanf("%d%d%d%d%d%d", k+0, k+1, k+2, t+0, t+1, t+2);

  scanf("%d", &n);
  for(int i=0; i<n; ++i) scanf("%d",v+i), c[i]=v[i];
  for(int i=0; i<3; ++i) {

    for(int offset=0; offset<n; offset+=k[i]) {
      for(int j=0; j<k[i] && j+offset<n; ++j) {
        int id = offset+j;
        if(offset==0) c[id]+=t[i];
        else if(c[id] > c[id-k[i]]) c[id]+=t[i];
        else c[id]=c[id-k[i]]+t[i];
      }
    }
  }

  long long res=0;
  for(int i=0; i<n; ++i) if(res < c[i]-v[i]) res=c[i]-v[i];
  printf("%I64d\n", res);
  return 0;
}
