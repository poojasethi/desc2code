#include <cstdio>
#include <algorithm>
#include <cstring>
#include <queue>
#define LL long long int
#define FOR(i,s,e) for (int i=(s); i<(int)(e); i++)
#define FOE(i,s,e) for (int i=(s); i<=(int)(e); i++)
using namespace std;

int num[3], ser[3], n;
LL arr[100005], x, t, ret;
priority_queue<LL> q[3];

int main(){

	for (int i=0; i<3; i++) scanf("%d", &num[i]);
	for (int i=0; i<3; i++) scanf("%d", &ser[i]);
	
	scanf("%d", &n);
	for (int i=0; i<n; i++){
		scanf("%I64d", &arr[i]);
		t = arr[i];
		
		for (int j=0; j<3; j++){
			if (q[j].size() < num[j]){
				t += ser[j];
				q[j].push(-t);
			}
			else{
				x = -q[j].top();
				q[j].pop();
				t = max(t, x) + ser[j];
				q[j].push(-t);
			}
		}
		
		ret = max(ret, t - arr[i]);
	}
	
	printf("%I64d\n", ret);

	return 0;
}
