#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>

using namespace std;
typedef long long ll;

const int N = 100010;

int k[3], t[3];

int n;

ll c[N], tm[N], d[N];

int main(){
	for(int i = 0; i < 3; i++)
		scanf("%d", &k[i]);
	for(int i = 0; i < 3; i++)
		scanf("%d", &t[i]);
	scanf("%d", &n);
	ll ret = 0;
	for(int i = 0; i < n; i++)
		scanf("%I64d", &c[i]), d[i] = c[i];
	sort(c, c + n);
	sort(d, d + n);
	for(int i = 0; i < 3; i++){
		memset(tm, 0, sizeof(tm));
		for(int j = 0; j < n; j++){
			int x = j % k[i];
			tm[x] = max(d[j], tm[x]) + t[i];
			d[j] = tm[x];
			ret = max(d[j] - c[j], ret);
		}
	}
	cout << ret << endl;
	return 0;
}
