#include <iostream>
#include <cstdio>
#include <cstdlib>
#define maxn 100005
using namespace std;

int c[maxn];
long long a[maxn];
int n, k[3], t[3];

int main()
{
	for (int i = 0; i < 3; ++i)	cin >> k[i];
	for (int i = 0; i < 3; ++i)	cin >> t[i];
	cin >> n;
	for (int i = 0; i < n; ++i) {scanf("%d", &c[i]); a[i] = c[i];}
	for (int p = 0; p < 3; ++p) {
		int l = 0;
		for (; l < k[p] && l < n; ++l) a[l] += t[p];
		for (; l < n; ++l) {
			long long x = a[l - k[p]];
			if (x > a[l]) a[l] = x;
			a[l] += t[p];
		}
	}
	long long ans = 0;
	for (int i = 0; i < n; ++i) if (a[i] - c[i] > ans) ans = a[i] - c[i];
	cout << ans;
	return 0;
}
