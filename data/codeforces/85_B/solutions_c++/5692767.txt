// In the name of God
#include <iostream>
#include <vector>
#include <algorithm>
#include <string>

using namespace std;
typedef long long ll;
const int maxn = 100000 + 20;
ll n , k[3] , t[3] , ans , c[maxn] , a[3][maxn];
int main()
{
  for (int i = 0 ; i < 3 ; i++)
    cin >> k[i];
  for (int j = 0 ; j < 3 ; j++)
    cin >> t[j];
  cin >> n;
  for (int i = 0 ; i < n ; i++) cin >> c[i];
  sort (c , c + n);
  for (int i = 0 ; i < n ; i++){
    ll cur = c[i];
    for (int j = 0 ; j < 3 ; j++){
      cur = max (cur , a[j][i % k[j]]) + t[j];
      a[j][i % k[j]] = cur;
    }
    if (i == n - 1)
      ans = cur - c[i];
  }
  cout << ans << endl;
  return 0;
}
