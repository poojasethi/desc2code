#include <stdio.h>
#include <queue>
using namespace std;
__int64 star[100010],end[100010];
int k1,k2,k3;
int t1,t2,t3;
int n;

void f(int a,int b)
{
	int i;
	queue<__int64> q;
	for(i=0;i<n;i++)
	{
		while(!q.empty()&&q.front()<=end[i])
			q.pop();
		if(q.size()<a)  {end[i]=end[i]+b;q.push(end[i]);}
		else 
		{
			end[i]=q.front()+b;
			q.pop();
			q.push(end[i]);
		}
	}
}

int main()
{
	int i;
	__int64 ma;
	while(scanf("%d%d%d",&k1,&k2,&k3)!=EOF)
	{
		scanf("%d%d%d",&t1,&t2,&t3);
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%I64d",&star[i]);
			end[i]=star[i];
		}
		f(k1,t1);
		f(k2,t2);
		f(k3,t3);
		ma=0;
		for(i=0;i<n;i++)
			if(end[i]-star[i]>ma)  ma=end[i]-star[i];
		printf("%I64d\n",ma);
	}
	return 0;
}
