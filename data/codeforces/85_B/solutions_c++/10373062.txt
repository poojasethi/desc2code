#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int k[4];
int t[4];
int n;
long long time[100000 + 5][4];
int main() {
	//freopen("test.in", "r", stdin);
	//freopen("test.out", "w", stdout);
	cin >> k[1] >> k[2] >> k[3];
	cin >> t[1] >> t[2] >> t[3];
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> time[i][0];
	for (int i = 1; i <= 3; i++)
		for (int j = 1; j <= n; j++) {
			if (j <= k[i]) {
				time[j][i] = time[j][i-1] + t[i];
				continue;
			}
			time[j][i] = max(time[j][i-1], time[j-k[i]][i]) + t[i];
		}
	long long ans = 0;
	for (int i = 1; i <= n; i++)
		ans = max(ans, time[i][3]-time[i][0]);
	cout << ans;
	return 0;
}
