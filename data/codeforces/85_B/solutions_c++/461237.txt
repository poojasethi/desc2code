#include <iostream>
#include <cstdio>
#include <math.h>
#include <string>
#include <string.h>
#include <algorithm>
#include <vector>
using namespace std;
const long maxn = 100000+5;
long long k1, k2, k3, t1, t2, t3;
long long c[maxn], f[maxn];
long n;

void work(long long k, long long t)
{
	for (long i = 1; i <= n; ++i) {
		if (i-k<=0) {
			f[i] = f[i] + t;
		} else {
			if (f[i-k]<=f[i]) f[i] = f[i] + t;
			else f[i] = f[i-k] + t; 
		}
	}
}

int main()
{

	cin >> k1 >> k2 >> k3;
	cin >> t1 >> t2 >> t3;
	cin >> n;
	for (long i = 1; i <= n; ++i) {
		cin >> f[i];
		c[i] = f[i];
	}
	work(k1,t1);
	work(k2,t2);
	work(k3,t3);

	long long ans = -1;
	for (long i = 1; i <= n; ++i)
		if (f[i]-c[i]>ans) ans = f[i]-c[i];
	cout << ans << endl;


	return 0;
}
