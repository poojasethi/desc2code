#include <iostream>
#include <stdio.h>
#include <queue>
#include <vector>
#include <set>
#include <memory.h>

using namespace std;

#define FOR(i,b) for (int i = 0; i < (b); i++)
#define MP(A,B) make_pair(A,B)

long long room[3];
long long t[3];
int N;

long long c[4][100000];



int main()
{
	FOR(n,3)
		scanf("%I64d",&room[n]);
		
	FOR(n,3)
		scanf("%I64d",&t[n]);
		
	scanf("%d",&N);
	
	FOR(n,N)
	{
		scanf("%I64d",&c[0][n]);
	}


	for(int k=0;k<3;k++)
	{
		for(int n=0;n<N;n++)
		{
			if(n<room[k])
			{
				c[k+1][n]=c[k][n]+t[k];
			}
			else
			{
				c[k+1][n]=max(c[k+1][n-room[k]]+t[k],c[k][n]+t[k]);
			}
		}
		
	}

	long long res =0;
	
	for(int n=0;n<N;n++)
	{
		res = max(res,c[3][n]-c[0][n]);;
	}
	
	printf("%I64d\n",res);
	

	

	
}
