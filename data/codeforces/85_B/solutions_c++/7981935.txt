#include <bits/stdc++.h>

using namespace std;

typedef pair<long long, long long> pie;

#define L first
#define R second
#define MP make_pair
#define PB push_back

const long long maxn = 100000 + 100;

long long n, a[maxn], b[maxn];

void fun(long long k, long long t){
	for(long long i=1;i<=min(k, n);i++){
		long long tim = 0;
		for(long long j=i;j<=n;j+=k){
			a[j] = max(tim + t, a[j] + t);
			tim = a[j];
		}
	}
}

int main(){
	ios_base::sync_with_stdio(false);
	long long k1, k2, k3;
	cin >> k1 >> k2 >> k3;
	long long t1, t2, t3;
	cin >> t1 >> t2 >> t3;
	cin >> n;
	for(long long i=1;i<=n;i++){
		cin >> a[i];
		b[i] = a[i];
	}
	fun(k1, t1);
	fun(k2, t2);
	fun(k3, t3);
	long long ans = 0;
	for(long long i=1;i<=n;i++)
		ans = max(ans, a[i] - b[i]);
	cout << ans << endl;
}
