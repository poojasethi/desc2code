#include <iostream>
#include <queue>
#include <algorithm>
using namespace std;

typedef long long  ll;

#define REP(i,b,n) for(int i=b;i<n;i++)
#define rep(i,n) REP(i,0,n)

ll solve() {
  ll k[3], t[3], n;
  
  rep(i,3) cin>>k[i];
  rep(i,3) cin>>t[i];
  cin>>n;
  
  queue<ll> que[3];
  rep(i,3) for(int j=0;j<k[i] && j<n;j++) {
    que[i].push(0);
  }
  
  ll res = 0, c, rec;
  while (n--) {
    cin>>c;
    rec = c;
    //tt = c;
    rep(i,3) {
      c = max(que[i].front(),c) + t[i];
      que[i].pop();
      que[i].push(c);
    }
    res = max(res,c-rec);
  }
  
  return res;
}

int main(int argc, char *argv[]) {
  cout<<solve()<<endl;
  return 0;
}
