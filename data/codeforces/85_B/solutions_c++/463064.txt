#include <iostream>
#include <deque>

using namespace std;

long long c[4][100500]={};

int main()
{
    int k[3],t[3],n;
    cin >> k[0] >> k[1] >> k[2] >> t[0] >> t[1] >> t[2] >> n;
    for (int s=0;s<3;++s)
    {
        int q=0;
        for (int i=0;i<n;++i)
        {
            if (s==0)
                cin >> c[s][i];
            c[s+1][i]=c[s][i]+t[s];
            while(q<i && c[s+1][q]<=c[s][i])
                q++;
            long long ql = i-q;
            long long qo = ql%k[s];
            long long qt = ql/k[s];
            c[s+1][i]=c[s+1][q+qo]+t[s]*qt;
        }
    }
    long long ans = 0;
    for (int i=0;i<n;++i)
        if (c[3][i]-c[0][i]>ans)
            ans = c[3][i]-c[0][i];
    cout << ans;
    return 0;
}
