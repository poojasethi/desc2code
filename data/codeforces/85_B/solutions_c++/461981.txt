#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define ll long long
#define lim 100001
#define lim2 4

ll i,j,mx,s,n;
ll A[lim],B[lim],E[lim],W[lim2],T[lim2];

int main()
{
  for(i=0;i<3;i++)
    scanf("%I64d",&W[i]);    
  for(i=0;i<3;i++)
    scanf("%I64d",&T[i]);
  scanf("%I64d",&n);
  for(i=0;i<n;i++)
    scanf("%I64d",&A[i]);
  sort(A,A+n);
  memcpy(B,A,sizeof A);
  for(i=0;i<3;i++)
  {
    for(j=s=0;j<n;j++)
    {
      mx=B[j];
      while(j-s+1>W[i])
      {
        mx=max(mx,E[s]);
        s++;
      }
      E[j]=mx+T[i];
    }
    memcpy(B,E,sizeof B);
  }
  for(i=0,mx=E[0]-A[0];i<n;i++)
    mx=max(mx,E[i]-A[i]);
  printf("%I64d\n",mx);
  return 0;
}
