#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<iostream>
using  namespace std;
using namespace std;
const int N = 100005;
__int64 t[N],t1[N],t2[N],t3[N];
int main()
{
	int k1,k2,k3,a,b,c,n,i;
	scanf("%d%d%d",&k1,&k2,&k3);
	scanf("%d%d%d",&a,&b,&c);
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%I64d",t+i);
	sort(t,t+n);
	__int64 res=0,r;
	for(i=0;i<n;i++)
	{
		if(k1<=i)
		{
			if(t[i]+a>t1[i-k1]+a)t1[i]=t[i]+a;
			else t1[i]=t1[i-k1]+a;
		}
		else t1[i]=t[i]+a;
		
		if(k2<=i)
		{
			if(t1[i]+b>t2[i-k2]+b)t2[i]=t1[i]+b;
			else t2[i]=t2[i-k2]+b;
		}
		else t2[i]=t1[i]+b;
		if(k3<=i)
		{
			if(t2[i]+c>t3[i-k3]+c)t3[i]=t2[i]+c;
			else t3[i]=t3[i-k3]+c;
		}
		else t3[i]=t2[i]+c;
		

		r=t3[i]-t[i];
		if(r>res)res=r;
	}

	printf("%I64d\n",res);
}
