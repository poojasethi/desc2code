#include<stdio.h>
#include<algorithm>

using namespace std;

const int maxn = 100000 + 5;

long long k[3];
long long t[3];
int n;
long long c[maxn], d[maxn], e[maxn];

int main()
{
	for(int i=0; i<3; i++)
		scanf("%I64d", &k[i]);
	for(int i=0; i<3; i++)
		scanf("%I64d", &t[i]);
	scanf("%d", &n);
	for(int i=0; i<n; i++)
		scanf("%I64d", &c[i]);

	for(int i=0; i<n; i++)
		e[i] = c[i];

	int end = 0;
	for(int p=0; p<3; p++)
	{
		for(int i=0; i<n; i++)
		{
			d[i] = max(i<k[p] ? 0 : d[i-k[p]], c[i]) + t[p];
		}
		for(int i=0; i<n; i++) c[i] = d[i];
	}

	long long ans = 0;

	for(int i=0; i<n; i++) ans = max(ans, c[i] - e[i]);

	printf("%I64d\n", ans);

	return 0;
}
