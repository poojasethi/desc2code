#include <string>
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int main() {
	long long k[3];
	long long t[3];
	cin >> k[0] >> k[1] >> k[2];
	cin >> t[0] >> t[1] >> t[2];
	int n;
	cin >> n;
	vector<long long> c(n);
	for (int i = 0; i < n; ++i) {
		cin >> c[i];
	}
	vector<long long> d(n);
	vector<long long> x = c;
	for (int w = 0; w < 3; ++w) {
		for (int i = 0; i < n; ++i) {
			d[i] = x[i] + t[w];
			if (i >= k[w] && d[i - k[w]] > x[i]) {
				d[i] = d[i - k[w]] + t[w];
			}
		}
		x = d;
	}
	long long ans = 0;
	for (int i = 0; i < n; ++i) {
		if (x[i] - c[i] > ans) {
			ans = x[i] - c[i];
		}
	}
	cout << ans;
}
