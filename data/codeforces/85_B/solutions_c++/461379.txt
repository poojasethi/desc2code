#include<cstdio>
#include<iostream>
#include<queue>
using namespace std;

priority_queue<long long> q[3];
long long k[5],t[5],N;
long long c[100005],ans = 0;
int main(){
	for (int i=0;i<3;++i) cin>>k[i];
	for (int i=0;i<3;++i) cin>>t[i];
	cin>>N;
	
	for (int i=0;i<3;++i){
		int M = min(N,k[i]);
		for (int j=0;j<M;++j){
			q[i].push(0);
		}
	}
	
	for (int i=0;i<N;++i){
		cin>>c[i];
		long long tmp = c[i];
		for (int j=0;j<3;++j){
			long long tt = -q[j].top();
			q[j].pop();
			tmp = max(tt,tmp);
			tmp += t[j];
			q[j].push(-tmp);
		}
		
		if (tmp-c[i]>ans) ans = tmp-c[i];
		
	}
	cout << ans << endl;

	return 0;
}
