#include <cstdio>
#include <cstdlib>
#include <cstring>

typedef long long LL;

int k[4], l[4], b[100005], n;
LL a[100005], wdw[100005];

int main()
{
    scanf("%d%d%d", &k[1], &k[2], &k[3]);
    scanf("%d%d%d", &l[1], &l[2], &l[3]);
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) {
        scanf("%d", &b[i]);
        a[i] = b[i];
    }

    for (int o = 1; o < 4; ++o) {
        memset(wdw, 0, sizeof(wdw));
        int now = 1;
        for (int i = 0; i < n; ++i) {
            if (wdw[now] > a[i]) a[i] = wdw[now];
            a[i] += l[o];
            wdw[now] = a[i];
            if (++now > k[o]) now = 1;
        }
    }

    LL ans = 0;
    for (int i = 0; i < n; ++i)
    if (a[i] - b[i] > ans) ans = a[i] - b[i];

    printf("%I64d\n", ans);

    return 0;
}
