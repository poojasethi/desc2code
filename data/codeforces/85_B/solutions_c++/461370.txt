#include <iostream>
#include <queue>
using namespace std;
typedef long long ll;
#define int ll
int ks[3];
int ts[3];
int n;
priority_queue<int,vector<int>,greater<int> > fs[3];
main()
{
	cin.sync_with_stdio(0);
	for(int i=0; i<3; ++i) cin>>ks[i];
	for(int i=0; i<3; ++i) cin>>ts[i];
	cin>>n;
	for(int i=0; i<3; ++i) {
		int a = min(ks[i],n);
		for(int j=0; j<a; ++j) {
			fs[i].push(0);
		}
	}
	int r=0;
	for(int i=0; i<n; ++i) {
		int c;cin>>c;
		int t=c;
		for(int j=0; j<3; ++j) {
			int f = fs[j].top();
			fs[j].pop();
			t = max(t,f);
			t += ts[j];
			fs[j].push(t);
		}
		r = max(r, t-c);
	}
	cout<<r<<'\n';
}
