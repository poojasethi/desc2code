#include <stdio.h>

long long a[100010]={0},b[100010]={0},c[100010]={0},n,t1,t2,t3,k1,k2,k3;

int main(){
    long long i,j=0,t,s1,s2,s3;
    scanf("%I64d%I64d%I64d",&k1,&k2,&k3);
    scanf("%I64d%I64d%I64d",&t1,&t2,&t3);
    scanf("%I64d",&n);
    s1=0;
    s2=0;
    s3=0;
    for (i=0;i<n;i++){
        scanf("%I64d",&t);
        if (t<a[s1]) a[s1]+=t1;
        else a[s1]=t+t1;
        if (a[s1]<b[s2]) b[s2]+=t2;
        else b[s2]=a[s1]+t2;
        if (b[s2]<c[s3]) c[s3]+=t3;
        else c[s3]=b[s2]+t3;
        if (c[s3]-t>j) j=c[s3]-t;
        s1=(s1+1)%k1;
        s2=(s2+1)%k2;
        s3=(s3+1)%k3;
    }
    printf("%I64d\n",j);
    return 0;
}
