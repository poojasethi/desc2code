#! /usr/bin/env python

k1, k2, k3 = map(int, raw_input().split())
t1, t2, t3 = map(int, raw_input().split())
n = input()
oq = map(int, raw_input().split())

def gen_new_que(r, k, t):
    q = []
    ft = 0
    nt = 0
    for x in r:
        while ft < len(q) and q[ft] <= x:
            ft += 1
            k += 1
        nt = max(nt, x)
        if k:
            q.append(nt + t)
            k -= 1
        else:
            nt = max(nt, q[ft])
            ft += 1
            q.append(nt + t)
    return q

q = gen_new_que(oq, k1, t1)
q = gen_new_que(q, k2, t2)
q = gen_new_que(q, k3, t3)

print max(q[i] - oq[i] for i in xrange(n))
