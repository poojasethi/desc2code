k1,k2,k3 = map(int,raw_input().split())
t1,t2,t3 = map(int,raw_input().split())
n = input()
r = map(int,raw_input().split())
def f (r,k,t):
    rr = []
    rt = 0    
    for x in r:
        if k==0:
            x = max(rr[rt],x)
            rt +=1
            k += 1
        rr.append(x+t)
        k-=1
    return rr

rr = f(r,k1,t1)
rr = f(rr,k2,t2)
rr = f(rr,k3,t3)
print max(rr[i]-r[i] for i in xrange(n))
