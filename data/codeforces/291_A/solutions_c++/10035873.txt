#include <bits/stdc++.h>
using namespace std;

int n, x, ans;
map <int, int> a;

int main()
{
    cin >> n;
    for (int i = 1; i <= n; ++i)
    {
        cin >> x;
        a[x]++;
        if (a[x] == 2 && x) ++ans;
        else if (a[x] > 2 && x)
        {
            cout << -1;
            return 0;
        }
    }
    cout << ans;
    return 0;
}
