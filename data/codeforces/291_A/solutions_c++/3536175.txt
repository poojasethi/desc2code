#include<cstdio>
#include<set>

using namespace std;

int main(){
	
	multiset<int> st;
	int n,a;
	bool ans=true;
	int num=0;
	scanf("%d",&n);
	while(n--){
		scanf("%d",&a);
		if(a&&true)st.insert(a);
		if(st.count(a)==2)num++;
		if(st.count(a)>2)ans=false;
	}
	if(!ans)num=-1;
	printf("%d\n",num);
	return 0;
}
