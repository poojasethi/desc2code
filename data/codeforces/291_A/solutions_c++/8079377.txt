#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<map>

using namespace std;

map<int,int> v;
int n,i,a,k=0;

int main()

{
	cin>>n;

	for(i=1;i<=n;i++) {
		cin>>a;
		if(a>0) {
			v[a]++;
			if(v[a]==2) k++;
			if(v[a]==3) {
				cout<<-1;
				exit(0);
			}
		}
	}

	cout<<k;

	return 0;
}
