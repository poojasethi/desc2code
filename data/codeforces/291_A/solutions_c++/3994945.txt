#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int a[1001],cnt=0,tmp=1,n;
	cin>>n;
	for(int i=0;i<n;i++)
	  cin>>a[i];
	sort(a,a+n);
	for(int j=0;j<n;j++)
	{while(a[j]==a[j+tmp]&&a[j]!=0){tmp++;cnt++;}
	  if(tmp>2){cout<<-1;return 0;}
	  tmp=1;
	}
    cout<<cnt;
	return 0;
}
