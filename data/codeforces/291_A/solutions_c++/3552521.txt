#include <cstdio>
#include <iostream>
#include <map>

using namespace std;

int N,res;

map<int,int> cnt;

int main(){
	
	scanf(" %d",&N);
	
	for( int a,i=0 ; i<N ; i++ ){
		scanf(" %d",&a);
		if( a>0 )
			cnt[a]++;
		if( cnt[a]==2 )
			res++;
		if( cnt[a]==3 ){
			printf("-1\n");
			return 0;
		}
	}
	
	printf("%d\n",res);
	return 0;
}
