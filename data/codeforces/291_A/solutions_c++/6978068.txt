#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int n,i;
	cin >> n;
	int a[n],cnt=0;
	for(i=0;i<n;i++) cin >> a[i];
	sort(a,a+n);
	for(i=0;i<n;i++){
		if(i+2<n && a[i]==a[i+2] && a[i]!=0){cout << -1 << endl; return 0;}
		if(i+1<n && a[i]==a[i+1] && a[i]!=0) cnt++;
	}
	cout << cnt << endl;
	return 0;
}
