#include<stdio.h>
int i,j,s,ans,n,arr[2000];
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&arr[i]);
	for(i=1;i<=n;i++){
		s=0;
		for(j=i+1;arr[i] && j<=n;j++)
			if(arr[i] == arr[j]) s++;
		if(s==1) ans++;
		if(s>1){
			puts("-1");
			return 0;
		}
	}
	printf("%d",ans);
	return 0;
}
