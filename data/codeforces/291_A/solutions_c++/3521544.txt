#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <map>
using namespace std;
int n,ans; map<int,int> c;
int main()
{
    scanf("%d",&n); int x;
    while(n--)
    {
        scanf("%d",&x); if(x) ++c[x];
        if(c[x]==2) ++ans; if(c[x]>2) {ans=-1; break;}
    }
    printf("%d\n",ans);
    return 0;
}
