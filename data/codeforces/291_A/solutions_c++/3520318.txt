#include <iostream>
#include <map>
using namespace std;

int main(){
    map < long , long > a;
    long i,r=0,j,n;
    cin>>n;
    for(i=0;i<n;i++){
        cin>>j;
        a[j]++;
        if(a[j]>2&&j){
            cout<<-1;
            return 0;
        }
        if(a[j]==2&&j)r++;
    }
    cout<<r;
    return 0;
}
