#include <map>
#include <iostream>

using namespace std;

int n, x, ret;
map<int, int> cnt;

int main()
{
  cin >> n;
  for (int i = 0; i < n; i++)
  {
    cin >> x;
    if (!x) continue;
    if (++cnt[x] > 2)
    {
      cout << -1;
      return 0;
    }
    ret += cnt[x] == 2;
  }
  cout << ret;
  return 0;
}
