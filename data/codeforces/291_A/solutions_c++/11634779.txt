#include <bits/stdc++.h>
using namespace std;
int main(){
	int n , a[1010] = {} , ans = 0;
	cin >> n;
	for(int i = 0 ; i < n ; i++)
		cin >> a[i];
	sort(a , a+n);
	for(int i = 0 ; i < n ; i++){
		if(a[i] && a[i] == a[i+1]){
			ans ++;
			if(a[i] == a[i+2]){
				cout << -1;
				return 0;
			}
		}
	}
	cout << ans;
}
