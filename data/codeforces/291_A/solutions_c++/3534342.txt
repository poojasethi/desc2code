#include <iostream>
#include <map>

using namespace std;

map<int, int> hash;

int main() {
	int n;
	int ans = 0;
	cin >> n;
	for (int i = 0; i < n; i++) {
		int x;
		cin >> x;
		if (x == 0) continue;
		hash[x]++;
		if (hash[x] > 2) {
			cout << -1 << endl;
			return 0;
		}
		if (hash[x] == 2) {
			ans++;
		}
	}
	cout << ans << endl;
}
