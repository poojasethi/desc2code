#include <iostream>
#include <map>
using namespace std;
map<int,int> M;
int x,t,c,v;
int main(){
    cin>>x;
    while(x--){
        cin>>t;
        if (t==0) continue;
        v=M[t];
        if (v==2){
            cout<<-1;
            return 0;
        }
        if (v==1) c++;M[t]++;
    }
    cout<<c;
    return 0;
}
