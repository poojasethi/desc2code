#include <cstdio>
#include <map>
using namespace std;

int gao() {
	int n, id, cnt, ret = 0;
	map<int,int> mp;
	scanf("%d", &n);
	for(int i=0; i<n; ++i) {
		scanf("%d", &id);
		if(!id) continue;
		cnt = ++mp[id];
		if(cnt > 2) return -1;
		if(cnt == 2) ++ret;
	}
	return ret;
}

int main() {
	printf("%d\n", gao());
}
