#include<iostream>
#include<map>
using namespace std;
map<int,int>a;
int main(){
    int n;
    cin>>n;
    int x,mx=-(int)1e6;
    int ans=0;
    for(int i=0;i<n;i++){
        cin>>x;
        a[x]++;
        if(x!=0 && a[x]>2){
            ans=-1;break;
        }
        if(x!=0 && a[x]==2)ans++;
    }
    cout<<ans;
return 0;}
