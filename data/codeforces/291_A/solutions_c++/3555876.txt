#include<iostream>
#include<algorithm>
using namespace std;
int a[1000000],n;
int main(){
   cin>>n;
   for(int i=0;i<n;i++) cin>>a[i];
   sort(a,a+n);
   int ans=0;
   for(int i=1;i<n;i++)
      if(a[i-1]==a[i] && a[i]){
         if(a[i]==a[i+1]) {cout<<-1;return 0;}
         ans++;
      }
   cout<<ans;
}
