#include<iostream>
#include<algorithm>
using namespace std;
int n,kil,sum,i,a[1001];
int main(){

cin>>n;
for(i=1;i<=n;i++){cin>>a[i];}
sort(a+1,a+n+1);
kil=0,sum=0;
for(i=2;i<=n;i++)
 {
if((a[i]==a[i-1]) && (a[i]!=0))kil++;
if(a[i]!=a[i-1]){kil=0;}
if(kil==1)sum++;
if(kil>=2) { cout<<-1; return 0;}
}
cout<<sum<<endl;
}
