#include <iostream>
#include <map>
using namespace std;
int main()
{
    map < int, int> m;
    int n, i, x, ans = 0;
    cin >>n;
    for(i = 0; i < n; i ++)
    {
    cin>>x;
    if(x){
        m[x] ++;
        if(m[x] == 2)ans ++;
        if(m[x] == 3){
            cout<<-1;
            return 0;}}
    }
    cout<<ans;
}
