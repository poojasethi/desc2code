#include<iostream>
#include<algorithm>
using namespace std;

int a[1100];

int main()
{
	int n, odp=0;
	cin>>n;
	for(int i=0; i<n; i++)
	{
		cin>>a[i];
	}
	sort(a, a+n);
	for(int i=0; i<n; i++)
	{
		if(a[i])
		{
			if(a[i]==a[i+1] && a[i+1]==a[i+2])
			{
				cout<<"-1";
				return 0;
			}
			if(a[i]==a[i+1])
			odp++;
		}
	}
	cout<<odp;
	return 0;
}
