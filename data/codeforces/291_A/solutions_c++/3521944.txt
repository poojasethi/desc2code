#include <map>
#include <cstdio>
using namespace std;

map<int,int> d;
int ans = 0;

int main() {
	int n;
	scanf("%d",&n);
	for (int i = 0; i < n; i++) {
		int x;
		scanf("%d",&x);
		if (x == 0) continue;
		d[x]++;
		if (d[x]==2)
			ans++;
		if (d[x]==3) {
			puts("-1");
			return 0;
		}
	}
	printf("%d",ans);
	return 0;
}
