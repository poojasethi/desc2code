#include <iostream>
#include <algorithm>
using namespace std;
int main(){
	int a[1010],i,n,k=0;
	cin>>n;
	for( i=0; i<n; i++ )
		cin>>a[i];
	sort(a,a+n);
	for( i=1; i<n; i++ ){
		if (a[i]!=0 && a[i]==a[i-1] && a[i]!=a[i+1] )k++;
		else if ( a[i]!=0 && a[i]==a[i-1] && a[i]==a[i+1] ){k=-1; break;}
	}
	cout<<k;
	return 0;
}
