#include<cstdio>
#include<map>
using namespace std;
map<int,int> m;
int main(){
	int n,i,ans,a;
	while(~scanf("%d",&n)){
		bool f=1;ans=0;
		m.clear();
		for(i=0;i<n;i++){
			scanf("%d",&a);
			if(!a)continue;
			m[a]++;
			if(m[a]==2)ans++;
			if(m[a]>2)f=0;
		}
		printf("%d\n",f?ans:-1);
	}
	return 0;
}
