#include<stdio.h>
#include<map>
int n;
std::map<int,int>p;
int main(){
	scanf("%d",&n);
	int b=0,flag=1;
	for(int i=0;i<n;i++){
		int tmp;
		scanf("%d",&tmp);
		if(tmp==0)
			continue;
		p[tmp]++;
		if(p[tmp]==2)
			b++;
		else if(p[tmp]>2)
			flag=0;
		}
	if(flag)
		printf("%d\n",b);
	else
		puts("-1");
	return 0;
	}
