#include <cstdio>
#include <algorithm>
#include <map>

using namespace std;

int N;
int sol;
map<int,int> M;

int main()
{
	scanf("%d",&N);
	for(int c = 0 ; c < N ; c++)
	{
		int a;
		scanf("%d",&a);
		if(a)
		{
			int tmp = ++M[a];
			if(tmp > 2)
			{
				printf("-1\n");
				return 0;
			}
			if(tmp == 2)
				sol++;
		}
	}
	printf("%d\n", sol);
}
