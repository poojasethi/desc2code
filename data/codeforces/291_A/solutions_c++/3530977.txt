#include <iostream>
#include <stdio.h>
#include <map>
using namespace std;
int n,x,ans,flag;
map<int,int>f;
int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
       cin>>x;
       f[x]++;
       if(x!=0&&f[x]==2)ans++;
       if(f[x]>2&&x!=0)flag=true;
    }
    if(flag)cout<<-1<<endl;
    else cout<<ans<<endl;
}
