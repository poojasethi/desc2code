#include <iostream>
#include <algorithm>
using namespace std;
int main(){
  int n, j, a[10000],k=0;
  cin>>n;
  for(j=0; j<n; j++) cin>>a[j];
sort(a,a+n);
for(j=0; j<n; j++)
if (a[j]==a[j+1] && a[j]==a[j+2] && a[j]!=0) return cout<<"-1", 0;
else if (a[j]==a[j+1] && a[j]!=0) k++;
cout<<k;
return 0;
}
