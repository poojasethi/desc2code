import bisect
n = input()
worms = map(int, raw_input().split())
m = input()
query = map(int, raw_input().split())
wsum = [0] * n
for i in range(n):
    if i == 0:
        wsum[i] = worms[i]
    else:
        wsum[i] = wsum[i - 1] + worms[i]

for i in range(m):
    print bisect.bisect_left(wsum, query[i]) + 1
