import bisect
n = int(raw_input())
a = map(int,raw_input().split())
lis = []
lis.append(a[0])
for i in range(1,n):
	lis.append(a[i]+lis[-1])

m = int(raw_input())
b = map(int,raw_input().split())
for i in xrange(m):
	val = bisect.bisect(lis, b[i])
	if lis[val-1] == b[i]:
		print val
	else:
		print val+1
