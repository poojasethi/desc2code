#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
typedef long long ll;
ll mod=1e9+7;
ll n,m,k,p,sum,l,r;
ll h[100050], a[100050], tot[5500], days[100050], knif[100050];
bool pan(ll mid)
{
    sum=0;
    for(int i=0;i<n;i++)
    {
        knif[i]=max(0LL, h[i]+m*a[i]-mid);
        days[i]= knif[i]/p;
        if(days[i]*p<knif[i]) days[i]++;
        sum+=days[i];
    }
    memset(tot,0,sizeof(tot));

    if(sum>m*k) return 0;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<days[i];j++)
        {
            if(knif[i]<h[i])
            {
                tot[0]++;
                knif[i]-=p;
            }
            else
            {
                ll day=(knif[i]-h[i])/a[i];
                if(day*a[i]< knif[i]-h[i]) day++;
                tot[day]++;
                knif[i]-=p;
            }
        }
    }
    for(int i=0;i<m;i++)
    {
        if(tot[i]>k) tot[i+1]+=(tot[i]-k);
    }
    return tot[m]==0;
}
int main()
{
    scanf("%d%d%d%d",&n,&m,&k,&p);
    for(int i=0;i<n;i++)
    {
        cin>>h[i]>>a[i];
        r=max(r,h[i]+m*a[i]);
    }
    while(l<=r)
    {
        ll mid=(l+r)>>1;
        if(pan(mid))
            r=mid-1;
        else
            l=mid+1;
    }
    printf("%I64d\n",l);
    return 0;
}
