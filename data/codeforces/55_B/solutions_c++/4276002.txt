#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
char op[4][10];
long long mn=100000000000007;
void dfs(int d, vector<long long> &x) {
	if(d==3){
		mn=min(mn,x[0]);
		return;
	}
	for(int i=0;i<x.size();++i){
		for (int j=i+1;j<x.size();++j) {
			vector<long long> c;
			for (int k=0;k<x.size();++k)if(k!=i&&k!=j)c.push_back(x[k]);
			if(op[d][0]=='+')c.push_back(x[i]+x[j]);
			else c.push_back(x[i]*x[j]);
			dfs(d+1,c);
		}
	}
}

int main(){
	vector<long long> x(4);
	for (int i=0;i<4; ++i)cin >> x[i];
	for (int i=0;i<3; ++i)cin >> op[i];
	dfs(0,x);
	cout<<mn<<endl;
	return 0;
}
