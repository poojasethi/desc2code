#include <stdio.h>
#include <algorithm>
using namespace std;

typedef long long LL;
const LL INF = 12345678912345678LL;
LL ans;

char op[3][10];

void go(int n, LL *arr) {
	LL brr[4];
	int bn=0;
	if(n==1) ans = min(ans, arr[0]);
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++) {
			bn = 0;
			if(op[n-1][0] == '+') brr[bn++] = arr[i] + arr[j];
			else brr[bn++] = arr[i] * arr[j];
			for(int k=0;k<n;k++)
				if(k!=i && k!=j)
					brr[bn++] = arr[k];
			go(n-1, brr);
		}
}

int main(void) {
	ans = INF;
	LL a[4];
	for(int i=0;i<4;i++)
		scanf("%I64d", &a[i]);
	for(int i=3;i>0;i--)
		scanf("%s", op[i]);
	go(4, a);
	printf("%I64d\n", ans);
	return 0;
}
