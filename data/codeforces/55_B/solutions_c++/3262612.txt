#include <cstdio>
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

typedef long long int ll;

ll a,b,c,d;

char tab[3];

ll res(int x,vector<ll>v)
{
	if(v.size()==1) return v[0];
	ll wyn=1e18;
	for(int i=0;i<v.size();i++)
	{
		for(int j=0;j<i;j++)
		{
			vector<ll>u;
			for(int k=0;k<v.size();k++)
			{
				if((k!=i) && (k!=j)) u.push_back(v[k]);
			}
			if(tab[x]=='+') u.push_back(v[i]+v[j]);
			else u.push_back(v[i]*v[j]);
			wyn=min(wyn,res(x+1,u));
		}
	}
	return wyn;
}

int main()
{
	vector<ll>v;
	cin >> a >> b >> c >> d;
	v.push_back(a);
	v.push_back(b);
	v.push_back(c);
	v.push_back(d);
	cin >> tab[0] >> tab[1] >> tab[2];
	cout << res(0,v) << "\n";
	return 0;
}
