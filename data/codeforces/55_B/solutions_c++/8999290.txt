#include<iostream>
#include<algorithm>
using namespace std;
char ch1,ch2,ch3;
long long f(long long a,long long b,long long c,long long d){
    long long ans=0,s,s1,s2,s3;
    if(ch1=='+')s=a+b;else s=a*b;
    if(ch2=='+'){
        s1=s+c;s2=s+d;s3=c+d;
    }else{
        s1=s*c;s2=s*d;s3=c*d;
    }
    if(ch3=='+')ans=min(s2+c,min(s1+d,s+s3));
    else ans=min(s2*c,min(s1*d,s*s3));
    return ans;
}
int main(){
    long long a,b,c,d,s,m=10000000000001;
    cin>>a>>b>>c>>d;cin>>ch1>>ch2>>ch3;
    s=f(a,b,c,d);if(s<m)m=s;
    s=f(a,c,b,d);if(s<m)m=s;
    s=f(a,d,b,c);if(s<m)m=s;
    s=f(b,c,a,d);if(s<m)m=s;
    s=f(b,d,a,c);if(s<m)m=s;
    s=f(c,d,a,b);if(s<m)m=s;
    cout<<m;
    return 0;
}
