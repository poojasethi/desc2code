#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

long long p[100][100];
char st[100];
long long res=2000000000000LL;

void f(int dep)
{
	if (dep==1)
	{
		res=min(res,p[dep][0]);
		return;
	}

	for (int i=0;i<dep;i++)
	for (int j=i+1;j<dep;j++)
	{
		int c=0;
		for (int  k=0;k<dep;k++)
		if (k!=i && k!=j)
		{
			p[dep-1][c++]=p[dep][k];
		}
		p[dep-1][c++]=(st[4-dep]=='+'?p[dep][i]+p[dep][j]:p[dep][i]*p[dep][j]);
		f(dep-1);
	}
}

int main()
{
	for (int i=0;i<4;i++) cin >> p[4][i];
	for (int i=0;i<3;i++)
	{
		char buf[10];
		scanf("%s",buf);
		st[i]=buf[0];
	}
	
	f(4);
	cout << res << endl;
}
