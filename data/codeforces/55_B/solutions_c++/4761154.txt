#include <iostream>
#include <algorithm>

using namespace std;

long long f(long long a, long long b, char c)
{
	if (c == '+')
		return a + b;
	else
		return a * b;
}

int a[4];
char b[3];
long long ans = 1000ll * 1000 * 1000 * 1000 + 1000;

int main()
{
	for (int i = 0; i < 4; i++)
		cin >> a[i];
	for (int j = 0; j < 3; j++)
		cin >> b[j];
	sort(a, a + 4);
	do
	{
		ans = min(ans, f(f(a[0], a[1], b[0]), f(a[2], a[3], b[1]), b[2]));
		ans = min(ans, f(f(a[0], a[2], b[0]), f(a[1], a[3], b[1]), b[2]));
		ans = min(ans, f(f(a[0], a[3], b[0]), f(a[1], a[2], b[1]), b[2]));
		ans = min(ans, f(f(f(a[0], a[1], b[0]), a[2], b[1]), a[3], b[2]));
	} while (next_permutation(a, a + 4));
	cout << ans << endl;
	return 0;
}
