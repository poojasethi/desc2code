#include<iostream>
#include<cstdio>
using namespace std;
long long a[5];
long long ans=10000000000000;
char c[10];
void dfs(int k)
{
	long long  tmp[10];
	for(int i=1;i<=4;i++)tmp[i]=a[i];
	if(k>3)
	{
		if(a[1]<ans)ans=a[1];
		return;
	}
	for(int i=1;i<=4-k;i++)
	{
		for(int j=i+1;j<=4-k+1;j++)
		{
			if(c[k]=='+')a[i]+=a[j];
			else a[i]*=a[j];
			if(j!=4-k+1)a[j]=a[4-k+1];
			dfs(k+1);
			for(int p=1;p<=4;p++)a[p]=tmp[p];
		}
	}
}
int main()
{
	cin>>a[1]>>a[2]>>a[3]>>a[4];
	for(int i=1;i<=3;i++)cin>>c[i];
	dfs(1);
	cout<<ans;
	return 0;
}
