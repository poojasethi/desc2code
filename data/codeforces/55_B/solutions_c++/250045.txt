#include <iostream>
#include <vector>
using namespace std;

char op[3];

long long ans;

void recur (vector <long long> v, int pos)
{
	if (pos == 3)
	{
		
		if (ans == -1 || ans > v[0])
		{
			ans = v[0];
		}
	}
	else
	{
		for (int i = 0; i < v.size(); i++)
		{
			for (int j = i+1; j < v.size(); j++)
			{
				vector <long long> z;
				if (op[pos] == '+') 
				{
					z.push_back(v[i]+v[j]);
				}
				else
				{
					z.push_back(v[i]*v[j]);
				}
				for (int k = 0; k < v.size(); k++)
				{
					if (k != i && k != j) z.push_back(v[k]);
				}
				recur(z,pos+1);
			}
		}
	}
}
				

main ()
{
	vector <long long int> v(4);
	cin >> v[0] >> v[1] >> v[2] >> v[3];

	cin >> op[0] >> op[1] >> op[2];

	ans = -1;
	recur(v,0);
	cout << ans << "\n";
}
