#include<iostream>
#include<algorithm>
using namespace std;
#define A 0x7ffffffffffff
__int64 ans,n;
char ch[3];
__int64 op(__int64 x,__int64 y,char z)
{
	if(z=='*')
		return x*y;
	if(z=='+')
		return x+y;
}
void dfs(__int64 a,__int64 b,__int64 c,__int64 d,int k){
	__int64 t[4];
	t[0]=a;t[1]=b;t[2]=c;t[3]=d;
	sort(t,t+4);
	if(k==1){
	if(ans>t[0])
		ans=t[0];
	return ;
	}
	for(int i=0;i<k;++i)
		for(int j=i+1;j<k;++j){
			a=t[i];
			b=t[j];
			t[i]=op(a,b,ch[4-k]);
			t[j]=A;
			dfs(t[0],t[1],t[2],t[3],k-1);
			t[i]=a;
			t[j]=b;
		}
}
int main(){
	int i;
	__int64 a,b,c,d;
	while(cin>>a>>b>>c>>d>>ch[0]>>ch[1]>>ch[2])
	{
		ans=A;n=4;
		dfs(a,b,c,d,4);
		cout<<ans<<endl;
	}
	return 0;
}
