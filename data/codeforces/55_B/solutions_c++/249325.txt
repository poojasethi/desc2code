#include <iostream>
#include <algorithm>

using namespace std;

long long A[5], sol = 1000000000000000LL;
char o[4];

void result(int ind, long long A[5], int n) {
	if (ind == 4) {
		sol = min(sol, A[1]);
    	return;
	}

	long long B[5] = {0, 0, 0, 0, 0};
	
	for (int i = 1; i < n; i++)
		for (int j = i + 1; j <= n; j++) {
		    if (o[ind] == '*')
           	    B[1] = 1LL * A[i] * A[j];
		    else
				B[1] = A[i] + A[j];

			int x = 2;
			for (int k = 1; k <= n; k++)
				if (k != i && k != j)
					B[x++] = A[k];
			
			result(ind + 1, B, n - 1);
		}
	
}

int main() {

	for (int i = 1; i <= 4; i++)
		cin >> A[i];
	for (int i = 1; i <= 3; i++)
		cin >> o[i];

	result(1, A, 4);

	cout << sol << "\n";

	return 0;
}
