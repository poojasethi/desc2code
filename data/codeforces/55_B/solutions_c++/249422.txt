#include <cstdio>
#include <algorithm>
#include <cstring>
#include <vector>
using namespace std;

int x;
long long v[4],ret;
char s[4][5];
vector<long long> a;

void dfs(vector<long long> p,int n){
	vector<long long> T;
	if (n==3) ret=min(ret,p[0]);
	else{
		for (int i=0;i<p.size();i++)
			for (int j=i+1;j<p.size();j++){
				T = p;
				T.erase(T.begin()+i);
				T.erase(T.begin()+j-1);
				if (s[n][0]=='+') T.push_back(p[i]+p[j]);
				else T.push_back(p[i]*p[j]);
				dfs(T,n+1);
			}
	}
}

int main(){
	a.clear();
	for (int i=0;i<4;i++){
		scanf("%d",&x);
		a.push_back(x);
	}
	scanf("%s%s%s",s[0],s[1],s[2]);
	ret = 1LL<<60;
	dfs(a,0);
	printf("%I64d\n",ret);
	return 0;
}
