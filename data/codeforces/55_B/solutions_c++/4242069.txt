#include<iostream>
#include<vector>
using namespace std;
string op[4];
long long ans=10000000000000;
long long p;
void dfs(long long d,vector<long long> x)
{
	if(d==3) 
	{
		ans=min(ans,x[0]);
		return;
	}
	for (long long i=0;i<x.size();i++) 
	{
		for (long long j=i+1;j<x.size();j++) 
		{
			vector<long long> c; 
			for(long long k=0;k<x.size();k++) 
			{
				if(k!=i && k!=j) c.push_back(x[k]);
			}
			if(op[d]=="+") c.push_back(x[i]+x[j]);
			else c.push_back(x[i]*x[j]);
			dfs(d+1,c);
		}
	}
}
int main() 
{
	vector<long long> x;
	for(long long i=0;i<4;i++)
	{
		long long y;
		cin>>y;
		x.push_back(y);
	}
	for(long long i=0;i<3;i++)
	{
		cin>>op[i];
	}
	dfs(0,x);
	cout<<ans<<endl;
	return 0;
}
