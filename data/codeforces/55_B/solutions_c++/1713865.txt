#include<iostream>
#include<vector>
using namespace std;
vector<long long> a;
char op[4];
long long ans=9999999999999999LL;
void dfs(vector<long long> f,int x){
	if ( f.size() == 1 ){
		ans = min(f[0],ans);
		return;
	}
	long long val;
	for(int i=0;i<f.size();i++){
		for(int j=i+1;j<f.size();j++)
		{
			if (op[x] == '+' ) val = f[i]+f[j];
			else if ( op[x] == '*' ) val = f[i]*f[j];
			
			vector<long long> temp;
			for(int k=0;k<f.size();k++){
				if ( k ==i || k == j ) continue;
				temp.push_back(f[k]);
			}
			temp.push_back(val);
			dfs(temp,x+1);
		}
	}
}
int main(){
	
	a = vector<long long>(4);
	for(int i=0;i<4;i++) cin>>a[i];
	for(int i=0;i<3;i++) cin>>op[i];	
	
	dfs(a,0);
	cout<<ans<<endl;
	return 0;
}
