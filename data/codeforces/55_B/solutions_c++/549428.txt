#include <iostream>

using namespace std;

long long r = 1ll << 62;

void f(int c, long long a[4], char o[3])
{
	if (c == 3)
		r = min(r, a[0]);
	else
	{
		long long n[4];
		for (int i = 0; i < 4 - c; i++)
		{
			for (int j = i + 1; j < 4 - c; j++)
			{
				for (int k = 0; k < 4 - c; k++)
					n[k] = a[k];

				if (o[c] == '+')
					n[i] = n[i] + n[j];
				else
					n[i] = n[i] * n[j];

				for (int k = j; k < 3 - c; k++)
					n[k] = a[k + 1];
				
				f(c + 1, n, o);
			}
		}
	}
}

int main(int argc, char* argv[])
{
	long long a[4];
	char o[3];

	cin >> a[0] >> a[1] >> a[2] >> a[3] >> o[0] >> o[1] >> o[2];

	f(0, a, o);
	cout << r << endl;

	return 0;
}
