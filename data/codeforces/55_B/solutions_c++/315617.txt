#include<vector>
#include<iostream>
using namespace std;
typedef long long ll;
char op[3];

ll opera(ll a, ll b, char c){
	if(c == '+') return a + b;
	else return a * b;
}

ll get(int pos, vector<ll> nums){
	if(pos == 3) return nums[0];
	ll res = 1LL << 62;
	for(int i = 0; i < nums.size(); i++) for(int j = i + 1; j < nums.size(); j++){
		vector<ll> next = nums;
		next.erase(next.begin() + j);
		next.erase(next.begin() + i);
		next.push_back(opera(nums[i], nums[j], op[pos]));
		res = min(res, get(pos + 1, next));
	}
	return res;
}

int main(){
	vector<ll> num(4);
	for(int i = 0; i < 4; i++) cin >> num[i];
	for(int i = 0; i < 3; i++) cin >> op[i];
	
	cout << get(0, num) << endl;	
}
