#include<iostream>
using namespace std;
typedef long long ll; 
const ll INF = (ll)2*1000*1000*1000*1000*1000;
ll a[4];
char com[3];
ll ans = INF;
struct my
{
	ll d[4];
};

void done(int k,my A)
{
	if(k==1)
	{
		ans = min(ans,A.d[0]);
		return;
	}
	for(int i=0;i<k;i++)
		for(int j=0;j<k;j++)
			if(i!=j)
			{
				my temp;
				temp.d[0] = ((com[4-k]=='+')?A.d[i]+A.d[j] :A.d[i]*A.d[j]);
				int ind=1;
				for(int a=0;a<k;a++)
					if(a!=i && a!=j)
						temp.d[ind++] = A.d[a];
				done(k-1,temp);
			}

}

int main()
{
	for(int i=0;i<4;i++)
		cin >> a[i];
	for(int i=0;i<3;i++)
		cin >> com[i];
	my dd;
	for(int i=0;i<4;i++)
		dd.d[i] = a[i];

	done(4,dd);
	cout << ans << endl;
	return 0;
}
