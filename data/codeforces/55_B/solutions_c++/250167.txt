#include <iostream>
#include <string>
#include <cstdio>
#include <vector>
using namespace std;

vector <__int64> a;
string str="";
__int64 ans=10000000000000;

void gen (vector <__int64> a)
{
	if (a.size()==1) {if (ans>a[0]) ans=a[0];}
	else
	{
		for (int i=0;i<a.size()-1;i++)
			for (int j=i+1;j<a.size();j++)
			{
				vector <__int64> b;
				for (int k=0;k<a.size();k++)
					if (k!=i&&k!=j) b.push_back(a[k]);
				b.push_back(str[2*(4-a.size())]=='*'?a[i]*a[j]:a[i]+a[j]);
				gen(b);
			}
	}
}
int main()
{
	int aa;
	for (int i=0;i<4;i++) {cin >> aa; a.push_back(aa);}
	scanf("\n");
	getline(cin,str);
	gen(a);
	cout << ans;
	return 0;
}
