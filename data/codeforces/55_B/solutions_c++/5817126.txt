#include<stdio.h>
__int64 a[4],min;
char c[3];
void dfs(int n){
	int i,j;
	if(n==3){
		for(i=0;i<4;i++)
			if(a[i]!=-1)
				break;
		if(a[i]<min) min=a[i];
	}
	__int64 x,y;
	for(i=0;i<3;i++)
		if(a[i]>=0)
			for(x=a[i],j=i+1;j<4;j++)
				if(a[j]>=0){
					y=a[j];
					a[j]=-1;
					if(c[n]=='+')	a[i]=x+y;
					else	a[i]=x*y;
					dfs(n+1);
					a[i]=x,a[j]=y;
				}
}
int main(){
	while(scanf("%I64d%I64d%I64d%I64d",&a[0],&a[1],&a[2],&a[3])!=-1){
		getchar();
		min=999999999999999;
		scanf("%c %c %c",&c[0],&c[1],&c[2]);
		dfs(0);
		printf("%I64d\n",min);
	}
	return 0;
}
