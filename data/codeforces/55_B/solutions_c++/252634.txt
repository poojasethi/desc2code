long long f(long long x, long long y, char op){
	long long ret;
	if(op == '*')
		ret = x*y;
	else ret = x+y;
	return ret;
}

	FOR(i,0,4)
		cin >> a[i];
	FOR(i,0,3)
		cin >> c[i];

	long long rec = 1000*1000;
	rec *= 1000*1000;
	long long cur;
	sort(a, a+4);
	do{
		b[0] = f(a[0],a[1],c[0]);
		b[1] = a[2];
		b[2] = a[3];
		sort(b, b+3);
		do{
			cur = f(b[0], b[1], c[1]);
			cur = f(cur, b[2], c[2]);
			if(rec > cur)
				rec = cur;
		}while(next_permutation(b, b+3));
	}while(next_permutation(a, a+4));

	cout << rec << endl;

}
