#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <ctime>
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,T,te;
int64 a[5],an;
char p[5][5];
void dfs(int64 a[5],int n) {
	if (n==1) {
		an=min(an,a[1]);
		return;
	}
	int i,j,k;
	int64 b[5];
	For(i,1,n)For(j,i+1,n) {
		For(k,1,n) b[k]=a[k];
		if (p[n][0]=='+') b[i]=a[i]+a[j];
		else b[i]=a[i]*a[j];
		swap(b[j],b[n]);
		dfs(b,n-1);
	}
}
int main() {
	an=(int64)1e18;
	For(i,1,4) scanf("%I64d",&a[i]);
	for (i=4;i>1;i--) scanf("%s",p[i]);
	dfs(a,4);
	printf("%I64d\n",an);
	//for(;;);
	return 0;
}
