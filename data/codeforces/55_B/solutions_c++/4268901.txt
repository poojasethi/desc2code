#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
char c[4];
long long a[5];
long long mn=1LL<<60;
int main()
{
    for (int i=0; i<4; i++) cin >>a[i];
    sort(a,a+4);
    for (int i=1; i<=3; i++) cin >>c[i];
    do
    {
        long long ans=a[0];
        for (int i=1; i<=3; i++)
        {
            if (c[i]=='+') ans+=a[i]; else ans*=a[i];
        }
        mn=min(mn,ans);
        long long s1=0,s2=0;
        if (c[1]=='+') s1=a[0]+a[1]; else s1=a[0]*a[1];
        if (c[2]=='+') s2=a[2]+a[3]; else s2=a[2]*a[3];
        if (c[3]=='+') ans=s1+s2; else ans=s1*s2;
        mn=min(mn,ans);
    }while (next_permutation(a,a+4));
    cout <<mn <<endl;
    return 0;
}
