#include <iostream>
using namespace std;
int n, i;
long long a[5], ans;
char ch[5];
void ss(long long *a, int k)
{
	 long long b[5], k1;
	 if (k == 1)
	 {
		 if ((ans == -1) || (ans > a[1])) ans = a[1];
		 return;
	 }
	 for (int i = 1; i < k; i++)
	 {
		 for (int j = i + 1; j <= k; j++)
		 {
			 k1 = 0;
			 for (int h = 1; h <= k; h++)
			 {
				 if ((h != i) && (h != j))
				 {
					 k1++; b[k1] = a[h];
				 }
			 }
			 if (ch[5 - k] == '+') b[k - 1] = a[i] + a[j];
			 else b[k - 1] = a[i] * a[j];
			 ss(b, k - 1);
		 }
	 }
}
int main()
{
	 for (i = 1; i <= 4; i++) cin >> a[i];
	 for (i = 1; i <= 3; i++) cin >> ch[i];
	 ans = -1;
	 ss(a, 4);
	 cout << ans;
	 return 0;
}
