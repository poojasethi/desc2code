#include <cstdlib>
#include <cstdio>
#include <cstring>
#include <iostream>
#define LL long long
using namespace std;
LL A[10],Ans=1LL<<60;
char B[10];
int UK[10],U[10];

void DFS(int X)
{
	if (X==7)
	{
		Ans=min(Ans,A[7]);
		return ;
	}
	for (int i=1;i<=X;++i)
		if (!U[i])
		for (int j=i+1;j<=X;++j)
			if (!U[j])
			
				{
					int k=X-3;
					A[X+1]=(B[k]=='*'?(A[i]*A[j]):(A[i]+A[j]));
					UK[k]=U[i]=U[j]=true;
					DFS(X+1);
					UK[k]=U[i]=U[j]=false;
				}	
}

int main()
{
	cin>>A[1]>>A[2]>>A[3]>>A[4];
	scanf(" \n %c %c %c",&B[1],&B[2],&B[3]);
	DFS(4);
	cout<<Ans<<endl;
}
