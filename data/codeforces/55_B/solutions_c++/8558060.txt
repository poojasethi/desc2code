#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int s[4];
long long f(long long a,long long b,char x)
{
    if (x=='+')
    {
        return a+b;
    }
    else
    {
        return a*b;
    }
}
int main()
{
    char x,y,z;
    long long ans=1e13;
    cin>>s[0]>>s[1]>>s[2]>>s[3]>>x>>y>>z;
    sort(s,s+4);
    do
    {
        long long a=s[0];
        long long b=s[1];
        long long c=s[2];
        long long d=s[3];
        ans=min(ans,f(f(a,b,x),f(c,d,y),z));
        ans=min(ans,f(f(a,c,x),f(b,d,y),z));
        ans=min(ans,f(f(a,d,x),f(b,c,y),z));
        ans=min(ans,f(f(f(a,b,x),c,y),d,z));
    }while (next_permutation(s,s+4));
    cout<<ans<<endl;
    return 0;
}
