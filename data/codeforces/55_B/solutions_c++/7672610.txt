#include<cstdio>

char op[5];
long long v[5];

long long doop(long long x, long long y, char op){
	return op == '+' ? x + y : x * y;
}

long long calc(int s){
	if(s == 3)
		for(int i = 0; i < 4;i++)
			if(v[i] > -1)
				return v[i];

	long long best = -1;
	for(int i = 0; i < 4; i++)
		for(int j = i+1; j < 4; j++)
			if(v[i] > -1 && v[j] > -1){
				long long  ovi = v[i], ovj = v[j];
				v[i] = doop(v[i], v[j], op[s]);
				v[j] = -1;
				long long cr = calc(s+1);
				if(best == -1 || cr < best)
					best = cr;
				v[i] = ovi, v[j] = ovj;
			}
	return best;
}

int main(void){
	scanf("%I64d%I64d%I64d%I64d %c %c %c", v, v+1, v+2, v+3, op, op+1, op+2);
	printf ("%I64d", calc(0));
	return 0;
}
