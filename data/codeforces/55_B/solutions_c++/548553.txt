#include <stdio.h>
__int64 ans=10000000000000;
__int64 a[5],b[5];
void dfs(int k)
{
	__int64 tmp[5];
	int i,j,s;
	for (i=1;i<=4;i++) tmp[i]=a[i];
	if (k>3)
	{
		if (a[1]<ans) ans=a[1];
		return;
	}
	for (i=1;i<=4-k;i++)
		for (j=i+1;j<=4-k+1;j++)
		{
			if (b[k]==0) a[i]=a[i]+a[j];
			else a[i]=a[i]*a[j];
			if (j!=4-k+1) a[j]=a[4-k+1];
			dfs(k+1);
			for (s=1;s<=4;s++) a[s]=tmp[s];
		}
}
int main()
{
    char str[10];
	scanf("%I64d%I64d%I64d%I64d",&a[1],&a[2],&a[3],&a[4]);
	for (int i=1;i<=3;i++)
	{
		scanf("%s",str);
		if (str[0]=='+') b[i]=0;else b[i]=1;
	}
	dfs(1);
	printf("%I64d\n",ans);
	return 0;
}
