#include <algorithm>
#include <cstdlib>
#include <iostream>

#define O(a, b, i)    ((op[i] == '*') ? a * b : a + b)

using namespace std;

typedef long long ll;

int main()
{
    ll num[4], ans = 0xfffffffffffffffll;
    char op[5] = {0};
    
    for (int i = 0; i < 4; ++i)
        cin >> num[i];

    for (int i = 0; i < 3; ++i)
        cin >> op[i];
        
    sort(num, num + 4);

    do
    {
        ans = min(ans, O(O(num[0], num[1], 0), O(num[2], num[3], 1), 2));
        ans = min(ans, O(O(O(num[0], num[1], 0), num[2], 1), num[3], 2));
        
    }while (next_permutation(num, num + 4));
    
    cout << ans << endl;
    return 0;
}
