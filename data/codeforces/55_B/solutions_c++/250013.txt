#include <iostream>
#include <cstdio>
using namespace std;

__int64 a[4][4], ans;
char ch[3];

void dfs(int l)
{
	if (l == 1) {
		if (a[3][0] < ans) ans = a[3][0];
		return;
	}
	for (int i = 0; i < l; ++i)
		for (int j = 0; j < l; ++j)
			if (j != i) {
				int p = 0;
				for (int k = 0; k < l; ++k)
					if (k != i && k != j)
						a[4 - l + 1][p++] = a[4 - l][k];
				if (ch[4 - l] == '+')
					a[4 - l + 1][p] = a[4 - l][i] + a[4 - l][j];
				else a[4 - l + 1][p] = a[4 - l][i] * a[4 - l][j];
				dfs(l - 1);
			}
}

int main()
{
	for (int i = 0; i < 4; ++i)	scanf("%I64d", &a[0][i]);
	scanf("\n%c %c %c", &ch[0], &ch[1], &ch[2]);
	ans = 1000000; ans *= 1000000;
	dfs(4);
	printf("%I64d", ans);
	return 0;
}
