import itertools

nums = map(int,raw_input().split())
oop = raw_input().split()

def getBest(curNums,pos):
	if pos == 3:
#		print curNums, pos
		return curNums[0]
	else:
		bestHere = 1005**4
		for i in xrange(len(curNums)-1):
			for j in xrange(i+1,len(curNums)):
				thisNums = list(curNums)
				del thisNums[j]
				del thisNums[i]
				if oop[pos] == '*':
					thisNums.append(curNums[i]*curNums[j])
				else:
					thisNums.append(curNums[i]+curNums[j])
				bestHere = min(bestHere,getBest(thisNums,pos+1))
#		print "getBest",curNums,pos,bestHere
		return bestHere

print getBest(nums,0)
