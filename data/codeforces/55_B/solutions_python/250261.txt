from itertools import *
l = map(int,raw_input().strip().split())
ops = raw_input().strip().split()

op = lambda o, a, b: a*b if o=='*' else a+b

print min(\
    min(op(ops[2],op(ops[1],op(ops[0],p[0],p[1]),p[2]),p[3]) for p in permutations(l,4)),\
    min(op(ops[2],op(ops[0],p[0],p[1]),op(ops[1],p[2],p[3])) for p in permutations(l,4)))
