import sys
from itertools import permutations
def readint(): return int(raw_input())
def readarray(foo): return [foo(x) for x in raw_input().split()]


def do(a, b, o):
	return (a + b) if o == "+" else (a * b)

def run():
	a = readarray(int)
	op = readarray(str)	
	r = []
	for p in permutations(a):
		sum = p[0]
		for i, o in enumerate(op):
			sum = do(sum, p[i + 1], o)
		r.append(sum)
		r.append(do(do(p[0], p[1], op[0]), do(p[2], p[3], op[1]), op[2]))
	print min(r)
run()
