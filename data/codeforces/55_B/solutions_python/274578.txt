from itertools import combinations
four, op, ans = raw_input ().split (), raw_input ().split (), 2 ** 64
for i, j in combinations (range (4), 2):
    three = four[:]
    three[i] = str (eval (three[i] + op[0] + three.pop (j))) 
    for i, j in combinations (range (3), 2):
        two = three[:]
        two[i] = str (eval (two[i] + op[1] + two.pop (j)))
        ans = min (ans, eval (two[0] + op[2] + two[1]))
print ans
