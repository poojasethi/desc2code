#!/usr/bin/env python
# -*- coding: utf-8 -*-
import sys

a = map(int, sys.stdin.readline().split())
op = sys.stdin.readline().split()
m = 10 ** 13
def r(a, k):
    global m
    l = len(a)
    if l == 1 and a[0] < m:
        m = a[0]
    for i in xrange(l):
        for j in xrange(l):
            if i == j:
                continue
            if op[k] == '+':
                b = [a[i] + a[j]]
            elif op[k] == '*':
                b = [a[i] * a[j]]
            for q, c in enumerate(a):
                if q != i and q != j:
                    b.append(c)
            r(b, k + 1)
r(a, 0)
print m
