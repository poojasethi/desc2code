a=map(int,raw_input().split())
o=raw_input().split()
t=1000000000000000007L
def search(x,p=0):
  global t
  if p<3:
    for i in range(len(x)):
      for j in range(i+1,len(x)):
        y=[(x[i]+x[j]) if '+'==o[p] else (x[i]*x[j])]
        y+=[x[u] for u in range(len(x)) if u!=i and u!=j]
        search(y,p+1)
  else:
    t=min(t,x[0])
search(a)
print t
