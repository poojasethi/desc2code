#!/usr/bin/python
import sys

def recur (a, s):
	if s == '':
		assert len (a) == 1
		return a[0]
	b = []
	n = len (a)
	for i in range (n - 1):
		for j in range (i + 1, n):
			if s[0] == '+':
				k = a[i] + a[j]
			else:
				k = a[i] * a[j]
			c = [k] + a[:i] + a[i + 1:j] + a[j + 1:]
			b += [recur (c, s[1:])]
	return min (b)

while True:
	s = sys.stdin.readline ().strip ()
	if s == '':
		s = sys.stdin.readline ().strip ()
		if s == '':
			break

	a = [int (x) for x in s.split ()]
	s = ''.join (sys.stdin.readline ().split ())
	print recur (a, s)
