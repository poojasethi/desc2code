from itertools import *
n,r=map(int,raw_input().split())
if n%2==0:
	print (n*r)**2
	for i in xrange(n/2):
		print r,0
		print -r,0
	exit()
p=[[x*x+y*y,x,y]for x in xrange(-r,r+1) for y in xrange(-r,r+1) if (x*x+y*y<=r*r)]
p.sort()
p=p[2*n-30:]
ans,rs=0,[]
for d in combinations(range(len(p)+n-1),n):
	q=[p[d[i]-i][1:3]for i in xrange(n)]
	d=sum((q[i][k]-q[j][k])**2 for i in xrange(n)for j in xrange(i+1,n)for k in[0,1])
	if d>ans: ans,rs=d,q
print ans
for i in xrange(n):print rs[i][0],rs[i][1]
