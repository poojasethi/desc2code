#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<vector>
#include<algorithm>
#include<climits>
#define delta 300
#define x first
#define y second
using namespace std;
bool vis[10][600][600];
int dp[10][600][600];
vector<pair<int,int> > P;
int n,r;
int dfs(int rem,int sumx,int sumy)
{
	if(rem==0)return dp[rem][sumx+delta][sumy+delta]=-(sumx*sumx+sumy*sumy);
	int& ret=dp[rem][sumx+delta][sumy+delta];
	if(vis[rem][sumx+delta][sumy+delta])return ret;
	vis[rem][sumx+delta][sumy+delta]=1;
	ret=INT_MIN;
	for(int i=0;i<P.size();i++)ret=max(ret,n*(P[i].x*P[i].x+P[i].y*P[i].y)+dfs(rem-1,sumx+P[i].x,sumy+P[i].y));
	return ret;
}
void print(int rem,int sumx,int sumy)
{
	if(rem==0)return;
	for(int i=0;i<P.size();i++)
		if(dp[rem-1][sumx+P[i].x+delta][sumy+P[i].y+delta]+n*(P[i].x*P[i].x+P[i].y*P[i].y)==dp[rem][sumx+delta][sumy+delta])
		{
			cout<<P[i].x<<' '<<P[i].y<<'\n';
			print(rem-1,sumx+P[i].x,sumy+P[i].y);
			return;
		}
}
int main()
{
	cin>>n>>r;
	P.clear();
	for(int x=-r;x<=r;x++)
		for(int y=-r;y<=r;y++)
			if(x*x+y*y+8>=r*r&&x*x+y*y<=r*r)P.push_back(make_pair(x,y));
	memset(vis,0,sizeof(vis));
	cout<<dfs(n,0,0)<<'\n';
	print(n,0,0);
	return 0;
}
