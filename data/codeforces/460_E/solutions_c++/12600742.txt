#include<stdio.h>
#include<stdlib.h>
#include<iostream>
using namespace std;
int tot; 
int a[5000],b[5000],d[5000],c[10],getans[10],ans,num,n; 
int getdis()
{
    int temp=0,i,j; 
    for (i=1; i<=n; i++)
       for (j=i+1; j<=n; j++)
       temp+=((a[c[i]]-a[c[j]])*(a[c[i]]-a[c[j]])+(b[c[i]]-b[c[j]])*(b[c[i]]-b[c[j]])); 
    return temp; 
} 

int search(int p,int k)
{
    int tt,i; 
    if (k==0)
    {
        tt=getdis(); 
        if (tt>ans)
        {
            ans=tt; 
            for (i=1; i<=n; i++)
            getans[i]=c[i]; 
        }
        return 1; 
    }
    if (p>num) return 1; 
    search(p+1,k); 
    tot++; 
    c[tot]=p; 
    search(p,k-1);
    c[tot]=0; 
    tot--;  
}

int main()
{
    int r,x,y,i,j,t; 
    cin>>n>>r; 
    tot=0; 
    for (x=-r; x<=r; x++)
       for (y=-r; y<=r; y++) 
       if (x*x+y*y<=r*r)
       {
         tot++; 
         a[tot]=x; 
         b[tot]=y; 
         d[tot]=x*x+y*y; 
       }
    for (i=1; i<=tot; i++)
       for (j=i+1; j<=tot; j++) 
       if (d[i]<d[j])
       {
          t=a[i]; a[i]=a[j]; a[j]=t; 
          t=b[i]; b[i]=b[j]; b[j]=t; 
          t=d[i]; d[i]=d[j]; d[j]=t; 
       }
    if (tot<18) 
    {
        tot++; a[tot]=0; b[tot]=0; d[tot]=0; 
    }
    num=18; if (num>tot) num=tot; 
    tot=0;  ans=0; 
    for (i=1; i<=n; i++) c[i]=getans[i]=0; 
    search(1,n); 
    cout<<ans<<endl; 
    for (i=1; i<=n; i++)
    {
        cout<<a[getans[i]]<<" "<<b[getans[i]]<<endl; 
    }
    i=1; 
}
