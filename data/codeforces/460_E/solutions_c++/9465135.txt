#include<iostream>
#include<cmath>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<iomanip>
using namespace std;
struct point
{
    int x,y,d;
}a[900],p[900],t[900];
int n,r;
int ans;
int cnt;
bool cmp(point a,point b)
{
    return a.d>b.d;
}
void dfs(int f,int s)
{
    int i,j;
    if(s==n)
    {
        int add=0;
        for(i=0;i<s;i++)
            for(j=i+1;j<s;j++)
                add+=(t[i].x-t[j].x)*(t[i].x-t[j].x)+(t[i].y-t[j].y)*(t[i].y-t[j].y);
        if(add>ans)
        {
            ans=add;
            for(i=0;i<s;i++)
            {
                a[i].x=t[i].x;
                a[i].y=t[i].y;
            }
        }
        return;
    }
    for(i=f;i<cnt;i++)
    {
        t[s].x=p[i].x;
        t[s].y=p[i].y;
        dfs(i,s+1);
    }
}
int main()
{
    cin>>n>>r;
    for(int i=-r;i<=r;i++)
        for(int j=-r;j<=r;j++)
        {
            if(i*i+j*j<=r*r&&i*i+j*j>(r-1)*(r-1))
            {
                p[cnt].x=i;
                p[cnt].y=j;
                p[cnt++].d=i*i+j*j;
            }
        }
    sort(p,p+cnt,cmp);
    cnt=17;
    dfs(0,0);
    cout<<ans<<endl;
    for(int i=0;i<n;i++)
        cout<<a[i].x<<' '<<a[i].y<<endl;
return 0;
}
