#include<iostream>
#include<cstring>
#include<string>
#include<cmath>
#include<algorithm>
#include<cstdio>
#include<cstdlib>
using namespace std;

struct re{int x,y,l;};
re a[1001];
int k,m,n,i,tot,r,ans,ansa[101],ansb[101],ga[101],gb[101];

inline bool cmp(re a,re b) {return a.l>b.l;}
inline int sq(int x) {return x*x;};
inline void add(int x,int y)
{
	a[++tot].x=x;
	a[tot].y=y;
	a[tot].l=x*x+y*y;
}

void work(int k,int t,int tt)
{
	int i;
	if (t==n)
	{
		if (tt>ans)
		{
			ans=tt;
			for (i=1;i<=n;i++) ansa[i]=ga[i],ansb[i]=gb[i];
		}
		return;
	}
	int s=0;
	for (i=1;i<=t;i++)
		s+=sq(a[k].x-ga[i])+sq(a[k].y-gb[i]);
	for (i=t+1;i<=n;i++)
	{
		ga[i]=a[k].x;
		gb[i]=a[k].y;
		work(k+1,i,tt+s*(i-t));
	}
	if (m-k+t>=n) work(k+1,t,tt);
}

int main()
{
	scanf("%d%d",&n,&r);
	add(-r,0);
	add(r,0);
	for (i=-r+1;i<r;i++)
	{
		k=int(sqrt(r*r-i*i));
		add(i,k);
		add(i,-k);
	}
	sort(a+1,a+tot+1,cmp);
	m=min(tot,20);
	work(1,0,0);
	printf("%d\n",ans);
	for (i=1;i<=n;i++) printf("%d %d\n",ansa[i],ansb[i]);
	return 0;
}
