import string
a, b = raw_input().split()
p = max(max([int(i) for i in a]), max([int(j) for j in b])) + 1
an, bn = 0, 0
for i in range(len(a)):
    an += int(a[i])*(p**(len(a)-i-1))
for i in range(len(b)):
    bn += int(b[i])*(p**(len(b)-i-1)) 
n = ""
x = an + bn
while x > 0:
    y = str(x % p)
    n = y + n
    x = int(x / p)

print len(n)
