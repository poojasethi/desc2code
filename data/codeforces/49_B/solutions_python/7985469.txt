line = raw_input().split()
base = 0

def cal(n):
    res = 0
    while n > 0:
        res = max(res, n % 10)
        n /= 10
    return res
a, b = int(line[0]), int(line[1])
base = max(base, cal(a)+1)
base = max(base, cal(b)+1)
car, lene= 0, 0
while a != 0 or b != 0:
    tmp = a % 10 + b % 10 + car
    car = tmp / base
    a /= 10
    b /= 10
    lene += 1
if car > 0:
    lene += 1

print lene
