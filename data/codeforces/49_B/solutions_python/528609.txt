s = raw_input().split()
mx, up = -1, 0
if len(s[0])>len(s[1]): s.reverse()
s[0] = (len(s[1])-len(s[0]))*'0' + s[0]
for i in xrange(len(s[0])):
    mx = max(mx,int(s[0][i]))
    mx = max(mx,int(s[1][i]))
p = mx + 1
for i in xrange(len(s[0])):
    if int(s[0][len(s[0])-1-i]) + int(s[1][len(s[0])-1-i]) + up >= p:    up = 1
    else: up = 0
print len(s[0]) + up
