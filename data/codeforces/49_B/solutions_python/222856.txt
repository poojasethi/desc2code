
if __name__ == '__main__':
    s1, s2 = raw_input().split()
    c = ord(max(s1 + s2)) - ord('0') + 1
    if len(s1) < len(s2):
        s1 = s1.rjust(len(s2), '0')
    else:
        s2 = s2.rjust(len(s1), '0')
    n1 = map(int, s1)
    n2 = map(int, s2)
    p = 0
    for a, b in zip(n1[::-1], n2[::-1]):
        if a + b + p >= c:
            p = 1
        else:
            p = 0
    print len(n1) + p
