#!/usr/bin/env python
# -*- coding: utf-8 -*-

def test_len(n1, n2, base):
    if len(n1) > len(n2):
        s = map(int, n1)
        n = n2
    else:
        s = map(int, n2)
        n = n1
    s.insert(0, 0)
    for i in xrange(len(n)):
        s[-i-1] += int(n[-i-1])
    for i in xrange(1, len(s)+1):
        while s[-i] >= base:
            s[-i-1] += 1
            s[-i] -= base
    if s[0] == 0:
        s.pop(0)
    return len(s)

def main():
    a, b = raw_input().split()
    mbase = max([int(s) for s in a + b])
    print max([test_len(a, b, base) for base in xrange(16, mbase, -1)])
    return 0

if __name__ == '__main__':
    main()
