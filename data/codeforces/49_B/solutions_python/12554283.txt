[a,b]=raw_input().split();
maxi=0;
for i in a:
    if(i>maxi):
        maxi=i;
for i in b:
    if(i>maxi):
        maxi=i;
base=int(maxi)+1;
deca=0;
decb=0;
multi=len(a)-1;
for i in a:
    deca+=int(i)*pow(base,multi);
    multi-=1;
    
multi=len(b)-1;
for i in b:
    decb+=int(i)*pow(base,multi);
    multi-=1;

sum=deca+decb;

digits=1;
while(sum>=pow(base,digits)):
    digits+=1;


print(digits)
