import string

s = raw_input()
a, b = [i for i in s.split()]
base = max(sorted(i, reverse=True)[0] for i in s.split())
if base in string.digits:
    base = ord(base) - ord('0') + 1 
else:
    base = ord(base) - ord('A') + 1 
base10 = int(a, base) + int(b, base)

sol = ""
while base10:
    sol = str(base10 % base) + sol 
    base10 /= base

print len(sol)
