#!/usr/bin/python

def pos_to_10(n, p):
    x = 0
    pp = 1
    for i in reversed(list(n)):
        r = int(i)
        x += r * pp
        pp *= p
    return x

a, b = raw_input().split()
p = max([int(c) for c in a + b]) + 1
a, b = pos_to_10(a, p), pos_to_10(b, p)
s = a + b
l = 0
while s > 0:
    s /= p
    l += 1
print l
