#include<stdio.h>
int main()
{
	__int64 a,b,x,y,i,k,j,d;
	scanf("%I64d%I64d",&a,&b);
	x=a;y=b;k=0;
	for(i=0;x!=0;i++)
	{
		if(x%10>k)k=x%10;
		x/=10;
	}
	for(j=0;y!=0;j++)
	{
		if(y%10>k)k=y%10;
		y/=10;
	}
	k+=1;d=1;
	for(i=0;a!=0;i++)
	{
		x+=a%10*d;
		a/=10;d*=k;
	}
	d=1;
	for(i=0;b!=0;i++)
	{
		y+=b%10*d;
		b/=10;d*=k;
	}
	x+=y;
	for(i=0;x!=0;i++)
		x/=k;
	printf("%I64d\n",i);
	return 0;
}
