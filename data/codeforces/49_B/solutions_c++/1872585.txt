#include <stdio.h>
#include <stdlib.h>

int main()
{
	char a[10];
	char b[10];
	char *erra;
	char *errb;
	int x;
	int y;
	int ans;
	int ansC;
	int i;

	scanf("%s", a);
	scanf("%s", b);

	for (i = 2; i <= 36; i++)
	{
		x = strtol(a, &erra, i);
		y = strtol(b, &errb, i);
		if (erra[0] == '\0' && errb[0] == '\0')
			break;
	}

	ans = x + y;
	ansC = 0;
	while (ans)
	{
		ansC++;
		ans /= i;
	}
	printf("%d\n", ansC);
	return 0;
}
