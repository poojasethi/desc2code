#include <iostream>

using namespace std;

int pos(int a)
{
	int p = 2;
	while (a != 0)
	{
		p = max(p, a % 10 + 1);
		a /= 10;
	}

	return p;
}

int dec(int a, int p)
{
	int r = 0;
	int m = 1;
	while (a != 0)
	{
		r += (a % 10) * m;
		m *= p;
		a /= 10;
	}
	return r;
}

int main(int argc, char* agrv[])
{
	int a, b;
	cin >> a >> b;

	int p = max(pos(a), pos(b));
	int c = dec(a, p) + dec(b, p);

	int r = 0;
	while (c != 0)
	{
		c /= p;
		r++;
	}

	cout << r << endl;

	return 0;
}
