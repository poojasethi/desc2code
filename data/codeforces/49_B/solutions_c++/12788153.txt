#include <bits/stdc++.h>
using namespace std;

int a,b;

int main () {

    cin >> a >> b;

    int base=0,x;
    x=a; while (x) {base = max(base,x%10); x/=10;}
    x=b; while (x) {base = max(base,x%10); x/=10;}
    base++;

    int t=0;
    while (a || b) {
        //r = (a%10+b%10+r)/base;
        //a/=10; b/=10;
        //a+=r;
        if ( (a%10+b%10)/base ) {
            a/=10; b/=10; a++;
        } else {a/=10; b/=10;}
        t++;
    }

    cout << t << endl;
}
