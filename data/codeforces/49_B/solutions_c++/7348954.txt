#include<iostream>

using namespace std;

int mn( int a )
{
    int s = 0;
    while( a )
    {
           s = max( s , a%10 );
           a/=10;
    }
    return s;
}

int x( int a , int c )
{
    if( a==0 ) return 0;
    return a%10 + c*x( a/10,c );
}

int main()
{
    int a ,b;
    cin >> a >> b;
    int c = max( mn( a ) , mn( b ) );
    c++;
    int ans = x( a , c ) + x( b , c );
    int s = 0;
    while( ans )
    {
           ans/=c;
           s++;
    }
    cout << s << endl;
}
