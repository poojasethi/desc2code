#include<iostream>
#include<cstdio>
using namespace std;
int main()
{ int a,b,i,res,t, max=0;;
    int ans=0;
    cin>>a>>b;
    t=a;
    while(t)
    {  res=t%10;  if(res>=max) max=res+1;
        t/=10;
    }
    t=b;
    while(t)
    {
        res=t%10; if(res>=max) max=res+1;
        t/=10;
    }
  res=0;
    while(a||b||res)
    {
        res=(a%10+b%10+res)/max;
        a/=10; b/=10;

        ans++;

    }
cout<<ans;
    return 0;
}
