#include <iostream>

using namespace std;

int m=0;

inline void find (int k)
{
	while (k>0)
	{
		if (k%10>m)
			m=k%10;
		k/=10;
	}
}

inline int toten (int a , int base)
{
	int ans=0,k=1;
	while (a)
	{
		ans+=(a%10)*k;
		k*=base;
		a/=10;
	}
	return ans;
}

int main()
{
	int a,b,ans=0;
	cin >> a >> b;
	find(a);
	find(b);
	m++;
	a=toten(a,m);
	b=toten(b,m);
	a+=b;
	while (a)
	{
		a/=m;
		ans++;
	}
	cout << ans << endl;
	return 0;
}
