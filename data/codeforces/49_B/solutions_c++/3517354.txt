#include<stdio.h>
#include<algorithm>
#include<string.h>
using namespace std;
int main()
{
	int n,m;
	while(scanf("%d%d",&n,&m)!=-1)
	{
		int i,j,t=0,a[111]={0},b[111]={0},c=0,l;
		for(i=0;n>0;i++)
		{
			a[i]=n%10;
			n/=10;
			if(a[i]>t)
				t=a[i];
		}
		l=i;
		for(i=0;m>0;i++)
		{
			b[i]=m%10;
			m/=10;
			if(b[i]>t)
				t=b[i];
		}
		if(i>l)
			l=i;
		for(i=0;i<l;i++)
		{
			if(a[i]+b[i]+c>t)
				c=1;
			else
				c=0;
		}
			if(c==1)
				printf("%d\n",l+1);
			else
				printf("%d\n",l);
	}
	return 0;
}
