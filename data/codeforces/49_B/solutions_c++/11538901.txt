#include<stdio.h>
char a[55],b[55];
int main()
{
scanf("%s%s",&a,&b);
int ba = -1;
int i;
for(i=0;a[i];i++)
if(a[i] -'0' > ba)ba =a[i]-'0';
for(i=0;b[i];i++)
if(b[i]-'0'>ba)ba = b[i]-'0';
ba++;
int fir,sec;
fir = sec = 0;
for(i=0;a[i];i++)
fir = fir*ba + a[i]-'0';
for(i=0;b[i];i++)
sec = sec*ba + b[i] -'0';
fir = fir+sec;
int ans = 0;
while(1)
{
fir = fir/ba;
ans++;
if(fir==0)break;
}
printf("%d\n",ans);
return 0;
}
