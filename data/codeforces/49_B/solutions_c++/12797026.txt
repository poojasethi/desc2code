#include <iostream>
using namespace std;

int main(){
    int a,b,r=0,m=0,k=0;
    cin>>a>>b;
    int i=a;
    while(i){
        m=max(m,i%10+1);
        i/=10;
    }
    i=b;
    while(i){
        m=max(m,i%10+1);
        i/=10;
    }

    while(a||b||r){
        i*=10;
        i+=(a%10+b%10+r)%m;
        r=(a%10+b%10+r)/m;
        a/=10;
        b/=10;
        k++;
    }
    cout<<k;
    return 0;

}
