#include <cstdio>
#include <vector>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <cctype>
using namespace std;

int calc(int a) {
	int ret = 0;
	while(a) {
		ret = max(ret , a % 10);
		a /= 10;
	}
	return ret + 1;
}
int main() {
	int a , b;
	while(~scanf("%d%d",&a,&b)) {
		int base = max(calc(a) , calc(b));
		int add = 0;
		int cnt = 0;
		while(a || b || add) {
			int x = a % 10 + b % 10 + add;
			add = x / base;
			a /= 10;
			b /= 10;
			cnt ++;
		}
		printf("%d\n",cnt);
	}
	return 0;
}
