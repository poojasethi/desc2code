#include<stdio.h>

char a[20];
char b[20];
int max;
void input()
{
	int i, j;
	scanf("%s %s",a,b);
	for(i=0; a[i]; i++)
	{
		if(max < a[i])
			max = a[i];
	}
	for(i=0; b[i]; i++)
	{
		if(max < b[i])
			max = b[i];
	}
	max -= '0';
	max++;
}
void process()
{
	int i, j, k;
	for(i=j=0; a[i]; i++)
	{
		j*=max;
		j+=a[i]-'0';
	}
	for(i=k=0; b[i]; i++)
	{
		k*=max;
		k+=b[i]-'0';
	}
	j+=k;
	k=0;
	while(j)
	{
		k++;
		j/=max;
	}
	printf("%d\n",k);
}
int main(){
	input();
	process();
	return 0;
}
