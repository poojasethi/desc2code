#include <iostream>
using namespace std;
int main()
{
	string a, b;
	int m=0, i, aa=0, bb=0;
	cin>>a>>b;
	for (i=0; i<a.size(); i++) m=max(m, a[i]-'0');
	for (i=0; i<b.size(); i++) m=max(m, b[i]-'0');
	if (m>9) m=m-'A'+'0'+10;
	m++;
	for (i=0; i<a.size(); i++) aa = (a[i]>'9') ? aa*m+a[i]-'A'+10 : aa*m+a[i]-'0';
	for (i=0; i<b.size(); i++) bb = (b[i]>'9') ? bb*m+b[i]-'A'+10 : bb*m+b[i]-'0';
	aa+=bb;
	for (i=0; aa>0; i++) aa/=m;
	if (i==0) i=1;
	cout<<i<<endl;
	return 0;
}
