#include<iostream>
#include<cmath>
using namespace std;
int main()
{
int a , b ,c ,p,h=0,arra[20]={ 0 },arrb[20]={ 0 };
cin>>a>>b;
p=0;
int i=a;
int sizea=0,sizeb=0;
while(i){
if(i%10 > p) p=i%10;
arra[sizea++]=i%10;
i/=10;
}
i=b;
while(i){
if(i%10 > p) p=i%10;
arrb[sizeb++]=i%10;
i/=10;
}
int carry=0;
p++;
for(i=0; i< max(sizea,sizeb); ++i){
carry=(arra[i]+arrb[i]+carry)/p;
h++;
}
if(carry>0) h++;
cout<<h<<"\n";
return 0;
}
