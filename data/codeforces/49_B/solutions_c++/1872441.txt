#include<stdio.h>
#include<stdlib.h>
int main()
{
	int a,b,max,c,d,aa[10]={0},bb[10]={0},t=0,t2=0,i,v,l=0;
   scanf("%d%d",&a,&b);
   max=a%10;c=a;d=b;
   while(c)
   {
     if(c%10>max)max=c%10;
	 aa[t++]=c%10;
	 c=c/10;
   }
   while(d)
   {
     if(d%10>max)max=d%10;
	 bb[t2++]=d%10;
	 d=d/10;
   }
   if(t2>t)t=t2;
   max++;v=0;
  for(i=0;i<t;i++)
  {
    int q=aa[i]+bb[i]+v;
    v=q/max;
	l++;
  }
  if(v)l++;
   printf("%d\n",l);
}
