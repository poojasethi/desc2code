#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
int Get(int x){
	int cnt=0;
	while(x>0) cnt=max(x%10,cnt),x/=10;
	return(cnt+1);
}
int main(){
	int a,b,cnt,cur,tmp,d;
	scanf("%d%d",&a,&b);
	d=max(Get(a),Get(b));
	tmp=cnt=0;
	//printf("%d\n",d);
	while(a>0||b>0){
		cnt++;cur=0;
		if(a>0) tmp+=(a%10),a/=10;
		if(b>0) tmp+=(b%10),b/=10;
		//printf("%d %d %d\n",a,b,tmp);
		if(tmp>=d) tmp=1,cur=1;
			else tmp=0;
	}
	cnt+=cur;
	printf("%d\n",cnt);
}
