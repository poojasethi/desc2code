#include<stdio.h>
#include<algorithm>

using namespace std;

int main(){
	int x,y,z,t,base=0;
	scanf("%d%d",&x,&y);
	
	z = x;
	while(z){
		base = max(base,z%10);
		z = z/10;
	}
	z = y;
	while(z){
		base = max(base,z%10);
		z = z/10;
	}

	base++;

	z = 0;
	t = 1;
	while(x){
		z+= x%10*t;
		t*=base;
		x/=10;
	}
	t = 1;
	while(y){
		z+= y%10*t;
		t*=base;
		y/=10;
	}

	t=0;
	while(z){
		z/=base;
		t++;
	}

	printf("%d\n",t);

	return 0;
}
