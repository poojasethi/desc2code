#include <iostream>
using namespace std;
int a[12];
int get(int x)
{
	int ans = 0;
	while (x)
	{
		ans = max(ans, x % 10);
		x /= 10;
	}
	return ans;
}
int main()
{
	ios::sync_with_stdio(false);
	int x, y;
	cin >> x >> y;
	int p = max(get(x), get(y)) + 1;
	for (int i = 0; i < 10; i++)
	{
		a[i] = x % 10 + y % 10;
		x /= 10;
		y /= 10;
	}
	int res = 0;
	for (int i = 0; i < 10; i++)
	{
		a[i + 1] += a[i] / p;
		a[i] %= p;
		if (a[i])
			res = i + 1;
	}
	cout << res << endl;
	return 0;
}
