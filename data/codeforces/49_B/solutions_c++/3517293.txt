#include<cstdio>
int cal(int x,int y,int b){
	int l=0,t1,t2,t=0;
	while(x||y){
		t1=x%10,t2=y%10;
		x/=10,y/=10;
		t=(t1+t2+t)/b;
		l++;
	}
	return t?l+1:l;
}
int main(){
	int a,b,c,B,t;
	while(~scanf("%d%d",&a,&b)){
		B=0;c=a+b;
		t=a;while(t){if(t%10>B)B=t%10;t/=10;}
		t=b;while(t){if(t%10>B)B=t%10;t/=10;}
		B++;
		printf("%d\n",cal(a,b,B));
	}
	return 0;
}
