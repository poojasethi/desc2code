#include <iostream>
#include <algorithm>

using namespace std;

int A,B,base,result;

int max_digit(int x)
{
	int ret=0;
	while (x)
	{
		ret=max(ret,x%10);
		x/=10;
	}
	return ret;
}

int main()
{
	cin >> A >> B;
	base=max(max_digit(A),max_digit(B))+1;
	int add=0;
	while (A||B||add)
	{
		int x=A%10+B%10+add;
		add=x/base;
		result++;
		A/=10;
		B/=10;
	}
	cout << result << endl;
	return 0;
}
