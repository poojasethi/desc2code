#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int n1,n2,maxD=0,carry=0,ind=0,count0=0;
    scanf("%d%d",&n1,&n2);
    int m1=n1,m2=n2;
    while(m1 || m2)
    {
        maxD=max(maxD,m1%10);
        maxD=max(maxD,m2%10);
        m1/=10,m2/=10;
    }
   maxD+=1;
   while(n1||n2||carry)
   {
        ind*=10;
        ind+=((n1%10)+(n2%10)+carry)%maxD;
        carry=((n1%10)+(n2%10)+carry)/maxD;
        n1/=10,n2/=10;
        count0++;
   }
   printf("%d\n",count0);
return 0;
}
