#include<iostream>
using namespace std;
int main()
{
	string a,b;
	cin>>a>>b;
	bool w=true;
	int maxx=0;
	for(int i=0;i<a.size();i++)maxx=max(maxx,a[i]-'0');
	for(int i=0;i<b.size();i++)maxx=max(maxx,b[i]-'0');
	while(b.size()<a.size())b='0'+b;
	while(a.size()<b.size())a='0'+a;
	//cout<<a<<" "<<b<<endl;
	int i=0;
	while(a[i]+b[i]-'0'-'0'==maxx)i++;
	if(a[i]+b[i]-'0'-'0'<maxx+1)cout<<max(a.size(),b.size())<<endl;
	else cout<<max(a.size(),b.size())+1<<endl;
}
