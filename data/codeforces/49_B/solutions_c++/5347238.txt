#include <stdio.h>
int x(int a,int b){return a>b?a:b;}
int s(int a,int d)
{
    int r=0,b=1;
    do r+=a%10*b,a/=10,b*=d;while(a>0);
    return r;
}
int g(int a)
{
    int t=0;
    do t=x(t,a%10),a/=10; while(a>0);
    return t;
}
int main()
{
    int a,b,d,c,l;
    while(scanf("%d%d",&a,&b)>0)
    {
        d=x(g(a),g(b))+1;
        a=s(a,d);b=s(b,d);
        c=a+b;l=0;
        do l++,c/=d;while(c>0);
        printf("%d\n",l);
    }
}
