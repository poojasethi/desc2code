#include <cstdio>
#include <cstring>
int a,b,c,d,date[100];
int main ()
{
   //freopen ("con","r",stdin);
   scanf("%d%d",&a,&b);
   int i,tem;
   int maxe=0;
   for (i=1;a||b;i++)
   {
      c=a%10;
      d=b%10;
      date[i]=c+d;
      a/=10;
      b/=10;
      tem=c>d?c:d;
      maxe=maxe>tem?maxe:tem;
   }
   int jinzhi=maxe+1;
   for (int j=1;j<i;j++)
   {
      if (date[j]>=jinzhi) date[j+1]++;
   }
   if (date[i]) printf("%d\n",i);
   else {i--;printf("%d\n",i);}
   return 0;
}
