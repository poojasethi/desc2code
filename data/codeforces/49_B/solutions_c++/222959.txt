#include<string.h>
#include<stdlib.h>

int main(){
	int a,b,t,p,x;
	scanf("%d%d",&a,&b);
	t=0; x=a;
	while(x) { t=t<x%10?x%10:t; x/=10; }
	x=b;
	while(x) { t=t<x%10?x%10:t; x/=10; }
	p=t+1;
	t=1; x=0;
	while(a){ x+=(a%10)*t; t*=p; a/=10; }
	t=1; while(b){ x+=(b%10)*t; t*=p; b/=10; }
	t=0; //printf("%d",x);
	while(x) { t++; x/=p; }
	printf("%d\n",t);
	return 0;
}
