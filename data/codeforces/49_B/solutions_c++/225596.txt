#include <iostream>

using namespace std;

int main() {
    int maxi = 2;

    int a, b;
    cin >> a >> b;

    int a2 = a, b2 = b;
    while(a2 > 0) maxi = max(maxi, a2 % 10 + 1), a2 /= 10;
    while(b2 > 0) maxi = max(maxi, b2 % 10 + 1), b2 /= 10;

    int c = 0, ans = 0;
    while(a > 0 || b > 0 || c > 0) {
        int s = (a%10 + b%10 + c);
        c = s/maxi; ans++;
        a /= 10; b /= 10;
    }

    cout << ans << endl;
}
