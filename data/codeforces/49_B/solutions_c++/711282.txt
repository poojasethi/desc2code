#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int maxdig(int n) {
    if (n==0) return 0;
    int m = 0;
    while (n>0) {
        m = max(m, n%10);
        n /= 10;
    }
    return m;
}

int main() {

    int a, b;

    scanf("%d %d",&a,&b);
    int base = max(maxdig(a),maxdig(b)) + 1;
    int carry = 0;
    int digs=0;
    while (a > 0 or b > 0 or carry > 0) {
        digs++;
        carry = (carry+(a%10)+(b%10))/base;
        a/= 10;
        b/= 10;
    }
    printf("%d\n",digs);

    return 0;
}
