#include <iostream>
using namespace std;

int base(int n)
{
	int m=0;
	while (n){
		m=max(m,n%10);
		n/=10;
	}
	return m;
}

int main()
{
	int a,b,ba,add,x,ans;
	while (cin>>a>>b)
	{
		ba=max(base(a), base(b))+1;
		add=0,ans=0;
		while (a || b|| add)
		{
			x=a%10+b%10+add;
			add=x/ba;
			ans++;
			a/=10;
			b/=10;
		}
		cout<<ans<<endl;
	}
	return 0;
}
