#include<iostream>
using namespace std;
int maxi(int n){
	int z=0;
	while(n!=0){
		z=max(z,n%10);
		n/=10;
	}
	return z;
}
int main(){
	
	int x,y,s,ans,z,k=0,l=0,i=1;
	
	cin>>x>>y;
	ans=0;
	s=max(maxi(x),maxi(y))+1;
	while(x){
		k+=x%10*i;
		x/=10;
		i*=s;
	}
	i=1;
	while(y){
		l+=y%10*i;
		y/=10;
		i*=s;
	}
	z=k+l;
	while(z){
		z/=s;
		ans++;
	}
	cout<<ans<<endl;
}
