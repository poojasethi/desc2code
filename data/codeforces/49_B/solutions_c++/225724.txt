#include <iostream>
using namespace std;
int a, b, A, B, p, k, i = 1;
int main()
{
    cin >> a >> b; A = a; B = b;
    while(a | b) {
        if(a)p = max(a % 10, p), a /= 10;
        if(b)p = max(b % 10, p), b /= 10;
    }
    p++;
    while(A) k += i * (A % 10), i *= p, A /= 10;
    i = 1;
    while(B) k += i * (B % 10), i *= p, B /= 10;
    i = 0;
    while(k) k /= p, i++;
    cout << i;
    return 0;
}
