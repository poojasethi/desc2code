n = input ()
targets = sorted ([map (eval, raw_input ().split ()) for _ in xrange (n)], key=lambda x: x[2])
E = [targets[i][3] for i in xrange (n)]
d = lambda a, b: ((a[0] - b[0]) ** 2 + (a[1] - b[1]) ** 2) ** 0.5
for i in xrange (1, n):
    for j in xrange (i):
        if targets[j][2] + d (targets[i], targets[j]) <= targets[i][2]:
            E[i] = max (E[i], E[j] + targets[i][3])
print max (E)
