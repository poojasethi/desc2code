#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<cstdlib>
using namespace std;
typedef long long ll;
int x[1000], y[1000];
ll t[1000];
double p[1000], dp[1000];
int n;

#define sqr(x) ((x) * (x))

double solve(int u){
    double &res = dp[u];
    if(res >= 0) return res;
    res = p[u];
    for(int v = 0; v < n; v++) if(t[v] > t[u] && sqr(x[u] - x[v]) + sqr(y[u] - y[v]) <= sqr(t[u] - t[v]))
        res = max(res, p[u] + solve(v));
    return res;
}
    
int main(){
    cin >> n;
    for(int i = 0; i < n; i++) scanf("%d %d %lld %lf", &x[i], &y[i], &t[i], &p[i]);
    
    for(int i = 0; i < n; i++) dp[i] = -1;
    
    double res = 0;
    for(int i = 0; i < n; i++) res = max(res, solve(i));
    
    printf("%.010lf\n", res);
    
}
