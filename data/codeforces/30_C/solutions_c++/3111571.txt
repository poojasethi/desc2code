#include<iostream>
#include<cstring>
#include<iomanip>
#include<algorithm>
using namespace std;
const int mm=1100;
const double ex=-0.0000001;
class node
{
public:
  double x,y,p,t;
}f[mm];
double dp[mm];
bool cmp(node a,node b)
{
  return a.t<b.t;
}
inline double sqr(double x)
{
  return x*x;
}
int n;
int main()
{
  while(cin>>n)
  {
    for(int i=0;i<n;i++)
      cin>>f[i].x>>f[i].y>>f[i].t>>f[i].p;
    double sum=0.0;
    sort(f,f+n,cmp);
    for(int i=0;i<n;i++)
    {
      dp[i]=f[i].p;
      for(int j=0;j<i;j++)
        {
          if(sqr(f[i].t-f[j].t)-sqr(f[i].x-f[j].x)-sqr(f[i].y-f[j].y)>ex)
            dp[i]=dp[i]>dp[j]+f[i].p?dp[i]:dp[j]+f[i].p;
        }
        sum=sum>dp[i]?sum:dp[i];
    }
    cout.setf(ios::fixed);
    cout<<setprecision(8)<<sum<<"\n";
  }
}
