#include<bits/stdc++.h>
using namespace std;
const int maxn = 1010;
struct point
{
	int x,y,t;
	double p;
}tt[maxn];
bool cmp(point a,point b)
{
	return a.t<b.t;
}
double dis(int a,int b)
{
	return sqrt(0.0+tt[a].x*tt[a].x+tt[b].x*tt[b].x-2*tt[a].x*tt[b].x+tt[a].y*tt[a].y+tt[b].y*tt[b].y-2*tt[a].y*tt[b].y);
}
double dp[maxn];
int n;
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d%d%d%lf",&tt[i].x,&tt[i].y,&tt[i].t,&tt[i].p);
	sort(tt,tt+n,cmp);
	dp[0]=tt[0].p;
	for(int i=1;i<n;i++)
	{
		dp[i]=tt[i].p;
		for(int j=i-1;j>=0;j--)
			if(dis(i,j)<=tt[i].t-tt[j].t)
		    	dp[i]=max(dp[i],dp[j]+tt[i].p);
	}
	double ans=0;
	for(int i=0;i<n;i++)
		ans=max(ans,dp[i]);
	printf("%.12f\n",ans);
	return 0;
}
