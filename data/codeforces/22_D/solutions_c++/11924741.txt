#include<bits/stdc++.h>
using namespace std;
const int maxn = 1010;
int n;
struct node
{
    int l,r;
}tt[maxn];
bool cmp(node a,node b)
{
    return a.r<b.r||(a.r==b.r&&a.l>b.l);
}
int ans[maxn],cur;
int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d%d",&tt[i].l,&tt[i].r);
        if(tt[i].l>tt[i].r)
            swap(tt[i].l,tt[i].r);
    }
    cur=0;
    sort(tt,tt+n,cmp);
    int tail = -100010;
    for(int i=0;i<n;i++)
        if(tt[i].l>tail)
        {
            tail=tt[i].r;
            ans[cur++]=tail;
        }
    printf("%d\n",cur);
    for(int i=0;i<cur;i++)
        printf("%d ",ans[i]);
    puts("");
    return 0;
}
