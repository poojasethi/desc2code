#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
  int n;
  cin>>n;
  vector<pair<int,int> > v(n);
  for (int i=0;i<n;i++) {
    cin>>v[i].first>>v[i].second;
    if (v[i].first>v[i].second) swap(v[i].first,v[i].second);
  }
  int lastnail=100000;
  sort(v.begin(),v.end());
  int nails=0;
  vector<int> posnail;
  for (int i=n-1;i>=0;i--) {
    if (v[i].second<lastnail) {
      nails++;
      posnail.push_back(v[i].first);
      lastnail=v[i].first;
    }
  }
  cout<<nails<<endl;
  for (int i=int(posnail.size())-1;i>=0;i--) {
    cout<<posnail[i];
    if (i>0) cout<<" ";
  }
  cout<<endl;
}
