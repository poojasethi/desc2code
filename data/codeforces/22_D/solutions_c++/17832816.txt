#include<cstdio>
#include<algorithm>
using namespace std;
const int N=1001;
struct po{
	int l,r;
	bool operator<(const po&b)const{
		if(r==b.r)
			return l<b.l;
		return r<b.r;
	}
}p[N];
int a[N];
int main(){
	int n,l,r,la,cnt;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d%d",&l,&r);
		if(l>r)
			swap(l,r);
		p[i].l=l,p[i].r=r;
	}
	sort(p,p+n);
	cnt=0;
	la=-1;
	for(int i=0;i<n;){
		cnt++;
		a[++la]=p[i].r;
		i++;
		while(i<n&&p[i].l<=a[la]&&a[la]<=p[i].r)
			i++;
	}
	printf("%d\n%d",cnt,a[0]);
	for(int i=1;i<=la;i++)
		printf(" %d",a[i]);
	return 0;
}
