#include <cstdio>
#include <vector>
#include <map>
#include <algorithm>

using namespace std;

int main() {

//	freopen("D.in","r",stdin);

    int N;
    scanf("%d", &N);
    pair<int,int> V[1010];
    vector<int> RET;
    for (int i = 0; i < N; i++) {
        int a, b;
        scanf("%d %d", &a, &b);
        if (b < a) swap(a,b);
        V[i] = make_pair(b,a);
    }
    sort(V, V+N);
    int l = -101010;
    for (int i = 0; i < N; i++) {
        if (V[i].second <= l && l <= V[i].first) continue;
        l = V[i].first;
        RET.push_back(l);
    }

    printf("%d\n", RET.size());
    for (int i = 0; i < RET.size(); i++) printf("%d ", RET[i]);

    return 0;
}
