#include <iostream>
using namespace std;
int main()
{
    int n,i,j,k,res,a[1100],b[1100],r[1100];
    cin>>n;
    for (i=1; i<=n; i++) 
    { cin>>a[i]>>b[i]; if (a[i]>b[i]) swap(a[i],b[i]); }
    
    for (i=1; i<n; i++) for(j=i+1; j<=n; j++) 
    if ( a[i]>a[j] ) { swap (a[i],a[j]); swap (b[i],b[j]); }
    
    res=0; k = b[1]; 
    for (i=2; i<=n; i++)
    {
        if ( k<a[i] )  {  res++; r[res] = k; k = b[i]; }                
        else  k = min(k,b[i]);
    }
    res++; r[res]=k;
    cout<<res<<endl;
    for (i=1; i<=res; i++) cout<<r[i]<<" "; 
    //system("pause");
    return 0;
}
