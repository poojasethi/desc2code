#include <cstdio>
#include <algorithm>
using namespace std;

int N;
int a, b;
pair< int, int > p[1005];
int now;
int ans[1005], cans;

int main(){
    int i;

    scanf("%d", &N);
    for( i = 0; i < N; i++){
        scanf("%d %d", &a, &b);
        if( a < b)p[i].first = b, p[i].second = a;
        else p[i].first = a, p[i].second = b;
    }

    sort( p, p + N);

    now =  -2000000000;
    for( i = 0; i < N; i++){
        if( p[i].second <= now && now <= p[i].first)continue;
        now = p[i].first;
        ans[cans++] = now;
    }

    printf("%d\n", cans);
    for( i = 0; i < cans; i++)
        printf("%d ", ans[i]);
    printf("\n");
}
