#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
    int n;
    pair<int,int> a[1000];
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d%d",&a[i].first,&a[i].second);
        if(a[i].first<a[i].second) 
            swap(a[i].first,a[i].second);
    }
    sort(a,a+n);
    int ans[1001],s=0;
    for ( int i=0 ; i<1001 ; i++ )
        ans[i]=-999999999;
    for(int i=0;i<n;i++)
        if(a[i].second>ans[s]) 
            ans[++s]=a[i].first;
    printf("%d\n",s);
    for(int i=1;i<s;i++) 
        printf("%d ",ans[i]);
    printf("%d\n",ans[s]);
}
