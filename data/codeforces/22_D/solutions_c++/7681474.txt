#include <algorithm>
#include <vector>
#include <stdio.h>
#include <string.h>
using namespace std;

struct Line {
 int l, r;
} a[11111];

inline bool cmp(const Line &a, const Line &b) {
 return a.r < b.r;
}

int main() {
 int n;
 int i, j, k, l, p, q, x, y;
 scanf("%d", &n);
 for (i = 0; i < n; ++i) {
  scanf("%d%d", &x, &y);
  if (x > y) { p = x; x = y; y = p; }
  a[i].l = x; a[i].r = y;
 }
 sort(a, a + n, cmp);

 vector<int> f;
 x = -111111;
 for (i = 0; i < n; ++i) {
  if (a[i].l <= x) continue;
  x = a[i].r;
  f.push_back(x);
 }

 n = f.size();
 printf("%d\n", n);
 for (i = 0; i < n; ++i) printf("%d ", f[i]);
 return 0;
}
