#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
const int maxn=1010;
int pin=1,pins[maxn];
struct interval{
	int i,j;
}arr[maxn];
bool cmp(interval x,interval y){
	if(x.j<y.j)
		return true;
	return false;
}
int main(){
	int n,s=0;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>arr[i].i>>arr[i].j;
		if(arr[i].i>arr[i].j)
			swap(arr[i].i,arr[i].j);
	}
	sort(arr,arr+n,cmp);
	int l=arr[0].j;
	pins[s]=arr[0].j;
	s++;
	for(int i=1;i<n;i++)
		if(arr[i].i>l){
			l=arr[i].j;
			pin++;
			pins[s++]=arr[i].j;
		}
	cout<<pin<<endl;
	for(int i=0;i<s;i++)
		cout<<pins[i]<<' ';
	cout<<endl;
}
