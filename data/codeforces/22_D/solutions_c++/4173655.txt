#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;

typedef pair<int, int> pii;
const int N = 1000 + 5;

int n;
vector<int> ans;
pii s[N];

bool cmp(pii x, pii y) { return x.second < y.second; }
int main() {
	ios_base::sync_with_stdio(false);
	cin >> n;
	for (int i = 0; i < n; i++) {
		int u, v;
		cin >> u >> v;
		s[i].first = min(u, v);
		s[i].second = max(u, v);
	}
	sort(s, s + n, cmp);
	ans.push_back(s[0].second);
	for (int i = 1; i < n; i++)
		if (ans.back() < s[i].first)
			ans.push_back(s[i].second);
	cout << ans.size() << '\n';
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i] << ' ';
	cout << '\n';
	return 0;
}
