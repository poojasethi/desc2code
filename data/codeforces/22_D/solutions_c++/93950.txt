#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main(){
    int n; cin >> n;
    vector<pair<int, int> > a(n);
    for(int i = 0; i < n; i++) scanf("%d %d", &a[i].first, &a[i].second);
    for(int i = 0; i < n; i++) if(a[i].first < a[i].second) swap(a[i].first, a[i].second);
    
    sort(a.begin(), a.end());
    int last = -1100000;
    vector<int> res ;
    for(int i = 0; i < n; i++){
        if(a[i].second > last){
            last = a[i].first;
            res.push_back(last);
        }
    }
    cout << res.size() << endl;
    for(int i = 0; i < res.size(); i++)
        printf("%d%c", res[i], i + 1 == res.size() ? '\n': ' ');
}
