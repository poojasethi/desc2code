#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

class otr{
public:
  int x, y;
  bool operator < (const otr &t) const{
    if(y < t.y)
      return true;
    if(y > t.y)
      return false;
    return x < t.x;
  }
};

int main(){
  vector <otr> v;
  vector <int> vi;
  otr o;
  int n, a, b, i;
  cin >> n;
  for(i = 0; i < n; i++){
    cin >> a >> b;
    o.x = min(a, b);
    o.y = max(a, b);
    v.push_back(o);
  }
  sort(v.begin(), v.end());
  a = v[0].y;
  vi.push_back(a);
  for(i = 1; i < n; i++){
    if(v[i].x > a){
      a = v[i].y;
      vi.push_back(a);
    }
  }
  n = vi.size();
  cout << n << endl;
  for(i = 0; i < n; i++){
    cout << vi[i] << ' ';
  }
}
