#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

const int maxn=1000+10;

int n,mx=-(1<<30);

pair <int,int> a[maxn];
vector <int> ans;

int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i].first>>a[i].second;
        if(a[i].first<a[i].second)
            swap(a[i].first,a[i].second);
    }
    sort(a+1,a+n+1);
    for(int i=1;i<=n;i++)
        if(a[i].second>mx){
            mx=a[i].first;
            ans.push_back(a[i].first);
        }
    cout<<ans.size()<<endl;
    for(int i=0;i<ans.size();i++)
        cout<<ans[i]<<" ";
    cout<<endl;


    return 0;
}
