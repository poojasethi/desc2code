#include<iostream>
#include<map>

using namespace std;

pair <int , int > seg[1000];

#define x first
#define y second

void sort(int n)
{
	for(int i=0 ; i<n ; i++)
	{
		for(int j=1 ; j<n ; j++)
		{
			if(seg[j].y < seg[j-1].y)
			{
				swap(seg[j] , seg[j-1]);
			}
		}
	}
}
int main()
{
	int n;
	cin>> n;
	for(int i=0 ; i<n ; i++)
	{
		cin>> seg[i].x >> seg[i].y;
		if(seg[i].x > seg[i].y)
			swap(seg[i].x, seg[i].y);
	}
	sort(n);
	int arrey[n];
	int q=0;
	int last=-100000;
	for(int i=0 ; i<n ; i++)
		if(last < seg[i].x)
		{
			last = seg[i].y;
			arrey[q++] = last;
		}
	cout<< q << endl;
	for(int i=0 ; i<q ; i++)
	{
		cout<< arrey[i] << " ";
	}
	cout<< endl;
}
