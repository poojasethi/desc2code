#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
int n;
pair<int,int> a[1000];
int main() 
{ 
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a[i].first>>a[i].second;
        if (a[i].first<a[i].second)
          swap(a[i].first,a[i].second);
    }
    sort(a,a+n);
    int ans[1001]={-999999999},count=0;
    for(int i=0;i<n;i++)
        if(a[i].second>ans[count]) ans[++count]=a[i].first;
    cout<<count<<endl;
    for(int i=1;i<=count;i++) cout<<ans[i]<<" ";
    return 0;
}
