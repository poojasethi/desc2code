#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;
vector<pair<int, int> > a;

int main(void) {
	int n, i, j;
	scanf("%d", &n);
	for(i=0;i<n;i++){
		int x, y;
		scanf("%d%d", &x, &y);
		if(y<x) swap(x, y);
		a.push_back(make_pair(y, x));
	}
	sort(a.begin(), a.end());
	vector<int> p, mark(n, 0);
	for(i=0;i<n;i++){
		if(mark[i]) continue;
		p.push_back(a[i].first);
		for(j=i+1;j<n;j++)
			if(a[j].second <= a[i].first)
				mark[j] = 1;
	}
	printf("%d\n", p.size());
	for(i=0;i<(int)p.size();i++)
		printf("%d ", p[i]);
	puts("");
	return 0;
}
