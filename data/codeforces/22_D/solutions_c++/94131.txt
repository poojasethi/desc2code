#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
struct point{
	int x, x1;
};
point p[10000];
bool use[10000] = {0};
bool f(point a, point b)
{
	return a.x1 < b.x1;
}
int n;
vector < int > a;
int main()
{
	cin >> n;
	for(int i = 0; i < n; i++)
	{
		cin >> p[i].x >> p[i].x1;
		if(p[i].x1 < p[i].x)swap(p[i].x1, p[i].x);
	}
	sort(p, p + n, f);
	int kol = 0;
	for(int i = 0; i < n; i++)
		if(!use[i])
		{
			kol++;
			
			int x = p[i].x1;
			a.push_back(x);
			for(int j = 0; j < n; j++)
				if(x >= p[j].x && x <= p[j].x1)
					use[j] = 1;
		}
	cout << kol << endl;
	for(int i = 0; i < kol; i++)
		cout << a[i] << ' ';
	return 0;
}
