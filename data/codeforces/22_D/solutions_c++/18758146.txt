#include<iostream>
#include<algorithm>
using namespace std;
struct sajal{long long s,en;};
sajal a[1004];long long b[100005];
int mysort(sajal x,sajal y){
if(x.en<y.en)return 1;return 0;
}
int main(){
long n;
cin>>n;
for(long i=0;i<n;i++){
    cin>>a[i].s>>a[i].en;
    if(a[i].s>a[i].en){long long h=a[i].en;a[i].en=a[i].s;a[i].s=h;}

}
sort(a,a+n,mysort);
long k=a[0].en;long d=1;b[d]=k;
for(long i=1;i<n;i++){
       // cout<<a[i].s<<" "<<a[i].en<<endl;
    if(k<a[i].s){
        d++;b[d]=a[i].en;k=a[i].en;
    }
}
cout<<d<<endl;
for(long i=1;i<=d;i++)cout<<b[i]<<" ";
cout<<endl;

















}
