#include <stdio.h>
#include <utility>
#include <algorithm>
using namespace std;
#define rep(i, N) for (int i = 0; i < N; i++)

pair<int, int> seg[1010];
int ans[1010];
int nans;
int main() {
	int N;
	scanf("%d", &N);
	rep(i, N) scanf("%d %d", &seg[i].first, &seg[i].second);
	rep(i, N) seg[i] = make_pair(max(seg[i].first, seg[i].second), min(seg[i].first, seg[i].second));
	nans = 0;
	int ult = -1000000000;
	sort(seg, seg+N);
	rep(i, N) {
		if (seg[i].second > ult) {
			ult = seg[i].first;
			ans[nans++] = seg[i].first;
		}
	}
	printf("%d\n", nans);
	rep(i, nans) {
		if (i) printf(" ");
		printf("%d", ans[i]);
	}
	printf("\n");
	return 0;
}
