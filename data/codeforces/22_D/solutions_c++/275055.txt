#include<iostream>
#include<algorithm>
using namespace std;

#define X first
#define Y second
pair<int, int> p[2000];
int x[2000], ans = 0, down = -1001000;

bool cmp(pair<int, int> a, pair<int, int> b){
	if(a.Y != b.Y)
		return a.Y < b.Y;
	return a.X < b.X;
}

int main(){
	int n;
	cin >> n;
	for(int i = 0; i < n; i++){
		cin >> p[i].X >> p[i].Y;
		if(p[i].X > p[i].Y)
			swap(p[i].X, p[i].Y);
	}
	sort(p, p+n, cmp);
	for(int i = 0; i < n; i++)
		if(p[i].X > down)
			down = x[ans++] = p[i].Y;
	cout << ans << endl;
	for(int i = 0; i < ans; i++)
		cout << x[i] << " ";
	cout << endl;
	return 0;
}
