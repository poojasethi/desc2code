#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
pair<int,int> a[1000+25];
int main(){
    int n;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>a[i].first>>a[i].second;
        if(a[i].first<a[i].second)swap(a[i].first,a[i].second);
    }
    sort(a,a+n);
    vector<int>ans;
    ans.push_back(a[0].first);
    int temp=a[0].first;
    for(int i=1;i<n;i++){
        if(a[i].second<=temp){
            //temp=a[i].first;
        }
        else{
            ans.push_back(a[i].first);
            temp=a[i].first;
        }
    }
    cout<<ans.size()<<endl;
    for(int i=0;i<ans.size();i++){
        cout<<ans[i]<<" ";
    }
}
