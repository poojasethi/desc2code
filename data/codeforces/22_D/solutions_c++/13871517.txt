#include<bits/stdc++.h>

using namespace std;

typedef long long LL;
typedef long long unsigned LLU;

int main()
{
    int n;
    cin>>n;
    vector< pair< int ,int > > m(n);
    vector<int>ans;
    for(int i=0;i<n;i++)
    {
        cin>>m[i].first>>m[i].second;
        if(m[i].first<m[i].second)
            swap(m[i].first,m[i].second);

    }
    sort(m.begin(),m.end());
    int pos=0;
    while(pos<n){
        int R=m[pos].first;
        ans.push_back(R);
        while( m[pos].first>=R && m[pos].second<=R )pos++;
    }
    cout<<ans.size()<<endl;
    for(int i=0;i<ans.size();i++)cout<<ans[i]<<" ";cout<<endl;

    return 0;
}
