#include<iostream>
#include<algorithm>

#define F first
#define S second

using namespace std;

pair<int, int> a[1005];
int ans[1005], in;
bool mark[1005];

int main()
{
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
	{
		cin >> a[i].F >> a[i].S;
		if(a[i].F < a[i].S)
			swap(a[i].F, a[i].S);
	}
	sort(a, a + n);
	for(int i = 0; i < n; i++)
		if(!mark[i])
		{
			ans[in++] = a[i].F;
			for(int j = 0; j < n; j++)
				if(a[j].S <= a[i].F && a[i].F <= a[j].F)
					mark[j] = true;
		}
	cout << in << endl;
	for(int i = 0; i < in; i++)
		cout << ans[i] << ' ';
	cout << endl;
	return 0;
}
