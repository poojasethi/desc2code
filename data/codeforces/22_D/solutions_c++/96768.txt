#include <stdio.h>
#include <vector>
#include <iostream>
#include <algorithm>

using namespace std;

vector< pair<int, int> > p;
vector<int> ans;

int main(){
	int n, a, b;
	cin >> n;
	for(int i = 0; i < n; i++){
		cin >> a >> b;
		if(b < a) swap(a, b);
		p.push_back(make_pair(b, a));
	}
	sort(p.begin(), p.end());
	int min = -10002;
	for(int i = 0; i < n; i++)
		if(p[i].second > min){
			min = p[i].first;
			ans.push_back(p[i].first);
		}
	printf("%d\n", ans.size());
	for(int i = 0; i < ans.size(); i++){
		printf("%d ", ans[i]);
	}
	printf("\n");
	return 0;
}
