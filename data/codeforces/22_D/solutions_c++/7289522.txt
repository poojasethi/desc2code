#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
int n,sv[100001],c1,tmp;
vector <pair <int,int> > v;
vector <int> ans;
int main(){
  scanf("%d",&n);
  for(int i=1,a,b;i<=n;i++){
    scanf("%d%d",&a,&b);
    if(a > b) swap(a,b);
    v.push_back(make_pair(a,-i));
    v.push_back(make_pair(b,i));
  }
  sort(v.begin(),v.end());
  for(int i=0;i<v.size();i++){
    if(v[i].second < 0) c1++, sv[-v[i].second] = c1;
    else{
      if(sv[v[i].second] > tmp) tmp = c1,ans.push_back(v[i].first);
    }
  }
  printf("%d\n",ans.size());
  for(int i=0;i<ans.size();i++)
    printf("%d ",ans[i]);
  return 0;
}
