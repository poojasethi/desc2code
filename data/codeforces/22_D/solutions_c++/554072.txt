#include <cstdio>
#include <utility>
#include <algorithm>
using namespace std;
pair < int, int > a[1010];
int n, x, y, k = -100000, p = 0, b[1010];
int main() {
    scanf("%d", &n);
    for(int i = 0; i < n; i++) {
        scanf("%d%d", &x, &y);
        if(x > y)swap(x, y);
        a[i] = make_pair(y, x);
    }
    sort(a, a + n);
    for(int i = 0; i < n; i++) {
        if(a[i].second > k)
            k = a[i].first,
            b[p++] = a[i].first;
    }
    printf("%d\n", p);
    for(int i = 0; i < p; i++)
        printf("%d ", b[i]);
    puts("");
    return 0;
}
