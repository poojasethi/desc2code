# include <bits/stdc++.h>

using namespace std;

# define L first
# define R second

const int maxn = 1000 + 10;

pair <int,int> a[maxn];

bool cmp(pair<int,int> A, pair<int,int> B){
	return (A.R < B.R);
}

int main(){
	ios_base::sync_with_stdio(false);
	int n;
	cin >> n;
	for (int i = 0; i < n; i++){
		cin >> a[i].L >> a[i].R;
		if(a[i].L > a[i].R)
			swap(a[i].L, a[i].R);
	}
	sort(a, a + n, cmp);

	vector <int> ans;
	for (int i = 0; i < n; i++)
		if(ans.size() == 0 || ans.back() < a[i].L)
			ans.push_back(a[i].R);

	cout << ans.size() << endl;
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i] << " ";
	cout << endl;

	return 0;
}
