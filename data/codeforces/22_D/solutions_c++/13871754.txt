#include<bits/stdc++.h>
using namespace std;
long long n, i, j, k, l, num1, num2, fix[1000];
pair<long long, long long>arr[1000];
vector<long long>ans;
main(){
    cin>>n;
    for(i=0;i<n;i++){
        cin>>num1>>num2;
        arr[i].second=min(num1,num2);
        arr[i].first=max(num1,num2);
    }
    sort(arr,arr+n);
    ans.push_back(arr[0].first);
    for(i=0;i<n;i++){
        j=i;
        if(arr[i].first>=ans[ans.size()-1] && arr[i].second<=ans[ans.size()-1])continue;
        ans.push_back(arr[i].first);
    }
    cout<<ans.size()<<endl;
    for(i=0;i<ans.size();i++)cout<<ans[i]<<" ";
    return 0;
}
