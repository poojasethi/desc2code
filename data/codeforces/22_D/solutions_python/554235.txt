n = int(raw_input())
i = 0
s = []
while i < n:
	i += 1
	p = raw_input().split()
	a = min(int(p[0]), int(p[1]))
	b = max(int(p[0]), int(p[1]))
	s.append((a, b))

s.sort()
pos = -1000000
ans = []
for i in range(len(s)):
	if pos >= s[i][0] and pos <= s[i][1]:
		continue
	skip = False
	for x in s[i+1:]:
		if x[0] <= s[i][1] and x[1] <= s[i][1]:
			skip = True
	if skip:
		continue
	pos = s[i][1]
	ans.append(pos)

print len(ans)
s = ''
for x in ans:
	s += str(x)+' '
print s
