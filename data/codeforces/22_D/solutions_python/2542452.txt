v=[]
for i in xrange(input()):
  a,b=map(int,raw_input().split())
  if a>b:a,b=b,a
  v.extend([(a,0,i),(b,1,i)])
t,k,u=[],[],set()
for x in sorted(v):
  if x[1]:
    if x[2] not in u:
      t.append(x[0])
      for j in k:u.add(j)
      k=[]
  else:
    k.append(x[2])
print len(t)
print ' '.join(map(str,t))
