#! /usr/bin/python

from sys import stdin


if __name__=='__main__':
	n=int(stdin.readline())
	segments=[map(int,stdin.readline().split()) for i in xrange(n)]
	segments=[(min(a,b),max(a,b)) for (a,b) in segments]
	segments=[(b,a) for a,b, in segments]
	segments.sort()#key=lambda x:x[-1])
	segments=[(a,b) for (b,a) in segments]
	#nails=1
	output=[segments[0][-1]]
	last=segments[0][-1]
	for cur in segments[1:]:
		if last<cur[0]:
			#nails+=1
			last=cur[-1]
			output.append(cur[-1])
	print len(output)
	print " ".join(map(str,output))
