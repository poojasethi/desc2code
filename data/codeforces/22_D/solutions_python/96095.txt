def readint(): return int(raw_input())
def readlinearray(foo): return map(foo, raw_input().split())

def get_nails(segments_list):
    segments_list.sort()
    xs = set()
    for b, e in segments_list:
        xs.add(b)
        xs.add(e)
    
    r = []
    while len(segments_list) > 0:
        max_x = None
        max_segments_to_remove = []
        for x in xs:
            segments_to_remove = []
            for b, e in segments_list:
                if b <= x <= e:
                    segments_to_remove.append((b, e))
                else:
                    break
            if len(segments_to_remove) > len(max_segments_to_remove):
                max_x = x
                max_segments_to_remove = segments_to_remove
        xs.remove(max_x)
        r.append(max_x)
        for segment in max_segments_to_remove:
            segments_list.remove(segment)
    return r


if __name__ == '__main__':
    n = readint()
    segments = []
    for _ in xrange(n):
        b, e = readlinearray(int)
        if b > e:
            b, e = e, b
        segments.append((b, e))
    nails = get_nails(segments)
    print len(nails)
    print ' '.join(map(str, nails))
