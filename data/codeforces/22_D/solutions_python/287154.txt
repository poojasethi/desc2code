n = input ()
ox = list ()
for i in xrange (n):
    l, r = sorted (map (int, raw_input ().split ()))
    ox.append ([l, 0, i])
    ox.append ([r, 1, i])
ox.sort ()
stack, removed, nails = [], [], []
for x, closing, index in ox:
    if index in removed:
        continue
    if not closing:
        stack.append (index)
    else:
        nails.append (x)
        removed += stack
        stack = []
print len (nails)
print ' '.join (map (str, nails))
