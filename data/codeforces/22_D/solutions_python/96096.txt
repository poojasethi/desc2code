def readint(): return int(raw_input())
def readlinearray(foo): return map(foo, raw_input().split())

def get_nails(segs):
    segs.sort()
    xs = set()
    for b, e in segs:
        xs.add(b)
        xs.add(e)
    
    r = []
    while len(segs) > 0:
        max_x = None
        max_segs_to_remove = []
        for x in xs:
            segs_to_remove = []
            for b, e in segs:
                if b <= x <= e:
                    segs_to_remove.append((b, e))
                else:
                    break
            if len(segs_to_remove) > len(max_segs_to_remove):
                max_x = x
                max_segs_to_remove = segs_to_remove
        xs.remove(max_x)
        r.append(max_x)
        for segment in max_segs_to_remove:
            segs.remove(segment)
    return r

if __name__ == '__main__':
    n = readint()
    segs = []
    for _ in xrange(n):
        b, e = readlinearray(int)
        if b > e:
            b, e = e, b
        segs.append((b, e))
    nails = get_nails(segs)
    print len(nails)
    print ' '.join(map(str, nails))
