#include <cstdio>
#include <algorithm>
#include <memory.h>
using namespace std;

int n, m, d, i, h, l, r, j, n1[160000], a[1000], b[1000], t[1000];
long long f[160000], f1[160000], ans;

int main() {
	scanf("%d%d%d", &n, &m, &d);
	t[0] = 1;
	for (i = 1; i <= m; i++) {
		h = 0;
		l = 1; r = 0;
		scanf("%d%d%d", &a[i], &b[i], &t[i]);
		for (j = 1; j <= n; j++) {
			while (h < j + 1LL * (t[i] - t[i - 1]) * d && h < n) {
				h++;
				while (l <= r && f[h] >= f[n1[r]])	r--;
				n1[++r] = h;
			}
			while (n1[l] < j - 1LL * (t[i] - t[i - 1]) * d)	l++;
			f1[j] = f[n1[l]] + b[i] - abs(a[i] - j);
		}
		memcpy(f, f1, sizeof(f1));
	}
	for (ans = -1000000000000000000LL, i = 1; i <= n; i++)	ans = max(ans, f[i]);
	printf("%I64d\n", ans);
}
