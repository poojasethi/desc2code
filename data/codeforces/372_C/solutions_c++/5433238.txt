#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long LL;
LL n, m, d, t, t0, a;
LL f[2][150010], q[150010], qq[150010], l, r;
LL ans;
void add(LL x, LL y)
{
    while (r >= l && q[r] >= x) --r;
    q[++r] = x;
    qq[r] = y;
}
int main()
{
    cin >> n >> m >> d;
    t = 0;
    for (LL i = 1; i <= m; ++i)
    {
        int ii = i & 1, iii = 1 - ii;
        LL j;
        t0 = t;
        cin >> a >> j >> t;
        ans += j;
        l = 0;
        r = -1;
        LL dd = (t-t0) * d;
        for (j = 1; j <= min(n, 1 + dd); ++j)
            add(f[iii][j], j);
        for (j = 1; j <= n; ++j)
        {
            while (l < r && qq[l] < j - dd) ++l;
            f[ii][j] = q[l] + abs(a - j);
            if (j+dd+1<=n)
                add(f[iii][j+dd+1], j+dd+1);
        }
    }
    LL mm = f[m&1][1];
    for (LL i = 2; i <= n; ++i)
        mm = min(mm, f[m&1][i]);
    cout << ans - mm << endl;
    return 0;
}
