#!/usr/bin/env python
# -*- coding: utf-8 -*-

def getans(n):
    nc = (len(n)+1) / 2
    if nc*2 > len(n):
        return '4'*nc + '7'*nc
    rep = -1
    nf = ns = nc
    for i in xrange(len(n)):
        if (n[i] <= '4' or rep != -1) and nf > 0:
            if n[i] < '4' and rep == -1:
                rep = i
            n[i] = '4'
            nf -= 1
        elif (n[i] <= '7' or rep != -1) and ns > 0:
            if n[i] < '7' and rep == -1:
                rep = i
            n[i] = '7'
            ns -= 1
        else:
            if ns > 0:
                ns -= 1
                js = i
            else:
                nf -= 1
                n[i] = '4'
                js = i - 1
                while js >= 0:
                    if n[js] == '7':
                        break
                    js -= 1
            if js != -1:
                jf = js - 1
                while jf >= 0:
                    if n[jf] == '4':
                        break
                    jf -= 1
            if js == -1 or jf == -1:
                nc += 1
                return '4'*nc + '7'*nc
            n[jf] = '7'
            n[js] = '4'
            rep = jf
    if rep != -1:
        n = n[:rep+1] + sorted(n[rep+1:])
    return ''.join(n)

print getans(list(raw_input()))
