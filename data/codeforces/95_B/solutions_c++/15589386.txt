#include<cstdio>
#include<cstring>
char s[100010];
int main(){
	scanf("%s",s);
	int l=strlen(s);
	int i=0,ll,seve=0,four=0;
	bool flg=false,bigger=false;
	while(s[i]>='7'){
		if(s[i]>'7'){flg=true;break;}
		i++;seve++;
	}
	while(!flg&&s[i]>='4'){
		if(s[i]>'7'){
			for(ll=i-1;ll>=0;ll--)if(s[ll]=='4')break;
			s[ll]='7';seve++;
			while(++ll<i)if(s[ll]=='7'){
				s[ll]='4';
				seve--;four++;
			}
			s[i]='4';i++;
			break;
		}
		if(s[i]>'4'){
			if(s[i]<'7')bigger=true;
			s[i]='7';seve++;
			if(bigger){i++;break;}
		}
		else four++;
		i++;
	}
	int j=i-1;
	while(seve>l/2){
		for(;j>=0;j--)if(s[j]=='4')break;
		if(j<0)break;
		s[j]='7';seve++;ll=j;
		while(++ll<i)if(s[ll]=='7'){
			s[ll]='4';
			seve--;four++;
		}
	}
	if(flg||l&1||seve>l/2){
		l++;
		if(l&1)l++;
		for(i=0;i<(l>>1);i++)s[i]='4';
		for(;i<l;i++)s[i]='7';s[i]=0;
		printf("%s",s);
		return 0;
	}
	ll=l>>1;
	while(i<l){
		if(four<ll)four++,s[i]='4';
		else seve++,s[i]='7';
		i++;
	}
	while(seve<ll){
		--i;
		if(s[i]=='4')s[i]='7',seve++;
	}
	printf("%s",s);
	return 0;
}
