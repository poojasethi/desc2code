#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <functional>
using namespace std;

string n, ans;
int main() {
	ios::sync_with_stdio(false);
	cin>>n;
	if(n.size() & 1) {
		for(int i = 0; i < (n.size() >> 1) + 1; i++) putchar('4');
		for(int i = 0; i < (n.size() >> 1) + 1; i++) putchar('7');
		putchar('\n');
		return 0;
	}
	if(string(n.size() >> 1, '7') + string(n.size() >> 1, '4') < n) {
		for(int i = 0; i < (n.size() >> 1) + 1; i++) putchar('4');
		for(int i = 0; i < (n.size() >> 1) + 1; i++) putchar('7');
		putchar('\n');
		return 0;
	}
	int cnt4 = n.size() >> 1, cnt7 = n.size() >> 1;
	bool fg = 0;
	for(int i = 0; i < n.size(); i++) {
		bool can = 0;
		if(cnt4 > 0) {
			if(fg || n[i] < '4') can = 1;
			else if(n[i] == '4') {
				for(int j = i + 1; j <= n.size(); j++) {
					if(j == n.size()) { can = 1; break; }
					char ch = (j <= i + cnt7 ? '7':'4');
					if(ch > n[j]) can = 1;
					if(ch != n[j]) break;
				}
			}
		}
		if(can) { ans += '4'; cnt4--; }
		else { ans += '7'; cnt7--; }
		if(ans[i] > n[i]) fg = 1;
	}
	cout<<ans<<endl;
	return 0;
}
