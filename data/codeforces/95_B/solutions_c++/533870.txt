using namespace std;

#define FOR(i,a,b) for (int i=(a),_b=(b); i<=_b; i++)
#define REP(i,n) for (int i=0,_n=(n); i<_n; i++)

char s[1110000], g[1110000];
int n;

bool gen(int i, int n4, int n7, bool larger){
	if (i==n){
		g[i] = '\0';
		puts(g);
		return true;
	}

	if (larger){
		if (n4){
			g[i] = '4';
			return gen(i+1, n4-1, n7, larger);
		} else {
			g[i] = '7';
			return gen(i+1, n4, n7-1, larger);
		}
	} else {
		g[i] = '4';
		if (n4 && s[i]<='4' && gen(i+1, n4-1, n7, s[i]<'4')) return true;
		g[i] = '7';
		if (n7 && s[i]<='7' && gen(i+1, n4, n7-1, s[i]<'7')) return true;
		return false;
	}
}

int main(){
	while (scanf("%s",s)!=EOF){
		n = strlen(s);
		if (n%2==1 || !gen(0,n/2,n/2,0)){
			n = n/2 + 1;
			REP(i,n) printf("4");
			REP(i,n) printf("7");
			puts("");
		}
	}
}
