#include<cstdio>
#include<cstring>
char s[100005],g[100005];
int up,i;
bool gen(int n,int n4,int n7)
{
    if (n4==0&&n7==0) return 1;
    if(up)
        if(n4)
        {
            g[n]='4';
            return gen(n+1,n4-1,n7);
        }
        else
        {
            g[n]='7';
            return gen(n+1,n4,n7-1);
        }
    else
    {
        if(s[n]<='4'&&n4)
        {
            g[n]='4';
            if(s[n]<'4')up=1;
            if (gen(n+1,n4-1,n7)) return 1;
            up=0;
        }
        if(s[n]<='7'&&n7)
        {
            g[n]='7';
            if(s[n]<'7')up=1;
            if(gen(n+1,n4,n7-1)) return 1;
            up=0;
        }
    }
    return 0;
}
int main()
{
    gets(s);
    int l=strlen(s);
    if (l&1||!gen(0,l/2,l/2))
    {
        for(i=0;i<l/2+1;i++) printf("4");
        for(i=0;i<l/2+1;i++) printf("7");
        puts("");
    }
    else puts(g);
}
