#include <bits/stdc++.h>
#define MAXN 100005

using namespace std;

int len, counter = 1;
char res[MAXN], number[MAXN];
bool process(int i, int four, int seven, bool is_large) {
    if(i == len) {
        res[i] = '\0';
        return true;
    }
    if(is_large) {
        if(four) {
            res[i] = '4';
            return process(i+1, four-1, seven, is_large);
        }
        else {
            res[i] = '7';
            return process(i+1, four, seven-1, is_large);
        }
    }
    else {

        if(four > 0 && number[i] <= '4' && process(i+1, four-1, seven, number[i] < '4')) {
            res[i] = '4';
            return true;
        }

        if(seven > 0 && number[i] <= '7' && process(i+1, four, seven-1, number[i] < '7')) {
            res[i] = '7';
            return true;
        }
        return false;
    }
}

int main() {
    scanf("%s", number);

    len = strlen(number);

    if(len % 2 == 0 && process(0, len/2, len/2, false)) printf("%s", res);
    else {
        len = (len / 2) + 1;
        for(int i=0; i<len; ++i) printf("4");
        for(int i=0; i<len; ++i) printf("7");
    }
    return 0;
}
