#include <cstdio>
#include <iostream>
#include <cstring>
using namespace std;
char s[100001];
bool judge(int x,int y){
	int l=strlen(s);
	for(int i=x;i<l;++i)
		if(i<x+y){
			if(s[i]<'7')return 1;
			if(s[i]>'7')return 0;
		}
		else{
			if(s[i]<'4')return 1;
			if(s[i]>'4')return 0;
		}
	return 1;
}
int main(){
	scanf("%s",s);
	int l=strlen(s);
	if(l%2==1){
		for(int i=0;i<=l/2;++i)printf("4");
		for(int i=0;i<=l/2;++i)printf("7");
		return 0;
	}
	if(!judge(0,l/2)){
		for(int i=0;i<=l/2;++i)printf("4");
		for(int i=0;i<=l/2;++i)printf("7");
		return 0;
	}
	else{
		int t=0,a=0,b=0;
		for(int i=0;i<l;++i)
			if(t)
				if(a<l/2)printf("4"),++a;
				else printf("7"),++b;
			else
			    if(s[i]<'4')t=1,printf("4"),++a;
			    else if(s[i]=='4')
			    	if(a<l/2)
			    		if(judge(i+1,l/2-b))printf("4"),++a;

	}
}
