#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
char a[100010], b[100010];
bool dfs(int i,int f,int s,bool z) {
  if(f==0 && s==0) return true;
  if(f<0 || s<0) return false;
  if(z) {
    if(a[i]<='4' && dfs(i+1,f-1,s,a[i]=='4'))
      return b[i] = '4', true;
    if(a[i]<='7' && dfs(i+1,f,s-1,a[i]=='7'))
      return b[i] = '7', true;
    if(a[i]>7) return false;
  } else {
    if(dfs(i+1,f-1,s,z))
      return b[i] = '4', true;
    if(dfs(i+1,f,s-1,z))
      return b[i] = '7', true;
    return false;
  }
}
int main() {
  scanf("%s",a);
  int l = strlen(a);
  if(l&1) for(int i(0);i!=l+1;++i)
    a[i] = i?'1':'0';
  l+=l&1;
  if(!dfs(0,l/2,l/2,true)) {
    for(int i(0);i!=l+2;++i)
      a[i] = i?'1':'0';
    l+=2;
    dfs(0,l/2,l/2,true);
  }
  printf("%s",b);
  return 0;
}
