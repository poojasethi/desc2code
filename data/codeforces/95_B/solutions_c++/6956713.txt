#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
string s,ans;
int n;
bool solve(int p,int n4,int n7,bool ok){
    //cout<<p<<" "<<n4<<" "<<n7<<" "<<ok<<endl;
    if(p==n)return 1;
    if(ok){
        for(int i=0;i<n7;++i)ans+='7';
        for(int i=0;i<n4;++i)ans+='4';
        return 1;
    }
    if(n4>=1 && s[p]<='4' && solve(p+1,n4-1,n7,s[p]<'4')){
        ans+='4';
        return 1;
    }else if(n7>=1 && s[p]<='7' && solve(p+1,n4,n7-1,s[p]<'7')){
        ans+='7';
        return 1;
    }
    return false;
}
int main(){
    while(cin >> s){
    n=s.size();
    ans.clear();
    //cout<<"n"<<n<<endl;
    //cout<<s<<endl;
    if(n%2==0 && solve(0,n/2,n/2,false)){
        reverse(ans.begin(),ans.end());
        cout<<ans<<endl;
    }else{
        if(n%2==0)n+=2;else ++n;
        for(int i=0;i<n/2;++i)cout<<4;
        for(int i=0;i<n/2;++i)cout<<7;
        cout<<endl;
    }
    }
    return 0;
}
