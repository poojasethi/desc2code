#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
#include<cmath>
#include<vector>
#include<map>
#include<queue>
using namespace std;
typedef long long ll;
const int N=1e5+10;
string s,t;
int main()
{
    int i,j,n;
    cin>>s;
    n=s.length();
    if(string(n/2,'7')+string(n/2,'4')<s||n%2)
        cout<<string(n/2+1,'4')<<string(n/2+1,'7')<<endl;
    else
    {
        int n4=n/2,n7=n/2;
        bool bo=0;
        for(i=0; i<n; ++i)
        {
            bool bo1=0;
            if(n4>0)
            {
                if(bo||s[i]<'4')bo1=1;
                else if(s[i]=='4')
                    for(j=i+1; j<=n; ++j)
                    {
                        if(j==n){bo1=1;break;}
                        char c=(j<=i+n7?'7':'4');
                        if(c>s[j])bo1=1;
                        if(c!=s[j])break;
                    }
            }
            if(bo1)t+="4",--n4;
            else t+="7",--n7;
            if(t[i]>s[i])bo=1;
        }
        cout<<t<<endl;
    }
}
