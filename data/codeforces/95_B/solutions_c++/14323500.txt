#include <bits/stdc++.h>
using namespace std;
string s;
char ret[1000000];
int n;
bool dfs(int a,int b,bool great){
	int ind = n-a-b;
	if(a==0 &&b==0)return true;
	if(great){
		if(a)
			ret[ind]='4',dfs(a-1,b,true);
		else
			ret[ind]='7',dfs(a,b-1,true);
		return true;
	}
	int temp = s[ind]-'0';
	if(temp>7)return false;
	if( temp<=7 && 4<temp){
		if(b!=0){
			bool more = temp==7 ? false :true;
			if(dfs(a,b-1,more)){
				ret[ind]='7';
				return true;
			}
		}
		else 
			return false;
	}
	if(temp<=4){
		if(a){
			bool more = temp==4 ? false : true;
			if( dfs(a-1,b,more))
				ret[ind]='4';
			else
				if(b==0)return false;
				else
					ret[ind]='7',dfs(a,b-1,true);
		}
		else
			ret[ind]='7',dfs(a,b-1,true);
		return true;
	}
}
void solve(){
	if(n%2==1){
		n++;
		dfs(n/2,n/2,true);
	}
	else{
		string ret;for(int i=0;i<n/2;++i)ret+='7';for(int i=0;i<n/2;++i)ret+='4';
		if(ret<s){
			n+=2;
			dfs( (n/2),(n/2),true);
		}
		else
			dfs(n/2,n/2,false);
	}
	ret[n]=0;
	printf("%s\n",ret);
}
int main(){
	cin>>s;n = s.size();
	solve();
}
