#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<algorithm>
using namespace std;
#define maxn 100005

int n;
char st[maxn],ans[maxn];

bool dfs(int l,int n4,int n7,int flag)
{
     if (l==n) return 1;
     if (flag)
     {
        for (int i=0;i<n4;i++) ans[l+i]='4';
        l+=n4;
        for (int i=0;i<n7;i++) ans[l+i]='7';
        return 1;
     }
     if (n4&&st[l]<='4')
     { 
        if (dfs(l+1,n4-1,n7,st[l]<'4')){ ans[l]='4'; return 1; }
     }
     if (n7&&st[l]<='7')
     {
        if (dfs(l+1,n4,n7-1,st[l]<'7')){ ans[l]='7'; return 1; } 
     }
     return 0;
}

int main()
{
    scanf("%s",st);
    n=strlen(st);             
    if ((n&1)||!dfs(0,n/2,n/2,0))
    {
       n+=2;
       for (int i=1;i<=n/2;i++) printf("4");
       for (int i=1;i<=n/2;i++) printf("7");
    }
    else
    {
        ans[n]='\0';
        printf("%s",ans);
    }
    printf("\n");
    return 0;
}
