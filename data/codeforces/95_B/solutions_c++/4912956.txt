#include <cstdlib>
#include <cstdio>
#include <iostream>
using namespace std;

string num,ans;

int main(){
//	freopen("B.in","r",stdin);
	cin >> num;
	int n = num.size();
	if(n&1){
		for (int i = 0; i < (n+1)/2; i++)
			printf("4");
		for (int i = 0; i < (n+1)/2; i++)
			printf("7");
		puts("");
		return 0;
	}
	
	if(string(n/2,'7') + string(n/2,'4') < num) {
		cout << string((n+2)/2,'4') << string((n+2)/2,'7') << endl;
		return 0;
	}
	int rest4 = n / 2, rest7 = n / 2;
	for(int i = 0; i < n; i++) {
		bool four = false;
		if(rest4) {
			if(num[i] < '4') 
				four = true;
			else if(num[i]=='4'){
				for(int j = i + 1; j < n; j++){
					char ch = (j<=i+rest7)?'7':'4';
					if (ch > num[j]) {four = true;break;}
					if (ch < num[j]) break;
					if (j == n - 1) four = true;
				}
			}
		}
		if(four) 
			ans += '4', rest4--;
		else ans += '7', rest7--;
		if(ans[i] > num[i] || !rest7) {
			ans += string(rest4,'4') + string(rest7,'7');
			break;
		}
	}
	cout << ans << endl;
	return 0;
}
