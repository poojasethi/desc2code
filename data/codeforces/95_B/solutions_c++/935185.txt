#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

string res;
string s;
int n;

// 上位桁から順番に決めていく
bool solve(int pos,int n4,int n7,bool bigger){
    if(pos==n)return true;
    else if(bigger){
        for(int i=0;i<n7;i++)res+='7';
        for(int i=0;i<n4;i++)res+='4';
        return true;
    }
    if(s[pos]<='4'&&n4>=1&&solve(pos+1,n4-1,n7,s[pos]<'4')){
        res+='4';
        return true;
    }
    else if(s[pos]<='7'&&n7>=1&&solve(pos+1,n4,n7-1,s[pos]<'7')){
        res+='7';
        return true;
    }
    return false;
}

int main(){
    cin>>s;
    n=s.size();
    if(s.size()%2==0&&solve(0,n/2,n/2,false)){
        reverse(res.begin(),res.end());
        cout<<res<<endl;
    }
    else{
        if(s.size()%2==0)n+=2;
        else n++;
        for(int i=0;i<n/2;i++)cout<<4;
        for(int i=0;i<n/2;i++)cout<<7;
        cout<<endl;
    }
    return 0;
}
