#include <cstdio>
#include <cctype>
#include <string>
#include <iostream>

using namespace std;


string n;
string res;

bool getNext(int pos, int n4, int n7, bool cuidado){
    if(pos == n.size()){
        
        return true;
    }else{
        
        if((!cuidado || n[pos] <= '4') && n4 > 0){
            res[pos] = '4';
            if(getNext(pos+1,n4-1,n7, cuidado && (n[pos] =='4')))return true;
        }
        if((!cuidado || n[pos] <= '7') && n7 > 0){
            res[pos] = '7';
            if(getNext(pos+1,n4,n7-1, cuidado && (n[pos] =='7')))return true;
        }
        return false;
    }
}

int main(){
    
    
    cin >> n;
    res = n;
    bool achei = false;
    if(n.size()%2 == 0){
        if(getNext(0, n.size()/2, n.size()/2, true)){
            achei = true;
        }
        n += "00";
        for(int i = 0; i < n.size(); i++)
            n[i] = '0';
    }else{
        n += "0";
        for(int i = 0; i < n.size(); i++)
            n[i] = '0';
    }
    
    if(!achei){
        res = n;
        getNext(0,n.size()/2, n.size()/2, true);
    }
    cout << res << endl;
    
    return 0;
}
