#include<cstdio>
int n,s4,s7,a[100000];
void output()
{
	int i=1;
	while(i<=s4+s7)printf("%d",a[i-1]),i++;
	while(i-s7<=n/2)printf("4"),i++;
	while(i<=n)printf("7"),i++;
	printf("\n");
}
bool clac()
{
	s4=s7=0;
	int i,x4=-1,flag=0;
	for(i=0;i<n;i++)
	{
		if(a[i]==7)s7++;
		else if(a[i]==4)
		{
			if(s7+1<=n/2)
				x4=i;
		}
		else break;
	}
	if(x4==-1)return 1;
	for(s4=s7=i=0;i<x4;i++)
	{
		if(a[i]==7)s7++;
		else s4++;
	}
	a[i]=7;
	s7++;
	return 0;
}
bool judge()
{
	int x4;
	x4=-1;
	for(int i=0;i<n;i++)
	{
		if(a[i]>7)
		{
			if(x4==-1)return 1;
			else return clac();
		}
		else if(a[i]==7)
		{
			s7++;
			if(s7>n/2)
			{
				if(x4==-1)return 1;
				else return clac();
			}
		}
		else if(a[i]>4){a[i]=7,s7++;break;}
		else if(a[i]==4)
		{
			s4++;
			if(s4>n/2)
			{
				a[i]=7;
				return 0;
			}
			x4=i;
		}
		else{a[i]=4,s4++;break;}
	}
	return s4>n/2||s7>n/2;
}
int main()
{
	while(~scanf("%1d",&a[n]))n++;
	if(n&1)	n++,output();
	else if(judge())s4=s7=0,n+=2,output();
	else output();
	return 0;
}
