#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int a[101000],i,len;
char s[101000];
void dfs(int o,int p,int w){
	if(w==len){
		for(i=0;i<len;++i)printf("%d",a[i]);
		exit(0);
	}
	if(s[w]-'0'<4&&o){
		a[w]=4;
		for(i=0;i<=w;++i)printf("%d",a[i]);
		for(i=1;i<o;++i)printf("4");
		for(i=0;i<p;++i)printf("7");
		printf("\n");exit(0);
	}
	if(s[w]-'0'==4&&o){
		a[w]=4;dfs(o-1,p,w+1);
	}
	if(s[w]-'0'<7&&p){
		a[w]=7;
		for(i=0;i<=w;++i)printf("%d",a[i]);
		for(i=0;i<o;++i)printf("4");
		for(i=1;i<p;++i)printf("7");
		printf("\n");exit(0);
	}
	if(s[w]-'0'==7&&p){
		a[w]=7;dfs(o,p-1,w+1);
	}
}
int main(){
	scanf("%s",s);
	len=strlen(s);
	if(len&1){
		len++;
		for(i=1;i+i<=len;++i)printf("4");
		for(i=1;i+i<=len;++i)printf("7");
		printf("\n");
	}else{
		dfs(len>>1,len>>1,0);
		len+=2;
		for(i=1;i+i<=len;++i)printf("4");
		for(i=1;i+i<=len;++i)printf("7");
	}
	return 0;
}
