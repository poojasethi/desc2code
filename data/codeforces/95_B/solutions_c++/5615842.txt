#include <iostream>
#include <cstdio>
#include <cmath>
#include <string>
#include <cstring>
using namespace std;
string s;
int i,n,need[100100];
char res[100100];
int main()
{
	cin>>s;n=s.size();
	for (i=n-1;i>=0;i--)
	{
		if (s[i]<'4') need[i]=0; else
		if (s[i]=='4') need[i]=min(1,need[i+1]); else
		if (s[i]<'7') need[i]=1; else
		if (s[i]=='7') need[i]=need[i+1]+1;else
		need[i]=1000000000;
	}
	if (n%2==0)
	{
		bool ok=1;
		int a=0,b=0,flag=0;
		for (i=0;i<n;i++)
			if (flag)
				if (a+1<=n/2) res[i]=4,a++;
				else res[i]=7,b++;
			else {
				if (s[i]<'4' && a+1<=n/2) res[i]=4,a++,flag=1;else
				if (s[i]=='4' && a+1<=n/2 && need[i+1]<=n/2-b) res[i]=4,a++;else
				if (s[i]<'7' && b+1<=n/2) res[i]=7,b++,flag=1;else
				if (s[i]=='7' && b+1<=n/2) res[i]=7,b++;
				else ok=0;
			}
			if (ok)
			{
				for (i=0;i<n;i++) printf("%d",res[i]);
				printf("\n");
				return 0;
			}
			n+=2;
	} else n+=1;
	for (i=1;i<=n/2;i++) printf("4");
	for (i=1;i<=n/2;i++) printf("7");


return 0;
}
