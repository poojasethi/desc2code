#define li       	long long int
#define rep(i,to)	for(li i=0;i<((li)(to));i++)
#define pb       	push_back
#define sz(v)    	((li)v.size())
#define bit(n)   	(1ll<<(li)(n))


#define MAX 100005ll
char ans[MAX];
string str;

bool recur(li pos,bool full,li rem0,li rem1){
	if(sz(str)==pos) return true;
	ans[pos]='4';
	if((!full || str[pos]<='4') && rem0 ){
		if(recur(pos+1,full & str[pos]=='4',rem0-1,rem1)) return true;
	}
	ans[pos]='7';
	if((!full || str[pos]<='7') && rem1 ){
		if(recur(pos+1,full & str[pos]=='7',rem0,rem1-1)) return true;
	}
	return false;
}


string cal(){
	string res="";
	if(sz(str)%2==0 && recur(0,true,sz(str)/2,sz(str)/2)){
		rep(i,sz(str)) res+=ans[i];
		return res;
	}
	li len=(sz(str)/2+1)*2;
	rep(i,len/2) res+="4";
	rep(i,len/2) res+="7";
	return res;
}

int main(){
	cin>>str;
	cout<<cal()<<endl;
}
