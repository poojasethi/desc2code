#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int f[N][2];
char p[N];
inline bool pan(int x,int a,int b) {
	int A=f[x][1];
	if (A<x+a) {
		if (p[A]>'7') return 0;
		if (p[A]<'7') return 1;
	}
	
	x+=a;
	A=f[x][0];
	if (A<x+b) {
		if (p[A]>'4') return 0;
		if (p[A]<'4') return 1;
	}
	return 1;
}
int main() {
	scanf("%s",p+1);
	n=strlen(p+1);
	f[n+1][0]=f[n+1][1]=n+1;
	for (i=n;i;i--) {
		For(j,0,1) f[i][j]=f[i+1][j];
		if (p[i]!='4') f[i][0]=i;
		if (p[i]!='7') f[i][1]=i;
	}
	if (n&1||!pan(1,n/2,n/2)) {
		for (n++;n&1;n++);
		For(i,1,n/2) printf("4");
		For(i,1,n/2) printf("7");
		printf("\n");
	} else {
		int A,B;
		A=B=n/2;
		For(i,1,n) {
			char T;
			if (A&&(p[i]<'4'||p[i]=='4'&&pan(i+1,B,A-1))) printf("4"),A--,T='4';
			else printf("7"),B--,T='7';
			if (p[i]<T) {
				For(j,1,A) printf("4");
				For(j,1,B) printf("7");
				break;
			}
		}
		printf("\n");
	}
	return 0;
}
