#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<string>
using namespace std;

int main() {
	static string s, t;
	cin>>s;
	int n = s.size();
	if (!(n&1)) {
		if (string(n/2, '7') + string(n/2, '4') < s) {
			cout<<string(n/2+1, '4')+string(n/2+1, '7')<<endl;
		} else {
			int n4 = n/2, n7 = n/2;
			bool f = 0;
			for (int i=0; i<n; ++i) {
				bool ff = 0;
				if (n4 > 0) {
					if (f || s[i] < '4') ff = 1;
				    else if (s[i] == '4') {
						for (int j=i+1; j<=n; ++j){
							if (j == n) {ff = 1; break;}
							char ch = (j <= i+n7?'7':'4'); 
							if (ch > s[j]) ff = 1;
							if (ch != s[j]) break;
						}
				    }
				}
				if (ff) {t += "4"; n4--;} else {t += "7"; n7--;}
				if (t[i] > s[i]) f = 1;
			}
			cout<<t<<endl;
		}
		
	}
	else cout<<string(n/2+1,'4')<<string(n/2+1,'7')<<endl;
}
