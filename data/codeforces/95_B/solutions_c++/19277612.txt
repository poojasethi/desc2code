#include<cstdio>
#include<stdio.h>
#include<string>
#include<iostream>
using namespace std;
	string s,t;
int main(){
	cin>>s;
	int n=s.size();
	if(n%2==0){
		if(string(n/2,'7')+string(n/2,'4')<s){
			cout<<string(n/2+1,'4')<<string(n/2+1,'7')<<endl;
		}
		else{
			int fo=n/2,se=n/2;
			int flag=0;
			for(int i=0;i<n;i++){
				int ok=0;
				if(fo>0){
					if(flag==1||s[i]<'4'){
						ok=1;
					}
					else if(s[i]=='4'){
						for(int j=i+1;j<=n;j++){
							if(j==n){
								ok=1;
								break;
							}
							char ch=(j<=i+se?'7':'4');
							if(ch>s[j]){
								ok=1;
							}
							if(ch!=s[j]){
								break;
							}
						}
					}
				}
				if(ok){
					t+="4";
					fo--;
				}
				else{
					t+="7";
					se--;
				}
				if(t[i]>s[i]){
					flag=1;
				}
			}
			cout<<t<<endl;
		}
	}
		else{
			cout<<string(n/2+1,'4')<<string(n/2+1,'7')<<endl;
		}
		return 0;
}
