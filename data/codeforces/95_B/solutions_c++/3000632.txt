#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<string>
using namespace std;
const int maxn = 100055;
int dp[maxn];
int len,finish;
string ans,tmp,fans;
void dfs(int l,int c4,int c7,int e)
{
	if(finish) return;
	if(l == len )
	{
		fans = ans;
		finish = 1;
		return;
	}
	if(e)
	{
		//cout<<c4<<endl;
		if(c4*2 < len)	ans[l]='4',dfs(l+1,c4+1,c7,e);
		else ans[l]='7',dfs(l+1,c4,c7+1,e);
	}
	else
	{
		//cout<<tmp[l]<<endl;
		if(tmp[l] <= '4'&&c4*2 < len) ans[l]='4',dfs(l+1,c4+1,c7,tmp[l]<'4');
		if(tmp[l] <= '7'&&c7*2 <len) ans[l]='7', dfs(l+1,c4,c7+1,tmp[l]<'7');		
	}
}
int main()
{
	char str[maxn];
	
	
	while(scanf("%s",str)!=EOF)
	{
		len = strlen(str);
		if(len%2 == 1) 
		{
			for(int i = 1; i <=len+1 ; i++)
			{
				if(i <= (len+1)/2) printf("4");
				else printf("7");
			}
			printf("\n");
		}
		else
		{
			tmp = str;
			fans = "";
			finish = 0;
			
			while(fans == "")
			{
				ans = string(len,'0');
				dfs(0,0,0,0);
				tmp = "00"+tmp;
				len+=2;
			}
			cout<<fans<<endl;
		}
	}
	return 0;
}
