#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;

string num;
string ans;

int n;
bool ok;

void complete( int pos, int n4, int n7 ) {
	for ( int i = 0; i < n4; i ++ ) ans[pos + i] = '4';
	for ( int i = 0; i < n7; i ++ ) ans[pos + n4 + i] = '7';
	ok = true;
}

void solve ( int pos, int n4, int n7 ) {

	if ( pos == n ) {
		ok = true;
		return;
	}

	int act = num[pos] - '0';

	if ( act < 4 ) {
		complete( pos, n4, n7 );
		return;
	}

	if ( act == 4 && n4 ) {
		ans[pos] = '4';
		solve( pos + 1, n4 - 1, n7 );
		if ( ok ) return;
	}

	if ( act < 7 && n7 ) {
		ans[pos] = '7';
		complete( pos + 1, n4, n7 - 1 );
		return;
	}

	if ( act == 7 && n7 ) {
		ans[pos] = '7';
		solve( pos + 1, n4, n7 - 1);
	} 
}

int main() {
	
	cin.tie(0);
	ios_base::sync_with_stdio(0);

	cin >> num;
	ans = num;

	n = num.size();

	ok = false;

	if ( (n & 1) == 0 ) solve( 0, n / 2, n / 2 );

	if ( !ok ) {
		n ++;
		for ( int i = 0; i < (n + 1) / 2; i ++ ) cout << "4";
		for ( int i = 0; i < (n + 1) / 2; i ++ ) cout << "7";
		cout << "\n";
	} else
		cout << ans << "\n";

}
