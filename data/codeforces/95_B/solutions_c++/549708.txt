#include <iostream>
#include <cstring>

using namespace std;
string s,res;
int kq=0;
int n;

void find(int pos,int ok,int cnt4,int cnt7)
{

    if (pos==n) kq++;
    if (kq) return;

    if (ok)
     if (cnt4)
     { res[pos]='4'; find(pos+1,ok,cnt4-1,cnt7);}
     else
     { res[pos]='7'; find(pos+1,ok,cnt4,cnt7-1);}
    else
    {
        if ('4'>=s[pos] && cnt4)
        {
            res[pos]='4';
            find(pos+1,'4'>s[pos],cnt4-1,cnt7);
        };
        if (kq) return;
        if ('7'>=s[pos] && cnt7)
        {
            res[pos]='7';
            find(pos+1,'7'>s[pos],cnt4,cnt7-1);
        }
    }
}

int main()
{
  //  freopen("LuNu.inp","r",stdin);
   // freopen("LuNu.out","w",stdout);
    cin>>s;
    n=s.length();
    if (n%2==1)
    {
        n++;
        cout << string(n/2,'4')<<string(n/2,'7');
    }
    else
    if (string(n/2,'7')+string(n/2,'4')<s)
     cout << string(n/2+1,'4')<<string(n/2+1,'7');
    else
    {
        res=string(n,'0');
        find(0,0,n/2,n/2);
        cout << res;
    }
 }
