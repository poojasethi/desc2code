#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;

const int MAXN = 100010;

int n;
char s[MAXN], ans[MAXN];

bool gao(int i, int n4, int n7, bool larger) {
	if (i == n) {
		ans[i] = '\0';
		printf("%s\n", ans);
		return true;
	}
	if (larger) {
		if (n4) {
			ans[i] = '4';
			return gao(i + 1, n4 - 1, n7, larger);
		} else {
			ans[i] = '7';
			return gao(i + 1, n4, n7 - 1, larger);
		}
	} else {
		ans[i] = '4';
		if (n4 && s[i] <= '4' && gao(i + 1, n4 - 1, n7, s[i] < '4'))
			return true;
		ans[i] = '7';
		if (n7 && s[i] <= '7' && gao(i + 1, n4, n7 - 1, s[i] < '7'))
			return true;
		return false;
	}
}

int main() {
	while (scanf("%s", s) != EOF) {
		n = strlen(s);
		if (n % 2 == 1 || !gao(0, n / 2, n / 2, false)) {
			n = n / 2 + 1;
			for (int i = 0; i < n; ++i)
				putchar('4');
			for (int i = 0; i < n; ++i)
				putchar('7');
			putchar('\n');
		}
	}
	return 0;
}
