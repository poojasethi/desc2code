#include<string>
#include<algorithm>
#include<iostream>
#include<cstdio>
using namespace std;
string s;
const int maxn=1e5+7;
char t[maxn];
int l;
//need =0 ganghao 1 binida -1 binixiao

void print()
{
    for(int i=0;i<l;i++)printf("%c",t[i]);
    exit(0);
}

void dfs(int pos,int flg,int num4,int num7)
{
    if(pos==l)
    {
        print();
    }
    if(flg)
    {
        if(num4>0)
        {
            t[pos]='4';
            dfs(pos+1,flg,num4-1,num7);
        }
        else
        {
            t[pos]='7';
            dfs(pos+1,flg,num4,num7-1);
        }
    }
    if(s[pos]<='4'&&num4>0)
    {
        t[pos]='4';
        dfs(pos+1,t[pos]>s[pos]?1:0,num4-1,num7);
    }
    if(s[pos]<='7'&&num7>0)
    {
        t[pos]='7';
        dfs(pos+1,t[pos]>s[pos]?1:0,num4,num7-1);
    }
}

void sb_print(int a)
{
    for(int i=0;i<a;i++)printf("4");
    for(int i=0;i<a;i++)printf("7");
    exit(0);
}
int main()
{
    cin>>s;
    l=s.length();
    if(l%2)
    {
        sb_print(l/2+1);
    }
    dfs(0,0,l/2,l/2);
    sb_print(l/2+1);
}
