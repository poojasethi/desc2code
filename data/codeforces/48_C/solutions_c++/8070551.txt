#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;

int a[10005];
int main() {
	int n; scanf("%d", &n);
	for (int i=1; i<=n; ++i) scanf("%d", &a[i]);
	double mi = 0x7fffffff;
	double ma = 1;
	for (int i=1; i<=n; ++i) {
		mi = min(mi, (a[i]+1)*1.0/i);
		ma = max(ma, a[i]*1.0/i);
	}
	int ans1 = floor((n+1)*mi-0.0001);
	int ans2 = floor((n+1)*ma);
	if (ans1 == ans2) printf("unique\n%d\n", ans1);
	else printf("not unique\n");
}
