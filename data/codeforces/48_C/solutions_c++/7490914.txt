#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int n,a,b,c = 1e9;
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&a);
        b = max(b,(n+1)*a/(i+1));
        c = min(c,(int)((n+1)*(a+1-1e-9)/(i+1)));
    }
    if(b == c) printf("unique\n%d",b);
    else printf("not unique");
    return 0;
}
