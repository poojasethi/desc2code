#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
double mins=1e18,maxs=0;
int n;
int main() 
{
	cin>>n;
	double t;
	for(int i=1;i<=n;i++) 
	{
		scanf("%lf",&t);
		mins=min(mins,(t+1-1e-8)/i);
		maxs=max(maxs,t/i);
	}
	n++;
	int low=(int)(n*maxs+1e-9);
	int high=(int)(n*mins+1-1e-9);
	if(high-low==1) 	cout<<"unique\n"<<low<<endl;
	else 	cout<<"not unique"<<endl;
	return 0;
}
