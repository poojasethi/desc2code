#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	double x,mmin,mmax,a,b;
	int n;
	cin>>n>>x;
	mmin = x;
	mmax = x + 1.;
	for(int i = 2; i <= n; i++)
	{
		cin>>x;
		mmin = max(mmin,x / double(i));
		mmax = min(mmax,(x + 1.) / double(i));
	}
	a = (n + 1.) * mmin;
	b = (n + 1.) * mmax;
	if (b == floor(b)) b--;
	if (int(a) == int(b)) cout<<"unique"<<endl<<int(a); else cout<<"not unique";
	return 0;
}
