#include<cstdio>

int N;
int station[1001];
long double mina,maxa;
long double a;
int s1;

int main(){
 int i;
 mina=1;
 maxa=1000001;
 scanf("%d",&N);
 for(i=1;i<=N;i++){
  scanf("%d",&station[i]);
  a=((long double)station[i]*10)/i;
  if(a>mina) mina=a;
 }
 s1=(int)((mina*i)/10);
 maxa=(((long double)s1+1)*10)/i;
 for(i=1;i<=N;i++){
  if((int)((maxa*i)/10)!=station[i]) break;
 }
 if(i<=N){
  printf("unique\n");
  printf("%d\n",s1);
 }
 else{
  printf("not unique\n");
 }

 return 0;
}
