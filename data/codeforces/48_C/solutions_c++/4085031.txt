#include <iostream>
#include <math.h>
#include <cmath>
#include <cstring>
#include <algorithm>
#include <cstdio>
#include <string>
using namespace std;

double mn,mx,A,B;
int ans1,ans2,n;
int a[10010];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	mx=10000000;
	mn=1;
	for (int i=1;i<=n;i++)
	{
		A=i;
		B=a[i];
		mx=min(mx,(1+B)/A);
		mn=max(mn,B/A);
	}
	ans1=int((n+1)*(mx)-0.0001);
	ans2=int((n+1)*mn);
	if (ans1==ans2) printf("unique\n%d\n",ans1);
	else
		printf("not unique\n");
}
