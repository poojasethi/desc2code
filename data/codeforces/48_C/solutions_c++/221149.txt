#include <iostream>

using namespace std;

int n;
double a, l = 1e+9, r = -(1e-9);

int main()
{
    cin >> n;
    for ( int i = 0; i < n; ++i )
    {
        cin >> a;
        l = min (l, (a+1)/(i+1));
        r = max (r, a/(i+1));
    }

    if (  (int) ( l*(n+1) - (1e-7) ) != (int) (r*(n+1))  )
        cout << "not unique";
    else
        cout << "unique" << endl << (int) (r*(n+1));
    return 0;
}
