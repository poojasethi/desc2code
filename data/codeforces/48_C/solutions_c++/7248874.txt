#include<iostream>
#include<cmath>
using namespace std;
double eps=0.000001;
int main()
{
	int n;
	cin>>n;
	double l=0,r=-1;
	int a;
	for(int i=1;i<=n;i++)
	{
		cin>>a;
		if(a/(double)i>l)l=a/(i*1.0);
		if(r<0||(a+1)/(i*1.0)<r)r=(a+1)/(i*1.0);
	}
	int lb=floor(l*(n+1)+eps),rb=ceil((r*(n+1)-1)-eps);
	if(lb==rb)cout<<"unique\n"<<lb;
	else cout<<"not unique\n";
}
