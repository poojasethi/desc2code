#include <iostream>
using namespace std;
#define EPS (1e-9)

int main() {
	int n,stop; cin>>n;
	double UB=1e10, LB=-1,temp;
	for(int i=1;i<=n;i++) {
		cin>>stop;
		temp = stop * 10.0 / i;
		if(LB < temp) LB = temp;
		temp = ( stop + 1 ) * 10.0 / i;
		if(UB > temp) UB = temp;
	}

	int min = ( n+1 ) * (LB+EPS) / 10;
	int max = ( n+1 ) * (UB-EPS) / 10;
	if(max != min) cout<<"not unique"<<endl;
	else cout<<"unique"<<endl<<min<<endl;

	return 0;
}
