#include <iostream>
using namespace std;
int n, ans1, ans2;
double amax=1.0e9, amin=1.0, x, y, i;
int main()
{
	cin>>n;
	for (i=x=1.0; x<n+0.1; x++) {
		cin>>y;
		for (; i<=y; i++) {
			amin = max(amin, i/x);
		}
		amax = min(amax, i/x);
	}
	ans1=(n+1)*(amax-1e-9);
	ans2=(n+1)*(amin+1e-9);
	if (ans1==ans2) cout<<"unique"<<endl<<ans1<<endl;
	else cout<<"not unique"<<endl; 
	return 0;
}
