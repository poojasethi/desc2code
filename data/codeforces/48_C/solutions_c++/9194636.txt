#include <iostream>
using namespace std;
int i,n,a;
double e,b=0,c=1e9;
int main(){
	cin>>n;
	for(;i<n;i++){
		cin>>a;
		e=a/(i+1.0);
		if(b<e)b=e;
		e=(a+1-1e-9)/(i+1.0);
		if(c>e)c=e;
	}
	int bb=b*(n+1),cc=c*(n+1);
	if (bb==cc) cout<<"unique\n"<<bb;
	else cout<<"not unique\n";
}
