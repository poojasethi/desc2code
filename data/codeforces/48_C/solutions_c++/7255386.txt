#include <cstdio>
#include <algorithm>
using namespace std;
int main() {
	int n;
	scanf("%d", &n);
	double Max = 1e9, Min = 1.0, x = 1.0;
	for(int i = 1, p; i <= n; i++) {
		scanf("%d", &p);
		for( ; x <= p; x ++) {
			Min = max(Min, x / i);
		}
		Max = min(Max, x / i);
	}
	int ans1 = (int)((n+1) * (Max - 1e-9));
	int ans2 = (int)((n+1) * (Min + 1e-9));

	if(ans1 == ans2) {
		puts("unique");
		printf("%d\n", ans1);
	} else puts("not unique");
	return 0;
}
