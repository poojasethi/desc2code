#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
    int i,n,x;
    double a,b,ans;
    while(cin >> n)
    {
        a = 10;
        b = 10000000000.0;
        for(i=1;i<=n;i++)
        {
            cin >> x;
            a = max((x*10.0)/i,a);
            b = min((x*10.0+10)/i,b);
        }
        a = a*(n+1);
        b = b*(n+1);
        ans = floor(a/10)*10;
        if(ans >= b || b-ans > 10.0)
        cout << "not unique" << endl;
        else
        cout << "unique" << endl << (int)(ans/10) << endl;
    }
    return 0;
}
