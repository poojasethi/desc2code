#include <cstdio>
#include <algorithm>

using namespace std;

int n;
double lb, rb;

int main()
{
//	freopen("c.in", "r", stdin);
	scanf("%d", &n);
	lb = 0;
	rb = 1e9;
	int x;
	for (int i = 0; i < n; i++) {
		scanf("%d", &x);
		lb = max(lb, (double)x * 10.0 / ((double)i + 1));
		rb = min(rb, ((double)x * 10.0 + 10.0) / ((double)i + 1));
	}
	int a = (int)((lb * (double)(n + 1) - (double)x * 10.0) / 10.0 + 1e-10), b = (int)(((rb - 1e-10) * (double)(n + 1) - (double)x * 10.0) / 10.0);
//	printf("%f %f\n", lb, rb);
//	printf("%d %d\n", a, b);
	if (a == b) {
		printf("unique\n");
		printf("%d\n", x + a);
	} else printf("not unique\n");
}
