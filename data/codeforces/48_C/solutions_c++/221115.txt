#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<cstring>
#include<cmath>
using namespace std;

int n,a[1005];

int main() {
   scanf("%d",&n);
   for (int i=0; i<n; i++) scanf("%d",&a[i]);
   double mx=(a[0]+1)*10.0,mi=a[0]*10.0;
   for (int i=1; i<n; i++) {
      mx=min(mx,(a[i]+1)*10.0/(i+1));
      mi=max(mi,(a[i])*10.0/(i+1));
   }
   mi+=1e-9; mx-=1e-9;
   if ((int)(mx*(n+1)/10.0)==(int)(mi*(n+1)/10.0)) printf("unique\n%d\n",(int)(mi*(n+1)/10.0));
      else printf("not unique\n");
}
