#include<stdio.h>
const double eps=1e-8;
int arr[1010];
int main()
{
	int n,i;
	double d=0,t=1e100,dd,tt,now;
	scanf("%d",&n);
	for(i=1;i<=n;i++) scanf("%d",&arr[i]);
	for(i=1;i<=n;i++)
	{
		dd=arr[i]*10.0/i;
		if(dd>d) d=dd;
		tt=(arr[i]+1)*10.0/i;
		if(tt<t) t=tt;
	}
	dd=((n+1)*d-arr[n]*10+eps)/10;
	tt=((n+1)*t-arr[n]*10-eps)/10;
	int aa=dd;
	int bb=tt;
	if(aa==bb)
	{
		printf("unique\n%d\n",arr[n]+aa);
	}
	else printf("not unique\n");
	return 0;
}
