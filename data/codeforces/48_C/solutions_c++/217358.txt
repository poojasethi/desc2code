#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const double EPS = 1e-8;

int n;
int a[1000];

int main() {
	scanf("%d", &n);
	double l = -1e10, r = 1e10;
	for (int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
		l = max(l, (double)a[i] / (i + 1));
		r = min(r, (double)(a[i] + 1) / (i + 1));
	}
	l *= n + 1;
	r *= n + 1;
	if ((int)l != (int)(r - EPS)) {
		puts("not unique");
	} else {
		puts("unique");
		printf("%d\n", (int)l);
	}
	return 0;
}
