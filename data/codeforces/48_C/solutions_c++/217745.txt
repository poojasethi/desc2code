#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<vector>

using namespace std;

const double eps=1e-7,inf=1e20;

double mina,maxa,a;
int x,y;
int n;

void work(){
	scanf("%d",&n);
	mina=0; maxa=inf;
	
	for (int i=1;i<=n;i++) {
		scanf("%lf",&a);
		mina=max(mina,a*(double (n+1))/(double (i)));
		maxa=min(maxa,(a+1.0)*(double (n+1))/(double (i))-eps);
	}
	
	x=floor(mina)+eps;
	y=floor(maxa)+eps;
	
	//printf("%lf %lf %d %d\n",mina,maxa,x,y);
	
	if (x==y) {
		printf("unique\n");
		printf("%d\n",x);
	}
	else printf("not unique\n");
}

int main(){
	work();

	return 0;
}
