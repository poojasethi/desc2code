#include<iostream>
#include<cmath>
using namespace std;

double eps=1e-9;

int main()
{
    int n;
    cin>>n;
    double kaf=0;
    double saqf=1e12;
    for(int i=1;i<=n;i++)
    {
       double x;
       cin>>x;
       x++;
       saqf=min(saqf,(x)/(double)i);
       kaf=max(kaf,(x-1)/(double)(i));
    }
    int maxi=(int)floor(kaf*(n+1)+eps);
    int mini=(int)floor(saqf*(n+1)-eps);
    
    if(maxi==mini)
       cout<<"unique"<<endl<<mini<<endl;
    else
       cout<<"not unique"<<endl;
    
    return 0;
}
