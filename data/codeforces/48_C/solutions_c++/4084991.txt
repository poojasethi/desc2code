#include <cstdio>
#include <algorithm>

using namespace std;

int a[10000];

int main()
{
	int N, i, j, k;
	double ll, rr;

	while( scanf("%d", &N) != EOF )
	{
		for( i = 1; i <= N; ++i )
			scanf("%d", &a[i]);
		
		k = a[N];
		if( N <= 2 )
		{
			puts("not unique");
			continue;
		}

		ll = -1e30;
		rr = 1e30;
		for( i = 1; i <= N; ++i )
		{
			ll = max(ll, (a[i]+0.)/i);
			rr = min(rr, (a[i]+1.)/i);
		}

		ll = ll*(N+1);
		rr = rr*(N+1);

		i = (int)(ll+1e-10);
		j = (int)(rr-1e-10);

		if( i == j )
			printf("unique\n%d\n", i);
		else
			printf("not unique\n");
	}

	return 0;
}
