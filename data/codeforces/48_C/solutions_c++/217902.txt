using namespace std ;
int n ;

int main()
{
 double mn = -1e9,mx = 1e9 ;
 int x ;
 scanf("%d",&n) ;
 for(int i = 1;i <= n;i++)
 {
  scanf("%d",&x) ;
  mn = max(mn,1. * x / i) ;
  mx = min(mx,1. * (x + 1) / i - 1e-9) ;
 }
// cout << mn << " " << mx << endl ;
 int r1 = floor((n + 1) * mn + 1e-9) ;
 int r2 = floor((n + 1) * mx + 1e-9) ;
 if(r1 == r2) printf("unique\n%d\n",r1) ;
 else printf("not unique\n") ;
 
 return 0 ;
}
