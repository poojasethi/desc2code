#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <cassert>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define ll long long

int n,i,a;
double mn,mx;
ll x,y;

int main()
{
  scanf("%d",&n);
  for(i=1;i<=n;i++)
  {
    scanf("%d",&a);
    if(i==1)
      mn=10.*(a+1)/i;
    mn=min(mn,10.*(a+1)/i-1e-10);
    mx=max(mx,10.*a/i);
  }
  assert(mx>=10);
  mn=max(mn,10.);
  x=(ll)(floor((n+1)*mn/10)+.5);
  y=(ll)(floor((n+1)*mx/10)+.5);
  if(x==y)
  {
    printf("unique\n");
    cout<<x;
  }
  else
    printf("not unique\n");
  return 0;
}
