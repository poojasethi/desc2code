#include<cstdio>
using namespace std;
int main(){
    //freopen("in.txt","r",stdin);
    int n,a,k=1;
    double max=1e+10,min=0.0,b,c;
    scanf("%d",&n);
    while(k<=n){
        scanf("%d",&a);
        b=10.0*a;c=10.0*(a+1);
        if(min<b/k)min=b/k;
        if(max>c/k)max=c/k;
        k++;
    }
    min=(n+1)*min/10+1e-9;max=(n+1)*max/10-1e-9;//printf("%llf %llf",min,max);
    if((int)min<(int)max)printf("not unique\n");
    else{printf("unique\n%d",(int)min);}
    return 0;
}
