#include<cstdio>
#include<cmath>
int N,k;
double low , up ;
bool ok = true;

int main(){
	scanf("%d",&N);
	low = 10;
	up = 10000000000.0;
	for (int i=1;i<=N;++i){
		scanf("%d",&k);
		if ((k*10.0)/i>low) low=(k*10.0)/i;
		if ((k*10.0+10)/i<up) up = (k*10.0+10)/i;
	}
	double ans;
//	printf("%f  %f  %f\n",low,up,ans);
	
	low *= N+1;
	up *= N+1;
	//if (low<k*10) low=k*10;

	ans = floor(low/10)*10;
	
//	printf("%f  %f  %f\n",low,up,ans);
	if (ans>=up || up-ans>10.0){
		printf("not unique\n");
	}else{
		printf("unique\n");
		printf("%d\n",(int)(ans/10));
	}
	
//	printf("%f  %f\n",low,up);
	
	
	return 0;
}
