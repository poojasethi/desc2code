#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<cmath>
#include<algorithm>
using namespace std;
double eps=0.000001;
int main()
{
	int n;
	while(cin>>n)
	{
		double l=0,r=-1;
		int k;
		for(int i=1;i<=n;i++)
		{
			scanf("%d",&k);
			if(k/(double)i>l)l=k/(i*1.0);
			if(r<0||(k+1)/(i*1.0)<r)r=(k+1)/(i*1.0);
		}
		int lb=floor(l*(n+1)+eps);
		int rb=ceil((r*(n+1)-1)-eps);
		if(lb==rb)
			printf("unique\n%d\n",lb);
		else
			printf("not unique\n");
	}
}
