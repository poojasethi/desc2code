#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
	double a[10009],min,max,t;
	int n,i,j;
	while(scanf("%d",&n)!=-1)
	{
		for(i=0;i<n;i++)
			scanf("%lf",&a[i]);
		min=0;
		for(i=0;i<n;i++)
		{
            t=10.0*a[i]/(i+1);
			if(t>min) min=t;
		}
		max=100000000;
		for(i=0;i<n;i++)
		{
            t=10.0*(a[i]+1)/(i+1)-0.0000001;
			if(t<max) max=t;
		}
		int sa=min*(n+1)/10.0;
		int sb=max*(n+1)/10.0;
		if(sa==sb)
		{
			printf("unique\n");
			printf("%d\n",sa);
		}
		else printf("not unique\n");
	}
	return 0;
}
