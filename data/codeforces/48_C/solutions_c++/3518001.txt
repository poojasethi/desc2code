#include<cstdio>
const double eps=1e-8;
int main(){
	int n,i,x,pre,p,q;
	double l,r,nl,nr;
	while(~scanf("%d",&n)){
		pre=0;
		l=0,r=100000000;
		for(i=1;i<=n;i++){
			scanf("%d",&x);
			nl=(x-pre)*10,nr=nl+10;
			nl+=pre*10,nr+=pre*10;
			nl/=i,nr/=i;
			if(nl>l)l=nl;
			if(nr<r)r=nr;
			pre=x;
		}
		p=((n+1)*l-pre*10+eps)/10,q=((n+1)*r-pre*10-eps)/10;
		if(q!=p)puts("not unique");
		else printf("unique\n%d\n",x+p);
	}
	return 0;
}
