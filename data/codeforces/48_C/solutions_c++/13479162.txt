#include<stdio.h>
#include<cmath>
//#include <algorithm>
using namespace std;
int main()
{
	int n,i;
	double a[1001],b[1001],c[1001],max,min;
	scanf("%d",&n);
	a[0]=0;
	for (i=1;i<=n;i++)
	{
		scanf("%lf",&a[i]);
	}
	for (i=1;i<=n;i++)
	{
		c[i]=a[i]/i;
		b[i]=(a[i]+1-1e-7)/i;
	}
	min=0; max=1000000;
	for (i=1;i<=n;i++)
	{
		if (b[i]<max) max=b[i];
		if (c[i]>min) min=c[i];
	}
	max=floor(max*(n+1));
	min=floor(min*(n+1));
	if (fabs(max-min)<1e-7 )
	{
		printf("unique\n");
		printf("%.0lf\n",max);
	}
	else printf("not unique\n");
}
