#include <iostream>
#include <cmath>

using namespace std;

const int MAXn=1000;
const int INF=~0u/2;
double list[MAXn+10];

int main()
{
  int n;
  double mina=0, maxa=INF;
  cin >> n;
  
  for(int i=0; i<n; i++)
    cin >> list[i];

  for(int i=0; i<n; i++)
    {
      mina=max(mina, list[i]*10/(i+1));
      maxa=min(maxa, (list[i]+1)*(10)/(i+1));
    }

  int k1, k2;
  
  k1=(mina)*(n+1)/10;
  k2=ceil((maxa)*(n+1)-1)/10;

  //cout << k1 << " " << k2 << endl;

  if(k1!=k2)
    cout << "not unique" << endl;
  else
    {
      cout << "unique" << endl;
      cout << k1 << endl;
    }
  return 0;
}
