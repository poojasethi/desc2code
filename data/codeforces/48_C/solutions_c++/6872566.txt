#include<iostream>
#include<cmath>
using namespace std;
#define eps 1e-9
int main(){
    int n,prev;
    cin>>n;
    double mxl,mnu,x;
    mnu=1000000;
    mxl=-mnu;
    for(int i=1;i<=n;i++){
        cin>>x;
        mxl=max(mxl,x/i);
        
        mnu=min(mnu,(x+1)/i);
        //cout<<mxl<<" "<<mnu<<endl;
    }
    n++;
    int a=(int)(mxl*n);
    int b=(int)(mnu*n);
    if(((mnu*n)-b) < eps){
        b--;
    }
    if( abs(a-b)<=0){
        cout<<"unique\n";
        cout<<int(mxl*n)<<endl;
    }
    else{
        cout<<"not unique\n";
    }   
}
