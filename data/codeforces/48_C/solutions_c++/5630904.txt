#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
double l,r;
int L,R,n,x,i;
int main(){
	l=1; r=1e9;
	scanf("%d",&n);
	for(i=1;i<=n;++i){
		scanf("%d",&x);
		l=max(l,x*1.0/i);
		r=min(r,(x+1)*1.0/i);
	}
	L=int(l*(n+1));
	R=int(r*(n+1)-1e-8);
	if(L==R)printf("unique\n%d\n",L);
	else printf("not unique\n");
}
