#include <bits/stdc++.h>
typedef long long  ll;
const int inf=1e+9;
const double eps=1e-12;
using namespace std;
double a[1010];
int main(){
    int n;cin>>n;
    for (int i=1; i<=n; i++) {
        cin>>a[i];
    }
    double l=a[1]*10,r=(a[1]+1)*10-1;
    for (int i=1; i<=n; i++) {
        double t=(a[i]*10)/i;
        double t2=(a[i]+1)*10/i;
        l=max(l, t);
        r=min(r,t2);
    }
    double lp=(n+1)*l-10*a[n];
    double rp=(n+1)*r-10*a[n];
    int k=lp/10;
    if (rp-eps-k*10.0>10) {
        printf("not unique\n");
    }
    else    printf("unique\n%d\n",(int)a[n]+k);
    return 0;
}
