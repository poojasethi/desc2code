#include <cstdio>
#include <cstring>
#include <vector>
#include <cmath>
#include <algorithm>
using namespace std;

int a[1005];
double L=10, R=1e9;
int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=1;i<=n;i++) scanf("%d", &a[i]);
	for(int i=1;i<=n;i++) {
		L = max(10.*a[i]/i, L);
		R = min((10.*a[i]+10. - 1e-9)/i, R);
	}
	int tL = (int)floor(((n+1)*L)/10.);
	int tR = (int)floor(((n+1)*R)/10.);
	if(tL != tR) puts("not unique");
	else {
		puts("unique");
		printf("%d\n", tL);
	}
	return 0;
}
