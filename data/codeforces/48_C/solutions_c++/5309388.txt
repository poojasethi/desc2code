#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main()
{
int n,p[1500];
int i,o,t1,t2;
double x,y,t;
while(scanf("%d",&n)==1)
{
   memset(p,0,sizeof(p));
   x=0;
   y=9999999;
   for(i=1;i<=n;i++)
   {
      scanf("%d",&p[i]);
      t=p[i];
      if(x<(t+0.00001)/i)
      {
         x=(t+0.0001)/i;
      }
      if(y>(t+0.99999)/i)
      {
         y=(t+0.99999)/i;
      }
   }
   t1=(x*(n+1))/1;
   t2=(y*(n+1))/1;
   if(t1==t2)
   {
      printf("unique\n%d\n",t1);
   }   
   else
   {
      printf("not unique\n");
   }
}
return 0;
}
