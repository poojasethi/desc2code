#include <cstdio>
#include <vector>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <cctype>
using namespace std;

int main() {
	int n;
	while (~scanf("%d",&n)) {
		double lo = 0;
		double hi = 1e20;
		for (int i = 1 ; i <= n ; i ++) {
			double pos;
			scanf("%lf",&pos);
			lo = max(lo , pos / i);
			hi = min(hi , (pos + 1) / i);
		}
	//	printf("%lf %lf\n",lo,hi);
		lo *= n + 1;
		hi *= n + 1;
		if ((int)lo != (int) (hi - 1e-8)) {
			puts("not unique");
		} else {
			puts("unique");
			printf("%d\n",(int)lo);
		}
	}
	return 0;
}
