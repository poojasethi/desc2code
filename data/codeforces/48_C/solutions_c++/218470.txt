#include <cstdio>
#include <cmath>

using namespace std;

int main()
{
    int n,x;
    double la=10,ra=1E9;
    scanf("%d",&n);
    for (int i=0;i<n;++i)
    {
        scanf("%d",&x);
        double lla = x*10./(i+1);
        double rra = (x+1)*10./(i+1);
        if (lla>la)
            la=lla;
        if (rra<ra)
            ra=rra;
    }
    double xl = floor(la*(n+1)/10);
    double xr = ceil(ra*(n+1)/10)-1;
    if (fabs(xr-xl)>0.5)
        printf("not unique");
    else
        printf("unique\n%0.0lf",xl);
    return 0;
}
