#include <iostream>
#include <cmath>

using namespace std;

int main ()
{
    int n;
    cin >> n;
    double lo=0.0, hi=10000000.0;
    for (int i=1; i<=n; i++)
    {
        int sta;
        cin >> sta;
        lo = max(lo, (double) 10.0 * sta / i);
        hi = min(hi, (double) 10.0 * (sta+1) / i);
    }
    int nextlo = floor(lo * (n+1) / 10.0);
    int nexthi = ceil(hi * (n+1) / 10.0);
    //cout << lo << " "<< hi << " " << nextlo << " " << nexthi << endl;
    if (nextlo +1 >= nexthi)
    {
        cout << "unique" << endl << nextlo << endl;
    }
    else{ cout << "not unique" << endl;}
    return 0;
}
