#include<iostream>
#include<algorithm>
using namespace std;
int main() {
	double mn=1e20,mx=0;
	int n;
	cin>>n;
	for(int i=1;i<=n;i++) {
		double t;
		cin>>t;
		mn=min(mn,(t+1-1e-8)/i);
		mx=max(mx,t/i);
	}
	n++;
	int low=(int)(n*mx+1e-9),high=(int)(n*mn+1-1e-9);
	if(low+1==high) {
		cout<<"unique\n"<<low;
	} else {
		cout<<"not unique";
	}
}
