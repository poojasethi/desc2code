n = input()
S = map(int,raw_input().split())
amin = 0
amax = 99999999
for i in range(n):
	amin = max(amin,float(S[i])/(i+1))
	amax = min(amax,float(S[i]+1)/(i+1))

smin = S[-1]+int(amin*(n+1)-S[-1]+1e-9)
smax = S[-1]+int(amax*(n+1)-S[-1]-1e-9)

if smin==smax:
	print "unique"
	print smin
else:
	print "not unique"
