#!/usr/bin/python
import sys
n = int (sys.stdin.readline ())
a = [int (x) for x in sys.stdin.readline ().split ()]

n += 1
a = [0] + a
lop, loq = 1, 1
hip, hiq = 10 ** 8, 1
for i in range (1, n):
	cp = a[i] - a[0]
	cq = i
	if lop * cq < cp * loq:
		lop, loq = cp, cq
	if hip * cq > (cp + 1) * hiq:
		hip, hiq = cp + 1, cq
x = (lop * n) / loq
y = (hip * n - 1) / hiq
if x == y:
	print 'unique'
	print x
else:
	print 'not unique'
