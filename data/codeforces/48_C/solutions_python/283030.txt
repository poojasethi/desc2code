n, a = input (), map (int, raw_input ().split ())
minp = max (float (a[i]) / (i + 1) for i in xrange (n))
maxp = min (float (a[i] + 1) / (i + 1) for i in xrange (n))
x = int (minp * (n + 1) + 1e-9)
y = int (maxp * (n + 1) - 1e-9)
if x == y:
    print 'unique'
    print x
else:
    print 'not unique'
