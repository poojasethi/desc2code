n = input()
S = map(int,raw_input().split())
amin = 0
amax = 1e20
t = 0
for s in S:
	t += 1
	amin = max(amin,float(s)/t)
	amax = min(amax,float(s+1)/t)

#print amin,amax

smin = S[-1] + int(amin*(n+1)-S[-1]+1e-9)
smax = S[-1] + int(amax*(n+1)-S[-1]-1e-9)

#print smin,smax

if smin==smax:
	print "unique"
	print smin
else:
	print "not unique"
