#include<bits/stdc++.h>
using namespace std;
int l, r;
string s, t;
int main()
	{
	cin >> s >> t;
	int n = s.size();
	int m = t.size();
	int ind = 0;
	for(int i = 0; i < m; i++)
		{
		if(t[i] == s[ind])
			ind++;
		if(ind >= n)
			{
			l = i;
			break;
			}
		}
	ind = n - 1;
	for(int i = m - 1; i > l; i--)
		{
		if(t[i] == s[ind])
			ind--;
		if(ind < 0)
			{
			r = i;
			break;
			}
		}
	if(r)
		cout << r - l;
	else
		cout << 0;   
	}
