#include <stdio.h>
#include <algorithm>
#include <iostream>

#define read(x) scanf("%d",&x)

using namespace std;


int n,pos,q1,q2;
string str,str2;

int main(){
    cin>>str>>str2;
    n=str2.size();
    for(int k=0;k<n;k++){
        if(str2[k]==str[pos]){pos++;if(pos==str.size()){q1=k;break;}}
    }
    pos=str.size()-1;
    for(int k=n-1;k>=0;k--){
        if(str2[k]==str[pos]){pos--;if(pos==-1){q2=k;break;}}
    }
    cout<<max(0,q2-q1)<<endl;

	return 0;
}
