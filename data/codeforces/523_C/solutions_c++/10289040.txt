#include<iostream>
#include<string>
using namespace std;

int main()
{
	string s,s1;
	cin>>s;
	cin>>s1;
	int n=s.length();
	int n1=s1.length();
	if(2*n>n1){cout<<0;return 0;}
	int i=0,j=0;
	while(i<n and j<n1){
		if(s1[j]==s[i]) i++;
		j++;
	}
	if(i<n){cout<<0;return 0;}
	int l=j-1;
	i=n-1;j=n1-1;
	while(i>=0 and j>=0){
		if(s1[j]==s[i]) i--;
		j--;
	}
	int r=j+1;
	if(r<=l) cout<<0;
	else cout<<r-l;
	return 0;
}
