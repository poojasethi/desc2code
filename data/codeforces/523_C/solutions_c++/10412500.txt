#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    char s[1010],t[1000010];
    scanf("%s",s);
    scanf("%s",t);
    int i,j,a,b,c;
    for(i=0,j=0;i<strlen(s)&&j<strlen(t);j++)
        {
            if(s[i]==t[j]) i++;
        }
    a=j-1;
    for(i=strlen(s)-1,j=strlen(t)-1;i>=0&&j>=a;j--)
    {
        if(s[i]==t[j]) i--;
    }
    b=j+1;
    if(b>a) c=b-a;
    else c=0;
    printf("%d\n",c);
    return 0;
}
