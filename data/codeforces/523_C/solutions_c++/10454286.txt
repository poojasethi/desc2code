#include <cstdio>
#include <cstring>
#define maxn 1000010
char s[1010],t[maxn];
int main()
{
	scanf("%s%s",s,t);
	int len1=strlen(s);
	int len2=strlen(t);
	int i,k=0,flag=0,lef=0,rig=0;
	for (i=0;i<len2;i++) {
		if (s[k]==t[i]) {
			if (k==len1-1) {
				flag=1;
				lef=i,rig=i;
				break;
			}
			k++;
		}
	}
	if (flag) {
//		k--;
		for (i=len2-1;i>lef;i--) {
			if (s[k]==t[i]) {
				if (k==0) {
					rig=i;
					break;
				}
				k--;
			}
		}
	}
	printf("%d",rig-lef);
	return 0;
}
