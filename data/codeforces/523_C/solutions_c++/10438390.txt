#include<bits/stdc++.h>
using namespace std;
char s1[1005],s2[1000005];

int main(void){
	int i,j;
	while(~scanf("%s%s",s1,s2)){
		int len1=strlen(s1),len2=strlen(s2);
		int l=0,r=len2-1,ok1=0,ok2=0;
		for(i=0,j=0;i<len2;i++){
			if(s1[j]==s2[i]) j++;
			if(j==len1){
				ok1=1,l=i;break;
			}
		}
		for(i=len2-1,j=len1-1;i>=0;i--){
			if(s1[j]==s2[i]) j--;
			if(j==-1){
				ok2=1,r=i;break;
			}
		}
		if(l>=r || ok1==0 || ok2==0) puts("0");
		else printf("%d\n",r-l);
	}
	return 0;
}
