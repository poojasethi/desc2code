#include<stdio.h>
#include<string.h>
char t[1000001];
int main()
{
	char s[1001];
	scanf("%s%s", s, t);
	int l1=strlen(s), l2=strlen(t);
	int i=0, j=0;
	while(i<l1 && j<l2)
	{
		if(s[i]==t[j]){i++;j++;}
		else j++;
	}
	if(i<l1){printf("0\n");return 0;}	
	i=l1-1;
	int last=-1;
	for(int k=l2-1;k>=j;k--)
	{
		if(s[i]==t[k])i--;
		if(i<0){last=k-1;break;}
	}
	if(last==-1){printf("0\n");return 0;}
	printf("%d\n", last-j+2);
	return 0;
}
