#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

string s, t;

int main() {
	ios_base::sync_with_stdio(0);
	cin >> s >> t;

	int left_s = 0, left_t = 0;
	while (left_s < s.length() && left_t < t.length())
	{
		if (s[left_s] == t[left_t]) ++left_s;
		++left_t;
	}

	int right_s = s.length() - 1, right_t = t.length() - 1;
	while (right_s >= 0 && right_t >= 0) {
		if (s[right_s] == t[right_t]) --right_s;
		--right_t;
	}

	cout << max(right_t - left_t + 2, 0) << endl;
	return 0;
}
