#include <iostream>
#include <string>
using namespace std;

int main()
{
	string s, t;
	cin >> s >> t;
	int p1 = 0, p2 = t.size() - 1;
	for (int i = 0; i < s.size() && p1 < t.size(); p1++)
		if (s[i] == t[p1])
			i++;
	for (int i = s.size() - 1; i >= 0 && p2 >= 0; p2--)
		if (s[i] == t[p2])
			i--;
	cout << max(p2 - p1 + 2, 0) << endl;
	return 0;
}
