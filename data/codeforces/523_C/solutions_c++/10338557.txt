#include <iostream>
#include <string>

using namespace std;

int main()
{
	string name, str;

	cin >> name >> str;

	int pos(0), pos2(name.size() - 1);
	int l(0), r(0);

	for(int i = 0, j = str.size() - 1; i < str.size(); i++, j--) {
		if(str[i] == name[pos]) {
			pos++;
			if(pos == name.size())
				l = i;
		}

		if(str[j] == name[pos2]) {
			if(pos2 == 0)
				r = j;
			pos2--;
		}
	}
	
	if(r <= l) cout << 0 << endl;
	else cout << r - l << endl;
	return 0;
}
