#include <stdio.h>
#include <string.h>
#include <math.h>
int main()
{
   char a[1500],b[1000000];
   int num=0,i,x,y=0,z=0,l1,l2;
   gets(a);
   gets(b);
   l1=strlen(a);
   l2=strlen(b);
	for(i=0,x=0;x<l2;x++)
   {
   	if(a[i]==b[x])
		i++;
	if(i==l1)
	{
		y=x;
		break;
	}
   }
   for(i=(l1-1),x=(l2-1);x>y;x--)
   {
   	if(a[i]==b[x])
		i--;
	if(i==-1)
	{
		z=x;
		break;
	}
   }
   if(z>=y)
   num=num+(z-y);
   printf("%d\n",num);
	return 0;
}
