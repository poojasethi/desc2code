#include <bits/stdc++.h>
using namespace std;
int main()
{
    string s,t;
    int i,beha,x=0,y=0;

    cin>>s>>t;

    beha=s.size()-1;

    for( i=t.size()-1; i>=0; i-- ){
        if( t[i]==s[beha] )beha--;
        if( beha<0 ){
            x=i;
            break;
        }
    }

    beha=0;

    for( i=0; i<t.size(); i++ ){
        if( t[i]==s[beha] )beha++;
        if( beha==s.size() ){
            y=i;
            break;
        }
    }

    if( x-y<0 )cout<<0;
    else cout<<abs(x-y);
}
