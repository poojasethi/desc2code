#include <iostream>
#include <string>

using namespace std;

int main(){
	int j=0,st=0,en=0;
	string s,t;
	cin >> s;
	cin >> t;
	for (int i = 0; i < t.size(); ++i)
	{
		if (t[i] == s[j])
		{
			j++;
		}
		if (j == s.size())
		{
			st=i;
			break;
		}
	}
	for (int i = t.size(); i > 0; --i)
	{
		if (t[i] == s[j])
		{
			j--;
		}
		if (j < 0)
		{
			en=i;
			break;
		}
	}
	if (en-st>0)
	{
		cout << en-st;
	}
	else
		cout << 0;
	return 0;
}
