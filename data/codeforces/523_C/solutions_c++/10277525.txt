#include <bits/stdc++.h>

using namespace std;

int main()
{
	string s, t;
	cin >> s >> t;
	int i1 = 0, i2 = 0;
	while (i2 < t.length())
	{
		if (s[i1] == t[i2])
			i1++;
		if (i1 == s.length())
			break;
		i2++;
	}
	int ii1 = s.length() - 1, ii2 = t.length() - 1;
	while (ii2 >= 0)
	{
		if (s[ii1] == t[ii2])
			ii1--; 
		if (ii1 == -1)
			break;
		ii2--;
	}
	if (ii2 - i2 > 0)
		cout << ii2 - i2;
	else 
		cout << 0;
	return 0;
}
