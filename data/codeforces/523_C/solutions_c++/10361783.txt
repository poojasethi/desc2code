#include <bits/stdc++.h>
using namespace std;
const int maxn = 1000005;
char s[maxn], t[maxn];
int main() {
	scanf("%s%s", s, t);
	int ls = strlen(s);
	int lt = strlen(t);
	int p = 0, l = -1;
	for(int i = 0; i < lt; i++) {
		if(s[p] == t[i]) p++;
		if(p == ls) {
			l = i;
			break;
		}
	}
	int q = ls - 1, r = -1;
	for(int i = lt - 1; i >= 0; i--) {
		if(s[q] == t[i]) q--;
		if(q == -1) {
			r = i;
			break;
		}
	}
	if(~l && ~r && l < r) {
		printf("%d\n", r - l);
	} else printf("0\n");
	return 0;
}
