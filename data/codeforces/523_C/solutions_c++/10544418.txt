#include <iostream>
#include <vector>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
int main()
{
	string s,t;
	cin>>s>>t;
	int i=-1,j=t.size();
	int a=0,b=s.size()-1;
	while((a<s.size()||b>=0)&&i<j)
	{
		if(a<s.size())
		{
			i++;
			if(s[a]==t[i])
				a++;
		}
		if(b>=0)
		{
			j--;
			if(s[b]==t[j])
				b--;
		}
	}
	if(i<j)
		cout<<j-i<<endl;
	else
		cout<<0<<endl;
}
