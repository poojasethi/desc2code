#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    char a[1001],b[1000001];
    long int m,p,s,q;
    long int l,i,j;
    cin>>a>>b;
    m=strlen(a);
    l=strlen(b);
    p=0;
    q=0;
    for(i=0;i<l;i++)
    {if(b[i]==a[p])p++;
    if(p==m)break;
    }
    for(j=l-1;j>=0;j--)
    {
     if(b[j]==a[m-1-q])
     q++;
     if(q==m)break;
    }
    s=j-i;
    if(s>=0)
    cout<<s<<endl;
    else
    cout<<0<<endl;
    return 0;
    }
