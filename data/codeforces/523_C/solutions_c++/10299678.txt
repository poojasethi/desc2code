#include <stdio.h>
using namespace std;
int n, m, i, j, k;
char s[10000], t[2000000];
int main(){
    for (n=0, scanf("%c", &s[n]); s[n]!='\n'; n++, scanf("%c", &s[n]));
    for (m=0, scanf("%c", &t[m]); t[m]!='\n'; m++, scanf("%c", &t[m]));
    for (i=0, j=0; j<n && i<m; j+=(s[j]==t[i])?1:0, i++);
    if (j<n) printf("0"); else{
        for (j=m-1, k=n-1; k>=0 && j>=0; k-=(s[k]==t[j])?1:0, j--);
        if (j+1>=i-1) printf("%d", j-i+2); else printf("0");
    }

    return 0;
}
