#include<stdio.h>
#include<string.h>


int main()
{
char a[1004],b[1000004];
int l1,l2,i=0,j=0,x,y;
scanf("%s %s",a,b);
l1=strlen(a);
l2=strlen(b);
while(i<l1 && j<l2)
{
if(b[j]==a[i])
i++;
j++;
}


if(i==l1)
x=j-1;


else
{
printf("0");
return 0;
}

i=l1;j=l2;
while(j>=x && i>=0)
{
if(b[j]==a[i])
i--;
j--;
}


if(i>=0)
{
printf("0");
return 0;
}

y=j+1;   
printf("%d",y-x);
return 0;
}
