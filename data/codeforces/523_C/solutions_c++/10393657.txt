#include <cstdio>
#include <cstring>
char t[1005],s[1000006];
int l,L;
int solve()
{
    int i,j,k;
    for(i=0,j=0;j<L;++j)
    {
        if(t[i]==s[j])
        {
            ++i;
            if(i==l)break;
        }
    }
    if(j==L)return 0;
    for(i=l-1,k=L-1;k>j;--k)
    {
        if(t[i]==s[k])
        {
            --i;
            if(i<0)break;
        }
    }
    return k-j;
}
int main()
{
    scanf(" %s %s",t,s);
    l=strlen(t);L=strlen(s);
    printf("%d\n",solve());
    return 0;
}
