#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main() {
	string s, t;

	cin >> s >> t;

	int i = 0, j = 0;
	int len1 = s.length(), len2 = t.length();
	while (i<len1 && j<len2)
	{
		if (s[i] == t[j])
			i++;
		j++;
	}

	int p1 = j-1 , p2;
	i = s.length() - 1;
	j = t.length() - 1;
	while (i >= 0 && j >= 0){
		if (s[i] == t[j])
			i--;
		j--;
	}
	p2 = j + 1;

	cout << (p2 - p1 >= 0 ? p2 - p1 : 0);
}
