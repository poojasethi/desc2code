#include <stdio.h>
#include <string.h>
char a[10005];
char b[1000006];
int main(){
	gets(a);	gets(b);
	int n = strlen(a);
	int m = strlen(b);
	int I = 0, J = m-1, k = 0;
	for(int i = 0; i < m; ++ i){
		if(b[i] == a[k]){
			k ++;
		}
		if(k == n) break;
		I ++;
	}
	k = n-1;
	for(int i = m-1; i >= 0; -- i){
		if(b[i] == a[k]){
			k --;
		}
		if(k == -1) break;
		J --;
	}
	if(I >= J){
		puts("0");
		return 0;
	}
	printf("%d", J - I);
	return 0;
}
