#include <iostream>
#include <string.h>

using namespace std;

int main()
{
    string s,t;
    cin >> s >> t;
    int l=0,r=0;
    int i=0,j=0;
    int s_size=s.length();
    int t_size=t.length();
    while (i<s_size && j<t_size)
         if (s[i]==t[j++])
             i++;
    l=j-1;
    i=s_size-1;
    j=t_size-1;
    while (i>=0 && j>=0)
         if (s[i]==t[j--])
             i--;
    r=j+1;
    r-=l;
    if (r>0)
       cout << r;
    else
        cout << 0;
    return 0;
}
