#include<bits/stdc++.h>
using namespace std;
string a,b;
int main(){
    cin>>a>>b;
    int pos=0,m,n,i,j,k;
    m=0;
    while(pos<a.length()&&m<b.length()){
        if(b[m]==a[pos]) pos++;
        m++;
    } 
    n=b.length()-1;
    pos=a.length()-1;
    while(pos>=0&&n>=0){
        if(b[n]==a[pos]) pos--;
        n--;
    }
    cout<<max(n-m+2,0)<<endl;
}
