#include<cstdio>
#include<cstring>
#include<algorithm>
#define MAXN 1000010
using namespace std;
int n,m,l,r;
char name[MAXN],s[MAXN];
int main()
{
	int i,j;
	scanf("%s",name+1);
	n=strlen(name+1);
	scanf("%s",s+1);
	m=strlen(s+1);
	for(i=j=1;i<=m&&j<=n;i++)
		if(s[i]==name[j])
			j++;
	l=i-1;
	for(i=m,j=n;i&&j;i--)
		if(s[i]==name[j])
			j--;
	r=i+1;
	printf("%d\n",max(r-l,0));
	return 0;
}
