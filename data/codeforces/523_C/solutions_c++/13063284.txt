#include<cstdio>
#include<cstring>

char s[1111], STR[1111111];
int len_s, len_STR, l, r;

int main() {
	scanf("%s", s);
	scanf("%s", STR);
	
	len_s = strlen(s);
	len_STR = strlen(STR);
	
	l = 0;
	for(int i=0;i<len_s && l < len_STR;i++, l++)
		while(l < len_STR && s[i] != STR[l])l++;
	l--;
	r = len_STR-1;
	for(int i=len_s-1;i>=0 && r>=0; i--, r--)
		while(r>=0 && s[i] != STR[r]) r--;
	r++;
	
	if(l>r) printf("0\n");
	else printf("%d\n", r-l);
	
	return 0;
}
