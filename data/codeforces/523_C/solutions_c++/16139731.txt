#include <bits/stdc++.h>
using namespace std;
string s,t;
int main(){
	cin>>s>>t;
	int i,j=0;
	for(i=0;i<t.size();++i){
		if(j==s.size())break;
		if(s[j]==t[i])++j;
	}
	int a= i-1,l;
	j= s.size()-1;
	for(i=t.size()-1;i>=0;--i){
		if(j==-1)break;
		if(s[j]==t[i])--j;
	}
	l = i;
	l=i+1;
	if(a>=l){
		cout<<0<<endl;
	}
	else
		cout<<l-a<<endl;
}
