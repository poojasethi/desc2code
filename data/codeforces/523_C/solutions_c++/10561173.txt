#include <bits/stdc++.h>

using namespace std;

int main()
{
    string s, t;
    int i, j, k;
    cin >> s >> t;
    for ( i = 0, j = 0; i < t.size(); ++i )
        if ( t[i] == s[j] )
            if ( ++j == s.size() ) break;
    for ( k = t.size() - 1, j = s.size() - 1; k >= 0; --k )
        if ( t[k] == s[j] )
            if ( --j < 0 ) break;
    cout << max( 0, k - i );
    return 0;
}
