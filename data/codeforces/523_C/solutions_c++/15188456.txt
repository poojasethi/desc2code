#include <bits/stdc++.h>
using namespace std;
int main()
{
	string a, b;
	cin>>a>>b;
	int x=0;
	for (int i=0; i<a.size(); i++)
	{
		for (int j=x; j<b.size(); j++)
		{
			if (a[i]==b[j])
			{
				x=j+1;
				break;
			}
			else if (j==b.size()-1)
			{
				cout<<0;
				return 0;
			}
		}
	}
	x--;
	int y=b.size()-1;
	for (int i=a.size()-1; i>=0; i--)
	{
		for (int j=y; j>=0; j--)
		{
			if (a[i]==b[j])
			{
				y=j-1;
				break;
			}
		}
	}
	y++;
	cout<<max(0, y-x);
	return 0;
}
