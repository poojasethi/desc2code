#include <iostream>
#include <string>
#include <cmath>
using namespace std;
string s,s1;
int main() {
    int i,j,o=0,k=0,n,m;
    cin>>s>>s1;
    n=s1.size();
    m=s.size();
    for (i=0; i<n; i++)
    {
        if(s[o]==s1[i]) o++;
        if (o==m) break;
    }
    for (j=n-1; j>=0; j--)
    {
        if (s[m-1-k]==s1[j]) k++;
        if (k==m) break;
    }
    cout<<max(j-i,0)<<endl;
    return 0;
}
