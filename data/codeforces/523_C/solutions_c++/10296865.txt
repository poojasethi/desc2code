#include<iostream>
using namespace std;
string s, c;
int l,len, t, r;
int main(){
    cin>>s>>c;
    len = c.size();
    for(int i = 0; i < len; i ++){
        if(s[t] == c[i]) t ++;
        if(t == s.size()){ l = i; break;}
    }
    t = s.size() - 1;
    for(int i = len - 1; i >= 0; i --){
        if(s[t] == c[i]) t --;
        if(t < 0){ r = i; break;}
    }
    if(l > r){
        cout<<0;
        return 0;
    }
    cout<<r - l;
    return 0;
}
