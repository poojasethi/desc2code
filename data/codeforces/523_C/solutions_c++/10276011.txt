#include <bits/stdc++.h>
using namespace std;
string s, t;
int ls, lt, l, r, i, j;
void read(){
	cin >> s >> t;
}
void solve(){
	ls = s.size();
	lt = t.size();
	i = 0;	l = 0;
	while(l < lt && i < ls){
		if(s[i] == t[l]){
			i ++;
		}
		l ++;
	}
	i = ls - 1;	r = lt - 1;
	while(r >= 0 && i >= 0){
		if(s[i] == t[r]){
			i --;
		}
		r --;
	}
}
void write(){
	printf("%d", max(r - l + 2, 0));
}
int main(){
	read();
	solve();
	write();
	return 0;
}
