from sys import stdin

def main():
    name, search = stdin.next().strip(), stdin.next().strip()

    first, last = 0, len(search) - 1

    if len(search)<2*len(name) or len(name)==0:
        print 0
        return
    
    for l in name:
        while search[first] != l:
            first += 1
            if first > last:
                print 0
                return
        first += 1

    for l in reversed(name):
        while search[last] != l:
            last -= 1
            if first > last:
                print 0
                return
        last -= 1

    print last - first + 2


if __name__ == "__main__":
    main()
