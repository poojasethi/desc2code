s = raw_input()
t = raw_input()

j = 0
for i1,d in enumerate(t):
    if d == s[j]:
        j += 1
    if j == len(s):
       break
       
j = len(s)-1
for i2 in range(len(t)-1,-1,-1):
    d = t[i2]
    if d == s[j]:
        j -= 1
    if j == -1:
       break

if i2 - i1 > 0:
    print i2-i1
else:
    print 0
