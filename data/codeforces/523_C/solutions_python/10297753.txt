s = raw_input()
t = raw_input()

t_length = len(t)
s_length = len(s)

l = -1
r = -1

j = 0
for i in xrange(0, t_length):
    if j < s_length and s[j] == t[i]:
        j += 1
        if j == s_length:
            l = i

j = s_length - 1
for i in xrange(0, t_length):
    if j >= 0 and s[j] == t[t_length - i - 1]:
        j -= 1
        if j == -1:
            r = t_length - i - 1


if l == -2 or r == -1 or r < l:
    print 0
else:
    print r - l
