name = raw_input()
word = raw_input()

left_end = -1
right_start = -1
current_letter_i = 0

for i, letter in enumerate(word):
    if letter == name[current_letter_i]:
        current_letter_i += 1
        if current_letter_i >= len(name):
            left_end = i
            break

current_letter_i = len(name) - 1
for i, letter in enumerate(reversed(word)):
    if letter == name[current_letter_i]:
        current_letter_i -= 1
        if current_letter_i < 0:
            right_start = len(word) - i - 1
            break

if right_start > left_end:
    print right_start - left_end
else:
    print 0
