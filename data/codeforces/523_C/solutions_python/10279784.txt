s = list(raw_input())
t = list(raw_input())

j = 0
k = -1
len_s = len(s)
len_t = len(t)
for i in xrange(len_t):
    if s[j] == t[i]:
        j += 1
        if j == len_s:
            k = i
            break
l = -1
j = len_s-1
for i in xrange(len_t-1, -1, -1):
    if i == k:
        break
    if s[j] == t[i]:
        j -= 1
        if j == -1:
            l = i
            break
if l>k and l!=-1 and k!=-1:
    print l-k
else:
    print 0
