#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

int main()
{
    int n,a[105];
    cin>>n;
    for(int i = 0; i<n; i++)
        cin>>a[i];
    sort(a,a+n);
    int s = 0, e = n-1;
    cout<<"1 "<<a[s++]<<endl;
    if(a[e]>0)
        cout<<"1 "<<a[e--]<<endl;
    else
        cout<<"2 "<<a[s++]<<" "<<a[s++]<<endl;
    cout<<e-s+1<<" ";
    for(;s<=e;s++)
        cout<<a[s]<<endl;
    return 0;
}
