#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n; cin>>n;
    int a[102];
    for (int i=1; i<=n; i++)
        cin>>a[i];
    sort(a+1, a+n+1);

    cout<<"1 "<<a[1]<<endl;;
    if (a[n]>0) {cout<<"1 "<<a[n]<<endl; cout<<n-2; for (int i=2; i<n; i++) cout<<' '<<a[i]; cout<<endl;}
        else {cout<<"2 "<<a[2]<<' '<<a[3]<<endl; cout<<n-3; for (int i=4; i<=n; i++) cout<<' '<<a[i]; cout<<endl;}

    return 0;
}
