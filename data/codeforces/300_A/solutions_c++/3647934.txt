#include<iostream>
#include<algorithm>
#include<cstdio>
#define N 110
using namespace std;
int n,a[N];

int main()
{
	cin>>n;
	for (int i=0;i<n;i++) cin>>a[i];
	sort(a,a+n);
	printf("1 %d\n",a[0]);
	if (a[n-1]>0)
	{
		printf("1 %d\n",a[n-1]);
		printf("%d",n-2);
		for (int i=1;i<n-1;i++) printf(" %d",a[i]);
	} 
	else 
	{
		printf("2 %d %d\n",a[1],a[2]);
		printf("%d",n-3);
		for (int i=3;i<n;i++) printf(" %d",a[i]);
	}
}
