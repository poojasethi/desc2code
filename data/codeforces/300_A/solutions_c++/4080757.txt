#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	int n, a[110], i=0, j;
	cin>>n;
	for(int i=0; i<n; i++) cin>>a[i];
	sort(a, a+n);
	j=n-1; i=0;
	cout<<"1 "<<a[i++]<<endl;
	if(a[j]>0) cout<<"1 "<<a[j--]<<endl;
	else cout<<"2 "<<a[i++]<<" "<<a[i++]<<endl;
	cout<<j-i+1<<" ";
	for( ; i<=j; i++) cout<<a[i]<<" ";
	cout<<endl;
	return 0;
}
