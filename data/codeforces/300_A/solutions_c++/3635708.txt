#include <cstdio>
#include <algorithm>
using namespace std;
int main()
{
    int n,a[111];
    scanf("%d",&n);
    for(int i=0;i<n;i++) {
        scanf("%d",&a[i]);
        if(!a[i])
            i--,n--;
    }
    sort(a,a+n);
    int p=n-1;
    printf("1 %d\n",a[0]);
    printf("%d %d ",(a[n-1]<0?2:1),a[p--]);
    if(a[n-1]<0)
        printf("%d",a[p--]);
    printf("\n%d ",p+1);
    while(p>0)
        printf("%d ",a[p--]);
    putchar('0');
    return 0;
}
