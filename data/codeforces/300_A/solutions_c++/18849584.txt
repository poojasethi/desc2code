#include<stdio.h>
#include<algorithm>
using namespace std; 
int main()
{
	int n,i,a[105];
	scanf("%d",&n);
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	sort(a,a+n);
	printf("1 %d\n",a[0]);
	if(a[n-1]>0)
	{
		printf("1 %d\n",a[n-1]);
		printf("%d",n-2);
		for(i=1;i<n-1;i++)
		{
			printf(" %d",a[i]);
		}
	}
	else
	{
		printf("2 %d %d\n",a[1],a[2]);
		printf("%d",n-3);
		for(i=3;i<n;i++)
		printf(" %d",a[i]);
	}
	printf("\n");
}
