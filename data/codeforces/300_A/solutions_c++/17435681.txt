#include <iostream>
#include <algorithm>
using namespace std;

int main()
{

	long n,a[100],i,c;
	cin >> n;
	for (i=0;i<n;i++) {
		cin >> a[i];
	}
	sort(a,a+n);
	cout << 1 << " " << a[0] << endl;
	c=1;
	if (a[n-1]>0) cout << 1 << " " << a[n-1];
	else {
		c++;
		cout << 2 << " " << a[1] << " " << a[2];
		swap(a[1],a[n-1]);
		swap(a[2],a[n-2]);
	}
	cout << endl << n-c-1;
	for (i=1;i<n-c;i++) cout << " " << a[i];




    return 0;
}
