#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int main(){
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++) cin>>a[i];
    sort(a, a+n);
    cout<<"1 "<<a[0]<<"\n";
    if (a[n-1]==0){
         cout<<"2 "<<a[1]<<" "<<a[2]<<"\n"<<n-3;
         for(int i=3;i<n;i++) cout<<" "<<a[i];
    }
    else{
         cout<<"1 "<<a[n-1]<<"\n"<<n-2;
         for(int i=1;i<n-1;i++) cout<<" "<<a[i];
    }
}
