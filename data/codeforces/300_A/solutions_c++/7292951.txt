#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int n,a[110];
    while(cin>>n)
    {
        for(int i=0;i<n;i++)
        cin>>a[i];
        sort(a,a+n);
        int p=0;
        while(p<n&&a[p]<=0) p++;
        cout<<1<<' '<<a[0]<<endl;
        if(p==n) cout<<2<<' '<<a[1]<<' '<<a[2]<<'\n'<<n-3;
        else cout<<1<<' '<<a[p]<<'\n'<<n-2;
        for(int i=(p==n?3:1);i<n;i++)
        if(i!=p) cout<<' '<<a[i];
        cout<<endl;
    }
}
