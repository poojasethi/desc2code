import sys

num = int(sys.stdin.readline())
nums = sorted([int(x) for x in sys.stdin.readline().split()])
zindex = nums.index(0)

print 1, nums[0]
if zindex == num - 1:
    print 2, nums[1], nums[2]
    print num-3, ' '.join([str(x) for x in nums[3:]])     
else:
    print 1, nums[num-1]
    print num-2, ' '.join([str(x) for x in nums[1:-1]])
