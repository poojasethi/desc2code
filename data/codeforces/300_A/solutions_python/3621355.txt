n = input()
a = map(int, raw_input().split())

b = []
c = []
d = []

for i in a:
  if len(b) == 0 and i < 0:
    b.append(i)
  elif len(c) == 0 and i > 0:
    c.append(i)
  else:
    d.append(i)

if len(c) == 0:
  d.sort()
  c = d[0:2]
  d = d[2:]

print len(b), " ".join(map(str, b))
print len(c), " ".join(map(str, c))
print len(d), " ".join(map(str, d))
