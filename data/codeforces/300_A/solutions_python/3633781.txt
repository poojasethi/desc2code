#!/usr/bin/python

n = input()
a = map(int, raw_input().split())
f = filter(lambda x: x > 0, a)
s = filter(lambda x: x <= 0, a)
s.sort()
if f and s:
	print 1, s[0]
	print 1, f[0]
	print n-2, " ".join(map(str, f[1:])), " ".join(map(str, s[1:]))
else:
	print 1, s[2]
	print 2, s[0], s[1]
	print n-3, " ".join(map(str, s[3:]))
