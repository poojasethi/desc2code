#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <ctime>
#include <cmath>
#define ll long long
#define mk make_pair
#define lb lower_bound
#define maxn 100020
#define rep(i,n) for(int i=1;i<=n;++i)
using namespace std;

pair<ll, ll> a[maxn];
ll b[maxn];
int x[maxn],t[maxn];
int n,V,ans;
int main()
{
    scanf("%d",&n);
    rep(i,n)scanf("%d%d",&x[i],&t[i]);
    scanf("%d",&V);
    rep(i,n)a[i]=mk((ll)V*t[i]-x[i],(ll)V*t[i]+x[i]);
    ans=0;
    sort(a+1,a+1+n);
    
    rep(i,n)
    {
        if(a[i].first<0||a[i].second<0)continue;
        int p=lb(b,b+ans,a[i].second+1)-b;
        b[p]=a[i].second;
        if(p==ans)++ans;
    }
    printf("%d ",ans);
    ans=0;
    rep(i,n)
    {
    //  if(a[i].first<0||a[i].second<0)continue;
        int p=lb(b,b+ans,a[i].second+1)-b;
        b[p]=a[i].second;
        if(p==ans)++ans;
    }
    printf("%d\n",ans);
    return 0;
}
