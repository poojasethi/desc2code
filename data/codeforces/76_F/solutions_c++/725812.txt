#include <cstdio>
#include <vector>
#include <algorithm>

using namespace std;

#define MP make_pair
#define PB push_back
#define A first
#define B second

typedef long long LL;
typedef pair<LL, LL> PLL;

int main() {
	int n; scanf("%d", &n);

	vector<PLL> g;
	for (int i = 0; i < n; ++i) {
		int x, t; scanf("%d%d", &x, &t);
		g.PB(MP(x, t));
	}

	int v; scanf("%d", &v);
	for (int i = 0; i < n; ++i) g[i] = MP(g[i].A + g[i].B * v, - g[i].A + g[i].B * v);
	g.PB(MP(0, 0)); sort(g.begin(), g.end());

	int ans1 = 0, ans2 = 0;
	vector<LL> f;
	for (int i = n; i >= 0 ; --i) {
		LL x = -g[i].B;
		int pos = upper_bound(f.begin(), f.end(), x) - f.begin();

		if (pos == f.size()) f.PB(x);
		else f[pos] = x;

		if (g[i].A == g[i].B) ans1 = max(ans1, pos);
		else ans2 = max(ans2, pos + 1);
	}

	printf("%d %d\n", ans1, ans2);

	return 0;
}
