#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define fi first
#define se second
using namespace std;
typedef long long ll;
typedef pair<ll,ll> PLL;
const int maxn=100512,
    inf=1e9;
PLL a[maxn]; 
int n,v;
ll f[maxn];
int main()
{
    //freopen("AC.in","r",stdin);
    //freopen("AC.out","w",stdout);
    ios::sync_with_stdio(false);
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i].fi>>a[i].se;
	cin>>v;
	for(int i=0;i<n;i++)
		a[i]=PLL(a[i].fi+a[i].se*v,-a[i].fi+a[i].se*v);
	a[n++]=PLL(0,0);
	sort(a,a+n);
	memset(f,0x3f,sizeof f);
	for(int i=n;i--;)
	{
		int p=upper_bound(f,f+n,-a[i].se)-f;
		if(a[i]==PLL(0,0))cout<<lower_bound(f,f+n,0)-f<<' ';
		else  f[p]=-a[i].se;
	}
	cout<<lower_bound(f,f+n,inf)-f<<endl;
	return 0;
}
