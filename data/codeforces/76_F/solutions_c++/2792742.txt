#include<cstring>
#include<iostream>
#include<algorithm>
#define X first
#define Y second
#define N 100010
#define For(i,n) for(int i=1;i<=n;i++)
using namespace std;
typedef pair <int,int> PII;
int n,v,x[N],t[N];
PII a[N];
int main()
{
	cin>>n;
    For(i,n) cin>>x[i]>>t[i];
	cin>>v;
	For(i,n) a[i]=PII(x[i]+t[i]*v,-x[i]+t[i]*v);
	a[++n]=PII(0,0);
	sort(a+1,a+n+1);
	memset(t,0x7F,sizeof(t));t[0]=1<<31;
	for(int i=n;i;i--)
	{
		v=upper_bound(t,t+n,-a[i].Y)-t;
		if (a[i]==PII(0,0))
			cout<<v-1<<' ';
		else
			t[v]=-a[i].Y;
	}
	cout<<lower_bound(t,t+n,0x7F7F7F7F)-t-1<<endl;
	return 0;
}
