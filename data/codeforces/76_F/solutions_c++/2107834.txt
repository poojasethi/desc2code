#include <cstdio>
#include <algorithm>

using namespace std;

typedef pair<int,int> pnt;

pnt p[100005];
int dp[100005],m;

int main()
{
    int n,v;
    int x,t;
    int i;
    m = 0;
    scanf("%d",&n);
    for(i=0;i<n;i++) scanf("%d %d",&p[i].first,&p[i].second);
    scanf("%d",&v);
    for(i=0;i<n;i++) {
        x = p[i].first;
        t = p[i].second;
        p[i] = pnt(x+v*t, -(x-v*t));
    }
    p[n++] = pnt(0,0);
    sort(p,p+n);
    for(i=n-1;i>=0;i--) {
        x = upper_bound(dp,dp+m,-p[i].second) - dp;
        if(p[i]==pnt(0,0)) printf("%d\n",x);
        else {
            dp[x] = -p[i].second;
            if(x==m) m++;
            //printf("ADD %d at %d\n",dp[x],x);
        }
    }
    printf("%d\n",m);
    return 0;
}
