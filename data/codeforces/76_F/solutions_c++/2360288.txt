#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn = 100000+5;
const int inf = 1e10;
struct data{
	int p,q;
}g[maxn];
int q[maxn],x[maxn],t[maxn],v;
int n,r,ans;

bool cmp(data a,data b){
	return a.p<b.p||(a.p==b.p&&a.q<b.q);
}

inline int find(int l,int r,int x){
	int m,ret=0;
	while(l<=r){
		m=(l+r)>>1;
		if(q[m]>x)ret=m,r=m-1;
		else l=m+1;
	}
	return ret;
}

inline void init(){
	for(int i=0;i<n;++i){
		g[i].p=-x[i]+t[i]*v;
		g[i].q=x[i]+t[i]*v;
	}
	sort(g,g+n,cmp);
	q[0]=-inf,q[r=1]=inf;	
}

int main(){
	while(~scanf("%d",&n)){
		for(int i=0;i<n;++i)
			scanf("%d%d",&x[i],&t[i]);
		scanf("%d",&v);
		init();
		for(int i=0;i<n;++i){
			int j=find(0,r,g[i].q);
			if(q[j]>g[i].q)q[j]=g[i].q;
			if(j>=r)q[++r]=inf;
		}
		ans=r-1;
		q[r=1]=inf;
		for(int i=0;i<n;++i)
			if(g[i].p>=0 && g[i].q>=0){
				int j=find(0,r,g[i].q);
				if(q[j]>g[i].q)q[j]=g[i].q;
				if(j>=r)q[++r]=inf;
			}
			printf("%d %d\n",r-1,ans);
	}
	return 0;
}
