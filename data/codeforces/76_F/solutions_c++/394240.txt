#include<cstdio>
#include<algorithm>
using namespace std;
int n,v,f[100003];
//                                                                                                                            Last Change:  2011-04-16 21:33:59
pair <int,int> a[100003];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
		scanf("%d%d",&a[i].first,&a[i].second);
	scanf("%d",&v);
	for(int i=1;i<=n;++i)
	{
		pair <int,int> t=a[i];
		a[i].first=v*t.second-t.first;
		a[i].second=v*t.second+t.first;
	}
	++n,a[n].first=a[n].second=0;
	sort(a+1,a+n+1);
	int cnt=1,l,r;
	f[0]=2000000000;
	for(int i=n;i;--i)
	{
		for(l=0,r=cnt;l<r;)
			if(f[l+r>>1]>=a[i].second)l=(l+r>>1)+1;
			else r=l+r>>1;
		if(a[i].first||a[i].second)
		{
			f[l]=a[i].second;
			if(l==cnt)++cnt;
		}
		else printf("%d ",l-1);
	}
	printf("%d\n",cnt-1);
	return 0;
}
