#include<iostream>
#include<algorithm>
using namespace std;
struct rec{int x,t;}a[100010];
int c[100010],b,n,i,j,m,v;
bool operator <(rec a,rec b)
{
	return a.t*v-a.x<b.t*v-b.x||a.t*v-a.x==b.t*v-b.x&&a.t*v+a.x<b.t*v+b.x;
}
int main()
{
	ios::sync_with_stdio(false);
	cin>>n;
	for(i=1;i<=n;i++) cin>>a[i].x>>a[i].t;
	a[++n]=(rec){0,0};
	cin>>v;
	sort(a+1,a+n+1);
	for(i=n;i;i--)
	{
		b=-a[i].t*v-a[i].x;
		j=upper_bound(c+1,c+m+1,b)-c;
		if(!a[i].t&&!a[i].x) cout<<j-1<<' '; else c[j]=b,m=max(m,j);
	}
	cout<<m<<endl;
}
