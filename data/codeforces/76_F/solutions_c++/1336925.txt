#include <vector>
#include <algorithm>
#include <iostream>
 
using namespace std;
 
typedef pair <int,int> pii ;
const int maxn = 100 * 1000 ;

int x[maxn] , t[maxn] ;
int V ;
pii a[maxn] ;
int N ;

int LIS (bool flag) {
    vector <int> v ;
    for (int i = 0 ; i < N ; ++i) {
        if (flag || (0 <= a[i].first && 0 <= a[i].second) ) {
            int p = upper_bound (v.begin(),v.end(),a[i].second)-v.begin() ;
            if (p == (int)v.size())
                v.push_back (a[i].second) ;
            else
                v[p] = a[i].second ;
        }
    }
    return v.size() ;
}

int main() {
    ios_base::sync_with_stdio(false) ;
    cin >> N ;
    for (int i = 0 ; i < N ; ++i)
        cin >> x[i] >> t[i] ;
    cin >> V ;
    for (int i = 0 ; i < N ; ++i) {
        a[i] = pii(V*t[i]-x[i] , x[i]+V*t[i]) ;
    }
    sort (a,a+N) ;
    cout << LIS(0) << " " << LIS(1) << endl ;
    return 0 ;
}
