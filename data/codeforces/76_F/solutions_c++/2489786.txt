#include<cstdio>
#include<utility>
#include<algorithm>
#define fi first
#define se second
#define mp make_pair
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=101000;
int n,m,v,ans1,ans2;
int x[mn],t[mn],q[mn];
pair<int,int>a[mn];
int calc()
{
    int i,j,L=0;
    q[0]=-2147483647;
    fo(i,1,n)
    {
        j=upper_bound(q,q+1+L,a[i].se)-q-1;
        if(j==L)q[++L]=a[i].se;
        else q[j+1]=min(q[j+1],a[i].se);
    }
    return L;
}
int main()
{
    int i,m;
    scanf("%d",&n);
    fo(i,1,n)scanf("%d%d",x+i,t+i);
    scanf("%d",&v);
    fo(i,1,n)a[i]=mp(-x[i]+t[i]*v,x[i]+t[i]*v);
    sort(a+1,a+1+n);
    ans2=calc();
    ++n,a[n]=mp(0,0);
    sort(a+1,a+1+n);
    m=n,n=0;
    fo(i,1,m)
        if(a[i].fi>=0&&a[i].se>=0)
            a[++n]=a[i];
    ans1=calc()-1;
    printf("%d %d\n",ans1,ans2);
    return 0;
}
