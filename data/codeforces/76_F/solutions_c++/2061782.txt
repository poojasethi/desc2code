#include<cstdio>
#include<algorithm>

using namespace std;

#define long long long

const int MAX_N = 100005;

struct Event{
  long x, y;
  bool operator < (const Event &o) const{
    return x != o.x? x > o.x: y < o.y;
  }
};

int n, v;
Event a[MAX_N];
long b[MAX_N];
int nb;

int main(){
  scanf("%d", &n);
  for(int i = 0; i < n; i++){
    scanf("%I64d %I64d", &a[i].x, &a[i].y);
  }
  scanf("%d", &v);
  n++;
  for(int i = 0; i < n; i++){
    Event t = a[i];
    a[i].x = t.x + v * t.y;
    a[i].y = t.x - v * t.y;
  }
  sort(a, a + n);
  for(int i = 0; i < n; i++){
    int it = upper_bound(b, b + nb, a[i].y) - b;
    if(a[i].x == 0 and a[i].y == 0){
      printf("%d ", it);
    }
    else{
      b[it] = a[i].y;
      if(it == nb){
	nb++;
      }
    }
  }
  printf("%d\n", nb);
}
