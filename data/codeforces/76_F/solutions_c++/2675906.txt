#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<time.h>
#include<math.h>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
#define N 100005
struct rec{int x,t;}a[N];
int c[N],b,n,i,j,m,v;
bool operator <(const rec &a,const rec &b){
	return a.t*v-a.x<b.t*v-b.x||a.t*v-a.x==b.t*v-b.x&&a.t*v+a.x<b.t*v+b.x;
}
int main()
{
	scanf("%d",&n);
	for(i=1;i<=n;i++) scanf("%d%d",&a[i].x,&a[i].t);
	a[++n]=(rec){0,0};
	cin>>v;
	sort(a+1,a+n+1);
	for(i=n;i;i--){
		b=-a[i].t*v-a[i].x;
		j=upper_bound(c+1,c+m+1,b)-c;
		if(!a[i].t&&!a[i].x) cout<<j-1<<' '; else c[j]=b,m=max(m,j);
	}
	cout<<m<<endl;
	//system("pause");
	return 0;
}
