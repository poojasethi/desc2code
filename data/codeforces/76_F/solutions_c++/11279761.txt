#include <cstring>
#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;

#define int long long
const int maxn=100010;
int t[maxn],x[maxn],id[maxn],a[maxn];
int v,n,len;

bool cmp(int i, int j)
{
	return v*t[i]-x[i]<v*t[j]-x[j] || v*t[i]-x[i]==v*t[j]-x[j] && v*t[i]+x[i]<v*t[j]+x[j];
}
#undef int

int main()
{
	#define int long long
	cin>>n;
	for (int i=1; i<=n; i++) cin>>x[i]>>t[i];
	cin>>v; x[++n]=0, t[n]=0;
	for (int i=1; i<=n; i++) id[i]=i; 
	
	sort(id+1,id+n+1,cmp);
	for (int p=n; p; p--)
	{
		int i=id[p], xx=-v*t[i]-x[i], pos;
		if (xx>=a[len] || len==0)
		{
			if (t[i]==0 && x[i]==0) cout<<len<<" "<<endl;
			else a[++len]=xx;
		}
		else
		{
			int l=0, r=len, mid;
			while (l+1<r)
			{
				mid=(l+r)>>1;
				if (xx>=a[mid]) l=mid; else r=mid;
			}
			if (t[i]==0 && x[i]==0) cout<<r-1<<" "<<endl;
			else a[r]=xx;
		}
		
	}
	cout<<len<<endl;
	//for (;;);
}
