#include<cstdio>
#include<algorithm>
using namespace std;

const int MAX_N = 1e5 + 5;

#define long long long

struct Event{
  long x, y;
  bool operator < (const Event &o) const{
    return x != o.x? x < o.x: y < o.y;
  }
};

int n;
long v;
Event a[MAX_N];
int b[MAX_N], k;

int main(){
  
  scanf("%d", &n);
  for(int i = 0; i < n; i++){
    scanf("%I64d %I64d", &a[i].x, &a[i].y);
  }
  scanf("%I64d", &v);

  a[n].x = 0;
  a[n].y = 0;
  n++;
  for(int i = 0; i < n; i++){
    long x = a[i].x, t = a[i].y;
    a[i].x = x - v * t;
    a[i].y = -x - v * t; 
  }
  sort(a, a + n);

  for(int i = 0; i < n; i++){
    int x = upper_bound(b + 1, b + k + 1, a[i].y) - b;
    if(a[i].x == 0 and a[i].y == 0){
      printf("%d ", x - 1);
    }
    else{
      b[x] = a[i].y;
      k = max(k, x);
    }
  }
  
  printf("%d\n", k);  
}
