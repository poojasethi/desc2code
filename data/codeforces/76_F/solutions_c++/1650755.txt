#include<cstdio>
#include<algorithm>
#include<iostream>
using namespace std;
#define x first
#define y second
typedef pair<int,int> per;
const int N = 100005;
per a[N];
int b[N];
bool cmp(per a,per b){if(a.x==b.x)return a.y>b.y;return a.x<b.x;}
int main()
{
    int n,i,v;
    scanf("%d",&n);
    for(i=0;i<n;i++)scanf("%d%d",&a[i].x,&a[i].y);
    scanf("%d",&v);
    for(i=0;i<n;i++)
    {
        per t=a[i];
        a[i].x=t.x+v*t.y;
        a[i].y=t.x-v*t.y;
    }
    a[n++]=make_pair(0,0);
    sort(a,a+n,cmp);
    int k=0,r=0;
    for(i=n-1;i>=0;i--)
    {
        int p=upper_bound(b,b+k,a[i].y)-b;
        if(!a[i].x&&!a[i].y)printf("%d ",p);
        else {b[p]=a[i].y;if(p==k)k++;} 
    }
    printf("%d\n",k);
    return 0;
}
