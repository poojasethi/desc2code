#include <cstdio>
#include <cstring>
#include <algorithm>
#include <climits>
#include <iostream>
using namespace std;
const int maxn = 100010;
typedef pair<int,int> ii;
ii num[maxn];
int lis[maxn];
int n;
int solve(bool flag)
{
	int len = 0;
	lis[len++] = INT_MIN;
	for(int i=0;i<n;i++)
	{
		if(flag || num[i].first >=0 && num[i].second >= 0)
		{
			int idx = upper_bound(lis,lis+len,num[i].second)-lis;
			if(idx == len)
			{
				lis[len++] = num[i].second;
			}
			else 
			{
				lis[idx] = num[i].second;
			}
		}
	}
	return len - 1;
}
int main()
{
	int v;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		scanf("%d%d",&num[i].first,&num[i].second);
	}
	cin>>v;
	for(int i=0;i<n;i++)
	{
		num[i] = ii(-num[i].first+num[i].second*v,num[i].first+num[i].second*v);
	}
	sort(num,num+n);
	int ans1 = solve(false);
	int ans2 = solve(true);
	printf("%d %d\n",ans1,ans2);
	return 0;
}
