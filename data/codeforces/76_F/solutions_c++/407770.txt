#include <cstdio>
#include <algorithm>
#include <utility>
using namespace std;
const int maxn = 100002;
int f[maxn];
pair<int , int> a[maxn];
class cmp
{
    public :
    bool operator () (const int &a , const int &b)
    {
        return a > b;
    }
};
int main()
{
    int i , j , x , t , v , n;
    scanf("%d" , &n);
    for (i = 1 ; i <= n ; ++i)
        scanf("%d%d" , &a[i].first , &a[i].second);
    scanf("%d" , &v);
    for (i = 1 ; i <= n ; ++i)
    {
        x = a[i].first , t = a[i].second;
        a[i] = make_pair(t * v  - x , t * v + x);
    }
    a[++n] = make_pair(0 , 0);
    sort(a + 1 , a + n + 1);
    t = 1 , f[0] = 0x7fffffff;
    for (i = n ; i >= 1 ; --i)
    {
        j = upper_bound(f , f + t , a[i].second , cmp()) - f;
        if (a[i] != make_pair(0 , 0))
            f[j] = a[i].second , t += int(j == t);
        else  x = j;
    }
    printf("%d %d\n" , x - 1 , t - 1);
    return 0;
}
