#include <algorithm>
#include <cstring>
#include <cstdio>

using namespace std;

pair<long long, long long> e[100000];
int x[100000], t[100000];
long long l[100000];

int main() {
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i ++)
		scanf("%d%d", &x[i], &t[i]);
	int V;
	scanf("%d", &V);
	
	for (int i = 0; i < n; i ++)
		e[i] = make_pair((long long)V * t[i] + x[i], (long long)V * t[i] - x[i]);
	sort(e, e + n);
	
	int tot = 0;
	for (int i = 0; i < n; i ++) {
		if (e[i].first < 0 || e[i].second < 0) continue;
		int loc = lower_bound(l, l + tot, e[i].second + 1) - l;
		l[loc] = e[i].second;
		if (loc == tot) tot ++;
	}
	printf("%d ", tot);
	
	tot = 0;
	for (int i = 0; i < n; i ++) {
		int loc = lower_bound(l, l + tot, e[i].second + 1) - l;
		l[loc] = e[i].second;
		if (loc == tot) tot ++;
	}
	printf("%d\n", tot);
	
	return 0;
}
