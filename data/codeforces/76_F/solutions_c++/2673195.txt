#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
struct eve{
    ll d,t;
} a[100000];
ll t[100000];
bool cmp(eve a,eve b){
    return a.d-a.t>b.d-b.t || a.d-a.t==b.d-b.t && a.d+a.t<b.d+b.t;
}
int main(){
    int n,i,ans;
    ll v,*now;
    scanf("%d",&n);
    for (i=0;i<n;i++)
        scanf("%I64d%I64d",&a[i].d,&a[i].t);
    scanf("%I64d",&v);
    for (i=0;i<n;i++) a[i].t*=v;
    sort(a,a+n,cmp);
    for (ans=0,i=0;i<n;i++) if (a[i].d<=a[i].t && -a[i].d<=a[i].t){
        *(now=upper_bound(t,t+ans,a[i].d+a[i].t))=a[i].d+a[i].t;
        if (now==t+ans) ans++;
    }
    printf("%d ",ans);
    for (ans=0,i=0;i<n;i++){
        *(now=upper_bound(t,t+ans,a[i].d+a[i].t))=a[i].d+a[i].t;
        if (now==t+ans) ans++;
    }
    printf("%d\n",ans);
    return 0;
}
