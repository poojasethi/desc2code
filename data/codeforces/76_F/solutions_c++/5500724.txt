#include <cstdio>
#include <algorithm>

using namespace std;
struct INC{
  int x,y;
  bool operator <(const INC &a) const{
    return (x<a.x) || (x==a.x && y<a.y);
  }
}A[100100];
int f[200100],n,i,j,Vel,maxx,K;

int main(){
  scanf("%d",&n);
  for (i=1;i<=n;i++) scanf("%d%d",&A[i].x,&A[i].y);
  scanf("%d",&Vel);
  
  n++;
  for (i=1;i<=n;i++){
    A[i].y=A[i].y*Vel-A[i].x;
    A[i].x=A[i].x*2+A[i].y;
  }
  sort(A+1,A+n+1);
  
  for (i=n;i>0;i--){
    K=upper_bound(f+1,f+maxx+1,-A[i].y)-f;
    if (A[i].x==0 && A[i].y==0)
      { printf("%d ",K-1);continue; }
    f[K]=-A[i].y;
    if (K>maxx) maxx=K;
  }
  printf("%d\n",maxx);
  return 0;
}
