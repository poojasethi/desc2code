#include <cstdio>
#include <cstring>
#include <algorithm>
#include <climits>
using namespace std;

const int N = 100000 + 10;
typedef pair<int,int> ii;

int x[N], t[N];
ii a[N];
int last[N];
int n;

int solve(bool flag)
{
	int len = 0;
	last[len++] = INT_MIN;
	for (int i = 0; i < n; i++)
		if (flag || (a[i].first >= 0 && a[i].second >= 0))
		{
			int idx = upper_bound(last, last+len, a[i].second) - last;
			if (idx == len)
				last[len++] = a[i].second;
			else
				last[idx] = a[i].second;
		}
	return len-1;
}

int main()
{
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		scanf("%d%d", &x[i], &t[i]);
	int v;
	scanf("%d", &v);
	for (int i = 0; i < n; i++)
		a[i] = ii(-x[i]+t[i]*v, x[i]+t[i]*v);
	sort(a, a+n);
	int ans1 = solve(false);
	int ans2 = solve(true);
	printf("%d %d\n", ans1, ans2);
}
