#include <cstdio>
#include <iostream>
#include <algorithm>
#define N 100010
#define INF 2147483647
using namespace std;
struct stu{int a,b;}e[N];
int n,x[N],t[N],v,Min[N];
bool cmp(stu x,stu y){
	if(x.a==y.a)return x.b<y.b;
	return x.a<y.a;
}
int find(int key){
	int res,l=0,r=n;
	while(l<=r){
		int mid=(l+r)>>1;
		if(Min[mid]<=key)res=mid,l=mid+1;
		else r=mid-1;
	}
	return res;
}
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;++i)scanf("%d%d",&x[i],&t[i]);
	scanf("%d",&v);
	for(int i=1;i<=n;++i){
		e[i].a=t[i]*v-x[i];
		e[i].b=t[i]*v+x[i];
	}
	sort(e+1,e+n+1,cmp);
	int ans=0;
	Min[0]=-INF;
	for(int i=1;i<=n;++i)Min[i]=INF;
	for(int i=1;i<=n;++i)if(e[i].a>=0 && e[i].b>=0){
		int key=find(e[i].b)+1;
		Min[key]=min(Min[key],e[i].b);
		ans=max(ans,key);
    }
    printf("%d ",ans);
    ans=0;
    Min[0]=-INF;
    for(int i=1;i<=n;++i)Min[i]=INF;
    for(int i=1;i<=n;++i){
    	int key=find(e[i].b)+1;
		Min[key]=min(Min[key],e[i].b);
		ans=max(ans,key);
    }
    printf("%d",ans);
}
