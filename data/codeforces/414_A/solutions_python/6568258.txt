def solve(n, k):
    if n == 1:
        if k == 0:
            return "1"
        return "-1"
    m = n/2
    if m > k:
        return "-1"
    t = k-m+1
    res = range(3*t-1, 3*t-1+n)
    res[0:2] = [2*t, 3*t]
    return str(res)[1:-1].replace(',', '')

n, k = map(int, raw_input().split())
print solve(n, k)
