n, k = map(int, raw_input().split())
p = n/2 - 1
t = k - p

res = ""
if (t <= 0):
	print "-1"
elif (n < 2):
	if (k > 0):	
		print "-1"
	else:
		print "1"
else:
	res += str( 2 * t) + " " + str( 3 * t) + " "
	for i in xrange(2 * t + 1, 2 * t + p + 1):
		res += str( 2 * i) + " " + str( 2 * i + 1) + " "
	
	if (n % 2 ==1):
		res += "1"
	print res
