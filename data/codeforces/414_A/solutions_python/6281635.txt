n, k = map(int, raw_input().split())

if n == 1 and k > 0:
  print -1
elif k == 0:
  if n == 1:
    print 1
  else:
    print -1
else:
  kk = k-((n/2)-1)
  if kk < 1:
    print -1
    exit()
  l = [kk, kk*2]
  n -= 2
  i = 1
  while n > 1:
    if i == kk or i == kk*2 or i+1 == kk or i+1 == kk*2:
      i += 1
    else:
      l.append(i)
      l.append(i+1)
      i += 2
      n -= 2
  if n == 1:
    l.append(10**9)
  print ' '.join(map(str, l))
