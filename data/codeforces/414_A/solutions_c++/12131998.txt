#include <bits/stdc++.h>
using namespace std;

int main(){
  int n, k;
  cin >> n >> k;
  if (n == 1 && k == 0){
    puts("1");
    return 0;
  }
  if (n / 2 > k || n == 1){
    puts("-1");
    return 0;
  }
  k -= (n - 2) / 2;
  cout << k << " " << 2 * k << " ";
  for (int i = 0; i < n - 2; i++)
    cout << 2 * k + i + 1 << " ";
  puts("");
}
