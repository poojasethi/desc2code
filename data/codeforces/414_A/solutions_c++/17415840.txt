#include <cstdio>
using namespace std;
int main()
{
    int n,k ;
    scanf("%d %d",&n,&k) ;
    if(k<n/2 || (n==1 && k!=0))
	printf("%d",-1) ;
    else if(n==1 && k==0)
	printf("1") ;
    else
    {
	int u=n/2,v=k-u+1 ;
	printf("%d %d",v,v*2) ;
	for(int i=1;i<u;i++)
	    printf(" %d %d",(i+v)*2+1,(i+v)*2+2) ;
	if(n%2)
	    printf(" %d",(int)1e9) ;
    }
}
