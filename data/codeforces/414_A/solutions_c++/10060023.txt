#include<bits/stdc++.h>
using namespace std;
int main()
{
  cin.tie(NULL);
  ios_base::sync_with_stdio(0);

  int n,k; cin>>n>>k;
  if((n/2)>k || (n==1&&k!=0)) cout<<"-1"<<"\n";
  else if(n==1 &&k==0) cout<<"1";
  else{
    int z=k-(n/2)+1;
    cout<<z<<" "<<z*2<<" ";
    for(int i=1;i<=(n-2);i++){
      cout<<i+z*2<<" ";
    }
  }
  return 0;
}
