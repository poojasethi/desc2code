#include <iostream>
using namespace std;

int n,k;

int main(){
	cin >> n >> k;
	
	if(n == 1){
		if(k == 0) cout << 1;
		else cout << -1;
		return 0;
	}
	
	int val = k - n/2 + 1;
	if(val <= 0) {cout << -1; return 0;}
	cout << val << ' ' << val*2;
	val *= 2;
	for(int i=2; i<n; i++){
		cout << ' ' << ++val;
	}
	return 0;
}
