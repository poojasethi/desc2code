#include <iostream>
using namespace std;
int n,k;
int main(){
	cin >> n >> k;
	if(n==1&&k==0){
		cout << "1\n";
		return 0;
	}
	if(n==1&&k!=0){
		cout << "-1\n";
		return 0;
	}
	
	int t = n/2-1;
	int r = k-t;
	if(r<1){
		cout << "-1\n";
		return 0;
	}
	cout << r << ' ' << r*2;
	for(int i = 2; i < n; ++i)
		cout << ' ' << 2*r+i-1;
	cout << '\n';
}
