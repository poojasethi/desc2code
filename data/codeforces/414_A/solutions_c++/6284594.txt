#include <cstdio>
int main() {
	int n, k, t;
	while( ~scanf("%d %d", &n, &k) ) {
		if( n==1 ) {
			if( k==0 ) puts("1");
			else puts("-1");
			continue;
		}
		t = k - (n-2)/2;
		if( t<=0 ) {
			puts("-1");
			continue;
		}
		printf("%d %d", t, t*2);
		for(int i=n-2; i>0; --i) printf(" %d", t*2+i);
		putchar('\n');
	}
}
