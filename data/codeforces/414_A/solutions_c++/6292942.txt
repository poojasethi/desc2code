#include <cstdio>
int main()
{
	int n,k,nn;
	scanf("%d%d",&n,&k);
	nn = n / 2;
	if(n == 1){
		if(k) printf("-1");
		else printf("1");
		return 0;
	}
	if(k < nn){
		printf("-1");
	}
	else{
		k -= nn - 1;
		n -= 2;
		printf("%d %d ",k,k * 2);
		for(int i = 1e9; i >= 1e9 - n + 1; --i)
			printf("%d%c",i,i == 1e9 - n + 1 ? '\n' : ' ');
	}
	return 0;
}
