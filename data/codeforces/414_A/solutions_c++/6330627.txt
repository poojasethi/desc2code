#include <stdio.h>
#include <iostream>
using namespace std;
int n,m,v;
int main( ){
	cin>>n>>m;
	if (n==1 && m==0){
		cout<<1<<endl;
		return 0;
	}
	if (m<n/2 || n==1){
		cout<<-1<<endl;
		return 0;
	}
	v=m-n/2+1;
	cout<<v<<' '<<v*2;
	for (int i=1,j=1;j<=n-2;i++)
		if (i!=v && i!=v*2){
			cout<<' '<<i;
			j++;
		}
	cout<<endl;
	return 0;
}
