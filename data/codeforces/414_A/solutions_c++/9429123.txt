#include <iostream>

using namespace std;
int n,k,s;

int main()
{
    cin>>n>>k;
    if(k<n/2||(n==1&&k))
        cout<<-1<<endl;
    else if(n==1)
        cout<<2<<endl;
    else
    {
        s=(k+1-(n/2));
        cout<<s<<" "<<2*s;
        for(int i=2;i<n;i++)
            cout<<" "<<2*s+i-1;
        cout<<" ";
    }
    return 0;
}
