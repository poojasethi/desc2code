#include<iostream>
using namespace std;
int main()
{
	int n,k,i;
	cin>>n>>k;
	if(n==1)
	{
		if(k==0)cout<<1<<endl;
		else cout<<-1<<endl;
		return 0;
	}
	if(n/2>k)
	{
		cout<<-1<<endl;
		return 0;
	}
	cout<<k-(n/2)+1<<" "<<(k-(n/2)+1)*2;
	k=(k-(n/2)+1)*2;
	for(i=2;i<n;i++)
		cout<<" "<<k+i;
	cout<<endl;
}
