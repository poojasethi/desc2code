#include<iostream>

using namespace std;

int main()
{
	int n, k;
	cin >> n >> k;
	int m = n/2;
	if(m > k || (m == 0 && k != 0))
	{
		cout << -1 << endl;
		return 0;
	}
	if(m == 0 && k == 0)
	{
		cout << 1 << endl;
		return 0;
	}
	int f = k-m+1;
	int s = 2*f;
	cout << f << " ";
	cout << s << " ";
	for(int i = 0; i < n-2; i++)
		cout << i+s+1 << " ";
	cout << endl;
}
