#include<cstdio>

int n, m, x, y;

int main() {
	scanf("%d%d", &n, &m);
	if(n/2>m){
		printf("-1\n");
		return 0;
	}
	if(n==1){
		if(m==0)
			printf("1\n");
		else printf("-1\n");
		return 0;
	}
	x=m-((n/2)-1);
	y=(111111111/x)*x;
	printf("%d %d", y, y+x);
	for(int i=1;i<=n-2;i++)
		printf(" %d", i);
	printf("\n");

	return 0;
}
