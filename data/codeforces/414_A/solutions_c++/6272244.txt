#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
int n,k,x,i,cnt;
int main(){
	scanf("%d%d",&n,&k);
	if(n==1){
		if(k==0)printf("1\n");
		else puts("-1");
		return 0;
	}
	x=k-(n/2-1);
	if(x<=0)return puts("-1"),0;
	printf("%d %d ",x,x*2);
	cnt=1e9;
	for(i=3;i<=n;++i)printf("%d ",cnt),cnt--;
}
