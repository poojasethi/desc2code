#include<bits/stdc++.h>
using namespace std;

int main(){
	int N,K;cin>>N>>K;
	if(K<(N/2)){
		cout<<-1<<endl;return 0;
	}
	if(N==1){
		if(K==0) cout<<1<<endl;
		else cout<<-1<<endl;
		return 0;
	}
	cout<<(K-(N-2)/2)<<" "<<2*(K-(N-2)/2)<<" ";
	for(int i=0;i<(N-2);i++) cout<<2*(K-(N-2)/2) + i+1<<" ";
	cout<<endl;
	return 0;
}
