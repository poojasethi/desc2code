#include <iostream>
#include <cstdio>

using namespace std;

typedef long long ll;

const int mod=1e9+7;
const int N=2000;

ll dp[N+1][N+1],c[N+1][N+1],fac[N+1];

int main()
{
	for(int i=0;i<=N;++i)
	{
		c[i][0]=c[i][i]=1;
		for(int j=1;j<i;++j)
			c[i][j]=(c[i-1][j]+c[i-1][j-1])%mod;
	}
	fac[0]=1;
	for(int i=1;i<=N;++i) fac[i]=fac[i-1]*i%mod;
	dp[0][0]=1;
	for(int i=1;i<=N;++i)
		for(int j=1;j<=i;++j) dp[i][j]=(dp[i-j][j]+dp[i-j][j-1])%mod;
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
		int result=0;
		for(int s=(k+1)*k/2;s<=n;++s)
		{
			result=(result+dp[s][k]*c[n-s+k][k])%mod;
		}
		printf("%d\n",result*fac[k]%mod);
	}
	return 0;
}
