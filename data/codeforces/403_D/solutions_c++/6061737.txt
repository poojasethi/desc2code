#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
using namespace std;
const int mo=1e9+7;
int f[47][1010],i,j,C[1100][1100],k,ans[1100][1100],T,n,x;
void plus(int &x,int y){
	x+=y;
	if(x>=mo)x-=mo;
}
int main(){
	for(i=0;i<=1000;++i)
		for(j=0;j<=i;++j)C[i][j]=!j?1:(C[i-1][j-1]+C[i-1][j])%mo;
	f[0][0]=1;
	for(i=0;i<=1000;++i)
		for(j=45;j>=0;--j)
			for(k=1000-i;k>=0;--k)if(f[j][k])plus(f[j+1][k+i],f[j][k]);
	for(i=1;i<=45;++i)
		for(j=0;j<=1000;++j)if(f[i][j])
			for(k=j;k<=1000;++k)plus(ans[i][k],1ll*f[i][j]*C[k-j-1][i-1]%mo);
	x=1;
	for(i=1;i<=45;++i){
		x=1ll*x*i%mo;
		for(j=1;j<=1000;++j)plus(ans[i][j],ans[i][j-1]);
		for(j=1;j<=1000;++j)ans[i][j]=1ll*ans[i][j]*x%mo;
	}
	for(scanf("%d",&T);T--;){
		scanf("%d%d",&n,&k);
		printf("%d\n",ans[k][n]);
	}
}
