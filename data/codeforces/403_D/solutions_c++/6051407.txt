#include <iostream>
using namespace std;
#define rep(i,n) for(int _n=n,i=0;i<_n;++i)
#define forn(i,a,b) for(int _b=(b),i=(a);i<=_b;++i)
typedef long long int64;

const int64 mod = 1000000007; // 1e+9 + 7
const int maxn = 2001;

int mem[maxn][maxn];
int f(int s, int k) {
  if (s < 0) return 0;
  if (k == 1) return s + 1;
  int& res = mem[s][k];
  if (res != -1) return res;
  if (k & 1) return res = (f(s - 1, k) + f(s, k - 1)) % mod;
  return res = (f(s - k / 2, k) + f(s, k - 1)) % mod;
}

int64 fact[maxn];
void init_fact(int k) {
  fact[0] = 1;
  forn(i, 1, k)
    fact[i] = (fact[i - 1] * i) % mod;
}

int main() {
  ios::sync_with_stdio(false);
  init_fact(maxn);
  int n, k, t;
  rep(i, maxn) rep(j, maxn) mem[i][j] = -1;
  cin >> t;
  while (t--) {
    cin >> n >> k;
    cout << (fact[k] * f(n - k - ((k * k - k) / 2), 2 * k)) % mod << endl;
  }
  return 0;
}
