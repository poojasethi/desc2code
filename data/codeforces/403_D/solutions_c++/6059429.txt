#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;

const long long mo = 1000000007;
int t, n, k;
long long f[2100][1100], c[1100][1100], g[1100];

int main()
{
	for (int i = 0; i <= 1000; i++) {
		c[i][0] = 1;
		for (int j = 1; j <= i; j++) 
			c[i][j] = (c[i-1][j] + c[i-1][j-1]) % mo;
	}
	f[0][0] = 1;
	for (int i = 1; i <= 2000; i++) {
		for (int j = 1; j <= min(i, 1000); j++) {
			f[i][j] = (f[i - j][j - 1] + f[i - j][j]) % mo;
		}
	}
	g[1] = 1;
	for (int i = 2; i <= 1000; i++) g[i] = (g[i - 1] * i) % mo;
	scanf("%d", &t);
	int w;
	long long ans, tmp;
	while (t--) {
		scanf("%d%d", &n, &k);
		ans = 0;
		for (int s = k * (k - 1) / 2; s + k <= n; s++) {
			tmp = (f[s + k][k] * g[k]) % mo;
			tmp = (tmp * c[n - s][k]) % mo;
			ans = (ans + tmp) % mo;
		}
		w = ans; printf("%d\n", w);
	}
    return 0;
}
