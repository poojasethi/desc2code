#include <cstdio>
#define mo 1000000007
int t,n,k;
long long C[1005][1005],A[1005][1005],fac[1005];
int main(){
    A[1][1]=1;
    for (int i=2;i<=1000;i++)
        for (int j=1;j<i;j++)
        {
            A[i][j]=(A[i-j][j-1]+A[i-j][j])%mo;
        }
    for (int i=1;i<=1000;i++)
    {
        C[i][0]=C[i][i]=1;
        for (int j=1;j<i;j++)
            C[i][j]=(C[i-1][j-1]+C[i-1][j])%mo;
    }
    fac[1]=1;
    for (int i=2;i<=1000;i++)
        fac[i]=(fac[i-1]*i)%mo;
    scanf("%d",&t);
    while (t--)
    {
        scanf("%d%d",&n,&k);
        long long ans=0;
        for (int s=(k*(k-1))/2;s+k<=n;s++)
        ans=(ans+((fac[k]*A[s+k][k])%mo*C[n-s][k])%mo)%mo;
        if (ans<0)
            ans+=mo;
        printf("%I64d\n",ans);
    }
}
