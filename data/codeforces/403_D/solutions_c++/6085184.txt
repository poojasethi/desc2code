#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define rep(i,n) for (int i=0;i<n;++i)
typedef long long ll;
const int P=1000000007;
int T,n=1001,m=50;
int f[55][1005],c[1005][55];
ll ans;
int main()
{
	f[0][0]=1;
	for (int k=0;k<=n;++k) for (int i=m;i>=0;--i) rep(j,n)
		if (j+k<=n) f[i+1][j+k]=(f[i+1][j+k]+f[i][j])%P;
	rep(i,n) rep(j,min(i,m)+1) c[i][j]=i&&j?(c[i-1][j]+c[i-1][j-1])%P:1;
	scanf("%d",&T);
	while (T--){
		scanf("%d%d",&n,&m),m=min(m,50),ans=0;
		rep(j,n+1) ans=(ans+1LL*f[m][j]*c[n-j][m])%P;
		rep(i,m) ans=ans*(i+1)%P;
		printf("%I64d\n",ans);
	}
	return 0;
}
