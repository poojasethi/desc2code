#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
const int mm=1000000007,lim=1050;
int i,j,k,n,m,c[1101][1101],tests,f[1101][1101][2],g[1101][1101],ans,fac[1101];
int main(){
	memset(f,0,sizeof(f));f[0][0][0]=1;
	for (i=1;i<=lim;i++)
	 for (j=(i-1)*i/2;j<=lim;j++) f[i][j][0]=(f[i][j-i][0]+f[i][j-i][1])%mm,f[i][j][1]=f[i-1][j][0],g[i][j]=(f[i][j][0]+f[i][j][1])%mm;
	memset(c,0,sizeof(c));
	for (c[1][1]=c[1][0]=1,i=2;i<=lim;i++)
	 for (c[i][0]=1,j=1;j<=i;j++) c[i][j]=(c[i-1][j-1]+c[i-1][j])%mm;
	for (i=1,fac[0]=1;i<=lim;i++) fac[i]=(ll)fac[i-1]*(ll)i%mm;
	for (scanf("%d",&tests);tests--;){
		scanf("%d%d",&n,&m);
		for (i=m*(m-1)/2,ans=0;i<=n-m;i++){
			ans+=(ll)g[m][i]*(ll)fac[m]%mm*(ll)c[n-i][m]%mm;
			ans%=mm;
		}
		printf("%d\n",ans);
	}
	return 0;
}
