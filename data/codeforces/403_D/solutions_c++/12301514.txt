#include<cstdio>
#include<cstring>
#define LL long long
#define N 1001
#define M 2001
#define MOD 1000000007
LL a[N],c[M][M];
LL f[N][N];
LL g[N][N];
void init(){
	int i,j,k;
	a[0]=1;
	for(i=1;i<N;++i) a[i]=a[i-1]*i%MOD;
	for(i=0;i<M;++i)
	for(j=0;j<=i;++j)
	if(j==0 || j==i) c[i][j]=1;
	else c[i][j]=(c[i-1][j-1]+c[i-1][j])%MOD;
	f[0][0]=1;
	for(k=1;k<N;++k){
		for(i=k;i>=1;--i)
		for(j=k;j<N;++j){
			f[i][j]=(f[i][j]+f[i-1][j-k])%MOD;
		}
	}
	memset(g,-1,sizeof(g));
}
int main(){
	int T;
	int n,k,i;
	init();
	scanf("%d",&T);
	while(T--){
		scanf("%d%d",&n,&k);
		if(g[n][k]==-1){
			g[n][k]=0;
			for(i=k*(k+1)>>1;i<=n;++i){
				g[n][k]+=f[k][i]*c[n-i+k][k]%MOD*a[k]%MOD;
				g[n][k]%=MOD;
			}
		}
		printf("%I64d\n",g[n][k]);
	}
	return 0;
}
