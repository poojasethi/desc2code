#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdlib>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long LL;
const  LL MOD=1000000007;
LL dp[1001][1001];
LL fat[1001];
int main(){
    memset(dp,0,sizeof dp);
    memset(fat,0,sizeof fat);
    fat[0]=1; fat[1]=1;
    LL sum;
    for (int i=2;i<=1000;i++)
        fat[i]=(fat[i-1]*LL(i))%MOD;
    for (int n=0;n<=1000;n++)
        dp[n][0]=1;
    for (int k=1;k<=1000;k++){
        sum=0;
        for (int n=k;n<=1000;n++){
            sum=(sum+dp[n-k][k-1])%MOD;
            dp[n][k]=(dp[n-k][k]+sum)%MOD;
        }
    }
    int T,a,b;
    scanf("%d",&T);
    while (T--){
        scanf("%d%d",&a,&b);
        printf("%I64d\n",(dp[a][b]*fat[b])%MOD);
    }
}
