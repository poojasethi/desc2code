#include <cstdio>
typedef long long LL;
const int N = 1100,p = 1000000007;
LL dp[N][N],c[N][N],f[N];
int main() {
    dp[0][0] = c[0][0] = 1;
    for (int n = 1;n < N;n++) {
        dp[n][0] = 1;
        for (int k = 1;k <= n;k++)
            dp[n][k] = (dp[n-k][k-1]+dp[n-k][k])%p;
    }
    for (int n = 1;n < N;n++)
        for (int m = 1;m <= n;m++)
            c[n][m] = (c[n-1][m-1]+c[n-1][m])%p;
    f[0] = 1;
    for (int i = 1;i < N;i++) f[i] = f[i-1]*i%p;
    int t,n,k;
    scanf("%d",&t);
    while (t--) {
        scanf("%d%d",&n,&k);
        LL ans = 0;
        for (int s = k*(k-1)/2;s+k <= n;s++)
            ans = (ans+dp[s+k][k]*c[n-s][k])%p;
        ans = ans*f[k]%p;
        printf("%I64d\n",ans);
    }
    return 0;
}
