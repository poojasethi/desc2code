#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
typedef long long LL;
const LL mod=1e9+7;
using namespace std;
LL fac[1100],c[1100][1100],dp[1100][1100],ans;
int n,k,T;
int main()
{
	fac[0]=1;
	for (int i=1;i<=1005;++i)
		fac[i]=(fac[i-1]*i)%mod;
	for (int i=0;i<=1005;++i)
	{
		c[i][0]=c[i][i]=1;
		for (int j=1;j<i;++j)
			c[i][j]=(c[i-1][j]+c[i-1][j-1])%mod;
	}
	dp[0][0]=1;
	for (int i=1;i<=1005;++i)
		for (int j=1;j<=i;++j) dp[i][j]=(dp[i-j][j]+dp[i-j][j-1])%mod;
	scanf("%d",&T);
	while (T--)
	{
		scanf("%d%d",&n,&k);
		ans=0;
		for (LL sum=k*(k+1)/2;sum<=n;++sum)
		{
			ans=(ans+dp[sum][k]*c[n-sum+k][k])%mod;
		}
		printf("%I64d\n",ans*fac[k]%mod);
	}
	return 0;
}
