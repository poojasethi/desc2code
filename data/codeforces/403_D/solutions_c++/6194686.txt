#include <bits/stdc++.h>
#define fr(i,a,b) for(int i=a;i<b;i++)
#define rep(i,b) fr(i,0,b)
#define pb push_back
#define MOD 1000000007

using namespace std;
typedef long long ll;

#define N 1010

int pd[N][N];
int fat[N];

int main(){
    fat[0] = 1;
    fr(i,1,N) fat[i] = (fat[i-1]*ll(i))%MOD;
    
    fr(n,0,N) pd[n][0]=1;
    fr(k,1,N){      
        int sum=0;
        fr(n,1,N){  
            if(k<=n){
                sum = (sum + pd[n-k][k-1])%MOD; 
                pd[n][k] = (pd[n-k][k] + sum)%MOD;          
            }
        }
    }
    
    int q;
    scanf("%d",&q);
    rep(i,q){
        int a, b;
        scanf("%d %d",&a,&b);
        printf("%d\n",int((pd[a][b]*ll(fat[b]))%MOD));
    }
    
    return 0;
}
