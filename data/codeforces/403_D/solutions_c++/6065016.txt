#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

const int P=1000000007;
int f[1001][51],g[1001][51];
int t[2501][51],c[51];

int main(){
	memset(f,0,sizeof(f));
	f[0][0]=1;
	for (int i=1; i<=1000; ++i)
		for (int k=50; k>=1; --k)
			for (int j=i; j<=1000; ++j)
				f[j][k]=(f[j-i][k-1]+f[j][k])%P;
	t[0][0]=1;
	for (int i=1; i<=2500; ++i)
		for (int j=0; j<=min(i,50); ++j)
			t[i][j]=(j==0)?1:(t[i-1][j]+t[i-1][j-1])%P;
	c[0]=1;
	for (int i=1; i<=50; ++i)
		c[i]=(long long)c[i-1]*i%P;
	for (int i=1; i<=1000; ++i)
		for (int j=1; j<=50; ++j)
			f[i][j]=(long long)f[i][j]*c[j]%P;
	for (int i=1; i<=1000; ++i)
		for (int j=1; j<=min(i,50); ++j){
			g[i][j]=0;
			for (int k=1; k<=i; ++k){
				int l=i-k;
				int ans=(long long)t[l+j][j]*f[k][j]%P;
				g[i][j]=(g[i][j]+ans)%P;
			}
		}
	int T;
	scanf("%d",&T);
	for (; T;--T){
		int n,k;
		scanf("%d%d",&n,&k);
		if (k>50) printf("0\n");
		else printf("%d\n",g[n][k]);
	}
}
