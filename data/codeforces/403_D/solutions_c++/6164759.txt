#include<stdio.h>
#define mod 1000000007
#define ll long long
ll i,j,k,n,t;
ll f[1100][55],tot[1100][55],c[1100][55],jie[55];
int main(){
	for(i=0;i<=1000;i++)
		for(j=0;j<=50;j++)
			f[i][j]=0;
	for(i=1;i<=1050;i++)
		for(j=1;j<=50&&j<=i;j++){
			if(j==1||j==i)c[i][j]=1;
			else c[i][j]=(c[i-1][j-1]+c[i-1][j])%mod;
		}
	f[0][0]=1;
	for(i=1;i<=1000;i++)
		for(j=1000-i;j>=0;j--)
			for(k=0;k<=50;k++)
				if(f[j][k])
					f[j+i][k+1]=(f[j+i][k+1]+f[j][k])%mod;
	for(i=1;i<=1000;i++)
		for(j=1;j<=50;j++){
			tot[i][j]=0;
			for(k=1;k<=i;k++)
			tot[i][j]=(tot[i][j]+f[k][j]*c[i-k+j][j])%mod;
		}
	for(i=1;i<=1000;i++)
		for(j=1;j<=50;j++){
			tot[i][j]=(tot[i][j]+tot[i-1][j])%mod;
		}
	jie[1]=1;
	for(i=2;i<=50;i++)jie[i]=jie[i-1]*i%mod;
	scanf("%lld",&t);
	while(t--){
		scanf("%lld%lld",&n,&k);
		//printf("%lld\n",f[n][k]);
		if(k>49)printf("0\n");
		else printf("%lld\n",jie[k]*tot[n][k]%mod);
	}
}
