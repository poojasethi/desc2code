#include<cstring>
#include<iostream>
#define M 1000000007
#define N 1004
using namespace std;
int n,k,c[N*2][N*2],f[50][N][N],s[N],ans;
int main(){
	memset(f,0,sizeof f);
	memset(c,0,sizeof c);
	for(int i=0;i<N;i++)
	f[0][0][i]=1;
	for(int k=1;k<50;k++)
	for(int j=(k+1)*k/2;j<N;j++)
	for(int i=k;i<N;i++){
		f[k][j][i]+=f[k][j][i-1],f[k][j][i]%=M;
		if(i<=j) f[k][j][i]+=f[k-1][j-i][i-1],
		f[k][j][i]%=M;
	}
	c[1][0]=c[1][1]=1;
	for(int i=2;i<N*2;i++){
	c[i][0]=1;
	for(int j=1;j<=i;j++)
	c[i][j]=(c[i-1][j]+c[i-1][j-1])%M;
	}
	s[0]=1;
	for(int i=1;i<N;i++)
	s[i]=(long long)s[i-1]*i%M;
	int T;cin>>T;
	while(T--){
		cin>>n>>k;
		if((k*k+1)/2>n) {
			cout<<0<<endl;
			continue;
		}
		ans=0;
		for(int i=1;i<=n;i++)
		ans+=(long long)f[k][i][N-1]*c[n-i+k][k]%M,
		ans%=M;
		ans=(long long)ans*s[k]%M;
		cout<<ans<<endl;
	}
	return 0;
}
