#include<cstdio>
#define N 1010
#define P 1000000007
long long f[N][N],c[N][N],ans[N][N],g[N];
int i,j,k,n,t,len;
int main()
{
	
	f[0][0]=1;
	c[0][0]=1;
	for (i=1;i<=1000;i++)
	{
	c[i][0]=1;
	for (j=1;j<=i;j++)
	c[i][j]=(c[i-1][j-1]+c[i-1][j])%P;
	}
	
	for (j=1;j<=1000;j++)
	for (k=50;k>=1;k--)
	for (i=j;i<=1000;i++)
	f[i][k]=(f[i][k]+f[i-j][k-1])%P;
	
	g[0]=1;
	for (i=1;i<=1000;i++)
	g[i]=(g[i-1]*i)%P;
	for (i=1;i<=1000;i++)
	for (j=1;j<=50;j++)
	f[i][j]=(f[i][j]*g[j])%P;
	for (k=1;k<=50;k++)
	{
		len=k*(k+1)/2;
		for (i=len;i<=1000;i++)
		for (j=i;j<=1000;j++)
		ans[j][k]=(ans[j][k]+f[i][k]*c[j-i+k][k])%P;
	}
	
	scanf("%d",&t);
	for (i=1;i<=t;i++)
	{
		scanf("%d%d",&n,&k);
		printf("%I64d\n",ans[n][k]);
	}
}
