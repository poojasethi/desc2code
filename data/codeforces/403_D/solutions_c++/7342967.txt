#include<cstdio>
using namespace std;

const int maxn = 1001, M = 1e9+7;
int n, k, T, C[maxn][47], pow[47], opt[47][maxn];
int main()
{
  opt[0][0] = C[0][0] = pow[0] = 1;
  for (int i = 1000; i >= 0; i--)
    for (int j = 1000-i; j >= 0; j--)
      for (int k = 45; k >= 0; k--)
        opt[k+1][i+j] = (opt[k+1][i+j]+opt[k][j])%M;
  for (int i = 1; i <= 46; i++) pow[i] = (1LL*pow[i-1]*i)%M;
  for (int i = 1; i <= 1000; i++) {
    C[i][0] = 1;
    for (int k = 1; k <= 46; k++) C[i][k] = (C[i-1][k]+C[i-1][k-1])%M;
  }
  scanf("%d", &T);
  while (T--) {
    scanf("%d%d", &n, &k);
    n--;
    if (1LL*k*(k-1)/2 > n) {
      printf("0\n");
    } else {
      long long ans = 0;
      for (int i = k*(k-1)/2; i <= n-k+1; i++) {
        ans += ((1LL*opt[k][i]*pow[k])%M)*C[n-i+1][k];
        ans %= M;
      }
      printf("%d\n", (int)ans);
    }
  }
}
