#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <algorithm>

#define MOD 1000000007

using namespace std;

long long jc[1105],F[1105][1105],c[1105][1105];
int N,K;

int main()
{
	N=1001;
	jc[0]=1; 
	for (int i=1; i<=N; i++) jc[i]=(jc[i-1]*i)%MOD;
	F[0][0]=1;
	for (int v=1; v<=N; v++)
    {
		for (int j=N; j>=v; j--)
        {
			for (int i=1; i<=45; i++)
			{
                F[i][j]+=F[i-1][j-v];
				F[i][j]%=MOD;
			}
        }
    }
	for (int i=0; i<=N; i++) c[i][0]=1;
	for (int i=1; i<=N; i++)
		for (int j=1; j<=i; j++)
        c[i][j]=(c[i-1][j-1]+c[i-1][j])%MOD;
	int test;
    for (scanf("%d",&test); test>0; test--)
    {
		scanf("%d%d",&N,&K);
        long long Ans=0;
        for (int i=K*(K+1)/2; i<=N; i++) Ans+=(F[K][i]*c[N-i+K][K])%MOD;
        Ans%=MOD; 
		Ans=(Ans*jc[K])%MOD;
		cout << Ans << endl;
    }
	return 0;
}
