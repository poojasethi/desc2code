#include <cstdio>
#define mo 1000000007
int t,n,k;
long long a[1005][1005],b[1005][1005],f[1005];
int main(){
    a[1][1]=1;
    for (int i=2;i<=1000;i++)for (int j=1;j<i;j++){a[i][j]=(a[i-j][j-1]+a[i-j][j])%mo;}
    for (int i=1;i<=1000;i++){b[i][0]=b[i][i]=1;for (int j=1;j<i;j++)b[i][j]=(b[i-1][j-1]+b[i-1][j])%mo;}
    f[1]=1;for (int i=2;i<=1000;i++)f[i]=(f[i-1]*i)%mo;scanf("%d",&t);
    while (t--){scanf("%d%d",&n,&k);long long ans=0;for (int s=(k*(k-1))/2;s+k<=n;s++)ans=(ans+((f[k]*a[s+k][k])%mo*b[n-s][k])%mo)%mo;if (ans<0) ans+=mo;printf("%I64d\n",ans);}
}
