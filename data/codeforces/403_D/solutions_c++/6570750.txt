using namespace std;
int T,A,B,sum[55][1005],dp[55][1005],i,j,k;
long long b[55];
const int MOD=1000000007;
int main()
{
    for (i=0; i<=1000; i++)dp[0][i]=1;
    for (i=0; i<=1000; i++)
    {
        for (j=49; j>=0; j--)
        {
            sum[j][0]=dp[j][0];
            for (k=1; k<=1000; k++) sum[j][k]=(sum[j][k-1]+dp[j][k])%MOD;
            for (k=i+1; k<=1000; k++)
              dp[j+1][k]=(dp[j+1][k]+sum[j][k-i-1])%MOD;
        }
    }
    scanf("%d",&T);
    b[0]=1;
    for (i=1; i<=50; i++) b[i]=b[i-1]*i%MOD;
    while (T--)
    {
        scanf("%d%d",&A,&B);
        if (B>50) {cout<<0<<endl; continue;}
        printf("%d\n",(long long)dp[B][A]*b[B]%MOD);
    }
    return 0;
}
