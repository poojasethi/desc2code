#include <iostream>
#include <cstdio>
using namespace std;

typedef long long LL;

#define N 1000
#define Mod 1000000007

int comb[N+5][N];
int f[N+5][N+5];

void init()
{
	for(int i=0;i<=N;i++)
	{
		comb[i][0]=1;
		for(int j=1;j<=i&&j<=N;j++)
			comb[i][j]=(comb[i-1][j]+comb[i-1][j-1])%Mod;
	}
	f[0][0]=1;
	for(int l=1;l<=N;l++)
		for(int s=0;s<=N;s++)
		{
			f[l][s]=0;
			for(int val=0;val+(val+1)*(l-1)<=s;val++)
				f[l][s]=(f[l][s]+f[l-1][s-val-(val+1)*(l-1)])%Mod;
			f[l][s]=(LL)f[l][s]*l%Mod;
		}
}
int query(int n, int l)
{
	int res=0;
	for(int s=0;s<=n;s++)
		res=(res+(LL)f[l][s]*comb[n-s][l])%Mod;
	return res;
}

int main()
{
	init();

	int nCases;
	cin >> nCases;
	while(nCases--)
	{
		int n, l;
		scanf("%d %d", &n, &l);
		printf("%d\n", query(n, l));
	}

	return 0;
}
