#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <ctime>
#define N 1010
#define M 50
#define mo 1000000007
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,x;
int f[N][M],C[N+M][M];
int main() {
	For(i,0,N+M-1) {
		C[i][0]=1;
		For(j,1,min(i,M-1)) C[i][j]=(C[i-1][j-1]+C[i-1][j])%mo;
	}
	f[0][0]=1;
	For(i,1,N-1)for (k=M-2;k>=0;k--)For(j,0,N-1-i) if ((x=f[j][k])) {
		f[j+i][k+1]=(1ll*x*(k+1)+f[j+i][k+1])%mo;
	}
	for (i=N-1;i;i--)For(j,1,M-1) {
		For(k,1,i-1) if ((x=f[k][j])) {
			f[i][j]=(1ll*x*C[i-k+j][j]+f[i][j])%mo;
		}
	}
	//printf("%d\n",clock());
	scanf("%d",&n);
	For(i,1,n) {
		int x,y;
		scanf("%d%d",&x,&y);
		if (y>=M) printf("0\n");
		else printf("%d\n",f[x][y]);
	}
	return 0;
}
