#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int f[1001][51]={0};
#define mo 1000000007
int fact[1001],c[1001][1001];
int F(int n,int k){
	int ans=0;
	for(int i=k*(k-1)/2;i+k<=n;i++)	ans=(ans+1ll*f[i+k][k]*fact[k]%mo*c[n-i][k]%mo)%mo;
	return ans;
}
int main(){
	fact[0]=1;
	for(int i=1;i<=1000;i++)	fact[i]=1ll*fact[i-1]*i%mo;
	for(int i=1;i<=1000;i++)	f[i][1]=1;
	for(int i=0;i<=1000;i++)	c[i][0]=1,c[i][1]=i;
	for(int i=1;i<=1000;i++)
		for(int j=1;j<=i;j++)
			c[i][j]=(c[i-1][j-1]+c[i-1][j])%mo;
	for(int i=2;i<=1000;i++)
		for(int j=2;j<=min(i,50);j++){
			f[i][j]=(f[i-j][j-1]+f[i-j][j])%mo;
		}
	int tc,n,k;
	scanf("%d",&tc);
	while(tc--){
		scanf("%d%d",&n,&k);
		if(k>50)	printf("0\n");
		else	printf("%d\n",F(n,k));
	}
	return 0;
}
