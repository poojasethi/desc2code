#include<cstdio>
#define p 1000000007
int n,i,j,k,t,ans,h[1001],f[1001][1001],g[1001][1001];
int main()
{
	for(i=0;i<1001;++i)f[0][i]=1;
	for(i=1;i<1001;++i){
		for(j=1;j<=i;++j)f[i][j]=(f[i][j-1]+f[i-j][j])%p;
		for(j=i+1;j<1001;++j)f[i][j]=f[i][j-1];
	}
	for(g[0][0]=i=1;i<1001;++i)
		for(g[i][0]=j=1;j<=i;++j)
			g[i][j]=(g[i-1][j]+g[i-1][j-1])%p;	
	for(h[0]=i=1;i<1001;++i)h[i]=(long long)h[i-1]*i%p;
	scanf("%d",&t);
	while(t--){
		scanf("%d%d",&n,&k);
		for(i=(j=k*(k-3)/2)>0?j:0,ans=0;i<n;++i){
			ans+=(long long)(f[i-j][k]-f[i-j][k-1]+p)*g[n-i][k]%p*h[k]%p;
			if(ans>=p)ans-=p;
		}
		printf("%d\n",ans);
	}
}
