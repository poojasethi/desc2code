#include<cstdio>
int f[50][1002][1002];
int fact[1001];
int MOD=1e9+7;
int c[2002][2002];
int main()
{
	fact[0]=1;
	for(int i=1;i<=1000;i++)fact[i]=(long long)fact[i-1]*i%MOD;
	
	for(int i=0;i<=1000;i++)f[0][0][i]=1;
	for(int i=1;i<50;i++)
	for(int j=i*(i+1)/2;j<=1000;j++)
	for(int k=i;k<=1000;k++)
	{
		f[i][j][k]=f[i][j][k-1];
		if(j>=k)
		{
			f[i][j][k]+=f[i-1][j-k][k-1];
			f[i][j][k]%=MOD;
		}
	}
	for(int i=0;i<=2000;i++)
	{
		c[i][0]=c[i][i]=1;
		for(int j=1;j<i;j++)
		{
			c[i][j]=(c[i-1][j]+c[i-1][j-1])%MOD;

		}
	}
	int t,n,k;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&k);
		int ans=0;
		if(k>=50)
		{
			printf("0\n");
			continue;
		}
		for(int i=1;i<=n;i++)
		{
			ans=(ans+(long long)f[k][i][1000]*c[n-i+k][k]%MOD)%MOD;
		}
		ans=(long long)ans*fact[k]%MOD;
		printf("%d\n",ans);
	}
	return 0;
}
