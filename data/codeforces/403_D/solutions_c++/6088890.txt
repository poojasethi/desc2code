#include <cstdio>

#define MOD 1000000007
#define MX 1005
long long a[50][MX], c[MX][MX], f[MX];

void init() {
	int i, j;
	
	a[1][0] = 1;
	for (i = 1; i < 50; i++) {
		for (j = 0; j < MX; j++) {
			if (j >= i - 1) a[i][j] = (a[i][j] + a[i - 1][j - i + 1]) % MOD;
			if (j >= i) a[i][j] = (a[i][j] + a[i][j - i]) % MOD;
		}
	}
	
	for (i = 0; i < MX; i++) {
		c[i][0] = c[i][i] = 1;
		for (j = 1; j < i; j++) c[i][j] = (c[i - 1][j - 1] + c[i - 1][j]) % MOD;
	}
	
	f[0] = 1;
	for (i = 1; i < MX; i++) f[i] = f[i - 1] * i % MOD; 
}

main() {
//	freopen("in.txt", "r", stdin);
//	freopen("out.txt", "w", stdout);
	
	int n, k, T, x;
	long long rlt;
	
	init();
	for (scanf("%d", &T); T--; ) {
		scanf("%d%d", &n, &k);
		if (n < (k + 1) * k / 2) {
			puts("0");
			continue;
		}
		rlt = 0;
		for (x = 0; x <= n; x++) rlt = (rlt + a[k][x] * c[n - x][k] % MOD) % MOD;
		rlt = rlt * f[k] % MOD;
		printf("%d\n", rlt);
	}
}
