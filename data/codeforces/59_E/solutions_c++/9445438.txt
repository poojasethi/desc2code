#include <iostream>
#include <vector>
#include <queue>
#include <map>
#include <algorithm>
#include <cstdio>
#define N 3002
using namespace std;
int n, m, k, par[N][N], vis[N][N];
queue< pair<int, int> >q;
vector<int>adj[N], R;
map< pair<int, pair<int, int> >, bool >f;
void S(int a, int b){
    R.push_back(b);
    if (a == 1)
        return;
    S(par[a][b], a);
}
int main(){
    scanf("%d%d%d", &n, &m, &k);
    for (int i = 0; i < m; i++){
        int a, b;
        scanf("%d%d", &a, &b);
        adj[a].push_back(b);
        adj[b].push_back(a);
    }
    for (int i = 0; i < k; i++){
        int a, b, c;
        scanf("%d%d%d", &a, &b, &c);
        f[make_pair(a, make_pair(b, c))] = 1;
    }
    q.push(make_pair(0, 1));
    par[0][1] = -1;
    while (!q.empty()){
        int x = q.front().first, y = q.front().second;
        q.pop();
        for (int i = 0; i < adj[y].size(); i++){
            if (f[make_pair(x, make_pair(y, adj[y][i]))] == 1 || vis[y][adj[y][i]]) continue;
            par[y][adj[y][i]] = x;
            vis[y][adj[y][i]] = 1;
            q.push(make_pair(y, adj[y][i]));
            if (adj[y][i] == n){
                S(y, adj[y][i]);
                printf("%d\n1 ", R.size());
                for (int j = R.size() - 1; j>=0; j--)
                    printf("%d ", R[j]);
                return 0;
            }

        }       
    }
    printf("-1");
    //system("pause");
}
