#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<map>
#include<vector>
#define ll long long
using namespace std;

const int maxn=3010;
const int maxm=50010;

struct edge
{
	int y,next;
	edge(){y=next=0;}
}el[maxm];

struct node
{
	int x,p,pre;
	node(){x=p=pre=0;}
}list[2*maxm];

bool v[maxn][maxn];
int d[maxn][maxn];
int first[maxn];
int len,n;
map  < ll , bool >Map;
vector < int > ans;

void ins(int x,int y)
{
	len++;
	el[len].y=y; 
	el[len].next=first[x]; first[x]=len;
}

{
	int tail;

	for(int i=1;i<=tail;i++)

		node &now=list[i];

			while(i!=0)

				ans.push_back(list[i].x);
				i=list[i].pre;
			}
			printf("%d\n",ans.size()-1);
			for(i=ans.size()-1 ; i>=0; i--)
				printf("%d ",ans[i]);
			printf("\n");
			return ;
		}
		for(int k=first[now.x];k;k=el[k].next)
		{
			int y=el[k].y;
			if(Map[(ll)now.p*3000*3000+now.x*3000+y] || v[now.x][y] )continue;
			v[now.x][y]=true;
			tail++;
			list[tail].p=now.x; list[tail].x=y;
			list[tail].pre=i;

}
int main()
{
	int m,k,x,y,z;
	scanf("%d%d%d",&n,&m,&k);

		scanf("%d%d",&x,&y);
