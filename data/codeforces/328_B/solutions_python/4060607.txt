t = raw_input()
x = raw_input()
need = [0] * 10
have = [0] * 10
for c in t:
  need[ord(c) - ord('0')] += 1
for c in x:
  have[ord(c) - ord('0')] += 1
need[2] += need[5]
need[6] += need[9]
need[5] = need[9] = 0
have[2] += have[5]
have[6] += have[9]
have[5] = have[9] = 0
print min([int(have[i] / need[i]) for i in range(10) if i != 5 and i != 9 and need[i] > 0])
