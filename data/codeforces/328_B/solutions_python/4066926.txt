#coding=utf-8

t = raw_input()
pieces = raw_input()

def getCnt(s):
  cnt = [0 for i in range(10)]
  for c in s:
    n = int(c)
    if n == 5: n = 2
    if n == 9: n = 6
    cnt[n] += 1
  return cnt

tCnt = getCnt(t)
pCnt = getCnt(pieces)

num = len(pieces) / len(t)
for i in range(10):
  if tCnt[i] != 0:
    num = min(num, pCnt[i] / tCnt[i])

print num
