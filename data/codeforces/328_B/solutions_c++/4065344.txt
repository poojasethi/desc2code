#include <iostream>
#include <cmath>
#include <string>

using namespace std;

int lazem[10], tedad[10];

int main()
{
	int t;
	string s;
	cin>>t>>s;
	while (t)
	{
		if (t%10 == 9)
			lazem[6]++;
		else if (t%10 == 5)
			lazem[2]++;
		else
			lazem[t%10]++;
		t /= 10;
	}
	for (int i = 0; i<s.length(); i++)
		if (s[i] == '9' || s[i] == '5')
			tedad[s[i]-'0'-3]++;
		else
			tedad[s[i]-'0']++;
	int Min = 0;
	for (int i = 0; i<10; i++)
		if (lazem[i] && (Min == 0 || tedad[i]/lazem[i]<Min))
			Min = tedad[i]/lazem[i];
	cout<<Min<<endl;
	return 0;
}
