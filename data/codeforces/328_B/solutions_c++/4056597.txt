#include <stdio.h>
#include <string.h>

int a[10];
int b[10];

int qq(int x)
{
	if (x==9)
		return 6;
	if (x==5)
		return 2;
	return x;
}

int main()
{
	int i,j,result,len,t;
	char s[300];
	scanf("%d",&t);
	for (i=0; i<10; ++i)
		a[i]=b[i]=0;
	while (t)
	{
		++a[qq(t%10)];
		t/=10;
	}
	scanf("%s",s);
	len=strlen(s);
	for (i=0; i<len; ++i)
		++b[qq(s[i]-'0')];
	result=-1;
	for (i=0; i<10; ++i)
		if (a[i]>0)
		{
			t=b[i]/a[i];
			if ((result==-1)||(t<result))
				result=t;
		}
	if (result==-1)
		result=0;
	printf("%d\n",result);
	return 0;
}
