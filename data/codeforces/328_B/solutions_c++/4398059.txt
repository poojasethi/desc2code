#include <iostream>
#include <algorithm>

using namespace std;

int needed[10];
int counted[10];

int main() {
	string s;
	cin >> s;
	for (int i = 0; i < s.length(); i++) {
		needed[(s[i] - '0')]++;
	}

	needed[2] += needed[5];
	needed[6] += needed[9];

	cin >> s;
	for (int i = 0; i < s.length(); i++)
		counted[(s[i] - '0')]++;

	counted[2] += counted[5];
	counted[6] += counted[9];

	int res = 100000000;
	for (int i = 0; i < 9; i++) {
		if (i == 5)
			continue;

		res = min(res, (needed[i] == 0) ? res : (counted[i] / needed[i]));
	}

	cout << res << endl;
}
