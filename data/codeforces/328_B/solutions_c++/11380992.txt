# include <bits/stdc++.h>

using namespace std;

int a[10] , n[10];

int main(){
	int t;
	cin >> t;
	while(t){
		int m = t %10;
		if(m == 9)
			a[6]++;
		if(m == 5)
			a[2]++;
		if(m != 5 and m!= 9)
			a[m]++;
		t /= 10;
	}

	string s;
	cin >> s;
	for(int i=0 ; i<s.size() ; i++){
		if(s[i] == '9')
			n[6]++;
		if(s[i] == '5')
			n[2]++;
		if(s[i] != 5 and s[i] != 9)
			n[s[i]-'0']++;
	}

	int ans = 100000;
	for(int i=0 ; i<=10 ; i++)
		if(a[i] != 0)
			ans = min(int(n[i] / a[i]) , ans);
	cout << ans << endl;

	return 0;

}
