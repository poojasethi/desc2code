#include<stdio.h>
char a[100],b[205];
int i,c[10]={0},r,d[10]={0};
int main()
{
    scanf("%s %s",a,b);
    for(i=0;b[i];i++)
    {c[b[i]-'0']++;if(b[i]=='5') c[2]++;else if(b[i]=='9') c[6]++;}
    for(i=0;a[i];i++)
    {d[a[i]-'0']++;if(a[i]=='5') d[2]++;else if(a[i]=='9') d[6]++;}
    for(r=1000000,i=0;i<10;i++)
        if(i==5||i==9) continue;
        else if(d[i]!=0&&r>c[i]/d[i])
            r=c[i]/d[i];
    printf("%d\n",r);
    return 0;
}
