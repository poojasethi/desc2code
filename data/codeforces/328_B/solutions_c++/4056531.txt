//In the Name of God
#include <iostream>
#include <cstring>
#include <string>
#include <vector>

using namespace std;

string s , t;
int ted[10] , need[10] , ans = 10000000;

int main(){
  cin >> s >> t;
  for(int i = 0 ; i < t.size() ; i++)
    ted[t[i] - '0']++;
  ted[6] += ted[9];
  ted[2] += ted[5];
  for(int i = 0 ; i < s.size() ; i++)
    need[s[i] - '0']++;
  need[6] += need[9];
  need[2] += need[5];
  for(int i = 0 ; i < 10 ; i++){
    if(i == 9 || i == 5 || need[i] == 0)
      continue;
    ans = min(ans , ted[i] / need[i]);
  }
  cout << ans << endl;
  return 0;
}
