#include<iostream>
#include<string>
using namespace std;

int a[10],d[10]={-1,-1,5,-1,-1,2,9,-1,-1,6};

int main()
{
	
	int t;
	string n;
	cin>>t>>n;
	for(int i=0;i<n.length();i++)
	{
		a[n[i]-'0']++;
		if(n[i]=='9'||n[i]=='5') a[n[i]-'3']++;
		if(n[i]=='2'||n[i]=='6') a[n[i]-'0'+3]++;
	}
	int ans=0;
	while(true)
	{
		int tmp=t;
		while(tmp>0)
		{
			if(a[tmp%10]) {a[tmp%10]--;if(d[tmp%10]!=-1) a[d[tmp%10]]--;} else break;
			tmp/=10;
		}
		if(tmp!=0) break;
		ans++;
	}
	cout<<ans;
	return 0;
}
