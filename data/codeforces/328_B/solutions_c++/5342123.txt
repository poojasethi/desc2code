#include <bits/stdc++.h>
using namespace std;

int c1[10], c2[10];
char s[222];

int main() {
	while (cin >> s) {
		memset(c1, 0, sizeof(c1));
		memset(c2, 0, sizeof(c2));
		int l = strlen(s);
		for (int i = 0; i < l; i++) c1[s[i] - '0']++;
		c1[2] += c1[5]; c1[6] += c1[9];
		c1[5] = c1[9] = 0;
		cin >> s;
		l = strlen(s);
		for (int i = 0; i < l; i++) c2[s[i] - '0']++;
		c2[2] += c2[5]; c2[6] += c2[9];
		c2[5] = c2[9] = 0;
		int ans = 222;
		for (int i = 0; i < 10; i++) {
			if (c1[i] == 0) continue;
			ans = min(c2[i] / c1[i], ans);
		}
		cout << ans << endl;
	}
	return 0;
}
