#include <string>
#include <iostream>

using namespace std;

int ics[10];
int occ[10];
int main(void){
	string t, str;
	cin >> t >> str;
	for(int i = 0; i < (int)str.length(); ++i){
		ics[str[i] - '0']++;
	}	
	for(int i = 0; i < (int)t.length(); ++i){
		occ[t[i] - '0']++;
	}
	ics[6] = ics[6] + ics[9];
	ics[2] = ics[2] + ics[5];
	occ[6] = occ[6] + occ[9];
	occ[2] = occ[2] + occ[5];
	int mn = 100000000;
	for(int i = 0; i < (int)10; ++i){
		if(i == 5 || i == 9 || occ[i] == 0) continue;
		int d = ics[i] / occ[i];
		mn = (mn) < (d) ? (mn) : (d);
	}
	cout << mn << endl;
	return 0;
}
