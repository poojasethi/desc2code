#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int numa[11], numb[11];
char str[1000];

void input(int a[]) {
    scanf("%s", str);
    int n = strlen(str);
    for (int i = 0; i < n; ++i)
	a[str[i] - '0']++;
    a[9] += a[6];
    a[2] += a[5];
    a[6] = a[5] = 0;
}

int main() {
    input(numa);
    input(numb);
    int ans = 100000;
    for (int i = 0; i < 10; ++i) {
	if (!numa[i]) continue;
	ans = min(ans, numb[i] / numa[i]);
    }
    printf("%d\n", ans);
    return 0;
}
