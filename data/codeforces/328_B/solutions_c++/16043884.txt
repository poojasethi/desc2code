#include <cstdio>
#include <cstring>
#include <algorithm>

int c[2][10];

int main()
{
  memset(c, 0, sizeof c);
  int a = 100000000;
  char s[2][205];
  scanf("%s%s", s[0], s[1]);
  for (int i = 0; i < 2; ++i)
    for (int j = 0; s[i][j]; ++j)
  	  if (s[i][j] == '6' || s[i][j] == '9') c[i][6]++;
      else if (s[i][j] == '2' || s[i][j] == '5') c[i][2]++;
      else c[i][s[i][j]-'0']++;
  for (int i = 0; i < 10; ++i) if (c[0][i])
    a = std::min(a, c[1][i]/c[0][i]);
  printf("%d\n", a == 100000000 ? 0 : a);
}
