#include <iostream>
#include <algorithm>
#include <vector>
#include <set>
#include <queue>
#include <cmath>

using namespace std;

int a[10], b[10];

int main()
{
	int t;
	string s;
	cin >> t >> s;
	
	for (; t > 0; t /= 10)
		a[t % 10]++;
	for (int i = 0; i < s.size(); i++)
		b[int(s[i] - '0')]++;
		
	a[2] += a[5];
	a[6] += a[9];
	a[5] = a[9] = 0;
	b[2] += b[5];
	b[6] += b[9];
	b[5] = b[9] = 0;
		
	int res = 1000;
	for (int i = 0; i < 9; i++)
		if (a[i] != 0)
			res = min(res, b[i] / a[i]);
	
	cout << res << '\n';
	
	{char _; cin >> _;}
	return 0;
}
