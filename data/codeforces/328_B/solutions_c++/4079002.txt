#include<iostream>
#include<string>
#include<cmath>
using namespace std;
int main()
{
    int t;
    int a[10]={},b[10]={};
    string s;
    cin>>t>>s;

    for(int i=0;i<s.size();i++)
        a[s[i]-'0']++;
    a[2]+=a[5],a[6]+=a[9];

    if(!t)b[0]++;
    for(int i=t;i>0;i/=10)
        b[i%10]++;
    b[2]+=b[5],b[6]+=b[9];

    int minN(201);
    for(int i=0;i<10;i++)
    if(i==5||i==9)continue;
    else
    {
        if(b[i]!=0)
            if(a[i]==0)minN=0;
            else minN=min(minN,a[i]/b[i]);
    };
    cout<<minN<<endl;
    return 0;
}
