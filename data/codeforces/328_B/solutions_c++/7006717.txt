#include <iostream>
#include <cmath>
#include <string>
using namespace std;
int main () {
	int t, temp, b[10]={0}, min=100000, len, a[11]={0};
	string s;
	cin >> t >> s;
	len = s.length();
	for (int i = 0; i < len; ++i) {
		a[s[i]-'0']++;
	}
	while(t>0) {
		temp = t%10;
		t /= 10;
		b[temp]++;
	}
	b[2] += b[5];
	b[5] = 0;
	b[6] += b[9];
	b[9] = 0;
	a[2] += a[5];
	a[5] = 0;
	a[6] += a[9];
	a[9] = 0;
	for (int i = 0; i < 10; i++) {
		if(b[i]!=0) {
			temp = a[i]/b[i];
			if(temp<min){min=temp;}
		}
	}
	// for (int i = 0; i < 10; ++i) {
	// 	cout << a[i] << " " << b[i] << endl;
	// }
	cout << min << endl;
}
