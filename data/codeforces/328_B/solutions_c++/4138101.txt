#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    char a[6],b[210];cin>>a>>b;
    int la=strlen(a),lb=strlen(b);
    int aa[10],bb[10];memset(aa,0,sizeof(aa));memset(bb,0,sizeof(bb));
    for(int i=0;i<la;i++) aa[a[i]-'0']++;
    for(int i=0;i<lb;i++) bb[b[i]-'0']++;
    aa[2]+=aa[5];aa[5]=0;aa[6]+=aa[9];aa[9]=0;
    bb[2]+=bb[5];bb[5]=0;bb[6]+=bb[9];bb[9]=0;
    int mm=99999;
    for(int i=0;i<=9;i++)
    {
        if(!aa[i]) continue;
        mm=min(mm,bb[i]/aa[i]);
    }
    cout<<mm<<"\n";
}
