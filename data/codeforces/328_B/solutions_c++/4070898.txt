#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

int ad[10];
int rd[10];

int main()
{
   int t;
   string s;
   cin >> t >> s;
   
   while (t > 0) {
       int d = t % 10;
       if (d == 9) d = 6;
       if (d == 5) d = 2;
       rd[d]++;
       t /= 10;
   }
   
   for (int i = 0 ; i<s.length() ; i++) {
       int d = s[i] - '0';
       if (d == 9) d = 6;
       if (d == 5) d = 2;
       ad[d]++;
   }
   
   int res = 1000000;
   for (int d = 0 ; d < 10 ; d++) if (rd[d] > 0)
    res = min(res, ad[d] / rd[d]);
   
   cout << res;
}
