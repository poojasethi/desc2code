#include<iostream>
#include<string>
using namespace std;
int main(){
string t,a;
int min=-1,i,Have[10]={0},Needed[10]={0};
cin>>t>>a;
for(i=0;i<a.length();i++){
    if(a[i]=='9')Have[6]++;
    else if(a[i]=='5')Have[2]++;
    else Have[a[i]-48]++;
}
for(i=0;i<t.length();i++){
    if(t[i]=='9')Needed[6]++;
    else if(t[i]=='5')Needed[2]++;
    else Needed[t[i]-48]++;
}
for(i=0;i<9;i++){
    if(i==5||!Needed[i])continue;
    if(min==-1||Have[i]/Needed[i]<min)
        min=Have[i]/Needed[i];
}
cout<<min<<endl;
return 0;
}
