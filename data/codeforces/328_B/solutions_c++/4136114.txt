#include<stdio.h>
#include<algorithm>
using namespace std;
char t[13],s[256];
int cnt_t[10],cnt_s[10];
int main(){
	scanf("%s",&t);
	scanf("%s",&s);
	for(int i=0;t[i];i++)
		cnt_t[t[i]-'0']++;
	for(int i=0;s[i];i++)
		cnt_s[s[i]-'0']++;
	cnt_t[2]+=cnt_t[5];cnt_t[5]=0;
	cnt_t[6]+=cnt_t[9];cnt_t[9]=0;
	cnt_s[2]+=cnt_s[5];cnt_s[5]=0;
	cnt_s[6]+=cnt_s[9];cnt_s[9]=0;
	int ans=1000;
	for(int i=0;i<10;i++)
		if(cnt_t[i])
			ans=min(ans,cnt_s[i]/cnt_t[i]);
	printf("%d\n",ans);
	return 0;
}
