#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
using namespace std;

int t,a[11],b[11];
char s[201];
int k;

int main()
{
	scanf("%d",&t);
	while (t)
	{
		k=t%10;
		a[k]++;
		t/=10;
	}
	scanf("%s",s);
	int len=strlen(s);
	for (int i=0;i<len;i++)
	{
		b[s[i]-'0']++;
	}
	a[6]+=a[9],a[2]+=a[5];
	b[6]+=b[9],b[2]+=b[5];
	a[9]=0,a[5]=0;
	b[9]=0,b[5]=0;
	
	int min=999999999;
	for (int i=0;i<=9;i++)
		if (a[i]!=0)
		{
			k=b[i]/a[i];
			if (k<min) 
				min=k;
		}
	printf("%d\n",min);
	return 0;
}
