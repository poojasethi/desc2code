#include<iostream>
using namespace std;
int main()
{
	string num,dig;
	cin>>num;
	cin>>dig;
	int n=dig.length();
	int m=num.length();
	int arr[10]={0};
	for(int i=0;i<n;i++)
	{
		arr[dig[i]-48]++;
	}
	arr[2]+=arr[5];
	arr[5]=arr[2];
	arr[6]+=arr[9];
	arr[9]=arr[6];
	
	int need[10]={0};
	for(int i=0;i<m;i++)
	{
		need[num[i]-48]++;
	}
	need[2]+=need[5];
	need[5]=need[2];
	need[6]+=need[9];
	need[9]=need[6];
	int min=2000;
	for(int i=0;i<10;i++)
	{
		if(need[i])
		{
			int tmp=arr[i]/need[i];
			if(tmp<min)
				min=tmp;
		}
	}
	cout<<min;
	return 0;
}
