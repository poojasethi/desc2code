#include <iostream>
#include <algorithm>
using namespace std;
int a[20],b[20];
int main()
{
  string s,t;
  cin>>t>>s;
  for(int i=0;i<s.length();i++)
    {
      if(s[i]=='5')
	a[2]++;
      else if(s[i]=='9')
	a[6]++;
      else
	a[s[i]-'0']++;
    }
  for(int i=0;i<t.length();i++)
    if(t[i]=='5')
      b[2]++;
    else if(t[i]=='9')
      b[6]++;
    else
      b[t[i]-'0']++;
  int Min=100000;
  for(int i=0;i<10;i++)
    {
      if(i!=9 && i!=5 && b[i]!=0)
	Min=min(a[i]/b[i],Min);
    }
  cout<<Min<<endl;
  return 0;
}
