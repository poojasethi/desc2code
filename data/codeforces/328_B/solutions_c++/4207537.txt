#include <iostream>
using namespace std ;

int T[10] = {0} ;
int c[10] = {0} ;
int get(int n){
    if(n == 5)  return 2 ;
    if(n == 9)  return 6 ;
    return n ;
}

int main(){
    int t ;
    cin >> t ;
    while(t){
        c[get(t % 10)]++ ;
        t /= 10 ;
    }
    string s ;
    cin >> s ;
    for(int i=0; s[i]; i++)
        T[get(s[i] - '0')]++ ;
    int ans = 1000 ;
    for(int i=0; i<10; i++){
        if(c[i]){
            ans = min(ans, T[i] / c[i]) ;
        }
    }
    cout << ans << endl ;

    return 0 ;
}
