#include <cstdio>
#include<algorithm>
using namespace std;
char t[10],s[210],*p;
int cnt[10],Cnt[10],i,sol;
int main()
{
    scanf("%s",t);
    for(p=t;*p;p++)cnt[*p-'0']++;
    scanf("%s",s);
    for(p=s;*p;p++)Cnt[*p-'0']++;
    cnt[2]+=cnt[5];cnt[5]=0;
    cnt[6]+=cnt[9];cnt[9]=0;
    Cnt[2]+=Cnt[5];Cnt[5]=0;
    Cnt[6]+=Cnt[9];Cnt[9]=0;
    for(i=0,sol=1000;i<10;i++)
    {
        if(!cnt[i])continue;
        sol=min(sol,Cnt[i]/cnt[i]);
    }
    printf("%d\n",sol);
    return 0;
}
