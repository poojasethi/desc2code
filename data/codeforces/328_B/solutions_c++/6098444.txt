#include <cstdio>
int p[10], q[10];
char ch;
int i, j, k;
int main(){
    do{
        ch = getchar();
        if (ch < '0' || ch > '9') break;
        if (ch == '5') ch = '2';
        if (ch == '9') ch = '6';
        p[ch - '0']++;
    } while (1);
    do{
        ch = getchar();
        if (ch < '0' || ch > '9') break;
        if (ch == '5') ch = '2';
        if (ch == '9') ch = '6';
        q[ch - '0']++;
    } while (1);
    k = 10000000;
    for (i = 0; i < 9; i++) if (p[i] != 0){
        if (q[i] / p[i] < k) k = q[i] / p[i];
    }
    printf("%d", k);
    return 0;
}
