#include<iostream>
#include<string>
#include<cstring>
using namespace std;
int c[10];
int c1[10];

int main()
{
	int n;
	string s;
	cin>>n>>s;
	int i;
	while(n)
	{ 
		c[n%10]++;
		n/=10;
	}

	int l=s.length();
	for(i=0;i<l;i++)
	{
		c1[s[i]-'0']++;
	}
	int max=200000;
	for(i=0;i<10;i++)
	{
		if(i!=2&&i!=5&&i!=6&&i!=9&&c[i]!=0)
		{
			if(c1[i]/c[i]<max)
				max=c1[i]/c[i];
		}
	}
	int a=c[2]+c[5];
	int b=c1[2]+c1[5];
	if(a!=0&&b/a<max)
		max=b/a;
	a=c[6]+c[9];
	b=c1[6]+c1[9];
	if(a!=0&&b/a<max)
		max=b/a;

	cout<<max<<endl;
	return 0;
}
