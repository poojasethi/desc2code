#include <bits/stdc++.h>
using namespace std;

const int maxN=1e4+100;
char s[maxN],t[maxN];
int a[10],b[10];
int n,m;

int main() {
    scanf("%s\n%s",s,t);
    n=strlen(s);
    for(int i=0;i<n;i++) if (s[i]=='5') ++a[2]; else if (s[i]=='9') ++a[6]; else ++a[s[i]-'0'];
    m=strlen(t);
    for(int i=0;i<m;i++) if (t[i]=='5') b[2]++; else if (t[i]=='9') ++b[6]; else ++b[t[i]-'0'];
    int ans=10000;
    for(int i=0;i<9;i++) {
        if (a[i]>0) ans=min(ans,b[i]/a[i]);
//        printf("%i %i\n",b[i],a[i]);
    }
    cout<<ans;
}
