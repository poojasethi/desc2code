#include <stdio.h>
#include <algorithm>
using namespace std;
char A[32],B[256];
int f1[10],f2[10];
int main()
{
    int r=1024;
    scanf("%s%s",A,B);
    for(int i=0;A[i];++i)
        ++f1[A[i]-'0'-3*(A[i]=='5' || A[i]=='9')];
    for(int i=0;B[i];++i)
        ++f2[B[i]-'0'-3*(B[i]=='5' || B[i]=='9')];
    for(int i=0;i<10;++i)
        if(f1[i])
            r=min(r,f2[i]/f1[i]);
    printf("%d",r);
    return 0;
}
