#include <iostream>
#include <cstdio>
#include <cstring>
#include <map>
using namespace std;

map <int,int> X,Y;
char A[1000],B[1000];

int main(){
  scanf("%s %s",A,B);
  for(int i=0;i<strlen(A);++i)
    if(A[i] == '5') X[2]++;
    else if(A[i] == '9') X[6]++;
    else X[A[i]-'0']++;
  for(int i=0;i<strlen(B);++i)
    if(B[i] == '5') Y[2]++;
    else if(B[i] == '9') Y[6]++;
    else Y[B[i]-'0']++;
  int ans = 10000;
  for(map<int,int>::iterator it=X.begin();it!=X.end();++it)
    if(Y.find(it->first) == Y.end()) ans = 0;
    else ans = min(Y[it->first] / it->second, ans);
  printf("%d\n",ans);
}
