#include<stdio.h>
#include<string.h>
int len,t1,t2;
char str[1000];
int main(){
	scanf("%s",str);
	len=strlen(str);
	for(int i=0;i<len;i++){
		if(str[i]=='+'){
			if(t1)t1--,t2++;
			else t2++;
		}else {
			if(t2)t1++,t2--;
			else t1++;
		}
	}
	printf("%d\n",t1+t2);
}
