seq = raw_input().strip()

out_set = []
in_set = []

cnt = 0

for c in seq:
    if c == '+':
        if len(out_set) == 0:
            cnt += 1
            out_set.append(cnt)
        in_set.append(out_set.pop()) 
    else:
        if len(in_set) == 0:
            cnt += 1
            in_set.append(cnt)
        out_set.append(in_set.pop())
print cnt
