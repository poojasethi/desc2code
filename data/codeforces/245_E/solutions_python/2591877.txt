s=raw_input()
out = 0
inside = 0
total = 0
for x in s:
    if x=='+':
        if out>0:
            out-=1
            inside+=1
        else:
            total+=1
            inside+=1
    else:
        if inside>0:
            inside-=1
            out+=1
        else:
            total+=1
            out+=1
print total
