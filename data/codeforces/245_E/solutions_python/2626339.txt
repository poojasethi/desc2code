
out_club , in_club=  0, 0
max_out_club, max_in_club = 0, 0
for ch in raw_input():
  if ch == '+':
    if in_club:
      in_club -= 1
    else:
      out_club += 1
  else:
    if out_club:
      out_club -= 1
    else:
      in_club += 1
  max_out_club = max(max_out_club, out_club)
  max_in_club = max(max_in_club, in_club)
print(max_out_club + max_in_club)
