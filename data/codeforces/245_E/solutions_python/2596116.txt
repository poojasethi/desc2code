def prob4():
	ppl = str(raw_input())
	
	stack_min = []
	stack_pl = []

	cnt = 0
	for ch in ppl:
		if ch == '+':
			if len(stack_min) == 0:
				cnt += 1
			else:
				stack_min.pop()
			stack_pl.append(0)
		else:
			if len(stack_pl) == 0:
				cnt += 1
			else:
				stack_pl.pop()
			stack_min.append(0)

	print cnt

if __name__ == '__main__':
	prob4()
