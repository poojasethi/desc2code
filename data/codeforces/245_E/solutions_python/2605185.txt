def solve(inp):
    m = 0
    l = []
    for x in inp:
        if x == '-':
            if not len(l):
                m = m + 1
            else:
                l.pop()
        else:
            l.append(x)
        m = max(m,len(l))
    return m
    
def main():
    a = raw_input()
    print solve(a)
    return

if __name__ == "__main__":
    main()
