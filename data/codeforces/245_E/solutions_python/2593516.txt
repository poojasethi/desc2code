string = raw_input()
i =0
outdoor =0
indor =0
maxout=0
maxin=0
for a in string:
    if a=='-':
        if indor>0:
            indor-=1
        outdoor+=1
    elif a=='+':
        if outdoor>0:
            outdoor-=1
        indor+=1
    maxin=max(indor,maxin)
    maxout=max(outdoor,maxout)
print(max(maxin,maxout))
