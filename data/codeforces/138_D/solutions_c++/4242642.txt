#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int f[45][45][45][45][5];
char a[25][25];
int v[10005];
int n,m,c;
int sg(int lx,int rx,int ly,int ry,int k)
{	int i,j,x,y;
	if (f[lx][rx][ly][ry][k]!=-1)
		return f[lx][rx][ly][ry][k];
	for (i=1;i<=n;++i)
		for (j=1;j<=m;++j)
		{	x=i+j-1,y=n-i+j;
			if ((x&1)==k&&x>=lx&&x<=rx&&y>=ly&&y<=ry)
				if (a[i][j]=='L')
					sg(lx,x-1,ly,ry,k),sg(x+1,rx,ly,ry,k);
				else if (a[i][j]=='R')
					     sg(lx,rx,ly,y-1,k),sg(lx,rx,y+1,ry,k);
					 else sg(lx,x-1,ly,y-1,k),sg(lx,x-1,y+1,ry,k),sg(x+1,rx,ly,y-1,k),sg(x+1,rx,y+1,ry,k);
		}
	++c;
	for (i=1;i<=n;++i)
		for (j=1;j<=m;++j)
		{	x=i+j-1,y=n-i+j;
			if ((x&1)==k&&x>=lx&&x<=rx&&y>=ly&&y<=ry)
				if (a[i][j]=='L')
					v[sg(lx,x-1,ly,ry,k)^sg(x+1,rx,ly,ry,k)]=c;
				else if (a[i][j]=='R')
					     v[sg(lx,rx,ly,y-1,k)^sg(lx,rx,y+1,ry,k)]=c;
					 else v[sg(lx,x-1,ly,y-1,k)^sg(lx,x-1,y+1,ry,k)^sg(x+1,rx,ly,y-1,k)^sg(x+1,rx,y+1,ry,k)]=c;
		}
	i=0;
	while (v[i]==c)
		++i;
	f[lx][rx][ly][ry][k]=i;
	return f[lx][rx][ly][ry][k];
}

int main()
{	int i;
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;++i)
		scanf("%s",a[i]+1);
	memset(f,-1,sizeof(f));
	if (sg(1,n+m-1,1,n+m-1,0)^sg(1,n+m-1,1,n+m-1,1))
		printf("WIN\n");
	else printf("LOSE\n");
	return 0;
}
