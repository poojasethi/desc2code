#!/usr/bin/python
# coding: utf-8

n, m, mod = map(int, raw_input().split())
res = [2] * n
for i in xrange(m):
  tmp = raw_input()
  for x, y in list(enumerate(tmp)):
    if y == '1':
      res[x] -= 1
num = [0] * 3
for tmp in res:
  if tmp < 0:
    print 0
    exit(0)
  else:
    num[tmp] += 1
mat = [[0] * 505 for _ in xrange(505)]
mat[0][0] = 1
for j in xrange(n + 1):
  for i in xrange(n + 1):
    if i >= 2:
      mat[i][j] += i * (i - 1) // 2 * mat[i - 2][j]
    if j >= 1:
      mat[i][j] += i * j * mat[i][j - 1]
    if j >= 2:
      mat[i][j] += j * (j - 1) // 2 * mat[i + 2][j - 2]
    mat[i][j] %= mod
print mat[num[1]][num[2]]
