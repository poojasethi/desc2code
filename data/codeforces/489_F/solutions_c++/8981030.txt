#include<bits/stdc++.h>

int dp[1111][1111];
int cnt[555],c[4];
int mod;
int dfs(int i,int j)
{
    if(i<0||i>555||j<0||j>555)
        return dp[i][j]=0;
    if(dp[i][j]!=-1)
        return dp[i][j];
    long long tmp=0;
    tmp+=(long long)dfs(i+1,j)*(i+1)*j%mod;
    tmp+=(long long)dfs(i,j+2)*(j+2)*(j+1)/2%mod;
    tmp+=(long long)dfs(i+2,j-2)*(i+2)*(i+1)/2%mod;
    return dp[i][j]=tmp%mod;
}
int main()
{
    memset(dp,-1,sizeof dp);
    int n,m;
    char a[555];
    scanf("%d%d%d",&n,&m,&mod);
    for(int i=0;i<m;i++)
    {
        scanf("%s",a);
        for(int j=0;j<n;j++)
        {
            cnt[j]+=a[j]-'0';
        }
    }
    for(int i=0;i<n;i++)
       c[cnt[i]]++;
    dp[c[0]][c[1]]=1;
    printf("%d\n",dfs(0,0));
    return 0;
}
