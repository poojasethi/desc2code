#include <iostream>
using namespace std;
#define ull unsigned long long
ull n,m,dp[555][555],mof;
int now[555],p1;
char st;
int main(){
    cin>>n>>m>>mof;
    for(int i=1;i<=m;++i)
        for(int j=1;j<=n;++j)
            cin>>st,now[j]+=st-'0';
    for(int i=1;i<=n;++i)
    if(!now[i])
        p1++;
    dp[m][p1]=1;
    for(ull i=m;i<n;++i)
    for(ull j=0;j<=n;++j){
        if(j>=2)
            dp[i+1][j-2]=(j*(j-1)/2*dp[i][j]%mof+dp[i+1][j-2])%mof;
        if(j>=1&&2*n-2*i-j*2>=1)
            dp[i+1][j-1]=(j*(2*n-2*i-j*2)*dp[i][j]%mof+dp[i+1][j-1])%mof;
        if(2*n-2*i-j*2>=2)
            dp[i+1][j]=((2*n-2*i-j*2)*(2*n-2*i-j*2-1)/2*dp[i][j]%mof+dp[i+1][j])%mof;
    }
    cout<<dp[n][0]<<endl;
}
