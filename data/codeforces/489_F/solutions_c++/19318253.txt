#include<cstdio>
using namespace std;

int n,m,cnt[550];
long long f[550][550],mod;
char c[550][550];

int main(){
	scanf("%d%d%I64d",&n,&m,&mod);
	for(int i = 1;i <= m;i++)
		scanf("%s",c[i]+1);
	for(int i = 1;i <= m;i++)
		for(int j = 1;j <= n;j++)
			if(c[i][j]=='1')cnt[j]++;
	int tmp=0;
	for(int i = 1;i <= n;i++)
		if(cnt[i]==1)tmp++;
	f[m][tmp]=1;
	for(int i = m;i < n;i++)
		for(int j = 0;j <= n;j+=2)
		{
			int k=n-i-j/2;
			if(j>0 && k>0)f[i+1][j]+=f[i][j]*k%mod*j%mod;
			if(j>1)f[i+1][j-2]+=f[i][j]*j*(j-1)/2%mod;
			if(k>1)f[i+1][j+2]+=f[i][j]*k*(k-1)/2%mod;
		}
	printf("%I64d",f[n][0]%mod);
	return 0;
}
