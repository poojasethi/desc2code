#include<bits/stdc++.h>

using namespace std;

#define ll long long

int n,m,mod;
int col[505];
ll dp[505][505];
char st[505];

ll playdp(int a, int b){
//	printf("%d %d\n",a,b);
	if(a==0 && b==0) return 1;
	if(dp[a][b]!=-1) return dp[a][b];
	ll tmp=0;
	if(a>1) tmp+=(playdp(a-2,b)%mod*(a*(a-1)/2)%mod)%mod;
	if(b>0) tmp+=((playdp(a,b-1)%mod*b)%mod*a)%mod;
	if(b>1) tmp+=(playdp(a+2,b-2)%mod*(b*(b-1)/2)%mod)%mod;
	tmp%=mod;
	return dp[a][b]=tmp;
}

int main(){
	scanf("%d%d%d",&n,&m,&mod);
	memset(dp,-1,sizeof dp);
	for(int x=0;x<n;x++) col[x]=2;
	for(int x=0;x<m;x++){
		scanf("%s",st);
		for(int y=0;y<n;y++) col[y]-=st[y]-'0';
	}
	int satu=0; int dua=0;
	for(int x=0;x<n;x++) if(col[x]==1) satu++; else if(col[x]==2) dua++;
	printf("%lld\n",playdp(satu,dua));
}
