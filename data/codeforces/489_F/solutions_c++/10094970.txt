#include <cstdio>

using namespace std;

typedef long long ll;

int n,m,nol,satu,col[550];
ll k,dp[550][550];
char dum[550];

int main(){
	scanf("%d %d %lld",&n,&m,&k);
	gets(dum);
	for (int i=1;i<=m;i++){
		gets(dum);
		for (int j=0;j<n;j++){
			if (dum[j]=='1') col[j+1]++;
		}
	}
	for (int i=1;i<=n;i++){
		if (col[i]==0) nol++; else
		if (col[i]==1) satu++;
	}
	dp[nol+2][satu+2]=1;
	for (int i=n+2;i>=2;i--){
		for (int j=n+2;j>=2;j--){
			dp[i-2][j+2]=(dp[i-2][j+2]+dp[i][j]*(i-2)*(i-3)/2)%k;
			dp[i-1][j] = (dp[i-1][j] + dp[i][j]*(i-2)*(j-2))%k;
			dp[i][j-2] = (dp[i][j-2] + dp[i][j]*(j-2)*(j-3)/2)%k;
		}
	}
	printf("%lld\n",dp[2][2]);
}
