#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int mn=505;
int n,m,mod,p,q;
char a[mn][mn];
int s[mn],f[mn][mn];
int main(){
	scanf("%d%d%d",&n,&m,&mod);
	for(int i=0;i<m;++i){
		scanf("%s",a[i]);
		for(int j=0;j<n;++j)
			if(a[i][j]=='1')++s[j];
	}
	for(int i=0;i<n;++i){
		if(s[i]==0)++p;
		if(s[i]==1)++q;
	}
	f[p][q]=1;
	for(int i=p;i>=0;--i)
		for(int j=p+q;j>=0;--j){
		 	if(i>=2)f[i-2][j+2]=(f[i-2][j+2]+(long long)f[i][j]*i*(i-1)/2)%mod;
		 	if(i>=1&&j>=1)f[i-1][j]=(f[i-1][j]+(long long)f[i][j]*i*j)%mod;
		 	if(j>=2)f[i][j-2]=(f[i][j-2]+(long long)f[i][j]*j*(j-1)/2)%mod;
		}
	printf("%d\n",f[0][0]);
	return 0;
}
