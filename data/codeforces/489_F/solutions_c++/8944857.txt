#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll dp[1024][1024];
int column[512];
int v[3];
char re[512];
int n,m,mod;
ll dfs(int x,int y){
	if(dp[x][y]!=-1)	
		return dp[x][y];
	ll ret = 0;
	if(x>0){
		ret += dfs(x-1,y) * x * y;
		ret %= mod;
	}
	if(x>=2){
		ret += dfs(x-2,y+2) * x * (x-1) /2;
		ret %= mod;
	}
	if(y>=2){
		ret += dfs(x,y-2) * y * (y-1) /2;
		ret %= mod;
	}
	return dp[x][y]=ret;
}
int main(){
	scanf("%d%d%d",&n,&m,&mod);
	for(int i=1;i<=m;i++){
		scanf("%s",re);
		for(int k=0;k<n;k++)
			column[k+1]+=re[k]-'0';
	}
	for(int i=1;i<=n;i++)
		v[ column[i] ]++;
	memset(dp,-1,sizeof(dp));
	dp[0][0]=1,dp[0][1]=0,dp[1][0]=0,dp[1][1]=0;
	cout<< dfs(v[0],v[1]) <<endl;
	return 0;
}
