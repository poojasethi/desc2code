#include<iostream>
#include<algorithm>

using namespace std;

const int M = 502;
long long dp[M][M];
int n, m, mod, a[1000];

int main()
{
	cin >> n >> m >> mod;
	for(int i = 0; i < m; i++)
		for(int j = 0; j < n; j++)
		{
			char t;
			cin >> t;
			a[j] += t - '0';
		}
	int k = 0, t = 0;
	for(int i = 0; i < n; i++)
		if(a[i] == 1)
			k++;
		else if(a[i] == 0)
			t++;
	dp[0][0] = 1;
	for(int l = 0; l <= n; l++)
		for(int j = 0; j <= n; j++)
		{
			if(j >= 2)
				dp[j][l] = dp[j - 2][l] * j * (j - 1) / 2;
			if(l >= 1)
				dp[j][l] += dp[j][l - 1] * j * l;
			if(l >= 2)
				dp[j][l] += dp[j + 2][l - 2] * l * (l - 1) / 2;
			dp[j][l] %= mod;
			}
	cout << dp[k][t] << endl;
	return 0;
}
