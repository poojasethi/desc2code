#include <bits/stdc++.h>
using namespace std;
int ar[501],c[3],dp[505][505];
int main(){
	string s;int n,m,mod;cin>>n>>m>>mod;
	memset(ar,0,sizeof(ar));
	for(int i=0;i<m;++i){
		cin>>s;
		for(int j=0;j<n;++j)
			ar[j]+=s[j]-'0';
	}
	for(int j=0;j<n;++j)
		++c[ar[j]];
	dp[c[0]+2][c[1]+2]=1;
	for(int i=n+2;i>=2;--i)
		for(int j=n+2;j>=2;--j){
			dp[i-1][j] = ((1LL*dp[i][j]*(i-2)*(j-2) )+dp[i-1][j] )%mod;
			dp[i][j-2]= ((1LL*dp[i][j]*(j-2)*(j-3))/2 + dp[i][j-2] )%mod;
			dp[i-2][j+2]= ((1LL*dp[i][j]*(i-2)*(i-3))/2 + dp[i-2][j+2] )%mod;
		}
	cout<<dp[2][2]<<endl;
}
