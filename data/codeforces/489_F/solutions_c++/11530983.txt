#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#define N 505
#define LL long long
#define oo (1<<30)
#define TAT "%I64d"
using namespace std;
LL f[N][N];
int n,m,A[N][N],u,v,mo;
int main()
{
	int i,j,k;
	scanf("%d %d %d\n",&n,&m,&mo);
	for(i=1;i<=m;i++){
		for(j=1;j<=n;j++)
			A[i][j]=getchar()-'0';
		scanf("\n");
	  }
	for(i=1;i<=n;i++){
		for(j=1,k=0;j<=m;j++)
			k+=A[j][i];
		if(k==1) v++;
		if(!k) u++;
	  }
	f[0][0]=1;
	for(i=0;i<=u;i++)
		for(j=0;j<=n;j++)
			if(!f[i][j]){
			if(j>1) (f[i][j]+=f[i][j-2]*j*(j-1)/2)%=mo;
			if(i&&j) (f[i][j]+=f[i-1][j]*i*j)%=mo;
			if(i>1) (f[i][j]+=f[i-2][j+2]*i*(i-1)/2)%=mo;
			}
	printf(TAT,f[u][v]);
	return 0;
}
