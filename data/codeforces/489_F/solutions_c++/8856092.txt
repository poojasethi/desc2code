#include <cstdio>

const int N = 503;

long long int dp[N][N];
char slo[N];
int ile[N];

int main()
{
	int n,m,mod;
	scanf("%d%d%d",&n,&m,&mod);
	for (int z=0;z<m;z++)
	{
		scanf("%s",slo);
		for (int i=0;i<n;i++)
			if (slo[i]=='1') ile[i]++;
	}
	int x = 0;
	for (int i=0;i<n;i++) if (ile[i]==0) x++;
	dp[x][m] = 1;
	for (int x=m;x<n;x++)
	{
		int ile = (n-x)*2;
		for (int i=0;i<=n;i++)
		{
			dp[i][x]%=mod;
			//dwa razy dwa
			if ( i > 1 ) dp[i-2][x+1] +=( (long long)i*(i-1)/2 * dp[i][x] )%mod;
			//dwa + 1
			if ( i > 0 ) dp[i-1][x+1] +=( (long long)i*(ile-2*i) * dp[i][x] )%mod;
			dp[i][x+1] +=( (long long)(ile-2*i)*(ile-2*i-1)/2 * dp[i][x] )%mod;
		}
	}
	printf("%lld\n",dp[0][n]);
}
