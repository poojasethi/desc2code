#include <bits/stdc++.h>

using namespace std;

typedef long long int lint;
const int MAXN = 510;

char ar[MAXN];
lint N,M,mod,d[MAXN],dn[MAXN][MAXN];

lint f(int zero,int one,int row){
	if(zero<0 || one<0)
		return 0;
	if(row==N+1)
		return (!zero)&&(!one);
	lint &ret=dn[zero][one];
	if(ret!=-1)
		return ret;
	ret=0;
	ret+=f(zero-2,one+2,row+1)*(zero*(zero-1)/2); ret%=mod;
	ret+=f(zero,one-2,row+1)*(one*(one-1)/2); ret%=mod;
	ret+=f(zero-1,one,row+1)*(one*zero); ret%=mod;
	return ret;
}

int main(){
	
	scanf(" %lld %lld %lld",&N,&M,&mod);
	for(int i=1;i<=M;i++){
		scanf(" %s",ar+1);
		for(int j=1;j<=N;j++)
			d[j]+=ar[j]-'0';
	}
	
	int zero=0,one=0;
	for(int i=1;i<=N;i++)
		if(d[i]==0) zero++;
		else if(d[i]==1) one++;
	
	memset(dn,-1,sizeof dn);
	printf("%lld\n",f(zero,one,M+1));
	
	return 0;
}
