#include<iostream>
#include<string>
using namespace std;
int main(){
	int n,m;
	int dp[510][510], mod;
	int cnt[3];
	int t[510];
	cin>>n>>m>>mod;
	for(int i=0;i<m;i++){
		string tmp;
		cin>>tmp;
		for(int j=0;j<n;j++) t[j]+=tmp[j]-'0';
	}
	for(int i=0;i<n;i++)
		cnt[t[i]]++;
	dp[cnt[0]+2][cnt[1]+2]=1;
	for(int i=n+2;i>=2;i--){
		for(int j=n+2;j>=2;j--){
			dp[i-2][j+2]=(1LL*dp[i-2][j+2]+1LL*dp[i][j]*(i-2)*(i-3)/2)%mod;
			dp[i-1][j]=(1LL*dp[i-1][j]+1LL*dp[i][j]*(i-2)*(j-2))%mod;
			dp[i][j-2]=(1LL*dp[i][j-2]+1LL*dp[i][j]*(j-2)*(j-3)/2)%mod;
		}
	}
	cout<<dp[2][2];
	return 0;
}
