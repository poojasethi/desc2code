#include<bits/stdc++.h>
using namespace std;

long long n, m, mod;
int satu[505]={};
long long dp[505][505];

long long nC2(int n)
{
    return n * (n-1) / 2;
}

long long f(int x, int y){
    if (x < 0 || y < 0) return 0;
    if (x == 0 && y == 0) return 1;
    if (dp[x][y] != -1) return dp[x][y];
    return dp[x][y] = (nC2(x) * f(x-2, y) + nC2(y) * f(x+2, y-2) + x * y * f(x, y-1)) % mod;
}

int main(){
    cin >> n >> m >> mod;
    char c[505];
    gets(c);
    for (int i=0; i<m; i++) {
        gets(c);
        for (int j=0; j<n; j++) if (c[j] == '1') satu[j]++;
    }

    int one = 0, two = 0;
    for (int i=0; i<n; i++)
        if (satu[i] == 1) one++; else if (satu[i] == 0) two++;

    memset(dp, -1, sizeof(dp));

    printf("%lld\n", f(one, two));

    return 0;
}
