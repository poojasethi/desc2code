#include <cstdio>
using namespace std;
const int maxn=1001 ;
long long int dp[maxn][maxn] ;//[bagh2][bagh1]
char s[maxn][maxn] ;
int main()
{
	int n,m,mode ;
	scanf("%d %d %d",&n,&m,&mode) ;
	for(int i=0;i<m;i++)
		scanf("%s",s[i]) ;
	int n2=n,n1=0 ;
	int k ;
	for(int i=0;i<n;i++)
	{
		k=0 ;
		for(int j=0;j<m;j++)
			k+=s[j][i]=='1' ;
		n2-=k>0 ;
		if(k==1)
			n1++ ;
	}
	dp[n2][n1]=1 ;
	long long int k1 ;
	for(long long int i=m+1;i<n+1;i++)
		for(long long int j=0;j<n+1;j++)
		{
			k1=2*n - 2*(i+j) ;
			if(k1>-1 && k1+j<n+1 )
			{
				dp[j][k1]=(dp[j+1][k1]*(j+1)*(k1)+(dp[j][k1+2]*(k1+2)*(k1+1)/2))%mode ;
				if(k1>1)
					dp[j][k1]=(dp[j][k1]+dp[j+2][k1-2]*(j+2)*(j+1)/2)%mode ;
			//	printf("%lld %lld %lld :%lld\n",i,j,k1,dp[j][k1]) ;
			}
		}
	printf("%I64d",dp[0][0]) ;
}
