#include <stdio.h>
int n, m, mod, kk;
long long cnt[501], f[501][501];
char matrix[510];
int main() {
  scanf("%d%d%d", &n, &m, &mod);
  for (int i = 1; i <= m; i++) {
    scanf("%s", matrix+1);
    for (int j = 1; j <= n; j++)
      if (matrix[j] == '1')
        ++cnt[j];
  }
  for (int i = 1; i <= n; i++) if (cnt[i] == 1) ++kk;
  f[m][kk] = 1;
  for (int i = m; i <= n; i++)
    for (int k = 0; k <= n; k++) {//枚举当前行中列1的个数k
      int j = i - k / 2;
      int l = n - j - k;
      if (k >= 2) f[i+1][k-2] = (f[i+1][k-2] + f[i][k]*k*(k-1)/2) % mod;
      if (l >= 2) f[i+1][k+2] = (f[i+1][k+2] + f[i][k]*l*(l-1)/2) % mod;
      if (k && l) f[i+1][k] = (f[i+1][k] + f[i][k]*l*k) % mod;
    }
  printf("%d", f[n][0]%mod);
  return 0;
}
