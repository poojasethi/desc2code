#include <bits/stdc++.h>

using namespace std;

typedef pair<int,  int> pii;
#define NN 111
pii id[NN][NN];
int res[NN][NN];
int main() {
	
	int N, M, Q, i, j, u, v, x, t;
	pii tmp;
	scanf("%d%d%d", &N, &M, &Q);
	for (i=0; i<N; i++) for (j=0; j<M; j++) id[i][j]=pii(i, j);
	
	while (Q--) {
		scanf("%d", &t);
		if (t==1) {
			scanf("%d", &u); u--; 
			tmp=id[u][0];
			for (j=0; j<M-1; j++) id[u][j]=id[u][j+1];
			id[u][M-1]=tmp;
		} else if (t==2) {
			scanf("%d", &v); v--;
			tmp=id[0][v];
			for (i=0; i<N-1; i++) id[i][v]=id[i+1][v];
			id[N-1][v]=tmp;
		} else {
			scanf("%d%d%d", &u, &v, &x); u--; v--;
			res[id[u][v].first][id[u][v].second]=x;
		}
	}
	
	for (i=0; i<N; i++, puts("")) {
		for (j=0; j<M; j++) {
			if (j) printf(" ");
			printf("%d", res[i][j]);
		}
	}
	
	return 0;
}
