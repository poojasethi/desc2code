#include <bits/stdc++.h>

#define NN 110
using namespace std;
typedef pair <int, int> pii;

int mat[NN][NN];
pii p[NN][NN];

int main (){
	
	int N, M, Q, i, j, id, r, c, v;
	
	scanf("%d %d %d", &N, &M, &Q);
	
	for(i=1; i<=N; i++){
		for(j=1; j<=M; j++){
			p[i][j]=pii(i, j);
		}
	}
	
	while(Q--){
		scanf("%d", &id);
		if(id==1){
			scanf("%d", &r);
			pii tmp=p[r][1];
			for(i=1; i<M; i++){
				p[r][i]=p[r][i+1];
			}
			p[r][i]=tmp;
		}
		if(id==2){
			scanf("%d", &c);
			pii tmp=p[1][c];
			for(i=1; i<N; i++){
				p[i][c]=p[i+1][c];
			}
			p[i][c]=tmp;
		}
		if(id==3){
			scanf("%d %d %d", &r, &c, &v);
			mat[p[r][c].first][p[r][c].second]=v;
		}
	}
	
	for(i=1; i<=N; i++){
		for(j=1; j<=M; j++){
			printf("%d ", mat[i][j]);
		}
		puts("");
	}
	
	return 0;
}
