#include <iostream>
#include <cstdio>
#define MAXN 105
#define MAXM 10005
using namespace std;

int n, m, q, num[MAXN][MAXN], opt[MAXM], x[MAXM], y[MAXM], z[MAXM], tmp;

int main()
{
	scanf("%d%d%d",&n,&m,&q);
	for(int i=1;i<=q;++i)
	{
		scanf("%d%d",&opt[i],&x[i]);
		if(opt[i]==3)scanf("%d%d",&y[i],&z[i]);
	}
	for(int i=q;i;--i)
	{
		if(opt[i]==3)num[x[i]][y[i]]=z[i];
		else if(opt[i]==1)
		{
			tmp=num[x[i]][m];
			for(int j=m;j>1;--j)num[x[i]][j]=num[x[i]][j-1];
			num[x[i]][1]=tmp;
		}
		else
		{
			tmp=num[n][x[i]];
			for(int j=n;j>1;--j)num[j][x[i]]=num[j-1][x[i]];
			num[1][x[i]]=tmp;
		}
	}
	for(int i=1;i<=n;++i)
	{
		for(int j=1;j<m;++j)printf("%d ",num[i][j]);
		printf("%d\n",num[i][m]);
	}
	return 0;
}
