#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define MAX 1000010
#define mod 1000000007

int n,m,q;
int a[110][110],b[10010][4];

int main() {
	cin >> n >> m >> q;
	for (int i=0;i<q;++i) {
		cin >> b[i][0];
		if (b[i][0]<3)
			cin >> b[i][1];
		else
			cin >> b[i][1] >> b[i][2] >> b[i][3];
	}
	for (int i=q-1;i>=0;--i) {
		if (b[i][0]==1) {
			int tmp=a[b[i][1]][m];
			for (int j=m;j>1;--j)
				a[b[i][1]][j]=a[b[i][1]][j-1];
			a[b[i][1]][1]=tmp;
		}
		else if (b[i][0]==2) {
			int tmp=a[n][b[i][1]];
			for (int j=n;j>1;--j)
				a[j][b[i][1]]=a[j-1][b[i][1]];
			a[1][b[i][1]]=tmp;
		}
		else
			a[b[i][1]][b[i][2]]=b[i][3];
	}
	for (int i=1;i<=n;++i) {
		for (int j=1;j<=m;++j)
			cout << a[i][j] << ' ';
		cout << endl;
	}
	return 0;
}
