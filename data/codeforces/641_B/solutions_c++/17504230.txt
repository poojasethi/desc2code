#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
#define rep(i,a,b) for(int i=a;i<=b;++i)
#define erp(i,a,b) for(int i=a;i>=b;--i)
#define LL long long
using namespace std;

int a[105][105];
int Q, n, m;
int op[10005];
int x[10005], y[10005], z[10005];
int main()
{
	scanf("%d%d%d",&n,&m,&Q);
	rep(i, 1, Q)
	{
		scanf("%d%d", op+i, x+i);
		if (op[i]==3) scanf("%d%d", y+i, z+i);
	}
	erp(t, Q, 1)
	{
		if (op[t]==3) a[x[t]][y[t]]= z[t];
		else if (op[t]==1)
		{
			int tt = a[x[t]][m];
			for (int i = m; i>1; --i)
				a[x[t]][i] = a[x[t]][i-1];
			a[x[t]][1] = tt;
		}
		else
		{
			int tt=a[n][x[t]];
			for (int i = n; i>1; --i)
				a[i][x[t]] = a[i-1][x[t]];
			a[1][x[t]] = tt;
		}
	}
	rep(i,1,n)rep(j,1,m)printf("%d%c",a[i][j],j==m?'\n':' ');
	return 0;
}
