#!/usr/bin/python

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n, m, q = ia()
res = [[0     for j in range(m)] for i in range(n)]
a   = [[(i,j) for j in range(m)] for i in range(n)]

for it in range(q):
    line = ia()
    ty   = line[0]
    if ty == 1:
        r = line[1]
        r = r - 1
        tmp = a[r][0]
        for j in range(1, m):
            a[r][j-1] = a[r][j]
        a[r][m-1] = tmp
    elif ty == 2:
        c = line[1]
        c = c - 1
        tmp = a[0][c]
        for i in range(1, n):
            a[i-1][c] = a[i][c]
        a[n-1][c] = tmp
    elif ty == 3:
        i = line[1]; i = i - 1
        j = line[2]; j = j - 1
        x = line[3]
        res[ a[i][j][0] ] [ a[i][j][1] ] = x

for i in range(n):
    for j in range(m):
        print res[i][j],
    print
