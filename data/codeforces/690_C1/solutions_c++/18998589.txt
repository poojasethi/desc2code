#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define fo(i,a,b) for(int i=a;i<=b;i++)
#define fd(i,a,b) for(int i=a;i>=b;i--)
#define maxn 1005
using namespace std;

int fa[maxn];

int n,m;

int get(int x){
	return fa[x]==x ? x : fa[x]=get(fa[x]);
}

int main(){
	scanf("%d%d",&n,&m);
	if (m!=n-1) {
		puts("no");
		return 0;
	}
	fo(i,1,n) fa[i]=i;
	fo(i,1,m) {
		int x,y;
		scanf("%d%d",&x,&y);
		int fx=get(x);
		int fy=get(y);
		fa[fx]=fy;
	}
	fo(i,1,n) get(i);
	fo(i,2,n) if (fa[i]!=fa[i-1]) {
		puts("no");
		return 0;
	}
	puts("yes");
	return 0;
}
