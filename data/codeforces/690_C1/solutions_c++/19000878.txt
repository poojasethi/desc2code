#include <cstdio>
#include <algorithm>
#include <cstring>
#include <iostream>
#define rep(j,k,l) for (int j=k;j<=l;j++)

using namespace std;
int fa[1111];

int ff(int k){
	
	if (k==fa[k]) return k;
	return fa[k]=ff(fa[k]);
	
}

int main(){
	
	int n,m;
	scanf("%d%d",&n,&m);
	if (m!=n-1){printf("no\n");return 0;}
	rep(i,1,n) fa[i]=i;
	rep(i,1,m){
		
		int k,l;
		scanf("%d%d",&k,&l);
		int o=ff(k),p=ff(l);
		if (o!=p) fa[o]=p;
		else{
			
			printf("no\n");
			return 0;
			
		}
		
	}
	printf("yes\n");
	return 0;
	
}
