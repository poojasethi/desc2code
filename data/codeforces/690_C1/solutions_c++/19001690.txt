#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
typedef long long ll;
int fat[1005];
int finds(int x){
	if(fat[x]!=x)return fat[x]=finds(fat[x]);
	return fat[x];
}
int main(){
	int n,m;
	int i;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)fat[i]=i;
	for(i=1;i<=m;i++){
		int a,b;
		scanf("%d%d",&a,&b);
		int root1=finds(a);
		int root2=finds(b);
		if(root1!=root2){
			fat[root1]=root2;
		}
	}
	int flag=1;
	if(n-1!=m)flag=0;
	for(i=2;i<=n;i++){
		if(finds(fat[i])!=finds(fat[i-1]))flag=0;
	}
	if(flag)printf("yes\n");
	else printf("no\n");
	return 0;
}
