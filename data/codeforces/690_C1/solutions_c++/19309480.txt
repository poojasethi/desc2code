#include <bits/stdc++.h>

using namespace std;
int n,m,fa[1005];
int find(int x) {return fa[x]==x?x:fa[x]=find(fa[x]);}
int main()
{
	
	scanf("%d%d\n",&n,&m);
	if(m!=n-1){
		printf("no\n");
		return 0;
	}
	for(int i=1;i<=n;i++) fa[i]=i;
	for(int i=1;i<=m;i++){
		int x,y;
		scanf("%d%d",&x,&y);
		x= find(x) ,y= find(y);
		fa[x]=y;
	}

	int cnt=0;
	for(int i=1;i<=n;i++){if(fa[i]==i)	cnt++;}
	if(cnt>1)
		printf("no\n");
	else
		printf("yes\n");

	return 0;
}
