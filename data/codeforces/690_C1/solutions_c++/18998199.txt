#include <cstdio>

using namespace std;

int p[1001], rank[1001];

int getanc(int x) {
	return p[x] == x ? x : (p[x] = getanc(p[x]));
}

inline void unite(int x, int y) {
	if (rank[x] > rank[y]) p[y] = x;
	else if (rank[x] < rank[y]) p[x] = y;
	else {
		++rank[x];
		p[y] = x;
	}
}

int main() {
	int n, m, i, x, y;
	
	scanf("%d%d", &n, &m);
	if (n != m + 1) {
		printf("no\n");
		return 0;
	}
	for (i = 0; i <= n; ++i) p[i] = i;
	for (i = 0; i < m; ++i) {
		scanf("%d%d", &x, &y);
		if ((x = getanc(x)) == (y = getanc(y))) {
			printf("no\n");
			return 0;
		}
		unite(x, y);
	}
	
	printf("yes\n");
	
	return 0;
}
