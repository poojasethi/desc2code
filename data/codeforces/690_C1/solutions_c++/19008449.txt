#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <cstring>
#include <string>
using namespace std;

int n, m, fa[1004], cnt = 0;

int f(int x){
	return x == fa[x] ? x : (fa[x] = f(fa[x]));
}

int main(){
	cin >> n >> m;
	for(int i = 1; i <= n; i++) fa[i] = i;
	int x, y;
	for(int i = 1; i <= m; i++){
		cin >> x >> y;
		if(f(x) == f(y)){
			printf("no\n");
			return 0;
		}
		else {
			cnt++;
			fa[f(y)] = f(x);
		}
	}
	if(cnt == n - 1){
		printf("yes\n");
		return 0;
	}
	printf("no\n");
	return 0;
}
