#include<stdio.h>
int flag[1005] = {0};
int Find(int x);
int main(void)
{
	int i, a, b, t1, t2, n, m, ans, sum;
	scanf("%d%d", &n, &m);
	sum = n-1;
	ans = 1;
	for(i=1;i<=m;i++)
	{
		scanf("%d%d", &a, &b);
		t1 = Find(a);
		t2 = Find(b);
		if(t1==t2)
			ans = 0;
		else
		{
			flag[t1] = t2;
			sum--;
		}
	}
	if(sum==0 && ans==1)
		printf("yes\n");
	else
		printf("no\n");
	return 0;
}

int Find(int x)
{
	if(flag[x]==0)
		return x;
	return Find(flag[x]);
}
