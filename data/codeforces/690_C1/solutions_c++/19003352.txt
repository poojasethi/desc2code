#include<iostream>
#define MAX_N 1000
using namespace std;
bool a[MAX_N][MAX_N];
bool used[MAX_N];
int n,m;
void dfs(int x){
  used[x]=true;
  for(int i=0;i<n;i++){
    if(!used[i]&&a[x][i]){
      dfs(i);
    }
  }
}
void solve(){
  if(m!=n-1){
    cout<<"no"<<endl;
  }
  else{
    int cnt=0;
    for(int i=0;i<n;i++){
      if(!used[i]){
        dfs(i);
        cnt++;
      }
    }
    if(cnt==1){
      cout<<"yes"<<endl;
    }
    else{
      cout<<"no"<<endl;
    }
  }
}
int main(){
  cin>>n>>m;
  for(int i=0;i<m;i++){
    int x,y;
    cin>>x>>y;
    a[x-1][y-1]=a[y-1][x-1]=true;
  }
  solve();
  return 0;
}
