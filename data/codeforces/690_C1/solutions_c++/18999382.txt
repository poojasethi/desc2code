#include<iostream>
#include<vector>
using namespace std;

const int oo=1010;
int n,m,x,y;
bool mark[oo];
vector<int> edg[oo];

void dfs(int v){
  mark[v]=true;
  for(int i=0;i<edg[v].size();i++)
    if(!mark[edg[v][i]])
      dfs(edg[v][i]);
}

int main(){
  cin>>n>>m;
  if(n-1!=m){
    cout<<"no";
    return 0;
  }
  for(int i=0;i<m;i++)
    cin>>x>>y,edg[x].push_back(y),edg[y].push_back(x);
  dfs(1);
  for(int i=1;i<=n;i++)
    if(!mark[i]){
      cout<<"no";
      return 0;
    }
  cout<<"yes";
  return 0;
}
