#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
#define N 1005
int n,m;

struct node
{
	int a,b;
}edge[N];

int r[N];

int find(int x)
{
	return x==r[x]?x:r[x]=find(r[x]);
}

int main()
{
	while(cin>>n>>m)
	{
		for(int i=1;i<=m;i++)
		{
			cin>>edge[i].a>>edge[i].b;
		}
		for(int i=1;i<=n;i++)
		    r[i]=i;
	    if(m!=n-1)cout<<"no"<<endl;
		else
		{
			int flag=1;
			for(int i=1;i<=m;i++)
			{
				int x=find(edge[i].a);
				int y=find(edge[i].b);
				if(x==y)
				{
					flag=0;
					break;
				}
				else r[y]=x;
			}
			if(flag)cout<<"yes"<<endl;
			else cout<<"no"<<endl;
		}
	}
	return 0;
}
