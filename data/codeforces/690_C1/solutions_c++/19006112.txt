#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<string.h>

using namespace std;

const int N=12345;
int fa[N];

int find(int x)
{
    if(x==fa[x]) return x;
    return fa[x] = find(fa[x]);
}

int main()
{
    int n,m;
    cin>>n>>m;
    if(n-1 != m){
        return 0*puts("no");
    }

    for(int i=1;i<=n;i++) fa[i] = i;
    for(int i=0;i<m;i++){
        int x,y;
        cin>>x>>y;
        x=find(x),y=find(y);
        if(x!=y) fa[x] = y;
    }
    int cnt=0;
    for(int i=1;i<=n;i++){
        cnt += fa[i]==i;
    }
    puts(cnt==1?"yes":"no");
    return 0;
}
