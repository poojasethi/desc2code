#include <bits/stdc++.h>
using namespace std;
int n,m,pr[1010];
int parent(int x){
	if(pr[x]==0) return x;
	return pr[x] = parent(pr[x]);
}
int main(){
	//freopen("in.in","r",stdin);
	cin >> n >> m;
	if(m!=n-1) cout << "no";
	else{
		int cnt = n;
		for(int i = 1; i <= m; i++){
			int u,v ; cin >> u >> v;
			u = parent(u);
			v = parent(v);
			if(u!=v){
				cnt--;
				if(u>v) swap(u,v);
				pr[v] = u;
			}
		}
		if(cnt==1) cout << "yes";
		else cout << "no";
	}
}
