#include<cstdio>
int p[1001];
/*dsu*/find(int x){return p[x]==x?x:find(p[x]);}
/*dsu*/join(int x,int y){x=find(x);y=find(y);p[x]=y;}
main(){
    int n,m,i=1,a,b;
    bool loop=false;
    for(scanf("%d%d",&n,&m);i<=n;i++)p[i]=i;
    if(n!=m+1)puts("no");
    else{
        for(i=1;i<=m&&!loop;i++){
            scanf("%d%d",&a,&b);
            if(find(a)==find(b))loop=true;
            join(a,b);
        }
        puts(loop?"no":"yes");
    }
}
