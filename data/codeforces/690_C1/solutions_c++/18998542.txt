#include<cstdio>
using namespace std;
int per[1005];
int find(int x)
{
    return per[x] == x ? x : find(per[x]);
}
void link(int a,int b)
{
    int s = find(a);
    int t = find(b);
    per[s] = t;
    return;
}
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    if(m != n-1){
        printf("no\n");
        return 0;
    }
    for(int i = 1;i <= n;i++){
        per[i] = i;
    }
    int a,b;
    for(int i = 1;i <= m;i++){
        scanf("%d%d",&a,&b);
        if(find(a) != find(b)){
            link(a,b);
        }
        else{printf("no\n");return 0;}
    }
    printf("yes\n");
    return 0;
}
