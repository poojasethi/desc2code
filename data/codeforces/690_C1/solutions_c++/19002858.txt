#include<string.h>
#include<stdio.h>
#define MAX 1005
int f[MAX];
int find(int x)
{return f[x]==x?x:f[x]=find(f[x]);}
int main()
{
	int m,n,a,b;
	while(~scanf("%d%d",&m,&n))
	{
		int cnt=0,ff=1;
		for(int i=0;i<=m;i++)f[i]=i;
		for(int i=0;i<n;i++)
		{
			scanf("%d%d",&a,&b);
			if(find(a)!=find(b))f[find(a)]=find(b),cnt++;
			else ff=0;
		}
		if((cnt==m-1)&&ff)puts("yes");
		else puts("no");
	}
	return 0;
}
