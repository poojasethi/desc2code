#include<bits/stdc++.h>
using namespace std;

const int MAXN = 1005;

namespace disset
{
  int fa[MAXN];
  void init(int n) { for(int i=1;i<=n;i++) fa[i] = i; }
  int find(int x) { return x == fa[x] ? x : fa[x] = find(fa[x]); }
  void merge(int x, int y) { fa[find(x)] = find(y); }
  bool judge(int x, int y) { return find(x) == find(y); }
}

int N, M;

void work()
{
  int u, v;
  scanf("%d%d", &N, &M);
  disset::init(N);
  if(M != N-1) { puts("no"); return; }
  while(M--) {
    scanf("%d%d", &u, &v);
    if(disset::judge(u, v)) { puts("no"); return; }
    disset::merge(u, v);
  }
  puts("yes");
}

int main()
{
  //freopen("in.txt", "r", stdin);
  work();
  return 0;
}
