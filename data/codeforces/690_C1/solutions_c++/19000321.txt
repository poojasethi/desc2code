#include <bits/stdc++.h>
using namespace std;
#define N 1010
vector<int> g[N];
bool vis[N];
int cnt;
void dfs(int u){
	vis[u]=true;
	for(int i=0;i<g[u].size();i++){
		if(vis[g[u][i]]) continue;
		dfs(g[u][i]);
		cnt++;
	}
}
int main(){
	///freopen("tmp.txt","r",stdin);
	int n,m,a,b;
	while(~scanf("%d%d",&n,&m)){
		memset(vis,false,sizeof(vis));
		cnt=1;
		for(int i=1;i<=m;i++){
			scanf("%d%d",&a,&b);
			g[a].push_back(b);
			g[b].push_back(a);
		}
		if(m!=n-1){
			printf("no\n");
			continue;
		}
		dfs(1);
		if(cnt==n)printf("yes\n");
		else printf("no\n");
		for(int i=1;i<=n;i++) g[i].clear();
		}
	return 0;
}
