#include <iostream>
using namespace std;

int cid[1000];

int find(int i)
{
	if(cid[i]==i) return i;
	return cid[i] = find(cid[i]);
}
void join(int i,int j)
{
	i = find(i), j = find(j);
	cid[i] = j;
}

int main()
{
	int N,a,b,M;
	cin >> N >> M;
	for(int i=0;i<N;i++)
		cid[i] = i;
	for(int i=0;i<M;i++)
	{
		cin >> a >> b;
		a--,b--;
		join(a,b);
	}
	if(M!=N-1)
	{
		cout << "no\n";
		return 0;
	}
	for(int i=0;i<N;i++)
	{
		if(find(i)!=find(0))
		{
			cout << "no\n";
			return 0;
		}
	}
	cout << "yes\n";
	return 0;
}
