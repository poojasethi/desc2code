#include<bits/stdc++.h>
using namespace std;

int n,m;
vector<int> v[10000];
bool used[10000];

void dfs(int x,int p)
{
    used[x]=1;
    for(int i=0;i<v[x].size();i++)
        if(!used[v[x][i]])
            dfs(v[x][i],x);
}

int main()
{
    cin>>n>>m;
    if(m+1!=n)
    {
        cout<<"no\n";
        return 0;
    }
    for(int i=1;i<n;i++)
    {
        int x,y;
        cin>>x>>y;
        v[x].push_back(y);
        v[y].push_back(x);
    }
    dfs(1,1);
    for(int i=1;i<=n;i++)
        if(!used[i])
        {
            cout<<"no\n";
            return 0;
        }
    cout<<"yes\n";
}
