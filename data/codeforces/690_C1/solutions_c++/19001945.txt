#include <bits/stdc++.h>

#define NN 1010

using namespace std;

int root[NN];

int find_root(int u){
	if(root[u]==u) return u;
	return root[u]=find_root(root[u]);
}

int main() {
	
	int N, M, i, u, v;
	scanf("%d %d", &N,&M);
	
	for(i=1; i<=N; i++) root[i]=i;
	
	int tot=N;
	for(i=0; i<M; i++){
		scanf("%d %d", &u, &v);
		int uu=find_root(u), vv=find_root(v);
		if(uu==vv) continue;
		root[uu]=vv;
		tot--;
	}
	
	if(M==N-1 && tot==1) puts("yes");
	else puts("no");
	
	return 0;
}
