#include <iostream>
#include <cstdlib>
#include <cstdio>
using namespace std;
const int N = 1010;
int n,m;
int fa[N];
int find(int a)
{
	if(fa[a]==a) return a;
	return fa[a] = find(fa[a]);
}
int main()
{
	scanf("%d%d",&n,&m);
	if(m!=n-1) {printf("no");return 0;}
	for(int i = 1; i <= n; i++) fa[i] = i;
	int a,b;
	for(int i = 1; i <= m; i++)
	{
		scanf("%d%d",&a,&b);
		fa[find(a)] = find(b);
	}
	int tot = 0;
	for(int i = 1; i <= n; i++) tot += (find(i) == i);
	if(tot == 1) printf("yes");
	else printf("no");
	return 0;
}
