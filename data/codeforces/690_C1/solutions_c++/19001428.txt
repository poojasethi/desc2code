#include <iostream>
#include <cstdio>

using namespace std;

int father[1005];

int find(int x)
{
    if(father[x] == x) return x;
    return father[x] = find(father[x]);
}

void Union(int x,int y)
{
    int rx,ry;
    rx = find(x);
    ry = find(y);
    father[rx] = ry;
}

int main()
{
	int n, m;
	scanf("%d%d", &n, &m);
	bool flag = (n == m+1);
	for (int i = 1; i <= n; i++) {
		father[i] = i;
	} 
	for (int i = 0; i < m; i++) {
		int a, b;
		scanf("%d%d", &a, &b);
		int fa = find(a), fb = find(b);
		if (fa == fb)	flag = false;
		else Union(fa, fb); 
	}
	if (flag) cout << "yes\n";
	else cout << "no\n";
	return 0;
}
