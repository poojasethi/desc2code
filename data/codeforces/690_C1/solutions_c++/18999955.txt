#include<cstdio>

int n, m, f[1005];
bool ok = true;

int find(int x) {
	if (f[x] == x) return x;
	return f[x] = find(f[x]);
}

int main() {
	scanf("%d %d", &n, &m);
	for (int i = 1; i <= n; i++) f[i] = i;
	if (m != n - 1) ok = false;
	for (int i = 0; i < m; i++) {
		int u, v;
		scanf("%d %d", &u, &v);
		if(f[find(u)] == find(v)) ok = false;
		f[find(u)] = find(v);
	}
	printf(ok ? "yes\n" : "no\n");
}
