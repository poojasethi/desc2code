#include <stdio.h>
#include <vector>
using namespace std;

void find(int now, bool* visit, vector<int> r[])
{
	visit[now] = true;
	for (int i=0;i<r[now].size();i++) if (!visit[r[now].at(i)]) find(r[now].at(i), visit, r);
}

int main()
{
	int n, m, i, a, b;
	scanf("%d%d", &n, &m);
	vector<int> road[n+1];
	bool visit[n+1], ok = true;
	for (i=1;i<=n;i++) visit[i] = false;
	for (i=1;i<=m;i++)
	{
		scanf("%d%d", &a, &b);
		road[a].push_back(b);
		road[b].push_back(a);
	}
	
	if (m!=n-1) printf("no\n"); else
	{
		find(1, visit, road);
		for (i=1;i<=n;i++) if (!visit[i]) ok = false;
		if (ok) printf("yes\n"); else printf("no\n");
	}
	return 0;
}
