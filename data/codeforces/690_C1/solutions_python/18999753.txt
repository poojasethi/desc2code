#include <iostream>
#include <vector>

using namespace std;

int used[1001];
vector<int> a[1001];
int n,m;
int count;

void dfs(int start) {
	int i;
	used[start] = 1;
	for (i = 0; i < a[start].size(); i++)
		if (used[a[start][i]] == 0)
			dfs(a[start][i]);
}

int main() {
	int i,flag,x,y;
	cin >> n >> m;
	flag = m == n-1;
	for (i = 1; i <= m; i++) {
		cin >> x >> y;
		a[x].push_back(y);
		a[y].push_back(x);
	}
	dfs(1);
	for (i = 1; i <= n; i++)
		flag = flag && used[i];
	if (flag)
		cout << "yes" <<endl;
	else
		cout << "no" << endl;
	return 0;
}
