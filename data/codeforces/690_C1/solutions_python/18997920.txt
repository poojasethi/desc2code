int fa[1010];
int find(int x) { return fa[x]==x?x:fa[x]=find(fa[x]);}
int main()
{
	int n,m; scanf("%d%d",&n,&m);
	if (m!=n-1) { printf("no\n"); return 0;}
	for (int i=1;i<=n;i++) fa[i]=i;
	for (int i=1;i<=m;i++)
	{
		int x,y; scanf("%d%d",&x,&y);
		int f1=find(x),f2=find(y);
		if (f1==f2) { printf("no\n"); return 0;}
		fa[f1]=f2;
	}
	printf("yes\n");
	return 0;
}
