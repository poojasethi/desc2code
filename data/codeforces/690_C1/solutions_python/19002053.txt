#include <iostream>
#include <vector>

using namespace std;

int n,m,res=0,f[2000];
vector<int > a[2000];

void dfs(int u)
{
    f[u]=1;
    res++;
    for (int i=0;i<a[u].size();i++)
        if (!f[a[u][i]]) dfs(a[u][i]);
}

int main()
{
    cin >> n >> m;
    for (int i=1;i<=m;i++)
    {
        int u,v;
        cin >> u >> v;
        a[u].push_back(v);
        a[v].push_back(u);
    }
    if (m>=n)
    {
        cout << "no";
        return 0;
    }
    dfs(1);
    if (res<n) cout << "no";
    else cout << "yes";
    return 0;
}
