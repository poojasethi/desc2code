#include<bits/stdc++.h>
using namespace std;
vector<int>adj[1001];
bool visited[1001];
int cnt;

void dfs(int s)
{
    visited[s]=true;
    --cnt;
    for(int i=0;i<adj[s].size();i++)
    {
        if(visited[adj[s][i]]==false)
            dfs(adj[s][i]);
    }
}

int main()
{
    int nodes,edges,x,y;
    cin>>nodes>>edges;
    cnt=nodes;
    for(int i=0;i<edges;i++)
    {
        cin>>x>>y;
        x--;
        y--;
        adj[x].push_back(y);
        adj[y].push_back(x);
    }
    dfs(0);
    if(nodes==edges+1 && !cnt)
        cout<<"yes";
    else
        cout<<"no";
    return 0;
}
