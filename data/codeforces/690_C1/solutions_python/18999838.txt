#include <bits/stdc++.h>
using namespace std;
int n,m,cnt;
bool visited[1005];
vector <int> G[1005];
void dfs(int u,int p){
    cnt++;
    visited[u] = 1;
    for(int i=0;i<G[u].size();i++){
        int v = G[u][i];
        if (v == p) continue;
        if (visited[v]) {cout << "no"; exit(0);}
        dfs(v,u);
    }
}
int main(){
    cin >> n >> m;
    for(int i=1;i<=m;i++){
        int x,y;
        cin >>  x >> y;
        G[x].push_back(y);
        G[y].push_back(x);
    }
    dfs(1,1);
    if (cnt != n) cout << "no";
    else cout << "yes";
}
