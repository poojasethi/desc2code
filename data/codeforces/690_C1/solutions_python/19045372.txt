#include<bits/stdc++.h>
using namespace std;
int n,m,i,x,y,fa[1010];
int get(int x){return fa[x]?fa[x]=get(fa[x]):x;}
int main(){
	cin>>n>>m;
	if (m!=n-1){
		puts("no\n");
		return 0;
	}
	for (i=1;i<n;i++){
		cin>>x>>y;
		if (get(x)==get(y)){
			puts("no\n"); return 0;
		}
		fa[get(x)]=get(y);
	}
	puts("yes\n");
	return 0;
}
