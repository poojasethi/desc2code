#include<bits/stdc++.h>
using namespace std;
const int N=100100;
int n,m,root[N]={};
int get_root(int r)
{
	if(r!=root[r])
		root[r]=get_root(root[r]);
	return root[r];
}
int main()
{
	scanf("%d%d",&n,&m);
	if(m!=n-1)
	{
		puts("no");
		return 0;
	}
	for(int i=1;i<=n;++i)
		root[i]=i;
	for(int i=1,u,v;i<=m;++i)
	{
		scanf("%d%d",&u,&v);
		int ru=get_root(u),rv=get_root(v);
		if(ru==rv)
		{
			puts("no");
			return 0;
		}
		root[ru]=rv;
	}
	puts("yes");
}
