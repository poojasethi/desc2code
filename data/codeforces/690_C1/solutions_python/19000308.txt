#include <cstdio>
using namespace std;
int fa[1050];
int n,m,i;
int gf(int x)
{
	return fa[x]==x?x:fa[x]=gf(fa[x]);
}
int main()
{
	scanf("%d%d",&n,&m);
	if (m!=n-1){
		printf("no\n");
		return 0;
	}for (i=1;i<=n;i++) fa[i]=i;
	for (i=1;i<n;i++){
		int x,y;scanf("%d%d",&x,&y);
		int f1=gf(x),f2=gf(y);
		if (f1==f2){
			printf("no\n");
			return 0;
		}else fa[f1]=f2;
	}
	printf("yes\n");
	return 0;
}
