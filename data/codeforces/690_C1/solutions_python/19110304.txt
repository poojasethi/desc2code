#include <bits/stdc++.h>

using namespace std;

const int maxn = 1005;

int p[maxn];

int Find(int x){return x == p[x]?x:p[x] = Find(p[x]);}

int main(){
    int n,m,x,y;
    cin >> n >> m;
    bool ok = true;
    for(int i =1;i <= n;i++) p[i] = i;
    for(int i = 0;i < m;i++){
        scanf("%d %d",&x,&y);
        x = Find(x);
        y = Find(y);
        if(x == y) ok = false;
        else p[y] = x;
    }
    if(ok&&m == n-1) puts("yes");
    else puts("no");
}
