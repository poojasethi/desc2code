#include <bits/stdc++.h>
using namespace std;
int n,m;
bool vis[1050];
bool G[1050][1050];

void dfs(int x)
{
	vis[x]=true;
	for(int i=1;i<=n;i++)
	{
		if(G[x][i]&&!vis[i]){dfs(i);}
	}
	return ;
}
int main()
{
	while(~scanf("%d%d",&n,&m))
	{
		memset(vis,false,sizeof(vis));
		memset(G,false,sizeof(G));
		int x,y;
		for(int i=1;i<=m;i++)
		{
			scanf("%d%d",&x,&y);
			G[x][y]=G[y][x]=true;
		}
		if(n!=m+1)printf("no\n");
		else {
			dfs(1);
			bool ok=true;
			for(int i=1;i<=n;i++)
			{
				if(!vis[i]){ok=false;break;}
			}
			if(ok)printf("yes\n");
			else printf("no\n");
		}
	}
	return 0;
}
