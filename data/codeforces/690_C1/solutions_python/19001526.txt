#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <vector>
using namespace std;
vector<int>son[2000];
int n,m;
int cnt;
int vis[2000];
void dfs(int x){
	cnt++;
	vis[x]=1;
	for (int i=0;i<son[x].size();i++)
		if (!vis[son[x][i]])
			dfs(son[x][i]);
}
int main(int argc, char const *argv[])
{
	memset(vis,0,sizeof(vis));
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++) son[i].clear();
	for (int i=0;i<m;i++){
		int u,v;
		scanf("%d%d",&u,&v);
		son[u].push_back(v);
		son[v].push_back(u);
	}
	if (m!=n-1)
		printf("no\n");
	else{
		cnt=0;
		dfs(1);
		if (cnt!=n) 
			printf("no\n");
		else printf("yes\n");
	}
	return 0;
}
