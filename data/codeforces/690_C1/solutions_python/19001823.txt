#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
int n, m;
vector <int> E[1010];
int vis[10010];
void dfs( int u ) {
	for ( int i = 0; i < (int)E[u].size(); i ++ ) {
		int v = E[u][i];
		if ( vis[v] ) continue;
		vis[v] = true;
		dfs( v );
	}
}
int main() {
	//freopen( "A.in", "r", stdin );
	scanf( "%d%d", &n, &m );
	for ( int i = 1; i <= m; i ++ )	{
		int x, y;
		scanf( "%d%d", &x, &y );
		E[x].push_back( y );
		E[y].push_back( x );
	}
	int ret = 0;	
	dfs( 1 );
	for ( int i = 1; i <= n; i ++ ) ret += vis[i];

	if ( ret == n && m == n - 1 ) {
		puts( "yes" );
	} else {
		puts( "no" );
	}
	return 0;	
}
