#include <iostream>
#include <string>
#include <map>
#include <algorithm>
#include <vector>
using namespace std;
map<string,int> score;
int main(){
	int n;
	string me,x,y,ll,l;
	cin>>me;
	cin>>n;
	while (n--){
		cin>>x>>l;
		if (l[0]!='l') cin>>ll;
		cin>>y>>ll;
		y=y.substr(0,y.size()-2);
		if (x==me||y==me){
			if (l[0]=='p'){
				score[x]-=15;
				score[y]-=15;
			}
			if (l[0]=='c'){
				score[x]-=10;
				score[y]-=10;
			}
			if (l[0]=='l'){
				score[x]-=5;
				score[y]-=5;
			}
		}
		else{
			score[x]+=0;
			score[y]+=0;
		}
	}
	vector<pair<int,string> > p;
	for(map<string,int>::iterator it=score.begin();it!=score.end();it++){
		if(it->first==me) continue;
		p.push_back(make_pair(it->second,it->first));
	}
	sort(p.begin(),p.end());
	for(int i=0;i<(int)p.size();i++)
		cout<<p[i].second<<endl;
	return 0;
}
