#include <iostream>
#include <string>
#include <map>
#include <algorithm>
#include <vector>
using namespace std;
map<string,int> score;
int main(){
	string me;
	string x,y,garbage,cmd;
	int n;
	cin>>me;
	cin>>n;
	while(n--){
		cin>>x>>cmd;
		if (cmd[0]!='l') cin>>garbage;
		cin>>y;
		y=y.substr(0,y.size()-2);
		cin>>garbage;
		if(x==me||y==me){
			score[x]-=(cmd[0]=='p'?15:cmd[0]=='c'?10:5);
			score[y]-=(cmd[0]=='p'?15:cmd[0]=='c'?10:5);
		}else score[x]+=0,score[y]+=0;
	}
	vector<pair<int,string> > p;
	for(map<string,int>::iterator it=score.begin();it!=score.end();it++) {
		if(it->first==me) continue;
		p.push_back(make_pair(it->second,it->first));
	}
	sort(p.begin(),p.end());
	for(int i=0;i<(int)p.size();i++)
		cout<<p[i].second<<endl;
	return 0;
}
