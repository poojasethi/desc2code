#include <cstdio>
#include <string>
#include <map>
#include <iostream>
#include <vector>
#include <utility>
#include <cstring>
#include <algorithm>
using namespace std;

int main()
{
    string name,t1,t2,str,op;
    map <string,int> mp;
    int n,val;
    cin>>name>>n;
    while(n--)
    {
        cin>>t1>>op;
        if(op=="likes") val=5;
        else if(op=="commented") val=10;
        else val=15;
        if(op!="likes") cin>>str;
        cin>>t2>>str;
        t2.resize(t2.size()-2);
        if(t1!=name && t2!=name) val=0;
        if(t2!=name) 
        {
            if(mp.find(t2)==mp.end()) mp[t2]=-1*val;
            else mp[t2]-=val;
        }
        if(t1!=name) 
        {
            if(mp.find(t1)==mp.end()) mp[t1]=-1*val;
            else mp[t1]-=val;
        }
    }
    vector <pair <int,string> > res;
    for(map<string,int>::iterator it=mp.begin();it!=mp.end();it++)
    {
        res.push_back(make_pair((*it).second,(*it).first));
    }
    sort(res.begin(),res.end());
    for(int i=0;i<res.size();i++)
    {
        cout<<res[i].second<<endl;
    }
    return 0;
}
