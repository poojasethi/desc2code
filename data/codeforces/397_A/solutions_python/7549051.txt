num_students = int(raw_input())
coord_array = [0 for i in range(0, 100)] #List of segments
alex_tokens = raw_input().split(' ')
alex_left = int(alex_tokens[0])
alex_right = int(alex_tokens[1])
for i in range(alex_left, alex_right):
	coord_array[i] = 1
for i in range(1, num_students):
	tokens = raw_input().split(' ')
	left = int(tokens[0])
	right = int(tokens[1])
	for j in range(left, right):
		coord_array[j] = 0

print sum(coord_array)
