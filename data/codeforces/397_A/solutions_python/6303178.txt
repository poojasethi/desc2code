#!/usr/bin/env python


n = int(raw_input())

l1, r1 = [int(_) for _ in raw_input().split()]

a = [0]*100

for i in range(1, n):
    l, r = [int(_) for _ in raw_input().split()]
    for j in range(l, r):
        a[j] = 1

ans = 0
for i in range(l1, r1):
    if a[i] == 0:
        ans += 1


print ans
