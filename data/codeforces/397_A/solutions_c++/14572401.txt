#include <iostream>
using namespace std;

bool chck[100 + 10];

int main()
{
	int n;
	cin>> n;

	int x, y;
	cin>> x >> y;

	for(int i = 0; i < n - 1; i ++)
	{
		int a, b;
		cin>> a >> b;
		for(int i = a; i < b; i ++)
			chck[i] = true;
	}

	int ans = 0;
	for(int i = x; i < y; i ++)
		if(!chck[i])
			ans ++;

	cout<< ans << endl;
}
