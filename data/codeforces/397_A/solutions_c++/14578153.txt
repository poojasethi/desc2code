#include<iostream>

using namespace std;

bool check[101];

int main(){
	int n, m, f ,g , s = 0, a, b;
	cin >> n;	
	cin >> f >> g;
	for(int i = 1; i < n; i++)
		{
			cin >> a >> b;
			for(int j  =a; j < b; j++)
				if( j >= f && j < g && check[j] == false )
					{
						check[j] = true;
						s++;
					}
		}
	cout << g - f - s;
}
