#include <cstdio>

int main() {
	int n, l, r;
	int A[100] = {0};
	scanf("%d", &n);
	scanf("%d %d", &l, &r);
	for (int i = l; i < r; i++) {
		A[i] = 1;
	}
	while (--n > 0) {
		scanf("%d %d", &l, &r);
		for (int i = l; i < r; i++) {
			A[i] = 0;
		}
	}
	for (int i = 0; i < 100; i++) {
		n += A[i];
	}
	printf("%d\n", n);
	return 0;
}
