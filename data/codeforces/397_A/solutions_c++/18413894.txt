#include <bits/stdc++.h>
using namespace std;

int main()
{

int n,r,l,a[100],x,y;
cin>>n;
cin>>x>>y;
for(int i=x ; i<y ; i++)
{
    a[i]=1;
}

for(int i=0 ; i<n-1 ; i++)
{
    cin>>r>>l;
    for(int j=r ; j<l ; j++)
    {
        a[j]=0;
    }
}
int res=0;
for(int i=x ; i<y ; i++)
{
    res+=a[i];
}
cout<<res;

return 0;

}
