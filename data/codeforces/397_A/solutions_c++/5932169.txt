#include<iostream>
#define For(i,s,e) for(__typeof(s)i=(s);i<(e);i++)
using namespace std;
int n,li,ri,x,y,res;
bool v[105];
int main()
{
	cin>>n;
	cin>>x>>y;
	For(i,0,101) v[i]=true;
	For(i,1,n)
		{
			cin>>li>>ri;
			For(j,li,ri)
				v[j]=false;
		}
	For(i,x,y)
		if(v[i]) res++;
	 cout<<res;
     return 0;
}
