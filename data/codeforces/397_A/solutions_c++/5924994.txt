
#include <iostream>
#include <algorithm>

using namespace std;

bool segment[110];

int main()
{

	int n, maxi = 0, a, b, t1, t2;

	cin>>n>>a>>b;
	for (int i = 1; i < n; i++) 
	{

		cin>>t1>>t2;

		for (int j = t1; j < t2; j++) segment[j] = 1; 

	}

	for (int i = a; i < b; i++) if (!segment[i]) ++maxi;

	cout<<maxi;

	return 0;

}
