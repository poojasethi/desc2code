#include<stdio.h>
#include<string.h>
int main()
{
	int n,l[105],a,b,i,sum=0;
	scanf("%d",&n);
		scanf("%d %d",&a,&b);
		memset(l,0,sizeof(l));
		for(i=a;i<b;i++)l[i]=1;
		for(i=0;i<n-1;i++)
		{
			scanf("%d %d",&a,&b);
			for(int j=a;j<b;j++)l[j]=0;
		}
		for(i=0;i<105;i++)sum+=l[i];
		printf("%d\n",sum);
	return 0;
}
