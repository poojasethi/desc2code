#include<iostream>
using namespace std;

bool p[101];
int n,l,r,ans;
int main()
{
    cin>>n>>l>>r;
    int a,b;
    for (int i=1;i<n;i++)
    {
       cin>>a>>b;
       for (int j=a;j<b;j++)
         p[j]=1;
    }
    for (int i=l;i<r;i++)
       if(!p[i])
         ans++;
    cout<<ans<<endl;
    return 0;
}
