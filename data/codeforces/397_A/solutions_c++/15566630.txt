#include <stdio.h>
#include <string.h>
int main(){
	int n,a,b,ans,num[101]={0};
	scanf("%d%d%d",&n,&a,&b);
	ans=0;
	for(int i=a;i<b;i++)num[i]=1;
	for(int i=1;i<n;i++){
		scanf("%d%d",&a,&b);
		for(int j=a;j<b;j++)num[j]=0;
	}
	for(int i=0;i<100;i++)if(num[i]==1)ans++;
	printf("%d\n",ans);
	return 0;
}
