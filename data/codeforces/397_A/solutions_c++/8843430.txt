#include<iostream>

using namespace std;

int a[105];

int main()
{
	int n;
	cin >> n;
	int l, r;
	cin >> l >> r;
	for(int i = 1; i < n; i++)
	{
		int l, r;
		cin >> l >> r;
		for(int i = l; i < r; i++)
			a[i] = 1;
	}
	int ans = 0;
	for(int i = l; i < r; i++)
		if(!a[i])
			ans++;
	cout << ans << endl;
	return 0;
}
