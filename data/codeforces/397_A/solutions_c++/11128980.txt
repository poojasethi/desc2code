#include<bits/stdc++.h>
using namespace std;
bool arr[101];
int main()
{
    int n,a,b,res=0;cin>>n>>a>>b;
    int arr[101]={0};
    for(int i=1;i<n;i++)
    {
        int x,y;
        cin>>x>>y;
        for(int l=x;l<y;l++)
            if(l<b)
                arr[l]=1;
    }
    for(int i=a;i<b;i++)
        if(!arr[i])res++;
    cout<<res;
}
