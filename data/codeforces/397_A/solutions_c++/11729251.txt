#include <iostream>

using namespace std;

int main()
{
    int ans=0,x,y,i,k,n,a,b,r[101];
    for(i=0;i<101;i++) r[i]=1;

    cin >> n >> x >> y;
    for(i=0;i<n-1;i++)
    {
        cin >> a >> b;
        for(k=a;k<b;k++) r[k]=0;
    }



    for(i=x;i<y;i++) ans += r[i];
    cout<<ans;

    return 0;
}
