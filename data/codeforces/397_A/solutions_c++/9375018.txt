#include <iostream>
#include <vector>
using namespace std;
int main(){
	int n;
	cin>>n;
	int al,ar;
	cin>>al>>ar;
	vector<int> cen(101,0);
	for(int i=0;i<n-1;++i){
		int l,r;
		cin>>l>>r;
		for(int i=l;i<r;++i)
			cen[i]=1;
	}
	int res=0;
	for(int i=al;i<ar;++i) if(!cen[i])++res;
	cout<<res;
	return 0;
}
