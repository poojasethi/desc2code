#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n, x, y;
	cin >> n;

	int mark[101] = {0};

	for (int i = 1; i <= n; ++i) {
	    	cin >> x >> y;

	    	for (int j = x; j < y; ++j) {
	    	    	mark[j] = i;
	    	}
	}

	int cnt = 0;
	for (int i = 0; i < 100; ++i) {
	    	if (mark[i] == 1) cnt++;
	}

	cout << cnt;

    return 0;
}
