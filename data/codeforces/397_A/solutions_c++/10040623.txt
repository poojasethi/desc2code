#include <bits/stdc++.h>
using namespace std;
long long a[1001];
int main(){
    long long i,j,n,l,r,o,p,k=0;
    cin>>n>>l>>r;
    for(i=2;i<=n;i++)
    {
        cin>>o>>p;
      for(j=o;j<p;j++)
        a[j]=1;
    }
    for(i=l;i<r;i++)
        if(a[i]==0)
        k++;
    cout<<k;
    return 0;
}
