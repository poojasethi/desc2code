#include<bits/stdc++.h>
using namespace std;
int v[100+7];
int main(){
    int n,a,b,l,r;
    cin>>n;
    cin>>a>>b;
    for(int i=0;i<n-1;i++){
        cin>>l>>r;
        for(int j=l;j<r;j++)v[j]++;
    }
    int sol = 0;
    for(int i=a;i<b;i++){
        if(v[i]==0)sol++;
    }
    cout<<sol<<endl;
    return 0;
}
