#include <iostream>
using namespace std;
int main() {
    int n, i, a, b, x, y, num = 0, j;
    int seg[100] = {0};
    cin >> n >> a >> b;
    for(i = 1; i < n; i++)
    {
        cin >> x >> y;
        for(j = x; j < y; j++)
        seg[j] = 1;
    }
    for(i = a; i < b; i++)
        if (seg[i] == 0) num++;
    cout << num;
    return 0;
}
