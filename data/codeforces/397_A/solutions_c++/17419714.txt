#include<bits/stdc++.h>
using namespace std;

int vis[105] = {0};

int main() {
	int n, sl, sr, ans = 0;
	cin >> n;
	cin >> sl >> sr;
	for(int i = 0; i < n-1; i++) {
		int l, r;
		scanf("%d%d", &l, &r);
		for(int i = l+1; i <= r; i++) vis[i] = 1;
	}
	for(int i = sl+1; i <= sr; i++) if(!vis[i]) ans++;
	cout << ans;
}
