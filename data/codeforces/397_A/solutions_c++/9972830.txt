#include <cstdio>

bool rope[128];

int main() {
    int n; scanf("%d",&n);
    
    int al, ar, l, r;
    scanf("%d%d",&al,&ar);
    for(int i=1; i<n; i++) {
        scanf("%d%d",&l,&r);
        for(int i=l; i<r; i++) rope[i] = 1;
    }
    
    int k=0;
    for(int i=al; i<ar; i++) if(!rope[i]) k++;
    printf("%d\n",k);
    
    return 0;
}
