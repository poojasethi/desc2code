#include<iostream>
using namespace std;
int mark[101];
int main()
{
    int n,l,r,nowl,nowr,ans=0;
    cin>>n>>l>>r;
    for(int i=1;i<n;i++)
    {
        cin>>nowl>>nowr;
        for(int j=nowl;j<nowr;j++)
                mark[j]=true;
    }
    for(int i=l;i<r;i++)
        if(!mark[i])
            ans++;
    cout<<ans;
}
