#include <iostream>
using namespace std;
int arr[103];
int main() {
    int n,b,e;cin>>n;
    n--;cin>>b>>e;for(int i=b;i<e;i++)arr[i]++;
    int bb,ee;
    for(int i=0;i<n;i++)
    {
        cin>>bb>>ee;for(int j=bb;j<ee;j++)arr[j]++;
    }
    int sol=0;
    for(int i=b;i<e;i++)if(arr[i]==1)sol++;
    cout<<sol<<endl;
    return 0;
}
