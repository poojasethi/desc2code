#include<cstdio>
int x[105];
int main()
{
    int n;scanf("%d",&n);
    int l,r;
    scanf("%d%d",&l,&r);
    for(int i=l;i<r;i++)x[i]=1;
    for(int i=2;i<=n;i++){
      int s,t;scanf("%d%d",&s,&t);
      for(int i=s;i<t;i++)x[i]=0;
    }
    int ans=0;
    for(int i=l;i<r;i++) ans+=x[i];
    printf("%d\n",ans);
}
