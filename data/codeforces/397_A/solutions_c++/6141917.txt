#include<iostream>
using namespace std;

bool seg[105];
int main(){
 int i,j,n,left,right,p,q;
 cin>>n;
 cin>>left>>right;
 for(i=left+1;i<=right;++i) seg[i]=true; 
 for(i=1;i<n;++i){
  cin>>p>>q;
  for(j=p+1;j<=q;++j) seg[j] = false;
 }
 int sum=0;
 for(i=left+1;i<=right;++i)
  if(seg[i]) ++sum;
 cout<<sum<<endl;
 return 0;
}
