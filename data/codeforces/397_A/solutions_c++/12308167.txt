#include<bits/stdc++.h>
using namespace std;
int x[100],y[100],ans[1001],cnt=0;
int main(){
int n;
cin>>n;
for(int i=0;i<n;i++){
    cin>>x[i]>>y[i];
}
for(int i=1;i<n;i++){
    for(int v=x[i];v<y[i];v++){
        ans[v]=1;
    }
}
for(int i=x[0];i<y[0];i++){
    if(ans[i]==0){
        cnt++;
    }
}
cout<<cnt<<endl;
}
