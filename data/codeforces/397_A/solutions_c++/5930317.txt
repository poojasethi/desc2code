#include <cstdio>

int n, a, b, ans; char g[123];
int main() {
    scanf("%d %d %d", &n, &a, &b);
    for (int i = a; i < b; i++) g[i] = 1;
    for (int i = 1; i < n; i++) {
        scanf("%d %d", &a, &b);
        for (int j = a; j < b; j++) g[j] = 0;
    }
    for (int i = 0; i < 123; i++) ans += g[i];
    printf("%d\n", ans);

    return 0;
}
