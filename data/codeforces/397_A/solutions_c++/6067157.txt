#include<iostream>
#include<cstring>
using namespace std;

int main()
{
	int n;
	cin>>n;
	bool a[101];
	memset(a,false,sizeof(a));
	int a1,a2,s1,s2;
	cin>>a1>>a2;
	for(int i=1;i<n;i++)
	{
		cin>>s1>>s2;
		for(int j=s1;j<s2;j++)
			a[j]=true;
	}
	int res=0;
	for(int i=a1;i<a2;i++)
		if(a[i]==false)
			res++;
	cout<<res<<endl;
	return 0;
}
