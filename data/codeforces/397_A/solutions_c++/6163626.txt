#include <iostream>

using namespace std;

int uso[200];

int main()
{
  int n;
  cin>>n;
  int l,r;
  cin>>l>>r;
  for (int i=l;i<r;i++)
    uso[i]=1;
  for (int j=2;j<=n;j++) {
    cin>>l>>r;
    for (int i=l;i<r;i++)
      uso[i]=0;
  }
  int c=0;
  for (int i=0;i<200;i++)
    c+=uso[i];
  cout<<c<<endl;
}
