#include<iostream>
using namespace std;
int main()
{
	int n,t,i,j,count=0,arr1[102]={0};
	cin>>n;
	int arr[n][2];
	cin>>arr[0][0]>>arr[0][1];
	for(i=1;i<n;i++)
	{
		cin>>arr[i][0]>>arr[i][1];
		for(j=arr[i][0];j<arr[i][1];j++)
		arr1[j]=1111;
	}
	for(i=arr[0][0];i<arr[0][1];i++)
	{
		if(arr1[i]==0)
		count++;
	}
	cout<<count;
}
