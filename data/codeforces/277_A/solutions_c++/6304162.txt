#include<stdio.h>

#define FOR(aa,bb,cc) for(int aa=bb ; aa<=cc ; aa++)

int n,m,k;
int arr[104][104];
int H[104];

void f(int x){

	FOR(i,1,n){
		FOR(j,1,arr[x][0])
			FOR(z,1,arr[i][0])
				if(!H[i] && arr[i][z]==arr[x][j]){
					H[i]=1;
					f(i);
				}

	}

}

int main(){

	int s=0,a=0;

	scanf("%d %d",&n,&m);

	FOR(i,1,n){
		scanf("%d",&k);
		if(!k){
			s++;
			continue;
		}
		a=1;
		arr[i][0]=k;
		FOR(j,1,k){
			scanf("%d",&arr[i][j]);
		}
	}

	FOR(i,1,n){
		if(!H[i] && arr[i][0]){
			H[i]=1;
			f(i);
			s++;
		}
	}
	if(!a)
		s++;
	printf("%d",s-1);

	return 0;
}
