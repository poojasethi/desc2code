#include<bits/stdc++.h>
using namespace std;
vector<int> g[222];
bool vis[222];
void dfs(int i) {
vis[i]=true; {
for(int k=0;k<g[i].size();k++)
    if(!vis[g[i][k]])
    dfs(g[i][k]);
}
}
int main() {
int k,n,m,a,r=0,t=0;
scanf("%d%d",&n,&m);
for(int i=1;i<=n;i++) {
    scanf("%d",&k);
    if(k>0) t=1;
        for(int j=0;j<k;j++) {
        scanf("%d",&a);
        g[i].push_back(n+a);
        g[n+a].push_back(i);
        }
}
for(int i=1;i<=n;i++) {
if(!vis[i]) {r++; dfs(i);}
}
cout<<r-t;
return 0;
}
