#include <cstdio>
#include<vector>
using namespace std;
vector <int> a[200];
bool used[200];
int m,n,r,tr;
void dfs(int i){
	used[i]=true;
	for (int j=0;j<a[i].size();j++){
		if (!used[a[i][j]]) dfs(a[i][j]);
	}
}
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	for (int i=0;i<n;i++){
		int k,x;
		scanf("%d",&k);
		if (k>0) tr=1;
		for (int j=0;j<k;j++){
			scanf("%d",&x);
			a[i].push_back(n+x);
			a[n+x].push_back(i);
		}
	}
	for (int i=0;i<n;i++){
		if (!used[i]) {
			r++;
			dfs(i);
		}
	}
	r-=tr;
	printf("%d",r);
	return 0;//silly cross
}
