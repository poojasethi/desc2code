#include<cstdio>
#include<cstring>
using namespace std;
int f[200];bool b1[100];
int fd(int x)
{
	if(!~f[x])return x;
	return f[x]=fd(f[x]);
}
int main()
{
	int n,m,a,b,ans;bool flag=1;
	scanf("%d%d",&n,&m);
	memset(f,-1,sizeof(f));
	for(int i=0;i<n;++i)
	{
		scanf("%d",&a);
		if(a)flag=0;
		while(a--)
		{
			scanf("%d",&b);
			b=fd(b+n-1);
			if(i!=b)f[b]=i;
		}
	}
	ans=flag?1:0;
	for(int i=0;i<n;++i)
	{
		a=fd(i);
		if(!b1[a])
		{b1[a]=1;++ans;}
	}
	printf("%d\n",ans-1);
}
