#include<cstdio>
using namespace std;
int fd[201];
bool b1[201];
int f(int x)
{return fd[x]?fd[x]=f(fd[x]):x;}
int main()
{
	int n,m,a,b,o=0;bool bl=0;
	scanf("%d%d",&n,&m);
	n+=m;
	for(int i=m+1;i<=n;++i)
	{
		scanf("%d",&a);
		if(a)bl=1;
		while(a--)
		{
			scanf("%d",&b);
			b=f(b);
			if(b!=i)fd[b]=i;
		}
	}
	for(int i=m+1;i<=n;++i)
	{
		a=f(i);
		if(!b1[a])
		{b1[a]=1;++o;}
	}
	printf("%d\n",o-bl);
}
