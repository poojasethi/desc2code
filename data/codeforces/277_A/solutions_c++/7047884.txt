#include<bits/stdc++.h>
using namespace std;
bool adj[110][110]={{0}},v[110]={0};
int n,m;
void dfs(int p)
{
	int i,j;
	v[p]=1;
	for(i=0;i<n;i++)
		if(!v[i])
			for(j=0;j<m;j++)
				if(adj[i][j]==1 && adj[p][j]==1)
					dfs(i);
}
int main()
{
	int i,j,o,num,ans;
	bool flag=0;
	cin>>n>>m;
	for(i=0;i<n;i++)
	{
		cin>>o;
		if(o)
			flag=1;
		for(j=0;j<o;j++)
		{
			cin>>num;
			adj[i][num-1]=1;
		}
	}
	(flag) ? ans=0 : ans=1;
	for(i=0;i<n;i++)
	{
		if(!v[i])
		{
			dfs(i);
			ans++;
		}
	}
	cout<<ans-1<<endl;
	return 0;
}
