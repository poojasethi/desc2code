#include <iostream>
using namespace std;

#define rep(i,n) for (int i = 1; i <= int(n); ++i)

int a[201], b[201][201], n, m, z, k, x;

void f(int i) {
    if (!a[i]) {
        a[i] = 1;
        rep (j,n+m) if (b[i][j]) f(j);
    }
}

int main() {
    cin >> n >> m;
    rep (i,n) {
        cin >> k;
        if (k == 0) ++z;
        rep (j,k) cin >> x, b[i][n+x] = b[n+x][i] = 1;
    }
    if (z == n)
        cout << n << endl;
    else {
        z = -1;
        rep (i,n) if (!a[i]) ++z, f(i);
        cout << z << endl;
    }
}
