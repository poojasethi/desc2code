//In the Name of God

#include<iostream>
#include<vector>

using namespace std;

int n,m,T,cnt;
vector<int> g[555];
bool vis[555];

void dfs(int v)
{
  vis[v]=true;
  for(int i=0;i<g[v].size();i++)
    if(!vis[g[v][i]])
      dfs(g[v][i]);
}

int main()
{
  cin>>n>>m;
  for(int x,l,i=1;i<=n;i++)
    {
      cin>>x;
      T+=!x;
      while(x--)
	{
	  cin>>l;
	  g[i].push_back(n+l);
	  g[n+l].push_back(i);
	}
    }
  for(int i=1;i<=n;i++)
    if(!vis[i])
      dfs(i),cnt++;
  cout<<cnt-(T!=n)<<endl;
  return 0;
}
