#include <bits/stdc++.h>
using namespace std;

int p[215], n, m;
int find(int x) { return x==p[x]?x:p[x]=find(p[x]); }
void un(int x, int y) {
	int fx=find(x), fy=find(y);
	if(fx==fy) return;
	p[fx]=fy;
}
int main() {
	scanf("%d%d", &n, &m);
	for(int i=1; i<=n+m; ++i) p[i]=i;
	int flag=1;
	for(int i=1; i<=n; ++i) {
		int k; scanf("%d", &k);
		for(int j=1; j<=k; ++j) {
			int y;
			scanf("%d", &y);
			un(i, n+y);
			flag=0;
		}
	}
	for(int i=1; i<=n; ++i) p[i]=find(p[i]);
	sort(p+1, p+1+n);
	printf("%d\n", (int)(unique(p+1, p+1+n)-p)-2+flag);
	return 0;
}
