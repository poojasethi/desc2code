#include <iostream>

using namespace std;

int n,m,c;
int adj[200][200];
int vis[200];

void dfs(int k) {
	vis[k] = 1;
	for (int i=0; i<n+m; i++)
		if (adj[i][k] && !vis[i]) 
			dfs(i);
}

int main() {
	int a,b; cin >> n >> m;
	int f = 0;
	c = 0;
	for (int i=0; i<n+m; i++) {
		vis[i] = 0;
		for (int j=0; j<n+m; j++)
			adj[i][j] = 0;
	}

	for (int i=0; i<n; i++) {
		cin >> a; f += a;
		for (int j=0; j<a; j++) {
			cin >> b;
			adj[i][n+b-1] = adj[n+b-1][i] = 1;
		}
	}

	for (int i=0; i<n; i++) {
		if (!vis[i]) {
			dfs(i);
			c++;
		}
	}

	cout << c-(f > 0 ? 1 : 0) << "\n";
}
