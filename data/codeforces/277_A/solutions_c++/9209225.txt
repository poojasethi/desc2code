#include<iostream>
using namespace std;

int n,m,know[110],root[110],v[110],ans;

int get_root(int x){return x==root[x]?x:get_root(root[x]);}

int main()
    {
    int t,a;
    cin>>n>>m;
    for(int i=1;i<=m;i++) root[i]=i;
    for(int i=1;i<=n;i++)
        {
        cin>>t;
        if(!t) ans++;
        for(int j=1;j<=t;j++) cin>>know[j],v[know[j]]=1;
        for(int j=1;j<t;j++) root[get_root(know[j])]=get_root(know[j+1]);
        }
    if(ans==n) ans++;
    else for(int i=1;i<=m;i++) if(v[i]&&root[i]==i) ans++;
    cout<<ans-1;
    return 0;
    }
