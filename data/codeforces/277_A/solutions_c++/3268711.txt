#include <iostream>
#include <vector>
#include <set>
using namespace std;

#define rep(i,n) for (int i = 1; i <= int(n); ++i)

int a[101], r[101], n, m, k, x, z, c;

int root(int i) {
    if (r[i] == i) return i;
    return r[i] = root(r[i]);
}

int main() {
    cin >> n >> m;
    rep (i,n) {
        cin >> k;
		if (k == 0) { ++z; continue; }
		rep (j,k) cin >> a[j], r[a[j]] == 0 ? r[a[j]] = a[j] : 0;
		rep (j,k) r[root(a[j])] = root(a[1]);
    }
	rep (i,m) if (r[i] == i) c++;
    cout << (z == n ? n : c + z - 1) << endl;
}
