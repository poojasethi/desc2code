#include<stdio.h>
int arr[105][105],h[105],n,m;

void f(int x){
	int j,k,t;

	for(j=1;j<=n;j++){
		if(x==j)
			continue;
		for(k=1;k<=m;k++)
			if(!h[j]&&arr[j][k]&&arr[x][k]){
				h[j]=1;
				f(j);
			}
	}
}

int main(){
	int i,j,k,x,s=0;
	scanf("%d %d",&n,&m);
	for(i=1;i<=n;i++){
		scanf("%d",&k);
		for(j=1;j<=k;j++){
			scanf("%d",&x);
			arr[i][x]=1;
		}
		if(k)s++;

	}

	if(!s)
		s=1;
	else 
		s=0;
	for(i=1;i<=n;i++){
		if(!h[i]){
			h[i]=1;
			f(i);
			s++;
		}
	}
	printf("%d",s-1);

	return 0;
}
