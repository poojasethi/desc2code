#include <cstdio>
#include <vector>
#include <iostream>
using namespace std;
int i,j,n,m,x,r,k;
vector <int> g[222];
bool u[222],q;
void dfs(int i)
{
    u[i]=true;
    for (int j=0;j<g[i].size();j++)
        if (!u[g[i][j]]) dfs(g[i][j]);
}
int main()
{
    cin>>n>>m;
    for (i=0;i<n;i++){
        cin>>k;
        if (k>0) q=true;
        for (j=0;j<k;j++){
            cin>>x;
            g[i].push_back(n+x);
            g[n+x].push_back(i);
        }
    }
    for (i=0;i<n;i++) if (!u[i]) {r++;dfs(i);}
        cout<<r-int(q);
    return 0;
}
