#include<iostream>
#include<set>
using namespace std;
int n,m,a[105][105]={0},i,j,l,k,p[105],z=0;
set<int> v;
int fs(int x)
{
if(x==p[x]) return x;
p[x]=fs(p[x]);
return p[x];
}
int main()
{
cin>>n>>m;
for(i=1;i<=n;i++)
{
cin>>l;
p[i]=i;
for(j=1;j<=l;j++) cin>>k,a[i][k]=1,z=k;
}
for(i=1;i<=n;i++)
{
for(j=1;j<=n;j++)
{
for(l=1;l<=m;l++)
{
if(a[i][l] && a[j][l]) p[fs(j)]=fs(i);
}
}
}
for(i=1;i<=n;i++)
{
v.insert(fs(i));
}
cout<<v.size()-1+(z==0);
return 0;
}
