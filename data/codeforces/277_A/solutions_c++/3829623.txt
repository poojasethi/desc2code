#include<iostream>
#include<cstdio>
#include<cmath>
#define rep(i,n) for((i)=1;(i)<=(n);i++)
using namespace std;

int n,m,i,k,ans,f[110],a[110];

int fa(int x)
{
    if (x!=f[x]) f[x]=fa(f[x]); return f[x];
}

int main()
{
    cin>>n>>m;
    rep(i,m) f[i]=i;
    rep(i,n) 
    {
      int u,v;
      cin>>k;
      if (!k) {ans++; continue;}
      cin>>u; a[u]=1;
      while (--k) {cin>>v; a[v]=1; f[fa(u)]=fa(v); u=v;}
    }
    k=0;
    rep(i,m) if (a[i]==1 && f[i]==i) k++;
    if (k) k--;
    cout<<ans+k<<endl;
}
