#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int fa[210],a[210],n,m,i,j,k,ans,v[210];

int get(int x)
{
	if(x==fa[x]) return x;
	return fa[x]=get(fa[x]);
}

int main()
{
	cin>>n>>m;
	for(i=1;i<=m;i++) fa[i]=i;
	for(i=1;i<=n;i++)
	{
		cin>>k;
		if(!k) ans++;
		for(j=1;j<=k;j++) cin>>a[j],v[a[j]]=1;
		for(j=1;j<k;j++) fa[get(a[j])]=get(a[j+1]);
	}
	if(ans==n) ans++;
	else
		for(j=1;j<=m;j++)
			if(v[j]&&get(j)==j) ans++;
	cout<<ans-1<<endl;
}
