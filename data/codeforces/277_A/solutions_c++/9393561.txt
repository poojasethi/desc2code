#include<iostream>
#include<vector>

#define PB push_back

using namespace std;

vector<int> v[205];
bool mark[205];

void DFS(int n)
{
	if(mark[n])
		return;
	mark[n] = true;
	for(int i = 0; i < v[n].size(); i++)
		DFS(v[n][i]);
}

int main()
{
	int n, m;
	cin >> n >> m;
	int ans = 1;
	for(int i = 0; i < n; i++)
	{
		int t;
		cin >> t;
		for(int j = 0; j < t; j++)
		{
			int x;
			cin >> x;
			v[i].PB(n - 1 + x);
			v[n - 1 + x].PB(i);
			if(ans)
				ans--;
		}
	}
	for(int i = 0; i < n; i++)
		if(!mark[i])
		{
			DFS(i);
			ans++;
		}
	cout << ans - 1 << endl;
	return 0;
}
