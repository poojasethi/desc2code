#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int n,m,a[210],v[210],r[210],ans,k;

int get(int x)
	{
	return r[x]==x?x:get(r[x]);
	}

int main()
	{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++) r[i]=i;
	for(int i=1;i<=n;i++)
		{
		scanf("%d",&k);
		if(!k) ans++;
		for(int j=1;j<=k;j++) cin>>a[j],v[a[j]]=1;
		for(int j=1;j<k;j++) r[get(a[j])]=get(a[j+1]);
		}
	if(ans==n) ans++;
	else for(int j=1;j<=m;j++) if(v[j]&&get(j)==j) ans++;
	printf("%d\n",ans-1);
	return 0;
	}
