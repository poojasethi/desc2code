#include<iostream>
#include<vector>
#include<cstring>
using namespace std;

vector<int> a[1001];
bool u[1001];

void dfs(int x)
{
	u[x]=true;
	for(int i=0;i<a[x].size();i++)
	  if(!u[a[x][i]]) dfs(a[x][i]);
}
int main()
{
	memset(u,false,sizeof(u));
	int x,y,z=0,p=0,n,m;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	{
		cin>>x;if(x!=0) z=1;
		for(int j=1;j<=x;j++)
		{
			cin>>y;
			a[i].push_back(n+y+1);
			a[n+y+1].push_back(i);
		}
	}
	for(int i=1;i<=n;i++)
		if(!u[i]) {dfs(i);p++;}
	cout<<p-z;
	return 0;
}
