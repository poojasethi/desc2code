#include<bits/stdc++.h>
using namespace std;
int fa[101],b[101],num=0;
int find(int x){
	return (fa[x]==x?fa[x]:fa[x]=find(fa[x]));
}
int main(){
	int n,m;
	while(cin>>n>>m){
		memset(b,0,sizeof(b));
		num=0;
		int flag=0;
		for(int i=1;i<=n;i++){
			fa[i]=i;
			int t;
			cin>>t;
			for(int j=0;j<t;j++){
				flag=1;
				int k;
				cin>>k;
				if(b[k]==0)
					b[k]=i;
				else 
					fa[find(b[k])]=i;
			}
		}
		for(int i=0;i<n;i++){
			if(fa[i]==i)
				num++;
		}
		cout<<num-flag<<endl;	
	}
	return 0;
}
