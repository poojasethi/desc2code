
n,m = map(int, raw_input().split())

langs = []
emps = [[] for _ in xrange(m)]

for emp in xrange(n):
    l = map(lambda s: int(s)-1 , raw_input().split())[1:]
    langs.append(l)
    for lang in l:
        emps[lang].append(emp)

visited = set()
def dfs(emp):
    visited.add(emp)
    for lang in langs[emp]:
        for oemp in emps[lang]:
            if oemp not in visited:
                dfs(oemp)


cost = 0
for emp in xrange(n):
    if emp not in visited:
        dfs(emp)
        cost += 1 

if all(len(ls) == 0 for ls in langs):
    print n
else:
    print cost -1
