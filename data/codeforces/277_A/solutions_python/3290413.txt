trees = []
flg = []
def dfs(p):
    flg[p] = True
    for item in trees[p]:
        if(flg[item] == False):
            dfs(item)
            
def solve(n):
    ans = 0
    global flg
    flg = [False for i in range(0, n + 1)]
    for item in range(1, n + 1):
        if(flg[item] == False):
            ans += 1
            dfs(item)
    return ans

def main():
    
    zero = 0
    langs = []
    n, m = map(int, raw_input().split())
    langs.append([])
    for i in range(1, n + 1):
        tmp = map(int, raw_input().split())
        del tmp[0]
        langs.append(tmp)
        if(langs[i] == list([])):
            zero += 1
            
    trees.append([])
    for i in range(1, n + 1):
        trees.append([j for j in range(1, n + 1) if(set(langs[i]) & set(langs[j]) != set([]))])
        
    if(zero == n):
        print n
    else:
        print solve(n) - 1
        
main()
