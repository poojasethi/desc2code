R=lambda:map(int,raw_input().split())
n,m=R()
G=range(n)
r=map(int,G)
def rt(x):
  if x!=r[x]:r[x]=rt(r[x])
  return r[x]
e=[0]*n
for i in G:e[i]=set(R()[1:])
for i in G:
  for j in G[:i]:
    u=rt(j)
    if e[u].intersection(e[i]):
      r[u]=i
      e[i]=e[i].union(e[u])
print sum(1 for i in G if i==rt(i) or not e[i])-(sum(map(len,e))>0)
