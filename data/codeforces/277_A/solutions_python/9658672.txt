from sys import stdin, exit
from collections import defaultdict

def readNums():
    return [int(x) for x in stdin.readline().split()]

n_employees, n_languages = readNums()

any_know_any = False
# Construct a biparte graph. Employees are represented as (n, True), languages as (n, False)
graph = defaultdict(set)
for employee in range(1, n_employees+1):
    info = readNums()
    if len(info) > 1:
        any_know_any = True
    for language in info[1:]:
        graph[(employee, True)].add((language, False))
        graph[(language, False)].add((employee, True))

if not any_know_any:
    print n_employees
    exit(0)

def dfs_color(node, visited):
    # add all nodes connected to node to visited
    visited.add(node)
    for child in graph[node]:
        if child not in visited:
            dfs_color(child, visited)

visited = set()
connected_components = 0
for employee in xrange(1, n_employees+1):
    if (employee, True) not in visited:
        connected_components += 1
        dfs_color((employee, True), visited)

print connected_components - 1
