import pdb
R = lambda: map(int, raw_input().split())

n, m = R()
Map = []
status = [0 for i in xrange(n)]


def dfs(index, l):
    stack = []
    status[index] = 1
    stack.append(l)
    while len(stack):
        temp = stack.pop()
        for i in xrange(len(Map)):
            if status[i] == 0:
                for j in temp:
                    if j in Map[i]:
                        stack.append(Map[i])
                        status[i] = 1
                        break


for i in xrange(n):
    templ = R()
    Map.append(templ[1:len(templ)])

flag=False
for item in Map:
    if len(item):
        flag =True
        break
ans = 0
if flag == False:
    ans = 1
index = status.index(0)
dfs(index, Map[index])
while 0 in status:
    ans += 1
    index = status.index(0)
    dfs(index, Map[index])

print ans
