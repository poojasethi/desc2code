graph_emp={}
graph_lang={}
employers,lang=map(int,raw_input().split())
null=0
for i in range(employers):
    nl_l=map(int,raw_input().split())
    if nl_l[0]==0:
        null+=1
    graph_emp[i]=nl_l[1:]
    for j in nl_l[1:]:
        if not j in graph_lang:
            graph_lang[j]=[i]
        else:
            graph_lang[j].append(i)
    
visited = set()
def dfs(emp):
    visited.add(emp)
    for lang in graph_emp[emp]:
        for d_emp in graph_lang[lang]:
            if d_emp not in visited:
                dfs(d_emp)


cost = 0
for emp in range(employers):
    if emp not in visited:
        dfs(emp)
        cost += 1 

if null==employers:
    print null
else:
    print cost -1
