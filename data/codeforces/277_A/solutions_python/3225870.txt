n, m = map(int, raw_input().split())
r = [[] for _ in range(n)]
c = [[] for _ in range(m)]
for i in range(n):
	for j in map(int, raw_input().split())[1 : ]:
		r[i].append(j - 1)
		c[j - 1].append(i)
u = [0 for _ in range(n)]
v = [0 for _ in range(m)]

def R(x):
	if not u[x]:
		u[x] = 1
		for y in r[x]:
			C(y)

def C(x):
	if not v[x]:
		v[x] = 1
		for y in c[x]:
			R(y)

p = -1
for i in range(n):
	if not u[i]:
		p += 1
		R(i)
print [n, p][sum(len(x) for x in r) > 0]
