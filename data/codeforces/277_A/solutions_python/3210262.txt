n, m = map(int, raw_input().split())
par = range(n+m)
def find(a):
    if a == par[a]:
        return a
    else:
        par[a] = find(par[a])
        return par[a]
def merge(a, b):
    a = find(a)
    b = find(b)
    if a != b: par[b] = a
f = False
for i in xrange(n):
    a = map(int, raw_input().split())[1:]
    for b in a:
        merge(b-1, m+i)
    if len(a): f = True
s = set()
for x in par[m:]:
    s.add(find(x))
if f:
    print len(s) - 1
else:
    print len(s)
