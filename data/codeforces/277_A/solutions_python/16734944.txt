n,m = map(int,raw_input().split())
G = [[] for x in xrange(n+m)]
flag = True
for i in range(n):
	G[i].extend(list(map(int,raw_input().split()))[1:])
	G[i] = [x-1+n for x in G[i]]
	for x in G[i]:
		G[x].append(i)
	if flag and len(G[i]): flag = False

visited = [False for x in range(n+m)]
cc = 0
for i in range(n):
	if visited[i]: continue
	s = [i]
	while s:
		v = s.pop()
		visited[v] = True
		for u in G[v]:
			if not visited[u]:
				s.append(u)
	cc += 1

if flag: print n
else: print cc - 1
