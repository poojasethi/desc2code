s = list(raw_input().strip())
num = [0]*10
for i in s:
    num[int(i)]+=1
for i in range(len(a)):
    maximum = 0
    for j in range(9,-1,-1):
        if num[j]!=0:
            maximum=j
            break;
    if a[i]<str(maximum):
        a[i]=str(maximum)
        num[maximum]-=1
