import sys
a = raw_input()
c = raw_input()
la = len(a)
lb = len(c)
b = []
for i in range(0,lb):
	b.append(c[i])	
b.sort()
b = b[::-1]
th = 0
d = []
for i in range(0,la):
	if (th >= lb):
		d.append(a[i])
		continue
	if (a[i] >= b[th]):
		d.append(a[i])
		continue
	d.append(b[th])
	th = th+1
for i in d:
	sys.stdout.write(i)
print
