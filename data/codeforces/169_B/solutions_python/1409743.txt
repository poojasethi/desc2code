from collections import Counter
import sys

a = raw_input()
s = raw_input()

cnt = Counter()
for c in s:
    cnt[c] += 1

if '0' in cnt:
    del cnt['0']

for n in a:
    if len(cnt) == 0:
        sys.stdout.write(n)
        continue
    m = max(cnt)
    if m > n:
        sys.stdout.write(m)
        cnt[m] -= 1
        if not cnt[m]:
            del cnt[m]
    else:
        sys.stdout.write(n)
