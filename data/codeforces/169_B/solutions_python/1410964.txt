import sys
a = map(int, raw_input())
b = map(int, raw_input())
b.sort(reverse = -1)
#print b
lengthb = len(b)
lengtha = len(a)
ia = 0
ib = 0
while ia < lengtha and ib < lengthb:
    if b[ib] > a[ia]:
        a[ia] = b[ib]
        ia += 1
        ib += 1
    else:
        ia += 1
for each in a: sys.stdout.write(str(each))
