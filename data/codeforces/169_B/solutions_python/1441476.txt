a = raw_input()
s = raw_input()

a = list(a)
s = sorted(s, reverse=True)

a_index = 0
s_index = 0
a_len  = len(a)
s_len  = len(s)

while s_index < s_len:
    while a_index < a_len:
        if a[a_index] < s[s_index]:
            a[a_index] = s[s_index]
            a_index += 1
            s_index += 1
            break
        else :
            a_index += 1
    if a_index == a_len:
        break

print "".join(a)
