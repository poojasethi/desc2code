#!/usr/bin/env python
a=map(int,list(raw_input()))
b=raw_input()
l=[0]*10
for i in b:
    l[9-int(i)]+=1
c=''
for i in range(len(a)):
    if a[i]==9:
        continue
    else:
        for j in range(10):
            if l[j]>0 and (9-j)>a[i]:
                a[i]=9-j
                l[j]-=1
for i in a:
    c+=str(i)
print c
