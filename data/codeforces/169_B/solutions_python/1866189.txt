import sys

file = sys.stdin
#file = open("test", "r")

p1 = map(int, list(file.readline().rstrip()))
p2 = map(int, list(file.readline().rstrip()))
p2.sort()
p2.reverse()

index2 = 0
for i1 in range(len(p1)):
    if p2[index2] > p1[i1]:
        p1[i1] = p2[index2]
        if index2 == len(p2) - 1:
            break
        else:
            index2 += 1

print "".join(map(str, p1))
