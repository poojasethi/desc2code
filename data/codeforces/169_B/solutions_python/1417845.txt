a = raw_input()
s = raw_input()
m = {}
for c in s:
    if not int(c) in m:
        m[int(c)] = 0
    m[int(c)] += 1
ret = ""
for i in range(len(a)):
    c = a[i]
    for x in range(9, 0, -1):
        if x in m and m[x] > 0 and x > int(c):
            m[x] -= 1
            if m[x] == 0:
                del m[x]
            ret += chr(ord('0') + x)
            break
    else:
        ret += a[i]
print(ret)
