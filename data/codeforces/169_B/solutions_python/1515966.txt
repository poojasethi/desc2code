import sys
s1 = raw_input()
s2 = raw_input()
ss2 = []
ss1 = []
for i in s1:
    ss1.append(int(i))
for i in s2:
    ss2.append(int(i))
ss2.sort(None,None,True)
l1 = len(s1)
l2 = len(s2)
i1 = 0
i2 = 0
while i1<l1 and i2 < l2:
    if ss1[i1] < ss2[i2]:
        ss1[i1] = ss2[i2]
        i2+=1
    else:
        i1+=1
for i in ss1:
    sys.stdout.write(str(i))
