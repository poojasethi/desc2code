#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

string A,S;

int main()
{
	int i=0,j=0;
	cin >> A >> S;
	sort(S.rbegin(),S.rend());
	while (i<A.size()&&j<S.size())
	{
		if (A[i]<S[j])
		{
			A[i]=S[j];
			i++;
			j++;
		}
		else
			i++;
	}
	cout << A << endl;
	return 0;
}
