#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
char A[100010],B[100010];

int main(){
	cin >> A;
	cin >> B;
	int a=strlen(A);
	int b=strlen(B);
	sort(B,B+b);
	int j=0;
	for(int i=b-1;i>=0;i--){
		while(B[i]<=A[j] && j<a)
			j++;
		if(j>=a)
			break;
		A[j]=B[i];
	}
	cout<<A<<endl;
}
