#include<iostream>
#include<algorithm>
using namespace std;
string a,s;
int main(){
    cin>>a>>s;
    sort(s.begin(),s.end(), greater<int>()); // sort from least to greatest
    for(int i=0,j=0; i<a.size(); i++)
        if(j < s.size() && s[j] > a[i])
            a[i]=s[j++];
    cout<<a;
    return 0;
}
