#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	string a,s;
	cin >> a >> s;
	sort(s.begin(),s.end());
	int i=0,j=s.size()-1;
	while (i<a.size() && j>=0) {
		while (i<a.size() && s[j]<=a[i]) {
			i++;
		}
		if (i<a.size()) {
			a[i] = s[j];
			i++;
		}
		j--;
	}
	cout << a << endl;
	return 0;
}
