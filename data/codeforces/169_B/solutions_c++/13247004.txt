#include <bits/stdc++.h>

using namespace std;

const int MAXN = 111111;

char a[MAXN], b[MAXN];

int main(){
    scanf("%s%s", a, b);
    int lb = strlen(b);
    sort(b, b + lb, greater<char>());
    for (char *i = a, *j = b; *i && *j; ++i)
        if (*i < *j) *i = *j++;
    puts(a);
    return 0;
}
