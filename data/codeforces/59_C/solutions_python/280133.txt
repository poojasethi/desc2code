import string
allowed = input()
letters = set(string.lowercase[:allowed])
template = [ x for x in raw_input() ]
length = len(template)
count = 0
for x in xrange(length):
    if template[x] == '?':
        template[x] = template[length - x - 1]
    if template[length - x - 1] == '?':
        template[length - x - 1] = template[x]
    if template[x] == '?':
        count += 1
    if template[x] in letters:
        letters.remove(template[x])
letters = sorted(list(letters))
count = (count + 1) / 2
for x in xrange(length):
    if template[x] == '?':
        if count - len(letters) <= 0:
            template[x] = template[length - x - 1] = letters.pop(0)
        else:
            template[x] = template[length - x - 1] = "a"
        count -= 1
if len(letters) > 0:
    print "IMPOSSIBLE"
else:
    template = ''.join(template)
    print template if template == template[::-1] else "IMPOSSIBLE"
