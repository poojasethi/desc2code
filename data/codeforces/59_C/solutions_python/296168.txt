import string
def f(a,b):
    if a=='?':return b
    if b=='?':return a
    if a!=b:return'.'
    return a
p=lambda s:map(f,s,reversed(s))
n=input()
A=set(string.letters[:n])
s=p(raw_input())
A=sorted(A-set(s))
for i in range((len(s)+1)/2)[::-1]:
    if s[i]=='?':s[i]=A.pop() if A else'a'      
print 'IMPOSSIBLE' if A or'.'in s else''.join(p(s))
