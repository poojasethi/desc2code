#include <cstdio>
#include <cstring>

int n, len;
char s[101];
bool used[26];

int main() {
	scanf("%d", &n);
	scanf("%s", s);
	len = strlen(s);
	memset(used, 0, sizeof(used));
	bool succ = true;
	for (int i = 0; i < len && succ; i++) {
		if (s[i] != '?' && s[len - i - 1] != '?' && s[i] != s[len - i - 1]) {
			succ = false;
		}
		if (s[i] == '?') {
			s[i] = s[len - i - 1];
		}
		if (s[len - i - 1] == '?') {
			s[len - i - 1] = s[i];
		}
		if (s[i] != '?') {
			used[s[i] - 'a'] = true;
		}
	}
	for (int i = (len - 1) / 2, j = len - i - 1, k = n - 1; i >= 0 && succ; i--, j++) {
		if (s[i] == '?') {
			for (; used[k] && k > 0; k--);
			s[i] = s[len - i - 1] = k + 'a';
			used[k] = true;
		}
	}
	for (int i = 0; i < n && succ; i++) {
		succ = used[i];
	}
	if (succ) {
		printf("%s\n", s);
	} else {
		puts("IMPOSSIBLE");
	}
	return 0;
}
