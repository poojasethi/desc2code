#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"

int k,i,z,n,r,j;
char s[101];
bool B[27];

int main()
{
  scanf("%d%s",&k,s);
  for(i=0,n=strlen(s);i<n;i++)
    if(s[i]!='?' && s[n-1-i]=='?')
      s[n-1-i]=s[i];
  for(i=z=0,r=k;i<=n-1-i;i++)
  {
    z+=s[i]=='?';
    if(s[i]!='?' && !B[s[i]-'a'])
    {
      B[s[i]-'a']=1;
      r--;
    }
    if(s[i]!=s[n-1-i])
      break;
  }
  if(z<r || i<=n-1-i)
  {
    printf("IMPOSSIBLE");
    return 0;
  }
  for(i=0;i<=n-1-i;i++)
    if(s[i]=='?')
      if(z==r)
      {
        while(B[j])
          j++;
        s[i]=(char)j+'a';
        s[n-1-i]=s[i];
        z--;
        r--;
        j++;
      }
      else
      {
        s[i]='a';
        s[n-1-i]=s[i];
        z--;
      }
  printf("%s\n",s);
  return 0;
}
