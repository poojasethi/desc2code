#include <bits/stdc++.h>
using namespace std;

int n, k, p;
bool use[26] = {};
string s;

int main(){
  cin >> k >> s;
  n = s.size();
  for(int i = 0; i < n; i++){
    int j = n - i - 1;
    if(s[i] == '?' && s[j] == '?') continue;
    if(s[i] != s[j]){
      if(s[i] == '?') s[i] = s[j];
      else if(s[j] == '?') s[j] = s[i];
      else {
        cout << "IMPOSSIBLE\n";
        return 0;
      }
    }
    if(s[i] - 'a' >= k){
      cout << "IMPOSSIBLE\n";
      return 0;
    }
    use[s[i] - 'a'] = 1;
  }
  p = k - 1;
  for(int i = n / 2; i >= 0; i--){
    while(p >= 0 && use[p]) p--;
    if(s[i] == '?'){
      if(p >= 0){
        s[i] = s[n - i - 1] = 'a' + p;
        use[p] = 1;
      } else {
        s[i] = s[n - i - 1] = 'a';
      }
    }
  }
  while(p >= 0 && use[p]) p--;
  if(p >= 0){
    cout << "IMPOSSIBLE\n";
    return 0;
  }
  cout << s << '\n';
}
