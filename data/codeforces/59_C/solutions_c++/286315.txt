#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
bool letters[26];
char inp[110];
int main()
{
	int k;
	scanf("%d",&k);
	scanf("%s",inp);
	int len=strlen(inp);
	for(int i=0;i<len;i++){
		if(inp[i]=='?') continue;
		letters[inp[i]-'a']=true;
	}
	bool ok=true;
	int elim=len/2;
	if(len&1){}else elim--;
	for(int i=elim;i>=0;i--){
		int li=len-1-i;
		if(inp[i]=='?' && inp[li]=='?'){
			int tc=k-1;
			for(;tc>=0;tc--) if(!letters[tc]) break;
			if(tc<0) tc=0;
			inp[i]=inp[li]=tc+'a';
			letters[tc]=true;
		}
		else if(inp[i]=='?'){
			inp[i]=inp[li];
		}
		else if(inp[li]=='?') inp[li]=inp[i];
		else{
			if(inp[li]!=inp[i]) {ok=false;break;}
		}
	}
	if(ok)
	for(int i=0;i<k;i++) if(!letters[i]) ok=false;
	if(ok) puts(inp);
	else puts("IMPOSSIBLE");
	return 0;
}
