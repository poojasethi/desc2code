#include <iostream>
#include <string>
#include <cstdlib>
using namespace std;
void error(){
   cout << "IMPOSSIBLE" << endl;
   exit(0);
}
int main(){
    int n,k,j,i,good[255]={0};
    string a;
    cin >> k >> a;
    n=a.length();
    for(i=0; i<n; i++)
       good[a[i]]=1;
    char start='a'+k-1;
    for(i=(n-1)/2; i>=0; i--){
       if(a[i]!='?'&&a[n-i-1]=='?')
          a[n-i-1]=a[i];
       if(a[i]!='?'&&a[n-i-1]!='?'&&a[i]!=a[n-i-1])
          error();
       if(a[i]=='?'&&a[n-i-1]!='?')
          a[i]=a[n-i-1];
       if(a[i]=='?'&&a[n-i-1]=='?'){
          j=start;
          while(good[j]&&j>'a')j--;
          a[i]=a[n-i-1]=j;
          good[j]=1;
       }
    }
    for(i='a'; i<'a'+k-1; i++)
       if(!good[i])error();
    cout << a;
    //system("pause");
    return 0;
}
