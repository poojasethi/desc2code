#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

string s;
int n;
bool exist[256];
main()
{
//	freopen("input.inp","r",stdin);
	cin >> n >> s;
	char limit = 'a' + n - 1;
	int i, j;
	for(i = 0, j = s.length() - 1; i <= j; i++, j--)
	{
		exist[s[i]] = exist[s[j]] = true;
		if(s[i] != s[j])
		{
			if(s[i] == '?') s[i] = s[j];
			else if(s[j] == '?') s[j] = s[i];
			else {cout << "IMPOSSIBLE"; return 0;}
		}
	}
	for(i--, j++; i >= 0; i--, j++)
	{
		if(s[i] == '?')
		{
			for(char ch = limit; ch >= 'a'; ch--)
			if(exist[ch] == false) {s[i] = s[j] = ch; exist[ch] = true; break;}
		}
	}
	for(char ch = 'a'; ch <= limit; ch++) if(exist[ch] == false) {cout << "IMPOSSIBLE"; return 0;}
	for(int k = 0; k < s.length(); k++) if(s[k] == '?') s[k] = 'a';
	cout << s;
	return 0;
}
