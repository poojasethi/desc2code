#include <cstdio>
#include <cstdlib>
#include <set>
#include <cstring>
#include <string>
#include <algorithm>

using namespace std;

set <int> v;
int n, k, i;
char s[111];

void bad ()
{
    puts ("IMPOSSIBLE");
    exit (0);
}

int main ()
{
    scanf ("%d%s", &k, s);
    n = strlen (s);

    for (i = 0; i < k; i++)
        v.insert (-i-'a');
    for (i = 0; i < (n+1)/2; i++)
    {
        if (isalpha (s[i]) && isalpha (s[n-i-1]) && s[i] != s[n-i-1]) bad ();
        else if (s[i] < s[n-i-1]) s[i] = s[n-i-1];
        else s[n-i-1] = s[i];
        if (isalpha (s[i]) && v.count (-s[i])) v.erase (v.find (-s[i]));
    }

    for (i=(n+1)/2; i--;)
    {
        if (isalpha (s[i])) continue;
        s[i] = s[n-i-1] = v.size () ? -*v.begin () : 'a';
        if (v.size ()) v.erase (v.begin ());
    }

    if (v.size ()) bad ();
    else printf ("%s\n", s);
}
