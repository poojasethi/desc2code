#include <iostream>
#include <string>
#include <stack>
using namespace std;
int mark[26]={false};
int main()
{
    int k,ok=true,cnt=0;
    string a;
    cin>>k>>a;
    for (int i=0;i<a.length();++i){
		if (a[i]=='?' && a[a.length()-i-1]=='?')
			++cnt;
		if (a[i]=='?' && a[a.length()-i-1]!='?')
			a[i]=a[a.length()-i-1];
		if (a[i]!='?' && a[a.length()-i-1]!='?' && a[i]!=a[a.length()-i-1])
			ok=false;
		if (a[i]!='?')
			mark[a[i]-'a']=true;
	}
	for (int i=a.length()/2;i>=0;--i){
		if (a[i]=='?' && a[a.length()-i-1]=='?'){
			bool find=false;
			for (int j=k-1;j>=0;--j){
				if (mark[j]==false){
					mark[j]=find=true;
					a[i]=a[a.length()-i-1]='a'+j;
					break;
				}
			}
			if (find==false)
				a[i]=a[a.length()-i-1]='a';
		}
	}
	for (int i=0;i<k;++i){

		if (mark[i]==false)
			ok=false;
	}
	if (ok)
		cout<<a<<endl;
	else cout<<"IMPOSSIBLE"<<endl;
    return 0;
}
