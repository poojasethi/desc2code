#include<cstdio>
#include<cstring>
#include<cmath>
#include<iostream>
#define FOR(i,n) for(i=0;i<n;i++)
#define maxn 100000
#define maxm 500
#define change(a,b) a+=b;b=a-b;a=a-b;
using namespace std;
int main()
{
	char s[105];
	int n,len,i,j;
	int vis[26],flag;
	while(scanf("%d",&n)!=EOF)
	{
		memset(vis,0,sizeof(vis));flag=0;
		scanf("%s",s);
		len=strlen(s);
		FOR(i,len)if(s[i]!='?')vis[s[i]-'a']++;
		//1if(len==1)flag++;
		//printf("%d\n",len);
		for(i=len/2;i>=0&&!flag;i--)
		{
			if(s[i]!='?'&&s[len-1-i]!='?')
			{
				if(s[i]!=s[len-1-i]){flag++;}
			}
			if(s[i]!='?'&&s[len-1-i]=='?'){s[len-i-1]=s[i];}
			else if(s[i]=='?'&&s[len-i-1]!='?'){s[i]=s[len-i-1];}
			else if(s[i]=='?'&&s[len-i-1]=='?')
			{
				for(j=n-1;j>=0;j--)
					if(!vis[j]){s[i]=j+'a';s[len-1-i]=j+'a';vis[j]++;break;}
				if(s[i]=='?')s[i]='a',s[len-1-i]='a';
			}
		}
		FOR(i,n)if(!vis[i]){flag++;break;}
		if(!flag)puts(s);
		else printf("IMPOSSIBLE\n");
	}
	return 0;
}
