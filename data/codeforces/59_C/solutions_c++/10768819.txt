#include <iostream>
using namespace std;
int mark[26]={0};
int main()
{
    int k,ok=1,cnt=0;
    string a;
    cin>>k>>a;
    for (int i=0;i<a.length();++i){
		if (a[i]=='?' && a[a.length()-i-1]=='?')
			++cnt;
		if (a[i]=='?' && a[a.length()-i-1]!='?')
			a[i]=a[a.length()-i-1];
		if (a[i]!='?' && a[a.length()-i-1]!='?' && a[i]!=a[a.length()-i-1])
			ok=0;
		if (a[i]!='?')
			mark[a[i]-'a']=1;
	}
	for (int i=a.length()/2;i>=0;--i){
		if (a[i]=='?' && a[a.length()-i-1]=='?'){
			bool find=0;

			for (int j=k-1;j>=0;--j){

				if (mark[j]==0){
					mark[j]=find=1;
					a[i]=a[a.length()-i-1]='a'+j;
					break;
				}
			}
			if (find==0)
				a[i]=a[a.length()-i-1]='a';
		}
	}
	for (int i=0;i<k;++i){
		if (mark[i]==0)
			ok=0;
	}
	if (ok)
		cout<<a<<endl;
	else cout<<"IMPOSSIBLE"<<endl;
    return 0;
}
