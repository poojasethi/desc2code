#include <iostream>

using namespace std;

int used[1000];

int main() {
	int k;
	string s;
	cin >> k >> s;
	int n = s.size();
	for (int i = 0; i < n; i++) {
		used[s[i]] = 1;
		if (s[i] != '?' && s[n - i - 1] == '?')
			s[n - i - 1] = s[i];
	}
	
	for (int i = 0; i < n / 2; i++) {
		if (s[i] != s[n - i - 1]) {
			cout << "IMPOSSIBLE";
			return 0;
		}
	}
	
	for (int i = k; i < 26; i++) {
		if (used['a' + i] == 1){
			cout << "IMPOSSIBLE";
			return 0;
		}
		
	}
	char ch = 'a' + k - 1;
	for (int i = n / 2; i >= 0; i--) {
		if (ch != 'a') {
			while (ch > 'a' && used[ch]) 
				ch--;
		}
		if (s[i] == '?') {
			s[i] = ch;
			s[n - i - 1] = ch;
			used[ch] = true;
		}
	}
	for (int i = 0; i < k; i++) {
		if (used['a' + i] == 0) {
			cout <<  "IMPOSSIBLE";
			return 0;
		}
	}
	cout << s;
	
	return 0;
}
