#include <iostream>
#include <string>
#include <set>

using namespace std;

int N;
string S;
set<char> alphabet;

void init()
{
	for (int i=0;i<N;i++)
		alphabet.insert('a'+i);
	for (int i=0;i<S.size();i++)
		alphabet.erase(S[i]);
}

string solve()
{
	for (int i=S.size()/2-(S.size()%2==0);i>=0;i--)
	{
		int j=S.size()-i-1;
		if (S[i]==S[j]&&S[i]=='?')
		{
			if (!alphabet.empty())
			{
				set<char>::iterator it=alphabet.end();
				it--;
				S[i]=S[j]=*it;
				alphabet.erase(it);
			}
			else
				S[i]=S[j]='a';
		}
		else if (S[i]=='?'&&S[j]!='?')
			S[i]=S[j];
		else if (S[i]!='?'&&S[j]=='?')
			S[j]=S[i];
		else if (S[i]!=S[j])
			return "IMPOSSIBLE";
	}
	return (alphabet.empty()?S:"IMPOSSIBLE");
}

int main()
{
	cin >> N >> S;
	init();
	cout << solve() << endl;
	return 0;
}
