#include <iostream>
int k,f,c[29];
std::string s;
int main(){
    std::cin>>k>>s;
    int l=s.size();
    for(int i=0;i<l;i++)if(s[i]!='?')c[s[i]-'a']=1;
    int ch=k-1;
    while(c[ch]&&ch>=0)ch--;
    for(int i=(l-1)/2;i>=0;i--){
        if(s[i]=='?'){
            if(s[l-1-i]!='?')s[i]=s[l-1-i];
            else{
                s[i]=s[l-1-i]=ch>=0?'a'+ch:'a';
                if(ch>=0){
                    c[ch]=1;
                    while(c[ch]&&ch>=0)ch--;
                }
            }
        }else{
            if(s[l-1-i]!='?'&&s[i]!=s[l-1-i]){f=1;break;}
            s[l-1-i]=s[i];
        }
    }
    std::cout<<(f||ch>=0?"IMPOSSIBLE":s);
}
