#include<iostream>
#include<stack>

using namespace std;

int main()
{
	bool ar[26];
	int k, i;
	cin >> k;
	string a;
	cin >> a;
	int l = a.size();
	int p = (l+1)/2;
	for (i = 0; i < p; i++)
	{
		if (a[i] == '?' && a[l-i-1] != '?') a[i] = a[l-i-1];
		if (a[i] != '?' && a[l-i-1] == '?') a[l-i-1] = a[i];
		if (a[i] != a[l-i-1]) 
		{
			cout << "IMPOSSIBLE\n";
			return 0;
		}
	}
	for (i = 0; i < k; i++) ar[i] = 0;
	for (i = 0; i < p; i++) 
	{
		ar[a[i]-97] = 1;
	}
	stack<int> s;
	int c;
	for (i = 0; i<k; i++) 
	{
		if(ar[i] == 0) s.push(i);
	}
	for (i = p-1; i >= 0; i--)
	{
		c = 0;
		if (a[i] == '?') 
		{
			if (!s.empty())
			{
				c = s.top();
				s.pop();
			}
			a[i] = a[l-i-1] = 'a'+c;
		}
	}
	if (!s.empty()) cout << "IMPOSSIBLE\n";	
	else cout << a << endl;
	return 0;
}
