#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <cstring>
using namespace std;

bool letters[27];

int main()
{
	//freopen("input.in", "r", stdin);
	memset(letters, 0, sizeof(letters));
	int k;
	cin >> k;
	string s;
	cin >> s;
	int n = s.size();
	for(int i = 0; i < n; i++)
	{
		if(s[i] != '?')
			letters[s[i] - 'a' + 1] = true;
		if(s[i] == '?' && s[n - i - 1] != '?')
			s[i] = s[n - i - 1];
		if(s[i] != '?' && s[n - i - 1] != '?' && s[n - i - 1] != s[i])
		{
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}
	}
	int idx = 1;
	for(idx = k; idx > 0 && letters[idx]; idx--);
	for(int i = (n - 1) / 2; i >= 0; i--)
	{
		if(s[i] == '?')
		{
			if(idx < 1) 
				idx = 1;
			s[i] = s[n - i - 1] = idx + 'a' - 1;
			letters[idx] = true;
			while(idx > 0 && letters[idx])
				idx--;
		}
	}
	if(idx < 1)
		cout << s << endl;
	else
		cout << "IMPOSSIBLE" << endl;
	return 0;
}
