#include<stdio.h>
#include<string.h>

int k,i,j;
char s[110];
bool a[28];
int main()
{
    scanf("%d",&k);
    scanf("%s",s);
    int n = strlen(s);
    for(i = 0; i <= n/2; i++){
        if(s[i] == '?' && s[n-1-i] != '?') s[i] = s[n-i-1];
        if(s[i] != '?'){
                a[s[i]-'a'] = true;
                if(s[n-1-i] == '?') s[n-1-i] = s[i];
                if(s[n-1-i] != s[i]){
                            puts("IMPOSSIBLE");
                            return 0;
                }
        }
    }
    for(j = k-1; j >= 0 && a[j]; j--);
    for(i = (n-1)/2; i >= 0; i--)
        if(s[i] == '?'){
            if(j < 0) j = 0;
            s[i] = s[n-1-i] = j+'a';
            a[j] = true;
            while(j >= 0 && a[j]) j--;
        }
    if(j<0) puts(s);
    else puts("IMPOSSIBLE");
    return 0;
}
