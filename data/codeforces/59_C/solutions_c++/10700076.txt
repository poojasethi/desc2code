#include <cstdio>
#include <cstring>
#include <cstdlib>
using namespace std;
int main()
{
    int n;
    char s[105];
    scanf("%d%s",&n,s);
    bool used[30];
    int i,j,k;
    k=strlen(s);
    bool flag=true;
    memset(used,false,sizeof(used));
    for(i=0;i<k&&flag;i++)
    {
        if(s[i]!='?'&&s[k-i-1]!='?'&&s[i]!=s[k-i-1])
            flag=false;
        if(s[k-i-1]=='?')
            s[k-i-1]=s[i];
        if(s[i]=='?')
            s[i]=s[k-i-1];
        if(s[i]!='?')
            used[s[i]-'a']=true;
    }
    for(i=(k-1)/2,j=n-1;i>=0&&flag;i--)
    {
        while(used[j]&&j>0)
            j--;
        if(s[i]=='?')
        {
            s[i]='a'+j;
            s[k-i-1]=s[i];
            used[j]=true;
        }
    }
    for(i=0;i<n;i++)
        if(!used[i])
            flag=false;
    if(flag)
        printf("%s\n",s);
    else
        printf("IMPOSSIBLE");
    return 0;
}
