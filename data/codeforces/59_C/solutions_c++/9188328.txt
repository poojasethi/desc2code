#include <iostream>
#include <algorithm>
#include <string>
#include <vector>

using namespace std;

string s;
int len, k;
vector<int> ind;
bool mark[30];

int main() {
	ios :: sync_with_stdio(false);
	cin.tie(0);

	cin >> k >>  s;
	len = s.length();

	for (int i = 0; i < len; i++) {
		if (s[i] == '?')
			continue;
		if (s[len - i - 1] == '?')
			s[len - i - 1] = s[i];
		if (s[len - i - 1] != s[i]) {
			cout << "IMPOSSIBLE\n";
			return 0;
		}
		mark[s[i] - 'a'] = true;
	}
	for (int i = 0; i < 26; i++) 
		if (mark[i] && i >= k) {
			cout << "IMPOSSIBLE\n";
			return 0;
		}
	for (int i = 0; i < k; i++)
		if (!mark[i])
			ind.push_back(i);
	for (int i = (len - 1) / 2; i >= 0; i--) 
		if(s[i] == '?') {
			if (ind.size()) {
				s[i] = s[len - i - 1] = 'a' + ind[ind.size() - 1];
				ind.pop_back();
			}
			else
				s[i] = s[len - i - 1] = 'a';
		}
	if (ind.size() || !k) {
		cout << "IMPOSSIBLE\n";
		return 0;
	}
	cout << s << endl;
}
