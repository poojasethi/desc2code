#include<iostream>
using namespace std;
int main()
{
 int k,m[30]={0},i,b,l,c=0,p,p1=0;
 string s;
 cin>>k;
 cin>>s;
 p=k-1;
 l=s.length();
 b=l/2;
 if(l%2==0)
 b--;
 for(i=0;i<l;i++)
 {
  if(s[i]!='?')
  {
   if(m[s[i]-'a']==0)
   {
    m[s[i]-'a']=1;
    c++;
   }
  }
 }
 for(i=b;i>=0;i--)
 {
  if((s[i]!=s[l-i-1]&&s[i]!='?')&&s[l-i-1]!='?')
  {
   cout<<"IMPOSSIBLE";
   return 0;
  }
  else if(s[i]=='?'&&s[l-i-1]!='?')
   s[i]=s[l-i-1];
  else if(s[l-i-1]=='?'&&s[i]!='?')
   s[l-i-1]=s[i];
  else if(s[i]=='?'&&s[l-i-1]=='?')
  {
   if(c!=k)
   {
    while(m[p]&&p>0)
    p--;
    //s[i]=s[l-i-1]=(char)(p+97);
    p1=p;
    m[p--]=1;
    c++;
   }
   s[i]=s[l-i-1]=(char)(p1+97);
   p1=0;
  }
 }
 if(c==k)
 {
  cout<<s<<endl;
  return 0;
 }
 cout<<"IMPOSSIBLE";
 return 0;
}
