#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
char s[205];
int len,i,n,used[35],flag,k;
int main(){
	memset(used,0,sizeof(used));
	scanf("%d%s",&n,s);
	flag = 1;
	len = strlen(s);
	for(i = 0;i < len && flag;i++){
		if (s[i] != '?' && s[len - i - 1] != '?' && s[i] != s[len - i - 1]) {
			flag = 0;
		}
		if (s[i] == '?') {
			s[i] = s[len - i - 1];
		}
		if (s[len - i - 1] == '?') {
			s[len - i - 1] = s[i];
		}
		if (s[i] != '?') {
			used[s[i] - 'a'] = 1;
		}
	}
	for(i = (len - 1) / 2,k = n - 1;i >= 0 && flag;i--){
		if(s[i] == '?'){
			while(used[k] != 0 && k > 0)
			k--;
			s[i] = k + 'a';
			s[len - i - 1] = s[i];
			used[k] = 1;
		}
	}
	for(i = 0;i < n;i++)
	if(used[i] == 0){
	flag = 0;
	break;
	}
	if(flag)
	printf("%s\n",s);
	else
	puts("IMPOSSIBLE");
	return 0;
}
