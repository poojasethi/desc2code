#include <string>
using namespace std;

int main() {
  int k,has[26]={0};
  string pat;
  cin>>k>>pat;
  for(int i=0;i<pat.size();++i) {
    if(pat[i]=='?')
    if(pat[pat.size()-1-i]!='?')
    pat[i]=pat[pat.size()-1-i];
  }
  for(int i=0;i<pat.size();++i)if(pat[i]!='?')
    if(pat[i]>=k+'a'||pat[i]<'a') return im();
    else if(pat[i]!=pat[pat.size()-1-i]) return im();
    else has[pat[i]-'a']=1;  
  for(int i=pat.size()/2,j=k-1;i<pat.size();++i)
    if(pat[i]=='?') { while(has[j]&&j)j--; pat[pat.size()-1-i]=pat[i]='a'+j; has[j]=1; }
  for(int i=0;i<k;++i) if(!has[i]) return im();
  cout << pat;
  return 0;
}
