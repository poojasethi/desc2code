#include <cstdio>

#define N 500
#define M 1000

int a[N], b[M], d[N], e[N], h[N], x[M], y[M], n, m, i, j, s, t, min = 1000000000;
bool c[M];

bool go(int k)
{
	int v;
	for (int i = h[k]; ~i; i = b[i])
		if (!c[i] && d[y[i]] != j)
		{
			d[v = y[i]] = j;
			if (!~e[v] || go(e[v]))
			{
				e[v] = k;
				return true;
			}
		}
	return false;
}

int main()
{
	scanf("%d%d", &n, &m);

	t = 3 * n + m - 2;

	for (i = 0; i < m; i++)
		b[i] = -1;
	for (i = 0; i < n; i++)
		h[i] = -1;

	for (i = 0; i < m; i++)
	{
		scanf("%d%d", &x[i], &y[i]);
		--x[i], --y[i];
		b[i] = h[x[i]];
		h[x[i]] = i;
	}

	for (i = 0; i < n; i++)
	{
		s = 0;
		for (j = 0; j < m; j++)
			c[j] = 0;
		for (j = 0; j < n; j++)
			d[j] = e[j] = -1;
		for (j = 0; j < m; j++)
			if (x[j] == i || y[j] == i)
				c[j] = 1, ++s;
		for (j = 0; j < n; j++)
			s += go(j);
		s = t - 2 * s;
		if (s < min)
			min = s;
	}

	printf("%d\n", min);

	return 0;
}
