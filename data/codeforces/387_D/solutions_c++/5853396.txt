#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,x,y,mi,v,s1,num;
int f[N][N],a[N],next[N],head[N],ff[N],F[N];
bool dfs(int x) {
	int v; F[x]=num;
	for (v=head[x];v;v=next[v]) {
		int A=a[v]+n;
		if (F[A]!=num) {
			F[A]=num;
			if (!ff[A]||F[ff[A]]!=num&&dfs(ff[A])) {
				ff[A]=x; return 1;
			}
		}
	}
	return 0;
}
int main() {
	//freopen("387d.in","r",stdin);
	scanf("%d%d",&n,&m);
	For(i,1,m) {
		scanf("%d%d",&x,&y),f[x][y]=1;
		a[i]=y,next[i]=head[x],head[x]=i;
	}
	mi=10000;
	For(i,1,n) {
		int s=f[i][i];
		For(j,1,n) if (i!=j) s+=f[i][j]+f[j][i];
		s1=0; For(j,1,n*2) ff[j]=0;
		For(j,1,n) if (j!=i) F[i]=F[i+n]=++num,s1+=dfs(j);
		mi=min(mi,(n-1)*2+1+n-1+m-(s+s1)*2);
	}
	printf("%d\n",mi);
	return 0;
}
