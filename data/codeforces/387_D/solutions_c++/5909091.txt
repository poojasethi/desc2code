#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
int n,m;
int b[2005],e[2005],h[1005],p[1005],match[1005];
int c[505][505],g[505];
int tot;

void add(int x,int y){
    b[++tot]=y; e[tot]=h[x]; h[x]=tot;
}

bool dfs(int k){
    for (int go=h[k];go;go=e[go])
        if (p[b[go]]==0){
            p[b[go]]=1;
            if (match[b[go]]==0||dfs(match[b[go]])){
                match[b[go]]=k;
                return 1;
            }
        }
    return 0;
}

int main(){
    //freopen("in","r",stdin);
    cin>>n>>m;
    int x,y;
    for (int i=0;i<m;i++){
        scanf("%d%d",&x,&y);
        add(x,y);
        c[x][y]=1;
        g[x]++; if (x!=y) g[y]++;
    }
    int ans=168430090,tmp;
    for (int i=1;i<=n;i++){
        memset(match,0,sizeof(match));
        tmp=0;
        for (int j=1;j<=n;j++)
            if (i!=j){
                if (c[i][j]==0) tmp++;
                if (c[j][i]==0) tmp++;
            }
        if (c[i][i]==0) tmp++;
        int F=0;
        for (int j=1;j<=n;j++)if (i!=j){
            memset(p,0,sizeof(p));
            p[i]=1;
            if (dfs(j)) F++;
        }
        ans=min(ans,tmp+(m-g[i]-F)+(n-1-F));
    }
    cout<<ans;
    return 0;
}
