#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
int ans,tmp,n,m,x,y,i,j,k,cnt,g;
int vis[505], mat[505];
bool a[505][505], b[505][505];

bool pd(int x){
  if (vis[x]==j) return 0;
  vis[x] = j;
  for (int i=1;i<=n;i++)
  if (b[x][i] && (mat[i]==0 || pd(mat[i]))){
    mat[i]=x; return 1;
  }
  return 0;
}

int main(){
  scanf("%d%d",&n,&m);
  for (i=1;i<=m;i++){
    scanf("%d%d",&x,&y);
    a[x][y] = 1;
  }
  ans = 1000000000;
  for (i=1;i<=n;i++){
    tmp = 0;
    if (!a[i][i]) tmp++;
    for (j=1;j<=n;j++)
    if (i!=j){
      if (!a[i][j]) tmp++;
      if (!a[j][i]) tmp++;
    }
    memset(b,0,sizeof(b));
    for (j=1,g=0;j<=n;j++) if (j!=i)
      for (k=1;k<=n;k++) if (k!=i)
      if (a[j][k]) b[j][k] = 1, g++;
    cnt = 0;
    memset(vis,0,sizeof(vis));
    memset(mat,0,sizeof(mat));
    for (j=1;j<=n;j++)
      if (j!=i && pd(j)) cnt++;
    cnt = (n-1-cnt) + (g-cnt);
    if (ans>cnt+tmp) ans = cnt+tmp;
  }
  printf("%d\n",ans);
  return 0;
}
