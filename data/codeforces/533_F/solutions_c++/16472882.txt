#include<bits/stdc++.h>

using namespace std;

#define pii pair<int,int>
#define ll long long
#define N (int)(2e5+10)
#define mod 1000000007
#define mp make_pair
#define pb push_back
#define nd second
#define st first
#define inf mod
#define endl '\n'
#define sag (sol|1)

int h1,h2,i,j,k,n,m,x,y,z,h3,h4;
int A[N][30];
int H[N],pw[N];
vector<int> v;
string str,str2;
char str1[N];

int main(){
	cin >> n >> m;

	scanf("%s",str1);
	str = str1;
	scanf("%s",str1);
	str2 = str1;

	pw[0]=1;

	for(i=1 ; i<N ; i++)
		pw[i] = pw[i-1]*1453;

	for(i='a' ; i<='z' ; i++)
		for(j='a' ; j<='z' ; j++){
			h1=h2=h3=h4=0;
			for(k=0 ; k<str2.size() ; k++)
				h2 = h2*1453+(str2[k]==j)+1;
			for(k=0 ; k<str2.size() ; k++)
				h3 = h3*1453+(str2[k]==i)+1;

			for(k=0 ; k<str.size() ; k++){
				if(k<str2.size()){
					h1 = h1*1453+(str[k]==i)+1;
					h4 = h4*1453+(str[k]==j)+1;
				}
				else{
					h1 = (h1-pw[str2.size()-1]*((str[k-str2.size()]==i)+1))*1453+(str[k]==i)+1;
					h4 = (h4-pw[str2.size()-1]*((str[k-str2.size()]==j)+1))*1453+(str[k]==j)+1;
				}

				if(k>=str2.size()-1 and h1 == h2 and h3==h4)
					A[k-str2.size()+1][j-'a']=1;

			}
		}

	for(i=0 ; i<str2.size() ; i++)
		H[str2[i]-'a']=1;

	for(i=0 ; i<str.size() ; i++){
		bool t=1;
		for(j=0 ; j<=25 ; j++)
			if(H[j] and !A[i][j])
				t=0;
		if(t)
			v.pb(i+1);
	}

	cout << v.size() << endl;

	for(i=0 ; i<v.size() ; i++)
		cout << v[i] << ' ';
}
