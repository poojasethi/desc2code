#include <stdio.h>
#include <algorithm>
#include <string.h>
#include <string>
#include <map>
#include <vector>
#include <iostream>
using namespace std;
typedef long long ll;
const int N = (int)2e5 + 5;
const int p = 41;
const int mod = 7654321;

char s[N],t[N];
int hash1[27],hash2[27];

bool isok()
{
    int cnt = 0;
    for(int i=1;i<=26;i++)
    {
        for(int j=1;j<=26;j++)
        {
            if(hash1[i]==hash2[j] && hash1[j]==hash2[i])
            {
                cnt ++;
                break;
            }
        }
    }
    return cnt==26;
}

int main()
{
    int n,m,base=1;
    scanf("%d%d",&n,&m);
    scanf("%s%s",s+1,t+1);
    for(int i=1;i<=n;i++) s[i] -= 'a' - 1;
    for(int i=1;i<=m;i++) t[i] -= 'a' - 1;
    for(int i=1;i<=m;i++) base = base * p % mod;

    for(int i=1;i<=m;i++)
    {
        for(int j=1;j<=26;j++)
        {
            if(t[i]==j) hash1[j] = (hash1[j] * p + 1) % mod;
            else hash1[j] = (hash1[j] * p) % mod;
        }
    }
    for(int i=1;i<=m;i++)
    {
        for(int j=1;j<=26;j++)
        {
            if(s[i]==j) hash2[j] = (hash2[j] * p + 1) % mod;
            else hash2[j] = (hash2[j] * p) % mod;
        }
    }

    vector<int> ans;
    if(isok()) ans.push_back(1);

    for(int i=m+1;i<=n;i++)
    {
        for(int j=-1;j<=26;j++)
        {
            if(s[i-m]==j) hash2[j] = ((hash2[j] * p - base) % mod + mod) % mod;
            else hash2[j] = (hash2[j] * p) % mod;
            if(s[i]==j) hash2[j] = (hash2[j] + 1) % mod;
        }
        if(isok()) ans.push_back(i-m+1);
    }
    printf("%d\n",ans.size());
    for(int i=0;i<ans.size();i++)
    {
        if(i) printf(" ");
        printf("%d",ans[i]);
    }
    if(ans.size() != 0) puts("");
    return 0;
}
