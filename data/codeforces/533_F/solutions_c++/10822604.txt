#include<cstdio>
#define M 1000000007
using namespace std;

char s[200100],t[200100];
int vis[200100],a[200100],b[200100],ans[200100];
long long hash[200100];
int main(){
	int n,m;scanf("%d%d",&n,&m);
	scanf("%s",&s);scanf("%s",&t);
	long long p=1;
	for(int i=0;i<m;i++)p*=3,p%=M;
	for(char i='a';i<='z';i++)
		for(char j=i;j<='z';j++){
			for(int k=0;k<n;k++){
				if(s[k]==i)a[k]=1;
				else if(s[k]==j)a[k]=2;
				else a[k]=0;
			}
			for(int k=0;k<m;k++){
				if(t[k]==j)b[k]=1;
				else if(t[k]==i)b[k]=2;
				else b[k]=0;
			}
			int cnt=0;
			for(int k=0;k<m;k++)cnt+=(b[k]!=0);
			long long h=0;
			for(int k=0;k<m;k++)
				h=h*3+b[k],h%=M;
			long long wa=0;
			for(int k=0;k<m;k++)
				wa=wa*3+a[k],wa%=M;
			if(wa==h)vis[m-1]+=cnt;
			for(int k=m;k<n;k++){
				wa=wa*3+a[k];wa-=M;
				wa-=a[k-m]*p;wa%=M;wa+=M;wa%=M;
				if(wa==h)vis[k]+=cnt;
			}
		}
	int ansn=0;
	for(int i=0;i<n;i++)
		if(vis[i]>=m)ans[ansn++]=i+2-m;
	printf("%d\n",ansn);
	if(ansn>0){
		for(int i=0;i<ansn-1;i++)printf("%d ",ans[i]);printf("%d\n",ans[ansn-1]);
	}
	return 0;
}
