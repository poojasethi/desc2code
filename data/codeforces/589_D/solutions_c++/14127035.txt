#include<cstdio>
#include<vector>
#include<algorithm>

using namespace std;

int a[1111];
int b[1111];
int c[1111];
int r[1111];

int cal(int x,int y){
	if(b[x]<c[x])return b[x]+y-a[x];
	return b[x]-y+a[x];
}

int main(){
	int i,j,n;
	scanf("%d",&n);
	for(i=0;i<n;i++) scanf("%d%d%d",&a[i],&b[i],&c[i]);
	for(i=0;i<n;i++)for(j=i+1;j<n;j++){
		if(a[i]<=a[j]+abs(b[j]-c[j])&&a[j]<=a[i]+abs(b[i]-c[i])&&
																	1LL*(cal(i,max(a[i],a[j]))-cal(j,max(a[i],a[j])))*
																	(cal(i,min(a[i]+abs(b[i]-c[i]),a[j]+abs(b[j]-c[j])))-cal(j,min(a[i]+abs(b[i]-c[i]),a[j]+abs(b[j]-c[j]))))<=0){
																		r[i]++;
																		r[j]++;
																	}
	}
	for(i=0;i<n;i++)printf("%d ",r[i]);
}
