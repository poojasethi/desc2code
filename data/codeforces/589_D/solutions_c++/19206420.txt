#include<bits/stdc++.h>
using namespace std;
struct p{
    double x,y;
};
struct aa{
    p a,b;
}cnt[1050];
double cross(p a,p b,p o){
    return (a.x-o.x)*(b.y-o.y)-(a.y-o.y)*(b.x-o.x);
}
bool is(aa u,aa v){
    return (cross(v.a,u.b,u.a)*cross(u.b,v.b,u.a)>=0)&&(cross(u.a,v.b,v.a)*cross(v.b,u.b,v.a)>=0)&&max(u.a.x,u.b.x)>=min(v.a.x,v.b.x)&&max(v.a.x,v.b.x)>=min(u.a.x,u.b.x)&&max(u.a.y,u.b.y)>=min(v.a.y,v.b.y)&&max(v.a.y,u.a.y)>=min(u.a.y,v.b.y);
}
int ans[1050];
int main(){
    int n,t,s,f;
    cin>>n;
    for(int i=0;i<n;i++){
        scanf("%d%d%d",&t,&s,&f);
        cnt[i].a.x=t;
		cnt[i].a.y=s;
        if(s<=f){
            cnt[i].b.x=t+f-s;
            cnt[i].b.y=f;
        }
        else {
            cnt[i].b.x=t+s-f;
            cnt[i].b.y=f;
        }
    }
    memset(ans,0,sizeof(ans));
    for(int i=0;i<n;i++)
        for(int j=i+1;j<n;j++)
            if(is(cnt[i],cnt[j])){
            	ans[i]++;ans[j]++;
			}
    for(int i=0;i<n;i++){
        cout<<ans[i]<<" ";
    }
    cout<<endl;
}
