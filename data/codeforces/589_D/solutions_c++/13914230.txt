#include <bits/stdc++.h>

using namespace std;

const int N=1005;

int n;
int t[N],s[N],f[N];
int ans[N];

bool meet(int x,int y)
{
    int lx,rx,ly,ry;
    lx=t[x],rx=t[x]+abs(s[x]-f[x]);
    ly=t[y],ry=t[y]+abs(s[y]-f[y]);

    if (rx<ly || ry<lx) return false;
    if (f[x]>s[x] && f[y]>s[y])
    {
        if (lx-ly!=s[x]-s[y]) return false;
        return true;
    }
    if (f[x]<s[x] && f[y]<s[y])
    {
        if (ly-lx!=s[x]-s[y]) return false;
        return true;
    }
    double k;
    if (f[y]>s[y] && f[x]<s[x])
    {
        k=(s[x]-s[y]+lx+ly)/2.0;
        if (k>=lx && k<=rx && k>=ly && k<=ry) return true;
        return false;
    }
    k=(s[y]-s[x]+lx+ly)/2.0;
    if (k>=lx && k<=rx && k>=ly && k<=ry) return true;
    return false;
}

int main()
{
    int i,j;
    scanf("%d",&n);
    for (i=1;i<=n;++i)
    {
        scanf("%d%d%d",t+i,s+i,f+i);
        for (j=1;j<i;++j)
          if (meet(i,j)) ++ans[i],++ans[j];
    }
    printf("%d",ans[1]);
    for (i=2;i<=n;++i)
        printf(" %d",ans[i]);
    printf("\n");
    return 0;
}
