typedef long long int LL;

#define x first

int P[2000],K[2000],D[2000],S[2000];

inline bool f(int a,int b)
{
    int tp=max(P[a],P[b]),tk=min(K[a],K[b]);
    if (tp>tk) return false;
    int ap=S[a]+D[a]*(tp-P[a]);
    int ak=S[a]+D[a]*(tk-P[a]);
    int bp=S[b]+D[b]*(tp-P[b]);
    int bk=S[b]+D[b]*(tk-P[b]);
    //printf ("%d %d %d %d\n",ap,ak,bp,bk);
    return 1LL*(ap-bp)*(ak-bk)<=0;
}
    
int main(){
    int n;
    scanf ("%d",&n);
    for (int i=1;i<=n;++i)
    {
        int t,s,f;
        scanf ("%d %d %d",&t,&s,&f);
        P[i]=t;
        K[i]=t+abs(s-f);
        D[i]=abs(s-f)/(f-s);
        S[i]=s;
        //printf ("%d %d %d %d\n",P[i],K[i],D[i],S[i]);
    }
    for (int i=1;i<=n;++i)
    {
        int wyn=0;
        for (int j=1;j<=n;++j) if (i!=j) wyn+=f(i,j);
        printf ("%d ",wyn);
    }
    return 0;
}
