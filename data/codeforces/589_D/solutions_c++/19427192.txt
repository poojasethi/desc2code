#include<cstdio>
#include<algorithm>
using namespace std;
#define N 1010
int t[N],s[N],f[N],e[N],n,anss[N],dir[N];
int Abs(int x)
{
	return x<0 ? (-x) : x;
}
int main()
{
	//freopen("d.in","r",stdin);
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
	  {
	  	scanf("%d%d%d",&t[i],&s[i],&f[i]);
	  	e[i]=t[i]+Abs(f[i]-s[i]);
	  	dir[i]=(f[i]>s[i] ? 1 : -1);
	  }
	for(int i=1;i<=n;++i)
	  for(int j=1;j<i;++j)
	    if(!(e[i]<t[j] || t[i]>e[j]))
	      {
	      	bool flag=0;
	      	if(t[i]>t[j])
	      	  {
	      	  	flag=1;
	      	  	swap(i,j);
	      	  }
	      	int tsi=s[i],tti=t[i];
	      	s[i]+=(dir[i]*(t[j]-t[i]));
	      	t[i]=t[j];
	      	int yikuai=min(e[i]-t[i],e[j]-t[j]);
	      	if(s[i]==s[j] || (s[j]>s[i] && dir[j]==-1 && dir[i]==1 && s[j]-yikuai<=s[i]+yikuai)
			|| (s[i]>s[j] && dir[i]==-1 && dir[j]==1 && s[i]-yikuai<=s[j]+yikuai))
	      	  {
	      	  	++anss[i];
	      	  	++anss[j];
	      	  }
	      	s[i]=tsi;
	      	t[i]=tti;
	      	if(flag)
	      	  swap(i,j);
	      }
	for(int i=1;i<n;++i)
	  printf("%d ",anss[i]);
	printf("%d\n",anss[n]);
	return 0;
}
