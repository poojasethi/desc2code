#include<cstdlib>
#include<cstdio>
#include<algorithm>

using namespace std;

const int N=1024;

struct data{
    int x,y;
};

data a[N],b[N];
int s[N];
int n,m,i,j,k;

long long cross(data u,data v,data w){
    return (long long)(v.x-u.x)*(w.y-u.y)-(long long)(w.x-u.x)*(v.y-u.y);
}

bool check(int u,int v){
    long long c1,c2,c3,c4;
    if(a[u].x>b[v].x || a[v].x>b[u].x)return false;
    c1=cross(a[u],b[u],a[v]);
    c2=cross(a[u],b[u],b[v]);
    c3=cross(a[v],b[v],a[u]);
    c4=cross(a[v],b[v],b[u]);
    if((c1>0 && c2>0) || (c1<0 && c2<0))return false;
    if((c3>0 && c4>0) || (c3<0 && c4<0))return false;
    return true;
}

int main(){
    scanf("%d",&n);
    for(i=1;i<=n;i++){
        scanf("%d%d%d",&a[i].x,&a[i].y,&b[i].y);
        b[i].x=a[i].x+abs(a[i].y-b[i].y);
    }
    for(i=1;i<n;i++)
        for(j=i+1;j<=n;j++)
            if(check(i,j)){
                s[i]++;
                s[j]++;
            }
    for(i=1;i<=n;i++)printf("%d ",s[i]);
    return 0;
}
