#include <cstdio>
#include <algorithm>
using namespace std;

int abs(int a){
    return (a<0)?(-a):(a);
}

int N=0;
int Ts[1001], Tf[1001], Ws[1001], Wf[1001], R[1001];

int GetPos(int i, int t){
    if(Ws[i]<Wf[i])return Ws[i]+t-Ts[i];
    else return Ws[i]-t+Ts[i];
}

int main(){
    scanf("%d", &N);
    for(int i=1;i<=N;i++){
        scanf("%d%d%d", &Ts[i], &Ws[i], &Wf[i]);
        Tf[i]=Ts[i]+abs(Ws[i]-Wf[i]);
    }
    for(int i=1;i<=N;i++)
    for(int j=i+1;j<=N;j++){
        if(Ts[i]>Tf[j] || Ts[j]>Tf[i])continue;
        int Ps=max(Ts[i], Ts[j]);
        int Pf=min(Tf[i], Tf[j]);
        int GP1s=GetPos(i, Ps);
        int GP1f=GetPos(i, Pf);
        int GP2s=GetPos(j, Ps);
        int GP2f=GetPos(j, Pf);

        if(GP1s==GP2s || GP1f==GP2f || ((GP1s<GP2s) != (GP1f<GP2f))){
            R[i]++;
            R[j]++;
            continue;
        }
    }
    for(int i=1;i<=N;i++)printf("%d ", R[i]);
    printf("\n");
    return 0;
}
