using namespace std;

long long n , t[2000] , s[2000] , f[2000] , sgn[2000];

int main()
{
	cin >> n;
	for( int i = 0 ; i < n ; i++ )
	{
		cin >> t[i] >> s[i] >> f[i];
		sgn[i] = (f[i]-s[i])/abs( f[i]-s[i] );
	}
	for( int i = 0 ; i < n ; i++ )
	{
		int ans = 0;
		for( int j = 0 ; j < n ; j++ )
			if( i != j )
			{
				if( sgn[i] == sgn[j] )
					ans += s[i]-sgn[i]*t[i] == s[j]-sgn[j]*t[j] && max( t[i] , t[j] ) <= min( t[i]+abs( f[i]-s[i] ) , t[j]+abs( f[j]-s[j] ) );
				else
				{
					double x = ((double)(s[i]-sgn[i]*t[i]-s[j]+sgn[j]*t[j]))/(sgn[j]-sgn[i]);
					ans += x >= max( t[i] , t[j] ) && x <= min( t[i]+abs( f[i]-s[i] ) , t[j]+abs( f[j]-s[j] ) );
				}
			}
		cout << ans << " ";
	}
	return 0;
}
