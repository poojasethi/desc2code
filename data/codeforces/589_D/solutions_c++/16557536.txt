#include <bits/stdc++.h>

using namespace std;

struct person
{
  int t, s, f;

  int duration() const
  {
    return abs(s - f);
  }

  int dir() const
  {
    return f > s ? 1 : -1;
  }

  int end() const
  {
    return t + duration();
  }
};

int check(person a, person b)
{
  int t0 = max(a.t, b.t);
  int t1 = min(a.end(), b.end());
  if (t1 < t0)
    return 0;
  if (a.t < t0)
    a.s += a.dir() * (t0 - a.t);
  else if (b.t < t0)
    b.s += b.dir() * (t0 - b.t);
  if (a.dir() == b.dir())
    return a.s == b.s ? 1 : 0;
  else {
    if (a.s == b.s)
      return 1;
    if ((a.s < b.s && a.dir() == -1 && b.dir() == 1) ||
	(a.s > b.s && a.dir() == 1 && b.dir() == -1))
      return 0;
    if ((t1 - t0) * 2 >= abs(b.s - a.s))
      return 1;
    return 0;
  }
}

int main()
{
  int n; cin >> n;
  static person v[1000];
  for (int i = 0; i < n; ++i)
    cin >> v[i].t >> v[i].s >> v[i].f;
  for (int i = 0; i < n; ++i) {
    int s = 0;
    for (int j = 0; j < n; ++j)
      if (i != j)
	s += check(v[i], v[j]);
    cout << s << " ";
  }
  cout << endl;
  return 0;
}
