#include<bits/stdc++.h>

using namespace std;

struct obj
{
    int t,s,e;
};

bool meet(int t1,int s1,int e1,int t2,int s2,int e2)
{
    int d1=(e1-s1)/abs(e1-s1);
    int d2=(e2-s2)/abs(e2-s2);
    if(t1<t2)
    {
        s1+=(t2-t1)*d1;
        t1=t2;
    }
    else if(t2<t1)
    {
        s2+=(t1-t2)*d2;
        t2=t1;
    }
    if(d1==1 && s1>e1 || d1==-1 && s1<e1)
        return 0;
    if(d2==1 && s2>e2 || d2==-1 && s2<e2)
        return 0;
    if(d1==d2)
        return s1==s2;
    if(d1==-1)
        swap(d1,d2),swap(s1,s2),swap(e1,e2);
    double intersection=(max(s1,s2)-min(s1,s2))/2.0;
    return s1+intersection<=e1 && s2-intersection>=e2 && s1+intersection==s2-intersection;
}

obj arr[1005];
int ans[1005];

int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>arr[i].t>>arr[i].s>>arr[i].e;
        for(int j=0;j<i;j++)
            if(meet(arr[i].t,arr[i].s,arr[i].e,arr[j].t,arr[j].s,arr[j].e))
                ans[i]++,ans[j]++;
    }
    for(int i=0;i<n;i++)
        cout<<ans[i]<<" \n"[i==n-1];
}
