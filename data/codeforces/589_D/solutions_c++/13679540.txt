#include<cstdio>
#include<algorithm>

using namespace std;

int N;
int ts[1010],ss[1010],fs[1010];

bool check(int t1,int s1,int f1,int t2,int s2,int f2){
	if(t1>t2){
		swap(t1,t2);
		swap(s1,s2);
		swap(f1,f2);
	}
	int d1=s1>f1?-1:1;
	int d2=s2>f2?-1:1;
	int a=s1+d1*(t2-t1);
	if(t2-t1>abs(f1-s1)) return false;
	if(a==s2) return true;
	if(d1==d2) return false;
	if(d1==-1){
		swap(a,s2);
		swap(f1,f2);
	}
	if(a>s2) return false;
	return s2-a<=min(s2-f2,f1-a)*2;
}

int get(int id){
	int res=0;
	for(int i=0;i<N;i++){
		if(i==id) continue;
		if(check(ts[id],ss[id],fs[id],ts[i],ss[i],fs[i])){
			res++;
		}
	}
	return res;
}

int ans[1010];

int main(){
	scanf("%d",&N);
	for(int i=0;i<N;i++){
		scanf("%d%d%d",ts+i,ss+i,fs+i);
	}
	for(int i=0;i<N;i++) ans[i]=get(i);
	for(int i=0;i<N;i++){
		printf("%d%c",ans[i],i==N-1?'\n':' ');
	}
	return 0;
}
