#define F first
#define S second
#define ll long long


using namespace std;
ll ans[50000],T[50000],t[50000],l[50000],r[50000],v[50000],n;
int main()
{
    cin >> n;
    for (ll i=1;i<=n;i++){
        cin >> t[i] >> l[i] >> r[i];
        if (r[i]>l[i]) v[i]=1; else v[i]=-1;
        T[i]=t[i]+abs(l[i]-r[i]);
    }
    for (ll i=1;i<=n;i++)
    for (ll j=1;j<=n;j++)
        if (i!=j) {
            if (v[i]==v[j] && l[i]==l[j]+v[j]*(t[i]-t[j]) && T[i]>=t[j] && T[j]>=t[i])
                ans[i]++;
            else
            if (v[i]!=v[j])
            {
                double k=l[i]-v[i]*t[i]-l[j]+v[j]*t[j], g=v[j]-v[i];
                double m=k/g,e=0.1;
                if (t[i]-e<=m && T[i]+e>=m &&
                    t[j]-e<=m && T[j]+e>=m) ans[i]++;
            }
    }
    for (ll i=1;i<=n;i++)
        cout << ans[i] << " ";
}
