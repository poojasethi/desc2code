#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <cmath>
#include <iostream>
#include <algorithm>
#define rep(i,x,y) for(int i=x;i<=y;++i)
#define mk make_pair
#define N 1200
using namespace std;
int a[N],t[N],s[N],f[N],d[N];
int n;
bool cross(int i,int j){
    int L=max(min(s[i],f[i]),min(s[j],f[j])),R=min( max(s[i],f[i]),max(s[j],f[j]) );
    if(L>R)return 0;
    int t1=t[i],t2=t[j];
    if(s[i]<=L)t1+=abs(L-s[i]);
        else t1+=abs(R-s[i]);

    if(s[j]<=L)t2+=abs(L-s[j]);
        else t2+=abs(R-s[j]);

    if(d[i]==d[j])return t1==t2;
    return abs(t1-t2)<=R-L;
}
int main(){
    scanf("%d",&n);
    rep(i,1,n){
        scanf("%d%d%d",&t[i],&s[i],&f[i]);
        if(s[i]<f[i])d[i]=1;
        else d[i]=-1;
    }
    rep(i,1,n)
        rep(j,i+1,n)if(cross(i,j)){a[i]++;a[j]++;}
    rep(i,1,n)printf("%d ",a[i]);
}
