#include <bits/stdc++.h>
using namespace std;
#define ll long long
struct T{
  double t,s,f;
  int v;
};
int f(T x, T y){
  T A,B;
  double d;
  if(x.t>y.t){
    A=y;
    B=x;
  }else{
    A=x;
    B=y;
  }
  d=A.s+abs(A.t-B.t)*A.v;
  if(d<min(A.s,A.f) || d>max(A.s,A.f)) return 0;
  A.s=d;
  if(A.v==B.v){
    if(A.s==B.s)return 1;
    return 0;
  }else{
    if((A.s<B.s && A.v==-1) || (A.s>B.s && A.v==1)) return 0;
    else{
      d=abs(A.s+B.s)/2;
      if((d>=min(A.s,A.f) && d<=max(A.s,A.f)) && (d>=min(B.s,B.f) && d<=max(B.s,B.f)))
        return 1;
      else return 0;
    } 
  }
  return 0;
}
int main(){
  int n;
  scanf("%d",&n);
  T v[n];
  int m[n];
  memset(m,0,sizeof(m));
  for(int i=0;i<n;i++){
    cin>>v[i].t>>v[i].s>>v[i].f;
    if(v[i].s<v[i].f)v[i].v=1;
    else v[i].v=-1;
  }
  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
      if(i!=j){
        m[i]+=f(v[i],v[j]);
      }
    }
  }
  for(int i=0;i<n;i++){
    if(i) cout<<" ";
    cout<<m[i];
  }
  cout<<endl;
  return 0;
}
