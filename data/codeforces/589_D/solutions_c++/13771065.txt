#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <queue>
#include <math.h>
#include <vector>
#include <map>
using namespace std;
const int N=10010;
int n,m,t[N],s[N],e[N],d[N],ans[N];
int t1,t2;
int gao(int i,int j)
{
	int L=max(min(s[i],e[i]),min(s[j],e[j]));
	int R=min(max(s[i],e[i]),max(s[j],e[j]));
	if(L>R) return 0;
	
	if(s[i]<=L)	t1=t[i]+L-s[i];
	else		t1=t[i]+s[i]-R;

	if(s[j]<=L)	t2=t[j]+L-s[j];
	else		t2=t[j]+s[j]-R;

	if(d[i]==d[j])	return	t1==t2;
	else			return	abs(t2-t1)<=R-L; 
}
int main ()
{
	while(scanf("%d",&n)!=EOF)
	{
		for(int i=1;i<=n;i++)
		{
			scanf("%d %d %d",&t[i],&s[i],&e[i]);
			if(s[i]<e[i]) d[i]=1;
			else d[i]=-1;
		}
		memset(ans,0,sizeof(ans));
		for(int i=1;i<=n;i++)
			for(int j=i+1;j<=n;j++)
				if(gao(i,j))
				{
					ans[i]++;
					ans[j]++;
				}
		for(int i=1;i<=n;i++)
			printf("%d ",ans[i]);
	}
	return 0;
}
