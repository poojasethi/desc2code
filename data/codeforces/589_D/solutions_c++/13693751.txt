#include<bits/stdc++.h>
#define TOPE 10000000
using namespace std;
typedef unsigned long long ull;

bool hello(int t1,int s1,int f1,int t2,int s2,int f2){
	int dir1 =1,dir2=1;
	if(f1<s1)dir1=-1;
	if(f2<s2)dir2=-1;
	if(t1>t2){
		if(t1-t2 > max(f2,s2) - min(f2,s2))return false;
		s2 += (t1-t2)*dir2;
	}else {
		if(t2-t1 > max(f1,s1) - min(f1,s1))return false;
		s1 += (t2-t1)*dir1;	
	}
	
	if(dir1==dir2 && s1==s2)return true;
	else{
		double p;
		if(dir1==-1){
			p = (s1-s2)/2.0;
		}else{
			p= (s2-s1)/2.0;
		}
		if(p<0)return false;
		if(dir1==1&&f1-s1>=p&&s2-f2>=p)return true;
		else if(dir1==-1&&s1-f1>=p&&f2-s2>=p)return true;
	}
	
	return false;
}
int main(){
	int n;
	cin>>n;
	vector<int>t(n),s(n),f(n),r(n);
	for(int i=0;i<n;++i)cin>>t[i]>>s[i]>>f[i];
	
	ull hi = 0;
	for(int i = 0;i<n-1;++i){
		for(int j=i+1;j<n;++j){
			if(hello(t[i],s[i],f[i],t[j],s[j],f[j]))r[i]++,r[j]++;
		}
	}
	for(int i = 0;i<n;++i){
		if(i>0)cout<<" ";
		cout<<r[i];
	}
	cout<<endl;
   return 0;
}
