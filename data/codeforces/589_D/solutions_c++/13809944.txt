#include <iostream>
#include <fstream>

using namespace std;

ifstream fin("input.txt");

int t[1001], s[1001], f[1001], a[1001];

bool greet(int i, int j)
{
	if (s[i] < f[i] && s[j] < f[j])
		return s[i] - t[i] == s[j] - t[j] && min(f[i], f[j]) >= max(s[i], s[j]);

	if (s[i] > f[i] && s[j] > f[j])
		return t[i] + s[i] == t[j] + s[j] && max(f[i], f[j]) <= min(s[i], s[j]);

	if (s[j] > f[j]) swap(i, j);

	if (s[j] > s[i]) return 0;

	int mid = s[i] + t[i] + s[j] - t[j];
	return mid <= s[i] * 2 && mid >= f[i] * 2 && s[j] * 2 <= mid && f[j]*2 >= mid;
}

int main()
{
	int n;

	cin >> n;

	for (int k = 0; k < n; ++k)
	{
		a[k] = 0;
	}

	for (int i = 0; i < n; ++i)
	{
		cin >> t[i] >> s[i] >> f[i];
		for (int j = 0; j < i; ++j)
		{
			if (greet(i , j))
			{
				a[i]++;
				a[j]++;
			}
		}
	}

	for (int l = 0; l < n; ++l)
	{
		cout << a[l] << ' ';
	}



	return 0;
}
