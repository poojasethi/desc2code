#include <iostream>*
#include <stdio.h>
#include <algorithm>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <vector>
using namespace std;
int n;
int t[1100],s[1110],f[1100];
bool in(int i, int j) {
    int sti=t[i], stj=t[j];
    int eti=t[i]+abs(f[i]-s[i]), etj=t[j]+abs(f[j]-s[j]);
    if(eti<stj || sti>etj) return false;
    int st=max(sti,stj),ed=min(eti,etj);
    int di=1,dj=1;
    if(f[i]<s[i])di=-1;
    if(f[j]<s[j])dj=-1;
    int b_st_i = s[i] + (st - t[i])*di;
    int b_ed_i = s[i] + (ed - t[i])*di;
    int b_st_j = s[j] + (st - t[j])*dj;
    int b_ed_j = s[j] + (ed - t[j])*dj;
    if((b_st_i-b_st_j)*(long long)(b_ed_i-b_ed_j)<=0)return true;
    return false;
}
int main()
{
    scanf("%d", &n);
    for(int i=0;i<n;i++) scanf("%d%d%d", &t[i],&s[i],&f[i]);
    for(int i=0;i<n;i++){
        int sum=0;
        for(int j=0;j<n;j++){
            if(i==j)continue;
            if(in(i,j)){
                    sum++;
            }
        }
        if(i==0)
        printf("%d", sum);
        else printf(" %d", sum);
    }
    printf("\n");
    return 0;
}
