#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <cstdlib>
#include <iostream>

using namespace std;

int N;
int T[1010]={0};
int S[1010]={0};
int F[1010]={0};
int ans[1010]={0};

bool check(int a,int b)
{
	if(T[a]>T[b]) swap(a,b);
	if(T[a]+abs(F[a]-S[a])<T[b]) return false;
	int va,vb;
	if(S[a]>F[a]) va=-1;
	else va=1;
	if(S[b]>F[b]) vb=-1;
	else vb=1;
	int xa=S[a]+(T[b]-T[a])*va,xb=S[b];
	if(va==vb)
	{
		if(xa==xb) return true;
		else return false;
	}
	int times=min((F[a]-xa)/va,(F[b]-xb)/vb);
	int ya=xa+times*va,yb=xb+times*vb;
	if(ya<xa) swap(ya,xa);
	if(yb<xb) swap(yb,xb);
	if(ya<xb || xa>yb) return false;
	else return true;
}

int main()
{
	cin>>N;
	for(int i=1;i<=N;i++)
	{
		scanf("%d%d%d",&T[i],&S[i],&F[i]);
		for(int j=1;j<i;j++)
		{
			if(check(i,j)==true)
				ans[i]++,ans[j]++;
		}
	}
	for(int i=1;i<=N;i++)
		printf("%d%c",ans[i],i==N?'\n':' ');
	return 0;
}
