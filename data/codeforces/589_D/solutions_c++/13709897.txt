#include<cstdio>
#define MAXN 1001
int t[MAXN],s[MAXN],f[MAXN];
int dir[MAXN];
int n;
void swap(int &x,int &y){int tmp=x;x=y;y=tmp;}
int main(){
    int n; scanf("%d",&n);
    for(int i=1;i<=n;++i)scanf("%d%d%d",&t[i],&s[i],&f[i]),dir[i]=s[i]<f[i];
    for(int i=1;i<=n;++i){
        int ans=0;
        for(int j=1;j<=n;++j){
            if(i==j)continue;
            int t1=t[i],s1=s[i],f1=f[i],t2=t[j],s2=s[j],f2=f[j];
            if(t1<t2){
                s1+=(t2-t1)*(dir[i]-0.5)*2;
                t1=t2;
                if((s1<f1)!=dir[i]&&s1!=f1)continue;
            }
            if(t1>t2){
                s2+=(t1-t2)*(dir[j]-0.5)*2;
                t2=t1;
                if((s2<f2)!=dir[j]&&s2!=f2)continue;
            }
            if(dir[i]==dir[j]){
                ans+=s1==s2;
                continue;
            }
            if(!dir[i]){
                swap(t1,t2); swap(s1,s2); swap(f1,f2);
            }
            if(s1>s2)continue;
            if((f1>=((double)(s1+s2)/2))&&(f2<=((double)(s1+s2)/2)))++ans;
        }
        printf("%d ",ans);
    }
}
