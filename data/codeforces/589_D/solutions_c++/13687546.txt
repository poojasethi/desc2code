#include <bits/stdc++.h>
using namespace std;

const int maxN=1000+100;
int a[maxN],t[maxN],s[maxN],f[maxN];
int n;

bool gap(int x1,int x2) {
    int d1=t[x1]+abs(s[x1]-f[x1]);
    int d2=t[x2]+abs(s[x2]-f[x2]);
    int l=max(t[x1],t[x2]);
    int r=min(d1,d2);
    int v1,v2;
    if (s[x1]<f[x1]) v1=1; else v1=-1;
    if (s[x2]<f[x2]) v2=1; else v2=-1;
    while (l<=r) {
        int mid=r;
        int sd1=s[x1]+(r-t[x1])*(v1);
        int sd2=s[x2]+(r-t[x2])*(v2);
        int td1=s[x1]+(l-t[x1])*v1;
        int td2=s[x2]+(l-t[x2])*v2;
        if (1ll*(sd1-sd2)*(td1-td2)<=0) return true;
        break;
    }
    return false;
}

int main() {
    //freopen("input.inp","r",stdin);
    scanf("%i\n",&n);
    for(int i=0;i<n;i++) scanf("%i%i%i",&t[i],&s[i],&f[i]);
    for(int i=0;i<n;i++) for(int j=i+1;j<n;j++)
        if (gap(i,j)) ++a[i],++a[j];
    for(int i=0;i<n;i++) printf("%i ",a[i]);
}
