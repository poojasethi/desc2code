#include <cstdio>
#include <cstring>
#include <algorithm>
#define ele int
using namespace std;
#define maxn 1010
struct ren{
	ele t,s,f,v;
};
ele n,r[maxn];
ren a[maxn];
inline ele abs(ele x){
	if (x>0) return x;
	return -x;
}
inline bool meet(ren a,ren b){
	if (a.t>b.t) swap(a,b);
	ele p=a.s+a.v*(b.t-a.t);
	if ((long long)(a.s-p)*(long long)(a.f-p)>0) return false;
	if (p==b.s) return true;
	if (a.v==b.v) return false;
	if ((long long)(p-b.s)*(a.v-b.v)>0) return false;
	ele k=abs(p+b.s);
	if (k&1){
		double d=(double)k/2;
		if ((d-a.s)*(d-a.f)<=0 && (d-b.s)*(d-b.f)<=0) return true;
	}
	else{
		ele d=k>>1;
		if ((long long)(a.s-d)*(long long)(a.f-d)<=0 && (long long)(b.s-d)*(long long)(b.f-d)<=0) return true;
	}
	return false;
}
int main(){
	scanf("%d",&n);
	for (int i=0; i<n; ++i){
		scanf("%d%d%d",&a[i].t,&a[i].s,&a[i].f);
		if (a[i].s<a[i].f) a[i].v=1; else a[i].v=-1;
	}
	memset(r,0,sizeof(r));
	for (int i=0; i<n-1; ++i)
		for (int j=i+1; j<n; ++j)
			if (meet(a[i],a[j])) ++r[i],++r[j];
	for (int i=0; i<n-1; ++i) printf("%d ",r[i]);
	printf("%d\n",r[n-1]);
	return 0;
}
