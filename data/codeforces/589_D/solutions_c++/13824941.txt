#include <cstdio>

int n;

struct node{
	int t,s,f;
	int d;
	void operator = (node n2){
		t=n2.t;
		s=n2.s;
		f=n2.f;
		d=n2.d;
	}
}a[1100];

bool check(int i,int j){
	int t1,t2;
	if (i==j) return false;
	node n1=a[i],n2=a[j];
	if (n1.t<n2.t) t1=n2.t;
	else t1=n1.t;
	t2=(n1.t+(n1.f-n1.s)*n1.d)<(n2.t+(n2.f-n2.s)*n2.d)?(n1.t+(n1.f-n1.s)*n1.d):(n2.t+(n2.f-n2.s)*n2.d);
	if (t1>t2) return false;
	int p1,p2,p3,p4;
	p1=n1.s+(t1-n1.t)*n1.d;
	p2=n2.s+(t1-n2.t)*n2.d;
	p3=n1.s+(t2-n1.t)*n1.d;
	p4=n2.s+(t2-n2.t)*n2.d;
	if (n1.d==n2.d)
		if (p1==p2&&p3==p4) return true;
		else return false;
	else
		if ((p1<=p2&&p3>=p4)||(p1>=p2&&p3<=p4)) return true;
		else return false;
}

int main()
{
	scanf("%d",&n);
	for (int i=0;i<n;i++)
	{
		scanf("%d%d%d",&a[i].t,&a[i].s,&a[i].f);
		if (a[i].f==a[i].s) a[i].d=0;
		if (a[i].f>a[i].s) a[i].d=1;
		if (a[i].f<a[i].s) a[i].d=-1;
	}
	for (int i=0;i<n;i++){
		int ans=0;
		for (int j=0;j<n;j++)
			if (check(i,j)){
				ans++;
				//printf("%d %d\n",i,j);
			}
		if (i) printf(" %d",ans);
		else printf("%d",ans);
	}
	printf("\n");
}
