#include<cstdio>
int t[1010],s[1010],f[1010],v[1010],r[1010];
int main(){
	int n;scanf("%d",&n);
	int i;
	for(i=0;i<n;i++){
		scanf("%d%d%d",&t[i],&s[i],&f[i]);
		if(s[i]<f[i])v[i]=1;
		else v[i]=-1;
	}
	int j,x,t1,t2,t3;
	for(i=0;i<n;i++)
		for(j=i+1;j<n;j++){
			if(v[i]==v[j]){
				if(t[i]<=t[j]){
					x=s[i]+v[i]*(t[j]-t[i]);
					if(x==s[j]&&
					((v[i]==1&&x<=f[i])
					||(v[i]==-1&&x>=f[i]))){
						r[i]++;r[j]++;
					}
				}
				else{
					x=s[j]+v[j]*(t[i]-t[j]);
					if(x==s[i]&&
                    ((v[j]==1&&x<=f[j])
                    ||(v[j]==-1&&x>=f[j]))){
                        r[i]++;r[j]++;
                    }
				}
			}
			else{
                t1=(s[j]-s[i])/v[i]+t[i]+t[j];
                t2=((f[i]-s[i])/v[i]+t[i])<<1;
                t3=((f[j]-s[j])/v[j]+t[j])<<1;
                if(2*t[i]<=t1&&2*t[j]<=t1&&t1<=t2&&t1<=t3){
                    r[i]++;r[j]++;
                }
            }
		}
    printf("%d",r[0]);
    for(i=1;i<n;i++)printf(" %d",r[i]);
	return 0;
}
