#include <iostream>
#include <algorithm>
#include <map>
#include <string>
using namespace std;

int n;
int t[1111];
int s[1111];
int f[1111];
int sol[1111];
int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>t[i]>>s[i]>>f[i];
	}

	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			if(i==j){
				continue;
			}

			if( (s[i]<f[i] && s[j]<f[j])   ){
				if(s[i]<=s[j]){
				if(t[i]-s[i] == t[j]-s[j] && f[i]>=s[j])sol[i]++;
				}
				else
					if(t[i]-s[i] == t[j]-s[j] && s[i]<=f[j])sol[i]++;

			} else if(s[i]>f[i] && s[j]>f[j]){
				if(s[i]<=s[j]){
					if( s[i]+t[i] == s[j]+t[j] && f[j]<=s[i])sol[i]++;
				}else if( s[i]+t[i] == s[j]+t[j] && f[i]<=s[j])sol[i]++;
			} else {

				int ind1,ind2;
				if(s[i]<f[i]){
					ind1=i;
					ind2=j;
				} else {
					ind1=j;
					ind2=i;
				}
				int dd=s[ind1]+t[ind2]+s[ind2]-t[ind1];

				int tt=dd-s[ind1]+t[ind1];

				if( max(min(s[i],f[i]),min(s[j],f[j]))*2  <= dd && dd <= 2*min(max(s[i],f[i]),max(s[j],f[j]))){
					if( tt>= t[i] && tt>=t[j])sol[i]++;
				}
			}

		}
	}
	for(int i=0;i<n;i++){
		cout<<sol[i]<< " ";
	}
}
