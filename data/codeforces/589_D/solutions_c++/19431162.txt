#include <bits/stdc++.h>

using namespace std;

#define NN 1100
#define inf 0x3f3f3f3f
#define eps 1e-8
#define eprintf(...) fprintf(stderr, __VA_ARGS__)

typedef pair<int, int> pii;
typedef long long INT;

int N;
int t[NN], s[NN], f[NN];

int sgn(int t) {
	if (t<0) return -1;
	if (t==0) return 0;
	return 1;
}
bool meet(int u, int v) {
	int l=max(min(s[u], f[u]), min(s[v], f[v]));
	int r=min(max(s[u], f[u]), max(s[v], f[v]));
	
	if (r<l) return false;
	
	int tul=(l-s[u])*sgn(f[u]-s[u])+t[u];
	int tur=(r-s[u])*sgn(f[u]-s[u])+t[u];
	int tvl=(l-s[v])*sgn(f[v]-s[v])+t[v];
	int tvr=(r-s[v])*sgn(f[v]-s[v])+t[v];
	
	return (INT)(tul-tvl)*(tur-tvr)<=0;
}

int main() {
	scanf("%d", &N);
	for (int i=0; i<N; i++) scanf("%d%d%d", &t[i], &s[i], &f[i]);
	
	for (int i=0; i<N; i++) {
		int cnt=0;
		for (int j=0; j<N; j++) if (i!=j && meet(i, j)) cnt++;
		printf("%d ", cnt); 
	}
	puts("");
	
	return 0;
}
