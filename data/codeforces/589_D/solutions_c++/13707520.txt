#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#include <string.h>

struct human {
        long long int t,f,s;
        long long int x1,y1,x2,y2;
        };

typedef struct human HUMAN;

int n;
HUMAN input[1000];

long long  fy(HUMAN h, long long  x){
	if (h.y1<h.y2)
		return h.y1 + (x-h.x1);
	else 
		return h.y1 + (h.x1-x);
}

int inter(int i, int j){
	long long int maxx1 = input[i].x1>input[j].x1?input[i].x1:input[j].x1;
	long long int minx2 = input[i].x2<input[j].x2?input[i].x2:input[j].x2;
	if (maxx1>minx2) return 0;
	long long  a=fy(input[i],maxx1) - fy(input[j],maxx1);
	long long  b=fy(input[i],minx2) - fy(input[j],minx2);
	if (a*b<=0) return 1;
	return 0;
}

main(){
	scanf("%d",&n);
	for (int i=0;i<n;i++){
		scanf("%I64d %I64d %I64d",&input[i].t,&input[i].s,&input[i].f);
		input[i].x1=input[i].t;
		input[i].y1=input[i].s;
		input[i].x2=input[i].t+abs(input[i].s-input[i].f);
		input[i].y2=input[i].f;
	}
	
	for (int i=0;i<n;i++){
		int count=0;
		for (int j=0;j<n;j++)
			if (i!=j && inter(i,j)) count++;
		printf("%d ",count);
	}
}
