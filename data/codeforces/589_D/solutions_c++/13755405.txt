#include <bits/stdc++.h>
using namespace std;
const int MAXN = 1005;
int t[MAXN], s[MAXN], f[MAXN], ans[MAXN];
int get_time(int sp, int st, int fp)
{
    return st + abs(sp-fp);
}
int main()
{
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; ++i)
    {
        scanf("%d %d %d", &t[i], &s[i], &f[i]);
    }
    for (int i = 0; i < n; ++i)
    {
        for (int j = i+1; j < n; ++j)
        {
            int l1 = min(s[i],f[i]), r1 = max(s[i],f[i]);
            int l2 = min(s[j],f[j]), r2 = max(s[j],f[j]);
            int l = max(l1,l2), r = min(r1,r2);
            if(l > r)
                continue;
            int t1_l = get_time(s[i],t[i],l);
            int t1_r = get_time(s[i],t[i],r);
            int t2_l = get_time(s[j],t[j],l);
            int t2_r = get_time(s[j],t[j],r);
            if((t1_l <= t2_l && t1_r >= t2_r) || (t1_l >= t2_l && t1_r <= t2_r))
            {
                ans[i]++;
                ans[j]++;
            }
        }
    }
    for (int i = 0; i < n; ++i)
        printf("%d ", ans[i]);
    return 0;
}
