#include <cstdio>
#include <algorithm>

using namespace std;

#define MAXN 1005

int t[MAXN],s[MAXN],f[MAXN],ans[MAXN];

bool check(int t1, int s1, int f1, int t2, int s2, int f2){
  int t12,t22;

  if(s1 < f1){
    t12 = t1 + f1 - s1;
    s1 -= t1;
  }else{
    t12 = t1 + s1 - f1;
    s1 += t1;
  }

  if(s2 < f2){
    t22 = t2 + f2 - s2;
    s2 -= t2;
  }else{
    t22 = t2 + s2 - f2;
    s2 += t2;
  }

  if(s1 < f1 && s2 < f2){
    return s1 == s2 && max(t1,t2) <= min(t12,t22);
  }else if(s1 < f1){
    if(s1 > s2) return false;
    return 2 * max(t1,t2) <= (s2 - s1) && (s2 - s1) <= 2 * min(t12,t22);
  }else if(s2 < f2){
    if(s2 > s1) return false;
    return 2 * max(t1,t2) <= (s1 - s2) && (s1 - s2) <= 2 * min(t12,t22);
  }else{
    return s1 == s2 && max(t1,t2) <= min(t12,t22);
  }
}

int main(){
  int n;

  scanf("%d",&n);

  for(int i = 0;i < n;++i){
    scanf("%d %d %d",&t[i],&s[i],&f[i]);

    for(int j = 0;j < i;++j){
      if(check(t[i],s[i],f[i],t[j],s[j],f[j])){
        ++ans[i];
        ++ans[j];
      }
    }
  }

  for(int i = 0;i < n;++i)
    printf("%d ",ans[i]);
  printf("\n");

  return 0;
}
