#include <iostream>

using namespace std;
const  int maxn = 1005;
int t[maxn], x[maxn], y[maxn], d[maxn];

bool check(int i, int j) {
    if (x[i] == x[j] && t[i] == t[j]) return true;
    if (x[i] > x[j]) swap(i,j);
    if (d[i] * d[j] == 1) {
        if (d[i] == 1) {
            return (t[j] - t[i] == x[j] - x[i] && x[j] <= y[i]);
        }
        else {
            return (t[i] - t[j] == x[j] - x[i] && x[i] >=  y[j]);
        }
    }
    else {
        if (d[i] == -1) return false;
        float tmp = t[j] + x[j] + x[i] - t[i];
        if (tmp < 0) return false;
        tmp = tmp/2;
        return (x[i] <= tmp && tmp <= y[i] && x[j] >= tmp && tmp >= y[j]);
    }
}

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> t[i] >> x[i] >> y[i];
        d[i] = x[i] < y[i] ? 1 : -1;
    }
    for (int i = 0; i < n; ++i) {
        int res = 0;
        for (int j = 0; j < n; ++j)
            if (i != j && check(i,j)) res++;
        cout << res << " ";
    }
    return 0;
}
