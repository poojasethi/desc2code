#include <bits/stdc++.h>

#define FOR(a,b,c) for(int a=(b),_for=(c);a<_for;++a)
#define ll long long
#define vi vector < int >

#define PCNT(a) __builtin_popcount(a);

using namespace std;

const int MAXN = 1000;

class c{
public:
	int a,b,dir,t;
};

int n;
c p[MAXN+5];

int ABS(int a){ return a>0?a:-a; }

bool f(c a,c b){
	if (a.t>b.t) swap(a,b);
	int dif=b.t-a.t;
	if (dif>ABS(a.b-a.a)) return 0;
	a.a+=dif*a.dir;
	if (a.dir==b.dir){
		if (a.a!=b.a) return 0;
		if (a.dir>0 && a.a>a.b) return 0;
		if (a.dir<0 && a.a<a.b) return 0;
		return 1;
	}
	int m=a.a+b.a;
	a.a*=2,a.b*=2,b.a*=2,b.b*=2;
	if (a.a<m && a.b<m) return 0;
	if (a.a>m && a.b>m) return 0;
	if (b.a<m && b.b<m) return 0;
	if (b.a>m && b.b>m) return 0;
	return 1;
}

int main(){
	scanf ("%d",&n);
	FOR(i,0,n){
		scanf ("%d%d%d",&p[i].t,&p[i].a,&p[i].b);
		if (p[i].a<p[i].b) p[i].dir=1;
		else p[i].dir=-1;
	}
	FOR(i,0,n){
		int sol=0;
		FOR(j,0,n){
			if (i==j) continue;
			if (f(p[i],p[j])) sol++;
		}
		printf ("%d\n",sol);
	}
	return 0;
}
