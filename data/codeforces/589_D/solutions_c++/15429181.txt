#include <bits/stdc++.h>
using namespace std;
const int N = 1e3;
int ret[N];
void add(int i,int j){
	++ret[i],++ret[j];
}
struct pp{
	int t,s,f,d;
 	pp(){}
	void init(){
		cin>>t>>s>>f;
		d=1;if(s>f)d=-1;
	}
}a[N];
int main(){
	int u,v,i,n,j,c,d,maxi,last,st;
	cin>>n;pp x,y;
	for(i=0;i<n;++i)a[i].init();
	for(i=0;i<n;++i){
		for(j=i+1;j<n;++j){
			y= a[j],x=a[i];
			if(x.t>y.t)swap(x,y);
			if(x.d==y.d){
				st=y.t-x.t;
				assert(st>=0);
				if( (x.d*st +x.s)== y.s && abs(x.f-x.s)>=st)
					add(i,j);
			}
			else{
				if(x.s>x.f)swap(x,y);
				int mid = x.s+y.s+y.t-x.t;
				if( x.s*2<=mid && mid <= x.f*2 && y.f*2<=mid && mid <= 2*y.s)
					add(i,j);
			}
		}
	}
	for(i=0;i<n;++i)cout<<ret[i]<<" ";
	cout<<endl;
}
