#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int, int> ii;
typedef vector<int> vi;
typedef vector<ii> vii;

int N, t[1010], s[1010], f[1010];

int main() {
	int i, j, ct[1010];
	scanf("%d", &N);
	for (i=0; i<N; i++) {
		scanf("%d %d %d", &t[i], &s[i], &f[i]);
	}
	memset(ct, 0, sizeof ct);
	for (i=0; i<N; i++) {
		int st1 = t[i];
		int et1 = t[i] + abs(f[i] - s[i]);
		int d1 = (f[i] > s[i] ? 1 : -1);
		for (j=i+1; j<N; j++) {
			int st2 = t[j];
			int et2 = t[j] + abs(f[j] - s[j]);
			int d2 = (f[j] > s[j] ? 1 : -1);
			if (et1 < st2 || st1 > et2) continue;
			int st = max(st1, st2);
			int et = min(et1, et2);
			int s1 = s[i]+d1*(st-st1);
			int s2 = s[j]+d2*(st-st2);
			int e1 = s1+d1*(et-st);
			int e2 = s2+d2*(et-st);
			if ((s1 >= s2 && e1 <= e2) ||
				(s1 <= s2 && e1 >= e2)) {
				ct[i]++;
				ct[j]++;
			}
		}
	}
	for (i=0; i<N; i++) {
		if (i) printf(" ");
		printf("%d", ct[i]);
	}
	printf("\n");
	return 0;
}
