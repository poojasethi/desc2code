#include <stdio.h>


struct walker {
	int t;
	int s;
	int f;
	int d;//direction
	int count;

}m[1000], fwd, bck;//men
int main() {
	int N, count=0, ci, cj, t;
	double x;
	scanf("%d", &N);
	for (int i = 0; i < N; i++) {
		scanf("%d%d%d", &m[i].t, &m[i].s, &m[i].f);
		m[i].d = m[i].s <= m[i].f ? 1 : -1;
	}
	for (int i = 0; i < N; i++) {
		count = 0;
		for (int j = 0; j < N; j++) {
			if (i == j)continue;
			ci = m[i].s - m[i].d*m[i].t;
			cj = m[j].s - m[j].d*m[j].t;
			if (m[i].d != m[j].d) {
				x = (ci + cj)/2.0;// crossing point (x, t)
				//t = x - ci;
				if (m[i].d == 1) { fwd = m[i]; bck = m[j]; }
				else { fwd = m[j]; bck = m[i]; }
				if (x >= fwd.s&&x <= fwd.f&&x <= bck.s&&x >= bck.f)count++;
			}
			else {
				if (ci == cj) {
					if (m[i].d == 1) {
						if (m[i].f >= m[j].s&&m[i].f <= m[j].f || m[j].f >= m[i].s&&m[j].f <= m[i].f)count++;
					}
					else if (m[i].f <= m[j].s&&m[i].f >= m[j].f || m[j].f <= m[i].s&&m[j].f >= m[i].f)count++;
				}
			}

		}
		printf("%d ", count);
	}
			



	return 0;
}
