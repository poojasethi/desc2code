#include <bits/stdc++.h>
#define maxn 1009
#define inf 0x3f3f3f3f
using namespace std;

int main()
{
    // freopen("d:\\in.txt","r",stdin);
    int t[maxn],f[maxn],s[maxn],n,v[maxn];
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d%d%d",&t[i],&s[i],&f[i]);
        if(s[i]<=f[i])
            v[i]=1;
        else
            v[i]=-1;
    }
    int ans[maxn];
    memset(ans,0,sizeof(ans));
    for(int i=1;i<=n;i++)
    {
        for(int j=i+1;j<=n;j++)
        {
          //printf("i=%d j=%d\n",i,j);
              int t11=t[i],t12=abs(f[i]-s[i])+t[i];
              int t21=t[j],t22=abs(f[j]-s[j])+t[j];
              int tx=max(t11,t21);
              int ty=min(t22,t12);
              if(tx>ty)
                continue;
              long long l1=s[i]+(tx-t[i])*v[i],l2=s[i]+(ty-t[i])*v[i];
              long long ll1=s[j]+(tx-t[j])*v[j],ll2=s[j]+(ty-t[j])*v[j];
              if((l1-ll1)*(l2-ll2)<=0)
              {
                  ans[i]++;
                  ans[j]++;
              }
        }
    }
    for(int i=1;i<=n;i++)
        printf("%d ",ans[i]);
    return 0;
}
