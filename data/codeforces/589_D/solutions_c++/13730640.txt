#include <bits/stdc++.h>

#define ll long long
#define fi freopen("", "r", stdin)
#define fo freopen("", "w", stdout)

using namespace std;

const int N = 1010;

int n;
int t[N], s[N], f[N], ans[N], d[N];

int check(int i, int j)
{
  int L = max(min(s[i],f[i]),min(s[j],f[j]));
  int R = min(max(s[i],f[i]),max(s[j],f[j]));
    if(L > R)return 0;
    int t1 = t[i], t2 = t[j];
    if(s[i] <= L) t1 += abs(L-s[i]);
        else t1 += abs(R-s[i]);

    if(s[j] <= L) t2 += abs(L-s[j]);
        else t2 += abs(R-s[j]);

    if(d[i] == d[j]) return t1==t2;
    return abs(t1-t2) <= (R-L);
}

int main()
{
  fi;fo;
  scanf("%d", &n);
  for(int i = 1; i <= n; i++) {
    scanf("%d%d%d", &t[i], &s[i], &f[i]);
    d[i] = -1;
    if(s[i] < f[i]) d[i] = 1;
  }
  memset(ans, 0, sizeof(ans));
  for(int i = 1; i <= n; i++)
    for(int j = i + 1; j <= n; j++)
      if(check(i, j)) {ans[i]++; ans[j]++;}
  for(int i = 1; i <= n; i++) cout << ans[i] << " ";
  return 0;
}
