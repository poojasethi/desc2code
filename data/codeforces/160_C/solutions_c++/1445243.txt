#include<iostream>
#include<algorithm>
using namespace std;

int n,i=0,a[1<<17];
long long k,t;

int main()
{
	for(cin >> n >> k;i<n;++i) cin >> a[i];
	sort(a,a+n);--k;
	for(i=0;i<n;i+=t)
	{
		for(t=1;a[i]==a[i+t];++t);
		if (t*n>k) break;
		k-=t*n;
	}
	cout << a[i] << ' ' << a[k/t] << endl;
	return 0;
}
