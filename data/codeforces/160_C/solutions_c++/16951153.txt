#include <bits/stdc++.h>
using namespace std;
int arr[100001],n;
long long k,t;
main()
{
    cin>>n>>k;
    int i ;
    for( i=0 ; i<n; ++i)
        cin>>arr[i];
    sort(arr,arr+n);
    --k;
    for(i=0; i<n; i+=t)
    {
      for(t=1; arr[i]==arr[i+t]; ++t);
        if(t*n>k)break;
        k-=t*n;
    }
    cout<<arr[i]<<" "<<arr[k/t];
    return 0;
}
