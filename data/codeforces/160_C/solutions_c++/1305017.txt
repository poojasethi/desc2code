#include <iostream>
#include <algorithm>
#include <map>

using namespace std;

int a[100100];

map<int,int> l,c;

int main() {
	int n;
	long long k;
	cin >> n >> k; k--;
	for(int i=0;i<n;i++) cin >> a[i];
	sort(a,a+n);
	for(int i=n-1;i>=0;i--) l[a[i]]=i, c[a[i]]++;
	int x=k/n;
	k -= (long long)l[a[x]]*n;
	k /= c[a[x]];
	cout << a[x] << " " << a[k] << endl;
	return 0;
}
