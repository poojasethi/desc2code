#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,i;
long long a[100010],k;
	while(cin>>n>>k)
	{
		for(i=0;i<n;i++)
			cin>>a[i];
		sort(a,a+n);
	long	long l=0,s=0;
		for(i=0;i<n;i++)
		{
			if(a[i]==a[(k-1)/n])
				l++;
			if(a[i]<a[(k-1)/n])
				s++;
		}
		cout<<a[(k-1)/n]<<" "<<a[(k-1-s*n)/l]<<endl;
	}
	return 0;
}
