#include<iostream>
#include<algorithm>
using namespace std;
long long int S[1000000];
int main(){
    long long int N,K,r,s;
    cin >> N >> K;
    K--;
    for(long long int i=0;i<N;i++) cin >> S[i];
    sort(S,S+N);
    cout << S[K/N] << ' ';
    r=lower_bound(S,S+N,S[K/N])-S;
    s=upper_bound(S,S+N,S[K/N])-S;
    K-=N*r;
    K/=(s-r);
    cout << S[K];
}
