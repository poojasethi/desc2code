#include<stdio.h>
#include<algorithm>
using namespace std;
long long a[100005];
int main()
{
    long long n,k,i,t;
    scanf("%I64d%I64d",&n,&k);
    k--;
    for(i=0;i<n;i++)
    scanf("%I64d",&a[i]);
    sort(a,a+n);
    for(i=0;i<n;i+=t)
    {
        for(t=1;a[i]==a[i+t];t++);
        if(t*n>k) break;
        k-=(t*n);
    }
    printf("%I64d %I64d",a[i],a[k/t]);
}
