#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll a[100005],i,n,k,t;
main()
{
    scanf("%I64d%I64d",&n,&k);
    for (i=0;i<n;i++) scanf("%I64d",&a[i]);
    sort(a,a+n);k--;
    for (i=0;i<n;i+=t)
    {
        for (t=1;a[i]==a[i+t];++t);
           if (t*n>k) break;
        k-=t*n;
    }
    printf("%I64d %I64d",a[i],a[k/t]);
}
