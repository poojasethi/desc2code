#include<cstdio>
#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
__int64 a[1000001],n,k,cnt,num;
int main(){
  scanf("%I64d%I64d",&n,&k);
  for(int i=0;i<n;++i)
    scanf("%I64d",&a[i]);
  sort(a,a+n);
  k--;
  for(int i=0;i<n;++i){
    if(a[i]<a[k/n])
      ++cnt;
    if(a[i]==a[k/n])
      ++num;
  }
  printf("%I64d %I64d",a[k/n],a[(k-n*cnt)/num]);
}
