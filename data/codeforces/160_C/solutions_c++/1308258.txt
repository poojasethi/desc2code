#include <iostream>
#include <algorithm>
using namespace std;
int X[100000];
int main(void)
{
    int i,n;
    long long k,t=0,r=0;
    cin>>n>>k;
    --k;
    for (i=0;i<n;++i)
        cin>>X[i];
    sort(X,X+n);
    for (i=0;i<n;++i)
        if (X[i]==X[k/n])
            r++;
        else if (X[i]<X[k/n])
            t++;
    cout<<X[k/n]<<" "<<X[(k-t*n)/r];
    return 0;
}
