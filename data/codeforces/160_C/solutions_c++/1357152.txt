#include <iostream>
#include <algorithm>

using namespace std;

long long N,K,A[100010],P1,P2,cnt,num;

int main()
{
	cin >> N >> K;
	for (int i=0;i<N;i++)
		cin >> A[i];
	sort(A,A+N);
	P1=A[(K-1)/N];
	for (int i=0;i<N;i++)
	{
		if (A[i]<P1)
			cnt++;
		else if (A[i]==P1)
			num++;
	}
	P2=A[(K-N*cnt-1)/num];
	cout << P1 << " " << P2 << endl;
	return 0;
}
