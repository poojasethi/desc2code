#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long n,k,less=0,eq=0,a[100010];
    scanf("%I64d%I64d",&n,&k);
    for(int i=0;i<n;i++)
        scanf("%I64d",&a[i]);
    sort(a,a+n);
    k--;
    for(int i=0;i<n;i++)
    {
        if(a[i]<a[k/n]) less++;
        if(a[i]==a[k/n]) eq++;
    }
    printf("%I64d %I64d\n",a[k/n],a[(k-n*less)/eq]);
    return 0;
}
