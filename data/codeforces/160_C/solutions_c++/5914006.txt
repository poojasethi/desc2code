#include<cstdio>
#include<algorithm>
using namespace std;

int a[1<<17],i,n;
__int64 k,t;

int main(){
	while(~scanf("%d%I64d",&n,&k))
    {
		for(i = 0;i<n;++i)
			scanf("%d",a+i);
		sort(a,a+n);
		--k;
		for(i=0;i<n;i+=t){
			for(t=1;a[i]==a[i+t];++t);    
			if(t*n>k)break;k-=t*n;   
		}
		printf("%d %d\n",a[i],a[k/t]);
    }
	return 0;
}
