#include <bits/stdc++.h>
using namespace std;

#define ll long long
const int N=1e5+5;

ll v[N],x,y;


int main()
{
	ll n,k;
	cin>>n>>k;k--;
	for (ll i=0;i<n;i++)
		cin>>v[i];
	sort(v,v+n);
	x=y=0;
	for (int i=0;i<n;i++){
		if (v[i]<v[k/n]) x++;
		if (v[i]==v[k/n]) y++;
	}
	cout<<v[k/n]<<" "<<v[(k-n*x)/y]<<endl;
}
