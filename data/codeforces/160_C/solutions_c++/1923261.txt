#include<iostream>
#include<algorithm>
using namespace std;
int s[100005];
int main()
{
	__int64 n,k; 
	while(cin>>n>>k)
	{
		int i;
		for(i=0;i<n;i++)
			cin>>s[i];
		sort(s,s+n);
		__int64 a,b;
		a=(k-1)/n;
		__int64 num=0;
		for(i=n-1;i>=0;i--)
			if(s[i]==s[a])
				num++,a=i;
		b=(k-n*a-1)/num;
		cout<<s[a]<<" "<<s[b]<<endl;
	}
	return 0;
}
