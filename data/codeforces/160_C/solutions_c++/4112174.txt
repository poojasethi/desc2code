#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	long long k,a[100010],n,i,x,y;
	x=y=0;
	cin>>n>>k;
	k--;
	for(i=0;i<n;i++)
		cin>>a[i];
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		if(a[i]<a[k/n])
			x++;
		if(a[i]==a[k/n])
			y++;
	}
	cout<<a[k/n]<<' '<<a[(k-n*x)/y]<<endl;
}
