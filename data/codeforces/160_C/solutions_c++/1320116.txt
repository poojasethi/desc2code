#include<cstdio>
#include<algorithm>
using namespace std;
int arr[100005];
int main()
{
    int n;
    __int64 i,j,k,w;
    scanf("%d%I64d",&n,&k);
    k--;
    for(i=0;i<n;i++)scanf("%d",arr+i);
    sort(arr,arr+n);
    for(i=w=0;arr[i]<arr[k/n];i++,w++);
    for(j=0;arr[i]==arr[k/n];i++,j++);
    printf("%d %d\n",arr[k/n],arr[(k-w*n)/j]);
    return 0;
}
