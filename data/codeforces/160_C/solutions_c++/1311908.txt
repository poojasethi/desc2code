#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	long long n,k;
	cin>>n>>k;
	int a[100001],c=0,u;
	for (int i=0;i<n;++i) cin>>a[i];
	sort(a,a+n);
	for (;a[c]<a[(k-1)/n];++c);
	for (u=c;u<n && a[u]==a[(k-1)/n];++u);
	cout<<a[(k-1)/n]<<' '<<a[(k-n*c-1)/(u-c)]<<endl;
	return 0;
}
