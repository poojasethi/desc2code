#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod  1000000007;
int main(){
ll n,k;
cin>>n>>k;
ll a[n];
for(int i=0;i<n;i++) cin>>a[i];
sort(a,a+n);
for(int i=0;i<n;i++){
int t=1;
for(t=1;i+t<n;t++){
if(a[i]!=a[i+t]) break;
}
if(k<=n*t){
  cout<<a[i]<<" "<<a[(k-1)/t];
  break;
}
k-=n*t;
i=i+t-1;
}
}
