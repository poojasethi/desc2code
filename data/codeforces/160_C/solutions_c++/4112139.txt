#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	long long a[100010],x,y,k;
	int n,i;
	x=y=0;
	cin>>n>>k;
	k--;
	for(i=0;i<n;i++)
		cin>>a[i];
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		if(a[i]<a[k/n])
			x++;
		if(a[i]==a[k/n])
			y++;
	}
	cout<<a[k/n]<<' '<<a[(k-n*x)/y]<<endl;
}
