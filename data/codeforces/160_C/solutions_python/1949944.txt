def r(): return raw_input().strip()
def ri(): return int(r())
def riv(): return map(int, r().split())

from itertools import product

def main():
    n, k = riv() 
    a = list(sorted(riv()))

    fpos = (k-1)/n
    c = 0
    for i in reversed(range(n)):
        if a[i] == a[fpos]:
           c += 1
           fpos = i

    spos = (k-1) % n
    for i in range(n):
        if (fpos*n + c*(i+1) >= k):
            print a[fpos], a[i]
            exit(0)
    
if __name__ == "__main__":
    main()
