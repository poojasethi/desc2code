def init():
    n, k = map(int, raw_input().split())
    a = map(int, raw_input().split())
    a.sort()
    return n, k, a

def solve(n, k, a):
    p = (k - 1) / n
    tc = 0
    lc = 0
    for i in xrange(0, n):
        if a[i] == a[p]:
            tc = tc + 1
        elif a[i] < a[p]:
            lc = lc + 1
    k = k - n * lc
    q = (k - 1) / tc
    return [a[p], a[q]]

if __name__ == "__main__":
    n, k, a = init()
    pair = solve(n, k, a)
    print pair[0], pair[1]
