# main

n, k = map(int, raw_input().split())
cnt = {}
for a in map(int, raw_input().split()):
    if cnt.has_key(a):
        cnt[a] += 1
    else:
	cnt[a] = 1

key = sorted(cnt.iterkeys())
for a in key:
    if k <= cnt[a] * n:
	# first is a
	# find second
        for b in key:
            if k <= cnt[a] * cnt[b]:
		print a, b
		break
            else:
		k -= cnt[a] * cnt[b]
	break
    else:
	k -= cnt[a] * n
