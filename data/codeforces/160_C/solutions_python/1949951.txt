import math

from operator import attrgetter, itemgetter, methodcaller, add, gt, lt
from collections import OrderedDict, Counter
from decimal import Decimal, getcontext
getcontext().prec = 7

def r(): return raw_input().strip()
def ri(): return int(r())
def riv(): return map(int, r().split())

from itertools import product

def main():
    n, k = riv() 
    a = list(sorted(riv()))
    k -= 1
    
    fpos = k / n
    count = a.count(a[fpos])
    k -= a.index(a[fpos]) * n
    print a[fpos], a[k / count]
    
    
if __name__ == "__main__":
    main()
