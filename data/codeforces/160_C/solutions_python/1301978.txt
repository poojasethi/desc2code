n, k = map(int, raw_input().split())
a = map(int ,raw_input().split())
a.sort()
same = [1] * n
for i in xrange(n):
    if i > 0 and a[i] == a[i-1]:
        same[i] = same[i-1] + 1
    else:
        same[i] = 1
p1 = (k - 1) / n
cnt1 = same[p1]
p11 = p1 + 1
while p11 < n and a[p11] == a[p1]:
    p11 += 1
    cnt1 += 1
p0 = p11 - cnt1
p2 = (k - n*p0 - 1)/cnt1
print a[p1], a[p2]
