def r(): return raw_input().strip()
def riv(): return map(int, r().split())

def main():
    n, k = riv() 
    a = list(sorted(riv()))
    k -= 1
    
    fpos = k / n
    count = a.count(a[fpos])
    k -= a.index(a[fpos]) * n
    print a[fpos], a[k / count]
    
    
if __name__ == "__main__":
    main()
