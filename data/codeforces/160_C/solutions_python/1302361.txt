n,k=map(int,raw_input().split())
k-=1
a=map(int,raw_input().split())
a.sort()
b=[]
for i in a:
    b.append(i)
base=0
cnt=0
dic={}
for i in a:
    if not dic.has_key(i):
        dic[i]=1
    else:        
        dic[i]+=1
a=list(set(a))
a.sort()
while True:
    # found
    if base+n*dic[a[cnt]]>k:
        e=(k-base)/dic[a[cnt]]
        print a[cnt],b[e]
        break
    base+=n*dic[a[cnt]]
    cnt+=1
