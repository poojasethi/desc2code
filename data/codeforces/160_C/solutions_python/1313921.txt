import sys

def run(inputStream):
    numPairs, resultPairNumber = map(int, inputStream.readline().split())
    numbers = sorted(map(int, inputStream.readline().split()))
    a = (resultPairNumber - 1) / numPairs
    b = (resultPairNumber - 1) % numPairs
    aValue = numbers[a]
    aCount = numbers.count(aValue)
    if aCount != 1:
        b = (numPairs * (a - numbers.index(aValue)) + b) / aCount

    return "{} {}".format(numbers[a], numbers[b])

print run(sys.stdin)
