n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
d = dict()
k -= 1
for i in a:
    d[i] = d.get(i, 0) + 1
b = d.items()
b.sort()
c = 1
for x in b:
    if 0 <= k < n * x[1]:
        print x[0],
        c = x[1]
        break
    else:
        k -= n * x[1]
for x in b:
    if 0 <= k < c * x[1]:
        print x[0]
        break
    else:
        k -= c * x[1]
