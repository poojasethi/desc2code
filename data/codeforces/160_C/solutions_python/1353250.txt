from __future__ import division
from bisect import bisect_left, bisect_right

n, k = map(int, raw_input().split())
k -= 1
a = map(int, raw_input().split())
a.sort()

i1 = k // n

left = bisect_left(a, a[i1])
right = bisect_right(a, a[i1])

cnt = left
num = right - left

i2 = (k - n * cnt) // num

print "%d %d" % (a[i1], a[i2])
