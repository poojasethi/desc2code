n, k = map(int, raw_input().split())
t = list(map(int, raw_input().split()))
k -= 1

t.sort()
i = j = k // n
x = t[i]

if t[0] == x: i = 0
else:
    i -= 1
    while t[i] == x: i -= 1
    i += 1

if t[-1] == x: j = n
else:
    j += 1
    while t[j] == x: j += 1
    
print x, t[(k - i * n) // (j - i)]
