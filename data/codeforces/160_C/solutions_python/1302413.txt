import sys

n, k = [int(s) for s in sys.stdin.readline().split()]
k -= 1

nums = [int(s) for s in sys.stdin.readline().split()]
nums.sort()

for k in [k]:  # xrange(n * n):
    kn = k / n
    st = min(s for s in xrange(k / n + 1) if nums[s] == nums[kn])
    en = max(s for s in xrange(k / n, n) if nums[s] == nums[st]) + 1
    idx = k - n * st
    print nums[k / n], nums[idx / (en - st)]
