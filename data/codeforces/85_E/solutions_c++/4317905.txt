#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
const int N = 5000 ;
const int mod = 1000000007;
int x[N],y[N];
int l,r,t,b;
int main(){
    int n;
    scanf("%d",&n);
    for (int i=0;i<n;i++){
        int a,b;
        scanf("%d%d",&a,&b);
        x[i]=a+b;y[i]=a-b;
    }
    r=t=-mod;l=b=mod;
    for (int i=0;i<n;i++){
        l = min(x[i],l) ; r = max(x[i],r) ;
        b = min(y[i],b) ; t = max(y[i],t) ;
    }
    int p1=0,p2=0;
    for (int i=0;i<n;i++){
        p1 = max(p1 , min(max(t-y[i],x[i]-l) , max(y[i]-b,r-x[i]))) ;
        p2 = max(p2 , min(max(t-y[i],r-x[i]) , max(y[i]-b,x[i]-l))) ;
    }
    int ans=min(p1,p2),L=ans;
    cout<<ans<<endl ;
    ans=2 ;
    for (int i=0;i<n;i++)
        if (max(max(t-y[i],x[i]-l),max(y[i]-b,r-x[i]))<=L) ans=ans*2%mod;
    if (p1==p2&&t-b>L&&r-l>L)
        ans=ans*2%mod;
    cout<<ans<<endl;
    return 0;
}
