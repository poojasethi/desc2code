#include<cstdio>
#include<algorithm>
#define lim 1000000007
using namespace std;
struct point
{	int x,y;
};
point p[5005];
int n,left,right,top,bot;
void work()
{	int i,d1,d2,ans,type,cnt,s;
	d1=d2=cnt=0;
	type=s=1;
	for (i=1;i<=n;++i)
		d1=max(d1,min(max(p[i].x-left,top-p[i].y),max(right-p[i].x,p[i].y-bot)));
	ans=d1;
	if (top!=bot)
	{	for (i=1;i<=n;++i)
			d2=max(d2,min(max(p[i].x-left,p[i].y-bot),max(right-p[i].x,top-p[i].y)));
		if (ans==d2)
			type=3,++cnt;
		if (ans>d2)
			type=2,ans=d2;
		
	}
	for (i=1;i<=n;++i)
	{	if (type!=2&&max(max(p[i].x-left,top-p[i].y),max(right-p[i].x,p[i].y-bot)) <= ans)
			++cnt;
		if (type!=1&&max(max(p[i].x-left,p[i].y-bot),max(right-p[i].x,top-p[i].y)) <= ans)
			++cnt;
	}
	s=1;
	while (cnt>=0)
	{	s<<=1;
		s%=lim;
		--cnt;
	}
	printf("%d\n%d\n",ans,s);
}
int main()
{	int i,x,y;
	scanf("%d",&n);
	bot=left=0x7ffffff;
	top=right=-0x7ffffff;
	for (i=1;i<=n;++i)
	{	scanf("%d%d",&x,&y);
		p[i].x=x+y,p[i].y=x-y;
		top=max(top,p[i].y);
		bot=min(bot,p[i].y);
		left=min(left,p[i].x);
		right=max(right,p[i].x);
	}
	work();
	return 0;
}
