#include<cstdio>
#include<iostream>
using namespace std;
const int p=1000000007;
struct arr
{
    int x,y;
}a[10002];
int n,l=210000000,r=-210000000,s=-210000000,ans1,ans2,x,y,re1,re2;
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%d %d",&x,&y),a[i].x=x+y,a[i].y=x-y;
    x=210000000;
    for (int i=1;i<=n;i++) l=min(l,a[i].x);
    for (int i=1;i<=n;i++) r=max(r,a[i].x);
    for (int i=1;i<=n;i++) x=min(x,a[i].y);
    for (int i=1;i<=n;i++) s=max(s,a[i].y);
    for (int i=1;i<=n;i++) ans1=max(ans1,min(max(a[i].x-l,a[i].y-x),max(r-a[i].x,s-a[i].y)));
    for (int i=1;i<=n;i++) ans2=max(ans2,min(max(a[i].x-l,s-a[i].y),max(r-a[i].x,a[i].y-x)));
    re1=min(ans1,ans2);re2=2;
    for(int i=1;i<=n;i++)
        if(max(max(a[i].x-l,a[i].y-x),max(r-a[i].x,s-a[i].y))<=re1) re2=re2*2%p;
    if(ans1==ans2&&r-l>re1&&s-x>re1) re2=re2*2%p;
    printf("%d\n%d",re1,re2);
    return 0;
}
