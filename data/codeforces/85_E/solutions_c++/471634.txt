#include<cstdio>
#include<cstring>
#include<algorithm>
#define MAXN 5010
#define MOD 1000000007
using namespace std;
typedef long long ll;
int n;
int x[MAXN],y[MAXN];
int vis[MAXN];
int ans,mid;
bool dfs(int cur,int color)
{
    vis[cur]=color;
    for(int i=0;i<n;i++)
        if(abs(x[cur]-x[i])+abs(y[cur]-y[i])>mid)
        {
            if(!vis[i]) {
                if(!dfs(i,3-color)) return false;
            }
            else if(vis[i]==color) return false;
        }
    return true;
}
bool ok()
{
    memset(vis,0,sizeof(vis));
    ans=1;
    for(int i=0;i<n;i++)
        if(!vis[i]) {
            if(!dfs(i,1)) return false;
            ans=ans*2%MOD;
        }
    return true;
}
int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d%d",x+i,y+i);
    int low=0,high=10000;
    while(low<high)
    {
        mid=(low+high)/2;
        if(ok()) high=mid;
        else low=mid+1;
    }
    mid=low;
    ok();
    printf("%d\n%d\n",low,ans);
    return 0;
}
