#include<cstdio>
#include<cstring>
const int MOD=1e9+7;
const int N=5001;
struct pinfo{
	int x,y;
}p[N];
struct einfo{
	int l,r,nxt;
}e[N*N/2];
int n,es=0,maxdis=0,head[N*2],f[N],d[N];
inline int abs(const int x){
	return x>=0?x:-x;
}
inline int getfather(const int v){
	if (!f[v]) return v;
	int tmp=getfather(f[v]);
	d[v]^=d[f[v]];
	return f[v]=tmp;
}
int calc(){
	int res=1;
	for (int i=0;i<n;i++) res=res*2%MOD;
	return res;
}
int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d%d",&p[i].x,&p[i].y);
	for (int i=1;i<n;i++)
		for (int j=i+1;j<=n;j++){
			int dis=abs(p[i].x-p[j].x)+abs(p[i].y-p[j].y);
			if (dis>maxdis) maxdis=dis;
			es++;
			e[es].l=i;
			e[es].r=j;
			e[es].nxt=head[dis];
			head[dis]=es;
		}
	for (;maxdis;maxdis--){
		int tmp=0;
		for (int i=head[maxdis];i;i=e[i].nxt){
			int fl=getfather(e[i].l),fr=getfather(e[i].r);	
			if (fl==fr){
				if (d[e[i].l]==d[e[i].r]){
					printf("%d\n%d",maxdis,calc());
					return 0;
				}
			}else d[fl]=d[e[i].l]==d[e[i].r],f[fl]=fr,tmp++;
		}
		n-=tmp;
	}
	printf("0\n2");
}
