#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
using namespace std;
int n, l, r, mid, ans, x[5010], y[5010], v[5010];
bool work(int u, int kind)
{
    v[u] = kind;
    for (int i = 1; i <= n; i++)
    {
        if (abs(x[i] - x[u]) + abs(y[i] - y[u]) <= mid) continue;
        if (v[i] == kind) return 0;
        if (v[i] == 0) if (!work(i, 3 - kind)) return 0;
    }
    return 1;
}
int ck()
{
    int ans = 1;
    memset(v, 0, sizeof(v));
    for (int i = 1; i <= n; i++)
    {
        if (v[i] == 0)
        {
            ans <<= 1;
            if (ans >= 1000000007) ans -= 1000000007;
            if (!work(i, 1)) return 0;
        }
    }
    return ans;
}
int main()
{
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) scanf("%d%d", x + i, y + i);
    l = -1, r = 10000;
    while (l + 1 < r)
    {
        mid = l + r >> 1;
        if (ck()) r = mid;
        else l = mid;
    }
    mid = r;
    printf("%d\n%d\n", mid, ck());
    return 0;
}
