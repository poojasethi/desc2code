#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

const int mod = 1000000007;
const int maxn = 5005;
int x[maxn], y[maxn];
int n, ans, sum;

int main() {
    int i, a, b, x1, x2, y1, y2, r1, r2;
    scanf("%d", &n);
    x1 = y1 = mod; x2 = y2 = -mod;
    for (i = 1; i <= n; ++i) {
        scanf("%d%d", &a, &b);
        x[i] = a + b; y[i] = a - b;
        x1 = min(x1, x[i]); x2 = max(x2, x[i]);
        y1 = min(y1, y[i]); y2 = max(y2, y[i]);
    }
    for (i = 1, r1 = r2 = 0; i <= n; ++i) {
        r1 = max(r1, min(max(x[i] - x1, y[i] - y1), max(x2 - x[i], y2 - y[i])));
        r2 = max(r2, min(max(x[i] - x1, y2 - y[i]), max(x2 - x[i], y[i] - y1)));
    }
    ans = min(r1, r2); sum = 2;
    for (i = 1; i <= n; ++i)
        if (max(max(x[i] - x1, x2 - x[i]), max(y[i] - y1, y2 - y[i])) <= ans)
            sum = (sum + sum) % mod;
    if (r1 == r2 && x2 - x1 > ans && y2 - y1 > ans)
        sum = (sum + sum) % mod;
    printf("%d\n%d\n", ans, sum);
    return 0;
}
