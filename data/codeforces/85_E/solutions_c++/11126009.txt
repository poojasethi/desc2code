#include<cstdio>
#include<cstring>
#include<algorithm>
#define MAXN 5010
#define mod 1000000007
using namespace std;
int n,l=0,r=10000,mid,ans=1,cnt;
int x[MAXN],y[MAXN];
bool vis[MAXN],col[MAXN];
inline int dis(int u,int v)
{
	return abs(x[u]-x[v])+abs(y[u]-y[v]);
}
inline bool dfs(int u,int c,int x)
{
	if(vis[u])
		return col[u]==c;
	vis[u]=1;
	col[u]=c;
	for(int i=1;i<=n;i++)
		if(dis(u,i)>x&&!dfs(i,c^1,x))
			return 0;
	return 1;
}
inline bool check(int x)
{
	memset(vis,0,sizeof(vis));
	cnt=0;
	for(int i=1;i<=n;i++)
		if(!vis[i])
		{
			cnt++;
			if(!dfs(i,0,x))
				return 0;
		}
	return 1;
}
int main()
{
	int i,j;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d%d",x+i,y+i);
	while(l<=r)
	{
		mid=l+r>>1;
		if(check(mid))
			r=mid-1;
		else
			l=mid+1;
	}
	check(l);
	while(cnt--)
		ans=(ans<<1)%mod;
	printf("%d\n%d\n",l,ans);
	return 0;
}
