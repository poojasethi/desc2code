#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 5001

int n,s,i,p;
bool f;
int X[lim],Y[lim],C[lim];

int dfs(int i,int k)
{
  int j;
  
  for(j=0;j<n;j++)
    if(abs(X[i]-X[j])+abs(Y[i]-Y[j])>k)
      if(C[j]==-1)
      {
        C[j]=C[i]^1;
        dfs(j,k);
      }
      else if(C[j]==C[i])
        f=0;
}
int bs(int i,int j)
{
  int k=(i+j)/2,p;
  
  memset(C,-1,sizeof C);
  for(p=s=0,f=1;p<n;p++)
    if(C[p]==-1)
    {
      C[p]=0;
      dfs(p,k);
      s++;
    } 
  if(i==j)
    return i;
  if(f)
    return bs(i,k);
  return bs(k+1,j);
}
int main()
{
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d%d",&X[i],&Y[i]);   
  printf("%d\n",bs(0,10000));
  p=1;
  while(s--)
    p=p*2%1000000007;
  printf("%d\n",p);
  return 0;
}
