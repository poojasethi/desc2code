#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
int mxn,n;
bool p[5000];
struct poi{
    int x,y;
} v[5000];
void cmin(int &a,int b){
    if (b<a) a=b;
}
void cmax(int &a,int b){
    if (b>a) a=b;
}
void dfs(int now){
    p[now]=false;
    for (int i=0;i<n;i++)
        if (p[i] && (abs(v[now].x-v[i].x)>mxn || abs(v[now].y-v[i].y)>mxn)) dfs(i);
}
int main(){
    int i,xa,xb,ya,yb,mxn1,mxn2,x,y,ans;
    scanf("%d",&n);
    xa=ya=1<<25,xb=yb=-(1<<25);
    for (i=0;i<n;i++){
        scanf("%d%d",&x,&y);
        v[i].x=x+y,v[i].y=x-y;
        cmin(xa,v[i].x),cmin(ya,v[i].y),cmax(xb,v[i].x),cmax(yb,v[i].y);
    }
    mxn1=mxn2=-(1<<25);
    for (i=0;i<n;i++){
        cmax(mxn1,min(max(v[i].x-xa,v[i].y-ya),max(xb-v[i].x,yb-v[i].y)));
        cmax(mxn2,min(max(v[i].x-xa,yb-v[i].y),max(xb-v[i].x,v[i].y-ya)));
    }
    mxn=min(mxn1,mxn2);
    printf("%d\n",mxn);
    for (i=0;i<n;i++) p[i]=true;
    ans=1;
    for (i=0;i<n;i++) if (p[i]) dfs(i),ans=ans*2%1000000007;
    printf("%d\n",ans);
    return 0;
}
