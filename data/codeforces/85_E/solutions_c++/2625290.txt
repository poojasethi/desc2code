#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
#define dbg(...) fprintf(stderr,__VA_ARGS__)
#define REP(i,n) for(int i=0;i<(n);i++)
#define mp make_pair
#define fs first
#define sc second

typedef pair<int,int> pii;

const int MAXN=5000;
const int MODER=1000000007 ;
int n;
pii p[MAXN+10];

int color[MAXN+10];
int mid;
void fun(int x,int clr){
	color[x]=clr;
	REP(i,n)
		if(abs(p[i].fs-p[x].fs)+abs(p[i].sc-p[x].sc)>mid)
			if(!color[i])
				fun(i,-clr);
			else if(color[i]==clr)
				throw "123";
}
int fang;
bool can(){
	memset(color,0,sizeof(color));
	fang=1;
	try{
		REP(i,n)if(!color[i]){
			fang=fang*2%MODER;
			fun(i,1);
		}
	}catch(...){
		return false;
	}
	return true;
}

int main(){
	scanf("%d",&n);
	REP(i,n)
		scanf("%d%d",&p[i].fs,&p[i].sc);
	
	int l=0,r=5000*2;
	while(l<=r){
        mid=(l+r)/2;
        if(can())
			r=mid-1;
		else
			l=mid+1;
    }
    mid=l;
	can();
	printf("%d\n%d\n",mid,fang);
	return 0;
}
