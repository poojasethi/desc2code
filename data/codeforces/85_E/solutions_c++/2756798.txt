#include<stdio.h>
#include<algorithm>
#include<string.h>
using namespace std;
int getint()
{
	int ret=0;bool ok=false;
	for(;;)
	{
		int c=getchar();
		if(c>='0'&&c<='9')ret=(ret<<3)+ret+ret+c-'0',ok=true;
		else if(ok)return ret;
	}
}
int n;
int x[5010],y[5010];
int d[5010][5010];
int limit;
int vis[5010];
bool dfs(int u)
{
	for(int i=1;i<=n;i++)if(u!=i&&d[u][i]>=limit)
		if(vis[i]==-1)
		{
			vis[i]=1^vis[u];
			if(dfs(i))return 1;
		}
		else if(vis[i]==vis[u])return 1;
	return 0;
}
bool check(int t)
{
	limit=t;
	memset(vis,-1,sizeof(vis));
	for(int i=1;i<=n;i++)if(vis[i]==-1)
	{
		vis[i]=0;
		if(dfs(i))return 1;
	}
	return 0;
}
int main()
{
	n=getint();
	for(int i=1;i<=n;i++)x[i]=getint(),y[i]=getint();
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)d[i][j]=abs(x[i]-x[j])+abs(y[i]-y[j]);
	int l=-1,r=10000;
	while(l+1!=r)
	{
		int mid=(l+r)>>1;
		if(check(mid))l=mid;
		else r=mid;
	}
	int ans0=max(l,0);
	int ans=1;
	limit=r;
	memset(vis,-1,sizeof(vis));
	for(int i=1;i<=n;i++)if(vis[i]==-1)ans=ans*2%1000000007,vis[i]=0,dfs(i);
	printf("%d\n%d\n",ans0,ans);
	return 0;
}
