#include <cstdio>
#include <cstring>

using namespace std;
const int mod=1000000007;
#define dis(x1,y1,x2,y2) (abs( (x1) - (x2) )+abs( (y1) - (y2) ))
int i,l,r,mid,j,n,m,Q[5001],x[5001],y[5001],v[5001],ans;

int abs(int a){
  if (a>=0) return a;
  return -a;
}

bool pd(){
  memset(v,-1,sizeof(v));
  for (i=1,m=0;i<=n;i++)
  if (v[i]<0){
    m++;
    int l,r;
    for (Q[l=r=1]=i,v[i]=0;l<=r;l++)
      for (j=1;j<=n;j++)
        if (dis(x[Q[l]],y[Q[l]],x[j],y[j])>mid){
          if (v[j]<0) v[j]=(v[Q[l]]^1),Q[++r]=j;
            else if (v[Q[l]]==v[j]) return 0;
        }
  }
  return 1;
}

void work(){
  l=1;r=10000;
  mid=(l+r)>>1;
  while (l<=r){
    if (pd()) r=mid-1;
      else l=mid+1;
    mid=(l+r)>>1;
  }
  if (!pd()) mid++;
  pd();
  printf("%d\n",mid);
  ans=1;
  while (m--) ans=ans*2%mod;
  printf("%d\n",ans);
}

int main(){
  scanf("%d",&n);
  for (i=1;i<=n;i++) scanf("%d%d",x+i,y+i);
  
  work();
  return 0;
}
