#include<cstdio>
#include<string>
#include<cstring>
#include<map>
#define rep(i,n) for(int i=0;i<n;i++)
#define FOR(i,a,b) for(int i=a;i<=b;i++)
#define FORD(i,b,a) for(int i=b;i>=a;i--)
using namespace std;
int n,len,m;
char s[110],id[110];
map <string,int> h;

void read(char s[110])
{
	gets(s);
	len=strlen(s);m=0;
	rep(i,len)
	    if(s[i]!=' ') s[m++]=s[i];
}
int Calc(int l,int r)
{
	int p=0;
	FORD(i,r,l)
	{
		if (s[i]=='(') p++;
		if (s[i]==')') p--;
		if (!p && (s[i]=='+'||s[i]=='-'))
		{
			int L=Calc(l,i-1),R=Calc(i+1,r);
			if (L==4||R==4|| s[i]=='-'&&R==1 ) return 4;
			return 1;
		}
	}
	FORD(i,r,l)
	{
		if (s[i]=='(') p++;
		if (s[i]==')') p--;
		if (!p && (s[i]=='*'||s[i]=='/'))
		{
			int L=Calc(l,i-1),R=Calc(i+1,r);
			if (L==4||R==4||L==1||R==1|| s[i]=='/'&&R==2 ) return 4;
			return 2;
		}
	}
	if (s[l]=='(')
	    return Calc(l+1,r-1)<4?3:4;
	string str="";
    FOR(i,l,r) str+=s[i];
	if (h[str]==0)
		return h[str]=3;
	else return h[str];
	
}
int main()
{
	for(scanf("%d",&n);n;n--)
	{
		scanf(" #%s%s",s,id);
		read(s);
		h[id]=Calc(0,m-1);
	}
	read(s);
	if (Calc(0,m-1)<4) printf("OK\n");
	else printf("Suspicious\n");
	return 0;
}
