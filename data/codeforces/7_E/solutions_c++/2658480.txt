#include<iostream>
#include<algorithm>
#include<map>
#include<string>
#include<cstdio>
#include<cstdlib>
#include<cstring>

using namespace std;

const int fin=0,N=103;
int n;
int sus[N],opt[300];
char s[N];
string str;
map<string,int>G;
map<string,int>::iterator it;

int sol(int n,bool w){
  int m,i,t,p=3,dep=0;
  char ch,l=0;
  if(w){
    str="";
    for(i=0;i<8;++i)while(ch=getchar(),ch==' ');
    while(ch!=' ')str+=ch,ch=getchar();
    G[str]=n;
  }
  str="",gets(s),m=strlen(s);
  for(i=0;i<m;++i)if(s[i]!=' ')
    if(opt[s[i]]){
      if(it=G.find(str),it!=G.end())
        if(t=it->second,sus[t]==0||
          l&&(l!='+'&&sus[t]<=1||l=='/'&&sus[t]<=2)||
          opt[s[i]]!=3&&sus[t]<opt[s[i]])
          return sus[n]=0;
      str="",l=0;
      if(s[i]=='(')++dep;
      else if(s[i]==')')--dep;
      else if(l=s[i],!dep)p=min(p,opt[s[i]]);
    }else str+=s[i];
  if(str!="")
    if(it=G.find(str),it!=G.end())
      if(t=it->second,sus[t]==0||
        l&&(l!='+'&&sus[t]<=1||l=='/'&&sus[t]<=2))
        return sus[n]=0;
  return sus[n]=p;
}

int main(){
  if(fin){
    freopen("t.in","r",stdin);
    freopen("t.out","w",stdout);
  }
  opt['+']=opt['-']=1,opt['*']=opt['/']=2,opt['(']=opt[')']=3;
  scanf("%d\n",&n);
  for(int i=1;i<=n;++i)sol(i,1);
  puts(sol(n+1,0)?"OK":"Suspicious");
  return 0;
}
