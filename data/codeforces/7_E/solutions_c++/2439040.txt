int work(int l,int r)
{
	int i,zz=0;
	for(i=r;i>l;i--)
	{
		if (s[i]=='(') zz++; 
		else if (s[i]==')') zz--;
		else if ((s[i]=='+'||s[i]=='-')&&zz==0)
		{
			int ll=work(l,i-1);
			int rr=work(i+1,r);
			if (ll==4||rr==4||(s[i]=='-'&&rr==3))
				return 4; else 
				return 3;
		}
	}
	zz=0;
	for(i=r;i>l;i--)
	{
		if (s[i]=='(') zz++; 
		else if (s[i]==')') zz--;
		else if ((s[i]=='*'||s[i]=='/')&&zz==0)
		{
			int ll=work(l,i-1);
			int rr=work(i+1,r);
			if (ll==4||rr==4||ll==3||rr==3||(s[i]=='/'&&rr==2))
				return 4; else 
				return 2;
		}
	}
	if (s[l]=='(')
		return((work(l+1,r-1)==4)?4:1);
	string sa; 
	sa="";
	for(i=l;i<=r;i++)
	sa=sa+s[i];
	if(ma[sa]==0)
	{
		ma[sa]=1;
	//	cout<<s<<endl;
		return(1);
	}
	else return ma[sa];
}
