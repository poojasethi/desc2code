#include<cstdio>
#include<cstring>
#include<iostream>
#include<map>
using namespace std;
map<string,int> Safe;
string Token,s;
char Expr[105];
int n,m,Test;
int get(int l,int r)
{
	int L,R,b=0;
	for (int i=r-1;i>=l;i--){
		b+=(s[i]=='(')-(s[i]==')');
		if (!b && (s[i]=='-' || s[i]=='+')){
			L=get(l,i),R=get(i+1,r);
			return L && R && (s[i]=='+'||R>1);
			}
		}
	for (int i=r-1;i>=l;i--){
		b+=(s[i]=='(')-(s[i]==')');
		if (!b && (s[i]=='/' || s[i]=='*')){
			L=get(l,i),R=get(i+1,r);
			return L>1 && R>1 && (s[i]=='*'||R>2)?2:0;
			}
		}
	if (s[l]=='(') return get(l+1,r-1)?3:0;
	string t=s.substr(l,r-l); return Safe.count(t)?Safe[t]:3;
}
int Work()
{
	gets(Expr),m=strlen(Expr),s="";
	for (int i=0;i<m;i++) if (Expr[i]!=' ') s+=Expr[i];
	return get(0,s.size());
}
int main()
{
	scanf("%d",&n); Safe.clear();
	for (int i=1;i<=n;i++) scanf(" #%*s"),cin>>Token,Safe[Token]=Work();
	puts(Work()?"OK":"Suspicious"); return 0;
}
