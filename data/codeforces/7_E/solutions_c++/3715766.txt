#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<string>
#include<map>
#define maxn 1000
using namespace std;
string S,T;
map<string,int> dp;
map<char,int> hash;
char s[maxn];
int Dp(int l,int r)
{
	for(int i=r,x=0;i>=l;i--)
	{
		x+=hash[s[i]];
		if(x==0&&(s[i]=='+'||s[i]=='-'))
		{
			int v1=Dp(l,i-1),v2=Dp(i+1,r);
			return (v1==-2||v2==-2||(s[i]=='-'&&v2==2))?-2:2;
		}
	}
	for(int i=r,x=0;i>=l;i--)
	{
		x+=hash[s[i]];
		if(x==0&&(s[i]=='*'||s[i]=='/'))
		{
			int v1=Dp(l,i-1),v2=Dp(i+1,r);
			return (v1==-2||v2==-2||v1==2||v2==2||(s[i]=='/'&&v2==1))?-2:1;
		}
	}
	if(s[l]=='(')
		return Dp(l+1,r-1)==-2?-2:-1;
	T="";
	for(int i=l;i<=r;i++)
		T+=s[i];
	return dp.count(T)?dp[T]:-1;
}
/*
	if Dp(i,j)=-2,it is "Suspicious"
	if Dp(i,j)=2,it's flag is '+' or '-'
	if Dp(i,j)=1,it's flag is '*' or '-'
	if Dp(i,j)=-1,it's flag is NULL
*/
void read()
{
	int n;
	scanf("%d",&n);
	gets(s);
	n++;
	while(n--)
	{
		if(n!=0)
		{
			scanf("%s",s);
			if(s[1]!='d')
				scanf("%s",s);
			scanf("%s",s);
			S=s;
		}
		gets(s);
		int tot=0;
		for(int i=0;s[i];i++)
			if(s[i]!=' ')
				s[tot++]=s[i];
		s[tot]=0;
		int temp=Dp(0,tot-1);
		if(n!=0)
			dp[S]=temp;
		else
			printf("%s\n",temp==-2?"Suspicious":"OK");
	}
}
int main()
{
	hash['(']=1,hash[')']=-1;
	read();
	return 0;
}
