int get(int l, int r)
{
    int L, R, b = 0;
    for (int i = r - 1; i >= l; --i)
    {
        if (t[i] == '(') b++;
        if (t[i] == ')') b--;
        if (!b && (t[i] == '+' || t[i] == '-'))
        {
            L = get(l, i), R = get(i + 1, r);
            return L && R && (t[i] == '+' || R > 1);
        }
    }
    for (int i = r - 1; i >= l; --i)
    {
        if (t[i] == '(') b++;
        if (t[i] == ')') b--;
        if (!b &&(t[i] == '*' || t[i] == '/'))
        {
            L = get(l, i), R = get(i + 1, r);
            return (L > 1 && R > 1 && (t[i] == '*' || R == 3))? 2 : 0;
        }
    }
    if (t[l] == '(') return get(l + 1, r - 1)? 3 : 0;
    string x = t.substr(l, r - l);
    return mp.count(x)? mp[x]: 3;
}
