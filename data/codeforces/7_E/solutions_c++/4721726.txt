#include <cstdio>
#include <cstring>
struct info{
    char s[201];
    int l,t;
}im[101];
int n,lon;
char s[201];
void getins(){
    gets(s);
    int x=strlen(s);
    lon=0;
    for (int i=0;i<x;i++)
    if (s[i]!=' ') s[lon++]=s[i];
}
int gett(int l,int r){
    for (int i=1;i<=n;i++){
        bool bo=(im[i].l==r-l+1);
        for (int j=0;j<im[i].l && bo;j++)
        if (im[i].s[j]!=s[l+j]) bo=0;
        if (bo) return im[i].t;
    }
    return 1;
}
int getans(int l,int r){
    int k=0,min=10000,j,bo=0;
    for (int i=l;i<=r;i++){
        if (s[i]=='(') ++k,bo=1;
        if (s[i]==')') --k,bo=1;
        if (s[i]=='+' || s[i]=='-'){
            int x=k*10+1;
            if (min>=x) min=x,j=i;
        }
        if (s[i]=='*' || s[i]=='/'){
            int x=k*10+2;
            if (min>=x) min=x,j=i;
        }
    }
    if (min==10000) return bo ? (getans(l+1,r-1) ? 1 : 0) : gett(l,r);
    if (min>10) return getans(l+1,r-1) ? 1 : 0;
    int x=getans(l,j-1),y=getans(j+1,r);
    if (!x || !y) return 0;
    if (s[j]=='+') return 2;
    if (s[j]=='-') return y==2 ? 0 : 2;
    if (x==2 || y==2) return 0;
    if (s[j]=='*') return 3;
    if (s[j]=='/') return y==3 ? 0 : 3;
    return 0;
}
int main(){
    scanf("%d",&n);
    for (int i=1;i<=n;i++){
        scanf(" #%s%s",im[i].s,im[i].s);
        im[i].l=strlen(im[i].s);
        getins();
        if (i==4)
        i=4;
        im[i].t=getans(0,lon-1);
    }
    getins();
    int x=getans(0,lon-1);
    if (x) printf("OK");
    else printf("Suspicious");
}
