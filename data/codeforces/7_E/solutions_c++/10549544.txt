int n;
map<string,int> mp;
int get(int l,int r,string s)
{
    for(int i=r-1,w=0;i>=l;i--)
    {
        if(s[i]==')') w++;
        if(s[i]=='(') w--;
        if(w==0&&(s[i]=='+'||s[i]=='-'))
        {
            int L=get(l,i,s),R=get(i+1,r,s);
            return (L&&R&&(s[i]=='+'||R>1));
        }
    }
    for(int i=r-1,w=0;i>=l;i--)
    {
        if(s[i]==')') w++;
        if(s[i]=='(') w--;
        if(w==0&&(s[i]=='*'||s[i]=='/'))
        {
            int L=get(l,i,s),R=get(i+1,r,s);
            return ((L>=2&&R>=2&&(s[i]=='*'||R>2))?2:0);
        }
    }
    if(s[l]=='(') return get(l+1,r-1,s)?3:0;
    string  u=s.substr(l,r-l);
	return (mp.count(u)?mp[u]:3);
}
int gao(string s)
{
    int len=s.size(),cur=0;
    char temp[1000000];
    for(int i=0;i<len;i++)
        if(s[i]!=' ')
            temp[cur++]=s[i];
    s=temp;
    return get(0,cur,s);
}
