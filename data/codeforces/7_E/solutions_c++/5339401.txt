#include <map>
#include <string>
#include <cstdio>
using namespace std;

int N,a[105];
char s[1005];
map<string,int> F;

void work(int t)
{
	char c,p=0; string x=""; int y=0,d=0,m=3;
	if (t) scanf("%s",s),F[string(s)]=t;
	for (;c=getchar(),1;) if (m==-1&&c!='\n') continue;
	else if (isspace(c))
	{
		if ((y=F[x])&&(a[y]==-1||(a[y]==1&&(p=='*'||p=='/'))||(p=='-'&&a[y]==1)||(p=='/'&&a[y]==2))) m=-1;
		x=""; if (c=='\n'||c==EOF) break;
	}
	else if (c=='('||c==')'||c=='+'||c=='-'||c=='*'||c=='/')
	{
		if (c=='(') d++; else if (c==')') d--; else if (!d) m=min(m,c=='+'||c=='-'?1:2);
		y=F[x],x="";
		if (y)
		{
			if (a[y]==-1) m=-1;
			else if (((c=='*'||c=='/')&&a[y]==1)||(a[y]==1&&(p=='*'||p=='/'))||(p=='-'&&a[y]==1)||(p=='/'&&a[y]==2)) m=-1;
		}
		p=c;
	}
	else x+=c;
	a[t]=m;
}

int main()
{
	scanf("%d",&N);
	for (int i=1; i<=N; i++)
	{
		scanf("%s",s);
		if (s[1]!='d') scanf("%s",s);
		work(i);
	}
	work(0),puts(a[0]>0?"OK":"Suspicious");
}
