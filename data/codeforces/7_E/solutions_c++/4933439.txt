int calc(int l,int r)
{
    for(int i=r-1,w=0;i>=l;i--)
    {
            w+=(S[i]=='(');
            w-=(S[i]==')');
            if(w==0&&(S[i]=='+'||S[i]=='-'))
            {
               int L=calc(l,i),R=calc(i+1,r);
               return L&&R&&(S[i]!='-'||R>1);
            }
    }for(int i=r-1,w=0;i>=l;i--)
    {
            w+=(S[i]=='(');
            w-=(S[i]==')');
            if(w==0&&(S[i]=='*'||S[i]=='/'))
            {
               int L=calc(l,i),R=calc(i+1,r);
               return (L>1)&&(R>1)&&(S[i]!='/'||R>2)?2:0;
            }
    }if(S[l]=='('&&S[r-1]==')')
                             return calc(l+1,r-1)?3:0;       
    string u(S+l,S+r);
    if(M.count(u))return M[u];
    else return 3;
}

    if(S[i]!=' ')S[m++]=S[i];
    S[m]=0;
    return calc(0,m);
}
