using namespace std;
int n;
int len;
string f;
map <string ,int> g;
char ch[10010];
int work(int l,int r)
{
    int m;
    for (int i = r,m = 0;i >= l;i--)
    {
        if (ch[i] == '(') m++;
        if (ch[i] == ')') m--;
        if (m == 0 && (ch[i] == '+' || ch[i] == '-'))
        {
            int L = work(l,i-1),R = work(i+1,r);
            return (L && R && (ch[i] == '+' || R > 1));
        }
    }
    for (int i = r,m = 0;i >= l;i--)
    {
        if (ch[i] == '(') m++;
        if (ch[i] == ')') m--;
        if (m == 0 && (ch[i] == '*' || ch[i] == '/'))
        {
            int L = work(l,i-1),R = work(i+1,r);
            return ((L > 1 && R > 1&& (ch[i] == '*' || R > 2))?2:0);
        }
    }
    if (ch[l] == '(')
        return (work(l+1,r-1)?3:0);
    string u(ch+l,ch+r+1);
    //cout<<l<<' '<<r<<' '<<u<<endl;
    return (g.count(u)?g[u]:3);
}
int solve()
{
    gets(ch);
    len = 0;
    for (int i = 0;i < strlen(ch);i++)
    if (ch[i] != ' ')
        ch[len++] = ch[i];
    //for (int i = 0;i < len;i++)
    //printf("%c",ch[i]);printf("\n");
    return work(0,len-1);
}
int main()
{
  //  freopen("in.txt","r",stdin);
    scanf("%d",&n);
    for (int i = 1;i <= n;i++)
    {
        scanf(" #%*s");

        g[f] = solve();
       // printf("%d\n",g[f]);
