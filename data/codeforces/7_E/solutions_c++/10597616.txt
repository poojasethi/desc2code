int cal(int l, int r)
{
	int cnt = 0;
	for(int i = r; i >= l; i --)
	{
		if(s[i] == '(')
			cnt ++;
		if(s[i] == ')')
			cnt --;
		if(!cnt && (s[i] == '+' || s[i] == '-'))
		{
			int fro = cal(l, i - 1), bes = cal(i + 1, r);
			if(fro == 1 || bes == 1 || s[i] == '-' && bes == 2)
				return 1;
			return 2;
		}
	}
	for(int i = r; i >= l; i --)
	{
		if(s[i] == '(')
			cnt ++;
		if(s[i] == ')')
			cnt --;
		if(!cnt && (s[i] == '*' || s[i] == '/'))
		{
			int fro = cal(l, i - 1), bes = cal(i + 1, r);
			if(fro == 1 || bes == 1 || fro == 2 || bes == 2 || s[i] == '/' && bes == 3)
				return 1;
			return 3;
		}
	}
	if(s[l] == '(')
		return cal(l + 1, r - 1) == 1 ? 1 : 0;
	string str = "";
	for(int i = l; i <= r; i ++)
		str += s[i];
	return v[str];
}
