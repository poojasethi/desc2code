#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<map>
using namespace std;
map<string,int> h;
char s[1010],str[1010],t[1010];
int n,m,i;

int solve(int l,int r)
{
	int i,cnt=0,L,R;
	for(i=r;i>=l;i--)
	{
		if(s[i]=='(') cnt++; else if(s[i]==')') cnt--;
		if(!cnt&&(s[i]=='+'||s[i]=='-'))
		{
			L=solve(l,i-1),R=solve(i+1,r);
			return L==-1||R==-1||s[i]=='-'&&R==1?-1:1;
		}
	}
	for(i=r;i>=l;i--)
	{
		if(s[i]=='(') cnt++; else if(s[i]==')') cnt--;
		if(!cnt&&(s[i]=='*'||s[i]=='/'))
		{
			L=solve(l,i-1),R=solve(i+1,r);
			return abs(L)==1||abs(R)==1||s[i]=='/'&&!R?-1:0;
		}
	}
	if(s[l]=='(') return solve(l+1,r-1)==-1?-1:2;
	for(i=l;i<=r;i++) t[i-l]=s[i];
	t[r-l+1]=0;
	return h.count(t)?h[t]:2;
}

int main()
{
	for(cin>>n;n>=0;n--)
	{
		if(n) scanf("\n#%*s%s",str);
		gets(s);
		for(m=i=0;s[i];i++)
			if(s[i]!=' '&&s[i]!='\n') s[m++]=s[i];
		s[m]=0;
		m=solve(0,m-1);
		if(n) h[str]=m; else puts(m==-1?"Suspicious":"OK");
	}
	return 0;
}
