#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;
const int MAXN = 105,MAXL = 105;
char s[MAXL],name[MAXN][MAXL],c;
int len[MAXN],pri[MAXN],n,tot = 0;

bool cmp(int i,int id,int l){
	for(int j = 0;j < l;++ j)
	if(s[i + j] != name[id][j])
		return false;
	return true;
}

int Safe(int l,int r){
	int bucket = 0,pre,suf;
	for(int i = r;i >= l;-- i){
		bucket += s[i] == ')'; bucket -= s[i] == '(';
		if(!bucket && (s[i] == '+' || s[i] == '-')){
			pre = Safe(l,i - 1); suf = Safe(i + 1,r);
			if((s[i] == '+' && (pre < 0 || suf < 0)) || (s[i] == '-' && (pre < 1 || suf < 2))) return -1;//Debug
			return 1;
		}
	}
	for(int i = r;i >= l;-- i){
		bucket += s[i] == ')'; bucket -= s[i] == '(';
		if(!bucket && (s[i] == '*' || s[i] == '/')){
			pre = Safe(l,i - 1); suf = Safe(i + 1,r);
			if((s[i] == '*' && (pre < 2 || suf < 2)) || (s[i] == '/' && (pre < 2 || suf < 3))) return -1;
			return 2;
		}
	}
	if(s[l] == '(' && s[r] == ')') return Safe(l + 1,r - 1) > 0 ? 3 : -1;
	for(int i = 0;i < tot;++ i)
	if(r - l + 1 == len[i] && cmp(l,i,len[i]))
		return pri[i];
	return 3;
}

int Judge(){
	gets(s);
	int tmp = strlen(s),r = 0;
	for(int i = 0;i < tmp;++ i)
	if(s[i] != ' ')
		s[r ++] = s[i];
	return Safe(0,r - 1);
}

int main(){
	scanf("%d",&n);
	for(int i = 0;i < n;++ i){
		while((c = getchar()) && c != '#');
		scanf("%s",s); scanf("%s",name[tot]);
		len[tot] = strlen(name[tot]);
		pri[tot ++] = Judge();
	}
	printf("%s\n",Judge() > 0 ? "OK" : "Suspicious");
	return 0;
}
