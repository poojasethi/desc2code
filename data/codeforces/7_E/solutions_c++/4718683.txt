#include <cstdio>
#include <cstring>
#include <string>
#include <map>
using std::string;
using std::map;
int n;
char s[200],name[200];
map<string,int>f;
//0: YES
//1: NO
//2: +
//3: *
int dp(const int l,const int r)
{
	int k = 0;
	for (int i = r - 1; i >= l; --i) {
		k += s[i] == '(';
		k -= s[i] == ')';
		if (!k && (s[i] == '+' || s[i] == '-')) {
			int L = dp(l,i),R = dp(i + 1,r);
			if (L == 1 || R == 1) return 1;
			//if (R == 2) return 1;
			if (s[i] == '-' && R == 2) return 1;
			return 2;
		}
	}
	for (int i = r - 1; i >= l; --i) {
		k += s[i] == '(';
		k -= s[i] == ')';
		if (!k && (s[i] == '*' || s[i] == '/')) {
			int L = dp(l,i),R = dp(i + 1,r);
			if (L == 1 || R == 1) return 1;
			if (R == 2 || L == 2) return 1;
			if (s[i] == '/' && R == 3) return 1;
			return 3;
		}
	}
	if (s[l] == '(') return dp(l + 1,r - 1)==1;
	return f[string(s + l,s + r)];
}
int work()
{
	gets(s);
	int len = strlen(s);
	int m = 0;
	for (int i = 0; i < len; ++i)
		if (s[i] != ' ') s[m++] = s[i];
	return dp(0,m);
}
int main()
{
	scanf("%d\n",&n);
	for (int i = 0; i < n; ++i) {
		scanf(" #%*s");
		scanf("%s",name);
		f[string(name)] = work();
	}
	printf((work() != 1)?"OK":"Suspicious");
}
