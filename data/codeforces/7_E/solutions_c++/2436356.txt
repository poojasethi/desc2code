#include <cstdio>
#include <map>
#include <string>
using namespace std; 
#define rep(i,n) for (int i=0;i<n;i++)
int n;char nam[110],S[110];string s;
map<string,int> dp;
void del(char *S) {char *s,*ss;for (s=S,ss=S;*ss;ss++) if (*ss!=' ') *(s++)=*ss;*s='\0';}
int solve(int l,int r) {
	int p=0;
	for (int i=r;i>=l;i--) if (s[i]=='(') p++; else if (s[i]==')') p--;
	else if (!p&&(s[i]=='+'||s[i]=='-')) {
		int L=solve(l,i-1),R=solve(i+1,r);
		return (L==-1 || R==-1 || (s[i]=='-' && R<1))?-1:0;
	}
	for (int i=r;i>=l;i--) if (s[i]=='(') p++; else if (s[i]==')') p--;
	else if (!p&&(s[i]=='*'||s[i]=='/')) {
		int L=solve(l,i-1),R=solve(i+1,r);
		return (L<1 || R<1 || (s[i]=='/' && R<2))?-1:1;
	}
	if (s[l]=='(') return (solve(l+1,r-1)!=-1)?2:-1;
	if (dp.count(s.substr(l,r-l+1))) return dp[s.substr(l,r-l+1)]; else return 2;
}
int main() {
	scanf("%d ",&n);
	rep(i,n) scanf("#%*s%s ",&nam),gets(S),scanf(" "),del(S),s=S,dp[nam]=solve(0,s.length()-1);
	gets(S);del(S);s=S;
	puts(solve(0,s.length()-1)==-1?"Suspicious":"OK");
}
