#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cstdlib>
using namespace std;
string s,name[120];
int f[120][3],i,j,k,l,n,m;
string Get(int &l){
	string res="";
	while((s[l]>='a' && s[l]<='z') || (s[l]>='A' && s[l]<='Z')){
		res+=s[l];
		l++;
	}
	return res;
}
void work(int o){
	int len=s.length(),l=s.find("define")+6,i,tot=0;
	char last;
	while(s[l]==' ')l++;
	name[o]=Get(l);
	for(;l<=len;++l){
		if(s[l]==' ')continue;
		if(s[l]=='(')tot++;
		if(s[l]==')')tot--;
		if((s[l]>='a' && s[l]<='z') || (s[l]>='A' && s[l]<='Z')){
			string res=Get(l);
			for(i=1;i<o;++i)if(name[i]==res)break;
			if(i<o){
				f[o][2]|=f[i][2];
				if(last=='/')f[o][2]|=f[i][0]|f[i][1];
				if(last=='*' || last=='-')f[o][2]|=f[i][0];
				int p=l;
				while(s[p]==' ')p++;
				if(s[p]=='*' || s[p]=='/')f[o][2]|=f[i][0];
			}
			l--;
		}
		if(!tot){
			if(s[l]=='+' || s[l]=='-') f[o][0]=1;
			if(s[l]=='*' || s[l]=='/') f[o][1]=1;
		}
		last=s[l];
	}
}
int main(){
	scanf("%d ",&n);
	for(i=1;i<=n+1;++i){
		getline(cin,s);
		if(i==n+1)s="#define wjhsb "+s;
		work(i);
	}
	puts(f[n+1][2]?"Suspicious":"OK");
}
