#include<cstdio>
#include<map>
#include<string>
#include<algorithm>
using namespace std;
#define N 110
map<string,int> h;
string s;
char t[N], r[N];
int n;

int calc(int l,int r)
{
    int cnt=0;
    for (int i=r; i>=l; i--)
    {
        if (s[i]=='(') cnt++; else if (s[i]==')') cnt--;
        if (!cnt && (s[i]=='+'||s[i]=='-'))
        {
            int L=calc(l,i-1), R=calc(i+1,r);
            return L==-1||R==-1||(s[i]=='-'&&R==1)?-1:1;
        }
    }
    for (int i=r; i>=l; i--)
    {
        if (s[i]=='(') cnt++; else if (s[i]==')') cnt--;
        if (!cnt && (s[i]=='*'||s[i]=='/'))
        {
            int L=calc(l,i-1), R=calc(i+1,r);
            return abs(L)==1||abs(R)==1||(s[i]=='/'&&!R)?-1:0;
        }
    }
    if (s[l]=='(') return calc(l+1,r-1)==-1?-1:2;
    string t=s.substr(l,r-l+1);
    return h.count(t)?h[t]:2;
}

string deal(string ch)
{
    string s=""; int len=ch.length();
    for (int i=0; i<len; i++) if (ch[i]!=' ') s=s+ch[i];
    return s;
}

int main()
{
    for (scanf("%d",&n); n; n--)
    {
        scanf("\n#%s%s",t,t), gets(r), s=deal(r);
        h[t]=calc(0,s.length()-1);
    }
    gets(r), s=deal(r);
    puts(calc(0,s.length()-1)==-1?"Suspicious":"OK");
    return 0;
}
