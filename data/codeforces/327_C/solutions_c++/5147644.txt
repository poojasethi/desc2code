#include <cstdio>
#include <string>
#include <iostream>
using namespace std;
const long long M=(long long)(1e9)+7;
long long bigmod(long long a,long long b,long long m){
	a%=m;
	if (b==0) return 1;
	if (b==1) return a;
	long long tmp=bigmod(a,b/2,m);
	if (b%2) return (tmp*a%m)*tmp%m;
	return tmp*tmp%m;
}
string s;
long long k;
int main(){
	cin>>s>>k;
	int sl=s.length();
	long long ans=0;
	for(int i=0; i<sl; ++i)
		if (s[i]=='5'||s[i]=='0') ans=(ans+bigmod(2,i,M))%M;
	ans=((ans*(bigmod(2,k*sl,M)+M-1))%M)*bigmod((bigmod(2,sl,M)+M-1),M-2,M)%M;
	cout<<ans<<endl;
	return 0;
}
