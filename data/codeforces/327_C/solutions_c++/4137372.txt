#include<iostream>
#include<string>
#define ll long long
using namespace std;
const int MOD=1000000007;
int m[100001];
ll pow(ll b,ll k){
    if(!k)
        return 1;
    if(k%2)
        return pow(b,k-1)*b%MOD;
    else{
        ll t=pow(b,k/2);
        return t*t%MOD;
    }
}
int main(){
    ll k,ans=0;
    string s;
    cin>>s>>k;
    int t=s.length();
    m[0]=1;
    for(int i=1;i<=100000;i++)
        m[i]=(m[i-1]*2)%MOD;
    for(int i=t-1;i>=0;i--)
        if(s[i]=='0'||s[i]=='5')
            ans=(ans+m[i])%MOD;
    cout<<ans*(pow(m[t],k)-1)%MOD*pow((m[t]-1),MOD-2)%MOD;
}
