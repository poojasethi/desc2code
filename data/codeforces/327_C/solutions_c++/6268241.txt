#include <stdio.h>
#include <iostream>
#include <cstring>
using namespace std;
#define ll long long
const ll MOD=1000000007;
ll k,l,v=0,x,y;
char str[100001];
ll pow(ll x,ll y){
	ll s=1,t=y,w=x;
	while (t>0){
		if (t&1) s=s*w%MOD;
		t>>=1;
		w=w*w%MOD;
	}
	return s;
}
int main( ){
	cin>>str>>k;
	l=strlen(str);
	for (int i=l-1;i>=0;i--)
		if (str[i]=='0' || str[i]=='5')
			v=(v+pow(2,i))%MOD;
	x=pow((pow(2,l)-1),MOD-2);
	y=(x*(pow(2,k*l)-1))%MOD;
	v=(v*y)%MOD;
	cout<<v<<endl;
	return 0;
}
