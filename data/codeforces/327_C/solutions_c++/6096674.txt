#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long LL;
const LL MOD=1e9+7;
LL i,k,x,n,ans;
char str[100010];
LL power(long long a,long long x)
{
	LL ans=1;
	while(x>0)
	{
		if(x&1) ans=(ans*a)%MOD;
		a=(a*a)%MOD;
		x>>=1;
	}
	return ans;
}
int main()
{
	scanf("%s %d",str,&k);
	int n=strlen(str);
	for (i=0;i<n;i++)
	if ((str[i]=='0')||(str[i]=='5'))
	ans=(ans+power(2,i))%MOD;
	x=power(2,n*k)%MOD-1;
	x=x*power(power(2,n)-1,MOD-2)%MOD;
	ans=ans*x%MOD;
	printf("%I64d\n",ans);
	return 0;
}
