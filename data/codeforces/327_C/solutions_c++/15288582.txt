#include<iostream>
#include<stdio.h>
#include<string.h>
#define MOD 1000000007
#define LL long long
using namespace std;
int k,l,x,y;
LL cnt;
string a;
LL pow(LL a,LL b)
{
    LL r=1;
    while(b)
    {
        if(b&1)r*=a,r%=MOD;
        a*=a,a%=MOD;
        b>>=1;
    }
    return r;
}
int main(void)
{
    cin>>a>>k;
    l=a.size();
    for(int i=0;i<l;i++)
        if(a[i]=='0' || a[i]=='5')cnt+=pow(2,i),cnt%=MOD;
    x=pow(2,l);
    y=pow(x,k);
    x=((1-x)%MOD+MOD)%MOD;
    y=((1-y)%MOD+MOD)%MOD;
    cout<<(cnt*((y*pow(x,MOD-2))%MOD))%MOD<<endl;
    return 0;
}
