#include<iostream>
#define M 1000000007
#define ll long long
#define R(a) ((a)%M)
using namespace std;
ll powm(ll b,ll p)
{
    if(p==0) return 1L;
    ll ret=powm(b,p/2L);
    ret=R(ret*ret);
    if(p&1) ret=R(b*ret);
    return ret;
}
int main ()
{
    int k;
    ll n,cnt=0L;
    string s;
    cin>> s>>k;
    n=(ll)s.size();
    for(int i=0; i<n; i++)
    {
        if(s[i]=='0' || s[i]=='5') cnt=R(cnt+powm(2,i));
    }
    cnt=R(cnt*R((powm(2,k*n)-1L) * powm(powm(2,n)-1L,M-2)));
    cout<<cnt<<endl;
}
