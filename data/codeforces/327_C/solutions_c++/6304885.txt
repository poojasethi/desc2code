#include<iostream>
#include<algorithm>
using namespace std;
typedef unsigned long long ll;
string a;
ll k,res=0,mod=1000000007;
ll pow(ll a,ll p){
	ll res=1,t=a;
	for(ll i=0;i<64;i++)
	{
		if(p&(1LL<<i)) res=res*t%mod;
		t=t*t%mod;
	}
	return res;
}
ll inv(ll x){
	return pow(x,mod-2);
}
int main()
{
	cin>>a>>k;
	for(ll i=0;i<a.size();i++)
	{
		if(a[i]=='0'||a[i]=='5'){
			res=(res+pow(2,i))%mod;
		}
	}
	cout<<res*(pow(pow(2,a.size()),k)-1)%mod*(inv(pow(2,a.size())-1))%mod<<endl;
	//system("pause");
	return 0;
}
