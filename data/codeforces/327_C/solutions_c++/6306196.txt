#include<iostream>
using namespace std;

#define mod 1000000007


long long pow(long long x,long long n)
{
	long long res=1,t=x;
	for(int i=0;i<64;i++)
	{
		if (n&(1LL<<i)) res=res*t%mod;
		t=t*t%mod;
	}
	return res;
}

long long inverse(long long x)
{
	return pow(x,mod-2);
}


string a;
long long K;

int main()
{
	cin>>a>>K;
	long long ans=0;
	for(int i=0;i<a.length();i++)
	{
		if (a[i]=='0' || a[i]=='5')
		{
			ans=(ans+pow(2,i))%mod;
		}
	}
	cout<<ans*(pow(pow(2,a.length()),K)-1)%mod*(inverse(pow(2,a.size())-1))%mod;

	return 0;
}
