#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long unsigned us(int a,int b)
{
	if(b==1)
		return a;
	if(!b)
		return 1;
	long long unsigned res=us(a,b/2);
	res=(res%mod*res%mod)%mod;
	if(b%2)
		res*=a;
	return res%mod;
}
int s[100005],n,k;
string a;
long long unsigned ans;
int main ()
{
	int i;
	cin>>a>>k;
	n=a.length();
	for(i=0; i<n; i++)
	{
		if(a[i]=='0' || a[i]=='5')
			ans+=(us(2,i)*(us(us(2,k),n)-1)%mod)*us((us(2,n)-1),mod-2);
		ans%=mod;
	}
	cout<<ans;
}
