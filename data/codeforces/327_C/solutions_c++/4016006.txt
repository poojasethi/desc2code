#include<iostream>
#define M 1000000007
#define llu long long int
using namespace std;
llu pow(llu a,llu b)
{
	llu y=a,ans=1;
	while(b!=0)
	{
		llu d=b%2;
		b=b/2;
		if(d) ans=(ans*y)%M;
		y=(y*y)%M;
	}
	return ans;


}
int main()
{
	llu k,k1,k2,i,n,ans=0;
	string a;
	cin>>a;
	cin>>k;
	for(i=a.size()-1;i>=0;i--)
	{
		if(a[i]=='5' || a[i]=='0')
		{
			ans=(ans+pow(2,i))%M;
		}
	}
	n=a.size()*k;
	k1=(pow(2,n)-1+M)%M;
	k2=(pow(2,a.size())-1+M)%M;
	k1=(k1*(pow(k2,M-2)))%M;
	//cout<<k1<<ans<<endl;
	ans=(ans*k1)%M;
	cout<<ans<<endl;
}
