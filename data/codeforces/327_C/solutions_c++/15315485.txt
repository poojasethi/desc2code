#include<iostream>
using namespace std;

typedef long long ll;




string a;
ll k,mo=1e9+7;

ll sum;

ll pow(ll a,ll n)
{
	ll ans=1;
	while(n)
	{
		if (n&1)
		ans=ans*a%mo;
		a=a*a%mo;
		n=n>>1;
	}
	return ans;
}

ll reimu(ll n,ll k)
{
	ll b=pow(2,n);
    ll a=pow(b,k);
    a=(1-a+mo)%mo;
    b=(1-b+mo)%mo;
    return a*pow(b,mo-2)%mo;
}

ll szc;
int main()
{
	
	cin>>a>>k;
	szc=a.size();
	for (int i=0;i<szc;i++)
	{
		if ((a[i]-48)%5==0)
		{
			sum=(sum+pow(2,i))%mo;
		}
	}
	cout<<(sum*reimu(szc,k))%mo;
}
