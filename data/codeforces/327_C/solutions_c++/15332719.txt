#include <iostream>
#include <cstdio>
#include <cstdlib>
#define LL long long
#define N 1000000007
using namespace std;

string s;
int k,len;

LL qpow(LL a,LL b)
{
	LL res=1;
	while(b!=0)
	{
		if(b&1)
		  res=(res*a)%N;
		b=b>>1;
		a=(a*a)%N;
	}
	return res;
}

LL run()
{
	LL a,b;
	a=qpow(2,len);
	b=((qpow(a,k)-1)%N+N)%N;
	a=qpow(a-1,N-2);
	return (a*b)%N;
}

int main()
{
	int i;
	LL t=0;
	cin>>s>>k;
	len=s.size();
	for(i=0;i<len;i++)
	{
		if(s[i]=='5'||s[i]=='0')
		  t=(t+qpow(2,i))%N;
	}
	cout<<(t*run())%N;
}
