#include <stdio.h>
#include <iostream>
#include <string>
#include <cstring>
#define clr(a,b) memset(a,b,sizeof(a))
#define ll long long
using namespace std;

const int mod=1e9+7;
int k;
string str;

ll pow(ll a,ll b){
	ll ret=1;
	while(b){
		if(b&1)ret=ret*a%mod;
		a=a*a%mod;b>>=1;
	}return ret;
}

int main(){
	cin>>str>>k;
	ll ans=0;
	ll u=(pow(2,(ll)k*str.size())-1+mod)%mod;
	ll g=(pow(2,(ll)str.size())-1+mod)%mod;
	ll d=pow(g,mod-2);
	for(int i=0;i<str.size();i++){
		if(str[i]=='0'||str[i]=='5'){
			ans=(ans+pow(2,i)*u%mod*d%mod)%mod;
		}
	}
	cout<<ans<<endl;
}
