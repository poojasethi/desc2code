#include<iostream>
#include<cstdio>
#include<string>
#define ll long long
using namespace std;

const ll mod=1e9+7;
string a;
ll n,m,res=0;
ll exp(ll x,ll y){
	if(!y) return 1;
	ll ans=exp(x,y/2);
	ans=(ans*ans)%mod;
	if(y&1) ans=(ans*x)%mod;
	return ans;
}

int main()
{
	cin>>a;
	cin>>n;
	for(int i=0;i<a.length();i++)
		if(a[i]=='0'||a[i]=='5') {res=(res+exp(2,i))%mod;
//		cout<<res<<endl;
		}
//	cout<<res<<endl;
	res=((res*((exp(2,a.length()*n)-1)%mod+mod))%mod)*exp((exp(2,a.length())-1+mod)%mod,mod-2)%mod;
	cout<<res<<endl;
	return 0;
}
