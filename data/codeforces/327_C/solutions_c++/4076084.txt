#include <iostream>
#include <algorithm>
#include <cmath>
#include <string>

#define rep(i,n) for(int i=0;i<n;i++)
#define ll long long
#define MOD 1000000007LL

using namespace std;

ll n,r,k,t,t1,t2;
string s;

ll pow(ll a,ll b){
    ll x = 1;
    ll y = a;

    while(b){
    if(b&1==1)x = (x*y)%MOD;
    b>>=1;
    y = y*y%MOD;
    }
    return x;
}

int main(){

    cin>>s>>k;n=(ll)s.size();t=1;
    rep(i,n){
    if(s[i]=='0' || s[i]=='5')r = (r+t)%MOD;
    t = t*2%MOD;
    }
    t1 = (pow(t,k)+MOD-1)%MOD;
    t2 = pow(t-1,MOD-2);
    cout<<(t1*t2%MOD)*r%MOD<<endl;

}
