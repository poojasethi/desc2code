#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
#define MOD 1000000007
char S[111111];
long long K,L;
long long ans;
long long pow(long long a,long long m)
{
    long long ret=1,p=a;
    while(m)
    {
        if(m&1)
            ret=ret*p%MOD;
        p=p*p%MOD;
        m>>=1;
    }
    return ret;
}
int main()
{
    cin>>S>>K;
    L=strlen(S);
    for(int l=0;l<L;++l)
        if(S[l]=='5'||S[l]=='0')
            ans=(ans+((pow(2,K*L)-1)*pow(pow(2,L)-1,MOD-2))%MOD*pow(2,l)%MOD)%MOD;
    cout<<ans<<endl;
    return 0;
}
