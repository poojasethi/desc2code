#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
const long p=1000000007;
long long i,k,x,n,ans;
string s;
long long power(long long a,long long x) {
	if (x==0) return 1;
	long long ans=power(a,x/2);
	ans=ans*ans%p;
	if (x%2!=0) ans=a*ans%p;
	return ans;
}
int main() {
	cin >> s >> k;
	n=s.length();
	for (i=0;i<n;i++)
		if ((s[i]=='0')||(s[i]=='5'))
			ans=(ans+power(2,i))%p;
	x=power(2,n*k)%p-1;
	x=x*power(power(2,n)-1,p-2)%p;
	ans=ans*x%p;
	cout << ans << endl;
	return 0;
}
