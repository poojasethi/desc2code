#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;

long long pows(int n,int k){
    if (k==0) return 1;
    long long ret = pows(n,k>>1);
    ret = (ret*ret)%MOD;
    if (k%2) ret = (ret*n)%MOD;
    return ret;
}

int main()
{
    string s;
    int k,n;
    cin>>s>>k;
    n = s.length();
    long long tmp = 0;
    for (int i=0;i<n;i++) {
        if (s[i]=='0'||s[i]=='5') tmp = (tmp+pows(2,i))%MOD;
    }
    long long top = pows(pows(2,n),k)-1;
    long long inv = pows(pows(2,n)-1,MOD-2);
    cout<<((inv*top)%MOD)*tmp%MOD<<endl;
    return 0;
}
