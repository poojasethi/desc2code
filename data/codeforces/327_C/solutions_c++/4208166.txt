#include <cstdio>
#include <cstring>
#define N 1000000007
#define LL long long

char c[100005];
int k, n, x = 1, t, p;

int func(int x, int k){
	if(k == 1) return x;
	int z = func(x, k / 2);
	if(k % 2) return (LL) x * z % N * z % N;
	else return (LL) z * z % N;
}

int main(){
	scanf("%s %d", c, &k);
	n = strlen(c);
	for(int j = 0; j < n; j++){
		if(c[j] == '0' || c[j] == '5') t = (t + x) % N;
		x = x * 2 % N;
	}
	printf("%d\n", (t + ((LL) t * (func(x, k) + N - x) % N * func(x - 1, N - 2) % N)) % N);
	scanf("\n");
}
