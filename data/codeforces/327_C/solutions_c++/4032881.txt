#include <iostream>
#include <cstdio>

using namespace std;

typedef long long ll;
#define Mod 1000000007

ll pow_mod(ll a,ll n)
{
	ll ans=1;
	while(n>0)
	{
		if(n & 1) ans=(ans*a)%Mod;
		a=(a*a)%Mod;
		n>>=1;
	}
	return ans;
}

int main()
{
	string str;
	ll k;
	cin>>str>>k;
	int ans=0;
	for(int i=0;i<str.size();++i)
		if(str[i]=='5' || str[i]=='0') ans=(ans+pow_mod(2,i))%Mod; 
	ll inv=pow_mod((pow_mod(2,str.size())-1),Mod-2);
	ll tmp=(inv*(pow_mod(2,str.size()*k)-1))%Mod;
	ans=(ans*tmp)%Mod;
	cout<<ans<<endl;
	return 0;
}
