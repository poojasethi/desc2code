#include <iostream>
#include <string>
#define MOD 1000000007
#define LL long long
using namespace std;

string s;
int k,n,c;
LL rez;

LL lgput(LL b,LL e) {
  LL r=1,a=b;
  for(;e;e>>=1) {
    if(e&1) r=(r*a)%MOD;
    a=(a*a)%MOD;
  }
  return r;
}

int main() {
  cin>>s>>k;
  n=s.size();
  for(int i=0; i<s.size(); ++i) if(s[i]=='0' || s[i]=='5') {
    c=i;
    LL nr=lgput(2,k*1LL*n)-1LL,num=lgput(2,n)-1LL;
    nr=(nr*lgput(2,c))%MOD;
    nr=(nr*lgput(num,MOD-2))%MOD;
    rez+=nr;
    if(rez>=MOD) rez-=MOD;
  }
  cout<<rez;
}
