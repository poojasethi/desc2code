#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll hi = 1000000007;
ll po(ll a,ll b)
{
	if(b==0)return 1;
	ll x = po(a,b/2);
	x = (x*x)%hi;
	if(b%2)x = (a*x)%hi;
	return x;
}
char a[100005];
int main()
{
	int k;
	scanf("%s%d",&a,&k);
	ll fir = 0;
	int l;
	for(l=0;a[l];l++)
	if(a[l]=='0' || a[l]=='5')
	{
		fir+= po(2,l);
		fir = fir%hi;
	}
	ll num,den;
	ll ho = l;
	ho = ho*k;
	num = po(2,ho);
	num = (num + hi -1)%hi;
	den = po(2,l);
	den = (den +hi -1)%hi;
	den = po(den,hi-2);
	num = (num*den)%hi;
	num = (num*fir)%hi;
	cout<<num;
	
	
}
