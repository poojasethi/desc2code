#include<stdio.h>
#include<stdlib.h>
#include<string.h>
typedef long long LL;
const LL Mod=1000000007;
char s[100010];
LL k,l,i,ans,p,q;
LL pow(LL x,LL y){
   if (y==0) return 1;
   if (y==1) return x;
   LL t=pow(x,y/2);
   return t*t%Mod*((y&1)?x:1)%Mod;        
}
int main(){
    gets(s);
    scanf("%I64d",&k);
    l=strlen(s);
    for (i=0;i<l;i++)
        if (s[i]=='0' || s[i]=='5')
           ans=(ans+pow(2,i))%Mod;
    if (k<2){printf("%I64d\n",ans*k);return 0;}
    q=pow(2,l);
    p=ans*(pow(q,k)-1)%Mod;
    p=p*pow((q-1),Mod-2)%Mod;
    printf("%I64d\n",p);
    return 0;
}
