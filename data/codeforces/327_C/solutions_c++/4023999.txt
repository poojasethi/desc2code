#include <iostream>
using namespace std;

typedef unsigned long long ll;

string a;
ll k, res = 0, MOD = 1000000007;

ll pow(ll a, ll p){
    ll res = 1, t = a;
    for(ll i = 0; i<64; i++){
        if(p&(1LL<<i)) res = res*t%MOD;
        t = t*t%MOD;
    }
    return res;
}

ll inv(ll x){
    return pow(x, MOD-2);
}

int main(){

    cin >> a >> k;
    for(ll i = 0; i<a.size(); i++){
        if(a[i]=='0'||a[i]=='5'){
            res = (res+pow(2, i))%MOD;
        }
    }

    cout << res*(pow(pow(2, a.size()), k)-1)%MOD*(inv(pow(2, a.size())-1))%MOD << endl;
    
}
