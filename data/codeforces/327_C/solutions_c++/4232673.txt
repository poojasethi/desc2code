#include<stdio.h>
#include<string.h>

const __int64 p = 1000000007;
char n[100000];

__int64 C(__int64 x,__int64 n)
{
	__int64 ans=1;
	while(n)
    {
		if(n&1)
               ans=ans*x%p;
		n>>=1;
		x=x*x%p;
	}
	return ans;
}

int main()
{
	__int64 k,l;
	scanf("%s %I64d",n,&k);	
	l=strlen(n);
	__int64 ans=0,t1,t2;
	for(int i=l-1;i>=0;i--)
        if(n[i]=='0'||n[i]=='5')
           ans=(ans+C(2,i))%p;
	t1=C((C(2,l)-1),p-2);
	t2=(t1*(C(2,k*l)-1))%p;
	ans=(ans*t2)%p;
	printf("%I64d\n",ans);
	return 0;
}
