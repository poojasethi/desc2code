#include<iostream>
#include<cstring>
#include<cmath>
typedef long long int ll;
using namespace std;

ll a,b,l,i,s=2,t=2,p=1000000007;
char o[100010];

ll qp(ll u,ll n)
{
	ll z;
	if(n==0)
	{
		z=1;
		return z;
	}
	if(n==1)
	{
		z=u;
		return z;
	}
	z=qp(u,n>>1);
	z*=z;
	z%=p;
	if((n&1)==1)
	z*=u;
	z%=p;
    return z;
}

int main()
{
    cin>>o>>a;
	l=strlen(o);
	while(i<l)
	{
	    if(o[i]=='0'||o[i]=='5')
	    b+=qp(2,i);
	    b%=p;
	    i++;
	}
	s=1-qp(s,l*a);
	t=1-qp(t,l);
	t=qp(t,p-2);
	cout<<((s*t)%p)*b%p;
}
