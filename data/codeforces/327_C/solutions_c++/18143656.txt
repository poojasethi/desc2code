#include<iostream>
#include<string.h>
#define mod 1000000007
using namespace std;
long long int power(long long int,long long int);
int main()
{
	char c[100001];
	cin>>c;
	long long int k,sum=0,i=0,sum2,n=strlen(c);
	cin>>k;
	for(i=0;i<n;i++)if(c[i]=='0'||c[i]=='5')sum=(sum+power(2,i))%mod;
	sum2=((power(2,n*k)-1)*power((power(2,n)-1),mod-2))%mod;
	cout<<(sum*sum2)%mod;
}
long long int power(long long int m,long long int n)
{
	long long int i,sum=1,j=m;
	while(n)
	{
		i=1;j=m;
		while(2*i<=n)j=(j*j)%mod,i*=2;
		sum=(sum*j)%mod;
		n-=i;
	}
	return sum;
}
