#include<bits/stdc++.h>
using namespace std;

const long long mod = 1e9 + 7;

int k;
long long ct = 1, ans;
char s[100001];

long long P(long long a, int now) {
  if(now == 0) return 1;
  long long x = P(a, now / 2);
  x *= x;
  x %= mod;
  if(now % 2) x *= a, x %= mod;
  return x;
}

int main() {
  scanf("%s", s);
  for(int i = 0; s[i]; i++) {
    if(s[i] == '0' || s[i] == '5') ans += ct, ans %= mod;
    ct *= 2;
    ct %= mod;
  }
  scanf("%d", &k);
  printf("%I64d\n", (((P(ct, k) - 1) * P(ct - 1, mod - 2)) % mod * ans) % mod);
  return 0;
}
