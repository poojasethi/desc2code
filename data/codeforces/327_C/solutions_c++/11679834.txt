#include<bits/stdc++.h>
using namespace std;
#define ll long long
const ll p=1000000007;
char s[100001];
ll C(ll x,ll n){
    ll ans=1;
    while(n){
        if(n&1)ans=ans*x%p;
        n>>=1;
        x=x*x%p;
    }
    return ans;
}
int main(){
    ll k,l;
        cin>>s;
        cin>>k;
        l=strlen(s);
        ll ans=0,t1,t2;
        for(int i=l-1;i>=0;i--)if(s[i]=='0'||s[i]=='5')ans=(ans+C(2,i))%p;
        t1=C((C(2,l)-1),p-2);
        t2=(t1*(C(2,k*l)-1))%p;
        ans=(ans*t2)%p;
        printf("%lld\n",ans);
    return 0;
}
