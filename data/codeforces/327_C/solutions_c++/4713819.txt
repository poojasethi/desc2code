 #include<cstdio>
 #include<algorithm>
 using namespace std;
 int k;
 long long mo = 1e9+7,ct = 1ll,ans;
 char s[100001];
 long long P(long long a,int now){
    if(now == 0) return 1ll;
    long long x = P(a,now/2);
    x *= x; x %= mo;
    if(now % 2) x *= a, x %= mo;
    return x;
 }
 int main(){
    scanf("%s",s);
    for(int i=0;s[i];i++){
        if(s[i] == '0' || s[i] == '5') ans += ct,ans %= mo;
        ct *= 2; ct %= mo;
    }
    scanf("%d",&k);
    printf("%I64d\n",(((P(ct,k) - 1) * P(ct-1,mo-2))%mo * ans)%mo);
    return 0;
 }
