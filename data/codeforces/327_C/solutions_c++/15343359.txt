#include<iostream>
using namespace std;


typedef long long ll;

ll mo=1e9+7;


ll pow(ll a,ll n)
{
	ll ans=1;
	while(n)
	{
		if (n&1)
		ans=(ans*a)%mo;
		a=(a*a)%mo;
		n>>=1;
	}
	return ans;
}

ll dengbi(ll n,ll k)
{
	if (n==1)
	return 1;
	if (n%2==0)
	return (dengbi(n/2,k)%mo*(pow(k,n/2)+1))%mo;
	if (n%2==1)
	return (dengbi(n-1,k)%mo+pow(k,n-1))%mo;
}


ll n,k,ans;

string s;

int main()
{
	cin>>s>>k;
	int x=1;
	for (int i=0;i<s.size();i++)
	{
		if (s[i]=='5'||s[i]=='0')
		ans=(ans+x)%mo;
		x=(x+x)%mo;
	}
	ll k1=pow(2,s.size());
	cout<<(dengbi(k,k1)*ans)%mo;
}
