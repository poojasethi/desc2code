#include <iostream>
#include <string>
using namespace std;

typedef unsigned long long ull;
const ull MOD = 1000000007;

string s;
ull k;
ull l;
ull r;

ull gpow(ull b,ull x){
	if(x==0) return 1;
	if(x==1) return b;
	ull a = gpow(b,x/2);
	a = (a*a) % MOD;
	if(x%2 != 0) a = (a*b) % MOD;
	return a;
}

int main(){
	cin >> s;
	cin >> k;
	l = s.length();
	
	r = 0;
	for(ull i=0; i<l; i++){
		if(s[i]=='5' || s[i]=='0') r = (r + gpow(2,i)) % MOD;
	}
	
	r = (r * (gpow(2,l*k) - 1)) % MOD;
	r = (r * gpow(gpow(2,l)-1, MOD-2)) % MOD;
	
	cout << r;
	return 0;
}
