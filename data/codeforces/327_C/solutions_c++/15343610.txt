#include<iostream>
using namespace std;

typedef long long ll;

const ll mo=1e9+7;

ll pow(ll a,ll n)
{
	ll ans=1;
	while(n)
	{
		if (n&1)
		ans=(ans*a)%mo;
		a=(a*a)%mo;
		n>>=1;
	}
	return ans;
}

ll bi(ll n,ll k)
{
	if (n==1)
	return 1;
	if (n%2==0)
	return (bi(n/2,k)%mo*(pow(k,n/2)+1)%mo)%mo;
	else
	return (bi(n-1,k)%mo+pow(k,n-1))%mo;
}
string s;
ll k,x=1,ans;

int main()
{
	cin>>s>>k;
	for (int i=0;i<s.size();i++)
	{
		if ((s[i]-48)%5==0)
		ans=(ans+x)%mo;
		x=(x+x)%mo;
	}
	cout<<(ans*bi(k,pow(2,s.size())))%mo;
}
