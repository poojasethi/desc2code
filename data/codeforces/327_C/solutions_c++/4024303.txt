#include<stdio.h>
typedef long long ll;
ll mod;
ll pow(ll a,ll b){
    ll as=1;
    while(b){
        if(b&1)as=(as*a)%mod;
        a=(a*a)%mod;b/=2;
    }
    return as;
}

    mod=1000000007;
    ll as=0;

    for(n=0;s[n];n++)
        if(s[n]=='0'||s[n]=='5')
            as=(as+pow(2,n))%mod;
    as=as*((pow(2,n*k)-1+mod)%mod)%mod;

}
