#include<bits/stdc++.h>
#define M7 1000000007
#define ifor(s,n) for(int i=(s);i<(n);i++)
using namespace std;
typedef long long ll;

ll qpow(ll a,ll p){
	ll ans=1;
	ll mul=a;
	while(p>0){
		if(p&1LL)
			ans=(mul*ans)%M7;
		mul=(mul*mul)%M7;
		p>>=1;
	}
	return ans;
}
ll inv(ll x){
	return qpow(x,M7-2);
}
