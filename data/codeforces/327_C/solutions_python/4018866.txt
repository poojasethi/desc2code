import sys
MOD = 1000000007

A = sys.stdin.readline().strip()
k = int(sys.stdin.readline().strip())

base = 0
for i, ch in enumerate(A):
	if ch == "0" or ch == "5":
		base += pow(2, i, MOD)
		base %= MOD

n = len(A)
sn = ((pow(1 << n, k, MOD) - 1) % MOD)
sd = (((1 << n) - 1) % MOD)
sdinv = pow(sd, MOD-2, MOD)
s = (sn * sdinv) % MOD
print (base * s) % MOD
