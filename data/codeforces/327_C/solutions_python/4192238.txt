def pow(a,n,m):
    res = 1
    while n > 0:
        if (n & 1) == 1:
            res = (res * a) % m
        a = (a*a) % m
        n >>= 1
    return res

a = map(int,raw_input())
k = int(raw_input())
l = len(a)
m = 1000000007
f = pow(2,l,m)
f = (pow(f,k,m) - 1) * pow(f - 1,m-2,m)
f %= m
p = 1
ans = 0

for d in a:
    if d == 0 or d == 5:
        ans += p
        ans %= m
    p = (p * 2) % m

print((ans * f)%m)
