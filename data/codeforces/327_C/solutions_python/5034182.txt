#!/usr/bin/env python2

MOD = 10**9+7

def power(d, n):
    rst = 1
    while n != 0:
        if n%2 == 1:
            rst = rst*d%MOD
        d = d*d%MOD
        n /= 2
    return rst

def cal(n, m, l):
    if n == 1:
        return m
    tmp = cal(n/2, m, l)
    tmp += power(power(2, l), n/2)*tmp
    tmp %= MOD
    if n%2 == 1:
        tmp += m*power(2, l*(n-1))
    tmp %= MOD
    return tmp

n = raw_input()
k = int(raw_input()) # < 10**9
l = len(n)           # < 10**5

m = 0
t = 1
for i in xrange(0, l):
    if n[i] == '0' or n[i] == '5':
        m += t
        m %= MOD
    t *= 2
    t %= MOD

print cal(k, m, l)
