mod = 1000000007

def inv_mod(a, m):
    a %= m
    return pow(a, m-2, m)

a = raw_input()

t = len(a)
d,x = 0,1
for i, c in enumerate(a):
    if c in ('0', '5'):
        d = (d + x) % mod 
    x = (2 * x) % mod
