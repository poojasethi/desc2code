s = raw_input()
k = input()
LEN = len(s)
md = int(1e9 + 7)
p2 = [1 for i in xrange(LEN)]
for i in xrange(1, LEN):
    p2[i] = p2[i-1]<<1
    if p2[i] >= md:
        p2[i] -= md
c = []
for i in xrange(LEN):
    if s[i] == '0' or s[i] == '5':
        c.append(i)
sa, sb = 0, 0
for ci in c:
    sa += p2[ci]
    if md <= sa:
        sa -= md
u = pow(2, LEN, md)
sb = (pow(u, k, md) - 1) * pow(u - 1, md-2, md) % md
if sb < 0:
    sb += md
print (sa * sb) % md
