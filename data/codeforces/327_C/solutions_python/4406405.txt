MOD = 1000000007 

find_list = lambda s,c:[i for i in range(len(s)) if s[i] == c]
def solver():
    s = raw_input()
    n = input()

    place5 = find_list(s,'5')
    place0 = find_list(s,'0')
    ans = 0
    #print place5, place0
    A = (pow(2,len(s)*n,MOD)-1) * pow((pow(2,len(s),MOD)-1),MOD-2,MOD)
    for x in place5:  
        ans += pow(2,x,MOD) 

    for x in place0:
        ans += pow(2,x,MOD)

    ans *= A

    print ans % MOD
    
    




if __name__ == "__main__":
    solver()
