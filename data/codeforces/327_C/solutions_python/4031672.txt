s = raw_input()
k = input()
LEN = len(s)
md = int(1e9 + 7)
c = []
for i in xrange(LEN):
    if s[i] == '0' or s[i] == '5':
        c.append(i)
sa, sb = 0, 0
for ci in c:
    sa += pow(2, ci, md)
    if md <= sa:
        sa -= md
u = pow(2, LEN, md)
sb = (pow(u, k, md) - 1) * pow(u - 1, md-2, md) % md
if sb < 0:
    sb += md
print (sa * sb) % md
