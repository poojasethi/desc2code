a = raw_input()
k = input()

s1 = 0
s2 = 0
MOD = 1000000007
for i, x in enumerate(a):
    if x == '0' or x == '5':
        s1 += pow(2, i, MOD)
        s1 %= MOD

n = len(a)
num = pow(2, n * k, MOD) - 1
denom = pow(2, n, MOD) - 1
multinv = pow(denom, MOD - 2, MOD)
tmp = (num * multinv) % MOD

print (s1 * tmp) % MOD
