#include <iostream>
#include <algorithm>
using namespace std;

int a[500500], n;
int main()
{
	cin>>n;
	for (int i=0;i<n;i++) {
	    cin>>a[i];
	}
	sort(a,a+n);
	int m=n-1;//the last
	for (int i=n/2-1;i>=0;i--){
		if (a[i] * 2 <= a[m]){
			m--;
		}
	}
	cout<<m+1<<endl;
	return 0;
}
