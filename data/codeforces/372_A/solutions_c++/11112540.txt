#include<iostream>
#include<cmath>
#include<cstdio>
#include<algorithm>
using namespace std;
int a[500001];
int n,l,r;
int main()
{
	cin>>n;
	int i;
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	l=n/2,r=n;
	for(;l>=1;l--)
		if(a[l]*2<=a[r])
			r--;
	cout<<r;
	return 0;
}
