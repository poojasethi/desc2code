#include <cstdio>
#include <algorithm>
using namespace std;
int main()
{
  int a[500005], n, i;
  scanf("%d", &n);
  for(i = 0; i < n; i++) 
    scanf("%d", &a[i]);
  sort(a, a+n);
  for(i = n / 2 - 1; i >= 0; i--)
    if(a[i] * 2 <= a[n - 1])
        n--;
  printf("%d\n", n);
}
