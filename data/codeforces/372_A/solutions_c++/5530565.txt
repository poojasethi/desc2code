#include<stdio.h>
#include<algorithm>
using namespace std;
main()
{
      int i,j,n,a[500000],x;
      scanf("%d",&n);
      x=n;
      j=n-1;
      for(i=0;i<n;i++)scanf("%d",&a[i]);
      sort(&a[0],&a[n]);
      for(i=n/2-1;i>=0;i--)if(a[i]*2<=a[j])j--,x--;
      printf("%d",x);
      scanf(" ");
}
