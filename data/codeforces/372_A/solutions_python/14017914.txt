n = int(raw_input())

k = []
for i in xrange(n):
    k.append(int(raw_input()))

k.sort()
v = len(k)
j = v / 2
r = n

for i in xrange(n / 2):
    while True:
        if (k[j] >= k[i] * 2):
            r -= 1
            j += 1
            break
        else:
            j += 1

        if (j == v): break

    if (j == v): break

print r
