#!/usr/bin/python
# Kangaroos

num = int(input())
sizes = []
for n in range(num):
    sizes.append(int(raw_input()))

sizes.sort()
counter, i, j  = 0 ,0 , num/2

while i<num/2 and j < num:
    while j < num and sizes[i]*2 > sizes[j]:
        j += 1
    if j < num:
        i += 1
        j += 1
        counter += 1

                    

print num - counter
