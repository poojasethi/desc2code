n = int(raw_input())
kangaroos = []
for i in xrange(n):
    kangaroos.append(int(raw_input()))

kangaroos = sorted(kangaroos, reverse = True)

import math
h = int(math.ceil(float(n)/2.0))
parents = kangaroos[:h]
ind_parent = 0

for k in kangaroos[h:]:
    if k * 2 > parents[ind_parent]:
        parents.append(k)
    else:
        ind_parent += 1

print len(parents)
