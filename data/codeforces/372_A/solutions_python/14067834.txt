kangaroos = int(raw_input())
weights = []

for i in xrange(kangaroos):
    weight = int(raw_input())
    weights.append(weight)
    

weights.sort()

visible = kangaroos
pocket = kangaroos / 2

for i in xrange(kangaroos/2):
    while True:
        if weights[pocket] >= (weights[i] * 2):
                visible -= 1
                pocket += 1
                break
        else:
            pocket += 1
            
        if pocket == kangaroos:
            break
    if pocket == kangaroos:
        break
