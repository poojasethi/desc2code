def main():
    n=input()
    s=sorted(int(raw_input()) for i in xrange(n))
    m=n/2-1
    r=n-1
    R=0
    while m>=0:
        if s[m]*2 <= s[r]:
            r-=1
            R+=1
        m-=1
    print n-R
    
if __name__ == '__main__':
    main()
