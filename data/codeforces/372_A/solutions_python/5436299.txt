import bisect
def main():
    n = map(int, raw_input().split())[0]
    size = [int(raw_input()) for i in xrange(n)]
    size.sort()
    n = len(size)
    high = n - 1
    low = n / 2
    low -= 1
    mid = low
    count = 0
    while high > mid and low >= 0:
        if size[high] >= 2*size[low]:
            count += 1
            high -= 1
        low -= 1
    print n - count

if __name__ == '__main__':
    main()
