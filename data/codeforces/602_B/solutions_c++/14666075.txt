#include <iostream>

using namespace std;

int main()
{
    long n,a[100005],cp,f,d,pp,ans;
    cp=f=pp=ans=0;
    cin>>n;
    for(long i=0;i<n;i++) cin>>a[i];
    for(long i=1;i<n;i++)
    {
        d=a[i]-a[i-1];
        if(d)
        {
            if(d==f) pp=cp;
            cp=i;
            f=d;
        }
        ans=(ans>=i-pp+1)?ans:(i-pp+1);
    }
    cout<<ans;
    return 0;
}
