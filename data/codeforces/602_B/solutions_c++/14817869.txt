#include <cstdio>
#include <iostream>
using namespace std;
int main(){
	int n, a1, a2, tmp, tmp2, tf=1, from=0, ans=2;
	scanf("%d %d %d",&n,&a1,&a2);
	tmp=a2-a1;
	a1=a2;
	for(int i=2; i<n; i++){
		scanf("%d",&a2);
		tmp2=a2-a1;
		if(tmp2!=0){
			if(tmp==tmp2) from=tf;
			else tmp=tmp2;
			tf=i;
		}
		ans=max(ans, i-from+1);
		a1=a2;
	}
	printf("%d\n",ans);
	return 0;
}
