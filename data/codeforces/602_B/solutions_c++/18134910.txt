#include <bits/stdc++.h>

using namespace std;

const int maxN=1e5+10;

int n,v[maxN],a[maxN];

int main()
{
    scanf("%d",&n);
    for (int i=0; i<maxN; i++) v[i]=0;
    for (int i=1; i<=n; i++)
        scanf("%d",&a[i]);
    int ans=0;
    for (int i=1; i<=n; i++)
    {
        v[a[i]+1]=i;
        ans=max(ans,max(i-max(v[a[i]+3],v[a[i]]),i-max(v[a[i]+2],v[a[i]-1])));
    }
    cout << ans;
}
