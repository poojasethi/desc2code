#include <cstdio>
#include <algorithm>
#include <set>
using namespace std;
int const MAX = 1e5 + 5;
int a[MAX];
multiset <int> s;

int main()
{
	int n, ans = 0, st = 0;
	scanf("%d", &n);
	for(int i = 0; i < n; i++)
		scanf("%d", &a[i]);
	for(int i = 0; i < n; i++)
	{
		s.insert(a[i]);
		while(*s.rbegin() - *s.begin() > 1)
			s.erase(s.find(a[st ++]));
		ans = max(ans, i - st + 1);
	}
	printf("%d\n", ans);
}
