#include<stdio.h>
#include<math.h>

int main()
{
	int a[100002],b[100002];	
	int i,n,l=1,r=1,ans=0,x=0,t=0;
	
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
	}
	for(i=1;i<n;i++)
	{
		b[i]=a[i+1]-a[i]; 
	}
	for(i=1;i<n;i++)
	{
		r=i;		
		if(x==b[i]&&x!=0)	l=t+1;
		if(b[i]!=0)	
		{
			x=b[i];
			t=i;
		}
		if(ans<r-l+2)	ans=r-l+2;
	}
	printf("%d\n",ans);
	return 0;
}
