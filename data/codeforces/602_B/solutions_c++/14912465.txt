#include<bits/stdc++.h>
using namespace std;

int n,d1,re,d2,d3,te,tes;
int main(){
cin>>n>>tes;d1=d2=1;d3=0;re=1;
for(int i=1;i<n;i++){
    cin>>te;
    if(te==tes) d1++,d2++;
    else if(te>tes){
        if(d3>=0) d2++;
        else d2=d1+1;
        d1=1;d3=-1;
    }
    else{
        if(d3<=0) d2++;
        else d2=d1+1;
        d1=1;d3=1;
    }
    re=max(re,d2);tes=te;
}
cout<<re<<endl;
return 0;
}
