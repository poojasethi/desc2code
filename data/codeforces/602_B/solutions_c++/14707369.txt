#include<bits/stdc++.h>
using namespace std;
int a[100010],l,x,n,i;
int main(){
    cin >> n;
    multiset<int> s;
    for(i=0;i<n;){
        if(!a[i]) cin >> a[i], s.insert(a[i]);
        if(*s.rbegin()-*s.begin()>1) s.erase(s.find(a[l++]));
        else x = max(x,(i++)-l+1);
    }
    cout << x << "\n";
    return 0;
    
}
