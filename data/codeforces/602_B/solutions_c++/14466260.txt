#include<stdio.h>
int a[100005];
int main()
{
	int i,j,k,p,count,min,max,maxi,n;
	scanf("%d",&n);
	for(i=0;i<n;i++) scanf("%d",&a[i]);
	maxi=2;
	for(i=0;i<n;i++)
	{
		min=max=a[i];
		for(j=i+1;j<n;j++)
		{
			if(a[j]>max)
			max=a[j];
			
			else if(a[j]<min)
			min=a[j];
			
			if(max-min>1)
			break;
		}
		
		if(j-i>maxi)
		maxi=j-i;
		
		if(maxi==n-i)
		break;
	}
	printf("%d",maxi);
	return 0;
}
