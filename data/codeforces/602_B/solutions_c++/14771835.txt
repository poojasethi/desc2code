#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int maxn = 100000 + 10;

int n;
int pos[maxn];

int main()
{
	scanf("%d", &n);
	int ans = 2;
	for(int i = 1; i <= n; i++) {
		int x; scanf("%d", &x);
		int t = min(max(pos[x-1], pos[x+2]), max(pos[x-2], pos[x+1]));
		ans = max(ans, i - t);
		pos[x] = i;
	}

	printf("%d\n", ans);

	return 0;
}
