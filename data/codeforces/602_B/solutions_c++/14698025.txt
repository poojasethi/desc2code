#include <iostream>
#include <cmath>

using namespace std;

int n;

int main()
{
  ios::sync_with_stdio(false);
  cin>>n;
  int sol=1;
  int aa;
  cin>>aa;
  int c1=1,c2=1;
  for (int i=1;i<n;i++) {
    int a;
    cin>>a;
    if (a==aa-1) {
      c2=c1+1;c1=1;
    } else if (a==aa) {
      c1++;c2++;
    } else if (a==aa+1) {
      c1=c2+1;c2=1;
    }
    aa=a;
    sol=max(sol,max(c1,c2));
  }
  cout<<sol<<endl;
}
