#include<bits/stdc++.h>
using namespace std;
#define ll long long 

ll a[1000000];
multiset <ll> m;
int main()
{
ll n,i,l=0,ans=0;
cin>>n;

for(i=0;i<n;i++)
cin>>a[i];

for(i=0;i<n;i++)
{
    m.insert(a[i]);
    
    while(l<i&&*(m.rbegin())-*(m.begin())>1)
    {
        m.erase(m.find(a[l++]));
    }
    ans=max(ans,(ll)m.size());
    
}
cout<<ans;
return 0;
}
