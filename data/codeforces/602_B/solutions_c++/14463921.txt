#include <bits/stdc++.h>
using namespace std;

multiset <int> ms;
int arr[100005];
int n;

int main(){
	cin >> n;
	int mx = 0;
	for(int i=1;i<=n;i++){
		cin >> arr[i];
	}
	int j=1;int i=1;
	while(j<=n){
		ms.insert(arr[j]);
		if(abs(*(ms.begin())-*(ms.rbegin()))>1){
			ms.erase(ms.find(arr[i]));
			i++;
		}
		j++;
		mx = max((int)ms.size(),mx);
	}
	cout << mx << endl;
}
