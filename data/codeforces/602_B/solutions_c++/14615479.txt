#include<bits/stdc++.h>
using namespace std ;
int main()
{
	int p1,p2,ans=0,current,prev,n ;
	cin>>n ;
	cin>>prev ;p1=p2=1 ;
	for(int i=2;i<=n;++i)
	{
		cin>>current ;
		if(current==prev+1)	p2=p1,p1=i ;
		else if (current==prev-1)  p1=p2,p2=i ;
		ans=max(ans,i-min(p1,p2)+1) ;
		prev=current ;		
	}
	cout<<ans<<'\n' ;
	return 0 ;
}
