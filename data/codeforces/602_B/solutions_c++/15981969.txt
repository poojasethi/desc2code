#include <stdio.h>
int main()
{
	int n,a[100010],b[100010]={0},c=0,l=0,ll,max=0,i,j;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		l++;
		b[a[i]]++;
		if(b[a[i]]==1) c++;
		if(c==3)
		{
			ll=l;
			for(j=i-ll+1;j<=i;j++)
			{
				l--;
				b[a[j]]--;
				if(b[a[j]]==0) c--;
				if(c==2) break;
			}
		}
		if(l>max) max=l;
	}
	printf("%d\n",max);
	return 0;
}
