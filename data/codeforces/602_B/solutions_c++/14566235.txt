#include <stdio.h>
#include <iostream>
using namespace std;

int n,i,ans,x,oldx,a,b;
main(){
         
 scanf("%d",&n); 
 oldx=-1;
 
 for (i=1;i<=n;i++){
  scanf("%d",&x);
  if (x<oldx-1 || x>oldx+1) 
    a=b=i;
  else
  if (x==oldx-1) {b=a; a=i;} 
  else
  if (x==oldx+1) {a=b; b=i;} 

  ans=max(ans,i-a+1);   ans=max(ans,i-b+1);
  oldx=x;     
  
 }
 
 printf("%d\n",ans);       
}
