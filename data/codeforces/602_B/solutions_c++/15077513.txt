#include<iostream>
#include<algorithm>
using namespace std;
long long n,i,a[1000050],ans,k;
main()
{
    cin>>n;
    for(i=1; i<=n; i++)
    {
        cin>>k;
        a[k]++;
        a[k-1]++;
        a[k+1]=0;
        a[k-2]=0;
        ans=ans>a[k]?ans:a[k];
        ans=ans>a[k-1]?ans:a[k-1];
    }
    cout<<ans<<endl;
}
