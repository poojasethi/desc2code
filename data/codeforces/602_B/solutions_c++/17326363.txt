#include <bits/stdc++.h>

using namespace std;

const int maxN = 1E5+5;

int n, a[maxN], x, ans = 0;
int main(){
    //freopen("input.txt", "r", stdin);
    cin>>n;
    for(int i = 0; i < n; ++i){
        cin>>x;
        a[x]++;a[x + 1]++;a[x-1] = 0;a[x + 2] = 0;
        ans = max(ans, max(a[x], a[x + 1]));
    }
    cout<<ans;

    return 0;
}
