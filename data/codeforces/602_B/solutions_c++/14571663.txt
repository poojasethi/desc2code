//In the Name of God

#include<iostream>
#include<set>

using namespace std;

int n,i,l,a[1<<17],ans;
multiset<int> s;

main()
{
  for(cin>>n;i<n;i++)
    {
      cin>>a[i];
      s.insert(a[i]);
      while(*s.rbegin()-*s.begin()>1)
	s.erase(s.find(a[l++]));
      ans=max(ans,(int)s.size());
    }
  cout<<ans<<endl;
  return 0;
}
