#include <iostream>
using namespace std;
int x=-1,x1,a,b,ans;
int main ()
{
    int n;
    cin>>n;
    
    for (int i=0;i<n;i++){
     cin>>x1;
     if(x1<x-1 || x1>x+1)
      a=b=i;
     else
     if (x1==x-1)
      b=a,a=i;
     else
     if (x1==x+1)
      a=b,b=i;
     
     ans=max(ans,i-a+1);
     ans=max(ans,i-b+1);
     x=x1;
}
cout<<ans<<endl;
//system ("pause");
return 0;
}
