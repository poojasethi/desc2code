#include <iostream>
#include <cmath>
#include <set>
typedef __int64 ull;
using namespace std;
int n, x, l, i, a[100000];

int main() {
	cin >> n;
	multiset<int> s;
	for(i = 0; i < n; i++){
		if(!a[i]) cin >> a[i], s.insert(a[i]);
		if(*s.rbegin() - *s.begin() > 1) s.erase(s.find(a[l++]));
		else x = max(x, i - l + 1);
	}
	cout << x << "\n";
	return 0;
	
}
