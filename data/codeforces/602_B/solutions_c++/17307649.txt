#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
main()
{
    long long n,a[100000],k[100000],c=0,i,l;
    cin>>n;
    for(i=0;i<n;i++) {
            cin>>l;
    a[l]++;
    a[l-1]++;
    a[l+1]=0;
    if(l!=1) a[l-2]=0;
    c=max(a[l],a[l-1]);
    k[i]=c;
    } sort(k,k+n);
    cout<<k[n-1];
    return 0;
}
