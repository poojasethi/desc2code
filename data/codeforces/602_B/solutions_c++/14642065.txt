#include <bits/stdc++.h>
using namespace std;
int a[100001];
int main()
{
    int n,i,ans=0,l=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    multiset<int> st;
    for(i=0;i<n;i++)
    {
        st.insert(a[i]);
        while(*st.rbegin()-*st.begin()>1)
            st.erase(st.find(a[l++]));
        ans=max(ans,(int)st.size());
    }
    printf("%d",ans);
    return 0;
}
