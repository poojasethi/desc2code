#include <bits/stdc++.h>
using namespace std;
#define N 100005
int main() {
	int n, a, p[N], mx, i;
	cin >> n;
	memset(p, -1, sizeof(p));
	for(i = 0; i < n; i++) { 
		cin >> a;
		a += 1;
		mx = max(mx, i - min(max(p[a-1], p[a+2]), max(p[a+1], p[a-2])));
		p[a] = i;
	}
	cout << mx << endl;
	return 0;
}
