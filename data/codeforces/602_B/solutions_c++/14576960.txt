#include<cstdio>

using namespace std;

int n, a[100005], j, k, x, xmax;

int main(){
	scanf("%d", &n);
	for(int i = 0; i < n; ++i)scanf("%d", a + i);
	
	for(int i = 0; i < n; ++i){
		if(i != 0 && a[i] - a[i - 1] == j && j != 0)x = i - k;
		if(i != 0 && a[i] - a[i - 1] != 0){k = i; j = a[i] - a[i - 1];}
		x++;
		if(x > xmax)xmax = x;
	}
	
	printf("%d", xmax);
	
	return 0;
}
