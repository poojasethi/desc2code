#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
#define MX 100100
int n;
int pos[MX];
int main(){
	scanf("%d",&n);
	memset(pos,-1,sizeof(pos));
	int ans=1;
	for(int i=0;i<n;++i){
		int x;
		scanf("%d",&x);
		int l=min(max(pos[x-1],pos[x+2]),max(pos[x+1],pos[x-2]));
		ans=max(ans,i-l);
		pos[x]=i;
	}
	printf("%d\n",ans);
	return 0;
}
