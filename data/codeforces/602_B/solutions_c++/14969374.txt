#include<bits/stdc++.h>
using namespace std;
int main()
{
int n;
cin>>n;
int a[n];
for(int i=0;i<n;i++)
cin>>a[i];
int m=1,l=0,r=0,c=0,d=0;
for(int i=1;i<n;i++)
{
if(abs(a[i]-a[i-1])==1)
{

if(d==(a[i]-a[i-1]))
{
l=c;
}
else
{
d=a[i]-a[i-1];
}
r=i;
c=i;
}
else if(a[i]-a[i-1]==0)
{
r=i;
}
else
{
l=i;
r=i;
c=i;
d=0;
}
m=max(m,r-l+1);
}
cout<<m;
return 0;
}
