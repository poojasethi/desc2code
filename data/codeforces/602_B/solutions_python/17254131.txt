n = input()
x = map(int, raw_input().split())

last_diff = 0
last_change = 0
current_length = 1
max_length = 1

for i in xrange(1,len(x)):
	if x[i] > x[i-1]:
		if last_change == 1:
			current_length = i - last_diff
		last_diff = i
		last_change = 1
	elif x[i] < x[i-1]:
		if last_change == -1:
			current_length = i - last_diff
		last_diff = i
		last_change = -1
	current_length += 1
	max_length = max(max_length, current_length)
print max_length
