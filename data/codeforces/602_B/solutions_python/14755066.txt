n = input()
a = map(int , raw_input().split())

N = -1
last = {}
last[a[0]] = 0;
l = 1
for i in range(1,n):
    p = [N for x in range(4)]
    if a[i] - 1 in last:
        p[0] = last[a[i] - 1]
    if a[i] - 2 in last:
        p[1] = last[a[i] - 2]
    if a[i] + 1 in last:
        p[2] = last[a[i] + 1]
    if a[i] + 2 in last:
        p[3] = last[a[i] + 2]

    p.sort()
    if p[2] == N:
        l = max(l , i + 1)
    else:
        l = max(l , i - p[2])
    last[a[i]] = i
print l
