n = int(raw_input().strip())
nums = map(int, raw_input().strip().split(' '))

taili = 0
mininsegment = 100001
maxinsegment = 0
headi = 0
best = 0
while headi < len(nums):
    if nums[headi] > maxinsegment:
        maxinsegment = nums[headi]
    if nums[headi] < mininsegment:
        mininsegment = nums[headi]
    if maxinsegment - mininsegment > 1:
        if headi - taili > best:
            best = headi - taili
        ttaili = taili
        while ttaili < headi:
            if (nums[headi] == maxinsegment and nums[ttaili] == mininsegment) or (nums[headi] == mininsegment and nums[ttaili] == maxinsegment):
                taili = ttaili + 1
            ttaili += 1
        if nums[headi] == maxinsegment:
            mininsegment = nums[taili]
        else:
            maxinsegment = nums[taili]
    headi += 1
if headi - taili > best:
    best = headi - taili
print best
