n=input()
a=map(int,raw_input().split())
b=c=a[0]
ans=1
p=[-1]*100005
p[b]=l=0
for i in range(1,n):
    if a[i] not in (b,c):
        if a[i]==c+1:
            if b!=c: l=p[b]+1
            b,c=c,a[i]
        elif a[i]==b-1:
            if b!=c: l=p[c]+1
            b,c=a[i],b
        else:
            l=i
    ans=max(ans,i-l+1)
    p[a[i]]=i
print ans
