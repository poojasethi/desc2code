n = input()
a = map(int,raw_input().split())
da = [0]*(n-1)
cnt,prev,mx,e = 0,0,0,0
v = [0,0,0]
for i in xrange(1,n):
    e = a[i] - a[i-1] 
    if e == 0:
        cnt += 1
        #print 'in 0 ',i,cnt,mx,e,prev
    else:
        if v[e] <= v[-e]: 
            cnt += 1
            #print 'in ok ',i,cnt,mx,e,v,prev
        else:
            prev = v[e]+1
            cnt = i - prev + 1 
            #print 'in notok ',i,cnt,mx,e,v,prev
        v[e] = i
    mx = max(cnt,mx)
print mx+1
