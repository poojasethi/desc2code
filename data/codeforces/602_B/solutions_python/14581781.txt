n = input()
ar = map(int,raw_input().split())
ar0 = ar[:]
ar1 = ar[:]
for i in xrange(n):
    if ar[i]%2==1:
        ar0[i] += 1
    else:
        ar1[i] += 1
ans = 1
cnt0 = 1
cnt1 = 1
for i in xrange(1,n):
    if ar0[i]==ar0[i-1]:
        cnt0 += 1
    else:
        cnt0 = 1
    if ar1[i]==ar1[i-1]:
        cnt1 += 1
    else:
        cnt1 = 1
    ans = max(cnt0,cnt1,ans)
print ans
