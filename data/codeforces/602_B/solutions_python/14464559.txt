n = int(raw_input())
aa = map(int, raw_input().split())

bb = [a / 2 for a in aa]
cc = [(a + 1) / 2 for a in aa]
cntb = 1
cntc = 1
ans = 2
for i in range(1, n):
    if bb[i] == bb[i - 1]:
        cntb += 1
        ans = max(ans, cntb)
    else:
        cntb = 1        
    if cc[i] == cc[i - 1]:
        cntc += 1
        ans = max(ans, cntc)
    else:
        cntc = 1

print ans
