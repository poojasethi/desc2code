n = input()
a = map(int , raw_input().split())

ans = 1
last_v = 0
last_i = 0
tmp = 1
for i in range(1 , n):
    s = a[i] - a[i-1]
    if s == 0:
        tmp += 1
        ans = max(ans , tmp)
    else :
        if last_v == 0 or s + last_v == 0:
            last_v = s
            last_i = i
            tmp += 1
            ans = max(ans , tmp)
        else :
            tmp = i - last_i + 1
            ans = max(ans , tmp)
            last_v = s
            last_i = i
print ans
