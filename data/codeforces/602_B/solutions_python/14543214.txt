import sys

n = input()
l = raw_input().split()
a = [int(i) for i in l]
b = [0 for i in range(100000 + 2)]

ans = 1
l=a[0]
r=a[0]
b[a[0]]=1
j=0

for i in range(1,n):
	b[a[i]]+=1
	while l>0 and b[l-1]>0:
		l-=1
	while b[r+1]>0:
		r+=1
	while r-l>1:
		b[a[j]]-=1
		while b[l]==0: l+=1
		while b[r]==0: r-=1
		j+=1
	ans=max(ans,i-j+1)

print ans
