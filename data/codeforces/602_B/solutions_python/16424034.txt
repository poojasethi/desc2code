#!/usr/bin/env python
# -*- coding: UTF-8 -*-

from sys import stdin
n = int(stdin.readline())
a = map(int, stdin.readline().split())
p = a[0]
cb = 0
cl = 0
ce = 0
max_c = 0

for e in a:
    if e == p:
        cb +=1
        cl +=1
        ce +=1
    elif e<p:
        cb = cl + 1
        cl = 1
        ce = 1
    else:
        cl = cb +1
        cb = 1
        ce = 1
    p = e
#    print e, ":" , cl, ce, cb 
    if max_c < cl: max_c = cl
    if max_c < cb: max_c = cb
    if max_c < ce: max_c = ce

print max_c
