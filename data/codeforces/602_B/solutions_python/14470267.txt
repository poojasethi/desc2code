import os

n = int(raw_input())

num = map(int, raw_input().split()) 

cou = [0] * n 
val = [0] * n
id = 0
cou[0] = 1
val[0] = num[0]
last = num[0]
ans = 2

for i in num[1:] :
    if i == last :
        cou[id] = cou[id] + 1
    else :
        id = id + 1
        cou[id] = 1
        val[id] = i
        last = i 

Max = val[0]
Min = val[0]
len = cou[0]
for i in range(1,id+1):
    Max = max(val[i], Max)
    Min = min(val[i], Min) 
    if Max - Min > 1:
        ans = max(ans, len)
        Max = max(val[i], val[i-1]) 
        Min = min(val[i], val[i-1])
        len = cou[i] + cou[i-1]
    else :
        len = len + cou[i]
ans = max(ans, len)

print ans
