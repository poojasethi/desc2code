n = input()
a = map(int, raw_input().split(" "))
idx = dict()
dp = range(n)
dp[0] = 0
ans = 1
idx[a[0]] = 0
for i in range(1, n):
    mi = idx.get(a[i] - 2, -1)
    Mi = idx.get(a[i] + 2, -1)
    lmt = max(mi + 1, Mi + 1)
    dp[i] = max(dp[i - 1], lmt)
    ans = max(ans, i - dp[i] + 1)
    idx[a[i]] = i
    # print 'i:', i, 'mi:', mi, 'Mi:', Mi, 'dp[i-1]:', dp[i - 1], 'dp[i]:', dp[i], 'idx', idx
print ans
