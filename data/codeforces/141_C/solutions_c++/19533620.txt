#include <bits/stdc++.h>

#define NN 3010
using namespace std;
typedef pair <int, int> pii;

pii p[NN];
int H[NN], a[NN], N;
char str[NN][20];

int main (){
	
	scanf("%d", &N);
	for(int i=0; i<N; i++){
		scanf("%s %d", str[i], &p[i].first);
		p[i].second=i;
	}
	
	sort(p, p+N);
	
	int h=N;
	for(int i=0; i<N; i++){
		if(i && p[i].first!=p[i-1].first) h--;
		if(p[i].first>i){
			puts("-1");
			return 0;
		}
		H[p[i].second]=h;
		for(int j=i; j>p[i].first; j--){
			a[j]=a[j-1];
		}
		a[p[i].first]=p[i].second;
	}
	
	for(int i=0; i<N; i++){
		printf("%s %d\n", str[a[i]], H[a[i]]);
	}
	
	return 0;
}
