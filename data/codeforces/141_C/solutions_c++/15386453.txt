#include<bits/stdc++.h>
using namespace std;
   
int main()
{
	int n;
	cin >> n;
	pair<int,string> p[n];
	for(int i=0;i<n;++i)
	{
		cin >> p[i].second;
		cin >> p[i].first;
	}
	sort(p,p+n);
	
	for(int i=0;i<n;++i)
{
		if(p[i].first>i)
	{
		cout << "-1";
		return 0;
	}	
}
	pair<string,int> ans[n];
		for(int i=0;i<n;++i)
{
	ans[i].first=p[i].second;
	ans[i].second=i+1-p[i].first;
	for(int j=i-1;j>=0;--j)
{
	if(ans[j].second>=ans[i].second)
		++ans[j].second;
}
	
}
	for(int i=0;i<n;++i)
{
	cout << ans[i].first << " " << ans[i].second << "\n";
}
	
}
