#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

int n,i=0,A[3003],j=0;
pair <int,string> C[3003];
pair <int,string> res[3003];
int main(){

for(i;i<n;i++)cin>>C[i].second>>C[i].first;
for(i=0;i<n;i++)res[i].first=0;
sort(C,C+n);
for(i=n-1;i>=0;i--){
    for(j=C[i].first;res[j].first!=0;j++){
        if(j+1==n){cout<<-1;return 0;}}
    res[j].first=n-i;
    res[j].second=C[i].second;
    }
for(i=0;i<n;i++){
    cout<<res[i].second<<" "<<res[i].first<<endl;
    }
    return 0;
    }
