#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

struct Node
{
	string name;
	int a, h;
}p[3005];
vector<int> h;

bool cmp(const Node &a, const Node &b)
{
	return a.a < b.a;
}

int main()
{
	int n, i;
	cin >> n;
	for(i=0; i<n; i++)
		cin >> p[i].name >> p[i].a;
	sort(p, p+n, cmp);
	int pn = 0;
	int hp = n;
	for(i=0; i<n; i++)
	{
		if(p[i].a > pn++)
		{
			cout << "-1\n";
			break;
		}
		h.insert(h.begin()+p[i].a, i);
		p[i].h = hp--;
	}
	if(i >= n)
		for(int j=0; j<n; j++)
		{
			int k = h[j];
			cout << p[k].name << ' ' << p[k].h << endl;
		}
	return 0;
}
