#include <iostream>
#include <string>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
vector < int > res;
vector < pair < int , string > > dat;
int n,a;
int heights[5000];
string S;
int main(){
cin >> n;
for(int i=0;i<n;i++){
 cin >> S >> a;
 dat.push_back(make_pair(a,S));
}    
sort(dat.begin(),dat.end());
for(int i=0;i<n;i++)
 if(dat[i].first>i) {
	cout << -1 << endl; return 0;
 }
 for(int i=0;i<n;i++){
  heights[i] = i - dat[i].first + 1;
  for(int j=0;j<i;j++)
   if(heights[i]<=heights[j])heights[j]++;
 }
 for(int i=0;i<n;i++)
  cout << dat[i].second << ' ' << heights[i] << endl;

return 0;
}
