#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
struct Que
{
	char name[15];
	int a;
}q[3005];
int cmp(Que x,Que y)
{
	return x.a<y.a;
}
int main()
{
	int n,i,j,height[3005];
	while(scanf("%d",&n) != EOF)
	{
		for(i=0;i<n;i++)
			scanf("%s%d",q[i].name,&q[i].a);
		sort(q,q+n,cmp);
		for(i=0;i<n;i++)
			if(q[i].a>i)
				break;
		memset(height,0,sizeof(height));
		if(i>=n)
		{
			for(i=0;i<n;i++)
			{
				for(j=0;j<i;j++)
					if(height[j]>=i-q[i].a)
					{
						height[j]++;
					}
				height[i]=i-q[i].a;
			}
			for(i=0;i<n;i++)
				printf("%s %d\n",q[i].name,height[i]+1);
		}
		else
			printf("-1\n");
	}
	return 0;
}
