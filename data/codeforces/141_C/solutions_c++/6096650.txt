#include<iostream>
#include<map>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;

typedef pair< int, string > pis; 

vector< pis > v;
vector< pis > ans;

int main(){
	int n;
	cin>>n;
	for(int i=0; i<n ; i++){
		int a;
		string name;
		cin>>name>>a;
		v.push_back( pis(a,name) );
	}
	sort(v.begin(), v.end());
	int h = 5000;
	for(int i=0 ; i<n ; i++){
		
		pis toIns = v[i];
		if(toIns.first > ans.size()){
			cout<<"-1"<<endl;
			return 0;
		}

		vector< pis >::iterator iter = ans.begin();
		ans.insert(iter+toIns.first, pis(h, toIns.second));
		h--;

	}

	for(int i=0; i<n ; i++){
		cout<<ans[i].second<<" "<<ans[i].first<<endl;
	}
}
