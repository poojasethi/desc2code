#include<iostream>
#include<algorithm>

#define F first
#define S second
using namespace std;

pair<int, string> a[10000];
int ans[10000];

int main()
{
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
		cin >> a[i].S >> a[i].F;
	sort(a, a + n);
	for(int i = 0; i < n; i++)
	{
		if(a[i].F > i)
		{
			cout << -1 << endl;
			return 0;
		}
		int x = i - a[i].F;
		for(int j = 0; j < i; j++)
			if(ans[j] >= x)
				ans[j]++;
		ans[i] = x;
	}
	for(int i = 0; i < n; i++)
		cout << a[i].S << ' ' << ans[i] + 1 << endl;
	return 0;
}
