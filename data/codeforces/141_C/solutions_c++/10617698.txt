#include <iostream>
#include <algorithm>
#include <string>
using namespace std ;
const int MAX = 3000+5 ;
int n , sum ;
struct node
{
	string s ;
	int x ;
} a[MAX] ;

bool cmp( node & a , node & b )
{
	return a.x < b.x ;
}

int main()
{
	cin >> n ;
	bool f = true ;
	for( int i = 0 ; i < n ; i++ )
	{
		cin >> a[i].s >> a[i].x ;
	}
	sort( a , a+n , cmp ) ;
	for( int i = 0 ; i < n ; i++ )
	{
		if( a[i].x > i )
		{
			cout << -1 << endl ;
			return 0 ;
		}
		for( int j = 0 ; j < i ; j++ )
		{
			if( a[j].x >= a[i].x ) ++a[j].x ;
		}
	}
	for( int i = 0 ; i < n ; i++ )
	{
		cout << a[i].s << ' ' << 10005 - a[i].x << endl ;
	}
	return 0 ;
}
