#include<string>
#include<iostream>
#include<algorithm>

using namespace std;

const int MAXN = 3005;

struct Person
{
	string name;
	int high;

	bool operator < (const Person & v) const
	{
		return high<v.high;
	}
};
Person person[MAXN];

int low[MAXN];

int main()
{
	int n;

	cin>>n;
	for(int i=0;i<n;i++) cin>>person[i].name>>person[i].high;
	sort(person,person+n);
	for(int i=0;i<n;i++)
	{
		if(person[i].high>i)
		{
			cout<<"-1"<<endl;
			return 0;
		}
		low[i]=i-person[i].high;
		for(int j=0;j<i;j++) if(low[j]>=low[i]) low[j]++;
	}
	for(int i=0;i<n;i++) cout<<person[i].name<<' '<<low[i]+1<<endl;
	return 0;
}
