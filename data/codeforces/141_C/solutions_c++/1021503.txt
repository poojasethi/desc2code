#include <cstdio>
#include <algorithm>
using namespace std;

int n;
bool u[3001], f;
struct e{
	char s[12];
	int a, h;
} x[3001];

bool c(e q, e p){ return q.a < p.a;}

int main(){
	scanf("%d", &n);
	for(int i = 0; i < n; i++) scanf("%s %d", x[i].s, &x[i].a);
	sort(x, x + n, c);
	for(int i = n - 1; i >= 0 && !f; i--){
		for(int j = n; j && x[i].a != -1; j--) if(!u[j]){
			x[i].a--;
			if(x[i].a == -1) x[i].h = j, u[j] = 1;
		}
		if(!x[i].h) f = 1;
	}
	if(f) printf("-1\n");
	else for(int i = 0; i < n; i++) printf("%s %d\n", x[i].s, x[i].h);
}
