#include<bits/stdc++.h>
using namespace std;
int main()
{
   // freopen("a.txt","r",stdin);
    int n,i,h[5000],indx,j;
    pair<int,string>p[5000];
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>p[i].second>>p[i].first;
        h[i]=i+1;
    }
    sort(p,p+n);
    for(i=n-1;i>=0;i--)
    {
        if(i<p[i].first)
        {
            cout<<"-1";
            return 0;
        }
        indx=i-1;
        for(j=p[i].first;j>0;j--)
        {
            swap(h[i],h[indx]);
            indx--;
        }
    }
    for(i=0;i<n;i++)
        cout<<p[i].second<<" "<<h[i]<<endl;
return 0;
}
