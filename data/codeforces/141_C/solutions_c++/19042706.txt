#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;cin>>n;
vector<pair<int,pair<string,int>    > >kill(n);
    for(int i=0;i<n;i++){
        cin>>kill[i].second.first>>kill[i].first;
        kill[i].second.second=1;
    }
sort(kill.begin(),kill.end());
for(int i=0;i<n;i++){
    if(kill[i].first>i){
        cout<<-1<<endl;
        return 0;
    }
    int k=i-kill[i].first;
    for(int j=0;j<i;j++){
        if(kill[j].second.second>=k)
            kill[j].second.second++;
    }
    kill[i].second.second=k;
}
for(int i=0;i<n;i++){
    cout<<kill[i].second.first<<' '<<kill[i].second.second+1<<endl;
}
return 0;}
