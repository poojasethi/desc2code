#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <conio.h>
using namespace std;
int main(){
	int i, n, t;
	string name;
	vector < pair <int, string> > v;
	vector < pair<string, int> > h;
	cin >> n;
	for(i = 0; i < n; ++i){
		cin >> name >> t;
		v.push_back(make_pair(t, name));
	}
	sort(v.begin(), v.end());
	for(i = 0; i < n; ++i)
		if(v[i].first > i){
			cout << "-1";
			return 0;
		}
	for(i = 0; i < n; ++i){
		h.insert(h.begin() + v[i].first, pair <string, int> (v[i].second, n - i));
	}
	for(i = 0; i < h.size(); ++i)
		cout << h[i].first << " " << h[i].second << endl;
	// _getch();
	return 0;
}
