#include<cstdio>
#include<algorithm>
using namespace std;
int n,h[3333];

struct node
{
	int w,h;
	char name[20];
} a[3333];

bool cmp(node a,node b)
{
	return a.w<b.w;
}

int main()
{
	scanf("%d",&n);
	for (int i=1; i<=n; i++)
	    scanf("%s%d",&a[i].name,&a[i].w);
	sort(a+1,a+n+1,cmp); 
	for (int i=1; i<=n; i++)
	  {
		if (a[i].w>=i)
		  {
			printf("-1\n"); return 0;
		  }
		h[i]=i;
		for (int j=1; j<=a[i].w; j++)
		    swap(h[i-j+1],h[i-j]);
	  }
	for (int i=1; i<=n; i++)
	    a[h[i]].h=i;
	for (int i=1; i<=n; i++)
	    printf("%s %d\n",a[i].name,a[i].h);
	//while (1);
	return 0;
}
