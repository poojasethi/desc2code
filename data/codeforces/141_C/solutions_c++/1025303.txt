#include<iostream>
#include<algorithm>
using namespace std;
struct man {
string name;
int num,z;
bool operator < (const man q) const {
return num<q.num;
}
} a[3005];
int main()
{
int n,i,j,b[3005]={0};
cin>>n;
for(i=1;i<=n;i++)
{
cin>>a[i].name>>a[i].num;
}
sort(a+1,a+n+1);
for(i=n;i>=1;i--)
{
if(a[i].num>=i)
{
cout<<-1;
return 0;
}
j=n+1;
do{
do{
j--;
}while(b[j]==1);
a[i].num--;
}while(a[i].num+1);
a[i].z=j;
b[j]=1;
}
for(i=1;i<=n;i++)
{
cout<<a[i].name<<" "<<a[i].z<<endl;
}
return 0;
}
