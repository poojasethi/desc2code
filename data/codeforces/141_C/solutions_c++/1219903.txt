#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

int main () {

    int n;
    int h[10000];
    pair <int, string> p[4000];

    cin >> n;
    for(int i = 0; i < n; i++)
        cin >> p[i].second >> p[i].first;

    sort(p,p + n);

    for(int i = 0; i < n; i++)
        if(p[i].first > i) {
            cout << -1;
            return 0;
        }

    vector <int> res;

    for(int i = 0; i < n; i++) {
        h[i] = n - i;
        res.insert(res.begin() + p[i].first,i);
    }

    for(int i = 0; i < res.size(); i++)
        cout << p[res[i]].second << " " << h[res[i]] << endl;

return 0;
}
