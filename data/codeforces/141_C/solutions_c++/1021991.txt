#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
const int MAXN=3333;
int n;
vector <int> jav;
struct afrad
{
	int a;
	string s;
	int h;
};
afrad A[MAXN];
bool cmp(afrad x,afrad y)
{
	if(x.a==y.a)
		return x.h<y.h;
	return x.a<y.a;
}
int main()
{
	cin >> n;
	for(int i=0;i<n;i++)
		cin >> A[i].s >> A[i].a;
	sort(A,A+n,cmp);
	for(int i=0;i<n;i++)
		if(A[i].a>i)
		{
			cout << -1 << endl;
			return 0;
		}
	int temp=0;
	for(int i=0;i<n;i++)
	{
		jav.insert(jav.end()-A[i].a,i);
		temp++;
	}
	for(int i=0;i<n;i++)
		A[jav[i]].h=10*(i+1);
	for(int i=0;i<n;i++)
		cout << A[i].s << " " << A[i].h << endl;
	return 0;
}
