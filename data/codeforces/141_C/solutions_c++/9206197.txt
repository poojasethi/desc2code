#include <bits/stdc++.h>
using namespace std;

int main () {
	int n;
	scanf("%d",&n);

	int a;
	string s;
	vector < pair < int, string > > v, ans;

	for (int i=0; i<n; i++) {
		cin >> s >> a;
		v.push_back(make_pair(a,s));
	}

	bool flag = true;
	sort(v.begin(),v.end());

	for (int i=0; i<n and flag; i++) {
		flag &= v[i].first <=i;
		if (flag) {
			ans.insert(ans.begin() + v[i].first, make_pair(n-i, v[i].second));
		}
	}
	if (!flag) {
		printf("-1\n");
		return 0;
	}
	else {
		for (int i=0; i<n; i++) {
			printf("%s %d\n", ans[i].second.c_str(),ans[i].first);
		}
	}
	return 0;
}
