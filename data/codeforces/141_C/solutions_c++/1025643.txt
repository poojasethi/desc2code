#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
#include<utility>
using namespace std;

int n,a,pplr[3001];
string name;
vector<pair<int,string> >ppl;
vector<string> ho;

int main(void){
	cin >>n;
	for(int i=0;i<n;++i){
		cin >>name >>a;
		ppl.push_back(make_pair(a,name));
	}
	sort(ppl.begin(),ppl.end());
	for(int i=0;i<ppl.size();++i){
		a=ppl[i].first;
		if(a>i){
			cout<< "-1\n";
			return 0;
		}
		ho.insert(ho.begin()+a,ppl[i].second);
	}
	for(int i=0;i<ppl.size();++i){
		cout<< ppl[i].second<< " "<< n-(int)(find(ho.begin(),ho.end(),ppl[i].second)-ho.begin())<< "\n";
	}
	return 0;
}
