#include <iostream>
#include <algorithm>

using namespace std;
struct People
{
  string s;
  int a,b;
} p[3010];
inline bool cmp(People p1, People p2) {return p1.a<p2.a;}

int main()
{ int n; cin>>n;
 for (int i=0; i<n;i++) {cin>>p[i].s>>p[i].a;}
 sort(p,p+n,cmp);
 int i,j;
 for (i=0;i<n;i++)
 { if (p[i].a<i+1)
   {
    for (j=0;j<i;j++) if (p[j].b>=i-p[i].a) p[j].b++;
    p[i].b=i-p[i].a;
   }
   else break;
 }
 if (i>=n)
   for (int i=0;i<n;i++) cout<<p[i].s<<' '<<p[i].b+1<<endl;
 else cout<<-1<<endl;
return 0;
}
