#include <iostream>
#include <string>
#include <algorithm>

using namespace std;
pair<int, string> T[3000];
int H[3000];
int main() {
  int n;
  cin >> n;
  for (int i = 0; i < n; ++i) {
    string s;
    int h;
    cin >> s >> h;
    T[i] = make_pair(h, s);
  }
  sort(T, T + n);
  for (int i = 0; i < n; ++i) {
    int f = T[i].first;
    string s = T[i].second;
    if (f > i) {
      cout << "-1";
      return 0;
    }
    for (int j = 0; j < i; ++j) {
      if (H[j] >= i - f) H[j]++;
    }
    H[i]=i-f;
  }
  for (int i = 0; i < n; ++i) {
    cout << T[i].second << " " << H[i] + 1 << endl;
  }
  return 0;
}
