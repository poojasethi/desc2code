#include<cstdio>
#include<algorithm>
using namespace std;
typedef pair<int,int> PII;
PII nr[3100];
char nume[3100][12];
int sol[3100];
int N;
int solve(){
	int i,j;
	for(i=1;i<=N;++i){
		sol[i]=i-nr[i].first;
		if(nr[i].first>=i)
			return 0;
		for(j=1;j<i;++j)
			if(sol[j]>=sol[i])
				++sol[j];
	}
	return 1;
}
int main(){
	//freopen("d.in","r",stdin);
	int i;
	scanf("%d",&N);
	for(i=1;i<=N;++i){
		scanf("%s%d",nume[i],&nr[i].first);
		nr[i].second=i;
	}
	sort(nr+1,nr+1+N);
	
	if(!solve()){
		printf("-1\n");
		return 0;
	}
	for(i=1;i<=N;++i)
		printf("%s %d\n",nume[nr[i].second],sol[i]);
	return 0;
}
