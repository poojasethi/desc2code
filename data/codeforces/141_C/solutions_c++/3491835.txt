#include <vector>
#include <stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;

pair<int,string> P[3005];

int h[3005];

int main()
{
    int n;
    cin >> n;
    for(int i=0;i<n;i++)
        cin >> P[i].second >> P[i].first;
    sort(P,P+n);
    for(int i=0;i<n;i++)
        if(P[i].first > i)
        {
            puts("-1");
            exit(0);
        }
    vector<int> ans;
    for(int i=0;i<n;i++)
    {
        h[i] = n-i;
        ans.insert(ans.begin()+P[i].first,i);
    }
    for(int i=0;i<n;i++)
        cout << P[ans[i]].second << ' ' << h[ans[i]] << '\n';
	return 0;
}
