#include<bits/stdc++.h>
using namespace std;

vector<pair<int, string> >ans(3001);
string str;
pair<int, string> names[3001];

int main() {
    int n,x;
    
    cin>>n;
    for(int i=0;i<n;i++) {
        cin>>str>>x;
        names[i] = make_pair(x,str);
    }
    
    sort(names, names+n);
    
    for(int i=0;i<n;i++) {
        if(names[i].first > i) {
            puts("-1");
            return 0;
        }
        ans.insert(ans.begin() + names[i].first, make_pair(n-i,names[i].second));
    }
    
    for(int i=0;i<n;i++) {
        cout<<ans[i].second<<" "<<ans[i].first<<endl;
    }
    return 0;
}
