#include<iostream>
#include<stdio.h>
#include<string>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;
struct node
{
	int n;
	string name;
}a[3010];
int qb[3010];
bool cmp(node aa,node bb)
{
	return aa.n<bb.n;
}
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		cin>>a[i].name>>a[i].n;
	sort(a,a+n,cmp);
	bool Sakura=0;
	for(int i=0;i<n;i++)
	{
		if(i<a[i].n)
		{
			Sakura=1;
			break;
		}
		for(int j=i;j>a[i].n;j--)
			qb[j]=qb[j-1];
		qb[a[i].n]=i;
	}
	if(Sakura)
		printf("-1\n");
	else
	{
		for(int i=0;i<n;i++)
			cout<<a[qb[i]].name<<" "<<n-qb[i]+1<<endl;
	}
}
