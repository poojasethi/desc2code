#include <algorithm>
#include <iostream>
#include <string>
using namespace std;
const int N=4000;
int n,i,j,a[N],f[N],g[N],ans[N];
string s[N];
bool cmp(int x,int y) {return a[x]<a[y];}
int main() {
	cin >> n;
	for (i=1;i<=n;i++) {
		cin >> s[i] >> a[i];
		f[i]=i;
	}
	sort(f+1,f+n+1,cmp);
	for (i=1;i<=n;i++) g[i]=f[i];
	for (i=1;i<=n;i++) {
		if (i<=a[f[i]]) break;
		for (j=i;j>i-a[f[i]];j--)
			swap(g[j],g[j-1]);
	}
	for (j=1;j<=n;j++) ans[g[j]]=j;
	if (i<=n) cout << -1 << endl;
	else for (j=1;j<=n;j++)
		cout << s[f[j]] << " " << ans[f[j]] << endl;
	return 0;
}
