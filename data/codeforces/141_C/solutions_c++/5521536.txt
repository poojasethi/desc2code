#include <bits/stdc++.h>
using namespace std;

int main(){
  ios::sync_with_stdio(0);
  int n, a;
  while(cin >> n){
    string x;
    vector< pair<int, string> > V, res;
    for(int i = 0; i < n; i++){
      cin >> x >> a;
      V.push_back(make_pair(a, x));
    }
    sort(V.begin(), V.end());
    bool good = true;
    for(int i = 0; i < n && good; i++){
      good &= V[i].first <= i;
      if(good)
	res.insert(res.begin() + V[i].first, make_pair(n - i, V[i].second));
    }
    if(!good) puts("-1");
    else
    for(int i = 0; i < n; i++)
      printf("%s %d\n", res[i].second.c_str(), res[i].first);
  }
  return 0;
}
