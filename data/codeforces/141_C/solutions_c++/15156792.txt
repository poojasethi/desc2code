#include<stdio.h>
#include<algorithm>

using namespace std;

struct xy{
	char name[20];
	int x;
	int h;
}a[3001];

bool sort_xy(xy a,xy b){
	if(a.x<b.x)
		return 1;
	return 0;
}

int main(){
	int n;
	int i,j;
	bool flag;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%s %d",a[i].name,&a[i].x);
	}
	sort(a,a+n,sort_xy);
	for(i=0;i<n;i++){
		if(a[i].x>i){
			printf("-1");
			return 0;
		}
	}
	for(i=0;i<n;i++){
		a[i].h=i-a[i].x+1;
		for(j=0;j<i;j++){
			if(a[j].h>=a[i].h)
				a[j].h++;
		}
	}
	for(i=0;i<n;i++){
		printf("%s %d\n",a[i].name,a[i].h);
	}


	return 0;
}
