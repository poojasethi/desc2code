#include <bits/stdc++.h>

using namespace std;

#define pb push_back

int main()
{
	int n, i, x;
	cin >> n;
	vector <int> v;
	pair <int, string> p[n];
	for (i = 0; i <  n; i++)
		cin >> p[i].second >> p[i].first;
	sort(p, p+n);
	for (i = 0; i < n; i++) {
		if (p[i].first > i) {
			cout << "-1" << endl;
			return 0;
		}
	}
	for (i = n; i > 0; i--)
		v.pb(n-i+1);
	vector <int> ans;
	for (i = n-1; i >= 0; i--) {
		x = p[i].first;
		ans.pb(v[v.size()-x-1]);
		v.erase(v.end()-x-1);
	}
	for (i = 0; i < n; i++)
		cout << p[i].second << " " << ans[ans.size()-i-1] << endl;
	return 0;
}
