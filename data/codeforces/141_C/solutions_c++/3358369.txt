#include <vector>
#include <string>
#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	string s;
	int x, n;
	cin >> n;
	vector < pair < int , string > > v( n ), a;
	for( int i = 0 ; i < n ; i ++ )
		cin >> v[i].second >> v[i].first;
	sort( v.begin() , v.end() );
	for( int i = 0 ; i < v.size() ; i ++ )
	{
		int cur = v[i].first;
		if( cur > i )
		{
			cout << -1 << endl;
			return 0;
		}
		a.insert( a.begin()+cur , make_pair( n - i , v[i].second ) );
	}
	for( int i = 0 ; i < v.size() ; i ++ )
		cout << a[i].second << " " << a[i].first << endl;
	return 0;
}
