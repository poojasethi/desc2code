#include <cstdio>
#include <map>
#include <vector>
#include <algorithm>
#include <deque>
#include <queue>
#include <string>

using namespace std;

int h[3000];

int main(){
	int n;
	scanf("%d",&n);

	vector<pair<int,string> > v;
	pair<int, string> p;
	char name[20];

	for(int i=0;i<n;i++){
		scanf("%s%d",name,&(p.first));
		p.second=name;
		v.push_back(p);
	}
	sort(v.begin(),v.end());


	for(int i=0;i<n;i++){
		if(v[i].first>i){
			printf("-1\n");
			return 0;
		}
		h[i]=v[i].first;
		for(int j=0;j<i;j++){
			if(h[j]>=h[i]) h[j]++;
		}
	}
	
	for(int i=0;i<n;i++){
		printf("%s %d\n",v[i].second.c_str(),5000-h[i]);
	}

	return 0;
}
