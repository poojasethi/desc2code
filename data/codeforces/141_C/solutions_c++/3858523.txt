#include <cstdio>
#include <algorithm>
#include <string>
#include <iostream>

using namespace std;

#define N 3333

int n;
pair<int, string> a[N];
int h[N];

int main()
{
    cin>> n;
    for (int i = 0; i < n; i++)
    {
        h[i] = i + 1;
        cin>> a[i].second>> a[i].first;
    }
    sort(a, a + n);
    for (int i = n - 1; i >= 0; i--)
    {
        if (i < a[i].first)
        {
            puts("-1");
            return 0;
        }
        for (int j = a[i].first - 1; j >= 0; j--)
            swap(h[i - j], h[i - j - 1]);
    }
    for (int i = 0; i < n; i++)
        cout<< a[i].second<< " "<< h[i]<< endl;
    return 0;
}
