import itertools as it
import operator as op
from sys import stdin

def solve(n, A):
    H = []
    for i, (name, a) in enumerate(A):
        if a > i: return None
        H.insert(a, (name, n-i))
    return H

def main():
    for n in it.imap(int, stdin):
        A = (l.split() for l in it.islice(stdin, n))
        A = [(name, int(a)) for name, a in A]
        A.sort(key = op.itemgetter(1))

        H = solve(n, A)
        if H:
            print '\n'.join('%s %d' % (name, h) for name, h in H)
        else:
            print -1

if __name__ == '__main__':
    main()
