f = lambda: raw_input().strip()

n, a = int(f()), []
for i in range(n):
  s = f().split(" ")
  a.append((int(s[1]), s[0]))
a.sort()
v = []
for i in range(0, n):
  s, t = a[i][1], a[i][0]
  if t > i: break
  v.insert(t, (n - i, s, t))

if i < n - 1:
  print -1
else:
  for x, y, z in v:
    print y, x
