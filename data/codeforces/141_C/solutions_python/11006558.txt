n = input()

q = []
for i in range(n):
    nam, h = raw_input().split()
    q.append([nam, int(h)])

q.sort(key=lambda x: x[1])

res = []
f = True

for k, v in enumerate(q):
    if v[1] > k:
        f = False
        break
    res.insert(v[1], [v[0], n - k])

if f:
    for i in res:
        print "%s %d" % (i[0], i[1])
else:
    print -1
