numPeople = int(raw_input())
pList = []
toprint = []


for i in range(numPeople):
	next = raw_input().split(' ')
	pList.append((next[0], int(next[1])))

pList.sort(key=lambda x:x[1])

for i in pList:
	if i[1] > len(toprint):
		print "-1"
		exit()
	else:
		toprint.insert(i[1],(i[0], numPeople))
		numPeople -= 1

for p in toprint:
	print (str(p[0]) + " " + str(p[1]))
