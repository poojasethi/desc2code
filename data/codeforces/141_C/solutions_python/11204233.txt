import math

def main():
    n = input()
    arr = []
    for _ in range(n):
        name, x = raw_input().split()
        arr.append([name, int(x)])

    arr.sort(key=lambda x: x[1])
    res = []
    for i in range(n):
        if i + 1 <= arr[i][1]:
            print -1
            return

        h = n - i
        res.insert(arr[i][1], (arr[i][0], h))

    for item in res:
        print item[0], item[1]

main()
