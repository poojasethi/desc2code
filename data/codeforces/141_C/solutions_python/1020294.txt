n = int(raw_input())
people = [0] * n

for i in xrange(n):
    people[i] = raw_input().split(' ')

people.sort(key=lambda person: int(person[1]))

result = []

i = 0

for person in people:
    nr = int(person[1])

    if nr > len(result):
        print "-1"
        exit(0)

    result.insert(nr, (person[0], n - i))
    i += 1

for i in result:
    print i[0], i[1]
