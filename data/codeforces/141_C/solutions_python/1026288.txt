# Codeforces 101 - C
n=int(raw_input())
d = []
for i in range(n):
    line = raw_input().split()
    d.append((line[0], int(line[1])))

res = []
for item in sorted(d, key=lambda x: x[1]):
    if item[1] > len(res): print -1; exit()
    else:
        res.insert(item[1], (item[0], n))
        n -= 1

for item in res:
    print item[0], item[1]
