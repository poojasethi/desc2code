import sys

N = input()
d = sorted(((int(j), i) for i, j in (raw_input().split(' ') for i in xrange(N))))

if d[0][0] != 0:
    print -1
    sys.exit()

    ret = []
    for i, (a, name) in enumerate(d):
        if not (len(ret) >= a):
            print -1
            sys.exit()

    A = sorted([(a, name , i) for i, (name, a) in enumerate(ret)])
