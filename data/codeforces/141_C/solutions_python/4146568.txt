n = int(raw_input())
data = []
for i in xrange(0,n):
    a,b = tuple(raw_input().split())
    data.append((a,int(b)))

data.sort(key=lambda x:x[1])
#print data
ans = []
flag = True
for i in data:
    if i[1] > len(ans):
        flag = False
        break
    else:
        ans.insert(i[1], i[0])
if flag:
    h = {}
    for i in xrange(0,n):
        h[ans[i]] = n-i
    for i in data:
        print i[0], h[i[0]]
else:
    print -1
