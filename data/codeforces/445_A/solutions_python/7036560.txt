#!/usr/bin/env python
# coding=utf-8
n,m = map(int,raw_input().split(' '))
for i in range(n):
    raw = raw_input()
    ans = ''
    for j in range(m) :
        if raw[j] == '.':
            if ((i + j) & 1):
                ans += 'B'
            else: ans += 'W'
        else: ans += raw[j]
    print ans
