import sys
s=raw_input()
n,m=s.split(' ')
n=int(n)
m=int(m)
l=[]
for i in range(0,n):
	l.append(raw_input())
#print l
for i in range(0,n):
	for j in range(0,m):
		if(l[i][j]=='-'):
		        sys.stdout.write('-')	
		else:
			if((i+j)%2==0):
				sys.stdout.write('B')
			else:
				sys.stdout.write('W')
        sys.stdout.write('\n')
