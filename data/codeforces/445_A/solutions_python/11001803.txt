n,m = tuple(int(i) for i in raw_input().strip().split(" "))
for i in range(n):
    x = raw_input()
    s = ""
    for j in range(m):
        if x[j] == ".":
            if (i-j)%2 == 0:
                s += "B"
            else:
                s += "W"
        else:
            s += "-"
    print s
