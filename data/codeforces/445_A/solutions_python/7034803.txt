n,m = raw_input().split(" ")
for i in range(0,int(n)):
    x = list(raw_input())
    for j in range(0,int(m)):
        if(x[j]=='.'):
            if(i%2==0):
                if(j%2==0):
                    x[j]="W"
                else:
                    x[j]="B"
            else:
                if(j%2==0):
                    x[j]="B"
                else:
                    x[j]="W"
    print ''.join(x)
