import sys

[n, m] = map(int, sys.stdin.readline().split())

board = []
for i in range(0,n):
    line = sys.stdin.readline().strip()
    board.append(line)
    

for i in range(0,n):
    line = ''
    for j in range(0,m):
        if (board[i][j] == '-'):
            line = line + '-'
        elif ((i + j) % 2 == 0):
            line = line + 'B'
        else:
            line = line + 'W'
    
    print line
