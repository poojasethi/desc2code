import sys
n,m = map(int, sys.stdin.readline().split())
board = []
for x in range(n):
	board.append(list(sys.stdin.readline().strip()))
for row in range(len(board)):
	for col in range(len(board[row])):
		if board[row][col] == '.':
			if (row+col) % 2 == 0:
				board[row][col] = 'W'
			else:
				board[row][col] = 'B'
for x in board:
	print ''.join(x)
