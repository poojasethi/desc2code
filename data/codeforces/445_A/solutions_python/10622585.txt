n,m=map(int,raw_input().split())
for i in xrange(n):
	L=list(raw_input())
	for j in xrange(m):
		if(L[j]=='-'):
			continue
		else:
			 if(i%2==0 and j%2==0):
				 L[j]="B"
                     	 if(i%2==0 and j%2==1):
				 L[j]="W"
                     	 if(i%2==1 and j%2==0):
				 L[j]="W"
                      	 if(i%2==1 and j%2==1):
				 L[j]="B"
	print "".join(L)
