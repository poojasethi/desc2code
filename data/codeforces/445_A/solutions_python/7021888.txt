sp = raw_input().split(' ')
n = int(sp[0])
m = int(sp[1])

for i in xrange(n):
    s = raw_input()
    out = ''
    for j in xrange(m):
        if s[j] == '-':
           out += '-'
        else:
            if (i + j) & 1 == 1:
                out += 'B'
            else:
                out += 'W'
    print out
