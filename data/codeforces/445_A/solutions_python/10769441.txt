n,m = raw_input().split()

n = int(n)
m = int(m)

color = 'WB'

for i in range(n):
    x = raw_input()
    out = ''
    for j in range(0,m,2):
        if x[j] == '-':
            out+='-'
        else:
            out+=color[(j+i)%2]
        if j == m-1:
            break
        if x[j+1] == '-':
            out+='-'
        else:
            out+=color[(j+i+1)%2]
    print out
