n,m = map(int , raw_input().split())

for  i in range(n):
    st=""
    if i % 2 ==0 :
        res = "BW" * (m/2)
        if m % 2 !=0:
            res += "B"
    else :
        res = "WB" *( m/2)
        if m% 2 !=0:
            res +="W"

    k = raw_input()
    for j in range(len(k)):
        if k[j] == '.':
            st+=res[j]
        else:
            st+='-'
    print st
