n,m=map(int,raw_input().split())
t=''
for i in xrange(n):
        s=raw_input()
        for j in xrange(m):
            if(s[j]=='.'):
                if((i+j)%2==0):
                    t+='B'
                else:
                    t+='W'
            else:
                t+=s[j]
i=0
while(i<n*m):
    print t[i:i+m]
    i+=m
