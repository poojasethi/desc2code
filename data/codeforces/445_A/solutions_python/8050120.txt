n, m = map(int, raw_input().split())

board = []

for i in xrange(n):
	board.append(list(raw_input()))

p = ['W', 'B']
t = 0
y = 0

for i in xrange(n):
	t = y
	for j in xrange(m):
		if board[i][j] == ".":
			board[i][j] = p[t]

		t += 1
		t %= 2
	y += 1
	y %= 2


for i in xrange(n):
		print ''.join(board[i])
