#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int n,m;
int a[110];
int ans[10010];
int main(){
	cin>>n>>m;
	for(int i=0;i<n;i++){
		int x;
		cin>>x;
		a[x]+=2;
		ans[2*i]=x;
		ans[2*i+1]=x;
	}
	sort(ans,ans+2*n);
	int an=n;
	for(int i=1;i<=m;i++){
		if (a[i]>n){an=2*n-a[i];break;}
	}
	cout<<an<<endl;
	for(int i=0;i<n;i++){
		if (i&1)
		cout<<ans[i]<<" "<<ans[i+n]<<endl;
		else
		cout<<ans[i+n]<<" "<<ans[i]<<endl;
	}
	return 0;
}
