#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int num[105],a[10005];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++)
	{
		int x;
		cin>>x;
		num[x]++;
		a[i*2]=x;
		a[i*2+1]=x;
	}
	int ans=n;
	for(int i=1;i<=m;i++) 
	{
		if(num[i]*2>n) ans=2*(n-num[i]);
	}
	cout<<ans<<endl;
	sort(a,a+n+n);
	for(int i=0;i<n;i++)
	{
		if(i%2==0) cout<<a[i]<<" "<<a[i+n]<<endl;
		else cout<<a[i+n]<<" "<<a[i]<<endl;
	}
	return 0;
}
