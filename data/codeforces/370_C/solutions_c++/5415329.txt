#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxn=5000+10;
int chi[maxn];

int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++)scanf("%d",&chi[i]);
    sort(chi,chi+n);
    int ans=0;
    for(int i=0;i<n;i++)
        if(chi[i]!=chi[(i+n/2)%n])ans++;
    printf("%d\n",ans);
    for(int i=0;i<n;i++)
        printf("%d %d\n",chi[i],chi[(i+n/2)%n]);
    return 0;
}
