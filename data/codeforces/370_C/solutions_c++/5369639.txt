#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <iostream>
using namespace std;
int a[5001],b[5001],n,m;
int main()
{
	scanf("%d%d",&n,&m);
	int i,j,ans=0;
	for(i=1;i<=n;i++){scanf("%d",&a[i]);b[i]=a[i];}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			if(a[i]!=b[j]&&b[i]!=a[j])
			{
				if(a[i]==b[i]||a[j]==b[j])
					swap(b[j],b[i]);
			}
		}
	}
	for(i=1;i<=n;i++)if(a[i]!=b[i])ans++;
	printf("%d\n",ans);
	for(i=1;i<=n;i++)printf("%d %d\n",a[i],b[i]);
	
}
