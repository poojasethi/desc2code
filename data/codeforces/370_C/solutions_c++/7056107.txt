#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
    int n,m,i,j,num,c[5005];
    while(scanf("%d%d",&n,&m) != EOF)
    {
        for(i=0;i<n;i++)
            scanf("%d",&c[i]);
        sort(c,c+n);
        for(i=0,num=0;i<n;i++)
            if(c[i]!=c[(i+n/2)%n])
                num++;
        printf("%d\n",num);
        for(i=0;i<n;i++)
            printf("%d %d\n",c[i],c[(i+n/2)%n]);
    }
    return 0;
}
