#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n, m; cin >> n >> m;
    int use[110] = {0};
    int a[5050], ans = 0;
    for (int i=0; i<n; i++) cin >> a[i];
    sort(a, a+n);
    for (int i=0; i<n; i++)
        if (a[i] != a[(i+n/2)%n]) ans ++;
    cout << ans << endl;
    for (int i=0; i<n; i++)
        cout << a[i] << ' ' << a[(i+n/2)%n] << endl;

    return 0;
}
