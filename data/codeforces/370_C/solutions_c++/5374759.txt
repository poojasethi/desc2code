#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int a[10010];
int n,m;
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1; i<=n; i++)
	{
		scanf("%d",&a[i]);
		a[i+n]=a[i];
	}
	sort(a+1,a+2*n+1);
	int cnt=0;
	for (int i=1; i<=n; i++)
	{
		if (a[i]!=a[i+n]) cnt++;
	}
	printf("%d\n",cnt);
	for (int i=1; i<=n; i++)
	{
		if (i%2==1) printf("%d %d\n",a[i],a[i+n]);
		else printf("%d %d\n",a[i+n],a[i]);
	}
	return 0;
}
