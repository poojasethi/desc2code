#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<algorithm>
#define N 5002
using namespace std;
int n,m;
int c[N];
int main(){
    while(cin>>n>>m){
         for(int i = 0;i<n;++i)
                 cin>>c[i];
         sort(c,c+n);
         int ans = 0;
         for(int i = 0;i<n;++i)
                 if(c[i]!=c[(i+n/2)%n])ans++;
         cout<<ans<<endl;
         for(int i = 0;i<n;++i)
                 cout<<c[i]<<" "<<c[(i+n/2)%n]<<endl;
    }
    return 0;
}
