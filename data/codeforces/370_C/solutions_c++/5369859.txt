#include <cstdio>

#define N 5000

int a[N], b[N], n, m, i, j, k = 0;

int main()
{
	scanf("%d%d", &n, &m);

	for (i = 0; i < n; i ++)
	{
		scanf("%d", &a[i]);
		b[i] = a[i];
	}

	for (i = 0; i < n; i ++)
		if (a[i] == b[i])
			for (j = 0; j < n; j ++)
				if (a[j] != b[i] && a[i] != b[j])
				{
					int z = b[i];
					b[i] = b[j];
					b[j] = z;
					break;
				}

	for (i = 0; i < n; i ++)
		if (a[i] != b[i])
			++k;

	printf("%d\n", k);

	for (i = 0; i < n; i ++)
		printf("%d %d\n", a[i], b[i]);

	return 0;
}
