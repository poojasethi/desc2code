#include<cstdio>
#include<algorithm>
using namespace std;
const int MAX=5010;
int num[MAX];
int main()
{
    int n,m;
    while(scanf("%d%d",&n,&m)!=EOF)
    {
        for(int i=0;i<n;i++)
            scanf("%d",&num[i]);
        int ans=0;sort(num,num+n);
        for(int i=0;i<n;i++)
            if(num[i]!=num[(i+n/2)%n])
                ans++;
        printf("%d\n",ans);
        for(int i=0;i<n;i++)
            printf("%d %d\n",num[i],num[(i+n/2)%n]);
    }
    return 0;
}
