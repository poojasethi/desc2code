#include<bits/stdc++.h>
using namespace std;
int n,m,a[5002],cnt,i;
int main()
{
    ios::sync_with_stdio(false);cin.tie(0);
    cin>>n>>m;
    for(i=0;i<n;i++)cin>>a[i];
    sort(a,a+n);
    for(i=0;i<n;i++)if(a[i]!=a[(i+n/2)%n])cnt++;
    cout<<cnt<<'\n';
    for(i=0;i<n;i++)cout<<a[i]<<" "<<a[(i+n/2)%n]<<'\n';	
	return 0;
}
