#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
int a[5005];
int main ( ) {
	int n,m,i,s=0;
	scanf("%d%d",&n,&m);
	for (i=0;i<n;i++) scanf("%d",&a[i]);
	sort(a,a+n);
	for (i=0;i<n;i++) {
		if (a[i]!=a[(i+n/2)%n]) s++;
	}
	printf("%d\n",s);
	for (i=0;i<n;i++) printf("%d %d\n",a[i],a[(i+n/2)%n]);
	return 0;
}
