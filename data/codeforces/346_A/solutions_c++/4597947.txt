#include <bits/stdc++.h>

using namespace std;

int n,maxx=-1,gcd=0;

void solve()
{
	int i,a;
	cin >> n ;
	for(i=1 ; i<=n ; i++)
	{
		cin >> a ;
		maxx=max(maxx,a);
		if(!gcd) gcd=a;
		else gcd=__gcd(gcd,a);
	}
	(((maxx/gcd)-n)%2) ? puts("Alice") : puts("Bob") ;
}

int main()
{
	solve();
	return 0;
}
