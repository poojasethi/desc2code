#include <bits/stdc++.h>
using namespace std;
int n;
int main()
{
	scanf("%d", &n);
	int fpb=0;
	int maks = 0 ;
	for(int i=0;i<n;i++)
	{
		int a;
		scanf("%d", &a);
		maks = max(maks,a);
		if(fpb) fpb = __gcd(a,fpb);
		else fpb=a;
	}
	int banyak = maks/fpb;
	puts(((banyak-n)&1) ? "Alice" : "Bob");
	return 0;
}
