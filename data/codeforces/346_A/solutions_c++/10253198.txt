#include <bits/stdc++.h>
using namespace std;
#define gcd(x, y) __gcd(x,y)
int main(){
	ios_base::sync_with_stdio(0), cin.tie(0);
	int n, m = 0, g = 0;
	cin >> n;
	for(int i = 0, t ; i < n ; i++){
		cin >> t;
		m = max(m, t);
		g = gcd(g, t);
	}
	cout << ((m / g - n) % 2 ? "Alice":"Bob") << endl;
}
