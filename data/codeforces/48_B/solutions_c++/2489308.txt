#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <iostream>

using namespace std;

int n,m,a,b,c[59][59],sum[59][59];

int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			scanf("%d",c[i]+j);
	for(int i=1;i<=n;i++)
		for(int j=1,s=0;j<=m;j++)
			s+=c[i][j],sum[i][j]=sum[i-1][j]+s;
	scanf("%d%d",&a,&b);
	int ret=n*m;
	for(int i=a;i<=n;i++)
		for(int j=b;j<=m;j++)
			ret=min(ret,sum[i][j]-sum[i-a][j]-sum[i][j-b]+sum[i-a][j-b]);
	for(int i=b;i<=n;i++)
		for(int j=a;j<=m;j++)
			ret=min(ret,sum[i][j]-sum[i-b][j]-sum[i][j-a]+sum[i-b][j-a]);
	printf("%d\n",ret);
	return 0;
}
