#include<iostream>
using namespace std;
int a[100][100];
int main()
{
	int n,m,x,y,ans=10000;
	cin>>n>>m;
	for(int i=1;i<=n;i++)for(int j=1;j<=m;j++)cin>>a[i][j],a[i][j]+=a[i][j-1]+a[i-1][j]-a[i-1][j-1];
	cin>>x>>y;
	for(int i=x;i<=n;i++)for(int j=y;j<=m;j++)
	{
		ans=min(ans,a[i][j]-a[i][j-y]-a[i-x][j]+a[i-x][j-y]);
	}
	for(int i=y;i<=n;i++)for(int j=x;j<=m;j++)
	{
		ans=min(ans,a[i][j]-a[i][j-x]-a[i-y][j]+a[i-y][j-x]);
	}
	cout<<ans<<endl;
	return 0;
}
