#include <iostream>
using namespace std;
int n, m, c, b, y, k, sum, minn = 1000000000;
int a[500][500],i,j;
int main()
{
	cin >> n >> m;
	for(i = 0; i < n; i++)
		for(j = 0; j < m; j++)
			cin >> a[i][j];
	cin >> b >> c;
	for (i=0;i<=n-b;i++)
		for (j=0;j<=m-c;j++)
		{
			sum=0;
			for (k=0;k<b;k++)
				for (y=0;y<c;y++)
					sum+=a[i+k][j+y];
			if (sum<minn)
				minn=sum;
		}
	for (i=0;i<=n-c;i++)
		for (j=0;j<=m-b;j++)
		{
			sum=0;
			for (k=0;k<c;k++)
				for (y=0;y<b;y++)
					sum+=a[i+k][j+y];
			if (sum<minn)
				minn=sum;
		}
	cout << minn;
	return 0;
}
