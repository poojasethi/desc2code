#include<stdio.h>
int main()
{
	int m,n,a,b,i,j,min=2500,cnt,p,q;
	int arr[51][51];
	scanf("%d%d",&m,&n);
	for(i=0;i<m;i++){
		for(j=0;j<n;j++)
			scanf("%d",&arr[i][j]);
	}
	scanf("%d%d",&a,&b);
	for(i=0;i<=m-a;i++){
		for(j=0;j<=n-b;j++){
			cnt=0;
			for(p=i;p<i+a;p++){
				for(q=j;q<j+b;q++){
					if(arr[p][q]==1)
						cnt++;
				}
			}
			if(cnt<min)
				min=cnt;
		}
	}
	if(a!=b){
		for(i=0;i<=m-b;i++){
			for(j=0;j<=n-a;j++){
				cnt=0;
				for(p=i;p<i+b;p++){
					for(q=j;q<j+a;q++){
						if(arr[p][q]==1)
							cnt++;
					}
				}
				if(cnt<min)
					min=cnt;
			}
		}
	}
	printf("%d\n",min);
	return 0;
}
