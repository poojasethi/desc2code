#include <iostream>
#include <algorithm>
using namespace std;

#define REP(i,b,n) for(int i=b;i<n;i++)
#define rep(i,n) REP(i,0,n)

int solve() {
  int n,m,a,b,garden[50][50];
  cin>>n>>m;
  rep(i,n) rep(j,m) cin>>garden[i][j];
  cin>>a>>b;
  
  int mcnt = 1<<28;
  rep(i,n-a+1) rep(j,m-b+1) {
    int cnt = 0;
    rep(ii,a) rep(jj,b) if (garden[i+ii][j+jj]) cnt++;
    mcnt = min(mcnt,cnt);
  }
  rep(i,n-b+1) rep(j,m-a+1) {
    int cnt = 0;
    rep(ii,b) rep(jj,a) if (garden[i+ii][j+jj]) cnt++;
    mcnt = min(mcnt,cnt);
  }
  
  return mcnt;
}

int main(int argc, char *argv[]) {
  cout<<solve()<<endl;
  return 0;
}
