#include <cstdio>
#define REP(i,a,b) for(int i=(a); i<(b); ++i)
#define rep(i,n) REP(i,0,n)
#define INF (1<<20)
int g[51][51];
int main()
{
	int n,m,a,b,res=INF;
	scanf("%d%d", &n,&m);
	rep(i,n)rep(j,m) scanf("%d", &g[i][j]);
	scanf("%d%d", &a,&b);

	rep(i,n-a+1) rep(j, m-b+1) {
		int t=0;
		REP(r,i,i+a) REP(c,j,j+b) if (g[r][c]) t++;
		if (res > t) res = t;
	}
	rep(i,n-b+1) rep(j, m-a+1) {
		int t=0;
		REP(r,i,i+b) REP(c,j,j+a) if (g[r][c]) t++;
		if (res > t) res = t;
	}
	printf("%d\n", res);
	return 0;
}
