#include <iostream>

using namespace std;

int n,m;
int a[60][60];

int main(){
	cin >> n >> m;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			cin >> a[i][j];
	int x,y;
	int ans=1000000;
	cin >> x >> y;
	//	cout << "Salam " << x <<" " << y << endl;
	for(int tt=0;tt<=1;tt++){
		int c=x;
		x=y;
		y=c;
		for(int i=1;i<=n-x+1;i++)
			for(int j=1;j<=m-y+1;j++){
				int sum=0;
				for(int k=0;k<x;k++)
					for(int h=0;h<y;h++){
						//					cout << i+k << " " << j+h << endl;
						sum+=a[i+k][j+h];
					}
				ans=min(ans,sum);
			}
	}
	cout << ans << endl;
}
