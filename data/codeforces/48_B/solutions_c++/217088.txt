#include<iostream>
#include<cstdlib>

using namespace std;


int main(void)
{
	int s[50][50];
	int min;
	int t;
	int n,m;
	int a,b;
	int i,j,k,l;

	cin>>n>>m;
	min=n*m;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			cin>>s[i][j];

	cin>>a>>b;

	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
		{	
			t=0;
			if((i+a<=n)&&(j+b<=m))
			{
				for(k=0;k<a;k++)
					for(l=0;l<b;l++)
						t+=s[i+k][j+l];
				if(t<min) min=t;
			}

			t=0;
			 if((i+b<=n)&&(j+a<=m))
			{
				for(k=0;k<a;k++)
					for(l=0;l<b;l++)
						t+=s[i+l][j+k];
				if(t<min) min=t;
			}
		}
	cout<<min;
//	cout<<endl;
//	system("pause");
	return 0;
}
