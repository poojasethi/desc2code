#include <iostream>
#include <cstring>
using namespace std;
int a[55][55];
int main()
{
	int n,m,n1,m1;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)cin>>a[i][j];
	cin>>n1>>m1;
	int mini=2000000000;
	
	for(int i=1;i<=n-n1+1;i++)
		for(int j=1;j<=m-m1+1;j++)
		{
			int sum=0;
			for(int i1=i;i1<=i+n1-1;i1++)
				for(int j1=j;j1<=j+m1-1;j1++)sum+=a[i1][j1];
			if (sum<mini)mini=sum;
		}
		
	for(int i=1;i<=n-m1+1;i++)
		for(int j=1;j<=m-n1+1;j++)
		{
			
			int sum=0;
			for(int i1=i;i1<=i+m1-1;i1++)
				for(int j1=j;j1<=j+n1-1;j1++)sum+=a[i1][j1];
			if (sum<mini)mini=sum;
		}
	cout<<mini;
	return 0;
}
