#include <iostream>

using namespace std;

const int Maxn = 50 + 10;
int map[Maxn][Maxn];
int m, n;
int a, b;

int num(int sx, int sy, int a, int b)
{
	int ct = 0;
	for(int i = sy;i < sy + b;i++)
		for(int j = sx;j < sx + a;j++)
		{
			if(map[j][i])
				ct++;
		}

	return ct;
}

int main()
{
	cin >> n >> m;

	for(int i = 0;i < n;i++)
		for(int j = 0;j < m;j++)
			cin >> map[j][i];

	cin >> a >> b;

	int ans = 50 * 50 + 1000;

	for(int i = 0;i <= n - b;i++)
		for(int j = 0;j <= m - a;j++)
			ans = min(ans, num(j, i, a, b));
	
	for(int i = 0;i <= n - a;i++)
		for(int j = 0;j <= m - b;j++)
			ans = min(ans, num(j, i, b, a));

	cout << ans << endl;
}
