#include<iostream>
using namespace std;
int g[100][100],n,m;
int rec(int a,int b){
  int f=0,min=100000;
  for(int s=0;s<=n;s++){
   for(int t=0;t<=m;t++){
     f=0;
    for(int i=s;i<a+s && a+s<=n;i++){
     for(int j=t;j<b+t && b+t<=m;j++){
     if(g[i][j]==1)f++;
      }
     }
   
    if(f<min &&a+s<=n &&b+t<=m)min=f;
    }
   }

 return min;
}

int main(){
  int a,b,l,o;
  cin>>n>>m;
  for(int i=0;i<n;i++){
    for(int j=0;j<m;j++){
      cin>>g[i][j];
   }
  }


    cin>>a>>b;
    l=rec(a,b);
    o=rec(b,a);
    if(l<o)cout<<l<<endl;
    else cout<<o<<endl;
    return 0;
    
}
