#include<iostream>

using namespace std;

int n,m,b[51][51];

int f(int x,int y)
{
	int i,j,s=m*n+1;

	for(i=1;i<=n-x+1;i++)
		for(j=1;j<=m-y+1;j++)
			s=min(s,b[i+x-1][j+y-1]-b[i+x-1][j-1]-b[i-1][j+y-1]+b[i-1][j-1]);

	return s;
}

int main()
{
	int i,j;

	cin>>n>>m;

	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			cin>>b[i][j];	

	for(j=2;j<=m;j++)
		b[1][j]+=b[1][j-1];
	for(i=2;i<=n;i++)
		b[i][1]+=b[i-1][1];

	for(i=2;i<=n;i++)
		for(j=2;j<=m;j++)
			b[i][j]+=b[i-1][j]+b[i][j-1]-b[i-1][j-1];

	cin>>i>>j;

	cout<<min(f(i,j),f(j,i));

	return 0;
}
