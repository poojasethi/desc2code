#include <iostream>

using namespace std;

const int MAXn=50;
const int INF=2501;

int map[MAXn+10][MAXn+10];

int main()
{
  int n, m, a, b;
  cin >> n >> m;

  for(int i=0; i<n; i++)
    for(int j=0; j<m; j++)
      cin >> map[i][j];

  cin >> a >> b;

  int result=INF;
  int counter;
  for(int x=0; x<2; x++)
    {
      for(int i=0; i<=n-a; i++)
	for(int j=0; j<=m-b; j++)
	  {
	    counter=0;
	    for(int l=0; l<a; l++)
	      for(int k=0; k<b; k++)
		counter+=map[i+l][j+k];
	    result=min(result, counter);
	  }
      swap(a, b);
    }
  
  cout << result << endl;
  return 0;
}
