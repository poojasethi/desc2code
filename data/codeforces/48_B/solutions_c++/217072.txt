#include<cstdio>

int main(void){
	int i,j,k,l;
	int n,m,a,b;
	int cnt,min;
	int c[60][60]={0};
	scanf("%d %d",&n,&m);
	for(i=0;i<n;i++)for(j=0;j<m;j++)scanf("%d",&c[i][j]);
	scanf("%d %d",&a,&b);
	cnt=0;
	min=a*b;
	for(i=0;i<=n-a;i++){
		for(j=0;j<=m-b;j++){
			for(k=i;k<i+a;k++){
				for(l=j;l<j+b;l++){
					if(c[k][l]==1)cnt++;
				}
			}
			if(cnt<min)min=cnt;
			cnt=0;
		}
	}
	for(i=0;i<=n-b;i++){
		for(j=0;j<=m-a;j++){
			for(k=i;k<i+b;k++){
				for(l=j;l<j+a;l++){
					if(c[k][l]==1)cnt++;
				}
			}
			if(cnt<min)min=cnt;
			cnt=0;
		}
	}

	printf("%d\n",min);
	return 0;
}
