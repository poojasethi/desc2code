#include<iostream>
using namespace std;
int main()
{
int n,m,k,l,i,j,mn,x,y,a[55][55],p;
cin>>n>>m;
mn=n*m;
for(i=1;i<=n;i++)
{
for(j=1;j<=m;j++)
{
cin>>a[i][j];
}
}
cin>>x>>y;
for(i=1;i<=n-x+1;i++)
{
for(j=1;j<=m-y+1;j++)
{
p=0;
for(k=i;k<=i+x-1;k++)
{
for(l=j;l<=j+y-1;l++)
{
if(a[k][l])
{
p++;
}
}
}
mn=min(mn,p);
}
}
for(i=1;i<=n-y+1;i++)
{
for(j=1;j<=m-x+1;j++)
{
p=0;
for(k=i;k<=i+y-1;k++)
{
for(l=j;l<=j+x-1;l++)
{
if(a[k][l])
{
p++;
}
}
}
mn=min(mn,p);
}
}
cout<<mn;
return 0;
}
