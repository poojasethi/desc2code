#include <stdio.h>
#include <map>
#include <string>
using namespace std;
int p[100][100];
int main(){
	int n,m,a,b,k,l;
	scanf("%d%d",&n,&m);
	int i,j;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			scanf("%d",&p[i][j]);
	scanf("%d%d",&a,&b);
	int ans=500000;
	for(i=0;i<=n-a;i++)
		for(j=0;j<=m-b;j++){
			int tem=0;
			for(k=0;k<a;k++)
				for(l=0;l<b;l++)
					tem+=p[i+k][j+l];
			if(tem<ans) ans=tem;
		}

	for(i=0;i<=n-b;i++)
		for(j=0;j<=m-a;j++){
			int tem=0;
			for(k=0;k<b;k++)
				for(l=0;l<a;l++)
					tem+=p[i+k][j+l];
			if(tem<ans) ans=tem;
		}
	printf("%d\n",ans);



}
