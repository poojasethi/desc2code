#include<cstdio>
int f[51][51];
int main(){
	int n,m,a,b,i,j,x,y,p,q,ans,t;
	while(~scanf("%d%d",&n,&m)){
		for(i=0;i<n;i++)for(j=0;j<m;j++)scanf("%d",&f[i][j]);
		scanf("%d%d",&a,&b);
		ans=2500;
		for(i=0;i<n;i++)for(j=0;j<m;j++){
			p=i+a-1,q=j+b-1;
			if(p<n&&q<m){
				t=0;
				for(x=i;x<=p;x++)for(y=j;y<=q;y++)t+=f[x][y];
				if(t<ans)ans=t;
			}
			p=i+b-1,q=j+a-1;
			if(p<n&&q<m){
				t=0;
				for(x=i;x<=p;x++)for(y=j;y<=q;y++)t+=f[x][y];
				if(t<ans)ans=t;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
