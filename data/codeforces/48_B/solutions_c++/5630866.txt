#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
int a[51][51],sum[51][51],ans,x,y,i,j,n,m;
void work(int x,int y){
	int i,j,s;
	for(j=1;j+y-1<=m;++j){
		s=0;
		for(i=1;i<=n;++i){
			s+=sum[i][j+y-1]-sum[i][j-1];
			if(i>=x){
				s-=sum[i-x][j+y-1]-sum[i-x][j-1];
				ans=min(ans,s);
			}
		}
	}
}
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i)for(j=1;j<=m;++j){
		scanf("%d",&a[i][j]);
		sum[i][j]=sum[i][j-1]+a[i][j];
	}
	scanf("%d%d",&x,&y);
	ans=1<<30;
	work(x,y);
	work(y,x);
	printf("%d\n",ans);
}
