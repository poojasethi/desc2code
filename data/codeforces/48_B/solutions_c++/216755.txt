#include <iostream>
#include <cstring>
using namespace std;

int h,w;
int a,b;
int t[52][52];

int solve(int si,int sj){
	int res = 0;
	for(int i=si;i<si+a;i++){
		for(int j=sj;j<sj+b;j++){
			res += t[i][j];
		}
	}
	return res;
}

int main(void){
	cin>>h>>w;

	for(int i=0;i<h;i++){
		for(int j=0;j<w;j++){
			cin>>t[i][j];
		}
	}
	cin>>a>>b;

	int ans = 999999999;
	for(int i=0;i<=h-a;i++){
		for(int j=0;j<=w-b;j++){
			ans = min(ans,solve(i,j));
		}
	}
	swap(a,b);
	for(int i=0;i<=h-a;i++){
		for(int j=0;j<=w-b;j++){
			ans = min(ans,solve(i,j));
		}
	}
	cout<<ans<<endl;

	return 0;
}
