#include <stdio.h>
#include <algorithm>
using namespace std;
int a[55][55];

int main(void) {
	int n, m;
	scanf("%d%d", &n, &m);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++) {
			scanf("%d", &a[i][j]);
			a[i][j] += a[i-1][j] + a[i][j-1] - a[i-1][j-1];
		}
	int x, y;
	scanf("%d%d", &x, &y);
	if(x>n || y>m) swap(x, y);
	int ans = a[x][y];
	for(int i=x;i<=n;i++)
		for(int j=y;j<=m;j++)
			ans = min(ans, a[i][j]-a[i-x][j]-a[i][j-y]+a[i-x][j-y]);
	for(int i=y;i<=n;i++)
		for(int j=x;j<=m;j++)
			ans = min(ans, a[i][j]-a[i-y][j]-a[i][j-x]+a[i-y][j-x]);
	printf("%d\n", ans);
	return 0;
}
