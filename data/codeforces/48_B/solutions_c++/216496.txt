#include<cstdio>

int plan[50][50];
int min=2501;
int sum;
int N,M,A,B;

int main(){
 int i,j,k,l,c;
 scanf("%d%d",&N,&M);
 for(i=0;i<N;i++) for(j=0;j<M;j++) scanf("%d",&plan[i][j]);
 scanf("%d%d",&A,&B);
 for(i=0;i<N-A+1;i++){
  for(j=0;j<M-B+1;j++){
   sum=0;
   for(k=0;k<A;k++){
    for(l=0;l<B;l++){
     sum+=plan[i+k][j+l];
    }
   }
   if(sum<min) min=sum;
  }
 }

 c=A;A=B;B=c;
 for(i=0;i<N-A+1;i++){
  for(j=0;j<M-B+1;j++){
   sum=0;
   for(k=0;k<A;k++){
    for(l=0;l<B;l++){
     sum+=plan[i+k][j+l];
    }
   }
   if(sum<min) min=sum;
  }
 }


 printf("%d\n",min);

 return 0;
}
