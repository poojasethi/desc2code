#include<cstdio>
#include<algorithm>
using namespace std;

#define INF 0x3f3f3f3f

int Map[100][100];

int n,m,ans=INF;

int cal(int x,int y)
{
	int ans=0;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			if(Map[x+i][y+j]) ans++;
	return ans;
}

int main()
{
	int N,M;
	scanf("%d%d",&N,&M);
	for(int i=1;i<=N;i++)
		for(int j=1;j<=M;j++)
			scanf("%d",&Map[i][j]);

	scanf("%d%d",&m,&n);
	for(int i=0;i<=N-n;i++)
		for(int j=0;j<=M-m;j++)
			ans=min(ans,cal(i,j));

	swap(m,n);
	for(int i=0;i<=N-n;i++)
		for(int j=0;j<=M-m;j++)
			ans=min(ans,cal(i,j));

	printf("%d",ans);
	return 0;
}
