#include <iostream>
#include <cstring>
#include <cmath>
#include <stdio.h>
#include <algorithm>

using namespace std;


int i,j,k,n,m,mas[101][101],a,b,res=200000000;
int main()
{
//	freopen("input.txt","r",stdin);
//	freopen("output.txt","w", stdout);
	cin>>n>>m;
	for(i=0;i<n;++i)
		for(j=0;j<m;++j)
			cin>>mas[i][j];
	cin>>a>>b;
	for(k=0;k<2;++k)
	{
	for(i=0;i<n;++i)
		for(j=0;j<m;++j)
			if (i+a<=n && j+b<=m)
			{
				int sum=0;			
				for(int ii=i;ii<i+a;ii++)
					for(int jj=j;jj<j+b;++jj)
						sum+=mas[ii][jj];
				res=min(sum,res);	
			}
		swap(a,b);
	}
	cout<<res;
	return 0;
}
