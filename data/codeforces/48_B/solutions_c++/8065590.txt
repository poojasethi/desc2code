#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int map[60][60];
int n,m; 
int a,b;
int ans;
int solve(int a, int b) 
{
	int best=50*50;
	for(int i=0;i<n-b+1;i++)
		for(int j=0;j<m-a+1;j++)
		{
			int cnt=0;
			for(int k=0;k<b;k++)
				for(int l=0;l<a;l++)
					cnt+=map[i+k][j+l];
			best=min(best,cnt);
		}
	return best;
}
int main()
{
	cin>>n>>m;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			scanf("%d",&map[i][j]);
	cin>>a>>b;
	ans=min(solve(a,b),solve(b,a));
	cout<<ans<<endl;
	return 0;
}
