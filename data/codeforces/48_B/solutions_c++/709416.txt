#include <iostream>
using namespace std;
int n,m,x[2],a[60][60],ans=1000000000,z;
int f(int k,int q,int l){
    int ret=0;
    for(int i=k;i<k+x[l];i++)for(int j=q;j<q+x[!l];j++)ret+=a[i][j];
    return ret;}
int main(){
    cin>>n>>m;
    for(int i=0;i<n;i++)for(int j=0;j<m;j++)cin>>a[i][j];
    cin>>x[0]>>x[1];
    for(int k=0;k<=1;k++)for(int i=0;i<n-x[k]+1;i++)for(int j=0;j<m-x[!k]+1;j++)
        ans=((z=f(i,j,k))<ans?z:ans);
    cout<<ans;}
