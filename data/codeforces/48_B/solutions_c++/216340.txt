#include <iostream>
#include <cstdio>
using namespace std;

int R,C,G[50][50],A,B,ans,tmp;

int main(){
	scanf("%d%d",&R,&C);
	for(int i=0;i<R;++i)
		for(int j=0;j<C;++j) scanf("%d",&G[i][j]);
	ans = R*C;
	scanf("%d%d",&A,&B);
	for(int i=0;i<R-A+1;++i)
		for(int j=0;j<C-B+1;++j){
			tmp = 0;
			for(int a=i;a<i+A;++a)
				for(int b=j;b<j+B;++b) tmp += G[a][b];
			ans = min(tmp,ans);
		}
	for(int i=0;i<R-B+1;++i)
		for(int j=0;j<C-A+1;++j){
			tmp = 0;
			for(int a=i;a<i+B;++a)
				for(int b=j;b<j+A;++b) tmp += G[a][b];
			ans = min(tmp,ans);
		}
	printf("%d\n",ans);
//	system("pause");
}
