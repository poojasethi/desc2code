#include <iostream>
using namespace std;

#define rep(i,n) for (int i = 0; i < (n); i++)

int n, m;
bool t[50][50];

int solve(int a, int b) {
	int best = 50*50;
	rep(i,n-b+1) rep(j,m-a+1) {
		int cnt = 0;
		rep(ii,b) rep(jj,a) {
			cnt += t[i+ii][j+jj];
		}
		best = min(best, cnt);
	}
	return best;
}

int main() {
	cin >> n >> m;
	rep(i,n) rep(j,m) {
		cin >> t[i][j];
	}

	int a, b;
	cin >> a >> b;

	cout << min(solve(a,b), solve(b,a)) << endl;
	return 0;
}
