#include<cstdio>
#include<algorithm>
#include<cstring>
#include<iostream>
using namespace std;
int n,m,i,j,f[200][200],q[200][200],a,b,ans;
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i){
		for(j=1;j<=m;++j){
			scanf("%d",&f[i][j]);
		}
	}
	for(i=1;i<=n;++i){
		for(j=1;j<=m;++j){
			q[i][j]=f[i][j]+q[i-1][j]+q[i][j-1]-q[i-1][j-1];
		}
	}
	scanf("%d%d",&a,&b);
	ans=0x3f3f3f3f;
	for(i=a;i<=n;++i){
		for(j=b;j<=m;++j){
			ans=min(ans,q[i][j]-q[i-a][j]-q[i][j-b]+q[i-a][j-b]);
		}
	}
	for(i=b;i<=n;++i){
		for(j=a;j<=m;++j){
			ans=min(ans,q[i][j]-q[i-b][j]-q[i][j-a]+q[i-b][j-a]);
		}
	}
	printf("%d",ans);
	return 0;	
}
