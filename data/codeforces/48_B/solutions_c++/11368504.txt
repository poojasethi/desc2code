#include <iostream>
#include <cmath>

using namespace std;

int n,m;
int suma[100][100];

int main()
{
  cin>>n>>m;
  for (int i=1;i<=n;i++) {
    for (int j=1;j<=m;j++) {
      cin>>suma[i][j];
      suma[i][j]+=suma[i-1][j]+suma[i][j-1]-suma[i-1][j-1];
    }
  }
  int a,b;
  cin>>a>>b;
  int sol=1000000000;
  for (int i=1;i<=n;i++) {
    for (int j=1;j<=m;j++) {
      if (i>=a and j>=b) sol=min(sol,suma[i][j]-suma[i-a][j]-suma[i][j-b]+suma[i-a][j-b]);
      if (i>=b and j>=a) sol=min(sol,suma[i][j]-suma[i-b][j]-suma[i][j-a]+suma[i-b][j-a]);
    }
  }
  cout<<sol<<endl;
}
