#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 60

int n,m,i,j,a,b;
int A[lim][lim];

int f(int a,int b)
{
  int i,j,mn=a*b;

  for(i=a;i<=n;i++)
    for(j=b;j<=m;j++)
      mn=min(mn,A[i][j]-A[i][j-b]-A[i-a][j]+A[i-a][j-b]);
  return mn;
}
int main()
{
  scanf("%d%d",&n,&m); 
  for(i=1;i<=n;i++)
    for(j=1;j<=m;j++)
    {
      scanf("%d",&A[i][j]);
      A[i][j]+=A[i][j-1]+A[i-1][j]-A[i-1][j-1];
    }
  scanf("%d%d",&a,&b);
  printf("%d\n",min(f(a,b),f(b,a)));
  return 0;
}
