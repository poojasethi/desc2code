import sys

n, m = map(int, raw_input().split())
grid = []
for _ in range(n):
    grid += [map(int, raw_input().split())]
a,b = map(int, raw_input().split())

t = cur = 99999

for x, y in ((a, b), (b,a)):
    for i in xrange(n-x+1):
        for j in xrange(m-y+1):
            t = 0
            for k in xrange(x):
                t += sum(grid[i+k][j:j+y])
            cur = min(cur, t)

print cur
