n,m = map(int,raw_input().split())
G = []
for i in range(n):
	G += [map(int,raw_input().split())]
a,b = map(int,raw_input().split())

def check(a,b):
	r = 99999999
	for i in range(n-a+1):
		for j in range(m-b+1):
			c = 0
			for k in range(a):
				for l in range(b):
					c += G[i+k][j+l]
			r = min(r,c)
	return r

print min(check(a,b),check(b,a))
