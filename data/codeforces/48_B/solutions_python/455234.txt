#!/usr/bin/env python
# -*- coding: utf-8 -*-

def test(g, i, j, a, b):
    if len(g) - i < a or len(g[0]) - j < b:
        return a*b
    ret = 0
    for li in xrange(i, i+a):
        ret += g[li][j:j+b].count(1)
    return ret

def main():
    n, m = map(int, raw_input().split())
    g = list()
    for _ in xrange(n):
        g.append(map(int, raw_input().split()))
    a, b = map(int, raw_input().split())
    ans = a*b
    for i in xrange(n):
        for j in xrange(m):
            ans = min(ans, test(g, i, j, a, b), test(g, i, j, b, a))
    print ans
    return 0

if __name__ == '__main__':
    main()
