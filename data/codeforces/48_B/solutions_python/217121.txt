#!/usr/bin/python
import sys
n, m = [int (x) for x in sys.stdin.readline ().split ()]
g = [[] for i in range (n)]
for i in range (n):
	g[i] = [int (x) for x in sys.stdin.readline ().split ()]
a, b = [int (x) for x in sys.stdin.readline ().split ()]
res = a * b
for i in range (n - a + 1):
	for j in range (m - b + 1):
		temp = 0
		for k in range (a):
			for l in range (b):
				temp += g[i + k][j + l]
		res = min (res, temp)
for i in range (n - b + 1):
	for j in range (m - a + 1):
		temp = 0
		for k in range (b):
			for l in range (a):
				temp += g[i + k][j + l]
		res = min (res, temp)
print res
