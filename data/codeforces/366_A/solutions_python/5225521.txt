n = raw_input()
n = int(n)
lst = []
for i in xrange(4):
    try:
        s = raw_input()
    except:
        break
    a, b, c, d = s.split(' ')
    lst.append((min(int(a),int(b)), min(int(c),int(d))))

found = False
for i in xrange(len(lst)):
    itm = lst[i]
    if itm[0] + itm[1] <= n:
        print i + 1, itm[0], n - itm[0]
        found = True
        break
if not found:
    print -1
