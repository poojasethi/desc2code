import sys
import fractions
# http://codeforces.com/problemset/problem/366/A
a = sys.stdin.readline()
n = int(a)
ret = -1
retstr = ""
for i in range(1,5):
    a = sys.stdin.readline()
    a = a.strip().split()
    a = [ int(k) for k in a]
    cost = min(a[0],a[1]) + min(a[2],a[3])
    if cost <= n:
        ret = i
        retstr = str( min(a[0],a[1])) + " "+ str(n - min(a[0],a[1]))
print ret, retstr
