import sys

def min(a, b):
        return a if a < b else b

n = input()
s = list()
for i in range(0, 4):
        [x1, y1, x2, y2] = map(int, raw_input().split(' '))
        s.append((min(x1, y1), min(x2, y2)))

k = 1
for (x, y) in s:
        if x + y <= n:
                print k, x, n-x
                sys.exit(0)
        else:
                k += 1

print -1
