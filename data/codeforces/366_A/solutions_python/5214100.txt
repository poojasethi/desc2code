#!/usr/bin/env python
from sys import stdin as cin

def main():
    n = int(next(cin))
    res = False

    for i in range(1,5):
        a1, a2, b1, b2 = map(int, next(cin).split())
        a = min(a1, a2)
        b = min(b1, b2)
        if a + b <= n:
            res = (i, a, n-a)

    return ' '.join(map(str, res)) if res else -1

print main()
