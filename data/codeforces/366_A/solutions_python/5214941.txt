n=int(raw_input())
ans=-1
for i in range(1,5):
	m=map(int,raw_input().split())
	c1=m[0]+m[2]
	c2=m[0]+m[3]
	c3=m[1]+m[2]
	c4=m[1]+m[3]
	if c1<=n:
		ans=str(i)+" "+str(m[0])+" "+str(m[2]+n-c1)
	if c2<=n:
		ans=str(i)+" "+str(m[0])+" "+str(m[3]+n-c2)	
	if c3<=n:
		ans=str(i)+" "+str(m[1])+" "+str(m[2]+n-c3)
	if c4<=n:
		ans=str(i)+" "+str(m[1])+" "+str(m[3]+n-c4)
print ans
