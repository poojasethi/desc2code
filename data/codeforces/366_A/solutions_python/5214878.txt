__author__ = 'Shailesh'


n = int(raw_input())

minV = (float("inf"), 0, 0)
minAt = 0
for i in xrange(4):
    data = raw_input().split()
    data = map(int, data)
    first = min(data[0], data[1])
    sec = min(data[2], data[3])
    if first + sec < minV[0]:
        minV = (first + sec, first)
        minAt = i + 1

if minV[0] > n:
    print -1
else:
    print minAt, minV[1], n - minV[1]
