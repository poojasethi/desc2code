n = int(raw_input())

ans = False

for i in xrange(1, 5):
  a, b, c, d = map(int, raw_input().split(' '))
  if a + c <= n:
    print i, a, n-a
    ans = True
    break
  if a + d <= n:
    print i, a, n-a
    ans = True
    break
  if b + c <= n:
    print i, b, n-b
    ans = True
    break
  if b + d <= n:
    print i, b, n-b
    ans = True
    break

if not ans:
    print -1
