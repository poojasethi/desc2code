def solve(n, g):
  for i in range(4):
    a, b = g[i][:2], g[i][2:]
    for j in range(2):
      for k in range(2):
        if a[j]+b[k] <= n:
          return i+1, a[j], n-a[j]
  return [-1]

n = int(raw_input())
g = [map(int, raw_input().split()) for i in range(4)]
print ' '.join(map(str, solve(n, g)))
