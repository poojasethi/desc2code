n = int(raw_input())
flag = True
k = -1
c = 0
for i in range(4):
	c1, j1, c2, j2 = map(int, raw_input().split())
	if flag:
		if c1 + c2 <= n:
			k = i + 1
			c = c1
		elif c1 + j2 <= n:
			k = i + 1
			c = c1
		elif j1 + c2 <= n:
			k = i + 1
			c = j1
		elif j1 + j2 <= n:
			c = j1
			k = i + 1
if k != -1: print k, c, n - c
else: print -1
