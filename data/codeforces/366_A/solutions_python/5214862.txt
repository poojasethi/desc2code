#/usr/bin/env python2

n = int(raw_input())

def solve(i):
	a,b,c,d = tuple(int(x) for x in raw_input().split(" "))
	if (min(a,b) + min(c,d)) <=n:
		print "%d %d %d"%(i+1,min(a,b),n-min(a,b))
		return True
	else:
		return False

r = False
for i in xrange(4):
	if solve(i):
		r = True
	if r:
		break
if not r:
	print "-1"
