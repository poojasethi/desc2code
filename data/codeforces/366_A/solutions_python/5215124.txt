possible = False

for num in xrange(1,5):

    for price1 in [a,b]:
        price2 = n - price1
        if price2 >= c:
            print num, price1, price2
            possible = True
        elif price2 >= d:
            print num, price1, price2
            possible = True
        if possible == True:
            break
    if possible == True:
        break
