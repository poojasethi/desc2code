n = int(raw_input())

s = []

for i in range(4):
    a,b,c,d = map(int,raw_input().split())
    t = [min(a,b),min(c,d)]
    s.append(t)
    
min = 300000

for i in range(4):
    if s[i][0] + s[i][1] < min:
        min =  s[i][0] + s[i][1]
        imin = i

if min > n:
    print "-1"
else:
    print imin+1,s[imin][0],n - s[imin][0]
