#include <iostream>
#include <cstdlib>
#include <cmath>

using namespace std;

int main()
{
  int n;
  cin>>n;
  for (int i=1;i<=4;i++) {
    int a,b,c,d;
    cin>>a>>b>>c>>d;
    int e=min(a,b);
    int f=min(c,d);
    if (e+f<=n) {
      cout<<i<<" "<<e<<" "<<n-e<<endl;
      exit(0);
    }
  }
  cout<<-1<<endl;
}
