#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
    int n;scanf("%d",&n);
    bool ok=0;
    for (int i=0;i<n;i++)
    {
	int a,b,c,d;
	scanf("%d%d%d%d",&a,&b,&c,&d);
	a=min(a,b),c=min(c,d);
	if(a+c<=n)
	{
	    ok=1;
	    printf("%d %d %d\n",i+1,a,n-a);
		break;
	}
    }
    if(!ok)printf("-1\n");
}
