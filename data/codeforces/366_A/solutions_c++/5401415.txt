#include<iostream>
using namespace std;
int main()
{
	int n; cin>>n;
	bool print=false;

	for(int i=1;i<5;i++)
	{
		int a,b,c,d; cin>>a>>b>>c>>d;
		if(print) continue;
		if(min(a,b)+min(c,d)<=n)
		{
			cout<<i<<" "<<min(a,b)<<" "<<n-min(a,b)<<endl;
			print=true;
		}
	}

	if(!print) cout<<-1<<endl;
	return 0;
}
