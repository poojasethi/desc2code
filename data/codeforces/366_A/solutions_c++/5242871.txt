#include<iostream>
using namespace std;
int n,x,x1,t,w,e,i;
main()
{
 cin>>n;
 for(i=1;i<=4;i++)
 {
  cin>>x>>x1;
  w=min(x,x1);
  x=x1=0;
  cin>>x>>x1;
  e=min(x,x1);
  if(w+e<=n){
             t=1;
             cout<<i<<" "<<w<<" "<<n-w;
             break;
            }
 }
 if(t==0)cout<<"-1";
}
