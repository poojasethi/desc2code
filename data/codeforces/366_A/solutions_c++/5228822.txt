#include <cstdio>
#include <algorithm>

using namespace std;

int main(){
	int n,c1,c2,j1,j2;
	scanf(" %d",&n);
	for(int i=0;i<4;i++){
		scanf(" %d %d %d %d" ,&c1,&j1,&c2,&j2);
		if (min(c1,j1) + min(c2,j2) <= n){
			printf("%d %d %d\n",i+1,min(c1,j1),n-min(c1,j1));
			return 0;
		}
	}
	printf("-1\n");
	return 0;
}
