#include <bits/stdc++.h>

using namespace std;

int main(){
	int n; cin >> n;
	for(int i = 0; i < 4; ++i) {
		int a, b, c, d;
		cin >> a >> b >> c >> d;
		int m1 = min(a, b);
		int m2 = min(c, d);
		if(m1 + m2 > n) continue;
		cout << i + 1 << " " << m1 << " " << n - m1;
		return 0;
	}
	cout << -1;
	return 0;
}
