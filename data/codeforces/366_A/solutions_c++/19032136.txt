#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n, i, a, b, c, d;
	scanf("%d", &n);
	for(i = 1; i <= 4; i++)
	{
		scanf("%d %d %d %d", &a, &b, &c, &d);
		a = min(a, b);
		c = min(c, d);
		if(n >= a + c)
		{
			printf("%d %d %d", i, a, n - a);
			return 0;
		}
	}
	printf("-1");
	return 0;
}
