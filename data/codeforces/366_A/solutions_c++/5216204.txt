#include <cstdio>
#include <algorithm>
using namespace std;
int n,a,b,c,d;
int main(){
    scanf("%d",&n);
    for(int i=1;i<=4;i++){
        scanf("%d%d%d%d",&a,&b,&c,&d);
        a=min(a,b);c=min(c,d);
        if(a+c<=n){printf("%d %d %d",i,a,n-a);return 0;}
    }
    printf("-1");
    return 0;
}
