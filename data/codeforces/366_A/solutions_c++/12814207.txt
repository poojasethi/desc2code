#include <iostream>

using namespace std;

int main() {
	int n;
	int a,b,c,d;
	cin >> n;
	int flg=0;
	for(int i=0; i<4; ++i) {
		cin >> a >> b >> c >> d;
		a=min(a,b);
		c=min(c,d);
		if(!flg && a+c<=n) {
			cout << i+1 << " " << a << " " << n-a << endl;
			flg=1;
		}
		
	}
	if(!flg)
		cout << -1 << endl;
	return 0;
}
