#include <cstdio>

#define N 4

int n, a, a1, b, b1, i;

int main()
{
	scanf("%d", &n);

	for (i = 1; i <= N; i ++)
	{
		scanf("%d%d%d%d", &a, &a1, &b, &b1);
		a = a < a1 ? a : a1;
		b = b < b1 ? b : b1;
		if (a + b <= n)
		{
			printf("%d %d %d\n", i, a, n - a);
			return 0;
		}
	}

	puts("-1");

	return 0;
}
