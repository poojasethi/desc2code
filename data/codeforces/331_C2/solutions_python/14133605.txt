#include<bits/stdc++.h>
		 
using namespace std;
#define foreach(i,x) for(type(x)i=x.begin();i!=x.end();i++)
#define FOR(ii,aa,bb) for(int ii=aa;ii<=bb;ii++)
#define ROF(ii,aa,bb) for(int ii=aa;ii>=bb;ii--)
#define type(x) __typeof(x.begin())

#define dbg(x) cerr << (#x) << " --> " << (x) << endl
#define y1 sdfsdfsdf
#define ll long long
#define pii pair<int,int>
#define mod 1000000007
#define N (int)(1e6)
#define mp make_pair
#define pb push_back
#define sd second
#define ft first
#define endll puts("")
#define endl '\n'
#define inf mod
ll dp[10][N+1],son[10][N+1],ans,mx,x,n;
int f(int x){
	int mx = 0;
	while(x){
		mx = max(mx,x%10);
		x/=10;	}
	return mx;	}
int main(){	
	FOR(j,0,9){
		dp[j][0] = 0;
		FOR(i,1,1000000){
			dp[j][i] = inf;
			mx = max(f(i),j);
			if(mx > i){
				 mx = i;
				 dp[j][i] = 1;
				 son[j][i] = i;
			}
			else{
				dp[j][i] = dp[j][i-mx]+1;
				son[j][i] = son[j][i-mx];
			}
		}
	}
	cin >> n;
	x = n%N;
	ROF(i,n/N,0){
		mx = f(i);
		ans += dp[mx][x];
		if(!son[mx][x] && i){
			ans++;
			x = N-mx;
		}
		else
			x = N + son[mx][x] - mx;
	}
	cout << ans << endl;
}
