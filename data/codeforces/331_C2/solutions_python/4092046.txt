#include<iostream>
#include<cstring>
#include<cstdio>
#define X 1000000

using namespace std;

int big[X];
long long n,jawab;
pair<int,long long> tuju[X][10];

int main() {
    cin>>n;
    for (int i=0;i<X;++i) {
        int sem = i;
        while (sem) {
            big[i] = max(big[i],sem % 10);
            sem /= 10;
        }
    }
    
    for (int i=0;i<10;++i) {
        for (int j=0;j<i;++j) {
            tuju[j][i] = make_pair((X-i+j) % X,1);
        }

        for (int j=i;j<X;++j) {
            tuju[j][i] = tuju[j-max(i,big[j])][i];
            if (j || i) ++tuju[j][i].second;
        }
    }
    
    if (n == 1000000000000) {
        --n;
        ++jawab;
    }
    
    long long mulai = n / X;
    long long sisa = n % X;
    
    for (long long i=mulai;i>-1;--i) {
        jawab += tuju[sisa][big[i]].second;
        sisa = tuju[sisa][big[i]].first;
    }
    
    cout<<jawab<<endl;
    return 0;
}
