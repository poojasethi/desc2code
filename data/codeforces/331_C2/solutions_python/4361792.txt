#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#define LL long long
using namespace std;

const int BASE = 1000000;
const int N = 1000050;

int need[N][10],cost[N][10];
LL n;

int cal(int x){
    int ret = 0;
    while(x) ret = max(ret,x % 10) , x /= 10;
    return ret;
}

int main(){
    // freopen("input.txt","r",stdin);
    for(int i = 1;i <= BASE;i ++)
        for(int j = 0;j < 10;j ++){
            int m = max(cal(i),j);
            if(i < m) need[i][j] = 1 , cost[i][j] = m;
            else need[i][j] = need[i - m][j] + 1 , cost[i][j] = cost[i - m][j] + m;
        }
    cin >> n;
    LL ans = 0;
    while(n > BASE){
        int ret = cal(n / BASE);
        if(n % BASE == 0) n -= ret , ans ++;
        else ans += need[n % BASE][ret] , n -= cost[n % BASE][ret];
        // cout << n << endl;
    }
    if(n) ans += need[n][0];
    /*while(n){
        int m = cal(n);
        n -= m;
        ans ++;
    }*/
    cout << ans << endl;
    return 0;
}
