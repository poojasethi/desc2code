#include <iostream>
#include <algorithm>
#include <cstdlib>
#include <cstdio>
using namespace std;
int ope[1000000][10]={};
int res[1000000][10]={};
int get[1000010]={};
int search(int n)
{
	int maxn=n%10;
	while(n)
	{
		maxn=max(maxn,n%10);
		n/=10;
	}
	return maxn;
}

int main()
{
	for (int i=0;i<1000010;++i)
		get[i]=search(i);
	for (int i=0;i<=9;++i)
		for (int j=0;j<1000000;++j)
		{
			if (i==0 && j==0)
				ope[j][i]=0;
			else
			{
				int n=j;
				int n2=n-max(get[n],i);
				if (n2<0)
				{
					res[j][i]=n2;
					ope[j][i]=1;
				}	
				else
				{
					res[j][i]=res[n2][i];
					ope[j][i]=ope[n2][i]+1;	
				}
			}
		}
	long long n;
	cin>>n;
	long long ans=0;
	while (n/1000000)
	{
		ans+=ope[n%1000000][get[n/1000000]];
		int temp=res[n%1000000][get[n/1000000]];
		n=n/1000000*1000000+temp;
	}
	ans+=ope[n][0];
	cout<<ans<<endl;
	return 0;
}
