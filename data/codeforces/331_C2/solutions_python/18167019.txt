#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<ll,int> Pair;
ll n;
map<Pair,Pair>Map;
Pair Calc(ll x,int y)
{
	if(Map.count(make_pair(x,y))) return Map[make_pair(x,y)];
	ll tmp=1;while(x/10>=tmp)tmp*=10;
	if(tmp==1) return Map[make_pair(x,y)]=make_pair(1,max(int(y-x),0));
	Pair a=Calc(x%tmp,max(y,int(x/tmp)));
	Pair b=Calc(x/tmp*tmp-a.second,y);
	return Map[make_pair(x,y)]=make_pair(a.first+b.first,b.second);
}
int main()
{
	scanf("%I64d",&n);if(!n) return !printf("0\n");
	printf("%I64d\n",Calc(n,0).first);
	return 0;
}
