#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;

typedef long long LL;
const int SQ = (int)1e6;

LL n, Ans;
int f[10][SQ+10];
int g[10][SQ+10];

void prepare(int pr, int f[], int g[])
{
     for (int i = 1, t, c; i <= SQ; i ++) {
	  c = pr, t = i;
	  while (t) c = max(c, t%10), t /= 10;
	  if (i-c >= 0)
	       f[i] = f[i-c] + 1, g[i] = g[i-c];
	  else f[i] = 0, g[i] = i;
     }
}

int main()
{
#ifndef ONLINE_JUDGE
     freopen("cf.in" , "r", stdin);
     freopen("cf.out", "w", stdout);
#endif
     cin >> n;

     for (int x = 0; x <= 9; x ++)
	  prepare(x, f[x], g[x]);

     if (n <= SQ) Ans = f[0][n];
     else {
	  int R = n % SQ, L = (n - R) / SQ;
	  while (L > 0 || R > 0) {
	       if (!L) Ans += f[0][R] + (g[0][R] > 0), R = 0;
	       else {
		    int c = 0, t = L;
		    while (t) c = max(c, t%10), t /= 10;
		    Ans += f[c][R], R = g[c][R];
		    Ans ++, L --, R = (SQ+R)-c;
	       }
	  }
     }
     
     cout << Ans << endl;
}
