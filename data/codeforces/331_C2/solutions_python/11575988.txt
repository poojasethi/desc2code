#include <bits/stdc++.h>
using namespace std;
#define PII pair <int, int> 
int maxdigit [1000005]; 
int minans [1000005]; 
PII dp [10][1000005]; 
int main()
{
    for (int g=1; g<=1000000; g++)
    {
        maxdigit[g]=max(g%10, maxdigit[g/10]);  
    }
    for (int g=1; g<=1000000; g++) minans[g]=minans[g-maxdigit[g]]+1; 
    for (int g=0; g<=9; g++)
    {
        for (int y=0; y<=1000000; y++)
        {
            int t=max(maxdigit[y], g); 
            if (y-t<0)
            {
                dp[g][y]=PII(1, y-t); 
            }
            else
            {
                PII m=dp[g][y-t]; 
                dp[g][y]=PII(m.first+1, m.second); 
            }
        }   
    }   
    long long x; cin >> x;
    if (x==0)
    {
        cout << 0; return 0; 
    }
    long long ans=0; 
    while (x>0)
    {
        int r=x%1000000; 
        PII m=dp[maxdigit[x/1000000]][r];
        ans+=m.first; 
        x=((x/1000000)*1000000)+m.second; 
    }
    cout << ans-1; 
    return 0; 
}
