#define mod 100000007
#define oo 0x3f3f3f3f
#define LL long long

map<pair<LL,int>,pair<LL,int> > mp;

pair<LL,int> cal( LL n, int x ) {
	if( n == 0 ) return make_pair( 0, 0 );
	if( n < 10 ) 
		return make_pair( 1, max( 0, (int)(x - n) ) );
	pair<LL,int> now = make_pair( n, x );
	if( mp.count(now) ) return mp[now];
	pair<LL,int> ans = make_pair( 0, 0 );
	LL ten = 1;
		while( ten  <= n / 10 ) ten *= 10;
	while( true ) {
		int df = n / ten;
		int nx = max( df, x );
		pair<LL,int> cur = cal( n % ten, nx );
		ans.first += cur.first;
		n -= n % ten;
		if( cur.second > 0 )
			n -= cur.second;
		else if( nx > 0 )
			n -= nx, ++ans.first;
		if( df == 0 )
			break;
	}
	ans.second = -n;
	mp[now] = ans;
	return ans;
} 

int main()
{
	LL n;
	cin >> n;
	cout << cal( n, 0 ).first << endl;
	return 0;
}
