#include<cstdio>
#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
typedef long long ll;
typedef pair<ll,ll> pr;
#define mk make_pair
#define x first
#define y second
map<pr,pr> f;
pr dfs(pr a)
{
	if (a.y<10) return mk(a.x || a.y,a.y-max(a.x,a.y));
	if (f.find(a)!=f.end()) return f[a];
	ll ans=0,p=1; while (p<=a.y/10) p*=10;
	pr r=dfs(mk(max(a.x,a.y/p),a.y%p)); ans+=r.x;
	pr b=mk(a.x,a.y/p*p+r.y); r=dfs(b);
	return f[a]=mk(r.x+ans,r.y);
}
int main(){ll n; cin>>n; cout<<dfs(mk(0,n)).x<<endl; return 0;}
