#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <set>
#include <map>
#define lowbit(x) (x & (-x))
using namespace std;
typedef long long LL;
const int N = 1000000;
LL n;
int cnt[N][10] , num[N][10];
int gao (LL x) {
    int ret = 0;
    while (x) ret = max (1LL * ret , x % 10) , x /= 10;
    return ret; 
}
int main () {
    #ifndef ONLINE_JUDGE
        freopen ("input.txt" , "r" , stdin);
    #endif
    cin >> n;
    for (int i = 1 ; i < N ; i ++) {
        for (int j = 0 ; j < 10 ; j ++) {
            int r = max (j , gao (i));
            if (i <= j) cnt[i][j] = 1 , num[i][j] = r;
            else cnt[i][j] = cnt[i - r][j] + 1 , num[i][j] = num[i - r][j] + r;
        }
    }
    LL ans = 0;
    while (n > N + 10) {
        int r = gao (n / N);
        if (n % N == 0) {
            n -= r;
            ans ++;
        }
        else {
            ans += cnt[n % N][r];
            n -= num[n % N][r];
        }
    }
    while (n > 0) {
        n -= gao (n);
        ans ++;
    }
    cout << ans << endl;
    return 0;
}
