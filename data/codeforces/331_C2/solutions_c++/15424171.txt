#include<bits/stdc++.h>

using namespace std;

#define mp make_pair
#define pb push_back 
#define nd second
#define st first

typedef long long ll;
typedef long long int lli;
typedef long long unsigned llu;
typedef pair < int,int > pii;
typedef pair < lli,lli > plli;

const int MAX = 1e6;

lli n;
lli dp[10][MAX+1];
lli p[10][MAX+1];
lli ans;

int main ()
{
	cin>>n;
	for(int j=0;  j<=9; j++)
		for(int i=0; i<=MAX; i++)
		{
			if(!i) continue;
			int t=i,k=j;
			
			while(t)	k=max(k,t%10),t/=10;

			if(i>=k)	dp[j][i] = dp[j][i-k] + 1, p[j][i] = p[j][i-k];	
			else		dp[j][i] = 1 , p[j][i] = i;
		}
	int r  = n % MAX;
	n /= MAX;
	while(n+1)
	{
		lli m=0,t=n;
		while(t) m=max(m,t%10),t/=10;
		ans += dp[m][r];
		if(!p[m][r] && n)
		{
			ans++;
			r = MAX - m;
		}
		else
			r = MAX + p[m][r] - m;
		n--;
	}
	cout<<ans;
}
