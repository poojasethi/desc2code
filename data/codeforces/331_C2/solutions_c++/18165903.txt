#include<map>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#define pr pair<ll,int>
using namespace std;
typedef long long ll;
map<pr,pr>mp;
pr calc(ll n,int m)
{
	if(n<=0)	return pr(0,n);
	if(n<10)	return pr(1,n-max((int)n,m));
	if(mp.find(pr(n,m))!=mp.end())	return mp[pr(n,m)];
	ll temp=n,M=1;
	while(temp)
	{
		temp/=10;
		M*=10;
	}
	M/=10;
	ll a=n/M*M;
	pr ret=calc(n%M,max(m,(int)(n/M)));
	a+=ret.second;
	temp=a;
	int tm=m;
	while(temp)
	{
		tm=max(tm,(int)(temp%10));
		temp/=10;
	}
	pr tret=calc(a-tm,m);
	ret=pr(ret.first+tret.first+1,tret.second);
	return mp[pr(n,m)]=ret;
}
int main()
{
	ll n;
	scanf("%I64d",&n);
	printf("%I64d\n",calc(n,0).first);
	fclose(stdin);
	fclose(stdout);
	return 0;
}
