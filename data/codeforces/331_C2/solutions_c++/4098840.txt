#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
#include<map>
#include<cstring>

#define ll long long

using namespace std;

int f[1000009][10],g[1000009][10];

int main()
{
    ll n;
    cin>>n;
    for(int i=1;i<10;i++)
        for(int j=0;j<10;j++)
        {
            f[i][j]=1;
            g[i][j]=max(0,j-i);
        }
   for(int i=10;i<1000009;i++)
       for(int j=0;j<10;j++)
       {
           int d=j;
           for(int k=i;k;k/=10) d=max(d,k%10);
           f[i][j]=f[i-d][j]+1;
           g[i][j]=g[i-d][j];
       }
   int x=n/1000000;
   int y=n%1000000;
   ll res=0;
   for(;x>=0;x--)
   {
       int d=0;
       for(int i=x;i;i/=10) d=max(d,i%10);
       res+=f[y][d];
       if(g[y][d])
           y=1000000-g[y][d];
       else if(x)
       {
           res++;
           y=1000000-d;
       }           
   }
   cout<<res<<endl;
}
