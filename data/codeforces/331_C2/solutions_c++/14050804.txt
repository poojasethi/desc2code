#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;
#define LL long long
#define pll pair<LL,LL>
#define x first
#define y second
map<pll,pll>f;
pll dfs(pll a){
	if(a.y<10) return pll(a.x||a.y,a.y-max(a.x,a.y));
	if(f[a].x) return f[a];
	LL ans=0,p=1;
	while(p<=a.y/10) p*=10;
	pll r=dfs(pll(max(a.x,a.y/p),a.y%p));
	ans+=r.x;
	pll b=pll(a.x,a.y/p*p+r.y);
	r=dfs(b);
	return f[a]=pll(r.x+ans,r.y);
}
LL n;
int main(){
	 scanf("%I64d",&n);
	 printf("%I64d\n",dfs(pll(0,n)).x);
}
