#include<iostream>
using namespace std;

int exp(int y){
    if(y==0) return 1;
    int ans=exp(y/2);ans=ans*ans;
    if(y&1) ans*=2;return ans;
}
int tree[262144],n,m,p,b;
void up(int x){
    int k=x,num=0;
    while(k/2){k/=2;num++;if(num&1) tree[k]=tree[k*2]|tree[k*2+1]; else tree[k]=tree[k*2]^tree[k*2+1];}
}
int main()
{
    cin>>n>>m;
    for(int i=exp(n);i<exp(n+1);i++)
       cin>>tree[i],up(i);
    for(int i=1;i<=m;i++){
       cin>>p>>b;tree[exp(n)+p-1]=b;
       up(exp(n)+p-1);cout<<tree[1]<<endl;
    }
    return 0;
}
