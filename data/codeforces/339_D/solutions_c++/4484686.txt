#include <iostream>
using namespace std;
int main(){
  long int q=1,w,e,r,t,a[262144];
  cin>>t>>w;
  while(t){
  t--;
  q*=2;}
  for(e=q,r=q*2;e<r;e++)
  cin>>a[e];
  bool c;
  for(e=q-1,r=q/2,c=0;e;e=r-1,r/=2,c=!c)
  for(t=e;t>=r;t--)
  if(c)
   a[t]=a[t*2]^a[t*2+1];
  else
   a[t]=a[t*2]|a[t*2+1];
  for(e=0;e<w;e++){
  cin>>r>>t;
  r+=q-1;
  a[r]=t;
  c=0;
  while(r){
  r/=2;
  c=!c;
  if(c)
  a[r]=a[r*2]|a[r*2+1];
  else
  a[r]=a[r*2]^a[r*2+1];}
  cout<<a[1]<<"\n";}
  return 0;}
