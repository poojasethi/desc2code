
#include <cstdio>

const int N = 1<<17;

int tree[2*N+10];
int n,m;

bool isOr(int id)
{
	int level=0;
	while( (1<<level) <= id) level++;
	level--;	
	return (n-level)%2;
}

void update(int id)
{
	id/=2;
	if(id==0)return;
	int L = tree[2*id], R = tree[2*id+1];
	tree[id] = isOr(id) ? L | R : L^R;
	update(id);
}

int main()
{
	scanf("%d%d",&n,&m);
	int len=1<<n;
	for(int i=0;i<len;++i){
		int a;
		scanf("%d",&a);	
		tree[len+i]=a;
		update(len+i);
	}

	while(m--){
		int index, val;	
		scanf("%d%d",&index,&val);
		tree[len+index-1] = val;
		update(len+index-1);
		printf("%d\n",tree[1]);
	}

	return 0;	
}
