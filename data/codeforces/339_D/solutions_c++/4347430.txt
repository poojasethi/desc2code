#include <iostream>

using namespace std;

int calculo[1<<20];
int primero;

void inserta(int pos,int val)
{
  pos+=primero;
  calculo[pos]=val;
  int paso=0;
  while (pos!=1) {
    pos/=2;
    if (paso%2==0)
      calculo[pos]=calculo[2*pos]|calculo[2*pos+1];
    else
      calculo[pos]=calculo[2*pos]^calculo[2*pos+1];
    paso++;
  }
}

int main()
{
  int n,m;
  cin>>n>>m;
  primero=1<<n;
  for (int i=0;i<(1<<n);i++) {
    int x;
    cin>>x;
    inserta(i,x);
  }
  for (int i=0;i<m;i++) {
    int pos,val;
    cin>>pos>>val;
    pos--;
    inserta(pos,val);
    cout<<calculo[1]<<endl;
  }
}
