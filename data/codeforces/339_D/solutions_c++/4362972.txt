#include <cstdio>

using namespace std;

int i, j, m, n, f[1 << 19];

void modify(int x, int v, int p){
  if(p == 1)f[x] = v;
  else
    if(p % 2)f[x] = f[x << 1 | 1] ^ f[x << 1];
    else f[x] = f[x << 1 | 1] | f[x << 1];
  if(x > 1)modify(x >> 1, v, p + 1);
}

int main(){
  scanf("%d %d", &n, &m);
  n = 1 << n;
  for(i = 0; i < n; i++){
    int c;
    scanf("%d", &c);
    modify(n + i, c, 1);
  }
  while(m--){
    int pos, b;
    scanf("%d %d", &pos, &b);
    modify(n + pos - 1, b, 1);
    printf("%d\n", f[1]);
  }
  return 0;
}
