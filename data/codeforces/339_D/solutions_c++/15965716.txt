#include <bits/stdc++.h>
using namespace std;
const int N = 1<<17;
int tree[N*2],ar[N],n;
void build(int pos,int v){
	int x=0;
	pos+=n;
	tree[pos]=v;
	while(pos!=1){
		pos>>=1;
		if(x)
			tree[pos]= (tree[pos<<1]^tree[pos<<1|1]);
		else
			tree[pos]=(tree[pos<<1]|tree[pos<<1|1]);
		x^=1;
	}
}
int main(){
	int a,b,i,m,j;cin>>n>>m;
	n = 1<<n;
	for(i=0;i<n;++i)
		cin>>ar[i],build(i,ar[i]);
	while(m--){
		scanf("%d%d",&a,&b);--a;
		build(a,b);
		printf("%d\n",tree[1]);
	}
}
