#include<cstdio>
#include<iostream>
#include<string>
using namespace std;
int a[1000000];
void u(int i)
{
    bool t=true;
    i/=2;
    while(i)
    {
        if(t) a[i]=a[i*2]|a[i*2+1];
        else a[i]=a[i*2]^a[i*2+1];
        t=!t;
        i/=2;
    }
}
int main()
{
    int i,j,n,m,p,b;
    cin>>n>>m;
    n=(1<<n);
    for(i=n;i<2*n;i++)
    {
        cin>>a[i];
        u(i);
    }
    for(i=0;i<m;i++)
    {
       cin>>p>>b;
       a[p+n-1]=b;
       u(p+n-1);
       cout<<a[1]<<endl;
    }
    return 0;
}
