#include <iostream>

using namespace std;
int num[(1<<17)*2+1];
int put(int v,int p)
{
	num[p]=v;
	bool tt=true;
	while(p)
	{
		
		p/=2;
		if(tt)
			num[p]=num[p*2]|num[p*2+1];
		else
			num[p]=num[p*2]^num[p*2+1];
		tt=!tt;
	}
	return 0;
}

int main()
{
	int n,TC,v,p;
	cin >> n >>TC;
	int nn = 1<<n;
	for(int i=0;i<nn;i++)
	{
		cin >> v;
		put(v,nn+i);
	}
	while(TC--)
	{
		cin >> p >> v;
		put(v,nn+p-1);
		cout << num[1] << endl;
	}
	

	return 0;
}
