#include <cstdio>
int T[262200];
int n, m, tn = 1;
int i, j, k;
int modify(int p, int b){
    int i, dep = 0;
    T[p + tn] = b;
    for (i = (p + tn) >> 1; i; i >>= 1){
        if (dep & 1) T[i] = T[i << 1] ^ T[i << 1 | 1];
        else T[i] = T[i << 1] | T[i << 1 | 1];
        dep++;
    }
    return T[1];
}
int main(){
    scanf("%d %d\n", &n, &m);
    while (n--) tn <<= 1; tn--;
    for (i = 1; i <= tn + 1; i++){
        scanf("%d", &j);
        modify(i, j);
    }
    while (m--){
        scanf("%d %d\n", &j, &k);
        printf("%d\n", modify(j, k));
    }
    return 0;
}
