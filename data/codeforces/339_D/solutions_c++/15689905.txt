#include <stdio.h>
int n,m,v[18][1<<17];
main(){
  scanf("%d%d",&n,&m);
  n=1<<n;
  for(int i=0;i<n;++i)scanf("%d",&v[0][i]);
  for(int i=1,w=1;i<18;++i,w<<=1)for(int j=0,*t=v[i-1];j<n;j+=w+w)v[i][j]=(i&1)?(t[j]|t[j+w]):(t[j]^t[j+w]);
  for(int i=0,p,b;i<m&&2==scanf("%d%d",&p,&b);++i){
    v[0][--p]=b;
    for(int i=1,w=1,*t;i<18;++i,w<<=1)p^=p&w,t=v[i-1],v[i][p]=(i&1)?(t[p]|t[p+w]):(t[p]^t[p+w]);
    printf("%d\n",v[17][0]);
  }
}
