#include<iostream>
#include<vector>
#include<cmath>
using namespace std;

int n, m;
int tree[400000];
void solve(int p, int a) {
    int t=1;
    tree[p] = a;
    while(p>1) {
	p/=2;
	if(t==1) 
	    tree[p] = tree[p*2] | tree[p*2+1];
	else
	    tree[p] = tree[p*2] ^ tree[p*2+1];

	//cout << p << "->" << tree[p] << endl;
	t=t^1;
    }
}

int main()
{
    cin >> n >> m;
    int a, b;
    n = 1<<n;  //(2,n)

    for(int i = 1; i <= n; i++) {
	cin >> a;
	solve(i+n-1, a);
	//cout << tree[a] << " ";
    } //cout << endl;

    while(m--) {
	cin >> b >> a;
	solve(b+n-1, a);
	cout << tree[1] << endl;
    }

    return 0;
}
