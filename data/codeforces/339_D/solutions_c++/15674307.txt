#include<cstdio>
#include<algorithm>
#include<iostream>
using namespace std;

const int K = 18;

int n, m, a[1<<K];
int ai, p, b;

void Insert(int Pos, int y)
{
	Pos += 1 << (K-1);
	a[Pos] = y;
	bool orr = 1;
	while(true)
	{
		Pos /= 2;
		if(Pos == 0)
			break;
		if(orr) a[Pos] = a[2*Pos] | a[2*Pos+1];
		else
			a[Pos] = a[2*Pos] ^ a[2*Pos+1];
		orr = !orr;
	}
}


int main()
{
	scanf("%d %d", &n, &m);

	for(int i = 0; i < (1<<n); i++)
	{
		cin >> ai;
		Insert(i, ai);
	}
	while(m--)
	{
		scanf("%d %d", &p, &b);
		Insert(p-1, b);
		printf("%d\n", a[1]);
	}

	return 0;
}
