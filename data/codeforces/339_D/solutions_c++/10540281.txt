#include<bits/stdc++.h>
using namespace std;
int a[(1<<18)];//O(2^18)=O(262144) memory
void insert(int pos,int y){
  pos+=1<<17;
  a[pos]=y;//leaf values
  bool t=1;
  while(true){
    pos/=2;
    if(pos==0) break;
    if(t) a[pos]=a[2*pos] | a[2*pos+1];//first or
    else
      a[pos]=a[2*pos] ^ a[2*pos+1];//then xor
    t=!t;
  }
}
int main()
{
  cin.tie(NULL);
  ios_base::sync_with_stdio(0);
  int n,m,ai,p,b; cin>>n>>m;
  for(int i=0;i<(1<<n);i++){//2^n times
    cin>>ai;
    insert(i,ai);
  }
  while(m--){
    cin>>p>>b;
    insert(p-1,b);
    cout<<a[1]<<"\n";
  }
  return 0;
}
