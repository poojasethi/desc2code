#include <cstdio>

using namespace std;

int i, j, m, n, f[1 << 18];

void modify(int x, int p){
  if(p > 1)f[x] = p - 2, p = 1;
  else
    if(p & 1)f[x] = f[x << 1 | 1] ^ f[x << 1];
    else f[x] = f[x << 1 | 1] | f[x << 1];
  if(x > 1)modify(x >> 1, 1 - p);
}

int main(){
  scanf("%d %d", &n, &m);
  n = 1 << n;
  for(i = 0; i < n; i++){
    int c;
    scanf("%d", &c);
    modify(n + i, c + 2);
  }
  while(m--){
    int pos, b;
    scanf("%d %d", &pos, &b);
    modify(n + pos - 1, b + 2);
    printf("%d\n", f[1]);
  }
  return 0;
}
