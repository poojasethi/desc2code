#include<stdio.h>
#include<algorithm>
using namespace std;
int c[1000000];
int segtree[1000000];
int update(int a,int b){
	a+=131072;
	segtree[a]=b;
	a/=2;
	int count=0;
	while(a){
		
		if(count%2)segtree[a]=segtree[a*2]^segtree[a*2+1];
		else segtree[a]=segtree[a*2]|segtree[a*2+1];
		a/=2;
		count++;
	}
}
int main(){
	int a,b;
	scanf("%d%d",&a,&b);
	for(int i=0;i<(1<<a);i++){
		scanf("%d",c+i);
		update(i,c[i]);
	}
	for(int i=0;i<b;i++){
		int d,e;
		scanf("%d%d",&d,&e);
		update(d-1,e);
		printf("%d\n",segtree[1]);
	}
}
