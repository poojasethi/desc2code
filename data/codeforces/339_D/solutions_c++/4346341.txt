#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<cmath>
#include<ctime>
using namespace std;
int a,b,n,m;
int l[301000];
int wei(int x,int y){
    int t=1;
    l[x]=y;
    while(x>1){
        x=x/2;
        if(t==1)l[x]=l[x*2]|l[x*2+1];
        else l[x]=l[x*2]^l[x*2+1];
        t=t^1;
    }
}
int main(){
    scanf("%d%d",&n,&m);
    n=1<<n;
    for(int i=1;i<=n;i++){
        scanf("%d",&a);
        wei(i+n-1,a);
    }
    for(int i=1;i<=m;i++){
        scanf("%d%d",&a,&b);
        wei(a+n-1,b);
        printf("%d\n",l[1]);
    }
}
