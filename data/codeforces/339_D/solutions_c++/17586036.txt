#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int n,m,a[262150]={0},b[262150]={0},i,j;
int main()
{
scanf("%d%d",&n,&m);
for(i=0;i<(1<<n);i++) scanf("%d",&a[(1<<n)+i]),b[(1<<n)+i]=1;
for(i=(1<<n)-1;i>=1;i--)
{
if(b[i*2]) a[i]=(a[i*2]|a[i*2+1]);
else a[i]=(a[i*2]^a[i*2+1]),b[i]=1;
}
for(i=1;i<=m;i++)
{
scanf("%d",&j);
j=(1<<n)+j-1;
scanf("%d",&a[j]);
while(j>=1)
{
j/=2;
if(b[j*2]) a[j]=(a[j*2]|a[j*2+1]);
else a[j]=(a[j*2]^a[j*2+1]);
}
printf("%d\n",a[1]);;
}
return 0;
}
