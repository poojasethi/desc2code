#include <iostream>
using namespace std;

int n, m, arr[(1<<18)+5], pow, x, v;

void upd(int p, int dep){
    if(dep%2) arr[p] = arr[2*p]|arr[2*p+1];
    else arr[p] = arr[2*p]^arr[2*p+1];
    if(p>1) upd(p/2, dep+1);
}

int main(){
    ios::sync_with_stdio(false);
    cin >> n >> m;
    pow = 1<<n;
    for(int i = 0; i<pow; i++) cin >> arr[pow+i];
    for(int i = 0; i<pow/2; i++) upd(pow/2+i, 1);
    for(int i = 0; i<m; i++){
        cin >> x >> v;
        arr[pow+x-1] = v;
        upd((pow+x-1)/2, 1);
        cout << arr[1] << '\n';
    }
}
