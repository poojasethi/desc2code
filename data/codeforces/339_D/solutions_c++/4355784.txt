#include <cstdio>
const int N = 1<<18;
int n, m, a[N];
int main()
{
    scanf("%d%d", &n, &m);
    for(int i = 1<<n; i < 1<<(n+1); i++)
        scanf("%d", a+i);
    for(int i = n-1; i >= 0; i--)
        for(int j = 1<<i; j < 1<<(i+1); j++)
            a[j] = n-i & 1 ? a[j<<1] | a[j<<1|1] : a[j<<1] ^ a[j<<1|1];
    for(int p, b, i = 0; i<m; i++)
    {
        scanf("%d%d", &p, &b);
        p += (1<<n)-1;
        for(int x = 1; p > 1; x ^= 1, p >>= 1)
        {
            a[p] = b;
            b = x ? a[p^1] | b : a[p^1] ^ b;
        }
        a[p] = b;
        printf("%d\n", b);
    }
}
