#include<iostream>
#define MAX 262145
using namespace std;
int a[MAX];
int pw(int x){return 1<<(x);}
bool state=0;
int main()
{
	int i,n,p,v,s,e,m;
	cin>>n>>m;
	s=pw(n);
	e=pw(n+1);
	for(i=s;i<e;i++)cin>>a[i];
	for(i=n,state=0;i>0;i--,state=1-state)
		for(int j=(pw(i))-1;j>=(pw(i-1));j--)
			if(state)a[j]=a[j*2]^a[2*j+1];
			else a[j]=a[2*j]|a[2*j+1];
	while(m--){
		cin>>p>>v;
		a[s+p-1]=v;
		for(i=(s+p-1)/2,state=0;i!=0;i/=2,state=1-state)
			if(state)a[i]=a[i*2]^a[2*i+1];
			else a[i]=a[2*i]|a[2*i+1];
		cout<<a[1]<<endl;
	}
}
