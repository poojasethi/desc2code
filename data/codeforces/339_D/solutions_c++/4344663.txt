#include<stdio.h>
int a[18][131072];
int main(){
	int i,j,k,h,p,b;
	int n,m,q,nowm;
	scanf("%d%d",&n,&q);
	m=1<<n;
	for(i=0;i<m;i++){
		scanf("%d",&a[0][i]);
	}
	for(h=1;h<=n;h++){
		nowm=1<<(n-h);
		if(h%2)
			for(i=0;i<nowm;i++)
				a[h][i]=a[h-1][i+i]|a[h-1][i+i+1];
		else
			for(i=0;i<nowm;i++)
				a[h][i]=a[h-1][i+i]^a[h-1][i+i+1];
	}
	while(q--){
		scanf("%d%d",&p,&b);
		p--;
		a[0][p]=b;
		for(h=1;h<=n;h++){
			p/=2;
			if(h%2)
				a[h][p]=a[h-1][p+p]|a[h-1][p+p+1];
			else
				a[h][p]=a[h-1][p+p]^a[h-1][p+p+1];
		}
		printf("%d\n",a[n][0]);
	}
}
