#include <cstdio>

int N, n, m, a[1 << 18];

int main() {
  scanf("%d%d", &n, &m);
  N = 1 << n;
  for (int i = 0; i < N; ++i)
    scanf("%d", a + N + i);
  for (int i = N - 1, f = 1; i >= 1; f ^= 1)
    for (int j = i + 1 >> 1; i >= j; --i)
      a[i] = f ? a[i << 1] | a[i << 1 | 1] : a[i << 1] ^ a[i << 1 | 1];
  for (int i = 0; i < m; ++i) {
    int p, b;
    scanf("%d%d", &p, &b);
    a[p - 1 + N] = b;
    for (int j = p - 1 + N >> 1, f = 1; j >= 1; j >>= 1, f ^= 1)
      a[j] = f ? a[j << 1] | a[j << 1 | 1] : a[j << 1] ^ a[j << 1 | 1];
    printf("%d\n", a[1]);
  }
}
