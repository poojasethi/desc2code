#include <cstdio>
#include <cstring>

char S[10][10];
int tmp;

int main()
{
	for (int i=0;i<4;i++)
		gets(S[i]);
	bool ans=0;
	for (int i=0;i<3;i++)
		for (int j=0;j<3;j++)
		{
			tmp=0;
			for (int k1=0;k1<=1;k1++)
				for (int k2=0;k2<=1;k2++)
					if (S[i+k1][j+k2]=='.') tmp++;
			if (tmp==4||tmp==0||tmp==1||tmp==3) ans=1;
		}
	if (ans) puts("YES");
	else puts("NO");
}
