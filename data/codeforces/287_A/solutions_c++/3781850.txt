#include <iostream>
using namespace std;
int main()
{string s; int a[4][4];
for (int i=0;i<4;i++)
{cin>>s;
for (int j=0;j<4;j++)
if (s[j]=='.') a[i][j]=1; else a[i][j]=0;
}
for (int i=0;i<=2;i++)
for (int j=0;j<=2;j++)
{
int k=a[i][j]+a[i][j+1]+a[i+1][j]+a[i+1][j+1];
if (k<=1||k>=3)   {cout<<"YES"; return 0;}
}
cout<<"NO";
return 0;
}
