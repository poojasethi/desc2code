# include <stdio.h>
# include <algorithm>

using namespace std;

int z;

char a[9][9];

int main()
{
	for(int i=0;i<4;i++)
	scanf("%s",a[i]);
	
	for(int i=1;i<4;i++)
	for(int j=1;j<4;j++)
	{
		z=0;
		if(a[i-1][j]=='#')
		z++;
		if(a[i-1][j-1]=='#')
		z++;
		if(a[i][j-1]=='#')
		z++;
		if(a[i][j]=='#')
		z++;
		
		if(z>=3 || z<=1)
		{printf("YES");return 0;}
	}
	
	printf("NO");
}
