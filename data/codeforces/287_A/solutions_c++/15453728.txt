#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
main(){
char a[4][4];
int t=0;
for(int i=0;i<4;i++){
	cin>>a[i];
}
for(int i=0;i<3;i++){
	for(int j=0;j<3;j++){
		if(a[i][j]==a[i+1][j])t++;
		if(a[i][j]==a[i][j+1])t++;
		if(a[i][j]==a[i+1][j+1])t++;
		if(t>=2||t==0){
			printf("YES");
			exit(0);
		}
		else t=0;
	}
}
	
	printf("NO");
	
}
