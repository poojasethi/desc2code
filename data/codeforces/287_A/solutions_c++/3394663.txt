#include <iostream>
using namespace std;

int main(){
int a[4][4],i,j,z;
char q;
for (i=0; i<4; i++){
for (j=0; j<4; j++){
cin>>q;
if (q=='#') a[i][j]=1; else a[i][j]=0;
}
}
for (i=0; i<3; i++){
for (j=0; j<3; j++){
z=a[i][j]+a[i+1][j]+a[i][j+1]+a[i+1][j+1];
if (z>=3 ||z<=1) {cout<<"YES"; return 0;}
}
}
cout<<"NO";
return 0;
}
