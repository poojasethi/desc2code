#include<cstdio>

char map[5][5];

int main() {
	int cnt,i,j;
	for(i=1;i<=4;i++)
		scanf("%s",map[i]+1);

	for(i=1;i<=3;i++) {
		for(j=1;j<=3;j++) {
			cnt=0;
			cnt+=(map[i][j]=='.');
			cnt+=(map[i][j+1]=='.');
			cnt+=(map[i+1][j]=='.');
			cnt+=(map[i+1][j+1]=='.');
			if(cnt!=2) {
				printf("YES\n");
				return 0;
			}
		}
	}
	printf("NO\n");
	return 0;
}
