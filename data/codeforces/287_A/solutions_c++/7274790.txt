#include <cstdio>
#define REP(x,n) for(int x=0; x<(n); ++x)
using namespace std;

const int N = 4;

char s[N][N+1];


int main()
{
	REP(i, N) scanf("%s", s[i]);
	REP(i, N) REP(j, N) s[i][j] = s[i][j]=='#' ? 1 : 0;
	
	REP(i, N-1) REP(j, N-1) {
		int sum = s[i][j] + s[i+1][j] + s[i][j+1] + s[i+1][j+1];
		if (sum != 2) {
			puts("YES");
			return 0;
		}
	}
	
	puts("NO");
	return 0;
}
