#include<stdio.h>
#include<string.h>
char arr[10][10];
int yon[10][3] = {{0,1},{1,0},{1,1}},i,j,k,s;
int main(){
	scanf("%s %s %s %s",arr[0],arr[1],arr[2],arr[3]);

	for(i=0;i<3;i++)
		for(j=0;j<3;j++){
			s=0;
			for(k=0;k<3;k++)
				if(arr[i][j] != arr[i + yon[k][0]][j + yon[k][1]])
					s++;

			if(!s || s == 1 || s== 3){
				puts("YES");
				return 0;
			}
		}
	puts("NO");
	return 0;
}
