#include<iostream>
using namespace std;
int main()
{
char sq[4][5];
int c[50]={0},i,j;
for(i=0; i<4; i++)
cin>>sq[i];
for(i=0; i<3; i++)
for(j=0; j<3; j++)
{
c[46]=0;
c[35]=0;
c[sq[i][j]-0]++;
c[sq[i+1][j]-0]++;
c[sq[i][j+1]-0]++;
c[sq[i+1][j+1]-0]++;
if(c[46]>2 || c[35]>2)
{
cout<<"YES";
return 0;
}
} 
cout<<"NO";
return 0;
}
