#include<iostream>
using namespace std;
int main(){
    int b[4][4];
    char a;
    for(int i=0;i<4;i++)
    for(int j=0;j<4;j++){
        cin>>a;
        if(a=='.')b[i][j]=0;else b[i][j]=1;
    }
    for(int i=0;i<3;i++)
    for(int j=0;j<3;j++){
        int s=b[i][j]+b[i][j+1]+b[i+1][j]+b[i+1][j+1];
        if(s!=2){cout<<"YES";return 0;}
    }
    cout<<"NO";
    return 0;
}
