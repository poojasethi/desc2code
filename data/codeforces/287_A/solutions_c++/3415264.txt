#include <iostream>
using namespace std;
char a[10][10];
int k,i,j,d,l,o;
main()
{
for (i=1; i<=4; i++)
for (j=1; j<=4; j++)
cin>>a[i][j];

for (i=1; i<=3; i++)
for (j=1; j<=3; j++)
{
d=0;
for (k=i; k<=i+1; k++)
for (o=j; o<=j+1; o++)
if (a[k][o]=='#') d++;
if (d!=2) l=1;
}
if (l==1) cout<<"YES"; else cout<<"NO";

}
