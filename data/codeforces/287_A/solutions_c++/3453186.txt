#include <iostream>
#include <cstdio>
#define C '#'
using namespace std;
char a[5][5];
int main(){
 	for(int i=0;i<4;i++) cin>>a[i];
 	for(int i=0;i<3;i++)
 		for(int j=0;j<3;j++) {
 		 	int x=(a[i][j]==C)+(a[i+1][j]==C)+(a[i][j+1]==C)+(a[i+1][j+1]==C);
 		 	if(x!=2) {
 		 		cout<<"YES";
 		 		return 0;
 		 	}
 		}
 	cout<<"NO";
 	return 0;
}
