#include<iostream>
using namespace std;
int main()
{
int a[5][5],i,j;
char c;
  for(i=1;i<=4;i++)
  {
  for(j=1;j<=4;j++)
  {
     cin>>c;
   if(c=='#')
   a[i][j]=1;
   else
   a[i][j]=0;
  }
   }
for(i=1;i<4;i++)
{
for(j=1;j<4;j++)
{
    if(a[i][j]+a[i+1][j]+a[i][j+1]+a[i+1][j+1]!=2)
{
cout<<"YES";
return 0;
}
}
}
cout<<"NO";
return 0;
}
