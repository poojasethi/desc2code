#include<iostream>
using namespace std;
char c[4][4];
short s[4][4];
int main(){
	for(int i=0;i<4;i++)cin>>c[i];
	for(int i=0;i<4;i++)for(int j=0;j<4;j++)if(c[i][j]=='#')s[i][j]=1;else s[i][j]=0;
	for(int i=0;i<3;i++)for(int j=0;j<3;j++)if(s[i][j]+s[i][j+1]+s[i+1][j]+s[i+1][j+1]==2)continue;else {cout<<"YES";return 0;}
	cout<<"NO";
	return 0;
}
