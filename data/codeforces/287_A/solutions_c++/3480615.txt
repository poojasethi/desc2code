#include<iostream>
using namespace std;

int a[5][5];

int main()
{
	char c;
	for(int i=0; i<4; i++)
	{
		for(int j=0; j<4; j++)
		{
			cin>>c;
			if(c=='#')
			{
				a[i][j]=1;
			}
		}
	}
	for(int i=0; i<3; i++)
	{
		for(int j=0; j<3; j++)
		{
			if(a[i][j]+a[i+1][j]+a[i][j+1]+a[i+1][j+1]!=2)
			{
				cout<<"YES";
				return 0;
			}
		}
	}
	cout<<"NO";
	return 0;
	}
