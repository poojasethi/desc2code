#include <bits/stdc++.h>
using namespace std;

int main() {
    string s; int a[4][4];
    for (int i=0;i<4;i++) {
        getline(cin,s);
        for (int j=0;j<4;j++) if (s[j]=='#') a[i][j]=1; else a[i][j]=0;
    }
    for (int i=0;i<3;i++)
        for (int j=0;j<3;j++)
            if (a[i][j]+a[i+1][j]+a[i][j+1]+a[i+1][j+1]!=2)
                return cout << "YES", 0;
    cout << "NO";
}
