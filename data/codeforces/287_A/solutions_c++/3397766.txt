#include<stdio.h>
int i,j,k,r;
char s[5][5];
int main()
{
	for(i=1;i<=4;i++)
	    scanf("%s",s[i]);
	r=0;
	for(i=1;i<=3;i++)
	{
		for(j=0;j<3;j++)
		{
			k=0;
			if(s[i][j]=='.')k++;
			if(s[i+1][j]=='.')k++;
			if(s[i][j+1]=='.')k++;
			if(s[i+1][j+1]=='.')k++;
			if(k!=2){r=1;break;}
		}
		if(r==1)break;
	}
	if(r==1)printf("YES\n");
	else printf("NO\n");		
	return 0;
}
