#include <iostream>
#include <cstdio>
using namespace std;
bool board[4][4];
char c;
int main(){
    for(int i=0;i<4;i++)for(int j=0;j<4;j++){
        cin>>c; board[i][j]=(c=='#')?1:0;
    }
    
    for(int i=0;i<3;i++)for(int j=0;j<3;j++){
        if ((board[i][j]+board[i][j+1]+board[i+1][j]+board[i+1][j+1])!=2)
        {puts("YES"); return 0;}
    }puts("NO"); return 0;
}
