#include <cstdio>
#include <iostream>

using namespace std;

int main(){
	char m[4][4];
	
	for(int i=0; i<4; i++)
		cin >> m[i];
	
	bool ans = false;
	for(int x=0; x<3; x++)
		for(int y=0; y<3; y++){
			int count = (m[y][x]=='#') + (m[y][x+1]=='#') + (m[y+1][x]=='#' ) + (m[y+1][x+1]=='#');
			if( count != 2 ){
				ans = true;
				break;
			}
		}

	cout << (ans?"YES":"NO") << endl;

	return 0;
}
