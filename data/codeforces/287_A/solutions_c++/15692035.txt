#include<bits/stdc++.h>
using namespace std;
int main(){
    char ara[4][5];
    for(int i=0;i<4;i++) scanf("%s",ara[i]);

    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            int w=(ara[i][j]=='.')+(ara[i+1][j]=='.')+(ara[i][j+1]=='.')+(ara[i+1][j+1]=='.');
            if(w!=2) { printf("YES"); return 0; }
        }
    }
    printf("NO");
}
