#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j;
	bool a[5][5],flag=0;
	char c;
	for(i=0;i<4;i++)
	{
		for(j=0;j<4;j++)
		{
			cin>>c;
			(c=='#') ? a[i][j]=1 : a[i][j]=0;
		}
		getchar();
	}
	for(i=0;i<3;i++)
	for(j=0;j<3;j++)
	{
		if(a[i][j]+a[i][j+1]+a[i+1][j]+a[i+1][j+1]!=2)
		{
			flag=1;
			break;
		}
	}
	if(flag)
		cout<<"YES\n";
	else
		cout<<"NO\n";
	return 0;
}
