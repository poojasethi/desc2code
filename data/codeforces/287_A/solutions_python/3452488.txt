a = []

for i in range(4):
    s = raw_input()
    a += [s]

for i in range(3):
    for j in range(3):
        c = 0
        for di in range(2):
            for dj in range(2):
                if a[i + di][j + dj] == '#':
                    c += 1
        if c != 2:
            print "YES"
            break
    else:
        continue
    break
else:
    print "NO"
