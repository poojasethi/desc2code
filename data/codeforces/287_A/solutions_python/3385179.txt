import sys
s = []
for i in xrange(4):
    s.append(raw_input())
for i in xrange(1, 4):
    for j in xrange(1, 4):
        freq = {"#":0, ".":0}
        freq[s[i][j-1]] += 1
        freq[s[i-1][j-1]] += 1
        freq[s[i-1][j]] += 1
        freq[s[i][j]] += 1
        if freq["#"] >= 3 or freq["."] >= 3:
            print "YES"
            sys.exit(0)
