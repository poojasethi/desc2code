chas = ['' for i in range(4)]

def isOK(a, b, c, d):
	mn = [a, b, c, d]
	if mn.count(a) == 3 or mn.count(a) ==1 or mn.count(a) == 4:
		return True
	else:
		return False
	
for i in range(4):
	chas[i] = raw_input()
	
out = 'NO'
for i in range(3):
	for j in range(3):
		if isOK(chas[i][j], chas[i][j+1], chas[i+1][j], chas[i+1][j+1]):
			out = 'YES'
			break
print out
