
def main():
    s = [raw_input() for _ in xrange(4)]
    for i in xrange(3):
        for j in xrange(3):
            cnt = 0
            for ii in xrange(2):
                for jj in xrange(2):
                    if s[i+ii][j+jj]=='.':
                        cnt += 1
            if cnt != 2:
                print 'YES'
                return
    print 'NO'

if __name__ == '__main__':
    main()
