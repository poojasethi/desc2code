def solve(lis):
    points=[(1,0),(1,1),(0,1)]
    sol=[["#","#","#","."],["#",".",".","."],["#"]*4,["."]*4]
    for i,x in enumerate(lis[:-1]):
        for j,y in enumerate(x[:-1]):
            if sorted([y]+[lis[q+i][w+j] for q,w in points]) in sol:
                      return "YES"
    return "NO"
print solve([[x for x in raw_input()]for _ in xrange(4)])
