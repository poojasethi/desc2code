s = []
for i in xrange(4):
  s.append(raw_input())

for i in xrange(3):
  for j in xrange(3):
    w, b = 0, 0
    for ii in xrange(2):
      for jj in xrange(2):
        if s[i + ii][j + jj] == '#':
          w += 1
        else:
          b += 1
      if w >= 3 or b >= 3:
        print 'YES'
        exit(0)
print 'NO'
