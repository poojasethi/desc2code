import sys

grid = [list(sys.stdin.readline().strip()) for x in xrange(4)]

for i in xrange(3):
    for j in xrange(3):
        v = 0
        for x, y in [(0, 0), (0, 1), (1, 1), (1, 0)]:
            v += grid[i + x][j + y] == '#' and 1 or 0
        if v != 2:
            print 'YES'
            exit(0)
print 'NO'
