board = []
for i in xrange(4): board.append(raw_input())
for i in xrange(3):
	for j in xrange(3):
		cnt = 0
		if '.' == board[i][j]: cnt += 1
		if '.' == board[i][j + 1]: cnt += 1
		if '.' == board[i + 1][j]: cnt += 1
		if '.' == board[i + 1][j + 1]: cnt += 1
		if cnt != 2: 
			print "YES"
			exit()
print "NO"
