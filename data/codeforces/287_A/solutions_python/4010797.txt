from collections import defaultdict
#br = open('a.in')
a = [raw_input().strip() for i in range(4)]
for i in range(0, 3):
  for j in range(0, 3):
    di = defaultdict(int)
    for k in range(i, i + 2):
      for l in range(j, j + 2):
        di[a[k][l]] += 1
    for k in di:
      if di[k] == 3 or di[k] == 4:
        print "YES"
        exit()
print "NO"
