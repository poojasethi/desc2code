S = 4
B = 2

if __name__ == '__main__':
	board = []
	for i in range(S):
		board.append(raw_input())
	ok = False
	for i in range(S - B + 1):
		for j in range(S - B + 1):
			num = [0] * 2
			for r in range(i, i + B):
				for c in range(j, j + B):
					num[0 if board[r][c] == '.' else 1] += 1
			if max(num[0], num[1]) >= 3:
				ok = True
	print("YES" if ok else "NO")
