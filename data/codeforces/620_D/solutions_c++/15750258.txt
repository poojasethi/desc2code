#include<iostream>
#include<cstdio>
#include<cstring>
#include<iostream>
#include<set>
using namespace std;
const long long inf=(1ll<<61);
int n,m;
int a[2011],b[2011];
long long suma=0ll,sumb=0ll,sb;
long long labs(long long a)
{
	if (a<0) return -a;
	else return a;
}
long long v[3];
int x11,y11,x21,y21,x22,y22;
set<long long> ms;
set<long long>::iterator iter;
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		suma+=a[i];
	}
	scanf("%d",&m);
	for (int i=1;i<=m;i++)
	{
		scanf("%d",&b[i]);
		sumb+=b[i];
	}
	v[0]=labs(suma-sumb);
	v[1]=v[2]=inf;
	for (int i=1;i<=n;i++)
	{
		for (int j=1;j<=m;j++)
		{
			if (v[1]>labs(suma-sumb-2*a[i]+2*b[j]))
			{
				x11=i;y11=j;
				v[1]=labs(suma-sumb-2*a[i]+2*b[j]);
			}
		}
	}
	ms.clear();
	for (int i=1;i<=m;i++)
	{
		for (int j=i+1;j<=m;j++)
		{
			ms.insert(2ll*(b[i]+(long long)b[j]));
		}
	}
	long long ta,tb,tem;
	int index;
	for (int i=1;i<=n;i++)
	{
		for (int j=i+1;j<=n;j++)
		{
			ta=a[i]+a[j];
			tem=suma-sumb-2*ta;
			tem=-tem;
			iter=ms.lower_bound(tem);
			if (iter!=ms.end())
			{
				ta=labs(tem-(*iter));
				if (ta<v[2])
				{
					v[2]=ta;
					x21=i;x22=j;sb=*iter;
				}
			}
			if (iter!=ms.begin())
			{
				iter--;
				ta=labs(tem-(*iter));
				if (ta<v[2])
				{
					v[2]=ta;
					x21=i;x22=j;sb=*iter;
				}
			}
		}
	}
	for (int i=1;i<=m;i++)
	{
		for (int j=i+1;j<=m;j++)
		{
			if (2ll*(b[i]+(long long)b[j])==sb)
			{
				y21=i;y22=j;
			}
		}
	}
	if (v[0]<=v[1]&&v[0]<=v[2])
	{
		printf("%I64d\n0",labs(suma-sumb));
		return 0;
	}
	if (v[1]<=v[2])
	{
		printf("%I64d\n1\n",v[1]);
		printf("%d %d",x11,y11);
		return 0;
	}
	else
	{
		printf("%I64d\n2\n",v[2]);
		printf("%d %d\n%d %d",x21,y21,x22,y22);
	}
}
