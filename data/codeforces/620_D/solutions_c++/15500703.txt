#define ll long long
#define mod 1000000007LL
#define F(a,b,c) for(a=b;a<c;a++)
#define Fr(a,b,c) for(a=b;a>=c;a--)
#define pf printf
#define sfd(a) scanf("%d",&a)
#define sfdd(a,b) scanf("%d%d",&a,&b)
#define sfl(a) scanf("%lld",&a)
#define sfll(a,b) scanf("%lld%lld",&a,&b)
#define pfd(a) printf("%d",a)
#define pfl(a) printf("%lld",a)
#define sf scanf
#define line printf("\n")
#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp(a,b) make_pair(a,b)
#define let(x,a) __typeof(a) x(a)
#define forall(it,v) for(it=v.begin();it!=v.end();it++)
pair<ll,pair<int,int> > a2[1<<21],b2[1<<21];
int a[1<<12],b[1<<12];
int main(){
int n,i,j,m2=0,n2=0,p1,p2,f=0,mi,m;
ll sa=0,sb=0,sa2=0,sb2=0;
cin>>n;
F(i,0,n)
	sfd(a[i]),sa=sa+a[i];
cin>>m;
F(i,0,m)
	sfd(b[i]),sb=sb+b[i];
F(i,0,n)
	F(j,i+1,n)
		a2[n2++]=mp(a[i]+a[j],mp(i,j)),sa2=sa2+a[i]+a[j];
F(i,0,m)
	F(j,i+1,m)
		b2[m2++]=mp(b[i]+b[j],mp(i,j)),sb2=sb2+b[i]+b[j];
sort(a2,a2+n2);
sort(b2,b2+m2);	
if(sa==sb)
	return pfd(0),line,pfd(0),0;
ll res=abs(sa-sb);
F(i,0,n)
	F(j,0,m)
		if(res>abs(sa-sb-2*a[i]+2*b[j]))
			res=abs(sa-sb-2*a[i]+2*b[j]),f=1,p1=i,p2=j;
i=0,j=0;
while(i<n2&&j<m2){
	if(res>abs(sa-sb-2*a2[i].first+2*b2[j].first))
		{
			res=abs(sa-sb-2*a2[i].first+2*b2[j].first);
			f=2;
			p1=i,p2=j;
		}
	if(i<n2-1&&(j==m2-1||(abs(sa-sb-2*a2[i+1].first+2*b2[j].first)<abs(sa-sb-2*a2[i].first+2*b2[j+1].first)))) i++;
	else j++;
}
if(!f)
	pfl(res),line,pfd(0);
else if(f==1){
	pfl(res),line,pfd(1),line,pf("%d %d",p1+1,p2+1);
}
else {
	pfl(res),line,pfd(2),line;
	pf("%d %d\n",a2[p1].second.first+1,b2[p2].second.first+1);
	pf("%d %d",a2[p1].second.second+1,b2[p2].second.second+1);
}
return 0;
}
