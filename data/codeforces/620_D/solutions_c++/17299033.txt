#include<bits/stdc++.h>

using namespace std;

#define mp make_pair
#define nd second
#define st first
#define N 2003

pair < int , pair < int , int > > A[N*N],B[N*N];

pair < int , int > a[N],b[N];

int m,n,r,s,x,y,i,j,asd;

long long t1,t2,mn;

bool h;

int main(){

	cin >> m;

	for(i=1;i<=m;i++){
		cin >> a[i].st;
		t1 += a[i].st;
		a[i].nd = i;
	}

	cin >> n;

	for(i=1;i<=n;i++){
		cin >> b[i].st;
		t2 += b[i].st;
		b[i].nd = i;
	}

	for(i=1;i<=m;i++)
		for(j=i+1;j<=m;j++)
			A[++r] = mp( a[i].st + a[j].st , mp(i,j) );

	for(i=1;i<=n;i++)
		for(j=i+1;j<=n;j++)
			B[++s] = mp( b[i].st + b[j].st , mp(i,j) );

	sort(a+1 , a+m+1);
	sort(b+1 , b+n+1);
	sort(A+1 , A+r+1);
	sort(B+1 , B+s+1);

	if(t1 < t2){ h = 1;
		swap(t1 , t2);
		swap(m , n);
		swap(r , s);
		swap(a , b);
		swap(A , B);
	}

	mn = t1 - t2;

	for(i=j=1;i<=m;i++){

		while(abs(t1 - a[i].st + b[j].st - (t2 - b[j].st + a[i].st)) > abs(t1 - a[i].st + b[j+1].st - (t2 - b[j+1].st + a[i].st)) && j < n)
			j++;

		if( mn > abs(t1 - a[i].st + b[j].st - (t2 - b[j].st + a[i].st)) ){
			mn = abs(t1 - a[i].st + b[j].st - (t2 - b[j].st + a[i].st));
			asd = 1;  x = i;  y = j;
		}
	}

	for(i=j=1; i <= r && s ;i++){

		while(abs(t1 - A[i].st + B[j].st - (t2 - B[j].st + A[i].st)) > abs(t1 - A[i].st + B[j+1].st - (t2 - B[j+1].st + A[i].st)) && j < s)
			j++;

		if( mn > abs((t1 - A[i].st + B[j].st) - (t2 - B[j].st + A[i].st)) ){
			mn = abs((t1 - A[i].st + B[j].st) - (t2 - B[j].st + A[i].st));
			asd = 2;  x = i;  y = j;
		}
	}

	cout << mn << endl << asd << endl;

	if(asd == 1){
		if(!h)
			cout << a[x].nd << " " << b[y].nd;
		else
			cout << b[y].nd << " " << a[x].nd;
	}

	else if(asd){
		if(!h){
			cout << A[x].nd.st << " " << B[y].nd.st << endl;
			cout << A[x].nd.nd << " " << B[y].nd.nd;
		}
		else{
			cout << B[y].nd.st << " " << A[x].nd.st << endl;
  			cout << B[y].nd.nd << " " << A[x].nd.nd;
		}
	}

	return 0;
}
