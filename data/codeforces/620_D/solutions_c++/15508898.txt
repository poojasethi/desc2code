#include <bits/stdc++.h>
#define REP(i,a,b) for(i=a;i<=b;i++)
#define x first
#define y second
using namespace std;
typedef long long ll;
typedef pair<ll,ll> ii;
map<ll,ii> mp;
map<ll,ii>::iterator it;
vector<ii> ar;
ll a[2010],b[2010];
int main(){
	ios::sync_with_stdio(false);
	ll i,j,k,n,m,sa=0,sb=0,ans,mm,v;
	ii p;
	cin>>n;
	REP(i,0,n-1){
		cin>>a[i];
		sa+=a[i];
	}
	cin>>m;
	REP(i,0,m-1){
		cin>>b[i];
		sb+=b[i];
	}
	ans=abs(sa-sb);
	//cout<<sa<<endl;
	REP(i,0,n-1)
		REP(j,0,m-1){
			mm=abs(sa-sb+2*b[j]-2*a[i]);
			if(mm<ans){
				ar.clear();
				ans=mm;
				ar.push_back(ii(i,j));
			}
		}
	REP(i,0,n-1)
		REP(j,i+1,n-1)
			mp[2*(a[i]+a[j])]=ii(i,j);
	REP(i,0,m-1)
		REP(j,i+1,m-1){
			v=sa-sb+2*(b[i]+b[j]);
			it=mp.lower_bound(v);
			if(it!=mp.begin())
				it--;
			REP(k,0,1){
				if(it==mp.end())
					break;
				mm=abs(v-it->x);
				p=it->y;
				if(mm<ans){
					ar.clear();
					ans=mm;
					ar.push_back(ii(p.x,i));
					ar.push_back(ii(p.y,j));
				}
				it++;
			}
		}
	cout<<ans<<endl<<ar.size()<<endl;
	for(i=0;i<ar.size();i++)
		cout<<ar[i].x+1<<" "<<ar[i].y+1<<endl;
}
