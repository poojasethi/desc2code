#include<bits/stdc++.h>
#define mp make_pair
using namespace std;
typedef long long ll;

const int N = 2001;
int n, m, k, res[4], a[N], b[N];
ll sa, sb, diff, ans;
pair<int, pair<int, int> >  A[N*N/2], B[N*N/2];
int sza, szb;

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d", &n);
	for(int i=0 ; i<n ; ++i) scanf("%d", a+i);
	scanf("%d", &m);
	for(int j=0 ; j<m ; ++j) scanf("%d", b+j);
	for(int i=0 ; i<n ; ++i) sa += a[i];
	for(int i=0 ; i<m ; ++i) sb += b[i];
	diff = sb - sa;
	ans = abs(diff);
	for(int i=0 ; i<n ; ++i)
		for(int j=0 ; j<m ; ++j)  {
			if(abs(diff + 2*(a[i]-0LL-b[j])) < ans) {
				ans = abs(diff + 2*(a[i]-0LL-b[j]));
				k = 1;
				res[0] = i, res[1] = j;
			}
		}
	for(int i=0 ; i<n ; ++i)
		for(int j=i+1 ; j<n ; ++j)
			A[sza++] = mp(a[i]+a[j], mp(i, j));
	for(int i=0 ; i<m ; ++i)
		for(int j=i+1 ; j<m ; ++j)
			B[szb++] = mp(b[i]+b[j], mp(i, j));
	sort(A, A+sza);
	sort(B, B+szb);
	int j=0;
	for(int i=0 ; i<sza ; ++i) {
		while(j < szb && diff+2*(A[i].first - 0LL - B[j].first) > 0) ++j;
		if(j < szb) {
			if(abs(diff + 2 * (A[i].first - 0LL - B[j].first)) < ans) {
				ans = abs(diff + 2 * (A[i].first - 0LL - B[j].first));
				k = 2;
				res[0] = A[i].second.first;
				res[2] = A[i].second.second;
				res[1] = B[j].second.first;
				res[3] = B[j].second.second;
			}
		}
		if(j>=1) {
			--j;
			if(abs(diff + 2 * (A[i].first - 0LL - B[j].first)) < ans) {
				ans = abs(diff + 2 * (A[i].first - 0LL - B[j].first));
				k = 2;
				res[0] = A[i].second.first;
				res[2] = A[i].second.second;
				res[1] = B[j].second.first;
				res[3] = B[j].second.second;
			}
			++j;
		}
	}
	cout << ans << endl;
	if(k == 0) puts("0");
	else if(k == 1) {
		puts("1");
		printf("%d %d\n", res[0]+1, res[1]+1);
	} else {
		puts("2");
		printf("%d %d\n", res[0]+1, res[1]+1);
		printf("%d %d\n", res[2]+1, res[3]+1);
	}

	return 0;
}
