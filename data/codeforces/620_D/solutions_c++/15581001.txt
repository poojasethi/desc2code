#include <bits/stdc++.h>
using namespace std;
const int N = 2005;
#define ll long long
ll a[N],b[N];
#define pp pair<pair<int,int>,pair<int,int> >
#define x first
#define y second
ll sa=0,sb=0;
ll get(pp t){
	ll ret= sa-sb;
	if(t.x.x!=-1)
		ret= sa-sb-a[t.x.x]*2+b[t.x.y]*2;
	if(t.y.x!=-1)
		ret= sa-sb-a[t.x.x]*2+b[t.x.y]*2-a[t.y.x]*2+b[t.y.y]*2;
	ret= abs(ret);
	return ret;
}
void print(pp t){
	ll a=t.x.x,b=t.x.y,c=t.y.x,d=t.y.y;
	if(a!=-1 && c!=-1)cout<<2<<endl;
	else if (c==-1 && a!=-1)cout<<1<<endl;
	else cout<<0<<endl;
	if(a!=-1)cout<<a+1<<" "<<b+1<<endl;
	if(c!=-1)cout<<c+1<<" "<<d+1<<endl;
}
#define mp make_pair
pair<ll,pair<int,int> > aa[N*N],bb[N*N];
int main(){
	int n,m,i,j;
	cin>>n;for(i=0;i<n;++i)cin>>a[i],sa+=a[i];
	cin>>m;for(i=0;i<m;++i)cin>>b[i],sb+=b[i];

	pp ret= mp(mp(-1,-1),mp(-1,-1)),tt;
	
	for(i=0;i<n;++i)
		for(j=0;j<m;++j){
			tt= mp(mp(i,j),mp(-1,-1));
			if(get(ret)>get(tt))
				ret= tt;
		}
	int ap=0,bp=0;
	for(i=0;i<n;++i)
		for(j=i+1;j<n;++j)
			aa[ap++]=mp(sa-sb-(a[i]*2+a[j]*2),mp(i,j));

	for(i=0;i<m;++i)
		for(j=i+1;j<m;++j)
			bb[bp++]= mp(-(b[i]*2+b[j]*2),mp(i,j));
	sort(aa,aa+ap);
	sort(bb,bb+bp);
	if(ap!=0 && bp!=0){
		for(i=0;i<ap;++i){
			int x=lower_bound(bb,bb+bp,mp(aa[i].x,mp(0,0)) )-bb;
			for(j=x-1;j<=x+1;++j){
				if(j<0 || j>=bp)continue;
				tt=mp(mp(aa[i].y.x,bb[j].y.x),mp(aa[i].y.y,bb[j].y.y));
				if(get(ret)>get(tt) )
					ret= tt;
			}
			x = upper_bound(bb,bb+bp,mp(aa[i].x,mp(0,0)))-bb;
			for(j=x-1;j<=x+1;++j){
				if(j<0 || j>=bp)continue;
				tt= mp( mp(aa[i].y.x,bb[j].y.x),mp(aa[i].y.y,bb[j].y.y));
				if(get(ret)>get(tt))
					ret= tt;
			}
		}
	}
	cout<<get(ret)<<endl;
	print(ret);
}
