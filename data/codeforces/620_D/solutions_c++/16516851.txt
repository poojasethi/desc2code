#include <bits/stdc++.h>
using namespace std;
const int maxn = 2010;
int n,a[maxn],m,b[maxn],ida[5],idb[5],k,la,lb,p,zz;
pair<int,pair<int,int> > tmpb[maxn*maxn];
long long sa,sb,ans,z;
int main(){
    //freopen("vao.inp", "r", stdin);
    scanf("%d",&n);
    for (int i=1; i<=n; i++) scanf("%d",&a[i]),sa+=1LL*a[i];
    scanf("%d",&m);
    for (int i=1; i<=m; i++) scanf("%d",&b[i]),sb+=1LL*b[i];
    ans=abs(sa-sb);k=0;
    for (int i=1; i<=n; i++)
    for (int j=1; j<=m; j++){
        if (ans>abs(sa-sb+2LL*b[j]-2LL*a[i])){
            ans=abs(sa-sb+2LL*b[j]-2LL*a[i]);
            k=1;
            ida[1]=i;
            idb[1]=j;
        }
    }
    for (int i=1; i<=m; i++)
    for (int j=i+1; j<=m; j++){
        tmpb[++lb]=make_pair(b[i]+b[j],make_pair(i,j));
    }
    sort(tmpb+1,tmpb+lb+1);
    for (int i=1; i<=n; i++)
    for (int j=i+1; j<=n; j++){
        /// sa-sb +2*z -2*(a[i]+a[j]) == 0
        z=(2LL*(a[i]+a[j])-(sa-sb))/2LL;
        z=max(z,-2000000000LL);
        z=min(z,2000000000LL);

            zz=z;
            p=upper_bound(tmpb+1,tmpb+lb+1,make_pair(zz,make_pair(m+1,m+1)))-tmpb;
            for (int r=max(p-2,1); r<=min(lb,p+2); r++){
                if (ans>abs(sa-sb+2LL*tmpb[r].first-2LL*(a[i]+a[j]))){
                    ans=abs(sa-sb+2LL*tmpb[r].first-2LL*(a[i]+a[j]));
                    k=2;
                    ida[1]=i;ida[2]=j;
                    idb[1]=tmpb[r].second.first;
                    idb[2]=tmpb[r].second.second;
                }
            }

    }
    printf("%I64d\n",ans);
    printf("%d\n",k);
    for (int i=1; i<=k; i++) printf("%d %d\n",ida[i],idb[i]);
}
