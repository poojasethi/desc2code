#include<stdio.h>
#include<cmath>
#include<map>
#include<utility>
#define N 2005
#define ll long long
#define F(a,b,c) for(int a=b;a<=c;a++)
#define hh  if(abs(c-it->first)<v){\
        ans=2;\
        v=abs(c-it->first);\
        swap1=it->second;\
        swap2={i,j};}
using namespace std;
ll a[N],b[N],n,m,suma,sumb,v,ans,c;
map<ll,pair<int,int> >u;
map<ll,pair<int,int> >::iterator it;
pair<int,int>swap1,swap2;
int main()
{
    scanf("%lld",&n);
    F(i,1,n)scanf("%lld",&a[i]),suma+=a[i];
    scanf("%lld",&m);
    F(i,1,m)scanf("%lld",&b[i]),sumb+=b[i];
    v=abs(suma-sumb);
    if(!v){printf("0\n0\n");return 0;}
    F(i,1,n-1)F(j,i+1,n)u[(a[i]+a[j])*2LL]= {i,j};
    F(i,1,n)F(j,1,m)
    if(abs(suma-sumb-2LL*(a[i]-b[j]))<v)
    {
        ans=1;
        v=abs(suma-sumb-2LL*(a[i]-b[j]));
        swap1= {i,j};
    }
    F(i,1,m-1)F(j,i+1,m)
    {
        c=suma-sumb+2LL*(b[i]+b[j]);
        it=u.lower_bound(c);
        if(it!=u.end()){hh}
        if(it!=u.begin()){it--;hh}
    }
    printf("%lld\n%lld\n",v,ans);
    if(ans==1)printf("%d %d\n",swap1.first,swap1.second);
    if(ans==2)printf("%d %d\n%d %d\n",swap1.first,swap2.first,swap1.second,swap2.second);
    return 0;
}
