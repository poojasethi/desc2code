#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int N = 2005;
int n, m, a[N], b[N];
ll x, y, xx, yy, d, k;
int ans[4];
map<ll, pair<int, int> > mp;
map<ll, pair<int, int> >::iterator it, jt;

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif

	scanf("%d", &n);
	for(int i=0 ; i<n ; ++i) {
		scanf("%d", &a[i]);
		x += a[i];
	}

	scanf("%d", &m);
	for(int j=0 ; j<m ; ++j) {
		scanf("%d", &b[j]);
		y += b[j];
	}
	d = abs(x - y);
	for(int i=0 ; i<n ; ++i)
		for(int j=0 ; j<m ; ++j) {
			xx = x - a[i] + b[j];
			yy = y - b[j] + a[i];
			if(abs(xx-yy) < d) {
				d = abs(xx-yy);
				k = 1;
				ans[0] = i;
				ans[1] = j;
			}
		}
	for(int i=0 ; i<m ; ++i)
		for(int j=i+1 ; j<m ; ++j)
			mp[(ll)b[i]+b[j]] = make_pair(i, j);
	ll A;
	for(int i=0 ; i<n ; ++i)
		for(int j=i+1 ; j<n ; ++j) {
			A = (ll)a[i] + a[j];
			it = jt = mp.lower_bound((2*A-x+y)/2);
			for(int kk=0 ; kk<3 ; ++kk) {
				if(it != mp.end()){
					if(abs(x - y - 2*A + 2*(it->first)) < d) {
						d = abs(x - y - 2*A + 2*(it->first));
						k = 2;
						ans[0] = i;
						ans[1] = it->second.first;
						ans[2] = j;
						ans[3] = it->second.second;
					}
					++it;
				}
				if(jt != mp.begin()) {
					--jt;
					if(abs(x-y-2*A+2*(jt->first)) < d){
						d = abs(x - y - 2*A + 2*(jt->first));
						k = 2;
						ans[0] = i;
						ans[1] = jt->second.first;
						ans[2] = j;
						ans[3] = jt->second.second;
					}
				}
			}
		}
	cout<<d<<endl;
	if(k == 0) puts("0");
	else if(k == 1) {
		puts("1");
		printf("%d %d\n", ans[0]+1, ans[1]+1);
	} else {
		puts("2");
		for(int i=0 ; i<4 ; i+=2)
			printf("%d %d\n", ans[i]+1, ans[i+1]+1);
	}
	return 0;
}
