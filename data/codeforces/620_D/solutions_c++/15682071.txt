#include<cstdio>
#include<map>

using namespace std;
const int nmax = 2010;

long long abs(long long x){
  return x>0 ? x : -x;
}

int n,m,a[nmax],b[nmax];
long long sum_a,sum_b,res = -1;
map<long long,pair<int,int> > mp;
map<long long,pair<int,int> >::iterator it;
pair<int,int> sol[3];

int main(){
scanf("%d",&n);
for(int i = 0; i < n; ++i){
  scanf("%d",&a[i]);
  sum_a += a[i];
}
scanf("%d",&m);
for(int i = 0; i < m; ++i){
  scanf("%d",&b[i]);
  sum_b += b[i];
}
res = abs(sum_a - sum_b);
sol[0].first = -1; sol[0].second = -1;
sol[1].first = -1; sol[1].second = -1;

for(int i = 0; i < n; ++i)
  for(int j = 0; j < m; ++j){
    if(abs( (sum_a-a[i]+b[j]) - (sum_b-b[j]+a[i]) ) < res){
       res = abs( (sum_a-a[i]+b[j]) - (sum_b-b[j]+a[i]) );
       sol[0].first = i; sol[0].second = j;
    }
  }

for(int i = 0; i < n; ++i)
  for(int j = i+1; j < n; ++j)
    mp[2ll*(a[i] + a[j])] = make_pair(i,j);

if(n > 1 && m > 1){
for(int i = 0; i < m; ++i)
  for(int j = i+1; j < m; ++j){
   // if(i == 0 && j == 53) printf("a[%d] = %d,b[%d]=%d,a[%d] = %d,b[%d]=%d\n",0,a[0],5,b[5],0,a[0],53,b[53]);
    long long val = sum_a-sum_b+2ll*(b[i]+b[j]);
    it = mp.lower_bound(val);
    if (it != mp.begin()) it--;
    for(int k = 0; k < 2; ++k){
      if (it == mp.end()) break;
      if(abs(val - it->first) < res){
        res = abs(val - it->first);
        sol[0].first = it->second.first; sol[0].second = i;
        sol[1].first = it->second.second; sol[1].second = j;
      }
      it++;
    }
  }
}

printf("%I64d\n",res);
if(sol[0].first != -1 && sol[1].first != -1) printf("2\n");
else if(sol[0].first != -1) printf("1\n");
else printf("0");

if(sol[0].first != -1) printf("%d %d\n",sol[0].first + 1,sol[0].second + 1);
if(sol[1].first != -1) printf("%d %d\n",sol[1].first + 1,sol[1].second + 1);

return 0;
}
