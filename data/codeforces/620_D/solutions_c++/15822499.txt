#include<iostream>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<cstdio>
#include<iomanip>
using namespace std;
long long suma=0,sumb=0;
long long a[2005],b[2005];
long long ans1;
int a1,b1;
struct hehe{int i,j;long long sum;} he[2000*1000+5];
long long ans2;
int a21,a22,b21,b22;
int n,m,num;
void solve1()
{
	int i,j;
	long long tmp;
	a1=b1=0;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{
			tmp=abs((suma-a[i]+b[j])-(sumb-b[j]+a[i]));
			if(tmp<ans1)
			{
				ans1=tmp;
				a1=i;b1=j;
			}
		}
}
bool ttmp(hehe x,hehe y)
{
	return x.sum<y.sum;
}
int zhao(long long x)
{
	int l,r,mid;
	l=1;r=num;
	while(l<r)
	{
		mid=(l+r)/2;
		if(he[mid].sum>=x) r=mid;
		if(he[mid].sum<x) l=mid+1;
	}
	return l;
}
void solve2()
{
	int i,j,bj;
	long long x,tmp;
	num=0;
	for(i=1;i<=n;i++)
		for(j=i+1;j<=n;j++)
		{
			num++;
			he[num].i=i;
			he[num].j=j;
			he[num].sum=2*(a[i]+a[j]);
		}
	sort(he+1,he+num+1,ttmp);
	for(i=1;i<=m;i++)
		for(j=i+1;j<=m;j++)
		{
			x=suma-sumb+2*(b[i]+b[j]);
			bj=zhao(x);
			tmp=abs(x-he[bj].sum);
			if(tmp<ans2)
			{
				ans2=tmp;
				a21=he[bj].i;
				a22=he[bj].j;
				b21=i;b22=j;
			}
			tmp=abs(x-he[bj-1].sum);
			if(tmp<ans2&&bj>=2)
			{
				ans2=tmp;
				a21=he[bj-1].i;
				a22=he[bj-1].j;
				b21=i;b22=j;
			}
		}
}
void solve()
{
	if(ans1<=ans2)
	{
		if(ans1==abs(suma-sumb))
		{
			cout<<ans1<<endl;
			cout<<0<<endl;
		}
		else
		{
			cout<<ans1<<endl;
			cout<<1<<endl;
			cout<<a1<<" "<<b1<<endl;
		}
	}
	else
	{
		cout<<ans2<<endl;
		cout<<2<<endl;
		cout<<a21<<" "<<b21<<endl;
		cout<<a22<<" "<<b22<<endl;
	}
}
int main()
{
	int i;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
		suma+=a[i];
	}	
	cin>>m;
	for(i=1;i<=m;i++)
	{
		cin>>b[i];
		sumb+=b[i];
	}
	ans1=ans2=abs(suma-sumb);
	solve1();
	if(n>=2&&m>=2)
		solve2();
	solve();
	return 0;
}
