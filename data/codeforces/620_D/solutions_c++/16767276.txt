#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define mp make_pair
const int N=2050;
int a[N],b[N];
pair<int,int> ans1,ans21,ans22;
pair<ll,pair<int,int> >two[N*N];
int tot=0;
int main()
{
    ll s=0;
    //freopen("f.txt","r",stdin);
    int n;scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]),s+=a[i];
    int m;scanf("%d",&m);
    for(int i=1;i<=m;i++)scanf("%d",&b[i]),s-=b[i];
    ll s2=1e18;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            ll cur=abs(s+2*b[j]-2*a[i]);
            if(cur<s2)s2=cur,ans1=mp(i,j);
        }
    }
    for(int i=1;i<=m;i++)
        for(int j=i+1;j<=m;j++)
        two[++tot]=mp(2ll*(b[i]+b[j]),mp(i,j));
    sort(two+1,two+tot+1);
    ll s3=1e18;
    for(int i=1;i<=n;i++){
        for(int j=i+1;j<=n;j++){
            ll tmp=2ll*a[i]+2ll*a[j]-s;
            int p=lower_bound(two+1,two+1+tot,mp(tmp,mp(0,0)))-two;
            for(int t=max(1,p-2);t<=min(tot,p+2);t++){
                ll cur=abs(s+two[t].first-2ll*a[i]-2ll*a[j]);
                if(cur<s3){
                    s3=cur;
                    ans21=mp(i,two[t].second.first);
                    ans22=mp(j,two[t].second.second);
                }
            }
        }
    }
    ll ans=min(min(abs(s),abs(s2)),abs(s3));
    cout<<ans<<endl;
    if(ans==s)printf("0\n");
    else if(ans==abs(s2)){
        printf("1\n");
        printf("%d %d\n",ans1.first,ans1.second);
    }
    else{
        printf("2\n");
        printf("%d %d\n%d %d\n",ans21.first,ans21.second,ans22.first,ans22.second);
    }
    return 0;
}
