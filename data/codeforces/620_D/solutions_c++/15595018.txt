#include<bits/stdc++.h>
#include<cstring>
#include<iostream>
#include<cstdio>
#include<set>
#include<cstdlib>
#include<algorithm>
#include<map>
using namespace std;
#define ll long long
map<ll,pair<int,int> >x;
map<ll,pair<int,int> >::iterator it;
ll a[2005],b[2005],n,m;
ll sum1,sum2;
ll minn=1e18;
pair<int,int>ans1,ans2;int num;
void check(ll a,ll b,pair<int,int>c,pair<int,int>d){
    if(abs(a-b)<minn){
        minn=abs(a-b);num=2;
        ans1=c;ans2=d;
    }
}
int main(){
    cin>>n;sum1=0;sum2=0;num=0;
    for(int i=1;i<=n;i++){
        scanf("%lld",&a[i]);sum1+=a[i];
    }cin>>m;
    for(int i=1;i<=m;i++){
        scanf("%lld",&b[i]);sum2+=b[i];
    }
    minn=abs(sum1-sum2);
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            if(abs(sum1-2ll*a[i]-sum2+2ll*b[j])<minn){
                num=1;
                minn=abs(sum1-a[i]*2ll-sum2+2ll*b[j]);ans1={i,j};
            }
        }
    }
    for(int i=1;i<=n;i++){
     for(int j=i+1;j<=n;j++){
            x[2ll*(a[i]+a[j])]={i,j};
        }
    }
    for(int i=1;i<=m;i++){
     for(int j=i+1;j<=m;j++){
            ll sum=sum1-sum2+2ll*(b[i]+b[j]);
            it=x.lower_bound(sum);
            if(it!=x.end()) check(sum,it->first,it->second,{i,j});
            if(it!=x.begin()) {it--;check(sum,it->first,it->second,{i,j});}
        }
    }
    printf("%lld\n",minn);
    if(num==1){
        printf("1\n");printf("%d %d\n",ans1.first,ans1.second);
    }
    else if(num==2){
        printf("2\n");printf("%d %d\n",ans1.first,ans2.first);
        printf("%d %d\n",ans1.second,ans2.second);
    }
    else printf("0\n");
    return 0;
}
