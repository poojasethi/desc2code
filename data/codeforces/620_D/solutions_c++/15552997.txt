#include <bits/stdc++.h>
using namespace std;
#define ll long long 
ll sa,sb,ans,k;
ll n,a[2009],m,b[2009];
struct node
{
	ll s;
	int i,j;
	node(ll a,int b,int c):s(a),i(b),j(c){}
	node(){}
	bool operator<(struct node a)
	{
		return this->s<a.s;
	}
};
bool cmp(node a,node b)
{
	return a.s<b.s;
}
node p[4000009];
int main() 
{
	cin>>n;
	for(int i=1;i<=n;i++){cin>>a[i];sa+=a[i];}
	cin>>m;
	for(int i=1;i<=m;i++){cin>>b[i];sb+=b[i];}
	vector<int> f,s;
	ans=abs(sa-sb);
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			if(ans>abs(sa-sb-2*a[i]+2*b[j]))
			{
				ans=abs(sa-sb-2*a[i]+2*b[j]);
				f.clear();s.clear();
				f.push_back(i);
				s.push_back(j);
			}
		}
	}
	for(int i=1;i<=n;i++)
	{
		for(int j=i+1;j<=n;j++)
		{
			p[++k]=node(2*a[i]+2*a[j],i,j);
		}
	}
	sort(p+1,p+k+1,cmp);
	//for(int i=1;i<=k;i++)cout<<p[i].s<<" "<<p[i].i<<" "<<p[i].j<<endl;
	for(int i=1;i<=m;i++)
	{
		for(int j=i+1;j<=m;j++)
		{
			ll t=(sa-sb+2*(b[i]+b[j]));
			int id=(int)(upper_bound(p+1,p+k+1,node(t,0,0),cmp)-p);
			if(id<k+1 && ans>abs(t-p[id].s))
			{
				ans=abs(t-p[id].s);
				f.clear();s.clear();
				f.push_back(p[id].i);
				f.push_back(p[id].j);
				s.push_back(i);
				s.push_back(j);
			}
			id--;
			if(id>=1 && ans>abs(t-p[id].s))
			{
				ans=abs(t-p[id].s);
				f.clear();s.clear();
				f.push_back(p[id].i);
				f.push_back(p[id].j);
				s.push_back(i);
				s.push_back(j);
			}
		}
	}
	cout<<ans<<endl;
	cout<<f.size()<<endl;
	for(int i=0;i<f.size();i++)cout<<f[i]<<" "<<s[i]<<endl;
	return 0;
}
