#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<vector>
#include<map>
#define Maxn 2010
#define ll long long
using namespace std;

int a[Maxn],b[Maxn];
int a1=-1,a2=-1,a3=-1,a4=-1;
map<ll,int> mp;
map<ll,int>::iterator it;
int main()
{
    int n,m;
    ll sa=0,sb=0;
    cin>>n;
    for(int i=1;i<=n;i++) cin>>a[i],sa+=a[i];
    cin>>m;
    for(int i=1;i<=m;i++) cin>>b[i],sb+=b[i];
    ll ans=llabs(sa-sb);
    //swap()
    //cout<<sa<<" ** "<<sb<<endl;
    //cout<<sa-sb-2*a[3]+2*b[3]-2*a[8]+2*b[9]<<endl;;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++){
            ll sub=llabs(sa-sb-2*a[i]+2*b[j]);
            if(sub<ans){
                ans=sub;
                a1=i,a2=j;
            }
        }
    //cout<<ans<<"***"<<endl;
    for(int i=1;i<m;i++)
        for(int j=i+1;j<=m;j++)
            mp[sa-sb+2*b[i]+2*b[j]]=(i-1)*m+(j-1);
    for(int i=1;i<n;i++)
        for(int j=i+1;j<=n;j++){
            ll tmp=(ll)2*a[i]+2*a[j];
            it=mp.lower_bound(tmp);
            int fr,se;
            if(it!=mp.end()){
                se=it->second%m+1;
                fr=it->second/m+1;
                if(llabs(it->first-tmp)<ans){
                    ans=llabs(it->first-tmp);
                    a1=i,a2=fr,a3=j,a4=se;
                    //if(!ans) cout<<i<<"+++"<<j<<" "<<tmp<<" "<<it->first<<" "<<fr<<" "<<se<<endl;
                }
            }
            if(it!=mp.begin()){
                --it;
                se=it->second%m+1;
                fr=it->second/m+1;
                if(llabs(it->first-tmp)<ans){
                    ans=llabs(it->first-tmp);
                    a1=i,a2=fr,a3=j,a4=se;
                }
            }
        }
    cout<<ans<<endl;
    if(a1==-1) puts("0");
    else if(a3==-1)
        printf("1\n%d %d",a1,a2);
    else
        printf("2\n%d %d\n%d %d\n",a1,a2,a3,a4);
	return 0;
}
