from sys import stdin
def cmpz(x,y):
    if x[1] != y[1]:
        return x[1]-y[1]
    else:
        return y[0]-x[0]

n = int(stdin.readline())
zak = []
while n:
    z = map(int, stdin.readline().split())
    zak.append(z)
    n -=1

Z = sorted(zak, cmp = cmpz )

R = -1
N = 0
for l, r in Z:
    if l > R:
        N +=1
        R = r
print N
