
def main():
  num = int(raw_input())
  entries = [None] * num
  for i in xrange(num):
    v1, v2 = raw_input().split()
    entries[i] = (int(v1), int(v2))
  entries.sort(key=lambda t: t[1])
  cur = 0
  res = 0
  for (v1, v2) in entries:
    if v1 >cur:
      res += 1
      cur = v2
  return res
print main()
