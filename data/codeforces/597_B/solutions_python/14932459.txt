from math import *
from Queue import *
from sys import *
from datetime import *




n = int(raw_input())
time = []
for i in range(n):
    l, r = map(int, raw_input().split())
    time.append((r,l))
time.sort()
current = 0
c = 0
for t in time:
    if t[1] > current:
        c += 1
        current = t[0]
print(c)
