class Order(object):
    def __init__(self, l_, r_):
        self.l = l_
        self.r = r_

n = int(raw_input())

orders = []
for i in range(n):
    line = raw_input().split()
    orders.append(Order(int(line[0]), int(line[1])))

orders.sort(key=lambda o: o.r)


result = 0
last_end = 0

for o in orders:
    if o.l > last_end:
        last_end = o.r
        result += 1

print result
