def GI(): return int(raw_input())
def GIS(): return map(int, raw_input().split())

def main():
    n = GI()
    ts = []
    for i in xrange(n):
        ts.append(tuple(reversed(GIS())))
    ts.sort()

    cr = 0
    count = 0
    for e, s in ts:
        if s > cr:
            cr = e
            count += 1

    return count


print main()
