n=input();
x=[]
for i in range(n) :
    s,f=map(int,raw_input().split());
    x.append((f,s));
    
x.sort();
c=x[0][0];
d=1;
for i in range(1,n) :
    if (x[i][1]>c) :
        c=x[i][0];
        d+=1;
print d
