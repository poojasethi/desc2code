#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
typedef pair<int,int> P;
int main(){
  int n,l,r;
  vector<P> v,s,e;
  cin >> n;
  for(int i=0;i<n;i++){
    cin >> l >> r;
    v.push_back(P(r,l));
  }
  sort(v.begin(),v.end());
  int ans = 0,t = 0;
  for(int i=0;i<n;i++){
    if(t < v[i].second){
      ans++;
      t = v[i].first;
    }
  }
  cout << ans << endl;
}
