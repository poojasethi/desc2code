#include<bits/stdc++.h>
using namespace std;
pair<int,int>a[5000050];
int i,j,n,k,ans=1,l;
int main()
{
   cin>>n;
   for(i=0;i<n;i++)
    cin>>a[i].first>>a[i].second;
   sort(a,a+n);
   l=a[0].second;
  for(i=1;i<=n;i++)
   {
       if(l<a[i].first)
       {
           ans++;
           l=a[i].second;
       }
       else
        l=min(l,a[i].second);

   }
   cout<<ans<<endl;
}
