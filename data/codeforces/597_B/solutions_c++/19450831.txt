#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
	int n, ans=1, pos=0;
	cin >> n;
	vector <pair <int,int> >v(n);
	for(int i=0;i<n;++i){
		cin >> v[i].second >> v[i].first;
	}
	sort(v.begin(),v.end());
	for(int i=1;i<n;++i){
        
		if(v[i].second>v[pos].first){
			++ans;
			pos=i;
		}
	}
	cout << ans << endl;
}
