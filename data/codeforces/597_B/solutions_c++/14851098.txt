#include<iostream>
#include<algorithm>
#define f first;
#define s second;
using namespace std;
pair<int,int> a[500000];
int main()
{
    int n,cnt=0;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i].second>>a[i].f;
    sort(a,a+n);
    int now=-1;
    for(int i=0;i<n;i++)
    {
        if(a[i].second>now)
        {
            cnt++;
            now=a[i].f;
        }
    }
    cout<<cnt;
}
