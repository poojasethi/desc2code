#include<bits/stdc++.h>
using namespace std;
pair<int,int>a[500005];
int n;
int ans=0;
int l=0;
int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++)
        scanf("%d%d",&a[i].second,&a[i].first);
    sort(a,a+n);
    for(int i=0;i<n;i++)
    {
        if(a[i].second>l)
        {
            ans++;
            l=a[i].first;
        }
            
    }
    printf("%d\n",ans);
    return 0;
}
