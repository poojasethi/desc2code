#include<iostream>
#include<utility>
#include<algorithm>
using namespace std;

int main()
{
    pair<int, int>pi[500005];
    int a, b, ans=1, n, x;

    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a>>b;
        pi[i]=make_pair(b, a);
    }

    sort(pi, pi+n);
    x=pi[0].first;
    for(int i=1;i<n;i++)
    {
        if(pi[i].second>x) ans++, x=pi[i].first;
    }

    cout<<ans<<endl;
    return 0;
}
