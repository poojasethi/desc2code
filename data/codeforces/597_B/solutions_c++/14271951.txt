#include<bits/stdc++.h>
#define ll long long int

using namespace std;

pair<ll,ll> a[500005];

bool cmp(pair<ll,ll> p1,pair<ll,ll> p2){

	return p1.second<p2.second; 
}

int main(){
	ll n,c,end,i;
	cin>>n;

	for(i=0;i<n;i++)
		cin>>a[i].first>>a[i].second;

	sort(a,a+n,cmp);
    
    c=1;
    end=a[0].second;

	for(i=1;i<n;i++){

		if(a[i].first>end){
			c++;
			end=a[i].second;
		}
	}
	cout<<c<<endl;
	return 0;
}
