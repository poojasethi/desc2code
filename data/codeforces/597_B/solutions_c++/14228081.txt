#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n, c = 1 , b , ch = 0 ;
    pair <int,int> p[500000];
    cin >> n;

    for(int i=0 ; i<n ; i++)
        cin >> p[i].second >> p[i].first ;

    sort( p , p+n ) ;

    b = p[0].first ;
    for(int i=1 ; i<n ; i++)
    {
       if (p[i].second > b )
        c++ , b = p[i].first ;
    }

    cout << c ;

}
