#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;
const int nmax=500000;

struct X {int a,b;};
bool operator<(X x,X y){
 return x.b<y.b;     
}

int n,i,B,ans;
X x[nmax];
main(){
       
 cin>>n;
 for (i=0;i<n;i++)
  cin>>x[i].a>>x[i].b;
 
 sort(x,x+n);
 for (i=0;i<n;i++)
  if (x[i].a>B) {
  ans++;B=x[i].b;
 }
 
 cout<<ans<<endl;
 //system("pause");      
}
