#include <iostream>
#include <algorithm>
using namespace std;
pair <long long, long long> a[500000];
int main()
{
int n,ans=1,k;
cin>>n;
for(int i=0; i<n; i++)
    cin>>a[i].second>>a[i].first;

sort(a,a+n);
k=a[0].first;

for (int i=0; i<n;i++)
    if (k<a[i].second) {ans++; k=a[i].first;}
    cout<<ans;
    return 0;
}
