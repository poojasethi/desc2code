#include<bits/stdc++.h>
using namespace std;
#define fr first
#define pb push_back
#define sc second
#define pa pair<int,int>
vector < pa > v;
int n,l,r,res;
int main(){
	scanf("%d",&n);
	for(int i=1 ; i<=n ; i++){
		scanf("%d %d",&l,&r);
		v.pb(pa(r,l));
	}
	sort(v.begin(),v.end());
	for(int i=0,son=0 ; i<n ; i++){
		if(v[i].sc>son){
			res++;
			son=v[i].fr;
		}
	}
	printf("%d\n",res);
	return 0;
}
