#include<iostream>
#include<algorithm>
#define ll long long
#define pii pair<ll,ll>
#define X first
#define Y second
using namespace std;
pii a[500100];
int main()
{
    ll n,ans,mx=0;
    cin>>n;
    for(ll i=0;i<n;i++)
        cin>>a[i].Y>>a[i].X;
    sort(a,a+n);
    for(ll i=0;i<n;i++)
        if(a[i].Y>mx)
            mx=a[i].X,ans++;
    cout<<ans<<endl;
    return 0;
}
