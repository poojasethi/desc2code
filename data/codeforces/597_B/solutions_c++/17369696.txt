#include <bits/stdc++.h>
using namespace std;
vector < pair<int,int> > v;
int main()
{
	int n,x,y;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		v.push_back(make_pair(y,x));
	}
	sort(v.begin(),v.end());
	int ans = 1,lst = v[0].first;
	for(int i=1;i<n;i++)
	{
		if(v[i].second > lst)
			ans++;
		else continue;
		lst = v[i].first;
	}
	cout<<ans;
	return 0;
}
