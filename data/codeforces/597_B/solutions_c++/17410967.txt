#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
int n;
cin>>n;
pair < int  , int  >a[n];
int i;
memset(a,-1,sizeof(a));
for(i=0;i<n;i++)
   {
    scanf("%d",&a[i].second);
    scanf("%d",&a[i].first);
   }
sort(a,a+n);
int sum=0,last=a[0].first;
for(i=1;i<n;i++)
{
  if(a[i].second >last)
  {
      last = a[i].first;
  }
  else sum++;
}
cout<<n-sum;
}
