#include <bits/stdc++.h>
#define l __int64

l n;

struct xy{
	l x, y;
}a[500005];

bool check(xy c, xy d){
	return c.y < d.y;
}

main(){
	std::cin >> n;
	
	for(int i = 0; i < n; i++)
		std::cin >> a[i].x >> a[i].y;
	
	std::sort(a, a + n, check);
	
	l x = -1, s = 0;
	for(int i = 0; i < n && x < a[n - 1].y; i++){
		if(x < a[i].x){
			s++;
			x = a[i].y;
		}
	}
	std::cout << s;
	return 0;
}
