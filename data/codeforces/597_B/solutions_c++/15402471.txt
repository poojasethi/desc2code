#include<bits/stdc++.h>

using namespace std;

int main()
{
    long long  n,lim=0,i,cnt=0;
    cin>>n;
    pair<long long ,long long >a[n+5];

    for(i=0;i<n;i++){

        cin>>a[i].second;
        cin>>a[i].first;
    }
    sort(a,a+n);
    for(i=0;i<n;i++){
        if(a[i].second>lim){
            cnt++;
            lim=a[i].first;
        }
    }
    cout<<cnt;
    return 0;
}
