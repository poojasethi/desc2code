#include <bits/stdc++.h>
using namespace std;
int n,i,x,y,c;
pair <int , int> p[500000];
int main()
{
    cin>>n;
    for(i=0;i<n;i++)
	{
	cin>>x>>y;
	p[i]=make_pair(y,x);
	}
    sort(p,p+n);
    x=p[0].second,y=0;
    for(i=0;i<n;i++)if(p[i].second>y)y=p[i].first,x=p[i].second,c++;
    cout<<c<<endl;
    return 0;
}
