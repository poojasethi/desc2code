#include <bits/stdc++.h>
using namespace std;
vector<pair<int,int> > v;
int main(){
    int n; cin >> n;
    for(int i = 0 ; i < n ; i++){
        int x , y; cin >> x >> y;
        v.push_back(make_pair(y,x));
    }
    sort(v.begin(),v.end());
    int cnt = 0;
    int lst = 0;
    for(int i = 0 ; i < n ; i++){
        if(v[i].second >  lst) cnt++, lst = v[i].first;
    }
    cout << cnt << endl;
}
