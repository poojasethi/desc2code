#include<stdio.h>
#include<algorithm>
using namespace std;
struct s
{
	int n,m;
}a[500010];
bool cmp(s x,s y)
{
	return x.m<y.m;
}
int main()
{
	int i,n,num=0,t;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	scanf("%d%d",&a[i].n,&a[i].m);
	sort(a,a+n,cmp);
	t=a[0].m;
	for(i=1;i<n;i++)
	{
		if(a[i].n<=t) num++; 
		else t=a[i].m;
	}
	printf("%d\n",n-num);
	return 0; 
}
