#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	vector<pair<int , int> >v;
	int t1 , t2;

	for(int i = 0 ; i<n ; i++)
	{
		cin >> t1 >> t2;
		v.push_back(make_pair(t2 , t1));
	}

	sort(v.begin() , v.begin()+n);

	int prev = v[0].first;
	int count = 1;

	for(int i = 1 ; i<n ; i++)
	{
		if(v[i].second>prev)
		{
			count++;
			prev = v[i].first;
		}
	}

	cout << count << endl;
}
