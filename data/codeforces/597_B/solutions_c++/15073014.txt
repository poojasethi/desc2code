#include<bits/stdc++.h>
using namespace std;
pair<long long,long long>arr[500000];
int main(){
    int n,count=1;
    
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>arr[i].second>>arr[i].first;
    }
    sort(arr,arr+n);
    long long next=arr[0].first;
    for(int i=0;i<n-1;i++){
        if(next<arr[i+1].second){
            count++;
            next=arr[i+1].first;
        }
    }
    cout<<count;
}
