#include <iostream>
#include <algorithm>
#define NMAX 500005


using namespace std;

pair<int,int> P[NMAX];

int N,m,cnt;

int main()
{
	int i,j;
	cin >> N;
	for(i=1 ; i<=N ; i++)
		cin >> P[i].second >> P[i].first;
	sort(P+1,P+N+1);
	for(i=1 ; i<=N ; i++)
		if(m < P[i].first)	
		{
			cnt++;
			m = P[i].first;
			while(m >= P[i+1].second)
				++i;
		}
	cout << cnt << endl;
	return 0;
}
