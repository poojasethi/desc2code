#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(0);
	int n;
	cin>>n;
	vector<pair<int,int> >v(n);
	int f=0;
	int cnt=0;
	
	for(int i=0;i<n;i++)
	{
		cin>>v[i].second>>v[i].first;
	}
	sort(v.begin(),v.end());
	for(int i=0;i<n;i++)
	{
		if(v[i].second>f)
		{
			cnt++;
			f=v[i].first;
		}
	}
	cout<<cnt;
	return 0;
}
