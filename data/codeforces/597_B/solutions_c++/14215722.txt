#include<cstdio>
#include<algorithm>
#include<utility>
using namespace std;
#define r first
#define l second
int n,ans=0,last=-1;
pair<int,int> a[500005];

int main(){
	scanf("%d",&n);
	for(int i=0; i<n; i++)scanf("%d %d",&a[i].l,&a[i].r);
	sort(a,a+n);
	for(int i=0; i<n; i++){
		if(last<a[i].l){ans++; last=a[i].r;}
	}
	printf("%d\n",ans);
	return 0;
}
