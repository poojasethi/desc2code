#include <bits/stdc++.h>
using namespace std;
pair<int,int> ar[500000];
int n,i=0;
int main(){
    cin>>n;
    for(i=0;i<n;++i)
        scanf("%d%d",&ar[i].second,&ar[i].first);
    sort(ar,ar+n);
    int ret=0,start=-1;
    for(i=0;i<n;++i){
        if(ar[i].second>start)
            start= ar[i].first,++ret;
    }
    cout<<ret<<endl;
}
