#include <bits/stdc++.h>
using namespace std;
pair<int,int> o[500005];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;++i)
        scanf("%d%d",&o[i].second,&o[i].first),
        ++o[i].first; 
    sort(o,o+n);
    int prev=0;
    int cnt=1;
    for(int i=1;i<n;++i)
        if(o[i].second>=o[prev].first)
            prev=i,
            ++cnt;
    cout<<cnt;
}
