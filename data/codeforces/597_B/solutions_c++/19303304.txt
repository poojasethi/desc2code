#include<stdio.h>
#include<algorithm>
using namespace std;
struct stu
{
	int a,b;
}p[510000];
bool cmp(stu n,stu m)
{
	return n.b<m.b;
}
int main()
{
	int n;
	while(scanf("%d",&n)!=EOF)
	{
		for(int i=0;i<n;i++)
		scanf("%d%d",&p[i].a,&p[i].b);
		sort(p,p+n,cmp);
		int s=1,j=0;
		for(int i=1;i<n;i++)
		{
			if(p[i].a>p[j].b)
		{
			j=i;
			s++;
			}	
		}
	printf("%d\n",s);	
	}
	return 0;
}
