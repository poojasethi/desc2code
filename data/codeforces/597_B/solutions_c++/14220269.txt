#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    vector < pair <int, int> > a;
    ll n, x, y, e = 0, ans = 0;
    cin >> n;
    for(int i = 0 ;i < n;i++)
        cin >> x >> y, a.push_back({y, x});
    sort(a.begin(), a.end());
    for(int i = 0;i < n;i++)
        if(a[i].second > e)
            ans +=1 , e = a[i].first;
    cout << ans <<endl;
    return 0;
}
