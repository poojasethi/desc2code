#include<stdio.h>
#include<algorithm>
using namespace std;
struct x
{
	int st,end;
}f[500001];
bool cmp(x a,x b)
{
	return a.end<b.end;
}
int main()
{
	int m,n,t;
	while(~scanf("%d",&t))
	{
		for(int i=0;i<t;i++)
		scanf("%d%d",&f[i].st,&f[i].end);
		sort(f,f+t,cmp);
		m=0;
		n=1;
		for(int j=1;j<=t;j++)
		if(f[j].st>f[m].end)
		{
			m=j;
			n++;
		}
		printf("%d\n",n);
	}
}
