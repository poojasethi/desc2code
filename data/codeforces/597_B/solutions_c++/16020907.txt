#include <cstdio>
#include <queue>
using namespace std;

typedef pair<int, int> ii;

int main()
{
  priority_queue<ii> pq;
  int n, l, r, f = -100000000, a = 0;
  scanf("%d", &n);
  while (n--)
    scanf("%d%d", &l, &r), pq.push(ii(-r, -l));
  for (; !pq.empty(); pq.pop()) {
  	ii p = pq.top();
  	if (-p.second > f) f = -p.first, a++;
  }
  printf("%d\n", a);
}
