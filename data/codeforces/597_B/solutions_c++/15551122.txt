#include <stdio.h>
#include <algorithm>
using namespace std;
struct m
{
	long long e,f;
}a[500005];
bool cmp(m a,m b)
{
	if(a.f==b.f)
	return a.e<b.e;
	return a.f<b.f;
}
int main()
{
	int n,i,k=1;
	long long c;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	scanf("%I64d%I64d",&a[i].e,&a[i].f);
	sort(a,a+n,cmp);
	c=a[0].f;
	for(i=0;i<n;i++)
	if(a[i].e>c)
	{
		k++;
		c=a[i].f;
	}
	printf("%d\n",k);
	
}
