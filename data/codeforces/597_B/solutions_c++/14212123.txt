#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin >> n;
	int a,b,i,ans=0,ft=0;
	vector < pair < int , int > > v;
	for(i=0;i<n;i++)
	{
		cin >> a >> b;
		v.push_back(make_pair(b,a));
	}
	sort(v.begin(),v.end());
	for(i=0;i<n;i++)
	{
		if(v[i].second>ft)
		{	
			ans++;
			ft=v[i].first;
		}
	}
	cout << ans << endl;
	return 0;
}
