#include<bits/stdc++.h>
using namespace std;
int x,ans,e=0;
pair<int,int>a[500001];
int main(){
cin>>x;
   for(int i=0;i<x;i++){
       cin>>a[i].second>>a[i].first;
   }
  sort(a,a+x);
  for(int i=0;i<x;i++){
    //cout<<a[i].second<<" "<<a[i].first<<endl;
      if(e<a[i].second){
         ans++;
         e=a[i].first;
     }
  }
cout<<ans<<endl;
}
