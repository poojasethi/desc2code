#include <iostream>
#include <algorithm>
#include  <stdio.h>
using namespace std;
struct X{int a;int b;};
X x[500000];
int i,j,n,g=0,d=0;
bool operator < (X p , X q){
return p.b<q.b;
}
main()
{
   scanf("%d",&n);
   
   for(i=0;i<n;i++)
   scanf ("%d %d",&x[i].a,&x[i].b);
   
   sort(x,x+n);

 for (i=0;i<n;i++)
   if(x[i].a>d)
  {
     d=x[i].b;     
     g++;
  }
  
  cout<<g<<endl;
   


}
