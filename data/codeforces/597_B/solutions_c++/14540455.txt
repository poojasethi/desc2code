#include <bits/stdc++.h>
#define fi first
#define se second
using namespace std;

typedef pair < int , int > ii;

int main(){
	int n;
	scanf("%d",&n);
	vector < ii > v(n);
	for( int i=0 ; i < n ; i++ )
		scanf("%d%d",&v[i].se,&v[i].fi);
	sort(v.begin() , v.end());
	int ans = 0;
	for( int i=0,last=-1 ; i < n ; i++ )
		if( v[i].se > last )
			last = v[i].fi , ans++;
	printf("%d\n",ans);
	return 0;
}
