#include <iostream>

using namespace std;

int solve(int n, int m, int x){
	if ( n<0 || m<0 ) return 0;
	if ( x == 0 ) return 0;
	if ( x == 1 ) return (n * m + 1)/2;
	return solve( n-2, m-2, x-1 );
}

int main(){
	int n, m, x;
	
	cin >> n >> m >> x;
	
	cout << solve(n, m, x)-solve(n, m, x+1) << endl;

	return 0;
}
