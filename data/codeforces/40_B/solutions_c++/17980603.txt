#include<stdio.h>
int n, m, k, ans;
int main(){
   scanf("%d %d %d", &n, &m, &k);
   ans = 0;
   for(int i = 0; i < n; i ++){
      for(int j = 0; j < m; j ++){
         if((i+j)%2==0 and (i == k-1 or j == k-1 or i == n-k or j == m-k) and i >= k-1 and j >= k-1 and i <= n-k and j <= m-k){ ans++; }
      }
   }printf("%d\n", ans);
}
