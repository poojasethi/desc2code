#include <bits/stdc++.h>
using namespace std;
int calc(int x, int y){
	return (x * y + 1) / 2;
}
int main(){
	int n, m, x;
	scanf("%d %d %d", &n, &m, &x);
    int ret = calc(max(n - 2 * x + 2, 0), max(m - 2 * x + 2, 0)) - calc(max(n - 2 * x, 0), max(m - 2 * x, 0));
    printf("%d\n", ret);
    return 0;
}
