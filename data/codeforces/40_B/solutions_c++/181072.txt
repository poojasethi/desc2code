#include<cstdio>
#include<algorithm>

using namespace std;

int r, c, x, answer;

int main(){
	scanf("%d%d", &r, &c);
	scanf("%d", &x);
	for(int i=0; i<r; i++)
		for(int j=0; j<c; j++)
			if((i+j)%2==0)
				if( min( min(i+1, j+1), min(r-i, c-j) ) == x )
					answer++;
	printf("%d\n", answer);
	return 0;
}
