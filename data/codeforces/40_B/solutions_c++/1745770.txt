#include <iostream>
#include <algorithm>

using namespace std;

int main(){
    int m,x,n,ans=0;
    cin>>n>>m;
    cin>>x; 
    for (int i=1; i<=n;i++) for (int j=1;j<=m;j++) if (x==min(min(i,j),min(n-i+1,m-j+1)) && (i+j)%2==0) ans++;
    cout<<ans;
    return 0;
}
