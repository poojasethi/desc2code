#include <iostream>
#include <cmath>
using namespace std;
int n, m, x;
int f(int a, int b){
    if(a < 0 || b < 0)  return 0;
    return (int)ceil(a * b / 2.0);
}
int main(){
    cin >> n >> m >> x;
    x--;
    cout << f((n - 2 * x) , (m - 2 * x)) -  f((n - 2 * (x + 1)) , (m - 2 * (x + 1)))<< endl;
    return 0;
}
