#include <iostream>
#include <cmath>

using namespace std;

int main()
{
	long long int n, m, x;
	
	cin >> n >> m >> x;
	
	n = n - (x-1)*2;
	m = m - (x-1)*2;
	
	if (n <= 0 || m <= 0)
	{
		cout << 0;
	}
	else
	{
		if (n == 1 || m == 1)
			cout << (n + m) / 2;
		else
			cout << (n + m - 2);
	}
	return 0;
}
