#include <cstdio>
#include <cstring>

using namespace std;

int n,m,x;

int calc(int n,int m){
    if (n<=0||m<=0) return 0;
    if (n*m%2==0) return n*m/2;
    return (n*m+1)>>1;
}

int main(){
    scanf("%d%d%d",&n,&m,&x);x--;
    printf("%d\n",calc(n-x*2,m-x*2)-calc(n-x*2-2,m-x*2-2));
    return 0;
}
