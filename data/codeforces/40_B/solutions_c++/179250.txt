#include <iostream>
using namespace std;

int ans (int n, int m, int x)
{
	if (x == 1) return (n * m + 1) / 2;
	if (n <= 2 || m <= 2) return 0;
	return ans (n - 2, m - 2, x - 1);
};

int main ()
{
	int n, m;
	int x;
	cin >> n >> m >> x;
	cout << ans (n, m, x) - ans (n, m, x + 1) << endl;
	return 0;
}
