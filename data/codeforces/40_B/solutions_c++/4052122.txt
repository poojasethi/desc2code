#include <iostream>
using namespace std;

int main() {
    int N, M, X;

    cin >> N >> M >> X; X--;
    while (X > 0 && M > 0 && N > 0) {
        N -= 2; M -=2; X--;
    }
    if (N == 1 && M == 1) cout << 1 << endl;
    else if (M > 0 && N > 0) cout << N+M-2 << endl;
    else cout << 0 << endl;

    return 0;
}
