n , m = map(int , raw_input().split())
x = int(raw_input())

def change():
    global n , m
    n -= 2
    m -= 2
    n = max(n , 0)
    m = max(m , 0)

def calc(n , m):
    ret = 0
    for i in range(1 , n+1):
        ret += m/2
        if m%2 == 1:
            ret += (i&1)
    return ret

for i in range(1 , min(x , 10000)):
    change()
ans = calc(n , m)
change()
ans -= calc(n , m)
print(ans)
