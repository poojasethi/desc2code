#! /usr/bin/env python
from sys import stdin, setrecursionlimit as depth

depth(10000)

def count_black(length,init):
	if length<0:
		return 0
	if init: #black
		ret=length/2
		if length%2:
			ret+=1
	else:
		ret=length/2
	return ret
	
def paint(n,m,x,step):
	#print n,m,x,step
	if n<=0 or m<=0:
		return 0
	elif x==1:
		if n==1:
			return count_black(m,True)
		elif m==1:
			return count_black(n,True)
		else:
			top_row=count_black(m,True)
			bottom_row=count_black(m,((n-1)%2==0))
			left_col=count_black(n-2,False)
			right_black=((n+m-1)%2)==0
			right_col=count_black(n-2,right_black)
			#print ">>> %d %d %d %d"%(top_row,bottom_row,left_col,right_col)
			return sum((top_row,bottom_row,left_col,right_col))
	else:
		return paint(n-2,m-2,x-1,step+1)
			

if __name__=='__main__':
	n,m=map(int,stdin.readline().split())
	x=int(stdin.readline())
	print paint(n,m,x,0)
