n, m = map(int, raw_input().split())
x = input()
def black(x1, y1, x2, y2):
  if x1 > x2 or y1 > y2:
    return 0
  h, w = x2 - x1 + 1, y2 - y1 + 1
  return ((h + 1) / 2) * ((w + 1) / 2) + (h / 2) * (w / 2)
if min(n, m) >= x + x - 1:
  print black(x - 1, x - 1, n - x, m - x) - black(x, x, n - x - 1, m - x - 1)
else:
  print 0
