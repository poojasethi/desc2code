#include<cstdio>
#include<cstring>
#include<algorithm>
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=4100;
int ti,td,tr,te,n,m,i,j,x,y;
int f[mn][mn];
int la[30],lb[30];
char a[mn],b[mn];
int main()
{
    scanf("%d%d%d%d",&ti,&td,&tr,&te);
    scanf("%s%s",a+1,b+1);
    n=strlen(a+1),m=strlen(b+1);
    fo(i,1,n)f[i][0]=i*td,a[i]-=96;
    fo(i,1,m)f[0][i]=i*ti,b[i]-=96;
    fo(i,1,n)
    {
        memset(lb,0,sizeof lb);
        fo(j,1,m)
        {
            f[i][j]=f[i-1][j-1]+tr*(a[i]!=b[j]);
            f[i][j]=min(f[i][j],min(f[i-1][j]+td,f[i][j-1]+ti));
            x=la[b[j]],y=lb[a[i]];
            if(x&&y)f[i][j]=min(f[i][j],f[x-1][y-1]+ti*(j-y-1)+td*(i-x-1)+te);
            lb[b[j]]=j;
        }
        la[a[i]]=i;
    }
    printf("%d\n",f[n][m]);
    return 0;
}
