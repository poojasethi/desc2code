#include<cstdio>
#include<cstring>
#define N 4005
int f[N][N],g1[N][255],g2[N][255],t1,t2,t3,t4;
char a[N],b[N];
inline void Min(int &x,int y){if (y<x) x=y;}
int main()
{
	scanf("%d%d%d%d",&t1,&t2,&t3,&t4); scanf("%s%s",a,b);
	int n=strlen(a),m=strlen(b); memset(f,6,sizeof(f)); f[0][0]=0;
	for (int i='a';i<='z';i++) g1[n][i]=n,g2[m][i]=m;
	for (int i=n-1;i>=0;i--) memcpy(g1[i],g1[i+1],sizeof(g1[i])),g1[i][a[i]]=i;
	for (int i=m-1;i>=0;i--) memcpy(g2[i],g2[i+1],sizeof(g2[i])),g2[i][b[i]]=i;
	for (int i=0;i<=n;i++)
		for (int j=0;j<=m;j++){
			int x=g1[i][b[j]],y=g2[j][a[i]],z=f[i][j];
			Min(f[i][j+1],z+t1); Min(f[i+1][j],z+t2);
			Min(f[i+1][j+1],z+(a[i]==b[j]?0:t3));
			if (a[i]-b[j]) Min(f[x+1][y+1],z+t4+(x-i-1)*t2+(y-j-1)*t1);
			}
	printf("%d\n",f[n][m]); return 0;
}
