#include <cstdio>
#include <cstring>
#include <algorithm>
#define update(a,b,c,d,x) (f[a][b]=std::min(f[a][b],f[c][d]+x))
int N,M,A,B,C,D,f[4005][4005],p[4005][26],q[4005][26];
char a[4005],b[4005];
int main()
{
	scanf("%d%d%d%d%s%s",&A,&B,&C,&D,a+1,b+1),N=strlen(a+1),M=strlen(b+1);
	for (int i=0; i<26; i++) p[N+1][i]=N+1,q[M+1][i]=M+1;
	for (int i=N; i; i--)
		for (int j=0; j<26; j++) p[i][j]=a[i]==j+97?i:p[i+1][j];
	for (int i=M; i; i--)
		for (int j=0; j<26; j++) q[i][j]=b[i]==j+97?i:q[i+1][j];
	memset(f,63,sizeof(f)),f[0][0]=0;
	for (int i=0; i<=N; i++)
		for (int j=0,x,y; j<=M; j++)
		{
			if (a[i+1]==b[j+1]) update(i+1,j+1,i,j,0);
			update(i,j+1,i,j,A),update(i+1,j,i,j,B),update(i+1,j+1,i,j,C);
			if (i<N&&j<M) x=p[i+2][b[j+1]-97],y=q[j+2][a[i+1]-97],update(x,y,i,j,B*(x-i-2)+D+A*(y-j-2));
		}
	printf("%d\n",f[N][M]);
}
