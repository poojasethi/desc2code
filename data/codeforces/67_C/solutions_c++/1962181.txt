#include <iostream>
#include <stdio.h>
#include <string.h>
#include <fstream>

using namespace std;

int ti,td,tr,te;
char s1[4040],s2[4040];
int len1,len2;
int dp[4040][4040];
int la[30],lb[30];

int main()
{
  //  freopen("in.txt","r",stdin);
    cin>>ti>>td>>tr>>te;
    scanf("%s%s",s1+1,s2+1);
    len1=strlen(s1+1);
    len2=strlen(s2+1);
    for(int i=1;i<=len1;i++) dp[i][0]=i*td;
    for(int i=1;i<=len2;i++) dp[0][i]=i*ti;
    for(int i=1;i<=len1;i++)
    {
        memset(lb,0,sizeof(lb));
        for(int j=1;j<=len2;j++)
        {
            dp[i][j]=dp[i-1][j-1];
            if(s1[i]!=s2[j]) dp[i][j]+=tr;
            dp[i][j]=min(dp[i][j],dp[i][j-1]+ti);
            dp[i][j]=min(dp[i][j],dp[i-1][j]+td);
            int ii,jj;
            ii=la[s2[j]-'a'];
            jj=lb[s1[i]-'a'];
            if(ii&&jj)dp[i][j]=min(dp[i][j],dp[ii-1][jj-1]+(i-ii-1)*td+(j-jj-1)*ti+te);
            lb[s2[j]-'a']=j;
        }
        la[s1[i]-'a']=i;
    }
    cout<<dp[len1][len2]<<endl;
    return 0;
}
