#include<cstdio>
#include<cstring>
#define add(a,b) if(b<a)a=b
#define min(a,b) (a<b?a:b)
using namespace std;
int t1,t2,t3,t4,l1,l2,f[4010][4010],i,j,sg1[4010][30],sg2[4010][30],p1,p2;
char s1[4010],s2[4010];
int main()
{
    scanf("%d%d%d%d",&t1,&t2,&t3,&t4);
    scanf("%s",s1+1);scanf("%s",s2+1);
    l1=strlen(s1+1);l2=strlen(s2+1);
    for(i=1;i<=l1;i++)
    {
        f[i][0]=t2*i;
        for(j=0;j<26;j++)sg1[i][j]=sg1[i-1][j];
        if(i>1)sg1[i][s1[i-1]-'a']=i-1;
    }for(i=1;i<=l2;i++)
    {
        f[0][i]=t1*i;
        for(j=0;j<26;j++)sg2[i][j]=sg2[i-1][j];
        if(i>1)sg2[i][s2[i-1]-'a']=i-1;
    }for(i=1;i<=l1;i++)
        for(j=1;j<=l2;j++)
        {
            f[i][j]=min(f[i-1][j]+t2,f[i][j-1]+t1);
            add(f[i][j],((s1[i]!=s2[j])*t3+f[i-1][j-1]));
            if((p2=sg2[j][s1[i]-'a'])&&(p1=sg1[i][s2[j]-'a']))
                add(f[i][j],(f[p1-1][p2-1]+(i-p1-1)*t2+(j-p2-1)*t1+t4));
        }printf("%d\n",f[l1][l2]);
}
