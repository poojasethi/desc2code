#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int f[4001][4001],p[26],q[26];
char a[4001],b[4001];

int main(void)
{
  int ti,td,tr,te;
  scanf("%d %d %d %d",&ti,&td,&tr,&te);
  scanf("%s",a+1);
  scanf("%s",b+1);
  int n=(int)strlen(a+1),m=(int)strlen(b+1);
  f[0][0]=0;
  for(int i=1;i<=n;++i)
    f[i][0]=i*td;
  for(int i=1;i<=m;++i)
    f[0][i]=i*ti;
  memset(p,0,sizeof(p));
  for(int i=1;i<=n;++i)
  {
    memset(q,0,sizeof(q));
    for(int j=1;j<=m;++j)
    {
      f[i][j]=min(f[i][j-1]+ti,f[i-1][j]+td);
      if(a[i]==b[j])
        f[i][j]=min(f[i][j],f[i-1][j-1]);
      else
        f[i][j]=min(f[i][j],f[i-1][j-1]+tr);
      int k=p[b[j]-'a'],l=q[a[i]-'a'];
      if((k>0)&&(l>0))
        f[i][j]=min(f[i][j],f[k-1][l-1]+td*(i-k-1)+te+ti*(j-l-1));
      q[b[j]-'a']=j;
    }
    p[a[i]-'a']=i;
  }
  printf("%d\n",f[n][m]);
  return 0;
}
