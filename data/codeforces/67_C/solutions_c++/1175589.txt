#include <cstdio>
#include <algorithm>
using namespace std;

int n,m,ti,td,tr,te,dp[4001][4001],ca[26],cb[26];

int main(){
    char a[4005],b[4005];
    scanf("%d%d%d%d%s%s",&ti,&td,&tr,&te,a+1,b+1);
    for(n=1;a[n];n++) dp[n][0]=td*n;
    for(m=1;b[m];m++) dp[0][m]=ti*m;
    n--,m--;
    for(int i=1;i<=n;i++){
        fill(cb,cb+26,0);
        for(int j=1;j<=m;j++){
            dp[i][j]=dp[i-1][j-1]+(a[i]!=b[j])*tr;
            dp[i][j]=min(dp[i][j],dp[i-1][j]+td);
            dp[i][j]=min(dp[i][j],dp[i][j-1]+ti);
            int x=ca[b[j]-'a'],y=cb[a[i]-'a'];
            if(x && y){
                int c=td*(i-x-1)+ti*(j-y-1)+te;
                dp[i][j]=min(dp[i][j],dp[x-1][y-1]+c);
            }
            cb[b[j]-'a']=j;
        }
        ca[a[i]-'a']=i;
    }
    printf("%d\n",dp[n][m]);
}
