#include<cstdio>
#include<cstring>
int T1,T2,T3,T4,i,j,I,J,l1,l2;
char S1[4005],S2[4005];
int f[4005][4005],Next1[30],Next2[30];
int Min(int a,int b)
{
	if(a<b) return a;
	return b;
}
int main()
{
	scanf("%d%d%d%d",&T1,&T2,&T3,&T4);
	scanf("%s",S1),scanf("%s",S2);
	l1=strlen(S1),l2=strlen(S2);
	for(i=0;i<=l2;++i) f[0][i]=T1*i;
	for(i=0;i<=l1;++i) f[i][0]=T2*i;
	for(i=1;i<=l1;++i)
	{
		for(j=0;j<26;++j) Next2[j]=0;
		for(j=1;j<=l2;++j)
		{
			f[i][j]=Min(f[i][j-1]+T1,f[i-1][j]+T2);
			if(S1[i-1]==S2[j-1]) f[i][j]=Min(f[i][j],f[i-1][j-1]);
			else f[i][j]=Min(f[i][j],f[i-1][j-1]+T3);
			I=Next1[S2[j-1]-'a'],J=Next2[S1[i-1]-'a'];
			if(I&&J) f[i][j]=Min(f[i][j],f[I-1][J-1]+T2*(i-I-1)+T1*(j-J-1)+T4);
			Next2[S2[j-1]-'a']=j;
		}
		Next1[S1[i-1]-'a']=i;
	}
	printf("%d\n",f[l1][l2]);
	return 0;
}
