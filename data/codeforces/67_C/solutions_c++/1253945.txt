#include<iostream>
#include<cmath>
#include<algorithm>
#include<cstring>
#include<cstdio>
#include<set>
#include<vector>
using namespace std;

typedef long long LL;

#define mp make_pair
#define pb push_back
const int maxn = 4111;
int p[2][99];
int t[5];
int n, m;
char A[ maxn ], B[ maxn ];
// 0-ins 1-del 2-change 3-exchange
int f[ maxn ][ maxn ];

void gao(){
	scanf("%s%s",A+1, B+1);
	n = strlen(A+1), m = strlen(B+1);
	memset(f,0x3f,sizeof f);
	for(int i=0;i<=n;++i) f[i][0] = i*t[1];
	for(int i=0;i<=m;++i) f[0][i] = i*t[0];
	memset(p,0,sizeof p);
	for(int i=1;i<=n;++i){
		memset( p[1],0,sizeof p[1]);
		for(int j=1;j<=m;++j){
			f[i][j]=min(f[i][j], f[i-1][j] + t[1]);
			f[i][j]=min(f[i][j], f[i][j-1] + t[0]);
			f[i][j]=min(f[i][j], f[i-1][j-1] + (A[i]==B[j]?0:t[2]));
			int pa, pb;
			pa=p[0][B[j]-'a'];
			pb=p[1][A[i]-'a'];
			if( pa && pb) {
				f[i][j] = min(f[i][j], f[pa-1][pb-1] + t[3] + (i-pa-1)*t[1] + (j-pb-1)*t[0]);
			}
			p[1][ B[j]-'a'] = j;
		}
		p[0][ A[i]-'a' ] = i;
	}
	cout << f[n][m]<<endl;
}

int main(){
	while(cin>>t[0]>>t[1]>>t[2]>>t[3]){
		gao();
	}
	return 0;
}
