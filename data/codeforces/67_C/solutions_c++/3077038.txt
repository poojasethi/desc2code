#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

char s[4002], t[4002];
int ti, td, tr, te, f[4001][4001], lt[4001][26], ls[4001][26];

int main() {
    scanf("%d%d%d%d%s%s",&ti,&td,&tr,&te,s+1,t+1);
    int n = strlen(s+1), m = strlen(t+1);
    for(int i = 1; i <= n; ++i)
        for(int j = i; j <= n; ++j)
            ls[j][s[i] - 0x61] = i;
    for(int i = 1; i <= m; ++i)
        for(int j = i; j <= m; ++j)
            lt[j][t[i] - 0x61] = i;
    for(int j = 0; j <= m; ++j) f[0][j] = j * ti;
    for(int i = 0; i <= n; ++i) f[i][0] = i * td;
    for(int i = 1; i <= n; ++i)
        for(int j = 1; j <= m; ++j)
            if(s[i] == t[j]) f[i][j] = f[i-1][j-1];
            else {
                f[i][j] = min(f[i][j-1] + ti, min(f[i-1][j] + td, f[i-1][j-1] + tr));
                int last1 = ls[i][t[j] - 0x61], last2 = lt[j][s[i] - 0x61];
                if(last1 && last2)
                    f[i][j] = min(f[i][j], f[last1-1][last2-1] + te + (i-last1-1)*td + (j-last2-1)*ti);
            }
    printf("%d\n", f[n][m]);
    return 0;
}
