#include <cstdio>
#include <algorithm>
#include <cstring>
#define INFI 999999999
using namespace std;
int cur[26],preS[5000][26],preT[5000][26],f[5000][5000];
char s[5000],t[5000];
int main()
{
	//freopen("input.txt","r",stdin);
	int I,D,R,E,i,j,x,y;
	scanf("%d%d%d%d",&I,&D,&R,&E);
	scanf("%s%s",s+1,t+1);
	for(i=1;s[i]!='\0';i++)
	{	
		for(j=0;j<26;j++)
			preS[i][j] = cur[j];
		cur[s[i]-'a'] = i;
	}
	memset(cur,0,sizeof(cur));
	for(i=1;t[i]!='\0';i++)
	{	
		for(j=0;j<26;j++)
			preT[i][j] = cur[j];
		cur[t[i]-'a'] = i;
	}
	for(i=0;i==0||s[i]!='\0';i++)
		for(j=0;j==0||t[j]!='\0';j++)
			f[i][j] = INFI;
	f[0][0] = 0; 
	for(i=0;i==0||s[i]!='\0';i++)
		for(j=0;j==0||t[j]!='\0';j++)
		if(i+j)
		{
			if(j) f[i][j] = min(f[i][j],f[i][j-1]+I);
			if(i) f[i][j] = min(f[i][j],f[i-1][j]+D);
			if(i && j) f[i][j] = min(f[i][j],f[i-1][j-1]+(s[i]==t[j]?0:R));
			x = preS[i][t[j]-'a'];
			y = preT[j][s[i]-'a'];
			if(x && y)
				f[i][j] = min(f[i][j],f[x-1][y-1]+D*(i-x-1)+E+I*(j-y-1));
			//printf("%d %d %d\n",i,j,f[i][j]);
		}
	printf("%d\n",f[i-1][j-1]);
}
