#include <algorithm>
#include <cstdio>
#include <cstring>
using namespace std;
const int INF=~0U>>2;
int ti,td,tr,te,la,lb,na[4003][27],nb[4003][27];
char sa[4003],sb[4003];
int f[4003][4003];
int main()
{
	scanf("%d%d%d%d",&ti,&td,&tr,&te);
	scanf("%s%s",sa+1,sb+1);
	la=strlen(sa+1),lb=strlen(sb+1);
	for (int i=1;i<=la;i++)
	{
		for (int j=0;j<26;j++) na[i][j]=na[i-1][j];
		na[i][sa[i]-'a']=i;
	}
	for (int i=1;i<=lb;i++)
	{
		for (int j=0;j<26;j++) nb[i][j]=nb[i-1][j];
		nb[i][sb[i]-'a']=i;
	}
	for (int i=0;i<=la;i++)
		for (int j=0;j<=lb;j++) f[i][j]=INF;
	f[0][0]=0;
	for (int i=1;i<=la;i++) f[i][0]=td*i;
	for (int i=1;i<=lb;i++) f[0][i]=ti*i;
	for (int i=1;i<=la;i++)
		for (int j=1;j<=lb;j++)
		{
			f[i][j]=min(f[i-1][j]+td,f[i][j-1]+ti);
			if (sa[i]==sb[j]) f[i][j]=min(f[i][j],f[i-1][j-1]);
			else f[i][j]=min(f[i][j],f[i-1][j-1]+tr);
			int u=na[i-1][sb[j]-'a'],v=nb[j-1][sa[i]-'a'];
			if (u && v)
				f[i][j]=min(f[i][j],f[u-1][v-1]+te+td*(i-u-1)+ti*(j-v-1));
		}
	int Ans=INF;
	printf("%d\n",f[la][lb]);
	return 0;
}
