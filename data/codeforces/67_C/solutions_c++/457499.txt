#include <iostream>
#include <string>
#include <vector>
#include <limits.h>

using namespace std;

typedef vector<int> vi;

int main()
{
	int in,dl,rp,ex;
	cin >> in >> dl >> rp >> ex;
	
	string from,to; cin >> from >> to;
	
	int n = from.size(), m = to.size();
	
	from = '$' + from;
	to = '$' + to;
	
	vector<vi> a(n+1,vi(m+1,0));
	vi afrom(26,0), ato(26,0);
	

	for(int i=1; i<=m; ++i)
		a[0][i] = i * in;

	for(int i=1; i<=n; ++i)
		a[i][0] = i * dl;

	for(int i=1; i<=n; ++i)
	{
		ato.assign(26,0);
		for(int j=1; j<=m; ++j)
		{
			int& v = a[i][j];
			v = INT_MAX;
			
			v = min(v, a[i-1][j-1] + (from[i] == to[j] ? 0 : rp));
			v = min(v, a[i][j-1] + in);
			v = min(v, a[i-1][j] + dl);
			int jj = ato[from[i]-'a'];
			int ii = afrom[to[j]-'a'];
			if (ii>0 && jj>0)
				v = min(v, a[ii-1][jj-1] + dl * (i-ii-1) + in * (j-jj-1) + ex);
			
			ato[to[j]-'a'] = j;
		}
		afrom[from[i]-'a'] = i;
	}

	cout << a[n][m] << endl;

	return 0;
}
