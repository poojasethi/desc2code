#include"stdio.h"
#include"string.h"
#include"algorithm"
#include"queue"
#include"map"
using namespace std;
char s1[4444],s2[4444];
int dp[4444][4444];
int main()
{
	int a,b,c,d;
    scanf("%d %d %d %d",&a,&b,&c,&d);
	s1[0]=s2[0]=3;
	scanf("%s",s1+1);
	scanf("%s",s2+1);
	int l1=strlen(s1),l2=strlen(s2);
	for(int i=1;i<l1;i++) 
	dp[i][0]=b*i;
		for(int i=1;i<l2;i++) 
		dp[0][i]=a*i;
	int fa[33]={
	0
	},fb[33]={
	0
	};
	for(int i=1;i<l1;i++) 
	 {
	 memset(fb,0,sizeof(fb));
	 for(int j=1;j<l2;j++)
     {
	      dp[i][j]=min(dp[i][j-1]+a,dp[i-1][j]+b);
          if(s1[i]==s2[j])
		  dp[i][j]=min(dp[i][j],dp[i-1][j-1]); 
		  int x1=fa[s2[j]-'a'],x2=fb[s1[i]-'a'];
		  if(x1&&x2)
		  {
		      int t=d+(i-x1-1)*b+(j-x2-1)*a;
		      dp[i][j]=min(dp[i][j],dp[x1-1][x2-1]+t);
   	      } 
		  if(s1[i]!=s2[j])
		  dp[i][j]=min(dp[i][j],min(dp[i-1][j-1]+c,min(dp[i][j-1]+a,dp[i-1][j]+b)));
		  fb[s2[j]-'a']=j;		     
      }
      fa[s1[i]-'a']=i;}
	  printf("%d\n",dp[l1-1][l2-1]); 
	return 0;
}
