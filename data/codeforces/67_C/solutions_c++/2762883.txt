#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
using namespace std;
const int maxn=4010;
int f[maxn][maxn],ga[maxn][150],gb[maxn][150];
int n,m,ti,td,tr,te,k,l;
char a[maxn],b[maxn];
int main(){
    //freopen("AC.in","r",stdin);
    //freopen("AC.out","w",stdout);
	cin>>ti>>td>>tr>>te>>a+1>>b+1;
	n=strlen(a+1);
    m=strlen(b+1);
    memset(ga,0,sizeof ga);
    memset(gb,0,sizeof gb);
    memset(f,0,sizeof f);
	for(int i=1;i<=n;i++){
		f[i][0]=i*td;
		for(int j='a';j<='z';j++)
            ga[i][j]=ga[i-1][j];
		ga[i][a[i]]=i;
	}
	for(int i=1;i<=m;i++){
		f[0][i]=i*ti;
    	for(int j='a';j<='z';j++)
            gb[i][j]=gb[i-1][j];
		gb[i][b[i]]=i;
	}
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++){
			f[i][j]=min(f[i][j-1]+ti,f[i-1][j]+td);
			f[i][j]=min(f[i-1][j-1]+tr*(a[i]!=b[j]),f[i][j]);
			if((k=ga[i-1][b[j]])&&(l=gb[j-1][a[i]]))
				f[i][j]=min(f[i][j],f[k-1][l-1]+te+td*(i-k-1)+ti*(j-l-1));
		}
	cout<<f[n][m]<<endl;
	return 0;
}
