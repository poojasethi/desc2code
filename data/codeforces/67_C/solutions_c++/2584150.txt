#include<cstdio>
#include<cstring>

int Next[30],Next2[30],f[4100][4100],n1,n2,tI,tD,tR,tE;

char Str1[4100],Str2[4100];

void Update(int&a,int b){if(a>b)a=b;}

int main()
{
	int i,j,k;
	scanf("%d%d%d%d",&tI,&tD,&tR,&tE);
	scanf("%s%s",Str1,Str2);
	n1=strlen(Str1);
	n2=strlen(Str2);
	memset(f,63,sizeof f);
	f[n1][n2]=0;
	for(j=0;j<26;j++)Next2[j]=100000;
	for(i=n1;i>=0;i--)
	{
		for(j=0;j<26;j++)Next[j]=100000;
		for(j=n2;j>=0;j--)
		{
			if(i<n1&&j<n2)Update(f[i][j],f[i+1][j+1]+tR);
			if(i<n1)Update(f[i][j],f[i+1][j]+tD);
			if(j<n2)Update(f[i][j],f[i][j+1]+tI);
			if(i<n1&&j<n2&&Str1[i]==Str2[j])
				Update(f[i][j],f[i+1][j+1]);
			if(i<n1&&j<n2&&Next[Str1[i]-'a']!=100000&&Next2[Str2[j]-'a']!=100000)
				Update(	f[i][j],f[Next2[Str2[j]-'a']+1][Next[Str1[i]-'a']+1]+tE
						+tI*(Next[Str1[i]-'a']-j-1)+tD*(Next2[Str2[j]-'a']-i-1));
			if(j!=n2)Next[Str2[j]-'a']=j;
		}
		if(i!=n1)Next2[Str1[i]-'a']=i;
	}
	printf("%d\n",f[0][0]);
	return 0;
}
