#include<cstdio>
#include<cstring>
int t1,t2,t3,t4,la[26],lb[26],f[4003][4003];
char s[4003],t[4003];
//                        Last Change:  2011-03-21 13:31:33
inline void update(int &x,int y){if(y<x)x=y;}
int main()
{
	scanf("%d%d%d%d",&t1,&t2,&t3,&t4);
	scanf("%s%s",s+1,t+1);
	int n=strlen(s+1);
	int m=strlen(t+1);
	for(int i=1;i<=n;++i)f[i][0]=t2*i;
	for(int i=1;i<=m;++i)f[0][i]=t1*i;
	for (int i=1;i<=n;++i)
	{
		memset(lb,0,sizeof(lb));
		for (int j=1;j<=m;++j)
		{
			f[i][j]=f[i-1][j-1]+t3*(s[i]!=t[j]);
			update(f[i][j],f[i][j-1]+t1);
			update(f[i][j],f[i-1][j]+t2);
			int ii=la[t[j]-'a'];
			int jj=lb[s[i]-'a'];
			if(ii>0&&jj>0)
				update(f[i][j],f[ii-1][jj-1]+t2*(i-ii-1)+t1*(j-jj-1)+t4);
			lb[t[j]-'a']=j;
		}
		la[s[i]-'a']=i;
	}
	printf("%d\n",f[n][m]);
	return 0;
}
