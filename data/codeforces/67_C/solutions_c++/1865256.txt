#include <iostream>
#include <cstdio>
#include <string.h>
using namespace std;
int ti, td, tr, te;
string A, B;
int dp[4001][4001];
int lastA['z'+1], lastB;
int main(){
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    scanf("%d %d %d %d",&ti,&td,&tr,&te);
    cin>>A>>B;
    A=" "+A, B =" "+B;
    dp[0][0] = 0;
    for (int i = 1; i < max(A.size(),B.size()); i++)
        dp[i][0] = dp[i-1][0]+td, dp[0][i] = dp[0][i-1]+ti;
    int alfa, beta;
    for (int a = 1; a < A.size(); a++){
        lastB = 0;
        for (int b = 1; b < B.size(); b++){
            alfa = lastA[B[b]], beta = lastB;
            if (A[a]==B[b])
                dp[a][b] = dp[a-1][b-1], lastB = b;
            else
                dp[a][b] = min(min(dp[a-1][b-1]+tr, dp[a-1][b]+td), dp[a][b-1]+ti);
            if (alfa && beta)
                dp[a][b] = min(dp[a][b], dp[alfa-1][beta-1]+te+(a-alfa-1)*td+(b-beta-1)*ti);
        }
        lastA[A[a]] = a;
    }
    printf("%d", dp[A.size()-1][B.size()-1]);
    return 0;
}
