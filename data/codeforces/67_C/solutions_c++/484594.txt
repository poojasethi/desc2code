#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int f[4005][4005];
char s1[4005],s2[4005];
int p[26],q[26];
int ti,td,tr,te;
int n,m;

int main()
{
    scanf("%d%d%d%d%s%s",&ti,&td,&tr,&te,s1+1,s2+1);
    n=strlen(s1+1);
    m=strlen(s2+1);
    for (int i=0;i<=n;i++)
    {
        for (int j=0;j<26;j++) q[j]=0;
        for (int j=0;j<=m;j++)
        {
            f[i][j]=1000000000;
            if (!i && !j) {f[i][j]=0;continue;}
            if (i) f[i][j]=min(f[i][j],f[i-1][j]+td);
            if (j) f[i][j]=min(f[i][j],f[i][j-1]+ti);
            if (i && j) 
            {
                f[i][j]=min(f[i][j],f[i-1][j-1]+(s1[i]!=s2[j])*tr);
                int u=p[s2[j]-'a'],v=q[s1[i]-'a'];
                if (u && v)
                    f[i][j]=min(f[i][j],f[u-1][v-1]+(i-u-1)*td+(j-v-1)*ti+te);
            }
            if (j) q[s2[j]-'a']=j;
        }
        if (i) p[s1[i]-'a']=i;
    }
    printf("%d\n",f[n][m]);
}
