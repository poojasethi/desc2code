#include <cstdio>
#include <cstring>
#include <algorithm>
#define maxn 4100
#define get_min(a,b) { a = min(a,b); }
using namespace std;
int i,j,k,l,m,n,a,b,c,d,x,y;
int f[maxn][maxn],A_[maxn][26],B_[maxn][26];
char A[maxn],B[maxn];
int main() {
	scanf("%d %d %d %d",&a,&b,&c,&d); // a -> insert b -> delete
	scanf("%s",A + 1); n = strlen(A + 1);
	scanf("%s",B + 1); m = strlen(B + 1);
	memset(f,1,sizeof(f)); f[0][0] = 0;
	for (i = n; i >= 1; i--) {
		for (j = 0; j < 26; j++) A_[i][j] = A_[i + 1][j];
		A_[i][A[i] - 'a'] = i;
	}
	for (i = m; i >= 1; i--) {
		for (j = 0; j < 26; j++) B_[i][j] = B_[i + 1][j];
		B_[i][B[i] - 'a'] = i;		
	}

	for (i = 0; i <= n; i++)
		for (j = 0; j <= m; j++) {
			if (A[i + 1] == B[j + 1]) get_min(f[i + 1][j + 1],f[i][j]);
			get_min(f[i + 1][j],f[i][j] + b);
			get_min(f[i][j + 1],f[i][j] + a);
			get_min(f[i + 1][j + 1],f[i][j] + c);
			if ((x = A_[i + 2][B[j + 1] - 'a']) && (y = B_[j + 2][A[i + 1] - 'a'])) {
				get_min(f[x][y],f[i][j] + (x - i - 2) * b + (y - j - 2) * a + d);
			}
		}
	printf("%d",f[n][m]);
	return 0;
}
