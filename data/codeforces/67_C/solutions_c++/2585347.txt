#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

const int maxn=4010;

int ti,td,tr,te,f[maxn][maxn],p1[300],p2[300];

char s1[maxn],s2[maxn];

int main()
{
	scanf("%d%d%d%d",&ti,&td,&tr,&te);
	scanf("%s%s",s1+1,s2+1);
	int l1=strlen(s1+1);
	int l2=strlen(s2+1);
	memset(f,0x3f,sizeof(f));
	for (int a=0;a<=l1;a++)
		f[a][0]=a*td;
	for (int a=0;a<=l2;a++)
		f[0][a]=a*ti;
	f[0][0]=0;
	for (int a=1;a<=l1;a++)
	{
		memset(p2,0,sizeof(p2));
		for (int b=1;b<=l2;b++)
		{
			f[a][b]=min(min(f[a-1][b]+td,f[a][b-1]+ti),f[a-1][b-1]+tr*(s1[a]!=s2[b]));
			int lastp1=p1[s2[b]],lastp2=p2[s1[a]];
			if (lastp1 && lastp2) f[a][b]=min(f[a][b],f[lastp1-1][lastp2-1]+(a-lastp1-1)*td+(b-lastp2-1)*ti+te);
			p2[s2[b]]=b;
		}
		p1[s1[a]]=a;
	}
	printf("%d\n",f[l1][l2]);

	return 0;
}
