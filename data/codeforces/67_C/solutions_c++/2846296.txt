#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <algorithm>
#include <iostream>
#define maxn 4020
#define rep(i,n) for(int i=1;i<=n;++i)
using namespace std;
int f[maxn][maxn];
char s[maxn],t[maxn];
int n,m;
int ti,td,tr,te;
int pre[2][300];
int main()
{
    scanf("%d%d%d%d\n",&ti,&td,&tr,&te);
    
    scanf("%s\n",s);
    n=strlen(s);
    scanf("%s\n",t);
    m=strlen(t);
    rep(i,n)f[i][0]=f[i-1][0]+td;
    rep(i,m)f[0][i]=f[0][i-1]+ti;
    rep(i,n)
    {
        memset(pre[1],0,sizeof(pre[1]));
        rep(j,m)
        {
            f[i][j]=min(min(f[i][j-1]+ti,f[i-1][j]+td),f[i-1][j-1]+tr*(s[i-1]!=t[j-1]));
            int x=pre[1][s[i-1]];
            int y=pre[0][t[j-1]];
            if(x&&y)
            {
                f[i][j]=min(f[i][j],f[y-1][x-1]+(i-y-1)*td+(j-x-1)*ti+te);
            }
            pre[1][t[j-1]]=j;
        }
        pre[0][s[i-1]]=i;
    }
    printf("%d\n",f[n][m]);
    return 0;
}
