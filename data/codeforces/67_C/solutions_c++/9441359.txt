#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define N 4010
using namespace std;
int t1, t2, t3, t4, n, m, i, j;
char s[N], t[N];
int A[N][200], B[N][200], f[N][N];
int main()
{
	scanf("%d%d%d%d", &t1, &t2, &t3, &t4);
	scanf("%s\n%s", s + 1, t + 1);
	n = strlen(s + 1), m = strlen(t + 1);
	for (i = 1; i <= n; i++)
	{
		f[i][0] = t2 * i;
		for (j = 1; j <= 26; j++)
			A[i][j] = A[i - 1][j];
		A[i][s[i] - 'a' + 1] = i;
	}
	for (i = 1; i <= m; i++)
	{
		f[0][i] = t1 * i;
		for (j = 1; j <= 26; j++)
			B[i][j] = B[i - 1][j];
		B[i][t[i] - 'a' + 1] = i;
	}
	for (i = 1; i <= n; i++)
		for (j = 1; j <= m; j++)
		{
			f[i][j] = min(f[i][j - 1] + t1, f[i - 1][j] + t2);
			if (s[i] == t[j]) f[i][j] = min(f[i][j], f[i - 1][j - 1]);
			f[i][j] = min(f[i][j], f[i - 1][j - 1] + t3);
			int a = A[i - 1][t[j] - 'a' + 1], b = B[j - 1][s[i] - 'a' + 1];
			if (!a || !b) continue;
			f[i][j] = min(f[i][j], f[a - 1][b - 1] + t4 + t2 * (i - a - 1) + t1 * (j - b - 1));
		}
	printf("%d", f[n][m]);
    return 0;
}
