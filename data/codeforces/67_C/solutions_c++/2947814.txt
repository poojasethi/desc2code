#include<stdio.h>
#include<string.h>
long min(long x,long y){return x<y?x:y;}
long I,D,R,E,i,lS,lT,j,u,v;
long f[4010][4010],nS[4010][26],nT[4010][26];
char S[4010],T[4010];
int main(){
    scanf("%ld%ld%ld%ld",&I,&D,&R,&E);
    scanf("\n");
    gets(S+1);lS=strlen(S+1);
    gets(T+1);lT=strlen(T+1);
    for (i=1;i<=lS;i++){
        f[i][0]=i*D;
        memcpy(nS[i],nS[i-1],sizeof(nS[i]));
        nS[i][S[i]-'a']=i;
        }
    for (i=1;i<=lT;i++){
        f[0][i]=i*I;
        memcpy(nT[i],nT[i-1],sizeof(nT[i]));
        nT[i][T[i]-'a']=i;
        }
    for (i=1;i<=lS;i++)
        for (j=1;j<=lT;j++){
            f[i][j]=min(min(f[i-1][j]+D,f[i][j-1]+I),f[i-1][j-1]+R*(S[i]!=T[j]));
            if ((u=nS[i-1][T[j]-'a'])&&(v=nT[j-1][S[i]-'a']))
                f[i][j]=min(f[i][j],f[u-1][v-1]+E+D*(i-u-1)+I*(j-v-1));
            }
    printf("%ld\n",f[lS][lT]);
    return 0;
}
