#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=4005;

char s[maxn],t[maxn];
int I,D,R,E;
int f[maxn][maxn],lt[maxn][26],ls[maxn][26];

int main(){
#ifndef ONLINE_JUDGE
	freopen("1.in","r",stdin);
	freopen("1.out","w",stdout);
#endif
    scanf("%d%d%d%d%s%s",&I,&D,&R,&E,s+1,t+1);
    int n=strlen(s+1),m=strlen(t+1);
    for(int i=1;i<=n;i++)
        for(int j=i;j<=n;j++)
            ls[j][s[i]-97]=i;
    for(int i=1;i<=m;i++)
        for(int j=i;j<=m;j++)
            lt[j][t[i]-97]=i;
    for(int j=0;j<=m;j++) 
    	f[0][j]=j*I;
    for(int i=0;i<=n;i++) 
    	f[i][0]=i*D;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            if(s[i]==t[j]) 
            	f[i][j]=f[i-1][j-1];
            else{
                f[i][j]=min(f[i][j-1]+I,min(f[i-1][j]+D,f[i-1][j-1]+R));
                int last1=ls[i][t[j]-97],last2=lt[j][s[i]-97];
                if(last1 && last2)
                    f[i][j]=min(f[i][j],f[last1-1][last2-1]+E+(i-last1-1)*D+(j-last2-1)*I);
            }
    printf("%d\n",f[n][m]);
    return 0;
}
