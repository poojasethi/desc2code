#include <cstdio>
#include <cstring>

const int maxn = 4010;
int L1, L2, ins, del, rpl, swp, f[maxn][maxn], last1[130], last2[130];
char s1[maxn], s2[maxn];

inline int min(int x, int y){return x < y ? x : y;}

int main(){
	scanf("%d%d%d%d\n", &ins, &del, &rpl, &swp);
	scanf("%s", s1 + 1); L1 = strlen(s1 + 1);
	scanf("%s", s2 + 1); L2 = strlen(s2 + 1);
	memset(f, 60, sizeof(f));
	for (int i = 0; i <= L1; i++) f[i][0] = i * del;
	for (int i = 0; i <= L2; i++) f[0][i] = i * ins;
	for (int i = 1; i <= L1; i++){
		memset(last2, 0, sizeof(last2));
		for (int j = 1; j <= L2; j++){
			f[i][j] = min(f[i][j], min(min(f[i - 1][j] + del, f[i][j - 1] + ins), f[i - 1][j - 1] + rpl * (s1[i] != s2[j])));
			int p1 = last1[s2[j]], p2 = last2[s1[i]];
			if (p1 && p2) f[i][j] = min(f[i][j], f[p1 - 1][p2 - 1] + swp + (i - p1 - 1) * del + (j - p2 - 1) * ins);
			last2[s2[j]] = j;
		}
		last1[s1[i]] = i;
	}
	printf("%d\n", f[L1][L2]);
}
