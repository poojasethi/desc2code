#include<stdio.h>
#include<math.h>
double Helper(double w,double v,double a,double l)
{
	return (l-(v*v-w*w)/(2*a))/v + (v-w)/a;
}
int main()
{
	int a,v,l,d,w;
	scanf("%d%d%d%d%d",&a,&v,&l,&d,&w);
	if(w>=v || w*w>=2*a*d)
	{
		if(v*v>=2*a*l)
			printf("%.7f",sqrt(2.*l/a));
		else
			printf("%.7f",Helper(0,v,a,l));
	}
	else
	{
		double t;
		if(v*v-w*w>=2*a*(l-d))
			t=(sqrt(2.*a*(l-d)+w*w)-w)/a;
		else
			t=Helper(w,v,a,l-d);
			
		if(2*v*v>=w*w+2*a*d)
			t+=(sqrt(1.*(w*w+2*a*d)/2)*2.-w)/a;
		else
			t+=1.*v/a+Helper(w,v,a,d-1.*v*v/(2*a));
		
		printf("%.7f",t);
	}
	return 0;
}
