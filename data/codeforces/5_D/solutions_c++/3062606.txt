#include <stdio.h>
#include <math.h>

int main()
{
	double time,vv;
	double a,v,l,d,w;
	scanf("%lf%lf",&a,&v);
	scanf("%lf%lf%lf",&l,&d,&w);
	if(w>=v)
	{
		if(a*(v/a)*(v/a)/2>=l) time=sqrt(2*l/a);
		else time=v/a+(l-a*(v/a)*(v/a)/2)/v;
	}
	else
	{
		if(a*(w/a)*(w/a)/2>=d)
		{
			time=sqrt(2*d/a);
			vv=time*a;
		}
		else
		{
			if(v*v-w*w>a*(d-a*(w/a)*(w/a)/2))
			{
				vv=sqrt((2*a*d+w*w)/2);
				time=vv/a+(vv-w)/a;
			}
			else time=v/a+(v-w)/a+(d-a*(v/a)*(v/a)/2-w*(v-w)/a-a*((v-w)/a)*((v-w)/a)/2)/v;
			vv=w;
		}
		if(vv*((v-vv)/a)+a*((v-vv)/a)*((v-vv)/a)/2>=l-d)
		{
			time+=(sqrt(vv*vv+2*a*(l-d))-vv)/a;
		}
		else
		{
			time+=(v-vv)/a+(l-d-vv*((v-vv)/a)-a*((v-vv)/a)*((v-vv)/a)/2)/v;
		}
	}
	printf("%.6f\n",time);
	return 0;
}
