#include <iostream>
#include <cmath>
#include <iomanip>

using namespace std;

double a, v, l, d, w, t;
double v1, v2, v3, ans, s;

int main() {
	cin >> a >> v >> l >> d >> w;

	v1 = min(min(v, w), sqrt(2 * a * d));
	t = v1 / a;
	ans = t;
	
	s  = d - a * t * t / 2;
	v2 = sqrt(a * s + v1 * v1);

	if (v2 <= v) ans += 2 * (v2 - v1) / a;
	else ans += 2 * (v - v1) / a + (s - (v * v - v1 * v1) / a) / v;

	s = l - d;
	v3 = sqrt(2 * a * s + v1 * v1);

	if (v3 <= v) ans += (v3 - v1) / a;
	else ans += (v - v1) / a + (s - 0.5 * (v * v - v1 * v1) / a) / v;

	cout << fixed << setprecision(5) << ans << endl;

	return 0;
}
