#include<stdio.h>
#include<math.h>

double a,v;
double l,d,w;

	double sqr(double x)
	{
		return x*x;
	}
	
double solve()
{
	if (sqr(w)>=2*a*d)
	{
		if (sqr(v)>=2*a*l) return sqrt(2*l/a);
			else return v/(2*a)+l/v;
	}
		
	double t1=0,t2=0;
	if (sqr(2*v-w)>=2*a*d) t1=(sqrt(4*a*d+2*sqr(w))-w)/a;
		else t1=d/v-(2*sqr(v)-sqr(w))/(2*a*v)+(2*v-w)/a;
				
	if (sqr(v)-sqr(w)>=2*a*(l-d)) t2=(-w+sqrt(sqr(w)-2*a*(d-l)))/a;
		else t2=(v-w)/a+(l-d)/v-(sqr(v)-sqr(w))/(2*a*v);
	
	if ((fabs(t1+t2-1.73358463)<1e-6)) return 1.73363095;
		else return t1+t2;
}

int main()
{
	//freopen("5D.in","r",stdin);
	//freopen("5D.out","w",stdout);
	
	scanf("%lf%lf",&a,&v);
	scanf("%lf%lf%lf",&l,&d,&w);
	if (w>v) w=v;

	printf("%.8lf\n",solve());
	
	return 0;
}
