#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
#include<vector>
#include<map>
#include<cmath>
using namespace std;
int main()
{
    double a,v,l,d,w,ans,t1,t2;
    cin>>a>>v>>l>>d>>w;
    if(w>=v||0.5*w*w/a>d)
    {
      if(0.5*v*v/a>l)
          ans=sqrt(2.0*l/a);
      else ans=l/v+0.5*v/a;
    }
    else 
    {
     if(0.5*(v*v-w*w)/a>(l-d))
     {
         t2=(sqrt(w*w+2*a*(l-d))-w)/a;
     }
     else 
     {
         t2=(l-d)/v-(v*v-w*w)/(2*a*v)+(v-w)/a;
     }
     if(v*v/a-0.5*w*w/a>d)
     {
         t1=(2*sqrt(a*(d+0.5*w*w/a))-w)/a;
     }
     else 
     {
         t1=v/a-w/a+d/v+w*w/(2*a*v);
     }
     ans=t1+t2;
    }
    printf("%.6lf\n",ans);
    return 0;
}
