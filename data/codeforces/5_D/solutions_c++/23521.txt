#include <stdio.h>
#include <math.h>

double calc(double a,double v,double &w,double d){
	if(w>v) w=v;
	double s=v*v/(2*a)+(v-w)*(v+w)/(2*a);
	if(s<=d){
		return (d-s)/v+v/a+(v-w)/a;
	}else if(d<=w*w/(2*a)){
		double t=sqrt(2*d/a);
		w=a*t;
		return t;
	}else{
		double ts=sqrt((2*d+w*w/a)/(2*a));
		return ts+(ts*a-w)/a;
	}
}

double calc2(double a,double v,double w,double d){
	double t=(v-w)/a;
	if(d<=t*(v+w)/2){
		return (sqrt(w*w+2*a*d)-w)/a;
	}
	return (d-t*(v+w)/2)/v+t;
}

int main(){
	double a,v,l,d,w;
	scanf("%lf%lf%lf%lf%lf",&a,&v,&l,&d,&w);
	double t1=calc(a,v,w,d);
	double t2=calc2(a,v,w,l-d);
	printf("%.9lf",t1+t2);
	scanf("%*s");
	return 0;
}
