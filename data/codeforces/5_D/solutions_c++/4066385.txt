#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <iostream>
using namespace std;
int main()
{
    double a,v,w,l,d;
    cin>>a>>v;
    cin>>l>>d>>w;
    double ans=0,v1,v2,v3;
    v1=min(v,w);
    v1=min(v1,sqrt(2*a*d));
    double t=v1/a;
    ans+=t;
    double s=d-0.5*a*t*t;
    v2=sqrt(v1*v1+s*a);
    if(v2<=v)
        ans+=2*(v2-v1)/a;
    else
        ans+=2*(v-v1)/a+(s-(v*v-v1*v1)/(a))/v;
    s=l-d;
    v3=sqrt(2*a*s+v1*v1);
    if(v3<=v)
        ans+=(v3-v1)/a;
    else
        ans+=(v-v1)/a+(s-(v*v-v1*v1)/(2*a))/v;
    printf("%.6lf\n",ans);
    return 0;
}
