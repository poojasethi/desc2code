#include<iostream>
#include<cmath>
using namespace std;
int a,v,l,d,w;
double ans;
int main(){
	cin>>a>>v>>l>>d>>w;
	if (w>=v||0.5*w*w/a>=d){
		if (0.5*v*v/a>l) ans=sqrt(2.0*l/a);
		else ans=(v+0.0)/a+(l-0.5*v*v/a)/v;
	}else{
		double r=d-0.5*w*w/a, t=(sqrt(w*w+a*r)-w+0.0)/a;
		if (w+a*t>v){
			t=(v-w+0.0)/a;
			ans=(w+0.0)/a+(v-w+0.0)/a*2+(r-2*(w*t+0.5*a*t*t)+0.0)/v;		
		}else ans=(w+0.0)/a+t+t;
		t=(v-w+0.0)/a; double dis=w*t+0.5*a*t*t;
		if (dis>l-d) ans+=(sqrt(w*w+2*a*(l-d))-w)/a;
		else ans+=(v-w+0.0)/a+(l-d-dis)/v;
	}
	cout.setf(ios::fixed); cout.precision(12);
	cout<<ans<<endl;
}
