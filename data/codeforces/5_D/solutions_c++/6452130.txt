#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

double a,b,l,d,w,rs=0;

int main(){
    cin >> a >> b >> l >> d >> w;
    w=min(b,w);
    double s1,s2,s3,u;
    s1=b*b/a/2;
    s2=w*w/a/2;
    s3=s1-s2;
    if (s2>d){
        w=sqrt(2*a*d);
        rs+=w/a;
    }else if (s1+s3>d){
        u=sqrt(a*d+w*w/2);
        rs+=u/a;
        rs+=(u-w)/a;
    }else{
        rs+=b/a;
        rs+=(b-w)/a;
        rs+=(d-s1-s3)/b;
    }
    d=l-d;
    s3=(b*b-w*w)/a/2;
    if (d<s3){
        u=sqrt(2*a*d+w*w);
        rs+=(u-w)/a;
    }else{
        rs+=(b-w)/a;
        rs+=(d-s3)/b;
    }




    cout << setprecision(13) << rs;

return 0;
}
