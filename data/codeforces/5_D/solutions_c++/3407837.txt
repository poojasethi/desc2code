#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
double a[5],ans,v1,s1,t1,v2;
int main(){
    for(int i=0;i<5;i++) scanf("%lf",&a[i]);
    v1 = min(a[4],min(a[1],sqrt(2.0*a[0]*a[3])));
    s1 = v1*v1/(2.0*a[0]);
    ans = v1/a[0];
    v2 = sqrt(a[0]*(a[3]-s1)+v1*v1);
    if(v2 > a[1])
        ans += 2.0*(a[1]-v1)/a[0]+((a[3]-s1)-(a[1]*a[1]-v1*v1)/a[0])/a[1];
    else 
        ans += 2.0*(v2-v1)/a[0];
    v2 = v1;
    if((a[1]*a[1]-v2*v2)/(2.0*a[0])>a[2]-a[3])
        ans += (sqrt(v2*v2+2.0*a[0]*(a[2]-a[3]))-v2)/a[0];
    else 
        ans += (a[1]-v2)/a[0]+(a[2]-a[3]-(a[1]*a[1]-v2*v2)/(2.0*a[0]))/a[1];
    printf("%.12lf",ans);
}
