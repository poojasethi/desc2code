#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

double a,v,l,d,w,rs=0;

int main(){
	cin >> a >> v >> l >> d >> w;
	w=min(v,w);
	double s1,s2,s3,u;
	s1=v*v/a/2;
	s2=w*w/a/2;
	s3=s1-s2;
	if (s2>d){
		w=sqrt(2*a*d);
		rs+=w/a;
	}else if (s1+s3>d){
		u=sqrt(a*d+w*w/2);
		rs+=u/a;
		rs+=(u-w)/a;
	}else{
		rs+=v/a;
		rs+=(v-w)/a;
		rs+=(d-s1-s3)/v;
	}
	d=l-d;
	s3=(v*v-w*w)/a/2;
	if (d<s3){
		u=sqrt(2*a*d+w*w);
		rs+=(u-w)/a;
	}else{
		rs+=(v-w)/a;
		rs+=(d-s3)/v;
	}




	cout << setprecision(13) << rs;

return 0;
}
