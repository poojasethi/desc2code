#include <cstdio>
#include <cmath>

using namespace std;

double A, V, L, D, W;

double dist(double v0, double t) {
    return v0*t + A*t*t*.5;
}

double travelTime(double D, double v0) {
    double tmax = (-v0 + sqrt(v0*v0 + 2*A*D)) / A;
    double tV = (V - v0) / A;
    if (tV >= tmax) return tmax;
    return tV + (D-dist(v0,tV)) / V;
}

int main() {

    scanf("%lf %lf %lf %lf %lf", &A, &V, &L, &D, &W);
    if (V <= W) { printf("%.10lf\n", travelTime(L, 0)); return 0; }
    double tw = W / A;
    double dw = dist(0, tw);
    if (dw >= D) { printf("%.10lf\n", travelTime(L, 0)); return 0; }
    printf("%.10lf\n", tw + travelTime(.5 * (D-dw), W) * 2. + travelTime(L-D, W));

    return 0;
}
