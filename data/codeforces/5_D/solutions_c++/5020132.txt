#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
#include <cmath>
using namespace std;

int a, v, l, d, w;

void DirectGO()
{
	if (2*a*l <= v*v)
		printf("%.9lf\n", sqrt(2.0*l/a));
	else
		printf("%.9lf\n", 1.0*v/a + (l-0.5*v*v/a)/v);
	exit(0);
}
double HalfGO()
{
	if ((v*v-w*w) >= (l-d)*a*2)
		return (-w+sqrt(w*w+2.0*a*(l-d))) / a;
	return (double)(v-w)/a + (l-d-(double)(v*v-w*w)/(a*2))/v;
}
int main()
{
	//freopen("input.txt", "r", stdin);
	scanf("%d%d%d%d%d%d", &a, &v, &l, &d, &w);
	if (v <= w || 2*a*d <= w*w) DirectGO();
	double t0, t = sqrt((double)(w*w+2*a*d)/(a*a*2));
	if (a*t <= v)
		t0 = t+(a*t-w)/a;
	else
		t0 = (d-(double)(v*v)/a/2-(double)(v*v-w*w)/a/2)/v + (double)v/a + (double)(v-w)/a;
	printf("%.9lf\n", t0 + HalfGO());
	return 0;
}
