#include <stdio.h>
#include <iostream>
#include <cmath>

using namespace std;

int main()
{
	double a, v;
	cin>>a>>v;
	double l, d, w;
	cin>>l>>d>>w;
	double ans;
	if (w >= v || 0.5*w*w/a > d) {
		if (0.5*v*v/a > l) ans = sqrt(2*l/a);
		else ans = l/v + 0.5*v/a;
	}
	else {
		double t1, t2;
		if (0.5*(v*v-w*w)/a > (l-d)) {
			t2 = (sqrt(w*w+2*a*(l-d))-w)/a;
		}
		else {
			t2 = (l-d)/v - (v*v-w*w)/(2*a*v) + (v-w)/a;
		}
		if (v*v/a - 0.5*w*w/a > d) {
			double vx = sqrt(a*(d+0.5*w*w/a));
			t1 = (2 * vx - w) / a;
		}
		else {
			t1 = v/a - w/a + d/v + w*w/(2*a*v);
		}
		ans = t1 + t2;
	}

//	cout<<ans;
	printf("%.6f\n", ans);

	return 0;
}
