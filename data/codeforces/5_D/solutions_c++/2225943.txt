#include<stdio.h>
#include<math.h>
int main()
{
	double a,v,l,d,w;
	scanf("%lf%lf%lf%lf%lf",&a,&v,&l,&d,&w);
	double ss=w*w/a/2.0,t,s1,t1;
	if(w>=v)
	{
		t1=v/a;
		s1=v*v/a/2.0;
		if(s1<=l)
		{
			t=t1+(l-s1)/v;
		}
		else
		{
			t=sqrt(l*2.0/a);
		}
	}
	else if(ss>=d)
	{
		t1=v/a;
		s1=v*v/a/2;
		if(s1<=l)
		{
			t=t1+(l-s1)/v;
		}
		else
		{
			t=sqrt(l*2.0/a);
		}
	}
	else
	{
		s1=(v*v-w*w)/2.0/a;
		if(s1>=d-v*v/a/2.0)
		{
			t1=(-w+sqrt(w*w+(d-ss)*a))*2.0/a+w/a;
		}
		else
		{
			t1=(d-s1-v*v/2.0/a)/v+v/a+(v-w)/a;
		}
		if(s1<=l-d)
		{
			t=t1+(l-d-s1)/v+(v-w)/a;
		}
		else
		{
			t=t1+(-w+sqrt(w*w+(l-d)*2.0*a))/a;
		}
	}
	printf("%.12lf\n",t);
}
