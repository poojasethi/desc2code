#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>

using namespace std;

int main()
{
	double a,v,l,d,w,t=0;
	scanf("%lf%lf%lf%lf%lf",&a,&v,&l,&d,&w);
	if(v<=w)
	{
		if(v*v<=2*a*l)
			t+=v/a+(l-v*v/2/a)/v;
		else
			t+=sqrt(2*l/a);
	}
	else
	{
		if(w*w>=2*a*d)
		{
			if(v*v<=2*a*l)
				t+=v/a+(l-v*v/2/a)/v;
			else
				t+=sqrt(2*l/a);
		}
		else
		{
			double x=sqrt((2*a*d+w*w)/2);
			if(x<=v)
				t+=x/a+(x-w)/a;
			else
				t+=v/a+(v-w)/a+(d-v*v/2/a-(v*v-w*w)/2/a)/v;
			if((v*v-w*w)>=2*a*(l-d))
				t+=(sqrt(w*w+2*a*(l-d))-w)/a;
			else
				t+=(v-w)/a+(l-d-(v*v-w*w)/2/a)/v;
		}
	}
	printf("%.6lf\n",t);
}
