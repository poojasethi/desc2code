#include <cstdio>
#include <cmath>

double a, v, l, d, w;

double calc(double v0, double s) {
	double smax = (v*v - v0*v0) / 2 / a;
	if (smax > s)
		return (-v0 + sqrt(v0 * v0 + 2*a*s)) / a;
	else
		return (s-smax) / v + (v-v0) / a;
}

int main() {
//	freopen("input.txt", "r", stdin);
	
	scanf("%lf%lf%lf%lf%lf", &a, &v, &l, &d, &w);
	
	if (v <= w || sqrt(2*a*d) <= w) {
		printf("%.5lf\n", calc(0.0, l));
	} else {
		double s1 = v * v / 2 / a, s2 = (v*v - w*w) / 2 / a;
		double t;
		if (s1 + s2 < d)
			t = (v*2-w) / a + (d - s1 - s2) / v;
		else
			t = sqrt(d / a + w * w / a / a / 2) * 2 - w / a;
		t += calc(w, l - d);
		printf("%.5lf\n", t);
	}
	
//	for(;;);
	return 0;
}
