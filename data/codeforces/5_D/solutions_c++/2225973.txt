#include<stdio.h>
#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    double a,v,w,l,d,mv,t,ds,vv,ta,dd,taa;
    while(~scanf("%lf%lf%lf%lf%lf",&a,&v,&l,&d,&w)) 
	{
        mv=min(v,w);ta=mv/a;ds=mv*ta*0.5;
        if(ds>d) t=sqrt(2*d/a),vv=a*t;
        else
        {
        	taa=(v-mv)/a;ta=v/a;ds=v*ta*0.5;
        	dd=ds+taa*mv+0.5*taa*(v-mv);
        	if(dd<d) t=ta+taa+(d-dd)/v;
        	else 
        	{
	        	ta=mv/a;
	        	ds=d-0.5*ta*mv;
	        	t=ta+(sqrt(mv*mv+a*ds)-mv)/(0.5*a);
	        }
        	vv=mv;
        }	
        d=l-d;ta=(v-vv)/a;ds=vv*ta+0.5*a*ta*ta;
        if(ds>d) t+=(sqrt(vv*vv+4*0.5*a*d)-vv)/a;
        else t+=ta+(d-ds)/v;
        printf("%.12f\n",t);
    }
}
