#include <cstdio>
#include <cmath>

using namespace std;

double a,v,l,d,w,vmax,t1,t2,s,s1,s2;

int main(){
	scanf("%lf%lf%lf%lf%lf",&a,&v,&l,&d,&w);
	if (w>v) w=v;
	if (w*w/2/a>=d){
		if (v*v/a/2>l) printf("%.9lf\n",sqrt(2*l/a));
		else printf("%.9lf\n",v/a+(l-v*v/2/a)/v);
		return 0;
	}
	double vmax=sqrt(a*d+w*w/2);
	if (vmax<=v){
		t1=vmax/a+(vmax-w)/a;
		s=(v*v-w*w)/2/a;
		if (s<=l-d){
			t2=(v-w)/a+(l-d-s)/v;
			printf("%.9lf\n",t1+t2);
		}
		else printf("%.9lf\n",t1+(-2*w+sqrt(4*w*w-8*a*(d-l)))/2/a);
		return 0;
	}
	s1=d-(2*v*v-w*w)/2/a;
	t1=s1/v+v/a+(v-w)/a;
	s2=(v*v-w*w)/2/a;
	if (s2<=l-d){
		t2=(v-w)/a+(l-d-s2)/v;
		printf("%.9lf\n",t1+t2);
	}
	else printf("%.9lf\n",t1+(-2*w+sqrt(4*w*w-8*a*(d-l)))/2/a);
	return 0;
}
