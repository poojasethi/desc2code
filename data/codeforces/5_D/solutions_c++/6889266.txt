#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main() {
  int a, v, l, d, w;
  double t, p;
  cin >> a >> v;
  cin >> l >> d >> w;
  if (v<=w || 2*d*a<=w*w) {
    if (2*l*a <= v*v) {
      t = sqrt(2.*l/a);
    } else {
      t = (double)l/v + (double)v/a*0.5;
    }
  } else {
    if (w*w + 2*a*d <= 2*v*v) {
      t = (sqrt(2.*w*w + 4.*a*d) - w)/a;
    } else {
      t = (double)(2*v-w)/a + (d - 0.5*(2*v*v - w*w)/a)/v;
    }
    if (w*w + 2*a*(l-d) <= v*v) {
      t += (sqrt(w*w + 2.*a*(l-d)) - w)/a;
    } else {
      t += (double)(v - w)/a + (l - d - 0.5*(v*v - w*w)/a)/v;
    }
  }
  cout << fixed << setprecision(12) << t << endl;
  return 0;
}
