#include<iostream>
#include<iomanip>
#include<cstring>
#include<cmath>
using namespace std;

long double a,v,l,d,w,ans,t1,v1,t2;

int main()
{
	//freopen("yoshi.in","rb",stdin);
	//freopen("yoshi.out","wb",stdout);
	cin>>a>>v>>l>>d>>w;
	if(w>v)w=v;
	l-=d;
	t1=sqrt(2*d/a);
	if(t1*a<=w)
		l+=d;
	else
	{
		t1=sqrt((2*a*d+w*w)/(2*a*a));
		if(t1*a<=v)
		{
			ans+=2*t1-w/a;
			v1=w;
		}
		else
		{
			t1=v/a;
			t2=(v-w)/a;
			d-=t1*t1*a/2;
			d-=(w+v)/2*t2;
			ans+=d/v+t1+t2;
			v1=w;
		}
	}
	t1=(sqrt(v1*v1+2.0*a*l)-v1)/a;
	if(t1*a+v1>v)
	{
		t1=(v-v1)/a;
		l-=t1*(v1+v)/2;
		ans+=l/v+t1;
	}
	else ans+=t1;
	cout<<fixed<<setprecision(6)<<ans<<endl;
	return 0;
}
