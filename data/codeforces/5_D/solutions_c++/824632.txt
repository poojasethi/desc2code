#include <math.h>
#include <stdio.h>
using namespace std;

double a; 
double ans = 0;
double d;
double l;
double v;
double w;
double tt;
double tv;
double tv2;

int main()
{
	scanf("%lf%lf%lf%lf%lf", &a, &v, &l, &d, &w);
	if (w > v) w = v;
	tt = w / a;
	l -= d;
	if (tt * tt * a <= 2 * d)
	{
		d -= tt * tt * a / 2;
		ans += tt;
		tv = a * tt;
		tv2 = sqrt(tv * tv + a * d);
		if (tv2 <= v)
			ans += (tv2 - tv) / a * 2;
		else
			ans += (v - tv) / a * 2 + (d - (v - tv) * (v + tv) / a) / v;
		tv = w;
	}
	else
	{
		tt = sqrt(2 * d / a);
		ans += tt;
		tv = a * tt;
	}
	tt = (v - tv) / a;
	if (tt * (v + tv) < 2 * l)
	{
		l -= tt * (v + tv) / 2;
		ans += tt;
		ans += l / v;
	}
	else
	{
		tt = (-tv + sqrt(tv * tv + 2 * a * l)) / a;
		ans += tt;
	}
	printf("%.5lf\n", ans);
	return 0;
}
