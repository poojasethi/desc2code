#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);

	long double a,v,l,d,w;
	cin>>a>>v>>l>>d>>w;

	long double v1;
	v1=sqrt(2*d*a);

	long double T=0;
	if(v<=w || v1<=w)
	{
		long double s;
		s= v*v/(2*a);
		if(s<l)
			T+=2*s/v+(l-s)/v;
		else T+=sqrt(2*a*l)/a;
	}
	else
	{
		v1=sqrt((2*d*a+w*w)/2);
		if(v1>v) v1=v;

		long double s1,s2;
		s1= v1*v1/(2*a);
		s2= (v1*v1-w*w)/(2*a);

		T=2*s1/v1+2*s2/(v1+w)+(d-s1-s2)/v;

		long double s3;
		s3= (v*v-w*w)/(2*a);
		if(s3<l-d)
			T+=2*s3/(v+w)+(l-d-s3)/v;
		else T+=(sqrt(2*a*(l-d)+w*w)-w)/a;
	}

	cout.precision(10);
	cout<<T;
	return 0;
}
