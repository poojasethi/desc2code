import math
def getfloats(): return map(float, raw_input().split())

def calc(v0, v, a, x):
	t = (v - v0) / a
	x0 = v0 * t + 0.5 * a * t * t
	if x0 > x: return (x, (math.sqrt(v0 * v0 + 2 * a * x) - v0) / a)
	return (x0, t)

def go(v0, v, a, x):
	x0, t = calc(v0, v, a, x)
	return t + (x - x0) / v

a, v = getfloats()
l, d, w = getfloats()
if w > v: w = v

x, t = calc(0, w, a, d)
if x == d: print go(0, v, a, l)
else: print t + go(w, v, a, (d - x) * 0.5) * 2 + go(w, v, a, l - d)
