#include <bits/stdc++.h>
#define MAXN 1000010
using namespace std;

int a[MAXN],b[MAXN],dir[MAXN];
vector<int> ans;
int n,c;
void func(int l,int r)
{
    if(l>r) return;
    if(l==r){
        ans.push_back(l);
        return;
    }
    //split at i
    for(int split=l;split<=r;++split){
        bool flag=true;
        for(int j=0;j<c;++j){
            if(a[j]>l && b[j]<=r){
                if(a[j]<=split && b[j] > split){
                    flag=false;
                    break;
                }
            }
            if(a[j]==l){
                if(dir[j] == 1 && b[j]>split){
                    flag=false;
                    break;
                }
                if(dir[j] == 2 && b[j]<=split){
                    flag=false;
                    break;
                }
            }
        }
        if(flag){
            func(l+1,split);
            ans.push_back(l);
            func(split+1,r);
            return;
        }
    }
    puts("IMPOSSIBLE");
    exit(0);
}
int main()
{
    string s;
    scanf("%d %d",&n,&c);
    for(int i=0;i<c;++i)
    {
        scanf("%d %d",&a[i],&b[i]);
        cin>>s;
        if(a[i]>=b[i]){
            puts("IMPOSSIBLE");
            return 0;
        }
        if(s[0]=='L'){
            dir[i]=1;
        }
        else{
            dir[i]=2;
        }
    }
    func(1,n);
    for(int i=0;i<n;++i) printf("%d ",ans[i]);
    return 0;
}
