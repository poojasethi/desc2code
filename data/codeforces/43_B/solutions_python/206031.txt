h = raw_input()
t = raw_input()
d={}
for c in h:
  if c!=" ":
    if d.has_key(c):
      d[c]+=1
    else:
      d[c]=1
poss = True
for c in t:
  if c!=" ":
    if not d.has_key(c) or d[c]==0:
      poss = False
      break
    else:
      d[c]-=1
if poss == True:
  print "YES"
else:
  print "NO"
h = "hi"
