T1 = raw_input().strip()
T2 = raw_input().strip()
L = [0]*512
for c in T1:
    if ord(c) != ord(' '):
        L[ord(c)] = L[ord(c)] + 1
for c in T2:
    if ord(c) != ord(' '):
        L[ord(c)] = L[ord(c)] - 1
chk = 1
for i in range(512):
    if L[i] < 0:
        chk = 0
if chk:
    print "YES"
else:
    print "NO"
