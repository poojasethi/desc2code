#!/usr/bin/env python2.6

import sys

S1=sys.stdin.readline()
S2=sys.stdin.readline()

C1={}

for i in range(0,len(S1)):
    k=S1[i]
    if k==' ': continue
    C1[k]=C1.get(k,0)+1

for i in range(0,len(S2)):
    k=S2[i]
    if k==' ': continue
    C1[k]=C1.get(k,0)-1
    if C1[k]<0:
        print('NO')
        exit()

print('YES')
