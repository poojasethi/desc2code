#!/usr/bin/python
import sys
import re
from collections import *

s1 = raw_input()
s2 = raw_input()

def make_counter(s):
    return Counter("".join(s.split()))

c1 = make_counter(s1)
c2 = make_counter(s2)

# check for 'c2 in c1'
for i in c2:
    if c2[i] > c1[i]:
        print "NO"
        exit()

print "YES"
