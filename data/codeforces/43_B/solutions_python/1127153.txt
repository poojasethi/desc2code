from collections import defaultdict
s1 = raw_input()
s2 = raw_input()
d = defaultdict(int)

tag = True

for ch in s1:
    if ch != ' ':
        d[ch] += 1

for ch in s2:
    if ch != ' ':
        d[ch] -= 1
        if d[ch] < 0:
            tag = False
            break

print 'YES' if tag else 'NO'
