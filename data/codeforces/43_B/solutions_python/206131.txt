import sys
ltr = raw_input()
b = raw_input()

ltr = ''.join(ltr.split())
b = ''.join(b.split())
ltr = [i for i in ltr]
b = [i for i in b]
ok = True
for i in range(len(b)):
    if not (b[i] in ltr):
        ok = False
        break
    else:
        ltr.remove(b[i])
if ok:
    print "YES"
else:
    print "NO"
