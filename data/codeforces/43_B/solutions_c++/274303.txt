#include<iostream>
#include<string>
using namespace std;
int a[300];
string s;
int main()
{
	getline(cin,s);
	for(int i=s.length()-1;i>=0;i--)
		a[int(s[i])]++;
	getline(cin,s);
	for(int i=s.length()-1;i>=0;i--)
		if(s[i]!=' ')
		a[int(s[i])]--;
	for(int i=0;i<300;i++)
		if(a[i]<0)
		{
			cout<<"NO"<<endl;
			return 0;
		}
	cout<<"YES"<<endl;
	return 0;
}
