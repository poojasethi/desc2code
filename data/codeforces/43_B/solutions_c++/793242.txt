#include<cstdio>
int cnt[1<<8];
char s[201],t[201];
int main()
{
    int i;
    bool f=1;
    gets(s);
    for(i=0;s[i];i++) cnt[s[i]]++;
    gets(t);
    cnt[' ']=300;
    for(i=0;t[i];i++)
    {
        if(--cnt[t[i]]<0)
        {
            f=0;break;
        }
    }
    puts(f?"YES":"NO");
    return 0;
}
