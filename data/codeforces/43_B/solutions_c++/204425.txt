#include"cstdio"
#include"map"
using namespace std;
int main(){
    map<char,int>m;
    char c;
    while((c=getchar())!='\n'){
        m[c]++;
    }
    while((c=getchar())!='\n'){
        if(c==' ')continue;
        if(m[c]>0)m[c]--;
        else {puts("NO");return 0;}
    }
    puts("YES");
    return 0;
}
