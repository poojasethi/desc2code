#include <cstdio>

const int MAXN=200;

int occ[256];
char s1[MAXN+1],s2[MAXN+1];

int main(){
	gets(s1);
	gets(s2);
	for (int i=0;s1[i];i++)
		occ[(int)s1[i]]++;
	bool ok=true;
	for (int i=0;s2[i];i++){
		occ[(int)s2[i]]--;
		if (occ[(int)s2[i]]<0 && s2[i]!=' ')
			ok=false;
	}
	puts(ok ? "YES" : "NO");
	return 0;
}
