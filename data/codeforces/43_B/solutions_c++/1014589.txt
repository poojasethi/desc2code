#include <iostream>
using namespace std;

string s, t;

int main() {
 
 getline(cin, s);
 getline(cin, t);
  for(int i = 0; i < t.size(); i++) {
    if(t[i] != ' ') {
     int pos =  s.find(t[i]);
     if(pos == -1) {
      cout << "NO";
      return 0;
     }
     else s[pos] = '!';
    }
  }
   cout << "YES";
  return 0;
}
