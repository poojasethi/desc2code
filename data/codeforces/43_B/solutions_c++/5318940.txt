
#include <stdio.h>
int main(void) {
    char a[500];
    int c[128]={};
    char b[500];
    int i;
    gets(a);
    for(i=0;a[i];i++)
        c[a[i]]++;
    gets(b);
    int flag=0;
    for(i=0;b[i];i++)
        if(b[i]!=' ' && --c[b[i]]<0){flag=1;break;}
    puts(flag?"NO":"YES");
    return 0;
}
