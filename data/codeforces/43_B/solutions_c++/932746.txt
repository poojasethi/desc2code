#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{ char s1[1001],s2[1001];
  gets(s1);
  gets(s2);
  int k1[256]={0},k2[256]={0};
  for (int i=0; s1[i];i++) k1[s1[i]]++;
  for (int i=0; s2[i];i++) k2[s2[i]]++;
  bool f=true;
  for (int i=33;i<=255;i++) if (k2[i]>k1[i]) f=false;
  if (f) cout<<"YES"; else cout<<"NO";
  return 0;
}
