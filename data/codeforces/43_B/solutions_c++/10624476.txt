#include <iostream>
using namespace std;
int main(){
string a,b;
int s=0;
getline(cin,a);
getline(cin,b);
for(int i=0;i<b.size();i++){
for(int i2=0;i2<a.size();i2++){
if(b[i]==' '){
s+=1;
break;
}
if(b[i]==a[i2]){
s+=1;;
a[i2]='0';
break;
}
}
if(s>=b.size()){
cout<<"YES";
return 0;
}
}
cout<<"NO";
return 0;
}
