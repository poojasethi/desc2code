#include<bits/stdc++.h>
using namespace std;
char A[500], B[500];
int H[1000];
int main(){
	gets(A);
	gets(B);

	for(int i=0;i<strlen(A); i++)
		H[A[i]]++;

	for(int i=0;i<strlen(B);i++)
		if(B[i] != ' ')
			H[B[i]]--;

	for(int i=1;i<=500;i++)
		if(H[i] < 0){
			cout << "NO" << endl;
			return 0;
		}
	cout << "YES"<<endl; 
	
}
