#include<iostream>
using namespace std;
int main(){
  int i,a[200];
  for(i=0;i<200;i++)a[i]=0;
 string s,l;
 getline(cin,s);
 getline(cin,l);
  for(i=0;i<s.length();i++){
    a[s[i]]++;
  }
  for(i=0;i<l.length();i++){
    if(a[l[i]]<1 && l[i]!=' ')
   {
     cout<<"NO";
     return 0;
}
    a[l[i]]--;
  }
  cout<<"YES";
  return 0;
}
