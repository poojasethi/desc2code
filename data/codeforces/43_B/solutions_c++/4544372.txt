#include<bits/stdc++.h>
int main()
{
	int count[2][CHAR_MAX]={0},i;
	char temp;
	for (i=0;i<2;i++)
	{
		while (1)
		{
			scanf("%c",&temp);
			if (temp=='\n')
				break;
			if (temp==' ')
				continue;
			count[i][temp]++;
		}
	}
	for (i=0;i<CHAR_MAX;i++)
		if (count[0][i]<count[1][i])
		{
			puts("NO");
			return 0;
		}
	puts("YES");
	return 0;
}
