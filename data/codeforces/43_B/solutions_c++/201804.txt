#include <iostream>
#include <map>
#include <cstdio>
using namespace std;

int main(void){
	map<char,int> count;
	char ch;

	while((ch=getchar()) != '\n'){
		count[ch]++;
	}

	while(cin>>ch){
		if(count[ch] > 0){
			count[ch]--;
		}
		else{
			cout<<"NO\n";
			return 0;
		}
	}
	cout<<"YES\n";

	return 0;
}
