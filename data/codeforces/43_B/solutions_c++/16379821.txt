#include <bits/stdc++.h>
using namespace std;

char a[256], b[256];
int c[256], d;

int main () {
  gets(a);
  gets(b);
  for (int i = 0; a[i]; i++) {
    if (a[i] != ' ') c[a[i]]++;
  }
  for (int i = 0; b[i] && d == 0; i++) {
    if (b[i] != ' ') c[b[i]]--;
    if (c[b[i]] < 0) d = 1;
  }
  printf("%s\n", d ? "NO" : "YES");
	return 0;
}
