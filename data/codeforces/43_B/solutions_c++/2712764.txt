#include<stdio.h>
#include<stdlib.h>
int main()
{
	
	char a[201],b[1000];

	gets(b);
	gets(a);
	int i,j,t=0,k=0;
    for(i=0;a[i]!='\0';i++)
	{
		if(a[i]!=' ')
		k++;
		for(j=0;b[j]!='\0';j++)
			if(a[i]==b[j]&&a[i]!=' ')
			{	t++;
			b[j]=' ';
			break;
			}
	}
		if(t==k)
			printf("YES\n");
		else
			printf("NO\n");
			
			return 0;
}
