#include <cstdio>
#include <cstring>

int main(){
	char ch;
	
	int cnt[10000];
	
	while ((ch = getchar()) != '\n'){
		if (ch == ' ') continue;
		cnt[(int)ch]++;
	}
	
	while ((ch = getchar()) != '\n'){
		if (ch == ' ') continue;
		if (cnt[(int)ch] == 0) {
			printf("NO\n");
			return 0;
		}
		else cnt[(int)ch]--;
	}
	
	printf("YES\n");
	return 0;
}
