#include<cstdio>
#include<iostream>
#include<string>
using namespace std;

int chars[257];
string s1,s2;

int main(){
 getline(cin,s1);
 getline(cin,s2);
 int i;
 for(i=0;i<s1.length();i++){
  chars[s1[i]]++;
 }
 for(i=0;i<s2.length();i++){
  if(s2[i]!=' ') chars[s2[i]]--;
  if(chars[s2[i]]<0){
   printf("NO\n");
   return 0;
  }
 }
 printf("YES\n");
 return 0;
}
