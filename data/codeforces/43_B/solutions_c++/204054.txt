#include <iostream>
#include <cstdio>
using namespace std;

int A[255], B[255];
char c;

int main(){
	while (scanf("%c", &c)==1 && c!='\n') {
		if (c==' ') continue;
		A[c]++;
	}
	
	while (scanf("%c", &c)==1 && c!='\n') {
		if (c==' ') continue;
		//cout << c << endl;
		A[c]--;
		if (A[c] < 0) {
			cout << "NO" << endl;
			return 0;
		}
	}
	cout << "YES" << endl;
}
