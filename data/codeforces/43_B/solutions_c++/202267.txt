#include<cstdio>
#include<cstring>

char S1[201],S2[201];

int main(){
	char* j;
	bool can=1;

	gets(S1);
	gets(S2);
	int L2=strlen(S2);
	for(int i=0;i<L2;i++){
		if(S2[i]==' ')
			continue;
		if((j=strchr(S1,S2[i])))
			*j='-';
		else{
			can=0;
			break;
		}
	}

	printf("%s\n",can?"YES":"NO");

	return 0;
}
