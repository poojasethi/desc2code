

#include <cstdio>
#include <cstring>
#include <cctype>
#include <utility>
#include <algorithm>
#include <vector>
#include <map>
#include <set>
using namespace std;

typedef pair<int,int> pii;

pii d[10240];

set<pii> ans;
void gao(int l,int r)
{
    if (l == r)
        return ;
    int mid = (l+r)/2;
    for (int i = l; i <= r; ++i)
            ans.insert(pii(d[mid].first,d[i].second));
    gao(l, mid);
    gao(mid+1,r);
}


int main()
{
    int n;
    scanf("%d",&n);
    for (int i = 0; i < n; ++i){
        scanf("%d%d",&d[i].first,&d[i].second);
        ans.insert(d[i]);
    }
    sort(d,d+n);

    gao(0,n-1);
    printf("%d\n",ans.size());
    for (set<pii>::iterator it = ans.begin(); it != ans.end(); ++it)
        printf("%d %d\n",it->first,it->second);
    
}
