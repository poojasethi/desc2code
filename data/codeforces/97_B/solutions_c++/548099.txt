#include<cstdio>
#include<set>
#include<algorithm>

using namespace std;

set<pair<int,int> > A;
pair<int,int> P[10002];
int N,a,b;

void Work( int l, int r){
    if (l==r) return ;
    else {
        int m=(l+r)>>1;
        Work(l,m);
        Work(m+1,r);
        for (int i=l;i<=r;i++) A.insert(make_pair(P[m].first,P[i].second));
    }
}

int main(){
    scanf("%d",&N);
    for (int i=1;i<=N;i++){
        scanf("%d%d",&a,&b);
        P[i]=make_pair(a,b);
        A.insert(P[i]);
    }
    sort(P+1,P+N+1);
    Work(1,N);
    printf("%d\n",A.size());
    for (set<pair<int,int> >::iterator i=A.begin();i!=A.end();i++) printf("%d %d\n",i->first,i->second);
}
