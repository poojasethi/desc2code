#include <bits/stdc++.h>

using namespace std;

const int MAXN = 100*100 + 3;
int n;
pair <int, int> a[MAXN];
set <pair <int, int> > s;

void sups (int st, int e){
    if (st == e + 1 || st == e)
        return;

    int m = (st + e) / 2;

    for (int i = st; i <= e; i++)
        s.insert (make_pair (a[m].first, a[i].second) );
    
    sups (st, m);
    sups (m + 1, e);

}

int main(){
    cin >> n;
    for (int i = 0; i < n; i++){
        cin >> a[i].first >> a[i].second;
        s.insert (a[i]);
    }
    
    sort (a, a + n);

    sups (0, n-1);

    cout << s.size() << endl;
    for (set< pair <int, int> > :: iterator it = s.begin(); it != s.end(); it++)
        cout << it -> first << " " << it -> second << endl;


    return 0;   
}
