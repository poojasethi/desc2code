#include<cstdio>
#include<utility>
#include<cstdlib>
#include<set>
#include<algorithm>
using namespace std;

int n;
pair<int, int> a[10010];
set<pair<int, int> > poi;

void work(int l, int r)
{
	int mid = (l + r) >> 1;
	for (int j = l; j <= r; ++j)
		poi.insert(make_pair(a[mid].first, a[j].second));
	if (l < mid) work(l, mid);
	if (mid + 1 < r) work(mid + 1, r);
}

int main()
{
//	freopen("b.in","r",stdin);
	scanf("%d",&n);
	int x, y;
	poi.clear();
	for (int i = 1; i <= n; ++i)
	{
		scanf("%d%d",&x,&y);
		a[i] = make_pair(x,y);
		poi.insert(a[i]);
	}
	sort(a + 1, a + n + 1);
	if (1 < n) work(1, n);
	printf("%d\n",poi.size());
	for (set<pair<int, int> >::iterator i = poi.begin(); i != poi.end(); ++i)
		printf("%d %d\n",i->first,i->second);
	return 0;
}
