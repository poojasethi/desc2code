#include <cstdio>
#include <algorithm>
#include <set>

using namespace std;

struct point
{
  int x,y;
  bool operator<(const point& a) const
  {
    if(x!=a.x)
      return x<a.x;
    return y<a.y;
  }
};

set<point> ans;
point p[10001];

void sol(int l,int r)
{
  if(l==r)
    return;
  int mid=(l+r)/2;
  for(int i=l;i<=r;++i)
  {
    point t;
    t.x=p[mid].x;
    t.y=p[i].y;
    ans.insert(t);
  }
  if(l<mid)
    sol(l,mid);
  if(mid+1<r)
    sol(mid+1,r);
}

int main(void)
{
  int n;
  scanf("%d",&n);
  for(int i=1;i<=n;++i)
  {
    scanf("%d %d",&p[i].x,&p[i].y);
    ans.insert(p[i]);
  }
  sort(p+1,p+n+1);
  sol(1,n);
  printf("%d\n",(int)ans.size());
  for(set<point>::iterator i=ans.begin();i!=ans.end();++i)
    printf("%d %d\n",(*i).x,(*i).y);
  return 0;
}
