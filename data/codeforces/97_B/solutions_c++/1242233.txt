#include <iostream>
#include <set>
#include <algorithm>
using namespace std;
typedef pair<int, int> pii;
pii arr[10010];
set<pii> ans;
void split(int low, int high) {
        if (low + 1 == high) return;
        int mid = (low + high) / 2;
        for (int i = low; i < high; i++) {
                ans.insert(pii(arr[mid].first, arr[i].second));
        }
        split(low, mid);
        split(mid, high);
}
int main() {
        int n;
        cin >> n;
        for (int i = 0; i < n; i++) {
                cin >> arr[i].first >> arr[i].second;
                ans.insert(arr[i]);
        }
        sort(arr, arr + n);
        split(0, n);
        cout << ans.size() << '\n';
        for (set<pii>::iterator it = ans.begin(); it != ans.end(); it++) {
                cout << it->first << ' '  << it->second << '\n';
        }
}
