#include <iostream>
#include <algorithm>
#include <set>

using namespace std;

const int MAXN = 1e4 + 10;

int n;
pair <int, int> p[MAXN];
set <pair <int, int> > ans;

void dc(int s, int e)
{
    if(e - s < 2)
        return;
    int m = (s + e) / 2;
    for(int i = s; i < e; i++)
        ans.insert(make_pair(p[m].first, p[i].second));
    dc(s, m);
    dc(m, e);
}

int main()
{
    cin >> n;
    for(int i=0;i<n;i++)
    {
        cin >> p[i].first >> p[i].second;
        ans.insert(p[i]);
    }
    sort(p, p + n);
    dc(0, n);
    cout << ans.size() << endl;
    for(set <pair <int, int> >::iterator i = ans.begin(); i != ans.end(); i++)
        cout << i->first << " " << i->second << endl;
    return 0;
}
