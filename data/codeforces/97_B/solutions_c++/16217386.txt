#include<bits/stdc++.h>
using namespace std;
int n;
struct node{
	int x,y;
	node(int a=0,int b=0){
		x=a,y=b;
	}
	friend bool operator<(node a,node b){
		if(a.x==b.x)return a.y<b.y;
		return a.x<b.x;
	}
};
node p[10010];
set <node> st;
void solve(int l,int r){
	if(l>=r)return;
	int mid=(l+r)>>1;
	for(int i=l;i<=r;++i)
		st.insert(node(p[mid].x,p[i].y));
	solve(l,mid-1);solve(mid+1,r);
}
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;++i)scanf("%d%d",&p[i].x,&p[i].y);
	sort(p+1,p+n+1);
	solve(1,n);
	for(int i=1;i<=n;++i)
		st.erase(p[i]);
	printf("%d\n",st.size()+n);
	for(int i=1;i<=n;++i)
		printf("%d %d\n",p[i].x,p[i].y);
	set<node>::iterator it;
	for(it=st.begin();it!=st.end();it++){
		node a=*it;
		printf("%d %d\n",a.x,a.y);
	}
	return 0;
}
