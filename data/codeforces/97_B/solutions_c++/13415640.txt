
#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<ll, ll> pii;
typedef map< ll, vector<ll> > miv;

ll n;
vector<pii> p;
set<pii> ans;

void solve(ll l, ll r){
	if (l + 1 >= r) return;
	ll mid = (l + r) / 2;

	for (ll i = l; i < r; i++)
		ans.insert(pii(p[mid].first, p[i].second));

	solve(l, mid);
	solve(mid, r);
}

int  main(){
	ios::sync_with_stdio( false );

	cin >> n;

	for (ll i = 0; i < n; i++){
		ll x, y; cin >> x >> y;
		p.push_back(pii(x, y));
		ans.insert(pii(x, y));
	}
	sort(p.begin(), p.end());

	solve(0, n);

	cout << ans.size() << "\n";

	for (set<pii> :: iterator iter = ans.begin(); iter != ans.end(); iter++)
		cout << iter-> first << " " << iter->second << "\n";

	cout << endl;

	
}
