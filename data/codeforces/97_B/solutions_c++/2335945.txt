#include<iostream>
#include<set>
#include<algorithm>

using namespace std;

#define X first
#define Y second

typedef pair<int,int> pii;

const int MAX = 10000;
int n;
set<pii> a;
set<pii>::iterator it;
pii b[MAX];

void dive(int s, int e)
{
    if(e - s == 1)
        return ;
    int mid = s + e;
    mid /= 2;

    dive(s,mid);
    dive(mid,e);

    for(int i = s; i < e; i++)
        a.insert(make_pair(b[mid].X,b[i].Y));
}

int main()
{
    cin >> n;

    int x,y;
    for(int i = 0; i < n; i++)
    {
        cin >> b[i].X >> b[i].Y;
        a.insert(b[i]);
    }

    sort(b,b+n);

    dive(0,n);

    cout << a.size() << endl;
    for(it = a.begin(); it != a.end(); it++)
        cout << it->X << ' ' << it->Y << endl;
    return 0;
}
