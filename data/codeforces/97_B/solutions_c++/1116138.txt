#include<iostream>
#include<vector>
#include<set>
#include<map>
#include<algorithm>

using namespace std;

#define pb push_back
#define mk make_pair
#define x first
#define y second

vector <pair<int , int> > point;
set <pair<int , int  > > ans; 

void f(int l , int r)
{
	if(l > r)
		return;
	int mid = (l+r)/2;
	for(int i=l ; i<=r ; i++)
		ans.insert(mk(point[mid].x , point[i].y));
	f(l , mid-1);
	f(mid+1 , r);
	return;
}

int main()
{
	int n;
	cin >> n;
	for(int i=0 ; i<n ; i++)
	{
		int p , q;
		cin >> p >> q;
		point.pb(mk(p , q));
	}
	sort(point.begin() , point.end());
	set<pair<int , int> >::iterator it;
	f(0 , n-1);
	int res = ans.size();
	cout << res << endl;
	for(it = ans.begin() ; it!=ans.end() ; it++)
		cout << it->x  << " " << it->y << endl;

}
