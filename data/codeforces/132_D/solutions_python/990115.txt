s=list(raw_input()[::-1])
neg=[]
car=0
for i in xrange(len(s)):
	if car==1:
		if s[i]=="0":
			car=0
			s[i]="1"
		else:
			s[i]="0"
	elif i!=0 and s[i]==s[i-1]=="1":
		neg.append(i-1)	
		s[i]=s[i-1]="0"
		car=1
if car==1: s.append("1")
print s.count("1")+len(neg)
for i,c in enumerate(s):
	if c=="1":
		print "+2^"+repr(i)
for i in neg:
	print "-2^"+repr(i)
