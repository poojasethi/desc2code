#include<cstdio>
#include<cstring>
const int N=1000005;
int a[N];
char s[N];
int main()
{
	int i,j,n,ans=0;
	scanf("%s",s);
	n=strlen(s);
	i=0;
	while(1)
	{
		while(i<n&&s[i]=='0')i++;
		if(i==n)break;
		a[n-i]=1;
		while(i<n&&s[i]=='1')i++;
		a[n-i]=-1;
	}
	for(i=0;i<=n;i++)
	{
		if(a[i])
		{
			ans++;
			if(a[i+1])a[i]*=-1,a[i+1]=0;
		}
	}
	printf("%d\n",ans);
	for(i=0;i<=n;i++)
	{
		if(a[i]==1)
			printf("+2^%d\n",i);
		else if(a[i]==-1)
			printf("-2^%d\n",i);
	}
	return 0;
}
