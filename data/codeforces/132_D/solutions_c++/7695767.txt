#include <string.h>
#include <iostream>
#include <stdio.h>
using namespace std;

int k[2000010]={0};
char c[2000010];

int main()
{
	scanf("%s",c);
	int n=strlen(c);
	for(int i=0;i<n;i++)
		if(c[i]=='1')
			k[n-1-i]=1;

	int pos=0;
	for(int i=0;i<n+2;)
	{
		if(k[i]==0)
			i++;
		else
		{
			pos=i;
			while(k[pos])
				pos++;
			if(pos-i>=2)
			{
				for(int j=i+1;j<pos;j++)
					k[j]=0;
				k[i]=-1;
				k[pos]=1;
			}
			i=pos;
		}
	}
	
	int res=0;
	for(int i=0;i<n+2;i++)
		if(k[i]!=0)
			res++;
	
	printf("%d\n",res);
	
	for(int i=0;i<n+2;i++)
	{
		if(k[i]==1)
		{
			printf("+2^%d\n",i);
		}
		if(k[i]==-1)
		{
			printf("-2^%d\n",i);
		}
	}
 		
	return 0;
}
