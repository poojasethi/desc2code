#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
int a,b,c,d,e,f[1000005],g,h,i,j,k,l,m,n;
char s[1000005];
int main(){
	scanf("%s",s);n=strlen(s);
	for(i=0;i<n;i++)f[n-i-1]=s[i]=='1';
	for(i=0;i<n;i++){
		if(!f[i])continue;
		for(j=i;j<n&&f[j];j++);
		if(j-i>=2){
			f[j]=1;
			for(k=i;k<j;k++)f[k]=0;
			f[i]=-1;
		}
		i=j-1;
	}
	for(i=0;i<=n;i++)if(f[i])h++;
	printf("%d\n",h);
	for(i=0;i<=n;i++)if(f[i]){
		if(f[i]>0)printf("+2^%d\n",i);
		else printf("-2^%d\n",i);
	}
	return 0;
}
