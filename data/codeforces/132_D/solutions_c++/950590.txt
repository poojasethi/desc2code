#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char s[1000005],c[1000005];
int n,a[1000005];

int main()
{
    scanf("%s", s);
        n = strlen(s);
        reverse(s,s+n);
        s[n++] = '0';
        int ret = 0;
        for(int i = 0; i < n; )
        {
            if(s[i]=='1')
            {
                if(s[i+1]=='1')
                {
                    c[ret] = '-';
                    a[ret++] = i;
                    while(i<n && s[i]=='1')i++;
                    s[i] = '1';
                }
                else
                {
                    c[ret] = '+';
                    a[ret++] = i++;
                }
            }
            else i++;
        }
        printf("%d\n",ret);
        for(int i = 0; i < ret; i++)
            printf("%c2^%d\n",c[i],a[i]);
    return 0;
}
