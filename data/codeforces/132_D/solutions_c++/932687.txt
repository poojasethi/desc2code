#include<cstdio>
#include<cstring>
const int N = 1000005;
char s[N];
int r1[N],r2[N];
int main()
{
	int i,k=0,cnt=0,n;
	scanf("%s",s);
	n=strlen(s);
	for(i=n;i>0;i--)s[i]=s[i-1];
	s[0]='0';
	for(i=n;i>=0;i--)
	{
		if(s[i]=='1')
		{
			cnt++;	
		}
		else
		{
			if(cnt==1)
			{
				r1[k]=n-i-1;
				r2[k]=1;
				k++;
				cnt=0;
			}
			else if(cnt>1)
			{
				if(!i||s[i-1]=='0')
				{
					r1[k]=n-i;
					r2[k]=1;
					k++;	
					r1[k]=n-i-cnt;
					r2[k]=0;
					k++;	
					cnt=0;
				}
				else
				{
					cnt++;
					r1[k]=n-i;
					r2[k]=0;
					k++;
				}
			}
		}
	}
	printf("%d\n",k);
	for(i=0;i<k;i++)
	{
		if(r2[i])putchar('+');
		else putchar('-');
		printf("2^%d\n",r1[i]);
	}
	return 0;
}
