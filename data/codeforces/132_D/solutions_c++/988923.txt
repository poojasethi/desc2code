#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

int main()
{
    string str;
    vector< pair<char,int> > ans;

    cin>>str;
    reverse(str.begin(),str.end());

    str += "000";
    for(size_t i=0;i<str.size();i++)
        if(str[i]=='1')
            if(str[i+1]=='1')
            {
                ans.push_back( make_pair('-',i) );
                while(str[i]=='1')str[i++]='0';
                str[i--]='1';
            }
            else
                ans.push_back( make_pair('+',i) );

    cout << ans.size() << '\n';
    for(size_t i=0;i<ans.size();i++)
        cout<<ans[i].first<<"2^"<<ans[i].second<<'\n';

    return 0;
}
