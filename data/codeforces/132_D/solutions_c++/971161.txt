#include<iostream>
#include<sstream>
#include<string>
using namespace std;
int main(){
    string s;
    stringstream o;
    cin>>s;
    s="00"+s;
    int i,ans=0,l=s.size()-1;
    for(i=l;i>=1;i--){
        if(s[i]=='1'){
            if(s[i-1]=='1'){
                o<<"-2^"<<l-i<<endl;
                while(s[i]=='1')s[i--]='0';
                s[i++]='1';
                ans++;
            }else{
                o<<"+2^"<<l-i<<endl;
                ans++;
            }
        }
    }
    cout<<ans<<endl<<o.str();
}
