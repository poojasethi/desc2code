#include <cstdio>
#include <cstdlib>
#include <cstring>

char s[1000100];
int length, number, count;
int a[1000100];

int main() {
	scanf("%s", s);
	length = strlen(s);

	for (int i = 0; i < length; i++)
		if (s[i] == '1')
			a[length - i - 1] = 1;

	for (int i = 0; i < length + 2; i++)
		if (a[i]) {
			int j = i;
			while (a[j])j++;
			if (j - i >= 2) {
				for (int k = i; k < j; k++)
					a[k] = 0;
				a[j] = 1;
				a[i] = -1;	
			}
			i = j - 1;
		}

	for (int i = 0; i < length + 2; i++)
		if (a[i])
			number++;

	printf("%d\n", number);
	for (int i = 0; i < length + 2; i++)
		if (a[i])
			printf("%c2^%d\n", a[i] + 1 ? '+' : '-', i);

	return 0;
}
