#include <stdio.h>
#include <string.h>

char str[1<<20];
int num[1<<20],num3[1<<20],len;

int main() {
    scanf("%s",str);
    len = strlen(str);
    memset(num,0,sizeof(num));
    for(int i = 0;str[i];++ i)
        num[i] = str[len-1-i] - '0';
    memset(num3,0,sizeof(num3));
    for(int i = 0;str[i];++ i)
        num3[i] = num[i]*3;
    for(int i = 0;i < len + 10;++ i)
        num3[i+1] += num3[i]/2,num3[i] %= 2;
    int ans = 0;
    for(int i = 0;i < len + 10;++ i)
        ans += num[i]^num3[i];
    printf("%d\n",ans);
    for(int i = len + 10;i >= 0;-- i)
        if(num[i] ^ num3[i]) {
            if(num[i] == 1)
                printf("-2^%d\n",i - 1);
            else
                printf("+2^%d\n",i - 1);
        }

    for(int i = len - 1;i >= 0;-- i)
    return 0;
}
