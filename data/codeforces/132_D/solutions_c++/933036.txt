#include "cstdio"
#include "cstring"
char s[1000010];
int n,f[1000010];


int main()
{
    gets(s);
    n=strlen(s);
    for (int i=0;i<n;i++)
        if (s[i]=='1') f[n-1-i]=1;
    int i=0,j;
    for (;i<=n+1;)
        if (!f[i]) i++;
    else
        {
        j=i;
        while (f[j]) j++;
        if (j-i>=2)
           {
           f[i]=-1;
           for (int k=i+1;k<j;k++)
               f[k]=0;
           f[j]=1;
           }
        i=j;
        }
    long ans=0;
    for (int i=0;i<n+2;i++)
        if (f[i]!=0) ans++;
    printf("%d\n",ans);
     for(int i=0;i<n+2;i++)if(f[i]!=0)
        if(f[i]==1)printf("+2^%d\n",i);
        else printf("-2^%d\n",i);
    //getchar();
    return 0;
}
