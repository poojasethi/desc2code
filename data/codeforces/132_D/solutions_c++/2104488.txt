#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
int l;
vector <int> a;
char s[1000005];
int main(){
    scanf("%s",s+1);
    l = strlen(s+1);
    for(int i=1,j;i<=l;i++){
       if(s[i] == '0') continue;     
       j = i;
       while(j<=l && s[j] == '1') j++;
       if(!a.empty() && a.back() == -(i-2) ){
          a.push_back(-(i-1)); swap(a.back(),a[a.size()-2]); a.pop_back();
          a.push_back(-(j-1));
       }
       else if( j-i == 1) a.push_back(i);
       else a.push_back(i-1), a.push_back(-(j-1));
       i = j;     
    }
    printf("%d\n",a.size());
    for(int i=0;i<a.size();i++){
       a[i] >= 0 ? printf("+") : printf("-");
       printf("2^"); printf("%d\n",max(0,l-abs(a[i])));
    }
    return 0;
}
