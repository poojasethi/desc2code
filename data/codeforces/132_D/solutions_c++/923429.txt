#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>

using namespace std;

int n;
char s[1000001];
vector <pair <int, int> > v;

void push(int a, int b) {
	if (!v.empty() && v.back().first == a) {
		v.back().second = b;
	} else {
		v.push_back(make_pair(a, b));
	}
}

void gao() {
	reverse(s, s + n);
	for (int i = 0; i < n; ) {
		if (s[i] == '0') {
			++i;
			continue;
		} else {
			int count = 0, t = i;
			for (; i < n && s[i] == '1'; ++count, ++i);
			if (count >= 2) {
				push(t, -1);
				push(i, 1);
				s[i] = '1';
			} else {
				for (int j = t; j < i; ++j) {
					push(j, 1);
				}
			}
		}
	}
}

int main() {
	scanf("%s", s);
	n = strlen(s);
	gao();
	printf("%d\n", (int)v.size());
	for (vector <pair <int, int> >::const_iterator i = v.begin(); i != v.end(); ++i) {
		printf("%c2^%d\n", i->second == 1 ? '+' : '-', i->first);
	}
	return 0;
}
