#include <cstdio>
#include <cstring>

char s[1000010];
int flag[1000010];

int main() {
	int len;
	scanf("%s",s);
	len = strlen(s);
	for (int i = 0; i < len; i++)
		flag[len - 1 - i] = s[i]=='1';
	for (int i = 0; i <= len + 1; i++){
		if (!flag[i]) continue;
		int j = i;
		while (flag[j]) j++;
		if (j - i >= 2) {
			flag[i] = -1;
			for (int k = i + 1; k < j; k++)
				flag[k] = 0;
			flag[j] = 1;
		}
		i = j - 1;
	}
	int ans = 0;
	for (int i = 0; i < len + 2; i++)
		ans += flag[i]!=0;
	printf("%d\n",ans);
	for(int i = 0; i < len + 2;i++)
		if(flag[i] != 0)
			if(flag[i] == 1)
				printf("+2^%d\n",i);
			else 
				printf("-2^%d\n",i);
	return 0;
}
