#include	<stdlib.h>
#include	<stdio.h>

using namespace std;
#define CC(x,i) do{memset(x,i,sizeof(x));}while(0)
#define SWAP(a,b) do{int _tmp=a;a=b;b=_tmp;}while(0)
#define INFI 2000000000
#define INFF 1e100
typedef long long  LL;
char s[1234567]="0";
int f[1234567],l;
int main ( int argc, char *argv[] )
{
	scanf("%s",s+1);l=strlen(s);
	for(int i=0;i<l;i++)f[i]=s[i]-'0';
	int i=l-1,j=l-1;
	while(i>=0&&j>=0){
		while(i>=0&&f[i]!=1)i--;
		if(i<0)break;
		j=i;while(j>=0&&f[j]==1)j--;
		if(i-j>1){
			f[i]=-1;
			for(int k=i-1;k>j;k--)f[k]=0;f[j]=1;
		}
		i=j;
	}
	int cnt=0;
	for(int i=0;i<l;i++)if(f[i])cnt++;
	printf("%d\n",cnt);
	for(int i=0;i<l;i++){
		if(f[i]){
			printf("%c2^%d\n",f[i]>0?'+':'-',l-1-i);
		}
	}
	return EXIT_SUCCESS;
}
