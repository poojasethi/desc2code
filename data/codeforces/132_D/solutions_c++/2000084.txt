#include <iostream>
#include <vector>
#include <string>

using namespace std;


const int Maxn = 1000 * 1000 + 10;

int ret[Maxn];
bool sig[Maxn];
int siz = -1;
char a = '1';
char b = '0';

string s;
bool is1;

int main()
{
	cin >> s;
//	if(s[0] == a && s[1] == a)
//		ret[++siz] = s.size(), is1++ , sig[siz] = 1;

	for(int i=0; i<s.size(); i++)
	{
		if(is1)
		{
			if(s[i] == b && s[i+1] == a)
				ret[++siz] = s.size() - i - 1;
			else if(s[i] == b)
				ret[++siz] = s.size() - i , is1 = 0;
		}
		else
		{
			if(s[i] == a && s[i + 1] == a)
				ret[++siz] = s.size() - i , is1 = 1 , sig[siz] = 1;

			else if(s[i] == a)
				ret[++siz] = s.size() - i - 1 , sig[siz] = 1;
		}
	}

	if(is1 == 1)
		ret[++siz] = 0;
	cout << siz + 1 << endl;
	for(int i=0; i<siz+1; i++)
		if(sig[i])
			cout << "+2^" << ret[i] << endl;
		else
			cout << "-2^" << ret[i] << endl;	
	return 0;
}
