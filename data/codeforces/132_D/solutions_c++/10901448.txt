#include <iostream>
#include <cmath>
#include <algorithm>
#include <string.h>
#include <stdio.h>
#define maxn 1000000 + 10
using namespace std;
int ans[maxn];
string ch;
int main()
{
//#ifndef ONLINE_JUDGE
	//      freopen("in.txt","r",stdin);
//#endif // ONLINE_JUDGE
	cin>>ch;
	reverse(ch.begin(),ch.end());
	ch += '00';
	int len = ch.length();
	for(int i=0; i<len;)
	{
		if(ch[i] == '1')
		{
			int l = i;
			while(i<len && ch[i] == '1') i++;
			if(i - l > 1)
			{
				ans[l] = -1;
				ans[i] = 1;
				ch[i] = '1';
			}
			else
			{
				ans[l] = 1;
			}
		}
		else i++;
	}
	int tt = 0;
	for(int i=0; i<len; i++)
	{
		if(ans[i]) tt++;
	}
	printf("%d\n",tt);
	for(int i=0; i<len; i++)
	{
		if(ans[i] >0) printf("+2^%d\n",i);
		else if(ans[i] == -1) printf("-2^%d\n",i);

	}
	return 0;
}
