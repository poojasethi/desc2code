#include <iostream>
#include <string>
#include <vector>
#include <cstdio>
#include <utility>
using namespace std;
const int maxn = 1000000 + 10;
string l,s;
vector<pair<char,int> >v;
int main () {
    cin >> l;
    s += "00";
    s += l;
    int n = s.size();
    for(int i = n-1,j ; i >= 0 ; i = j-1) {
        j = i;
        while(s[j] == '1') j--;
        if(j == i-1) v.push_back(make_pair('+',n-i-1)); 
        else if(j < i-1) {
            v.push_back(make_pair('-',n-i-1));
            s[j] = '1';
            j++;
        }
    }
    cout << v.size() << endl; 
    for(int i = 0 ; i < v.size() ; i++) 
        cout << v[i].first << "2^" << v[i].second <<endl;
    return 0;
}
