#include <cstdio>
#include <cstring>
#include <algorithm>
#include <sstream>
#include <iostream>
using namespace std;

char str[1000010];

int main() {
    gets(str);
    reverse(str, str + strlen(str));
    strcat(str, "00");
    int n = strlen(str);
    stringstream out;
    int sign = 0;
    int cnt = 0;
    for (int i = 0; i < n; i++) {
        int d = (str[i] - '0') ^ sign;
        if (d == 1) {
            int d2 = (str[i + 1] - '0') ^ sign;
            if (d2 == 1)
                sign ^= 1;
            out << (sign == 0 ? '+' : '-') << "2^" << i << endl;
            cnt++;
        }
    }
    cout << cnt << endl;
    cout << out.str();
}
