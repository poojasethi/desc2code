#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

int main() {
	string s;
	cin >> s;
	vector<long long> v;
	reverse(s.begin(), s.end());
	long long t1, t2, tmp;
	for(int i = 0; i < s.size(); i++) {
		if(s[i] == '1') {
			v.push_back(i+1);
		}
		if(v.size() >= 2) {
			if(v[v.size()-1] == v[v.size()-2]+1) {
				t2 = v[v.size()-1];
				t1 = v[v.size()-2];
				v.pop_back();
				v.pop_back();
				v.push_back(-1*(t1));
				v.push_back(t2+1);
			} else if(v[v.size()-1] == v[v.size()-2]) {
				t1 = v[v.size()-1];
				v.pop_back();
				v.pop_back();
				if(t1 < 0) {
					v.push_back(t1-1);
				} else {
					v.push_back(t1+1);
				}
			}
		}
	}
	cout << v.size() << endl;
	for(int i = 0; i < v.size(); i++) {
		if(v[i] < 0) {
			cout << "-2^" << -1*v[i]-1 << endl;
		} else {
			cout << "+2^" << v[i]-1 << endl;
		}
	}
	return 0;
}
