#include<stdio.h>
#include<string.h>
const int maxn=1000005;
int p[maxn];
char str[maxn];
int main()
{
    int i,j,k;
    scanf("%s",str);
    int len=strlen(str);
    memset(p,0,sizeof(p));
    for(i=0;i<len;i++) if(str[i]=='1') p[len-1-i]=1;
    for(i=0;i<len+2;)
    {
        if(!p[i]) i++;
        else
        {
            j=i;
            while(p[j]) j++;
            if(j-i>=2)
            {
                p[i]=-1;
                for(k=i+1;k<j;k++)
                    p[k]=0;
                p[j]=1;
            }
            i=j;
        }
    }
    int m=0;
    for(i=0;i<len+2;i++) 
        if(p[i]!=0)
            m++;
    printf("%d\n",m);
    for(i=0;i<len+2;i++)
        if(p[i]==1)
            printf("+2^%d\n",i);
        else if(p[i]==-1)
            printf("-2^%d\n",i);
    return 0;
}
