#include <bits/stdc++.h>
using namespace std;
const int maxn = (1 << 20);
char s[maxn], st[maxn];
int ans[maxn];
int main()
{
	scanf("%s", st);
	int len = strlen(st);
	for (int i = 0; i < len; i++)
	  s[i] = st[len-i-1];
	s[len] = '0';
	int cnt = 0;
	//i == x ;   ~i == -(x+1)
	for (int i = 0; i <= len; i++)
	{
		if (s[i] == '2')
		  s[i+1]++;
		else if (s[i] == '1')
		{
			if (s[i+1] == '1')
			{
				s[i+1]++;
				ans[cnt++] = ~i;
			}else
			  ans[cnt++] = i;
		}
	}
	printf("%d\n", cnt);
	for (int i = 0; i < cnt; i++)
	{
		if (ans[i] < 0)
		  printf("-2^%d\n", ~ans[i]);
		else
		  printf("+2^%d\n", ans[i]);
	}
	return 0;
}
