#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
using namespace std;

const int N = 1100000;

char a[N];
vector<pair<char, int> > rez;
int n;

int main() {
	int i, j;
	
	//freopen("ttt", "r", stdin);
	
	a[0] = '0';
	cin >> a + 1;
	n = strlen(a);
	
	for(i = n - 1; i >= 0; i = j - 1) {
		j = i;
		while(j >= 0 && a[j] == '1') --j;
		if(j == i)
			continue;
		if(j == i - 1)
			rez.push_back(pair<char, int>('+', n - i - 1));
		else {
			rez.push_back(pair<char, int>('-', n - i - 1));
			a[j] = '1';
			++j;
		}
	}
	
	cout << rez.size() << "\n";
	for(i = 0; i < rez.size(); ++i)
		cout << rez[i].first << "2^" << rez[i].second << "\n";
	
	return 0;
}
