#include <cstdio>
#include <cstring>
#include <vector>
using namespace std;

#define mp make_pair
char s[1 << 20];

int main() {
	
	scanf("%s", s + 2);
	s[0] = s[1] = '0';
	int n = strlen(s);
	vector<pair<char, int> > v;
	for (int i = n - 1, j; i >= 0; i = j - 1) {
		j = i;
		while (s[j] == '1') {
			j--;
		}
		if (j == i - 1) {
			v.push_back(mp('+', n - 1 - i));
		} else if (j < i - 1) {
			v.push_back(mp('-', n - 1 - i));
			s[j] = '1';
			j++;
		}
	}
	printf("%d\n", v.size());
	for (int i = 0; i < v.size(); i++) {
		printf("%c2^%d\n", v[i].first, v[i].second);
	}
}
