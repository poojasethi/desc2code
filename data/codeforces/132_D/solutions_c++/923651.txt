#include <iostream>
#include <string>
#include <cstring>
#include <algorithm>
#include <vector>
#include <cmath>
#include <utility>
using namespace std;

int main() {
  string ss; cin >> ss;
  string s;
  for (int i = 0; i < ss.size(); i++) s += ss[ss.size()-1-i];
  s += '0';
  int n = s.size();

  vector<pair<int, int> > res;
  for (int i = 0; i < s.size(); i++) {
    if (s[i] == '0') continue;
    int j = i+1;
    while (j < s.size() && s[j] == '1') j++;
    if (j < i+2) {
      res.push_back(make_pair(i, 1));
      continue;
    }

    res.push_back(make_pair(i, -1));
    s[j] = '1';
    i = j-1;
  }

  cout << res.size() << endl;
  for (int i = 0; i < res.size(); i++)
    cout << (res[i].second > 0 ? "+" : "-") << "2^" << res[i].first << endl;
  return 0;
}
