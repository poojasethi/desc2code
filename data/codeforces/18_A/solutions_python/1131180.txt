arr = map(int, raw_input().split())

def dis(a, b):
    return (arr[a] - arr[b])**2 + (arr[a+1] - arr[b+1])**2

def check(l):
    A = dis(0, 2)
    B = dis(2, 4)
    C = dis(4, 0)

    A, B, C = sorted((A, B, C))
    return A > 0 and A + B == C

if check(arr):
    print 'RIGHT'
    exit()

for i in xrange(6):
    for j in [1, -1]:
        arr[i] += j
        if check(arr):
            print 'ALMOST'
            exit()
        arr[i] -= j

print 'NEITHER'
