def hypot(x,y):
    return x*x+y*y
p=map(int,raw_input().split())
def c(P):
    h=[hypot(P[0]-P[2],P[1]-P[3]),hypot(P[0]-P[4],P[1]-P[5]),hypot(P[4]-P[2],P[5]-P[3]) ]
    h.sort()
    return int(h[0] + h[1])==int(h[2]) and min(h)
if c(p): print "RIGHT";exit()
for i in range(6):
    for d in [-1,1]:
        q=p[:]
        q[i]+=d
        if c(q): print "ALMOST";exit()
print "NEITHER"
