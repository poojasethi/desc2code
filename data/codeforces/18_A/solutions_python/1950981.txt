#!/usr/bin/env python

from sys import stdin

def dist2(x0, y0, x1, y1):
   return (x0-x1)*(x0-x1) + (y0-y1)*(y0-y1)

def check(x1, y1, x2, y2, x3, y3):
   d1 = dist2(x1, y1, x2, y2)
   d2 = dist2(x2, y2, x3, y3)
   d3 = dist2(x3, y3, x1, y1)
   if d1 == 0 or d2 == 0 or d3 == 0: return False
   A = [d1, d2, d3]
   A.sort()
#  print A
   return A[0] + A[1] == A[2]

def main():
   x1, y1, x2, y2, x3, y3 = map(int, stdin.readline().split())
   if check(x1, y1, x2, y2, x3, y3):
      print 'RIGHT'
      return 0
   if check(x1-1, y1, x2, y2, x3, y3) or \
      check(x1+1, y1, x2, y2, x3, y3) or \
      check(x1, y1-1, x2, y2, x3, y3) or \
      check(x1, y1+1, x2, y2, x3, y3) or \
      check(x1, y1, x2-1, y2, x3, y3) or \
      check(x1, y1, x2+1, y2, x3, y3) or \
      check(x1, y1, x2, y2-1, x3, y3) or \
      check(x1, y1, x2, y2+1, x3, y3) or \
      check(x1, y1, x2, y2, x3-1, y3) or \
      check(x1, y1, x2, y2, x3+1, y3) or \
      check(x1, y1, x2, y2, x3, y3-1) or \
      check(x1, y1, x2, y2, x3, y3+1) :
      print 'ALMOST'
      return 0

   print 'NEITHER'
   return 0

if __name__ == '__main__': main()
