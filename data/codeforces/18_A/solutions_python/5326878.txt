def l(a, b): return (a[0] - b[0]) ** 2 + (a[1] - b[1]) ** 2
def f(t):
    p = [l(t[1], t[2]), l(t[0], t[2]), l(t[0], t[1])]
    return 0 in p or sum(p) - 2 * max(p)

t = map(int, raw_input().split())
a, b, c = (t[0], t[1]), (t[2], t[3]), (t[4], t[5])

if f((a, b, c)):
    u, v = ((a, b, c), (b, c, a), (c, a, b)), ((0, 1), (0, -1), (1, 0), (-1, 0))
    print 'NEITHER' if all(f(t) for t in [((i[0] + x, i[1] + y), j, k) for i, j, k in u for x, y in v]) else 'ALMOST'
else: print 'RIGHT'
