#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

a = ia()

def r(x1, y1,  x2, y2,  x3, y3):
    a2 = (x1-x2)**2 + (y1-y2)**2
    b2 = (x1-x3)**2 + (y1-y3)**2
    c2 = (x2-x3)**2 + (y2-y3)**2
    a2, b2, c2 = sorted([a2, b2, c2])
    return a2!=0 and c2 == a2 + b2

def solve():
    if r(*a): print 'RIGHT'; return
    n = len(a)
    for i in range(n):
        b = a[:]; b[i] += 1
        if r(*b): print 'ALMOST'; return

        b = a[:]; b[i] -= 1
        if r(*b): print 'ALMOST'; return
    print 'NEITHER'

solve()
