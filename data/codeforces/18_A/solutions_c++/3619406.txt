const int maxn=100000;

double f[7];

int ang(){
    double a=(f[1]-f[3])*(f[1]-f[3])+(f[2]-f[4])*(f[2]-f[4]);
    double b=(f[1]-f[5])*(f[1]-f[5])+(f[2]-f[6])*(f[2]-f[6]);
    double c=(f[5]-f[3])*(f[5]-f[3])+(f[6]-f[4])*(f[6]-f[4]);
    if(a<b)swap(a,b);
    if(a<c)swap(a,c);//cout<<a<<' '<<b<<' '<<c<<endl;
    if(!a||!b||!c)return 0;
    if(b+c-a==0)return 1;
    return 0;
}

int main(){
    for(int i=1;i<7;++i) cin>>f[i];
    int k=ang();
    k*=2;
    for(int i=1;i<7&&!k;++i){
        f[i]+=1;
        k=ang();
        f[i]-=2;
        if(!k)k=ang();
        f[i]+=1;
    }
    if(!k)puts("NEITHER");
    if(k==2)puts("RIGHT");
    if(k==1)puts("ALMOST");
    return 0;
}
