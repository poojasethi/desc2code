# include <bits/stdc++.h>

using namespace std;

int x[4], y[4];

int dis(int a,int b,int c,int d){
	return (a - c)*(a - c) + (b - d)*(b - d);
}

bool T(){
	int a[4];
	a[1] = dis(x[1], y[1], x[2], y[2]);
	a[2] = dis(x[1], y[1], x[3], y[3]);
	a[3] = dis(x[2], y[2], x[3], y[3]);

	sort(a+1, a+4);

	if(a[1] + a[2] == a[3] and a[1] * a[2] != 0)
		return true;
	return false;
}

int main(){
	for(int i=1 ; i<=3 ; i++)
		cin >> x[i] >> y[i];
	
	if(T()){
		cout << "RIGHT" << endl;
		return 0;
	}

	for(int i=1 ; i<=3 ; i++)
		for(int j=-1 ; j<=1 ; j+=2){
			x[i] += j;
			if(T()){
				cout << "ALMOST" << endl;
				return 0;
			}
			x[i] -= j;

			y[i] += j;
			if(T()){
				cout << "ALMOST" << endl;
				return 0;
			}
			y[i] -= j;
		}
	
	cout << "NEITHER" << endl;

	return 0;
}
