#include <iostream>
#define multi(a) ((a)*(a))
#define dis(x1,y1,x2,y2) (multi(x1-x2)+multi(y1-y2))
#define isright(x1,y1,x2,y2,x3,y3) (dis(x1,y1,x2,y2)>0&&dis(x1,y1,x3,y3)>0&&dis(x1,y1,x2,y2)+dis(x1,y1,x3,y3)==dis(x2,y2,x3,y3))
#define judge(x1,y1,x2,y2,x3,y3) isright(x1,y1,x2,y2,x3,y3)||isright(x2,y2,x3,y3,x1,y1)||isright(x3,y3,x1,y1,x2,y2)
using namespace std;



int main()
{
    int s[6],i,j;
    for (i=0;i<6;i++) cin>>s[i];
    for (i=0;i<6;i++)
    {
        if (judge(s[0],s[1],s[2],s[3],s[4],s[5]))
        {
            cout<<"RIGHT";
            return 0;
        }
        for (j=-1;j<=1;j+=2)
        {
            s[i]+=j;
            if (judge(s[0],s[1],s[2],s[3],s[4],s[5]))  {
            cout<<"ALMOST";
            return 0;}
            s[i]-=j;
        }

    }
    cout<<"NEITHER";
    return 0;
}
