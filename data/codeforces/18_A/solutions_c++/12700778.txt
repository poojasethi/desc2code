#include<iostream>
#include<algorithm>
using namespace std;

int x[7],y[3];
bool dis()
{
	y[0]=(x[0]-x[2])*(x[0]-x[2])+(x[1]-x[3])*(x[1]-x[3]);
	y[1]=(x[4]-x[2])*(x[4]-x[2])+(x[5]-x[3])*(x[5]-x[3]);
	y[2]=(x[0]-x[4])*(x[0]-x[4])+(x[1]-x[5])*(x[1]-x[5]);
	sort(y,y+3);
	if(y[2]==y[1]+y[0] && y[0]!=0 && y[1]!=0 && y[2]!=0)
	return true;
	return false;
}

int main()
{
	
	for(int i=0;i<6;i++)cin>>x[i];
	if(dis()){cout<<"RIGHT"<<endl;goto end;}
	for(int i=0;i<6;i++)
	{
		x[i]++;
		if(dis())
		{cout<<"ALMOST"<<endl;goto end;}
		x[i]-=2;
		if(dis())
		{cout<<"ALMOST"<<endl;goto end;}
		x[i]++;
	}
	cout<<"NEITHER"<<endl;
	end:
	return 0;
}
