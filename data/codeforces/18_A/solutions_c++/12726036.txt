#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int x[6],y[6];
int dx[4]={0,1,0,-1};
int dy[4]={1,0,-1,0};
bool test(){
  if(abs(x[0]*(y[1]-y[2])+x[1]*(y[2]-y[0])+x[2]*(y[0]-y[1]))==0) return false;
  for(int i=3;i<6;i++) x[i]=x[i-3],y[i]=y[i-3];
  for(int i=0;i<3;i++){
    if((y[i+1]-y[i])*(y[i+2]-y[i])==-1*(x[i+1]-x[i])*(x[i+2]-x[i])) return true;
  }
  return false;
}
int main(){
  for(int i=0;i<3;i++) scanf("%d %d",&x[i],&y[i]);
  if(test()){
    printf("RIGHT\n");
    return 0;
  }
  for(int i=0;i<3;i++){
    for(int j=0;j<4;j++){
      x[i]+=dx[j];
      y[i]+=dy[j];
      if(test()){
        printf("ALMOST\n");
        return 0;
      }
      x[i]-=dx[j];
      y[i]-=dy[j];
    }
  }
  printf("NEITHER\n");
  return 0;
}
