#include<bits/stdc++.h>
#define lp(i, n) for(int i=0 ; i<n ; ++i)

using namespace std;

int a[6], x, y, z, u, v;
int dx[] = { 0, 0, 1, -1 };
int dy[] = { 1, -1, 0, 0 };


bool check() {
	x = (a[0] - a[2]) * (a[0] - a[2]) + (a[1] - a[3]) * (a[1] - a[3]);
	y = (a[0] - a[4]) * (a[0] - a[4]) + (a[1] - a[5]) * (a[1] - a[5]);
	z = (a[2] - a[4]) * (a[2] - a[4]) + (a[3] - a[5]) * (a[3] - a[5]);
	return x && y && z && (x == y + z || y == x + z || z == x + y);
}

int main(int argc, char **argv) {
	lp(i, 6) scanf("%d", a + i);
	if (check()) {
		puts("RIGHT");
		return 0;
	}
	bool yes = 0;
	lp(i, 3) lp(j, 4) {
		u = a[i + i], v = a[i + i + 1];
		a[i + i] += dx[j], a[i + i + 1] += dy[j];
		yes |= check();
		a[i + i] = u, a[i + i + 1] = v;
	}
	puts(yes ? "ALMOST" : "NEITHER");
	return 0;
}
