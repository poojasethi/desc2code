#include <iostream>
#include<stdio.h>
#include<cmath>

using namespace std;
int sq(int x) { return x*x; }
int d[6];
void check(char *s){
	int a=sq(d[0]-d[2])+sq(d[1]-d[3]);
	int b=sq(d[2]-d[4])+sq(d[3]-d[5]);
	int c=sq(d[0]-d[4])+sq(d[1]-d[5]);
	if(!(a&&b&&c)) return ;
	if(a+b==c||a+c==b||c+b==a){
		printf("%s",s);
		exit(0);
	}
}


int main() {
	/*freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);*/
	for(int i=0;i<6;scanf("%d",d+i),++i){ } 
	check("RIGHT");
	for(int i=0;i<6;++i){
		d[i]-=1;
		check("ALMOST");
		d[i]+=2;
		check("ALMOST");
	d[i]-=1;	
	}
	
	printf("NEITHER");
	return 0;
}
