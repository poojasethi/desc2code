#include <stdio.h>

class Point{
	public:
		int x, y;
		Point(int _x=0, int _y=0){x=_x;y=_y;}
		Point operator-(const Point &p) {return Point(x-p.x, y-p.y);}
		Point operator+(const Point &p) {return Point(x+p.x, y+p.y);}
		int len(){return x*x+y*y;}
		void read(){scanf("%d%d", &x, &y);}
};

int ok(Point A[]) {
	int a = (A[0]-A[1]).len(), b = (A[1]-A[2]).len(), c = (A[2]-A[0]).len();
	return (a&&b&&c&&(a+b==c||a+c==b||b+c==a));
}

int main(void) {
	Point A[3], D[4];
	D[0]=Point(1, 0); D[1] = Point(0, 1); D[2] = Point(-1, 0); D[3]=Point(0, -1);
	for(int i=0;i<3;i++) A[i].read();
	if(ok(A)) puts("RIGHT");
	else {
		int flag=0;
		for(int i=0;i<3;i++)
			for(int j=0;j<4;j++) {
				A[i]=A[i]+D[j];
				if(ok(A)) flag = 1;
				A[i]=A[i]-D[j];
			}
		puts(flag?"ALMOST":"NEITHER");
	}
	return 0;
}
