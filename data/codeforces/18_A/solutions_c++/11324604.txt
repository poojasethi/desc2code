# include <iostream>
using namespace std;

int x[10];
int c[3];
bool ok() {
    int a = (x[0]-x[2])*(x[0]-x[2]) + (x[1]-x[3])*(x[1]-x[3]);
    int b = (x[2]-x[4])*(x[2]-x[4]) + (x[3]-x[5])*(x[3]-x[5]);
    int c = (x[4]-x[0])*(x[4]-x[0]) + (x[5]-x[1])*(x[5]-x[1]);

    if( !a || !b || !c ) return false;

    if( a==b+c || b==c+a || c==a+b )
        return true;

    return false;
}

int main() {

    for( int i=0; i<6; i++ )
        cin>>x[i];

    if( ok() ) { cout << "RIGHT" << endl; return 0; }

    c[0]=1;
    c[1]=-1;
    for( int i=0; i<6; i++ ) {
        for( int j=0; j<2; j++ ) {
            x[i] += c[j];
            if( ok() ) { cout << "ALMOST" << endl; return 0; }
            x[i] -= c[j];
        }
    }

    cout << "NEITHER" << endl;
    return 0;
}
