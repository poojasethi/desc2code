#include<stdio.h>
#include<math.h>

int main(int argc, char* argv[])
{   int xy[6],i,a,b,c;
    for(i=0;i<6;i++)
        scanf("%d",xy+i);
	a=pow(xy[0]-xy[2],2)+pow(xy[1]-xy[3],2);
	b=pow(xy[0]-xy[4],2)+pow(xy[1]-xy[5],2);
	c=pow(xy[2]-xy[4],2)+pow(xy[3]-xy[5],2);
    for(i=1;(a+b!=c)&&(a+c!=b)&&(b+c!=a);i++)
	{  
		if(i>=13)
		{  i=0;
		   break;
		}
		if((i-1)%2==0)
		    xy[(i-1)/2]--;
	    else if((i-1)%2==1)
		         xy[(i-1)/2]++;
		a=pow(xy[0]-xy[2],2)+pow(xy[1]-xy[3],2);
	    b=pow(xy[0]-xy[4],2)+pow(xy[1]-xy[5],2);
	    c=pow(xy[2]-xy[4],2)+pow(xy[3]-xy[5],2);
		if((i-1)%2==0)
		    xy[(i-1)/2]++;
	    else if((i-1)%2==1)
		         xy[(i-1)/2]--;
		if(a==0||b==0||c==0)
		{  a=1;
		   b=1;
		   c=1;
		}
	}
    if(i==0)
	   printf("NEITHER\n");
	else if(i==1)
		    printf("RIGHT\n");
	else
	   printf("ALMOST\n");
	return 0;
}
