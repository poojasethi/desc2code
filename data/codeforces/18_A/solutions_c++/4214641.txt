#include<iostream>
#include<cmath>
#define x first
#define y second
#define show(x) cerr << #x << "=" << x << endl
using namespace std;
typedef pair<int, int> pii;
int diff[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
int dist2(pii a, pii b)
{
	return (pow(a.x-b.x,2) + pow(a.y-b.y,2));
}
bool is_right(pii a, pii b, pii c)
{
	int ab = dist2(a, b), bc = dist2(b, c), ca = dist2(c, a);
	return ((ab == bc + ca) || (bc == ca + ab) || (ca == ab + bc)) && ((ab*bc*ca)); 
}
int main()
{
	pii p[3];
	cin >> p[0].x >> p[0].y >> p[1].x >> p[1].y >> p[2].x >> p[2].y;
	if(is_right(p[0], p[1], p[2])) {cout << "RIGHT" << endl; return 0;}
	for(int i = 0; i < 3; i++)
		for(int j = 0; j < 4; j++)
			if(is_right(pii(p[i].x+diff[j][0], p[i].y+diff[j][1]), p[(i+1)%3], p[(i+2)%3]))
			{
				cout << "ALMOST" << endl;
				return 0;
			}
	cout << "NEITHER" << endl;
}
