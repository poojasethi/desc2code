#include<stdio.h>

inline int sqr(int a){
  return a*a;
}

bool check(int x[], int y[]){
  int d[3];
  for(int i=0; i<3; i++){
    d[i] = sqr(x[i]-x[(i+1)%3]) + sqr(y[i]-y[(i+1)%3]);  
  }
  if(d[0] == 0 || d[1] == 0 || d[2] == 0) return false;
  return (d[0] == d[1]+d[2] || d[1] == d[0]+d[2] || d[2] == d[0]+d[1]);
}

int main(int agrc, char *argv[]){
  int x[3], y[3];
  scanf("%d %d %d %d %d %d", x, y, x+1, y+1, x+2, y+2);
  if(check(x, y)){
    printf("RIGHT\n");
    return 0;
  }
  bool ok = false;
  for(int p=0; p<3; p++){
    x[p]++; ok |= check(x, y); x[p]--;
    y[p]++; ok |= check(x, y); y[p]--;
    x[p]--; ok |= check(x, y); x[p]++;
    y[p]--; ok |= check(x, y); y[p]++;
  }
  if(ok){
    printf("ALMOST\n");
  }else{
    printf("NEITHER\n");
  }
  return 0;
}
