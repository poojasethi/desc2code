#include<cstdio>
int dir[4][2]={0,1,1,0,0,-1,-1,0};
int main(){
	int x[3],y[3];
	for(int i=0;i<3;i++)
		scanf("%d%d",&x[i],&y[i]);
	int flg=0;
	for(int i=0;i<3;i++){
		int j=(i+2)%3,k=(i+1)%3;
		if((x[j]-x[i])*(x[k]-x[i])+(y[j]-y[i])*(y[k]-y[i])==0){
			flg|=1;
			break;
		}
		for(int d=0;d<4;d++){
			int tx=x[i]+dir[d][0],ty=y[i]+dir[d][1];
			if(tx==x[j]&&ty==y[j]||tx==x[k]&&ty==y[k])
				continue;
			if((x[j]-tx)*(y[k]-ty)==(x[k]-tx)*(y[j]-ty))
				continue;
			if((x[j]-tx)*(x[k]-tx)+(y[j]-ty)*(y[k]-ty)==0 || (tx-x[k])*(x[j]-x[k])+(ty-y[k])*(y[j]-y[k])==0 || (x[k]-x[j])*(tx-x[j])+(y[k]-y[j])*(ty-y[j])==0){
				flg|=2;
				break;
			}
		}
	}
	if(flg&1)
		puts("RIGHT");
	else if(flg==0)
		puts("NEITHER");
	else
		puts("ALMOST");
	return 0;
}
