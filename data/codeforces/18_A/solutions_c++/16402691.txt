#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

bool isRight(int a[])
{
	int s[3];
	s[0] = pow(a[2] - a[0], 2) + pow(a[3] - a[1], 2);
	s[1] = pow(a[4] - a[2], 2) + pow(a[5] - a[3], 2);
	s[2] = pow(a[0] - a[4], 2) + pow(a[1] - a[5], 2);
	sort(s, s+3);
	return (s[0] > 0 && s[2] == s[0] + s[1]);
}
int main()
{
	int k[6];
	for (int i = 0; i < 6; i++)
		cin >> k[i];
	if (isRight(k))
	{
		cout << "RIGHT" << endl;
		return 0;
	}
	
	for (int i = 0; i < 6; i++)
	{
		k[i]++;
		if (isRight(k))
		{
			cout << "ALMOST" << endl;
			return 0;
		}
		k[i] -= 2;
		if (isRight(k))
		{
			cout << "ALMOST" << endl;
			return 0;
		}
		k[i]++;
	}
	cout << "NEITHER" << endl;
	return 0;
}
