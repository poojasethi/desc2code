#include<stdio.h>

int x[3],y[3];
int n_x[3],n_y[3]; 

int dist(int a,int b){
 return (n_x[a]-n_x[b])*(n_x[a]-n_x[b])+(n_y[a]-n_y[b])*(n_y[a]-n_y[b]);
}
int is_right(){
 if(dist(0,1)==0 || dist(1,2)==0 || dist(2,0)==0) return 0;
 if(dist(0,1)==dist(1,2)+dist(2,0)) return 1;
 if(dist(1,2)==dist(2,0)+dist(0,1)) return 1;
 if(dist(2,0)==dist(0,1)+dist(1,2)) return 1;
 return 0;
}

int main(){
 scanf("%d%d%d%d%d%d",x,y,x+1,y+1,x+2,y+2);
 int i;
 for(i=0;i<3;i++){
  n_x[i] = x[i];
  n_y[i] = y[i];
 }
 if(is_right()){
  printf("RIGHT\n");
  return 0;
 }
 for(i=0;i<3;i++){
  n_x[i]--;
  if(is_right()){ printf("ALMOST\n"); return 0; }
  n_x[i]++; n_y[i]--;
  if(is_right()){ printf("ALMOST\n"); return 0; }
  n_y[i]+=2;
  if(is_right()){ printf("ALMOST\n"); return 0; }
  n_x[i]++; n_y[i]--;
  if(is_right()){ printf("ALMOST\n"); return 0; }
  n_x[i]--;
 }

 printf("NEITHER\n");
 return 0;
}
