#include<stdio.h>
#define K(a) ((a)*(a))
#define dis(x1,y1,x2,y2) (K(x1-x2)+K(y1-y2))
#define tri(x1,y1,x2,y2,x3,y3) (dis(x1,y1,x2,y2)>0&&dis(x2,y2,x3,y3)>0&&dis(x1,y1,x2,y2)+dis(x2,y2,x3,y3)==dis(x3,y3,x1,y1))
#define dene(x1,y1,x2,y2,x3,y3) (tri(x1,y1,x2,y2,x3,y3)||tri(x2,y2,x3,y3,x1,y1)||tri(x3,y3,x1,y1,x2,y2))
int c[6];
int main()
{
	int i,j;
	for(i=0;i<6;i++) scanf("%d",c+i);
	if(dene(c[0],c[1],c[2],c[3],c[4],c[5])) {puts("RIGHT"); return 0;}
	for(i=0;i<6;i++)
	{
		for(j=-1;j<=1;j+=2)
		{
			c[i]+=j;
			if(dene(c[0],c[1],c[2],c[3],c[4],c[5])) {puts("ALMOST"); return 0;}
			c[i]-=j;
		}
	}
	puts("NEITHER");
	return 0;
}
