#include<iostream>
#include<cstdio>
#include<cstdlib>
using namespace std;
int x[20];
int ck()
{
    int i;
    int a,b,c,d;
    for (i=1;i<=6;++i) x[i+6]=x[i];
    for (i=1;i<6;i+=2)
    {
        a=x[i]-x[i+2]; b=x[i+1]-x[i+3];
        if (a==0&&b==0) return 0;
    }
    for (i=1;i<6;i+=2)
    {
        a=x[i]-x[i+2]; b=x[i+1]-x[i+3];
        c=x[i+2]-x[i+4]; d=x[i+3]-x[i+5];
        if (a*c+b*d==0) return 1;
    }
    return 0;
}
int main()
{
    int i;
    for (i=1;i<=6;++i) scanf("%d",&x[i]);
    if (ck()) cout <<"RIGHT"<<endl;
    else 
    {
        for (i=1;i<=6;++i)
        {
            x[i]--; if (ck()) {cout <<"ALMOST"<<endl;return 0;}
            x[i]+=2; if (ck()) {cout <<"ALMOST"<<endl;return 0;}
            x[i]--;
        }
        cout <<"NEITHER"<<endl; return 0;
    }
    return 0;
}
