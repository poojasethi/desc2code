#include<stdio.h>
struct point{
	int x,y;
}a,b,c;
int dir[4][2]={{1,0},{0,1},{-1,0},{0,-1}};
int jud(point a,point b,point c){
	point pa,pb;
	pa.x=b.x-a.x;pa.y=b.y-a.y;
	pb.x=c.x-a.x;pb.y=c.y-a.y;
	if((pa.x==0&&pa.y==0)||(pb.x==0&&pb.y==0))
	    return 0;
	if(pa.x*pb.x+pa.y*pb.y==0)
	    return 1;
	return 0;
}
int check(point a,point b,point c){
	int i;
	point d;
	if(jud(a,b,c))  return 100000;
	for(i=0;i<4;i++){
		d.x=a.x+dir[i][0];
		d.y=a.y+dir[i][1];
		if(jud(d,b,c)||jud(b,d,c)||jud(c,b,d)) return 1;
	}
	return 0;
}
int main(){
	int i,t;
	while(~scanf("%d%d%d%d%d%d",&a.x,&a.y,&b.x,&b.y,&c.x,&c.y)){
		t=check(a,b,c)+check(b,a,c)+check(c,a,b);
		if(t>=100000)
		    puts("RIGHT");
		else if(t>0)
		    puts("ALMOST");
		else
		    puts("NEITHER");
		}
		return 0;
	}
