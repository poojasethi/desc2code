#include <stdio.h>
#include<math.h>
int func(long a,long b,long c,long d,long e,long f)
{
	float x,y,z;
	x=((pow(a-c,2))+(pow(b-d,2)));
	y=((pow(c-e,2))+(pow(d-f,2)));
	z=((pow(a-e,2))+(pow(b-f,2)));
	if(x==y+z || y==x+z || z==y+x)
	{
		if(x!=0 && y!=0 && z!=0 )
		return 1;
		else
			return 0;
	}
	else
	return 0;
}

int main ()
{
	long a,b,c,e,f,d,m[]={-1,1,0,0},n[]={0,0 ,1,-1},p,q,r,s;
	while(~scanf("%ld %ld %ld %ld%ld%ld",&a,&b,&c,&d,&e,&f))
	{
		p=func(a,b,c,d,e,f);
		if(p==1)
				printf("RIGHT\n");
		else
		{
			for(int i=0;i<4;i++)
			{
				q=func((a+m[i]),(b+n[i]),c,d,e,f);
				r=func(a,b,(c+m[i]),(d+n[i]),e,f);
				s=func(a,b,c,d,(e+m[i]),(f+n[i]));
				if(q==1 || r==1 || s==1)
				{
					printf("ALMOST\n");
					break;
				}
			}
		}
		if(p==0 && q==0 && r==0 && s==0 )
		printf("NEITHER\n");
	}
	return 0;
}
