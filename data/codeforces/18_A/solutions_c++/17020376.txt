int x[1000],y[3];
bool check_rightangle(){
y[0]=pow(x[0]-x[2],2)+pow(x[1]-x[3],2);
y[1]=pow(x[4]-x[2],2)+pow(x[5]-x[3],2);
y[2]=pow(x[0]-x[4],2)+pow(x[1]-x[5],2);
sort(y,y+3);
if(y[2]==y[1]+y[0]&& y[0]!=0 && y[1]!=0 && y[2]!=0)return true;
else return false;
}
int main()
{
    for(int i=0;i<6;i++)cin>>x[i];
    if(check_rightangle()){cout<<"RIGHT";return 0;}
    for(int i=0;i<6;i++){
        x[i]++;
        if(check_rightangle()){cout<<"ALMOST";return 0;}
        x[i]-=2;
        if(check_rightangle()){cout<<"ALMOST";return 0;}
        x[i]++;
    }
    cout<<"NEITHER";
    return 0;
}
