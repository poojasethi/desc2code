#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int x[4], y[4];
int incx[] = {0,0,+1,-1};
int incy[] = {-1,+1,0,0};

int go() {
	int v[4];
	v[0] = (x[0]-x[1])*(x[0]-x[1]) + (y[0]-y[1])*(y[0]-y[1]);
	v[1] = (x[1]-x[2])*(x[1]-x[2]) + (y[1]-y[2])*(y[1]-y[2]);
	v[2] = (x[0]-x[2])*(x[0]-x[2]) + (y[0]-y[2])*(y[0]-y[2]);
	sort(v,v+3);
	if (v[0] == 0) return 0;
	return v[2] == v[0] + v[1];
}

int main() {
	for (int i=0; i<3; i++)
		scanf("%d%d",&x[i],&y[i]);
	
	if (go()) printf("RIGHT\n");
	else {
		int flag = 0;
		
		for (int i=0; i<3; i++)
			for (int j=0; j<4; j++) {
				x[i] += incx[j];	y[i] += incy[j];
				if (go()) { flag=1;	break;	}
				x[i] -= incx[j];	y[i] -= incy[j];
			}
		
		flag ? printf("ALMOST\n") : printf("NEITHER\n");
	}
	
	return 0;
}
