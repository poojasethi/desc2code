#include<iostream>
using namespace std;

int x[3],y[3];
inline int sq(int x)
{
	return x*x;
}
bool ok()
{
	int a,b,c;
	a = sq(x[0]-x[1])+sq(y[0]-y[1]);
	b = sq(x[1]-x[2])+sq(y[1]-y[2]);
	c = sq(x[2]-x[0])+sq(y[2]-y[0]);
	if(a==0 || b==0 || c==0)
		return false;
	if( a+b == c || a+c==b || b+c ==a)
		return true;
	else
		return false;
}

int main()
{
	for(int i=0;i<3;i++)
		cin >> x[i] >> y[i];
	if(ok())
	{
		cout << "RIGHT" << endl;
		return 0;
	}
	else
	{
		bool done = false;
		for(int i=0;i<3;i++)
			for(int j=-1;j<=1;j+=2)
			{
				x[i] +=j;
				if(ok())
					done = true;
				x[i]-=j;
				y[i]+=j;
				if(ok())
					done = true;
				y[i]-=j;
			}
		if(done)
		{
			cout << "ALMOST" << endl;
			return 0;
		}
	}
	cout << "NEITHER" << endl;

	return 0;
}
