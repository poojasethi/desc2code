#include<iostream>
#include<algorithm>
using namespace std;

int arr[6];
int dist(int p1,int p2)
{
	return (arr[p1]-arr[p2])*(arr[p1]-arr[p2])+(arr[p1+1]-arr[p2+1])*(arr[p1+1]-arr[p2+1]);
}

bool right()
{
	int length[3]={dist(0,2),dist(0,4),dist(2,4)};
	sort(length,length+3);
	return (length[2]==length[0]+length[1])&&(length[0]!=0);
}

int main()
{
	for(int i=0;i<6;i++)
	cin>>arr[i];
	
	int inc[]={0,-1,1};
	
	for(int i=0;i<6;i++)
	{
		for(int j=0;j<3;j++)
		{
			arr[i]+=inc[j];
			if(right())
			{
				if(j==0)
				{
					cout<<"RIGHT";
					goto end;
				}
				else
				{
					cout<<"ALMOST";
					goto end;
				}
			}
			arr[i]-=inc[j];
		}
	}
	cout<<"NEITHER";
	end:
	return 0;
}
