
#include<iostream>

using namespace std ;


int a[6] ; 

int sq ( int x ) { return x*x ; } 

bool isright() { 
    
    
    int x1 = a[0] ;
    int y1 = a[1] ;
    int x2 = a[2], y2 = a[3], x3 = a[4], y3 = a[5] ; 

    int x = sq(x1-x2) + sq(y1-y2);
    int y = sq(x1-x3) + sq(y1-y3);
    int z = sq(x3-x2) + sq(y2-y3);


    if ( x+y == z || x+z == y || z+y == x ) { 
	int area = x1*(y2-y3) + x2*(y3-y1) + x3*(y1-y2) ; 	
	if ( area != 0 ) { 
	    return true ;
	}
    }
    return false ; 

}
int main () { 
    for ( int i = 0 ; i < 6 ; i ++ ) { cin >> a[i] ; } 
    if ( isright() ) { 
	cout << "RIGHT" << endl ;
	return 0 ;
    }
    for ( int i = 0 ; i < 6 ; i ++ ) { 
	for ( int j = -1 ; j < 2 ; j += 2 ) { 
	    a[i] += j ;
	    if ( isright() ) { 
		cout << "ALMOST" << endl ;
		return 0 ;
	    }
	    a[i] -= j;
	}
    }
    cout << "NEITHER" << endl ; 
 	return 0 ;
}
