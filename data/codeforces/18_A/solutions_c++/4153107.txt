#include <iostream>
using namespace std;
int d[4][2]={{1,0},{-1,0},{0,1},{0,-1}};

int judge(int x1,int y1,int x2,int y2,int x3,int y3)
{
   int px,py,qx,qy;
   px=x2-x1,py=y2-y1;
   qx=x3-x1,qy=y3-y1;
   if(px*qx+py*qy==0&&px*qy-py*qx!=0)
   return 1;
   px=x1-x2,py=y1-y2;
   qx=x3-x2,qy=y3-y2;
   if(px*qx+py*qy==0&&px*qy-py*qx!=0)
   return 1;
   px=x2-x3,py=y2-y3;
   qx=x1-x3,qy=y1-y3;
   if(px*qx+py*qy==0&&px*qy-py*qx!=0)
   return 1;
   return 0;
}
int main()
{
    int x1,y1,x2,y2,x3,y3;
	cin>>x1>>y1>>x2>>y2>>x3>>y3;
	if(judge(x1,y1,x2,y2,x3,y3))
	{
	cout<<"RIGHT"<<endl;
	return 0;
	}
	int i;
	for(i=0;i<4;i++)
	{
		if(judge(x1+d[i][0],y1+d[i][1],x2,y2,x3,y3)||
		judge(x1,y1,x2+d[i][0],y2+d[i][1],x3,y3)||
		judge(x1,y1,x2,y2,x3+d[i][0],y3+d[i][1]))
		{
			cout<<"ALMOST"<<endl;
			return 0;
		}
	}
	cout<<"NEITHER"<<endl;
	return 0;
}
