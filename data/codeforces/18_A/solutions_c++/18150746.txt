#include <bits/stdc++.h>
using namespace std;

int const dx[] = {0, 0, 1, -1};
int const dy[] = {1, -1, 0, 0};

int x[4], y[4];

bool right() {
    x[3] = x[0], y[3] = y[0];
    int z[3];
    for (int i=0; i<3; ++i) {
        z[i] = (x[i] - x[i+1]) * (x[i] - x[i+1]) + (y[i] - y[i+1]) * (y[i] - y[i+1]);
    }
    sort(z, z + 3);
    if (!z[0]) return false;
    return z[0] + z[1] == z[2];
}

int main() {
    for (int i=0; i<3; ++i) cin >> x[i] >> y[i];
    if (right()) {
        cout << "RIGHT";
        return 0;
    }
    for (int i=0; i<3; ++i) for (int k=0; k<4; ++k) {
            x[i] += dx[k], y[i] += dy[k];
            if (right()) {
                cout << "ALMOST";
                return 0;
            }
            x[i] -= dx[k], y[i] -= dy[k];
        }
    cout << "NEITHER";
    return 0;
}
