#include <iostream>
#include <cstdio>

using namespace std;

int dist (int a, int b){
  return a*a+b*b;
}

bool check(  int a,int b,int c,int d,int e,int f){
  int dist1 = dist(a-c,b-d);
  int dist2 = dist(c-e,d-f);
  int dist3 = dist(a-e,b-f);

  if(dist1 * dist2 * dist3 == 0){
    return false;
  }

  return (dist1  + dist2) == dist3 || (dist2+dist3 == dist1) || (dist1+dist3 == dist2) ;
}


int main(){
  int a,b,c,d,e,f;
  scanf("%d %d %d %d %d %d", &a, &b, &c, &d, &e, &f);
  if(check(a,b,c,d,e,f)){
    cout << "RIGHT" <<endl;
    return 0;
  }
  int dx[4] = {0,0,1,-1};
  int dy[4] = {1,-1,0,0};
  for(int i = 0;i<4;i++){
    if(check(a+dx[i], b +dy[i], c,d,e,f) ||
       check(a, b, c +dx[i],d +dy[i],e,f) ||
       check(a, b , c,d,e +dx[i],f +dy[i]) ){
      
      cout << "ALMOST" <<endl;
      return 0;
    }
  }
  cout << "NEITHER" << endl;
  return 0;
  
  return 0;
}
