#include <iostream>
using namespace std;

main()
{
	int n,i,p,q,ma=0,mi=1000000001,r;
	bool f=false;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>p>>q;
		if(p<mi || q>ma)
		{
			if((p<=mi)+(q>=ma)<2)f=false;
			mi=min(p,mi);ma=max(q,ma);
		}
		if((p<=mi)+(q>=ma)==2)f=true,r=i+1;
	}
	if(f)cout<<r;else cout<<-1;
	return 0;
}
