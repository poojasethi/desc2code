#include <iostream>
using namespace std;
int main()
{static long int i,n,x,y,x1,y1,a,b,t;
  cin>>n>>x>>y;
  x1=x;  y1=y;
  for(i=1;i<n;i++)
  { cin>>a>>b;
     if(a<=x && b>=y)
       {x=a;y=b;t=i;}
      if(b>y1)
         y1=b;
         if(a<x1)
            x1=a;  
  }
if(y1==y && x1==x)
cout<<t+1<<endl;
 else cout<<"-1";
    return 0;}
