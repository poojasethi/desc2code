#include<iostream>
using namespace std;

int main(){
int place=0;
int i,n,ar,de,nar,nde;
cin>>n;
for(i=0;i<n;i++){
if(i==0){
place=i+1;
cin>>ar;
cin>>de;
}
else{
cin>>nar;
cin>>nde;
if((ar>=nar)&&(nde>=de)){
de=nde;
ar=nar;
place=i+1;
}
else{
if(ar>nar){
ar=nar;
place=-1;
}
if(de<nde){
de=nde;
place=-1;
}
}
}
}
cout<<place;
}
