#include <iostream>
#include <algorithm>
using namespace std;
int n,i,L,R,a[100001],b[100001];
int main()
{
    cin>>n;
    for(i=1;i<=n;i++)cin>>a[i]>>b[i];
    L=a[1];R=b[1];
    for(i=1;i<=n;i++){L=min(L,a[i]);R=max(R,b[i]);}
    for(i=1;i<=n;i++)if(a[i]==L&&b[i]==R){cout<<i;return 0;}
    cout<<-1;
    return 0;
}
