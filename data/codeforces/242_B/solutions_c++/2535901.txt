#include<iostream>
using namespace std;
int main(){
	int n,lmin=0x7fffffff,rmax=0xffffffff;
	int l[100000],r[100000];
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>l[i]>>r[i];
		lmin=min(lmin,l[i]);
		rmax=max(rmax,r[i]);
	}
	int res=-1;
	for(int i=0;i<n;i++){
		if(lmin==l[i] && rmax==r[i]){
			res=i+1;
		}
	}
	cout<<res<<endl;
}
