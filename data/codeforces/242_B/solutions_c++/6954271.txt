#include<iostream>
#define FOR(i,n) for(int i=0;i<n;i++)
using namespace std;
int main()
{
 int n,s=1000000000,l=0;  cin>>n;
 int *B=new int[n],*E=new int[n];
 FOR(i,n)
 { cin>>B[i]>>E[i];
   if(B[i]<s)s=B[i];
   if(E[i]>l)l=E[i];
 }
 FOR(i,n)
 { if(B[i]==s&&E[i]==l)
    {cout<<i+1; return 0;}
 }
 cout<<-1;
 delete[]B; delete[]E;
}
