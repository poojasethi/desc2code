#include<stdio.h>
int main()
{
	int n,a,b,c,d,ans=-1,i;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&a,&b);
		if(i==0)
		{
			c=a;
			d=b;
		}
		if(a<c)
		{
			c=a;
			ans=-1;
		}
		if(d<b)
		{
			d=b;
			ans=-1;
		}
		if(a==c&&b==d)
		{
			ans=i+1;
		}
	}
	printf("%d\n",ans);
	return 0;
}
