#include<iostream>
using namespace std;
int main(){
	int n,a,b,l,r,i;
	int tl,tr,order;
	while(cin>>n){
		for(i=1;i<=n;i++){
			cin>>a>>b;
			if(i==1) l=a,r=b;
			if(a<=l) l=a;
			if(b>=r) r=b;
			if(a==l && b==r){
				tl=l,tr=r;
				order=i;
			}
		}
		if(tl==l && tr==r) cout<<order<<endl;
		else cout<<-1<<endl;
	}
}
