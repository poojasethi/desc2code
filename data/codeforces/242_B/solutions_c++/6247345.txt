#include<cstdio>
using namespace std;
int a1[100000],a2[100000];
int main()
{
	int n,a=0,b=0;
	scanf("%d",&n);
	for(int i=0;i<n;++i)
	{
		scanf("%d%d",a1+i,a2+i);
		if(a1[i]<a1[a])a=i;
		if(a2[i]>a2[b])b=i;
	}
	for(int i=0;i<n;++i)
	if(a1[i]==a1[a]&&a2[i]==a2[b])
	{
		printf("%d\n",i+1);
		return 0;
	}
	puts("-1");
}
