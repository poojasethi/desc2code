#include<cstdio>
#include<algorithm>
using namespace std;
int a[100000],b[100000],n,lenk;
int main(){
	int i;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d%d",a+i,b+i);
		if(b[i]-a[i]>b[lenk]-a[lenk])
			lenk=i;
	}
	for(i=0;i<n;i++){
		if(a[i]<a[lenk]||b[i]>b[lenk]){
			printf("-1\n");
			return 0;
		}
	}
	printf("%d\n",lenk+1);
}
