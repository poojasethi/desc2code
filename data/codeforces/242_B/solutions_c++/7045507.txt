#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,a=INT_MAX,b=INT_MIN,l[100010],r[100010],ans=-1;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&l[i],&r[i]);
		a=min(a,l[i]);
		b=max(b,r[i]);
	}
	for(i=1;i<=n;i++)
	{
		if(a==l[i] && b==r[i])
		{
			ans=i;
			break;
		}
	}
	cout<<ans<<endl;
	return 0;
}
