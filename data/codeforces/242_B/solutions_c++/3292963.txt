#include<iostream>
using namespace std;

long long n,x=2147483647,y=-1;

int main()
	{
	long long num=1,a,b;
	cin>>n>>a>>b;
	x=a,y=b;
	for(int i=2;i<=n;i++)
		{
		cin>>a>>b;
		if(x<=a&&y>=b&&num!=-1) continue;
		if(a<=x&&b>=y) num=i,x=a,y=b;
		else x=min(x,a),y=max(b,y),num=-1;
		}
	cout<<num;
	return 0;
	}
