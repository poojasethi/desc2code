#include <stdio.h>
#define F(I,B,E) for(int I=B; I<E; I++)
main() {
  int n,l[100100],L=1<<30,r[100100],R=1<<31;
  scanf("%d",&n);
  F(i,0,n)
    scanf("%d%d",l+i,r+i),
    L=L<l[i]?L:l[i],
    R=R>r[i]?R:r[i];
  F(i,0,n)
    if(l[i]==L && r[i]==R) {
      printf("%d",i+1);
      return 0;
    }
  puts("-1");
  return 0;
}
