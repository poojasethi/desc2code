#include<iostream>
using namespace std;
int main()
{
	int n,x=-1,left,right,a,b;
	cin>>n>>a>>b;
	left=a;right=b;x=1;
	for(int i=1;i<n;i++)
	{
		cin>>a>>b;
		if(a<=left&&b>=right) {left=a; right=b; x=i+1;}
		else if(a<left&&b<right) {left=a; x=-1;}
		else if(a>left&&b>right) {right=b; x=-1;}
	}
	cout<<x<<endl;
}
