#include<iostream>

using namespace std;
int main()
{
	int n;
	cin>>n;
	long long int a[n],b[n],min=1000000001,max=0;
	for(int i=0;i<n;i++)
	{
		cin>>a[i]>>b[i];
		if(min>a[i]) min=a[i];
		if(max<b[i]) max=b[i];
	}
	for(int i=0;i<n;i++)
	{
		if(a[i]<=min&&b[i]>=max)
		{cout<<i+1;return 0;}
	}
	cout<<-1;
	
}
