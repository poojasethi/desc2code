#include <iostream>
using namespace std;
int q[100005],w[100005],tot;
int ans=-2;
int p=-1,c=1000000009;
int main(){
    int n;
    cin >> n;
    for(int i=0;i<n;i++)cin >> q[i] >> w[i];
    for(int i=0;i<n;i++){p=max(p,w[i]);c=min(q[i],c);}
    for(int i=0;i<n;i++){if(q[i]<=c&&p<=w[i])ans=i;}
    cout << ans+1;

}
