#include <iostream>
using namespace std;
 
int main()
{
 int n,i,al,ar,ans,x,y;
     bool ok=false;
 cin>>n;
 for(i=1;i<=n;i++)
  {cin>>x>>y;
   if (i==1) {al=x;ar=y;}
   if (x<al){al=x;ok=false;}
   if(y>ar) {ar=y;ok=false;}
   if ( (al==x)&&(ar==y)) {ok=true;ans=i;}
  }
 if (ok) cout<<ans; else cout<<-1;
 return 0;   
}
