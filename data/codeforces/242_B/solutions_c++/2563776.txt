#include <stdio.h>
#include <algorithm>

using namespace std;

int mr,ml,ans=-1,n,i,l[100001],r[100001];

int main()
{    
  scanf("%d",&n);
  for (i=0;i<=n-1;i++) scanf("%d%d",&l[i],&r[i]);
  mr=*max_element(r,r+n);
  ml=*min_element(l,l+n);
  for (i=0;i<=n;++i) if (l[i]==ml && r[i]==mr) ans=++i;
  printf("%d",ans);
  return 0;
}
