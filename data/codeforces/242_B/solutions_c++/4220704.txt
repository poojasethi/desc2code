#include <cstdlib>
#include <iostream>

using namespace std;

int main(int argc, char *argv[])
{long long  int n,m=0,j,k,l=1e9,o=1e9,p=0,c=0,y=0;
cin>>n;
while(n--)
{cin>>j>>k;
y++;
if(j<=l&&k>=m)
{l=j;
m=k;
c=y;
}
if(j<=o)
{o=j;
}
if(k>=p)
{p=k;
}
}

if(l==o&&m==p)
{
cout<<c;
}
else
cout<<"-1";
    return(0);
}
