#include<stdio.h>
int x[100010],y[100010];
int main(){
	int i,k,n,max,min;
	k=-1;
	scanf("%d",&n);
	max=0;min=0x7fffffff;
	for(i=1;i<=n;i++){
		scanf("%d%d",&x[i],&y[i]);
		min=min<x[i]?min:x[i];
		max=max>y[i]?max:y[i];
	}
	for(i=1;i<=n;i++)
	    if(x[i]==min&&y[i]==max)
	        break;
	printf("%d\n",i<=n?i:-1);
	return 0;
}
