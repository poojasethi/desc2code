#include <bits/stdc++.h>
using namespace std;
int n;
int main() {
	cin>>n;
	int m1=numeric_limits<int>::max(),m2=numeric_limits<int>::min(),p1,p2,x,y;
	for(int i=1;i<=n;i++)
	{
		cin>>x>>y;
		if(x<=m1 && y>=m2)m1=x,m2=y,p1=p2=i;
		if(x<m1)m1=x,p1=i;
		if(y>m2)m2=y,p2=i;
	}
	if(p1==p2)cout<<p1;
	else cout<<-1;
	return 0;
}
