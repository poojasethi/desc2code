#include <cstdio>

int n, a = 1000000001, b = 0, A[100000], B[100000];

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d %d", A + i, B + i);
		if (A[i] < a) a = A[i];
		if (B[i] > b) b = B[i];
	}
	for (int i = 0; i < n; i++) {
		if (A[i] == a && B[i] == b) {
			printf("%d\n", i + 1);
			return 0;
		}
	}
	puts("-1");
	return 0;
}
