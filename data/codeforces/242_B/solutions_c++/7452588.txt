#include <bits/stdc++.h>
using namespace std;


int main()
{
	int n, i, l, r, k, mn = 1000000001, mx = 0;

	scanf("%d", &n);
	
	for(i = 0; i < n; i++) {
		scanf("%d%d", &l, &r);

		if(l < mn) {
			mn = l;
			k = -2;
		}
		if(r > mx) {
			mx = r;
			k = -2;
		}
		if(l == mn && r == mx) k = i;
	}

	printf("%d\n", k+1);

	return 0;
}
