#include<stdio.h>
int main(){
    int l,r;
    int max,min,ind,i,n;
    scanf("%d",&n);
    scanf("%d %d",&min,&max);
    ind=1;
    for(i=1;i<n;i++){
        scanf("%d %d",&l,&r);
        if(l<min){min=l;ind=-1;}
        if(r>max){max=r;ind=-1;}
        if(l<=min&&r>=max)ind=i+1;
    }
    printf("%d\n",ind);
    return 0;
}
