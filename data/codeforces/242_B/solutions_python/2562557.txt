s = int(raw_input())
left, right, tl, tr, ans = 1<<31, 0, -1, -1, -1
for x in range(s):
	a, b = map(int, raw_input().split())
	if a<=left and b>=right:
		ans = x+1
		left, right = a, b
		tl, tr = a, b
	elif a<=left:
		left = a
	elif b>=right:
		right = b

if tl==left and tr==right:
	print ans
else:
	print -1
