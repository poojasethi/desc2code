import sys

n = int(sys.stdin.readline().strip())

ans = -1
mx = 0
mn = 1000000001

for i in range(0, n):
    (l, r) = [int(x) for x in sys.stdin.readline().strip().split()]
    if l < mn or r > mx:
        ans = -1
    mx = max(r, mx)
    mn = min(l, mn)
    if (l, r) == (mn, mx):
        ans = i + 1

print(ans)
