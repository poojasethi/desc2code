n = int(raw_input())

ranges = []
for i in range(n):
    ranges.append(tuple(map(int, raw_input().split())))

l = None
r = None
for a, b in ranges:
    if l is None or a < l:
        l = a
    if r is None or b > r:
        r = b

answer = -1
for i, (a, b) in enumerate(ranges):
    if a == l and b == r:
        answer = i + 1

print answer
