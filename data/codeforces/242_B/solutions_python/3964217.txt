n = input()
lst = []
for i in xrange(n):
	l,r = map(int,raw_input().split(' '))
	lst.append((l,r))
max_r = max([x[1] for x in lst])
l2 = sorted(lst)
i = 0
counter = 0
while(l2[i][0]==l2[0][0]):
	if (l2[i][1] == max_r):
		counter = 1
		break
	i = i + 1
if (counter == 1): print (lst.index((l2[0][0],max_r)) + 1)
else : print '-1'
