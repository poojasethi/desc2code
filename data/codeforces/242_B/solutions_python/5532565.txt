n = input()
ans = -1
maxright, minleft = 0,10**10
curl = 0 
curr = 0
for i in xrange(1,n+1):
	l,r = map(int,raw_input().split())
	maxright = max(maxright,r)
	minleft = min(minleft,l)
	if curl>minleft or curr<maxright:
		ans = -1
	
	if l<=minleft and r>=maxright:
		ans = i
		curl = l
		curr = r
	
print ans
