if __name__ == '__main__':
	N = int(raw_input())
	mins = {}
	maxs = {}
	for i in range(N):
		a, b = [int(_) for _ in raw_input().split()]
		mins.setdefault(a, set()).add(i + 1)
		maxs.setdefault(b, set()).add(i + 1)
	l = list(mins[sorted(mins)[0]] & maxs[sorted(maxs)[-1]])
	if not l:
		print(-1)
	else:
		print(l[0])
