n = int(input())

segments = []
minL = 10 ** 9 + 1
maxR = -1
for i in range(n) :
    l, r = map(int, raw_input().split())
    segments.append([l, r])
    if l < minL :
        minL = l
    if r > maxR :
        maxR = r
if [minL, maxR] in segments :
    print segments.index([minL, maxR]) + 1
else :
    print -1
