def solve(s):
    minl, maxr = s[0]
    for l, r in s[1:]:
        minl, maxr = min(minl, l), max(maxr, r)
    for i in range(len(s)):
        if s[i][0] <= minl and s[i][1] >= maxr:
            return i+1
    return -1

n = int(raw_input())
s = [map(int, raw_input().split()) for i in range(n)]
print solve(s)
