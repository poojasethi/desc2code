n = int(raw_input())
lst = map(int, raw_input().split())
one = 0
two = 0
three = 0
for i in range(n):
    if lst[i] == 1:
        one += 1
    elif lst[i] == 2:
        two += 1
    else:
        three += 1
a = one + two
b = one + three
c = two + three
lst2 = []
lst2.append(a)
lst2.append(b)
lst2.append(c)
lst2.sort()
print(lst2[0])
