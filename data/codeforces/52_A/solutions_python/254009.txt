def main():
    
    
    n = int(raw_input())
    cnt = [0] * 4
    num = map(int, raw_input().split())
    for i in range(0, n):
        cnt[num[i]] += 1
    ans = 100000000
    for i in range(1, 4):
        if ans > (n - cnt[i]):
            ans = n - cnt[i]
    print ans
    
    return 0

if __name__ == '__main__': main()
