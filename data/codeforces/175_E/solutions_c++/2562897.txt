#include <cstdio>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;
#define rep(i,n) for (int i=0;i<n;i++)
int nf,ne,ns,rf,re,rs,df,de,p1,p2;
double lf,le,ls,p,ans;
vector<double> t;
double crs(double l,double r,double x,double y) {
	return max(0.,min(r,y)-max(l,x));}
int main() {
	scanf("%d%d%d%d%d%d%d%d",&nf,&ne,&ns,&rf,&re,&rs,&df,&de);
	lf=sqrt(rf*rf-1);le=sqrt(re*re-1);ls=sqrt(rs*rs-1);
	int n=nf+ne+ns,u=(1<<n)-1;
	rep(i,(1<<n)) {
		if (__builtin_popcount(i)!=ns) continue;
		t.clear();p=0;
		for (int x=u^i;x;x-=x&-x) {
			p1=__builtin_ctz(x)>>1;
			double pf=2*lf*df,pe=2*le*de;
			for (int y=i;y;y-=y&-y) {
				p2=__builtin_ctz(y)>>1,
				pf+=df*crs(p2-ls,p2+ls,p1-lf,p1+lf);
				pe+=de*crs(p2-ls,p2+ls,p1-le,p1+le);
			}
			p+=pf;t.push_back(pf-pe);
		}
		sort(t.begin(),t.end());
		rep(x,ne) p-=t[x];
		if (p>ans) ans=p;
	}
	printf("%.10f\n",ans);
}
