#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
int nf,ne,ns,rf,re,rs,df,de,tot,n,a[1<<21],c[1<<21],g[1<<21],p[21];
double Rf,Re,Rs,ans,res[55];
double cross(double l,double r,double x,double y){return max(0.,min(r,y)-max(l,x));}
void work()
{
	for (int i=0;i<tot;i++){
		int j=(1<<n)-1^a[i],cnt=0; double sum=0,F,E;
		while (j){
			int x=j&-j,y,k=a[i]; j-=x,x=g[x]>>1; F=Rf*2*df,E=Re*2*de;
			while (k) y=k&-k,k-=y,y=g[y]>>1,
				F+=cross(x-Rf,x+Rf,y-Rs,y+Rs)*df,
				E+=cross(x-Re,x+Re,y-Rs,y+Rs)*de;
			res[cnt++]=F-E,sum+=F;
			}
		std::sort(res,res+cnt);
		for (int i=0;i<ne;i++) sum-=res[i];
		if (sum>ans) ans=sum;
		}
}
int main()
{
	scanf("%d%d%d%d%d%d%d%d",&nf,&ne,&ns,&rf,&re,&rs,&df,&de),n=nf+ne+ns;
	Rf=sqrt(rf*rf-1.),Re=sqrt(re*re-1.),Rs=sqrt(rs*rs-1.),tot+=!ns;
	for (int i=0;i<n;i++) g[1<<i]=i;
	for (int i=1;i<1<<n;i++) g[i]=g[i&-i],c[i]=c[i-(i&-i)]+1,c[i]==ns?a[tot++]=i:0;
	work(),printf("%.9lf\n",ans); return 0;
}
