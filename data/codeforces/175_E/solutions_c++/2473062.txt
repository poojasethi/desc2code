#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<iostream>
#include<algorithm>
using namespace std;
int nf,ne,ns,n,df,de;
double rf,re,rs,c[100],ans=0;
int e[40],w[40],q[1<<21],h[1<<21];
double ask(const double A,const double B,const double X,const double Y)
{
	return max(0.0,min(B,Y)-max(A,X));
}
void G(const int x)
{
	if ( h[x]!=ns ) return;
	int cp=0;double E,F,now=0;
	for(int i=(1<<n)-1-x,k;i>0;i-=i&-i )
	{
		k=(w[(i&-i)%29]+1)/2;
		E=0,F=0;
		for(int j=x,l;j>0;j-=j&-j)
		{
			l=(w[(j&-j)%29]+1)/2;
			E+=de*ask(k-re,k+re,l-rs,l+rs);
			F+=df*ask(k-rf,k+rf,l-rs,l+rs);
		}
		now+=E;
		c[++cp]=E-F;
	}
	sort(c+1,c+1+cp);
	for(int i=1;i<=nf;++i) now-=c[i];
	if ( now>ans ) ans=now;
}
int main()
{
	scanf("%d%d%d",&nf,&ne,&ns);
	n=nf+ne+ns;
	e[1]=1,w[1]=1;
	for(int i=2;i<=n;++i) e[i]=e[i-1]<<1,w[e[i]%29]=i;
	scanf("%lf%lf%lf",&rf,&re,&rs);
	scanf("%d%d",&df,&de);
	rf=sqrt(rf*rf-1);re=sqrt(re*re-1);rs=sqrt(rs*rs-1);
	h[0]=0;
	G(0);
	for(int i=1;i<(1<<n);++i) h[i]=h[i-(i&-i)]+1,G(i);
	ans+=nf*rf*df*2+ne*re*de*2;
	printf("%.10lf\n",ans);
	return 0;
}
