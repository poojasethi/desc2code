#include<cstdio>
#include<cmath>
#include<algorithm>
#define low(x) x&(-x)
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=22,mm=1210000;
int n,m,i,j,s,tot,r,t;
int nf,ne,ns,x,y,L;
int cnt[mm],lg[mm];
double rf,re,rs,df,de;
double ans,now,v[mn];
double cross(double a,double b,double x,double y)
{
    return max(0.0,min(b,y)-max(a,x));
}
int main()
{
    scanf("%d%d%d",&nf,&ne,&ns);
    scanf("%lf%lf%lf",&rf,&re,&rs);
    scanf("%lf%lf",&df,&de);
    rf=sqrt(rf*rf-1),re=sqrt(re*re-1),rs=sqrt(rs*rs-1);
    n=nf+ne+ns,tot=(1<<n)-1;
    cnt[1]=1;
    fo(i,2,tot)cnt[i]=cnt[i>>1]+(i&1),lg[i]=lg[i>>1]+1;
    ans=0.0;
    fo(s,0,tot)
    {
        if(cnt[s]!=ns)continue;
        L=0;
        now=0.0;
        for(r=s^tot;r;r-=low(r))
        {
            x=lg[low(r)]>>1;
            double vf=2*rf*df,ve=2*re*de;
            for(t=s;t;t-=low(t))
            {
                y=lg[low(t)]/2;
                vf+=cross(x-rf,x+rf,y-rs,y+rs)*df;
                ve+=cross(x-re,x+re,y-rs,y+rs)*de;
            }
            now+=vf,v[++L]=vf-ve;
        }
        sort(v+1,v+1+L);
        fo(i,1,ne)now-=v[i];
        ans=max(now,ans);
    }
    printf("%.10lf\n",ans);
    return 0;
}
