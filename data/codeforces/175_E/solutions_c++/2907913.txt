double rf,re,rs,df,de,ans;
double srt[22];
int N,nf,ne,ns;
double lt(double x1,double y1,double x2,double y2){
	if (x1<x2) x1=x2;
	if (y1>y2) y1=y2;
	y1-=x1;
	if (y1<0) y1=0;
	return y1;
}
int que[23];
bool vis[23];
void Chk(){
	double now=0;
	int l=0;
	for (int i=0;i<nf+ne+ns;i++)
	if (!vis[i]){
		double L1=rf*2,L2=re*2;
		for (int j=0;j<ns;j++){
			L1+=lt(-rs+que[j]/2,rs+que[j]/2,-rf+i/2,rf+i/2);
			L2+=lt(-rs+que[j]/2,rs+que[j]/2,-re+i/2,re+i/2);
		}
		L1*=df;L2*=de;
		now+=L1;
		srt[++l]=L2-L1;
	}
	sort(srt+1,srt+l+1);
	for (int i=l;i>l-ne;i--) now+=srt[i];
	if (now>ans) ans=now;
}
void dfs(int x,int y){
	if (y==ns||x==nf+ne+ns){
		if (y==ns)Chk();
		return;
	}
	dfs(x+1,y);
	que[y]=x;
	vis[x]=true;
	dfs(x+1,y+1);
	vis[x]=false;
}

	rf=sqrt(rf*rf-1);re=sqrt(re*re-1);rs=sqrt(rs*rs-1);

}
