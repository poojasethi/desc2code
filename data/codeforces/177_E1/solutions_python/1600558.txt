#include <iostream>

using namespace std;

int main()
{
    int i, j, k, n, c, *a, *b, x;
    cin >> n >> c;
    a = new int[n];
    b = new int[n];
    x = 0;
    for(i = 0; i < n; i++){
        cin >> a[i] >> b[i];
    }
    //
    x = 1;
    j = 0;
    do{
        k = n;
        for(i = 0; i < n; i++){
            k += a[i]*x / b[i];
        }
        if (k == c)
            j++;
        x++;
    }while(k <= c);
    //
    cout << j << "\n";
    return 0;
}
