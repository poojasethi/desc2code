#include<stdio.h>
int main(void)
{
	int i,n;
	int count;
    int c;
	int  a[10005],b[10005];
	double sum=0;
	double r;
	int k,s;
	
	scanf("%d%d",&n,&c);
	
	for(i=0;i<n;i++){
		scanf("%d%d",&a[i],&b[i]);
		//if(b[i]!=0){
		sum=sum+a[i]*1.0/b[i];
	//	}

		
		
		
		
	}

	c=c-n;
	r=c*1.0/sum;
	k=(int)r+1;
	
	s=0;
	count=0;
	while(s<=c){
		
		for(i=0;i<n;i++){
			//if(k<a[i]){
			//break;
			//}
			
			
			s=s+k*a[i]/b[i];
			
		}
		
		if(s==c){
			count++;
			s=0;
			k++;
		}
		
		else if(s>c){
			break;
		}
		
		else {
			k++;
			s=0;
		}
		
		
		
		
	}
	
	
	printf("%d\n",count);
	
	return 0;
}
