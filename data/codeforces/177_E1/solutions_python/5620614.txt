#include<iostream>

using namespace std;

int a[101],b[101],c,n;

int main()
{
    cin>>n>>c;
    for (int i=0;i<n;i++)
        cin>>a[i]>>b[i];
    int res = 0,s;
    for (int x=1;x<=500000;x++)
    {
        s = 0;
        for (int i=0;i<n;i++)
            s += 1 + a[i]*x/b[i];
        res += (c == s);
    }
    
    cout<<res<<endl;
    return 0;
}
