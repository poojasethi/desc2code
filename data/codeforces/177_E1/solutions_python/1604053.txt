#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <queue>
#include <stack>
#include <vector>
#include <map>
#include <string>
#define INF 2000000000
#define ii pair<int, int>

using namespace std;

int main() {
	int n,c,a[105],b[105],res;
	while (scanf("%d %d",&n,&c)!=EOF) {
		res=0;
		for (int i=0;i<n;i++) {
			scanf("%d %d",&a[i],&b[i]);	
		}
		for (int x=1;x<=10000;x++) {
			int jum=0;
			for (int i=0;i<n;i++) {
				jum++;
				jum=jum+((a[i]*x)/b[i]);
			}
			
			if (jum==c) {res++;}
		}
		printf("%d\n",res);	
	}
    return 0;
}
