#include <cstdio>
#include <iostream>

using namespace std;

typedef long long ll;

const int maxn = 10 * 1000 + 5;

int a[maxn], b[maxn];

int n;

ll func(ll x)
{
  ll res = 0;
  for (int i = 0; i < n; ++i) {
    res += 1LL + x * a[i] / (ll)b[i];
  }
  return res;
}

int bin_search(ll c)
{
  int l = 0, r = 1 << 30;
  while (l + 1 < r) {
    int x = (l + r) >> 1;
    if (func(x) < c) {
      l = x;
    }
    else {
      r = x;
    }
  }

  return l;
}

int main()
{
  int c;
  cin >> n >> c;
  for (int i = 0; i < n; ++i) {
    cin >> a[i] >> b[i];
  }

  int res = bin_search(c + 1) - bin_search(c);
  cout << res << '\n';
  
  return 0;
}
