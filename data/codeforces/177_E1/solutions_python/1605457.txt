#include <iostream>
#include <memory.h>
#include <algorithm>
#include <cstdio>
#include <cstdlib>
using namespace std;

int a[101],b[101];

int main()
{
  int n,c;
  while(scanf("%d %d",&n,&c) != EOF)
    {
   
      for(int i = 1;i <= n;i ++)
	scanf("%d %d",a + i,b + i);
      int ans = 0;
      for(int i = 1;i <= 10000;i ++)
	{
	  int sum = 0;
	  for(int j = 1;j <= n;j ++)
	    {
	      sum += a[j] * i / b[j] + 1;
	    }
	  if(sum == c) ans ++;
	}
      printf("%d\n",ans);
    }
}
