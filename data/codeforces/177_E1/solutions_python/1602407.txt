#include <cstdio>
#define len1 20001
#define oo 1000000000000000000LL
long long n , a[len1] , b[len1] , c ;
long long ans ;

int main()
{
    int i , j ;
    long long l , r , mid , t ;
    int allzero = 0 ;
    
    scanf("%I64d %I64d",&n,&c);
    c -= n ;
    
    for ( i = 1 ; i <= n ; ++i )
    {
        scanf("%I64d %I64d",&a[i],&b[i]);
        if ( a[i] != 0 )
            allzero = 1 ;
    }
    ans = 0 ;
    for ( i = 1 ;  ; ++i )
    {
        t = 0 ;
        for ( j = 1 ; j <= n ; ++j )
        {
            t += i*a[j]/b[j] ;
        }
        if ( t == c )
            ans ++ ;
        if ( t > c )
            break;
    }
    printf("%I64d",ans);
    
    return 0;
}
