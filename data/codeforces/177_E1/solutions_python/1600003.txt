#include <iostream>
#include <stdio.h>

using namespace std;

long long n, c, a[100005], b[100005], mx, s, ans;

int main()
{
    //freopen("input.txt", "r", stdin);
    cin >> n >> c;
    for (long long i = 0; i < n; i++)
        cin >> a[i] >> b[i];
    for (int x = 1; s <= c; x++)
    {
        s = 0;
        for (int j = 0; j < n; j++)
            s += a[j] * x / b[j] + 1;
        if (s == c)
            ans++;
    }
    cout << ans;
}
