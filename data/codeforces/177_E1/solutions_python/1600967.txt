#include <cstdlib>
#include <cstring>
#include <cstdio>

int n,c,a[10010],b[10010];

void init()
{
	int i;
	scanf("%d%d",&n,&c);
	for (i=0;i<n;i++)scanf("%d%d",&a[i],&b[i]);
}

void small()
{
	int ans=0;
	int i,j,x,day;
	for (x=1;;x++)
	{
		day=0;
		for (i=0;i<n;i++)
		{
			day+=1+a[i]*x/b[i];
		}
		if (day==c)ans++;
		if (day>c)break ;
	}
	printf("%d\n",ans);
}

int main()
{
	init();
	small();
	return 0;
}
