#include <iostream>
#include <vector>
using namespace std;

int main() {
	int n, c;
	cin >> n >> c;
	vector< pair<int, int> > arr(n);
	bool t = false;
	for (int i = 0; i < n; ++i) {
		cin >> arr[i].first >> arr[i].second;
		if (arr[i].first) {
			t = true;
		}
	}
	if (!t) {
		cout << "-1";
		return 0;
	}
	c -= n;
	int x = 0, mn = 0, mx = 0;
	while (t) {
		++x;
		int f = 0;
		for (int i = 0; i < n; ++i) {
			f += arr[i].first * x / arr[i].second;
		}
		if (f < c) {
			mn = x;
		}
		if (f > c) {
			mx = x;
			t = false;
		}
	}
	cout << mx - mn - 1;
	return 0;
}
