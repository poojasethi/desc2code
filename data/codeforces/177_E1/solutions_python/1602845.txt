#include <iostream>
#include <stdio.h>
using namespace std;
typedef long long LL;
LL n,x;
LL a[11111],b[11111];


LL calc(LL x)
{
    LL ret=0;
    for(int i=0;i<n;i++)
    {
        ret+=(x*a[i]/b[i])+1;
    }
    return ret;
}


int main()
{
    cin>>n>>x;
for(int i=0;i<n;i++) cin>>a[i]>>b[i];
LL ans=0;
   if(x<n) ans=0;
   else
   {
        for(int i=1; ;i++)
        {
            LL t=calc(i);
            if(t==x) ans++;
            if(t>x) break;
        }
   }
    cout<<ans<<endl;

}
