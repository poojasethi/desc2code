#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

int n, c;
int a[1000], b[1000];

int go(int x)
{
    int res = 0;
    for (int i=1; i<=n; i++)
        res += a[i]*x / b[i] + 1;
    return res;
}

int main()
{
#ifndef ONLINE_JUDGE
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
#endif
    cin >> n >> c;
    for (int i=1; i<=n; i++)
        cin >> a[i] >> b[i];
    int ans = 0;
    for (int x = 1;; x++)
    {
        int k = go(x);
        if (k == c)
            ans++;
        else
            if (k > c)
                break;
    }
    cout << ans;
    return 0;
}
