#include <cstdio>
#include <iostream>
#include <climits>

using namespace std;

#define LL unsigned long long

int n;
LL c;
LL a[10000];
LL b[10000];

LL calc(LL mid)
{
	LL res = 0;
	for (int i = 0; i < n; ++i)
		res += 1+a[i]*mid/b[i];
	return res;
}

int main()
{
	cin >> n >> c;
	for (int i = 0; i < n; ++i)
		cin >> a[i] >> b[i];
	
	LL bot = INT_MAX, top = 0;
	
	for (LL i = 1; i <= 10000; ++i)
	{
		if (calc(i) == c) bot = min(bot, i), top = max(top, i);
	}
	
	if (bot <= top) cout << top-bot+1 << endl;
	else cout << 0 << endl;
	
	return 0;
}
