#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;

#define FOR(i, a, n) for (int i = (a); i < (n); ++i)
#define REP(i, n) FOR(i, 0, n)

int n, c;
int a[10010], b[10010];

int main() {
    scanf("%d%d", &n, &c);
    REP(i, n) scanf("%d%d", &a[i], &b[i]);

    if (c < n) {
        printf("0\n");
        return 0;
    }
    
    int ans = 0;
    for (int x = 1; ; ++x) {
        int sum = 0;
        REP(i, n)
            sum += a[i] * x / b[i];
        if (sum == c - n)
            ++ans;
        if (sum > c - n)
            break;
    }
    printf("%d\n", ans);
    return 0;
}
