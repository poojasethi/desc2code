#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<string.h>
#include<algorithm>
#include<iostream>
#define clr(x,y) memset(x,y,sizeof(x))
#define ll long long
using namespace std;

struct item{
    ll a,b;
}p[100000];
int i,n,c;

int main(){
//    freopen("e:\\in.txt","r",stdin);
    scanf("%d%d",&n,&c);
    for(i=1;i<=n;i++)
        scanf("%I64d%I64d",&p[i].a,&p[i].b);
    int ans=0;
    for(int x=1;x<=100000;x++){
        int t=0;
        for(i=1;i<=n;i++){
            t+=(p[i].a*x)/p[i].b+1;
        }
        if(t==c) ans++;
    }
    printf("%d\n",ans);
    return 0;
}
