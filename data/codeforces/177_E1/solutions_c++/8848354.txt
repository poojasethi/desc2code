#include<iostream>

using namespace std;

#define int long long
const int MAXN = 10004;
int a[MAXN], b[MAXN];
int n, c;

int check(int x)
{
	int cnt = 0;
	for(int i = 0; i < n; i++)
		cnt += x * a[i] / b[i] + 1;
	return cnt;
}

main()
{
	cin >> n >> c;
	for(int i = 0; i < n; i++)
		cin >> a[i] >> b[i];
	int i = 1;
	while(check(i) < c)
		i++;
	int ans = 0;
	while(check(i) == c)
	{
		i++;
		ans++;
	}
	cout << ans << endl;
	return 0;
}
