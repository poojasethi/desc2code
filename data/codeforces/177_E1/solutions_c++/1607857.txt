#include<stdio.h>

int main()
{
    int n,x,c;
    int xx[1111][2];
    int i;
    scanf("%d %d",&n,&c);
    int max=10000000;
    for(i=0;i<n;i++)
    {
        scanf("%d %d",&xx[i][0],&xx[i][1]);
    }
    int l=1,r=max,cnt=0;
    for(i=l;i<=r;i++)
    {
        int sum=0;
        for(int j=0;j<n;j++)sum=sum+(1+xx[j][0]*i/xx[j][1]);
        if(sum==c)
        {
            cnt++;
        }
        else if(sum>c)break;
    }
    printf("%d\n",cnt);
}
