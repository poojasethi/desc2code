#include <iostream>
#define N 100000

using namespace std;

int n, c, a[N+1], b[N+1], ans = 0;

int main (int argc, const char * argv[])
{
    cin >> n >> c;
    for(int i = 1; i <= n; i++)
        cin >> a[i] >> b[i];
    for(int x = 1; x <= N; x++)
    {
        int s = 0;
        for(int i = 1; i <= n; i++)
            s += (a[i] * x) / b[i];
        if(s == c - n)
            ans++;
    }
    cout << ans;
    return 0;
}
