#include<iostream>

using namespace std;


// there could not be infinite ans- as 1(suitcase)*1000000(x)/100(max value of people)>100 so x > 1000000
int main()
{
    int a[101],b[101],c,n;
    cin >> n >> c;
    for (int i=0;i<n;i++)
        cin>>a[i]>>b[i];
    int res = 0,s;
    for (int x=1;x<=100000;x++)
    {
        s = 0;
        for (int i=0;i<n;i++)
            s += 1 + a[i]*x/b[i];
        res += (c == s);
    }
    
    cout<<res<<endl;
    return 0;
}
