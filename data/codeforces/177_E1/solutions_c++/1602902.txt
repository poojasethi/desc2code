#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cmath>
#include <vector>
#define ll long
#define maxn 10003
#define FOR(i,a,b) for(ll i=a;i<=b;i++)
#define FO(i,a,b) for(ll i=a;i<b;i++)
#define FORD(i,a,b) for(ll i=a;i>=b;i--)
#define FOD(i,a,b) for(ll i=a;i>b;i--)
#define pb push_back
#define mp make_pair
using namespace std;
int a[maxn],b[maxn];
int n,c,ans;
main() {
	//freopen("test.in","r",stdin);
	//freopen("test.out","w",stdout);
	cin >> n >> c;
	FOR (i,1,n) cin >> a[i] >> b[i];
	int x=0,d;
	while (1) {
		x++;
		d=0;
		FOR (i,1,n) d+=1+a[i]*x/b[i];
		if (d==c) ans++;
		if (d>c) break;
	}
	cout << ans;
}
