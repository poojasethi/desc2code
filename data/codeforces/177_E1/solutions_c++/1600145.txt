#include <iostream>
#include <cstdio>
#include <cmath>
#define MAXN 111111
#define MAXX 11111

using namespace std;

int main(){
//    freopen("a.in", "rt", stdin);
//    freopen("a.out", "wt", stdout);
    int n, c, a[MAXN], b[MAXN], res = 0, s;
    scanf("%d%d", &n, &c);
    for (int i = 1; i <= n; i++) scanf("%d%d", &a[i], &b[i]);
    for (int i = 1; i <= MAXX; i++){
        s = 0;
        for (int j = 1; j <= n; j++)
            s += (a[j] * i) / b[j] + 1;
        if (s == c) res++;
    }
    printf("%d\n", res);
    
    return 0;
}
