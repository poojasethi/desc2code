#include<iostream>
#include<vector>
#include<cmath>
#include<queue>
#include<algorithm>
using namespace std;
int n,c,res,x,a[101],b[101],i,j;
bool solve(int x)
{
    int day=0;
    for(int i=1;i<=n;i++)
       {
        day++;
        int r=a[i]*x;
        day+=r/b[i]; 
    }
    return day==c;
}
int main()
{
    cin>>n>>c;
    for(i=1;i<=n;i++)
        cin>>a[i]>>b[i];
    for(x=1;x<=100000;x++)
        if(solve(x)) res++;
    cout<<res<<endl;
}
