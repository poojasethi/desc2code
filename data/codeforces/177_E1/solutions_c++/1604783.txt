#include<iostream>
#include<algorithm>
#include<vector>
#include<string>
#include<set>
using namespace std;

#define rep(i,n) for(int i=0;i<n;i++)
#define rev(i,n) for(int i=n-1;i>=0;i--)

int main(){
   int n,c;
   cin >> n >> c;
   vector<int> a,b;
   a.reserve(n);
   b.reserve(n);
   int aa,bb;
   rep(i,n){
      cin >> aa >> bb;
      a.push_back(aa);
      b.push_back(bb);
   }

   int count=0;
   int xmax=20000;
   int sum;
   for(int x=1;x<=xmax;x++){
      sum=0;
      rep(i,n){
         sum+=1+(a[i]*x)/b[i];
      }
      if(sum==c){
	 count++;
	 //cout << "x=" << x << endl;
      }
   }

   cout << count << endl;

   return 0;
}
