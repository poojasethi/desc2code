#include <iostream>
#include <stdio.h>
#define Max 10010
#define Max_c 1000000001
using namespace std;

struct Node{
    int a;
    int b;
}node[Max];
int main()
{
    int n,c,i,j,count,k=0;
    int x;
    cin>>n>>c;
    for(i =1 ;i <= n; i++) scanf("%d%d",&node[i].a,&node[i].b);
    for(x =1; x< Max_c; x++){
        count =0;
        for(i =1; i<= n; i++){
            count+= 1 + (int)(node[i].a/(node[i].b*1.0)*x);
            if(count > c) break;
        }
        if(count == c) k++;
        if(count >c) break;
    }
    cout<<k<<endl;
    return 0;
}
