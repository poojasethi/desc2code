#include <iostream>
using namespace std;
int n,m,i,c;
long long a[111111],b[111111],be;
long long ok(long long x)
{
	long long k0=0;
	for(int i=0;i<n;i++)
	{
		k0+=(x*a[i])/b[i]+1;
		if (k0>c) return k0;
	}
	return k0;
}
int main()
{
	cin>>n>>c;
	for(i=0;i<n;i++)
		cin>>a[i]>>b[i];
	long long l=0,r=1000000000000LL;
	while (l+1<r)
	{
		long long lr=(l+r)/2,k=ok(lr);
		if (k<c)
			l=lr;
		else r=lr;
	}
	be=r;
	l=0; r=1000000000000LL;
	while (l+1<r)
	{
		long long lr=(l+r)/2,k=ok(lr);
		if (k>c)
			r=lr;
		else l=lr;
	}
	cout<<l-be+1<<endl;
    return 0;
}
