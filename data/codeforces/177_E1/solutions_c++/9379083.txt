#include <stdio.h>
int n,c;
int a[10010],b[10010];
int main(void){
    while(scanf("%d%d",&n,&c)!=EOF){
        for(int i=0;i<n;i++){
            scanf("%d%d",&a[i],&b[i]);
        }
        if(n>c){
            printf("0\n");
            continue;
        }
        c = c - n;
        int res = -1;
        int x = 1;
        int ans = 0;
        while(x<=200000){
            if(res==c) ans++;
            res = 0;
            for(int i=0;i<n;i++)
                res += (a[i]*x)/b[i];
            x++;
        }
        printf("%d\n",ans);
    }
    
    return 0;
}
