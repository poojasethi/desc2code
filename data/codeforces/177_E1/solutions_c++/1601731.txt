#include<cstdio>
#include<iostream>
#include<cmath>
using namespace std;
#define MAXN 10010
#define ll long long

int N,C;
int a[MAXN],b[MAXN];

bool check(int k)
{
    int t=0;
    for(int i=1;i<=N;i++)
    {
        t+=(k*a[i])/b[i]+1;
        if(t>C) return false;
    }
    return t==C;
}
int main()
{
    cin>>N>>C;
    for(int i=1;i<=N;i++) scanf("%d%d",a+i,b+i);
    int ans=0;
    for(int i=1;i<=40000;i++)
        if(check(i)) ans++;
    cout<<ans<<endl;
    return 0;
}
