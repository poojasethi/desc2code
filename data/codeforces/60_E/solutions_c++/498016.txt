	matrix(LL a = 0, LL b = 0, LL c = 0, LL d = 0)
	{
		v[0][0] = a;
		v[0][1] = b;
		v[1][0] = c;
		v[1][1] = d;
	}
	matrix operator * (matrix t)
	{
		matrix res;
		REP(i, 2)
			REP(j, 2)
				REP(k, 2)
					res.v[i][j] = (res.v[i][j] + mul(v[i][k], t.v[k][j])) % p;
		return res;
	}

	{
		printf("|%d %d|\n|%d %d|\n", v[0][0], v[0][1], v[1][0], v[1][1]);
	}
};

void Pow(matrix &m, LL b)
{
	matrix t = m;
	m = matrix(1, 0, 0, 1);
	while (b)
	{
		if (b & 1)
			m = m*t;
		t = t*t;
		b >>=  1;
	}
}
