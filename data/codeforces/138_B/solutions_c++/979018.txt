#include	<stdlib.h>
#include	<stdio.h>
#include	<string.h>
#include	<math.h>
#include	<map>
#include	<algorithm>
#include	<vector>
#include	<set>
using namespace std;
#define CC(x,i) do{memset(x,i,sizeof(x));}while(0)
#define INFI 2000000000
#define INFF 1e100
#define PII pair<int,int>
#define MP make_pair
#define X first
#define Y second
typedef long long  LL;
char s[123456],s1[123456],s2[123456];
int cnt[12],mx=0,mxz=0,mxt=-1,xx[12];
int main ( int argc, char *argv[] )
{
	CC(cnt,0);
	scanf("%s",s);int ll=strlen(s);
	for(int i=0;s[i];i++){
		cnt[s[i]-'0']++;
	}
	int zo=cnt[0],tmp;
	for(int i=0;i<10;i++)xx[i]=cnt[i];
	for(int z=0;z<=cnt[0];z++){
		cnt[0]=xx[0]=zo-z;tmp=z;
		if(tmp>mx){
			mx=tmp;mxz=z;mxt=-1;
		}
		for(int t=1;t<=5;t++){
			if(cnt[t]&&xx[10-t]){
				int ttmp=tmp+1;
				cnt[t]--;xx[10-t]--;
				for(int s=0;s<10;s++){
					ttmp+=min(cnt[s],xx[9-s]);
				}
				if(ttmp>mx){
					mx=ttmp;
					mxz=z;
					mxt=t;
				}
				cnt[t]++;xx[10-t]++;
			}
		}
		cnt[0]=xx[0]=zo;
	}
	int pt=ll-1;s1[ll]=s2[ll]=0;
	for(int i=0;i<mxz;i++){
		cnt[0]--,xx[0]--;s1[pt]=s2[pt]='0';pt--;
	}
	if(mxt>-1){
		cnt[mxt]--;xx[10-mxt]--;s1[pt]=mxt+'0';s2[pt]=10-mxt+'0';pt--;
	}
	for(int i=0;i<10;i++){
		int dd=min(cnt[i],xx[9-i]);
		for(int j=0;j<dd;j++){
			cnt[i]--;xx[9-i]--;
			s1[pt]=i+'0';s2[pt]=9-i+'0';
			pt--;
		}
	}
	for(int i=0;i<10;i++){
		while(cnt[i]){printf("%d",i);cnt[i]--;}
	}
	printf("%s\n",s1+pt+1);
	for(int i=0;i<10;i++){
		while(xx[i]){printf("%d",i);xx[i]--;}
	}
	printf("%s\n",s2+pt+1);

	return EXIT_SUCCESS;
}
