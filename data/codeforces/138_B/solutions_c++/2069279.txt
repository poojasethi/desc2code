#include<cstdio>
#include<string>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
    char s[100001]; gets(s);

    int f[10]={};
    for(int i=0;s[i];i++) f[s[i]-'0']++;

    int z_ans=-1;
    string s_ans1,s_ans2;
    for(int d=1;d<=9;d++){
        int f1[10],f2[10];
        copy(f,f+10,f1);
        copy(f,f+10,f2);

        int z=0;
        string s1,s2;
        if(f1[d]>0 && f2[10-d]>0){
            z++;
            s1+='0'+d;
            s2+='0'+10-d;
            f1[d]--;
            f2[10-d]--;

            rep(e,10){
                int tmp=min(f1[e],f2[9-e]);
                z+=tmp;
                s1+=string(tmp,'0'+ e );
                s2+=string(tmp,'0'+9-e);
                f1[ e ]-=tmp;
                f2[9-e]-=tmp;
            }
        }
        if(f1[0]>0 && f2[0]>0){
            int tmp=min(f1[0],f2[0]);
            z+=tmp;
            s1=string(tmp,'0')+s1;
            s2=string(tmp,'0')+s2;
            f1[0]-=tmp;
            f2[0]-=tmp;
        }
        rep(e,10){
            s1+=string(f1[e],'0'+e);
            s2+=string(f2[e],'0'+e);
        }

        if(z_ans<z){
            z_ans=z;
            s_ans1=s1;
            s_ans2=s2;
        }
    }
    reverse(s_ans1.begin(),s_ans1.end());
    reverse(s_ans2.begin(),s_ans2.end());

    puts(s_ans1.c_str());
    puts(s_ans2.c_str());

    return 0;
}
