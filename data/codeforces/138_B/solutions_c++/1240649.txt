#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;
char a[100005];
char ra[100005];
char rb[100005];
string aa, ab;
int c[10];
int main(void) {
	scanf("%s", a);
	for(int i=0;a[i];i++) c[a[i]-'0']++;
	int bestv=-1;
	sort(a, a+strlen(a));
	aa = a;
	ab = a;
	for(int i=1;i<=5;i++) {
		if(c[i]&&c[10-i]) {
			int da[10]={}, db[10]={};
			for(int j=0;j<10;j++) da[j]=db[j]=c[j];
			--da[i]; --db[10-i];
			int m=0, v=0;
			
			ra[m] = i+'0';
			rb[m++] = 10-i+'0';
			for(int j=0;j<10;j++) {
				while(da[j]>0 && db[9-j]>0) {
					--da[j]; --db[9-j];
					ra[m] = j+'0'; rb[m++] = 9-j+'0';
					++v;
				}
			}
			int r1=m, r2=m;
			for(int j=1;j<10;j++) while(da[j]>0) {--da[j]; ra[r1++]=j+'0';}
			for(int j=1;j<10;j++) while(db[j]>0) {--db[j]; rb[r2++]=j+'0';}
			reverse(ra, ra+r1);
			reverse(rb, rb+r2);
			while(da[0]>0 && db[0]>0) {
				--da[0]; --db[0];
				ra[r1++]='0'; rb[r2++]='0';
				++v;
			}
			reverse(ra, ra+r1);
			reverse(rb, rb+r2);
			while(da[0]>0) { --da[0]; ra[r1++]='0';}
			while(db[0]>0) { --db[0]; rb[r2++]='0';}

			ra[r1]=0; rb[r2]=0;
			if(bestv < v) {
				bestv = v;
				aa = ra; ab = rb;
			}
		}
	}
	reverse(aa.begin(), aa.end());
	reverse(ab.begin(), ab.end());
	printf("%s\n%s\n", aa.c_str(), ab.c_str());
	return 0;
}
