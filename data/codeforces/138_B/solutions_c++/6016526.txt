#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int ocurrencias[11];

int o1[11],o2[11];

int calidadsol=-1;
vector<int> solfin1,solfin2;

void prueba(int d)
{
  for (int i=0;i<10;i++)
    o1[i]=o2[i]=ocurrencias[i];
  vector<int> sol1,sol2;
  int calidad=0;
  if (o1[d]>0 and o2[10-d]>0) {
    sol1.push_back(d);
    sol2.push_back(10-d);
    o1[d]--;
    o2[10-d]--;
    calidad++;
  }
  if (calidad) {
    for (d=0;d<10;d++) {
      while (o1[d] and o2[9-d]) {
	o1[d]--;
	o2[9-d]--;
	sol1.push_back(d);
	sol2.push_back(9-d);
	calidad++;
      }
    }
  }
  vector<int> nextsol1,nextsol2;
  int mi=min(o1[0],o2[0]);
  for (int i=0;i<mi;i++) {
    nextsol1.push_back(0);
    nextsol2.push_back(0);
    if (calidad) calidad++;
  }
  o1[0]-=mi;
  o2[0]-=mi;
  for (int i=0;i<int(sol1.size());i++) {
    nextsol1.push_back(sol1[i]);
    nextsol2.push_back(sol2[i]);
  }
  sol1=nextsol1;
  sol2=nextsol2;
  for (d=0;d<10;d++) {
    for (int d2=0;d2<10;d2++) {
      while (o1[d] and o2[d2]) {
	o1[d]--;
	o2[d2]--;
	sol1.push_back(d);
	sol2.push_back(d2);
      }
    }
  }
  if (calidad>calidadsol) {
    solfin1=sol1;
    solfin2=sol2;
    calidadsol=calidad;
  }
}

int main()
{
  {
    char c;
    while (cin>>c)
      ocurrencias[c-'0']++;
  }
  for (int d=1;d<10;d++)
    prueba(d);
  for (int i=int(solfin1.size())-1;i>=0;i--)
    cout<<solfin1[i];
  cout<<endl;
  for (int i=int(solfin2.size())-1;i>=0;i--)
    cout<<solfin2[i];
  cout<<endl;
}
