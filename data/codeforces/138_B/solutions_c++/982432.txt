#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;
int main() {
    string n;
    cin >> n;
    vector<int> a(10);
    for (int i = 0; i < (int)n.length(); ++ i) {
        ++ a[n[i]-'0'];
    }
    int r = a[0];
    int ry = 0, rz = a[0];
    for (int z = 0; z <= a[0]; ++ z) {
        for (int y = 1; y <= 9; ++ y) {
            if (a[y] && a[10-y]) {
                vector<int> X = a, Y = a;
                X[0] -= z; Y[0] -= z;
                -- X[y]; -- Y[10-y];
                int rr = z+1;
                for (int x = 0; x <= 9; ++ x) {
                    rr += min(X[x], Y[9-x]);
                }
                if (rr > r) { r = rr; rz = z; ry = y; }
            }
        }
    }
    vector<int> X = a;
    vector<int> Y = a;
    string sx(rz, '0');
    string sy(rz, '0');
    X[0] -= rz; Y[0] -= rz;
    if (ry) {
        sx += (char)('0'+ry);
        sy += (char)('0'+10-ry);
        -- X[ry]; -- Y[10-ry];
        for (int x = 0; x <= 9; ++ x) {
            int nx = min(X[x], Y[9-x]);
            sx += string(nx, (char)('0'+x));
            sy += string(nx, (char)('0'+9-x));
            X[x] -= nx; Y[9-x] -= nx;
        }
    }
    for (int i = 0; i < 10; ++ i) {
        for (int j = 0; j < X[i]; ++ j) sx += (char)('0'+i);
        for (int j = 0; j < Y[i]; ++ j) sy += (char)('0'+i);
    }
    reverse(sx.begin(),sx.end());
    reverse(sy.begin(),sy.end());
    cout << sx << endl;
    cout << sy << endl;
}
