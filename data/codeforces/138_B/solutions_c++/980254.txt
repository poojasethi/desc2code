#include <iostream>
#include <cstdio>

using namespace std;

char str[200005]={0};
long have[10]={0};
long have2[10]={0};
long a[2000005]={0};
long b[2000005]={0};

int main()
{
 long i,j;
 long ans=-1;
 long p=0;
 long o1=0;
 long o2=0;
 
 scanf("%s",str);
 for(i=0;str[i];i++)
   {
    have[str[i]-'0']++;
    have2[str[i]-'0']++;
   }
 for(i=1;i<=5;i++)
   if(have[i]&&have2[10-i])
   {
    have[i]--;
    have2[10-i]--;
    long sum=1;
    for(j=1;j<=8;j++)
      {
       sum+=min(have[j],have2[9-j]);
      }
    sum+=min(have[9],have2[0]);
    sum+=min(have[0],have2[9]);
    sum+=min(have[0]-min(have[0],have2[9]),have2[0]-min(have[9],have2[0]));
    if(sum>ans)
     ans=sum,p=i;
    
    have[i]++;
    have2[10-i]++;
   }
 if(ans==-1)
  {
   for(i=0;i<=9;i++)
     for(;have[i];have[i]--)
       a[o1++]=i,b[o2++]=i;
  }
 else
  { 
   have[p]--;
   have2[10-p]--;
   for(i=1;i<=min(have[0]-min(have[0],have2[9]),have2[0]-min(have[9],have2[0]));i++)
     a[o1++]=b[o2++]=0;
   a[o1++]=p;
   b[o2++]=10-p;
  
   have[0]-=i-1;
   have2[0]-=i-1;
   for(i=0;i<=9;i++)
     for(;have[i]&&have2[9-i];have[i]--,have2[9-i]--)
       {
        a[o1++]=i;
        b[o2++]=9-i;
       }
   for(i=0;i<=9;i++)
     for(;have[i];have[i]--)
       a[o1++]=i;
   for(i=0;i<=9;i++)
     for(;have2[i];have2[i]--)
       b[o2++]=i;
  }
 
 for(i=o1-1;i>=0;i--)
   printf("%ld",a[i]);
 printf("\n");
 for(i=o2-1;i>=0;i--)
   printf("%ld",b[i]);
 
 return 0;
}
