#include <bits/stdc++.h>

using namespace std;

int a[100005],b[100005],cnt[10];
char k[100005];

int main()
{
    while(scanf("%s",k+1)!=EOF)
	{
		memset(cnt,0,sizeof(cnt));
		int l=strlen(k+1);
		int pos=l;
		for(int i=1;i<=l;i++)
			cnt[k[i]-'0']++;
		int ca[10],cb[10];
		for(int i=0;i<=9;i++)
			ca[i]=cb[i]=cnt[i];
		for(int i=1;i<=ca[0]-ca[9];i++)
			a[l-i+1]=b[l-i+1]=0;
		pos=pos-max(0,ca[0]-ca[9]);
		cb[0]=ca[0]=min(ca[0],ca[9]);
		int ans=0,tag=-1;
		for(int i=1;i<=5;i++)
		{
			int cal=1;
			if(ca[i]==0||cb[10-i]==0)
				continue;
			else
				ca[i]--,cb[10-i]--;
			for(int j=0;j<5;j++)
				cal+=min(ca[j],cb[9-j]);
			if(cal>ans)
				ans=cal,tag=i;
			ca[i]++,cb[10-i]++;
		}
		if(tag>0)
		{
			if(tag==1&&ca[0]==ca[9]&&ca[0]>0)
				a[pos]=0,b[pos]=0,pos--,ca[0]--,cb[0]--;
			a[pos]=tag,b[pos]=10-tag,pos--,ca[tag]--,cb[10-tag]--;
		}
		for(int i=0;i<=9;i++)
			while(ca[i]>0&&cb[9-i]>0)
				a[pos]=i,b[pos]=9-i,pos--,ca[i]--,cb[9-i]--;
		int pa=pos,pb=pos;
		for(int i=0;i<=9;i++)
			while(ca[i]>0)
				ca[i]--,a[pa]=i,pa--;
		for(int i=0;i<=9;i++)
			while(cb[i]>0)
				cb[i]--,b[pb]=i,pb--;
		for(int i=1;i<=l;i++)
			printf("%d",a[i]);
		puts("");
		for(int i=1;i<=l;i++)
			printf("%d",b[i]);
		puts("");
	}
	return 0;
}
