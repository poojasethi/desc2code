#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int n,tmp,cnt1[10],cnt2[10],dig[10],ans,po,zero;
char s[101000];

int main()
{
  //  freopen("input.txt","r",stdin);

    cin>>s;
    n=strlen(s);
    ans=-1;
    for (int i=0;i<n;i++) { cnt1[int(s[i])-48]++;cnt2[int(s[i])-48]++; }
    for (int dif=1;dif<=5;dif++)
    {
  		if (cnt1[dif]==0 || cnt2[(10-dif)%10]==0) continue;
        tmp=0;
        cnt1[dif]--;cnt2[(10-dif)%10]--;
        for (int i=0;i<10;i++) tmp+=min(cnt1[i],cnt2[9-i]);
        if (tmp>ans) { ans=tmp;po=dif; }
        cnt1[dif]++;cnt2[(10-dif)%10]++;
    }
 //   if (ans==-1) cout<<s<<endl<<s<<endl;
  //  else {
    if (ans>=0) {  cnt1[po]--;cnt2[(10-po)%10]--;
    		for (int i=0;i<10;i++) { tmp=min(cnt1[i],cnt2[9-i]); cnt1[i]-=tmp; cnt2[9-i]-=tmp; dig[i]=tmp; }
    }
    zero=min(cnt1[0],cnt2[0]);
    cnt1[0]-=zero; cnt2[0]-=zero;
    
    for (int i=9;i>=0;i--) while (cnt1[i]) { cout<<i;cnt1[i]--; }
    for (int i=9;i>=0;i--) for (int j=1;j<=dig[i];j++) cout<<i;
    if (ans>=0) cout<<po;
    for (int i=1;i<=zero;i++) cout<<0;
    cout<<endl;
    
    for (int i=9;i>=0;i--) while (cnt2[i]) { cout<<i;cnt2[i]--; }
    for (int i=0;i<10;i++) for (int j=1;j<=dig[9-i];j++) cout<<i;
    if (ans>=0) cout<<(10-po)%10;
    for (int i=1;i<=zero;i++) cout<<0;
    cout<<endl;
   // }
    
    return 0;
}
