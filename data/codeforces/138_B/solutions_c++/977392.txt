#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

void Output(int x,int y)
{
  while(y--)putchar(x+48);
}

int main()
{
  char c;int A[10],B[10],C[10],D[10];
  memset(A,0,sizeof(A));memset(B,0,sizeof(B));
  c=getchar();while(c<'0'||c>'9')c=getchar();
  while(c>='0'&&c<='9'){A[c-48]++;B[c-48]++;c=getchar();}
  int Max=-1,Choice=-1;
  for(int i=1;i<10;i++)
    if(A[i]&&B[10-i])
      {
        for(int j=0;j<10;j++)C[j]=A[j];
        for(int j=0;j<10;j++)D[j]=B[j];
        C[i]--;D[10-i]--;
        int Sum=1;for(int j=0;j<10;j++)Sum+=min(C[j],D[9-j]);
        Sum+=min(C[0]-min(C[0],D[9]),D[0]-min(D[0],C[9]));
        if(Sum>Max){Max=Sum;Choice=i;}
      }
  if(Choice==-1)
    {
      for(int i=1;i<10;i++)Output(i,A[i]);
      Output(0,A[0]);putchar('\n');
      for(int i=1;i<10;i++)Output(i,A[i]);
      Output(0,A[0]);putchar('\n');
    }
  else
    {
      A[Choice]--;B[10-Choice]--;
      for(int i=0;i<10;i++){C[i]=min(A[i],B[9-i]);A[i]-=C[i];B[9-i]-=C[i];}
      int End=min(A[0],B[0]);A[0]-=End;B[0]-=End;
      for(int i=0;i<10;i++)Output(i,A[i]);
      for(int i=0;i<10;i++)Output(i,C[i]);
      putchar(Choice+48);Output(0,End);putchar('\n');
      for(int i=0;i<10;i++)Output(i,B[i]);
      for(int i=0;i<10;i++)Output(9-i,C[i]);
      putchar(10-Choice+48);Output(0,End);putchar('\n');
    }
  return 0;
}
