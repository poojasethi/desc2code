#include <stdio.h>
#include <string.h>
#define NMAX 100005
inline int cif(char x)
{
	return x>='0' && x<='9';
}
inline int min(int x,int y)
{
	return x<y ? x : y;
}
char A[NMAX];
int n,B[10],C[NMAX],D[NMAX],ap1[10],ap2[10],r,rez,st,rest,rst;
int main()
{
	//freopen("test.in","r",stdin);
	//freopen("test.out","w",stdout);
	fgets(A+1,NMAX,stdin);
	while (cif(A[n+1])){ n++; B[A[n]-'0']++;}
	int i,j,k;
	for (i=1; i<=9; i++)
	{
		for (j=0; j<=9; j++)
			ap1[j]=ap2[j]=B[j];
		if (!ap1[i] || !ap2[10-i])
			continue ;
		r=0;
		ap1[i]--; ap2[10-i]--; r++;
		for (j=0; j<=9; j++)
		{
			while (ap1[j] && ap2[9-j])
				r++,ap1[j]--, ap2[9-j]--;
		}
		rst=min(ap1[0],ap2[0]);
		if (r>rez)
		{
			rez=r; rest=rst;
			st=i;
		}
	}
	if (st==0)
	{
		r=0;
		for (i=0; i<=9; i++)
			for (j=1; j<=B[i]; j++)
				C[++r]=D[r]=i;
		for (i=n; i>=1; i--)
			printf("%d",C[i]);
		printf("\n");
		for (i=n; i>=1; i--)
			printf("%d",D[i]);
		printf("\n");
		return 0;
	}

	r=rest;
	for (j=0; j<=9; j++)
		ap1[j]=ap2[j]=B[j];
	ap1[0]-=r; ap2[0]-=r;
	C[++r]=st; D[r]=10-st; ap1[st]--; ap2[10-st]--; 
	for (j=0; j<=9; j++)
	{
		while (ap1[j] && ap2[9-j])
		{
			C[++r]=j; D[r]=9-j;
			ap1[j]--; ap2[9-j]--;
		}
	}
	st=r;
	for (j=0; j<=9; j++)
		for (k=1; k<=ap1[j]; k++)
			C[++r]=j;
	r=st;
	for (j=0; j<=9; j++)
		for (k=1; k<=ap2[j]; k++)
			D[++r]=j;
	for (i=n; i>=1; i--)
		printf("%d",C[i]);
	printf("\n");
	for (i=n; i>=1; i--)
		printf("%d",D[i]);
	printf("\n");
	return 0;
}
