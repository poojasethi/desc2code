#define PB push_back
#define MP make_pair
#define BG begin()
#define ED end()
#define SZ(x) ((int)((x).size()))
#define FF first
#define SS second
#define foreach(i,x) for (__typeof((x).begin()) i=(x).begin();i!=(x).end();i++)
#define FOR(i,l,r) for (int i=(l);i<=(r);i++)
#define ROF(i,r,l) for (int i=(r);i>=(l);i--)
using namespace std;

int o[10],o2[10];

int main() {
  string a;
  cin>>a;
  FOR(i,0,SZ(a)-1) o[a[i]-'0']++;
  int mx=-1,ii=0;
  FOR(i,0,9) o2[i]=o[i];
  FOR(i,1,5) if (o[i] && o2[10-i]) {
    o[i]--,o2[10-i]--;
    int ans=1;
    FOR(j,0,9) ans+=min(o[j],o2[9-j]);
    if (ans>mx) mx=ans,ii=i;
    o[i]++,o2[10-i]++;
  };
  string ans,ans2;
  if (ii) {
    ans.PB(ii+'0');o[ii]--;
    ans2.PB(10-ii+'0');o2[10-ii]--;
  };
  FOR(i,0,9)
    while (o[i] && o2[9-i]) {
      o[i]--,o2[9-i]--;
      ans.PB(i+'0');
      ans2.PB(9-i+'0');
    };
  string m0(min(o[0],o2[0]),'0');
  int ll=min(o[0],o2[0]);
  o[0]-=ll,o2[0]-=ll;
  ans=m0+ans,ans2=m0+ans2;
  FOR(i,0,9) 
    FOR(j,1,o[i]) ans.PB(i+'0');
  FOR(i,0,9) 
    FOR(j,1,o2[i]) ans2.PB(i+'0');
  ROF(i,SZ(ans)-1,0) cout<<ans[i];cout<<endl;
  ROF(i,SZ(ans2)-1,0) cout<<ans2[i];cout<<endl;
  return 0;
}
