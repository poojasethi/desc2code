#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;

string s,s1,s2;
int cnt[15],c1[15],c2[15];

int main()
{
	int i,j;
	cin>>s;
	for(i=0;i<s.length();i++)
		cnt[s[i]-'0']++;
	int max=0,f=0,ff=0;
	for(i=1;i<=9;i++)
	{
		if(cnt[i]>=1)
		{
			for(j=0;j<=9;j++) c1[j]=cnt[j];
			c1[i]--;
		}
		else continue;
		if(cnt[10-i]>=1)
		{
			for(j=0;j<=9;j++) c2[j]=cnt[j];
			c2[10-i]--;
		}
		else 
			continue;
		int tmp=1+cnt[0];
		for(j=1;j<9;j++) 
			tmp+=min(c1[j],c2[9-j]);
		if(tmp>max)
			max=tmp,f=i;
		tmp=1;
		for(j=0;j<=9;j++)
		{
			int m=min(c1[j],c2[9-j]);
			c1[j]-=m,c2[9-j]-=m;
			tmp+=m;
		}
		int kk=min(c1[0],c2[0]);
		if(tmp+kk>max)
			max=tmp+kk,f=10+i,ff=kk;
	}
	for(j=1;j<10;j++) c1[j]=c2[j]=cnt[j];
	c1[0]=c2[0]=0;
	for(i=0;i<cnt[0];i++)
		s1+="0",s2+="0";
	if(f)
	{
		if(f>10) 
		{ 
			s1="",s2="";
			for(i=0;i<ff;i++) 
				s1+="0",s2+="0";
			f-=10; c1[0]=c2[0]=cnt[0]-ff; 
		}
		c1[f]--; c2[10-f]--;
		s1.push_back(f+'0');
		s2.push_back(10-f+'0');
		for(j=0;j<=9;j++)
		{
			int m=min(c1[j],c2[9-j]);
			c1[j]-=m,c2[9-j]-=m;
			for(i=0;i<m;i++)
			{
				s1.push_back(j+'0');
				s2.push_back(9-j+'0');
			}
		}
	}
	for(i=0;i<=9;i++)
	{
		for(j=0;j<c1[i];j++) s1.push_back(i+'0');
		for(j=0;j<c2[i];j++) s2.push_back(i+'0');
	}
	for(i=s1.length()-1;i>=0;i--) cout<<s1[i];
	cout<<endl;
	for(i=s2.length()-1;i>=0;i--) cout<<s2[i];
	cout<<endl;
	return 0;
}
