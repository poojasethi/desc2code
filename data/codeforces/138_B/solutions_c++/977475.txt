#include <cstdio>
#include <cstring>
#include <algorithm>
#include <functional>

using namespace std;

int n;
int c1[10], c2[10];
char s[100005];
char a[100005];
char b[100005];

int main() {
	scanf("%s", s);
	n = strlen(s);
	for (int i = 0; i < n; ++i) {
		++c1[s[i] - '0'];
		++c2[s[i] - '0'];
	}
	bool noTen = true;
	for (int i = 1; i <= 4 && noTen; ++i) {
		noTen = c1[i] == 0 || c1[10 - i] == 0;
	}
	noTen = noTen && c1[5] == 0;
	if (noTen) {
		sort(s, s + n, greater <char>());
		puts(s);
		puts(s);
		return 0;
	}
	a[n] = b[n] = '\0';
	int pos = n - 1;
	for (; c1[0] > c2[9]; a[pos] = b[pos] = '0', --pos, --c1[0], --c2[0]);
	bool changed = false;
	for (int i = 1; i <= 5 && !changed; ++i) {
		int x = i, y = 10 - i;
		if (c1[x] > c2[9 - x] && c1[y] > c2[9 - y]) {
			a[pos] = x + '0';
			b[pos] = y + '0';
			--pos;
			--c1[x];
			--c2[y];
			changed = true;
		}
	}
	if (!changed) {
		for (int i = 1; i <= 4 && !changed; ++i) {
			int x = i, y = 10 - i;
			if (c1[x] > 0 && c2[y] > 0) {
				a[pos] = x + '0';
				b[pos] = y + '0';
				--pos;
				--c1[x];
				--c2[y];
				changed = true;
			}
		}
	}
	if (!changed) {
		a[pos] = b[pos] = '5';
		--pos;
		--c1[5];
		--c2[5];
	}
	for (int i = 0; i <= 9; ++i) {
		int x = i, y = 9 - i;
		for (; c1[x] > 0 && c2[y] > 0; a[pos] = x + '0', b[pos] = y + '0', --c1[x], --c2[y], --pos);
	}
	for (int i = pos, j = 9; i >= 0; --i, --c1[j]) {
		for (; c1[j] == 0; --j);
		a[i] = j + '0';
	}
	for (int i = pos, j = 9; i >= 0; --i, --c2[j]) {
		for (; c2[j] == 0; --j);
		b[i] = j + '0';
	}
	puts(a);
	puts(b);
	return 0;
}
