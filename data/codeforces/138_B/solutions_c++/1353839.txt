typedef long long LL;

#define mp make_pair
#define pb push_back

int n;

const int maxn = 100005;
int ct[2][11];
char s[ maxn ];
int a[ maxn],b[maxn];

int cal(int x, int y) {
	if(ct[0][x] == 0 || ct[1][y] == 0 ) return - 1;
	-- ct[0][x];
	-- ct[1][y];
	int r = 0;
	for(int i=0;i<10;++i) r+=min(ct[0][i],ct[1][9-i]);
	++ ct[0][x];
	++ ct[1][y];
	return r;
}

void gao(){
	memset(ct,0,sizeof ct);
	n=strlen(s);
	for(int i=0;i<n;++i) ++ct[0][ s[i] -'0'];
	for(int i=0;i<10;++i) ct[1][i] = ct[0][i];
	int add = 0;
	for(int k=0;k<n;++k){
		int s=10-add;
		int i0,j0;
		i0=j0=-1;
		int best=-1;
		if(s ^ 9) {
			if(ct[0][0] && ct[1][0]) i0=j0=0, best=cal(0,0);
		}
		for(int i=0;i<10;++i){
			int x = i, y = s - i, c = cal( x, y );
			if(c < 0) continue;
			if(c > best){
				best=c;
				i0=x;
				j0=y;
			}
		}
		
		if(best < 0){
			for(int i=0;i<10;++i) if(ct[0][i]!=0) {i0=i; break;}
			for(int i=0;i<10;++i) if(ct[1][i]!=0) {j0=i; break;}
		}
		a[k]=i0;
		b[k]=j0;
		--ct[0][i0];
		--ct[1][j0];
		add=(add+i0+j0)/10;
	}
	for(int i=n-1;i>=0;--i)printf("%d",a[i]);puts("");
	for(int i=n-1;i>=0;--i)printf("%d",b[i]);puts("");
}
int main(){
	while(1==scanf("%s", s)) gao();
	return 0;
}
