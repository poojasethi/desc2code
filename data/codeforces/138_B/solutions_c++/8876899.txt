#include <bits/stdc++.h>

#define itrep(i, v)	for(typeof(v.begin()) i = v.begin(); i != v.end(); i ++)
std::list<int> A, B;
char sr[100010];

int main() {
	int cntA[10] = {0}, cntB[10] = {0}, k = 0, max = 0;
	scanf("%s", sr);
	int n = strlen(sr);
	for(int i = 0; i < n; i ++) {
		cntA[sr[i] - '0'] ++;
		cntB[sr[i] - '0'] ++;
	}
	for(int i = 1; i <= 9; i ++) {
		if(cntA[i] && cntB[10 - i]) {
			int sum = 0;
			cntA[i] --;
			cntB[10 - i] --;
			for(int j = 0; j <= 9; j ++) {
				sum += std::min(cntA[j], cntB[9 - j]);
			}
			if(sum >= max) {
				max = sum;
				k = i;
			}
			cntA[i] ++;
			cntB[10 - i] ++;
		}
	}
	if(!k) {
		for(int i = 9; ~i; i --) while(cntA[i] --)	putchar('0' + i);
		puts("");
		for(int i = 9; ~i; i --) while(cntB[i] --)	putchar('0' + i);
		return 0;
	}
	A.push_back(k);
	B.push_back(10 - k);
	cntA[k] --;
	cntB[10 - k] --;
	for(int i = 0; i < 10; i ++) {
		int c = std::min(cntA[i], cntB[9 - i]);
		for(int t = 0; t < c; t ++) {
			A.push_front(i);
			B.push_front(9 - i);
		}
		cntA[i] -= c;
		cntB[9 - i] -= c;
	}
	int c = std::min(cntA[0], cntB[0]);
	for(int t = 0; t < c; t ++) {
		A.push_back(0);
		B.push_back(0);
	}
	cntA[0] -= c;
	cntB[0] -= c;
	for(int i = 0; i < 10; i ++) {
		while(cntA[i] --) 
			A.push_front(i);
		while(cntB[i] --)
			B.push_front(i);
	}
	itrep(c, A)	putchar(*c + '0');
	puts("");
	itrep(c, B)	putchar(*c + '0');
	puts("");
	return 0;
}
