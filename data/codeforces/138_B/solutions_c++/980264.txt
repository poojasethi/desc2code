#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <utility>
#include <algorithm>
#include <iostream>
#include <vector>
#include <string>
#include <set>
#include <map>
using namespace std;

string n;
string a;
string b;

int ac[10];
int bc[10];

int main()
{
	a = b = "";
	getline(cin, n);
	memset(ac, 0, sizeof(ac));
	memset(bc, 0, sizeof(bc));
	for (int i=0; i<n.length(); ++i)
	{
		++ac[n[i]-'0'];
		++bc[n[i]-'0'];
	}
	int mz = 0;
	int bs = -1;
	for (int i=1; i<10; ++i)
	{
		if (ac[i] == 0 || bc[10-i] == 0)
			continue;
		--ac[i]; --bc[10-i];
		int r = 1;
		for (int j=0; j<10; ++j)
			r += min(ac[j], bc[9-j]);
		if (mz < r)
		{
			mz = r;
			bs = i;
		}
		++ac[i]; ++bc[10-i];
	}
	if (bs != -1)
	{
		a += bs + '0';
		b += 10 - bs + '0';
		--ac[bs]; --bc[10-bs];
		for (int j=0; j<10; ++j)
		{
			int t = min(ac[j], bc[9-j]);
			for (int k=0; k<t; ++k)
			{
				a += j + '0';
				b += 9 - j + '0';
			}
			ac[j] -= t;
			bc[9-j] -= t;
		}
	}
	int tmem = min(ac[0], bc[0]);
	ac[0] -= tmem;
	bc[0] -= tmem;
	/* Other */
	int ap = 0;
	int bp = 0;
	for (;;)
	{
		while (ap < 10 && ac[ap] == 0) ++ap;
		while (bp < 10 && bc[bp] == 0) ++bp;
		if (ap == 10 || bp == 10) break;
		a += ap + '0';
		b += bp + '0';
		--ac[ap];
		--bc[bp];
	}
	reverse(a.begin(), a.end());
	reverse(b.begin(), b.end());
	cout << a;
	for (int i=0; i<tmem; ++i)
		putchar('0');
	cout << endl << b;
	for (int i=0; i<tmem; ++i)
		putchar('0');
	cout << endl;
	return 0;
}
