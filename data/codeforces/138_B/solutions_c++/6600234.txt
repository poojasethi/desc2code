#include <bits/stdc++.h>

using namespace std;

int main() {
    vector<int> d1(10, 0), d2, d3;
    string s, res1, res2, A, B;
    int Max = -1;
    cin >> s;

    for (int i = 0; i < (int) s.size(); i++)
        d1[s[i] - '0']++;

    for (int i = 1; i <= 9; i++) {
        d3 = d2 = d1;
        int cnt0 = 0 , Min;
        A = B = "";
        if (d2[i] && d3[10 - i]) {
            A += (char) (i + '0');
            B += (char) (10 - i + '0');
            d2[i]--,d3[10 - i]-- , cnt0++;
            for (int j = 0; j < 10; j++) {
                Min = min(d2[j], d3[9 - j]);
                d2[j] -= Min , d3[9 - j] -= Min,cnt0 += Min;
                A = string(Min, (char) (j + '0')) + A;
                B = string(Min, (char) (9 - j + '0')) + B;
            }
        }
        if (d2[0] && d3[0]) {
            Min = min(d2[0], d3[0]);
            A += string(Min, '0');
            B += string(Min, '0');
            cnt0 += Min , d2[0] -= Min, d3[0] -= Min;
        }

        for (int j = 0; j < 10; j++) {
            A = string(d2[j], (char) (j + '0')) + A;
            B = string(d3[j], (char) (j + '0')) + B;
        }
        if (cnt0 > Max)
            Max = cnt0 ,res1 = A , res2 = B;
    }
    cout << res1 << endl;
    cout << res2 << endl;
}
