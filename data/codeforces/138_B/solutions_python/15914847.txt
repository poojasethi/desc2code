s=raw_input()
a=[s.count(`x`) for x in range(10)]+[0]
b=a[:]
u,v=0,0
F=range(10)
R=lambda x:x[::-1]
for i in F:
  if a[i] and b[10 - i]:
    a[i]-=1;b[10-i]-=1
    c=1+sum(min(a[x],b[9-x]) for x in F)
    if c>v:u,v=i,c
    a[i]+=1;b[10-i]+=1
d=max(0,a[0]-a[9])
a[0]-=d;b[0]-=d
x=y='0'*d
if v:
  x+=`u`;y+=`10-u`
  a[u]-=1;b[10-u]-=1
for i in F:
  j=min(a[i],b[9-i])
  x+=`i`*j;y+=`9-i`*j
  a[i]-=j;b[9-i]-=j
for i in F:x+=`i`*a[i];y+=`i`*b[i]
print R(x)
print R(y)
