def near(n):
    lo,hi=0,100000000000000
    while hi > lo + 1:
        mid = (lo + hi) / 2
        if mid * mid + mid > 2 * n:
            hi = mid - 1
        elif mid * mid + mid == 2 * n:
            return mid - 1
        else:
            lo = mid
    if hi * hi + hi < 2 * n:
        return hi
    return lo
n=input()
pos=near(n)
n -= pos*(pos+1)/2
