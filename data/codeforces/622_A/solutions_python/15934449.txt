def main():
    n=int(raw_input())
    start=-1;end=n;mid=(end+start)/2
    while end-start>1:
        mid=(start+end)/2
        if ((mid*(mid+1))/2)>n:
            end=mid
        else:
            start=mid
    temp=(end*(end-1))/2
    if temp==n:
        temp=((end-2)*(end-1))/2
    n=n-temp
    print n
main()
