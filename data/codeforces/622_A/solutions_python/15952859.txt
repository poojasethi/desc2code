from math import sqrt


def read_int():
    return int(raw_input())


def sum_of_natural(n):
    return n * (n + 1)/2


def get_nth_number(x):
    n = int(sqrt(2 * x))
    while sum_of_natural(n) < x:
        n += 1
    return n - (sum_of_natural(n) - x)


def main():
    x = read_int()
    print get_nth_number(x)

if __name__ == '__main__':
    main()
