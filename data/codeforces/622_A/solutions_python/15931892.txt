n = int(raw_input())
left, right = 0, n


if(n == 1):
	print(1)
else:
	while(True):
		middle = (left + right) // 2
		if(middle * (middle + 1) / 2 <= n and (middle + 1)* (middle + 2) / 2 > n):
			break
		elif(middle * (middle + 1) / 2 > n):
			right = middle
		else:
			left = middle

	if(n == middle * (middle + 1) / 2):
		print(middle)
	else:
		print(n - middle * (middle + 1) / 2)
