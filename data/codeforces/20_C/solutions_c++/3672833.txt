#include<iostream>
#include<vector>
#include<set>
#define pb push_back
using namespace std;
typedef long long ll;
const ll MAX=100010;
ll n,m,d[MAX],INF=1000000000010LL,p[MAX];
vector<ll>a,g[MAX],c[MAX];
struct cmp
{
	bool operator()(int i,int j)
	{
		if(d[i]!=d[j])return d[i]<d[j];
		return i<j;
	}
};
set<ll,cmp> s;
int main()
{
	cin>>n>>m;
	for(int i=0;i<m;i++)
	{
		ll u,v,w;
		cin>>u>>v>>w;
		u--;
		v--;
		g[u].pb(v);
		g[v].pb(u);
		c[u].pb(w);
		c[v].pb(w);
	}
	d[0]=0;
	s.insert(0);
	for(int i=1;i<n;i++)
	{
		d[i]=INF;
		s.insert(i);
	}
	while(!s.empty())
	{
		ll v=*(s.begin());
		s.erase(v);
		for(int i=0;i<g[v].size();i++)
		{
			ll u=g[v][i];
			if(d[v]+c[v][i]<d[u])
			{
				p[u]=v;
				s.erase(u);
				d[u]=d[v]+c[v][i];
				s.insert(u);
			}
		}
	}
	if(d[n-1]==INF)
	{
		cout<<-1;
		return 0;
	}
	ll v=n-1;
	while(v!=0)
	{
		a.pb(v);
		v=p[v];
	}
	cout<<1<<" ";
	for(int i=a.size()-1;i>=0;i--)
	{
		cout<<a[i]+1<<" ";
	}
	return 0;
}
