#include<iostream>
#include<vector>
#include<set>
using namespace std;
typedef long long ll;
const ll MAXN=1000010;
ll n,m;
ll p[MAXN],d[MAXN];
ll INF=1000000000000LL;
vector<ll> c[MAXN],g[MAXN],a;
struct cmp
{
	bool operator()(int i,int j)
	{
		if(d[i]!=d[j])return (d[i]<d[j]);else return i<j;
	}
};
set<ll,cmp> s;
int main()
{
	cin>>n>>m;
	for(int i=0;i<m;i++)
	{
		int x,y,w;
		cin>>x>>y>>w;
		x--;
		y--;
		g[x].push_back(y);
		g[y].push_back(x);
		c[x].push_back(w);
        c[y].push_back(w);
	}
	d[0]=0;
	s.insert(0);
	for(ll i=1;i<n;i++)
	{
		d[i]=INF;
		s.insert(i);
	}
	while(!s.empty())
	{
		ll v=*(s.begin());
		s.erase(v);
		for(ll i=0;i<g[v].size();i++)
		{
			ll u=g[v][i];
			if((d[v]+c[v][i])<d[u])
			{
				p[u]=v;
				s.erase(u);
				d[u]=d[v]+c[v][i];
				s.insert(u);
			}
		}
	}
	if(d[n-1]==INF)
	{
		cout<<-1;
		return 0;
	}
	ll v=n-1;
	while(v!=0)
	{
		a.push_back(v);
		v=p[v];
	}
	cout<<1<<' ';
	for(ll i=a.size()-1;i>=0;i--)
	{
		cout<<a[i]+1<<' ';
	}
	return 0;
}
