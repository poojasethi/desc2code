#include <iostream>
#include <vector>
#include <set>
#include <cstring>
#include <cstdio>
using namespace std;
long long r[100000];
vector<pair<int,int> > v[100000];
set<pair<long long,int> > all;
int p[100000],ma[100000];
int out(int x){
	if (x!=0) out(p[x]);
	printf("%d ",x+1);
}
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=0;i<m;i++){
		int a,b,c;
		scanf("%d%d%d",&a,&b,&c);
	    a--;
		b--;
		v[a].push_back(make_pair(b,c));
		v[b].push_back(make_pair(a,c));
	}
	memset(ma,0,sizeof(ma));
	ma[0]=1;
	r[0]=0;
	all.insert(make_pair(0,0));
	while (!all.empty()){
		int i=all.begin()->second;
		long long d=all.begin()->first;
		all.erase(all.begin());
		ma[i]=2;
		for(int j=0;j<v[i].size();j++){
			int k=v[i][j].first;
			long long D=d+v[i][j].second;
			if (ma[k]==0||ma[k]==1&&r[k]>D){
				if (ma[k]==1) all.erase(make_pair(r[k],k));
				ma[k]=1;
				r[k]=D;
				p[k]=i;
				all.insert(make_pair(D,k));
			}
		}
	}
	if (ma[n-1]==2) out(n-1); else printf("%d",-1);
	return 0;
}
