#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <bitset>
using namespace std;

typedef long long LL;
const int maxn = 125, mod = 1000000007;
int a[maxn], n, m, t, ans;

void dfs(int u, bitset<120> f, int s) {
    if (u > m / 2) {
        ans = (ans + s) % mod; return ;
    }
    dfs(u + 1, f, s);
    if (a[u] && !f[u])
        dfs(u + 1, f | f >> u | f << u | f >> (m - u) | f << (m - u), (LL)s * a[u] % mod);
}

int main() {
    int x;
    scanf("%d%d%d", &n, &m, &t);
    while (t--) {
        scanf("%d", &x);
        x %= m; ++a[min(x, m - x)];
    }
    dfs(0, bitset<120>(1), 1);
    printf("%d\n", ans);
    return 0;
}
