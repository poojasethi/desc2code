#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<bitset>
using namespace std;
const int maxt=10000+10;
const int mod=1000000007;
const int maxm=65;
int n,m,t,p,cnt[maxm],hash[maxm];
long long ans;

void dfs(int x,bitset<120> s,long long p)
{
	if(x>m/2) { ans=(ans+p)%mod; return; }	
	dfs(x+1,s,p);
	if(cnt[x] && !s[x])
		dfs(x+1,s|s<<x|s>>x|s<<(m-x)|s>>(m-x),p*cnt[x]%mod);
}

int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	scanf("%d %d %d",&n,&m,&t);
	for(int i=1;i<=t;i++)
	{
		scanf("%d",&p);
		cnt[min(p%m,m-p%m)]++;
	}
	bitset<120> s=1;
	dfs(0,s,1);
	cout<<ans<<endl;
	return 0;
}
