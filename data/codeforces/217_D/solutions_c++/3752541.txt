#include<bitset>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define LL long long
using namespace std;
const int mod=1000000007;
int a[125],m,n;
LL ans;
void dfs(int x,bitset<120>f,LL now)
{
	if(x>m/2)
	{
		ans=(ans+now)%mod;
		return;
	}
	dfs(x+1,f,now);
	if(a[x]&&!f[x])
		dfs(x+1,f|f>>x|f<<m-x|f<<x|f>>m-x,now*a[x]%mod);
}
int main()
{
	int x;
	scanf("%d%d%d",&x,&m,&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&x);
		a[min(x%m,m-x%m)]++;
	}
	dfs(1,bitset<120>(1),1);
	printf("%I64d\n",ans);
}
