#include <cstdio>
#include <algorithm>
#include <bitset>
using namespace std;

const int M=120,mo=1e9+7;

int m,t,ans,c[M];

void dfs(int x,bitset<M> f,int w){
	if(x>m-x){ ans=(ans+w)%mo; return; }
	dfs(x+1,f,w);
	if(c[x]&&!f[x]) dfs(x+1,f|f<<x|f>>m-x|f>>x|f<<m-x,1LL*w*c[x]%mo);
}

int main(){
	scanf("%*d%d%d",&m,&t);
	while(t--){ int x; scanf("%d",&x); x%=m; c[min(x,m-x)]++; }
	dfs(0,bitset<M>(1),1);
	printf("%d\n",ans);
}
