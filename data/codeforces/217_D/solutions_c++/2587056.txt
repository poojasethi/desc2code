#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<algorithm>
#include<bitset>
#include<map>

using namespace std;

int m,T,a[200],bl[200];

bitset<120>S;

long long Ans;

void Dfs(int l,int f,bitset<120>Now)
{
	if(l==T){Ans=(Ans+f)%1000000007;return;}
	Dfs(l+1,f,Now);
	if(!Now[a[l]])
		Dfs(l+1,(1ll*f*bl[a[l]])%1000000007,Now|(Now<<a[l])|(Now>>a[l])|(Now<<m-a[l])|(Now>>m-a[l]));
}

int main()
{
	int i,Pat;
	scanf("%*d%d%d",&m,&T);
	for(i=0;i<T;i++)
	{
		scanf("%d",&Pat);
		Pat%=m;
		Pat=Pat<(m-Pat)?Pat:(m-Pat);
		bl[Pat]++;
	}
	T=0;
	for(i=1;i<120;i++)if(bl[i])a[T++]=i;
	S[0]=1;
	Dfs(0,1,S);
	printf("%I64d\n",Ans);
	return 0;
}
