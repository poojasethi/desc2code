#include <cstdio>
#include <cstring>
#include <algorithm>
#define FOR(i,s,e) for (int i=(s); i<(e); i++)
#define FOE(i,s,e) for (int i=(s); i<=(e); i++)
#define FOD(i,s,e) for (int i=(s)-1; i>=(e); i--)
#define N 10005
using namespace std;

int n, cur, t, ret;
int a[N], suf[N], mn[N];

int main(){
	scanf("%d", &n);
	FOR(i,0,n) scanf("%d", &a[i]);
	FOR(i,0,n){
		FOR(j,0,n) mn[j] = j - a[j];
		suf[n] = n;
		FOD(j,n,0) suf[j] = min(suf[j+1], mn[j]);
		
		cur = 0;
		while (cur < n - 1){
			t = cur + 1;
			while (t + 1 < n && suf[t + 1] <= cur) ++t;
			ret++;
			cur = t;
		}
		
		rotate(a, a + 1, a + n);
	}
	
	printf("%d\n", ret);
	return 0;
}
