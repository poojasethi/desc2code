using namespace std;

const double PI = acos(-1.0);
const int MN = 250111;

int n, a[MN], first[MN];

int main() {
    while (scanf("%d", &n) == 1) {
        FOR(i,1,n) scanf("%d", &a[i]);

        int res = 0;

        for(int start = 1; start <= n; ++start) {
            int j = start;
            for(int i = 1; i <= n; ++i) {
                first[i] = i - a[j];
                ++j; if (j > n) j -= n;
            }

            // PR(first, n);
            int cur = n, next;
            while (cur > 1) {
                ++res;
                next = first[cur];
                if (next <= 1) break;
                for(int i = first[cur]+1; i <= cur-1; ++i)
                    if (first[i] < first[next]) next = i;
                cur = next;
            }
        }
        cout << res << endl;
    }
    return 0;
}
