#include <stdio.h>
#define N 10010
int a[2*N], p[2*N];
int main()
{
    int r, i, j, k, n;
    for(scanf("%d", &n), i=0; i<n; scanf("%d", &a[i]), a[n+i]=a[i], i++);
    for(i=0; i<2*n; p[i]=2*n, i++);
    for(r=0, i=n; i>0; i--)
        for(j=i; j!=i+n-1; r++)
        {
            if(p[j]>=i+n)
            {
                for(k=i+n-1; k>j && a[k]<k-j; k--);
                p[j]=k;
            }
            j=p[j];
        }
    printf("%d\n", r);
    return 0;
}
