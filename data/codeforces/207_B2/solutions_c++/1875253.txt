#include <iostream>
#include <algorithm>
using namespace std;
int main(){
   int n,nxt[10000],a[100000];
   cin >> n;
    for (int i=0; i<n; i++)
        cin >> a[i];
    int ans=0;
    for (int q=n; q--; ) {
        for (int i=n-2,j=n-1; i>=0; i--) {
            for (;j>i && i<j-a[j];j--) ;
            nxt[i]=j;
        }
        for (int i=0; i<n-1; i=nxt[i]) 
            ans++;
        rotate(a, a+n-1, a+n);
    }
   cout << ans;
}
