#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mp(x,y) make_pair(x,y)
#define pr pair<int,int>
vector<ll> v[300001],pre[300001],suf[300001];
ll d[300001],ans[300001];
ll mark[300001];
ll parent[300001];
ll mod=1000000007;
ll dfs(ll i,ll j)
{
	ll s=v[i].size();
	d[i]=1;
	parent[i]=j;
	for(ll j=0;j<s;j++)
		d[i]=(d[i]*(dfs(v[i][j],i)+1))%mod;
	pre[i].push_back(1);
	suf[i].push_back(1);
	for(ll j=1,k=s-2;j<s;j++,k--)
	{
		pre[i].push_back((pre[i][j-1]*(d[v[i][j-1]]+1))%mod);
		suf[i].push_back((suf[i][j-1]*(d[v[i][k+1]]+1))%mod);
	}
	reverse(suf[i].begin(),suf[i].end());
	return d[i];
}
void dfs1(ll i)
{
	ans[i]=d[i];
	ll s=v[i].size();
	for(ll j=0;j<s;j++)
	{
		ll sti=d[i],stj=d[v[i][j]];
		d[i]=(pre[i][j]*suf[i][j])%mod;
		if(parent[i]!=0) d[i]=(d[i]*(d[parent[i]]+1))%mod;
		d[v[i][j]]=(d[v[i][j]]*(d[i]+1))%mod;
		dfs1(v[i][j]);
		d[i]=sti,d[v[i][j]]=stj;
	}
}
int main()
{
	std::ios_base::sync_with_stdio(false);
	ll n,x;
	cin>>n;
	for(ll i=2;i<=n;i++)
	{
		cin>>x;
		v[x].push_back(i);
	}
	dfs(1,0);
	dfs1(1);
	for(ll i=1;i<=n;i++)
		cout<<ans[i]<<" ";
	return 0;
}
