#include <iostream>
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <vector>
using namespace std;
#define mod 1000000007
#define LL long long
vector<int> e[200010];
LL u[200010];
LL v[200010];
LL d[200010];
LL a[200010];

void dfs(int x)
{
    d[x]=1;
    for(int i=0;i<e[x].size();i++)
    {
        int to=e[x][i];
        dfs(to);
        d[x]=d[x]*(d[to]+1)%mod;
    }
    LL p=1;
    for(int i=0;i<e[x].size();i++)
    {
        int to=e[x][i];
        u[to]=p;
        p=p*(d[to]+1)%mod;
    }
    p=1;
    for(int i=e[x].size()-1;i>=0;i--)
    {
        int to=e[x][i];
        v[to]=p;
        p=p*(d[to]+1)%mod;
    }
}

void dfs(int x,int y)
{
    a[x]=(y+1)*d[x]%mod;
    for(int i=0;i<e[x].size();i++)
    {
        int to=e[x][i];
        dfs(to,u[to]*v[to]%mod*(y+1)%mod);
    }
}

int main()
{
    int n,x;
    scanf("%d",&n);
    for(int i=2;i<=n;i++)
    {
        scanf("%d",&x);
        e[x].push_back(i);
    }
    dfs(1);
    dfs(1,0);
    for(int i=1;i<=n;i++)
        printf("%d ",a[i]);
    return 0;
}
