#!/usr/bin/env python
# -*- coding: utf-8 -*-

N = int(raw_input())

sum_x = 0
sum_x2 = 0
sum_y = 0
sum_y2 = 0

for i in range(N):
    x, y = (raw_input().split())
    x = int(x)
    y = int(y)
    sum_x += x
    sum_x2 += x**2
    sum_y += y
    sum_y2 += y**2

print(N * sum_x2 - sum_x**2 + N * sum_y2 - sum_y**2)
