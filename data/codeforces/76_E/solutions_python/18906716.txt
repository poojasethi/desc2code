#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
sx, sy = 0, 0
sx2, sy2 = 0, 0
for i in range(n):
    x, y = ia()
    sx += x; sx2 += x*x
    sy += y; sy2 += y*y

ans = 0
ans += sx2*n + sx2*n - 2*sx*sx
ans += sy2*n + sy2*n - 2*sy*sy

print ans / 2
