

def readints():
    return map(int,raw_input().split())

def main():
    n=input()
    xs=[]
    ys=[]
    for i in xrange(n):
        v1,v2=readints()
        xs.append(v1)
        ys.append(v2)
    print comp(xs)+comp(ys)

def comp(xs):
    return len(xs)*sum([i*i for i in xs])-sum(xs)**2    



if __name__=='__main__':
    main()
