#include <stdio.h>
#include <math.h>

int main() {
    int n,i;
    long long sum,sa,sb,x,y;
    scanf("%d",&n);
    sum = sa = sb = 0;
    for( i=0; i<n; i++ ) {
        scanf("%I64d %I64d",&x,&y);
        sum+=x*x*n+y*y*n;
        sa+=x;
        sb+=y;
    }
    printf("%I64d\n",sum-sa*sa-sb*sb);
    return 0;
}
