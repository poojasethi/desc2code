#include <cstdlib>
#include <iostream>

using namespace std;

int main() {
  long long i, j, n, x, y, X, Y, sum;
  X = 0;
  Y = 0;
  sum = 0;
  cin >> n;
  for (i = 0; i < n; i++) {
    cin >> x >> y;
    sum += n * (x*x+y*y);
    X += x;
    Y += y;
  }
  cout << sum - X*X - Y*Y << endl;
  cin.get();
  cin.get();
}
