#include <iostream>
#include <math.h>
using namespace std;

int main ()
{
    long n;
    unsigned long long a=0;
    cin>>n;
    long long x1,y1;
    long long x=0,y=0;
    for(long i=0;i<n;i++){
        cin>>x1>>y1;
        x+=x1;
        y+=y1;
        a+=x1*x1+y1*y1;
    }
    cout<<a*n-x*x-y*y;
    return 0;
}
