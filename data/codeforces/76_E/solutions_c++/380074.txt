#include <iostream>
#include <cstdio>
using namespace std;

typedef long long LL;
int n;
LL x2y2,xy, sumx, sumy;

int main(){
    scanf("%d", &n);
    int x,y;
    for (int i=0; i<n; i++){
	scanf("%d%d", &x, &y);
	x2y2+=x*x+y*y;
	xy+=(sumx*x)+(sumy*y);
	sumx+=x;
	sumy+=y;
    }
    x2y2*=(n-1);
    xy*=2;
    cout<<x2y2-xy<<endl;
    return 0;
}
