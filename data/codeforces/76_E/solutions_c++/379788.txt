#include <iostream>
#include <cstdio>

using namespace std;

typedef long long huge;

int main() {
	huge Sx2 = 0, Sy2 = 0, Sx = 0, Sy = 0;
	int n;
	cin >> n;
	for (int i = 1; i <= n; ++i) {
		huge x, y;
		cin >> x >> y;
		Sx += x;
		Sy += y;
		Sx2 += x * x;
		Sy2 += y * y;
	}
	cout << n * (Sx2 + Sy2) - Sx * Sx - Sy * Sy << endl;
	return 0;
}
