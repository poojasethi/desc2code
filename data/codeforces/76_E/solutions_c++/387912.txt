#include<iostream>
#include<stdio.h>
#include<cmath>
using namespace std;
int x,y,i,n;
long long xs,ys,x2s,y2s,res;
int main()
{
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		xs+=x;
		x2s+=(long long)x*x;
        ys+=y;
        y2s+=(long long)y*y;
    }
	res=n*x2s-xs*xs+n*y2s-ys*ys;
    cout<<res<<endl;
}
