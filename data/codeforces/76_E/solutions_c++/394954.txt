#include <iostream>
using namespace std;
long long ans, sumx, sumy;
long n;

int main()
{
	cin >> n;
	long long x, y;
	ans = sumx = sumy = 0;
	for (long i = 1; i <= n; ++i) {
		cin >> x >> y;
		ans += (x*x+y*y)*n;
		sumx += x;
		sumy += y;
	}
	ans -= (sumx*sumx+sumy*sumy);
	cout << ans << endl;

	return 0;
}
