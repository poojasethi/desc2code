#include<cstdio>
__int64 N,x,y,X,Y,Xx,Yy,p,q;
int main()
{
	while(~scanf("%I64d",&N))
	{
		Xx=Yy=X=Y=p=q=0;
		for(int i=0;i<N;i++)
		{
			scanf("%I64d%I64d",&x,&y);
			X+=x*x;Y+=y*y;
			Xx+=x*p;Yy+=y*q;
			p+=x;q+=y;
		}
		X*=(N-1);Y*=(N-1);
		X-=2*Xx;Y-=2*Yy;
		printf("%I64d\n",X+Y);
	}
}
