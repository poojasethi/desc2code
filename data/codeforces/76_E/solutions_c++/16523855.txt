#include <stdio.h>

using namespace std;

long long n,x,y,xc,yc,sx,sy,res,res2;

int main(){
    scanf("%lld",&n);
    for(int i=0;i<n;i++){
        scanf("%lld %lld",&x,&y);
        sx+=x;
        sy+=y;
        xc+=x*x;
        yc+=y*y;
    }
    res=(n*xc)-(sx*sx);
    res2=(n*yc)-(sy*sy);
    printf("%lld",res+res2);
}
