#include <iostream>
using namespace std;
const int N=100000;
int main()
{
    int n,i;
    long long x,y,sumx,sumy,qx,qy;
    while(cin>>n)
    {
		sumx=0,sumy=0,qx=0,qy=0;
		for(i=0;i<n;i++)
		{
			cin>>x>>y;
			sumx+=x,sumy+=y;
			qx+=x*x,qy+=y*y;
		}
		cout<<n*(qx+qy)-sumx*sumx-sumy*sumy<<endl;;
    }
    return 0;
}
