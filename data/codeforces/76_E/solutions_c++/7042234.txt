#include <iostream>
#include<cstdio>
using namespace std;

int main(){
	long long xx,yy,ss;
	int n,t;
	long long x,y;
	scanf("%d",&n);
	xx=0;
	yy=0;
	ss=0;
	t=n;
	while(t--){
		scanf("%lld%lld",&x,&y);
		ss+=(x*x+y*y);
		xx+=x;
		yy+=y;
	}
	ss=n*ss-xx*xx-yy*yy;
	printf("%lld\n",ss);
	return 0;
}
