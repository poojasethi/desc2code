#include <cstdio>
#include <algorithm>
#include <cstring>
#define FOR(i,s,e) for (int i=(s); i<(int)(e); i++)
using namespace std;

int n;
long long x,y,X,Y,ret;

int main(){

	scanf("%d",&n);
	FOR(i,0,n){
		scanf("%I64d%I64d",&x,&y);
		X+=x;
		Y+=y;
		ret+=n*x*x+n*y*y;
	}
	ret -= X*X + Y*Y;
	printf("%I64d\n",ret);
	
	return 0;
}
