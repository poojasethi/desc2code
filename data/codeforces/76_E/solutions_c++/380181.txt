#include <iostream>
#include <cstdio>
using namespace std;

__int64 sum1,sum2,s1,s2,ans;

int main()
{
	int N,x,y;
	scanf("%d",&N);
	s1 = s2 = sum1 = sum2 = 0;
	for(int i = 0; i<N; i++)
	{
		scanf("%d %d",&x,&y);
		sum1+=x*x;
		sum2+=y*y;
		s1+=x;
		s2+=y;
	}
	ans = N*(sum1+sum2)-s1*s1-s2*s2;
	printf("%I64d\n",ans);
	return 0;
}
