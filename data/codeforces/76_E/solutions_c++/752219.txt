#include <cstdio>
typedef long long LL;
int main()
{
    int n; LL s1=0,s2=0,s3=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        LL x,y;
        scanf("%I64d%I64d",&x,&y);
        s1+=x*x+y*y;
        s2+=x;
        s3+=y;
    }
    s2*=s2, s3*=s3;
    printf("%I64d\n",n*s1-s2-s3);
}
