#include <iostream>
using namespace std;

int main()
{
	int x, y, n;
	while( cin>>n )
	{
		__int64 kx = 0, ky = 0, sx = 0, sy = 0;
		for( int i = 1; i <= n; i++)
		{
			cin>>x>>y;
			kx +=  x * x; ky += y * y;
			sx += x; sy += y;
		}
		cout<<(n * kx + n * ky - sx * sx - sy * sy)<<endl;
	}
	return 0;
}
