#include <iostream>
#include <cstdlib>
#define MAXN 100010

using namespace std;

int n,a[MAXN],q;
int st[4*MAXN][60];

void update(int i, int l, int r, int pos, int val) {
    if (l > pos || r < pos) return;
    if (l == r) {
        for (int j = 0; j < 60; j++)
            st[i][j] = j % val == 0 ? 2 : 1;
        return;
    }
    int mid = (l+r)/2;
    update(2*i,l,mid,pos,val);
    update(2*i+1,mid+1,r,pos,val);
    for (int j = 0; j < 60; j++) {
        st[i][j] = st[2*i][j] + st[2*i+1][(j+st[2*i][j])%60];
    }
}

int query(int i, int l, int r, int ll, int rr, int t) {
    if (l > rr || r < ll) return 0;
    if (l >= ll && r <= rr) return st[i][t];
    int mid = (l+r)/2;
    int t1 = query(2*i,l,mid,ll,rr,t);
    int t2 = query(2*i+1,mid+1,r,ll,rr,(t+t1)%60);
    return t1+t2;
}

int main() {
    ios::sync_with_stdio(0);
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        update(1,1,n,i,a[i]);
    }
    cin >> q;
    while (q--) {
        char op;
        int x,y;
        cin >> op >> x >> y;
        if (op == 'A') { //query
            cout << query(1,1,n,x,y-1,0) << '\n';
        }
        else { //update
            a[x] = y;
            update(1,1,n,x,y);
        }
    }
}
