#include<cstdio>
const int N=200001;
int a[N],b[N];
int main()
{
	int i,v,n;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&v);
		b[v]=i;
	}
	for(i=2;i<=n;i++)
	{
		if(b[a[i]]<b[a[i-1]])
			break;
	}
	printf("%d\n",n-i+1);
	return 0;
}
