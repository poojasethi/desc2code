#include<cstdio>
#include<iostream>
#include<iostream>
using namespace std;
int a[200005][2];
int main()
{
	int n;
	int i,j;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&a[i][0]);
	for(i=0;i<n;i++)
		scanf("%d",&a[i][1]);
	for(i=0,j=0;i<n;i++)
		if(a[i][1]==a[j][0])j++;
	printf("%d\n",n-j);
	return 0;
}
