#include<cstdio>

int arr[200011];

int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&arr[i]);
    int stable=0;
    for(int i=0;i<n;i++)
    {
         int temp;
         scanf("%d",&temp);
         if(temp==arr[stable]) stable++;
    }
    printf("%d\n",n-stable);
}
