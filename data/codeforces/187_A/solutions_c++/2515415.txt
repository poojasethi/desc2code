#include <iostream>
#include <algorithm>
using namespace std;
const int maxn=200001;
int a[maxn];
int main()
{
	int n;
	cin>>n;
	int bank=1;
	for(int i=1;i<=n;i++)   
		cin>>a[i];
	for(int i=1;i<=n;i++)   
	{
		int q;
		cin>>q;
		if(q==a[bank])
			bank++;
	}
	cout<<n-bank+1;
}
