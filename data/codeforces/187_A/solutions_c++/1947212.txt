#include<cstdio>
#include<map>
std::map<int,int>m;int x[2][200001],N,i,j,k;
int main(){
	scanf("%d",&N);
	for(i=0;i<N;i++){
		scanf("%d",x[0]+i);
	}
	for(i=0;i<N;i++){
		scanf("%d",x[1]+i);
		m[x[1][i]]=i;
	}
	for(i=0;i<N-1;i++){
		if(m[x[0][i]]>m[x[0][i+1]])break;
	}
	printf("%d",N-i-1);
}
