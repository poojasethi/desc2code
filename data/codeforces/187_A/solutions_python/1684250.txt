n = input()
a = map(int, raw_input().split())
b = map(int, raw_input().split())
c = [0] * (n+1)
d = []
i = 0
for j in xrange(n):
    i += 1
    c[ a[j] ] = i
i = 0
for j in xrange(n):
    tt = b[j]
    b[j] = c[tt]
    c[tt] = i
    i += 1
    d.append([b[j], c[tt]])
d.sort()

ans, r, can = 0, None, 1
for j in xrange(n):
    if j == 0 or (can == 1 and d[j][1] >= r):
        r = d[j][1]
    else:
        ans += 1
        can = 0
print ans
