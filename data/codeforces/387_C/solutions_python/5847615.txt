s = raw_input();
num = int(s)
cnt = 0

lastpos = len(s) - 1

while True:
	#print "num = ",num,"   pos = ",pos
	pos = lastpos
	while s[pos] == '0':
		pos -= 1

	firstlen = pos
	lastlen = lastpos - pos + 1

	if(firstlen < lastlen):
		break
	else:
		if(firstlen == lastlen):
			firstOneIsBigger = True
			for i in range(firstlen):
				if( s[i] < s[i+firstlen] ):
					firstOneIsBigger = False;
					break;



			if(firstOneIsBigger):
				cnt += 1
			else:
				break;
		else:
			cnt+= 1

		lastpos = pos - 1



cnt += 1

print cnt
