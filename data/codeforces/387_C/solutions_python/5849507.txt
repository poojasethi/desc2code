p = raw_input()
n = [p[0]]
for x in xrange(1,len(p)):
    if p[x] == '0':
        n[-1]+='0'
    else:
        n.append(p[x])
b = 1
p = 1
now = n[0]
while p < len(n):
    if len(now)>len(n[p]):
        b+=1
    elif int(now)>=int(n[p]):
        b+=1
    else:
        b=1
    now+=n[p]
    p+=1
print b
