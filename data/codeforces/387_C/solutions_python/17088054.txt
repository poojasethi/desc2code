p = raw_input()
count = 0
zeros = 0
skip = []

for x in xrange(len(p)-1,-1,-1):
	if x in skip:
		continue
	if p[x] != '0':
		if zeros != 0:
			tmp = p[x-zeros-1:x]
			q = p[x] + '0'*zeros
			if tmp < q and len(q) >= x:
				skip = [x for x in range(max(x-zeros-1,0), x)]
		count += 1
		zeros = 0
	else:
		zeros += 1

if len(p) > 1 and p[0] < p[1] and 0 not in skip: count -= 1
print count
