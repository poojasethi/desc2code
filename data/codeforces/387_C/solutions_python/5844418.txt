from sys import stdin
def main():
    s = stdin.readline().strip()
    if len(s) == 1:
        print 1
        return
    l = len(s)
    ans = 1
    j = 0
    while j < l:
        b = j
        j = j + 1
        while j < l and s[j] == '0':
            j = j + 1
        if b > j - b or (b == j - b and s[:b] >= s[b:j]):
            ans = ans + 1
        else:
            ans = 1
    print ans
main()
