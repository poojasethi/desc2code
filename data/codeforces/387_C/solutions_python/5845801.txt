import sys
import datetime


def main():
    p = sys.stdin.readline().strip()
    l = len(p)
    c = 1
    for i in range(l - 1, -1, -1):
        if i < l - i or i == l - i and p[:i] < p[i:l]:
            break
        if p[i] == '0':
            continue
        c += 1
        l = i
    print c

if __name__ == '__main__':
    main()
