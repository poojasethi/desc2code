s = ''
def cmp(a,b,c):
        if b-a>c-b: return True
        if b-a<c-b: return False
        for i in range(b-a):
                if s[i+a]>s[i+b]: return True
                if s[i+a]<s[i+b]: return False
        return True

s = raw_input()
ans,cur = 0,0
for i in range(len(s)):
        if i!=len(s)-1 and s[i+1]=='0': continue
        if cmp(0,cur,i+1)==True: ans += 1
        else: ans = 1
        cur =  i+1
print ans
