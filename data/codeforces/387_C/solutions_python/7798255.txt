#!/usr/bin/env python

def less_than(a , b):
    if len(a) > len(b):
        return False
    elif len(b) > len(a):
        return True
       
    return a < b 

def get_num_in_array(number):
    if not number:
        return 0
    ans = 1
    i = len(number) - 1
    while i >= 0:
        if number[i] == '0':
            i -= 1
            continue
        if not less_than(number[:i], number[i:]):
            ans += 1
            number = number[:i]
        i -= 1
    return ans

if __name__ == '__main__':
    number = raw_input()
    print get_num_in_array(number)
