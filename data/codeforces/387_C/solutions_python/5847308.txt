s = raw_input()
l = []
prev = 0
for i in s:
#    i = s[x]
#    print i
    if i=='0':
        l[-1] = l[-1] + '0'
        continue
    l.append(i)
count = 0
prev = l[0]
first = True
#print l
for n in l:
    if len(n) < len(prev) :
        count = count +1
    elif len(n) == len(prev) and n<=prev:
        count = count +1
    else:
        #print 'x'
        count = 1
    if not first:
        prev = prev+n
    first = False
