#!/usr/bin/env python


def read():
    return raw_input()


def work(s):
    lengL = len(s)
    lengR = 0
    idx = len(s) - 1
    cnt = 0
    
    while idx >= 0:
        lengL -= 1
        lengR += 1

        if s[idx] != '0':
            if lengL > lengR or (lengL == lengR and s[0] >= s[idx]):
                lengR = 0
                cnt += 1
        elif lengL < lengR:
            break

        idx -= 1
    
    print cnt + 1
    

if __name__ == "__main__":
    work(read())
