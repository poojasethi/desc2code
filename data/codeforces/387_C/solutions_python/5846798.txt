import sys

x = raw_input()
n = len(x)
b = []
count = 0
j = -1

def larger():
	for ii in xrange(j):
		if b[ii] > x[ii]:
			return True
		elif b[ii] < x[ii]:
			return False
	return False

for i in xrange(n+1):
	if i >= n or x[i] != '0':
		if len(b) > j or (len(b) == j and larger()):
			count = 1
		else:
			count += 1
		j = i
		b = []
	if i < n:
		b.append(x[i])

print count
