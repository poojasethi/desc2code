# nice problem :-)

def lesser(a,b):
    if len(a) != len(b):
        return len(a) < len(b)
    return a < b

p = raw_input()

n = len(p)
i = 0
previous = ''
answer = 0
while i < n:
    t = 1
    while i + t < n and p[i + t] == '0':
        t += 1
    curr = p[i:i+t]
    if lesser(previous,curr):
        answer = 1
    else:
        answer += 1
    previous += curr
    i = i+t

print answer
