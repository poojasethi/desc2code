#!/usr/bin/python 

nums = raw_input()
# print nums

cnt = 0 
pos = len(nums) - 1

while pos >= 0 : 
    if pos == 0 : 
        cnt += 1
        pos -= 1
    elif int(nums[pos]) != 0 and \
            (pos > len(nums) - pos or 
            int(nums[pos:]) <= int(nums[:pos])) : 
        cnt += 1
        nums = nums[:pos]
        pos -= 1
    else : 
        pos -= 1

print cnt
