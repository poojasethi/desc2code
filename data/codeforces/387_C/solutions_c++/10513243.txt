#include <bits/stdc++.h>

using namespace std;

int main()
{
    string s;
    cin >> s;
    int i = s.length()-1, t, l = 0;;
    while (i > -1) {
        t = 1;
        while (s[i] == '0') i--, t++;
        if (t < i) l++;
        else if (t == i) {
            if (s[0] >= s[i] ) l++;
            else break;
        }
        else break;
        i--;
    }
    cout << l + 1;
    return 0;
}
