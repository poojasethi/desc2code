#include <bits/stdc++.h>
using namespace std;
#define MAXN 10000007
string s;
int main(){
    cin>>s;
    int i,l=s.length(),ret = 1;
    for(i=l-1;i>=0;i--){
        if(s[i]=='0')   continue;
        if(2*i<l)   break;
        if(2*i==l && s[i]>s[0]) break;
        ret++;
        l = i;
    }
    cout<<ret<<endl;
}
