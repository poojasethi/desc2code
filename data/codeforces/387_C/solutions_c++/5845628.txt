#include<cstdio>
#include<cstring>

int b[100005], ans = 0;
char a[100005];

int main() {
    scanf("%s", a);
    for (int i = strlen(a) - 1 ; i >= 0 ; i--) if (a[i] == '0') b[i] = b[i + 1] + 1;
    for (int i = b[1] ; a[i] ; i += b[i + 2] + 1) {
        if (i > b[i + 2] || i == b[i + 2] && a[0] >= a[i + 1]) ans++;
        else ans = 0;
    }
    printf("%d\n", ans);
    scanf("\n");
    return 0;
}
