#include <cstdio>
#include <cstdlib>
#include <ctime>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
char str[1000001];
int ans = 1, len;
int main( )
{
	scanf("%s", str);
	len = strlen(str);
	for (int i = len - 1; i >= 0; i --)
		if (str[i] != '0')
		{
			if (2 * i < len || (2 * i == len && str[i] > str[0])) break;
			++ ans, len = i;
		}
	printf("%d\n", ans);
	return 0;
}
