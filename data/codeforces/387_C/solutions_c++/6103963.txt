#include<cstdio>

char str[100005];

int main()
{
    gets(str);
    int i, j, res = 0;
    for (i = 0; str[i]; i = j)
    {
        for (j = i + 1; str[j] == '0'; j++);
        
         if (j - i > i || j - i == i && str[0] < str[i]) res = 1;
        else res++;
       // printf("%d %d\n",j,res);
    }
    printf("%d", res);
    return 0;
}
