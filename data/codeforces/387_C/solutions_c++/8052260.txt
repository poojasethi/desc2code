#include<iostream>
using namespace std;

int main()
{
    string s;

    int i , j , ans = 0;

    cin >> s;

    for( i = 0; s[i] ;i = j)
    {
        for(j = i + 1 ; s[j] == '0';j ++);

        if(j - i > i || j - i == i && s[0] < s[i]) ans = 1;
        else ans ++;
    }

    cout << ans << endl;
}
