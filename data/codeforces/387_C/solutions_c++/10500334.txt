#include <bits/stdc++.h>

using namespace std;
string s;int ans = 0;
int main()
{
    cin >> s;
    int i = s.length()-1, t;
    while (i > -1) {
        t = 1;
        while (s[i] == '0') i--, t++;
        if (t < i) ans++;
        else if (t == i) {
            if (s[0] >= s[i] ) ans++;
            else break;
        }
        else break;
        i--;
    }
    cout << ans+1;
    return 0;
}
