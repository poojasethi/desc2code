#include <stdio.h>
#include <string.h>
#define N 100010
char s[N];
int main()
{
	int i,j,ans,n;
	scanf("%s",s);
	n=strlen(s);
	ans=i=0;
	while(i<n)
	{
		j=i+1;
		while(j<n&&s[j]=='0')j++;
		if(j-i>i||(j-i==i&&s[0]<s[i]))
			ans=1;
		else
			ans++;
		i=j;
	}
	printf("%d\n",ans);
	return 0;
}
