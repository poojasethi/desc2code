#include<iostream>
using namespace std;

int main() {
  int ans=0;
  string s;
  cin>>s;
  int j=s.length()-1;
  while(j>=0){
    int k=j;
    while(s[k]=='0')
      k--;
    int len=j-k+1;
    if(len>k){
      ans++;
      j=-1;
      continue;
    }
    if(len==k){
      if(s[0]>=s[k])
	j=k-1;
      else
	j=-1;
      ans++;
      continue;
    }
    if(len<k){
      ans++;
      j=k-1;
    }
  }
  cout<<ans;
}
