#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    string a;
    cin>>a;
    int len=a.size();
    int i,ans=1;
    for(i=len-1;i>=0;i--)
    {
        if(a[i]=='0')
            continue;
        else if(i*2<len||(i*2==len&&a[i]>a[0]))
        {
            break;
        }
        else
        {
            len=i; ans++;
        }
    }
    cout<<ans<<endl;
    return 0;
}
