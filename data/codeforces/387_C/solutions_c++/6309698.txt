#include<iostream>
#include<string>
using namespace std;
main()
{
    string s;
    cin >> s;
    int len = s.size();
    int ans = 1;
    for(int i = len - 1; i > 0; i--)
    {
        if(s[i] == '0') continue;
        if(i * 2 < len || i * 2 == len && s[i] > s[0]) break;
        ans++;
        len = i;
    }
    cout << ans;
}
