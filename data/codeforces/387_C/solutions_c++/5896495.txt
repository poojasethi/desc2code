#include <iostream>
#include <cstring>
using namespace std;



int main()
{
    ios::sync_with_stdio(false);
    string s;
    cin>>s;
    int n=s.length();
    int i,cnt=1;
    for(i=n-1;i>=0;--i)
    {
        if(s[i]=='0')
            continue;
        if(2*i<n||(2*i==n&&s[i]>s[0]))
            break;
        cnt++;n=i;
    }
    cout<<cnt;
    return 0;
}
