#include<iostream>
#include<string>
using namespace std;
int main()
{
	string s;
	cin>>s;
	int n=s.length(),count=1,i=0,j=0,k=0;
	while(i<n)
	{
		while(i+1<n and s[i+1]=='0')i++;
		j=i+1;
		while(j+1<n and s[j+1]=='0')j++;
		if(j-i<i+1 or (j-i==i+1 and s[i+1]<=s[0]))count++;
		else count=1;
		i=j;
	}cout<<count-1;
}
