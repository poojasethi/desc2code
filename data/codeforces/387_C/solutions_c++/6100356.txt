#include <iostream>
#include <cstdio>
#include <string>

using namespace std;

int main() {
    int ans = 0;
    string s;
    cin >> s;
    for (int i = 0; i < s.size(); ) {
        int l = 1, pos = i++;
        for (; i < s.size() && s[i] == '0'; i++)
            l++;
        ans = (pos == 0 || pos > l || (pos == l && s[0] >= s[pos])) ? ans + 1 : 1;
    }
    cout << ans;
    return 0;
}
