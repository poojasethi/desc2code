
#include <iostream>

using namespace std;

int main()
{
	string s;
	cin>>s;
	
	int t = 0 , i = s.size()-1 , k;
	while (i>=0)
	{
		k = i;
		while (s[k]=='0') k--;
		
		int len = i - k + 1;
		
		if (k>len)
		{
			t++;
			i = k-1;
		}
		else if (k==len)
		{
			t++;
			if (s[0] >= s[k])
				i=k-1;
			else
				break;
		}
		else
		{
			t++;
			break;
		}
	}
	cout<<t;
	return 0;
}
