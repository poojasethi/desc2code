#include <iostream>
using namespace std;

string s;
int ans(int n) {
	int pt = n;
	while (s[pt] == '0')
		pt--;
	int x = (n - pt + 1) - pt;
	if (x < 0 || (!x && s[0] >= s[pt]))
		return ans(pt - 1) + 1;
	return 1;
}
int main() {
	ios_base::sync_with_stdio(false);
	cin >> s;
	cout << ans(s.size() - 1) << '\n';
	return 0;
}
