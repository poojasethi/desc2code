
#include <iostream>
#include <string>
#include <algorithm>
#include <utility>
#include <vector>

using namespace std;

int main()
{
	
	string p;
	
	cin>>p;

	int len = p.length();
	int ans = 0, i = 0;

	while (i < len)
	{

		int j = i + 1;

		while (p[j] == '0') ++j;
		if (j > 2 * i || (j == 2 * i && p[i] > p[0])) ans = 1;
		else ++ans;

		i = j;

	}

	cout<<ans;

	return 0;

}
