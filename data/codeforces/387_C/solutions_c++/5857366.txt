#include <iostream>
using namespace std;

char str[100005];

int main()
{
  cin>>str;
  int r=0;
  for(int i=0,j=0;str[i] ^ '\0';)
    {
      for(j=i+1;str[j] ^ '\0' && str[j]=='0';j++);
      if(j-i>i || (j-i==i && str[0]<str[i])) r=1;
      else r++;
      i=j;
    }
  cout<<r<<endl;
  return 0;
}
