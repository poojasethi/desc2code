#include <cstdio>

#define N 100000

char a[N], c;
int s, n, k;

int main()
{
	while ((c = getchar()) >= 48 && c <= 57)
	{
		a[n] = c - 48;
		if (a[n] > 0)
			k = n;

		if (k > n - k + 1 || (k == n - k + 1 && a[0] >= a[k]))
		{
			if (a[n])
				++ s;
		}
		else
			s = 1;
		++n;
	}
	printf("%d\n", s);
}
