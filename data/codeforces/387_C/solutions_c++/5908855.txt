#include <iostream>
#include <string>

int main() {
	std::string str;
	std::getline(std::cin, str);

	int i = 0, j, res = 0;

	while(i < str.length()) {
		for (j = i + 1; j < str.size() && (str[j] == '0'); j++);

		if (j - i > i || (j - i == i && str[0] < str[i]))
			res = 1;
		else
			res++;
		i = j;
	}

	std::cout << res << std::endl;
	return 0;
}
