#include <iostream>
using namespace std;
typedef long long ll;
#define Size(x) (int)(x).size()

int main() {
	ios::sync_with_stdio(false);
	string s;
	int L = 1, ans = 1;
	cin >> s;
	for(int i = Size(s) - 1;i > -1;--i) {
		if(s[i] == '0')
			++L;
		else {
			if(L < i || (L == i && s[0] >= s[i]))
				++ans;
			else
				break;
			L = 1;
		}
	}
	cout << ans << endl;
	return 0;
}
