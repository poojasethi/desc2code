# include <stdio.h>
# include <iostream>
# include <string.h>

# define MAXN 100009

using namespace std;

string s;
int n, ans;

int main(){
	
	cin >> s;
	
	n = s.length();
	
	for(int i=n-1; i>=0; i--){
		
		ans++;
		
		int r = i;
		
		while(s[i] == '0'){
			i--;
		}
		
		if(r-i > i-1)	break;
		
		if(r-i == i-1  &&  s.substr(0,i) < s.substr(i, r-i+1))	break;
	}
	
	printf("%d\n",ans);
}
