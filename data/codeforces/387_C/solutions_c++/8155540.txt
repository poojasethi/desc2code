# include <bits/stdc++.h>
using namespace std;
const int nmax = 1e6 + 5;
char s[nmax];
int main(void)
{
    cin>>s;
    long long p=0;
    for (int i=0,j;s[i];i=j)
    {
        for (j=i+1;s[j]=='0' && s[j];j++);
        if (j-i>i || (j-i==i && s[0]<s[i])) p=1;else ++p;
    }
    cout<<p<<'\n';
    return 0;
}
