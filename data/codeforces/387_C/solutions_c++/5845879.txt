#include <cstdio>

char in[100005];
int ans;

int main(){
    int i, j;

    scanf("%s", in);
    for( i = 0; in[i] != '\0'; i = j){
        for( j = i + 1; in[j] != '\0' && in[j] == '0'; j++);

        if( j - i > i || ( j - i == i && in[0] < in[i]))ans = 1;
        else ans++;
    }
    printf("%d\n", ans);
}
