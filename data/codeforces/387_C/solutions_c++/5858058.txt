#include<bits/stdc++.h>

using namespace std;
    

int main(){
    int n, i, cnt = 1;
    string s;
    cin >> s;
    n = s.length();
    
    for (i = n - 1; i >= 0; i--){
        if (s[i] == '0') continue;
        if (2 * i < n) break;
        if (2 * i == n && s[i] > s[0]) break;
        n = i;
        cnt++;
    }
    cout << cnt;
}
