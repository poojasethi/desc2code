#include <iostream>
#include <string>
using namespace std;
int main(){
    string p;
    cin >> p;
    long res = 1,len = 0;
    for (long i = p.size()-1;i>0;i--){
        len ++;
        if (p[i]=='0') continue;
        if (i>len || (p[0]>=p[i] && i==len))
            res++,len=0;
        else
            break;
    }
    cout << res << endl;
    return 0;
}
