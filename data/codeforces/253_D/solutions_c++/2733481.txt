#include <cstdio>
#define MAXN 440
using namespace std;
typedef long long big;
int n,m,K;
big res;
char ar[MAXN][MAXN];
int arr[MAXN][MAXN];
int H[128];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int i,j,k;
	int bas;
	scanf(" %d %d %d",&n,&m,&K);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{
			scanf(" %c",&ar[i][j]);
			arr[i][j]=arr[i-1][j]+arr[i][j-1]-arr[i-1][j-1];
			if(ar[i][j]=='a') arr[i][j]++;
		}
	for(i=1;i<=m;i++)
		for(j=i+1;j<=m;j++)
		{
			bas=0;
			for(k=1;k<=n;k++)
			{
				while(arr[k][j]-arr[k][i-1]-arr[bas][j]+arr[bas][i-1] > K)
				{
					bas++;
					if(bas && ar[bas][i]==ar[bas][j]) H[ar[bas][i]]--;
				}
				if(ar[k][i]==ar[k][j])
				{
					H[ar[k][i]]++;
					if(bas==k || bas+1==k || H[ar[k][i]]-1==0) continue;
					res+=H[ar[k][i]]-1;
				}
			}
			for(k=0;k<128;k++) H[k]=0;
		}
	printf("%I64d\n",res);
	return 0;
}
