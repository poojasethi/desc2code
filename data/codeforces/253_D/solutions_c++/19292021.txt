#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn = 405;
string str[maxn];
int cnt[maxn][maxn], t[maxn<<1];
int n, m, k;

int main(){
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	ios::sync_with_stdio(false);
	cin.tie(0);
	cin >> n >> m >> k;
	for (int i=0; i<n; ++i)
		cin >> str[i];
	memset(cnt, 0, sizeof(cnt));
	for (int i=0; i<n; ++i)
		for (int j=0; j<m; ++j){
			cnt[i+1][j+1] = cnt[i+1][j]+cnt[i][j+1]-cnt[i][j];
			if (str[i][j] == 'a')
				++cnt[i+1][j+1];
		}
	ll ans = 0;
	for (int i=0; i<n; ++i)
		for (int j=i+1; j<n; ++j){
			memset(t, 0, sizeof(t));
			for (int p=0, s=0; p<m; ++p){
				if (str[i][p] != str[j][p])
					continue;
				--t[str[i][p]];
				while (s<m && cnt[j+1][s+1]-cnt[j+1][p]-cnt[i][s+1]+cnt[i][p]<=k){
					if (str[i][s] == str[j][s])
						++t[str[i][s]];
					++s;
				}
				if (t[str[i][p]] > 0)
					ans += t[str[i][p]];
			}
		}
	cout << ans << endl;
	return 0;
}
