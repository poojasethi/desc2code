#include<cstdio>
#include<algorithm>
#define N 410
using namespace std;
char s[N][N];
int n, m, k, dp[N][N], alp[30]; long long ans;
int main(){
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    scanf("%d%d%d", &n, &m, &k);
    for(int i = 1; i <= n; i++) scanf("%s", s[i]+1);
    for(int i = 1; i <= n; i++) for(int j = 1; j <= m; j++)
        dp[i][j] = dp[i][j-1]+(s[i][j] == 'a');
    for(int x = 1; x <= m; x++) for(int y = x+1; y <= m; y++){
        int cnt = 0;
        for(int i = 0; i < 30; i++) alp[i] = 0;
        for(int i = 1, j = 1; i <= n; i++){
            for( ; j <= n && cnt+dp[j][y]-dp[j][x-1] <= k; j++){
                cnt += dp[j][y] - dp[j][x-1];
                if(s[j][x] == s[j][y]) alp[s[j][x]-'a']++;
            }
            if(i == j){ j++; continue; }
            if(s[i][x] == s[i][y]){
                alp[s[i][x]-'a']--;
                if(alp[s[i][x]-'a'] > 0) ans += alp[s[i][x]-'a'];
            }
            cnt -= dp[i][y] - dp[i][x-1];
        }
    }
    printf("%I64d\n", ans);
}
