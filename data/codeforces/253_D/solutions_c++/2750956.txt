#include <iostream>
#include <vector>
#include <stdio.h>
using namespace std;

vector <long> q[28];
long long n,m,i,j,k,l,ans,u,dp[401][401],last;
string a[401];

main()
{
      freopen ("input.txt","r",stdin);
      freopen ("output.txt","w",stdout);      
 cin>>n>>m>>k;

 for (i=1;i<=n;i++)
 cin>>a[i];

 for (i=1;i<=n;i++)
 for (j=0;j<a[1].size();j++)
{ dp[i][j+1]=dp[i][j]+dp[i-1][j+1]-dp[i-1][j]; 
if(a[i][j]=='a') dp[i][j+1]++;}


 for (i=1;i<=n-1;i++)
 for (j=i+1;j<=n;j++)
 {
   for (u=1;u<=26;u++)
    q[u].push_back(0);
     
 for (l=1;l<=m;l++)
 if (a[i][l-1]==a[j][l-1]) q[a[i][l-1]-'a'+1].push_back(l);

 for (u=1;u<=26;u++)
 {
 last=1; 
  for (l=1;l<=q[u].size()-1;l++)
  {
while (last<q[u].size()-1 && dp[j][q[u][last+1]]-dp[i-1][q[u][last+1]]-dp[j][q[u][l]-1] + dp[i-1][q[u][l]-1]<=k)last++;
     ans+=max(last-l,0LL);
  }
 }
 for (l=1;l<=26;l++)
  q[l].clear();
} 

cout<<ans<<endl;

//system ("pause");
}
