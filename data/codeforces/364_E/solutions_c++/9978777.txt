#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <vector>
#include <ctime>
#define fi first
#define se second
#define PA pair<int,int>
#define VI vector<int>
#define VP vector<PA >
#define mk(x,y) make_pair(x,y)
#define N 2510
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,K;
int64 an;
int a[N][N],b[N][N],c[N][N][10];
char p[N][N];
inline int cal(int x,int xx,int y,int yy) {
	return b[xx][yy]-b[x-1][yy]-b[xx][y-1]+b[x-1][y-1];
}
inline void suan(int x,int y,int xx,int yy) {
	int n=xx-x+1,m=yy-y+1,i,j,k,mid;
	For(i,x,xx)For(j,y,yy) a[i-x+1][j-y+1]=p[i][j]=='1';
	if (n<m) {
		For(i,1,n)For(j,1,m) b[j][i]=a[i][j];
		swap(n,m);
		For(i,1,n)For(j,1,m) a[i][j]=b[i][j];
		mid=(y+yy)/2-y+1;
	} else mid=(x+xx)/2-x+1;
	For(i,1,n)For(j,1,m) b[i][j]=b[i-1][j]+b[i][j-1]-b[i-1][j-1]+a[i][j];
	For(i,1,m) {
		int zhi[10];
		zhi[0]=mid;
		For(k,1,K+1) {
			zhi[k]=zhi[k-1]-1;
			if (zhi[k]<0) zhi[k]=0;
			for (;zhi[k]&&!a[zhi[k]][i];zhi[k]--);
		}
		For(j,i,m) {
			For(k,0,K) c[i][j][k]=zhi[k+1]-zhi[k];
			if (j==m) break;
			For(k,1,K+1)
				for (;zhi[k]<=mid&&cal(zhi[k]+1,mid-1,i,j+1)>=k;zhi[k]++);
		}
	}
	//
	For(i,1,m) {
		int zhi[10];
		zhi[0]=mid;
		For(k,1,K+1) {
			zhi[k]=zhi[k-1]+1;
			if (zhi[k]>n+1) zhi[k]=n+1;
			for (;zhi[k]<=n&&!a[zhi[k]][i];zhi[k]++);
		}
		zhi[0]=mid;
		For(j,i,m) {
			int si=cal(mid,mid,i,j);
			if (si>K) break;
			For(k,0,K-si) an+=-1ll*(zhi[k+1]-zhi[k])*c[i][j][K-si-k];
			if (j==m) break;
			For(k,1,K+1)
				for (;zhi[k]>=mid&&cal(mid+1,zhi[k]-1,i,j+1)>=k;zhi[k]--);
		}
	}
}
void work(int x,int y,int xx,int yy) {
	if (x>xx||y>yy) return;
	suan(x,y,xx,yy);
	if (xx-x>=yy-y) {
		int mid=(x+xx)/2;
		work(x,y,mid-1,yy);
		work(mid+1,y,xx,yy);
	} else {
		int mid=(y+yy)/2;
		work(x,y,xx,mid-1);
		work(x,mid+1,xx,yy);
	}
}
int main() {
	scanf("%d%d%d",&n,&m,&K);
	For(i,1,n) scanf("%s",p[i]+1);
	work(1,1,n,m);
	printf("%I64d\n",an);
	return 0;
}
