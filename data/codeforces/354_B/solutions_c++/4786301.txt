#include <iostream>
#include <vector>
#include <string>
#include <cmath>

using namespace std;

int n;
char t[50][50];

int memcomputa[50][1<<20];
int computado[50][1<<20];

int computa(int paso,int c)
{
  int &mem=memcomputa[paso][c];
  if (computado[paso][c]) return mem;
  computado[paso][c]=1;
  if (paso==n+n-2) {
    if (t[n-1][n-1]=='a') mem=1;
    else if (t[n-1][n-1]=='b') mem=-1;
    else mem=0;
    return mem;
  }
  int nextc[26];
  for (int k=0;k<26;k++)
    nextc[k]=0;
  int i=0,j=paso;
  int resta=0;
  if (paso>=n-1) resta=1;
  while (j>=n) {j--;i++;};
  int ic=0;
  while (i<n and j>=0) {
    if (c&(1<<ic)) {
      char car=t[i][j];
      if (j<n-1)
	nextc[car-'a']|=(1<<(ic-resta));
      if (i<n-1)
	nextc[car-'a']|=(1<<(ic-resta+1));
    }
    j--;i++;ic++;
  }
  bool hihasol=false;
  for (int inextc=0;inextc<26;inextc++) {
    if (nextc[inextc]) {
      int acum=0;
      if (inextc==0) acum=1;
      else if (inextc==1) acum=-1;
      int nextmem=computa(paso+1,nextc[inextc]);
      if (not hihasol) {
	hihasol=true;
	mem=acum+nextmem;
      } else if (paso%2==0) {
	mem=max(mem,acum+nextmem);
      } else {
	mem=min(mem,acum+nextmem);
      }
    }
  }
  return mem;
}

int main()
{
  cin>>n;
  for (int i=0;i<n;i++)
    for (int j=0;j<n;j++)
      cin>>t[i][j];
  int sol=computa(0,1);
  if (sol>0)
    cout<<"FIRST"<<endl;
  else if (sol<0)
    cout<<"SECOND"<<endl;
  else
    cout<<"DRAW"<<endl;
}
