#include <cstdio>
#include <iostream>
using namespace std;
#define inf (1<<29)
#define rep(i,a,b) for (int i=a;i<=b;++i)
#define dep(i,a,b) for (int i=a;i>=b;--i)
#define d(x) cout<<x<<endl
typedef long long ll;

int n;
int f[42][(1<<20)+10], mark[42][30];
char bo[30][30];

int calc(int x, int y) {
	if (x+y <= n+1) return x;
	else return x - (x+y - (n+1));
}

int work(int x, int y,int ch) {
	if (f[x][y] < inf) return f[x][y];
	if (!(x & 1)) f[x][y] = -inf; else f[x][y] = inf;
	if (x==2*n-1) f[x][y] = 0;
	else {
		rep(next,0,25) {
			int tmp = y | (y<<1);
			if (x>=n) tmp = y | (y>>1);
			int y1 = mark[x+1][next] & tmp;
			if (!y1) continue;
			int ans = work(x+1, y1, next);
			if (x%2==1) f[x][y] = min(f[x][y], ans);
			else f[x][y] = max(f[x][y], ans);
		}
	}
	if (ch== 0) f[x][y] ++; else if (ch == 1) f[x][y] --;
	return f[x][y];
}

int main() {
	scanf("%d",&n);
	rep(i,1,n) scanf("%s",bo[i]+1);
	rep(i,1,n*2-1) rep(j,0,1<<n) f[i][j] = inf;
	rep(i,1,n)
		rep(j,1,n)
			mark[i+j-1][bo[i][j] - 'a'] |= 1<<(calc(i,j)-1);
	int ans = work(1,1, bo[1][1]-'a');
	if (ans > 0) puts("FIRST");
	else puts(ans ? "SECOND" : "DRAW");
	return 0;
}
