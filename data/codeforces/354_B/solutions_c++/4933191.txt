#include <cstdio>
using namespace std;

#define min(x, y) ((x) < (y) ? (x) : (y))
int n, i, j, l, p[200], dp[41][1 << 20];
char a[30][30];
bool used[41][1 << 20];

void calc(int l, int S) {
	if (used[l][S])	return ;
	if (l == 2 * n) {
		dp[l][S] = p[a[n][n]];
		return ;
	}
	dp[l][S] = 1000000000;
	int delta;
	for (int i = 0; i < n; i++)
		if (S & (1 << i)) {
			delta = ((l & 1) ? -p[a[i + 1][l - i - 1]] : p[a[i + 1][l - i - 1]]);
			break;
		}
	for (int i = 'a'; i <= 'z'; i++) {
		int ss = 0;
		for (int j = 1; j <= n; j++)
			if (S & (1 << (j - 1))) {
				if (a[j][l + 1 - j] == i)	ss |= (1 << (j - 1));
				if (a[j + 1][l - j] == i)	ss |= (1 << j);
			}
		if (ss) {
			calc(l + 1, ss);
			dp[l][S] = min(dp[l][S], -dp[l + 1][ss] + delta);
		}
	}
	used[l][S] = true;
}

int main() {
	scanf("%d", &n);
	p['a'] = 1;
	p['b'] = -1;
	for (i = 1; i <= n; i++)
		for (j = 1; j <= n; j++)
			for (a[i][j] = getchar(); a[i][j] <= 32; a[i][j] = getchar());

	calc(2, 1);
	
	if (dp[2][1] > 0)	printf("FIRST\n");
	else	if (dp[2][1] < 0)	printf("SECOND\n");
	else	printf("DRAW\n");
}
