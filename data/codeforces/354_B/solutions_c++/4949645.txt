#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int inf=1000000000;
char w[51];
bool v[51][(1<<20)+50];
int i,j,k,n,m,a[51][51],f[51][(1<<20)+50],b[51][51],p[52][51][2];
int work(int x,int s){
	if (v[x][s]) return f[x][s];
	v[x][s]=1;
	int i,j,k=0,t,x1,y1;
	for (i=1;!((s>>(i-1))&1);i++);
	if (a[p[x][i][0]][p[x][i][1]]==0) k=1;
	if (a[p[x][i][0]][p[x][i][1]]==1) k=-1;
	if (x==2*n-1) return f[x][s]=k;
	f[x][s]=(x&1)?inf:-inf;
	for (i=0;i<26;i++){
		for (t=0,j=1;j<=n;j++)
		 if ((s>>(j-1))&1){
		 	x1=p[x][j][0];y1=p[x][j][1];
		 	if (x1<n&&a[x1+1][y1]==i) t|=1<<(b[x1+1][y1]-1);
		 	if (y1<n&&a[x1][y1+1]==i) t|=1<<(b[x1][y1+1]-1);
		 }
		if (!t) continue;
		if (x&1) f[x][s]=min(f[x][s],work(x+1,t));else f[x][s]=max(f[x][s],work(x+1,t));
	}
	f[x][s]+=k;
	return f[x][s];
}
int main(){
	scanf("%d",&n);memset(v,0,sizeof(v));
	for (i=1;i<=n;i++){
		scanf("%s",w);
		for (j=1;j<=n;j++) a[i][j]=w[j-1]-'a',b[i][j]=(i+j-1<=n)?j:j-(i+j-1-n),p[i+j-1][b[i][j]][0]=i,p[i+j-1][b[i][j]][1]=j;
	}
	work(1,1);
	if (f[1][1]==0) puts("DRAW");
	if (f[1][1]>0) puts("FIRST");
	if (f[1][1]<0) puts("SECOND");
	return 0;
}
