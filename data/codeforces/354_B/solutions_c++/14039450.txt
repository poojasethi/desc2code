#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int n;
char T[40][25],buf[25];
int f[40][1<<20],used[40][1<<20],flag[40],mask[40][26];

inline int SCORE(char a){return a=='a' ? 1 : a=='b' ? -1 : 0;}
int F(int k,int st){
	if(used[k][st]) return f[k][st];
	used[k][st]=1;
	int tt=(st|(st<<1))&flag[k+1];
	int &t=f[k][st];t=(k&1 ? 0x80000000 : 0x7fffffff);
	for(int j=0;j<26;j++)
		if(tt & mask[k+1][j])
			if(k&1) t=max(t,F(k+1,tt&mask[k+1][j]));
			else t=min(t,F(k+1,tt&mask[k+1][j]));
	t+=SCORE(T[k][__builtin_ctz(st)]);
	return t;
}
int main()
{
	scanf("%d",&n);
	
	for(int i=0;i<n;i++){
		scanf("%s",buf);
		for(int j=0;j<n;j++)
			T[i+j][j]=buf[j],flag[i+j]|=1<<j,mask[i+j][buf[j]-'a']|=1<<j;
	}
	f[n+n-2][1<<n-1]=SCORE(T[n+n-2][n-1]);used[n+n-2][1<<n-1]=1;
	int res=F(0,1);
	puts(res<0 ? "SECOND" : res==0 ? "DRAW" : "FIRST");
	return 0;
}
