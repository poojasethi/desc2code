#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1100000
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int f[40][N],h[30][30],v[40][N];
char p[30][30];
int dfs(int x,int y,int z) {
	z=!z?1:z==1?-1:0;
	if (v[x][y]) return f[x][y];
	if (x==2*n-1) return z;
	v[x][y]=1;
	int i;
	x++;
	int ll=x<=n?x:n+n-x;
	int s=0,an=x&1?-10000000:+10000000;
	For(i,0,ll-1) if ((y>>i&1)||x<=n&&(y>>i-1&1)||x>n&&(y>>i+1&1)) s|=1<<i;
	For(i,0,ll-1) if (s>>i&1) {
		int X=x<=n?x-i:n-i;
		int Y=x<=n?i+1:x-n+i+1;
		if (x&1) an=max(an,dfs(x,h[X][Y]&s,p[X][Y]-'a')+z);
		else an=min(an,dfs(x,h[X][Y]&s,p[X][Y]-'a')+z);
	}
	f[x-1][y]=an;
	return an;
}
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%s",p[i]+1);
	For(i,1,2*n-1)For(j,0,25) {
		int x=min(i,n),y=max(1,i-n+1),s=0;
		for (k=0;x&&y<=n;x--,y++,k++) if (p[x][y]-'a'==j) s|=1<<k;
		x=min(i,n),y=max(1,i-n+1);
		for (;x&&y<=n;x--,y++) if (p[x][y]-'a'==j) h[x][y]=s;
	}
	k=dfs(1,1,p[1][1]-'a');
	printf("%s\n",!k?"DRAW":k>0?"FIRST":"SECOND");
	//for(;;);
	return 0;
}
