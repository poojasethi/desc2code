#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int n;
char grid[50][25];
char buf[25];
int f[50][1<<20],used[50][1<<20];
int state[50][30];

int val(char a){return a=='a' ? 1 : (a=='b' ? -1 : 0);}

int F(int num,int st){
	if(used[num][st]) return f[num][st];
	used[num][st]=1;int q,cnt;
	if(num&1){
		if(num==n*2-1) return f[num][st]=val(grid[num][n]);
		int tst=st|(st<<1),tmp=0x7fffffff;
		for(int j=0;j<26;j++) if(state[num+1][j]&tst)
			tmp=min(tmp,F(num+1,tst&state[num+1][j]));
		for(q=st&(-st),cnt=0;q;q>>=1,cnt++);
		
		return f[num][st]=tmp+val(grid[num][cnt]);
	}
	else{
		int tst=st|(st<<1),tmp=0x80000000;
		for(int j=0;j<26;j++) if(state[num+1][j]&tst)
			tmp=max(tmp,F(num+1,tst&state[num+1][j]));
		for(q=st&(-st),cnt=0;q;q>>=1,cnt++);
	//	printf("%d %d %d\n",num,st,f[num][st]);
		return f[num][st]=tmp+val(grid[num][cnt]);
	}
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%s",buf);
		for(int j=0;j<n;j++) grid[i+j][j+1]=buf[j],state[i+j][buf[j]-'a']|=1<<j;
	}
	int res=F(1,1);
	//printf("%d\n",res);
	puts(res>0 ? "FIRST" :(res==0 ? "DRAW" : "SECOND"));
	return 0;
}
