#include<cstdio>
#include<iostream>
#include<cmath>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<queue>
#define mk make_pair
using namespace std;
int i,j,k,l,n,m,sum,an;
const int N=21;
char a[N][N];
int f[N*2][1<<N];
bool ff[N*2][1<<N];
int dfs(int dep,int now)
{
    if (dep==2*n-1) return 0;
    if (ff[dep][now]) return f[dep][now];
    ff[dep][now]=true;
    if (dep&1) f[dep][now]=999999;
    else f[dep][now]=-999999;
    for (char x='a';x<='z';x++)
    {
        int v=0;
        if (x=='a') v=1;
        if (x=='b') v=-1;
        int tt=0;
        for (int i=1;i<=n;i++)
        if (now&(1<<(i-1)))
        {
            if (a[i][dep-i+1+1]==x) tt|=(1<<(i-1));
            if (a[i+1][dep-i+1]==x) tt|=1<<i;
        }
        if (tt)
        {
            if (dep&1) f[dep][now]=min(f[dep][now],dfs(dep+1,tt)+v);
            else  f[dep][now]=max(f[dep][now],dfs(dep+1,tt)+v);
        }
    }
    return f[dep][now];
}
int main()
{
    scanf("%d",&n);
    for (int i=1;i<=n;i++)
    scanf("%s",a[i]+1);
    int v=0;
    if (a[1][1]=='a') v=1;
    if (a[1][1]=='b') v=-1;
    an=v;
    if (n!=1)
    an+=dfs(1,1);
    if (an>0) printf("FIRST\n");
    else if (an==0) printf("DRAW\n");
    else printf("SECOND\n");
    //system("pause");
    return 0;
}
