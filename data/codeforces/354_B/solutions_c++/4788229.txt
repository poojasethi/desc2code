#include<cstdio>
#include<algorithm>
#define oo 2000000
using namespace std;
int mp[40][40];
int f[40][40];
int i,j,tmp1,tmp2,flag;
int n; char ch;
bool vis[42][(1<<19)+5];
int dp[42][(1<<19)+5];
int dfs(int x,int y,int who,int zm)
{ 
	 if (x==2*n-1)
		 if (zm==1) return 1;
		 else if (zm==2)return -1;
		 else return 0;
	 if (vis[x][y]) return dp[x][y];
	 vis[x][y]=1;
	 int go[30],i;
	 for(i=1;i<=26;i++)
		 go[i]=0;
	 if (x<n){
		 for(i=0;i<x;i++)
			 if ((y>>i)&1){
				 go[mp[x+1-i][i+1]]|=(1<<i);
				 go[mp[x-i][i+2]]|=(1<<(i+1));
			 }
	 }else{
		 for(i=1;i<2*n-x;i++)
			 if ((y>>i)&1)
				 go[mp[n+1-i][x-n+i+1]]|=(1<<(i-1));
		 for(i=0;i<2*n-x-1;i++)
			 if ((y>>i)&1)
				 go[mp[n-i][x-n+i+2]]|=(1<<(i));
	 }
	 int final=who*oo; int tmp;
	 for(i=1;i<=26;i++)
		 if (go[i]){
			 tmp=dfs(x+1,go[i],-who,i);
			 if (who==1) final=min(final,tmp);
			 else final=max(final,tmp);
		 }
	 dp[x][y]=final;
	 if (zm==1) dp[x][y]++;
	 else if (zm==2) dp[x][y]--;
	 return dp[x][y];
} 
int main()
{
	//freopen("cf.in","r",stdin);
	//freopen("cf.out","w",stdout);
	scanf("%d\n",&n);
	for(i=1;i<=n;i++){
		for(j=1;j<=n;j++){
			ch=getchar();
			mp[i][j]=ch-'a'+1;
		}
		ch=getchar();
	}
	flag=dfs(1,1,1,mp[1][1]);
	if (flag>0)
		printf("FIRST");
	else if (flag<0)
		printf("SECOND");
	else printf("DRAW");
	return 0;
}
