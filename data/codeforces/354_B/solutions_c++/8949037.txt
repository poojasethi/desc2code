#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <string>
#include <cstring>
using namespace std;
int dp[42][1048576],sum[42],i,j,Q[1005],n,k;
char s[105][105];
const int inf=10000005;
int work(char a)
{
    if (a=='a') return 1;
    if (a=='b') return -1;
    return 0;
}
int main()
{
    scanf("%d",&n);
    for (i=1; i<=n*2-1; i++) sum[i]=(1<<n-abs(n-i))-1;
    for (i=1; i<=n*2-1; i++) for (j=1; j<=sum[i]; j++) if (i%2==1) dp[i][j]=inf; else dp[i][j]=-inf;
    for (i=0; i<n; i++) scanf("%s",s[i]);
    dp[n*2-1][1]=0;
    for (i=n*2-2; i>=1; i--)
      for (j=1; j<=sum[i]; j++)
        {
            for (k='a'; k<='z'; k++) Q[k]=0;
            if (i<n)
            {
                for (k=1; k<=i; k++) if (((1<<k-1)&j) || k!=1 && ((1<<k-2)&j))Q[s[k-1][i-k+1]]|=(1<<k-1);
                if ((1<<i-1)&j) Q[s[i][0]]|=(1<<i);
            } else
              for (k=i-n+2; k<=n; k++)
                  if (i==n && ((j&(1<<k-1)) || (j&(1<<k-2))) || i!=n && (((1<<(n-k))&j) || ((1<<n-k+1)&j))) Q[s[k-1][n-(k-(i-n+2))-1]]|=(1<<(n-k));

            for (k='a'; k<='z'; k++) if (Q[k])
            {
                if (i%2==0) dp[i][j]=max(dp[i][j],dp[i+1][Q[k]]+work(k)); else
                  dp[i][j]=min(dp[i][j],dp[i+1][Q[k]]+work(k));
            }
        }
    dp[1][1]+=work(s[0][0]);
    if (dp[1][1]>0) cout<<"FIRST"; else if (dp[1][1]<0) cout<<"SECOND"; else cout<<"DRAW";
    return 0;
}
