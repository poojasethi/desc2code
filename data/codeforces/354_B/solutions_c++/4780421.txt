#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#define score(s)  (s == 'a' ? 1 : (s == 'b' ? -1 : 0))
const int inf = (int) 1e9 + 10;
using namespace std;

char s[110][110], e[110][110];
bool vis[40][1 << 20];
int g[40][1 << 20], d[1 << 20], si[110];
int n;

int f(int x, int t)
{
	if (vis[x][t])  return g[x][t];
	vis[x][t] = 1;

	int &r = g[x][t], lt = 0;
	r = (x & 1 ? 1 : -1) * inf;

	for (int i = 0; i < si[x + 1]; ++i)  {
		if (x == 2  &&  i == 2)
			i = 2;
		lt = 0;
		for (int j = 0; j < si[x + 1]; ++j)
			if (e[x + 1][j] == e[x + 1][i])
				if (t & (1 << j)  ||  t & (1 << (j + (x < n ? -1 : 1))))
					lt += 1 << j;

		if (lt)
			if (x & 1)  r = min(r, f(x + 1, lt));
			else  r = max(r, f(x + 1, lt));
	}

	return r += score(e[x][d[t & -t]]);
}
int main()
{
	for (int i = 0; i < 20; ++i)  d[1 << i] = i;
	//freopen("B.in", "r", stdin);
	//freopen("B.out", "w", stdout);

	scanf("%d", &n);
	for (int i = 0; i < n; ++i)
		scanf("%s", s[i]);

	for (int i = 0; i < n; ++i)
		for (int j = 0; j < n; ++j)
			e[i + j + 1][si[i + j + 1] ++] = s[i][j]; 

	vis[n * 2 - 1][1] = 1;
	g[n * 2 - 1][1] += score(s[n - 1][n - 1]);

	f(1, 1);
	if (g[1][1] < 0)  printf("SECOND");
	else if (g[1][1] == 0)  printf("DRAW");
	else printf("FIRST");
}
