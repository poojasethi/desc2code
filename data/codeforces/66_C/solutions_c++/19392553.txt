#include<iostream>
#include<cstdio>
#include<map>
#include<cstring>
#include<string>
using namespace std;
int findchar(string str,char ch){
	for(int i=str.length()-1; i>=0; i--)
		if(str[i]==ch) return i;
	return -1;
} 
map<string,int> FILE_NUM,SUBFOLDER_NUM;
int main(){
	string str;
	int a=0,b=0;
	while(cin>>str){
		int tmp_num=0;
		while(true){
			int index = findchar(str,'\\');
			if(index==2) break;
			str=str.substr(0,index);
			SUBFOLDER_NUM[str] += tmp_num;
			if(!FILE_NUM[str])
				tmp_num++;
			FILE_NUM[str]++;
			a = max(a,SUBFOLDER_NUM[str]);
			b = max(b,FILE_NUM[str]);
		}
	}
	cout<<a<<" "<<b<<endl;
	return 0;
}
