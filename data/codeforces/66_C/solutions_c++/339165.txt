#include <iostream>
#include <string>
#include <map>
#include <algorithm>
using namespace std;
typedef pair<int, int> ii;
const ii FileNULL = ii(0, 0);
int main()
{
	map<string, ii> FS;
	string str;
	
	while (cin >> str)
	{
		int numFolder = 0, numFile = 1;
		int pos;
		while ( (pos = str.find_last_of('\\')) != 2)
		{
			str = str.substr(0, pos);
			bool newFolder = FS[str] == FileNULL;
			FS[str].first += numFolder;
			FS[str].second += numFile;
			if (newFolder)
				numFolder ++;
		}
	}
	int maxFiles = 0, maxFolders = 0;
	for (map<string, ii>::iterator iter = FS.begin(); iter!=FS.end(); iter++)
	{
		if (iter->second.first > maxFolders) maxFolders = iter->second.first;
		if (iter->second.second > maxFiles) maxFiles = iter->second.second;
	}	
	cout<<maxFolders<<' '<<maxFiles<<endl;
}
