#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
#include<queue>
#include<algorithm>
#include<set>
#include<map>
using namespace std;
#define INF 0x3f3f3f3f

map<string,int> file;
map<string,int> folder;
string s;
int maxfolder,maxfile;
int work(){
	int len = s.length();
	int now = len-1;
	while (now >= 0){
		if (s[now]=='\\')
			return now;
		else now--;
	}
}
int main(){
	folder.clear();
	file.clear();
	maxfolder = maxfile = 0;
	while (cin>>s){
		int sumfold = 0;
		int flag;
		while (1){
			int pos = work();
			if (pos == 2)
				break;
			else {
				s = s.substr(0,pos);
				flag = file[s];
				folder[s] += sumfold;
				file[s]++;
				maxfolder = max(maxfolder,folder[s]);
				maxfile = max(maxfile,file[s]);
				if (!flag) sumfold++;
			}
		}
	}
	cout<<maxfolder<<" "<<maxfile<<endl;
	return 0;
}
