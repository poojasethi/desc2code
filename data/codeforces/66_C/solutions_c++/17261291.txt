
#define N (1<<7)

#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <queue>
#include <vector>
#include <stack>
#include <map>
#include <set>
#include <deque>
#include <string>
#include <cstring>
#include <utility>
using namespace std;

char tmp[N];
map<string,int> mp1;
map<string,int> mp2;
int Ans1,Ans2;

int main()
{
	while(scanf("%s",tmp)==1)
	{
		string s=tmp;
		int F=0;
		while(true)
		{
			int pos=s.find_last_of('\\');
			if(pos==2)
			{
				break;
			}
			s=s.substr(0,pos);
			int f=mp1[s];
			mp2[s]+=F;
			mp1[s]++;
			if(!f)
			{
				F++;
			}
		}
	}
	for(map<string,int>::iterator it=mp1.begin();it!=mp1.end();it++)
	{
		Ans1=max(Ans1,it->second);
	}
	for(map<string,int>::iterator it=mp2.begin();it!=mp2.end();it++)
	{
		Ans2=max(Ans2,it->second);
	}
	printf("%d %d\n",Ans2,Ans1);
	return 0;
}
