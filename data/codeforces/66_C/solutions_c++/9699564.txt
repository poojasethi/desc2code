#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<stdlib.h>
#include<map>
#include<string>
#include<queue>
#include<algorithm>
#define mod 1000000007
#define maxn 100005
#define ll __int64
using namespace std;
string s;
map<string,int> f,b;
map<string,int>::iterator it;

int main()
{
    f.clear() , b.clear();
    int i,j,n,cnt,x,y;
	while(cin>>s)
	{
		cnt=0;
		while(1)
		{
			int pos=s.find_last_of('\\'),flag;
			if(pos == 2) break;
			s.erase(s.begin()+pos,s.end());
			flag= f[s]==0;
			b[s]+=cnt;
			if(flag) cnt++;
			f[s]++;
		}
	}
	x=y=0;
	for(it=b.begin();it!=b.end();it++)
		x=max(x,it->second);
	for(it=f.begin();it!=f.end();it++)
		y=max(y,it->second);
	cout<<x<<" "<<y<<endl;
    return 0;
}
