#include<iostream>
#include<cstring>
#include<cstdio>
#include<map>
#include<algorithm>
using namespace std;
string s;
map<string,int>fold,file;
map<string,int>::iterator it;
int main()
{
    fold.clear();
    file.clear();
    while(cin>>s)
    {
        int v=0;
        while(1)
        {
            int p=s.find_last_of('\\');
            if(p==2)
                break;
            s.erase(s.begin()+p,s.end());
            bool tag=fold[s]==0&&file[s]==0;
            fold[s]+=v;
            if(tag)
                v++;
            file[s]++;
        }
    }
    int fo=0,fi=0;
    for(it=fold.begin();it!=fold.end();it++)
        fo=max(fo,it->second);
    for(it=file.begin();it!=file.end();it++)
        fi=max(fi,it->second);
    cout<<fo<<" "<<fi<<endl;
    return 0;
}
