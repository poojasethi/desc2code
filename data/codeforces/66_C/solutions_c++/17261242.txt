using namespace std;

string s;
map<string,int> m,n;
int x,c=0,a,b;

int main()
{
	while(cin>>s)
	{
		c=0;
		while(true)
		{
			x=s.find_last_of('\\');
			if(x<=2) break;
			s=s.substr(0,x);
			int t=m[s];
			m[s]++;
			n[s]+=c;
			if(t==0) c++;
			a=max(a,m[s]);
			b=max(b,n[s]);
		}
	}
	cout<<b<<" "<<a;
	return 0;
}
