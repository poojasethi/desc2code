#include <iostream>
#include <sstream>
#include <iomanip>
#include <string>
#include <cmath>
#include <map>
#define EPS 1e-7

using namespace std;

int main() {
	map<string,int> folder;
	map<string,int> checked;
	map<string,int> file;
	string str;
	while (cin >> str) {
		int it=0;
		while(str[it]!='\\')it++;
		it++;
		while(str[it]!='\\')it++;
		it++;
		string topfolder=str.substr(0,it);
		file[topfolder]++;
		for (int i=it; i<str.size(); i++) {
			if (str[i]=='\\'&&checked.find(str.substr(0,i+1))==checked.end()){
				folder[topfolder]++;
				checked[str.substr(0,i+1)]=1;
			}
		}
	}
	int foldernum=0,filenum=0;
	for (typeof(folder.begin()) ite=folder.begin(); ite!=folder.end(); ++ite) {
		foldernum=max(foldernum,ite->second);
	}
	for (typeof(file.begin()) ite=file.begin(); ite!=file.end(); ++ite) {
		filenum=max(filenum,ite->second);
	}
	cout << foldernum << " " << filenum << endl;
}
