#include <cstdio>
#include <string>
#include <cstring>
#include <map>
#include <iostream>
using namespace std;

#define REP(i,n) for(int i = 0; i < n; i++)
#define FOR(i,a,b) for(int i = a; i <= b; i++)

int ans1,ans2,k;
int count1[10015], count2[10015];
map<string,int> id;
string file;

int main()
{
	int c = 0;
	ans1 = ans2 = 0;
	while(cin >> file){
		int i = 0;
		while(file[i] != '\\' || file[i-1] == ':')i++;
		string dir = file.substr(0,i);
		
		if(!id.count(dir)) id[dir] = c++;
		k = id[dir];
		
		while(i < file.length()){
			dir += file[i++];
			while(i < file.length() && file[i] != '\\') dir += file[i++];
			
			if(i >= file.length())break;
			
			if(!id.count(dir)){
				ans1 = max(ans1, ++count1[k]);
				id[dir] = c++;
			}
		}
		
		ans2 = max(ans2, ++count2[k]);
	}
	
	printf("%d %d\n",ans1,ans2);
}
