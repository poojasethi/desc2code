#include <iostream>
#include <vector>
#include <string>
#include <map>
#include <set>
#include <algorithm>
using namespace std;

vector<string> a;
map<string, set<string> > c;
map<string, int> d;

void func() {
	int r1 = 0, r2 = 0;
	for (int i = 0; i < a.size(); ++ i) {
		vector<string> b;
		for (int j = 3; j < a[i].length(); ++ j) {
			if (a[i][j] == '\\') {
				b.push_back(a[i].substr(0,j));
			}
		}
		for (int k = 0; k < b.size(); ++ k) {
			for (int j = 0; j < k; ++ j) {
				c[b[j]].insert(b[k]);
			}
			r2 = max(r2, ++ d[b[k]]);
		}
	}
	typedef map<string, set<string> > c_t;
	for (c_t::iterator i = c.begin(); i != c.end(); ++ i) {
		r1 = max<int>(r1, i->second.size());
	}
	cout << r1 << " " << r2 << endl;
}

int main() {
	string buf;
	while (getline(cin, buf)) {
		a.push_back(buf);
	}
	func();
}
