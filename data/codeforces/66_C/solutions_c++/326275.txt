#include <map>
#include <string>
#include <iostream>
using namespace std;

map<string, int> subfolder, subfile;

int main(void) {
	string str;
	int i;
	cin.sync_with_stdio(false);
	while(cin >> str) {
		int f = 0;
		int sum=0;
		for(i=str.length(); i>=3; i--) {
			if(str[i] == '\\') {
				string v = str.substr(0, i);
				int rsum = sum;
				if(subfolder.find(v) == subfolder.end()) {
					++sum;
				}
				subfolder[v] += rsum;
				subfile[v] += 1;
			}
		}
	}
	map<string, int>::iterator p;
	int mfolder=0, mfile=0;
	for(p = subfolder.begin(); p != subfolder.end(); ++p) {
		if(mfolder < p->second)
			mfolder = p->second;
	}
	for(p = subfile.begin(); p != subfile.end(); ++p) {
		if(mfile < p->second)
			mfile = p->second;
	}
	cout << mfolder << ' ' << mfile << endl;
	return 0;
}
