#include<iostream>
#include<map>
#include<cstring>
using namespace std;
map<string,int>m,n;
string s;
int f,F,a,b;
int main()
{
	while(cin>>s)
	{
		F=0;
		while(1)
		{
			int x=s.find_last_of('\\');
			if(x==2)break;
			s=s.substr(0,x);
			f=n[s];
			m[s]+=F;
			n[s]++;
			a=max(a,m[s]);
			b=max(b,n[s]);
			if(!f)F++;
		}
		
	}
	cout<<a<<" "<<b;
	return 0;
}
