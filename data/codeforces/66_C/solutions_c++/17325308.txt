    #include <iostream>
    #include <string>
    #include <set>  
    #include <algorithm>
    #include <vector>
    #include <cmath>
    #include <queue>
    #include <cstring>
    #include <cstdio>
    #include <map>
using namespace std;
map<string,set<string> >sub;
map<string,int>cnt;
char buff[200],s[200];
int i,j,k,ansfo,ansfi;
int main()
{
	while(gets(buff))
	{
		for(k=3;buff[k]!='\\';k++);
		for(j=0;j<k;j++)s[j]=buff[j];
		s[k]=0;
		cnt[s]++;
		for(i=strlen(buff)-1;i>k;i--)
		{
			if(buff[i]=='\\')
			{
				buff[i]=0;
				sub[s].insert(buff);
			}
		}
	}
	for(map<string,set<string> >::iterator it=sub.begin();it!=sub.end();it++)
		ansfo=max(ansfo,(int)((it->second).size()));
	for(map<string,int>::iterator it=cnt.begin();it!=cnt.end();it++)
		ansfi=max(ansfi,it->second);
	cout<<ansfo<<" "<<ansfi;
	return 0;
}
