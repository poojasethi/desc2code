#include<iostream>
#include<string>
#include<map>
#include<vector>
#include<algorithm>
#include<set>
using namespace std;

map< string , set<string> > subfolders;

map<string,int> files;

void add_subfolders(string folder, string line)
{
	size_t i;
	while( ( i = line.find_last_of('\\') ) !=  string::npos )
	{
		line = line.substr(0,i);
		subfolders[folder].insert(line);
	}
}

int main()
{
	int m1 = 0;
	int m2 = 0;
	string s;
	while(cin>>s)
	{
		s[2] = '*';
		int l=s.find('\\');
		string folder = s.substr( 0, l);
		add_subfolders(folder,s.substr(  l+1, s.size()-l-1  ));
		
		files[folder]++;
		
		if(subfolders[folder].size() > m2) m2 = subfolders[folder].size();
		if(files[folder] > m1) m1 = files[folder];
	}
	
	cout<<m2<<" "<<m1<<endl;
}
