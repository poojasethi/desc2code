using namespace std;

#define REP(i,n) for(int i=0;i<(int)n;++i)
#define FOR(i,c) for(__typeof((c).begin())i=(c).begin();i!=(c).end();++i)
#define ALL(c) (c).begin(), (c).end()
const int INF = 1<<30;

int main() {  
  string s;
  map<string, int> mp;
  while(cin >> s) {
    while(true) {
      s = s.substr(0, s.rfind('\\'));
      if (s[s.size()-1] == ':') break;
      mp[s+"\\"]++;
    }
  }
  int a1 = 0, a2 = 0;
  FOR(it, mp) {
    a2 = max(a2, it->second);
    int now = 0;
    FOR(jt, mp)
      if (it->first < jt->first && jt->first.find(it->first) == 0)
        now++;
    a1 = max(a1, now);
  }
  printf("%d %d\n", a1, a2);
}
