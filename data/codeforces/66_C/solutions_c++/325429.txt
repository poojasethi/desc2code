#include<iostream>
#include<cstdio>
#include<vector>
#include<map>
#include<set>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#include<string>

using namespace std;

int main(){
  map<string,set<string> > m1,m2;
  
  string s;
  while(cin >> s){
    int i=0;
    while(s[i]!='\\')i++;
    i++;
    while(s[i]!='\\')i++;
    int j=i+1; i++;
    while(s[j]!='\0'){
      if(s[j]=='\\'){
	m1[s.substr(0,i)].insert(s.substr(0,j));
	//	cout << s.substr(0,j) << endl;
      }
      j++;
    }
    m2[s.substr(0,i)].insert(s);
  }
  int a=0,b=0;
  for(map<string,set<string> >::iterator it=m1.begin();it!=m1.end();
      ++it){
    a=max(a,(int)((*it).second.size()));
  }
  for(map<string,set<string> >::iterator it=m2.begin();it!=m2.end();
      ++it){
    b=max(b,(int)((*it).second.size()));
  }
  cout << a << " " << b << endl;
  return 0;
}
