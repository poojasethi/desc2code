#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
#include <cstdio>
#include <cstring>
#include <set>
using namespace std;
string s;
int ansa,ansb;
map <string,int> a,b;
int main()
{
	while(cin>>s)
	{
	int sum=0;
	for(;;)
	{
	int k=s.find_last_of('\\');
	if(k==2)
	break;
	s=s.substr(0,k);
	int t=b[s];
	a[s]+=sum;
	b[s]++;
	ansa=max(a[s],ansa);
	ansb=max(b[s],ansb);
	if(!t)
	sum++;
	}
	}
	cout<<ansa<<" "<<ansb;
	return 0;
}
