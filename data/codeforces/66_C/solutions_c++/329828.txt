#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <string>
#include <map>
#include <set>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 6
#define lim2 101

int i,j,f,d;
string str;
map <string,int> M[lim];
int S[lim];
set <string> D[lim][lim2],F[lim][lim2];

int main()
{
  while(cin>>str)
  {
    for(j=3;str[j]!='\\';j++);
    if(!M[str[0]-'C'][str.substr(0,j)])
      M[str[0]-'C'][str.substr(0,j)]=++S[str[0]-'C'];
    for(i=j;i<str.size();i++)
      if(str[i]=='\\')
        D[str[0]-'C'][M[str[0]-'C'][str.substr(0,j)]].insert(str.substr(0,i));
    F[str[0]-'C'][M[str[0]-'C'][str.substr(0,j)]].insert(str);
  }  
  for(i=0;i<5;i++)
    for(j=1;j<=S[i];j++)
    {
      d=max(d,(int)D[i][j].size());
      f=max(f,(int)F[i][j].size());
    }
  printf("%d %d\n",d-1,f);
  return 0;
}
