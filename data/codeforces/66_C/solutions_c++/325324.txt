#include <iostream>
#include <map>
#include <vector>
#include <string>
#include <algorithm>
#include <cstdio>
using namespace std;

vector<string> x;
map<string, int> a, b;
int main()
{
	string s;
	while (cin >> s)
	{
		string t;
		t.push_back(s[0]);
		t.push_back(s[1]);
		t.push_back(s[2]);
		x.clear();
		for (int i = 3; i < s.size(); ++i)
		{
			t.push_back(s[i]);
			if (s[i] == '\\')
			{
				if (a.find(t) == a.end())
				{
					a[t] = 0;
					for (int j = 0; j < x.size(); ++j)
						++a[x[j]];
				}
				x.push_back(t);
			}
		}
		for (int j = 0; j < x.size(); ++j)
			if (b.find(x[j]) == b.end())
				b[x[j]] = 1;
			else
				++b[x[j]];
	}
	
	map<string, int>::iterator I;
	int q = 0, w = 0;
	for (I = a.begin(); I != a.end(); ++I)
		q = max(q, I->second);
	for (I = b.begin(); I != b.end(); ++I)
		w = max(w, I->second);
	cout << q << ' ' << w << '\n';
	return 0;
}
