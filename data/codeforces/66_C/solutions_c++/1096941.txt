#include <iostream>
#include <map>
#include <string>
#include <cstring>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
#define rep2(i,n) for(int i=3;i<n;i++)

string s;
map<string,int> d,d2;
map<string,int>::iterator it;
int r1,r2,cnt,pos,bi,sz;

int main(){

  while(cin>>s){

    cnt=0;pos=-1;bi=0;sz=s.size();
    rep(i,sz)if(s[i]=='\\')cnt++;
    if(cnt<=1)continue;
    else rep2(i,sz)if(s[i]=='\\'){pos=i;break;}    
    rep2(i,sz)if(s[i]=='\\'&&++d[s.substr(0,i)]==1)d2[s.substr(0,pos)]++;

  }
  
  it=d2.begin();
  while(it!=d2.end()){r1=max(r1,it->second);it++;}
  it=d.begin();
  while(it!=d.end()){r2=max(r2,it->second);it++;}

  cout<<(r1-1)<<" "<<r2<<endl;


  return 0;
}
