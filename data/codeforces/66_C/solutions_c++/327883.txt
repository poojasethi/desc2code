#include <cstdio>
#include <iostream>
#include <vector>
#include <string>
#include <set>

using namespace std;

int main() {
	string s,s2;
	int p,m=0,n=0,i=0;
	multiset<string> v;
	multiset<string>::iterator it;
	set<string> v2;
	set<string>::iterator it2;
	while (cin >> s) {
		p=s.find_last_of("/\\");
		while (p<s.size()) {
			s.erase(p+1,s.size()-p+1);
			if (s.size()==3) break;
			v.insert(s);
			v2.insert(s);
			s.erase(s.size()-1,1);
			p=s.find_last_of("/\\");			
		}				
	}
	for (it=v.begin(); it!=v.end(); ++it) {
		if (v.count(*it)>m) m=v.count(*it);
	}
	s=*v2.begin();
	for (it2=v2.begin(); it2!=v2.end(); ++it2) {
		s2=*it2;
		if (s2.find(s,0)<s2.size()) i++;
		else {
			if (i>n) n=i-1;
			i=0;			
			s=*it2;
			--it2;
		}
	}
	if (i>n) n=i-1;
	printf("%d %d",n,m);
	return 0;
}
