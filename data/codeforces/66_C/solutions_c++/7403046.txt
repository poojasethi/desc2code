#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
#include <map>
#include <algorithm>
using namespace std;
typedef long long ll;
const int N = 1005;
string s;
map<string, int> length, num;
map<string, int>::iterator it;
int main() {
	while(cin >> s) {
		int v = 0;
		while(true) {
			int pos = 2;
			for(int i = s.length()-1; i >= 2; i --) {
				if(s[i] == 92) {
					pos = i;
					break;
				}
			}
			if(pos == 2) break;
			s.erase(s.begin() + pos, s.end());
			bool flag = (length[s] == 0 && num[s] == 0);
			length[s] += v;
			if(flag) v++;
			num[s] ++;
		}
	}
	int l = 0, n = 0;
	for(it = length.begin(); it != length.end(); it ++) {
		if(it->second > l) l = it->second;
	}
	for(it = num.begin(); it != num.end(); it ++) {
		if(it->second > n) n = it->second;
	}
	printf("%d %d\n", l, n);
	return 0;
}
