#include<iostream>
#include<string>
#include<cstdio>
#include<set>
#include<map>
#include<vector>
#define pb push_back
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;
map<string,int> hsh;
set<int> has[10004],have[10004];
string str;
vector<int> v;
int tot;
int main(){
//	int N;
//	cin>>N;
	while(cin>>str){
		int i=3;
		v.clear();
		for(i;i<(int)str.size();i++){
			if(str[i]=='\\'){
				string t=str.substr(0,i);
				if(!hsh[t])
					hsh[t]=++tot;
				rep(i,v.size())
					has[v[i]].insert(hsh[t]);
				v.pb(hsh[t]);
			}
		}
		if(!hsh[str])
			hsh[str]=++tot;
		rep(i,v.size())
			have[v[i]].insert(hsh[str]);
	}
	int mx=0;
	rep(i,tot)
		if(has[i].size()>mx)
			mx=has[i].size();
	printf("%d ",mx);
	mx=0;
	rep(i,tot)
		if(have[i].size()>mx)
			mx=have[i].size();
	printf("%d\n",mx);
	return 0;
}
