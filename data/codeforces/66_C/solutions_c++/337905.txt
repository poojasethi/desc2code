#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <map>
#include <set>
#include <string>
#include <vector>

std::map<std::string, std::set<std::string> > folder;
std::map<std::string, int> file;
std::vector<std::string> l;
char line[100 + 1];
char *ptr;

int main() {
  while (scanf("%s", line) != EOF) {
    l.clear();
    for (char *ptr = strtok(line, "\\"); ptr; ptr = strtok(NULL, "\\"))
      l.push_back(ptr);
    std::string root = l[0] + l[1];
    std::string path = "";
    for (size_t i = 2; i < l.size() - 1; ++i) {
      path += '\\' + l[i];
      folder[root].insert(path);
    }
    ++file[root];
  }
  size_t a = 0;
  int b = 0;
  for (__typeof(folder.begin()) itr = folder.begin(); itr != folder.end(); ++itr)
    a = std::max(a, itr->second.size());
  for (__typeof(file.begin()) itr = file.begin(); itr != file.end(); ++itr)
    b = std::max(b, itr->second);
  printf("%u %d\n", a, b);
  return 0;
}
