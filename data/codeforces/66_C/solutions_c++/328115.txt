#include <cstdio>
#include <cstring>
#include <algorithm>
#include <map>
#include <string>
#include <set>
using namespace std;
map<string,int> mp;
set<string>s[100];
int ans[100];
int main()
{
	char str[300];
	int cnt=0,k;
	while(gets(str))
	{
		char tmp[300];
		int len=strlen(str);
		int i;
		for(i=0;i<len;i++)
		{
			if(str[i]=='\\'&&i!=2)
				break;
			tmp[i]=str[i];
		}
		tmp[i]='\0';
		if(mp.find(tmp)==mp.end())
			mp[tmp]=cnt++;
		ans[mp[tmp]]++;
		k=mp[tmp];
		tmp[i]='\\';
		for(i=i+1;i<len;i++)
		{
			if(str[i]=='\\')
			{
				tmp[i]='\0';
				s[k].insert(tmp);
			}
			tmp[i]=str[i];
		}
		
	}
	int mx1=0,mx2=0;
	for(int i=0;i<cnt;i++)
	{
			if(ans[i]>mx2) mx2=ans[i];
			if(s[i].size()>mx1) mx1=s[i].size();
	}	
	printf("%d %d\n",mx1,mx2);
	//system("pause");
	return 0;
}
