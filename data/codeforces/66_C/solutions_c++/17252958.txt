#include <bits/stdc++.h>
using namespace std;

#define debug(x) cout<<#x<<"="<<x<<endl
#define ll long long
#define mp makepair
#define pii pair<int,int>

map <string,int> q,p;
int fd,temp,a,b,tt=0;

int main(){
	string s;
	while(cin>>s){
		int f=0;
		while(true){
			fd=s.find_last_of('\\');
			if (fd==2) break;
			s=s.substr(0,fd);
			temp=p[s];
			q[s]+=f;
			p[s]++;
			a=max(a,q[s]);
			b=max(b,p[s]);
			if (!temp) f++;
		}
	}
	cout<<a<<" "<<b;
	return 0;
}
