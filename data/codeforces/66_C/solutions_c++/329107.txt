#include <algorithm>

#include <cstdio>
#include <cmath>
#define sz(a) int((a).size())
#define foreach(i, v) for(typeof((v).begin()) i=(v).begin(); i!=(v).end(); i++)
using namespace std;

int main()
{
	string s;
	int i, ans1=0, ans2=0;
	map<string, int> cnt;
	map< string, set<string> > sub;
    ios::sync_with_stdio(0);

    while( cin>>s )
    {
    	string t = s.substr(0, 3);
    	for(i=3; s[i]!='\\'; i++)
    		t +=s[i];
		cnt[t]++;
		for(int j=sz(s)-1; j>i; j--)
			if( s[j] == '\\' )
				sub[t].insert(s.substr(0, j));
    }
    foreach(p, sub)
    	ans1 = max(ans1, sz(p->second));
	foreach(p, cnt)
		ans2 = max(ans2, p->second);
	cout<<ans1<<" "<<ans2<<endl;

    return 0;
}
