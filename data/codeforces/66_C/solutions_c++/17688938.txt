#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <map>
using namespace std;

const int maxn=105;
string s;
map<string,int> m1,m2;
int ansfolder,ansfile;
int cnt;

int main()
{
	while(cin>>s)
	{
		cnt=0;
		while(true)
		{
			int next=s.find_last_of('\\');
			if(next==2)
				break;
			s=s.substr(0,next);
			bool flag=m2[s];
			m1[s]+=cnt;
			m2[s]++;
			ansfolder=max(ansfolder,m1[s]);
			ansfile=max(ansfile,m2[s]);
			cnt+=1-flag;
		}
	}
	cout<<ansfolder<<" "<<ansfile<<endl;
	return 0;
}
