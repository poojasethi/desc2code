using namespace std;
string s;
int a = 0,b = 0;
map <string,int> files;
map <string,int> folders;
int main()
{
	ios::sync_with_stdio(0);cin.tie(0);
	while (cin >> s)
	{
		int fold = 0;
		while(1)
		{
			int i = int(s.size())-1;
			for (;s[i]!='\\';i--);
			if (s[i-1]==':')
				break;
			s = s.substr(0,i);
			int file = files[s];
			folders[s]+=fold;
			files[s]++;
			a = max(a,folders[s]);
			b = max(b,files[s]);
			fold+=!file;
		}
	}
	cout << a << " " << b << endl;
	return 0;
}
