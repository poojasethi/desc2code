#include<map>
#include<string>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
//                                            Last Change:  2011-09-20 13:51:47
string s;
map<string,int> h,hh;
int main(){
	while(cin>>s){
		for(int v=0;;){
			size_t found=s.find_last_of('\\');
			if(found==2)
				break;
			else{
				s.erase(s.begin()+found,s.end());
				bool flag=(h[s]==0&&hh[s]==0);
				h[s]+=v;
				if(flag)++v;
			}
			hh[s]++;
		}
	}
	int a1=0,a2=0;
	for(map<string,int>::iterator i=h.begin();i!=h.end();++i)
		a1=max(a1,i->second);
	for(map<string,int>::iterator i=hh.begin();i!=hh.end();++i)
		a2=max(a2,i->second);
	cout<<a1<<" "<<a2<<endl;
	return 0;
}
