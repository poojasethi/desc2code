import sys
from collections import defaultdict
x = lambda: defaultdict(x)
d = x()
f = defaultdict(int)
for s in sys.stdin:
    if not s.strip(): continue
    t = s.split('\\')    
    t[1]+=t[0]
    f[t[1]]+=1
    eval('d'+''.join('["'+x+'"]' for x in t[1:-1]))
r = 0
for q in d.values():
    x = q.values()
    for i in x: x.extend(i.values())
    r = max(r, len(x))
print r,max(f.values())
