import sys
dd,ff,gg={},{},{}
for s in sys.stdin:
    hd=s[:s.index('\\',s.index('\\')+1)]
    ff[hd]=ff.get(hd,0)+1
    e=len(s)
    while e>len(hd):
        e=s.rfind('\\',0,e)
        t=s[:e]
        if t in gg: break
        else: dd[hd],gg[t]=dd.get(hd,0)+1,1
print max(dd.values())-1,max(ff.values())
