from sys import stdin
w = [s.replace(':\\','',1).rsplit('\\',1)[0] for s in stdin.readlines()]
f,d,p,b = {}, dict(a=0), set(), '\\'
for s in w:
    u = s.split(b, 1)[0]
    if u in f: f[u] += 1
    else: f[u] = 1
    while s.rfind(b)>-1:
        p.add(s)
        s = s.rsplit(b, 1)[0]
for s in p:
    u = s.split(b, 1)[0]
    if u in d: d[u] += 1
    else: d[u] = 1
print max(d.values()), max(f.values())
