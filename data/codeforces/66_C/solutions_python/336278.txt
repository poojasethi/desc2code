import sys

def parc(x):
  a, b = 0, 0
  for v in x.itervalues():
    c, d = parc(v)
    a += 1
    a += c
    b += d
  if b == 0:
    b += 1
  return a, b

dex = {}
for line in sys.stdin.readlines():
  line = line.replace(':\\', 'x')[:-1]
  a = dex
  for fold in line.split('\\'):
    if fold not in a:
      a[fold] = {}
    a = a[fold]
sa, sb = 0, 0
for v in dex.itervalues():
  a, b = parc(v)
  sa = max(sa, a - b)
  sb = max(sb, b)

print sa, sb
