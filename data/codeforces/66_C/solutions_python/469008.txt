#!/usr/bin/env python
# -*- coding: utf-8 -*- 

import sys, collections

def main():
    x = lambda: collections.defaultdict(x)
    folders = x()
    files = collections.defaultdict(int)
    for path in sys.stdin.readlines():
        path = path[:-1].split('\\')
        path[1] += path[0]
        files[path[1]] += 1
        eval('folders'+''.join(['["%s"]' % s for s in path[1:-1]]))
    f = 0
    for folder in folders.values():
        lst = folder.values()
        for e in lst:
            lst.extend(e.values())
        f = max(f, len(lst))
    print f, max(files.values())
    return 0

if __name__ == '__main__':
    main()
