from sys import stdin

n = int(stdin.readline())
seen = [[False for i in range(n)] for i in range(n)]
used = [False for i in range(n)]

i = 0
d = 0
v = 0
while v < n and not seen[i][d]:
  seen[i][d] = True
  if not used[i]:
    used[i] = True
    v += 1
  i = (i + d) % n
  d = (d + 1) % n
  
print "YES" if v == n else "NO"
