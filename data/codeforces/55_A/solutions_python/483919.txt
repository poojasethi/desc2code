n=input()

visit=[]
current=0
loop=True
while loop:
    

    for i in range(n):
        current=(current+i)%n
        if visit.count(current)==0:
            visit.append(current)
    if len(visit)==n:
        print'YES'
        loop =False
    elif current==0:
        print'NO'
        loop=False
