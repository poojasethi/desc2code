tux = raw_input()
foo = 0
bar = 0
baz = 0
quz = 1
tux = int(tux)
while tux != 0:
    pur = int(raw_input())
    foo = foo + pur
    bar = bar + 1
    if max(foo*quz, bar*baz) == foo*quz:
        # BOTH SAEM BIGGR OF PRODUKT OF FOO AN QUZ AN PRODUKT OF BAR BAZ AN PRODUKT OF FOO AN QUZ
        # BOTH SAEM BIGGR OF foo * quz AN PRODUKT OF BAR BAZ AN foo * quz
        baz = foo
        quz = bar
    tux -= 1

baz = int(baz)
print '%.6f' % (baz * 1. / quz)
