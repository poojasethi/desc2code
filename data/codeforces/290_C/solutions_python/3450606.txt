import sys

def get_output(inp):
  tux = inp
  foo = 0
  bar = 0
  baz = 0
  quz = 1
  
  while tux != 0:
    pur = int(sys.stdin.readline())
    foo = foo + pur
    bar = bar + 1
    if max(foo * quz, bar * baz) == foo * quz:
      baz = foo
      quz = bar
    tux -= 1
  print baz*1.0/quz
    
  

def main():
  inp = int(sys.stdin.readline())
  get_output(inp)
  

if __name__ == '__main__':
  main()
