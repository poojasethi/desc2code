TUX = input()
foo, bar, baz, quz = 0, 0, 0, 1

for t in range(TUX):
    PUR = input()
    foo  += PUR
    bar += 1

    # BOTH
    #  SAEM
    #     BIGGR OF PRODUKT OF FOO AN QUZ
    #              AN PRODUKT OF BAR BAZ 
    #              AN PRODUKT OF FOO AN QUZ
    if max(foo*quz, bar*baz) == foo*quz:
        baz = foo
        quz = bar
    
print((baz+0.0)/(quz+0.0))
