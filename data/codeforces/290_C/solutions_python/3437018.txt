from itertools import *
from functools import *
from sys import stdin, stdout

l = map(int, stdin.read().split())
tux = l[0]
l = l[1:]

foo = 0
bar = 0
baz = 0
quz = 1

for i in range(tux):
    pur = l[i]
    foo += pur
    bar += 1
    if foo * quz >= bar * baz:
        baz = foo
        quz = bar

# print(baz)
# print(quz)

print(baz * 1.0 / quz)
