import sys

dat = [int(i) for i in sys.stdin.read().strip().splitlines()]
#print sum(dat[1:])/len(dat[1:])

tux = dat[0]
dat = dat[1:]
foo, bar, baz, quz = 0, 0, 0, 1
while tux != 0:
    tux -= 1
    pur = dat[0]
    dat = dat[1:]
    foo += pur
    bar += 1
    if max(foo*quz, bar*baz) == foo*quz:
        baz = foo
        quz = bar

print float(baz)/float(quz)
