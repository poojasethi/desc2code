foo = 0
bar = 0
baz = 0
quz = 1
for i in range(tax-1,-1,-1):
    pyour = int(raw_input())
    foo += pyour
    bar += 1
    # BOTH SAEM BIGGR OF [PRODUKT OF FOO AN QUZ] AN [PRODUKT OF BAR BAZ] AN (PRODUKT OF FOO AN QUZ)
    if max(foo * quz, bar*baz) == (foo*quz):
        baz = foo
        quz = bar
