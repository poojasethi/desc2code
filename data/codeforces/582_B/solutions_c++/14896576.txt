#include <cstdio>
#include <algorithm>
#include <iostream>
#include <string.h>
using namespace std;
const int inf = 0x3f3f3f3f;
const int maxn = 110;
int dp[maxn * maxn], a[maxn * maxn], n, k;
int solve(int l) {
    fill(dp, dp + l, inf);
    for (int i = 0; i < l; ++i) {
        *upper_bound(dp, dp + l, a[i]) = a[i];
    }
    return lower_bound(dp, dp + l, inf) - dp;
}
int main() {
    scanf("%d %d", &n, &k);
    for (int i = 0; i < n; ++i) {
        scanf("%d", &a[i]);
        for (int j = 1; j <= min(n, k); ++j) a[i + n * j] = a[i];
    }
    int m = solve(n * min(n, k));
    if (k > n) m += (solve(n * (n, n + 1)) - m) * (k - n);
    printf("%d\n", m);
    return 0;
}
