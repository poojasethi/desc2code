#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<iostream>
using namespace std;


__int64 a[10005],dp[10005],have[305],maxn,sum;

int main()
{
	__int64 n,t,i,j,stop,ac;
	scanf("%I64d%I64d",&n,&t);
	for(i=1;i<=n;i++){
		scanf("%I64d",&a[i]);
		have[a[i]]++;
		dp[i]=1;
	}
	for(i=n+1;i<=n*n;i++){
		a[i]=a[i-n];
		dp[i]=1;
	}
	stop=min(n*n,n*t);
	for(i=1;i<=stop;i++){
		for(j=i+1;j<=stop;j++){
			if(a[j]>=a[i])dp[j]=max(dp[j],dp[i]+1);
		}
		sum=max(sum,dp[i]);
	}
	if(t>n){
		for(i=1;i<=300;i++)
			if(have[i]>maxn)maxn=have[i];
		sum+=maxn*(t-n);
	}
	printf("%I64d\n",sum);
	return 0;
}
