#include <iostream>
using namespace std;
int main() {
int a[10000];
int dp[10000];
int n;
int T; 
int b[10000];
int h[10000];
cin >> n >> T;
	int ma = 0;
	for (int i = 0; i < n; i++) {
		 cin >> a[i];
		 h[a[i]]++;
		 if (h[a[i]] > ma) ma = h[a[i]];
	}		 

	int len = n * min(T, n);
	for (int i = 0; i < len; i++) b[i] = a[i % n];

	for (int i = 0; i < len; i++) dp[i] = 1;
	for (int i = 0; i < len; i++) for (int j = 1; j <= n; j++) {
		if (i + j >= len) break;

		if (b[i] <= b[i + j]) dp[i + j] = max(dp[i + j], dp[i] + 1);
	}
	int ans = 0;
	for (int i = 0; i < len; i++) ans = max(ans, dp[i]);

	ans += ma * (T - min(T, n));

	cout << ans << endl;


	return 0;
}
