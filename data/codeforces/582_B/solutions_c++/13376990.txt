#include <cstdio>
#include <algorithm>

using namespace std;

const int INF=~0u>>1;

int n,t;
int a[100],num[301],dp[10001];

int main() {
	scanf("%d%d",&n,&t);
	for (int i=0;i<n;i++) {
		scanf("%d",&a[i]);
		num[a[i]]++;
	}
	for (int i=0;i<10001;i++)
		dp[i]=INF;
	dp[0]=-INF;
	int ans=0;
	for (int i=0;i<min(n,t);i++) {
		for (int j=0;j<n;j++) {
			int tmp=upper_bound(dp,dp+10001,a[j])-dp;
			dp[tmp]=a[j];
			ans=max(ans,tmp);
		}
	}
	int maxn=0;
	for (int i=0;i<=300;i++)
		if (num[i]>maxn) maxn=num[i];
	if (t>n) ans+=maxn*(t-n);
	printf("%d\n",ans);
	return 0;
}
