#include <bits/stdc++.h>
using namespace std;
int data[105], dp[305][10005], apr[305];
int main() {
	int n, t, ml, ma = 0;
	cin >> n >> t;
	ml = n * min(n, t);
	for(int i = 0; i < n; i++) {
		cin >> data[i];
		apr[data[i]]++;
	}
	for(int i = 0; i < 305; i++)
		ma = max(ma, apr[i]);
	for(int i = ml - 1; ~i; i--)
		for(int j = 0; j < 305; j++) {
			dp[j][i] = dp[j][i + 1];
			if(j <= data[i % n]) dp[j][i] = max(dp[j][i], dp[data[i % n]][i + 1] + 1);
		}
	cout << dp[0][0] + (t - ml / n) * ma;
}
