#include<cstdio>
#include<algorithm>
using namespace std;
int a[10001];
int dp[10001];
int num[301];
int n,m,t,mx,ans;

int main()
{
	int i,j,k;
	scanf("%d%d",&n,&t);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		num[a[i]]++;
	}
	for(i=1;i<=300;i++)
	{
		mx=max(num[i],mx);
	}
	for(i=1;i<=min(n,t)-1;i++)
	{
		for(j=1;j<=n;j++)
			a[i*n+j]=a[j];
	}
	m=n*min(n,t);
	dp[0]=0;a[0]=0;
	for(i=1;i<=m;i++)
	{
		for(j=0;j<i;j++)
		{
			if(a[i]>=a[j])
				dp[i]=max(dp[i],dp[j]+1);
		}
	}
	for(i=1;i<=m;i++)
	{
		ans=max(ans,dp[i]);
	}
	ans+=mx*(t-min(n,t));
	printf("%d",ans);	
	return 0;
}
