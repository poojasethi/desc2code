#include <bits/stdc++.h>

#define FOR(i,n) for(int (i)=0;(i)<(n);++(i))
#define FORI(i,n) for(int (i)=1;(i)<=(n);++(i))
#define REP(i,a,b) for(int (i)=(a);(i)<=(b);++i)
#define REPD(i,a,b) for(int (i)=(a); (i)>=(b);--i)
#define X first
#define Y second
using namespace std;
typedef long long ll;
// --------------------------------------------
int main(void)
{
    int c, h;
    ll t1, t2, t3, t4, ans, p, t, x, y;
    t1 = t2 = t3 = t4 = -4e9-7;
    ans = 4e9+7;
    scanf("%d%d",&x,&y);
    scanf("%d",&c);
    FOR(_,c) {
        scanf("%I64d%I64d",&x,&y);
        t1 = max(t1,x+y);
        t2 = max(t2,-x-y);
        t3 = max(t3,x-y);
        t4 = max(t4,-x+y);
    }
    scanf("%d",&h);
    REP(i,1,h) {
        t = -4e8-7;
        scanf("%I64d%I64d",&x,&y);
        t = max(t,t1 -x-y);
        t = max(t,t2 +x+y);
        t = max(t,t3 -x+y);
        t = max(t,t4 +x-y);
        if (t < ans) {
            ans = t;
            p = i;
        }
    }
    cout << ans << '\n' << p;
    return 0;
}
