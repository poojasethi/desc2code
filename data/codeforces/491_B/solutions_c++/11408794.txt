#include <stdio.h>
#include <algorithm>

using namespace std;

const int inf = -1u/2;

int n, m;
int mx1, mx2, mx3, mx4;
int c, h;

int main() {
  scanf("%d%d", &n, &m);
  scanf("%d", &c);
  mx1 = mx2 = mx3 = mx4 = -inf;
  for(int i = 0;i < c;i++) {
    int x, y;
    scanf("%d%d", &x, &y);
    mx1 = max(mx1, x+y);
    mx2 = max(mx2, x-y);
    mx3 = max(mx3, -x+y);
    mx4 = max(mx4, -x-y);
  }

  scanf("%d", &h);
  int mn = inf, idx;
  for(int i = 1;i <= h;i++) {
    int x, y;
    scanf("%d%d", &x, &y);
    int ans = -inf;
    ans = max(ans, mx1-(x+y));
    ans = max(ans, mx2-(x-y));
    ans = max(ans, mx3-(-x+y));
    ans = max(ans, mx4-(-x-y));
    if(ans < mn) {
      mn = ans;
      idx = i;
    }
  }
  printf("%d\n%d\n", mn, idx);
  return 0;
}
