#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;

int main(){

    int n;
    cin >> n >> n; //无用数据

    int i, a, b, c, d;

    a = b = c = d = INT_MAX;

    cin >> n;
    while(n--){
        int x, y;
        cin >> x >> y;
        //找到四个角落的房子
        a = min(a, x+y);
        b = min(b, -x-y);
        c = min(c, x-y);
        d = min(d, -x+y);
    }

    cin >> n;
    int ans = INT_MAX, idx = -1;
    for(i=1;i<=n;i++){

        int x, y, dist;
        cin>>x>>y;
        //相对应的对比餐馆得出最远距离
        dist = abs(a + (-x-y));
        dist = max(dist, abs((x+y) + b));
        dist = max(dist, abs(c + (-x+y)));
        dist = max(dist, abs(d + (x-y)));

        if(dist < ans){
            ans = dist; idx = i;
        }
    }

    cout << ans << endl << idx << endl;

    return 0;
}
