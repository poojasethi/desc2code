#include <cstdio>
#include <algorithm>
using namespace std;

const int N = 1e5 + 10;
int n,m,sx,sy,hx,hy,ans,ansn;

int main() {
    int a,b;
    scanf("%d%d",&a,&b);
    scanf("%d",&n);
    hx = 1e9 + 1;
    hy = 1e9 + 1;
    sx = -hx;
    sy = -hy;
    for (int i = 1;i <= n;i ++) {
        int x,y;
        scanf("%d%d",&x,&y);
        int xx = x + y,yy = x - y;
        sx = max(xx,sx);
        hx = min(xx,hx);
        sy = max(sy,yy);
        hy = min(hy,yy);
    }
    scanf("%d",&m);
    ans = 2e9 + 1;
    for (int i = 1;i <= m;i ++) {
        int x,y;
        scanf("%d%d",&x,&y);
        int xx = x + y,yy = x - y;
        int cur = max(sx - xx,xx - hx);
        cur = max(sy - yy,cur);
        cur = max(cur,yy - hy);
        if (cur < ans) {
            ans = cur;
            ansn = i;
        }
    }
    printf("%d\n%d",ans,ansn);
}
