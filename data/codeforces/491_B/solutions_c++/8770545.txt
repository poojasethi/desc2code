#include <cstdio>
#include <iostream>

using namespace std;

int main()
{
	int n, m, c, h, i, x, y;
	int Mp, mp, Ms, ms, temp;
	int Md, Mdi;
	scanf("%d%d%d", &n, &m, &c);
	Mp=0;
	mp=m+n;
	Ms=0-m;
	ms=n;
	Md=m+n;
	for(i=0;i!=c;i++)
	{
		scanf("%d%d", &x, &y);
		if(x+y>Mp) Mp=x+y;
		if(x+y<mp) mp=x+y;
		if(x-y>Ms) Ms=x-y;
		if(x-y<ms) ms=x-y;
	}
	scanf("%d", &h);
	for(i=0;i!=h;i++)
	{
		temp=0;
		scanf("%d%d", &x, &y);
		if((Mp-(x+y))>temp)
			temp=Mp-(x+y);
		if(((x+y)-mp)>temp)
			temp=(x+y)-mp;
		if((Ms-(x-y))>temp)
			temp=Ms-(x-y);
		if(((x-y)-ms)>temp)
			temp=(x-y)-ms;
		if(temp<Md)
		{
			Md=temp;
			Mdi=i+1;
		}
	}
	printf("%d\n%d",Md,Mdi);
}
