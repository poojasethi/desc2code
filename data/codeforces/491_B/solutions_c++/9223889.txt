#include <bits/stdc++.h>

using namespace std;

#define INF 1LL<<50
long long v[4] = {INF,-INF,INF,-INF};


long long maxi(long long a , long long b , long long c , long long d){
  return max(max(a,b),max(c,d));
}

int main(){
  #ifndef ONLINE_JUDGE
  freopen("in.txt","r",stdin);
  #endif

  long long n,m,c,h,x,y;
  cin >> n >> m >> c;
  for(int i=0; i<c; ++i){
    cin >> x >> y;
    v[0] = min(v[0],x+y);
    v[1] = max(v[1],x+y);
    v[2] = min(v[2],x-y);
    v[3] = max(v[3],x-y);
  }

  cin >> h;
  long long a = INF, j = -1;
  for(int i=0;i<h;++i){
    cin >> x >> y;
    long long cur = maxi(abs(v[0] - (x+y)), abs(v[1] - (x+y)),
                         abs(v[2] - (x-y)), abs(v[3] - (x-y)));
    if(cur < a){
      a = cur;
      j = i;
    }
  }

  cout << a << endl << j+1 << endl;

  return 0;
}
