#include <stdio.h>
#include <algorithm>
using namespace std;

int main(void) {
    int n, m;
    int c, h;
    int x, y;
    int d1, d2, d3, d4;
    int idx;
    int aux;
    int best;


    scanf(" %d %d", &n, &m);
    d1 = d2 = d3 = d4 = -(n + m + 1);
    best = n + m + 1;
    idx = 0;

    scanf(" %d", &c);
    for (int i = 0; i < c; i++) {
        scanf(" %d %d", &x, &y);
        d1 = max(d1, x + y);
        d2 = max(d2, x - y);
        d3 = max(d3, -x + y);
        d4 = max(d4, -x - y);
    }
    scanf(" %d", &h);
    for (int i = 0; i < h; i++) {
        scanf(" %d %d", &x, &y);
        aux = 0;
        aux = max(aux, d1 - x - y);
        aux = max(aux, d2 - x + y);
        aux = max(aux, d3 + x - y);
        aux = max(aux, d4 + x + y);
        if (aux < best) {
            idx = i + 1;
            best = aux;
        }
    }
    printf("%d\n", best);
    printf("%d\n", idx);

    return 0;
}
