#include<iostream>
#include<cstring>
#include<climits>
using namespace std;
long long minp , minq , maxp , maxq , n  , x , y , ans=LONG_LONG_MAX , best;
int main(){
    cin>>n>>n;
    minp=minq=LONG_LONG_MAX;
    maxp=maxq=LONG_LONG_MIN;
    cin>>n;
    while(n--){
        cin>>x>>y;
        maxq=max(maxq,x+y);
        minq=min(minq,x+y);
        maxp=max(maxp,x-y);
        minp=min(minp,x-y);
    }
    cin>>n;
    for(int j=1;j<=n;j++){
        cin>>x>>y;
        long long q=x+y , p=x-y , cur=LONG_LONG_MIN;
        cur=max(q-minq , maxq-q);
        cur=max(cur , max(p-minp , maxp - p));
        if(cur<ans){
            ans=cur;
            best=j;
        }
    }
    cout<<ans<<endl<<best<<endl;
}
