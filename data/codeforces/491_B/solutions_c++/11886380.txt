#include <bits/stdc++.h>
#define foreach(it,v) for(__typeof(v.begin()) it = v.begin(); it != v.end(); ++it)
using namespace std;
typedef long long ll;
typedef pair<int,int> Point;
#define x first
#define y second
const int maxn = 1e5 + 10;
int a[maxn],b[maxn];
int main()
{	
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	int n,m;
	while(cin>>n>>m) {
		int C;cin>>C;
		for(int i = 0; i < C; i++) {
			int c,d;cin>>c>>d;
			a[i] = c + d;
			b[i] = c - d;
		}
		sort(a,a+C);
		sort(b,b+C);
		int H; cin>>H;
		Point ans(2e9+7,0);
		for(int i = 1; i <= H; i++) {
			int c,d;cin>>c>>d;
			int t1 = max(abs(c+d-a[0]),abs(c+d-a[C-1]));
			int t2 = max(abs(c-d-b[0]),abs(c-d-b[C-1]));
			t1 = max(t1,t2);
			if(ans.x > t1) {
				ans.x = t1;
				ans.y = i;
			}
		}
		cout<<ans.x<<endl<<ans.y<<endl;
	}
    return 0;
}
