

#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<queue>
#include<cmath>
using namespace std;
typedef long long ll;
const ll inf =(ll) 1<<60;
ll a=-inf,b=-inf,c=-inf,d=-inf;
int n,m;
int main()
{
    cin>>n>>m;
    int cnt;
    cin>>cnt;
    for(int i=1;i<=cnt;i++)
    {
        ll x,y;
        scanf("%I64d%I64d",&x,&y);
        a=max(a,x+y);
        b=max(b,-x+y);
        c=max(c,-x-y);
        d=max(d,x-y);
    }
    cin>>cnt;
    ll ans=inf,p;
   // printf("a=%d b=%d c=%d d=%d \n",a,b,c,d);
    for(int i=1;i<=cnt;i++)
    {
        int x,y;
        scanf("%d%d",&x,&y);
        ll tmp=max(max(a-x-y,c+x+y),max(x-y+b,-x+y+d));
        if(ans>tmp) ans=tmp,p=i;
    }
    cout<<ans<<endl<<p<<endl;


}
