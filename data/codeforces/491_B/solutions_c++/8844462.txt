#include<bits/stdc++.h>
using namespace std ;

vector<int> x , y ;
int a , b ;

int main() {

    int n , m , c , h , u , v ;
    cin >> n >> m ;
    cin >> c ;
    x.resize(c) ;
    y.resize(c) ;    
    for( int i = 0 ; i < c ; i++ ) {
        cin >> u >> v ;
        x[i] = u+v ;
        y[i] = u-v ;
    }
    sort(x.begin(),x.end()) ;
    sort(y.begin(),y.end()) ;
    
    cin >> h ;
    int pos = 1 , min_dist = (1<<31)-1 ;
    for( int i = 0 ; i < h ; i++ ) {
        cin >> u >> v ;
        a = u+v ;
        b = u-v ;
        int dist = max( max( abs(x[0]-a) , abs(x[c-1]-a) ) , max( abs(y[0]-b) , abs(y[c-1]-b) ) ) ;
        if( min_dist > dist ) {
            min_dist = dist ; pos = i+1 ;
        }
    }
    cout << min_dist << endl ;
    cout << pos << endl ;
    

    
    return 0 ;
}
