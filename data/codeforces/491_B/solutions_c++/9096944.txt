#include <iostream>
#include <algorithm>

using namespace std;

const long int maxn = 1000 * 1000 * 1000 + 1000;

int main ()
{
    long long int temp = -1, n, m, c, h, x, y;
    cin >> n >> m >> c;
    long long int br = -maxn, bc = maxn, pr = -maxn, pc = maxn, fasele = m + n;
    for (int i = 0; i < c; i++)
    {
        cin >> x >> y;
        pr = max (pr, x + y);
        bc = min (bc, x + y);
        pc = min (pc, x - y);
        br = max (br, x - y);
    }
    cin >> h;
    for (int i = 0; i < h; i++)
    {
        cin >> x >> y;
        fasele = min (fasele, max (max (abs (x + y - pr), abs(x + y - bc)), max ( abs (x - y - pc), abs (x - y - br))));
        if (fasele == max (max (abs (x + y - pr), abs(x + y - bc)), max ( abs (x - y - pc), abs (x - y - br))))
        temp = i;
    }
    cout << fasele << endl;
    cout << temp + 1;
}
