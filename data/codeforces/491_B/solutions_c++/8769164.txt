#include <iostream>
using namespace std;

int N, M;
int C, H;
int indx;
long long max1, max2, max3, max4, best;

int main() {

    cin.sync_with_stdio(false);
    
    max1 = -(1LL << 60LL);
    max2 = -(1LL << 60LL);
    max3 = -(1LL << 60LL);
    max4 = -(1LL << 60LL);
    best = +(1LL << 60LL);
    
    cin >> N >> M;
    cin >> C;
    for (int i = 0; i < C; i++) {
        long long x, y;
        cin >> x >> y;
        max1 = max(max1, +x+y);
        max2 = max(max2, -x+y);
        max3 = max(max3, +x-y);
        max4 = max(max4, -x-y);
    }
    
    cin >> H;
    for (int i = 0; i < H; i++) {
        long long x, y;
        cin >> x >> y;
        long long dst = 0;
        dst = max(dst, max1 + (-x-y));
        dst = max(dst, max2 + (+x-y));
        dst = max(dst, max3 + (-x+y));
        dst = max(dst, max4 + (+x+y));
        if (dst < best) {
            best = dst;
            indx = i+1;
        }
    }
    
    cout << best << endl;
    cout << indx << endl;
    
    return 0;
}
