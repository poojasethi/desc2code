const long long A=100000000000000LL,N=228228;

long long k,x[2],ot,o[4],otv[2]={A,-1},y[2],i,j,n,m;

int main(){
	cin>>n>>m>>k;
	o[0]=o[1]=o[2]=o[3]=A;
	for(i=0;i<k;i++){
		cin>>x[0]>>y[0],x[1]=n-x[0],y[1]=m-y[0];
		o[0]=min(o[0],x[0]+y[0]);
		o[1]=min(o[1],x[0]+y[1]);
		o[2]=min(o[2],x[1]+y[0]);
		o[3]=min(o[3],x[1]+y[1]);
	}
	cin>>k;
	for(i=0;i<k;i++){
		cin>>x[0]>>y[0],x[1]=n-x[0],y[1]=m-y[0];
		ot=x[0]+y[0]-o[0];
		ot=max(ot,x[0]+y[1]-o[1]);
		ot=max(ot,x[1]+y[0]-o[2]);
		ot=max(ot,x[1]+y[1]-o[3]);
		if(ot<otv[0])otv[0]=ot,otv[1]=i+1;
	}
	cout<<otv[0]<<"\n"<<otv[1];
}
