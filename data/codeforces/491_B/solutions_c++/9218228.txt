#include<iostream>
#include<algorithm>
typedef long long LL;

const int N = (1LL << 31) - 1;
using namespace std;
int main(){
	int n , m;
	cin>>m>>n;
	int c;
	cin>>c;
//	vector<int , int > a;
	int xmin = N,ymin =N,xmax = -N,ymax = -N;
	for(int i = 0 ; i < c;i++){
		int x, y ;
		cin>>x>>y;
		 xmin = min(xmin,x-y);
		 xmax = max(xmax,x-y);
		 ymin = min(ymin ,x+y);
		 ymax = max(ymax , x+y);
	}
	int h;
	cin>>h;
	
	int dist = N;
	int rest;
	for (int i = 1; i <= h; i++){
		int x, y;
		cin >> x >> y;
		int curr = -N;
		curr = max(curr, abs(x - y - xmin));
		curr = max(curr, abs(x + y - ymin));
		curr = max(curr, abs(x - y - xmax));
		curr = max(curr, abs(x + y - ymax));
		if (dist > curr){
			dist = curr;
			rest = i;
		}
	}
	cout << dist << "\n" << rest;
	return 0;
}
