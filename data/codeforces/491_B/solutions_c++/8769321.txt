#include <iostream>
#include <cstdlib>
#define INF 3000000000ll

using namespace std;

int n,m,c,h;
long long xa=INF,xb=-INF,ya=INF,yb=-INF;
long long ans=INF,idx;

int main() {
    cin >> n >> m >> c;
    for (int i = 0; i < c; i++) {
        long long x,y;
        cin >> x >> y;
        long long xx = x+y;
        long long yy = x-y;
        xa = min(xa,xx);
        xb = max(xb,xx);
        ya = min(ya,yy);
        yb = max(yb,yy);
    }
    cin >> h;
    for (int i = 0; i < h; i++) {
        long long x,y;
        cin >> x >> y;
        long long xx = x+y;
        long long yy = x-y;
        long long d = abs(xx-xa);
        d = max(d,abs(xx-xb));
        d = max(d,abs(yy-yb));
        d = max(d,abs(yy-ya));
        if (d < ans) {
            ans = d;
            idx = i+1;
        }
    }
    cout << ans << '\n' << idx << '\n';
}
