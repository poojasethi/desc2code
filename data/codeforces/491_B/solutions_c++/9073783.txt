#include <bits\stdc++.h>

using namespace std;

vector <int> w, q;
int a, b;

int main(void)
{
    int n, m, c, h, u, v;
    cin >> n >> m ;
    cin >> c ;
    w.resize(c) ;
    q.resize(c) ;
    for( int i = 0 ; i < c ; ++i) {
        cin >> u >> v ;
        w[i] = u + v ;
        q[i] = u - v ;
    }
    sort(w.begin(),w.end()) ;
    sort(q.begin(),q.end()) ;
    cin >> h ;
    int pos(1), min_dist((1<<31)-1);
    for( int i = 0 ; i < h ; ++i) {
        cin >> u >> v ;
        a = u + v ;
        b = u - v ;
        int dist = max( max( abs(w[0] - a), abs(w[c - 1] - a) ), max( abs(q[0]-b), abs(q[c-1]-b))) ;
        if( min_dist > dist ) {
            min_dist = dist ; pos = i + 1 ;
        }
    }
    cout << min_dist << '\n' ;
    cout << pos << '\n' ;
    return 0 ;
}
