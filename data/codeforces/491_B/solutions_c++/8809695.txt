#include <iostream>
#include <cstdio>
#include <map>
#include <cstring>
#include <climits>
#include <vector>
#include <stack>
#include <algorithm>
#include <queue>
#include <string>

using namespace std;

int mn[2], mx[2];

int main()
{
    int n, x, y, h;
    cin >> h >> h;
    mn[0] = mn[1] = INT_MAX;
    mx[1] = mx[0] = INT_MIN;
    int r = INT_MAX, j;
    cin >> n;
    for(int i=0; i < n; i++)
    {
        cin >> x >> y;
        mn[0] = min(mn[0], x-y);
        mn[1] = min(mn[1], x+y);
        mx[0] = max(mx[0], x+y);
        mx[1] = max(mx[1], x-y);
    }
    cin >> n;
    for(int i=0; i < n; i++)
    {
        cin >> x >> y;
        int d1 = x-y, d2 = x+y;
        int w = max(mx[1] - d1, max(mx[0] - d2, max(d2 - mn[1], d1 - mn[0])));
        if(w < r)
            r = w, j = i+1;
    }
    cout << r << "\n" << j << "\n";
    return 0;
}
