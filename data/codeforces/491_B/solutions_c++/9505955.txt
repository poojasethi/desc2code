#include<bits/stdc++.h>
using namespace std;
   #include <bits/stdc++.h>

using namespace std;

int N, M, C, H;
int xmin = 2000000000, xmax = 0, ymin = 100000000, ymax = -1000000000;
int best = 2000000000, val = -1;

int main() {
  scanf ("%d %d", &N, &M);
  scanf ("%d", &C);
  for (int i = 0; i < C; i++) {
    int x, y;
    scanf ("%d %d", &x, &y);
    int X = x + y;
    int Y = x - y;
    x = X, y = Y;
    xmin = min (x, xmin);
    ymin = min (y, ymin);
    xmax = max (x, xmax);
    ymax = max (y, ymax);
  }
  //  printf ("%d %d %d %d\n", xmin, ymin, xmax, ymax);
  scanf ("%d", &H);
  for (int i = 1; i <= H; i++) {
    int x, y;
    scanf ("%d %d", &x, &y);
    int X = x + y;
    int Y = x - y;
    x = X, y = Y;
    int a = max (abs (x - xmin), abs (xmax - x));
    int b = max (abs(y - ymin), abs (ymax - y));
    if (best > max (a, b)) {
      best = max (a, b);
      val = i;
    }
  }
  printf ("%d\n%d\n", best, val);
}
