#include<vector>
#include<cstdio>
#include<algorithm>
using namespace std;

long long mn,m,n,i,j,a,b,sum,k,c,mnx,mny,mxx,mxy,mx;

int main(){
    mn=2000000002;
    mxx=-2000000002;
    mxy=-2000000002;
    mnx=2000000002;
    mny=2000000002;
    scanf("%lld %lld",&m,&n);
    scanf("%lld",&m);
    for(i=1;i<=m;i++){
        scanf("%lld %lld",&a,&b);
        if(a+b>mxx)mxx=a+b;
        if(a+b<mnx)mnx=a+b;
        if(a-b>mxy)mxy=a-b;
        if(a-b<mny)mny=a-b;
    }
    scanf("%lld",&n);
    for(i=0;i<n;i++){
        scanf("%lld %lld",&a,&b);
        mx=0;
        if(a+b-mnx>mx)mx=a+b-mnx;
        if(mxx-a-b>mx)mx=mxx-a-b;
        if(mxy-a+b>mx)mx=mxy-a+b;
        if(a-b-mny>mx)mx=a-b-mny;
        if(mx<mn){
            mn=mx;
            c=i+1;
        }
    }
    printf("%lld\n%lld\n",mn,c);
}
