#include<stdio.h>
#include<cstring>
#include<iostream>
#include<cstdlib>

using namespace std;

int main(){
    int N,M;
    cin >> N >> M;
    int c;
    cin >> c;
    int maxxy = -N-M-1;
    int minxy = N+M+1;
    int maxxny = -N-M-1;
    int minxny = N+M+1;
    for(int i = 0; i < c; i++){
        int x,y;
        cin >> x >> y;
        maxxy = max(x+y, maxxy);
        minxy = min(x+y, minxy);
        maxxny = max(x-y, maxxny);
        minxny = min(x-y, minxny);
    }
    int h;
    cin >> h;
    int mind = N+M+1;
    int minh = -1;
    for(int i = 0; i < h; i++){
        int x,y;
        cin >> x >> y;
        
        int dist = max(max(maxxy - x-y, maxxny - x + y), max(x - y - minxny,  x + y - minxy));
        if(dist < mind){
            mind = dist;
            minh = i;
        }   
    }
    cout << mind << endl;
    cout << minh+1 << endl;
}
