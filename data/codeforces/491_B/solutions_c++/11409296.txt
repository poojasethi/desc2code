#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
int main()
{
    int n,m,c;
    long long x,y,cx,cy,cd;
    long long lx=1000000000LL,hx=-10000000000LL,ly=lx,hy=hx;
    long long bsm=20000000000LL;
    int bsi;
    scanf("%d%d",&n,&m);
    scanf("%d",&c);
    for(int i=1;i<=c;i++)
    {
        scanf("%lld%lld",&x,&y);
        lx=min(lx,x+y);
        hx=max(hx,x+y);
        ly=min(ly,x-y);
        hy=max(hy,x-y);
    }
    int h;
    scanf("%d",&h);
    for(int i=1;i<=h;i++)
    {
        scanf("%lld%lld",&x,&y);
        cx=x+y;
        cy=x-y;
        cd=max(max(cx-lx,hx-cx),max(cy-ly,hy-cy));
        if(cd<bsm)
        {
            bsm=cd;
            bsi=i;
        }
    }
    printf("%lld\n%d",bsm,bsi);

}
