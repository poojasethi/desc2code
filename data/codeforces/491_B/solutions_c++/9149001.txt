#include <iostream>

using namespace std;

const long long INF=(4e9)+10;

int main()
{
    long long ans=INF,maxh=-INF,maxc=-INF,minh=INF,minc=INF;
    long long n,m,c,h,p;
    cin>>n>>m;
    cin>>c;
    for(int i=0;i<c;i++)
    {
        long x,y;
        cin>>x>>y;
        if(x+y>maxh) maxh=x+y;
        if(x+y<minh) minh=x+y;
        if(x-y>maxc) maxc=x-y;
        if(x-y<minc) minc=x-y;
    }
    cin>>h;
    for(int i=1;i<=h;i++)
    {
        long x,y;
        cin>>x>>y;
        long long tmph=max(x+y-minh,maxh-x-y);
        long long tmpc=max(x-y-minc,maxc-x+y);
        long long tmp=max(tmpc,tmph);
        if(tmp<ans)
        {
            ans=tmp;
            p=i;
        }
    }
    cout<<ans<<endl;
    cout<<p<<endl;
    return 0;
}
