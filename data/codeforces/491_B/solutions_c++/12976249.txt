#include <bits/stdc++.h>
using namespace std;
long long int INF = 10000000000000000ll;
int main()
{
    // freopen("input.txt","r",stdin);
    ios_base::sync_with_stdio(0); 
    int n,m;
    cin>>n>>m;
    long long int minx = INF, maxx = -INF, miny = INF, maxy = -INF;
    int c,x,y;
    cin>>c;
    for (int i = 0; i < c; ++i)
    {
        cin>>x>>y;
        long long int a = x+y, b = x-y;
        minx = min(minx,a);
        maxx = max(maxx,a);
        miny = min(miny,b);
        maxy = max(maxy,b);
    }
    int h;
    cin>>h;
    long long int mindist = INF, ans = -1;
    for (int i = 0; i < h; ++i)
    {
        cin>>x>>y;
        long long int a = x+y, b = x-y;
        long long int dist = max(abs(maxx-a),max(abs(minx-a),max(abs(miny-b),abs(maxy-b))));
        if(dist < mindist)
        {
            mindist = dist;
            ans = i+1;
        }
    }
    cout<<mindist<<"\n"<<ans<<"\n";
    return 0 ;
}
