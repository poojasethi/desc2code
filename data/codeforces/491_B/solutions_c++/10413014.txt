#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
    long long n,m,c,h,x,y;
    while(cin>>n>>m)
    {
    	cin>>c;
        vector<int> axis1,axis2;
        for(int i=1;i<=c;i++)
        {
        	cin>>x>>y;
            //将点(x,y)分别投影到1 3 ， 2 4 象限
            axis1.push_back(x+y);
            axis2.push_back(x-y);
        }
        sort(axis1.begin(),axis1.end());
        sort(axis2.begin(),axis2.end());
        long long res=-1,id=0;
        cin>>h;
        for(int i=1;i<=h;i++)
        {
        	cin>>x>>y;
        	int dis1=max(abs(x+y-axis1[0]),abs(x+y-axis1[c-1]));
        	int dis2=max(abs(x-y-axis2[0]),abs(x-y-axis2[c-1]));
        	int tmpmax=max(dis1,dis2);
        	if(res>tmpmax||res==-1){
        		res=tmpmax;
        		id=i;
        	}
        }
        cout<<res<<endl;
        cout<<id<<endl;

    }
}
