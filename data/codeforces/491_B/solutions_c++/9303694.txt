#pragma comment(linker, "/STACK:268435456")
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <set>
#include <algorithm>

using namespace std;

long long mp1=-1e12, mp2=1e12, mm1=-1e12, mm2=1e12;
long long ans=1e12, ansi;

int main(){
    int n,m,c,h;
    long long x,y;
    scanf("%d %d", &n, &m);
    scanf("%d", &c);
    for(int i=0;i<c;i++){
        scanf("%I64d %I64d", &x, &y);
        mp1=max(mp1,x+y);
        mp2=min(mp2,x+y);
        mm1=max(mm1,x-y);
        mm2=min(mm2,x-y);
    }
    //cout << mp1 << " " << mp2 << " " << mm1 << " " << mm2 << endl;
    scanf("%d", &h);
    for(int i=0;i<h;i++){
        scanf("%I64d %I64d", &x, &y);
        long long temp=max(max(abs(x+y-mp1), abs(x+y-mp2)), max(abs(x-y-mm1), abs(x-y-mm2)));
        if(temp<ans) ans=temp, ansi=i;
        //printf("%d %I64d %I64d\n", i, ans, ansi);
    }
    printf("%I64d\n%I64d\n", ans, ansi+1);
    //system("pause");
    return 0;
}
