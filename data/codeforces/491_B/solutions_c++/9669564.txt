#include<iostream>
using namespace std;

const int inf=2e9;
int main() {
	ios_base::sync_with_stdio(false); cin.tie(NULL);
	int n,m,c,x,y,xx,yy; cin>>n>>m>>c;
	int p=inf,q=inf,r=inf,s=inf;
	while(c--) {
		cin>>x>>y, xx=n-x,yy=m-y;
		p=min(p,x+y), q=min(q,x+yy), r=min(r,xx+y), s=min(s,xx+yy);
	}

	int h, key=-1, val=inf; cin>>h;
	for (int i=1; i<=h; i++) {
		cin>>x>>y, xx=n-x,yy=m-y;
		int cur=x+y-p;
		cur=max(cur,x+yy-q);
		cur=max(cur,xx+y-r);
		cur=max(cur,xx+yy-s);
		if(val>cur) val=cur, key=i;
	}
	cout << val << endl << key << endl;
	return 0;
}
