#include <cstdio>
#include <climits>
#define abs(A) (((A)>0)?(A):(-(A)))
int main(){
	int X, Y, n, m, xmin = INT_MAX, xmax = INT_MIN, ymax = INT_MIN, ymin = INT_MAX, ans = INT_MAX, index;
	scanf("%d%d%d", &X, &Y, &n);
	for (int i = 1; i <= n; i ++ ){
		int t1, t2;
		scanf("%d%d", &t1, &t2);
		if (t1 + t2 < xmin) xmin = t1 + t2;
		if (t1 + t2 > xmax) xmax = t1 + t2;
		if (t1 - t2 < ymin) ymin = t1 - t2;
		if (t1 - t2 > ymax) ymax = t1 - t2;
	}
	scanf("%d", &m);
	for (int i = 1; i <= m; i ++ ){
		int t1, t2;
		scanf("%d%d", &t1, &t2);
		int x = t1 + t2, y = t1 - t2, now = 0;
		if (abs(x - xmin) > now) now = abs(x - xmin);
		if (abs(x - xmax) > now) now = abs(x - xmax);
		if (abs(y - ymin) > now) now = abs(y - ymin);
		if (abs(y - ymax) > now) now = abs(y - ymax);
		if (now < ans){
			ans = now;
			index = i;
		}
	}
	printf("%d\n%d\n", ans, index);
}
