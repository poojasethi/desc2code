#include <iostream>
#include <cstdio>
using namespace std;

const int N = 111111;

int n, m;
int c, h;
int pp, pm, mp, mm;

int main() {
    cin >> n >> m;
    cin >> c;
    pp = pm = mp = mm = -2000000010;
    for (int i = 1; i <= c; ++i) {
        int x, y;
        scanf("%d %d", &x, &y);
        pp = max(pp, x + y);
        pm = max(pm, x - y);
        mp = max(mp, -x + y);
        mm = max(mm, -x - y);
    }
    cin >> h;
    int ans1;
    int ans2;
    ans1 = 2000000010;
    for (int i = 1; i <= h; ++i) {
        int x, y;
        scanf("%d %d", &x, &y);
        int t = 0;
        t = max(t, -x - y + pp);
        t = max(t, -x + y + pm);
        t = max(t, x - y + mp);
        t = max(t, x + y + mm);
        if (t < ans1) {
            ans1 = t;
            ans2 = i;
        }
    }
    cout << ans1 << endl << ans2 << endl;
    return 0;
}
