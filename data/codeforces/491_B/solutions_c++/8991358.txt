#include <cstdio>
#include <algorithm>

using namespace std;

const int inf = 2000000010;
int n, m, c, h, d = inf;
int mi0 = inf, mi1 = inf, ma0 = -inf, ma1 = -inf;

int main()
{
    scanf("%d %d %d", &n, &m ,&c);
    while (c--)
    {
        int a, b, d0, d1;
        scanf("%d %d", &a, &b);
        d0 = a + b; d1 = a - b;
        mi0 = min(mi0, d0);
        ma0 = max(ma0, d0);
        mi1 = min(mi1, d1);
        ma1 = max(ma1, d1);
    }
    scanf("%d", &h);
    for (int i = 1; i <= h; i++)
    {
        int a, b, d0, d1;
        scanf("%d %d", &a, &b);
        d0 = a + b; d1 = a - b;
        int val = max(max(abs(d0 - mi0), abs(d1 - mi1)), max(abs(d0 - ma0), abs(d1 - ma1)));
        if (val < d) d = val, c = i;
    }
    printf("%d\n%d\n", d, c);
    return 0;
}
