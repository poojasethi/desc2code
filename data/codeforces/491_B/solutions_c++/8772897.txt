#include<bits/stdc++.h>
using namespace std;

#define rep(i,a,b) for(i=a;i<=b;i++)
#define rev(i,b,a) for(i=b;i>=a;i--)
#define pb push_back
#define mp make_pair
#define pi 2*acos(0.0)

const int M=2e9;

int main()
{
 int n,m;
 scanf("%d %d",&n,&m);
 int sum_1=M,sum_2=-M,sub_1=-M,sub_2=M;
 int i,j;
 scanf("%d",&n);
 rep(i,1,n)
  {
    int x,y;
    scanf("%d %d",&x,&y);
     int p=x+y;
     int q=x-y;
     if(p<sum_1)sum_1=p;
     if(p>sum_2)sum_2=p;
     if(q>sub_1)sub_1=q;
     if(q<sub_2)sub_2=q;
  }
  scanf("%d",&m);
  int res=M+1,pos;
  rep(i,1,m)
  {
      int x,y;
      scanf("%d %d",&x,&y);
      int temp=-M-1;
      int p=x+y;
      int q=x-y;
      if(abs(p-sum_1)>temp)temp=abs(p-sum_1);
     if(abs(p-sum_2)>temp)temp=abs(p-sum_2);
     if(abs(q-sub_1)>temp)temp=abs(q-sub_1);
     if(abs(q-sub_2)>temp)temp=abs(q-sub_2);
     if(temp<=res)
     {
         res=temp;pos=i;
     }
  }
  cout<<res<<endl<<pos<<endl;
  return 0;
}
