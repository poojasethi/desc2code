#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;

int main(){
    
    int dummy;
    cin>>dummy>>dummy;
    
    int i, a, b, c, d;
    
    a = b = c = d = INT_MAX;
    
    int n;
    cin>>n;
    while(n--){
        int x, y;
        cin>>x>>y;
        
        a = min(a, x+y);
        b = min(b, -x-y);
        c = min(c, x-y);
        d = min(d, -x+y);
    }
    
    cin>>n;
    int ans = INT_MAX, idx = -1;
    for(i=1;i<=n;i++){
        
        int x, y, dist;
        cin>>x>>y;
        
        dist = abs(a + (-x-y));
        dist = max(dist, abs((x+y) + b));
        dist = max(dist, abs(c + (-x+y)));
        dist = max(dist, abs(d + (x-y)));
        
        if(dist < ans){
            ans = dist; idx = i;
        }
    }
    
    cout<<ans<<endl<<idx;
    
    return 0;
}
