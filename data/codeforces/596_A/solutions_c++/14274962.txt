#include <iostream>
using namespace std;

int main(){
    int n,x[4],y[4],a;
    cin>>n;
    for (int i=0;i<n;i++){
        cin>>x[i]>>y[i];
    }
    for(int i=0;i<n-1;i++){
        for(int j=i+1;j<n;j++){
            if(x[i]!=x[j] && y[i]!=y[j]){a=(x[j]-x[i])*(y[j]-y[i]);if(a<0)a*=-1;cout<<a;return 0;}
        }
    }
    cout<<-1;
    return 0;
}
