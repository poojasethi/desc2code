int x[5],y[5];

{
	int i,j,k,n,m;

	{
		for(i=0;i<n;i++)
		scanf("%d%d",&x[i],&y[i]);
		sort(x,x+n);
		sort(y,y+n);
		if(x[0]!=x[n-1]&&y[0]!=y[n-1])
		{
			m=(x[n-1]-x[0])*(y[n-1]-y[0]);
			printf("%d\n",m);
		}

	}
