#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for (int i = 0; i < n; i ++)

int x[4], y[4];

int main ()
{
	int n;
	cin >> n;
	rep (i, n)
		cin >> x[i] >> y[i];	
	int dx = 0, dy = 0;
	rep (i, n)
		rep (j, n)
		{
			dx = max (abs(x[i]-x[j]), dx);
			dy = max (abs(y[i]-y[j]), dy);
		}
	cout << (dx*dy ? dx*dy : -1) << endl;
}
