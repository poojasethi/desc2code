#include<bits/stdc++.h>
using namespace std;
long long i,j,xm=-20005,xn=10005,ym=-20005,n,x,y,ny=100005;
int main()
{
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>x>>y;
        xm=max(x,xm);
        xn=min(x,xn);
        ym=max(y,ym);
        ny=min(y,ny);
    }
    if((xm-xn)&&(ym-ny))
        cout<<(xm-xn)*(ym-ny);
    else
        cout<<-1;

}
