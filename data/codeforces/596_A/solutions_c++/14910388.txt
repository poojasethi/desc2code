#include <cstdio>
#include <cstdlib>
int x[5],y[5];
int main()
{
    int n,ans=-1;
    scanf("%d",&n);
    for(int i=0; i<n; i++)
        scanf("%d%d",&x[i],&y[i]);
    for(int i=0; i<n; i++)
        for(int j=i+1; j<n; j++)
            if(x[i]!=x[j]&&y[i]!=y[j])
                ans=abs(x[i]-x[j])*abs(y[i]-y[j]);
    printf("%d",ans);
    return 0;
}
