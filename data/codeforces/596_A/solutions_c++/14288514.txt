#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,x[4],y[4];
    scanf("%d",&n);
    for(int i=0; i<n; ++i)
        scanf("%d%d",&x[i],&y[i]);
    sort(x,x+n);
    sort(y,y+n);

    if(n == 1 || y[n-1] == y[0] || x[n-1] == x[0]) {
        printf("-1");
    } else {
        printf("%d",(y[n-1] - y[0]) * (x[n-1] - x[0]));
    }
    return 0;
}
