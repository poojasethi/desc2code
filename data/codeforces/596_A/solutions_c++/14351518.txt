#include<bits/stdc++.h>
using namespace std;
int main()
{
    int x[5],y[5];
    int n,area;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d %d",&x[i],&y[i]);
    sort(x,x+n);
    sort(y,y+n);
    area=(abs(x[n-1]-x[0])*abs(y[n-1]-y[0]));
    if(area) cout<<area<<endl;
    else cout<<"-1"<<endl;
    return 0;
}
