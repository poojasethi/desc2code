#include <bits/stdc++.h>
#define ll long long

using namespace std;

int main(){
	int n;
	cin>>n;
	long long x[n],y[n];
	for(int i=0;i<n;i++)cin>>x[i]>>y[i];
	if(n==1){
		cout<<-1;
		return 0;
	}
	sort(x,x+n);
	sort(y,y+n);
	long long a=x[n-1]-x[0];
	long long b=y[n-1]-y[0];
	if(a*b>0)
		cout<<a*b;
	else
		cout<<-1;
	return 0;
}
