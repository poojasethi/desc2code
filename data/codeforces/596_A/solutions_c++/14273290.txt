#include<bits/stdc++.h>
using namespace std;

long long n,m,a;
long long x[4],y[4];

int main(){
    scanf("%I64d",&n);


    for(int i=0;i<n;i++)
        scanf("%I64d %I64d",&x[i],&y[i]);

    for(int i=0;i<n-1;i++){
        for(int j=i+1;j<n;j++)
            a=max(a,abs(x[i]-x[j])*abs(y[i]-y[j]));
    }

    if(a>0)
        printf("%I64d\n",a);
    else
        printf("-1\n");

}
