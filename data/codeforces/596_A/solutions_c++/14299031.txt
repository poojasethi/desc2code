#include <iostream>
#include <cmath>
using namespace std;


int main()
{
    int n, x1, y1, x, y, w = 0, h= 0;
    cin >> n;
    cin >> x1 >> y1;
    for (int i = 1; i < n; i++) {
        cin >> x >> y;
        if (x != x1) w = abs(x1 - x);
        if (y != y1) h = abs(y1 - y);
    }
    if (w*h==0)
        cout << -1;
    else
        cout << w*h;
}
