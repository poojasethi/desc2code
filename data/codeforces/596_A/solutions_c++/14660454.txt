#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{  int x[5],y[5];int i,n;
    scanf("%d",&n);

    for (i=1;i<=n;i++)
        scanf("%d%d",&x[i],&y[i]);
       sort(x+1,x+n+1);
       sort(y+1,y+n+1);
       if ((x[n]-x[1])*(y[n]-y[1])!=0) printf("%d",(x[n]-x[1])*(y[n]-y[1]));
       else printf("-1");

}
