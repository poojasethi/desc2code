#include "iostream"

using namespace std;

int main(){
	int n, res=0;
	cin >> n;
	int x[n], y[n];
	for (int i=0; i<n; i++) {
		cin >> x[i] >> y[i];
	}
	for (int i=0;i<n;i++){
		for (int j = i+1;j<n;j++){
			res=(x[j]-x[i])*(y[j]-y[i]);
			if (res<0) res=-res;
			if (res>0) {
				i=n;
				j=n;
			}
		}
	} 

	if (res == 0) cout << -1 << endl;
	else cout << res << endl;
}
