#include <bits/stdc++.h>

using namespace std;

int n, x[4], y[4], ans;

int main() {
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> x[i] >> y[i];
    for (int i = 0; i < n; i++)
        for (int j = 0; j < i; j++)
            ans = max (ans , abs(x[i] - x[j]) * abs(y[i] - y[j]));
    cout << ( ans ? ans : -1);
    return 0;
}
