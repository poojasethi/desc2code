#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int n, ans;
int x[6], y[6];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
		scanf("%d%d",&x[i], &y[i]);
	sort(x+1, x+n+1);
	sort(y+1, y+n+1);
	ans=(x[n]-x[1])*(y[n]-y[1]);
	if (ans==0)
		printf("-1\n");
	else
		printf("%d\n",ans);
	return 0;
}
