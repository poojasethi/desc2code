#include <bits/stdc++.h>
using namespace std;

int value(set<int>& s)
{
	return (*s.rbegin() - *s.begin());
}

int main()
{
	int n, x, y; cin>>n;
	set<int> a,b;
	for (int i = 0;  i < n; i++)	
	{
		cin>>x>>y;
		a.insert(x); b.insert(y);
	}
	if (a.size() != 2 || b.size() != 2)
		cout<<"-1\n";
	else
	{
		printf("%d\n", value(a)*value(b));
	}
	return (0);
}
