#include<bits/stdc++.h>
using namespace std;

set<int> a,b;

int main()
{
    int n,x,y;
    cin >> n;
    for(int i=0;i<n;i++) cin >> x >> y,a.insert(x),b.insert(y);
    if(a.size()!=2 || b.size()!=2) cout << "-1";
    else cout << (*a.rbegin()-*a.begin())*(*b.rbegin()-*b.begin());


    return 0;
}
