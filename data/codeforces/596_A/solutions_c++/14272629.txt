#include <iostream>
#include <algorithm>

using namespace std;

int main() {
	
	int x[4], y[4];
	int n;
	
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> x[i] >> y[i];
	}
	sort(x, x + n);
	sort(y, y + n);
	
	if (x[0] < x[n - 1] && y[0] < y[n - 1]) {
		int s = (x[n - 1] - x[0]) * (y[n - 1] - y[0]);
		cout << s << "\n";
	}
	else {
		cout << "-1\n";
	}
	
	return 0;
}
