#include <iostream>
using namespace std;
int n,x1=-1000000000,y1=-100000000,x2=100000000000,y2=10000000000;
int main() {
    cin>>n;
    for(int i=0;i<n;i++){
        int a,b;cin>>a>>b;
        x1=max(x1,a);
        x2=min(x2,a);
        y1=max(y1,b);
        y2=min(y2,b);
    }
    int ans=(x1-x2)*(y1-y2);
    if(ans)cout<<ans;
    else cout<<-1;
}
