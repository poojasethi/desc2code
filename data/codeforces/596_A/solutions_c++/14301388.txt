#include <iostream>
#include <cstdlib>
using namespace std;

int main(){
	int n;
	long long area = -1;
	cin >> n;
	int x[n],y[n];
	for(int i = 0; i < n; ++i){
		cin >> x[i] >> y[i];
	}	
	for(int i = 1; i < n; ++i){
		for(int j = 0; j < i; ++j){
			if(x[i] != x[j] && y[i] != y[j]) area = abs((x[i]-x[j])*(y[i]-y[j]));
		}
	}
	cout << area;
	return 0;
}
