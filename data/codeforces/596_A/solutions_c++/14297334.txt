#include<bits/stdc++.h>
using namespace std; 
int x[4], y[4];
int main() {
    int n,ans=-1;cin>>n;
    for(int i=0;i<n;i++)cin>>x[i]>>y[i];
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            if(x[i]!=x[j]&&y[i]!=y[j])
                ans=abs(x[i]-x[j])*abs(y[i]-y[j]);
    cout<<ans<<endl;
}
