#include<bits/stdc++.h>
#define Rep(i,n) for(int i = 1;i <= n;i ++)
using namespace std;
int n,x,y,mxx = -1e9,mxy = -1e9,mnx = 1e9,mny = 1e9;
int main ()
{
	scanf("%d",&n);
	Rep(i,n)scanf("%d%d",&x,&y),mxx = max(mxx,x),mxy = max(mxy,y),mnx = min(x,mnx),mny = min(mny,y);
	int ans = (mnx - mxx) *(mny - mxy);
	if(!ans)ans = -1;
	printf("%d\n",ans);
	return 0;
}
