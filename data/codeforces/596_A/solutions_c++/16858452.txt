#include<stdio.h>
#include<algorithm>
using namespace std;
main()
{
  int n,ax,ay,bx,by,x,y;
  scanf("%d",&n);
  ax=ay=1000000000;
  bx=by=-ax;
  for(int i=0;i<n;i++)
  {
    scanf("%d%d",&x,&y);
    ax=min(ax,x);
    ay=min(ay,y);
    bx=max(bx,x);
    by=max(by,y);
  }
   x=bx-ax;
   y=by-ay;
   if(x*y)
   printf("%d",x*y);
   else
   puts("-1");
}
