#include <iostream>

using namespace std;
int main(){
		int n,x,y,maxx=-10000,minx=100000,maxy=-100000,miny=100000;
		cin>>n;
		for(int i=1;i<=n;i++){
			cin>>x>>y;
			minx=min(x,minx);
			miny=min(y,miny);
			maxx=max(x,maxx);
			maxy=max(y,maxy);
		}
		if(maxx-minx==0 || maxy-miny==0)cout<<-1;
		else cout<<(maxx-minx)*(maxy-miny);

}
