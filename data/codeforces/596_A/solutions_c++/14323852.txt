#include <cstdio>
#include <set>
using namespace std;
set<int> X, Y;
int main() {
    int n, i, x, y;
    scanf("%d", &n);
    for(i=1;i<=n;i++) {
        scanf("%d%d", &x, &y);
        X.insert(x); Y.insert(y);
    }
    if(X.size()==2&&Y.size()==2) {
        printf("%d", (*X.rbegin()-*X.begin())*(*Y.rbegin()-*Y.begin()));
    } else printf("-1");
    return 0;
}
