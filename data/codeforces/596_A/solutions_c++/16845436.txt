#include <bits/stdc++.h>

using namespace std;

int main(){
	set<int> xs;
	set<int> ys;
	set<int>::iterator it;
	int i,n,x,y;
	scanf("%i",&n);
	for(i = 0; i < n; i++){
		scanf("%d %d",&x,&y);
		xs.insert(x);
		ys.insert(y);
	}
	if(xs.size() == 2 && ys.size() == 2){
		printf("%d\n",(*xs.rbegin()-*xs.begin())*(*ys.rbegin()-*ys.begin()));
	}
	else printf("-1\n");
	return 0;
}
