#include <iostream>
using namespace std;
int f(int a)
{
	if(a<0)
		a=-a;
	return a;
}
int main()
{
    int n,i,j;
    cin>>n;
    int x[4],y[4];
    for(i=0;i<n;i++)
        cin>>x[i]>>y[i];
    for(i=0;i<n;i++)
		for(j=0;j<n;j++)

			if(x[i]!=x[j]&&y[i]!=y[j])
			{
				
				cout<<f(x[i]-x[j])*f(y[i]-y[j]);
				return 0;

			}
	cout<<-1;
    return 0;
}
