#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int n;
	cin>>n;
	int  x,y,minx=100000,miny=100000,maxx=-100000,maxy=-1000000;
	for(int i=0;i<n;i++){
		cin>>x>>y;
		minx=min(x,minx);
		miny=min(y,miny);
		maxx=max(x,maxx);
		maxy=max(y,maxy);
	}
	if((maxx-minx)*(maxy-miny)==0){
		cout<<-1;
	}
	else cout<<(maxx-minx)*(maxy-miny);
return 0;
}
