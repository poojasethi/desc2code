n = int(raw_input())
maxx = -2000
minx = 2000
maxy = -2000
miny = 2000
for i in range(n):
    a, b = map(int, raw_input().split())
    if a < minx:
        minx = a
    if a > maxx:
        maxx = a
    if b < miny:
        miny = b
    if b > maxy:
        maxy = b

s = (maxx-minx)*(maxy-miny)
if s == 0:
    print -1
else:
    print s
