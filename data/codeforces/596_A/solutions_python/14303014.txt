'''input
1
1 1
'''
x1, y1 = 10000, 10000
x2, y2 = -10000, -10000

A = [map(int, raw_input().split()) for _ in xrange(input())]
for i in xrange(len(A)):
	x1 = min(x1, A[i][0])
	x2 = max(x2, A[i][0])
	y1 = min(y1, A[i][1])
	y2 = max(y2, A[i][1])

if not x2 - x1 or not y2 - y1:
	print -1
else:
	print (x2 - x1) * (y2 - y1)
