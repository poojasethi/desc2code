n=input()
if(n==1):
	print(-1)
else:
	lastx=-10000
	lasty=-10000
	sidea=10000
	sideb=10000
	ans=0
	for i in range(n):
		x,y=raw_input().split()
		x=int(x)
		y=int(y)
		lastx=max(x,lastx)
		lasty=max(y,lasty)
		sidea=min(x,sidea)
		sideb=min(y,sideb)
	ans=(lastx-sidea)*(lasty-sideb)
	if ans==0:
		print(-1)
	else: 
		print(ans)
