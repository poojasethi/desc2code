n = input()
res = []
for i in range(n):
	k = map(int, raw_input().strip().split(' '))
	res.append(k)
xma = -1111
yma = -1111
ymi = 1111
xmi = 1111
for x in res:
	xma = max(xma, x[0])
	xmi = min(xmi, x[0])
	yma = max(yma, x[1])
	ymi = min(ymi, x[1])
ans = (xma - xmi) * (yma - ymi)
if ans > 0:
	print ans
else:
	print -1
