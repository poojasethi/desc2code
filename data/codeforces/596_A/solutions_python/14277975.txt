n = int(raw_input())
pt = map(int, raw_input().split(' '))
minx = maxx = pt[0]
miny = maxy = pt[1]

if n > 1:
	for i in range(n - 1):
		pt = map(int, raw_input().split())
		if pt[0] < minx: minx = pt[0]
		if pt[0] > maxx: maxx = pt[0]
		if pt[1] < miny: miny = pt[1]
		if pt[1] > maxy: maxy = pt[1]

area = (maxx - minx) * (maxy - miny)
if area > 0:
	print area
else:
	print -1
