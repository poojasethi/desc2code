n = int(raw_input())
points = [tuple(map(int,raw_input().split(" "))) for _ in xrange(n)]
def find_corners(points):
    for i in range(len(points)-1):
        for j in range(i+1, len(points)):
            if points[i][0] != points[j][0] and points[i][1] != points[j][1]:
                return (i,j)
    return (None, None)

a,b = find_corners(points)
if a is None:
    print "-1"
else:
    print abs(points[a][0]-points[b][0]) * abs(points[a][1]-points[b][1])
