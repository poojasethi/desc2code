a = input()
xmax = -10000000000
ymax = -10000000000
xmin = 100000000000
ymin = 100000000000
for i in range(0,a) :
	x,y = raw_input().split()
	x = int(x)
	y = int(y)
	xmax = max(x,xmax)
	xmin = min(xmin,x)
	ymax = max(y,ymax)
	ymin = min(y,ymin)

ar = (ymax-ymin)*(xmax-xmin)
if ar==0 :
	print -1
else :
	print ar
