n = int(raw_input())
c = []
for _ in range(n):
    x, y = map(int, raw_input().split())
    c.append((x, y))
d = [c[0]] + [(c[i][0], c[i][1]) for i in range(1, n) if c[0][0] != c[i][0] and c[0][1] != c[i][1]]
if n > 2 and len(d) < 2:
    c[0], c[1] = c[1], c[0]
    d = [c[0]] + [(c[i][0], c[i][1]) for i in range(1, n) if c[0][0] != c[i][0] and c[0][1] != c[i][1]]
if len(d) < 2:
    print -1
else:
    print abs(d[0][0] - d[1][0]) * abs(d[0][1] - d[1][1])
