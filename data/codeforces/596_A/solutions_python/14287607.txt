n = input()
l = []
sq = 0
for i in xrange(n):
	l.append(tuple(map(int, raw_input().split())))
if (n == 1):
	print -1
	exit()
if (n == 2):
	if (l[0][0] == l[1][0] or l[0][1] == l[1][1]):
		print -1
		exit()
	else:
		sq = abs((l[0][1] - l[1][1]) * (l[0][0] - l[1][0]))
else:
	for i in xrange(n):
		for j in xrange(n):
			if l[j][0] != l[i][0] and l[j][1] != l[i][1]:
				sq = abs((l[j][1] - l[i][1]) * (l[j][0] - l[i][0]))
				if sq != 0:
					print sq
					exit()
print sq
