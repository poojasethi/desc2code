n = int(raw_input())
xs = []
ys = []
res = -1

if (n > 1):
	for x in xrange(0,n):
		inp = [int(i) for i in raw_input().split()]
		xs.append(inp[0])
		ys.append(inp[1])

	xs.sort()
	ys.sort()

	a = 0
	b = 0

	for x in xrange(1,n):
		a1 = xs[x] - xs[x-1]
		if (a1 != 0):
			a = a1
		b1 = ys[x] - ys[x-1]
		if (b1 != 0):
			b = b1

	if (a != 0 and b != 0):
		res = abs(a * b)

print res
