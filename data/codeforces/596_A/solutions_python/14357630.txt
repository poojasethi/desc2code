ip,rp = input,raw_input
n = ip()
pt = []
for i in range(n):
    x,y = map(int,rp().split())
    pt.append((x,y))
if(n==1):
    print -1
elif(n==2):
    if(pt[0][0]!=pt[1][0] and pt[0][1]!=pt[1][1]):
        print abs(pt[1][0]-pt[0][0])*abs(pt[1][1]-pt[0][1])
    else:
        print -1
else:
    ans = 0
    for i in range(n):
        for j in range(i+1,n):
            ans = max(ans,abs(pt[j][0]-pt[i][0])*abs(pt[j][1]-pt[i][1]))
    print ans
