
n = int(raw_input())

points= []
for i in range(n):
        points.append([int(j) for j in raw_input().split(" ")])


side1 = [p[0] for p in  points]
l1 = max(side1) - min(side1)
side2 = [p[1] for p in  points]
l2 = max(side2) - min(side2)

if l2*l1 == 0:
        print -1
else:
        print l2*l1
