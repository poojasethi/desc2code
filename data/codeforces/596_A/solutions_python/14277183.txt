n=int(raw_input())
a=[]
b=[]
c=[]
p=False
q=False
for i in range(n):
    a.append(map(int, raw_input().split()))
for i in range(n):
    b.append(a[i][0])
    c.append(a[i][1])
d=b[0]
f=c[0]
for i in range(n):
    if b[i]!=d:
        d-=b[i]
        p=True
        break
for i in range(n):
    if c[i]!=f:
        f-=c[i]
        q=True
        break
if p==True and q==True:
    print abs(d*f)
else:
    print -1
