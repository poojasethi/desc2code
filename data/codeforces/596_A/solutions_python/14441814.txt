n = int(raw_input())
if n ==1:
    print -1
elif n ==2:
    x1,y1 = map(int ,raw_input().split())
    x2,y2 = map(int ,raw_input().split())

    if x1 == x2 or y1 == y2:
        print -1
    else:
        print abs((x2-x1) * (y2-y1))

else:
    x= [0]*n
    y = [0]*n
    for i in range(0,n):
        x[i],y[i] = map(int , raw_input().split())
    x = sorted(x)
    y = sorted(y)
    print abs((x[0]- x[-1]) * (y[0]-y[-1]))
