n = int(raw_input())


xs = []
ys = []
for i in range(n):
    s = raw_input().split(" ")
    x = int(s[0])
    y = int(s[1])
    xs.append(x)
    ys.append(y)

ans = -1
   
for i in range(n):
    x1 = xs[i]
    y1 = ys[i]
    for j in range(i, n):
        x2 = xs[j]
        y2 = ys[j]
        if x1 != x2 and y1 != y2:
            ans = abs(x2-x1)*abs(y2-y1)
            break
print ans
