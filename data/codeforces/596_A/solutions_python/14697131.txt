n=input()
li=[map(int,raw_input().split()) for i in xrange(n)]
flag=0
for i in xrange(n-1):
    for j in xrange(i+1,n):
        if(li[i][0]-li[j][0]!=0 and li[i][1]-li[j][1]!=0):
            print abs(li[i][0]-li[j][0])*abs(li[i][1]-li[j][1])
            flag=1
            break
    if(flag==1):
        break
if(flag==0):
    print '-1'
