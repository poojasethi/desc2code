n = input()

ans = ['']

for i in xrange(n):
    s = raw_input()

    if s == 'pwd':
        print '/'.join(ans) + '/';
    else:
        s = s[3:]
        ary = s.split('/')

        if ary[0] == '':
            ans = []

        for ts in ary:
            if ts == '..':
                ans.pop()
            else:
                ans.append(ts)
