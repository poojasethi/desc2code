def ltos(ls):
	res='/'
	for x in ls:
		res+=x+'/'
	return res

count=int(raw_input())

path=[]

for i in range(count):
	s=raw_input()
	if s=='pwd':
		print ltos(path)
		continue 
	s=s[3:]
	ls=s.split('/')
	if s[0]=='/':
		path=[]
	for x in ls:
		if x=='..':
			path.pop()
		else:
			if x!='':
				path.append(x)
