d = [""]
for x in xrange(input()):
    cmd = raw_input()
    if cmd == "pwd":
        print "/".join(d) + '/'
    elif cmd[:2] == "cd":
        c = cmd[3:].split("/")
        for x in c:
            if x == "":
                d = [""]
            else:
                d=d[:-1] if x=='..' else d+[x]
