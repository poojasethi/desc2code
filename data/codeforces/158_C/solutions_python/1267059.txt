#!/ usr/bin/env python
path = []
t = (int)(raw_input())
while t:
	t -= 1
	cmd = raw_input().split();
	if cmd[0] == "pwd":
		print "/" + "".join(path)
	else:
		if cmd[1].startswith("/"):
			path = []
		for d in cmd[1].split("/"):
			if d == "..":
				path.pop()
			elif len(d) > 0:
				path.append(d + "/")
