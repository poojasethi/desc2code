n = int(raw_input())

path = ['']

while n > 0:
	text = raw_input()
	if text == "pwd":
		print "/".join(path) + "/"
	else:
		text = text.split(' ')[1].split('/')
		if text[0] == '':
			path = ['']
		for i in [k for k in text if k != '']:
			if i == "..":
				path.pop()
			else:
				path.append(i)
	n -= 1
