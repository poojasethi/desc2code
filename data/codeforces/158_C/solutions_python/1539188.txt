n = input()
stk = ['']
for i in range(n):
  s = raw_input()
  if s == 'pwd':
    print '/'.join(stk) + '/'
  else:
    cur = s.split()[1].split('/')
    if cur[0] == '':
      stk = ['']
    for s in cur:
      if s == '..':
        stk.pop()
      elif s == '':
        pass
      else:
        stk.append(s)
