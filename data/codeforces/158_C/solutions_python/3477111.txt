
def main():
	path = []
	n = input()
	for i in xrange(n):
		cmd = raw_input()
		if (cmd=='pwd'):
			print '/' + '/'.join(path+[''])
		else:
			cmd = cmd.split()
			if cmd[1][0]=='/':
				path = []
			cmd = cmd[1].split('/')
			for q in cmd:
				if q:
					if q!='..':
						path.append(q)
					else:
						path.pop()


main()
