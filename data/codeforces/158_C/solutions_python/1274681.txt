n = int(raw_input())

path = ['', '']
for i in range(n):
	line = raw_input().split()
	if line[0] == 'pwd':
		print '/'.join(path)
	else:
		path.pop()
		if line[1][0] == '/':
			path = ['']
			D = line[1][1:].split('/')
		else:
			D = line[1].split('/')
		for d in D:
			if d == '..':
				path.pop()
			elif d != '':
				path.append(d)
		path.append('')
