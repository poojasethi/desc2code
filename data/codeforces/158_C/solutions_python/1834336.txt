n = input()
cur = ['']

for i in xrange(n):
    cmd = raw_input()
    if cmd == 'pwd':
        print "/".join(cur) + "/"
    else:
        moves = cmd.split()[1].split('/')
        for m in moves:
            if not m:
                cur = ['']
            elif m == '..':
                cur.pop()
            else:
                cur.append(m)
