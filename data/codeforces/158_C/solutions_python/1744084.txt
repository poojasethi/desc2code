pwd = ['/']
for i in range(input()):
    cmd = raw_input().split()
    if len(cmd) > 1:
        for x in cmd[1].split('/'):
            if not x: pwd = ['/']
            elif x=='..': pwd.pop()
            elif x=='.': pass
            else: pwd.append(x)
    else:
        print '/'.join(pwd)[1:]+'/'
