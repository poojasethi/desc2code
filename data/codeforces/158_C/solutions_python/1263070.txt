import posixpath

N = int(raw_input())

curpath = '/'
for i in xrange(N):
    cmd = raw_input().split()
    if cmd[0] == "pwd":
        if posixpath.abspath(curpath)[-1] != '/':
            print posixpath.abspath(curpath) + '/'
        else:
            print posixpath.abspath(curpath)
    else:
        curpath = posixpath.join(curpath, cmd[1])
