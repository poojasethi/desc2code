d = []
n = input()
for i in xrange(n):
	inp = raw_input()
	if(inp[:3] == "pwd"):
		if not d:
			print "/"
		else:
			print "/" + "/".join(d) + "/"
	else:
		inp = inp[3:]
		com = inp.split("/")
		for c in com:
			if(c == ""):
				d = []
			elif(c == ".."):
				if not d:
					continue
				else:
					d.pop()
			else:
				d.append(c)
