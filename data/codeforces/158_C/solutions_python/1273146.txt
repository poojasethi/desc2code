#!/usr/bin/python2
pwd = ['']
n = input()
for i in xrange(n):
  cmd = raw_input().split()
  if (cmd[0] == 'pwd'):
    print '/'.join(pwd) + '/'
  else:
    npwd = cmd[1].split('/')
    if (npwd[0] == ''):
      pwd = ['']
    for i in npwd:
      if (i == '..'):
        pwd.pop()
      elif (i != ''):
        pwd.append(i)
