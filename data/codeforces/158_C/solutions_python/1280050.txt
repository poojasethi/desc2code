n = input()
ans = []
for i in range(n):
	cmd = raw_input().split()
	if cmd[0] == 'pwd':
		if len(ans) == 0:
			print '/'
		else:
			print '/' + '/'.join(ans) + '/'
	else:
		if cmd[1][0] == '/':
			ans = []
		path = cmd[1].split('/');
		for i in path:
			if(i == ""):
				pass
			elif i == '..':
				del ans[-1]
			else:
				ans.append(i)
