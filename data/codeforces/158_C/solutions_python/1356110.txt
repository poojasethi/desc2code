n=input()
s=[]
for t in xrange(n):
    cur=raw_input().split()
    if cur[0]=="cd":
        st=cur[1]
        if st[0]=='/': s=[]
        L=st.replace('/',' ').split()
        for ch in L:
            if ch=='..': del s[-1]
            else: s.append(ch)
    else:
        ans='/'
        for ch in s:
            ans+=ch+'/'
        print ans
