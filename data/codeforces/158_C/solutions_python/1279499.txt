from sys import stdin

comms = stdin.readlines()[1:]

cur_dir=[]
for com in comms:
  com = com.split()
  if com[0]=="pwd":
    print "/".join([""] + cur_dir + [""])
  else:
    dirs = com[1].split("/")
    if dirs[0]=="":
      cur_dir = []
      del dirs[0]
    for d in dirs:
      if d=="..":
        del cur_dir[-1]
      else:
        cur_dir.append(d)
