#include <iostream>
#include <cstring>

using namespace std;

void processd(string &s){
	int i;
	while( (i = s.find("..")) != string::npos){
		int j = s.rfind("/", i-2);
		s.erase(j, i-j+2);
	}
}

int main(){
	int n;
	cin >> n;
	
	string cur("/");
	string in, d;
	
	for(int i = 0; i < n; i++){
			cin >> in;
			if( in.compare("pwd") == 0 )
				cout << cur << endl;
			else if( in.compare("cd" ) == 0){
				cin >> d;
				if(d[0] == '/')
					cur = d + "/";
				else
					cur = cur + d + "/";
				
				processd(cur);
			}
	}
}
