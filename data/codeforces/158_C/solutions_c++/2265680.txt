#include <iostream>
#include <string>
using namespace std;

int main(){
    int n,i,j,k;
    string a,s="";
    cin>>n;
    for(i=0;i<n;i++){
        cin>>a;
        if(a[0]=='p'){
            cout<<s+'/'<<endl;
        }else{
            cin>>a;
            if(a[0]=='/')s=a;
            else s+='/'+a;
        }
        k=s.find('.');
        while(k!=-1){
            k++;
            j=k;
            while(s[j--]!='/');
            while(j>=0 && s[j]!='/')j--;
            s.erase(j,k-j+1);
            k=s.find('.');
        }
    }
}
