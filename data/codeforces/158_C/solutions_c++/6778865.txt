#include <stdio.h>
#include <string.h>
char s[200000];
int n,l,i;
char str[500];

int main()
{
	scanf("%d",&n);
	s[0]='/';
	l=1;
	while(n--)
	{
		scanf("%s",&str);
		if(str[0]=='p')
		{
			s[l]='\0';
			printf("%s\n",s);
			continue;
		}
		memset(str,0,sizeof(str));
		scanf("%s",&str);
		if(str[0]=='/')
			l='\0';
		for(i=0;str[i];i++)
		{
			if(str[i]=='.')
			{
				l--;
				while(s[l-1]!='/')
					l--;
				i+=2;
				continue;
			}
			s[l]=str[i];
			l++;
		}
		if(s[l-1]!='/')
		{
			s[l]='/';
			l++;
		}
	}
	return 0;
}
