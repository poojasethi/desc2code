#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
#include <cstdio>
#include <cstring>
#include <map>
#include <set>
using namespace std;
string ans;
int main()
{
	int n,k;
	cin>>n;
	while(n--)
	{
	string s;
	cin>>s;
	if(s=="pwd")
	cout<<ans<<"/"<<endl;
	else {
	cin>>s;
	if(s[0]=='/')
	ans=s;
	else ans+="/"+s;
	}
	while((k=ans.find('.'))!=-1)
	{
	int m=++k;
	while(ans[m--]!='/');
	while(m>=0&&ans[m]!='/')
	m--;
	ans.erase(m,k-m+1);
	}
	}
	return 0;
}
