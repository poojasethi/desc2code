#include<iostream>
#include<cstdio>
#include<vector>
#include<string>
using namespace std;

int main(){
   int n; scanf("%d",&n);
   vector<string> v;
   string s;
   for(int i=0; i<n; i++){
      cin>>s;
	  if (s=="pwd"){
	     putchar('/');
		 for(int i=0; i<v.size(); i++) cout<<v[i]<<"/";
		 puts("");
	  }
	  else{
	     cin>>s; s+='/';
		 string tmp="";
		 for(int i=0; i<s.size(); i++){
		    if (s[i]=='/'){
			   if (tmp=="") v.clear();
			   else if (tmp=="..") v.pop_back();
			   else v.push_back(tmp);
			   tmp="";
			}
			else tmp+=s[i];
		 }
	  }
   }
}
