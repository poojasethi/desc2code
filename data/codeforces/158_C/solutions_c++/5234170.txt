#include<iostream>
#include<string>
using namespace std;

int main (){
int n;
int k;
string s,p="/",t;
cin >> n;
	for(int i = 1; i <= n; i ++){
		cin >> s;
		if(s == "pwd"){
			cout<<p<<endl;
		}
		if(s == "cd"){
			cin>>s;
			s += '/';
			for(int j = 0; j < s.size(); j ++){
				t += s[j];
				if(s[j] == '/'){
					if(t == "/")
						p = t;
					else if(t == "../"){
						for(k = p.size() - 1; p[k - 1] != '/'; k --);
						p.resize(k);
					}
					else
						p += t;
					t = "";
				}
			}	
		}
	}
	return 0; 
}
