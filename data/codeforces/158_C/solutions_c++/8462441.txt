#include<iostream>
using namespace std;
int main()
{
int n,i,j,k;
string s,temp="/",t;
cin>>n;
for(i=1;i<=n;i++)
{
cin>>s;
if(s=="cd")
{
cin>>s;
s+='/';
for(j=0;j<s.length();j++)
{
t+=s[j];
if(s[j]=='/')
{
if(t=="/") 
temp=t;
else if(t=="../")
{
for(k=temp.length()-1;temp[k-1]!='/';k--);
temp.resize(k);
}
else temp+=t;
t="";
}
}
}
else
{
cout<<temp<<endl;
}
}
return 0;
}
