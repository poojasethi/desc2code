#include<iostream>
#include<string.h>

using namespace std;

int n,i,j,ndir=0;
char cmd[5],par[205],dir[5000][205],*p;

int main() {
  cin>>n;
  for(i=0;i<n;++i) {
    cin>>cmd;
    if(strcmp(cmd,"cd")==0) {
      cin>>par;
      if(par[0]=='/') ndir=0;
      p=strtok(par,"/");
      while(p) {
        if(strcmp(p,"..")==0) --ndir;
        else strcpy(dir[ndir++],p);
        p=strtok(NULL,"/");
      }
    }
    else {
      cout<<'/';
      for(j=0;j<ndir;++j)
       cout<<dir[j]<<'/';
      cout<<endl;
    }
  }
  return 0;
}
