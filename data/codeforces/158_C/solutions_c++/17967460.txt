int main(){
	int n;
	string p="/",s,t;
	cin>>n;
	getchar();
	while(n--){
		cin>>s;
		if(s=="pwd"){
			cout<<p<<endl;
		}
		else{
			cin>>s;
			s+='/';
			for(int i=0;i<s.length();i++){
				t+=s[i];
				if(s[i]=='/'){
					if(t=="/") p=t;
					else if(t=="../"){
						int k;
						for(k=p.length()-1;p[k-1]!='/';k--);
						p.resize(k);
					}
					else p+=t;
					t="";
				}
			}
		}
	}
	return 0;
