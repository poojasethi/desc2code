#include<iostream>
#include<vector>
#include<string>
#include<cstdio>
using namespace std;
string s,t;
char w[256];
vector<string>	v;
int n,i;
int main(){
	for(gets(w),sscanf(w,"%d",&n),v.push_back("");n--;){
		s.assign(gets(w));
		if(s[0]=='c'){
			s.erase(0,3);
			if(s[0]=='/')v.clear();
			for(size_t x;(x=s.find_first_of("/"))!=string::npos;s.erase(0,x+1))
				((t=s.substr(0,x))=="..")?v.pop_back():v.push_back(t);
			(s=="..")?v.pop_back():v.push_back(s);
		}else{
			for(i=0;i<v.size();++i)cout<<v[i]<<"/";
			cout<<endl;
		}
	}
	return 0;
}
