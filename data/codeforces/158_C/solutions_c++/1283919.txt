#include <iostream>
#include <string>
using namespace std;

int main() {
	string path = "/";
	int N;
	cin >> N;
	while (N--) {
		string A, B;
		cin >> A;
		if (A == "cd") {
			cin >> B; B += "/";
			string temp = "";
			for (int i = 0; i < B.length(); i++) {
				if (B[i] == '/') {
					if (temp == "") path = "/";
					else if (temp == "..") {
						int pos = path.find_last_of('/', path.length() - 2);
						path = path.substr(0, pos + 1);
					}
					else path += (temp + "/");
					temp = "";
				}
				else temp += B[i];
			}
		}
		else cout << path << endl;
	}
	return 0;
}
