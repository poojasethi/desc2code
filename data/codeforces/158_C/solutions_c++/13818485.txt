#include<iostream>
#include<string>
#include<vector>

using namespace std;

int main(){
	int n;
	cin >> n;
	vector<string> a;
	string b, c, d;
	while( n-- ){
		cin >> b;
		if( b=="pwd" ){
			cout << '/';
			for( int i=0 ; i<a.size() ; i++ ) cout << a[i];
			cout << endl;
		}
		else{
			int x;
			cin >> c;
			c += '/';
			if( c[0]=='/' ){
				a.clear();
				c.erase(0,1);
			}
			while( c!="" ){
				x = c.find('/');
				d = c.substr(0,x+1);
				c.erase(0, x+1);
				if(	d=="../" ) a.pop_back();
				else{
					a.push_back(d);
				}
			}
		}
	}
	return 0;
}
