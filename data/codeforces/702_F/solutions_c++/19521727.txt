struct Node {
	Node();
	Node *l, *r, *p;
	int size;
	pair<int, int> key;
	int rev, lz, lz2;
	int ans;
};
Node* nil = new Node();
Node::Node() {
	l = r = p = nil;
	size = 1; rev = lz = lz2 = 0;
	key = make_pair(0, 0);
	ans = 0;
}
void init() {
	nil->l = nil->r = nil->p = nil;
	nil->size = 0;
}
int isrt(Node* x) {
	return x->p == nil || (x->p->l != x && x->p->r != x);
}
void setchild(Node* p, Node* c, int l) {
	c->p = p; l ? p->l = c : p->r = c;
}
void updatelz(Node* x, int val) {
	if (x == nil) return;
	x->lz += val;
	x->key.fi += val;
}
void updatelz2(Node* x, int val) {
	if (x == nil) return;
	x->lz2 += val;
	x->ans += val;
}
void pushdown(Node* x) {
	if (x == nil) return;
	Node *u = x->l, *v = x->r;
	if (x->rev) {
		if (u != nil) {swap(u->l, u->r); u->rev ^= 1;}
		if (v != nil) {swap(v->l, v->r); v->rev ^= 1;}
		x->rev = 0;
	}
	if (x->lz) {
		if (u != nil) updatelz(u, x->lz);
		if (v != nil) updatelz(v, x->lz);
		x->lz = 0;
	}
	if (x->lz2) {
		if (u != nil) updatelz2(u, x->lz2);
		if (v != nil) updatelz2(v, x->lz2);
		x->lz2 = 0;
	}
}
void pushup(Node* x) {
	x->size = x->l->size + x->r->size + 1;
}
void rotate(Node* x) {
	Node* y = x->p;
	int l = x->p->l == x;
	if (!isrt(y)) {
		setchild(y->p, x, y->p->l == y);
	}
	else {
		x->p = y->p;
	}
	setchild(y, l ? x->r : x->l, l);
	setchild(x, y, !l);
	pushup(y);
}
void splay(Node* x) {
	pushdown(x);
	while (!isrt(x)) {
		pushdown(x->p->p); pushdown(x->p); pushdown(x);
		if (!isrt(x->p)) rotate((x->p->l == x) == (x->p->p->l == x->p) ? x->p : x);
		rotate(x);
	}
	pushup(x);
}

const int MAXN = 400000 + 10;
int ans[MAXN];
struct SplayTree {
	Node mem[MAXN], *root;
	int cur;
	SplayTree() {
		root = nil; cur = 0;
		init();
	}
	Node* alloc(pair<int, int> key) {
		mem[cur].key = key;
		return mem + (cur++);
	}
	Node* insert(Node* x, pair<int, int> key) {
		Node* p = nil;
		while (x != nil) {
			pushdown(x);
			p = x;
			if (x->key < key) x = x->r;
			else x = x->l;
		}
		x = alloc(key);
		x->p = p;
		if (p == nil) root = x;
		else if (p->key < x->key) p->r = x;
		else p->l = x;
		splay(x);
		return root = x;
	}
	Node* insert(Node* x, Node* y) {
		Node* p = nil;
		while (x != nil) {
			pushdown(x);
			p = x;
			if (x->key < y->key) x = x->r;
			else x = x->l;
		}
		y->p = p;
		if (p == nil) root = y;
		else if (p->key < y->key) p->r = y;
		else p->l = y;
		splay(y);
		return y;
	}
	Node* insert(int key, int id) {
		return insert(root, make_pair(key, id));
	}
	Node* findkey(Node* x, pair<int, int> key) {
		while (x != nil) {
			pushdown(x);
			if (x->key < key) x = x->r;
			else if (key < x->key) x = x->l;
			else return x;
		}
		return nil;
	}
	Node* findkey(pair<int, int> key) {
		return findkey(root, key);
	}
	Node* findpos(Node* x, int pos) {
		while (1) {
			pushdown(x);
			int k = x->l ? x->l->size + 1 : 1;
			if (pos == k) return x;
			else if (pos < k) x = x->l;
			else {x = x->r; pos -= k;}
		}
		return nil;
	}
	Node* findpos(int pos) {
		return findpos(root, pos);
	}
	Node* join(Node* x, Node* y) {
		if (x == nil) return root = y;
		if (y == nil) return root = x;
		pushdown(y);
		while (1) {
			pushdown(x);
			if (x->r == nil) break;
			x = x->r;
		}
		splay(x);
		setchild(x, y, 0);
		pushup(x);
		return root = x;
	}
	void split(Node* x, int pos, Node*& l, Node*& r) {
		if (!pos) {l = nil; r = x; return;}
		Node* y = findpos(x, pos);
		splay(y);
		if ((r = y->r) != nil) r->p = nil;
		if ((l = y) != nil) {l->r = nil; pushup(l);}
	}
	void split(Node* t, int pos1, int pos2, Node*& t1, Node*& t2, Node*& t3) {
		split(t, pos1 - 1, t1, t2);
		split(t2, pos2 - pos1 + 1, t2, t3);
	}
	void reverse(int l, int r) {
		Node *x, *y, *z, *t;
		split(root, r, t, z);
		split(t, l - 1, x, y);
		if (y) {
			swap(y->l, y->r);
			y->rev ^= 1;
		}
		join(join(x, y), z);
	}
	void update(int l, int r, int val) {
		Node *x, *y, *z;
		split(root, l, r, x, y, z);
		updatelz(y, val);
		join(join(x, y), z);
	}
	void update(int val) {
		insert(root, make_pair(val, -INF));
		Node* x = root->l;
		Node* y = root->r;
		x->p = y->p = nil;
		if (y != nil) {
			updatelz(y, -val);
			updatelz2(y, 1);
			if (x == nil) {
				root = y;
				return;
			}
			while (x != nil && y != nil) {
				while (x->r != nil) pushdown(x), x = x->r;
				while (y->l != nil) pushdown(y), y = y->l;
				splay(x), splay(y);
				if (x->key > y->key) {
					Node* z = x;
					x = x->l; x->p = nil;
					z->l = z->r = nil;
					Node* t = y;
					y = y->r; y->p = nil;
					t->l = t->r = nil;
					x = insert(x, t);
					y = insert(y, z);
				}
				else {
					break;
				}
			}
			root = join(x, y);
		}
		else {
			root = x;
		}
	}
	void trace(Node* x) {
		if (x == nil) return;
		pushdown(x);
		ans[x->key.se] = x->ans;
		trace(x->l);
		trace(x->r);
	}
	void trace() {
		trace(root);
	}
} slt;

const int maxn = 200000 + 10;
int n, k;
int c[maxn];
int q[maxn];
int b[maxn];

int compare(int i, int j) {
	if (q[i] != q[j]) return q[i] > q[j];
	if (c[i] != c[j]) return c[i] < c[j];
	return i < j;
}

void solve() {
	cin >> n;
	vi v;
	FOR(i, 0, n) cin >> c[i] >> q[i], v.pb(i);
	sort(all(v), compare);
	cin >> k;
	FOR(i, 0, k) {
		cin >> b[i];
		slt.insert(b[i], i);
	}
	FOR(j, 0, sz(v)) {
		int i = v[j];
		slt.update(c[i]);
	}
	slt.trace();
	FOR(i, 0, k) cout << ans[i] << " \n"[i == k - 1];
}
