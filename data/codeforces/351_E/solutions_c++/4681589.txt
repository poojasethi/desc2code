#include<cstdio>
using namespace std;
int n,a,b,p[2010],ans;
int main(){
	scanf("%d",&n);
	for (int i=0;i<n;i++){
		scanf("%d",&p[i]);
		if (p[i]<0)p[i]=-p[i];
	}
	for (int i=0;i<n;i++){
		a=b=0;
		for (int j=0;j<n;j++)
			if (p[i]>p[j]){
				if (i>j)a++;
				if (i<j)b++;
			}
		ans+=(a<b?a:b);
	}
	printf("%d\n",ans);
	return 0;
}
