#include <cstdio>
#include <algorithm>
using namespace std;

int n, a[2005];

int main() {
	scanf("%d", &n);
	for(int i = 0; i < n; ++i) {
		scanf("%d", a+i);
		a[i] = abs(a[i]);
	}
	int ans = 0;
	for(int i = 0; i < n; ++i) {
		int cnt0 = 0, cnt1 = 0;
		for(int j = 0; j < n; ++j) if(a[j] < a[i]) {
			if(j < i) ++cnt0;
			if(j > i) ++cnt1;
		}
		ans += min(cnt0, cnt1);
	}
	printf("%d\n", ans);
}
