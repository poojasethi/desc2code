#include <iostream>
#include <cstdio>

using namespace std;

int A[2005];

int main() {
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> A[i];
	for (int i = 1; i <= n; i++)
		if (A[i] < 0)
			A[i] = -A[i];
	int ans = 0;
	for (int i = 1; i <= n; i++) {
		int a = 0, b = 0;
		for (int j = 1; j <= n; j++)
			if (A[j] < A[i]) {
				if (j < i) a++;
				else b++;
			}
		ans += min(a, b);
	}
	cout << ans << endl;
	return 0;
}
