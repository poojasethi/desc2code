#include <cstdio>
#include <algorithm>
using namespace std;
const int maxn=2010;
int a[maxn];
int main(){
    int n,i,j,x,y,s=0;
    scanf("%d",&n);
    for (i=1;i<=n;i++){
        scanf("%d",&a[i]);
        if (a[i]<0) a[i]=-a[i];
    }
    for (i=1;i<=n;i++){
        x=y=0;
        for (j=1;j<i;j++) if (a[j]<a[i]) x++;
        for (j=i+1;j<=n;j++) if (a[j]<a[i]) y++;
        s+=min(x,y);
    }
    printf("%d\n",s);
    return 0;
}
