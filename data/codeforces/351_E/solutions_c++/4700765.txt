#include <iostream>
#include <algorithm>
using namespace std;
int a[2000];
int main(){
	int n;
	cin >> n;
	for(int i=0;i<n;i++){
		cin >> a[i];
		if(a[i]<0) a[i]=-a[i];
	}
	int ans=0;
	for(int i=0;i<n;i++){
		int cnt1=0,cnt2=0;
		for(int j=0;j<i;j++){
			if(a[i]>a[j]) cnt1++;
		}
		for(int j=i+1;j<n;j++){
			if(a[i]>a[j]) cnt2++;
		}
		ans+=min(cnt1,cnt2);
	}
	cout << ans;
	return 0;
}
