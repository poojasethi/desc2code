#include <cmath>
#include <cstdio>
#include <algorithm>

using namespace std;

int n;
int a[2048];

int main()
{
    scanf("%d",&n);
    for (int i=1;i<=n;i++) scanf("%d",a+i);
    int ans=0;
    for (int i=1;i<=n;i++)
	{
		int l=0,r=0;
		for (int j=1;j<=n;j++)
		{
			if (j<i && abs(a[j])<abs(a[i])) l++;
			if (j>i && abs(a[j])<abs(a[i])) r++;
		}
		ans+=min(l,r);
	}
	printf("%d\n",ans);
    return 0;
}
