#include <iostream>
#include <cmath>
#include <cstdio>

using namespace std;

int p[3001], n;

int main() {
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> p[i];
		p[i] = abs(p[i]);
	}
	
	int ans = 0;
	
	for (int i = 0; i < n; i++) {
		int l = 0, r = 0;
		for (int j = 0; j < i; j++) 
			if (p[j] < p[i])
				l++;
		for (int j = i + 1; j < n; j++)
			if (p[j] < p[i])
				r++;
		ans += min(l, r);
	}
	
	cout << ans << endl;
	
	return 0;
}
