#include<cstdio>
int n,a[2001],v[2001];
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
      scanf("%d",&a[i]);
      if(a[i]<0)a[i]=-a[i];
    }
    for(int i=1;i<=n;i++)
      for(int j=i+1;j<=n;j++)
      {
        if(a[i]>a[j])v[i]--;
        else if(a[i]<a[j])v[j]++;
      }
    for(int i=1;i<=n;i++)
      if(v[i]<0)a[i]=-a[i];
    int ans=0;
    for(int i=1;i<=n;i++)
      for(int j=i+1;j<=n;j++)
        if(a[i]>a[j])ans++;
    printf("%d\n",ans);
    return 0;
}
