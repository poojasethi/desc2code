#include <bits/stdc++.h>
using namespace std;
#define REP(a, b, c) for(int a = b; a < c; a++)
int a[2002];
int main(){

    int n;
    cin >> n;
    REP(i, 0, n) { cin >> a[i]; a[i] = abs(a[i]); }

    int ans = 0;
    REP(i, 0, n){
        int p = 0, q = 0;
        REP(j, 0, n) {
            if(abs(a[i]) > abs(a[j])){
                if(i < j) p++;
                else q++;
            }
        }
        if(p < q) a[i] *= -1;
        ans += min(p, q);
    }
    cout << ans << endl;

    return 0;
}
