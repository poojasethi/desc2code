
#include <cstdio>

int n;
int d[2010];

int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;++i){
		scanf("%d",d+i);
		if(d[i]<0) d[i] *=-1;
	}

	int ans=0;
	for(int i=0;i<n;++i) for(int j=i;j<n;++j) if(d[i]>d[j]) ans ++;

	for(int i=0;i<n;++i){
		int j=0;
		int res = 0;
		while(j<i){
			if(d[i]>d[j++]) res++;
		}
		while(j<n){
			if(d[i]>d[j++]) res --;
		}
		if(res<0) ans += res;
	}

	printf("%d\n",ans);

	return 0;
}
