#include <stdio.h>

#define M 2001
int a[M];
int main(){
    //freopen("in.txt","r",stdin);
    int i,n,x,y;
    scanf("%d",&n);

    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
        if(a[i]<0) a[i]=-a[i];
    }

    int ans=0;
    for(i=0;i<n;i++){
        int x=0;
        int y=0;
        for(int j=0;j<n;j++) if(a[i]>a[j]){
            if(i>j) x++;
            if(i<j) y++;
        }
        ans+=(x>y?y:x);
    }
    printf("%d\n",ans);

}
