#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>

using namespace std;

int a[2000+3],n,ans;

int main()
{
scanf("%d",&n);
for (int i=1;i<=n;i++) scanf("%d",&a[i]),a[i]=abs(a[i]);
for (int i=1;i<=n;i++)
  {
  int x=0,y=0;
  for (int j=1;j<=n;j++)
    if (a[j]<a[i])
      {
      if (j<i) x++;
        else y++;
      }
  ans+=min(x,y);
  }
printf("%d",ans);
}
