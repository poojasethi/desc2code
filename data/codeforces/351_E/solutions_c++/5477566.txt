#include <cstdio>
#include <iostream>

using namespace std;

int n, a[1000001], ans;

int main()
{
	scanf("%d", &n);

	for (int i = 1; i <= n; ++i)
	{
		scanf("%d", &a[i]);
		if (a[i] < 0)
			a[i] = -a[i];
	}

	for (int i = 1; i <= n; ++i)
	{
		int x = 0, y = 0;
		for (int j = 1; j < i; ++j)
			if (a[j] < a[i])
				++x;

		for (int j = i + 1; j <= n; ++j)
			if (a[j] < a[i])
				++y;

		ans += min(x, y);
	}

    cout << ans;
}
