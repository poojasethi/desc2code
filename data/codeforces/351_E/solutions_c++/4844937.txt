#include<stdio.h>
int a[2001];
int main(){
	int T,n,ans;
	int cnt1,cnt2;
	//scanf("%d",&T);
	while(scanf("%d",&n)==1){
		
		for(int i=0;i<n;i++){
			scanf("%d",&a[i]);
			if(a[i]<0) a[i]*=-1;
		}
		ans=0;
		for(int i=0;i<n;i++){
			cnt1=cnt2=0;
			for(int j=0;j<i;j++){
				if(a[j]<a[i]) cnt1++;
			}
			for(int j=i+1;j<n;j++){
				if(a[j]<a[i]) cnt2++;
			}
			if(cnt1<cnt2) ans+=cnt1;
			else ans+=cnt2;
		}
		printf("%d\n",ans);
	}
}
