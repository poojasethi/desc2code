#include <cstdio>
#include <algorithm>
using namespace std;

int a[123456];
int main() {
    int n, i, j, af, b, ans = 0;

    scanf("%d", &n);
    for (i = 0; i < n; i++) {
        scanf("%d", a+i); a[i] = abs(a[i]);
    }
    for (i = 0; i < n; i++) {
        af = b = 0;
        for (j = 0; j < i; j++) af += (a[j] < a[i]);
        for (j = i+1; j < n; j++) b += (a[j] < a[i]);
        ans += min(af, b);
    }
    printf("%d\n", ans);

    return 0;
}
