#include <cstdio>
#include <algorithm>

using namespace std;

const int maxn=2010;

int a[maxn],n,s;

int main() {
    s=0;
    scanf("%d",&n);
    for (int i=0;i<n;i++) 
        scanf("%d",a+i),a[i]=abs(a[i]);
    for (int i=0;i<n;i++) {
        int x=0,y=0;
        for (int j=0;j<n;j++)
            if (a[i]>a[j]) {
                if (j<i) 
                    x++;
                if (j>i)
                    y++;
            }
        s+=min(x,y);
    }
    printf("%d\n",s);
}
