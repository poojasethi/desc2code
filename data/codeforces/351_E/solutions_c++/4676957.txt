#include <bits/stdc++.h>
using namespace std;

int n, a[2000], ans;

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d", a + i);
		a[i] = abs(a[i]);
	}
	for (int i = 0; i < n; i++) {
		int pos = 0, neg = 0;
		for (int j = 0; j < n; j++)
		if (a[j] < a[i]) {
			if (j < i) neg++;
			if (j > i) pos++;
		}
		ans += min(pos, neg);
	}
	printf("%d", ans);
}
