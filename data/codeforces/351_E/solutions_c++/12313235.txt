#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<math.h>
#include<stack>
#include<queue>
using namespace std ;
int a[2005];

int main()
{
	int n ;
	int sum=0 ;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
			scanf("%d",&a[i]);
			if(a[i]<0)
			a[i]=-a[i];
	}
	for(int i=0;i<n;i++)
	{
		int c=0,b=0;
		for(int j=0;j<i;j++)
		{
			if(a[j]<a[i])
			c++;
		}
		for(int j=i+1;j<n;j++)
		{
			if(a[j]<a[i])
			b++;
		}
	    sum+=min(c,b);
	}
	cout<<sum<<endl ;
	return 0 ; 
}
