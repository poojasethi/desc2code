#include <cstdio>
#include <algorithm>
int a[2012];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i = 1; i <= n; i++) {
        scanf("%d",&a[i]);
        if(a[i]<0) a[i] = -a[i];
    }
    int ans = 0;
    for(int i = 1; i <= n; i++) {
        int x=0,y=0;
        for(int j = 1; j <= n; j++)if(a[j] < a[i]) {
            if(j < i) x++;
            if(j > i) y++;
        }
        ans += std::min(x,y);
    }
    printf("%d\n",ans);
    return 0;
}
