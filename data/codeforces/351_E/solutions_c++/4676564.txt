#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
int a[3005];
int main()
{
    int n;
    scanf("%d", &n);
    for (int i=0; i<n; i++) scanf("%d", &a[i]);
    for (int i=0; i<n; i++)
        if (a[i]<0)
            a[i]=-a[i];
    int ans=0;
    for (int i=0; i<n; i++) {
        int l=0, r=0;
        for (int j=0; j<i; j++) l+=(a[j]<a[i]);
        for (int j=i+1; j<n; j++) r+=(a[j]<a[i]);
        ans+=min(l, r);
    }
    printf("%d\n", ans);
    return 0;
}
