#include <stdio.h>
#include <string.h>
#include <math.h>
int main()
{
	int n, l, r;
	int i,j;
	int a[2000];
	int sum = 0;
	scanf("%d", &n);
	for (i = 0; i < n; i++)
	{
		scanf("%d", &a[i]);
		if (a[i] < 0)
			a[i] = -a[i];
	}
	for (i = 0; i < n; i++)
	{
		l = 0;
		r = 0;
		for (j = 0; j < i ; j++)
		{
			if (a[i] > a[j])
				l++;
		}
		for (j = i + 1; j < n; j++)
		{
			if (a[i] > a[j])
				r++;
		}
		if (r < l)
			sum += r;
		else 
			sum += l;
	}
	printf("%d\n", sum);
	return 0;
}
