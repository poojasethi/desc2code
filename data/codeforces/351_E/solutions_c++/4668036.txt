#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;

int a[2010], n, ans;

int comp(int d, int ord){
	int s = 0;
	for (int j=1; j<=ord-1; j++) if (d<a[j]) s++;
	for (int j=ord+1; j<=n; j++) if (a[j]<d) s++;
	return s;
}

int main(){
	cin >> n;
	for (int i=1; i<=n; i++){
		cin >> a[i]; a[i] = abs(a[i]);
	}
	for (int i=1; i<=n; i++)
		if (comp(a[i], i) > comp(-a[i], i)) a[i] = -a[i];
	for (int i=1; i<=n; i++) ans += comp(a[i], i);
	ans /= 2;
	cout << ans << endl; return 0;
}
