#include <cstdio>
#include <algorithm>
using std::abs;
using std::min;
int a[2002];
int main(){
	int n,cnt = 0; 
	scanf("%d",&n);
	for (int i = 0; i < n; ++i) scanf("%d",a+i) , a[i] = abs(a[i]);
	for (int i = 0; i < n; i++){
		int l = 0, r = 0;
		for (int j = 0; j < i; j++)
			if (a[j] < a[i]) l++;
		for (int j = i + 1; j < n; j++)
			if (a[j] < a[i]) r++;
		cnt += min(l, r);
	}
	printf("%d\n",cnt);
}
