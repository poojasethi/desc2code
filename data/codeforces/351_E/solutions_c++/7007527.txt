#include <stdio.h>
#define N 2009
int n,p[N],ans;
main() {
	//freopen("204_E.in","r",stdin);
	scanf("%d",&n);
	for (int i=0;i<n;i++) {
		scanf("%d",p+i);
		if (p[i]<0) p[i]*=-1;
	}
	for (int i=0;i<n;i++) {
		int a=0,b=0;
		for (int j=0;j<n;j++) if (p[i]>p[j]) {
			if (i<j) a++;
			if (i>j) b++;
		}
		ans+=(a<b?a:b);
	}
	printf("%d\n",ans);
}
