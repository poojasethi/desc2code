#include<cstdio>
#include<cstdlib>
#include<algorithm>

using namespace std;

const int MAXN=2000+9;

int a[MAXN];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
#endif
	int n,i,j;
	scanf("%d",&n);
	for(i=1;i<=n;++i)
		scanf("%d",a+i);
	int ans=0;
	for(i=1;i<=n;++i)
	{
		int left=0;
		for(j=1;j<i;++j)
			if(abs(a[i])>abs(a[j]))
				++left;
		int right=0;
		for(++j;j<=n;++j)
			if(abs(a[i])>abs(a[j]))
				++right;
		ans+=min(left,right);
	}
	printf("%d\n",ans);
	return 0;
}
