#include <cstdio>
using namespace std;

int n, i, j, x, y, ans, p[3100];

#define abs(x) ((x) > 0 ? (x) : -(x))
#define min(x, y) ((x) < (y) ? (x) : (y))

int main() {
	scanf("%d", &n);
	for (i = 1; i <= n; i++)	scanf("%d", &p[i]), p[i] = abs(p[i]);
	for (i = 1; i <= n; i++) {
		for (x = 0, j = 1; j < i; j++)	x += (p[j] < p[i]);
		for (y = 0, j = i + 1; j <= n; j++)	y += (p[j] < p[i]);
		ans += min(x, y);
	}
	printf("%d\n", ans);
}
