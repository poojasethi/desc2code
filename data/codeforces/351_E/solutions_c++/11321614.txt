#include <cstdio>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstring>

using namespace std;

int n,a[2005],l,r,ans;
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;++i)
	{
		scanf("%d",&a[i]);
		a[i]=abs(a[i]);
	}
	for (int i=1;i<=n;++i)
	{
		l=0;r=0;
		for (int j=1;j<i;++j) if (a[j]<a[i]) l++;
		for (int j=i+1;j<=n;++j) if (a[j]<a[i]) r++;
		ans+=min(l,r); 
	}
	printf("%d\n",ans);
	return 0;
}
