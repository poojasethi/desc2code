#include <stdio.h>
#define N 2010

int a[N];

inline int Abs(int a) { if (a<0) return -a; return a; }

int main() {
//	freopen ("in.txt", "r", stdin);
	int n, ans=0;
	scanf("%d", &n);
	for (int i=0; i<n; i++) scanf("%d", a+i), a[i]=Abs(a[i]);
	for (int i=0; i<n; i++) {
		int x=0, y=0;
		for (int j=0; j<i; j++) if (a[i]>a[j]) x++;
		for (int j=i+1; j<n; j++) if (a[i]>a[j]) y++;
		if (x>y) ans+=y;
		else ans+=x;
	}
	printf("%d\n", ans);
}
