#include<stdio.h>
#include<iostream>

using namespace std;

const int N = 2010;

int val[N];

int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&val[i]);
		if(val[i]<0) val[i]=-val[i];
	}
	int ans = 0;
	for(int i=1;i<=n;i++)
	{
		int l = 0, r = 0;
		for(int j=1;j<i;j++)
			if(val[j]<val[i])
				l++;
		for(int j=i+1;j<=n;j++)
			if(val[j]<val[i])
				r++;
		ans+=min(l,r);
	}
	printf("%d\n",ans);
	return 0;
}
