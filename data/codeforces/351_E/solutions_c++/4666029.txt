#include <cstdio>
#include <algorithm>

using namespace std;

int n;

int tab[2005];

int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d",tab+i);
		if(tab[i]<0) tab[i]*= -1;
	}
	int wyn=0;
	for(int i=0;i<n;i++)
	{
		int a=0,b=0;
		for(int j=0;j<i;j++) if(tab[j]<tab[i]) a++;
		for(int j=i+1;j<n;j++) if(tab[j]<tab[i]) b++;
		wyn+=min(a,b);
	}
	printf("%d\n",wyn);
	return 0;
}
