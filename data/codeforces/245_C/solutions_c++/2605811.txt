#include<cstdio>
#include<algorithm>
using namespace std;
int m,a[105],n,i,sol;
int main()
{
    scanf("%d",&n);
    if(n==1||n%2==0) {printf("-1\n"); return 0;}
    for(i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
    }
    for(i=n/2;i>=1;i--)
    {
        m=max(a[i*2],a[i*2+1]);
        a[i]=a[i]>m?a[i]-m:0;
        sol+=m;
    }
    printf("%d",sol+a[1]);
    return 0;
}
