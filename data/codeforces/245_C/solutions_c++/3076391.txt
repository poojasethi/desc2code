#include<stdio.h>
int max(int a,int b)
{
	return a>b?a:b;
}
int main()
{
	int a[1005],i,n,times,m;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	times=0;
	if(n%2==0||n<3)
	{
		printf("-1\n");
	}
	else
	{
		for(i=n/2;i>0;i--)
		{
			m=max(a[2*i],a[2*i+1]);
			a[i]=max(0,a[i]-m);
			a[2*i]=0;
			a[2*i+1]=0;
			times+=m;
		}
		times+=a[1];
		printf("%d\n",times);
	}
	return 0;
}
