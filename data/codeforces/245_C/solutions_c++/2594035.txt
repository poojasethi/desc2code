#include <string>
#include <iostream>
using namespace std;
int main(int argc, char **argv)
{int n,i,x,a[10000];
	cin >> n;
	for (i=1;i<=n;i++) cin >> a[i];
	
	if (n<3 or n%2==0) cout << -1; else
	{	x=0;
		for (i=n;i>=1;i--)
		while (a[i]>0) 
		{if (i%2==0) {a[i]-=1;a[i+1]-=1;a[i/2]-=1;x++;} else
		{a[i]-=1;a[i-1]-=1;a[(i-1)/2]-=1;x++;}} 
		cout << x;
	}

	
	return 0;
}
