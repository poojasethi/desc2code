#include<iostream>
using namespace std;
int max(int a, int b) { return a > b ? a : b; }
int main()
{
int n,a[105],i,k=0,p;
cin>>n;
if(n<3 || n%2==0) cout<<-1;
else
{
for(i=1;i<=n;i++) cin>>a[i];
for(i=(n-1)/2;i>=1;i--)
{
p=0;
p=max(p,a[i*2]);
p=max(p,a[i*2+1]);
a[i]=max(0,a[i]-p);
a[i*2]=max(0,a[i*2]-p);
a[i*2+1]=max(0,a[i*2+1]-p);
k+=p;
//cout<<"k = " << k<<endl;
}
cout<<k+a[1]<<endl;
}
return 0;
}
