#include <iostream>

using namespace std;

const int MAXN = 101;
int a[MAXN];

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i + 1];
	if (n < 2 || n % 2 == 0) {
		cout << -1 << endl;
		return 0;
	}
	int ans = 0;
	for (int i = n / 2; i > 0; i--) {
		ans += max(a[i * 2], a[i * 2 + 1]);
		a[i] = max(0, a[i] - max(a[i * 2], a[i * 2 + 1]));
	}
	ans += a[1];
	cout << ans << endl;
}
