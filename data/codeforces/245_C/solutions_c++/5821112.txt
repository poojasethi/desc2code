#include <iostream>
#include <algorithm>
using namespace std;
int a[110];
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	if(n%2==0 || n<=2)
	{
		cout<<-1<<endl;
		return 0;
	}
	int ans=0;
	while(n>2)
	{
		int y=max(a[n],a[n-1]);
		a[n]=a[n-1]=0;
		a[n/2]-=y;
		ans+=y;
		if(a[n/2]<0)
			a[n/2]=0;
		n--;
		n--;
	}
	ans+=max(a[1],a[2]);
	cout<<ans<<endl;	
	return 0;
}
