#include<stdio.h>
int a[110];
int max(int a,int b){
	return a>b?a:b;
}
int main(){
	int i,j,n,as,e;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	if(n%2==0||n==1){puts("-1");return 0;}
	as=0;
	for(i=n;i>0;i-=2)
		if(a[i]>=0||a[i-1]>=0){
			e=max(a[i],a[i-1]);
			as+=e;
			a[i]-=e;
			a[i-1]-=e;
			a[(i-1)/2]-=e;
		}
	printf("%d\n",as);
	return 0;
}
