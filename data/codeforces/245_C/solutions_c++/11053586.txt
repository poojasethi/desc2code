#include <iostream>
using namespace std;
int n,a[105],res=0;
int main(){
    cin>>n;
    for (int i=1; i<=n; i++) cin>>a[i];
    if (n%2==0 || (n-1)/2<1) {cout<<-1; return 0;}
    for (int i=n; i>2; i-=2){
        res+=max(a[i],a[i-1]);
        a[(i-1)/2]>max(a[i],a[i-1]) ? a[(i-1)/2]-=max(a[i],a[i-1]):a[(i-1)/2]=0;
        a[i]=0, a[i-1]=0;
    }
    cout<<(a[1]>0 ? res+a[1]:res);
    return 0;
}
