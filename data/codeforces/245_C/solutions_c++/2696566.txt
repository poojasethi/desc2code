#include<iostream>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;
int a[111];
int main(){
   int n,x,ans=0;
   cin>>n;
   rep(i,n) cin>>a[i+1];
   if(n%2==0||n==1) {cout<<-1;return 0;}
   while(n>1){
      x=(n-1)/2;
      a[x]-=max(a[n],a[n-1]);
      a[x]=max(a[x],0);
      ans+=max(a[n],a[n-1]);
      n-=2;  
   }
   if(a[1]>0) ans+=a[1];
   cout<<ans<<endl;
   return 0; 
}
