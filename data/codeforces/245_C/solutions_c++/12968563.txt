#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int a[1000];
int main()  
{  
int n,i,sum,k,ans;
scanf("%d",&n);
if(n%2==0||n<3)
{
	printf("-1\n");
	return 0;
}
for(i=1;i<=n;i++)
{
	scanf("%d",&a[i]);
}
sum=0;
for(i=n;i>=1;i-=2)
{   
    if(a[i]>0||a[i-1]>0)
    {
	k=(i-1)/2;
	ans=max(a[i],a[i-1]);
	sum+=ans;
	a[k]=max(a[k]-ans,0);
    }	
}
printf("%d\n",sum);

 
}
