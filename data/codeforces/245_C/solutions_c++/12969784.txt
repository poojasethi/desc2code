#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int a[1100];
int main(){
	int n,i;
	int ans,x;
	ans=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		scanf("%d",&a[i]);
	}
	if(n<3||n%2==0) {
		printf("-1\n");
	}
	else {
	for(i=n;i>=1;i-=2){
	x=max(0,max(a[i],a[i-1]));
	a[(i-1)/2]-=x;
	ans+=x;
	//printf("%d****\n",x);
	}
	printf("%d\n",ans);
	}
}
