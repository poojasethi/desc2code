#include<stdio.h>
int main()
{
	int n,i,x,ans=0,a[110];
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);		
	}
	if(n%2==0||n<3)
	printf("-1\n");
	else 
	{
		for(i=n;i>=1;i-=2)
		{
			if(a[i]>a[i-1])
			x=a[i];
			else x=a[i-1];
			if(x<0)
			x=0;
			a[(i-1)/2]-=x;
			ans+=x;
		}
		printf("%d\n",ans);
	}
}
