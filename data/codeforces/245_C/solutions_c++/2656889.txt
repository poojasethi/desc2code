#include <iostream>
using namespace std;
int main(){
	int n,a[100],ans=0,m;
	cin >> n;
	for(int i=0;i<n;i++) cin >> a[i];
	if(n==1 || n%2==0){
		cout << -1;
		return 0;
	}
	for(int i=(n-1)/2;i>1;i--){
		m=max(a[2*i],a[2*i-1]);			//0-index
		a[2*i]=0;
		a[2*i-1]=0;
		a[i-1]=max(0,a[i-1]-m);
		ans+=m;
	}
	m=max(a[0],max(a[1],a[2]));
	ans+=m;
	cout << ans;
	return 0;
}
