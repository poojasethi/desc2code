#include <cstring>
#include <cstdio>
using namespace std;
int main()
{
	int n,x,a[10000];
	scanf ("%d", &n);
	for (int i=1;i<=n;i++)scanf ("%d", a+i);
	
	if (n<3 or n%2==0)return puts ("-1"), 0;
	else
	{	x=0;
		for (int i=n;i>=1;i--)
		while (a[i]>0) 
		{if (i%2==0) {a[i]-=1;a[i+1]-=1;a[i/2]-=1;x++;} else
		{a[i]-=1;a[i-1]-=1;a[(i-1)/2]-=1;x++;}} 
		printf ("%d\n", x);
	}
}
