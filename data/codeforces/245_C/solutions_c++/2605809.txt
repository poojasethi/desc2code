#include<cstdio>
#define max(a,b) a>b?a:b
int i,n,A[110],M,SOL;
int main()
{
    scanf("%d",&n);
    for(i=1;i<=n;i++)
        scanf("%d",&A[i]);
    if(n==1||!(n%2)){printf("-1\n");return 0;}
    for(i=n/2;i>=1;i--)
    {
        M=max(A[i+i],A[i+i+1]);
        SOL+=M;A[i+i]=0;A[i+i+1]=0;
        if(A[i]>M)A[i]-=M; else A[i]=0;
    }
    printf("%d\n",SOL+A[1]);
    return 0;
}
