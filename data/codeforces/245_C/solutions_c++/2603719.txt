#include <iostream>
using namespace std;

#define MAXN 110

int N, sir[MAXN];

int main () {
	int rez = 0;
	cin >> N;
	
	if (N % 2 == 0 || N == 1) {
		cout << -1;
		return 0;
	}
	
	for (int i = 1; i <= N; ++i)
		cin >> sir[i];
	for (int i = (N - 1) / 2; i >= 1; --i) {
		rez += max (sir[i * 2], sir[i * 2 + 1]);
		sir[i] -= max (sir[i * 2], sir[i * 2 + 1]);
		sir[i] = max(0, sir[i]);
	}
	rez += sir[1];
	
	cout << rez;
	return 0;
}
