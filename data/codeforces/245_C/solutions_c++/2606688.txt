#include<cstdio>
#include<algorithm>
using namespace std;
int n,a[110],i,M,sol;
int main()
{
    scanf("%d",&n);
    if(n==1||!(n%2)) {printf("-1\n");return 0;}
    for(i=1;i<=n;i++) scanf("%d",&a[i]);
    for(i=n/2;i>=1;i--)
    {
        M=max(a[2*i],a[2*i+1]);
        sol+=M;
        if(a[i]<M) a[i]=0;
        else a[i]-=M;
    }
    printf("%d\n",sol+a[1]);
    return 0;
}
