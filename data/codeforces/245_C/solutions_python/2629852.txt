

n = int(raw_input())
a = list(map(int, raw_input().split()))
if n < 3 or n % 2 == 0:
  print(-1)
else:
  a.insert(0, 0)
  ans = 0
  for y in range(len(a) - 1, 0, -1):
    ans += a[y]
    if y == 1:
      break
    x = int(y/2)
    a[x] = max(0, a[x] - a[y])
    a[2*x] = max(0, a[2*x] - a[y])
    #print(a[y])
  print(ans)
