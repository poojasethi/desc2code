n, q = map(int, raw_input().split())
a = map(int, raw_input().split())

cnt = [0] * (n + 1)
for i in range(q):
    l, r = map(int, raw_input().split())
    cnt[l - 1] += 1
    cnt[r] -= 1
for i in range(1, n + 1): cnt[i] += cnt[i - 1]

a.sort()
a.reverse()

j, res = 0, 0
for i in reversed(sorted(cnt)):
    res += i * a[j]
    j += 1
    if j == n: break
print res
