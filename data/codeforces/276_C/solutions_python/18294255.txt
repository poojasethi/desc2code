#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n, q = ia()

a = ia(); a.sort(reverse=True)

d = [0 for i in range(n+1)]

for i in range(q):
    l, r = ia(); l-=1; r-=1
    d[l] += 1; d[r+1] -=1

c = [0 for i in range(n+1)]
s = 0
for i in range(n+1):
    s += d[i]
    c[i] = s

c.sort(reverse=True)
ans = 0
for i in range(n):
    ans += a[i]*c[i]

print ans
