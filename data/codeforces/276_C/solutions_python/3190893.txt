from sys import stdin

def line():
	return stdin.readline().strip()

n, q = map(int, line().split())
A = map(int, line().split())
Q = []
for _ in xrange(q):
	l, r = map(int, line().split())
	Q.append((l-1, r-1))

open = [0] * n
close = [0] * (n + 1)
for l, r in Q:
	open[l] += 1
	close[r+1] += 1

F = [0] * n
count = 0
for i in xrange(n):
	count += open[i] - close[i]
	F[i] = count
	
print sum(map(lambda (x, y): x*y, zip(sorted(F), sorted(A))))
