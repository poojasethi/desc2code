#!/usr/bin/env python
from __future__ import print_function
input = raw_input

if __name__ == '__main__':
    n, q = map(int, input().split())
    a = map(int, input().split())
    assert len(a) == n
    a.sort()

    start = [0]*q
    end = [0]*q
    for x in range(q):
        i, j = map(int, input().split())
        start[x] = i-1
        end[x] = j-1
    start.sort()
    end.sort()

    cnt = [0]*n
    current = 0 # open intervals
    j = 0
    k = 0
    for i in xrange(n):
        while j < len(start) and start[j] == i:
            current += 1
            j += 1
        cnt[i] = current
        while k < len(end) and end[k] == i:
            current -= 1
            k += 1

    cnt.sort()

    result = sum([x*y for x,y in zip(a, cnt)])
    print(result)
