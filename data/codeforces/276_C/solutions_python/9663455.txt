def iarr(): return map(int,raw_input().split())
[n,q] = iarr()
arr = iarr()
arr.sort()
count = [0]*(n+1)
for i in xrange(q):
    [l,h] = iarr()
    count[l-1] += 1
    count[h] -= 1
for i in xrange(1,n+1):
    count[i] = count[i]+ count[i-1]
count.pop()
count.sort()
print sum([i*j for i,j in zip(count,arr)])
