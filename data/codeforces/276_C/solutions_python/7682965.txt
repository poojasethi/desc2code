n, q = list(map(int, raw_input().split(' ')))
a = list(map(int, raw_input().split(' ')))
a.sort()
s = list(0 for i in range(n + 1))
for i in range(q):
	l, r = list(map(int, raw_input().split(' ')))
	s[l - 1] += 1
	s[r] -= 1
# for j in range(l, r + 1):
# 	s[j] += 1
for i in range(1, n):
	s[i] += s[i - 1]

s.sort()
ans = 0
for i in range(1, n + 1):
	ans += s[i] * a[i - 1]
print(ans)
