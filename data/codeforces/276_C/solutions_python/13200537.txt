import sys
n, q = map(int, sys.stdin.readline().split())
a = map(int, sys.stdin.readline().split())
l = [0] * (n + 1)
for i in range(q):
  x,y = map(int, sys.stdin.readline().split())
  l[x-1] += 1
  l[y] -= 1
ll = []
s= 0
for i in range(n):
  s += l[i]
  ll.append(s)
print sum(map(lambda (a, b): a * b, zip(sorted(a), sorted(ll))))
