n, q = map(int, raw_input().split())
a = sorted(map(int, raw_input().split()))
b = [0] * n
e = [0] * n
for i in xrange(q):
  l, r = map(int, raw_input().split())
  b[l - 1] += 1
  e[r - 1] += 1
k = 0
r = [0] * n
for i in xrange(n):
  r[i] = k + b[i]
  k += b[i] - e[i]
r = sorted(r)
res = 0
for i in xrange(n):
  res += a[i] * r[i]
print res
