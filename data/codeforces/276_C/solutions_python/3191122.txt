n,q = [int(x) for x in raw_input().split()]
a = [int(x) for x in raw_input().split()]
lr = [[int(x) for x in raw_input().split()] for _ in range(q)]

b = [0]*n
for l,r in lr:
    b[l-1] += 1
    if r != n:
        b[r] -= 1
if n != 1:
    for i in range(1,n):
        b[i] += b[i-1]
b.sort(reverse=True)
a.sort(reverse=True)
sum = 0
for j,k in zip(a,b):
    sum += j*k
    
print sum
