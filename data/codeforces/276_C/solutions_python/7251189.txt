n, q = map(int ,raw_input().split())
a = map(int, raw_input().split())
t = [0] * (n + 1)
for i in range(q) :
    l, r = map(int, raw_input().split())
    l -= 1
    t[l] += 1
    t[r] -= 1
cur = 0
for i in range(n + 1) : 
    cur += t[i]
    t[i] = cur
t.sort(reverse = True)
a.sort(reverse = True)
result = 0
for i in range(n) : 
    result += a[i] * t[i]
print(result)
