def main():
    def line():
        return map(int, raw_input().split(' '))

    n, q = line()

    a = sorted(line())

    aa = [0] * (n + 2)
    while q > 0:
        l, r = line()
        aa[l] += 1
        aa[r + 1] += -1
        q -= 1

    for i in range(1, n + 2):
        aa[i] += aa[i - 1]

    aa = aa[1:n + 1]
    aa.sort()

    sol = 0
    for i in range(n):
        sol += a[i] * aa[i]

    print sol


main()
