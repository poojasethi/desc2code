def main():
	n,m = [int(i) for i in raw_input().split()]
	xs = sorted([int(i) for i in raw_input().split()])
	deltas = [0]*(n+1)
	for i in xrange(m):
		l,r = [int(i)-1 for i in raw_input().split()]
		deltas[l]+=1
		deltas[r+1]-=1
	d = 0
	ys = []
	for i in deltas[:-1]:
		d+=i
		ys.append(d)
	ys.sort()
	print sum([xs[i]*ys[i] for i in xrange(n)])
	

main()
