n,q=map(int,raw_input().split())
a=map(int,raw_input().split())
v=[0]*(n+1)
b=[0]*(n+1)
for i in range (q):
    l,r=map(int,raw_input().split())
    v[l-1]+=1
    if r<(n):
        v[r]-=1
a.sort(reverse=True)
j=0
for i in range(n):
    j+=v[i]
    b[i]=j
b.sort(reverse=True)
k=0

    k+=a[i]*b[i]
