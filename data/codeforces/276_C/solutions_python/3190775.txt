#!/usr/bin/env python
import sys

OPEN = 0
CLOSE = 1

n, q = map(int, sys.stdin.readline().strip().split())
a    = map(int, sys.stdin.readline().strip().split())

opened = [0] * n
closed = [0] * n
for _ in xrange(q):
    l, r = map(int, sys.stdin.readline().strip().split())
    opened[l-1] += 1
    closed[r-1] += 1

freq = [0] * n
count = 0
for i in xrange(n):
    count += opened[i]
    freq[i] = count
    count -= closed[i]

a.sort()
freq.sort()

print sum(a[i] * freq[i] for i in xrange(n))
