#! /usr/bin/env python
read = lambda: map(int, raw_input().split())
n, q = read()
a = sorted(read())
cnt = [0] * (n + 1)
for i in xrange(q):
    l, r = read()
    cnt[l - 1] += 1
    cnt[r] -= 1
for i in xrange(n):
    cnt[i + 1] += cnt[i]
cnt.pop()
cnt.sort()
print sum(x * y for x, y in zip(a, cnt))
