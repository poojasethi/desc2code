ri = lambda: raw_input().strip()

#n = int(ri())
n, q = map(int, ri().split())
ai = map(int, ri().split())

ai.sort(reverse = True)

#print ai

pos = [0 for x in range(n+1)]

for x in xrange(1, q+1):
    l, r = map(int, ri().split())
    pos[l-1] += 1
    pos[r] -= 1

for x in xrange(1, len(pos)):
    pos[x] += pos[x-1]

pos.sort(reverse = True)

#print pos

total = 0
for x in xrange(0, n):
    total += ai[x] * pos[x]

print total
