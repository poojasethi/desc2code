def main():
    n,q = map(int,raw_input().split())
    a = list(map(int,raw_input().split()))

    a.sort(reverse=True)
    d = [0]*(n+1)
    res = 0
    for _ in range(q):
        l,r = map(int,raw_input().split())
        d[l-1] += 1
        d[r] -= 1
    now = 0
    for i in range(n):
        d[i] += now
        now = d[i]
    d.sort(reverse=True)
    for i in range(n):
        res += a[i] * d[i]
    print(res)

main()
