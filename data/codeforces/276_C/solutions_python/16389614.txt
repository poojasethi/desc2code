n ,q = map(int,raw_input().split())
a = map(int,raw_input().split())
b =[0]*(n+1)
c = [0]*n
for i in xrange(q):
	x,y = map(int,raw_input().split())
	b[x-1] +=1
	b[y] -= 1
#print "b",b
count = b[0]
for i in xrange(n):
	c[i] = count
	count = count + b[i+1]
#print "c", c
c.sort()
a.sort()
sum = 0
for i in xrange(n):
	sum += a[i]*c[i]
print sum
