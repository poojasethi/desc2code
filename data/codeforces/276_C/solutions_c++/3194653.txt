#include <iostream>
#include <algorithm>
using namespace std;
int a[200010],b[200010];
int main()
{
	ios_base::sync_with_stdio(0);
	int n, q; cin>>n>>q;
	for(int i=0;i<n;++i) cin>>a[i];
	sort(a, a+n);
	while(q--) {
		int l, r; cin>>l>>r;
		b[--l]++;
		b[r]--;
	}
	for(int i=1;i<n;++i) b[i]+=b[i-1];
	sort(b, b+n);

	long long res=0;
	for(int i=0;i<n;++i) res+=1LL*a[i]*b[i];
	cout<<res<<endl;

	return 0;
}
