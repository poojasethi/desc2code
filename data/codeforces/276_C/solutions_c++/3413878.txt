#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,q,l,r;
    long long a[200002]={},b[200002]={},t(0);
    cin>>n>>q;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    for(int i=0;i<q;i++)
        cin>>l>>r,b[l]++,b[r+1]--;
    for(int i=1;i<=n;i++)
        b[i]+=b[i-1];
    sort(a,a+n+1),sort(b,b+n+1);
    for(int i=1;i<=n;i++)
        t+=a[i]*b[i];
    cout<<t<<endl;
    return 0;
}
