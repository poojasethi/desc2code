#include<bits/stdc++.h>

using namespace std;
long long int a[2000006],n,m,x,y,sayac,i,j,k,t,h[200006],arr[200006];
int main(){
	scanf("%lld %lld",&n,&m);
	for(i=1;i<=n;i++)
		scanf("%lld",&a[i]);
	
	
	for(i=1;i<=m;i++){
		scanf("%lld %lld",&x,&y);
		h[x]++,h[y+1]--;
	}

	for(i=1;i<=n;i++)
		arr[i]=arr[i-1]+h[i];
	
	sort(arr+1,arr+n+1);
	sort(a+1,a+n+1);
	for(i=1;i<=n;i++){
		t+=a[i]*arr[i];
	}
	printf("%lld",t);
		
	return 0;
}
