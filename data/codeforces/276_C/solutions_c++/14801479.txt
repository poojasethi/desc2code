#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define s(n) scanf("%llu",&n)
#define p(n) printf("%llu ",n)
ll a[200005],f[200005];
int main()
{ll n,i,q,l,r,sum=0;
s(n);s(q);
for(i=1;i<=n;++i) s( a[i] );
sort( a,a+n+1,greater<ll>());
for(i=1;i<=q;++i){ s(l);s(r); f[l]++; if(r+1<=n) f[r+1]--;}
for(i=1;i<=n;++i) f[i]+=f[i-1]; sort(f,f+n+1,greater< ll >());
for(i=0;i<n;++i) sum+=f[i]*a[i];
cout<<sum;
}
