#include<stdio.h>
#include <algorithm>
using namespace std;

int n,q,l,r;
__int64  a[200010],t[200010],ans;

int main(){
	int i;
	scanf("%d%d",&n,&q);
	for(i=0;i<n;i++)
		scanf("%I64d",&a[i]);
	sort(a,a+n);

	for(i=0;i<q;i++)
	{
		scanf("%d%d",&l,&r);
		t[l]++,t[r+1]--;
	}
	for(i=1;i<=n;i++)
		t[i]+=t[i-1];
	sort(t+1,t+n+1);

	for(i=1;i<=n;i++)
		ans+=a[i-1]*t[i];
	printf("%I64d\n",ans);
	return 0;
}
