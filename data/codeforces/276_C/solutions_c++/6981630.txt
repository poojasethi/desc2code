#include <bits/stdc++.h>
using namespace std;
const int NMAX=2e5+1;
long long n,q,A[NMAX],flags[NMAX]={0},NA[NMAX];
int main() {
	cin>>n>>q;
	for(int i=0;i<n;++i) {
		cin>>A[i];
	}
	sort(A,A+n);
	while(q--) {
		int l,r;
		cin>>l>>r;
		l--,r--;
		flags[l]++,flags[r+1]--;
	}
	long long V=0;
	for(int i=0;i<n;++i) {
		V+=flags[i];
		NA[i]=V;
	}
	sort(NA,NA+n);
	long long ans=0;
	for(int i=0;i<n;++i) {
		ans+=(NA[i]*A[i]);
	}
	cout<<ans<<"\n";
	return 0;
}
