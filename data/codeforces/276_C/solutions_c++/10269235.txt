#include<iostream>
#include<cstdio>

#include<algorithm>
using namespace std;
long long i,k,d,n,m,a[200005],b[200005],mx; //n-elementebis raod m-moqmedebebis  raod.
int main()
{
scanf("%d%d",& n , & m);
for(i=0;i<n;i++)
scanf("%d", & a[i]);
sort(a,a+n);
for(i=0;i<m;i++)
{scanf("%d%d", & k, & d);
b[k-1]++;
b[d]--;}
for(i=1;i<n;i++)
b[i]+=b[i-1];
sort(b,b+n);
for(i=0;i<n;i++)
{
mx+=b[i]*a[i];}
cout<<mx;
//system("pause");
return 0;}
