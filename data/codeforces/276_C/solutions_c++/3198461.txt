 #include <iostream>
  #include <algorithm>
 #include <stdio.h>
using namespace std;
int main()
{
    int a[300005]={},b[300005]={},n,q;
long long ans=0;
cin>>n>>q;
for(int i=0; i<n; i++) cin>>a[i];
sort(a,a+n);
for(int i=0; i<q; i++)
{
  int x,y; cin>>x>>y;   b[x-1]++; b[y]--;
}
for(int i=1; i<n; i++) b[i]+=b[i-1];
sort(b,b+n);
for(int i=0; i<n; i++)  ans+=(1LL*a[i]*b[i]);
cout<<ans;

}
