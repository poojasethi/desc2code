#include<stdio.h>
#include <algorithm>

int main()
{
long long A[200010]={0},B[200010]={0};
int n,q,l,r;
long long ans=0;
scanf("%d%d",&n,&q);
for(int i=0;i<n;i++)
scanf("%d",&A[i]);
for(int i=1;i<=q;i++)
{
    scanf("%d%d",&l,&r);
    B[l-1]++;
    B[r]--;
}
for(int i=1;i<n;i++)
B[i]+=B[i-1];
std::sort(A,A+n);
std::sort(B,B+n);
for(int i=0;i<n;i++)
ans=ans+A[i]*B[i];
printf("%I64d",ans);



 }
