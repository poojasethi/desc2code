#include<iostream>
#include<algorithm>
using namespace std;
long long n,q,i,s,l,r;
long long a[2000005],t[2000005],in[2000005];
int main()
{
  s = 0;

  cin>>n>>q;

  for(i=1;i<=n;i++)
    cin>>a[i];

  for(i=1;i<=q;i++)
  {
    cin>>l>>r;
    t[l]++;
    t[r+1]--;
  }

  for(i=1;i<=n;i++)
    in[i] = in[i-1] + t[i];

  sort(a+1,a+n+1);
  sort(in+1,in+n+1);

  for(i=1;i<=n;i++)
   s += a[i]*in[i];

  cout<<s<<endl;
}
