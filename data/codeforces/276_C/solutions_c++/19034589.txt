#include <iostream>
#include <algorithm>
#include <stdio.h>
using namespace std;
long long n,m,x,y,q,ans,a[200005],D[200005];
int main(){
     cin>>n>>q;
   for(int i=0;i<n;i++)   scanf("%d",a+i);
   for(int i=0;i<q;i++)scanf("%d%d",&x,&y),D[x-1]++,D[y]--;
   for(int i=0;i<n;i++)D[i]+=D[i-1];
   sort(a,a+n);
   sort(D,D+n);
   for(int i=0;i<n;i++)ans+=a[i]*D[i];
   cout<<ans;
  
}
