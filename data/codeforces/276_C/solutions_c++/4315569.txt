#include <iostream>
#include<algorithm>
#define N 200005
using namespace std;
int n,q,i,j;
long long x[N],cnt[N],sol;
int main()
{
    cin>>n>>q;
    for(i=1;i<=n;i++)cin>>x[i];
    sort(x+1,x+n+1);
    for(;q;q--)
    {
        cin>>i>>j;
        cnt[i]++;
        cnt[j+1]--;
    }
    for(i=1;i<=n;i++)cnt[i]+=cnt[i-1];
    sort(cnt+1,cnt+n+1);
    for(i=1;i<=n;i++)sol+=cnt[i]*x[i];
    cout<<sol;

    return 0;
}
