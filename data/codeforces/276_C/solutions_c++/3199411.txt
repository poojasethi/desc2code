#include <iostream>
#include <algorithm>
using namespace std;
#define N 200005
int a[N],b[N],n,q;
int main(){
    cin>>n>>q;for(int i=0;i<n;i++) cin>>a[i];sort(a,a+n);
    int x,y;
    while(q--){
        cin>>x>>y;
        b[x]++,b[y+1]--;
    }
    for(int i=1;i<=n;i++) b[i]+=b[i-1];sort(b+1,b+n+1);
    long long c=0;
    for(int i=0;i<n;i++) c+=1LL*a[i]*b[i+1];cout<<c<<endl;

}
