#include<iostream>
#include<algorithm>
using namespace std;
long long  a[200005],use[200005];
int n,m;
int p,q;
int main()
{
	cin>>n>>m;
	for(int i=0;i<n;i++)
	cin>>a[i];
	for(int i=0;i<m;i++)
	{
		cin>>p>>q;
		use[p-1]++;
		use[q]--;
	}
	for(int i=1;i<n;i++)
	use[i]+=use[i-1];
	sort(use,use+n);
	sort(a,a+n);
	long long sum=0;
	for(int i=0;i<n;i++)
	sum+=use[i]*a[i];
	cout<<sum;
	//system("pause");
	return 0;
}
