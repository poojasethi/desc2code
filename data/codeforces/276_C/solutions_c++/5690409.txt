#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
int v=0,n,q,l,r,i;
cin>>n>>q;
int a[n],b[n];
for(i=0;i<n;i++) {b[i]=0;cin>>a[i];}
sort(a,a+n);
while(q--)
{
 cin>>l>>r;
 b[l-1]++;
if(n-r) b[r]--;
}
for(i=0;i<n;i++)
{
v+=b[i];
b[i]=v;
}
sort(b,b+n);
long long ans=0;
for(i=0;i<n;i++)
ans+=(a[i]*1LL*b[i]);
cout<<ans;
return 0;
}
