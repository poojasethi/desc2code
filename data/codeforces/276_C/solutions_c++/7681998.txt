#include<cstdio>
#include<algorithm>
using namespace std;
int a1[200000],a2[200001];
#define ll long long
int main()
{
	int n,q,a,b;ll o=0;
	scanf("%d%d",&n,&q);
	for(int i=0;i<n;++i)
		scanf("%d",a1+i);
	sort(a1,a1+n);
	while(q--)
	{
		scanf("%d%d",&a,&b);
		--a2[a-1],++a2[b];
	}
	for(int i=n;i;--i)
		a2[i-1]+=a2[i];
	sort(a2+1,a2+n+1);
	for(int i=0;i<n;++i)
		o+=(ll)a1[i]*a2[i+1];
	printf("%I64d\n",o);
}
