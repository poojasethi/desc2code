#include <iostream>
#include <algorithm>
using namespace std;

int n,q,l,r;
long long a[200010],t[200010],ans;

int main(){
    cin>>n>>q;
    for(int i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    for(int i=0;i<q;i++){
        cin>>l>>r;
        t[l-1]++,t[r]--;
    }
    for(int i=1;i<n;i++)
        t[i]+=t[i-1];
    sort(t,t+n);
    for(int i=0;i<n;i++)
        ans+=a[i]*t[i];
    cout<<ans;
    return 0;
}
