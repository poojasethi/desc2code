#include <bits/stdc++.h>
using namespace std;
const int N = 2e5+5;
#define ll long long
int add[N],ar[N],pp[N],mm[N];
int main(){
	int q,n,i,a,b;cin>>n>>q;
	for(i=1;i<=n;++i)scanf("%d",&ar[i]);
	for(i=0;i<q;++i)
		scanf("%d%d",&a,&b),mm[a]++,pp[b]++;
	int cnt = 0;
	for(i=n;i>=1;--i){
		cnt += pp[i];
		add[i]+=cnt;
		cnt-=mm[i];
	}
	sort(add+1,add+1+n);
	sort(ar+1,ar+1+n);
	ll ret=0,x;
	for(i=1;i<=n;++i)
		x= add[i],x*= ar[i],ret+=x;
	cout<<ret<<endl;
}
