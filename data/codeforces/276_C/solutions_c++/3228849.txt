#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
long long n,q,x,y,i,m[200005],k[200005],sum,j,t;
main()
{
	cin>>n>>q;
	for(i=1; i<=n; i++)
	{
		cin>>k[i];
	}
	for(i=1; i<=q; i++)
	{
		cin>>x>>y;
		m[x]++;
		m[y+1]--;	
	}
	for(i=1; i<=n; i++)
	{
		t+=m[i];
		m[i]=t;
	}
	sort(m,m+n+1);
	sort(k,k+n+1);
	for(i=1; i<=n; i++)
	{
		sum+=k[n+1-i]*m[n+1-i];
	}
	cout<<sum;
}
