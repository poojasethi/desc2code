#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
long long i,n,m,x,s,y,b[200010],a[200010];
main()
{
cin>>n>>m;
for (i=1; i<=n; i++)
cin>>a[i];
for (i=1; i<=m; i++)
{
cin>>x>>y;
b[y+1]--;
b[x]++;
}

for (i=2; i<=n+1; i++)
b[i]+=b[i-1];


sort(a+1,a+n+1);
sort(b+1,b+n+2);

for (i=n; i>=1; i--)
s+=a[i]*b[i+1];

cout<<s;
//system("pause");      
}
