#include<iostream>
#include<algorithm>
using namespace std;
long long n,m,a[200005],b[200005],c[200005],i,j,k,p,d[200005];
int main()
{
    cin>>n>>m;
    for(i=1;i<=n;i++) cin>>a[i];
    for(i=1;i<=m;i++) { cin>>k>>p; d[k]++; d[p+1]--; }
    
    for(i=1;i<=n+1;i++) d[i]+=d[i-1];
    
    sort(d,d+n+1);
    sort(a,a+n+1);
    
    for(i=1;i<=n+1;i++) j+=a[i]*d[i];
    
    cout<<j;
    
    return 0;
}
