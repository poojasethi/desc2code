#include<bits/stdc++.h>
using namespace std;
long long int a[200003],h[200003],n,m,i,j,t,s,d,x,y;
int main()
{
	scanf("%lld %lld",&n,&m);
	for(i=0;i<n;i++)
		scanf("%lld",a+i);
	sort(a,a+n);
	for(i=0;i<m;i++)
	{
		scanf("%lld %lld",&x,&y);
		h[x-1]++;h[y]--;
	}
	for(i=0;i<n;i++)
	{t+=h[i];h[i]=t;}
	sort(h,h+n);
	for(i=0;i<n;i++)
		d+=h[i]*a[i];
	printf("%lld",d);
	return 0;
}
