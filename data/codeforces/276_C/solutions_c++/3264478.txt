#include <iostream>
#include <algorithm>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n,a[200010],q,s[200010],l,r;
long long res;

int main(){
  cin>>n>>q;
  rep(i,n)cin>>a[i];
  sort(a,a+n);
  rep(i,q){
    cin>>l>>r;
    s[l-1]++;
    s[r]--;
  }
  rep(i,n)s[i+1]+=s[i];
  sort(s,s+n);
  rep(i,n)res += (long long)a[i]*s[i];
  cout<<res<<endl;

}
