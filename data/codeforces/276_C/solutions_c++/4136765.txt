#include<iostream>
#include<algorithm>
long long b[200005],ans;
using namespace std;
int main()
{
    long long n,q,i,x,y;
    cin>>n>>q;
    long long a[n+2];
    for(i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    for(i=0;i<q;i++)
    {
        cin>>x>>y;
        b[x-1]++;
        b[y]--;
    }
    for(i=1;i<n;i++)
    b[i]=b[i]+b[i-1];
    sort(b,b+n);
    for(i=0;i<n;i++)
    ans+=a[i]*b[i];
    cout<<ans<<endl;

}
