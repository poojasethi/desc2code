#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
ll z,ans,q,n,i,x,y,a[200010],f[200010];
int main()
{
 scanf("%I64d%I64d",&n,&q);
 for (i=1;i<=n;i++)scanf("%I64d",&a[i]);
 sort(a+1,a+n+1);
 while (q--){scanf("%I64d%I64d",&x,&y);
  f[x]++;f[y+1]--;
            }
  z=n;
 for (i=1;i<=n;i++)
   f[i]+=f[i-1];
 sort(f+1,f+n+1);
 for (i=n;i>=1;i--)ans+=f[i]*a[z--];   
 printf("%I64d\n",ans);
return 0;
}
