#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long n,q,a[200005],b[200005]={0},d[200005]={0};
	scanf("%lld%lld",&n,&q);
	for(int i=1;i<=n;i++)scanf("%lld",&a[i]);
	for(int i=0;i<q;i++)
	{
		long long l,r;
		scanf("%lld%lld",&l,&r);
		d[l]++;
		d[r+1]--;
	}
	for(int i=1;i<=n;i++)
		b[i]=b[i-1]+d[i];
	sort(a+1,a+n+1);
	sort(b+1,b+n+1);
	long long ans=0;
	for(int i=1;i<=n;i++)
		ans+=a[i]*b[i];
	cout << ans << endl;
	return 0;
}
