#include <bits/stdc++.h>
#define FOR(i, j, k) for (int i = j; i <= k; i ++)
#define FORD(i, j, k) for (int i = j; i >= k; i --)
#define mn 200005
#define maxC 1000000007
#define base 100007
#define pb push_back

using namespace std;
int n, m, q, mt[1003][1003], u[mn], v[mn];
int l[mn], r[mn], s[mn], t[mn];
bool ans[mn];
vector<int> qid[mn];

void setup()
{
    cin >> n >> m >> q;
    FOR(i, 1, m)
        cin >> u[i] >> v[i];
    FOR(i, 1, q)
    {
        cin >> l[i] >> r[i] >> s[i] >> t[i];
        qid[r[i]].pb(i);
    }
}

void xuly()
{
    FOR(i, 1, m)
    {
        FOR(j, 1, n)
            mt[u[i]][j] = mt[v[i]][j] = max(mt[u[i]][j], mt[v[i]][j]);
        mt[v[i]][u[i]] = mt[u[i]][v[i]] = i;
        FOR(j, 0, int(qid[i].size()) - 1)
        {
            int u = qid[i][j];
            if (mt[t[u]][s[u]] >= l[u])
                ans[u] = true;
        }
    }
    FOR(i, 1, q)
        cout << (ans[i]? "Yes\n" : "No\n");
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(NULL);
    cout.tie(NULL);
    setup ();
    xuly ();
    return 0;
}
