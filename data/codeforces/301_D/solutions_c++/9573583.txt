#include <cstdio>
#include <vector>
#define repu(i,x,y) for (int i=x; i<=y; ++i)
using namespace std;

int n,m,a[200100],p[200100],bit[200100],cnt,l[200100],r,ans[200100];
vector<int> t[200100],q[200100];

void add(int x)
{
    for (; x<=n; x+=x&-x)
        ++bit[x];
}

int query(int x)
{
    int ret=0;
    for (; x; x-=x&-x)
        ret+=bit[x];
    return ret;
}

int main()
{
    scanf("%d%d",&n,&m);
    repu(i,1,n)
        scanf("%d",&a[i]),p[a[i]]=i;
    repu(i,1,n)
        for (int j=i; j<=n; j+=i)
            if (p[i]<p[j])
                t[p[j]].push_back(p[i]);
            else
                t[p[i]].push_back(p[j]);
    repu(i,1,m)
        scanf("%d%d",&l[i],&r),q[r].push_back(i);
    repu(i,1,n)
    {
        repu(j,0,int(t[i].size())-1)
            ++cnt,add(t[i][j]);
        repu(j,0,int(q[i].size())-1)
            ans[q[i][j]]=cnt-query(l[q[i][j]]-1);
    }
    repu(i,1,m)
        printf("%d\n",ans[i]);
    return 0;
}
