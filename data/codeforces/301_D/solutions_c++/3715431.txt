#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define N 210000
using namespace std;
struct ppp{
	int x,y,id;
} a[N];
int pos[N],tree[N],x,i,tmp,ans[N],n,m,head[N],e[N*18],next[N*18],j,len,k,l;
bool cmp(const ppp&a,const ppp&b){
	return a.x<b.x;
}
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i)scanf("%d",&x),pos[x]=i;
	for(i=1;i<=n;++i)
		for(j=i;j<=n;j+=i){
			int x=pos[i],y=pos[j];
			if(x>y)swap(x,y);
			next[++len]=head[x];
			head[x]=len;
			e[len]=y;
		}
	for(i=1;i<=m;++i)scanf("%d%d",&a[i].x,&a[i].y),a[i].id=i;
	sort(a+1,a+m+1,cmp);
	l=m;
	for(i=n;i>=1;--i){
		for(j=head[i];j;j=next[j])
			for(k=e[j];k<=n;k+=k&(-k)) tree[k]++;
		while(l>=1 && a[l].x==i){
			int tmp=0;
			for(k=a[l].y;k;k-=k&(-k)) tmp+=tree[k];
			ans[a[l].id]=tmp;
			l--;
		}
	}
	for(i=1;i<=m;++i)printf("%d\n",ans[i]);
//	system("pause");
}
