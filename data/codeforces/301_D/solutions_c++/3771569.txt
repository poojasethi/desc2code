#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int n, m, p[200001], l[200000], r[200000], ans[200000], c[200001];
vector<int> a[200001], b[200001];

int main() {
  cin >> n >> m;
  for (int i(1); i <= n; ++i) {
    int x;
    cin >> x;
    p[x] = i;
  }
  for (int i(0); i < m; ++i) {
    cin >> l[i] >> r[i];
    b[l[i]].push_back(i);
  }
  for (int i(1); i <= n; ++i)
    for (int j(i); j <= n; j += i) {
      a[min(p[i], p[j])].push_back(max(p[i], p[j]));
      ++c[max(p[i], p[j])];
    }
  for (int i(1); i <= n; ++i)
    c[i] += c[i - 1];
  for (int i(n); i >= 1; --i)
    c[i] -= c[i ^ i & -i];
  for (int i(1); i <= n; ++i) {
    for (vector<int>::iterator j(b[i].begin()); j != b[i].end(); ++j)
      for (int k(r[*j]); k; k ^= k & -k)
        ans[*j] += c[k];
    for (vector<int>::iterator j(a[i].begin()); j != a[i].end(); ++j)
      for (int k(*j); k <= n; k += k & -k)
        --c[k];
  }
  for (int i(0); i < m; ++i)
    cout << ans[i] << endl;
}
