#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
const int n1=200010,n2=n1*20;
struct ww {
    int x,y,id;
} a[n2],q[n1];
int i,j,k,n,m,t,x,y;
int b[n1],c[n1],tr[n1],re[n1];
inline bool cc1(const ww &a,const ww &b) {
    return a.y<b.y;
}
inline void inc(int x) {
    for (;x<=n;x+=x&-x) tr[x]++;
}
inline int calc(int x) {
    int an=0;
    for (;x;x-=x&-x) an+=tr[x];
    return an;
}
int main() {
    scanf("%d%d",&n,&m);
    For(i,1,n) scanf("%d",&b[i]),c[b[i]]=i;
    For(i,1,n) {
        for (j=b[i];j<=n;j+=b[i]) {
            a[++t].x=i,a[t].y=c[j];
            if (a[t].x>a[t].y) swap(a[t].x,a[t].y);
        }
    }
    sort(a+1,a+t+1,cc1);
    For(i,1,m) scanf("%d%d",&q[i].x,&q[i].y),q[i].id=i;
    sort(q+1,q+m+1,cc1);
    j=0;
    For(i,1,m) {
        for (;j<t&&a[j+1].y<=q[i].y;j++) inc(n-a[j+1].x+1);
        re[q[i].id]=calc(n-q[i].x+1);
    }
    For(i,1,m) printf("%d\n",re[i]);
    return 0;
}
