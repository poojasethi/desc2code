#include <cstdio>
#include <vector>
using namespace std;
const int MAX = 200005;
vector<pair<int, int> > end[MAX];
vector<int> divi[MAX];
int ind[MAX], fen[MAX], ans[MAX], n, q;
void add(int p)
{
	for (int i = p; i <= n; i += (i & -i))
		fen[i]++;
}
int sum(int p)
{
	int ans = 0;
	for (int i = p; i > 0; i -= (i & -i))
		ans += fen[i];
	return ans;
}
int main()
{
	scanf("%d %d", &n, &q);
	for (int i = 1; i <= n; i++)
	{
		int val;
		scanf("%d", &val);
		ind[val] = i;
	}
	for (int i = 1; i <= n; i++)
		for (int j = i; j <= n; j += i)
			if (ind[i] < ind[j])
				divi[ind[j]].push_back(ind[i]);
			else
				divi[ind[i]].push_back(ind[j]);
	for (int i = 0; i < q; i++)
	{
		int l, r;
		scanf("%d %d", &l, &r);
		end[r].push_back(make_pair(i, l));
	}
	for (int i = 1; i <= n; i++)
	{
		for (int j = 0; j < divi[i].size(); j++)
			add(divi[i][j]);
		for (int j = 0; j < end[i].size(); j++)
			ans[end[i][j].first] = sum(i) - sum(end[i][j].second - 1);
	}
	for (int i = 0; i < q; i++)
		printf("%d \n", ans[i]);
	return 0;
}
