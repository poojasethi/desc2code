#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;

#define pb push_back
#define pr pair<int,int>
#define X first
#define Y second
#define mp make_pair
#define N 200010
int n,Q,pos[N],p[N];
vector <int> g[N];
vector < pr > q[N];
int an[N],bit[N],d[N];
int main(){
	scanf("%d%d",&n,&Q);
	for(int i = 1;i <= n;i ++)	scanf("%d",&p[i]), pos[p[i]] = i;
	int u,v;
	for(int i = 1;i <= n;i ++){
		for(int j = i;j <= n;j += i){
			u = pos[i], v = pos[j];
			if(u > v)	swap(u,v);
			d[v] ++;
			g[u].pb(v);
		}
	}
	for(int i = 1;i <= n;i ++)	for(int j = i;j <= n;j += j & -j)	bit[j] += d[i];
	for(int i = 1;i <= Q;i ++){
		scanf("%d%d",&u,&v);
		q[u].pb(mp(v,i));
	}
	for(int i = 1;i <= n;i ++){
		for(int j = 0;j < q[i].size();j ++)	for(int k = q[i][j].X;k;k -= k & -k)	an[q[i][j].Y] += bit[k];
		for(int j = 0;j < g[i].size();j ++)	for(int k = g[i][j];k <= n;k += k & -k)	bit[k] --;
	}
	for(int i = 1;i <= Q;i ++)	printf("%d\n",an[i]);
	return 0;
}
