#include <iostream>
#include<cstring>
#include<cstdio>
#include<vector>
#include<algorithm>
#define pb push_back
#define mp make_pair
using namespace std;
const int MAXN=200010;
vector<int> to1[MAXN],to2[MAXN];
int n,m,l[MAXN],r[MAXN],id[MAXN],a[MAXN],c[MAXN],ans[MAXN];
void add(int x)
{
	for(;x<=n;x+=x&(-x))
		c[x]++;
}
int getsum(int x)
{
	int sum=0;
	for(;x;x-=x&(-x))
		sum+=c[x];
	return sum;
}
int main()
{
	int i,j,x;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		id[a[i]]=i;
	}
	for(i=1;i<=m;i++)
	{
		scanf("%d%d",&l[i],&r[i]);
		to1[l[i]].pb(i);
		to2[r[i]].pb(i);
	}
	for(i=1;i<=n;i++)
	{
		for(j=0;j<(int)to1[i].size();j++)
			ans[to1[i][j]]-=getsum(r[to1[i][j]])-getsum(i-1);
		for(j=1;j*a[i]<=n;j++)
			add(id[j*a[i]]);
		for(j=0;j<(int)to2[i].size();j++)
			ans[to2[i][j]]+=getsum(i)-getsum(l[to2[i][j]]-1);
	}
	for(i=1;i<=m;i++)
		printf("%d\n",ans[i]);
}
