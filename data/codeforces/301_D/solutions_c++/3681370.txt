#include <cstdio>
#include <vector>
#include <utility>
#define FOR(i,n) for (__typeof(n.begin())i=n.begin();i!=n.end();++i)
using namespace std;

const int MAXN=200020;
vector< pair<int,int> > v[MAXN],q[MAXN];
int n,m,x,y,a[MAXN],b[MAXN],t[MAXN];
long long ans[MAXN];

void Push(int x,int y) {
	x=b[x]; y=b[y];
	if ((!x)||(!y)) return;
	if (x>y) swap(x,y);
	v[1].push_back(make_pair(y,1));
	v[x+1].push_back(make_pair(y,-1));
}

int main() {
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i) scanf("%d",&a[i]),b[a[i]]=i;
	for (int i=1;i<=n;++i)
		for (int j=a[i];j<=n;j+=a[i]) Push(a[i],j);
	for (int i=1;i<=m;++i) {
		scanf("%d%d",&x,&y);
		q[x].push_back(make_pair(y,i));
	}
	for (int i=1;i<=n;++i) {
		FOR(p,v[i])
			for (int j=(*p).first;j<=n;j+=(j&-j)) t[j]+=(*p).second;
		FOR(p,q[i]) {
			long long tmp=0;
			for (int j=(*p).first;j;j-=j&(-j)) tmp+=(long long)t[j];
			ans[(*p).second]=tmp;
		}
	}
	for (int i=1;i<=m;++i) printf("%I64d\n",ans[i]);
	return 0;
}
