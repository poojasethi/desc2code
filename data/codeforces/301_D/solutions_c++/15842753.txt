//In the Name of God 

#include<bits/stdc++.h>
#define pii pair<int,int>
#define val first
#define id second
#define pb push_back

using namespace std;

const int maxn=200007;

int n,q;
int pos[maxn];
int s[maxn];
vector<int> edge[maxn];
vector<pii> query[maxn];
int ans[maxn];

void add(int i,int x=1)
{
  for(i++;i<maxn;i+=i&-i)
    s[i]+=x;
}

int get(int i)
{
  int ret=0;
  for(i++;i;i-=i&-i)
    ret+=s[i];
  return ret;
}

int main()
{
  ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
  cin>>n>>q;
  for(int p,i=0;i<n;i++)
    {
      cin>>p;
      p--;
      pos[p]=i;
    }
  for(int i=1;i<=n;i++)
    for(int j=i;j<=n;j+=i)
      if(pos[i-1]<pos[j-1])
	edge[pos[i-1]].pb(pos[j-1]);
      else
	edge[pos[j-1]].pb(pos[i-1]);
  for(int l,r,i=0;i<q;i++)
    {
      cin>>l>>r;
      l--,r--;
      query[l].pb(pii(r,i));
    }
  for(int i=n;i--;)
    {
      for(int j=0;j<edge[i].size();j++)
	add(edge[i][j]);
      for(int j=0;j<query[i].size();j++)
	ans[query[i][j].id]=get(query[i][j].val);
    }
  for(int i=0;i<q;i++)
    cout<<ans[i]<<endl;
  return 0;
}
