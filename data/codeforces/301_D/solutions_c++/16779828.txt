#include<bits/stdc++.h>

using namespace std;

#define pr pair < int , pair < int , int > >
#define sgmn int v, int tl, int tr
#define tm  (tl + tr >> 1)
#define sol v*2, tl, tm
#define sag v*2+1, tm+1, tr
#define mp make_pair
#define st first
#define nd second
#define N 200005

vector < pair < int , int > > Q[N]; 

vector < int > V[N];

int A[N],W[N],F[N],T[N],n,q,i,j,x,y,t;

void up(){
	for(; y <= n ; y += y & -y)
		F[y] ++;
}

void qry(){
	t = 0;
	for(; y ; y -= y & -y)
		t += F[y];
}

int main(){

	scanf("%d%d",&n,&q);

	for(i=1;i<=n;i++)
		for(j=i+i;j<=n;j+=i)
			V[j].push_back(i);

	for(i=1;i<=n;i++){
		scanf("%d",&A[i]);
		W[ A[i] ] = i;
	}

	for(i=1;i<=q;i++){
		scanf("%d%d",&x,&y);
		Q[y].push_back( make_pair(x,i) );
	}

	for(i=1;i<=n;i++){

		x = A[i];

		for(j=0;j<V[x].size();j++)
			if(W[ V[x][j] ] <= i)
				y = W[ V[x][j] ], up();

		for(j=x;j<=n;j+=x)
			if(W[j] <= i)
				y = W[j], up();
		
		for(j=0;j<Q[i].size();j++){

			y = i;                qry();  T[ Q[i][j].second ] =  t;
			y = Q[i][j].first-1;  qry();  T[ Q[i][j].second ] -= t;

		}
	}

	for(i=1;i<=q;i++)
		printf("%d\n",T[i]);

   return 0;
}
