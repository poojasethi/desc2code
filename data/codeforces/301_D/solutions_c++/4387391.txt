#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn = 200100;
int v[maxn],p[maxn],rp[maxn],l[maxn],r[maxn],th[maxn],ans[maxn];
int cmp(int i,int j){
	return r[i]<r[j];
}
void add(int x){
	for(;x>0;x-=x&(-x))v[x]++;
}
int sum(int x,int n){
	int ans=0;
	for(;x<=n;x+=x&(-x))ans+=v[x];
	return ans;
}
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%d",&p[i]);
	}
	for(int i=0;i<m;i++){
		th[i]=i;
		scanf("%d%d",&l[i],&r[i]);
	}
	sort(th,th+m,cmp);
	int k=0;
	for(int i=1;i<=n;i++){
		rp[p[i]]=i;
		int t=p[i],j;
		for(j=1;j*j<t;j++){
			if(t%j!=0)continue;
			add(rp[j]);add(rp[t/j]);
		}
		if(j*j==t)add(rp[j]);
		for(j=2;j*t<=n;j++)add(rp[j*t]);
		while(k<m&&r[th[k]]==i){
			ans[th[k]]=sum(l[th[k]],i);
			k++;
		}
	}
	for(int i=0;i<m;i++)printf("%d\n",ans[i]);
}
