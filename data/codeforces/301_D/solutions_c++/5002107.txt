#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 200010
#define M 3000000
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int a,b,id;
} a[N];
int i,j,k,n,m,v,t;
int b[N],c[N],tr[N],head[N],d[M],next[M],an[N];
inline bool cc1(const ww &a,const ww &b) {
	return a.b<b.b;
}
inline void add(int x,int y) {
	d[++t]=y,next[t]=head[x],head[x]=t;
}
inline void inc(int x) {
	x=n-x+1;
	for (;x<=n;x+=x&-x) tr[x]++;
}
inline int calc(int x) {
	x=n-x+1;
	int an=0;
	for (;x;x-=x&-x) an+=tr[x];
	return an;
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n) scanf("%d",&b[i]),c[b[i]]=i;
	For(i,1,m) scanf("%d%d",&a[i].a,&a[i].b),a[i].id=i;
	sort(a+1,a+m+1,cc1);
	k=1;
	For(i,1,n) {
		for (j=1;j*j<=b[i];j++) if (b[i]%j==0) {
			if (c[j]<=i) inc(c[j]);
			else add(c[j],i);
			if (j*j<b[i]) {
				int A=b[i]/j;
				if (c[A]<=i) inc(c[A]);
				else add(c[A],i);
			}
		}
		for (v=head[i];v;v=next[v]) inc(d[v]);
		for (;k<=m&&a[k].b==i;k++) an[a[k].id]=calc(a[k].a);
	}
	For(i,1,m) printf("%d\n",an[i]);
	//for(;;);
	return 0;
}
