#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
#define rep(i,n) for (int i=1,_=n;i<=_;i++)
#define N 200005
vector<int> v[N];
int n,m,x,y,Q,a[N],c[N],d[N],r[N],w[N],ans[N],son[N],next[N];
int main()
{
	scanf("%d%d",&n,&m); rep(i,n) scanf("%d",a+i),w[a[i]]=i;
	rep(i,n) for (int j=i;j<=n;j+=i){
		x=w[i],y=w[j]; if (x>y) swap(x,y);
		d[y]++; v[x].push_back(y);
		}
	rep(i,n) for (x=i;x<=n;x+=x&-x) c[x]+=d[i];
	rep(i,m) scanf("%d%d",&x,r+i),next[i]=son[x],son[x]=i;
	rep(i,n){
		for (int j=son[i];j;j=next[j])
			for (x=r[j];x;x-=x&-x) ans[j]+=c[x];
		rep(j,v[i].size())
			for (x=v[i][j-1];x<=n;x+=x&-x) c[x]--;
		}
	rep(i,m) printf("%d\n",ans[i]); return 0;
}
