#include<iostream>
#include<vector>
using namespace std;
const int N=200010;
int l[N],r[N],c[N],p[N],ans[N],a[N],n,m;
vector<int> L[N],R[N];
void upd(int pos){
	for(;pos<=n;pos+=(pos&-pos))c[pos]++;
}
int read(int pos){
	int ret=0;
	while(pos>0){
		ret+=c[pos];
		pos-=(pos&-pos);
	}
	return ret;
}
int main(){
	cin.sync_with_stdio(false);
	cin.tie(0);
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		cin>>a[i];
		p[a[i]]=i;
	}
	for(int i=1;i<=m;i++){
		cin>>l[i]>>r[i];
		L[l[i]].push_back(i);
		R[r[i]].push_back(i);
	}
	for(int i=1;i<=n;i++){
		int j;
		for(j=0;j<L[i].size();j++)
			ans[L[i][j]]-=read(r[L[i][j]])-read(i-1);
		j=1;
		while(j*a[i]<=n){
			upd(p[j*a[i]]);
			j++;
		}
		for(j=0;j<R[i].size();j++)
			ans[R[i][j]]+=read(i)-read(l[R[i][j]]-1);
	}
	for(int i=1;i<=m;i++)
		cout<<ans[i]<<'\n';
	return 0;
}
