#include <iostream>
#include <vector>
using namespace std;
#define N 222222
#define pb push_back
#define forup(i,a,n) for(int i=a;i<=n;i++)
int O[N],a[N],B[N],w[N],tt=0;
vector<int> q[N],A[N];
void add(int x) { tt++; while(x<N) { B[x]++; x += (x&-x); }}
int get(int x) { int r=0; while(x) { r+=B[x]; x-=(x&-x);}return r;}
int main() {
    int n,Q,l,r;
    cin>>n>>Q;
    forup(i,1,n) { cin>>a[i]; w[a[i]]=i; }
    forup(i,1,Q) { cin>>l>>r; q[r].pb(l); q[r].pb(i); }
    forup(i,1,n) {
        for(int j=a[i];j<=n;j+=a[i]) if(w[j]<=i) add(w[j]); else A[w[j]].pb(i);
        forup(j,0,(int)A[i].size()-1) add(A[i][j]);
        forup(j,0,(int)q[i].size()-1) O[q[i][j+1]] = tt-get(q[i][j++]-1);
    }
    forup(i,1,Q) cout<<O[i]<<endl;
}
