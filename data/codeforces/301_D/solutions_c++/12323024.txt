typedef long long ll;
using namespace std;
const int mx = 200005;
int l[mx],r[mx],a[mx],c[mx],p[mx],ans[mx];
vector <int> L[mx],R[mx];
int n;
void add(int x)
{
    for(; x <= n; x += x&-x) c[x]++;
}

int get(int x)
{
    int sum = 0;
    for(; x >= 1; x -= x&-x) sum += c[x];
    return sum;
}

int main ()
{
    int i,j,k,m;
    scanf("%d%d",&n,&m);
    for(i = 1; i <= n; i++) scanf("%d",&a[i]),p[a[i]] = i;
    for(i = 1; i <= m; i++)
    {
        scanf("%d%d",&l[i],&r[i]);
        L[l[i]].push_back(i);
        R[r[i]].push_back(i);
    }

    for(i = 1;i <= n;i++)
    {
        for(j = 0;j < L[i].size();j++) ans[L[i][j]] -= get(r[L[i][j]]) - get(i - 1);
        for(j = a[i];j <= n;j += a[i]) add(p[j]);
        for(j = 0;j < R[i].size();j++) ans[R[i][j]] -= get(l[R[i][j]] - 1) - get(i);
    }
    for(i = 1;i <= m;i++) printf("%d\n",ans[i]);
}
