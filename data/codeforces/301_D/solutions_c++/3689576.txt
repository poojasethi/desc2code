#include<cstdio>
#include<cstring>
#include<cmath>
#include<vector>
#include<iostream>
#include<algorithm>
#define pb push_back
using namespace std;
const int N=200005;
int l[N],r[N],a[N],c[N],p[N],ans[N];
vector <int> L[N],R[N];
int n;
void add(int x)
{
    for(;x<=n;x+=x&-x) c[x]++;
}
int get(int x)
{
    int sum=0;
    for(;x;x-=x&-x) sum+=c[x]; 
    return sum;
}
int main()
{
	int i,j,m;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i) scanf("%d",&a[i]),p[a[i]]=i;
	for(i=1;i<=m;++i)
	{
		scanf("%d%d",&l[i],&r[i]);
		L[l[i]].pb(i);
		R[r[i]].pb(i);
	}
	for(i=1;i<=n;++i)
	{
        for(j=0;j<L[i].size();++j) ans[L[i][j]]-=get(r[L[i][j]])-get(i-1);
        for(j=1;j*a[i]<=n;++j) add(p[j*a[i]]);
        for(j=0;j<R[i].size();++j) ans[R[i][j]]+=get(i)-get(l[R[i][j]]-1); 
    }
    for(i=1;i<=m;++i) printf("%d\n",ans[i]);
    return 0;
}
