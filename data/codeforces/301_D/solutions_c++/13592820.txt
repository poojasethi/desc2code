#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;

const int MAX_N = 2000000 + 10;

struct info {
	int l, r, id;
}q[MAX_N];

vector<int> d[MAX_N];
int n, m, vis[MAX_N], ans[MAX_N], c[MAX_N];

bool cmp(info i, info j) {
	return i.l < j.l;
}

int main() {
	scanf("%d%d", &n, &m);
	for(int i = 1; i <= n; i++) {
		int x; scanf("%d", &x);
		vis[x] = i;
	}
	for(int i = 1; i <= n; i++)
		for(int j = i; j <= n; j += i) {
			int a = vis[i], b = vis[j];
			if(a > b) swap(a, b);
			d[a].push_back(b);
		}
	for(int i = 1; i <= m; i++) {
		scanf("%d%d", &q[i].l, &q[i].r);
		q[i].id = i;
	}
	sort(q + 1, q + m + 1, cmp);
	int p = m;
	for(int i = n; i >= 1; i--) {
		for(int j = 0; j < d[i].size(); j++)
			for(int k = d[i][j]; k <= n; k += k & -k)
				c[k]++;
		while(p >= 1 && q[p].l == i) {
			int res = 0;
			for(int j = q[p].r; j > 0; j -= j & -j)
				res += c[j];
			ans[q[p].id] = res;
			p--;
		}
	}
	for(int i = 1; i <= m; i++)
		printf("%d\n", ans[i]);
	return 0;
}
