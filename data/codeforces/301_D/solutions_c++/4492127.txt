#include <iostream>
#include <vector>
using namespace std;
const int N=200200;
int n,m,p[N],q[N],c[N],e[N],a[N];
vector<int>v[N],t[N];
main(){
  cin>>n>>m;
  for(int i=1;i<=n;++i)cin>>p[i],q[p[i]]=i;
  for(int i=1,l,r;i<=n;++i)for(int j=i;j<=n;j+=i){
    l=q[i],r=q[j];
    if(l>r)r^=l^=r^=l;
    ++c[r],v[l].push_back(r);
  }
  for(int i=n;i-->1;)for(int j=i+(i&-i);j<=n;j+=j&-j)c[j]+=c[i];
  for(int i=0,l;i<m&&cin>>l>>e[i];++i)t[l].push_back(i);
  for(int i=1;i<=n;++i){
    for(int j=0;j<t[i].size();++j){
      int u=t[i][j],s=0;
      for(int k=e[u];k>0;k-=k&-k)s+=c[k];
      a[u]=s;
    }
    for(int j=0;j<v[i].size();++j)for(int k=v[i][j];k<=n;k+=k&-k)--c[k];
  }
  for(int i=0;i<m;++i)cout<<a[i]<<endl;
}
