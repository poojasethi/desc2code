#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <vector>
#include <string.h>
#define N 200005
using namespace std;

#define LB(x) ((x) & (-x))
int a[N],c[N];
vector<int> p[N];
vector<pair<int,int> > q[N];
int idx[N];
int n;
int ins(int x,int delta) {
	for(int i=x;i<=n; i+=LB(i)) c[i] += delta; 		
}

int sum(int n) {
	int ret = 0;
	for(int i=n;i>0;i-=LB(i)) ret += c[i];
       return ret;	
}

int ans[N];

int main() {
	int m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;++i) {
		scanf("%d",a + i);
		idx[a[i]] = i;
	}
	
	for(int i=1;i<=n;++i) {
		for(int j=a[i];j<=n;j+=a[i]) {
			int k = idx[j];
			if(k > i) p[k].push_back(i);
			else p[i].push_back(k);
		}	
	}
	
	for(int i=0,l,r;i<m;++i) {
		scanf("%d%d",&l,&r);
		q[r].push_back(make_pair(i,l));
	}

	memset(c,0,sizeof(c));
	for(int i=1;i<=n;++i) {
		for(int j=0;j<p[i].size();++j) ins(p[i][j],1);
		for(int j=0;j<q[i].size();++j) {
			ans[q[i][j].first] = sum(i) - sum(q[i][j].second-1); 
		}
	}
	for(int i=0;i<m;++i) printf("%d\n",ans[i]);

	return 0;
}
