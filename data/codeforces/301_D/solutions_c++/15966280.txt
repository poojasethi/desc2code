#include <bits/stdc++.h>
using namespace std;
const int N = 5e5+5;
int p[N],r[N];
long long ret[N],fen[N];
vector<int> v[N],rig[N];
void update(int x){
	while(x<N){
		fen[x]++;
		x+=x&-x;
	}
}
long long get(int x){
	int R=0;
	for(;x;x=x&(x-1))
		R+=fen[x];
	return R;
}
int main(){
	int n,m,i,j,a,l;
	memset(ret,0,sizeof(ret));
	memset(fen,0,sizeof(fen));
	cin>>n>>m;
	for(i=1;i<=n;++i)scanf("%d",&a),p[a]=i;
	for(i=0;i<m;++i)
		scanf("%d%d",&l,&r[i]),rig[l].push_back(i);
	for(i=1;i<=n;++i)
		for(j=i;j<=n;j+=i)
			v[min(p[i],p[j])].push_back(max(p[i],p[j]));
	for(i=n;i>=1;--i){
		for(j=0;j<v[i].size();++j)update(v[i][j]);
		for(j=0;j<rig[i].size();++j)
			ret[rig[i][j]]= get(r[rig[i][j]]);
	}
	for(i=0;i<m;++i)
		cout<<ret[i]<<endl;
}
