#include<bits/stdc++.h>
using namespace std;
int main(){
   int n,m,ans=0;
   pair<int,int>a[1000];
   cin>>n>>m;
   for(int i=1;i<=m;i++){
      cin>>a[i].second>>a[i].first;
   }
   sort(a+1,a+m+1);
   for(int i=m;i>=1;i--){
      if(n>0){
         int k=min(n,a[i].second);
         ans+=(k*a[i].first);
         n-=k;
      }
      else break;
  }
  cout<<ans<<endl;
  return 0;
}
