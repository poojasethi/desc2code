#include <iostream>
#include <bits/stdc++.h>
#define ll long long int
using namespace std;
int main()
{
ll n,m;
cin >> n >> m;
pair <long long int,long long int > pr[m];
for(int i=0;i<m;i++)
    cin >> pr[i].second >> pr[i].first;
    ll ans=0;
    sort(pr,pr+m);
for(int i=m-1;i>=0 && n>0;i--)
{
ans+=pr[i].first*min(pr[i].second,n);
n-=min(pr[i].second,n);

}
cout << ans << endl;
}
