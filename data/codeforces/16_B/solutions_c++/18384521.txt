#include<bits/stdc++.h>
using namespace std;
vector<pair<int,int> >x;
int main()
{
    int n,m,sum=0; cin >> m >> n;
    for(int i=0;i<n;i++)
    {
        int a,b; cin >> b >> a;
        x.push_back(make_pair(a,b));
    }
    sort(x.begin(),x.end()); reverse(x.begin(),x.end());
    for(int i=0;i<n;i++)
        sum  += min(x[i].second,m) * x[i].first , m -= min(x[i].second,m);
    cout << sum << endl;
}
