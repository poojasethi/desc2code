#include <stdio.h>
#include <algorithm>
using namespace std;
int main()
{
	pair<int,int> v[20];
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=0;i<m;++i)
		scanf("%d%d",&v[i].second,&v[i].first);
	sort(v,v+m);
	int r=0;
	for(int i=m-1;i>=0;--i){
		r+=min(n,v[i].second)*v[i].first;
		n-=min(n,v[i].second);
	}
	printf("%d",r);
	return 0;
}
