#include<algorithm>
#include<iostream>
using namespace std;
struct mat{int x,y;}a[20];
bool tmp(mat a,mat b){return a.y>b.y;}
int main()
{
	int n,m,ans=0,now=0;
	cin>>n>>m;
	for(int i=0;i<m;i++)cin>>a[i].x>>a[i].y;
	sort(a,a+m,tmp);
	while(n>0&&now<m)
	{
		if(n>a[now].x)ans+=a[now].x*a[now].y,n-=a[now].x;
		else ans+=n*a[now].y,n=0;
		now++;
	}
	cout<<ans<<endl;
}
