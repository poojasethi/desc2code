#include<iostream>
#include<string>
#include<stdio.h>
#include<cstdio>
#include<map>
#include<cmath>
#include<algorithm>
using namespace std;
int s[11];
int main(){
int n,m,ans=0,a,b;cin>>n>>m;
for(int i=0;i<m;i++){
    cin>>a>>b;
    s[b]+=a;
}
for(int i=10;i>0;i--){
    ans+=min(n,s[i])*i;
    n-=min(n,s[i]);
}
cout<<ans;
}
