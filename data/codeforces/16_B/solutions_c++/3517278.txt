#include <iostream>
using namespace std;

int main() {
int i,j,m;
long n,k=0,s=0;
cin>>n>>m;
long a[m];int b[m];
for(i=0;i<m;i++) {
    cin>>a[i]>>b[i]; }
for(i=1;i<m;i++) 
    for(j=1;j<m;j++) {
        if(b[j-1]<b[j]) { swap(b[j-1],b[j]); swap(a[j-1],a[j]); }}

for(i=0;i<m;i++) {
    if(n-k>=a[i]) {s+=a[i]*b[i]; k+=a[i];}
    else  { s+=(n-k)*b[i]; break; }
    }
cout<<s;
return 0;       
    
}
