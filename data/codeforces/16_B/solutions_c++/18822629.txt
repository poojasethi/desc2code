#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
int  x[11];

int main()
{
    int n, m, a, b, s=0, mn;
	cin >> n >> m;
	for( int i = 0; i < m; i++ )
	{
		cin >> a >> b;
		x[b] += a;
	}
	for( int i = 10; i>=0; i-- )
	{
		 mn= min( x[i], n );
		s += i*mn;
		n -= mn;
	}

	cout << s;


	return 0;
}
