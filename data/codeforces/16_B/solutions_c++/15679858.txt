#include<iostream>
#include<algorithm>
using namespace std;
pair<int,int> a[21];
main(){long long n,m,i,s=0,j=0,k,l;
cin>>n>>m;
for(i=1; i<=m; i++){cin>>k>>l; a[i].first=l; a[i].second=k;}
sort(a+1,a+m+1); i=m+1; 
while(j<=n && i>0){i--; 
j=j+a[i].second; s=s+a[i].first*a[i].second;
} 
cout<<s-(j-n)*a[i].first;
}
