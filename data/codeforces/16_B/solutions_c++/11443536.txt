#include <bits/stdc++.h>
using namespace std;

int main(){
    vector<pair<int,int> >v;
    int n,m,a,b;
    cin>>n>>m;
    for(int i=0;i<m;i++){
        cin>>a>>b;
        v.push_back(make_pair(b,a));
    }
    sort(v.rbegin(),v.rend());

    int y=0,res=0;
    for(int i=0;i<m &&  y<n ;i++){
        int x=min(v[i].second,n-y);
        y+=x;
        res+=x*v[i].first;
    }
    cout<<res<<"\n";

}
