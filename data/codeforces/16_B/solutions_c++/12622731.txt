# include <bits/stdc++.h>

using namespace std;

typedef long long int ll;

ll a[11];

int main(){
	ll n, m;
	cin >> n >> m;
	for(int i=1 ; i<=m ; i++){
		ll fi, se;
		cin >> fi >> se;
		a[se] += fi;
	}
	
	ll ans = 0;
	for(int i=10 ; i>=1 ; i--)
		if(a[i] <= n){
			n -= a[i];
			ans += a[i] * i;
		}
		else{
			ans += n * i;
			break;
		}
	
	cout << ans << endl;
			


	return 0;
}
