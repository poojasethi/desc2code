#include<cstdio>
#include<algorithm>
using namespace std;

int n,m,a,b,ans,got;
int s[111];

int main() {
	scanf("%d%d", &n, &m);
	for(int i=0;i<m;i++){
		scanf("%d%d", &a, &b);
		s[b]+=a;
	}
	ans=0;
	for(int i=10;i>0;i--){
		got = min(n,s[i]);
		ans+=got*i;
		n-=got;
	}
	printf("%d\n",ans);

	return 0;
}
