#include<bits/stdc++.h>
using namespace std;
int main()
{

    int a,b;
    cin>>a>>b;
  	 vector<pair<int,int> >v(b);
  	 for(int i=0;i<b;++i)
  		 cin>>v[i].second>>v[i].first;
  	 sort(v.rbegin(),v.rend());
  	 int rem=0;
  	 for(int i=0;i<b;++i){
  		 if(a<=0)
            break;
  		rem+=v[i].first*min(v[i].second, a);
  		 a-=v[i].second;
  	 }
  	 cout<<rem<<endl;
    return 0;
}
