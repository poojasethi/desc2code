#include<iostream>
using namespace std;
int main(){
	int b[10]={0};
    int m,n,aa,bb,s=0,kk=9;
	cin>>n>>m;
	for(int i=0;i<m;i++){
		cin>>aa>>bb;
        b[bb-1]+=aa;
    }
	for(int j=0;j<n;j++)
		for(int k=kk;k>=0;k--){
			if(b[k]){
				s+=k+1;
				b[k]--;
				break;
			}
			else
				kk--;
		}
    cout<<s<<endl;
}
