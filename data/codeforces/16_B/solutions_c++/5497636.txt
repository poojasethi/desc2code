#include<cstdio>
int a[11];
int main()
{
    int n,m,i,b,c,sum=0;
    scanf("%d %d",&n,&m);
    while(m--)
    {
        scanf("%d %d",&b,&c);
        a[c]+=b;
    }
    for(i=10;i>0;i--)
    {
        if(n<a[i])
        {
            sum+=n*i;
            n=0;
        }
        else
        {
            n-=a[i];
            sum+=a[i]*i;
        }
    }
    printf("%d\n",sum);
}
