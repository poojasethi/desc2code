# include<iostream>
#include<algorithm>
using namespace std;
struct node{int b,a;}a[100];
int cmp(node a, node c)
{
    return a.a>c.a;
}
int main()
{
    int n, m,i;
    long long s=0;
    cin>>n>>m;
    for(i=0; i<m; i++)cin>>a[i].b>>a[i].a;
    sort(a,a+m,cmp);
    for(i=0; i<m&&n>0; i++)
    {
        s += a[i].a*(min(a[i].b,n));
        n-=min(a[i].b,n);
    }
    cout<<s<<'\n';
    return 0;
}
