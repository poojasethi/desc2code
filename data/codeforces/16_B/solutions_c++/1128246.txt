#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
    pair<int,int>  a[25];
    int n,m,res=0;
    cin >> n >> m;
    for(int i=0;i<m;i++)
        cin >> a[i].second >> a[i].first;
    sort(a,a+m);
    for(int i=m-1;i>=0;i--)
        while(a[i].second and n){
            res+=a[i].first;
            a[i].second--;n--;
        }

    cout << res << endl;
    return 0;
}
