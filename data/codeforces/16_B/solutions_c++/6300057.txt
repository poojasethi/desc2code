#include<iostream>
using namespace std;
int main()
{int n,m,i,j,c,d=0;
int a[20],b[20];
while(cin>>n>>m)
{
for(i=0;i<m;i++)
	cin>>a[i]>>b[i];
for(i=0;i<m;i++)
	for(j=i;j<m;j++)
		if(b[i]<b[j])
		{c=b[i];b[i]=b[j];b[j]=c;
         c=a[i];a[i]=a[j];a[j]=c;}
c=0;
for(i=0;;i++)
{c=c+a[i];
if(c>=n){d=d+a[i]*b[i]-(c-n)*b[i];goto l;}
else d=d+a[i]*b[i];
if(i==m-1) goto l;}
l:cout<<d<<endl;}}
