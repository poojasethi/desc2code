#include <iostream>
#include <algorithm>
using namespace std;

#define f first
#define s second

const int MAXN = 20;
pair <int, int> p[MAXN];

int main()
{
	int n, k;
	cin>> n >> k;

	for(int i = 0; i < k; i ++)
		cin>> p[i].s >> p[i].f;

	sort(p, p + k);

	long long ans = 0;
	for(int i = k - 1; i >= 0; i --)
	{
		ans += min(p[i].s, n) * p[i].f;
		n = max(0, n - p[i].s);
		if(n == 0)
			break;
	}

	cout<< ans << endl;
}
