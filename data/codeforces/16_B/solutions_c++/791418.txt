#include <iostream>
using namespace std;
int n,m,a,b,i,ans=0,x[11];
int main()
{
    //ifstream cin("input.txt");
    //ofstream cout("output.txt");
	cin >> n >> m;
	for (i=0;i<m;i++)
	{
		cin >> a >> b;
		x[b]+=a;
	}
	for (i=10;i>=1;i--)
		if (n>x[i]) {n-=x[i];ans+=i*x[i];}
		else
			{ans+=i*n; break;}
	cout << ans << endl;
    return 0;
}
