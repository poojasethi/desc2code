#include<iostream>
using namespace std;
main()
{
long long int n,m,a[20],b[20],k=0,i,max,pos;
cin>>n>>m;
for(i=0;i<m;i++)
{
cin>>a[i]>>b[i];
}
dk:
max=0;
pos=0;
for(i=0;i<m;i++)
{
if(max<b[i])
{
{max=b[i];
pos=i;
}
}}
b[pos]=0;
if(n-a[pos]>=0)
{
k=k+a[pos]*max;
n=n-a[pos];
goto dk;
}
else
{
k=k+n*max;
}
cout<<k;
return(0);
}
