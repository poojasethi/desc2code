#include<stdio.h>
int x[15];
main()
{
    int a,b,c,d,e,f;
    scanf("%d %d",&a,&b);
    for(c=0;c<b;c++)
    {
        scanf("%d %d",&d,&e);
        x[e]+=d;
    }
    f=0;
    for(c=10;c>0;c--)
    {
        if(a>=x[c])
        {
            a-=x[c];
            f=f+(x[c]*c);
        }
        else
        {
            f=f+(c*a);
            break;
        }
    }
    printf("%d",f);
}
