#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
const int maxn=10000;
pair<int,int> p[maxn];
int main(){
	int n,m;
	cin>>n>>m;
	for(int i=0;i<m;i++)
		cin>>p[i].second>>p[i].first;
	sort(p,p+m);
	int ans=0;
	int w=m-1;
	while(n&&w+1){
		ans+=(p[w].first*(min(n,p[w].second)));
		n-=min(n,p[w--].second);
	}
	cout<<ans<<endl;
}
