#include<stdio.h>
typedef __int64 LL;
	
int c[11];
int main(){
	LL x,i,j,n,m,k,t;
	scanf("%I64d%I64d",&n,&m);
	for(i=0;i<m;i++){
		scanf("%I64d%I64d",&x,&t);
		c[t]+=x;
	}
	LL res=0;
	for(i=10;i>0&&n;i--){
		if(n>=c[i]){
			n-=c[i];
			res+=c[i]*i;
		}
		else{
			res+=n*i;
			n=0;
		}
	}
	printf("%I64d\n",res);
	return 0;
}
