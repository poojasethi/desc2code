#include<bits/stdc++.h>
#define fi first
#define se second
using namespace std;
int main ()
{
    int a,i,j,cnt=0,k;
    pair < int, int > p[21];
    cin>>k>>a;
    for( i=1; i<=a; i++ )cin>>p[i].se>>p[i].fi;
    sort( p+1, p+a+1 );
    for( i=a; i>=1; i-- ){
        int x=min( k, p[i].se );
        k-=x;
        cnt+=x*p[i].fi;
    }
    cout<<cnt;
}
