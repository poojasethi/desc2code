# include<iostream>
#include<algorithm>
using namespace std;
struct cm{
	int b, nm;
}a[100];
int cmp(cm a, cm c){
	return a.nm>c.nm;
}
int main(){
	int n, m,i;
	long long s=0;
	cin>>n>>m;
	for(i=0;i<m;i++)
		cin>>a[i].b>>a[i].nm;
	sort(a, a+m,cmp);
	for(i=0;i<m&&n>0;i++){
		s += a[i].nm*(min(a[i].b,n));
		n-=min(a[i].b,n);
	}
	cout<<s<<'\n';
	return 0;
}
