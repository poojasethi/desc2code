#include <bits/stdc++.h>
using namespace std;
pair <int,int> a[22];
int main()
{
	int n,m,x,y;
	cin>>n>>m;
	for(int i=0;i<m;i++)
	{
		cin>>x>>y;
		a[i] = make_pair(y,x);
	}
	sort(a,a+m);
	reverse(a,a+m);
	long long ans = 0;
	for(int i=0;i<m && n>0;i++)
	{
		x = min(n , a[i].second);
		n -= x;
		ans += 1ll*x*a[i].first;
	}
	cout<<ans;
	return 0;
}
