#include<stdio.h>
using namespace std;
long long n,m,a,b,ar[11],mino,ok;
int main(){
	scanf("%lld%lld",&n,&m);
	for(int i=0;i<m;++i){
		scanf("%lld%lld",&a,&b);
		ar[b]+=a;
	}
	for(int i=10;i>0 && n;--i){
		if(n>ar[i]) mino=ar[i];
		else mino=n;
		ok+=i*mino;
		n-=mino;
	}
	printf("%lld\n",ok);
	return 0;
}
