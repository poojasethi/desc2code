#include<iostream>
#include<algorithm>
using namespace std;
int mas[1000002], i, a, b, c, n, m, sum, j;
pair <int, int> p[1000003];
int main(){
	cin>>a>>b;
	for(i=0; i<b; i++){
		cin>>p[i].second>>p[i].first;	
	}
	sort(p, p+b);
	reverse(p, p+b);
	while(a>0 && j<=b){
		while(p[j].second > a){
		p[j].second--;
		}
		sum = sum + p[j].first * p[j].second;
		a = a - p[j].second;
		j++;
	}
	cout<<sum;
	
}
