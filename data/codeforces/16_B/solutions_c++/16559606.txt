#include <bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

using namespace std;

int n,m,ans,c[11],a,b;

int main(){
  scanf("%d %d",&n,&m);
  FI(i,1,m) scanf("%d %d",&a,&b),c[b]+=a;
  FD(i,10,1){
    if(c[i]<=n){
      n-=c[i];
      ans+=c[i]*i;
    }else{
      ans+=n*i;
      n=0;
    }
  }
  printf("%d\n",ans);
  return 0;
}
