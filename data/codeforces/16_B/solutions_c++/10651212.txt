#include <bits/stdc++.h>

using namespace std;
long n,m,i,sum;
int main()
{cin>>n>>m;pair<int,int>a[m];
for(i=0;i<m;i++)
{
    cin>>a[i].second>>a[i].first;
}sort(a,a+m);reverse(a,a+m);
for(i=0;i<m;i++)
{
if(a[i].second>=n){cout<<sum+(n*a[i].first);return 0;}
sum+=a[i].second*a[i].first;
n-=a[i].second;
}cout<<sum;
    return 0;
}
