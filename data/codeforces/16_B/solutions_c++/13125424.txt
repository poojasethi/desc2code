#include <iostream>
using namespace std;

long long m,n,a,b,ar[11]={0},ress;

void res()
{
    cin>>n>>m;
    for(int i=0;i<m;i++)
    {
        cin>>a;
        cin>>b;
        ar[b] += a;
    }
    for(int i=10;i>0 && n>0;i--)
    {
        if(n > ar[i])
            ress+=ar[i]*i;
        else
            ress+=n*i;
        n-=ar[i];
    }
    cout<<ress;
}

int main()
{
    res();

}
