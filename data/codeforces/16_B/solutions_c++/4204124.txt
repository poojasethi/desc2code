#include<stdio.h>
int main()
{
 long int n,b,sum=0,d=0,count=0;
 int a[11]={0},c,i,m;
 scanf("%ld",&n);
 scanf("%d",&m);
 for(i=0;i<m;i++)
{
  scanf("%ld",&b);
  scanf("%d",&c);
  a[c]+=b;
}
for(c=10;c>0;c--)
{
  d=(n-count)>a[c]?a[c]:(n-count);
  count+=d;
  sum+=d*c;
  if(count==n)
  break;
}
printf("%d",sum);
return 0;
}
