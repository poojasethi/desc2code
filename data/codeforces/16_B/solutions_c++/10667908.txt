#include <iostream>
using namespace std;
int arr[11],sum;
int main(){
    int x,y,n,m; cin >> n >> m ;
    for (int i=0; i<m; i++){
        cin >> x >> y ;
        arr[y]+=x;
    }
    for ( int i=10; n && i>=0; --i  ){
        if (n>arr[i]) sum+=arr[i]*i,n-=arr[i];
        else sum+=n*i,n=0;
    }

    cout << sum ;
    return 0;
}
