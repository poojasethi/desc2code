#!/usr/bin/env python

from sys import stdin

N, M = map(int, stdin.readline().split())

v = []
for i in xrange(M):
   a, b = map(int, stdin.readline().split())
   v.append((b, a))
v.sort()
v.reverse()
res = 0
for t in v:   
#  print t[0], t[1], res
   if t[1] >= N:
      res += t[0] * N
      break
   N -= t[1]
   res += t[0] * t[1]

print res
