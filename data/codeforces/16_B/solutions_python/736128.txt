import operator

total_size, containers = map(int, raw_input().split())
matchboxes = sorted([map(int, raw_input().split()) for _ in range(containers)], key=operator.itemgetter(1))[::-1]

volume = 0
size = 0
for s, v in matchboxes:
    delta = min(total_size-size, s)
    volume += v*delta
    size += delta

    if size == total_size:
        break

print volume
