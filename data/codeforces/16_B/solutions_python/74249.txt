import sys

n,m = (int(v) for v in sys.stdin.readline().strip().split())
lst = [0 for i in xrange(11)]
for i in xrange(m):
    a,b = (int(b) for b in sys.stdin.readline().strip().split())
    lst[b] += a
    
ret = 0
for i in xrange(0,10):
    num = min(n, lst[10-i])
    ret += num * (10-i)
    n -= num
    
print ret
