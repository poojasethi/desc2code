total_box, m = map(int, raw_input().split()) 
container = [0] * 11
for _ in range(m):
    box, matches = map(int, raw_input().split()) 
    container[matches] += box
r = 0
for i, box in enumerate(container[ : : -1]):
    r += (10 - i) * min(box, total_box)
    total_box -= min(box, total_box)
    if total_box == 0: break
