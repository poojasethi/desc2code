def readint(): return int(raw_input())
def readfloat(): return float(raw_input())
def readarray(N, foo):
        res = []
        for _ in xrange(N):
                res.append(foo())
        return res
def readlinearray(foo): return map(foo, raw_input().split())

def get_max(containers, n):
    r = 0
    containers.sort(reverse=True)
    for bi, ai in containers:
        if n == 0: break
        take = min(ai, n)
        r += take * bi
        n -= take
    return r

if __name__ == '__main__':
    n, m = readlinearray(int)
    
    containers = []
    for ci in xrange(m):
        ai, bi = readlinearray(int)
        containers.append((bi, ai))
    print get_max(containers, n)
