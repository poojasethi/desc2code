from operator import itemgetter
n,m=map(int,raw_input().split())
a=[]
for x in xrange(m):
    i,j=map(int,raw_input().split())
    a.append((i,j))
taken=0
ans=0
for (i,j) in sorted(a,key=itemgetter(1),reverse=True):
    if taken==n:
        break
    ans+=min(i,n-taken)*j
    taken=min(n,taken+i)
print ans
