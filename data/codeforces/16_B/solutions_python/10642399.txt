from operator import itemgetter
n,m = map(int, raw_input().split())
container = [[int(y) for y in raw_input().split()] for x in xrange(m)]
container = sorted(container, key=itemgetter(1), reverse=True)
matches = 0
for c in container:
    if c[0] <= n:
        n -= c[0]
        matches += c[1] * c[0]
    else:
        matches += c[1] * n
        break
print matches
