m, n = tuple([int(x) for x in raw_input().split()])
boxes = []
for i in xrange(n):
    boxes.append([int(x) for x in raw_input().split()])
    boxes[i].reverse()
boxes.sort()
boxes.reverse()
sum = 0
i = 0
while m > 0 and i < len(boxes):
    sum += min(boxes[i][1], m) * boxes[i][0]
    m -= boxes[i][1]
    i += 1
print sum
