n, m = map(int, raw_input().split())
containers = [reduce(lambda a, b: (int(a), int(b)), raw_input().split()) for _ in range(m)]
containers.sort(key=lambda (a, b): b)
total = 0
while n > 0 and len(containers) > 0:
	total += containers[-1][1] * min(n, containers[-1][0])
	n -= min(n, containers[-1][0])
	containers.pop()
print total
