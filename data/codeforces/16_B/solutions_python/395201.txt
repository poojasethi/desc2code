#!/usr/bin/python
s = raw_input().split(" ")
n, m = int(s[0]), int(s[1])
a = []
res = 0
for i in range(0, m):
  x = list(map(lambda x: int(x), raw_input().split(" ")))
  a.append(x)
a.sort(key=lambda x: -x[1])
#print(a)
for q in a:
  if n == 0: break;
  t = min(n, q[0])
  res += q[1] * t
  n -= t
print(res)
