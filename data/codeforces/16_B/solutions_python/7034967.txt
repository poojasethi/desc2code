from operator import itemgetter
n,m=map(int,raw_input().split())
p=[]
for i in range(m):
    p.append(map(int,raw_input().split()))
    
p.sort(key=itemgetter(1,0),reverse=True)
t,s=0,0
for i in range(m):
    t+=p[i][0]
    if t<=n:
        s+=p[i][0]*p[i][1]
    else:
        s+=(n-(t-p[i][0]))*p[i][1]
        break
    
print s
