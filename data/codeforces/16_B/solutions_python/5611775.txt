#!/usr/bin/env python
# -*- coding: UTF-8 -*-
import time
import sys, io
import re, math
start = time.clock()
l=[]
chk,ans=0,0
(n,m)=map(int, raw_input().split())
for _ in range(m):
    i=map(int, raw_input().split())
    l.append(i)
#print l
l1=sorted(l, key=lambda l:l[1])
# print l1
l1.reverse()
for i in l1:
    if n==chk: break
    elif n>=chk+i[0]:
        ans+=i[0]*i[1]
        chk+=i[0]
    else:
        ans+=(n-chk)*i[1]
        chk+=(n-chk)
print ans
