#!/usr/bin/python
class Container:
	def __init__(self, a, b):
		self.a = a
		self.b = b

n, m = map(lambda(x): int(x), raw_input().split())
l = []
for i in range(m):
	a, b = map(lambda(x): int(x), raw_input().split())
	l.append(Container(a, b))

l.sort(key = lambda x: -x.b)
remaining = n
total = 0
for x in l:
	if remaining <= x.a:
		total += remaining * x.b
		break
	else:
		total += x.a * x.b
		remaining -= x.a
print total
