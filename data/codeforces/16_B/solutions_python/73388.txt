n, m = [int(x) for x in raw_input().split()]
boxs = []
for x in xrange(m):
    tp = [int(x) for x in raw_input().split()]
    boxs += [(tp[0], tp[1])]
boxs.sort(key = lambda a: a[1])
ans = 0
i = m - 1
while n > 0 and i > -1:
    taken = min(n, boxs[i][0])
    ans += taken * boxs[i][1]
    n -= taken
    i -= 1
print ans
