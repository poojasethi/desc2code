def func(str):
    x, y = 0, 0
    for c in str:
        if c == 'U': y += 1
        elif c == 'D': y -= 1
        elif c == 'R': x += 1
        elif c == 'L': x -= 1
    if x == 0 and y == 0:
        return 1
    else:
        return 0

n = int(raw_input())
s = raw_input()

ans = 0
for x in xrange(n+1):
    for y in xrange(x+1, n+1):
        ans += func(s[x:y])

print ans
