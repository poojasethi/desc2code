import sys

f = sys.stdin

n = int(f.readline())
s = f.readline().strip()
x = y = 0
h = {}
for ch in s:
    if ch == 'U':
        x += 1
    elif ch == 'D':
        x -= 1
    elif ch == 'L':
        y -= 1
    elif ch == 'R':
        y += 1
    if x * 250 + y not in h:
        h[x * 250 + y] = 1
    else:
        h[x * 250 + y] += 1
if 0 in h:
    ans = h[0]
else:
    ans = 0
for x in h.values():
    ans += x * (x - 1) / 2
print ans
