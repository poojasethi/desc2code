def solve(moves):
	if len(moves) == 0 or moves == "":
		return False

	x, y = 0, 0

	for move in moves:
		if move == "U":
			y += 1
		elif move == "D":
			y -= 1
		elif move == "R":
			x += 1
		elif move == "L":
			x -= 1

	if x == 0 and y == 0:
		return True
	else:
		return False

n = int(raw_input())
moves = raw_input()

x, c = 1, 0
for i in range(n):
	for j in range(len(moves[i:]) + x):
		if solve(moves[i:j]):
			c += 1
	x += 1
print c
