n = int(raw_input())
s = raw_input()
ans = 0
for i in xrange(len(s)):
    x, y = 0, 0
    for j in xrange(i, len(s)):
        if s[j] == 'U':
            y += 1
        elif s[j] == 'D':
            y -= 1
        elif s[j] == 'L':
            x -= 1
        elif s[j] == 'R':
            x += 1
        if x == 0 and y == 0:
            ans += 1
print ans
