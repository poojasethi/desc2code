PROBLEM = "A"
FILE = "%s" %(PROBLEM) 
try:
    inFile = open(FILE+".txt")
except:
    pass

def read():
    try:
        return inFile.readline().strip()
    except:
        return raw_input().strip()

n = int(read())
string = read()
ways = 0

for i in range(n):
    for j in range(i+1,n+1):
        substr = string[i:j]
        if substr.count("D") == substr.count("U") and substr.count("L") == substr.count("R"):
            ways += 1
print ways
