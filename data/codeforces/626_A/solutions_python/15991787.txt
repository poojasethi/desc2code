n=input()
s=raw_input()

d={
        'U': lambda p: (p[0]+0,p[1]+1),
        'D': lambda p: (p[0]+0,p[1]-1),
        'R': lambda p: (p[0]+1,p[1]),
        'L': lambda p: (p[0]-1,p[1]),
}
total=0
for i in range(n):
        f=(0,0)
        for j in range(i,n):
                f= d[s[j]](f)
                if f==(0,0):
                        total+=1

print total
