chars = int(input())
def abc(s):
	
	num = 0
	for i in xrange(0,len(s)):
		count1 =0
		count2 = 0
		for k in s[i:]:
			if(k=='U'):
				count1+=1
			elif(k=='D'):
				count1-=1
			elif(k=='R'):
				count2+=1
			else:
				count2-=1
			if(count1==0 and count2==0):
				num+=1
				
	print num

abc(raw_input())
