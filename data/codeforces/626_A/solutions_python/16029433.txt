n=int(raw_input())
a=raw_input()
ans=0
for i in range (0, n):
    x=0
    y=0
    for j in range (i, n):
        if a[j]=='U':
            y=y+1
        elif a[j]=='D':
            y=y-1
        elif a[j]=='R':
            x=x+1
        elif a[j]=='L':
            x=x-1
        if x==0 and y==0:
            ans=ans+1
print ans
