if __name__=="__main__":
    n = int(raw_input())
    s = raw_input()

    c = 0
    for i in range(n):
       p = (0,0)
       for j in range(i,n):
            if s[j] == 'D':
                p = (p[0],p[1]-1)
            elif s[j] == 'U':
                p = (p[0],p[1]+1)
            elif s[j] == 'R':
                p = (p[0]+1,p[1])
            elif s[j] == 'L':
                p = (p[0]-1,p[1])
            if p == (0,0):
                c+=1
    print c
