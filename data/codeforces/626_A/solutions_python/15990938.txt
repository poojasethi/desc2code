#! /usr/bin/env python
from math import acos,asin,atan,sqrt,pi,sin,cos,tan,ceil,floor,log
import re
from operator import itemgetter as ig

#map(int,raw_input().split())

n=input()
s=raw_input()
res=0
for i in range(n):
    for j in range(i+1,n):
        x=y=0
        for k in range(i,j+1):
            if s[k]=='U':
                y+=1
            elif s[k]=='D':
                y-=1
            elif s[k]=='R':
                x+=1
            else:
                x-=1
        if x==0 and y==0:
            res+=1
print res

exit(0)
