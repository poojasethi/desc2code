n = int(input())
steps = raw_input()
l = len(steps)

res = 0
for i in range(0, l):
	for j in range(i, l):
		x = 0
		y = 0
		for k in range(i, j + 1):
			c = steps[k]
			if c == 'U':
				x -= 1
			if c == 'D':
				x += 1
			if c == 'L':
				y -= 1
			if c == 'R':
				y += 1
		if x == 0 and y == 0:
			res += 1

print res
