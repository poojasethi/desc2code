import sys
n = int( sys.stdin.readline().strip() )
secuence = str( sys.stdin.readline() )
answer = 0

for i in xrange(n):
    h = v = 0
    for j in xrange(i,n):
        if secuence[j] == 'U':
            v += 1
        elif secuence[j] == 'D':
            v -= 1
        elif secuence[j] == 'R':
            h += 1
        else:
            h -= 1
        if v == 0 and h == 0:
            answer += 1

print answer
