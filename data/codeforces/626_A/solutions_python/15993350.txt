n = input()
st = list(raw_input())
h, v = 0, 0
count = 0
for i in xrange(n):
	if st[i] == 'U':
		v += 1
	if st[i] == 'D':
		v -= 1
	if st[i] == 'L':
		h -= 1
	if st[i] == 'R':
		h += 1
	for j in xrange(i+1, n):
		if st[j] == 'U':
			v += 1
		if st[j] == 'D':
			v -= 1
		if st[j] == 'L':
			h -= 1
		if st[j] == 'R':
			h += 1
		if v == 0 and h== 0:
			count += 1
	v,h = 0,0
print count
