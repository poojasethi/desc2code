from collections import defaultdict
n=int(raw_input())
l = list(str(raw_input()))
d=defaultdict(tuple)
d[-1] = [0,0]
for i in range (0, n):
    x,y=d[i-1]
    if l[i] == 'U':
        d[i] = (x-1,y)
    if l[i]== 'D':
        d[i] = (x+1,y)
    if l[i] == 'L':
        d[i] = (x, y-1)
    if l[i] == 'R':
        d[i] = (x,y+1)

e = defaultdict(int)
e[ (0,0)]=1
for i in range (0, n):
    l=d[i]
    e[l] += 1

s=0
for i in e.values():
    s += 0.5 * i * (i-1)
print int(s)
