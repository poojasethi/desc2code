#!/usr/bin/python
n = input()
point = (0,0)
dictPoints = dict()
dictPoints[point] = 1
for i in raw_input().strip():
	if i=='U':
		point = (point[0]-1, point[1])
	elif i=='D':
		point = (point[0]+1, point[1])
	elif i=='L':
		point = (point[0], point[1]-1)
	else:
		point = (point[0], point[1]+1)
	try:
		dictPoints[point] += 1
	except:
		dictPoints[point] = 1
#print dictPoints
answer = 0
for i in dictPoints.values():
	answer += i*(i-1)/2
print answer
