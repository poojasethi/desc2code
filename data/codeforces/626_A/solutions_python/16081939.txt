import sys

n = int(sys.stdin.readline())

s = [0, 0]
history = [[0, 0]]
count = 0

for c in sys.stdin.readline():
    if c == 'U':
        s[1] += 1
    elif c == 'D':
        s[1] -= 1
    elif c == 'R':
        s[0] += 1
    elif c == 'L':
        s[0] -= 1
    else:
        continue
    #print s
    for h in history:
        if h == s:
            count += 1
            #break
    history.append(s[:])

print count
