n = int(raw_input())
ins = raw_input()
out = 0

for i in xrange(n):
	countX = 0
	countY = 0
	for j in xrange(i, n):
		if ins[j] == 'U':
			countY += 1
		elif ins[j] == 'D':
			countY -= 1
		elif ins[j] == 'R':
			countX += 1
		else:
			countX -= 1
		if countX == 0 and countY == 0:
			out += 1
print out
