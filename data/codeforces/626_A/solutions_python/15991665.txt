def sim(com, a, b):
	x, y = 0, 0
	for i in xrange(a, b):
		c = com[i]
		if c == 'U':
			y += 1
		if c == 'D':
			y -= 1
		if c == 'L':
			x -= 1
		if c == 'R':
			x += 1
	return x == 0 and y == 0

n = int(raw_input())
com = raw_input().strip()
tot = 0 
for a in xrange(n):
	for b in xrange(a+1, n+1):
		if sim(com, a, b):
			tot += 1
print tot
