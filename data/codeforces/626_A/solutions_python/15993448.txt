import sys

def check(sub):
    d = dict()
    d['U'] = 0
    d['R'] = 0
    d['D'] = 0
    d['L'] = 0
    for elem in sub:
	d[elem] += 1
    return d['U'] == d['D'] and d['R'] == d['L']

def main():
    n = int(sys.stdin.readline().strip())
    s = sys.stdin.readline().strip()

    r = 0
    for i in xrange(n):
	if i >= 1:
	    for j in xrange(i):
		if (i-j+1) % 2 != 0:
		    continue
		if check(s[j:(i+1)]):
		    r += 1
    print r

if __name__ == "__main__":
    main()
