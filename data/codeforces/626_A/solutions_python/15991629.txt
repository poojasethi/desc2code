n = int(raw_input())
s = raw_input();
ans = 0;
curr = [0,0];
for i in xrange(n):
	curr = [0,0];
	for j in xrange(i,n):
		if s[j] == 'U' :
			curr[1]+=1;
		elif s[j] == 'D' :
			curr[1]-=1;
		elif s[j] == 'R' :
			curr[0]+=1;
		else:
			curr[0]-=1;
		if curr[0] == 0 and curr[1] == 0:
			ans+=1;
print ans;
