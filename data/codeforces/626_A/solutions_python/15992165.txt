n = int(raw_input())
s = raw_input()
ans = [(0,0)]*(n+1)
j = 0
for i in range(0,n):
	j = j+1
	if s[i] == "U":
		ans[i] = (ans[i-1][0],ans[i-1][1]+1)
	if s[i] == "D":
		ans[i] = (ans[i-1][0],ans[i-1][1]-1)
	if s[i] == "R":
		ans[i] =(ans[i-1][0]+1,ans[i-1][1])
	if s[i] == "L":
		ans[i] = (ans[i-1][0]-1,ans[i-1][1])
res = 0
for i in range(1,len(ans)):
	for j in range(0,i):
		if ans[i][0] == ans[j][0] and ans[i][1] == ans[j][1]:
			res += 1
print res
