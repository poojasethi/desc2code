i = input()
dic = {"0,0":1}
pos = [0,0]
n = 0
for j in raw_input():
	if (j == "U"):
		pos = [pos[0],pos[1]+1]
	elif (j == "D"):
		pos = [pos[0],pos[1]-1]
	elif (j == "R"):
		pos = [pos[0]+1,pos[1]]
	elif (j == "L"):
		pos = [pos[0]-1,pos[1]]
	has = str(pos[0])+","+str(pos[1])
	if (not has in dic):
		dic[has] = 1
	else:
		n += dic[has]
		dic[has] += 1
print(n)
