import sys
n = int(sys.stdin.readline())
inst = sys.stdin.readline()
res = 0

for i in xrange(n):
	d = [0] * 2
	for j in xrange(i, n):
		if inst[j] == 'L':
			d[0] += 1
		elif inst[j] == 'R':
			d[0] -= 1
		elif inst[j] == 'U':
			d[1] += 1
		else:
			d[1] -= 1
		if not d[0] and not d[1]:
			res += 1
			
print res
