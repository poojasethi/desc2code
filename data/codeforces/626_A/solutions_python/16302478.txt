n = int(raw_input())
moves = raw_input()

count = 0
ud = 0
lr = 0
for i in range(n):
    ud = 0
    lr = 0
    for j in range(i, n):
        if moves[j] == 'U':
            ud += 1
        elif moves[j] == 'R':   
            lr += 1
        elif moves[j] == 'D':
            ud -= 1
        elif moves[j] == 'L':
            lr -= 1
        # print ud, lr
        if ud == 0 and lr == 0:
            count += 1

print count
