n = int(raw_input())
commands = raw_input()

def add(x, y):
    return (x[0] + y[0], x[1] + y[1])

dx = {
    "L": (-1, 0),
    "R": (1, 0),
    "U": (0, 1),
    "D": (0, -1)
}

count = 0

for s in range(n):
    for e in range(s + 1, n):
        pos = (0, 0)
        for i in range(s, e + 1):
            pos = add(pos, dx[commands[i]])
        if pos == (0, 0):
            count += 1

print count
