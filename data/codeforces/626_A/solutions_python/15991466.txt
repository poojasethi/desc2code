n = int(raw_input())
s = raw_input().strip()
x = [0]
y = [0]
dd = {'U': (-1, 0), 'D': (1, 0), 'L': (0, 1), 'R': (0, -1)}
for c in s:
    dx, dy = dd[c]
    x += [x[-1] + dx]
    y += [y[-1] + dy]
ans = 0
for i in xrange(n):
    for j in xrange(i + 1, n + 1):
        if x[j] == x[i] and y[j] == y[i]:
            ans += 1
print ans
