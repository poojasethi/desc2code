n = input()
s = raw_input()
cnt = 0
for i in range(n):
  for j in range(i, n):
    dx, dy = 0, 0
    for k in range(i, j + 1):
      if s[k] == 'U': 
        dx -= 1
      if s[k] == 'D':
        dx += 1
      if s[k] == 'L':
        dy -= 1
      if s[k] == 'R':
        dy += 1
    if dx == 0 and dy == 0:
      cnt += 1
print cnt
