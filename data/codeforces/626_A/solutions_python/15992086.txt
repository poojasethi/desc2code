#!/usr/bin/env python

n = int(raw_input())
moves = list(raw_input())

ways = 0
for i in range(len(moves)-1):
	y = 0
	x = 0
	for j in range(i, len(moves)):
		m = moves[j]
		if m == 'U':
			y+=1
		elif m == 'R':
			x+=1
		elif m == 'L':
			x-=1
		else:
			y-=1
		if y == x == 0:
			ways+=1

print ways
