from math import *
from Queue import *
from sys import *




n = int(raw_input())
s = raw_input()
p = [(0,0)]
res = 0
for i in range(len(s)):
    x = p[-1][0]
    y = p[-1][1]
    if s[i] == 'U':
        p.append((x,y+1))
    if s[i] == 'D':
        p.append((x,y-1))
    if s[i] == 'R':
        p.append((x+1,y))
    if s[i] == 'L':
        p.append((x-1,y))
for i in range(1,len(p)):
    for j in range(i):
        if p[i] == p[j]:
            res += 1
print res
