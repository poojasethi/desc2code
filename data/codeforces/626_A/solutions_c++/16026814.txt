#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll n,ans,x,y;
string s;
int main() {
	cin>>n>>s;
	for(ll i=0;i<n;i++){
	    x=0;y=0;
        for(ll j=i;j<n;j++){
            if(s[j]=='U')y--;
            if(s[j]=='D')y++;
            if(s[j]=='L')x--;
            if(s[j]=='R')x++;
            if(x==0 && y==0)ans++;
        }
	}
	cout<<ans<<endl;
}
