#include  <bits/stdc++.h>

using namespace std;

int main () 
{
	int n;
	cin >> n;
	string s;
	cin >> s;
	int sz = s.size();
	int cnt = 0;

	for(int i = 0; i < sz; i++) {
		int x = 0, y = 0;
		for (int j = i; j < sz; j++) {
			if (s[j] == 'U') y++;
			if (s[j] == 'D') y--;
			if (s[j] ==  'R')x++; 
			if (s[j] ==  'L')x--; 
			if (x == 0 and y == 0) cnt++;
		}
	}
	cout <<cnt<<endl;
	return 0;
}
