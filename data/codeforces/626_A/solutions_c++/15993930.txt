#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	string s;
	cin>>s;
	int x=0,y=0,l=s.size(),count=0;
	for(int i=0;i<l;i++)
	{
		x=0;y=0;
		for(int j=i;j<l;j++)
		{
			if(s[j]=='R')
				x++;

			else if(s[j]=='L')
				x--;

			else if(s[j]=='U')
				y++;

			else if(s[j]='D')
				y--;

			if(x==0 && y==0)
				count++;
		}
	}

	cout<<count;
}
