#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int N=250;
int n,ans=0;
char s[N];
int main()
{
	scanf("%d",&n);
	scanf("%s",s);
	for(int i=0;i<n-1;++i)
	for(int j=i+1;j<n;++j)
	{
		int h=0,l=0;
		for(int k=i;k<=j;++k)
		{
			if(s[k]=='L')h++;
			if(s[k]=='R')h--;
			if(s[k]=='U')l++;
			if(s[k]=='D')l--;
		}
		if(!h&&!l)ans++;
	}
	printf("%d\n",ans);
}
