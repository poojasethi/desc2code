#include <cstdio>

int n,x,y,ans;
char s[201];

int main()
{
	scanf("%d%s",&n,s);
	ans=0;
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			x=0;
			y=0;
			for(int k=i;k<=j;k++)
			{
				if(s[k]=='U') x--;
				else if(s[k]=='D') x++;
				else if(s[k]=='L') y--;
				else y++;
			}
			if(!(x||y)) ans++;
		}
	}
	printf("%d",ans);
	return 0;
}
