#include<iostream>
using namespace std;
int main()
{
	int n,i,j,ct=0;
	int u,r;
	cin>>n;
	char a[n];
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<(n-1);i++)
	{
		u=r=0;
		for(j=i;j<n;j++)
		{
			switch(a[j])
			{
				case 'U':u++;break;
				case 'D':u--;break;
				case 'R':r++;break;
				case 'L':r--;break;
			}
			if(u==0&&r==0) ct++;
		}
	}
	cout<<ct<<"\n";
	return 0;

}
