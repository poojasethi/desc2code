#include<cstring>
#include<iostream>
#include<cstdio>
using namespace std;
string s;
int n,x,y,ans;
int main(){
	scanf("%d",&n);
	cin>>s;
	for(int i=0;i<n;i++)
		for(int j=i;j<n;j++){
			x=0;
			y=0;
			for(int k=i;k<=j;k++)
				if(s[k]=='L') x--;
				else if(s[k]=='R') x++;
				else if(s[k]=='U') y++;
				else y--;
			if(x==0 && y==0) ans++;
		}
	cout<<ans<<endl;
}
