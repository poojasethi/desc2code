#include<cstdio>
int main(void){
	int n;
	scanf("%d",&n);
	char s[300];
	scanf("%s",s);
	int sum=0;
	for(int i=0;i<n;i++){
		int x=0,y=0;
		for(int j=i;j<n;j++){
			if(s[j]=='D') y--;
			if(s[j]=='U') y++;
			if(s[j]=='R') x++;
			if(s[j]=='L') x--;
			if(y==0&&x==0) sum++;
		}
	}
	printf("%d",sum);
	
	return 0;
}
