#include<bits/stdc++.h>
using namespace std;
int main(){
	
	int n;cin>>n;
	string str;cin>>str;
	int l,r,u,d,ans=0;
	
	for(int i=0;i<n-1;i++){
		l=r=u=d=0;
		for(int j=i;j<n;j++) {
			if(str[j]=='R') r++;
			else if(str[j]=='L') l++;
			else if(str[j]=='U') u++;
			else if(str[j]=='D') d++;
			if(l==r && u==d) ans++;
		}
	}
	cout<<ans;
}
