#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main(){
	char a[205];
	int n,i,j,sum=0;
	cin>>n;
	scanf("%s",a+1);
	for(i=1;i<=n;i++){
		int x=0,y=0;
		for(j=i;j<=n;j++){
			if(a[j]=='D')x+=1;
			else if(a[j]=='L')y-=1;
			else if(a[j]=='U')x-=1;
			else y+=1;
			if(x==0&&y==0)sum++;
		}
	}
	cout<<sum<<endl;
	return 0;
}
