#include<bits/stdc++.h>

using namespace std;

string s;

int main()
{
	scanf(" %*d");
	cin >> s;
	int r = 0;
	for(int i=0; i<s.size(); i++)
	{
		int x=0, y=0;
		for(int j=i; j<s.size(); j++)
		{
			if(s[j]=='U') y++;
			else if(s[j]=='D') y--;
			else if(s[j]=='L') x--;
			else if(s[j]=='R') x++;
			if(x==0 and y==0) r++;
		}
	}	
	printf("%d\n",r);
	return 0;
}
