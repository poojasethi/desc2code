#include <iostream>
using namespace std;

int main() {
	int i,j,n,h=0, v=0, c=0;
	cin>>n;
	string s;
	cin>>s;
	
	for(i=0; i<n; i++){
		h=0; v=0;
		for(j=i;j<n;j++){
	      if(s[j]=='L') h++;
	      if(s[j]=='R') h--;
	      if(s[j]=='U') v++;
	      if(s[j]=='D') v--;
	      if(!h and !v) c++;
		}
	}
	cout<<c<<endl;
	
	
	return 0;
}
