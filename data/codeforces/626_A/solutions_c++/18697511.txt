#include <bits/stdc++.h>
using namespace std;
int main()
{
	string s;
	int n;
	cin>>n>>s;
	int ans=0;
	for(int i=1;i<=n;i++){
		for(int j=0;j<n-i;j++){
			map<char,int> m;
			m['U']=0;
			m['D']=0;
			m['R']=0;
			m['L']=0;
			for(int k=j;k<=j+i;k++){
				m[s[k]]++;
			}
			if(m['U']==m['D'] && m['R']==m['L']) ans++;
		}
	}
	cout<<ans;
}
