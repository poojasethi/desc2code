#include<stdio.h>
int main()
{
	int a[402][402];
	int n,i,j,k,c,b;

	scanf("%d",&n);
	getchar();
	for(i=1;i<=401;i++)
		for(j=1;j<=401;j++)
			a[i][j]=-1;
	i=j=201;
	a[i][j]++;
	b=0;
	for(k=1;k<=n;k++)
	{
		if((c=getchar())=='U')i--;
		else if(c=='R')j++;
		else if(c=='D')i++;
		else if(c=='L')j--;
		a[i][j]++;
		b+=a[i][j];
	}
	printf("%d",b);
	return 0;
}
