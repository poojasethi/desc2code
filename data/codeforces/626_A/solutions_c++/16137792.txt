#include <cstdio>
int n, ans, x, y;
char s[205];

int main() {
	scanf("%d\n%s", &n, s);
	for (int i = 0; i < n; i++) {
		x = y = 0;
		for (int j = i; j < n; j++) {
			if (s[j] == 'U') y--;
			else if (s[j] == 'D') y++;
			else if (s[j] == 'L') x--;
			else x++;
			if (!x && !y) ans++;
		}
	}
	printf("%d", ans);
}
