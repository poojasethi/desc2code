#include <bits/stdc++.h>
 using namespace std;
int main(){
ios_base :: sync_with_stdio (false);
int a,h=0,v=0,c=0;
string b;
cin>>a;
cin>>b;
for(int j=0;j<a;j++){
for(int i=j;i<a;i++){
if(b[i]=='U')
v++;
else if(b[i]=='D')
v--;
else if(b[i]=='R')
h++;
else if(b[i]=='L')
h--;
if(v==0 && h==0)
c++;
}
v=0;
h=0;
}
cout<<c;
}
