#include <bits/stdc++.h>
using namespace std; 
int main()
{
	int i,j,k,n,x=0,y=0,count=0;
	char s[300];
	cin>>n;
	cin>>s;
	for(i=0;i<n;i++)
	{
		x=y=0;
		for(j=i;j<n;j++)
		{
			if(s[j]=='U')
				y++;
			if(s[j]=='D')
				y--;
			if(s[j]=='L')
				x--;
			if(s[j]=='R')
				x++;
			if(x==0&&y==0)
				count++;
		}
	}
	cout<<count<<endl;
	return 0;
}
