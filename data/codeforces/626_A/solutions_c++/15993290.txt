#include <cstdio>

int main()
{
	int n;
	char com[210];
	scanf("%d\n%s", &n, com);
	int res = 0;
	for(int i=0;i<n-1;i++)
	{
		int sum[2] = {0};
		for(int j=i;j<n;j++)
		{
			if(com[j] == 'U') sum[0]++;
			else if(com[j] == 'R') sum[1]++;
			else if(com[j] == 'D') sum[0]--;
			else sum[1]--;
			if(!sum[0] && !sum[1]) res++;
		}
	}
	printf("%d\n", res);
	return 0;
}
