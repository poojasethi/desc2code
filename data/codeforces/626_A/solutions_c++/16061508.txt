#include <iostream>
using namespace std;

int main() {
	string str;
	int n;
	cin>>n;
	cin>>str;
	int ans=0;
	for(int i=0;i<n;++i){
		int x=0,y=0;
		for(int j=i;j<n;++j){
			if(str[j]=='U') ++y;
			if(str[j]=='D') --y;
			if(str[j]=='R') ++x;
			if(str[j]=='L') --x;
			if(x==0&&y==0) ++ans;
		}
	}
	cout<<ans;
	return 0;
}
