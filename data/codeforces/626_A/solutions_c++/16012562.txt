#include<bits/stdc++.h>
using namespace std;
char s[201];
long n, res;
int main(){
	cin>>n;	for(int i=1; i<=n; i++)cin>>s[i];
	for(int i=1; i<n; i++)for(int j=i+1; j<=n; j++){
			int x=0, y=0;
			for(int k=i; k<=j; k++){
				if (s[k]=='L')y--;else if (s[k]=='R')y++;else
				if (s[k]=='U')x--;else if (s[k]=='D')x++;
			}	if (x==0&&y==0)res++;
		}
	cout<<res;
	return 0;
}
