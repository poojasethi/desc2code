#include <bits/stdc++.h>
using namespace std;

int main(){
	string s;int n,i,j,x,y;cin>>n>>s;
	int ret=0;
	for(i=0;i<n;++i){
		x=y=0;
		for(j=0;i+j<n;++j){
			if(s[i+j]=='U')++y;
			if(s[i+j]=='D')--y;
			if(s[i+j]=='L')++x;
			if(s[i+j]=='R')--x;
			if(x==0 && y==0)++ret;
		}
	}
	cout<<ret<<endl;
}
