#include <bits/stdc++.h>
using namespace std;

char s[1000];
int n;

int main() {
	scanf("%i\n%s",&n,s);
	int res=0;
	for(int i=0;i<n;i++) {
		int x=0,y=0;
		for(int j=i;j<n;j++) {
			if (s[j]=='U') ++y;
			if (s[j]=='D') --y;
			if (s[j]=='L') --x;
			if (s[j]=='R') ++x;
			if (x==0 && y==0) ++res;
		}
	}
	cout<<res;
	return 0;
}
