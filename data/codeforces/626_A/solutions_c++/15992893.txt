#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int n, ans=0;
	cin>>n;
	string s;
	cin>>s;
	int x=0, y=0;
	for(int j=0; j<s.size() ; j++)
	{
		x=0,y=0;
	for(int i=j ; i<s.size() ; i++)
	{
		if(s[i] == 'U')
			y++;
		else if(s[i] == 'D')
			y--;
		else if(s[i] == 'R')
			x++;
		else
			x--;
		if(x == 0 && y == 0)
			ans++;
			
	}
	}
	cout<<ans<<endl;
}
