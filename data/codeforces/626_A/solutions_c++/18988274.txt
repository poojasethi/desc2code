#include<iostream>
#include<string>
using namespace std;
int main(){
	int n,ql=0,qu=0,qr=0,qd=0,d=0;
	string s;
	cin>>n;
	cin>>s;
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			for(int y=i;y<=j;y++){
				if(s[y]=='U')qu++;
				if(s[y]=='L')ql++;
				if(s[y]=='R')qr++;
				if(s[y]=='D')qd++;
			}
			if(qd==qu&&ql==qr)d++;
			ql=0;qu=0;qr=0;qd=0;
		}
	}
	cout<<d;
}
