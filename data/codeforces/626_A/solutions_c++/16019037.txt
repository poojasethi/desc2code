#include <bits/stdc++.h>
using namespace std;
int main(){
	int n,ans=0,d=0,r=0;
	char ar[1000000];
	scanf("%d %s",&n,ar+1);
	for(int i=1 ; i<=n ; i++){
		d=0;
		r=0;
		for(int j=i ; j<=n ; j++){
			if(ar[j]=='D')
				d++;
			if(ar[j]=='U')
				d--;
			if(ar[j]=='R')
				r++;
			if(ar[j]=='L')
				r--;
			if(!d && !r)
				ans++;
		}
	}
	printf("%d\n",ans);
	return 0;
}
