#include<bits/stdc++.h>
using namespace std;
int main()
{
	int x=0,y=0,ans=0,n;
	cin>>n;
	string a;
	cin>>a;
	for(int j=0;j<a.size();j++)
	{
		x=0;
		y=0;
	for(int i=j;i<a.size();i++)
	{
		if(a[i]=='U')y++;
		if(a[i]=='D')y--;
		if(a[i]=='L')x++;
		if(a[i]=='R')x--;
		if(x==0&&y==0)ans++;
	}
    }
	cout<<ans;
}
