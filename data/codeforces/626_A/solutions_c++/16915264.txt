#include<cstdio>
#include<cstring>


int main()
{
	int n,i,j,u,r,d,l;
	__int64 sum;
	char c[205];
	sum=0;
	scanf("%d",&n);
	scanf("%s",&c);
	for(i=0;i<n;i++)
	{
		u=r=d=l=0;
		for(j=i;j<n;j++){
			if(c[j]=='U')u++;
			if(c[j]=='R')r++;
			if(c[j]=='D')d++;
			if(c[j]=='L')l++;
			if(u==d&&r==l)sum++;
		}
	}
	printf("%I64d\n",sum);
	return 0;
}
