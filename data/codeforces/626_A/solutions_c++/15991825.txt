#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,i;
	scanf("%d",&n);
 	char p[201];
	scanf("%s",p);
	int ans=0;
	for(i=0;i<n;i++)
		for (int j=i+1;j<n;j++)
		{
			int a1=0,a2=0,a3=0,a4=0;
			for(int l=i;l<=j;l++)
			{
				if(p[l]=='U') a1++;
				else if (p[l]=='D') a1--;
				else if(p[l]=='R') a2++;
				else a2--;
			}
			if(a1==0 and a2==0) ans++;
		}
	printf("%d\n",ans);
}
