#include <stdio.h>
int main()
{
	int n,x,y,a,b,sum;
	char z[205];
	sum=0;
	scanf("%d",&n);
	scanf("%s",&z);
	for(a=0;n>a;a++){
		x=y=0;
		for(b=a;n>b;b++){
			if(z[b]=='U'){
				y++;
			}
			if(z[b]=='D'){
				y--;
			}
			if(z[b]=='L'){
				x--;
			}
			if(z[b]=='R'){
				x++;
			}
			if(x==0&&y==0){
				sum++;
			}
		}		
	}
	printf("%d\n",sum);
	return 0;
}
