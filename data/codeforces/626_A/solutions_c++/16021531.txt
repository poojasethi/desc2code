#include<bits/stdc++.h>

using namespace std;

int s[98] , ans , n;

string c;

int main() {
	cin >> n >> c;
	for (int i = 0; i < n; i++)
		for (int j = i + 1; j < n; j++) {
			fill (s , s + 98 , 0);
			for (int k = i; k <= j; k++)
				s[c[k]]++;
			if (s[76] == s[82] && s[68] == s[85])
				ans++;
		}
	cout << ans;		
	return 0;
}
