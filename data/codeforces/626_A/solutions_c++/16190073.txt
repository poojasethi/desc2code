#include<stdio.h>

int main()
{
	int n;
	scanf("%d",&n);
	char str[201];
	scanf("%s",str);
    int i,j,l=0,r=0,u=0,d=0,ans=0;
    for(i=0;i<n;i++)
	{
		l=0,r=0,u=0,d=0;
		for(j=i;j<n;j++)
		{
			if(str[j]=='L') l++;
			else if(str[j]=='R') r++;
			else if(str[j]=='U') u++;
			else if(str[j]=='D') d++;
			if(l==r && u==d)
				ans++;
		}
	}
	printf("%d\n",ans);

	return 0;
}
