#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,i,count1=0,x=0,y=0,j;
cin>>n;
string str;
cin>>str;
for(i=0;i<n;i++)
{
for(j=i;j<n;j++)
{
    if(str[j]=='R')
        x++;
    else if(str[j]=='L')
        x--;
    else if(str[j]=='U')
        y++;
    else
        y--;
    if(y==0&&x==0)
        count1++;

}
x=0;
y=0;
}
cout<<count1<<endl;
return 0;
}
