#include<iostream>
using namespace std;
int main()
{ int n,x=0,y=0,k=0; cin>>n;
  string s; cin>>s;
  for(int i=0;i<n;i++)
  { x=0; y=0;
    for(int j=i;j<n;j++)
    { if(s[j]=='R') x++;
      if(s[j]=='L') x--;
      if(s[j]=='U') y++;
      if(s[j]=='D') y--;
      if(x==0 && y==0) k++;
    }
  }
  cout<<k;
}
