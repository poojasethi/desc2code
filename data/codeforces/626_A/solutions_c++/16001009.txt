#include<bits/stdc++.h>

using namespace std;
   int r, d, u, l;
int main () {
 int n; 
   int ans=0;
	string s;
cin>>n;
cin>>s;
 for (int i=0; i<n; i++){
   for (int j=i; j<n; j++){
  		if(s[j]=='U') u++;
  		if(s[j]=='D') d++;
       	if(s[j]=='R') r++;
		if(s[j]=='L') l++;
	   
	   	if(u==d && r==l) ans++;
	 
	   }
 u=0; d=0; r=0; l=0;
}


cout<<ans;
return 0;
}
