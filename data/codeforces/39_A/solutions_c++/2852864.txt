#include <cstdio>
#include <iostream>
#include <string>
#include <vector>
#include <cstdlib>
#include <algorithm>
#define ll long long
#define pb push_back

using namespace std;

string s;
ll a, Opt;
ll ans;
vector<ll> f;

int main()
{
	//freopen( "39A.in", "r", stdin ); freopen( "39A.out", "w", stdout );
	cin>>a>>s;
	s = '+' + s;
	
	for ( int i = 0; i < s.length(); i += 3 )
	{
		if ( s[i++] == '+' ) Opt = 1; else Opt = -1;
		
		int x = 0;
		if ( s[i] <= '9'&& s[i] >= '0' )
		{
			while ( s[i] <= '9'&& s[i] >= '0' )
			x = x*10 + s[i] - '0', ++i;
		}
		else x = 1, --i;
		
		++i;
		x = x * Opt;
		if ( s[i] == 'a' )	ans -= x;
		ans += x * a;
		f.pb( x );
	}
	
	sort( f.begin(), f.end() );
	for ( vector<ll>::iterator i = f.begin(); i != f.end(); ++i )
	ans += (i-f.begin()+1)*(*i);
	
	cout<<ans;
	
	//fclose( stdin ); fclose( stdout );
	return 0;
}
