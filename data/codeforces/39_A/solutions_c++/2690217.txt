#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;

int a,ans,n;
int res[100010];
char s[100010];

int main()
{
	//freopen("a.in","r",stdin);
	//freopen("")
	scanf("%d",&a);
	scanf("%s",s+1);
	int len=strlen(s+1);
	
	int pos=1,sign=1;
	for (;pos<=len;)
	{
		if (s[pos]=='-') sign=-1, pos++;
		else if (s[pos]=='+') sign=1, pos++;
		int x=0;
		for (;s[pos]<='9' && s[pos]>='0' && pos<=len;pos++) x=x*10+s[pos]-'0';
		//x*=sign;
		
		if (s[pos]=='*')
		{
			x*=sign;
			pos++;
			if (s[pos]=='+') ans+=x;
			res[++n]=x;
		}
		else
		{
			x=sign;
			if (s[pos]=='+') ans+=x;
			res[++n]=x;
		}
		pos+=3;
	}
	sort(res+1,res+n+1);
	for (int i=1; i<=n; i++) ans+=(a+i-1)*res[i];
	printf("%d\n",ans);
	//for (;;);
}
