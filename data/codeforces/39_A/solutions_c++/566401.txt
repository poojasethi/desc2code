#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cctype>
#include <cstring>
using namespace std;
char s[10001];
__int64 re=0;
bool isnum(char c){
	return c>='0'&&c<='9'||c=='-';
}
int read(int x,int &m){
	if (!isnum(s[x]))
	{
		m=1;
		if (s[x]!='a')
		{
			re+=1;
		}
		return x+3;
	}
	if (!isnum(s[x+1])&&s[x]=='-')
	{
		m=-1;
		if (s[++x]!='a')
		{
			re-=1;
		}
		return x+3;
	}
	sscanf(s+x,"%d",&m);
	while(isnum(s[x])){x++;}
	if (s[++x]!='a')
	{
		re+=m;
	}
	return x+3;
}
int an[10001];
int main(){
	int a;
	scanf("%d",&a);
	scanf("%s",s);
	int m;
	int n=strlen(s);
	int i=0,j=0;
	while(i<n)
	{
		i=read(i,an[j++]);
		if (s[i]=='+')
		{
			i++;
		}
	}
	sort(an,an+j);
	for (i=0;i<j;i++)
	{
		re+=an[i]*a++;
	}
	printf("%d\n",re);
	return 0;
}
