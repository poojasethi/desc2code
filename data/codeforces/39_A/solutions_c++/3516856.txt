#include<stdio.h>
int main(){
    char s[100000];
    int e;
    scanf("%d%s",&e,s);
    char* s1=s;
    int mul=1;
    int a[50];
    int an=0;
    int d=0;
    while(*s1){
        int num=0;
        if (*s1>='0'&&*s1<='9'){
            while (*s1!='*')
                num=num*10+*s1-'0',s1++;
            a[an]=mul*num;
            s1++;
            
        }else
            a[an]=mul*1;
        
        
        if (*s1=='+')
            d+=a[an];
        s1+=3;
        if (*s1=='+')
            mul=1;else
        if (*s1=='-')
            mul=-1;
        
        an++;
        if (*s1==0)
            break;
        s1++;
        
    }
    for (int i=0;i<an;i++)
        for (int j=i+1;j<an;j++)
            if (a[i]>a[j])
            {
                int t=a[i];
                a[i]=a[j];
                a[j]=t;
            }
    int sum=d;
    for (int i=0;i<an;i++)
        sum+=a[i]*(e+i);
        printf("%d",sum);
    return 0;
}
