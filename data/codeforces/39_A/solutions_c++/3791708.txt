int main(void) {

cin>>a;
cin>>s;
s="+"+s;
vector<int>v;
int n=s.size();
int ans=0;
for(int i=0;i<n;i+=3){
  int flag=(s[i++]=='+')?1:-1;
  int x=0;
  if(s[i]>='0'&&s[i]<='9'){
    while(s[i]>='0'&&s[i]<='9')
      x=x*10+s[i++]-'0';
      i++;
 }
 else
  x=1;
 x=x*flag;
 v.push_back(x);
 if (s[i]=='a')
    ans-=x;
}
sort(v.begin(),v.end());
for(int i=0;i<v.size();i++){
 ans+=(i+a+1)*v[i];
}
printf("%d",ans);
puts("");
return 0;
}
