#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<cstring>

using std::sort;

const int maxn=1005,maxm=maxn*10;
int p[maxm],g[maxm],a[maxn];
char s[maxm];
bool v[maxm];

int main(){
  int m=0,n,ans=0,i,C;
  scanf("%d\n",&C);
  gets(s),n=strlen(s);
  for(i=0;i<n;++i)
    if(s[i]=='a')
      if(i>=2&&s[i-1]=='+'&&s[i-2]=='+'&&!v[i-1]&&!v[i-2])
        ans+=a[m++]=(i>=3&&s[i-3]=='*')?p[i-4]*g[i-4]:i>=3?g[i-3]:1;
      else
        v[i+1]=v[i+2]=1,a[m++]=(i>=1&&s[i-1]=='*')?p[i-2]*g[i-2]:i?g[i-1]:1;
    else if(s[i]=='+')g[i]=1;
    else if(s[i]=='-')g[i]=-1;
    else if(s[i]>='0'&&s[i]<='9')
      p[i]=(i?p[i-1]*10:0)+s[i]-48,g[i]=i?g[i-1]:1;
  sort(a,a+m);
  for(i=0;i<m;++i)ans+=(i+C)*a[i];
  printf("%d",ans);
  return 0;
}
