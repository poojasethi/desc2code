#include <stdio.h>
#include <cstdlib>
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

#define F first
#define S second
#define mp make_pair
typedef pair<int,int> pii;

int charge = 1, a, n, i, j, l, coef, p, ans, maxi, mini;
pii sum[1100];
string s;
bool used[1100];

void Find()
{
    coef = 0;
    if (('0' <= s[i] && s[i] <= '9'))
    {
        while (('0' <= s[i] && s[i] <= '9'))
            coef = coef * 10 + (s[i++] - '0');
        i++;
    }
    else coef = 1;
    if (i == n) return;
    p = (s[i] == '+');
    i += 3;
}

int main()
{
    cin >> a >> s;
    n = s.size();
    while (1)
    {
        Find();
        sum[l++] = mp(charge * coef, p);
        if (i == n) break;
        if (s[i++] == '+')
            charge = 1;
        else
            charge = -1;
    }
    sort(sum, sum + l);
    for (i = 0; i < l; i++)
        ans += sum[i].F * (a++ + sum[i].S);
    cout << ans;
    return 0;
}
