#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
const int maxn=10005;

int l[maxn];
int ac[maxn],av[maxn];
char s[maxn];
int tot;
int a,ans;
bool cmp(int x,int y){
	return av[x]<av[y];
}
int main(){
#ifndef ONLINE_JUDGE
	freopen("1.in","r",stdin);
	freopen("1.out","w",stdout);
#endif
	int p,v;
	scanf("%d",&a);
	scanf("%s",s);
	int len=strlen(s);
	if(s[0]=='-') p=-1;
	else p=1;
	for(int i=0;i<len;){
		v=0;
		if(s[i]>='0' && s[i]<='9')
			for(;s[i]>='0' && s[i]<='9';i++)
				v=v*10+s[i]-'0';
		else v=1;
		if(s[i]=='*') i++;
		av[++tot]=v*p;
		ac[tot]=s[i]!='a';
		i+=3;
		if(s[i]=='-') p=-1;
		else p=1;
		i++;	
	}
	for(int i=1;i<=tot;i++)
		l[i]=i;
	sort(l+1,l+1+tot,cmp);
	for(int i=1;i<=tot;i++){
		int k=l[i];
		if(ac[k]) ans+=av[k]*++a;
		else ans+=av[k]*a++;
	}
	printf("%d\n",ans);
	return 0;
}
