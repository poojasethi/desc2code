#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char s[10002];
int b[1001];
int main(){
	int a,n,i,j,sgn,ans;
	while(~scanf("%d%s",&a,s+1)){
		s[0]='+';ans=n=0;
		for(i=0;s[i];i+=3){
			sgn=s[i++]=='+'?1:-1;
			for(j=0;s[i]>='0'&&s[i]<='9';i++)j=j*10+s[i]-'0';
			if(s[i]=='*')i++;
			else j=1;
			b[++n]=(j*=sgn);
			ans+=(a-(s[i]=='a'))*j;
		}
		sort(b+1,b+n+1);
		for(i=1;i<=n;i++)ans+=i*b[i];
		printf("%d\n",ans);
	}
	return 0;
}
