#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define ft first 
#define sd second
using namespace std;
typedef pair<int,int> PR;
int n,m=0,len,ans=0;
PR a[110000];
char s[110000];
int G(int x,const int y)
{
    if ( s[y]!='*' ) return s[x]=='-'?-1:1;
    int k=1,p=0;
    if ( s[x]=='+' ) ++x;
    else if ( s[x]=='-' ) ++x,k=-1;
    for(int i=x;i<y;++i)
        p=p*10+s[i]-'0';
    return k*p;
}
int main()
{
    scanf("%d",&n);
    scanf("%s",s+1);
    len=strlen(s+1);
    int i=1;
    if ( s[1]!='-' ) s[--i]='+';
    for(;i<=len;)
    {
        int k=i;
        while ( s[k]!='a' ) ++k;
        if ( k>i+2 && s[k-1]=='+' &&  s[k-2]=='+' )a[++m]=make_pair(G(i,k-3),2),i=k+1;
            else a[++m]=make_pair(G(i,k-1),1),i=k+3;
    }
    sort(a+1,a+1+m);
    for(int i=1;i<=m;++i)
    {
        if ( a[i].sd==2 ) ++n;
        ans+=a[i].ft*n;
        if ( a[i].sd==1 ) ++n;
    }
    printf("%d\n",ans);
    return 0;
}
