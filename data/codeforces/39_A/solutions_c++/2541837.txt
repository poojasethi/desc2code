#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
struct pp{
    int co;
    bool ty;
}   a[10010];
int t,l,ans,n;
char s[10010];
bool cmp(pp a,pp b){
    return a.co<b.co;
}
int main(){
    //freopen("AC.in","r",stdin);
    //freopen("AC.out","w",stdout);
    scanf("%d\n",&n);
    scanf("%s",s);
    l=strlen(s);
    t=0;
    for(int i=0;i<l;){
        int sym=1,x=0,f=0;
        if((s[i]=='+'&&s[i+1]!='+')||(s[i]=='+'&&s[i+1]=='+'&&s[i+2]=='+'))i++;
        else if(s[i]=='-')sym=-1,i++;
        while(s[i]<='9'&&s[i]>='0'){
            f=1;
            x=x*10+s[i]-48;
            i++;
        }
        if(s[i]=='*')i++;
        if(!f)x=1;
        x*=sym;
        a[t].co=x;
        if(s[i]=='+')a[t++].ty=0;
        else a[t++].ty=1;
        i+=3;
    }
    sort(a,a+t,cmp);
    for(int i=0;i<t;i++)
        if(a[i].ty)ans+=a[i].co*n++;
        else ans+=a[i].co*++n;
    printf("%d\n",ans);
    return 0;
}
