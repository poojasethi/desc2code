#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

int z[10100],v,ans;

char s[10100];

bool isnum(char c)
{
	return (c<='9' && c>='0') || c=='-';
}

int get(int p,int &v)
{
	if (!isnum(s[p]))
	{
		v=1;
		if (s[p]!='a') ans++;
		return p+3;
	}
	if (!isnum(s[p+1]) && s[p]=='-')
	{
		v=-1;
		if (s[p+1]!='a') ans--;
		return p+4;
	}
	sscanf(s+p,"%d",&v);
	while (isnum(s[p]))
		p++;
	p++;
	if (s[p]!='a') ans+=v;
	return p+3;
}

int main()
{
	scanf("%d%s",&v,s+1);
	int l=strlen(s+1);
	int nowp=1;
	int cnt=0;
	while (nowp<=l)
	{
		cnt++;
		nowp=get(nowp,z[cnt]);
		if (s[nowp]=='+') nowp++;
	}
	sort(z+1,z+cnt+1);
	for (int a=1;a<=cnt;a++)
		ans+=z[a]*v++;
	printf("%d\n",ans);

	return 0;
}
