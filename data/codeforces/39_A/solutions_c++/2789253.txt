#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int mn=12000;
struct node{long long v;bool c;}a[mn];
long long L,n,st,ans;
char s[mn];
bool cmp(node a,node b){return a.v<b.v;}
int main()
{
    scanf("%I64d",&st);
    scanf("%s",s+1);
    long long i,j,p,v;
    n=0,L=strlen(s+1);
    s[0]='+';
    for(i=0;i<=L;i+=3)
    {
        v=0,p=s[i++]=='+'?1:-1;
        if(s[i]>='0'&&s[i]<='9')
            for(;s[i]>='0'&&s[i]<='9';++i)v=v*10+s[i]-48;
        else v=1;
        if(s[i]=='*')++i;
        ++n;
        a[n].v=v*p,a[n].c=s[i]=='a';
    }
    sort(a+1,a+1+n,cmp);
    for(i=1;i<=n;++i)
        if(a[i].c)ans+=a[i].v*st++;
        else ans+=a[i].v*++st;
    printf("%I64d\n",ans);
    return 0;
}
