using namespace std;
#define li       	long long int
#define rep(i,to)	for(li i=0;i<((li)(to));i++)
#define pb       	push_back
#define sz(v)    	((li)v.size())
#define bit(n)   	(1ll<<(li)(n))


vector<pair<li,li> > vec;
int main(){
	li a,res=0;
	string exp;
	cin>>a>>exp;
	li minus=false;
	rep(i,sz(exp)){
		if(i!=0) minus=(exp[i++]=='-');
		li co=1;
		if(isdigit(exp[i])){
			co=0;
			while(isdigit(exp[i])) co=co*10+exp[i++]-'0';
			i++;
		}
		if(minus) co=-co;
		if(exp.substr(i,3)=="++a") vec.pb(make_pair(co,0));
		else vec.pb(make_pair(co,1));
		i+=2;
	}
	sort(vec.begin(),vec.end());
	rep(i,sz(vec)){
		if(vec[i].second==0) a++;
		res+=a*vec[i].first;
		if(vec[i].second==1) a++;
	}
	cout<<res<<endl;
}
