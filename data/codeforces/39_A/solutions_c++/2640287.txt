#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
string s;
int n,ans,i,k,sw,num,a[10000];

int main()
{
//	freopen("1.in","r",stdin);
//	freopen("1.out","w",stdout);
	cin>>n>>s;
	if (s[0]!='-')s='+'+s;
	//cout<<s<<endl;
	for(i=0;i<s.length();)
	{
		sw=s[i++]=='-'?-1:1;
		k=0;
		for(;s[i]>='0'&&s[i]<='9';i++)
		k=k*10+s[i]-'0';
		if(s[i]=='*')i++;
		else k=1;
		a[++num]=k*sw;
		if (s[i]=='a')ans-=a[num];
		i+=3;
	//	cout<<a[num]<<' '<<ans<<endl;
	}
	sort(a+1,a+num+1);
	for(i=1;i<=num;i++)
//	{
	ans+=a[i]*(i+n);
	//cout<<ans<<' '<<(a[i]*(i+n))<<endl;
//	}
	cout<<ans;
}
