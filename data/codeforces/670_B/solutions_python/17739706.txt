from sys import stdin, stdout
import math

n, k = [int(x) for x in stdin.readline().rstrip().split(" ")]
ids = [int(x) for x in stdin.readline().rstrip().split(" ")]
d = (-1 + math.sqrt(1 + (k<<3))) / 2
s = int(math.ceil(d)) if d > 0 else 0

sum = (s >> 1) * (s + 1) if s % 2 == 0 else s * ((s + 1) >> 1)

print ids[s - sum + k - 1]
