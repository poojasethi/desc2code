# -*- coding: utf-8 -*-

import math


def row(n):
    return int(math.ceil((math.sqrt(1+8*n)-1)/2))

def number(n):
    return n*(n+1)/2


if __name__ == '__main__':
    n, k = map(int, raw_input().split())
    ids = raw_input().split()

    r = row(k)
    previous = number(r-1)
    index = k-previous-1
    result = ids[index]
    print(result)
