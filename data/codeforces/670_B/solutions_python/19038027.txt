n, k = map(int, raw_input().split(" "))
num = map(int, raw_input().split(" "))
menor, mayor, aux = 0, 0, 0
def buscar(numero):
    print num[numero]
    exit()
    return 0
for i in range(n + 1):
    menor = mayor + 1
    mayor = mayor + aux
    aux += 1
    if menor <= k <= mayor:
        buscar(k - menor)
