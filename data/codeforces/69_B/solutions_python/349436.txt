import sys

n, m = map(int, sys.stdin.readline().strip().split(' '))
A = [map(int, sys.stdin.readline().strip().split(' ')) for i in xrange(m)]

s = 0
for i in xrange(1, n+1):
  m = sys.maxint
  t = 0
  for a in A:
    if a[0] <= i <= a[1] and a[2] < m:
      m = a[2]
      t = a[3]
  s += t
print s
