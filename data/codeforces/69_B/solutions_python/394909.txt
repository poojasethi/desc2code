nums = raw_input().split()
n = int(nums[0])
m = int(nums[1])
ans = 0

tm = [5000]*n
win = [-1]*n

c = [0]*m

for i in range(m):
    nums = raw_input().split()
    lf = int(nums[0]) - 1
    rg = int(nums[1]) - 1
    t  = int(nums[2])
    c[i] = int(nums[3])

    for j in range(lf, rg + 1):
        if tm[j] > t:
            tm[j] = t;
            win[j] = i;

for i in range(n):
    if win[i] != -1:
        ans = ans + c[win[i]]

print ans
