#!/usr/bin/env python
# -*- coding: utf-8 -*-

from collections import namedtuple

Man = namedtuple("Man", ['n', 'l', 'r', 't', 'c'])
mans = list()
ans = 0
n, m = map(int, raw_input().split())
for i in xrange(m):
    mans.append(Man(*([i]+map(int, raw_input().split()))))
for x in xrange(1, n+1):
    try:
        ans += min([man for man in mans if man.l <= x <= man.r], key=lambda man: (man.t, man.n)).c
    except ValueError:
        pass
print ans
