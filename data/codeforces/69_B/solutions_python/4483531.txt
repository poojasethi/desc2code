I=lambda:map(int, raw_input().split())
n, m = I()
a = [[1000000000, 20000000, 0] for _ in xrange(n+1)]
for _ in xrange(m):
    l, r, t, c = I()
    for i in xrange(l, r+1):
        if t < a[i][0] or (t == a[i][0] and i <a[i][1]):
            a[i][0], a[i][1], a[i][2] = t, i, c
print sum([x[2] for x in a])
