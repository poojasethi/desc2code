n,m = map(int, raw_input().split())
i=0
cost=[]
time=[]
while i<=1000000:
        cost.insert(i,0)
        time.insert(i,10000000)
        i+=1
    
while m>0:
    a,b,c,d = map(int, raw_input().split())
    i=a
    while i<=b:
        if time[i]>c:
            time[i]=c
            cost[i]=d
        i+=1  
    m-=1

i=1
sum1=0
while i<=n:
    sum1+=cost[i]
    i+=1
print sum1
