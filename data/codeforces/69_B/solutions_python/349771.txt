n, m = [int(_1) for _1 in raw_input().split()]
l,r,t,c = [],[],[],[]
for i in xrange(1, m + 1):
    _l, _r, _t, _c = [int(_2) for _2 in raw_input().split()]
    l.append(_l)
    r.append(_r)
    t.append(_t)
    c.append(_c)
res = 0
for i in xrange(1, n + 1):
    time = 2000
    profit = 0
    for j in xrange(0, m):
        if (l[j] <= i and i <= r[j] and t[j] < time):
            time = t[j]
            profit = c[j]
    res += profit
print res
