n,m = map(int,raw_input().split())
u,b = [[] for i in xrange(n)],[]
for i in xrange(m):
    b.append(map(int,raw_input().split()))
for i in b:
    for j in xrange(i[0]-1,i[1]):
        u[j].append((i[2],i[3]))
money = 0
for i in u:
    if i:
        mt = min(i)[0]
        for j in i:
            if j[0]==mt:
                money += j[1]
                break
print money
