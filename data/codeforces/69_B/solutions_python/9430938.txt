n,m=map(int,raw_input().split())
ans=0
tm=[5000]*n
win=[-1]*n
p=[0]*m
for i in xrange(m):
    m-=1
    l,r,t,c=map(int,raw_input().split())
    p[i]=c
    for j in xrange(l-1,r):
        if tm[j]>t:
            tm[j]=t
            win[j]=i
for i in xrange(n):
    if win[i]!=-1:
        ans+=p[win[i]]
print ans
