m, n = map(int, raw_input().split())
a = []
b = [0, n]
for i in range(n):
    a.append(map(int, raw_input().split()))
    a[-1][0] -= 1
    b.append(a[-1][0])
    b.append(a[-1][1])
b = list(set(b))
b.sort()
res = 0
for i in range(1, len(b)):
    best = 1001
    profit = 0
    for k in a:
        if k[0] < b[i] and k[1] >= b[i] and k[2] < best:
            best = k[2]
            profit = k[3]
    res += profit * (b[i] - b[i - 1])
print res
