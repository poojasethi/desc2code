[n,m]=map(int,raw_input().split())
ans=[[1000000,0] for _ in xrange(n+1)]
for _ in xrange(m):
    [l,r,t,c] = map(int,raw_input().split())
    for i in xrange(l,r + 1):
        if t < ans[i][0]:
            ans[i][0] = t
            ans[i][1] = c
cnt = 0
for i in xrange(1,n + 1):
    cnt += ans[i][1]
print cnt
