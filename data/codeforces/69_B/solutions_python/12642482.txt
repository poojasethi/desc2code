n, m = map(int, raw_input().split())
f = [[0 for _ in range(2)] for _ in range(n + 1)]
for i in range(n + 1):
  f[i][0] = 10000
  f[i][1] = 0
for i in range(m):
  l, r, t, c = map(int, raw_input().split())
  for j in range(l, r + 1):
    if (f[j][0] > t):
      f[j][0] = t
      f[j][1] = c
res = 0
for i in range(1, n + 1):
  res += f[i][1]
