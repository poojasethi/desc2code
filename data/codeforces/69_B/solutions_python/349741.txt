#!/usr/bin/python

import sys, os
sys.setrecursionlimit(10000)

def readline():
	return sys.stdin.readline().strip()
def readrow():
	return readline().split(' ')

(n, m) = map(int, readrow())
ws = [(1001, 0, -1)] * n
as_ = []
for i in range(m):
	as_.append(tuple([i]+map(int, readrow())))
as_.reverse()
for a in as_:
	(i, l, r, t, c) = a
	for s in range(l, r+1):
		(time, profit, ath) = ws[s-1]
		if t <= time:
			ws[s-1] = (t, c, i)

print sum(map(lambda x: x[1], ws))
