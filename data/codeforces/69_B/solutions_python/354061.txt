import sys


lines = [i.rstrip() for i in sys.stdin.readlines()]
size = int(lines[0].split(" ")[0])
sections = [1001 for i in range(size)]
profits = [0 for i in range(size)]

for line in lines[1:]:
    s = line.split(" ")
    l = int(s[0])
    r = int(s[1])
    t = int(s[2])
    c = int(s[3])

    for i in range(l - 1, r):
        if sections[i] > t:
            sections[i] = t
            profits[i] = c
print sum(profits)
