from itertools import *
from collections import defaultdict

a, b = map(int, raw_input().split())
spo = list(enumerate([map(int, raw_input().split()) for i in xrange(b)]))
pol = defaultdict(list)

for i, data in spo:
    for j in xrange(data[0], data[1]+1):
        if pol[j] or pol[j] == 0:
            if spo[pol[j]][1][2] > data[2]:
                pol[j] = i
        else:
            pol[j] = i
profit = 0            
for i in pol:
    profit += spo[pol[i]][1][3]

print profit
