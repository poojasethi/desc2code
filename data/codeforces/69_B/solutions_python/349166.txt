

n,m = map(int, raw_input().split())

arr = []
for i in xrange(m):
    arr.append(map(int, raw_input().split()))
    arr[i].append(i)

summ = 0
for i in xrange(1,n+1):
    uch = []
    for sport in arr:
        if i in xrange(sport[0], sport[1]+1):
            uch.append((sport[2], sport[4], sport[3]))
    uch.sort(cmp=lambda x,y: x[0]!=y[0] and cmp(x[0],y[0]) or cmp(x[1], y[1]))
    try:
        summ += uch[0][2]
    except IndexError:
        pass
print summ
