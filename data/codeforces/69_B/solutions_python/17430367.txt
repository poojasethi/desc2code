section_count,player_count= map(int, raw_input().split())
from collections import namedtuple
Section = namedtuple('Section', ['winner','time','prize'])
S = {i: Section(0,float('inf'),0) for i in xrange(1,section_count+1)}
for i in xrange(1,player_count+1):
	l,r,t,c = map(int, raw_input().split())
	for j in xrange(l,r+1):
		if t < S[j].time:
			S[j] = Section(i, t, c)
print sum([s.prize for s in S.values()])
