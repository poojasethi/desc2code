#include <cstdio>
#include <cstring>

int main() {

	int n,m,l,r,t,c,i,s;
	int vt[101], vc[101];
	
	memset(vt, -1, sizeof(vt));
	memset(vc, 0, sizeof(vc));
	
	scanf("%d %d ", &n, &m);
	
	while(m--) {
	
		scanf("%d %d %d %d ", &l, &r, &t, &c);
		
		for (i=l; i<=r; i++) {
			if (vt[i]<0 || vt[i]>t) {
				vt[i]=t;
				vc[i]=c;
			}
		}
	
	}
	
	s=0;
	for(i=1; i<=n; i++)
		s += vc[i];
		
	printf("%d\n", s);
		

}
