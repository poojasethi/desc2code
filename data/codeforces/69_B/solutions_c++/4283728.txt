#include <iostream>
#include <cstdio>
using namespace std;
int n,m,i,l[101],r[101],t[101],c[101],j,ma,e,sum;
const int inf=1000000000;
int main()
{
cin>>n>>m;
for (i=0;i<m;i++)
cin>>l[i]>>r[i]>>t[i]>>c[i];
for (i=1;i<=n;i++)
{
	ma=inf;
	for (j=0;j<m;j++)
	{
	if (l[j]<=i && r[j]>=i)
	{
		if (t[j]<ma)
			ma=t[j],e=j;
	}
	}
    if (ma!=inf)
    sum+=c[e];
}
cout<<sum;


return 0;
}
