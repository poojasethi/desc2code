#include <iostream>
using namespace std;
int C[101],T[101];
int n,i,m,j,a,b,t,c,ans;
main(){
 cin>>n>>m;
 for (i=1;i<=n;i++) 
  T[i]=1001;
 
 for (i=1;i<=m;i++){
  cin>>a>>b>>t>>c;
  for (j=a;j<=b;j++)
   if (t<T[j]) {T[j]=t; C[j]=c;}
 }        
 
 for (i=1;i<=n;i++)
  ans+=C[i];
  
 cout<<ans<<endl;
 
// system("pause"); 
       
       
       
       
       
}
