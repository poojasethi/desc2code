#include<stdio.h>

int n,m;
int l[100],r[100],t[100],c[100];
int maxim;
int minim;

int main(){
 scanf("%d%d",&n,&m);
 int i,j;
 for(i=0;i<m;i++) scanf("%d%d%d%d",l+i,r+i,t+i,c+i);
 for(i=1;i<=n;i++){
  for(j=0;j<m;j++) if(l[j]<=i && r[j]>=i) break;
  if(j==m) continue;
  minim = j;
  for(j=j+1;j<m;j++){
   if(l[j]<=i && r[j]>=i){
    if(t[j] < t[minim]) minim =j;
   }
  }
  maxim += c[minim];
 }
 printf("%d\n",maxim);

 return 0;
}
