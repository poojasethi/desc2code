#include <iostream> 
using namespace std;
int main()
{
    long long i,j,n,m,l,r,t,c,sum=0,a[101][2];
	cin >> n >> m;
	for (i=0; i<n; i++) {a[i][1]=1001; a[i][2]=0;}
	for (i=0; i<m; i++){
	cin >> l >> r >> t >> c;
	for (j=l-1; j<r; j++)
		if (a[j][1]>t) {a[j][1]=t; a[j][2]=c;}
	}
	for (i=0; i<n; i++) {sum+=a[i][2];}
	cout << sum;
	return 0;
}
