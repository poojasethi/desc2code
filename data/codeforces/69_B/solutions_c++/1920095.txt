#include<iostream>
#define N 101
#define FOR(i,a,b) for(int i=a;i<=b;i++)
using namespace std;
int l[N],r[N],c[N],t[N];
int main()
{
    int n,m,q,min,sum=0;
    cin>>n>>m;
    FOR(i,1,m)
    cin>>l[i]>>r[i]>>t[i]>>c[i];
    FOR(i,1,n)
{
    q=0; min=10000;
    FOR(j,1,m)
    if (l[j]<=i && r[j]>=i && min>t[j]) {min=t[j];q=j;}
    if (q>0) sum+=c[q];
}
    cout<<sum;
    return 0;                       
}
