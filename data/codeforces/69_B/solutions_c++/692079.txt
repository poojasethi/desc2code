#include <stdio.h>

int w[105];
int c[105];
int main(void) {
	int n, m, ans=0;
	scanf("%d%d", &n, &m);
	for(int i=0;i<m;i++) {
		int l, r, t, v;
		scanf("%d%d%d%d", &l, &r, &t, &v);
		for(int j=l;j<=r;j++)
			if(w[j]==0 || w[j]>t) {
				ans += v-c[j];
				w[j]=t;
				c[j]=v;
			}
	}
	printf("%d\n", ans);
	return 0;
}
