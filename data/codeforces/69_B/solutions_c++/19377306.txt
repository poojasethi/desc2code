#include<iostream>
using namespace std;
int main(){
	int n,m,l,r,t[105],k[105],a[105]={0},p=0;
	cin>>n>>m;
	for(int i=1;i<=m;i++)
	{
		cin>>l>>r>>t[i]>>k[i];
		for(int j=l;j<=r;j++)
		{
			if(!a[j]||t[i]<t[a[j]]) 
				a[j]=i;
		}
	}
	k[0]=0;
	for(int i=1;i<=n;i++)
	{
		p+=k[a[i]];
	}
	cout<<p;
	return 0;
}
