#include <algorithm>
#include <numeric>
#include <iostream>
using namespace std;

int main() {
	int n, m, l, r, t, c;
	int tm[100];
	int cm[100];

	cin >> n >> m;

	fill_n(tm, n, 9999);
	fill_n(cm, n, 0);

	for (int i = 0; i < m; ++i) {
		cin >> l >> r >> t >> c;
		for (int j = l - 1; j <= r - 1; ++j)
			if (t < tm[j]) {
				tm[j] = t;
				cm[j] = c;
			}
	}

	cout << accumulate(cm, cm + n, 0);

	return 0;
}
