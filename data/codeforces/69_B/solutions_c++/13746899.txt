#include <iostream>
using namespace std;
int main()
{
	int n,m;
	int pr[110]={0};
	int ti[110]={0};
	int i;
	cin >> n >> m;
	int x,y,c,p,j;
	for(i=0;i<110;i++)
	{
		ti[i]=1000000000;
	}
	for(i=0;i<m;i++)
	{
		cin >> x >> y >> c >> p;
		for(j=x;j<=y;j++)
		{
			if(ti[j]>c)
			{
				ti[j]=c;
				pr[j]=p;
			}
		}
	}
	int sum=0;
	for(j=1;j<=n;j++)
	{
		sum=sum+pr[j];
	}
	cout << sum << endl;
	return 0;
}
