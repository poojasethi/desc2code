#include<cstdio>
#include<algorithm>
using namespace std;

int n,m,l,r,t,c;
pair<int,int> ans[1111];
int res;

int main() {
	scanf("%d%d", &n, &m);
	for(int i=1;i<=n;i++)
		ans[i]=make_pair(1111,0);
	for(int i=0;i<m;i++){
		scanf("%d%d%d%d", &l, &r, &t, &c);
		for(int j=l;j<=r;j++)
			if(ans[j].first>t)
				ans[j]=make_pair(t,c);
	}
	for(int i=1;i<=n;i++)
		res+=ans[i].second;
	printf("%d\n",res);

	return 0;
}
