#include <cstdio>
int n, m, r = 0, a, b, c, d, t[111], s[111];
int main() {
    scanf("%d%d", &n, &m);
    for (int i = 0;i < m;i++) {
        scanf("%d%d%d%d", &a, &b, &c, &d);
        for (int j = a;j <= b;j++) if (t[j] == 0 || c < t[j]) { t[j] = c; s[j] = d; }
    }
    for (int i = 1;i <= n;i++) if (t[i]) r += s[i];
    printf("%d\n", r);
}
