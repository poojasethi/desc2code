#include <stdio.h>

int main()
{
	int l[110],r[110],t[110],c[110];
	int t_min,c_max;
	int n,m,sum;
	scanf("%d%d",&n,&m);
	for (int i = 1; i<=m;i++)
		scanf("%d%d%d%d",&l[i],&r[i],&t[i],&c[i]);
	sum=0;
	for (int i = 1; i<=n;i++)
	{	
		c_max=0;
		t_min=10000;
		for (int j = 1; j<=m;j++)
		if (i>=l[j] && i<=r[j] && t[j]<t_min) 
		{
			c_max=c[j];
			t_min=t[j];
		}
		sum+=c_max;
	}
	printf("%d",sum);
	return 0;
}
