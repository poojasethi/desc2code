#include<bits/stdc++.h>
using namespace std;
int x[110],y[110];
int main()
{
    int n,m,l,r,t,c,ans=0;
    cin >> n >> m;
    for(int i=0;i<m;i++)
    {
    cin >> l >> r >> t >> c;
    for(int j=l;j<=r;j++)
        if(x[j]==0 or x[j]>t)
    {
        ans+=c-y[j];
        x[j]=t;
        y[j]=c;
    }
    }
   cout << ans;
   return 0;
}
