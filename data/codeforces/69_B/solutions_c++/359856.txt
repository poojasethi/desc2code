#include <iostream>
#include <vector>

using namespace std;

int main()
{
	int n, m;
	cin >> n >> m;
	vector <int> times(n, 1001), profit(n, 0);

	for(int i = 0; i < m; i++){
		int l, r, t, c;
		cin >> l >> r >> t >> c;
		for(int j = l - 1; j <= r - 1; j++){
			if(t < times[j]){
				times[j] = t;
				profit[j] = c;
			}
		}
	}

	int ans = 0;
	for(int i = 0; i < n; i++)
		ans += profit[i];
	cout << ans << endl;

	return 0;
}
