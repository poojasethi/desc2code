#include <iostream>
#include <cstdio>

using namespace std;

int n,m;
int a[110][4];

int main()
{
	cin >>n>>m;
	for(int i=0; i<m; i++)	
		for(int j=0; j<4; j++)
			cin >>a[i][j];

	int ans=0;
	for(int i=1; i<=n; i++)
	{
	    int tim=999999,t=0;
		for(int j=0; j<m; j++)
			if((a[j][0]<=i)&&(a[j][1]>=i)&&(a[j][2]<tim))	
				{
					tim=a[j][2];
					t=a[j][3];
				}
		ans+=t;
	}
	cout <<ans;
	return 0;
}
