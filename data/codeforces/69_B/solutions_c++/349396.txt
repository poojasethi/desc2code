#include<iostream>
#include<vector>
using namespace std;

pair<int,int> best[1000];

int main()
{
	int n,m;
	cin>>n>>m;
	while(m--)
	{
		int a,b,t,p;
		cin>>a>>b>>t>>p;
		for(int i = a; i <= b; i++)
			if(best[i].first > t || best[i].first == 0) best[i] = make_pair(t,p);
	}
	int sum = 0;
	for(int i = 1; i <= n; i++)
	{
		if(best[i].first!=0) sum += best[i].second;
	}
	cout<<sum<<endl;
}
