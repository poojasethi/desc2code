#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
int t[1005],c[1005];
int main()
{
    int n,m;
    cin>>n>>m;
    for(int i=0;i<=n;++i)
        t[i]=1005;
    for(int i=0;i<m;++i)
    {
        int l,r,a,b;
        cin>>l>>r>>a>>b;
        for(int j=l;j<=r;++j)
            if(a<t[j])t[j]=a,c[j]=b;
    }
    int ans=0;
    for(int i=1;i<=n;++i)
        ans+=c[i];
    cout<<ans<<endl;
}
