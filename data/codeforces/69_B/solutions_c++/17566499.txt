#include <iostream>
using namespace std;

long long l[101], r[101], t[101], c[101];

int main(){
	long long n, m, x, i, j, y, ans=0;
	cin >> n >> m;
	for(i=0;i<m;i++){
		cin >> l[i] >> r[i] >> t[i] >> c[i];
	}
	for(i=1;i<=n;i++){
		x = 10001;
		for(j=0;j<m;j++){
			if(l[j]<=i and r[j]>=i){
				if(t[j]<x){
					x = t[j];
					y = c[j];
				}
			}
		}
		if(x!=10001){
			ans+=y;
		}
	}
	cout << ans << '\n';
	return 0;
}
