#include <iostream>
using namespace std;
int l[105],r[105],c[105],t[105],n,m,ans=0,st[105],mint[105];
int main()
{
  cin>>n>>m;
  for(int i=0;i<m;i++)cin>>l[i]>>r[i]>>t[i]>>c[i];
  for(int i=1;i<=n;i++)mint[i]=1000000000;
  for(int i=0;i<m;i++)
    for(int j=l[i];j<=r[i];j++)
      if (t[i]<mint[j])
      {
        mint[j]=t[i];
        st[j]=c[i];
      }
  for(int i=1;i<=n;i++)ans+=st[i];
  cout<<ans;
  return 0;
}
