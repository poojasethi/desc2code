#include <iostream>
#include <vector>

using namespace std;

int main()
{
  int n,m;
  cin>>n>>m;
  vector<int> tiempo(n,10000);
  vector<int> gana(n,0);
  for (int i=0;i<m;i++) {
    int l,r,t,c;
    cin>>l>>r>>t>>c;
    for (int j=l;j<=r;j++) {
      if (t<tiempo[j-1]) {
	tiempo[j-1]=t;
	gana[j-1]=c;
      }
    }
  }
  int g=0;
  for (int i=0;i<n;i++) {
    g+=gana[i];
    //cout<<gana[i]<<",";
  }
  // cout<<endl;
  cout<<g<<endl;
}
