#include <cstdio>
int castig[3000],timp[3000];
const int inf=1<<29;
int main()
{
  int n,x,y,z,t,m,i,j,sum;
  scanf("%d%d",&n,&m);
  for (i=1;i<=n;i++)
    timp[i]=inf;
  for (i=1;i<=m;i++)
  {
    scanf("%d%d%d%d",&x,&y,&z,&t);
    for (j=x;j<=y;j++)
      if ( z< timp[j])
      {
        timp[j]=z;
        castig[j]=t;
      }
  }
  sum=0;
  for (i=1;i<=n;i++)
  {
    sum+=castig[i];
  }
  printf("%d\n",sum);
  return 0;
}
