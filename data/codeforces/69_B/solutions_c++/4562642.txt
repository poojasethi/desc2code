#include <iostream>
using namespace std;
int main(){
  int q,w,e,r,t,y,u,c,a[101],s[101];
  cin>>q>>w;
  for(e=1;e<=q;e++){
  a[e]=2000;
  s[e]=0;}
  
  for(e=0;e<w;e++){
  cin>>r>>t>>y>>u;
  for(c=r;c<=t;c++)
  if(a[c]>y){
   a[c]=y;
   s[c]=u;}}
  r=0;
  for(e=1;e<=q;e++)
  r+=s[e];
  cout<<r;
  return 0;}
