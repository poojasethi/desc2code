#include<cstdio>

const int maxn=10000+10;
int t[maxn],c[maxn];
int n,m;

int main()
{
    scanf("%d%d",&n,&m);
    for (int i=1;i<=n;i++) t[i]=1000000,c[i]=0;
    for (int l,r,tt,cc;m--;)
    {
        scanf("%d%d%d%d",&l,&r,&tt,&cc);
        for (int i=l;i<=r;i++)
        if (tt<t[i]) t[i]=tt,c[i]=cc;
    }
    int res=0;
    for (int i=1;i<=n;i++) res+=c[i];
    printf("%d\n",res);
}
