#include<iostream>
using namespace std;
struct a{
	int ti,in;
	a(int b=10000,int c=0){
		ti=b;
		in=c;
	}
}aa[105];
int main(){
	int n,m;
	cin>>n>>m;
	int l,r,t,c;
	int i,j;
	for(i=0;i<m;i++){
		cin>>l>>r>>t>>c;
		for(j=l-1;j<r;j++){
			if(aa[j].ti>t){
				aa[j].ti=t;
				aa[j].in=c;
			}
		}
	}
	int cnt=0;
	for(i=0;i<n;i++) cnt+=aa[i].in;
	cout<<cnt<<endl;
}
