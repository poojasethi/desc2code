#include<stdio.h>
int main(){
	int a[110][2],n,m,i,j,l,r,t,c,sum=0;
	scanf("%d%d",&n,&m);
	for(i=0;i<=n;i++){
		a[i][0]=10000;
		a[i][1]=0;
	}
	for(i=0;i<m;i++){
	    scanf("%d%d%d%d",&l,&r,&t,&c);
	    for(j=l;j<=r;j++)
	        if(a[j][0]>t){
				a[j][0]=t;
				a[j][1]=c;
			}
	}
	for(i=1;i<=n;i++)
		sum+=a[i][1];
	printf("%d\n",sum);
	return 0;
}
