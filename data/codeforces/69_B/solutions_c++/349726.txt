#include<stdio.h>

long l,r,t,c,i,j,n,m,s;
long A[105]={0},b[105]={0};

main()
{
  
  scanf("%ld%ld",&n,&m);
  for (i=1;i<=m;i++)
  {
    scanf("%ld%ld%ld%ld",&l,&r,&t,&c);  
    for (j=l;j<=r;j++)
    {
      if (A[j]==0) {A[j]=t;b[j]=c;}
        else if (A[j]>t) {A[j]=t;b[j]=c;}  
    }
  }
  for (i=1;i<=n;i++)
    s=s+b[i];
  printf("%ld\n",s);  
}
