#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <queue>
#include <map>
using namespace std;

priority_queue<int, vector<int>, greater<int> > q1, q2;
int n;
map<int, int> m;
long long ans = 0;

int main()
{
	scanf("%d", &n);
	int x;
	for(int i = 1; i <= n; ++i)
	{
		scanf("%d", &x);
		ans += x;
		++m[x];
	}
	int now = 0;
	map<int,int> :: reverse_iterator it;
	for(it = m.rbegin(); it != m.rend(); ++it)
	{
		int a = it->first;
		int b = it->second;
		while(b && now)
		{
			q2.push(a);
			--b;
			--now;
		}
		while(b >= 2 && !q1.empty() && q1.top() < 2 * a)
		{
			if(q1.top() < a)
			{
				q2.push(a);
				q2.push(a);
			}
			else
			{
				q2.push(a + a - q1.top());
				q2.push(q1.top());
			}
			q1.pop();
			b -= 2;
		}
		
		if(b && !q1.empty() && q1.top() < a)
		{
			q1.pop();
			q2.push(a);
		}
		while(!q2.empty())
		{
			q1.push(q2.top());
			q2.pop();
		}
		now += b;
	}
	while(!q1.empty())
	{
		ans -= q1.top();
		q1.pop();
	}
	printf("%I64d\n", ans);
	
	return 0;
}
