from math import *
from Queue import *


def rotate(s,l,r,k):
    new = ''
    for i in range(r-l):
        new += s[(i-k+r-l)%(r-l)+l]
    return s[:l] + new + s[r:]


s = raw_input()
m = int(raw_input())
for i in range(m):
    l = map(int, raw_input().split())
    s = rotate(s,l[0]-1,l[1],l[2])
print(s)
