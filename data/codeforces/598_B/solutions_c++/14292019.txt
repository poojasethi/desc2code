#include<bits/stdc++.h>
using namespace std;

int main(){
string s;
cin>>s;
int T,l,r,k,x;
cin>>T;
while(T--){
    cin>>l>>r>>k;
    l--;r--;
    string s1=s;
    x=r-l+1;
    k%=x;
    for(int i=l;i<=r;i++){
        if(i+k<=r) s1[i+k]=s[i];
        else s1[i+k-x]=s[i];
    }
    s=s1;
}
cout<<s<<endl;
return 0;
}
