#include <iostream>
#include <string>
using namespace std;

int main(){
	long l,r,k,m,x;
	string S;
	cin >> S >> m;
	while(m--){
		cin >> l >> r >> k;
		l--;r--;
		x = r-l+1;
		k = k%x;
		char A[x];
		for(int i = l; i <= r; ++i){
			A[(i-l+k)%x] = S[i];
		}
		for(int i = l; i <= r; ++i){
			S[i] = A[i-l];
		}
	}
	cout << S << endl;
	return 0;
}
