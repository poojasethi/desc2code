#include<stdio.h>
int main()
{
	char s[10005], s2[10005];
	int m, l, r, k, i, j;
	scanf("%s", s);
	scanf("%d", &m);
	while (m--)
	{
		scanf("%d%d%d", &l, &r, &k); j = r - l + 1;
		for (i = 0; i <= r-l; i ++)
			s2[(i + k) % j] = s[i+l-1];
		for (i = 0; i <= r - l; i++)
			s[i + l - 1] = s2[i];
	}
	printf("%s\n", s);
	return 0;
}
