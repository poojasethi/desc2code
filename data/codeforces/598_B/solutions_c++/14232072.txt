#include <bits/stdc++.h>
using namespace std;
int main() {
	string s, t;
	int m, l, r, k, i, d, j;
	cin >> s;
	cin >> m;
	while(m--) {
		cin >> l >> r >> k;
		d = r - l + 1;
		t.resize(d);
		for(i = l - 1, j = 0; i < r; i++, j++) t[(k + j) % d] = s[i];
		for(i = l - 1, j = 0; i < r; i++, j++) s[i] = t[j];
	}
	cout << s << endl;
}
