#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main() {
    string s;
    cin>>s;
    int m;
    cin>>m;
    int l,r,k;
    while(m--) {
        cin>>l>>r>>k;
        int t=(r-l+1)-k%(r-l+1);
        rotate(s.begin()+l-1,s.begin()+l-1+t,s.begin()+r);
    }
    cout<<s<<endl;
    return 0;
}
