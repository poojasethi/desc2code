#include<iostream>
using namespace std;
int main()
{
    string s;
    int m;
    cin>>s>>m;
    for(int i=0;i<m;i++)
    {
        int a,b,c;
        cin>>a>>b>>c;
        c%=(b-a+1);
        s=s.substr(0,a-1)+s.substr(b-c,c)+s.substr(a-1,b-a+1-c)+s.substr(b);
    }
    cout<<s<<endl;
    return 0;
}
