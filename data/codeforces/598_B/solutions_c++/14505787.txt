#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
char c[10005],a[10005];
int m,l,r,k,j,i;
int main()
{
	cin>>a+1;
	cin>>m;
	for (i=1;i<=m;i++)
	{
		cin>>l>>r>>k;
		k%=(r-l+1);
		for (j=l;j<=r;j++)
			c[(j-l+k)%(r-l+1)+l]=a[j];
		for (j=l;j<=r;j++) a[j]=c[j];
	}
	cout<<a+1;
}
