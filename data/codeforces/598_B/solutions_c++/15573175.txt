#include <iostream>
#include <string>
using namespace std;

int main()
{
	string s,s1;
	int i,m,l,j,r,k;
	cin>>s;
	cin>>m;
	for (i=1;i<=m;i++)
	{
		s1="";
		cin>>l>>r>>k;
		k=k%(r-l+1);
		for (j=r-k;j<=r-1;j++) s1+=s[j];
		s.insert(l-1,s1);
		s.erase(r-k+s1.length(),s1.length());
	}
	cout<<s<<endl;
	return(0);
}
