#include "bits/stdc++.h"
using namespace std;

const int MAXLENGTH = 11111;
char s[MAXLENGTH];

int main() {
	while (cin >> s) {
		int n, l, r, k;

		cin >> n;
		while (n--) {
			cin >> l >> r >> k;
			--l;
			--r;
			k %= (r - l + 1);
			rotate(s + l, s + r + 1 - k, s + r + 1);
		}
		cout << s << endl;
	}

	return 0;
}
