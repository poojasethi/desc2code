#include<iostream>
#include<math.h>
#include<string>
using namespace std;
string temp;

int main(){

	string s;
	cin>>s;
	int t;
	cin>>t;
	int l , r , k;
	while(t--){
		cin>>l>>r>>k;
		 temp = s.substr(l-1,r-l+1);
		 for(int i = 0 ; i<temp.length() ; i++)
		 {
			 s[l+((i+k)%temp.length())-1] = temp[i];
		 }
	}
	cout<<s<<endl;
}
