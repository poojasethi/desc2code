#include <bits/stdc++.h>
using namespace std;
int main()
{
	int m,l,r,k,i,j,n;
	char s[1000000],t[1000000];
	cin>>s;
	cin>>m;
	while(m--)
	{
		cin>>l>>r>>k;
		strcpy(t,s);
		k%=r-l+1;
		for(i=l-1;i<r;i++)
		{
			if(i-k>=l-1)
				s[i]=t[i-k];
			else
				s[i]=t[r-l+i-k+1];
		}
	}
	cout<<s<<endl;
	return 0;
}
