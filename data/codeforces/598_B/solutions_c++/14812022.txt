#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

char s[10005];
int m, l, r, k;

int main() {
    scanf("%s", s);
    scanf("%d", &m);
    while (m--) {
        scanf("%d%d%d", &l, &r, &k); --l;
        rotate(s + l, s + l + (r-l-k%(r-l)), s + r);
    }
    printf("%s\n", s);
}
