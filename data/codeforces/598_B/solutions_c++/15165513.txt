#include <bits/stdc++.h>
using namespace std;

ifstream fi("598B.INP");
ofstream fo("598B.OUT");

string s;
int n,l,r,k;

int main()
{
    cin>>s>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>l>>r>>k;
        l--;
        k%=(r-l);
        rotate(s.begin()+l,s.begin()+r-k,s.begin()+r);
    }
    cout<<s;
}
