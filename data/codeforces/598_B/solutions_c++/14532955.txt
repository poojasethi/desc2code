#include <bits/stdc++.h>

using namespace std;

int main(){
    string s;
    cin >> s;
    int tt,l,r,k;

    scanf("%d",&tt);
    while(tt--){
        scanf("%d %d %d",&l,&r,&k);
        --l, r;
        k %=(r - l);
        rotate(s.begin()+l, s.begin()+r-k,s.begin()+r);
    }
    cout << s << "\n";

    return 0;
}
