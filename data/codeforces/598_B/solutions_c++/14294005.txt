#include <iostream>
#include <string>
using namespace std;
int m, l, r, k, t;
int main() {
	string s;
	cin >> s >> m;
	for (int i = 0; i < m; ++i) {
		cin >> l >> r >> k;
		k %= (r-- - --l);
		s = s.substr(0, l) + s.substr(r - k + 1, k) + s.substr(l, r - l - k + 1) + s.substr(r + 1);
	}
	cout << s;
	return 0;
}
