#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int m,l,r,k,len;
string s;
int main(){
    cin>>s>>m;
    while(m--){
        scanf("%d%d%d",&l,&r,&k);
        l--;
        len = r-l;
        k %= len;
        rotate(s.begin()+l, s.begin()+r-k, s.begin()+r);
    }
    cout<<s;
    return 0;
}
