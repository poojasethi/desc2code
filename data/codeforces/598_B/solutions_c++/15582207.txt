#include <bits/stdc++.h>
using namespace std;
char temp[10005];
int main(){
	string s;cin>>s;
	int next,m,l,r,k,i,j,len;cin>>m;
	for(i=0;i<m;++i){
		cin>>l>>r>>k;
		k%=(r-l+1);
		len=r-l+1;
		if(k==0)continue;
		for(j=l-1;j<r;++j)temp[j]= s[j];
		for(j=l-1;j<r;++j){
			next= j+k;
			if(next>=r)next-=r-l+1;
			s[next]= temp[j];
		}
	}
	cout<<s<<endl;
}
