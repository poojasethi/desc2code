#include <iostream>
#include <cmath>
#include <string.h>

using namespace std;

string s;
long long int l,t,r,k;
int main()
{
	cin>>s>>t;
	for(int i=0;i<t;i++)
	{
		cin>>l>>r>>k;
		if(l!=r&&k%(r-l+1)!=0)
		{
			k=k%(r-l+1);
			s=s.substr(0,l-1)+s.substr(r-k,k)+s.substr(l-1,r-l-k+1)+s.substr(r,s.length()-r);
		}
	}
	cout<<s<<endl;
	return 0;
}
