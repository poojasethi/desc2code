using namespace std;

#define REP(i,n) for((i)=0;(i)<(int)(n);(i)++)
#define foreach(c,itr) for(__typeof((c).begin()) itr=(c).begin();itr!=(c).end();itr++)

typedef long long ll;
#define MOD 1000000007ll

int X[110],Y[110];
ll dp[110][1010];
ll C[1010][1010];

int main(void){
    int M,i,j,k;
    
    scanf("%d",&M);
    REP(i,M) scanf("%d",&X[i]);
    REP(i,M) scanf("%d",&Y[i]);
    REP(i,M-1) X[i+1] += X[i];
    
    REP(i,1010) REP(j,i+1){
        if(j == 0 || j == i) C[i][j] = 1;
        else C[i][j] = (C[i-1][j-1] + C[i-1][j]) % MOD;
    }
    
    dp[M][X[M-1]] = 1;
    for(i=M-1;i>=0;i--) REP(j,X[i]+1){
        REP(k,Y[i]+1) if(k <= X[i]-j) dp[i][j] = (dp[i][j] + dp[i+1][j+k] * C[X[i]-j][k]) % MOD;
    }
    
    ll ans = dp[0][0];
    REP(i,M-1) ans = ans * C[X[i+1]][X[i]] % MOD;
    printf("%d\n",(int)ans);
    
    return 0;
}
