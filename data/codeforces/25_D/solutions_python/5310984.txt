n=input()
g=[[] for i in (n+1)*[0]]
for i in (n-1)*[0]:
    a,b=map(int,raw_input().split())
    g[a]+=[b]
    g[b]+=[a]

def d(i,p,v,g):
    r=set()
    v.add(i)
    for j in g[i]:
        if j != p:
            if j in v: r.add(tuple(sorted((i,j))))
            else:
                r|=d(j,i,v,g)
    return r
v=set()
r=set()
V=[]

for i in range(1,n+1):
    if i in v: continue
    r|=d(i,0,v,g)
    V+=[i]
