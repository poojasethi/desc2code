#include <cstdio>
#include <vector>

using namespace std;

int p[1001];

int gr(int x){
	if(p[x]==-1)return x;
	return (p[x]=gr(p[x]));
}

int main(){
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)p[i]=-1;
	vector <pair<int,int> > r1;
	for(int i=0;i<n-1;i++){
		int a,b;
		scanf("%d%d",&a,&b);
		a--;b--;
		if(gr(a)==gr(b)){
			r1.push_back(make_pair(a+1,b+1));
		} else {
			p[gr(b)]=a;
		}
	}
	int p1=0;
	printf("%d\n",(int)r1.size());
	for(int i=1;i<n;i++){
		if(gr(0)==gr(i))continue;
		printf("%d %d %d %d\n",r1[p1].first,r1[p1].second,gr(0)+1,gr(i)+1);
		p[gr(i)]=gr(0);
		p1++;
	}
	return 0;
}
