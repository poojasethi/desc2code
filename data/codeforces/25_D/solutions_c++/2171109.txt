#include<iostream>
#include<cstdio>
#include<vector>

using namespace std;

const int maxn = 1010;

int n , par[maxn];
vector<pair<int,int> > V;
vector<int> P;

int anc( int t ) {
	if ( par[t] != t ) par[t] = anc( par[t] );
	return par[t];
}

int main() {
	scanf( "%d" , &n );
	for ( int i = 1 ; i <= n ; i++ )
		par[i] = i;
		
	for ( int i = 1 ; i <= n - 1 ; i++ ) {
		int u , v; scanf( "%d%d" , &u , &v );
		if ( anc( u ) != anc( v ) ) par[anc( u )] = anc( v );
			else V.push_back( make_pair( u , v ) );
	}
	for ( int i = 1 ; i <= n ; i++ )
		if ( par[i] == i ) P.push_back( i );
			
	printf( "%d\n" , ( int ) V.size() );
	for ( int i = 0 ; i < ( int ) V.size() ; i++ )
		printf( "%d %d %d %d\n" , V[i].first , V[i].second , P[i] , P[i + 1] );
	
	return 0;
}
