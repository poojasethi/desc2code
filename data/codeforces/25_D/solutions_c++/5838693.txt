#include <iostream>
#include <vector>
#define pb push_back
using namespace std;
int n,fl[1001],kd;
int ms[1001][1001]; vector <int> ans_del,ans_add;

int fun(int cur,int pre){
 if (fl[cur]==kd+1) {ms[pre][cur]=ms[cur][pre]=0; ans_del.pb(cur*10000+pre); return 0;}
 fl[cur]=kd+1;int kil=1;
 for (int j=1;j<=n;j++) if (j!=pre && ms[cur][j]) kil+=fun(j,cur);
 return kil;}

int main()
{int ot,ku,j; ios_base::sync_with_stdio(false);
cin>>n;
for (int i=1;i<n;i++) {cin>>ot>>ku; ms[ku][ot]=ms[ot][ku]=1;}
while (fun(1,0)!=n) {kd++; for(j=1;j<=n;j++)if (fl[j]<kd) break; ans_add.pb(j); ms[1][j]=ms[j][1]=1;}

cout<<kd<<endl;
for(int i=1; i<=kd; i++) {cout<<ans_del[i-1]/10000<<" "<<ans_del[i-1]%10000<<" "<<"1 "<<ans_add[i-1]<<endl;}
}
