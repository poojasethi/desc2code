#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int Set[1005];
int u[1005],v[1005];
int n;
int find(int x)
{
	if(Set[x] == x)return x;
	return Set[x] = find(Set[x]);
}

int main()
{

	scanf("%d",&n);
	for(int i = 1; i <= n; i++)Set[i]=i;
	int m=0;
	for(int i = 0; i < n-1; i++)
	{
		int a,b;scanf("%d%d",&a,&b);
		int x=find(a),y=find(b);
		if(x==y)u[m]=a,v[m++]=b;
		else Set[x]=y;
	}
	printf("%d\n",m);
	for(int i = 0; i < m; i++)
	{
		printf("%d %d",u[i],v[i]);
		for(int j = 1; j <= n; j++)
			if(find(j)!=find(1))
			{
				Set[find(j)]=find(1);
				printf(" %d %d\n",1,j);
				break;
			}
	}
	return 0;
}
