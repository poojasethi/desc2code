#include <cstdio>
#include <vector>
#include <set>
using namespace std;
int fa[1010], n, a, b;
vector<pair<int,int> > e;
vector<int> pa;
int find(int x) {return fa[x]==x ? x : fa[x]=find(fa[x]);}
void solve()
{
    scanf("%d", &n);
    for (int i=1; i<=n; i++) fa[i]=i;
    for (int i=1; i<n; i++)
    {
    	scanf("%d%d", &a, &b);
    	int x = find(a);
    	int y = find(b);
    	if (x != y) fa[x] = y;
    	else e.push_back(make_pair(a, b));
    }
    for (int i=1; i<=n; i++)
        if (find(i)==i)
            pa.push_back(i);

    printf("%d\n", e.size());
    for (int i=0; i<e.size(); i++)
    	printf("%d %d %d %d\n", e[i].first, e[i].second, pa[i], pa[i+1]);
}

int main()
{
    solve();
    return 0;
}
