#include<cstdio>
#include<iostream>
using namespace std;
int r[1000000],s[10000][4];
void init(int n)
{
    for(int i=0;i<=n;i++)
           r[i]=i;
}
int root(int a)
{
    if(r[a]==a) return a;
    else return r[a]=root(r[a]);
}
void merge(int a,int b)
{
    int ra,rb;
    ra=root(a);
    rb=root(b);
    if(ra!=rb) r[rb]=ra;
}
int main()
{
	int t,m,n,j,k,l,s1,s2;
	cin>>t;
	s1=s2=0;
	init(100000);
	for(j=0;j<t-1;j++)
	{
		cin>>m>>n;
		if(root(m)==root(n))s[s1][0]=m,s[s1][1]=n,s1++;
		else merge(m,n);
	}
	for(j=1;j<t;j++)
	{
		if(root(j)!=root(j+1)){
		s[s2][2]=j,s[s2][3]=j+1,s2++;
		merge(j,j+1);
		}
	}
	cout<<s2<<endl;
	for(j=0;j<s2;j++)
	cout<<s[j][0]<<' '<<s[j][1]<<' '<<s[j][2]<<' '<<s[j][3]<<endl;
	return 0;
}
