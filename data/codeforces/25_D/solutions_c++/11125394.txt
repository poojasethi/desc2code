#include <stdio.h>
#include <vector>
using namespace std;

int f[1005];
int find(int x)
{
  return f[x]==x?f[x]:f[x]=find(f[x]);
}
void he(int x, int y)
 {
   f[find(x)]=find(y);
 }

int main(void) 
{
	int n, i;
	scanf("%d", &n);
	for(i=1;i<=n;i++)
		f[i] = i;
	vector<pair<int, int> > p;
	vector<int> q;
	for(i=1;i<n;i++)
    {
		int x, y;
		scanf("%d%d", &x, &y);
		if(find(x)==find(y)) 
		p.push_back(make_pair(x, y));///有环记录这条边，待会删去
		else he(x, y);
    }
	for(i=1;i<=n;i++)///找有几个不同集合，就需要改几调边
		if(find(i)==i) q.push_back(i);
	printf("%d\n", q.size()-1);
	for(i=1;i<(int)q.size();i++)///去个环，合并两个集合
		printf("%d %d %d %d\n", p[i-1].first, p[i-1].second, q[i-1], q[i]);
	return 0;
}
