#include <stdio.h>
#define oo 0xfffffff
int f[1500],ding[1100];
int find(int x)
{
	int r=x,j;
	while(f[r]!=r)
	{
		r=f[r];
	}
	while(x!=r)
	{
		j=f[x];
		f[x]=r;
		x=j;
	}
	return r;
}
struct node
{
	int x,y;
}city[100000];
int main()
{
	int i,j,n,a,b;
	while(scanf("%d",&n)!=EOF)
	{
		for(i=1;i<=1010;i++)
		{
			f[i]=i;
		}
		int t=n-1,ans=0;
		while(t--)
		{
			scanf("%d%d",&a,&b);
			if(find(a)-find(b))
			{
				f[find(a)]=find(b);
			}
			else
			{
				ans++;
				city[ans].x=a;city[ans].y=b;
			}	
		}
		printf("%d\n",ans);
		for(i=1,j=1;i<=n;i++)
		{
			if(f[i]==i)
			{
				ding[j++]=i;	
			}
		}
		for(i=1,j=1;i<=ans;i++,j++)
		{
			printf("%d %d ",city[i].x,city[i].y);
			printf("%d %d\n",ding[j],ding[j+1]);
		}
	}
	return 0;
}
