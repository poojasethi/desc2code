#include <bits/stdc++.h>

#define st first
#define nd second
#define mp make_pair
#define lli long long int
#define FP( ii,aa,bb ) for( int ii=aa;ii<=bb;ii++ )
#define FM( ii,aa,bb ) for( int ii=aa;ii>=bb;ii-- )

using namespace std;

int ata[2000];
int atabul(int nod){
	if( nod==ata[nod] ) return nod;
	return ata[nod] = atabul( ata[nod] );
}

int n,ind;
pair<int,int> v[2000];

int main(){
	
	int x,y;
	cin >> n;
	FP( i,1,n ) ata[i]=i;
	FP( i,1,n-1 ){
		cin >> x >> y;
		if( atabul(x)!=atabul(y) )
			ata[atabul(x)] = atabul(y);
		else
			v[++ind] = mp( x,y );
	}

	cout << ind << endl;

	FP( i,1,n-1 )
		FP( j,i+1,n )
			if( atabul(i)!=atabul(j) ){
				ata[atabul(i)] = atabul(j);
				cout << v[ind].st << " " << v[ind].nd << " " << i << " " << j << endl;
				ind--;
			}

}
