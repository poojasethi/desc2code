#include <stdio.h>

int n;
int c[1001];

void helper (int a, int b)
{
	int t = c[a];
	for (int i = 1; i <= n; i++){
		if (c[i] == t){
			c[i] = c[b];
		}
	}
}

int main ()
{
	int A[1001], B[1001], t = 0;

	scanf ("%d", &n);
	for (int i = 1; i <= n; i++)
		c[i] = i;
	
	int a,b;	
	for (int i = 1; i < n; ++i){
		scanf ("%d%d", &a, &b);
		if (c[a] == c[b]){
			A[t] = a;
			B[t] = b;
			++t;
		}
		else{
			helper(a,b);
		}
	}
	
	printf ("%d\n", t);
	for (int j = 0; j < t; j++){
		for (int i=1; i<=n; i++){
			if (c[i] != c[A[j]]){
				printf ("%d %d  %d %d\n", A[j], B[j], A[j], i);
				helper(A[j], i);
				break;
			}
		}
	}
	
	return 0;
}
