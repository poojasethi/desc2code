#include<cstdio>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
using namespace std;
int n,i,j,l[10001],r[10001],f[10001],num,a[10001];
int fa(int x)
{
    if (f[x]==x) return x;
    f[x]=fa(f[x]);
    return f[x];
}
int main()
{
    scanf("%d",&n);
    for (i=1;i<=n-1;i++)
        scanf("%d%d",&l[i],&r[i]);
    for (i=1;i<=n;i++)
        f[i]=i;
    for (i=1;i<=n-1;i++)
    {
        int x=fa(l[i]);int y=fa(r[i]);
        if (x!=y) f[x]=y;
        else a[++num]=i;
    }
    printf("%d\n",num);
    int t=0;
    for (i=2;i<=n;i++)
        if (fa(i)!=fa(1))
        {
           ++t;
           printf("%d %d %d %d\n",l[a[t]],r[a[t]],1,i);
           f[fa(i)]=fa(1);
        }
}
