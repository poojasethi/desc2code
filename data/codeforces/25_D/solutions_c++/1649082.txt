#include"stdio.h"
int way[1002][1002],clo[1002][2],reb[1002][2],re=0,cl=0,flag[1002],n;
int find(int me,int father){
	int i;
	flag[me]=1;
	for(i=1;i<=n;i++){
		if(i!=father)
			if(way[me][i])
				if(flag[i])clo[cl][0]=me,clo[cl++][1]=i,way[me][i]=0,way[i][me]=0;
				else find(i,me);
	}
}
int main(){
	int i,t1,t2,pre=0,f=1;
	scanf("%d",&n);
	for(i=1;i<n;i++){
		scanf("%d%d",&t1,&t2);
		way[t1][t2]=1;
		way[t2][t1]=1;	
	}

	for(i=1;i<=n;i++)
		if(!flag[i]){
			find(i,pre);
			if(pre)
				reb[re][0]=pre,reb[re++][1]=i;
			pre=i;
			f=1;
		}	
	printf("%d\n",cl);
	for(i=0;i<cl;i++){
		printf("%d %d %d %d\n",clo[i][0],clo[i][1],reb[i][0],reb[i][1]);
	}
	//getchar();getchar();
}
