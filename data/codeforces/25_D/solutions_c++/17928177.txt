#include<cstdio>
#include<set>
using namespace std;
const int N = 1001;
int fa[N];
int find(int x){
	return x == fa[x] ? x : find(fa[x]);
}
struct po{
	int u, v;
}p[N];
set<int>st;
int main(){
	int n, u, v, fu, fv;
	scanf("%d", &n);
	for (int i = 1; i <= n; i++)
		fa[i] = i;
	int lp = 0;
	po tp;
	for (int i = 1; i < n; i++){
		scanf("%d%d", &u, &v);
		fu = find(u), fv = find(v);
		if (fu != fv)
			fa[fu] = fv;
		else{
			tp.u = u, tp.v = v;
			p[lp++] = tp;
		}
	}
	for (int i = 1; i <= n; i++)
		st.insert(find(i));
	printf("%d\n", st.size() - 1);
	set<int>::iterator it = st.begin();
	for (int i = 0; i < lp; i++){
		printf("%d %d %d", p[i].u, p[i].v, *it);
		printf(" %d\n", *(++it));
	}
	return 0;
}
