#include<bits/stdc++.h>
using namespace std;
const int maxn = 1010;
int n;
int f[maxn];
int find(int x)
{
	return x==f[x]?x:find(f[x]);
}
struct node
{
	int a,b;
}tt[maxn];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		f[i]=i;
	int a,b;
	int cur=0;
	for(int i=0;i<n-1;i++)
	{
		scanf("%d%d",&a,&b);
		int ta=find(a),tb=find(b);
		if(ta==tb)
			tt[cur++]=(node){a,b};
		else
			f[ta]=tb;
	}
	printf("%d\n",cur);
	cur=0;
	for(int i=1;i<=n;i++)
		if(find(i)!=find(1))
	    {
	    	printf("%d %d",tt[cur].a,tt[cur].b);
	    	cur++;
	    	printf(" 1 %d\n",i);
	    	f[find(i)]=find(1);
	    }
	return 0;
}
