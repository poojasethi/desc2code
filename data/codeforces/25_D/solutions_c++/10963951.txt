#include <bits/stdc++.h>

using namespace std;

int n,rep[1009];
vector<pair<int,int> > p;
vector <int> res,res1;

int func( int x ){
	if ( x == rep[x] )	return x;
	
	return rep[x]=func( rep[x] );
}

int main (){

	cin >> n;	p.resize(n);
	
	for ( int i=1; i<=n; i++ )	rep[i]=i;
	
	for ( int i=0; i<n-1; i++ ){
		cin >> p[i].first >> p[i].second;
		int x=func(p[i].first),y=func(p[i].second);
		if ( x != y )	rep[y]=x;
		else res.push_back( i );
	}
	
	for ( int i=1; i<=n; i++ )	if ( rep[i] == i ) res1.push_back(i);
	
	cout << res1.size()-1 << endl;
	for ( int i=1; i<res1.size(); i++ )
		cout << p[res[i-1]].first << ' ' << p[res[i-1]].second << ' ' << res1[i-1] << ' ' << res1[i] << '\n';
	
	return 0;
}
