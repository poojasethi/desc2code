#include <iostream>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)

int n,d[1010][2],a,b,u[1010],t,p;

int main(){

  cin>>n;rep(i,n+1)u[i]=i;
  rep(i,n-1){
    cin>>a>>b;
    if(u[a]==u[b]){d[p][0]=a;d[p++][1]=b;}
    else {
      t=u[b];
      rep(j,n+1)if(u[j]==t)u[j]=u[a];
    }
  }
  

  cout<<p<<endl;

  rep(i,p){
    cout<<d[i][0]<<" "<<d[i][1]<<" "<<u[1]<<" ";
    rep(j,n)if(u[j+1]!=u[1]){cout<<u[j+1]<<endl;t=u[j+1];break;}
    rep(j,n)if(u[j+1]==t)u[j+1]=u[1];
  }

  return 0;
}
