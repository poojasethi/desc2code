#include<iostream>
#include<vector>
#define pb push_back
using namespace std;
vector< pair<int,int> > cle;
int u[1010];
bool mark[1010],p[1010];
int main(){
   int n,i,j,t,x,y,ans=0;
   cin>>n;
   for(i=1;i<=n;i++) u[i]=i;
   for(i=0;i<n-1;i++){
      cin>>x>>y; 
      if(u[x]==u[y]) {cle.pb(make_pair(x,y));}
      else{
         t=u[y];      
         for(j=1;j<=n;j++) if(u[j]==t) u[j]=u[x];   
      }  
   }
   for(i=1;i<=n;i++) if(!p[u[i]]) {ans++;p[u[i]]=1;}    
   cout<<ans-1<<endl;
   if(cle.size()==0) return 0;
   mark[u[1]]=true;
   t=0;
   for(i=2;i<=n;i++){
      j=u[i];  
      if(!mark[j]) {cout<<cle[t].first<<" "<<cle[t].second<<" "<<1<<" "<<i<<endl;mark[j]=1;t++;}  
   }
   //system("pause");
   return 0; 
}
