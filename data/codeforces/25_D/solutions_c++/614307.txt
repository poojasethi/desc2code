#include <stdio.h>
#include <vector>
using namespace std;

int g[1005];
int FIND(int x) {return g[x]==x?g[x]:g[x]=FIND(g[x]);}
void UNION(int x, int y) {g[FIND(x)]=FIND(y);}

int main(void) {
	int n, i;
	scanf("%d", &n);
	for(i=1;i<=n;i++)
		g[i] = i;
	vector<pair<int, int> > p;
	vector<int> q;
	for(i=1;i<n;i++){
		int x, y;
		scanf("%d%d", &x, &y);
		if(FIND(x)==FIND(y)) p.push_back(make_pair(x, y));
		else UNION(x, y);
	}
	for(i=1;i<=n;i++)
		if(FIND(i)==i) q.push_back(i);
	printf("%d\n", q.size()-1);
	for(i=1;i<(int)q.size();i++)
		printf("%d %d %d %d\n", p[i-1].first, p[i-1].second, q[i-1], q[i]);
	return 0;
}
