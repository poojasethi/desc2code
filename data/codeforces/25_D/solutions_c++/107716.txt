#include <cstdio>

int root[2000], close[2000][2], open[2000][2];
int used[2000];
int find(int a){return a==root[a] ? a : root[a] = find(root[a]);}

int n, a, b, ccnt=1, ocnt;

int main(){
	scanf("%d", &n);
	for(int i=1; i<=n; i++)root[i] = i;
	for(int i=1; i<n; i++){
		scanf("%d%d", &a, &b);
		int ra = find(a), rb = find(b);
		if(ra == rb){
			close[ccnt][0] = a;
			close[ccnt][1] = b;
			ccnt++;
		}
		else{
			root[ra] = rb;
		}
	}
	for(int i=1; i<=n; i++){
		int r = find(i);
		if(!used[r]){
			open[ocnt][1] = i;
			open[++ocnt][0] = i;
			used[r] = 1;
		}
	}
	printf("%d\n", ccnt-1);
	for(int i=1; i<ccnt; i++)
		printf("%d %d %d %d\n", close[i][0], close[i][1], open[i][0], open[i][1]);
}
