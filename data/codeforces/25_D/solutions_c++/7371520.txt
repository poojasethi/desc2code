#include <iostream>
#include <vector>
#include <cstdlib>
using namespace std;
const int MAXN=1010;
int n,par[MAXN];
vector <pair<int,int> > c;
int dad(int x)
{
  while(x!=par[x])
    x=par[x];
  return x;
}
int main()
{
  cin>>n;
  for(int i=1;i<=n;i++)
    par[i]=i;
  for(int i=1;i<n;i++)
    {
      int x,y;
      cin>>x>>y;
      int a1=dad(x),a2=dad(y);
      if(a1==a2)
	c.push_back(make_pair(x,y));
      else
	par[a2]=a1;
    }
  cout<<c.size()<<endl;
  for(int i=1;i<=n;i++)
    for(int j=1;j<=n;j++)
      {
	int x=dad(i),y=dad(j);
	if(x!=y)
	  {
	    cout<<c.back().first<<" "<<c.back().second<<" "<<i<<" "<<j<<endl;
	    c.pop_back();
	    par[y]=x;
	  }
      }
  
  return 0;
}
