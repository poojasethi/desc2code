#include <iostream>
#include <map>
#include <cstring>
#include <vector>
using namespace std;
int f[1005];
int find( int x ){ return f[x] == -1 ? x : f[x] = find( f[x] ); }
int n;
vector< pair<int, int> > no;
int main(){
    cin >> n;
	memset( f, -1, sizeof f );
	for( int i=1; i<n; ++i ){
		int a, b;
		int x, y;
		cin >> a >> b;
		x = a, y = b;
		a = find( a );
		b = find( b );
		if( a != b ){
			f[a] = b;
		}
		else{
			no.push_back( make_pair( x, y ) );
		}
	}
	cout << no.size() << endl;
	for( int i=0; i<no.size(); ++i ){
		int a = no[i].first;
		int b = no[i].second;
		cout << a << " " << b << " ";
		for( int j=1; j<=n; ++j ){
			if( find(j) != find(a) ){
				cout << a << " " << j << endl;
				f[find(j)] = find(a);
				break;
			}
		}
	}
}
