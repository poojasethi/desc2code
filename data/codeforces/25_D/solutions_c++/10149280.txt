#include<iostream>
#include<conio.h>
#include<vector>

using namespace std;

const int maxn=100010;
int d[maxn],D,n,p;
vector<int> v1,v2;

int dad(int v)
{
  if(v==d[v])
    return v;
  int a=dad(d[v]);
  d[d[v]]=a;
  d[v]=a;
  return a;
}

void join(int v,int u)
{
  d[dad(v)]=dad(u);
}

int main()
{
  int v,u;
  cin>>n;
  for(int i=0;i<=n;i++)
  d[i]=i;
  for(int i=0;i<n-1;i++)
    {
      cin>>u>>v;
      if(dad(u)==dad(v))
	{
	  v1.push_back(u);
	  v2.push_back(v);
	}
      else
	join(v,u);
    }
  p=0;
  D=dad(1);
  cout<<v1.size()<<endl;
  for(int i=2;i<=n;i++)
    if(dad(i)!=D)
      {
	cout<<v1[p]<<" "<<v2[p]<<" "<<1<<" "<<i<<endl;
	d[dad(i)]=D;
	p++;
      }
  //getch();
  return 0;
}
