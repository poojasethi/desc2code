#include<cstdio>
#include<vector>
using namespace std;
typedef long long INT;
typedef pair<int,int> pii;
#define NN 1010
int root[NN],b[NN];
pii a[NN];
int find_root(int u){
    if(root[u]==u) return u;
    else root[u]=find_root(root[u]);
}
int main()
{
    //freopen("in.txt","r",stdin);
    int n;int u,v,uu,vv;
    int m=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++) root[i]=i;
    for(int i=0;i<n-1;i++){
        scanf("%d %d",&u,&v);u--;v--;
        uu=find_root(u);vv=find_root(v);
        if(uu==vv) a[m++]=pii(u,v);
        else root[uu]=vv;
    }
    int k=0;
    for(int i=0;i<n;i++){
        if(root[i]==i) {b[k++]=i;}
    }
    printf("%d\n",m);
    for(int i=0;i<m;i++){
        printf("%d %d %d %d\n",a[i].first+1,a[i].second+1,b[i]+1,b[i+1]+1);
    }
    return 0;
}
