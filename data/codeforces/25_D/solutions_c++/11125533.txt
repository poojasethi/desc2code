#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;

int pre[1005];
vector<int>cq;
vector<int>s;
int find(int x){
	if(x==pre[x])return x;
	pre[x]=find(pre[x]);
	return pre[x];
}
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)pre[i]=i;
	
	int ans=0;
	for(int i=1;i<n;i++){
		int x,y;
		cin>>x>>y;
		int xx=find(x);
		int yy=find(y);
		
		if(xx==yy){
			cq.push_back(x);
			cq.push_back(y);
			ans++;
		}
		else {
			pre[xx]=yy;
		}
	}
	
	for(int i=1;i<=n;i++){
		if(pre[i]==i){
			s.push_back(i);
		}
	}
	printf("%d\n",ans);
	for(int i=1,j=0;j<cq.size();i++,j+=2){
		printf("%d %d ",cq[j],cq[j+1]);
		printf("%d %d\n",s[i],s[0]);
	}
	return 0;
}
