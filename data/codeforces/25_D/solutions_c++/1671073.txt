#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;

#define MAXN 2020

typedef pair<int,int> ii;

int n,orig[MAXN],dest[MAXN],id[MAXN];
vector<ii> del;
vector<int> vet;

int find(int x) {
	if (id[x] == x) return x;
	return id[x] = find(id[x]);
}

int main() {
	scanf("%d",&n);
	
	for (int i=0; i<n; i++)
		id[i] = i;
	
	for (int i=0; i<n-1; i++) {
		int u,v,a,b;
		scanf("%d%d",&a,&b);
		u = find(a-1);	v = find(b-1);
		if (u == v)	del.push_back(ii(a,b));
		else id[max(u,v)] = min(u,v);
	}
	
	for (int i=0; i<n; i++)
		if (id[i] == i) vet.push_back(i);
		
	printf("%d\n",int(del.size()));
	for (int i=0,j=0; i<int(vet.size())-1; i++,j++) {
		printf("%d %d %d %d\n",del[j].first,del[j].second,vet[i]+1,vet[i+1]+1);
	}
	
	return 0;
}
