#include<stdio.h>
long nc,j,i,Color,color[1001],f[1001][2],q=0,a,b;
int main(void)
 {
  long n,z=1;
  scanf("%ld", &n);
  nc=n;
  for(i=1;i<=n;i++) color[i]=i;
  for(i=1;i<n;i++)
   {
	scanf("%ld %ld",&a,&b);
	if(color[a]==color[b]) { q++; f[q][0]=a; f[q][1]=b; }
	else
	 {
	  nc--;
	  Color=color[b];
	  for(j=1;j<=n;j++) if(color[j]==Color) color[j]=color[a];
	 }
   }
  printf("%ld\n", nc-1);
  for(i=1;i<=nc-1;i++)
   {
	for(j=2;j<=n;j++) if(color[j]!=color[1]) break;
	printf("%ld %ld %ld %ld\n", f[z][0], f[z][1], 1, j); z++;
	Color=color[j];
	for(j=2;j<=n;j++) if(color[j]==Color) color[j]=color[1];
   }
  return 0;
 }
