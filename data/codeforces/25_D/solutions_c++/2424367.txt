#include <iostream>
#include <algorithm>
using namespace std;
#define N 1005
int fa[N],n;
int q[N];
int find(int u) {return fa[u]?fa[u]=find(fa[u]):u;}
int x[N*2],y[N*2],k;
int m;
int main(){
    k=0;
    cin>>n;
    for(int i=1;i<n;i++) {
        int a,b;
        cin>>a>>b;
        int aa=find(a),bb=find(b);
        if(aa==bb)
        {
           x[k]=a,y[k]=b;
           k++;
        }
        else  fa[aa]=bb;
    }
    m=k;
    for(int i=0;i<n;i++) q[i]=find(i+1);
    sort(q,q+n);
    n=unique(q,q+n)-q;
    for(int i=0;i<n-1;i++){
        x[k]=q[i],y[k]=q[i+1];
        k++;
    }
    k/=2;
    cout<<k<<endl;
   for(int i=0;i<k;i++) cout<<x[i]<<" "<<y[i]<<" "<<x[m+i]<<" "<<y[m+i]<<endl;

}
