#include <stdio.h>
#include <vector>
using namespace std;
int g[1005];
int FN(int x) {return g[x]==x?g[x]:g[x]=FN(g[x]);}
void UN(int x, int y) {g[FN(x)]=FN(y);}
int main(void) {
    int n, i;
    scanf("%d", &n);
    for(i=1;i<=n;i++)g[i] = i;
    vector<pair<int, int> > p;
    vector<int> q;
    for(i=1;i<n;i++){
        int x, y;
        scanf("%d%d", &x, &y);
        if(FN(x)==FN(y)) p.push_back(make_pair(x, y));
        else UN(x, y);
    }
    for(i=1;i<=n;i++)
        if(FN(i)==i) q.push_back(i);
    printf("%d\n", q.size()-1);
    for(i=1;i<(int)q.size();i++)
        printf("%d %d %d %d\n", p[i-1].first, p[i-1].second, q[i-1], q[i]);
    return 0;
}
