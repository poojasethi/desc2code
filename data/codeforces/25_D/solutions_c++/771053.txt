#include<cstdio>
int n;
int fa[1005];
int x[1005],y[1005],len;
int get(int a)
{
    while(fa[a]!=a)a=fa[a];
    return a;
}
int un(int a,int b)
{
    a=get(a);
    b=get(b);
    if(a==b)return 0;
    if(a<b)fa[b]=a;
    else fa[a]=b;
    return 1;
}
int main()
{
    scanf("%d",&n);
    int a,b,i;
    for(i=1;i<=n;i++)fa[i]=i;
    int cnt=n;
    for(i=1;i<n;i++)
    {
        scanf("%d%d",&a,&b);
        if(un(a,b))cnt--;
        else
        {
            x[len]=a;
            y[len++]=b;
        }
    }
    printf("%d\n",cnt-1);
    for(i=2;i<=n;i++)
    {
        if(get(i)!=1)
        {
            un(1,i);
            printf("%d %d %d %d\n",x[len-1],y[len-1],1,i);
            len--;
        }
    }
    return 0;
}
