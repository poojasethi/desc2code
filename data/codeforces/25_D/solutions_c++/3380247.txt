#include<cstdio>
int set[1001],x[1001],y[1001],u[1001];
int find(int x){
	if(set[x]!=x)set[x]=find(set[x]);
	return set[x];
}
bool Union(int x,int y){
	int fx=find(x),fy=find(y);
	if(fx==fy)return false;
	set[fy]=fx;
	return true;
}
int main(){
	int n,i,j,a,b;
	while(~scanf("%d",&n)){
		for(i=1;i<=n;i++)set[i]=i;
		j=0;
		for(i=1;i<n;i++){
			scanf("%d%d",&a,&b);
			if(!Union(a,b))x[j]=a,y[j]=b,j++;
		}
		printf("%d\n",j);
		if(!j)continue;int k=0;
		for(i=1;i<=n;i++)if(set[i]==i)u[k++]=i;
		for(i=0;i<j;i++){
			printf("%d %d %d %d\n",x[i],y[i],u[i],u[i+1]);
		}
	}
	return 0; 
}
