# include <stdio.h>
# include <string.h>
# include <iostream>

using namespace std;

int n,par[1500];

int un(int a,int b)
{
    int i,k;
    k = par[a];
    for(i=1;i<=n;i++)
    if(k == par[i])
    par[i] = par[b];


}
int x[1500],y[1500];

int main()
{
  int i,j,k=1,l,a,b;
  cin >> n;

  for(i=1;i<=n;i++)
  par[i] = i;

  for(i=1;i<=n-1;i++)
  {
      cin >> a>>b;
      if(par[a] == par[b])
      {
          x[k] = a;
          y[k] = b;
          ++k;
      }
      else un(a,b);
  }
  cout << k-1<<endl;

  for(i=1;i<k;i++)
   for(j=1;j<=n;j++)
   {
       if(par[j] != par[x[i]])
       {
           cout << x[i]<<" "<<y[i]<<" "<<x[i]<<" "<<j<<endl;
           un(x[i],j);
           break;
       }
   }
  return 0;
}
