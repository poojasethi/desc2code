#include<stdio.h>
struct NN
{
	int a,b;
};
NN nn[1005];
int root[1005];
int find(int x)
{
	if(x!=root[x]) root[x]=find(root[x]);
	return root[x];
}
int arr[1005],len_a;
int main()
{
	int n,i,a,b,f1,f2;
	int len;
	while(scanf("%d",&n)!=EOF)
	{
		len=0;
		for(i=1;i<=n;i++) root[i]=i;
		for(i=0;i<n-1;i++)
		{
			scanf("%d%d",&a,&b);
			f1=find(a);
			f2=find(b);
			if(f1==f2)
			{
				nn[len].a=a;
				nn[len].b=b;
				len++;
			}
			else
			{
				root[f1]=f2;
			}
		}
		printf("%d\n",len);
		len_a=0;
		for(i=1;i<=n;i++)
		{
			root[i]=find(i);
			if(root[i]==i) arr[len_a++]=i;
		}
		for(i=1;i<len_a;i++)
		{
			printf("%d %d %d %d\n",nn[i-1].a,nn[i-1].b,arr[i-1],arr[i]);
		}
	}
	return 0;
}
