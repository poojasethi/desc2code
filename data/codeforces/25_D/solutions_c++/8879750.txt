#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std ;
int uset[1005];
int find(int x) {
    if (x != uset[x]) uset[x] = find(uset[x]);
    return uset[x];
}

int main()
{
    int n,i,j=0,k=0,a,b,c[1005],d[1005],f[1005]={0};
    scanf("%d",&n);
    for(i=0;i<=n;i++)
        uset[i]=i;
    for(i=1;i<n;i++)
    {
        scanf("%d%d",&a,&b);
        if(find(a)==find(b))
            {c[j]=a;d[j++]=b;}
        else{uset[find(b)]=find(a);}
    }
       for(i=1;i<=n;i++)
        if(i==uset[i])f[k++]=i;
        printf("%d\n",j);
       for(i=0;i<j;i++)
       {
           printf("%d %d %d %d\n",c[i],d[i],f[i],f[i+1]);

       }

}
