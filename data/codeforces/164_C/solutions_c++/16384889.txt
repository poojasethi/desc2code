#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
#include <deque>
#include <cstring>
#include <cmath>
#include <queue>
using namespace std;
struct ed
{
	long long fr,to,cp,fl,co;
};
vector<ed> es;
vector<long long> g[3010];
long long inf=1000000000000LL;
long long ds[3010];
long long pa[3010];
long long fa[3010];
int iq[3010];
int ln[3010];
queue<int> q;
void ade(int fr,int to,int cp,int co)
{
	es.push_back((ed){fr,to,cp,0,co});
	es.push_back((ed){to,fr,0,0,-co});
	g[fr].push_back(es.size()-2);
	g[to].push_back(es.size()-1);
}
int s,t;
bool bf(long long&flow,long long&cost)
{
	for(int i=0;i<3000;i++)
		ds[i]=inf,iq[i]=0;
	ds[s]=0,iq[s]=1,pa[s]=0,fa[s]=inf;
	q.push(s);
	while(!q.empty())
	{
		int u=q.front();
		iq[u]=0;
		q.pop();
		for(int i=0;i<g[u].size();i++)
		{
			ed&e=es[g[u][i]];
			if(e.cp>e.fl&&ds[e.to]>ds[u]+e.co)
			{
				ds[e.to]=ds[u]+e.co;
				pa[e.to]=g[u][i];
				fa[e.to]=min(e.cp-e.fl,fa[u]);
				if(!iq[e.to])
				{
					q.push(e.to);
					iq[e.to]=1;
				}
			}
		}
	}
	if(ds[t]==inf)
		return 0;
	flow+=fa[t];
	cost+=ds[t]*fa[t];
	int u=t;
	while(u!=s)
	{
		es[pa[u]].fl+=fa[t];
		es[pa[u]^1].fl-=fa[t];
		u=es[pa[u]].fr;
	}
	return 1;
}
long long se[1010],te[1010],cs[1010];
int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	vector<int> oh,h;
	for(int i=0;i<n;i++)
	{
		scanf("%I64d%I64d%I64d",&se[i],&te[i],&cs[i]);
		te[i]+=se[i];
		oh.push_back(se[i]);
		oh.push_back(te[i]);
	}
	sort(oh.begin(),oh.end());
	for(int i=0;i<oh.size();i++)
		if(i==0||oh[i]!=oh[i-1])
			h.push_back(oh[i]);
	t=h.size(),s=t+1;
	for(int i=0;i<h.size();i++)
		ade(i,i+1,k,0);
	ade(s,0,k,0);
	for(int i=0;i<n;i++)
	{
		se[i]=lower_bound(h.begin(),h.end(),se[i])-h.begin();
		te[i]=lower_bound(h.begin(),h.end(),te[i])-h.begin();
		ade(se[i],te[i],1,-cs[i]);
		ln[i]=es.size()-2;
	}
	long long flow=0,cost=0;
	while(bf(flow,cost)==1);
	for(int i=0;i<n;i++)
	{
		if(es[ln[i]].fl!=0)
			printf("1 ");
		else
			printf("0 ");
	}
}
