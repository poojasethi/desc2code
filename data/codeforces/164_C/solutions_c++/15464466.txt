#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define rep(i,s,e) for (int i=(s); i<(e); i++)
#define clr(a,x) memset(a, x, sizeof(a))
#define N 2005
#define M 10005
using namespace std;
int n,m,S,T,V,num,x,y,t;
int a[N],b[N],c[N],l[N],v[N],u[N],q[N],f[N],r[N],dis[N];
int qd[M],qn[M],qc[M],ql[M];
void flow(){
	for (int i=0;i<V;i++) v[i]=-(1<<30),u[i]=0;
	v[S]=0;q[0]=S;
	for (int i=0,j=0,s=1;s;i++){
		if (i==N) i=0;
		u[q[i]]=0;s--;
		for (int k=l[q[i]];k;k=qn[k])
		if (qc[k] && v[q[i]]+ql[k]>v[qd[k]]){
			v[qd[k]]=v[q[i]]+ql[k];
			f[qd[k]]=k;
			if (!u[qd[k]]){
				if (++j == N) j=0;
				s++;
				u[q[j]=qd[k]]=1;
			}
		}
	}
	if (v[T]==-(1<<30)) return ;
	for (int i=T;i!=S;i=qd[f[i]^1]) qc[f[i]]--,qc[f[i]^1]++;
 }
void add(int x,int y,int c,int s){
	qd[++num]=y;qn[num]=l[x];qc[num]=c;ql[num]=s;l[x]=num;
	qd[++num]=x;qn[num]=l[y];qc[num]=0;ql[num]=-s;l[y]=num;
}
int main(){
	scanf("%d%d",&n,&m);
	rep(i,0,n){
		scanf("%d%d%d",a+i,b+i,c+i);
		dis[t++]=a[i];
		dis[t++]=a[i]+b[i];
	}
	sort(dis,dis+t);
	V=unique(dis,dis+t)-dis;
	S=0;T=V-1;
	num=1;
	rep(i,0,T) add(i,i+1,m,0);
	rep(i,0,n){
		x=lower_bound(dis,dis+V,a[i])-dis;
		y=lower_bound(dis,dis+V,a[i]+b[i])-dis;
		add(x,y,1,c[i]);
		r[i]=num-1;
	}
	rep (i,0,m) flow();
	rep (i,0,n) printf("%d ",1-qc[r[i]]);
	return 0;
}
