#include<iostream>
using namespace std;
int main()
{
	int sum,n,p1,p2,p3,t1,t2,power=0,l[100],r[100];
	cin>>n>>p1>>p2>>p3>>t1>>t2;
	for(int i=0;i<n;i++)
	{
		cin>>l[i];
		cin>>r[i];
		power+=(r[i]-l[i])*p1;
	}
	for(int i=0;i<n-1;i++)
	{
		power+=min(t1,l[i+1]-r[i])*p1;
		sum=min(t2,l[i+1]-r[i]-t1);
		power+=((sum>=0)?sum:0)*p2;
		sum=(l[i+1]-r[i]-t1-t2);
		power+=((sum>=0)?sum:0)*p3;
	}
	cout<<power;
}
