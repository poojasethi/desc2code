#include <iostream>
using namespace std;
int main()
{
	int son=0,n,a,b,c,t1,t2,x,y,sum=0;
	cin>>n>>a>>b>>c>>t1>>t2;
	for(int i=1;i<=n;i++)
	{
		cin >> x >> y ;
		sum+=(y-x)*a;
		if(i==1){son=y;  continue;}
		if(son+t1>x){sum+=(x-son)*a; son=y; continue;}
		sum+=t1*a;
		son+=t1;
		if(son+t2>x){sum+=(x-son)*b; son=y; continue;}
		sum+=t2*b;
		son+=t2;
		sum+=(x-son)*c;
		son=y;
	}
	cout << sum << endl;
	return 0;
}
