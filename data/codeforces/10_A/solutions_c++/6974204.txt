#include <iostream>
using namespace std;

int main()
{	
	int n, P1, P2, P3, T1, T2, T=0, ans=0, x, y;
	cin>>n>>P1>>P2>>P3>>T1>>T2;
	while (n--) {
		cin>>x>>y;
		if (ans>0) {
			ans+=min(x-T,T1)*P1;
			if (x-T>T1) ans+=min(x-T1-T,T2)*P2;
			if (x-T>T1+T2) ans+=(x-T1-T2-T)*P3;
		}
		ans+=(y-x)*P1;
		T=y;
	}
	cout<<ans<<endl;
	return 0;
}
