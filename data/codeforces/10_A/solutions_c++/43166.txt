#include <iostream>
using namespace std;

int main()
{
	int n,P1,P2,P3,T1,T2;
	cin >> n >> P1 >> P2 >> P3 >> T1 >> T2;

	int answer = 0;

	int p = -1;
	for(int i=0; i<n; ++i)
	{
		int li, ri;
		cin >> li >> ri;

		if( p == -1 )
			p = li;

		answer += min(li-p, T1)*P1;
		answer += max(0, min(li-p-T1, T2))*P2;
		answer += max(0, li-p-T1-T2)*P3;
		answer += (ri-li)*P1;
		p = ri;
	}
	cout << answer << endl;
}
