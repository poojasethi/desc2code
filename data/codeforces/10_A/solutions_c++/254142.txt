#include <iostream>
using namespace std;

int n,p1,p2,p3,t1,t2;

int main()
{
	int a,b;
	cin >> n >> p1 >> p2 >> p3 >> t1 >> t2;
	cin >> a >> b;
	int result = (b-a)*p1;
	while(--n)
	{
		cin >> a;
		int diff = a-b;
		cin >> b;
		result += (b-a)*p1;
		int x = min(t1,diff);
		result += x*p1;
		diff -= x;
		x = min(t2,diff);
		result += x*p2;
		diff -= x;
		result += diff*p3;
	}
	cout << result << endl;
	return 0;
}
