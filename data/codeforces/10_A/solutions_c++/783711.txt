#include<iostream>

using namespace std;

int n,P1,P2,P3,T1,T2;
int A,B,L,sum=0;

int main()
{
	cin>>n>>P1>>P2>>P3>>T1>>T2;
	
	for (int i=1;i<=n;i++)
	{
		cin>>A>>B;
		sum+=(B-A)*P1;
		
		if (i>1)
		{
			if (A-L<T1) sum+=(A-L)*P1;
				else
				{
					sum+=T1*P1;
					if (A-L-T1<T2) sum+=(A-L-T1)*P2;
						else sum+=T2*P2+(A-L-T1-T2)*P3;
				}
		}
		
		L=B;
	}
	
	cout<<sum<<endl;
	
	return 0;
}
