#include <iostream>
int main()
{
int n,p1,p2,p3,t1,t2,d; int ar[100][2],i;
long long total=0;
std::cin >> n >> p1 >> p2 >> p3 >> t1 >> t2 ;
for(i=0;i<n;i++)
{
std::cin >> ar[i][0] >> ar[i][1];
}
for(i=0;i<n;i++)
{
total+=p1*(ar[i][1]-ar[i][0]);
if(i!=0) 
{
 d=ar[i][0]-ar[i-1][1];
if(d>=(t1+t2)) total+=p1*t1+p2*t2+p3*(d-t2-t1);
else if(d>t1) total+=p1*t1+p2*(d-t1);
else if(d<=t1) total+=p1*d;
}
}
std::cout << total << "\n";
return 0;
}
