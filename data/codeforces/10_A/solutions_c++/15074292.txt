#include<bits/stdc++.h>

using namespace std;

int main ()
{
    int n,p1,p2,p3,t1,t2,sum=0,a,b,c;
    cin >> n >> p1 >> p2 >> p3 >> t1 >> t2;

	for(int i=0;i<n;i++){

		cin >> a >> b;

		sum+=(b-a)*p1;
		if(i!=0){
			if(c+t1<=a){
				sum+=t1*p1;
				c+=t1;
				if(c+t2<=a){
					sum+=t2*p2+(a-c-t2)*p3;
				}
				else sum+=(a-c)*p2;
			}
			else sum+=(a-c)*p1;
		}
		c=b;
	}
	printf("%d\n",sum);
	return 0;

 return 0;
}
