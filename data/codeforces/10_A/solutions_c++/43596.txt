#include<iostream>

using namespace std;

int n,p1,p2,p3,t1,t2;

int main()
{
	cin>>n>>p1>>p2>>p3>>t1>>t2;
	int last=-1,ans=0;
	for(int i=0;i<n;i++)
	{
		int x,y;
		cin>>x>>y;
		ans+=p1*(y-x);
		if(last!=-1)
		{
			int d=x-last;
			if(d<=t1)
			{
				ans+=p1*d;
			}else
			{
				ans+=p1*t1;
				d-=t1;
				if(d<=t2)
				{
					ans+=p2*d;
				}else
				{
					d-=t2;
					ans+=p2*t2;
					ans+=p3*d;
				}
			}
		}
		last=y;
	}
	cout<<ans<<endl;
}
