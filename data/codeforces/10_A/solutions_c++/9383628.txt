#include <iostream>
#include <cstdlib>
#include <stdio.h>
using namespace std;
int main(){
	short int n, p1,p2,p3,t1,t2, a, b,lb;
	int s=0;
	cin>>n>>p1>>p2>>p3>>t1>>t2;
	for(int i=0;i<n;i++){
		cin>>a>>b;
		if(i==0) lb=a;
		if(a-lb>t1){
			s+=t1*p1;
			if(a-lb>t2+t1){
				s+=t2*p2;
				s+=(a-lb-t1-t2)*p3;
			}else{
				s+=(a-lb-t1)*p2;
			}
		}else{
			s+=(a-lb)*p1;
		}
		s+=(b-a)*p1;
		lb=b;
	}
	cout<<s;
}
