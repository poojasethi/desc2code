#include<iostream>
#include<algorithm>
using namespace std;
const int N=111;
int main()
{
	int ans,n,p1,p2,p3,t1,t2,l[N],r[N];
	while(cin>>n>>p1>>p2>>p3>>t1>>t2)
	{
		ans=0;
		for(int i=1;i<=n;++i)cin>>l[i]>>r[i],ans+=(r[i]-l[i])*p1;
		for(int i=1;i<n;++i)
		{
			int cha=l[i+1]-r[i];
			if(cha>=t1+t2)ans+=t1*p1+t2*p2+(cha-t1-t2)*p3;
			else if(cha>=t1)ans+=t1*p1+(cha-t1)*p2;
			else ans+=cha*p1;
		}
		cout<<ans<<endl;
	}
	return 0;
}
