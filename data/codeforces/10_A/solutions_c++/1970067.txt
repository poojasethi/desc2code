#include<stdio.h>
int main()
{
	int n,p1,p2,p3,t1,t2,i;
	int a[150],b[150];
	scanf("%d%d%d%d%d%d",&n,&p1,&p2,&p3,&t1,&t2);
	for(i=1;i<=n;i++)
		scanf("%d %d",&a[i],&b[i]);
	int sum=p1*(b[1]-a[1]);
	for(i=2;i<=n;i++)
	{
		sum+=p1*(b[i]-a[i]);
		int t=a[i]-b[i-1];
		if(t>t1)
		{
			sum+=t1*p1;
			if(t-t1>t2)
				sum+=t2*p2+(t-t1-t2)*p3;
			else
				sum+=(t-t1)*p2;
		}
		else
			sum+=t*p1;
	}
	printf("%d\n",sum);
	return 0;
}
