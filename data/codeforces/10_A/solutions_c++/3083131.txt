#include <iostream>
using namespace std;
int main()
{int n,p1,p2,p3,t1,t2,l,r,l1,r1;
cin>>n>>p1>>p2>>p3>>t1>>t2;
cin>>l>>r;
int ans=(r-l)*p1;
for (int i=1;i<n;i++)
{
 cin>>l1>>r1;
 ans+=(r1-l1)*p1;
 int temp=l1-r;
 if (temp>t1)
  {temp-=t1;ans+=t1*p1;
   if (temp>t2)
    {
     temp-=t2;ans+=t2*p2;ans+=temp*p3;
    }
   else ans+=temp*p2;
  }
  else ans+=temp*p1;
 r=r1;
}
cout<<ans<<endl;
return 0;
}
