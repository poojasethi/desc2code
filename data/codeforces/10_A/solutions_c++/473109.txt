#include <stdio.h>
int main(int argc, char* argv[])
{
    int n,P1,P2,P3,T1,T2,l,r,ro=0,t,p=0;
    scanf("%d%d%d%d%d%d",&n,&P1,&P2,&P3,&T1,&T2);
    for (int i=0; i<n; i++) {
        scanf("%d%d",&l,&r);
        t=l-ro;
        if (i>0)
            p+=(t<=T1?t*P1:(T1*P1+(t-T1<=T2?(t-T1)*P2:(T2*P2+(t-T1-T2)*P3))));
        p+=(r-l)*P1;
        ro=r;
    }
    printf("%d\n", p);
    return 0;
}
