#include <iostream>

using namespace std;

int n,p_1,p_2,p_3,t_1,t_2,ans,l,r,r_2;

int main(){
  cin>>n>>p_1>>p_2>>p_3>>t_1>>t_2>>l>>r;
  ans+=p_1*(r-l);
  r_2=r;
  for(int i=1;i<n;i++){
    r_2=r;
    cin>>l>>r;
    ans+=p_1*(r-l);
    if(l-r_2<=t_1)
      ans+=(l-r_2)*p_1;
    else if (l-r_2<=t_1+t_2)
      ans+=t_1*p_1+(l-r_2-t_1)*p_2;
    else ans+=t_1*p_1+t_2*p_2+(l-r_2-t_1-t_2)*p_3;
  }
  cout<<ans<<endl;
}//  :(
