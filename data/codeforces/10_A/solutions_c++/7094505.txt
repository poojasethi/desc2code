#include<iostream>
using namespace std;
int main()
{
int n,p1,p2,p3,t1,t2,i,t;
long int p=0;
cin>>n>>p1>>p2>>p3>>t1>>t2;
int *l= new int [n];
int *r= new int [n];
for(i=0; i<n; i++)
cin>>l[i]>>r[i];
for(i=0; i<n; i++)
{
t=r[i]-l[i];
p+=p1*t;
if(i<n-1)
{
t=l[i+1]-r[i];
if(t<=t1)
p+=p1*t;
else if(t<=t1+t2)
p+=p1*t1+p2*(t-t1);
else
p+=p1*t1+p2*t2+p3*(t-t1-t2);
}
}
cout<<p;
delete r;
delete l;
return 0;
}
