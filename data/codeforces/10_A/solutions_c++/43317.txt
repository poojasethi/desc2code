#include<iostream>
using namespace std;
int p[3],t[2],n;
int sum;
int main(){
	cin>>n>>p[0]>>p[1]>>p[2]>>t[0]>>t[1];
	int bf=-1;
	for(int i=0;i<n;i++){
		int st,ed;
		cin>>st>>ed;
		if(bf!=-1){
			int tl=st-bf;
			if(tl<=t[0]){
				sum+=p[0]*tl;
			}else{
				tl-=t[0];
				sum+=p[0]*t[0];
				if(tl<=t[1]){
					sum+=p[1]*tl;
				}else{
					tl-=t[1];
					sum+=p[1]*t[1]+p[2]*tl;
				}
			}
		}
		sum+=(ed-st)*(p[0]);
		bf=ed;
	}
	cout<<sum<<endl;
}
