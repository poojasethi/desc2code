#include<stdio.h>
int main()
{
	int n,p1,p2,p3,t1,t2;
	scanf("%d%d%d%d%d%d",&n,&p1,&p2,&p3,&t1,&t2);
	int pr ,i,x,y;
	int ans = 0;
	for(i=0;i<n;i++)
	{
	scanf("%d%d",&x,&y);
	if(i==0)pr = x;
	ans+=(y-x)*p1;
	int rem = x-pr;
	if(rem >= t1){ans+=t1*p1;rem-=t1;}
	else {ans+=rem*p1; rem=0;}
	if(rem>=t2){ans+=t2*p2; rem-=t2;}
	else {ans+=rem*p2; rem=0;}
	ans+=rem*p3;
	pr = y;


	}
	printf("%d\n",ans);
}
