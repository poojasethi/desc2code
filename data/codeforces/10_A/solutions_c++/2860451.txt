#include<iostream>
using namespace std;
int n,ans,p1,p2,p3,t1,t2,l[110],r[110];
int main()
{
    cin>>n>>p1>>p2>>p3>>t1>>t2>>l[0]>>r[0];
    for (int i=1; i<n; i++)
    {
        cin>>l[i]>>r[i];
        ans+=(r[i]-l[i])*p1;
        int t=l[i]-r[i-1];
        ans+=min(t,t1)*p1,t-=min(t,t1);
        ans+=min(t,t2)*p2,t-=min(t,t2);
        ans+=t*p3;
    }
    cout<<ans+(r[0]-l[0])*p1<<endl;
}
