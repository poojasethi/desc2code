#include <iostream>
using namespace std;
int main()
{
	int a[101][2];
	int n,p1,p2,p3,t1,t2,i,sum=0,x;
	cin>>n>>p1>>p2>>p3>>t1>>t2;
	for(i=0;i<n;i++)
	{
		cin>>a[i][0]>>a[i][1];
		sum += (a[i][1]-a[i][0])*p1;
	}
	for(i=0;i<n-1;i++)
	{
		x = a[i+1][0] - a[i][1];
		if(x<=t1) sum+=x*p1;
		else if(x<=t1+t2) sum=sum+t1*p1+(x-t1)*p2;
		else sum = sum+t1*p1+t2*p2+(x-t1-t2)*p3;
	}
	cout<<sum<<endl;
}
