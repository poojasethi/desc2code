#include <iostream>     //IN THE NAME OF GOD
int n,p1,p2,p3,t1,t2,s[100+10],t[100+10],sum,ans,x;
int main(){std::cin>>n>>p1>>p2>>p3>>t1>>t2;
    for(int i=0;i<n;i++){std::cin>>s[i]>>t[i];sum+=t[i]-s[i];}
    ans+=p1*sum;
    for(int i=0;i<n-1;i++){x=s[i+1]-t[i];
        if(x>t1+t2)ans +=t1*p1+t2*p2+(x-t1-t2)*p3;
        else if(x>t1)ans+=p1*t1+p2*(x-t1);else ans+=x*p1;}
    std::cout<<ans;}
