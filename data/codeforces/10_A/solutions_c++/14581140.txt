#include <iostream>
using namespace std;

int main() {
	int n, i, p1, p2, p3, t1, t2, res = 0, l[100], r[100], t;
	cin >> n >> p1 >> p2 >> p3 >> t1 >> t2;
	for(i = 0; i < n; i++) cin >> l[i] >> r[i];
	for(i = 0; i < n - 1; i++) {
		res += p1 * (r[i] - l[i]);
		t = l[i + 1] - r[i];
		res += min(t, t1) * p1;
		t -= t1;
		if(t > 0) res += min(t, t2) * p2;
		t -= t2;
		if(t > 0) res += t * p3;
	}
	res += p1 * (r[n - 1] - l[n - 1]);
	cout << res;
}
