#include<iostream>
using namespace std;
int main()
{
	int n,t1,t2,p1,p2,p3,s=0,r[111],l[111],i;
	cin>>n>>p1>>p2>>p3>>t1>>t2;
	for(i=0;i<n;i++)
	{
		cin>>l[i]>>r[i];
		s+=(r[i]-l[i])*p1;
	}	
	for(i=0;i<n-1;i++)
	{
		if((l[i+1]-r[i])>=t1+t2)
			s+=t1*p1+t2*p2+p3*(l[i+1]-r[i]-t1-t2);
		else if((l[i+1]-r[i])>=t1)
			s+=t1*p1+p2*(l[i+1]-r[i]-t1);
		else
			s+=p1*(l[i+1]-r[i]);
	}
	cout<<s<<endl;
	return 0;
}
