#include <cstdio>
int ans,n,i,p1,p2,p3,t1,t2,l[200],r[200];
int count(int x) {
	if (x<=t1) return p1*x;
	if (x<=t1+t2) return (x-t1)*p2+t1*p1;
	return (x-t1-t2)*p3+t1*p1+p2*t2;
}
int main() {
	scanf("%d%d%d%d%d%d",&n,&p1,&p2,&p3,&t1,&t2);
	for (i=1;i<=n;i++) scanf("%d%d",l+i,r+i);
	ans=(r[1]-l[1])*p1;
	for (i=2;i<=n;i++)
		ans+=count(l[i]-r[i-1])+(r[i]-l[i])*p1;
	printf("%d\n",ans);
	return 0;
}
