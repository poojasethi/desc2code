#include <cstdio>
main()
{
    int n,a,b,c,d,e,l,r,t,x;
    scanf("%u%u%u%u%u%u",&n,&a,&b,&c,&d,&e);
    c-=b;
    b-=a;
    e+=d;
    t=1e4;
    x=0;
    while(n--)
    {
        scanf("%u%u",&l,&r);
        x+=a*(r-l);
        l-=t;
        t=r;
        x+=a*(l>0)*l+b*(l>d)*(l-d)+c*(l>e)*(l-e);
    }
    return !printf("%d\n",x);
}
