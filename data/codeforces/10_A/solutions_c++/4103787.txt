#include<stdio.h>


int main(){
	int  n, P1, P2, P3, T1, T2;
	int res=0,q,r,l;
	scanf("%d%d%d%d%d%d",&n,&P1,&P2,&P3,&T1,&T2);
	scanf("%d%d",&r,&l);
	res=P1*(l-r);
	q=l;n--;
	while(n--){
		scanf("%d%d",&r,&l);
		res+=P1*(l-r);
		if(r-q>T1){
			res+=T1*P1;
			if(r-q>T1+T2){
				res+=P2*T2+(r-q-T1-T2)*P3;
			}else res+=(r-q-T1)*P2;
		}else res+=(r-q)*P1;
		q=l;
	}
	printf("%d\n",res);
	return 0;
}
