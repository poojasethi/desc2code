#include<iostream>
using namespace std;
int main()
{
	int n,p1,p2,p3,t1,t2,i,w=0;
	int l[1500],r[1500];
	cin>>n>>p1>>p2>>p3>>t1>>t2;
	for(i=0;i<n;i++)
	{
		cin>>r[i]>>l[i];
		w+=(l[i]-r[i])*p1;
	}
	for(i=0;i<n-1;i++)
	{
		if(r[i+1]-l[i]>=(t1+t2))
		{
			w+=p1*t1+p2*t2+(r[i+1]-l[i]-t1-t2)*p3;
		}
		else if(r[i+1]-l[i]>=t1)
		{
			w+=p1*t1+(r[i+1]-l[i]-t1)*p2;
		}
		else
		{
			w+=(r[i+1]-l[i])*p1;
		}
	}
	cout<<w<<endl;
	return 0;
}
