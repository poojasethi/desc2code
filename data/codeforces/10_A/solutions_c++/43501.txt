#include<iostream>
using namespace std;
int n,p1,p2,p3,t1,t2,r[1000],l[1000],i,t,ans;
int main()
{
	cin>>n>>p1>>p2>>p3>>t1>>t2;
     for (i=0;i<n;i++)
		 cin>>l[i]>>r[i];
	 ans=0;
	 for (i=0;i<n-1;i++)
	 {
		 ans+=(r[i]-l[i])*p1;
		 t=l[i+1]-r[i];
		 if (t>=t1+t2) {ans+=t1*p1+t2*p2+(t-t1-t2)*p3; continue;}
		 if (t>=t1) {ans+=t1*p1+(t-t1)*p2; continue;}
		 ans+=t*p1;
		 
		 

	 }
	 ans+=(r[n-1]-l[n-1])*p1;
	 cout<<ans<<endl;
}
