#include <cstdio>
using namespace std;
int n,p1,p2,p3,t1,t2;
int ans=0;
int main()
{
	scanf("%d%d%d%d%d%d",&n,&p1,&p2,&p3,&t1,&t2);
	int l,r;
	scanf("%d%d",&l,&r);
	int rr=r;
	ans+=p1*(r-l);
	for (int i=2;i<=n;i++)
	{
		scanf("%d%d",&l,&r);
		ans+=p1*(r-l);
		if (l-rr<=t1) ans+=(l-rr)*p1;
		else if (l-rr<=t1+t2) ans+=t1*p1+(l-rr-t1)*p2;
		else ans+=t1*p1+t2*p2+(l-rr-t1-t2)*p3;
		rr=r;
	}
	printf("%d\n",ans);
	return 0;
}
