#include <iostream>

using namespace std;
int n, p1,p2,p3,t1,t2, l[100],r[100];
int res=0;

int foo (int &m, int t) {
      if(m < t) t = m;
      m -= t;
      return t;
}

int main()
{

cin>>n>>p1>>p2>>p3>>t1>>t2;

for(int i=0; i<n; i++)
cin>>l[i]>>r[i];


int tmp;

for(int i=0; i<n; i++){
 
res+=((r[i]-l[i])*p1);
if(i+1<n){
tmp=l[i+1]-r[i];

res+=p1*foo(tmp,t1)+p2*foo(tmp,t2)+p3*tmp;
}

}


 cout<<res;


 return 0;
}
