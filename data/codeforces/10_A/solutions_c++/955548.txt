#include<iostream>
using namespace std;
int main(){
	int n,p1,p2,p3,t1,t2,ans=0;;
	cin>>n>>p1>>p2>>p3>>t1>>t2;
	int l=0;
	for(int i=0;i<n;i++){
		int x,y;
		cin>>x>>y;
		ans+=(y-x)*p1;
		if(i){
			int w=l;
			while(w<=x){
				if(w>l+t1+t2)
					ans+=p3;
				else if(w>l+t1)
					ans+=p2;
				else if(w-l)
					ans+=p1;
				w++;
			}
		}
		l=y;
	}
	cout<<ans<<endl;
}
