def standby(t):
    if t <= t1:
        return t * p1
    elif t <= t1 + t2:
        return t1 * p1 + (t - t1) * p2
    else:
        return t1 * p1 + t2 * p2 + (t - t1 - t2) * p3

n, p1, p2, p3, t1, t2 = map(int, raw_input().split())
t = 0
p = 0
for i in range(n):
    l, r = map(int, raw_input().split())
    if i != 0:
        p += standby(l - t)
    p += (r - l) * p1
    t = r
print p
