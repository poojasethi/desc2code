n, p1, p2, p3, t1, t2 = map(int, raw_input().split())

power_used = 0
r = 0

for x in xrange(n):
	cur_l, cur_r = map(int, raw_input().split())
	cur_power = (cur_r - cur_l) * p1
	power_used += cur_power

	if r != 0:
		between_time = cur_l - r
		for time in xrange(between_time):
			if time < t1:
				power_used += p1
			elif time < t2 + t1:
				power_used += p2
			else:
				power_used += p3

	r = cur_r

print power_used
