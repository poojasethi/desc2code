nums=map(int,raw_input().split())
n2s=[map(int,raw_input().split()) for i in range(nums[0])]
sum=0
for i in range(nums[0]):
	sum+=(n2s[i][1]-n2s[i][0])*nums[1]
	if i>0:
		sum+=min(n2s[i][0]-n2s[i-1][1],nums[4])*nums[1] + min(max(n2s[i][0]-n2s[i-1][1]-nums[4],0),nums[5])*nums[2]+max(n2s[i][0]-n2s[i-1][1]-nums[5]-nums[4],0)*nums[3]
print sum
