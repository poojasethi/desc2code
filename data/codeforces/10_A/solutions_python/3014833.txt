def rdint():
	return map(int,raw_input().split())
def rdmat(n):
	return [rdint() for _ in range(n)]
####################
n,p1,p2,p3,t1,t2 = rdint()
mp = rdmat(n)
d = [mp[i][0]-mp[i-1][1] for i in range (1,n)]
print p1*(mp[-1][1]-mp[0][0]) + (p2-p1)*sum(max(0,dd-t1) for dd in d)+(p3-p2)*sum(max(0,dd-t1-t2)
		for dd in d)
