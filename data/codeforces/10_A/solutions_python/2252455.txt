def cntProstoi(p1, p2, p3, t1, t2, dur):
 if(dur>t2+t1): return t1*p1+t2*p2+p3*(dur-t1-t2)
 if(dur>t1): return t1*p1+p2*(dur-t1)
 return dur*p1

(n, p1, p2, p3, t1, t2) = [int(t) for t in raw_input().split()]
res=0
lr=-1  #last right
for k in xrange(n):
 (l,r)=[int(t) for t in raw_input().split()]
 res+=p1*(r-l)
 if lr!=-1 : res+=cntProstoi(p1,p2,p3,t1,t2,l-lr)
 lr=r
print res
