power = 0
n, p1, p2, p3, t1, t2 = map(int, raw_input().split())
l, r = map(int, raw_input().split())
power += p1 * (r-l)
rlast = r

for i in range(1,n):
    l, r = map(int, raw_input().split())
    power += p1 * (r-l)
    t = l-rlast
    if t > (t1 + t2):
        power += p3 * (t-(t1 + t2)) + (p2 * t2) + (p1 * t1)
    elif t > t1:
        power += p2 * (t-t1) + (p1*t1)
    else:
        power += p1 * t
    rlast = r

print power
