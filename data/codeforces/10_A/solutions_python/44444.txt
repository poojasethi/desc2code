n, p1, p2, p3, t1, t2 = map(eval, raw_input().split(' '))
res = 0
l1, r1 = map(eval, raw_input().split(' '))
t = r1
for i in range(1,n):
    l, r = map(eval, raw_input().split(' '))
    dt = l - t
    dt -= t1
    if dt > 0:
        res -= dt * (p1 - p2)
        dt -= t2
        if dt > 0:
            res -= dt * (p2 - p3)
    t = r
res += (t - l1) * p1

print res
