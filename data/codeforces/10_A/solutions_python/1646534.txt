#!/usr/bin/python
import sys

(n, p1, p2, p3, t1, t2) = sys.stdin.readline().split()
n  = int(n)
p1 = int(p1)
p2 = int(p2)
p3 = int(p3)
t1 = int(t1)
t2 = int(t2)

p     = 0
rprev = 0
for i in range(1,n+1):
	(l, r) = sys.stdin.readline().split()
	l = int(l)
	r = int(r)
	p += p1 * (r - l)
	if i == 1:
		rprev = l
	delta = l - rprev
	if delta > t1 + t2:
		p += (delta-t1-t2)*p3 + t2*p2 + t1*p1
	elif delta > t1:
		p += (delta-t1)*p2 + t1*p1
	else:
		p += delta*p1
	rprev = r

print p
