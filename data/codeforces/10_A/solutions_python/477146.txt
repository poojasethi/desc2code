def cal(t):
    sums = 0
    sums += min(t, t1) * p1
    t -= min(t,t1)
    sums += min(t, t2) * p2;
    t -= min(t, t2)
    sums += t * p3
    return sums

n, p1, p2, p3, t1, t2 = map(int, raw_input().split())



segs = [map(int, raw_input().split()) for _ in range(n) ]
ans = p1 * (segs[0][1] - segs[0][0])
for i in range(1, n):
    ans += cal( segs[i][0] - segs[i - 1][1] )
    ans += p1 * (segs[i][1] - segs[i][0] )
print ans
