n, P1, P2, P3, T1, T2 = map (int, raw_input ().split())
segs = [map (int, raw_input ().split()) for _ in range (n)]
ans = 0
for i in range (n - 1):
    delta = segs[i + 1][0] - segs[i][1]
    ans += (segs[i][1] - segs[i][0]) * P1 + min (delta, T1) * P1 + max (0, min (delta - T1, T2)) * P2 + max (0, delta - T1 - T2) * P3
print ans + (segs[n - 1][1] - segs[n - 1][0]) * P1
