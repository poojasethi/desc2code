#!/usr/bin/env python

from sys import stdin

n, P1, P2, P3, T1, T2 = map(int, stdin.readline().split())
last = map(int, stdin.readline().split())

res = (last[1]-last[0]) * P1
# print '*', res

for i in xrange(1, n):
   cur = map(int, stdin.readline().split())
   d = cur[0] - last[1]
#  print '*', min(d, T1) * P1
   res += min(d, T1) * P1
#  print '*', min(max(d - T1, 0), T2) * P2
   res += min(max(d - T1, 0), T2) * P2
   res += max(d - (T1+T2), 0)  * P3
   res += (cur[1]-cur[0]) * P1
   last = cur

   
print res
