import math

n, p1, p2, p3, t1, t2  = map(int, raw_input().split())

ret = 0
ii = []
for i in range(n):
    ii.append( map(int, raw_input().split()) )

for i in range(n-1):
    betw = ii[i][1] - ii[i][0]
    next = ii[i+1][0] - ii[i][1]
    tot = next + betw
    ret += min(tot, betw + t1) * p1  + min( t2, max(0, next - t1) ) * p2 + max(0, next - t1 - t2) * p3
ret += (ii[n-1][1] - ii[n-1][0])*p1
print ret
