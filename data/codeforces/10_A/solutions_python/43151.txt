import sys
rl = lambda: sys.stdin.readline().split()
n, p1, p2, p3, t1, t2 = map(int, rl())
last_b = -1
ret = 0
for i in range(n):
    a, b = map(int, rl())
    if last_b != -1:
        idle = a - last_b
        a3 = max(0, idle - t1 - t2)
        a2 = max(0, min(t2, idle - t1))
        a1 = min(idle, t1)
        ret += a1 * p1 + a2 * p2 + a3 * p3
    ret += (b - a) * p1
    last_b = b
print ret
