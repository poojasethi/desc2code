n, p1, p2, p3, t1, t2 = map(int, raw_input().split())
power_consumption = 0
currrent_r = 0

for _ in xrange(n):
  li, ri = map(int, raw_input().split())
  current_power = (ri - li) * p1
  power_consumption += current_power

  if currrent_r != 0:
    difference = li - currrent_r
    for time_interval in xrange(difference):
      if time_interval < t1:
        power_consumption += p1
      elif time_interval < t2 + t1:
        power_consumption += p2
      else:
        power_consumption += p3

  currrent_r = ri

print power_consumption
