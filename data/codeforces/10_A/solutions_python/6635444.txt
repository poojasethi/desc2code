[n, p1, p2, p3, t1, t2]=map(int, raw_input().split())
list, ans, a, b = [], 0, 0, 0 
for i in range(n):
	list = map(int, raw_input().split())
	ans +=  p1 * (list[1]-list[0])
	if(i>0):
		b = list[0]-a
		if(b <= t1):
			ans += p1 * b
		elif(t1 < b <= (t1+t2)):
			ans += p1*t1 + p2*(b-t1)
		elif((t1+t2) < b):
			ans += p1*t1 + p2*t2 + p3*(b-(t1+t2))
	a = list[1]
print ans
