n,p1,p2,p3,t1,t2 = map(int, raw_input().split(" "))
p3-=p2
p2-=p1
last = 1000000
ans = 0
for i in xrange(0,n):
    l, r = map(int, raw_input().split(" "))
    if (l>last):
        ans+=max(l - last, 0)*p1
        ans+=max(l - last - t1, 0)*p2
        ans+=max(l - last - t1 - t2, 0)*p3
    ans+=(r-l)*p1
    last = r
print ans
