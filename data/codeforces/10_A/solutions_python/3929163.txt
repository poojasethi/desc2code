I=lambda:map(int, raw_input().split())
n,p1,p2,p3,t1,t2=I()
def get(n):
    if n<=t1: return p1*n
    elif n<=t1+t2: return p1*t1+(n-t1)*p2
    else: return p1*t1+p2*t2+(n-t1-t2)*p3

ans,prev=0,-1

for _ in xrange(n):
    l,r=I()
    ans+=(r-l)*p1
    if prev!=-1:
        ans+=get(l-prev)
    prev=r
print ans
