n, p1, p2, p3, t1, t2 = map(int, raw_input().split())
l, r = [0]*n, [0]*n
for i in range(n):
	l[i], r[i] = map(int, raw_input().split())
ans = [0]*3
for i in range(n-1):
	delta = l[i+1] - r[i]
	_t1 = min(t1, delta)
	delta -= _t1
	_t2 = min(t2, delta)
	delta -= _t2
	_t3 = delta
	ans[0] += _t1 + r[i] - l[i]
	ans[1] += _t2
	ans[2] += _t3
ans[0] += r[n-1] - l[n-1]
print p1 * ans[0] + p2 * ans[1] + p3 * ans[2]
