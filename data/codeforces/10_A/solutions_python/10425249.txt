n,p1,p2,p3,t1,t2 = map(int, raw_input().split())
r = consumed = 0
for _ in xrange(n):
    li,ri = map(int, raw_input().split())
    cur_consumed = (ri - li) * p1
    consumed += cur_consumed
    if r != 0:
        for i in xrange(li - r):
            if i < t1:
                consumed += p1
            elif i < t1 + t2:
                consumed += p2
            else:
                consumed += p3
    r = ri
print consumed
