s=raw_input().split()
n=int(s[0])
P1=int(s[1])
P2=int(s[2])
P3=int(s[3])
T1=int(s[4])
T2=int(s[5])

r=0
A=[0,0]

for i in range(n):
	B=[int(x) for x in raw_input().split()]
	if i>0:
		rest=B[0]-A[1]
		r+=P1*min(rest,T1)
		rest-=T1
		if(rest>0):
			r+=P2*min(rest,T2)
			rest-=T2
			if(rest>0):
				r+=P3*rest
	r+=P1*(B[1]-B[0])
	A=B
print r
