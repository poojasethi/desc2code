#!/usr/bin/env python
# -*- coding: UTF-8 -*-
import time
import sys, io
import re, math
start = time.clock()
(n,P1,P2,P3,T1,T2)=map(int, raw_input().split())
ans=0
l=[raw_input().split() for i in range(n)]
for j in range(n):
    ans+=P1*( int(l[j][1]) - int(l[j][0]))
    if j:
        k=int(l[j][0])-int(l[j-1][1])
        if k<=T1:
            ans+=P1*k
        elif k<=T1+T2:
            ans+=P1*T1
            ans+=P2*(k-T1)
        else:
            ans+=P1*T1
            ans+=P2*T2
            ans+=P3*(k-(T1+T2))
print ans
