n, p1, p2, p3, t1, t2 = map(int, raw_input().split())
ans = 0
last = 0
for i in range(n):
    s, t = map(int, raw_input().split())
    if i == 0:
        last = s
    ans += (t-s)*p1
    if s-last > t1:
        ans += t1*p1
        if s-last <= t1+t2:
            ans += (s-last-t1)*p2
        else:
            ans += t2*p2 + (s-last-t1-t2)*p3
    else:
        ans += (s-last)*p1
    last = t
    
print ans
