def main():
    n, P1, P2, P3, T1, T2 = map(int, raw_input().split())
    tm = [ map(int, raw_input().split()) for _ in xrange(n) ]
    ans = 0
    for i in range(n-1):
        det = tm[i+1][0]-tm[i][1]
        ans += (tm[i][1]-tm[i][0]+min(det, T1))*P1
        ans += max(0, min(det-T1, T2))*P2
        ans += max(0, det-T1-T2)*P3
    ans += (tm[n-1][1]-tm[n-1][0])*P1
    print ans

if __name__ == '__main__':
    main()
