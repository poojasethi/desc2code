(n,p1,p2,p3,T1,T2) = map(int,raw_input().split())
a = []
ans = 0
for i in range ( n ) :
	(c,d) = map(int,raw_input().split())
	a.append( (c,d) ) 
	ans += p1 * (d - c)

for i in range (1,n) :
	delta = a[i][0] - a[i-1][1]
	if delta >= T1 :
		ans += p1 * T1
		if delta >= T1 + T2 :
			ans += p2 * T2
			ans += (delta - T1 - T2) * p3
		else:
			ans += p2 * (delta - T1)
	else:
		ans += delta * p1
print ans
