R=lambda:map(int,raw_input().split())
n,P1,P2,P3,T1,T2=R()
s=[R() for i in ' '*n]
def calc(t):
    res=0
    if t>=T1:
        t-=T1
        res+=T1*P1
        if t>=T2:
            t-=T2
            res+=T2*P2
            res+=t*P3
        else:
            res+=t*P2
    else:
        res+=t*P1
    return res
ans=0
for a,b in s:
    ans+=(b-a)*P1
for i in xrange(n-1):
    ans+=calc(s[i+1][0]-s[i][1])
print ans
