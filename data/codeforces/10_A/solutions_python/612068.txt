n, p1, p2, p3, t1, t2 = map(int, raw_input().split())
l = [0] * 110
r = [0] * 110
for i in range(n):
    l[i], r[i] = map(int, raw_input().split())
k = 0
ans = 0
t = 0
#print l[0], r[n - 1]
for i in range(l[0], r[n - 1]):
    if(i >= l[k] and i < r[k]):
        ans += p1
        t = 0
    else:
        t += 1
        if(i >= r[k]):
            k += 1
        if(t <= t1):
            ans += p1
        elif(t <= t1 + t2):
            ans += p2
        else:
            ans += p3
print ans
