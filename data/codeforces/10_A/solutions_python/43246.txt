import sys

def ints():
    return [int(x) for x in sys.stdin.readline().split()]

n, p1, p2, p3, t1, t2 = ints()

a = []

for i in range(n):
    l, r = ints()
    a.append((l, r))

last = a[0][1]
power = (a[0][1] - a[0][0]) * p1

for (l, r) in a[1:]:
    c1 = min([l - last, t1])
    last += c1
    power += c1 * p1
    c2 = min([l - last, t2])
    last += c2
    power += c2 * p2
    power += (l - last) * p3
    power += (r - l) * p1
    last = r

print(power)
