import sys

def main():
	n, p1, p2, p3, t1, t2 = map (int, sys.stdin.readline().split())
	power = 0
	prev_r = 0
	first = True
	for i in xrange (n):
		l, r = map (int, sys.stdin.readline().split())
		if first:
			first = False
		else:
			d = l - prev_r
			power += min (d, t1) * p1
			d -= min (d, t1)
			power += min (d, t2) * p2
			d -= min (d, t2)
			power += d * p3
		power += (r - l) * p1
		prev_r = r
	print power


if len (sys.argv) > 1:
	sys.stdin = open (sys.argv[1], 'r')
main()
