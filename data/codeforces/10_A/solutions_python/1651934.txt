
def solve(time):
	ans=0
	if time>(t1+t2):
		ans+=(time-(t1+t2))*p3
		time=(t1+t2)
	if time>t1:
		ans+=(time-t1)*p2
		time=t1
	if time>0:
		ans+=time*p1
	return ans

# main
n,p1,p2,p3,t1,t2=map(int,raw_input().split(" "))
s=[0]
e=[0]
for i in range(n):
	a,b=map(int,raw_input().split(" "))
	s.append(a)
	e.append(b)
ans=0
for i in range(n):
	ans+=solve(s[i+1]-e[i]) if i!=0 else 0
	ans+=(e[i+1]-s[i+1])*p1
print ans
