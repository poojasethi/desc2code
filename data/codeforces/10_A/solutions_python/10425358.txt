n,p1,p2,p3,t1,t2 = map(int, raw_input().split())

totalPower = 0
lastR = 0

for i in xrange(n):
	l,r = map(int, raw_input().split())
	power = (r - l)*p1
	totalPower += power

	if lastR != 0:
		interval = l - lastR
		for j in xrange(interval):
			if j < t1:
				totalPower += p1
			elif j < t2 + t1:
				totalPower += p2
			else:
				totalPower += p3

	lastR = r

print totalPower
