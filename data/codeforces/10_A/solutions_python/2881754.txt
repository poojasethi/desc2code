n, p1, p2, p3, t1, t2 = map(int, raw_input().split())

last = -1
consumption = 0
for i in range(n):
    l, r = map(int, raw_input().split())
    consumption += (r - l) * p1

    if last != -1:
        consumption += min(l - last, t1) * p1
        consumption += min(max(l - last - t1, 0), t2) * p2
        consumption += max(l - last - t1 - t2, 0) * p3

    last = r

print consumption
