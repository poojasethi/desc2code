#include <bits/stdc++.h>

using namespace std;

int main () {
	string s;
	cin>>s;
	long long ans=0;
	long long sum=0;
	for(int i=0;i<s.size();i++){
		string q=s.substr(i,5);
		if(q=="heavy"){
			ans++;
		}
		if(q=="metal"){
			sum+=ans;
		}
	}
	cout<<sum;
	return 0;
}
