#include<cstdio>
#include<iostream>
#include<string>
using namespace std;
int main()
{
    string s;
    cin>>s;
    long long n=s.length(),count=0,ans=0;
    for(int i=0;i<=n-5;i++)
    {
        if(s.substr(i,5)=="heavy")count++;
        else if(s.substr(i,5)=="metal")ans+=count;
    }
    cout<<ans;
}
