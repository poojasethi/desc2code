#include <iostream>
#include <string>

using namespace std;

int main()
{
    long long h=0,s=0;
    string t,q,w;
    cin >> t;

    for(int i=0 ;i<t.size(); i++){
        q=t.substr(i,5);
        w=t.substr(i,5);
        if(q=="heavy")h++;
        if(w=="metal" && h>0)s+=h;
    }

    cout << s;
}
