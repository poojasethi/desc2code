#include <iostream>
using namespace std;

int main(){
  string s;
  cin >> s;
  long long heavy = 0;
  long long sum = 0;
  if(s.size()<5){cout << 0; return 0;}
  for(int i=0;i<s.size()-4;i++){
    if(s.substr(i,5) == "heavy") heavy++;
    if(s.substr(i,5) == "metal") sum += heavy;
  }
  cout << sum;
}
