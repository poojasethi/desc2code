#include<iostream>
#include<string>
using namespace std;

int main()
{
	long long i,a=0,ans=0;
	string s;
	cin>>s;
	if (s.size() <5){cout<<0<<endl;return 0;}
	for (i=0;i<s.length()-4;i++)
	{
		if (s.substr(i,5) == "heavy")a++;
		else if (s.substr(i,5)=="metal")ans+=a;
	}
	cout<<ans<<endl;
	return 0;
}
