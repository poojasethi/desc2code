#include <bits/stdc++.h>
using namespace std;

int main(){
    string s;
    long long res = 0, count = 0;
    cin >> s;
    for(int i = 0; i<(int)s.size()-4; i++)
    {
        if(s.substr(i, 5)=="metal") res+=count;
        else if(s.substr(i, 5)=="heavy") count++;
    }
    cout << res << endl;
}
