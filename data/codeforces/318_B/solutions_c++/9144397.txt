#include<iostream>
#include<string>
using namespace std;
int main()
{
    long long int cnt=0,st=0;
    string s;
    cin>>s;
    for(long long int i=0;i<s.length();i++)
    {
        if(s.substr(i,5)=="heavy") st++;
        if(s.substr(i,5)=="metal") cnt+=st;
    }
    cout<<cnt<<endl;
    return 0;
}
