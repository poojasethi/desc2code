#include <iostream>

using namespace std;

int main()
{
	string s;
	cin >> s;
	long long ans = 0, h = 0, m = 0;
	for (int i = 4; i < s.size(); i++)
	{
		string x = s.substr(i - 4, 5);
		if (x == "metal")
		{
			ans += h;
			m++;
		}
		else if (x == "heavy")
			h++;
	}
	cout << ans << endl;
	return 0;
}
