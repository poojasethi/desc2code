#include <iostream>
#include <string>
using namespace std;
 
string str, t;
int main() {
    long long c=0, h=0;
    cin >> str;
    for (int i=0; i<str.size(); ++i) {
        t = str.substr(i, 5);
        if (t=="heavy") ++h, i+=4;
        if (t=="metal") c+=h, i+=4;
    }
    cout << c;
    return 0;
}
