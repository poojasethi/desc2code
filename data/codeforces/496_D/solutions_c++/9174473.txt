#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<queue>
using namespace std;
int n,s[200005],a[200005],b[200005],sa[200005],sb[200005],as,bs,ss,ls;
pair<int,int>p[100005];
int main()
{scanf("%d",&n);
 for(int i=1;i<=n;i++)
    {scanf("%d",&s[i]);
	 sa[i]=a[0];sb[i]=b[0];
	 if(s[i]==1)a[++a[0]]=i;
	 else b[++b[0]]=i;
    }
 for(int i=a[0]+1;i<=200000;i++)a[i]=1000000;for(int i=b[0]+1;i<=200000;i++)b[i]=1000000;
 for(int i=n;i>=1;i--)
    {int j=1;as=bs=0;//cout<<endl<<i<<":";
     while(j<=n)
	  {//cout<<sa[j]<<" "<<a[sa[j]+i-1]<<" "<<b[sb[j]+i]<<endl;
	   if(a[sa[j]+i]<b[sb[j]+i])
         as++,j=a[sa[j]+i]+1,ls=1;
       else
         bs++,j=b[sb[j]+i]+1,ls=2;
  	  }
     if(j==n+1&&as!=bs&&((as>bs&&ls==1)||(as<bs&&ls==2)))
       {ss++;
        p[i].first=max(as,bs),p[i].second=i;
       }
     else
       p[i].first=1000000;
    }printf("%d\n",ss);
 sort(1+p,1+p+n);
 for(int i=1;i<=ss;i++)
    printf("%d %d\n",p[i].first,p[i].second);
 return 0;
}
