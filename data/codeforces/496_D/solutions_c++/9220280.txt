#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
const int maxn=100010;
const int INF=0x3f3f3f3f;
typedef pair<int,int> pii;
int n,a,ans,x[maxn<<1],y[maxn<<1],xn,yn;
pii p[maxn];
void run(int t)
{
    int i=0,j=0,sx=0,sy=0;
    while(i!=xn || j!=yn)
    {
        if(x[i+t]==INF && y[j+t]==INF) return ;
        if(x[i+t]<y[j+t])
        {
            sx++;
            i+=t;
            while(y[j+1]<x[i]) j++;
        }
        else
        {
            sy++;
            j+=t;
            while(x[i+1]<y[j]) i++;
        }
    }
    if(a==1 && sx<=sy || a==2 && sx>=sy) return ;
    if(a==1) p[ans++]=pii(sx,t);
    else p[ans++]=pii(sy,t);
}
int main()
{
    ios_base::sync_with_stdio(0);cin.tie(0);
    cin>>n;
    memset(x,INF,sizeof x);
    memset(y,INF,sizeof y);
    for(int i=0;i<n;i++)
    {
        cin>>a;
        if(a==1) x[++xn]=i;
        else y[++yn]=i;
    }
    for(int i=1;i<=max(xn,yn);i++) run(i);
    sort(p,p+ans);
    cout<<ans<<endl;
    for(int i=0;i<ans;i++) cout<<p[i].first<<" "<<p[i].second<<endl;
    return 0;
}
