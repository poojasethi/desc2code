#include<bits/stdc++.h>
using namespace std;
int cnt1[100009],cnt2[100009],a[100009];
vector<int> factor;
vector<pair<int,int> > ans;
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
		cnt1[i]=cnt1[i-1];
		cnt2[i]=cnt2[i-1];
		if(a[i]==1) cnt1[i]+=1;
		else cnt2[i]+=1;
	}
	for(int t=1;t<=n;t++){
		int last=0;
		int j=1,tmp=1,c1=0,c2=0,set1=0,set2=0;
		while(1){
		int x1=lower_bound(cnt1+1,cnt1+n+1,t+c1)-cnt1;
		int x2=lower_bound(cnt2+1,cnt2+n+1,t+c2)-cnt2;
		if(x1>n&&x2>n){
		//	tmp=0;
			break;
		}
		if(x1<x2){
			last=1;
			set1++;
		}
		else last=2,set2++;
		j=min(x1,x2);
		c1=cnt1[j];
		c2=cnt2[j];
	}
	int v=max(set1,set2);
	if(c1+c2<n||set1==set2||(set1<set2&&last==1)||(set1>set2&&last==2)) continue;
	ans.push_back(make_pair(v,t));

}
    sort(ans.begin(),ans.end());
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++){
		cout<<ans[i].first<<" "<<ans[i].second<<endl;
	}
}
