#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int sum1[100005],sum2[100005],a[100005];
int n;
vector<pair<int,int> > ans;

int main(void){
	int i,j,t;
	while(~scanf("%d",&n)){
		sum1[0]=sum2[0]=0;
		for(i=1;i<=n;i++){
			scanf("%d",&a[i]);
			sum1[i]=sum1[i-1];
			sum2[i]=sum2[i-1];
			if(a[i]==1) sum1[i]++;
			else sum2[i]++;
		}
		for(t=1;t<=n;t++){
			int p=0,g=0,last;
			int x=0,y=0,pos=0;
			while(pos<n){
				x=lower_bound(sum1,sum1+n+1,sum1[pos]+t)-(sum1);
				y=lower_bound(sum2,sum2+n+1,sum2[pos]+t)-(sum2);
				if(sum1[x]!=sum1[pos]+t && sum2[y]!=sum2[pos]+t) break;
				if(x<y) p++,last=1;
				else g++,last=2;
				pos=min(x,y);
			}
			if(pos<n) continue;
			if(p>g && last==1) ans.push_back(make_pair(p,t));
			if(p<g && last==2) ans.push_back(make_pair(g,t));
		}
		sort(ans.begin(),ans.end());
		printf("%d\n",ans.size());
		for(i=0;i<ans.size();i++)
		    printf("%d %d\n",ans[i].first,ans[i].second);
	}
	return 0;
}
