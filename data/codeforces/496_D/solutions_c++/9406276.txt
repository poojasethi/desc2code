#include <bits/stdc++.h>

using namespace std;
pair<int,int> ans[112345];
int s[112345],ocnt[112345],tcnt[112345],acnt=0;
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)cin>>s[i];
	for(int i=1;i<=n;i++)
	{
		ocnt[i]=ocnt[i-1];
		tcnt[i]=tcnt[i-1];
		if(s[i]==1)
			ocnt[i]++;
		else
			tcnt[i]++;
	}
	for(int i=1;i<=n;i++)
	{
		int ow=0,tw=0;
		bool ff=0;
		for(int j=1;j<=n;)
		{
			int opost=lower_bound(ocnt+1,ocnt+n+1,ocnt[j-1]+i)-ocnt;
			int tpost=lower_bound(tcnt+1,tcnt+n+1,tcnt[j-1]+i)-tcnt;
			if(opost>n&&tpost>n){
				ff=1;break;
			}
			if(opost<tpost)
				j=opost+1,ow++;
			else
				j=tpost+1,tw++;
		}
		if(ff)continue;
		if(ow>tw&&s[n]==1)
		{
			ans[acnt].first=ow;
			ans[acnt++].second=i;
		}
		else if(ow<tw&&s[n]==2)
		{
			ans[acnt].first=tw;
			ans[acnt++].second=i;
		}
	}
	sort(ans,ans+acnt);
	cout<<acnt<<endl;
	for(int i=0;i<acnt;i++)
	{
		cout<<ans[i].first<<' '<<ans[i].second<<endl;
	}
	return 0;
}
