num_officials = int(raw_input())
query_list = []
for i in range(0, num_officials):
	tokens = raw_input().split(' ')
	tokens = [int(token) for token in tokens]
	query_list.append(tokens)

def can_pay(query):
	n = query[0]
	l = query[1]
	r = query[2]
	if n%l <= n/l * (r-l):
		return 'Yes'
	else:
		return 'No'

for i in range(0, len(query_list)):
	print can_pay(query_list[i])
