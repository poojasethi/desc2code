import sys

t = int(sys.stdin.readline())

for i in range(0,t):
    line = sys.stdin.readline()
    a = line.split()
    (n,l,r) = (int(a[0]), int(a[1]), int(a[2]))
    k = n / l
    diff = n - k*l
    dendiff = r - l    
    if diff <= dendiff * k:
        print "Yes"
    else:
        print "No"
