#define F1(x,y,z) for(int x=y;x<z;x++)
#define F2(x,y,z) for(int x=y;x<=z;x++)
#define F3(x,y,z) for(int x=y;x>z;x--)
#define F4(x,y,z) for(int x=y;x>=z;x--)
#define pb push_back
#define LL long long
#define co complex<double>
 
#define MAX 100005
#define AMAX 1500
#define MOD 1000000007

#define f(c,d) ((1<<(c))*(d))

using namespace std;

char x[2][2005],y[2005];
int n,m;
LL ans,M1=MOD-7+9,M2=MOD-7+21,dp[2005][2005][2];
pair<LL,LL> tim[2005],wq[2005],ta,tb;

void go(bool fst){
	F2(a,0,n){
		F1(b,0,2)dp[a][0][b]=1;
		if(a){
			F2(b,0,m-2)F1(c,0,2)if(x[c][a-1]==y[b]&&x[c^1][a-1]==y[b+1])dp[a][b+2][c^1]=(dp[a][b+2][c^1]+dp[a-1][b][c])%MOD;	
			F2(b,0,m-1)F1(c,0,2)if(x[c][a-1]==y[b])dp[a][b+1][c]=(dp[a][b+1][c]+dp[a-1][b][c])%MOD;
			F1(c,0,2){
				ta=tb=make_pair(0LL,0LL);
				for(int b=1;b*2<=m&&a-b>=0;b++){
					ta=make_pair((ta.first*26+x[c][a-b])%M1,(ta.second*26+x[c][a-b])%M2);
					tb=make_pair((tb.first+x[c^1][a-b]*tim[b-1].first)%M1,(tb.second+x[c^1][a-b]*tim[b-1].second)%M2);
					if(fst||b*2!=m)if(b>1)if(make_pair((ta.first*tim[b].first+tb.first)%M1,(ta.second*tim[b].second+tb.second)%M2)==wq[b*2])dp[a][b*2][c^1]=(dp[a][b*2][c^1]+1)%MOD;
				}
			}
		}
		F1(c,0,2){
			ta=tb=make_pair(0LL,0LL);
			for(int b=1;b*2<=m&&a-1+b<n;b++){
				ta=make_pair((ta.first*26+x[c][a-1+b])%M1,(ta.second*26+x[c][a-1+b])%M2);
				tb=make_pair((tb.first+x[c^1][a-1+b]*tim[b-1].first)%M1,(tb.second+x[c^1][a-1+b]*tim[b-1].second)%M2);
				if(b>1)if(make_pair((ta.first*tim[b].first+tb.first)%M1,(ta.second*tim[b].second+tb.second)%M2)==make_pair((wq[m].first-wq[m-b*2].first*tim[b*2].first%M1+M1)%M1,(wq[m].second-wq[m-b*2].second*tim[b*2].second%M2+M2)%M2)&&(fst||b*2!=m))ans=(ans+dp[a][m-b*2][c])%MOD;
			}
		}
		ans=(ans+dp[a][m][0]+dp[a][m][1])%MOD;
	}
}

int main(){
	scanf("%s%s%s",x[0],x[1],y);
	while(x[0][n])n++;
	while(y[m])m++;
	F1(a,0,2)F1(b,0,n)x[a][b]-='a';
	F1(a,0,m)y[a]-='a';
	if(m==1){
		F1(a,0,2)F1(b,0,n)if(x[a][b]==y[0])ans++;
	}else if(m==2){
		F1(a,0,2)F1(b,1,n)if(x[a][b]==y[0]&&x[a][b-1]==y[1])ans++;
		F1(a,0,2)F1(b,1,n)if(x[a][b]==y[1]&&x[a][b-1]==y[0])ans++;
		F1(a,0,n)if(x[0][a]==y[0]&&x[1][a]==y[1])ans++;
		F1(a,0,n)if(x[0][a]==y[1]&&x[1][a]==y[0])ans++;
	}else{
		tim[0]=make_pair(1LL,1LL);
		F2(a,1,m)tim[a]=make_pair(tim[a-1].first*26%M1,tim[a-1].second*26%M2);
		wq[0]=make_pair(0LL,0LL);
		F1(a,0,m)wq[a+1]=make_pair((wq[a].first*26+y[a])%M1,(wq[a].second*26+y[a])%M2);
		go(1);
		reverse(y,y+m);
		F1(a,0,m)wq[a+1]=make_pair((wq[a].first*26+y[a])%M1,(wq[a].second*26+y[a])%M2);
		memset(dp,0,sizeof(dp));
		go(0);
	}
	printf("%I64d\n",ans);
	//system("pause");
	return 0;
}
