n = input()
a = raw_input()
keys = {}
count = 0
for i in range(1, len(a), 2):
    if a[i-1] not in keys:
        keys[a[i-1]] = 1
    else:
        keys[a[i-1]] += 1

    if a[i].lower() not in keys or keys[a[i].lower()] == 0:
        count += 1
    else:
        keys[a[i].lower()] -= 1

print count
