import sys

N = int(sys.stdin.readline())

inp = sys.stdin.readline()

count = 0
keys = {}

for i in range(N-1):
  k = inp[i*2]
  D = inp[i*2 + 1]

  if k not in keys:
    keys[k] = 0
  keys[k] += 1
  
  if D.lower() in keys and keys[D.lower()] > 0:
    keys[D.lower()] -= 1
  else:
    count += 1

print count
