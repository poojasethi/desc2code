n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
b = filter(lambda x: x<1, a)
if k <= len(b):
    print sum(a) - 2 * sum(b[:k])
else:
    fac = ((k - len(b)) % 2) * 2
    fac2 = a[len(b)] if len(a) - len(b) else abs(b[-1])
    print sum(a) - 2 * sum(b) - fac *(min(abs(b[-1]), fac2) if len(b) else a[0])
