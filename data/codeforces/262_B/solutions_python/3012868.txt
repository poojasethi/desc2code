
myin = map(int, raw_input().split())

nums = map(int, raw_input().split())

nums.sort()

count = 0

while (count < myin[1]) and (nums[count] <= 0) and (count < myin[0] - 1):
    nums[count] = -nums[count]
    count += 1

nums.sort()

while count < myin[1]:
    nums[0] = -nums[0]
    count += 1

print sum(nums)
