def raw():
    return raw_input().strip()


def ints():
    return map(int, raw().split())

n, k = ints()
a = ints()
rk = 0

c = 0
while rk < k and a[rk] < 0:
    a[rk] = a[rk] * -1
    rk += 1
    if rk >= n:
        break

rest = k - rk
if rest <= 0:
    print sum(a)
    exit(0)

m = min(a)
mi = a.index(m)
if rest % 2 != 0:
    a[mi] = m * -1
print sum(a)
