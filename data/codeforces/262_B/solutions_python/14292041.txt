def roma(n, k, numbers):
    i = 0
    while i < n and k > 0 and numbers[i] < 0:
        numbers[i] *= -1
        k -= 1
        i += 1
    
    if i > 0:
        numbers.sort()

    if k % 2 == 1:
        numbers[0] *= -1

    return sum(numbers)

n, k = map(int, raw_input().split())
lst = map(int, raw_input().split())
print roma(n, k, lst)
