import sys
n,k = map(int, raw_input().split())
inp = map(int, raw_input().split())
cnt = 0
for i in xrange(len(inp)):
    if inp[i] < 0:
        inp[i] *= -1
        cnt += 1
    else:
        break
    if cnt == k:
        print sum(inp)
        sys.exit(0)
inp.sort()
inp[0] *= (-1)**(k-cnt)
print sum(inp)
