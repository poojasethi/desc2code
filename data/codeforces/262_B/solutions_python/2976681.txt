r = raw_input().split()
swap = int(r[1])
s = raw_input().split()
for x in range(len(s)):
    s[x] = int(s[x])
s.sort()
a = -1
for x in range(swap):
    if x == len(s) or s[x] >= 0:
        a = x
        break
    s[x] = -1*s[x]
if not a == -1 and (swap - a) % 2 == 1:
    s.sort()
    s[0] = -1*s[0]
total = 0
for x in s:
    total += x
print total
