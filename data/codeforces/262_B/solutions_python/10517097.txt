n,k=map(int,raw_input().split())
L=map(int,raw_input().split())
count=0
L.sort()
for x in L:
	if(x<0):
		count+=1
if(count==k):
	s=0
	for x in L:
		s+=abs(x)
elif(count<k):
	for x in xrange(n):
		L[x]=abs(L[x])
	k-=count
	if(k%2==0):
		s=sum(L)
	else:
		s=sum(L)-2*min(L)
else:
	s=0
	c=0
	for x in L:
		if(c==k):
			s+=x
			continue
		s+=abs(x)
		c+=1	
print s
