N,S = map(int,raw_input().split())
tmp = raw_input().split()
lst = [int(x) for x in tmp]

#count -ve elements
i = 0
while S > 0 and i<N:
    t = S
    if lst[i]<=0:
        lst[i] *= -1
        S-=1
    i+=1
    if t == S:
        break
lst.sort()
if S>0:
    if S%2 == 1:
        lst[0]*=-1
print sum(lst)
