#include <bits/stdc++.h>
using namespace std;

int main() {
	int n,k,sum=0;
	cin>>n>>k;
	int a[n];
	for(int i=0;i<n;i++){
		cin>>a[i];
		if(a[i]<0&&k>0){
			a[i]*=-1;
			k--;
		}}
		sort(a,a+n);
		if(k>0){
			for(int j=0;j<k;j++){
				a[0]*=-1;
			}
		}
	for(int i=0;i<n;i++){
		sum+=a[i];
	}
	cout<<sum<<endl;
	
	return 0;
}
