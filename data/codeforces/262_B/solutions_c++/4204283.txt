#include <iostream>
#include <stdlib.h>

using namespace std;

int main(){
	int n , m , i , s=0, *a; cin >> n >> m; a=new int[n]; for(int i=0 ; i<n ; i++) cin >> a[i];
	for(i=0 ; i<m && a[i]<0 ; i++) a[i]*=-1;
	if(i!=m && m%2!=i%2){ if(a[i]>a[i-1]) a[i-1]*=-1; else a[i]*=-1;}
	for( int i=0 ; i<n ; i++) s+=a[i];
	cout << s << endl;
}
