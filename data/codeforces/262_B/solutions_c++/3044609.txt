#include <iostream>
#include <algorithm>

using namespace std;

int n, k, ret, niz[111111];

int main()
{
  cin >> n >> k;
  for (int i = 0; i < n; i++)
  {
    cin >> niz[i];
    if (k && niz[i] < 0) niz[i] *= -1, k--;
  }
  sort(niz, niz + n);
  for (; k; k--) *niz *= -1;
  for (int i = 0; i < n; i++) ret += niz[i];
  cout << ret << endl;
  return 0;
}
