#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define INF 99999999
using namespace std;

int main(){
    int sum=0,a,n,k,m=INF;
    cin>>n>>k;
    for(int i=0;i<n;++i){
        scanf("%d",&a);
        if(a<0 && k)--k,a=-a;
        sum+=a;
        m=min(m,a);
    }
    if(k && k%2)sum-=2*m;
    cout<<sum<<endl;
    return 0; 
}
