#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int n, k;
	cin >> n >> k;
	int a[n];
	for (int i=0; i<n; ++i)
		cin >> a[i];
	for (int i=0; i<n; ++i)
		if (k && a[i]<0)
		{
			a[i]*=-1;
			--k;
		}
	sort (a, a+n);
	if (k&1 && k>0)
		a[0]*=-1;
	int s=0;
	for (int i=0; i<n; ++i)	s+=a[i];
	cout << s;
	return 0;
}
