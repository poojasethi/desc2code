#include<cstdio>
#include<algorithm>
using namespace std;
int a[100005];
int main()
{
	int n,i,k,min,sum;
	while(~scanf("%d%d",&n,&k))
	{
	min=10000000;sum=0;
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	for(i=1;i<=n;i++)
	{
	if(a[i]<0&&k>0){a[i]=-a[i];k--;}
	sum+=a[i];
	min=min<a[i]?min:a[i];
	}
	if(k%2)sum-=2*min;
	printf("%d\n",sum);
	}
}
