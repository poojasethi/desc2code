#include<iostream>
#include<algorithm>
using namespace std;
int a[100100];
int main()
{
    int n,k,i,j,sum=0;
    cin>>n>>k;
    for(i=0;i<n;i++) cin>>a[i];
    sort(a,a+n);
    i=0;
    while(a[i]<0&&k>0) {a[i]=-a[i];k--;i++;}
    sort(a,a+n);
    for(i=0;i<n;i++) sum+=a[i];  
    if(k%2) sum-=2*a[0];  
    cout<<sum<<endl;
    //system("pause");
    return 0;
}
