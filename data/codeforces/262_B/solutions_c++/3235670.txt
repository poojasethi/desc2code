#include <stdio.h>
#include <stdlib.h>

int main (void) {
	int n, k, i, m,  v[100000], ans;
	scanf ("%d%d", &n, &k);
	ans = 0;
	m = 100000;
	for (i = 0; i < n; i++) {
		scanf ("%d", &v[i]);
		if (m > abs(v[i]))	m = abs(v[i]);
		ans += v[i];
	}
	for (i = 0; i < k && i < n && v[i] <= 0; i++)	ans += 2*(-v[i]);
	if ((k-i)&1)	ans -= 2*m;
	printf ("%d\n", ans);
	return 0;
}
