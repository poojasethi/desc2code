#include<stdio.h>
#define REP(i,a) for(i=0;i<a;i++)
using namespace std;
int main() {
	int n,k,i,indx,A[100005];
	scanf("%d%d",&n,&k);
	int min=100000;
	REP(i,n) {
		scanf("%d",&A[i]);
		if(A[i]<0 && k) {
			A[i]=-1*A[i];
			k--;
		}
		if(A[i]<min) {
			min=A[i];
			indx=i;
		}
	}
	while(k--) A[indx]=-A[indx];
	int sum=0;
	REP(i,n) sum+=A[i];
	printf("%d\n",sum);
	return 0;
}
