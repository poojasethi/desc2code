#include <algorithm>
#include <iostream>
using namespace std;
int main ()
{
	int n , k , arr [100000] , sum = 0 , a = 0;
	cin>>n>>k;
	for (int i = 0; i < n; i++)
	{
		cin>>arr [i];
		if (a < k && arr [i] < 0)
		{
			arr [i] *= -1;
			a++;
		}
		sum += arr [i];
	}
	sort (arr , arr + n);
	if ((k - a) % 2 != 0)
		sum -= 2 * arr [0];
	cout<<sum<<endl;
	return 0;
}
