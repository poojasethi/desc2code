#include <iostream>
#include <algorithm>

using namespace std;

int main(){

	int i, n, k, max=0;
	
	cin >> n >> k;
	
	int a[n];
	
	for(i=0; i<n; i++){
		cin>>a[i];
		
		if (k>0 && a[i]<0){
			a[i]*=-1;
			k--;
		}
	}
	
	sort(a, a+n);
	
	while(k>0){
		a[0]*=-1;
		k--;
	}
	
	for(i=0; i<n; i++){
		max+=a[i];
	}
	
	cout << max << endl;
	return 0; 
}
