#include <iostream>
#include <algorithm>
using namespace std;
int a[100000],i,n,s,m,k,z;
main()
{
cin>>n>>m;
for (i=1; i<=n; i++)
cin>>a[i];
sort(a+1,a+n+1);
k=m;
z=0;
while (k>0)
{
z++;
if (a[z]<0) {a[z]=a[z]*-1; k--;}
if (z==n) break;  
}
if ((k>0) && (k % 2==1))
{
sort(a+1,a+n+1);
a[1]=a[1]*-1;         
}
for (i=1; i<=n; i++)
s+=a[i];
cout<<s;

}
