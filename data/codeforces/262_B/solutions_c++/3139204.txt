#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

int n, k, a[100010], ans;

int main (){
	scanf("%d%d", &n, &k);
	for(int i = 0; i < n; i++)
		scanf("%d", &a[i]);
	for(int i = 0; i < n; i++)
		if(k > 0 && a[i] < 0) a[i] *= -1, k--;
	sort(a, a+n); if(k&1) a[0] *= -1;
	for(int i = 0; i < n; i++) ans += a[i];
	printf("%d\n", ans);
}
