#include <iostream>
#include <algorithm>
using namespace std;

int main(){
	int n,k,a[100000],i,s = 0;
	cin>>n>>k;
	for(i=0;i<n;i++) cin>>a[i];
	for(i=0;i<n;i++){
		if(a[i] < 0 && k > 0){
			a[i] = -a[i];
			k--;
		}
		if(k == 0) break;
	}
	sort(a,a + n);
	if(k % 2 == 1) a[0] = -a[0];
	for(i=0;i<n;i++) s += a[i];
	cout<<s;
}
