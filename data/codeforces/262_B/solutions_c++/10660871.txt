#include <iostream>
#include <algorithm>
using namespace std;
int main(){
	int n,k,a[100100],cnt=0;
	cin>>n>>k;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	for(int i=0;i<n;i++){
		if(k>0&&a[i]<0){
			a[i]*=-1;
			k--;
		}
	}
	if(k>0){
		sort(a,a+n);
		if(k%2==1)a[0]*=-1;	
	}
	for(int i=0;i<n;i++)cnt+=a[i];
	cout<<cnt<<endl;
	
}
