#include <iostream>
using namespace std;


int main()
{
	int n,k;
	cin >> n >> k;
	long long Sum = 0;
        int Min = (int)1e5 +1;
	for ( int i = 0 ; i < n ; i++ ){
		int x;
		cin >> x;
		if ( x < 0 && k ){
			x = -x;
			k--;
		}
		Min = min( Min , x );
		Sum += x;

	}
	if ( k%2 )
		Sum -= 2*Min;
	cout << Sum << endl;

	return 0;
}
