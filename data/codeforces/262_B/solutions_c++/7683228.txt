#include<cstdio>
#include<algorithm>
using namespace std;
int a1[100000];
int main()
{
	int n,k,a=-1,b=10000,o=0;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;++i)
	{
		scanf("%d",a1+i);
		if(a1[i]<0)
		{if(k)--k,a1[i]=-a1[i];a=i;}
		o+=a1[i];
	}
	if(k&1)
	{
		if(a+1<n)b=a1[a+1];
		if(~a)b=min(b,a1[a]);
		o-=2*b;
	}
	printf("%d\n",o);
}
