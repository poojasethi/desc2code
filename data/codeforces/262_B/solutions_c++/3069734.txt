#include <iostream>
#include <algorithm>
using namespace std;
int a[100000];
int main() {
   int n,k;
   cin>>n>>k;
   for(int i=0;i<n;++i) cin>>a[i];
   
   for(int i=0;i<n && k && a[i]<0;++i) {
      a[i]*=-1;
      --k;
   }
   if(k%2) {
      sort(a,a+n);
      a[0]*=-1;
   }
   int res=0;   
   for(int i=0;i<n;++i) res+=a[i];
   cout<<res;
   return 0;
}
