#include<iostream>
#include<algorithm>
using namespace std;

int a[100010];

int main()
{
	int n, k, s=0;
	cin>>n>>k;
	for(int i=0; i<n; i++)
	{
		cin>>a[i];
	}
	sort(a, a+n);
	for(int i=0; i<n && a[i]<0 && k; i++)
	{
		a[i]*=-1;
		k--;
	}
	if(k%2)
	{
		sort(a, a+n);
		a[0]*=-1;
	}
	for(int i=0; i<n; i++)
	{
		s+=a[i];
	}
	cout<<s;
	return 0;
}
