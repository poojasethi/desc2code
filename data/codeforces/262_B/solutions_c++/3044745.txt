#include <iostream>

using namespace std;

int main()
{
    long n, k, a, m = 100000, sum = 0;
    cin >> n >> k;
    while (n--) {
        cin >> a;
        if (k && a < 0) {
            a = -a;
            k--;
        }
        if (a < m) {
            m = a;
        }
        sum += a;
    }
    if (k%2) {
        sum -= 2*m;
    }
    cout << sum;
}
