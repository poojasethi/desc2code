#include<cstdio>
int main()
{
    int n,k,w=0;
    int v[100005];
    scanf("%d %d",&n,&k);
    for(int i=0;i<n;i++)
        scanf("%d",&v[i]);
    for(int i=0;i<n && k>0;i++)
     if(v[i]<0){v[i]=-v[i];k--;}
     else break;
    int m=100000;
    for(int i=0;i<n;i++){w+=v[i];if(m>v[i]) m=v[i];}
    printf("%d",w-2*((k&1))*m);
    return 0;
}
