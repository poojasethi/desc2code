#include<stdio.h>
int abs(int a)
{
	return a<0?-a:a;
}
int main()
{
	int n,i,k,a,b,sum;
	sum=0;
	scanf("%d %d",&n,&k);
	b=999999999;
	for(i=0;i<n;i++)
	{
		scanf("%d",&a);
		if(b>abs(a))
		{
			b=abs(a);
		}
		if(a<0&&k>0)
		{
			k--;
			sum+=-a;
		}
		else
		{
			sum+=a;
		}
	}
	if(k%2)
	{
		sum-=b*2;
	}
	printf("%d\n",sum);
	return 0;
}
