#include <cstdio>
#include <algorithm>
using namespace std;

int n,k;
int m[123456];

int main(){
int i,j;
while(~scanf("%d%d",&n,&k)){
for(i=0;i<n;i++){
scanf("%d",&m[i]);
if(k&&m[i]<=0){
k--;
m[i]=-m[i];
}
}

if(k%2){
sort(m,m+n);
m[0]=-m[0];
}

for(i=j=0;i<n;i++)
j+=m[i];
printf("%d\n",j);

}

return 0;
}
