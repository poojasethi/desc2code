#include<iostream>
#include<algorithm>
using namespace std;
int mass[100002];
int main()
{
int n,k,i,x,sum=0;
cin>>n>>k;
x=k;
for(i=0;i<n;i++)cin>>mass[i];
for(i=0;i<k;i++)
{
if(mass[i]<0){mass[i]=-mass[i];x--;}
else break;
}
k=x;
sort(mass,mass+n);
if(k%2)mass[0]=-mass[0];
for(i=0;i<n;i++)sum+=mass[i];
cout<<sum;
}
