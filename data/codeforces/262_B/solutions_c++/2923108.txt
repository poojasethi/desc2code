#include<iostream>
using namespace std;
int main()
{
    int n,k,nMin(10000),total(0);
    cin>>n>>k;
    for(int i=0;i<n;i++)
    {
        int x;
        cin>>x;
        if(x<0&&k){x=-x;k--;}
        if(x<nMin)nMin=x;
        total+=x;
    }
    if(k%2==1)total-=(2*nMin);
    cout<<total<<endl;
    return 0;
}
