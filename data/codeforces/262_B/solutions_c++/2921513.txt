#include <iostream>
#include <cstdlib>
using namespace std;
int n,k,a[100000],s,m;
int main(){
	cin>>n>>k;
	m=1000000;
	for(int i=0;i<n;i++) { 
		cin>>a[i];
		if(m>abs(a[i])) m=abs(a[i]);
	}
	for(int i=0;i<n;i++) {
		if((a[i]<0)&&(k)) a[i]*=-1, k--;
		s+=a[i];	
	}
	if(k&1) s-=2*m;
	cout<<s;
	return 0;
}
