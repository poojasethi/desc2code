#include<cstdio>
#include<algorithm>
using namespace std;

int n,m;
int a[111111];
int x,ans;

int main() {
	scanf("%d%d", &n, &m);
	for(int i=0;i<n;i++)
		scanf("%d", &a[i]);
	x=m;
	for(int i=0;i<m;i++){
		if(a[i]<0){a[i]=-a[i];x--;}
	}
	if(x&1){
		sort(a,a+n);
		a[0]=-a[0];
	}
	ans=0;
	for(int i=0;i<n;i++)
		ans+=a[i];
	printf("%d\n",ans);

	return 0;
}
