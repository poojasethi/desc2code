#include <iostream>
using namespace std;

int n;

int main() {
	int k, n;
	cin >> n >> k;
	long long sum = 0;
	int tmp;
	int min = 10001;
	for (int i = 0; i < n; i++) {
		cin >> tmp;
		if (tmp <= 0 && k-- > 0) tmp = -tmp;
		sum += tmp;
		if (min > tmp) min = tmp;
	}
	if (k > 0 && k %2 != 0) {
		sum -= 2 * min;
	}
	cout << sum << endl;

	return 0;
}
