#include <cstdio>
int main(int argc, char const *argv[])
{
	int n, k, a;
	int s = 0;
	int smallest = 10000;
	scanf("%d%d", &n, &k);
	for (int i = 0; i < n; ++i)
	{
		scanf("%d", &a);
		if (a < 0 && k > 0) {
			a = -a;
			--k;
		}
		if (a >= 0 && a < smallest) {
			smallest = a;
		}
		s += a;
	}
	if (k % 2 != 0) {
		s -= smallest * 2;
	}
	printf("%d\n", s);
	return 0;
}
