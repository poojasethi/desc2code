#include<iostream>
#include<algorithm>
using namespace std;
int s[100001];
int main(){
	int n,k;
	cin >>n>>k;
	for(int i=0;i<n;i++)
		cin >>s[i];
	int q=0;
	for(int i=0;i<n && q<k && s[i]<0;i++){
		s[i]*=-1;
		q++;
	}
	int sum=0;
	sort(s,s+n);
	for(int i=0;i<k-q;i++)
		s[0]*=-1;
	for(int i=0;i<n;i++)
		sum+=s[i];
	cout <<sum<<endl;
}
