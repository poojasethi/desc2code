#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,k,a[100010],i,sum=0;
	cin>>n>>k;
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(i=0;i<n && k>0;i++)
	{
		if(a[i]>0)
			break;
		a[i]=-a[i];
		k--;
	}
	sort(a,a+n);
	if(k>0 && k&1)
		a[0]=-a[0];
	for(i=0;i<n;i++)
		sum+=a[i];
	cout<<sum<<endl;
	return 0;
}
