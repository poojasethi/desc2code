n = int(raw_input())
a = map(int, raw_input().split(' '))

neg_count = 0
clen = 0
folders = []
for i in xrange(len(a)):
	if a[i] < 0:
		if neg_count == 2:
			folders.append(clen)
			clen = 1
			neg_count = 1
		else:
			clen += 1
			neg_count += 1
	else:
		clen += 1
folders.append(clen)
print len(folders)
print " ".join(map(str, folders))
