n = input()
profit = map(int, raw_input().split())
ans = []
cnt = 0
negative = 0
for i in range(len(profit)):
  if profit[i] < 0:
    negative += 1 
  if negative >= 3:
    ans.append(cnt)
    cnt = 0
    negative = 1
  cnt += 1
ans.append(cnt)
print(len(ans))
print(' '.join(map(str, ans)))
