n = int(raw_input())
a = [ int(x) for x in raw_input().split() ]

ans = []
count = 0
mi = 0
for x in a :
    if x < 0:
        if mi >= 2:
            ans.append(count)
            mi = 0
            count = 0
        mi += 1
    count += 1

if count > 0:
    ans.append(count)
print len(ans)
print " ".join([ str(x) for x in ans] )
