n = raw_input()
a = raw_input()

a = a.split(" ")
count_all = 0
count_minus = 0
foldersize_list = []
for i, a_i in enumerate(a):
    count_all += 1
    if int(a_i) < 0:
        if count_minus == 2:
            foldersize_list.append(count_all-1)
            count_minus = 1
            count_all = 1
        else:
            count_minus += 1
if count_all > 0:
    foldersize_list.append(count_all)

print(len(foldersize_list))
print(' '.join(str(a_i) for a_i in foldersize_list))
