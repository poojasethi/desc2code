def main():
    n = int(raw_input())
    data = map(int, raw_input().split())
    res = []
    i, acc = 0, 0
    while i < n:
        if data[i] < 0:
            acc += 1
        if acc >= 3:
            acc = 1
            res.append(i - sum(res))
        i += 1
    res.append(i - sum(res))
    print len(res)
    print ' '.join(map(str, res))


if __name__ == '__main__':
    main()
