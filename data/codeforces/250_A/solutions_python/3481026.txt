def divide(seq):
    res = []
    cnt = 0
    neg = 0
    for item in seq:
        cnt += 1
        if item < 0:
            neg += 1
            if neg > 2:
                res.append(cnt - 1)
                neg = 1
                cnt = 1
    res.append(cnt)
    return res

if __name__ == "__main__":
    items_cnt = int(raw_input())
    seq = map(int, raw_input().split(' '))
    res = map(str, divide(seq))
    print len(res)
    print ' '.join(res)
