n = input()
a = map(int, raw_input().strip().split())
minn = 0
co = 0
coop = []
for i in range(n):
    if a[i] < 0:
        minn = minn+1
    if minn <= 2 and i+1 < n:
        co = co+1
    elif minn > 2 and i+1 < n:
        coop.append(co)
        co = 1
        minn = 1
    elif minn > 2 and i+1 == n:
        coop.append(co)
        co = 1
        coop.append(co)
    elif i+1 == n:
        co = co+1
        coop.append(co)
print len(coop)        
print ' '.join(str(h) for h in coop)
