def main():
    T = int(raw_input())
    a = map(int,raw_input().split())
    b = [0]
    minus = 0 # number of minus data in each b
    result = ""
    for i in range(T):
        if a[i] >= 0:
            b[-1] += 1
        else:
            minus += 1
            if minus == 3:
                result += str(b[-1]) + " "
                b.append(0)
                minus = 1
            b[-1] += 1
            
    result += str(b[-1])
    print(len(b))
    print(result)

main()
