import sys

n = int(sys.stdin.readline())
days = [int(x) for x in sys.stdin.readline().strip().split()]

l = []
l.append([])
index = 0
minuses = 0

for i in range(len(days)):
    if days[i] < 0:
        minuses += 1
        if minuses > 2:
            minuses = 1
            l.append([])
            index += 1
            
    l[index].append(days[i])

##print l
print len(l)

s = ''
for x in l:
    s += str(len(x)) + ' '
