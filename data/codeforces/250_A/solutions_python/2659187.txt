import sys

n = int(sys.stdin.readline().strip())

a = [int(x) for x in sys.stdin.readline().strip().split()]

neg = 0
ans = []

for x in a:
    if x < 0:
        neg += 1
    if neg > 2:
        neg -= 2
        ans.append(1)
    else:
        if len(ans) == 0:
            ans.append(1)
        else:
            ans[-1] += 1

print(len(ans))
print(' '.join([str(x) for x in ans]))
