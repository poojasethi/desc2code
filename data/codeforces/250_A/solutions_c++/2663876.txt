#include<stdio.h>
int main()
{
  int n,a[100],i,k=0,l=0,k1,l1=0;
  scanf("%i",&n);
  for(i=0;i<n;i++)
  {
     scanf("%i",&k1);
     if(k1<0)k++;
     if(k==3)
     {
        a[l++]=i-l1;
        k=1;
        l1=i;
     }
  }
  a[l++]=n-l1;
  printf("%i\n",l);
  for(i=0;i<l;i++)printf("%i ",a[i]);
  return 0;
}
