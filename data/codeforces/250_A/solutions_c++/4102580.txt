#include<stdio.h>
int main()
{
	int n,i,j,l,sum;
	int a[1005],b[1005];
	
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	j=1;l=0;
	sum=0;
	for(i=1;i<=n;i++)
	{
		if(a[i]<0)sum++;
		if(sum==3)
		{
			sum=1;
			b[j]=i-l-1;
			j++;
			l=i-1;
		}
	}
	printf("%d\n",j);
	
	for(i=1;i<j;i++)
		printf("%d ",b[i]);
	printf("%d\n",n-l);
}
