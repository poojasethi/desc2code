#include <stdio.h>
int n,t,ans[100],k,z,v;
int main(){
    scanf("%d",&n);
    while(n--){
        scanf("%d",&t);
        if(t<0) z++;
        if(z==3) {
            ans[k++]=v;
            v=0;
            z=1;
        }
        v++;
    }
    ans[k++]=v;
    printf("%d\n",k);
    int i;
    for(i=0;i<k;i++) printf("%d ",ans[i]);
    return 0;
}
