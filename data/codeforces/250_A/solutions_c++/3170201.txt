#include<iostream>
using namespace std;
int main()
{
int n,a[105];
int neg=0,cur=0,b[105]={0};
cin>>n;
int j=0;
for(int i=0;i<n;i++)
	{
		cin>>a[i];
        if(a[i]>=0) cur++;
		else
			if(neg==2)
			{b[j++]=cur;neg=1;cur=1;}
			else
			{cur++;neg++;}
		if(i==n-1)
		b[j]=cur;
    }
cout<<j+1<<endl;
for(int i=0;i<j+1;i++)
	cout<<b[i]<<" "<<endl;
}
