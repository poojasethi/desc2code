#include <iostream>
#include <vector>

using namespace std;

int main()
{
	int n, c1=0, a, p = 0;
	cin >> n;
	vector <int> g;

	for(int i = 0; i < n; i++)
	{
		cin >> a;
		p++;
		if (a < 0) c1++;

		if (c1 == 3) c1 = 1, g.push_back(p-1), p=1;

	}
	g.push_back(p);
	cout << g.size() << endl;
	for(int i = 0; i < (int)g.size(); i++) cout << g[i] << " ";


    return 0;
}
