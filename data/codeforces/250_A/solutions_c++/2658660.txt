#include <cstdio>

int n,k,now,fu,a[110];

int main() {
	scanf("%d",&n);
	for (int i=1;i<=n;++i) {
		scanf("%d",&k);
		if (k>=0) ++now;
		else
			if (fu<2) ++now,++fu;
			else a[++a[0]]=now,now=fu=1;
	}
	if (now) a[++a[0]]=now;
	printf("%d\n",a[0]);
	for (int i=1;i<a[0];++i) printf("%d ",a[i]);
	printf("%d\n",a[a[0]]);
	return 0;
}
