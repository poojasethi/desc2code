int a[101],b[101],n,i,k,s,l;

{
    cin>>n;
    k=0; s=1; l=0;
    for (i=1;i<=n;i++)
    {
        cin>>a[i];
        if (a[i]<0) k++;
        if (k==3) {b[s]=i-l-1;k=1;l=i-1;s++;}
    }
    if(l<n){b[s]=n-l;s++;}
    cout<<s-1<<endl;
    for(i=1;i<=s-1;i++) 
     cout<<b[i]<<" ";
 return 0;
}
