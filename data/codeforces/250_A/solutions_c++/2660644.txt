#include<cstdio>
int n,x,cnt,len,res,o[10005];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++){
		scanf("%d",&x); if (x<0) cnt++;
		if (cnt==3) o[++len]=res,res=0,cnt=1;
		res++;
		}
	o[++len]=res;
	printf("%d\n",len);
	for (int i=1;i<=len;i++) printf("%d%s",o[i],i<len?" ":"\n");
	return 0;
}
