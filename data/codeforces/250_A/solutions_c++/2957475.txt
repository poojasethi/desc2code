#include<iostream>
#include<vector>
using namespace std;

int main() {
	int n;
	cin >> n;
	vector<int> ans;
	int j=0;
	for (int i=0,bad=0,a; i<n; i++,j++) {
		cin >> a;
		if (a<0) bad++;
		if (bad>2) {
			ans.push_back(j);
			j=0;
			bad=1;
		}
	}
	ans.push_back(j);
	cout << ans.size() << '\n';
	for (int i=0; i< ans.size(); i++)
		cout << ans[i] << ' ';
	cin.get();
	cin.get();
}
