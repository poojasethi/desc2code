#include<stdio.h>
int main()
{
  int n,a[100],i,b[100],l=0,left=-1,k=0;
  scanf("%i",&n);
  for(i=0;i<n;i++)scanf("%i",&a[i]);
  for(i=0;i<n;i++)
  {
     if(a[i]<0)k++;
     if(k==3)
     {
        b[l++] = i-left-1;
        k = 1;
        left = i-1;
     }
  }
  b[l++] = i-left-1;
  printf("%i\n",l);
  for(i=0;i<l;i++)printf("%i ",b[i]);
  return 0;
}
