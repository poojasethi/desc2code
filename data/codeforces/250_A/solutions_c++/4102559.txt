#include<stdio.h>
int cd[101]={0};
int ans[101]={0};
int d=0;
int i;
	main(){
int n,a;
int num=0;
scanf("%d",&n);
for(a=1;a<=n;a++)scanf("%d",&cd[a]);
for(a=1;a<=n;a++){
if(cd[a]<0)num++;
if(num==3){ans[++d]=a-1;num=1;}
}
ans[++d]=n;
printf("%d\n",d);
for(a=0;a<d;a++)
{
if(a==0)printf("%d",ans[1]);
else printf(" %d",ans[a+1]-ans[a]);
}
printf("\n");
}
