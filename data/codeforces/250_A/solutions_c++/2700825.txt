#include<iostream>
using namespace std;
int sort[100];
int main()
{
	int n;
	cin>>n;
	int day[100];
	int neg=0;
	int folder=1;
	for(int i=0;i<n;i++)
	{
		cin>>day[i];
		if(day[i]<0)
			neg++;
		if(neg==3)
		{
			folder++;
			neg=1;
		}
		//if(day[i]<0)
		//	neg++;
		sort[folder]++;
	}
	cout<<folder<<endl;
	for(int i=1;i<=folder;i++)
		cout<<sort[i]<<" ";
	return 0;
}
