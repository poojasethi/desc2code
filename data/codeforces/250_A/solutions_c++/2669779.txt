#include<cstdio>
#include<vector>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n; scanf("%d",&n);

	int pre=0,cnt=0;
	vector<int> ans;
	rep(i,n+1){
		int a; scanf("%d",&a);
		if(i==n || a<0){
			cnt++;
			if(i==n || cnt==3) cnt=1, ans.push_back(i-pre), pre=i;
		}
	}

	printf("%d\n",ans.size());
	rep(i,ans.size()) printf("%d ",ans[i]);

	return 0;
}
