#include<stdio.h>
#include<stdlib.h>
#include<string.h>

int a[10001],aa[10001];
int main()
{
	int n,i,sum=0,ans=1;
	scanf("%d",&n);
	for(i=1;i<=n;++i)
		scanf("%d",&a[i]);
	for(i=1;i<=n;++i)
	{
		if(a[i]<0)
			if(++sum>2)
			{
				aa[ans++]=i-1;
				sum=1;
			}
	}
	aa[ans]=n;
	printf("%d\n",ans);
	for(i=1;i<=ans;++i)
		printf("%d ",aa[i]-aa[i-1]);
	return 0;
}
