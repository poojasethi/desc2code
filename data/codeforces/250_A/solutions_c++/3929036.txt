//In the name of God
#include <iostream>
using namespace std;

int b[40], n, cnt, k;

int main() {
	cin >> n;
	while (n--) {
		int t; cin >> t;
		cnt += (t < 0);
		if (cnt == 3)
			b[++k] = cnt = 1;
		else
			b[k]++;
	}
	cout << k + 1 << '\n';
	for (int i = 0; i <= k; i++)
		cout << b[i] << ' ';
	cout << '\n';
	return 0;
}
