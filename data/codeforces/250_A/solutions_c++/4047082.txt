#include<cstdio>
#include<iostream>
using namespace std;

int n;
int A[105];
int main(){
	//sdafasdf
	int i,x,s=0,j=0,y=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>x;
		y++;
		if(x<0) s++;
		if(s>2){
			A[j++]=y-1;
			y=1;
			s=0;
			if(x<0) s++;
		}
	}
	A[j++]=y;
	printf("%d\n",j);
	
	for(i=0;i<j;i++)
		printf("%d ",A[i]);
	printf("\n");
	return 0;
}
