#include<cstdio>
int main()
{
	int n,i,c=0,ans[100]={0},temp,negc=0;
	scanf("%d",&n);
	for (i=0;i<n;i++)
	{
		scanf("%d",&temp);
		if (temp<0)
			negc++;
		if (negc==3)
		{
			ans[c++]=i;
			negc=1;
		}
	}
	ans[c++]=n;
	printf("%d\n",c);
	if (c)
		printf("%d ",ans[0]);
	for (i=1;i<c;i++)
		printf("%d ",ans[i]-ans[i-1]);
	return 0;
}
