#include<stdio.h>
int main(){
	int a[1000],f[1000],i,j,t,n,tot,y;
	while(scanf("%d",&n)!=EOF){
		for(i=1;i<=n;i++)scanf("%d",&a[i]);
		tot=1;t=0;y=1;
		for(i=1;i<=n;i++){
			if(a[i]<0)t++;
			if(t==3){
				f[tot]=i-y;
				y=i;t=1;tot++;
			}
		}
		if(y<=n)f[tot]=n-y+1;
		printf("%d\n",tot);
		for(i=1;i<tot;i++)printf("%d ",f[i]);printf("%d\n",f[tot]);
	}
}
