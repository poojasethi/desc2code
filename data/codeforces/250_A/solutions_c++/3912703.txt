#include <cstdio>
int f[50];
int main(int argc, char const *argv[])
{
	int n, a;
	scanf("%d", &n);
	int r = 0, neg = 0;
	for (int i = 0; i < n; ++i)
	{
		scanf("%d", &a);
		if (a < 0) ++neg;
		if (neg == 3) {
			++r;
			f[r] = neg = 1;
		} else {
			++f[r];
		}
	}
	printf("%d\n", r + 1);
	for (int i = 0; i <= r; ++i)
	{
		printf("%d ", f[i]);
	}
	putchar('\n');
	return 0;
}
