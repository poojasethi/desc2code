#include<iostream>
#include<vector>
using namespace std;
int g,sum;
vector<int> v;
int main(){
	int n;
	cin>>n;
	int a[200];
	for(int i=1;i<=n;i++){
		sum++;
		cin>>a[i];
		if(a[i]<0)g++;
		if(g==3){
			v.push_back(sum-1);
			sum=1;
			g=1;
		}
	}
	if(sum!=0)v.push_back(sum);
	cout<<v.size()<<endl;
	for(int i=0;i<v.size();i++)cout<<v[i]<<" ";
}
