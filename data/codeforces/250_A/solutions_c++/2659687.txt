#include <iostream>
using namespace std;

int n;
int a[101];
int c=0, ch=0;

int main(){
//freopen("INP.txt","r",stdin);
cin>>n;
for(int i=0; i<n; i++) a[i]=0;
int x;
for(int i=0; i<n; i++){
cin>>x;
if(x<0) if(ch==2){
c++; a[c]++; ch=1; continue;
}

a[c]++;
if(x<0) ch++;
}
cout<<c+1<<endl;
for(int i=0; i<=c; i++){
cout<<a[i]<<" ";
}
cout<<endl;
return 0;
}
