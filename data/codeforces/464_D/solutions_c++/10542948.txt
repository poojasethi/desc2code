#include <bits/stdc++.h>

using namespace std;

int n,k,cur;
double f[2][710];

int main()
{
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++,cur^=1)
		for(int j=1;j<=700;j++)
			f[cur^1][j]=1./k/(j+1)*(j*f[cur][j]+f[cur][j+1]+j*(j+3)/2)+1.*(k-1)/k*f[cur][j];
	printf("%.10lf\n",f[cur][1]*k);
	return 0;
}
