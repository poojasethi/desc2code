#include<cstdio>
using namespace std;

#define maxn 100010
double f[maxn];

int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    
    int lim=555;
    
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=lim;j++)
        {
            f[j]=(1/(j+1.0)*(f[j+1]+j)+j/(j+1.0)*(f[j]+(j+1)/2.0))/k+(k-1.0)/k*f[j];
        }
    }
    printf("%.10f\n",f[1]*k);
    return 0;
}
