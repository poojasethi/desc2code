#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
using namespace std;
double dp[1005],ans,Q[1005],QQ[1005],QQQ[10005];
int i,j,n,m;
int main()
{
    scanf("%d%d",&n,&m);
    dp[1]=1;
    for (j=1; j<=550; j++) Q[j]=((j+1)*j/2.0+j)/(j+1.0);
    for (j=1; j<=550; j++) QQ[j]=(1/(m+0.0))*(1/(j+0.0));
    for (j=1; j<=550; j++) QQQ[j]=(1-(1/(m+0.0))*(1/(j+1.0)));
    for (i=1; i<=n; i++)
    {
         for (j=min(i+1,550); j>=1; j--)
         {
           ans+=dp[j]*Q[j];
           dp[j]=dp[j-1]*QQ[j]+dp[j]*QQQ[j];
         }
    }
    printf("%.15f\n",ans);
    return 0;
}
