#define dep 700
using namespace std;

double a[dep+5],b[dep+5];
int n,m;
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=n-1;i>=0;i--)
	{
		memset(b,0,sizeof(b));
		for (int j=1;j<=dep;j++)
		  b[j]=a[j]*(m-1)/m+(a[j]*j/(j+1)+1.0*j/2+a[j+1]/(j+1)+1.0*j/(j+1))/m;
		memcpy(a,b,sizeof(b));
	}
	printf("%.10lf\n",a[1]*m);
	return 0;
}
