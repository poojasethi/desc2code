#include<bits/stdc++.h>
using namespace std;

const int N = 100009;
int sqN = 560;
double dp[2][N]={0.0}, ex = 0.0, sum[N],K2,Kl,f[N]={0.0},f2[N]={0.0};

int main()
{
	sum[0] = 0;
	for(int i=1;i<N;++i)
		sum[i] = sum[i-1] + i;
	int n;
	double K;
	cin >> n >> K;
	Kl = K-1;
	K2 = 1.0 / K;
	dp[0][1] = 1.0;

	for(int j=1;j<=sqN;++j)
		f[j] = Kl+ j*1.0/(j+1)*1.0;
	for(int j=1;j<=sqN;++j)
		f2[j] = (sum[j] + j)/(j+1)*1.0;

	for(int i=1;i<=n;++i){
		int id = i % 2;
		int limit = min( i+1, sqN);

		for(int j=1;j<=limit;++j){
			ex += f2[j] * dp[!id][j]; 
			dp[id][j] = ( dp[!id][j]*f[j] + dp[!id][j-1]/j) * K2;
		}

	}
	printf("%.12lf\n",ex);
	return 0;
}
