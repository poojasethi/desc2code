#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=100005,maxl=705;
int n,K;
double ans;
double f[2][maxn];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j;
	scanf("%d%d",&n,&K);
	for(i=maxl-1;i;i--)
	{
		for(j=1;j<=n;j++)
			f[i&1][j]=0;
		for(j=1;j<=n;j++)
			f[i&1][j]=((i*f[i&1][j-1]+f[!(i&1)][j-1]+(double)i*(i+3)/2))/(i+1)/K+f[i&1][j-1]*(K-1)/K;
	}
	ans=f[1][n]*K;
	printf("%.10lf\n",ans);
	return 0;
}
