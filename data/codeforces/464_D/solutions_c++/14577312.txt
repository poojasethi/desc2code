#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int SZ = 700;

double a[SZ+1];
double b[SZ+1];

int main()
{
    //freopen("input.txt", "r", stdin);

    int n;
    double k;

    cin >> n >> k;

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= SZ; j++) b[j] = a[j];

        for (int j = 1; j <= SZ; j++) {
            a[j] = (k-1) / k * b[j] + 1 / (k * (j+1)) * ( b[j+1] + j * b[j] + j *(j+3)/2.0 );
        }
    }

    double ans = a[1] * k;

    cout << setprecision(15) << fixed << ans;
}
