#include <stdio.h>
double d[2][701];
int main(){
	int n,k,i,j,togle=0;
	scanf("%d %d",&n,&k);
	for(i=0;i<n;i++){
		for(j=1;j<=700;j++){
			d[!togle][j]=1.0/k/(j+1)*(j*d[togle][j]+d[togle][j+1]+j*(j+3)/2)+1.0*(k-1)/k*d[togle][j];
		}
		togle=!togle;
	}
	printf("%.11lf\n",d[togle][1]*k);
	return 0;
}
