#define Inf 1e9
#define M 750
#define Mod 1000000007
using namespace std;
int main()
{
    int i,j,n;
    double k,t,ans,dp[M];
    cin>>n>>k;
    for(i=0;i<M;i++) dp[i]=(double) 0;
    for(i=0;i<n;i++)
    {
        for(j=0;j<M-1;j++)
        {
            t=j+1;
            dp[j]=(((k-1)/k)+(t/((t+1)*k)))*dp[j]+(1/(((t+1)*k)))*dp[j+1]+(t*(t+3))/(2*(t+1)*k);
        }
    }
    ans=dp[0]*k;
    printf("%.12f",ans);
    return 0;
}
