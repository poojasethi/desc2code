#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cstdlib>
using namespace std;
int n,k;
double dp[2][705];
int main()
{
	scanf("%d%d",&n,&k);
	int u=min(700,n);
	for(int i=1;i<=n;++i)
	  for(int j=1;j<=u;++j)
	    dp[i&1][j]=(1.0/k)*((1.0*j/(j+1))*(dp[i&1^1][j]+(j+1)*1.0/2)+(1.0/(j+1))*(dp[i&1^1][j+1]+j))+(1.0*(k-1)/k)*dp[i&1^1][j];
    printf("%.10lf\n",dp[n&1][1]*k);
    return 0;
}
