#include <bits/stdc++.h>

#define MAX 100005
#define MAXL 600

using namespace std;

int n,k;
double pd[2][MAXL];

int main(){

    scanf("%d %d",&n,&k);

    for(int i=1;i<=n;i++){
        for(int j=MAXL-1;j>=0;j--){
            double aux1=j*pd[ (i&1)^1 ][j]+pd[(i&1)^1][j+1]+1.0*j*(j+3)/2;
            pd[i&1][j]=(aux1/(k*(j+1)) + 1.0*(k-1)/k*pd[(i&1)^1][j]);
        }
    }

    printf("%.10f\n",k*pd[n&1][1]);

    return 0;
}
