#include<bits/stdc++.h>
using namespace std;
double ff[800][6];
double k;
int main()
{
    int n,i,j;
    scanf("%d %lf",&n,&k);
    for(i=0;i<n;i++) for(j=1;j<=700;j++) {
    ff[j-1][(i+1)&1]+=1/k*1/(j)*(ff[j][i&1]+j-1);
    ff[j][(i+1)&1]=(ff[j][(i)&1]*j+(j+1)*j/2)*1/k*1/(j+1)+(k-1)/k*ff[j][i&1];
    }
//    ff[lv][turn]=1/k*((f(lv+1,turn-1)+lv)*1/(lv+1)+(lv*f(lv,turn-1)+(lv+1)*lv/2)*1/(lv+1))+(k-1)/k*f(lv,turn-1);
    printf("%.9lf",ff[1][n&1]*k);
}
