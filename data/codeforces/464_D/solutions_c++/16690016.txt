#include<iostream>
#include<stdio.h>
#include<vector>
#include<cmath>
using namespace std;
double f[1002][2];
int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    int can=1;
    for (int j=1;j<=n;j++)
    {
        can=1-can;
        for (int i=1;i<=1000;i++)
        {

            f[i][can]=((1.0*i+f[i][1-can]*i+(f[i+1][1-can]))/(i+1)  +0.5*i )/k+f[i][1-can]*(k-1)/k;
        }
    }
    printf("%.12lf\n",f[1][can]*k)
;

}
