#include<algorithm>
#include<cstdio>
using namespace std;

const int maxn=100010;

double f[2][1010];
int n,k;

int main(){
	scanf("%d%d",&n,&k);
	int LIM=min(n,1000);
	int p=0;
	for(int i=1;i<=n;++i){
		p^=1;
		for(int j=1;j<=LIM;++j){
			f[p][j]=1.0/k*(1.0/(j+1)*(f[p^1][j+1]+j)+1.0*j/(j+1)*(f[p^1][j]+(j+1)/2.0))+(1.0-1.0/k)*f[p^1][j];
		}
	}
	printf("%.10f\n",f[p][1]*k);
//	for(;;);
	return 0;
}
