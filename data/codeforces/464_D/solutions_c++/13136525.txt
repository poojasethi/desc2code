#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
#include <algorithm>
using namespace std;
#define N 2050
#define ld long double
ld f[N],ans,inv[N],m;int n;
int main()
 {
    cin >>n>>m;
    for (int i=1;i<N;i++)
      inv[i]=1.0/i;
    f[1]=1.0;ld q=1/m,eps=1e-100;
    while (n--)
     for (int i=N-1;i>=1;i--)
      if (f[i]<eps) f[i]=0;else
       {
          ans+=f[i]*i*(0.5+inv[i+1]);
          ld j=f[i]*q*inv[i+1];
          f[i+1]+=j;f[i]-=j;
       }
    printf("%.10lf\n",(double)(ans));
    return 0;
 }
