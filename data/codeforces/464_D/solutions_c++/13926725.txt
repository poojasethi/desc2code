#include<cstdio>
typedef double LD;
#define LIM 700
LD f[2][LIM+20],n,k;
int main()
{
    scanf("%lf%lf",&n,&k);
    for(int i(1);i<=n;i++)
    {
        for(int j(1);j<=LIM;j++)
            f[1][j]=1/(k*(j+1))*(j*f[0][j]+f[0][j+1]+j*(j+3)/2)+(k-1)/k*f[0][j];
        for(int j(1);j<=LIM;j++)
            f[0][j]=f[1][j],f[1][j]=0;
    }
    printf("%.12lf\n",f[0][1]*k);
    return 0;
}
