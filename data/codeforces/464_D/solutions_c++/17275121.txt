#include <bits/stdc++.h>
using namespace std;
const int maxn = 1e5 + 10, len = 800;
int n, k;
long double dp[maxn+10];

int main() {
    //freopen("input.txt", "r", stdin);
    cin >> n >> k;
    for(int i = 1; i <= n; ++i)
    for(int j = 1; j <= min(len, n); ++j)
        dp[j] = dp[j] * (k-1) / k + (dp[j+1] + j) / ((j+1) * k) + (dp[j]*j + j*(j+1)/2) / ((j+1) * k);
    cout << setprecision(10) << fixed << dp[1]*k;
}
