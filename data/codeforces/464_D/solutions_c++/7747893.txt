#include <cstdio>
#include <algorithm>

using namespace std;

const int MAXLV = 600;

int n;
double k;
double dp[2][MAXLV + 10];

int main() {
    scanf("%d %lf", &n, &k);
    int sw = 0;
    for (int j = n; j >= 1; --j) {
        sw ^= 1;
        for (int i = MAXLV; i >= 1; --i)
            dp[sw][i] = dp[sw ^ 1][i] * (k - 1) / k + (dp[sw ^ 1][i] * i + dp[sw ^ 1][i + 1] + i) / k / (i + 1) + (double)i / 2 / k;
    }
    printf("%.12lf\n", dp[sw][1] * k);
}
