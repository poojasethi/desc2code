#include<bits/stdc++.h>
#define MAX   200200
#define FOR(i,a,b) for (int i=(a);i<=(b);i=i+1)
#define REP(i,n) for (int i=0;i<(n);i=i+1)
#define fi   first
#define se   second
using namespace std;
typedef long long ll;
const ll INF=(ll)1e18+7LL;
typedef pair<ll,int> ii;
int n,k;
ll s[MAX];
int c[MAX];
ii a[MAX];
void init(void) {
    cin >> n >> k;
    k=n-k+1;
    FOR(i,1,n) cin >> a[i].se >> a[i].fi;
    sort(a+1,a+n+1);
    FOR(i,1,n) c[a[i].se]++;
    FOR(i,1,MAX-10) c[i]+=c[i-1];
    FOR(i,1,n) s[i]=s[i-1]+a[i].fi;
}
ll choose(int x) {
    int ngr;
    if (x<2) ngr=0; else ngr=c[x-2];
    if (ngr>=k) return (s[x]);
    if (c[x]<k) return (INF);
    int y=k-ngr;
    vector<ll> vx,vy;
    FOR(i,1,n) if (a[i].se==x || a[i].se==x-1) vy.push_back(a[i].fi); else vx.push_back(a[i].fi);
    ll ret=0LL;
    sort(vy.begin(),vy.end());
    REP(i,y) ret+=vy[i];
    FOR(i,y,vy.size()-1) vx.push_back(vy[i]);
    sort(vx.begin(),vx.end());
    REP(i,x-y) ret+=vx[i];
    return (ret);
}
void process(void) {
    ll res=INF;
    REP(x,n+1) res=min(res,choose(x));
    if (res>=INF) cout << -1; else cout << res;
}
int main(void) {
    init();
    process();
    return 0;
}
