#include <iostream>
#include <iomanip>
using namespace std;

struct pt{
    int x, y;
} arr[305];

int n;

int cross(pt a, pt b, pt c){
    return (a.x-b.x)*(c.y-b.y)-(a.y-b.y)*(c.x-b.x);
}

int main(){

    int res = 0;
    cin >> n;
    for(int i = 0; i<n; i++) cin >> arr[i].x >> arr[i].y;
    for(int i = 0; i<n; i++){
        for(int j = 0; j<n; j++){
            int hi = 0, lo = 0;
            for(int k = 0; k<n; k++){
                hi = max(hi, cross(arr[i], arr[j], arr[k]));
                lo = min(lo, cross(arr[i], arr[j], arr[k]));
            }
            if(hi>0&&lo<0) res = max(res, hi-lo);
        }
    }

    cout << setprecision(16) << fixed << res/2.0 << endl;

}
