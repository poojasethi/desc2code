#include<stdio.h>
int x[301],y[301];
int cross(int ax,int ay,int bx,int by)
{
    return ax*by-ay*bx;
}
int main()
{
    int n;
    int max=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d %d",&x[i],&y[i]);
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<i;j++)
        {
            int mp=-1,mm=-1;
            for(int k=0;k<n;k++)
            {
                if(k==i || k==j) continue;
                int a=cross(x[j]-x[i],y[j]-y[i],x[k]-x[i],y[k]-y[i]);
                if(a<0 && mm<=-a) mm=-a;
                if(a>0 && mp<=a) mp=a;
            }
            if(mp>=0 && mm>=0 && max<mm+mp) max=mm+mp;
        }
    }
    printf("%lf\n",max/2.0+1e-10);
    return 0;
}
