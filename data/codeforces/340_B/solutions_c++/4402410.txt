#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;

long long n ,x[100010] ,y[100010] ; double m , M , area , ans=0.0 ; 

int main()
{
     cin >> n ;
     for(int i=0 ; i<n ; i++) cin >> x[i] >> y[i] ;   
     
     for(int i=0 ; i<n ; i++) 
      for(int j=0 ; j<n ; j++) if( i!=j )
      {
         m=11278970.0 , M=0.0 ;
         for(int k=0; k<n ; k++) if( k!=i && k!=j )
         {
             area = x[i]*(y[j]-y[k]) - y[i]*(x[j]-x[k]) + (x[j]*y[k]-x[k]*y[j]) ;
             m = min(m,area) , M = max(M,area) ;
         }
         if( ans < M-m && M>0 ) ans = M-m ; 
      }
        
    printf("%.6lf\n",ans/2.0); 
  return 0;
}
