#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;

double x[305],y[305];

double cross(double x1,double y1,double x2,double y2)
{
    return x1*y2-x2*y1;
}

int main()
{
    int n;cin>>n;
    for(int i=0;i<n;++i) cin>>x[i]>>y[i];
    double s=0;
    for(int i=0;i<n;++i)
    {
        for(int j=i+1;j<n;++j)
        {
            double u=0,v=0,t;
            for(int k=0;k<n;++k)
            {
                t=cross(x[k]-x[i],y[k]-y[i],x[k]-x[j],y[k]-y[j]);
                if(t>0) u=max(u,t);
                else v=max(v,-t);
            }
            if(u&&v) s=max(s,u+v);
        }
    }
    printf("%.6lf\n",s/2);
    return 0;
}
