#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <math.h>
using namespace std;
struct node
{
	double x,y;
}p[400];
int main(int argc, char const *argv[])
{
	int n;
	scanf("%d",&n);
	for (int i=0;i<n;i++) scanf("%lf%lf",&p[i].x,&p[i].y);
	double ans=0;
	for (int i=0;i<n;i++)
		for (int j=i+1;j<n;j++){
			double MIN=99999999999,MAX=-99999999999;
			for (int k=0;k<n;k++)
				if (k!=i && k!=j){
				double d=0.5*(p[i].x*p[j].y+p[j].x*p[k].y+p[k].x*p[i].y-p[i].x*p[k].y-p[j].x*p[i].y-p[k].x*p[j].y);
				MIN=min(d,MIN);
				MAX=max(d,MAX);
			}
			if (MIN<0 && MAX>0) ans=max(ans,MAX-MIN);
		}
	printf("%.12f\n",ans );
	return 0;
}
