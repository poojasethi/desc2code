#include <cstdio>
#include <algorithm>
using namespace std;

int n, x[300], y[300];
int ans;

int main() {
  scanf("%d", &n);
  for(int i = 0; i < n; i++) {
    scanf("%d %d", x+i, y+i);
  }
  for(int i = 0; i < n; i++) {
    for(int j = 0; j < n; j++) {
      int l = 0, r = 0;
      for(int k = 0; k < n; k++) {
    int cross = (x[j]-x[i])*(y[k]-y[j]) - (y[j]-y[i])*(x[k]-x[j]);
    l = min(l, cross);
    r = max(r, cross);
      }
      if(l < 0 and r > 0) {
    ans = max(ans, r-l);
      }
    }
  }
  printf("%lf\n", ans/2.0);
}
