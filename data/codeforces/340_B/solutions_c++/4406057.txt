#include<iostream>
#include<stdio.h>
using namespace std;
int x[300];
int y[300];
double Area(int a,int b,int c);
double Area(int a,int b,int c){
 double ans;
 ans= (x[b]-x[a])*1.0*(y[b]-y[c])+(x[c]-x[b])*(y[b]-y[a]);
 return ans;
}
int main(){
 int n;
 double ans=0;
 cin>>n;
 for(int i=0;i<n;i++){
  cin>>x[i]>>y[i];
 }
 for(int i=0;i<n;i++){
  for(int j=i+1;j<n;j++){
   double s1=0; 
   double s2=0;
   for(int z=0;z<n;z++){
    if(z!=i&&z!=j){
     double s=Area(i,j,z);
     if(s>0&&s>s1)
	s1=s;
     else{
	if(s<0&&s<s2) s2=s;
	}
     }
    }
   if(s1-s2>ans&&s1*s2<0) ans=s1-s2;
  }
 }
 printf("%.6lf\n",ans/2.0);
}
