#include"stdio.h"

int x[305],y[305];

double cal(int x1,int y1,int x2,int y2)
{
	return (x1*y2 - y1*x2) * 0.5;
}

double ab(double a)
{
	return a<0?-a:a;
}


int main( )
{
	int n,i,j,k;
	double max=0,min=0,tmp,ans=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d%d",&x[i],&y[i]);
	for(i=0;i<n;i++)
		for(j=i+1;j<n;j++)
		{
			max=min=0;
			for(k=0;k<n;k++)
			{
				if(k==i||k==j)
				  continue;
				tmp = cal(x[i]-x[j],y[i]-y[j],x[j]-x[k],y[j]-y[k]);
				max = max>tmp?max:tmp;
				min = min<tmp?min:tmp;
				if(max!=0&&min!=0)
					ans = ans>(max-min)?ans:(max-min);
			}
		}
	printf("%f\n",ans);
	return 0;
}
