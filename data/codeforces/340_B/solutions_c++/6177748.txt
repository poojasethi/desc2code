#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#include<math.h>
#include<string.h>
#include<time.h>
using namespace std;
int x[300],y[300];
double s[300];
int main(){
    int n,i,j,k;
    double res=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
      scanf("%d%d",x+i,y+i);
    for(i=0;i<n;i++){
      for(j=i+1;j<n;j++){
        int N=0;
        for(k=0;k<n;k++){
          if(k==i)continue;
          if(k==j)continue;
          s[N++]=(x[k]-x[i])*(y[j]-y[i])-(x[j]-x[i])*(y[k]-y[i]);
        }
        sort(s,s+n-2);
        res=max(res,s[n-3]-s[0]);
      }
    }
    printf("%.12f\n",res/2);
}
