#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define db double
#define N 310
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	db a,b;
} a[N];
int i,j,k,n,m;
db an;
inline db cross(int i,int j,int k) {
	return (a[j].a-a[i].a)*(a[k].b-a[i].b)-(a[j].b-a[i].b)*(a[k].a-a[i].a);
}
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%lf%lf",&a[i].a,&a[i].b);
	For(i,1,n)For(j,i+1,n) {
		db mi,ma; mi=ma=0;
		For(k,1,n) {
			db A=cross(i,j,k);
			mi=min(mi,A),ma=max(ma,A);
		}
		if (!mi) mi=1e15;
		if (!ma) ma=-1e15;
		an=max(an,ma-mi);
	}
	printf("%.12lf\n",an/2);
	//for(;;);
	return 0;
}
