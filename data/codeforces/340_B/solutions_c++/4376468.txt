#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const int maxn=3e2+9;
int x[maxn],y[maxn];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    scanf("%d %d",&x[i],&y[i]);

    double ans=0;
    for(int i=1;i<=n;i++)
    for(int j=i+1;j<=n;j++)
    {
        double lmax=0,rmax=0,ret;
        for(int k=1;k<=n;k++)
        if(k!=i&&k!=j)
        {
            ret=(x[j]-x[i])*(y[k]-y[i])-(y[j]-y[i])*(x[k]-x[i]);
            ret/=2;
            if(ret<0)
            lmax=max(lmax,-ret);
            else
            rmax=max(rmax,ret);
        }
        if(lmax>0&&rmax>0)
        ans=max(ans,lmax+rmax);
    }
    printf("%f\n",ans);

    return 0;
}
