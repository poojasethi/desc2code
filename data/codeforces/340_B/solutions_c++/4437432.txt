#include<stdio.h>

int a[305][2];
int  area(int p,int q,int r)
{
    return a[p][0]*a[q][1]-a[p][1]*a[q][0]+a[q][0]*a[r][1]-a[q][1]*a[r][0]+a[r][0]*a[p][1]-a[r][1]*a[p][0];
}
int main()
{
    int n,i,j,k,dap=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d%d",&a[i][0],&a[i][1]);
    }
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            int max=-2000000000;
            int min=2000000000;

            for(k=0;k<n;k++)
            {
                if(i==k || j==k)continue;
                int kk=area(i,j,k);
                if(kk>max)max=kk;
                if(kk<min)min=kk;
            }
            if(dap<max-min)dap=max-min;
        }
    }
    printf("%f",dap/2.0);

}
