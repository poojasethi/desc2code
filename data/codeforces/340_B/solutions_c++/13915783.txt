#include<iostream>
#include<cstdio>
#include<string>
#include<cstring>
#include<algorithm>
#include<set>
#include<cmath>
using namespace std;

int n,x[310],y[310];

int cross(int x1,int y1,int x2,int y2)
{
	return x1*y2-x2*y1;
}

int main()
{
	int i,j,k,maxn,minn,ans = 0;
	cin >> n;
	for(i = 1; i<=n; i++) scanf("%d%d",&x[i],&y[i]);
	for(i = 1; i<=n; i++)
		for(j = i+1; j<=n; j++) {
			for(maxn = -(int)1e9,minn = (int)1e9,k = 1; k<=n; k++) if(k!=i && k!=j) {
				int s = cross(x[i]-x[k], y[i]-y[k], x[j]-x[k], y[j]-y[k]);
				maxn = max(maxn, s);
				minn = min(minn, s);
			}
			ans = max(ans, maxn-minn);
		}
	printf("%lf", (double)ans/2);
	return 0;
}
