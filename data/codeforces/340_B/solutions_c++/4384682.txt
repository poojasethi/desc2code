#include <stdio.h>

struct vect{
	int x,y;
} P[1000];

vect operator - (vect A, vect B){
	vect aux;
	aux.x=A.x-B.x;
	aux.y=A.y-B.y;
	return aux;
}

double cruz(vect A, vect B){
	return double(A.x)*B.y-double(A.y)*B.x;
}

int main(){
	int N;
	vect b,h;
	scanf("%d",&N);
	for(int i=0; i<N; i++)
		scanf("%d %d",&P[i].x,&P[i].y);
	double Mp,Mn,MA,val;
	
	MA=0;
	for(int i=0; i<N; i++)
		for(int j=i+1; j<N; j++){
			b=P[j]-P[i];
			Mn=Mp=0;
			Mn=1; Mp=-1;
			for(int k=0; k<N; k++)
			if(k!=i && k!=j){
				h=P[k]-P[i];
				val=cruz(b,h);
				if(val<Mn) Mn=val;
				if(val>Mp) Mp=val;
			}
			val=Mp-Mn;
			if(Mn>0 || Mp<0) val=0;
			if(val>MA) MA=val;
		}
	
	printf("%.9lf\n",MA/2);
	return 0;
}
