#include <cstdio>
#include <cmath>

void mx(double &x, double y) {
  if (y > x)
    x = y;
}

int main() {
  int n;
  double x[300];
  double y[300];
  scanf("%d", &n);
  for (int i = 0; i < n; ++i)
    scanf("%lf%lf", x + i, y + i);
  double ans = 0;
  for (int i = 1; i < n; ++i)
    for (int j = 0; j < i; ++j) {
      double m[2] = {0, 0};
      for (int k = 0; k < n; ++k)
        if (k != i && k != j) {
          double p = (x[k] - x[i]) * (y[j] - y[i]) - (y[k] - y[i]) * (x[j] - x[i]);
          mx(m[p > 0], fabs(p) * 0.5);
        }
      if (m[0] > 0 && m[1] > 0)
        mx(ans, m[0] + m[1]);
    }
  printf("%f\n", ans);
}
