#include <bits/stdc++.h>

using namespace std;
int x[301], y[301];
int mn=1e9, mx=-1e9;
void cross(int i, int j, int k){
    int c=(x[j]-x[i])*(y[k]-y[i])-(x[k]-x[i])*(y[j]-y[i]);
    if(c>0)mx=max(mx, c);
    else if(c<0)mn=min(mn, c);
}
int main()
{
    int n;
    double area=0;
    cin>>n;
    for(int i=0;i<n;i++)
        scanf("%d%d", &x[i], &y[i]);
    for(int i=0;i<n;i++){
        for(int j=i+1;j<n;j++){
            for(int k=0;k<n;k++){
                if(k==i||k==j)continue;
                cross(i, j, k);
            }
            if(mx!=-1e9&&mn!=1e9)area = max(area, (mx-mn+0.0)/2);
            mn=1e9, mx=-1e9;
        }
    }
    printf("%.9lf", area);
    return 0;
}
