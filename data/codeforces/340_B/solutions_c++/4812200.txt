#include <cstdio>
#include <algorithm>
using namespace std;

struct point
{
	int x,y;
	point(int _x=0,int _y=0) {x=_x,y=_y;}
	friend int operator *(const point&A,const point&B) {return A.x*B.y-A.y*B.x;}
	friend point operator -(const point&A,const point&B) {return point(A.x-B.x,A.y-B.y);}
} a[305];
int N;

int main()
{
	scanf("%d",&N);
	int ans=0;
	for (int i=1; i<=N; i++) scanf("%d%d",&a[i].x,&a[i].y);
	for (int i=1; i<=N; i++)
		for (int j=i+1; j<=N; j++)
		{
			int x=-1e9,y=-1e9;
			for (int k=1,s; k<=N; k++) if (k!=i&&k!=j) if ((s=(a[k]-a[i])*(a[j]-a[i]))>0) x=max(x,s); else y=max(y,-s);
			ans=max(ans,x+y);
		}
	printf("%.9lf\n",ans/2.0);
}
