#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
using namespace std;
double x[310],y[310];

int main()
{
    int i,j,k,n;
    double ans=0;
    double tt;
    double l,r;
    scanf("%d",&n);
    for (i=0;i<n;i++) scanf("%lf%lf",&x[i],&y[i]);
    for (i=0;i<n;i++)
    {
        for (j=i+1;j<n;j++)
        {
            l=r=0;
            for (k=0;k<n;k++)
            {
                if (k==i||k==j) continue;
                tt=(x[i]-x[j])*(y[i]-y[k])-(x[i]-x[k])*(y[i]-y[j]);
                l=min(l,tt);
                r=max(r,tt);
            }
            if (l<0&&r>0)
                ans=max(ans,r-l);
        }
    }
    printf("%.6f\n",ans/2);
    return 0;
}
