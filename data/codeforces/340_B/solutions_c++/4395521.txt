#include <cstdio>
#include <complex>
#include <algorithm>
using namespace std;
typedef complex<int> V;
const int inf=1000010000;
V v[330];
int crs(V a,V b){return (a.real()*b.imag()-a.imag()*b.real());}
main(){
  int n,s=0;
  scanf("%d",&n);
  for(int i=0,x,y;i<n&&2==scanf("%d %d",&x,&y);++i)v[i]=V(x,y);
  for(int i=0;i<n;++i)for(int j=i+1;j<n;++j){
    int a=inf,b=-inf;
    for(int k=0;k<n;++k){
      int c=crs(v[k]-v[i],v[j]-v[i]);
      if(c<0)a=min(a,c);else if(c>0)b=max(b,c);
      if(a<0&&b>0)s=max(s,b-a);
    }
  }
  printf("%.8lf\n",s/2.0);
}
