#include <algorithm>
#include <iostream>
using namespace std;
#define U(x) (x>0?1:0)
int n,r[2],x[300],y[300],i,j,k,t,a;
int main(){
  cin>>n; for(;t<n;t++)cin>>x[t]>>y[t];
  for(;i<n-1;i++) for(j=i+1;j<n;j++) for (k=0;k<n;k++){
    k==0?(r[0]=r[1]=0):0;
    if (j==k||i==k)continue;
    t=(x[j]-x[i])*(y[k]-y[i])-(y[j]-y[i])*(x[k]-x[i]);
    r[U(t)]=(r[U(t)]<abs(t)?abs(t):r[U(t)]);
    a=(a<r[0]+r[1]&&r[0]*r[1]!=0?r[0]+r[1]:a);
  }
  cout <<a/2<<(a%2==1?".5":"");    
  return 0;
}
