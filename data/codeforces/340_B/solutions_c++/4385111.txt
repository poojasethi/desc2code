#include<stdio.h>
#include<algorithm>
using namespace std;
typedef double dou;
typedef struct{dou x,y;}dot;
dot operator-(dot a,dot b){return(dot){a.x-b.x,a.y-b.y};}
dou operator*(dot a,dot b){return a.x*b.y-a.y*b.x;}
dot d[514];
int n;
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%lf %lf",&d[i].x,&d[i].y);
	dou ans=-1;
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			dou m1=-12345678;
			dou m2=-12345678;
			for(int k=0;k<n;k++){
				if(k==i||k==j)
					continue;
				dou tmp=(d[j]-d[i])*(d[k]-d[i]);
				if(tmp<0)
					m1=max(m1,-tmp);
				else
					m2=max(m2,tmp);
				ans=max(ans,m1+m2);
				}
			}
		}
	printf("%.10f\n",ans/2.0);
	return 0;
	}
