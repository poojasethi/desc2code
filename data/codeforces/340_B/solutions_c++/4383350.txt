#include<cstdio>
#include<algorithm>
using namespace std;
const int maxn=305;
int x[maxn],y[maxn];
int main()
{
    int n,ans=0,s,l,r;
    scanf("%d",&n);
    for(int i=0; i<n; i++)
        scanf("%d%d",&x[i],&y[i]);
    for(int i=0; i<n; i++)
        for(int j=i+1; j<n; j++)
        {
            l=r=0;
            for(int k=0; k<n; k++)
                if(k!=i&&k!=j)
                {
                    s=(x[j]-x[i])*(y[k]-y[i])-(y[j]-y[i])*(x[k]-x[i]);
                    if(s<0)
                        l=max(l,-s);
                    else
                        r=max(r,s);
                }
            if(l&&r)
                ans=max(ans,l+r);
        }
    printf("%f",ans/2.0);
}
