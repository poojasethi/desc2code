#include <cstdio>
int in[300][2];
int main(){
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d%d",in[i],in[i]+1);
	}
	int s=0;
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			int mas=0x80000000,mis=0x7FFFFFFF;
			for(int k=0;k<n;k++){
				if(k!=i&&k!=j){
					int s=(in[j][0]-in[i][0])*(in[k][1]-in[i][1])-(in[k][0]-in[i][0])*(in[j][1]-in[i][1]);
					if(s>mas){
						mas=s;
					}
					if(s<mis){
						mis=s;
					}
				}
			}
			if(mas-mis>s){
				s=mas-mis;
			}
		}
	}
	printf("%f",(float)s/2);
	return 0;
}
