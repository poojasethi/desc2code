#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;

struct point{ int x,y; };

int main(){
 int n,i,j,k,val,mini,maxi,ans=0;
 cin>>n;
 vector <point> input(n);
 for(i=0;i<n;i++) cin>>input[i].x>>input[i].y;

 for(i=0;i<n;i++)
 for(j=0;j<n;j++)
 if(i!=j){
    maxi=mini=0;
    for(k=0;k<n;k++){
        if(k!=i && k!=j){
            val=(input[j].x-input[i].x)*(input[k].y-input[i].y)-(input[j].y-input[i].y)*(input[k].x-input[i].x);
            mini=min(mini,val);  maxi=max(maxi,val);
        }
        if(maxi && mini) ans=max(ans,maxi-mini);
    }
 }
 printf("%.6f",ans/2.0);
 return 0;
}
