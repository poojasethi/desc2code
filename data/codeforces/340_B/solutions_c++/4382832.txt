#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=302;
int x[maxn],y[maxn];
int n,ans;

int main(){
  int i,j,k,s,l,r;
  scanf("%d",&n);
  for (i=0;i<n;++i) scanf("%d%d",&x[i],&y[i]);
  for (i=0;i<n;++i)
    for (j=i+1;j<n;++j){
      l=r=0;
      for (k=0;k<n;++k)
        if (k!=i && k!=j){
          s=(x[j]-x[i])*(y[k]-y[i])-(y[j]-y[i])*(x[k]-x[i]);
          if (s<0) l=max(l,-s);
          else r=max(r,s);
        }
      if (l && r) ans=max(ans,l+r);
    }
  printf("%f\n",ans/2.0);
  return 0;
}
