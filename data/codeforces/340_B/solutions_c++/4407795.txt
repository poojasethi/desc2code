#include<iostream>
#include<algorithm>
using namespace std;
const int N=310;
int x[N], y[N];
int main()
{
	int n, i, j, k, ans=0;
	cin>>n;
	for (i=0; i<n; i++)
		cin>>x[i]>>y[i];
	for (i=0; i<n; i++)
		for (j=i+1; j<n; j++)
		{
			int maxA=-100000000, minA=100000000;
			for (k=0; k<n; k++)
				if (k!=i && k!=j)
				{
					int tmp=(y[i]-y[k])*(x[j]-x[k])-(y[j]-y[k])*(x[i]-x[k]);
					maxA=max(maxA,tmp);
					minA=min(minA,tmp);
				}
			ans=max(abs(maxA-minA),ans);
		}
	cout<<ans/2;
	if (ans &1)
		cout<<".5";
	cout<<endl;
}
