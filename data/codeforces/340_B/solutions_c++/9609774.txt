#include <iostream>
#include <cstdio>
using namespace std;

int n, x[301], y[301];
double area = 0;

int main () {
    //freopen("test01.in", "r", stdin);
    //freopen("", "w", stdout);
    scanf("%d", &n);
    for (int i = 1; i <= n; i++)
        scanf("%d%d", &x[i], &y[i]);

    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= n; j++) {
            double l = 0, r = 0;
            for (int k = 1; k <= n; k++) {
                double s = ((x[j]-x[i])*(y[k]-y[i]) - (y[j]-y[i])*(x[k]-x[i])) * 0.5;
                l = min(l, s);
                r = max(r, s);
            }
            if (l < 0 && r > 0) area = max(area, r-l);
        }
    printf("%.9lf\n", area);
    return 0;
}
