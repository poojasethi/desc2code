#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

const int maxn=310;
const int INF=0x3f3f3f3f;

int n,x[maxn],y[maxn];

int cross(int a,int b,int c)
{
	return (x[c]-x[b])*(y[b]-y[a])-(x[b]-x[a])*(y[c]-y[b]);
}

int main()
{
	scanf("%d",&n);
	for (int a=1;a<=n;a++)
		scanf("%d%d",&x[a],&y[a]);
	int ans=0;
	for (int a=1;a<=n;a++)
		for (int b=a+1;b<=n;b++)
		{
			int maxv=-INF,minv=INF;
			for (int c=1;c<=n;c++)
				if (c!=a && c!=b)
				{
					int v=cross(a,b,c);
					maxv=max(maxv,v);
					minv=min(minv,v);
				}
			if (minv<=0 && maxv>=0) ans=max(ans,maxv-minv);
		}
	printf("%lf\n",(double)ans/2.0);

	return 0;
}
