#include<bits/stdc++.h>
using namespace std;

int x[300], y[300], A[2];

int main()
{
    int n, i, j, k, tmp, maxA = 0;
    scanf("%d", &n);
    for (i = 0; i < n; i++) scanf("%d%d", &x[i], &y[i]);
    for (i = 0; i < n - 1; i++)
        for (j = i + 1; j < n; j++)
        {
            A[0] = A[1] = 0;
            for (k = 0; k < n; k++) 
            {
                if (k == i || k == j) continue;
                tmp = (x[j] - x[i]) * (y[k] - y[i]) - (y[j] - y[i]) * (x[k] - x[i]);
                A[tmp > 0] = max(A[tmp > 0] , abs(tmp));
                if (A[0] && A[1]) maxA = max(maxA, A[0] + A[1]);
            }
        }
    printf("%d", maxA >> 1);
    if (maxA & 1) printf(".5");
    return 0;
}
