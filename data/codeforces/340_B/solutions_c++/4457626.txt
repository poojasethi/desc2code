#include<stdio.h>
#define INF -10000000000
struct
{
	double x,y;
}v[309];
double area(int i,int j,int k){  
    return ((v[k].x-v[i].x)*(v[k].y-v[j].y)-(v[k].y-v[i].y)*(v[k].x-v[j].x))/2.0;  
} 
int main()
{
	//freopen("two2.in","r",stdin);
	int n,i,j,k;
	double temp,max=0,max1,max2;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%lf%lf",&v[i].x,&v[i].y);
	}
	for(i=0;i<n;i++)
	for(j=0;j<n;j++)
	{
		if(i==j)
		continue;
		max1=INF;max2=INF;
		for(k=0;k<n;k++)
		{
			if(i==k||j==k)
			continue;
			temp=area(i,j,k);
			if(temp>0)
				max1=(temp>max1)?temp:max1;
			else
				max2=(temp+max2<0)?-temp:max2;
		}
		max=(max>max1+max2)?max:max1+max2;
	}
	printf("%.6f\n",max);
	return 0;
}
