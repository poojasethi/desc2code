#include<stdio.h>
int x[305],y[305];
int main()
{
    int i,j,k,N;
    double ans=0;
    scanf("%d",&N);
    for(i=1;i<=N;i++) scanf("%d%d",&x[i],&y[i]);
    for(i=1;i<=N-1;i++)
        for(j=i+1;j<=N;j++)
        {
            int ax=x[j]-x[i];
            int ay=y[j]-y[i];
            double Max=0,Min=0;
            for(k=1;k<=N;k++)
            {
                int bx=x[k]-x[i];
                int by=y[k]-y[i];
                double S=(double)(ax*by-ay*bx)/2.0;
                if(S>Max) Max=S;
                if(S<Min) Min=S;
            }
            if(Max*Min<0 && ans<Max-Min) ans=Max-Min;
        }
    printf("%f\n",ans);
}
