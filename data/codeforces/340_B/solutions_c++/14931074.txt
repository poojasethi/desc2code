#include <bits/stdc++.h>
#define ii pair<int,int>
using namespace std;
double area=0,pos_loc=0,neg_loc=0;
void ccw(ii a,ii b,ii c){
	double d=(double)((b.first-a.first)*(c.second-b.second)*1.0-(b.second-a.second)*(c.first-b.first)*1.0)*0.5;
	if(d<0&&neg_loc<abs(d))neg_loc=abs(d);
	else if(d>0&&pos_loc<abs(d))pos_loc=abs(d);
}
int main(){
	int n;
	pair<int,int>p[310];
	cin>>n;
	for(int i=0;i<n;i++)cin>>p[i].first>>p[i].second;
	for(int i=0;i<n-1;i++){
		for(int j=i+1;j<n;j++){
			pos_loc=neg_loc=0;
			for(int k=0;k<n;k++){
				if(k!=i&&k!=j){
					ccw(p[i],p[j],p[k]);
				}
			}
			if(pos_loc&&neg_loc){
				if(pos_loc + neg_loc>area)area=pos_loc + neg_loc;
			}
		}
	}
	cout<<fixed<<setprecision(12)<<area<<endl;
}
