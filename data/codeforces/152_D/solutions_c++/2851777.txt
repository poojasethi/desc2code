#include<cstdio>
#include<cstdlib>
#define rep(i,n) for(int i=0;i<n;i++)
#define FOR(i,a,b) for(int i=a;i<=b;i++)
#define N 1001
char s[N][N];
int n,m,cnt,tot,T,lx,ly,x[N],y[N];
void Check(int x1,int y1,int x2,int y2,int x3,int y3,int x4,int y4)
{
    if (x4-x3<2 || y4-y3<2) return;
	T++;cnt=0;
	FOR(j,y1,y2)
	{
		if (s[x1][j]<T) s[x1][j]=T,cnt++;
		if (s[x2][j]<T) s[x2][j]=T,cnt++;
	}
	FOR(i,x1,x2)
	{
		if (s[i][y1]<T) s[i][y1]=T,cnt++;
		if (s[i][y2]<T) s[i][y2]=T,cnt++;
	}
	FOR(j,y3,y4)
	{
		if ((s[x3][j]*s[x4][j])==0) return;
		if (s[x3][j]<T) s[x3][j]=T,cnt++;
		if (s[x4][j]<T) s[x4][j]=T,cnt++;
	}
	FOR(i,x3,x4)
	{
        if ((s[i][y3]*s[i][y4])==0) return;
		if (s[i][y3]<T) s[i][y3]=T,cnt++;
		if (s[i][y4]<T) s[i][y4]=T,cnt++;
	}
	if (cnt==tot)
	{
		puts("YES");
		printf("%d %d %d %d\n%d %d %d %d",x1+1,y1+1,x2+1,y2+1,x3+1,y3+1,x4+1,y4+1);
		exit(0);
	}
}
void Calc(int x1,int y1,int x2,int y2)
{
	if (x2-x1<2 || y2-y1<2) return;
	FOR(i,x1,x2)
	    if ((s[i][y1]*s[i][y2])==0) return;
	FOR(j,y1,y2)
	    if ((s[x1][j]*s[x2][j])==0) return;
	rep(x3,4)
	    rep(x4,4)
	        rep(y3,4)
	            rep(y4,4)
	                Check(x1,y1,x2,y2,x[x3],y[y3],x[x4],y[y4]);
}
int main()
{
	scanf("%d%d\n",&n,&m);
	rep(i,n) gets(s[i]);
	rep(i,n) rep(j,m) tot+=s[i][j]=s[i][j]=='#';
	rep(i,n) rep(j,m-2)
		if (s[i][j]&s[i][j+1]&s[i][j+2])
		{
			x[lx++]=i;
			break;
		}
	rep(j,m) rep(i,n-2)
	    if (s[i][j]&s[i+1][j]&s[i+2][j])
	    {
			y[ly++]=j;
			break;
		}
	if (lx>3)
		x[2]=x[lx-2],x[3]=x[lx-1];
	if (ly>3)
		y[2]=y[ly-2],y[3]=y[ly-1];
	T=1;
	rep(x1,4)
	    rep(x2,4)
	        rep(y1,4)
	            rep(y2,4)
	                Calc(x[x1],y[y1],x[x2],y[y2]);
	puts("NO");
	return 0;
}
