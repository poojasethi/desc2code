#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
char s[1010][1010]; 
vector<pair<int,int> > a;
int b[1010][1010],c[1010][1010],v[1010][1010],n,m,i,j,k,lx,rx,ly,ry,x,y,num;
#define F first
#define S second

bool calc()
{
	lx=n,rx=1,ly=m,ry=1;
	for(int i=0;i<a.size();i++)
		if(v[a[i].F][a[i].S]!=num)
			lx=min(lx,a[i].F),rx=max(rx,a[i].F),ly=min(ly,a[i].S),ry=max(ry,a[i].S),v[a[i].F][a[i].S]=num-1;
	if(lx>rx)
	{
		lx=i,rx=x,ly=j,ry=y;
		return 1;
	}
	if(ly==ry) lx=i,rx=x,ry<y?ry=y:ly=j; else if(lx==rx) ly=j,ry=y,rx<x?rx=x:lx=i;
	if(rx-lx<2||ry-ly<2) return 0;
	while(lx>1&&b[lx][ry]-b[lx][ly-1]<ry-ly+1) lx--;
	while(rx<n&&b[rx][ry]-b[rx][ly-1]<ry-ly+1) rx++;
	while(ly>1&&c[rx][ly]-c[lx-1][ly]<rx-lx+1) ly--;
	while(ry<m&&c[rx][ry]-c[lx-1][ry]<rx-lx+1) ry++;
	for(k=lx;k<=rx;k++)
		if(v[k][ly]<num-1||v[k][ry]<num-1) return 0; else v[k][ly]=v[k][ry]=num;
	for(k=ly;k<=ry;k++)
		if(v[lx][k]<num-1||v[rx][k]<num-1) return 0; else v[lx][k]=v[rx][k]=num;
	for(int i=0;i<a.size();i++)
		if(v[a[i].F][a[i].S]<num) return 0;
	return 1;
}

int main()
{
	cin>>n>>m;
	for(i=1;i<=n;i++)
	{
		scanf("%s",s[i]+1);
		for(j=1;j<=m;j++)
		{
			if(s[i][j]=='#') a.push_back(make_pair(i,j));
			b[i][j]=b[i][j-1]+(s[i][j]=='#');
			c[i][j]=c[i-1][j]+(s[i][j]=='#');
		} 
	}
	if(a.size()>4*(n+m-2)) {puts("NO"); return 0;}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
			if(s[i][j]=='#') break;
		if(j<=m)
		{
			if(s[i][j+1]!='#') break;
			for(y=j+2;y<=m;y++)
				if(s[i][y]=='#')
					for(x=i+1;x<=n&&s[x][j]=='#'&&s[x][y]=='#';x++)
					{
						for(k=j;k<=y;k++)
							if(s[x][k]=='.') break;
						if(x==i+1||k<=y) continue;
						num+=2;
						for(k=i;k<=x;k++) v[k][j]=v[k][y]=num;
						for(k=j;k<=y;k++) v[i][k]=v[x][k]=num;
						if(calc())
						{
							puts("YES");
							cout<<i<<' '<<j<<' '<<x<<' '<<y<<endl;
							cout<<lx<<' '<<ly<<' '<<rx<<' '<<ry<<endl;
							return 0;
						}
					}
				else break;
			break;
		}
	}
	puts("NO");
}
