#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<iostream>

using namespace std;

#define PI pair<int,int>
#define N 2005
#define x first
#define y second
#define MP(x,y) make_pair(x,y)

PI point1[N],point[N];
int dy[6][2]={{0,0},{0,0},{1,0},{2,0},{0,1},{0,2}},sum,sum1;
int cnt=0,cnt1=0,dy1[6][2]={{0,0},{0,0},{-1,0},{-2,0},{0,-1},{0,-2}},tot=0;
int used[N][N];
char c[N][N];

bool cmp(PI a,PI b)
{
	return a.x>b.x||(a.x==b.x&&a.y>b.y);
}

int check(int a,int b)
{
	if(point1[b].x-point[a].x<2) return -0x3f3f3f;
	if(point1[b].y-point[a].y<2) return -0x3f3f3f;
	int anss=0;
	for(int i=point[a].x;i<=point1[b].x;i++)
	{
		if(c[i][point[a].y]=='.') return -0x3f3f3f;
		if(c[i][point1[b].y]=='.') return -0x3f3f3f;
		if(used[i][point[a].y]!=tot)	used[i][point[a].y]=tot,anss++;
		if(used[i][point1[b].y]!=tot)	used[i][point1[b].y]=tot,anss++;
	}
	for(int i=point[a].y;i<=point1[b].y;i++)
	{
		if(c[point[a].x][i]=='.') return -0x3f3f3f;
		if(c[point1[b].x][i]=='.') return -0x3f3f3f;
		if(used[point[a].x][i]!=tot)	used[point[a].x][i]=tot,anss++;
		if(used[point1[b].x][i]!=tot)	used[point1[b].x][i]=tot,anss++;
	}	
	return anss;
}

void print(int a,int b)
{
	printf("%d %d %d %d\n",point[a].x,point[a].y,point1[b].x,point1[b].y);
}

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
		scanf("%s",c[i]+1);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			if(c[i][j]=='#') sum++;
	int summ=0;
	for(int i=1;i<=n-2;i++)
	{
		summ=0;
		for(int j=1;j<=m-2;j++)
		{
			for(int k=1;k<=5;k++)
			{
				int nowx=i+dy[k][0],nowy=j+dy[k][1];
				if(c[nowx][nowy]=='.')	goto syf;
			}
			summ++;
			point[++cnt]=MP(i,j);
			syf:i;
			if(summ>=3)	break;
		}
	}
	for(int i=3;i<=n;i++)
	{
		summ=0;
		for(int j=m;j>=3;j--)
		{
			for(int k=1;k<=5;k++)
			{
				int nowx=i+dy1[k][0],nowy=j+dy1[k][1];
				if(c[nowx][nowy]=='.')	goto syf1;
			}
			summ++;
			point1[++cnt1]=MP(i,j);
			syf1:i;
			if(summ>=3) break;
		}
	}		
	sort(point+1,point+1+cnt);
	sort(point1+1,point1+1+cnt1,cmp);
	for(int i=1;i<=6;i++)
		for(int j=1;j<=6;j++)
		{
			tot++;
			int k=check(i,j),k1=check(1,1);
			if(k+k1==sum)
			{
				printf("YES\n");
				print(1,1),print(i,j);
				exit(0);
			}
			tot++;
			k=check(1,j),k1=check(i,1);
			if(k+k1==sum)
			{
				printf("YES\n");
				print(1,j),print(i,1);
				exit(0);
			}
		}
	printf("NO\n");
	return 0;
}
