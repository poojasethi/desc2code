#include<cstdio>
#include<cstring>
#include<bitset>
using namespace std;
const int t=1111;
bitset<t> a[t],q[t];
int x[t],y[t];
char s[t];
void cl(int l,int r,int u,int d){
	int i;
	for (i=l;i<=r;i++) q[i][u]=q[i][d]=1;
	for (i=u;i<=d;i++) q[l][i]=q[r][i]=1;
}
int main(){
	int n,m,i,j,cx,cy,l1,r1,u1,d1,l2,r2,u2,d2,cnt;
//	freopen("152d.in","r",stdin);
//	freopen("152d.out","w",stdout);
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++){
		a[i]=0;
		scanf("%s",s);
		for (j=0;j<m;j++) a[i][j+1]=s[j]=='#'?1:0;
	}
	for (cx=0,i=1;i<=n;i++){
		for (cnt=0,j=1;j<=m && cnt<3;j++) cnt=a[i][j]?cnt+1:0;
		if (cnt==3) x[cx++]=i;
	}
	if (cx>4){
		x[2]=x[cx-2];
		x[3]=x[cx-1];
		cx=4;
	}
	for (cy=0,i=1;i<=m;i++){
		for (cnt=0,j=1;j<=n && cnt<3;j++) cnt=a[j][i]?cnt+1:0;
		if (cnt==3) y[cy++]=i;
	}
	if (cy>4){
		y[2]=y[cy-2];
		y[3]=y[cy-1];
		cy=4;
	}
	for (l1=0;l1<cx;l1++) for (r1=l1+1;r1<cx;r1++) if (x[r1]-x[l1]>1)
	for (u1=0;u1<cy;u1++) for (d1=u1+1;d1<cy;d1++) if (y[d1]-y[u1]>1)
	for (l2=l1;l2<cx;l2++) for (r2=l2+1;r2<cx;r2++) if (x[r2]-x[l2]>1)
	for (u2=0;u2<cy;u2++) for (d2=u2+1;d2<cy;d2++) if (y[d2]-y[u2]>1){
		for (i=1;i<=n;i++) q[i]=0;
		cl(x[l1],x[r1],y[u1],y[d1]);
		cl(x[l2],x[r2],y[u2],y[d2]);
		for (i=1;i<=n && q[i]==a[i];i++);
		if (i==n+1){
			puts("YES");
			printf("%d %d %d %d\n%d %d %d %d\n",
				x[l1],y[u1],x[r1],y[d1],x[l2],y[u2],x[r2],y[d2]);
			return 0;
		}
	}
	puts("NO");
	return 0;
}
