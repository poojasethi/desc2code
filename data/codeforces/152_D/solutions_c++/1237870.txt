#include <stdio.h>
#include <string.h>
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

typedef pair<int,int> PII;
typedef long long int64;

int n,m,total;
char f[1005][1005];
int row[1005],col[1005];

int g[1005][1005];
int paint(int y1, int x1, int y2, int x2, int c) {
	int st=0;
	for (int i=x1;i<=x2;i++) {
		if (g[y1][i]!=c) {
			g[y1][i]=c; st++;
			if (c==1 && f[y1][i]!='#') return -1;
		}
		if (g[y2][i]!=c) {
			g[y2][i]=c; st++;
			if (c==1 && f[y2][i]!='#') return -1;
		}
	}
	for (int i=y1;i<=y2;i++) {
		if (g[i][x1]!=c) {
			g[i][x1]=c; st++;
			if (c==1 && f[i][x1]!='#') return -1;
		}
		if (g[i][x2]!=c) {
			g[i][x2]=c; st++;
			if (c==1 && f[i][x2]!='#') return -1;
		}
	}
	return st;
}

int main() {
	scanf("%d %d",&n,&m);
	for (int i=0;i<n;i++) {
		scanf("%s",f[i]);
		for (int j=0;j<m;j++) if (f[i][j]=='#') total++;
	}
	vector<PII> c1;
	memset(row,0,sizeof(row));
	memset(col,0,sizeof(col));
	for (int i=0;i+2<n;i++) {
		for (int j=0;j+2<m;j++) {
			if (f[i][j]=='#' && f[i][j+1]=='#' && f[i][j+2]=='#' && f[i+1][j]=='#' && f[i+2][j]=='#') {
				if (row[i]<3 && col[j]<3) {
					c1.push_back(make_pair(i,j));
					row[i]++; col[j]++;
				}
			}
		}
	}
	vector<PII> c2;
	memset(row,0,sizeof(row));
	memset(col,0,sizeof(col));
	for (int i=n-1;i>=2;i--) {
		for (int j=m-1;j>=2;j--) {
			if (f[i][j]=='#' && f[i][j-1]=='#' && f[i][j-2]=='#' && f[i-1][j]=='#' && f[i-2][j]=='#') {
				if (row[i]<3 && col[j]<3) {
					c2.push_back(make_pair(i,j));
					row[i]++; col[j]++;
				}
			}
		}
	}
	if (c1.size()*c2.size()>100) {
		printf("NO\n");
		return 0;
	}
	for (int i1=0;i1<c1.size();i1++) for (int j1=0;j1<c2.size();j1++) {
		if (c2[j1].first-c1[i1].first<2 || c2[j1].second-c1[i1].second<2) continue;
		for (int i2=0;i2<c2.size();i2++) for (int j2=0;j2<c2.size();j2++) {
			if (c2[j2].first-c1[i2].first<2 || c2[j2].second-c1[i2].second<2) continue;
			int p1=paint(c1[i1].first, c1[i1].second, c2[j1].first, c2[j1].second, 1);
			int p2=paint(c1[i2].first, c1[i2].second, c2[j2].first, c2[j2].second, 1);
			if (p1!=-1 && p2!=-1 && p1+p2==total) {
				printf("YES\n");
				printf("%d %d %d %d\n", c1[i1].first+1, c1[i1].second+1, c2[j1].first+1, c2[j1].second+1);
				printf("%d %d %d %d\n", c1[i2].first+1, c1[i2].second+1, c2[j2].first+1, c2[j2].second+1);
				return 0;
			}
			paint(c1[i1].first, c1[i1].second, c2[j1].first, c2[j1].second, 0);
			paint(c1[i2].first, c1[i2].second, c2[j2].first, c2[j2].second, 0);
		}
	}
	printf("NO\n");
	return 0;
}
