#include<cstdio>
#include<cstdlib>
#include<vector>
#include<cstring>
#include<algorithm>
using namespace std;
typedef pair<int,int> II;
vector<II> a,b;
char st[1005][1005];
int f[1005][1005];
int n,m;
int color(II a, II b)
{	if (b.first-a.first<2||b.second-a.second<2)
		return 0;
	int i,j;
	for (i=a.first;i<=b.first;++i)
		for (j=a.second;j<=b.second;++j)
			if (i==a.first||i==b.first||j==a.second||j==b.second)
				f[i][j]=1;
	return 1;
}
void doit(II a0, II b0, II a1, II b1)
{	int i,j;
	memset(f,0,sizeof f);
	if (a1.first-a0.first<2||a1.second-a0.second<2)
		return ;
	if (b1.first-b0.first<2||b1.second-b0.second<2)
		return ;
	color(a0,a1),color(b0,b1);
	for (i=1;i<=n;++i)
		for (j=1;j<=m;++j)
			if (f[i][j]&&st[i][j]!='#'||!f[i][j]&&st[i][j]!='.')
				return;
	puts("YES");
	printf("%d %d %d %d\n",a0.first,a0.second,a1.first,a1.second);
	printf("%d %d %d %d\n",b0.first,b0.second,b1.first,b1.second);
	exit(0);
}	
void pre()
{	int i,j,c;
	for (i=1;i<=n-2;++i)
		if (a.size()<5)
		{	c=0;
			for (j=1;j<=m-2;++j) 
				if (a.size()<5&&c<3)
					if (st[i][j]=='#'&&st[i+1][j]=='#'&&st[i+2][j]=='#'&&st[i][j+1]=='#'&&st[i][j+2]=='#')
					{	a.push_back(make_pair(i,j));
						++c;
					}
		}
	for (i=n;i>=3;--i) 
		if (b.size()<5)
		{	c=0;
			for (j=m;j>=3;--j)
				if (b.size()<5&&c<3)
					if (st[i][j]=='#'&&st[i-1][j]=='#'&&st[i-2][j]=='#'&&st[i][j-1]=='#'&&st[i][j-2]=='#')
					{	b.push_back(make_pair(i,j));
						++c;
					}
		}
}
int main()
{	int i,j;
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;++i)
		scanf("%st",st[i]+1);
	pre();
	for(i=0;i<a.size();++i)
		for (j=0;j<b.size();++j)
			doit(a[0],a[i],b[0],b[j]),doit(a[0],a[i],b[j],b[0]);
	puts("NO");
	return 0;
}
