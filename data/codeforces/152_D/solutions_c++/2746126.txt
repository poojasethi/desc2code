#include <cstdio>
#include <cstring>
#include <algorithm>
#include <set>
using namespace std;

const int maxn = 1010;

#define foto(it,s) for (set < int > ::iterator it = s.begin(); it != s.end(); ++it) 

set < int > X,Y;

int n,m;
char st[maxn][maxn],sz[maxn][maxn];

bool fill(int x1,int y1,int x2,int y2)
{
    int i;
    for (i = x1; i <= x2; ++i) 
    {
        if (st[i][y1] == '.') return 0;
        if (st[i][y2] == '.') return 0;
        sz[i][y1] = sz[i][y2] = '#';
    }
    for (i = y1; i <= y2; ++i) 
    {
        if (st[x1][i] == '.') return 0;
        if (st[x2][i] == '.') return 0;
        sz[x1][i] = sz[x2][i] = '#';
    }
    return true;
}

int main()
{
    int i,j;

    scanf("%d %d\n",&n,&m);
    for (i = 1; i <= n; ++i)
        scanf("%s",st[i]+1);
    for (i = 1; i <= n; ++i)
        for (j = 1; j <= m; ++j)
        {
            if (st[i][j-1] == '#' && st[i][j] == '#' && st[i][j+1] == '#') X.insert(i);
            if (st[i-1][j] == '#' && st[i][j] == '#' && st[i+1][j] == '#') Y.insert(j);
        }

    if (X.size() > 4) X.erase(++++X.begin(),----X.end());
    if (Y.size() > 4) Y.erase(++++Y.begin(),----Y.end());

    foto(x1,X) foto(x2,X) foto(x3,X) foto(x4,X)
        foto(y1,Y) foto(y2,Y) foto(y3,Y) foto(y4,Y)
        if (*x1+1 < *x2 && *x3+1 < *x4 && *y1+1 < *y2 && *y3+1 < *y4)
        {
            memset(sz,'.',sizeof(sz));
            if (fill(*x1,*y1,*x2,*y2) && fill(*x3,*y3,*x4,*y4))
            {
                bool ok = true;
                for (i = 1; i <= n; ++i)
                    for (j = 1; j <= m; ++j)
                        if (st[i][j] != sz[i][j]) ok = false;
                if (ok) 
                {
                    printf("YES\n");
                    printf("%d %d %d %d\n",*x1,*y1,*x2,*y2);
                    printf("%d %d %d %d\n",*x3,*y3,*x4,*y4);
                    return 0;
                }
            }
        }
    printf("NO\n");
    return 0;
}
