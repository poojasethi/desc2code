#include<iostream>
#include<cstdio>
#include<cstring>
#include <algorithm>
using namespace std;
int a[1001][1001],b[1001][1001];
int n,m,bbb[1000],tb,c[1000],tc;
int xx[4],yy[4],sum;
int getch()
{
	char c=getchar();
	while (c!='#' && c!='.') c=getchar();
	if (c=='#') sum++;
	if (c=='#') return 1;
	else return 0;
}
bool panduan(int xx,int yy,int aa,int bb)
{
	for (int i=xx;i<=aa;++i)
		if (a[i][yy]!=1 || a[i][bb]!=1) return false;
	for (int i=yy;i<=bb;++i)
		if (a[xx][i]!=1 || a[aa][i]!=1) return false;
	return true;
}
int jiao(int a1,int a2,int a3,int a4,int a5,int a6,int a7,int a8)
{
	int ax1=xx[a1],ax2=xx[a2],ay1=yy[a3],ay2=yy[a4];
	int bx1=xx[a5],bx2=xx[a6],by1=yy[a7],by2=yy[a8];
	int sum=0;
	for (int i=ax1;i<=ax2;++i) 
	{
		if (!b[i][ay1]) sum++,b[i][ay1]=1;
		if (!b[i][ay2]) sum++,b[i][ay2]=1;
	}
	for (int i=ay1;i<=ay2;++i)
	{
		if (!b[ax1][i]) sum++,b[ax1][i]=1;
		if (!b[ax2][i]) sum++,b[ax2][i]=1;
	}
	for (int i=bx1;i<=bx2;++i)
	{
		if (!b[i][by1]) sum++,b[i][by1]=1;
		if (!b[i][by2]) sum++,b[i][by2]=1;
	}
	for (int i=by1;i<=by2;++i)
	{
		if (!b[bx1][i]) sum++,b[bx1][i]=1;
		if (!b[bx2][i]) sum++,b[bx2][i]=1;
	}
	//++++++++++++++++++++++++++++++++++++++++++
	for (int i=ax1;i<=ax2;++i) b[i][ay1]=0,b[i][ay2]=0;
	for (int i=ay1;i<=ay2;++i) b[ax1][i]=0,b[ax2][i]=0;
	for (int i=bx1;i<=bx2;++i) b[i][by1]=0,b[i][by2]=0;
	for (int i=by1;i<=by2;++i) b[bx1][i]=0,b[bx2][i]=0;
	return sum;
}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i)
		for (int j=1;j<=m;++j)
			a[i][j]=getch();
	for (int i=1;i<=n;++i)
		for (int j=1;j<=m-2;++j)
			if (a[i][j]+a[i][j+1]+a[i][j+2]==3)
				{ bbb[++tb]=i;break;}
	if (tb<2) { cout<<"NO"<<endl;return 0;}
	xx[0]=bbb[1];xx[1]=bbb[2];xx[2]=bbb[tb-1];xx[3]=bbb[tb];
	for (int i=1;i<=m;++i)
		for (int j=1;j<=n-2;++j)
			if (a[j][i]+a[j+1][i]+a[j+2][i]==3)
				{ c[++tc]=i;break;}
	if (tc<2) { cout<<"NO"<<endl;return 0;}
	yy[0]=c[1];yy[1]=c[2];yy[2]=c[tc-1];yy[3]=c[tc];
	for (int a1=0;a1<4;++a1)
		for (int a2=a1+1;a2<4;++a2)
			if (xx[a2]-xx[a1]>1)
				for (int a3=0;a3<4;++a3)
					for (int a4=a3+1;a4<4;++a4)
						if (yy[a4]-yy[a3]>1)
						if (panduan(xx[a1],yy[a3],xx[a2],yy[a4]))
						{
							for (int a5=0;a5<4;++a5)
								for (int a6=a5+1;a6<4;++a6)
								if (xx[a6]-xx[a5]>1)
									for (int a7=0;a7<4;++a7)
										for (int a8=a7+1;a8<4;++a8)
										if (yy[a8]-yy[a7]>1)
										if (panduan(xx[a5],yy[a7],xx[a6],yy[a8]))
										if (jiao(a1,a2,a3,a4,a5,a6,a7,a8)==sum)
										{
											cout<<"YES"<<endl;
											cout<<xx[a1]<<" "<<yy[a3]<<" "<<xx[a2]<<" "<<yy[a4]<<endl;
											cout<<xx[a5]<<" "<<yy[a7]<<" "<<xx[a6]<<" "<<yy[a8]<<endl;
											return 0;
										}
						}
	cout<<"NO"<<endl;
	return 0;
}
