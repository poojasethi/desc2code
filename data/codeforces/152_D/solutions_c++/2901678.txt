using std::vector;
  #define pb push_back
  #define forI_(V,i,_) for(typeof(V.begin())i=_;i!=V.end();++i)
  #define forI(V,i) forI_(V,i,V.begin())
  #define forD(V,i0,i1,i2,i3) \
    forI(V,i0)forI_(V,i1,i0)if(*i1-*i0>1) \
      forI(V,i2)forI_(V,i3,i2)if(*i3-*i2>1)

const int maxn=1010;
int n,m;
char g[maxn][maxn],s[maxn][maxn];
vector<int> A,B;

void pre(){
  int i,j;
  scanf("%d%d\n",&n,&m);
  for(i=0;i<n;++i)gets(s[i]);
  for(i=0;i<n;++i){
    for(j=2;j<m;++j)
      if(s[i][j-2]=='#'&&s[i][j-1]=='#'&&s[i][j]=='#')break;
    if(j<m)A.pb(i);
  }
  for(j=0;j<m;++j){
    for(i=2;i<n;++i)
      if(s[i-2][j]=='#'&&s[i-1][j]=='#'&&s[i][j]=='#')break;
    if(i<n)B.pb(j);
  }
  while(A.size()>4)A.erase(++++A.begin());
  while(B.size()>4)B.erase(++++B.begin());
  //for(i=0;i<A.size();++i)printf("%d ",A[i]);puts("");
  //for(i=0;i<B.size();++i)printf("%d ",B[i]);puts("");
}

inline void add(int x0,int y0,int x1,int y1){
  for(int i=x0;i<=x1;++i)
    for(int j=y0;j<=y1;++j)
      g[i][j]='#';
}

void sol(){
  int i,x0,y0,x1,y1,x2,y2,x3,y3;
  forD(A,ix0,ix1,ix2,ix3)
    forD(B,iy0,iy1,iy2,iy3){
      x0=*ix0,x1=*ix1,x2=*ix2,x3=*ix3;
      y0=*iy0,y1=*iy1,y2=*iy2,y3=*iy3;
      // Brute force two rectangles : (x0,y0)-(x1,y1) & (x2,y2)-(x3,y3)
      for(i=0;i<n;++i)
        memset(g[i],'.',sizeof(char)*m);
      add(x0,y0,x1,y0),add(x0,y1,x1,y1);
      add(x0,y0,x0,y1),add(x1,y0,x1,y1);
      add(x2,y2,x3,y2),add(x2,y3,x3,y3);
      add(x2,y2,x2,y3),add(x3,y2,x3,y3);
      for(i=0;i<n;++i)
        if(strcmp(s[i],g[i])!=0)break;
      if(i==n)goto ret;
    }
  puts("NO");return;
  ret:printf("YES\n%d %d %d %d\n%d %d %d %d\n",x0+1,y0+1,x1+1,y1+1,x2+1,y2+1,x3+1,y3+1);
}

int main(){
#ifdef FILEIO
  freopen("t.in","r",stdin);
  freopen("t.out","w",stdout);
#endif
  pre();
  sol();
  return 0;
}
