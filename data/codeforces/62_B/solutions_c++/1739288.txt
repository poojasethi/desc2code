#include<stdio.h>
#include<string.h>
#include<vector>
#include<iostream>

using namespace std;

vector<int> myv[30];

void ini(){
	for(int i=0;i<29;i++){
		myv[i].clear();		
	}
}

int abss(int a){
	if(a>0)return a;
	return -1*a;
	
}

int main(){
	__int64 ans;
	char orgi[200005],buff[200005];
	int i,j,n,k,a,b,len,l,pres,max;
	scanf("%d%d%s",&n,&k,orgi);
	
	ini();

	for(i=0;i<k;i++){
		myv[orgi[i]-'a'].push_back(i);	
	}

	for(i=0;i<n;i++){
		scanf("%s",&buff);
		
		len=strlen(buff);
		
		ans=0;

		for(j=0;j<len;j++){
			pres=buff[j]-'a';
			if(myv[pres].size()==0){ans+=len;}
			else{
				max=k+len;
				for(l=0;l<myv[pres].size()&&max;l++){
					if(abss(myv[pres][l]-j)<max){max=abss(myv[pres][l]-j);}
				}
				ans+=max;
			}
		}
		printf("%I64d\n",ans);

	}

	return 0;
}
