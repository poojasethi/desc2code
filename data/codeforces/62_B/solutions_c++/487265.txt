#include<cstdio>
#include<set>
#include<cstring>
#include<algorithm>
using namespace std;

set<int> ch[26];
char buf[1000000];
int n,m;

int main()
{
    scanf("%d%d",&n,&m);
    scanf("%s",buf);
    for (int i=0;i<m;i++)
        ch[buf[i]-'a'].insert(i);
    for (;n--;)
    {
        scanf("%s",buf);
        int len=strlen(buf);
        long long res=0;
        for (int i=0;buf[i];i++)
        {
            if (!ch[buf[i]-'a'].size()) {res+=len;continue;}
            int w=len;
            set<int> :: iterator it=ch[buf[i]-'a'].lower_bound(i);
            if (it!=ch[buf[i]-'a'].end()) w=*it-i;
            if (it!=ch[buf[i]-'a'].begin())
            {
                it--;w=min(w,i-*it);
            }
            res+=w;
        }
        printf("%I64d\n",res);
    }
}
