#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
  int n, k;
  string s;
  cin >> n >> k >> s;

  vector< vector<int> > p(26);
  for (unsigned int i = 0; i < s.size(); ++i)
    p[s[i] - 'a'].push_back(i);

  for (int i = 0; i < n; ++i) {
    long long int ans = 0;
    cin >> s;
    for (int j = 0; j < s.size(); ++j) {
      vector<int>& t = p[s[j] - 'a'];
      if (t.empty()) {
    ans += s.size();
      } else {
    vector<int>::iterator it = lower_bound(t.begin(), t.end(), j);
    if (it != t.begin()) {
      if (abs(j - *it) > abs(j - *(it-1)))
        ans += abs(j - *(it-1));
      else
        ans += abs(j - *it);
    } else {
      ans += abs(j - *it);
    }
      }
    }
    cout << ans << endl;
  }

  return 0;
}
