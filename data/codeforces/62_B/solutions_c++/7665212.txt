#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 200010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int a[26][N];
int64 an;
char p[N];
inline int cal(int x,int y) {
	if (!(*a[x])) return m;
	int l=1,r=*a[x],mid;
	for (;l<=r;) {
		mid=(l+r)/2;
		if (a[x][mid]<=y) l=mid+1;
		else r=mid-1;
	}
	l--;
	int an=N;
	if (l) an=min(an,abs(y-a[x][l]));
	if (l<*a[x]) an=min(an,abs(y-a[x][l+1]));
	return an;
}
int main() {
	scanf("%d%d",&n,&m);
	scanf("%s",p+1);
	For(i,1,m) {
		int A=p[i]-'a';
		a[A][++*a[A]]=i;
	}
	For(i,1,n) {
		scanf("%s",p+1);
		m=strlen(p+1);
		an=0;
		For(j,1,m) {
			int A=p[j]-'a';
			an+=cal(A,j);
		}
		printf("%I64d\n",an);
	}
	//for(;;);
	return 0;
}
