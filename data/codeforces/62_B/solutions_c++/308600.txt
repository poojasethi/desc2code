#include<iostream>
#include<vector>
#include<string>
typedef long long llong;

using namespace std;

int const INF = 123456789;

inline int abs(int a)
{
	return (a < 0) ? -a : a;
}

int main()
{
	int n, k;
	string user_url;
	cin >> n >> k;
	cin >> user_url;
	vector<vector<int> > smap(256, vector<int>());
	for(int i = 0; i < k; i++)
		smap[user_url[i]].push_back(i);

	for(int c = 0; c < n; c++)
	{
		string c_url;
		cin >> c_url;
		llong f = 0;
		for(int j = 0; j < (int)c_url.length(); j++)
		{
			if(smap[c_url[j]].size() == 0)
				f += (llong)c_url.length();
			else
			{
				int pos = INF;
				for(int i = 0; i < smap[c_url[j]].size(); i++)
					pos = min(pos, abs(j - smap[c_url[j]][i]));
				f += (llong)pos;
			}
		}
		cout << f << endl;
	}
	return 0;
}
