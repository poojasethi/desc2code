#include <bits/stdc++.h>
using namespace std;

#define LL long long
#define PB push_back
#define MP make_pair

int n, k;
string s, t;
vector<int> have[26];

void pre() {
	for (int i = 0; i < k; i++)
		have[ s[i] - 'a' ].PB(i);
}
LL FError(string t) {
	LL ans = 0LL;
	for (int i = 0; i < t.size(); i++) {
		int cur = t[i] - 'a';
		if (have[cur].size() == 0) {
			ans += (int)t.size();
			continue;
		}
		int idx = lower_bound(have[cur].begin(), have[cur].end(), i) - have[cur].begin();
		if (idx == 0) ans += abs(have[cur][idx] - i);
		else ans += min(abs(have[cur][idx] - i), abs(have[cur][idx - 1] - i));
	}
	return ans;
}
int main() {
	cin >> n >> k;
	cin >> s;
	pre();
	for (int i = 0; i < n; i++) {
		cin >> t;
		cout << FError(t) << endl;
	}
	return 0;
}
