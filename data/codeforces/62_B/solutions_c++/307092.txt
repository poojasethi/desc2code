#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
#include<cmath>
using namespace std;
#define pb push_back
#define ll long long
vector<int> a[300];
string k,b;
int n,l;
ll wh(int u) {
	if(a[int(b[u])].size() == 0)
		return b.length() ;
	int le = 0 , m ;
	int ri = a[int(b[u])].size() - 1 ;
	if( a[ int(b[u]) ][ ri ] >= u && a[ int(b[u]) ][ le ] <= u ) {
		while ( le + 1 < ri ) {
			m = ( le + ri ) / 2 ;
			if(a[ int(b[u]) ][ m ] >= u)
				ri = m ;
			else
				le = m ;
		}
	}
	return min( abs( a[int(b[u])][le] -  u )  ,  abs( a[int(b[u])][ri] -  u ) );
}
ll ans() {
	int len = b.length();
	ll t = 0 ;
	for(int i = 0 ; i < len ; i++ )
		t += wh(i) ;
	return t ;
}
int main() {
	cin >> n >> l ;
	cin >> k ;
	for(int i = 0 ; i < l ; i++ )
		a[ int(k[i]) ].pb(i);
	for(int i = 0 ; i < n ; i++ ) {
		cin >> b;
		cout << ans() << endl;
	}
	return 0;
}
