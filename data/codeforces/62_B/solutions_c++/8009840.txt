#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
using namespace std;

vector <int> v[500];

int BS(int l,int r,int n,char c)
{
	if(l+1>=r)
	{
		if(abs(v[c][l]-n)<=abs(v[c][r]-n))
			return abs(v[c][l]-n);
		return abs(v[c][r]-n);
	}
	int m=(l+r)/2;
	if(v[c][m]>n)
		return BS(l,m,n,c);
	return BS(m,r,n,c);
}

int main()
{
	int n,k;
	bool ch[500]={};
	string s;
	cin>>n>>k>>s;
	for(int i=0;i<k;i++)
	{
		ch[s[i]]=1;
		v[s[i]].push_back(i);
	}
	for(int i=0;i<n;i++)
	{
		long long ans=0;
		string t;
		cin>>t;
		for(int j=0;j<t.length();j++)
		{
			//cout<<t[j]<<" "<<v[t[j]].size()<<endl;
			if(ch[t[j]])
				ans+=BS(0,v[t[j]].size()-1,j,t[j]);
			else
				ans+=t.length();
		}
		cout<<ans<<endl;
	}
	return 0;
}
