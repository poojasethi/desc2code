#include <cstdlib>
#include <algorithm>
#include <set>
#include <iostream>
using namespace std;

#define REP(i,b,n) for(int i=b;i<n;i++)
#define rep(i,n) REP(i,0,n)

const int inf = 1<<28;

int main(int argc, char *argv[]) {
  int n,k;
  cin>>n>>k;
  string s,t;
  cin>>s;
  set<int> l[26];
  rep(i,k) l[s[i] - 'a'].insert(i);
  rep(i,26) {
    l[i].insert(inf);
    l[i].insert(-inf);
  }
  while (n--) {
    long long cnt = 0, tmp, ll;
    set<int>::iterator lhs, rhs;
    cin>>t;
    rep(i,t.size()) {
      tmp = t[i] - 'a';
      lhs = rhs = l[tmp].upper_bound(i);
      lhs--;
      ll = min(*rhs - i, i - *lhs);
      if (ll < inf / 2)
        cnt += ll;
      else
        cnt += t.size();
    }
    cout<<cnt<<endl;
  }
  return 0;
}
