#include<cstdio>
#include<cstring>
#include<cstdlib>

using namespace std;

#define MAXN 400000

typedef long long lld;

int P[MAXN],R[MAXN],now[MAXN];
char S[MAXN],ts[MAXN];
lld tmp,min;
int len,N,K;

int main(){
	scanf("%d%d\n",&N,&K);
	gets(S+1);
	len=strlen(S+1);
	for (int i=1;i<=len;i++){
		P[i]=R[S[i]-'a'+1];R[S[i]-'a'+1]=i;
	}

	for ( int i=1;i<=N;i++){
		gets(ts+1);
		len=strlen(ts+1);
		tmp=0;
		for ( int i=1;i<=26;i++)
			now[i]=R[i];
		for ( int i=len;i;i--){
			if (!now[ts[i]-'a'+1]){
				tmp+=len;
				continue;
			}
			min=abs(now[ts[i]-'a'+1]-i);
			while (P[now[ts[i]-'a'+1]]){
				if (abs(P[now[ts[i]-'a'+1]]-i)<min){
					min=abs(P[now[ts[i]-'a'+1]]-i);
					now[ts[i]-'a'+1]=P[now[ts[i]-'a'+1]];
				} else break;
			}
			tmp+=min;
		}
		printf("%I64d\n",tmp);
	}

}
