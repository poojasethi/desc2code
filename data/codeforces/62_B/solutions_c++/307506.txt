#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

long long f(string a,string b) {
	vector<int> vec[26];
	for (int i=0; i<a.size(); i++) {
		vec[a[i]-'a'].push_back(i);
	}
	long long r=0;
	for (int i=0; i<b.size(); i++) {
		int n=b[i]-'a';
		while(vec[n].size()>1) {
			if (abs(i-vec[n][0])>=abs(i-vec[n][1]))vec[n].erase(vec[n].begin());
			else break;
		}
		if (vec[n].size()!=0)
		{
			r+=abs(i-vec[n][0]);
		} else {
			r+=b.size();
		}
	}
	return r;
}

int main(){
	int n,k;
	cin >> n >> k;
	string a;
	cin >> a;
	for (int i=0; i<n; i++)
	{
		string b;
		cin >> b;
		cout << f(a,b) << endl;
	}
	return 0;
}
