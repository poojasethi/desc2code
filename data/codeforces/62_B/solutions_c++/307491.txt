#include<stdio.h>
#include<string.h>
int a[26][100001],f[27];
char str[200002];
int abs(int x)
{
	return x>0?x:-x;
}
int min(int a,int b)
{
	return a<b?a:b;
}
int main()
{
	int n,k,i,j,low,high,mid,c;
	__int64 res;
	scanf("%d%d",&n,&k);
	scanf("%s",str);
	for(i=0;str[i];i++)
	{
		j=str[i]-'a';
		a[j][f[j]++]=i;
	}
	while(n--)
	{
		res=0;
		scanf("%s",str);
		c=strlen(str);
		for(i=0;str[i];i++)
		{
			j=str[i]-'a';
			if(!f[j])res+=c;
			else if(f[j]==1)res+=abs(a[j][0]-i);
			else
			{
				low=0;
				high=f[j];
				while(low<high-1)
				{
					mid=(low+high)/2;
					if(a[j][mid]>i)
						high=mid;
					else
						low=mid;
				}
				if(high==f[j])
					res+=abs(i-a[j][low]);
				else
					res+=min(abs(i-a[j][low]),abs(a[j][high]-i));
			}
		}
		printf("%I64d\n",res);
	}
}
