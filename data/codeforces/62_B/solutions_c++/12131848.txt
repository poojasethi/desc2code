#include <stdio.h>
#include <bits/stdc++.h>

using namespace std;

vector<pair<char,int> > ci;

long long busqueda(int j, char c, int k) {
	int s = 0;
	while (s<k) {
		int m = (s+k)/2;
		if (ci[m].first>=c) {
			k = m;
		} else {
			s = m+1;
		}
	}
	s;
	int resu = 1000000;
	while(ci[s].first == c) {
		resu = min(resu,abs(ci[s].second-j));
		s++;
	}
	return resu;
}

int main() {
	int n,k;
	cin >> n >> k;
	string pal;
	cin >> pal;
	for (int i=0;i<k;i++) {
		ci.push_back(pair<char,int>(pal[i],i));
	}
	sort(ci.begin(),ci.end());
	for (int i=0;i<n;i++) {
		string pol;
		cin >> pol;
		long long s = 0;
		int le = pol.length();
		for (int j=0;j<le;j++) {
			if (j >= k || pal[j] != pol[j]) {
				long long res = busqueda(j,pol[j],k);
				s+= (res==1000000) ? le : res; 
			}
		}
		printf("%I64d\n",s);
	}
	return 0;
}
