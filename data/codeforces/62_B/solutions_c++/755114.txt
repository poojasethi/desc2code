#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

int MAXLEN = 200000;
char s[200005];
int v[200005][26];
int d[26], INF = 1234123;

int main(void) {
	int n, k;
	scanf("%d%d", &n, &k);
	scanf("%s", s);
	for(int i=0;i<26;i++) d[i]=-INF;
	for(int i=0;i<MAXLEN;i++) {
		if(i<k)
			d[s[i]-'a'] = i;
		for(int j=0;j<26;j++) {
			if(d[j]==-INF) v[i][j] = INF;
			else v[i][j] = i - d[j];
		}
	}
	for(int i=0;i<26;i++) d[i] = INF;
	for(int i=MAXLEN-1;i>=0;--i) {
		if(i<k)
			d[s[i]-'a'] = i;
		for(int j=0;j<26;j++) {
			if(d[j]<INF)
				v[i][j] = min(v[i][j], d[j] - i);
		}
	}
	while(n--) {
		scanf("%s", s);
		int l = strlen(s);
		long long F = 0;
		for(int i=0;s[i];i++) {
			if(v[i][s[i]-'a'] == INF) F += l;
			else F += v[i][s[i]-'a'];
		}
		printf("%I64d\n", F);
	}
	return 0;
}
