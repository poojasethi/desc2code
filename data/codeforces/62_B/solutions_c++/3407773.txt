#include <stdio.h>
#include <string.h>
int ab(int x)
{
    if(x>=0)
    return x;
    return -x;
}
int main()
{
    char s[100111],a[101010];
    int n,k,flag[30]={0};
    int p[26][100000]={0};
    int l[26]={0};
    scanf ("%d%d",&n,&k);
    scanf ("%s",s);
    for(int i=0;i<k;i++)
    {
        flag[s[i]-'a']=1;
        p[s[i]-'a'][l[s[i]-'a']++]=i;
    }
    while(n--)
    {
        scanf ("%s",a);
        int la=strlen(a);
        long long ans=0;
        for(int j=0;j<la;j++)
        {
            int key=a[j]-'a';
            int min=999999;
            if(flag[key]==0)
            {ans+=la;continue;}
            else
            {
                for(int i=0;i<l[key];i++)
                if(ab(p[key][i]-j)<min)
                min=ab(p[key][i]-j);
                ans+=min;
            }
        }
        printf("%I64d\n",ans);
    }
}
