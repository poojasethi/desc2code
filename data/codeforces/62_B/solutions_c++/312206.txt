#include<cstdio>
#include<set>
#include<algorithm>
#include<vector>
#include<cstring>
using namespace std;
typedef long long ll;

char s[400000];

int main(){

	vector<int> pos[26];
	int n, k;
	scanf("%d %d", &n, &k);

	scanf("%s", s);
	for(int i = 0; s[i]; i++) pos[s[i] - 'a'].push_back(i);
	
	while(n--){
		scanf("%s", s);
		int len = strlen(s);
		ll res = 0;
		for(int i = 0; s[i]; i++){
			int best = -1;
			int ii = s[i] - 'a';
			int x = lower_bound(pos[ii].begin(), pos[ii].end(), i) - pos[ii].begin();
			if(x != pos[ii].size()) best = pos[ii][x] - i;
			x--;
			if(x >= 0) best = min(best == -1 ? i - pos[ii][x] : best, i - pos[ii][x]);
			if(best == -1) best = len;
			res += best;
		}
		printf("%I64d\n", res);
	}
}
