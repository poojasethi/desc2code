#include<iostream>
#include<stdio.h>
#include<string>
#include<set>
#include<math.h>
#include<string.h>
#define ll long long
//#define debug
using namespace std;

set<int>s[300];
set<int>::iterator it;
int abs(int x)
{
	return x>0?x:-x;
}
char str[1000000];
int main()
{
	int n,k;
	scanf("%d%d",&n,&k);

	for(int i=0;i<k;++i)
	{
		char c;
		cin>>c;
		s[c-'a'].insert(i);
	}
	gets(str);
	for(int i=0;i<n;++i)
	{
		gets(str);
		ll sum=0;
		int len = strlen(str);
		for(int j=0;j<len;++j)
		{
			it = s[str[j]-'a'].lower_bound(j);
			ll add = 1000000;
			if( it != s[str[j]-'a'].end() )
			{
				add = min(add,(ll)abs(*it-j));
			}
			if( it != s[str[j]-'a'].begin() )
			{
				add = min(add,(ll)abs(*(--it)-j));
			}
			if( add == 1000000 )
				add = len;
			sum += add;
		}
		cout<<sum<<endl;
	}
	return 0;
}
