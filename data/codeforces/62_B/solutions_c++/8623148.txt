#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
#include <cstring>
using namespace std;
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,*a[26],d[26];
  long long t;
  for(w=0;w<26;w++){
   d[w]=2000001;
   a[w]=new int[200001];}
  char s[200001];
  cin>>q>>w;
  scanf("%s",s);
  for(e=200000;e>=0;e--){
   if(e<w)
    d[s[e]-97]=-1;
   for(r=0;r<26;r++){
    d[r]++;
    a[r][e]=d[r];}}
  for(e=0;e<26;e++)
   d[e]=200000;
  for(e=0;e<200001;e++){
   if(e<w)
    d[s[e]-97]=-1;
   for(r=0;r<26;r++){
    d[r]++;
    a[r][e]=min(d[r],a[r][e]);}}
  for(;q;q--){
   t=0;
   scanf("%s",s);
   r=strlen(s);
   for(e=0;s[e];e++)
    if(a[s[e]-97][e]>200000)
     t+=r;
    else
     t+=a[s[e]-97][e];
   printf("%I64d\n",t);}
  return 0;}
