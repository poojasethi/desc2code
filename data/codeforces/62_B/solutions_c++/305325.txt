#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
typedef long long ll;
int N,K;
string s;
vector<int> pos[256];
int main()
{
	cin.sync_with_stdio(0);
	cin>>N>>K;
	cin>>s;
	for(int i=0; i<K; ++i) {
		pos[(int)s[i]].push_back(i);
	}
	string q;
	for(int i=0; i<N; ++i) {
		cin>>q;
		ll r=0;
		for(int j=0; j<(int)q.size(); ++j) {
			int c = q[j];
			if (pos[c].empty()) {
				r += q.size();
				continue;
			}
			int a = lower_bound(pos[c].begin(), pos[c].end(), j) - pos[c].begin();
			int f = 1e9;
			if (a>0) f = j - pos[c][a-1];
			if (a<(int)pos[c].size()) f = min(f, pos[c][a]-j);
			r += f;
		}
		cout<<r<<'\n';
	}
}
