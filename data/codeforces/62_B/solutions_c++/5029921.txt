#include<iostream>
#include<vector>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
typedef unsigned long long int L;
int main()
{
    vector<L> v[26];
    L i,j,n,k,d,ct,f,min;
    char c;
    cin>>n>>k;
    c=getchar();
    i=0;
    while((c=getchar())!='\n')
    {
        v[c-97].push_back(i);
        i++;
    }
    while(n--)
    {
        i=f=ct=0;
        while((c=getchar())!='\n')
        {
            if(((L)v[c-97].size())==0)
                ct++;
            else
            {
                min=abs(i-v[c-97][0]);
                for(j=1; (j<(L)v[c-97].size()&&(abs(i-v[c-97][j])<min)); j++)
                    min=abs(i-v[c-97][j]);
                f+=min;
            }
            i++;
        }
        f=f+i*ct;
        cout<<f<<endl;
    }
    return 0;
}
