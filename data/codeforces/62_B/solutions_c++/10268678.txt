#include<iostream>
#include<algorithm>
#include<cstdio>
#include<vector>
#include<string>
#include<cmath>
using namespace std;
int n, k;
vector<int> dic[200];
int main() {
	//freopen("test.in", "r", stdin);
	//freopen("test.out", "w", stdout);
	cin >> n >> k;
	getchar();
	string s;
	//cin >> s;
	getline(cin, s);
	//cout << s << endl << s.size() << endl;
	for (int i = 0; i < s.size(); i++)
		dic[s[i]].push_back(i);
	for (int i = 0; i < n; i++) {
		//cin >> s;
		getline(cin, s);
		long long ans = 0;
		for (int j = 0; j < s.size(); j++) {
			if (dic[s[j]].size() == 0) {
				ans += s.size();
				continue;
			}
			int cnt = 100000;
			for (int t = 0; t < dic[s[j]].size(); t++)
				cnt = min(cnt, abs(j-dic[s[j]][t]));
			ans += cnt;
		}
		cout << ans << endl;
	}
	return 0;
}
