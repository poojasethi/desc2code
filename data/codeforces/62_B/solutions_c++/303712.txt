using namespace std;
#define li		long long
#define rep(i,to)	for(li i=0;i<(li)to;i++)
#define pb(a)		push_back(a)
#define sz(v)		((li)v.size())
int main(){
	ios_base::sync_with_stdio(0);
	int n,k;
	string str;
	int c[26][100000];
	int len[26];
	rep(i,26) len[i]=0;
	cin>>n>>k;
	cin>>str;
	rep(i,sz(str))	c[str[i]-'a'][len[str[i]-'a']++]=i;
	while(n--){
		cin>>str;
		li res=0;
		rep(i,sz(str)){
			int s=str[i]-'a';
			if(len[s]==0){
				res+=sz(str);
			}else{
				int low=0,high=len[s]-1;
				while(high-low>1){
					int mid=(high+low)/2;
					if(c[s][mid]<i) low=mid;
					else high=mid;
				}
				res+=min(abs(c[s][low]-i),abs(c[s][high]-i));
			}
		}
		cout<<res<<endl;
	}
}
