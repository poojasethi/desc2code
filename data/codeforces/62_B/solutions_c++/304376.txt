#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
#include <string.h>
#include <math.h>
using namespace std;

char a[100040],b[100040];
int c[27][100040];
int n,m,k;
int l,r,mid;
long long int ans;

int find(int k,int o,int ll){
	if (c[k][0]==0) return ll;
	l=0;r=c[k][0];
	while (l<r-1){
		mid=(l+r)>>1;
		if (c[k][mid]<o) l=mid;
		else r=mid;
	}
	if (r==1) return abs(c[k][r]-o);
	if (abs(c[k][r]-o)<abs(o-c[k][r-1])) return abs(c[k][r]-o);
	else return abs(o-c[k][r-1]);
}

int main(){
	scanf("%d%d",&m,&n);
	scanf("%s",a);
	for (int i=1;i<=26;++i) c[i][0]=0;
	for (int i=0;i<n;++i)
	 c[a[i]-96][++c[a[i]-96][0]]=i;
	
	while (m--){
		scanf("%s",b);
		ans=0;
		k=strlen(b);
		for (int i=0;i<k;++i)
		 ans+=find(b[i]-96,i,k);
		printf("%I64d\n",ans);
	}
	return 0;
}
