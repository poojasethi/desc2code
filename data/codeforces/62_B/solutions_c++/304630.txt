#include<iostream>
#define inf 1000000
using namespace std;
long long k=0,p;
int n,m,i,j,l,a[100005][30],b[100005][30];
string s,t;
int main()
{
cin>>n>>m>>s;
for(i=0;i<m;i++)
{
if(i==0)
{
for(j=0;j<26;j++)
{
a[i][j]=-inf*10;
}
}
else
{
for(j=0;j<26;j++)
{
a[i][j]=-inf*10;
if(i>0) a[i][j]=a[i-1][j];
}
}
a[i][s[i]-'a']=i;
}
for(i=m-1;i>=0;i--)
{
if(i==m-1)
{
for(j=0;j<26;j++)
{
b[i][j]=inf*10;
}
}
else
{
for(j=0;j<26;j++)
{
b[i][j]=b[i+1][j];
}
}
b[i][s[i]-'a']=i;
}
for(l=1;l<=n;l++)
{
cin>>t;
k=0;
for(i=0;i<t.length();i++)
{
p=inf*10;
if(i<m) p=min(p,(long long)b[i][t[i]-'a']-i);
p=min(p,(long long)i-a[min(i,m-1)][t[i]-'a']);
if(p>inf) p=t.length();
k+=p;
}
cout<<k<<endl;
}
return 0;
}
