# -*- coding: koi8-r -*-

def readarray(foo): return [foo(x) for x in raw_input().split()]

def search(a, x):
        l = 0
        r = len(a)
        while r - l > 1:
                m = (l + r) / 2
                if (a[m] <= x):
                        l = m
                else:
                        r = m
        r1 = abs(a[l] - x)
        r2 = abs(a[l + 1] - x) if l + 1 < len(a) else r1
        return min(r1, r2)

def main():
        (n, k) = readarray(int)
        s = raw_input()
        pos = {}
        for (i, c) in enumerate(s):
                if c not in pos:
                        pos[c] = []
                pos[c].append(i)
        for i in xrange(n):
                t = raw_input()
                res = 0
                for (i, c) in enumerate(t):
                        res += search(pos[c], i) if c in pos else len(t)
                print res
main()
