#include <iostream>
#include <string>
#include <set>
using namespace std;

const int N = 2036;
string a[N], b[N];
int t[N];
set<string> ans;

int main() {
    int n, d;
    cin >> n >> d;
    for (int i = 0; i < n; i++) {
        cin >> a[i] >> b[i] >> t[i];
    }
    
    for (int i = 0; i < n; i++) for (int j = 0; j < n; j++) if (i != j) {
        if (t[j] - t[i] <= d && 0 < t[j] - t[i] && a[i] == b[j] && a[j] == b[i]) {
            string s = min(a[i], b[i]) + " " + max(a[i], b[i]);
            ans.insert(s);
        }
    }
    
    cout << ans.size() << endl;
    for (set<string>::iterator it = ans.begin(); it != ans.end(); ++it) {
        cout << *it << endl;
    }
    
    return 0;
}
