#include <iostream>
#include <string>
#include <set>

using namespace std;
string W1[1024];
int D[1024], P[1024];
set<string> sol;
int main() {
  int n, k;
  cin >> n >> k;
  for (int i = 0; i < n; ++i) {
    string w1, w2; int d;
    cin >> w1 >> w2 >> d;
    P[i] = 0;
    if (w1 < w2) swap(w1, w2), P[i] = 1;
    W1[i] = w1 + " " + w2; D[i] = d;
  }
  for (int i = 0; i < n; ++i)
    for (int j = i + 1; j < n; ++j) if (D[j] - D[i] > 0 && D[j] - D[i] <= k && W1[i] == W1[j] && P[i] + P[j] == 1) {
      sol.insert(W1[i]);
    }
  cout << sol.size() << '\n';
  for (set<string>::iterator it = sol.begin(); it != sol.end(); ++it)
    cout << *it << '\n';
  return 0;
}
