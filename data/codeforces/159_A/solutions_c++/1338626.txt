#include<iostream>
#include<set>
using namespace std;

string a[1000], b[1000];
int t[1000];

int main()
{
	int n, d;
	set<pair<string, string> > ans;

	cin >> n >> d;

	for(int i = 0; i < n; ++i)
	{
		cin >> a[i] >> b[i] >> t[i];

		for(int j = 0; j < i; ++j)
		{
			if((a[i] == b[j])&&(b[i] == a[j])&&(t[i] > t[j])&&(t[i] - t[j] <= d))
			{
				if(a[i] > b[i]) swap(a[i], b[i]);
				ans.insert(pair<string, string> (a[i], b[i]));
			}
		}
	}

	cout << ans.size() << endl;
	for(set<pair<string, string> >::iterator it = ans.begin(); it != ans.end(); ++it)
		cout << it->first << ' ' << it->second << endl;

	return 0;
}
