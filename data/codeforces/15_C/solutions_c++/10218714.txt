#include<cstdio>
int main()
{
    int n;
    long long x,m,ans=0;
    scanf("%d",&n);
    while(n--)
    {
        scanf("%I64d %I64d",&x,&m);
        while(x%4&&m)
        {
            ans^=x;
            m--;
            x++;
        }
        x=x+m-1;
        m%=4;
        while(m--)
        {
            ans^=x;
            x--;
        }
    }
    printf("%s\n",ans?"tolik":"bolik");
}
