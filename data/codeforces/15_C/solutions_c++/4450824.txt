#include <algorithm>
#include <stdio.h>
using    namespace std;
typedef  long long LL;

LL n,ans;

LL f(LL X)
{
  if(X%4==0) return X;
  if(X%4==1) return 1;
  if(X%4==2) return X+1;
  if(X%4==3) return 0;
}

int main()
{
  scanf("%I64d",&n);
  for(LL i=0,x,m ; i<n ; i++)
  {
    scanf("%I64d%I64d",&x,&m);
    ans^=f(x+m-1);
    ans^=f(x-1);
  }
  if(ans)  printf("tolik\n");
  else     printf("bolik\n");
  return 0;
}
