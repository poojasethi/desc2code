#include <bits/stdc++.h>

int main() {
    int n;
    long long ans = 0;
    scanf("%d", &n);
    while (n--) {
        long long x, m;
        scanf("%I64d%I64d", &x, &m);
        if (x&1) ans ^= x^(((m-1)/2)&1)^((m-1)&1?x+m-1:0);
        else ans ^= ((m/2)&1)^(m&1?x+m-1:0);
    }
    printf("%s\n", ans?"tolik":"bolik");
    return 0;
}
