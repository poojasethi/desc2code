#include <cstdio>
#include <cassert>
typedef long long LL;
LL f(LL x){
  if(x % 4 == 0) return x;
  if(x % 4 == 3) return 0;
  if(x % 4 == 2) return x + 1;
  if(x % 4 == 1) return 1;
  assert(0);
}
int n;
LL m, x;
LL g, res;
int main(){
  scanf("%d", &n);
  g = 0;
  for(int i = 0; i < n; i++){
    scanf("%I64d %I64d", &x, &m);
    res ^= f(x-1) ^ f(x+m-1);
  }
  puts(res ? "tolik" : "bolik");
  return 0;
}
