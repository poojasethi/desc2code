#include <cstdio>
#include <cstring>
long long int a[100500];
long long int yihuo(long long int x)
{
	if(x%4==0)return x;
	if(x%4==1)return 1;
	if(x%4==2)return x+1;
	if(x%4==3)return 0;
}
int main()
{
	int n;
	scanf("%d",&n);
	a[0]=0;
	for(int i=1;i<=n;i++)
	{
		long long int x,m;
		scanf("%I64d %I64d",&x,&m);
		a[i]=(yihuo(x+m-1))^(yihuo(x-1))^a[i-1];
	}
	if(a[n])printf("tolik\n");
	else printf("bolik\n");
	return 0;
}
