#include <iostream>
#include <vector>

using namespace std;

int main() {
	int n;
	cin >> n;
	long long ans = 0;
	while (n-- > 0) {
		long long a, b;
		cin >> a >> b;
		long long ed = a + b - 1;
		if (a & 1) {
			ans ^= a;
			a++, b--;
		}
		if (b) {
			long long len = b >> 1;
			if (len & 1)
				ans ^= 1;
			if (b & 1)
				ans ^= ed;
		}
	}
	if (ans)
		cout << "tolik" << endl;
	else
		cout << "bolik" << endl;
}
