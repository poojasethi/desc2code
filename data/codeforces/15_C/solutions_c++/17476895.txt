#include<cstdio>
typedef long long ll;
int main(){
	int n,tm;
	ll x,m,p=0;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%I64d%I64d",&x,&m);
		if(x&1){
			p=p^x;
			m--;
			x++;
			if(m<=0)continue;
		}
		if(m&1==0){
			m=m/2;
			if(m&1)
				p=p^1;
		}
		else{
			tm=m%4;
			for(int j=0;j<tm;j++){
				p=p^(m/4*4+j+x);
			}
		}
	}
	puts(p?"tolik":"bolik");
	return 0;
}
