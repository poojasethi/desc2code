#include<cstdio>
int main()
{
	__int64 x,m,i,n;
	while(~scanf("%I64d",&n))
	{
		__int64 s=0,y1,y2;
		while(n--)
		{
			scanf("%I64d%I64d",&x,&m);
			y1=x/4;y1=(y1+1)*4;
			for(i=x;i<y1;i++) s^=i;
			y2=(x+m-1)/4;y2=y2*4;
			for(i=y2;i<x+m;i++) s^=i;
		}
		if(s) puts("tolik");
		else puts("bolik");
	}
	return 0;
}
