#include <stdio.h>
#include <inttypes.h>

typedef int64_t Int;

Int f(Int n) {
  if(n & 1) return ((n & 3) == 1) ? 1 : 0;
  return (!(n & 3) ? n : n+1);
}

int main() {
  int n;
  Int xored = 0, x, m;
  scanf("%d", &n);
  while(n--) {
    scanf("%lld %lld", &x, &m);
    xored ^= f(x+m-1) ^ f(x-1);
  }
  puts(xored ? "tolik" : "bolik");
  return 0;
}
