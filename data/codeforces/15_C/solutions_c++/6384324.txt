#include<stdio.h>
#include <iostream>
using namespace std;
long long sum(long long x)
{
  switch (x % 4)
  {
  case 0:
    return 0;
  case 1:
    return x - 1;
  case 2:
    return 1;
  case 3:
    return x;
  }
}
int main(void)
{
  long long xorsum = 0;
  int n;
  cin>>n;
  while (n--) 
  {
    long long x, m;
    cin>>x>>m;
    xorsum ^= sum(x) ^ sum(x + m);
  }
  puts(xorsum ? "tolik" : "bolik");
  return 0;
}
