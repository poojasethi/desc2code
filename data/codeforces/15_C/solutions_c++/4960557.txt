#include <stdio.h>


typedef long long ll;

inline ll func(ll x)
{
    ll res=0;
    for(ll i=x-x%4;i<=x;++i)
        res^=i;
    return res;
}

int main()
{
    int n;
    ll m,x;
    scanf("%d",&n);
    ll ans=0;
    for(int i=0;i<n;++i)
    {
        scanf("%lld%lld",&x,&m
              );
        ll d = func(x-1);
        ans = ans^(d^func(m+x-1));
    }
    if (ans) puts("tolik");
    else puts("bolik");
    return 0;
}
