#include<iostream>
using namespace std;
typedef long long LL;
LL ans,n,x,m;

LL calc(LL x)
{
	LL ret=0;
	for (LL i=2,j=0,t;j<60;j++,i<<=1)
	{
		if (x%(i<<1)>i) t=x%(i<<1)-i;
		else t=0;
		if (t&1) ret|=i;
	}
	if (x%4>1) ret++;
	return ret;
}

int main()
{
	cin>>n;
	for (int i=0;i<n;i++)
	{
		cin>>x>>m;
		ans^=calc(x)^calc(x+m);
	}
	cout<<(ans?"tolik":"bolik")<<endl;
	return 0;
}
