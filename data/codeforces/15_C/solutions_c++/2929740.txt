#include <iostream>
using namespace std;
long long go(long long n) {
	if (n<=(long long)1) return n;
	if(n%2) return (n+1)/2%2;
	return n^(n/2%2);
}

int main(void) {
	int n;
	while(cin>>n)
	{long long ans = 0;
    while(n--) {
		long long x, y;
		cin>>x>>y;
		ans ^= go(x+y-1) ^ go(x-1);
    }
    if(ans)cout<<"tolik\n";
    else cout<<"bolik\n";
	}
	return 0;
}
