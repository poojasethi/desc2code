#include <cstdio>
#include <iostream>

using namespace std;

typedef long long int Lint;

int N,ans;

int f( Lint k ){
	
	if( k%4==0 )
		return k;
	if( k%4==1 )
		return 1;
	if( k%4==2 )
		return k+1;
	return 0;
}

int main(){
	
	scanf(" %d",&N);
	
	Lint a,b;
	
	for( int i=1 ; i<=N ; i++ ){
		scanf(" %I64d %I64d",&a,&b);
		ans^=f(a+b-1LL)^f(a-1LL);
	}
	
	printf( ans ? "tolik\n" : "bolik\n" );
	return 0;
}
