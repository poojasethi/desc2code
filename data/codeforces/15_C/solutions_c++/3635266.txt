#include<iostream>
using namespace std;
typedef long long LL;
LL ans,n,x,m;
LL calc(LL x)
{
	LL ret=x%4>1;
	for (LL i=2,t;i<=x;i<<=1)
		if (max(0ll,x%(i<<1)-i)&1) ret|=i;
	return ret;
}
int main()
{
	cin>>n;
	for (LL i=0;i<n;i++)
	{
		cin>>x>>m;
		ans^=calc(x)^calc(x+m);
	}
	cout<<(ans?"tolik":"bolik")<<endl;
	return 0;
}
