#include <iostream>
#include <cstdio>
using namespace std;

int N;
long long int X,M,ans,tmp;

int main(){
  scanf("%d",&N);
  for(int i=0;i<N;++i){
    cin >> X >> M;
    tmp = X+M-1ll;
    while(tmp%4ll != 3ll && tmp > 0){
      ans ^= (tmp--);
    }
    tmp = X-1ll;
    while(tmp%4ll != 3ll && tmp > 0){
      ans ^= (tmp--);
    }
  }
  printf("%s\n",(ans == 0)?"bolik":"tolik");
}
