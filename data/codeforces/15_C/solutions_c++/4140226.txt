#include <cstdio>
#include <iostream>

using namespace std;

int n;
long long ans;

long long fin(long long a)
{
    return ( a% 2)*a-(a % 2)+(a % 4) / 2;
}
int main()
{
    long long x = 0;
    long long y = 0;
    ans = 0;
    for ( cin>> n; n--;)
    {
        cin>> x>> y;
        ans ^= fin(x) ^ fin(x+y);
    }
    puts( ans ? "tolik" : "bolik" );
    return 0;
}
