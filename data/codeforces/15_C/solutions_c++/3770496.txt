#include <iostream>

using namespace std;

long long int f(long long int a)
{
        long long res[] = {a,1,a+1,0};
        return res[a%4];
}

int main()
{
    int n;

    long long int x, y;

    cin >> n;
    int ans = 0;
    for(int i = 0; i < n; i++) {
        cin >> x >> y;
        ans ^= f(x+y-1)^f(x-1);
    }

    if(ans) {
        cout << "tolik\n";
    }
    else {
        cout << "bolik\n";
    }
}
