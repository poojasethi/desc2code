#include <stdio.h>

typedef long long LL;

LL go(LL n) {
	if (n<=1LL) return n;
	if(n%2) return (n+1)/2%2;
	return n^(n/2%2);
}

int main(void) {
	int n;
	scanf("%d", &n);
	LL ans = 0;
	while(n--) {
		LL x, y;
		scanf("%I64d%I64d", &x, &y);
		ans ^= go(x+y-1) ^ go(x-1);
	}
	puts(ans? "tolik": "bolik");
	return 0;
}
