#include <iostream>
using namespace std;
#define LL unsigned long long

int main(){
    int n;
    LL x,m,ans=0ll;
    cin >>n;
    for(int i=0;i<n;i++){
        cin >>x>>m;
        m=x+m-1ll;
        for(;x%4    && x<=m;x++) ans^=x;
        for(;m%4!=3 && m>=x;m--) ans^=m;
    }
    cout <<(ans?"tolik":"bolik")<<endl;
    return 0;
}
