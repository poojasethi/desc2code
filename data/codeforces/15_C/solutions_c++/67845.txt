#include <iostream>
#include <cstdio>

using namespace std;

#define getBit(x) (((x)&1)?(1-(((x)>>1)&1)):((x)+(((x)>>1)&1)))

int main()
{
	int n;
	cin >> n;
	long long bit = 0, x, m;
	for (int i = 0; i < n; ++i)
	{
		cin >> x >> m;
		bit ^= getBit(x - 1) ^ getBit(x + m - 1);
	}
	if (bit) printf("tolik\n"); else printf("bolik\n");
}
