#include <cstdio>
typedef long long ll;
#define i64 "%I64d"

ll xorf (ll r){
	if (r % 4 == 1)
		return 1;
	if (r % 4 == 2)
		return r + 1;
	if (r % 4)
		return 0;
	return r;
}

int n;
ll ans, x, m;
int main ()
{
	scanf("%d", &n);
	for (int i = 0; i < n; i++){
		scanf(i64 i64, &x, &m);
		ans ^= xorf(x - 1);
		ans ^= xorf(x + m - 1);
	}
	puts(ans ? "tolik" : "bolik");
	return 0;
}
