#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
LL cal(LL x)
{
    if(x%4==0) return x;
    if(x%4==1) return 1;
    if(x%4==2) return x+1;
    return 0;
}
int main()
{
    LL ret=0;
    LL n,m,x;
    scanf("%lld",&n);
    while(n--)
    {
        scanf("%lld%lld",&x,&m);
        ret^=(cal(x-1)^cal(x+m-1));
    }
    if(ret==0)
        puts("bolik");
    else
        puts("tolik");
    return 0;
}
