# include <stdio.h>
int main()
{
	long long int n,ans=0,k,x,temp;
	scanf("%lld",&n);
	for (int i = 0; i < n; ++i)
	{
		// printf("Y\n");
		scanf("%lld %lld",&x,&k);
		long long int temp1=(x-1 >> 1) & 1 ^ (x-1&1 ? 1 : x-1);
		x=x+k;
		long long int temp2=(x-1 >> 1) & 1 ^ (x-1&1 ? 1 : x-1);
		// printf("%lld %lld\n",temp1,temp2);
		temp2=temp1^temp2;
		ans=ans^temp2;
	}
	if(ans==0)
		printf("bolik\n");
	else
		printf("tolik\n");
}
