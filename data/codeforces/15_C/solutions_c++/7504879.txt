#include <cstdio>
using namespace std;
int n;
long long x,m;
long long ans=0;
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%I64d%I64d",&x,&m);
		long long L=x,R=x+m-1;
		if (L&1) ans^=L,m--;
		if ((R&1)==0) ans^=R,m--;
		if ((m>>1)&1) ans^=1;
	}
	if (ans) puts("tolik");
	else puts("bolik");
	return 0;
}
