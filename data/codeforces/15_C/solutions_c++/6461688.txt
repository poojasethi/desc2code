#include <iostream>
#include <stdio.h>

using namespace std;

long long i,res,n,a,b;
long long f(long long x)
 {
  if (x%4 == 0) return x;
  if (x%4 == 1) return 1;
  if (x%4 == 2) return x+1;
  if (x%4 == 3) return 0;
 }

main()
 {
  cin>>n;
  for (i=1;i<=n;i++)
   {
    cin>>a>>b;
    res^=f(a+b-1)^f(a-1);
   }
  if (res) cout<<"tolik"<<endl; else
           cout<<"bolik"<<endl;
 }
