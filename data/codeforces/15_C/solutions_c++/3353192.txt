#include <cstdio>
#include <iostream>

using namespace std;

typedef long long Lint;

int N;
Lint res;

Lint f( Lint a ){
	switch( a%4 ){
		case 0: return a;
		case 1: return 1;
		case 2:	return a+1LL;
		case 3: return 0;
	}
}

int main()
{
	scanf(" %d",&N);
	
	Lint a,b;
	
	for( int i=0 ; i<N ; i++ ){
		cin >> a >> b;
		res ^= f(a+b-1LL)^f(a-1LL);
	}
	
	if( !res )	cout << "bolik" << endl;
	else	cout << "tolik" << endl;
	
	return 0;
}
