def xor(n):
    x = (n+1)&3
    if x == 0 : return 0
    if x == 1 : return n
    if x == 2 : return 1
    return n+1
n = int(raw_input())
ans = 0 
while n>0:
    n-=1
    x,m= map(int,raw_input().split())
    ans = ans ^ xor(x-1) ^ xor(x-1+m)
if ans ==0 : 
    print "bolik"
else :
    print "tolik"
