







def value(x):
	if x%4 == 0:
		return x
	if x%4 == 1:
		return 1
	if x%4 == 2:
		return x+1
	if x%4 == 3:
		return 0


n = input()
xorvalue = 0
while n:
	n-=1
	x,m = map(int,raw_input().split(" "))
	xorvalue = xorvalue ^ value(x+m-1) ^ value(x-1)

if xorvalue:
	print("tolik")
else:
	print("bolik")
