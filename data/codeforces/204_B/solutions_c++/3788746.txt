#include <cstdio>
#include <iostream>
#include <map>

#define fi first
#define se second

using namespace std;

typedef pair<int,int> ii;

int N,res=1e8;

map<int,ii> cnt;

int main(){
	
	scanf(" %d",&N);
	
	for( int a,b,i=1 ; i<=N ; i++ ){
		scanf(" %d %d",&a,&b);
		cnt[a].fi++;
		if( a!=b )
			cnt[b].se++;
	}
	
	for( map<int,ii>::iterator it=cnt.begin() ; it!=cnt.end() ; it++ )
		if( it->se.fi+it->se.se>=(N+1)/2 )
			res=min(res,max(((N+1)/2)-it->se.fi,0));
	
	if( res!=1e8 )
		printf("%d\n",res);
	else
		printf("-1\n");
	return 0;
}
