#include<stdio.h>
#include<stdlib.h>
#include<map>
using namespace std;
struct card{
	int up,sum;
};
map<int,card> m;
int n;
int main()
{
	int i,j,tmp,a,b,min;
	scanf("%d",&n);
	tmp=(n+1)/2;
	min=tmp+1;
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&a,&b);
		m[a].up++;
		m[a].sum++;
		if(a!=b)
			m[b].sum++;
		if(m[a].sum>=tmp&&tmp-m[a].up<min)
		{
			min=tmp-m[a].up;
		}
		if(m[b].sum>=tmp&&tmp-m[b].up<min)
		{
			min=tmp-m[b].up;
		}
	}
	if(min<0)
		min=0;
	if(min!=tmp+1)
		printf("%d\n",min);
	else
		printf("-1\n");
	return 0;
}
