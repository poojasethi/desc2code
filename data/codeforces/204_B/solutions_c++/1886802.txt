#include <iostream>
#include <map>
#include <set>

using namespace std;

int main()
{
  map<int,int> veces,debajo;
  set<int> s;
  int n;
  cin>>n;
  for (int i=0;i<n;i++) {
    int x,y;
    cin>>x>>y;
    s.insert(x);
    s.insert(y);
    veces[x]++;
    if (x!=y) debajo[y]++;
  }
  int minimo=1000000000;
  for (set<int>::iterator it=s.begin();it!=s.end();it++) {
    int x=*it;
    if (2*veces[x]>=n) minimo=0;
    else if (2*(veces[x]+debajo[x])>=n)
      minimo=min(minimo,(n+1)/2-veces[x]);
  }
  if (minimo==1000000000)
    cout<<-1<<endl;
  else
    cout<<minimo<<endl;
}
