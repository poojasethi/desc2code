#include<iostream>
#include<map>
#include<utility>
using namespace std;
int n,temp1,temp2;
map<int,int> p1,p2;

int main(){
	cin>>n;
	int m=(n+1)/2;
	int max=-1,i=0;
	bool a=false;
	for(int i=0;i<n;i++){
		cin>>temp1>>temp2;
		p1[temp1]++;
		if(temp2!=temp1)
			p2[temp2]++;
		if((p1[temp1]+p2[temp1])>=m && p1[temp1]>max){
			max=p1[temp1];
		}
		if((p1[temp2]+p2[temp2])>=m && p1[temp2]>max){
			max=p1[temp2];
		}
	}
	if(max==-1)
		cout<<"-1"<<endl;
	else if(max>=m)
		cout<<"0"<<endl;
	else 
		cout<<(m-max)<<endl;
	return 0;
}
