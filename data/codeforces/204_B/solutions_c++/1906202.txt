#include<iostream>
#include<map>
#include<vector>
#include<algorithm>
using namespace std;
map<int,int>p,b;
vector<int>r;
int main()
{
   int n;
   cin>>n;
   int px,bx;
   for(int i=0;i<n;i++)
   {
     cin>>px>>bx;
     p[px]++;
     if(px!=bx)
        b[bx]++; 
     r.push_back(px);
     r.push_back(bx);
   }
   (n+=1)>>=1;
   int res=n+1;
   for(int i=0;i<r.size();i++)
   {
      if(p[r[i]]+b[r[i]]>=n) 
         res=min(n-p[r[i]],res);
   }
   if(res>n) cout<<"-1"<<endl;
   else cout<<max(res,0)<<endl;
   return 0;
}
