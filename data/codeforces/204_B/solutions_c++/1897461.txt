#include<iostream>
#include<map>

using namespace std;

int main()
{

	map<int,int> b,f;
	int n,u[100001]={},v[100001]={},ans;
	cin>>n;
	for(int i = 0 ; i < n ; ++i )
	{
		cin>>u[i]>>v[i];
		b[u[i]]++;
		if(u[i] != v[i] )
		{
			b[v[i]]++;
		}
		f[u[i]]++;
	}

	ans = n+1;
	for(int i = 0 ; i < n ; ++i )
	{
		if(b[u[i]]*2 >= n)
		{
			ans = min( ans , max( (n+1)/2 - f[u[i]] , 0 ) );
		}
		if(b[v[i]]*2 >= n)
		{
			ans = min( ans , max( (n+1)/2 - f[v[i]] , 0 ) );
		}
	}

	if(ans == n+1)
		cout<<-1<<endl;
	else
		cout<<ans<<endl;





	
}
