#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<map>
#define F first
#define S second
using namespace std;
map <int,pair<int,int> > q;
int i,j,k,n,m,x,y,ans=10000000;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i){
		scanf("%d%d",&x,&y);
		q[x].F++;
		if(q[x].F+q[x].S>=(n+1)/2)
			ans=min(ans,(n+1)/2-q[x].F);
		if(x==y)continue;
		q[y].S++;
		if(q[y].F+q[y].S>=(n+1)/2)
			ans=min(ans,(n+1)/2-q[y].F);
	}
	printf("%d\n",ans==10000000?-1:max(0,ans));
}
