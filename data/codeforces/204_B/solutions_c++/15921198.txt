#include <bits/stdc++.h>

using namespace std;

int n , a[111111] , b[111111] , ans ;
map<int,int>up,dn;

void solve(int x){
    int UP=up[x] , DN=dn[x] ;

    if( DN+UP <(n+1)/2 )return;

ans=min(ans,  max( ( (n+1)/2 )-UP , 0) );
}

int main()
{
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>a[i]>>b[i];
        up[a[i]]++;
        if( a[i]!=b[i] )dn[ b[i] ]++;
    }
ans=999999999;

    for(int i=0;i<n;i++){
        solve(a[i]);
        solve(b[i]);
    }
    cout<<((ans==999999999)?-1:ans)<<endl;

    return 0;
}
