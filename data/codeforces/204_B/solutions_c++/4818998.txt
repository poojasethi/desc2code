#include <iostream>
#include <map>
#include <stdio.h>
using namespace std;
int main()
{//freopen("i.txt","r",stdin);
int n; cin>>n;
int kol=(n+1)/2;
map<int,int> a,b;
for (int i=0;i<n;i++) {int x,y; cin>>x>>y; a[x]++; if (x!=y) b[y]++;}
map<int, int>::iterator it;
int p=100000;
for (it=a.begin();it!=a.end();it++)
{int i1=it->first;
 if (a[i1]>=kol) p=0;
  else if (a[i1]+b[i1]>=kol) p=min(p,kol-a[i1]);
}
for (it=b.begin();it!=b.end();it++)
{int i1=it->first;
 if (a[i1]==0&&b[i1]>=kol) p=min(p,kol);
}
if (p!=100000) cout<<p; else cout<<-1;
return 0;
}
