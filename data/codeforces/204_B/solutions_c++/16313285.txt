#include<bits/stdc++.h>

using namespace std;

map < int , int > M, N;

vector < int > v;

int n,i,a,b,mn = 1000000009;

int main(){

	scanf("%d",&n);

	for(i=1;i<=n;i++){
		scanf("%d%d",&a,&b);

		N[a]++;
		if(a != b)
			N[b]++;

		M[a]++;

		if(N[a] == n/2 + (n&1))
			v.push_back(a);

		if(N[b] == n/2 + (n&1))
			v.push_back(b);
	}

	for(i=0;i<v.size();i++)
		mn = min(mn , max(0 , n/2 + (n&1) - M[ v[i] ]));


	printf("%d",mn == 1000000009 ? -1 : mn);
	return 0;
}
