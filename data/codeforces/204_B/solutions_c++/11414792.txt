#include <iostream>
#include <vector>
#include <string>
#include <map>
#include <algorithm>
using namespace std;
map<int,int> f,b;
int N;
const int INFINITY=1000*1000*1000;
int main()
{
    int res=INFINITY;
    cin>>N;
    int m=(N+1)/2;
    for(int i=0;i<N;i++)
    {
        int a,c;
        cin>>a>>c;
        f[a]++;
        if(c!=a)
            b[c]++;
        if(m<=f[a]+b[a])
            res=min(res,m-f[a]);
        if(m<=f[c]+b[c])
            res=min(res,m-f[c]);
    }
    if(res==INFINITY)
    {
        cout<<-1;
        return 0;
    }
    cout<<max(res,0);
}
