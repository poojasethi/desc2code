#include <algorithm>
#include <iostream>
#include <map>
using namespace std;
int N, a, b;
int main() {
    while (cin >> N) {
	typedef map<int,pair<int,int> > map_t;
	map_t C;
	for (int i=0; i<N; ++i) {
	    cin >> a >> b;
	    C[a].first++;
	    C[a].second++;
	    if (a != b)
		C[b].first++;
	}
	int inf = N*2, n = inf;
	for (map_t::iterator p=C.begin(); p!=C.end(); ++p) {
	    if (p->second.first*2 >= N) {
		int t = p->second.second,
		    s = (t*2 < N) ? (N-t*2+1)/2 : 0;
		n = min(n, s);
	    }
	}
	cout << (n==inf ? -1 : n) << endl;
    }
}
