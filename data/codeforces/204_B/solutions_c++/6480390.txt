#include <cstdio>
#include <algorithm> 
#include <map>
using namespace std;

int n , x[100001] , y[100001] , ans=(1<<29);
map<int,int> cnt ;
map<int,int>::iterator it ; 

int main()
{
   scanf("%d",&n) ; 
   for(int i=1 ; i<=n  && scanf("%d%d",&x[i],&y[i]) ; i++) { cnt[x[i]]++ ;  if(x[i]!=y[i]) cnt[y[i]]++ ; }  
   for( it=cnt.begin() ; it!=cnt.end() ; it++) if(it->second >= (n+1)/2 ) 
   {
       int a=0 ;  
       for(int i=1 ; i<=n ; i++) if(x[i]==it->first) a++ ;
       ans=min(ans,max(0,(n+1)/2-a)) ; 
   }
 
   printf((ans<(1<<29))?"%d":"-1",ans) ;
   
 return 0;
}
