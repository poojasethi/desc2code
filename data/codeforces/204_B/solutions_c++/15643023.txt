#include<bits/stdc++.h>
using namespace std;
#define N 1111111
#define ii pair<int,int>
#define f first
#define s second
map<int,int> mp1,mp2;
int a[211111];
int main(){
	int n,b;
	cin>>n;
	int k=(n+1)/2;
	for(int i=0;i<2*n;i+=2){
		cin>>a[i]>>a[i+1];
		if(a[i]!=a[i+1]) mp2[a[i+1]]++;
		mp1[a[i]]++;
	}
	int mn=N;
	for(int i=0;i<2*n;i++){
		int m=mp1[a[i]];
		int m1=mp2[a[i]];
		if(m1+m>=k) {
			if(m>=k){ mn=0;break;}
			else mn=min(mn,k-m);
		}
	}
	if(mn==N) cout<<"-1"<<endl;
	else cout<<mn<<endl;
}
