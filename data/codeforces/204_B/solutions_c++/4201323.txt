#include <iostream>
#include <map>

using namespace std;

int cards[200005], front[200005];
map <int, int> m;

int main()
{
	int n;
	cin >> n;
	int maxid = 0;
	for (int i = 0; i < n; i++)
	{
		int a, b;
		cin >> a >> b;
		if (m.find(a) == m.end())
			m[a] = maxid++;
		if (m.find(b) == m.end())
			m[b] = maxid++;
		cards[m[a]]++;
		front[m[a]]++;
		if (a != b)
			cards[m[b]]++;
	}
	int ans = -1;
	int half = n / 2;
	if (n % 2 != 0)
		half++;
	for (int i = 0; i < maxid; i++)
		if (half <= cards[i])
			if (ans == -1 || ans > max(half - front[i], 0))
				ans = max(0, half - front[i]);
	cout << ans << endl;
}
