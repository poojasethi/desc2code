#include<bits/stdc++.h>
using namespace std;
int main()
{
    map<int ,int> a,b;
    int n;
    cin >>n;
    int mid = (n+1)/2;
    int x,y;

    for(int i=0;i<n;i++)
    {
        cin >> x>>y;
        a[x]++;  b[x]++;
        if(x!=y)
            a[y]++;
    }
    map<int,int>::iterator it;
    int ans = 999999999;

    for(it = a.begin(); it != a.end(); it++)
        if(it->second >= mid)
        {
            ans = min(ans, max(0, mid - b[it->first]));
        }

    if(ans == 999999999)
        ans =-1;
    cout << ans<< endl;
}
