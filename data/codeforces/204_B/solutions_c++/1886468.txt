#include <stdio.h>
#include <cstdlib>
#include <algorithm>
#include <map>
using namespace std;

map <int,int> c,u;
int x[100000+10],y[100000+10],n,i,res;

main()
{
 scanf("%d",&n);
 for ( i=1;i<=n;i++ )
  {
   scanf("%d%d",&x[i],&y[i]);
   c[x[i]]++;
   if ( y[i]!=x[i] ) c[y[i]]++;
   u[x[i]]++;
  }
 res=n+1;
 for ( i=1;i<=n;i++ )
  {
   if ( c[x[i]]*2>=n ) res=min(res,max((n+1)/2-u[x[i]],0));
   if ( c[y[i]]*2>=n ) res=min(res,max((n+1)/2-u[y[i]],0));
  }
 if ( res==n+1 ) res=-1;
 printf("%d\n",res);
}
