#include <cstdio>
#include <cstring>
#include <map>
#include <algorithm>

using namespace std;

int main() {
	int n, m, a, b;
	map<int, int> front, all;
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d%d", &a, &b);
		++front[a];
		++all[a];
		if (a != b) {
			++all[b];
		}
	}
	m = (n + 1) / 2;
	int res = 0x3f3f3f3f;
	for (map<int, int>::const_iterator i = all.begin(); i != all.end(); ++i) {
		if (i->second >= m) {
			int t = front[i->first];
			res = min(res, t >= m ? 0 : m - t);
		}
	}
	if (res == 0x3f3f3f3f) {
		res = -1;
	}
	printf("%d\n", res);
	return 0;
}
