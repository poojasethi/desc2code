#include <cstdio>
#include <map>
#include <set>

using namespace std;

map<int, int> up;
map<int, int> down;
set<int> col;

int main()
{
	int n, a, b, res;
	scanf("%d", &n);
	res = n+1;
	
	for (int i = 0; i < n; ++i)
	{
		scanf("%d%d", &a, &b);
		up[a]++;
		if (a != b) down[b]++;			
		col.insert(a);
		col.insert(b);
	}
	
	int l = n/2 + (n%2);
	
	for (typeof(col.begin()) it = col.begin(); it != col.end(); ++it)
	{
		int c = *it;
		if (up[c]+down[c] < l) continue;
		res = min(res, max(0, (l-up[c])));
	}
	
	if (res == n+1) printf("-1\n");
	else printf("%d\n", res);
	
	return 0;
}
