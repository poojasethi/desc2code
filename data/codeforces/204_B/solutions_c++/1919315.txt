#include <iostream>
#include<algorithm>
#include<map>
using namespace std;
map<int,int>a,b;

int m=1000000001,n;

int main()
{
   int j,x,y;cin>>n;
   for(j=0;j<n;j++)
   {
      cin>>x>>y;a[x]++;a[y]++;b[x]++;
      if(x==y)a[x]--;
   }
   map<int,int>::iterator i;
   
   for(i=a.begin();i!=a.end();i++)
   {
       if(i->second>=(n+1)/2)
       {if(b[i->first]>(n+1)/2) m=0;
       else
       m=min(m,(n+1)/2-b[i->first]);
   }
   }
   if(m==1000000001) cout<<-1;
   else cout<<m;

    return 0;
}
