#include<iostream>
#include<cstdio>
using namespace std;
int n,i,x,y,ma,ans,a[11000000],b[11000000],ha[11000000];
int hash(int k)
{
	int p=k%999777;
	while (ha[p]&&ha[p]!=k)	p++;
	ha[p]=k;
	ma=max(ma,p);
	return p;
}
int main()
{
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	{
		scanf("%d%d",&x,&y);
		a[hash(x)]++;
		if (x!=y)
		b[hash(y)]++;
	}
	ans=2000000000;
	for (i=0;i<=ma;i++)
	if (a[i]+b[i]>=(n+1)/2&&(n+1)/2-a[i]<ans)
		ans=(n+1)/2-a[i];
	if (ans==2000000000)
		printf("-1");
	else
		printf("%d",max(0,ans));
}
