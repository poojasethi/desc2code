#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int x[n], y[n];
	map<int, int>A;
	map<int, int>B;
	for (int i=0; i<n; i++)
	{
		cin>>x[i]>>y[i];
		A[x[i]]++;
		if (x[i]!=y[i])
		B[y[i]]++;
	}
	int ans=1e9+228;
	int q=(n+1)/2;
	for (int i=0; i<n; i++)
	{
		if (A[x[i]]>=q)
		{
			ans=0;
		}
		else if (q-A[x[i]]<=B[x[i]])
		{
			ans=min(ans, q-A[x[i]]);
		}
	}
	if (ans==1e9+228)
	{
		for (int i=0; i<n; i++)
		{
			if (B[y[i]]>=q)
			{
				cout<<q;
				return 0;
			}
		}
		cout<<-1;
		return 0;
	}
	cout<<ans;
	return 0;
}
