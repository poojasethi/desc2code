#include<stdio.h>
#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
#define F first
#define S second
map<int,pair<int, int> >m;
int n,i,a,b,ans=10000000;
int main()
{
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d%d",&a,&b);
        m[a].F++;
        if(m[a].F+m[a].S>=(n+1)/2)
        ans=min(ans,((n+1)/2)-m[a].F);
        if(a==b)
        continue;
        m[b].S++;
        if(m[b].F+m[b].S>=(n+1)/2)
        ans=min(ans,((n+1)/2)-m[b].F);
        //cout<<ans<<endl;
    }
    printf("%d\n",ans==10000000?-1 : max(0,ans));

}
