#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    map <int  ,int> a,b;
    int  n, x, y, ans,mid;
    cin>>n; 
    mid = (n+1)/2;
    for(int  i = 0; i<n ;i++)
    {
        cin>>x>>y;
        a[x]++;b[x]++;
        if(x!=y)
        a[y]++; 
    
    }
    ans = 999999;
    map<int,int>::const_iterator it;
    for( it = a.begin(); it!=a.end(); it++)
    {
        if(it->second>=mid)
        {
        ans = min(ans,max(0,mid-b[it->first]));
        
        }
    }
    if(ans == 999999)
    cout<<"-1";
    else
    cout<<ans;
    
}
