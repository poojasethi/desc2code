#include<map>
#include<cstdio>
#include<algorithm>
#define N 100010
#define INF 999999999
using namespace std;
int n, x, y, ans = INF;
map<int,int> a, b;
int main(){
	scanf("%d", &n);
	int nd = (n+1)/2;
	for(int i = 0; i < n; i++){
		scanf("%d%d", &x, &y);
		a[x]++, b[x]++;
		if(x != y) a[y]++;
	}
	for(map<int,int>::iterator it = a.begin();it != a.end(); it++)
		if(it->second >= nd) ans = min(ans, max(0, nd-b[it->first]));
	printf("%d\n", ans==INF?-1:ans);
}
