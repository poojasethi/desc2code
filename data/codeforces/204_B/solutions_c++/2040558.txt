#include<iostream>
#include<cstdio>
#include<map>
#include<climits>
#include<algorithm>
using namespace std;
int main()
{
int a,b,n;
cin>>n;
map<int,int> m1,m2;
map<int,int>::iterator it;
	for(int i=0;i<n;++i)
	{
		cin>>a>>b;
		m1[a]++;
		m2[a]++;
		if(a!=b)
		m2[b]++;
	
	}

	int ans=INT_MAX;

	int mid=(n+1)/2;

	for(it=m2.begin();it!=m2.end();it++)
	{
		int val1=m1[it->first];
		if(it->second>=mid)
		{
			if(val1>=mid)
			ans=0;
			else
			ans=min(mid-val1,ans);
		}
	}
if(ans==INT_MAX)
cout<<"-1"<<endl;
else
cout<<ans<<endl;
return 0;
}
