#include <cstdio>
#include <set>
#include <algorithm>
using namespace std;
int i,j,m,n,o;
multiset<int> up,down;
set<int> color;
void odp(int c) {
  int a=up.count(c),b=down.count(c);
  if(a+b>=n) {
    if(a>=n) m=0;
    else m=min(m,n-a);
  }
}
int main() {
  scanf("%d",&n);
  for(i=0;i<n;i++) {
    scanf("%d%d",&m,&j);
    up.insert(m),color.insert(m);
    if(m!=j)down.insert(j),color.insert(j);
  }
  m=1e9,n=n+1>>1;
  for_each(color.begin(),color.end(),odp);
  if(m==1e9) printf("-1\n");
  else printf("%d\n",m);
  return 0;
}
