#include<cstdio>
#include<map>
typedef std::pair<int,int> pr;
int main(){
	int n;
	while(~scanf("%d",&n)){
		std::map<int,pr>mp;
		std::map<int,pr>::iterator it;
		int i;
		int f,b;
		for(i=0;i<n;++i){
			scanf("%d%d",&f,&b);
			mp[f].first++;
			if(b!=f)mp[b].second++;
		}
		int min=0x7fffffff;
		int cur;
		for(it=mp.begin();it!=mp.end();++it){
			if((it->second.first+it->second.second)*2>=n){
				cur=n/2+(n&1)-it->second.first;
				if(cur<min)min=cur;
			}
			if(min<=0)break;
		}
		if(min<0)min=0;
		if(min==0x7fffffff)min=-1;
		printf("%d\n",min);
	}
	return 0;
}
