n=int(raw_input())
a=[]
for i in xrange(n):
    a.append(map(int,raw_input().split()))
dd,dd2={},{}
z=set()
for i,j in a:
    dd[i]=dd.get(i,0)+1
    if i!=j:
        dd2[j]=dd2.get(j,0)+1   
    z.add(i)
    z.add(j)
x=(n+1)/2
mm=-1
for i in z:
#    print i,dd.get(i,0),dd2.get(i,0),x
    if dd.get(i,0)+dd2.get(i,0)>=x:
        tre=x-dd.get(i,0)
        if tre<0:
            tre=0
        if (mm==-1)or(mm>tre):
            mm=tre    
print mm
