#!/usr/bin/env python
n = int(raw_input())
m = (n+1) // 2
colors = set()
free = {}
paid = {}
for i in xrange(n):
	a,b = map(int,raw_input().split())
	colors.add(a)
	colors.add(b)
	free[a] = free.get(a,0) + 1
	if a != b:
		paid[b] = paid.get(b,0) + 1
mi = m+1
for color in colors:
	mm = max(0,m - free.get(color,0))
	if mm == 0:
		mi = 0
	elif color in paid and paid[color] >= mm:
		mi = min(mi,mm)
print mi if mi <= m else -1
