#!/usr/bin/python



def add_map(sums,a):
        if a in sums:
                sums[a] += 1
        else :
                sums[a] = 1


def main():
        n = int(raw_input())
        front = {}
        back = {}
        sums = {}
        data = []
        for i in range(n):
                a,b = map(int,raw_input().split())
                add_map(front,a)
                add_map(back,b)
                if(a == b):
                        add_map(sums,a)
                else:
                        add_map(sums,b)
                        add_map(sums,a)

                data.append((a,b))

        ans = n + 1
        count = (n + 1) // 2
        for i in range(n):
                if(front[data[i][0]] >= count):
                        ans = 0
                        break
                else:
                        if(data[i][0] in back and sums[data[i][0]] >= count):
                                ans = min(ans,count - front[data[i][0]])
        for i in range(n):
                if data[i][1] not in front:
                        if(back[data[i][1]] >= count):
                                ans = min(ans,count)
        if ans == n + 1:
                ans = -1
        print ans

if __name__ == '__main__':
        main()
