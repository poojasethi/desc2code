n = int(raw_input())
m = (n + 1) / 2
a = {} 
b = {}
s = set()

for i in xrange(n):
	x, y = map(int, raw_input().split())
	s.add(x)
	s.add(y)
	a[x] = a.get(x, 0) + 1
	if x != y: b[y] = b.get(y, 0) + 1

r = n + 1
for i in s:
	if a.get(i, 0) + b.get(i, 0) >= m:
		r = min(r, m - a.get(i, 0))
r = max(r, 0)
if r > n: r = -1
print r
