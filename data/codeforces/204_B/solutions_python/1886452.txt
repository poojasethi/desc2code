import sys
n = int(sys.stdin.readline())
a = dict()
b = dict()
c = set()
m = (n+1)/2
for i in xrange(n):
	x, y = map(int, sys.stdin.readline().split())
	if x != y: a[y] = a.get(y, 0) + 1
	b[x] = b.get(x, 0) + 1
	c.add(x); c.add(y)
ans = n + 10
for k in c:
	if a.get(k, 0) + b.get(k, 0) >= m:
		ans = min(ans, (m - b.get(k, 0)))
ans = max(ans, 0)
if ans > n: ans = -1
print ans
