#!/usr/bin/env python
# -*- coding: utf-8 -*-
import sys

def f2(s):
    for i in xrange(len(s)):
        yield s[i:] + s[:i]

def t(s):
    for i, c in enumerate(s):
        if c is 'X':
            return (i+1)%2

#sum([t(s) for s in f2('..XXX')])

n, k, p = map(int, raw_input().split())
for lp in (n-int(x) for x in  sys.stdin.readlines()):
    ln = n
    lk = k
    if n%2:
        if lp == 0:
            sys.stdout.write('X' if k >= 1 else '.')
            continue
        ln -= 1
        lk -= 1
        lp -= 1
    if lk == 0:
        sys.stdout.write('.')
    else:
        if ln >= lk*2:
            sys.stdout.write('X' if lp%2 == 0 and lp/2+1 <= lk else '.')
        else:
            if lp%2 == 0:
                sys.stdout.write('X')
            else:
                lk -= ln/2
                sys.stdout.write('X' if (lp+1)/2 <= lk else '.')
sys.stdout.write('\n')
