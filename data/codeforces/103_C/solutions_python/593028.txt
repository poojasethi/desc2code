#!/usr/bin/python
import itertools, math, os, random, re, sys
random.seed (1234)

while True:
	s = sys.stdin.readline ().strip ()
	if s == '':
		s = sys.stdin.readline ().strip ()
		if s == '':
			break
	n, k, p = [int (x) for x in s.split ()]
	m = k * 2
	if n < m:
		lo = 0
		hi = n * 2 - m
	elif n % 2 == 0 or k == 0:
		lo = n - m
		hi = n
	else:
		lo = n - m + 1
		hi = n - 1
#	print lo, hi
	s = ''
	for i in range (p):
		j = int (sys.stdin.readline ())
		j -= 1
		if (j < lo) or ((j < hi) and ((j - lo) % 2 == 0)):
			s += '.'
		else:
			s += 'X'
	print s
