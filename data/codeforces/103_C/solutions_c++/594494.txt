#include <stdio.h>
#include <iostream>
#define LL long long
using namespace std;

LL calc(LL n, LL p)
{
    if (n&1) {
        if (p == n) return 1;
        if (p&1) return n - p/2;
        return n/2 - p/2 + 2;
    }
    if (p&1) return n - p/2;
    return n/2 - p/2 + 1;
}

int main()
{
    LL n, k, p;
    int q;
    while (cin >> n >> k >> q) {
        for (int i = 0; i < q; ++i) {
            cin >> p;
            if (calc(n, p) <= k) putchar('X');
            else putchar('.');
        }
        puts("");
    }
    return 0;
}
