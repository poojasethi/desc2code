#include <bits/stdc++.h>
using namespace std;

long long n,k;
int t;


int main()
{
    ios::sync_with_stdio(0);
    cin >> n >> k >> t;
    long long d =0;
    long long p = min(k,n-k);
    long long c =0;
    if(k>n-k) c = k+k-n;
    else d = n - k- k;
    if(d%2 && p) p--, d++ ,c++;
    while(t--)
    {
        long long x;
        cin >> x;
        if(x<=d) cout << ".";
        else if(x<=d+p*2)
        {
            if((x-d)%2) cout <<".";
            else cout << "X";
        } else cout << "X";
    }


}
