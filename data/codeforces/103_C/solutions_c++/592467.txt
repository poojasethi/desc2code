#include <cstdio>
using namespace std;
typedef long long LL;
char solve(LL n,LL k,LL x)
{
    if(k==0) return '.';
    if(x==n) return 'X';
    if(n&1LL)
    {
        return solve(n-1,k-1,x);
    }
    LL y=n/2-(x+1)/2;
    if(x&1)
    {
        if(y<k-n/2) return 'X'; else return '.';
    }
    else
    {
        if(y<k) return 'X'; else return '.';
    }
}
int main()
{
    LL n,k;
    int p;
    scanf("%I64d%I64d%d\n",&n,&k,&p);
    for(int i=0;i<p;i++)
    {
        LL x;
        scanf("%I64d",&x);
        putchar(solve(n,k,x));
    }
    puts("");
    return 0;
}
