
#include<stdio.h>

using namespace std;
long long n , m , k;
bool IS(long long x)
{
if(n %2 == 1)
{
  if(x % 2 == 1)
   {
     if(x == n)return k == 0;
     return n-k >= x/2+1;
    }
 return n-k >= (x/2 + n / 2);
}

if(x%2==1)
return n-k >= x/2+1;

return n-k >= (x/2 + n / 2);
}

int main()
{
    cin >> n >> k >> m;
    
    while(m--){
      long long x;
      cin >> x;
      if(IS(x))printf(".");
      else printf("X");
    }
    printf("\n");
 return 0;
}
