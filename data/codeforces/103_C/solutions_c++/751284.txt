#include <fstream>
#include <iostream>

using namespace std;

long long f(long long e, long long n){
    if (n % 2 == 0) {
        if (e % 2 == 0) return n / 2 + (n - e) / 2 - 1;
        return (n - e) / 2;
    }
    if (e == n - 1) return 0;
    return f(e, n - 1) + 1;
}

int main(){
    long long n, k, p;
    cin >> n >> k >> p;

    for (int i = 0; i < p; i++) {
        long long _x;
        cin >> _x;
        _x--;

        if (f(_x, n) < k) cout << "X";
        else cout << ".";
    }
}
