#include<bits/stdc++.h>

using namespace std;

#define LL long long
#define mod 1000000007
#define N 100005

LL cross(LL x, LL n)
{
    LL res = 0,nn = n;
    if(n&1) {res++,nn--;}
    if(x==n)    return 1;

    if(x%2==0)
        return res+(nn-x)/2+1;

    return nn/2+(nn-x)/2+res+1;
}

int main()
{
    LL n,k,x;
    int p;

    cin>>n>>k>>p;
    while(p--)
    {
        cin>>x;
        if(cross(x,n)>k)    cout<<'.';
        else    cout<<'X';
    }
    return 0;
}
