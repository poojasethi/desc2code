#include <iostream>
using namespace std;

typedef long long ll;

ll n, k, p, x;

bool ok(ll x){
  if(k == 0) return 0;
  ll y = n - x;
  if(n % 2 == 0){
    if(y % 2 == 0){
      return y / 2 < k;
    } else {
      return y / 2 < k - n / 2;
    }
  } else {
    if(y % 2 == 0){
      if(y == 0) return 1;
      return y / 2 < k - n / 2;
    } else {
      return y / 2 < k - 1;
    }
  }
}

int main(){
  cin >> n >> k >> p;
  for(int i = 0; i < p; i++){
    cin >> x;
    cout << (ok(x) ? "X" : ".");
  }
}
