#include <iostream>

using namespace std;

const char model[] = ".X";

int main() {
  long long n, k, x;
  int p;
  cin >> n >> k >> p;
  for (int i = 0; i < p; ++i) {
    cin >> x;
    --x;
    if (n - k < k) {
      if (x >= 2 * (n - k)) cout << 'X';
      else cout << model[x % 2];
    } else if ((n - 2 * k) % 2 == 0) {
      if (x >= n - 2 * k) cout << model[(n - x) % 2];
      else cout << '.';
    } else {
      if (x == n - 1 && k > 0) cout << model[1];
      else if (k > 1) {
        if (x >= n - 2 * k + 1) cout << model[(n - 1 - x) % 2];
        else cout << '.';
      } else cout << '.';
    }
  }
  cout << endl;
  return 0;
}
