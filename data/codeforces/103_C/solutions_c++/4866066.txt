#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;
typedef long long LL;
int main(){
	LL n,m,p;
	// freopen("C.in","r",stdin);
	cin >> n >> m >> p;
	while (p--){
		LL pos;
		cin >> pos;pos--;
		bool stu = m>(n+(n&1))/2;
		if (stu){
			if(pos&1 || pos>=(n-m)*2)
				printf("X");
			else
				printf(".");
		} else {
			if ((n&1)==0){
				if((pos&1) && pos>=(n-m*2)+1)
					printf("X");
				else
					printf(".");
			} else {
				if(m&&(((pos&1) && pos>=n-(m-1)*2) || pos == n-1))
					printf("X");
				else
					printf(".");
			}
		}
	}
	return 0;
}
