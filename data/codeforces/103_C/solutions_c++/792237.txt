#include <iostream>
using namespace std;
int main()
{
    //ifstream cin("input.txt");
    //ofstream cout("output.txt");
    long long n,k,p,m,x;
	int i;
	cin >> n >> k >> p;
    if (n%2 && k) {n--; k--;}
    if (n<2*k) {m=2*k-n; n-=m; k-=m;}
    for (i=0;i<p;i++)
	{
        cin >> x;
        if (x>n || x%2==0 && (n-x)/2<k) cout << "X"; else cout << ".";
    }
	cout << endl;
    return 0;
}
