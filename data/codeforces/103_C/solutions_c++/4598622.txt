#include <iostream>
#include <cstdio>
using namespace std;
typedef long long ll ;
ll n,k,p;
int main()
{
    cin>>n>>k>>p;
    while (p--)
    {
        ll pos,te;cin>>pos;
        if (n&1)
        {
            if (pos==n) te=1;
            else if (pos&1) te=n/2+(n-pos)/2+1;
            else te=(n-pos+1)/2+1;
            if (te<=k) printf("X");
            else printf(".");
        }
        else
        {
            if (pos&1) te=n/2+(n-pos+1)/2;
            else te=(n-pos)/2+1;
            if (te<=k) printf("X");
            else printf(".");
        }
    }
    return 0;
}
