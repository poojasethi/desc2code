#include <iostream>
using namespace std;

typedef long long ll;

ll bullet(ll n, ll k) {
  if (n%2 == 0) {
    ll p = n/2 - k/2 - 1;
    if (k%2 == 0) return n/2 + p;
    return p;
  }
  
  if (k == n - 1) return 0;
  return 1 + bullet(n - 1, k);
}

int main() {
  ll n, k, p;
  cin >> n >> k >> p;
  for (int i = 0; i < p; ++i) {
    ll x;
    cin >> x;
    --x;
    if (bullet(n, x) < k) cout << "X";
    else cout << ".";
  }
  cout << endl;
}
