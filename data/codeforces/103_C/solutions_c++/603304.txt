#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
#define ll long long
using namespace std;
ll num(ll n, ll q){
    ll o = n % 2ll;
    if (q == n) return 1;
    if (n % 2ll) n--;
    
    if (q % 2ll) return n - q / 2ll + o;
    else return n / 2ll - q / 2ll + o + 1ll;
}
int main()
{
    ll n, k, p, q;
    cin >> n  >> k >> p;
    for (int i = 0;i < p;++i){
        cin >> q;
        if (num(n, q) > k) cout << '.';
        else cout << 'X';
    }
    cout << endl;
    cin >> n;
    return 0;
}
