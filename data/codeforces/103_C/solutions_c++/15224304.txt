#include <bits/stdc++.h>
#define ll long long

using namespace std;

int main(){
	ll n,k,m;
	int p;
	cin>>n>>k>>p;
	bool flag=false;
	m=n;
	if(n%2==1 and k>0)
		m--,flag=true,k--;
	ll x;
	ll b;
	string s="";
	for(int i=0;i<p;i++){
		cin>>x;
		b=0;
		if(x==n and flag==true)
			s=s+'X';
		else{
			if(x%2==1)
				b=m/2+(m/2-(x+1)/2);
			else
				b=m/2-x/2;	
			b=max(b,(ll)0);
			if(b>=k)
				s=s+'.';
			else
				s=s+'X';
		}
	}
	cout<<s;
	return 0;
}
