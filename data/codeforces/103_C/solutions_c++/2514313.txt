#include <cstdio>
using namespace std;

typedef long long ll;

ll n, k, x;

int p;

bool echo(ll x) {
    if(k == 0) return true;
    ll rep = k;
    if(n % 2) rep--;
    if(x % 2 == 0) {
        if(x / 2 + rep > n / 2) return false;
        return true;
    }
    else {
        if(x == n) return false;
        if(x / 2 + rep - n / 2 + 1 > n / 2) return false;
        return true;
    }
}

int main() {
//    freopen("in", "r", stdin);
    scanf("%I64d%I64d%d", &n, &k, &p);
    while(p--)
    {
        scanf("%I64d", &x);
        putchar(echo(x) ? '.' : 'X');
    }
    puts("");
    return 0;
}
