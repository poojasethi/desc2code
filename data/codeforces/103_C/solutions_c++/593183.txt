#include<iostream>

using namespace std;

typedef long long ll;

inline void out(bool b){ cout<<(b?'.':'X'); }

int main(){
	ll n,k;
	int p; cin>>n>>k>>p;
	while(p--){
		ll x; cin>>x;

		if(k==0){ out(1); continue; }

		if(n%2==0){ // n is even
			if(k<=n/2) out(!(x%2==0 && n-2*(k-1)<=x));
			else       out(!(x%2==0 || 2*n-2*k+1<=x));
		}
		else{ // n is odd
			if(x==n) out(0);
			else{
				if(k<=(n+1)/2) out(!(x%2==0 && n-2*(k-1)<=x));
				else           out(!(x%2==0 || 2*n-2*k+1<=x));
			}
		}
	}

	return 0;
}
