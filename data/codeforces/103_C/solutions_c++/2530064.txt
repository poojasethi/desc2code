#include <cstdio>
#include <algorithm>
using namespace std;

__int64 n, k, p, od, ev, x;

int main() {
	scanf("%I64d%I64d%I64d", &n, &k, &p);
	if ( (n&1) && k) n--, k--;
	if (k) {
		ev = max((__int64)2, n-(k-1)*2);
		od = min(n+1, n-(k-(n/2))*2+1);
	}
	else {
		ev = od = n+1;
	}
	while (p--) {
		scanf("%I64d", &x);
		printf("%c", ((x&1)&&(x>=od))||((!(x&1))&&(x>=ev))? 'X': '.');
	}
	printf("\n");
	return 0;
}
