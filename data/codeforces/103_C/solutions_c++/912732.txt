#include <iostream>

using namespace std;

typedef long long int lint;

int main()
{
  lint n,k,p;
  cin>>n>>k>>p;
  for (int i=0;i<p;i++) {
    lint input;
    cin>>input;
    lint pos=n-input;
    if (k==0) {
      cout<<".";
    } else if (2*k>=n) {
      lint x=2*k-n;
      if (pos<x)
	cout<<"X";
      else if ((pos-x)%2==0)
	cout<<"X";
      else
	cout<<".";
    } else if (n%2==0) {
      if (2*k>pos and pos%2==0)
	cout<<"X";
      else
	cout<<".";
    } else {
      if (pos==0)
	cout<<"X";
      else if (2*(k-1)+1>pos and pos%2==1)
	cout<<"X";
      else
	cout<<".";
    }
  }
  cout<<endl;
}
