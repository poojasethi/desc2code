#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,q;
int64 n,m;
inline int64 suan(int64 x) {
	int64 nn=n,an=0;
	if (nn&1) {
		if (x==nn) return 1;
		an++;
		nn--;
	}
	if (x%2==0) return an+(nn-x)/2+1;
	an+=nn/2;
	return an+(nn-x)/2+1;
}
int main() {
	scanf("%I64d%I64d%d",&n,&m,&q);
	For(i,1,q) {
		int64 x;
		scanf("%I64d",&x);
		int64 Rank=suan(x);
		if (Rank>m) printf(".");
		else printf("X");
	}
	printf("\n");
	return 0;
}
