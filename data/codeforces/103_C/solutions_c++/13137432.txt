#include <iostream>

using namespace std;

typedef long long LL;

LL n, k;
int p;

void work(LL n, LL k, LL x){
	LL t;
	if (x % 2 == 0) t = (n - x) / 2 + 1;
	else t = n / 2 + (n - 1 - x) / 2 + 1;
	if (t <= k) cout << 'X';
	else cout << '.';
}

int main(){
	cin >> n >> k >> p;
	if (k == 0){
		for(int i = 0; i < p; i++){
			cout << '.';
		}
		cout << endl;
		return 0;
	}
	for(int i = 0; i < p; i++){
		LL x;
		cin >> x;
		if (x == n){
			cout << 'X';
		}
		else{
			if (n % 2 == 1) work(n - 1, k - 1, x);
			else work(n, k, x);
		}
	}
	cout << endl;
}
