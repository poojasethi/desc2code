#include <cstdio>
#include <cstdlib>
#include <iostream>

using namespace std;

int n=0, m=0;
int sum=0;
int k[200];

int deb=0;

int main() {
	scanf("%d%d", &n, &m);
	if (m<1) {
		printf("No\n");
		return 0;
	}

	for (int i=0; i<n; ++i) {
		scanf("%d", &k[i]);
		sum+=k[i];
	}
	
	if (sum<m || (n==1 && k[0]!=m) ) {
		printf("No\n");
		return 0;
	}

	int t=-1;
	for (int i=0; i<n; ++i)
		if (k[i]<=m) {t=i; break;}

	if (t!=-1) {
		printf("Yes\n");
		m-=k[t];
		for (int i=0; i<k[t]-1; ++i) {
			printf("%d %d ", t+1, t+1); deb+=2;
		}
		k[t]=0;
		
		int p=0;
		int s=0;
		while (s<m) {
			if (k[p]<1) ++p;
			printf("%d %d ", p+1, p+1); deb+=2;
			k[p]--;
			++s;
		}

		printf("%d ", t+1); deb+=1;
		
		for (int i=0; i<n; ++i) {
			for (int j=0; j<k[i]*2; ++j) {
				printf("%d ", i+1); deb+=1;
			}
		}
	
		printf("%d\n", t+1); deb+=1;
	} else {
		if (m==1) {
			printf("No\n"); return 0;
		}
		printf("Yes\n");
		m-=2;
		k[0]--;
		printf("1 "); deb+=1;
		for (int i=1; i<n-1; ++i)
			for (int j=0; j<k[i]*2; ++j) {
				printf("%d ", i+1); deb+=1;
			}
		
		for (int j=0; j<(k[n-1]-1)*2; ++j) {
			printf("%d ", n-1+1); deb+=1;
		}
		printf("1 "); deb+=1;
		
		printf("%d ", n); deb+=1;
		for (int j=0; j<(k[0]-m)*2; ++j) {
			printf("1 ");deb+=1;
		}
		printf("%d ", n); deb+=1;

		for (int j=0; j<(m*2); ++j) {
			printf ("1 ");deb+=1;
		}
		
		if (deb!=sum*2) //cerr << "FAIL\n";
			return 100;
	}

	return 0;
}
