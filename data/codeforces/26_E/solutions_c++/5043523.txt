#include<cstdio>
const int maxn=1011;
int n,w,s;
int a[maxn];
int main()
{
    scanf("%d%d",&n,&w);
    for(int i=1;i<=n;i++)
    {
      scanf("%d",&a[i]);
      s+=a[i];
    }
    if(n==1)
    {
      if(w==a[1])
      {
        puts("Yes");
        for(int i=0;i<a[1]<<1;i++)printf("1 ");
      }
      else puts("No");
      return 0;
    }
    int p;
    if(w==1)
    {
      for(p=1;p<=n&&a[p]!=1;p++);
      if(p==n+1)puts("No");
      else
      {
        puts("Yes");
        printf("%d ",p);
        for(int i=1;i<=n;i++)
          if(i!=p)
            while(a[i]--)printf("%d %d ",i,i);
        printf("%d ",p);
      }
      return 0;
    }
    if(w<2||w>s)
    {
      printf("No\n");
      return 0;
    }
    puts("Yes");
    int d=s-w;
    a[1]--;
    a[2]--;
    printf("1 ");
    for(p=n;p>1&&d;d--)
    {
      printf("%d %d ",p,p);
      a[p]--;
      while(p&&!a[p])p--;
    }
    printf("1 2 ");
    while(d)
    {
      printf("%d %d ",p,p);
      d--;
      a[p]--;
      while(p&&!a[p])p--;
    }
    printf("2 ");
    while(p)
    {
      printf("%d %d ",p,p);
      a[p]--;
      while(p&&!a[p])p--;
    }
    return 0;
}
