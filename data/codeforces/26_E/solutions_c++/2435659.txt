#include<cstdio>
#include<algorithm>
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=210;
int n,m,a[mn];
int main()
{
    scanf("%d%d",&n,&m);
    int i,j,k;
    fo(i,1,n)scanf("%d",a+i);
    if(m<=0){puts("No");return 0;}
    if(n<=1)
    {
        if(m!=a[1])puts("No");
        else
        {
            printf("Yes\n1 1");
            fo(i,1,a[1]-1)printf(" 1 1");
            printf("\n");
        }
        return 0;
    }
    if(m==1)
    {
        k=0;
        fo(i,1,n)if(a[i]==1)k=i;
        if(!k)puts("No");
        else
        {
            --a[k];
            puts("Yes");
            printf("%d",k);
            fo(i,1,n)fo(j,1,a[i])printf(" %d %d",i,i);
            printf(" %d\n",k);
        }
        return 0;
    }
    m=-m;
    fo(i,1,n)m+=a[i];
    if(m<0)puts("No");
    else
    {
        puts("Yes");
        --a[1],--a[2];
        printf("1");
        fo(i,2,n)
        {
            if(!m)break;
            k=min(m,a[i]);
            m-=k,a[i]-=k;
            fo(j,1,k)printf(" %d %d",i,i);
        }
        printf(" 1 2");
        k=min(m,a[1]);
        a[1]-=k;
        fo(i,1,k)printf(" 1 1");
        printf(" 2");
        fo(i,1,n)fo(j,1,a[i])printf(" %d %d",i,i);
    }
    return 0;
}
