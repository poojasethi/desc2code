#define maxn 105

int     n,w,data[maxn],tol;
void    writ(int x){
    for (int i = 0; i<n; i++) if ( i != x )
        while ( data[i]  ) {
            printf(" %d %d" , i+1 , i+1 );
            data[i]--;
        }
}

        printf("Yes\n%d" , i+1 );

}
int     main(){
