#include <cstdio>
int S,N,M,a[105],b[105];
int main()
{
	scanf("%d%d",&N,&M);
	int p=1,q;
	for (int i=1; i<=N; i++) scanf("%d",&a[i]),b[i]=a[i]*2,S+=a[i],p=a[i]<a[p]?i:p;
	if (M>S||M<=0||(N==1&&S!=M)||(M==1&&a[p]>1)) {puts("No"); return 0;}
	puts("Yes");
	if (N>1)
	if (M==1)
	{
		printf("%d ",p),b[p]--;
		for (int i=1; i<=N; i++) if (i!=p)
			for (;b[i]; b[i]--) printf("%d ",i);
	}
	else if (M<=a[p])
	{
		q=(p==1?2:1),printf("%d ",q);
		for (int i=1; i<=a[p]-M+1; i++) printf("%d %d ",p,p),b[p]-=2;
		printf("%d %d ",q,p),b[p]--,b[q]-=2;
		for (int i=1; i<=N; i++) if (i!=p)
			for (;b[i]; b[i]--) printf("%d ",i);
	}
	else
	{
		for (int i=1,t=0; i<=N; i++) if (i!=p)
			for (;b[i]; b[i]-=2)
			{
				printf("%d %d ",i,i),t++;
				if (t==M-a[p]) printf("%d ",p),b[p]--;
			}
	}
	for (;b[p]; b[p]--) printf("%d%c",p,b[p]==1?'\n':' ');
}
