#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
struct rec{int x,y;}a[110];
int b[110],n,w,i,j,k;
bool operator <(rec a,rec b)
{
	return a.x<b.x;
}
void print(int p,int q,int x,int y)
{
	for(j=n;j>q;j--)
		for(k=1;k<=a[j].x*2;k++) printf("%d ",a[j].y);
	for(j=1;j<2*y;j++) printf("%d ",a[q].y);
	for(j=1;j<=2*(a[p].x-x);j++) printf("%d ",a[p].y);
	for(j=1;j<p;j++)
		for(k=1;k<=a[j].x*2;k++) printf("%d ",a[j].y);
	printf("%d %d ",a[q].y,a[p].y);
	for(j=1;j<=2*(a[q].x-y);j++) printf("%d ",a[q].y);
	for(j=1;j<2*x;j++) printf("%d ",a[p].y);
	cout<<endl;
}
int main()
{
	cin>>n>>w;
	for(i=1;i<=n;i++) scanf("%d",&a[i].x),a[i].y=i;
	sort(a+1,a+n+1);
	for(i=n;i;i--) b[i]=b[i+1]+a[i].x;
	if(w<=0||w==1&&a[1].x!=1||b[1]<w||b[1]>w&&n==1) {puts("No"); return 0;}
	puts("Yes");
	if(w==1)
	{
		printf("%d ",a[1].y);
		for(i=2;i<=n;i++)
			for(j=1;j<=a[i].x*2;j++) printf("%d ",a[i].y);
		printf("%d\n",a[1].y);
		return 0;
	}
	for(i=n;i;i--)
		if(b[i]==w)
		{
			for(j=n;j>i;j--)
				for(k=1;k<=a[j].x*2;k++) printf("%d ",a[j].y);
			for(j=1;j<2*a[i].x;j++) printf("%d ",a[i].y);
			for(j=1;j<i;j++)
				for(k=1;k<=a[j].x*2;k++) printf("%d ",a[j].y);
			printf("%d ",a[i].y);
			cout<<endl;
			break;
		}
		else if(b[i]>w)
		{
			if(i<n) print(i,i+1,w-b[i+1]+(b[i+1]==w-1),a[i+1].x-(b[i+1]==w-1)); else print(i-1,i,1,w-1);
			break;
		}
	return 0;
}
