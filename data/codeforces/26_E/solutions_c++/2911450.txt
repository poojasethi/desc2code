#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

#define X first
#define Y second
typedef pair<int, int> PII;
const int maxn = 1005;
PII a[maxn];
int n, w, s;

int main() {
    int i, j;
    scanf("%d%d", &n, &w);
    for (i = 1; i <= n; ++i) {
        scanf("%d", &a[i].X);
        a[i].Y = i; s += a[i].X;
    }
    sort(a + 1, a + n + 1);
    if (w < 0 || w > s || (n == 1 && w < a[1].X) ||
        (w == 1 && a[1].X > 1)) {
        puts("No"); return 0;
    }
    puts("Yes");
    if (w >= a[1].X) {
        printf("%d ", a[1].Y); w = s - w; --a[1].X;
        for (i = 2; i <= n; ++i)
            while (a[i].X && w) {
                --w; --a[i].X;
                printf("%d %d ", a[i].Y, a[i].Y);
            }
        printf("%d ", a[1].Y);
        for (i = 1; i <= n; ++i)
            for (; a[i].X; --a[i].X)
                printf("%d %d ", a[i].Y, a[i].Y);
    } else {
        --a[2].X; printf("%d ", a[1].Y);
        for (i = 2; i <= n; ++i)
            for (; a[i].X; --a[i].X)
                printf("%d %d ", a[i].Y, a[i].Y);
        printf("%d %d ", a[1].Y, a[2].Y);
        for (i = 0; i < a[1].X - w + 1; ++i)
            printf("%d %d ", a[1].Y, a[1].Y);
        printf("%d ", a[2].Y);
        for (i = 0; i < w - 2; ++i)
            printf("%d %d ", a[1].Y, a[1].Y);
    }
    printf("\n");
    return 0;
}
