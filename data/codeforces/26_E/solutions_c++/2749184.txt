int main()
{
    scanf("%d%d",&n,&m);
    rep(i,n)
    {
        scanf("%d",&a[i]);
        sum+=a[i];
    }
    if(n==1)
    {
        if(m!=a[1])printf("No\n");
        else
        {
            printf("Yes\n");
            printf("1 1");
            rep(i,a[1]-1)printf(" 1 1");
            printf("\n");
        }
        return 0;
    }
    if(m<=0||m>sum)
    {
        printf("No\n");
        return 0;
    }
    if(m==1)
    {
        int ans=0;
        rep(i,n)if(a[i]==1)ans=i;
        if(!ans)printf("No\n");
        else
        {
            printf("Yes\n");
            printf("%d",ans);
            --a[ans];
            rep(i,n)rep(j,a[i])printf(" %d %d",i,i);
            printf(" %d\n",ans);
        }
        return 0;
    }
    sum=sum-m;
    printf("Yes\n");
    --a[1];--a[2];
    printf("1");
    for(int i=2;i<=n;++i)
    {
        if(!sum)break;
        int k=min(a[i],sum);
        sum-=k;
        a[i]-=k;
        rep(j,k)printf(" %d %d",i,i);
    }
    printf(" 1 2");
