#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
int sum,n,w,a[100];
int main(){
    //freopen("AC.in","r",stdin);
    //freopen("AC.out","w",stdout);
    cin>>n>>w;
    for(int i=0;i<n;i++){
        scanf("%d",a+i);
        sum+=a[i];
    }
    if(sum<w||w<=0||(n==1&&sum!=w)||(w==1&&find(a,a+n,1)==a+n))
        return puts("No"),0;
    puts("Yes");
    if(n==1)while(a[0]--)printf("1 1 ");
    else
    if(w==1){
        int x=find(a,a+n,1)-a;
        a[x]--;
        printf("%d ",x+1);
        for(int i=0;i<n;i++)
            while(a[i]--)printf("%d %d ",i+1,i+1);
        printf("%d ",x+1);
    }
    else{
        a[0]--,a[1]--;
        for(int i=0;i<n;i++)
            while(a[i]&&w>2){
                printf("%d %d ",i+1,i+1);
                w--;
                a[i]--;
            }
        printf("1 ");
        for(int i=1;i<n;i++)
            while(a[i]--)printf("%d %d ",i+1,i+1);
        printf("1 2 ");
        while(a[0]--)printf("%d %d ",1,1);
        puts("2");
    }
    return 0;
}
