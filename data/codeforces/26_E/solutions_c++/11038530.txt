#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
using namespace std;
const int MAXN = 105;
int n,w,sum,cnt[MAXN],rest,ord[MAXN];

inline bool cmp(const int &x,const int &y){
	return cnt[x] < cnt[y];
}

int main(){
	scanf("%d%d",&n,&w);
	sum = 0;
	for(int i = 1;i <= n;++ i) scanf("%d",&cnt[i]),sum += cnt[i],ord[i] = i;
	sort(ord + 1,ord + n + 1,cmp);
	if(w <= 0 || w > sum || (w != sum && n == 1)){
		puts("No");
	}else{
		if(w < cnt[ord[1]]){
			if(w == 1){
				puts("No"); return 0;
			}
			puts("Yes");
			printf("%d ",ord[2]);
			for(int i = 1;i <= cnt[ord[1]] - w + 1;++ i)
				printf("%d %d ",ord[1],ord[1]);
			printf("%d %d ",ord[2],ord[1]);
			-- cnt[ord[2]];
			for(int i = 2;i <= n;++ i)
				for(int j = 1;j <= cnt[ord[i]];++ j)
					printf("%d %d ",ord[i],ord[i]);
			printf("%d ",ord[1]);
			for(int i = 3;i <= w;++ i)
				printf("%d %d ",ord[1],ord[1]);
		}else{
			puts("Yes");
			printf("%d ",ord[1]);
			rest = sum - w;
			for(int i = 2;i <= n && rest;++ i)
				for(int j = 1,lim = cnt[ord[i]];j <= lim && rest;++ j,-- rest)
					printf("%d %d ",ord[i],ord[i]),-- cnt[ord[i]];
			printf("%d ",ord[1]); -- cnt[ord[1]];
			for(int i = 1;i <= n;++ i)
				for(int j = 1;j <= cnt[ord[i]];++ j)
					printf("%d %d ",ord[i],ord[i]);
		}
		puts("");
	}
	return 0;
}
