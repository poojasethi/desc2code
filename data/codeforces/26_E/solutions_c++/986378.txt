#include <iostream>
#include <cstdio>
using namespace std;
#define FOR(i,a,b) for(i=a;i<=b;i++)

int i,j,k,n,m;
int a[110];

int main(){
        scanf("%d%d",&n,&m);
        FOR(i,1,n) scanf("%d",&a[i]);
        if (m<=0) {puts("No");return 0;}
        if (n==1){
                if (m!=a[1]) {puts("No");return 0;}
                puts("Yes");
                a[1]--;printf("1 1");
        }else if (m==1){
                k=-1;
                FOR(i,1,n) if (a[i]==1) k=i;
                if (k==-1) {puts("No");return 0;}
                a[k]--;
                puts("Yes");
                printf("%d",k);
                FOR(i,1,n)FOR(j,1,a[i]) printf(" %d %d",i,i);
                printf(" %d\n",k);
                return 0;
        }else {
                m=-m;
                FOR(i,1,n) m+=a[i];
                if (m<0) {puts("No");return 0;}
                puts("Yes");
                a[1]--;a[2]--;
                printf("1");
                FOR(i,2,n){
                        if (m<=0) break;
                        k=min(m,a[i]);
                        m-=k;a[i]-=k;
                        FOR(j,1,k) printf(" %d %d",i,i);
                }
                printf(" 1");
                printf(" 2");
                k=min(m,a[1]);
                if (k){
                        a[1]-=k;
                        FOR(j,1,k) printf(" 1 1");
                }
                printf(" 2");
        }
        FOR(i,1,n)FOR(j,1,a[i]) printf(" %d %d",i,i);
}
