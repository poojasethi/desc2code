#include <iostream>
#include <algorithm>
using namespace std;
int N,n[111],o[111],w,i,j,k;
int main()
{
    cin >> N >> w;
    for (i=1;i<=N;i++) 
    {
        cin >> n[i];
        j+=n[i];
        if (n[i]==1) k=i;
    }
    if (w<=0 || w>j || (N==1 && n[1]!=w) || (N>=2 && w==1 && k==0))
    {
       cout << "No" << endl;
       return 0;
    }
    cout << "Yes" << endl;
    if (N==1)
    {
       for (i=1;i<=2*j;i++) cout << 1 << " ";
       return 0;
    }
    if (w==1)
    {
       cout << k << " ";
       for (i=1;i<=N;i++)
           if (i!=k)
           for (j=1;j<=2*n[i];j++) cout << i << " ";
       cout << k << " ";
       return 0;
    }
    n[1]--;
    n[2]--;
    k=w-2;
    for (i=1;i<=N && k>0;i++)
    {
        j=min(n[i],k);
        n[i]-=j;
        k-=j;
        o[i]+=j;
    }
    cout << 1 << " ";
    for (i=2;i<=N;i++)
        for (j=1;j<=2*n[i];j++)
            cout << i << " ";
    cout << 1 << " " << 2 << " ";
    for (i=1;i<=2*n[1];i++) cout << 1 << " ";
    cout << 2 << " ";
    for (i=1;i<=N;i++)
    for (j=1;j<=2*o[i];j++) cout << i << " ";
    cout << endl;
    return 0;
}
