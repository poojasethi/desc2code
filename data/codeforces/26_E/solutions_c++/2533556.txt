#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<cstring>

const int fin=0,maxn=1011;
int n,w;
int a[maxn];

int main(){
  if(fin){
    freopen("t.in","r",stdin);
    freopen("t.out","w",stdout);
  }
  int s=0,i,j,p,d;
  scanf("%d%d",&n,&w);
  for(i=1;i<=n;++i)
    scanf("%d",a+i),s+=a[i];
  if(n==1){
    if(w==a[1]){
      puts("Yes");
      for(i=0;i<2*a[1];++i)
        printf("1 ");
    }else puts("No");
    return 0;
  }
  if(w==1){
    for(p=1;p<=n&&a[p]!=1;++p);
    if(p==n+1)puts("No");
    else{
      puts("Yes");
      printf("%d ",p);
      for(i=1;i<=n;++i)if(i!=p)
        while(a[i]--)
          printf("%d %d ",i,i);
      printf("%d ",p);
    }
    return 0;
  }
  if(w<2||w>s)return puts("No"),0;
  puts("Yes");
  d=s-w,--a[1],--a[2];
  printf("1 ");
  for(p=n;p>1&&d>0;){
    printf("%d %d ",p,p),--d;
    --a[p];
    while(p>0&&!a[p])--p;
  }
  printf("1 2 ");
  while(d){
    printf("%d %d ",p,p),--d;
    --a[p];
    while(p>0&&!a[p])--p;
  }
  printf("2 ");
  while(p){
    printf("%d %d ",p,p);
    --a[p];
    while(p>0&&!a[p])--p;
  }
  return 0;
}
