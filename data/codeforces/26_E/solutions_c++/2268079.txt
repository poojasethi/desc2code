#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <queue>
#include <map>
#include <string.h>

const int N=100010;
const int inf=0x3f3f3f3f;

using namespace std;

int n,w,a[N];

int main()
{
	scanf("%d%d",&n,&w);
	int sum=0;
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]),sum+=a[i];
	if(w>sum || w<=0) return puts("No"),0;
	if(n==1)
	{
		if(a[1]!=w) return puts("No"),0;
		cout<<"Yes"<<endl;
		for(int i=1;i<=2*a[1];i++)
			cout<<1<<" ";
		cout<<endl;
		return 0;
	}
	if(w==1)
	{
		int id=-1;
		for(int i=1;i<=n;i++)
			if(a[i]==1) id=i;
		if(id==-1) return puts("No"),0;
		cout<<"Yes"<<endl;
		cout<<id<<" ";
		for(int i=1;i<=n;i++)
			for(int j=1;j<=2*a[i];j++)
				if(i-id) cout<<i<<" ";
		cout<<id<<endl;
		return 0;
	}
	cout<<"Yes"<<endl;
	int t=sum-w;
	a[1]--;
	cout<<1<<" ";
	while(a[2]>1)
	{
		if(t==0) break;
		cout<<2<<" "<<2<<" ";
		a[2]--;t--;
	}
	for(int i=3;i<=n;i++)
		while(a[i])
		{
			if(t==0) break;
			cout<<i<<" "<<i<<" ";
			a[i]--;t--;
		}
	a[2]--;
	cout<<1<<" "<<2<<" ";
	while(t && a[1])
	{
		cout<<1<<" "<<1<<" ";
		a[1]--;t--;
	}
	cout<<2<<" ";
	for(int i=1;i<=n;i++)
		while(a[i]) cout<<i<<" "<<i<<" ",a[i]--;
	cout<<endl;
	return 0;
}
