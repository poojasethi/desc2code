#include <cstdio>
int n,w,a[1000],b[1000],s;
void work()
{
	int i,j,k,now;
	if(n == 1 && a[1] == w)
	{
		printf("Yes\n");
		for(i=1;i<=a[1];i++)
			printf("1 1 ");
		return;
	}
	if(w == 1 && n >= 2)
	{
		for(i=1;i<=n;i++)
			if(a[i] == 1) break;
		if(i<=n)
		{
			printf("Yes\n");
			printf("%d ",i);
			for(j=1;j<=n;j++)
				if(j!=i)
				for(k=1;k<=a[j];k++)
					printf("%d %d ",j,j);
			printf("%d",i);
			return;
		}
	}
	if(n >= 2 && w >= 2 && w <= s)
	{
		b[1] = b[2] = 1; now = 2;
		for(i=1;i<=n && now<=w;i++)
		{
			if(a[i]-b[i] < w-now)
				now += a[i]-b[i],b[i]=a[i];
			else
				b[i] += w-now, now = w;
		}
		printf("Yes\n1 ");
		for(i=2;i<=n;i++)
		{
			for(j=1;j<=a[i]-b[i];j++)
				printf("%d %d ",i,i);
		} 
		printf("1 ");
		printf("2 ");
		for(j=1;j<=a[1]-b[1];j++)
			printf("1 1 ");
		printf("2 ");
		for(i=1;i<=n;i++)
			for(j=(i<=2)?2:1;j<=b[i];j++)
			printf("%d %d ",i,i);
		return;
	}
	printf("No");
}
int main()
{
	//freopen("input.txt","r",stdin);
	scanf("%d%d",&n,&w);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		s += a[i];
	}
	work();
}
