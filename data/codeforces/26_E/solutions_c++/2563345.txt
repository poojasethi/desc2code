#include <cstdio>

using namespace std;

const int maxn = 105;

int cnt[maxn];
int n, sum, w;

void init()
{
	scanf("%d%d", &n, &w);
	sum = 0;
	for (int i = 1; i <= n; i ++){
		scanf("%d", &cnt[i]); sum += cnt[i];
	}
}

void print()
{
	if (w < 1 || w > sum){
		printf("No\n"); return;
	}
	if (n == 1){
		if (w != sum){
			printf("No\n"); return;
		}
		printf("Yes\n");
		for (int i = 0; i < cnt[1]; i ++) printf("1 1 ");
		putchar('\n'); return;
	}
	if (w == 1){
		int v = -1;
		for (int i = 1; i <= n; i ++)
			if (cnt[i] == 1) v = i;
		if (v == -1){
			printf("No\n"); return;
		}
		printf("Yes\n%d ", v);
		for (int i = 1; i <= n; i ++) if (i != v)
			for (int j = 0; j < cnt[i]; j ++) printf("%d %d ", i, i);
		printf("%d\n", v); return;
	}
	printf("Yes\n");
	int extra = sum - w;
	printf("1 ");
	for (int i = 3; i <= n; i ++)
		while (cnt[i] > 0 && extra > 0){
			cnt[i] --; extra --; printf("%d %d ", i, i);
		}
	cnt[2] --;
	while (cnt[2] > 0 && extra > 0){
		cnt[2] --; extra --; printf("2 2 ");
	}
	cnt[1] --;
	printf("1 ");
	printf("2 ");
	while (cnt[1] > 0 && extra > 0){
		cnt[1] --; extra --; printf("1 1 ");
	}
	printf("2 ");
	for (int i = 1; i <= n; i ++)
		for (int j = 0; j < cnt[i]; j ++) printf("%d %d ", i, i);
	putchar('\n');
}

int main()
{
	init();
	print();
	return 0;
}
