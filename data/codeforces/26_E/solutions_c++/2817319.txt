#include <cstdio>
#include <algorithm>
using	namespace	std;
const	int	maxn = 100 + 10;
int	n, w, sig;
struct	rec
{
	int	key, wei;
}	a[maxn];
bool	cmp(rec a, rec b)	{return a.key < b.key;}
int	main()
{
//	freopen("26e.in", "r", stdin);
//	freopen("26e.out", "w", stdout);
	scanf("%d%d", &n, &w);
	sig = 0;
	for (int i=1; i<=n; i++) scanf("%d", &a[i].key), a[i].wei = i, sig += a[i].key;
	if (w <= 0 || w > sig)	{printf("No\n"); return 0;}
	if (n == 1 && w != a[1].key) {printf("No\n"); return 0;}
	sort(a + 1, a + n + 1, cmp);
	if (w == 1 && a[1].key > 1) {printf("No\n"); return 0;}
	printf("Yes\n");
	if (w < a[1].key)
	{
		for (int i=1; i<=2*w-4; i++) printf("%d ", a[1].wei);
		printf("%d ", a[2].wei); a[2].key --;
		for (int i=2*w-3; i<=2*a[1].key-2; i++) printf("%d ", a[1].wei);
		printf("%d ", a[2].wei);
		printf("%d ", a[1].wei);
		for (int i=2; i<=n; i++)
			for (int j=1; j<=2*a[i].key; j++) printf("%d ", a[i].wei);
		printf("%d ", a[1].wei);
		printf("\n");
		return 0;
	}
	w = sig - w;
	for (int i=1; i<2*a[1].key; i++) printf("%d ", a[1].wei);
	for (int i=2; i<=n; i++)
	{
		if (w >= a[i].key)
		{
			for (int j=1; j<=2*a[i].key; j++) printf("%d ", a[i].wei);
			w -= a[i].key; a[i].key = 0;
		}
		else
		{
			for (int j=1; j<=2*w; j++) printf("%d ", a[i].wei);
			a[i].key -= w; w = 0; break;
		}
	}
	printf("%d ", a[1].wei);
	for (int i=2; i<=n; i++)
		for (int j=1; j<=2*a[i].key; j++) printf("%d ", a[i].wei);
	printf("\n");
	return 0;
}
