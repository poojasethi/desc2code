#include<cstdio>
int a[111];
int main(){
//	freopen("26e.in","r",stdin);
//	freopen("26e.out","w",stdout);
	int i,j,n,m,sum,ls;
	scanf("%d%d",&n,&m);
	if (m<1) {printf("No\n"); return 0;}
	for (i=1;i<=n;i++) scanf("%d",a+i);
	sum=0;
	for (i=1;i<=n;i++) sum+=a[i];
	if (sum<m) {printf("No\n"); return 0;}
	if (n==1){
		if (a[1]!=m) {printf("No\n"); return 0;}
		printf("Yes\n");
		for (i=1;i<=2*m;i++) printf(i==2*m?"1\n":"1 ");
	} else
	if (m==1){
		for (i=1;i<=n;i++) if (a[i]==1) break;
		if (i==n+1) {printf("No\n"); return 0;}
		printf("Yes\n");
		a[i]--;
		printf("%d",i);
		for (j=1;j<=n;j++)
			for (;a[j]>=1;a[j]--) printf(" %d %d",j,j);
		printf(" %d\n",i);
	} else{
		printf("Yes\n");
		a[1]--;
		printf("1");
		ls=sum-m;
		for (i=2;i<n;i++)
			for (;ls && a[i];ls--,a[i]--) printf(" %d %d",i,i);
		for (;ls && a[n]>=2;ls--,a[n]--) printf(" %d %d",n,n);
		printf(" 1");
		a[n]--;
		printf(" %d",n);
		for (;ls && a[1];ls--,a[1]--) printf(" 1 1");
		printf(" %d",n);
		for (i=1;i<=n;i++)
			for (;a[i];a[i]--) printf(" %d %d",i,i);
		printf("\n");
	}
	return 0;
}
