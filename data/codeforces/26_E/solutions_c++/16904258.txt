#include<stdio.h>
#include<string.h>
#include<algorithm>
#define N 110
using namespace std;
typedef long long ll;
int a[N];
int st1[210000],top1;
int st2[210000],top2;
int main()
{
	int n,w,m,i,k;
	scanf("%d%d",&n,&w);
	m=0;
	for(i=1;i<=n;i++)
	{
		scanf("%d",a+i);
		m+=a[i];
	}
	if(w<1||w>m)
	{
		puts("No");
		return 0;
	}
	if(n==1)
	{
		if(w!=m)
			puts("No");
		else
		{
			puts("Yes");
			for(i=1;i<=m;i++)
				printf("1 1 ");
		}
		return 0;
	}
	if(w==1)
	{
		k=0;
		for(i=1;i<=n;i++)
			if(a[i]==1)
				k=i;
		if(k==0)
			puts("No");
		else
		{
			puts("Yes");
			printf("%d ",k);
			for(i=1;i<=n;i++)
			{
				if(i==k)	continue;
				while(a[i])
				{
					a[i]--;
					printf("%d %d ",i,i);
				}
			}
			printf("%d ",k);
			puts("");
		}
		return 0;
	}
	a[1]--;
	a[2]--;
	for(i=1,k=2;i<=n&&k!=w;i++)
	{
		while(k!=w&&a[i])
		{
			a[i]--;
			k++;
			st1[++top1]=i;
			st1[++top1]=i;
		}
	}
	i--;
	st1[++top1]=1;
	st2[++top2]=2;
	for(;i<=n;i++)
	{
		if(i!=1)
		while(a[i])
		{
			a[i]--;
			st1[++top1]=i;
			st1[++top1]=i;
		}
		else
		while(a[i])
		{
			a[i]--;
			st2[++top2]=i;
			st2[++top2]=i;
		}
	}
	st1[++top1]=1;
	st2[++top2]=2;
	puts("Yes");
	for(i=1;i<=top1;i++)
	{
		printf("%d ",st1[i]);
	}
	for(i=1;i<=top2;i++)
	{
		printf("%d ",st2[i]);
	}
	puts("");
	fclose(stdin);
	fclose(stdout);
	return 0;
}
