#include<cstdio>
#include<algorithm>
using namespace std;
int N,W,i,j,sum,P;
int A[105];
int main()
{
	scanf("%d%d",&N,&W);
	for(i=1;i<=N;++i) scanf("%d",&A[i]);
	for(i=1;i<=N;++i) sum+=A[i];
	P=-1;
	for(i=1;i<=N;++i) if(A[i]==1) P=i;
	if(sum<W||W<=0||N==1&&A[1]!=W||W==1&&P==-1)
	{
		printf("No\n");
		return 0;
	}
	printf("Yes\n");
	if(N==1)
	{
		for(i=1;i<2*A[1];++i) printf("1 ");
		printf("1\n");
		return 0;
	}
	if(W==1)
	{
		printf("%d ",P);
		for(i=1;i<=N;++i) if(i!=P)
			for(j=1;j<=2*A[i];++j)
				printf("%d ",i);
		printf("%d\n",P);
		return 0;
	}
	A[1]--,A[2]--;
    for(i=1;i<=N;++i)
		for(;A[i]&&W>2;--W,--A[i])
            printf("%d %d ",i,i);
    printf("1 ");
    for(i=2;i<=N;++i)
    	for(;A[i]--;) printf("%d %d ",i,i);
    printf("1 2 ");
    for(;A[1]--;) printf("1 1 ");
    printf("2\n");
	return 0;
}
