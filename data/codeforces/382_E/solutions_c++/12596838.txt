#include<iostream>
#include<cstdio>
#define lld long long
using namespace std;
const int Mod = 1000000007;
int n, k;
int f[55][55][2], C[55][55];
int main()
{
	scanf("%d%d", &n, &k);
	C[0][0] = 1;
	for(int i = 1; i <= n; ++i)
		for(int j = 0; j <= i; ++j)
			C[i][j] = (C[i - 1][j - 1] + C[i - 1][j]) % Mod;
	f[0][0][1] = f[1][0][0] = 1;
	int v, cn, rv;
	for(int i = 2; i <= n; ++i)
	{
		for(int l = 0, r = i - 1; l <= r; ++l, --r)
		{
			if(l != r) v = C[i - 1][l];
			else v = C[i - 2][l - 1]; // = C[n - 1][l] / 2
			if(i != n) v = ((lld)v * (lld)i) % Mod;
			for(int pl = 0; pl * 2 <= l; ++pl)
				for(int pr = 0; pr * 2 <= r; ++pr)
					for(int cl = 0; cl <= 1; ++cl)
						for(int cr = 0; cr <= 1; ++cr)
							if(f[l][pl][cl] && f[r][pr][cr])
							{
								cn = !(cl && cr);
								rv = ((((lld)f[l][pl][cl] * (lld)f[r][pr][cr]) % Mod) * v) % Mod;
								f[i][pl + pr + cn][cn] = (f[i][pl + pr + cn][cn] + rv) % Mod;
							}
		}
	}
	cout << (f[n][k][0] + f[n][k][1]) % Mod << endl;
	return 0;
}
