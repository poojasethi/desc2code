#include <iostream>

using namespace std;

typedef long long int ll;

ll modulo=1e9+7;

ll combi[51][51];

ll memcuantos[51][51][2];
int calculado[51][51][2];

ll cuantos(int n,int k,int t)
{
  if (k>=n) return 0;
  if (n==1)
    return t==0;
  if (n==2)
    return k==1 and t==1;
  ll &mem=memcuantos[n][k][t];
  if (calculado[n][k][t]) return mem;
  calculado[n][k][t]=1;
  mem=(n-1)*cuantos(n-1,k-t,1-t);
  for (int n1=1;n1<n-1;n1++) {
    int n2=n-1-n1;
    for (int k1=0;k1+t<=k;k1++) {
      int k2=k-k1-t;
      if (k1>=n1 or k2>=n2) continue;
      for (int t1=0;t1<=1;t1++) {
	for (int t2=0;t2<=1;t2++) {
	  if ((t1 and t2 and t) or (not t1 and not t) or (not t2 and not t)) continue;
	  mem+=combi[n-2][n1-1]*n1%modulo*n2%modulo*cuantos(n1,k1,t1)%modulo*cuantos(n2,k2,t2)%modulo;
	}
      }
    }
  }
  mem%=modulo;
  return mem;
}

int main()
{
  {
    for (int n=0;n<51;n++)
      for (int k=0;k<=n;k++)
	if (k==0 or k==n) combi[n][k]=1;
	else combi[n][k]=(combi[n-1][k]+combi[n-1][k-1])%modulo;
  }
  int n,k;
  cin>>n>>k;
  cout<<(cuantos(n,k,0)+cuantos(n,k,1))%modulo<<endl;
}
