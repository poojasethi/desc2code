#include<iostream>
#include<cstdio>
#define mod 1000000007
#define LL long long
using namespace std;
int n,m;

int f[55][55][2];
int C[55][55];

int main()
{
//	freopen("E.in","r",stdin);
	C[0][0]=1;
	for (int i=1;i<55;++i) {
		C[i][0]=1;
		for (int j=1;j<55;++j) C[i][j]=(C[i-1][j]+C[i-1][j-1])%mod;
	}
	
	scanf("%d%d",&n,&m);
	f[1][0][0]=1;
	f[2][1][1]=1;
	f[0][0][0]=1;
	for (int i=3;i<=n;++i) {
		for (int j=0;j<i;++j) {
			for (int cnt_L=0;cnt_L<=(i-1)/2;++cnt_L) {
				int cnt_R=i-cnt_L-1;
				for (int match_L=0;match_L<=j;++match_L) {
					int t=(LL)max(cnt_L,1)*max(cnt_R,1)*C[i-1][cnt_L]%mod;
					if (cnt_L==cnt_R) t=(LL)t*500000004%mod;
					
					(f[i][j][0]+=(LL)f[cnt_L][match_L][cnt_L>0]*f[cnt_R][j-match_L][1]%mod*t%mod)%=mod;
					
//					f[i][j][0]+=f[cnt_L][match_L][0]*f[cnt_R][j-match_L][1];
					if (j-match_L-1<0) continue;
//					if (i==3&&j==2&&f[cnt_L][match_L][0]*f[cnt_R][j-match_L-1][1]*t) {
//						printf("G %d %d\n",cnt_L,match_L);
//					}
//					
					if (cnt_L) (f[i][j][1]+=(LL)f[cnt_L][match_L][0]*f[cnt_R][j-match_L-1][1]%mod*t%mod)%=mod;
					if (j-match_L-1) (f[i][j][1]+=(LL)f[cnt_L][match_L][1]*f[cnt_R][j-match_L-1][0]%mod*t%mod)%=mod;
					(f[i][j][1]+=(LL)f[cnt_L][match_L][0]*f[cnt_R][j-match_L-1][0]%mod*t%mod)%=mod;
//					f[i][j][1]+=f[cnt_L][match_L][0]*f[cnt_R][j-match_L-1][0]*C[i-1][cnt_L];
				}
			}
//			printf("%d %d   %d %d\n",i,j,f[i][j][0],f[i][j][1]);
		}
	}
	printf("%d\n",(f[n][m][1]+f[n][m][0])%mod);
	return 0;
}
