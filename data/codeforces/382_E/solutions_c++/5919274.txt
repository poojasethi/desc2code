#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define mo 1000000007
#define N 60
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,u,i1,i2,s;
int64 f[N][N][2],C[N][N];
inline void M(int j,int k,int64 A) {
	A%=mo; f[i][j][k]=(f[i][j][k]+A)%mo;
}
inline int64 pow(int64 x,int64 y) {
	int64 an=1;
	for (;y;y>>=1) {
		if (y&1) an=an*x%mo;
		x=x*x%mo;
	}
	return an;
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,0,n) {
		C[i][0]=1;
		For(j,1,i) C[i][j]=(C[i-1][j]+C[i-1][j-1])%mo;
	}
	f[1][0][0]=1;
	int64 P=pow(2,mo-2);
	For(i,2,n) {
		s=i==n?1:i;
		For(j,0,i-1) {
			M(j+1,1,s*f[i-1][j][0]);
			M(j,0,s*f[i-1][j][1]);
		}
		for (j=1;j+1<i;j++)For(k,0,j)For(i1,0,1) {
			int A=i-1-j;
			For(u,0,A)For(i2,0,1) {
				int64 B=f[j][k][i1]*C[i-1][j]%mo*f[A][u][i2]%mo*s%mo*P%mo;
				int C=k+u+(!i1||!i2),D=!(i1&&i2);
				M(C,D,B);
			}
		}
	}
	printf("%I64d\n",(f[n][m][0]+f[n][m][1])%mo);
	//for(;;);
	return 0;
}
