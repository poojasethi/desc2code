#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cstring>

using namespace std;

typedef long long LL;
const int N = 50 + 1;
const int MOD = 1e9+7;

int n, k, c[N][N], f[N][N][2];

int main()
{
	scanf( "%d%d", &n, &k );
	c[0][0] = 1;
	for ( int i = 1; i <= 50; i ++ )
		for ( int j = 0; j <= i; j ++ )
			c[i][j] = (c[i-1][j]+c[i-1][j-1])%MOD;
	f[1][0][0] = f[0][0][1] = 1;
	for ( int i = 2; i <= 50; i ++ )
		for ( int l = 0; l*2 < i; l ++ ){
			int r = i-l-1, s = c[i-1][l];
			if ( l == r ) s = c[i-2][l-1];
			if ( i != n ) s = (LL)s*i%MOD;

			for ( int pl = 0; pl <= l/2; pl ++ )
				for ( int pr = 0; pr <= r/2; pr ++ )
					for ( int cl = 0; cl < 2; cl ++ )
						for ( int cr = 0; cr < 2; cr ++ ){
							int t = !( cl & cr );
							int val = (LL)f[l][pl][cl]*(LL)f[r][pr][cr]%MOD*s%MOD;
							f[i][pl+pr+t][t] = (f[i][pl+pr+t][t]+val)%MOD;
						}
		}
	printf( "%d\n", (f[n][k][0]+f[n][k][1])%MOD );
	return 0;
}
