#include<cstdio>
#include<iostream>
#define min(a,b) (a<b?a:b)
#define max(a,b) (a>b?a:b)
using namespace std;
typedef long long LL;
const LL p=1e9+7;
LL g[60][60][2],C[60][60];
int main(){
	for(int i=0;i<=50;i++){
		C[i][0]=1;
		for(int j=1;j<=i;j++)C[i][j]=(C[i-1][j-1]+C[i-1][j])%p;
	}
	int n0,k0;
	scanf("%d%d",&n0,&k0);
	g[0][0][1]=g[1][0][0]=1;
	for(int n=2;n<=n0;n++){
		for(int k=0;2*k<=min(2*k0,n);k++){
			for(int nl=0;2*nl<n;nl++){
				for(int kl=0;2*kl<=min(2*k,nl);kl++){
					int nr=n-nl-1,kr=k-kl;
					LL mul=C[n-1][nl]*max(nl,1)*nr%p;
					if(nl==nr)mul=mul*(p+1)/2%p;
					g[n][k][0]=(g[n][k][0]+g[nl][kl][1]*g[nr][kr][1]%p*mul)%p;
					if(kr--){
						g[n][k][1]=(g[n][k][1]+g[nl][kl][0]*g[nr][kr][0]%p*mul)%p;
						g[n][k][1]=(g[n][k][1]+g[nl][kl][0]*g[nr][kr][1]%p*mul)%p;
						g[n][k][1]=(g[n][k][1]+g[nl][kl][1]*g[nr][kr][0]%p*mul)%p;
					}
				}
			}
		}
	}
	printf("%lld\n",(g[n0][k0][0]+g[n0][k0][1])%p);
	return 0;
}
