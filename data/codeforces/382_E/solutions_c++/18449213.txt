#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<cmath>
#include<iostream>
#define N 55
#define mmod 1000000007
#define LL long long
using namespace std;
LL f[N][N][2],n,m,ans,c[N][N];
LL qmod(LL a,LL b)
{
	LL t=1;
	while(b)
	{
		if(b%2) t=((LL)t*(LL)a)%mmod;
		b/=2;
		a=((LL)a*(LL)a)%mmod;
	}
	return t;
}
int main()
{
	scanf("%I64d%I64d",&n,&m);
	c[0][0]=1;
	for(LL i=1;i<=n;i++)
	{
		c[i][0]=1;
		for(LL j=1;j<=i;j++) c[i][j]=(c[i-1][j]+c[i-1][j-1])%mmod;
	}
	f[1][0][0]=1;
	for(LL i=2;i<=n;i++)
		for(LL j=0;j<=m;j++) 
		{
			f[i][j][0]=((LL)f[i][j][0]+(LL)f[i-1][j][1]*(LL)i)%mmod;
			if(j) f[i][j][1]=((LL)f[i][j][1]+(LL)f[i-1][j-1][0]*(LL)i)%mmod;
			for(LL k=1;k<=(i-1)/2;k++)
			{
				for(LL l=0;l<=j;l++)
				{
					if(k==i-k-1 && l>j-l) break;
					LL t=((LL)f[k][l][1]*(LL)f[i-1-k][j-l][1])%mmod;
					t=((t*(LL)i)%mmod*(LL)c[i-1][k])%mmod;
					if(k==i-1-k && l==j-l) t=(t*(LL)(mmod+1)/2)%mmod;
					f[i][j][0]=(f[i][j][0]+t)%mmod;
					if(k==i-1-k && l>j-l-1) break;
					if(l<j && j)
					{
						LL tt=0;
						tt+=((LL)f[k][l][0]*(LL)f[i-1-k][j-l-1][1])%mmod;
						tt+=((LL)f[k][l][1]*(LL)f[i-1-k][j-l-1][0])%mmod;
						tt+=((LL)f[k][l][0]*(LL)f[i-1-k][j-l-1][0])%mmod;
						tt%=mmod;
						tt=((tt*(LL)i)%mmod*(LL)c[i-1][k])%mmod;
						if(k==i-1-k && l==j-l-1) tt=(tt*(LL)(mmod+1)/2)%mmod;
						f[i][j][1]=(f[i][j][1]+tt)%mmod;
					}
				}
			}
		}
	ans=(f[n][m][0]+f[n][m][1])%mmod;
	ans=(ans*(LL)qmod(n,mmod-2))%mmod;
	printf("%I64d\n",ans);
	return 0;
}
