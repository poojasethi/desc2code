#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define mo 1000000007
using namespace std;
const int N=55;
long long f[N][N][2],c[N][N];
int n,k;
int main()
{
//	freopen("382.in","r",stdin);
//	freopen("382.out","w",stdout);
	for (int i=0;i<=50;i++) c[i][0]=1;
	for (int i=1;i<=50;i++)
		for (int j=1;j<=i;j++)
			c[i][j]=(c[i-1][j-1]+c[i-1][j])%mo;
	f[0][0][1]=f[1][0][0]=1;
	for (int i=2;i<=50;i++)
		for (int j=0;j*2<i;j++)
		{
			k=i-j-1;
			long long s=(c[i-1][j]*k)%mo;
			if (j) s=(s*j)%mo;
			if (j==k) s=(s*500000004)%mo;
			for (int l1=0;l1<=j/2;l1++)
				for (int l2=0;l2<=k/2;l2++)
				{
					f[i][l1+l2+1][1]=(f[i][l1+l2+1][1]+(f[j][l1][0]*f[k][l2][0]%mo*s)%mo)%mo;
					f[i][l1+l2+1][1]=(f[i][l1+l2+1][1]+(f[j][l1][1]*f[k][l2][0]%mo*s)%mo)%mo;
					f[i][l1+l2+1][1]=(f[i][l1+l2+1][1]+(f[j][l1][0]*f[k][l2][1]%mo*s)%mo)%mo;
					f[i][l1+l2][0]=(f[i][l1+l2][0]+(f[j][l1][1]*f[k][l2][1]%mo*s)%mo)%mo;
				}
		}
	scanf("%d%d",&n,&k);
	printf("%I64d\n",(f[n][k][0]+f[n][k][1])%mo);
	return 0;
}
