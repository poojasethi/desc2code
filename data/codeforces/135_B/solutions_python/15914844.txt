l=[tuple(map(int,raw_input().split()))for i in [0]*8]
from itertools import permutations as P

def D(a,b):
    return (l[a-1][0]-l[b-1][0])**2+(l[a-1][1]-l[b-1][1])**2

def p(a,b,c,d):
    return D(a,b)==D(c,d) and D(b,c)==D(a,d) and D(a,c) == D(b,d)
def q(a,b,c,d):
    return p(a,b,c,d) * (D(a,b) == D(b,c))
for s in P(range(1,9)):
    if p(*s[:4]) * q(*s[4:]):
        print "YES"
        X=lambda x:' '.join(map(str,x))
        print X(s[4:])
        print X(s[:4])
        exit()
print "NO"
