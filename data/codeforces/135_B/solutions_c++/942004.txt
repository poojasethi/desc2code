int dis(int i,int j)
{
    return (x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
}
int ch()
{
    for(int i=0;i<4;i++)
    if(dis(p[i],p[(i+1)%4])!=dis(p[(i+1)%4],p[(i+2)%4]))return 0;

    if(dis(p[4],p[6])!=dis(p[5],p[7]))return 0;

}
