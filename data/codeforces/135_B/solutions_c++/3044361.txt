int a[9],b[9],p[9];
int d2(int i, int j)
{
 return (a[i]-a[j])*(a[i]-a[j])+(b[i]-b[j])*(b[i]-b[j]);
}

bool square(int i1, int i2,int i3,int i4)
{
if ((d2(i1,i2)==d2(i2,i3))&&(d2(i2,i3)==d2(i3,i4))&&(d2(i3,i4)==d2(i4,i1))&&
  (d2(i1,i3)==d2(i2,i4))) return 1;
  else return 0;
}
bool rectangle(int i1, int i2,int i3,int i4)
{
 if ((d2(i1,i2)==d2(i3,i4))&&(d2(i2,i3)==d2(i4,i1))&&
  (d2(i1,i3)==d2(i2,i4))) return 1;
  else return 0;

}
