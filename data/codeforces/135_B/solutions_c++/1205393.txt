int dot(point a,point b,point c)
{
  return !((a.x-b.x)*(c.x-b.x)+(a.y-b.y)*(c.y-b.y));
}
int dis(point a,point b)
{
  return (a.x-b.x)*(a.x-b.x)+(a.y-b.y)*(a.y-b.y);
}
int rec(point a,point b,point c,point d)
{
  return dot(a,b,c)&&dot(b,c,d)&&dot(c,d,a)&&dot(d,a,b);
}
int sq(point a,point b,point c,point d)
{
  return rec(a,b,c,d)&&dis(a,b)==dis(b,c)&&dis(b,c)==dis(c,d);
}
