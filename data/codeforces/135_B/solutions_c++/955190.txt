#include <cstdio>
#include <algorithm>
using namespace std;

int x[8], y[8], a[8];

int sqr(int x) {
    return x * x;
}

bool check(int * a, bool checklen) {
    bool quadro = true, oklen = true;
    int ovx = x[a[0]] - x[a[3]], ovy = y[a[0]] - y[a[3]], len = sqr(ovx) + sqr(ovy);
    for (int i = 0; i < 4; ++i) {
        int vx = x[a[(i + 1) % 4]] - x[a[i]], vy = y[a[(i + 1) % 4]] - y[a[i]];
        oklen &= (sqr(vx) + sqr(vy) == len);
        quadro &= (ovx * vx + ovy * vy == 0);
        ovx = vx, ovy = vy;
    }
    return quadro && (!checklen || oklen);
}

int main() {
    for (int i = 0; i < 8; ++i)
        scanf("%d%d", &x[i], &y[i]), a[i] = i;
    do {
        if (check(a, true) && check(a + 4, false)) {
            printf("YES\n");
            for (int i = 0; i < 8; ++i)
                printf("%d%c", a[i] + 1, (i % 4 != 3) ? ' ' : '\n');
            return 0;
        }
    } while (next_permutation(a, a + 8));
    printf("NO\n");
    return 0;
}
