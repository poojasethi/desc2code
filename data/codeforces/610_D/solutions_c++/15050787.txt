#include<bits/stdc++.h>
#define N 888888
using namespace std;int n,i,l,r,t[N];double ans,gg[N],sm[N],x1,x2,y2,X1,X2,Y1,Y2;
struct P{double x,z,y;int f;}a[N];
bool cmp(P a,P b){return a.y<b.y;}
void ps(int k,int l,int r){if(t[k])sm[k]=gg[r+1]-gg[l];else if(l==r)sm[k]=0;else sm[k]=sm[k<<1]+sm[k<<1|1];}
void up(int k,int l,int r,int x,int y,int f){
	if(x==l&&y==r){t[k]+=f;ps(k,l,r);return;}
    int mid=(l+r)>>1;
    if(x>mid)up(k<<1|1,mid+1,r,x,y,f);else if(y<=mid)up(k<<1,l,mid,x,y,f); 
    else up(k<<1,l,mid,x,mid,f),up(k<<1|1,mid+1,r,mid+1,y,f);
    ps(k,l,r);
}
int df(double x){for(int l=1,r=2*n,mid;l<=r;)if(gg[mid=l+r>>1]<x)l=mid+1;else if(gg[mid]==x)return mid;else r=mid-1;}
int main(){
	double y1;double uu;
    for(scanf("%d",&n),i=1;i<=n;i++){
        scanf("%lf%lf%lf%lf",&X1,&Y1,&X2,&Y2);
		x1=min(X1,X2);x2=max(X1,X2);y1=min(Y1,Y2);y2=max(Y1,Y2);
        x1-=1;y1-=1;a[2*i-1].x=a[2*i].x=x1;
        gg[2*i-1]=x1;gg[2*i]=x2;
        a[2*i-1].y=y1;a[2*i].y=y2;a[2*i-1].z=a[2*i].z=x2;
        a[2*i-1].f=1;a[2*i].f=-1;
    }
    for(sort(gg+1,gg+1+n*2),sort(a+1,a+1+n*2,cmp),i=1;i<=2*n;i++){
        l=df(a[i].x),r=df(a[i].z)-1;if(l<=r)up(1,1,2*n,l,r,a[i].f);
        ans=ans+sm[1]*(a[i+1].y-a[i].y);
    }printf("%lld",(long long)ans);
}
