import re
from calendar import monthrange
s = raw_input()
g = {}
for t in re.findall('(?=(0[1-9]|[12][0-9]|3[01])-(0[1-9]|1[012])-(201[345]))', s):
    if int(t[0]) <= monthrange(int(t[2]), int(t[1]))[1]:
        d = '-'.join(t)
        if not g.has_key(d):
            g[d] = 1
        else:
            g[d] += 1
print sorted([i for i in g.iteritems()], key=lambda x: x[1], reverse=True)[0][0]
