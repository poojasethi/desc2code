#include <cstdio>
#include <cstring>

int cnt[21000000], mx=-1, ans=0;
char a[100005];
int N;
int M[] = {0,31,28,31,30,31,30,31,31,30,31,30,31};
int main() {
	gets(a);
	int N = strlen(a);
	for (int i=0;i+9<N;++i) {
		bool ok = true;
		for (int j=0;j<10;++j) {
			if (j==2 || j==5) {
				if (a[i+j]!='-') ok = false;
			} else if (a[i+j]<'0' || a[i+j]>'9') ok = false;
		}
		if (!ok) continue;
		int d,m,y;
		sscanf(a+i, "%2d-%2d-%4d", &d,&m,&y);
		if (y>=2013 && y<=2015) {
			if (m>=1 && m<=12) {
				if (d>=1 && d<=M[m]) {
					int key = y*10000 + m*100 + d;
					cnt[key]++;
					if (cnt[key]>mx){
						mx = cnt[key];
						ans = key;
					}
				}
			}
		}
	}	
	printf("%02d-%02d-%d\n", ans%100, ans/100%100, ans/10000);	
	return 0;
}
