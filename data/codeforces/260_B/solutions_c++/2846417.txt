#include<stdio.h>
#include<string.h>
#include<ctype.h>
int ans[5][50][50]={0};
int ansy=0,ansm=0,ansd=0;
char a[200000];
int cor[12]={31,28,31,30,31,30,31,31,30,31,30,31};

void work(char *x)
{
	if 	(isdigit(x[0])&&isdigit(x[1])
		&&isdigit(x[3])&&isdigit(x[4])
		&&(x[9]>='3'&&x[9]<='5')   ) 
	{
		int y=x[9]-'3';
		int m=(x[3]-'0')*10+x[4]-'0';
		int d=(x[0]-'0')*10+x[1]-'0';
		if (m>0&&m<13&&d>0&&d<=cor[m-1]) 
			if (++ans[y][m][d]>ans[ansy][ansm][ansd]) {
				ansy=y;
				ansm=m;
				ansd=d;
			}
	} 
}
int main(){
	scanf("%s",a);
	int len=strlen(a);
	for (int i=0;i<len-9;i++) 
	if (a[i+2]=='-'&&a[i+5]=='-'&&a[i+6]=='2'&&a[i+7]=='0'&&a[i+8]=='1') work(&a[i]);
	
	if (ansd<10) printf("0");printf("%d-",ansd);
	if (ansm<10) printf("0");printf("%d-",ansm);
	printf("%d\n",2013+ansy);
	return 0;
}
