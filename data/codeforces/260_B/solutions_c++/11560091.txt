#include <stdio.h>
#include <iostream>
#include <map>
#include <string>
using namespace std;

map<string,int> u;
const int day[15]={0,31,28,31,30,31,30,31,31,30,31,30,31};

int main() {
    string n,x;
    cin >> n;
    int d,m,y,r,mx = 0;
    string ans = "";
    for(int i = 0 ; i + 10 <=n.length();i++){
        x = n.substr(i,10);
        if(sscanf((x + "*1").c_str(),"%2d-%2d-%4d*%d",&d,&m,&y,&r) == 4){
            if(m > 0 && m <= 12 && y >= 2013 && y <= 2015 && d > 0 && d <= day[m]){
                u[x]++;
                if(u[x] > mx){
                    mx = u[x];
                    ans = x;
                }
            }
        }
    }
    cout << ans << endl;
    return 0;
}
