#include <iostream>
#include <cstring>
#include <string>
#include <cstdio>
using namespace std;

string a;
int dates[32][13][3];
int months[] = { 0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };

int main() {
	cin >> a;
	int year, month, day, dummyend;
	string result, tmp;
	int n = a.length();
	int maxdate = 0;
	for (int i = 0; i < n - 9; i++) {
		tmp = a.substr(i, 10);
		if (sscanf((tmp + "&0").c_str(), "%2d-%2d-%4d&%d", &day, &month, &year, &dummyend) == 4) {
			if (year >= 2013 && year <= 2015) {
				if (month > 0 && month <= 12) {
					if (day > 0 && day <= months[month]) {
						if (++dates[day][month][year - 2013] > maxdate) {
							maxdate = dates[day][month][year - 2013];
							result = tmp;
						}
					}
				}
			}
		}
	}
	cout << result.c_str() << endl;
	return 0;

}
