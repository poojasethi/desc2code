#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
#include<set>
#include<map>
#include<vector>
using namespace std;
map<string ,int>mp;
const int day[13]={0,31,28,31,30,31,30,31,31,30,31,30,32};
int n,d,f,m,j,y;
string str,v,x;
int main(){
	cin>>str;
	for(int i=0;i<=str.length()-10;i++)
	{
		x=str.substr(i,10);
		if(sscanf((x+"*1").c_str(),"%2d-%2d-%4d*%d",&d,&m,&y,&j)!=4) continue ;
		if (y<2013 || y>2015 || m<1 || m>12 || d<1 || d>day[m]) continue ;
		mp[x]++;
		if (n<mp[x]) n=mp[x],v=x;
	}
	cout<<v;
}
