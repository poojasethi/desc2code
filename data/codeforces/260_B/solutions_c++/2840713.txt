#include<iostream>
#include<map>
using namespace std;
int main()
{
int i,j,a[]={31,28,31,30,31,30,31,31,30,31,30,31},x,y,z,p=0;
map<int,int> v;
string s;
getline(cin,s);
for(i=0;i<s.length()-9;i++)
{
if(s[i+2]=='-' && s[i+5]=='-')
{
for(j=i;j<i+10;j++) if(s[j]=='-' && j!=i+2 && j!=i+5) break;
if(j<i+10) continue;
x=(s[i]-'0')*10+s[i+1]-'0';
y=(s[i+3]-'0')*10+s[i+4]-'0';
z=(s[i+6]-'0')*1000+(s[i+7]-'0')*100+(s[i+8]-'0')*10+s[i+9]-'0';
if(z>=2013 && z<=2015 && y>=1 && y<=12 && x>=1 && x<=a[y-1])
{
v[x*1000000+y*10000+z]++;
if(v[x*1000000+y*10000+z]>v[p]) p=x*1000000+y*10000+z;
}
}
}
cout<<p/10000000<<(p/1000000)%10<<"-"<<(p/10000%100)/10<<(p/10000%100)%10<<"-"<<p%10000;
return 0;
}
