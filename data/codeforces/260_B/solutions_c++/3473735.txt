#include <iostream>
#include <string>
#include <map>

using namespace std;

int stoi(string s)
{
  int x=0;
  for (int i=0;i<int(s.size());i++)
    x=x*10+s[i]-'0';
  return x;
}

int dias[]={31,28,31,30,31,30,31,31,30,31,30,31};

int main()
{
  string s;
  cin>>s;
  map<string,int> mapa;
  for (int i=0;i<=int(s.size())-10;i++) {
    if (s[i+2]=='-' and s[i+5]=='-') {
      int numdig=0;
      for (int j=0;j<10;j++)
	if (s[i+j]>='0' and s[i+j]<='9')
	  numdig++;
      if (numdig==8) {
	int d=stoi(s.substr(i,2));
	int m=stoi(s.substr(i+3,2));
	int y=stoi(s.substr(i+6,4));
	if (y>=2013 and y<=2015 and m>=1 and m<=12 and d>=1 and d<=dias[m-1])
	  mapa[s.substr(i,10)]++;
      }
    }
  }
  string smax="";
  for (map<string,int>::iterator it=mapa.begin();it!=mapa.end();it++)
    if (it->second>mapa[smax])
      smax=it->first;
  cout<<smax<<endl;
}
