#include <stdio.h>
#include <iostream>
#include <string>
#include <map>
#include <cstdlib>
using namespace std;

int dy[13] = {0,31,28,31,30,31,30,31,31,30,31,30,31};
map<string,int> mp;
int main()
{
    string s;
    cin >> s;
    string c;
    int d , m , y;
    int j;
    int max = 0;
    string ans;
    for(int i = 0 ; i <= (int)s.length()-10; i++)
    {
        c = s.substr(i,10);
        if(sscanf((c+"_0").c_str(),"%2d-%2d-%4d_%d",&d,&m,&y,&j)!=4)continue;
        if(y< 2013 || y > 2015)continue;
        if(m < 1 || m > 12)continue;
        if(d < 1 || d > dy[m])continue;
        mp[c]+=1;
        if(max < mp[c])max = mp[c],ans=c;
        }
    cout << ans ;
    return 0;
}
