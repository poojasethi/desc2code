#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
const int maxn=100100;
char s[maxn];
int ansM,ansD,ansY,d[9]={0,0,1,3,4,6,7,8,9};
int rem,n,a[maxn],c[maxn],day[13]={0,31,28,31,30,31,30,31,31,30,31,30,31};
int main()
{
	s[0]=' ';scanf("%s",&s[1]);n=strlen(s)-1;
	for(int i=1;i<=n;i++)if(s[i]=='-')a[i]=10;else a[i]=s[i]-'0';
	for(int i=1;i<=n-9;i++)
	{
		int f=1;
		for(int j=1;j<=8;j++)if(a[i+d[j]]==10)f=0;
		if(a[i+2]!=10||a[i+5]!=10)f=0;
		if(!f)continue;
		int d=a[i]*10+a[i+1],m=a[i+3]*10+a[i+4];
		int y=1000*a[i+6]+100*a[i+7]+10*a[i+8]+a[i+9];
		if(!(2013<=y&&y<=2015&&1<=m&&m<=12&&d<=day[m]&&d>=1))continue;
		if((f=++c[(y-2013)*10000+m*100+d])>rem)rem=f,ansM=m,ansD=d,ansY=y;	
	}
	if(ansD<10)cout<<0;cout<<ansD<<"-";
	if(ansM<10)cout<<0;cout<<ansM<<"-";
	cout<<ansY;return 0;
}
