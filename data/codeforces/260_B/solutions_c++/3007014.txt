#include <cstdio>
#include <iostream>
#include <map>
#include <string>
using namespace std;

int main(void)
{
	int year, month, date, max_count=0;
	string number1, answer;
	map<string,int> data;
	const int day[13]={0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

	cin >> number1;
	for (int i=0; i+10<=number1.size(); ++i)	{
		int j;
		string number2=number1.substr(i,10);
		if (sscanf( (number2+"*1").c_str(),"%2d-%2d-%4d*%d", &date, &month, &year, &j ) != 4)	continue;
		if (year<2013 || 2015<year)	continue;
		if (month<1 || 12<month)	continue;
		if (date<1 || day[month]<date)	continue;
		if ( max_count<++data[number2] )	max_count=data[number2], answer=number2;
	}
	cout << answer << endl;
	return 0;
}
