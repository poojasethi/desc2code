#include <bits/stdc++.h>

using namespace std;

int mns[12] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

vector<int> is(string s2){
    vector<int> x(3);
    char c[2];
    stringstream ss(s2);
    ss >> x[0] >> c[0] >> x[1] >> c[1] >> x[2];
    if(c[0] != '-' || c[1] != '-'|| s2.find("-") != 2 || s2.find_last_of("-") != 5)x[2] = -1;
    return x;
}

int main()
{
    vector<int> nms;
    string s, tmp;
    map<string,int> crc;
    cin >> s;
    string mx = "";
    for(int i = 0; i + 9 < s.size(); i++){
        tmp = s.substr(i, 10);
        nms = is(tmp);
        if(nms[2] >= 2013 && nms[2] <= 2015 && nms[1] > 0 && nms[1] < 13 && nms[0] > 0 && nms[0] <= mns[nms[1] - 1]){
            crc[tmp]++;
            if(crc[tmp] > crc[mx])mx = tmp;
            i += 7;
        }
    }
    cout << mx;
}
