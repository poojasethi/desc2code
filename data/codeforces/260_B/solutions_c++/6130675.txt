#include<iostream>
#include<map>
#include<vector>
#include<cstdio>
#include<cstring>
#include<string>
#include<stdio.h>
#include<cstdlib>
using namespace std;
int day[]={0,31,28,31,30,31,30,31,31,30,31,30,31},n;
map<string,int> a;
string x;
int main()
{
	int q,m,d,y;
	string date,max;
	cin>>x;
	for(int i=0;i<=((int)x.size()-10);i++)
	{
		date=x.substr(i,10);
		if(sscanf((date+"*1").c_str(),"%2d-%2d-%4d*%d",&d,&m,&y,&q) !=4) continue;
		if(y<2013||y>2015||m<1||m>12||d<1||d>day[m]) continue;
		if(n<++a[date]) n=a[date],max=date;
	}
	cout<<max;
	return 0;
}
