#include <iostream>
#include <string>
using namespace std;

int main(){
  //ymmdd
  int a[21232] = {0}, max=0, i;
  string s, maxday="";
  cin >> s;
  int dom[13] = {0,31,28,31,30,31,30,31,31,30,31,30,31};
  int size = s.size(), day, month, year, tag;
  for(i=9; i<size; i++){
    if(s[i-7] == '-' && s[i-4] == '-'
    && s[i-3] =='2'&& s[i-2] == '0' && s[i-1] == '1'
    && s[i-9] >= '0' && s[i-8] >= '0' && s[i-6] >= '0' 
    && s[i-5] >= '0' && s[i] >= '0'){
      day = (s[i-9] - '0')*10 + (s[i-8] - '0');
      month = (s[i-6] - '0')*10 + (s[i-5] - '0');
      year = s[i] - '3';
      if(0<=year && year<=2 && 1<=month && month<=12
      && 1<=day && day<=dom[month]){
        tag = 10000*year + 100*month + day;
        a[tag]++;
        if(a[tag]>max){
          max = a[tag];
          maxday = s.substr(i-9, 10);
        }
      }
    }
  }
  cout << maxday;
}
