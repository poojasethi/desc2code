#include <bits/stdc++.h>

using namespace std;

string s;
int dd[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
map <string, int> p;

int main()
{
    getline(cin, s);
    for (int i = 0; i <= s.length() - 10; ++i)
    {
        string t = s.substr(i, 10);
        int d, m, y, f;
        if (sscanf((t + "#1").c_str(), "%2d-%2d-%4d#%1d", &d, &m, &y, &f) != 4) continue;
        if (y < 2013 || y > 2015 || m < 1 || m > 12 || d < 1 || d > dd[m]) continue;
//        cout << "ok\n";
        p[t]++;
    }
    map<string, int>::iterator ans = p.begin();
    for (map<string, int>::iterator it = p.begin(); it != p.end(); ++it)
        if (it->second > ans->second) ans = it;
    cout << ans->first << endl;
    return 0;
}
