#include<iostream>
#include<cstring>
#include<map>
using namespace std;
int res;
string s,ans;
map<string,int>g;
int month[13]={0,31,28,31,30,31,30,31,31,30,31,30,31};
bool num(char x)
{
	return(x>='0'&&x<='9');
}
bool ok(string x)
{
	int d=(x[0]-'0')*10+x[1]-'0',m=(x[3]-'0')*10+x[4]-'0';
	if(m<=0||m>12)return false;
	if(d<=0||d>month[m])return false;
	return true;
}
int main()
{
	cin>>s;
	for(int i=2;i<s.size();i++)
	if(s[i]=='-')
	{
		string a="";
		for(int j=i-2;j<i+8;j++)a+=s[j];
		if(num(s[i-2])&&num(s[i-1]))
		if(num(s[i+1])&&num(s[i+2]))
		if(s[i+3]=='-')
		if(s[i+4]=='2'&&s[i+5]=='0'&&s[i+6]=='1')
		if(s[i+7]=='3'||s[i+7]=='4'||s[i+7]=='5')
		if(ok(a))g[a]++;
	}
	for(map<string,int>::iterator i=g.begin();i!=g.end();i++)
		if(i->second>res)
			res=i->second,ans=i->first;
	cout<<ans<<endl;
	return 0;
}
