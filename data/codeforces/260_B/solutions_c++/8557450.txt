#include<iostream>
#include<cstdio>
using namespace std;
const int c[11][2]={{1,0},{2,0},{2,3},{4,0},{5,0},{2,6},{7,0},{8,0},{9,6},{10,3},{2,3}};
const int day[12]={31,28,31,30,31,30,31,31,30,31,30,31};
int hash[3][12][31],d[1200],cnt,ans=1,w;
char a[120000],res[1200][11];
void pushdata(const char *p)
{
	static int dd,mm,yy;
	static int *t;
	if(*(p+6)!='2'||*(p+7)!=48||*(p+8)!='1')return;
	dd=10*(*(p+0)-48)+(*(p+1)-48);
	mm=10*(*(p+3)-48)+(*(p+4)-48);
	yy=2010+(*(p+9)-48);
	if(yy<2013||yy>2015)return;
	if(mm<1||mm>12)return;
	if(dd<1||dd>day[mm-1])return;
	t=&hash[yy-2013][mm-1][dd-1];
	if(*t==0)*t=++cnt,sprintf(res[cnt],"%02d-%02d-%d",dd,mm,yy);
	d[*t]++;
}
int main()
{
	cin>>a;
	for(int i=0;a[i];i++)
	{
		w=c[w][(a[i]=='-')];
		if(w==10)pushdata(a+(i-9));
	}
	for(int i=2;i<=cnt;i++)
		if(d[i]>d[ans])ans=i;
	cout<<res[ans]<<endl;
	return 0;
}
