#include <bits/stdc++.h>

using namespace std;

int main()
{
    int month[13]={0,31,28,31,30,31,30,31,31,30,31,30,31};
    string s , ans,tmp;
    int c=0,i,l,dd,mm,yyyy,j;
    map<string,int> mp;
    cin>>s;
    l=s.length();
    for(i=0;i+10<=l;i++){
        tmp=s.substr(i,10);
        if(sscanf((tmp+"*1").c_str(),"%2d-%2d-%4d*%d",&dd,&mm,&yyyy,&j)!=4)continue;
        if(yyyy>2015||yyyy<2013||mm<1||mm>12||dd<1||dd>month[mm])continue;

        mp[tmp]++;
        if(mp[tmp]>c){c=mp[tmp]; ans=tmp;}
    }
    cout<<ans<<endl;







    return 0;
}
