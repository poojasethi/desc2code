#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
string convert(int a)
{
	string str = "";
	if(a<10)
	{
		str+="0";
		str+= (a+48);
		return str;
	}
		
	int i=0;
	while(a!=0)
	{
		str += (a%10)+48;
		a/=10;
		++i;
	}
	reverse(&str[0],&str[i]);
	return str;
}
main()
{
	string s;
	cin>>s;
	string final = "";
	int max=0;
	
	int A[13]={0,30,28,31,30,31,30,31,31,30,31,30,31};
	for(int i=2013;i<=2015;++i)
		for(int j=1;j<=12;++j)
			for(int k=1;k<=A[j];++k)
			{
				int count =0;
				string st = "";
				st += convert(k)+"-"+convert(j)+"-"+convert(i);
				size_t found = -1;
				do
				{
					found=s.find(st,found+1);
					if(found!=string::npos)
					{
						count++;
					}
				}while(found < s.length());
				if(count > max)
				{
					max = count;
					final = st;
				}
			}
		cout << final;
}
