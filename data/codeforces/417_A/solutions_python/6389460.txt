c,d = [int(x) for x in raw_input().split()]
n,m = [int(x) for x in raw_input().split()]
k = int(raw_input())
probs = 0
go = d
add = 1
if float(c)/n < float(d):
    go = c
    add = n
    
while 1:
    
    if k >= m*n:
        break
    if k + 1 == m*n:
        probs += min(c,d)
        break
    elif k + n >= m*n:
        probs += min(c, (m*n - k)*d)
        break
    else:
        k+=add
        probs+=go
print probs
