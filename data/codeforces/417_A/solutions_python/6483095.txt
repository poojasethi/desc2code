c,d = map(int,raw_input().split())
n,m = map(int,raw_input().split())
k = int(raw_input())
need = m * n - k
main_need = c/float(n)
add_need = d
if need <= 0 :
	print 0
else :
	if add_need <= main_need :
		print add_need * need
	else :
		s = c * ( need / n )
		last_need = need % n
		if c <= add_need * last_need :
			print s + c
		else :
			print s + add_need * last_need
