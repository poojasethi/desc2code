num = raw_input().strip().split(" ")
n, m = int(num[0]), int(num[1])
k = int(raw_input())
total = n * m
need = total - k
prob = 0
while need > 0:
    if n <= need:
        if c < d * n:
            need -= n
            prob += c
        else:
            need -= 1
            prob += d
    else:
        if c < d * need:
            prob += c
        else:
            prob += d * need
        need = 0
print prob
