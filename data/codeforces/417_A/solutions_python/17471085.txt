from math import ceil

c, d = map(int, raw_input().split())
n, m = map(int, raw_input().split())
k = input()

def f(c, d, n, m, k):
    p = n*m - k
    if p <= 0: return 0
    #print p, d, n
    a = (
        p * d,
        int(ceil(float(p)/n)) * c,
        (p/n) * c + (p%n) * d
    )
    #print a
    return min(a)
print f(c, d, n, m, k)
