import sys, re

f = sys.stdin

c, d = map(int, f.readline().strip().split())
n, m = map(int, f.readline().strip().split())
k = int(f.readline().strip().split()[0])

need = n * m - k

if need <= 0:
    print 0
    exit(0)

result = sys.maxint
for x in xrange(need + 1):
    for y in xrange(need + 1):
        if x * n + y >= need:
            result = min(x * c + y * d, result)
    if x * n >= need:
        break

print result
