#!/usr/bin/env python

import sys
import logging

c, d = map(int, raw_input().split())
n, m = map(int, raw_input().split())
k = input()

t = n * m - k

candidates = []

candidates += [c * (t / n + (1 if t%n != 0 else 0))]
candidates += [d * t]

candidates += [c * (t/n) + (t - n*(t/n)) * d]

print max(0, min(candidates))
