
c,d = [int(item) for item in raw_input().split(' ')]
n,m = [int(item) for item in raw_input().split(' ')]
k = int(raw_input())
for i in range(0, m + 1):
    x = i
    if x * n + k >= m * n:
        y = 0
    else:
        y = m * n - k - n * x
    ans = c * x + d * y
    if i == 0:
        mx = ans
        continue
    mx = ans if mx > ans else mx
print mx
