import sys
l=[]
for i in sys.stdin:
    l+=[int(k) for k in i.rstrip().split()]


c,d,n,m,k=l[0],l[1],l[2],l[3],l[4]

if n*m<=k:
    print 0
else:
    v=[d*i for i in range(n*m-k+1)] #quest
    
    for i in range(n,len(v)):
        v[i]=min([v[i-n]+c,v[i-1]+d])

    for i in range(n):
        if len(v)-i-1<0:
            break
        v[-1]=min([v[-1-i]+c,v[-1]])
    print v[-1]
