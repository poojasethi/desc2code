input = raw_input
c, d = map(int, input().split())
n, m = map(int, input().split())
k = int(input())

s = n*m - k
res = 0

if s > 0:
    if (c/n) > d:
        res += d*s
    else:
        os = s - (s // n) * n
        res += (s // n) * c
        if c < d*os:
            res += c
        else:
            res += d*os

print( res )
