c, d = map(int, raw_input().split())
n, m = map(int, raw_input().split())
k = int(raw_input())

problems = 0
people = n * m - k

while people > 0:
    if (d < float(c) / n) or (d < float(c) / people):
        people -= 1
        problems += d
    else:
        people -= n
        problems += c
print problems
