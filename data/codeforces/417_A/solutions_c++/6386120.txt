#include<iostream>
#include<vector>
#include<cstdio>
#include<algorithm>
using namespace std;
main()
{
	int c,d,n,m,k;
	cin>>c>>d>>n>>m>>k;
	int res=n*m-k;
	int ret=1000000000;
	for(int i=0;i<=m;i++)
	{
		int res1=n*i,ret1=i*c;
		if(res1<res)
			ret1+=(res-res1)*d;
		ret=min(ret,ret1);
	}
	cout<<ret<<"\n";
}
