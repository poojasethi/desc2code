#include<bits/stdc++.h>
using namespace std;

int main()
{
    int c, d, n, m, k, z;
    cin >> c >> d >> n >> m >> k;
    z = n*m - k;
    if (k >= n*m)cout << 0;
    else {
        int x = min((int)ceil((double)z / n) * c, z / n * c + z%n *d),
             y = z*d;
        cout << min(x, y);
    }
}
