#include <stdio.h>
int main()
{
	int c,d,n,m,k;
	scanf("%d%d",&c,&d);
	scanf("%d%d",&n,&m);
	scanf("%d",&k);
	int ans = 0x7ccccccc;
	int x,tmp;
	for (int a=0;a<=m;a++)
	{
		tmp = n * m - k - a * n;
		if (tmp<0) tmp = 0;
		x=tmp * d + a * c ;
		if (x<ans) ans = x;
	}
	printf("%d\n",ans);
	return 0;
}
