#include<iostream>

using namespace std;

int main(){
	int c,d,n,m,k,ans(0);
	cin>>c>>d;
	cin>>n>>m;
	cin>>k;
	int total=n*m;
	total-=k;
	while(total>0){
		if(c<=d*n&&c<=d*total){
			total-=n;
			ans+=c;
			//cout<<ans<<endl;
			continue;
		}
		else{
			total--;
			ans+=d;
			continue;
		}
	}
	cout<<ans;


}
