#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

int main(){
	int c,d,n,m,k;
	cin>>c>>d;
	cin>>n>>m;
	cin>>k;
	int res=max(0,n*m-k);
	int tmp=2e9;
	int i=0;
	while((i-1)*n<=res){
		int j=max(m*n-k-i*n,0);
		tmp=min(tmp,c*i+d*j);
		i++;
	}
	cout<<tmp<<endl;
	return 0;
}
