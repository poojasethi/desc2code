#include <stdio.h>
#include <string.h>
#define min(a,b) ((a) < (b) ? (a) : (b))
int main(){
    int tot, n, m, k, c, d;
    scanf("%d%d%d%d%d", &c, &d, &n, &m, &k);
    tot = n * m - k;
    printf("%d\n", tot <= 0 ? 0 : min(tot * d, min(((tot - 1) / n + 1) * c, tot / n * c + tot % n * d)));
    return 0;
}
