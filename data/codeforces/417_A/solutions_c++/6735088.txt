#include<bits/stdc++.h>

int i,j,k,n,m,c,d,top=INT_MAX,t,v;

int main(){
	scanf("%d %d %d %d %d",&c,&d,&n,&m,&k);
	v=n*m;
	v-=k;
	for(i=0;i<=v/n+1;i++){
		if(i*c+(((v-(i*n)<0)? 0 :(v-i*n))*d)<top)
			top=i*c+(((v-(i*n)<0)? 0 :(v-i*n))*d);
	}
	if(top==INT_MAX)
		top=0;
	printf("%d",top);	
	
	return 0;
}
