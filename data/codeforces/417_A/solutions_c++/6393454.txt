# include <stdio.h>
# include <algorithm>
using namespace std;

int c,d,m,n,k,t,ans;

int main()
{
	scanf("%d %d",&c,&d);
	scanf("%d %d",&n,&m);
	scanf("%d",&k);
	t=n*m-k;
	while(t>0)
	{
		if(t<=n && d*t<=c)
		{ans=ans+d*t;break;}
		
		if(c/n<d)
		{t=t-n;ans+=c;}
		else
		{t--;ans+=d;}
		
	}
	printf("%d",ans);
}
