#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxl=250;
char s[maxl];
int len,K,ans;
bool check(int x)
{
	int i,j,z;
	for(i=0;i+2*x<=len+K;i++)
	{
		z=1;
		for(j=0;x+i+j<len&&j<x;j++)
			if(s[i+j]!=s[x+i+j]) { z=0;break; }
		if(z) return 1;
	}
	return 0;
}
int main()
{
	int i,j;
	scanf("%s%d",s,&K);
	len=strlen(s);
	for(i=(len+K)>>1;i;i--)
		if(check(i)) { ans=i<<1;break; }
	printf("%d\n",ans);
	return 0;
}
