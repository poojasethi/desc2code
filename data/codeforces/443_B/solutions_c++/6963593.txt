#include <iostream>
using namespace std;
string t;
int k,r=0;
int main(){
	cin>>t;
	cin>>k;
	for(int i=0;i<t.size();i++){
		string s=t.substr(i);
		for(int j=1;;j++){
			if (2*j>s.size()+k) break;
			bool ok=1;
			for(int m=0;m<j;m++)
				if (j+m<s.size()) ok&=(s[m]==s[j+m]);
			if (ok) r=max(j,r);
		}
		
	}
	cout<<2*r;
	return 0;
}
