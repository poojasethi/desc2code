#include <iostream>
using namespace std;
string t;
int k,res=0;
int main(){
	cin>>t;
	cin>>k;
	for(int a=0;a<t.size();a++){
		string s=t.substr(a);
		for(int i=1;;i++){
			if (2*i>s.size()+k) break;
			bool ok=1;
			for(int j=0;j<i;j++){
				if (i+j<s.size()){
					ok&=(s[j]==s[i+j]);
				}
			}
			if (ok) res=max(res,i);
		}
	}
	cout<<2*res;
	return 0;
}
