#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    char s[1000];
    gets(s);
    int n,ans=0,len=0;
    cin>>n;
    for(len=0;s[len]!=0;len++);
    for(int i=0;i<len;i++)for(int j=i+1;j<len+n;j+=2)
    {
        int k=(i+j+1)/2;
        bool w=true;
        for(int l=i,m=k;l<k;l++,m++)if(s[l]!=s[m]&&m<len)w=false;
        if(w)ans=max(ans,j-i+1);
    }
    cout<<ans<<endl;
    return 0;
}
