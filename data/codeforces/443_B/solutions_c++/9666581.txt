#include<bits/stdc++.h>

using namespace std;

int main()
{
string s;
int kk;
int ans,i,j;
cin>>s;
cin>>kk;
ans=max(1,kk/2);
int n=s.size();
for(i=0;i<n;i++)
{
for(j=1;j<=n;j++)
{
int p=i,q=i+j;
while(p<i+j && p<n && q<n && s[p]==s[q])
{
p++;q++;
}
if(p==(i+j) || (i+j-p<=kk && q==n))
{
ans=max(ans,2*j);
}
}
}
if(kk>=n)
{
int u=kk+n;
if(u%2)u--;
ans=max(ans,u);
}
cout<<ans<<"\n";
return 0;
}
