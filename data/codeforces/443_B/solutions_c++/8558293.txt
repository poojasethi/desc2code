#include<stdio.h>
#include<string.h>
char str[300];
int main(void){
    int len;
    int l=0,k;
    int ok;
    scanf("%s%d",str,&k);
    len = strlen(str);
    for(int i=0;i<len;i++)
        for(int j=l+1;i+2*j<=len+k;j++){
            ok=1;
            for(int m=0;m<j&&i+j+m<len&&ok;m++)
                if(str[i+m]!=str[i+j+m])
                    ok=0;
            if(ok) l=j;
        }
    printf("%d\n",2*l);
    return 0;
}
