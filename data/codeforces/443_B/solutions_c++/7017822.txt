#include <iostream>
#include <string>
using namespace std;

int main() {
	string s ;
	cin >> s;
	int k;
	cin >> k;

	for(int i = 0 ;i < k ;i++)s+="?";

	int out = 0;
	int cnt;
	int len=s.size();
	for(int n = 1 ; n < len ;n++)
	{
		cnt = 0 ;
		for(int j = 0 ; j + n < len;j++)
		{
			if(s[j] == s[j+n] || s[j+n] == '?')
			cnt++;
			else
			cnt = 0;
			if(cnt == n)out = max(out,n);
		}


	}
	cout << 2*out << endl;
	return 0;
}
