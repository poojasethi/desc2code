#include<bits/stdc++.h>
using namespace std;
int n,len;
char s[10000];
int ma;
int j;
int ch(int x,int y){
	for(int i=0;x-y+i<x;i++){
		if(s[x-y+i]!=s[x+i]&&s[x+i]!='@')return 0;
	}
	return 1;
}
int main(){
	scanf("%s %d",s,&n);
	len=strlen(s);
	for(int i=0;i<n;i++){
		s[i+len]='@';
	}
	s[n+len]=0;
	for(int i=0;i<len+n;i++){
		for(j=1;i-j>=0;j++){
			if(ch(i,j))ma=max(ma,j);
		}
	}
	printf("%d\n",ma*2);
	return 0;
}
