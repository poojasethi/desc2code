#include<iostream>
using namespace std;

int main() {
    string in;
    int k;
    cin>>in>>k;
    int m = 0, l = in.length();
    for (int i = 0; i < l; i++)
        for (int j = m+1; j <= i && i+j<=k+l; j++)
            if (in.substr(i-j, j).find(in.substr(i, j)) == 0)
                m = j;
    m = max(m, min(k, k+l>>1));
    cout<<m*2;
}
