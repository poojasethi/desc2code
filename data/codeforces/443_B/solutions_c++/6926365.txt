#include <iostream>
using namespace std;

int main()
{
	string s;
	int k, i, n, j, ans, f;
	cin>>s>>k;
	while (k--) s+='.'; 
	for (ans=0,i=0; i<s.size()-1; i++) {
		for (n=ans+1; i+n+n<=s.size(); n++) {
			for (f=1,j=0; j<n; j++) 
				if (s[i+j]!='.' && s[i+n+j]!='.' && s[i+j]!=s[i+n+j]) {
					f=0; break;
				}
			if (f) ans=n;
		}
	}
	cout<<ans*2<<endl;
	return 0;
}
