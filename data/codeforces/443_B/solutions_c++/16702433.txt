#include<iostream>
using namespace std;
int main(){
	string s;
	long n;
	cin>>s>>n;string c="*";
	for(long i=1;i<=n;i++){
	s=s+c;
	}
	long max=0;
	for(long i=0;i<=s.size();i++){
		for(long j=i+1;j<n+s.size();j=j+2){
			long h=i;long e=j;
			long r=(e+h)/2;
		
			long f=r;r=r+1;
			while(s[h]==s[r]||s[r]=='*'&&h<=f&&r<=e){r=r+1;h=h+1;}
			if(r>e){if(max<j-i+1)max=j-i+1;
		
			}
			
		}
	}
	cout<<max<<endl;
}
