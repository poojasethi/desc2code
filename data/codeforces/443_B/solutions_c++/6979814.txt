#include <iostream>
using namespace std;
int main(){
    string a;
    int e,n,t_l,i,j;
    cin>>a>>e;
    n = a.size();
    for(t_l=(n+e)/2;t_l>=0;t_l--){
        for(i=0;i<=(n+e-2*t_l);i++){
            for(j=i;j<i+t_l;j++){
                if(a[j]!=a[j+t_l] && j+t_l < n) break;
            }
            if(j==i+t_l){
                cout<<2*t_l;
                return 0;
            }
        }
    }
}
