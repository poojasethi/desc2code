#include <bits/stdc++.h>
using namespace std;
string s;
int main(){
 	int k,n,l,c,x;
 	bool ok;	
 	cin>>s>>k;
 	n=s.length();
    for(int i=0;i<n;i++)
		for(c=l+1;i+c*2<=n+k;c++){
           for(ok=1,x=0; x<c && i+c+x<n && ok; x++)
              if(s[i+x]!=s[i+c+x])ok=0;
           if(ok)l=c;
        }
    cout<<l*2<<endl;
}
