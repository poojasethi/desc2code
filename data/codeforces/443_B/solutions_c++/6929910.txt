#include <cstdio>
#include <string.h>

char s[220];
int n, k, max;
bool t;

int main(int argc, char* argv[])
{
	scanf("%s", s);
	scanf("%d", &k);
	n = strlen(s);
	for(int i = 0; i < n+k; i++)
	{
		for(int m = 1; m < (k+n-i)/2+1; m++)
		{
			t = true;
			for(int j = i; j < i+m; j++)	
			{
				if(j+m >= n && j+m < n+k) continue;
				if(s[j] == s[j+m]) continue;
				t = false;
			}	
			if(t && m > max)
				max = m;
		}			
	}
	printf("%d\n", max*2);
	return 0;
}
