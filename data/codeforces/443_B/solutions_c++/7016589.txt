#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

int main(){
  char s[512]={0};
  int k;
  scanf("%s %d",s,&k);
  int n=strlen(s),t=(n+k)/2;
  while(t>0){
    for(int i=n+k-t*2,j;i>=0;--i){
      for(j=0;j<t;++j)if(!(i+t+j>=n||s[i+j]==s[i+t+j]))break;
      if(j>=t)goto Z;
    }
    --t;
  }
  Z:printf("%d\n",t*2);
  return 0;
}
