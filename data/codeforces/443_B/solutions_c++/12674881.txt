# include <iostream>
# include <cstdio>
# include <cstring>
using namespace std;
char s[222];

int main(void)
{
	scanf("%s",s+1);
	int k; scanf("%d",&k);
	int len=strlen(s+1);
	for(int i=(len+k)/2;i>=0;i--)
	{
		int flag;
		for(int j=1;j+2*i-1<=len+k;j++)
		{
			flag=1;
			for(int k=j;k<j+i;k++)
				if(k<=len&&k+i<=len&&s[k]!=s[k+i])
					{flag=0;break;}
			if(flag) break;
		}
		if(flag) {printf("%d\n",2*i);break;}
	}
	return 0;
}
