#include <bits/stdc++.h>
using namespace std;
int main()
{
    string s;
    int cnt,n,ans=0,k,l,i;
    cin>>s;
    cin>>k;
    n=s.length();
    for(l=1;l<=(n+k)/2;l++)
    {
        cnt=0;
        for(i=0;i<n+k-l;i++)
        {
            if(i+l >= n || s[i]==s[i+l] ) cnt+=1;
            else cnt = 0;
            if(cnt >= l) ans = l;
        }
    }
    cout<<2*ans;
    return 0;
}
