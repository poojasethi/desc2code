#include <stdio.h>
#include <iostream>
#include <cstring>
using namespace std;
char s[1024]={0};
int k,l,t;
int main( )
	{
	scanf("%s %d",s,&k);
	l=strlen(s);
	t=(l+k)/2;
	while (t>0)
	{
		int i,j; 
		for (i=l+k-t*2;i>=0;i--)
		{
			for (j=0;j<t;j++)
			if (i+t+j<l&&s[i+j]!=s[i+t+j])
			break ;
			if (j>=t)
			{
				cout<<t+t<<endl;
				return 0;
			}
		}
		t--;
	}
	return 0;
}
