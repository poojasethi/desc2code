#include<iostream>
#include<string.h>
#include<algorithm>

#define fr(i,m,n) for(int i=m; i<=n; i++)
#define dfr(i,n,m) for(int i=n; i>=m; i--)

using namespace std;


int main()
{
	char s[203];
	cin>>s; int k,ls=strlen(s);
	cin>>k; int ans=min(ls,k)*2;
	fr(i,0,ls+k-2)
	{
		fr(j,1,(ls+k-i)/2)
		{
			int f=1;
			fr(k,0,j-1) if (i+k+j<ls && s[i+k]!=s[i+k+j]) {f=0; break;}
			if (f && ans<j*2) ans=j*2;
		}
	}
	cout<<ans;
	return 0;
}
