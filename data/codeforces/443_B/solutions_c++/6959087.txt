#include <iostream>
using namespace std;
int main()
{
	int k,l=0,i,j;
	string s;
	cin>>s;
	cin>>k;
	for (i=0;i<s.size();++i)
		for (j=1;i+j+j<=s.size()+k;++j)
			if (i+j>=s.size()) l=max(l,j*2);
			else if (i+j+j>=s.size())
			{
				if (s.substr(i,s.size()-i-j)==s.substr(i+j,s.size()-i-j)) l=max(l,j*2);
			}
			else
			{
				if (s.substr(i,j)==s.substr(i+j,j)) l=max(l,j*2);
			}
	cout<<l<<endl;
	return 0;
}
