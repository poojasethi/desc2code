#include<stdio.h>
#include<string.h>
char st[500];
int ans,n,len,i,j,k,flag,temp,max;
main()
{
	scanf("%s",&st);
	len=strlen(st);
	scanf("%d",&n);
	ans=n+len;
	max=0;
	for(i=0;i<ans;i++)
		for(j=0;i+2*j-1<ans;j++)
		{
			temp=j*2;
			flag=0;
			for(k=i;k<i+j;k++)
				if(k>=len||j+k>=len||st[k]==st[j+k]) continue;
					else{flag=1;break;}
			if(flag==0)
				if(2*j>max) max=2*j;
		}
	printf("%d\n",max);
}
