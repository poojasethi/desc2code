#include<stdio.h>
#include<string.h>
int i,j,k,n,l;
char s[401];
int f(int i , int n)
{
	for(j = i ; j < i+n ; j++){
		if(s[j] != s[n+j] && n+j < l)
			return 0;
	}
	return 1;
}

int main(){
	scanf("%s",s);
	scanf("%d",&k);
	l = strlen(s);
	for(n = (l+k)/2;n>0;n--){
		for(i = 0 ; i<= l+k-2*n ; i++)
		{
			if(f(i,n) == 1){
				printf("%d\n",n*2);
				return 0;
			}
		}
	}
	return 0;
}
