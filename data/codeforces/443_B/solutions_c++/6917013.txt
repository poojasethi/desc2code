#include <iostream>
#include <string>

using namespace std;

string s;
int k,ans;

int main(void)
{
	cin>>s>>k;

	int n = s.size();
	if(n==1)
	{
		cout<<(n+k)/2<<endl;
		return 0;
	}

	for(int i=0;i<n+k;i++)
	{
		for(int d=1;d<n+k-i;d++)
		{
			int ok=0;
			for(int j=i+d;j<n+k&&j<i+2*d;j++)
			{
				if(j>=n ||j-d>=n||s[j]==s[j-d])
					ok++;
			}
			if(ok==d) ans = max(2*d,ans);
		}
	}
	cout<<ans<<endl;
	return 0;
}
