#include <iostream>
#include <stdio.h>
#include <string>

using namespace std;

int main(){
	string s;
	int k;
	cin >> s >> k;
	int ans = 0;
	s.append(k,'?');
	for(int i = 0;i<s.length()-1; i++){
		for(int j = 1; i+2*j <= s.length(); j++){
			bool allSame = true;
			for(int k = i; k < i+j;k++)
				if(s[k] != s[k+j] && !(s[k] == '?' || s[k+j] == '?'))
					allSame = false;
			if(allSame)
				ans = max(ans,2*j);
		}
	}
	cout << ans << endl;
}
