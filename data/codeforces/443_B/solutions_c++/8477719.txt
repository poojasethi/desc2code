#include<iostream>
#include<string>
#include<algorithm>

using namespace std ;

int main()
{
  string s ;
  int k , m=0 ;
  cin >> s >> k ;
  int l=s.size() ;
  for(int i=0 ; i<l ; i++)
    for(int j=m+1 ; j<=i && j+i<=l+k ; j++)
      if(s.substr(i-j,j).find(s.substr(i,j))==0)
	m=j ;
  m=max(m,min(k,k+l>>1)) ;
  cout << m*2 << endl ;
  return 0 ;
}
