#include <cstdio>
#include <cstring>

int main()
{
	char s[201];
	gets(s);
	int k; scanf("%d",&k);
	int len = strlen(s);
	for(int n = (len+k)/2;n>=1;n--)
	{
		for(int i=0;i<=len+k-2*n;i++)
		{
			int flag = 0;
			for(int a=i;a<i+n;a++)
			{
				if(a+n >= len) break;
				if(s[a] != s[a+n]) { flag = 1; break; }
			}
			if(!flag) { printf("%d\n",2*n); return 0; }
		}
	}
	return 0;
}
