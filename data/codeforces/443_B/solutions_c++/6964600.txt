#include <iostream>
#include <string>

using namespace std;

int k,l,i,c,ok,x,n;
string S;

int main ()
{
    cin >> S>> k;

    for(n=S.size();i<n;i++)
    for(c=l+1;i+c*2<=n+k;c++)
    {
        for(ok=1,x=0; x<c && i+c+x<n && ok; x++)
            if(S[i+x]!=S[i+c+x])ok=0;
        if (ok>0)
            l= c;
    }
    cout<<l*2;
}
