#include<bits/stdc++.h>
using namespace std;
string s;
long long n,mn=1;



int main () {
cin>>s>>n;

for(int i=0 ; i<s.length() ; i++){
	for(int j=mn ; 2*j<=(s.length()+n-i) ; j++){
	      int k=0 ;
		for(int qq=i ; qq<j+i && k==0 && qq+j<s.length(); qq++){
			if(s[qq]!=s[qq+j]){
				k=1;
			}
		}
		if(k==0) mn=mn>j?mn:j;
	}
	
	
	
}

cout<<2*mn;




	return 0;
}
