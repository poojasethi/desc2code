#include <bits/stdc++.h>

using namespace std;
int main(){
	string st;int l;cin>>st>>l;int s=st.size();
	for(int i=(s+l)/2;i>=1;i--){
		//seraching
		bool yon=false;
		for(int j=0;j<=(s+l)-i*2;j++){
			bool tmp=true;
			for(int k=j;k<j+i;k++){
				if(k>=s||k+i>=s)break;
				tmp=tmp&&(st[k]==st[k+i]);
			}
			yon=yon||tmp;
		}
		if(yon){
			cout<<i*2<<endl;
			return 0;
		}
	}
	return 0;
}
