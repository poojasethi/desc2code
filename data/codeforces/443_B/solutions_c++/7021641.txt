#include <iostream>
#include <vector>
#include <string>

using namespace std;

int main () {
	string s;
	int k, n;
	cin >> s;
	cin >> k;
	n = (s.length() + k) / 2;
	while(n){
		bool flag;
		for(int j = 0; j <= s.length() + k - 2*n; j++){
				flag = true;
				for(int i = j; i < j + n; i++) if(i + n < s.length() ) if(s[i] != s[i + n]){flag = false; break;}
				if(flag) break;
			}
		if(flag) break;
		n --;
	}
	cout << 2*n;
	return 0;
}
