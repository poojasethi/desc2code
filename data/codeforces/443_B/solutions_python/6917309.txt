s = raw_input().strip()
n = int(raw_input())
totallen = len(s) + n
for i in xrange(totallen / 2, 0, -1):
	for j in xrange(totallen - i * 2 + 1):
		found = True
		for k in xrange(i):
			if j + k + i >= len(s):
				break
			if s[j + k] != s[j + k + i]:
				found = False
				break
		if found:
			print i * 2
			break
	if found:
		break
