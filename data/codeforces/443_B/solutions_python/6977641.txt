import sys

def check_tandem(s):
	if len(s)%2 == 1: return False
	h = len(s)/2
	for i in range(h):
		if s[i] != '*' and s[i+h] != '*' and s[i] != s[i+h]:
			return False
	return True

lines = sys.stdin.readlines()

for i in range(len(lines)/2):
	s = lines[i*2].rstrip('\n')
	n = int(lines[i*2+1].rstrip('\n'))

	s += '*'*n
	maxs = 0
	for j in range(len(s)):
		for k in range(j+2, len(s)+1):
			if check_tandem(s[j:k]):
				maxs = max(maxs, k-j)
	print maxs
