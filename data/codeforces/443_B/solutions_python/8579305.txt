String = raw_input()
k = int(raw_input())

s = list(String)

for i in range(k):
	s.append("X")

contador = 0
c=1
max_temp = 0
max = 0

while contador<len(s)-1:
	for i in range((len(s)-contador)/2):
		for x in range(contador,c+contador):
			if s[x]!=s[x+c] and s[x]!="X" and s[x+c]!="X":
				max_temp=0
				break
			max_temp+=2
			if x==c+contador-1:
				if max_temp>max:
					max = max_temp
				max_temp=0
		c+=1
	c=1
	contador+=1

print max
