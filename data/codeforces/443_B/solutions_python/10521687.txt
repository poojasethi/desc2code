def judge(s,e):
    global r,has
    L = e-s+1
    if L&1 > 0:
        return 0
    else:
        half = L/2
        for i in range(s,s+half):
            if i+half >= has or r[i] == r[i+half]:
                continue
            else:
                return 0
        return 1


r = raw_input()
k = int(raw_input())
has = len(r)
len = has+k
res = 1
for i in range(len-1):
    for j in range(i+1,len):
        if judge(i,j):
            if j-i+1 > res:
                res = j-i+1
print res
