from sys import stdin

inp = stdin.readlines()

s = inp[0][:-1]
k = int(inp[1])

l = len(s)
tl = l+k

mt = (l+k - (l+k)%2)
ms = mt/2



def eq(i,j):
	if j < tl:
		if(j>=l):
			return True
		return s[i]==s[j]
	return False

maxt = 0
for i in range(tl):
	for k in range(1,ms+1):
		p = True
		for j in range(k):
			if not eq(i+j,i+k+j):
				p = False
				break
		if p:
			maxt = max(maxt,k)
			continue

print maxt*2
