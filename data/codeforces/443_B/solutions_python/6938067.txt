s = raw_input()
k = int(raw_input())
k1 = ''
for i in xrange(k):
    k1 += '1'
s = s + k1

def sravn(s):
    for i in xrange(len(s)/2):
        if s[i] != s[i+len(s)/2] and s[i+len(s)/2] != '1':
            return 'no'
    print len(s)
    return 'yes'

l = len(s)/2*2-1
while l >= 1:
    for i in xrange(len(s)-l):
        t = sravn(s[i:i+l+1])
        if t == 'yes':
            l = -1
            break
    l -= 2
