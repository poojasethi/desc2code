#!/usr/bin/python

import sys
from math import *

def Ni(): return tuple(map(int, sys.stdin.readline().split()))
def Ns(): return sys.stdin.readline()

s = Ns().strip()
k = Ni()[0]

for n in range((len(s) + k)/2, 0, -1):
	if k >= n:
		L = 2 * n
		break
	else:
		si = len(s) + k - n
		l = len(s) - si
		if s[si:si+l] == s[si-n:si-n+l]:
			L = 2 * n
			break

		# maybe there's one already
		L = 0
		for i in range(len(s)):
			if s[i:i+n] == s[i+n:i+2*n]:
				L = 2 * n
				break
		if L > 0:
			break

print L
