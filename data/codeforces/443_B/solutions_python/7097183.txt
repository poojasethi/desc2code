import sys

def check(s, i, j):
	return s[j] == ' ' or s[i] == ' ' or s[i] == s[j]

s = raw_input()
k = int(raw_input())
ns = list(s) + [' ' for i in xrange(k)]

for sz in xrange(2*(len(ns)/2), 1, -2):
	for j in xrange(len(ns) - (sz) + 1):
		good = True
		for i in xrange(sz/2):
			if not check(ns, j + i, j + i + sz/2):
				good = False
				break

		if good:
			print sz
			sys.exit()

print 0
