def solve(s, k):
    s += '?'*k
    n = len(s)
    res = 0
    for i in xrange(n):
        for j in xrange(i+1, n):
            res = max(res, tr(s, i, j))
    return res

def tr(s, l, r):
    d = r-l+1
    if d % 2 != 0: return -1
    m = l+d/2
    for i in range(d/2):
        a, b = s[l+i], s[m+i]
        if a == '?' or b == '?': continue
        if a != b: return -1
    return d

s = raw_input()
k = int(raw_input())
print solve(s, k)
