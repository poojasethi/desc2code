#!/usr/bin/env python

s = raw_input()
k = input()

s += "?"*k
ans = 0

i = 0
while i < len(s):
    j = 1
    while j < len(s) - i:
        cut = s[i:i+j+1:]
        j += 1
        if not len(cut)%2==0:
            continue
        tandem = True
        for k in range(len(cut)/2):
            if cut[k] != cut[k + len(cut)/2] and cut[k + len(cut)/2] != "?":
                tandem = False
                break
        if (tandem):
            ans = max(ans, len(cut))
    i += 1
    
print ans
