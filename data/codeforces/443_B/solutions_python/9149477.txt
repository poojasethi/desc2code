def isok(s, q):
    for i in range(0, len(s) - 2 * q + 1):
        j = 0
        while j < q and (s[i + j + q] == "?" or s[i + j] == s[i + j + q]):
            j += 1
        if j == q: return True
    return False

s, k = raw_input(), int(raw_input())
s += "?" * k

for i in range(len(s) // 2, -1, -1):
    if isok(s, i):
        print 2 * i
        break
