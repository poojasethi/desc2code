def fuzzy_eq(a, b):
    if len(a) != len(b):
        return False
    for i in xrange(len(a)):
        if a[i] != b[i] and b[i] != '*':
            return False
    return True

world = raw_input() + ('*' * int(raw_input()))
best_len = 0
for i in xrange(len(world)):
    ll = 0
    while i + 2*ll <= len(world):
        if fuzzy_eq(world[i:i+ll], world[i+ll:i+2*ll]) and ll > best_len:
            best_len = ll
        ll += 1
print best_len * 2
