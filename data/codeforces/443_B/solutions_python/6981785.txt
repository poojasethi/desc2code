str = raw_input()
n = len(str)
m = int(raw_input())
ans = 0
flag = 0
for x in range(0, m): str += '*'
len = n+m
for i in range(len/2, 0, -1):
    for j in range(0, len):
        if j+2*i > len: continue
        tmp = 0
        for k in range(j,j+i):
            if str[k] != str[k+i] and str[k+i] != '*':
                tmp = 1
                break
        if tmp == 0:
            ans = 2 * i
            flag = 1
            break
    if flag == 1:
        break
print ans
