input_str = input_str + '*'*no_of_char

def isTandem(substr):
    n = len(substr)//2
    for i in xrange(0, n):
        if not substr[i] == substr[i+n] and substr[i+n]!='*' and substr[i] != '*':
     #       print "Returning false for ", substr
            return False
    #print "REturning true for ", substr
    return True

def getMaxTandem(input_str):
    #print "Received ", input_str
    start_len = len(input_str)
    if start_len%2: start_len -= 1
    
    while(start_len>=2*no_of_char):
        #print start_len
        for i in xrange(0, len(input_str)):
            if not i+start_len<=len(input_str):
                break
            substr = input_str[i:i+start_len]

            #print substr
            if isTandem(substr):
                return start_len, substr
        start_len -= 2
