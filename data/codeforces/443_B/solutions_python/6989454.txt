s = raw_input()
k = int(raw_input())
los = len(s)

# helper function: determines if string s (length los) has a tandem string with length l 	
def tandem(s, los, lot):
	n = lot / 2
	strlen = len(s)
	for x in xrange(0, los - lot + 1): 
		for pos in xrange(x, x + n):
			if pos + n >= strlen:
				return True
			if s[pos] != s[pos + n]:
				break
			if pos == (x + n - 1):
				return True
	return False
	
# length of tandem, must be even
lot = k + los
if lot % 2 != 0:
	lot -=1

# iterate through all possible permutations for tandems, breaks at highest valid tandem length
while lot > 0:
	if tandem(s, los + k, lot):
		break
	lot-=2

print(lot)
