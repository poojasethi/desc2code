from collections import defaultdict
import sys

s = raw_input()
k = int(raw_input())

if k >= len(s) - 1:
    print ((k + len(s)) / 2) * 2
    sys.exit(0)

c = defaultdict(list)
for i, ch in enumerate(s):
    c[ch].append(i)

index = 0
maxL = k * 2
last = len(s) - 1
while index < len(s):
    cand = [i for i in c[s[index]] if i > index]
    for can in cand:
        # if (can - index) * 2 <= maxL:
        #     continue
        add = can - index
        if can + add < len(s):
            if s[index: can] == s[can: can + add]:
                length = add * 2
                if maxL < length:
                    maxL = length
        else:
            add = 0
            while can + add < last:
                if s[index + add + 1] == s[can + add + 1]:
                    add += 1
                else:
                    break
            if can + add < last:
                continue
            diff = can - add - index - 1
            if diff > k:
                continue
            length = (can - index) * 2
            if length > maxL:
                maxL = length

    index += 1

print maxL
