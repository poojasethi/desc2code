s = raw_input()
k = int(raw_input())

for i in xrange(k):
    s += "*"

ans = 0

for i in xrange(len(s)):
    for j in xrange(i+1,len(s)):
        ok = True
        le = j-i+1
        if le&1:
            continue
        for kk in xrange(le/2):
            n = le/2
            if s[i+kk]!=s[i+kk+n] and s[i+kk+n]!="*":
                ok = False
        if ok:
            ans = max(ans,le)

print ans
