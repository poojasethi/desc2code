def isok(s, q):
    for i in range(0, len(s) - 2 * q + 1):
        for j in range(0, q):
            if s[i + j + q] != "?" and s[i + j] != s[i + j + q]:
                break
            j += 1
        else:
            return True
    return False

s, k = raw_input(), int(raw_input())
s += "?" * k

for i in range(len(s) // 2, -1, -1):
    if isok(s, i):
        print 2 * i
        break
