def isok(s, q):
    i = 0
    while i + 2 * q <= len(s):
        j = 0
        while j < q and (s[i + j + q] == "?" or s[i + j] == s[i + j + q]):
            j += 1
        if j == q: return True
        i += 1
    return False

s, k = raw_input(), int(raw_input())
s += "?" * k

for i in range(len(s) // 2, -1, -1):
    if isok(s, i):
        print 2 * i
        break
