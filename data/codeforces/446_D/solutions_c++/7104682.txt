#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef double matr[111][111];
struct edge{
	int a,b;
}g[110001];
matr a,b,c;
double w[511][1111],f[511][511],ans;
int i,j,k,n,m,d[511],lim,tp[511],p[511],num;
void gauss(){
	int i,j,k;
	double t;
	for (i=1;i<=n;i++){
		for (j=i;w[j][i]==0;j++);
		for (k=1;k<=2*n;k++) swap(w[i][k],w[j][k]);
		for (k=i,t=w[i][i];k<=2*n;k++) w[i][k]/=t;
		for (j=i+1;j<=n;j++){
			if (w[j][i]==0) continue;
			for (k=i,t=w[j][i];k<=2*n;k++) w[j][k]-=w[i][k]*t;
		}
	}
	for (i=n;i;i--)
	 for (j=1;j<i;j++){
	 	if (w[j][i]==0) continue;
	 	for (t=w[j][i],w[j][i]=0,k=n+1;k<=2*n;k++) w[j][k]-=w[i][k]*t;
	 }
}
void mmul(matr &a,matr b){
	int i,j,k;
	memset(c,0,sizeof(c));
	for (i=1;i<=num;i++)
	 for (j=1;j<=num;j++)
	  for (k=1;k<=num;k++) c[i][j]+=a[i][k]*b[k][j];
	memcpy(a,c,sizeof(a));
}
void mpw(int p){
	int i;
	memset(a,0,sizeof(a));
	for (i=1;i<=num;i++) a[i][i]=1;
	for (;p;p>>=1,mmul(b,b))
	 if (p&1) mmul(a,b);
}
int main(){
	scanf("%d%d%d",&n,&m,&lim);memset(d,0,sizeof(d));
	for (i=1,num=0;i<=n;i++){
		scanf("%d",&tp[i]);
		if (tp[i]) p[i]=++num;
	}
	for (i=1;i<=m;i++) scanf("%d%d",&g[i].a,&g[i].b),d[g[i].a]++,d[g[i].b]++;
	memset(w,0,sizeof(w));
	for (i=1;i<=n;i++) w[i][i]=w[i][i+n]=1;
	for (i=1;i<=m;i++){
		if (!tp[g[i].a]){
			w[g[i].b][g[i].a]-=(double)1/d[g[i].a];
		}
		if (!tp[g[i].b]){
			w[g[i].a][g[i].b]-=(double)1/d[g[i].b];
		}
	}
	gauss();
	memset(f,0,sizeof(f));memset(b,0,sizeof(b));
	for (i=1;i<=n;i++)
	 if (!tp[i]){
		for (j=1;j<=n;j++) f[i][j]=w[j][i+n];
	 }else f[i][i]=1;
	for (i=1;i<=m;i++){
		if (tp[g[i].a]){
			for (j=1;j<=n;j++)
			 if (p[j]) b[p[j]][p[g[i].a]]+=f[g[i].b][j]/d[g[i].a];
		}
		if (tp[g[i].b]){
			for (j=1;j<=n;j++)
			 if (p[j]) b[p[j]][p[g[i].b]]+=f[g[i].a][j]/d[g[i].b];
		}
	}
	mpw(lim-2);
	for (ans=0,i=1;i<=n;i++)
	 if (p[i]) ans+=a[num][p[i]]*f[1][i];
	printf("%.9lf\n",ans);
	return 0;
}
