#define EPS 1e-9
#define INF MOD
#define MOD 1000000007LL

int a,b,ax,ay,bx,by,n;
int x1[1010],y1[1010],x2[1010],y2[1010],d[1010];
int dist2(int A,int B){
    int dx = 0,dy = 0;
    if(x2[A] < x1[B]) dx = x1[B]- x2[A];
    else if(x2[B] < x1[A]) dx = x1[A]- x2[B];
    if(y2[A] < y1[B]) dy = y1[B]- y2[A];
    else if(y2[B] < y1[A]) dy = y1[A]- y2[B];
    return dx*dx +dy*dy;
}

double dist(int A,int B){
    int dx = 0,dy = 0;
    if(x2[A] < x1[B]) dx = x1[B]- x2[A];
    else if(x2[B] < x1[A]) dx = x1[A]- x2[B];
    if(y2[A] < y1[B]) dy = y1[B]- y2[A];
    else if(y2[B] < y1[A]) dy = y1[A]- y2[B];
    return sqrt((double)(dx*dx +dy*dy));
}

int main(){

   //freopen("1.txt","r",stdin);

    cin>>a>>b>>ax>>ay>>bx>>by>>n;
    for(int i = 0; i < n; i++){
        cin>>x1[i]>>y1[i]>>x2[i]>>y2[i];
        if(x1[i] == x2[i]){
            if(y1[i] > y2[i]) swap(y1[i],y2[i]);
        }else{
            if(x1[i] > x2[i]) swap(x1[i],x2[i]);
        }
    }
    x1[n] = x2[n] = ax;
    y1[n] = y2[n] = ay;
    x1[n+1] = x2[n+1] = bx;
    y1[n+1] = y2[n+1] = by;
    queue<int>Q;
    Q.push(n);
    fill(d,d+n+2,INF);
    d[n] = 0;
    while(Q.sz){
        int cur = Q.front(); Q.pop();
        if(cur == n+1) break;
        for(int i = 0; i < n+2; i++){
            if(d[i] > d[cur]+1&&dist2(cur,i)<= a*a){
                d[i] = d[cur]+1;
                Q.push(i);
            }
        }
    }
    if(d[n+1] == INF) {
        cout<<-1<<endl;
        return 0;
    }
    double ans = INF;
    for(int i = 0; i < n+1; i++)
        if(d[i] == d[n+1]-1){
            ans = min(ans,(a+b)*d[i]+dist(i,n+1));
        }
    cout<<ans<<endl;
    return 0;
}
