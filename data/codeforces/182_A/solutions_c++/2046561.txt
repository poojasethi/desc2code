#include<cstdio>
#include<cmath>
#include<algorithm>
#define inf 200000000
using namespace std;

double aa,bb,sx,sy,ex,ey;
double x[1005][2],y[1005][2],c[3000000],d[3000];
int n,b[3000000],e[3000000],h[3000],tot,p[3000],q[3000];

void Add(int x,int y,double z)
{
	b[++tot]=y; c[tot]=z; e[tot]=h[x]; h[x]=tot;
}

double Sqr(double x){return x*x;}

double Dist(double x1,double y1,double x2,double y2)
{
	return sqrt(Sqr(x1-x2)+Sqr(y1-y2));
}

void Deal(int i,int j)
{
	double d=inf;
	if (i==n-2&&j==n-1)
	{	
		d=Dist(sx,sy,ex,ey);
		if (d<=aa+1e-8) Add(i,j,d);
	}
	int x1=(max(x[i][0],x[i][1])==x[i][0]?0:1),y1=(max(y[i][0],y[i][1])==y[i][0]?0:1);
	int x2=(max(x[j][0],x[j][1])==x[j][0]?0:1),y2=(max(y[j][0],y[j][1])==y[j][0]?0:1);
	if (x[i][0]==x[i][1]&&x[j][0]==x[j][1])
	{
		if (y[i][1-y1]<=y[j][y2]&&y[i][y1]>=y[j][1-y2])
			d=abs(x[i][0]-x[j][0]);
	} else 
	if (y[i][0]==y[i][1]&&y[j][0]==y[j][1])
	{
		if (x[i][1-x1]<=x[j][x2]&&x[i][x1]>=x[j][1-x2])
			d=abs(y[i][0]-y[j][0]);
	} else 
	if (x[i][0]==x[i][1]&&y[j][0]==y[j][1])
	{
		if (y[i][y1]>=y[j][0]&&y[i][1-y1]<=y[j][0])
			d=min(abs(x[j][x2]-x[i][0]),abs(x[j][1-x2]-x[i][0]));
		if (x[j][x2]>=x[i][0]&&x[j][1-x2]<=x[i][0])
			d=min(abs(y[j][0]-y[i][y1]),abs(y[j][0]-y[i][1-y1]));
	} else 
	if (y[i][0]==y[i][1]&&x[j][0]==x[j][1])
	{
		if (y[j][y2]>=y[i][0]&&y[j][1-y2]<=y[i][0])
			d=min(abs(x[j][0]-x[i][x1]),abs(x[j][0]-x[i][1-x1]));
		if (x[j][0]<=x[i][x1]&&x[j][0]>=x[i][1-x1])
			d=min(abs(y[j][y2]-y[i][0]),abs(y[j][1-y2]-y[i][0]));
	} 
	if (abs(d-inf)<1e-8) 
	{
		for (int a=0;a<2;a++)
			for (int b=0;b<2;b++)	
				d=min(d,Dist(x[i][a],y[i][a],x[j][b],y[j][b]));
	}
	if (d<=aa+1e-8)
		if (j!=n-1)Add(i,j,aa+bb); else Add(i,j,d);
}

void Spfa()
{
	for (int i=0;i<n;i++)d[i]=inf;
	int l=0,r=1;
	q[1]=n-2;d[n-2]=0;
	while (l<r)
	{
		l++; p[q[l]]=0;
		for (int go=h[q[l]];go;go=e[go])
			if (d[b[go]]>d[q[l]]+c[go])
			{
				d[b[go]]=d[q[l]]+c[go];
				if (!p[b[go]])
				{
					q[++r]=b[go];
					p[b[go]]=1;
				}
			}
	}
	if (abs(d[n-1]-inf)<1e-8) printf("-1"); else printf("%.10lf",d[n-1]);
}
	

			
int main()
{
	scanf("%lf%lf",&aa,&bb);
	scanf("%lf%lf%lf%lf",&sx,&sy,&ex,&ey);
	scanf("%d",&n);
	for (int i=0;i<n;i++)
		scanf("%lf%lf%lf%lf",&x[i][0],&y[i][0],&x[i][1],&y[i][1]);
	x[n][0]=sx;x[n][1]=sx;y[n][0]=sy;y[n][1]=sy;
	x[n+1][0]=ex;x[n+1][1]=ex;y[n+1][0]=ey;y[n+1][1]=ey;
	n+=2;
	for (int i=0;i<n-1;i++)
		for (int j=0;j<n;j++)
			if (i!=j) Deal(i,j);
	Spfa();
	return 0;
}
