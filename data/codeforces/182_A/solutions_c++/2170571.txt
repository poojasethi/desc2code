#include<stdio.h>
#include<math.h>
#define X 1010
struct point{
	__int64 x1,y1,x2,y2;
	int t,f;
}p[X];
int map[X][X];
int f[X],n;
int sqr(int x){
	return x*x;
}
int max(int a,int b){
	return a>b?a:b;
}
int min(int a,int b){
	return a<b?a:b;
}
void dfs(int a){
	int i;
	for(i=0;i<n;i++)
	    if(map[a][i]&&f[i]>f[a]+1){
			f[i]=f[a]+1;
	        dfs(i);
		}
}
int main(){
	int i,j,x1,x2,y1,y2,t,a,b;
	double as;
	scanf("%d%d",&a,&b);
	scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
	scanf("%d",&n);
	p[n].x1=p[n].x2=x1;
	p[n].y1=p[n].y2=y1;
	p[n+1].x1=p[n+1].x2=x2;
	p[n+1].y1=p[n+1].y2=y2;
	for(i=0;i<n;i++){
		scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
		if(x1==x2){
			if(y1>y2){t=y1;y1=y2;y2=t;}
			p[i].f=1;
		}
		else{
			if(x1>x2){t=x1;x1=x2;x2=t;}
	        p[i].f=0;
		}
		p[i].x1=x1;p[i].y1=y1;
		p[i].x2=x2;p[i].y2=y2;
		f[i]=n+2;
	}
	n+=2;
	for(i=0;i<n;i++)
	    for(j=0;j<n;j++){
			map[i][j]=sqr(max(0,max(p[i].x1,p[j].x1)-min(p[i].x2,p[j].x2)))+
                      sqr(max(0,max(p[i].y1,p[j].y1)-min(p[i].y2,p[j].y2)));
			if(map[i][j]>a*a)
			    map[i][j]=0;
			}
	f[n-1]=n;f[n-2]=0;
	dfs(n-2);
	if(f[n-1]==n){
		puts("-1");
		return 0;
	}
	t=n+2;
	for(i=0;i<n-1;i++)
	    if(map[i][n-1]&&f[i]*(a+b)+sqrt(map[i][n-1])<as){
			as=f[i]*(a+b)+sqrt(map[i][n-1]);
		}
	printf("%.8lf\n",as);
}
