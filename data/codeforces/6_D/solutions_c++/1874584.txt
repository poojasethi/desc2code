#include <iostream>

using namespace std;

const int MAXN = 100;

int n, a, b, h[MAXN], temp[MAXN], ans[MAXN], count = 1000, cnt;

void BT(int i)
{
	if (i+1 == n)
	{
		if (cnt < count)
		{
			for (int j = 0 ; j < cnt ; j++)
				ans[j] = temp[j];
			count = cnt;
		}
		return;
	}
	if (h[i] < 0)
		return BT(i+1);
	for (int aa = 0 ; aa < 15 ; aa++)
	{
		int hp = h[i]-a*aa, tempcnt = cnt, bb;
		if (hp < 0)
			bb = 0;
		else
			bb = hp/b + 1;
		for (int j = 0 ; j < aa ; j++)
			temp[cnt++] = i;
		for (int j = 0 ; j < bb ; j++)
			temp[cnt++] = i+1;
		h[i] -= (aa*a+bb*b);
		h[i+1] -= (aa*b+bb*a);
		h[i+2] -= (bb*b);
		BT(i+1);
		h[i] += (aa*a+bb*b);
		h[i+1] += (aa*b+bb*a);
		h[i+2] += (bb*b);
		cnt = tempcnt;
	}
}

int main()
{
	ios::sync_with_stdio(0);
	cin >> n >> a >> b;
	for (int i = 0 ; i < n ; i++)
		cin >> h[i];
	while (h[0] >= 0)
	{
		h[0] -= b;
		h[1] -= a;
		h[2] -= b;
		temp[cnt++] = 1;
	}
	while (h[n-1] >= 0)
	{
		h[n-1] -= b;
		h[n-2] -= a;
		h[n-3] -= b;
		temp[cnt++] = n-2;
	}
	BT(1);
	cout << count << endl;
	for (int i = 0 ; i < count ; i++)
		cout << ans[i]+1 << " ";
	cout << endl;
	return 0;
}
