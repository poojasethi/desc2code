#include <iostream>
using namespace std;
int n, a, b;
int h[15];
int dp[15][15][15];
int th[15];
int ans[15];
int ret;
int tret;
int flag = 0;
int cnt = 0;
int dfs(int p)
{
    cnt++;
    if(p == n)
    {
        if(h[p] >= 0 || h[p - 1] >= 0)
            return 0;
        if(ret > tret)
        {
            for(int i = 1; i <= n; i++)
                ans[i] = th[i];
            ret = tret;
        }
        return 0;
    }
    if(cnt > 60000005) return 0;
    if(h[p - 1] < 0)
        dfs(p + 1);
    for(;h[p + 1] >= 0 || h[p] >= 0 || h[p - 1] >= 0;)
    {
        tret++;
        th[p]++;
        h[p]-=a;
        h[p+1]-=b;
        h[p-1]-=b;
        if(h[p - 1] < 0)dfs(p + 1);
    }
    for(;th[p]>0; th[p]--)
    {
        tret--;
        h[p]+=a;
        h[p+1]+=b;
        h[p-1]+=b;
    }
}
int main()
{
    ret = 1000000;
    cin >> n >> a >> b;
    for(int i = 1; i <= n; i++)
        cin >> h[i];
    dfs(2);
    cout << ret << endl;
    for(int i = 1; i <= n; i++)
        for(int j = 0; j < ans[i]; j++)
        cout << i <<" ";
}
