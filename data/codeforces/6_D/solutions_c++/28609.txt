#include <cstdio>
#include <cstring>



int n,a,b;
int HP[20];
int ANS[10000];
int STEP[10000];
int s=0;
int ans=2000000000;

inline void fire(int &x,int &y,int &z)
{
	x-=b; y-=a; z-=b;
	if (x<0) x=0;
	if (y<0) y=0;
	if (z<0) z=0;
}

void dfs(int now,int line1,int line2,int line3,int step)
{
	if (step>=ans) return;
	if (now==n) {
		if (line1+line2+line3==0) {
			if (ans>step) {
				ans=step;
				memcpy(ANS,STEP,ans<<2);
			}
		}
		return;
	}

	while (line1) {
		fire(line1,line2,line3); STEP[step++]=now;
	}

	dfs(now+1,line2,line3,HP[now+2],step);
	while (line2+line3) {
		fire(line1,line2,line3); STEP[step++]=now;
		dfs(now+1,line2,line3,HP[now+2],step);
	}
}

int main()
{
	scanf("%d%d%d",&n,&a,&b);
	for (int i=1;i<=n;i++) {
		scanf("%d",HP+i);
		HP[i]++;
	}
	dfs(2,HP[1],HP[2],HP[3],0);
	printf("%d\n",ans);

	for (int i=0;i<ans;i++) {
		if (i) {
			putchar(' ');
		}
		int tt=ANS[i];
		printf("%d",tt);
	}
	putchar(10);
	return 0;
}
