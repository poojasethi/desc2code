#include<cstdio>
#include<cstring>
int minn=0x7fffffff,n,a,b,h[22],ans[22],tmp[22];
void dfs(int i,int step){
    if(i==n-1){
        if(h[n-2]>=0)return;
        int t=0;
        if(h[n-1]>=0)
            t=h[n-1]/b+1;
        tmp[n-2]+=t;
        if(minn>step+t){
            minn=step+t;
            memcpy(ans,tmp,sizeof(tmp));
        }
        tmp[n-2]-=t;
        return;
    }
    if(h[i-1]>=0){
        int t=h[i-1]/b+1;
        tmp[i]+=t;
        h[i-1]-=t*b;
        h[i]-=t*a;
        h[i+1]-=t*b;
        dfs(i,step+t);
        h[i+1]+=t*b;
        h[i]+=t*a;
        h[i-1]+=t*b;
        tmp[i]-=t;
    }
    else{
        dfs(i+1,step);
        if(h[i]>=0){
            ++tmp[i];
            h[i]-=a;
            h[i+1]-=b;
            dfs(i,step+1);
            h[i+1]+=b;
            h[i]+=a;
            --tmp[i];
        }
    }
}
int main(){
    scanf("%d%d%d",&n,&a,&b);
    int i;
    for(i=0;i<n;++i)scanf("%d",h+i);
    dfs(1,0);
    printf("%d\n",minn);
    int j;
    for(i=1;i<n-1;++i)
        for(j=0;j<ans[i];++j)
            printf("%d ",i+1);
    return 0;
}
