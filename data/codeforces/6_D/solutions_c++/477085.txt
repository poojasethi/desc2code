#include <cstdio>
#include <cmath>
#include <vector>

using namespace std;

int N, A, B, H[20];
vector<int> SOL, CUR;

void go(int k) {
    int c = 0;
    if (k == N-1) {
        while (H[k] >= 0 || H[k-1] >= 0) { CUR.push_back(k); c++; H[k-1] -= A; H[k] -= B; H[k-2] -= B; }
        if (!SOL.size() || CUR.size() < SOL.size()) SOL = CUR;
        while (c--) { CUR.pop_back(); H[k-1] += A; H[k] += B; H[k-2] += B; }
        return;
    }
    while (H[k-1] >= 0) { CUR.push_back(k+1); c++; H[k-1] -= B; H[k] -= A; H[k+1] -= B; }
    while (H[k] >= 0) {
        go(k+1);
        CUR.push_back(k+1); c++; H[k-1] -= B; H[k] -= A; H[k+1] -= B;
    }
    go(k+1);
    while (c--) { CUR.pop_back(); H[k-1] += B; H[k] += A; H[k+1] += B; }
}

int main() {

    scanf("%d %d %d", &N, &A, &B);
    for (int i = 0; i < N; i++) scanf("%d", H+i);
    go(1);
    printf("%d\n%d", SOL.size(), SOL[0]);
    for (int i = 1; i < SOL.size(); i++) printf(" %d", SOL[i]);
    putchar('\n');

    return 0;
}
