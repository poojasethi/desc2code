#include<stdio.h>
#include<algorithm>
using namespace std;
int ans[10]={0},arr[15]={0},cur[10]={0},Min=999,n,a,b;
void pmu(int c,int sum)
{
   if(sum>Min) return;
   if(c==n)
   {
      for(int i=1;i<=n;i++)
      {
        if(arr[i]>=0) return;
      }
      if(sum<Min)
      {
        Min=sum;
        for(int i=2;i<=n-1;i++)
        {
          ans[i]=cur[i];
        }
      }
   }
   else
   {
      int M=max(max((arr[c-1]+1)/b+1,(arr[c]+1)/a+1),arr[c+1]/b+1);
      //printf("%d %d %d\n",c,M,(arr[c-1]+1)/b);
      for(int i=max((arr[c-1]+1)/b,0);i<=M;i++)
      {
          arr[c]-=i*a;
          arr[c-1]-=i*b;
          arr[c+1]-=i*b;
          cur[c]=i;
         // printf("chose %d %d\n",c,i);
          pmu(c+1,sum+i);
          arr[c]+=i*a;
          arr[c-1]+=i*b;
          arr[c+1]+=i*b;
          cur[c]=0;
      }
   }
}
int main()
{
   int i,j;
   scanf("%d %d %d",&n,&a,&b);
   for(i=1;i<=n;i++) scanf("%d",&arr[i]);
   pmu(2,0);
   printf("%d\n",Min);
   for(i=2;i<=n-1;i++)
   {
     for(j=1;j<=ans[i];j++) printf("%d ",i);
   }
return 0;
}
