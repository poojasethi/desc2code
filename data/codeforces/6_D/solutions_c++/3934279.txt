#include <cstdio>
#include <algorithm>

#define INF 1000000000
using namespace std;

int f[100][100][100],n,a,b,h[1000],ans;

int dp(int x,int y,int z)
{
   // printf("**%d %d %d\n",x,y,z);
    if(x==n)
    {
        if(h[n]>z*b || y>0)return INF;
        return 0;
    }
    int &now=f[x][y][z];
    if(now!=0)return now;
    now=INF;
    for(int i=(y+b-1)/b;i<=16;i++)
    {
        int k=h[x]-i*a-b*z;
        if(k<0)k=0;
        now=min(now,dp(x+1,k,i)+i);
    }
   return now;
}

void print(int x, int y, int z)
{
    if(x==n) return;
    int check=dp(x,y,z);
    for(int i=(y+b-1)/b;i<=16;i++)
    {
        int k=h[x]- z*b-i*a;
        if(k < 0) k = 0;
        if(check==(dp(x+1,k,i)+i))
        {
            for(int j=0;j<i;j++)printf("%d ",x);
            print(x+1,k,i);
            return;
        }
    }
}


int main()
{
    scanf("%d %d %d",&n,&a,&b);
    for(int i=1;i<=n;i++){scanf("%d",&h[i]);h[i]++;}
    ans=dp(2,h[1],0);
    printf("%d\n",ans);
    print(2,h[1],0);
    return 0;
}
