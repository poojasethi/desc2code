#include<stdio.h>
#include<string.h>
#define max(a,b) ((a)>(b)?(a):(b))

int a, b, n, h[15], num[15];

int dfs(int d, int x, int l){
	int i;
	if(d==0){
		for(i=1; i<=n; i++)
			if(h[i]>=0)
				return 0;
		return 1;
	}
	
	if(h[x]<0)
		return dfs(d, x+1, l);
	
	if(x==n){
		h[x]-=b, num[x-1]++;
		if(dfs(d-1, x, l))
			return 1;
		h[x]+=b, num[x-1]--;
		return 0;
	}
	
	for(i=max(x, l); i<=x+1; i++){
		if(i<=1 || i>=n)
			continue;
		num[i]++;
		h[i]-=a, h[i+1]-=b, h[i-1]-=b;
		if(dfs(d-1, x, i))
			return 1;
		h[i]+=a, h[i+1]+=b, h[i-1]+=b;
		num[i]--;
	}
	return 0;
}

int main()
{
	int ans, i;
	scanf("%d %d %d", &n, &a, &b);
	for(i=1; i<=n; i++)
		scanf("%d", &h[i]);
	for(ans=0; ; ans++){
		memset(num, 0, sizeof(num));
		if(dfs(ans, 1, 2)){
			printf("%d\n", ans);
			int k=2;
			while(num[k]==0)
				k++;
			num[k]--;
			printf("%d", k);
			for(; k<12; k++){
				while(num[k]--)
					printf(" %d", k);
			}
			printf("\n");
			return 0;
		}
	}
}
