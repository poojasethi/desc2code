#include<cstdio>
#include<algorithm>
using namespace std;
const int MAXN=100;
const int INF=1000000000;
int life[MAXN];
int lo[MAXN];
int resp[MAXN];
int minimun=INF;
int mn[MAXN];
int n,a,b;
void rec(int pos,int qtd){
    if(pos==n){
        if(qtd<minimun){
            for(int i=2;i<n;i++){
                if(life[i]>=lo[i]*a+lo[i-1]*b+lo[i+1]*b)return;
            }
            for(int i=1;i<=n;i++)resp[i]=lo[i];
            minimun=qtd;
        }

        return;
    }
    for(int i=0;i<=8;i++){
        lo[pos]+=i;
        rec(pos+1,qtd+i);   
        lo[pos]-=i;
    }
}
int main(){

    scanf("%d %d %d",&n,&a,&b);
    for(int i=1;i<=n;i++) scanf("%d",&life[i]);
    int qtd=0,fkboard;
    fkboard=(life[1]/b)+1;
    lo[2]=max(lo[2],fkboard);
    fkboard=(life[n]/b)+1;
    lo[n-1]=max(lo[n-1],fkboard);
    qtd+=lo[2];
    if(n-1!=2)qtd+=lo[n-1];
    rec(2,0);
    printf("%d\n",minimun+qtd);
    for(int i=1;i<=n;i++){
        for(int j=0;j<resp[i];j++){
            printf("%d ",i);
        }
    }
    printf("\n");
}
