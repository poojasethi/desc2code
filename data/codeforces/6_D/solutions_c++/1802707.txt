#include<cstdio>
#include<cstring>
int hp[15];
int ret[10000];
int path[10000];
int ans=~0u>>2;
int n,a,b;
void fire(int &x,int &y,int &z){
    x-=b;if(x<0) x=0;
    y-=a;if(y<0) y=0;
    z-=b;if(z<0) z=0;
}
void dfs(int now,int fir,int sec,int thi,int step){
    if(step>=ans) return ;
    if(now==n){
        if(fir+sec+thi==0){
            if(step<ans){
            ans=step;
            memcpy(ret,path,ans<<2);
           }
        }
        return ;
    }
    while(fir){
        fire(fir,sec,thi);  path[step++]=now;
    }
    dfs(now+1,sec,thi,hp[now+2],step);
    while(sec+thi){
        fire(fir,sec,thi); path[step++]=now;
        dfs(now+1,sec,thi,hp[now+2],step);
    }
}
int main()
{
  
    scanf("%d%d%d",&n,&a,&b);
    for(int i=1;i<=n;i++){
        scanf("%d",&hp[i]);
        hp[i]++;
    }
    dfs(2,hp[1],hp[2],hp[3],0);
    printf("%d\n",ans);
    for(int i=0;i<ans;i++)
    {
        printf("%d ",ret[i]);
    }
    scanf("%d",&n);
    return 0;
}
