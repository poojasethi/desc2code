using namespace std;
int n,a,b,h[20],dp[20][20][20],ne[20][20][20],INF=0x3f3f3f3f;
int dfs(int q,int w,int e)
{
    if(dp[q][w][e]!=INF)return dp[q][w][e];
    if(q==n-1)
    {
        if(h[q]>=b*w+e*a||h[n]>=e*b)return dp[q][w][e]=INF-1;
        return dp[q][w][e]=e;
    }
    dp[q][w][e]--;
    for(int i=0;i<17;i++)
    {
        if(h[q]>=b*(w+i)+e*a)continue;
        if(dp[q][w][e]>dfs(q+1,e,i)+e)
        {
            dp[q][w][e]=dp[q+1][e][i]+e;
            ne[q][w][e]=i;
        }
    }
    return dp[q][w][e];
}
int main()
{
    memset(dp,0x3f,sizeof(dp));
    scanf("%d%d%d",&n,&a,&b);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&h[i]);
    }
    dfs(1,0,0);
    printf("%d\n",dp[1][0][0]);
    for(int i=1,w=0,e=0,q;i<n;q=ne[i++][w][e],w=e,e=q)
    {
        for(int j=0;j<e;j++)printf("%d ",i);
    }
    return 0;
}
