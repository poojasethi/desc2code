#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;
int ans[150],mi,n,f[150],re[150],a,b;
void dfs(int pos,int num)
{
	if(num>=mi) return;
	if(pos==n){
		if(f[pos-1]<0&&f[pos]<0){
			mi=num;
			for(int i=0;i<mi;i++)
				ans[i]=re[i];
		}
		return;
	}
	int x,y,z;
	x=f[pos-1];
	y=f[pos];
	z=f[pos+1];
	int c=0;
	if(f[pos-1]>=0){
		c=f[pos-1]/b+1;
	}
	f[pos]-=c*a;
	f[pos+1]-=c*b;
//	re[step]=c;
	for(int i=0;i<c;i++)
		re[num+i]=pos;
	dfs(pos+1,num+c);
	f[pos]+=c*a;
	f[pos+1]+=c*b;
	while(f[pos]-c*a>=0||f[pos+1]-c*b>=0){
		c++;
		f[pos]-=c*a;
		f[pos+1]-=c*b;
		re[num+c-1]=pos;
		dfs(pos+1,num+c);
		f[pos]+=c*a;
		f[pos+1]+=c*b;
	}
}
int main()
{
	scanf("%d%d%d",&n,&a,&b);
	mi=2100000000;
	memset(f,-1,sizeof(f));
	for(int i=1;i<=n;i++)
		scanf("%d",&f[i]);
	dfs(2,0);
	printf("%d\n",mi);
	for(int i=0;i<mi;i++){
		printf("%d",ans[i]);
		if(i==mi-1) printf("\n");
		else printf(" ");
	}
	//system("pause");
	return 0;
}
