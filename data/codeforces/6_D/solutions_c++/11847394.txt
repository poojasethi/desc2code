#include<iostream>
#include<cstring>
#include<cstdio>

using namespace std; 

struct node{
	int id,x,y,cnt; 
}pre[20][20][20]; 

int h[20],a,b,f[20][20][20],n; 
int const oo=1<<29; 

int main(){
	
	scanf("%d%d%d",&n,&a,&b); 
	for(int i=1;i<=n;i++)
		scanf("%d",&h[i]),h[i]++; 
	for(int i=0;i<=n;i++) 
		for(int j=0;j<20;j++) 
			for (int k=0;k<20;k++) 
				f[i][j][k]=oo; 
	f[2][h[1]][h[2]]=0;
	for(int i=2;i<n;i++) 
		for(int j=0;j<20;j++)
			for(int k=0;k<20;k++) 
				if (f[i][j][k]!=oo) {
					for(int c=0;c<20;c++)
						if ( j-c*b<=0) {
							int j2=max(0,k-c*a); 
							int k2=max(0,h[i+1]-c*b);		
							if( f[i+1][j2][k2]>f[i][j][k]+c) {
								f[i+1][j2][k2]=f[i][j][k]+c; 
								pre[i+1][j2][k2].id=i; 
								pre[i+1][j2][k2].x=j; 
								pre[i+1][j2][k2].y=k; 
								pre[i+1][j2][k2].cnt=c; 
							}
						}
				}
	printf("%d\n",f[n][0][0]); 
	int id=n,x=0,y=0; 
	while ( id>1) { 
		int  cnt=pre[id][x][y].cnt; 
		for(int i=1;i<=cnt;i++) printf("%d ",id-1); 
		int xx=pre[id][x][y].x;
		int yy=pre[id][x][y].y; 
		x=xx;y=yy; 
		id--; 
	}
	return 0; 
}
