#include<bits/stdc++.h>
using namespace std;

int h[15];
int a,b,n;
vector<int>T,T1;
int ans = 1e9;
void dfs(int num,int times)
{
    if (times>=ans)
		return;
	if (num==n)
	{
		if (h[num]<0)
		{
		   	T1 = T;
		    ans = times;
		}
        return;
	}
    for (int i = 0;i<=max(h[num-1]/b+1,max(h[num]/a+1,h[num+1]/b+1));i++)
	{
		if (h[num-1]<b*i)
		{
			h[num-1]-=b*i;
			h[num]-=a*i;
			h[num+1]-=b*i;
			for (int j = 0;j<i;j++)
			  T.push_back(num);
			dfs(num+1,times+i);
			for (int j= 0;j<i;j++)
				T.pop_back();
			h[num-1]+=b*i;
			h[num]+=a*i;
			h[num+1]+=b*i;
		}
	}
}
int main()
{
    scanf("%d%d%d",&n,&a,&b);
	for(int i = 1;i<=n;i++)
		scanf("%d",&h[i]);
    dfs(2,0);
	printf("%d\n",ans);
	for (int i = 0;i<T1.size();i++)
		printf("%d ",T1[i]);
	printf("\n");
}
