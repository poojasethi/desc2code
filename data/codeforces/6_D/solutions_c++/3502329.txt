#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int ans[165],res,l[165];
int h[15],n,a,b;
void dfs(int x,int tmp)
{
    int i,j;
    if(tmp>=res) return;
    for(i=1;i<=n;i++)
    if(h[i]>=0) break;
    if(i>n)
    {
        res=tmp;
        for(i=0;i<tmp;i++)
            l[i]=ans[i];
        return;
    }
    if(x==n) return ;
    for(i=0;i<=16;i++)
    {
        if(h[x-1]-i*b>=0) continue;
        if(x==n-1&&h[x+1]-i*b>=0) continue;
        for(j=tmp;j<tmp+i;j++)
        ans[j]=x;
        h[x-1]-=(i*b);
        h[x]-=(i*a);
        h[x+1]-=(i*b);
        dfs(x+1,tmp+i);
        h[x-1]+=(i*b);
        h[x]+=(i*a);
        h[x+1]+=(i*b);
        if(h[x]-i*a<0&&h[x-1]-i*b<0&&h[x+1]-i*b<0) break;
    }
}
int main()
{
    while(~scanf("%d%d%d",&n,&a,&b))
    {
        memset(h,0,sizeof(h));
        for(int i=1;i<=n;i++)
        scanf("%d",&h[i]);
        res=165;
        dfs(2,0);
        printf("%d\n",res);
        for(int i=0;i<res;i++)
        printf("%d ",l[i]);
        printf("\n");
    }
    return 0;
}
