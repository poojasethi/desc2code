#include<iostream>
#include<stdio.h>
#include<string.h>
#include<vector>
#include<algorithm>
using namespace std;
int n,a,b,h[150];
vector<int>pp,ans;
void dfs(int x,int flag){
	if(x==n&&flag==0){
	 if(pp.size()<ans.size())
		 ans=pp;
	 return;
	}
	int size=0;
	while(h[x-1]>0){
	 h[x]-=a;
	 h[x-1]-=b;
	 h[x+1]-=b;
	 size++;
	 pp.push_back(x);
	}
	while(h[x]>0){
	 dfs(x+1,1);
	 h[x]-=a;
	 h[x+1]-=b;
	 h[x-1]-=b;
	 size++;
	 pp.push_back(x);
	}
	dfs(x+1,0);
	h[x]+=size*a;
	h[x-1]+=size*b;
	h[x+1]+=size*b;
	while(size--)
	pp.erase(pp.end()-1);
}
int main(){
	while(cin>>n>>a>>b){
	 pp.clear();
	 ans.clear();
	 int i;
	 for(i=1;i<=n;i++){
		 scanf("%d",&h[i]);
	 h[i]++;
	 }
while(h[1]>0){
 h[2]-=a;
 h[1]-=b;
 h[3]-=b;
 ans.push_back(2);
}
while(h[n]>0){
 h[n-1]-=a;
 h[n]-=b;
 h[n-2]-=b;
 ans.push_back(n-1);
}
pp=ans;
for(i=1;i<=100;i++)
ans.push_back(i);
dfs(2,0);
printf("%d\n",ans.size());
printf("%d",ans[0]);
for(i=1;i<ans.size();i++)
printf(" %d",ans[i]);
printf("\n");
	}
	return 0;
}
