#include <bits/stdc++.h>
using namespace std;

int a, b, n, h[15], f[200];

bool dfs(int d, int x, int l) {
  if (d == 0) {
    for (int i = 1; i <= n; ++ i) if (h[i] >= 0) return false;
    return true;
  }
  if (h[x] < 0) return dfs(d, x + 1, l);
  if (x == n) {
    h[x] -= b; f[d] = x - 1;
    if (dfs(d - 1, x, l)) return true;
    h[x] += b;
    return false;
  }
  for (int i = max(x, l); i <= x + 1; ++ i) {
    if (i <= 1 || i >= n) continue;
    f[d] = i;
    h[i] -= a; h[i + 1] -= b; h[i - 1] -= b;
    if (dfs(d - 1, x, i)) return true;
    h[i] += a; h[i + 1] += b; h[i - 1] += b;
  }
  return false;
}

int main() {
  cin >> n >> a >> b;
  for (int i = 1; i <= n; i++) scanf("%d", h + i);
  for (int ans = 0;; ++ ans) {
    if (dfs(ans, 1, 2)) {
      cout << ans << endl;
      for (int i = ans; i > 0; -- i) cout << f[i] << " ";
      return 0;
    }
  }
}
