#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int n,a,b,h[20],ans=2139062143,ANS[20],w[20];

void dfs(int now, int step)
{
	if(now==n-1)
	{
		if(h[n-2]>=0)return;
		int q=0;
		if(h[n-1]>=0)q=h[n-1]/b+1;
		w[n-2]+=q;
		if(step+q<ans)
		{
			ans=step+q;
			memcpy(ANS,w,sizeof w);
		}
		w[n-2]-=q;
		return;
	}
	if(h[now-1]>=0)
	{
		int q=h[now-1]/b+1;
		h[now-1]-=q*b;
		h[now]-=q*a;
		h[now+1]-=q*b;		
		w[now]+=q;
		dfs(now,step+q);
		h[now-1]+=q*b;
		h[now]+=q*a;
		h[now+1]+=q*b;
		w[now]-=q;
	}
	else
	{
		dfs(now+1,step);
		if(h[now]>=0)
		{
			h[now]-=a;
			h[now+1]-=b;
			w[now]++;
			dfs(now,step+1);
			h[now]+=a;
			h[now+1]+=b;
			w[now]--;
		}
	}
}

int main()
{
	//freopen("yoshi.in","rb",stdin);
	//freopen("yoshi.out","wb",stdout);
	cin>>n>>a>>b;
	for(int i=0;i<n;++i)cin>>h[i];
	dfs(1,0);
	cout<<ans<<endl;
	for(int i=1;i<n-1;++i)
		for(int j=0;j<ANS[i];++j)cout<<i+1<<' ';
	return 0;
}
