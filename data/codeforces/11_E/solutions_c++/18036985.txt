#include <bits/stdc++.h>
using namespace std;

double calc(string s) {
	int p = 0, q = 0, non = 0, one = 0;
	for (int i = 0; i < s.size(); i ++) {
		char w = q & 1 ? 'R' : 'L';
		if (s[i] == 'X') q ++;
		else if (s[i] == w) p ++, q ++, one = 0;
		else p ++, q += 2, non += one, one = !one;
	}
	if (q & 1) q ++, non += one;
	if (p * 2 > q) p -= non, q -= 2 * non;
	return 1e2 * p / q;
}

string s, t;
double ans;

int main() {
//	freopen("e.in", "r", stdin);
	cin >> t;
	for (int i = 0; i < t.size(); i ++) {
		if (i && t[i] != 'X' && t[i] == t[i-1]) s += 'X';
		s += t[i];
	}
	if (s[0] == s[s.size()-1] && s[0] != 'X') ans = max(calc(s + 'X'), calc('X' + s));
	else ans = calc(s);
	printf("%.6lf\n", 1e-6 * floor(ans * 1e6));
	return 0;
}
