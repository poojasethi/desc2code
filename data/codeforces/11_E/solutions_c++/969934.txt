#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define REP(i,n) for(i=0;i<n;i++)

const int MOD=1000000;

int solve(string s){
        int a=0,b=0,c=0,f=0,i;
        REP(i,s.size()){
                if(s[i]=='X') b++;
                else{
                        if (s[i]==(b%2?'R':'L')){
                                a++;b++;f=0;
                        }else{
                                a++;b+=2;c+=f;f^=1;
                        }
                }
        }
        if (b%2) b++,c+=f;
        if (a*2>b) a-=c,b-=2*c;
        return (long long)a*MOD*100/b;
}

int main(){
        string st,s;
        int i;
        cin>>st;
        s="";
        REP(i,st.size()){
                if (i&&st[i]==st[i-1]&&st[i]!='X') s+='X';
                s+=st[i];
        }
        int ans;
        if (s[0]==s[s.size()-1]&&s[0]!='X') ans=max(solve(s+'X'),solve('X'+s));else ans=solve(s);
        printf("%d.%06d",ans/MOD,ans%MOD);
}
