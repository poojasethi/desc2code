ld calc(string s)
{
    int p=0,q=0,o=0,f=0;
    for(int i=0;i<s.length();i++)
    {
        if(s[i]=='X')q++;
        else
            if(s[i]==((q&1)?'R':'L'))p++,q++,f=0;
            else p++,q+=2,o+=f,f=!f;
    }
    if(q&1)q++,o+=f;
    if(2*p>q)p-=o,q-=2*o;
    return 1.0*p/q;
}

    if(t[0]==t[t.length()-1]&&t[0]!='X')
        ans=max(calc('X'+s),calc(s+'X'));

        ans=calc(s);
