#include <iostream>
#include <string>
#include <vector>
#include <cmath>

using namespace std;

int signo(int x)
{
  if (x>0) return 1;
  else if (x<0) return -1;
  return 0;
}

vector<int> agrupa(vector<int> &v)
{
  //if (int(v.size())==0) return v;
  vector<int> vnext;
  vnext.push_back(v[0]);
  for (int i=1;i<int(v.size());i++) {
    if (signo(v[i])==signo(v[i-1]))
      vnext[int(vnext.size())-1]+=v[i];
    else
      vnext.push_back(v[i]);
  }
  return vnext;
}

double solucio(vector<int> v)
{
  int len=int(v.size());
  {
    vector<int> vnext;
    int s=-1;
    for (int i=0;i<int(v.size());i++) {
      if (v[i]!=0) {
	if (signo(v[i])==s)
	  vnext.push_back(1);
	else
	  vnext.push_back(-1);
      }
      s=-s;
    }
    v=vnext;
  }
  if (int(v.size())==0)
    return 0.0;
  v=agrupa(v);
  {
    int s=1;
    for (int i=0;i<int(v.size());i++) {
      v[i]=s*v[i];
      if (signo(v[i])==-1 and v[i]!=-1) {
	v[i]=-v[i];
	s=-s;
	len++;
      }
    }
  }
  int quantsmenys1=0;
  int sumapositius=0;
  for (int i=0;i<int(v.size());i++) {
    if (v[i]==-1) quantsmenys1++;
    else sumapositius+=v[i];
  }
  int menys1alfinal;
  if (v[int(v.size())-1]==-1)
    menys1alfinal=1;
  else
    menys1alfinal=0;
  if (len%2==1) {
    len++;
    if (menys1alfinal) {
      sumapositius++;
      menys1alfinal=0;
      quantsmenys1--;
    }
  }
  if (double(sumapositius)/double(len)<0.5) {
    sumapositius+=quantsmenys1;
    len+=2*quantsmenys1;
  }
  return 100.0*double(sumapositius)/double(len);
}

    if (input[i]!=0 and input[i]==input[i-1])
      v.push_back(0);
    v.push_back(input[i]);
  }
  vector<int> v1=v;
  vector<int> v2=v;
  if (int(input.size())!=0 and input[0]!=0 and
      input[int(input.size())-1]==input[0]) {

  cout<<int(1000000*(max(solucio(v1),solucio(v2))))/1000000.0<<endl;
