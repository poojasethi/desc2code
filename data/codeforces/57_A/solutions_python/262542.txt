s = raw_input()
s = s.split()
n = int(s[0])
p1 =  (int(s[1]), int(s[2]))
p2 = (int(s[3]),int(s[4]))
ans = 0
v = (p2[0] - p1[0], p2[1] - p1[1])
if abs(v[0]) != n and abs(v[1]) != n:
    ans = abs(v[0]) + abs(v[1])
else:
    if abs(p1[0] - p2[0]) == n:
        l1 = p1[1] + p2[1]
        l2 = 2 * n -l1
        l = min(l1, l2)
        ans = n + l
    else:
        l1 = p1[0] + p2[0]
        l2 = 2 * n -l1
        l = min(l1, l2)
        ans = n + l
print ans
