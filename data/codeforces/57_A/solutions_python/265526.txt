n, x1, y1, x2, y2 = map(int, raw_input().split())
if x1 == x2 == 0 or x1 == x2 == n:
    print abs(y1-y2)
elif y1 == y2 == 0 or y1 == y2 == n:
    print abs(x1-x2)
elif abs(x1-x2) < n and abs(y1-y2) < n or abs(x1-x2) == n and abs(y1-y2) == n:
    print abs(x1-x2) + abs(y1-y2)
elif abs(x1-x2) == n:
    print min(y1+y2, n-y1+n-y2) + n
else: print min(x1+x2, n-x1+n-x2) + n
