m,x1,y1,x2,y2 = map(int,raw_input().split())
if (x1==0 and x2==m) or (x2==0 and x1==m):
    d1 = m-y1 + m + m-y2
    d2 = y1 + m + y2
    d = min(d1,d2)
elif (y1==0 and y2==m) or (y2==0 and y1==m):
    d1 = m-x1 + m + m-x2
    d2 = x1 + m + x2
    d = min(d1,d2)
else:
    d = abs(y1-y2) + abs(x1-x2)    
print d
