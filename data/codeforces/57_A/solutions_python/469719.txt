n,x1,y1,x2,y2=map(int,raw_input().split())

if (x1==y1)and (x2==y2):
    ans = 2*n
elif (x1==0 and x2==0) or (x1==n and x2==n):
    ans=abs(y1-y2)

elif (y1==0 and y2==0) or (y1==n) and (y2==n):
    ans=abs(x1-x2)

elif (x1==0 and x2==n) or (x1==n and x2==0):
    one = y1+n+y2
    two = 4*n-one
    ans=min(one,two)

elif(y1==0 and y2==n) or (y1==n and y2==0):
    one = x1+n+x2
    two = 4*n-one
    ans=min(one,two)

else: ans=abs(x1-x2)+abs(y1-y2)


print ans
