n,x1,y1,x2,y2 = map(int,raw_input().split())
if abs(x1-x2) == n:
    d1 = y1 + y2 + n
    d2 = 3*n - y1 - y2
    print min(d1,d2)
elif abs(y1-y2) == n:
    d1 = x1 + x2 + n
    d2 = 3*n - x1 - x2
    print min(d1,d2)
elif x1 == x2 == n or x1 == x2 == 0: print abs(y2-y1)
elif y1 == y2 == n or y1 == y2 == 0: print abs(x2-x1)        
else: print abs(x2-x1) + abs(y2-y1)
