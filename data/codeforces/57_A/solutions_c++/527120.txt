#include<cstdio>
#include<algorithm>
using namespace std;

int n,x1,y1,x2,y2;

int main() {
	scanf("%d%d%d%d%d",&n,&x1,&y1,&x2,&y2);
	int cnt=0;
	while (!(x1==x2 && y1==y2)) {
		if (x1==0 && y1>0) y1--;
		else if (x1==n && y1<n) y1++;
		else if (y1==0 && x1<n) x1++;
		else x1--;
		cnt++;
	}
	printf("%d\n",min(4*n-cnt,cnt));
}
