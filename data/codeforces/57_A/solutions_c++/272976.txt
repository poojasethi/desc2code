#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int n,x,y,X,Y,ans1,ans2;
	cin>>n>>x>>y>>X>>Y;
	ans1=(x+y+X+Y); ans1=min(ans1,4*n-ans1);
	ans2=abs((double)(n-x)+y+(double)(n-X)+Y);  ans2=min(ans2,4*n-ans2);
	if(x==X && x%n==0) ans1=min(ans1,int(abs((double)(Y-y))));
	if(y==Y && y%n==0) ans2=min(ans2,int(abs((double)(x-X))));
	cout<<min(ans1,ans2)<<endl;
	return 0;
}
