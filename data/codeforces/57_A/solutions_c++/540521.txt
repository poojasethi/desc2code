#include<stdio.h>
#include<algorithm>
using namespace std;
int test ( int x, int y, int n){
    if ( x >= y ) return x+y;
    return 4*n-(x+y);
}
int main(void)
{
    int n,a,b,c,d;
    scanf("%d%d%d%d%d", &n,&a,&b,&c,&d);
    a = test(a,b,n);
    c = test(c,d,n);
    if ( a > c)
       swap(a,c);
    printf("%d\n", min(c-a,4*n-(c-a)));
    return 0; 
}
