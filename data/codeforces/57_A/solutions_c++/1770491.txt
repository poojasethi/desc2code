#include <iostream>
#include <windows.h>
using namespace std;

int main()
{
	int n, x1, y1, x2, y2, t1, t2;

	cin >> n >> x1 >> y1 >> x2 >> y2;
	t1 = (x1 <= y1 ? x1 + y1 : 4 * n - x1 - y1);
	t2 = (x2 <= y2 ? x2 + y2 : 4 * n - x2 - y2);
	int ans = abs(t1 - t2);
	cout << min(ans, 4 * n - ans) << endl;
	return 0;
}
