#include <iostream>

using namespace std;

int abs(int i) { return (i < 0 ? -i : i); }

int main()
{
	int n, x1, y1, x2, y2, t1, t2, l;
	cin >> n >> x1 >> y1 >> x2 >> y2;
	
	t1 = (x1 <= y1 ? x1 + y1 : 4 * n - x1 - y1);
	t2 = (x2 <= y2 ? x2 + y2 : 4 * n - x2 - y2);
	
	l = abs(t2 - t1);
	
	cout << min(l, 4 * n - l) << endl;
	
	return 0;
}
