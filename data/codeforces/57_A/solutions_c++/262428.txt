#include <iostream>
#include <cmath>
using namespace std;
int n,x1,Y1,x2,y2;
double rast(int x,int y)
{
	if (x==0)return y;
	else if (y==n)return n+x;
	else if (x==n)return 3*n-y;
	return 4*n-x;
}
int main()
{
	cin>>n>>x1>>Y1>>x2>>y2;
	cout<<(int)min(fabs(rast(x1,Y1)-rast(x2,y2)),4*n-fabs(rast(x1,Y1)-rast(x2,y2)));
	return 0;
}
