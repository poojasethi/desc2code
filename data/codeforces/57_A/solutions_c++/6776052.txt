
#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;

int main()
{
   int n,x1,x2,y1,y2;
   cin>>n>>x1>>y1>>x2>>y2;
   int s;
   if(abs(x1-x2)==n||abs(y1-y2)==n)
   s=x1+x2+y1+y2;
    else
   s=abs(x1-x2)+abs(y1-y2);

   if(s>2*n)
    cout<<4*n-s<<endl;
   else
    cout<<s<<endl;

}
