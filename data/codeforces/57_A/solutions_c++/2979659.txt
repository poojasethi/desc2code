#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n,x1,y1,x2,y2;
    
    cin>>n>>x1>>y1>>x2>>y2;
    
    int l;
    
    if (abs(y1-y2)==n) l = n + x1+x2; else
    if (abs(x1-x2)==n) l = n + y1+y2; else l = abs(x1-x2) + abs(y1-y2);
    
    cout<<min(l,4*n-l)<<endl;
    //system("pause");
    return 0;
}
