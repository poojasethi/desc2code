#include<stdio.h>
#define abs(x) ((x)>0?(x):-(x))
#define INF 1<<30
int main()
{
	int n,x1,x2,y1,y2;
	scanf("%d%d%d%d%d",&n,&x1,&y1,&x2,&y2);
	int sum1=INF,sum2=INF;
	if(abs(x1-x2)==n||abs(y1-y2)==n)
		sum1=x1+x2+y1+y2,sum2=(n<<2)-sum1;
	else
		sum1=abs(y1-y2)+abs(x1-x2);
    printf("%d\n",sum1<sum2?sum1:sum2);
}
