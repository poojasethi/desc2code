#include <iostream>
#include <cmath>

using namespace std;

int conv(int x, int y, int n) {
    if(x >= y) return x + y;
    return 4*n - x - y;
}

int main() {
    int n, x1, y1, x2, y2;
    cin >> n >> x1 >> y1 >> x2 >> y2;
    int v1 = conv(x1, y1, n), v2 = conv(x2, y2, n);

    if(v2 < v1) swap(v1, v2);
    cout << min(v2 - v1, 4*n - (v2 - v1)) << endl;
}
