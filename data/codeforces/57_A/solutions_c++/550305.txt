#include<iostream>
#include<cmath>
int main()
{
	using namespace std;
	int n,x1,y1,x2,y2;
	cin>>n>>x1>>y1>>x2>>y2;
	if(abs(x1-x2)==n||abs(y1-y2)==n)
	{
		int sum=abs(n-x1)+abs(n-y1)+abs(n-x2)+abs(n-y2);
		if(sum>2*n)
			sum=4*n-sum;
		cout<<sum<<endl;
	}
	else
		cout<<abs(x1-x2)+abs(y1-y2)<<endl;
	return 0;
}
