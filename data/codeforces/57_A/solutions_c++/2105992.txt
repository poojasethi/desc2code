#include<stdio.h>

int abs(int a){return a>0?a:-a;}

int change(int n,int a,int b){
	if(a==0){return -b;}
	if(b==0){return a;}
	if(a==n){return n+b;}
	if(b==n){return 3*n-a;}
	return 0;
}

int main(){
	int n,a,b,c,d,ans;
	scanf("%d%d%d%d%d",&n,&a,&b,&c,&d);
	ans=abs(change(n,a,b)-change(n,c,d));
	if(ans>2*n){ans=4*n-ans;}
	printf("%d\n",ans);
	return 0;
}
