#include<stdio.h>
#include<stdlib.h>
int main()
{
	int n,x1,x2,y1,y2,len;
	while(scanf("%d%d%d%d%d",&n,&x1,&y1,&x2,&y2) != EOF)
	{
		if(abs(y1-y2)==n)
			len=n+x1+x2;
		else
		{
			if(abs(x1-x2)==n)
				len=n+y1+y2;
			else
				len=abs(x1-x2)+abs(y1-y2);
		}
		if(4*n-len<len)
			len=4*n-len;
		printf("%d\n",len);
	}
	return 0;
}
