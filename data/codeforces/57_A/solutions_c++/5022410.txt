#include<iostream>
#include<cstdlib>
using namespace std;

int main()
{
    int n,x1,y1,x2,y2;
    int ans;
    cin>>n>>x1>>y1>>x2>>y2;
    if(abs(x1-x2)==n)
        ans=n+y1+y2;
    else if(abs(y1-y2)==n)
        ans=n+x1+x2;
    else
        ans=abs(x1-x2)+abs(y1-y2);
    cout<<min(ans,4*n-ans)<<endl;
    return 0;
}
