#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,x1,y1,x2,y2;

int main(){
  int i,j,k;
  scanf("%d%d%d%d%d",&n,&x1,&y1,&x2,&y2);
  if (x1>=y1 && x2>=y2 || x1<=y1 && x2<=y2) printf("%d\n",abs(x1-x2)+abs(y1-y2));
  else printf("%d\n",min(x1+x2+y1+y2,4*n-x1-x2-y1-y2));
  return 0;
}
