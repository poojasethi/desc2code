#include <cstdlib>
using namespace std;
int main()
{
    int n,x1,y1,x2,y2,t1,t2;
    cin >> n >> x1 >> y1 >> x2 >> y2;
    if (x1<=y1) t1 = x1+y1;
    else t1 = 4*n-x1-y1;
    if (x2<=y2) t2 = x2+y2;
    else t2 = 4*n-x2-y2;
	int fuck = abs(t2-t1);
