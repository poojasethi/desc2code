#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n , a , b , c , d ;
    scanf("%d%d%d%d%d",&n,&a,&b,&c,&d);
    if( abs(a-c)==n )
        {
        cout<<n+min( b+d ,(2*n)-b-d )<<endl;
        }
    else if( abs(b-d)==n ){
        cout<<n+min( a+c ,(2*n)-a-c )<<endl;
    }else {
     cout<<abs(a-c)+abs(b-d)<<endl;
    }

    return 0;
}
