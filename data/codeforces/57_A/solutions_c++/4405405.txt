#include<iostream>
using namespace std;
int main()
{
    int s,i,x1,y1,x2,y2,len,m1,m2,t,p;
    while(cin>>s>>x1>>y1>>x2>>y2)
    {
        m1=x1+y1;m2=x2+y2;
        t=max(m1,m2);
        p=min(m1,m2);
        if((y1<x1&&y2<x2)||(y1>x1&&y2>x2))len=t-p;
        else len=min(t+p,4*s-t-p);
        cout<<len<<endl;
    }
    return 0;
}
