#include<stdio.h>
int abs(int a)
{
	return a>0?a:-a;
}
int main()
{
	int x1,x2,y1,y2,n;
	scanf("%d%d%d%d%d",&n,&x1,&y1,&x2,&y2);
	if(abs(x1-x2)==n)printf("%d\n",y1+y2>n?3*n-y1-y2:y1+y2+n);
	else if(abs(y2-y1)==n)printf("%d\n",x1+x2>n?3*n-x1-x2:x1+x2+n);
	else printf("%d\n",abs(x1-x2)+abs(y1-y2));
	return 0;
}
