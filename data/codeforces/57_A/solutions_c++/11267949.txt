#include <iostream>
#include <algorithm>

using namespace std;



int main()
{
    int d1,n,x1,x2,y1,y2,dx,dy;

    cin >> n >> x1 >> y1 >> x2 >> y2;

    dx = abs(x1-x2);
    dy = abs(y1-y2);

    if(dx==n) d1 = n +y1 +y2;
    else if (dy==n) d1 = n +x1 +x2;
    else d1 = dy + dx;

    cout << min(4*n-d1,d1) << endl;




    return 0;
}
