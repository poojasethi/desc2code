#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	int n,x1,x2,y1,y2;
	cin>>n>>x1>>y1>>x2>>y2;
	int sum1=1000000,sum2=1000000;
	if(abs(x1-x2)==n || abs(y1-y2)==n)
	{
		sum1=x1+x2+y1+y2;
		sum2=n-x1+n-x2+n-y1+n-y2;
	}
	else
	{
		sum1=abs(y1-y2)+abs(x1-x2);
	}
	cout<<min(sum1,sum2)<<endl;
	return 0;
}
