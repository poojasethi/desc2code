#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,x1,y1,x2,y2,res1;
    cin>>n>>x1>>y1>>x2>>y2;
    if(x1==0 && x2==0 || x1==n && x2==n)res1=abs(y1-y2);
    else
        if(y1==0 && y2==0 || y1==n && y2==n)res1=abs(x1-x2);
        else res1=min(y1+y2,n-y1+n-y2)+min(x1+x2,n-x2+n-x1);
    cout<<res1<<"\n";
    return 0;
}
