#include <bits/stdc++.h>

using namespace std;

const int maxN = 1000 + 321;
 
int main()
{
  int n, x1, x2, y1, y2;
  cin >> n >> x1 >> y1 >> x2 >> y2;
  if(x1 <= y1) x1 += y1;
  else x1 = 4 * n - x1 - y1;
  if(x2 <= y2) x2 += y2;
  else x2 = 4 * n - x2 - y2;
  int ans = abs(x1 - x2);
  cout << min(ans, 4 * n - ans) << endl;
  return 0;
}
