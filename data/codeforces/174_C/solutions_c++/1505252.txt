#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
#include <cmath>
#define N 111111
using namespace std;
int n,a[N],c[N],kol,k;
pair<int,int> d[N];
int main()
{
    scanf("%d",&n);
    for (int i=1;i<=n;i++) scanf("%d",&a[i]);
    for (int i=1;i<=a[1];i++)c[++kol]=1;
    for (int i=2;i<=n;i++){
        if (a[i]>=a[i-1])
          for (int j=a[i-1]+1;j<=a[i];j++)c[++kol]=i;
        else 
          for (int j=a[i]+1;j<=a[i-1];j++)
              d[++k]=make_pair(c[kol--],i-1);
        }
    for (int i=kol;i>=1;i--)d[++k]=make_pair(c[i],n);
    printf("%d\n",k);
    for (int i=1;i<=k;i++) printf("%d %d\n",d[i].first,d[i].second);
    cin>>n;
    return 0;
}
