#include <cstdio>
#include <iostream>
using namespace std;
const int N = 101000;
int n, a[N], b[N], c[N], d[N], cc, cd;
int main ()
{
	scanf ("%d", &n);
	for (int i = 1; i <= n; i ++)
		scanf ("%d", &a[i]);
	for (int i = 1; i <= n + 1; i ++)
		b[i] = a[i] - a[i - 1];
	for (int i = 1; i <= n + 1; i ++)
	{
		if (b[i] == 0) continue;
		if (b[i] > 0)
			for (int j = 1; j <= b[i]; j ++)
				c[++ cc] = i;
		else
			for (int j = 1; j <= -b[i]; j ++)
				d[++ cd] = i - 1;
	}
	printf ("%d\n", cc);
	for (int i = 1; i <= cc; i ++)
		printf ("%d %d\n", c[i], d[i]);
	return 0;
}
