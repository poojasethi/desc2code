#include <cstdio>
#include <cstdlib>
using namespace std;

const int N = 100003;
int a[N];

int main()
{
    int n;
    scanf("%d", &n);
    for(int i = 1; i <= n; ++i)
        scanf("%d", a+i);
    int res = 0;
    for(int i = 1; i <= n+1; ++i)
        res += abs(a[i-1]-a[i]);
    printf("%d\n", res/2);
    
    int c = 0;
    int b = 1, e = 2;
    while(e <= n+1) {
        while(b < e && a[b]-c == 0)
            ++b;
        if(b == e || a[e-1]-c <= a[e]) {
            a[e] += c;
            ++e;
        }
        else {
            printf("%d %d\n", b, e-1);
            ++c;
        }
    }
    return 0;
}
