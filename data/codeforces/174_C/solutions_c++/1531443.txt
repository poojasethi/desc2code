#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <vector>
#include <ctime>
using namespace std;
#define f first
#define s second

int n,a[100004],m,k,x;
pair <int,int> ans[200000];

int main () {
	#ifdef LOCAL
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	#endif
	scanf("%d",&n);
	for(int i=1;i<=n;++i) {
		scanf("%d",&a[i]);
	}
	for(int i=1;i<=n+1;++i) {
		if(a[i] > x) {
			int l = a[i] - x;
			for(int j=0;j<l;++j)
				ans[m++].f = i;
		}
		else if(a[i] < x) {
			int l = x - a[i];
			for(int j=0;j<l;++j)
				ans[k++].s = i - 1;
		}
		x = a[i];
	}
	printf("%d\n",m);
	for(int i=0;i<m;++i)
		printf("%d %d\n",ans[i].f,ans[i].s);
	return 0;
}
