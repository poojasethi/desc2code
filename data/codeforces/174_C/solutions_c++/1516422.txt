#include <iostream>
using namespace std;
int num[100010],p[100010][2],c[2]={};
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;++i)
		cin>>num[i];
	int i=0,j;
	for(j=0;num[i]-j>0;p[c[0]++][0]=i,++j);
	for(++i;i<n;++i)
	{
		if(num[i]>num[i-1])
			for(j=0;num[i]-num[i-1]-j>0;p[c[0]++][0]=i,++j);
		else if(num[i]<num[i-1])
			for(j=0;num[i-1]-num[i]-j>0;p[c[1]++][1]=i-1,++j);
	}
	for(j=0;num[n-1]-j>0;p[c[1]++][1]=n-1,++j);
	cout<<c[0]<<endl;
	for(i=0;i<c[0];++i)
		cout<<p[i][0]+1<<" "<<p[i][1]+1<<endl;
	return 0;
}
