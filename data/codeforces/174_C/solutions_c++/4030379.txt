#include <stdio.h>
#include <stdlib.h>
#include <string.h>
using namespace std;
const int maxn = 100010;
int ans[maxn][2];
void run(){
	int i = 0 , n;
	scanf("%d",&n);
	int x , p = 0 , q = 0 , cur = 0;
	for(int i = 1 ; i <= n + 1 ; i ++){
		if(i <= n) 
			scanf("%d",&x);
		else 
			x = 0;
		while(cur < x){
			ans[p++][0] = i;
			cur ++;
		}
		while(cur > x){
			ans[q++][1] = i-1;
			cur --;
		}
	}
	printf("%d\n",p);
	for(i = 0 ; i < p; i++){
		printf("%d %d\n",ans[i][0],ans[i][1]);
	}
}
int main(){
	run();
	return 0;
}
