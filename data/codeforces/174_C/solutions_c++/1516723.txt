#include <iostream>
#include <stack>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	stack<int> s;
	int n;
	cin >> n;
	vector<pair<int, int> > v;
	for (int i = 0; i < n; i++) {
		int a;
		cin >> a;
		while (a > s.size()) {
			s.push(i + 1);
		}
		while (a < s.size()) {
			v.push_back(make_pair(s.top(), i));
			s.pop();
		}
	}
	while (s.size()) {
		v.push_back(make_pair(s.top(), n));
		s.pop();
	}
	cout << v.size() << endl;
	for (int i = 0; i < v.size(); i++) {
		cout << v[i].first << " " << v[i].second << endl;
	}
	return 0;
}
