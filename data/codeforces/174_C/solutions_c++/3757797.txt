#include<stdio.h>
#include<stack>
#include<vector>
#include<map>
using namespace std;
typedef pair<int,int> P;
int ar[100002];
vector<P> ans;
int main(){
    int n,i,j;
    scanf("%d",&n);
    ar[0]=0;
    ar[n+1]=0;
    for(i=1;i<=n;i++)scanf("%d",&ar[i]);
    stack<int> s;
    for(i=1;i<=n+1;i++){
        if(ar[i]>ar[i-1]){
            for(j=0;j<ar[i]-ar[i-1];j++)s.push(i);
        }
        else if(ar[i]<ar[i-1]){
            for(j=0;j<ar[i-1]-ar[i];j++){
                ans.push_back(P(s.top(),i-1));
                s.pop();
            }
        }
    }
    printf("%d\n",ans.size());
    for(i=0;i<ans.size();i++)printf("%d %d\n",ans[i].first,ans[i].second);
    return 0;
}
