#include <iostream>
#include <stack>
#include <utility>
#include <vector>

using namespace std;

stack<int> pos;
int N,P;
vector<pair<int,int> > ans;

int main()
{
	cin >> N;
	for (int i=1;i<=N+1;i++)
	{
		int a;
		if (i<N+1)
			cin >> a;
		else
			a=0;
		if (P<a)
			for (int j=P;j<a;j++)
				pos.push(i);
		else if (P>a)
			for (int j=P;j>a;j--)
			{
				ans.push_back(make_pair(pos.top(),i-1));
				pos.pop();
			}
		P=a;
	}
	cout << ans.size() << endl;
	for (int i=0;i<ans.size();i++)
		cout << ans[i].first << " " << ans[i].second << endl;
	return 0;
}
