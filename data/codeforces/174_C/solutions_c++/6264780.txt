#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>

using namespace std;

const int maxn = 100005;

int data[maxn],n,p,q,x,l[maxn],r[maxn];
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d",data+i);
    for(int i=1;i<=n+1;i++)
    {
        while(x<data[i])
            l[p++]=i,x++;
        while(x>data[i])
            r[q++]=i-1,x--;
    }
    printf("%d\n",p);
    for(int i=0;i<p;i++)
        printf("%d %d\n",l[i],r[i]);
}
