#include<bits/stdc++.h>
#define MAX 123456
using namespace std;

int main() {
    int n, tmp = 0, l = 0, h = 0, arr[MAX];
    pair<int, int> ret[MAX];
    cin >> n;
    for (int i = 1; i < n + 1; i++)
        cin >> arr[i];
    arr[n + 1] = 0;
    for (int i = 1; i <= n + 1; i++) {
        while (tmp < arr[i])
            ret[l++].first = i, tmp++;
        while (tmp > arr[i])
            ret[h++].second = i - 1, tmp--;
    }
    cout << l << endl;
    for (int i = 0; i < l; i++)
        cout << ret[i].first << " " << ret[i].second << endl;

    return 0;
}
