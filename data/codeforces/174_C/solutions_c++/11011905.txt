// 174 C
#include <bits/stdc++.h>
#define MAX 111111
#define pii pair<int, int>
using namespace std;

int st[MAX], en[MAX];
int A[MAX];
stack<int> S;

int main() 
{
    int N;
    cin >> N;
    int cnt = 0;
    for (int i = 1; i <= N; i++)
    {
        cin >> A[i];
        while (S.size() > A[i])
        {
            st[cnt] = S.top();
            en[cnt] = i - 1;
            S.pop();
            cnt++;
        }
        while (S.size() < A[i])
            S.push(i);
    }
    while (!S.empty())
    {
        st[cnt] = S.top();
        en[cnt] = N;
        S.pop();
        cnt++;
    }
    cout << cnt << endl;
    for (int i = 0; i < cnt; i++)
        printf("%d %d\n", st[i], en[i]);
}
