#include <cstdio>
#include <iostream>

using namespace std;
const int MAXN = 100010;

int N, inc[MAXN], dc[MAXN];

int main() {
    cin >> N;
    int curlevel = 0;
    int ind1 = 0, ind2 = 0;
    for(int i = 1; i <= N+1; i++) {
        int a = 0;
        if(i <= N) cin >> a;
        while(curlevel < a) {
            curlevel++;
            inc[ind1++] = i;
        }
        while(a < curlevel) {
            curlevel--;
            dc[ind2++] = i-1;
        }
    }
    cout << ind1 << "\n";
    for(int i = 0; i < ind1; i++) {
        cout << inc[i] << " " << dc[i] << "\n";
    }
}
