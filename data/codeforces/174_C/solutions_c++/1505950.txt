#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
#include <cmath>
#define N 111111
using namespace std;
int n,a[N],c[N],kol,k,e[N],f[N];
int main()
{
    while( scanf( "%d",&n ) == 1 )
    {
           k = 0;
        for (int i=1;i<=n;i++) scanf("%d",&a[i]);
        a[0] = a[n+1] = 0;
        for (int i=1;i<=n+1;i++){
            if (a[i-1] <= a[i])
              for (int j=a[i-1]+1;j<=a[i];j++)c[++kol]=i;
            else 
              for (int j=a[i]+1;j<=a[i-1];j++)
                  e[++k] = c[kol--],f[k] = i-1;
            }
        printf("%d\n",k);
        for (int i=1;i<=k;i++) printf("%d %d\n",e[i],f[i]);
    }
    return 0;
}
