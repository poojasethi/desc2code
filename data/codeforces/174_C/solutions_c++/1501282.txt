#include <utility>
#include <vector>
#include <iostream>
using namespace std;

vector<pair<int, int> > z[100000];

int main() {
	int n;
	int a = 0, b;
	int s = 0;

	ios_base::sync_with_stdio(false);

	cin >> n;
	for (int i = 0; i < n; ++i) {
		cin >> b;
		if (b > a)
			for (int j = a; j < b; ++j)
				z[j].push_back(make_pair(i, -1));
		else if (b < a)
			for (int j = a - 1; j >= b; --j)
				z[j].back().second = i;
		a = b;
	}
	if (0 < a)
		for (int j = a - 1; j >= 0; --j)
			z[j].back().second = n;
	for (int i = 0; i < 100000; ++i)
		s += z[i].size();
	cout << s << "\n";
	for (int i = 0; i < 100000; ++i)
		for (int j = 0; j < (int)z[i].size(); ++j)
			cout << z[i][j].first + 1 << " " << z[i][j].second << "\n";
	return 0;
}
