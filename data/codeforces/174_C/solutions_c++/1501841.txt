#include <cstdio>
#include <cstring>

int prev[111111];
int l[111111];
int r[111111];

int main(){
  int n;
  int cnt = 0;
  scanf("%d", &n);
  memset(prev, -1, sizeof(prev));
  for(int i = 0; i < n; ++i){
    int x;
    scanf("%d", &x);
    for(int j = x + 1; ~prev[j]; ++j){
      l[cnt] = prev[j];
      r[cnt] = i - 1;
      prev[j] = -1;
      ++cnt;
    }
    for(int j = x; j > 0 && prev[j] == -1; --j){
      prev[j] = i;
    }
  }
  for(int i = 1; ~prev[i]; ++i){
      l[cnt] = prev[i];
      r[cnt] = n - 1;
      prev[i] = -1;
      ++cnt;
  }
  printf("%d\n", cnt);
  for(int i = 0; i < cnt; ++i)
    printf("%d %d\n", l[i] + 1, r[i] + 1);
}
