#include<cstring>
#include<cstdio>
int a,b;
int x[100002];
int l[100002];
int r[100002];
int main()
{
	int n,ans,c;
	int f;
	while(scanf("%d",&n)==1)
	{
		for(int i=0;i<n;i++) scanf("%d",&x[i]);
		x[n]=0;
		f=0;
		a=b=0;
		for(int i=0;i<=n;i++)
		{
			if(x[i]>f)
			{
				c=x[i]-f;
				for(int j=0;j<c;j++) 
				{
					l[a]=i;
					a++;
				}
			}
			if(x[i]<f)
			{
				c=f-x[i];
				for(int j=0;j<c;j++) 
				{
					r[b]=i-1;
					b++;
				}
			}
			f=x[i];
		}
		printf("%d\n",b);
		for(int i=0;i<a;i++) printf("%d %d\n",l[i]+1,r[i]+1);
	}
	return 0;
}
