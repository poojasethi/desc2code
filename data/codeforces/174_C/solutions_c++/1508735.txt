#include <cstdio>

int vet[100001], n;
int res[100001][2];

int main()
{
	scanf("%d", &n);
	for (int i = 0; i < n; ++i)
		scanf("%d", &vet[i]);
	
	int x = 0, a = 0, b = 0;
	
	for (int i = 0; i <= n; ++i)
	{
		if (vet[i] > x) for (int j = vet[i]-x; j > 0; --j) res[a++][0] = i;
		else if (vet[i] < x) for (int j = x-vet[i]; j > 0; --j) res[b++][1] = i-1;
		x = vet[i];
	}
	
	printf("%d\n", a);
	for (int i = 0; i < a; ++i)
		printf("%d %d\n", res[i][0]+1, res[i][1]+1);
	
	return 0;
}
