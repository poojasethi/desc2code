#include <cstdio>
#define N 100010

int st[N],tp=0,x[N],y[N],a[N],bf[N];

int main()
{
  int n,i,j,re,cnt=0;
  scanf("%d",&n);
  for (i=0;i<=n;i++)
  {
    bf[i]=i;
    if (i!=n)
      scanf("%d",&a[i]);
    else
      a[i]=0;
    while(tp>0&&a[i]<a[st[tp-1]])
    {

      if (tp>1&&a[st[tp-2]]>a[i])re=a[st[tp-1]]-a[st[tp-2]];
      else
      {
        re=a[st[tp-1]]-a[i];
        bf[i]=bf[st[tp-1]];
      }
      for (j=cnt;j<cnt+re;j++)
      {
        x[j]=bf[st[tp-1]]+1;
        y[j]=i;
      }
      cnt=j;
      tp--;
    }
    if (!(a[i]==0||(tp>0&&a[i]==a[st[tp-1]])))
      st[tp++]=i;
  }
  printf("%d\n",cnt);
  for (i=0;i<cnt;i++)
    printf("%d %d\n",x[i],y[i]);
}
