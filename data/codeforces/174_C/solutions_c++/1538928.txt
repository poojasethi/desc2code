#include<stdio.h>

#define maxN 100004

int n;
int a[maxN];
int stack[maxN],pos[maxN],last;
int sum;

int main(){
 scanf("%d",&n);
 int i,j;
 a[0] = 0;
 for(i=1;i<=n;i++){
  scanf("%d",a+i);
  if(a[i]>a[i-1]) sum+=a[i]-a[i-1];
 }
 printf("%d\n",sum);
 last = 0;
 for(i=1;i<=n;i++){
  if(a[i]>a[i-1]){
   for(j=a[i-1]+1;j<=a[i];j++){
    stack[last]=j;
    pos[last++]=i;
   }
  }
  while(last>0 && a[i]<stack[last-1]){
   printf("%d %d\n",pos[--last],i-1);
  }
 }
/*
printf("last: %d\n",last);
for(i=0;i<last;i++) printf(" %d",stack[i]);
printf("\n");
*/
 while(last>0 && 0<stack[last-1]){
  printf("%d %d\n",pos[--last],i-1);
 }

 return 0;
}
