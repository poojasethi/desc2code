#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

vector<int>v1,v2;

const int max_n = 100010;
int a[max_n];
int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    for(int i=0;i<=n+1;i++)
    {
        if(a[i]>a[i+1])
            for(int j=0;j<a[i]-a[i+1];j++)
                v2.push_back(i);
    
        if(a[i]<a[i+1])
            for(int j=0;j<a[i+1]-a[i];j++)
                v1.push_back(i+1);
    }
    sort(v1.begin(),v1.end());
    sort(v2.begin(),v2.end());
    cout<<v1.size()<<endl;
    for(int i=0;i<v1.size();i++)
        cout<<v1[i]<<" "<<v2[i]<<endl;
    return 0;
}
