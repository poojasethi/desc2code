#include <iostream>
#include <cstdio>
using namespace std;
int a[100005]={0},b[100005]={0},c[100005];
int main(){
    int n,i,ans=0,e=0,j;
    scanf("%d",&n);
    for(i=1;i<=n;i++){
        scanf("%d",&a[i]);
    }
    for(i=1;i<=n+1;i++){
        b[i]=a[i]-a[i-1];
    }
    for(i=1;i<=n+1;i++){
        if(b[i]>0){
            ans+=b[i];
            }
        }
    printf("%d\n",ans); 
    for(i=1;i<=n+1;i++){
        if(b[i]>0){
            for(j=1;j<=b[i];j++){
                e++;
                c[e]=i;
            }
        }
        if(b[i]<0){
            for(j=1;j<=-b[i];j++){
                printf("%d %d\n",c[e],i-1);
                e--;
            }
        }
    }
    return 0;
}
