n = int(raw_input())
a = map(int, raw_input().split())
a.append(0)

stack = []
result = []
for i, val in enumerate(a):
    x = len(stack)
    if x > val:
        for _ in xrange(x - val):
            result.append((stack.pop(), i))
    elif x < val:
        for _ in xrange(val - x):
            stack.append(i + 1)

print len(result)
print '\n'.join("%d %d" % p for p in result)
