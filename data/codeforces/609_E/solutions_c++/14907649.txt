//In The Name of God ;

#include <bits/stdc++.h>

using namespace std;

#define mp make_pair
#define pb push_back

const int Maxn=2e5 + 7;

int p[Maxn],h[Maxn];
long long ret;
int par[20][Maxn],mx[20][Maxn];
vector< pair<int,pair<int,int> > >e,E;
vector< pair<int,int> >G[Maxn];

void dfs(int u,int dad = 0,int ww = 0)
{
  par[0][u]=dad;
  mx[0][u]=ww;
  for(int i=1;i<20;i++)
    par[i][u]=par[i-1][par[i-1][u]],
      mx[i][u]=max(mx[i-1][u],mx[i-1][par[i-1][u]]);
  for(int i=0;i<G[u].size();i++)
    {
      int v=G[u][i].second,w=G[u][i].first;
      if(v!=dad)
	h[v]=h[u]+1,dfs(v,u,w);
    }
}

int LCA(int u,int v)
{
  if(h[v]<h[u])
    swap(u,v);
  int ans=0;
  for(int i=19;i>=0;i--)
    if(h[par[i][v]]>=h[u])
      ans=max(ans,mx[i][v]),v=par[i][v];
  if(v==u)
    return ans;
  for(int i=19;i>=0;i--)
    {
      int p1=par[i][v],p2=par[i][u];
      if(p1!=p2)
	{
	  ans=max(mx[i][u],ans);
	  ans=max(mx[i][v],ans);
	  v=p1,u=p2;
	}
    }
  ans=max(mx[0][u],ans);
  ans=max(mx[0][v],ans);
  return ans;
}

int Par(int v)
{
  if(v==p[v])
    return v;
  return p[v]=Par(p[v]);
}

main()
{
  ios_base::sync_with_stdio(0);
  int n,m;
  cin>>n>>m;
  for(int i=1;i<=n;i++)
    p[i]=i;
  
  for(int u,v,w,i=0;i<m;i++)
    {
      cin>>u>>v>>w;
      e.pb(mp(w,mp(u,v)));
      E.pb(mp(w,mp(u,v)));
    }
  sort(e.begin(),e.end());
  for(int u,v,w,i=0;i<e.size();i++)
    {
      u=e[i].second.first;
      v=e[i].second.second;
      w=e[i].first;
      if(Par(v)!=Par(u))
	{
	  p[p[v]]=p[u];
	  G[v].pb(mp(w,u));
	  G[u].pb(mp(w,v));
	  ret+=w;
	}
    }
  h[1]=1;
  dfs(1);
  for(int u,v,w,c,i=0;i<m;i++)
    {
      v=E[i].second.first;
      u=E[i].second.second;
      w=E[i].first;
      c=LCA(u,v);
      cout<<w-c+ret<<endl;
    }
  return 0;
}
