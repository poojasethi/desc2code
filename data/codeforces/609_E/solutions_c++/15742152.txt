#include <bits/stdc++.h>
using namespace std;
const int N = 2e5+5;
struct E{
	int u,v,w,id;
}e[N];
bool operator<(const E a,const E b){
	return a.w<b.w;
}
int con[N],par[N],sz[N],add[N];
long long mst=0,ret[N];
int get(int u,int v,int w){
	int maxi = 0;
	while( (par[u]!=u || par[v]!=v) && u!=v){
		if( par[u]==u || (par[v]!=v&& sz[v]<=sz[u]) )
			maxi = max(maxi,con[v]),v= par[v];
		else
			maxi= max(maxi,con[u]),u= par[u];
	}
	if(u==v)return w-maxi;
	if(sz[u]<sz[v])swap(u,v);
	sz[u]+=sz[v],par[v]=u;con[v]=w;mst+=w;
	return 0;
}
int main(){
	int u,v,w,n,m,i,j;cin>>n>>m;
	for(i=0;i<m;++i){
		scanf("%d%d%d",&u,&v,&w);
		e[i].u = u,e[i].v=v,e[i].w=w;
		e[i].id=i;
	}
	sort(e,e+m);
	for(i=1;i<=n;++i)
		par[i]=i,sz[i]=1;
	for(i=0;i<m;++i){
		add[e[i].id]= get(e[i].u,e[i].v,e[i].w);
	}
	for(i=0;i<m;++i)
		cout<<mst+add[i]<<endl;
}
