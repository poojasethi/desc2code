#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
const int n=200000;
class sp{
  public:
  int q,e;
  sp *w;
  sp(int q1,sp *w1,int e1){
   q=q1;w=w1;e=e1;
  }
};
int numrb,rb[n],rb1[n],rb2[n],rb3[n];
void qs(int q,int w){
  int e=q,r=w,t=rb[q+(rand()%(w-q+1))],y;
  do{
   while(rb[e]<t)e++;
   while(rb[r]>t)r--;
   if(e<=r){
    y=rb[e];rb[e]=rb[r];rb[r]=y;
    y=rb1[e];rb1[e]=rb1[r];rb1[r]=y;
    y=rb2[e];rb2[e]=rb2[r];rb2[r]=y;
    y=rb3[e];rb3[e]=rb3[r];rb3[r]=y;
    e++;r--;
   }
  }while(e<=r);
  if(q<r)qs(q,r);
  if(e<w)qs(e,w);
}
sp *a[n];
int i=0,s[n],s1[n],f[n][18],f1[n][18],g[n];
int k(int w){
  if(g[w]!=w)
   g[w]=k(g[w]);
  return g[w];
}
void ini(int q,int w){
  sp *c;
  s[q]=i;
  i++;
  int e,r=w;
  for(c=a[q];c;c=c->w)
   if(c->q==w)
    f1[q][0]=c->e;
  f[q][0]=r;
  for(e=1;e<18;e++){
   if(r<0){
    f[q][e]=-1;
    continue;
   }
   f1[q][e]=max(f1[q][e-1],f1[r][e-1]);
   r=f[r][e-1];
   f[q][e]=r;
  }
  for(c=a[q];c;c=c->w)
   if(c->q!=w)
    ini(c->q,q);
  s1[q]=i;
}
int sch(int e,int r){
  r=s[r];
  int c,v,ans=0;
  for(c=17;c>=0;c--)
   if(f[e][c]>=0){
    v=f[e][c];
    if(!((s[v]<=r)&&(r<s1[v]))){
     ans=max(ans,f1[e][c]);
     e=v;
    }
   }
  if(!((s[e]<=r)&&(r<s1[e])))
   ans=max(ans,f1[e][0]);
  return ans;
}
long long h[n];
int main(){
  #ifdef Vlad_kv
   freopen("input.txt","r",stdin);
   freopen("output.txt","w",stdout);
  #endif // Vlad_kv
  int q,w,e,r,t;
  long long subans=0;
  scanf("%d%d",&q,&numrb);
  for(w=0;w<q;w++){
   a[w]=0;
   g[w]=w;
  }
  for(w=0;w<numrb;w++){
   scanf("%d%d%d",&rb1[w],&rb2[w],&rb[w]);
   rb1[w]--;
   rb2[w]--;
   rb3[w]=w;
  }
  if(numrb>0)
   qs(0,numrb-1);
  for(w=0;w<numrb;w++){
   if(k(rb1[w])!=k(rb2[w])){
    g[k(rb1[w])]=k(rb2[w]);
    a[rb1[w]]=new sp(rb2[w],a[rb1[w]],rb[w]);
    a[rb2[w]]=new sp(rb1[w],a[rb2[w]],rb[w]);
    subans+=rb[w];
   }
  }
  ini(0,-1);
  for(w=0;w<numrb;w++){
   e=rb1[w];
   r=rb2[w];
   h[rb3[w]]=subans+rb[w]-max(sch(e,r),sch(r,e));
  }
  for(w=0;w<numrb;w++)
   printf("%I64d\n",h[w]);
  return 0;
}
