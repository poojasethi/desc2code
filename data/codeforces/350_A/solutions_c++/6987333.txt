#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,amax=0,amin=101,bmin=101,num;
	cin>>n>>m;
	while(n--)
	{
		cin>>num;
		amin=min(amin,num);
		amax=max(amax,num);
	}
	while(m--)
	{
		cin>>num;
		bmin=min(bmin,num);
	}
	if(max(2*amin,amax) < bmin)
		cout<<max(2*amin,amax)<<endl;
	else
		cout<<"-1\n";
	return 0;
}
