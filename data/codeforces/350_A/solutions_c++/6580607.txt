#include<iostream>
#include<set>
using namespace std;
int main()
{ ios_base::sync_with_stdio(0);
 int n,m,x;  cin>>n>>m;
 set<int>c,w;
 for(int i=0;i<n;i++)
 { cin>>x; c.insert(x);}
 for(int i=0;i<m;i++)
 { cin>>x; w.insert(x);}
 int mx=max((*c.begin()*2),*c.rbegin());
 if(mx<*w.begin())cout<<mx;
 else cout<<-1;
 return 0;
}
