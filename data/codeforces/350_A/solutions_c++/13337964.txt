#include<bits/stdc++.h>
using namespace std;

int main()
{ int n,m,i,a[100],b[100],v;
 cin>>n>>m;
 for(i=0;i<n;i++)
 cin>>a[i];
 for(i=0;i<m;i++)
 cin>>b[i];
 sort(a,a+n);
 sort(b,b+m);
 v=b[0]-1;
 while(v>a[n-1]&&v>2*a[0])
 {
  v--;
 }
 if(2*a[0]<=v&&v>=a[n-1])
 cout<<v<<endl;
 else 
 cout<<"-1"<<endl;
 
 return 0;
}
