#include <iostream>
#include <algorithm>

using namespace std;

int main()
{int n,m;
cin>>n>>m;
int a[n],b[m];
for(int i=0;i<n;i++)
    cin>>a[i];
sort(a,a+n);
for(int i=0;i<m;i++)
    cin>>b[i];
sort(b,b+m);

if(max(a[0]*2,a[n-1])<b[0])
    cout<<max(a[0]*2,a[n-1]);
else
    cout<<-1;
    return 0;
}
