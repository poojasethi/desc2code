#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	int a[n],b[m];
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<m;i++)cin>>b[i];
	sort(a,a+n);
	sort(b,b+m);
	if(a[0]*2<b[0]&&a[n-1]<b[0])
	{
		cout<<max(a[0]*2,a[n-1])<<endl;
		return 0;
	}
	else cout<<-1<<endl;
	return 0;
}
