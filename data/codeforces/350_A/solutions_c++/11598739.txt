#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	int s1[n],s2[m],a,i;
	for(i=0;i<n;i++)
	cin>>s1[i];
	for(i=0;i<m;i++)
	cin>>s2[i];

	sort(s1,s1+n);
	sort(s2,s2+m);
	int maxi=max(2*s1[0],s1[n-1]);
	if(maxi<s2[0])
	a=maxi;
	else
	a=-1;
	cout<<a;
	return 0;
}
