#include<bits/stdc++.h>
using namespace std;
long long a[1000005],b[1000005],i,j,m,n;
int main()
{
    cin>>n>>m;
    for(i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    for(i=0;i<m;i++)
        cin>>b[i];
    sort(b,b+m);
    j=max(2*a[0],a[n-1]);
    if(j>=b[0])
        cout<<"-1";
    else
        cout<<j;
}
