#include <algorithm>
#include <cstdio>
using namespace std;

int a, b, t, A[100], B[100];

int main() {
	scanf("%d %d", &a, &b);
	for (int i = 0; i < a; i++) scanf("%d", A + i);
	for (int i = 0; i < b; i++) scanf("%d", B + i);
	sort(A, A + a);
	sort(B, B + b);
	t = max(2 * A[0], A[a - 1]);
	if (t >= B[0]) t = -1;
	printf("%d\n", t);
	return 0;
}
