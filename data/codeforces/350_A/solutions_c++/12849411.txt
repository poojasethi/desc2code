#include <iostream>
#include<algorithm>
using namespace std;
int main() {
    int n,m;
    int a[100],b[100];
    cin>>n>>m;
    for (int i=0;i<n;i++) cin>>a[i];
    for (int i=0;i<m;i++) cin>>b[i];
    sort(a,a+n);
    int v=max(2*a[0],a[n-1]);
    sort(b,b+m);
    if (v<b[0])cout<<v<<endl;
    else cout<<-1<<endl;
}
