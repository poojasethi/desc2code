#include <iostream>
using namespace std;
int main()
{
int n,m,v,i,x,l,h;
cin>>n>>m;
v=0;
h=0;
l=101;
for(i=1; i<=n; i++)
{
cin>>x;
if(x<l) l=x;
if(x>h) h=x;
}
if(h==l) v=2*h;
else if(h>=(2*l)) v=h;
else if(h<(2*l)) v=2*l;
else v=-1;
if(v!=-1)
for(i=1; i<=m; i++)
{
cin>>x;
if(x<=v) v=-1;
}
cout<<v;
return 0;
}
