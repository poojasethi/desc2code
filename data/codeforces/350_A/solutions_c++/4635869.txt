#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int a[1000],b[1000];
int v,n,m;
int main()
{
	cin>>n>>m;
	for (int i=1;i<=n;++i)
		scanf("%d",&a[i]);
	for (int i=1;i<=m;++i)
		scanf("%d",&b[i]);
	sort(a+1,a+n+1);
	sort(b+1,b+m+1);
	v=max(a[1]*2,a[n]);
	if (v<b[1])
		cout<<v<<endl;
	else cout<<-1<<endl;
	return 0;
}
