#include<iostream>
#include<algorithm>
using namespace std;
main()
{
    long long i,j,k,n,m,t;
    cin>>n>>m;
    long long a[n+10],b[m+10];
    for(i=0;i<n;i++)cin>>a[i];
    for(i=0;i<m;i++)cin>>b[i];
    sort(a,a+n);
    sort(b,b+m);
    t=max(a[n-1],2*a[0]);
    if(t<b[0]) {cout<<t; return 0;} else {cout<<-1; return 0;}
}
