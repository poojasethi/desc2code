#include<cstdio>
#include<algorithm>

using namespace std;

int main()
{
	int n,m,A[100],B[100];
	scanf("%d %d",&n,&m);
	int i;
	for(i=0;i<n;i++)
		scanf("%d",&A[i]);
	for(i=0;i<m;i++)
		scanf("%d",&B[i]);
	sort(A,A+n);
	int v=max(2*A[0],A[n-1]);
	sort(B,B+m);
	if(B[0]<=v)
		printf("-1\n");
	else
		printf("%d\n",v);
	return 0;
}
