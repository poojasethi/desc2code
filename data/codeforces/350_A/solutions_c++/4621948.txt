#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
   int n, m; cin>>n>>m;
   int a[n], b[m];
   for(int i = 0; i<n; ++i) cin>>a[i];
   for(int i = 0; i<m; ++i) cin>>b[i];
   sort(a, a+n); sort(b, b+m);

   int lo = max(2*a[0], a[n-1]);
   int hi = b[0];
   if(lo >= hi) cout<<"-1\n";
   else cout<<lo<<endl;
   return 0;
}
