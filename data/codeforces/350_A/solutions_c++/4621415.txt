#include <bits/stdc++.h>

using namespace std;

int n,m;

int c[105], w[105];

int main () {

	cin>>n>>m;
	for (int i = 0; i < n; ++i) {
		cin>>c[i];
	}
	for (int i = 0; i < m ; ++i) {
		cin>>w[i];
	}
	sort(c,c+n);
	sort(w,w+m);
	int m = 2*c[0];
	m = max(m,c[n-1]);
	if (m >= w[0]) {
		m = -1;
	}
	cout<<m<<endl;
	
}
