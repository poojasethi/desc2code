#include<iostream>
#include<algorithm>
//#include<math.h>
using namespace std;
int main(){
	int n,m;
	cin>>n>>m;
	int c[n],w[m];
	for(int i=0;i<n;i++)
		cin>>c[i];
	for(int i=0;i<m;i++)
		cin>>w[i];
	int v=max(2*(*min_element(c,c+n)),*max_element(c,c+n));
	if(v< *min_element(w,w+m) )
		cout<<v<<endl;
	else
		cout<<-1<<endl;
	return 0;
}
