# include <iostream>

using namespace std;

int n,m,k,x,y=109,z=109;

int main ()
{
	cin >> n >> m;
	
	for ( int i = 0; i < n; i++ )
		cin >> k, x = max ( x, k ), y = min ( y, k );
	
	for ( int i = 0; i < m; i++ )
		cin >> k, z = min ( z, k );

	x = max ( x, y*2 );
	
	cout << ( x < z ? x : -1 );
	
return 0;	
}
