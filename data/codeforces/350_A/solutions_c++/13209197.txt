#include <bits/stdc++.h>
using namespace std;
int main()
{
	int i,n,m;
	scanf("%d%d",&n,&m);
	int a[n],b[m];
	for(i=0;i<n;i++) scanf("%d",&a[i]);
	for(i=0;i<m;i++) scanf("%d",&b[i]);
	sort(a,a+n);sort(b,b+m);
	int maxx = max(2*a[0],a[n-1]);
	if( b[0] <= maxx ) printf("-1\n");
	else printf("%d\n",maxx);
	return 0;
}
