#include <iostream>
#include <algorithm>
using namespace std;

int n,m,a[101],b[101];

int main()
{
    int v;
    cin>>n>>m;
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int i=0;i<m;i++)
        cin>>b[i];
    sort(a,a+n),sort(b,b+m);
    v=max(a[0]*2,a[n-1]);
    cout<<(b[0]<=v?-1:v)<<endl;
    return 0;
}
