#include<bits/stdc++.h>
using namespace std;
int main(){
int m,n;
cin>>m>>n;
int a[101],b[101];
for(int i=0;i<m;i++)cin>>a[i];
for(int j=0;j<n;j++)cin>>b[j];
sort(a,a+m);
sort(b,b+n);

int as=a[m-1];
int we=a[0]*2;
int sum=max(as,we);
if(sum>=b[0])cout<<"-1"<<endl;
else cout<<sum<<endl;

return 0;
}
