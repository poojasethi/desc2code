#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
  int n,m,x;int a[105],b[105];
  cin>>n>>m;
  for(int i=0;i<n;i++){
    cin>>a[i];
  }
  for(int i=0;i<m;i++){
    cin>>b[i];
  }
  sort(a,a+n);
  sort(b,b+m);
  x=max(a[0]*2,a[n-1]);
  if(b[0]<=x)
     cout<<-1;
  else  cout<<x;
  return 0;
}
