#include <bits/stdc++.h>

using namespace std;

int n,m;
int A[105];
int B[105];

int main()
{
	int i,j,mx;
	cin >> n >> m;
	for(i=1 ; i<=n ; i++)
		cin >> A[i];
	for(j=1 ; j<=m ; j++)
		cin >> B[j];
	sort(A+1,A+1+n);
	sort(B+1,B+1+m);
	mx=max(A[1]*2,A[n]);
	if(mx>=B[1]) cout << -1 << endl ;
	else cout << mx << endl ;
	return 0;
}
