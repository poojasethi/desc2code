#include <algorithm>
#include <iostream>
using namespace std;
int a[100],b[100];
int main(){
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<m;i++)cin>>b[i];
	sort(a,a+n);
	sort(b,b+m);
	for(int i=a[n-1];i<b[0];i++){
		if(i>=2*a[0]){
			cout<<i<<endl;
			return 0;
		}
	}
	cout<<"-1\n";
	return 0;
}
