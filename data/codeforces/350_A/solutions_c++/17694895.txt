#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,m,a[105],b[105];
    cin>>n>>m;
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int j=0;j<m;j++)
        cin>>b[j];
    sort(a,a+n);
    sort(b,b+m);

    int t=max(2*a[0],a[n-1]);
    if(t>=b[0])cout<<"-1"<<endl;
    else cout<<t<<endl;
}
