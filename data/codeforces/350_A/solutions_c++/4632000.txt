#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,m,v=0,i,a[200],b[200];
    cin>>n>>m;
    for(i=0;i<n;i++)
    cin>>a[i];
    for(i=0;i<m;i++)
    cin>>b[i];
    sort(a,a+n);
    sort(b,b+m);
    v=max(a[0]*2,a[n-1]);
    if(v>=b[0])cout<<-1;
    else cout<<v<<endl;
}
