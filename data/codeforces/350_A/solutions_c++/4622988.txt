#include <iostream>
#include <algorithm>
using namespace std;

int m, n, m1=101, m2=0, i, j;

int main() {
	cin >> n >> m;
	for (i=0; i<n; i++) {
		cin >> j;
		m1=min(m1,j);
		m2=max(m2,j);
	}
	m1=max(m1*2,m2);
	m2=101;
	for (i=0; i<m; i++){
		cin >> j;
		m2=min(m2,j);
	}
	if (m1<m2) cout << m1;
	else cout << -1;
	return 0;
}
