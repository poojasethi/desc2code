#include <iostream>;
#include <algorithm>;

using namespace std;

int main(){
    int n,m,a[100],b[100];
    cin>>n>>m;
    for (int i=0;i<n;i++) cin>>a[i];
    for (int i=0;i<m;i++) cin>>b[i];
    sort(a,a+n);
    sort(b,b+m);
    int x;
    x = max(a[n-1],a[0]*2);
    if (x<b[0]) cout<<x; else cout<<"-1";

    return 0;
}
