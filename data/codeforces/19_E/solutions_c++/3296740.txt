#include<cstdio>
#include<algorithm>

using namespace std;
const int maxn=20005;
int next[maxn],e[maxn],head[maxn],tot=1;
int f[maxn],g[maxn],vis[maxn],d[maxn];
int ct,n,m;
void add(int u,int v){
	next[++tot]=head[u];
	head[u]=tot;
	e[tot]=v;
}
void dfs(int u,int v){
	vis[u]=1;
	for(int i=head[u];i;i=next[i]){
		int w=e[i];
		if(w==v || d[w]>d[u]) continue;
		if(!vis[w]){
			d[w]=d[u]+1; dfs(w,u);
			f[u]+=f[w]; g[i]=f[w];
		}else if((d[u]^d[w])&1) f[u]--,f[w]++;
			else{
				f[u]++,f[w]--;
				g[i]=1,ct++;
			}
	}
}
int main() {
#ifndef ONLINE_JUDGE
	freopen("1.in","r",stdin);
	freopen("1.out","w",stdout);
#endif
	scanf("%d%d",&n,&m);
	for(int i=1,x,y;i<=m;i++){
		scanf("%d%d",&x,&y);
		add(y,x),add(x,y);
	}
	for(int i=1;i<=n;i++) 
		if(!vis[i]) dfs(i,0);
	if(!ct){
		printf("%d\n",m);
		for(int i=1;i<=m;i++)
			printf("%d ",i);
	}else{
		tot=0;
		for(int i=2;i<=m+m+1;i++) 
			if(g[i]==ct) f[++tot]=i/2;
		printf("%d\n",tot);
		for(int i=1;i<=tot;i++) 
			printf("%d ",f[i]);
	}
	return 0;
}
