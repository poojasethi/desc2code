#include <algorithm>
#define re(i,A) for(int i=0;i<(int)(A);i++)
#define de(i,A) for(int i=(A)-1;~i;i--)
#define inf (1e9)
#define mn 1011
#define P 1000000007
#define eps 1e-9
#define si(x) ((int)x.size())
#define MAX(A,B) A=max(A,B)
#define MIN(A,B) A=min(A,B)
#define maxn 200005
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
typedef pair<int,int> pii;

/*****************************************************************************************************************************************************************************************************************************************************************************************************************************************************************************************************************************************************************/
int tot,N,L,R,U,V,M,n,head[maxn],lk[maxn],next[maxn],bg[maxn],w[maxn],q[maxn],d[maxn],fa[maxn],e[maxn],s[maxn],f[maxn],ff[maxn],g[maxn],gg[maxn],h[maxn],b[maxn];

void add(int u,int v,int x,int t) {lk[t]=v,w[t]=x,next[t]=head[u],head[u]=t;}

void init()
{
	scanf("%d%d%d",&N,&L,&R);
	for (int i=1,u,v; i<N; i++) scanf("%d%d%d",&u,&v,&bg[i]),add(u,v,bg[i],i),add(v,u,bg[i],i+N-1);
}

int bfs(int p)
{
	int l=1,r=1;
	for (q[1]=p,fa[p]=0; l<=r; l++)
		for (int u=q[l],i=head[u],v; v=lk[i],i; i=next[i]) if (b[v]!=tot&&v!=fa[u]) fa[v]=u,q[++r]=v,d[v]=d[u]+1,e[v]=e[u]+(w[i]<bg[M]?-1:1);
	return r;
}

int find(int p)
{
	int mi=1e9,r=bfs(p),o;
	for (int x=r,u; u=q[x],x; x--)
	{
		int ma=0; s[u]=1;
		for (int i=head[u],v; v=lk[i],i; i=next[i]) if (b[v]!=tot&&v!=fa[u]) s[u]+=s[v],ma=max(ma,s[v]);
		ma=max(ma,r-s[u]);
		if (ma<mi) mi=ma,o=u;
	}
	return o;
}

bool work(int u)
{
	int s=bfs(u);
	for (int i=1; i<=s; i++) if (e[q[i]]>ff[d[q[i]]]) ff[d[q[i]]]=e[q[i]],gg[d[q[i]]]=q[i];
	bool ok=0;
	int l=1,r=0;
	for (int i=s,j=0,ll,rr; ll=L-i,rr=R-i,!ok&&i&&ll<=n; i--)
	{
		for (;j<ll; j++);
		for (;j<=rr&&j<=n; j++)
		{
			for (;l<=r&&f[j]>f[h[r]]; r--);
			h[++r]=j;
		}
		for (;l<=r&&h[l]<ll; l++);
		if (l<=r&&f[h[l]]+ff[i]>=0) U=g[h[l]],V=gg[i],ok=1;
	}
	n=max(n,s);
	for (int i=1; i<=s; i++)
	{
		if (ff[i]>f[i]) f[i]=ff[i],g[i]=gg[i];
		ff[i]=-1e9;
	}
	return ok;
}

bool dfs(int u)
{
	int p=find(u),s;
	d[p]=e[p]=0,s=bfs(p),b[p]=tot,n=0,f[0]=0,g[0]=p;
	for (int i=1; i<=s; i++) f[i]=ff[i]=-1e9;
	for (int i=head[p],v; v=lk[i],i; i=next[i]) if (b[v]!=tot&&work(v)) return 1;
	for (int i=head[p],v; v=lk[i],i; i=next[i]) if (b[v]!=tot&&dfs(v)) return 1;
	return 0;
}

void doit()
{
	int l=1,r=N-1;
	sort(bg+1,bg+N);
	for (;l<=r;)
	{
		M=(l+r)>>1,tot++;
		if (dfs(1)) l=M+1; else r=M-1;
	}
	M=r,tot++,dfs(1),printf("%d %d\n",U,V);
}

int main()
{
	init();
	doit();
	return 0;
}
