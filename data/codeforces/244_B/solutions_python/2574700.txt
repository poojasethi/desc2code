n = int(raw_input())
ans = {}

def dfs(k, a, b):
        if k <= n:
                ans[k] = 1
                if k != k * 10 + a:
                        dfs(k * 10 + a, a, b)
                if k != k * 10 + b:
                        dfs(k * 10 + b, a, b)

for i in range(10):
        for j in range(i + 1):
                dfs(0, i, j)

print len(ans) - 1
