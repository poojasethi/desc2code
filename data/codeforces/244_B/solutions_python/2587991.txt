if __name__ == '__main__':
	vn = [int(_) for _ in list(raw_input())]
	n = len(vn)
	s = 81 * (1 << (n - 1)) - 72 * n - 9
	for i in range(n):
		for r in range(vn[i]) if i > 0 else range(1, vn[0]):
			k = len(set(vn[:i] + [r]))
			if k == 1:
				s += 9 * (1 << (n - 1 - i)) - 8
			elif k == 2:
				s += 1 << (n - 1 - i)
	if len(set(vn)) <= 2:
		s += 1
	print s
