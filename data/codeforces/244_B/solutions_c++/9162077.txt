#include <cstdio>
#include <set>
#include<iostream>
using namespace std;
set<int>lis;
int n ,i ,j;
void dfs(long long x, int y)
{
    if(x>n||y>10)
        return;
    lis.insert(x);
    dfs(x*10+j,y++);
    dfs(x*10+i,y++);
}
int main()
{
    cin>>n;
    for(i=0;i<10;i++)
    for(j=i+1;j<10;j++)
        dfs(0,0);
    cout<<lis.size()-1<<endl;
}
