#include<stdio.h>
#include<map>
long long int ans;
int n,a,b;
std::map<int,int> S;
void DFS(long long int value,int length){
	if( value>n||length>10 )	return;
	S[value]++;
	DFS(value*10+a,length+1);
	DFS(value*10+b,length+1);
}
int main(){
	while( scanf("%d",&n)==1 ){
		ans=0;
		for(a=0;a<10;a++)
			for(b=a;b<10;b++)
				DFS(0,0);
		printf("%lld\n",S.size()-1); // delete 0
	}
	return 0;
}
