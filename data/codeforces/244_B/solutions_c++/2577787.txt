#include <iostream>
#include <set>

using namespace std;

set<long long> s;
int n,a,b;

void gen(long long x, int st){
    if (x>n || st>10) return;
    s.insert(x);
    gen(10*x+a,st+1);
    gen(10*x+b,st+1);
}

int main(){
    cin >> n;
    for (a=0; a<=9; a++){
        for (b=a+1; b<=9; b++) gen(0,0);
    }
    cout << s.size() - 1 << endl;;
    return 0;
}
