#include<iostream>
#include<algorithm>
#include<set>

using namespace std;

int a,b,n;

set<__int64> S;

void dfs(__int64 x,int len)
{
    if(x>n||len>10) return ;
    S.insert(x);
    dfs(x*10+a,len+1);
    dfs(x*10+b,len+1);
}

int main()
{
    while(cin>>n)
    {
        for(a=0;a<=9;a++)
         for(b=0;b<=9;b++)
          dfs(0,0);
    cout<<S.size()-1<<endl;
    }
    return 0;
}
