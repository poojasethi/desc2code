#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<set>

using namespace std;

int n,l;
set<long long> ans;

void work(int x, int y,long long z,int l)
{
	if (z>n) return;
	if (l>10) return;
	ans.insert(z);
	work(x,y,z*10+x,l+1);
	work(x,y,z*10+y,l+1);
}

int main()
{
	cin>>n;
	int i,j;
	for (i=0;i<=9;i++)
		for (j=i+1;j<=9;j++)
			work(i,j,0,0);
	cout<<ans.size()-1<<"\n";
}
