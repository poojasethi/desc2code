#include<iostream>
#include<set>
using namespace std;

int n,x,y;
set<int> s;


void dfs(long long a,int b){
   if(a>n||b>10)
      return ;
   s.insert(a);
   dfs(10*a+x,b+1);
   dfs(10*a+y,b+1);
}

int main(){
   cin>>n;
   for(x=0;x<9;x++)
      for(y=x+1;y<=9;y++)
         dfs(0,0);
   cout<<s.size()-1;
   cin.get();
   cin.get();
   return 0;
}
