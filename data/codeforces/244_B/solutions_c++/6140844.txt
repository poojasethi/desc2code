#include <iostream>
using namespace std;

int ans=0,n;

void dfs(int x){
	if(x>n)return;
	int a=-1,b=-1,c=x;
	while(c){
		if(a==-1)a=c%10;
		else if(a!=c%10){
			if(b==-1)b=c%10;
			else if(b!=c%10)
				return;
		}
		c/=10;
	}
	ans++;
	for(int i=0;i<=9;i++){
		if(x==0&&i==0)continue;
		dfs(x*10+i);
	}
}

int main(){
	cin>>n;
	dfs(0);
	cout<<ans-1;
}
