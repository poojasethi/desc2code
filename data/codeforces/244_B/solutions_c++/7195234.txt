#include <iostream>
#include <set>

using namespace std;

typedef long long int ll;

int main()
{
  ll n;
  cin>>n;
  set<ll> s;
  for (int c=0;c<(1<<11);c++) {
    for (int d1=0;d1<=9;d1++) {
      for (int d2=0;d2<=9;d2++) {
	ll x=0;
	int cc=c;
	while (cc>0) {
	  ll b;
	  if (cc%2) b=d1;
	  else b=d2;
	  cc/=2;
	  x=x*10+b;
	}
	if (x>0 and x<=n)
	  s.insert(x);
      }
    }
  }
  cout<<int(s.size())<<endl;
}
