#include <iostream>
#include <cstring>
#include <cstdio>
#include <set>
using namespace std;
set<long long>v;
int n,a,b;

void dfs(long long x,int y){
	if(x>n||y>10)
		return ;
	v.insert(x);
	dfs(x*10+a,y+1);
	dfs(x*10+b,y+1);
}
int main(){
	while(cin>>n){
		v.clear();
		for(a=0;a<=9;a++)
			for(b=a+1;b<=9;b++)
				dfs(0,0);
		cout<<v.size()-1<<endl;
	}
	return 0;
}
