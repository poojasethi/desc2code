#include <iostream>
#include <set>
using namespace std;
int a,b,n;
set<long long> s;
void gen(long long x,int st)
{
	if (x>n||st>10) return;
	s.insert(x);
	gen(x*10+a,st+1);
	gen(x*10+b,st+1);
}
int main()
{
	cin>>n;
	for (a=0;a<=9;a++)
	{
		for (b=a+1;b<=9;b++) gen(0,0);
	}
	cout<<s.size()-1;
	return 0;
}
