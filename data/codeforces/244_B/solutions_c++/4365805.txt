#include<stdio.h>
#include<set>

using namespace std;
set<long long> S;
int n;
void f(long long int v, int x, int y)
{
    if( v > n || !v ) return;
    S.insert(v);
    f( v*10+x, x, y );
    f( v*10+y, x, y );
}
int main()
{
    int i ,j;
    scanf("%d", &n);
    for( i = 0 ; i < 9 ; i++ ) for( j = i+1 ; j < 10 ; j++ )
        f( (long long)i, i, j ), f( j, i, j);
    printf("%ld", S.size());
}
