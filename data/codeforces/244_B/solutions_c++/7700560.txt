#include <iostream>
#include <cstdio>
#include <set>
using namespace std;

set<long long>v;
int n,a,b;

void gen(long long x,int st)
{
	if (x>n||st>10)
	{
		return;
	}
	
	v.insert(x);
	
	gen(x*10+a,st+1);
	gen(x*10+b,st+1);
	
}

int main()
{
	scanf("%d",&n);
	
	for (a=0;a<=9;a++)
	{
		for (b=a+1;b<=9;b++)
		{
			gen(0,0);
		}
	}
	
	printf("%d",v.size()-1);
	
	return 0;
	
}
