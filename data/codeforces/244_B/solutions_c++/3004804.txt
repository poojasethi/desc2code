#include <iostream>
#include <set>
using namespace std;

set <long long> s;
long long n, i, j;

void dfs(long long x, long long y){
    if (x>n || y>10)
        return;
    s.insert(x);
    dfs(x*10+i, y+1);
    dfs(x*10+j, y+1);
}

int main(){
    cin>>n;
    for (i=0; i<=9; i++)
        for (j=i+1; j<=9; j++)
            dfs(0, 0);
    cout<<s.size()-1<<endl;
    return 0;
}
