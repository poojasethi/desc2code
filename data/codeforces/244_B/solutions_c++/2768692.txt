#include <iostream>
#include <set>
using namespace std;
set <long long> s;
int a, b;
long long n;

void def(long long x = 0, int y = 0){
    if (x > n || y > 10) return;
    s.insert(x);
    def(x*10+a, y+1);
    def(x*10+b, y+1);
}

int main()
{
    cin >> n;
    for (a = 0; a < 10; a++)
        for (b = 0; b < 10; b++) def();
    cout << s.size()-1 << endl;
    return 0;
}
