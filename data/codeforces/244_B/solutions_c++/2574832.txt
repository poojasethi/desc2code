#include<iostream>
#include<set>

using namespace std;
set<int> ans;
int n;
void dfs(int x,int y,long long cur,int lim){
  if(cur > n)return;
  if(lim > 10)return;
  ans.insert(cur);
  dfs(x,y,cur*10+y,lim+1);
  dfs(x,y,cur*10+x,lim+1);

}

int main(){
  cin >> n;

  for(int i=0;i<=9;i++){
    for(int j=i+1;j<=9;j++){
      dfs(i,j,0,0);
    }
  }

  cout << ans.size()-1 << endl;


  return 0;
}
