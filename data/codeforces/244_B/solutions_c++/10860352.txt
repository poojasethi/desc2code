#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int n;
set<ll>v;
int i,j;

void dfs(ll x,int y) {
  if(x>n||y>10) {
    return;
  }
  v.insert(x);
  dfs(x*10+i,y+1);
  dfs(x*10+j,y+1);
  return;
}
int main() {
  scanf("%d",&n);
  v.clear();
  for(i=0;i<=9;i++) {
    for(j=i+1;j<=9;j++) {
      dfs(0,0);
    }
  }
  printf("%d\012",v.size()-1);
  return 0;
}
