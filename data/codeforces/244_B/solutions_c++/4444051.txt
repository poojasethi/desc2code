#include<iostream>
#include<set>
using namespace std;
long n;
set<long long> se;
void sinhso(long x,long y, long long s){
    if(s > n) return;
    se.insert(s);
    sinhso(x,y,s*10 + x);
    sinhso(x,y,s*10 + y);
}
int main(){
    cin>>n;
    for(int i=1;i<10;i++){
        for(int j=i+1;j<10;j++)
        {
            sinhso(i,j,0);
        }
        sinhso(i,0,i);
    }
    cout<<se.size() - 1;
}
