#include<iostream>
#include<set>
#include<algorithm>
using namespace std;
set<long long> s;
int n,b,a;
void count(long long x,long w)
{
	if(x>n || w>10) return;
	s.insert(x);
	count(10*x+a,w+1);
	count(10*x+b,w+1); 
}
int main()
{
	cin>>n;
	for(a=0;a<10;a++)
	{
		for(b=a+1;b<10;b++)
		{
			count(0,0);
		}
	}
	cout<<s.size()-1;
	return 0;
}
