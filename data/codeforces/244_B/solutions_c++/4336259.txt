#include<iostream>
#include<set>
using namespace std;
set<long long> ans;
int n;
int i,j;
void dfs(long long num,int cs)
{
	if(num>n || cs>10) return;
	ans.insert(num);
	dfs(num*10+i,cs+1);
	dfs(num*10+j,cs+1);
}
int main()
{
	cin>>n;
	for(i=0;i<=9;i++)
		for(j=0;j<=9;j++)
			dfs(0,0);
	cout<<ans.size()-1;
	return 0;
}
