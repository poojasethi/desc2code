#include <iostream>
#include <set>
using namespace std;
set<long long> s;
long long nmax;

void dfs(long long n,int x,int y)
{
    if(n>nmax)return;
    s.insert(n);
    if(10*n+x!=n)dfs(10*n+x,x,y);
    if(10*n+y!=n)dfs(10*n+y,x,y);
    return;
}

int main()
{
    cin>>nmax;
    for(int i=0;i<10;i++)for(int j=0;j<i;j++)dfs(0,i,j);
    cout<<s.size()-1;
    return 0;
}
