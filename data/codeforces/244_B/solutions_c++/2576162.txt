#include <stdio.h>
#include <set>

using namespace std;

set<long long> s;
int n,a,b;

void gen(long long x, int st){
    if (x>n || st>10) return;
    s.insert(x);
    gen(10*x+a,st+1);
    gen(10*x+b,st+1);
}

int main(){
    scanf(" %d",&n);
    for (a=0; a<=9; a++){
        for (b=a+1; b<=9; b++) gen(0,0);
    }
    printf("%d\n",s.size()-1);
    return 0;
}
