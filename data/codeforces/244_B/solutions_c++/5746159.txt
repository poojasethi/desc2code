#include <iostream>
#include <set>
using namespace std;
set<long long>countl;
int n,x,y;
void dfs(long long num,int t)
{
    if(num>n||t>10)return;
    countl.insert(num);
    dfs(10*num+x,t+1);
    dfs(10*num+y,t+1);
}
int main()
{
    cin>>n;
    for(x=0;x<10;x++)
        for(y=x+1;y<10;y++)
            dfs(0,0);
    cout<<countl.size()-1<<endl;
    return 0;
}
