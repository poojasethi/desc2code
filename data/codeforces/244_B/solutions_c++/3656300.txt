#include<cstdio>
#include<set>
using namespace std;

set<long long> a;
int n, x, y;

void dfs(long long num, int t)
{
	if (num > n || t > 10)
		return;
	a.insert(num);
	dfs(num * 10 + x, t + 1);
	dfs(num * 10 + y, t + 1);
}

int main()
{
	scanf("%d", &n);
	for (x = 0; x < 10; x++)
		for (y = x + 1; y < 10; y++)
			dfs(0, 0);
	printf("%d\n", a.size() - 1);
	return 0;
}
