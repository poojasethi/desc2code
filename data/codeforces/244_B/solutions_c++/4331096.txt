#include<iostream>
#include<algorithm>
#include<set>
using namespace std;
int n,a,b;
set<long long> s;
long long inf=100000000;
void dfs(long long x)
{
	if(x>0 && x<=n) s.insert(x);
	else return;
	if(x>inf) return;
	dfs(x*10+a);
	dfs(x*10+b);
	return;
}
int main()
{
	cin>>n;
	for(a=0;a<10;a++)
	{
		for(b=0;b<10;b++)
		{
			dfs(a);
			dfs(b);
		}
	}
	cout<<s.size();
	return 0;
}
