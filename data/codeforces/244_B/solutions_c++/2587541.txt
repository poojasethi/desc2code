#include<stdio.h>
#include<set>

using namespace std;

set<int> ans;
int n,i,j;
void dfs(long long sum)
{
	long long temp;
	temp=sum*10+i;
	if(temp<=n){
		ans.insert(temp);
		dfs(temp);
	}
	temp=sum*10+j;
	if(temp>0 && temp<=n){
		ans.insert(temp);
		dfs(temp);
	}
}
int main()
{
	scanf("%d",&n);
	for(i=0;i<10;i++)
		for(j=0;j<i;j++)
	            dfs(0);
	printf("%d\n",ans.size());
	return 0;
}
