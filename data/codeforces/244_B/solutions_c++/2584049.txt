using namespace std;

#include <iostream>
#include <set>

int a, b, n;
set <long long> ans;

void dfs (long long x, int y) {
	if (x > n || y > 10) return;
	ans.insert (x);
	dfs (x * 10 + a, ++y);
	dfs (x * 10 + b, y);
}

int main () {
	cin >> n;

	for (a = 0; a < 10; ++a)
		for (b = a + 1; b < 10; ++b)
			dfs (0, 0);

	cout << ans.size() - 1 << endl;
	return 0;
}
