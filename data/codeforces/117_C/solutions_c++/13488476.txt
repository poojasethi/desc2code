#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
using namespace std;
#define N 5005
int vis[N],n;
char mp[N][N];
int dfs(int u,int p){
    vis[u] = 1;
    for (int i=0;i<n;i++){
        if (mp[u][i]=='1'){
            if (mp[i][p]=='1'){
                printf("%d %d %d\n",p+1,u+1,i+1);
                return 1;
            }
            if (!vis[i]&&dfs(i,u)) return 1;
        }
    }
    return 0;
}

int main()
{
    cin>>n;
    for (int i=0;i<n;i++) scanf("%s",mp[i]);
    for (int i=0;i<n;i++)
        if (!vis[i]&&dfs(i,i)) return 0;
    cout<<-1<<endl;

    return 0;
}
