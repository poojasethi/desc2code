
#include <cstdio>
#include <cstring>

const int N = 5050;

int n;
char a[N][N],vst[N];

bool dfs(int u,int v)
{
    vst[u]=1;
    for(int i=1;i<=n;i++)   if(a[u][i]-'0')
    {
        if(a[i][v]-'0')
        {
            printf("%d %d %d\n",v,u,i);
            return 1;
        }
        if(!vst[i] && dfs(i,u)) return 1;
    }
    return 0;
}
int main(void)
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)   scanf("%s",a[i]+1);
    for(int i=1;i<=n;i++)   if(!vst[i] && dfs(i,i)) return 0;
    puts("-1");
    return 0;
}
