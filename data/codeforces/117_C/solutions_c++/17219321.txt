#include <bits/stdc++.h>
using namespace std;
#define N 5005
int cnt[N];
char a[N][N];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)scanf("%s",&a[i]);
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			if(a[i][j] == '1')
				cnt[i]++;
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			if( a[i][j] == '1' && cnt[j]>=cnt[i])
			{
				for(int k=0;k<n;k++)
				{
					if(k ==i || k==j)continue;
					if(a[j][k] == '1' && a[k][i] == '1')
					{
						cout<<i+1<<" "<<j+1<<" "<<k+1;
						return 0;
					}
				}
			}
	puts("-1");
	return 0;
}
