#include<bits/stdc++.h>
using namespace std;
const int N=5005;
int vis[N],n;
char s[N][N]; 
void dfs(int u,int pre)
{
   int i,j;
   vis[u]=1;
   for(i=0;i<n;i++)
        if(s[u][i]-'0')
        {
          if(!vis[i])dfs(i,u);
          else if(pre!=-1 && s[i][pre]-'0')
           {
             cout<<i+1<<" "<<pre+1<<" "<<u+1<<"\n";
             exit(0);
           }
        }
}
int main()
{
	ios::sync_with_stdio(false);cin.tie(0);
	int i,j;
    cin>>n;
    for(i=0;i<n;i++)cin>>s[i];
    for(i=0;i<n;i++)
    	if(!vis[i])
    		dfs(i,-1);
    cout<<"-1"<<'\n';
	return 0;
}
