#include <bits/stdc++.h>
using namespace std;
#define N 5005
int n;
char a[N][N];
bool used[N];
bool dfs(int cur , int prev)
{
	used[cur] = 1;
	for(int i=0;i<n;i++)
	{
		if(a[cur][i] == '1')
		{
			if(a[i][prev] == '1'){
				printf("%d %d %d\n",prev+1,cur+1,i+1);
				return 1;
			}
			if(!used[i] && dfs(i,cur))return 1;
		}
	}
	return 0;
}
int main()
{
		scanf("%d",&n);
		for(int i=0;i<n;i++)scanf("%s",a[i]);
		for(int i=0;i<n;i++)if(!used[i] && dfs(i,i))return 0;
		puts("-1");
		return 0;
}
