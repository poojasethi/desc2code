#include <cstdio>
#include <cstring>

const int N = 5005;
int n;
char g[N][N],vis[N];

bool dfs(int u,int pre)
{
	vis[u]=1;
	for(int i=1;i<=n;i++)	if(g[u][i]-'0')
	{
		if(g[i][pre]-'0')
		{
			printf("%d %d %d\n",pre,u,i);
			return 1;
		}
		if(!vis[i] && dfs(i,u))	return 1;
	}
	return 0;
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)	scanf("%s",g[i]+1);
	for(int i=1;i<=n;i++)	if(!vis[i] && dfs(i,i))	return 0;
	puts("-1");
	return 0;
}
