#include<cstdio>
#include<cstring>
#include<cstdlib>
using namespace std;
int n;
char s[5010][5010];
bool done[5010];

bool dfs(int a,int b)
{
	done[a]=1;
	for(int i=0;i<n;++i)if(s[a][i]=='1')
		if(!done[i])dfs(i,a);
		else if(s[i][b]=='1')
			exit((printf("%d %d %d\n",a+1,i+1,b+1),0));
}

int main()
{
	//freopen("input.txt","rb",stdin);
	//freopen("output.txt","wb",stdout);
	scanf("%d\n",&n);
	for(int i=0;i<n;++i)gets(s[i]);
	for(int i=0;i<n;++i)if(!done[i])dfs(i,-1);
	puts("-1");
	return 0;
}
