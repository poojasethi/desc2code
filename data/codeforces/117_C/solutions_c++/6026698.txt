#include <stdio.h>
#include <string.h>
#define N 5007
char s[N][N];
int p[N],c[N],t,n;
int dfs(int u)
{
	int v;
	c[u]=-1;
	for(v=0;v<n;v++)if(s[u][v]=='1')
	{
		p[v]=u;
		if(c[v]<0){t=v;return 0;}
		else if(!c[v] && !dfs(v))return 0;
	}
	c[u]=1;
	return 1;
}
int toposort()
{
	int u;
	memset(p,-1,sizeof(p));
	memset(c,0,sizeof(c));
	for(u=0;u<n;u++)if(!c[u])
		if(!dfs(u))return 0;
	return 1;
}
int main()
{
	int i,x;
	scanf("%d",&n);
	for(i=0;i<n;i++)scanf("%s",s[i]);
	if(toposort()){printf("-1\n");return 0;}
	x=p[t];
	printf("%d %d %d\n",t+1,p[x]+1,x+1);
	return 0;
}
