#include<bits/stdc++.h>
using namespace std ;
int n ;
char s [5002][5002] ;
bool vis[5001]= {false} ;

void dfs (int i ,int pre){
    vis[i] = true ;
    for(int x = 0 ; x < n ; x++){
        if(s[i][x]=='1'){
            if(s[x][pre]=='1'){
                cout << pre + 1 << " " << i + 1 << " " << x + 1 ;
                exit(0) ;
            }
            if(!vis[x])dfs(x,i) ;
        }
    }
}
int main() {
    cin >> n ;
    for(int i = 0 ;i < n ; i++){
        scanf("%s",&s[i]);
    }
    for(int i = 0 ;i < n; i++){
        if(!vis[i])dfs(i , -1) ;
    }
    cout << -1 ;
    return 0 ;
}
