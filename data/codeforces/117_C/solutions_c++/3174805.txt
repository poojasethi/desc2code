#include<stdio.h>
#include<string.h>
#define MAXN 5005
int N,vis[MAXN];
char mat[MAXN][MAXN];
int dfs(int x,int fa){
  int i;
  if(vis[x]) return 0;
  vis[x]=1;
  for(i=1;i<=N;i++)
    if(mat[x][i]){
      if(fa!=-1&&mat[i][fa]) {printf("%d %d %d\n",fa,x,i); return 1;}
      if(dfs(i,x)) return 1;
    }
  return 0;
}
int main(){
  int i,j;
  scanf("%d",&N);
  for(i=1;i<=N;i++) scanf("%s",mat[i]+1);
  for(i=1;i<=N;i++)
    for(j=1;j<=N;j++)
      mat[i][j]-='0';
  memset(vis,0,sizeof(vis));
  for(i=1;i<=N;i++) if(dfs(i,-1)) return 0;
  printf("-1\n");
  return 0;
}
