#include <cstdio>
#include <cstdlib>
#include <iostream>

using namespace std;

const int MAXN = 5005;

int N;
bool used[MAXN];

char ways[MAXN][MAXN];

void dfs( int nd, int dad )
{
	used[nd] = true;
	
	for( int i=1 ; i<=N ; i++ )
		if( ways[nd][i]-'0' ){
			
			if( ways[i][dad]-'0' ){
				printf("%d %d %d",dad,nd,i);
				exit(0);
			}
			
			if( !used[i] )	dfs( i, nd );
		}
}

int main()
{
	scanf(" %d",&N);
	
	for( int i=1 ; i<=N ; i++ )
		scanf(" %s",ways[i]+1);
	
	for( int i=1 ; i<=N ; i++ )
		if( !used[i] )
			dfs( i, i );
	
	puts("-1");
	
	return 0;
}
