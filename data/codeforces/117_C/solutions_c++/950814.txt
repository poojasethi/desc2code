#include <stdio.h>
#include <stdlib.h>

int n;
char g[ 5003 ][ 5003 ];
bool seen[ 5003 ];

void dfs(int u, int p){
    seen[u] = true;
    for (int v=0; v<n; v++) if ( g[u][v] == '1' ){
        if ( !seen[v] ) dfs( v, u );
        else if ( g[v][p]=='1' )
            exit( (printf("%d %d %d\n",v+1, p+1,u+1), 0) );
    }
}

int main(){
    
    scanf("%d\n",&n);
    for (int i=0; i<n; i++) gets( g[i] );
    for (int i=0; i<n; i++) if ( !seen[i] ) dfs( i, -1 );
    puts("-1");
        
    return 0;
}
