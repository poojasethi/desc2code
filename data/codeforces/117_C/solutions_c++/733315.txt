#include <cstdio>
#include <cstdlib>

char way[5001][5001];
int visit[5001];
int n;

int go(int u,int v)
{
    int i,j,k;
    visit[u] = 1;
    for(i=0;i<n;i++) {
        if(way[u][i]=='1' and visit[i]==0) go(i,u);
        else if(way[u][i]=='1' and way[i][v]=='1') exit(!printf("%d %d %d\n",u+1,i+1,v+1));
    }
    return 0;
}

int main()
{
    int i,j;
    scanf("%d",&n);
    for(i=0;i<n;i++) {
        scanf("%s",way[i]);
        visit[i] = 0;
    }
    for(i=0;i<n;i++) if(visit[i]==0) go(i,-1);
    printf("-1\n");
    return 0;
}
