#include <cstdio>
#include <cstdlib>
#include <iostream>

using namespace std;

const int MAXN=5005;

int N;

bool used[MAXN];

char way[MAXN][MAXN];

inline void dfs( int nd , int dad ){
	
	used[nd]=1;
	
	for( int i=1 ; i<=N ; i++ )
		if( way[nd][i]-'0' ){
			
			if( way[i][dad]-'0' ){
				printf("%d %d %d\n",dad,nd,i);
				exit(0);
			}
			
			if( !used[i] )
				dfs(i,nd);
				
		}
}

int main(){
	
	scanf(" %d",&N);
	
	for( int i=1 ; i<=N ; i++ )
		scanf(" %s",way[i]+1);
	
	for( int i=1 ; i<=N ; i++ )
		if( !used[i] )
			dfs(i,i);
	
	printf("-1\n");
	return 0;
}
