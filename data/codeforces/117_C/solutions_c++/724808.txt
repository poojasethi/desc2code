#include<cstdio>
#include<cstdlib>
#include<cstring>
char map[5010][5010];
bool vis[5010];
int n;
void dfs(int x,int pre)
{
    int i;
    vis[x]=true;
    for(i=1;i<=n;i++)
    if(map[x][i]=='1')
    {
        if(!vis[i])dfs(i,x);
        else if(pre!=-1&&map[i][pre]=='1')
        {
            printf("%d %d %d\n",i,pre,x);
            exit(0);
        }
    }
}
int main()
{
    int i,j;
    scanf("%d",&n);getchar();
    for(i=1;i<=n;i++)gets(map[i]+1);
    for(i=1;i<=n;i++)
    dfs(i,-1);
    puts("-1");
}
