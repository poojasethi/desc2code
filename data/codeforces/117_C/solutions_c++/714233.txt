#include<cstdio>

const int maxn=5000+10;
char g[maxn][maxn];
int d[maxn];
int A[maxn],B[maxn];
int n;

int main()
{
	scanf("%d",&n);
	for (int i=0;i<n;i++)
		scanf("%s",g[i]);
	for (int i=0;i<n;i++)
	for (int j=0;j<n;j++)
		d[j]+=(g[i][j]=='1');

	for (int i=0;i<n;i++)
	{
		int u=0,v=0;
		for (int j=0;j<n;j++)
		if (i!=j)
		{
			if (g[i][j]=='1') A[u++]=j;
			else B[v++]=j;
		}

		int tot=0;
		for (int j=0;j<u;j++) tot+=d[A[j]];
		if (tot!=u+u*(u-1)/2+v*u)
		{
			for (int a=0;a<u;a++)
			for (int b=0;b<v;b++)
			if (g[A[a]][B[b]]=='1')
			{
				printf("%d %d %d\n",i+1,A[a]+1,B[b]+1);
				return 0;
			}
		}
	}
	printf("-1\n");
}
