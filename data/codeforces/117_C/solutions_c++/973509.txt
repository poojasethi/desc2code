#include<cstdio>
#include<cstdlib>
using namespace std;
int n;
char t[5010][5010];
bool v[5010];
void dfs(int p,int i) {
    v[i]=true;
    for(int j=1;j<=n;j++) {
        if(t[i][j]=='1') {
            if(t[j][p]=='1') {
                printf("%d %d %d",i,j,p);
                exit(0);
            } else if(!v[j]) {
                dfs(i,j);
            }
        }
    }
}
int main() {
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%s",t[i]+1);
    for(int i=1;i<=n;i++)if(!v[i])dfs(0,i);
    puts("-1");
}
