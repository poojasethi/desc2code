#include<iostream>
#include<string.h>
#include<stdio.h>
#define MAX 5005
#define REP(i,a,b) for(int i=a;i<=b;i++)
using namespace std;
int n;
char m[MAX][MAX],vis[MAX];
bool bfs(int u,int pre){
    vis[u]=1;
    REP(i,1,n) if(m[u][i]=='1'){
        if(m[i][pre]-'0'){
            printf("%d %d %d\n",pre,u,i); 
            return 1;
        }
        if(!vis[i] && bfs(i,u)) return 1;
    }
    return 0;
}
int main(){
    scanf("%d",&n);
    REP(i,1,n)scanf("%s",m[i]+1);
    REP(i,1,n)if(!vis[i] && bfs(i,i)) return 0;
    puts("-1");

}
