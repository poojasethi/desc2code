#include <cstdio>
#define nmax 5010

char s[nmax][nmax], u[nmax];
int sol, n;

void df(int nod, int t)
{
	int i, v;
	u[nod]=1;
	for (v=1; v<=n && !sol; v++)
		if (s[nod][v-1]=='1')
		{
			if (t)
				if (s[v][t-1]=='1')
				{
					printf("%d %d %d\n",t,nod,v);
					sol=1;
					return;
				}
			if (!u[v])
				df(v, nod);
		}
}

int main()
{
	scanf("%d\n", &n);
	int i, j;
	for (i=1; i<=n; i++)
		fgets(s[i], nmax, stdin);
	for (i=1; i<=n && !sol; i++) 
		if (!u[i]) df(i, 0);
	if (!sol) printf("-1\n");
}
