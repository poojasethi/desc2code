#include <iostream>
#include <string.h>
#include <stdlib.h>
#include <stdio.h>
using namespace std;

int N;
char ar[5004][5004];
bool used[5004];

bool DFS(int node , int prev)
{
  used[node]=1;
  for(int i=0 ; i<N ; i++)
    if(ar[node][i]=='1')
    {
      if(!used[i])               DFS(i , node);
      else if(ar[i][prev]=='1')  exit((printf("%d %d %d\n",node+1,i+1,prev+1) , 0));
    }
}

int main()
{
  scanf("%d",&N);
  for(int i=0 ; i<N ; i++)
    scanf("%s",ar[i]);
  for(int i=0 ; i<N ; i++)
    if(!used[i])
      DFS(i , -1);
  puts("-1");
  return 0;
}
