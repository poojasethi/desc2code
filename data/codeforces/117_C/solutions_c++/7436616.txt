#include <cstdio>
#include <cstdlib>
int g[5005][5005]; bool v[5005]; int n;
void dfs(int i, int p){
    v[i] = true;
    for(int j = 0; j < n; j++) if(g[i][j]){  if(!v[j]) dfs(j,i);  else if(g[j][p]){ printf("%d %d %d\n",j+1,p+1,i+1); exit(0); }}
}
int main(){
    scanf("%d",&n);
    getchar();
    for(int i = 0; i < n; i++){ for(int j = 0; j < n; j++){   g[i][j]=getchar()-'0';}    getchar();}
    for(int i = 0; i < n; i++) if(!v[i]) dfs(i,-1);
    puts("-1");
}
