#include<stdio.h>
#include<iostream>
#include<vector>
#include<stack>
using namespace std;
int n;
char a[5005][5005];
int known[5005]={0};
int  dfs(int i,int prev)
{
	known[i]=1;
	int k;
	for(k=1;k<=n;k++)
	{
		if(a[i][k]-'0'==1)
		{
			if(a[k][prev]-'0'==1)
			{
				printf("%d %d %d\n",prev,i,k);
				return 1;
			}
			if(known[k]==0&&dfs(k,i))
			return 1;
		}			
	}
	return 0;
}
int main()
{
	char t;
	scanf("%d\n",&n);
	int i,j;
	for(i=1;i<=n;i++)
	{
	scanf("%s",a[i]+1);
	//printf("%s\n",a[i]+1);
	}
	for(i=1;i<=n;i++)
	{
		if(known[i]==0&&dfs(i,i))
		return 0;
	}
	printf("-1\n");
	return 0;
}
