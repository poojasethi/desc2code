#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;
const int N=5005;
char a[N][N];
int v[N],n;

void dfs(int x,int p)
{
    int i;
    v[x]=1;
    for (i=0; i<n; i++)
     if (a[x][i]=='1')
     {
      if (v[i]==0) dfs(i,x);
      else if (p!=-1 && a[i][p]=='1')
      {printf("%d %d %d",p+1,x+1,i+1); exit(0);};
     }
}

int main()
{
    int i;
    scanf("%d",&n);
    for (i=0; i<n; i++) scanf("%s",a[i]);
    for (i=0; i<n; i++)
     if (v[i]==0)
      dfs(i,-1);
    printf("-1\n");
}
