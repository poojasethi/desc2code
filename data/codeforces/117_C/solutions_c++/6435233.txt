#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

const int mxn = 5005;
char g[mxn][mxn];
bool vis[mxn];
int n;
bool dfs(int now, int pre)
{
	vis[now] = true;
	for(int j = 1; j <= n; j++) 
		if(g[now][j] == '1') 
		{	if(g[j][pre] == '1')
			{
				printf("%d %d %d\n", pre, now, j);
				return true;
			}
			if(!vis[j]) 
				if(dfs(j,now)) 
					return true;
		}
	return false;
	
}

int main()
{
	cin >> n;
	for(int i = 1; i <= n; i++)
		scanf("%s", g[i]+1);
	for(int i = 1; i <= n; i++)
		if(!vis[i])
			if(dfs(i,i)) return 0;
	puts("-1");
	return 0;
}
