#include <cstdio>
#include <cstring>

const int MaxN = 5050;

int n;
char map[MaxN][MaxN],vst[MaxN];

bool dfs(int u,int pre){
	vst[u] = 1;
	// pre -- > u --> i --> pre
	for(int i = 1; i <= n; i++)
		if (map[u][i]) {
			if(map[i][pre]) {
				printf("%d %d %d\n",pre,u,i);
				return true;
			}
			if(!vst[i] && dfs(i,u))	return 1;
		}
	return false;
}

int main(){
	scanf("%d",&n);

	for(int i = 1; i <= n; i++)
		scanf("%s",map[i]+1);
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= n; j++)
			map[i][j] -= '0';
	for(int i = 1; i <= n; i++)
		if(!vst[i] && dfs(i,i))
			return 0;
	puts("-1");
	return 0;
}
