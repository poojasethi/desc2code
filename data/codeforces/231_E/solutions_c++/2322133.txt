#include <cstdio>
#define N 100005
#define FI(a, b, c) for(int a = (b); a <= (c); a++)
#define fe(a, b, c) for(int a = (b); a; a = c[a])

int n, m, f[N], v[N + N], o[N + N], s[N], p;
int z[N], Z[N + N], q, F[N], V[N + N], O[N + N], t[N], A[N], d[N];
bool u[N], ue[N], w[N];

void bd(int a, int b){
	v[++p] = b; o[p] = f[a]; f[a] = p;
}

void BD(int a, int b){
	V[++p] = b; O[p] = F[a]; F[a] = p;
}

int fa(int x){
	return x == s[x] ? x : s[x] = fa(s[x]);
}

void cyc(int y){
	s[p++] = y;
	u[y] = 1;

	fe(i, f[y], o){
		if(ue[i + 1 >> 1]) continue;
		ue[i + 1 >> 1] = 1;
		
		if(u[v[i]]){
			w[++q] = 1;
			for(int j = p - 1; s[j + 1] != v[i]; j--) z[s[j]] = q;
		}else cyc(v[i]);
	}
	
	p--;
	if(!z[y]) z[y] = ++q;
}

void dfs(int y, int P){
	u[y] = 1;
	s[y] = y;
	d[y] = d[P] + w[y];
	
	fe(i, f[y], o) if(s[v[i]]){
		int x = fa(v[i]);
		A[Z[i]] = d[y] + d[v[i]] - d[x] * 2 + w[x];
	}
	
	fe(i, F[y], O) if(!s[V[i]])
		dfs(V[i], y);
	
	s[y] = P;
}

int main(){
	scanf("%d %d", &n, &m);
	while(m--){
		int a, b;
		scanf("%d %d", &a, &b);
		bd(a, b);
		bd(b, a);
	}
	
	p = 0;
	cyc(1);
	FI(i, 1, n)
		fe(j, f[i], o) if(z[v[j]] != z[i])
			BD(z[v[j]], z[i]);
	
	t[0] = 1;
	FI(i, 1, q) t[i] = t[i - 1] * 2 % 1000000007;
	
	FI(i, 1, q) f[i] = s[i] = 0;
	p = 0;
	
	scanf("%d", &m);
	FI(i, 1, m){
		int a, b;
		scanf("%d %d", &a, &b);
		a = z[a]; b = z[b];
		bd(a, b); Z[p] = i;
		bd(b, a); Z[p] = i;
	}
	
	dfs(1, 0);
	FI(i, 1, m) printf("%d\n", t[A[i]]);
}
