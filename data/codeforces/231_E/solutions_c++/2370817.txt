#include <stdio.h>
#include <vector>
#include <stdlib.h>
#include <string.h>
using namespace std;

#define pb push_back
#define D 1000000007

vector<int> v[100005];
vector<int> V[100005];
int n,m,N;
int via[100005];
int cy[100005];
int wh[100005];
long long bi[100005];
int S[100005];
int L[100005];
int P[100005][25];
bool ck[100005];
bool in[100005];

void go(int x){
	ck[x]=1;
	in[x]=1;
	wh[x]=x;
	int i,k;
	for(i=0;i<v[x].size();i++){
		k=v[x][i];
		if(k==via[x])continue;
		if(in[k]){
			int T=k;
			cy[T]=1;
			for(k=x;k!=T;k=via[k])wh[k]=T;
		}if(!ck[k])via[k]=x,go(k);
	}
	in[x]=0;
}

void GO(int x){
	int i,k=via[x];
	P[x][0]=k;
	L[x]=L[k]+1;
	S[x]=S[k]+cy[x];
	ck[x]=1;
	for(i=0;i<V[x].size();i++){
		k=V[x][i];
		if(k!=via[x] && !ck[k])via[k]=x,GO(k);
	}
}
int LCA(int x,int y){
	if(L[x]<L[y])swap(x,y);
	int i;
	for(i=0;i<18;i++)if(L[x]-L[y]>>i&1)x=P[x][i];
	if(x==y)return x;
	for(i=17;i>=0;i--)if(P[x][i]!=P[y][i]){
		x=P[x][i];
		y=P[y][i];
	}return P[x][0];
}
int main(){
	scanf("%d%d",&n,&m);
	int i,j,s,e;
	while(m--){
		scanf("%d%d",&s,&e);
		v[s].pb(e);
		v[e].pb(s);
	}
	go(1);
	for(i=1;i<=n;i++)for(j=0;j<v[i].size();j++){
		s=wh[i];
		e=wh[v[i][j]];
		if(s!=e)
			V[s].pb(e),
			V[e].pb(s);
	}
	memset(ck,0,sizeof(ck));
	GO(1);
	for(int k=1;k<18;k++)for(i=1;i<=n;i++)if(wh[i]==i){
		P[i][k]=P[P[i][k-1]][k-1];
	}
	bi[0]=1;
	for(i=1;i<=n;i++)bi[i]=bi[i-1]*2%D;

	scanf("%d",&m);while(m--){
		scanf("%d%d",&s,&e);
		s=wh[s];
		e=wh[e];
		int x=LCA(s,e);
		int K=S[s]+S[e]-S[x]*2+cy[x];
		printf("%I64d\n",bi[K]);
	}
}
