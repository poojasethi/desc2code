#include <algorithm>
#include <stdio.h>
using    namespace std;

int a,b,x1,y1,x2,y2;

int f(int x , int y)
{
  if(x>=0)  return x/y;
  return x/y-1;
}

int main()
{
  scanf("%d%d%d%d%d%d",&a,&b,&x1,&y1,&x2,&y2);
  int c1=abs(f(x1+y1,a*2)-f(x2+y2,a*2));
  int c2=abs(f(x1-y1,b*2)-f(x2-y2,b*2));
  printf("%d\n",max(c1,c2));
  return 0;
}
