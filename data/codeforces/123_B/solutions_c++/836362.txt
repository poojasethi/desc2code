#include<iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
int a,b,x1,y1,x2,y2,i,j;
int main()
{
    scanf("%d%d%d%d%d%d",&a,&b,&x1,&y1,&x2,&y2);
    i=(x2+y2)/(2*a)-(x1+y1)/(2*a);
    if (x2+y2<0) i--;
    if (x1+y1<0) i++;
    j=(x2-y2)/(2*b)-(x1-y1)/(2*b);
    if (x2-y2<0) j--;
    if (x1-y1<0) j++;
    if (i<0) i=-i;
    if (j<0) j=-j;
    if (i<j) printf("%d\n",j); else printf("%d\n",i);
    return 0;
}
