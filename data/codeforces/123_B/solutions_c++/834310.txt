#include <iostream>
#include <algorithm>
using namespace std;
int a,b,x,y,x2,y2;
int f(int x){
    return (x>=0?x:-x);
}
int g(int x,int y){
    if(x>=0)  return x/y;
    return x/y-1;
}
int main()
{
    cin>>a>>b>>x>>y>>x2>>y2;
    a*=2; b*=2;
    cout<<max(f(g(x+y,a)-g(x2+y2,a)),f(g(x-y,b)-g(x2-y2,b)));
}
