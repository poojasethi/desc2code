#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
long long f(int a,int b){
	if(a>=0){
		return a/b;
	}
	else{
		return (a/b - 1);
	}
}
int main(){
	int a,b,x1,y1,x2,y2;
	cin>>a>>b>>x1>>y1>>x2>>y2;
	a*=2;b*=2;
	long long c,d;
	c=abs(f(x1+y1,a)-f(x2+y2,a));
	d=abs(f(x1-y1,b)-f(x2-y2,b));
	cout<<max(c,d)<<endl;
	return 0;
}
