#include <iostream>
#include <cmath>

using namespace std;

int mydiv(int a,int b){

  return a>=0?a/b:-1+a/b;

}

int main(){

  int a,b,x1,x2,y1,y2;
  cin>>a>>b>>x1>>y1>>x2>>y2;
  
  int n1=abs(mydiv(x1+y1,2*a)-mydiv(x2+y2,2*a));
  int n2=abs(mydiv(x1-y1,2*b)-mydiv(x2-y2,2*b));
  
  cout<<max(n1,n2)<<endl;

  return 0;


}
