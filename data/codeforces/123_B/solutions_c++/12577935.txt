#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;

int a,b,x_1,y_1,x_2,y_2;

int main()
{
	int k1,k2;
	while(cin>>a>>b>>x_1>>y_1>>x_2>>y_2)
	{
		
		k1=abs((x_1+y_1)/(2*a)-(x_2+y_2)/(2*a));
		if((long long)(x_1+y_1)*(x_2+y_2)<=0)
			k1++;
		k2=abs((x_1-y_1)/(2*b)-(x_2-y_2)/(2*b));
		if((long long)(x_1-y_1)*(x_2-y_2)<=0)
			k2++;
		cout<<(max(k1,k2))<<endl;
	}
	return 0;
}
