#include<algorithm>
#include<iostream>

using namespace std;

int f(int a,int b){
    return a>=0? a/b : -1 + a/b;
}
int main(){
    int a,b,x1,y1,x2,y2;
    cin >> a >> b >> x1 >> y1 >> x2 >> y2;
    a <<= 1; b <<= 1;
    cout << max(abs(f(x1 + y1, a) - f(x2 + y2, a)),abs(f(x1 - y1, b) - f(x2 - y2, b))); 
    return 0;   
}
