#include<cstdio>
#include<algorithm>
#include<cmath>

int main() {
	double a,b,x1,y1,x2,y2;
	scanf("%lf%lf%lf%lf%lf%lf", &a, &b, &x1, &y1, &x2, &y2);
	double n1 = floor((x1+y1)/(2.0*a)), m1 = floor((x1-y1)/(2.0*b));
	double n2 = floor((x2+y2)/(2.0*a)), m2 = floor((x2-y2)/(2.0*b));
	int n = std::abs(n1-n2), m = std::abs(m1-m2);
	printf("%d\n", std::max(n,m));
	return 0;
}
