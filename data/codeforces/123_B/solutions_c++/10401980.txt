#include <iostream>
#include <cmath>
int dv(int n,int m)
{
	if(n>0)return n/m;
	else
	{
		if(n%m==0)return n/m;
		else return n/m-1;
	}
}
using namespace std;
int main()
{
	int a,b,x1,y1,x2,y2;
	cin>>a>>b>>x1>>y1>>x2>>y2;
	int g1=x1+y1;
	int g2=x2+y2;
	int h1=x1-y1;
	int h2=x2-y2;
	int m1=abs(dv(g1,2*a)-dv(g2,2*a));
	int m2=abs(dv(h1,2*b)-dv(h2,2*b));
	cout<<max(m1,m2)<<endl;
	return 0;
}
