#include <iostream>
#include <algorithm>

using namespace std;

int main(void)
{
	long long int a,b,x1,y1,x2,y2,r1,r2,p1,p2;
	cin>>a>>b>>x1>>y1>>x2>>y2;
	r1=(x1+y1)/(2*a);
	r2=(x2+y2)/(2*a);
	if (x1+y1<0)
		r1--;
	if (x2+y2<0)
		r2--;	
	p1=(x1-y1)/(2*b);
	p2=(x2-y2)/(2*b);
	if (x1-y1<0)
		p1--;
	if (x2-y2<0)
		p2--;
//	cout<<r1<<" "<<r2<<" "<<p1<<" "<<p2<<endl;
	cout<<max(abs(r1-r2),abs(p1-p2));
	return 0;
}
