#include <cstdio>
#include <iostream>
#include <vector>
#include <cmath>
#include <stdlib.h>
using namespace std;
int main()
{
	int a,b,x1,y1,x2,y2;
	int _x1,_y1,_x2,_y2;
	cin>>a>>b>>x1>>y1>>x2>>y2;
	_x1=(x1+y1)/(2*a);	if (x1+y1<0) _x1--;
	_y1=(x1-y1)/(2*b);	if (x1-y1<0) _y1--;
	_x2=(x2+y2)/(2*a);	if (x2+y2<0) _x2--;
	_y2=(x2-y2)/(2*b);	if (x2-y2<0) _y2--;

	_x1=abs(_x2-_x1); _y1=abs(_y2-_y1);
	cout<<max(_y1,_x1)<<endl;
	return 0;
}
