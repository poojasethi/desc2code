#include <cstdio>
#include <algorithm>

using namespace std;

int mydiv (long long x, long long y) {
  if (x >= 0) return x / y; else return - ((-x + y - 1) / y);
}

int main () {
  int a, b, x1, y1, x2, y2;
  scanf ("%d%d%d%d%d%d", &a, &b, &x1, &y1, &x2, &y2);
  int c1 = mydiv (x1 + y1, 2 * a), c2 = mydiv (x2 + y2, 2 * a);
  int d1 = mydiv (x1 - y1, 2 * b), d2 = mydiv (x2 - y2, 2 * b);
  printf ("%d\n", max (abs (c2 - c1), abs (d2 - d1)));

  return 0;
}
