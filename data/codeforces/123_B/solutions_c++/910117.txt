#include <algorithm>
#include <iostream>
#include <stdio.h>
using    namespace std;
typedef  long long llint;

llint a,b,x1,y1,x2,y2;

llint F(llint a , llint b) 
{ 
  return (a>=0) ? a/b : (-1 + a/b); 
} 
 
int main() 
{ 
  cin >> a >> b >> x1 >> y1 >> x2 >> y2;
  a*=2;b*=2;
  llint m1,m2;
  m1 = abs ( F(x1+y1 , a) - F(x2+y2 , a) ); 
  m2 = abs ( F(x1-y1 , b) - F(x2-y2 , b) ); 
  cout << max(m1,m2) << endl; 
  return 0; 
}
