#include<algorithm>
#include<iostream>

typedef long long ll;

ll floor( ll a, ll b )
{
    if( a >= 0 )
        a /= b;
    else
        a = (a-b+1)/b;

    return a;
}

int main()
{
    ll a, b, x1, y1, x2, y2;
    std::cin >> a >> b >> x1 >> y1 >> x2 >> y2;

    std::cout << std::max(abs(floor(x2-y2, 2*b)-floor(x1-y1, 2*b)), abs(floor(x2+y2, 2*a)-floor(x1+y1, 2*a))) << std::endl;

    return 0;
}
