#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
long long floor(long long p,int a){
    return (p-(p%a+a)%a)/a;
}
int main(){
    long long a,b,x1,y1,x2,y2; long long X1,X2,Y1,Y2;
    cin>>a>>b>>x1>>y1>>x2>>y2;
    a*=2; b*=2;
    X1=floor(x1+y1,a);
    X2=floor(x2+y2,a);
    Y1=floor(y1-x1,b);
    Y2=floor(y2-x2,b);
    printf("%lld\n",max(abs(X1-X2),abs(Y1-Y2)));
    return 0;
}
