#include <iostream>
#include <cmath>
using namespace std;
int main()
{
    int a,b,x1,y1,x2,y2;
    int k1,k2,k3,k4;
    cin>>a>>b>>x1>>y1>>x2>>y2;
    k1=(x1+y1)/(2*a);  if (x1+y1<0) k1--;
    k2=(x1-y1)/(2*b);  if (x1-y1<0) k2--;
    k3=(x2+y2)/(2*a);  if (x2+y2<0) k3--;
    k4=(x2-y2)/(2*b);  if (x2-y2<0) k4--;

    k1=abs(k3-k1);
     k2=abs(k4-k2);
    cout<<max(k2,k1)<<endl;
    //system("pause");
}
