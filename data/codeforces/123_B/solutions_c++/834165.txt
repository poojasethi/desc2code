#include<iostream>
#include<cstdlib>
#include<string>
using namespace std;
long long a,b,x1,y1,x2,y2,c,d;
int main()
{
	//freopen("input.txt","rb",stdin);
	//freopen("output.txt","wb",stdout);
	cin>>a>>b>>x1>>y1>>x2>>y2;
	a*=2,b*=2;
	c=abs(min(x1-y1,x2-y2))/b+1;
	d=abs(min(x1+y1,x2+y2))/a+1;
	cout<<max(abs((x1+y1+d*a)/a-(x2+y2+d*a)/a),abs((x1-y1+c*b)/b-(x2-y2+c*b)/b));
	return 0;
}
