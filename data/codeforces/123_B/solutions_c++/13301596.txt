#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    long long a,b,x1,y1,x2,y2;
    cin>>a>>b>>x1>>y1>>x2>>y2;
    cout<<(long long)max(abs(floor((long double)(x1+y1)/(2*a))-floor((long double)(x2+y2)/(2*a))),abs(floor((long double)(x1-y1)/(2*b))-floor((long double)(x2-y2)/(2*b))))<<endl;
}
