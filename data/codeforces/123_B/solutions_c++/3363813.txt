#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	long long a,b,c,d,x1,y1,x2,y2;
	while(cin>>a>>b>>x1>>y1>>x2>>y2)
	{
		a*=2;b*=2;
		if((x1+y1)>0&&(x2+y2)>0||(x1+y1)<0&&(x2+y2)<0)
		{
			c=abs((x1+y1)/a-(x2+y2)/a);
		}
		else c=abs((x1+y1)/a-(x2+y2)/a)+1;
		if((x1-y1)>0&&(x2-y2)>0||(x1-y1)<0&&(x2-y2)<0)
		{
			d=abs((x1-y1)/b-(x2-y2)/b);
		}
		else d=abs((x1-y1)/b-(x2-y2)/b)+1;
		cout<<max(d,c)<<endl;
	}
}
