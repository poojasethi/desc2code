#include<iostream>
#include<algorithm>
using namespace std;

int a,b,x1,y1,x2,y2;

long long f(int a,int b)
{
	return (a>=0)?a/b:(-1+a/b);
}

int main()
{
	cin>>a>>b>>x1>>y1>>x2>>y2;
	a*=2;b*=2;
	long long m1,m2;
	m1=abs(f(x1+y1,a)-f(x2+y2,a));
	m2=abs(f(x1-y1,b)-f(x2-y2,b));
	cout<<max(m1,m2)<<endl;
	return 0;
}
