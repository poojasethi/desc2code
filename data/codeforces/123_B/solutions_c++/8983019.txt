#include <bits/stdc++.h>

int main() {
	int a, b, x1, y1, x2, y2;
	std::cin >> a >> b >> x1 >> y1 >> x2 >> y2;
	int x = x1, y = y1;
	x1 = x + y;
	y1 = x - y;
	x = x2, y = y2;
	x2 = x + y;
	y2 = x - y;
	a *= 2;
	b *= 2;
	int numx = x2 / a + (x2 > 0) - x1 / a - (x1 > 0);
	int numy = y2 / b + (y2 > 0) - y1 / b - (y1 > 0);
	printf("%d\n", std::max(abs(numx), abs(numy)));
	return 0;
}
