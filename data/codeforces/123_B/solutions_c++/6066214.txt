#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

int a, b, X1, Y1, X2, Y2;
int p1, q1, p2, q2;
int Ans;

int main()
{
    scanf("%d %d %d %d %d %d", &a, &b, &X1, &Y1, &X2, &Y2);

    p1 = floor( (double)(X1+Y1)/(2*a));
    q1 = floor( (double)(X1-Y1)/(2*b));

    p2 = floor( (double)(X2+Y2)/(2*a));
    q2 = floor( (double)(X2-Y2)/(2*b));

    Ans = max(abs(p1-p2), abs(q1-q2));

    printf("%d\n", Ans);
}
