#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;

int a,b,x1,y1,x2,y2;

long long DIV(int a,int b)
{
	if(a>=0)
		return a/b;
	return -1-(-a)/b;
}

int main()
{
	scanf("%d%d%d%d%d%d",&a,&b,&x1,&y1,&x2,&y2);
	a*=2;
	b*=2;
	long long t1=abs(DIV(x1+y1,a)-DIV(x2+y2,a));
	long long t2=abs(DIV(x1-y1,b)-DIV(x2-y2,b));
	cout<<max(t1,t2)<<endl;
	return 0;
}
