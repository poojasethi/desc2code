#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;
ll Count(ll f,ll t,ll div){
  if(f>t)swap(f,t);
  ll fp=(f/div)*div,tp=(t/div)*div;
  if(tp>t)tp-=div;if(fp<f)fp+=div;
  if(fp>tp)return 0;
  return (tp-fp)/div+1; 
}
int main(){
  ll a,b,x1,y1,x2,y2;
  cin>>a>>b>>x1>>y1>>x2>>y2;
  cout<<max(Count(x1+y1 , x2+y2 , 2*a) , Count(x1-y1 , x2-y2 , 2*b))<<'\n';
  return 0;
}
