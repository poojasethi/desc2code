#include<cstdio>
#include<cstdlib>
#include<cstring>
__int64 max(__int64 a,__int64 b){return a>b?a:b;}
__int64 myabs(__int64 x)
{
    return x>0?x:-x;
}
__int64 f(__int64 x,__int64 a)
{
    return (x>=0)?(x/a):(x/a-1);
}
int main()
{
    __int64 x1,x2,y1,y2,a,b;
    scanf("%I64d%I64d%I64d%I64d%I64d%I64d",&a,&b,&x1,&y1,&x2,&y2);
    a*=2,b*=2;
    printf("%I64d\n",max(myabs(f(x2+y2,a)-f(x1+y1,a)),myabs(f(x2-y2,b)-f(x1-y1,b))));
}
