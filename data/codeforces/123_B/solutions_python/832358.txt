#!/usr/bin/env python

from sys import stdin, stderr

def rot(x, y):
   return (.5*x - .5*y, .5*x + .5*y)

def cand(X, Y, A, B):
   P = []
   if X % B == 0:
      if Y % A == 0:
         P.append( (X-1, Y-1) )
         P.append( (X-1, Y+1) )
         P.append( (X+1, Y-1) )
         P.append( (X+1, Y+1) )
      else:
         P.append( (X-1, Y) )
         P.append( (X+1, Y) )
   else:
      if Y % A == 0:
         P.append( (X, Y-1) )
         P.append( (X, Y+1) )
      else:
         P.append( (X, Y) )
   return P

def dist(p1, p2, A, B):
   print >>stderr, p1, p2
   x1 = p1[0]/B
   y1 = p1[1]/A
   x2 = p2[0]/B
   y2 = p2[1]/A
   dx = abs(x1-x2)
   dy = abs(y1-y2)
   print >>stderr, 'dx: %d  dy: %d' % (dx, dy)
   return max(dx, dy)


a, b, x1, y1, x2, y2 = map(int, stdin.readline().split())
A = a*2
B = b*2

(X1, Y1) = rot(x1, y1)
X1 = int(X1*2)
Y1 = int(Y1*2)
P1 = cand(X1, Y1, A, B)

(X2, Y2) = rot(x2, y2)
X2 = int(X2*2)
Y2 = int(Y2*2)
P2 = cand(X2, Y2, A, B)

res = 1000000000000000000
for p1 in P1:
   for p2 in P2:
      res = min(res, dist(p1, p2, A, B))

print res
