def numCandleBurning(t):
    count = 0
    for i in range(len(candleTimeArray)-1, -1, -1):
        if(candleTimeArray[i] + timeBurning > t):
            count += 1
        else:
            return count
    return count

numGhosts, timeBurning, minNumCandle = map(int, raw_input().split())
ghostTimeArray = map(int, raw_input().split())
if(timeBurning < minNumCandle): print -1
else:
    candleTimeArray = []
    for ghostTime in ghostTimeArray:
        numCandlesNeeded = max(minNumCandle - numCandleBurning(ghostTime), 0)
        for i in range(numCandlesNeeded):
            candleTimeArray.append(ghostTime - numCandlesNeeded + 1 + i)
    print len(candleTimeArray)
