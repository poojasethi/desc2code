def readlist(f): return map(f, raw_input().split())
def readint(): return int(raw_input())
def printlist(l): print ' '.join(map(str, l))

m, t, r = readlist(int)
W = readlist(int)

if t < r:
  print -1
  exit()

res = 0
lit = []
for w in W:
  lit = filter(lambda x: x + t >= w, lit)
  for k in range(r - len(lit)):
    res += 1
    lit.append(w - k - 1)
print res
