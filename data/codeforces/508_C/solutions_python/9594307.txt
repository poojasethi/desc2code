m, t, r = map(int, raw_input().split())
w = map(int, raw_input().split())
f = {i: 0 for i in xrange(w[0] - t, 301)}
for x in w:
    y = sum([f[h] for h in xrange(x - t, x)])
    if y >= r:
        continue
    for i in xrange(x - 1, x - min(r - y, t) - 1, -1):
        f[i] = 1
    if sum([f[h] for h in xrange(x - t, x)]) < r:
        print -1
        quit()
print sum(f.values())
