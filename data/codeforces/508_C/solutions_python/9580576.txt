m, t, r = map(int, raw_input().split())
G = map(int, raw_input().split())
if r > t:
    print -1
else:
    C = set()
    for x in G:
        h = 0
        for y in C:
            if y < x <= y + t:
                h += 1
        j = x - 1
        while h < r:
            if j not in C:
                C.add(j)
                h += 1
            j -= 1
    print len(C)
