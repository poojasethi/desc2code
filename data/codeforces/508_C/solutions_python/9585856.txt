import sys

m,t,r=map(int,raw_input().split())
arr=map(int,raw_input().split())
arr.sort()
brr=[0]*(r+1)
if r>t:
    print -1
else:
    for i in xrange(0,r):
        brr[i]=arr[0]-i-1
    ans=r 
    for i in xrange(1,m):
        s=arr[i]
        e=arr[i-1]-1
        for j in range(0,r):
            if arr[i]-brr[j]>t:
                ans+=1
                s-=1
                brr[j]=s
                
        if s<=e:
            print -1
            sys.exit(0)

    print ans
