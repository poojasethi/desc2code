(ghostCount, lightDuration, reqLight) = map(int, raw_input().split())
ghosts = {x: 1 for x in map(int, raw_input().split())}

light = 0
candles = {}

for time in xrange(300, 0, -1):
	if time in candles:
		light -= 1

	if time in ghosts and light < reqLight:
		lightStart = time - lightDuration
		lightEnd = lightStart + reqLight - light - 1

		if lightEnd >= time:
			print -1
			exit()

		for t in xrange(lightStart, lightEnd + 1):
			candles[t] = 1
			light += 1

print len(candles)
