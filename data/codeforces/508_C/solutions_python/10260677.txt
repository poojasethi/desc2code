N,T,R = map(int,raw_input().split())
ghost = map(int,raw_input().split())


time=[ghost[0]-i for i in range(R,0,-1)]
pos=0

if(T >= R):
    for i in range(1,len(ghost)):
        cnt=0
        for j in range(pos,len(time)):
             if(time[j]+T < ghost[i]):
                cnt+=1

        for j in range(cnt,0,-1):
            time.append(ghost[i]-j)
            pos+=1
          
    
    print len(time)

    
else:
    print -1
