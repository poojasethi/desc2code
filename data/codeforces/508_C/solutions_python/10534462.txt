m , t , r = map(int , raw_input().split());
l = map(int , raw_input().split());

if r > t :
	print "-1";
else:
	cand = [] ; ans = 0 ;
	for i in l:
		lite = 0
		for j in cand:
			if i - j <= t:
				lite += 1;
		to_lite = r - lite
		c = i - 1
		for j in range(to_lite):
			ans += 1
			cand.append(c);
			c -= 1
	print ans;
