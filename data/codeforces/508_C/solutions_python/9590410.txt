inp = [int(_) for _ in raw_input().split()]
m = inp[0]
t = inp[1]
r = inp[2]
visits = [int(_) for _ in raw_input().split()]
lightStart = []
lightEnd = []

def alreadyLit(time):
  tmp = 0
  for i in sorted(range(len(lightEnd)), reverse=True):
    if lightEnd[i] > time:
      tmp += 1
  return tmp

if r > t:
  print -1
else:
  ans = 0
  for i in range(m):
    # print "visits[i]", visits[i]
    a = alreadyLit(visits[i])
    # print "a", a
    add = r - a
    # print "add", add
    ans += add
    index = visits[i] - 1
    while add > 0:
      while index in lightStart:
        index -= 1
      lightStart.append(index)
      lightEnd.append(index + t + 1)
      add -= 1
    # print "lightStart", lightStart
    # print "lightEnd", lightEnd
  print ans
