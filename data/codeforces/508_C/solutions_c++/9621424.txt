#include<bits/stdc++.h>
using namespace std;

int ans, m, r, t, ghost[605], lamp[605];

main() {
    scanf("%d %d %d", &m, &t, &r);
    for(int i=0; i<m; i++) scanf("%d", &ghost[i]);

    if (r>t) {
        puts("-1");
        return 0;
    }

    for(int i=0; i<m; i++) {
        for(int j=ghost[i]-1; lamp[ghost[i]]<r; j--){
            for(int k=j+1; k<=j+t; k++) if (k>0) lamp[k]++;
            ans++;
        }
    }printf("%d\n", ans);
}
