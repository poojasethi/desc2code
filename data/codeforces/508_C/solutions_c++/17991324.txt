#include <stdio.h>
#define M 403
int m,t,r;
int g[M],s[M];
void f(int i){
	int k = r - s[i];
	for(int j = t;k;k--,j--)
		for(int a = 0;a<j;s[i+a++]++);		
}
int main(){
	int tmp,i,res;
	scanf("%d%d%d",&m,&t,&r);
	for(int i = 0;i<m;i++){
		scanf("%d",&tmp);
		g[tmp] = 1;
	}
	if(t<r) res=-1;
	else
		for(i = res = 0;;f(i)){
			for(i++;!g[i] && i<M;i++);
			if(i>=M) break;
			res += r - s[i];
		}	
	printf("%d\n",res);
}
