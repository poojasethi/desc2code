#include <bits/stdc++.h>
using namespace std;
int w[800];

int *tab;

int main()
{
	tab = w+300;
	int n,r,t,a;
	int ile = 0, suma = 0;
	scanf("%d%d%d",&n,&t,&r);
	for (int x=0;x<n;x++)
	{
		ile = 0;
		scanf("%d",&a);
		for (int i=0;i<t;i++)
			if ( tab[a-i] ) ile++;
		suma += r-ile;
		for (int i=0;i<t;i++)
			if ( tab[a-i] == 0 && ile < r )
			{
				tab[a-i] = 1;
				ile++;
			}
	}
	if ( r > t ) printf("-1\n");
	else printf("%d\n",suma);
}
