#include <iostream>

using namespace std;
int m,t,r,tl[305],g,candles=0,exist;
int main(){
	cin>>m>>t>>r;
	for(int q=0;q<m;q++){
		cin>>g;
		//CHECK HOW MANY EXISTING
		exist=tl[g];
		for(int w=g-r+exist;w<g;w++){
			for(int e=0;e<t;e++){
				if(w+1+e>300)break;

				if(w+1+e>0 && w+1+e<301){
					tl[w+1+e]++;
				}			
			}
			candles++;	
		}
		if(tl[g]!=r){
			cout<<"-1\n";
			return 0;
		}
	}
	cout<<candles<<endl;
	return 0;
}
