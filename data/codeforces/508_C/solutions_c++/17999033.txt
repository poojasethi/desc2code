#include<bits/stdc++.h>
using namespace std;
int a[1000],b[1000];
int main()
{
	int m,t,r,tmp;
	scanf("%d%d%d",&m,&t,&r);
	for(int i=1;i<=m;i++)
	{
		scanf("%d",&tmp);
		a[tmp]++;
	}
	int ans=0;
	for(int i=1;i<=300;i++)
	{
		if(a[i]&&t>=r)
		{
			ans+=r-b[i];
			for(int j=i+1;j<i+t;j++)
			b[j]+=r-b[i];
			for(int j=i+t-1,k=1;k<r-b[i];k++,j--)
			b[j]-=r-b[i]-k;
		}
	}
	if(t>=r)
	printf("%d",ans);
	else
	printf("-1");
	return 0;
}
