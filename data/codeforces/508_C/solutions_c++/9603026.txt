#include<bits/stdc++.h>
using namespace std;

set<int> s;

int m,t,r,x;

int main()
{
	cin >> m >> t >> r;
	
	if(r>t)
	{
		cout <<"-1"<<endl;
	}
	else
	{
		int cnt=0;
		for(int i=0;i<m;i++)
		{
			cin >> x;
			
			while(!s.empty() && *(s.begin())<=x)
			{
				s.erase(s.begin());
			}
			
			int temp=x;
					
			while(s.size()<r)
			{
				s.insert(temp+t);
				cnt++;
				temp--;
			}
			
		}
		
		cout <<cnt<<endl;
		
	}
	
	return 0;
}
