#include <iostream>
#include <set>
using namespace std;
set<int> s;
int main() {
	int n,t,r;
	cin>>n>>t>>r;
	if(t<r) {
		int x;
		for(int i=0;i<n;i++) cin>>x; 
		cout<<-1<<endl;
	}
	else {
		int ans=0;
		while(n--) {
			int x;
			cin>>x;
			while(!s.empty()&&*s.begin()<=x) s.erase(s.begin());
			int tmp=x;
			while(s.size()<r) {
				s.insert(tmp+t);
				tmp--;
				ans++;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
