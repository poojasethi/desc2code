#include <bits/stdc++.h>
using namespace std;
int n, t, r, tmp[1000], a[1000], x, ans;
int main(){
    cin>>n>>t>>r;
    if(t < r) return cout<<-1, 0;
    for(int i = 0; i < n; i++){
        cin>>a[i];
        x = tmp[a[i]] + a[i] - r + 1;
        ans += fabs(tmp[a[i]] - r);
        for(int j = x; j <= a[i]; j++)
            for(int k = j; k < j + t; k++) tmp[k]++;

    }
    cout<<ans;
}
