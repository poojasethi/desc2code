#include <bits/stdc++.h>
using namespace std;

int main() {
	int m, t, r, w[300];
	cin >> m >> t >> r;
	if (r > t) {
		cout << -1 << endl;
		return 0;
	}
	for (int i = 0; i < m; i++) cin >> w[i];
	int res = r, s[300];
	for (int j = 0; j < r; j++) s[j] = w[0]-j;
	for (int i = 1; i < m; i++) {
		// light new candles, if needed
		int k = 0;
		for (int j = 0; j < r; j++) {
			if (w[i] - s[j] >= t) {
				s[j] = w[i]-k++;
				res++;
			}
		}
	}
	cout << res << endl;
	return 0;
}
