#include <iostream>
#include <algorithm>
using namespace std;

#define MAX 1001
#define INF 1000000

int n,t,r,w[MAX],s[MAX*3];

int main(){
	int i,j,a,b=0;
	cin>>n>>t>>r;
	for (i=0; i<n; i++){ cin>>w[i]; w[i]+=1000	; }
	for (i=0; i<n; i++){
		for (a=0,j=w[i]-1; j>w[i]-1-t; j--) a+=s[j];
		for (j=w[i]-1; j>w[i]-1-t && a<r; j--) if (s[j]==0){ a+=(s[j]=1); b++;} 
		if (a<r) break;
	}
	if (i<n) b=-1;
	cout<<b<<endl;
	return 0;
}
