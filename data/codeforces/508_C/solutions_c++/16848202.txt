#include <bits/stdc++.h>
using namespace std;
int cdl[610],ans;
int main(){
	int m,t,r; cin >> m >> t >> r;
	if(t<r) return puts("-1");
	for(int i = 0 ; i < m ; i++){
		int w; cin >> w;
		for(int j = w-1 ; cdl[w] < r ; j--){
			ans++;
			for(int l = j+1 ; l <= j+t ; l++){
				if(l>0)cdl[l]++;
			}
		}
	}
	cout << ans << endl;
}
