#include<bits/stdc++.h>
using namespace std;
main()
{
	int m,t,r;
	scanf("%d%d%d",&m,&t,&r);
	int a[m];
	for(int i=0;i<m;i++)
	{
		scanf("%d",&a[i]);
	}
	if(t<r){printf("-1\n");}
	else{
	int ans=r;
	int b[r];
	for(int i=0;i<r;i++){if(a[0]-r>=0){b[i]=a[0]-r+i+1;}else{b[i]=a[0]-r+i+1;}}
	for(int i=1;i<m;i++)
	{
		int l=0;
		for(int j=0;j<r;j++)
		{
			if(b[j]+t<=a[i])
			{
				b[j]=a[i]-l;ans++;l++;
			}
		}
	}
	printf("%d\n",ans);
}
}
