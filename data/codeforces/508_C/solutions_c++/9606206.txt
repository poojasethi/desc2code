#include<stdio.h>
int main(){
	int m,t,r,i,k,u,z,j;
	scanf("%d%d%d",&m,&t,&r);
	int a[m+1],b[r+1];
	if(r>t){
		for(i=0;i<m;i++){
			scanf("%d",&a[i]);
		}
		printf("-1");
	}else{
		scanf("%d",&a[0]);
		for(i=0;i<r;i++){
			b[i]=a[0]-i-1;
		}
		z=r;
		for(i=1;i<m;i++){
			scanf("%d",&a[i]);
			u=0;
		for(j=0;j<r;j++){
			if(b[j]+t < a[i]){
				b[j] = a[i]-u-1;
				u++;
				z++;
			}
		}
	}
		printf("%d",z);	
	}
	return 0;
}
