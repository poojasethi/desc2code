#include <stdio.h>
const int maxn = 500;
int m,t,r,x,cnt,a[2*maxn];

int main()
{
	scanf("%d %d %d", &m, &t, &r);
	if(r > t)
	{
		puts("-1");
		return 0;
	}
	while(m--)
	{
		scanf("%d",&x);
		int k = 0;
		for(int i = x-t+maxn; i < x+maxn; ++i) if(a[i])k++;

		if(k < r)
		{
			for(int i = x-1+maxn; i>= x-t+maxn; --i)
			{
				if(a[i] == 0)
				{	
					a[i] = 1;
					cnt++;
					k++;
				}
				if(k == r) break;
			}
		}
	}

	printf("%d", cnt);

	return 0;
}
