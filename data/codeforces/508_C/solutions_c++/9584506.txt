#include<iostream>
#include<cstdio>

using namespace std;

int a[1024];
int main()
{
    int m,t,r,i,j,c=0;
    cin>>m>>t>>r;
    while(m--)
    {
        int x;
        cin>>x;
        int k=r-a[x];
        for(i=x;k>0;i--,k--)
        {
            c++;
            for(j=x;j<i+t;j++)
                a[j]++;
        }
        if(a[x]<r)
        {
            printf("-1");
            return 0;
        }
    }
    cout<<c;

    return 0;
}
