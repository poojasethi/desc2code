#include<stdio.h>
int w[700],num[700];
int main(){
	int m,t,r,i,j,cur,sum = 0;
	scanf("%d%d%d",&m,&t,&r);
	for(i = 0; i<m; i++){
		scanf("%d",&w[i]);
	}
	if(t<r){
		printf("-1\n");
	}
	else{
		for(i = 0; i<m; i++){
			for(j = w[i]-1; num[w[i]]<r; j--){
				sum++;
				for(cur = j+1; cur<=j+t; cur++){
					if(cur>0)  num[cur]++;
				}
			}
		}
		printf("%d\n",sum);
	}
	return 0;
}
