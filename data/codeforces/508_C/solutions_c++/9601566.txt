#include <cstdio>
bool v[1005];
int ar[305];
int main() {
	int m, t, r, ans = 0;
	scanf("%d %d %d", &m, &t, &r);
	if(r > t) { puts("-1"); return 0; }
	for(int i = 0; i < m; ++i) scanf("%d", &ar[i]);
	for(int i = 0; i < m; ++i) {
		int add = 0;
		for(int j = ar[i] - 1; j >= ar[i] - t; --j)
			if(v[j + 300]) ++add;
		add = r - add;
		for(int j = ar[i] - 1; j >= ar[i] - t && add > 0; --j)
			if(!v[j + 300]) { --add; ++ans; v[j + 300] = 1; }
	}
	printf("%d", ans);
}
