#include<iostream>
using namespace std;

int main(){
    int m,r,t,g[310],v[800]={0},ca=0,curr=0;
    cin>>m>>t>>r;
    if(r>t) cout<<-1<<endl;
    else{
    for(int i=1;i<=m;i++) cin>>g[i];
    int i=1;
    for(int j=1;j<=m;j++){
        while(i<=g[j]){ curr+=v[i]; i++;}
        if(curr<r) for(int k=i-r+curr+t;k<i+t;k++) v[k]=-1;
        ca+=r-curr;
        curr=r;
    }
    cout<<ca<<endl;}
    return 0;
}
