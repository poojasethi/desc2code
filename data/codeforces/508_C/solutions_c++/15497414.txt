#include<stdio.h>
#define maxn 2000

int w[maxn],num[maxn],ans;
int main(){
	int m,t,r;
	scanf("%d %d %d", &m, &t, &r);
	for(int i=0; i<m; i++){
		scanf("%d", &w[i]);
	}
	if(t<r){
		puts("-1");
	}else{
		for(int i=0; i<m; i++){
			for(int j=w[i]-1; num[w[i]]<r; j--){
				ans++;
				for(int k=j+1; k<=j+t; k++){	
					if(k>0) num[k]++;
				}
			}
		}
		printf("%d\n", ans);
	}

	return 0;
}
