#include<iostream>
#include<string>
#include<algorithm>
#include<cmath>
using namespace std;
int main(){
int m,t,r,a[305],upper[305],i,j,c,ans=0;
cin>>m>>t>>r;
for(i=0;i<m;i++)
cin>>a[i];
for(i=0;i<r;i++)
upper[i]=0;
for(i=0;i<m;i++){
    c=1;
    for(j=0;j<r;j++) if(a[i]>upper[j]) upper[j]=a[i]-c+t,c++,ans++;
    for(j=0;j<r;j++) if(a[i]>upper[j]) {cout<<"-1";return 0;}
}
cout<<ans;
return 0;
}
