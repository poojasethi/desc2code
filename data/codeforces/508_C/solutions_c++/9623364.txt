#include <bits/stdc++.h>
using namespace std;

int main()
{
	int m,t,r,a[4000],f[4000]={},ans=0,j=0;
	cin>>m>>t>>r;
	for(int i=0;i<m;i++)
		cin>>a[i];
	if(t<r)
	{
		cout<<-1<<endl;
		return 0;
	}
	int now=0;
	for(int i=1;i<=300;i++)
	{
		if(f[i])
			now--;
		if(i==a[j])
		{
			for(int k=a[j]-1;now<r;k--)
			{
				ans++;
				f[k+t+1]=1;
				now++;
			}
			j++;
		}
	}
	cout<<ans<<endl;
	return 0;
}
