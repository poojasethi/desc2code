#include <bits/stdc++.h>
using namespace std;
queue<int> a;
int main()
{ int m,t,r;
	cin >> m >> t >> r; bool thanhdz=1;
	if(t<r) { cout << -1 << endl;  thanhdz=0;} int ans=0;
	for(int x=0;x<m;++x)
	{
		int now;  cin >> now;
		if(thanhdz)
		{
			while(!a.empty()&&a.front()<now) a.pop();
			int tt=r-a.size();
			while(a.size()<r) { a.push(now-tt+t); tt--; ans++; }
		}
	}
	if(thanhdz) cout << ans << endl;
}
