#include <iostream>
#include <stdio.h>
using namespace std;

int t,r,i,n,k,j,ans,a[1005],b[1005];
bool f;

int main(){
//	freopen("input.txt","r",stdin);
//	freopen("output.txt","w",stdout);
	cin >> n >> t >> r;
	if (r>t)
	{
		cout << -1;
		return 0;
	}
	for (i=0; i<n; i++)
	{
		cin >> a[i];
		k = 0;
		while (b[a[i]]<r)
		{
			for (j=a[i]; j<a[i]+t-k; j++)
				b[j]++;
			k++;
			ans++;
		}
	}	
	cout << ans;
	return 0;
}
