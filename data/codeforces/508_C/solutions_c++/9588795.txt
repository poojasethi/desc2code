// VAXES
#include <iostream>
using namespace std;

int m, t, r, c;
int a[1000], b[1000];

void light(int k, int p) {
	while (a[p] <= k + t && p < m) {
		b[p]--; p++;
	}
}
int main() {
	cin >> m >> t >> r;
	int i, j, x, y;
	for(i = 0; i < m; i++) {
		cin >> a[i];
		b[i] = r;
	}
	if (t < r) {
		cout << -1 << endl; return 0;}
	for (i =0; i < m;i++) {
		x = a[i];
		y = b[i];
		for (j = 1; j <= y; j++) {
			light(x - j, i);
			c++;
		}
	}
	cout << c << endl;
	return 0;
}
