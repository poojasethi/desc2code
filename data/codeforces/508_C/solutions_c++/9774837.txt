#include <iostream>
#include <set>
using namespace std;

int main() {
	
	int m=0,t=0,r=0;
	set<int> a;
	
	cin >> m >> t >> r;
	
	if(t < r) {
		cout << -1 << endl;
		return 0;
	}
	
	int arriveTime = 0, candle = 0;
	
	for(int i=0; i < m; i++) {
		cin >> arriveTime;
		while(!a.empty() && *a.begin() < arriveTime) {
			a.erase(a.begin());
		}
		
		while(a.size() < r) {
			a.insert(arriveTime -1 + t);
			candle++;
			arriveTime--;
		}
	}
	
	cout << candle << endl;
		
	return 0;
}
