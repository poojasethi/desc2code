#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long m,t,r;
	cin>>m>>t>>r;
	vector<long long > arr(r,-1);
	long long ans=0;
	for(int i=0;i<m;i++)
	{
		long long tim;
		cin>>tim;
		long long need=0;
		for(int j=0;j<arr.size();j++)
		{
			if(arr[j]<tim)
			{
				need++;
				arr[j]=tim-need+t;
				if(arr[j]<tim)
				{
					cout<<-1<<endl;
					return 0;
				}
				ans++;
			}
		}
		
		
		
		
	}
	
	cout<<ans<<endl;
	
	
}
