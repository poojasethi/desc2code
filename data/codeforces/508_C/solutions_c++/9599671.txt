#include<iostream>  
#include<cstdio>  
#include<cstring>  
#include<algorithm>  
using namespace std;
int w[10010],f[10010];
int main()
{
	int m,t,r,i,j,k,ans=0;
	scanf("%d %d %d",&m,&t,&r);
	for(i=0;i<m;i++) 
	scanf("%d",&w[i]);
	sort(w,w+m);
	memset(f,0,sizeof(f));
	if(t>=r)
	{
		for(i=0;i<m;i++)
	{
		for(j=w[i]-1;f[w[i]]<r;j--)
		{
			for(k=j+1;k<=j+t;k++)
			 if(k>=1) f[k]++;			
			ans++;
		}
	}
	printf("%d\n",ans);	
	}
	else
	printf("-1\n");
}
