#include <bits/stdc++.h>

using namespace std;

int n,t,r;
int f[800],l[800],a[800];

int main(){
	cin >> n >> t>> r;
	for (int i=1;i<=n;i++){
		cin >> a[i];a[i]+=300;
	}
	int bo=1;
	int ans = 0;
	for (int i=1;i<=n;i++){
		int p = r-f[a[i]];
		for (int k=0;k<p;k++){
			ans++;
			if (l[a[i]-k]) bo=0;
				else l[a[i]-k]=1;
			for (int j=0;j<t;j++)
				f[a[i]-k+j]++;
		}
		if (f[a[i]]<r) bo=0;
	}
	if (bo) cout << ans<<endl;
		else cout << -1 << endl;
	return 0;
}
