#include <iostream>
using namespace std;
int w[305],c[1000],v[1000],ans,t;
void work(int temp){
	for(int i=0;i<t;i++)c[temp+i]++;
	ans++;
}
int main(){
	int m,r;
	cin>>m>>t>>r;
	for(int i=0;i<m;i++)cin>>w[i];
	if(t<r){cout<<"-1\n";return 0;}
	for(int i=0;i<m;i++){
		int temp=w[i]+350;
		while(temp+t>350 && c[w[i]+350]<r){
			if(!v[temp]){work(temp);v[temp]=1;}
			temp--;
		}
	}
	cout<<ans<<endl;
	return 0;
}
