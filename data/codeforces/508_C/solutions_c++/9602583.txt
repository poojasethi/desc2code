#include<iostream>
using namespace std;
int a[100000],b[100000];
int main()
{
	int m,t,r;
	cin>>m>>t>>r;
	int count=0;
	while(m--)
	{
		int temp;
		cin>>temp;
		temp+=310;
		if(b[temp]<r)
		{
			int h=temp-1,k=b[temp];
			for(int j=0;j<r-k;j++)
			{
				a[j]=-1;
				count++;
				for(int k=h+1;k<=h+t;k++)
				{
					b[k]++;
				}
				h--;
			}
			if(b[temp]<r)
			{
				cout<<-1;
				return 0;
			}
		}
	}
	cout<<count;
	return 0;
}
