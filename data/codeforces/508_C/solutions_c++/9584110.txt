#include <cstdio>

int m,t,r,w,s,k;
bool uu[200009];
bool* const u=uu+100005;

int main()
{
	scanf("%d%d%d",&m,&t,&r);
	for (int i=1;i<=m;++i)
	{
		scanf("%d",&w);
		k=0;
		for (int j=w-1;j+t>=w;--j) k+=u[j];
		if (k>=r) continue;
		for (int j=w-1;j+t>=w;--j)
		{
			if (!u[j])
			{		
				u[j]=true;
				++s;
				if (++k==r) break;
			}
		}
		if (k<r) goto NO;
	}
	printf("%d\n",s);
	goto EXIT;
	NO:puts("-1");
	EXIT:return 0;
}
