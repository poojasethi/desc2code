#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#define M 310
using namespace std;
int n,t,r,ans;
int mem1[M<<2],mem2[M<<2];
int *cnt=mem1+M,*v=mem2+M;
int main()
{
	int i,j,k,x;
	cin>>n>>t>>r;
	for(i=1;i<=n;i++)
	{
		scanf("%d",&x);
		j=x;
		while(cnt[x]<r)
		{
			for(;v[j];j--);
			if(j+t<=x)
				return puts("-1"),0;
			v[j]=1;++ans;
			for(k=j;k<j+t;k++)
				cnt[k]++;
		}
	}
	cout<<ans<<endl;
	return 0;
}
