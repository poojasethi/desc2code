#include <iostream>

using namespace std;

int m, t, r, n, ii, tot, sum, a[310], b[310];

int main() {
	cin >> m >> t >> r;

	if(r > t) {
		cout << -1 << endl;
		return 0;
	}
	for(int i = 0; i < m; i++) {
		cin >> a[i];
		while(a[ii] + t < a[i]) {
			tot -= b[ii++];
		}
		n = !i ? r : max(0, r - tot + max(0, b[ii] - a[ii] + a[i] - t));
		b[i] += n;
		tot += n;
		sum += n;
	}
	cout << sum << endl;
	return 0;
}
