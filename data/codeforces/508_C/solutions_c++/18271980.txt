#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int m,t,r,sum=0,count=0;
	cin>>m>>t>>r;
	int a[r],w[m];
	for(int i=0;i<m;i++)cin>>w[i];
	for(int i=0;i<r;i++)a[i]=-1;
	for(int i=0;i<m;i++)
	{
		count=0;
		for(int j=0;j<r;j++)if(a[j]<=w[i])count++;
		sort(a,a+r);sum+=count;
		for(int j=0;j<r-count;j++)a[j]=a[j+count];
		for(int j=r-count;j<r;j++)a[j]=w[i]+t-(j-r+count);
	}
	if(t>=r)cout<<sum;else cout<<-1;
}
