#include <iostream>
#include <set>

using namespace std;

set<int> s;
int main()
{
	int n, t, r;
	cin >> n >> t >> r;
	if (t < r)
	{
		cout << -1 << endl;
		return 0;
	}
	int ans = 0;
	for (int i = 0; i < n; i++)
	{
		int now;
		cin >> now;
		while (!s.empty() && *s.begin() <= now)
			s.erase(s.begin());
		int nt = now;
		while (s.size() < r)
		{
			s.insert(nt + t);
			nt--;
			ans++;
		}
	}
	cout << ans << endl;

}
