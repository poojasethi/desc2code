import sys
import copy	
import os


def main(data):
	n = int(data.readline().strip())
	v = map(int, data.readline().strip().split())
	res = 0
	for i in range(1,n):
		if (v[i-1]-v[i]>0):
			res+= v[i-1]-v[i]
	print res


if __name__ == "__main__":
	data = sys.stdin
	if (os.path.exists('best.txt')):
		data = open('best.txt')
	main(data)
