n=input()
c=map(int, raw_input().split())+[10**9]
p=c[0]
u=True
m=0
M=p
f=0
for i in range(1,n+1):
    if c[i] < p and u:
        if m:
            f += min(M,p) - m    
        M = max(M,p)
        u=False
    if c[i] > p and not u:
        m = p
        u=True
    if c[i] != p:
        p = c[i]
if m:
    f += min(M,p) - m       

print f
