NAME = "B"
try:
    inFile = open(NAME+".txt")
except:
    pass

def read():
    try:
        return inFile.readline().strip()
    except:
        return raw_input().strip()

n = int(read())
a = map(int,read().split())
opers = 0
for i in range(1,n):
    a[i] += opers
    if a[i]<a[i-1]:
        opers += a[i-1]-a[i]
        a[i] = a[i-1]
print opers
