#!/usr/bin/env python
def count():
    n = int(raw_input())
    l = raw_input().split()
    dp = [] 
    dp.append((0,int(l[0])))
    for i in range(1,n+1):
        if int(l[i-1])<dp[i-1][1]:
            left = dp[i-1][0]+max(0,int(l[i-2])-int(l[i-1]))
            r = dp[i-1][1]
            dp.append((left,r))
        else :
            left = dp[i-1][0]
            r = int(l[i-1])
            dp.append((left,r))

    print dp[n][0]
        
        
count()
