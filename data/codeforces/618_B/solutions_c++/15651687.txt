#include<bits/stdc++.h>

using namespace std;

int n , ans[100] , x;

int main()
{
	cin >> n;
	for( int i = 1 ; i <= n ; i++ )
		for( int j = 1 ; j <= n ; j++ )
		{
			cin >> x;
			ans[i] = max( ans[i] , x );
		}
	for( int i = 1 ; i <= n ; i++ )
		if( ans[i] == n-1 )
		{
			ans[i] = n;
			break;
		}
	for( int i = 1 ; i <= n ; i++ )
		cout << ans[i] << " ";
	return 0;
}
