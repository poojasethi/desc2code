#include<iostream>
#include<algorithm>
using namespace std;
int ans[55];
int main()
{
	int n;
	cin >> n;
	int x;
	for(int i=0;i<n;i++)
	{
		int maxnum=0;
		for(int j=0;j<n;j++)
		{
			int num;
			cin >> num;
			maxnum=max(maxnum,num);
		}
		if(maxnum==n-1)
		x=i;
		ans[i]=maxnum;
	}
	ans[x]=n;
	for(int i=0;i<n;i++)
	cout << ans[i] << " ";
	return 0;
}
