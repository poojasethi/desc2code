#include <cstdio>

#define N 55

int mx[N];

main() {
	int n;
	scanf("%d", &n);
	for (int i = 1; i <= n; i ++) {
		for (int j = 1, a; j <= n; j ++) {
			scanf("%d", &a);
			if (mx[i] < a) mx[i] = a;
		}
	}
	bool bf = 0;
	for (int i = 1; i <= n; i ++) {
		if (mx[i] == n - 1) {
			if (bf) printf("%d ", n);
			else printf("%d ", n - 1);
			bf = 1;
		} else printf("%d ", mx[i]);
	}
}
