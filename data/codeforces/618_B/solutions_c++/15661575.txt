#include <bits/stdc++.h>
using namespace std;
int n;

int main()
{
	cin >> n;
	int maxi = n, a;
	for(int i = 0; i < n; i++)
	{
		int resp = 0;
		map<int, int> m;
		for(int j = 0; j < n; j++)
		{
			cin >> a;
			//cout << a << endl;
			m[a]++;
			if(m[a] == 2) resp = a;
		}
		
		if(resp) cout << resp << " ";
		else cout << maxi-- << " "; 
	}	
}
