#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

bool ok[100];

int main()
{
	int n;scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		int ma=0;
		for (int j=1;j<=n;j++)
		{
			int x;scanf("%d",&x);
			ma=max(ma,x);
		}
		if (ok[ma]) ma++;
		ok[ma]=1;printf("%d ",ma);
	}
	return 0;
}
