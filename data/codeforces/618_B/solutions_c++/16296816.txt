#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,j,a[1111]={0};
	cin>>n;
	int fl=0;
	for(i=0;i<n;i++)
	{
		int max=0;
		for(j=0;j<n;j++)
		{
			int x;
			cin>>x;
			if(x>max)
				max=x;
		}
		a[i]=max;
		if(a[i]==n-1 and fl==0)
		{
			a[i]=n;
			fl=1;
		}
	}
	for(i=0;i<n;i++)
		cout<<a[i]<<' ';
	return 0;
}
