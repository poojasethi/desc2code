#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	int a;
	int flag=0;
	int ans[55];
	int max;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		max=0;
		for(int j=1;j<=n;j++)
		{
			cin>>a;
			if(a>max) max=a;
		}
		//cout<<i<<" "<<max<<"\n";
		if(max!=n-1||flag==1) ans[i]=max;
		else{
			flag=1;
			ans[i]=n;
		}
	}
	for(int i=1;i<=n;i++)
	{
		cout<<ans[i]<<" ";
	}
	cout<<"\n";
	return 0;
}
