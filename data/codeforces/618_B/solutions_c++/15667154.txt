#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,i,j,k,z=0;
	cin>>n;
	int a[n][n],b[n]={0},c[n];
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			cin>>a[i][j];
			b[i]=max(a[i][j],b[i]);
		}
	}
	for(i=0;i<n;i++){
		if(z==1&&b[i]==n-1)
		cout<<b[i]+1<<" ";
		else
		cout<<b[i]<<" ";
		if(b[i]==n-1)
		z=1;
	}
	
	return 0;	
}
