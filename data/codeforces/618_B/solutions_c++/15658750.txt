#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,a[51]={0},flag=0;
	cin >>n;
	for(int i=0;i<n;i++)
	{
		int st=INT_MIN;	
		for(int j=0;j<n;j++)
		{
			int temp;
			cin >>temp;
			st=max(st,temp);
		}
		if(st==n-1)
		{
			if(flag==1)	
				a[i]=n;
			else
				a[i]=n-1;
			flag=1;

		}
		else
		{
			a[i]=st;
		}
	}
	for(int i=0;i<n;i++)
		cout <<a[i]<<" ";
	return 0;
}
