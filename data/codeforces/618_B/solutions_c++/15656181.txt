#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int a[n][n];
	int b[n],k=0,c[n+1]={0};
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
		cin >> a[i][j];
		k = max(k,a[i][j]);
		
		}
		b[i] = k;
		k=0;
	}
	
	for(int i=0;i<n;i++){
		c[b[i]]++;
	}
	
	for(int i=0;i<n;i++){
		cout << (b[i]+c[b[i]]-1) << " ";
		c[b[i]]--;
	}
	
	
	
	
	
	  return 0;
}
