#include <bits/stdc++.h>

using namespace std;

int x, n, a[100];



int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) {
		for (int j = 1; j <= n; ++j) {
			scanf("%d", &x);
			a[i] = max(a[i], x);
		}
	}
	for (int i = 1; i <= n; ++i) {
		if (a[i] == n - 1) {
			a[i] = n;
			break;
		}
	}
	for (int i = 1; i <= n; ++i) {
		printf("%d ", a[i]);
	}
}
