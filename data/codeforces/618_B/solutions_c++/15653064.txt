#include<bits/stdc++.h>
using namespace std;

set <int> q;
int n, x;
bool t = false;

int main(){
	scanf("%d", &n);
	for (int i = 0; i < n; ++i){
		q.clear();
		for (int j = 0; j < n; ++j){
			scanf("%d", &x);
			q.insert(x);
		}
		if (q.size() != n){
			printf("%d ", (int)q.size() - 1);
		} else {
			if (!t){
				printf("%d ", n);
				t = true;
			} else {
				printf("%d ", n - 1);
			}
		}
	}
	return 0;
}
