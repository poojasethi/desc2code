#include <iostream>
 
using namespace std;
 
int i,j,a,s[55],n,p[55],f;
 
int main(){
  cin >> n;
  for(i=0;i<n;i++) {
	  for(j=1;j<=n;j++) 
		  cin>>a, s[a]++;
	  for(j=1;j<n;j++) {
		  if(s[j]==n-j)
			  p[i]=j;
		  s[j]=0;
	  }
  }
  for(i=0;i<n;i++)
	  if(p[i]==n-1&&!f)
		  cout<<n<<' ', f=1;
	  else
		  cout<<p[i]<< ' ';
  return 0;
}
