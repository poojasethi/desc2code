#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,t,i,j;
	scanf("%d",&n);
	int a[51]={0};
	for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		{
			scanf("%d",&t);
			a[i]+=t;
		}
	int ans[n];
	int k=1;
	for(i=0;i<n;i++)
	{
		int min = 0;
		for(j=0;j<n;j++)
		{
			if( a[j] < a[min] )
				min = j;
		}
		ans[min]=k++;
		a[min]=50000;
	}
	for(i=0;i<n;i++)
		printf("%d ",ans[i]);
	printf("\n");
	return 0;
}
