#include <iostream>

using namespace std;

int p[50][50];
int b[50], a[50];

int main() {
  int n;
  cin >> n;
  for (int i = 0; i < n; i++)
    for (int j = 0; j < n; j++) {
      cin >> p[i][j];
      b[j] = max(b[j], p[i][j]);
    }

  for (int i = 0; i < n; i++) {
    a[b[i]]++;
    if (a[b[i]] > 1)
      b[i] = n;
    cout << b[i] << " ";
  }

  return 0;
}
