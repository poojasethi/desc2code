#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,arr[60][60];
	cin>>n;
	int row[n],col[n],mx,mp[n+1];
	for(int i=0;i<=n;i++)
	mp[i]=0;
	for(int i=0;i<n;i++){
		mx=-1;
		for(int j=0;j<n;j++){
			cin>>arr[i][j];
			if(arr[i][j]>mx)
			mx=arr[i][j];
		}
		while(mp[mx])
		mx++;
		row[i]=mx;
		mp[mx]=1;
	}
	
	for(int i=0;i<n;i++){
		cout<<row[i]<<" ";
	}
	return 0;
}
