#include <bits/stdc++.h>
using namespace std;

int M[50];

int main() {
	int n;
	cin >> n;
	bool bylo = false;
	for(int i = 0; i < n; i++) {
		int mx = 0;
		for(int j = 0; j < n; j++) {
			scanf("%d", &M[j]);			
			mx = max(mx, M[j]);
		}
		if(mx < n-1) printf("%d ", mx);
		else {
			if(bylo) printf("%d ", n);
			else printf("%d ", mx);
			bylo = true;
		}
	}			
}
