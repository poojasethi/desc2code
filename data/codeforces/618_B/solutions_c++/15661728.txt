#include "bits/stdc++.h"
using namespace std;

int n,p[54],a[54][54];
int _i,_j;

int main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++){
			cin>>a[i][j];
			if(a[i][j]==n-1)
				_i=i,_j=j;
		}
	}
	p[_i]=n-1,p[_j]=n;
	for(int i=1;i<=n;i++){
		if(i!=_i&&i!=_j)
			p[i]=a[i][_i];
	}
	for(int i=1;i<=n;i++)
		cout<<p[i]<<" ";cout<<endl;
	return 0;
}
