#include<iostream>
using namespace std;
#define ll long long
ll a[60][60];
int main()
{
	ll n;
	cin>>n;
	bool flag=true;
	for(ll i=0;i<n;i++)
	{
		ll max=0;
		for(ll j=0;j<n;j++)
		{
			cin>>a[i][j];
			if(a[i][j]>max)
			max=a[i][j];
		}
		if(max==n-1&&flag)
		{
			cout<<n<<" ";
			flag=false;
		}
		else
		cout<<max<<" ";
	}
}
