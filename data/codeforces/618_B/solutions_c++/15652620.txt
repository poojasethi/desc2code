#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;

int main() {
	int n;
	cin >> n;

	int input[n][n];
	int r = 0;
	for(int i=0; i<n; i++) {
		for(int j=0; j<n; j++) {
			int a;
			cin >> a;
			input[i][j] = a;
			if(a == 0) {
				input[i][j] = n;
			}
			if(a == n-1) {
				r = i;
			}
		}
	}

	for(int k=0; k<n; k++) {
		cout << input[r][k] << " ";
	}
}
