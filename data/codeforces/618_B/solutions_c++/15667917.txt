#include <iostream>
using namespace std;

int main() {
	long long n, x;
	long long p[100]={0};
	long long flag = 0;
	cin>>n;
	for(long long i=1; i<=n; i++)
	{
		for(long long j=1; j<=n; j++)
		{
			cin>>x;
			if(i != j)
			{
				p[i] = max(p[i], x);
			}
			if(p[i] == n-1 && flag == 0)
			{
				flag = 1;
				p[i] = n;
			}
		}
	}
	for(long long i=1; i<=n; i++)
	{
		cout<<p[i]<<" ";
	}
	return 0;
}
