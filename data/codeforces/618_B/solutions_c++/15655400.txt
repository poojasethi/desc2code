
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,i,j,t,s;
cin>>n;
vector <pair<int,int> > v(n);
vector <int> u(n);

for(i=0;i<n;i++)
	{s=0;
	for(j=0;j<n;j++)
		{cin>>t;s+=t;}
	v[i].first=s;
	v[i].second=i;
	}
sort(v.begin(),v.end());
for(i=0;i<n;i++)
	u[v[i].second]=i+1;

for(i=0;i<n;i++)
	cout<<u[i]<<' ';
cout<<endl;

}
