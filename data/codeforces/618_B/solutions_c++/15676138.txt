#include<iostream>

using namespace std;

int main(){

	int n,x,temp = 0,p  = 0;
	cin>>n;
	int a[n];
	for(int i = 0; i<n; i++){
	
	temp = 0;
		for(int j = 0; j<n; j++){
		cin>>x;
			if(x>temp){
			temp = x;
			}
		
		}
		a[p] = temp;
		p++;
	
	}
	
	for(int i = 0; i<n; i++){
		if(a[i] == n-1){
		a[i]++;
		break;
		}
	}
	for(int i = 0; i<n; i++){
		cout<<a[i]<<" ";
	}
	
	

}
