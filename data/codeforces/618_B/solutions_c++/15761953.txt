#include <cstdio>
#include <algorithm>

using namespace std;

int a[51];

int main()
{
	int n,in;

	scanf("%d",&n);

	for(int i=0; i<n; i++) {
		for(int j=0; j<n; j++) {
			scanf("%d",&in);
			a[i] = max(a[i],in);
		}
	}

	for(int i=0; i<n; i++) {
		if(a[i] == n-1) {
			a[i] = n;
			break;
		}
	}

	for(int i=0; i<n; i++) 
		printf("%s%d",i?" ":"",a[i]);

	return 0;
}
