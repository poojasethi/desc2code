#include <bits/stdc++.h>
using namespace std;
bool all[55];
int n;
bool check(){
	int i;
	for(i=0;i<n;++i)
		if(!all[i])return false;
	return true;
}
int ar[55];
int main(){
	int j,i;cin>>n;
	for(i=0;i<n;++i){
		memset(all,false,sizeof(all));
		for(j=0;j<n;++j)
			cin>>ar[j],all[ar[j]]=true;
		if(check()){
			ar[i]= n;
			break;
		}
	}
	for(i=0;i<n;++i)
		cout<<ar[i]<<" ";
	cout<<endl;
}
