#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,a[100010],x;
	scanf("%d",&n);
	for (int i=1; i<=n; i++){
		int y=0;
	    for (int j=1; j<=n; j++){
		    scanf("%d",&x);
		    if (x>y) y=x;
		}
		if (y<=n-2) a[i]=y;
	}
    for (int i=1; i<=n; i++) if (a[i]==0) {a[i]=n-1; break;}
	for (int i=1; i<=n; i++) if (a[i]==0) a[i]=n;
	for (int i=1; i<=n; i++) printf("%d ",a[i]);
}
