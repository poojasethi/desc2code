#include <iostream>
using namespace std;

int main() {
  int n, a[55];
  cin >> n;
  for (int i = 0; i < n; i++) {
    int flag = 0;
    for (int j = 0; j < n; j++) {
      cin >> a[j];
      flag |= (a[j]==n-1);
    }
    if (flag) {
      for (int j = 0; j < n; j++)
        cout << (a[j]==0? n : a[j]) << ' ';
      break;
    }
  }
  cout << endl;
  return 0;
}
