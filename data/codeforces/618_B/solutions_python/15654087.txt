import math
from itertools import *
def bigest(x):
    return int(math.log(x,2))

n= int(raw_input())
m = []
for i in xrange(n):
    m.append(map(int,raw_input().split()))
    
done = [0]*n
for i, q in enumerate(m):
    for c in q:
        if sum([1 for x in q if x==c])>1:
            done[i]=c
    
pom=n-1
for i in xrange(len(done)):
    if done[i]==0:
        done[i]=pom
        pom+=1

print " ".join(map(str,done))
