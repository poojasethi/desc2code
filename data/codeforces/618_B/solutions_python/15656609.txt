from collections import Counter
n = input()
mat = []
op = []
val = 0
fg = False 
for i in xrange(n):
    row = map(int,raw_input().split())
    c = Counter(row)
    mxkey = max(c.values())
    for k in c.keys():
        if c[k] == mxkey and k != 0:
            val = k
            break
    if val == 1 and c[val] == 1 and not fg:
        val = n-1
        fg = True
    elif val == 1 and c[val] == 1 and fg:
        val = n
    op.append(val)
for v in op:
    print v,
