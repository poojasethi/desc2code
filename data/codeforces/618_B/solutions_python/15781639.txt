from collections import Counter

__author__ = 'aste'


def main():
    n = int(raw_input())
    p = [0] * n
    current = n - 1
    for i in range(0, n):
        c = Counter(map(int, raw_input().split()))
        (a, b) = c.most_common(1)[0]
        if b == 1:
            p[i] = current
            current += 1
        else:
            p[i] = a
    print " ".join([str(x) for x in p])

main()
