def solve():
    n=input()
    l=[0]*(n+1)
    m=[0]*(n+1)
    for x in xrange(1,n+1):
        a=map(int,raw_input().split())
        a.sort()
        for y in xrange(1,n):
            if a[y]==a[y-1]:
                l[x]=a[y]
                m[a[y]]=1
                break
    temp=[]
    for x in xrange(1,n+1):
        if m[x]==0:
            temp.append(x)
    i=0
    for x in xrange(1,n+1):
        if l[x]==0:
            l[x]=temp[i]
            i+=1
        print l[x],

solve()
