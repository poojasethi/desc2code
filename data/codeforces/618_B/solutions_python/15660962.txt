n = int(raw_input())

mat = []
for x in range(n):

    mat.append(map(int,raw_input().split()))

t_mat = []
for x in range(n):

    t_mat.append([mat[y][x] for y in range(n)])


li = []
i=n-1
for x in t_mat:
    l =[]
    for y in range(1,n+1):
        l.append((y,x.count(y)))

    l.sort(key=lambda x: x[1],reverse=True)
    if len(set(x))!=len(x):
        li.append(l[0][0])
    else:
        li.append(i)
        i+=1
