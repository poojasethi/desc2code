n=input()
mat = []
res = {}
vis = {}
for i in range(n):
    v=map(int,raw_input().split())
    mat.append(v)
bar = 1
while bar < n+1:
    for i in range(len(mat)):
        if i not in vis:
            flag = 1
            for j in mat[i]:
                if j > bar:
                    flag = 0
                    break
            if flag:
                res[i] = bar
                vis[i] = 1
                break
    bar += 1
