n = int(raw_input())

x = [0] * n
used = []
for i  in range(0,n):
    temp = map(int , raw_input().split())
    for j in range(1,n+1):
        if temp.count(j) > 1 :
            x[i]=j
            used.append(j)
for i in range(1,n+1):
    if used.count(i) == 0:
        for j in range(0,n):
            if x[j] == 0 :
                x[j] = i
                break
print " ".join(map(str , x))
