def sat(n,arr):
	for item in arr:
		if item>n:
			return False
	return True
n = input()
arr = []
for i in range(n):
	arr.append(map(int,raw_input().split()))
new = {}
help = [False]*(n)
for i in range(1,n+1):
	ind = i
	for j in range(n):
		item = arr[j]
		if sat(ind,item) and not help[j]:
			new[j] = ind
			help[j] = True
			break
