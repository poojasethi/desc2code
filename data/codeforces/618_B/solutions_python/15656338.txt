import sys

n = int(sys.stdin.readline())

l = [list(map(int,sys.stdin.readline().split()))    for i in xrange(n)]

ans = [0    for i in xrange(n)]
ma = n

for i in xrange(n):
    a = set()
    for j in xrange(n):
        if i != j:
            if l[i][j] in a:
                ans[i]= l[i][j]
                a = set()
                break
            else:
                a.add(l[i][j])
    if len(a) > 0:
        ans[i] = ma
        ma -= 1
