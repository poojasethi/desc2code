n = input()
ans=[0]*n
for i in xrange(n):
    a= map(int,raw_input().split())
    for j in xrange(n):
        if j!=i:
            if ans[j]==0:
                ans[j]=a[j]
            else:
                ans[j]=max(ans[j],a[j])

m=max(ans)
i = ans.index(m)
ans[i]+=1
ans=map(str,ans)
print " ".join(ans)
