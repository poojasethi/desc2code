n = input()
m=[]

for i in range(n):
        m.append([int(j) for j in raw_input().split(' ')])
find = 1
a=[0]*n
while find<=n:
        for j,row in enumerate(m):
                if find == max(row):
             
                        a[j]=find
                        break
        find+=1
for i in range(n):
        if a[i]==0:
                a[i]=str(n)
        else:
                a[i]=str(a[i])
        
print " ".join(a)
