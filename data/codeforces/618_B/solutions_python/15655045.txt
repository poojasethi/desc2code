#!/usr/bin/python

import sys

def Ni(): return tuple(map(int, sys.stdin.readline().split()))
def Nil(): return map(int, sys.stdin.readline().split())

n = Ni()[0]
a = []
for i in range(n):
    a.append(Nil())

s = [None] * n
for i in range(n):
    for j in range(i, n):
        if a[i][j] == n-1:
            s[i] = n-1
            s[j] = n

have = 2
while have < n:
    for i in range(n):
        for j in range(i, n):
            # a[i][j] = min(s[i], s[j])
            if s[i] == None and s[j] != None and a[i][j] < s[j]:
                s[i] = a[i][j]
                have += 1
            if s[j] == None and s[i] != None and a[i][j] < s[i]:
                s[j] = a[i][j]
                have += 1
    
print " ".join(map(str, s))
