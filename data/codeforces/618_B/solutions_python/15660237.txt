# -*- coding: utf-8 -*-

n = int(raw_input())
matrix = []

for i in xrange(n):
    line = raw_input()
    matrix.append(map(int, line.split(' ')))
    
for i in xrange(n):
    for j in xrange(n):
        if matrix[j][i] == 0:
            matrix[j][i] = n

right_ans = int((n*(n+1.0)/2.0))
ans = []
for line in matrix:
    if sum(line) == right_ans:
        ans = ''
        for i in line:
            ans = '{0} {1}'.format(ans, i)
        print ans
        # print once
        break
