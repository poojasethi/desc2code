from sys import stdin, stdout
import json


n = int(stdin.readline().rstrip())
max = n
matrix = []

for i in xrange(0, n):
    row = [int(x) for x in stdin.readline().rstrip().split()]
    matrix.append(row)

combination = [-1] * n

for i in range(0, n):
    d = {}

    for j in range(0, n):
        item = matrix[j][i]
        if item in d:
            combination[i] = item
            break
        d[item] = True

    if combination[i] == -1:
        combination[i] = max
        max -= 1

print " ".join(str(x) for x in combination)
