#!/usr/bin/env python
import math, sys, collections, heapq

n = int(raw_input())
a = [map(int, raw_input().split()) for _ in range(n)]

unused = set(range(1, n+1))
freq = sorted([(i, j, a[j].count(i)) for i in range(n) for j in range(n)], key=lambda x: x[2])
res = [0]*n
while freq:
    num, row, count = freq.pop()
    if num not in unused: continue

    res[row] = num
    freq = [(a,b,c) for a,b,c in freq if b != row]
    unused.remove(num)

print " ".join(map(str, [a if a != 0 else n for a in res]))
