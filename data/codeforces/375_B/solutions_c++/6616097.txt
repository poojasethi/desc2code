#include <cstdio>
#include <algorithm>
using namespace std;

int N,M,f[5005][5005],a[5005];
char s[5005][5005];

int main()
{
	scanf("%d%d",&N,&M);
	for (int i=1; i<=N; i++) scanf("%s",s[i]+1);
	for (int i=1; i<=N; i++)
	{
		f[i][M+1]=M;
		for (int j=M; j; j--) f[i][j]=s[i][j]=='1'?f[i][j+1]:j-1;
	}
	int ans=0;
	for (int j=1; j<=M; j++)
	{
		for (int i=0; i<=M; i++) a[i]=0;
		for (int i=1; i<=N; i++) a[f[i][j]-j+1]++;
		int s=0;
		for (int i=M,t=0; i; i--) t+=a[i],s=max(s,i*t);
		ans=max(ans,s);
	}
	printf("%d\n",ans);
}
