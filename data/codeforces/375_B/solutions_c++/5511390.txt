#include <algorithm>
#include <cstdio>
#include <iostream>
#include <string>
using namespace std;
const int N=5100;
int n,m,i,j,k,ans,a[N][N],f[N];
char s[N][N];
int main() {
    scanf("%d%d",&n,&m);
    for (i=1;i<=n;i++) scanf("%s",s[i]);
    for (j=m-1;j>=0;j--) {
        for (i=1;i<=n;i++)
            if (s[i][j]=='0') f[i]=a[i][j]=0;
            else f[i]=a[i][j]=a[i][j+1]+1;
        sort(f+1,f+n+1,greater<int>());
        for (i=1;i<=n;i++) ans=max(ans,f[i]*i);
    }
    printf("%d\n",ans);
    return 0;
}
