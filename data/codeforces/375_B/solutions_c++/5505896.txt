#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define rep(i,n) for (int i=1;i<=n;++i)
const int N=5005;
char s[N]; int n,m,ans,R,c[N],r[N][N];
int main()
{
	scanf("%d%d",&n,&m);
	rep(i,n){
		scanf("%s",s+1),R=m+1;
		for (int j=m;j;--j) if (s[j]=='1') r[i][j]=R-1; else R=j;
	}
	rep(j,m){
		rep(i,m) c[i]=0; int t=0;
		rep(i,n) if (r[i][j]) ++c[r[i][j]],++t;
		for (int i=j;i<=m && t;++i)
			ans=max(ans,(i-j+1)*t),t-=c[i];
	}
	printf("%d\n",ans);
	return 0;
}
