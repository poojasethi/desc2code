#include <bits/stdc++.h>
using namespace std;

#define N 5003

int dp[N][N];
char arr[N][N];
int main()
{
	int n , m;
	cin >> n >> m;

	for(int i = 0; i<n; i++)
	{
		scanf("%s",arr[i]);
	}

	for(int j = 0; j<n; j++)
	{
		dp[m][j] = 0;
		for(int i = m-1; i>=0; i--)
		{
			if(arr[j][i]=='1')
				dp[i][j] = dp[i+1][j] + 1;
			else
				dp[i][j] = 0;
		}
	}

	int ans = 0;

	for(int i = 0; i<m; i++)
	{
		sort(dp[i], dp[i]+n);
		for(int j = 0; j<n; j++)
		{
			if(dp[i][j]*(n-j)>ans)
				ans = dp[i][j]*(n-j);
		}
	}

	cout << ans << endl;
}
