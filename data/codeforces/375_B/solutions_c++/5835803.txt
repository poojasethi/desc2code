//In the name of God
#include <iostream>
using namespace std;
const int N = 5005;

int a[N], h[N], ans, n, m;
string mat[N];

int main() {
	ios_base::sync_with_stdio(false);
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		cin >> mat[i];
	for (int i = m - 1; i >= 0; i--) {
		fill(a, a + m, 0);
		for (int j = 0; j < n; j++)
			a[(mat[j][i] == '0'? h[j] = 0: ++h[j])]++;
		for (int j = m, sum = 0; j >= 0; j--)
			ans = max(ans, (sum += a[j]) * j);
	}
	cout << ans << '\n';
	return 0;
}
