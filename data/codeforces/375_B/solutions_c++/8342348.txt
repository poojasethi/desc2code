#include<cstdio>
#include<algorithm>
#define N 5005
using namespace std;
int n,m,i,j,f[N],x[N],a[N][N],ans;
char s[N];
int main()
{
  scanf("%d%d",&n,&m);
  for (i=1;i<=n;i++)
  {
    scanf("%s",s+1);
    for (j=1;j<=m;j++)
      a[j][i]=s[j]-'0';
  }
  int t=n;n=m;m=t;
  for (i=1;i<=m;i++) f[i]=a[1][i],ans+=(f[i]==1);
  for (i=2;i<=n;i++)
  {
    for (j=1;j<=m;j++)
    {
      if (!a[i][j]) f[j]=0;else f[j]++;
      x[j]=f[j];
    }
    sort(x+1,x+m+1);
    for (j=m;j;j--)
      ans=max(ans,(m-j+1)*x[j]);
  }
  printf("%d",ans);
}
