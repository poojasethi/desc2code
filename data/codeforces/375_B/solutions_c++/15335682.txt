#include<bits/stdc++.h>
using namespace std;
int s[5005][5005],arr[5005];
char a[5005][5005];
int main()
{
int n,m,i,j,maxi=0;
scanf("%d %d",&n,&m);
for(i=0;i<n;i++)
{
scanf("%s",a[i]);
for(j=m-1;j>=0;j--)
s[i][j]=(a[i][j]=='1'?s[i][j+1]+1:0);
}

for(i=0;i<m;i++)
{
for(j=0;j<n;j++)
{
arr[j]=s[j][i];
}
sort(arr,arr+n);
for(j=n-1;j>=0;j--)
maxi=max(maxi,arr[j]*(n-j));
}
printf("%d\n",maxi);
return 0;
}
