#include <bits/stdc++.h>

using namespace std;

char s[6001];
int c[6001][6001],n,m;
int main(int argc, char const *argv[]){

	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++){
		scanf("%s",s);
		int now=0;
		for(int j=m-1;j>=0;j--){
			if(s[j]=='1') ++now;
			else now=0;
			++c[j][now];
		}
	}
	int ans=0;
	for(int i=0;i<m;i++){
		int now=0;
		for(int j=m;j>=0;j--){
			now+=c[i][j];
			if(ans<now*j) ans=now*j;
		}
	}
	printf("%d\n",ans);
	return 0;
}
