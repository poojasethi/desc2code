#include <iostream>
#include <algorithm>

using namespace std;

const int MAXN = 5010;

int N, M;
int dp[MAXN][MAXN];

int main() {
  cin >> N >> M;
  for (int i = 0; i < N; ++i) {
    string in; cin >> in;
    int len = 0;
    for (int k = 1; k <= M; ++k) {
      if (in[k - 1] == '0') {
        len = 0;
      } else {
        ++dp[k][++len];;
      }
    }
  }
  int ans = 0;
  for (int i = 1; i <= M; ++i) {
    for (int k = M; k > 0; --k) {
      ans = max(ans, dp[i][k] * k);
      dp[i][k - 1] += dp[i][k];
    }
  }
  cout << ans << endl;
  return 0;
}
