#include <cstdio>

int n,m,tmp,ans,f[5003][5003];
bool c[5003][5003];

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i)
	{
		getchar();
		for (int j=1;j<=m;++j) c[i][j]=getchar()=='1';
	}
	for (int i=1;i<=n;++i) for (int j=1;j<=m;++j) if (c[i][j])
	{
		int k;
		for (k=j;c[i][k];++k);
		++f[j][k-1];
		j=k;
	}
	for (int i=1;i<=m;++i) for (int j=m;j>=i;--j)
	{
		f[i][j]+=f[i-1][j]+f[i][j+1]-f[i-1][j+1];
		tmp=f[i][j]*(j-i+1);
		if (ans<tmp) ans=tmp;
	}
	printf("%d",ans);
	return 0;
}
