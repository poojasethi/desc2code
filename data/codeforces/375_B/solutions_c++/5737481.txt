#include <cstdio>
#include <cstring>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

const int N = 5002;

int n, m, c[N][N];
string s;

int main() {
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		cin >> s;
		for (int t = 0, j = 0; j < m; j++) {
			if (s[j] == '1') t++;
			else t = 0;
			c[j][t]++;
		}
	}

	int ans = 0;
	for (int i = 0; i < m; i++)
		for (int j = m; j >= 1; j--) {
			c[i][j] += c[i][j + 1];
			int area = j * c[i][j];
			if (area > ans) ans = area;
		}

	cout << ans << endl;
}
