#include<cstdio>
#include<algorithm>
using namespace std;
const int X=5010;
char s[X][X];
int a[X],sum[X][X];
int main(){
    int i,j,n,m,as=0;
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;i++)
        scanf("%s",s[i]+1);
    for(i=1;i<=m;i++){
        for(j=1;j<=n;j++)
            a[j]=sum[j][i]=s[j][i]=='1'?sum[j][i-1]+1:0;
        sort(a+1,a+1+n);
        for(j=1;j<=n;j++)
            as=max(as,a[j]*(n-j+1));
    }
    printf("%d\n",as);
    return 0;
}
