#include<bits/stdc++.h>

#define FP( ii,aa,bb )	for( int ii=aa;ii<=bb;ii++ )
#define FM( ii,aa,bb )	for( int ii=aa;ii>=bb;ii-- )

using namespace std;

int m,n,res,s[6000],a[6000][6000];

char str[6000];

int main(){
	char x;
	int  k;
	scanf("%d %d",&m,&n);
	FP( i,1,m ){
		scanf("%s",str+1);
		FP( j,1,n )
			if( str[j]-'0' )	a[i][j] = a[i][j-1]+1;
	}
	FP( i,1,n ){
		FP( j,0,n )	s[j] = 0;
		FP( j,1,m )	s[ a[j][i] ]++;
		k = 0;
		FM( j,n,1 ){ k += s[j]; res = max( res,k*j ); }
	}
	printf("%d\n",res);
}
