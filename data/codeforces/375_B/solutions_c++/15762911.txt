#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char ch[5101];
int i,j,k,n,m,ans,p[5101][5101],a[5101];
int main(){
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++){
		scanf("%s",ch+1);
		for (p[i][m]=(ch[m]=='1')?m+1:m,j=m-1;j;j--) p[i][j]=(ch[j]=='1')?p[i][j+1]:j;
	}
	for (i=1,ans=0;i<=m;i++){
		memset(a,0,sizeof(a));
		for (j=1;j<=n;j++) a[p[j][i]]++;
		for (j=m+1,k=0;j>i;j--) k+=a[j],ans=max(ans,k*(j-i));
	}
	printf("%d\n",ans);
	return 0;
}
