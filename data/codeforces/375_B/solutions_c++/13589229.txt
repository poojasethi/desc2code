#include<bits/stdc++.h>

using namespace std;

#define N 5003

int M[N][N],B[N],m,n,i,j,s,mx;

char A[N][N];

int main(){

	cin >> m >> n;

	for(i=1;i<=m;i++)
		scanf("%s",A[i] + 1);

	for(i=m; i ;i--)
		for(s=0,j=n; j ;j--){
			if(A[i][j] == '1') s++;
			else s = 0;
			M[i][j] = s;
		}

	for(j=1;j<=n;j++){
		for(i=1;i<=m;i++)
			B[i] = M[i][j];
		
		sort(B + 1, B + m + 1);
		reverse(B+1,B+m+1);

		for(s=0,i=1;i<=m;i++)
			s = max(s , i*B[i]);

		mx = max(mx , s);
	}    

	cout << mx;
	return 0;
}
