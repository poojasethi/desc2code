#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
#define REP(i,n) for(int i = 0 ; i < n ; i++)
#define MAX 5005
int n, m , x;
char C[MAX];
vector<int> A[MAX];
 
int main(){
 scanf("%d%d",&n,&m);
 REP(i,n) {
   scanf("%s",C);
   x = 0;
   REP(j,m) {
      if (C[j] == '1') x++;
      else x = 0;
      A[j].push_back(-x);
   }
 }
 int r = 0;
 REP(i,m) {
   sort(A[i].begin(),A[i].end());
   REP(j,n) r = max(r, (-A[i][j]) * (j+1));
 }
 printf("%d\n",r);
 return 0;
}
