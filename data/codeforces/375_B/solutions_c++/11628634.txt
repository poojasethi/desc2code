#include<cstdio>
#include<algorithm>
using namespace std;

int n, m, ans;
char g[5555][5555];
int s[5555][5555], arr[5555];

int main() {
	scanf("%d%d", &n, &m);
	for(int i=0;i<n;i++) {
		scanf("%s",&g[i]);
		for(int j=m-1;j>=0;j--)
			s[i][j] = (g[i][j] == '1') ? s[i][j+1] + 1 : 0;
	}
	
	ans = 0;
	for(int i=0;i<m;i++) {
		for(int j=0;j<n;j++)
			arr[j] = s[j][i];
		sort(arr, arr+n);
		for(int j=n-1;j>=0;j--)
			ans = max(ans, arr[j] * (n-j));
	}
	printf("%d\n", ans);
	
	return 0;
}
