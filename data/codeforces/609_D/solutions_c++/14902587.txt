#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

ll n , m , s , k ;
ll a[211111] , b[211111] , c[211111] , t[211111];


bool check(int X){
int A=0 , B=0 ;
for(int i=1;i<X;i++){
   if( a[A]>a[i] ){A=i;}
   if( b[B]>b[i] ){B=i;}
}
vector<ll> OO;
for(int i=0;i<m;i++)
    OO.push_back( ((t[i]==1)?(a[A]*c[i]):(b[B]*c[i])) );

sort(OO.begin(),OO.end());

ll d=0;
for(int i=0;i<k;i++)
d+=OO[i];

if( d<=s )return true;
          return false;
}

void print(int X){
printf("%d\n",X);
//cout<<X<<endl;

int A=0 , B=0 ;
for(int i=1;i<X;i++){
   if( a[A]>a[i] ){A=i;}
   if( b[B]>b[i] ){B=i;}
}
vector<pair<ll,int> > OO;
for(int i=0;i<m;i++)
    OO.push_back( make_pair(((t[i]==1)?(a[A]*c[i]):(b[B]*c[i])), i+1 ));

sort(OO.begin(),OO.end());

for(int i=0;i<k;i++)
//cout<<(OO[i].second)<<" "<<1+((t[ OO[i].second-1 ]==1)?A:B)<<endl;
printf("%d %d\n",OO[i].second,1+((t[ OO[i].second-1 ]==1)?A:B));
}
int main()
{
   cin>>n>>m>>k>>s;

   for(int i=0;i<n;i++)scanf("%l64d",&a[i]);
   for(int i=0;i<n;i++)scanf("%l64d",&b[i]);

   for(int i=0;i<m;i++)scanf("%l64d %l64d",&t[i],&c[i]);

   int low=1 , mid , high=n+1;
   while( low < high ){

    mid=(low+high)/2;

    if( check(mid) )high=mid;
    else low=mid+1;

   }
   if( high<n+1 ){
    print(high);
   }else{printf("-1\n");}

    return 0;
}
