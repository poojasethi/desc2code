#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

#define all(a) (a).begin(),(a).end()
#define pb push_back
#define sz(a) ((int)(a).size())
#define mp make_pair
#define fi first
#define se second

typedef pair<int,int> pint;
typedef long long ll;
typedef vector<int> vi;


#define MAX_N 200005

int n,m,k,s;
int a[MAX_N],b[MAX_N];
int t[MAX_N],c[MAX_N];
pair<ll,int> can[MAX_N];
int da,db;

bool f(int x)
{
    int ca=1234567,cb=1234567;
    for (int i=1; i<=x; i++)
    {
        if (a[i]<ca)
            ca=a[i],da=i;
        if (b[i]<cb)
            cb=b[i],db=i;
    }
    for (int i=1; i<=m; i++)
        if (t[i]==1)
            can[i]=mp(ca*1LL*c[i],i);
        else
            can[i]=mp(cb*1LL*c[i],i);
    sort(can+1,can+1+m);
    ll sum=0;
    for (int i=1; i<=k; i++)
        sum+=can[i].fi;
    return sum<=s;
}

int main()
{
    scanf("%d %d %d %d",&n,&m,&k,&s);
    for (int i=1; i<=n; i++)
        scanf("%d",a+i);
    for (int i=1; i<=n; i++)
        scanf("%d",b+i);
    for (int i=1; i<=m; i++)
        scanf("%d %d",t+i,c+i);
    if (!f(n))
    {
        printf("-1\n");
        return 0;
    }
    int lo=1,hi=n,m;
    while (lo<hi)
    {
        m=(lo+hi)/2;
        if (f(m))
            hi=m;
        else
            lo=m+1;
    }
    printf("%d\n",lo);
    f(lo);
    for (int i=1; i<=k; i++)
        if (t[can[i].se]==1)
            printf("%d %d\n",can[i].se,da);
        else
            printf("%d %d\n",can[i].se,db);
    return 0;
}
