#include <cstdio>
#include <stack>
#include <vector>
#include <climits>
#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
int n,m,Ned;
struct node{long long v;int t,d;}b[200050];
long long s,Val[200050];
int a[3][200050],t[200050];
bool cmp(node a,node b)
{
    return a.v<b.v;
}
bool Check(int G)
{
    int mx[3];
    mx[1]=mx[2]=1;
    for(int i=2;i<=G;i++)
        if(a[1][i]<a[1][mx[1]])
            mx[1]=i;
    for(int i=2;i<=G;i++)
        if(a[2][i]<a[2][mx[2]])
            mx[2]=i;
    for(int i=1;i<=m;i++)
        b[i]=(node){a[t[i]][mx[t[i]]]*Val[i],i,mx[t[i]]};
    sort(b+1,b+m+1,cmp);
    long long tmp=0;
    for(int i=1;i<=Ned;i++)
        tmp+=b[i].v;
    return tmp<=s;
}
void ef(int l,int r)
{
    int mid;
    while(l<=r)
    {
        mid=(l+r)/2;
        if(Check(mid))
            r=mid-1;
        else
            l=mid+1;
    }
    if(l>n)
        printf("-1\n");
    else
    {
        printf("%d\n",l);
        Check(l);
        for(int i=1;i<=Ned;i++)
            printf("%d %d\n",b[i].t,b[i].d);
    }
}
int main()
{
    scanf("%d %d %d %I64d",&n,&m,&Ned,&s);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[1][i]);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[2][i]);
    for(int i=1;i<=m;i++)
        scanf("%d %I64d",&t[i],&Val[i]);
    ef(1,n+1);
}
