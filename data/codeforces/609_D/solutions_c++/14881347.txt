#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <cmath>
#include <map>
#include <vector>
#include <set>
#include <iomanip>
using namespace std;

const int size = 210000;
typedef long long ll;
typedef  double dd;
const int inf=1111111111;
typedef pair<ll,int> pp;

int n,m,k,s;
int a[size],b[size],t[size],c[size],aa[size],bb[size];
pp x[size];
int q[size],d[size];

int work(int p){
    for(int i=0;i<m;i++){
        if(t[i]==1) x[i].first=1ll*a[p]*c[i];
        else x[i].first=1ll*b[p]*c[i];
        x[i].second=i;
    }
    sort(x,x+m);
    int tot=0;
    int sum=s;
    for(int i=0;i<m&&sum>=x[i].first;i++){
        sum-=x[i].first;
        q[tot]=x[i].second;
        if(t[q[tot]]==1)d[tot]=aa[p];
        else d[tot]=bb[p];
        tot++;
    }
    return tot>=k;
}

int main(){
    cin>>n>>m>>k>>s;
    for(int i=1;i<=n;i++){
        scanf("%d",&a[i]);aa[i]=i;
        if(i>1&&a[i]>a[i-1]){
            a[i]=a[i-1];
            aa[i]=aa[i-1];
        }
    }
    for(int i=1;i<=n;i++){
        scanf("%d",&b[i]);bb[i]=i;
        if(i>1&&b[i]>b[i-1]){
            b[i]=b[i-1];
            bb[i]=bb[i-1];
        }
    }
    for(int i=0;i<m;i++)scanf("%d%d",&t[i],&c[i]);
    int l=1,h=n+1,mid;
    while(h>l){
        mid=l+h>>1;
        if(work(mid))h=mid;
        else l=mid+1;
    }
    if(l==n+1)cout<<-1<<endl;
    else {
        work(l);
        cout<<l<<endl;
        for(int i=0;i<k;i++)printf("%d %d\n",q[i]+1,d[i]);
    }
}
