#include <bits/stdc++.h>
using namespace std;
int n,m,k,s,RR=1e9;
const int N = 2e5+5;
int a[N],b[N];
vector<pair<int,int> > cost[2],ret;
vector<long long> total[2];
int lx,ly,xr,yr;
bool check(int day){
	int d=day,p = day,i;
	for(i=day;i>=1;--i){
		if(a[i]<a[d])d= i;
		if(b[i]<b[p])p=i;
	}
	int x=0,y=k,j;
	for(;x<total[0].size()&&y>=0;++x,--y){
		if(y>=total[1].size())continue;
		if( total[0][x]*a[d]+ total[1][y]*b[p]<=s){
			lx=x,ly= y;
			xr=d,yr=p;
			RR= min(RR,max(p,d));
			return true;
		}
	}
	return false;
}
int main(){
	int t,c,i,j;cin>>n>>m>>k>>s;
	for(i=1;i<=n;++i)scanf("%d",&a[i]);
	for(i=1;i<=n;++i)scanf("%d",&b[i]);
	for(i=0;i<m;++i)
		scanf("%d%d",&t,&c),cost[t-1].push_back(make_pair(c,i+1));
	for(i=0;i<2;++i){
		sort(cost[i].begin(),cost[i].end()),total[i].resize(1),total[i].back()=0;
		for(j=0;j<cost[i].size();++j)
			total[i].push_back(total[i].back()+cost[i][j].first);
	}
	int l=1,r=n,mid;
	if(!check(n)){
		cout<<-1<<endl;return 0;
	}
	while(l<r){
		mid= (l+r)/2;
		if(check(mid))
			r= mid;
		else
			l=mid+1;
	}
	check(r);
	cout<<RR<<endl;
	for(i=0;i<lx;++i)
		ret.push_back(make_pair(cost[0][i].second,xr));
	for(i=0;i<ly;++i)
		ret.push_back(make_pair(cost[1][i].second,yr));
	for(i=0;i<ret.size();++i)
		printf("%d %d\n",ret[i].first,ret[i].second);
}
