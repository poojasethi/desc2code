#include<iostream>
#include<algorithm>
#include<cstdio>
#include<vector>
#include<cmath>
using namespace std;

bool cal(int day,int *a,int *b,int n,int m,int k,int s,int *t,int *c,bool flag)
{
if(day>=n)
return false;
int dollar=0;
int pound=0;
for(int i=0;i<=day;i++)
{
if(a[i]<a[dollar])
dollar=i;
if(b[i]<b[pound])
pound=i;
}
long long int temp=0;
vector<pair<long long int,pair<int,int> > >v;
for(int i=0;i<m;i++)
if(t[i]==1)
v.push_back(make_pair((long long)c[i]*(long long)a[dollar],make_pair(i+1,dollar+1)));
else
v.push_back(make_pair((long long)c[i]*(long long)b[pound],make_pair(i+1,pound+1)));
sort(v.begin(),v.end());
for(int i=0;i<k;i++)
{
if(flag)
cout<<v[i].second.first<<" "<<v[i].second.second<<endl;
temp+=v[i].first;
}
return(temp<=(long long)s);
}

void func(int *a,int *b,int n,int m,int k,int s,int *t,int *c)
{
int beg=0;
int last=n-1;
int duck=20;
bool flag=false;
while(duck--)
{
int superduck=(beg+last)/2;
bool temp=cal(superduck,a,b,n,m,k,s,t,c,false);
flag=flag|temp;
if(temp)
last=superduck;
else
beg=superduck+1;
}
if(!flag)
cout<<"-1"<<endl;
else
{
cout<<last+1<<endl;
cal(last,a,b,n,m,k,s,t,c,true);
}
}

int main()
{
ios_base::sync_with_stdio(false);
//freopen("in.txt","r",stdin);
int n,m,k,s;
cin>>n>>m>>k>>s;
int a[n],b[n],t[m],c[m];
for(int i=0;i<n;i++)
cin>>a[i];
for(int i=0;i<n;i++)
cin>>b[i];
for(int i=0;i<m;i++)
cin>>t[i]>>c[i];
func(a,b,n,m,k,s,t,c);
return 0;
}
