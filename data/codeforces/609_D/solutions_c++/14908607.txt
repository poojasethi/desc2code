#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
const ll inf=1000000000;
ll tc[2],a[210001],w[2][210001],q[210001];
int d[210001],i,j,k,n,m,rc,c,l,r,mid,ans;
bool cmp(int x,int y){
	return a[x]*tc[d[x]]<a[y]*tc[d[y]];
}
bool check(int x){
	int i;
	ll s,t1,t2;
	for (i=1,t1=t2=inf;i<=x;i++) t1=min(t1,w[0][i]),t2=min(t2,w[1][i]);
	for (i=1;i<=m;i++) q[i]=(d[i])?t2*a[i]:t1*a[i];
	sort(q+1,q+1+m);
	for (s=0,i=1;i<=rc;i++) s+=q[i];
	return s<=c;
}
void opt(int x){
	int i,t1,t2;
	for (i=t1=t2=1;i<=x;i++){
		if (w[0][i]<w[0][t1]) t1=i;
		if (w[1][i]<w[1][t2]) t2=i;
	}
	for (i=1;i<=m;i++) q[i]=i;
	tc[0]=w[0][t1];tc[1]=w[1][t2];
	sort(q+1,q+1+m,cmp);
	for (i=1;i<=rc;i++) printf("%d %d\n",(int)q[i],(d[q[i]])?t2:t1);
}
int main(){
	scanf("%d%d%d%d",&n,&m,&rc,&c);
	for (i=0;i<=1;i++)
	 for (j=1;j<=n;j++) scanf("%d",&k),w[i][j]=k;
	for (i=1;i<=m;i++) scanf("%d%d",&d[i],&k),d[i]--,a[i]=k;
	if (!check(n)){
		puts("-1");return 0;
	}
	for (l=1,r=n;l<r-1;){
		mid=(l+r)>>1;
		if (check(mid)) r=mid;else l=mid+1;
	}
	ans=check(l)?l:r;
	printf("%d\n",ans);
	opt(ans);
	return 0;
}
