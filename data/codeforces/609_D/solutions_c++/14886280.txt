# include <bits/stdc++.h>

# define oo 1000000000000000
# define N 200001

using namespace std;

int n,m,k,s,a[N],b[N],t[N],l,r,tt,vx[N],vy[N];
long long q,res,d[N],x[N],y[N];
pair <long long,pair <int,int> > z[N];
pair <int,int> ress[N];

main()
{
	ios_base::sync_with_stdio(0);
	cin >> n >> m >> k >> s;
	for (int i=1;i<=n;i++) cin >> a[i];
	for (int i=1;i<=n;i++) cin >> b[i];
	for (int i=1;i<=m;i++) cin >> t[i] >> d[i];
	x[1]=a[1]; y[1]=b[1]; vx[1]=1; vy[1]=1;
	for (int i=2;i<=n;i++)
	{
		if (a[i] < x[i-1])
		{
			x[i]=a[i];
			vx[i]=i;
		} else
		{
			x[i]=x[i-1];
			vx[i]=vx[i-1];
		}
		if (b[i] < y[i-1])
		{
			y[i]=b[i];
			vy[i]=i;
		} else
		{
			y[i]=y[i-1];
			vy[i]=vy[i-1];
		}
	}
	res = oo;
	l=1; r=n;
	do
	{
		tt = (l+r) >> 1;
		for (int i=1;i<=m;i++)
		if (t[i]==1)
		{
			z[i-1] = make_pair(d[i] * x[tt],make_pair(i,vx[tt]));
		} else
		{
			z[i-1] = make_pair(d[i] * y[tt],make_pair(i,vy[tt]));
		}
		sort(z,z+m);
		q = 0;
		for (int i=0;i<k;i++) q += z[i].first;
		if (q <= s)
		{
			r = tt-1;
			res = tt;
			for(int i=0;i<k;i++) ress[i]=z[i].second;
		} else l = tt+1;
	} while (l<=r);
	if (res < oo)
	{
		cout << res << endl;
		for (int i=0;i<k;i++) cout << ress[i].first << " " << ress[i].second << endl;
	} else cout << -1;
}
