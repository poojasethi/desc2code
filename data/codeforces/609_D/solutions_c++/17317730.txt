#include<bits/stdc++.h>

using namespace std;

#define nd second
#define st first
#define N 200005LL

pair < long long , int > A[N],B[N],X[N],Y[N];

long long n,k,t,i,j,h,a,b,x,y,l,m,r;

void f(){
   j = t;
	for(a=b=1; j >= ( h = min( (a > x ? N*N : A[m].st * X[a].st) , (b > y ? N*N : B[m].st * Y[b].st) ) ) ;){

		j -= h;

		if(b == y+1 || (a <= x && A[m].st * X[a].st < B[m].st * Y[b].st))  a++;
		else                                                               b++;

	}
}

int main(){

	cin >> n >> b >> k >> t;

	for(i=1;i<=n;i++){
		cin >> A[i].st;
		A[i].nd = i;
		if(i > 1) A[i] = min(A[i] , A[i-1]);
	}

	for(i=1;i<=n;i++){
		cin >> B[i].st;
		B[i].nd = i;
		if(i > 1) B[i] = min(B[i] , B[i-1]);
	}

	for(i=1;i<=b;i++){

		cin >> h >> a;

		if(h == 1)  X[++x] = make_pair(a , i);
		else        Y[++y] = make_pair(a , i);

	}

	sort(X+1 , X+x+1);
	sort(Y+1 , Y+y+1);

	l = 0;
	r = n+1;

	while(l + 1 < r){

		m = l + r >> 1;

		f();

		if(a+b-2 >= k)  r = m;
		else   			 l = m;
	}

	if(r == n+1)
	{  cout << -1;  return 0;  }

	m = r;  f();

	cout << r << endl;  k++;

	for(i=1;i<min(a,k);i++)
		cout << X[i].nd << " " << A[r].nd << endl;

	for(i=1;i<min(k-a+1,b);i++)
		cout << Y[i].nd << " " << B[r].nd << endl;

	return 0;
}
