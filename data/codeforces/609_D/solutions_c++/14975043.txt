#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const int maxn = 2e5 + 10;
const int inf = 1e8;
int n, m, k, s;
int a[maxn], b[maxn], t[maxn], v[maxn];
vector<pair<ll, pair<int, int> > > item;

bool check(int day) {
  item.clear();
  int mina = inf, minb = inf;
  int posa, posb;
  for (int i = 1; i <= day; i++) {
    if (mina > a[i]) mina = a[i], posa = i;
    if (minb > b[i]) minb = b[i], posb = i;
  }
  for (int i = 1; i <= m; i++) {
    if (t[i] == 1) item.push_back(make_pair(1ll * v[i] * mina, make_pair(i, posa)));
    else item.push_back(make_pair(1ll * v[i] * minb, make_pair(i, posb)));
  }
  sort(item.begin(), item.end());
  ll sum = 0;
  for (int i = 0; i < k; i++) sum += item[i].first;
  return s >= sum;
}

int main() {
  //freopen("case.in", "r", stdin);
  cin >> n >> m >> k >> s;
  for (int i = 1; i <= n; i++)  cin >> a[i];
  for (int i = 1; i <= n; i++)  cin >> b[i];
  for (int i = 1; i <= m; i++)  cin >> t[i] >> v[i];
  int l = 0, r = n + 1;
  while (l + 1 < r) {
    int m = (l + r) / 2;
    if (check(m)) r = m;
    else l = m;
  }
  if (l == n) { puts("-1"); return 0; }
  printf("%d\n", r);
  check(r);
  for (int i = 0; i < k; i++) printf("%d %d\n", item[i].second.first, item[i].second.second);
  return 0;
}
