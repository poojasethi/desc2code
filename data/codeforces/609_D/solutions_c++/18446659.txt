#include <bits/stdc++.h>
using namespace std;

int n, m, k, s;
int a[200005];
int b[200005];

int ex[4];

pair<int, int> c[200005];
int ind[200005];

long long cost(int x)
{
	return c[x].first *1ll* ex[c[x].second];
}

bool cmp(int x, int y)
{
	long long c1 = cost(x);
	long long c2 = cost(y);
	return c1 < c2;
}

int main()
{
	scanf("%d %d %d %d", &n, &m, &k, &s);
	for (int i = 0; i < n; i++)
		scanf("%d", a+i);
	for (int i = 0; i < n; i++)
		scanf("%d", b+i);
	for (int i = 0; i < m; i++)
	{
		scanf("%d %d", &c[i].second, &c[i].first);
		c[i].second--;
		ind[i] = i;
	}
	int lo = 0;
	int hi = n-1;
	int md;
	while (lo <= hi)
	{
		md = (lo + hi) / 2;
		ex[0] = ex[1] = 1000000000;
		for (int i = 0; i <= md; i++)
		{
			ex[0] = min(ex[0], a[i]);
			ex[1] = min(ex[1], b[i]);
		}
		sort(ind, ind + m, cmp);
		long long sm = 0;
		for (int i = 0; i < k; i++)
		{
			sm += cost(ind[i]);
		}
		if (sm <= s)
			hi = md-1;
		else
			lo = md+1;
	}
	if (lo >= n)
	{
		printf("-1\n");
	}
	else
	{
		printf("%d\n", lo+1);
		md = lo;
		ex[0] = ex[1] = 1000000000;
		int mn[2];
		for (int i = 0; i <= md; i++)
		{
			if (a[i] < ex[0])
			{
				mn[0] = i;
				ex[0] = a[i];
			}
			if (b[i] < ex[1])
			{
				mn[1] = i;
				ex[1] = b[i];
			}
		}
		sort(ind, ind + m, cmp);
		for (int i = 0; i < k; i++)
		{
			int day = mn[c[ind[i]].second];
			printf("%d %d\n", ind[i]+1, day+1);
		}
	}
}
