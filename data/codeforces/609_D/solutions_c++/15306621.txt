using namespace std;
typedef long long ll;
const int inf = 1 << 29;
const int mod = (int)1e9 + 7;
const double eps = 1e-8;
const double pi = acos(-1.0);
int n, m, k, s;
struct node
{
    ll cost;
    int id, day;
    bool operator < (node b)const
    {
        return cost < b.cost;
    }
}v[200010];
int a[200010], b[200010];
int t[200010],c[200010];

bool check(int mid)
{
    int ma=1, mb=1;
    for(int i=1;i<=mid;i++) {
        if(a[ma] > a[i]) ma = i;
        if(b[mb] > b[i]) mb = i;
    }
    for(int i=1;i<=m;i++){
        if(t[i] == 1) v[i].cost = a[ma]*1ll*c[i], v[i].id = i, v[i].day = ma;
        else v[i].cost = b[mb]*1ll*c[i], v[i].id = i, v[i].day = mb;
    }
    //for(int i=1;i<=m;i++) cout<<v[i].cost<<" "<<v[i].id<<" "<<v[i].day<<endl;
    sort(v+1,v+m+1);
    ll sum = 0;
    for(int i=1;i<=k;i++) sum += v[i].cost;
    return sum <= s;
}

int main() {

#ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin);
    //freopen("out.txt", "w", stdout);
#endif
    cin>>n>>m>>k>>s;
    for(int i=1;i<=n;i++) scanf("%d", &a[i]);
    for(int i=1;i<=n;i++) scanf("%d", &b[i]);
    for(int i=1;i<=m;i++) scanf("%d%d", &t[i], &c[i]);
    int l=1, r = n+1;
    //if(!check(n)) return puts("-1");
    while( l < r){
        int mid = (l + r) >> 1;
        if(check(mid))  r = mid;
        else l = mid + 1;
    }
    if(l == n + 1) cout<<-1<<endl;
    else {
        check(l);
        cout<<l<<endl;
        for(int i=1;i<=k;i++) cout<<v[i].id<<" "<<v[i].day<<endl;
    }
    return 0;
}
