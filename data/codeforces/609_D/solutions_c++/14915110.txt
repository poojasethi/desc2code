#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const int N=200005;
struct node
{
    LL a;
    int b,c;
    node(LL aa=0,int bb=0,int cc=0):a(aa),b(bb),c(cc){}
    bool operator<(const node&tt)const
    {
        return a<tt.a;
    }
}t[N];
int a[N],b[N],c[N],d[N];
int n,m,k,s;
bool f(int x)
{
    int t1=1,t2=1,i,j;
    for(i=1;i<=x;++i)
    {
        if(a[t1]>a[i])
            t1=i;
        if(b[t2]>b[i])
            t2=i;
    }
    for(i=1;i<=m;++i)
    {
        if(c[i]==1)
            t[i]=node((LL)(d[i])*a[t1],i,t1);
        else t[i]=node((LL)(d[i])*b[t2],i,t2);
    }
    sort(t+1,t+1+m);
    LL sum=0;
    for(i=1;i<=k;++i)
        sum+=t[i].a;
    return sum<=s;
}
void bin(int l,int r)
{
    int mid;
    while(l<=r)
    {
        mid=l+r>>1;
        if(f(mid))
            r=mid-1;
        else l=mid+1;
    }
    if(l==n+1)
    {
        puts("-1");
        return;
    }
    f(l);
    printf("%d\n",l);
    for(int i=1;i<=k;++i)
        printf("%d %d\n",t[i].b,t[i].c);
}
int main()
{
    int i,j;
    scanf("%d%d%d%d",&n,&m,&k,&s);
    for(i=1;i<=n;++i)
        scanf("%d",a+i);
    for(i=1;i<=n;++i)
        scanf("%d",b+i);
    for(i=1;i<=m;++i)
        scanf("%d%d",c+i,d+i);
    bin(1,n+1);
    return 0;
}
