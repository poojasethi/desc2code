#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

#define mp(x, y) make_pair((x), (y))

int n, m, k, s;
int a[200005], b[200005];
int da[200005], db[200005];
int t[200005], c[200005];

int pos(int d)
{
	ll g[200005];
	for(int i=0; i<m; i++) {
		if(t[i]==1) {
			g[i]=((ll)a[d])*c[i];
		} else {
			g[i]=((ll)b[d])*c[i];
		}
	}
	sort(g, g+m);
	ll cs=0;
	for(int i=0; i<k; i++) {
		cs+=g[i];
	}
	if(cs>s) return 0;
	else return 1;
}

int main()
{
	scanf("%d%d%d%d", &n, &m, &k, &s);
	for(int i=0; i<n; i++) {
		scanf("%d", &a[i]);
		da[i]=i;
		if(i>0 && a[i-1]<a[i]) {
			a[i]=a[i-1];
			da[i]=da[i-1];
		}
	}
	for(int i=0; i<n; i++) {
		scanf("%d", &b[i]);
		db[i]=i;
		if(i>0 && b[i-1]<b[i]) {
			b[i]=b[i-1];
			db[i]=db[i-1];
		}
	}
	for(int i=0; i<m; i++) {
		scanf("%d%d", &t[i], &c[i]);
	}
	int l=0, r=n-1;
	while(l<r) {
		int m=(l+r)/2;
		if(pos(m)) r=m;
		else l=m+1;
	}
	if(pos(l)) {
		pair<ll,int> g[200005];
		for(int i=0; i<m; i++) {
			if(t[i]==1) {
				g[i]=mp(((ll)a[l])*c[i], i);
			} else {
				g[i]=mp(((ll)b[l])*c[i], i);
			}
		}
		sort(g, g+m);
		printf("%d\n", l+1);
		for(int i=0; i<k; i++) {
			if(t[g[i].second]==1) {
				printf("%d %d\n", g[i].second+1, da[l]+1);
			} else {
				printf("%d %d\n", g[i].second+1, db[l]+1);
			}
		}
		
	} else {
		printf("-1\n");
	}

	return 0;
}
