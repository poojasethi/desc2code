#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 110
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,S,T,x,y,num;
int a[N][N],b[N][N],s[N],t[N],an[N],h[N],f[N],c[N][N];
int dfs(int x,int y) {
	if (h[x]==num) return f[x];
	h[x]=num;
	int i,ans=-1;
	For(i,1,n) if (a[x][i]==1&&a[x][i]+a[i][y]==a[x][y]) ans=max(ans,dfs(i,y));
	if (ans<0) ans=a[0][0];
	f[x]=min(ans,an[x]+1);
	return f[x];
}
int main() {
	scanf("%d%d%d%d",&n,&m,&S,&T);
	memset(a,60,sizeof(a));
	For(i,1,n) a[i][i]=0;
	For(i,1,m) {
		scanf("%d%d",&x,&y);
		a[x][y]=1;
	}
	For(k,1,n)For(i,1,n)For(j,1,n) a[i][j]=min(a[i][j],a[i][k]+a[k][j]);
	scanf("%d",&m);
	For(i,1,m) {
		scanf("%d%d",&x,&y);
		For(j,1,n) if (a[x][j]<1e9&&a[x][j]+a[j][y]==a[x][y]) b[i][a[x][j]]++;
		For(j,1,n) if (a[x][j]<1e9&&a[x][j]+a[j][y]==a[x][y]&&b[i][a[x][j]]==1)
			c[i][j]=1;
		s[i]=x,t[i]=y;
	}
	memset(an,60,sizeof(an));
	an[T]=0;
	for (;;) {
		int F=0;
		For(i,1,m) {
			For(j,1,n) if (c[i][j]) {
				++num;
				int A=dfs(j,t[i]);
				if (A<an[j]) an[j]=A,F=1;
			}
		}
		if (!F) break;
	}
	printf("%d\n",an[S]>1e9?-1:an[S]);
	return 0;
}
