#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <vector>
using namespace std;
const int maxn=110,maxk=110,inf=0x38383838;
int ans[maxn],n,m,route,num,start,end,cut[maxn][maxn],bus[maxn];
int cnt[maxn][maxn],st[maxk],en[maxk],dis[maxn][maxn],vis[maxn],f[maxn];
vector<int>link[maxn];
int dfs(int cur,int en)
{
	if(vis[cur]==num)
		return f[cur];
	vis[cur]=num;
	int busnum=-1;
	for(int i=1;i<=n;i++)
		if(dis[cur][i]==1&&dis[cur][i]+dis[i][en]==dis[cur][en])
			busnum=max(busnum,dfs(i,en));
	if(busnum==-1)
		busnum=dis[0][0];
	f[cur]=min(busnum,bus[cur]+1);
	return f[cur];
}
int main()
{
	scanf("%d%d%d%d",&n,&m,&start,&end);
	memset(dis,0x38,sizeof(dis));
	for(int i=1;i<=m;i++)
	{
		int from,to;
		scanf("%d%d",&from,&to);
		link[from].push_back(to);
		dis[from][to]=1;
	}
	for(int i=1;i<=n;i++)
		dis[i][i]=0;
	for(int k=1;k<=n;k++)
		for(int i=1;i<=n;i++)
			for(int j=1;j<=n;j++)
				dis[i][j]=min(dis[i][j],dis[i][k]+dis[k][j]);
	scanf("%d",&route);
	for(int i=1;i<=route;i++)
	{
		int from,to;
		scanf("%d%d",&from,&to);
		for(int j=1;j<=n;j++)
			if(dis[from][j]<inf&&dis[from][j]+dis[j][to]==dis[from][to])
				cnt[i][dis[from][j]]++;
		for(int j=1;j<=n;j++)
			if(dis[from][j]<inf&&dis[from][j]+dis[j][to]==dis[from][to]&&cnt[i][dis[from][j]]==1)
				cut[i][j]=1;
		st[i]=from;
		en[i]=to;
	}
	memset(bus,0x38,sizeof(bus));
	bus[end]=0;
	for(;;)
	{
		bool flag=0;
		for(int i=1;i<=route;i++)
			for(int j=1;j<=n;j++)
				if(cut[i][j]==1)
				{
					num++;
					int busnum=dfs(j,en[i]);
					if(busnum<bus[j])
					{
						bus[j]=busnum;
						flag=1;
					}
				}
		if(flag==0)
			break;
	}
	printf("%d\n",bus[start]==inf?-1:bus[start]);
	return 0;
}
