#include<algorithm>
#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<iostream>
using namespace std;
#define N 200
#define INF 1000000000
int i,j,k,n,m,x,y,num;
int a[N][N],b[N][N],st[N],ed[N],ans[N],vis[N],f[N],p[N][N];
int dfs(int x,int y){
	if(vis[x]==num)return f[x];
	vis[x]=num;
	int i,Ans=-1;
	for(int i=1;i<=n;i++) 
      if(a[x][i]==1 && a[x][i]+a[i][y]==a[x][y])Ans=max(Ans,dfs(i,y));
	if(Ans<0)Ans=INF;
	f[x]=min(Ans,ans[x]+1);
	return f[x];
}
int main(){
	scanf("%d%d%d%d",&n,&m,&st[0],&ed[0]);
	for(int i=1;i<=n;i++)
	  for(int j=1;j<=n;j++)a[i][j]=INF*(int)(i!=j);
	for(int i=1;i<=m;i++){
		scanf("%d%d",&x,&y);
		a[x][y]=1;
	    }
	for(int k=1;k<=n;k++)
      for(int i=1;i<=n;i++)
        for(int j=1;j<=n;j++)a[i][j]=min(a[i][j],a[i][k]+a[k][j]);
	scanf("%d",&m);
	for(int i=1;i<=m;i++){
		scanf("%d%d",&x,&y);
		for(int j=1;j<=n;j++)
          if(a[x][j]<INF && a[x][j]+a[j][y]==a[x][y])b[i][a[x][j]]++;
		for(int j=1;j<=n;j++)
          if(a[x][j]<INF && a[x][j]+a[j][y]==a[x][y]&&b[i][a[x][j]]==1)p[i][j]=1;
		st[i]=x,ed[i]=y;
	}
	for(int i=1;i<=n;i++)ans[i]=INF;
	ans[ed[0]]=0;
	while(1){
		bool pdd=0;
		for(int i=1;i<=m;i++)
		  for(int j=1;j<=n;j++)
            if(p[i][j]){
				++num;
				int tt=dfs(j,ed[i]);
				if(tt<ans[j]){
                  ans[j]=tt;
                  pdd=1;
                  }
			}
		if (!pdd)break;
	}
	if(ans[st[0]]<INF)printf("%d\n",ans[st[0]]);
	else printf("-1\n");
	return 0;
}
