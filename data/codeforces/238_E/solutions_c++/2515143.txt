#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;

const int MAX=200+10;
const int INF=100000000;

int n,m,q,A,B,dist[MAX][MAX];
int s[MAX],t[MAX],md[MAX],mm[MAX];
vector<int> must[MAX],du[MAX][MAX];

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
#endif
	int i,j,k,a,b;
	scanf("%d %d %d %d",&n,&m,&A,&B);
	for(i=1;i<=n;++i)
		for(j=1;j<=n;++j)
			dist[i][j]=INF;
	for(i=1;i<=m;++i)
	{
		scanf("%d %d",&a,&b);
		dist[a][b]=1;
	}
	for(i=1;i<=n;++i)
		dist[i][i]=0;
	for(k=1;k<=n;++k)
		for(i=1;i<=n;++i)
			for(j=1;j<=n;++j)
				dist[i][j]=min(dist[i][j],dist[i][k]+dist[k][j]);
	scanf("%d",&q);
	for(int id=1;id<=q;++id)
	{
		scanf("%d %d",&a,&b);
		s[id]=a,t[id]=b;
		if(dist[a][b]==INF)
			continue;
		for(i=1;i<=n;++i)
		{
			if(dist[a][i]+dist[i][b]!=dist[a][b])
				continue;
			du[id][dist[a][i]].push_back(i);
			int flag=1;
			for(j=1;j<=n;++j)
				if(i!=j && dist[a][i]==dist[a][j] && dist[j][b]==dist[i][b])
				{
					flag=0;
					break;
				}
			if(flag)
				must[id].push_back(i);
		}
	}
	for(i=1;i<=n;++i)
		md[i]=INF;
	md[B]=0;
	int flag=1;
	while(flag)
	{
		flag=0;
		for(int id=1;id<=q;++id)
		{
			a=s[id],b=t[id];
			if(dist[a][b]==INF)continue;
			for(i=1;i<=n;++i)
				mm[i]=INF;
			mm[b]=min(md[b]+1,INF);
			for(i=dist[a][b]-1;i>=0;--i)
				for(int ij=0;ij<(int)du[id][i].size();++ij)
				{
					j=du[id][i][ij];
					int p=-1;
					for(int ik=0;ik<(int)du[id][i+1].size();++ik)
					{
						k=du[id][i+1][ik];
						if(dist[j][k]==1 && dist[a][k]+dist[k][b]==dist[a][b] && dist[a][k]==i+1)
							p=max(p,mm[k]);
					}
					if(p!=-1)
						mm[j]=min(md[j]+1,p);
				}
			for(int it=0;it<(int)must[id].size();++it)
			{
				i=must[id][it];
				if(mm[i]<md[i])
				{
					md[i]=mm[i];
					flag=1;
				}
			}
		}
	}
	if(md[A]==INF)printf("-1\n");
	else printf("%d\n",md[A]);
	return 0;
}
