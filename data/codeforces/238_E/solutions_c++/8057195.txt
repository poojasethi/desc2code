#include <bits/stdc++.h>
using namespace std;
#define rep(i,a,n) for (int i=a;i<n;i++)
#define per(i,a,n) for (int i=n-1;i>=a;i--)
#define pb push_back
#define mp make_pair
#define all(x) (x).begin(),(x).end()
#define SZ(x) ((int)(x).size())
#define fi first
#define se second
typedef vector<int> VI;
typedef long long ll;
typedef pair<int,int> PII;
const ll mod=1000000007;
ll powmod(ll a,ll b) {ll res=1;a%=mod;for(;b;b>>=1){if(b&1)res=res*a%mod;a=a*a%mod;}return res;}
// head

const int inf=0x20202020;

const int N=110;
int n,m,a,b,u,v,q;
int s[N],t[N],g[N][N],cnt[N],cost[N][N],can[N][N],must[N][N];
int main() {
	scanf("%d%d%d%d",&n,&m,&a,&b); --a; --b;
	rep(i,0,n) rep(j,0,n) if (i!=j) g[i][j]=inf;
	rep(i,0,m) {
		scanf("%d%d",&u,&v); g[u-1][v-1]=1;
	}
	rep(k,0,n) rep(i,0,n) rep(j,0,n) g[i][j]=min(g[i][j],g[i][k]+g[k][j]);
	scanf("%d",&q);
	rep(i,0,q) {
		scanf("%d%d",s+i,t+i); --s[i]; --t[i];
		if (g[s[i]][t[i]]>n) continue;
		memset(cnt,0,sizeof(cnt));
		rep(j,0,n) if (g[s[i]][j]+g[j][t[i]]==g[s[i]][t[i]])
			cnt[g[s[i]][j]]++,can[i][j]=1;
		rep(j,0,n) if (can[i][j]&&cnt[g[s[i]][j]]==1) must[i][j]=1;
	}
	memset(cost,0x20,sizeof(cost));
	rep(i,0,q) if (can[i][b]) cost[i][b]=1;
	while (1) {
		bool fg=0;
		rep(y,0,q) rep(x,0,n) if (can[y][x]) {
			if (x==b) continue;
			int mv=-1;
			rep(v,0,n) if (g[x][v]==1&&g[x][v]+g[v][t[y]]==g[x][t[y]]) mv=max(mv,cost[y][v]);
			if (mv>=0&&mv<cost[y][x]) { cost[y][x]=mv; fg=1;}
			mv=inf;
			rep(v,0,q) if (must[v][x]&&v!=y) mv=min(mv,cost[v][x]+1);
			if (mv<cost[y][x]) { cost[y][x]=mv; fg=1;}
		}
		if (!fg) break;
	}
	int mv=inf;
	rep(i,0,q) if (must[i][a]) mv=min(mv,cost[i][a]);
	if (mv>n) puts("-1"); else printf("%d\n",mv);
}
