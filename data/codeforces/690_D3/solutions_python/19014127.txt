#include <iostream>
using namespace std;

typedef long long ll;

const ll MOD = 1e6 + 3;
//const int MOD = 17;
const int MAXN = 1000, MAXW = 300;

ll dp[MAXW];
ll f[MAXW];

int main() {
    ll c, w, h; cin >> c >> w >> h;
    ll period = w + 2;

    f[0] = 1;
    for (int i = 1; i <= w; i ++)
        f[i] = 1LL * f[i - 1] * h % MOD;

    dp[0] = 1;
    for (ll n = 1; n <= period; n ++) {
        int mina = n % period;
        dp[mina] = 0;
        for (int i = 0; i <= min(n, w); i ++) {
            ll mem = 0;
            if (n - i - 1 == -1)
                mem = 1;
            else mem = dp[(n - i - 1) % period];
            dp[mina] += (1LL * mem * f[i]) % MOD;
            if (dp[mina] > MOD)
                dp[mina] -= MOD;
        }
//      cout << n << ' ' << mina << ' ' << dp[mina] << endl;
    }

    for (int n = period + 1; n <= c; n ++) {
        ll mina = n % period;
        ll shayan = (n - 1) % period;
        ll abtin = (n - w - 2) % period;

        dp[mina] = (dp[shayan] - (1LL * dp[abtin] * f[w]) % MOD + 3*MOD) % MOD;
        dp[mina] = (dp[mina] * h) % MOD + dp[shayan];
        dp[mina] %= MOD;
    }

    cout << dp[c % period] << endl;
    return 0;
}
