#include<cstdio>
#define MOD 1000003
long long mat1[110][110],mat2[110][110],mat3[110][110],c,w,h,i,j,k;

void pow(long long x){
    if(x==1)return;
    pow(x/2);
    for(i=0;i<=w;i++){
        for(j=0;j<=w;j++){
            mat2[i][j]=0;
            for(k=0;k<=w;k++){
                mat2[i][j]+=mat1[i][k]*mat1[k][j];
                mat2[i][j]%=MOD;
            }
        }
    }
    for(i=0;i<=w;i++){
        for(j=0;j<=w;j++){
            mat1[i][j]=mat2[i][j];
        }
    }
    if(x%2==1){
        for(i=0;i<=w;i++){
            for(j=0;j<=w;j++){
                mat2[i][j]=0;
                for(k=0;k<=w;k++){
                    mat2[i][j]+=mat1[i][k]*mat3[k][j];
                    mat2[i][j]%=MOD;
                }
            }
        }
        for(i=0;i<=w;i++){
            for(j=0;j<=w;j++){
                mat1[i][j]=mat2[i][j];
            }
        }
    }
}

int main(){
    scanf("%I64d %I64d %I64d",&c,&w,&h);
    for(i=0;i<=w;i++){
        mat1[0][i]=1;
        mat3[0][i]=1;
    }
    for(i=1;i<=w;i++){
        mat1[i][i-1]=h;
        mat3[i][i-1]=h;
    }
    pow(c+1);
    printf("%I64d",mat1[0][0]);
}
