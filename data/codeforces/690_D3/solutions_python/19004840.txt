#include <iostream>

using namespace std;

long long a[200][200],c,w,h,res[200][200],d[200][200],mod=1e6+3,p[200];

void mul1()
{
    for (long long i=0;i<=w;i++)
        for (long long j=0;j<=w;j++)
    {
        long long tmp=0;
        for (long long k=0;k<=w;k++)
            tmp+=res[i][k]*a[k][j],tmp%=mod;
        d[i][j]=tmp;
    }
    for (long long i=0;i<=w;i++)
        for (long long j=0;j<=w;j++)
            res[i][j]=d[i][j];
}

void mul2()
{
    for (long long i=0;i<=w;i++)
        for (long long j=0;j<=w;j++)
    {
        long long tmp=0;
        for (long long k=0;k<=w;k++)
            tmp+=a[i][k]*a[k][j],tmp%=mod;
        d[i][j]=tmp;
    }
    for (long long i=0;i<=w;i++)
        for (long long j=0;j<=w;j++)
            a[i][j]=d[i][j];
}

void exp(long long k)
{
    while (k)
    {
        if (k%2)
            mul1();
        mul2();
        k/=2;
    }
}

int main()
{
    cin >> c >> w >> h;
    for (long long i=0;i<=w;i++)
    {
        a[0][i]=1;
        if (i) a[i][i-1]=h;
        res[i][i]=1;
    }
    exp(c);
    p[0]=1;
    long long ret=0;
    for (long long i=0;i<=w;i++)
    {
        long long tmp=0;
        for (long long j=0;j<=w;j++)
            tmp+=res[i][j]*p[j],tmp%=mod;
        ret+=tmp;
        ret%=mod;
    }
    cout << ret;
}
