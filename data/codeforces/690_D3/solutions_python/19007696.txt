#include <iostream>
#include <cstdlib>
#include <cstdio>
using namespace std;
typedef long long ll;
const int P = 1000003;
ll a[101][101], s[101][101], d[101][101];
int main() {
  #ifdef Vlad_kv
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
  #endif //Vlad_kv
  int q, w, e, r, t, y;
  cin >> y >> q >> t;
  for (e = 0; e <= q; e++) {
    for (r = 0; r <= q; r++) { 
      a[e][r] = d[e][r] = 0;
    }
    d[e][e] = 1;
  }
  for (e = 0; e < q; e++) {
    a[e][e + 1] = t;
    a[e][0] = 1;
  }
  a[q][0] = 1;
  
  for (; y; y /= 2) {
    if (y % 2 == 1) {
      for (e = 0; e <= q; e++) {
        for (r = 0; r <= q; r++) {
          s[e][r] = 0;
        }
      }
      for (w = 0; w <= q; w++) {
        for (e = 0; e <= q; e++) {
          for (r = 0; r <= q; r++) {
            s[w][r] = (s[w][r] + a[w][e] * d[e][r]) % P;
          }
        }
      }
      for (e = 0; e <= q; e++) {
        for (r = 0; r <= q; r++) {
          d[e][r] = s[e][r];
        }
      }
      
    }
    
    for (e = 0; e <= q; e++) {
      for (r = 0; r <= q; r++) {
        s[e][r] = 0;
      }
    }
    for (w = 0; w <= q; w++) {
      for (e = 0; e <= q; e++) {
        for (r = 0; r <= q; r++) {
          s[w][r] = (s[w][r] + a[w][e] * a[e][r]) % P;
        }
      }
    }
    for (e = 0; e <= q; e++) {
      for (r = 0; r <= q; r++) {
        a[e][r] = s[e][r];
      }
    }
    
  }
  
  int res = 0;
  for (w = 0; w <= q; w++) {
    res = (res + d[0][w]) % P;
  }
  cout << res;
  return 0;
}
