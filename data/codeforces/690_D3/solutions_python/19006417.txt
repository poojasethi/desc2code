#include<cstdio>
#include<cstring>
typedef long long ll;
const int N=105,mod=1000003;
int C,W,H;
struct matrix{
	int a[N][N];
	int r,c;
	friend matrix operator *(matrix a,matrix b){
		static matrix c;
		memset(c.a,0,sizeof(c.a));
		for (int i=0;i<=a.r;i++)
		 for (int j=0;j<=b.c;j++)
		  for (int k=0;k<=a.c;k++)
		   c.a[i][j]+=((ll)a.a[i][k]*b.a[k][j])%mod,c.a[i][j]%=mod;
		c.r=a.r,c.c=b.c;
		return c;
	} 
}A,F;
int pow(int a,int k){
	int res=1;
	for (;k;k>>=1,a=(ll)a*a%mod) if (k&1) res=(ll)res*a%mod;
	return res;
}
void pow(matrix a,int k){
	for (;k;k>>=1,a=a*a) if (k&1) F=F*a; 
}
int main(){
	scanf("%d%d%d",&C,&W,&H);
	for (int i=1;i<=W;i++) A.a[i][i-1]=1; 
	for (int i=0;i<=W;i++) A.a[i][W]=pow(H,W-i);
	A.r=A.c=W;
	F.r=0,F.c=W; 
	for (int i=0;i<=W;i++) F.a[0][i]=pow(H+1,i);
	pow(A,C);
	printf("%d",F.a[0][0]);
	return 0;
}
