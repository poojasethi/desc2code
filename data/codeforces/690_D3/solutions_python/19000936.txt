#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

#define ll long long
#define inf 1e9
#define eps 1e-10
#define md 1000003
#define N 110
using namespace std;
struct Ju
{
	ll a[N][N];
	int x,y;
} a,ju,mid,ans;
bool operator *= (Ju &a,Ju &b)
{
	memset(mid.a,0,sizeof(mid.a));
	mid.x=a.x; mid.y=b.y;
	for (int i=0;i<=a.x;i++)
	  for (int k=0;k<=a.y;k++)
	    if (a.a[i][k])
	      for (int j=0;j<=b.y;j++)
	        mid.a[i][j]=(mid.a[i][j]+a.a[i][k]*b.a[k][j])%md;
	a=mid;
}
 
void kpow(Ju &a,int b)
{
	memset(ans.a,0,sizeof(ans.a));
	ans.x=a.x; ans.y=a.y;
	for (int i=0;i<=a.x;i++) ans.a[i][i]=1;
	while (b)
	{
		if (b&1) ans*=a;
		a*=a; b>>=1;
	}
	a=ans;
}
int main()
{
	int C,W,H;
	scanf("%d%d%d",&C,&W,&H);
	ju.x=W; ju.y=W;
	for (int i=0;i<W;i++) ju.a[i][i+1]=H;
	for (int i=0;i<=W;i++) ju.a[i][0]=1;
	kpow(ju,C+1);
	a.x=0; a.y=W; a.a[0][0]=1;
	a*=ju;
	printf("%lld\n",a.a[0][0]);
	return 0;
}
