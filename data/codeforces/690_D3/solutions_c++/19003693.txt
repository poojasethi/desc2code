#include<bits/stdc++.h>
using namespace std;
#define NN 100010
typedef long long INT;
const int MAX=110;
struct mat{
	INT m[MAX][MAX];
};
INT v[MAX],rst[MAX];
mat P;
const int mod=1e6+3;
mat mul(mat a,mat b,int n){
	int i,j,k;
	mat ans;
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			ans.m[i][j]=0;
			for(k=0;k<n;k++) ans.m[i][j]+=a.m[i][k]*b.m[k][j]%mod;
			ans.m[i][j]%=mod;
		}
	}
	return ans;
}
void print(mat a,int n){
	int i,j;
	for(i=0;i<n;i++){
		for(j=0;j<n;j++) printf("%d ",a.m[i][j]);
		puts("");
	}
}

mat paw(int k,int n){
	int i,j;
	mat ans;
	if(!k) {
		for(i=0;i<n;i++) for(j=0;j<n;j++) {
			if(i==j) ans.m[i][j]=1;
			else ans.m[i][j]=0;
		}
		return ans;
	}
	ans=paw(k/2,n);
	ans=mul(ans,ans,n);
	if(k&1) ans=mul(ans,P,n);
	return ans;
}
int main(){
//	freopen("in.in","r",stdin);
	int C,W,H,i,j;
	scanf("%d%d%d",&C,&W,&H);
	v[0]=1;v[1]=H+1;
	rst[0]=1;
	for(i=1;i<MAX;i++) rst[i]=rst[i-1]*H%mod;
	for(i=2;i<=W;i++){
		for(j=1;j<=i;j++) v[i]=(v[i]+rst[j-1]*v[i-j]%mod)%mod;
		v[i]+=rst[i];
	}
	if(C<=W) {printf("%I64d\n",v[C]);return 0;}
	for(i=0;i<=W;i++) P.m[0][i]=rst[i];
	for(i=1;i<=W;i++){
		for(j=0;j<=W;j++){
			if(i==j+1) P.m[i][j]=1;
			else P.m[i][j]=0;
		}
	}
	mat Q=paw(C-W,W+1);
	INT jin=0;
	for(i=0;i<=W;i++) jin=(jin+v[W-i]*Q.m[0][i]%mod)%mod;
	printf("%I64d\n",jin);
}
