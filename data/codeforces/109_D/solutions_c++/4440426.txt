#include <bits/stdc++.h>



using namespace std;





#define fr(i,a,b) for(int i=a;i<b;++i)
#define N 100100
typedef pair<int,int> pii;
#define F first
#define S second
#define mp make_pair
#define rep(i,b) fr(i,0,b)


int v[N];
pii vor[N];
int onde[N];
int pivo = -1;
bool ta = true;

pii resp[2*N];
int nresp = 0;


bool lucky(int a){
	while(a){
		int x = a%10;
		if(x == 4 || x == 7) a /= 10;
		else return false;
	}
	return true;
}


void tr(int a, int b){
	if(a != b){
		onde[v[b]] = a;
		onde[v[a]] = b;
		swap(v[a], v[b]);
		resp[nresp++] = pii(a,b);
	}
	if(pivo == a) pivo = b;
	else if(pivo == b) pivo = a;
}



int main(){
	int n;
	scanf("%d",&n);
	fr(i,0,n){
		scanf("%d",&v[i]);
		if(lucky(v[i])){
			pivo = i;
		}
		if(i && v[i] < v[i-1]) ta = false;
		vor[i] = pii(v[i],i);
	}
	if(ta) printf("0\n");
	else if(pivo == -1){
		printf("-1\n");
	}
	else{
		sort(vor, vor+n);
		fr(i,0,n){
			v[vor[i].S] = i;
		}
		fr(i,0,n){
			onde[v[i]] = i;
		}
		fr(i,0,n){
			tr(pivo,i);
			tr(i,onde[i]);
		}
		printf("%d\n",nresp);
		fr(i,0,nresp) printf("%d %d\n",resp[i].F+1,resp[i].S+1);
	}
	return 0;
}
