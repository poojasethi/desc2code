#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cmath>
#include<vector>
#define maxn 100100
using namespace std;
int n, i, key, pos[maxn], u;
pair<int, int> a[maxn], b[maxn], tmp;
vector<int> x,y;

int lucky(int x){
  if(x==0) return 0;
  while(x>0){
    if(x%10==4||x%10==7) x/=10;
    else return 0;
  }    
  return 1;
}

int main(){
    cin>>n;
    for(i=1;i<=n;i++)cin>>a[i].first;
    for(i=1;i<=n;i++){
      a[i].second=i;
      b[i]=a[i];
      pos[i] = i;
    }
    int ok=1;
    for(i=1;i<n;i++) if (a[i]>a[i+1]) ok = 0;
    if(ok==1){cout<<0;return 0;}
    sort(b+1,b+n+1);    
    key = 0;
    for(i = 1; i<= n;i++) if(lucky(a[i].first)) {key = i;break;}
    if(key==0){cout<<"-1";return 0;}
    for(i=1;i<=n;i++){
       u=b[i].second;                
       if(u==a[i].second)continue;
       if(key!=i){
          pos[a[key].second]=i;
          pos[a[i].second]=key;
          tmp = a[key]; a[key]=a[i];a[i]=tmp;
          x.push_back(key);
          y.push_back(i);          
          key = i;
       }
       int j = pos[u];
       if(key!=j){
          pos[a[key].second]=j;
          pos[a[j].second]=key;
          tmp = a[key]; a[key]=a[j];a[j]=tmp;
          x.push_back(key);
          y.push_back(j);          
          key = j;
       }
    }
    cout<<x.size()<<endl;
    for(i=0;i<x.size();i++){
      cout<<x[i]<<" "<<y[i]<<endl;
    }
    //system("pause");
}
