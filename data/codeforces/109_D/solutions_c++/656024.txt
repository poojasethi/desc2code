#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

int is_lucky(int x) {
  while (x) {
    if (x % 10 != 4 && x % 10 != 7) return 0;
    x /= 10;
  }
  return 1;
}

const int N = 100000 + 5;

int n, an;
int a[N];
int idx[N];
int ans[N + N][2];

bool cmp(int i, int j) {
  return a[i] < a[j];
}

int change(int x, int y) {
  if (x == y) return x;
  ans[an][0] = x;
  ans[an++][1] = y;
  swap(a[x], a[y]);
  idx[a[x]] = x;
  idx[a[y]] = y;
  return y;
}

int main() {
  scanf("%d", &n);
  for (int i = 0; i < n; i++)
    scanf("%d", &a[i]);

  int x = -1;

  for (int i = 0; x == -1 && i < n; i++)
    if (is_lucky(a[i])) x = i;

  if (x > -1) {
    for (int i = 0; i < n; i++)
      idx[i] = i;
    sort(idx, idx + n, cmp);
    for (int i = 0; i < n; i++)
      a[idx[i]] = i;

    an = 0;
    for (int i = 0; i < n; i++) {
      if (idx[i] != i) {
        x = change(x, i);
        x = change(x, idx[i]);
      }
    }

    printf("%d\n", an);
    for (int i = 0; i < an; i++)
      printf("%d %d\n", ans[i][0] + 1, ans[i][1] + 1);

  } else {
    int flag = 1;
    for (int i = 1; i < n; i++)
      if (a[i] < a[i - 1]) flag = 0;
    if (flag) {
      puts("0");
    } else {
      puts("-1");
    }
  }
  return 0;
}
