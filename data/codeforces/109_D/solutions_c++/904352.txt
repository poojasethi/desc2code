#include <iostream>
#include <algorithm>
using namespace std;

pair<int ,int > ans[210000],p[110000];
int a[110000];
int n,k;

void swp(int x,int y){
        if (x!=y){
                swap(p[x].first,p[y].first);
                p[p[x].first].second=x;
                p[p[y].first].second=y;
                ans[++k]=make_pair(x,y);
        }
}

bool lucky(int x){
        if (!x) return true;
        return ((x%10)==4||(x%10==7))&&lucky(x/10);
}

int main(){
        int i,x;
        cin>>n;
        for(i=1;i<=n;i++)cin>>a[i];
        for(i=1;i<n;i++) if (a[i]>a[i+1]) break;
        if (i==n) {cout<<0<<endl;return 0;}
        for(i=1;i<=n;i++)if (lucky(a[i])) break;
        if (i>n) {cout<<-1<<endl;return 0;}
        else {
                x=i;
                for (i=1;i<=n;i++) p[i]=make_pair(a[i],i);
                sort(p+1,p+1+n);
                for (i=1;i<=n;i++) p[p[i].second].first=i;
                for (i=1;i<=n;i++)if (p[i].second!=i){
                        int t;
                        t=p[i].second;swp(x,i);swp(i,t);x=t;
                }
                cout<<k<<endl;
                for (i=1;i<=k;i++) cout<<ans[i].first<<' '<<ans[i].second<<endl;
        }
}
