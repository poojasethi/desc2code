#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<iostream>
#include<map>
#define ll long long
using namespace std;
int tot,ans[222222][2],st,stv,pos;
int n,num1[111111];
map<int,int>mp;
struct node
{
	int v,id;
}num[111111];
int judge(int k)
{
	while(k>0)
    {
    	int t=k%10;
    	k/=10;
    	if(t!=4&&t!=7)return 0;
    }
    return 1;
}
void slove()
{
	int i;
	for(i=1;i<=n;i++)
	{
		if(i==pos)continue;
		if(num[i].v==i)continue;
		if(st!=i)
		{
			ans[tot][0]=i;
			ans[tot++][1]=st;
			num[st].v=num[i].v;
		    mp[num[i].v]=st;
		}
		ans[tot][0]=i;
		ans[tot++][1]=mp[i];
		num[mp[i]].v=-1;
		st=mp[i];
	}
	if(st!=pos)
	{
		ans[tot][0]=pos;
	    ans[tot++][1]=st;
	}
	
}
void output()
{
	int i;
	printf("%d\n",tot);
	for(i=0;i<tot;i++)
	{
		printf("%d %d\n",ans[i][0],ans[i][1]);
	}
}
bool cmp(node a,node b)
{
	if(a.v==b.v)return a.id<b.id;
	return a.v<b.v;
}
bool cmp1(node a,node b)
{
	return a.id<b.id;
}
int main()
{
	int i,j;
	scanf("%d",&n);
	st=-1;
	tot=0;
	for(i=1;i<=n;i++)
	{
		scanf("%d",&num[i].v);
		num[i].id=i;		
	    if(st==-1&&judge(num[i].v))st=i;
	}
	sort(num+1,num+n+1,cmp);
	for(i=1;i<=n;i++)num[i].v=i;
	sort(num+1,num+n+1,cmp1);
	for(i=1;i<=n;i++){mp[num[i].v]=i;}
	if(st==-1)
	{
		 int flag=0;
		 for(i=1;i<=n;i++)
		 {
 		    if(num[i].v!=num[i].id){flag=1;break;}
 		 }
 		 if(flag)printf("-1\n");
 		 else printf("0");
	}
	else 
	{
		pos=num[st].v;
		slove();
		output();
	}
}
