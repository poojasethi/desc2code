#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
#define P push_back
#define M make_pair
#define F first
#define S second
bool luck(int& x){
	int t;
	do{	t= x%10;
		if(t!=4 && t!=7)return false;
	}while(x/=10);
	return true;
}
vector<pair<int,int> > p,a;
void swp(int x,int y){
	if(x!=y){swap(p[x].F,p[y].F);p[p[x].F].S=x;p[p[y].F].S=y;a.P(M(x,y));}
}
int t,z=-1,n,i;
int main(){
	scanf("%d",&n);
	for(i=0;i<n;++i){
		scanf("%d",&t);
		p.P(M(t,i));
		if(z==-1)if(luck(t))z=i;
	}
	sort(p.begin(),p.end());
	for(i=0;i<n;++i)p[p[i].S].F=i;
	for(i=0;i<n;++i)p[p[i].F].S=i;
	for(i=0;i<n && p[i].F==i;++i);
	if(i==n)return puts("0");
	if(i<n && (z==-1))return puts("-1");
	for(;i<n;++i)if(p[i].F!=i){
		t=p[i].S;swp(z,i);
		swp(i,t);z=t;
	}
	printf("%d\n",a.size());
	for(i=0;i<a.size();++i)printf("%d %d\n",a[i].F+1,a[i].S+1);
	return 0;
}
