#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int len,outx[200010],outy[200010],n,t;

struct arr
{
	int x,lab;
} a[100010],b[100010];

bool comp(arr a,arr b)
{
	return a.x<b.x;
}

void exc(int x,int &y)
{
	if (x==y) return;
	outx[++len]=x; outy[len]=y;
	int p=a[x].lab,q=a[y].lab;
	swap(a[x].lab,a[y].lab);
	swap(b[p].lab,b[q].lab);
	swap(a[x].x,a[y].x);
	swap(x,y);
}

bool islucky(int k)
{
	if (!k) return false;
	while(k)
	{
		int t=k%10;
		k/=10;
		if (t!=4&&t!=7) return false;
	}
	return true;
}

int main()
{
	//freopen("109D.in","r",stdin);
	
	scanf("%d",&n);
	for(int i=0;i<n;i++) scanf("%d",&a[i].x),b[i].x=a[i].x,b[i].lab=i;
	sort(b,b+n,comp);
	for(int i=0;i<n;i++) a[b[i].lab].lab=i;
	t=-1;
	for(int i=0;i<n;i++) if (islucky(a[i].x)) t=i;
	if (t<0)
	{
		for(int i=1;i<n;i++) if (a[i].x<a[i-1].x) t=i;
		if (t>=0) puts("-1"); else puts("0");
		return 0;
	}
	for(int i=0;i<n;i++) if (a[i].lab!=i)
	{
		exc(i,t);
		exc(b[i].lab,t);
	}
	printf("%d\n",len);
	for(int i=1;i<=len;i++) printf("%d %d\n",outx[i]+1,outy[i]+1);
	/*
	for(int i=0;i<n;i++) printf("%d ",a[i].x);
	puts("");
	*/
	return 0;
}
