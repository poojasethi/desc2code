#include <iostream>
#include <algorithm>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;

int n,a[100010],k[100010],b[100010],m,mm,r,c,g[200020][2];

bool f(int s,int t){if(a[s]<a[t])return true;else return false;}
bool islucky(int s){
  while(s){
    if(s%10==4||s%10==7){s/=10;continue;}
    else return false;
  }
  return true;
}

void h(int s,int t){
  
  g[r][0]=s;g[r][1]=t;r++;
  b[k[s]]=t;b[k[t]]=s;
  swap(k[s],k[t]);
  swap(a[s],a[t]);

}


int main(){
  cin>>n;
  rep(i,n){cin>>a[i];b[i]=i;}

  m=-1; rep(i,n)if(islucky(a[i])){m=i;break;}

  if(m==-1){
    c=1;
    rep(i,n-1)if(a[i+1]-a[i]>=0&&c==1)continue;else {c=0;break;}
    if(c==0)cout<<-1<<endl;else cout<<0<<endl;
    return 0;
  }

  sort(b,b+n,f);rep(i,n)k[b[i]]=i;

  rep(i,n){
    if(b[i]==i)continue;
    if(i==m){m=b[i];h(i,b[i]);}
    else if(b[i]==m){h(i,m);m=i;}
    else {mm=b[i];h(i,m);h(i,mm);m=mm;}

  }

  cout<<r<<endl;
  rep(i,r)cout<<g[i][0]+1<<" "<<g[i][1]+1<<endl;

  return 0;

}
