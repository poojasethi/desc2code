#include <stdio.h>
int n;
int m;
int p[1001], can[1001];
int main() {
	int n, k, last = 0;
	scanf ("%d%d", &n, &k);
	for (int i = 2; i <= n; i++)
		if (!p[i]) {
			k -= can[i];
			if (last != 0 && last + i + 1 <= n) can[last + i + 1] = 1;
			last = i;
			for (int j = i * i; j <= n; j += i) p[j] = 1;
		}
	if (k > 0) printf ("NO\n"); else printf ("YES\n");
	return 0;
}
