#include<cstdio>
#include<vector>
using namespace std;

vector<int>p;
bool prime(int n){
	for(int d=2; d*d<=n; d++) if(n%d==0) return 0;
	return 1;
}
main(){
	int n,k;
	scanf("%d %d",&n,&k);
	for(int i=2; i<n; i++) if(prime(i)){
		p.push_back(i);
	}
	for(int i=0; i+1<p.size(); i++){
		if(p[i]+p[i+1]+1<=n && prime(p[i]+p[i+1]+1)) --k;
	}
	if(k<=0) printf("YES\n");
	else printf("NO\n");
}
