#include <iostream>
#include <cmath>
using namespace std;
int pr=2,ans=0,n,k;
bool prost(int x)
{
	for(int i=2; i<=(int)sqrt(x); i++)
		if(x%i==0)return 0;
	return 1;
}
int main()
{
	cin >> n >> k;
	for(int i=3; i<=n; i++)
		if(prost(i)){
			if(prost(pr+i+1) && pr+i+1<=n)ans++;
			pr = i;
		}
	if(ans>=k)cout<<"YES"; else cout<<"NO";
	return 0;
}
