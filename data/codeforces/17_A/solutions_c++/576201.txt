#include <stdio.h>

int main(void) {
	int n, k;
	int p[1005], np=0;
	scanf("%d%d", &n, &k);
	int i, j;
	for(i=2;i<=n;i++) {
		for(j=2;j<i;j++)
			if(i%j==0) break;
		if(j==i) p[np++] = i;
	}
	for(i=0;i<np;i++) {
		for(j=1;j<np;j++)
			if(p[j]+p[j-1]+1==p[i])
				break;
		if(j<np) --k;
	}
	puts(k<=0? "YES":"NO");
	return 0;
}
