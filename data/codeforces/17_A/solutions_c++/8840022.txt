#include <stdio.h>
#define MAX 1010

bool pc[MAX];
int p[MAX], cnt;

int main(void){
	int i, j, k, n;
	scanf("%d %d", &n, &k);
	for (i=2;i<=n;i++){
		if (pc[i]) continue;
		p[++cnt]=i;
		for (j=i*i;j<=n;j+=i){
			pc[j]=1;
		}
	}

	for (i=1;i<cnt;i++){
		if (p[i]+p[i+1]+1>n) break;
		if (!pc[p[i]+p[i+1]+1]) k--;
		if (k==0) break;
	}
	if (k==0) printf("YES\n");
	else printf("NO\n");
	return false;
}
