#include <iostream>
using namespace std;
int main()
{
	int a[1000],b[1000],i,j,s=0,n,k;
	cin>>n>>k;
	for(i=1;i<=n;i++)
		a[i]=1;
	a[1]=0;
	for(i=1;i<=n;i++)
		if (a[i]==1){
			s++;
			b[s]=i;
			for(j=2;j<=n/i;j++)
				a[i*j]=0;
		}
	for(i=s;i>=3;i--)
		for(j=i-1;j>=2;j--)
			if (b[i]==b[j]+b[j-1]+1) k--;
	if (k<=0) cout<<"YES"; else cout<<"NO";
	return 0;
}
