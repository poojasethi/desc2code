#include <iostream>
using namespace std;
int prime(int n){
    int i;
    for (i=2;i*i<=n;i++)
        if (n%i==0) return 0;
    return 1;
}
int n,m,p[1010],pn;
int main(){
    int i;
    cin>>n>>m;
    for (i=2;i<=n;i++)
        if (prime(i)) p[++pn]=i;
    for (i=1;i<pn;i++)
        if (p[i]+p[i+1]+1<=n && prime(p[i]+p[i+1]+1)) m--;
    if (m<=0) cout<<"YES";
    else cout<<"NO";
    return 0;
}
