#include <bits/stdc++.h>
using namespace std;
int main ()
{
int long long a,b,c1=1,c=0,d=0,d1[10001],j,i,l;
cin>>a>>b;
for (i=2;i<=a;i++)
{
for (j=2;j<=i;j++)
{
if (i%j!=0){c++;}
}
if (c==i-2)
{
d1[c1]=i;
c1++;
}
c=0;
}
for (i=1;i<c1;i++)
{
for (j=1;j<=c1;j++)
{
if (d1[i]+d1[i+1]+1==d1[j]){d++;}
}
}
if (d>=b){cout<<"YES"<<endl;}
else{cout<<"NO"<<endl;}
}
