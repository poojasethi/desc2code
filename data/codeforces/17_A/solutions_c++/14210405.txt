#include <iostream>
#include <cmath>

using namespace std;

int cnt,ans,n,k;

bool is_prime(int x){
  for(int i=2; i<=(int)sqrt(x); i++)
    if(x%i==0)
      return false;
  return true;
}

int main(){
  cin>>n>>k;
  cnt=2;
  for(int i=3;i<=n;i++)
    if(is_prime(i)){
      if(is_prime(cnt+i+1) && cnt+i+1<=n)
	ans++;
      cnt=i;
    }
  if(ans>=k)
    cout<<"yes"<<endl;
  else
    cout<<"no"<<endl;
}
