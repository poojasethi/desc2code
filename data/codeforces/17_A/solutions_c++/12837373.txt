#include<iostream>
using namespace std;
int i,n,m,k,sum,j,b[10005];
bool a[10005];
int main()
{
	cin>>n>>k;
	for (int i=2;i<=n;i++)
	if (!a[i])
	{
		b[m++]=i;
		for (int j=i*2;j<=n;j+=i)
		a[j]=1;
	}
	for (int i=0;i<m;i++)
	for (int j=0;j<i;j++)
	if (b[i]-1-b[j]==b[j+1])
	sum=sum+1;
	if (sum>=k)
	cout<<"YES";
	else
	cout<<"NO";
	return 0;
}
