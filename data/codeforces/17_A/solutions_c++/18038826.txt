#include <iostream>
int n,k,i,j,s,a[200],b,m;
int main()
{
 std::cin>>n>>k;
 for(i=2;i<=n;++i)
 {
  s=1;
  for(j=0;j<m;++j)
   if(!(i%a[j]))
    s=0;
  if(s)
   a[m++]=i;
 }
 for(i=0,j=0;;)
 {
  b=a[i]+a[i+++1]+1;
  if(b>n)
   break;
  while(b>a[j])
   ++j;
  if(a[j]==b)
   --k;
 }
 std::cout<<(k>0?"NO":"YES");
}
