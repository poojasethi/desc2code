#include <cstdio>

using namespace std;

int main() {
	int n,k;
	scanf("%d%d",&n,&k);
	bool prime[1010];
	bool check[1010];
	for(int i=2;i<=n;i++) prime[i]=true, check[i]=false;
	int prev=100000;
	int res=0;
	for(int i=2;i<=n;i++) {
		if (prime[i]) {
			if (prev+i+1<=n) check[prev+i+1]=true;
			prev=i;
			if (check[i]) res++;
		}
		for(int j=i*i;j<=n;j+=i) prime[j]=false;
	}
	printf(res>=k?"YES\n":"NO\n");
	return 0;
}
