#include<iostream>
using namespace :: std;
int main()
{
int x,i,j,s,z,d=0,k=0,b=0,a[168]={0},n=0;
cin >>x>>s;
for(i=2;i<=x;i++)
{
for(j=1;j<=i;j++)
{
if(i%j==0)
b++;
}
if(b==2)
{
a[k]=i;
k++;
n++;
}
b=0;
}


for(j=0;j<n-1;j++)
{
z=a[j]+a[j+1]+1;
for(i=j+1;i<n;i++)
{
if(z==a[i])
d++;
}
}



if(d>=s)
cout <<"YES";
else
cout <<"NO";


return 0;
}
