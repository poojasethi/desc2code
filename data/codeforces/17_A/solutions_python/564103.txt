n, k = map(int, raw_input().split())
p = []
for i in range(2, n + 1):
    for j in range(2, i - 1):
        if i % j == 0:
            break
    else:
        p.append(i)
t = 0
for i in range(len(p) - 1):
    if p[i] + p[i + 1] + 1 > n:
        break
    if p[i] + p[i + 1] + 1 in p:
        t += 1

if t >= k:
    print "YES"
else:
    print "NO"
