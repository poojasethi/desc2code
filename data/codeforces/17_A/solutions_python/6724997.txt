def prime_table(n):
  list = [True for _ in xrange(n + 1)]
  i = 2
  while i * i <= n:
    if list[i]:
      j = i + i
      while j <= n:
        list[j] = False
        j += i
    i += 1
 
  table = [i for i in xrange(n + 1) if list[i] and i >= 2]
  return table

n,k = map(int, raw_input().split())
pt= prime_table(n)
count=0
for i in range(len(pt)):
  for j in range(i):
	if pt[j] + pt[j+1] +1 == pt[i]:
		count = count + 1 

print "YES" if  k<=count else "NO"
