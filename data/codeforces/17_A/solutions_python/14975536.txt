def eratosthenes2(n):
    multiples = set()
    for i in range(2, n+1):
        if i not in multiples:
            yield i
            multiples.update(range(i*i, n+1, i))
 
p = list(eratosthenes2(1001))
n, k = map(int, raw_input().split())
i = 0
c = 0
while 1:
    if p[i] + p[i+1] + 1 > n: break
    if p[i] + p[i+1] + 1 in p:
        c += 1
        if c >= k:
            break
    i += 1    
print "YES"  if c >= k else "NO"
