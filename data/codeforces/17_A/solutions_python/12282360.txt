primes,cprimes = [],set()
n,k = map(int,raw_input().split())

def is_prime(n):
    if n == 1: return False
    for x in xrange(2,int(n**0.5)+1):
        if n % x == 0:
            return False
    return True

for x in xrange(1,n+1):
    if is_prime(x):
        primes.append(x)

for i in xrange(len(primes)-1):
    cprimes.add(primes[i]+primes[i+1]+1)

for prime in primes:
    if prime in cprimes:
        k -= 1

print ["NO","YES"][k <= 0]
