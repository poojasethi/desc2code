n, k = map(int, raw_input().split(' '))
P = []
p = [1]*(n+1)
p[0] = p[1] = 0
for i in xrange(n+1):
   if p[i]:
      for j in xrange(i*i, n+1, i):
	 p[j] = 0
      P.append(i)
for i in xrange(2, n+1):
   if p[i]:
      for j in xrange(len(P)-1):
	 if i == P[j]+P[j+1]+1:
	    k -= 1
	    break
print "YES" if k <= 0 else "NO"
