a, b = map(int, raw_input().split())
c = [0]*(a+1)
d = [2]
for k in range(3, a+1, 2):
    if c[k] == 0:
        d.append(k)
        for j in range(k*3, a+1, k*2):
            c[j] += 1
e = [d[k]+d[k+1] for k in range(len(d)-1)]
f = 0
for k in range(len(d)):
    if d[k]-1 in e: f += 1
if f >= b: print 'YES'
else: print 'NO'
