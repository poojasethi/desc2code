n,k = [int(x) for x in raw_input().split()]

primes = []
for i in xrange(2, n+1):
    for p in primes:
        if i%p == 0: break
    else:
        primes += [i]

cnt = 0
for i in xrange(len(primes)-1):
    if primes[i] + primes[i+1] + 1 in primes : cnt += 1

if cnt >= k:
    print "YES"
else:
    print "NO"
