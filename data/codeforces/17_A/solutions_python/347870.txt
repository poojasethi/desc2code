def  isprime(x):
    p = 2
    while p*p <= x:
        if x%p == 0:
            return 0
        p = p + 1
    return 1

L = map(int,raw_input().strip().split())
n = L[0]
k = L[1]
prime = []
cnt = 0
for i in range (2,n+1):
    if isprime(i):
        prime.append(i)
for i in range (0,len(prime)-1):
    if prime[i] + prime[i + 1] + 1 <= n and isprime(prime[i] + prime[i + 1] + 1):
        cnt = cnt + 1
if cnt >= k:
    print 'YES'
else:
    print 'NO'
