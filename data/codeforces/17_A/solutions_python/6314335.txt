import math

n,k=map(int,raw_input().split())

primes=[x for x in xrange(2,n+1) if all(x%t for t in xrange(2,int(math.sqrt(x))+1))]
#print primes
a=0
for i in range(len(primes)-1):
   if primes[i]+primes[i+1]+1 in primes:
      a+=1
      
if a>=k:
   print 'YES'
else:
   print 'NO'
