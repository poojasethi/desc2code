from itertools import *
from collections import defaultdict

def isprime(n):
    for x in range(2, int(n**0.5)+1):
        if n % x == 0:
            return False
    return True

a, b = map(int, raw_input().split())
primes = [i for i in xrange(a) if isprime(i)]
counter = 0
for i in xrange(len(primes)-1):
    if isprime(primes[i]+primes[i+1]+1) and primes[i]+primes[i+1]+1 <= a:
        counter += 1
if counter >= b:
    print "YES"
else:
    print "NO"
