# 17A
n, k = map(int, raw_input().split())
mask = 2 * [False] + (n - 1) * [True]
prime = []
for i, x in enumerate(mask):
    if (x):
        prime.append(i)
        for j in xrange(2 * i, len(mask), i):
            mask[j] = False
cnt = 0
for i in xrange(len(prime) - 1):
    if prime[i] + prime[i + 1] + 1 in prime:
        cnt += 1
print 'YES' if cnt >= k else 'NO'
