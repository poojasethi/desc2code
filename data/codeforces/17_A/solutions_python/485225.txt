n,k = map(int,raw_input().split());
check = [True]*2000;
check[0] = check[1] = False;
for i in range(2,2000):
	if check[i]:
		for j in range(2,2000//i):
			check[i*j] = False;
primes = [];
for i in range(1001):
	if check[i]:
		primes.append(i);
sums = set();
for i in range(len(primes)-1):
	if check[primes[i]+primes[i+1]+1]:
		sums.add(primes[i]+primes[i+1]+1);
count = 0;
for i in range(2,n+1):
	if i in sums: count+=1;
if count >= k: print "YES";
else: print "NO";
