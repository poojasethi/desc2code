n, k = map(int, raw_input().split())

primes, is_prime = [], [True for i in xrange(n + 1)]

for i in xrange(2, n + 1):
	if is_prime[i]:
		primes.append(i)
		j = 2 * i
		while j <= n:
			is_prime[j] = False
			j += i

count = 0
for i in xrange(len(primes) - 1):
	if count == k:
		break
	tmp = primes[i] + primes[i + 1] + 1
	if tmp > n:
		break
	if is_prime[tmp]:
		count += 1

print 'YES' if count == k else 'NO'
