def isPrime(x):
	if x==1:
		return 0
	i=2;
	while(i*i<=x):
		if x%i==0:
			return 0;
		i+=1;
	return 1;

inpt=raw_input().split();
n=int(inpt[0]);
k=int(inpt[1]);
primes=[];
for i in range(2,n+1):
	if isPrime(i)==1:
		primes=primes+[i];
number=0;
for i in range(0,len(primes)-1):
	if primes[i]+primes[i+1]>n:
		break;
	if isPrime(primes[i]+primes[i+1]+1):
		number+=1;
if(number>=k):
	print "YES"
else:
	print "NO"
