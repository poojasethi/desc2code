n, k = map(int, raw_input().split())
sieve = [-1] * (n + 1)
for i in range(3, n + 1, 2):
    if sieve[i] == -1:
        sieve[i] = 1
        for j in range(i * i, n + 1, i):
            sieve[j] = 0
p = 0
q = 2
z = 0
for i in range(3, n + 1, 2):
    if sieve[i] == 1:
        p, q = q, i
        if p + q + 1 > n:
            break
        if sieve[p + q + 1] == 1:
            z += 1
if z >= k:
    print 'YES'
else:
    print 'NO'
