def sieve(n):
        s = xrange(3, n + 1, 2)
        r = set(s)
        [r.difference_update(xrange(n << 1, s[-1] + 1, n)) for n in s if n in r]
        return r.union([2])
n,k=map(int,raw_input().split())
if k==0:
    print 'YES'
    exit()
A=sorted(list(sieve(n+1)))
c=0
for x in A[1:]:
    y=x-1
    for i in range(len(A)-1):
        if y==A[i]+A[i+1]:
            c+=1
            if c==k:
                print 'YES'
                exit()
print 'NO' if c!=k else 'YES'
