n, k = map(int, raw_input().strip().split())

co = 0
nums = []
for i in xrange(2, n+1):
    for j in xrange(2, i):
        if i % j == 0:
            break
    else:
        nums.append(i)

for i in range(1,len(nums)):
    a = nums[i-1]+nums[i]+1
    if a in nums:
        co = co+1

if co >= k:
    print 'YES'
else:
    print 'NO'
