from string import split
def prime_nums(n, p=2):
	nums=[x for x in range(2,n+1)]
	while p<=n:
		multply=2
		while True:
			n_n=multply*p
			if n_n<=n:
				if n_n in nums:
					nums.remove(n_n)
			else:
				break
			multply+=1
		p+=1
	return nums

a, b=map(int , split(raw_input()))
m=prime_nums(a)
new_m=[]
for x in range(0, len(m)-1):
	pff=m[x]+m[x+1]+1
	if pff not in new_m and pff in m:
		new_m.append(pff)
if len(new_m)>=b:
	print "YES"
else:
	print "NO"
