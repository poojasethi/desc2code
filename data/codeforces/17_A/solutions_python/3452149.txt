import math
isprime = [True]*1001
for i in range(2, int(math.sqrt(1000))+1):
    if isprime[i]:
        j = i*i
        while j <= 1000:
            isprime[j] = False
            j += i


prime = [i for i in range(2, 1001) if isprime[i]]

n, k = map(int, raw_input().split())
count = 0

for i in range(0, len(prime)-1):
    s = prime[i] + prime[i+1] + 1
    if s in prime and s <= n:
        count += 1
    if count >= k:
        break

if count >=k:
    print "YES"
else:
    print "NO"
