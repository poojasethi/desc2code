#!/usr/bin/env python
from string import split

n, k = map(int, split(raw_input()))

resheto = [True] * 1001
for i in range(2, 1000):
	c = 2
	while c * i <= 1000:
		resheto[c * i] = False
		c += 1

primes = []
for i in range(2, 1000):
	if resheto[i]:
		primes.append(i)

r = 0
for i in range(2, n + 1):
	if i in primes:
		a = i - 1
		j = 0
		while primes[j] + primes[j + 1] < a:
			j += 1
		if primes[j] + primes[j + 1] == a:
			r += 1

if r >= k:
	print 'YES'
else:
	print 'NO'
