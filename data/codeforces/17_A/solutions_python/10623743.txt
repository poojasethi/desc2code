def isPrimeNum(x):
	for i in xrange(2, int(x**0.5)+1):
		if x%i == 0:
			return False

	return True

n,k = map(int, raw_input().split())

primeNums = []

totalNums = 0

for i in xrange(2, n+1):
	if isPrimeNum(i):
		primeNums.append(i)

for i in xrange(len(primeNums)-1):
	val = primeNums[i] + primeNums[i+1] + 1
	if val <= n and isPrimeNum(val):
		totalNums += 1

if totalNums >= k:
	print 'YES'
else:
	print 'NO'
