es = [2, 3, 5, 7, 11]
for i in range(12, 1001):
    g = True
    for j in es:
        if i % j == 0:
            g = False
            break
    if g: es.append(i)

n, k = map(int, raw_input().split())
count = 0
for x in range(len(es)):
    i = es[x]
    if i > n: break
    g = False
    for j in range(len(es) - 1):
        if i == es[j] + es[j + 1] + 1:
            g = True
            break
    if g: count += 1
print(["NO", "YES"][count >= k])
