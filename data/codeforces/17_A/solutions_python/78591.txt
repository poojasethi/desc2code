n,k = [int(x) for x in raw_input().split()]

prime = []
for i in xrange(2,n+1):
        if all((i%p!=0 for p in prime)):
                prime += [i]

c = 0
for i in xrange(len(prime)-1):
        if prime[i]+prime[i+1]+1 in prime:
                c += 1

if c >= k:
        print "YES"
else:
        print "NO"
