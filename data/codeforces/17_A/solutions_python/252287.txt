N,K = map(int,raw_input().split())

primes = []

def isPrime(n):
	for i in xrange(2,int((n**0.5)+1)):
		if i == n:
			return True
		if n%i == 0:
			return False
	return True

def findPrimes():
	for i in xrange(2,N+1):
		if isPrime(i):
			primes.append(i)

findPrimes()

numNeighPrimes = set()

for i in xrange(len(primes)-1):
	if primes[i] + primes[i+1] + 1 in primes:
		numNeighPrimes.add(primes[i]+primes[i+1]+1)

if len(numNeighPrimes) >= K:
	print "YES"
else:
	print "NO"
