n=input()
num=n*(n-1)/2-1
d=[[] for _ in xrange(n+1)]
won=[0 for _ in xrange(n+1)]
for _ in xrange(num):
    x,y=map(int,raw_input().split())
    won[x]+=1
    x,y=min(x,y),max(x,y)    
    d[x].append(y)
for i in xrange(1,n+1):
    if len(d[i])!=n-i:
        j=i+1
        for z in sorted(d[i]):
            if z==j:j+=1
            else:break
        if won[i]>won[j]:
            print i,j
        else:print j,i
        break
