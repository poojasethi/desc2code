from itertools import *
n = input ()
a = [map (int, raw_input ().split ()) for _ in xrange (n * (n - 1) / 2 - 1)]
x, y = filter (lambda x: a.count (list (x)) + a.count (list (x[::-1])) == 0, combinations (range (1, n + 1), 2))[0]
print str (x) + ' ' + str (y) if len ([z for z in xrange (1, n + 1) if a.count ([x, z]) and a.count ([z, y])]) else str (y) + ' ' + str (x)
