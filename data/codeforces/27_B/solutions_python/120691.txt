n = input()
w = {}
s = set()
for _ in xrange(n*(n-1)/2 - 1):
	a, b = map(int, raw_input().split())
	w[a] = w.get(a, 0) + 1
	if a > b: s.add((b, a))
	else: s.add((a, b))
for a in xrange(1, n+1):
	for b in xrange(a+1, n+1):
		if a == b: continue
		_a, _b = a, b
		if _a > _b: _a, _b = _b, _a
		if not (_a, _b) in s:
			if w.get(_a, 0) > w.get(_b, 0):
				print _a, _b
			else:
				print _b, _a
