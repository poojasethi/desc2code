def main():
	n = input()
	t = [0] * (n + 1)
	win = [0] * (n + 1)
	for i in xrange(n * (n - 1) / 2 - 1):
		x, y = map(int, raw_input().split())
		t[x] += 1
		t[y] += 1
		win[x] += 1
	a, b = [i for i in xrange(1, n + 1) if t[i] == n - 2]
	if win[a] > win[b]: print a, b
	else: print b, a
 
if __name__ == "__main__": main()
