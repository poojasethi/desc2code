n=input()
a=[0 for i in xrange(n+1)]
b=[0 for i in xrange(n+1)]
for _ in xrange(n*(n-1)/2-1):
    x,y=map(int,raw_input().split())
    a[x]+=1
    b[y]+=1
c=[]
for i in xrange(n+1):
    if a[i]+b[i]==n-2:
        c.append(i)

if a[c[0]]>a[c[1]]:
    print '%d %d'%(c[0],c[1])
else:
    print '%d %d'%(c[1],c[0])
