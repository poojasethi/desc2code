n = input()
w = [0 for _ in xrange(n+1)]
l = [0 for _ in xrange(n+1)]
for _ in xrange(n*(n-1)/2-1):
    a,b = map(int, raw_input().split())
    w[a] += 1
    l[b] += 1
ans = []
for i in xrange(1, n+1):
    if w[i]+l[i] == n-2:
        ans.append(i)
if w[ans[0]] < w[ans[1]]:
    ans[0],ans[1] = ans[1],ans[0]
print "%d %d" % (ans[0], ans[1])
