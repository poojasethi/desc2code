n = int(raw_input())
less = [n*[False] for _ in range(n)]
cnt = n*[0]
for _ in range(n*(n-1)/2-1):
    i, j = map(lambda s: int(s)-1, raw_input().split())
    less[i][j] = True
    cnt[i] += 1
    cnt[j] += 1

x = []
for i in range(n):
    if cnt[i] == n-2:
        x.append(i)
x, y = x

for i in range(n):
    for j in range(n):
        for k in range(n):
            if less[i][k] and less[k][j]:
                less[i][j] = True
if less[x][y]:
    print x+1, y+1
else:
    print y+1, x+1
