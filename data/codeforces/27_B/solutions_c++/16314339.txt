#include <bits/stdc++.h>

using namespace std;
int x[55],y[55],w[55];

int main()
{
    int n ;
    cin>>n;
    int m=(n*(n-1))/2-1;
    for(int i=0;i<m;i++)
        {
            int a,b;
            cin>>a>>b;
            x[a]++,y[b]++,w[a]++;
        }
        int win[2],in=0;
    for(int i=1;i<=n;i++)
      if(x[i]+y[i]==n-2)
        win[in++]=i;
      if(w[win[0]]>w[win[1]])
        cout<<win[0]<< " "<<win[1]<<endl;
      else
        cout<<win[1]<< " "<<win[0]<<endl;
}
