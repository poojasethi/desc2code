#include <iostream>
#include <algorithm>
using namespace std;
int a[51];
int main()
{
	int n, x, y, t, i, j, u=0, v=0;
	cin>>n;
	for (i=1; i<n; i++) for (j=i+1; j<=n; j++) {
		u+=i;
		v+=j;
	}
	for (i=0; i<n*(n-1)/2-1; i++) {
		cin>>x>>y;
		a[x]++;
		u-=min(x,y);
		v-=max(x,y);
	}
	if (a[u]>a[v]) cout<<u<<" "<<v<<endl;
	else cout<<v<<" "<<u<<endl;
	return 0;
}
