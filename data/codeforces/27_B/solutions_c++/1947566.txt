#include<iostream>
#include<string.h>
using namespace std;

int main()
{
 bool w[55][55];
 int win[55],N,i,j,x,y;
 memset(w,0,sizeof w);
 memset(win,0,sizeof win);
 
 cin>>N;
 for(i=0;i<((N * (N-1)) / 2) -1;i++)
  {
   cin>>x>>y;
   w[x-1][y-1] = w[y-1][x-1] = 1; 
   win[x-1]++;
  }
 for(i=0;i<N;i++)
  for(j=0;j<N;j++)
  {
   if(i!=j && w[i][j]!=1)goto yla; 
  }
  yla:
      if(win[i]>=win[j])cout<<i+1<<" "<<j+1<<endl;
      else cout<<j+1<<" "<<i+1<<endl;
      return 0;
 
}
