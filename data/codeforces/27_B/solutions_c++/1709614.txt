#include<iostream>
using namespace std;
int n,x[10000],y[10000],i,j,m,a[10000],b[10000],k,q,p;
int main()
{
cin>>n;
for (i=1; i<=n*(n-1)/2-1; i++)
{
cin>>x[i]>>y[i];    
a[x[i]]++;
b[y[i]]++;
}    
for (i=1; i<=n; i++)
if (a[i]+b[i]!=n-1) {k=i;break; }    
for (i=k+1; i<=n; i++)    
if (a[i]+b[i]!=n-1) {p=i; break;}
if (a[p]>a[k]) swap(k,p);
cout<<k<<" "<<p<<endl; //system("pause");
}
