#include<stdio.h>
#include<string.h>
int i,j,k,m,n,w[51],l[51],x,y;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n*(n-1)/2-1;i++){
		scanf("%d%d",&x,&y);
		w[x]++;
		l[y]++;
	}
	x=0;
	for(i=1;i<=n;i++)
		if(w[i]+l[i]<n-1){
			if(x==0) 
				x=i;
			else
				y=i;
		}
	if(w[x]>w[y])
		printf("%d %d\n",x,y);
	else
		printf("%d %d\n",y,x);
}
