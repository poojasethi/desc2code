#include<stdio.h>
#include<string.h>
int main()
{
   int n,x,y,i;
   int a[55][55],b[55];
  
   while(~scanf("%d",&n)){ 
	   memset(a,0,sizeof(a));
   for(i=1;i<=n*(n-1)/2-1;i++){
      scanf("%d %d",&x,&y);
	  a[x][y]=1;
	  a[y][x]=1;
	  b[x]++;
   }
   for(x=1;x<=n;x++){
	   for(y=x+1;y<=n;y++){
	         if(a[x][y]==0)
			{
				if(b[x]>=b[y])
				printf("%d %d\n",x,y);
				else 
				printf("%d %d\n",y,x);
			}
		}

	   }
   }
return 0;
}
