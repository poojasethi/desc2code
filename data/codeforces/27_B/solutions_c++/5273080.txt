#include<iostream>
using namespace std;
#include<string>
#include<cstdio>
#include<map>
int main(){

	int t,n,k=1,x,y;
	map<int,int>M,W;
	cin>>t;
	n = t-1;
	int yy = (t*(t-1))/2-1;
	while(yy--){
		cin>>x>>y;
		M[x]++;M[y]++;
		W[x]++;
	}
	int i=1;
	for (; i <= n+1; i++)
	{
		if(M[i]!=n){x = i;break;}
	}
	i++;
	for (; i<= n+1; i++)
	{
		if(M[i]!=n){y = i;break;}
	}
	if(W[x]>=W[y])cout<<x<<" "<<y<<endl;
	else cout<<y<<" "<<x<<endl;
	return 0;
}
