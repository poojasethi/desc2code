#include <iostream>
#include <map>
#include <algorithm>
#include <cstring>
using namespace std;
int main()
{
	int n;
	cin >> n;
	int win[n+1];
	int play[n+1];
	memset(win,0,sizeof win);
	memset(play,0,sizeof play);
	
	for(int i=0;i<(n*(n-1)/2)-1;i++)
	{
		int x,y;
		cin >> x >> y;
		win[x]++;
		play[x]++;play[y]++;
	}
	int h=0,k=0;
	for(int i=1;i<=n;i++)
	{
		if(play[i]==n-2)
		{
			if(!h) h=i;
			else k=i;
		}
	}
	
	cout << ((win[h]>win[k])?h:k) << " " << ((win[h]>win[k])?k:h);
	
}
