#include<cstdio>
bool us[51][51];
int d[51];
int main(){
	int n,u,v;
	scanf("%d",&n);
	
	for(int i=0;i<(n-1)*n/2-1;i++){
		scanf("%d%d",&u,&v);
		us[v][u]=us[u][v]=true;
		d[u]++,d[v]--;
	}
	for(int i=1;i<n;i++)
		for(int j=i+1;j<=n;j++)
			if(!us[i][j]){
				if(d[i]>d[j])
					printf("%d %d",i,j);
				else
					printf("%d %d",j,i);
				return 0;
			}
	return 0;
}
