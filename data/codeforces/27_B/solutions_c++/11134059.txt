#include<stdio.h>
int a[55],b[55];

int main()
{
	int n ;
	int x ,y ;
	scanf("%d",&n);
	int i  ;
	for(i=1;i<n*(n-1)/2;i++) 
	{
		scanf("%d%d",&x,&y) ;
		a[x]++; //i赢的场次 
		b[y]++; //i 输的场次 
	}
	int p1=0 ,p2= 0 ;
	i=1 ;
	while(1)
	{
		if(a[i]+b[i]!=n-1)
		{
			p1= i ;
			break ;
		}
		i++ ;
	} 
	i++ ;
	while(1)
	{
		if(a[i]+b[i]!=n-1)
		{
			p2= i ;
			break ;
		}
		i++;
	}
	if(a[p1]>a[p2])
	printf("%d %d",p1,p2);
	else 
	printf("%d %d",p2,p1);
	return 0 ;	
}
