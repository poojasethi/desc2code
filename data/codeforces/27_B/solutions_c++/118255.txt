#include<iostream>
using namespace std;
int main(){
int n,k;
int b,c,f=0;
int d[51],bord[51],s[51],l[2];
for(int i=0;i<51;i++)
{
d[i]=0;
bord[i]=0;
}
cin>>n;
k=(n*(n-1))/2-1;
for(int i=0;i<k;i++)
{
cin>>b>>c;
d[b]++;
d[c]++;
bord[b]++;
}

for(int i=1;i<=n;i++)
{
if(d[i]==n-1)
{
s[bord[i]]=i;
}
else
{
l[f]=i;
f++;
}
}
if(bord[l[0]]>bord[l[1]])
cout<<l[0]<<" "<<l[1]<<endl;
else
cout<<l[1]<<" "<<l[0]<<endl;
return 0;
}
