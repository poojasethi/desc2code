#include<cstdio>
#include<cstring>

int main()
{
   int n,cnt[100],win[100];
   memset(cnt,0 ,sizeof(cnt));
  memset(win,0,sizeof(win)) ;
   scanf("%d",&n)  ;
   for (int i=0;i<n*(n-1)/2-1;++i) {
      int x,y; 
     scanf("%d%d",&x,&y) ;
     ++cnt[x] ;
    ++cnt[y] ;
   ++win[x] ;
   } 
  int a=-1,b;
 for (int i=1;i<=n;++i)  
if (cnt[i]<n-1) {
    if (a==-1) a=i;
    else b=i; 
}

if (win[a]>win[b]) printf("%d %d\n",a,b);
else printf("%d %d\n",b,a);
return 0;
}
