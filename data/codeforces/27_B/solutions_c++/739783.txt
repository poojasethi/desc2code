#include<stdio.h>
using namespace std;
int y[51],z[51];
int main()
{
    int i,j,n,a,b;
    scanf("%d",&n);
    for(i=1; i<=((n*(n-1))/2)-1; i++)
{
    scanf("%d %d",&a,&b);
    y[a]++;y[b]++;z[a]++;
}
    for(i=1; i<=n; i++)
    if(y[i]<(n-1)) break;

    for(j=i+1; j<=n; j++)
    if(y[j]<(n-1)) break;
    
    if(z[i]>z[j])printf("%d %d",i,j); else printf("%d %d",j,i);
    getchar();    getchar();
    return 0;
}
