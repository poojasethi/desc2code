#include <bits/stdc++.h>
using namespace std ;

int w[55] , l[55] , n , k , p , x , y , yo[3] , c ;
int main()
{
   cin >> n ;

   k = n * ( n - 1 ) / 2 - 1 ;

   p = n - 1 ;

   for( int i = 0 ; i < k ; i++ )
   {
       cin >> x >> y ;
       w[x]++ ;
       l[y]++ ;
   }

   for( int i = 1 ; i <= n ; i++ )
   {
       if( w[i] + l[i] != p ) yo[c] = i , c++ ;
   }

   if( w[yo[0]] > w[yo[1]] )cout << yo[0] << " "<<yo[1] ;
   else cout << yo[1] << " "<<yo[0] ;

return 0 ;
}
