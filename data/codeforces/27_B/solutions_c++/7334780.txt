#include<stdio.h>
int main()
{
	int w[55]={0},l[55]={0};
	int win,lose;
	int n;
	int i,j=0;
	int a[2];
	scanf(" %d",&n);
	for(i=0;i<(n*(n-1)/2-1);i++)
	{
		scanf(" %d %d",&win,&lose);
		w[win]++;
		l[lose]++;
	}
	for(i=0;i<=n;i++)
	{
		if(w[i]+l[i]==n-2)
			a[j++]=i;
	}
	if(w[a[1]]>=w[a[0]])
	printf("%d %d\n",a[1],a[0]);
	else printf("%d %d\n",a[0],a[1]);
	return 0;
}
