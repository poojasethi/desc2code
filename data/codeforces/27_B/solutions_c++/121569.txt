#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int n,i,j,k,l;
	int G[51],W[51];
	cin>>n;
	for(i=1;i<=n;i++)
		G[i]=W[i]=0;
	k=(n*(n-1))/2;
	for(i=1;i<k;i++)
	{
		cin>>j>>l;	
		G[j]++;
		G[l]++;
		W[j]++;
	}
	for(i=1;G[i]==n-1;i++);
	j=i;
	for(i++;G[i]==n-1;i++);
	if(W[i]<=W[j])
		cout<<j<<" "<<i;
	else
		cout<<i<<" "<<j;			
	return 0;		
}
