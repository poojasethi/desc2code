#include <iostream>
#include <map>
#include <vector>
using namespace std;
int main() {
	int n,a,b,aux;
	cin>>n;
	vector<pair<int,int> > v(n+1);
	aux=n;
	n=n*(n-1)/2-1;
	while(n--) {
		cin>>a>>b;
		v[a].first++;
		v[a].second++;
		v[b].first++;
	}
	n=aux;
	a=0;
	b=0;
	for(int i=1;i<=n;i++) {
		if(v[i].first==(n-2)) {
			if(a==0) a=i;
			else b=i;
		}
	}
	if(v[a].second>v[b].second) cout<<a<<" "<<b<<endl;
	else cout<<b<<" "<<a<<endl;

	return 0;
}
