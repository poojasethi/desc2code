#include<stdio.h>
#include<string.h>
int vis[1500][1500];
int main()
{
	int n,a,b,i,j,win[51];
	while(scanf("%d",&n)!=EOF)
	{
		memset(win,0,sizeof(win));
		memset(vis,0,sizeof(vis));
		for(i=1;i*2<n*(n-1);i++)
		{
			scanf("%d%d",&a,&b);
			win[a]++;
			vis[a][b]++;
			vis[b][a]++;
		}
		for(i=1;i<=n;i++)
		{
			for(j=i+1;j<=n;j++)
			{
				if(vis[i][j]==0)
				{
					if(win[i]>=win[j])
						printf("%d %d\n",i,j);
					else 
						printf("%d %d\n",j,i);
				}
			}
		}
	}
	return 0;
}
