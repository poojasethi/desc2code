#include <stdio.h>
#include <string.h>
int main(){
	int n;
	while(scanf("%d",&n)!=EOF){
		int i,j,x,y,a[109][109]={0},d[109]={0},b,c;
		for(i=1;i<=n*(n-1)/2-1;i++){
			scanf("%d%d",&x,&y);
			a[x][y]=1;a[y][x]=1;
			d[x]++;
		}
		for(i=1;i<=n;i++) a[i][i]=1;
		for(i=1;i<=n;i++)
			for(j=1;j<=n;j++)
				if(a[i][j]==0) b=i,c=j;
		if(d[b]>=d[c]) printf("%d %d\n",b,c);
		else printf("%d %d\n",c,b);
	}
	return 0;
}
