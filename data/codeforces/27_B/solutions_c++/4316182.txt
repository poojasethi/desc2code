#include<iostream>
#include<cstdio>

using namespace std;
int n;
int dizi[100];
bool H[55][55];
int main()
{
	int i,j,a,b;
	cin>>n;
	for(i=0; i<n*(n-1)/2-1; i++)
	{
		cin>>a>>b;
		H[a][b]=H[b][a]=true;
		dizi[a]++;
	}
	for(i=1; i<=n; i++)
		for(j=1; j<=n; j++)
			if(i!=j)
			{
				if(!H[i][j]){
					if(dizi[i]>=dizi[j])
					printf("%d %d\n",i,j);
					else
						printf("%d %d\n",j,i);

					return 0;
				}
			}




	return 0;
}
