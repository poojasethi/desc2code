#include <iostream>
#include <algorithm>
using namespace std;

long long arr[100], ar2[100];

int main(){
	long long n, t, a, b, i;
	cin >> n;
	t = (n*(n-1))/2 -1;
	while(t--){
		cin >> a >> b;
		arr[a]++;
		ar2[a]++;
		ar2[b]++;
	}
	a = b = 0;
	for(i=1;i<=n;i++){
		if(ar2[i]==n-2){
			if(!a) a = i;
			else if(!b){
				b = i;
				break;
			}
		}
	}
	if(arr[a]<arr[b]){
		cout << b << ' ' << a << '\n';
	}
	else cout << a << ' ' << b << '\n';
	return 0;
}
