#include<cstdio>

bool vis[100];
int n,x,y;
bool e[100][100];

bool dfs(int x,int y){
	if (x==y) return 1;
	vis[x]=1;
	for (int i=1;i<=n;i++)
		if (e[x][i] && !vis[i] && dfs(i,y)) return 1;		
	return 0;
}

int main(){
	scanf("%d",&n);
	for (int i=1;i<(n-1)*n/2;i++){
		scanf("%d%d",&x,&y);
		e[x][y]=1;
	}
	for (int i=1;i<=n-1;i++)
		for (int j=i+1;j<=n;j++)
			if (!e[i][j] && !e[j][i]){
				if (dfs(i,j)) printf("%d %d\n",i,j);
				else printf("%d %d\n",j,i);
			}
	return 0;
}
