#include<stdlib.h>
#include<string.h>
#include<stdio.h>
int main()
{
int a[55][55];
int b[55];
int k,i,t,j,n;
memset(a,0,sizeof(a));
memset(b,0,sizeof(b));
 scanf("%d",&n);
for(i=1;i<=(n*(n-1)/2-1);i++)
{  scanf("%d%d",&k,&t);
   a[k][t]=1;
   a[t][k]=1; 
   b[k]++; }
 for(i=1;i<=n;i++)
		for(j=i+1;j<=n;j++)
		{
			if(a[i][j]==0)
			{
				if(b[i]>=b[j])
				printf("%d %d\n",i,j);
				else 
				printf("%d %d\n",j,i);
			}
		}
  return 0;
}
