#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std ;

int W[50] ;
int J[50] ;

int main() {
    int N ;
    int i, j ;
    int a, b ;

    scanf("%d",&N ) ;
    for( i=1; i<(N*(N-1))/2; i++  )  {
        scanf("%d%d",&a,&b ) ;
        W[--a] ++ ;
        J[a] ++ ;
        J[--b] ++ ;
    }

    for( i=0; J[i]==N-1; i++ ) {}
    for( j=i+1; J[j]==N-1; j++ ) {}

    if( W[i] < W[j] )
        swap( i, j ) ;
    printf("%d %d\n",i+1,j+1 ) ;

    return 0 ;
}
