#include <iostream>
#include <algorithm>
using namespace std;
int a[51];
int main()
{
	int n, i, j, v, w, x, y;
	v=0;
	w=0;
	cin >> n;
	for(i=1; i<n; i++){
		for(j=i+1; j<=n; j++){
			v+=i;
			w+=j;
		}
	}
	for(i=0;i<n*(n-1)/2-1; i++){
		cin >> x >> y;
		a[x]++;
		if(x>y){
			x+=y;
			y=x-y;
			x-=y;
		}
		v-=x;
		w-=y;
	}
	if(a[v]>a[w])
		cout << v << " " << w << endl;
	else
		cout << w << " " << v << endl;
	return 0;
}
