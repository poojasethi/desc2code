#include<cstdio>
int win[50];
bool know[50][50];
int main()
{
	int n,i,j,t1,t2;
	scanf("%d",&n);
	for (i=0;i<n*(n-1)/2-1;i++)
	{
		scanf("%d %d",&t1,&t2);
		win[t1-1]++;
		know[t1-1][t2-1]=true;
		know[t2-1][t1-1]=true;
	}
	for (i=0;i<n-1;i++)
		for (j=i+1;j<n;j++)
			if (know[i][j]==false)
			{
				if (win[i]>win[j])
					printf("%d %d",i+1,j+1);
				else
					printf("%d %d",j+1,i+1);
				return 0;
			}
}
