#include<iostream>
using namespace std;
int main()
{
	int n, x, y, k;
	int w[55][55], p[55];
	cin>>n;
	for(int i = 1; i <= n; i++)
	{
		p[i] = 0;
		for(int j = 1; j <= n; j++)
			w[i][j] = -1;
	}
	for(int i = 1; i <= n * (n - 1) / 2 - 1; i++)
	{
		cin>>x>>y;
		p[x]++;
		w[x][y] = 1;
		w[y][x] = 0;		
	}
	for(int i = 1; i <= n; i++)
		for(int j = 1; j <= n; j++)
			if (i != j && w[i][j] == -1)
			{
				if (p[i] >= p[j]) cout<<i<<" "<<j; else cout<<j<<" "<<i;
				return 0;
			}
}
