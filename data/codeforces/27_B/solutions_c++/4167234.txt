#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
#include<map>
#include<cstdlib>
#include<cstring>
#include<string>

using namespace std;

bool a[100][100];
int b[100];
int main(){
	int n,i,j,tot,x,y;
	scanf("%d",&n);
	tot=n*(n-1)/2;
	for(i=1;i<tot;++i){
		scanf("%d%d",&x,&y);
		a[x][y]=true;
		a[y][x]=true;
		++b[x];
	}
	for(i=1;i<=n;++i){
		for(j=1;j<=n;++j)if(i!=j && !a[i][j]){
			if(b[i]<b[j])swap(i,j);
			printf("%d %d\n",i,j);
			return 0;
		}
	}
}
