#include<iostream>
using namespace std;

struct sajal{
	int m;int w;
};sajal a[104];
int main(){
	int n;
	cin>>n;long x,y;
	long h=n*(n-1)/2;h=h-1;
	for(int i=0;i<h;i++){cin>>x>>y;
	a[x].m=a[x].m+1;a[y].m=a[y].m+1;a[x].w=a[x].w+1;
	}int l=0;
	for(int i=1;i<=n;i++){
		if(a[i].m!=(n-1)){if(l==0)x=i;
		else y=i;
			l=1;
		}
	}int g=a[x].w;int r=a[y].w;
	if(g<r)cout<<y<<" "<<x<<endl;
	else cout<<x<<" "<<y<<endl;
}
