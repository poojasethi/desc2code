#include <iostream>
using namespace std;

int A[100],B[100];
bool C[100];
int main(){
	int n;
	int i,j,k;
	int x,y;
	
	cin>>n;
	k = (n*(n-1))/2;
	for(i=1;i<k;i++){
		cin>>x>>y;
		A[x]++; A[y]++;
		B[x]++;
	}
	x = y = -1;	
	for(i=1;i<=n;i++){
		if(A[i] != (n-1)){
			if(x == -1)
				x = i;
			else
				y = i;
		}
		else	
			C[B[i]] = true;
	}
	if(!C[B[x]+1] && !C[B[y]])
		cout<<x<<" "<<y<<endl;
	else	
		cout<<y<<" "<<x<<endl;
	return 0;
}
