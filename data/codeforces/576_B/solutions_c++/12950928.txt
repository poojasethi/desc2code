#include<bits/stdc++.h>
using namespace std;
int n,p[100005],col[100005];
bool vis[100005];
int main(){
  scanf("%d",&n);
  for(int i=1;i<=n;i++) scanf("%d",&p[i]);
  int one=-1,two=-1;
  for(int i=1;i<=n;i++){
    if(p[i]==i) one=i;
    else if(p[p[i]]==i) two=i;
  }
  if(one!=-1){
    printf("YES\n");
    for(int i=1;i<=n;i++){
      if(i==one) continue;
      printf("%d %d\n",one,i);
    }
  }
  else if(two==-1) printf("NO\n");
  else{
    memset(vis,0,sizeof(vis));
    vis[two]=vis[p[two]]=true;
    for(int i=1;i<=n;i++){
      if(vis[i]) continue;
      int j=i,len=0;
      while(!vis[j]){
        vis[j]=true;
        col[j]=len%2;
        len++;
        j=p[j];
      }
      if(len&1){
        printf("NO\n");
        return 0;
      }
    }
    printf("YES\n%d %d\n",two,p[two]);
    for(int i=1;i<=n;i++){
      if(i==two || p[i]==two) continue;
      if(col[i]) printf("%d %d\n",i,two);
      else printf("%d %d\n",i,p[two]);
    }
  }
  return 0;
}
