#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
int n, zh[100005] = {0}, ans[100005] = {0}, qaq, now, now1;
int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) scanf("%d", &zh[i]);
	for (int i = 1; i <= n; i++)
		if (zh[i] == i)
		{
			printf("YES\n");
			for (int j = 1; j <= n; j++)
				if (j != i)
					printf("%d %d\n", i, j);
			return 0;
		}
	for (int i = 1; i <= n; i++)
		if (zh[zh[i]] == i)
		{
			qaq = i;
			for (int j = 1; j <= n; j++)
				if (j != i && j != zh[i] && ans[j] == 0)
				{
					ans[j] = i;
					now = zh[j];
					now1 = zh[i];
					while (now != j)
					{
						ans[now] = now1;
						now = zh[now];
						now1 = zh[now1];
					}
					if (ans[now] != now1)
					{
						printf("NO\n");
						return 0;
					}
				}
			break;
		}
	if (qaq == 0)
	{
		printf("NO\n");
		return 0;
	}
	printf("YES\n%d %d\n", qaq, zh[qaq]);
	for (int i = 1; i <= n; i++)
		if (i != qaq && i != zh[qaq])
			printf("%d %d\n", ans[i], i);
	return 0;
}
