#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

vector<int> p[100005];
int n,num,t,a[100005];
bool ok[100005],flag;
int main() {
	scanf("%d",&n);
	for (int i=1;i<=n;i++) {
		scanf("%d",&a[i]);
		if (a[i]==i) {
			printf("YES\n");
			for (int j=1;j<=n;j++) if (i!=j) printf("%d %d\n",i,j); 
			return 0;
		}
	}
	for (int i=1;i<=n;i++) if (ok[i]==false) {
		ok[i]=true; num++;
		p[num].push_back(i);
		int j=i;
		while (1) {
			j=a[j];
			if (ok[j]) break;
			ok[j]=true;
			p[num].push_back(j);
		}
		if (p[num].size()==2) t=num,flag=true;
		if (p[num].size()%2==1) {
			printf("NO\n");
			return 0;
		}
	}
	if (!flag) {
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	printf("%d %d\n",p[t][0],p[t][1]);
	for (int i=1;i<=num;i++) if (i!=t) {
		for (int j=0;j<p[i].size();j++) {
			if (j%2==0) printf("%d %d\n",p[i][j],p[t][0]); else printf("%d %d\n",p[i][j],p[t][1]);
		}
	}
	return 0;
}
