#include<bits/stdc++.h>
using namespace std;
const int maxn=100010;
vector<int> t[maxn];
int a[maxn],n,sum;

bool vis[maxn];
int main(){
	scanf("%d",&n),sum=0; int k;
	for(int i=1;i<=n;++i)
		scanf("%d",a+i);
	memset(vis,false,sizeof(vis));
	for(int i=1;i<=n;++i)if(!vis[i]){
		t[++sum].clear();
		vis[i]=true,t[sum].push_back(i);
		for(int j=a[i];j!=i;j=a[j])
			vis[j]=true,t[sum].push_back(j);
	}	
	k=-1;
	for(int i=1;i<=sum;++i)
		if(t[i].size()==1){ k=i; break; }
	if(k!=-1){
		puts("YES");
		for(int i=1;i<=n;++i)
			if(i!=t[k][0]) printf("%d %d\n",i,t[k][0]);
		return 0;
	}
	
	for(int i=1;i<=sum;++i)
		if(t[i].size()&1){ puts("NO"); return 0; }
	
	k=-1;
	for(int i=1;i<=sum;++i)
		if(t[i].size()==2){ k=i; break; }
	if(k==-1){ puts("NO"); return 0; }
	puts("YES");
	printf("%d %d\n",t[k][0],t[k][1]);
	for(int i=1;i<=sum;++i)if(i!=k)
		for(int j=0;j<t[i].size();++j)
			if(j&1)
				printf("%d %d\n",t[k][0],t[i][j]);
			else printf("%d %d\n",t[k][1],t[i][j]);
	return 0;
}
