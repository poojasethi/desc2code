#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=1e5+10;
int color[mx],p[mx];

int main(){
   int n;
   scanf("%d",&n);
   int u=-1;
   for(int i=0;i<n;i++){
      scanf("%d",p+i);
      p[i]--;
      if(i==p[i])u=i;
   }
   if(u!=-1){
      puts("YES");
      for(int i=0;i<n;i++)if(i!=u)
         printf("%d %d\n",u+1,i+1);
      return 0;
   }   
   memset(color,-1,sizeof color);
   bool ok=true;
   int x=-1,y=-1;
   for(int i=0;i<n;i++)if(color[i]==-1){
      u=i;
      int c=0,len=0;;
      while(color[u]==-1){
         color[u]=c;
         u=p[u];
         c^=1;
         len++;
      }
      if(len==2) x=u,y=p[u];
      if(len&1) ok=false;
   }
   if(x==-1)ok=false;
   if(!ok){
      puts("NO");
      return 0;
   }
   puts("YES");
   printf("%d %d\n",x+1,y+1);
   for(int i=0;i<n;i++)if(i!=x && i!=y){
      if(color[i]==0) printf("%d %d\n",x+1,i+1);
      else printf("%d %d\n",y+1,i+1);
   }
   return 0;
}
