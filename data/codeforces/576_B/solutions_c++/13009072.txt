#include<bits/stdc++.h>
#define N 200005
using namespace std;
int ansx[N],ansy[N],p[N],f[N],L,R,c,n,i,flag,x,y,T;
void Link(int x,int y){
  ansx[++*ansx]=x;
  ansy[*ansx]=y;
}
void Print()
{
  puts("YES");
  for (int i=1;i<n;i++)
    printf("%d %d\n",ansx[i],ansy[i]);
  exit(0);
}
int main()
{
  scanf("%d",&n);
  for (i=1;i<=n;i++){
    scanf("%d",&p[i]);
    if (p[i]==i) flag=i;
  }
  for (i=1;i<=n;i++)
    if (i!=flag&&p[i]==i)
      Link(flag,i);
  if (flag){
    for (i=1;i<=n;i++)
        if (p[i]!=i)
            Link(flag,i);
    Print();
  }for (i=1;i<=n;i++)
    if (p[i]!=i&&p[p[i]]==i&&p[i]>i){
        f[i]=f[p[i]]=1;
        if (!L) Link(p[i],i);
        else Link(i,L),Link(p[i],R);
        L=i;R=p[i];
    }
  for (i=1;i<=n;i++)
    if (p[i]!=i&&!f[i]){
        if (!flag&&!L) return puts("NO"),0;
      for (x=i,c=0;!f[x]&&x;x=p[x],c++)
        Link(c&1?L:R,x),f[x]=1;
      if (!x||x!=i||(c&1)) return puts("NO"),0;
  }
  Print();
}
