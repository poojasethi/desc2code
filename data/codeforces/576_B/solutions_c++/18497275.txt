#include<cstdio>

bool vis[100001];
int a[100001],st[100001],top;
int main()
{
	bool flag=false;
	int n,i,pos=0,x,j,cnt;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		if(a[i]<i&&a[a[i]]==i)
			flag=true,x=i;
		else if(a[i]==i)
			pos=i;
	}
	if(pos)
	{
		puts("YES");
		for(i=1;i<pos;i++)
			printf("%d %d\n",pos,i);
		for(i=pos+1;i<=n;i++)
			printf("%d %d\n",pos,i);
	}
	else if(flag)
	{
		vis[pos=a[x]]=vis[x]=true;
		for(i=1;i<=n;i++)
		if(!vis[i])
		{
			cnt=1,vis[st[top++]=i]=true;
			for(j=a[i];j!=i;j=a[j])
				vis[st[top++]=j]=true,cnt++;
			if(cnt&1)return puts("NO"),0;
		}
		puts("YES");
		printf("%d %d\n",pos,x);
		for(i=0;i<top;i+=2)
			printf("%d %d\n%d %d\n",pos,st[i],x,st[i|1]);
	}
	else puts("NO");
	return 0;
}
