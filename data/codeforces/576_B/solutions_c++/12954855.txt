#include<cstdio>
#include<algorithm>
#define N 1001000
using namespace std;
int n,flag,i,a[N],vis[N],f[N],cnt,t;
int tmp1,tmp2;
int main()
{
    scanf("%d",&n);
    for (i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        if (a[i]==i) flag=i;
    }
    if (flag)
    {
        printf("YES\n");
        for (i=1;i<=n;i++)
        if (i!=flag)
        printf("%d %d\n",flag,i);
        return 0;
    }
    flag=0;
    for (i=1;i<=n;i++)
    if (vis[i]==0)
    {
        t=i;cnt=0;
        while (vis[t]==0)
        {
            vis[t]=1;
            f[t]=cnt%2;
            t=a[t];
            cnt++;
        }
        if (cnt%2)
        {
            printf("NO");
            return 0;
        }
        if (cnt==2)
        flag=i;
    }
    if (flag==0)
    {
        printf("NO");
        return 0;
    }
    printf("YES\n");
    for (i=1;i<=n;i++)
    if ((i!=flag)&&(i!=a[flag]))
    {
    if (f[i]==0)
    printf("%d %d\n",i,flag);
    else
    printf("%d %d\n",i,a[flag]);
    }
    printf("%d %d\n",flag,a[flag]);
}
