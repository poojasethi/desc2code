#include<bits/stdc++.h>
using namespace std;
int n,m,i,j,minv,w,cur;
int c,A[100100],B[100100],ST[100100], u[100100];
int main() {
    scanf("%d",&n);
    for (i=1; i<=n; i++) scanf("%d",&A[i]);
    for (i=1; i<=n; i++)
        if (!u[i]) {
            ST[c++]=m;
            for (j=i; !u[j]; j=A[j]) {
                u[j]=true;
                B[m++]=j;
            }
        }
    ST[c]=m; minv=ST[1]-ST[0];
    for (i=0; i<c; i++)
        if (ST[i+1]-ST[i]<minv) {
            minv=ST[i+1]-ST[i];
            w=i;
        }
    if (minv>2) {
        puts("NO"); return 0;
    }
    for (i=0; i<c; i++)
        if ((ST[i+1]-ST[i])%minv) {
            puts("NO"); return 0;
        }
    puts("YES");
    if (minv==2) printf("%d %d\n",B[ST[w]],B[ST[w]+1]);
    for (i=0; i<c; i++)
        if (i!=w) {
            cur=ST[i+1]-ST[i];
            for (j=0; j<cur; j++)
                printf("%d %d\n",B[ST[i]+j],B[ST[w]+j%minv]);
        }
    return 0;
}
