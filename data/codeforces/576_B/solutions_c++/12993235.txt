#include<cstdio>
using namespace std;
const int N=100005;
int p[N],t[N],n;
bool vis[N];
int main(){
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	scanf("%d",&n);
	for(int i=1;i<=n;i++){scanf("%d",p+i);t[i]=1;}
	for(int i=1;i<=n;i++)
		if(p[i]==i){
			puts("YES");//<<endl;
			for(int j=1;j<=n;j++)
				if(j!=i)printf("%d %d\n",i,j);//<<endl;
			return 0;
		}
	for(int i=1,j;i<=n;i++){
		if(!t[i])continue;
		for(j=p[i];j!=i;j=p[j]){
			t[i]++;t[j]--;
		}
	}
	int cp=0,flag=0;
	for(int i=1;i<=n;i++){
		if(t[i]==2) cp=i;
		if(t[i]&1) flag=1;
	}
	//for(int i=1;i<=n;i++)if(t[i])printf("%d\n",t[i]);
	if(flag||!cp){
		puts("NO");//<<endl;
		return 0;
	}
	puts("YES");//<<endl;
	printf("%d %d\n",cp,p[cp]);//<<endl;
	vis[cp]=vis[p[cp]]=1;
	for(int i=1,j;i<=n;i++){
		if(vis[i]) continue;
		printf("%d %d\n",i,cp);//<<endl;
		vis[i]=1; cp=p[cp];
		for(j=p[i];j!=i;j=p[j],cp=p[cp]){
			printf("%d %d\n",cp,j);//<<endl;
			vis[j]=1;
		}
	}
	return 0;
}
