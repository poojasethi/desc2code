#include <iostream>
#include <vector>
using namespace std;
const int MAX = 100005;
bool mark[MAX];
int p[MAX];
vector<pair<int, int> > ans;
int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> p[i];
		p[i]--;
		if (p[i] == i)
		{
			cout << "YES\n";
			for (int j = 0; j < n; j++)
				if (j != i)
					cout << i + 1 << " " << j + 1 << "\n";
			return 0;
		}
	}
	for (int i = 0; i < n; i++)
		if (p[p[i]] == i)
		{
			mark[i] = mark[p[i]] = true;
			ans.push_back(make_pair(i, p[i]));
			for (int j = 0; j < n; j++)
				if (!mark[j])
				{
					int x = i;
					int v = j;
					int len = 0;
					while (!mark[v])
					{
						mark[v] = true;
						ans.push_back(make_pair(x, v));
						x = p[x];
						v = p[v];
						len++;
					}
					if (len & 1)
					{
						cout << "NO\n";
						return 0;
					}
				}
			cout << "YES\n";
			for (int i = 0; i < ans.size(); i++)
				cout << ans[i].first + 1 << " " << ans[i].second + 1 << "\n";
			return 0;
		}
	cout << "NO\n";
	return 0;
}
