#include <bits/stdc++.h>
#define fi first
#define se second

using namespace std;

typedef pair <int,int> ii;

const int maxn=1e5+20;

int N,ar[maxn];
priority_queue <ii> heap;
bool es[maxn],used[maxn];
vector <ii> v;

int main(){
	scanf("%d",&N);
	
	for(int i=1;i<=N;i++){
		scanf("%d",&ar[i]);
		if(ar[i]==i){
			printf("YES\n");
			for(int j=1;j<=N;j++)
				if(j!=i)
					printf("%d %d\n",i,j);
			return 0;
		}
	}
	
	for(int i=1;i<=N;i++)
		if(!used[i]){
			int top=0;
			for(int j=i;!used[j];j=ar[j]){
				top++;
				used[j]=1;
			}
			if(top&1){
				printf("NO\n");
				return 0;
			}
			heap.push(ii(-top,i));
		}
	
	if(heap.top().fi!=-2){
		printf("NO\n");
		return 0;
	}
	
	int a=heap.top().se,b=ar[a];
	heap.pop();
	
	while(!heap.empty()){
		ii x=heap.top();
		heap.pop();
		int k=a;
		
		for(int j=x.se;!es[j];j=ar[j],k=ar[k])
			v.push_back(ii(k,j)) , es[j]=1;
	}
	
	printf("YES\n");
	
	printf("%d %d\n",a,b);
	for(int i=0;i<N-2;i++)
		printf("%d %d\n",v[i].fi,v[i].se);
	
	return 0;
}
