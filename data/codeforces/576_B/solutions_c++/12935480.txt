#include <cstdio>

int A[101000];
bool F[101000], G[101000];

int main () {
    int n = -1;
    scanf ("%d", &n);
    bool hasodd = false;
    int mk = n + 1, mi = -1;
    for (int i = 1; i <= n; i++) scanf ("%d", &A[i]);
    for (int i = 1; i <= n; i++)
	if (!F[i]) {
	    int k = 1;
	    F[i] = true;
	    for (int j = A[i]; j != i; j = A[j]) {
		F[j] = true;
		++k;
	    }
	    if (k < mk) {
		mk = k;
		mi = i;
	    }
	    if (k & 1) hasodd = true;
	}
    if (mk == 1) {
	puts ("YES");
	for (int i = 1; i <= n; i++)
	    if (i != mi) {
		printf ("%d %d\n", mi, i);
	    }
    } else if (mk == 2 && !hasodd) {
	puts("YES");
	G[mi] = G[A[mi]] = true;
	printf ("%d %d\n", mi, A[mi]);
	for (int i = 1; i <= n; i++) {
	    if (!G[i]) {
		printf ("%d %d\n", mi, i);
		int k = 1;
		G[i] = true;
		for (int j = A[i]; j != i; j = A[j]) {
		    G[j] = true;
		    ++k;
		    if (k & 1) printf ("%d %d\n", mi, j); else printf ("%d %d\n", A[mi], j);
		}
	    }
	}
    } else puts ("NO");
}
