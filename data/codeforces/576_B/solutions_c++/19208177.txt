#include<cstdio>
int a[100001],n,i,j,k,t;
bool f[100001],g[100001];
inline void find(int x){
	for(;!f[x];x=a[x],t=a[t])f[x]=1,printf("%d %d\n",t,x);
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i)scanf("%d",a+i);
	for(i=1;i<=n;++i){
		if(a[i]==i)break;
		if(a[a[i]]==i)t=i;
		if(g[i]||k&1)continue;
		for(j=i;!g[j];j=a[j])++k,g[j]=1;
	}
	if(i<=n||t&&(k&1)==0){
		printf("YES\n");
		if(i>n)printf("%d %d\n",t,a[t]);else t=i;
		f[t]=f[a[t]]=1;
		for(i=1;i<=n;++i)if(!f[i])find(i);
	}
	else printf("NO\n");
	return 0;
}
