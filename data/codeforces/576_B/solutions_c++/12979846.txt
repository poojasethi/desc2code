#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;
const int N=1E5+5;
int n,a[N],ans[N],k,u,v,p,sum;

int main()
{
    cin>>n;
    for (int i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        if (a[i]==i) k=i;
    }
    if (k)
    {
        cout<<"YES\n";
        for (int i=1;i<=n;i++) if (i!=k) printf("%d %d\n",k,i);
        return 0;
    }
    for (int i=1;i<=n;i++) if (a[a[i]]==i) k=i;
    u=k; v=a[u];
    a[u]=0; a[v]=0;
    for (int i=1;i<=n;i++) if (a[i])
    {
        sum=0;
        for (int j=i;a[j];)
        {
            sum++;
            ans[j]=u; p=a[j]; a[j]=0; j=p;
            swap(u,v);
        }
        if (sum%2) k=0; 
    }
    if (!k) { cout<<"NO"; return 0; }
    cout<<"YES\n";
    cout<<u<<" "<<v<<endl;
    for (int i=1;i<=n;i++) if (i!=u&&i!=v) printf("%d %d\n",i,ans[i]);
}
