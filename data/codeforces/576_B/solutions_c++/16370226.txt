#include <bits/stdc++.h>
#define f first
#define s second
#define mp make_pair
using namespace std;

const int N=100000+5;

int n,p[N],tail,q[N],S,T,cnt;
bool vis[N];
pair<int,int> e[N];

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;++i) scanf("%d",&p[i]);
	for (int i=1;i<=n;++i)
		if (p[i]==i){S=i;break;}
	if (S)
	{
		puts("YES");
		for (int i=1;i<=n;++i)
			if (i!=S) printf("%d %d\n",S,i);
		return 0;
	}
	for (int i=1;i<=n;++i)
		if (p[p[i]]==i){T=p[S=i];break;}
	if (!S || !T){puts("NO");return 0;}
	e[++cnt]=mp(S,T);
	vis[S]=vis[T]=1;
	for (int i=1;i<=n;++i)
		if (!vis[i])
		{
			tail=0;
			for (int j=i;!vis[j];j=p[j])
				vis[q[++tail]=j]=1;
			if (tail&1){puts("NO");return 0;}
			for (int j=1;j<=tail;++j)
				e[++cnt]=mp(q[j],j&1?S:T);
		}
	puts("YES");
	for (int i=1;i<=cnt;++i)
		printf("%d %d\n",e[i].f,e[i].s);
	return 0;
}
