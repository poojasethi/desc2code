#include <iostream>
#include <vector>
using namespace std;
int n,nxt[100005],mn=1000000000,loc;
bool vis[100005],odd,oe[100005];
vector<int> len;
int main() {
    cin>>n;
    for(int i=1;i<=n;i++)cin>>nxt[i];
    for(int i=1;i<=n;i++){
        if(vis[i])continue;
        vis[i]==true;
        int cnt=1,cur=nxt[i];
        oe[i]=true;
        while(cur!=i){
            vis[cur]=true;
            cnt++;
            oe[nxt[cur]]=!oe[cur];
            cur=nxt[cur];
        }
        len.push_back(cnt);
        if(mn>cnt){
            mn=cnt;
            loc=i;
        }
        if(cnt%2==1)odd=true;
    }
    if(mn==1){
        cout<<"YES\n";
        for(int i=1;i<=n;i++){
            if(i==loc)continue;
            cout<<i<<" "<<loc<<"\n";
        }
    }
    else if(mn==2&&odd==false){
        cout<<"YES\n";
        for(int i=1;i<=n;i++){
            if(i==loc)continue;
            if(!oe[i])cout<<i<<" "<<loc<<"\n";
            else cout<<i<<" "<<nxt[loc]<<"\n";
        }
    }
    else cout<<"NO";
}
