//In the Name of God

#include<iostream>
#include<cstdlib>
#define E exit(0)

using namespace std;

int n,p[100100],r,r1,r2;
bool vis[100100],c[100100];

void dfs(int v)
{
  vis[v]=true;
  if(!vis[p[v]])
    c[p[v]]=!c[v],dfs(p[v]);
  else if(c[v]==c[p[v]])
    cout<<"NO"<<endl,E;
}

int main()
{
  cin>>n;
  for(int i=1;i<=n;i++)
    {
      cin>>p[i];
      if(p[i]==i)
	r=i;
    }
  if(r!=0)
    {
      cout<<"YES"<<endl;
      for(int i=1;i<=n;i++)
	if(i!=r)
	  cout<<i<<" "<<r<<endl;
      cout<<endl;
      E;
    }
  for(int i=1;i<=n;i++)
    if(p[p[i]]==i)
      r1=i,r2=p[i];
  if(r1==0)
    {
      cout<<"NO"<<endl;
      E;
    }
  for(int i=1;i<=n;i++)
    if(!vis[i])
      dfs(i);
  cout<<"YES"<<endl;
  cout<<r1<<" "<<r2<<endl;
  for(int i=1;i<=n;i++)
    if(i!=r1 && i!=r2)
      if(c[i])
	cout<<i<<" "<<r1<<endl;
      else
	cout<<i<<" "<<r2<<endl;
  E;
}
