#include <stdio.h>
#include <string.h>
#include <iostream>
#include <algorithm>
using namespace std;
int n;
int p[111111];
bool mark[111111];
int main()
{
	scanf("%d",&n);
	int root=0;
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&p[i]);
		if(i==p[i])root=i;
	}
	if(root)
	{
		printf("YES\n");
		for(int i=1;i<=n;i++)
		{
			if(i==root)continue;
			printf("%d %d\n",i,root);
		}
	}
	else {
		memset(mark,false,sizeof(mark));
		bool key=false;
		for(int i=1;i<=n;i++)
		{
			if(!mark[i])
			{
				int c=0;
				for(int j=i;;j=p[j])
				{
					if(mark[j])break;
					c++;
					mark[j]=true;
				}
				if(c&1){
					root=0;
					break;
				}
				if(c==2)root=i;
				
			}
		}
		memset(mark,false,sizeof(mark));
		if(!root)printf("NO\n");
		else {
			
			printf("YES\n%d %d\n",root,p[root]);
			for(int i=1;i<=n;i++)
			{
				if(i!=root&&i!=p[root]&&!mark[i])
				{
					for(int j=i;;j=p[j])
					{
						if(mark[j])break;
						mark[j]=true;
						printf("%d %d\n",root,j);
						root=p[root];
					}
				}
			}
		}
	}
}
