#include<cstdio>
#include<vector>
using namespace std;

#define ran 111111

int n, x, y, z;
int a[ran];
bool visd[ran];
vector<int> c[ran];
int top;

int main() {
	scanf("%d", &n);
	for(int i=1;i<=n;i++) scanf("%d", &a[i]);
	x=y=-1;
	top=0;
	for(int i=1;i<=n;i++)
		if(!visd[i]) {
			c[top].push_back(i);
			visd[i] = true;
			for(int p=a[i]; p!=i;p=a[p]) {
				c[top].push_back(p);
				visd[p] = true;
			}
			if(c[top].size()==1) x=i;
			else if(c[top].size()==2) y=i;
			top++;
		}
	if(x>0) {
		printf("YES\n");
		for(int i=1;i<=n;i++)
			if(i!=x) printf("%d %d\n", x, i);
	} else if(y>0) {
		bool chk = true;
		for(int i=0;i<top && chk;i++)
			if (c[i].size()%2) chk = false;
		if(!chk) printf("NO\n");
		else {
			printf("YES\n");
			z=a[y];
			printf("%d %d\n", y, z);
			for(int i=0;i<top;i++) {
				if(c[i][0]==y) continue;
				for(int j=0;j<c[i].size();j+=2)
					printf("%d %d\n%d %d\n", y, c[i][j], z, c[i][j+1]);
			}
		}
	} else printf("NO\n");
	
	return 0;
}
