#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
int n,p[100005];
bool vis[100005];
vector <pair <int,int> > v;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&p[i]);
	}
	for(int i=1;i<=n;i++)
	{
		if(i==p[i])
		{
			puts("YES");
			for(int j=1;j<=n;j++)
			{
				if(j!=i)
				{
					printf("%d %d\n",i,j);
				}
			}
			return 0;
		}
	}
	for(int i=1;i<=n;i++)
	{
		if(p[p[i]]!=i)continue;
		vis[i]=true;
		vis[p[i]]=true;
		//cout<<i<<endl;
		v.push_back(make_pair(i,p[i]));
		for(int j=1;j<=n;j++)
		{
			if(vis[j])continue;
			int now=j;
			int cnt=0;
			//cout<<now<<endl;
			while(!vis[now])
			{
				vis[now]=true;
				cnt++;
				if(cnt&1)v.push_back(make_pair(now,i));
				else v.push_back(make_pair(now,p[i]));
				now=p[now];
			}
			if(cnt&1)
			{
				puts("NO");
				return 0;
			}
		}
		puts("YES");
		for(int j=0;j<n-1;j++)
		{
			printf("%d %d\n",v[j].first,v[j].second);
		}
		return 0;
	}
	puts("NO");
	return 0;
}
