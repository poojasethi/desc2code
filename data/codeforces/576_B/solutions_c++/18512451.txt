#include<cstdio>
#include<cstring>
using namespace std;
int _2[2],_1,n,a[100001];
bool ji,vis[100001];
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(int i=1;i<=n;i++)
		if(!vis[i]){
			int cnt=0;
			while(!vis[i]){
				vis[i]=true;
				cnt++;
				i=a[i];
			}
			if(cnt==1)
				_1=i;
			else if(cnt==2){
				_2[0]=i;
				_2[1]=a[i];
			}
			else if(cnt%2)
				ji=true;
		}
	if(_1){
		printf("YES\n");
		for(int i=1;i<_1;i++)
			printf("%d %d\n",a[i],a[_1]);
		for(int i=_1+1;i<=n;i++)
			printf("%d %d\n",a[i],a[_1]);
	}
	else if(!ji&&_2[0]){
		printf("YES\n");
		printf("%d %d\n",a[_2[0]],a[_2[1]]);
		memset(vis,0,sizeof(vis));
		for(int i=1;i<=n;i++)
			if(i!=_2[0]&&i!=_2[1]){
				int j=0;
				while(!vis[i]){
					vis[i]=true;
					printf("%d %d\n",a[i],_2[j]);
					j=1-j;
					i=a[i];
				}
			}
	}
	else
		printf("NO");
	return 0;
}
