#include <cstdio>
#include <cstring>
#include <algorithm>
#define maxn 201000
using namespace std;
int mark[maxn],p[maxn];
int flag1,flag2,flag3;
int n;

int main() {
	scanf("%d",&n);
	for (int i = 1; i <= n; i++) scanf("%d",p + i);
	flag1 = flag2 = flag3 = 0;
	for (int i = 1; i <= n; i++) {
		int js = 1;
		if (mark[i]) continue; mark[i] = 1;
		for (int j = p[i]; j != i; j = p[j]) mark[j] = 1, js++;
		if (js > 2 && (js & 1)) flag3 = 1;
		if (js == 1) flag1 = i;
		if (js == 2) flag2 = i;
	}
	if (flag1) {
		printf("YES\n");
		for (int i = 1; i <= n; i++)
			if (i != flag1) printf("%d %d\n",flag1,i);
	} else if (flag2 && !flag3) {
		memset(mark,0,sizeof(mark));
		printf("YES\n");
		int a = flag2, b = p[flag2];
		printf("%d %d\n",a,b);
		for (int i = 1; i <= n; i++) {
			if (mark[i] || i == a || i == b) continue;
			int js = 0;
			for (int j = i; (j != i) || (!js); j = p[j]) {
				mark[j] = 1;
				if ((++js) & 1) printf("%d %d\n",j,a);
				else printf("%d %d\n",j,b);
			}
		}
	} else {
		printf("NO\n");
	}	
	return 0;
}
