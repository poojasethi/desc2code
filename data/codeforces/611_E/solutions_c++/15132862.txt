#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<set>
using namespace std;

int n,x,y,z,a[200005],ans;
multiset<int> s;
int main() {
	scanf("%d%d%d%d",&n,&x,&y,&z);
	if (x>y) swap(x,y);
	if (y>z) swap(y,z);
	if (x>z) swap(x,z);
	if (x>y) swap(x,y);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]),s.insert(a[i]);
	while (!s.empty()) {
		set<int>::iterator now=--s.end(),t,t1,t2;
		int num=*now;
		if (num>x+y+z) {
			printf("-1\n"); return 0;
		}
		ans++;
		if (num>y+z) {
			s.erase(now);
			continue;
		}
		if (num>x+z) {
			s.erase(now);
			t=s.upper_bound(x);
			if (t!=s.begin()) {
				t--; s.erase(t);
			}
			continue;
		}
		if (num>x+y&&num>z) {
			s.erase(now);
			t=s.upper_bound(y);
			if (t!=s.begin()) {
				t--; s.erase(t);
			}
			continue;
		}
		if (num>z) {
			s.erase(now);
			t=s.upper_bound(z);
			if (t!=s.begin()) {
				t--; s.erase(t);
			}
			continue;
		}
		if (num>y) {
			s.erase(now);
			t1=s.upper_bound(x); t2=s.upper_bound(y);
			if ((t1!=s.begin())&&(t2!=s.begin())) {
				t1--; t2--;
				if ((t1==t2)&&(t1==s.begin())) {
					t=s.upper_bound(x+y);
					t--; s.erase(t);
					continue;
				} else {
					if (t1==t2) t1--;
					s.erase(t1); s.erase(t2);
					continue;
				}
			} else {
				t=s.upper_bound(x+y);
				if (t!=s.begin()) {
					t--; s.erase(t);
					continue;
				}
				continue;
			}
		}
		s.erase(now);
		t=s.upper_bound(y);
		if (t!=s.begin()) {
			t--; s.erase(t);
			t=s.upper_bound(x);
			if (t!=s.begin()) {
				t--; s.erase(t);
			}
		}
	}
	printf("%d\n",ans);
	return 0;
}
