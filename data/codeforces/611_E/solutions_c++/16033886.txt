#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long

using namespace std;

int n,a,b,c,t[200005],ans;
bool use[200005];

int j[7],v[7];

bool m(int i){
  while(j[i] && (use[j[i]] || t[j[i]]>v[i])) j[i]--;
  if(j[i]) return use[j[i]]=true;
  return false;
}
void u(int &p,int &q){if(p>q) swap(p,q);}
int main(){
  scanf("%d",&n);
  scanf("%d %d %d",&a,&b,&c);
  u(a,b); u(b,c); u(a,b);
  
  FI(i,1,n){
    scanf("%d",&t[i]);
    if(t[i]>a+b+c){
      printf("-1\n");
      return 0;
    }
  }
  sort(t+1,t+n+1);
  
  //a,b,c,ab,ac,bc,abc
  FI(i,0,6) j[i]=n;
  v[0]=a;
  v[1]=b;
  v[2]=c;
  v[3]=a+b;
  v[4]=a+c;
  v[5]=b+c;
  v[6]=a+b+c;
  
  FD(i,n,1){
    if(use[i]) continue;
    use[i]=true;
    if(t[i]>b+c);
    else if(t[i]>a+c) m(0);
    else if(t[i]>a+b && t[i]>c) m(1);
    else if(a+b>=c){
      if(t[i]>c) m(2);
      else if(t[i]>b) m(3*(1-m(1)));
      else if(t[i]>a) m(0),m(2);
      else m(1),m(2);
    }else{
      if(t[i]>a+b) m(3*(1-m(1)));
      else if(t[i]>b) m(2*(1-m(1)));
      else if(t[i]>a) m(0),m(2);
      else m(1),m(2);
    }
    ans++;
  }
  printf("%d\n",ans);
  return 0;
}
