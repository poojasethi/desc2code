#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
#include <set>
using namespace std;
#define ma (-(*ss.begin()))
multiset<int>ss;
multiset<int>::iterator it1;
int find(int v)
{
	it1=ss.lower_bound(-v);
	if(it1!=ss.end())
	{
		ss.erase(it1);
		return 1;
	}
	return 0;
}
int main()
{
	int n;int a,b,c,tmp;
	while(~scanf("%d",&n))
	{
		ss.clear();
		scanf("%d%d%d",&a,&b,&c);
		if(a>c)swap(a,c);if(b>c)swap(b,c);if(a>b)swap(a,b);
		for(int i=0;i<n;i++)
		{
		  scanf("%d",&tmp);
		  ss.insert(-tmp);
	    }
		if(ma>a+b+c)
		{
			printf("-1\n");
			continue;
		}
		    int cnt=0;
			while(ss.size()>0)
			{
				while(ma>b+c)find(a+b+c),cnt++;
				while(ma>a+c)find(b+c),find(a),cnt++;
				while(ma>a+b&&ma>c)find(a+c),find(b),cnt++;
				if(a+b>c)
				{
					while(ma>c)find(a+b),find(c),cnt++;
				}
				while(find(c))
				{
					cnt++;
					if(find(b))
					{
						find(a);
					}
					else 
					{
						find(a+b);
					}
				}
			}
			printf("%d\n",cnt);
	}
    
	return 0;
}
