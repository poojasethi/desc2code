int const N=111111;

int i,j,n,m,k;

multiset <int> s;
multiset <int> ::iterator it;

int fi()
{
  it=s.end();
  return *(--it);

}

void del(int x)
{
 if (s.size()==0) return;
 it=s.upper_bound(x);
 if (it==s.begin()) return;
 s.erase(--it);
 k++;
}

int main()
{
vector <int> v(3);
cin>>n>>v[0]>>v[1]>>v[2];
sort(v.begin(),v.end());
int ans=0;
for (int i=0;i<n;i++)
{
 int x;
 cin>>x;
if (x>v[0]+v[1]+v[2]) {cout<<-1;return 0;}
if (x>v[1]+v[2]) ans++; else
      s.insert(x);
}

while (s.size()>0)
{
  int t=fi();
  if (v[2]>=t) {
   s.erase(s.find(t));
  k=0;
  del(v[1]);
  del(v[0]);
 if (k==0) del(v[1]+v[0]);
  } else if (v[0]+v[1]>=t) {
   s.erase(s.find(t));
  del(v[2]);
 } else if (v[0]+v[2]>=t) {
   s.erase(s.find(t));
  del(v[1]);
  } else if (v[2]+v[1]>=t) {
   s.erase(s.find(t));
   del(v[0]);
  }


ans++;
//cout<<t<<' ';;
}
cout<<ans;

}
