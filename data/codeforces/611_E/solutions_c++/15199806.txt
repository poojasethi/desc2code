#include <bits/stdc++.h>
using namespace std;
#define maxn 200005

int a[maxn];
bool bt[maxn];
int n,p[5],t[5],ans,rem;

void update(int i) {
	for(;t[i]>=0;t[i]--)
		if(!bt[t[i]])
			break;
}

bool ok(int i) {
	return t[i]>=0;
}

void kill (int i){
	update(i);
	if(!ok(i))	return;
	bt[t[i]]=1;
	rem--;
}

int main () {
	int i;
	scanf("%d",&n);
	
	for(i=0;i<3;i++)
		scanf("%d",p+i);
	sort(p,p+3);
	p[3]=p[0]+p[1];
	p[4]=p[2]+p[3];

	for(i=0;i<n;i++)
		scanf("%d",a+i);
	sort(a,a+n);
	if(a[n-1]>p[4]) {
		puts("-1");
		return 0;
	}
	ans=0,rem=n;

	for(i=0;i<5;i++)
		for(t[i]=n-1;t[i]>=0;t[i]--)
			if(a[t[i]]<=p[i])
				break;
	
	while(rem){
		ans++;
		for(i=0;i<5;i++)
			update(i);
		kill(4);
		if(t[4]==t[2]) {
			update(1);
			if(!ok(1))
				kill(3);
			else {
				kill(1);kill(0);
			}
		}
		else if(a[t[4]]<=p[0]+p[1]) 
			kill(2);
		else if(a[t[4]]<=p[0]+p[2]) 
			kill(1);
		else if(a[t[4]]<=p[1]+p[2]) 
			kill(0);
	}
	cout<<ans;
}
