#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;
int q[200200],a,b,c,num[5];
int main(){
	int n;
	while(~scanf("%d",&n)){
		scanf("%d%d%d",&a,&b,&c);
		q[0]=a;q[1]=b;q[2]=c;
		sort(q,q+3);
		a=q[0];b=q[1];c=q[2];
		int flag=0;
		for(int i=0;i<n;i++){
			scanf("%d",&q[i]);
			if(q[i]>a+b+c) flag=1;
		}
		if(flag) puts("-1");
			else{
			sort(q,q+n);
				int l=1,r=n,ans=n;
				while(l<=r){
				//cout << l << " " << r << endl;
					int mid=(l+r)/2,tmp=1;
					num[0]=num[1]=num[2]=mid;
					int x=0;
					for(int i=n-1;i>=0;i--){
						if(q[i]<=a&&num[0]) {
							num[0]--;
							continue;
						}
						if(q[i]<=b&&num[1]){
							num[1]--;
							continue;
						}
						if(q[i]<=c&&num[2]){
							num[2]--;
							continue;
						}
						if(q[i]<=a+b&&num[1]&&num[0]){
							num[1]--;
							num[0]--;
							x++;
							continue;
						}
						if(q[i]<=a+c&&num[0]&&num[2]){
							num[2]--;
							num[0]--;
							x++;
							continue;
						}
						if(q[i]<=b+c&&num[2]&&num[1]){
							num[2]--;
							num[1]--;
							x++;
							continue;
						}
						if(num[1]&&num[2]&&num[0]){
							num[1]--;
							num[0]--;
							num[2]--;
							x++;
							continue;
						}
						tmp=0;
					}
					if(x>mid) tmp=0;
					if(tmp) ans=mid,r=mid-1;
						else l=mid+1;
				}
				cout << ans << endl;
			}
	}
}
