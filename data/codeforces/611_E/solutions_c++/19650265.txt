#include<cstdio>
#include<algorithm>
using namespace std;
#define N 2000010
int s[N],v[N],a[N],n,l,r,m,i,tmp;
int check(int x)
{
	int i,j,k,tmp;
	for (i=1;i<=3;i++)
	if (a[i]>=x)
		for (j=1;j<=7;j++)
		if ((s[j])&&((j&(1<<(i-1)))==(1<<(i-1))))
		{
			s[j]--;
			s[j-(1<<(i-1))]++;
			return 1;	
		}
	
	for (i=1;i<=3;i++)
	for (j=i+1;j<=3;j++)
	if (a[i]+a[j]>=x)
	for (k=1;k<=7;k++)
	if (s[k])
	{
		tmp=(1<<(i-1))+(1<<(j-1));
		if ((k&tmp)==tmp)
		{
			s[k]--;
			s[k-tmp]++;
			return 1;
		}
	}
	
	if (a[1]+a[2]+a[3]>=x)
	if (s[7])
	{
		s[7]--;
		return 1;
	}
	return 0;
}
int main()
{
	scanf("%d",&n);
	for (i=1;i<=3;i++)
	scanf("%d",&a[i]);
	sort(a+1,a+4);
	for (i=1;i<=n;i++)
	scanf("%d",&v[i]);
	sort(v+1,v+1+n);
	l=1;r=n;
	while (l<=r)
	{
		m=(l+r)>>1;
		s[7]=m;
		for (i=0;i<=6;i++)
		s[i]=0;
		for (i=n;i>=1;i--)
		if (check(v[i])==0) break;
		if (i==0) r=m-1;else l=m+1;
	}
	if (l==n+1) l=-1;
	printf("%d\n",l);

}
