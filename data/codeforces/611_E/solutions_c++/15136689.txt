#include <iostream>
#include <algorithm>
#include <cstdlib>

using namespace std;

int a,b,c,ab,ac,bc,abc;
int ia,ib,ic,iab,iac,ibc,iabc;

void escribe()
{
  cout<<ia<<" "<<ib<<" "<<ic<<" "<<iab<<" "<<iac<<" "<<iabc<<" "<<iabc<<endl;
}

int n;
int v[1000000];

void avanza(int &i,int val)
{
  while (i<n and (v[i]==0 or val<v[i]))
    i++;
}

int main()
{
  cin>>n>>a>>b>>c;
  for (int i=0;i<n;i++)
    cin>>v[i];
  if (a>b) swap(a,b);
  if (b>c) swap(b,c);
  if (a>b) swap(a,b);
  ab=a+b;
  ac=a+c;
  bc=b+c;
  abc=a+b+c;
  sort(v,v+n);
  reverse(v,v+n);
  if (v[0]>abc) {
    cout<<-1<<endl;
    exit(0);
  }
  int sol=0;
  for (;;) {
    avanza(iabc,abc);
    avanza(ibc,bc);
    avanza(iac,ac);
    avanza(iab,ab);
    avanza(ic,c);
    avanza(ib,b);
    avanza(ia,a);
    //escribe();
    if (iabc==n) break;
    sol++;
    if (iabc<ibc) {
      v[iabc]=0;
    } else if (ibc<iac) {
      v[ibc]=0;
      avanza(ia,a);
      if (ia<n)
	v[ia]=0;
    } else if (iac<iab and iac<ic) {
      v[iac]=0;
      avanza(ib,b);
      if (ib<n)
	v[ib]=0;
    } else if (iab<ic) {
      v[iab]=0;
      avanza(ic,c);
      if (ic<n)
	v[ic]=0;
    } else {
      //cout<<"caso c"<<endl;
      v[ic]=0;
      avanza(iab,ab);
      avanza(ia,a);
      avanza(ib,b);
      if (ib<n and ib==ia) {
	int ia2=ia+1;
	avanza(ia2,a);
	if (ia2<n)
	  ia=ia2;
      }
      //cout<<ic<<" "<<ia<<" "<<ib<<" "<<iab<<endl;
      if (ib<ia and ia<n) {
	v[ib]=0;
	v[ia]=0;
      } else if (iab<n) {
	v[iab]=0;
      }
    }
  }
  cout<<sol<<endl;
}
