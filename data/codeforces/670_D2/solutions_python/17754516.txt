def ok(u, a, b, n, k):
    for i in xrange(n):
        if (u * a[i] - b[i] > 0):
            k -= u * a[i] - b[i]
            if k < 0: return False
    return k >= 0
        
I = lambda: map(int, raw_input().split())
n, k = I()
a = I()
b = I()
l, r = 0, 2000000010
while l < r-1:
    g = l + (r-l+1) / 2
    if ok(g, a, b, n, k): l = g
    else: r = g

print l
