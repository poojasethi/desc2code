n, k = map(int,raw_input().split())
alist = map(int,raw_input().split())
blist = map(int,raw_input().split())

def check(n, m, alist, blist, k):
    for i in range(n):
        if blist[i] < alist[i]*m:
            k -= alist[i]*m - blist[i]
            if k < 0:
                return False
            
    return True

l, r = 0, 2000000001
while l<r:
    m = l + (r-l+1)/2
    if check(n, m, alist, blist, k):
        l = m
    else:
        r = m-1
    
print l
