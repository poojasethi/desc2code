n,k=map(int,raw_input().split())
a=map(int,raw_input().split())
b=map(int,raw_input().split())
lo,hi=0,1e10
ans=0
while lo<=hi:
	mid=int((lo+hi)/2)
	powder=k
	for i in range(0,n):
		if a[i]*mid>b[i]:
			powder=powder-(a[i]*mid-b[i])
		if powder<0:
			break
	if powder>=0:
		ans=mid
		lo=mid+1
	else:
		hi=mid-1
print ans
