rl = lambda : map(int, raw_input().split())
[n, k], a, b = rl(), rl(), rl()
l, r = 0, int(2e9)

def f(n_cakes):
	magic = k
	for i in xrange(n):
		qneeded = n_cakes * a[i]
		if qneeded > b[i]:
			magic -= (qneeded - b[i])
		if magic < 0: return False
	return True

while r - l > 1:
	mid = (l + r + 1) / 2
	if f(mid): l, r = mid, r
	else: l, r = l, mid
	
if r - l == 1:
	if f(r):
		print(r)
	else:
		print(l)
else: print(l)
