def test(m, n, k, a, b):
	for i in xrange(n):
		if b[i] >= a[i] * m:
			continue
		else:
			k -= a[i] * m - b[i]
			if k < 0:
				return False
	return k >= 0

I=lambda:map(int, raw_input().split())
n,k = I()
a = I()
b = I()
l, r = 0,2000000010
while l < r:
	m = l + (r-l+1) / 2
	if test(m, n, k, a, b):
		l = m
	else:
		r = m-1
print l
