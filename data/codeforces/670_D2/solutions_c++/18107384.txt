#include<iostream>
using namespace std;
long long int n,k,a[100000],b[100000],c,l=0,r=2000000000,m;
bool check (int x){
     for (int i=0;i<n;i++){
         if (a[i]*x>b[i])
         m-=a[i]*x-b[i];
          if (m<0) return 0;
}
     return true;
}
main()
{
    cin>>n>>k;
    for (int i=0;i<n;i++)
        cin>>a[i];
    for (int i=0;i<n;i++)
        cin>>b[i];
    while (l<r){
          c=(l+r+1)/2;
          m=k;
          if (check (c))
             l=c;
          else
             r=c-1;
    }
    cout<<l<<endl;
}
