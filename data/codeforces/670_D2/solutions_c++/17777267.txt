#include<iostream>
#include<cstdlib>
#include<algorithm>
using namespace std;

long long a[100005],b[100005];
int main()
{
	long long n,k,kk,mark,i,j,ans=0,l=1,r=0,mid;
	cin>>n>>k;
	for (i=1; i<=n; i++) cin>>b[i];
	for (i=1; i<=n; i++) cin>>a[i];
	for (i=1; i<=n; i++) 
		if (a[i]/b[i]>r) r=a[i]/b[i];
	r+=k; 
	while (l<=r){
		mid=(l+r)>>1; mark=1; kk=k;
		for (i=1; i<=n; i++){
			if (a[i]<mid*b[i]){
				kk-=mid*b[i]-a[i];
				if (kk<0) {mark=0; break;}
			}
		}
		if (mark){ans=mid; l=mid+1;}
		else r=mid-1;
	}
	cout<<ans<<endl;

	//system("pause");
}
