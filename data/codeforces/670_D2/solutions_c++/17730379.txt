#include <iostream>

using namespace std;

typedef long long int ll;

const int limite=1000000;

int n;

ll a[limite],b[limite];

bool puede(ll med,ll k)
{
  for (int i=0;i<n;i++) {
    ll bb=med*a[i];
    if (bb>b[i]) {
      if (bb>b[i]+k) return false;
      k-=bb-b[i];
    }
  }
  return true;
}

int main()
{
  ios::sync_with_stdio(false);
  ll k;
  cin>>n>>k;
  for (int i=0;i<n;i++)
    cin>>a[i];
  for (int i=0;i<n;i++)
    cin>>b[i];
  ll inf=0,sup=2000000001;
  while (inf+1<sup) {
    ll med=(inf+sup)/2;
    if (puede(med,k)) inf=med;
    else sup=med;
  }
  cout<<inf<<endl;
}
