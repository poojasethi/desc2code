#include<stdio.h>
int main()
{
	__int64 n,k,i,j,a[100005],b[100005];
	scanf("%I64d%I64d",&n,&k);
	for(i=0;i<n;i++)
 		scanf("%I64d",&a[i]);
	for(i=0;i<n;i++)
		scanf("%I64d",&b[i]);
	__int64 big,small;
	big=2000000000+8,small=0;
	while(small<big){
		__int64 mid=(big+small+1)/2;
		__int64 k1=k,k2;
		for(i=0;i<n;i++){
			k2=a[i]*mid-b[i];
			if(k2>0)k1-=k2;
			if(k1<0)break;
		}
		if(k1>=0)small=mid;
		else big=mid-1;
	}
	printf("%I64d\n",small);
	return 0;
}
