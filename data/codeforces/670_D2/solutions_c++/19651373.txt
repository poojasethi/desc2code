#include <iostream>
#include<algorithm>
using namespace std;
#define N 100010
long long a[N],b[N],n,k;
int main ()
{
	while(cin>>n>>k)
	{
		long long i;
		for(i=0;i<n;i++)
			cin>>a[i];
		for(i=0;i<n;i++)
            cin>>b[i];
		long long low=0,up=2000000001,mid;
			while(low<up)
			{
                mid=(low+up)/2+1;
				long long s=k;
				for(long long i=0;i<n;i++){
					if(mid>b[i]/a[i])
						s-=mid*a[i]-b[i];
                    if(s<0) break;
				}
				if(s<0) up=mid-1;
				else low=mid;
			}
			cout<<up<<endl;
	}
	return 0;
}
