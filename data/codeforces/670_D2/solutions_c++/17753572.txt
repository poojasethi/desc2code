#include<stdio.h>
int a[100005],b[100005];
int n;
int solve(int m,int k)
{
int i;
for(i=0;i<n;i++)
{
if(b[i]/a[i] >=m);
else {
if((b[i]+k)/a[i] <m)return 0;
k-= a[i]*m-b[i];
if(k<0)return 0;
}
}
return 1;
}
int main()
{
int k;
scanf("%d%d",&n,&k);
int i;
for(i=0;i<n;i++)scanf("%d",&a[i]);
for(i=0;i<n;i++)scanf("%d",&b[i]);
int lo,hi,mid;
lo = 0;
hi = 2000000009;
while(hi-lo>1)
{
mid = lo + (hi-lo)/2;
if(solve(mid,k))lo = mid;
else hi = mid;
}
int ans = lo;
if(solve(hi,k))ans = hi;
printf("%d\n",ans);
}
