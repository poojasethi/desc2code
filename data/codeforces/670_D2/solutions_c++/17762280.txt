#include <iostream>
#include <cstdio>
int n,m,l,r,mid,ans=0x7fffffff,a[100005],b[100005];
long long tot;
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	for(int i=1;i<=n;i++)
		scanf("%d",&b[i]),ans=std::min(ans,b[i]/a[i]);
	for(int i=1;i<=n;i++) b[i]-=a[i]*ans;
	l=0;r=m;
	while(l<r)
	{
		mid=(l+r+1)>>1;tot=0;int i;
		for(i=1;i<=n&&tot<=m;i++)
			if((long long)a[i]*mid>b[i])
				tot+=(long long)a[i]*mid-b[i];
		if(i==n+1&&tot<=m) l=mid; else r=mid-1;
	}
	printf("%d",ans+l);
}
