#include <bits/stdc++.h>
using namespace std;
#define ll long long
const int maxn = 100010;
int n;
ll k;
ll a[maxn],b[maxn];
ll solve(ll x){
    ll cur=k;
    for(int i=0; i<n; i++){
        ll temp = b[i]-a[i]*x;
        if(temp<0) cur+=temp;
        if(cur<0) return -1;
    }
    return 1;
}
int main(){
    cin>>n>>k;
    for(int i=0; i<n; i++) cin>>a[i];
    for(int i=0; i<n; i++) cin>>b[i];
    ll l=0,r=3e9;
    while(l<=r){
        ll mid = (l+r)>>1;
        if(solve(mid)<0) r=mid-1;
        else l=mid+1;
    }
    cout<<l-1<<endl;
    return 0;
}
