#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    ll n,k;
    cin>>n>>k;
    vector<int> a(n),b(n);
    for(ll i=0;i<n;i++)    cin>>a[i];
    for(ll i=0;i<n;i++)    cin>>b[i];

    ll l=0,h=(ll)2e9;
    while(h-l>0){
        ll m = (l+h+1)/2;
        ll cnt=0;
        for(ll i=0;i<n;i++){
            cnt += max(0LL,(ll)a[i]*m-b[i]);
            if(cnt>k)   break;
        }
        if(cnt<=k)  l=m;
        else        h=m-1;
    }
    cout<<l<<endl;
    return 0;
}
