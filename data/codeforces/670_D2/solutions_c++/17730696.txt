#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <vector>
#include <queue>
using namespace std;
typedef long long ll;
const int N=1E5+5;
int a[N],b[N],k,n;
ll x,sum,l,r,mid;

int main()
{
	cin>>n>>k;
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	for (int i=1;i<=n;i++) scanf("%d",&b[i]);
	l=0; r=2E9+1;
	while (l<r)
	{
		mid=(l+r)/2;
		sum=0;
		for (int i=1;i<=n&&sum<=k;i++)
		{
			x=max((ll)0,(ll)mid*a[i]-b[i]);
			sum+=x;
		}
		if (sum<=k) l=mid+1; else r=mid;
	}
	cout<<l-1;
}
