#include <cstdio>
int n,k,a[100010],b[100010];
bool check(long long m){
	long long s=0,t;
	for(int i=0;i<n;i++){
		t=m*a[i]-b[i];
		if(t>0){
			s+=t;
			if(s>k)return 0;
		}
	}
	return 1;
}
int main(){
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++)scanf("%d",a+i);
	for(int i=0;i<n;i++)scanf("%d",b+i);
	int left=0,right=2000000001,mid;
	while(right-left>1){
		mid=left+(right-left)/2;
		if(check(mid))left=mid;
		else right=mid;
	}
	printf("%d",left);
}
