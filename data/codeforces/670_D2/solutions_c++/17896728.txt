#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib> 
#include<algorithm>
using namespace std;

typedef long long ll;

ll n,k,minv=2e9+1;
ll a[100001],b[100001];

int main(){
	scanf("%ld%ld",&n,&k);
	for(int i=0;i<n;i++)
		scanf("%ld",&a[i]);
	for(int i=0;i<n;i++)
		scanf("%ld",&b[i]),
		minv=min(minv,(k+b[i])/a[i]);
	
	ll L=0,R=minv,mid,ans;
	while(L<=R){
		mid=(L+R)/2;
		
		ll sum=0;
		for(int i=0;i<n;i++)
			sum+=max((ll)0,mid*a[i]-b[i]);
		
		if(sum<=k) L=mid+1,ans=mid;
		else R=mid-1;
	}
	
	printf("%ld\n",ans);
	return 0;
}
