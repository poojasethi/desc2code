#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	long long int n,k,i,a[100005],b[100005],l,h,m,s=0,y,ans=0;
	cin>>n>>k;
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n;i++)
	cin>>b[i];
	l=0,h=2e9;
	while(l<=h)
	{
		m=(l+h)>>1;
	
		s=0;
		for(i=0;i<n;i++)
		{
			s+=max(0LL,(m*a[i])-b[i]);
		//	cout<<s<<" ";
		if(s>k)
		break;
		}
		
		if(s<=k)
		{
			ans=m;
			l=m+1;
		}
		else
		h=m-1;
	}
	cout<<ans<<endl;
	return 0;
}
