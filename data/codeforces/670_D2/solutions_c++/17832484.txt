#include<iostream>
using namespace std;
#include<vector>
#define MAX 1000
typedef long long ll;
ll amax(ll x,ll y) { return x>y?x:y;}
int main(){
	int n,k;
	while(cin>>n>>k){
		vector<int>a(n);
		vector<int>b(n);
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		for(int i=0;i<n;i++){
			cin>>b[i];
		}
		
		ll l=0,h=(ll)2e9;
		while(h-l>0){
			ll x=0;
			ll mid=(l+h+1)/2;
			for(int i=0;i<n;i++){
				x+=amax(0ll,((ll)a[i]*mid-b[i]));
				if(x>k)break;
			}
			if(x>k)
				h=mid-1;
			else l=mid;
		}
		cout<<l<<endl;
	}

	return 0;
}
