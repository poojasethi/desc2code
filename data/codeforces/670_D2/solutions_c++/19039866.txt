#include <iostream>
#include <cstdio>
using namespace std;
#define LL long long
LL a[100005],b[100005],l,r,mid,k;
int n;
bool fuc(LL p)
{
	LL ans=0;
	for(int i=0;i<n;i++)
	{
		if(a[i]*p>b[i])
		{
			ans+=a[i]*p-b[i];
			if(ans>k) return 0; 
		}
	}
	if(ans<=k) return 1;
	else return 0;
}
int main()
{
	scanf("%d%lld",&n,&k);
	for(int i=0;i<n;i++) scanf("%lld",&a[i]);
	for(int i=0;i<n;i++) scanf("%lld",&b[i]);
	l=1,r=1e10;
	while(l<=r)
	{
		mid=(l+r)>>1;
		if(fuc(mid)) l=mid+1;
		else r=mid-1;
	}
	printf("%lld\n",l-1);
}
