#include<iostream>
using namespace std;
typedef long long int ll;
int main()
{	//freopen("in.txt","r",stdin);
	int n,k;
	cin>>n>>k;
	int a[n],b[n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	for(int i=0;i<n;i++)
	cin>>b[i];
	ll left=0,right=(ll)2e9,md;
	while(left<right)
	{
		md=(left+right+1)>>1;
		ll mg=0;
		for(int i=0;i<n;i++)
		{
			mg+=max(0LL,(ll)a[i]*md-b[i]);
			if(mg>k)break;
		}
		if(mg<=k)
		left=md;
		else right=md-1;
	}
	cout<<left<<"\n";
	return 0;
}
