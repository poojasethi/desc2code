#include<cstdio>
long long n, m, dane[100003][2], stan;

int main(){
	long long i, j, p, k, s;
	scanf("%I64d%I64d", &n, &m);
	for(i=0; i<n; i++)
		scanf("%I64d", &dane[i][0]);
	for(i=0; i<n; i++)
		scanf("%I64d", &dane[i][1]);
		
	p=0; k=3*1e9;
	while(p<k){
		s=(p+k+1)/2;
		stan=m;
		for(i=0; i<n; i++){
			if(dane[i][0]*s>dane[i][1])
				stan+=dane[i][1]-dane[i][0]*s;
			if(stan<0)
				break;
		}
		if(stan>=0)
			p=s;
		else
			k=s-1;
		
	
	}
	printf("%I64d\n", p);

	return 0;
}
