#include<cstdio>
using namespace std;
int i,n;
long long l,r,mid,a[100010],b[100010],now,k;
bool check(long long x)
{
	long long now=k;
	for (i=1;i<=n;i++)
	{
		if (a[i]*x>b[i]) now-=(a[i]*x-b[i]);
		if (now<0) return 0;
	}
	return 1;
}
int main()
{
	scanf("%d%I64d",&n,&k);
	for (i=1;i<=n;i++) scanf("%I64d",&a[i]);
	for (i=1;i<=n;i++) scanf("%I64d",&b[i]);
	l=0;r=(b[1]+k)/a[1];
	for (i=2;i<=n;i++)
	{
		now=(b[i]+k)/a[i];
		if (now<r) r=now;
	}
	while (l<r)
	{
		mid=(l+r+1)/2;
		if (check(mid)) l=mid;
		else r=mid-1;
	}
	printf("%I64d\n",l);
 }
