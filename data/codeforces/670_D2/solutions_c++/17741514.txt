#include<iostream>
using namespace std;
typedef long long ll;
const int maxn=11111111;
ll inf=0x7f7f7f7f;
ll n,k;
ll a[maxn],b[maxn];
int check(ll x)
{
	ll m=k;
	for(int i=0;i<n;i++)
	{
		if(b[i]/a[i]>=x)continue;
		if((b[i]+m)/a[i]>=x){
			m-=(a[i]*x-b[i]);
		}
		else{
			return false;
		}
	}
	return true;
}
int main()
{
	cin>>n>>k;
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<n;i++)cin>>b[i];
	ll l=0,r=inf;
	ll ans;
	while(l<=r)
	{
		ll mid=(l+r)/2;
		if(check(mid)){
			ans=mid;
			l=mid+1;
		}
		else{
			r=mid-1;
		}
	}
	cout<<ans<<endl;
}
