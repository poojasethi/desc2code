#include<cstdio>
#include<algorithm>
using namespace std;
const int f=100010;
long long k,b[f],a[f],r;
int n,i;
int main()
{
	r=0;
	scanf("%d%lld",&n,&k);
	for(i=1;i<=n;i++)
		scanf("%lld",&a[i]);
	for(i=1;i<=n;i++)
	{	scanf("%lld",&b[i]);r=max(r,(b[i]+k)/a[i]);}
	long long l=0,mid,sum;
	while(l<=r)
	{
		mid=(l+r)/2;
		sum=0;
		for(i=1;i<=n;i++)
		{	
			if(mid*a[i]>b[i])
				sum=sum+mid*a[i]-b[i];
			if(sum>k)
				break;
		}
	if(sum==k)
		{r=mid;break;}
	else if(sum<k)
		l=mid+1;
	else
		r=mid-1;		
	}
	printf("%d\n",r);
	return 0;

}
