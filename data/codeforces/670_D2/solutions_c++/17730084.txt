#include<cstdio>
#include<algorithm>
#define ll long long
using namespace std;
ll n,m;
ll a[100100],b[110000];

bool check(ll x)
{
	ll i,j,k=m;
	for(i=1;i<=n;i++)
	{
		if(a[i]*x-b[i]>k) return 0;
		if(a[i]*x-b[i]>0) k-=a[i]*x-b[i];
	}
	return 1;
}

int main()
{
	ll i,j,k;
	scanf("%I64d%I64d",&n,&m);
	for(i=1;i<=n;i++)
		scanf("%I64d",&a[i]);
	for(i=1;i<=n;i++)
		scanf("%I64d",&b[i]);
	ll l=0,r=2000000001;
	while(r-l>1)
	{
		ll mid=l+r>>1;
		if(check(mid)) l=mid;
		else r=mid;
	}
	printf("%I64d",l);
	return 0;
}
