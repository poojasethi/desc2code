#include<stdio.h>

int a[100010], b[100010];

bool ok(int n, long long m, long long k)
{
	for(int i=0; i<n; i++){
		long long t=a[i]*m;
		if(b[i]<t){
			k-=t-b[i];
			if(k<0)
				return 0;
		}
	}
	return 1;
}

int main ()
{
	int n;
	long long k;

	scanf("%d %I64d", &n, &k);
	for(int i=0; i<n; i++)
		scanf("%d", &a[i]);
	for(int i=0; i<n; i++)
		scanf("%d", &b[i]);

	long long l=0, r=2e9+1;
	while(l+1<r){
		long long m=(l+r)/2;
		if(ok(n, m, k))	l=m;
		else	r=m;
	//	printf("%d %d\n", l, r);
	}
	printf("%I64d\n", l);
}
