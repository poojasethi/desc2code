#include <iostream>

using namespace std;

long long a[100000], b[100000];
long long n, k;

bool valid(long long m)
{
	long long sum = 0;
	for (int i = 0; i < n; ++i)
		if (a[i] * m > b[i])
		{
			sum += a[i] * m - b[i];
			if (sum > k)
				return false;
		}
	return true;
}

int main()
{
	cin >> n >> k;
	for (int i = 0; i < n; ++i)
		cin >> a[i];
	for (int i = 0; i < n; ++i)
		cin >> b[i];

	long long l = 0, r = 2000000000, mid;
	while (l < r)
	{
		mid = (l + r) / 2;
		if (valid(mid + 1))
			l = mid + 1;
		else
			r = mid;
	}
	cout << l;
}
