#include <bits/stdc++.h>
using namespace std;
const int maxn=100010;

int a[maxn],b[maxn];
int n,k;

bool Check(long long x)
{
	long long rest=k;
	for (int i=1;i<=n;i++)
	{
		if (b[i]<a[i]*x) rest-=a[i]*x-b[i];
		if (rest<0) return false;
	}
	return true;
}

int main()
{
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	for (int i=1;i<=n;i++) scanf("%d",&b[i]);
	long long l=0,r=2e9,mid;
	while (l<r)
		if (Check(mid=(l+r+1)>>1)) l=mid;else r=mid-1;
	printf("%d\n",l);
	return 0;
}
