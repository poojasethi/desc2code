#include <iostream>

using namespace std;
const int maxn=100010;
long long n,k,a[maxn],b[maxn],ans,l=0,r=2000000000,m,kk;


bool check(long long m)
{
    kk=k;
    for (int i=1;i<=n;++i)
    {
        kk-=max(0ll,a[i]*m-b[i]);
        if (kk<0) return false;
    }
    return kk>=0;
}

int main()
{
    cin>>n>>k;
    for (int i=1;i<=n;++i) cin>>a[i];
    for (int i=1;i<=n;++i) cin>>b[i];
    while (l<=r)
    {
        m=(l+r)>>1;
        if (check(m)) ans=max(ans,m),l=m+1;
        else r=m-1;
    }
    cout<<ans;
    return 0;
}
