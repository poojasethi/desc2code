#include<stdio.h>
long long n,k,i,sum,a[100005],b[100005];
long long search(long long l,long long r)
{
	long long m,sum;
	while(l<=r)
	{
		m=(l+r)/2;
		sum=0;
		for(i=1;i<=n;i++)
		{
			if(a[i]*m>b[i])
				sum+=(a[i]*m-b[i]);
			if(sum>k)
				break;
		}
		if(sum==k)
			return m;
		else if(sum<k)
			l=m+1;
		else
			r=m-1;
	}
	return r;
}
int main()
{
	while(scanf("%lld%lld",&n,&k)!=EOF)
	{
		for(i=1;i<=n;i++)
			scanf("%lld",&a[i]);
		for(i=1;i<=n;i++)
			scanf("%lld",&b[i]);
		printf("%lld\n",search(1,2000000002));   
	}
	return 0;
}
