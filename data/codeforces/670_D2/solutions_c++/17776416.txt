#include <iostream>

using namespace std;

int n, m;
int a[100010], b[100010];

int main() {
	cin >> n >> m;
	for (int i = 1; i <= n; ++i) cin >> a[i];
	for (int i = 1; i <= n; ++i) cin >> b[i];

	int l = 0, r = 2000000000;
	while (l != r) {
		int mid = ((long long)l + (long long)r) / 2 + 1;
		long long need = 0;
		for (int i = 1; i <= n; ++i) {
			need += max((long long)mid * a[i] - b[i], 0LL);
			if (need > m) break;
		}
		// cout << need << endl;
		if (need <= m) l = mid;
		else r = mid - 1;
		// cout << l << " " << r << endl;
	}
	cout << l << endl;
}
