#include <stdio.h>
#include <iostream>

using namespace std;
long long a[100010],b[100010];
int main()
{
long long n,k,ans,max,poc,kraj,mid;
long long suma;
scanf("%I64d%I64d",&n,&k);
for(int i=0;i<n;i++){
scanf("%I64d",&a[i]);}
for(int i=0;i<n;i++){
scanf("%I64d",&b[i]);}
max=0;
poc=0;
kraj=2000000001;
while(poc<=kraj){
mid=(poc+kraj)/2;
suma=0;
for(int i=0;i<n;i++){
if(a[i]*mid>b[i]){
if(a[i]*mid-b[i]>2000000001-suma){
i=n;
suma=k+1;}else{
suma+=(a[i]*mid-b[i]);}}}
if(suma<=k){
max=mid;
poc=mid+1;}else{
kraj=mid-1;}}
cout<<max;
return 0;}
