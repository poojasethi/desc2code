#include<stdio.h>
#define MAXN 110000
int main()
{
	__int64 t,m,n,l,r,mid,sum;
	__int64 a[MAXN],b[MAXN];
	while(scanf("%I64d%I64d",&n,&m)!=EOF)
	{
		for(int i=0;i<n;i++)  scanf("%I64d",&a[i]);
		for(int i=0;i<n;i++)  scanf("%I64d",&b[i]);
		l=0;
		r=2*1e9+10;
		while(l<=r)
		{
			mid=(l+r)/2;
	        sum=0;
			for(int i=0;i<n;i++)
			{
				if(a[i]*mid>b[i])  sum=sum+(a[i]*mid-b[i]);
				if(sum>m)  break; 
			}
			if(sum<m)  l=mid+1;
			else if(sum==m)
			{
				l=mid+1;
				break;
			}
			else  r=mid-1;
		}
		printf("%I64d\n",l-1);
	}
	return 0;
}
