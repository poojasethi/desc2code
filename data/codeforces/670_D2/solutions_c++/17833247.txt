#include <bits/stdc++.h>
using namespace std;

long long n, k;
long long a[1000005], b[1000005];
long long r, l;

int main() {
  scanf("%lld%lld", &n, &k);
  for (int i = 0; i<n; i++) scanf("%lld", &a[i]);
  for (int i = 0; i<n; i++) scanf("%lld", &b[i]);
  r=0, l=2000000000;
  int maxi = 0;
  while (r!=l) {
    long long m = (r+l+1)/2;
    long long c = 0;
    for (int i = 0; i<n; i++) {
      c+=max(a[i]*m-b[i], (long long)0);
      if (c>k) break;
    }
    if (c<=k) r=m;
    else l=m-1;
  }
  printf("%lld\n", l);
}
