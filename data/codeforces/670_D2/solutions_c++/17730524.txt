#include<bits/stdc++.h>

using namespace std;

#define sgmn int v, int tl, int tr
#define tm   (tl + tr >> 1)
#define sol  v*2, tl, tm
#define sag  v*2+1, tm+1, tr
#define N    200005

long long A[N],B[N],n,k,i,l,m,r,t;

int main(){

	cin >> n >> k;

	for(i=1;i<=n;i++)
		cin >> A[i];

	for(i=1;i<=n;i++)
		cin >> B[i];

	l = 0;
	r = N * 10004;

	while(l + 1 < r){

		m = l + r >> 1;

		t = k;

		for(i=1;i<=n;i++){

			t -= max(0LL , m * A[i] - B[i]);
			if(t < 0)
				break;
		}

		if(i == n+1)  l = m;
		else          r = m;
	}

	cout << l;
	return 0;
}
