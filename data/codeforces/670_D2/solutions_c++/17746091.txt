#include<bits/stdc++.h>
using namespace std;

int i,n,k,a[100005],b[100005];
long long st=1,dr=3e9+69,pivot,rs;

bool Check(long long x) {
    long long aux=k;

    for(int i=1;i<=n && aux>=0;++i)
    if(1LL*x*a[i]>b[i]) aux-=1LL*x*a[i]-b[i];

    return aux>=0;
}

int main()
{
  ios_base::sync_with_stdio(0); cin.tie(0);

  cin>>n>>k;
  for(i=1;i<=n;++i) cin>>a[i];
  for(i=1;i<=n;++i) cin>>b[i];

  while(st<=dr)
  {
    pivot=st+(dr-st)/2;
    if(Check(pivot)) rs=pivot,st=pivot+1;
    else dr=pivot-1;
  }

  cout<<rs<<'\n';

 return 0;
}
