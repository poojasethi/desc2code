#include<bits/stdc++.h>
#define ll long long int
#define pb push_back
#define mp make_pair

using namespace std;

ll a[100002],b[100002],n,k,i,j;
bool check(ll x){
    ll sum=0;
    for(i=0;i<n;i++){
        sum+=max((a[i]*x-b[i]),0LL);
        if(sum>k) return false;
    }
    return true;
}

int main(){
cin>>n>>k;
for(i=0;i<n;i++){
    cin>>a[i];
}
for(i=0;i<n;i++){
    cin>>b[i];
}
ll l=0,h=2000000001,mid;
while(l<h){
    mid=(l+h+1)/2;
    if(check(mid)) l=mid;
    else h=mid-1;
}
cout<<l<<endl;

return 0;
}
