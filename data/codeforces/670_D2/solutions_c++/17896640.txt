#include <cstdio>

typedef long long ll;
const int maxn=100000;
const int maxk=2000000000;
int n,k,ans,a[maxn],b[maxn];

int main()
{
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++)scanf("%d",&a[i]);
	for(int i=0;i<n;i++)scanf("%d",&b[i]);
	
	unsigned l=0,r=maxk;
	while(l<=r)
	{
		int m=(l+r)/2,s=k;
		for(int i=0;i<n;i++)if(s<(ll)a[i]*m-b[i])s=-1;else if((ll)a[i]*m-b[i]>0)s-=a[i]*m-b[i];
		if(s!=-1)ans=m,l=m+1;else r=m-1;
	}
	printf("%d\n",ans);
	
	return 0;
}
