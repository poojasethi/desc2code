#include <bits/stdc++.h>
using namespace std;
const int mod = 1e9+7;
#define ll long long
ll sum(ll x){
	if(x&1)
		return ((x%mod)*(((x+1)/2)%mod))%mod;
	return ( ((x/2)%mod)*((x+1)%mod) )%mod;
}
int main(){
	ll n,m,i,j,l,r;cin>>n>>m;
	ll ret=0;
	for(i=1;i<=min(n,m);){
		ll t = n/i;
		l = i,r = n/t;r= min(min(n,m),r);
		ll s = (sum(r)-sum(l-1) + mod)%mod;
		ret = ret + (s*t)%mod;
		ret%=mod;;
		i = r+1;
	}
	ret*=-1;
	ret=(mod+ ret+ ((n%mod)*(m%mod))%mod )%mod;
	cout<<ret<<endl;
}
