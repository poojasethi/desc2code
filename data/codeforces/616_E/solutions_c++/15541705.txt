#include <bits/stdc++.h>

#define MOD 1000000007

using namespace std;

long long x, n, k;

int main()

{



    cin >> k >> n;

    if(n >= k)

    {
        x = k % MOD * ((n - k) % MOD) % MOD;
        n = k - 1;
    }

    for(long long L = 1; L <= n; L++) {
        long long t = k / L, R = t ? k / t : n;
        if(R > n) R = n;
        (x += (k % MOD) * ((R - L + 1) % MOD) % MOD - (t % MOD) * ((L + R) % MOD) % MOD * ((R - L + 1) % MOD) % MOD * 500000004 % MOD) %= MOD;
        L = R;
    }

    cout << (x % MOD + MOD) % MOD << "\n";

    return 0;
}
