#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const int P=1e9+7;
LL n,m,last,i;int ok,ans;
int mul(LL a,LL b){
    a%=P;b%=P;return 1ll*a*b%P;
}
int Seq(LL a,LL b){
    if ((b-a+1)&1) return mul((a+b)/2,b-a+1);
    return mul((b-a+1)/2,a+b);
}
int main(){
    scanf("%I64d%I64d",&n,&m);
    for (i=ok=1;i<=n&&ok;i=last+1){
        last=n/(n/i);
        if (last>=m) ok=0,last=m;
        ans+=mul(n/i,Seq(i,last));
        ans%=P;
  }ans=mul(n,m)-ans;
  printf("%d\n",(ans+P)%P);
}
