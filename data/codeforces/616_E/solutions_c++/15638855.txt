#include<cstdio>
#include<iostream>
using namespace std;
typedef __int64 LL;
const int mod=1e9+7;
int main(){
    #ifdef DouBi
    freopen("in.cpp","r",stdin);
    #endif // DouBi
    LL n,m;
    while(scanf("%I64d%I64d",&n,&m)!=EOF){
        LL ans=0;
        LL r=m;
        while(r>0){
            LL k=n/r;
            LL l=n/(k+1)+1;
            //printf("%I64d %I64d\n",l,r);
            LL bns;
            if((l+r)%2==0){
                bns=((((l+r)/2)%mod)*((r-l+1)%mod))%mod;
            }
            else {
                bns=((((r-l+1)/2)%mod)*((r+l)%mod))%mod;
            }
            ans+=(bns*(k%mod))%mod;ans%=mod;
            r=l-1;
        }
        LL bns=((n%mod)*(m%mod))%mod;
        printf("%I64d\n",(bns-ans+mod)%mod);
    }
    return 0;
}
