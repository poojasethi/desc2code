#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#define mo 1000000007
#define I2 500000004
#define LL long long
#define oo (1<<30)
using namespace std;
LL n,m,ans;
LL S(LL l){ l%=mo;  return l*(l+1)%mo*I2%mo;}
LL S(LL l,LL r){ return (S(r)-S(l-1)+mo)%mo;}
int main()
{
    LL i,r;
    cin>>n>>m;
    if(m>n)
        ans=(n%mo)*((m-n)%mo),m=n;
    ans%=mo;
    (ans+=(m%mo)*(n%mo))%=mo;
    for(i=1;i<=m;i=r+1){
        r=min(m,n/(n/i));
        ans-=S(i,r)*(n/i)%mo;
        ans%=mo;
      }
    cout<<(ans+mo)%mo;
    return 0;
}
