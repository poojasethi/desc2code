#include <iostream>
#include <string.h>
#include <stdio.h>
#include <algorithm>
#include <vector>
#define ll long long
#define mod 1000000007
using namespace std;

int main()
{
    //freopen("dd.txt","r",stdin);
    ll n,m;
    cin>>n>>m;
    ll sum = 0;
    ll limit=min(n,m);
    for(ll i = 1, j; i <=limit ; i = j + 1) {

        j = n / (n / i);
        //cout<<i<<" "<<j<<endl;
        j=min(m,j);
        ll a=(i+j),b=(j-i+1);
        if(a%2==0)
            a/=2;
        if(b%2==0)
            b/=2;
        a%=mod;
        b%=mod;
        sum =(sum+a*b%mod*((n / i)%mod)%mod)%mod;
    }
    n%=mod;
    m%=mod;
    ll ans=((n*m%mod-sum)%mod+mod)%mod;
    cout<<ans<<endl;
    return 0;
}
