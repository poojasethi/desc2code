
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
const __int64 mod=1e9+7;
int main(){
	__int64 n,m,i,sum=0,shengxia,t=0;
	cin>>n>>m;
	shengxia=m+1;
	sum+=(n%mod)*(m%mod);
	m=min(m,n);
	for(i=1;i<=(__int64)sqrt(n*1.0);i++){
		__int64 l,r;
		l=n/(i+1)+1;r=n/i;
		r=min(m,r);
		if(l>r)continue;
		shengxia=min(shengxia,l);
		__int64 a,b;
		a=l+r;b=r-l+1;
		if(a%2==0)a/=2;
		else b/=2;
		a%=mod;b%=mod;
		a=a*b%mod;
		a=a*i%mod; 
		t=(t+a)%mod;
	}
	sum=(sum-t+mod)%mod;
	for(i=1;i<shengxia;i++){
		t=n/i%mod*i%mod;
		sum=(sum-t+mod)%mod;
	}
	cout<<sum<<endl;
	return 0;
}
