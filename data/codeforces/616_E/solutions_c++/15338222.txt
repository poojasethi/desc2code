#include<string>

#include<algorithm>
#include<cmath>

#define Int long long
#define Mod 1000000007
using namespace std;
Int r,Ans=0,n,k;
int main()
{
    cin>>k>>n;
    if(n>k) Ans=Ans+(((n-k)%Mod)*(k%Mod))%Mod,n=k;
    for(Int i=1;i<=n;i=r+1)
    {
        r=k/(k/i);
        if(r>n) r=n;
        Int d=k/i,x=k%i,y=k%r,A,B;
        A=x+y;
        B=r-i+1;
        if(A%2==0) A=A/2;
        else B=B/2;
        Ans+=((A%Mod)*(B%Mod))%Mod;
        Ans%=Mod;
    }
    cout<<Ans;
}
