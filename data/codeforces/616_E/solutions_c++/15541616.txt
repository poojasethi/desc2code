#include <bits/stdc++.h>

#define MOD 1000000007

using namespace std;

const double eps = 1e-10;
const double pi = acos(-1.0);

const int INF = 1e9+7;
const int N = 1e5+7;

int main()

{

    long long res = 0, n, k;

    cin >> k >> n;

    if(n >= k)

    {
        res = k % MOD * ((n - k) % MOD) % MOD;
        n = k - 1;
    }

    for(long long L = 1; L <= n; L++) {
        long long t = k / L, R = t ? k / t : n;
        if(R > n) R = n;
        (res += (k % MOD) * ((R - L + 1) % MOD) % MOD - (t % MOD) * ((L + R) % MOD) % MOD * ((R - L + 1) % MOD) % MOD * 500000004 % MOD) %= MOD;
        L = R;
    }

    cout << (res % MOD + MOD) % MOD << "\n";

}
