#include<bits/stdc++.h>
using namespace std;
#define ll long long
long long int n,m,i,ans,r,mo,MOD=1000000007,e;
int main()
{
	std::ios_base::sync_with_stdio(false);
	cin>>n>>m;
	//	if(n==20000000){cout<<176305083;return 0;}
	if(m>n){
		ans=(((m-n)%MOD)*(n%MOD))%MOD;
		m=n;}
	for(i=1;i<=m;i++){
		e = n/i;
		mo = n%i;
		r = mo/e;
		if(i+r>m )
			r=m-i;
		mo%=MOD;
		e%=MOD;
		if(r&1)
			ans=(((ans)+((((((((((mo))*((((r+1))%MOD))))%MOD))-((((((((e))*((((r%MOD))*((((((r+1))>>1))%MOD)))%MOD)))%MOD))%MOD))+((MOD))))%MOD))))%MOD;
		else 
			ans=(((ans)+((((((((((mo))*((((r+1))%MOD))))%MOD))-((((((((e))*((((((r>>1))%MOD))*((((r+1))%MOD)))%MOD)))%MOD))%MOD))+((MOD))))%MOD))))%MOD;

		i+=r;
	}

	cout<<ans;
	return 0;
}
