#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long ll;
const ll mod = 1e9+7;
const ll inv2= 5e8+4;
ll n,m,ans;

int main(){
    scanf("%I64d%I64d",&n,&m);
    ans = 0;
    for(ll i=1;i<=m;i++){
        ll p = n/i;
        if(p==0){
            ans = (ans + ((m-i+1)%mod*(n%mod))%mod)%mod;
            break;
        }
        ll j = min(n/p,m);
        ans = (ans + ((n%i+n%j)%mod*((j-i+1)%mod)%mod*inv2)%mod)%mod;
        i=j;
    }
    printf("%I64d\n",ans);
}
