#include<cstdio>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
using namespace std;
const long long mod=(long long)(1e9+7);
const long long inv=(long long)(5e8+4);
int main()
{
    long long n,m;
    cin>>n>>m;
    long long res=(n%mod)*(m%mod)%mod;
    long long mina=min(n,m);
    long long l=1;
    long long tmp=0;
    long long r;
    while(l<=mina)
    {
        long long r=min(m,(n/(n/l)));
        tmp+=((n/l)%mod)*((r+l)%mod)%mod*((r-l+1)%mod)%mod*inv%mod;
        tmp%=mod;
        l=r+1;
        //cout<<l<<endl;
    }
    res=(res-tmp+mod)%mod;
    cout<<res<<endl;

    return 0;
}
