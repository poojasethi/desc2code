#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
typedef __int64 ll;
const ll mod = 1e9+7;
ll n,m;

int main(void){
	
	while(~scanf("%I64d%I64d",&n,&m)){
		ll ans=(n%mod)*(m%mod)%mod;
		ll lim=(ll)sqrt(n);
		//printf("lim %I64d ans %I64d\n",lim,ans);
		for(ll i=1;i<=lim && i<=m;i++){
			ans=(ans-n/i*i%mod)%mod;
			ans=((ans%mod)+mod)%mod;
			//if(ans>=mod) ans%=mod;
		}
		if(lim>=m){
			printf("%I64d\n",ans%mod);
			continue;
		}
		for(ll i=lim+1;i<=min(n,m);){
			ll v=n/i;
			ll rg=min(n/v,m),lf=n/(v+1);
			ll sum=(rg+lf+1)%(2*mod)*((rg-lf)%(2*mod))%(2*mod)/2;
			ans=(ans-sum*v%mod)%mod;
			ans=((ans%mod)+mod)%mod;
			//if(ans>=mod) ans%=mod;
			i=rg+1;
		}
		printf("%I64d\n",ans%mod);
	}
	return 0;
}
