#include <bits/stdc++.h>
using namespace std;
const int mod = 1e9+7;
long long n,m,sum,l,r,tmp;
int sn;
int main(){
    cin>>n>>m;
    sn=trunc(sqrt(n));
    for (int i=1; i<=sn; i++){
        if (i<=m){
            sum+=1LL*(n/i)*i;
            sum%=mod;
        }
    }
    for (int i=1; i<=sn; i++){
        l=(n/(long long)(i+1))+1;
        r=n/i;r=min(r,m);
        l=max(l,1LL*sn+1);
        if (l<=r){
            /// i*(l+...+r)
            if ((l+r)%2==0) tmp=(((l+r)/2LL)%mod)*((r-l+1)%mod); else tmp=(((r-l+1)/2LL)%mod)*((l+r)%mod);
            tmp%=mod;
            tmp*=i;
            tmp%=mod;
            sum+=tmp;
            sum-=(sum>=mod?mod:0);
        }
    }
    tmp=(n%mod)*(m%mod);
    tmp%=mod;
    tmp-=sum;
    if (tmp<0) tmp+=mod;
    cout<<tmp;
}
