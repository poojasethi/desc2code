#include <iostream>  
#include <algorithm>  
#include <cstdio>  
#include <cstring>  
#include <string>  
using namespace std;  
#define ll __int64  
ll mod=1e9+7;  
int main()  
{  
    ll n, m;  
    scanf("%I64d%I64d", &n, &m);  
    ll ans = n%mod*(m%mod)%mod;  
    ll sum=0, j, mm;  
    for(ll i=1; i<=min(n, m); i++)  
    {  
        j = min(n/(n/i), m);  
        ll s1 = i+j, s2 = j-i+1;  
        if(s1%2==0) s1 = (s1/2)%mod, s2%=mod;  
        else s1%=mod, s2 = (s2/2)%mod;  
        mm = s1*s2%mod*(n/i%mod)%mod;  
        sum = (sum+mm)%mod;  
        i = j;  
    }  
    ans = (ans-sum+mod)%mod;  
    printf("%I64d\n", ans);  
    return 0;  
}
