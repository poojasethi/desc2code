#include <cstdio>
#include <vector>
#include <algorithm>
#include <iostream>
#include <string>
#include <cmath>

using namespace std;

const long long p = 1000000007;

long long Mul(long long a, long long b) {
	return ((a % p) * (b % p)) % p;
}

long long Sum(long long l, long long r) {
	return Mul(Mul(r + l + 1, r - l), (p + 1) / 2);
}

int main() {
	long long n, m;
	cin >> n >> m;
	long long answer = Mul(n, m);
	long long min_value = min(n, m);
	for (long long i = 1; i * i <= n; ++i) {
		long long l = n / (i + 1);
		long long r = min(m, n / i);
		if (l < r) {
			answer = (answer - Mul(i, Sum(l, r))) % p;
			min_value = l;
		}
	}
	for (long long i = 1; i <= min_value; ++i) {
		answer = (answer - Mul(i, n / i)) % p;
	}
	cout << (answer % p + p) % p;
	return 0;
}
