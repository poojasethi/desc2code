#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll fun(ll a)
{
    ll x;
    if(a&1)
        x=(a+1)/2,x=(x%mod)*(a%mod);
    else
        x=a/2,x=(x%mod)*((a+1)%mod);
    return x%mod;
}
int main()
{
    ll n,m,i,t,l,r,x;
    cin>>n>>m;
    t=(n%mod)*(m%mod)%mod;
    x=0;
    ll mi=min(n,m);
    for(i=1;i*i<=n;i++)
    {
       
        l=n/(i+1);
        r=min(m,n/i);
        if(l>=r)continue;
        mi=l;
        x=(x+i*fun(r)%mod-i*fun(l)%mod+mod)%mod;
        //cout<<x<<"\n";
    }
    for(i=1;i<=mi;i++)
    	 x=(x+(n/i)*i)%mod;
    //cout<<t<<"\t"<<x<<"\n";
    cout<<(t-x+mod)%mod;
}
