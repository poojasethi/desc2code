#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
typedef long long ll;
const int mm=1000000007;
ll n,m,k,i,j,iv,t;
int ans;
ll pw(ll b,ll p){
	ll res=1;
	for (;p;p>>=1,b=b*b%mm)
	 if (p&1) res=res*b%mm;
	return res;
}
int main(){
	cin>>n>>m;ans=0;iv=pw(2,mm-2);
	for (i=1,k=min(n,m);i<=k;i=j+1){
		t=n/i;j=min(k,n/t);
		ans=(ans+(ll)((n%i+n%j)%mm)*(ll)((j-i+1)%mm)%mm*iv%mm)%mm;
	}
	if (m>n) ans=(ans+(n%mm)*((m-n)%mm)%mm);
	printf("%d\n",ans);
	return 0;
}
