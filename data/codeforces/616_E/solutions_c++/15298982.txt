#include <bits/stdc++.h>
using namespace std;
long long mod = 1e9 + 7;
long long inv = 5e8 + 4;
long long mul(long long a, long long b) {
	a %= mod;
	b %= mod;
	return (a * b) % mod;
}
long long sum(long long l, long long r) {
	long long ret1 = mul(l, l - 1);
	ret1 = mul(ret1, inv);
	long long ret2 = mul(r, r + 1);
	ret2 = mul(ret2, inv);
	return (ret2 - ret1 + mod) % mod;
}
int main(){
	long long n, m;
	cin>>n>>m;
	long long ans = 0;
	if(m > n) ans += mul(n, m - n), m = n;
	long long l = 1, r = 1;
	while (l <= m){
	    long long tmp = n / l;
	    l = (n + 1 + tmp) / (tmp + 1);
	    r = min(n / tmp, m); 
	    assert(n / l == n / r && l <= r);
	    ans += mul(r - l + 1, n) - mul(sum(l, r), tmp);
	    ans %= mod;
	    ans = (ans + mod) % mod;
	    l = r + 1;
	}
	cout<<ans<<endl;
}
