#include <iostream>
#include <cstring>
#include <cstdio>
#define LL long long
using namespace std;
const LL MOD=1e9+7;
LL n,m;
int main(){
    cin>>n>>m;
    LL ans=((n%MOD)*(m%MOD))%MOD,sum=0,mmin=min(n,m);
    for(LL l=1,r;l<=mmin;l=r+1){
        LL tmp=n/l;
        r= min(mmin,n/tmp);
        LL lr=(l+r);
        LL rl=(r-l+1);
        if(lr%2 == 0) lr/=2;
        else rl/=2;
        lr%=MOD;rl%=MOD;tmp%=MOD;
        sum=( (sum%MOD) + ( ( (tmp*lr)%MOD) *rl)%MOD )%MOD;
    }
    ans=(ans-sum + MOD)%MOD;
    cout<<ans<<endl;
    return 0;
}
