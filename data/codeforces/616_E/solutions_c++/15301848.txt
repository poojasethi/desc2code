#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const uli mod=1e9+7;
const uli i2=500000004;
uli rng(uli x,uli y){
   x%=mod, y%=mod;
   uli l=x*(x+1)%mod*i2%mod;
   uli r=y*(y+1)%mod*i2%mod;
   uli ans=(r-l+x)%mod;
   return (ans+mod)%mod;
}
int main(){
   uli n,m;
   cin>>n>>m;
   uli s=0;
   uli r=m;
   while(r>0){
      uli v=n/r;
      uli l=n/(v+1)+1;
      uli vm=v%mod;
      uli inc=rng(l,r)*vm%mod;
      s=(s+inc)%mod;
      r=n/(v+1);
   }
   uli ans=(n%mod)*(m%mod)%mod;
   ans=(ans-s+mod)%mod;
   cout<<ans<<endl;
   return 0;
}
