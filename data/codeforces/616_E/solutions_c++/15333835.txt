#include<cstdio>
#include<cmath>
#include<iostream>
using namespace std;
typedef long long LL;
const LL MOD = 1e9+7;
#define M(x) ((x)%MOD)
LL n, m, sum;

int main() {
    scanf("%I64d%I64d", &n, &m);
    LL sqn = 1;
    while(sqn*sqn<=n) sqn++;
    if(m>n/sqn) {
        for(LL i = sqn; i>=2; --i)
        {
            LL l, r, addsum;
            l = n/i;
            r = min(n/(i-1),m);
            sum += M(r-l)*M(n);
            sum -= M((i-1)*M(r-l))*M(M(r+l+1)*M((MOD+1)/2));
            sum %= MOD;
            if(sum<0) sum += MOD;
            if(r==m) break;
        }
    }
    if(m>n) {
        sum += M(M(m-n)*M(n));
        sum %= MOD;
    }
    for(LL x = 1; x<=n/sqn&&x<=m; ++x)
    {
        sum = (sum+n%x)%MOD;
    }
    printf("%I64d\n", sum);
    return 0;
}
