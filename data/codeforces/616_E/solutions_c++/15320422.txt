#include <iostream>
#include <cmath>
using namespace std;
const int mod=int(1e9+7);
int main(){
    long long n,m,res=0;
    cin>>n>>m;
    for (long long i=1; i<=n;){
        long long v=n/i,l=n/(v+1)+1,r=min(n/v,m);
        if (l<=r){
            long long tmp1=r+l, tmp2=r-l+1;
            if (tmp2%(long long) 2) tmp1/=2;
            else tmp2/=2;
            tmp1%=mod, tmp2%=mod;
            res=(res+(((tmp1*tmp2)%mod)*v)%mod)%mod;
            i=r+1;
        }
        if (i>m) break;
    }
    cout<<((n%mod)*(m%mod)-res)%mod;
    return 0;
}
