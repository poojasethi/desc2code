#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define rep(i,a,b) for(int i=a;i<=b;++i)
#define LL long long
const int inf = 0x3f3f3f3f;
const LL P = 1e9+7;
const LL ni=500000004;

int main()
{
    LL n, m;  
    cin >> n >> m;
    LL res = 0;
    if(m>n) res+=(m-n)%P*(n%P), m=n, res%=P;
    for(LL i = 1; i <= n;)
    {
        LL r = min(m, n/i);
        LL l = (n/(i+1))+1;
        if (l<=r) res += (n%l + n%r)%P*((r-l+1)%P)%P * ni%P;
        res %= P;
        if (i==n) break;
        i = n / (n/(i+1));
    }
    cout << res << endl;
    return 0;
}
