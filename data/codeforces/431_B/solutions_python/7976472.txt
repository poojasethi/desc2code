b = [map(int,raw_input().split()) for x in xrange(5)]
mh = 0
from itertools import permutations
for x in permutations([0,1,2,3,4]):
    h = 0
    h += b[x[0]][x[1]] + b[x[1]][x[0]] + (b[x[2]][x[3]] + b[x[3]][x[2]])*2
    h += b[x[2]][x[1]] + b[x[1]][x[2]] + (b[x[3]][x[4]] + b[x[4]][x[3]])*2
    if h > mh:
        mh = h
print mh
