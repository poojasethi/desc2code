import itertools

a = [0,1,2,3,4]

g = []
for i in range(5):
    g.append([])
    for x in raw_input().split():
        g[i].append(int(x))


maxt = 0
for t in list(itertools.permutations(a)):
    tmp = g[t[0]][t[1]] + g[t[1]][t[0]] + g[t[2]][t[3]] + g[t[3]][t[2]]
    tmp += g[t[1]][t[2]] + g[t[2]][t[1]] + g[t[3]][t[4]] + g[t[4]][t[3]]
    tmp += g[t[2]][t[3]] + g[t[3]][t[2]]
    tmp += g[t[3]][t[4]] + g[t[4]][t[3]]
    if tmp > maxt:
        maxt = tmp
print maxt
