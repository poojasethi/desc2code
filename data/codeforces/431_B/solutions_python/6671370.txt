import sys,math,itertools

if __name__ == '__main__':
	bath = []
	for _ in range(5):
		bath.append(map(int,raw_input().split()))

	ans = 0
	perm = list(itertools.permutations([0,1,2,3,4], 5))
	for p in perm:
		pans = 0
		for i in range(4):
			for j in range(i,4,2):
				pans +=bath[p[j]][p[j+1]] + bath[p[j+1]][p[j]]
			ans = max(ans,pans)

	print ans
