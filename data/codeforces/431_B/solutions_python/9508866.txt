from itertools import permutations as p
l = list(p([1,2,3,4,5]))
g=[]
for i in range(5):
    g.append(map(int,raw_input().split()))
ma = 0
for it in l:
    hap = g[it[0]-1][it[1]-1] + g[it[1]-1][it[0]-1] + 2*g[it[2]-1][it[3]-1] + 2*g[it[3]-1][it[2]-1] + g[it[1]-1][it[2]-1] + g[it[2]-1][it[1]-1] + 2*g[it[3]-1][it[4]-1] + 2*g[it[4]-1][it[3]-1]
    if hap>ma:ma=hap
print ma
