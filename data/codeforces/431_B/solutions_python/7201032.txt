import itertools
ar = []
for i in xrange(5):
    ar +=[map(int,raw_input().split())]
rd = 0;
mx = 0
z = itertools.permutations([0,1,2,3,4],5)
for j in z:
    tc = []
    rd = 0;
    for q in j:
        tc += [q]
    while (len(tc) != 0):
        for i in range(len(tc))[::2]:
            try:
                rd += ar[tc[i]][tc[i+1]]
                rd += ar[tc[i+1]][tc[i]]
            except:
                pass
        del tc[0]
    if rd > mx:
        mx = rd
print mx
