import itertools, math

g = []

for _ in range(5): g.append(map(int, raw_input().split()))

seq = [0, 1, 2, 3, 4]
permut = itertools.permutations(seq)

ans = 0
i = 0
for p in permut:
    tmp_ans = g[p[0]][p[1]] + g[p[1]][p[0]] + 2 * (g[p[2]][p[3]] + g[p[3]][p[2]]) + g[p[1]][p[2]] + g[p[2]][p[1]] + 2 * (g[p[3]][p[4]] + g[p[4]][p[3]])
    ans = max(ans, tmp_ans)

print ans
