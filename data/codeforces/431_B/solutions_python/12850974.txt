from itertools import permutations

a = []

def happiness(l):
	acc = 0
	acc += a[l[0]][l[1]]
	acc += a[l[1]][l[0]]
	acc += a[l[1]][l[2]]
	acc += a[l[2]][l[1]]
	acc += 2*a[l[2]][l[3]]
	acc += 2*a[l[3]][l[2]]
	acc += 2*a[l[3]][l[4]]
	acc += 2*a[l[4]][l[3]]
	return acc


if __name__ == '__main__':
	for i in xrange(5):
		a += [[int(x) for x in raw_input().split()]]
	res = 0
	for i in permutations([0,1,2,3,4]):
		res = max(res,happiness(i))
