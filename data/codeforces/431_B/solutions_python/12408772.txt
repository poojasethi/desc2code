import sys
import itertools

lines = []
for line in sys.stdin:
    lines.append([int(n) for n in line.split()])

assert len(lines) == 5
for line in lines:
    assert len(line) == 5

bestHappiness = 0
for order in itertools.permutations(range(0, 5)):
    happiness = 0
    while len(order) >= 2:
        for i in xrange(0, len(order) - 1, 2):
            happiness += lines[order[i]][order[i+1]]
            happiness += lines[order[i+1]][order[i]]
        order = order[1:]
    if happiness > bestHappiness:
        bestHappiness = happiness

print bestHappiness
