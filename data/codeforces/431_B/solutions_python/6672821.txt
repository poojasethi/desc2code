from itertools import permutations as pt

f = [map(int, raw_input().split()) for i in range(5)]

sq = []
for i in pt('01234'):
    intt = map(int, i)
    sq.append(intt)

summ = 0
for i in sq:
    tmp = f[i[1]][i[2]] + f[i[2]][i[1]] + f[i[0]][i[4]] + f[i[4]][i[0]] \
          + f[i[0]][i[2]] + f[i[2]][i[0]] + f[i[4]][i[3]] + + f[i[3]][i[4]] \
          + f[i[0]][i[4]] + f[i[4]][i[0]] + + f[i[4]][i[3]] + f[i[3]][i[4]]
    if tmp > summ:
        summ = tmp

print summ
