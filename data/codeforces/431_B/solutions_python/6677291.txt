from itertools import permutations
gs = [map(int, raw_input().split()) for _ in range(5)]
def weight(s):
	w = 0
	w += gs[s[0]][s[1]] + gs[s[1]][s[0]] + gs[s[2]][s[3]] + gs[s[3]][s[2]]
	w += gs[s[1]][s[2]] + gs[s[2]][s[1]] + gs[s[3]][s[4]] + gs[s[4]][s[3]]
	w += gs[s[2]][s[3]] + gs[s[3]][s[2]]
	w += gs[s[3]][s[4]] + gs[s[4]][s[3]]
	return w
print max(map(weight, permutations(range(5))))
