import itertools

indices = [0, 1, 2, 3, 4]
lines = []
for i in xrange(5):
    line = map(int, raw_input().split())
    lines.append(line)

perm = list(itertools.permutations(indices))

ans = 0
for p in perm:
    happ = 0
    for j in xrange(4):
        for k in xrange(j, 4, 2):
            happ += lines[p[j]][p[j+1]] + lines[p[j+1]][p[j]]
    ans = max(ans, happ)

print ans
