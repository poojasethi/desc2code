from itertools import permutations
ans = 0
ma = [ map(int, raw_input().split()) for _ in range(5) ]

for p in permutations([0, 1, 2, 3, 4]):
	l = list(p)
	happiness = 0
	for i in range(5):
		for j in range(0, 5, 2):
			if j + 1 < len(l):
				happiness += ma[l[j]][l[j+1]] + ma[l[j+1]][l[j]]
			else:
				break
		l = l[1:]
	if happiness > ans: ans = happiness
print ans
