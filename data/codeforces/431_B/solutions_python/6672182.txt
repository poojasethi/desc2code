from itertools import permutations

lst = []
for i in range(5):
    a = [int(x) for x in raw_input().split()]
    lst.append(a)

ans = 0
for x in permutations(range(5)):
    #print x
    new_ans = (lst[x[1]][x[2]] + lst[x[2]][x[1]]) + (lst[x[0]][x[1]] + lst[x[1]][x[0]]) + 2 * (lst[x[2]][x[3]] + lst[x[3]][x[2]]) + 2 * (lst[x[3]][x[4]] + lst[x[4]][x[3]])
    if new_ans > ans:
        ans = new_ans

print ans
