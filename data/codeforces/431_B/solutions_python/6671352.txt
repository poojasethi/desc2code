from itertools import permutations
g = []
for _ in range(5):
    g.append(map(int, raw_input().split()))

orders = permutations(range(5))

greaterGood = 0
for p in orders:
    good = (g[p[0]][p[1]]+g[p[1]][p[0]] + g[p[2]][p[3]]+g[p[3]][p[2]] +
            g[p[1]][p[2]]+g[p[2]][p[1]] + g[p[3]][p[4]]+g[p[4]][p[3]] +
            g[p[2]][p[3]]+g[p[3]][p[2]] +
            g[p[3]][p[4]]+g[p[4]][p[3]])
    greaterGood = max(good, greaterGood)
print greaterGood
