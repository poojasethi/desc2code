g = [map(int, raw_input().split()) for i in range(5)]

import itertools as it
maxa = 0
for j in it.permutations('12345',5):

	j = map(int, j)
	j = [x - 1 for x in j]


	a = 0
	for i in range(4):
		a += g[j[i]][j[i+1]]
		a += g[j[i+1]][j[i]]
	a += (g[j[2]][j[3]]+g[j[3]][j[2]])
	a += (g[j[3]][j[4]]+g[j[4]][j[3]])

	if a > maxa:
		maxa = a

print maxa
