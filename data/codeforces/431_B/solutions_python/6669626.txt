#!/usr/bin/python

import sys
from itertools import permutations

def Ni(): return tuple(map(int, sys.stdin.readline().split()))
def Ns(): return tuple(map(int, sys.stdin.readline().split()))

g = [Ni() for i in range(5)]

answer = 0
for p in permutations(range(5)):
    value = 0
    value += g[p[0]][p[1]] + g[p[2]][p[3]]
    value += g[p[1]][p[2]] + g[p[3]][p[4]]
    value += g[p[2]][p[3]]
    value += g[p[3]][p[4]]
    value += g[p[1]][p[0]] + g[p[3]][p[2]]
    value += g[p[2]][p[1]] + g[p[4]][p[3]]
    value += g[p[3]][p[2]]
    value += g[p[4]][p[3]]
    answer = max(answer, value)

print answer
