from itertools import permutations
h, s, most = [], range(5), 0
for i in xrange(5): h.append(map(int, raw_input().split()))
for p in permutations(s):
	most = max(most, (h[p[0]][p[1]]+h[p[1]][p[0]]+h[p[2]][p[3]]+h[p[3]][p[2]]) +
		(h[p[1]][p[2]]+h[p[2]][p[1]]+h[p[3]][p[4]]+h[p[4]][p[3]]) +
		(h[p[2]][p[3]]+h[p[3]][p[2]]) + h[p[3]][p[4]]+h[p[4]][p[3]])
print most
