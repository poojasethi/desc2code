from sys import stdin
from itertools import permutations

h = [ [int(x) for x in l.split()] for l in stdin.readlines()]

#for l in h: print l

p = permutations(range(5))

m=0
t=0
for o in p:
	t= ( h[o[0]][o[1]] + h[o[1]][o[0]] +
		h[o[2]][o[3]]*2 + h[o[3]][o[2]]*2 +
		h[o[1]][o[2]] + h[o[2]][o[1]] +
		h[o[3]][o[4]]*2 + h[o[4]][o[3]]*2 )
	#print (o,t,m)

	if t>m:
		m=t

print m
