import itertools

g = [[0]*5]*5

def read():
    for i in range(5):
        g[i] = map(int, raw_input().split())

def cal(a):
    res = 0
    for i in range(len(a)-1):
        for j in range(i, len(a)-1, 2):
            res += g[int(a[j])][int(a[j+1])] + g[int(a[j+1])][int(a[j])]
    return res 

def main():
    read()
    ans = 0
    for i in itertools.permutations("01234", 5):
        ans = max(ans, cal(i))
    print ans
    
if __name__ == "__main__":
    main()
