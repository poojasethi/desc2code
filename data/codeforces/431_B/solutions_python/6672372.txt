import itertools
g = [[0 for i in range(5)] for j in range(5)]
for i in range(5):
	g[i] = map(int,raw_input().split(" "))
ans  = -1;
for i in itertools.permutations([0,1,2,3,4]):
	alocal = 0;
	for k in range(5):
		j=k
		while(j< len(i)):
			if j >= len(i)-1:
				break;
			alocal += g[i[j]][i[j+1]]+g[i[j+1]][i[j]]
			j+=2;
	ans = max(ans,alocal)
print ans
