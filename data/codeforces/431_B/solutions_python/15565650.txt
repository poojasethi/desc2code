from math import *
from itertools import *
import heapq
PI = acos(-1)
st = [0, 1, 2, 3, 4]
hap = []
for i in range(0, 5):
	s = map(int, raw_input().strip().split(' '))
	hap.append(s)
ans = 0
for cont in list(permutations(st)):
	res = 0
	for i in range(0, 5):
		j = i
		while(j<4):
			res += hap[cont[j+1]][cont[j]]+hap[cont[j]][cont[j+1]]
			j += 2
	ans = max(ans, res)
print ans
