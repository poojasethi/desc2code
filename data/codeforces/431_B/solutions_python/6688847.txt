import itertools
a=list(itertools.permutations([0,1,2,3,4]))

g=[]
for i in range(5):
    g.append(map(int,raw_input().split()))

hmax=0             

for i in a:
    h=[0,0,0,0,0]
    h[0]=g[i[0]][i[1]]
    h[1]=g[i[1]][i[0]]+g[i[1]][i[2]]
    h[2]=g[i[2]][i[3]]+g[i[2]][i[1]]+g[i[2]][i[3]]
    h[3]=g[i[3]][i[2]]+g[i[3]][i[2]]+g[i[3]][i[4]]+g[i[3]][i[4]]
    h[4]=g[i[4]][i[3]]+g[i[4]][i[3]]
    hmax=max(hmax,sum(h))
    
print hmax
