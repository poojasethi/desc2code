import itertools

mat = []
for _ in range(5):
    mat.append(map(int, raw_input().split()))

max_res = 0

for perm in itertools.permutations([0,1,2,3,4]):
    a,b,c,d,e = perm
    res = (mat[a][b] + mat[b][a] + mat[c][d] + mat[d][c])
    res += (mat[c][b] + mat[b][c] + mat[e][d] + mat[d][e])
    res += (mat[c][d] + mat[d][c] + mat[e][d] + mat[d][e])
    max_res = max([max_res, res])

print max_res
