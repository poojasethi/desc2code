from itertools import permutations
g = []
for i in range(5):
    t = map(int, raw_input().split())
    g.append(t)
perm = permutations(range(5))
ans = 0
for p in perm:
    t = g[p[0]][p[1]] + g[p[1]][p[0]] + g[p[2]][p[3]] + g[p[3]][p[2]]
    t += g[p[1]][p[2]] + g[p[2]][p[1]] + g[p[3]][p[4]] + g[p[4]][p[3]]
    t += g[p[2]][p[3]] + g[p[3]][p[2]] + g[p[3]][p[4]] + g[p[4]][p[3]]
    ans = max(ans,t)
print ans
