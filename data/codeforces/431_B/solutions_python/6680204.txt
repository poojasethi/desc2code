import itertools

n = 5
edges = []
for i in xrange(n):
    edges.append(map(int, raw_input().split()))

ret = 0
for students in itertools.permutations([0, 1, 2, 3, 4]):
    cnt = 0
    for i in xrange(n):
        for j in xrange(i, n - 1, 2):
            a = students[j]
            b = students[j + 1]
            cnt += edges[a][b] + edges[b][a]
    ret = max(ret, cnt)
print ret
