from itertools import permutations

g = []
for i in xrange(5):
	g.append(map(int, raw_input().split()))

happ_max = 0
for x in permutations(xrange(5), 5):
	temp = 0
	temp += g[x[0]][x[1]] + g[x[1]][x[0]] + g[x[2]][x[3]] + g[x[3]][x[2]]
	temp += g[x[1]][x[2]] + g[x[2]][x[1]] + g[x[3]][x[4]] + g[x[4]][x[3]]
	temp += g[x[2]][x[3]] + g[x[3]][x[2]]
	temp += g[x[3]][x[4]] + g[x[4]][x[3]]
	if temp > happ_max:
		happ_max = temp
print happ_max
