
import itertools
a = []
m = 0
for i in range(5): a.append(map(int, raw_input().split()))
p = list(itertools.permutations([0, 1, 2, 3, 4]))
for pi in p:
  c = 0
  c += a[pi[0]][pi[1]] + a[pi[1]][pi[0]] + a[pi[2]][pi[3]] + a[pi[3]][pi[2]]
  c += a[pi[1]][pi[2]] + a[pi[2]][pi[1]] + a[pi[3]][pi[4]] + a[pi[4]][pi[3]]
  c += a[pi[2]][pi[3]] + a[pi[3]][pi[2]]
  c += a[pi[3]][pi[4]] + a[pi[4]][pi[3]]
  m = max(m, c)
print m
