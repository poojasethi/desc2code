#include <iostream>
#include <algorithm>
using namespace std;
int main() {
    int a[5][5], ans = 0;
    for(int i = 0; i < 5; ++i) 
    for(int j = 0; j < 5; ++j) 
    cin>>a[i][j];
    int g[] ={0, 1, 2, 3, 4};
    do {
        ans = max(ans, a[g[0]][g[1]] + 2 * a[g[2]][g[3]] + a[g[1]][g[2]] + 2 * a[g[3]][g[4]] + a[g[1]][g[0]] + 2 * a[g[3]][g[2]] + a[g[2]][g[1]] + 2 * a[g[4]][g[3]] );
    }while(next_permutation(g,g+5));
    cout <<ans<<endl;
    return 0;
}
