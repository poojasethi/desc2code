#include<cstdio>
#include<algorithm>
using namespace std;
int g[6][6],p[6];
int main() {
	for(int i=1;i<=5;i++) for(int j=1;j<=5;j++) scanf("%d",&g[i][j]);
	for(int i=1;i<=5;i++) p[i]=i;
	int ans=0;
	do {
		int tmp=0;
		for(int i=1;i<=4;i++) tmp+=g[p[i]][p[i+1]]+g[p[i+1]][p[i]];
		for(int i=3;i<=4;i++) tmp+=g[p[i]][p[i+1]]+g[p[i+1]][p[i]];
		ans=max(ans,tmp);
	} while(next_permutation(p+1,p+6));
	printf("%d\n",ans);
	return 0;
}
