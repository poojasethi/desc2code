#include<bits/stdc++.h>
using namespace std;

int main()
{
	int a[6][6],x[6];
	for(int i=0; i<5; i++)
	{
		x[i] = i;
		for(int j=0; j<5; j++)
			cin>>a[i][j];
	}
	int ans=0,temp;
	for(int i=0; i<120; i++)
	{
		temp = 0;
		for(int j=0; j<5; j++)
		{
			for(int k=j; k<4; k+=2)
			{
				temp += a[x[k]][x[k+1]] + a[x[k+1]][x[k]];
			}
		}
		ans = max(ans,temp);
		next_permutation(x,x+5);
	}
	cout<<ans<<endl;
	return 0;
}
