#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
	int a[5][5],s[5]={0,1,2,3,4},max=0;
	for(int i=0;i<5;i++)
		for(int j=0;j<5;j++)
			cin>>a[i][j];
	for(int i=0;i<5;i++)
		for(int j=i+1;j<5;j++)
		{
			a[i][j]+=a[j][i];
			a[j][i]=a[i][j];
		}
	do
	{
		max=std::max(max,a[s[0]][s[1]]+a[s[1]][s[2]]+2*a[s[2]][s[3]]+2*a[s[3]][s[4]]);
	}while(next_permutation(s,s+5));
	cout<<max;
	return 0;
}
