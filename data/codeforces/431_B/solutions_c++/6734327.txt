#include<cstdio>
#include<algorithm>
using namespace std;
int edge[6][6];
int tab[]={1,2,3,4,5,6,7};
int main()
{
	for(int i=0;i<5;i++)
		for(int j=0;j<5;j++)
			scanf("%d",&edge[i][j]);
	int m=0;
	for(int q=0;q<5*4*3*2*1;q++)
	{
		int w=0;
		for(int i=0;i<4;i++)
			for(int j=i;j+1<5;j+=2)
				w+=edge[tab[j]-1][tab[j+1]-1]+edge[tab[j+1]-1][tab[j]-1];
		m=max(w,m);
		next_permutation(tab,tab+5);
	}
	printf("%d\n",m);
}
