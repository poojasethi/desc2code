#include<bits/stdc++.h>
using namespace std;
int main()
{
	int g[6][6],i,j,a[5],ans,m=INT_MIN;
	for(i=0;i<5;i++)
		a[i]=i+1;
	for(i=1;i<=5;i++)
	for(j=1;j<=5;j++)
		cin>>g[i][j];
	do
	{
		ans=g[a[0]][a[1]]+g[a[1]][a[0]]+2*g[a[2]][a[3]]+2*g[a[3]][a[2]]+g[a[1]][a[2]]+g[a[2]][a[1]]+2*g[a[3]][a[4]]+2*g[a[4]][a[3]];
		m=max(m,ans);
	}while(next_permutation(a,a+5));
	cout<<m<<endl;
	return 0;
}
