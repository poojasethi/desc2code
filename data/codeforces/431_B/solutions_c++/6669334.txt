#include <iostream>
#include <algorithm>

using namespace std;

int a[5][5], b[5];

int main() {
	for (int i = 0; i < 5; i++)
		b[i] = i;
	for (int i = 0; i < 5; i++)
		for (int j = 0; j < 5; j++)
			cin >> a[i][j];
	int ans = 0;
	do {
		int cur = 0;
		for (int i = 0; i < 5; i++)
			for (int j = i; j < 4; j += 2)
				cur += a[b[i]][b[i + 1]] + a[b[i + 1]][b[i]];
		ans = max(ans, cur);
	} while(next_permutation(b, b + 5));
	cout << ans << endl;
	return 0;
}
