#include<cstdio>
#include<algorithm>
using namespace std;

int ans, tot;
int v[11][11], p[11];

int c(int x, int y) {
	return v[p[x]][p[y]] + v[p[y]][p[x]];
}

int main() {
	for(int i=0;i<5;i++) {
		for(int j=0;j<5;j++) scanf("%d", &v[i][j]);
		p[i] = i;
	}
	
	ans = 0;
	do {
		tot = c(0, 1) + c(2, 3) + c(1, 2) + c(3, 4) + c(2, 3) + c(3, 4);
		ans = max(ans, tot);
	} while (next_permutation(p, p+5));
	printf("%d\n", ans);
	
	return 0;
}
