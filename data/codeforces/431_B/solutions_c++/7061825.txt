#include <iostream>
#include <algorithm>

using namespace std;

int a[5][5];
int p[5]={0,1,2,3,4};

int val(int i,int j)
{
	return a[p[i]][p[j]] + a[p[j]][p[i]];
}
int main()
{
	for(int i=0;i<5;i++)
        for(int j=0;j<5;j++) cin>>a[i][j];
	int best=0;
	do
	{
		best=max(best,val(0,1)+val(2,3)+val(1,2)+val(3,4)+val(2,3)+val(3, 4));
	}
	while(next_permutation(p,p+5));
	cout<<best;
}
