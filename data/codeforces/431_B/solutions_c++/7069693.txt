#include <cstdio>
#include <algorithm>
using namespace std;
int a[5][5], p[5] = {0, 1, 2 ,3 ,4}, Ans = 0, tmp = 0;

int main(){
	
	for (int i = 0; i < 5; i++)
		for (int j = 0; j < 5; j++)
			scanf("%d", &a[i][j]);
			
	do{
		tmp = 0;
		for (int i = 0; i < 4; i++)
			for (int j = i; j < 4; j += 2)
				tmp += a[p[j]][p[j + 1]] + a[p[j + 1]][p[j]];
		if (tmp > Ans)
			Ans = tmp;
	}while (next_permutation(p, p + 5));
	
	printf("%d", Ans);
	
	return 0;
}
