#include<bits/stdc++.h>
using namespace std;

int main()
{
    int h[5][5], q[5]={0,1,2,3,4}, toth, bestToth=-1;
    for(int i=0;i<5;i++) for(int j=0;j<5;j++) cin>>h[i][j];
    do{
        toth=0;
        for(int i=0;i<5;i++)
            for(int j=i;j<4;j+=2)
                toth+=(h[q[j]][q[j+1]]+h[q[j+1]][q[j]]);
        bestToth=((toth>bestToth)?toth:bestToth);
    }while(next_permutation(q,q+5));
    cout<<bestToth;
    return 0;
}
