{
  int a[5][5],ans=0,c[5]={0,1,2,3,4}; 
  FOR(i,0,5) FOR(j,0,5) cin>>a[i][j];   
  do
  { int s=0;
    FOR(i,0,4)
      FOR(j,i,4)
        s+=(a[c[j]][c[j+1]]+a[c[j+1]][c[j]]) , j++;
    ans=max(s,ans);    
  } while(next_permutation(c,c+5));
  cout<<ans;
}
