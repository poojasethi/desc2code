#include<iostream>
#include<algorithm>
using namespace std;
int main(void)
{
	int a[5][5],i,j,b[5]={0,1,2,3,4},max=0,sum;
	for(i=0;i<5;i++)
	{
		for(j=0;j<5;j++)
		{
			cin>>a[i][j];
		}
	}
	do
	{
		sum=a[b[0]][b[1]]+a[b[1]][b[0]]+2*a[b[2]][b[3]]+2*a[b[3]][b[2]]+a[b[1]][b[2]]+a[b[2]][b[1]]+2*a[b[3]][b[4]]+2*a[b[4]][b[3]];
		max=max>=sum?max:sum;
	}
	while(next_permutation(b,b+5));
	cout<<max<<endl;
	return 0;
}
