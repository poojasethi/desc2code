#include<iostream>
#include<algorithm>
using namespace std;
int v[5]={0,1,2,3,4};
int a[5][5];
int f(int i,int j)
{
	return a[v[i]][v[j]]+a[v[j]][v[i]];
}
int main()
{

	
	for(int i=0;i<5;i++)
		for(int j=0;j<5;j++)
			cin>>a[i][j];
	int ans=-1;
	do{
       ans=max(ans,f(0,1)+f(2,3)+f(1,2)+f(3,4)+f(2,3)+f(3,4));
	}while(next_permutation(v,v+5));
	cout<<ans<<endl;
	return 0;
}
