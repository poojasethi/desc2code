#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int g[5][5],l[5],mxn=0,t;
    for(int i=0;i<5;i++)
    {
        for(int j=0;j<5;j++)
            cin>>g[i][j];
        l[i]=i;
    }
    do
    {
        t=g[l[0]][l[1]]+g[l[1]][l[0]]+g[l[1]][l[2]]+g[l[2]][l[1]]+
        2*(g[l[2]][l[3]]+g[l[3]][l[2]]+g[l[3]][l[4]]+g[l[4]][l[3]]);
        if(t>mxn) mxn=t;
    }while(next_permutation(l,l+5));
    cout<<mxn<<endl;
}
