#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int a[5][5],ma=-1;

int rob(int b[]){
	int kolo=0;
	for(int i=0;i<5;i++){
		for(int j=i;j<4;j=j+2){
			kolo+=a[b[j]][b[j+1]]+a[b[j+1]][b[j]];
		}
	}
	return kolo;
}

int main(){
	for(int i=0;i<5;i++) for(int j=0;j<5;j++) cin>>a[i][j];
	int b[]={0,1,2,3,4};
	ma=max(ma,rob(b));
	while(next_permutation(b,b+5)) ma=max(ma,rob(b));
	cout<<ma<<endl;
	return 0;
}
