# include<iostream>
# include<algorithm>
using namespace std;
int main()
{
int j,k,g[6][6];
for(j=1;j<=5;j++)
for(k=1;k<=5;k++)
cin>>g[j][k];
int a[5]={1,2,3,4,5};
int t,ans=-1;
do
{
    t=0;
    for(int i=0;i<=4;i++)
    {
        for(j=i;j<=4;j+=2)
        {
            if(j+1<=4) t+=g[a[j]][a[j+1]]+g[a[j+1]][a[j]];
        }
    }
    if(ans<t) ans=t;
}
while(next_permutation(a,a+5));
cout<<ans<<endl;
return 0;
}
