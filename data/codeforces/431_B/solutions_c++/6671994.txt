#include <cstdio>
#include <algorithm>

using namespace std;

int main(){
	int h[5][5], i, j, s=0, max_s=0;

	for(i = 0;i<5;i++){
		scanf("%d %d %d %d %d", &h[i][0], &h[i][1], &h[i][2], &h[i][3], &h[i][4]);
	}

	int p[] = {0, 1, 2, 3, 4};
	sort(p, p+5);

	do{
		s = 0;
		for(i = 0;i<4;i++){
			for(j = i; j<4;j+=2){
				s+=h[p[j]][p[j+1]] + h[p[j+1]][p[j]];
			}
		}

		if(s > max_s)
			max_s = s;
	}while(next_permutation(p, p+5));

	printf("%d", max_s);

	return 0;
}
