#include<algorithm>
#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int a[6][6],ans=0,s=0,d[6]={0,1,2,3,4,5};
	for(int i=1;i<=5;i++)for(int j=1;j<=5;j++)cin>>a[i][j];
	for(int i=1;i<=5;i++)for(int j=1;j<i;j++)a[i][j]+=a[j][i],a[j][i]=a[i][j];
	while(next_permutation(d+1,d+6))
	{
		s=a[d[2]][d[3]]+a[d[1]][d[5]]*2+a[d[1]][d[3]]+a[d[4]][d[5]]*2;
		ans=max(ans,s);
	}
	cout<<ans<<endl;
	return 0;
}
