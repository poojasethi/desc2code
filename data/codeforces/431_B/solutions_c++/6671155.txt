#include<iostream>
#include<algorithm>

using namespace std;

int main(){
	int a[] = {0,1,2,3,4};
	int g[6][6];
	for(int i=0;i<5;i++)
		for(int j=0;j<5;j++)
			cin >> g[i][j];
			
	long long int h=0,hmax=h;
	do{
		h = g[a[0]][a[1]] + g[a[1]][a[0]] + 2*g[a[2]][a[3]]+2*g[a[3]][a[2]] + g[a[1]][a[2]] + g[a[2]][a[1]] + 2*g[a[3]][a[4]] + 2*g[a[4]][a[3]];
		
		if (h > hmax)
			hmax = h;
	}while(next_permutation(a,a+5));
	cout << hmax <<endl;
	return 0;
}
