#include <iostream>
#include <algorithm>

using namespace std;

int g[6][6],a[6],ans=0;
int main()
{
	for (int i=1;i<=5;i++) for (int j=1;j<=5;j++) cin>>g[i][j];
	for (int i=1;i<=5;i++) a[i]=i;
	do
	{
		int sum=0;
		for (int i=1;i<=4;i++)
		{
			for (int j=i;j<=5;j+=2)
			{
				if (j>=5) break;
				sum+=g[a[j]][a[j+1]]+g[a[j+1]][a[j]];
			}
		}
		ans=max(sum,ans);
	}while (next_permutation(a+1,a+6));
	cout<<ans<<endl;
	return 0;
}
