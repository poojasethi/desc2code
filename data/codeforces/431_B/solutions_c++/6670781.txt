#include <cstdio>
#include <algorithm>

int s,t,a[5],g[5][5];

int sum(int a,int b)
{
	return g[a][b]+g[b][a];
}
int main()
{
	for (int i=0;i<5;++i) for (int j=0;j<5;++j) scanf("%d",g[i]+j);
	for (int i=0;i<5;++i) a[i]=i;
	s=sum(a[0],a[1])+2*sum(a[2],a[3])+sum(a[1],a[2])+2*sum(a[3],a[4]);
	for (int r=1;r<120;++r)
	{
		std::next_permutation(a,a+5);
		s=std::max(s,sum(a[0],a[1])+2*sum(a[2],a[3])+sum(a[1],a[2])+2*sum(a[3],a[4]));
	}
	printf("%d",s);
	return 0;
}
