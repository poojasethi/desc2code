#include <bits/stdc++.h>
using namespace std;
int a[5][5],p[]={0,1,2,3,4},res=0;
int main() {
    // your code goes here
    for (int i=0;i<5;i++)
        for (int j=0;j<5;j++)
            cin>>a[i][j];
    do
    {
        int k=0;
        for(int i=0;i<5;i++)
            for(int j=i;j+1<5;j+=2)
                k+=a[p[j]][p[j+1]]+a[p[j+1]][p[j]];
        res=max(res,k)  ;
    }while(next_permutation(p,p+5));
    cout<<res;
    return 0;
}
