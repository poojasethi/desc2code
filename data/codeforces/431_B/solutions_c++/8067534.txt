#include<bits/stdc++.h>
using namespace std;
int main()
{	
	int a[5][5],i,j;
	for(i=0;i<5;i++)
		for(j=0;j<5;j++)
			scanf("%d",&a[i][j]);
	int p[5],ans=-1,temp;
	for(i=0;i<5;i++)
		p[i]=i;
	do
	{
		temp=0;
		temp=2*(a[p[2]][p[3]]+a[p[3]][p[2]])+2*(a[p[3]][p[4]]+a[p[4]][p[3]])+a[p[1]][p[2]]+a[p[2]][p[1]]+a[p[0]][p[1]]+a[p[1]][p[0]];
		if(temp>ans)
			ans=temp;
	}while(next_permutation(p,p+5));
	printf("%d\n",ans);
return 0;
}
