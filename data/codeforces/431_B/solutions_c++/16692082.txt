#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main()
{
    ll a[]={1,2,3,4,5},ans=INT_MIN,sum,b[6][6],i,j;
    for(i=1;i<=5;i++) for(j=1;j<=5;j++) cin>>b[i][j];
    do
    {
        sum=b[a[0]][a[1]]+b[a[1]][a[0]]+( 2* (b[a[2]][a[3]]+b[a[3]][a[2]] +b[a[3]][a[4]]+b[a[4]][a[3]]) ) + b[a[1]][a[2]]+b[a[2]][a[1]];
        if(sum>ans) ans=sum;
    }while(next_permutation(a,a+5));
    cout<<ans;
    return 0;
}
