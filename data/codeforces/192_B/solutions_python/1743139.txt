n = int(raw_input())
p = map(int,raw_input().split())
u = [False]*n
c = [False]*n
i,j = 0,10000
ans = 0
while i<j:
    t = (i+j)>>1
    for k in range(n): c[k] = (p[k]>=t)
    u[n-1] = c[n-1]
    if n>1: u[n-2] = c[n-2] and u[n-1]
    for k in range(n-3,-1,-1): u[k] = c[k] and (u[k+1] or u[k+2])
    if u[0]: ans,i = t,t+1
    else: j = t
    
print ans
