N = int(raw_input())
a = map(int, raw_input().split())
ret = 0
for d in xrange(0, min(a[0], a[N - 1]) + 1):
	i = 0
	result = True
	while i != N - 1:
		if a[i + 1] >= d:
			i += 1
		elif i + 2 < N and a[i + 2] >= d:
			i += 2
		else:
			result = False
			break
	if result and i == N - 1:
		ret = d
print ret
