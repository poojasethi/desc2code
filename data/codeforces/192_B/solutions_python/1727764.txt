def work():
    n = int(raw_input())
    a = map(int, raw_input().split())
    if n == 1:
        return a[0]
    if n == 2:
        return min(a[0], a[1])
    dp = [0] * (n + 1)
    dp[0] = a[0]
    dp[1] = min(a[1], a[0])
    for i in xrange(2, n):
        dp[i] = max(dp[i - 2], dp[i - 1])
        dp[i] = min(dp[i], a[i])
    return dp[n - 1]


print work()
