n = int(raw_input())
number = list(map(int , raw_input().split()))
if n == 1:
    print number[0]
else:
    dp = [number[0] , min(number[0] , number[1])]
    for i in range(2 , n):
       a = min(dp[i - 2] , number[i])
       b = min(dp[i - 1] , number[i])
       dp.append(max(a , b))
    print dp[n - 1]
