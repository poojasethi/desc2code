#coding=utf-8

def main():
    r = raw_input()
    n  = int(r)
    a = []
    s = raw_input().split()

    for ss in s:
        a.append(int(ss))

    minn = min(a[0], a[len(a)-1])

    for i in range(len(a)-1):
        m = max(a[i],a[i+1])
        if m < minn:
            minn = m

    print minn

if __name__ == "__main__":
    main()
