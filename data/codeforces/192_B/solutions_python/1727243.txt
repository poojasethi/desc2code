import sys
from math import sqrt, ceil
input = sys.stdin
#input = open('test1.in')

def solve(A):
  for d in xrange(0, 1010):
    if A[0] <= d:
      return d
    if A[len(A)-1] <= d:
      return d
    for i in xrange(1, len(A)-2):
      if A[i] <= d and A[i+1] <= d:
        return d

input.readline()
print solve(map(int, input.readline().split()))
