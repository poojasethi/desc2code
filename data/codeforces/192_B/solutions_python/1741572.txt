import sys


def solve(input_data):
    data = [input_data[0], input_data[-1]]
    for i in range(len(input_data) - 1):
      data.append(max(input_data[i], input_data[i + 1]))
    return min(data)  
    
if __name__ == '__main__':
    sys.stdin.readline()
    data = [int(x) for x in sys.stdin.readline().split()]
    sys.stdout.write(str(solve(data)))
