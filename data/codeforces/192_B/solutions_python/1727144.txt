n = int(raw_input())
a = map(int, raw_input().split(' '))
dead = set([])
m = max(a) + 5
stop = False
for i in xrange(1, m):
    for j in xrange(n):
        if a[j] == i - 1:
            if j == 0 or j == n - 1 or j - 1 in dead or j + 1 in dead:
                print i - 1
                stop = True
                break
            else:
                dead.add(j)
    if stop:
        break
