INF = 1005

def solve(n, L):
    answer = INF
    for i in range(n - 1):
        answer = min(answer, max(L[i], L[i + 1]))
    answer = min(answer, L[0])
    answer = min(answer, L[n - 1])
    return answer

if __name__ == '__main__':
    n = int(raw_input().strip())
    L = map(int, raw_input().strip().split())
    print '%g' %solve(n, L)
