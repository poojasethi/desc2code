n = int(raw_input())
nums = [int(x) for x in raw_input().split()]
day = 0
while True:
    if nums[0] <= 0:
        break
    if nums[-1] <= 0:
        break
    broke_idx = [i for i, x in enumerate(nums) if x <= 0]
    diff = [broke_idx[i] - broke_idx[i - 1] for i, x in enumerate(broke_idx) if i != 0]
    if diff.count(1) > 0:
        break
    day += 1
    nums = [x - 1 for x in nums]

print day
