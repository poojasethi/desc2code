n=int(raw_input())
#n,k=map(int,raw_input().split())
l=[int(x) for x in raw_input().split()]
ans=f=0
chk=max(l)+1
while max(l)<chk or f==0:
    k=l.count(min(l))
    ans=min(l)
    for i in range(k):
        b=l.index(min(l))
        l[b]=chk
        if b==0 or b==n-1:
            f=1
        else:
            if l[b-1]==chk or l[b+1]==chk:
                f=1
print ans
