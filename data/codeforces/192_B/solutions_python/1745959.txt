#! /usr/bin/python
import sys
import math

def read_int():
        tmp = raw_input().split()
        res = []
        for i in tmp:
                res.append(int(i))
        return res

n = int(raw_input())
a = read_int()
ans = min(a[0], a[-1])
for i in range(n - 1):
        ans = min(ans, max(a[i], a[i + 1]))
print ans
