#http://codeforces.com/problemset/problem/192/B

def solve():
    a=int(raw_input())
    ls=raw_input()
    ls=ls.split()
    ls = [int(el) for el in ls]
    m=ls[0]
    for i in range(a-1):
        m=min(m,max(ls[i],ls[i+1]))
    m=min(ls[0],m)
    m=min(ls[a-1],m)
    print m

if __name__=='__main__':
    solve()
