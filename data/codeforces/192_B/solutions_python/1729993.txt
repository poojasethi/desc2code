import sys

def main():
    _ = int(sys.stdin.readline())
    data = [int(x) for x in sys.stdin.readline().split()]
    
    L = [data[0], data[-1]]
    for i in xrange(len(data) - 1):
        L.append(max(data[i], data[i+1]))
    sys.stdout.write(str(min(L)))
    
if __name__ == "__main__":
    main()
