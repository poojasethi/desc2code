#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int n;
	int a[1000];
	int d[1000];

	cin >> n;
	for (int i = 0; i < n; ++i)
		cin >> a[i];

	d[0] = a[0];
	d[1] = min(d[0], a[1]);
	for (int i = 2; i < n; ++i)
		d[i] = min(max(d[i - 2], d[i - 1]), a[i]);

	cout << d[n - 1];
}
