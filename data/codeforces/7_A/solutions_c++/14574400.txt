#include<bits/stdc++.h>
using namespace std;
int main()
{
	int ans=0,tmp=0;
	for(int i=0;i<8;i++)
	{
		string s;
		cin>>s;
		int cnt=0;
		for(int j=0;j<8;j++)
		{
			if(s[j]=='B')
			{
				cnt++;
			}
		}		
		if(cnt==8)
			ans++;
		else
			tmp=cnt;
	}	
	if(ans<8)
		ans+=tmp;
	cout<<ans;
	return 0;
}
