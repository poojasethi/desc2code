#include<cstdio>
#include<cstring>
using namespace std;
int num,k;
char map[9][9],p[9]="BBBBBBBB";
main()
{
    for(int i = 0;i < 8;++i) {
        gets(map[i]);
        if(!strcmp(p,map[i]))num++;
        else if(!k) {
            for(int j = 0;j < 8;++j)if(map[i][j] == 'B')num++;
            k++;
        }
    }
    printf("%d\n",num);
}
