#include<stdio.h>

int main(void)
{
	int x = 0, y = 0, num;
	int i, j;
	char str[10];
	for (i = 0; i < 8; ++i)
	{

		scanf("%s", str);
		for (j = 0, num = 0; j < 8; ++j)
		{
			if (str[j] == 'B')
				num++;
		}
		if (num == 8)
			x++;
		else
			y = num;
	}
	printf("%d\n",x+y);
	return 0;
	
}
