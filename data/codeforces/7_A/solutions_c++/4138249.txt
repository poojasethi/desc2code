#include<iostream>
using namespace std;
int main()
{
    char a[9][9];int sum=0,mm=0;
    for(int i=0;i<8;i++)
    {
        cin>>a[i];
        int t=0;
        for(int j=0;j<8;j++) if(a[i][j]=='B')t++;
        if(t==8) sum++;
        else mm=t;
    }
    cout<<sum+mm<<"\n";
}
