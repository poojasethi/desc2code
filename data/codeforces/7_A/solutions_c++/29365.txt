#include<cstdio>
char maze[77][77];
int main(){
	int i;
	for(i=0;i<8;++i)scanf("%s",maze[i]);
	int j,cnt=0;
	for(i=0;i<8;++i){
		for(j=0;j<8;++j)
			if(maze[i][j]-'B')break;
		cnt+=j==8;
	}
	if(cnt==8){puts("8");return 0;}
	for(j=0;j<8;++j){
		for(i=0;i<8;++i)
			if(maze[i][j]-'B')break;
		cnt+=i==8;
	}
	printf("%d",cnt);
	return 0;
}
