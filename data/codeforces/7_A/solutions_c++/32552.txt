#include <iostream>

using namespace std;

int main() {

    int a=0,vert=1;
    char c;

    for (int i=0; i<8; i++) {
        int b=0;
        for (int i=0; i<8; i++) {
            cin >> c;
            if (c == 'B') b++;
        }
        if (b == 8) a++;
        else if (vert) a+=b, vert=0;
    }

    cout << a;
}
