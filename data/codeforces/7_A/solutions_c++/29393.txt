#include<iostream>
using namespace std;
int main()
{
int a[10],b[10],i,j,k=0,l=0;
char c;
for(i=1;i<=8;i++)
{
a[i]=0;
b[i]=0;
}
for(i=1;i<=8;i++)
{
for(j=1;j<=8;j++)
{
cin>>c;
if(c=='B')
{
a[i]++;
if(a[i]==8)
{
k++;
}
b[j]++;
if(b[j]==8)
{
l++;
}
}
}
}
if(k==8 || l==8)
{
cout<<8;
}
else
{
cout<<k+l;
}
return 0;
}
