#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char s[10];
int main()
{
    int a,b,c;
    for(int i=a=c=0;i<8;++i)
    {
        scanf("%s",s);
        for(int j=b=0;j<8;++j)b+=(s[j]=='B');
        if(b==8)++a;else c=b;
    }
    if(a<8)a+=c;
    printf("%d\n",a);
}
