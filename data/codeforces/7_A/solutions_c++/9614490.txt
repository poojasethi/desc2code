#include <iostream>
using namespace std;
char c[10][10];
int main()
{
	int ans=0 , t=0 ,tn=0;
	for(int i=0;i<8;i++)
		cin >> c[i];
	for(int i=0;i<8;i++)
	{
		t=0;
		for(int j=0;j<8;j++)
			if(c[i][j]=='B')
				t++;
		if(t==8)
			ans++;
		else
			tn=t;
	}
	if(ans==8)
		cout << ans << endl;
	else
		cout << ans+tn << endl;
}
