#include<iostream>
using namespace std;
#include<string.h>
int main()
{
 string s[8];
 int i,j,k=0,c=0,r=0;
 for(i=0;i<8;i++)
 {
     k=0;
  cin>>s[i];
  for(j=0;j<8;j++)
  {
   if(s[i][j]=='B')
   k++;
  }
 if(k==8)
  c=c+1;
  else
  r=k;
 }
 cout<<(c+r)<<endl;
 return 0;
}
