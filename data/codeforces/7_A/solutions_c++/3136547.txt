#include <iostream>
#include <string>
using namespace std;

int main(void)
{
	int column=0, answer=0;

	for (int i=0; i<8; ++i)	{
		int row=0;
		string board;
		cin >> board;
		for (int j=0; j<8; ++j)
			row+=(board[j]=='B');
		(row==8)?	++answer : column=row;
	}
	if (column < 8)	answer+=column;
	cout << answer << endl;
	return 0;
}
