#include <stdio.h>
int r[9],c[9];
int main()
{
	int i,j,k;
	char ch;
	for(i=1;i<=8;i++)
	{
		for(j=1;j<=8;j++)
		{
			scanf(" %c",&ch);
			if(ch=='B')
			{
				r[i]++;
				c[j]++;
			}
		}
	}
	for(k=0,i=1;i<=8;i++)
	{
		if(r[i]==8) k++;
		if(c[i]==8) k++;
	}
	if(k==16) k=8;
	printf("%d",k);
	return 0;
}
