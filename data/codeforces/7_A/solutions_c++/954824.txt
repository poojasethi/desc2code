#include<iostream>
using namespace std;
int x[8],y[8];
int main(){
	string a[8];
	for(int i=0;i<8;cin>>a[i],i++);
	for(int i=0;i<8;i++)
		for(int j=0;j<8;j++)
			if(a[i][j]=='B'){
				x[i]++;
				y[j]++;
			}
	int ans=0;
	for(int i=0;i<8;i++){
		if(x[i]==8)
			ans++;
		if(y[i]==8)
			ans++;
	}
	cout<<(ans==16?8:ans)<<endl;
}
