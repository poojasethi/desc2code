#include<bits/stdc++.h>
using namespace std;
int main(){

char arr[8][8];

for(int I=0 ; I<8 ; I++){
for(int j=0 ; j<8 ; j++){
cin >> arr[I][j];
}
}
int rows=0,col = 0;

for(int I=0 ; I<8 ; I++){
int line = 0;
for(int j=0 ; j<8 ; j++){
if(arr[I][j]=='B'){
line++;
}
}
if(line == 8)
rows++;
else col = line;
}

cout << rows+col;
}
