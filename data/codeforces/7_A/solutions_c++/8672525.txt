#include <iostream>
#include <cstdio>
using namespace std;
char s[20][20];
int main()
{
	int cnt=0,sum=0;
	for(int i=0;i<8;i++)
	{
		gets(s[i]);
		int t=0;
		for(int j=0;j<8;j++)
			if(s[i][j]=='B')
				t++;
		if(t==8) cnt++;
		else 
			sum=t;
	}
	cnt+=sum;
	cout<<cnt<<endl;
	return 0;
}
