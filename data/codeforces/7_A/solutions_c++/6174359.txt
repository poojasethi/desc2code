#include<iostream>
using namespace std;
int main()
{
	char a[8][8];
	int p=0,h=0;
	int i,j;
	for(i=0;i<8;i++)
		for(j=0;j<8;j++)
			cin>>a[i][j];
	for(i=0;i<8;i++)
	{	int n=0;
		for(j=0;j<8;j++)
			if(a[i][j]=='B')
				n++;
		if(n==8) p++;
		else h=h>n?h:n;
	}
	cout<<h+p<<endl;
}
