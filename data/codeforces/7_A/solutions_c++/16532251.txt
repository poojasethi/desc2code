#include <iostream>
#include <string>	
using namespace std;

int main(){
	string s;
	int answer = 0, cnt, f = 0;
	for (int i = 0; i < 8; i++){
		cin >> s;
		cnt = 0;
		for (int j = 0; j < s.size(); j++)
			if (s[j] == 'B') cnt++;
		if (cnt == 8) answer++;
		else if (!f) {
			answer += cnt;
			f = 1;
		}
	}
	cout << answer;
}
