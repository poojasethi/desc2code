#include<iostream>
using namespace std;
int main()
{
char line[8][9];
int i,j,count=0;
for(i=0; i<8; i++)
cin>>line[i];
for(i=0; i<8; i++)
{
for(j=0; line[i][j]=='B' && j<8; j++);
if(j==8)
count++;
}
if(count<8)
for(i=0; i<8; i++)
{
for(j=0; line[j][i]=='B' && j<8; j++);
if(j==8)
count++;
}
cout<<count;
return 0;
}
