#include<iostream>
using namespace std;
int main()
{

    string s;int i,j,a=0,r=0,c=0,k=-1;
    for(i=0;i<8;i++)
    {
        cin>>s;a=0;
        for(j=0;j<8&&k==-1;j++)
        if(s[j]=='B')
        a++;
        if(a<8)
        c=a;
        else
        r++;
    }
    cout<<(c+r);
    return 0;
}
