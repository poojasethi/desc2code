#include<iostream>
#include<string>
using namespace std;
int main()
{
	string a[9];
	for(int i=0;i<8;i++) {
		cin>>a[i];
	}
	int s=0,f=0;
	for(int i=0;i<8;i++) {
		int temp=0;
		for(int j=0;j<8;j++) {
			temp+=(a[i][j]=='B');
		}
		if(temp==8)
			s++;
			else if(f==0&&temp<8){
				s=s+temp;
				f=1;
			}
		
	}
	cout<<s;
}
