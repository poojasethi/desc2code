
from sys import stdin

board = [list(stdin.readline().strip()) for i in range(8)]

rc = 0
for row in board:
	if all([i=="B" for i in row]):
		rc += 1

cc = 0
for col in range(len(board[0])):
	if all([row[col]=="B" for row in board]):
		cc += 1

total = cc + rc
if total == 16:
	total = 8

print total
