#include <bits/stdc++.h>

using namespace std;

typedef long long LL;

const int N = 100000;

int n,d[64],cnt,p[N+1];
bitset<N+1> a[64];
LL x[N+1],res,r[2];

int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++) scanf("%I64d",&x[i]),res^=x[i];
	for(int i=62;i>=0;i--) if(~(res>>i)&1){
		for(int j=0;j<n;j++) if((x[j]>>i)&1) a[cnt][j]=1;
		cnt++;
	}
	for(int i=62;i>=0;i--) if((res>>i)&1){
		for(int j=0;j<n;j++) if((x[j]>>i)&1) a[cnt][j]=1;
		cnt++;
	}
	for(int i=0;i<cnt;i++){
		a[i][n]=1;
		for(int j=0;j<i;j++) if(d[j]<n && a[i][d[j]]) a[i]^=a[j];
		for(;d[i]<n && !a[i][d[i]];d[i]++);
		if(d[i]<n){
			for(int j=0;j<i;j++) if(a[j][d[i]]) a[j]^=a[i];
		}
	}
	for(int i=0;i<n;i++) p[i]=1;
	for(int i=0;i<cnt;i++) if(d[i]<n) p[d[i]]=a[i][n] ? 2 : 1;
//	for(int i=0;i<n;i++) r[p[i]-1]^=x[i];printf("res = %lld\n",r[0]+r[1]);
	for(int i=0;i<n;i++) printf("%d ",p[i]);puts("");
	return 0;
}
