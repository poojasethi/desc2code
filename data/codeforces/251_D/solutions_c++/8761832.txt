#include<cstdio>
#include<bitset>
#define LL long long
using namespace std;
bitset<64> b[100005],f;
int r[100005],w[64];
LL a[100005],S,T;
int use[100005];
int n,t;
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%I64d",&a[i]),
		r[i]=i,T^=a[i];
	for(int q=0;q<2;q++)
		for(int i=60;i>=0;i--)
			if(((T>>i)&1)==q)
			{
				for(int j=t;j<n;j++)
					if((a[r[j]]>>i)&1)
					{
						swap(r[t],r[j]);
						goto succ;
					}
				continue;
				succ:
				b[r[t]][t]=1;
				for(int j=t+1;j<n;j++)
					if((a[r[j]]>>i)&1)
					{
						a[r[j]]^=a[r[t]];
						b[r[j]]=b[r[j]]^b[r[t]];
					}
				w[t]=i;
				t++;
			}
	for(int i=0;i<t;i++)
		if((~S>>w[i])&1)
			S^=a[r[i]],
			f=f^b[r[i]];
	for(int i=0;i<t;i++)
		use[r[i]]=f[i];
	for(int i=0;i<n;i++)
		printf("%d ",use[i]+1);
	puts("");
}
