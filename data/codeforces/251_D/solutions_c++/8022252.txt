#include <bits/stdc++.h>
using namespace std;
#define rep(i,a,n) for (int i=a;i<n;i++)
#define per(i,a,n) for (int i=n-1;i>=a;i--)
#define pb push_back
#define mp make_pair
#define all(x) (x).begin(),(x).end()
#define SZ(x) ((int)(x).size())
#define fi first
#define se second
typedef vector<int> VI;
typedef long long ll;
typedef pair<int,int> PII;
const ll mod=1000000007;
ll powmod(ll a,ll b) {ll res=1;a%=mod;for(;b;b>>=1){if(b&1)res=res*a%mod;a=a*a%mod;}return res;}
// head

const int N=101000,T=60;
int n,bits,psb[70],pos[70],st[N];
ll ret,a[N],s[70],base[70],bs,cd;
int main() {
	scanf("%d",&n);
	rep(i,0,n) scanf("%I64d",a+i),ret^=a[i];
	bits=T-1;
	per(i,0,T) if ((ret&(1ll<<i))==0) cd|=(1ll<<bits),psb[bits--]=i;
	per(i,0,T) if (ret&(1ll<<i)) psb[bits--]=i;
	rep(i,0,n) {
		ll x=0;
		per(j,0,T) x|=((a[i]>>psb[j])&1)<<j;
		bs=0;
		per(k,0,T) {
			if ((x^base[k])<x) {
				x=x^base[k]; bs^=s[k];
			}
			if (x>=(1ll<<k)) {
				base[k]=x; pos[k]=i; s[k]=bs^(1ll<<k);
				break;
			}
		}
	}
	bs=0;
	per(k,0,T) if ((cd^base[k])<cd) cd^=base[k],bs^=s[k];
	rep(i,0,n) st[i]=2;
	rep(i,0,T) if (bs&(1ll<<i)) st[pos[i]]=1;
	rep(i,0,n) printf("%d ",st[i]); puts("");
}
