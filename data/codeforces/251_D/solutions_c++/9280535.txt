#include <cstdio>
#include <bitset>
std::bitset<100001>a[64],b[64];
int ans[100000 + 9],n,t[100];
long long sum;
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("251D.in","r",stdin);
	freopen("251D.out","w",stdout);
	#endif
	scanf("%d",&n);
	for (int i = 0; i < n; ++i) {
		long long x;
		scanf("%I64d",&x);
		sum ^= x;
		for (int j = 0; j < 63; ++j,x /= 2)
			b[j][i] = (x & 1);
	}
	int m = 0;
	for (int i = 0; i < 63; ++i)
		if (sum >> i & 1) a[m++] = b[i];
	for (int i = 0; i < 63; ++i)
		if ((sum >> i & 1) == 0) a[m++] = b[i];
	m = 63;
	for (int j,i = 62; i >= 0; --i) {
		a[i][n] = 1;
		for (j = 62; j >= m; --j)
			if (a[i][t[j]]) a[i] ^= a[j];
		for (j = 0; j < n && !a[i][j]; ++j);
		if (j < n) {
			for (int k = 62; k >= m; --k)
				if (a[k][j]) a[k] ^= a[i];
			t[--m] = j;
			a[m] = a[i];
		}
	}
	for (; m < 63; ++m) ans[t[m]] = a[m][n];
	for (int i = 0; i < n; ++i) printf("%d ",ans[i] + 1);
}
