#include<cstdio>
#include<cstring>
#include<bitset>
#define LL long long
using namespace std;
bitset<100050> A[65];
LL a[100050];
int ans[100050];
LL b[65];
LL X;
int n,tot;
int main()
{
	int i,j,k;
	scanf("%d",&n);
	X=0;
	for (i=1;i<=n;i++)
	{
		scanf("%I64d",&a[i]);
		X^=a[i];
	}
	for (i=0;i<=1;i++)
		for (j=59;j>=0;j--)
			if ((X&(1ll<<j))>>j==i)
			{
				tot++; A[tot][n+1]=1; b[tot]=0;
				for (k=1;k<=n;k++) A[tot][k]=(a[k]&(1ll<<j))>>j;
				for (k=1;k<=tot-1;k++)
					if (A[tot][b[k]]) A[tot]^=A[k];
				for (k=1;k<=n;k++)
					if (A[tot][k])
					{
						b[tot]=k; break;
					}
				if (b[tot])
				{
					for (k=1;k<=tot-1;k++)
						if (A[k][b[tot]]) A[k]^=A[tot];
				}
				else tot--;
			}
	memset(ans,0,sizeof(ans));
	for (i=1;i<=tot;i++) ans[b[i]]=A[i][n+1];
	for (i=1;i<=n;i++) printf("%d ",ans[i]+1);
	printf("\n"); return 0;
}
