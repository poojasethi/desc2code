#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
int n;
#define MAXN 100010
long long X[MAXN],o[MAXN];
#define MAXV 70
long long P[MAXV],c[MAXV],q[MAXV],b[MAXV];
long long ans;
long long res[MAXV];
int main(){
    scanf("%d",&n);for(int i=0;i<=62;i++)P[i]=-1;
    for(int i=1;i<=n;i++){
        scanf("%I64d",&X[i]);ans^=X[i];
    }
    for(int i=0;i<=62;i++)res[i]=ans>>i&1;
    for(int i=1;i<=n;i++){
        long long x=X[i],y=0;
        for(int j=62;j>=0;j--)if((!res[j])&&(x>>j&1))
        if(P[j]<0){P[j]=i,c[j]=x,q[j]=y,x=-1;break;}
            else x^=c[j],y|=1LL<<j;
            
        if(x<0)continue;
        
        for(int j=62;j>=0;j--)if((res[j])&&(x>>j&1))
        if(P[j]<0){P[j]=i,c[j]=x,q[j]=y,x=-1;break;}
            else x^=c[j],y|=1LL<<j;
    }long long x=0;
    
    for(int i=62;i>=0;i--)
    if((!res[i])&&!(x>>i&1)&&P[i]>=0)x^=c[i],b[i]^=1;
    for(int i=62;i>=0;i--)
    if((res[i])&&!(x>>i&1)&&P[i]>=0)x^=c[i],b[i]^=1;
    
    for(int i=0;i<=62;i++)if(b[i]&&res[i])
        for(int j=0;j<=62;j++)if(q[i]>>j&1)b[j]^=1;
    for(int i=0;i<=62;i++)if(b[i]&&!res[i])
        for(int j=0;j<=62;j++)if(q[i]>>j&1)b[j]^=1;
    for(int i=0;i<=62;i++)if(b[i])o[P[i]]=1;
    for(int i=1;i<=n;i++)printf("%d ",o[i]+1);
}
