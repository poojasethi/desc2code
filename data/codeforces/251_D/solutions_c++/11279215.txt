#include <cstdio>
typedef long long ll;
const int N=(int)1e5+5,B=61;
int n,ans[N];
ll a[N],key[B],ch[B],be;
int loc[B];
int main(){
	scanf("%d",&n);
	ll sum=0;
	for(int i=1;i<=n;i++) scanf("%I64d",&a[i]),sum^=a[i];
	for(int i=1;i<=n;i++){
		ll x=a[i],y=0;
		for(int j=B-1;j>=0;j--) if((sum>>j&1)==0&&(x>>j&1)){
			if(!key[j]){
				key[j]=x;
				loc[j]=i;
				ch[j]=y;
			}
			x^=key[j];
			y|=1LL<<j;
		}
		for(int j=B-1;j>=0;j--) if((sum>>j&1)==1&&(x>>j&1)){
			if(!key[j]){
				key[j]=x;
				loc[j]=i;
				ch[j]=y;
			}
			x^=key[j];
			y|=1LL<<j;
		}
	}
	ll tmp=sum;
	for(int j=B-1;j>=0;j--) if((sum>>j&1)==0&&(tmp>>j&1)==0&&key[j]){
		tmp^=key[j];
		be^=1LL<<j;
	}
	for(int j=B-1;j>=0;j--) if((sum>>j&1)==1&&(tmp>>j&1)==0&&key[j]){
		tmp^=key[j];
		be^=1LL<<j;
	}
	for(int i=0;i<B;i++) if((sum>>i&1)==1&&be>>i&1)
		be^=ch[i];
	for(int i=0;i<B;i++) if((sum>>i&1)==0&&be>>i&1)
		be^=ch[i];
	for(int i=0;i<B;i++) if(be>>i&1)
		ans[loc[i]]^=1;
	for(int i=1;i<=n;i++) printf("%d ",2-ans[i]);
}
