using namespace std;
#define N 110000
#define ll unsigned long long
int ans[N];
ll n,id[65],b[65],x,s,h[65],c[65],a[N];
int get(ll x,int y){
    return (x>>y)&1;
}
void debug(ll x){
   //  cout<<x<<endl;
     int d[100];
     d[0]=0;

int main(){
    cin>>n;
    //for(int i=1;i<=n;i++)cin>>w[i];
    for(int i=1;i<=n;i++){
      cin>>a[i];
      s^=a[i];
      ans[i]=1;
      }
    for(int i=60;i>=0;i--)
      if(get(s,i)==0)b[++b[0]]=i;
    for(int i=60;i>=0;i--)
      if(get(s,i)==1)b[++b[0]]=i;
    for(int i=1;i<=n;i++){
      ll x=a[i],t=0;
      for(int j=1;j<=61;j++)
        if(get(x,b[j])){
          if(id[j])x^=h[j],t^=c[j];
          else{
            h[j]=x;
            t|=(ll)1<<j;
            c[j]=t;
            id[j]=i;
            break;
          }
        }
    }
    ll x=0,t=0;
    for(int i=1;i<=61;i++)
      if(id[i] && get(x,b[i])==0)x^=h[i],t^=c[i];
    for(int i=1;i<=61;i++)

    //cout<<x+(s^x)<<endl;

    return 0;
}
