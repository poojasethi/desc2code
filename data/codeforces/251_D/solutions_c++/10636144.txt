#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

#define maxn 100010
#define maxl 60
typedef long long ll;
ll a[maxn];
bool g[maxl][maxn],res[maxn];
int id[maxl];
int n,m;
ll X;

int main()
{
	ios::sync_with_stdio(false);
	cin>>n;
	X=0;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		X^=a[i];
	}
	for(int i=maxl;i>=0;i--)
	{
		if (!((X>>i)&1))
		{
			for(int j=0;j<n;j++)
				g[m][j]=(a[j]>>i)&1;
			g[m][n]=true;
			for(int k=0;k<m;k++)
			{
				if (g[m][id[k]])
				{
					for(int j=0;j<=n;j++)
						g[m][j]^=g[k][j];
				}
			}
			id[m]=-1;
			for(int j=0;j<n;j++)
			{
				if (g[m][j])
				{
					id[m]=j;
					break;
				}
			}
			if (id[m]!=-1) m++;
		}
	}
	for(int i=maxl;i>=0;i--)
	{
		if ((X>>i)&1)
		{
			for(int j=0;j<n;j++)
				g[m][j]=(a[j]>>i)&1;
			g[m][n]=false;
			for(int k=0;k<m;k++)
			{
				if (g[m][id[k]])
				{
					for(int j=0;j<=n;j++)
						g[m][j]^=g[k][j];
				}
			}
			id[m]=-1;
			for(int j=0;j<n;j++)
			{
				if (g[m][j])
				{
					id[m]=j;
					break;
				}
			}
			if (id[m]!=-1) m++;
		}
	}
	for(int i=m-1;i>=0;i--)
	{
		res[id[i]]=g[i][n];
		for(int j=0;j<n;j++)
		{
			if (j!=id[i] && g[i][j])
				res[id[i]]^=res[j];
		}
	}
	for(int i=0;i<n;i++)
		cout<<(res[i]?1:2)<<' ';
	cout<<endl;
	return 0;
}
