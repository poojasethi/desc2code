#include <cstdio>
#include <cstring>
#include <bitset>

using namespace std;

typedef long long LL;

int n;
LL a[100010],res,ans;
bitset<100001> g[64];
int pos[100010],d[64];

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%I64d",&a[i]),res^=a[i];
	int cnt=0;
	for(int i=62;i>=0;i--)
		if(!(res&(1LL<<i))){
			for(int j=1;j<=n;j++)
				if(a[j]&(1LL<<i)) g[cnt][j-1]=1;
			cnt++;
		}
	for(int i=62;i>=0;i--)
		if(res&(1LL<<i)){
			for(int j=1;j<=n;j++)
				if(a[j]&(1LL<<i)) g[cnt][j-1]=1;
			cnt++;
		}
	for(int i=0;i<cnt;i++){
		g[i][n]=1;
		for(int j=0;j<i;j++) if(d[j]!=n && g[i][d[j]]) g[i]^=g[j];
		for(d[i]=0;d[i]<n && !g[i][d[i]];d[i]++);
		if(d[i]!=n){
			for(int j=0;j<i;j++) if(g[j][d[i]]) g[j]^=g[i];
		}
	}
	for(int i=1;i<=n;i++) pos[i]=1;
	for(int i=0;i<cnt;i++) if(d[i]!=n) pos[d[i]+1]= g[i][n] ? 2 : 1;
	for(int i=1;i<=n;i++) printf("%d ",pos[i]);puts("");
	return 0;
		
}
