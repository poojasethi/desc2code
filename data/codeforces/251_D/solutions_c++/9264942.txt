#include<bits/stdc++.h>
using namespace std;
struct equation
{
	bitset<100001> a;
	int r;
}x[101];
long long a[100001],t;
int n,ind,ans[100001],last[100001];
void add(int now,int as)
{
	++ind;x[ind].r=as;
	for(int i=1;i<=n;i++)if(a[i]>>now&1)x[ind].a[i]=1;
	for(int i=1;i<=n;i++)
		if(x[ind].a[i])
			if(last[i])
			{
				x[ind].a^=x[last[i]].a;
				x[ind].r^=x[last[i]].r;
			}
			else break;
	for(int i=1;i<=n;i++)if(x[ind].a[i]){last[i]=ind;return;}
	--ind;
}
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)cin>>a[i],t^=a[i];
	for(int i=62;i>=0;i--)if(!(t>>i&1))add(i,1);
	for(int i=62;i>=0;i--)if(t>>i&1)add(i,0);
	for(int j=n;j>=1;--j)
		if(x[last[j]].r)
		{
			ans[j]=1;
			for(int i=1;i<=ind;i++)
				if(x[i].a[j])
					x[i].r^=1;
		}
		else ans[j]=2;
	for(int i=1;i<=n;i++)printf("%d ",ans[i]);
}
