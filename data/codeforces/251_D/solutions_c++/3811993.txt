#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define N 120000
using namespace std;
int i,j,k,l,n,m,be[N],num[N],ans[N],tmp[66];
long long sum,a[N],p[N],o[N],x,y;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i)scanf("%I64d",&a[i]),sum^=a[i];
	for(i=0;i<=62;++i)tmp[i]=(sum>>i)&1;
	for(i=1;i<=n;++i){
		x=a[i]; y=0;
		for(j=62;j>=0;--j)if( ((sum>>j)&1)==0 && (x>>j)&1){
			if(!p[j])p[j]=x,num[j]=i,o[j]=y;
			x^=p[j];
			y|=1ll<<j;
		}
		for(j=62;j>=0;--j)if( ((sum>>j)&1) && (x>>j)&1){
			if(!p[j])p[j]=x,num[j]=i,o[j]=y;
			x^=p[j];
			y|=1ll<<j;
		}
	}
	for(j=62;j>=0;--j)if( tmp[j]==0 && !(sum>>j&1) && p[j]){
		sum^=p[j];
		be[j]=1;
	}
	for(j=62;j>=0;--j)if( tmp[j]==1 && (sum>>j&1)==0 && p[j]){
		sum^=p[j];
		be[j]=1;
	}
	for(j=0;j<=62;++j)if(tmp[j] && be[j]==1)
		for(k=0;k<=62;++k)if((o[j]>>k)&1)be[k]^=1;

	for(j=0;j<=62;++j)if(!tmp[j] && be[j]==1)
		for(k=0;k<=62;++k)if((o[j]>>k)&1)be[k]^=1;
	//
	for(i=0;i<=62;++i)if(be[i])ans[num[i]]=1;
		else ans[num[i]]=2;
	for(i=1;i<=n;++i)printf("%d ",ans[i]==0?2:ans[i]);
}
