#include<cstdio>
#include<iostream>
#include<bitset>
#define ll long long
#define N 100009
using namespace std;
int n,t,tt;
ll x[N],b[N],d[N],c;
bitset<N> a[70];
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>x[i],c^=x[i];
        for(int j=0;j<63;j++)
            a[62-j][i]=(x[i]&(1ll<<j))>0;
    }
    for(int i=0;i<63;i++)
    {
        ll mask=1ll<<(62-i);
        if(~c&mask)
        {
            a[69]=a[i];
            for(int j=i;j>tt;j--) a[j]=a[j-1];
            a[tt++]=a[69];
        }
    }
    for(int i=0;i<63;i++)
    {
        if(i<tt) a[i][n]=1;
        for(int j=0;j<t;j++) if(a[i][d[j]]) a[i]^=a[j];
        int p=0;
        for(;p<n&&!a[i][p];p++);
        if(p==n) a[i][n]=0;
        else
        {
            for(int j=0;j<t;j++) if(a[j][p]) a[j]^=a[i];
            d[t]=p,swap(a[t++],a[i]);
        }
    }
    while(t--) b[d[t]]=a[t][n];
    for(int i=0;i<n;i++) b[i]?printf("1 "):printf("2 ");
}
