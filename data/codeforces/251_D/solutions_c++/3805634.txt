#include<cstdio>
#define rep(i,n) for (int i=0;i<n;i++)
#define N 100005
int n,flag,rk[N],o[N],b[N],p[N]; long long a[N],c[65],q[65],e[3],res,x,y;
int main()
{
	scanf("%d",&n); rep(i,63) p[i]=-1;
	rep(i,n) scanf("%I64d",a+i),res^=a[i]; rep(i,63) rk[i]=res>>i&1;
	rep(i,n){
		x=a[i],y=0;
		for (int j=62;j>=0;j--) if (!rk[j] && x>>j&1)
			if (p[j]<0) {p[j]=i,c[j]=x,q[j]=y,x=-1; break;} else x^=c[j],y|=1LL<<j;
		if (x<0) continue;
		for (int j=62;j>=0;j--) if (rk[j] && x>>j&1)
			if (p[j]<0) {p[j]=i,c[j]=x,q[j]=y,x=-1; break;} else x^=c[j],y|=1LL<<j;
		}
	x=0;
	for (int i=62;i>=0;i--)
		if (!rk[i] && !(x>>i&1) && p[i]>=0) x^=c[i],b[i]^=1;
	for (int i=62;i>=0;i--)
		if (rk[i] && !(x>>i&1) && p[i]>=0) x^=c[i],b[i]^=1;
	rep(i,63) if (b[i] && rk[i]) rep(j,63) if (q[i]>>j&1) b[j]^=1;
	rep(i,63) if (b[i] && !rk[i]) rep(j,63) if (q[i]>>j&1) b[j]^=1;
	rep(i,63) if (b[i]) o[p[i]]=1; rep(i,n) e[o[i]]^=a[i];
	rep(i,n) printf("%d%s",o[i]+1,i<n-1?" ":"\n");
	return 0;
}
