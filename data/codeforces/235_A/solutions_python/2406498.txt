from fractions import gcd

def lcm(a, b):
    return a*b/gcd(a, b)

def main():
    n = input()
    ans = 0
    for i in xrange(max(1,n-50), n+1):
        for j in xrange(i, n+1):
            for k in xrange(j, n+1):
                ans = max(ans, lcm(lcm(i, j), k))
    print ans
    
if __name__ == '__main__':
    main()
