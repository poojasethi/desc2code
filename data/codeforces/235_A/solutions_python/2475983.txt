def cmm(a, b):
    if (b == 0):
        return a
    else:
        return cmm(b, a % b)

n = int(input())
result = 1
for x in range(n, max(n-50, 1), -1):
    for y in range(x, max(x-50, 1), -1):
        for z in range(y, max(y-50, 1), -1):
            a = x * y // cmm(x, y)
            result = max(result, a * z // cmm(a, z))

print(result)
