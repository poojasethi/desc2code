#!/usr/bin/python

def gcd(a, b) :
	if b == 0 : return a
	else : return gcd(b, a % b)

def lcm(a, b) :
	return a * b / gcd(a, b)

n = int(raw_input())
z = 0
for i in range(max(1, n - 30), n + 1) :
	for j in range(max(1, n - 30), n + 1) :
		for k in range(max(1, n - 30), n + 1) :
			z = max(z, lcm(i, lcm(j, k)))

print(z)
