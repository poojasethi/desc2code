from fractions import gcd
n = int(raw_input())
if n==1:
    print 1
elif n==2:
    print 2
else:
    if n%2==1:
        print n*(n-1)*(n-2)
    else:
        num = (n-1)*(n-2)*(n-3)
        x = n-2
        while x>=1:
            if gcd(n,x)==1 and gcd(n-1,x)==1:
                break
            x -=1
        print max(x*n*(n-1),num)
