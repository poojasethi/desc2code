#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
#include <cassert>

using namespace std;
typedef long long ll;
int j,i,n, p, k,m,a[1111][1111];

int main(){
ios_base::sync_with_stdio(0);
cin>>n>>m;
for (i=0;i<n;i++)
 for (j=0;j<m;j++) cin>>a[i][j];
int ans=min(min(a[1][1],a[n][1]),min(a[1][m],a[n][m]));
for (i=0;i<n;i++)
 for (int ii=i+1;ii<n;ii++)
 {
   int t1=-1,t2=-1,t;
   for (j=0;j<m;j++)
   if (a[i][j]>ans && a[ii][j]>ans) {
                                  t=min(a[i][j],a[ii][j]);
                                  if (t>t1) {t2=t1;t1=t;}
                                        else {if (t>t2) t2=t;}
   }
  ans=max(ans,t2);
 }


 cout<<ans;

}
