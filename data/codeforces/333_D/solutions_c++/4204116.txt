#include<algorithm>
#include<cstdio>
#define f(i,a,b) for(int i=a;i<b;i++)
int B,X,Y,N,M,A[1005][1005];
using namespace std;
int fastmin(int x,int y)
{
    return (((y-x)>>(31))&(x^y))^x;
}
int fastmax(int x,int y)
{
    return (((y-x)>>(31))&(x^y))^y;
}
int main()
{
	scanf("%d%d",&N,&M);
	f(i,0,N)f(j,0,M)scanf("%d",A[i]+j);
	f(i,0,N)f(j,i+1,N)
    {
        X=Y=0;
        f(k,0,M)
            if(A[i][k]>B && A[j][k]>B)
            {
                X=fastmax(X,fastmin(A[i][k],A[j][k]));
                if(X>Y)
                    swap(X,Y);
            }
        B=fastmax(B,X);
    }
	printf("%d",B);
}
