#include<algorithm>
#include<cstdio>
#define FP(ii,aa,bb) for(ii=aa;ii<=bb;ii++)
using namespace std;
int m,n,i,j,k,l,a1,a2,res;
int a[1005][1005];
int main(){

  scanf("%d %d",&m,&n);

  FP( i,1,m ) FP( j,1,n ) scanf("%d",a[i]+j);

  FP( i,1,m-1 )
    FP( j,i+1,m ){
      a1 = -1;
      a2 = -1;
      FP( k,1,n ){
        if( a[i][k]>res and a[j][k]>res ){
          l = min( a[i][k],a[j][k] );
          if( l>=a2 ){ a1 = a2; a2 = l; }
          else if( l>a1 ){ a1 = l; }
        }
      }
      res = max( res,a1 );
    }
  printf("%d\n",res);

}
