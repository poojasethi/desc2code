#include <stdio.h>

int N,M,A[1001][1001],ans;

int main()
{
    int i,j,k,mx1,mx2;
    scanf("%d%d",&N,&M);
    for (i=1;i<=N;i++) for (j=1;j<=M;j++) scanf("%d",A[i]+j);
    for (i=1;i<N;i++) for (j=i+1;j<=N;j++){
        mx1 = mx2 = 0;
        for (k=1;k<=M;k++) if (A[i][k] > ans && A[j][k] > ans){
            int v=A[i][k]<A[j][k]?A[i][k]:A[j][k];
            if (mx1 < v) mx2 = mx1, mx1 = v;
            else if (mx2 < v) mx2 = v;
        }
        if (ans < mx2) ans = mx2;
    }
    printf("%d\n",ans);
}
