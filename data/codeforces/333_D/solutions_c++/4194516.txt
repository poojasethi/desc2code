#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int er[1010][1010];
int main(){
	int n, m;
	int i, j, o;
	scanf("%d%d", &n, &m);
	for(i = 0; i < n; i ++){
		for(j = 0; j < m; j ++){
			scanf("%d", &er[i][j]) ;
		}
	}
	int ans = 0;
	for(i = 0; i < n; i ++){
		for(j = i+1; j < n; j ++){
			int mx1 = -1, mx2 = -1;
			for(o = 0; o < m; o ++){
				if(er[i][o] > ans && er[j][o] > ans){
					int tmp = min(er[i][o], er[j][o]);
					if(tmp > mx1){mx2 = mx1;mx1 = tmp;}
					else if(tmp > mx2) mx2 = tmp;
				}
			}
			ans = max(mx2, ans);
		}
	}
	printf("%d\n", ans);
	return 0;
}
