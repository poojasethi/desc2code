#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int MAXN = 1010;

int n, m;
int a[MAXN][MAXN];

int main() {
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < m; ++j)
			scanf("%d", &a[i][j]);

	int ans = 0;
	for (int r1 = 0; r1 < n; ++r1)
		for (int r2 = r1 + 1; r2 < n; ++r2) {
			int mx1 = -1, mx2 = -1;
			for (int c = 0; c < m; ++c) if (a[r1][c] > ans && a[r2][c] > ans) {
				int b = min(a[r1][c], a[r2][c]);
				if (b > mx1) mx2 = mx1, mx1 = b;
				else if (b > mx2) mx2 = b;
			}
			ans = max(ans, mx2);
		}
	printf("%d\n", ans);
}
