#include"iostream"
#include"string"
#define f(i,a,b) for(int i=a;i<=b;i++) 
using namespace std;
int n,m;
long x=0,a[1001][1001],x_pre=0,res;
main(){
       cin>>n>>m;
       f(i,1,n) f(j,1,m) cin>>a[i][j];
       f(i,1,n-1) f(j,i+1,n){
        x=0;x_pre=0;          
        f(k,1,m) if(a[i][k]>res&&a[j][k]>res){
                 x=max(x,min(a[i][k],a[j][k]));
                 if(x>x_pre) swap(x,x_pre);
                 }          
       res=max(res,x);
       }
       cout<<res;
       //system("pause");
       }
