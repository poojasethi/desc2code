#include<cstdio>
#define f(i,a,b) for(int i=a;i<b;i++)
int B,X,Y,N,M,A[1005][1005],Z;
int main()
{
    scanf("%d%d",&N,&M);
    f(i,0,N)f(j,0,M)scanf("%d",&A[i][j]);
    f(i,0,N)f(j,i+1,N)
    {
        X=Y=0;
        f(k,0,M)if(A[i][k]>B&&A[j][k]>B)
            if((Z=A[i][k]<A[j][k]?A[i][k]:A[j][k])>Y)
                X=Y,Y=Z;
            else if(Z>X)
                X=Z;
        B=X>B?X:B;
    }
    printf("%d",B);
}
