#include<stdio.h>
#include<algorithm>
using namespace std;
int ans,Max,Maxx,i,j,k,n,m,t;
int a[1010][1010];
int main(){
    scanf("%d%d",&n,&m);
    for (i=1;i<=n;i++)
        for (j=1;j<=m;j++)
            scanf("%d",a[i]+j);
    for (i=1;i<n;i++)
        for (j=i+1;j<=n;ans=max(ans,Maxx),j++)
            for (Max=Maxx=0,k=1;k<=m;k++)
                if (a[i][k]>ans && a[j][k]>ans)
                    if ((t=min(a[i][k],a[j][k]))>=Max) Maxx=Max,Max=t;
                    else Maxx=max(t,Maxx);
    printf("%d\n",ans);
}
