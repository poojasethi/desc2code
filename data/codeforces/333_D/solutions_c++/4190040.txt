#include<cstdio>
#include<algorithm>

using namespace std;

int g[1010][1010], n, m, ans;

int max(int x, int y) { return (((y-x)>>(32-1))&(x^y))^y; }
int min(int x, int y) { return (((y-x)>>(32-1))&(x^y))^x; }

int main(){
	scanf("%d%d", &n, &m);
	for(int i = 1; i <= n; i++)
		for(int j = 1; j <= m; j++)
			scanf("%d", &g[i][j]);
	for(int i = 1; i < n; i++)
		for(int j = i + 1; j <= n; j++){
			int a1 = 0, a2 = 0;
			for(int k = 1; k <= m; k++){
				const int tmp = min(g[i][k], g[j][k]);
				if(tmp > a1) a2 = a1, a1 = tmp;
				else if(tmp > a2) a2 = tmp;
			}
			ans = max(ans, a2);
		}
	printf("%d\n", ans);
	return 0;
}
