#include<algorithm>
#include<cstdio>
#define f(i,a,b) for(int i=a;i<b;i++)
#define min(a, b) ((a < b) ? a : b)
int B,X,Y,N,M,A[1005][1005], y1, y2, y3;
using namespace std;
int main()
{
    scanf("%d%d",&N,&M);
    f(i,0,N)f(j,0,M)scanf("%d",A[i]+j);
    f(i,0,N)f(j,i+1,N)
    {
        X=Y=0;
        f(k,0,M) {
        y1 = A[i][k];
        y2 = A[j][k];
        if (y1 <= B) continue;
        y3 = min(y1, y2);
        if(y3>B)
        {
            X=max(X,y3);
            if(X>Y)swap(X,Y);
        }
        }
        B=max(B,X);
    }
    printf("%d",B);
}
