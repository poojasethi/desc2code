#include <cstdio>
#include <cstring>
#include <algorithm>
const long M=1100;
const long inf=1<<30;
long a[M][M],n,m;
int main(){
	while (~scanf("%d%d",&n,&m)){
		for (long i=1;i<=n;++i)
			for (long j=1;j<=m;++j)
				scanf("%d",a[i]+j);
		long ans=-1;
		for (long l=1;l<n;++l)
			for (long r=l+1;r<=n;++r){
				long max1=-1,max2=-1;
				for (long i=1;i<=m;++i)
				if (a[l][i]>ans && a[r][i]>ans){
					max1=std::max(max1,std::min(a[l][i],a[r][i]));
					if (max1>max2) std::swap(max1,max2);
				}
				ans=std::max(ans,max1);
			}
		printf("%d\n",ans);
	}
	return 0;
}
