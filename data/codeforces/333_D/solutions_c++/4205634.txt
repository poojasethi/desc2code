#include<algorithm>
#include<cstdio>
#define f(i,a,b) for(int i=a;i<b;i++)
int B,X,Y,N,M,A[1005][1005], d;
using namespace std;
int main()
{
    d = 0;
    scanf("%d%d",&N,&M);
    f(i,0,N)f(j,0,M)scanf("%d",A[i]+j);
    f(i,0,N)f(j,i+1,N)
    {
        X=Y=0;
        
        f(k,0,M)if(A[i][k] > -1 && A[j][k]>B && A[i][k]>B)
        {
            X=max(X,min(A[i][k],A[j][k]));
            if(X>Y)swap(X,Y);
        }
        B=max(B,X);
    }
    B += d;
    printf("%d",B-d);
}
