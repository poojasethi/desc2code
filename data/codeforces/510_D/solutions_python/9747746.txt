from fractions import gcd

n = input()


l = map(int, raw_input().split())
c = map(int, raw_input().split())

d = {0:0}



for x in xrange(n):
    L = {}
    for j in d.keys():
        k, p = gcd(l[x],j), d[j] + c[x]
        if k in L.keys():
            L[k] = min(L[k], p)
        else: L[k] = p
    for r in L:
        if r in d.keys():
            d[r] = min(d[r], L[r])
        else: d[r] = L[r]

if 1 in d.keys():
    print d[1]
else: print -1
