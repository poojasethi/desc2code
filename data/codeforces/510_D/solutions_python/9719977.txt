import fractions
In = lambda:map(int, raw_input().split())
input()
ads = {0:0}
for vi in zip(In(), In()):
    for item in ads.items():
        a = fractions.gcd(item[0], vi[0])
        if ads.has_key(a):
            ads[a] = min(item[1] + vi[1], ads[a])
        else: ads[a] = item[1] + vi[1]

print -1 if not ads.has_key(1) else ads[1]
