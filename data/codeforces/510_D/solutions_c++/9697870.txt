#include<bits/stdc++.h>

using namespace std;

#define mp make_pair

map< pair<int,int> , int> dp;

int cost[1000];
int a[1000],i,n;

int gcd(int x,int y){
	if(!x)
		return y;
	return gcd(y%x,x);
}

int f(int x,int gc){

	if(dp[mp(x,gc)])
		return dp[mp(x,gc)];

	if(gc == 1)
		return 0;
	
	if(x == n+1)
		return (int)1e9+7;

	return dp[mp(x,gc)] = min( f(x+1,gcd(gc,a[x])) + cost[x] , f(x+1,gc));
}

int main(){

	cin >> n;

	for(i=1 ; i<=n ; i++)
		cin >> a[i];

	for(i=1 ; i<=n ; i++)
		cin >> cost[i];

	int t = f(1,0);

	if(t == (int)1e9+7)
		cout << -1 << endl;
	else
		cout << t << endl;

}
