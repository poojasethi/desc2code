#include<iostream>
#include<vector>
#include<map>
#define int long long 
using namespace std;

map<pair<int, int> ,int> M;
int l[500], c[500], n;

int gcd(int a, int b)
{
  return !b ? a : gcd(b, a % b);
}

int solve(int r,int g)
{
    if (r == n)
      return g == 1 ? 0 : 1000000000000000ll;
    if (M.count(make_pair(r,g)))
      return M[make_pair(r,g)];
    
    return M[make_pair(r,g)] = min(solve(r + 1, g), solve(r + 1, gcd(l[r], g)) + c[r]);
}

 main()
{
  cin>>n;
  for (int i=0;i<n;i++)
    cin>>l[i];
  for (int i=0;i<n;i++)
    cin>>c[i];
  int ans = solve(0, 0);
  if (ans >= 100000000000000ll)
    cout<<-1<<endl;
  else cout<<ans<<endl;
  return 0;
}
