#include <bits/stdc++.h>

using namespace std;

#define r dp[i][k]

typedef long long ll;
typedef pair < int, int > pii;

const ll linf = 1e18 + 9;
const int inf = 1e9 + 9;
const int N = 5e2 + 9;

map< ll, ll > dp[N];

ll n, a[N], b[N];

ll f(ll i, ll k) {
	
	if (k == 1) return r = 0;

	if (i > n) return r = inf;

	if (r) return r;

	int t;
	
	if (!k) t = a[i];
	
	else t=__gcd(k,a[i]);
	
	r = min(f(i+1, 0), min(f(i+1, t)+b[i], f(i+1, k)));
	
	return r;
}

int main() {

	scanf("%lld", &n);

	for (int i = 1; i <= n; i++)
		scanf("%lld", &a[i]);

	for (int i = 1; i <= n; i++)
		scanf("%lld", &b[i]);

	if (f(1, 0) != inf) cout << f(1, 0) << endl;
	else cout << -1 << endl;
	
	return 0;
}
