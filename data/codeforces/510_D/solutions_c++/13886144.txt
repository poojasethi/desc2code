#include<bits/stdc++.h>

using namespace std;

map < pair <int , int> , int > M;

int A[100005],B[100005],n,i;

int f(int i, int eb){
	
	if(eb == 1)
		return 0;

	if(i == n+1)
		return 985934802;

	int &r = M[ make_pair(i,eb) ];

	if(r)
		return r;

	return r = min( f(i+1,(eb ? __gcd(eb,A[i]) : A[i])) + B[i] , f(i+1,eb) );
}
int main(){

	cin >> n;

	for(i=1;i<=n;i++)
		cin >> A[i];

	 for(i=1;i<=n;i++)
		 cin >> B[i];

	 cout << (f(1,0) == 985934802 ? -1 : f(1,0));
	 return 0;
}
