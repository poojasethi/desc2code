#include <bits/stdc++.h>
#define INF 100000000000LL
using namespace std;
typedef long long ll;
typedef pair<ll,ll> ii;

ll n;
vector<ll> l;
vector<ll> c;
map<ii,ll> mp;
ll cdp(ll pos, ll d){
    if(d==1) return 0;
    if(pos==-1) return INF;
    ii tp=ii(pos,d);
    if(mp.count(tp)) return mp[tp];
    ll gcdt;
    if(d==0) gcdt=l[pos];
    else gcdt=__gcd(l[pos],d);
    return mp[tp]=min(c[pos]+cdp(pos-1,gcdt), cdp(pos-1,d)); 
}
int main(){
    cin>>n;
    l.resize(n);
    c.resize(n);
    for(int i=0;i<n;++i)
        cin>>l[i];
    for(int i=0;i<n;++i)
        cin>>c[i];
    ll res=cdp(n-1,0);
    if(res<INF) cout<<res;
    else cout<<-1;
}
