#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n],c[n];

	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(int i=0;i<n;i++)
	{
		cin>>c[i];
	}
	map <int,int> mapi;
	map <int,int> :: iterator it;
	for(int i=0;i<n;i++)
	{
		int &x=mapi[a[i]];
		if(x==0)
		{
			x=c[i];
		}
		else
		{
			x=min(x,c[i]);
		}
		for(it=mapi.begin();it!=mapi.end();it++)
		{
			int &y=mapi[__gcd(a[i],it->first)];
			if(y==0)
			{
				y=it->second+c[i];
			}
			else
			{
				y=min(y,it->second+c[i]);
			}
		}
	}
	if(mapi[1]==0)
	{
		cout<<"-1"<<endl;
	}
	else
	{
		cout<<mapi[1]<<endl;
	}
}
