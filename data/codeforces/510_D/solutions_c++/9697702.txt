#include<bits/stdc++.h>

using namespace std;

#define mp make_pair

int n;

int L[305],C[305];

map<pair<int,int> , int > DP;

int gcd(int x,int y){
	return !y ? x : gcd(y,x%y);
}

int f(int x,int g){

	if(DP[mp(x,g)])
		return DP[mp(x,g)];

	if(g==1)
		return 0;
	if(x==n+1)
		return INT_MAX/2;

	return DP[mp(x,g)]=min(f(x+1,gcd(g,L[x]))+C[x],f(x+1,g));

}

int main(){

	int i;

	scanf("%d",&n);

	for(i=1 ; i<=n ; i++)
		scanf("%d",L+i);

	for(i=1 ; i<=n ; i++)
		scanf("%d",C+i);
	
	int t=f(1,0);

	if(t==INT_MAX/2)
		return cout<<-1,0;

	cout<<t;

	return 0;
}
