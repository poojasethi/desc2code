#include <bits/stdc++.h>
#define INF 1000000002LL
using namespace std;
typedef pair<int,int> ii;

int n;
vector<int> l;
vector<int> c;
map<ii,int> mp;
int cdp(int pos, int d){
    if(d==1) return 0;
    if(pos==-1) return INF;
    ii tp=ii(pos,d);
    if(mp.count(tp)) return mp[tp];
    return mp[tp]=min(c[pos]+cdp(pos-1,__gcd(l[pos],d)), cdp(pos-1,d)); 
}
int main(){
    cin>>n;
    l.resize(n);
    c.resize(n);
    for(int i=0;i<n;++i)
        cin>>l[i];
    for(int i=0;i<n;++i)
        cin>>c[i];
    int res=cdp(n-1,0);
    if(res<INF) cout<<res;
    else cout<<-1;
}
