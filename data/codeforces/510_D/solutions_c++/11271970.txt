#include <bits/stdc++.h>
using namespace std;
int n ;
int c[100000] ;
long long dis[100000] ;
map< pair<int,long long> , long long >mp ;
long long solve(int x ,long long gcd){
    if(gcd == 1)return 0 ;
    else if(x == n )return 2e9 ;
    if(mp[{x,gcd}])return mp[{x,gcd}] ;
    return mp[{x,gcd}] = min(c[x] + solve(x+1,__gcd(gcd,dis[x])),solve(x+1,gcd)) ;
}
int main()
{
    cin >> n ;
    for(int i = 0 ;i < n ; i++)cin >> dis[i] ;
    for(int i = 0 ;i < n ; i++)cin >> c[i] ;
    long long x = solve(0,0) ;
    if(x == 2e9 )cout << -1 ;
    else {
        cout << x;
    }
    return 0 ;
}
