#include<bits/stdc++.h>
#define ll long long
using namespace std;
map<ll,ll> ans,ans1;
map<ll,ll> ::iterator it;


int main()
{
    ios_base::sync_with_stdio(0);
	ll a[301],c[301],i,j,k,l,m,n;
	cin>>n;
	for(i=0;i<n;i++) cin>>a[i];
	for(i=0;i<n;i++) cin>>c[i];
	
	for(i=0;i<n;i++)
	{  
		for(it=ans.begin();it!=ans.end();it++){
			j=it->first;
			k=__gcd(a[i],j);
			l=it->second;
			if(ans[k]==0)
			ans[k]=c[i]+l;
			else
			ans[k]=min(l+c[i],ans[k]);
		}
		if(!ans[a[i]])
		ans[a[i]]=c[i];
		else
		ans[a[i]]=min(ans[a[i]],c[i]);
		
}
	if(!ans[1])
	cout<<-1;
	else
	cout<<ans[1];
	

	
}
