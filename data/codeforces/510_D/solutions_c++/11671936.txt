#include <bits/stdc++.h>
using namespace std;
#define f first
#define s second
vector<pair<long long ,long long> > a(301);
map<long long ,long long> dp;
int main()
{
 int n;
 cin >> n;
 dp[0] = 0;
 map<long long ,long long>::iterator it;
 for(int i=0;i<n;++i)
    cin >> a[i].f;
 for(int i=0;i<n;++i)
    cin >> a[i].s;
 for(int i=0;i<n;++i)
 {
  for(it=dp.begin();it!=dp.end();++it)
  {
    long long g = __gcd(it->f,a[i].f);
   if(dp.find(g)!=dp.end()) dp[g] = min(dp[g],it->s+a[i].s);    
   else
    dp[g] = it->s+a[i].s;
  }
}
if(dp.find(1) == dp.end())
cout << -1 << endl;
else
cout << dp[1] << endl;
}
