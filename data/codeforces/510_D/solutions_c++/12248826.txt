#include <cstdio>
#include <algorithm>
#include <map>
using namespace std;
const int MAX_N = 310;
int l[MAX_N], c[MAX_N];
int main()
{
	int n;
	scanf("%d", &n);
	for (int i = 0;i < n;i++) scanf("%d", l+i);
	for (int i = 0;i < n;i++) scanf("%d", c+i);
	map<int, int> m;
	for (int i = 0;i < n;i++){
        if(m[l[i]]) m[l[i]] = min(m[l[i]], c[i]);
        else m[l[i]] = c[i];
        for(map<int, int>::iterator it = m.begin() ;it != m.end() ;it++){
            int g = __gcd(l[i],it->first);
            if(m[g]) m[g] = min(m[g], it->second+c[i]) ;
            else m[g] = it->second+c[i] ;
        }		
	}
	if (m[1])	printf("%d\n",m[1]);
	else	printf("-1\n");
	return 0;
}
