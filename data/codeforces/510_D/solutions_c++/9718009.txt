#include<iostream>
#include<algorithm>
#include<map>

using namespace std;

int l[333], c[333];
map<int, int> m[333], dp[333];

int rec(int n, int g)
{
	if(g == 1)
		return 0;
	if(n < 0)
		return 1e9;
	if(m[n][g])
		return dp[n][g];
	m[n][g] = true;
	return dp[n][g] = min(rec(n - 1, g), rec(n - 1, __gcd(l[n], g)) + c[n]);
}

int main()
{
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
		cin >> l[i];
	for(int i = 0; i < n; i++)
		cin >> c[i];
	int ans = rec(n, 0);
	if(ans == 1e9)
		cout << -1 << endl;
	else
		cout << ans << endl;
	return 0;
}
