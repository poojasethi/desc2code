#include <bits/stdc++.h>
using namespace std;

int N, l[333], c[333], sz;
map <int, int> cost;
vector <int> jump;

main()
{
    scanf("%d", &N);
    for (int i = 1; i <= N; i++) scanf("%d", &l[i]);
    for (int i = 1; i <= N; i++) scanf("%d", &c[i]);

    jump.push_back(0);      sz = 1;
    for (int i = 1; i <= N; i++)
        for (int j = 0; j < sz; j++)
        {
            int x = __gcd(l[i], jump[j]);
            if (cost[x] == 0)
            {
                jump.push_back(x);      sz++;
                cost[x] = 1 << 30;
            }
            cost[x] = min(cost[x], cost[jump[j]] + c[i]);
        }

    printf("%d", cost[1] ? cost[1] : -1);
}
