#include <algorithm>
#include <cstdio>
const int inf=1e9;
int l[305],c[305],a[20],f[512],i,j,k,n,m,x,t,ans=inf;
int main()
{
	scanf("%d\n",&n);
	for (i=1;i<=n;++i) scanf("%d",&l[i]);scanf("\n");
	for (i=1;i<=n;++i) scanf("%d",&c[i]);scanf("\n");
	for (i=1;i<=n;++i)
	{
		x=l[i];t=0;
		for (j=2;j*j<=x;++j)
			if (x%j==0)
				for (a[t++]=j;x%j==0;x/=j);
		if (x>1) a[t++]=x;
		f[m=(1<<t)-1]=c[i];
		for (j=0;j<m;++j) f[j]=inf;
		for (j=1;j<=n;++j)
		if (i!=j)
		{
			x=0;
			for (k=0;k<t;++k)
				if (l[j]%a[k]==0) x|=1<<k;
			for (k=0;k<=m;++k)
				f[k&x]=std::min(f[k&x],f[k]+c[j]);
		}
		ans=std::min(ans,f[0]);
	}
	if (ans==inf) ans=-1;	
	printf("%d\n",ans);
}
