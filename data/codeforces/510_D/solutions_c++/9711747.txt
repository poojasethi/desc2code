#include<stdio.h>
#include<bits/stdc++.h>
#include<vector>
#include<map>
using namespace std;
int n,a[1001],b[1001];
map<int,int> m;
int main()
{
    int i;
    scanf("%d",&n);    
    for(i=1;i<=n;i++)
        scanf("%d",&a[i]);
    for(i=1;i<=n;i++)
        scanf("%d",&b[i]);
    m[0]=0;
    for (i=1;i<=n;i++)
    {
        for(map<int,int>::iterator j=m.begin();j!=m.end();j++) 
        {
            int k=__gcd(j->first,a[i]);
            if(!m[k]) m[k]=j->second+b[i];
            m[k]=min(m[k],j->second+b[i]);
        }
    }
    if(!m[1])
        printf("-1\n");
    else
        printf("%d\n",m[1]);
    return 0;
}
