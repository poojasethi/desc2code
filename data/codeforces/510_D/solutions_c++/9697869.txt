#include<bits/stdc++.h>

#define mp make_pair

using namespace std;

int n,m,j,k,i,l;
int A[305],B[305];
map<pair<int,int>,int> Mp;

int gcd(int x,int y){
	return !y ? x : gcd(y,x%y);
}

int f(int x,int g){
	
	if(Mp[mp(x,g)])
		return Mp[mp(x,g)];

	if(g == 1)
		return 0;

	if(x == n+1)
		return INT_MAX/2;
	
	return Mp[mp(x,g)] = min(f(x+1,g),f(x+1,gcd(g,A[x])) + B[x]);

}

int main(){
	
	scanf("%d",&n);

	for(i=1 ; i<=n ; i++)
		scanf("%d",&A[i]);
	
	for(i=1 ; i<=n ; i++)
		scanf("%d",&B[i]);

	l = f(1,0);

	if(l == INT_MAX/2)
		return cout << -1,0;

	printf("%d",l);

	return 0;
}
