#include<bits/stdc++.h>
#define ll long long
using namespace std;
map<ll,ll> ans,ans1;
map<ll,ll> ::iterator it;
int main()
{

	ll a[301],c[301],i,j,k,l,m,n;
	cin>>n;
	for(i=0;i<n;i++) cin>>a[i];
	for(i=0;i<n;i++) cin>>c[i];
	
	for(i=0;i<n;i++)
	{  
		for(it=ans.begin();it!=ans.end();it++){
			j=it->first;
			k=__gcd(a[i],j);
			
			if(ans1[k])ans1[k]=min(c[i]+ans1[j],ans1[k]);
		    else 
			ans1[k]=c[i]+ans1[j];
		}
		if(ans1[a[i]]) ans1[a[i]]=min(ans1[a[i]],c[i]);
		else ans1[a[i]]=c[i];
		ans=ans1;
	}
	if(!ans[1])
	cout<<-1;
	else
	cout<<ans[1];
	
}
