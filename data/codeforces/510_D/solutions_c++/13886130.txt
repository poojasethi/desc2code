using namespace std;
typedef long long ll;
const int N = 3e2+3,M=1e9+4;
map< pii , int > dp;
int n,m,i,j,k,l,s,t,d,u,ii,jj,x,y,z,h[N],hh[N];
int f(int i,int g)
{
	if(g==1)return 0;
	if(i>n) return M;
	if(dp[mp(i,g)])return dp[mp(i,g)];
	int t,s;
	if(!g)t=h[i];
	else t=__gcd(g,h[i]);
	return dp[mp(i,g)]=min(f(i+1,g),f(i+1,t)+hh[i]);
}
int main()
{
	cin>>n;
	for(i=1;i<=n;i++)cin>>h[i];
	for(i=1;i<=n;i++)cin>>hh[i];
	x=f(1,0);
	if(x==M)cout<<-1<<endl;
	else cout<<x<<endl;
}
