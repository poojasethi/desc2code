#include <bits/stdc++.h>
using namespace std;

int n, ns[305], cs[305];

map<int,int> mem[305];

int solve(int id=0, int gcd=0){
    if(id>=n) return gcd==1?0:1<<30;
    if(mem[id].count(gcd))
        return mem[id][gcd];
    return mem[id][gcd]=min(
        solve(id+1, __gcd(gcd, ns[id])) + cs[id],
        solve(id+1, gcd)
    );
}

int main(){
    
    cin>>n;
    
    for(int i=0; i<n; i++)
        cin>>ns[i];
    
    for(int i=0; i<n; i++)
        cin>>cs[i];
    
    int ans = solve();
    if(ans >= (1<<30)) cout<<-1<<endl;
    else cout<<ans<<endl;
    
    return 0;
}
