#include <bits/stdc++.h>
using namespace std;
template <typename T>
void rv(int n, vector<T>& xs) {
	T x;
	while(n--) {
		cin >> x;
		xs.push_back(x);
	}
}
int main() {
	int n; cin >> n;
	vector<int> l, c;
	rv(n, l);
	rv(n, c);
	map<int,int> f;
	f[0] = 0;
	for (int i = 0; i < n; ++i) {
		for (map<int,int>::iterator it=f.begin();it!=f.end();++it) {
			int g = __gcd((*it).first, l[i]);
			f[g] = min(f[g]?f[g]:1<<29, (*it).second + c[i]);
		}
	}
	cout << (f[1]? f[1] : -1);
}
