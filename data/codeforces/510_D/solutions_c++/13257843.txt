#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main() 
{
    long long int n,i,g;
    map<long long int,long long int>m;
    set<long long int>s;
    cin>>n;
    long long int l[n],c[n];
    for(i=0;i<n;i++)
    cin>>l[i];
    for(i=0;i<n;i++)
    cin>>c[i];
    s.insert(0);
    set<long long int>::iterator it;
    for(i=0;i<n;i++)
    {
        for(it=s.begin();it!=s.end();it++)
        {
            g=__gcd(*it,l[i]);
            if(s.find(g)==s.end())
            s.insert(g);
            if(m[g]==0 || m[g]>c[i]+m[*it])
            m[g]=m[*it]+c[i];
        }
    }
    if(m[1]==0)
    cout<<"-1";
    else
    cout<<m[1];
    return 0;
}
