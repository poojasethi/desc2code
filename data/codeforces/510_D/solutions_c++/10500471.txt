#include<algorithm>
#include<iostream>
#include<cstdio>
#include<cstring>
#include<map>
using namespace std;
map<int , int> dp;
map<int , int>::iterator it;
int l[500] , c[500];
int main()
{
	int n;
	scanf("%d", &n);
	for(int i=1; i<=n; ++i)
		scanf("%d", &l[i]);
	for(int i=1; i<=n; ++i)
		scanf("%d", &c[i]);
	dp[0] = 0;
	for(int i=1; i<=n; ++i)
	{
		for(it = dp.begin(); it != dp.end(); ++it)
		{
			int x = it -> first;
			int t = __gcd(x , l[i]);
			if(!dp[t])
				dp[t] = it -> second + c[i];
			else dp[t] = min(dp[t] , it -> second + c[i]);
		}
	}
	if(!dp[1])
		cout << "-1" << endl;
	else cout << dp[1] << endl;
	return 0;
}
