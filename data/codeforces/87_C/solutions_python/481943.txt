n = input()
a,f = [[] for _ in xrange(n+1)],[0]*(n+1)
for k in xrange(2,n):
    for x in xrange(1,n):
        z = x*k+(k*k-k>>1)
        if z>n: break
        a[z].append((k,x))
q,s = [n],set([n])
for j in q:
    l = set([])
    for k,x in a[j]: l.update(range(x,x+k))
    q.extend(l-s)
    s |= l
q = lambda(k,x): reduce(int.__xor__,f[x:x+k],0)
for i in sorted(s): f[i]=next(j for j in xrange(n) if j not in set(map(q,a[i])))
print min(p[0] for p in a[n] if q(p)==0) if f[n] else -1
