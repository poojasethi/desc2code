#include<bits/stdc++.h>
using namespace std;
const int N = 1e5+10;
int gn[N];
int my_xr[N];
int main()
{
	int n;
	scanf("%d",&n);
	set<int> mex;
	int k;
	my_xr[0]=0;
	gn[0]=0;
	for(int i=1;i<=n;++i)
	{
		mex.clear();
		for(k=2;((k*(k+1))/2)<=i;++k)
		{
			int x = i - (k*(k+1))/2;
			if(x%k) continue;
			x/=k;
			int y=my_xr[x]^my_xr[x+k];
			mex.insert(y);
		}
		int j;
		for(j=0;mex.find(j)!=mex.end();++j)
			;
		gn[i]=j;
		my_xr[i]=my_xr[i-1]^gn[i];
	}

	if(gn[n])
	{
		for(k=2;((k*(k+1))/2)<=n;++k)
		{
			int x = n - (k*(k+1))/2;
			if(x%k) continue;
			x/=k;
			int y=my_xr[x]^my_xr[x+k];
			if(!y)
				break;
		}
		printf("%d\n",k);
	}
	else printf("-1\n");
	return 0;
}
