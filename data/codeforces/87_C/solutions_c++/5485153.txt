#include<cstdio>
#include<iostream>
#include<cstring>
#include<set>
#define N 100000
using namespace std;

int memo[N+5] , res=-1, orig;

int g(int n){
	int &ret=memo[n];
	if(ret!=-1)return ret;
	set<int>S; 
	for(int k=2;k<=n;++k){
		int num=2*n-k*k+k;
		if(num/ (2.0*k) <1)break;
		if(num%(2*k)==0){
			//uso k
			int sum=0;
			int a=num/(2*k);
			for(int i=0;i<=k-1;++i){
				sum^=g(a+i);
			}
			if(orig==n && sum==0 && res==-1)res=k;
			S.insert(sum);
		}	
	}
	int dev=0;
	while(S.count(dev)!=0)dev++;
	ret=dev;
	return ret;	
}

int main(){
	int n;
	cin>>n;
	orig=n;
	memset(memo,-1,sizeof(memo));
	if(g(n)==0)puts("-1");
	else cout<<res<<endl;
}
