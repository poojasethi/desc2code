#include <algorithm>
#include <iostream>
using namespace std;

const int N = 1e5 + 5, sq = 500;

int n, sum[N];
bool mark[sq];

int main() {
	ios::sync_with_stdio(false);
	cin >> n;
	for (int i = 1; i <= n; i++) {
		fill(mark, mark + sq, false);
		for (int k = 2; k * (k + 1) / 2 <= i; k++)
			if (!((2 * i - k + k * k) % (2 * k))) {
				int x = (2 * i - k + k * k) / (2 * k);
				mark[sum[x] ^ sum[x - k]] = true;
				if ((i == n) && !(sum[x] ^ sum[x - k])) {
					cout << k << '\n';
					return 0;
				}
			}
		sum[i] = sum[i - 1] ^ (find(mark, mark + sq, false) - mark);
	}
	cout << -1 << '\n';
	return 0;
}
