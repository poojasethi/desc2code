#include <iostream>
#include <cstdio>
using namespace std;
#define SZ 100000
int a[SZ + 5], u[SZ + 5], p[SZ + 5], r[SZ + 5];
main()
{
  int n, b;
  cin >> n;
  a[1] = 0;
  for(int i = 2;i <= n; i++) 
  {
    for(int k = 2;2 * i - k * (k - 1) > 0; k++)
      if ((2 * i - k * (k - 1)) % (2 * k) == 0) 
      {
        b  = (2 * i- k * (k - 1)) / (2 * k);
        int e = p[b + k - 1] ^ p[b - 1];
        u[e] = i;
        if (e == 0 && r[i] == 0) r[i] = k;
      }
    int m = 0;
    while (u[m] == i) m++;
    a[i] = m;
    p[i] = p[i - 1] ^ m;
  }
  cout << (a[n] == 0? -1:r[n]);
  return 0;
}
