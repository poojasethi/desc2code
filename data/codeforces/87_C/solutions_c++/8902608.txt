#include<bits/stdc++.h>
using namespace std ;

#define MAX 100009

int game(int n) {
    vector<int> grundy(MAX), prefix(MAX) ;
    int ans = -1 ;
    for(int i=3;i<=n;i++) {
        map<int,bool> done ;
        for(int j=2;;j++) {
            int x = i - (j*(j+1))/2 ;
            if(x<0) break ;
            if(x%j) continue ;
            x /= j ;
            int val = prefix[x]^prefix[x+j] ;
            if(!val && ans < 0 && i == n)   ans = j ;
            done[val] = 1 ;
        }
        while(done[grundy[i]]) ++grundy[i] ;
        prefix[i] = prefix[i-1]^grundy[i] ;
    } 
    return ans ;
}

int main() {
    int n ;
    cin >> n ;
    cout << game(n) << endl ;
}
