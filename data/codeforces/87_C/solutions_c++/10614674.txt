main()
{
	int i,j,k,a,b,c;

	scanf("%d",&N);

	FOR(i,3,N)
	{
		FOR(j,2,300)
		{
			a = ((2*i) / j - j + 1) / 2;

			if(a >= 1 && ( a + a + j - 1) * j == i * 2)
			{
				b = 0;
				FOR(k,a,a+j-1)
					b ^= gr[k];

				h[b] = 1;

				if(!b && i == N)
				{
					printf("%d\n",j);
					exit(0);
				}
			}
		}

		FOR(j,0,300)
			if(!h[j])
			{
				gr[i] = j;
				break;
			}

		FOR(j,0,300) h[j] = 0;
	}

	printf("-1\n");
}
