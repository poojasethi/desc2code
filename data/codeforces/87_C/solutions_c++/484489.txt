#include <cstdio>
using namespace std;

int N,u[100001],D[100001],C[100001],r[100001];
int main () {
    scanf("%d",&N);
    for (int i=3;i<=N;++i) {
        for (int j=2;2*i>=j*(j+1);++j) if ((i-j*(j+1)/2)%j==0) {
            int x=(i-j*(j+1)/2)/j,d=C[x+j]^C[x];
            u[d]=i;
            if (!r[i] && !d) r[i]=j;
        }
        while (u[D[i]]==i) ++D[i];
        C[i]=C[i-1]^D[i];
    }
    if (!D[N]) printf("-1");
    else printf("%d\n",r[N]);
    return 0;
}
