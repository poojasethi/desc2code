#include<cstdio>
#include<cstring>

#define NMAX 100000

int sg[1+NMAX];
int zum[1+NMAX];
int vis[1+NMAX];
int ans[1+NMAX];

void build() {
	int n,k,m,a,b;
	memset(ans,0xff,sizeof(ans));
	for(n=3;n<=NMAX;n++) {
		for(k=2;k*(k-1)<2*n;k++) {
			if(k&1) {
				if(n%k)continue;
				m=n/k;
				a=m-(k-1)/2;
				b=m+(k-1)/2;
			} else {
				if(n%k*2!=k)continue;
				m=n/k;
				a=m+1-k/2;
				b=m  +k/2;
			}
			if(b>n)continue;
			m=zum[b]^zum[a-1];
			if(!m&&ans[n]<0)ans[n]=k;
			vis[m]=n;
		}
		for(m=0;vis[m]==n;m++);
		sg[n]=m;
		zum[n]=sg[n]^zum[n-1];
	}
}

int main() {
	int n;
	build();
	scanf("%d",&n);
	printf("%d\n",ans[n]);
	return 0;
}
