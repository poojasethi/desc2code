#include <iostream>
#include <cstdio>
#include <queue>
#include <set>
using namespace std;
#define SZ 100000
int a[SZ + 5], p[SZ + 5], r[SZ + 5];
int main()
{
  int n, b;
  cin >> n;
  a[1] = 0;
  for(int i = 2; i <= n; i++)
  {
    priority_queue <int> q;
    //set <int> q;
    for(int k = 2; 2 * i - k * (k - 1) > 0; k++)
      if ((2 * i - k * (k - 1)) % (2 * k) == 0)
      {
        b  = (2 * i- k * (k - 1)) / (2 * k);
        int e = p[b + k - 1] ^ p[b - 1];
        q.push(-e);
        if (e == 0 && r[i] == 0) r[i] = k;
      }
    while (!q.empty() && q.top() == -a[i])
    {
      while (!q.empty() && q.top() == -a[i]) q.pop();
      a[i]++;
    }
    p[i] = p[i - 1] ^ a[i];
  }
  cout << (a[n] == 0? -1:r[n]);
  return 0;
}
