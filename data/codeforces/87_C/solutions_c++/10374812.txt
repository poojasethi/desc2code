#include <bits/stdc++.h>
using namespace std;
int arr[100005],val[100005],sum[100005];
int main(){
  int n,i,j,p,k,a[505];
  arr[0]=0;
  sum[0]=0;
  for(i=1;i<100001;++i){
    val[i]=1e9;
    memset(a,0,sizeof(a));
    for(j=2;(j*(j+1))/2<=i;++j){
      k=0;p=(j*(j+1))/2;
      if((i-p)%j==0){
        p=(i-p)/j;
        k=sum[p]^sum[p+j];
        if(k==0) val[i]=min(val[i],j);
        a[k]=1;
      }
    }
    k=0;
    while(a[k]!=0) k++;
    arr[i]=k;sum[i]=sum[i-1]^k;
  }
  scanf("%d",&n);
  if(arr[n]==0) printf("-1\n");
  else printf("%d\n",val[n]);
  return 0;
}
