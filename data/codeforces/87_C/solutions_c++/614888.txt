#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int dp[100100],used[100];
int ca[100100];
int gao(int x)
{
	if(dp[x]!=-1)
	{
		return dp[x];
	}
	memset(used,0,sizeof(used));
	for(int i=1;i*(i+1)/2<x;i++)
	{
		if((x-i*(i+1)/2)%(i+1)==0)
		{
			int u=(x-i*(i+1)/2)/(i+1);
			int y=0;
			for(int j=0;j<=i;j++)
			{
				y^=gao(u+j);
			}
			if(y==0&&ca[x]==0) ca[x]=i+1;
			used[y]=1;
		}
	}
	int i;
	for(i=0;used[i];i++);
	return dp[x]=i;
}
int main(void)
{
	memset(dp,-1,sizeof(dp));
	memset(ca,0,sizeof(ca));
	dp[0]=dp[1]=dp[2]=0;
	for(int i=1;i<=100000;i++) gao(i);
	int n;
	scanf("%d",&n);
	if(ca[n]==0) printf("-1\n");
	else printf("%d\n",ca[n]);
	return 0;
}
