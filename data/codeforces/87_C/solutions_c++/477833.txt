#include <iostream>
#include <cstdio>

#define MAX_N 100005
#define ROOT_N 405

using namespace std;

int n,i,j,k,l;
int tmp;
int num[MAX_N];
int answer[MAX_N];
int is[MAX_N];
int a,q;

int main()
{
	cin>>n;
	for(l=1;l<=n;l++)
	{
		i=2*l;
		for(j=2;j*j<=i;j++)
		{
			if(i%j==0)
			{
				q=i/j;
				if((q-j)%2==1)
				{
					a=(q-j+1)/2;
					tmp=0;
					for(k=a;k<a+j;k++)
					{
						tmp=(tmp^num[k]);
					}
					if(tmp==0&&answer[l]==0)
					{
						answer[l]=j;
					}
					is[tmp]=l;
				}
			}
		}
		for(i=0;is[i]==l;i++)
		{
		}
		num[l]=i;
	}
	if(answer[n]!=0)
		printf("%d\n",answer[n]);
	else
		printf("-1\n");
	return 0;
}
