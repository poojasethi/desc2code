#include <iostream>
#include <algorithm>
using namespace std;
#define N 100010
int i,k,n,j,f[N]={0},s[N]={0},ii,x,r[N]={-1},rc,m[N]={0};

int main() {
    cin >> n;
    for (i=1;i<=n;i++) {
        ii=i<<1;
        rc = 1;
        for (k=2;k<=ii;k++) {
            x = ii/k-k+1;
            if (x<2) break;
            if (x&1) continue;
            if (ii%k>0) continue;
            x/=2;
            r[rc]=s[x+k]^s[x];
            if (r[rc]==0&&!m[i]) m[i]=k;
            rc++;
        }
        sort(r,r+rc);
        for (j=0;j<rc-1;j++) {
            if (r[j]==r[j+1]) continue;
            if (r[j]+1!=r[j+1]) break;
        }
        f[i]=r[j]+1;
        s[i+1]=s[i]^f[i];
    }
    if (f[n]) cout << m[n];
    else cout << -1;
    return 0;
}
