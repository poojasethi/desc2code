#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<string>
#include<queue>
using namespace std;

int sg[100005],s[100005];

int SG(int x)
{
	int i,k;
	if(sg[x]!=-1) return sg[x];
	int vis[500]={0};
	for(k=2;k*(k-1)/2<x;k++) if((x-k*(k-1)/2)%k==0)
	{
		i=(x-k*(k-1)/2)/k;
		vis[s[i+k-1]^s[i-1]]=1;
	}
	for(i=0;;i++) if(!vis[i]) return sg[x]=i;
}

int main()
{
	int n,i,k;
	memset(sg,-1,sizeof(sg));
	cin>>n;
	for(i=1;i<=n;i++) s[i]=s[i-1]^SG(i);
	if(sg[n])
	{
		for(k=2;k*(k-1)/2<n;k++) if((n-k*(k-1)/2)%k==0)
		{
			i=(n-k*(k-1)/2)/k;
			if((s[i+k-1]^s[i-1])==0) break;
		}
		cout<<k<<endl;
	}
	else cout<<-1<<endl;
}
