#include<iostream>
using namespace std;
int g[100001];

int game(int n)
{
  int ans = -1;
  bool mex[50];
  for (int i = 3; i <= n; i++) {
    for (int k = 0; k < 50; k++) mex[k] = false;
    for (int k = 2; ; k++) {
      int x = i - k*(k-1)/2;
      if (x < 0) break;
      if (x%k) continue;
      x /= k;
      int sg = 0;
      for (int j = 0; j < k; j++) sg ^= g[x+j];
      if (!sg && i == n && ans < 0) ans = k;
      mex[sg] = true;
    }
    while (mex[g[i]]) ++g[i];
  }
  return ans;
}
int main()
{
  int n;
  cin>>n;
  cout<<game(n)<<endl;
  return 0;
}
