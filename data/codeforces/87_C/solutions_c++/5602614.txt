#include <iostream>
#include <set>

using namespace std;

const int MAXN = 1000 * 100 + 10;
int nim[MAXN], pxr[MAXN], ans[MAXN];
set<int> all;

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i <= n; i++) ans[i] = -1;
	for (int i = 3; i <= n; i++)
	{
		all.clear();
		for (int j = 2; j * (j + 1) <= 2 * i; j++)
		{
			int cur = i - j * (j + 1) / 2;
			if (cur % j == 0)
			{
				int l = cur / j, r = l + j;
				int tmp = pxr[r] ^ pxr[l];
				all.insert(tmp);
				if (tmp == 0 && ans[i] == -1) ans[i] = j;
			}
		}
		while (all.find(nim[i]) != all.end()) nim[i]++;
		pxr[i] = pxr[i - 1] ^ nim[i];
	}
	cout << ans[n] << endl;
	return 0;
}
