#include <algorithm>
#include <stdio.h>
#define  maxn      100003
using	 namespace std;

int n;
int dp[maxn];
int ans[maxn];
int mark[maxn];

void f(int X)
{
  ans[X]=-1;
  for(int i=2 ; (i*(i+1))/2<=X ; i++)
    if((X-(i*(i+1))/2)%i==0)
    {
	  int bas=(X-(i*(i+1))/2)/i+1;
	  int st=dp[bas+i-1]^dp[bas-1];
	  mark[st]=X;
	  if(!st && ans[X]==-1) ans[X]=i;
	}
  for(int i=0 ; ; i++)
  {
    if(mark[i]==X) continue;
    dp[X]=dp[X-1]^i;
    return;
  }
}

int main()
{
  scanf("%d",&n);
  for(int i=1 ; i<=n ; i++)
    f(i);
  printf("%d\n",ans[n]);
  return 0;
}
