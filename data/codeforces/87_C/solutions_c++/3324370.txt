#include <iostream>
#include <set>
#include <cmath>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n,grundy[100010],res=-1;

int main(){

  cin>>n;
  for(int i=3;i<=n;i++){
    set<int> s;
    int p = 0;
    for(int j=2;;j++){
      int sum = i-j*(j+1)/2;
      int ng  = 0;
      if(sum<0)break;
      if(sum%j)continue;
      for(int k=1;k<=j;k++)ng^=grundy[sum/j+k];
        if(i==n && !ng && res<0){
        res = j;
      }
      s.insert(ng);
    }
    while(s.count(p))p++;
    grundy[i] = p;
  }
  cout<<res<<endl;

}
