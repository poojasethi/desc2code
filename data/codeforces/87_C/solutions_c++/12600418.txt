#include <iostream>
#include <cstdio>
#include <algorithm>
#include <set>
using namespace std;
const int maxn = 1e5 + 5;
int n;
int sg[maxn];
int ans = maxn;
set <int> s[maxn];
void grundy(int x)
{
	int g = 0;
	while (s[x].count(g) != 0)
		g++;
	sg[x] = g;
	int temp = g;
	int sum = x;
	for (int i = 2; i <= x; i++)
	{
		sum += x - i + 1;
		if (sum > n)
			break;
		temp ^= sg[x - i + 1];
		s[sum].insert(temp);
		if (sum == n && temp == 0)
			ans = min(ans, i);
	}
}
int main()
{
	//freopen("test.in", "r", stdin);
	cin >> n;
	for (int i = 2; i <= n; i++)
		grundy(i);
	if (!sg[n])
		cout << -1 << endl;
	else
		cout << ans << endl;
	return 0;
}
