#include <iostream>
#include <algorithm>
#include <math.h>
using namespace std;
int G[100001];
int X[1000];
int main(int argc, char* argv[])
{
   int r,a,i,j,p,t,m,n,c,v;
   cin>>n;
   for (i=1;i<=n;i++) {
      t=2*i;
      c=0;
      p=sqrt((double)(t+1));
      v=-1;
      for (m=2;m<=p;m++)
         if (t%m==0 && (t/m-m+1)%2==0) {
            a=(t/m-m+1)/2;
            if (a>0) {
               r=0;
               for (j=0;j<m;j++)
                  r=r^G[a+j];
            X[c++]=r;
            if (r==0 && v==-1)
               v=m;
            }
         }
      sort(X,X+c);
      r=0;
      for (j=0;j<c && X[j]<=r;j++)
         if (X[j]==r)
            r++;
      G[i]=r;
   }
   cout<<v;
   return 0;
}
