#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,num,F;
int sg[N],f[N],h[N];
int main() {
	scanf("%d",&n);
	For(i,1,n) {
		++num;
		for (k=2;;k++) {
			int A=k*(k+1)/2;
			if (A>i) break;
			int B=(i-A)%k;
			if (B) continue;
			B=(i-A)/k;
			f[h[k+B]^h[B]]=num;
			if (i==n&&h[k+B]==h[B]&&!F) F=k;
		}
		for (;f[sg[i]]==num;) sg[i]++;
		h[i]=h[i-1]^sg[i];
	}
	if (!sg[n]) printf("-1\n");
	else printf("%d\n",F);
	return 0;
}
