#include <cstdio>
using namespace std;

int N,used[100001],dp[100001],dpc[100001],r[100001];
int main () {
    scanf("%d",&N);
    for (int i=3;i<=N;++i) {
        for (int j=2;2*i>=j*(j+1);++j) if ((i-j*(j+1)/2)%j==0) {
            int x=(i-j*(j+1)/2)/j,d=dpc[x+j]^dpc[x];
            used[d]=i;
            if (!r[i] && !d) r[i]=j;
        }
        while (used[dp[i]]==i) ++dp[i];
        dpc[i]=dpc[i-1]^dp[i];
    }
    if (!dp[N]) printf("-1");
    else printf("%d\n",r[N]);
    return 0;
}
