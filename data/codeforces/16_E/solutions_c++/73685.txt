#include <stdio.h>
#define N 20

int n;
double f[ N ][ N ], ans[ 1 << (N) ];

void input()
{
	scanf("%d\n", &n);
	int i, j;
	for( i = 0; i < n; i++ )
	for( j = 0; j < n; j++ ) 
		scanf("%lf", &f[ i ][ j ]);
}

void solve()
{
	int i, j, k;
	ans[ ( 1 << n ) - 1 ] = 1.0;
	for( k = ( 1 << n ) - 1; k > 0; k-- ) {
		int c = 0;
		for( i = 0; i < n; i++ ) if( k & (1<<i) ) c++;
		c = c*(c-1)/2;
		for( i = 0; i < n; i++ ) {
			if( k & (1<<i) ) {
				for( j = 0; j < n; j++ ) {
					if( k & (1<<j) ) ans[ k ^ (1<<i) ] += ans[ k ]*f[ j ][ i ]/c;
				}
			}
		}
	}
	
	for( i = 0; i < n; i++ ) printf("%lf ", ans[ 1<<i ]);
	putchar('\n');	
}

int main()
{
	input();
	solve();
	return 0;
}
