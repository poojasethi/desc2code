#include<stdio.h>
double p[20][20];
double dp[(1<<19)];
int n;
int main()
{
    scanf("%d",&n);
    dp[(1<<n)-1]=1.0;
    for(int i=0;i<n;i++)
    for(int j=0;j<n;j++)
    scanf("%lf",&p[i][j]);
    for(int mask=(1<<n)-1;mask>0;mask--)
    {
        int sum=0;
        for(int i=0;i<n;i++)
            if((1<<i)&mask)
                sum++;
        for(int i=0;i<n;i++)
        if((1<<i)&mask)
        for(int j=0;j<n;j++)
        if((1<<j)&mask)
        if(i!=j)
            dp[mask-(1<<i)]+=dp[mask]*p[j][i]*2.0/sum/(sum-1);
    }
    for(int i=0;i<n;i++)
    printf("%f ",dp[(1<<i)]);
}
