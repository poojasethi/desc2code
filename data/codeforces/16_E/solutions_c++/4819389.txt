//http://codeforces.com/problemset/problem/16/E
#include <iostream>
#include<cstdio>
using namespace std;

int n, i, j, m, c;
double prob[18][18], state[1<<18];

int main()
{
    cin >> n;
    for( i = 0; i < n; i++ ) for( j = 0; j < n; j++ ) cin >> prob[i][j];
    state[(1<<n)-1] = 1;
    for( m = (1<<n)-1; m > 0; m-- )
    {
	c = 0;
	for( i = 0; i < n; i++ ) if( (m>>i)&1 ) c++;
	for( i = 0; i < n; i++ ) if( (m>>i)&1 )
	    for( j = 0; j < n; j++ ) if( (m>>j)&1 )
		if( i!=j )
		    state[m-(1<<i)] += state[m]*prob[j][i]*2/c/(c-1);
    }
    for( i = 0; i < n; i++ ) printf("%.6f ",state[1<<i]);
    printf("\n");
    return 0;
}
