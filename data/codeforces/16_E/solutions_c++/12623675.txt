# include <bits/stdc++.h>

using namespace std;

const int maxn = (1 << 19);
const int maxl = 19;

double dp[maxn];
double a[maxl][maxl];

int main(){
	int n;
	cin >> n;
	for(int i=0 ; i<n ; i++)
		for(int j=0 ; j<n ; j++)
			cin >> a[i][j];
	
	dp[(1 << n) - 1] = 1.0;
	for(int i=(1 << n)-1 ; i>0 ; i--){
		int pop = 0;
		for(int j=0 ; j<n ; j++)
			if(i & (1 << j))
				pop++;
		
		for(int j=0 ; j<n ; j++)
			for(int k=0 ; k<n ; k++)
				if((i & (1 << j)) and (i & (1 << k)) and i - j)
					dp[i - (1 << k)] += dp[i] * a[j][k] * (1.0 / double(pop * (pop - 1) / 2));
	}
	
	cout.precision(9);
	for(int i=0 ; i<n ; i++)
		cout << fixed << dp[(1 << i)] <<" ";
	cout << endl;

	return 0;
}
