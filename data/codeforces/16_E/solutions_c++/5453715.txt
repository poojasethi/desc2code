#include <iostream>
#include <cstdio>

using namespace std;

const int maxN = 19;

double a[maxN][maxN], prob[1 << maxN];

int main()
{
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
		for(int j = 0; j < n; j++)
			cin >> a[i][j];
	prob[0] = 1;
	for(int i = 0; i < 1 << n; i++)
	{
		int t0 = 0;
		for(int j = 0; j < n; j++)
			t0 += !((i >> j) & 1);
		for(int j = 0; j < n; j++)
			if(!((i >> j) & 1))
				for(int k = 0; k < n; k++)
					if(k !=j && !((i >> k) & 1))
						prob[i | (1 << j)] += (double)prob[i] * a[k][j] / (t0 * (t0 - 1) / 2);
		//cerr << i << " : " << t0 << " " << prob[i] << endl;
	}
	for(int i = 0; i < n; i++)
		printf("%6f ", prob[((1 << n) - 1) ^ 1 << i]);
	cout << endl;
	return 0;
}
