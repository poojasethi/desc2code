#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<cmath>
using namespace std;

int n ;
double a[20][20] , dp[(1<<20)] ;

int main()
{
    scanf("%d",&n) ;
    for(int i=0 ; i<n ; i++) for(int j=0 ; j<n ; j++) scanf("%lf",&a[i][j]) ;

    dp[(1<<n)-1]=1.0 ;
    for(int i=(1<<n)-2 ; i>0 ; i--)  
    { 
        int x=0 ; 
        for(int k=0 ; k<n ; k++) if(i&(1<<k)) x++ ;
    
        for(int j=0 ; j<n ; j++) if((i&(1<<j))==0)
          for(int k=0 ; k<n ; k++) if(i&(1<<k)) 
            dp[i]+=2.0*dp[i|(1<<j)]*a[k][j]/x/(x+1) ;
    } 
 
    for(int i=0 ; i<n ; i++) printf("%.6lf ",dp[(1<<i)]) ;

 return 0 ; 
}
