#include <stdio.h>
#include <string.h>
#define maxn 18
double a[maxn][maxn];
double f[1<<maxn];
int n;
int calc(int x){
	int ret=0;
	while (x) ret+=(x&1),x>>=1;
	return ret;
}
int C(int n){
	return n*(n-1)>>1;
}
int main(){
	scanf("%d",&n);
	for (int i=0;i<n;i++)
		for (int j=0;j<n;j++) scanf("%lf",&a[i][j]);
	f[(1<<n)-1]=1;
	for (int mask=((1<<n)-1);mask;mask--){
		int k=calc(mask);
		if (k==1) continue;
		for (int i=0;i<n;i++) if ((mask&(1<<i))>0)
			for (int j=i+1;j<n;j++) if (i!=j && (mask&(1<<j))>0){
				f[mask^(1<<j)]+=f[mask]*a[i][j]/C(k);
				f[mask^(1<<i)]+=f[mask]*a[j][i]/C(k);
			}
	}
	for (int i=0;i<n;i++) printf("%.8lf ",f[1<<i]);
	puts("");
	return 0;
}
