#include <bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	double a[n][n];
	for (int i = 0; i < n; i++) for (int j = 0; j < n; j++) cin>>a[i][j];
	int x = (1 << n);
	double dp[x];
	for (int i = 0; i < x; i++) dp[i] = 0; 
	dp[x - 1] = 1;
	for (int i = x - 1; i >= 0; i--) {
		vector<int> v; 
		double count = 0;
		for (int j = 0; j < n; j++) if((i >> j) & 1) v.push_back(j), count++;
		count = (count * (count - 1) / 2.0); 
		for (int j = 0; j < v.size(); j++) {
			for (int k = j + 1; k < v.size(); k++) {
				dp[i & ~(1 << v[j])] += dp[i] * a[v[k]][v[j]] / count;
				dp[i & ~(1 << v[k])] += dp[i] * a[v[j]][v[k]] / count;	
			}		
		}	
	}  
	for (int i = 0; i < n; i++) printf("%.10lf ", dp[1 << i]);
	cout<<endl; 
}
