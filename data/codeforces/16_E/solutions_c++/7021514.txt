#include <cstdio>
#include <cstring>

using namespace std;

int n;
double g[20][20];
double f[1<<18];

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++) for(int j=1;j<=n;j++) scanf("%lf",&g[i][j]);
	f[(1<<n)-1]=1;
	for(int i=(1<<n)-1;i;i--){
		int t=__builtin_popcount(i);if(t<=1) continue;
		t=t*(t-1)/2;
		for(int j=1;j<=n;j++) if(i&(1<<j-1))
			for(int k=1;k<=n;k++) if((i&(1<<k-1)) && j!=k)
				f[i^(1<<k-1)]+=f[i]*g[j][k]/t;
	}
	for(int i=0;i<n;i++) printf("%.6lf ",f[1<<i]);puts("");
	return 0;
}
