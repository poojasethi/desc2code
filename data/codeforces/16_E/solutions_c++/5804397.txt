#include<iostream>
#include<cstdio>
#include<string.h>
#include<algorithm>
using namespace std;

double p[20][20],live[1<<18];

int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
		    scanf("%lf",&p[i][j]);
	live[(1<<n)-1]=1;
	for(int state=(1<<n)-1;state>0;state--){
		int cnt=0;//活着个数
		for(int i=0;i<n;i++)
			if((state>>i)&1)
				cnt++;
		int all=cnt*(cnt-1)/2;//鱼对数-相遇概率
		for(int i=0;i<n;i++){
			if(((state>>i)&1)==0) continue;
			for(int j=0;j<n;j++){
				if(((state>>j)&1)==0) continue;
				if(i==j) continue;
				live[state-(1<<j)]+=live[state]*p[i][j]/all;
			}
		}
	}
	for(int i=0;i<n;i++)
		printf("%.6lf ",live[1<<i]);
	printf("\n");
	return 0;
}
