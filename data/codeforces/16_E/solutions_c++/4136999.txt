#include <cstdio>
#include <iostream>
using namespace std;

double p[32][32];
double t[1<<19];
int n;

double prob(int mask){
	if(t[mask]>-0.5)return t[mask];
	int aliv=1,i,j;
	for(i=0;i<n;++i){
		if((mask>>i)&1)
			++aliv;
	}
	double npair=(double)aliv*(aliv-1)/2.;
	t[mask]=0;
	for(i=0;i<n;++i){
		for(j=0;j<n;++j){
			if(((mask>>i)&1)&&!((mask>>j)&1)){
				t[mask]+=prob(mask|(1<<j))*p[i][j]/npair;
			}
		}
	}
	return t[mask];
}

int main(){
	int i,j;
	cin>>n;
	for(i=0;i<n;++i){
		for(j=0;j<n;++j){
			cin>>p[i][j];
		}
	}
	for(i=0;i<(1<<n);++i)
		t[i]=-1;
	t[(1<<n)-1]=1;
	for(i=0;i<n;++i){
		printf("%.6f ",prob(1<<i));
	}
	i=2;
	cout<<endl;
	return 0;
}
