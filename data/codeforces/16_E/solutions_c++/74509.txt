#include <iostream>
#include <cstdio>
using namespace std;

double A[1<<18];
int main(){
	int i,j,k;
	int n;
	int C[20];
	double M[20][20],d,f;
	
	scanf("%d",&n);
	for(i=0;i<n;i++){
		for(j=0;j<n;j++)
			scanf("%lf",&M[i][j]);
	}
	
	A[(1<<n)-1] = 1;
	for(i = (1<<n)-1;i > 0;i--){
		int c = 0;
		for(j=0;j<n;j++){
			if((i&(1<<j)) != 0){
				C[c] = j;
				c++;
			}
		}
		
		f = 2.0/(c*(c-1));
		for(j=0;j<c;j++){
			for(k=j+1;k<c;k++){
				A[i^(1<<C[k])] += (f*A[i]*M[C[j]][C[k]]);
				A[i^(1<<C[j])] += (f*A[i]*M[C[k]][C[j]]);
			}
		}
	}
	
	for(i=0;i<n;i++){
		if(i != 0) printf(" ");
		printf("%lf",A[1<<i]);
	}
	printf("\n");
	return 0;
}
