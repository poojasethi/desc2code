#include <cstdio>

using namespace std;

double p[18][18],f[1<<18];

int main(void)
{
  int n;
  scanf("%d",&n);
  for(int i=0;i<n;++i)
    for(int j=0;j<n;++j)
      scanf("%lf",&p[i][j]);
  f[(1<<n)-1]=1;
  for(int k=(1<<n)-1;k>0;k--)
  {
    int t=0,temp=k;
    while(temp>0)
    {
      t+=temp%2;
      temp/=2;
    }
    for(int i=0;i<n;++i)
    {
      if((k&(1<<i))==0)
        continue;
      for(int j=0;j<n;++j)
      {
        if((j==i)||((k&(1<<j))==0))
          continue;
        f[k-(1<<j)]+=f[k]*p[i][j]/(t*(t-1)/2.0);
      }
    }
  }
  for(int i=0;i<n-1;++i)
    printf("%.6lf ",f[1<<i]);
  printf("%.6lf\n",f[1<<(n-1)]);
  return 0;
}
