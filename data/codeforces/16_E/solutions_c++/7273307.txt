#include "stdio.h"
double b[1<<18],a[18][18];
int main(){
	int i,j,k,n;scanf("%d",&n);
	for(i=0;i<n;i++)for(j=0;j<n;j++)scanf("%lf",&a[i][j]);
	int all=1<<n;b[all-1]=1;
	for(k=all-1;k>=0;k--)
	{	int siz=0;
		for(i=0;i<n;i++)siz+=(k>>i&1);
		for(i=0;i<n;i++)
			if(k>>i&1)
				for(j=0;j<n;j++)
					if(j!=i&&(k>>j&1))
						b[k^(1<<j)]+=b[k]*a[i][j]/((siz-1)*siz/2);}
	for(i=0;i<n;i++)printf("%.6lf%c",b[1<<i],i==n-1?'\n':' ');
	return 0;}
