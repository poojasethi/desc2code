#include<cstdio>
#define N 20
double a[N+5][N+5];
bool used[(1<<N)+5];
double dp[(1<<N)+5];
int n;
double dfs(int mask){
    if(mask==(1<<n)-1)return 1.0;
    if(used[mask])return dp[mask];
    used[mask]=true;
    double ans=0.0;
    int cnt=0;
    int p=__builtin_popcount(mask);
    for(int i = 0 ; i < n ; ++i )
        if( mask&(1<<i) ) 
            for( int j = 0 ; j < n ; ++j )
                if( !(mask&(1<<j)) )ans += 2*dfs(mask xor (1<<j) )*a[i][j]/(p*(p+1));
    ans;
    return dp[mask]=ans;
}
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;++i)
        for(int j=0;j<n;++j)scanf("%lf",a[i]+j);
    for(int i=0;i<n;++i)printf("%.6lf\n",dfs(1<<i));
}
