#include <cstdio>
#include <algorithm>
using namespace std;

double p[1<<18];
double a[18][18];

int main() {
	int n;

	scanf("%d", &n);
	for (int i=0; i<n; ++i) for (int j=0; j<n; ++j) scanf("%lf", a[i]+j);
	int t = (1 << n) - 1;
	p[t] = 1;
	for (int k=t; k; --k) {
		int x = __builtin_popcount(k);
		x = x * (x - 1) >> 1;
		for (int i=0; i<n; ++i) if (k & 1 << i) for (int j=i+1; j<n; ++j) if (k & 1 << j) {
			p[k & ~(1 << j)] += p[k] * a[i][j] / x, p[k & ~(1 << i)] += p[k] * a[j][i] / x;
		}
	}
	for (int i=0; i<n; ++i) printf("%.16lf%s", p[1<<i], i==n-1?"\n":" ");

	return 0;
}
