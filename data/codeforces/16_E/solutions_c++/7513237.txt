#include <cstdio>
using namespace std;
double pro[20][20];
double f[1<<20];
inline int Get_Fish(int x)
{
	int t=0;
	while (x)
	{
		if (x&1) t++;
		x>>=1;
	}
	return t;
}
int main()
{
	int n;
	scanf("%d",&n);
	for (int i=0;i<n;i++)
		for (int j=0;j<n;j++) scanf("%lf",&pro[i][j]);
	f[(1<<n)-1]=1.0;
	for (int i=(1<<n)-1;i;i--)
	{
		int fish=Get_Fish(i);
		if (fish==1) continue;
		double C=2.0*f[i]/fish/(fish-1);
		for (int j=0;j<n;j++)
			if ((1<<j)&i)
				for (int k=0;k<n;k++)
					if ((1<<k)&i)
						f[i-(1<<k)]+=pro[j][k]*C;
	}
	for (int i=0;i<n;i++)
		printf("%.6lf ",f[1<<i]);
	return 0;
}
