#include <iostream>
using namespace std;
int n;
double g[32][32],t[1<<18];
int bits(int x){
  int b=0;
  while(x)b+=x&1,x>>=1;
  return b;
}
main(){
  cin>>n;
  for(int i=0;i<n;++i)for(int j=0;j<n;++j)cin>>g[i][j];
  t[(1<<n)-1]=1;
  for(int i=(1<<n);i-->0;)if(i!=(i&-i)){
    int d=bits(i);
    d=d*(d-1)/2;
    for(int x=0;x<n;++x)if((1<<x)&i)for(int y=0;y<n;++y)if((1<<y)&i)t[i^(1<<y)]+=t[i]*g[x][y]/d;
  }
  for(int i=0;i<n;++i)cout<<t[1<<i]<<" ";cout<<endl;
}
