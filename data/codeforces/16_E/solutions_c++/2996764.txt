#include<stdio.h>
double p[20][20];
int n;
double dp[(1<<19)];
int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    for(int j=0;j<n;j++)
    scanf("%lf",&p[i][j]);
    dp[0]=1.0;
    for(int mask=0;mask<(1<<n);mask++)
    {
        int sum=0;
        for(int i=0;i<n;i++)
            if(((1<<i)&mask)==0)
                sum++;
        for(int i=0;i<n;i++)
        if(((1<<i)&mask)==0)
        for(int j=0;j<n;j++)
        if(((1<<j)&mask)==0)
        if(i!=j)
        dp[mask+(1<<i)]+=dp[mask]*p[j][i]*2/sum/(sum-1);
    }
    for(int i=0;i<n;i++)
    printf("%.9f ",dp[(1<<n)-1-(1<<i)]);
}
