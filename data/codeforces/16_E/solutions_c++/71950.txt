#include <cstdio>

int n, w[300000];
double d[300000], a[20][20];

double get (int u)
{
	if (u == (1 << n) - 1) return 1;
	if (w[u]) return d[u];
	w[u] = 1;
	double ret = 0;
	int v1 = 0;

	for ( int i = 0; i < n; i++ )
		if ((1 << i) & u)
			v1++;

	v1 = v1*(v1 + 1) / 2;

	for ( int i = 0; i < n; i++ ) if ((1 << i) & u) for ( int j = 0; j < n; j++ )
		if (!((1 << j) & u))
			ret += get (u | (1 << j))*a[i][j];

	d[u] = ret / v1;

	return d[u];
}

int main ()
{
	scanf ( "%d", &n );

	for ( int i = 0; i < n; i++ )
		for ( int j = 0; j < n; j++ )
			scanf ( "%lf", &a[i][j] );

	for ( int i = 0; i < n; i++ )
		printf ( "%.6lf ", get (1 << i) );

	return 0;
}
