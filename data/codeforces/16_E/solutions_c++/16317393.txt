//in the name of God

#include <iostream>
#include <iomanip>
using namespace std;

int n;
double a[20][20], p[1<<20];

int main()
{
	ios_base::sync_with_stdio(false);

	cin >> n;
	for(int i=0; i<n; i++)
		for(int j=0; j<n; j++)
			cin >> a[i][j];


	p[(1<<n)-1] = 1.0;
	for(int b=(1<<n)-2; b; b--)
	{
		for(int i=0; i<n; i++)
		{
			if((b & (1<<i)) == 0)
			{
				int nb = b | (1<<i);
				for(int j=0; j<n; j++)
				{
					if(b & (1<<j))
					{
						p[b] += p[nb]*a[j][i];
					}
				}
			}
		}

		int x = __builtin_popcount(b)+1;
		p[b] /= x*(x-1)/2;
	}

	cout << fixed << setprecision(9);
	for(int i=0; i<n; i++)
		cout << p[1<<i] << ' ';
	cout << endl;

	return 0;
}
