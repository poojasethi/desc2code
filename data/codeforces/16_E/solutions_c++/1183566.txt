#include<cstdio>
#include<cstring>
const int N = 18;
double dp[1<<N],p[N][N];
int main()
{
	int n,i,j,k,l;
	scanf("%d",&n);
	for(i=0;i<n;i++)for(j=0;j<n;j++)scanf("%lf",&p[i][j]);
	l=(1<<n);
	memset(dp,0,sizeof(dp));
	dp[l-1]=1.0;
	int cnt;
	for(i=l-1;i>0;i--)
	{
		cnt=0;
		for(j=0;j<n;j++)
					if(i&(1<<j))
				{
					for(k=j+1;k<n;k++)
						if((i&(1<<k)))
					{
						cnt++;
					}
				}
		for(j=0;j<n;j++)
			if(i&(1<<j))
		{
			for(k=j+1;k<n;k++)
				if((i&(1<<k)))
			{
				dp[i-(1<<j)]+=p[k][j]*dp[i]/(double)cnt;
				dp[i-(1<<k)]+=p[j][k]*dp[i]/(double)cnt;
			}
		}
	}
	for(i=0;i<n-1;i++)printf("%lf ",dp[1<<i]);printf("%lf\n",dp[1<<i]);
	return 0;
}
