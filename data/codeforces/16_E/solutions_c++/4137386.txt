#include <iostream>
#include <cstdio>
#define forn(i,n) for(int i=0;i<n;i++)
using namespace std;

double dp[1<<18];
int N;
double p[18][18];

double f(int mask){
    if (dp[mask] > -0.5) return dp[mask];
    dp[mask]=0;
    int vivos = 1;
    forn(i,N) if((mask>>i)%2==1) vivos++;
    double pares = (vivos*(vivos-1))/2;
    forn(i,N) forn(j,N){
        if ((mask & (1<<i))!=0 && (mask&(1<<j))==0)
            dp[mask] += f(mask | (1<<j)) * p[i][j] / pares;
    }
    return dp[mask];
}

int main (){
    cin >> N;
    forn(i,N)
        forn(j,N)
            cin >> p[i][j];
    forn(i,(1<<N))
        dp[i]=-1;
    dp[(1<<N)-1] = 1.;
    forn(i,N)
        printf(" %6lf\n",f(1<<i));
    return 0;
}
