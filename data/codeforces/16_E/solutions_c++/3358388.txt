#include <iomanip>
#include <iostream>

using namespace std;

const int max_n = 20;
const int max_val = 1 << max_n;

int n;
double dp[max_val], p[max_n][max_n];

int main()
{
  cin >> n;
  for (int i = 0; i < n; i++)
    for (int j = 0; j < n; j++) cin >> p[i][j];
  dp[(1 << n) - 1] = 1;
  for (int mask = (1 << n) - 1; mask >= 0; mask--)
  {
    int cnt = __builtin_popcount(mask);
    for (int i = 0; i < n; i++) if (mask & 1 << i)
      for (int j = 0; j < n; j++) if (mask & 1 << j)
        dp[mask - (1 << j)] += 2 * dp[mask] * p[i][j] / (cnt * (cnt - 1));
  }
  for (int i = 0; i < n; i++)
    cout << fixed << setprecision(10) << dp[1 << i] << " ";
  return 0;
}
