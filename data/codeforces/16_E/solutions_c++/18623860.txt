#include <bits/stdc++.h>

using namespace std;

long double A[19][19], bitmask[1<<19];

int n;

int num_ones(int x){
	int c=0;
	for (;x;x >>= 1) if(x&1) c++;
	return c;
}

int main(void) {
	cin>>n;
	for(int i=0; i<n; i++){
		for(int j=0; j<n; j++) cin>>A[i][j];
	} 
	
	bitmask[(1 << n) - 1] = 1;
	
	for(int k=(1 << n) - 1; k>=0; k--){
		int num_fish = num_ones(k);
		
		for(int i=0; i < n; i++){
			if((k>>i) & 1){
				for(int j=0; j<n; j++){
					if((k>>j) & 1 && i!=j){
						bitmask[k-(1<<i)] += bitmask[k]*A[j][i] / ((num_fish-1)*num_fish/2);
					}
				}
			}
		}
		
	}
	cout << setprecision(9);
	for (int i = 0; i < n; ++i) cout<<bitmask[1<<i]<<" ";

	return 0;
}
