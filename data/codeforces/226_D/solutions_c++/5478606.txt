#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#define N 120
using namespace std;
int n,m,a[N][N],row[N],col[N],ans1[N],ans2[N],i,j;
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i)
		for(j=1;j<=m;++j){
			scanf("%d",&a[i][j]);
			row[i]+=a[i][j];
			col[j]+=a[i][j];
		}
	for(;;){
		bool t=0;
		for(i=1;i<=n;++i)if(row[i]<0){
			ans1[i]^=1;
			for(j=1;j<=m;++j){
				col[j]-=2*a[i][j];
				a[i][j]=-a[i][j];
			}
			row[i]=-row[i];
			t=1;
		}
		for(j=1;j<=m;++j)if(col[j]<0){
			ans2[j]^=1;
			for(i=1;i<=n;++i){
				row[i]-=2*a[i][j];
				a[i][j]=-a[i][j];
			}
			col[j]=-col[j];
			t=1;
		}
		if(!t)break;
	}
	int p=0;
	for(i=1;i<=n;++i)if(ans1[i])p++;
	printf("%d ",p);
	for(i=1;i<=n;++i)if(ans1[i])printf("%d ",i);
	puts("");
	p=0;
	for(j=1;j<=m;++j)if(ans2[j])p++;
	printf("%d ",p);
	for(j=1;j<=m;++j)if(ans2[j])printf("%d ",j);
}
