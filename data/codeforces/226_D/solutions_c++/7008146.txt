#include<cstdio>
#define N 110
#define rep(i, n)	for(int i = 0; i < n; i ++)
int a[N][N], sr[N], sc[N];
bool fr[N], fc[N];

int n, m;

int main() {
	scanf("%d%d", &n, &m);
	rep(i, n)
		rep(j, m) {
			scanf("%d", &a[i][j]);
			sr[i] += a[i][j];
		}
	rep(j, m)
		rep(i, n)
			sc[j] += a[i][j];
	bool flag = true;
	while(flag) {
		flag = false;
		rep(i, n) {
			if(sr[i] < 0) {
				flag = true;
				sr[i] *= -1;
				fr[i] = !fr[i];
				rep(j, m) {
					sc[j] -= 2 * a[i][j];
					a[i][j] *= -1;
				}
			}
		}
		rep(j, m) {
			if(sc[j] < 0) {
				flag = true;
				sc[j] *= -1;
				fc[j] = !fc[j];
				rep(i, n) {
					sr[i] -= 2 * a[i][j];
					a[i][j] *= -1;
				}
			}
		}
	}
	int ct, ct2;
	ct = ct2 = 0;
	rep(i, n)	ct += fr[i];
	rep(j, m)	ct2 += fc[j];
	printf("%d ", ct);
	rep(i, n)if(fr[i]) {
		printf("%d ", i + 1);
	}
	puts("");
	printf("%d ", ct2);
	rep(j, m)if(fc[j]) {
		printf("%d ", j + 1);
	}
	puts("");
	return 0;
}
