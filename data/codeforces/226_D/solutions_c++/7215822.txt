#include <iostream>

using namespace std;

const int MAXN = 101;
int c[MAXN], r[MAXN], a[MAXN][MAXN];
bool markC[MAXN], markR[MAXN];

int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			cin >> a[i][j], r[i] += a[i][j], c[j] += a[i][j];
	bool found = true;
	while (found) {
		found = false;
		for (int i = 0; i < n && !found; i++)
			if (r[i] < 0) {
				for (int j = 0; j < m; j++)
					c[j] -= 2 * a[i][j], a[i][j] *= -1;
				r[i] *= -1;
				markR[i] = !markR[i];
				found = true;
			}
		for (int j = 0; j < m && !found; j++)
			if (c[j] < 0) {
				for (int i = 0; i < n; i++)
					r[i] -= 2 * a[i][j], a[i][j] *= -1;
				c[j] *= -1;
				markC[j] = !markC[j];
				found = true;
			}
	}
	int ansR = 0, ansC = 0;
	for (int i = 0; i < n; i++)
		ansR += markR[i];
	for (int j = 0; j < m; j++)
		ansC += markC[j];
	cout << ansR;
	for (int i = 0; i < n; i++)
		if (markR[i])
			cout << " " << i + 1;
	cout << endl;
	cout << ansC;
	for (int i = 0; i < m; i++)
		if (markC[i])
			cout << " " << i + 1;
	cout << endl;
	return 0;
}
