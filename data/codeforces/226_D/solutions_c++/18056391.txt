#include<cstdio>
#define N 110
using namespace std;

int n,m,a[N][N],sum0[N],sum1[N],vis0[N],vis1[N];

int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			scanf("%d",&a[i][j]);
			sum0[i]+=a[i][j];
			sum1[j]+=a[i][j];
		}
	}
	while(true){
		int get=0;
		for(int i=1;i<=n;i++){
			if(sum0[i]<0){
				get++;
				sum0[i]=-sum0[i];
				vis0[i]^=1;
				for(int j=1;j<=m;j++){
					sum1[j]-=2*a[i][j];
					a[i][j]=-a[i][j];
				}
			}
		}
		for(int j=1;j<=m;j++){
			if(sum1[j]<0){
				get++;
				sum1[j]=-sum1[j];
				vis1[j]^=1;
				for(int i=1;i<=n;i++){
					sum0[i]-=2*a[i][j];
					a[i][j]=-a[i][j];
				}
			}
		}
		if(get==0) break;
	}
	int cnt0=0,cnt1=0;
	for(int i=1;i<=n;i++) cnt0+=vis0[i];
	for(int j=1;j<=m;j++) cnt1+=vis1[j];
	printf("%d ",cnt0);
	for(int i=1;i<=n;i++) if(vis0[i]) printf("%d ",i);
	printf("\n%d ",cnt1);
	for(int j=1;j<=m;j++) if(vis1[j]) printf("%d ",j);
	return 0;
}
