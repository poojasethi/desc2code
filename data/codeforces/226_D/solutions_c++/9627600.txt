#include <bits/stdc++.h>
using namespace std;

int n , m;
int matrix[110][110];


bool a[110], b[110];

int rw[110] , col[110];

inline void solve()
{
  memset( rw  , 0 , sizeof rw);
  memset( col, 0 , sizeof col);
      
  for(int i=0; i<n; i++)
    for(int j=0; j<m; j++)
      rw[i] += matrix[i][j] , col[j] += matrix[i][j];



  for(int i=0; i<n; i++)
    if( rw[i] < 0)
      {
	a[i] = 1-a[i];
	for(int j=0; j<m; j++)matrix[i][j] *= -1;
	solve();
	return;
      }
  for(int i=0; i<m; i++)
    if( col[i] < 0)
      {
	b[i] = 1-b[i];
	for(int j=0; j<n; j++)matrix[j][i] *= -1;
	solve();
	return;
      }
  
}


int main() 
{
  cin>>n>>m;
  for(int i=0; i<n; i++)
    for(int j=0; j<m; j++)
      cin>>matrix[i][j];

  memset( a , 0, sizeof a);
  memset( b , 0, sizeof b);
  solve();
  int x , y = x = 0;
  for(int i=0; i<n; i++)if( a[i] == 1) x++;
  for(int i=0; i<m; i++)if( b[i] == 1) y++;
  cout<<x<<' ';
  for(int i=0; i<n; i++)if( a[i] == 1)cout<<i+1<<' ';
  cout<<endl<<y<<' ';
  for(int i=0; i<m; i++)if( b[i] == 1)cout<<i+1<<' ';
  cout<<endl;
  return 0;
}
