#include <cstdio>

using namespace std;

int a[101][101],sr[101],sc[101],fr[101],fc[101];

int main(void)
{
  int n,m;
  scanf("%d %d",&n,&m);
  for(int i=1;i<=n;++i)
  {
    for(int j=1;j<=m;++j)
    {
      scanf("%d",&a[i][j]);
      sr[i]+=a[i][j];
      sc[j]+=a[i][j];
    }
  }
  bool flag=false;
  while(!flag)
  {
    flag=true;
    for(int i=1;i<=n;++i)
    {
      if(sr[i]>=0)
        continue;
      flag=false;
      for(int j=1;j<=m;++j)
      {
        sc[j]-=2*a[i][j];
        a[i][j]=-a[i][j];
      }
      fr[i]^=1;
      sr[i]=-sr[i];
    }
    for(int j=1;j<=m;++j)
    {
      if(sc[j]>=0)
        continue;
      flag=false;
      for(int i=1;i<=n;++i)
      {
        sr[i]-=2*a[i][j];
        a[i][j]=-a[i][j];
      }
      fc[j]^=1;
      sc[j]=-sc[j];
    }
  }
  int ansr=0,ansc=0;
  for(int i=1;i<=n;++i)
    ansr+=fr[i];
  for(int i=1;i<=m;++i)
    ansc+=fc[i];
  printf("%d",ansr);
  for(int i=1;i<=n;++i)
    if(fr[i]>0)
      printf(" %d",i);
  printf("\n%d",ansc);
  for(int i=1;i<=m;++i)
    if(fc[i]>0)
      printf(" %d",i);
  printf("\n");
  return 0;
}
