#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;

const int MAX=200;
int n,m,g[MAX][MAX],sumr[MAX],sumc[MAX],cr[MAX],cc[MAX];

void rev(int i,int j)
{
	sumr[i]-=2*g[i][j];
	sumc[j]-=2*g[i][j];
	g[i][j]*=-1;
}

void changer(int i)
{
	cr[i]^=1;
	int j;
	for(j=1;j<=m;++j)
		rev(i,j);
}

void changec(int j)
{
	cc[j]^=1;
	int i;
	for(i=1;i<=n;++i)
		rev(i,j);
}

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
#endif
	int i,j,flag;
	scanf("%d %d",&n,&m);
	for(i=1;i<=n;++i)
		for(j=1;j<=m;++j)
		{
			scanf("%d",&g[i][j]);
			sumr[i]+=g[i][j];
			sumc[j]+=g[i][j];
		}
	flag=1;
	while(flag)
	{
		flag=0;
		for(i=1;i<=n;++i)
			if(sumr[i]<0)
			{
				flag=1;
				changer(i);
			}
		for(i=1;i<=m;++i)
			if(sumc[i]<0)
			{
				flag=1;
				changec(i);
			}
	}
	int tot=0;
	for(i=1;i<=n;++i)
		if(cr[i])++tot;
	printf("%d ",tot);
	for(i=1;i<=n;++i)
		if(cr[i])
			printf("%d ",i);
	printf("\n");
	tot=0;
	for(i=1;i<=m;++i)
		if(cc[i])++tot;
	printf("%d ",tot);
	for(i=1;i<=m;++i)
		if(cc[i])
			printf("%d ",i);
	printf("\n");
}
