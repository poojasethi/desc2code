#include <stdio.h>
int n,m,a[128][128],u[128],v[128];
void pr(int*p,int k){
  int z=0;
  for(int i=0;i<k;++i)if(p[i])++z;
  printf("%d",z);
  for(int i=0;i<k;++i)if(p[i])printf(" %d",i+1);
  puts("");
}
main(){
  scanf("%d%d",&n,&m);
  for(int i=0;i<n;++i)for(int j=0;j<m;++j)scanf("%d",&a[i][j]);
  for(;;){
    int ok=1;
    for(int i=0;i<n;++i){
      int s=0;
      for(int j=0;j<m;++j)s+=u[i]^v[j]?-a[i][j]:a[i][j];
      if(s<0)u[i]^=1,ok=0;
    }
    for(int j=0;j<m;++j){
      int s=0;
      for(int i=0;i<n;++i)s+=u[i]^v[j]?-a[i][j]:a[i][j];
      if(s<0)v[j]^=1,ok=0;
    }
    if(ok)break;
  }
  pr(u,n),pr(v,m);
}
