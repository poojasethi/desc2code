#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,a[110][110];
bool flag,A[110],B[110];
int rcalc(int i){
	int sum=0;
	for (int j=1;j<=m;j++) sum+=a[i][j];
	return sum;
}
int ccalc(int j){
	int sum=0;
	for (int i=1;i<=n;i++) sum+=a[i][j];
	return sum;
}
void print(bool *ans){
	int cnt=0;
	for (int i=1;i<=max(n,m);i++)
	  cnt+=ans[i];
	printf("%d",cnt);
	for (int i=1;i<=max(n,m);i++)
	  if (ans[i]==true)
	    printf(" %d",i);
	printf("\n");
}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++)
	  for (int j=1;j<=m;j++)
	    scanf("%d",&a[i][j]);
	while (true){
		flag=false;
		for (int i=1;i<=n;i++)
		  if (rcalc(i)<0){
			  for (int j=1;j<=m;j++) a[i][j]=-a[i][j];
			  A[i]^=1;flag=true;
		  }
		for (int j=1;j<=m;j++)
		    if (ccalc(j)<0){
		      for (int i=1;i<=n;i++) a[i][j]=-a[i][j];
		      B[j]^=1;flag=true;
		    }
		if (flag==false) break;
    }
    print(A);print(B);
    return 0;
}
