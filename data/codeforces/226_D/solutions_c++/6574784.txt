#include<cstdio>
using namespace std;

int v1[102],v2[102],s,a[102][102],flag,n,m,i,j;

void ptf(int a[],int n)
{
	int s=0;
	for(int i=1;i<=n;i++)
		s+=a[i];
	printf("%d ",s);
	for(int i=1;i<=n;i++)
		if (a[i]) printf("%d ",i);
	printf("\n");
}
int main()
{

	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			scanf("%d",&a[i][j]);

	while (1)
		{
			flag=0;
			for(i=1;i<=n;i++)
				{
					s=0;
					for(j=1;j<=m;j++)
						s+=(v1[i]^v2[j])?(-a[i][j]):a[i][j];
					if (s<0)
						{
							flag=1;
							v1[i]^=1;
						}
				}
			for(i=1;i<=m;i++)
				{
					s=0;
					for(j=1;j<=n;j++)
						s+=(v2[i]^v1[j])?(-a[j][i]):a[j][i];
					if (s<0)
						{
							flag=1;
							v2[i]^=1;
						}
				}
			if (!flag) break;
		}

	ptf(v1,n);
	ptf(v2,m);

	
	return 0;
}
