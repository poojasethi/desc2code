#include<cstdio>
#include<algorithm>
#include<string>
#include<iostream>
#include<set>
#include<cmath>
#define FOR(i,a,b) for(int i=(a);i<=(b);i++)
#define FORD(i,a,b) for (int i=(a);i>=(b);i--)

using namespace std;
typedef long long LL;
int n,m,i,j,a[110][110],sr[110],sc[110],ar[110],ac[110];
int main(){
  scanf("%d%d",&n,&m);
  FOR(i,1,n)FOR(j,1,m){
    scanf("%d",&a[i][j]);
    sr[i]+=a[i][j];
    sc[j]+=a[i][j];
  }
  bool kind=true;
  while(kind){
    kind=false;
    FOR(i,1,n)if(sr[i]<0){
      ar[i]^=1;
      FOR(j,1,m){
        sr[i]-=2*a[i][j];
        sc[j]-=2*a[i][j];
        a[i][j]=-a[i][j];
      }
      kind=true;
    }
    FOR(j,1,m)if(sc[j]<0){
      ac[j]^=1;
      FOR(i,1,n){
        sr[i]-=2*a[i][j];
        sc[j]-=2*a[i][j];
        a[i][j]=-a[i][j];
      }
      kind=true;
    }
  }
  int t=0;FOR(i,1,n)if(ar[i])++t;
  printf("%d ",t);
  FOR(i,1,n)if(ar[i])printf("%d ",i);printf("\n");
  t=0;FOR(j,1,m)if(ac[j])++t;
  printf("%d ",t);
  FOR(j,1,m)if(ac[j])printf("%d ",j);printf("\n");

  

return 0;}
