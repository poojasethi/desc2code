#include <cstdio>
const int N=105;

int n,m,a[N][N],sx[N],sy[N],fx[N],fy[N];

bool work(){
	for(int i=0;i<n;i++) if(sx[i]<0){
		fx[i]^=1;
		for(int j=0;j<m;j++){
			sx[i]-=a[i][j];
			sy[j]-=a[i][j];
			a[i][j]*=-1;
			sx[i]+=a[i][j];
			sy[j]+=a[i][j];
		}
		return true;
	}
	for(int j=0;j<m;j++) if(sy[j]<0){
		fy[j]^=1;
		for(int i=0;i<n;i++){
			sx[i]-=a[i][j];
			sy[j]-=a[i][j];
			a[i][j]*=-1;
			sx[i]+=a[i][j];
			sy[j]+=a[i][j];
		}
		return true;
	}
	return false;
}

int main(){
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++){
			scanf("%d",&a[i][j]);
			sx[i]+=a[i][j];
			sy[j]+=a[i][j];
		}
	while(work());
	int cnt=0;
	for(int i=0;i<n;i++) cnt+=fx[i];
	printf("%d",cnt);
	for(int i=0;i<n;i++) if(fx[i]) printf(" %d",i+1); puts("");
	cnt=0;
	for(int j=0;j<m;j++) cnt+=fy[j];
	printf("%d",cnt);
	for(int j=0;j<m;j++) if(fy[j]) printf(" %d",j+1); puts("");
}
