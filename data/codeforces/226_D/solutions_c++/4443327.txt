#include<cstdio>
#include<cstring>

#define ran 111

int n,m,s;
int v[ran][ran];
int r[ran],c[ran];
bool flip_r[ran],flip_c[ran];
bool finish;

int main(){
	memset(r,0,sizeof r);
	memset(c,0,sizeof c);
	memset(flip_r,false,sizeof flip_r);
	memset(flip_c,false,sizeof flip_c);
	
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++){
			scanf("%d",&v[i][j]);
			r[i]+=v[i][j];
			c[j]+=v[i][j];
		}
	
	finish=false;
	while(!finish){
		finish=true;
		for(int i=0;i<n;i++)
			if(r[i]<0){
				finish=false;
				r[i]=-r[i];
				flip_r[i]=!flip_r[i];
				for(int j=0;j<m;j++){
					c[j]-=2*v[i][j];
					v[i][j]=-v[i][j];
				}
			}
		for(int i=0;i<m;i++)
			if(c[i]<0){
				finish=false;
				c[i]=-c[i];
				flip_c[i]=!flip_c[i];
				for(int j=0;j<n;j++){
					r[j]-=2*v[j][i];
					v[j][i]=-v[j][i];
				}
			}
	}
	
	s=0;
	for(int i=0;i<n;i++)
		s+=flip_r[i];
	printf("%d",s);
	for(int i=0;i<n;i++)
		if(flip_r[i]) printf(" %d",i+1);
	printf("\n");
	
	s=0;
	for(int i=0;i<m;i++)
		s+=flip_c[i];
	printf("%d",s);
	for(int i=0;i<m;i++)
		if(flip_c[i]) printf(" %d",i+1);
	printf("\n");

	return 0;
}
