#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<int, int> pii;
const int N = 100 + 5;

int n, m, a[N][N], c[N], r[N], rans[N], cans[N];

int main() {
	ios_base::sync_with_stdio(false);
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++) {
			cin >> a[i][j];
			r[i] += a[i][j];
			c[j] += a[i][j];
		}
	while (true) {
		int x = min_element(r, r + n) - r;
		if (r[x] < 0) {
			for (int j = 0; j < m; j++) {
				a[x][j] *= -1;
				c[j] += 2 * a[x][j];
			}
			r[x] *= -1;
			rans[x] ^= 1;
			continue;
		}
		x = min_element(c, c + m) - c;
		if (c[x] < 0) {
			for (int i = 0; i < n; i++) {
				a[i][x] *= -1;
				r[i] += 2 * a[i][x];
			}
			c[x] *= -1;
			cans[x] ^= 1;
			continue;
		}
		break;
	}
	cout << count(rans, rans + n, 1) << ' ';
	for (int i = 0; i < n; i++)
		if (rans[i])
			cout << i + 1 << ' ';
	cout << '\n';
	cout << count(cans, cans + m, 1) << ' ';
	for (int i = 0; i < m; i++)
		if (cans[i])
			cout << i + 1 << ' ';
	cout << '\n';
	return 0;
}
