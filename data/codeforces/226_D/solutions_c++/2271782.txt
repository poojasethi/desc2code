#include<cstdio>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int h,w; scanf("%d%d",&h,&w);
	static int a[100][100];
	rep(i,h) rep(j,w) scanf("%d",a[i]+j);

	int row[100]={},col[100]={}; // 行和, 列和
	rep(i,h) rep(j,w) {
		row[i]+=a[i][j];
		col[j]+=a[i][j];
	}

	bool flip_row[100]={},flip_col[100]={};
	ANKOORU:
	rep(i,h) if(row[i]<0) {
		flip_row[i]^=true;
		row[i]*=-1;
		rep(j,w) col[j]-=2*a[i][j], a[i][j]*=-1;
		goto ANKOORU;
	}
	rep(j,w) if(col[j]<0) {
		flip_col[j]^=true;
		col[j]*=-1;
		rep(i,h) row[i]-=2*a[i][j], a[i][j]*=-1;
		goto ANKOORU;
	}

	printf("%d",count(flip_row,flip_row+h,true));
	rep(i,h) if(flip_row[i]) printf(" %d",i+1); puts("");
	printf("%d",count(flip_col,flip_col+w,true));
	rep(j,w) if(flip_col[j]) printf(" %d",j+1); puts("");

	return 0;
}
