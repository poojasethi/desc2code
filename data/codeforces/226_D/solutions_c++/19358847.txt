#include<cstdio>
const int N=101;
int a[N][N],b[N],c[N],g[N],h[N];
int main()
{
	int i,j,n,m,x,y;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			scanf("%d",&a[i][j]);
			b[i]+=a[i][j];
			c[j]+=a[i][j];
		}
	}
	while(1)
	{
		x=y=-1;
		for(i=0;i<n;i++)
		{
			if(b[i]<0)
			{
				x=i;
				break;
			}
		}
		if(x!=-1)
		{
			for(i=0;i<m;i++)
			{
				a[x][i]*=-1;
				c[i]+=2*a[x][i];
				b[x]+=2*a[x][i];
			}
			g[x]++;
			continue;
		}
		for(i=0;i<m;i++)
		{
			if(c[i]<0)
			{
				y=i;
				break;
			}
		}
		if(y==-1)break;
		for(i=0;i<n;i++)
		{
			a[i][y]*=-1;
			c[y]+=2*a[i][y];
			b[i]+=2*a[i][y];
		}
		h[y]++;
	}
	for(i=j=0;i<n;i++)if(g[i]&1)j++;
	printf("%d ",j);
	for(i=0;i<n;i++)
	{
		if(g[i]&1)
			printf("%d ",i+1);
	}
	printf("\n");
	for(i=j=0;i<m;i++)if(h[i]&1)j++;
	printf("%d ",j);
	for(i=0;i<m;i++)
	{
		if(h[i]&1)
			printf("%d ",i+1);
	}
	printf("\n");
	return 0;
}
