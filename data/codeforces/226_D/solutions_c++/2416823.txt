#include<cstdio>
int a[101][101],hs[101],ls[101],hf[101],lf[101],ans,n,m,b;
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            scanf("%d",&a[i][j]),hs[i]+=a[i][j],ls[j]+=a[i][j];
    b=1;
    while(b==1){
        b=0;
        for(int i=1;i<=n;i++)
            if(hs[i]<0){
                hs[i]*=-1;
                for(int j=1;j<=m;j++)
                    a[i][j]*=-1,ls[j]+=2*a[i][j];
                hf[i]=1-hf[i],b=1;    
            }
        for(int j=1;j<=m;j++)
            if(ls[j]<0){
                ls[j]*=-1;
                for(int i=1;i<=n;i++)
                    a[i][j]*=-1,hs[i]+=2*a[i][j];
                lf[j]=1-lf[j],b=1;
            }
    }
    for(int i=1;i<=n;i++)ans+=hf[i];
    printf("%d ",ans);
    for(int i=1;i<=n;i++)   if(hf[i])   printf("%d ",i);printf("\n");
    ans=0;
    for(int i=1;i<=m;i++)ans+=lf[i];
    printf("%d ",ans);
    for(int i=1;i<=m;i++)   if(lf[i])   printf("%d ",i);
    getchar(),getchar();
}
