#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
const int MAXN = 100 + 5;
int n, m;
int a[MAXN][MAXN];
int r[MAXN], c[MAXN];
void out(int arr[], int len) {
	int cnt = 0;
	for (int i = 0; i < len; i++)
		if (arr[i]) cnt++;
	printf("%d", cnt);
	for (int i = 0; i < len; i++)
		if (arr[i]) printf(" %d", i + 1);
	puts("");
}
int main() {
	//freopen("test.in", "r", stdin);
	while (~scanf("%d %d", &n, &m)) {
		for (int i = 0; i < n; i++)
			for (int j = 0; j < m; j++)
				scanf("%d", &a[i][j]);
		bool flg = true;
		while (flg) {
			flg = false;
			for (int i = 0; i < n; i++) {
				int s = 0;
				for (int j = 0; j < m; j++)
					s += r[i] ^ c[j] ? -a[i][j] : a[i][j];
				if (s < 0) r[i] ^= 1, flg = true;
			}
			for (int i = 0; i < m; i++) {
				int s = 0;
				for (int j = 0; j < n; j++)
					s += r[j] ^ c[i] ? -a[j][i] : a[j][i];
				if (s < 0) c[i] ^= 1, flg = true;
			}
		}
		out(r, n);
		out(c, m);
	}
	return 0;
}
