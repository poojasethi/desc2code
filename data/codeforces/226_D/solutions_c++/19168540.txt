#include<cstdio>
#include<cstring>
int main()
{
	int n,m,num[105][105],sum,flag=1,r[105],c[105];
	memset(r,0,sizeof(r));
	memset(c,0,sizeof(c));
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
	{
		sum=0;
		for(int j=1;j<=m;j++)
		{
			scanf("%d",&num[i][j]);
			sum+=num[i][j];
		}
		num[i][0]=sum;
	}
	for(int i=1;i<=m;i++)
	{
		sum=0;
		for(int j=1;j<=n;j++)
		{
			sum+=num[j][i];
		}
		num[0][i]=sum;
	}
	while(flag)
	{
		flag=0;
		for(int i=1;i<=n;i++)
		{
			if(num[i][0]<0)
			{
				flag=1;
				r[i]++;
				num[i][0]*=-1;
				for(int j=1;j<=m;j++)
				{
					num[0][j]-=2*num[i][j];
					num[i][j]*=-1;
				}
			}
		}//printf("2264");
		for(int i=1;i<=m;i++)
		{
			if(num[0][i]<0)
			{
				flag=1;
				c[i]++;
				num[0][i]*=-1;
				for(int j=1;j<=n;j++)
				{
					num[j][0]-=2*num[j][i];
					num[j][i]*=-1;
				}
			}
		}
	}
	for(int i=1;i<=n;i++)
	if(r[i]%=2)
	r[0]++;
	for(int i=1;i<=m;i++)
	if(c[i]%=2)
	c[0]++;
	printf("%d",r[0]);
	for(int i=1;i<=n;i++)
	if(r[i])
	printf(" %d",i);
	printf("\n%d",c[0]);
	for(int i=1;i<=m;i++)
	if(c[i])
	printf(" %d",i);
}
