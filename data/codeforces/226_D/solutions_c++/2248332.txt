#include<cstdio>
#define fo(i,a,b) for(i=a;i<=b;++i)
const int mn=120;
int n,m,i,j;
int a[mn][mn],sx[mn],sy[mn],w[mn];
bool rx[mn],ry[mn],upd;
void R_rot(int x)
{
    int i;
    sx[x]=-sx[x];
    rx[x]=!rx[x];
    fo(i,1,m)
        sy[i]-=a[x][i]*2,a[x][i]*=-1;
}
void C_rot(int x)
{
    int i;
    sy[x]=-sy[x];
    ry[x]=!ry[x];
    fo(i,1,n)
        sx[i]-=a[i][x]*2,a[i][x]*=-1;
}
void print(bool *r,int n)
{
    int i,ans=0;
    fo(i,1,n)if(r[i])w[++ans]=i;
    printf("%d",ans);
    fo(i,1,ans)printf(" %d",w[i]);
    printf("\n");
}
int main()
{
    scanf("%d%d",&n,&m);
    fo(i,1,n)
        fo(j,1,m)
        {
            scanf("%d",a[i]+j);
            sx[i]+=a[i][j],sy[j]+=a[i][j];
        }
    for(upd=1;upd;)
    {
        upd=0;
        fo(i,1,n)if(sx[i]<0)R_rot(i),upd=1;
        fo(i,1,m)if(sy[i]<0)C_rot(i),upd=1;
    }
    print(rx,n);
    print(ry,m);
    return 0;
}
