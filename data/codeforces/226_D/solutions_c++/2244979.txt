#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;

const int N = 120;

int n, m, x[N][N], pc[N], pl[N], nu, ver = 1, suml[N], sumc[N];

int main() {
	int i, j;
	
	cin >> n >> m;
	
	for(i = 1; i<=n; ++i)
		for(j = 1; j<=m; ++j)
			cin >> x[i][j], suml[i] += x[i][j], sumc[j] += x[i][j];
	
	while(ver) {
		ver = 0;
		
		for(i = 1; i<=n; ++i) if(suml[i] < 0) {
			pl[i] = 1 - pl[i];
			
			suml[i] = -suml[i]; ver = 1;
			
			for(j = 1; j<=m; ++j)
				x[i][j] = -x[i][j], sumc[j] = sumc[j] + 2*x[i][j];
		}
		
		for(j = 1; j<=m; ++j) if(sumc[j] < 0) {
			pc[j] = 1 - pc[j];
			
			sumc[j] = -sumc[j]; ver = 1;
			
			for(i = 1; i<=n; ++i)
				x[i][j] = -x[i][j], suml[i] = suml[i] + 2 * x[i][j];
		}
	}
	
	for(i = 1; i<=n; ++i)
		if(pl[i])
			++nu;
	cout << nu << " ";
	for(i = 1; i<=n; ++i)
		if(pl[i])
			cout << i << " ";
	cout << "\n";
	nu = 0;
	for(i = 1; i<=m; ++i)
		if(pc[i])
			++nu;
	
	cout << nu << " ";
	for(i = 1; i<=m; ++i)
		if(pc[i])
			cout << i << " ";
	
	return 0;
}
