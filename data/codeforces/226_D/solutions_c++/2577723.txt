#include<cstdio>
const int n=105;
int N,M,i,j,P,Q,flag;
int A[n][n],B[n],C[n],F[n],G[n],X[n],Y[n];
void Work1(int i)
{
	for(j=1;j<=M;++j) A[i][j]=-A[i][j],C[j]+=2*A[i][j];
	F[i]^=1,B[i]=-B[i];
}
void Work2(int j)
{
	for(i=1;i<=N;++i) A[i][j]=-A[i][j],B[i]+=2*A[i][j];
	G[j]^=1,C[j]=-C[j];
}
int main()
{
	scanf("%d%d",&N,&M);
	for(i=1;i<=N;++i) for(j=1;j<=M;++j)
		scanf("%d",&A[i][j]),B[i]+=A[i][j],C[j]+=A[i][j];
	for(;!flag;)
	{
		flag=1;
		for(i=1;i<=N;++i) if(B[i]<0) Work1(i),flag=0;
		for(j=1;j<=M;++j) if(C[j]<0) Work2(j),flag=0;
	}
	for(i=1;i<=N;++i) if(F[i]) X[++P]=i;
	for(i=1;i<=M;++i) if(G[i]) Y[++Q]=i;
	printf("%d",P); for(i=1;i<=P;++i) printf(" %d",X[i]); printf("\n");
	printf("%d",Q); for(i=1;i<=Q;++i) printf(" %d",Y[i]); printf("\n");
	return 0;
}
