#include<cstdio>
#define N 100010
int n,m,i,j,col[N],row[N],a[200][200],fr[N],fc[N],ii,ans;
int main()
{
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++)
	for (j=1;j<=m;j++)
	{
		scanf("%d",&a[i][j]);
		col[j]+=a[i][j];
		row[i]+=a[i][j];
	}
	
	for (ii=1;ii<=100000;ii++)
	{
		for (i=1;i<=n;i++)
		if (row[i]<0)
		{
			row[i]*=-1;
			fr[i]^=1;
			for (j=1;j<=m;j++)
			{
				col[j]-=2*a[i][j];
				a[i][j]*=-1;
			}
		}
		
		for (i=1;i<=m;i++)
		if (col[i]<0)
		{
			col[i]*=-1;
			fc[i]^=1;
			for (j=1;j<=n;j++)
			{
				row[j]-=2*a[j][i];
				a[j][i]*=-1;
			}
		}
	}
	for (i=1;i<=n;i++)
	if (fr[i]) ans++;
	printf("%d ",ans);
	for (i=1;i<=n;i++)
	if (fr[i]) printf("%d ",i);
	printf("\n");
	
	ans=0;
	for (i=1;i<=m;i++)
	if (fc[i]) ans++;
	printf("%d ",ans);
	for (i=1;i<=m;i++)
	if (fc[i]) printf("%d ",i);
	printf("\n");
}
