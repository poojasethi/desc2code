#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
#define LL int
#define R "%d"
#define N 100
using namespace std;

LL i, j, k, p, n, m, ans;
LL a[N + 10][N + 10];
bool r[N + 10], c[N + 10];

void Print(bool a[], LL sz) {
	LL res = 0;
	for (i = 1; i <= sz; ++i)
		res += a[i];
	printf(R, res);
	for (i = 1; i <= sz; ++i)
		if (a[i]) printf(" " R, i);
	printf("\n");
}

int main() {
	#ifndef ONLINE_JUDGE
	freopen("table.in", "r", stdin);freopen("table.out", "w", stdout);
	#endif
	scanf(R R "\n", &n, &m);
	for (i = 1; i <= n; ++i)
		for (j = 1; j <= m; ++j)
			scanf(R, &a[i][j]);
	for ( ; ; ) {
		bool flag = 0;
		for (i = 1; i <= n; ++i) {
			LL s = 0;
			for (j = 1; j <= m; ++j)
				s += r[i] ^ c[j] ? (-a[i][j]) : a[i][j];
			if (s < 0) flag = 1, r[i] ^= 1;
		}
		for (i = 1; i <= m; ++i) {
			LL s = 0;
			for (j = 1; j <= n; ++j)
				s += c[i] ^ r[j] ? (-a[j][i]) : a[j][i];
			if (s < 0) flag = 1, c[i] ^= 1;
		}
		if (!flag) break;
	}
	Print(r, n), Print(c, m);
	return 0;
}
