#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
int a[110][110],b[110][110],c[110][110],vx[110],vy[110],n,m,i,j,k,t,cnt,temp;

int main()
{
	cin>>n>>m;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++) scanf("%d",&a[i][j]);
	for(t=1;t;t++)
	{
		
		for(i=1;i<=n;i++)
			for(j=1;j<=m;j++) b[i][j]=b[i][j-1]+a[i][j],c[i][j]=c[i-1][j]+a[i][j];
		temp=0x3fffffff;
		for(i=1;i<=n;i++)
			if(b[i][m]<temp) temp=b[i][m],k=i;
		for(i=1;i<=m;i++)
			if(c[n][i]<temp) temp=c[n][i],k=-i;
		if(temp>=0) break; 
		if(k>0)
		{
			vx[k]^=1;
			for(i=1;i<=m;i++) a[k][i]*=-1;
		}
		if(k<0)
		{
			k=-k;
			vy[k]^=1;
			for(i=1;i<=n;i++) a[i][k]*=-1;
		}
	}
	for(i=1;i<=n;i++) if(vx[i]) cnt++;
	cout<<cnt;
	for(i=1;i<=n;i++) if(vx[i]) cout<<' '<<i;
	cout<<endl;
	cnt=0;
	for(i=1;i<=m;i++) if(vy[i]) cnt++;
	cout<<cnt;
	for(i=1;i<=m;i++) if(vy[i]) cout<<' '<<i;
	cout<<endl;
	return 0;
}
