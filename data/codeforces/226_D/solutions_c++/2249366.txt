#include <queue>
#define REP(i,n) for(int i=0; i<(int)(n); i++)

#include <set>
#include <cstdio>
inline int getInt(){ int s; scanf("%d", &s); return s; }

using namespace std;

int a[100][100];

int hh[100];
int ww[100];

int main(){
  const int h = getInt();
  const int w = getInt();
  REP(i,h) REP(j,w) a[i][j] = getInt();
  REP(i,h) hh[i] = 1;
  REP(i,w) ww[i] = 1;

  bool ok = false;
  while(!ok){
    ok = true;
    REP(i,h){
      int s = 0;
      REP(j,w) s += hh[i] * ww[j] * a[i][j];
      if(s < 0){
        ok = false;
        hh[i] *= -1;
      }
    }

    REP(i,w){
      int s = 0;
      REP(j,h) s += hh[j] * ww[i] * a[j][i];
      if(s < 0){
        ok = false;
        ww[i] *= -1;
      }
    }
  }
  {
    vector<int> b;
    REP(i,h) if(hh[i] < 0) b.push_back(i + 1);
    printf("%d", b.size());
    REP(i,b.size()) printf(" %d", b[i]);
    puts("");
  }
  {
    vector<int> b;
    REP(i,w) if(ww[i] < 0) b.push_back(i + 1);
    printf("%d", b.size());
    REP(i,b.size()) printf(" %d", b[i]);
    puts("");
  }

  return 0;
}
