#include<cstdio>
#define N 105
int a[N][N],c[N],d[N],o[N],n,m,flag=1; bool f[N],g[N];
void prt(int n,bool *f)
{
	int len=0; for (int i=1;i<=n;i++) if (f[i]) o[len++]=i;
	printf("%d",len); for (int i=0;i<len;i++) printf(" %d",o[i]); printf("\n");
}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++) for (int j=1;j<=m;j++)
		scanf("%d",&a[i][j]),c[i]+=a[i][j],d[j]+=a[i][j];
	while (flag){
		flag=0;
		for (int i=1;i<=n;i++) if (c[i]<0){
			for (int j=1;j<=m;j++) a[i][j]=-a[i][j],d[j]+=2*a[i][j];
			c[i]=-c[i],f[i]^=1,flag=1;
			}
		for (int j=1;j<=m;j++) if (d[j]<0){
			for (int i=1;i<=n;i++) a[i][j]=-a[i][j],c[i]+=2*a[i][j];
			d[j]=-d[j],g[j]^=1,flag=1;
			}
		}	
	prt(n,f); prt(m,g); return 0;
}
