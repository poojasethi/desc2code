#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

const int MAX = 100+2;
int a[MAX][MAX], n ,m, row[MAX], col[MAX];
int nr[MAX], nc[MAX];

void makeRow(int x) {
	row[x] = -row[x];
	for (int i=0; i<m; i++) {
		a[x][i] = -a[x][i];
		col[i] += 2*a[x][i];
	}
}
void makeCol(int x) {
	col[x] = -col[x];
	for (int i=0; i<n; i++) {
		a[i][x] = -a[i][x];
		row[i] += 2*a[i][x];
	}
}

int main() {
	cin >> n >> m;
	for (int i=0; i<n; i++)
		for (int j=0; j<m; j++) {
			cin >> a[i][j];
			row[i] += a[i][j];
			col[j] += a[i][j];
		}

	bool psb = 1;
	while (psb) {
		psb = 0;
		for (int i=0; i<n; i++)
			if (row[i] < 0) {
//				cerr << "A" << endl;
				nr[i]++;
				makeRow(i);
			}
		for (int i=0; i<m; i++)
			if (col[i] < 0) {
				psb = 1;
				nc[i]++;
				makeCol(i);
			}
	}
	int ansr = 0, ansc = 0;
	for (int i=0; i<n; i++)
		if (nr[i]%2 == 1)
			ansr++;
	for (int i=0; i<m; i++)
		if (nc[i]%2 == 1)
			ansc++;

	cout << ansr << " ";
	for (int i=0; i<n; i++)
		if (nr[i]%2 == 1)
			cout << i+1 << " ";
	cout << endl << ansc << " ";
	for (int i=0; i<m; i++)
		if (nc[i]%2 == 1)
			cout  << i+1 << " ";
	cout << endl;
	return 0;
}
