#include <iostream>
using namespace std;

#define rep(i, n) rep2(i, 0, n)
#define rep2(i, m, n) for(int i = m; i < (int)(n); ++i)

int n, m, aa, b, a[111][111], sr[111], sc[111], ir[111], ic[111];

void invrow(int i)
{
    ir[i] ^= 1; sr[i] = -sr[i];
    rep(j, m) a[i][j] = -a[i][j], sc[j] += 2 * a[i][j];
}

void invcol(int j)
{
    ic[j] ^= 1; sc[j] = -sc[j];
    rep(i, n) a[i][j] = -a[i][j], sr[i] += 2 * a[i][j];
}

int main()
{
    cin >> n >> m;
    rep(i, n) rep(j, m) cin >> a[i][j], sr[i] += a[i][j], sc[j] += a[i][j];
    while(true) {
        bool update = false;
        rep(i, n) if(sr[i] < 0) invrow(i), update = true;
        rep(j, m) if(sc[j] < 0) invcol(j), update = true;
        if(!update) break;
    }
    rep(i, n) if(ir[i]) ++aa;
    rep(j, m) if(ic[j]) ++b;
    cout << aa << " ";
    rep(i, n) if(ir[i]) cout << i + 1 << " ";
    cout << "\n" << b << " ";
    rep(j, m) if(ic[j]) cout << j + 1 << " ";
    cout << endl;
}
