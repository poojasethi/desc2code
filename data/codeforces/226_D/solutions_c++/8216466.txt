#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=120;
int n,m,mat[maxn][maxn],ans1[maxn],anss1,ans2[maxn],anss2;
int sum[2][maxn];
bool rev[2][maxn];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
	freopen("data.out","w",stdout);
#endif
	int i,j,x;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			scanf("%d",&mat[i][j]),sum[0][i]+=mat[i][j],sum[1][j]+=mat[i][j];
	while(1)
	{
		bool z=0;
		for(x=1;x<=n;x++)
			if(sum[0][x]<0)
				break;
		if(x<=n) 
		{
			z=1;
			rev[0][x]^=1;
			sum[0][x]=-sum[0][x];
			for(i=1;i<=m;i++)
				sum[1][i]+=2*mat[x][i]*((rev[0][x]^rev[1][i])?-1:1);
		}
		for(x=1;x<=m;x++)
			if(sum[1][x]<0)
				break;
		if(x<=m)
		{
			z=1;
			rev[1][x]^=1;
			sum[1][x]=-sum[1][x];
			for(i=1;i<=n;i++)
				sum[0][i]+=2*mat[i][x]*((rev[1][x]^rev[0][i])?-1:1);
		}
		if(!z) break;
	}
	for(i=1;i<=n;i++)
		if(rev[0][i])
			ans1[anss1++]=i;
	for(i=1;i<=m;i++)
		if(rev[1][i])
			ans2[anss2++]=i;
	printf("%d",anss1);
	for(i=0;i<anss1;i++)
		printf(" %d",ans1[i]);
	printf("\n");
	printf("%d",anss2);
	for(i=0;i<anss2;i++)
		printf(" %d",ans2[i]);
	printf("\n");
	return 0;
}
