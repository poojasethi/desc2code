#include<cstdio>
#include<cstring>
#include<numeric>
using namespace std;
int fr[110],fc[110];
int ma[110][110],m,n;
inline bool f(int id,bool r)
{
	int i,j,sum=0;
	if(r)//è¡Œ 
	{
		for(i=0;i<m;++i)
		{
			sum+=ma[id][i];
		}
		if(sum>=0)
		{
			return false;
		}
		fr[id]^=1;
		for(i=0;i<m;++i)
		{
			ma[id][i]=-ma[id][i];
		}
		return true;
	}
	for(i=0;i<n;++i)
	{
		sum+=ma[i][id];
	}
	if(sum>=0)
	{
		return false;
	}
	fc[id]^=1;
	for(i=0;i<n;++i)
	{
		ma[i][id]=-ma[i][id];
	}
	return true;
}
int main()
{
	int i,j;
	bool tag=true;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;++i)
	{
		for(j=0;j<m;++j)
		{
			scanf("%d",&ma[i][j]);
		}
	}
	memset(fr,0,sizeof(int)*(n));
	memset(fc,0,sizeof(int)*(m));
	while(tag)
	{
		tag=false;
		for(i=0;i<n;++i)
		{
			f(i,true)?tag=true:1;
		}
		for(i=0;i<m;++i)
		{
			f(i,false)?tag=true:1;
		}
	}
	printf("%d",accumulate(fr,fr+n,0));
	for(i=0;i<n;++i)
	{
		fr[i]?printf(" %d",i+1):1;
	}
	printf("\n%d",accumulate(fc,fc+m,0));
	for(i=0;i<m;++i)
	{
		fc[i]?printf(" %d",i+1):1;
	}
	putchar('\n');
	return 0;
}
