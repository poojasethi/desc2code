#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 110
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,ty,mi;
int a[N][N],f[N],h[N],an[N],re[N];
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n)For(j,1,m) scanf("%d",&a[i][j]),h[i]+=a[i][j],f[j]+=a[i][j];
	for (;;) {
		mi=0;
		For(i,1,n) if (h[i]<mi) mi=h[i],ty=1,k=i;
		For(i,1,m) if (f[i]<mi) mi=f[i],ty=2,k=i;
		if (!mi) break;
		if (ty==1) {
			h[k]*=-1;
			For(i,1,m) {
				f[i]-=a[k][i]*2;
				a[k][i]*=-1;
			}
			an[k]^=1;
		} else {
			f[k]*=-1;
			For(i,1,n) {
				h[i]-=a[i][k]*2;
				a[i][k]*=-1;
			}
			re[k]^=1;
		}
	}
	k=0; For(i,1,n) if (an[i]) k++;
	printf("%d ",k); For(i,1,n) if (an[i]) printf("%d ",i); printf("\n");
	k=0; For(i,1,m) if (re[i]) k++;
	printf("%d ",k); For(i,1,m) if (re[i]) printf("%d ",i); printf("\n");
	//for(;;);
	return 0;
}
