#include<stdio.h>
int r[10000];
int c[10000];
int tab[1001][1001];
int main(){
	int n,m;
	scanf("%d %d",&n,&m);
	for(int i =0 ; i < n ; i++ ){
		for(int j = 0;j < m ;j++ ) scanf("%d",&tab[i][j]);
	}
	for(int k = 0; k < 300 ; k++ ){
		for(int i = 0 ;i < n ; i++ ){
			int sum = 0;
			for(int j = 0 ;j < m ; j++ ) sum += tab[i][j];
			if( sum < 0 ){
				r[i] ^=1;
				for(int j = 0;j < m ;j++ ) tab[i][j] *=-1;
			}
		}
		for(int i = 0 ;i < m ;i++ ){
			int sum = 0;
			for(int j =0 ;j <n ; j++ ) sum += tab[j][i];
			if( sum < 0 ){
				c[i] ^=1;
				for(int j = 0 ;j < n ;j++ ) tab[j][i] *=-1;
			}
		}
	}
	int num = 0;
	for(int i = 0;i < n ; i++ ) if( r[i] == 1 ) num++;
	printf("%d ",num);
	for(int i = 0 ;i < n ;i++ ) if( r[i] == 1 ) printf("%d ",i+1);
	printf("\n");
	num = 0;
	for(int i = 0;i < m ; i++ ) if( c[i] == 1 ) num++;
	printf("%d ",num);
	for(int i = 0 ;i < m ;i++ ) if( c[i] == 1 ) printf("%d ",i+1);
	printf("\n");
}
