#include <iostream>
#include <cmath>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n,m,a[110][110],sr[110],sc[110],fr[110],fc[110],cr,cc,p=1;

void f(int v){
  rep(i,m){
    a[v][i]*=-1;
    sc[i]+=a[v][i]*2;
  }
  sr[v]*=-1;
  fr[v]=1-fr[v];
}

void g(int v){
  rep(i,n){
    a[i][v]*=-1;
    sr[i]+=a[i][v]*2;
  }
  sc[v]*=-1;
  fc[v]=1-fc[v];
}

int main(){
  cin>>n>>m;
  rep(i,n)rep(j,m){
    cin>>a[i][j];
    sr[i]+=a[i][j];
    sc[j]+=a[i][j];
  }
  while(p){
    p=0;
    rep(i,n)if(sr[i]<0){f(i);p=1;}
    rep(i,m)if(sc[i]<0){g(i);p=1;}
  }
  rep(i,n)cr+=fr[i];
  rep(i,m)cc+=fc[i];
  cout<<cr<<" ";rep(i,n)if(fr[i])cout<<i+1<<" ";cout<<endl;
  cout<<cc<<" ";rep(i,m)if(fc[i])cout<<i+1<<" ";cout<<endl;

  return 0;

}
