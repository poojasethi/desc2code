#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef unsigned __int64 lld;
const int maxn = 103;

int r[maxn], sr[maxn];
int c[maxn], sc[maxn];
int a[maxn][maxn];

void print(int *a, int n) {
	int i, tot = 0;
	for (i=1; i <= n; i++) if (a[i]&1) {
		tot++;
	}
	printf("%d",tot);
	for (i=1; i <= n; i++) if (a[i]&1) {
		printf(" %d",i);
	}
	printf("\n");
}
int main() {
	int n, m, i, j;
	scanf("%d%d",&n,&m);
	for (i=1; i <= n; i++) {
		for (j=1; j <= m; j++) 
			scanf("%d",&a[i][j]);
	}
	for (i=1; i <= n; i++) {
		r[i] = 0;
		for (j=1; j <= m; j++)
			r[i] += a[i][j];
	}
	for (i=1; i <= m; i++) {
		c[i] = 0;
		for (j=1; j <= n; j++) 
			c[i] += a[j][i];
	}
	bool flag = true;
	while (flag) {
		flag = false;
		for (i=1; i <= n; i++) if (r[i] < 0){
			flag = true;
			sr[i]++;
			r[i] = -r[i];
			for (j=1; j <= m; j++) {
				c[j] -= 2*a[i][j];
				a[i][j] = -a[i][j];
			}
		}
		for (i=1; i <= m; i++) if (c[i] < 0) {
			flag = true;
			sc[i]++;
			c[i] = -c[i];
			for (j=1; j <= n; j++) {
				r[j] -= 2*a[j][i];
				a[j][i] = -a[j][i];
			}
		}
	}
	print(sr, n);
	print(sc, m);
	return 0;
}
