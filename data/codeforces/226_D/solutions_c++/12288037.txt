#include<cstdio>
#include<cstring>
#define N 101
int sr[N],sc[N];
int a[N][N];
int calc(int x1,int x2,int y1,int y2){
	int res=0;
	for(int i=x1;i<=x2;++i)
	for(int j=y1;j<=y2;++j) res+=a[i][j];
	return res;
}
int main(){
	int n,m;
	int i,j,t;
	int ar,ac;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i)
	for(j=1;j<=m;++j) scanf("%d",&a[i][j]);
	bool flag=true;
	t=0;
	ar=ac=0;
	while(flag){
		t^=1;
		if(t){
			// row
			for(i=1;i<=n;++i)
			if(calc(i,i,1,m)<0){
				sr[i]?--ar:++ar;
				sr[i]^=1;
				for(j=1;j<=m;++j) a[i][j]=-a[i][j];
			}
			flag=false;
			for(j=1;j<=m;++j)
			if(calc(1,n,j,j)<0){
				flag=true;
				break;
			}
		}
		else{
			// col
			for(j=1;j<=m;++j)
			if(calc(1,n,j,j)<0){
				sc[j]?--ac:++ac;
				sc[j]^=1;
				for(i=1;i<=n;++i) a[i][j]=-a[i][j];
			}
			flag=false;
			for(i=1;i<=n;++i)
			if(calc(i,i,1,m)<0){
				flag=true;
				break;
			}
		}
	}
//	for(i=1;i<=n;++i){
//		for(j=1;j<=m;++j) printf("%d ",a[i][j]);
//		printf("\n");
//	}
	printf("%d",ar);
	for(i=1;i<=n;++i)
	if(sr[i]) printf(" %d",i);
	printf("\n");
	printf("%d",ac);
	for(j=1;j<=m;++j)
	if(sc[j]) printf(" %d",j);
	return 0;
}
