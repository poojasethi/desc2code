#include <algorithm>
#include <cstdio>
#include <iostream>
#include <cstring>
#include <string>

using namespace std;

int n, k;
string str;

void doit(string s){
	int now = 0, right = 1, last = -1;
	int cnt[10];
	memset(cnt, 0, sizeof(cnt));
	while(now >= 0 && now < s.length()){
		if(s[now] == '#') {
			now += right;
			continue;
		}
		if(s[now] >= '0' && s[now] <= '9'){
			cnt[s[now] - '0']++;
			last = -1;
			if(--s[now] < '0') s[now] = '#';
		}
		else{
			right = s[now] == '>' ? 1 : -1;
			if(~last) s[last] = '#';
			last = now;
		}
		now += right;
	}	
	for(int i = 0; i < 10; ++i)
		cout << cnt[i] << ' ';
	cout << endl;
}
int main(){
	cin >> n >> k;
	cin >> str;
	for(int i = 0, l, r; i < k; ++i){
		cin >> l >> r;
		doit(str.substr(l - 1, r - l + 1));
	}
	return 0;
}
