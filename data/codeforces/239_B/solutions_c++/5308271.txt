#include <bits/stdc++.h>
using namespace std;

int ok[128], f[16], n, q, l, r;
string s;

int main() {  
  cin >> n >> q >> s;
  for (int i = 0; i < q; ++i) {
    cin >> l >> r;
    
    string p = s.substr(l-1, r-l+1);
    memset(ok, 1, sizeof(ok));
    memset(f, 0, sizeof(f));
    int dp = 1, cp = 0, idx = 0;
    
    while (idx >= 0 && idx < r-l+1) {
      if (!ok[idx]) { 
        idx += dp;
        continue;
      }
      if (isdigit(p[idx])) {
        f[p[idx]-'0']++;
        if (--p[idx] < '0') ok[idx] = 0;
      } else {
        if (!isdigit(p[cp]) && cp != idx) ok[cp] = 0;
        dp = -1;
        if (p[idx] == '>') dp = 1;
      }
      cp = idx;
      idx += dp;      
    }
    for (int d = 0; d < 10; ++d) {
      cout << f[d] << " ";
    }
    cout << endl;
  }
  return 0;
}
