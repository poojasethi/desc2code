using namespace std;

char c[110];

void go(int st,int ed){
	char s[110];
	int a[10];
	memset(a,0,sizeof(a));
	strcpy(s,c);

	int ptr=st,d=1;
	char pre=-1;
	while(st<=ptr && ptr<=ed){
		if('0'<=s[ptr] && s[ptr]<='9'){
			a[s[ptr]-'0']++;

			if(s[ptr]=='0') s[ptr]=0;
			else s[ptr]--;
			pre=-1;
		}
		else if( s[ptr]=='<'){
			d=-1;
			if(pre>=0)s[pre]=0;
			pre=ptr;
		}
		else if( s[ptr]=='>'){
			d=1;
			if(pre>=0)s[pre]=0;
			pre=ptr;
		}
		ptr+=d;
	}
	for(int i=0;i<10;i++)printf("%d%c",a[i],i==9?'\n':' ');
}

int main(){

	int n,m;
	scanf("%d%d%s",&n,&m,c);
	for(int i=0;i<m;i++){
		int x,y;
		scanf("%d%d",&x,&y);
		go(x-1,y-1);
	}
	return 0;
}
