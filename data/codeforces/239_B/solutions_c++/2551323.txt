#include<stdio.h>
#include<string>
#include<iostream>
using namespace std;

string s,temp;
void solve(int l,int r)
{
	int dp=1,cp=l-1,pre;
	int count[10]={0};
	while(l<=cp+1 && cp<r){
		if(s[cp]!='<' && s[cp]!='>'){
			count[s[cp]-'0']++;
			if(s[cp]=='0'){
				s.erase(cp,1);
				r--;
				if(dp+1)
					cp-=dp;
			}
			else
				s[cp]--;
			cp+=dp;
		}
		else{
			if(s[cp]=='<'){
				pre=cp;
				dp=-1;
				cp+=dp;
				if(l<=cp+1)
					if(s[cp]=='<' || s[cp]=='>'){
						s.erase(pre,1);
						r--;
					}			
			}
			else{
				pre=cp;
				dp=1;
				cp+=dp;
				if(cp<r)
					if(s[cp]=='<' || s[cp]=='>'){
						s.erase(pre,1);
						r--;
						cp--;
					}
			}
		}
	}
	int i;
	for(i=0;i<10;i++)
		printf("%s%d",i?" ":"",count[i]);
	printf("\n");
}
int main()
{
	int n,q;
	scanf("%d%d",&n,&q);
	cin>>s;
	temp=s;
	int l,r;
	while(q--){
		scanf("%d%d",&l,&r);
		s=temp;
		solve(l,r);
	}
	return 0;
}
