#include<iostream>
#include<string>
using namespace std;
int n,q;
string s;
void Q(string s){
  int c[10]={};  int n=s.size();
  for(int cp=0,dp=1;cp>=0 && cp<s.size();){
    bool yes=false;
    for(int i=0;i<s.size();i++) if(s[i]>='0' && s[i]<='9') yes=true;
    if(!yes)break;

    if(s[cp]=='>'){
      dp=1; cp++;
      if(s[cp]=='>' || s[cp]=='<') s.erase(cp-dp,1),cp-=dp*(dp==1);
    } else if(s[cp]=='<'){
      dp=-1; cp--;
     if(s[cp]=='>' || s[cp]=='<') s.erase(cp-dp,1),cp-=dp*(dp==1);
    } else if(s[cp]!='^'){
      int tmp=s[cp]-'0'; 
      c[tmp]++; 
      if(tmp) s[cp]=char(tmp-1+'0'),cp+=dp; else s.erase(cp,1),cp+=dp*(dp==-1);
    }
  }
  for(int i=0;i<10;i++) cout<<c[i]<<" "; cout<<endl;
}
int main(){
  cin>>n>>q; cin>>s;
  while(q--){
    int l,r; cin>>l>>r;
    Q(s.substr(l-1,r-l+1));
  }
}
