#include<iostream>
using namespace std;
#include<string>
void print( string s)
{
int cp = 0;
int dp = 1;
int out[10] = {0}; //the new stuff
while ( cp >=0 && cp < s.length() )
{
   bool erase = 0;
 if( s[cp]>= '0'&& s[cp] <= '9')
 {
 out[s[cp]-'0']++;
 s[cp]--;
 if( s[cp] < '0')
 erase = 1;
 }
else
{
  if(s[cp] == '<')
  {
   dp = -1;
  }
  else
  dp = 1;
  if(cp+dp >=0 && cp+dp < s.size() && ( s[cp+dp] == '>' || s[cp+dp] == '<') )
  erase =1;
}
if(erase)
{
 s.erase(cp , 1);
 if (dp == -1)
 cp--;
 }
 else
 {
 cp  +=dp;
 }
 }
 for(int i = 0; i < 10; i++)
 cout << out[i] <<" ";
 cout << endl;
}

int main()
{
int n , q;
string s;
cin >> n >> q;
cin >> s;
while ( q--)
{
int l , r;
cin >> l >> r;
print(s.substr(l-1 , r-l+1));
}
return 0;
}
