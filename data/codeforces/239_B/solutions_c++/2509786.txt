#include <cctype>
#include <cstring>
#include <iostream>

using namespace std;

int n, q, l, r, cnt[11];
bool bio;
string unos, s;

int main()
{
  cin >> n >> q >> unos;
  while (q--)
  {
    cin >> l >> r;
    s = unos.substr(l - 1, r - l + 1);
    memset(cnt, 0, sizeof cnt);
    bio = false;
    for (int cp = 0, dp = 1; cp >= 0 && cp < s.size();)
    {
      if (isdigit(s[cp]))
      {
        bio = false;
        cnt[s[cp] - '0']++;
        if (s[cp] == '0')
        {
          s.erase(cp, 1);
          if (dp == -1) cp--;
        }
        else s[cp]--, cp += dp;
      }
      else
      {
        if (bio)
        {
          s.erase(cp - dp, 1);
          if (dp == 1) cp--;
        }
        bio = true;
        if (s[cp] == '<') dp = -1;
        else dp = 1;
        cp += dp;
      }
    }
    for (int i = 0; i < 10; i++) cout << cnt[i] << " ";
    cout << endl;
  }
  return 0;
}
