#include<stdio.h>
#include<string.h>
char s[105];
char t[105];
int mm[10];
int main()
{
	int n,q;
	while(scanf("%d %d",&n,&q)!=EOF)
	{
		int st,end,i;
		scanf("%s",s);
		while(q--)
		{
			scanf("%d %d",&st,&end);
			strcpy(t,s);
			int p=st-1,d=1;
			int lastd=-1;
			memset(mm,0,sizeof(mm));
			while(p>=st-1 && p<end)
			{
				if(t[p]<='9' && t[p]>='0')
				{
					lastd=-1;
					mm[t[p]-'0']++;
				}
				if(t[p]=='0')
					t[p]='#';
				if(t[p]>'0' && t[p]<='9')
					t[p]--;
				if(t[p]=='<')
				{
					d=-1;
					if(lastd!=-1)
						t[lastd]='#';
					lastd=p;
				}
				if(t[p]=='>')
				{
					d=1;
					if(lastd!=-1)
						t[lastd]='#';
					lastd=p;
				}
				p+=d;
			}
			printf("%d",mm[0]);
			for(i=1;i<10;i++)
				printf(" %d",mm[i]);
			printf("\n");
		}
	}
	return 0;
}
