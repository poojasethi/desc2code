#include<stdio.h>
int main()
{
	int n,q,i,l,r,dir,f,pre;
	char ss[20000],s[20000];
	//scanf("%d%d",&n,&q);
	while(scanf("%d%d",&n,&q)==2)
	{
		getchar();
		for(i=1;i<=n;i++)
			scanf("%c",&ss[i]);
		while(q--)
		{
			scanf("%d%d",&l,&r);
			for(i=0;i<=n;i++)
				s[i]=ss[i];
			int c[200]={0};
			dir=1,f=0,pre=0;
			for(i=l;i>=l&&i<=r;i+=dir)
			{
				if(s[i]>='0'&&s[i]<='9')
				{
					c[s[i]-'0']++;
					if(s[i]=='0')
						s[i]=0;
					else
						s[i]--;
					f=0;
				}
				else if(s[i]=='>')
				{
					dir=1;
					if(f)
						s[pre]=0;
					f=1;
					pre=i;
				}
				else if(s[i]=='<')
				{
					dir=-1;
					if(f)
						s[pre]=0;
					f=1;
					pre=i;
				}
			}
			printf("%d",c[0]);
			for(i=1;i<10;i++)
				printf(" %d",c[i]);
			printf("\n");
		}
	}
	return 0;
}
