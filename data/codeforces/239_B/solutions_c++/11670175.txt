#include <bits/stdc++.h>
using namespace std;
int res[15];
int main(){
    int n,q,l,r;
    string in,s;
    cin>>n>>q>>in;
    while(q--){

        cin>>l>>r;l--;r--;
        memset(res,0,sizeof res);
        s=in.substr(l,r-l+1);

        int i=0,dir=1;
        while(i>=0 && i<(int)s.size() && !s.empty()){

             bool rmv=0;int idx;

             if(isdigit(s[i])){
                if(s[i]>'0')res[s[i]-'0']++,s[i]--;
                else res[0]++,rmv=true,idx=i;
             }
             else{
                if(s[i]=='>')dir=1;
                else dir=-1;
                if(i+dir>=0 && (s[i+dir]=='<' || s[i+dir]=='>'))rmv=true,idx=i;
             }

             i+=dir;
             if(rmv){
                s.erase(idx,1);
                if(dir==1)i--;
             }

        }
        for(int i=0;i<10;i++)cout<<res[i]<<((i==9)?"\n":" ");


    }
}
