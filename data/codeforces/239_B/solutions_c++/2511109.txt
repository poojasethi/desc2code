#include <iostream>
#include <string>
using namespace std;
int main(void)
{
 string s,t;
 int len,Q,pl,pr;
 cin>>len>>Q>>s;
 while(Q--)
 {
  cin>>pl>>pr;
  t=s.substr(pl-1,pr-pl+1);
  int cp=0,dp=1,count[10]={0};
  while(cp>=0&&cp<t.length())
  {
   if(t[cp]>='0'&&t[cp]<='9')
   {
    count[t[cp]-'0']++;
    if(t[cp]=='0')
    {
     t.erase(cp,1);
     if(!dp)
      cp--;
    }
    else
    {
     t[cp]--;
     cp+=dp?1:-1;
    }
   }
   else
   {
    if(t[cp]=='>')
    {
     dp=1;
     cp++;
     if(cp<t.length()&&(t[cp]=='>'||t[cp]=='<'))
      t.erase(--cp,1);
    }
    else
    {
     dp=0;
     cp--;
     if(cp>=0&&(t[cp]=='>'||t[cp]=='<'))
      t.erase(cp+1,1);
    }
   }
  }
  for(int i=0;i<10;i++)
   cout<<count[i]<<" ";
  cout<<endl;
 }
 return 0;
}
