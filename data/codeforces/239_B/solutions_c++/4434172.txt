#include <iostream>
#include <vector>
using namespace std;

void sim(string str) {

	int arr[]  = {1, -1};
	vector <int> vec(10, 0);
	int pos = 0, dir = 1;

	while(pos >= 0 && pos < (int)str.size()) {
		if(str[pos] == '<' || str[pos] == '>') {
			dir = arr[str[pos] == '<'];
			if(pos+dir >= 0 && pos+dir < (int)str.size()
					&& (str[pos+dir] == '<' || str[pos+dir] == '>')) {
				str.erase(pos, 1);
				if(dir == 1) pos --;
			}
			pos += dir;
		} else {
			vec[str[pos] - '0'] ++;
			if(str[pos] == '0') {
				str.erase(pos, 1);
				if(dir == 1) pos --;
			} else {
				str[pos] --;
			}
			pos += dir;
		}
	}

	for(int i=0; i<10; i++)
		cout << vec[i] << " \n"[i == 9];
}

int main() {

	int N, Q, l, r;
	string tmp;
	cin >> N >> Q >> tmp;

	for(int q = 0; q < Q; q++) {
		cin >> l >> r;
		string str = tmp.substr(l-1, r-l+1);
		sim(str);
	}

	return 0;
}
