#include <cstdio>
#include <cstring>

char s[105];
char s_[105];
int t[10];
bool v[105];

int main()
{
	int n, q;
	scanf("%d %d", &n, &q);
	
	for (int i = 0; i < n; ++i)
		scanf(" %c ", &s_[i]);
		
	for (int i = 0; i < q; ++i)
	{
		int l, r;
		scanf("%d %d", &l, &r); l--; r--;
		
		memset(t, 0, sizeof(t));
		memset(v, true, sizeof(v));
		memcpy(s, s_, sizeof(s_));

		int j = l;
		int dir = 1;
		while (j >= l && j <= r)
		{
			if (!v[j])
			{
				j += dir;
				continue;
			}
			
			if (s[j] == '>' || s[j] == '<')
			{
				int oldJ = j;
				dir = s[j] == '>' ? 1 : -1;
				j += dir;
				while (!v[j] && j >= l && j <= r)
					j += dir;
				if (!(j >= l && j <= r))
					break;
				if (s[j] == '>' || s[j] == '<')
					v[oldJ] = false;
			}
			else
			{
				t[s[j] - '0']++;
				if (s[j] == '0')
					v[j] = false;
				else
					s[j]--;
				j += dir;
			}
		}
		
		for (int i = 0; i < 10; ++i)
			printf("%d ", t[i]);
		printf("\n");
	}
	
	return 0;
}
