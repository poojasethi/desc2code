#include<iostream>
#include<string>
#include<cstring>
using namespace std;
int main(){
	int n,q;
	string s;
	cin>>n>>q>>s;
	int x[11];
	for(int i=0;i<q;i++){
		int l,r;
		cin>>l>>r;
		memset(x,0,sizeof(x));
		string sub=s.substr(l-1,r-l+1);
		int now=0,move=1;
		while(now>=0 && now<sub.size()){
			char c=sub.at(now);
			if('0'<= c && c<='9'){
				x[c-'0']++;
				if(c=='0'){
					sub.erase(now,1);
					if(move==-1){
						now+=move;
					}
				}else{
					sub[now]--;
					now+=move;
				}
			}else{
				if(c=='>'){
					move=1;
				}else{
					move=-1;
				}
				int prev=now;
				now+=move;
				if(!(now>=0 && now<sub.size())){
					break;
				}
				char c=sub.at(now);
				if(c=='>' || c=='<'){
					sub.erase(prev,1);
					if(prev<now){
						now--;
					}
				}
			}
		}
		cout<<x[0];
		for(int i=1;i<10;i++){
			cout<<" "<<x[i];
		}
		cout<<endl;
	}
}
