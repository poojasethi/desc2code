#include<iostream>
#include<string.h>
using namespace std;
char ch[105],s[104];
int arr[12];
int n,q,l,r;
int main()
{
	cin>>n>>q;
	cin>>s;
	for(int i=0;i<q;i++)
	{
		memset(arr,0,sizeof(arr));
		memcpy(ch+1,s,sizeof(s));
		ch[0]='x';
		cin>>l>>r;
		int point=l,dir=1;
		while(point>=l&&point<=r)
		{
			if(ch[point]>='0'&&ch[point]<='9')
			{
				arr[ch[point]-'0']++;
				if(ch[point]=='0')ch[point]='.';
				else ch[point]--;
				point+=dir;
				while(ch[point]=='.'&&point<=r&&point>=l)point+=dir;
			}
			else
			{
				if(ch[point]=='<')dir=-1;
				else dir=1;
				int u=point;
				point+=dir;
				while(ch[point]=='.'&&point<=r&&point>=l)point+=dir;
				if(ch[point]=='<'||ch[point]=='>')
					ch[u]='.';
			}
		}
		for(int i=0;i<9;i++)cout<<arr[i]<<' ';
		cout<<arr[9]<<endl;
	}
	return 0;
}
