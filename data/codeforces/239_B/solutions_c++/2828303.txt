#include <string>
#include <iostream>
using namespace std;
void func(string s)
{
    int c[10]={0};
    int d=s[0]=='<'?-1:1;
    int p=0, f=0;
    while (0<=p && p<s.size()) {
        if (s[p]=='>' || s[p]=='<') {
            if (f) { s.erase(p-d, 1); if(d>0)p--; }
            f=1;
            d=s[p]=='<'?-1:1;
        }
        else {
            c[s[p]-'0']++;
            if(--s[p]<'0') { s.erase(p, 1); if(d>0)p--; }
            f=0;
        }
        p+=d;
    }
    for(int i=0;i<10;++i)cout<<c[i]<<' ';
    cout<<endl;
}
int main()
{
    string s;
    int n, q; cin>>n>>q>>s;
    while (q--) {
        int l, r; cin>>l>>r;
        func(s.substr(l-1,r-l+1));
    }
    return 0;
}
