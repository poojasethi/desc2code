#include<stdio.h>
long i,j,k,a,l,r,dir,d[10],cp,dp,n,q;
char o[104],c[104];
int main(){
 scanf("%ld %ld",&n,&q);
 scanf("%s",o);
 while(q){
  scanf("%ld %ld",&l,&r);
  for(i=l;i<=r;i++) c[i]=o[i-1];
  cp=l;
  dp=1;
  for(i=0;i<10;i++) d[i]=0;
  while(1){
   if(cp<l||cp>r)
    break;
   if(c[cp]){

    if(c[cp]>='0'&&c[cp]<='9'){

     d[c[cp]-'0']++;

     c[cp]--;
     if(c[cp]<'0')
        c[cp]=0;

     cp+=dp;

    }else{

     if(c[cp]=='<'){
      dp=-1;
     }else{
      dp=1;
     }

     i=cp;
     cp+=dp;
     if(cp<l||cp>r)
        break;
     while(c[cp]==0){
        cp+=dp;
        if(cp<l||cp>r)
            break;
     }
     if(c[cp]=='<'||c[cp]=='>')
        c[i]=0;

    }


   }else{
    cp+=dp;
   }
  }

  for(i=0;i<10;i++) printf("%ld ",d[i]);
  printf("\n");
  q--;
 }
}
