#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
	int len,n;
	string s;
	while(~(scanf("%d%d",&len,&n)))
	{
		cin >> s;
		int dp,cp,l,r,ans[10];
		while(n--)
		{
			scanf("%d%d",&l,&r);
			int flag=0,pos;
			string temp=s;
			dp=1;
			memset(ans,0,sizeof(ans));
			l--;r--;
			for(cp=l;cp>=l&&cp<=r;cp+=dp)
			{
				if(temp[cp]>='0'&&temp[cp]<='9')
				{
					ans[temp[cp]-'0']++;
					if(temp[cp]=='0')
						temp[cp]=' ';
					else 
						temp[cp]--;
					flag=0;
				}
				else if(temp[cp]=='<')
				{
					dp=-1;
					if(flag){
						temp[pos]=' ';
					}
					flag=1;
					pos=cp;
				}
				else if(temp[cp]=='>'){
					dp=1;
					if(flag){
						temp[pos]=' ';
					}
					flag=1;
					pos=cp;
				}
			}
			for(int i=0;i<=9;i++)
			{
				if(i)
					cout << " ";
				cout << ans[i];
			}
			cout << endl;
		}
	}
	return 0;
}
