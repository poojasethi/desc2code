#include <stdio.h>
#include <algorithm>
#include <string.h>
#include <iostream>

using namespace std;

#define sz(x) ((x).size())

int n,q,cnt[11];
string str;

void solve(string str)
{
	//cout<<str<<endl;
	int l=0,r=1;
	while(sz(str))
	{
		//cout<<l<<" "<<str[l]<<endl;
		if(l>=sz(str) || l<0) break;
		if(str[l]<='9' && str[l]>='0')
		{
			cnt[str[l]-'0']++;
			if(str[l]=='0') 
			{
				str.erase(l,1);
				if(r==-1) l--;
			}
			else str[l]--,l+=r;
		}
		else
		{
			if(str[l]=='>') r=1;
			else r=-1;
			int tmp=l;
			l+=r;
			if(l>=sz(str) || l<0) break;
			if(str[l]=='>' || str[l]=='<')
			{
				str.erase(tmp,1);
				if(r==1) l--;
			}
		}
	}
}

int main()
{
	cin>>n>>q>>str;
	for(int i=0;i<q;i++)
	{
		int l,r;
		cin>>l>>r;l--,r--;
		memset(cnt,0,sizeof(cnt));
		solve(str.substr(l,r-l+1));
		for(int j=0;j<10;j++)
			printf("%d ",cnt[j]);
		puts("");
	}
	return 0;
}
