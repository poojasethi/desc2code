#include<iostream>
#include<string>
#include<vector>
#include<cctype>
using namespace std;
string str(120,' ');
bool move(int &c,bool d,int L,int R)
{
	if(d)
		c++;
	else
		c--;
	if(c>=L&&c<=R)
		return true;
	else
		return false;
}
void deal(int L,int R)
{
	string s(str);
	int a[10]={0};
	int c=L;
	bool d=1;
	while(c>=L&&c<=R)
	{
		if(s[c]==' ')
		{
			if(!move(c,d,L,R))
				break;
		}
		else if(isdigit(s[c]))
		{
			a[s[c]-'0']++;
			if(s[c]=='0')
				s[c]=' ';
			else
				s[c]--;
			if(!move(c,d,L,R))
				break;
		}
		else
		{
			if(s[c]=='>')
				d=1;
			else
				d=0;
			int p=c;
			while(1)
			{
				if(!move(c,d,L,R))
					goto End;
				if(s[c]!=' ')
					break;
			}
			if(s[c]=='>'||s[c]=='<')
				s[p]=' ';
		}
	}
	End:
	for(int i=0;i<10;i++)
		cout<<a[i]<<" ";
	cout<<endl;
}
int main()
{
	int n,q;
	cin>>n>>q;
	cin>>&str[1];
	while(q--)
	{
		int L,R;
		cin>>L>>R;
		deal(L,R);
	}
	return 0;
}
