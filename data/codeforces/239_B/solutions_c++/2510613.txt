#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;

const int MAX=100+10;

int n,q,del[MAX],ans[MAX];
char s[MAX],c[MAX];

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
#endif
	scanf("%d %d",&n,&q);
	scanf("%s",s+1);
	int i,j,l,r;
	for(i=1;i<=q;++i)
	{
		scanf("%d %d",&l,&r);
		for(j=1;j<=n;++j)c[j]=s[j];
		memset(del,0,sizeof del);
		memset(ans,0,sizeof ans);
		int now=l,dir=1;
		while(1)
		{
			if(c[now]<='9' && c[now]>='0')
			{
				++ans[c[now]-'0'];
				if(c[now]=='0')del[now]=1;
				else --c[now];
			}
			int flag=0;
			if(c[now]=='<')flag=now,dir=-1;
			if(c[now]=='>')flag=now,dir=1;

			now+=dir;
			while(now<=r && now>=l && del[now])now+=dir;
			if(now>r || now<l)break;
			if( ( c[now]=='<' || c[now]=='>' ) && flag)
				del[flag]=1;
		}
		for(j=0;j<10;++j)
			printf("%d ",ans[j]);
		printf("\n");
	}
	return 0;
}
