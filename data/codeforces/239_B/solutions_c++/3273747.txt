#include <string>
#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>
#define F first
#define S second
#define PB push_back
#define MP make_pair
#define LL long long
#define FOR(it, c) for(__typeof((c).begin())it=(c).begin();it!=(c).begin();it++)
using namespace std;

int n, q;
string s;

int main ()
{
	cin >> n >> q >> s;
	while(q--)
	{
		int l, r, cnt[10];
		fill(cnt, cnt+10, 0);
		cin >> l >> r; l--, r--;
		int p = l, d = 1, fl = 0, id = 0;
		string t = s;
		while(p >= l && p <= r)
		{
			if(t[p] == '0'-1)
			{
				p += d;
				continue;
			}
			if(t[p] >= '0' && t[p] <= '9')
			{
				fl = 0;
				cnt[t[p]-'0'] ++;
				t[p] --;
				p += d;
				continue;
			}
			if(t[p] == '>' || t[p] == '<')
			{
				t[p] == '>'? d = 1: d = -1;
				if(fl) t[id] = '0'-1;
				fl = 1; id = p;
				p += d;
			}
		}
		for(int i = 0; i <= 9; i++)
			printf("%d%c", cnt[i], i == 9?'\n':' ');
	}
}
