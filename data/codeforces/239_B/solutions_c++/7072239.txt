#include <iostream>
#include <string.h>
#include <cstdio>
using namespace std;
int c[10],a[105],b[105];
int main(){
	int m,n,l,r;
	cin >> m >> n;
	char x;
	string y;
	cin >> y;

	while(n--){
		memset(c,0,sizeof(c));	
		for (int i=1;i<=m;i++)
		{
			x=y[i-1];
			if(x=='>') {b[i]=1;a[i]=0;}
			else if(x=='<') {b[i]=-1;a[i]=0;}
			else {b[i]=0;a[i]=x-'0';}
		}
		cin >> l >> r;
		int tt=l,dp=1,next;
		while(tt>=l&&tt<=r){
		//cout << tt <<endl;
			if(b[tt]) {
				dp=b[tt];
			next=tt+dp;		
			while(a[next]==-1&&next>=l&&next<=r) next+=dp;
			if(b[next]) { b[tt]=0;a[tt]=-1;}	
			}
			else{
				c[a[tt]]++;
				a[tt]--;
				next=tt+dp;		
			while(a[next]==-1&&next>=l&&next<=r) next+=dp;
			}				
			tt=next;
		}
		for(int i=0;i<9;i++)
		cout << c[i] <<" ";
		cout << c[9] << endl;
	}
}
