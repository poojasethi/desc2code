#include <iostream>
#include <string>

using namespace std;

int main()
{
	int n, q;
	cin >> n >> q;
	string t;
	cin >> t;
	for (int p = 0; p < q; p++)
	{
		int mark[10] = {};
		int l, r;
		cin >> l >> r;
		string s = t;
		l--;
		r--;
		int cp = l, dp = +1;
		bool ld = false;
		int lp = -1;
		while (l <= cp && cp <= r)
		{
			if (s[cp] == '.')
			{
				cp += dp;
				continue;
			}
			if (s[cp] == '<' || s[cp] == '>')
			{
				if (ld)
					s[lp] = '.';
				if (s[cp] == '>')
					dp = +1;
				else
					dp = -1;
				ld = true;
				lp = cp;
			}
			else
			{
				int d = s[cp] - int('0');
				mark[d]++;
				d--;
				if (d >= 0)
					s[cp] = char(d + int('0'));
				else
					s[cp] = '.';
				ld = false;
			}
			cp += dp;
		}
		for (int i = 0; i < 10; i++)
			cout << mark[i] << " ";
		cout << endl;
	}
}
