#include <string>
#include <cstring>
#include <iostream>
using namespace std;

int a[10];

void query(string s)
{
	int dp = 1, cp = 0;
	while (0 <= cp && cp < s.size())
	{
		int pos = -1;
		if (!(s[cp] == '>' || s[cp] == '<'))
		{
			a[s[cp]-'0']++;
			if (s[cp]-'0' == 0)
				pos = cp;
			else
				s[cp] = s[cp]-1;
			cp += dp;
		}
		else
		{
			if ((dp>0&&s[cp]=='<') || (dp<0 && s[cp]=='>'))
				dp *= -1;
			cp += dp;
			if (s[cp] == '>' || s[cp] == '<')
				pos = cp-dp;
		}
		if (pos >= 0)
		{
			if (cp > pos)
				cp--;
			s = string(s.begin(), s.begin()+pos)+string(s.begin()+pos+1, s.end());
		}
	}
}

int main()
{
	int n, q;
	cin >> n >> q;
	string s;
	cin >> s;
	for (int i=0; i<q; i++)
	{
		memset(a, 0, sizeof(a));
		int x, y;
		cin >> x >> y;
		x--;
		query(string(s.begin()+x, s.begin()+y));
		for (int k=0; k<10; k++)
			cout << a[k] << ' ';
		cout << endl;
	}
}
