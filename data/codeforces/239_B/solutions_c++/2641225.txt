#include <stdio.h>
#define F(I,B,E) for(int I=B; I<E; ++I)

void delt(char *s, int *n) {
  --*n;
  F(i,0,*n)
    s[i]=s[i+1];
}

void interpreter(char *s, int *k, int n) {
  char t[111],cp=0,dp=1;
  F(i,0,n)
    t[i]=s[i];
  while(cp>=0 && cp<n)
    if(t[cp]=='<' || t[cp]=='>') {
      dp=(t[cp]=='>')-(t[cp]=='<');
      if(cp+dp<0 || cp+dp>=n) 
        break;
      if(t[cp+dp]=='<' || t[cp+dp]=='>')
        delt(t+cp,&n),
        cp-=(dp<0);
      else
        cp+=dp;
    }
    else {
      ++k[t[cp]-'0'];
      if(t[cp]=='0')
        delt(t+cp,&n),
        cp-=(dp<0);        
      else
        --t[cp],
        cp+=dp;
    }
}

main() {
  int n,q;
  scanf("%d%d",&n,&q);
  char s[111];
  scanf("\n%s",s);
  F(i,0,q) {
    int l,r,k[10]={};
    scanf("%d%d",&l,&r);
    interpreter(s+l-1,k,r-l+1);
    F(p,0,10)
      printf("%d%c",k[p],p<9?' ':'\n');
  }
  return 0;
}
