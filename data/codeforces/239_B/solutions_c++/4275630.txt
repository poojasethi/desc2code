#include<iostream>
#include<string>
using namespace std;

int b[10];

int main()
{
	int n,q;
	cin >> n >> q;
	string s;
	cin >> s;
	char a[101];
	while(q-->0)
	{
		bool d=1;
		int f=-1;
		for(int i=0; i<10; i++)
			b[i]=0;
		int l,r;
		cin >> l >> r;
		for(int i=--l; i<r; i++)
			a[i-l]=s[i];
		int t=0;
		while(t>=0&&t<r-l)
		{
			if(a[t]>='0'&&a[t]<='9')
			{
				b[a[t]-'0']++;
				if(a[t] == '0')
					a[t]=0;
				else
					a[t]--;
				f=-1;
			}
			else if(!a[t])
			{
			}
			else if(f>-1)
			{
					a[f]=0;
				if(a[t]=='<')
					d=0;
				else
					d=1;
				f=t;
			}
			else
			{
				if(a[t]=='<')
					d=0;
				else
					d=1;
				f=t;
			}
			if(d)
				t++;
			else
				t--;
		}
		for(int i=0; i<9; i++)
			cout << b[i] << " ";
		cout << b[9] <<endl;
	}
}
