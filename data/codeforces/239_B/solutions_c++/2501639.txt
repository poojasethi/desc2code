#include <cstdio>
#include <cstring>
char str[110];
int n,q,l,r;
int vis[10],del[110];
int go(int cur,int d)
{
	cur = cur + d;
	while( cur<=r && cur>=l && del[cur]) cur+=d;
	return cur;
}
void work(int l,int r)
{
	int i,d=1,t;
	char tmp[110];
	memcpy(tmp,str,sizeof(str));
	memset(vis,0,sizeof(vis));
	memset(del,0,sizeof(del));
	for ( i=l; i<=r && i>=l ; )
	{
		if ('0'<= tmp[i] && tmp[i] <='9'){
			vis[tmp[i]-'0'] ++ ;
			if (tmp[i]=='0')del[i] = 1;
			else	tmp[i]--;
			i = go(i,d);
		}
		else
		{
			if (tmp[i]=='>') d = 1;
			else if (tmp[i]=='<') d = -1;
			t = go(i,d);
			if (tmp[t]=='>' || tmp[t]=='<')del[i] = 1;
			i = t;
		}
	}
	for ( i=0 ; i<9 ; i++ )printf("%d ",vis[i]);
	printf("%d\n",vis[i]);
}
int main()
{
	int i;
//	freopen("test.txt","r",stdin);
	scanf("%d%d%s",&n,&q,str);

	for ( i=0 ; i<q ; i++ )
	{
		scanf("%d%d",&l,&r);
		l--,r--;
		work(l,r);
	}
	return 0;
}
