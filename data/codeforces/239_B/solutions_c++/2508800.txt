#include <iostream>
#include <cstring>
#include <cstdio>
#include <set>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

int main()
{
	int n,q;
	cin>>n>>q;
	string seq;
	cin>>seq;
	for(int i=0,l,r;i<q;i++)
	{
		cin>>l>>r;
		l--;r--;
		int cnt[10]={0};
		string tmp(seq.begin()+l,seq.begin()+r+1);
		int cur=0;
		int pre=-1;
		int dp=1;
		while(cur>=0&&cur<tmp.length())
		{
			if(tmp[cur]=='<')
			{
				if(cur==0)break;
				dp=-1;
				if(pre!=-1&&(tmp[pre]=='<'||tmp[pre]=='>'))
					tmp[pre]='@';
			}
			else if(tmp[cur]=='>')
			{
				if(cur==tmp.length()-1)break;
				dp=1;
				if(pre!=-1&&(tmp[pre]=='<'||tmp[pre]=='>'))
					tmp[pre]='@';
			}
			else
			{
				cnt[tmp[cur]-'0']++;
				if(tmp[cur]=='0')
					tmp[cur]='@';
				else
					tmp[cur]--;
			}
			pre=cur;
			cur+=dp;
			while(cur>=0&&cur<tmp.length()&&tmp[cur]=='@')
				cur+=dp;
		}
		for(int i=0;i<10;i++)
			cout<<cnt[i]<<" ";
		cout<<endl;
	}	
	return 0;
}
