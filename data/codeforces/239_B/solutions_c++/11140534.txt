#include<cstdio>
#include<cstring>
#include<cctype>
const int maxn=110;
int n,q;
int ans[10];
char s[maxn],ss[maxn];
void solve(int l,int r)
{
    memset(ans,0,sizeof(ans));
    for (int i=l;i<=r;i++) s[i]=ss[i];
    int cp=l,dp=1,f=0,p;
    while (l<=cp&&cp<=r)
   {
       char&ch=s[cp];
       if (ch==0) cp+=dp;
       else if (isdigit(ch))
       {
           ans[ch-'0']++;
           if(ch!='0') ch--;
           else ch=0;
           cp+=dp;
           f=0;
        }
        else
        {
            if (ch=='<') dp=-1;
            else dp=1;
            if (f) s[p]=0;
            p=cp;
            f=1;
            cp+=dp;
        }
   }
   for(int i=0;i<10;i++)
   printf("%d%c",ans[i]," \n"[i==9]);
}
 int main()
 {
     while (~scanf("%d%d",&n,&q))
     {
         scanf("%s",ss+1);
         while (q--)
         {
             int l,r;
             scanf("%d%d",&l,&r); solve(l,r);
         }
    }
    return 0;
}
