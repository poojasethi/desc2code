#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t,c,v,d[10];
  char a[101],f[101];
  bool s[100];
  cin>>q>>t>>f;
  for(;t;t--){
   for(w=0;w<q;w++)
    a[w]=f[w];
   cin>>e>>r;
   e--;r--;
   for(w=e;w<=r;w++)
    s[w]=1;
   for(w=0;w<10;w++)
    d[w]=0;
   c=q;
   v=1;
   for(w=e-1;;){
    w+=v;
    while(((e<=w)&&(w<=r))&&(!s[w]))
     w+=v;
    if(!((e<=w)&&(w<=r)))
     break;
    if(((a[w]=='<')||(a[w]=='>'))&&((a[c]=='<')||(a[c]=='>')))
     s[c]=0;
    if((48<=a[w])&&(a[w]<=57)){
     d[a[w]-48]++;
     a[w]--;
     if(a[w]==47)
      s[w]=0;}
    if(a[w]=='<')
     v=-1;
    if(a[w]=='>')
     v=1;
    c=w;}
   for(w=0;w<10;w++)
    cout<<d[w]<<" ";
   cout<<"\n";}
  return 0;}
