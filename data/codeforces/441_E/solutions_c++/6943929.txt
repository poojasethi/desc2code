#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;

double dp[210][250][250];
double P;
int K,X;

int main()
{
	cin>>X>>K>>P;
	P=P/100.;
	dp[0][0][0]=1;
	double ans=0.;
	for(int k=0;k<K;++k)
		for(int p=0;p<35+K;++p)
			for(int s=0;s<35+K;++s)
			{
				if(s % 2 != 0)
				{
					ans+=dp[k][p][s]*P*p;
				}
				else
				{
					dp[k+1][p+1][s>>1]+=dp[k][p][s]*P;
				}
				dp[k+1][p][s+1]+=dp[k][p][s]*(1-P);
			}
	for(int p=0;p<35+K;++p)
		for(int s=0;s<35+K;++s)
		{
			ans+=dp[K][p][s]*(p+__builtin_ctz(X+s));
		}
	printf("%.6f\n",ans);
	return 0;
}
