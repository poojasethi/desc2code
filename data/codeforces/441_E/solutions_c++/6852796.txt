#include <stdio.h>
#include <math.h>
#include <algorithm>

double s[201][201];

int zeros(int x) {
  int s = 0;
  while ((x & 1) == 0) {
    s++;
    x = x >> 1;
  }
  return s;
}

int main() {
  int x, kmax, pint;
  scanf("%d %d %d", &x, &kmax, &pint);
  double p = pint / 100.0;
  double q = 1 - p;

  // populate s[0][da]
  for (int da = 0; da <= kmax; da++) {
    s[0][da] = zeros(x+da);
  }

  for (int k = 1; k <= kmax; k++) {
    for (int da = 0; da <= kmax; da++) {
      // for b == 1
      double s0 = 0;
      double qa = 1.0;
      for (int a = 0; a <= k-1; a++) {
        if ( (a+da) % 2 == 0) {
          int da2 = (a+da) / 2;
          s0 += p * qa * (1+s[k-1-a][da2]);
        } else {
          // s0 += 0
        }
        qa *= q;
      }
      // for b == 0
      s0 += zeros(x+k+da) * qa;
      s[k][da] = s0;
    }
  }

  printf("%0.10f\n", s[kmax][0]);
}
