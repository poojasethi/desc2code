#include <cstdio>
#pragma GCC optimize("O2")
using namespace std;

#define mark ((1<<9)-1)

int X, K, P, t;
double p;
double dp[202][1<<9];
double ze[202][1<<9];

int main(void) {
    scanf("%d%d%d", &X, &K, &P);
    p=P/100.0;
    dp[0][X&mark]=1.0;
    t=X>>8;
    if (t%2)
        t++;
    if (t) {
        while (t && t%2==0) {
            ze[0][X&mark]++;
            t/=2;
        }
    }
    for (int k=0; k<K; k++)
        for (int m=0; m<1<<9; m++) {
            double pr=dp[k][m]*p;
            int nm=(m<<1)&mark;
            dp[k+1][nm]+=pr;
            if (bool(m&(1<<8))==bool(m&(1<<7)))
                ze[k+1][nm]+=ze[k][m]*p+pr;
            else
                ze[k+1][nm]+=pr;
            pr=dp[k][m]*(1-p);
            nm=(m+1)&mark;
            dp[k+1][nm]+=pr;
            ze[k+1][nm]+=ze[k][m]*(1-p);
        }
    double a=.0;
    for (int m=0; m<1<<9; m++)
        if (!m) {
            a+=ze[K][0]+8*dp[K][m];
        } else {
            int c=0;
            for (int v=m; v%2==0; v/=2)
                c++;
            a+=c*dp[K][m];
        }
    printf("%.20f\n", a);
}
