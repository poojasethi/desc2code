double dp[257][203][2][233];
int x,k,p,i;

double f(int mask,int x,bool w,int kob){
	double &r = dp[mask][x][w][kob];
	if(r>=0)
		return r;
	if(x == k+1){
		if(mask){
			for(int i=0 ; i<8 ; i++)
				if(bit(mask,i))
					return r = i;
		}
		if(w==1)
			return r = 8;
		return r = 8+kob;
	}
	if(bit(mask,7) == w)
		r = p*f(mask*2%256,x+1,w,kob+1)/100;
	else
		r = p*f(mask*2%256,x+1,!w,1)/100;
	
	mask += 1;

	if(bit(mask,8)){
		if(w == 1)
			r += (100-p)*f(mask^256,x+1,0,kob)/100;
		else
			r += (100-p)*f(mask^256,x+1,1,1)/100;
	}
	else{
		r += (100-p)*f(mask,x+1,w,kob)/100;
	}

	return r;
}

int main(){
	cin >> x >> k >> p;

	memset(dp,-1,sizeof dp);

	for(i=9 ; i<=200 ; i++)
		if(bit(x,i) != bit(x,8))
			break;

	cout << setprecision(10) << fixed << f(x%256,1,bit(x,8),i-8) << endl;
}
