#include <cstdio>
using namespace std;

double dp[201][201];
int X, K, P;

double inline f(int S) {
    double re = 0.0;
    while(!(S & 1)) { S >>= 1; re += 1.0; }
    return re;
}

int main() {
    scanf("%d%d%d", &X, &K, &P);
    double p = (double)P / 100.0;
    for (int i = 0; i <= K; i++)
        dp[0][i] = f(X + i);
    for (int i = 1; i <= K; i++)
        for (int j = 0; j <= K; j++) {
        dp[i][j] = (1 - p) * (dp[i - 1][j + 1]);
        if (j % 2 == 0) dp[i][j] += p * (dp[i - 1][j / 2] + 1);
        }

    printf("%.10lf", dp[K][0]);
    return 0;
}
