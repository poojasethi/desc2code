double f[MAXN][MAXN];

int main()
{

	cin >> x >> n >> p;
	//predict what will happen...>.<
	p /= 100;
	for (int i = 0; i <= n; ++i)
		for (int j = 0; j <= n; ++j)
			if (!i)
				for (int a = x + j; !(a & 1); ++f[i][j], a >>= 1);
			else
			{
				f[i][j] += f[i - 1][j + 1] * (1.0 - p);
				if (!(j & 1))
					f[i][j] += (f[i - 1][j / 2] + 1) * p;
			}
	printf("%.10lf\n", f[n][0]);
	
	fclose(stdin);
	fclose(stdout);
	return 0;
}
