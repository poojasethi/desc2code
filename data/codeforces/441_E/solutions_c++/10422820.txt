#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int x,k,p;
double f[400][400];
int main()
{
	scanf("%d %d %d",&x,&k,&p);
	memset(f,0,sizeof(f));
	double P=p/100.0;
	for (int i=0;i<=k;i++)
	{
		int tt=x+i;
		while (tt % 2 == 0)
		{
			tt >>= 1;
			f[0][i]+=1;	
		}
	}
	
	for (int i=1;i<=k;i++)
	{
	  for (int j=0;j<=k;j++)
	  {
	  	f[i][j]+=(f[i-1][j+1])*(1.0-P);
	  	if (j % 2==0)
		     f[i][j]+=(f[i-1][j>>1]+1.0)*P;
	  }
    }
	printf("%.10f\n",f[k][0]);
	return 0;
}
