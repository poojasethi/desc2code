#include <cstdio>

int main() {
    static double dp[201][201]={0.};

    int n, k, a;
    scanf("%d %d %d", &n, &k, &a);
    double p = a / 100.;

    for(int i = 0; i <= k; i++) {
        for(int j = 0; j <= k; j++) {
            if(i == 0) {
                for(int x = n + j; x % 2 == 0; x /= 2) dp[0][j]+=1.0;
            } else {
                dp[i][j] += (dp[i-1][j+1] * (1-p));
                if(j % 2 == 0) dp[i][j] += (p * (dp[i-1][j / 2] + 1));
            }
        }
    }
    printf("%.10lf", dp[k][0]);
}
