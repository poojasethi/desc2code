#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

const int N = 205;

double dp[N][N];

int main() {
    int x, k;
    double p;
    while (cin >> x >> k >> p) {
        p /= 100;
        memset(dp, 0, sizeof (dp));
        for (int i = 0; i <= k; i++) {
            for (int j = x + i; j % 2 == 0; j >>= 1) dp[0][i]++;
        }
        for (int i = 1; i <= k; i++) {
            for (int j = 0; j <= k; j++) {
                dp[i][j] += (1 - p) * dp[i - 1][j + 1];
                if (j % 2 == 0) dp[i][j] += p * (dp[i - 1][j >> 1] + 1);
            }
        }
        printf("%.9f\n", dp[k][0]);
    }
    return 0;
}
