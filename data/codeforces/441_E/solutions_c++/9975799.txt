#include <cmath>
#include <cstdio>
#include <vector>
#include <string>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
#define ll long long
#define cls(a) memset(a,0,sizeof(a))
#define rise(i,a,b) for(int i=(a);i<=(b);i++)
#define fall(i,a,b) for(int i=(a);i>=(b);i--) 
double dp[2][201], rate;
int x, k, p; 
int main()
{
	cin >> x >> k >> p;
	rate = p * 1.0 / 100.0;
	rise( i , 0 , k ) for( int j = x + i ; !( j & 1 ) ; j >>= 1 ) dp[0][i] ++; 
	int halfk, flag;
	rise( i , 1 , k )
	{
		flag = i & 1;
		cls( dp[flag] );
		halfk = ( k - i + 1 ) >> 1;
		rise( j , 0 , k )
		{
			if( j > 0 ) dp[flag][j-1] += dp[!flag][j] * ( 1.0 - rate ); 
			if( j <= halfk ) dp[flag][j<<1] += ( dp[!flag][j] + 1.0 ) * rate;
		}
	}
	printf( "%.10f\n" , dp[flag][0] );
}
