#include<cstdio>
#include<iostream>
#define maxn 1000
using namespace std;
double dp[maxn][maxn];
int main()
{
    int x,k;
    double p;
    scanf("%d%d%lf",&x,&k,&p);
    p=1.0*p/100;
    for(int i=0;i<=k;i++)
        for(int j=i+x;(j&1)==0;j>>=1)dp[0][i]++;
    for(int i=1;i<=k;i++)
    {
        for(int j=0;j<=k;j++)
        {
            if(j!=0)dp[i][j-1]+=(dp[i-1][j])*(1-p);
            if(j*2<=k)dp[i][j*2]+=(dp[i-1][j]+1)*p;
        }
    }
    printf("%.6lf\n",dp[k][0]);
    return 0;
}
