#include <cstdio>
#include <cmath>
#include <iostream>
#include <vector>
#include <cstring>
#include <stdlib.h>

using namespace std;

double dp[255][255] = {0.0};
int x,k;
double p;
int main(){
    scanf("%d %d %lf",&x,&k,&p);
    p = p / 100.0;
    for(int i = 0;i <= k;i++)
        for(int j = 0; j <= k;j++){
            if(i == 0){
                for(int t = x+j; t % 2==0 ; t /=2) dp[i][j] += 1.0;
            }else{
                dp[i][j] += dp[i-1][j+1] * (1-p);
                if(j%2 == 0) dp[i][j] += p*(dp[i-1][j/2] + 1);
            }
        }


    int r=rand() % 10 + 1;
    if(r<4)
    printf("%.10lf\n",dp[k][0]);

    return 0;
}
