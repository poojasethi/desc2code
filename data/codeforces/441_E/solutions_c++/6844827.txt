#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

typedef long double ld;

int x, n, p;
int a[1005];
ld f[205][205][205];

int check(int m){
    int s= 0;
    while(m%2==0){
        s++;
        m/= 2;
    }
    return s;
}

int main(){
    ios_base::sync_with_stdio(false);
    int i, j, k;
    ld xx, yy;
    cin >> x >> n >> p;
    xx= (ld)p/100.0; yy= (ld)(100-p)/100.0;
    for(i=0;i<=1000;i++) a[i]= check(x+i);
    for(j=0;j<=201;j++){
        for(k=0;k<=201;k++){
            f[0][j][k]= (ld)(a[j]+k);
        }
    }

    for(i=1;i<=n;i++){
        for(j=0;j<=200;j++){
            for(k=0;k<=200;k++){
                f[i][j][k]= f[i-1][j+1][k]*yy;
                if(j%2==0) f[i][j][k]+= (f[i-1][j/2][k+1]*xx);
                else f[i][j][k]+= ((ld)k*xx);
            }
        }
    }

    cout.precision(9);
    cout << fixed << f[n][0][0] << endl;

    return 0;
}
