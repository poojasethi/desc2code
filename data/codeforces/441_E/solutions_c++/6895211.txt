#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <assert.h>
#include <vector>
using namespace std;
#define eps 10-8

double dp[202][202];
int main ()
{
    int x, n, pp;
    scanf("%d%d%d", &x, &n, &pp);
    double p = 1.0*pp/100;
    dp[0][0] = 1.0;
    double ans = 0.0;
    for(int i=0; i<n; ++i)
    {
        for(int j=0; j<=n; ++j)
        {
            dp[i+1][j+1] += dp[i][j]*(1-p);
            if(j%2 == 0)
            {
                dp[i+1][j/2] += dp[i][j]*p;
                ans += dp[i][j]*p;
            }
        }
    }
    for(int j=0; j<=n; ++j)
    {
        int xx = x+j;
        while(xx%2 == 0)
        {
            xx /= 2;
            ans += dp[n][j];
        }
    }
    printf("%.10lf\n", ans);
    return 0;
}
