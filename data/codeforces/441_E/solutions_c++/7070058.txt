#include<cstring>
#include<cstdio>
#include<algorithm>
#include<string>
#include<iostream>
using namespace std;
double dp[205][205][205][2];
int gao(int x)
{
    int sum=0;
    while(x)
    {
        if(x%2==0)
            sum++;
        else
            break;
            x=x/2;
    }
    return sum;
}
int main()
{
    int x,k,p;
    cin>>x>>k>>p;
    double p1=(double)p/100.0;
    memset(dp,0,sizeof(dp));
    dp[0][0][0][0]=1;
    for(int i=0;i<=k;i++)
    {
        for(int j=0;j<=k;j++)
        {
            for(int r=0;r<=k;r++)
            {
                for(int o=0;o<=1;o++)
                {
                    dp[i+1][j+1][r][o]+=dp[i][j][r][o]*(1-p1);
                    if(j%2==1)
                    dp[i+1][j/2][r][1]+=dp[i][j][r][o]*p1;
                    else if(o==0)
                    dp[i+1][j/2][r+1][0]+=dp[i][j][r][o]*p1;
                    else
                    dp[i+1][j/2][r][1]+=dp[i][j][r][o]*p1;
                }
            }
        }
    }
    double sum=0;
    for(int i=0;i<=k;i++)
    {
        for(int j=0;j<=k;j++)
        {
            sum+=dp[k][i][j][0]*(j+gao(x+i));
            sum+=dp[k][i][j][1]*j;
        }
    }
    printf("%.10f\n",sum);
}
