const int MAXN = 2e2+10;
double dp[MAXN][MAXN],pb;
bool vis[MAXN][MAXN];
int x;
double solve(int steps, int add){
    if(vis[steps][add]) return dp[steps][add];
    vis[steps][add]=true;
    double & ret = dp[steps][add] = 0.00;
    if(steps==0){
        int num = x+add; 
        while(num%2==0) { ret++; num/=2; }
    }
    else{
          if(add%2==0) ret=pb*(1+solve(steps-1,add/2));
          ret+=(1.00-pb)*(solve(steps-1,add+1)); 
    }
    return ret;
}

   printf("%.9lf\n",solve(steps, 0));
