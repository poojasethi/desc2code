#define REP(a,b,c) for(int a=b;a<c;a++)
#define asd(x)              cout<<__LINE__<<" :: "<<#x<< ": "<<x<<endl;
#define asdf(x, y)          cout<<__LINE__<<" :: "<<#x<< ": "<<x<<" | "<<#y<< ": "<<y<<endl;
typedef pair<int,int> ii;
typedef long long LL;

int vis[220][220];
double dp[220][220], prob;
int x, k, p;

double rec(int moresteps, int add){
    if(vis[moresteps][add]) return dp[moresteps][add];
     vis[moresteps][add] = 1;

     double &ret = dp[moresteps][add];
     if(moresteps == 0){
         int yolo = x + add;
         while(yolo % 2 == 0) yolo /= 2, ret++;
         return ret;
     }

     if(add % 2 == 0) ret += prob*(1.0 + rec(moresteps - 1, add/2));
     ret += (1.0 - prob)*(rec(moresteps - 1, add + 1));
     return ret;
}

int main(){
    cin >> x >> k >> p;
    prob = (double)p/100.0;
    printf("%.12lf\n", rec(k, 0));
    return 0;
}
