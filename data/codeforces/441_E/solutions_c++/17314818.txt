int x,k,q,w,s;

double p;
double dp[205][265][2][265];

double f(int x,int mask,bool w,int cnt){

	if(x==0){
		if(mask)
			for(int i=0 ; i<8 ; i++)
				if(mask&(1<<i))
					return i;
		if(w)
			return 8;
		return 8+cnt;
	}

	double &r=dp[x][mask][w][cnt];

	if(r>=0)
		return r;

	r=0;

	if(mask+1<256)
		r+=f(x-1,mask+1,w,cnt)*(1-p);
	else {
		if(!w)
			r+=f(x-1,0,1,1)*(1-p);
		else r+=f(x-1,0,0,cnt)*(1-p);
	}

	if(mask&(1<<7)){
		if(w)
			r+=f(x-1,mask*2%256,1,cnt+1)*p;
		else r+=f(x-1,mask*2%256,1,1)*p;
	}
	else {
		if(!w)
			r+=f(x-1,mask*2,0,cnt+1)*p;
		else r+=f(x-1,mask*2,0,1)*p;
	}
		
	return r;
}

main(){

	scanf("%d %d %lf",&x,&k,&p);

	for(int i=0 ; i<205 ; i++)
		for(int j=0 ; j<265 ; j++)
			for(int k=0 ; k<2 ; k++)
				for(int l=0 ; l<265 ; l++)
					dp[i][j][k][l]=-1;

	p/=100;

	w=((x&(1<<8))?1:0);

	int i=9;

	while(i<=200 and ((x&(1<<i))?1:0)==w)
		i++;

	printf("%.8lf",f(k,x%256,w,i-8));
}
