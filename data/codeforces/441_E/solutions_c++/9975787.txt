#include <cmath>
#include <cstdio>
#include <vector>
#include <string>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
#define ll long long
#define cls(a) memset(a,0,sizeof(a))
#define rise(i,a,b) for(int i=(a);i<=(b);i++)
#define fall(i,a,b) for(int i=(a);i>=(b);i--) 
double dp[201][201], rate;
int x, k, p; 
int main()
{
	cin >> x >> k >> p;
	rate = p * 1.0 / 100.0;
	rise( i , 0 , k ) for( int j = x + i ; !( j & 1 ) ; j >>= 1 ) dp[0][i] ++; 
//	rise( i , 0 , k ) cout << dp[0][i] << endl;
	int halfk = k >> 1;
	rise( i , 1 , k )
	rise( j , 0 , k )
	{
		if( j > 0 ) dp[i][j-1] += dp[i-1][j] * ( 1.0 - rate ); 
//		if ( j > 0 )cout << i << " " << " " << j - 1   << " " <<  dp[i][j-1] << endl;
		if( j <= halfk ) dp[i][j<<1] += ( dp[i-1][j] + 1.0 ) * rate;
	}
	printf( "%.10f\n" , dp[k][0] );
}
