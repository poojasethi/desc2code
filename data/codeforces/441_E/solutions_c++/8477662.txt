#include<cstdio>
#include<cmath>
#include<cstring>
const int K=200;
double f[K+5][K+5];
int main()
{
    int n,x,k,i,j,bm,l,cnt=0;
    double p;
    scanf("%d%d%lf",&x,&k,&p);
    p/=100.0;
    for(i=0;i<=k;++i) for(j=x+i;!(j&1);j>>=1) f[0][i]+=1.0;
    for(i=1;i<=k;++i)
        for(j=0;j<=k;++j)
    {
        f[i][j]=(1.0-p)*f[i-1][j+1];
        if(!(j&1)) f[i][j]+=p*(f[i-1][j>>1]+1.0);
    }
    printf("%0.6lf",f[k][0]);
}
