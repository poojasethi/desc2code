#include <stdio.h>
#include <time.h>
#include <algorithm>
using namespace std;
double dp[2][256]={{0}};
int main()
{
	int i,j;
	int x,k,p;
	int r;
	int yn;
	double ans=0,ans2;
	double pd,pp;
	int next=1;
	srand(time(NULL));
	scanf("%d %d %d",&x,&k,&p);
	pd=(double)p/100;
	pp=(double)(100-p)/100;
	dp[0][0]=1;
	for(i=0; i<k; i++)
	{
		dp[next][0]=0;
		for(j=0; j<255; j++)
		{
			dp[next][j+1]=dp[1-next][j]*pp;
		}
		for(j=0; j<255; j+=2)
		{
			ans+=dp[1-next][j]*pd;
			dp[next][j/2]+=dp[1-next][j]*pd;
		}
		next=1-next;
	}
	for(i=0; i<255; i++)
	{
		r=i+x;
		while(r%2==0 )
		{
			r/=2;
			ans+=dp[1-next][i];
		}
	}
	printf("%lf",ans);
	return 0;
}
