#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
#define maxn 240
#define maxm 512
using namespace std;
typedef long long LL;
int n,ori;
double prob,dp[2][maxm][maxn];
int Count(int x)
{
	int sum=0;
	for(;~x&1;x>>=1,++sum);
	return sum;
}
double Dp()
{
	double ans=0;
	int cur=0;
	prob/=100;
	int x=ori>>8,last=x&1,s=1;
	for(x>>=1;x&&(x&1)==last;x>>=1,++s);
	dp[cur][ori&511][s]=1;
	for(int i=1;i<=n+1;++i,cur^=1)
	{
		memset(dp[cur^1],0,sizeof(dp[cur^1]));
		for(int j=0;j<=511;++j)
			for(int k=1;k<=i+30;++k)
			{
				int x=j>>8&1;
				double &val=dp[cur][j][k];
				if(i==n+1)
				{
			 		if(j==0)
				 		ans+=val*(8+k);
			 		else
				 		ans+=val*Count(j);
			 		continue;
		 		}
				int newj,newk,newx;
				//a=a*2
				newj=j*2,newx=newj>>8,newj%=512;
				newk=newx%2==x?k+1:1;
				dp[cur^1][newj][newk]+=val*prob;
				//a=a+1
				newj=j+1,newx=x,newk=k;
				if(newj==512)
					newj=0;
				dp[cur^1][newj][newk]+=val*(1-prob);
			}
	}
	return ans;
}
int main()
{
	cin>>ori>>n>>prob;
	printf("%.10f\n",Dp());
	return 0;
}
