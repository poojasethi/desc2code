#include <stdio.h>
#include <stdlib.h>
using namespace std;

int n,m,p,i,j,k;
double f[205][205],p1,p2;

int main()
{
	scanf("%d%d%d",&m,&n,&p);
	p1=p/100.0;
	p2=1-p1;
	for(i=0;i<=n;++i)
	for(j=m+i;j%2==0;j/=2)
	++f[0][i];
	for(i=0;i<=n;++i)
	for(j=0;j<=n;++j)
	{
		if(j!=0)f[i+1][j-1]+=p2*f[i][j];
		if(j*2<=n)f[i+1][j*2]+=p1*(f[i][j]+1);
	}
	printf("%.9lf\n",f[n][0]);
}
