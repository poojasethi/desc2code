#include <iomanip>
#include <iostream>

using namespace std;

const int maxn = 2*100+3;

int x, k, p_tmp;
double dp[maxn][maxn], p;

int main()
{
	cin >> x >> k >> p_tmp;
	p = p_tmp / 100.0;
	
	for(int j = 0; j <= k; j++){
		int q = x + j;
		while(!(q & 1)){
			dp[0][j]++;
			q /= 2;
		}
	}

	for(int i = 1; i <= k; i++){
		for(int j = 0; j <= k; j++){
			dp[i][j] += (dp[i-1][j+1] * (1-p));
			if(!( j & 1 ))
				dp[i][j] += p* (dp[i-1][j>>1] + 1.0);
		}
	}

	cout << setprecision(9) << fixed << dp[k][0];
	return 0;
}
