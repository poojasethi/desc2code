#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
using namespace std;
double dp[205][205][205],ans;
int i,j,k,X,K,P,now,t;
int main()
{
    scanf("%d%d%d",&X,&K,&P);
    dp[0][0][0]=1;
    for (i=1; i<=K; i++)
      for (j=0; j<i; j++)
        for (k=0; k<=i; k++)
        {
            if (k%2==0) dp[i][j+1][k/2]+=dp[i-1][j][k]*(P/100.0); else
              ans+=dp[i-1][j][k]*(P/100.0)*j;
            dp[i][j][k+1]+=dp[i-1][j][k]*(1-P/100.0);
        }
    for (j=0; j<=K; j++)
      for (k=0; k<=K; k++)
      {
          now=k+X; t=0;
          while (now%2==0) {t++; now/=2;}
          ans+=dp[K][j][k]*(t+j);
      }
    printf("%.10f\n",ans);
    return 0;
}
