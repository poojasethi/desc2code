#include <cstdio>
#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdlib>
#include <cmath>
#define N 310
using namespace std;
typedef long long ll;
typedef double D;
int x,k,t,i,j;
D f[N][N],p,q,ans;
int main(){
  scanf("%d%d%d",&x,&k,&t);
  p=t/100.0;q=1-p;
  for (i=0;i<=k;i++) for (t=x+i;!(t&1);t>>=1) f[0][i]+=1.0;
  for (i=1;i<=k;i++) for (j=0;j<=k;j++){
    f[i][j]+=f[i-1][j+1]*q;
    if (!(j&1)) f[i][j]+=p*(f[i-1][j>>1]+1.0);
  }
  printf("%.10lf\n",f[k][0]);
  return 0;
}
