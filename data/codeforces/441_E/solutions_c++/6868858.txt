#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxk=210,maxb=1<<8,maxc=240;
int x,K,lim=maxb-1;
double p,ans=0;
double f[2][maxb][2][maxc]={0};
int bitcount(int x,bool b) { return !x||(x&1)!=b?0:bitcount(x>>1,b)+1; }
inline int getbit(int x,int a) { return (x>>a)&1; }
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j,k,l;
	scanf("%d%d%lf",&x,&K,&p);
	p/=100;
	f[0][x&lim][getbit(x,8)][bitcount(x>>8,getbit(x,8))]=1;
	for(i=0;i<K;i++)
	{
		memset(f[!(i&1)],0,sizeof(f[0]));
		for(j=0;j<=lim;j++)
			for(k=0;k<2;k++)
				for(l=0;l<maxc-1;l++)
				{
					double f0=f[i&1][j][k][l];
					if(getbit(j,7)==k)
						f[!(i&1)][(j<<1)&lim][k][l+1]+=f0*p;
					else
						f[!(i&1)][(j<<1)&lim][!k][1]+=f0*p;
					if(j<lim)
						f[!(i&1)][j+1][k][l]+=f0*(1-p);
					else
						f[!(i&1)][0][!k][k?l:1]+=f0*(1-p);
				}
	}
	for(i=1;i<=lim;i++)
		for(j=0;j<2;j++)
			for(k=0;k<maxc;k++)
				ans+=bitcount(i,0)*f[K&1][i][j][k];
	for(i=1;i<maxc;i++)
		ans+=8*f[K&1][0][1][i];
	for(i=1;i<maxc;i++)
		ans+=(8+i)*f[K&1][0][0][i];
	printf("%.8lf\n",ans);
	return 0;
}
