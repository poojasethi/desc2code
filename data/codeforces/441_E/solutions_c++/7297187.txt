#include <cstdio>
#include <cstdlib>
using namespace std;

int x,k;
double p,f[201][201];

int main()
{
	scanf("%d%d",&x,&k);
	scanf("%lf",&p);
	p/=100;
	for (int i=0;i<=k;i++)
		for (int j=x+i;(j&1)==0;j>>=1) f[0][i]++;
	for (int i=1;i<=k;i++)
		for (int j=0;j<=k;j++)
		{
			if (j-1>=0) f[i][j-1]+=f[i-1][j]*(1-p);
			if (j*2<=k) f[i][j*2]+=(f[i-1][j]+1)*p;
		}
	printf("%.12lf\n",f[k][0]);
}
