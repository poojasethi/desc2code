#include<cstdio>
#include<algorithm>
#define fo(i,a,b) for(i=a;i<=b;++i)
#define fd(i,a,b) for(i=a;i>=b;--i)
using namespace std;
const int mo=51123987,mn=152,mm=52;
int n,m,i,j,k,a,b,c,x,y,z,v,ans;
int f[mn][mm][mm][mm],next[mn][4];
char st[mn];
int abs(int x){return x<0?-x:x;}
void upd(int&x,int v){x=(x+v)%mo;}
int main()
{
    scanf("%d",&n);
    scanf("%s",st+1);
    m=(n+2)/3;
    f[1][0][0][0]=1;
    fd(i,n,1)
    {
        fo(j,1,3)next[i][j]=next[i+1][j];
        if(i)next[i][st[i]-96]=i;
    }
    fo(i,1,n)
    {
        fo(a,0,m)
        fo(b,0,m)
        fo(c,0,m)
        {
            v=f[i][a][b][c];
            if(!v)continue;
            if(a+b+c==n&&max(abs(a-b),max(abs(b-c),abs(a-c)))<=1)
                upd(ans,v);
            x=next[i][1],y=next[i][2],z=next[i][3];
            if(x)upd(f[x][a+1][b][c],v);
            if(y)upd(f[y][a][b+1][c],v);
            if(z)upd(f[z][a][b][c+1],v);
        }
    }
    printf("%d\n",ans);
    return 0;
}
