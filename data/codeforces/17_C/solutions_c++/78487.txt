#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
const int MOD = 51123987;
int n, lim;
int next[152][3], dp[151][52][52][52];

int solve(int ith, int a,int b,int c) {
    if(a>lim||b>lim||c>lim) return 0;
    if(a+b+c==n) return abs(a-b)<=1&&abs(a-c)<=1&&abs(b-c)<=1;
    int& res=dp[ith][a][b][c];
    if(res!=-1) return res;
    res=0;
    for(int i=0; i<3; i++) {
        if(next[ith][i]==-1) continue;
        res+=solve(next[ith][i], a+(i==0), b+(i==1), c+(i==2));
        if(res>=MOD) res-=MOD;
    }
    return res;
}
int main() {
    int i,j;
    char str[152];
    scanf("%d",&n);
    scanf("%s",str);
    
    lim=(n/3+1);
    
    for(i=0; i<3; i++) next[n][i]=-1;
    for(i=n-1; i>=0; i--) {
        for(j=0; j<3; j++) next[i][j]=next[i+1][j];
        next[i][str[i]-'a']=i;
    }
    
    memset(dp,-1,sizeof(dp));
    printf("%d\n",solve(0,0,0,0));
    
    return 0;
}
