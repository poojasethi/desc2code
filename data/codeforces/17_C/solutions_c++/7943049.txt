#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

const int maxn = 158, maxn_ = 58, mod = 51123987;
int f[maxn][maxn_][maxn_][maxn_], next[maxn][3], n, m, ans;
char S[maxn];

inline void Add(int &x, int y) 
	{x = (x + y) % mod;}

int main() {
	scanf("%d%s", &n, S);
	next[n][0] = next[n][1] = next[n][2] = n;
	for (int i = n - 1; i >= 0; i--)
		next[i][0] = next[i + 1][0], 
		next[i][1] = next[i + 1][1], 
		next[i][2] = next[i + 1][2],
		next[i][S[i] - 'a'] = i;
	m = (n + 2) / 3; f[0][0][0][0] = 1; 
	for (int i = 0; i < n; i++)
		for (int j = 0; j <= m; j++)
			for (int k = 0; k <= m; k++)
				for (int l = 0; l <= m; l++) {
					int tmp = f[i][j][k][l];
					if (j + k + l == n && abs(j - k) < 2 && abs(j - l) < 2 && abs(k - l) < 2)
						Add(ans, f[i][j][k][l]);
					Add(f[next[i][0]][j + 1][k][l], tmp),
					Add(f[next[i][1]][j][k + 1][l], tmp),
					Add(f[next[i][2]][j][k][l + 1], tmp);
				}
	cout << ans << endl;
	return 0;
}
