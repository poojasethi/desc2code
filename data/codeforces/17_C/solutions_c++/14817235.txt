//In the Name of God

#include<iostream>
#include<algorithm>

using namespace std;

const int M=51123987;
int N,n;
int next[155][3];
int dp[155][55][55][55];
int A,B,C,f;
int ans;
string s;

int main()
{
  ios_base::sync_with_stdio(0);cin.tie(0);
  cin>>N;
  cin>>s;
  s.resize(unique(s.begin(),s.end())-s.begin());
  n=s.size();
  for(int i=0;i<3;i++)
    next[n][i]=n;
  for(int i=n-1;i>=0;i--)
    {
      for(int j=0;j<3;j++)
	next[i][j]=next[i+1][j];
      next[i][s[i]-'a']=i;
    }
  dp[0][0][0][0]=1;
  for(int i=0;i<n;i++)
    for(int a=0;a<51;a++)
      for(int b=0;b<51;b++)
	for(int c=0;c<51;c++)
	  {
	    if(a+b+c==N and abs(a-b)<2 and abs(a-c)<2 and abs(b-c)<2)
	      ans=(ans+dp[i][a][b][c])%M;
	    for(int j=0;j<3;j++)
	      {
		f=next[i][j];
		A=a+(j==0),B=b+(j==1),C=c+(j==2);
		dp[f][A][B][C]=(dp[f][A][B][C]+dp[i][a][b][c])%M;
	      }
	  }
  cout<<ans<<endl;
  return 0;
}
