#include<stdio.h>
#include<stdlib.h>
const int MOD=51123987;
int n;
char s[300];
int ss[300],nn;
int dp[153][53][53][53];
int d;
int next[300][3];
inline void add(int &a,int b){a=(a+b)%MOD;}
int main()
{
	scanf("%d%s",&n,s+1);
	d=n/3+1;
	for(int i=1;i<=n;i++)if(s[i]!=s[i-1])ss[++nn]=s[i]-'a';
	for(int i=nn;i>=0;i--)
		for(int j=0;j<3;j++)next[i][j]=i&&ss[i]==j?i:next[i+1][j];
	dp[0][0][0][0]=1;
	int ans=0;
	for(int i=0;i<=nn;i++)
		for(int x=0;x<=d;x++)
			for(int y=0;y<=d;y++)
				for(int z=0;z<=d;z++)
				{
					int t=dp[i][x][y][z];
					if(!t)continue;
					if(x+y+z==n&&abs(x-y)<=1&&abs(y-z)<=1&&abs(x-z)<=1){add(ans,t);continue;}
					int p;
					if(p=next[i][0])add(dp[p][x+1][y][z],t);
					if(p=next[i][1])add(dp[p][x][y+1][z],t);
					if(p=next[i][2])add(dp[p][x][y][z+1],t);
				}
	printf("%d\n",ans);
	return 0;
}
