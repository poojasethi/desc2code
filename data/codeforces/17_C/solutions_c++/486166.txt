#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<cmath>
using namespace std;
int f[155][55][55][55],n,next[155][3];
char s[155];
void ck(int &x,int y)
{
	x=(x+y)%51123987;
}
int main()
{
	scanf("%d%s",&n,s);
	memset(f,0,sizeof(f));
	next[n][0]=next[n][1]=next[n][2]=n;
	for(int i=n-1;i>=0;i--)
	{
		for(int j=0;j<3;j++) next[i][j]=next[i+1][j];
		next[i][s[i]-'a']=i;
	}
	f[0][0][0][0]=1;
	int up=n/3+2;
	int ans=0;
	for(int i=0;i<n;i++)
		for(int j=0;j<=up;j++)
			for(int k=0;k<=up;k++)
				for(int l=0;l<=up;l++)
				{
					if(j+k+l==n&&abs(j-k)<=1&&abs(j-l)<=1&&abs(k-l)<=1) ck(ans,f[i][j][k][l]);
					ck(f[next[i][0]][j+1][k][l],f[i][j][k][l]);
					ck(f[next[i][1]][j][k+1][l],f[i][j][k][l]);
					ck(f[next[i][2]][j][k][l+1],f[i][j][k][l]);
				}
	printf("%d\n",ans);
	return 0;
}
