#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int n,m,i,j,k,l,o,p;
int f[152][52][52][52],ne[152][3],a[152];
char ch;
void add(int &x,int y)
{
    x=(x+y)%51123987;
}
int main( )
{
    cin >> n;
    for(i=1;i<=n;i++) cin >> ch,a[i]=ch-97;
    ne[n+1][0]=ne[n+1][1]=ne[n+1][2]=n+1;
    for(i=n;i>0;i--)
        ne[i][0]=ne[i+1][0],ne[i][1]=ne[i+1][1],ne[i][2]=ne[i+1][2],ne[i][a[i]]=i;
    f[1][0][0][0]=1;
    int lim=(n+2)/3;
    int ans=0;
    for(i=1;i<=n;i++)
        for(j=0;j<=lim;j++) for(k=0;k<=lim;k++) for(l=0;l<=lim;l++)
        {
            int w=f[i][j][k][l];
            if(j+k+l==n && abs(k-j)<=1 && abs(l-k)<=1 && abs(j-l)<=1) add(ans,w);
            add(f[ne[i][0]][j+1][k][l],w);
            add(f[ne[i][1]][j][k+1][l],w);
            add(f[ne[i][2]][j][k][l+1],w);
        }
    cout << ans << endl;
    cin >> n;
    return 0;
}
