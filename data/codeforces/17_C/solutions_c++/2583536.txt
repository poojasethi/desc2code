#include<cstdio>
const int oo=51123987;
int N,M,i,j,k,l,res,Ans;
int s[155][5],f[155][55][55][55];
char S[155];
int Abs(int a)
{
	if(a<0) return -a;
	return a;
}
int main()
{
	scanf("%d",&N);
	scanf("%s",S);
	for(j=0;j<3;++j) s[N][j]=N;
	for(i=N-1;i>=0;--i)
		for(j=0;j<3;++j)
			if(S[i]-'a'==j) s[i][j]=i;
			else s[i][j]=s[i+1][j];
	f[0][0][0][0]=1;
	M=(N+2)/3;
	for(i=0;i<N;++i)
		for(j=0;j<=M;++j)
			for(k=0;k<=M;++k)
				for(l=0;l<=M;++l)
				{
					if(j+k+l>N) continue;
					res=f[i][j][k][l];
					if(j+k+l==N&&Abs(j-k)<2&&Abs(j-l)<2&&Abs(k-l)<2) Ans=(Ans+res)%oo;
					f[s[i][0]][j+1][k][l]=(f[s[i][0]][j+1][k][l]+res)%oo;
					f[s[i][1]][j][k+1][l]=(f[s[i][1]][j][k+1][l]+res)%oo;
					f[s[i][2]][j][k][l+1]=(f[s[i][2]][j][k][l+1]+res)%oo;	
				}
	printf("%d\n",Ans);
	return 0;
}
