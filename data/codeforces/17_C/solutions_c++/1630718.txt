#include <cstdio>
#include <cstdlib>

int  n, ans;
int  dp[155][55][55][55];
int  next[155][3];
char s[155];

void plus(int &x, const int y)
{
    x = (x + y) % 51123987;
}

int main(void)
{
    scanf("%d %s", &n, s);
    next[n][0] = next[n][1] = next[n][2] = n;
    for (int i = n-1; i >= 0; --i)
    {
        for (int j = 0; j < 3; ++j)
            next[i][j] = next[i+1][j];
        next[i][s[i]-'a'] = i;
    }
    
    dp[0][0][0][0] = 1;
    int limit = (n+5) / 3;
    for (int i = 0; i < n; ++i)
        for (int x = 0; x < limit; ++x)
        for (int y = 0; y < limit; ++y)
        for (int z = 0; z < limit; ++z)
        {
            if (x+y+z == n && abs(x-y) < 2 && abs(x-z) < 2 && abs(y-z) < 2)
                plus(ans, dp[i][x][y][z]);
            plus(dp[next[i][0]][x+1][y][z], dp[i][x][y][z]);
            plus(dp[next[i][1]][x][y+1][z], dp[i][x][y][z]);
            plus(dp[next[i][2]][x][y][z+1], dp[i][x][y][z]);
        }
    
    printf("%d\n", ans);
    
    return 0;
}
