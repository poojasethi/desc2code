#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int mod=51123987;
int f[151][53][53][53];
int next[151][3];
int n;
char s[151];
int ans;
int main()
{
	scanf("%d",&n);
	scanf("%s",s+1);
	for(int i=n;i>0;i--)
	{
		next[i][0]=next[i+1][0];
		next[i][1]=next[i+1][1];
		next[i][2]=next[i+1][2];
		next[i][s[i]-'a']=i;
	}
	f[1][0][0][0]=1;
	int d=n/3+1;
	for(int i=1;i<=n;i++)
		for(int j=0;j<=d;j++)
			for(int k=0;k<=d;k++)
				for(int l=0;l<=d;l++)
					if(f[i][j][k][l])
					{
						if(j+k+l==n&&abs(j-k)<2&&abs(j-l)<2&&abs(l-k)<2)
							ans+=f[i][j][k][l],ans%=mod;
						f[next[i][0]][j+1][k][l]+=f[i][j][k][l];
						f[next[i][1]][j][k+1][l]+=f[i][j][k][l];
						f[next[i][2]][j][k][l+1]+=f[i][j][k][l];
						f[next[i][0]][j+1][k][l]%=mod;
						f[next[i][1]][j][k+1][l]%=mod;
						f[next[i][2]][j][k][l+1]%=mod;
					}
	printf("%d\n",ans);
	return 0;
}
