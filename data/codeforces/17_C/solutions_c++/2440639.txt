#include <cstdio>
#include <cstring>

#include <algorithm>
using namespace std;

const int N = 155, modu = 51123987;

int f[N][N / 3 + 2][N / 3 + 2][N / 3 + 2], next[N][3], n, m;
char S[N];

inline void I (int &x, int y) { x = (x + y) % modu; }

int main ()
{
#ifdef LOCAL
	freopen("in", "r", stdin);
#endif
	scanf("%d", &n);
	scanf("%s", S + 1);
	for (int i = n; i >= 1; --i)
	{
		memcpy(next[i], next[i + 1], sizeof next[i]);
		next[i][S[i] - 'a'] = i;
	}
	f[1][0][0][0] = 1;
	m = n / 3 + 1;
	int res(0);
	for (int i = 1; i <= n; ++i)
		for (int a = 0; a <= m; ++a)
			for (int b = 0; b <= m; ++b)
				for (int c = 0; c <= m; ++c)
				{
					if (a + b + c == n && abs(a - c) <= 1 && abs(b - c) <= 1 && abs(a - b) <= 1)
						I(res, f[i][a][b][c]);
					I(f[next[i][0]][a + 1][b][c], f[i][a][b][c]);
					I(f[next[i][1]][a][b + 1][c], f[i][a][b][c]);
					I(f[next[i][2]][a][b][c + 1], f[i][a][b][c]);
				}
	printf("%d\n", res);
	return 0;
}
