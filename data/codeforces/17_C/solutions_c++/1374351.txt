#include<iostream>
#include<cmath>
#include<cstdio>
#include<cstring>
using namespace std;
#define MAXN 54
#define MAXL 155
#define MOD 51123987

int N,idx[3][MAXL];;
int f[MAXL][MAXN][MAXN][MAXN];
string s;

void add(int &x,int y)
{
    x=(x+y)%MOD;
}
int main()
{
    cin>>N>>s;
    int sup=N/3+2;

    idx[0][N]=idx[1][N]=idx[2][N]=N;
    for(int i=N-1;~i;--i)
    {
        for(int j=0;j<3;j++) idx[j][i]=idx[j][i+1];
        idx[s[i]-'a'][i]=i;
    }
    int ans=0;
    f[0][0][0][0]=1;
    for(int i=0;i<N;i++) for(int x=0;x<=sup;x++)
        for(int y=0;y<=sup;y++) for(int z=0;z<=sup;z++)
        {
            int tmp=f[i][x][y][z];
            if(x+y+z==N&&abs(x-y)<=1&&abs(y-z)<=1&&abs(z-x)<=1)
                add(ans,tmp);
            add(f[idx[0][i]][x+1][y][z],tmp);
            add(f[idx[1][i]][x][y+1][z],tmp);
            add(f[idx[2][i]][x][y][z+1],tmp);
        }
    cout<<ans<<endl;
    return 0;
}
