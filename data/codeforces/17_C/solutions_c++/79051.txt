#include<cstdio>
#include<algorithm>
#include<cstring>
#define mod 51123987
using namespace std;

char s[155];
int f[151][52][52][52];
int next[151][3];
int ans;
int n;
int main()
{
	//freopen("C.in","r",stdin);
	scanf("%d",&n);
	scanf("%s",s);
	next[n][0]=next[n][1]=next[n][2]=n;
	for(int i=n-1;i>=0;--i)
	{
		for(int j=0;j<3;++j) next[i][j]=next[i+1][j];
		next[i][s[i]-'a']=i;
	}
	
	int m=(n+2)/3;
	memset(f,0,sizeof(f));
	f[0][0][0][0]=1;
	ans=0;
	
	for(int i=0;i<n;++i)	
	{
		for(int a=0;a<=m;++a)
			for(int b=0;b<=m;++b)
					for(int c=0;c<=m;++c)
					{
						if(a+b+c==n && abs(a-b)<2 && abs(a-c)<2 && abs(b-c)<2)
							ans=(ans+f[i][a][b][c])%mod;
						if(next[i][0]!=n)
							f[next[i][0]][a+1][b][c]=(f[next[i][0]][a+1][b][c]+f[i][a][b][c])%mod;
						if(next[i][1]!=n)
							f[next[i][1]][a][b+1][c]=(f[next[i][1]][a][b+1][c]+f[i][a][b][c])%mod;
						if(next[i][2]!=n)
							f[next[i][2]][a][b][c+1]=(f[next[i][2]][a][b][c+1]+f[i][a][b][c])%mod;
					}
	}
	printf("%d\n",ans);
	return 0;
}
