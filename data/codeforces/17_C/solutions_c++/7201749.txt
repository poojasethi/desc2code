#include<cstdio>
#include<algorithm>
using namespace std;
int dp[152][53][53][53],next[152][3];
int n,mod = 51123987,m,ans;
char s[152];
int main(){
  scanf("%d",&n); m =(n+2)/3;
  scanf("%s",s+1);
  for(int i=1;i<=n;i++) s[i] -= 'a';
  for(int i=n;i>=0;i--){
    for(int j=0;j<3;j++)
      next[i][j] = next[i+1][j];
    next[i][s[i]] = i;
  }
  dp[1][0][0][0] = 1;
  for(int i=1;i<=n;i++)
    for(int k1=0;k1<=m;k1++)
      for(int k2=0;k2<=m;k2++)
    for(int k3=0;k3<=m;k3++){
      dp[next[i][0]][k1+1][k2][k3] += dp[i][k1][k2][k3];
      dp[next[i][0]][k1+1][k2][k3] %= mod;
      dp[next[i][1]][k1][k2+1][k3] += dp[i][k1][k2][k3];
      dp[next[i][1]][k1][k2+1][k3] %= mod;
      dp[next[i][2]][k1][k2][k3+1] += dp[i][k1][k2][k3];
      dp[next[i][2]][k1][k2][k3+1] %= mod;
    }  
  for(int i=1;i<=n;i++)
    for(int k1=0;k1<=m;k1++)
      for(int k2=-1;k2<=1;k2++)
    for(int k3=-1;k3<=1;k3++)
      if(k1*3+k2+k3 == n && abs(k2-k3) != 2)
        ans += dp[i][k1][k1+k2][k1+k3], ans %= mod;
  printf("%d",ans);
  return 0;
}
