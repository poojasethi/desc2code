#include<cstdio>
#include<cstring>
#include<cstdlib>
const int mo=51123987;
char s[155];
int nxt[155][3];
int f[55][55][55][155];
void p(int &a,int b){
    a+=b;
    if (a>=mo) a-=mo;
}
int main(){
    int n,i,j,k,l,ans;
//  freopen("17c.in","r",stdin);
//  freopen("17c.out","w",stdout);
    scanf("%d",&n);
    scanf("%s",s+1);
    nxt[n+1][0]=nxt[n+1][1]=nxt[n+1][2]=n+1;
    for (i=n;i>=0;i--){
        for (j=0;j<3;j++) nxt[i][j]=nxt[i+1][j];
        if (i) nxt[i][s[i]-'a']=i;
    }
    ans=0;
    memset(f,0,sizeof f);
    f[0][0][0][0]=1;
    for (i=0;i<=n/3+1;i++) for (j=0;j<=n/3+1;j++) for (k=0;k<=n/3+1;k++){
        if (i+j+k==n && abs(i-j)<=1 && abs(j-k)<=1 && abs(i-k)<=1)
            for (l=1;l<=n;l++) p(ans,f[i][j][k][l]);
        for (l=0;l<=n;l++){
            p(f[i+1][j][k][nxt[l][0]],f[i][j][k][l]);
            p(f[i][j+1][k][nxt[l][1]],f[i][j][k][l]);
            p(f[i][j][k+1][nxt[l][2]],f[i][j][k][l]);
        }
    }
    printf("%d\n",ans);
    return 0;
}
