#include<iostream>
#include<cmath>
#include<cstdio>
#include<cstring>
using namespace std;
int f[152][52][52][52];
char a[151];
int e[151][4];
int n;
const int p = 51123987;
int main()
{
	scanf("%d", &n);
	scanf("%s", a+1);
	for(int i = 0; i <= n; i ++)
	{
		for(int j = 1; j <= 3; j++)
		{
			for(int k = i; k <= n; k++)
				if(a[k] == 'a' + j - 1)
				{
					e[i][j] = k;
					break;
				}
			if(e[i][j] == 0)
				e[i][j] = n+1;
		}
	}
	f[0][0][0][0] = 1;
	int ans = 0;
	for(int i = 0; i <= n; i++)
		for(int j = 0; j <= 50; j++)
			for(int k = 0;  k <= 50; k++)
				for(int w = 0; w <= 50; w++)
				{
					if(j + k + w == n && abs(w-j) <= 1 && abs(j-k)<=1 && abs(k-w)<=1)
					{
						ans += f[i][j][k][w];
						ans %= p;
					}
					f[e[i][1]][j+1][k][w] += f[i][j][k][w];
					f[e[i][1]][j+1][k][w] %= p;
					f[e[i][2]][j][k+1][w] += f[i][j][k][w];
					f[e[i][2]][j][k+1][w] %= p;
					f[e[i][3]][j][k][w+1] += f[i][j][k][w];
					f[e[i][3]][j][k][w+1] %= p;
				}
	printf("%d\n", ans);
	return 0;
}
