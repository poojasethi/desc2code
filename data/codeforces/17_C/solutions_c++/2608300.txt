#include<cstdio>
#include<cstring>
const int N=155,oo=51123987;
int f[55][55][55][N],w[5],next[N][5],n,m,z,ans;
char s[N];
void updata(int &x,int y){x+=y; if (x>=oo) x-=oo;}
inline int abs(int a){return a>0?a:-a;}
int main()
{
	scanf("%d",&n); scanf("%s",s+1); m=n/3+1; w[0]=w[1]=w[2]=n+1;
	for (int i=n;i;i--) w[s[i]-97]=i,memcpy(next[i],w,sizeof(w));
	f[0][0][0][0]=1; memcpy(next[0],w,sizeof(w));
	for (int i=0;i<=m;i++)
		for (int j=0;j<=m;j++)
			for (int k=0;k<=m;k++)
				for (int p=0;p<=n;p++) if (z=f[i][j][k][p])
					updata(f[i+1][j][k][next[p][0]],z),
					updata(f[i][j+1][k][next[p][1]],z),
					updata(f[i][j][k+1][next[p][2]],z);
	for (int i=0;i<=m;i++)
		for (int j=i-1;j<=i+1;j++) if (i+j<=n && j>=0){
			int k=n-i-j;
			if (abs(i-k)<=1 && abs(j-k)<=1)
				for (int p=0;p<=n;p++) updata(ans,f[i][j][k][p]);
			}
	printf("%d\n",ans); return 0;
}
