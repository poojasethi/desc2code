#include<queue>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#define N 155
#define mod 51123987
using namespace std;
typedef long long ll;
char str[N];
int n,a[N],ans;
int next[N][3];
int temp[3];
int f[N][55][55][55];
int main()
{
	int len,lim,i,j,k,l,s;
	scanf("%d",&n);
	scanf("%s",str+1);
	for(i=1;i<=n;i++)
	{
		a[i]=str[i]-'a';
	}
	len=unique(a+1,a+n+1)-a-1;
	for(i=len;i>=1;i--)
	{
		temp[a[i]]=i;
		memcpy(next[i],temp,sizeof(int)*3);
	}
	lim=(n+2)/3;
	f[1][0][0][0]=1;
	for(i=1;i<=len;i++)
	{
		for(j=0;j<=lim;j++)
		{
			for(k=0;k<=lim;k++)
			{
				for(l=0;l<=lim;l++)
				{
					if(j+k+l==n&&abs(l-j)<=1&&abs(l-k)<=1&&abs(j-k)<=1)
						(ans+=f[i][j][k][l])%=mod;
					(f[next[i][0]][j+1][k][l]+=f[i][j][k][l])%=mod;
					(f[next[i][1]][j][k+1][l]+=f[i][j][k][l])%=mod;
					(f[next[i][2]][j][k][l+1]+=f[i][j][k][l])%=mod;
				}
			}
		}
	}
	printf("%d\n",ans);
	fclose(stdin);
	fclose(stdout);
	return 0;
}
