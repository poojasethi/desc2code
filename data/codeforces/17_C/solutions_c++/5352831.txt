#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
using namespace std;
#define MOD 51123987
#define rep(i,s,t) for(int i=s;i<=t;++i)
#define vep(i,t,s) for(int i=t;i>=s;--i)
#define up(x,y) ((x+=y)%=MOD)
int n, t[155][4], m, f[155][55][55][55], c[4];
char s[155];
int main(){
    scanf("%d",&n);scanf("%s",s+1);
    rep(i,1,n)s[i] -= 'a';
    vep(i,n,1){c[s[i]]=i;rep(j,0,2)t[i][j]=c[j];}
    m=n/3+1;f[1][0][0][0]=1;int ans=0;
    rep(i,1,n)
        rep(j,0,m)rep(k,0,m)rep(l,0,m){
               int p=f[i][j][k][l];
               if (j+k+l==n&&abs(j-k)<2&&abs(j-l)<2&&abs(k-l)<2) (ans+=p)%=MOD;
               up(f[t[i][0]][j+1][k][l],p);up(f[t[i][1]][j][k+1][l],p);up(f[t[i][2]][j][k][l+1],p);
        }
    printf("%d\n", ans);
}
