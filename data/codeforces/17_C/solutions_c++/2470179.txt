#include <iostream>
#include <algorithm>
using namespace std;
int n,i,j,k,l,lim,w,ans;
int f[152][52][52][52],a[152],next[152][3];
char o;

void add(int &x,int y)
{
     x=(x+y)%51123987;
}

int main()
{
    cin >> n;
    for (i=1;i<=n;i++) cin >> o,a[i]=o-97;
    lim=(n+2)/3;
    next[n+1][0]=next[n+1][1]=next[n+1][2]=n+1;
    for (i=n;i>=1;i--)
        next[i][0]=next[i+1][0],next[i][1]=next[i+1][1],next[i][2]=next[i+1][2],next[i][a[i]]=i;
    f[1][0][0][0]=1;
    for (i=0;i<=n;i++)
        for (j=0;j<=lim;j++) for (k=0;k<=lim;k++) for (l=0;l<=lim;l++)
        {
            w=f[i][j][k][l];
            if (j+k+l==n && abs(j-k)<=1 && abs(j-l)<=1 && abs(k-l)<=1) add(ans,w);
            add(f[next[i][0]][j+1][k][l],w);
            add(f[next[i][1]][j][k+1][l],w);
            add(f[next[i][2]][j][k][l+1],w);
        }
    cout << ans << endl;
    return 0;
}
