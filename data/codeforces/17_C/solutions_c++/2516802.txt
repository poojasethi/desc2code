#include<cstdio>
#include<algorithm>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
#define For(i,n) for(int i=1;i<=n;i++)
#define ForD(i,n) for(int i=n;i;i--)
#define plus(a,b) (a+=b)%=mo
const int N=152,M=52,mo=51123987;
int n,m,f[N][M][M][M],p[N][3],ans;
char s[N];
int main()
{
	scanf("%d",&n);
	For(i,n)
	{
	 	scanf(" %c",&s[i]);
		s[i]-='a';
	}
	ForD(i,n)
	{
		rep(j,3) p[i][j]=p[i+1][j];
		p[i][s[i]]=i;
	}
	m=(n+2)/3+1;
	f[1][0][0][0]=1;
	For(i,n)
	    rep(x,m) rep(y,m) rep(z,m)
	    if (int t=f[i][x][y][z])
	    {
			if (x+y+z==n && abs(x-y)<2 && abs(x-z)<2 && abs(y-z)<2)
			    plus(ans,t);
			plus(f[p[i][0]][x+1][y][z],t);
			plus(f[p[i][1]][x][y+1][z],t);
			plus(f[p[i][2]][x][y][z+1],t);
		}
	printf("%d\n",ans);
	return 0;
}
