#include <cstdio>
#include <cstdlib>
#define P 51123987
#define up(x,y) ((x+=y)%=P)
int N,M,f[155][55][55][55],p[155][3],ans;
char S[155];
int main()
{ 
    scanf("%d%s",&N,S+1),M=N/3+2;
    p[N+1][0]=p[N+1][1]=p[N+1][2]=N+1;
    for (int i=N;i>=0;i--)
        for (int j=0;j<3;j++) p[i][j]=(j==S[i]-97)?i:p[i+1][j];
    f[0][0][0][0]=1;
    for (int i=0;i<=N;i++)
        for (int j=0;j<=M;j++)
            for (int k=0;k<=M;k++)
                for (int l=0,x;l<=M;l++) if (x=f[i][j][k][l])
                {
                    if (j+k+l==N&&abs(j-k)<=1&&abs(j-l)<=1&&abs(k-l)<=1) up(ans,x);
                    up(f[p[i][0]][j+1][k][l],x),up(f[p[i][1]][j][k+1][l],x),up(f[p[i][2]][j][k][l+1],x);
                }
    printf("%d\n",ans);
}
