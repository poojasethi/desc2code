#include<cstdio>
#include<cstdlib>
#include<cstring>

using namespace std;

const int mo=51123987;

#define inc(a,b) a=(a+b)%mo

int next[160][4],f[152][52][52][52],ans,n;

char s[160];

int main()
{
	scanf("%d",&n);
	scanf("%s",s+1);
	for (int a=1;a<=n;a++)
		s[a]-='a'-1;
	for (int a=1;a<=3;a++)
		next[n+1][a]=n+1;
	for (int a=n;a>=1;a--)
	{
		for (int b=1;b<=3;b++)
			next[a][b]=next[a+1][b];
		next[a][s[a]]=a;
	}
	f[1][0][0][0]=1;
	for (int a=1;a<=n;a++)
		for (int b=0;b*3<=n+2;b++)
			for (int c=0;c*3<=n+2;c++)
				for (int d=0;d*3<=n+2 && b+c+d<=n;d++)
					if (f[a][b][c][d])
					{
						if (b+c+d==n && abs(b-c)<=1 && abs(c-d)<=1 && abs(d-b)<=1) inc(ans,f[a][b][c][d]);
						inc(f[next[a][1]][b+1][c][d],f[a][b][c][d]);
						inc(f[next[a][2]][b][c+1][d],f[a][b][c][d]);
						inc(f[next[a][3]][b][c][d+1],f[a][b][c][d]);
					}
	printf("%d\n",ans);

	return 0;
}
