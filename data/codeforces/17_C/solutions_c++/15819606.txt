#include <bits/stdc++.h>
using namespace std;
const int MOD = 51123987;
int n, f[155][55][55][55], next[155][255];
char str[200];
int main()
{
	scanf("%d%s",&n,str+1);
	for (int i=1;i<=n;i++) {
		for (int j=i;j<=n;j++) {
			if (str[j]=='a'&&!next[i]['a']) next[i]['a'] = j;
			if (str[j]=='b'&&!next[i]['b']) next[i]['b'] = j;
			if (str[j]=='c'&&!next[i]['c']) next[i]['c'] = j;
		}
	}
	f[1][0][0][0] = 1;
	int m = n/3+1, ans = 0;
	for (int l=1;l<=n;l++) {
		for (int a=0;a<=m;a++) {
			for (int b=0;b<=m;b++) {
				for (int c=0;c<=m;c++) {
					int p = f[l][a][b][c];
					if (a+b+c==n&&abs(a-b)<2&&abs(a-c)<2&&abs(b-c)<2) ans = (ans + p) %MOD;
					f[next[l]['a']][a+1][b][c] += p;
					f[next[l]['a']][a+1][b][c] %= MOD;
					f[next[l]['b']][a][b+1][c] += p;
					f[next[l]['b']][a][b+1][c] %= MOD;
					f[next[l]['c']][a][b][c+1] += p;
					f[next[l]['c']][a][b][c+1] %= MOD;
				}
			}
		}
	}
	cout << ans << endl;
	return 0;
}
