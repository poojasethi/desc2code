#include <cstdio>
#include <algorithm>
#include <cstring>
#include <iostream>
#define mod 51123987
using namespace std;
char s[1000]; int f[152][52][52][52];
struct box{int c[5];}next[1000];
int main()
{
	//freopen("input.txt","r",stdin);
	long long int ans=0; int n,i,m,a,b,c,d[5];
	scanf("%d",&n);
	scanf("%s",s+1);
	next[n+1].c[1] = next[n+1].c[2] = next[n+1].c[3] = n+1;
	for(i=n;i>=1;i--)
	{
		next[i] = next[i+1];
		next[i].c[s[i]-'a'] = i;
	}	
	memset(d,0,sizeof(d)); d[s[1]-'a']=1;	
	f[1][0][0][0] = 1;
	m = (n+2)/3;
	for(i=1;i<=n;i++)
		for(a=0;a<=m;a++)
		for(b=0;b<=m;b++)
		for(c=0;c<=m;c++)
		if(f[i][a][b][c])
		{
			//printf("%d %d %d %d: %d\n",i,a,b,c,f[i][a][b][c]);
			for(int j=0;j<3;j++)
			{
				memset(d,0,sizeof(d)); d[j]=1;
				f[next[i].c[j]][a+d[0]][b+d[1]][c+d[2]] += f[i][a][b][c];
				f[next[i].c[j]][a+d[0]][b+d[1]][c+d[2]] %= mod;
			}
		}
	for(a=0;a<=m;a++)
		for(b=max(0,a-1);b<=m&&b<=a+1;b++)
		for(c=max(0,a-1);c<=m&&c<=a+1;c++)
		if(abs(b-c)<=1 && a+b+c == n)
		for(i=1;i<=n;i++)
			ans = (ans+f[i][a][b][c])%mod;
	cout<<ans;
}
