#include <iostream>
#include <algorithm>
using namespace std;
#define REP(i,a,b) for(i=a;i<b;i++)

const int MOD=51123987;
int f[160][53][53][53],next[160][3];

int main(){
        int n,i,a,b,c,m,j;
        string s;
        cin>>n>>s;
        REP(i,0,3) next[n][i]=n;
        for(i=n-1;i>=0;i--){
                REP(j,0,3) next[i][j]=next[i+1][j];
                next[i][s[i]-'a']=i;
        }
        f[0][0][0][0]=1;
        m=n/3+2;
        int ans=0;
        REP(i,0,n)REP(a,0,m)REP(b,0,m)REP(c,0,m)if(f[i][a][b][c]){
                if (a+b+c==n&&abs(a-b)<=1&&abs(a-c)<=1&&abs(b-c)<=1) ans=(ans+f[i][a][b][c])%MOD;
                f[next[i][0]][a+1][b][c]=(f[next[i][0]][a+1][b][c]+f[i][a][b][c])%MOD;
                f[next[i][1]][a][b+1][c]=(f[next[i][1]][a][b+1][c]+f[i][a][b][c])%MOD;
                f[next[i][2]][a][b][c+1]=(f[next[i][2]][a][b][c+1]+f[i][a][b][c])%MOD;
        }
        cout<<ans<<endl;
}
