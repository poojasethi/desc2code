# -*- coding: utf-8 -*-
c = [ '2', '3', '4', '5', '6', '7', '8', '9', 'T', 'J', 'Q', 'K', 'A' ]
a = raw_input()
x, y = map(str, raw_input().split())
if x[1] == y[1]:
  print 'YES' if c.index(x[0]) > c.index(y[0]) else 'NO'
elif x[1] == a:
  print 'YES'
elif y[1] == a:
  print 'NO'
else:
  print 'NO'
