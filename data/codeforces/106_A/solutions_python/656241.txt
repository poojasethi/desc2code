def value(c):
    s = "6789TJQKA"
    for i in range(len(s)):
        if s[i] == c:
            return i
    return -1

tr = raw_input()
f,s = raw_input().split()

if f[1] == s[1]:
    if value(f[0]) > value(s[0]):
        print "YES"
    else:
        print "NO"
else:
    if f[1] == tr[0]:
        print "YES"
    else:
        print "NO"
