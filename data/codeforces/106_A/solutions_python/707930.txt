koz = raw_input()
kk = raw_input()
k = kk.split()
k1 = k[0]
k2 = k[1]

mas = ['6', '7', '8', '9', 'T', 'J', 'Q', 'K', 'A']
answer = 'NO'

if k1[1] == k2[1]:
    if mas.index(k1[0]) > mas.index(k2[0]):
        answer = 'YES'    
else:
    if k1[1] == koz and k2[1] != koz:
        answer = 'YES'

print answer
