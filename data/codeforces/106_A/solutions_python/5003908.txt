cos = raw_input()
cards = raw_input().strip().split()
rules = ['6', '7', '8', '9', '10', 'T', 'J', 'Q', 'K', 'A']

if cards[0][1] == cards[1][1] and rules.index(cards[0][0]) > rules.index(cards[1][0]):
    print 'YES'
elif cards[0][1] == cos and cards[1][1] != cos:
    print 'YES'
else:
    print 'NO'
