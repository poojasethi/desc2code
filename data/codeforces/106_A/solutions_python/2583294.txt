import sys

rank = {'6':6,'7':7,'8':8,'9':9,'T':10,'J':11,'Q':12,'K':13,'A':14}

t = sys.stdin.readline().strip()

(a, b) = sys.stdin.readline().strip().split()

if a[-1] == b[-1] and rank[a[0]] > rank[b[0]]:
    print('YES')
elif a[-1] == t and b[-1] != t:
    print('YES')
else:
    print('NO')
