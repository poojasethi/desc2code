RANKS = "6789TJQKA"
def is_first_winner(tramp, first, second):
	if first[1] == second[1]:
		if RANKS.find(first[0]) > RANKS.find(second[0]):
			return "YES"
		else:
			return "NO"
	if first[1] == tramp:
		return "YES"
	return "NO"

tramp = raw_input()
first, second = raw_input().split()
print is_first_winner(tramp, first, second)
