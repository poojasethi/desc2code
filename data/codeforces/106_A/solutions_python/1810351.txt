k = raw_input()
card1, card2 = tuple(raw_input().split())
d = {
    '6': 1,
    '7': 2,
    '8': 3,
    '9': 4, 
    'T': 5,
    'J': 6,
    'Q': 7,
    'K': 8,
    'A': 9,
}

if (d[card1[0]] > d[card2[0]] and card1[1] == card2[1] or
    card1[1] == k and card2[1] != k):
    print 'YES'
else:
    print 'NO'
