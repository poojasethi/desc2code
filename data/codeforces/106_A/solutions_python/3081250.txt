k = raw_input()
card = raw_input().split()
v = '6789TJQKA'
if card[0][1] == k:
  if card[1][1] == k and v.index(card[1][0]) > v.index(card[0][0]):
    print 'NO'
  else:
    print 'YES'
else:
  if card[0][1] != card[1][1] or v.index(card[1][0]) > v.index(card[0][0]):
    print 'NO'
  else:
    print 'YES'
