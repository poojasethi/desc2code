z = ["6", "7", "8", "9", "T", "J", "Q", "K", "A"]

trmp = raw_input()
a, b = raw_input().split()

if a[1] != b[1]:
	if a[1] == trmp:
		print 'YES'
	else:
		print 'NO'
else:
	x, y = -1, -1
	for i in range(len(z)):
		if z[i] == a[0]:
			x = i
		if z[i] == b[0]:
			y = i
	if x > y:
		print "YES"
	else:
		print "NO"
