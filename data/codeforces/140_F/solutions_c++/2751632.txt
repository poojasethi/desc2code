#include<cstdio>
#include<algorithm>
using namespace std;
int N,K,i,j,Ans,res;
struct node
{
	int X,Y;
}A[200005],B[105];
int Cmp(const node &i,const node &j)
{
	if(i.X==j.X) return i.Y<j.Y;
	return i.X<j.X;
}
void Check(int x,int y)
{
	int i,j,s=0;
	i=1,j=N;
	for(;i<=j;)
	{
		if(2*A[i].X==x&&2*A[i].Y==y)
		{
			++s,++i;
			continue;
		}
		if(2*A[i].X==x&&2*A[i].Y==y)
		{
			++s,--j;
			continue;
		}
		if(A[i].X+A[j].X==x&&A[i].Y+A[j].Y==y)
		{
			s+=2;
			++i,--j;
			continue;
		}
		if(A[i].X+A[j].X>x||A[i].X+A[j].X==x&&A[i].Y+A[j].Y>y) --j;
		else ++i;
	}
	if(s+K>=N) B[++Ans].X=x,B[Ans].Y=y;
}
int main()
{
	scanf("%d%d",&N,&K);
	if(N<=K)
	{
		printf("-1\n");
		return 0;
	}
	for(i=1;i<=N;++i) scanf("%d%d",&A[i].X,&A[i].Y);
	sort(A+1,A+N+1,Cmp);
	for(i=0;i<=K;++i)
		for(j=0;i+j<=K;++j)
			Check(A[i+1].X+A[N-j].X,A[i+1].Y+A[N-j].Y);
	sort(B+1,B+Ans+1,Cmp);
	for(i=1;i<=Ans;++i)
		if(i==1||B[i].X!=B[i-1].X||B[i].Y!=B[i-1].Y)
			++res;
	printf("%d\n",res);
	for(i=1;i<=Ans;++i)
		if(i==1||B[i].X!=B[i-1].X||B[i].Y!=B[i-1].Y)
			printf("%.1lf %.1lf\n",B[i].X/2.0,B[i].Y/2.0);
	return 0;
}
