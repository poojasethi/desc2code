#include<iostream>
#include<algorithm>
#include<vector>
#include<cstdio>
#include<cstdlib>
#include<cstring>

#define X first
#define Y second

using namespace std;
typedef pair<int,int> ii;

const int fin=0,maxn=200010;
int n,k;
ii a[maxn];
vector<ii> b;

ii operator +(ii a,ii b){
  return ii(a.X+b.X,a.Y+b.Y);
}
ii operator -(ii a,ii b){
  return ii(a.X-b.X,a.Y-b.Y);
}

void check(ii s){
  int i,j,p=0;
  ii t;
  for(i=1,j=n;i<=n;++i){
    for(t=s-a[i];j>1&&a[j]>t;--j);
    if(j&&a[j]==t)--j;else if(++p,p>k)return;
  }
  b.push_back(s);
}

int main(){
  if(fin){
    freopen("t.in","r",stdin);
    freopen("t.out","w",stdout);
  }
  int i,j;
  scanf("%d%d",&n,&k);
  if(k>=n)return puts("-1"),0;
  for(i=1;i<=n;++i)
    scanf("%d%d",&a[i].X,&a[i].Y);
  sort(a+1,a+n+1);
  for(i=1;i<=k+1;++i)
    for(j=n-k;j<=n;++j)
      check(a[i]+a[j]);
  sort(b.begin(),b.end());
  b.erase(unique(b.begin(),b.end()),b.end());
  printf("%d\n",b.size());
  for(i=0;i<b.size();++i)
    printf("%lf %lf\n",b[i].X*0.5,b[i].Y*0.5);
  return 0;
}
