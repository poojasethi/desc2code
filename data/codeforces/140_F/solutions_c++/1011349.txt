#include<cstdio>
#include<set>
#include<algorithm>
using namespace std;
#define X	first
#define Y	second
typedef pair<int,int> pt;
pt p[200000];
set<pt>	v,a;
int n,k,i,j,f,t,e;
int main(){
	scanf("%d%d",&n,&k);
	if(n<=k)return puts("-1"),0;
	for(i=0;i<n;++i)scanf("%d%d",&p[i].X,&p[i].Y);
	sort(p,p+n);
	for(i=0;i<=k;++i)for(j=0;j<=k;++j){
		pt u(p[i].X+p[n-j-1].X,p[i].Y+p[n-j-1].Y);
		if(v.insert(u).Y){
			e=n-1;
			for(t=f=0;f<=e;){
				pt s(p[f].X+p[e].X,p[f].Y+p[e].Y);
				if(s==u)++f,--e;
				else ++t,s<u?++f:--e;
			}
			if(t<=k)a.insert(u);
		}
	}
	printf("%d\n",a.size());
	for(set<pt>::iterator i=a.begin();i!=a.end();++i)
		printf("%.1lf %.1lf\n",i->X*0.5,i->Y*0.5);
	return 0;
}
