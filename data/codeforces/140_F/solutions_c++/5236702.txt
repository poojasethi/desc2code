#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
const long long base=2e9;
long long a[210000],f[1200],x,y;
int i,n,k,ans,j;
bool check(){
	int i=1,j=n,vk=k;
	long long t1,t2;
	for(;i<=j;){
		t1=f[ans]-a[i];
		t2=a[j]-f[ans];
		if(t1==t2)i++,j--;
		else if(t1>t2)i++,vk--;
		else j--,vk--;
	}
	return vk>=0;
}
int main(){
	scanf("%d%d",&n,&k);
	if(k>=n)return puts("-1"),0;
	for(i=1;i<=n;++i)scanf("%I64d%I64d",&x,&y),a[i]=((x+5e8)*base+y+5e8)*2;
	sort(a+1,a+n+1);
	for(i=1;i<=k+1;++i)
		for(j=n-k;j<=n;++j){
			f[++ans]=(a[i]+a[j])/2;
			if(!check())ans--;
		}
	sort(f+1,f+ans+1);
	ans=unique(f+1,f+ans+1)-f-1;
	printf("%d\n",ans);
	for(i=1;i<=ans;++i)printf("%.1lf %.1lf\n",f[i]/base/2.0-5e8,f[i]%base/2.0-5e8);
}
