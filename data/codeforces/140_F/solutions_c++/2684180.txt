#include<cstdio>
#include<algorithm>
#define rep(i,n) for (int i=0;i<=n;i++)
#define N 200005
int n,m,len,ans;
struct note{int x,y;} a[N],f[N][11][11],o[N];
bool cmp(const note i,const note j){return i.x==j.x?i.y<j.y:i.x<j.x;}
int main()
{
	scanf("%d%d",&n,&m); if (n<=m) return puts("-1"),0;
	rep(i,n-1) scanf("%d%d",&a[i].x,&a[i].y); std::sort(a,a+n,cmp);
	rep(i,n/2) rep(j,m) rep(k,m-j) f[i][j][k].x=f[i][j][k].y=1<<30;
	rep(i,m) rep(j,m-i) f[0][i][j].x=a[i].x+a[n-1-j].x,f[0][i][j].y=a[i].y+a[n-1-j].y;
	rep(i,n/2) rep(j,m) rep(k,m-j) if (f[i][j][k].x<1<<30) rep(l,m-j-k) rep(r,m-j-k-l)
		if (a[i+1+j+l].x+a[n-2-i-k-r].x==f[i][j][k].x &&
			a[i+1+j+l].y+a[n-2-i-k-r].y==f[i][j][k].y) f[i+1][j+l][k+r]=f[i][j][k];
	rep(i,n/2) rep(j,m) rep(k,m-j)
		if (i+i+2+m>=n && f[i][j][k].x<1<<30) o[len++]=f[i][j][k];
	std::sort(o,o+len,cmp);
	for (int i=0;i<len;i++) ans+=!i || o[i].x!=o[i-1].x || o[i].y!=o[i-1].y;
	printf("%d\n",ans);
	for (int i=0;i<len;i++) if (!i || o[i].x!=o[i-1].x || o[i].y!=o[i-1].y)
		printf("%lf %lf\n",o[i].x/2.,o[i].y/2.);
	return 0;
}
