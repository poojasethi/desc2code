#include <cstdio>
#include <set>
#include <algorithm>
using namespace std;
#define mp(x,y) make_pair(x,y)
#define X first
#define Y second
typedef pair<int,int> pii;
pii p[201000];set<pii> w,a,v;
int n,k,x,y;
#define rep(i,n) for (int i=0;i<n;i++)
int main() {
	scanf("%d%d",&n,&k);if (k>=n) {puts("-1");return 0;}
	rep(i,n) scanf("%d%d",&x,&y),p[i]=mp(x,y),w.insert(p[i]);
	sort(p,p+n);
	rep(i,k+1) rep(j,k+1) {
		pii P=mp(p[i].X+p[n-1-j].X,p[i].Y+p[n-1-j].Y);int t=0;
		if (!v.insert(P).Y) continue;
		rep(l,n) {if (t>k) break;t+=!w.count(mp(P.X-p[l].X,P.Y-p[l].Y));}
		if (t<=k) a.insert(P);
	}
	printf("%d\n",a.size());
	for (typeof(a.begin()) it=a.begin();it!=a.end();it++)
		printf("%.1f %.1f\n",0.5*it->X,0.5*it->Y);
}
