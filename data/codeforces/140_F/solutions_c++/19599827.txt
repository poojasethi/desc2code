#include <iostream>
#include <algorithm>
#include <cctype>
#include <cmath>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <vector>
#include <set>
#define N 200100
#define LL long long
using namespace std;
int n, K, cnt;
pair<int, int> t[N], u;
int mark[N];
set<pair<int, int> > s, p;

int main(){
	int i, j, k;
	scanf("%d%d", &n, &K);
	for (i = 1; i <= n; ++ i){
		scanf("%d%d", &t[i].first, &t[i].second);
		s.insert(t[i]);
	}
	sort(t + 1, t + 1 + n);
	if (n <= K) puts("-1");
	else {
		int cnt = 0;
		for (i = 1; i <= K + 1; ++ i){
			for (j = n; j >= i && j >= n - K; -- j){
				u = make_pair(t[i].first + t[j].first, t[i].second + t[j].second);
				int num = 0;
				if (p.count(u)) continue;
				for (k = 1; k <= n; ++ k){
					num += !s.count(make_pair(u.first - t[k].first, u.second - t[k].second));
					if (num > K) break;
				}
				if (num <= K){
					p.insert(u);
				}
			}
		}
		printf("%d\n", p.size());
		for (set<pair<int, int> >::iterator iter = p.begin(); iter != p.end(); ++ iter)
			printf("%lf %lf\n", iter->first * 0.5, iter->second * 0.5);
	}
	return 0;
}
