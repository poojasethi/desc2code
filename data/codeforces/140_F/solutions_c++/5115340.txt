#include<cstdio>
#include<algorithm>
#include<set>
using namespace std;
pair<long long,long long> P[200001],tmp,tmp2;
set <pair<long long,long long> > S;
set <pair<long long,long long> > ::iterator it;
int n,k;
int main(){
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++)
        scanf("%I64d%I64d",&P[i].first,&P[i].second);
    sort(P,P+n);
    if(n <= k) printf("-1\n");
    else{
        for(int i=0;i<=k;i++)
            for(int j=n-1-k+i;j<n;j++){
                int s = 0;
                tmp.first = P[i].first+P[j].first;
                tmp.second = P[i].second+P[j].second;
                for(int l1=0,r1=n-1;l1<=r1;){
                    tmp2.first = P[l1].first+P[r1].first;
                    tmp2.second = P[l1].second+P[r1].second;
                    if(tmp == tmp2) l1++,r1--;
                    else{
                        s++;
                        if(tmp > tmp2) l1++;
                        else r1--;
                    }
                }
                if(s <= k)
                    S.insert(tmp);
            }
        printf("%d\n",S.size());
        for(it = S.begin();it != S.end();++it)
            printf("%lf %lf\n",0.5*it->first,0.5*it->second);
    }
    return 0;
}
