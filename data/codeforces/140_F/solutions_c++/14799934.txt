#include<bits/stdc++.h>
using namespace std;

struct point
{int x,y;
};
struct point a[200010],p0;

bool operator <(point p1,point p2)
{return p1.x<p2.x || (p1.x==p2.x && p1.y<p2.y);
}

int n,k,ans;
vector<point> w;

int check()
{int i,j,ret=0;
 point p1;
 
 j=n;
 for(i=1;i<=n;i++)
 {  //cout<<i<<" "<<j<<endl;
    if(i>j) break;
    p1.x=p0.x-a[i].x;
    p1.y=p0.y-a[i].y;
    while((p1.x!=a[j].x || p1.y!=a[j].y) && !(a[j]<p1) && j>=i) j--;
    if(j>=i && a[j].x==p1.x && a[j].y==p1.y)
    {   ret+=2-(j==i);
        j--;
    }
 }
 return ret>=n-k;
}

int main()
{int i,j;
 
 scanf("%d%d",&n,&k);
 for(i=1;i<=n;i++)
    scanf("%d%d",&a[i].x,&a[i].y);
 if(k>=n)
 {  printf("%d\n",-1);
    return 0;
 }
 sort(a+1,a+n+1);
 for(i=1;i<=k+1;i++)
    for(j=n;j>=n-k;j--)
    {   p0.x=a[i].x+a[j].x;
        p0.y=a[i].y+a[j].y;
        //cout<<i<<" "<<j<<endl;
        if(i<=j && check())
            w.push_back(p0);
    }
 sort(w.begin(),w.end());
 ans=w.size();
 for(i=1;i<w.size();i++)
    if(w[i].x==w[i-1].x && w[i].y==w[i-1].y)
        ans--;  
 printf("%d\n",ans);
 for(i=0;i<w.size();i++)
    if(i==0 || w[i].x!=w[i-1].x || w[i].y!=w[i-1].y)
        printf("%.1lf %.1lf\n",w[i].x/2.0,w[i].y/2.0);  
 return 0;
}
