#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
struct node{double x,y;}a[200005],ans[200005];
int b,c,d,e,f,g,h,i,j,k,l,m,n;
inline bool cmp(node a,node b){return a.x<b.x||(a.x==b.x&&a.y<b.y);}
bool check(double x,double y){
	g=m;
	for(int i=1,j=n;i<=j&&g>=0;){
		node A=(node){x-a[i].x,y-a[i].y};
		node B=(node){a[j].x-x,a[j].y-y};
		if(abs(A.x-B.x)<1e-8&&abs(A.y-B.y)<1e-8){i++;j--;}
		else if(cmp(A,B)){j--;g--;}
		else {i++;g--;}
	}
	return g>=0;
}
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)scanf("%lf%lf",&a[i].x,&a[i].y);
	if(check(-12345,12345)&&check(23333,-233333)){printf("-1");return 0;}
	sort(a+1,a+1+n,cmp);
	for(i=1;i<=m+1;i++)
	for(j=n;j>=n-m;j--){
		if(check((a[i].x+a[j].x)/2,(a[i].y+a[j].y)/2))
		ans[++h]=(node){(a[i].x+a[j].x)/2,(a[i].y+a[j].y)/2};
	}
	sort(ans+1,ans+1+h,cmp);ans[h+1].x=ans[h].x+1;
	for(i=1;i<=h;i++)if(abs(ans[i].x-ans[i+1].x)>1e-8||abs(ans[i].y-ans[i+1].y)>1e-8)ans[++l]=ans[i];
	printf("%d\n",l);
	for(i=1;i<=l;i++)printf("%.9lf %.9lf\n",ans[i].x,ans[i].y);
	return 0;
}
