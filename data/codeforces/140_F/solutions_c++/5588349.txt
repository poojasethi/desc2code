#include<iostream>
#include<cstdio>
#include<set>
#include<algorithm>
using namespace std;

#define X first
#define Y second
typedef pair<int, int> PII;
typedef set<PII> si;
typedef si::iterator sii;
int n, m;
PII p[222222];
si ans;

int solve(PII o){
    int j = n, ret = 0;
    for (int i = 1; i <= n && i <= j; ++i){
        for (; j >= i && p[j].X + p[i].X > o.X || (p[j].X + p[i].X == o.X && p[j].Y + p[i].Y > o.Y); --j, ret++);
        if (i > j) break;
        if (p[j].X + p[i].X == o.X && p[j].Y + p[i].Y == o.Y) j--; else ret++;
    }
    return ret;
}

int main(){
    scanf("%d%d", &n, &m);
    if (m >= n) {puts("-1"); return 0;}
    for (int i = 1; i <= n; ++i) scanf("%d%d", &p[i].X, &p[i].Y);
    sort(p + 1, p + 1 + n);
    for (int i = 1; i <= m + 2 && i <= n; ++i)
        for (int j = 1; j <= m + 2 && n - j >= 0; ++j){
            PII x = make_pair(p[i].X + p[n - j + 1].X, p[i].Y + p[n - j + 1].Y);
            if (solve(x) <= m) ans.insert(x);
        }
    printf("%d\n", ans.size());
    for (sii i = ans.begin(); i != ans.end(); i++)
        printf("%.10f %.10f\n", (*i).X * 0.5, (*i).Y * 0.5);
}
