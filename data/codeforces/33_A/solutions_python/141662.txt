#!/usr/bin/env python

import sys

(n, m, k) = map(int, sys.stdin.readline().rstrip('\n').split(' '))
row = [1000000] * (m + 1);

for i in xrange(0, n):
    (r, c) = map(int, sys.stdin.readline().rstrip('\n').split(' '))
    row[r] = min(row[r], c)

sum = 0
for i in range(1, m + 1):
    sum += row[i]

print min(sum, k)
