def readint(): return int(raw_input())
def readarray(foo): return [foo(x) for x in raw_input().split()]

def run():
	(n, m, k) = readarray(int)
	l = {}
	for i in xrange(n):
		(a, b) = readarray(int)
		if a not in l: l[a] = []
		l[a].append(b)
	res = min(sum([min(l[a]) for a in l]), k)
	print res
run()
