#!/usr/bin/env python
# -*- coding: utf-8 -*-

def main():
    n, m, k = map(int, raw_input().split())
    l = [10**7 for _ in xrange(m)]
    for _ in xrange(n):
        r, c = map(int, raw_input().split())
        l[r-1] = min(l[r-1], c)
    print min(k, sum(l))
    return 0

if __name__ == '__main__':
    main()
