T = map(int,raw_input().strip().split())
n = T[0]
m = T[1]
k = T[2]
L = []
for i in range (n):
    L.append(map(int,raw_input().strip().split()))
A = [1000000] * m
for i in range (n):
    A[L[i][0]-1] = min(A[L[i][0]-1],L[i][1])
ans = 0
for i in range (m):
    ans = ans + A[i]
if ans > k:
    ans = k
print ans
