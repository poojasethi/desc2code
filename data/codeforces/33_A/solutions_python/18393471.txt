#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n, m, k = ia()

rmin = [1e13 for e in range(m)]
for i in range(n):
    r, c = ia(); r-=1
    rmin[r] = min(rmin[r], c)


print min(k,   sum(rmin))
