from itertools import *
from collections import defaultdict


a, b, c = map(int, raw_input().split())
Too = [map(int, raw_input().split()) for i in xrange(a)]
levs = defaultdict(list)

for i in Too:
    if levs[i[0]] or levs[i[0]] == 0:
        levs[i[0]] = min(levs[i[0]], i[1])
    else:
        levs[i[0]] = i[1]

counter = 0
for i in levs:
    counter = counter + levs[i]
    
print min(counter, c)
