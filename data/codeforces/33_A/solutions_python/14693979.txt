import sys
import string

def main():
    n,m,p=map(int,sys.stdin.readline().split())
    a=[100000000]*(m+1)
    for i in range(n):
        u,v=map(int,sys.stdin.readline().split())
        a[u]=min(a[u],v)
    ans=0
    for i in range(1,m+1):
        ans=ans+a[i]
    print min(ans,p)

if __name__=='__main__':
    main()
