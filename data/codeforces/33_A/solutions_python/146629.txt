#!/usr/bin/python
import sys
(n,m,k) = map(int,sys.stdin.readline().split(' '))
MAX       = 1000000+10
mr      = [MAX]*1010
for i in xrange(n):
    (r, c) = map(int, sys.stdin.readline().split(' '))
    mr[r]  = min(mr[r],c)
tot    = 0
for s in range(1,m+1):
    if s != MAX:
        tot += mr[s]
tot   = min(tot, k)
print tot
