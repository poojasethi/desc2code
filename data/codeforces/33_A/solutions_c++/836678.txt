#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main(){
 long n,m,k,s=0;
 cin>>n>>m>>k;
 vector<long> P(m,-1);
 for(int i=0;i<n;i++){
  long d,s;
  cin>>d>>s;
  if(P[d-1]>s || P[d-1]==-1)P[d-1]=s;
 }
 for(int i=0;i<m;i++)s+=P[i];
 if(s>k)cout<<k;
 else cout<<s;
 //system("pause");
 return 0;
}
