#include <cstdio>
int low[1001],sum=0,n,m,k,r,c,i;
int main (){
    scanf("%d %d %d",&n,&m,&k);
    for (i=1;i<=m;i++) low[i]=2147483647;
    for (i=0;i<n;i++){
        scanf("%d %d",&r,&c);
        if (low[r]>c) low[r]=c;
    }
    for (i=1;i<=m;i++) sum+=low[i];
    if (sum<k) printf("%d\n",sum); else printf("%d\n",k);
    return 0;
}
