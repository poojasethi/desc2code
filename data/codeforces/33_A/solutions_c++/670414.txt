#include <cstdio>
int n, m, k, r, c, s, v[1001];
int main() {
    scanf("%d%d%d", &n, &m, &k);
    while (n--) {
        scanf("%d%d", &r, &c);
        if (c - 2000000 < v[r]) v[r] = c - 2000000;
    }
    for (int i = 1;i <= m;i++) if (v[i]) s += v[i] + 2000000;
    printf("%d\n", s > k ? k : s);
}
