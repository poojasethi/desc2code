#include <iostream>

using namespace std;

main()
{
	long m,n,k,r,c;
	long row[1001];
	long i,j;
	long s=0;
	cin >> n >> m >> k;
	for (i=1;i<=m;i++) row[i]=1000000;
	for (i=1;i<=n;i++)
	{
		cin >> r >> c;
		if (c<row[r]) row[r]=c;
	}
	for (i=1;i<=m;i++) s=s+row[i];
	if (s<k) cout << s << endl;
	else cout << k << endl;
}
