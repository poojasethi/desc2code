#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main()
{
  int n,m,k;
  cin>>n>>m>>k;
  vector<int> v(m,1000001);
  for (int i=0;i<n;i++) {
    int r,c;
    cin>>r>>c;
    r--;
    v[r]=min(v[r],c);
  }
  int s=0;
  for (int i=0;i<m;i++)
    s+=v[i];
  cout<<min(s,k)<<endl;  
}
