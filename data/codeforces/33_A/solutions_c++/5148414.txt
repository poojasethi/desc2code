#include<iostream>
using namespace std;
int M = 10000000;
int main()
{
   int a[1001]={0}, i,n,k,r,m, s=0, l;
   cin>>n>>k>>l;
   for(i=1;i<=k;i++) a[i]=M;
   for(i=0;i<n;i++) cin>>r>>m, a[r]=min(a[r],m);
   for(i=1;i<=k;i++) if (a[i]!=M) s+=a[i];
   cout<<min(s,l);
//   getchar(); getchar();
   return 0;
}
