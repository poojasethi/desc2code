#include<iostream>
using namespace std;
const int maxn=1010,inf=1000000000;
int arr[maxn];
int main(){
	fill(arr,arr+maxn,inf);
	int n,m,k,ans=0;
	cin>>n>>m>>k;
	for(int i=0;i<n;i++){
		int r,c;
		cin>>r>>c;r--;
		arr[r]=min(c,arr[r]);
	}
	for(int i=0;i<m;i++)
		if(arr[i]!=inf)
			ans+=arr[i];
	cout<<min(ans,k)<<endl;
}
