#include <iostream>
using namespace std;

int mn[10001];

int main(){
	int n,m,k,a,r;
	cin>>n>>m>>k;
	for (int i=0; i<m; ++i)
		mn[i]=1000000001;
	for (int i=0; i<n; ++i){
		cin>>r>>a;
		r--;
		if (mn[r]>a) mn[r]=a;
	}
	int ans=0;
	for (int i=0; i<m; ++i)
		ans+=mn[i];
	if (ans<k)
		cout<<ans;
	else
		cout<<k;
}
