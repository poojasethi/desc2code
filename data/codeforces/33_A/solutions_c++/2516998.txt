#include<iostream>
using namespace std;
int main()
{
long int n,m,k;
cin>>n>>m>>k;
int ar[m+1];
for(int i=1;i<=m;i++)
ar[i]=-100;
int a,b;
int sum=0;
for(int i=0;i<n;i++) {
	cin>>a>>b;
	if(ar[a]==-100)
	ar[a]=b;
	if(ar[a]>b)
	ar[a]=b;
}
for(int i=1;i<=m;i++) {
	sum=sum+ar[i];
}
if(sum<k)
cout<<sum;
else cout<<k;
}
