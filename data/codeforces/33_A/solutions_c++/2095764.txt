#include<stdio.h>
int main(){
	int s[1010];
	int i,m,n,k,a,b;
	__int64 sum;
	while(~scanf("%d%d%d",&n,&m,&k)){
		for(i=0;i<=m;i++)   s[i]=-1;
		for(i=0;i<n;i++){
			scanf("%d%d",&a,&b);
			if(s[a]>b||s[a]==-1)  s[a]=b;
		}
		sum=0;
		for(i=1;i<=m;i++)
		    sum+=s[i];
		printf("%I64d\n",sum<k?sum:k);
	}
	return 0;
}
