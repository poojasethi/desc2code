#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
  vector <int> v;
  int n, m, k, r, c, ma(0), i;
  cin >> n >> m >> k;
  v.resize(m, 1000000);
  for(i = 0; i < n; i++){
    cin >> r >> c;
    v[r - 1] = min(v[r - 1], c);
  }
  for(i = 0; i < m; i++){
    ma += v[i];
  }
  cout << min(k, ma) << endl;
}
