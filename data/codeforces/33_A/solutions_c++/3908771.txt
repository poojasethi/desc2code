#include<stdio.h>
#include<limits.h>
int n,m,k;
int A[1005];
void set()
{
	int i;
	for(i=0;i<1005;i++)
		A[i]=INT_MAX;
}
int main()
{
	int sum=0,i,r,c;
	scanf("%d %d %d",&n,&m,&k);
	set();
	for(i=0;i<n;i++)
	{
		scanf("%d %d",&r,&c);
		if(c<A[r])
			A[r]=c;
	}
	for(i=1;i<=m;i++)
		sum+=A[i];
	printf("%d\n",(sum<k)? sum : k);

	return 0;
}
