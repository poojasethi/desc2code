#include<iostream>
#include<vector>
using namespace std;

int main() {
  int n,m,k,r,c;
  cin>>n>>m>>k;
  vector<int> in(m,100000000);
  for(int i=0; i<n; ++i) {
    cin>>r>>c;
    in[r-1] = min(in[r-1], c);
  }

  int sum = 0;
  for(int i=0; i<m; ++i) {
    sum += in[i];
  }

  cout<<min(sum,k)<<endl;
}
