#include <iostream>
#include <vector>
using namespace std;
int main(){
	int i,j,x,y;
	int n,m,k;
	
	cin>>n>>m>>k;
	
	vector<int> v(m,10000000);
	for(i=0;i<n;i++){
		cin>>x>>y;
		v[x-1] = min(v[x-1],y);
	}
	
	x = 0;
	for(i=0;i<v.size();i++){
		x += v[i];
		if(x > k){
			x = k;
			break;
		}
	}
	cout<<x<<endl;
	return 0;
}
