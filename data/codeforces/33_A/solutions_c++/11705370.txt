#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,k,r,c;
	int ans=0;
	cin>>n>>m>>k;
	int a[1005];
	for(int i=0;i<1005;i++)
		a[i]=INT_MAX;
	for(int i=0;i<n;i++)
	{
		cin>>r>>c;
		a[r]=min(c,a[r]);
	}
	for(int i=0;i<1005;i++)
	{
		if(a[i]!=INT_MAX)
			ans+=a[i];
	}
	cout<<min(k,ans)<<endl;
	return 0;
}
