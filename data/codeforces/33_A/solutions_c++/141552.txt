#include <cstdio>

int mn[1000];

int main() {
	int n,m,k;
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0; i<m; ++i) mn[i]=1000005;
	for(int i=0; i<n; ++i) {
		int a,b;
		scanf("%d%d",&a,&b);
		if(b<mn[a-1]) mn[a-1]=b;
	}
	int res=0;
	for(int i=0; i<m; ++i) res+=mn[i];
	if(res<k) printf("%d\n",res);
	else printf("%d\n",k);
	return 0;
}
