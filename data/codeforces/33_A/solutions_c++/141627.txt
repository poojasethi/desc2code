#include <iostream>
#include <vector>

using namespace std;

int main ()
{
	int n, m, k; cin >> n >> m >> k;
	vector<int> c(m, 1000000000);
	for(int i = 0; i < n; i++)
	{
		int q, s; cin >> q >> s; q--;
		if(c[q] > s) c[q] = s;
	}

	int r = 0;
	for(int i = 0; i < m; i++)
		r += c[i];
	cout << min(k, r) << endl;
}
