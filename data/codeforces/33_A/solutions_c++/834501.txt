#include <iostream>

using namespace std;

main ()
{
    int i,a[1001],n,m,k,mm,h;
    cin >>n>>m>>k;
    for (i=1;i<=m;i++)
        a[i]=1000000;
    for (i=1;i<=n;i++)
    {
        cin>>mm>>h;
        a[mm]=min(a[mm],h);
    }
    h=0;
    for (i=1;i<=m;i++)
    {
        h+=a[i];
    }
    cout<<min(h,k);

}
