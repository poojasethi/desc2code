#include<cstdio>
using namespace std;
int a[1005];
int main(){
	int n,m,k,c,b,ans;
	ans=0;
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=m;i++){
		a[i]=2147483647;
	}
	for(int i=1;i<=n;i++){
		scanf("%d%d",&c,&b);
		if(a[c]>b)a[c]=b;
	}
	for(int i=1;i<=m;i++)ans+=a[i];
	if(ans>k)printf("%d",k);else printf("%d",ans);
	return 0;
}
