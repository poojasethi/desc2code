#include<iostream>
#include<vector>
using namespace std;
int main()
{
	int n,m,r,c;
	long long k,s = 0;
	cin>>n>>m>>k;
	vector<int> g(m,1000001);
	for(int i = 0; i < n; i++)
	{
		cin>>r>>c;
		g[r - 1] = min(g[r - 1],c);
	}
	for(int i = 0; i < m; i++)
		s += g[i];
	if (s > k) s = k;
	cout<<s;		
	return 0;
}
