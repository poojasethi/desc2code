#include<stdio.h>
int x[1005];
int main()
{
	int n,m,k,i,a,b,nr=0;
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=m;i++)
		x[i]=1000000;
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&a,&b);
		if(b<x[a])
			x[a]=b;
	}
	for(i=1;i<=m;i++)
		nr=nr+x[i];
	if(nr>k)
		printf("%d",k);
	else
		printf("%d",nr);
	return 0;
}
