#include<iostream>
using namespace std;
const int MN=1100;
int rw[MN],k,inf=1000000000;
int main(){
	int n,m,k;
	cin>>n>>m>>k;
	for(int i=0;i<m;i++){
		rw[i]=inf;
	}
	for(int i=0;i<n;i++){
		int x;
		cin>>x;
		int v;
		cin>>v;
		x--;
		rw[x]=min(rw[x],v);
	}
	int sum=0;
	for(int i=0;i<m;i++){
		sum+=rw[i];
	}
	cout<<min(sum,k)<<endl;
}
