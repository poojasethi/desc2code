#include <cstring>
#include <cstdio>
int main()
{
    int n,m,k,r,c,i,l[1010];
    while(~scanf("%d%d%d",&n,&m,&k))
    {
        memset(l,0x3f,sizeof l);
        for(i=1;i<=n;i++) scanf("%d%d",&r,&c),l[r]=l[r]<c?l[r]:c;
        n=0;
        for(i=1;i<=m;i++) n+=l[i];
        printf("%d\n",n<k?n:k);
    }
}
