#include <iostream>

using namespace std;

int main()
{
	int n,m,k;
	int r,c;
	cin >> n >> m >> k;
	int a[m+1];
	for (int i=1;i<=m;i++)
		a[i] = 1999999999;
	for (int i=0;i<n;i++)
	{
		cin >> r >> c;
		a[r] = min(a[r],c);
	}
	int sum=0;
	for (int i=1;i<=m;i++)
		sum+=a[i];
	cout << min(sum,k);
	return 0;
}
