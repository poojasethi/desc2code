#include<iostream>
using namespace std;
long n,m,k,r,c,rez,i,f[1001];

int main()
{
    cin>>n>>m>>k;
    for(i=1;i<=m;i++)
        f[i]=1100000;
    for(i=1;i<=n;i++)
    {
        cin>>r>>c;
        if(f[r]>c) f[r]=c;
    }
    for(i=1;i<=m;i++)
        rez=rez+f[i];
    if(rez<=k)cout<<rez;
    else cout<<k;
    return 0;
}
