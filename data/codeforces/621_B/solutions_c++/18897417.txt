#include<bits/stdc++.h>
using namespace std;

int a[2005], b[2005];
int main()
{
        long long int n, i, s = 0;
        int x, y;
        cin >> n;
        for(i = 0; i < n; i++) {
              scanf("%d%d", &x,&y);
              s += a[x + y]++ + b[1000 + x - y]++;
        }
        cout << s;
}
