#include<cstdio>
int a[2005],b[2005];
main(){
	int n,i,j;
	scanf("%d",&n);
	int x,y;
	for(i=1;i<=n;i++){
			scanf("%d %d",&x,&y);
			a[x+y]++;
			b[x-y+1000]++;
		}
	long long int count=0;
	for(i=1;i<=2000;i++){
			count=count+a[i]*(a[i]-1)/2;
			count=count+b[i]*(b[i]-1)/2;
		}
	printf("%I64d",count);		
}
