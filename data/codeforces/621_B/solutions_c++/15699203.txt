#include <cstdio>
#include <map>
using namespace std;
int main()
{
	int n,x,y;
	scanf("%d",&n);
	map<int,int> m1,m2;
	map<int,int>::iterator it;
	long long ans=0;
	for(int i=0;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		ans+=m1[x+y]+m2[x-y];
		m1[x+y]++;
		m2[x-y]++;
	}
	printf("%I64d\n",ans);
	return 0;
}
