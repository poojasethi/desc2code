#include<iostream>
#include<map>
using namespace std;

typedef long long ll;
ll n,ans=0;
map<ll,ll>M1,M2;
int main()
{
    cin>>n;
    ll x,y;
    for(int i=0;i<n;i++)
	{
        cin>>x>>y;
        ans+=M1[x+y];
        M1[x+y]++;
        ans+=M2[x-y];
        M2[x-y]++;
    }
    cout<<ans;
    return 0;
}
