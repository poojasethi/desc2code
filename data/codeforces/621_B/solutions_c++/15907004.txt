#include <iostream>
#include <map>

int main()
{
	std::map<int,int> diag, adiag;
	unsigned long long pairs = 0;
	int n, x, y;
	std::cin>>n;
	for(int i = 0; i < n; ++i)
	{
		std::cin>>x;
		std::cin>>y;
		pairs += diag[x-y]++;
		pairs += adiag[x+y]++;
	}
	std::cout<<pairs<<std::endl;
	return 0;
}
