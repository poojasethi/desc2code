#include <bits/stdc++.h>
using namespace std;

int d[10000];

int main() {
	// your code goes here
	int n,i,x,y;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>x>>y;
		d[x+y]++;
		d[x-y+3000]++;
	}
	long long int ans=0;
	
	for(i=0;i<=10000;i++)
	{
		ans+=(d[i]*(d[i]-1))/2;
	}
	cout<<ans<<endl;
	

	return 0;
}
