#include <iostream>

using namespace std;

long long v[4010];

int main()
{
    long long s=0,n,r,c,ll=1000,lr=2001;
    cin >> n;
    for(int i=0;i<n;i++)
    {
        cin>>r>>c;
        v[ll+c-r]++;
        v[lr+c+r]++;
    }
    for(int i=0;i<4010;i++) s+=(v[i]*(v[i]-1))/2;
    cout << s;
    return 0;
}
