#include<iostream>
#include<map>
using namespace std;
map<int,long long>a,b;
int main()
{
    int n,x,y;
    long long ans=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>x>>y;
        ans+=a[x-y]+b[x+y];
        a[x-y]++;
        b[x+y]++;
    }
    cout<<ans<<endl;
}
