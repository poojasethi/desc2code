#include<iostream>
using namespace std;
int main(){
int n,count=0,i,x1,y1,x[2005]={0},y[2005]={0};
cin>>n;
for(i=0;i<n;++i){
	cin>>x1>>y1;
	x[x1+y1]++;
	y[x1-y1+1001]++;
}
for(i=2;i<2001;++i){
	count+=x[i]*(x[i]-1)/2;
	count+=y[i]*(y[i]-1)/2;
}
cout<<count<<endl;
return 0;
}
