#include <iostream>
#include <stdio.h>
using namespace std;
#define Maxn 2010
long cc[Maxn]={0},cp[Maxn]={0};
long n;
int main()
{
    cin>>n;
    long rest=0;
    for(int i=1;i<=n;i++)
    {
        int x,y;
        cin>>x>>y;
        rest+=(cc[x-y+1001]++)+(cp[x+y]++);
    }
     cout<<rest<<endl;
}
