#include<bits/stdc++.h>
using namespace std;
int d1[2021],d2[2021];
int main(){
    int n,x,y,i,ans=0;
    cin>>n;;
    for(i=1;i<n+1;i++){
        scanf("%d%d",&x,&y);
        d1[x+y]++,d2[1001+(x-y)]++;
    }
    for(i=0;i<2010;i++)ans+=(d1[i]*(d1[i]-1)/2+d2[i]*(d2[i]-1)/2);
    cout<<ans;
    return 0;
}
