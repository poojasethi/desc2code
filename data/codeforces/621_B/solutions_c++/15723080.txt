#include <bits/stdc++.h>
typedef long long ll;
using namespace std;
map<int,int> mp1,mp2;
int main()
{
    long long ans=0;
    int T,x,y;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d %d",&x,&y);
        ans+=mp1[x+y]++;
        ans+=mp2[x-y]++;
    }
    printf("%I64d\n",ans);
    return 0;
}
