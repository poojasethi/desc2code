#include <bits/stdc++.h>
#define MAX_N 1010 

using namespace std;

int a[2 * MAX_N] , b[2 * MAX_N]; 
int res; 

int main ()
{
	int n; 
	
	cin >> n;
	
	for (int i = 1; i <= n; i++) 
	{
		int x,y; 
		cin >> x >> y; 
		
		res += (a[x + y]++ + b[1002 + x - y]++); 
	}
	
	cout << res;
	return 0;
}
