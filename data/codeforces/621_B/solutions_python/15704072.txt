import sys

n = int(sys.stdin.readline())

left = [0]*2000
right = [0]*2000
rot = range(1001, -1, -1)

for i in xrange(n):
    x, y = map(int, sys.stdin.readline().split())
    left[x - y + 1000] += 1
    right[x - rot[y] + 1000] += 1

s = 0
for x in left:
    if x == 0:
        continue
    s += x*(x-1) / 2

for x in right:
    if x == 0:
        continue
    s += x*(x-1) / 2

print >>sys.stdout, s
