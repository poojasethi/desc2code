n = int(raw_input())
add_arr = [0] * 1999
sub_arr = [0] * 1999

i = 0
attacks = 0
while i < n:
	point = raw_input()
	x,y = map(int,point.split())

	add_arr[x+y-2] += 1
	sub_arr[y-x+999] +=1

	i += 1

i = 0
for a,s in zip(add_arr,sub_arr):
	attacks += (a*(a-1)/2)
	attacks += (s*(s-1)/2)

print attacks
