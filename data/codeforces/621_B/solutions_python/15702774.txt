#!/usr/bin/env python
from heapq import heappush, heappop
from math import hypot, factorial

n = int(raw_input())
arr = [tuple(map(int, raw_input().split())) for _ in range(n)]

downdiagonal = [0]*3000
updiagonal = [0]*3000
offset = 1000
def calcrow(i, j):
    downdiagonal[offset + (i-j)] += 1
    updiagonal[i+j] += 1

for x, y in arr:
    calcrow(x, y)

pairs = 0
for i in downdiagonal: 
    pairs += i*(i-1)/2
for i in updiagonal:
    pairs += i*(i-1)/2

print pairs
