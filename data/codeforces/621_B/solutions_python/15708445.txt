a = {}
b = {}

k = 1000 - 1
for i in range(2000 - 1):
    a[k] = 0
    k -= 1

k = (1000 - 1) * 2
for i in range(2000 - 1):
    b[k] = 0
    k -= 1

attack = 0

n = int(raw_input())

for i in range(n):
    x, y = raw_input().split(' ')
    x = int(x) - 1
    y = int(y) - 1
    
    attack += a[x -y] + b[y+x]
    
    a[x -y] += 1
    b[y+x] += 1    
    
print attack
