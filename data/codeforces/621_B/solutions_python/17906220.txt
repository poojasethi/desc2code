x = int(raw_input())
diagonais = {}
dia = {}
for i in range(x):
	a,b = map(int, raw_input().split())
	if a+b in dia:
		dia[a+b] += 1
	if a+b not in dia:
		dia[a+b] = 1
	if a-b in diagonais:
		diagonais[a-b] += 1
	else:
		diagonais[a-b] = 1			
soma = 0
for i in diagonais:
	soma += diagonais[i]*(diagonais[i]-1)/2
for i in dia:
	soma += dia[i]*(dia[i]-1)/2
print soma
