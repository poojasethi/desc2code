l, r = {}, {}
for _ in xrange(input()):
    a = map(int, raw_input().split())
    if a[0]-a[1] in l:
        l[a[0]-a[1]] += 1
    else:
        l[a[0]-a[1]] = 1
    if a[0]+a[1] in r:
        r[a[0]+a[1]] += 1
    else:
        r[a[0]+a[1]] = 1
ans = 0
for x in l:
    ans += (l[x] * (l[x] - 1))/2
for x in r:
    ans += (r[x] * (r[x] - 1))/2
print ans
