from collections import defaultdict

n = input()
grid = [[False for i in xrange(1000)] for j in xrange(1000)]
dic1, dic2 = defaultdict(int), defaultdict(int)
result = 0
for i in xrange(n):
	x1, y1 = map(lambda x: int(x)-1, raw_input().split())
	dic1[x1 - y1] += 1
	dic2[x1 + y1] += 1
for n in dic1.itervalues():
	result += (n*(n-1))/2
for n in dic2.itervalues():
	result += (n*(n-1))/2
print result
