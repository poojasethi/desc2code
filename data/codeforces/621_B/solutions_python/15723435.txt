from math import *
from Queue import *
from sys import *




n = int(raw_input())
d = dict()
ad = dict()
for i in range(n):
    x, y = map(int, raw_input().split())
    try:
        d[x-y] += 1
    except:
        d[x-y] = 1
    try:
        ad[x+y] += 1
    except:
        ad[x+y] = 1
res = 0
for i in d:
    res += (d[i]*(d[i]-1))/2
for i in ad:
    res += (ad[i]*(ad[i]-1))/2
print res
