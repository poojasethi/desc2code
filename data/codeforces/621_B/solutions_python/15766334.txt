from collections import defaultdict

n = input()
diag1 = defaultdict(int)
diag2 = defaultdict(int)
for i in range(n):
    x = map(int, raw_input().split())
    diag1[x[0]-x[1]] += 1
    diag2[x[0]+x[1]] += 1

values = diag1.values()+diag2.values()
ans = sum(map(lambda v: v*(v-1)/2, values))

print ans
