n = int(raw_input())

p = {}
s = {}

for i in range(n):
	a, b = map(int, raw_input().split())
	if (a + b) in p:
		p[a + b] += 1
	else:
		p[a + b] = 1
	if (a - b) in s:
		s[a - b] += 1
	else:
		s[a - b] = 1

v1 = 0
v2 = 0


for k in p:
	v1 += (p[k] * (p[k] - 1))/2
for z in s:
	v2 += (s[z] * (s[z] - 1))/2

print v1 + v2
