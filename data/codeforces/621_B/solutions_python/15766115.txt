n = input()
diag1 = {}
diag2 = {}
for i in range(n):
    x = map(int, raw_input().split())
    value1 = x[0]-x[1]
    if value1 not in diag1:
        diag1[value1] = 1
    else:
        diag1[value1] += 1
    value2 = x[0]+x[1]
    if value2 not in diag2:
        diag2[value2] = 1
    else:
        diag2[value2] += 1

ans = 0
for value in diag1:
    num = diag1[value]
    ans += num*(num-1)/2;
for value in diag2:
    num = diag2[value]
    ans += num*(num-1)/2;

print ans
