n=input()
sum=[]
diff=[]
diff_d={}
sum_d={}

for _ in xrange(n):
    x,y=map(int,raw_input().split())
    sum.append(x+y)
    diff.append(x-y)
p=0
for i in sum:
    sum_d[i]=sum_d.get(i,0)+1

for i in diff:
    diff_d[i]=diff_d.get(i,0)+1

for i in sum_d:
    p+=(sum_d[i]*(sum_d[i]-1))/2

for i in diff_d:
    p+=(diff_d[i]*(diff_d[i]-1))/2
print p
