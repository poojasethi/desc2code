n = int(raw_input())
points = [tuple(map(int, raw_input().split())) for _ in xrange(n)]
up, down = {}, {}
for x, y in points:
    up[y+x] = up.get(y+x, 0) + 1
    down[y-x] = down.get(y-x, 0) + 1
total = 0
for k, v in up.items():
    total += v * (v-1) // 2
for k, v in down.items():
    total += v * (v-1) // 2
print total
