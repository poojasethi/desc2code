n = int( raw_input() )

memo1 = {}
memo2 = {}

ans = 0

for i in range( 0, n ):
	
	x, y = map( int, raw_input().split( ' ' ) )
	
	if (x+y) not in memo1:
		memo1[ (x+y) ] = 0
		
	if (x-y) not in memo2:
		memo2[ (x-y) ] = 0
		
	ans += memo1[ ( x+y ) ] + memo2[ ( x-y ) ]
		
	memo1[ ( x+y ) ] += 1
	memo2[ ( x-y ) ] += 1

print "%d" % (ans)
