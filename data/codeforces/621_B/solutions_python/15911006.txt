n = int(raw_input())

diag1 = [0 for i in range(2004)]
diag2 = [0 for i in range(2004)]

pos = [None for i in range(n)]

for i in range(n):
    pos[i] = map(int, raw_input().split())

ans = 0

for p in pos:
    ans += diag1[p[0]+p[1]] + diag2[p[0]-p[1]+1000]
    diag1[p[0]+p[1]] += 1
    diag2[p[0]-p[1]+1000] += 1

print ans
