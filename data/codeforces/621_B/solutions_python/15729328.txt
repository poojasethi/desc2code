from collections import Counter

n = int(raw_input())

pos = {}
neg = {}

counter = 0
for i in xrange(0, n):
    x,y = map(int, raw_input().split())
    p = x + y
    n = x - y
    if p not in pos: pos[p] = 0
    if n not in neg: neg[n] = 0
    pos[p] += 1
    neg[n] += 1

for i in pos.values():
    counter += (i * (i - 1) / 2)

for i in neg.values():
    counter += (i * (i - 1) / 2)

print counter
