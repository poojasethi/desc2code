def C2(n):
    return (n*(n-1))/2

def doit(l):
    ans = 0
    i = 0
    while i < len(l):
        j = i
        while j < len(l) and l[j] == l[i]: j += 1
        ans += C2(j-i)
        i = j
    return ans

n = int(raw_input())
d1 = []
d2 = []
for i in xrange(n):
    a, b = map(int, raw_input().split())
    d1.append(a-b)
    d2.append(a+b)
d1.sort()
d2.sort()

print doit(d1) + doit(d2)
