def comb(x):
    return x * (x-1) / 2

n = input()

store = []
diags = [0] * 2000
antidiags = [0] * 2000


for _ in range(n):
    q = map(int,raw_input().split())
    x = q[1] - q[0] + 1000
    diags[x] += 1
    antidiags[q[0]+q[1]-1] += 1


ans = sum(comb(x) for x in diags+antidiags)

print ans
