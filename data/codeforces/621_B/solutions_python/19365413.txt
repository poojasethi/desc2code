x = int(raw_input())

lista = []
dicio = {}
dicio2 = {}

for i in range(x):
	a,b = map(int,raw_input().split())
	lista.append((a,b))
k = 0
vet = 0 

for k in range(x):
	v = lista[k][0] - lista[k][1]
	if v not in dicio:
		dicio[v] = 0
	dicio[v] += 1
qtd = 0
for j in dicio:
	qtd +=  (dicio[j] * (dicio[j]-1))/2
	

for h in range(x):
	v2 = lista[h][0] + lista[h][1]
	if v2 not in dicio2:
		dicio2[v2] = 0
	dicio2[v2] += 1

for q in dicio2:
	qtd += (dicio2[q] * (dicio2[q]-1))/2
print qtd
