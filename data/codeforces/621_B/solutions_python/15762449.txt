__author__ = 'aste'

MAX = 1001


def main():
    n = int(raw_input())
    diag1 = [0] * (2*MAX)
    diag2 = [0] * (2*MAX)

    for i in range(0, n):
        x, y = map(int, raw_input().split())
        diag1[x + y] += 1
        diag2[MAX + x - y] += 1

    res = 0
    for i in range(1, 2*MAX):
        res += diag1[i]*(diag1[i] - 1)
        res += diag2[i]*(diag2[i] - 1)
    res /= 2
    print res

main()
