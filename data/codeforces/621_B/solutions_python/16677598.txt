n = int(raw_input())
a, b = {}, {}
ans = 0
for _ in range(n):
    x, y = map(int, raw_input().split())
    if a.get(x+y):
        a[x+y] += 1
    else:
        a[x+y] = 1
    if b.get(x-y+3000):
        b[x-y+3000] += 1
    else:
        b[x-y+3000] = 1

for i in a:
    ans += a[i]*(a[i] - 1) / 2
for i in b:
    ans += b[i]*(b[i] - 1) / 2

print ans
