def solve():
    mat1=[0]*(2000)
    mat2=[0]*(2001)
    n=input()
    for x in xrange(n):
        i,j=map(int,raw_input().split())
        mat1[j-i+1000]+=1
        mat2[i+j]+=1
    result=0
    for x in mat1:
        if x>1:
            result+=(x*(x-1))/2
    for x in mat2:
        if x>1:
            result+=(x*(x-1))/2
    print result

solve()
