n = input()
a = []
b = []
for _ in xrange(n):
    x,y = map(int,raw_input().split())
    a+=[(x+y)]
    b+=[(x-y)]
sa = set(a)
sb = set(b)
hs1 = {}
hs2 = {}
for i in sa:
    hs1[i]=0
for i in sb:
    hs2[i]=0
for i in a:
    hs1[i]+=1
for i in b:
    hs2[i]+=1
ans = 0
for i in sa:
    x= hs1[i]
    x *= (x-1)
    x >>= 1
    ans += x
for i in sb:
    x= hs2[i]
    x *= (x-1)
    x >>= 1
    ans += x
print ans
