from collections import Counter
n = int(raw_input())
pos =dict()
neg=dict()
ans = 0
for X in range(n):
	x,y = map(int, raw_input().split())
	p = x-y
	n = x+y
	if p in pos:
		pos[p]+=1
	else:
		pos[p] = 1
	
	if n in neg:
		neg[n]+=1
	else:
		neg[n] = 1
for i in pos.values():
	if i > 1:
		ans += (i*(i-1)/2)
for i in neg.values():
	if i > 1:
		ans += (i*(i-1)/2)
print ans
