from collections import defaultdict
n=int(raw_input())
sum = defaultdict(int)
diff = defaultdict(int)
for i in range (0, n):
    x,y = map(int, raw_input().split())
    sum[x+y] += 1
    diff[x-y] += 1

i = 0

for r in sum.values():
    i += float(0.5*r*(r-1))
for r in diff.values():
    i += float(0.5*r*(r-1))
print int(i)
