n=input()
p={};
s={};
for i in range(n):
    x,y=map(int,raw_input().split());
    b=x-y;
    c=x+y;
    if s.get(b,-1)==-1:
        s[b]=1;
    else:
        s[b]+=1;
    if p.get(c,-1)==-1:
        p[c]=1;
    else:
        p[c]+=1;
h=[];
z=0;
h=p.keys();
for i in range(len(h)):
    z+=(p[h[i]]*(p[h[i]]-1))/2;
    #print str(h[i])+"-->"+str(p[h[i]]);
h=s.keys();
for i in range(len(h)):
    z+=(s[h[i]]*(s[h[i]]-1))/2;
