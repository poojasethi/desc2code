n, k = map(int, raw_input().split())
a = list(reversed(map(int, raw_input().split())))

t = 0
while a[-1] != k:
    #print a
    r = a[0]
    if a[0] != k:
        a[0] += 1
    for i in range(1, n):
        if a[i] != r:
            a[i] += 1
            r = a[i] - 1
        else:
            r = a[i]
    t += 1
print t
