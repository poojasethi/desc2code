n, k = map(int, raw_input().split())
arr = []
arr = map(int, raw_input().split())
arr.append(k)
k_count = 0
for i in arr:
    if i == k:
        k_count += 1
n += 1
ans = 0
while(k_count != n):
    ans += 1
    for i in range(len(arr)):
        if arr[i] < k:
            if not (arr[i + 1] == arr[i]):
                arr[i] += 1
                if arr[i] >= k:
                    k_count += 1

print ans
