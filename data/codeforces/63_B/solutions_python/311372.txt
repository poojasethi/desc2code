#!/usr/bin/python
import math, random, re, sys
while True:
	s = sys.stdin.readline ().strip ()
	if s == '':
		s = sys.stdin.readline ().strip ()
		if s == '':
			break
	n, k = [int (x) for x in s.split ()]
	a = [int (x) for x in sys.stdin.readline ().split ()]
	v = [0] * k
	for c in a:
		v[c - 1] += 1
	res = 0
	while True:
		ok = False
		for c in range (k - 1) [::-1]:
			if v[c] > 0:
				ok = True
				v[c + 1] += 1
				v[c] -= 1
		if not ok:
			break
		res += 1
	print res
