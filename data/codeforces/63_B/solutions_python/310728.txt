import bisect

def soldier(k, ranks):
    n = 0
    while True:
        new_ranks = list(ranks)
        upgrade = False
        for rk in range(1, k):
            idx = bisect.bisect_right(ranks, rk)
            if idx > 0 and ranks[idx-1]==rk:
                upgrade = True
                new_ranks[idx-1] += 1
        if not upgrade:
            return n
        ranks = new_ranks
        n += 1

n, k = map(int, raw_input().split())
ranks = map(int, raw_input().split())
print soldier(k, ranks)
