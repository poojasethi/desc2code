import sys
from collections import deque

n_soldiers, highest = tuple(int(x) for x in sys.stdin.readline().split())
values = deque([int(x) for x in sys.stdin.readline().split()])
ranks, answer = {}, 0

for i in range(1, highest+1):
    ranks[i] = 0

for value in values:
    ranks[value] += 1

while ranks[highest] != n_soldiers:
#    print(ranks)
    answer += 1
    for i in range(1, highest)[::-1]:
        if ranks[i] != 0:
            ranks[i+1] += 1
            ranks[i] -= 1

print(answer)
