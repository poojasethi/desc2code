import sys
import array

#input

total_sol, level_k = map(int, raw_input().split())

soldier = [0 for _ in range(123)]

arr = map(int, raw_input().split())

for idx in arr :
    soldier[idx] += 1

#process 
total_move = 0
while soldier[level_k] != total_sol :
    total_move += 1
    
    idx = level_k

    while idx > 0 :
        if (soldier[idx-1] > 0) :
            soldier[idx-1] -= 1
            soldier[idx] += 1
        idx -= 1

print total_move
