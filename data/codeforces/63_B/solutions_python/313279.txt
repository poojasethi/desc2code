import sys
import string

lines = sys.stdin.readlines()
goal = int(lines[0].rstrip().split(" ")[1])
nums = [int(i) for i in lines[1].rstrip().split(" ")]
c = 0
while 1:
    t = []
    for i in range(len(nums)):
        if (nums[i] not in t) and nums[i] < goal:
            t.append(nums[i])
            nums[i] += 1
    if len(t) == 0:
        break
    c += 1
print c
