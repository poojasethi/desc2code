n,k = map(int,raw_input().split())
a,b,ans = map(int,raw_input().split()),[0 for _ in xrange(k+1)],0
for i in xrange(n):
    b[a[i]] = b[a[i]]+1
while not b[k]==n:
    up,ans = 0,ans+1
    for i in xrange(1,k+1):
        if up and not b[i]==0:
            if i==k: b[i]=b[i]+1
        elif not b[i]==0 and not i==k: up,b[i] = 1,b[i]-1
        elif up: b[i],up=b[i]+1,0
        else: continue
print ans
