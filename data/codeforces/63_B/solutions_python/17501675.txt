def main():
	n, k = map(int, raw_input().split())
	a = map(int, raw_input().split())
	rank = [0] * (k + 1)
	for x in a: rank[x] += 1
	ans = 0
	while rank[k] != n:
		ans += 1
		for i in xrange(k - 1, 0, -1):
			if rank[i] > 0:
				rank[i + 1] += 1
				rank[i] -= 1
	print ans

if __name__ == "__main__": main()
