#!/usr/bin/env python
# -*- coding: utf-8 -*-

def main():
    n, k = map(int, raw_input().split())
    s = [0 for i in xrange(k)]
    for i in raw_input().split():
        s[int(i)-1] += 1
    c = 0
    while s[-1] != n:
        c += 1
        for i in xrange(k-2, -1, -1):
            if s[i] > 0:
                s[i] -= 1
                s[i+1] += 1
    print c
    return 0

if __name__ == '__main__':
    main()
