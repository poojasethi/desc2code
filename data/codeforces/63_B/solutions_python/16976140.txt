n,k = map(int, raw_input().split())
A = map(int, raw_input().split())
coins = 0
promoted = 0
while promoted < n:
    s = set()
    for i in xrange(n):
        if A[i] != k and A[i] != None and A[i] not in s:
            s.add(A[i])
            A[i] += 1
        if A[i] == k:
            promoted += 1
            A[i] = None
    if len(s) > 0:
        coins += 1

print coins
