n,k=map(int,(raw_input('').split()))
a=map(int,raw_input().split())
cnt=0
while 1:
    isInc=set()
    ok=False
    for it in a:
        if it < k:
            ok=True
    if not ok:
        break
    for i in range(0,len(a)):
        if not a[i] in isInc:
            isInc.add(a[i])
            a[i]+=1
    cnt+=1


print cnt
