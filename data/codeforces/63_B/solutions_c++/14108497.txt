#include<bits/stdc++.h>
using namespace std;
int n, k, x, ans, a[105];

int main(int argc, char **argv) {
        scanf("%d%d", &n, &k);
	for(int i=0 ; i<n ; ++i){
		scanf("%d", &x);
		++a[x];
	}
	while(a[k] < n) {
		for(int i=k-1 ; i>0 ; --i)
			if(a[i]) --a[i], ++a[i+1];
		++ans;
	}
	printf("%d\n", ans);
	return 0;
}
