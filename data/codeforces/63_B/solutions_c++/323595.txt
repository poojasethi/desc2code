#include <cstdio>

const int MAX_N = 105;

int am[MAX_N];

int main() {
	int n, k; scanf("%d %d", &n, &k);
	for (int i = 0; i < n; i++) {
		int x; scanf("%d", &x);
		am[x]++;
	}

	int ans = 0;
	while (am[k] < n) {
		ans++;
		for (int x = k - 1; x >= 1; x--) if (am[x] > 0)
			am[x]--, am[x + 1]++;
	}

	printf("%d\n", ans);

	return 0;
}
