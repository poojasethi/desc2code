# include <iostream>
# include <cmath>
using namespace std;
int main(){
int n;
int k;
cin>>n>>k;
int h[100];
int l;
int r;
for(int i=0;i<n;i++){
	cin>>r;
	h[i]=r;
}
l=0;
while(h[0]!=k){
	int i=0;
	while(i<n){
		while (h[i]==h[i+1] && i<n-1 ) i++;
		h[i]++;
		i++;
	}
	l++;
}
	cout<<l;
return 0;
}
