#include<iostream>

using namespace std;

int main () {
	int n, k, a[100], b[100], t=0;
	cin >> n >> k;

	for(int i = 0; i<n; i++) 
		cin >> a[i];
	
	while( a[0] != k) {
		for(int i = n-1; i>=0 ; i--) {
			b[i] = a[i];
			if(a[i]!=b[i+1] && a[i]!=k) 
				a[i]++; 
	   		}
	   	t++;
	   	}


	cout << t;
   	
}
