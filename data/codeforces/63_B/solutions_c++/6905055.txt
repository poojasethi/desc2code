#include<stdio.h>
int main()
{
	int n,k,i,time,a[105];
	while(scanf("%d%d",&n,&k) != EOF)
	{
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		time=0;
		for(;;)
		{
			if(a[0]==k)
				break;
			for(i=0;i<n-1;i++)
				if(a[i]!=a[i+1])
					a[i]++;
			if(a[n-1]!=k)
				a[n-1]++;
			time++;
		}
		printf("%d\n",time);
	}
	return 0;
}
