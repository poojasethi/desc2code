#include<cstdio>
int n,k,cnt[101];
//          Last Change:  2011-07-13 20:21:58
int main()
{
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;++i)
	{
		int x;
		scanf("%d",&x);
		++cnt[x];
	}
	int ans=0;
	for(;;)
	{
		if(cnt[k]==n)break;
		++ans;
		for(int i=k-1;i>0;--i)
			if(cnt[i])
				--cnt[i],++cnt[i+1];
	}
	printf("%d\n",ans);
	return 0;
}
