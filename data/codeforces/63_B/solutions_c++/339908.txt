#include<iostream>//To Aman,311181 
using namespace std;


int main()
{
    int n,k,i,a[105],p=0;
    cin>>n>>k;
    ++n;
    a[n]=k;
    for(i=1;i<n;i++)
        cin>>a[i];
    while(a[1]<k)
    {
        p++;
        for(i=1;i<n;i++)
        {
            if(a[i]-a[i+1]) a[i]++;
        }
    }
    cout<<p;
    return 0;
}
