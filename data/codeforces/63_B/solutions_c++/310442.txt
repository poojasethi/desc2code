#include <iostream>
#include <stack>
#include <vector>
using namespace std;

int main()
{
	int n,k;
	cin>>n>>k;
	vector<int> a(k);
	for(int i=0;i<n;i++)
	{
		int t;
		cin>>t;
		a[t-1]++;
	}
	int res=0;
	while(a[k-1]!=n)
	{
		for(int i=k-2;i>=0;i--)
			if(a[i]>0)
			{
				a[i]--;
				a[i+1]++;
			}
		res++;
	}
	cout<<res<<'\n';
	return 0;
}
