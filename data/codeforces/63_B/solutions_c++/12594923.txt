#include<bits/stdc++.h>
using namespace std;
int x[100],delay,maxm;
int main()
{
    int n,i,k; cin >>  n >> k;  for(i=0;i<n;i++) cin >> x[i];
    for(i=0;i<n;i++){
        if(x[i]==k)
            break ;
        maxm = max(i+x[0] - x[i],0);
        delay = max(delay,maxm);
    }
    cout << k - x[0] + delay << endl;
    return 0;
}
