#include<iostream>
#include<algorithm>
using namespace std;
int a[101];
int main()
{
	int n,k,x,h = 0;
	cin>>n>>k;
	for(int i = 0; i < n; i++)
	{
		cin>>x;
		a[x]++;
	}
	while(a[k] != n)
	{
		h++;
		for(int i = k - 1; i >= 1; i--)
			if (a[i] > 0)
			{
				a[i]--;
				a[i + 1]++;
			}
	}
	cout<<h;
	return 0;
}
