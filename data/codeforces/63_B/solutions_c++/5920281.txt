#include<iostream>
using namespace std;

const int oo = 110;
int n, k, a[oo], x, ans;

int main(){
  cin >> n >> k;
  for(int i = 0; i < n; i++){
    cin >> x;
    a[x]++;
  }
  while(a[k] != n){
    for(int i = k - 1; i >= 1; i--)
      if(a[i]){
    a[i]--;
    a[i + 1]++;
      }
    ans++;
  }
  cout << ans;
  return 0;
}
