#include <iostream>

using namespace std;

int N,K,A[110],result;

int main()
{
	cin >> N >> K;
	for (int i=0;i<N;i++)
		cin >> A[i];
	A[N]=K;
	int flag=1;
	while (flag)
	{
		flag=0;
		for (int i=0;i<N;i++)
			if (A[i]!=A[i+1])
			{
				A[i]++;
				flag=1;
			}
		result++;
	}
	cout << result-1 << endl;
	return 0;
}
