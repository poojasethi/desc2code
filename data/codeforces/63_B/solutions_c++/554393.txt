#include <stdio.h>
int i,ans,n,k,a[102];
int main()
{ scanf("%d%d",&n,&k);
  for (i=1;i<=n;i++) scanf("%d",&a[i]);
  ans=0;
  while (a[1]<k)
  { ans++;
    for (i=1;i<n;i++)
    {if (a[i]==k) continue;
     if (a[i]<a[i+1]) {a[i]++;}
    }
    if (a[n]<k) a[n]++;
  }
  printf("%d\n",ans);
  return 0;
}
