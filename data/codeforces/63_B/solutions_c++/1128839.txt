#include<iostream>
using namespace std;
int main()
{
    int n,a[102],k,l=0;
    cin>>n>>k;
    for(int i=1;i<=n;i++)
    cin>>a[i];
    a[n+1]=k;
    while(a[1]!=k)
    {
                  for(int i=2;i<=n+1;i++)
                  if(a[i]!=a[i-1])
                  a[i-1]++;
                  l++;
    }
    cout<<l;
}
