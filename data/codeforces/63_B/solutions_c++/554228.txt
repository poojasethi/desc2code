#include<stdio.h>
int a[103];
int main()
{
	int n,l;
	scanf("%d %d",&n,&l);
	int i;
	for(i=0;i<=n-1;i++)
	{
		scanf("%d",&a[i]);
	}
	a[n]=l;
	int ans=0;
	while(a[0]!=l)
	{
		for(i=0;i<=n-1;i++)
		{
			if(a[i]<a[i+1])
				a[i]++;
		}
		ans++;
	}
	printf("%d\n",ans);
	return 0;
}
