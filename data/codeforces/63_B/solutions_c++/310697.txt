#include<iostream>
#include<vector>
using namespace std;
int a[200];
int main()
{
    int q,n,k;
    cin>>n>>k;
    for(int i=0;i<n;i++){
    cin>>q;
    a[q-1]++;}
    q=0;
    while(a[k-1]!=n){
    for(int i=k-1;i>=0;i--){
    if(a[i-1]!=0){
    a[i-1]--;
    a[i]++;}}
    q++;}
    cout<<q;
    return 0;
}
