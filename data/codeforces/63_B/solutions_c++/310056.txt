#include <iostream>
#include <string>

using namespace std;

int main() {
	int n, k;
	int a[101] = { 0, };
	cin >> n >> k;
	a[n] = k;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	int step = 0;
	while (a[0] != k) {
		step++;
		for (int i = 0; i < n; i++)
			if (a[i] != a[i+1])
				a[i]++;
	}
	cout << step << endl;
	return 0;
}
