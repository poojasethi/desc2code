#include<iostream>
using namespace std;
int main()
{
    int i,n,k,cnt=0;
    cin>>n>>k;
    int a[101];
    for(i=0;i<n;i++)
    cin>>a[i];
    a[n]=k;
    while(a[0]!=k)
    {
        cnt++;
        for(i=0;i<n;i++)
        {
            if(a[i]!=a[i+1])
            a[i]=a[i]++;
        }
    }
    cout<<cnt;
    return 0;
}
