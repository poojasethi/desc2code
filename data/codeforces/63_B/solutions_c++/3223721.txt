#include<iostream>
using namespace std;
int main()
{
    int n, i, k, dem=0;
    cin>>n;
    int a[n+1];
    a[n]=k;
    cin>>k;
    for (i=0; i<n; i++)
        cin>>a[i];
    while (a[0]<k)
    {
        for (i=0; i<n; i++)
            if (a[i]<a[i+1])
                a[i]++;
        dem++;
    }
    cout<<dem;
    return 0;
}
