#include <iostream>
using namespace std;

int main() {
	int n, k;
	cin >> n >> k;

	int a[101];
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}

	int cnt = 0;
	while (a[0] != k) {
		for (int i = 0; i < n; i++) {
			if (i == n-1 || a[i] != a[i+1]) {
				a[i] = min(k, a[i] + 1);
			}
		}
		cnt++;
	}

	cout << cnt << endl;
	return 0;
}
