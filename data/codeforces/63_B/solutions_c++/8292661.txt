#include <iostream>

using namespace std;

int main() {
	int n, k, cnt = 0;
	cin >> n >> k;
	int a[n];
	for (int i = 0; i < n; i++) cin >> a[i];

	while (a[0] != k) {
		int i = 0;
		while (i < n) {
			while (a[i] == a[i+1] && i < n-1) i++;
			a[i]++;
			i++;
		}
		cnt++;	
	}	
	cout <<cnt;

	return 0;
}
