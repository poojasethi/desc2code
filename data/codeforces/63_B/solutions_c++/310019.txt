#include <iostream>

using namespace std;

int main()
{
    int a[100];
    int n,k;
    cin>>n>>k;
    for(int i=0;i<n;i++)
        cin>>a[i];
    int ans=0;
    while(a[0]!=k)
    {
        for(int i=0;i<n&&a[i]!=k;i++)
            if(i==n-1||a[i]!=a[i+1])
                a[i]++;
        ans++;
    }
    cout<<ans;
    return 0;
}
