#include<iostream>
#include<algorithm>
using namespace std;
int a[110];
int main()
{
    int n,k,ans=0;
    cin>>n>>k;
    for(int i=1;i<=n;i++)
    cin>>a[i];
    while(a[1]<k)
    {
    sort(a+1,a+n);
    for(int i=1;i<=n;i++)
    if(a[i]!=a[i+1]&&a[i]<k)
    a[i]++;
    ans++;
}
cout<<ans;
return 0;
}
