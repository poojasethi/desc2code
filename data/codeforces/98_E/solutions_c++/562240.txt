#include <stdio.h>
#include <algorithm>
using namespace std;
double dp[1010][1010];
double f(int m,int n)
{
    if (dp[m][n]>-1) return dp[m][n];
    double &ans=dp[m][n];
    if (m==0) return ans=1.0/(1.0+n);
    if (n==0) return ans=1.0;
    double a,b,c,d;
    b=n/(n+1.0)*(1-f(n-1,m));
    d=b+1.0/(n+1.0);
    a=1.0;
    c=1-f(n,m-1);
    double p=(d-b)/(a+d-c-b);
    ans=(1-p)*(1-p)*b+(1-p)*p*d+p*(1-p)*a+p*p*c;
    ans=max(ans,min(a,c));
    ans=max(ans,min(b,d));
    return ans;
}
int main()
{
    int n,m;
    while (scanf("%d%d",&m,&n)==2)
    {
        for (int i=0;i<=1000;i++)
            for (int j=0;j<=1000;j++)
                dp[i][j]=-10;
        double ans=f(m,n);
        printf("%.10f %.10f\n",ans,1-ans);
    }
    return 0;
}
