#include<cstdio>
#include<cstring>
double f[1010][1010];int n,m;
double ff(int n,int m)
{
    if(n==0)return 1.0/(1+m);
    if(m==0)return 1.0;
    if(f[n][m]>-1e-8)return f[n][m];
    double A=1.0*m/(m+1)*(1-ff(m-1,n)),C=1.0,B=1.0/(m+1)+1.0*m/(m+1)*(1-ff(m-1,n)),D=1-ff(m,n-1);
    double p=(D-C)/(A-B-C+D);
    return f[n][m]=A*p+1-p;
}
int main()
{
    memset(f,-1,sizeof(f));
    scanf("%d%d",&n,&m);
    printf("%.9lf %.9lf\n",ff(n,m),1-ff(n,m));
    return 0;
}
