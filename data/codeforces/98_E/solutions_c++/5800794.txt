#include <iostream>
#include <cmath>

using namespace std;

double memprob[2000][2000];
int calculado[2000][2000];

double prob(int n,int m)
{
  if (m==0) return 1;
  if (n==0) return 1.0/(m+1);
  double &mem=memprob[n][m];
  if (calculado[n][m]) return mem;
  calculado[n][m]=1;
  double A=double(m)/(m+1)*(1.0-prob(m-1,n));
  double B=1.0/(m+1);
  double C=1-prob(m,n-1);
  double p=(C-1)/(A-1-(B+A)+C);
  mem=(p*(A+B+A)+(1-p)*(1+C))/2.0;
  mem=max(mem,1.0/(m+1));
  return mem;
}

int main()
{
  cout.setf(ios::fixed);
  cout.precision(15);
  int n,m;
  cin>>n>>m;
  double p=prob(n,m);
  cout<<p<<" "<<1-p<<endl;
}
