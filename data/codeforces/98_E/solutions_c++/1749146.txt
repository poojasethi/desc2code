# include <cstdio>
# include <iostream>
using namespace std;
const int N=1000+10;

int n,m;
double f[N][N];
bool v[N][N];

void init()
{
    cin>>n>>m;
}

double dfs(int n,int m)
{
    if(v[n][m])return f[n][m];
    v[n][m]=1;
    if(n*m==0)
    {
        if(n>0)f[n][m]=1.0000;
        else f[n][m]=1.0000/(m+1);
        return f[n][m];
    }
    double A=1.0000;
    double B=1.0000-dfs(m,n-1);
    double C=double(m)/(m+1)*(1.0000-dfs(m-1,n));
    double D=double(m)/(m+1)*(1.0000-dfs(m-1,n))+1.0000/(m+1);
    //A * P + B * (1 - P) = C * P + D * (1 - P)
    //AP + BP - CP + DP = D - B
    double P=(D-B)/(A-B-C+D);
    return f[n][m]=P*A+(1-P)*B;
}

void solve()
{
    double ans=dfs(n,m);
    printf("%.10f %.10f\n",ans,1-ans);
}

int main()
{
    init();
    solve();
    return 0;
}
