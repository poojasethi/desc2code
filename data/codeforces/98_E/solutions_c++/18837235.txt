#include<cstdio>
#define maxn 1010
bool v[maxn][maxn];
double f[maxn][maxn];
double solve(int p,int q)
{
	if(v[p][q]) return f[p][q];
	v[p][q]=true;
	if(p==0) return f[p][q]=1.0/(q+1);
	if(q==0) return f[p][q]=1.0;
	double C=1.0/((q+1)*solve(q,p-1)+1);
	double A=1-C;
	return f[p][q]=A*q/(q+1)*(1-solve(q-1,p))+C;
}
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	printf("%.10f %.10f\n",solve(n,m),1-solve(n,m));
	return 0;
}
