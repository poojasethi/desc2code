using namespace std;

int get()
{
  char c;
  while(c=getchar(),(c<'0'||c>'9')&&(c!='-'));
  bool flag=(c=='-');
  if(flag)
    c=getchar();
  int x=0;
  while(c>='0'&&c<='9')
    {
      x=x*10+c-48;
      c=getchar();
    }
  return flag?-x:x;
}

void output(int x)
{
  if(x<0)
    {
      putchar('-');
      x=-x;
    }
  int len=0,data[10];
  while(x)
    {
      data[len++]=x%10;
      x/=10;
    }
  if(!len)
    data[len++]=0;
  while(len--)
    putchar(data[len]+48);
  putchar('\n');
}

double f[1001][1001];
bool flag[1001][1001];

double DP(int n,int m)
{
  if(!n||!m)
    return 1.0/(m+1);
  if(flag[n][m])
    return f[n][m];
  flag[n][m]=true;
  double a=1,b=1-DP(m,n-1),c=(double)m/(m+1)*(1-DP(m-1,n)),d=1-(double)m/(m+1)*DP(m-1,n);
  double p=(d-b)/(a-b-c+d);
  return f[n][m]=max(1.0/(m+1),p*(a-b)+b);
}

int main()
{
  int n=get(),m=get();
  memset(flag,0,sizeof(flag));
  printf("%0.10lf %0.10lf\n",DP(n,m),1-DP(n,m));
  return 0;
}
