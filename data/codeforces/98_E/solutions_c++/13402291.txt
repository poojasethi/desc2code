#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

double f[1005][1005];
bool ok[1005][1005];
int n,m;
double solve(int n,int m) {
	if (ok[n][m]) return f[n][m];
	ok[n][m]=true;
	if (!n) return f[n][m]=1.0/(m+1);
	if (!m) return f[n][m]=1;
	double a=1,b=1-solve(m,n-1),c=(double)m/(m+1)*(1-solve(m-1,n)),d=(double)m/(m+1)*(1-solve(m-1,n))+1.0/(m+1);
	double p=(d-b)/(a-b-c+d);
	return f[n][m]=a*p+b*(1-p);
}
int main() {
	scanf("%d%d",&n,&m);
	printf("%.9lf %.9lf\n",solve(n,m),1-solve(n,m));
	return 0;
}
