#define li		long long int

#define pb		push_back
#define sz(v)		((li)v.size())


#define MAX 1005
double dp[MAX][MAX];

double recur(li n,li m);

double cal(li n,li m,double p){
	double M=m,N=n;
	double opp_match=M/(1.0+M)*p*(1.0-recur(m-1,n));
	double opp_not=p/(1.0+M);
	return opp_match+(opp_not+(1.0-p)-max(opp_not,(1.0-p)*recur(m,n-1)));
}

double recur(li n,li m){
	double N=n,M=m;
	double &res=dp[n][m];
	if(-1<res) return res;
	if(n==0) return res=1.0/(1.0+M);
	if(m==0) return res=1.0;
	res=1.0/(1.0+M);
	res=max(res,cal(n,m,0.0));
	res=max(res,cal(n,m,1.0));
	res=max(res,cal(n,m,(1.0+M)*recur(m,n-1)/((1.0+M)*recur(m,n-1)+1.0)));
	return res;
}
	

int main(){
	rep(i,MAX)rep(j,MAX) dp[i][j]=-1000;
	li n,m;
	cin>>n>>m;
	printf("%.20lf %.20lf\n",recur(n,m),1.0-recur(n,m));
}
