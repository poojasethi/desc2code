#include <stdio.h>
#include <string.h>

int main()
{
    char s[1010];
    int num[256],f[1010];
    int len,i,j,k,t;
    
    gets(s);
    len=strlen(s);
    memset(num,0,sizeof(num));
    for(i=0;i<len;i++)
      num[s[i]]++;
      
    k=0;
    for(i=1;i<256;i++)
    if(num[k]<num[i]) k=i;
    
    memset(f,1,sizeof(f));
    for(i=2;i+i<=len;i++)
      if(f[i])
      for(j=i;j<=len;j+=i)
        f[j]=0;
    
    f[1]=1;
    for(i=2;i+i<=len;i++)
      f[i]=0;
      
    for(i=1;i<=len;i++)
      if(!f[i])
      {
        if(num[k]==0)
        {
            puts("NO");
            return 0;
          }
          num[s[i-1]=k]--;
      }
      
    j=0;
    for(i=1;i<=len;i++)
      if(f[i])
      {
        while(num[j]==0) j++;
        num[s[i-1]=j]--;
      }
      
      puts("YES");
      puts(s);
      return 0;
}
