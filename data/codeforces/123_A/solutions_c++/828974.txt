#include <iostream>
#include <algorithm>

using namespace std;

char s[1024];
int M[1024];
int R[256];
int main(void)
{
	int n,k=1;
	char *t;
	cin>>s;
	fill(R,R+256,0);
	for (t=s;*t;t++) 
		R[*t]++;
	n=t-s;
	fill(M,M+n+1,0);
	for (int i=2;i<=n;i++) {
		if (M[i]==0) {
			if (2*i>n)
				k++;
			else
				for (int j=i;j<=n;j+=i)
					M[j]=1;
		}
	}
	int m=0;
	for (int i=0;i<256;i++)
		if (R[m]<R[i])
			m=i;
	if (R[m]<n-k)
		cout<<"NO";
	else {
		cout<<"YES"<<endl;
		R[m]-=n-k;
		for (int i=1,j=0;i<=n;i++) {
			if (M[i])
				cout<<(char)m;
			else {
				for (;R[j]==0;j++) {}
				cout<<(char)j;
				R[j]--;
			}
		}
	}
	return 0;
}
