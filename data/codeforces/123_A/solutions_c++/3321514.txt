#include <cstdio>
#include <cstring>

char s[1005];
int f[200];
bool vis[1005];
char ans[1005];

int main()
{
    scanf("%s",s+1);
    int l=strlen(s+1);
    memset(f,0,sizeof(f));
    for(int i=1;i<=l;i++)
        f[s[i]]++;
    int x=0;
    for(int i=1;i<200;i++)
        if(f[i]>f[x])x=i;
    memset(vis,0,sizeof(vis));
    for(int i=2;i<=l/2;i++)
    {
        if(vis[i])continue;
        for(int j=i;j<=l;j+=i)
            vis[j]=1;
    }
    int ss=0;
    for(int i=1;i<=l;i++)
        if(vis[i])ss++;
    if(ss>f[x])puts("NO");
    else
    {
        puts("YES");
        for(int i=1;i<=l;i++)
            if(vis[i])ans[i]=x,f[x]--;
        for(int i=1;i<=l;i++)
        {
            if(vis[i])continue;
            for(int j=0;j<200;j++)
                if(f[j]){ans[i]=j,f[j]--;break;}
        }
        puts(ans+1);
    }
    return 0;
}
