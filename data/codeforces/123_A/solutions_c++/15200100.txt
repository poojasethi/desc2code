#include<bits/stdc++.h>
#define N 1005
using namespace std;
char a[N];int f[N],flag[N],prime[N],g[N];
int n,num,ok[N],k,i,id,ans,cnt,j;
int main(){
    scanf("%s",a+1);
    n=strlen(a+1);num=1;ok[1]=1;
    for (i=1;a[i];i++)
        g[a[i]-'a']++;
    for (i=2;i<=n;i++){
        if (!flag[i])
            if ((prime[++cnt]=i)>n/2) ok[i]=1,num++;
        for (j=1;j<=cnt&&prime[j]*i<=n;j++)
            flag[prime[j]*i]=1;
  }for (i=0;i<26;i++)
    if (ans<g[i]) ans=g[i],id=i;
  num=n-num;
  if (ans<num) return puts("NO"),0;
  puts("YES");
  for (i=1;i<=n;i++)
    if (!ok[i]) putchar('a'+id);
    else{
        for (k=0;k<26;k++)
            if (k!=id&&g[k]){
                putchar('a'+k);g[k]--;
                break;
          }if (k==26) putchar('a'+id);
    }
}
