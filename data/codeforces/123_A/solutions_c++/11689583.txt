#include<iostream>
#include<queue>
#include<algorithm>
#include<string>
using namespace std;
bool mark[1001];
int counter[26];
int main()
{
	string str;
	cin>>str;
	int n = str.size();
	for (int i=0;i<n;i++)
		counter[str[i]-'a']++;
	int cnt = 0;
	for (int i=2;i<=n;i++)
	{
		if (mark[i])
			continue;
		if (n / i > 1)
		{
			for (int j=i;j<=n;j+=i)
			{
				if (!mark[j])
				{
					mark[j] = 1;
					cnt++;
				}
			}
		}
	}
	int *mx = max_element(counter,counter+26);
	if (*mx < cnt)
		cout<<"NO"<<endl;
	else
	{
		
		cout<<"YES"<<endl;
		
		for (int i=1;i<=n;i++)
		{
			if (mark[i])
			{
				cout<<char('a' + (mx-counter));
				counter[mx-counter]--;
				cnt--;
			}
			else
				for (int j=0;j<26;j++)
				{
					if (counter[j] == 0)
						continue;
					if (cnt == counter[j] && j == mx - counter)
						continue;
					
					counter[j]--;
					cout<<char('a' + j);
					break;
				}
		}
		cout<<endl;
	}
	return 0;
}
