#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char s[1005];
int w[26];
int d[1005];

int main(void) {
	scanf("%s", s);
	for(int i=0;s[i];i++) w[s[i]-'a']++;
	int n = strlen(s), c=n-1;
	d[0] = 1;
	for(int i=n/2+1;i<=n;i++) {
		for(int j=2;j<=i;j++)
			if(i%j==0) {
				if(i==j) {d[i-1]=1; --c;}
				break;
			}
	}
	if(*max_element(w, w+26) >= c) {
		puts("YES");
		char v = max_element(w,w+26)-w +'a';
		for(int i=0;i<n;i++)
			if(d[i]==0) s[i]=v, w[v-'a']--;
		for(int i=0;i<n;i++)
			if(d[i]==1) {
				int j;
				for(j=0;j<26;j++) if(w[j]) break;
				s[i] = j+'a';
				w[j]--;
			}
		puts(s);
	} else {
		puts("NO");
	}
	return 0;
}
