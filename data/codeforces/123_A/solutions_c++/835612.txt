#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char str[2000],res[2000];
int len;
int num[26];
bool isprime(int x){
	int i;
	for(i=2;i*i<=x;i++)if(x%i==0)return 0;
	return 1;
}
void run(){
	memset(num,0,sizeof(num));
	memset(res,0,sizeof(res));
	int i;
	for(i=0;i<len;i++)num[str[i]-'a']++;
	int mi=-1;
	for(i=0;i<26;i++)if(mi==-1||num[i]>num[mi])mi=i;
	int cnt=len-1;
	for(i=len/2+1;i<=len;i++){
		if(isprime(i))cnt--;
		else res[i-1]=mi+'a';
	}
	if(num[mi]<cnt){
		printf("NO\n");
		return;
	}
	for(i=2;i<=len/2;i++)res[i-1]=mi+'a';
	num[mi]-=cnt;
	int j=0;
	for(i=0;i<len;i++){
		if(res[i])continue;
		while(!num[j])j++;
		num[j]--;
		res[i]=j+'a';
	}
	printf("YES\n%s\n",res);

}
int main(){
	scanf("%s",str);
	len=strlen(str);
	run();

	return 0;
}
