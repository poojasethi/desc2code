typedef long long i64;

using namespace std;

char s[1010];
int a[26];
int b[1010];
int main() {
#ifdef pperm

#endif
	scanf("%s", &s);
	for (int i = 0; s[i]; i++) {
		a[s[i] - 'a']++;
	}
	int n = strlen(s);
	for (int i = 2; i + i <= n; i++) {
		for (int j = i; j <= n; j += i) {
			b[j] = 1;
		}
	}
	int k = 0;
	for (int i = 1; i <= n; i++) {
		if (b[i]) k++;
	}
	int c = 0;
	for (int i = 1; i < 26; i++) {
		if (a[i] > a[c]) {
			c = i;
		}
	}
	if (a[c] < k) {
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	a[c] -= k;
	for (int i = 1, j = 0; i <= n; i++) {
		if (b[i]) {
			printf("%c", 'a' + c);
		} else {
			while (!a[j]) j++;
			a[j]--;
			printf("%c", 'a' + j);
		}
	}
	return 0;
}
