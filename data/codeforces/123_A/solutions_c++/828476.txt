#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<list>
using namespace std;

int n;
int all;
bool b[1005];
char s[1005];

bool check(int x)
{
	for(int i=2;i*i<=x;i++)
		if(x%i==0)
			return false;
	return true;
}

int main()
{
	scanf("%s",s);
	n=strlen(s);
	b[1]=true;
	for(int i=n;i*2>n;i--)
	{
		if(check(i))
		{
			all++;
			b[i]=true;
		}
	}
	int t=n-all-1;
	int now=1;
	int ch=0;
	sort(s,s+n);
	for(int i=1;i<=n;i++)
	{
		if(s[i]==s[i-1])
			now++;
		else
		{
			if(now>=t)
			{
				ch=s[i-1];
				break;
			}
			else
				now=1;
		}
	}
	if(!ch)
		puts("NO");
	else
	{
		puts("YES");
		list<char> T;
		for(int i=0;i<n;i++)
			if(s[i]==ch)
				T.push_front(s[i]);
			else
				T.push_back(s[i]);
		for(int i=1;i<=n;i++)
			if(!b[i])
			{
				putchar(T.front());
				T.pop_front();
			}
			else
			{
				putchar(T.back());
				T.pop_back();
			}
		puts("");
	}
	return 0;
}
