#include <stdio.h>
#include <cstdlib>
#include <math.h>
#include <algorithm>
#include <string.h>
#include <iostream>
using namespace std;

string s,a;
int d[2000],cnt;
bool f[2000];
char r;

bool prime(int x)
{
 for ( int i=2;i<x;i++)
  if ( x%i==0 ) return false;
 return true;
}

main()
{
 cin>>s;
 for ( int i=0;i<s.size();i++ )
  {
   d[s[i]]++;
   if ( d[s[i]]>d[r] ) r=s[i];
  }
  
 for ( int k=1;k<=s.size();k++ )
  if ( ( prime(k) && k*2>s.size() ) || k==1 ) f[k]=true; 
  else cnt++;

 if ( cnt>d[r] ) cout<<"NO"<<endl;
 else
  {
   cout<<"YES"<<endl;
   a=s;
   for ( int k=0;k<s.size();k++ )
    if ( f[k+1]==false ) { a[k]=r; d[r]--; };
   char c='a';
   for ( int k=0;k<s.size();k++ )  
    if ( f[k+1]==true )
    {
     while ( d[c]==0 ) c=c+1;
     a[k]=c;
     d[c]--;
    };
   cout<<a<<endl;
  };
}
