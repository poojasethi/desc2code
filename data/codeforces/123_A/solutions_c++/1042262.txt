#include<iostream>
using namespace std; 

const int Maxn = 100 ; 

string s , out;
int n;
int c[Maxn];
char mmax;

bool not_prime(int num){
	for(int i=2;i<=n;i++)
		if((num%i)==0)
			return ( num-i>0 || num+i<=n);
}

void input(){
	cin>>s;
	n=s.length();
}	

bool calc(){
	for(int i=0;i<n;i++){
		c[s[i]-'a']++;
		if(c[s[i]-'a']>c[mmax])
			mmax=s[i]-'a';
	}
	out=s;
	for(int i=1;i<=n;i++)
		if(not_prime(i)){
			out[i-1]=mmax+'a';
			c[mmax]--;
			if(c[mmax]<0)
				return false ;
		}
	string temp="";
	for(int i=0;i<Maxn;i++)
		for(int j=0;j<c[i];j++)
			temp+=i+'a';
	int j=0;
	for(int i=1;i<=n;i++)
		if(!not_prime(i)){
			out[i-1]=temp[j];
			j++;
		}
	return true;
}

void output(){
	if(calc())
		cout<<"YES"<<endl<<out<<endl;
	else
		cout<<"NO"<<endl;
}

int main(){
	input();
	output();
	return 0;
}
