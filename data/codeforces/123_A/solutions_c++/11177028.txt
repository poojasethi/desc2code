{int j;
 
 for(j=2;j<i;j++)

 return 1;
}

 }
 
 s=l/2-1;
 for(i=l/2+1;i<=l;i++)
 	if(!check(i))
 		s++;
 
 //cout<<max1<<" "<<s<<endl;

 else
 {	printf("YES\n");

 		if(!check(i))

 			c[max1]--;
 		}
 	j=1;

		if(ans[i]==0)

			c[j]--;
		}
	
	for(i=1;i<=l;i++)
		printf("%c",(char)(ans[i]+96));

 }
