#include<bits/stdc++.h>
using namespace std;
const int N=1005;
char s[N],v;int d[N],hsh[26],n,i,j,cnt=1;
int main()
{
	ios::sync_with_stdio(false);cin.tie(0);
	cin>>s;n=strlen(s);d[0]=1;
	for(i=0;i<n;i++)hsh[s[i]-'a']++;
	for(i=n/2+1;i<=n;i++)
		for(j=2;j<=i;j++)
			if(i%j==0)
			{
				if(i==j)d[i-1]=1,cnt++;
        break;
			}
    if(n-cnt<=*max_element(hsh,hsh+26))
    {
          cout<<"YES\n";
          v=max_element(hsh,hsh+26)-hsh+'a';
          for(i=0;i<n;i++)
          	if(d[i]==0)
          		s[i]=v,hsh[v-'a']--;
          for(i=0;i<n;i++)
             if(d[i])
             {
             	for(j=0;j<26;j++)
             		if(hsh[j])break;
             	s[i]='a'+j;hsh[j]--;
             }
         cout<<s;
    }
    else
    	cout<<"NO\n";
	return 0;
}
