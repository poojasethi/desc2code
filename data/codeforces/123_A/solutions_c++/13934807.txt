#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
char a[10010];bool mark[10010];
int Ans[10010],C[26],num,P[10010],len,N;
inline void biao()
{
	memset(mark,1,sizeof(mark));
	for (int i=2;i<=1000;i++)
	if (mark[i])
	{
		P[++num]=i;
		for (int j=i+i;j<=1000;j+=i) mark[j]=0;
	}
}
int main()
{
	biao(); int N=1;
	scanf("%s",&a); int len=strlen(a);
	for (int i=0;i<len;i++) C[a[i]-'a']++;
	
	for (int i=1;i<=num;i++) if (P[i]<=len&&P[i]*2>len) N++;
	N=len-N; bool pd=false; int A=0;
	
	for (int i=0;i<26;i++) if (C[i]>=N) {pd=1; A=i; break;}
	if (!pd) {printf("NO\n"); return(0);} memset(Ans,-1,sizeof(Ans));
	
	for (int i=2;i<=len;i++) 
	if ((mark[i]&&i<=len&&i*2<=len)||(!mark[i])) Ans[i]=A,C[A]--;
	for (int i=1;i<=len;i++) 
	if (Ans[i]==-1)	{int j=0; while (C[j]==0) j++; C[j]--; Ans[i]=j;}
	
	printf("YES\n");
	for (int i=1;i<=len;i++) printf("%c",Ans[i]+'a');
}
