#include<iostream>
#include<string>
using namespace std;
int B[26],ans;
bool K[1010];
string A;
int main()
{
	//freopen("input.txt","rb",stdin);
	//freopen("output.txt","wb",stdout);
	cin>>A;
	int n=A.length();
	for(int i=0;i<n;++i)B[A[i]-'a']++;
	for(int i=2;i<=n;++i)
	{
		int j;
		for(j=2;i%j!=0&&i*j<=n;++j)K[i*j]=1;
		if(i*j<=n)K[i*j]=1;
	}
	ans=n-1;
	for(int i=n/2+1;i<=n;++i)if(!K[i])ans--;
	for(int i=2;i<=n/2;++i)K[i]=1;
	for(int i=0,j,k=0;i<26;++i)if(B[i]>=ans)
	{
		cout<<"YES"<<endl;
		for(j=1;j<=n;++j)if(K[j])B[i]--,A[j-1]=i+'a';
		for(j=1;j<=n;++j)if(!K[j])
		{
			for(;!B[k];k++);
			B[k]--;
			A[j-1]=k+'a';
		}
		cout<<A<<endl;
		return 0;
	}
	cout<<"NO"<<endl;
	return 0;
}
