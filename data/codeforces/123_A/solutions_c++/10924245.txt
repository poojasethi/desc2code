#include<bits/stdc++.h>
using namespace std;
char str[1010], ans[1010];
int M[255], ufs[1010], sz[1010], vis[1010];
int find(int x){
	return ufs[x] == x ? x : ufs[x] = find(ufs[x]);
}
int main(){
	cin >> str;
	int len = strlen(str);
	for(int i = 0; i < len; ++i)
		++M[str[i]-'a'], ufs[i+1] = i+1, sz[i+1] = 1;
	for(int i = 2; i <= len; ++i)
		for(int j = i*2; !vis[i] && j <= len; j += i){
			vis[j] = 1;
			if(find(i) != find(j)) sz[find(j)] += sz[find(i)], ufs[find(i)] = ufs[find(j)];
		}
	int Maxp = max_element(M, M+26) - M;
	int Max = max_element(sz+1, sz+len+1) - sz;
	if(sz[Max] <= M[Maxp]) puts("YES"); else puts("NO"), exit(0);
	for(int i = 0; i < len; ++i)
		if(find(i+1) == Max) ans[i] = 'a' + Maxp, M[Maxp]--;
	for(int i = 0, j = 0; i < len; ++j)
		for(; M[j] && i < len; ++i)
			if(!ans[i]) ans[i] = 'a' + j, M[j]--;
	cout << ans;
}
