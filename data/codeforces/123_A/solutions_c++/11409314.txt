#include<bits/stdc++.h>
using namespace std;
bool used[1100];
int main()
{
	int i,j,n=1,maxi=0,k;
	string s;
	cin>>s;
	int hash[1126]={0};
	int has[26]={0};
	for(i=0;i<s.length();i++)
		has[s[i]-'a']++;
	for(i=0;i<26;i++)
		if(maxi<has[i])
			maxi=has[i],k=i;
	memset(used,false,sizeof(false));	
	for(i=2;i<=s.length();i++)
	{
		if(!hash[i])
		{
			if(2*i>s.length())
				n++;
			else
				hash[i]=1;	
			for(j=i*i;j<=s.length();j+=i)
				hash[j]=1;
		}	
	}
	if(maxi>=(s.length()-n))
	{
		cout<<"YES\n";
		for(i=1;i<=s.length();i++)
		{
			if(hash[i]==1)
			{
				s[i-1]=k+'a';
				has[k]--;
				used[i-1]=true;
			}	
			else
			{
				for(j=0;j<26;j++)
				{
					if(j!=k&&has[j]>0)
					{
						s[i-1]=j+'a';
						has[j]--;
						used[i-1]=true;
						break;
					}	
				}
			}	
		}
		for(i=1;i<=s.length();i++)
			if(used[i-1]==false)
				s[i-1]=k+'a';
		cout<<s<<endl;		
	}	
	else
		cout<<"NO\n";			
	return 0;
}
