#include <cstdio>
#include <cstring>

int n;
int c[26];
char s[1001];

bool isp(int n) {
	for (int i = 2; i * i <= n; ++i) {
		if (n % i == 0) {
			return false;
		}
	}
	return true;
}

int main() {
	scanf("%s", s);
	n = strlen(s);
	int maxC = 0;
	for (int i = 0; i < n; ++i) {
		int ch = s[i] - 'a';
		++c[ch];
		if (c[ch] > c[maxC]) {
			maxC = ch;
		}
	}
	for (int i = 1; i <= n; ++i) {
		if (!(i == 1 || (i > n / 2 && isp(i)))) {
			s[i - 1] = maxC + 'a';
			--c[maxC];
		}
	}
	if (c[maxC] < 0) {
		puts("NO");
	} else {
		puts("YES");
		for (int i = 1, j = 0; i <= n; ++i) {
			for (; c[j] == 0; ++j);
			if (i == 1 || (i > n / 2 && isp(i))) {
				s[i - 1] = j + 'a';
				--c[j];
			}
		}
		puts(s);
	}
	return 0;
}
