// div 1
#include <iostream>
#include <string>
#include <string.h>
using namespace std;

string st;
int u[3000], f[100];

int main()
{
    int i, n, s, j;
    cin >> st;
    n=st.size();
    for (i=0; i<n; i++) f[(int)st[i]-(int)'a'+1]++;
    memset(u, 0, sizeof(u));
    for (i=2; i<=2000; i++) if (!u[i])
        for (j=i*i; j<=2000; j+=i) u[j]=1;      
    u[2]=1;
    for (i=n; i>2; i--)
        if (!u[i])
        {
            if (i*2<=n) u[i]=1;
        }
    s=0;
    for (i=1; i<=n; i++) if (u[i]) s++;
    for (int t=1; t<=26; t++) if (f[t]>=s)
    {
        f[t]-=s; j=1;
        cout << "YES" << endl;
        for (i=1; i<=n; i++) if (u[i]) cout << (char)((int)'a'-1+t);
        else
        {
            while (f[j]==0) j++;
            cout << (char)((int)'a'-1+j);
            f[j]--;
        }
        cout << endl;
        return 0;
    }
    cout << "NO" << endl;
    return 0;
}
