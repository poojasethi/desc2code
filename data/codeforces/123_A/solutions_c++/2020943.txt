#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int dr[]={-1,0,1,0},dc[]={0,1,0,-1};
int main(void) {
	char s[1024];
	bool prm[1024];
	memset(prm,true,sizeof(prm));
	for(int i=2;i<1024;++i)if(prm[i])for(int j=i*i;j<1024;j+=i)prm[j]=false;
	for(int ks=1; 1==scanf("%s",s);++ks){
		int n=strlen(s),c[128]={0},p=0;
		char t[1024]={0};
		for(int i=2;i<=(n>>1);++i)if(prm[i])for(int j=i;j<=n;j+=i)t[j]=1;
		for(int i=0;i<n;++i)++c[s[i]];
		for(int i=1;i<=n;++i)if(t[i])++p;
		bool ok=false;
		for(int i='a';!ok&&i<='z';++i)if(c[i]>=p){
			c[i]-=p;
			for(int j='a',u=1;j<='z';++j)for(int k=c[j];k>0;++u)if(!t[u])t[u]=j,--k;
			for(int u=1;u<=n;++u)if(1==t[u])t[u]=i;
			t[n+1]=0,puts("YES"),puts(t+1),ok=true;
		}
		if(!ok)puts("NO");
	}
	return 0;
}
