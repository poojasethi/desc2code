#include<cstdio>
#include<cstring>
using namespace std;

char s[2333],ans[2333];
bool f[2333];
int cnt[27];
int main(){
	scanf("%s",s);
	int n=strlen(s);
	for(int i=0;i<n;i++)cnt[s[i]-'a']++;
	memset(f,1,sizeof(f));
	for(int i=2;i<=n;i++)if(f[i]){
		for(int j=i+i;j<=n;j+=i)f[j]=0;
	}
	for(int i=0;(i<<1)<=n;i++)f[i]=0;f[1]=1;
	int mx=0;
	for(int i=1;i<27;i++)if(cnt[i]>cnt[mx])mx=i;
	for(int i=0;i<n;i++)if(!f[i+1]){
		cnt[mx]--;
		if(cnt[mx]<0){
			printf("NO\n");
			return 0;
		}
		ans[i]=mx+'a';
	}
	int now=0;
	for(int i=0;i<n;i++)if(f[i+1]){
		while(!cnt[now])now++;
		ans[i]=now+'a';
		cnt[now]--;
	}
	ans[n]='\0';
	printf("YES\n");
	printf("%s\n",ans);
	return 0;
}
