#include<iostream>
#include<string>
#include<algorithm>
using namespace std;

bool isp(int a,int i) {return a<=i || a%i!=0 && isp(a,i+1);}
int c[256];
int main() {
	string s;
	cin>>s;
	int N=s.size();
	for(int i=0;i<N;i++) {
		c[s[i]]++;
	}
	string t(N,0);
	int *m=max_element(c,c+256);
	for(int i=1;i<N;i++) {
		if(i<N/2 || !isp(i+1,2)) {
			t[i]=m-c;
			(*m)--;
		}
	}
	if(*m<0) {
		cout<<"NO";
		return 0;
	}
	int p=0;
	for(int i=0;i<N;i++) {
		if(!t[i]) {
			while(!c[p]) p++;
			c[t[i]=p]--;
		}
	}
	cout<<"YES\n"<<t;
}
