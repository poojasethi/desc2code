#include<iostream>
#include<stdio.h>
#include<cstring>
using namespace std;

int main()
{
	int len,m=0,need=0,equal[1001]={0},times[30]={0};
	char s[1001];
	bool check=true,prime[1001]={false};
	scanf("%s",s+1);
	len=strlen(s+1);
	prime[2]=true;
	for (int i=3;i<1000;i++)
	{
		prime[i]=true;
		for (int j=2;j<i;j++)
			if (i%j==0)
			{
				prime[i]=false;
				break;
			}
	}
	for (int i=1;i<=len;i++)
		times[s[i]-'a']++;
	for (int i=0;i<26;i++)
		m=times[m]<times[i]?i:m;
	for (int i=2;i<len;i++)
		if (prime[i])
			if (len/i>1)
				for (int j=1;j<=len/i;j++)
					equal[i*j]=1;
	for (int i=1;i<=len;i++)
		if (equal[i]==1)
			need++;
	if (need>times[m])
		cout<<"NO";
	else
	{
		for (int i=1;i<=len;i++)
			if (equal[i]==1)
			{
				s[i]=m+'a';
				need--;
				times[m]--;
			}
			else
			{
				for (int j=0;j<26;j++)
					if (times[j]>0&&(need<times[j]||j!=m))
					{
						s[i]=j+'a';
						times[j]--;
						break;
					}
			}
		cout<<"YES"<<endl;
		cout<<s+1;
	}
	return 0;
}
