from math import acos,asin,atan,sqrt,pi,sin,cos,tan,ceil,floor
import re
from operator import itemgetter as ig

dic={'[':1,'<':2,'{':3,'(':4,']':-1,'>':-2,'}':-3,')':-4}
s=raw_input()
st=[0]*len(s)
le=0
c=0
for i in range(len(s)):
	if le==0:
		st[le]=dic[s[i]]
		le+=1
	else:
		if st[le-1]>0 and dic[s[i]]<0:
			if st[le-1]+dic[s[i]]!=0:
				c+=1
			le-=1
		else:
			st[le]=dic[s[i]]
			le+=1
if le!=0:
	print "Impossible"
else:
	print c
exit(0)
