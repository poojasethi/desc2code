s = raw_input()
o = "[<{("
c = "]>})"


stack = []
ans = 0


def match(l, r):
    if (l + r == "[]") or (l + r == "<>") or (l + r == "{}") or (l + r == "()"):
        return True
    return False


for i in s:
    if not len(stack):
        stack.append(i)
    elif (i in c) and (stack[-1] in o):
        x = stack.pop()
        if not match(x, i):
            ans += 1
    else:
        stack.append(i)

if len(stack):
    print "Impossible"
else:
    print ans
