brackets = {'<': '>', '{': '}', '[': ']', '(': ')'}

seq = raw_input()
stack, replaces, invalid = list(), 0, False
for c in seq:
	if c in brackets: 
		stack.append(brackets[c])
	elif not stack:
		invalid = True
		break
	elif stack.pop() != c: 
		replaces += 1

print "Impossible" if invalid or stack else replaces
