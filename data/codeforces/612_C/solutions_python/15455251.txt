o = ['(', '[', '<', '{']
cl = [')', ']', '>', '}']

def f():
  seq = raw_input()
  v = []
  count = 0

  for c in seq:
    if c in o:
      v.append(c)
    elif c in cl:
      if len(v) is 0:
        print "Impossible"
        return
      x = v.pop()
      m = cl[o.index(x)]
      if c is not m:
        count += 1
  
  if len(v) is 0:
    print count
  else:
    print "Impossible"
  return

f()
