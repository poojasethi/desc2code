s = raw_input()
change,balanced , i , l  = 0,True, 0, []

while i < len(s) and balanced:
	if s[i] in '([{<':
		l.append(s[i])
	else:
		if l == []:
			balanced= False
		else:
			if '([{<'.index(l[-1]) != ')]}>'.index(s[i]):
				change += 1
			l.pop()
	i += 1
if balanced and l ==[]:
	print change
else:
	print 'Impossible'
