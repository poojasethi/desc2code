s = raw_input()
arl = ["(","<","[","{"]
arr = [")",">","]","}"]
st = []
f = 0
cnt = 0
for i in s:
    if i in arl:
        st.append(i)
    elif i in arr:
        if st == []:
            f = 1
            break
        else:
            tmp = st[-1]
            st.pop()
            if arr.index(i) != arl.index(tmp):
                cnt += 1
if f==1 or len(st) != 0:
    print "Impossible"
else:
    print cnt
