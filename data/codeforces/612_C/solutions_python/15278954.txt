s=raw_input()

expected=['.']*len(s)

res=0
cur=0

closed={'{':'}', '(':')', '<':'>','[':']'}
for ch in s:
    if ch in '{([<':
        expected[cur]=closed[ch]
        cur+=1
    else:
        cur-=1
        if cur<0:
            break
        if expected[cur]!=ch:
            res+=1

print 'Impossible' if cur!=0 else res
