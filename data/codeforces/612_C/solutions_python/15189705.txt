s = raw_input()
dic = { '}' : '{', ']' : '[', ')' : '(', '>' : '<' }

temp = []
ans = True
cnt = 0
for i in range(len(s)):
	if s[i] in '{<[(' :
		temp.append(s[i])
	else:
		if len(temp) == 0:
			ans = 0
			break
		else:
			if temp.pop() != dic[s[i]]:
				cnt += 1
if len(temp) == 0 and ans:
	print cnt
else:
	print 'Impossible'
