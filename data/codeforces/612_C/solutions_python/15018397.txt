s = str(raw_input())
size = len(s)
stack = list()

res = 0
checker = True

closing = ']})>'
opening = '[{(<'

for char in s:
    if char in closing:
        if len(stack) == 0:
            checker = False
            break
        if opening.index(stack[-1]) != closing.index(char):
            res += 1
        stack.pop()
    else:
        stack.append(char)

if len(stack) != 0:
    checker = False

if checker is False:
    print 'Impossible'
else:
    print res
