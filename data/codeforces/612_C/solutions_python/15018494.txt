#!/usr/bin/env python
def solve():
    s = raw_input()

    PAIRS = {
        '<': '>',
        '{': '}',
        '[': ']',
        '(': ')',
    }

    opened_brackets = []
    changes = 0
    for i, bracket in enumerate(s):
        if bracket in PAIRS:
            opened_brackets.append(i)
        else:
            if opened_brackets:
                open_bracket = s[opened_brackets.pop()]
                if PAIRS[open_bracket] != bracket:
                    changes += 1
            else:
                print 'Impossible'
                return
    if opened_brackets:
        print 'Impossible'
    else:
        print changes

solve()
