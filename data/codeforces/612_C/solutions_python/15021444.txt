a = raw_input()
a = list(a)

stack = []
rep = 0
braces = {}
braces['<'] = '>'
braces['('] = ')'
braces['['] = ']'
braces['{'] = '}'

openb = braces.keys()
closeb = braces.values()
for i in a:
	if i in openb:
		stack.append(i)
	elif len(stack) == 0 and i in closeb:
		print "Impossible"
		exit()
	elif i in closeb:
		if braces[stack[len(stack) - 1]] != i:
			rep += 1
		stack.pop()
if len(stack) == 0:
	print rep
else:
	print 'Impossible'
