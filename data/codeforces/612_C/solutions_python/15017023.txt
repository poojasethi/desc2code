def is_left(a):
	return a in ('{', '[', '(', '<')
def is_right(a):
	return a in ('}', ']', ')', '>')
def one_type(a, b):
	return a == '{' and b == '}' or a == '[' and b == ']' or a == '<' and b == '>' or a == '(' and b == ')'

s = raw_input()
st = []
q = 0
for i in s:
	if is_left(i):
		st.append(i)
	if is_right(i):
		if len(st) == 0:
			st.append(1)
			break
		cur = st.pop()
		if not one_type(cur, i):
			q += 1
print q if len(st) == 0 else 'Impossible'
