#include<bits/stdc++.h>
using namespace std;

int main()
{
	string st;
	cin>>st;
	stack <char> stak;
	int ans=0;
	for(int i=0;i<st.size();i++)
		{
			
			if(st[i]=='{'||st[i]=='['||st[i]=='('||st[i]=='<')
				stak.push(st[i]);
			else {
				if(!stak.size())
					return cout<<"Impossible" , 0;
				if(st[i]-stak.top()!=2 && st[i]-stak.top()!=1)
					ans++;
				stak.pop();
			}
				
		}
		if(stak.size())cout<<"Impossible"<<endl;
		else cout<<ans<<endl;
		
	return 0;
}
