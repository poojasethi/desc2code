#include <iostream>
#include <stack>

using namespace std;

int main()
{
	string a;
	cin >> a;
	int ans = 0;
	stack <char> x;
	for (int i = 0; i < a.size(); i++)
	{
		if (a[i] == '<' || a[i] == '{' || a[i] == '[' || a[i] == '(')
			x.push(a[i]);
		else
		{
			if (x.empty())
			{
				cout << "Impossible\n";
				return 0;
			}
			char t = x.top();
			if (!((t == '<' && a[i] == '>') || (t == '{' && a[i] == '}') || (t == '(' && a[i] == ')') || (t == '[' && a[i] == ']')))
				ans++;
			x.pop();
		}
	}
	if (x.empty())
		cout << ans << endl;
	else
		cout << "Impossible\n";
	return 0;
}
