#include<iostream>
#include<stack>
#include<cstring>
#include<string>
using namespace std;
int main()
{
	stack<char>p;
	string s;
	cin >> s;
	int count=0;
	for(int i=0;i<s.size();i++){
		if(s[i]=='('||s[i]=='['||s[i]=='<'||s[i]=='{')
			p.push(s[i]);
		else{
			if(p.size()==0||p.top()==')'||p.top()=='}'||p.top()=='>'||p.top()==']')	p.push(s[i]);
			else{
			char c=p.top();
			if(c==(s[i]-1)||c==(s[i]-2))	p.pop();
			else
				count++,p.pop();
			}
		}
	}
	if(p.size()==0)
	cout << count << endl;
	else	cout << "Impossible\n";
}
