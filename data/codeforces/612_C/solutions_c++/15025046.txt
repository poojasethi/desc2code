#include <iostream>
#include <stack>
#define MAX 200001
using namespace std;
string s;
int ans;
stack<char> p;

int get(char c)
{
	if (c == '(' || c == '{' || c == '[' || c == '<') return '(';
	return ')';
}

int main()
{		
	cin >> s;
	for(int i = 0; i < s.size(); i++)
	{
		char c = s[i];
		char t = get(c);
		if(t == '(') p.push(c);
		else
		{
			if(p.size())
			{
				t = p.top();
				if(get(t) == '(')
				{
					if(c - t != 2 && c - t != 1) ans++;
					p.pop();
				}
				else p.push(c);
			}
			else p.push(c);
		}
	}

	if(p.empty()) cout << ans;
	else cout << "Impossible";
}
