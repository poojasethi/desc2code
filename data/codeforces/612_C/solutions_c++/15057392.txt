#include <iostream>
#include <cstdio>
#include <stack>
using namespace std;

int s;
string a;
stack<char> st;
int main(){
    int i;
    cin>>a;
    for(i=0; i<a.size(); i++) {
        if(a[i]=='<' || a[i]=='{' || a[i]=='[' || a[i]=='(') st.push(a[i]);
        else {
            if(st.empty()) {s=-1; break;}
            if(a[i]-st.top()<1 || a[i]-st.top()>2) s++;
            st.pop();
        }
    }
    if(s==-1 || !st.empty()) cout<<"Impossible";
    else cout<<s;
    return 0;
}
