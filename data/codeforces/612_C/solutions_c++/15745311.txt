#include <bits/stdc++.h>
using namespace std;
stack <char> st;
char ch;
int ans;
int main()
{
	while(cin >> ch){
		if(ch == '(' || ch == '[' || ch == '{' || ch == '<'){
			st.push(ch);
		}
		else{
			if(st.empty()){
				cout <<"Impossible";
				return 0;
			}
			if(st.top() != ch - 1 && st.top() != ch - 2){
				++ ans;
			}
			st.pop();
		}
	}
	if(st.size() > 0){
		cout << "Impossible";
		return 0;
	}		
	cout << ans;
	return 0;
}
