#include<bits/stdc++.h>
using namespace std;
int ans;
string s;
stack<char> st;
int main()
	{
	cin >> s;
	for(int i = 0; i < s.size(); i++)
		{
		if(s[i] == '[' || s[i] == '{' || s[i] == '(' || s[i] == '<')
			st.push(s[i]);
		else 
			{
			if(st.size() == 0)
				puts("Impossible"), exit(0);
			if(st.top() == '[' && s[i] != ']')
				ans++;
			if(st.top() == '{' && s[i] != '}')
				ans++;
			if(st.top() == '(' && s[i] != ')')
				ans++;
			if(st.top() == '<' && s[i] != '>')
				ans++;
			st.pop();	
			}
		}
	if(st.size())
		puts("Impossible"), exit(0);
	cout << ans;
	}
