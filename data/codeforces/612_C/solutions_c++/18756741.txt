#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
long long n,rez,o=0;
char c[1000010];
string s;
int main(){
	cin>>s;
	n=s.length();
	for(int i=0;i<n;i++){
		if(s[i]=='{'||s[i]=='['||s[i]=='('||s[i]=='<'){
			c[o]=s[i];
			o++;
		}else{
			if(c[o-1]!='{'){
			    if(s[i]=='}')rez++;
			};
			if(c[o-1]!='['){
				if(s[i]==']')rez++;
			};
			if(c[o-1]!='('){
				if(s[i]==')')rez++;
			};
			if(c[o-1]!='<'){
				if(s[i]=='>')rez++;
			};
			o--;
			if(o<0){
				cout<<"Impossible";
				return 0;
			}
		}
	}
	if(o!=0)cout<<"Impossible";
	else cout<<rez;
}
