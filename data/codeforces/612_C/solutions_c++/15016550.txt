
#include<bits/stdc++.h>
using namespace std;

int isOpen(char ch)
{
	if(ch=='<' || ch=='{' || ch=='[' || ch=='(')
		return 1;
	return 0;
}

int isMatch(char ch1,char ch2)
{
	if( abs(ch1-ch2) <=2 )
		return 1;
	return 0;
}

int main()
{
	stack <char> arr;
	string s;
	cin >> s;
	int n=s.length(),count=0,flag=0;
	for(int i=0;i<n;i++)
	{
		if(isOpen(s[i]))
		{
			arr.push(s[i]);
		}
		else
		{
			if(arr.size()==0)
			{
				flag=1;
				break;
			}
			if(isMatch(s[i],arr.top())==0 )
				count++;
			arr.pop();
		}
	}
	if(flag==1 || arr.size()>0)
		cout << "Impossible\n";
	else
		cout << count << endl;
	return 0;
}
