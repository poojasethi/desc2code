#include<bits/stdc++.h>
using namespace std;

stack<char >ss;

int main()
{
    string s;
    cin>>s;

    int ans=0;
    for(int i=0; i<s.size(); i++)
    {
        if(s[i]=='<' ||s[i]=='('||s[i]=='{'||s[i]=='[' )
            ss.push(s[i]);
        else
        {
            if(ss.empty())   return cout<<"Impossible"<<endl,0;
            if(ss.top()+2==s[i] or ss.top()+1==s[i])
                ss.pop();
            else    ans++,ss.pop();

        }
    }
    if(ss.size())  return cout<<"Impossible"<<endl,0;
    cout<<ans<<endl;

    return 0;

}
