#include<bits/stdc++.h>
using namespace std;
char str[1000005];
int main(){
    stack<char>stk;
    char ch;
    int ans=0,i=0;
    scanf("%s",str);
    while(str[i]){
        ch=str[i++];
        if(ch=='<' || ch=='(' || ch=='{' || ch=='[') stk.push(ch);
        else{
            if(stk.empty()) { printf("Impossible"); return 0;}
            char p=stk.top();
            if((ch=='}' && p=='{') || (ch==']' && p=='[') || (ch==')' && p=='(') || (ch=='>' && p=='<')) ;
            else ans++;
            stk.pop();
        }
    }
    if(stk.empty()) printf("%d",ans);
    else printf("Impossible");
}
