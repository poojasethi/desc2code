#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
#define abs(x) (x<0) ? -x : x
int d[1000050],tail,ans,N;
char s[1000050];
inline bool l(char s)
{
	if (s=='<'||s=='{'||s=='('||s=='[') return 1;
	return 0;
}
int main()
{
	scanf("%s",s+1);
	N=strlen(s+1);
	for (int i=1;i<=N;++i) 
		if (l(s[i])) {
			d[++tail]=s[i];
		} else {
			if (!tail) {
				ans=-1; break;
			} else {
				if (abs(s[i]-d[tail])>2) ++ans;
				--tail;
			}
		}
	if (tail) ans=-1;
	if (ans<0) printf("Impossible\n");
	else printf("%d\n",ans);
}
