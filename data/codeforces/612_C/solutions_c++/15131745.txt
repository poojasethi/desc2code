#include <bits/stdc++.h>

using namespace std;

stack <char> open;

int main()
{
    string s;
    cin >> s;
    int ans = 0;
	for(int i = 0 ; i < s.size() ; i++){
		if(s[i] == '<' || s[i] == '[' || s[i] == '(' || s[i] == '{')
			open.push(s[i]);
		else{
			if(!open.size())
				return cout << "Impossible" , 0;
			if(s[i] - open.top() != 2 && s[i] - open.top() != 1)
				ans ++;
			open.pop();
		}
	}
	if(open.size()) cout << "Impossible";
	else cout << ans;
}
