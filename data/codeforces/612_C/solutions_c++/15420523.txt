#include <bits/stdc++.h>
using namespace std;
int main(){
    string s; cin >> s;
    stack<char>p;
    int ans = 0;
    for( int i = 0 ; i < (int)s.size() ; ++i ){
        if( s[i] == '[' || s[i] == '{' || s[i] == '<' || s[i] == '(' ) p.push(s[i]);
        else{
            if(p.empty()){ cout << "Impossible\n"; return 0; }
            if( abs( s[i] - p.top() ) > 2 ) ++ans;
            p.pop();
        }
    }
    (!p.empty() ? cout << "Impossible\n" : cout << ans << '\n' );
}
