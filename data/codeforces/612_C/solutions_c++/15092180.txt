#include<bits/stdc++.h>
using namespace std;
char a[1000100];
string s;
int h=0,ans=0;
int main(){
	cin>>s;
	for (int i=0; i<s.size(); i++){
		if (s[i]=='[' || s[i]=='{' || s[i]=='<' || s[i]=='(') a[++h]=s[i];
		else {
		    if (h==0){
			    printf("Impossible");
				return 0;
			}
			if (abs(s[i]-a[h])>2) ans++;
			h--;
		}
	}
	if (h){
	    printf("Impossible");
	    return 0;  
	}
	printf("%d",ans);
}
