#include <bits/stdc++.h>

using namespace std;

int main(){
	string s;
	cin>>s;
	stack<char> mystack;
	int ans = 0;
	for(int i=0; i<s.length(); i++){
		if(s[i] == '<' || s[i] == '{' || s[i] == '(' || s[i] == '[')
			mystack.push(s[i]);
		else
		{
			if(mystack.empty()){
				cout<<"Impossible"<<endl;
				return 0;		
			}
			if(s[i] - mystack.top() > 2 || s[i] - mystack.top() <= 0)
				ans++;
			mystack.pop();
		}
	}
	if(mystack.empty())
		cout<<ans<<endl;
	else
		cout<<"Impossible"<<endl;
	return 0;
}
