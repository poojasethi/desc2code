#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
#define maxn 1111111
char c[maxn],s[maxn];
int res,ans;
int main()
{
	while(~scanf("%s",c))
	{
		res=ans=0;
		int len=strlen(c);
		for(int i=0;i<len;i++)
		{
			if(c[i]=='{'||c[i]=='['||c[i]=='('||c[i]=='<')s[++res]=c[i];
			else 
			{
				if(c[i]!=s[res]+1&&c[i]!=s[res]+2)ans++;
				res--;
			}
			if(res<0)break;
		}
		if(res) printf("Impossible\n");
		else printf("%d\n",ans);
	}
	return 0;
}
