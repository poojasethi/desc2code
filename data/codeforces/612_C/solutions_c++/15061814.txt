#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <stack>
using namespace std;
string a;
stack<char> st;
int s;
int main(){
    cin>>a;
    for(int i=0; i<a.size(); i++) {
        if(a[i]=='<' || a[i]=='{' || a[i]=='[' || a[i]=='(') st.push(a[i]);
        else {
            if(st.empty()) {s=-1; break;}
            if(a[i]-st.top()<1 || a[i]-st.top()>2) s++;
            st.pop();
        }
    }
    if(s==-1 || st.size()) cout<<"Impossible";
    else cout<<s;
    return 0;
}
