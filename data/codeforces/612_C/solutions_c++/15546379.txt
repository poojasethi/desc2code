#include <algorithm>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <iostream>
using namespace std;
char s[1010000];
char stack[1010000];
char t[300];
int now,n;
int main()
{
	t['>']='<';
	t[')']='(';
	t['}']='{';
	t[']']='[';
	scanf("%s",s+1);
	n = strlen(s+1);
	int can = 1,ans = 0;
	for (int i = 1;i <= n;i++)
	{
		if (s[i]=='('||s[i]=='['||s[i]=='{'||s[i]=='<')
			stack[++now] = s[i];
		else 
		{
			if (now==0) {can = 0;break;}
			if (stack[now]!=t[s[i]]) ans++;
			now--;
		}
	}
	if (can == 0 || now > 0) 
	{
		puts("Impossible");return 0;
	}
	cout << ans <<endl;

}
