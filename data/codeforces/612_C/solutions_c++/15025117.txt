#include <stdio.h>
int sl=-1,re=0;
char st[1000001],s='a';
int main()
{
    while (s!='\n')
    {
        s='\n';
        scanf("%c",&s);
        if (s=='\n') break;
        if (s=='(' || s=='{' || s=='[' || s=='<') st[++sl]=s;
        else
        {
            if (sl<=-1) 
            {
            sl=-2;
            break;
            }
            if ((st[sl]=='(' && s!=')')||(st[sl]=='[' && s!=']')||(st[sl]=='{' && s!='}')||(st[sl]=='<' && s!='>')) re++;
            sl--;
        }
    }
    if (sl!=-1) printf("Impossible"); else printf("%d",re);
    return 0;
}
