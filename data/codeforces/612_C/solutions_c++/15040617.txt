#include<iostream>
#include<stack>
using namespace std;
int n,m,k;
string s;
stack <char> S;
main()
{
   cin>>s;
   for(int i=0;i<s.size();i++)
   {
       if (s[i]=='[')
       S.push(']');
       else if (s[i]=='(')
       S.push(')');
       else if (s[i]=='{')
       S.push('}');
       else if (s[i]=='<')
       S.push('>');
       else if (S.size()!=0)
       {
           if (s[i]!=S.top())
           k++;
           S.pop();
       }
      else
      {cout<<"Impossible"<<endl;
       return 0;
}
   }
   if (S.size()>0)
    cout<<"Impossible"<<endl;
  else cout<<k<<endl;
}
