#include <iostream>
#include <vector>
using namespace std;
int main()
{
	string s;
	cin>>s;
	long long int t=0;
	vector<char> A;
	for (int i=0; i<s.size(); i++) 
	if(s[i]=='{' or s[i]=='<' or s[i]=='[' or s[i]=='(') A.push_back(s[i]); else
	{
		if(A.size()==0) {cout<<"Impossible"; return 0;}
		if((A[A.size()-1]=='<' and s[i]=='>') or (A[A.size()-1]=='(' and s[i]==')') 
		or (A[A.size()-1]=='[' and s[i]==']') or (A[A.size()-1]=='{' and s[i]=='}')) t+=0;
		else t++;
		A.pop_back();
	}
	if(A.size()==0) cout<<t; else cout<<"Impossible";
}
