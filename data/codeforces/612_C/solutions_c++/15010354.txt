#include <bits/stdc++.h>

#define mp(x, y) make_pair((x), (y))

typedef long long ll;

char s[1000005];
char st[1000005];
int top;
int ans;

char op(char cl)
{
	if(cl==')') return '(';
	if(cl==']') return '[';
	if(cl=='}') return '{';
	if(cl=='>') return '<';
}

int main()
{
	scanf("%s", s);
	for(int i=0; s[i]; i++) {
		if(s[i]=='(' || s[i]=='[' || s[i]=='{' || s[i]=='<') {
			st[top++]=s[i];
		} else {
			if(top<1) {
				printf("Impossible\n");
				return 0;
			} else {
				ans+=(op(s[i])!=st[--top]);
			}
		}
	}
	if(top) {
		printf("Impossible\n");
	} else {
		printf("%d\n", ans);
	}

	return 0;
}
