#include <bits/stdc++.h>
using namespace std;
int main() {
	string s;
	cin >> s;
	stack<char> t;
	int ans = 0, i;
	map<char, char> mp;
	mp['<'] = '>';
	mp['('] = ')';
	mp['{'] = '}';
	mp['['] = ']';
	for(i = 0; i < s.size(); i++) {
		switch(s[i]) {
			case '<':
			case '{':
			case '(':
			case '[':
				t.push(s[i]);
				break;
			default:
				if(t.empty()) {
					cout << "Impossible" << endl;
					return 0;
				}
				if(mp[t.top()] != s[i])
					ans++;
				t.pop();
		}
	}
	if(!t.empty()) cout << "Impossible" << endl;
	else cout << ans << endl;
	return 0;
}
