#include <bits/stdc++.h>

int a[8];
int main() {
	int i,j,k,l,n,m,ptr=0;
	char c[1000006],v[1000006];
	cin>>c;
	n = strlen(c);
	int fl=1,ans=0;
	for(i=0;i<n;i++) {
		if(c[i]=='<')
			v[ptr++]='>';
		else if(c[i]=='(')
			v[ptr++]=')';
		else if(c[i]=='{')
			v[ptr++]='}';
		else if(c[i]=='[')
			v[ptr++]=']';
		else 
			if(ptr>0) {
				if(v[ptr-1]!=c[i])
					ans++;
				ptr--;
			}
			else {
				fl=0;
				break;
			}
	}
	if(!fl||ptr)
		cout<<"Impossible\n";
	else
		cout<<ans<<endl;
	return 0;
}
