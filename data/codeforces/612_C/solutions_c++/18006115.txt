#include <iostream>
#include <stack>
#include <string>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	string s;
	int ans(0);
	cin>>s;
	stack <char> v;
	for (int i=0;i<s.size();i++){
		if (s[i]=='<' || s[i]=='{'||s[i]=='[' || s[i]=='('){
			v.push(s[i]);
		}
		else{
			if (v.empty()){
				cout<<"Impossible";
				return 0;
			}
			if ((int)s[i]-(int)v.top()!=1 && (int)s[i]-(int)v.top()!=2){
				ans++;
			}
			v.pop();
		}
	}
	if (!v.empty())
		cout<<"Impossible";
	else
		cout<<ans;
}
