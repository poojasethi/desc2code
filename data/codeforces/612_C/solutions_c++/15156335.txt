#include<bits/stdc++.h>
using namespace std;
stack<char> st;
char str[1000006];
bool isOpen(char x)
	{
		return (x=='{'|| x=='[' || x=='(' || x=='<');
	}
int main()
{
	int ans=0;
	char x;
	cin>>str;
	for(int i=0;str[i];i++)
	{
		if(isOpen(str[i]))
			st.push(str[i]);
		else
		{
			if(st.empty())
				{
					cout<<"Impossible\n";
					return 0;
				}
			x=st.top();
			if(!((x=='{' && str[i]=='}' )||(x=='[' && str[i]==']' )||(x=='(' && str[i]==')' )||(x=='<' && str[i]=='>' )))
				ans++;
			st.pop();		
		}
	}
	if(!st.empty())
		cout<<"Impossible\n";
	else
		cout<<ans<<endl;
	return 0;
}
