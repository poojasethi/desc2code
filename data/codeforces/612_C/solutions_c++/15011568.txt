#include <bits/stdc++.h>
using namespace std;

int hasil = 0;
string s;
stack <char> kurung;

int main() {
	cin >> s;
	for (int i = 0; i < s.length(); i++) {
		if (s[i] == '<')
			kurung.push('>');
		else if (s[i] == '{')
			kurung.push('}');
		else if (s[i] == '[')
			kurung.push(']');
		else if (s[i] == '(')
			kurung.push(')');
		else {
			if (kurung.empty()) {
				printf("Impossible\n");
				return 0;
			}
			if (kurung.top() != s[i])
				hasil++;
			kurung.pop();
		}
	}
	if (!kurung.empty())
		printf("Impossible\n");
	else
		printf("%d\n", hasil);
	return 0;
}
