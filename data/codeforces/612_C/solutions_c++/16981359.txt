#include <stdio.h>
#include <stack>
#include <string.h>

using namespace std;

stack <char> st;
int n;
char lin[1000010];

int main()
{
	scanf(" %s", lin);
	n = strlen(lin);
	int i, ans = 0, f = 1;
	for(i = 0; i < n; i++)
	{
		if(lin[i] == '<' || lin[i] == '{' || lin[i] == '[' || lin[i] == '(')
			st.push(lin[i]);
		else
		{
			if(st.empty())
			{
				f = 0;
				break;
			}
			char c = st.top();
			st.pop();
			if(c == '(')
				c++;
			else
				c += 2;
			if(c != lin[i])
				ans++;
		}
	}
	if(f && st.empty())
		printf("%d\n", ans);
	else
		printf("Impossible\n");
	return 0;
}
