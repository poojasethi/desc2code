#include <bits/stdc++.h>
using namespace std;
stack <char> s;
int ans;
char str[1000005];
int main(){
    gets(str);
    int l=strlen(str);
    for(int x=0;x<l;x++){
        if(str[x]=='{'||str[x]=='['||str[x]=='('||str[x]=='<') s.push(str[x]);
        else{
            if(s.empty()){
                printf("Impossible\n");
                return 0;
            }
            else if(str[x]-s.top()>2||str[x]-s.top()<1) ans++;
            s.pop();
        }
    }
    if(!s.empty()){
        printf("Impossible\n");
    }
    else printf("%d\n",ans);
    return 0;
}
