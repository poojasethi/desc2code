#include <bits/stdc++.h>
using namespace std;

int a;
char S[1000100];
stack<char> s;

int main() {
//	freopen("in.txt", "r", stdin);
	scanf("%s", S);
	for (int i = 0; S[i] != '\0'; i++) {
		if (S[i] == '<' || S[i] == '{' || S[i] == '[' || S[i] == '(') {
			s.push(S[i]);
		} else {
			if (s.empty()) {
				cout << "Impossible";
				return 0;
			}
			char c = s.top(); s.pop();
			if ((c == '{' && S[i] != '}') || (c == '<' && S[i] != '>') || (c == '[' && S[i] != ']') || (c == '(' && S[i] != ')')) a++;
		}
	}
	if (s.size()) cout << "Impossible";
	else cout << a;
}
