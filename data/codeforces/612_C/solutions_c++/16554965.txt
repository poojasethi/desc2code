#include<iostream>
#include<stack>
#include<bits/stdc++.h>
using namespace std;
stack <char> sk;

char check(char a)
{
	if(a=='>') return '<';
	if(a==')') return '(';
	if(a=='}') return '{';
	if(a==']') return '[';
	return a;
}

main()
{
	string s;
	cin>>s;
	long d=0;
	for(long i=0;i<s.length();i++){
		if(check(s[i])==s[i]){
			sk.push(s[i]);
		}
		else{
			if(sk.empty()){
				cout<<"Impossible";
				return 0;
			}
			else{
				if(sk.top()!=check(s[i])){
					d++;
				}
				sk.pop();
			}
		}
		
	}
	if(sk.empty()) cout<<d;
	else cout<<"Impossible";
}
