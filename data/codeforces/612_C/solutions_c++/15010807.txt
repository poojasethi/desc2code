#include <bits/stdc++.h>
using namespace std;

bool eq( char a, char b )
{
	return ((a == '[' && b == ']') || (a == '{' && b == '}') || (a == '<' && b == '>')
		|| (a == '(' && b == ')'));
}
		

int main( void )
{
	string s;
	stack <char> st;
	int cnt = 0;

	cin >> s;
	for( int i = 0; i < (int)s.length(); i++ ) {
		if( s[i] == '[' || s[i] == '(' || s[i] == '{' || s[i] == '<' )
			st.push(s[i]);
		else {
			if( st.empty() ) {
				cout << "Impossible";
				return 0;
			}
			else if( !eq(st.top(),s[i]) )
				++cnt;
			st.pop();
		}
	}
	if( !st.empty() )
		cout << "Impossible";
	else
		cout << cnt;

	return 0;
}
