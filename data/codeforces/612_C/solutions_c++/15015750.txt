#include <iostream>
#include <stack>
#include <stdlib.h>

using namespace std;

int main(){
	string s;
	cin >> s;
	stack<char> bal;
	int num=0;
	for(int i=0; i<s.length(); i++){
		if(s[i] == '(' || s[i] == '[' || s[i] == '{' || s[i] == '<'){
			bal.push(s[i]);
		} else{
			if(bal.empty()){
				cout << "Impossible" << endl;
				return 0;
			} else{
				char c = bal.top();
				if(abs(s[i]-c) > 2)
					num++;
				bal.pop();
			}
		}
	}
	if(!bal.empty()){
		cout << "Impossible" << endl;
		return 0;
	}
	cout << num << endl;
}
