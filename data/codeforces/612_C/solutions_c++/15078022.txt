#include <bits/stdc++.h>
using namespace std;

int main() {
    char str[1000001];
    int len, ans=0;
    stack<char> st;
    gets(str);
    len=strlen(str);
    if(len%2==1) return cout<<"Impossible",0;
    for(int i=0;i<len;i++){
        if(str[i]=='(' || str[i]=='<' || str[i]=='{' || str[i]=='[')
            st.push(str[i]);
        else{
            if(st.empty())
                return cout<<"Impossible",0;
            else if(abs(str[i]-st.top())>2)
                ans++;
            st.pop();
        }
    }
    if(!st.empty())
        cout<<"Impossible";
    else
      cout<<ans;
}
