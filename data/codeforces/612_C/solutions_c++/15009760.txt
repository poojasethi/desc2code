#include <bits/stdc++.h>

using namespace std;

string s;
char a[2000020];
int st,ans,i;

char rev(char a)
 {
  if (a == '}') return '{';
  if (a == '>') return '<';
  if (a == ')') return '(';
  if (a == ']') return '[';
 }

int main()
 {
  cin>>s;
  for (i=0;i<s.size();i++)
   {
    if (s[i] == '{' || s[i] == '(' || s[i] == '[' || s[i] == '<') a[++st]=s[i]; else
     {
      if (st == 0) { cout<<"Impossible"<<endl; return 0;}
      if (a[st] != rev(s[i])) ans++;
      st--;
     }
   }
      if (st != 0) { cout<<"Impossible"<<endl; return 0;}
  cout<<ans<<endl;
 }
