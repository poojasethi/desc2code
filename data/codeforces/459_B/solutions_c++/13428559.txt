#include<iostream>

using namespace std;

long long int n,ma,mi=1000000001,nma,nmi;
long long int a[200001];

main(){
  cin>>n;
  for(int i=0;i<n;i++){
    cin>>a[i];
    mi=min(mi,a[i]);
    ma=max(ma,a[i]);
  }
  for(int i=0;i<n;i++){
    if(ma==a[i])
      nma++;
    if(mi==a[i])
      nmi++;
  }
  if(mi==ma)
    cout<<"0 "<<n*(n-1)/2;
  else
  cout<<ma-mi<<" "<<nma*nmi;
}
