#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
long long a[1000000],dif,x,y,n,i;
cin>>n;
for(x=0;x<n;x++)
cin>>a[x];
sort(a,a+n);
dif=a[n-1]-a[0];
x=1;
y=1;
for(i=1;i<n-1;i++)
{
if(a[i]==a[0])
x++;
if(a[i]==a[n-1])
y++;
}
if(dif==0)
cout<<0<<" "<<n*(n-1)/2;
else
cout<<dif<<" "<<x*y;
return 0;
}
