#include <bits/stdc++.h>

using namespace std;
long long int n, x, i, a = -1, b = 1e12;
int main()
{

    cin >> n;
    map<int, long long>m;
    for (i = 0; i < n; i++)
        cin >> x,a=max(a,x),b=min(b,x),m[x]++;
    if (a==b)cout << 0 << ' ' << m[a]*(m[a]-1)/2 << '\n';
        else
    cout << a-b << ' ' << m[a]*m[b] << '\n';
    return 0;
}
