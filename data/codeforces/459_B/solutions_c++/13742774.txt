#include<iostream>
#include<algorithm>
using namespace std;
main()
{
long long n;
cin>>n;
long long a[n],i,mn=1000000001,mx=0;
for(i=0;i<n;i++)
{
cin>>a[i];
mn=min(mn,a[i]);
mx=max(mx,a[i]);
}
long long t1=0,t2=0;
for(i=0;i<n;i++)
{
if(a[i]==mn)
t1++;
if(a[i]==mx)
t2++;
}
cout<<mx-mn<<" ";
if(mn==mx)
cout<<(n*(n-1))/2<<"\n";
else
cout<<t1*t2<<"\n";
}
