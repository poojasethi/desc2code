#include<cstdio>
using namespace std;
int main()
{
    int n,a=0,b=1000000001,c,d,x;
    scanf("%d",&n);
    while(n--)
    {
        scanf("%d",&x);
        if(x>a)a=x,c=1;else if(x==a)++c;
        if(x<b)b=x,d=1;else if(x==b)++d;
    }
    a-=b;
    printf("%d %I64d\n",a,(long long)c*(a?2*d:(c-1))/2);
}
