#include<cstdio>
#include<algorithm>
using namespace std;
main()
{
    int n,i,f,l;
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++) scanf("%d",&a[i]);
    sort(a,a+n);
    for(i=0;i<n-1;i++) if(a[i]!=a[i+1]) break;
    f=i+1;
    for(i=n-1;i>0;i--) if(a[i]!=a[i-1]) break;
    l=n-i;
    printf("%d %I64d",a[n-1]-a[0],f==n?((1LL*n)*(n-1))/2:(1LL*f)*l);
}
