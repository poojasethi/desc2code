#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
long long i,j;
long n;
cin>>n;
long int *b= new long int [n];
for(i=0; i<n; i++)
cin>>b[i];
sort(b,b+i);
cout<<b[n-1]-b[0]<<' ';
for(i=1; i<n && b[i-1]==b[i]; i++);
for(j=n-2; j>=0 && b[j+1]==b[j]; j--);
j=n-1-j;
if(i==n)
cout<<i*(i-1)/2;
else
cout<<i*j;
delete b;
return 0;
}
