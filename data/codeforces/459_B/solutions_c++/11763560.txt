#include<iostream>
using namespace std;
long long n, a[200001], i, ma, mi=1000000001, nma, nmi;
main(){
  cin>>n;
  for(i=0;i<n;i++){cin>>a[i];mi=min(mi,a[i]);ma=max(ma,a[i]);}
  for(i=0;i<n;i++){if(ma==a[i])nma++;if(mi==a[i])nmi++;}
  if(mi==ma)cout<<"0 "<<n*(n-1)/2;else
  cout<<ma-mi<<" "<<nma*nmi;
}
