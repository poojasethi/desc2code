#include<bits/stdc++.h>

using namespace std;

long long  i,sa,s,arr[200004];
long long  n;

int main(){

	cin >> n;

	for(i=0  ; i<n ; i++)
		cin >> arr[i];

	sort(arr,arr+n);

	i=0;
	s=n-1;

	while(arr[0]==arr[i])
		i++;


	while(arr[n-1]==arr[s]){
		s--;
		sa++;
	}

	if(i==n)
		cout << '0' << ' ' << i*(i-1)/2;

	else
	cout << arr[n-1]-arr[0] << ' ' << sa*i;

}
