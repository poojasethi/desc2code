#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,m,mmax=0,mmin=1e9,a=1,b=1;
	cin>>n;
	while(n--){
		cin>>m;
		if(m>mmax) {mmax=m;a=1;}
		else if(m==mmax) a++;
		if(m<mmin) {mmin=m;b=1;}
		else if(m==mmin) b++;
	}
	cout<<mmax-mmin<<' '<<(mmax==mmin? a*(a-1)/2:a*b);
	return 0;
}
