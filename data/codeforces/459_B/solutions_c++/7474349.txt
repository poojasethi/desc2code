#include <bits/stdc++.h>
using namespace std;

const int N = 2e5+8;
map<int, int> mp;
int n, b, x=0, y=1e9+1;

int main(){
	//freopen("input.in", "r", stdin);
	scanf("%d", &n);
	for(int i=1; i<=n; i++){
		scanf("%d", &b);
		if(b>x) x=b;
		if(b<y) y=b;
		mp[b]++;
	}
	if(x!=y) cout << x-y << ' ' << 1ll*mp[x]*mp[y];
	else cout << x-y << ' ' << 1ll*mp[x]*(mp[x]-1)/2;
}
