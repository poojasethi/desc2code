#include<cstdio>
int n,x,mx,mn,a,b,i;
int main()
{
	scanf("%d", &n);
	for (i=1,mn=1000000000; i<=n; i++)
	{
		scanf("%d", &x);
		if (x<mn) mn=x,a=0;
		if (x==mn) a++;
		if (x>mx) mx=x,b=0;
		if (x==mx) b++;
	}
	if (mx == mn) printf("0 %I64d\n", 1LL*n*(n - 1)/2);
	else printf("%d %I64d", mx-mn, 1LL*a*b);
}
