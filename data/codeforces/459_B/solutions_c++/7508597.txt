#include<stdio.h>

int main()
{
	int n, b, cmx, cmn, mx=0, mn=1<<30;
	scanf("%d", &n);
	while(n--)
	{
		scanf("%d", &b);
		mx<b?(cmx=1, mx=b):mx==b?cmx++:0;
		mn>b?(cmn=1, mn=b):mn==b?cmn++:0;
	}
	printf("%d %lld\n", mx-mn, mn==mx?1LL*cmx*(cmx-1)/2:(long long)cmx*cmn);
	return 0;
}
