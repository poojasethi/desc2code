#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	ll n;
	int i;
	ll k,l,r;
	ll a[200000];
	cin>>n;
	for(i=0;i<n;i++)
	cin>>a[i];
	sort(a,a+n);
	k=count(a,a+n,a[0]);
	l=count(a,a+n,a[n-1]);
	if(a[0]==a[n-1])
	r=(n*(n-1))/2;
	else
	r=l*k;
	cout<<a[n-1]-a[0]<<" "<<r;
	return 0;
}
