#include <bits/stdc++.h>
using namespace std;
vector<int> val;
int main()
{
    long long  n,x=0,y=0;cin>>n;
    for(int i=0;i<n;i++){int a;cin>>a;val.push_back(a);}
    sort(val.begin(),val.end());
    for(int i=0;i<n;i++){if(val[i]==val[0])x++;if(val[i]==val[n-1])y++;}
    cout<<val[n-1]-val[0]<<" "<<((val[n-1]==val[0])?(x*(x-1)/2):x*y);
    return 0;
}
