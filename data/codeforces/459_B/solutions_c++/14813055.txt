#include<bits/stdc++.h>
using namespace std;
long long int a[200005];
int main()
{
   long long int i,j,k,n,s=1,p=1,ans=0;
   cin>>k;
   for(i=0;i<k;i++)
    cin>>a[i];
   sort(a,a+k);
   s=count(a,a+k,a[0]);
   p=count(a,a+k,a[k-1]);
   if(a[0]==a[k-1])
      ans=k*(k-1)/2;
   else ans=s*p;
   cout<<a[k-1]-a[0]<<" "<<ans<<endl;
}
