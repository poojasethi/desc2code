#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long a, n, i, k, l, g, h;
    cin>>n;
    long long m[n];
    for(i=0; i<n; i++){
        cin>>m[i];
    }
    sort(m, m+n);
    k=count(m, m+n, m[0]);
    l=count(m, m+n, m[n-1]);
    if(m[n-1]-m[0]==0) cout<<"0 "<<k*(k-1)/2<<endl;
    else cout<<m[n-1]-m[0]<<" "<<k*l<<endl;
    return 0;
}
