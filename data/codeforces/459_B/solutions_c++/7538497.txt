#include <bits/stdc++.h>
using namespace std;

long long n, a[200005], x, y, cx = 0, cy = 0;

int main(){
cin >> n;
for(int i = 0; i < n; i++) cin >> a[i];
x = *min_element(a, a + n);
y = *max_element(a, a + n);
for(int i = 0; i < n; i++) cx += x == a[i], cy += y == a[i];
cout << y - x << " " << (x == y ? cx * (cx - 1) / 2 : cx * cy) << '\n';
}
