#include<bits/stdc++.h>
using namespace std;
int main(){
    long long  n,i,j,a[200001],min=0,max=0;
    cin>>n;
    for(i=0;i<n;i++){
        cin>>a[i];
    }
    sort(a,a+n);
    cout<<(a[n-1]-a[0])<<" ";
    min=count(a,a+n,a[0]);
    max=count(a,a+n,a[n-1]);
    cout<<(a[0]==a[n-1]? n*(n-1)/2 : max*min);
    return 0;
}
