#include <iostream>
#include <algorithm>
using namespace std;
long long int a,b[1000000],t1,t2;
int main() {
cin>>a;
for(int i=0;i<a;i++)cin>>b[i];
sort(b,b+a);
for(int i=0; i<a;i++){
   if(b[0]==b[i]){t1++;}
   if(b[a-1]==b[i]){t2++;}
 }
 cout<<b[a-1]-b[0]<<" ";
 if (b[a-1]-b[0]==0)cout<<(t2*(t2-1))/2;else cout<<t2*t1;
 
}
