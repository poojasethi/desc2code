#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int n,a[200000],i;
	long long ti=0,tx=0;
	cin>>n;
	for(int i=0;i<n;i++)cin>>a[i];
	sort(a,a+n);
	i=n-1;
	while(a[i]==a[n-1]){
		i--;
		tx++;
	}
	i=0;
	while(a[i]==a[0]){
		i++;
		ti++;
	}
	cout<<a[n-1]-a[0]<<" ";
	if(a[0]==a[n-1])cout<<ti*(ti-1)/2;
	else cout<<ti*tx;
}
