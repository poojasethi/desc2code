#include<bits/stdc++.h>

using namespace std;

long long  i,sa,s,arr[200004],m;

int main(){

	cin >> m;

	for(i=0  ; i<m ; i++)
		cin >> arr[i];

	sort(arr,arr+m);

	i=0;
	s=m-1;
	while(arr[0]==arr[i])
		i++;


	while(arr[m-1]==arr[s]){
		s--;
		sa++;
	}

	if(i==m)
		cout << '0' << ' ' << i*(i-1)/2;
	else	cout << arr[m-1]-arr[0] << ' ' << sa*i;

}
