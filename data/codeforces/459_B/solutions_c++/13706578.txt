#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main(){
	ll n,x,mi=1000000000,mx=0;
	cin>>n;
	map<ll,ll>m;
	for(int i=0;i<n;i++){
		cin>>x;
		mx=max(mx,x);
		mi=min(mi,x);
		m[x]++;
	}
	long long q;
	if(mx==mi) q=(m[mx]*(m[mx]-1))/2;
	else q=m[mx]*m[mi];
	cout<<mx-mi<<" "<<q<<endl;
	return 0;
}
