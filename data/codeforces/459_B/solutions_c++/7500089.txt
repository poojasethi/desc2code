#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <map>
using namespace std;
map <long long,long long> m;
long long  n,mi=1e15,ma=0,a,r;
int main(int argc, char *argv[])
{cin>>n;

for (int i=1;i<=n;i++) {cin>>a;ma=max(a,ma); mi=min(a,mi);m[a]++;}
r=ma-mi; cout<<r<<" "<<(r?m[ma]*m[mi]:n*(n-1)/2)<<endl;

    //system("PAUSE");
    return 0;
}
