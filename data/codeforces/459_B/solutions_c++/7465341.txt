#include<bits/stdc++.h>

using namespace std;

long long a,b,c,d,n,m;

long long A[200005];

int main(){
	int i,j,k;
	cin>>n;

	for(i=1;i<=n;i++)
		cin>>A[i];

	sort(A+1,A+n+1);

	a=A[n]-A[1];
	i=1;
	m=n;
	while(A[1]==A[i]){
		c++;
		i++;
	}
	while(A[n]==A[m]){
		d++;
		m--;
	}
	if(A[1]!=A[n])	cout<<a<<' '<<c*d;
	else cout<<0<<' '<<n*(n-1)/2;
	return 0;
}
