#include<iostream>
using namespace std;
long long n,mn=-1,mx=-1,l,r,a[2000005];
int main()
{
	cin>>n;
	for(long long i=1;i<=n;i++)
	{
		cin>>a[i];
		mx=max(mx,a[i]);
		if(mn==-1 || mn>a[i])mn=a[i];
	}
	cout<<mx-mn<<" ";
	for(long long i=1;i<=n;i++)
	{if(a[i]==mn)l++;if(a[i]==mx)r++;}
	if(mx==mn)cout<<l*(l-1)/2;
	else cout<<l*r;
	return 0;
}
