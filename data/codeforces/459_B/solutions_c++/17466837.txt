#include<iostream>
using namespace std;
int main ()
{
	long long n,b=0,c,d,g[200000],i,l=0,f=0,k=0;
	cin>>n;
	for(i=1;i<=n;i++)cin>>g[i];
	c=g[1];
	for(i=1;i<=n;i++){if(b<g[i])b=g[i];
					if (c>g[i])c=g[i];
						}
	for(i=1;i<=n;i++){if(b==g[i])l++;
	if(c==g[i])f++;
	}					
	if(b==c){for(i=1;i<n;i++)k=k+i;cout<<0<<' '<<k;}
	else cout<<b-c<<' '<<l*f;
}
