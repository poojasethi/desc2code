#include <iostream>
#include<algorithm>
using namespace std;
long long n,a[1000000],k,m,sum1,sum2,i,s;
main()
{
cin>>n;
for(i=0;i<n;i++)
cin>>a[i];
sort(a,a+n);
sum1=a[0];
sum2=a[n-1];
for(i=0;i<n;i++)
{
if(a[i]==sum1)
k++;
if(a[i]==sum2)
m++;
}
s=k*m;
if(k==n)
s=n*(n-1)/2;
cout<<sum2-sum1<<" "<<s;
}
