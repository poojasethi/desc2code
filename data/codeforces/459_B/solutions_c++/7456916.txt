#include <cstdio>

int n,x,l,r;
long long s,t;

int main()
{
	scanf("%d",&n);
	l=0X7FFFFFFF;
	r=0;
	while (n--)
	{
		scanf("%d",&x);
		if (l>x)
		{
			l=x;
			s=1;
		}
		else s+=l==x;
		if (r<x)
		{
			r=x;
			t=1;
		}
		else t+=r==x;
	}
	if (l==r) printf("0 %I64d",s*(s-1)/2);
	else printf("%d %I64d",r-l,s*t);
	return 0;
}
