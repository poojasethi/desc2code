#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
	long long n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)cin>>a[i];
	sort(a,a+n);
	long long x=1,y=1;
	while(a[x]==a[x-1])x++;
	while(a[n-y]==a[n-y-1])y++;
	if(a[n-1]!=a[0])cout<<a[n-1]-a[0]<<" "<<x*y<<endl;
	else cout<<0<<" "<<n*(n-1)/2<<endl;
	return 0;
}
