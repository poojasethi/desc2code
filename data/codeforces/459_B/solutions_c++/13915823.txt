#include<iostream>
using namespace std;
long long n, a[200001], i, mx, mi=1000000001, nma, nmi;
main(){
  cin>>n;
  for(i=0;i<n;i++){cin>>a[i];mi=min(mi,a[i]);mx=max(mx,a[i]);}
  for(i=0;i<n;i++){if(mx==a[i])nma++;if(mi==a[i])nmi++;}
  if(mi==mx)cout<<"0 "<<n*(n-1)/2;else
  cout<<mx-mi<<" "<<nma*nmi;
}
