#include<bits/stdc++.h>
using namespace std;
int ara[26];
int main(){
    string s;
    string t;
    int flag1=1,flag2=1,flag3=1;

    cin>>s>>t;
    for(int i=0;s[i];i++) ara[s[i]-'a']++;
    for(int i=0;t[i];i++) ara[t[i]-'a']--;

    for(int i=0,j=0;s[i];i++){
            if(s[i]==t[j]) j++;
            if(!t[j]) { printf("automaton"); return 0; }
    }

    for(int i=0;i<26;i++){
        if(ara[i]!=0) flag2=0;
        if(ara[i]<0) flag1=0;
    }

    if(flag2==1)  printf("array");
    else if(flag1==1) printf("both");
    else printf("need tree");
}
