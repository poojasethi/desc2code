#include <stdio.h>
#include <algorithm>
#include <string>
#include <iostream>

using namespace std;

bool sa(string &s, string &t) {
	int i, j;
	for (i = j = 0; i < s.length() && j < t.length(); i++) {
		if (s[i] == t[j])	j++;
	}
	return j == t.length();
}

int main (void) {
	string s, t;
	cin >> s >> t;
	if (sa(s, t)) {
		printf ("automaton\n");
		return 0;
	}
	sort (s.begin(), s.end());
	sort (t.begin(), t.end());
	if (s == t) {
		printf ("array\n");
		return 0;
	}
	if (sa(s, t)) {
		printf ("both\n");
		return 0;
	}
	printf ("need tree\n");
	return 0;

}
