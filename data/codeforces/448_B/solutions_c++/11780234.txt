#include<bits/stdc++.h>
using namespace std;
int main(){
  string s,t;
  int ii,mm,nn;
   cin>>s>>t;
   mm=s.length();
   nn=t.length();
  map<char,int> S,T;
  for(ii=0;ii<mm;ii++)
  ++S[s[ii]];
  for(ii=0;ii<nn;ii++)
  ++T[t[ii]];
  if(S==T)
  return cout<<"array\n",0;
  for(ii=0;ii<nn;ii++)
  {  char c = t[ii];
  if(T[c]>S[c])
  return cout<<"need tree\n",0;
  }
  for(int i=0,j=0;i<mm&&j<nn;){
    if(s[i]==t[j])++i,++j;
    else ++i;
    if(j==t.size())return cout<<"automaton\n",0;
  }
  cout<<"both\n";
}
