#include<cstdio>
#include<cstring>
using namespace std;
char s[101],t[101];
int a[26],b[26];
int main()
{
	int i,j,l1,l2;
	scanf("%s",&s);
	scanf("%s",&t);
	l1=strlen(s);
	l2=strlen(t);
	for(i=0;i<l1;i++)
		a[s[i]-'a']++;
	for(j=0;j<l2;j++)
		b[t[j]-'a']++;
	for(i=0;i<26;i++)
	{
		if(a[i]<b[i])
		{
			break;
		}
	}
	if(i<26)
		printf("need tree\n");
	else
	{
	for(i=j=0;i<l1&&j<l2;j++,i++)
	{
		while(i<l1&&s[i]!=t[j])
			i++;
	}
	if(j>=l2)
		printf("automaton\n");
	else if(l1==l2)
		printf("array\n");
	else
		printf("both\n");
	}
	return 0;
}
