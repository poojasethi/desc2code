#include <bits/stdc++.h>
#define rep(I, N) for (int I = 0; I < (N); ++I)
using namespace std;
int t[30],r,i;
int main()
{
    string a,b;
    cin>>a>>b;
    rep(i,a.size()) t[a[i]-'a']++;
    rep(i,b.size()) t[b[i]-'a']--;
    rep(i,26) {r+=t[i];if(t[i]<0) {cout<<"need tree";return 0;}}
    if(r==0) {cout<<"array";return 0;}
    rep(j,a.size()) if(b[i]==a[j]) i++;
    if(i==b.size()) {cout<<"automaton";return 0;}
    cout<<"both";
}
