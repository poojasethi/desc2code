#include <bits/stdc++.h>
using namespace std;

int a[26], b[26];
	
int main(){
	//freopen("input.in", "r", stdin);
	string s, t;
	cin >> s >> t;
	for(int i=0; i<s.size(); i++) a[s[i]-'a']++;
	for(int i=0; i<t.size(); i++) b[t[i]-'a']++;
	bool arr = 0;
	for(int i=0; i<26; i++) if(a[i] < b[i]){
		printf("need tree");
		return 0;
	}
	else if(a[i] > b[i]) arr = 1;
	if(!arr) printf("array");
	else {
		int i=0, j=0;
		for(; j<s.size() && i<t.size(); j++){
			if(s[j] == t[i]) i++;
		}
		if(i==t.size()){
			printf("automaton");
			return 0;
		}
		else printf("both");
	}
}
