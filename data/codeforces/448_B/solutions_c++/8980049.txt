#include <bits/stdc++.h>
using namespace std;

int a[26];
string s, t;

int main() {
  cin >> s >> t;
  for (int i = 0; i < s.length(); ++i)
    ++a[s[i] - 'a'];
  for (int i = 0; i < t.length(); ++i)
    --a[t[i] - 'a'];
  for (int i = 0; i < 26; ++i)
    if (a[i] < 0)
      return puts("need tree"), 0;
  int it = 0;
  for (int i = 0; i < s.length(); ++i) {
    if (it < t.length() && t[it] == s[i])
      ++it;
  }
  if (it == t.length()) {
    puts("automaton");
  }
  else {
    puts(s.length() == t.length() ? "array" : "both");
  }
  return 0;
}
