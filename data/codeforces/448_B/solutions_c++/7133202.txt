#include<cstdio>
char s[111];
char t[111];
int tmp[200]={0};
int fun(int a,int b) {
	if(!t[b]) return 1;
	for(int i=a;s[i];i++)
		if(s[i]==t[b])
			return fun(i+1,b+1);
	return 0;
}
int main () {
	scanf("%s%s",s,t);
	for(int i=0;s[i];i++) tmp[s[i]]++;
	if(fun(0,0)) {
		printf("automaton");
		return 0;
	}
	for(int i=0;t[i];i++)
		if(tmp[t[i]]>0) tmp[t[i]]--;
		else {
			printf("need tree");
			return 0;
		}
	for(int i=0;s[i];i++)
		if(tmp[s[i]]>0) {
			printf("both");
			return 0;
		}
	printf("array");
	return 0;
}
