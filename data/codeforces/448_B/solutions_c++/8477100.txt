#include<cstdio>
#include<cstring>
using namespace std;
char s[101],t[101];
int a[256];
int main()
{
	int i,j,k;
	scanf("%s%s",s,t);
	for(i=0;i<strlen(s);i++)a[s[i]]++;
	for(i=0;i<strlen(t);i++)a[t[i]]--;
	int fl=0;
	for(i=0;i<256;i++)if(a[i]>0)fl=1;else if(a[i]<0){printf("need tree\n");return 0;}
	if(!fl){printf("array\n");return 0;}
	for(i=0,j=0;i<strlen(s)&&j<strlen(t);i++)if(s[i]==t[j])j++;
	if(j>=strlen(t))printf("automaton\n");else printf("both\n");
	return 0;
}
