#include<cstdio>
#include<cstring>
int n,i,j,l1,l2,a[26],b[26];
char s[101],t[101];
int main()
{
    scanf("%s",s);
    scanf("%s",t);
    l1=strlen(s);
    l2=strlen(t);
    for(i=0;i<l1;++i)++a[s[i]-'a'];
    for(i=0;i<l2;++i)++b[t[i]-'a'];
    for(i=0;i<26;++i)
        if(a[i]<b[i]){printf("need tree\n");return 0;}
    for(i=j=0;i<l2&&j<l1;++i,++j)
        while(j<l1&&s[j]!=t[i]) ++j;
    printf("%s\n",i<l2?(l1==l2?"array":"both"):"automaton");
}
