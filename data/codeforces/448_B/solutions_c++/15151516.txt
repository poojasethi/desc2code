#include <cstdio>
#include <cstring>

char s[110],t[110];
int ind=0;
int contains[50];

int main() {
	scanf("%s",s);
	scanf("%s",t);
	
	for(int i=0;s[i];++i) {
		++contains[s[i]-'a'];
		if(s[i]==t[ind] && t[ind]) ++ind;
	}
	if(!t[ind]) printf("automaton");
	else {
		for(int j=0;t[j];++j)
			if(!contains[t[j]-'a']) {
				printf("need tree");
				return 0;
			} else --contains[t[j]-'a'];
		if(strlen(s) != strlen(t))
			printf("both");
		else printf("array");
	}
	return 0;
}
