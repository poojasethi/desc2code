#include<bits/stdc++.h>
using namespace std;
int main()
{
	int a[30]={0},b[30]={0},i,j;
	bool v=0,au=0,sw=0;
	string s,t;
	cin>>s>>t;
	for(i=0;i<s.size();i++)
		a[s[i]-'a']++;
	for(i=0;i<t.size();i++)
		b[t[i]-'a']++;
	for(i=0;i<26;i++)
	{
		if(a[i]!=b[i])
			v=1;
		if(a[i]<b[i])
		{
			cout<<"need tree\n";
			return 0;
		}
		else if(a[i]>b[i])
			au=1;
	}
	for(i=0,j=0;i<s.size();i++)
	{
		if(s[i]==t[j])
			j++;
	}
	if(j!=t.size())
		sw=1;
	if(au && sw)
		cout<<"both\n";
	else if(au)
		cout<<"automaton\n";
	else if(!v)
		cout<<"array\n";
	return 0;
}
