s=raw_input()
r=raw_input()

s_i=''.join(sorted(s))
r_i=''.join(sorted(r))

def subset(a,b):
    i = 0
    for c in a:
        if i < len(b) and c == b[i] : i+= 1
    return i == len(b)
if s_i==r_i: print "array"
elif subset(s,r): print "automaton"
elif subset (s_i,r_i): print "both"
else: print "need tree"
