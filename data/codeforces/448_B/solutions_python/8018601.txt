def sub(x,y):
    for c in y:
        if x:
            x=x[c==x[0]:]
    return x==""
s, t = raw_input(), raw_input()
if sub(t,s):
    print "automaton"
elif sorted(s) == sorted(t):
    print "array"
else:
    if all(s.count(x) >= t.count(x) for x in t):
        print "both"
    else:
        print "need tree"
