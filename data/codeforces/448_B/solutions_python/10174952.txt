s,t=raw_input(),raw_input()
f=lambda x:''.join(sorted(x))
def g(s,t):
    k=-1
    for i in t:
        if s[k+1:].find(i)!=-1:
            k=s[k+1:].find(i)+k+1
        else:
            k=-1;break
    return k
