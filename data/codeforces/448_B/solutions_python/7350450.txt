s = raw_input()
f = raw_input()
tree = 0
ar = 0
aut = 0
for c in 'abcdefghijklmnopqrstuvwxyz':
	a = s.count(c)
	b = f.count(c)
	if a>b: aut = 1
	elif a<b: tree = 1; break

if tree: print 'need tree'; exit(0)

j = 0
for i in xrange(len(f)):
	while j<len(s) and s[j]!=f[i]: j+=1
	if j>len(s): ar = 1; break
	else: j+=1
	
if ar and aut: print 'both'
else:
	if ar: print 'array'
	if aut: print 'automaton'
