a, b = list(raw_input()), list(raw_input())

def check_swap(a, b):
    a.sort()
    b.sort()
    return a == b
    
def auto(a, b):
    for ch in b:
        try:
            p = a.index(ch)
            a = a[p + 1:]
        except:
            return False
    return True
   
def contain(a, b):
    for ch in b:
        if a.count(ch) < b.count(ch): return False
    return True
    
if len(a) == len(b):
    if check_swap(a, b): print "array"
    else: print "need tree"
elif len(a) > len(b):
    if auto(a, b): print "automaton"
    elif contain(a, b): print "both"
    else: print "need tree"
    
else:
    print "need tree"
