s = raw_input()
t = raw_input()

frq = [s.count(chr(x))-t.count(chr(x)) for x in range(ord('a'), ord('z'))]
if min(frq) < 0 or len(s) < len(t):
	print "need tree"
	exit()

swaps, drops = False, False

i = 0
for x in t:
	while i < len(s) and s[i] != x: i += 1
	i += 1

if i > len(s): swaps = True
if len(s) > len(t): drops = True

if swaps and drops: print "both"
elif swaps: print "array"
else: print "automaton"
