#!/usr/bin/env python


def read():
    A = raw_input()
    B = raw_input()
    return A, B


def automaton(A, B):
    idxA = 0
    idxB = 0
    while idxA < len(A) and idxB < len(B):
        if A[idxA] == B[idxB]:
            idxA += 1
            idxB += 1
        else:
            idxA += 1
    return idxB == len(B)


def array(A, B):
    return sorted(A) == sorted(B)


def work((A, B)):
    if automaton(A, B):
        print "automaton"
    elif array(A, B):
        print "array"
    elif automaton(sorted(A), sorted(B)):
        print "both"
    else:
        print "need tree"


if __name__ == "__main__":
    work(read())
