def subseq (s, t):
	j = len(s) - 1
	i = len(t) - 1
	while i>=0 and j>=0:
		if s[j] == t[i]:
			i = i - 1
		j = j - 1
	if i < 0:
		return True
	return False

s = raw_input()
t = raw_input()
S = ''.join(sorted(s))
T = ''.join(sorted(t))
if subseq(s,t):
	print "automaton"
elif S == T:
	print "array"
elif subseq(S,T):
	print "both"
else:
	print "need tree"
