from collections import *

def subsequence(s,t):
  if s=='': return t==''
  if t=='': return True
  if s[0] == t[0]:
    return subsequence(s[1:],t[1:])
  else:
    return subsequence(s[1:],t)

def counter_leq(c,d):
  return all(c[i] <= d[i] for i in c)

s = raw_input()
t = raw_input()
cs = Counter(s)
ct = Counter(t)

if cs == ct:
  print "array"
elif subsequence(s,t):
  print "automaton"
elif counter_leq(ct,cs):
  print "both"
else:
  print "need tree"
