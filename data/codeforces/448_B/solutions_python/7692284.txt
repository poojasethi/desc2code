import re

def subsequence(s, t):
	j = len(s) - 1
	i = len(t) - 1
	while i >= 0 and j >= 0:
		if s[j] == t[i]:
			i -= 1
		j -= 1
	if i < 0:
		return True
	return False

def main():
	s = raw_input()
	t = raw_input()
	s_sorted = ''.join(sorted(s))
	t_sorted = ''.join(sorted(t))
	if subsequence(s, t):
		print "automaton"
	elif s_sorted == t_sorted:
		print "array"
	elif subsequence(s_sorted, t_sorted):
		print "both"
	else:
		print "need tree"

main()
