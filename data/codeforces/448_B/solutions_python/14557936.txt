from collections import Counter
s=raw_input()
t=raw_input()
if len(s)==len(t):
	s=list(s)
	t=list(t)
	s.sort()
	t.sort()
	if s==t:
		print'array'
	else:
		print'need tree'
	exit()
if len(s)<len(t):
	print'need tree'
	exit()
k=0
for i in t:
	k=s.find(i,k)+1
	if not k:
		break
else:
	print'automaton'
	exit()
s=Counter(list(s))
t=Counter(list(t))
if not set(s)>=set(t):
	print'need tree'
	exit()
for i in t:
	if s[i]<t[i]:
		print'need tree'
		exit()
print'both'
