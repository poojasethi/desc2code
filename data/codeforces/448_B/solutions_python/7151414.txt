s = raw_input()
t = raw_input()

def can_automaton(s, t):
    l = [-1]
    for c in t:
        res = s.find(c, l[-1]+1)
        if res == -1:
            return False
        l.append(res)
    return l == sorted(l)

if ''.join(sorted(s)) == ''.join(sorted(t)):
    print "array"
elif can_automaton(s, t):
    print "automaton"
elif can_automaton(''.join(sorted(s)), ''.join(sorted(t))):
    print 'both'
else:
    print 'need tree'
