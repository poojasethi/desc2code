#include <iostream>
using namespace std;
const int MAXD = 1000;
int a[MAXD] = {0};
int main()
{
    int n;
    cin >> n;
    int sum = 0;
    for (int i = 0; i < n; i++)
    {
        int b;
        cin >> b;
        for(int j = 0; j < MAXD; j++)
        {
            int  d = a[j] + 1;
            while (d < 10 && (b - (sum + d - a[j]) < 0 || b - (sum + d - a[j]) > 9*j)) d++;
            if (d<10)
            {
                sum = b - (sum + d -a[j]);
                a[j] = d;
                for (int k = 0; k < j; k++)
                {
                    a[k] = min(sum, 9);
                    sum -= a[k];
                }
                sum = b;
                break;
            }
            sum -= a[j];
        }
        int j = MAXD - 1; while(a[j] == 0) j--;
        for (; j >=0; j--) cout << a[j];
        cout << endl;
    }

    return 0;
}
