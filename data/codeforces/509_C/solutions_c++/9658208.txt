#include<bits/stdc++.h>
using namespace std;

int n,a[999]={0},l=0,b[999]={0};

inline void print(){
    for(int i=l;i>=1;i--)
        cout<<b[i];
    cout<<endl;
}

inline void add(int d){
    for(int i=1;d;i++){
        while(b[i]<9&&d>0)
            d--,b[i]++;
        if(!d)
            l=max(l,i);
    }
}

inline void get(int d){
    if(d>0){
        add(d);
    }
    else{
        for(int i=1;;i++){
            l=max(l,i);
            if(b[i]!=9 && d>0){
                b[i]++,d--;
                add(d);
                break;
            }
            d+=b[i],b[i]=0;
        }
    }
    print();
}

int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        get(a[i]-a[i-1]);
    }

}
