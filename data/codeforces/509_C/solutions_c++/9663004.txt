#include<iostream>
#include<cstdio>
using namespace std;

int a[310];
int res[500];
int len;

void next(int d)
{
	for(int i=1;d>0;i++)
	{
		while (d>0 && res[i]<9)
		{
			d--;
			res[i]++;
		}
		if (d==0)
			len=max(len,i);
	}
}

void solve(int d)
{
	if (d>0)
		next(d);
	else
	{
		for(int i=1;;i++)
		{
			len=max(len,i);
			if (res[i]!=9 && d>0)
			{
				d--;
				res[i]++;
				next(d);
				break;
			}
			d+=res[i];
			res[i]=0;
		}
	}
	for(int i=len;i>=1;i--)
		cout<<res[i];
	cout<<endl;
}
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	for(int i=1;i<=n;i++)
		solve(a[i]-a[i-1]);
	return 0;
}
