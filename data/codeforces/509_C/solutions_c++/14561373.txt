#include <bits/stdc++.h>
using namespace std;
bool can[10000][301];
char ret[10000],
	 pre[10000];
int n;
#define rem (remain-1)/9+1
bool dp[10000][2][302];
bool cons(int ind,int over,int remain){
	if(remain<0)return false;

	if(dp[ind][over][remain])return false;
	
	if(ind==n){
		if(remain!=0)return false;
		if(over){ret[ind]=0;return true;}
		return false;
	}
	int start= pre[ind]-'0';
	if(over)start=0;if(over&&ind==0)start=1;
	for(int i=start;i<=9;++i)
		if(cons(ind+1, (!(i==pre[ind]-'0')) || over ,remain-i) ){
			ret[ind]='0'+i;
			return true;
		}
	dp[ind][over][remain]=true;
	return false;
}
int main(){
	int x,d;cin>>x;int pp=-1;
	for(int i=0;i<x;++i){
		cin>>d;
		memset(dp,false,sizeof(dp));
		int cg = (d-1)/9 +1;
		if(pp<cg)
			n = cg,	cons(0,1,d);
		else{
			n=pp;
			if(!cons(0,0,d)){
				++n;
				memset(dp,false,sizeof(dp));
				cons(0,1,d);
			}
		}
		for(int j=0;j<n;++j)
			pre[j]= ret[j];
		ret[n]=0;
		puts(ret);
		pp = n;
	}
}
