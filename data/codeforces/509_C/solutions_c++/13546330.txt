#include<bits/stdc++.h>
using namespace std;

int posisi[400],d,d1,t;

//d=banyaknya digit
//d1=posisi digit paling kanan yang bukan 9

void tulis(){
	int i;
	for(i=d;i>=0;i--){
		printf("%d",posisi[i]);
	}
	printf("\n");
}

void cmd1(int x){
	x-=t;
	int l;
	while(x>0){
		if(posisi[d1]<9){
			l=posisi[d1];
			posisi[d1]=min(9,x+posisi[d1]);
			x=max(x+l-9,0);
		}
		else d1++;
	}
	if(d1>d)d=d1;
}

void cmd2(int x){
	int y=0,z=d,i;
	bool per=true;
	while(y<x){
		y+=posisi[z];
		z--;
	}
	y=y-posisi[z+1]+1;
	z+=2;
	while(posisi[z]==9){
		z++;
		y-=9;
	}
	posisi[z]++;
	if(z>d)d=z;
	y=x-y;
	for(i=0;i<z;i++){
		posisi[i]=min(9,y);
		y-=posisi[i];
		if(posisi[i]!=9&&per){
			d1=i;
			per=false;
		}
//		printf("%d ",d1);
//		tulis();
	}
}

int main(){
	int n,x;
	scanf("%d",&n);
	while(n--){
		scanf("%d",&x);
		if(x>t)cmd1(x);
		else cmd2(x);
		t=x;
		tulis();
	}
}
