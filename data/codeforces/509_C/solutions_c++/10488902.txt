#include <cstdio>
#include <cstring>
#include <algorithm>
char t[3300];
int add(int pos,int num){
	int r=num;
	t[pos]+=num;
	for(int i=pos;t[i]>=10;i++){
		t[i]=0;
		t[i+1]++;
		r-=9;
	}
	return r;
}
int main(){
	int n,sum=0;
	scanf("%d",&n);
	memset(t,0,sizeof(t));
	for(int ci=0;ci<n;ci++){
		int tgt;
		scanf("%d",&tgt);
		sum+=add(0,1);
		for(int i=0;sum>tgt;i++){
			sum+=add(i,10-t[i]);
		}
		for(int i=0;sum<tgt;i++){
			int a=std::min(9-t[i],tgt-sum);
			sum+=a;
			t[i]+=a;
		}
		bool p=false;
		for(int i=3200;i>=0;i--){
			if(p||t[i]){
				p=true;
				putchar('0'+t[i]);
			}
		}
		putchar('\n');
	}
	return 0;
}
