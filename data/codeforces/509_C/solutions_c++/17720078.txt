#include <bits/stdc++.h>
using namespace std;
const int maxn = 666;
long long a[maxn], b[maxn];

int main()
{
    int n, i, j;
    scanf("%d",&n);
    for(int i=1; i<=n; i++)
    {
        scanf("%d",&b[i]);
        long long  d = b[i] - b[i-1];
        long long x;
        if(d <= 0)
        {
            d = 1 - d;
            x = 0;
            for(j=600; j>=0; j--)
            {
                if(x < d || a[j] == 9)
                {
                    x += a[j];
                    a[j] = 0;
                }
                else
                    break;
            }
            d = x - d;
            a[j]++;
        }
        for (j=600; j>=0; j--)
        {
            x = min(a[j]+d, (long long)9);
            d -= (x - a[j]);
            a[j] = x;
        }
        j=0;
        while(!a[j])
            j++;
        for(; j<=600; j++)
            printf("%d",a[j]);
        printf("\n");
    }
    return 0;
}
