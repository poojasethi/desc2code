#include <iostream>
using namespace std;
int a[500],b[500];

int main() {
    int n;
    cin>>n;
    for(int i=0;i<n;i++) {
        cin>>b[i];
    }
    int pre=0,sz=0;
    for(int i=0;i<n;i++) {
        int d=b[i]-pre;
        int l=0;
        while(1) {
            if(l==sz) sz++;
            if(a[l]<9&&d>=1&&d<=9*l+9-a[l]) {
                d--,a[l]++;
                int q=0;
                while(q<l) {
                    if(d<=9) {
                        a[q]=d;
                        break;
                    }
                    a[q]=9;
                    d-=9;
                    q++;
                }
                if(q==l) a[q]+=d;
                break;
            }
            d+=a[l];
            a[l]=0;
            l++;
        }
        pre=b[i];
        for(int j=sz-1;j>=0;j--) cout<<a[j];
        cout<<endl;
    }
    return 0;
}
