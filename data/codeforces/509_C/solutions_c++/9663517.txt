#include <stdio.h>
#include <iostream>
#include <cstring>
using namespace std;
int n,z,x,f[505]={0},l=0,t,k;
inline void print(){
	for (int i=l;i>0;i--)
		printf("%d",f[i]);
	printf("\n");
}
void make(int t){
	for (int i=1;t>0;i++){
		while (f[i]<9 && t>0)
			f[i]++,t--;
		if (t==0 && i>l) l=i;
	}
}
int main( ){
	scanf("%d %d",&n,&z);
	make(z),print();
	for (int i=2;i<=n;i++){
		scanf("%d",&x);
		if (x-z>0){
			make(x-z);
			print();
		}
		else{
			t=x-z,k=1;
			while (true){
				if (k>l) l=k;
				if (f[k]!=9 && t>0){
					f[k]++,t--;
					make(t);
					break ;
				}
				t+=f[k],f[k]=0;
				k++;
			}
			print();
		}
		z=x;
	}
	return 0;
}
