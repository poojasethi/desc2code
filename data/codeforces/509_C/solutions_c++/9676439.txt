#include<stdio.h>
#include<cstring>
#include<iostream>
#include<cstdlib>
#include<string>
#include<map>
#include<utility>
#include<algorithm>
using namespace std;
int a[5012],length=0;
int main(){
	int n;
	int b;
	int pre=0;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&b);
		int dist=b-pre;
		int ind=0;
		while(1){
			if(ind>length)	length++;
			if(dist>=1 && a[ind]<9 && dist<=ind*9+9-a[ind]){
				a[ind]++,dist--;
				int x=0;
				while(x<=ind){
					if(dist>9){
						a[x]=9;
						dist-=9;
					}
					else{
						a[x]+=dist;
						dist=0;
						break;
					}
					x++;
				}
				break;
			}
			dist+=a[ind];
			a[ind]=0;
			ind++;
		}
		pre=b;
		for(int i=length;i>=0;i--)
			printf("%d",a[i]);
		puts("");
	}
	return 0;
}
