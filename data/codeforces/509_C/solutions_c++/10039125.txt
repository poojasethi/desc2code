#include <iostream>
#include <cstdio>
using namespace std;

int n, l, d, a[305], ans[1005];

void f(int p) {
    int i, k;
    for (i = 0; p > 0; i++) {
        k = min(p, 9 - ans[i]);
        ans[i] += k;
        p -= k;
    }
    l = max(i, l);
}

int main()
{
    int i, j;
//    freopen ("input.txt", "r", stdin);
    cin >> n;
    for (i = 1; i <= n; i++) {
        scanf ("%d", &a[i]);
        d = a[i] - a[i - 1];
        if (d > 0) {
            f(d);
        } else {
            for (j = 0; ans[j] == 9 || d <= 0; j++) {
                d += ans[j];
                ans[j] = 0;
            }
            ans[j]++;
            if (j == l) l++;
            f(d - 1);
        }
        for (j = l - 1; j >= 0; j--) {
            printf("%d", ans[j]);
        }
        puts("");
    }
}
