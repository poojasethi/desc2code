#include<bits/stdc++.h>
using namespace std;
int n,x,sz;
int s[1005];
void out() {
	for(int i = sz; i >= 1; i--) printf("%d",s[i]);
    printf("\n");
}
void add(int P) {
	int id = 1;
	while(P) {
		if (s[id] == 9) id++;
		else s[id]++,P--;
	}
	sz = max(sz, id);
}
void add2(int P) {
	int id = 1;
	while(P <= 0) {
		P += s[id];
		s[id] = 0;
		++id;
	}
	while(s[id] == 9) {
		P += s[id];
		s[id] = 0;
		++id;
	}
	s[id]++; P--;
	sz = max(sz, id);
	add(P);
}
int main() {
	scanf("%d",&n); sz = 0;
	int prev = 0;
	for(int i = 1; i <= n; i++) {
		scanf("%d",&x);
		if (x - prev > 0) add(x - prev);
		else add2(x - prev);
		out();
		prev = x;
	}
}
