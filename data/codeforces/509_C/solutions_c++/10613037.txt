

#include <stdio.h>
#include <iostream>
#include <math.h>
#include <algorithm>
using namespace std;
int l, a[500], b1, b2;
void solve1(int x){
    int i = 1;
    while (x){
        if (a[i] == 9){
			i ++;
		}
        else{
            a[i] ++;
			x --;
        }
        l = max(l, i);
    }
}
void solve2(int x){
    int i = 1;
    while (x <= 0){
        x += a[i];
        a[i ++] = 0;
        l = max(l, i);
    }
    while (a[i] == 9){
        x += a[i];
        a[i ++] = 0;
        l = max(l, i);
    }
    a[i] ++;
	x --;
    solve1(x);
}
int main(){
    int i, j, n;
    l = 1;
    scanf("%d%d", &n, &b1);
		solve1(b1);
   		for (i = l; i >= 1; i --){
			printf("%d",a[i]);
		}
		puts("");
		for(i = 2 ;i <= n; i ++){
			scanf("%d", &b2);
			int x = b2 - b1;
			b1 = b2;
			if (x > 0){
				solve1(x);
			}
			else{
				solve2(x);
			}
			for (j = l; j >= 1; j --){
		 		printf("%d",a[j]);
			}
			puts("");
		}
    return 0;
}
