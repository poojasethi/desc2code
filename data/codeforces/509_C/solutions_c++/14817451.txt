#include <bits/stdc++.h>
#define N 123456

using namespace std;

int n, b[N];
string tr, s;

void inc(int p){
  for (int i = s.size() - 1; i >= 0 && p > 0; --i){
    int c = 57 - s[i];
    s[i] = min(57, s[i] + p);
    p -= c;
  }
  while (p > 0){
    s.insert(s.begin(), min(57, p + 48));
    p -= 9;
  }
}

void process(int p){
  if (p > 0) inc(p);
  else{
    for (int i = s.size() - 1; i >= 0; --i){
      if (s[i] != '9' && p > 0){
        ++s[i];
        inc(--p);
        return;
      }
      p += (s[i] - 48);
      s[i] = '0';
      if (i == 0) s.insert(s.begin(), '0'), ++i;
    }
  }
}

int main()
{
  ios_base::sync_with_stdio(false);
  //freopen("c.in","r",stdin);
  //freopen("c.out","w",stdout);
  cin >> n;
  s = "";
  for (int i = 1; i <= n; ++i)
  {
    cin >> b[i];
    process(b[i] - b[i - 1]);
    cout << s << endl;
  }
}
