#include<stdio.h>
#define ll long long int

int last[609] = {0}, lastin =1;

int add(){
	int i =0, count=0;
	while( i < 607 ){
		last[i+1] += last[i]/10;
		last[i]%=10;
		count += last[i];
		if( last[i] )
			lastin = i;
		i++;
	}
	lastin += 1;
	return count;
}
