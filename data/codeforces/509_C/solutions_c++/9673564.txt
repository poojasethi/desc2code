#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

const int maxn = 310;

int a[maxn],b[3100],n,nowt;

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;++i) scanf("%d",a+i);
	for (int i=1;i<=n;++i)
	{
		int t = 0,flag = 1,tflag = 0;
		for (int j=nowt;j>0;--j)
		{
			t += b[j];
			if (t >= a[i])
			{
				flag = tflag = j + 1;
				t -= b[j];
				while (b[flag] == 9) t -= b[flag],flag++;
				break;
			}
			else if (j == 1) 
			{
				b[j] += a[i] - t;
				while (b[j] >= 10) b[j + 1] += b[j] - 9,b[j++] = 9;
				flag = 0;
				break;
			}
		}
		if (flag)
		{
			int j = 1;
			b[j] = a[i] - t - 1;
			b[flag] ++;
			for (int k=2;k<flag;++k) b[k] = 0;
			while (b[j] >= 10) b[j + 1] = b[j] - 9,b[j++] = 9;
		}
		while (b[nowt + 1]) nowt++;
		for (int j=nowt;j>0;--j) printf("%d",b[j]);
		printf("\n");
	}
	return 0;
}
