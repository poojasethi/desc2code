int arr[MAX];
int len;
int ans[MAX];
void add(int x){
	int i=0;
	while(x>0){
		while(ans[i]<9&&x>0){
			ans[i]++;
			x--;
		}
		if(!x)len=max(len,i);
		i++;
	}
}

void func(int t){
	if(t>0){
		add(t);
	}
	else{
		int j=0;
		while(1){
			if(ans[j]!=9&&t>0){
				ans[j]++;
				t--;
				if(t)add(t);
				len=max(len,j);
				break;
			}
			t+=ans[j];
			ans[j]=0;
			j++;
		}
	}
	for(int k=len;k>=0;k--){
		cout<<ans[k];
	}
	cout<<"\n";
}
