#include <iostream>
#include <cstdio>
using namespace std;

int n, l, d, s, a[305], ans[1005];

void f(int p) {
    int i, k;
    for (i = 0; p > 0; i++) {
        k = min(p, 9 - ans[i]);
        ans[i] += k;
        p -= k;
    }
    l = max(i, l);
}

int main()
{
    int i, j;
//    freopen ("input.txt", "r", stdin);
    cin >> n;
    for (i = 1; i <= n; i++) {
        scanf ("%d", &a[i]);
        d = a[i] - a[i - 1];
        s = 0;
        if (d > 0) {
            f(d);
        } else {
            d--;
            for (j = 0; j < l; j++) {
                s += ans[j];
                ans[j] = 0;
                if (d + s >= 0 && ans[j + 1] != 9) {
                    break;
                }
            }
            if (j < l) ans[j + 1]++;
            else ans[j]++;
            if (j >= l - 1) l++;
            f(s + d);
        }

        for (j = l - 1; j >= 0; j--) {
            printf("%d", ans[j]);
        }
        puts("");
    }
}
