#include <cstdio>
#include <cstdlib>
#include <iostream>
using namespace std;
int n, b[999], c[999];
int ln;
void print() {
    for (int i = ln; i >= 1; i--)
        printf("%d", c[i]);
    printf("\n");
}
void Add(int d) {
    for (int i = 1; d; i++) {
        while (d && c[i] < 9)
            d--, c[i]++;
        if (d == 0)
            ln = max(ln, i);
    }
}
void Get(int d) {
    if (d > 0)
        Add(d);
    else {
        for (int i = 1;; i++)  {
            ln = max(ln, i);
            if (c[i] != 9 && d > 0) {
                c[i]++; d--;
                Add(d);
                break;
            }
            d += c[i], c[i] = 0;
        }
    }
    print();
}
int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; i++)
        scanf("%d", b + i);
    for (int i = 1; i <= n; i++)
        Get(b[i] - b[i - 1]);
    return 0;
}
