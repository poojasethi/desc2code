#include <iostream>
#include <vector>

using namespace std;

int cur = -1, prev = -1;
int l = 0;
int res[500];

int process(int d){
    int pos = 0;
    while(res[pos] == 9 || d <= 0){
        d += res[pos];
        res[pos] = 0;
        pos++;
    }
    res[pos]++;
    d--;
    l = max(l, pos+1);
    return d;
}

void solve(int d){
    if(d <= 0){
        d = process(d);
    }
    int pos = 0;
    while(d > 0){
        if(d >= 9 - res[pos]){
            d -= 9 - res[pos];
            res[pos] = 9;
        }else{
            res[pos] += d;
            d = 0;
        }
        pos++;
    }
    l = max(l, pos);
    for(int i = l-1; i >= 0; i--){
        cout << res[i];
    }
    cout << endl;
}

int main() {
    int n;
    int cur, prev;
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> cur;
        if(i){
            solve(cur-prev);
        }else{
            solve(cur);
        }
        prev = cur;
    }
    return 0;
}
