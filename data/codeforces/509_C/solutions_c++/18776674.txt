#include <iostream>
#include <cstdio>
int f[10000],g[10000],a[310],i,j,n;
void work(int x,int y)
{
	int i,j=0;
	if (x==y)
	{
		++g[1];
		for (i=1;g[i]>=10;++g[++i]) g[i]-=10;
		if (i>g[0]) g[0]=i;
		x=0;
		for (i=1;i<=g[0];++i) x+=g[i];
	}
	if (x>y)
	{
		for (i=1;i<=g[0] && x+1-j>y;++i) j+=g[i];
		for (;g[i]==9;++i);
		if (i>g[0]) g[0]=i;
		for (++g[i--];i;--i) g[i]=0;
		x=0;
		for (i=1;i<=g[0];++i) x+=g[i];
	}
	for (i=1;x<y;++i)
		if (x+9-g[i]>=y) g[i]+=y-x,x=y;
			else x+=9-g[i],g[i]=9;
	if (i-1>g[0]) g[0]=i-1;
}
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%d\n",&n);
	for (i=1;i<=n;++i) scanf("%d\n",&a[i]);
	for (i=1;i<=n;++i)
	{
		for (j=0;j<=f[0];++j) g[j]=f[j];
		work(a[i-1],a[i]);
		for (j=0;j<=g[0];++j) f[j]=g[j];
		for (j=g[0];j;--j) printf("%d",f[j]);printf("\n");
	}
}
