#include <iostream>
#include <cstdio>
using namespace std;

int n, len, b[305], s[1005];

void f(int p) {
    int i;
    for (i = 1; p > 0; i++) {
        while (s[i] < 9 && p > 0) {
            s[i]++;
            p--;
        }
    }
    len = max(len, i - 1);
}

int main() {
    int i, j, d;
//    freopen("input.txt", "r", stdin);
    cin >> n;
    for (i = 1; i <= n; i++) {
        scanf("%d", &b[i]);
        d = b[i] - b[i - 1];
        if (d > 0) {
            f(d);
        } else {
            for (j = 1; ; j++) {
                len = max(len, j);
                if (s[j] != 9 && d > 0) {
                    s[j]++;
                    d--;
                    f(d);
                    break;
                }
                d += s[j];
                s[j] = 0;
            }
        }
        for (j = len; j > 0; j--) {
            printf("%d", s[j]);
        }
        puts("");
    }
}
