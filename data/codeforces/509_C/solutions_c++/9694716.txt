
#include<bits/stdc++.h>
using namespace std;
int L,a[500],b1,b2;
void add1(int x)
{
    int i=1;
    while(x)
    {
        if(a[i]==9) ++i;
        else
        {
            ++a[i],--x;
        }
        L=max(L,i);
    }
}
void add2(int x)
{
    int i=1;
    while(x<=0)
    {
        x+=a[i];
        a[i++]=0;
        L=max(L,i);
    }
    while(a[i]==9)
    {
        x+=a[i];
        a[i++]=0;
        L=max(L,i);
    }
    ++a[i],--x;
    add1(x);
}

void output()
{
    for(int i=L;i>=1;--i)
        printf("%d",a[i]);
    puts("");
}
int main()
{
    int n;
    L=1;
    scanf("%d%d",&n,&b1);
    add1(b1);
    output();
    for(int i=2;i<=n;i++)
    {
        scanf("%d",&b2);
        int x=b2-b1;
        b1=b2;
        if(x>0) add1(x);
        else add2(x);
        output();
    }
    return 0;
}
