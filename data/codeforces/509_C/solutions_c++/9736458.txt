#include<bits/stdc++.h>

using namespace std;

int n,a[999],b[999],l=0;

void print()
{
    for(int i=l; i>0; i--)cout<<b[i];
    cout<<endl;
}

void fill(int d)
{
    for(int i=1;; i++)
    {
        while(b[i]<9 and d>0)b[i]++,d--;
        l=max(l,i);
        if(d==0)break;
    }

}

void solve(int d)
{
    if(d>0)fill(d);
    else
    {
        for(int i=1;; i++)
        {
            if(d>0 and b[i]<9)
            {
                b[i]++;
                d--;
                l=max(l,i);
                fill(d);
                break;
            }
            d+=b[i];
            b[i]=0;
        }
    }
    print();
}


int main()
{
    //freopen("a.txt","r",stdin);

    cin>>n;
    a[0]=0;
    for(int i=1; i<=n; i++)
    {
        cin>>a[i];
        solve(a[i]-a[i-1]);
    }







}
