#include<cstdio>
#include<algorithm>
using namespace std;

int tc[305];
int num[350];
int l = 0;

void next(int dif){
    int id;
    for(id=1; dif>0; id++){
        l = max(l,id);
        while(dif>0 && num[id]<9){
            dif--;
            num[id]++;
        }
    }
}

void ans(int dif){
    int id;
    if(dif>0) next(dif);
    else{
        for(id=1; ; id++){
            l = max(l, id);
            if(dif>0 && num[id]<9){
                dif--;
                num[id]++;
                next(dif);
                break;
            }
            dif += num[id];
            num[id] = 0;
        }
    }
}

int main(){
    int n, i, j;
    scanf("%d", &n);
    for(i=0; i<n; i++){
        scanf("%d", &tc[i]);
        if(i == 0) ans(tc[i]);
        else ans(tc[i]-tc[i-1]);
        for(j=l; j>0; j--) printf("%d", num[j]);
        printf("\n");
    }
    return 0;
}
