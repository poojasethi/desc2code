#include <bits/stdc++.h>
using namespace std;
int a[999]={0},b[999]={0},n;
int l=0;
void print()
{
for(int i=l;i>=1;i--)
cout<<b[i];
cout<<endl;
}

void fill(int d)
{
for(int i=1;d;i++)
{
while(b[i]<9 && d>0)
b[i]++,d--;
if(d==0)
l=max(l,i);
}
}


void get(int d)
{
if(d>0)
fill(d);
else
{
for(int i=1;;i++)
{
l=max(l,i);
if(b[i]!=9 && d>0){
b[i]++,d--;
fill(d);
break;
}
d+=b[i],b[i]=0;
}
}
print();
}
int main() {
    int i,j,k;

  {  cin>>a[i];
