#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;
int a[1010],res[1010];
int n,w;
void next(int d)
{
    for (int i=1; d>0; i++)
    {
        while (d>0 && res[i]<9)
        {
            d--;
            res[i]++;
        }
        if (d==0) w=max(w,i);
    }
}
void work(int d)
{
    if (d>0) next(d);
    else
    {
        for (int i=1; ; i++)
        {
            w=max(w,i);
            if (res[i]!=9 && d>0)
            {
            	d--;
                res[i]++;
                next(d);
                break;
            }
            d+=res[i];
			res[i]=0;
        }
    }
    for (int i=w; i>=1; i--) printf("%d",res[i]);
    printf("\n");
}
int main()
{
    scanf("%d",&n);
    for (int i=1; i<=n; i++) scanf("%d",&a[i]);
    for (int i=1; i<=n; i++) work(a[i]-a[i-1]);
    return 0;
}
