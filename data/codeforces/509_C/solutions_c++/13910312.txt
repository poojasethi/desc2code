#include <stdio.h>        
#include <string.h>    
#include <iostream>    
#include <algorithm>

using namespace std;

int n,len;
int a[400],b[400];
void get(int num){
	for(int i=1; num; i++){
		while(a[i] < 9 && num){
			a[i]++;
			num--;
		}
		if(len < i && !num) len = i;
	}
}

void print(){
	for(int i=len; i>=1; i--)
		printf("%d",a[i]);
	puts("");
}

int main(){
	scanf("%d",&n);
	for(int i=1; i<=n; i++)
		scanf("%d",b + i);
	get(b[1]);
	print();
	for(int i=2; i<=n; i++){
		int gap = b[i] - b[i-1];
		if(gap > 0){
			get(gap);
			print();
		}
		else{
			int k = 1;
			while(1){
				if(k > len) len = k;
				if(a[k] < 9 && gap > 0){
					a[k]++;
					get(gap-1);
					print();
					break;
				}
				gap += a[k];
				a[k] = 0;
				k++;
			}
		}
	}
	return 0;
}
