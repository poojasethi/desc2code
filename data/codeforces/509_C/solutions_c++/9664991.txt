#include <bits/stdc++.h>
using namespace std;
int n;
int a[300+3],ans[10000];
int length;
void thuchien(int x)
{
	for(int i=0;x>0;++i)
	{
		while(ans[i]<9&&x>0)
		{
			ans[i]++; x--;
		}
		if(!x) length=max(length,i);
	}
}
int work(int x)
{
	if(x>0) thuchien(x);
	else{
		for(int t=0;;t++)
		{
			if(ans[t]!=9&&x>0)
			{
				x--; ans[t]++;
				thuchien(x);
				length=max(length,t); break;
			}
			x+=ans[t]; ans[t]=0;
		}
	}
		for(int i=length;i+1;--i)
			cout << ans[i];
		cout << endl;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin >> n;
	for(int x=0;x<n;++x) { cin >> a[x];} length=0;
	for(int x=0;x<n;++x) if(x) work(a[x]-a[x-1]); else work(a[x]);
}
