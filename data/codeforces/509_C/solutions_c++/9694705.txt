
#include<bits/stdc++.h>
using namespace std;
int L,a[500],b1,b2;
void add1(int x)
{
    int i=1;
    while(x)
    {
        if(a[i]==9) ++i;
        else
        {
            ++a[i],--x;
        }
        L=max(L,i);
    }
}
void add2(int x)
{
    int i=1;
    while(x<=0)
    {
        x+=a[i];
        a[i++]=0;
        L=max(L,i);
    }
    while(a[i]==9)
    {
        x+=a[i];
        a[i++]=0;
        L=max(L,i);
    }
    ++a[i],--x;
    add1(x);
}

void print()
{
    for(int i=L;i>=1;i--)
        printf("%d",a[i]);
    printf("\n");
}
int main()
{
    int n;
    cin>>n;
    cin>>b1;
    add1(b1);
    print();
    for(int i=2;i<=n;i++)
    {
        cin>>b2;
        int t=b2-b1;
        b1=b2;
        if(t>0)
            add1(t);
        else
        {
          add2(t);
        }
        print();
    }
    return 0;
}
