#include<cstdio>
#include<algorithm>
using namespace std;

int n, x, tot;
int v[555];

int main() {
	scanf("%d", &n);
	tot = 0; //枚举到当前位时前面所有东西的和
	for(int i=0;i<n;i++) {
		scanf("%d", &x);
		bool found = false;
		for(int p=0;!found;p++) {
			tot -= v[p];
			for(int j = v[p]+1;j<10;j++) {
				if (tot+j > x) continue;
				if (x-(tot+j) > 9*p) continue;
				
				v[p] = j;
				int rem = x-(tot+j);
				for(int k = 0; k<p;k++) {
					int det = min(9, rem);
					v[k] = det;
					rem -= det;
				}
				found = true;
				break;
			}
		}
		tot = x;
		
		int p = 500;
		while(!v[p]) p--;
		for(;p>=0;p--) printf("%d", v[p]);
		printf("\n");
	}
	
	return 0;
}
