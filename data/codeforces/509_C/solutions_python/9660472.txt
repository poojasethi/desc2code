def getnum(minimum, weight):
	minnumwidth = len(str(minimum))
	minimumAsList = map(int, list(str(minimum)))
	while True:
		for i in reversed(xrange(len(minimumAsList))):
			if minnumwidth == len(minimumAsList):
				minimumAsList[i] += 1
			while minimumAsList[i] <= 9:
				rangemin = 0
				rangemax = 9*(len(minimumAsList) - i - 1)
				if weight <= sum(minimumAsList[:i + 1]) + rangemax and weight >= sum(minimumAsList[:i + 1]):
					toFill = weight - sum(minimumAsList[:i + 1])
					for j in reversed(xrange(len(minimumAsList))):
						while minimumAsList[j] < 9 and toFill > 0: 
							minimumAsList[j] += 1
							toFill -= 1
					return "".join(map(str, minimumAsList))
				minimumAsList[i] += 1
			minimumAsList[i] = 0
		minimumAsList = [1] + [0]*len(minimumAsList)
