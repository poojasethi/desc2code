#include<iostream>
#include<string>
#include<vector>
#include<cstring>
#define pb push_back
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
string s1,s2,s3;
int l1,l2,l3;
struct node{
	bool f;
	short dep;
	int son[26];
}OsO4;
int tot=1,ans;
node pnt[2000006];
bool fns[2333];
inline void getans(int cur=0){
	if(pnt[cur].dep+l3>=l2)
		ans+=pnt[cur].f;
	rep(i,26)
		if(pnt[cur].son[i])
			getans(pnt[cur].son[i]);
	return;
}
inline void build(int cur,int ps,int d=0){
	if(fns[ps])
		pnt[cur].f=1;
	pnt[cur].dep=d;
	if(ps==l1)
		return;
	int nxt=s1[ps]-'a';
	if(!pnt[cur].son[nxt]){
		pnt[cur].son[nxt]=tot;
		pnt[tot++]=OsO4;
	}
	build(pnt[cur].son[nxt],ps+1,d+1);
	return;
}
int main(){
	pnt[0]=OsO4;
	cin>>s1>>s2>>s3;
	l1=(int)s1.size();
	l2=(int)s2.size();
	l3=(int)s3.size();
	rep(i,l1-l3+1)
		if(s1.substr(i,l3)==s3)
			fns[i]=1;
	rep(i,l1-l2+1)
		if(s1.substr(i,l2)==s2)
			build(0,i);
	getans();
	cout<<ans<<endl;
	return 0;
}
