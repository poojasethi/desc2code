#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
#include <string>
#include <iostream>
using namespace std;

string a, b, c;
string w[2005];

int main(void) {
	cin >> a >> b >> c;
	int n = (int)a.size();
	a = a+"\0";
	for(int i=0;i<(int)a.size();i++)
		w[i]=a.substr(i);
	sort(w, w+n);
	/*for(int i=0;i<n;i++)
		printf("%s\n", w[i].c_str());*/
	int cnt=0;
	for(int i=0;i<n;i++) {
		int last=0;
		if(i>0) for(last=0;last<(int)w[i].size() && last<(int)w[i-1].size();last++) if(w[i][last]!=w[i-1][last]) break;
		if(w[i].size()>=b.size() && w[i].substr(0,(int)b.size()) == b) {
			for(int j=max(0, max((int)b.size()-(int)c.size(), last-(int)c.size()+1));j+(int)c.size()<=(int)w[i].size();j++)
			{
				int f=0;
				for(int x=0;x<(int)c.size() && !f;x++) if(w[i][j+x]!=c[x]) f=1;
				cnt += !f;
				//cnt += (w[i].substr(j, (int)c.size())==c);
			}
		}
	}
	cout << cnt << endl;
	return 0;
}
