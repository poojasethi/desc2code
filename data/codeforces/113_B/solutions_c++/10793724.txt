#include <bits/stdc++.h>
using namespace std;

string st, st1, st2;
bool kata1[2005], kata2[2005];
map<long long, bool> mp;
vector<long long> v;
long long temp;
int len;

int main(){
	cin >> st;
	cin >> st1;
	cin >> st2;
	memset(kata1, false, sizeof kata1);
	memset(kata2, false, sizeof kata2);
	len = st.size();
	int maks = max(st1.length(), st2.length());
	for(int i = 0 ; i < len; i++){
		if(st.substr(i, st1.size()) == st1) kata1[i] = true;
		if(st.substr(i, st2.size()) == st2) kata2[i + st2.size() - 1] = true;
	}
	for(int i = 0 ; i < len; i++){
		if(!kata1[i]) continue;
		temp = 0;
		for(int j = i ; j < st.size(); j++){
			temp *= 31;
			temp += (st[j] - 'a' + 1);
			if(kata2[j] && (j - i + 1) >= maks) v.push_back(temp);
		}
	}
	sort(v.begin(), v.end());
	v.resize(unique(v.begin(), v.end())- v.begin());
	printf("%d\n", v.size());
}
