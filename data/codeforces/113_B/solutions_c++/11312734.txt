#include<bits/stdc++.h>
using namespace std;
int ans=0,final=1,len1,len2;
int visited[2010*1010][26],arr1[2005],arr2[2005];
char str[2005],start[2005],end[2005];
void find(int pos){
    int current=0;
    for(int i=pos;str[i];i++)
    {
        //printf("current%d\n",current );
        if(!visited[current][str[i]-'a']) 
        {
            if(arr2[i]&&i>=len1+pos-1&&i>=len2+pos-1) ans++;
            visited[current][str[i]-'a']=final++;
        }
        current=visited[current][str[i]-'a'];
        //printf("ans%d final%d\n",ans,final);
    }
}
int main(){
    scanf("%s%s%s",str,start,end);
    len1=strlen(start);
    len2=strlen(end);
    for(int i=0;str[i];i++)
    {
        //strncmp returns 0 if content is equal
        if(!strncmp(str+i,start,len1)) arr1[i]=1;
        if(!strncmp(str+i,end,len2)) arr2[i+len2-1]=1;
    }
    for(int i=0;str[i];i++) if(arr1[i]) find(i);
    printf("%d\n",ans);
return 0;
}
