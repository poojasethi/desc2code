# -*- coding: utf8 -*-
import sys

if __name__ == '__main__':
    n = int(sys.stdin.readline())
    s = sys.stdin.readline()
    t = 0
    for c in s:
        if c == 'T': t += 1
    m = sys.maxint
    for i in xrange(n):
        k = 0
        for j in xrange(t):
            if s[(i + j) % n] == 'H':
                k += 1
        if k < m:
            m = k
    print m
