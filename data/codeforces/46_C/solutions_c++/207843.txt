#include <cstdio>
#include <algorithm>
#include <cstring>

int n;
char s[1100];

int main()
{
	scanf("%d\n", &n);
	gets(s);
	int h = 0;
	for (int i = 0; i < n; i++)
		if (s[i] == 'H')
			h++;
	int ans = n + 1;
	for (int i = 0; i < n; i++)
	{
		int cans = 0;
		for (int j = 0; j < h; j++)
			cans += s[(i + j) % n] == 'T';
		ans = std::min(ans, cans);
	}
	printf("%d", ans);
}
