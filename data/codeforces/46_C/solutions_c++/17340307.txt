#include<bits/stdc++.h>

using namespace std;

#define N 1003

int P[N],n,i,mn = N;

char s[N];

int main(){

	cin >> n >> s + 1;

	for(i=1;i<=n;i++)	
		P[i] = P[i-1] + (s[i] == 'H');

	for(i=1;i<=n;i++){

		if(i >= P[n])
			mn = min(mn , P[n] - (P[i] - P[i-P[n]]));

		if(i >= n-P[n])
			mn = min(mn , n-P[n] - ((n-P[n]) - (P[i] - P[i-(n-P[n])])));
	}

	cout << mn;
	return 0;
}
