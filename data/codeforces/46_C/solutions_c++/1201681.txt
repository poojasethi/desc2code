#include <stdio.h>
int main()
{
	int n,i,j;
	char q;
	scanf("%d%c",&n,&q);
	char a[n],b[2*n];
	int k=0;
	for(i=0;i<n;i++)
	{
		scanf("%c",&a[i]);
		b[i]=a[i];
		if(a[i]=='H')k++;
	}
	for(;i<2*n;i++)
	{
		b[i]=b[i-n];
	}
	int o=0;
	for(i=0;i<2*n-k;i++)
	{	
		int u=0;
		for(j=i;j<i+k;j++)
		{
			if(b[j]=='H')u++;
		}
		if(u>o)
		{
			o=u;
		}
	}
	printf("%d\n",k-o);
	return 0;
}
