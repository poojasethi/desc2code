#include<cstdio>
int main()
{
	int n,h=0,i,j,k,t,m;
	char s[1001]={},b;
	scanf("%d",&n);
	scanf("%c",&b);
	for(i=0;i<n;i++)
	{
		scanf("%c",&s[i]);
		if(s[i]==72) h+=1;
	}
	m=h;
	for(j=0;j<n;j++)
	{
		t=0;
		for(k=0;k<h;k++)
		{
			if(s[(j+k)%n]==84) t+=1;
		}
		if(m>t) m=t;
	}
	printf("%d\n",m);
	return 0;
 }
