#include<iostream>
#include<string>
using namespace std;
const int INF = 1000*1000;
int n;
string inp;

int main()
{
	cin >> n >> inp;
	int t=0;
	for(int i=0;i<n;i++)
		if(inp[i]=='T')
			t++;

	int ans = INF;
	int temp=0;
	for(int i=0;i<n;i++,temp=0)
	{
		for(int j=0;j<t;j++)
			if(inp[(i+j)%n] =='H' && ++temp)
				;
		ans = min(ans,temp);
	}
	cout << ans << endl;
	return 0;
}
