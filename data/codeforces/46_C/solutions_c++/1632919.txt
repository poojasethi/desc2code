#include <stdio.h>
int main() {
	int n;
	char str[1100];
	int s,count[1100],i,j,c;
	scanf("%d",&n);
	scanf("%s",str);
	c=0;
	for(i=0;i<n;i++) count[i]=0;
	for(i=0;i<n;i++) {if(str[i]=='H') c++;}
	for(i=0;i<n;i++) {
		if(str[i]=='H')
		for(j=i;j<i+c;j++)
			count[j%n]++;
	}
	s=c;
	for(i=0;i<n;i++)
		if(c-count[i]<s) s=c-count[i];
	printf("%d\n",s);
	return 0;
}
