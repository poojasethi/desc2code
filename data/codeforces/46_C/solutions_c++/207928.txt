#include <iostream>
#include <string>
using namespace std;

int main(){
	int n;
	cin>>n;
	string s;
	cin>>s;
	int t=0;
	for (int i=0; i<n; i++)
		if (s[i]=='H')
			t++;
	int min=n;
	for (int i=0; i<n; i++){
		int temp=0;
		int j=i;
		for (int tj=0; tj<t; tj++){
			if (s[j]!='H')
				temp++;
			j=(j+1)%n;
		}
		if (temp<min)
			min=temp;
	}
	cout<<min;
}
