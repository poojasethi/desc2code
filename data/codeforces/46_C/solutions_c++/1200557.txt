#include<stdio.h>
#include<string.h>
char ani[10000];
int main()
{
	int n,i,j,num,cnt,ans;
	while(scanf("%d",&n)!=EOF)
	{
		scanf("%s",ani);
		num=0;
		for(i=0;i<n;i++)
		{
			if(ani[i]=='H')num++;
		}
		cnt=0;ans=n;
		for(i=0;i<n;i++)
		{
			cnt=0;
			for(j=0;j<num;j++)
			{
				if(ani[(i+j)%n]=='T')
				cnt++;
			}
			if(cnt<ans)ans=cnt;
		}
		printf("%d\n",ans);
	}
	return 0;
}
