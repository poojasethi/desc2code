#include<stdio.h>
int main()
{
	int n,i,j,t,s,x,ans;
	char c[1001];
	scanf("%d",&n);
	scanf("%s",&c);
	t=0; ans=0;
	for (i=0;i<=n-1;i++) if (c[i]=='H') t++;
	for (i=0;i<=n-1;i++)
	{
		x=0;
		for (j=i;j<=i+t-1;j++)
		{
			s=j%n;
			if (c[s]=='H') x++;
		}
		if (ans<x) ans=x;
	}
	printf("%d\n",t-ans);
}
