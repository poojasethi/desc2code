#include <iostream>
#include <string>
using namespace std;

int main()
{
	int n, h = 0;
	string s;
	cin >> n >> s;
	for(int i = 0; i < n; i++)
		if(s[i] == 'H')
			h++;
	int min = 10000;
	for(int i = 0; i < n; i++)
	{
		int c = 0;
		for(int j = i; j < i + h; j++)
		       if(s[j % n] == 'T')
			       c++;
		if(c < min)
			min = c;
	}
	cout << min << endl;
	return 0;
}
