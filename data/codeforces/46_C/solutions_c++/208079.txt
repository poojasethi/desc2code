#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
using namespace std;
main()
{
	string inp;
	int num,h=0,i,j,ans,cnt;
	cin>>num;
	cin>>inp;
	for(i=0;i<num;i++)
	{
		if(inp[i]=='H')
		h++;
	}
	ans=num;
	for(i=0;i<num;i++)
	{
		cnt=0;
		for(j=i;j<(i+h);j++)
		{
			if(inp[j%num]=='T')
			cnt++;
		}
		if(cnt<ans)
		ans=cnt;
	}
	printf("%d",ans);
}
