#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
int i,j,n,l,h=0,count,ans;string a;
cin>>n>>a;l=a.size();
for(i=0;i<l;i++)if(a[i]=='H')h++;
ans=h;

for(i=0;i<n;i++)
{
    count=0;
    for(j=i;j<i+h;j++)
    {
    if(a[j%n]=='T')count++;
    }
    if(count<ans)ans=count;
}
cout<<ans;
return 0;
}
