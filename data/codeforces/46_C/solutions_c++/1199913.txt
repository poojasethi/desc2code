#include <stdio.h>

main()
{
	int n,a[2001],i,j,k,sum,max,now,c;
	scanf("%d",&n); 
	getchar();
	sum=0;
	for(i=1;i<=n;i++)
	{
		c=getchar();
		if(c=='H') a[i]=a[i+n]=0;
		else a[i]=a[i+n]=1;
		sum+=a[i];
	}
	max=0;
	for(i=1;i<=sum;i++) max+=a[i];
	now=max;
	for(i=1;i<n;i++)
	{
		now+=a[i+sum]-a[i];
	     if(now>max) max=now;
	}
	
	printf("%d",sum-max);
	
	
	
	
	return 0;
}
