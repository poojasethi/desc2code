#include <bits/stdc++.h>
using namespace std;


int n,h,t;
string s;

int main()
{
    ios::sync_with_stdio(0);
    cin >> n;
    cin >> s;
    for(int i=0;i<n;++i) if(s[i]=='H') h++; else t++;
    int et=n+n;
    for(int i=0;i<n;++i)
    {
        int ch=0;
        for(int j=i;j<i+h;++j)
            if(s[j%n]=='H') ch++;
        et=min(et,h-ch);
    }
    cout << et << "\n";
}
