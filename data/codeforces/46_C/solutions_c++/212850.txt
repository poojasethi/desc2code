#include <iostream>
#include <stdio.h>
using namespace std;
int n,i,j,nrt,t,rez;
char X[1001];
int main()
{
	cin>>n;
	gets(X);
	gets(X);
	rez=1000;
	for (i=0;i<=n-1;i++)
		if (X[i]=='T')
			nrt++;
	for (i=0;i<=n-1;i++)
	{
		t=0;
		for (j=1;j<=nrt;j++)
			if (X[(i+j-1)%n]=='T')
				t++;
		if (nrt-t<rez)
			rez=nrt-t;
	}
	cout<<rez;
	return 0;
}
