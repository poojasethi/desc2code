#include <iostream>
#include <cstring>
using namespace std;
int main()
{
	string s;
	int mini=2000000000,h=0,n;
	cin>>n;
	cin>>s;
	
	for(int i=0;i<n;i++)h+=(s[i]=='H'?1:0);
	
	for(int i=0;i<n;i++)
	{
		int tig=0;
		for(int j=0;j<h;j++)
			if (s[(i+j)%n]=='T')tig++;
		mini=min(tig,mini);
	}
	cout<<mini;
	return 0;
}
