#include <iostream>
#include <string>
#include <cmath>

using namespace std;

int n;
string s;
int main()
{
  cin>>n>>s;
  int h=0,t=0;
  for (int i=0;i<n;i++) {
    h+=s[i]=='H';
    t+=s[i]=='T';
  }
  int sol=n+1;
  for (int i=0;i<n;i++) {
    int nh=0;
    for (int j=0;j<t;j++)
      nh+=s[(i+j)%n]=='H';
    sol=min(sol,nh);
  }
  cout<<sol<<endl;
}
