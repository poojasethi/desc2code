#include <stdio.h>
int main()
{
	int su=0,i,j,n;
	char c[2500];
	scanf ("%d",&n);
	scanf ("%s",c);
	for (i=n;i<2*n;i++){
		c[i]=c[i-n];
		if (c[i]=='T'){su++;}
	}
	c[2*n]='\0';
	int max=0;
	for (i=0;i<n;i++){
		int sum=0;
		for (j=i;j<su+i;j++){
			if (c[j]=='T'){sum++;}
		}
		if (sum>max){max=sum;}
	}
	printf ("%d\n",su-max);
	return 0;
}
