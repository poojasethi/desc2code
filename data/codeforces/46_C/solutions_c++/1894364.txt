#include<stdio.h>
int main()
{
	int n,h=0,i,min,j,ctr;
	char s[1001];
	int H[1001];
	scanf("%d%c",&n,&s[0]);
	for(i=0; i<n; i++)
	{
		scanf("%c",&s[i]);
		if(s[i]=='H')
			H[++h]=i;
	}
	min=1111;
	for(i=1; i<=h; i++)
	{
		ctr=0;
		for(j=H[i]; j<(H[i]+h); j++)
			if(s[j%n]=='T')
				ctr++;
		if(ctr<min)
			min=ctr;
	}
	printf("%d\n",min);
	return 0;
}
