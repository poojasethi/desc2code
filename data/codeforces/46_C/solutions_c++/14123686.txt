#include<iostream>
using namespace std;
int h,temp,ans,n,i,j;
int main()
{
	char a[1100];
	cin>>n>>a;
    for(i=0;i<n;i++)if(a[i]=='H')h++;
	ans=n-1;
	temp=0;
	for(i=0;i<n;i++)
	{
		temp=0;
	for(j=0;j<h;j++)
       if(a[(i+j)%n]=='T')temp++;
	   ans=ans>temp?temp:ans;
	   }
	cout<<ans<<endl;
	return 0;
}
