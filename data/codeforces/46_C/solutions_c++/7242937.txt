#include <iostream>
using namespace std;

int n, i, j, a[1001], ans;
string s;
int main()
{
	cin>>n>>s;
	for (i=0; i<n; i++) {
		if (s[i]=='H') a[i+1]=a[i]+1;
		else a[i+1]=a[i];
	}
	ans=n;
	for (i=0; i<n; i++) for (j=i+1; j<=n; j++) {
		if (n-a[n]+i-j==0) ans = min(ans, a[j]-a[i]);
		if (j-i==a[n]) ans = min(ans, a[n]-a[j]+a[i]);
	}
	cout<<ans<<endl;
	return 0;
}
