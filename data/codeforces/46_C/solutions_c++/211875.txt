#include<iostream>
#include<string>
#include<cmath>
using namespace std;
int main()
{
int N,h=0,ch=0,chm=10000;
string str;
cin>>N;
cin>>str;
for(int i=0 ;i< N ;++i) if(str[i]=='H') h++;
for(int i=0;i < N; i++){ ch=0;
for(int j=0; j< h; ++j) { 
if(str[(i+j)%N]=='T') ch++;
}
chm=min(chm,ch);
}
cout<<chm<<"\n";
return 0;
}
