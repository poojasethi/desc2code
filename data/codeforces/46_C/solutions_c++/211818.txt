#include<iostream>
#include <stdio.h>
using namespace std;
char S[1100];
int main(){
    int n, h=0, r=n+1, s; scanf("%d%s", &n, S);
    for(int i=0; i<n; i++)if(S[i]=='H')h++;
    for(int i=0; i<n; i++){
        s=0;
        for(int j=0; j<h; j++)s+=S[(i+j)%n]=='T'?1:0;
        r=::min(r,s);
    }
    printf("%d", r);
    return 0;
}
