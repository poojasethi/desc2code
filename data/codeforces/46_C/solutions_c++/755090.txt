#include <stdio.h>
#include <algorithm>
using namespace std;

char a[1005];

int main(void) {
	int n;
	scanf("%d", &n);
	scanf("%s", a);
	int c=0;
	for(int i=0;i<n;i++) if(a[i]=='T') ++c;
	int ans = n;
	for(int i=0;i<n;i++) {
		int w=0;
		for(int j=0;j<c;j++)
			if(a[(i+j)%n]!='T') ++w;
		if(w < ans) ans = w;
	}
	printf("%d\n", ans);
	return 0;
}
