#include<stdio.h>
int main()
{
	int n,i,j,h,k,a,max;
	char s[1001];
	h=0;
	max=0;
	scanf("%d",&n);
	scanf("%s",s);
	for(i=0;i<n;i++)
	{
		if(s[i]=='H')
		{
			h++;
		}
	}
	for(j=0;j<n;j++)
	{
		a=0;
		for(k=j;k<j+h;k++)
		{
			if(s[k%n]=='H')
			{
				a++;
			}
		}
		if(a>max)
		{
			max=a;
		}
	}
	printf("%d\n",h-max);
}
