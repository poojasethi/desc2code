#include <iostream>
using namespace std;
int n,ans,h,temp;
char a[1100];
int main(){
    cin>>n>>a;
    ans=n-1;
    for(int i=0;i<n;i++)    if(a[i]=='H')   h++;
    for(int i=0; i<n; i++){temp=0;
        for(int j=0;j<h;j++) if(a[(i+j)%n]=='T') temp++;
        ans=(temp<ans?temp:ans);
    }
    cout<<ans;
}
