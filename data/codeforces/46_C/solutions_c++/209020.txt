#include <cstdio>
const int N = 1005;
int n, cH;
char s[N];
int work(int d)
{
	int ret = 0;
	for (int i = 0; i < cH; ++i) if (s[(i + d) % n] == 'T') ++ret;
	return ret;
}

int main()
{
	int ans = 1001;
	scanf("%d\n", &n);
	gets(s);
	for (int i = 0; i < n; ++i) if (s[i] == 'H') ++cH;
	for (int i = 0; i < n; ++i) {
		int t = work(i);
		if (t < ans) ans = t;
	}
	printf("%d\n", ans);
}
