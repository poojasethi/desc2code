#!/usr/bin/python

n=int(raw_input())

line=raw_input().split(' ')
a=[]
for e in line:
    a.append(int(e))

a=sorted(a)

T=int(raw_input())
maxLen=0
for i in range(n):
    end=n-1
    begin=i
    while begin+1<end:
        mid=begin+(end-begin)/2
        if a[mid]>T+a[i]:
            end=mid
        else:
            begin=mid
    if a[end]>T+a[i]:
        curLen=begin-i+1
    else:
        curLen=end-i+1
    if curLen>maxLen:
        maxLen=curLen

print maxLen
