
def solve():
	n = int(raw_input())
	s = raw_input().split(" ")
	s = map(lambda x: int(x), s)
	t = int(raw_input())
	s.sort()
	longest = 1
	for i in range(n):
		sofar = 1
		for j in range(i+1,n):
			if s[j] - s[i] > t:
				break
			else:
				sofar += 1
		if sofar > longest:
			longest = sofar
	print longest


if __name__ == "__main__":
	solve()
