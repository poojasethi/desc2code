n = int(raw_input())
t = map(int,raw_input().split())
T = int(raw_input())
t.sort()
sizes = []
for i in xrange(n):
    window = t[i] + T 
    num = 1
    for j in xrange(i+1,n):
        if t[j] <= window:
            num += 1
        else:
            break;
    sizes.append(num)

print max(sizes)
