#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
//freopen("i.txt","r",stdin);
int n,ans=0,m1=0; cin>>n;
int a[3001]={0};
for (int i=0;i<n;i++) {int x; cin>>x; a[x]++;m1=max(m1,x);}
int T; cin>>T;
for (int i=1;i<=m1+T;i++) a[i]+=a[i-1];
for (int i=1;i<=m1;i++) ans=max(ans,a[i+T]-a[i-1]);
cout<<ans;
return 0;
}
