#include <cstdio>
#include <algorithm>
using namespace std;

int a[200];
int n,t,ans;
int main(){
    scanf("%d", &n);
    for (int i=0; i<n; ++i) scanf("%d", a+i);
    scanf("%d", &t);

    sort(a, a+n);
    ans = 0;
    int l=0;
    for (int i=0; i<n; ++i){
        while (a[i]-a[l]>t) ++l;
        ans = max(ans, i-l+1);
    }
    printf("%d\n", ans);
}
