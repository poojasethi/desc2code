#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int n,a[101],i,t,br=0,j;
    scanf("%d",&n);
    for(i=0;i<n;i++)scanf("%d",&a[i]);
    sort(a,a+n);
    scanf("%d",&t);
    for(i=0;i<n;i++)
    {
        j=i;
        while(a[i]+t>=a[j]&&j<n)j++;
        if(j-i>br)br=j-i;
    }
    printf("%d\n",br);
}
