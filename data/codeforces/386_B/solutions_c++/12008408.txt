#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	int n,i,t,s=0,m=0,j;
	int a[1050];
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	sort(a,a+n);
	//for(i=0;i<n;i++)
		//cout<<a[i];
    cin>>t;
    for(i=0;i<n;i++)
	{
		for(j=i;j<n;j++)
	{
		if(abs(a[i]-a[j])<=t) m++;
	}
	if (m>s) s=m;
	m=0;
	}
	cout<<s;
	return 0;
}
