#include <iostream>
#include <algorithm>
using namespace std;
main(){
    int n,i;
    cin>>n;
    int a[n+1];
    for(i=0;i<n;i++)
        cin>>a[i];
    sort(a , a + n);
    int t;
    cin>>t;
    int ans = 0;
    int j = 0;
    for(i=0;i<n;i++){
        while (j < n && a[j] - a[i] <= t) ++j;
        ans = max(ans, j - i);
    }
    cout<<ans;
}
