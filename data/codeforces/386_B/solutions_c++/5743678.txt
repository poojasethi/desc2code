#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define M 1010

int n,m,ans;
int p[M];

int main()
{
	int i,tmp;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&tmp);
		p[tmp]++;
	}
	scanf("%d",&m);
	for(i=1;i<M;i++) p[i]+=p[i-1];
	for(i=m;i<M;i++) ans=max(ans,p[i]-p[i-m-1]);
	printf("%d\n",ans);
	return 0;
}
