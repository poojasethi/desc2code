#include <iostream>
#include <algorithm>
using namespace std;
int n,t[1005],i,j,tt,maxkil=1;
int main()
{
    cin>>n;
    for(i=1;i<=n;i++)cin>>t[i];
    cin>>tt;
    sort(t+1,t+1+n);
    for(i=1;i<=n;i++){
     for(j=i+1;j<=n;j++){if ( (t[j]-t[i])<=tt ){if (j-i+1>maxkil) maxkil=j-i+1; }}
     }
    cout<<maxkil<<endl;
    return 0;
}
