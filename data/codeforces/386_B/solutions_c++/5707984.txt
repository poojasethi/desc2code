#include <cstdio>
#include <algorithm>
using namespace std;

int n, i, h, T, ans, t[1100];

int main() {
	scanf("%d", &n);
	for (i = 1; i <= n; i++)	scanf("%d", &t[i]);
	sort(t + 1, t + n + 1);
	h = 1;
	scanf("%d", &T);
	for (i = 1; i <= n; i++) {
		while (t[i] - t[h] > T)	h++;
		ans = max(ans, i - h + 1);
	}
	printf("%d\n", ans);
}
