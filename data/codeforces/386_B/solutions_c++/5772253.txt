
#include <iostream>
#include <algorithm>

using namespace std;

int t[101];

int main()
{

	int n, freebie, maxi = -1;
	cin>>n;
	for (int i = 0; i < n; i++) cin>>t[i];
	cin>>freebie;

	sort(t, t + n);

	for ( int i = 0; i < n; i++)
	{

		for (int j = i; j < n; j++)
		{

			if(t[j] - t[i] <= freebie) if(j - i + 1 > maxi) maxi = j - i + 1;

		}

	}

	cout<<maxi;

	return 0;

}
