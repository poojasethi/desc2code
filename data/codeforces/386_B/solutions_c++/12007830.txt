#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,ShoutTime[100],T,ans=0,t;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>ShoutTime[i];
	cin>>T;
	sort(ShoutTime,ShoutTime+n);
	for(int i=0;i<n;i++)
	{
		t=upper_bound(ShoutTime+i,ShoutTime+n,ShoutTime[i]+T)-&ShoutTime[i];
		if(t>ans)ans=t;
	}
	cout<<ans<<endl;
	return 0;
}
