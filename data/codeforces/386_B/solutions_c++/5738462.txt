#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>

using namespace std;

int s[1000+3],n,T,ans;

int main()
{
scanf("%d",&n);
for (int i=1;i<=n;i++){int x;scanf("%d",&x);s[x]++;}
scanf("%d",&T);
for (int i=1;i<=1000;i++) s[i]+=s[i-1];
for (int i=1;i<=1000;i++) ans=max(ans,s[i]-s[max(i-T-1,0)]);
printf("%d",ans);
}
