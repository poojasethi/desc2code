#include <bits/stdc++.h>

using namespace std;

const int maxl = 100 + 5;

int a [ maxl ];
int n;

int T;
int ans;

int main (){
	cin >> n;
	for ( int i = 0 ; i < n ; i++ )
		cin >> a [ i ];
	cin >> T;

	sort ( a, a + n );

	for ( int i = 0 ; i < n ; i++ ){
		int ind = upper_bound ( a, a + n, a [ i ] + T ) - a;
		ans = max ( ans, ind - i );
	}

	cout << ans << endl;
}
