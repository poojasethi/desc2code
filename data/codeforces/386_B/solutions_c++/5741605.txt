#include <iostream>
#include<algorithm>
using namespace std;
# define MAX 100000
int main()
{
    int n,i,a[102],t,tm,max=0;
    cin>>n;
    for(i=0;i<n;i++) cin>>a[i];
    cin>>t;
    sort(a,a+n);
    for(i=0;i<n;i++)
    {tm=upper_bound(a,a+n,a[i]+t)-a-i;
        if(tm>max) max=tm;
     }
     cout<<max;
    return 0;
}
