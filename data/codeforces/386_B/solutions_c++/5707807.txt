#include <iostream>
#include <algorithm>
using namespace std;
int a[101];
int n,t,ans=-1;
int main ()
{
	cin>>n;
	for (int i=0;i<n;i++)
		cin>>a[i];
	cin>>t;
	sort(a,a+n);
	int p=0;
	for (int i=0;i<n;i++)
	{
		while ((a[p]-a[i])<=t && p<n)
		{
			p++;
		}
		ans=max(ans,(p-i));
	}
	cout<<ans;
	return 0;
}
