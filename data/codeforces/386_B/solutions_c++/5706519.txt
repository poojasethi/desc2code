#include <iostream>
#include <algorithm>
using namespace std;

int main(){

    int n, arr[105];
    cin >> n;
    for(int i = 0; i<n; i++) cin >> arr[i];
    sort(arr, arr+n);
    int l = 0, r = 0, T;
    cin >> T;
    int res = 0;
    for( ; r<n; r++){
        while(arr[l]+T<arr[r]) l++;
        res = max(res, r-l+1);
    }
    cout << res;

}
