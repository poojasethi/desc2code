#include<stdio.h>
#include<algorithm>
using namespace std;
int c[1005];
main()
{
	int a,b,d,e,f,g,h;
	scanf("%d",&a);
	for(b=0;b<a;b++)
	{
		scanf("%d",&d);
		c[d]++;
	}
	scanf("%d",&e);
	h=0;
	for(b=1;b<1002-e;b++)
	{
		g=0;
		for(f=b;f<=b+e;f++)
		{
			g+=c[f];
		}
		if(g>h)
		h=g;
	}
	printf("%d",h);
	
}
