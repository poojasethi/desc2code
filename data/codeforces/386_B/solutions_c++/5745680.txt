#include<iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;
int main()
{int n,a,time[2001]={0};
int t,ans_at_i=0,ans=0;
cin>>n;
for(int i=0;i<n;i++)
{cin>>a;
time[a]++;}
cin>>t;
for(int i=1;i<=1000;i++)
{ans_at_i=0;
for(int j=0;j<=t;j++)
{ans_at_i+=time[i+j];
ans=max(ans_at_i,ans);
}}
cout<<ans;
}
