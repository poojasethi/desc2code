#include<iostream>
#include<algorithm>

using namespace std;

int n; int t[10000];

int main(){
  int i,time;
  cin>>n;
  for(i=0;i<n;i++){
    cin>>t[i];
  }
  sort(t,t+n);
  cin>>time;
  int q,ans = -1;
  for(i=0;i<n;i++){
    q = upper_bound(t,t+n,t[i]+time)-t-i;
    ans = max(q,ans);
  }
  cout<<ans<<endl; 
  return 0;
}
