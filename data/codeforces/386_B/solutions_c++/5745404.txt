#include <iostream>
#include <algorithm>
using namespace std;
int arr[111],n,t,now,ans,j;
int main()
{
    cin >> n;
    for (int i=0; i<n; i++) cin >> arr[i];
    cin >> t;
    sort(arr,arr+n);
    for (int i=0; i<n; i++)
        for (j=i; (j<n) && (arr[j]-arr[i]<=t); j++) ans=max(ans,j-i);
    cout << ans+1;
}
