#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int a[101],t;
	int n,max=0;
	cin>>n;
	for(int i=0;i<n;i++)
	{

		cin>>a[i];

	}
	cin>>t;
	sort(&a[0],&a[n]);
	for(int i=0;i<n;i++)
	{
			int cnt=0;
			for(int j=i;j<n;j++)
			{
				if((a[j]-a[i])<=t)
					cnt++;

			}
			if(cnt>max)
				max=cnt;

	}
	cout<<max<<endl;
	return 0;
}
