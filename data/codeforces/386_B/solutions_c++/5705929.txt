#include <iostream>
#include <algorithm>

using namespace std;

long n,m,i,a[111],t,j;

int main()
{
	cin >> n;
	for (i=1;i<=n;i++) cin >> a[i];
	cin >> t;
	sort(a+1,a+n+1);
	 m=1;
	 a[n+1]=100000;
	for (i=1;i<=n;i++){
		for (j=i+1;j<=n+1;j++){
			if (a[j]-a[i]>t){
				if (m<j-i) m=j-i;
				break;
			}
		}
	}		
	cout << m;
	return 0;
}
