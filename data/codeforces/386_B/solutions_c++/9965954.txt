#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,i,m,j,a[1001],max=-32769,k;
cin>>n;
for(i=1;i<=n;i++)
    cin>>a[i];
    cin>>m;
for(i=1;i<=n;i++){
        k=0;
    for(j=1;j<=n;j++)
    {
        if((a[i] <= a[j]) && (a[j]<=a[i]+m))
            k++;
        if(k>=max)
            max=k;
    }}
    cout<<max;
return 0;
}
