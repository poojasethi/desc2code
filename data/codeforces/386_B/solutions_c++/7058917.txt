#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int A[n];
	int i,j,sum=0,s=0,a,b;
	for(i=0;i<n;i++) cin>>A[i];
	sort(A,A+n);
	int T;
	cin>>T;
	for(i=0;i<n;i++)
	{
		a=A[i];
		for(j=i+1,s=n-i;j<n;j++) if(A[j]-A[i]>T) {s=j-i;break;}
		if(s>sum) sum=s;
	}
    cout << sum << endl;
    return 0;
}
