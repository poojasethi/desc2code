#include <iostream>
#include <algorithm>
#include<stdio.h>

#define PI 3.14159265359
using namespace std;
int n,D[500001],ans,x,m;
int main(){
    scanf("%d", &n);
for(int i=0;i<n;i++)scanf("%d", D+i);
sort(D,D+n);
 scanf("%d", &m);
 for(int i=0,j=0;i<n;i++)
 {
 while(j<n-1 && D[j+1]-D[i] <=m)j++;
 ans=(max(ans,j-i+1));
 }
    printf("%d\n", ans);
}
