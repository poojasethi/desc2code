#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

int main()
{
	int n,t;
	cin >> n;
	int i,j,mx=0;
	vector<int> v;
	for(i=0;i<n;i++)
	{
		cin >> t;
		v.push_back(t);
	}
	cin >> t;
	sort(v.begin(),v.end());
	for(i=0;i<v.size();i++)
	{
		for(j=i;j<v.size();j++)
		{
			if(v[j]-v[i]<=t && j-i+1>mx)	mx=j-i+1;
		}
	}
	cout << mx << endl;
	return 0;
}
