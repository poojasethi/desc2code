#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int i,j,n,a[100],t,m=0,k=0;
    cin>>n;
    for(i=0;i<n;i++)cin>>a[i];
    cin>>t;
    sort(a,a+n);
    for(i=0;i<n;i++)
    {
        for(j=i,k=0;(a[j]<=a[i]+t)&&j<n;k++,j++);
        if(k>m)m=k;
    }
    cout << m<< endl;
    return 0;
}
