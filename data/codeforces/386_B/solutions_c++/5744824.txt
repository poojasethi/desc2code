#include <cstdio>
#include <algorithm>

using namespace std;

const int maxn=1010;

int n,a[maxn],t,s,l,r;

int main() {
	scanf("%d",&n);
	for (int i=0;i<n;i++)
		scanf("%d",a+i);
	scanf("%d",&t);
	sort(a,a+n);
	s=0;
	l=0;
	r=0;
	while (l<n) {
		while (r<n && a[r]-a[l]<=t)
			r++;
		s=max(s,r-l);
		l++;
	}
	printf("%d\n",s);
}
