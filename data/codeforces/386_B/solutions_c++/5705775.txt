#include<iostream>
using namespace std;
int main()
{
	int N;
	cin >> N;
	int p[2001] = {0};
	for(int i=0; i<N; ++i)
	{
		int x;
		cin >> x;
		++p[x];
	}

	int T, maxi = 0;
	cin >> T;
	for(int i=0; i<1001; ++i)
	{
		int cnt = 0;
		for(int j=0; j<=T; ++j)
		{
			if(p[i+j])
			{
				cnt += p[i+j];
			}
		}

		maxi = max(cnt, maxi);
	}

	cout << maxi << endl;

	return 0;
}
