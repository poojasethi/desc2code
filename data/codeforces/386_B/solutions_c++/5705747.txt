#include<iostream>
#include<algorithm>
using namespace std;
int n, num[1001], T, mx, cur;
int main() {
	cin >> n;
	for(int i=0;i<n;i++)
		cin >> T, num[T]++;
	cin >> T;
	for(int i=0;i<=T;i++)
		cur += num[i];
	mx = max(mx, cur);
	for(int i=1;i+T <= 1000;i++)
		cur += num[i+T] - num[i-1], mx = max(mx, cur);
	cout << mx << '\n';
	return 0;
}
