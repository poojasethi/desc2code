#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	int n,t[1005],T,num,Max = -1;
	cin>>n;
	for(int i = 0; i < n; i++)
		cin>>t[i];
	cin>>T;
	sort(t,t+n);
	for(int i = 0; i < n; i++)
	{
		num = 0;
		for(int j = i; j<n && t[j]-t[i]<=T ;j++) num++;
		Max = Max > num ? Max : num;
	}
	cout<<Max<<endl;
	return 0;
}
