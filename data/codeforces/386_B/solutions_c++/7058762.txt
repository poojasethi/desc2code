#include <stdio.h>

int main()
{
	int n,T,max,i,j;
	int t[1010]={0};
	int ans=0;

	scanf("%d",&n);
	max=0;

	for (i=0;i<n;i++)
	{
		int x;
		scanf("%d",&x);
		t[x]++;
		if (x>max)	max=x;
	}

	scanf("%d",&T);

	for (i=1;i<=max;i++)
	{
		int x=0;
		for (j=i;j<=i+T&&j<=max;j++)
			x+=t[j];

		if (x>ans)
			ans=x;
	}

	printf("%d\n",ans);
	return 0;
}
