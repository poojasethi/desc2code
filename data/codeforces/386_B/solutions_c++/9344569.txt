#include<iostream>
#include<algorithm>
#include<conio.h>
using namespace std;

int t,a[1005],n,j,i,rs;

int main()
{
  cin>>n;
  for(i=1;i<=n;++i) cin>>a[i];
  cin>>t;

  sort(a+1,a+n+1);

  for(i=1;i<=n;++i)
  {
    j=i;
    while(a[j]<=a[i]+t && j<=n) ++j;
    if(a[j]>a[i]+t || j==n+1) --j;
    rs=max(rs,j-i+1);
  }

  cout<<rs<<'\n';

 return 0;
}
