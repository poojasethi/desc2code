#include <bits/stdc++.h>
using namespace std;
int s[200];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i =0 ;i < n; i++)
		scanf("%d",s+i);
	int k;
	scanf("%d",&k);
	sort(s,s+n);
	int i = 0,j = 0;
	int ans = 0;
	while(1)
	{
		while(j<n&&s[j] <= s[i]+k) j++;
		ans = max(ans, j - i);
		if(j >=n)
			break;
		i++;
	}
	printf("%d\n",ans);
	return 0;
}
