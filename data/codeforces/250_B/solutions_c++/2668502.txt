#include<stdio.h>
#include<string.h>
char address[150];

void solve()
{
	int i,j,n,count=0,p=0,sum=0;
	char ans[4];
	n=strlen(address);
	for(i=0;i<n;i++)
		if(address[i]==':') count++;
	for(i=0;i<n;i++){
		if(address[i]!=':') ans[p++]=address[i];
		if(i==n-1 || address[i]==':'){
			for(j=0;j<4-p;j++) printf("0");
			for(j=0;j<p;j++){
				printf("%c",ans[j]);
				ans[j]=0;
			}
			sum++; p=0;
		}
		if(address[i]==':' ){
			if(sum>0 && sum!=8)
				printf(":");
			if(address[i+1]==':'){
				if(count>=8) i++;
				while(++count<=7){
					printf("0000:"); sum++;
				}
			}
		}
	}
	while(++sum<=8) printf("0000");
	printf("\n");
}
int main()
{
	int n;
	scanf("%d",&n);
	while(n--){
		scanf("%s",address);
		solve();
	}
	return 0;
}
