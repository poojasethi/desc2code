#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <string>
#include <iostream>
#include <iomanip>
using namespace std;

int a[10];

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		string s;
		cin>>s;
		int len=s.size();
		int n=0,m=0;a[0]=-1;
		for (int i=0;i<len;i++)
			if (s[i]==':')
				a[++n]=i;
		for (int i=2;i<=n;i++)
		if (a[i]==a[i-1]+1) m=i;
		for (int i=1;i<=n;i++)
		{
			if (i==m)
			{
				if (n==8) continue;
				for (int j=1;j<=7-n;j++)
				cout<<setw(5)<<setfill('0')<<s.substr(a[i-1]+1,a[i]-a[i-1]);
			}
			cout<<setw(5)<<setfill('0')<<s.substr(a[i-1]+1,a[i]-a[i-1]);
		}
		cout<<setw(4)<<setfill('0')<<s.substr(a[n]+1,len-a[n])<<"\n";
	}
	return 0;
}
