#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char s[1000];
int next[1000];

int main()
{
    int n;
    scanf("%d",&n);
    while(n--)
    {
        cin >> s;
        int len = strlen(s);
        int cnt = 0,son = len;
        for(int i=len-1;i>=0;i--)if(s[i] == ':')cnt++,next[i] = son,son = i;
        for(int i=son;i<4;i++)printf("0");
        for(int i=0;i<len;i++)
        {
            if(s[i] == ':')
            {
                if(s[i+1] == ':')while(++cnt<=8)printf(":0000");
                else {printf(":");for(int j=next[i]-i-1;j<4;j++)printf("0");}
            }
            else cout << s[i];
        }
        cout << endl;
    }
    return 0;
}
