#include<stdio.h>
#include<stdlib.h>
#include<string.h>
char s[1001];
int main()
{
	int t,bk,len,i,j,k;
	scanf("%d",&t);
	for(bk=0;bk<t;++bk)
	{
		scanf("%s",s);
		len=strlen(s);
		int cnt=0;
		for(i=0;i<len;++i)
			if(s[i]==':')
				cnt++;
		if(s[len-1]!=':')
			cnt++;
		if(s[0]==':')
			cnt--; 
		s[len]=':';
		for(i=s[0]==':'?1:0;i<len;i=j+1)
		{
			if(s[i]==':')
			{
				for(k=0;k<=8-cnt;++k)
					if(k==8-cnt&&i==len-1)
						printf("0000");
					else
						printf("0000:");
				j=i;
			}
			else
			{
				int cnt2=0;
				for(j=i;s[j]!=':';++j)
					cnt2++;
				for(k=0;k<4-cnt2;++k)
					printf("0");
				for(j=i;s[j]!=':';++j)
					printf("%c",s[j]);
				if(j<len)
					printf(":");
			}
		}
		printf("\n");
	}
	return 0;
}
