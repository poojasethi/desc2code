#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<iostream>
using namespace std;
int i,j,k,l,n,m,T,num,len;
char c[120];
int main(){
	for(scanf("%d",&T);T;T--){
		scanf("%s",c+1);
		l=strlen(c+1);
		num=0;
		c[l+1]=':';
		for(i=1;i<=l;++i)if(c[i]!=':' && c[i+1]==':')num++;
		if(c[1]==':')i=2;else i=1;
		for(;i<=l;i=j+1){
			for(j=i;j<=l && c[j]!=':' ;++j);
			len=j-i;
			if(len==0){
				for(k=8;k>num;--k){
					printf("0000");
					if(k==num+1 && j==l)printf("\n");else printf(":");
				}
				continue;
			}
			for(k=4;k>len;--k)printf("0");
			for(k=i;k<j;++k)printf("%c",c[k]);
			if(j==l+1)printf("\n"); else printf(":");
		}
	}
}
