#include<iostream>
#include<vector>
#include<string>
using namespace std;

int main(){
    int n;
    cin>>n;
    for (int i=0;i<n;++i){
        string s;
        vector <string> a(10008);
        int count=0;
        bool res=false;        
        cin>>s;
        for (int i=0;i<s.size();++i)
            if (s[i] == ':')    ++count;
            else    a[count].push_back(s[i]);
        for (int i=0;i<=count;++i){
            if (a[i] == "" && !res){
                for (int j=1;j<=8-count;++j)
                    cout<<"0000:";
                res = true;
            }
            else{
                for (int j=1;j<=4-a[i].size();++j)
                    cout<<0;
                cout<<a[i];
                if (i != count) cout<<":";
            }
        }
        cout<<endl;
    }
}
