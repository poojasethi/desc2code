#include<cstdio>
#include<cstring>
using namespace std;
char s[128];
int next[128],t;
int main(){
	while(scanf("%d ",&t)!=EOF){
		while(t--){
			gets(s);
			int n=strlen(s);
			int flag=n,sum=0;
			for(int i=n-1;~i;i--){
				if(s[i]==':')
					next[i]=flag,sum++,flag=i;
			}
			for(int i=0;i<4-flag;i++) printf("0");
			for(int i=0;i<n;i++){
				if(s[i]==':'){
					if(s[i+1]==':')
						for(int j=sum;j<8;j++) printf(":0000");
					else{
						printf(":");
						for(int j=next[i]-i;j<=4;j++) printf("0");
					}
				}
				else printf("%c",s[i]);
			}
			puts(" ");
		}
	}
}
