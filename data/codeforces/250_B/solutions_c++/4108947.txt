#include <stdio.h>
#include <string.h>
int main()
{
	char str[1000];
	int f[10];
	int n,len,zeroid,sixteenbit,i,j;
	scanf("%d",&n);
	while(n--)
	{
		memset(str,0,sizeof(str));
		scanf("%s",str);
		memset(f,0,sizeof(f));
		len=strlen(str);
		zeroid=1;
		sixteenbit=0;
		for(i=0;i<len;i++)
		{
			if(str[i]==':')
			{
				f[zeroid++]=sixteenbit;
				sixteenbit=0;
				if(str[i+1]==':')
				{
					i+=1;
					zeroid=8;
					for(j=i+1;j<len;j++)
					{
						if(str[j]==':')
						zeroid--;
					}
				}
			}
			else
			{
				if(str[i]>='0'&&str[i]<='9')
				sixteenbit=sixteenbit*16+str[i]-'0';
				else
				sixteenbit=sixteenbit*16+str[i]-'a'+10;
			}
		}
		f[zeroid]=sixteenbit;
		for(i=1;i<8;i++)
		printf("%04x:",f[i]);
		printf("%04x\n",f[8]);
	}
}
