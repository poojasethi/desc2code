#include <iostream>
#include <stdio.h>
#include <string>
using namespace std;
int T,p,q,cnt,pos;
string str,cur;

int main(){
	cin>>T;
	while(T--){
		cin>>str;pos=-1;cnt=0;
		p=0;q=str.find(':');
		while(q!=-1){
			if(q==p)pos=p;else cnt++;
			if(q!=p&&q-p!=4){
				cur=str.substr(p,q-p);
				while(cur.size()!=4)cur='0'+cur;
				str=str.substr(0,p)+cur+str.substr(q);
				q=p+4;
			}
			p=q+1;q=str.find(':',p);
		}
		cur=str.substr(p);
		while(cur.size()!=4)cur='0'+cur;
		str=str.substr(0,p)+cur;
		if(pos!=-1){
			cur=":";
			for(int i=cnt;i<7;i++)cur+="0000:";
			str=str.substr(0,pos-1)+cur+str.substr(pos+1);
		}
		if(str[0]==':')str=str.substr(1);
		if(str[str.size()-1]==':')str=str.substr(0,str.size()-1);
		cout<<str<<endl;
	}
	return 0;
}
