# include <stdio.h>
# include <stdlib.h>
# include <string.h>


int main()
{
    int t;
    scanf("%d", &t);
    getchar();
    while(t--) {
	char s[40];
	gets(s);
	int next[40], cnt = 0, i, n = strlen(s);
	int son = n, j;
	for ( i = n-1; i >= 0; i-- ) if ( s[i] == ':' ) cnt++, next[i] = son, son = i;
       	for ( i = son; i < 4; i++ ) printf("0");
	for ( i = 0; i < n; i++ ) {
	    if ( s[i] == ':' ) {
		
		if ( s[i+1] == ':' )
		    
		    while (cnt++<8) printf( ":0000" );
		else {
		    printf(":");
		    for ( j = next[i]-i; j <= 4; j++ ) printf("0");
		}
	    }
	    else
		printf("%c", s[i]);
	}
	printf("\n");
    }
    

    return 0;
}
