#include<stdio.h>
#include<string.h>
int main(){
	char str[100];
	int i,n,l,tot,t,j,f[10];
	scanf("%d",&n);
	while(n--){
		scanf("%s",&str);
		l=strlen(str);
		for(i=1;i<=8;i++)f[i]=0;t=1;tot=0;
		for(i=0;i<l;i++){
			if(str[i]==':'){
				f[t]=tot;t++;tot=0;
				if(str[i+1]==':'){
					i++;t=8;
					for(j=i+1;j<l;j++)
						if(str[j]==':')t--;
				}
			}
			else {
				if(str[i]>='0'&&str[i]<='9')tot=tot*16+str[i]-'0';
				else tot=tot*16+str[i]-'a'+10;
			}
		}
		if(tot!=0)f[t]=tot;
		for(i=1;i<8;i++)printf("%04x:",f[i]);
		printf("%04x\n",f[8]);
	}
}
