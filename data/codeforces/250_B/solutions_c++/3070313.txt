#include<stdio.h>
#include<string.h>
char s[110]={':'};
char as[110],cpy[110]=":0000:0000:0000:0000:0000:0000:0000:0000";
void insert(int a,int b,int k){
	int i,c=b-a-1;k*=5;
	for(i=1;i<=c;i++)
		as[k-i]=s[b-i];
}
int main(){
	int i,j,n,m,k,cs;
	scanf("%d",&cs);
	while(cs--){
		strcpy(as,cpy);
		scanf("%s",s+1);
		n=strlen(s);
		s[n]=':';s[++n]=0;
		m=0;
		for(k=0,i=1;i<n;i++)
			if(s[i]==':')k++;
		for(j=0,i=1;i<n;i++)
			if(s[i]==':'){
				m++;
				insert(j,i,m);j=i;
				if(s[i+1]==':')break;
			}
		m=9;
		for(j=n-1,i=n-2;i>0;i--)
			if(s[i]==':'){
				m--;//printf("%d %d %d\n",i,j,m);puts(as);
				insert(i,j,m);j=i;
				if(s[i-1]==':')break;
			}
		puts(as+1);
	}
	return 0;
}
