#include<stdio.h>
#include<string.h>
#include<math.h>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t,n,cnt,k,next[105],i,j;
	char s[105];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",s);
		n=strlen(s);
		cnt=0;k=n;
		for(i=n-1;i>=0;i--) 
		if(s[i]==':')
		{
			cnt++;
			next[i]=k;
			k=i;
		}
		for(i=k;i<4;i++)
		printf("0");
		for(i=0;i<n;i++)
		if(s[i]==':')
		{
			if(s[i+1]==':')
			while(++cnt<=8) printf(":0000");
			else
			{
				printf(":");
				for(j=next[i]-i;j<=4;j++)
				printf("0");
			}	
		}
		else printf("%c",s[i]);
		printf("\n");
	}
	return 0;
}
