#include<cstring>
#include<iostream>
#include<algorithm>
#include<string>
using namespace std; 
int main()
{
	int r,n,q[105],xx;
	string s;
	cin>>n;
	while(n--)
	{
		cin>>s;
		r=0;
		xx=0;
		for(int i=0;i<s.length();i++){
			if(s[i]==':'){
				q[++r]=i;
				if(i>0)if(s[i-1]==':')	xx=r-1;
			}
		}
		q[0]=-1;
		q[++r]=s.length();
		for(int i=0;i<r;i++)
		{
			if(xx==i)
			{
				if(r>8)continue;
				for(int j=1;j<=8-r;j++) cout<<"0000:";
			}
			for(int j=1;j<=5-q[i+1]+q[i];j++) cout<<"0";
			for(int j=q[i]+1;j<q[i+1];j++) cout<<s[j];
			if(i==r-1) cout<<endl;
			else cout<<":";
		}
	}
	return 0;
}
