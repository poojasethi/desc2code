#include<iostream>
#include<string>
using namespace std;
string change(string a)
{
	while (a.length()<4)
		a='0'+a;
	return a;
}
int main()
{
	int n,i,j;
	cin>>n;
	while (n--)
	{
		string s,a[10];
		bool b[10]={false};
		cin>>s;
		int k,e,l=s.length();
		e=s.find("::");
		k=1;
		for(i=0;i<e;i++)
		{
			if (s[i]!=':')
			{
				a[k]=a[k]+s[i];
				b[k]=true;
			}
			else
				k++;
		}
		k=8;
		for(i=l-1;i>e;i--)
		{
			if (s[i]!=':')
			{
				a[k]=s[i]+a[k];
				b[k]=true;
			}
			else
				k--;
		}
		for(i=1;i<=8;i++)
		{
			if (b[i])
				cout<<change(a[i]);
			else
