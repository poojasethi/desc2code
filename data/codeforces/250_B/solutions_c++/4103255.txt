#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int main()
{
	int cas,next[105];
	char c[105];
	scanf("%d",&cas);
	while(cas--)
	{
		scanf("%s",c);
		int l=strlen(c);
		int s=0,k=l;
		for(int i=l-1;i>=0;i--)
		{
			if(c[i]==':')
			{
				s++;
				next[i]=k;
				k=i;
			}
		}
		for(int i=k;i<4;i++)
		printf("0");
		for(int i=0;i<l;i++)
		{
			if(c[i]==':')
			{
				if(c[i+1]==':')
				{
					for(int j=0;j<8-s;j++)
					printf(":0000");
				}
				else
				{
					printf(":");
					for(int j=0;j<4-(next[i]-i-1);j++)
					printf("0");
				}
			}
			else
			{
				printf("%c",c[i]);
			}
		}
		printf("\n");
	}
	return 0;
}
