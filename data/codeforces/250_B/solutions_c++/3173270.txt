#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;

void proc(char *S, vector<int> &V) {
	char *p = strtok(S, ":");
	while(p != NULL) {
		int val;
		sscanf(p, "%x", &val);
		V.push_back(val);
		p = strtok(NULL, ":");
	}
}

void solve() {
	char S[105];
	scanf("%s", S);
	vector<int> A, B;
	char *p = strstr(S, "::");
	if(p == NULL)
		proc(S, A);
	else {
		*p = NULL;
		proc(S, A);
		proc(p + 2, B);
	}

	int k = 8 - A.size() - B.size();
	for(int i = 0; i < k; ++i)
		A.push_back(0);
	A.insert(A.end(), B.begin(), B.end());

	for(int i = 0; i < 7; ++i)
		printf("%04x:", A[i]);
	printf("%04x\n", A[7]);
}

int main() {
	int n;
	scanf("%d", &n);
	for(int i = 0; i < n; ++i)
		solve();
	return 0;
}
