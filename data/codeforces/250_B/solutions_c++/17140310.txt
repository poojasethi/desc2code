#include<cstdio>
#include<vector>
#include<string>
using namespace std;
void output(string x){
	for(int i=0;i<4-x.size();i++) printf("0");
	printf("%s",x.c_str());
}
void doit(){
	vector<string> vec;
	char c;string now;
	while(1){
		c=getchar();
		if(c==':'||c=='\n') {
			if(now!="") vec.push_back(now); else{
				if(vec.begin()==vec.end()||*vec.rbegin()!="::") vec.push_back("::");
			}
			now="";
		}
		else now+=c;
		if(c=='\n') break;
	}
	for(int i=0;i<vec.size();i++){
		if(vec[i]=="::") 
			for(int j=0;j<9-vec.size();j++){
				output("0000");
				if(j<8-vec.size()) printf(":");
			}
		else output(vec[i]);
		printf("%c",(i<vec.size()-1?':':'\n'));
	}
}
int main(){
	int n;
	scanf("%d\n",&n);
	while(n--) doit();
	return 0;
}
