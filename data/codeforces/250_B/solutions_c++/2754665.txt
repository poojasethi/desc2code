#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
char s[50], frn[10][5];
int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		memset(frn, 0, sizeof(frn));
		scanf("%s", s);
		int n = strlen(s), cnt = 0, in = n, i = 0;
		if(s[0] == ':') i = 2, in = 0;
		for( ; i < n; ){
			int st = i;
			for(; i < n; i++){
				if(s[i] == ':') break;
				frn[cnt][i-st] = s[i];
			}i++; cnt++;
			if(i < n && s[i] == ':'){
				in = cnt;
				i++;
			}
		}
		bool fir = 1;
		for(i = 0; i <= cnt; i++){
			if(i == in){
				for(int j = 0; j < 8-cnt; j++){
					if(!fir) printf(":");
					printf("0000"); fir = 0;
				}
			}
			if(i == cnt) break;
			if(!fir) printf(":");
			int l = strlen(frn[i]);
			if(l < 4) for(int j = 0; j < 4-l; j++) printf("0");
			printf("%s", frn[i]); fir = 0;
		}puts("");
	}
}
