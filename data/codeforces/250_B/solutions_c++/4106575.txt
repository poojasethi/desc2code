#include<cstring>
#include<cstdio>
using namespace std;
int main()
{
	int n,i,j;
	int f[10],sum,t;
	scanf("%d",&n);
	while(n--)
	{
		char ch[1001];
		scanf("%s",ch);
		int len=strlen(ch);
		sum=0; t=1;
		memset(f,0,sizeof(f));
		for(i=0;i<len;i++)
		{
			if(ch[i]==':') 
			{
				f[t]=sum;
				sum=0;
				t++;
				if(ch[i+1]==':')
				 {
 					t=8; i++;
 					for(j=i+1;j<len;j++)
 					if(ch[j]==':') t--;
 				}		 
			}
			else {
				if(ch[i]<='9'&&ch[i]>='0') sum=sum*16+ch[i]-'0';
				else sum=sum*16+ch[i]-'a'+10;
			}
		}
		if(sum!=0) f[t]=sum;
		for(i=1;i<8;i++)
		printf("%04x:",f[i]);
		printf("%04x\n",f[i]);
		//getchar();
	}
}
