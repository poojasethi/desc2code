n = int(raw_input())
for tt in range(n):
	ip = raw_input()
	if(ip == '::'):
		print "0000:0000:0000:0000:0000:0000:0000:0000"
		continue
	ip = ip.split(':')
	rip = []
	sp = 8-len(ip)+ip.count('')
	#print len(ip)
	for i in range(len(ip)):
		if ip[i] == '':
			if(i > 0 and ip[i-1] == ''):
				continue
			for j in range(sp):
				rip.append('0000')
			if i+1 < len(ip) and ip[i+1] == '':
				i += 1
		else:
			s = (4-len(ip[i]))*'0'+ip[i]
			rip.append(s);
	ans = ':'.join(rip)
	print ans
