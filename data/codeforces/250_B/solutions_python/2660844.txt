import sys

n = int(sys.stdin.readline().strip())

for i in range(0, n):
    ip = sys.stdin.readline().strip()
    colons = 0
    for c in ip:
        if c == ':':
            colons += 1
    ip = ip.replace('::', ':'*(9-colons))
    vals = [int('0'+x, 16) for x in ip.split(':')]
    print(':'.join([('0000'+hex(x)[2:])[-4:] for x in vals]))
