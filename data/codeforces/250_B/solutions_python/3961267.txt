n = int(raw_input())
for i in xrange(n):
	line = raw_input().split(':')
	part = len(line)
	full = 8
	ans = ""
	#if line[0] == line[1] == line[2] == '':
	#	ans = '0000:' * 8
	#else :
	for x in line:
		if len(x)>0:
			ans += '0'*(4-len(x))
			ans += x + ':'
		elif len(x) == 0:
			ans += "0000:"*(full-part+1)
			full = 0
			part = 0
	print ans[:-1]
