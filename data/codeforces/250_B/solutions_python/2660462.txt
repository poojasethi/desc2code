n = input()

for _ in range(n):
    short = raw_input()
    if short[0] == ':':
        short = short[1:]
    if short[-1] == ':':
        short = short[:-1]
    short = short.split(':')

    ans = ''
    for a in short:
        if a == '':
            ans += '0000:'*(9-len(short))
        else:
            ans += '0'*(4-len(a))+a+':'
    print ans[:-1]
