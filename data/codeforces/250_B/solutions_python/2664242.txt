def p(v):
  print ':'.join(map(lambda x:'%04x'%x,v))
for _ in range(input()):
  v=map(lambda x:-1 if not x else int(x,base=16),raw_input().split(':'))
  if v[0]<0 and v[-1]<0:
    p([0]*8)
  elif v[0]<0:
    p(([0]*8+v[2:])[-8:])
  elif v[-1]<0:
    p((v[:-2]+[0]*8)[:8])
  elif -1 in v:
    u=v.index(-1)
    p(v[:u]+[0]*(9-len(v))+v[u-len(v)+1:])
  else:
    p(v)
