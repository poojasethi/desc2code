#!/usr/bin/python
n = int(raw_input()) 

def getval(vv):
    while len(vv) < 4:
        vv = '0' + vv
    return vv

for i in xrange(n):
    s = raw_input()
    li = s.split(":")
    ret = ["0000"]*8
    for pos,v in enumerate(li):
        if not v:
            break;
        ret[pos] = getval(v)
    for pos,v in enumerate(reversed(li)):
        if not v:
            break;
        ret[7-pos] = getval(v)
    print ":".join(ret)
