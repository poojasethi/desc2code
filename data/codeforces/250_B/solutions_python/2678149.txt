N = int(raw_input())

for i in range(0,N):
    line = raw_input()
    miss = 7 - line.count(':')
    
    if miss != 0:
        tok = line.split('::')
        if miss > 0:
            line = tok[0] + (':' * (miss + 2)) + tok[1]
        else:
            line = tok[0] + ':' + tok[1]
    
    tok = line.split(':')
    ans = ''
    
    for x in tok:
        if len(x) < 4:
            x = '0' * (4 - len(x)) + x
        ans += ':' + x
    
    print ans[1:]
