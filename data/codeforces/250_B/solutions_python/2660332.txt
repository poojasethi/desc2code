def work(s):
	ret = []
	for i in s.split(':'):
		j = i
		while len(j) < 4:
			j = '0' + j
		ret.append(j)
	return ret

N = input()
for i in xrange(N):
	s = raw_input()
	if s.find('::') != -1:
		s1 , s2 = s.split('::')
		ans1 = work(s1)
		ans2 = work(s2)
		ans3 = (8 - len(ans1) - len(ans2)) * ['0000']
		print ':'.join(ans1 + ans3 + ans2)
	else:
		print ':'.join(work(s))
