n=input()
l = [raw_input().split(":") for i in range(n)]
for i in range(len(l)):
    while len(l[i]) > 8:
        l[i].remove('')
    for j in range(len(l[i])):
        if len(l[i][j])==0:
            while len(l[i]) < 8:
                l[i].insert(j+1,'0')
for i in range(len(l)):
    for j in range(len(l[i])):
        if len(l[i][j])<4:
            l[i][j]=(4-len(l[i][j]))*'0'+l[i][j]
for x in l:
    print ':'.join(x)
