import re
n = input()
for i in xrange(n):
    s = raw_input()
    sp = s.split("::")
    list1 = []
    list2 = []
    if len(sp) == 2:
        for block in sp[1].split(":"):
            list2 += ["0"*(4-len(block))+block]
    for block in sp[0].split(":"):
        list1 += ["0"*(4-len(block))+block]
    for j in  xrange(8-len(list1)-len(list2)):
        list1 += ["0000"]
    print ":".join(list1+list2)
