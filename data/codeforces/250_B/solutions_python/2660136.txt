from sys import stdin

for i in range(int(raw_input())):
    ip = raw_input()
    lst = ip.split(':')
    res = []
    already = False
    c = len(filter(lambda x: x == '', lst))
    for j in  lst:
        if j == '':
            if already:
                continue
            already = True
            l = 8 - len(lst) + c
            s = ':'.join(['0000',]*l)
        else:
            s = '0' * (4-len(j)) + j
        res.append(s)

    print ':'.join(res)
