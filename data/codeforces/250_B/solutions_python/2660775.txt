n = int(raw_input())
for i in xrange(n):
    s = raw_input().strip()
    if '::' in s:
        a = list(s.partition('::'))
        if not a[0]: a[0] = '0'
        if not a[2]: a[2] = '0'
        s = a[0] + ':' + '0:' * (6 - a[0].count(':') - a[2].count(':')) + a[2]
    print ':'.join([b.zfill(4) for b in s.split(':')])
