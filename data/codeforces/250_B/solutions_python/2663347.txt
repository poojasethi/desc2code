N = int(raw_input())

for i in range(0,N):
    line = raw_input()
    miss = 7 - line.count(':')
    
    if miss > 0:
        tok = line.split('::')
        line = tok[0] + (':' * (miss + 2)) + tok[1]
    elif miss < 0 and line.count('::') > 0:
        tok = line.split('::')
        line = tok[0] + ':' + tok[1]
    
    tok = line.split(':')
    ans = ''
    
    for cur in tok:
        x = cur
        while len(x) < 4:
            x = '0' + x
        ans += ':' + x
    
    print ans[1:]
