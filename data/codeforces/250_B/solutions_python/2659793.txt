n = int(raw_input())

for i in range(n):
	str = raw_input().split('::')

	if len(str[0]) > 0:
		cnt = len(str[0].split(':'))
	else:
		cnt = 0

	if len(str) == 2 and str[1] != '':
		cnt += len(str[1].split(':'))

	cnt = 8 - cnt 

	addr = ''
	if len(str[0]) > 0:
		for j in str[0].split(':'):
			addr += j.rjust(4,'0') + ':'

	for j in range(cnt):
		addr += '0000:'

	if len(str) == 2 and len(str[1]) > 0:
		for j in str[1].split(':'):
			addr += j.rjust(4,'0') + ':'

	print addr.rstrip(':')
