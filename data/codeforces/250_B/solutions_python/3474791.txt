def do(t):
    ret = ""
    for item in t:
        ret += "0"*(4-len(item)) + item + ":"
    return ret

n = int(raw_input())    
for i in range(n):
    ip6 = str(raw_input()).split("::")
    if len(ip6) == 2:
        prefix,postfix = ip6[0].split(":"),ip6[1].split(":")
        res = do(prefix) +"0000:"*(8-len(prefix) - len(postfix))+ do(postfix)
        print(res[:-1])
    else:
        print(do(ip6[0].split(":"))[:-1])
