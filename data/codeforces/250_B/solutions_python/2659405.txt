n = input()

for k in range(n):
    ip = raw_input().split(':')
    if '' in ip:
        ip = ip[:ip.index('')] + (8 - len(ip) + 1) * [''] + ip[ip.index('') + 1:]

    ip = ['0000' if x == '' else x for x in ip]
    for (i, ipv6) in enumerate(ip):
        if len(ipv6) < 4:
            ip[i] = '0' * (4 - len(ipv6)) + ipv6

    print ':'.join(ip)
