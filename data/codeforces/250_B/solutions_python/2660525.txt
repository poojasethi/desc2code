n = input()
for i in xrange(n):
    s = raw_input()
    k = s.find('::') 
    if  k == -1:
        log = s.split(':')        
    else:
        log = s[:k].split(':') + ['0000']*(9-len(s.split(':'))) + s[k+2:].split(':')
    for j in xrange(len(log)):
        if len(log[j]) < 4:
            log[j] = '0'*(4-len(log[j])) + log[j]
    print ':'.join(log)
