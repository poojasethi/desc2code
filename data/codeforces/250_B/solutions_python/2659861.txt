
def longify(short):
	a = short.split(":")
	b = []
	lengtha = 0
	for i in xrange(len(a)):
		if a[i] != "":
			lengtha += 1
	for i in xrange(len(a)):
		if len(a[i]) == 0:
			for j in xrange(8 - lengtha):
				b.append("0000")
				lengtha += 1
		elif len(a[i]) < 4:
			b.append("0" * (4 - len(a[i])) + a[i])
		else:
			b.append(a[i])
	return ":".join(b)



if __name__ == "__main__":
	n = int(raw_input())
	for i in xrange(n):
		short = raw_input()
		print longify(short)
