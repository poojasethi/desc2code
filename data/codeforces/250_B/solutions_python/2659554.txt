import os,sys

n = input()
for i in xrange(n):
    ip = raw_input()
    if ip[0] == ':': ip = '0'+ip
    if ip[-1] == ':': ip = ip + '0'
    group = ip.split(':')
    miss = 8-len(group)
    ot = [0,0,0,0,0,0,0,0]
    pos = 0
    for t in group:
        if t == '':
            pos += miss+1
        else:
            ot[pos] = int(t,16)
            pos += 1
    for i in xrange(7): sys.stdout.write('%04x:'%ot[i])
    sys.stdout.write('%04x\n'%ot[7])
