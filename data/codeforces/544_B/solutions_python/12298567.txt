n,k = map(int,raw_input().split())
if k > n**2/2 + n**2%2:
    print "NO"
else:
    b = [['S']*n for _ in xrange(n)]
    for x in xrange(n):
        for y in xrange(n):
            if k and (x+y) % 2 == 0:
                b[x][y] = 'L'; k -= 1
    print "YES"
    for x in b:
        print ''.join(x)
