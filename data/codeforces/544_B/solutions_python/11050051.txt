

def find_islands(n, k):
    if k > n*n / 2 + n % 2:
        return None
    return [(i,j) for i in xrange(n) for j in xrange(n) if (i + j) % 2 == 0][:k]

def main():
    n,k = tuple(map(int,raw_input().split(' ')))
    inds = find_islands(n,k)
    if inds is None:
        print 'NO'
    else:
        print 'YES'
        for i in xrange(n):
            print ''.join('L' if (i,j) in inds else 'S' for j in xrange(n))

if __name__ == '__main__':
    main()
