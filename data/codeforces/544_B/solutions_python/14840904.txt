n, k = map(int, raw_input().strip().split(' '))
if float(k) > n*n/2.0+0.5:
	print "NO"
else:
	print "YES"
	ans = [[] for i in range(n)]
	for i in range(n):
		for j in range(n):
			if k > 0 and (i+j)%2 == 0:
				ans[i].append('L')
				k -= 1
			else:
				ans[i].append('S')
	for r in ans:
		ot = ""
		for c in r:
			ot += c
		print ot
