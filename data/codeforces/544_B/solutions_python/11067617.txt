import sys
[n,k] = raw_input().split()
n=int(n)
k=int(k)
def cal():
    if(n%2==0):
        return n*n/2
    else:
        return (n+1)/2*(n+1)/2+(n-1)/2*(n-1)/2
if(k>cal()):
    print('NO')
else:
    print('YES')
    cnt=0
    for i in range(n):
        for j in range(n):
            if((i+j)%2==0 and cnt<k):
                sys.stdout.write('L')
                cnt+=1
            else:
                sys.stdout.write('S')
        print('')
