n, k = map(int, raw_input().split())

ISLAND = 'L'
SEA = 'S'

my_map = []
for i in xrange(n):
    my_map.append([])
    for j in xrange(n):
        my_map[i].append(SEA)


def display(m):
    for line in m:
        print ''.join(line)


def gen_coords(num):
    for x in xrange(num):
        for y in xrange(num):
            if not (y + x) % 2:
                yield x, y

coords_getter = gen_coords(n)
for z in xrange(k):
    try:
        a, b = next(coords_getter)
        my_map[a][b] = ISLAND
    except StopIteration:
        print 'NO'
        exit(0)

print 'YES'
display(my_map)
