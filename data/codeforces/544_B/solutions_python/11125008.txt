#!/usr/bin/python
# -*-coding:utf-8 -*-

n,k = [int(item) for item in raw_input().split()]
tbl = [[None for j in range(n)] for i in range(n)]
i,j = (0,0)
go = [[-1,0],[1,0],[0,-1],[0,1]]
while i != n:
    for item in go:
        new_i = i + item[0]
        new_j = j + item[1]
        if new_i < 0 or new_j < 0 or new_i >= n or new_j >= n:
            continue
        if tbl[new_i][new_j] == 'L':
            tbl[i][j] = 'S'
    if k > 0 and tbl[i][j] == None:
        tbl[i][j] = 'L'
        k -= 1
    else:
        tbl[i][j] = 'S'
    j += 1
    if j == n:
        i += 1
        j = 0
if k:
    print "NO"
else:
    print "YES"
    for row in tbl:
        print ''.join(row)
