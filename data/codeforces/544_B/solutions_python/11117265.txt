import math
n,k = [int(i) for i in raw_input().split(" ")]
map = []
allow = True
offset = False
for r in range(n):
    map_r = ""
    for i in range(n):
        if offset and i==0 and k>0:
            map_r+="S"
            allow = True
            continue
        if not offset and i==0 and k>0:
            map_r+="L"
            allow = False
            k-=1
            continue                    
        if allow and k > 0:
            allow = False
            map_r+="L"
            k-=1
        else:
            map_r+="S"
            allow = True

    if offset:
        offset=False
    else:
        offset=True     
    map.append(map_r)               



if k <= 0:
    print "YES"
    for r in map: print r           
else:
    print "NO"
