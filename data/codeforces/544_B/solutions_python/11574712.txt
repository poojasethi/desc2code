def solve(n, k):
	ans = []
	islands = 0
	for i in range(n):
		temp = ""
		for x in range(n):
			if islands != k:
				if (i + x) % 2 == 0:
					temp += "L"
					islands += 1
				else:
					temp += "S"
			else:
				temp += "S"
		ans.append(temp)
	if islands < k:
		return "NO"
	return ans


n1, k1 = map(int, raw_input().split(" "))
answer = solve(n1, k1)
if answer == "NO":
	print answer
else:
	print "YES"
	for i in answer:
		print i
