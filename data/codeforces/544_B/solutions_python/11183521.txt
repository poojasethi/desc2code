n, k = [int(v) for v in raw_input().split(' ')]

if k > (n * n + 1) / 2:
  print 'NO'
else:
  rows = []
  for i in xrange(n):
    row = []
    for j in xrange(n):
      if i % 2 == j % 2 and k > 0:
        row.append('L')
        k -= 1
      else:
        row.append('S')
    rows.append(''.join(row))
  print 'YES'
  print "\n".join(rows)
