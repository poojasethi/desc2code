#!/usr/bin/python

from math import ceil, floor

n, k = [int(i) for i in str(raw_input()).strip().split()]
result = ''

if k > ceil(float((n ** 2)) / 2):
    print 'NO'
else:
    for i in range(n):
        for j in range(n):
            if k > 0:
                if i % 2 == 0 and j % 2 == 0:
                    result += 'L'
                    k -= 1
                elif i % 2 == 0 and j % 2 == 1:
                    result += 'S'
                elif i % 2 == 1 and j % 2 == 0:
                    result += 'S'
                else:
                    result += 'L'
                    k -= 1
            else:
                result += 'S'
        if i < n - 1:
            result += '\n'
if len(result) > 0:
    print 'YES'        
    print result
