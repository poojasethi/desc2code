n,k=map(int,raw_input().split())
grid=[]
for i in range(n):
    ne=[]
    for j in range(n):
        ne.append('S')
    grid.append(ne)
for i in range(n):
    for j in range(n):
        if(i+j)%2==0 and k>0:
            grid[i][j]="L"
            k-=1
if(k!=0):
    print "NO"
else:
    print "YES"
    for i in range(n):
        print ''.join(grid[i])
