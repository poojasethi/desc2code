import math
n , k = map(int , raw_input().split())
idx = 0
temp = [['S'] * n for i in xrange(0,n)]

max = n * (n / 2)
if n % 2 !=0 :
    max += math.ceil(n/2.0)


if k > max :
    print "NO"
    exit()
else:
    print "YES"

for i in range(0,n):
    idx = i % 2
    for j in range(0,n):
        if idx % n ==  j and k >0:
            temp[i][idx % n ] = "L"
            k-=1
            idx+=2
    print "".join(temp[i])
