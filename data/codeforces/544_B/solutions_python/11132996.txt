n,k = map(int , raw_input().split())
lis = [['S' for i in range(n)] for j in range(n)]
for i in range(n):
	f = True if i%2 == 0 else False
	if k == 0:
		break
	for j in range(n):
		if f and k:
			lis[i][j] = "L"
			k-=1
			f = False
		elif k:
			f = True
		else:
			break
if k:
	print "NO"
else:
	print "YES"
	for i in lis: print ''.join(i)
