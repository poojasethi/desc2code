import sys
import math

n, k = map(int, raw_input().split())

if k > math.ceil(n**2/2.0):
    print "NO"
else:
    print "YES"
    for i in range(n):
        for j in range(n):
            if k > 0 and (i+j)%2 == 0:
                sys.stdout.write("L")
                k -= 1
            else:
                sys.stdout.write("S")
        sys.stdout.write("\n")
    sys.stdout.flush()
