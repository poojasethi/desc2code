#include <iostream>
using namespace std;

int main() {
	int n, s;
	cin >> n >> s;
	if (s <= (n * n + 1) / 2) {
		cout << "YES\n";
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++)
				if (s > 0 && (i + j) % 2 == 0) {
					cout << "L";
					s--;
				} else cout << "S";
			cout << "\n";
		}
	} else cout << "NO\n";
}
