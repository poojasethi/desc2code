#include<stdio.h>
int main()
{
	int n,k,i,j;
	scanf("%d%d",&n,&k);
	if(k>(n*n+1)/2)puts("NO");
	else {
		puts("YES");
		for(i=1;i<=n;i++){
			for(j=1;j<=n;j++){
			if((i+j)%2==0&&k>0)
				{printf("L");
				k--;}
				else printf("S");
		}
		printf("\n");
		}		
	}
	return 0;
}
