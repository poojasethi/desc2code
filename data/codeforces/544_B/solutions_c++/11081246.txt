#include<stdio.h>
using namespace std;
int main(){
	int n,k;
	while(scanf("%d%d",&n,&k)==2){
		if(k>(n*n+1)/2)printf("NO\n");
		else {
			printf("YES\n");
			for(int i=0;i<n;i++){
				for(int j=0;j<n;j++){
					if(!(i+j&1)&&k){
						printf("L");
						k--;
					}
					else printf("S");
				}
				printf("\n");
			}
		}
	}
}
