#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,k;
	cin>>n>>k;
	if(k>(n*n+1)/2)
		puts("NO");
	else
	{
		puts("YES");
		for(int i=0,t=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
				if((i&1)==(j&1) && t!=k) putchar('L'),t++;
				else putchar('S');
			putchar('\n');
		}
	}
	return 0;
}
