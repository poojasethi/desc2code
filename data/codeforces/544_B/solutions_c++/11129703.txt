#include <iostream>

using namespace std;

int main() {
	int n, k;
	cin>>n>>k;
	if(2*k > n*n+1) {
		cout<<"NO";
		return 0;
	}
	int c = 0;
	cout<<"YES"<<endl;
	for(int i = 0; i < n;i++) {
		for(int j = 0; j < n;j++) {
			if(i % 2 == j % 2 && c < k) {
				c++;
				cout<<"L";
			} else {
				cout<<"S";
			}
		}
		cout<<endl;
	}
}
