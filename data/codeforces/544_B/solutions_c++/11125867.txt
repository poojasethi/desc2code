#include <cstdio>
int n, k;
int main () {
	scanf("%d %d", &n, &k);
	if ((n*n+1)/2<k) printf("NO\n");
	else {
		printf("YES\n");
		for (int i=1; i<=n; i++) {
			for (int j=1; j<=n; j++) {
				if (((i%2==1&&j%2==1)||(i%2==0&&j%2==0))&&k>0) {
					printf("L");
					k--;
				} else {
					printf("S");
				}
			}
			printf("\n");
		}
	}
	return 0;
}
