#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n,k;
	cin>>n>>k;
	if(n*n/2+n*n%2<k)
		cout<<"NO"<<'\n';
	else
	{
		cout<<"YES"<<'\n';
		for(i=0;i<n;++i)
		{
			for(j=0;j<n;++j)
			{
				if((i+j)%2==0&&k!=0)
				{
					cout<<'L';
					k--;
				}
				else
				{
					cout<<'S';
					
				}	
			}
			cout<<"\n";
		}
	}
}
