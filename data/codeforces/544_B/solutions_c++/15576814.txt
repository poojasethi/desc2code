#include<cstdio>
int main()
{
	int a,b,c,i,j,d=0;
	scanf("%d%d",&a,&b);
	if(a%2==0) c=a*a/2;else c=a*a/2+1;
	if(c<b) printf("NO");
	else{
		printf("YES\n");
		for(i=0;i<a;i++){
			for(j=0;j<a;j++){
				if((i+j)%2==0&&d<b) {printf("L");d++;}
				else printf("S");
			}printf("\n");
		}
	}
	return 0;
}
