#include<stdio.h>
int main()
{
	int n,k,i,j,flag=0;
	scanf("%d%d",&n,&k);
	if(k>(n*n+1)/2) printf("NO\n");
	else
	{
		printf("YES\n");
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				if(j%2==i%2&&k>0)
				{
					printf("L");
					k=k-1;
				}
				else
				{
					printf("S");
				}
			}
			printf("\n");
		}
	}
}
