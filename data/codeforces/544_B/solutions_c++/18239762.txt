#include <cstdio>

int n, k;

int main() {
	scanf("%d%d", &n, &k);
	if(2 * k - 1 > n * n) {
		puts("NO");
	}
	else {
		puts("YES");
		for(int i = 0; i < n; i++) {
			for(int j = 0; j < n; j++) {
				if(k > 0 && ((i ^ j) & 1) == 0) {
					putchar('L');
					k--;
				}
				else {
					putchar('S');
				}
			}
			puts("");
		}
	}
	return 0;
}
