#include<stdio.h>
int main()
{
	int n,m,i,j,count=0;
	scanf("%d %d",&n,&m);
	if(m>(n*n+1)/2)
	{
		printf("NO");
		return 0;
	}
	printf("YES\n");
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			if(((i+j)%2==0)&&(count<m))
			{
				count++;
				printf("L");
			}
			else
				printf("S");
		}
		printf("\n");
	}
	return 0;
}
