#include <iostream>
using namespace std;

int main()
{
  int n, k;
  cin>>n>>k;
  if((n*n+1)/2 < k)
  {
    cout<<"NO\n";
    return 0;
  }
  cout<<"YES\n";
  for(int i=0; i<n; ++i,cout<<"\n")
  for(int j=0; j<n; ++j)
  {
    if(k && (i+j)%2==0) {cout<<"L"; k--;}
    else cout<<"S";
  }
  return 0;
}
