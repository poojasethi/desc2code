#include<cstdio>
int n, k, lim;

int main(){
	scanf("%d%d", &n, &k);
	lim = (n*n+1)/2;
	if(k>lim){
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	for(int i=0;i<n;i++) {
		for(int j=0;j<n;j++)
			if((i+j)&1) printf("S");
			else if(k*2>i*n+j)printf("L");
			else printf("S");
		printf("\n");
	}
	
	return 0;
}
