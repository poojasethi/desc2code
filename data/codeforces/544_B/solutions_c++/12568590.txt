#include <stdio.h>

int i,j,m,n;

int main()
{
	scanf("%d%d",&n,&m);
	if (m>(n*n+1)/2)
	{
		printf("NO");
		return 0;
	}
	printf("YES\n");
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=n;j++)
		{
			if (((i+j)%2==0) && (m>0))
			{
				m--;
				printf("L");
			}
			else
				printf("S");
		}
		printf("\n");
	}
	return 0;
}
