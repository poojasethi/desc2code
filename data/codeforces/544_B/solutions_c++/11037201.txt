#include <iostream>
#include <string>

using namespace std;
int main(){
	int n,k;
	cin>>n>>k;
	int d=0;
	if ((n*n+1)/2 >=k){
		cout<<"YES\n";
		for (int i=0; i<n ; i++){
			for (int j=0; j<n; j++)
				if ((i+j)%2==0 && d<k) {
					cout<<"L";
					d++;
				}
				else cout<<"S";
			cout<<"\n";
		}
	}
	else cout<<"NO";
	return 0;
}
