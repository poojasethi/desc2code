#include<iostream>
#include<math.h>
#include<string>
using namespace std;
int main()
{
double n,k;
cin>>n>>k;
string s;

if(pow(ceil(n/2),2)+pow(floor(n/2),2)<k)
{
cout<<"NO"<<endl;
return 0;
}

cout<<"YES"<<"\n";
for(int i=0;i<n;i++)
{

for(int j=0;j<n;j++)
{
if(k!=0 && (i+j)%2==0)
{
s+="L";
k--;
}
else s+="S";
}
s+="\n";
}
cout<<s;
return 0;
}
