#include<iostream>
using namespace std;
int main()
{
	int n,k,ul,i,j;
	cin>>n>>k;
	if(n%2==0)
		ul=n*n/2;
	else
		ul=(((n+1)/2)*((n+1)/2))+((n/2)*(n/2));
	if(k>ul)
		cout<<"NO\n";
	else
	{
		cout<<"YES\n";
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
				if(((i+j)%2==0)&&(k>0))
				{
					cout<<'L';
					k--;
				}
				else
					cout<<'S';
			cout<<'\n';
		}
	}
	return 0;
}
