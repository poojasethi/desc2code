#include <cstdio>
#include <map>
#include <algorithm>
#include <vector>
using namespace std;

typedef map< int,vector<int> > map_vector;
map_vector point, same;

int input[100010];

int main(){
	int n, m;
	scanf("%d",&n);
	
	for(int i=0;i<n;i++){
		scanf("%d",&m);
		input[i] = m;
		point[m].push_back(i);
	}
	
	for(map_vector::iterator it = point.begin(); it != point.end(); it++){
		vector<int> p = it->second;
		for(int i=0;i<p.size();i++){
			for(int j=i+1;j<p.size();j++){
				same[p[j]-p[i]].push_back(p[i]);
			}
		}
	}
	
	int left = 0;
	
	for(map_vector::iterator it = same.begin();it != same.end(); it++){
		int n = it->first;
		vector<int> v = it->second;
		sort(v.begin(),v.end());
		
		for(int i=0;i + n <= v.size();){
			if(v[i] < left){ i++; continue; }
			int j;
			
			for(j=i+1;j<i+n;j++)if(v[j-1]+1 != v[j])break;
			if(j == i + n)left = v[j-1] + 1;
			
			i = j;
		}
	}
	
	printf("%d\n",n-left);
	for(int i=left;i<n;i++){
		printf("%d ",input[i]);
	}
	
	return 0;
}
