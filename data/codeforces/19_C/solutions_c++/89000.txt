#include<cstdio>
#include<map>
using namespace std;

int N,ans=1,a[100005];
map<int ,int > gnt;
int t[100005],nt[100005],len;

bool check(int i,int j,int l){
//	printf("%d %d %d %d\n",i,j,l,len);
	if (l==len){
//		printf("hi\n");
		return true;
	}
	
	if (i-l>=ans && j-l>=ans){
		if (a[i-l]==a[j-l]) return check(i,j,l+1);
	}else{
		return false;
	}
	return false;
}

int main(){
	gnt.clear();
	scanf("%d",&N);
	for (int i=1;i<=N;++i){
		
		scanf("%d",&a[i]);
		int GNT = gnt[a[i]];
		for (int j=GNT;j!=0;j=nt[j]){
			len = i - j;
			if (check(i,j,1)){
				ans = i-len+1;
			}
		}
		nt[i] = GNT;
		gnt[a[i]] = i;
//		printf("%d %d\n",i,ans);
	}
	printf("%d\n",N-ans+1);
	for (int i= ans;i<N;++i) printf("%d ",a[i]);
	printf("%d\n",a[N]);
	
	return 0;
}
