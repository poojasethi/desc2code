#include <bits/stdc++.h>

using namespace std;

unsigned long long mod = 13;

map<int,int> mp;

unsigned long long num[100010],b[100010];

int a[100010],nex[100010],c[100010];

int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    int ans=n+1,s;
    for(int i=n;i>=1;i--)
    {
        if(mp.find(a[i])==mp.end())
            nex[i]=n+1;
        else
            nex[i]=mp[a[i]];
        mp[a[i]]=i;
    }
    num[0]=1;
    for(int i=1;i<=n;i++)
        num[i]=num[i-1]*mod;
    for(int i=1;i<=n;i++)
        b[i]=b[i-1]+num[n-i]*a[i];
    c[n]=0;
    for(int i=n;i>=1;i--)
    {
        int now=i;
        while(nex[now]<=n)
        {
            int l=nex[now]-i;
            if(l>=ans)
                break;
            if(i+2*l>n+1)
                break;
            unsigned long long x = b[i+l-1]-b[i-1];
            unsigned long long y = b[i+2*l-1]-b[i+l-1];
            y*=num[l];
            if(x==y)
            {
                c[i]=max(i+l,c[i+l]);
                break;
            }
            now=nex[now];
        }
        if(c[i]==0)
            c[i]=c[i+1];
    }
    if(c[1]==0)
        c[1]=1;
    printf("%d\n",n-c[1]+1);
    for(int i=c[1];i<=n;i++)
        printf("%d ",a[i]);
    printf("\n");
    return 0;
}
