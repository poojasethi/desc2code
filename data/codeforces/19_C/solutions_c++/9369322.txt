#include <stdio.h>
#include <iostream>
#include <map>
#include <list>
using namespace std;
map<int,list<int> > t;
list<int>::iterator it;
int a[200000];
int main()
{
    int n;
    while(~scanf("%d",&n))
    {
        for (int i=0; i<n; i++)
        {
            scanf("%d",&a[i]);
            t[a[i]].push_back(i);
        }
        a[n]=-1;
        int ans;
        for (int i=0; i<n; i++)
        {
            list<int> e=t[a[i]];
            t[a[i]].pop_front();
            for (it=e.begin(); it!=e.end(); it++)
            {
                int x=i+1;
                int y=(*it)+1;
                while (x<(*it)&&a[x]==a[y])
                {
                    x++;
                    y++;
                }
                if (x==(*it))
                {
                    i=(*it)-1;
                    ans=(*it);
                }
            }
        }
        printf("%d\n",n-ans);
        for (int i=ans; i<n; i++)
            printf(i==n-1?"%d\n":"%d ",a[i]);
    }
    return 0;
}
