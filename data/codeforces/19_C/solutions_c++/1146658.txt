using namespace std;

map<int, int> M;
vector<int> pos[111111];
set<pair<int, int> > st;
int a[111111];

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        if (!M.count(a[i])) M[a[i]] = i;
        pos[M[a[i]]].push_back(i);
    }
    a[n] = -1;
    int s = 0;
    for (int i = 0; i < n; ++i) {
        vector<int> &w = pos[M[a[i]]];
        w.erase(w.begin());
        for (vector<int>::iterator j = w.begin(); j != w.end(); ++j) {
            int x = i;
            int y = *j;
            do {
                st.insert(make_pair(x, y));
                ++x;
                ++y;
            }
            while (x < *j && a[x] == a[y] && st.count(make_pair(x, y)) == 0);
            if (x == *j) {
                s = *j;
                i = s - 1;
                break;
            }
        }
    }
    printf("%d\n", n - s);
    for (int i = s; i < n; ++i) printf("%d ", a[i]);
    return 0;
}
