#include <bits/stdc++.h>
using namespace std;
#define N 100100
typedef pair <int, int> pii;

pii p[10*N];
map <pii, int> mp;
map <int, int> num;
int n, a[N], cnt, runs;
vector <int> v[N];

int calc(int i, int j) {
	if (a[i] != a[j]) return 0;
	pii q;
	q.first = i, q.second = j;
	if (mp.count(q)) return mp[q];
	return mp[q] = min(j - i, calc(i + 1, j + 1) + 1);
}

int main() {
//	freopen("c.in", "r", stdin);
	scanf("%d", &n);
	for (int i = 1; i <= n; i ++) scanf("%d", &a[i]);
	for (int i = 1; i <= n; i ++) {
		if (!num.count(a[i])) num[a[i]] = ++cnt;
		v[num[a[i]]].push_back(i);
	}
	for (int i = 1; i <= cnt; i ++) {
		for (int j = 0; j < v[i].size(); j ++) {
			for (int k = j + 1; k < v[i].size(); k ++) {
				int a = v[i][j], b = v[i][k];
				if (calc(a, b) == b - a) {
					p[runs].first = b - a, p[runs++].second = a;
				}
			}
		}
	}
	int x = 1;
	if (runs) {
		sort(p, p + runs);
		for (int i = 0; i < runs; i ++) {
			if (p[i].second >= x) x = p[i].first + p[i].second;
		}
	}
	printf("%d\n", n - x + 1);
	for (int i = x; i <= n; i ++) printf("%d%c", a[i], i < n ? ' ' : '\n');
	return 0;
}
