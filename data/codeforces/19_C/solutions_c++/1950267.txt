#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <cstdlib>
#include <cstring>
#include <string>
#include <vector>
#include <list>
#include <set>
#include <map>
using namespace std;

const int maxn=1e5+2;
map<int, list<int> > mapque;
int n;
int data[maxn];
int main()
{
    scanf("%d", &n);
    for(int i=1;i<=n;++i)
    {
        scanf("%d", &data[i]);
        mapque[data[i]].push_back(i);
    }    
    int step=1;
    list<int>::iterator it;
    for(int i=1;i<=n;++i)
    {
        list<int> &tmpque=mapque[data[i]];
        //tmpque.pop_front();
        for(it=tmpque.begin();it!=tmpque.end();++it)
        {
            int x=i+1;
            int y=*it+1;
            while(y<=n && x<*it && data[x]==data[y])
            {
                ++x;
                ++y;            
            }             
            if(x==*it)
            {
                i=*it-1;
                step=max(step, *it);          
            }                               
        }   
    }
    printf("%d\n", n-step+1);
    for(int i=step;i<n;++i)
        printf("%d ", data[i]);
    printf("%d\n", data[n]);
//    system("pause");
    return 0;
}
