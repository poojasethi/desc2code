#include <iostream>
#include <stdio.h>
#include <map>
#define MAXN 100010

using namespace std;

int n , a[ MAXN ] , next[ MAXN ] , cnt , pos = 1;
map < int , int > last;

int main()
{
	cin >> n;
	for( register int i = 1 ; i <= n ; i++ )
	{
		scanf( "%d" , &a[i] );
		last[ a[i] ] = n + 1;
	}
	for( register int i = n ; i ; i-- )
		next[i] = last[ a[i] ] , last[ a[i] ] = i;
	for( int i = 1 ; i <= n ; i++ )
	{
		int t = next[i];
		while( t <= n )
		{
			int l = i , r = t;
			while( a[l] == a[r] && l < t && r <= n ) l++ , r++;
			if( l == t ) pos = i = t;
			t = next[t];
		}
	}
	cout << n - pos + 1 << endl;
	for( register int i = pos ; i <= n ; i++ )
		printf( "%d " , a[i] );
	return 0;
}
