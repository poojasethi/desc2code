#include<iostream>
#include<stdio.h>
#include<set>
#include<vector>
#include<algorithm>
#include<math.h>
#include<map>
#define fori(a,b,c) for(a=b;a<c;a++)
#define ford(a,b,c) for(a=b;a>=c;a--)
#define pb push_back
#define pii pair<int,int>
#define mii map<int,int>
#define mp make_pair
using namespace std;
mii M;
int main(){
	int n,i,A[100008],l[100008],s=0,j,min,k,m=0,a;
	scanf("%d",&n);
	fori(i,0,n){
		scanf("%d",&A[i]);
		l[i]=-1;
		if(M.count(A[i])){
			l[i]=M[A[i]];	
		}
		M[A[i]]=i;
	}
	fori(i,0,n){
		min=1000000004;
		a=-1;
		for(j=l[i];j>-1 && j>=s ;j=l[j]){
			if(i-j>=min)
				break;
			m=0;
			fori(k,0,i-j){
				if(n<=k+i || A[k+i]!=A[k+j]){
					m=1;
					break;
				}			
			}
			if(!m && i+k<min){
				min=i+k;
				a=i;	
			}
			if(a>-1)
				s=max(a,s);
			//printf("%d ",s);
		}
	}
	printf("%d\n",n-s);
	fori(i,s,n)
		printf("%d ",A[i]);	
	return 0;
}
