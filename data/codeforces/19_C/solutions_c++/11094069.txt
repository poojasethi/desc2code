#include <bits/stdc++.h>

using namespace std;
const char FIN[] ="deletion.i1";
const char FOU[] ="deletion.o1";
const int MAXN = 100005;
int a[MAXN], next[MAXN], n;
map<int, int> mp;
int main()
{
    #ifndef ONLINE_JUDGE
    freopen(FIN,"r",stdin);
    freopen(FOU,"w",stdout);
    #endif // ONLINE_JUDGE
    scanf("%d",&n);
    for(int i = 0; i < n; i++){
        scanf("%d",&a[i]);
        mp[a[i]] = n;
    }
    for(int i = n-1; i >=0; i--){
        next[i] = mp[a[i]];
        mp[a[i]] = i;
    }
    int pos = 0;
    for(int i = 0; i < n; i++){
        int t = next[i];
        while( t < n){
            int x = i, y = t;
            do{ x++; y++;} while(x < t && a[x]==a[y]);
            if(x == t){
                pos = t; i = t-1;
            }
            t = next[t];
        }
    }
    printf("%d\n",n-pos);
    for(int i = pos; i < n; i++) printf("%d ",a[i]);
    return 0;
}
