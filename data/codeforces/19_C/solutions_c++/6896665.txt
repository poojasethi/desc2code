#include <algorithm>
#include <stdio.h>
#include <memory.h>
#include <map>

using namespace std;

#define NN 110000

int a[NN], next[NN];
map<int, int> mp;

int main() {
//	freopen("deletion.in", "r", stdin);
//	freopen("out.out", "w", stdout);
	
	int n;
	scanf("%d", &n);
	for (int i=0; i<n; i++) {
		scanf("%d", a+i);
		mp[a[i]]=n;
	}
	for (int i=n-1; i>=0; i--) {
		next[i]=mp[a[i]]; mp[a[i]]=i;
	}
	int pos=0;
	for (int i=0; i<n; i++) {
		int t=next[i];
		while (t<n) {
			int x=i, y=t;
			do x++, y++; while (x<t && a[x]==a[y]);
			if (x==t) {
				pos=t;
				i=t-1;
			}
			t=next[t];
		}
	}
	printf("%d\n", n-pos);
	while (pos<n) printf("%d ", a[pos++]);
	
	return 0;
}
