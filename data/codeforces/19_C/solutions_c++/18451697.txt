#include <cstdio>
#include <map>
#include <list>
using namespace std;

const int MAXN = 100001;
int a[MAXN];
map<int, list<int> > mp;


int main()
{
    int n, ans;

    // freopen("D:\\1.in", "r", stdin);
    // freopen("D:\\1.out", "w", stdout);


    scanf("%d", &n);
    for(int i = 0; i < n; i ++)
        scanf("%d", &(a[i])), mp[a[i]].push_back(i);
    a[n] = -1;


    ans = 0;
    for(int i = 0; i < n; i ++)
    {
        int orii = i;

        mp[a[i]].pop_front();

        for(list<int>::iterator iter = mp[a[i]].begin(); iter != mp[a[i]].end(); iter ++)
        {
            int x = i;
            int y = (*iter);

            while(x < (*iter) && a[x] == a[y])
                x ++, y ++;

            if(x == (*iter))
                i = ans = (*iter);
        }

        for(int j = orii + 1; j <= ans; j ++)
            mp[a[j]].pop_front();
    }

    ans = n - ans;
    printf("%d\n", ans);
    for(int i = n - ans; i < n; i ++)
        printf("%d ", a[i]);


    return 0;
}
