#include <string>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>

using namespace std;

bool used[105];
string cards[105];
vector <string> ans[105];
int n, m, t, cmd = 0, player = 0, curCard, c[105], d[105][105], a[105][2], b[105][2];

inline int next(int cur) {
	do {
	cur = (cur + 1) % m;
	}	while (used[cur]);
	return cur;
}

int main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

	scanf("%d%d", &n, &t);
	for (int i = 0; i < n; i++)
		scanf("%d%d%d%d", a[i] + 0, b[i] + 0, a[i] + 1, b[i] + 1);
	scanf("%d", &m);
	for (int i = 0; i < m; i++)
		cin >> cards[i] >> c[i];

	for (int rem = 0; rem != m; ) {
		int curTime = 0;
		while (curTime < t && rem != m) {
			int that = max(1, c[curCard] - (a[cmd][player] + b[cmd][!player]) - d[cmd][curCard]);
			if (that <= t - curTime) {
				curTime += that;
				used[curCard] = 1;
				ans[cmd].push_back(cards[curCard]);
				rem++;
				if (rem != m)
					curCard = curTime == t ? curCard : next(curCard);
			}
			else {
				d[cmd][curCard] += t - curTime;
				break;
			}
		}
		if (rem != m) {
			cmd++;
			if (cmd == n) {
				player++;
				cmd = 0;
				player = !(player == 2);
			}
			curCard = next(curCard);
		}
	}

	for (int i = 0; i < n; i++) {
		printf("%d ", (int)ans[i].size());
		for (int j = 0; j < (int)ans[i].size(); j++)
			cout << ans[i][j] << " ";
		puts("");
	}

	return 0;
}
