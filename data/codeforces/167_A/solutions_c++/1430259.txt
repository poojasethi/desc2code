#include <cstdio>
#include <cmath>

int main(){
	int n;
	double a, d, d1, x, t, t1, v;
	scanf("%d %lf %lf", &n, &a, &d);
	t1 = 0;
	for (int i = 0; i < n; i++){
		scanf("%lf %lf", &x, &v);
		t = x + v / a;
		d1 = v * v / a * 0.5;
		if (d1 >= d){
			t = x + sqrt(2.0 * d / a);
			if (t < t1) t = t1;
		} else {
			t += (d-d1)/v;
			if (t < t1) t = t1;
		}
		printf("%lf\n", t);
		t1 = t;
	}
//	while(1);
	return 0;
}
