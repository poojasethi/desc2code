#include<bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    double a,d,now=0,v,tt,t;
    scanf("%d%lf%lf",&n,&a,&d);
    for(int i=0;i<n;i++)
    {
        scanf("%lf%lf",&tt,&v);
        if(v*v/(2.0*a)<d)t=tt+(d*2.0/v+v/a)/2.0;else t=tt+sqrt(d*2.0/a);
        if(t>now) now=t;
        printf("%.10f\n",now);
    }
    return 0;
}
