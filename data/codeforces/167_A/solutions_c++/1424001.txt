#include <stdio.h>
#include <string.h>
#include <math.h>
int main(){
	double a,d;
	int n;
	scanf("%d%lf%lf",&n,&a,&d);
	double t=0;
	for (int i=0;i<n;i++){
		double v,st;
		scanf("%lf%lf",&st,&v);
		double sa=v*v/(2*a);
		double ct;
		if (sa>d-1e-8){
			ct=sqrt(2*d/a);
		}else{
			ct=(v/a)+(d-sa)/v;
		}
		ct+=st;
		if (ct<t) ct=t;
		t=ct;
		printf("%.8lf\n",t);
	}
	return 0;
}
