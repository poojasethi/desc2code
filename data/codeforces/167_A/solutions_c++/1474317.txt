#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;

int main ()
{
	int n;
	double a, d, maxt(0.0), t, v;
	scanf("%d%lf%lf", &n, &a, &d);
	while (n--)
	{
		scanf("%lf%lf", &t, &v);
		double s = v * v / (2 * a), ti;
		if (s > d) ti = sqrt(2 * d / a) + t;
		else ti = v / a + (d - s) / v + t;
		printf("%.8f\n", maxt = max(maxt, ti));
	}
	return 0;
}
