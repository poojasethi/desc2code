#include<cstdio>
#include<cmath>
using namespace std;
int n,a,d;
double v[100005],t[100005],dd,ac=0,tmp;
int main(){
	scanf("%d%d%d",&n,&a,&d);
	for (int i=1; i<=n; i++)
	    scanf("%lf%lf",&t[i],&v[i]);
	for (int i=1; i<=n; i++){
		dd=d;
		tmp=v[i]/a;
		if (tmp*tmp*a*0.5>dd){
		   tmp=sqrt(2.0*d/a);
		}
		else{
			 dd-=tmp*tmp*a*0.5;
			 tmp+=dd/v[i];
 		 }
		if (tmp+t[i]>ac) ac=tmp+t[i];
		printf("%lf\n",ac);
	}
	return 0;
}
