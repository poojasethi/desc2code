#include<cstdio>
#include<iostream>
#include<cmath>

using namespace std;

int n,i;
double a,d,t,v,dt,ans;

int main()
{
    scanf("%d %lf %lf", &n, &a, &d);
    for (i=0; i<n; i++)
    {
        scanf("%lf %lf", &t, &v);
        dt=(v*v)/(2*a);
        if (d-dt>=0) ans=max(ans,t+(d-dt)/v+(v/a));
        else ans=max(ans,t+sqrt(d*2/a));
        printf("%.20lf\n", ans);
    }
}
