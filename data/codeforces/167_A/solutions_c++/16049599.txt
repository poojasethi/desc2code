#include "bits/stdc++.h"
using namespace std;
long long n, a, d, t, v, i;
double maxt;
int main() {
	scanf("%I64d%I64d%I64d", &n, &a, &d);
	maxt = 0;
	for (i = 1; i <= n; i++) {
		scanf("%I64d%I64d", &t, &v);
		if (v * v >= 2 * a * d) maxt = max(maxt, sqrt(2 * (double)d / a) + t);
		else maxt = max(maxt, (double)(v * v + 2 * a * d) / (double)(2 * a * v) + t);
		printf("%.10f\n", maxt);
	}
	return 0;
}
