#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
int n; double a,d;
double tim(double v){
  if(d<=(v*v)/(2*a))return sqrt((2*d)/a);
  return v/a + (d-(v*v)/(2*a))/v; 
}
int main(){
  cin>>n>>a>>d;
  double last=-1E18;
  while(n--){
   double st,v; cin>>st>>v;
   last=max(last , st + tim(v));
   printf("%.9lf\n" , last);
            }
  //system("pause");
  return 0;
}
