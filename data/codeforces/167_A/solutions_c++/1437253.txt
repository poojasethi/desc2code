#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cmath>

using namespace std;

long long N,A,D;
double result;

int main()
{
	cin >> N >> A >> D;
	for (int i=0;i<N;i++)
	{
		long long T,V;
		cin >> T >> V;
		if (double(V*V)/(2*A)>D)
			result=max(result,sqrt(double(D*2)/A)+T);
		else
			result=max(result,double(2*A*D+V*V)/(2*A*V)+T);
		printf("%.4lf\n",result);
	}
	return 0;
}
