#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
int n;
double a,d;

int main(){
  double ans,s=0,t,v,tt;
  scanf("%d%lf%lf",&n,&a,&d);
  for (int i=0;i<n;i++){
    scanf("%lf%lf",&t,&v);
    tt=v/a;
    if (sqrt(2*d/a)<=tt) ans=t+sqrt(2*d/a);
    else ans=t+tt+(d-0.5*a*tt*tt)/v;
    printf("%.10lf\n",max(ans,s));
    s=max(ans,s);
  }
  return 0;
}
