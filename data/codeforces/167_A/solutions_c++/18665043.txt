#include<bits/stdc++.h>
using namespace std;
double a,d,t,v,x,y;
long long  n,i;
int main()
{
    scanf("%lld%lf%lf",&n,&a,&d);
    for(i=0;i<n;i++)
     {
     	scanf("%lf%lf",&t,&v);
     	if(v*v<2*a*d)
     		x=d/v+0.5*v/a;
     	else
     		x=sqrt(2*d/a);
     	x+=t;
     	if(x>y)y=x;
     		printf("%0.12lf\n",y);
     }
	return 0;
}
