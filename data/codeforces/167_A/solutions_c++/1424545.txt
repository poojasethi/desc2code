#include<iostream>
#include<iomanip>
#include<cmath>
using namespace std;


int main() {
	int n;
	long double a,d;
	cin>>n>>a>>d;
	long double ans = 0;
	for(int i=0;i<n;i++) {
		long double v,t;
		cin>>t>>v;
		long double cur;
		long double temp = v/a;
		if(a*temp*temp/2<d) {
			cur=t+temp+(d - a*temp*temp/2)/v;
		} else {
			cur=t+sqrt(2*d/a);
		}
		ans = max(cur,ans);
		cout<<fixed<<setprecision(10)<<ans<<endl;
	}
}
