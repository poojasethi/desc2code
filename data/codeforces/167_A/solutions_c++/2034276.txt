#include<stdio.h>    
#include<math.h>
int main()   {   
    int n,a,d,i,j;
	double t,v,s,tx,tt;
    while(scanf("%d %d %d",&n,&a,&d)!=EOF){
		t=0;
		for(i=0;i<n;i++){
			scanf("%lf %lf",&tx,&v);
			s=1.0*v*v/(2*a);
			if(s>=d){
				tt=sqrt(2.0*a*d)/a;
			}
			else{
				tt=1.0*v/a;
				tt+=(d-s)*1.0/v;
			}
			if(tt+tx-t>0.00001)t=tt+tx;
			printf("%lf\n",t);
		}
	}
   return 0;   
}
