#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

int main() {
	double n,a,d;
	cin >> n >> a >> d;
	double lasttime=0;
	for (int i=0;i<n;++i) {
		double t,v;
		cin >> t >> v;
		double taccel=v/a;
		double daccel=a*taccel*taccel/2;
		double mytime;
		if (daccel>d) {
			mytime=sqrt(2*d/a);
		}
		else {
			mytime=taccel+(d-daccel)/v;
		}
		mytime=max(mytime+t,lasttime);
		printf("%.10f\n",mytime);
		lasttime=mytime;
	}
}
