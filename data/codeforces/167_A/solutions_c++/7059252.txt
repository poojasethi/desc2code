#include <stdio.h>
#include <math.h>
int main ( ) {
	double n,a,s;
	while (scanf("%lf%lf%lf",&n,&a,&s)!=EOF) {
		double t,v,max=0;
		while (n--) {
			scanf("%lf%lf",&t,&v);
			double time=t;
			double sum=v*v*1.0/(2.0*a);
			if (sum<=s) {
				sum=s-sum;
				time+=(v*1.0/a+sum*1.0/v);
			}
			else
				time+=sqrt(2.0*s/a);
			if (time>max)
				max=time;
			if (time<max)
				time=max;
			printf("%.4f\n",time);
		}
	}
	return 0;
}
