from math import sqrt

n , a , d = map(int, raw_input().split() )

lasttime = 0.0

for i in xrange(n):
    ti , vi = map(int,raw_input().split())
    t = 1.0 * vi / a
    if(0.5 * a * t * t >= d):
        lasttime = max(lasttime,ti + sqrt(2.0 * d / a ))
    else:
        dd = d - 0.5 * a * t * t
        t += dd / vi
        lasttime = max(lasttime,ti + t) 
    print "%.12f" % lasttime
