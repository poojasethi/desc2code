import sys

n, a, d = [int(s) for s in sys.stdin.readline().split()]
a, d = float(a), float(d)

trolleys = []  # (t, v)

for i in xrange(n):
    t, v = [float(s) for s in sys.stdin.readline().split()]
    trolleys.append((t, v))

def time_of_arrival(t, v):
    ttma = v / a
    dtma = 0.5 * a * ttma * ttma
    if dtma >= d:
        return t + (2 * d / a) ** 0.5
    else:
        return t + ttma + (d - dtma) / v

mt = 0
for t, v in trolleys:
    mt = max(mt, time_of_arrival(t, v))
    print mt
