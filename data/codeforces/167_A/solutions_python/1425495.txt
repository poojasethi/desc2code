import math
n,a,d = map(int, raw_input().split())

q = []
for i in range(n):
    t,v = map(float, raw_input().split())
    q.append((t,v));

lst = -1000.0
ans = []
for i in range(n):
    t,v = q[i]

    t1 = float(v) / a
    s1 = a * t1 * t1 / 2.0
    cur = t
    if s1 < d:
        cur += t1 + (d - s1) / v
    else:
        cur += math.sqrt(2.0 * d / a)
    if cur < lst: cur = lst
    ans.append(cur) 
    lst = cur

for i in range(n):
    print "%.18f" % ans[i]
