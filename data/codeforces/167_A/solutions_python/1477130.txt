#!/usr/bin/env python

from __future__ import division
import math

n,a,d = map(int, raw_input().split())

min = 0

for i in xrange(n):

	t, v = map(int, raw_input().split())

	t_0 = 1.0 * (v/a)
	d_0 = .5 * (t_0 * t_0) * a

	if d_0 > d:
		t_1 = t + math.sqrt((2*d)/a)
	else:
		t_1 = t + t_0 + (d-d_0)/v

	if min < t_1:
		min = t_1
	
	print min
