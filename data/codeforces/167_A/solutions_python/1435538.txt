from math import *
n, a, d = map(int, raw_input().split())
pre = 0
for _ in xrange(n):
    t, v = map(int, raw_input().split())
    if v ** 2 > 2 * a * d:
        time = sqrt(2.0 * d / a) + t
    else:
        time = float(d) / v + v / (2.0 * a) + t
    time = max(time, pre)
    pre = time
    print time
