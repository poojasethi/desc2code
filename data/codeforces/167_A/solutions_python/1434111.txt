import math
n, a, d = map(int, raw_input().split())
prev = -1
for i in xrange(n):
    t, v = map(int, raw_input().split())
    if v ** 2 > 2 * a * d:
        time = math.sqrt(2. * d / a) + t
    else:
        time = float(d) / v + v / (2. * a) + t
    time = max(time, prev)
    print "%.10f" % time
    prev = time
