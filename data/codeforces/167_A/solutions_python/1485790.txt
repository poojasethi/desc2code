import math
[n,a,d]=map(lambda x:int(x),(raw_input().split(' ')))

time0=0
for i in range(0,n):
	[ti, vi]=map(lambda x:float(x),(raw_input().split(' ')))
	if (vi*vi>2*a*d):
		timei=ti+math.sqrt(2*float(d)/float(a))
	else:
		timei=float(0.5*float(vi)/float(a))+float(d)/float(vi)+ti
	if timei < time0:
		timei=time0
	else:
		time0=timei
	print timei
