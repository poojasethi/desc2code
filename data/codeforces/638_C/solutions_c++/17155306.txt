#include <iostream>
#include <string.h>
#include <stdio.h>
#include <stdlib.h>
#include <vector>
#define maxn 200005
using namespace std;
vector<pair<int,int> > Edge[maxn];
vector<int>ans[maxn];
int tot;

void dfs(int x, int fa, int te)
{
    int now = 0;
    for(int i=0;i<Edge[x].size();i++)
    {
        int v = Edge[x][i].first;
        if(v == fa)continue;
        now++;
        if(now ==  te)
            now++;
        tot = max(tot,now);
        ans[now].push_back(Edge[x][i].second);
        dfs(v,x,now);
    }

}
int main()
{
    int n,i,j,x,y;
    while(cin>>n)
    {
        tot = 0;
        for(i=0;i<maxn;i++)
        {
            Edge[i].clear();
            ans[i].clear();
        }
        for(i=1;i<n;i++)
        {
            cin>>x>>y;
            Edge[x].push_back(make_pair(y,i));
            Edge[y].push_back(make_pair(x,i));
        }

        dfs(1,-1,0);
        cout<<tot<<endl;
        for(i=1;i<=tot;i++)
        {
            cout<<ans[i].size();
            for(j=0;j<ans[i].size();j++)
                cout<<" "<<ans[i][j];
            cout<<endl;
        }

    }
    return 0;
}
