#include <bits/stdc++.h>

#define mp make_pair
#define pb push_back
#define fi first
#define se second

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;
typedef unsigned long long ull;
typedef long double ld;
typedef pair<ll, ll> pll;

const int N = 1e6 + 7;
const ll INF = 1e9 + 7;
const ld EPS = 1e-9;
const ll P = 485437;
const int C = 2016;

int last[N], nxt[N], a[N], m, num[N];
bool used[N];

vector <int> e[N];

void add(int x, int y, int v, int q){
    nxt[v] = last[x];
    last[x] = v;
    a[v] = y;
    num[v] = q;
}

void init(int n){
    int x, y, v = 0;
    for (int i = 0; i < n; i++)
        last[i] = -1;
    for (int i = 0; i < n - 1; i++){
        cin >> x >> y;
        add(x - 1, y - 1, v++, i + 1);
        add(y - 1, x - 1, v++, i + 1);
    }
}

void dfs(int u, int p){
    used[u] = true;
    int x = 0;
    for (int i = last[u]; i != -1; i = nxt[i]){
        if (x == p)
            x++;
        if (!used[a[i]]){
            dfs(a[i], x);
            e[x].pb(num[i]);
            m = max(m, ++x);
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(0);
    int n;
    cin >> n;
    init(n);
    dfs(0, -1);
    cout << m << endl;
    for (int i = 0; i < m; i++){
        cout << e[i].size() << " ";
        for (int j = 0; j < e[i].size(); j++)
            cout << e[i][j] << " ";
        cout << endl;
    }
    return 0;
}
