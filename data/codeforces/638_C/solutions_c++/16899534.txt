#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

using namespace std;

#define M 200005

int n,a[M],b[M],res;

vector<int> V[M],put[M];

void dfs(int id,int prev,int tp){
  
  int cur=1;
  
  res=max(res,tp);
  FI(i,0,(int)V[id].size()-1){
    int e=V[id][i];
    int to=a[e]+b[e]-id;
    if(to==prev) continue;
    
    if(cur==tp) cur++;
    put[cur].push_back(e);
    dfs(to,id,cur);
    
    cur++;
  }
  return;
}

int main(){
  scanf("%d",&n);
  FI(i,1,n-1){
    scanf("%d %d",&a[i],&b[i]);
    V[a[i]].push_back(i);
    V[b[i]].push_back(i);
  }
  dfs(1,0,0);
  
  printf("%d\n",res);
  FI(i,1,res){
    int sz=put[i].size();
    printf("%d ",sz);
    FI(j,0,sz-1) printf("%d%c",put[i][j],j==sz-1?'\n':' ');
  }
  return 0;
}
