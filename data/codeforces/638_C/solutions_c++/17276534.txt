#include <bits/stdc++.h>
using namespace std;
vector<vector<int> > ans;
vector<pair<int,int> > g[200005];
bool vis[200005];
void dfs(int u,int p)
{
    vis[u]=true;
    int c=0;
    for(int i=0;i<g[u].size();i++)
    {
        if(!vis[g[u][i].first])
        {
            if(c==p)
                c++;
            ans[c].push_back(g[u][i].second);
            dfs(g[u][i].first,c);
            c++;
        }
    }
}
int main()
{
    int n ,m=0 ,i , j ,x ,y;
    cin>> n;
    for(i = 0; i < n-1 ;i++)
    {
        cin>>x>>y;
        g[x].push_back(make_pair(y,i));
        g[y].push_back(make_pair(x,i));
        m=max(m,(int)g[x].size());
        m=max(m,(int)g[y].size());
    }
    ans.resize(m);
    dfs(1,-1);
    cout<<m<<endl;
    for(i=0;i<m;i++)
    {
        cout<<ans[i].size()<<" ";
        for(j=0;j<ans[i].size();j++)
            cout<<ans[i][j]+1<<" ";
        cout<<endl;
    }
    return 0;
}
