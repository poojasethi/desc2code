#include<stdio.h>
#include<vector>
using namespace std;
struct aj
{
   int ct;
   int r;
   aj(int a,int b)
   {
     ct=a;
     r=b;
   }
};
vector <aj> adj[200005];
vector <int> day[200005];
int road[200005]={0},maxd=0,n,i,a,b;
void repai(int i,int rped)
{
  //printf("%d %d\n",i,rped);
   int k=0;
   for(vector<aj>::iterator it=adj[i].begin();it!=adj[i].end();it++)
   {
     if(road[it->r]!=0) continue;
     k++;
     if(k==rped) k++;
     road[it->r]=k;
     day[k].push_back(it->r);
     maxd=max(maxd,k);
     repai(it->ct,k);
   }
}
int main()
{
  scanf("%d",&n);
  for(i=1;i<n;i++)
  {
     scanf("%d %d",&a,&b);
     adj[a].push_back(aj(b,i));
     adj[b].push_back(aj(a,i));
  }
  repai(1,0);
  printf("%d\n",maxd);
  for(i=1;i<=maxd;i++)
  {
     printf("%d ",day[i].size());
     for(vector<int>::iterator it=day[i].begin();it!=day[i].end();it++)
     {
       printf("%d ",*it);
     }
     printf("\n");
  }
return 0;
}
