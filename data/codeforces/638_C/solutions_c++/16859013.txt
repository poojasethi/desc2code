#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
#include <string.h>
#include <vector>

using namespace std;

vector <pair <int, int> > a[200001];
vector <int> days[200000];

void func_ (vector <int> * days, vector <pair<int, int> > * a, int u, int v, int k) {
    int i, j = 0;
    for (i = 0; i < a[u].size(); i++) {
        if (a[u][i].first == v) continue;
        if ((j + 1) < k) {
            days[j + 1].push_back(a[u][i].second);
            func_(days, a, a[u][i].first , u, j + 1);
        }
        else if ((j + 1) >= k) {
            days[j + 2].push_back(a[u][i].second);
            func_(days, a, a[u][i].first, u, j + 2);
        }
        j++;
    }
}

int main () {
    int n, i, j, max_ = 1, u, v, d;
    pair <int, int> p1, p2;
    cin >> n;
    for (i = 1; i < n; i++) {
        cin >> p1.first >> p2.first;
        p1.second = i;
        p2.second = i;
        a[p1.first].push_back(p2);
        a[p2.first].push_back(p1);
    }
    for (i = 2; i <= n; i++) {
        if (a[max_].size() < a[i].size()) max_ = i;
    }
    cout << a[max_].size() << endl;
    d = a[max_].size();
    func_(days, a, max_, 0, n + 1);
    for (i = 1; i <= d; i++) {
        cout << days[i].size() << " ";
        for (j = 0; j < days[i].size(); j++) {
            cout << days[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
