vector<int> res[500500];
vector<pair<int,int> > e[500500];


void go(int u,int pred,int day)
{
    int curr_day = 0;
    for(int i=0;i<e[u].size();i++)
    if(e[u][i].first!=pred)
    {
        if(curr_day==day) curr_day++;
        res[curr_day].push_back(e[u][i].second);
        go(e[u][i].first,u,curr_day);
        curr_day++;
    }
}

int main()
{

    {
        int u,v;

        e[u].push_back(make_pair(v,i));
        e[v].push_back(make_pair(u,i));
    }
    go(0,-1,-1);
    int r = n;
    while(res[r].size()==0) r--;
    printf("%d\n",r+1);
    for(int i=0;i<=r;i++)

        for(int j=0;j<res[i].size();j++)
            printf(" %d",res[i][j]+1);
        printf("\n");
    }
    return 0;
}
