#include <iostream>
using namespace std;
const int MAX = 2005;
int mx[MAX], a[MAX], b[MAX];
pair<int, int> ans[MAX * MAX];
int t;
int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		int val;
		cin >> val;
		val--;
		b[val] = i;
	}
	for (int i = 0; i < n; i++)
	{
		int val;
		cin >> val;
		val--;
		a[b[val]] = i;
	}
	int sum = 0;
	for (int i = 0; i < n; i++)
	{
		int pos = -1;
		for (int j = 0; j < n; j++)
		{
			mx[j] = j;
			if (j > 0 && a[mx[j]] < a[mx[j - 1]])
				mx[j] = mx[j - 1];
			if (a[j] == i)
				pos = j;
		}
		while (pos != i)
		{
			int ind = mx[pos - 1];
			sum += pos - ind;
			swap(a[pos], a[ind]);
			ans[t++] = make_pair(pos, ind);
			pos = ind;
		}
	}
	cout << sum << "\n" << t << "\n";
	for (int i = 0; i < t; i++)
		cout << ans[i].first + 1 << " " << ans[i].second + 1 << "\n";
	return 0;
}
