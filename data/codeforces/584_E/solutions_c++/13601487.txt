#include<bits/stdc++.h>
using namespace std;
int a[5005],b[5005],desired[5005];
int main()
{
	int n,i;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	scanf("%d",&a[i]);

	for(i=1;i<=n;i++)
	scanf("%d",&b[i]);

	for(i=1;i<=n;i++)
	desired[b[i]] = i;

	int total = 0;
	vector< pair<int,int> > ans;

	int j;
	for(i=n;i>=1;i--)
	{
	int pos = i;
	for(j = i+1; j<= desired[a[pos]];j++)
	{
	if(desired[a[j]] <= pos)
	{
	int dif = pos - j;
	total += j - pos;
	ans.push_back(make_pair(j,pos));
	swap(a[j],a[pos]);
	pos = j;
	}
	}
	}
	printf("%d\n",total);
	printf("%d\n",ans.size());
	for(i=0;i<ans.size();i++)
	printf("%d %d\n",ans[i].first,ans[i].second);
}
