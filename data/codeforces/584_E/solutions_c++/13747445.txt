#include <bits/stdc++.h>

using namespace std;

int abs(int a, int b){
	if(a>b)
		return a-b;
	else
		return b-a;
}

int main(){
	int n,i,j,k,x,y,z,s,t;

	scanf("%d", &n);

	int a[n+1], b[n+1];;
	queue< pair<int,int> > q;

	for(i=1;i<n+1;i++)
		scanf("%d", &a[i]);

	for(i=1;i<n+1;i++){
		scanf("%d", &j);

		b[j]=i;
	}

	for(i=1;i<n+1;i++)
		a[i]=b[a[i]];

	s=t=0;

	for(k=n;k>0;k--){
		for(i=k;i>0 && a[i]!=k;i--);

		while(a[k]!=k){
			for(j=i+1;a[j]>i;j++);

			s+=(j-i);
			t++;
			q.push(make_pair(i,j));

			a[i]=a[j];
			a[j]=k;
			i=j;
		}
	}

	printf("%d\n%d\n", s, t);

	while(!q.empty()){
		printf("%d %d\n", q.front().first, q.front().second);

		q.pop();
	}
}
