#include <iostream>
#include <queue>
#include <cstdio>
#include <cmath>
#include <vector>
using namespace std;
const int N=1e6+5;
int a[2002],b[2002],mk[2002];
int x[N],y[N];
int ans=0;
int t=0;
int main()
{
	int n;
	cin >> n;
	for(int i=1;i<=n;i++)
		cin >> a[i];
	for(int j=1;j<=n;j++)
		cin >> b[j],mk[b[j]]=j;
	for(int i=1;i<=n;i++)
	{
		int val=b[i];
		int h;
		for(int j=1;j<=n;j++)
			if(a[j]==val)
				h=j;
		int p=h;
		for(int j=h-1;j>=i;j--)
			if(mk[a[j]]>=p)
			{
				ans+=p-j;
				t++;
				x[t]=j;
				y[t]=p;
				swap(a[j],a[p]);
				p=j;
			}
	}
	cout<<ans<<endl;
	cout<<t<<endl;
	for(int i=1;i<=t;i++)
		printf("%d %d\n",x[i],y[i]);
	return 0;
}
