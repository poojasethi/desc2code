#include<bits/stdc++.h>
using namespace std;
const int N=2020;
int n,a[N]={},b[N]={};
vector<int> u,v;
int main()
{   
    cin>>n;
    for(int i=1,x;i<=n;++i)
    {
        cin>>x;
        a[x]=i;
    }
    for(int i=1;i<=n;++i)
    {
        cin>>b[i];
        b[i]=a[b[i]];
    }
    int ans=0;
    for(int i=1;i<=n;++i)
        ans+=abs(b[i]-i);
    cout<<ans/2<<endl;
    
    for(int i=n;i>=1;--i)
    {
        int p=find(b+1,b+n+1,i)-b;
        for(int j=p+1;j<=n;++j)
            if(b[j]<=p)
            {
                u.push_back(p),v.push_back(j);
                swap(b[p],b[j]);
                p=j;
            }
    }
    printf("%u\n",u.size());
    for(int i=u.size()-1;i>=0;--i)
        printf("%d %d\n",u[i],v[i]);
    return 0;
}
