#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
#include <algorithm>
using namespace std;
#define N 2050
#define M 2000050
int a[M][2],t[N],v[N],n,m,ans,st;
inline int Read()
 {
 	int x=0;char y;
 	do y=getchar(); while (y<'0'||y>'9');
 	do x=x*10+y-'0',y=getchar(); while (y>='0'&&y<='9');
 	return x;
 }
int main()
 {
 	n=Read();
 	for (int i=1;i<=n;i++) v[i]=Read();
 	for (int i=1;i<=n;i++) t[Read()]=i;
 	for (int i=1;i<=n;i++) v[i]=t[v[i]];
 	for (int i=1;i<=n;i++)
 	 {
 	 	int k;
 	 	for (int j=1;j<=n;j++)
 	 	 if (v[j]==i) {k=j;break;}
 	 	for (int j=k-1;j>=i;j--)
 	 	 if (v[j]>=k)
 	 	   swap(v[j],v[k]),a[++st][0]=j,a[st][1]=k,
 	 	   ans+=k-j,k=j;
 	 }
 	cout <<ans<<endl<<st<<endl;
 	for (int i=1;i<=st;i++)
 	  printf("%d %d\n",a[i][0],a[i][1]);
 	return 0;
 }
