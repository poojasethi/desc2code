#include <cstdio>

int n;
int id[2200],l[2200000],r[2200000];
int a[2200],b[2200],t[2200];

int main()
{
	long long ans=0;
	int k=0,temp;
	int x,u,flag=1;
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
		scanf("%d",a+i);
	for (int i=1;i<=n;i++)
		scanf("%d",b+i);
	for (int i=1;i<=n;i++)
		id[b[i]]=i;
	for (int i=1;i<=n;i++)
		t[i]=id[a[i]];
	/*for (int i=1;i<=n;i++)
		printf("%d ",t[i]);
	printf("\n");*/
	while(flag){
		flag=0;
		for (int i=1;i<=n;i++)
		{
			u=i;
			while (t[u]<u){
				for (int j=u-1;j>=1;j--)
					if (t[j]>j){
						x=j;
						break;
					}
				temp=t[u];
				t[u]=t[x];
				t[x]=temp;
				l[k]=u;
				r[k]=x;
				k++;
				ans+=u-x;
				flag=1;
				u=x;
			}
		}
	}
	printf("%I64d\n%d\n",ans,k);
	for (int i=0;i<k;i++)
		printf("%d %d\n",l[i],r[i]);
}
