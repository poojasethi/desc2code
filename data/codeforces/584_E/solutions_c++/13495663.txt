#include<bits/stdc++.h>
using namespace std;
int n,a[2010],b[2010],l[2010],p[2010],q[2010],s=0,ans=0,u[4000010],v[4000010];
int main(){
    scanf("%d",&n);
    for (int i=1; i<=n; i++) {scanf("%d",&a[i]); l[a[i]]=i;}
    for (int i=1; i<=n; i++) {scanf("%d",&b[i]); p[l[b[i]]]=i; q[i]=l[b[i]];}
    for (int i=n; i>=1; i--){
        int k=q[i];
        for (int j=k+1; j<=i; j++) if (p[j]<=k){
            u[++s]=k;
            v[s]=j;
            ans+=j-k;
            swap(p[k],p[j]);
            q[p[k]]=k;
            q[p[j]]=j;
            k=j;
        }
    }
    printf("%d\n",ans);
    printf("%d\n",s);
    for (int i=1; i<=s; i++) printf("%d %d\n",u[i],v[i]); 
}
