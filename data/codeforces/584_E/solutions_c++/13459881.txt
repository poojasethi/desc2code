#include <bits/stdc++.h>

using namespace std;

#define rep(i,j) (l.push_back(1+(i)), r.push_back(1+(j)), swap(p[j], p[i]), cost += abs((j)-(i)))

#define MAXN 2005

int p[MAXN], s[MAXN], pos[MAXN];

int main() {
	int n;
	scanf(" %d", &n);
	for (int i = 0; i < n; i++)
		scanf(" %d", &p[i]);
	for (int i = 0; i < n; i++)
		scanf(" %d", &s[i]), pos[s[i]] = i;

	vector<int> l, r;
	int cost = 0;
	for (int i = 1; i < n; i++) {
		if (p[i] == s[i]) continue;
		if (pos[p[i]] > i) continue;
		int k = i;
		for (int j = i-1; j >= 0; j--) {
			if (p[j] == s[j]) continue;
			if (pos[p[j]] < j) continue;
			rep(k, j), k = j;
			if (p[k] == s[k]) break;
		}
	}
	//for (int i = 0; i < n; i++) printf("%d ", p[i]);
	printf("%d\n%d\n", cost, l.size());
	for (int i = 0; i < l.size(); i++)
		printf("%d %d\n", l[i], r[i]);
	return 0;
}
