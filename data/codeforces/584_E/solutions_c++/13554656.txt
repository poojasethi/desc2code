#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
const int MAXN=2010,MAXM=2000010;
int n,a[MAXN],b[MAXN],target[MAXN],ans,cnt,u[MAXM],v[MAXM];
int main(void)
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    for(int i=1;i<=n;i++)scanf("%d",&b[i]);
    for(int i=1;i<=n;i++)target[b[i]]=i;
    for(int i=n;i>=1;i--)
    {
        int p=i;
        for(int j=i+1;j<=target[a[p]];j++)
            if(target[a[j]]<=p)
            {
                ans+=j-p;
                u[cnt]=p,v[cnt]=j;
                cnt++;
                swap(a[p],a[j]);
                p=j;
            }
    }
    printf("%d\n%d\n",ans,cnt);
    for(int i=0;i<cnt;i++)printf("%d %d\n",u[i],v[i]);
    return 0;
}
