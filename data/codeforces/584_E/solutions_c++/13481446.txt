#include <bits/stdc++.h>
using namespace std;
int a[2005], b[2005], n, m, ans;
stack <int> s, t;
int main () {
    scanf("%d", &n);
    for (int i=0; i<n; i++) {
        scanf("%d", &m);
        a[m]=i+1;
    }
    for (int i=0; i<n; i++) {
        scanf("%d", &m);
        b[i+1]=a[m];
    }
    for (int i=1; i<=n; i++) {
        int j=n; 
        while (b[j]!=i) j--;
        while (j!=i) {
            int k=j-1;
            while (b[k]<j) k--;
            swap(b[j], b[k]);
            s.push(j);
            t.push(k);
            ans+=j-k;
            j=k;
        }
    }
    printf("%d\n%d\n", ans, s.size());
    while (!s.empty()) {
        printf("%d %d\n", s.top(), t.top());
        s.pop();
        t.pop();
    }
    return 0;
}
