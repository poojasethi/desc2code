#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int, int> ii;
typedef vector<int> vi;
typedef vector<ii> vii;

int n, a[2010], b[2010], pa[2010], pb[2010], inv;
vii res;

int main() {
	int i, j, k, end;
	scanf("%d", &n);
	for (i=1; i<=n; i++) {
		scanf("%d", &a[i]);
		pa[a[i]] = i;
	}
	for (i=1; i<=n; i++) {
		scanf("%d", &b[i]);
		pb[b[i]] = i;
	}
	inv = 0;
	for (i=1; i<=n; i++) {
		if (pa[b[i]] == i) continue;
		end = pa[b[i]];
		for (j=pa[b[i]]-1; j>=i; j--) {
			if (pb[a[j]] >= end) {	
				swap(a[j], a[end]);
				pa[a[j]] = j;
				pa[a[end]] = end;
				res.push_back(ii(j, end));
				inv += end-j;
				end = j;
			}
		}
	}
	printf("%d\n", inv);
	printf("%d\n", res.size());
	for (i=0; i<(int)res.size(); i++) {
		printf("%d %d\n", res[i].first, res[i].second);
	}
	return 0;
}
