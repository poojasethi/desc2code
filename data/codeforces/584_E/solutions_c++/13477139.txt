#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <string>
using namespace std;
#define inf 2005
int a[inf],p[inf],xi[inf*inf],yi[inf*inf];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	for(int i=1;i<=n;i++)
	{
		int x;
		scanf("%d",&x);
		p[x]=i;
	}
	int ans=0;
	for(int i=1;i<=n;i++)
	{
		a[i]=p[a[i]];
		ans+=abs(i-a[i]);
	}
	for(int i=1;i<=n;i++) 
	{
		p[a[i]]=i;
	}
	printf("%d\n",ans/2);
	int li=0;
	for(int i=1;i<=n;i++)
	{
		for(int j=p[i]-1;j>=i;j--)
		{
			if(a[j]>=p[i])
			{
				li++;
				xi[li]=p[i];
				yi[li]=j;
				int res=p[i];
				swap(p[a[j]],p[i]);
				swap(a[res],a[j]);
			}
		}
	}
	printf("%d\n",li);
	for(int i=1;i<=li;i++)
	{
		printf("%d %d\n",xi[i],yi[i]);
	}
	return 0;
}
