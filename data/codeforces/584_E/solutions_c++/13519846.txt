#include <cstdio>
#include <algorithm>
#define N 10005
using namespace std;
int a[N],b[N],c[N],d,e,f,g,h,i,j,k,l,m,n;
struct nd{int a,b;}ans[4000005];
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=1;i<=n;i++)scanf("%d",&b[i]);
	for(i=1;i<=n;i++)c[b[i]]=i;
	for(i=1;i<=n;i++)a[i]=c[a[i]];
	for(i=1;i<=n;i++)c[a[i]]=i;
	for(i=n;i;i--){
		k=c[i];
		for(j=k+1;j<=i;j++){
			if(a[j]<=k){
				swap(a[k],a[j]);
				swap(c[a[k]],c[a[j]]);
				ans[++m]=(nd){j,k};
				h=h+j-k;
				k=j;
			}
		}
	}
	printf("%d\n",h);
	printf("%d\n",m);
	for(i=1;i<=m;i++)printf("%d %d\n",ans[i].a,ans[i].b);
	return 0;
}
