#include <cstdio>
#define N 2010
#define M 2000010
int n, ans = 0, a[N], b[N];
int k = 0, sa[M], sb[M];
inline void swap(int &a, int &b)
{
	a ^= b ^= a ^= b;
}
int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; i++)
		scanf("%d", a + i);
	for (int i = 1; i <= n; i++)
		scanf("%d", b),
		b[*b] = i;
	for (int i = 1; i <= n; i++)
		a[i] = b[a[i]];
	for (int i = n; i; i--)
			for (int j = i + 1, p = i; a[p] > p; j++)
				if (a[j] ^ j)
					swap(a[j], a[p]),
					ans += j - p,
					sb[++k] = p,
					p = sa[k] = j;
	printf("%d\n%d\n", ans, k);
	for (int i = 1; i <= k; i++)
		printf("%d %d\n", sa[i], sb[i]);
	return 0;
}
