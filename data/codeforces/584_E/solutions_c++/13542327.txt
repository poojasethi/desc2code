#include <bits/stdc++.h>

using namespace std;

typedef pair<int,int> ii;

const int maxn=2020;

int N,ar[maxn],w[maxn],ans;
vector <ii> v;

int main(){
    scanf("%d",&N);

    for(int i=1;i<=N;i++)
        scanf("%d",&ar[i]);

    for(int i=1,a;i<=N;i++)
        scanf("%d",&a) , w[a]=i;

    for(int i=1;i<=N;i++)
        ar[i]=w[ar[i]];

    for(int i=1;i<=N;i++)
        w[ar[i]]=i;

    for(int i=N;i>=1;i--){
        int ad=w[i]+1;

        for(;ad<=N;ad++)
            if(ar[ad]<=w[i]){
                ans+=abs(w[i]-ad);
                v.push_back(ii(w[i],ad));
                swap(w[i],w[ar[ad]]);
                swap(ar[ad],ar[w[ar[ad]]]);
            }
    }

    printf("%d\n%d\n",ans,v.size());
    for(int i=0;i<v.size();i++)
        printf("%d %d\n",v[i].first,v[i].second);

    return 0;
}
