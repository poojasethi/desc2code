#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cmath> 
using namespace std;
int a[5000], b[5000], c[5000];
int st[2000000], st1[2000000], tot;
int main(){
	int n;
	scanf("%d", &n);
	for (int i = 1; i <= n; i ++) scanf("%d", &a[i]);
	for (int i = 1; i <= n; i ++){
		scanf("%d", &b[i]);
		for (int j = 1; j <= n; j ++) if(a[j] == b[i]) c[j] = i;
	}
	int cost = 0;
	for (int i = 1; i <= n; i ++){
		int ax = 0;
		for (int j = i; j <= n; j ++) if(c[j] == i){
			ax = j;
			break;
		}
		for (int j = ax - 1; j >= i; j --) if(c[j] >= ax){
			swap(c[j], c[ax]);
			st[++ tot] = j;
			st1[tot] = ax;
			cost += ax - j;
			ax = j;
		}
	}
	printf("%d\n", cost);
	printf("%d\n", tot);
	for (int i = 1; i <= tot; i++) printf("%d %d\n", st[i], st1[i]);
	return 0;
}
