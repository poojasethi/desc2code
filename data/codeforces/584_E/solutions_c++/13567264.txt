#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <fstream>
using namespace std;
int i,j,n,k,now,tot,cost,t,s,d[2001],f[2001],p[3000000][2];
int main()
{
 	scanf("%d",&n);
	for(i=0;i<=n-1;i++)
		scanf("%d",&f[i]);
	for(i=0;i<=n-1;i++)
	{
		scanf("%d",&k);
		d[k]=i;
	}
	for(i=n-1;i>=0;i--)
	{	
		if(d[f[i]]>i)
		{ 
			now=i;
			while(d[f[now]]!=now)
			{
				j=now+1;
				while(d[f[j]]>now)
				{
					j++;
				}
				p[tot][0]=j;
				p[tot][1]=now;
				cost+=abs(j-now);
				swap(f[j],f[now]);
				tot++;
				now=j;
			}
		}
	}
	printf("%d\n",cost);
	printf("%d\n",tot);
	for(i=0;i<=tot-1;i++)
    printf("%d %d\n",p[i][0]+1,p[i][1]+1);
}
