#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
#include<cmath>
#include<cstdlib>
using namespace std;
struct node{
	int x,y;
}ans[2000010];
int A[2010],B[2010],mark[2010],mark2[2010],x=0;
int main(){
	memset(mark,0,sizeof(mark));
	int n,i,j,cnt=0;
	scanf("%d",&n);
	for(i=0;i<n;i++) scanf("%d",&A[i]);
	for(i=0;i<n;i++) {scanf("%d",&B[i]);mark[B[i]]=i;}
	int ok=0;
	while(!ok){
		ok=1;
		for(i=0;i<n;i++){
			int w=i;
			if(mark[A[i]]!=i){
				ok=0;
				for(j=w+1;j<=mark[A[i]];j++){
					if(mark[A[j]]<j&&mark[A[j]]<=w){
						cnt+=abs(w-j);
						ans[x].x=w,ans[x++].y=j;
						swap(A[w],A[j]);
						w=j;
					}
				}
			}
		}
	}
	printf("%d\n%d\n",cnt,x);
	for(i=0;i<x;i++) printf("%d %d\n",ans[i].x+1,ans[i].y+1);
	return 0;
}
