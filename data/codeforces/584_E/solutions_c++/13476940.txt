#include <cstdio>
#include <algorithm>
#define N 2005

using namespace std;

int n, a[N], p[N], z, x[N * N], y[N * N], c;

int main()
{
	scanf("%d", &n);
	for(int i=1;i<=n;i++) scanf("%d", &a[i]);
	for(int i=1;i<=n;i++)
	{
		int x;
		scanf("%d", &x);
		p[x] = i;
	}
	for(int i=1;i<=n;i++)
	{
		a[i] = p[a[i]];
		z += abs(i - a[i]);
	}
	for(int i=1;i<=n;i++) p[a[i]] = i;
	printf("%d\n", z / 2);
	for(int i=1;i<=n;i++)
	{
		for(int j=p[i]-1;j>=i;j--)
		{

			if(a[j] >= p[i])
			{
				x[c] = p[i];
				y[c++] = j;
				int temp = p[i];
				swap(p[a[j]], p[i]);
				swap(a[temp], a[j]);
			}
		}
	}
	printf("%d\n", c);
	for(int i=0;i<c;i++) printf("%d %d\n", x[i], y[i]);
	scanf("\n");
}
