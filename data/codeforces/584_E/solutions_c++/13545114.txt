#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int l[2000005][2],a[2005],b[2005],id[2005];
int n,ans,tot;
void work(int k)
{
    int x=0;
    if(k==1)return;
    for(int i=1;i<=n;++i)if(a[i]==k){x=i;break;}
    for(int i=x+1;i<=n;++i)if(a[i]<=x)
    {
        swap(a[i],a[x]);
        l[++tot][0]=i;l[tot][1]=x;
        ans=ans+abs(i-x);
        x=i;
    }
    work(k-1);
}
void init()
{
    scanf("%d",&n);
    for(int i=1;i<=n;++i)scanf("%d",&a[i]);
    for(int i=1;i<=n;++i)scanf("%d",&b[i]),id[b[i]]=i;
    for(int i=1;i<=n;++i)a[i]=id[a[i]];
    work(n);
    printf("%d\n",ans);printf("%d\n",tot);
    for(int i=1;i<=tot;++i)printf("%d %d\n",l[i][0],l[i][1]);
}
int main()
{
    init();
    return 0;
}
