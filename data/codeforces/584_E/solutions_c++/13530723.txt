#include<algorithm>
#include<cstdio>

using namespace std;

int a[2100],b[2100],tra[2100],s[6000100],e[6000100];
int i,j,n,p,cnt;
long long ans;

int main(){

  scanf("%d",&n);
  for (i=1;i<=n;i++) scanf("%d",&a[i]);
  for (i=1;i<=n;i++){
    scanf("%d",&b[i]);
    tra[b[i]]=i;
  }
  for (i=1;i<=n;i++) a[i]=tra[a[i]];

  for (i=n;i>0;i--){
    if (a[i]==i) continue;
    for (j=1;j<=n;j++)
      if (a[j]==i){
        p=j;
        break;
      }
    j=p+1;
    while (p!=i){
      if (a[j]<=p){
        swap(a[j],a[p]);
        ans+=j-p;
        s[++cnt]=p;
        e[cnt]=j;
        p=j;
      }
      else j++;
    }

  }

  printf("%I64d\n",ans);
  printf("%d\n",cnt);
  for (i=1;i<=cnt;i++) printf("%d %d\n",s[i],e[i]);

  return 0;


}
