#include<stdio.h>
#include<algorithm>
using namespace std;

#define maxn 2005
typedef pair<int,int>pii;
pii b[maxn],c[maxn],ans[maxn*maxn];
int cnt,val,pos[maxn],a[maxn];

int main(){
	int n,x;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d",&x);
		a[i]=x;
	}
	for(int i=1;i<=n;i++){
		scanf("%d",&x);
		b[i]=make_pair(x,i);  
	}
	sort(b+1,b+1+n);
	for(int i=1;i<=n;i++){
		a[i]=b[a[i]].second;
		c[i].first=a[i];
		c[i].second=i;
	}
	sort(c+1,c+1+n);
	for(int i=n;i>=1;i--){ 
		int s=c[i].second;
		for(int j=s+1;j<=i;j++){
			if(a[j]<=s){
				ans[++cnt]=make_pair(s,j);
				swap(a[s],a[j]);
				val+=abs(s-j);
				s=j;
			}
		}
	}
	printf("%d\n%d\n",val,cnt);
	for(int i=1;i<=cnt;i++)
		printf("%d %d\n",ans[i].first,ans[i].second);
	return 0;
}
