#include<bits/stdc++.h>
using namespace std;
const int maxn = 2000 + 5;
int a[maxn],b[maxn];
int n;
int ansx[2000000+10],ansy[2000000+10],cur;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	int tmp ;
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&tmp);
		b[tmp] = i;
	}
	for(int i=1;i<=n;i++)
		a[i] = b[a[i]];
	long long tot = 0;
	for(int j=1;j<=n;j++)
	{
		int pos;
		for(pos=1;pos<=n;pos++)
			if(a[pos] == j)
			    break;
		if(pos == j) continue;
		for(int i = pos-1;i>=1;i--)
			if(a[i]>=pos)
		    {
		    	ansx[cur] = pos;
		    	ansy[cur++] = i;
		    	tot += pos-i;
		    	swap(a[pos],a[i]);
		    	pos = i;
		    }

	}
	printf("%I64d\n",tot);
	printf("%d\n",cur);
	for(int i=0;i<cur;i++)
		printf("%d %d\n",ansx[i],ansy[i]);
	return 0;
}
