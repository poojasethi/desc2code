#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
const int N=2005;
int n,A[N],a[N],x,ans[N*N][2],sum,tot;

int main()
{
    cin>>n;
    for (int i=1;i<=n;i++) scanf("%d",&x),A[x]=i;
    for (int i=1;i<=n;i++) 
    {
        scanf("%d",&x);
        a[A[x]]=i-A[x];
        if (a[A[x]]>0) tot+=a[A[x]];
    }
    cout<<tot<<endl;
    for (int i=n;i;i--) 
        for (int j=i+1,k=i;a[k]>0&&j<=n;j++) if (a[j]<0)
        {
            ans[++sum][0]=k; ans[sum][1]=j;
            swap(a[j],a[k]);
            a[j]-=j-k; a[k]+=j-k;
            k=j;
        }
    cout<<sum<<endl;
    for (int i=1;i<=sum;i++) printf("%d %d\n",ans[i][0],ans[i][1]);
}
