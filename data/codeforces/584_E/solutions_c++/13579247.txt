#include <cstdio>
#define N 3000
#define M 3000000
int n, ans = 0, a[N], b[N];
int k = 0, sa[M], sb[M];
inline int abs(int x)
{
	return x > 0 ? x : -x;
}
inline void swap(int &a, int &b)
{
	a ^= b ^= a ^= b;
}
int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; i++)
		scanf("%d", a + i);
	for (int i = 1; i <= n; i++)
		scanf("%d", b),
		b[*b] = i;
	for (int i = 1; i <= n; i++)
		a[i] = b[a[i]];
	for (int i = n; i; i--)
		if (a[i] > i)
			for (int j = i + 1, p = i; a[p] ^ p; j++)
				if (a[j] ^ j)
					swap(a[j], a[p]),
					ans += abs(j - p),
					sa[++k] = j,
					sb[k] = p, p = j;
	printf("%d\n%d\n", ans, k);
	for (int i = 1; i <= k; i++)
		printf("%d %d\n", sa[i], sb[i]);
	return 0;
}
