#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

typedef pair<int,int> PII;
#define MP make_pair
#define Abs(a) ((a)>=(0)?(a):(-(a)))
const int MAX_N=2010;
int N,Sn;
int A[MAX_N],B[MAX_N],Pos[MAX_N];
PII Seq[MAX_N*MAX_N];

inline void Main()
{
	scanf("%d",&N);
	for(int i=1;i<=N;i++) scanf("%d",&A[i]);
	for(int i=1;i<=N;i++){
		scanf("%d",&B[i]);
		Pos[B[i]]=i;
	}

	int ans=0;
	for(int i=N;i>=1;i--){
		int pos=i;
		for(int j=i+1;j<=Pos[A[pos]];j++){
			if(Pos[A[j]]<=pos){
				ans+=Abs(j-pos);
				Seq[++Sn]=MP(j,pos);
				swap(A[j],A[pos]);
				pos=j;
			}
		}
	}

	printf("%d\n%d\n",ans,Sn);
	for(int i=1;i<=Sn;i++)
		printf("%d %d\n",Seq[i].first,Seq[i].second);
}

int main()
{
	Main();
	return 0;
}
