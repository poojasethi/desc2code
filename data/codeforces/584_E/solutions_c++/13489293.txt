#include <stdio.h>
#include <vector>
#include <math.h>
#include <stdlib.h>
#include <algorithm>
using namespace std;
int index[2001],aa[2001],rr[2001];
int main(){
	vector<pair<int,int> >ret;
	int n,a;
	scanf("%d",&n);
	for(int i=0;i<n;++i){
		scanf("%d",&a); index[a]=i;
	}
	for(int i=0;i<n;++i)
		scanf("%d",&rr[i]);
	bool ok = true;
	int sum = 0;
	while(ok){
		ok = false;
		for(int i=0;i<n;++i){
			if(i==index[rr[i]])continue;
			ok = true;
			for(int j=i+1;j<=index[rr[i]];++j){
				if( (abs(j-index[rr[j]])>= abs(i-j)+ abs(i-index[rr[j]]) )&&index[rr[j]]!=j){
					sum+= abs(i-j);
					swap(rr[i],rr[j]);
					ret.push_back(make_pair(i+1,j+1));
				}
			}
		}
	}
	printf("%d\n%d\n",sum,(int)ret.size());
	reverse(ret.begin(),ret.end());
	for(int i=0;i<ret.size();++i)
		printf("%d %d\n",ret[i].first,ret[i].second);
}
