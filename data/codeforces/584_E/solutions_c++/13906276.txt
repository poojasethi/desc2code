#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=2e3+10;
const int mxk=2e6+10;
int s[mx],t[mx],p[mx],q[mx];
int ans[2][mxk];
int main(){
   int n,k=0;
   scanf("%d",&n);
   for(int i=0;i<n;i++){
      scanf("%d",p+i);
      s[p[i]]=i;
   }
   for(int i=0;i<n;i++){
      scanf("%d",q+i);
      t[q[i]]=i;
   }
   int f=0;
   for(int i=0;i<n;i++){
      int x=s[q[i]];
      if(x==i)continue;
      for(int j=x-1;j>=i;j--){//i  j    x
         if(t[p[j]]>=x){
            f+=x-j;
            ans[0][k]=j;
            ans[1][k++]=x;
            swap(p[j],p[x]);
            swap(s[p[j]],s[p[x]]);
            x=j;
         }
      }
   }
   printf("%d\n",f);
   printf("%d\n",k);
   for(int i=0;i<k;i++){
      printf("%d %d\n",++ans[0][i],++ans[1][i]);
   }
   return 0;
}
