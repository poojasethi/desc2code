#include <bits/stdc++.h>
using namespace std;
int main() {
  int n;
  cin>>n;
  vector<int> a,b,idx(n),p(n);
  int t;
  for(int i=0;i<n;i++){
    cin>>t;
    a.push_back(t-1);
  }
  for(int i=0;i<n;i++){
    cin>>t;
    b.push_back(t-1);
    idx[t-1]=i;
  }
  for(int i=0;i<n;i++) p[i]=idx[a[i]];
  vector<pair<int,int> >result;
  int cost=0;
  for (int i=0;i<n;i++) {
    if (p[i] < i) {
      int k=i,to=p[i];
      for (int j=i-1;j>=to;j--) {
        if(p[j] > j) {
          result.push_back(make_pair(k,j));
          swap(p[k],p[j]);
          cost+=k-j;
          k=j;
        }
      }
    }
  }
  printf("%d\n%d\n", cost, result.size());
  for(int i=0;i<result.size();i++) 
    printf("%d %d\n", result[i].first + 1, result[i].second + 1);
  return 0;
}
