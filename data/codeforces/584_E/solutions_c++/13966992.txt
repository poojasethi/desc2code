#include<cstdio>
#include<algorithm>
#define N 2015
using namespace std;
int n,a[N],b[N],c[N],cnt[2000010][2],tot,ans;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&b[i]);
		c[b[i]]=i;
	}
	for(int i=1;i<=n;i++) a[i]=c[a[i]];
	for(int i=n;i>=1;i--)
	{
		if(a[i]==i) continue;
		int p=0;
		for(int j=1;j<=n;j++)
			if(a[j]==i)
			{
				p=j;
				break;
			}
		for(int j=p+1;p<i;j++)
			if(a[j]<=p)
			{
				swap(a[j],a[p]);
				cnt[++tot][0]=j;
				cnt[tot][1]=p;
				ans+=j-p;
				p=j;
			}
	}
	printf("%d\n%d\n",ans,tot);
	for(int i=1;i<=tot;i++)
		printf("%d %d\n",cnt[i][0],cnt[i][1]);
	return 0;
}
