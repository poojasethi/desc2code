#include <bits/stdc++.h>

#define ll long long
#define fi freopen("", "r", stdin)
#define fo freopen("", "w", stdout)

using namespace std;

const int N = 2010;

int n;
int a[N], b[N], pos[N];
vector<pair<int, int> > res;

int main()
{
  fi;fo;
  cin >> n;
  for(int i = 1; i <= n; i++) cin >> a[i];
  for(int i = 1; i <= n; i++){
    cin >> b[i];
    pos[b[i]] = i;
  }
  int ans = 0;
  for(int i = 1; i <= n; i++){
    int c = 0;
    for(int j = i; j <= n; j++)
      if(b[i] == a[j]) c = j;
    ans += c - i;
    for(int j = c - 1; j >= i; j--)
      if(pos[a[j]] >= c) {
        swap(a[j], a[c]);
        res.push_back(make_pair(j, c));
        c = j;
      }
  }
  cout << ans << endl << res.size() << endl;
  for(int i = 0; i < res.size(); i++)printf("%d %d\n", res[i].first, res[i].second);
  return 0;
}
