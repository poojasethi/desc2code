#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cstdlib>
using namespace std;
const int MAXN = 2030;
int p[MAXN],val[MAXN],pos[MAXN],n,l[MAXN * MAXN],r[MAXN * MAXN];
bool vis[MAXN];

int main(){
	scanf("%d",&n);
	for(int i = 1;i <= n;++ i) scanf("%d",&p[i]);
	for(int x,i = 1;i <= n;++ i){
		scanf("%d",&x); val[x] = i;
	}
	for(int i = 1;i <= n;++ i) p[i] = val[p[i]],pos[p[i]] = i;
	int ret = 0,cnt = 0;
	for(int i = n;i > 1;-- i){
		int now = pos[i],j = now + 1;
		while(now != i){
			while(p[j] > now) ++ j;
			ret += j - now;
			l[cnt] = now; r[cnt ++] = j;
			int tmp = p[j]; p[j] = p[now]; p[now] = tmp;
			pos[p[now]] = now; pos[p[j]] = j; now = j;
		}
	}
	printf("%d\n%d\n",ret,cnt);
	for(int i = 0;i < cnt;++ i) printf("%d %d\n",l[i],r[i]);
	return 0;
}
