#include <iostream>

using namespace std;

int main()
{
    int w,m;
    cin >> w >>m;
    while(m)
    {
        int t = m%w;
        m/=w;
        if(t==1 || t==0) continue;
        if(t==w-1) {m++;continue;}
        cout <<"NO"<<endl;
        return 0;
    }
    cout <<"YES"<<endl;
    return 0;
}
