w,m = raw_input().split()
w = int(w)
m=int(m)
a=0
b=m
cur = w
if(w==2):
    print "YES"
else:
    for i in range(0,101):
        if((a+cur+b*(cur-1))%cur == (cur/w)):
            b+=cur/w;
        elif((b+cur+(cur-1)*a)%cur == (cur/w)):
            a+=cur/w;
        cur = cur*w
    if((a==b)):
        print "YES"
    else:
        print "NO"
