#!/usr/bin/python

#
# simplified version of http://codeforces.com/contest/552/submission/11648455
#
import sys

def Ni(): return tuple(map(int, sys.stdin.readline().split()))

w, m = Ni()

def canWeigh(n):
    if w == 2:
        return True

    carry = 0
    while n > 0:
        d = n % w + carry
        if d % w not in [0, 1, w-1]:
            return False

        carry = 1 if d >= w-1 else 0
        n /= w

    return True
    

print "YES" if canWeigh(m) else "NO"
