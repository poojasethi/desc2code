from __future__ import division

w, m = [int(v) for v in raw_input().split()]
# w, m = 100,50

if w == 2:
	print('YES')
	exit()

csum = 0
for k in xrange(1, 101):
	for ak in (-1, 0, 1):
		if (m - csum - ak * w**(k - 1)) % (w**k) == 0:
			break
	else:
		print('NO')
		exit()
	# print(ak)
	csum += ak * w**(k - 1)

print('YES')
