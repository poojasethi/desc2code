#!/usr/bin/python

import sys
import math
import string

def solve(w,m):
    if m == 0:
        return True
    if m % w == 1:
        return solve(w,m-1)
    if m % w == w - 1:
        return solve(w,m+1)
    if m % w == 0:
        return solve(w, m/w)
    return False

if __name__ == "__main__":  
    w, m = map(int, sys.stdin.readline().split())
    print 'YES' if solve(w, m) else 'NO'
