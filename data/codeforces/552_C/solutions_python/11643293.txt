
from math import log

def search(i, w, m):
    if i == -1:
        return m == 0
    if abs(m) > (pow(w,i+1) - 1)/(w-1):
        return False
    for k in [0, -1, 1]:
        if search(i-1, w,m + k*pow(w,i)):
            return True
    return False

w,m = tuple(int(s) for s in raw_input().strip().split(' '))
b = int(log(m, w)) + 1
if w == 2:
    print "YES"
else:
    print "YES" if search(b, w, m) else "NO"
