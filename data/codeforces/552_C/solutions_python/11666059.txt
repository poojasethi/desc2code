import sys
weights,mass = map(int,raw_input().split())
if weights == 2 or weights == 3:
    print 'yes'
    exit(0)
while mass > 0:
    if mass%weights == weights-1:
        mass = mass/weights + 1
    elif mass%weights <= 1:
        mass = mass/weights
    else:
        print 'no'
        exit(0)
print 'yes'
