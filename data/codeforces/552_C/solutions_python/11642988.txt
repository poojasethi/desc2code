[w, m] = [int(i) for i in raw_input().split()]

S = [] 
while m > 0:
    S.append(m % w)
    m /= w

for i in range(len(S)):
    if len(S) > 101:
        print 'NO'
        exit()
    if S[i] > w - 1:
        if i == len(S) - 1:
            S.append(0)
        S[i + 1] += S[i] / w
        S[i] %= w
    if S[i] != 0 and S[i] != 1 and S[i] != w - 1:
        print 'NO'
        exit()
    if S[i] == w - 1:
        if i == len(S) - 1:
            S.append(0)
        S[i + 1] += 1

print 'YES'
