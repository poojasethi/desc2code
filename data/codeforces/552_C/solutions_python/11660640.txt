import string
import math

ln1 = raw_input()
ln1 = ln1.split()
w = int(ln1[0])
m = int(ln1[1])

if (w <= 3) :print "YES"

else :
	used = set()
	def F(w, m):
		global used
		if (m == 0) : return True
		expo = int(math.log(2*m, w))
		num = math.pow(w, expo)
		if (expo in used) : return False
		used.add(expo)
		if (num > m) :
			return F(w, num - m)
		else:
			return F(w, m - num)


	if (F(w, m)): print "YES"
	else : print "NO"
