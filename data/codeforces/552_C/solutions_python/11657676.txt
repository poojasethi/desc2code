
def Controlla(n,W):
    if n == 1:
        return True
    if n % W == 0:
        return Controlla(n / W, W)
    elif (n-1) % W == 0:
        return Controlla((n-1)/W,W)
    elif (n+1)%W == 0:
        return Controlla((n+1)/W,W)
    else:
        return False

   
    
W,M = raw_input().split()
W=int(W)
M=int(M)
if Controlla(M,W):
    print "YES"
else:
    print "NO"
