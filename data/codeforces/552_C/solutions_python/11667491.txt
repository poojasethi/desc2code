def solve(x,m):
    res = [1,0,-1]
    for i in range(101):
        f = False
        for j in res:
            if (m-j)%x == 0:
                f = True
                m = (m-j)/x
                break
        if not f:
            return 'NO'
    return 'YES'
x,m = map(int,raw_input().split())
print solve(x,m)
