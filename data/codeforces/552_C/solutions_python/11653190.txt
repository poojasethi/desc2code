w, m = raw_input().split()
w, m = int(w), int(m)

a = []
while m > 0:
	a.append(m%w)
	m = m//w
#print(a)
gran = w/2.0
for i in xrange(len(a)):
	if a[i] > gran:
		a[i] = a[i] - w
	else:
		continue
	if i+1 < len(a):
		j = i+1
		while a[j] + 1 >= w:
			a[j] = 0
			j = j + 1
			if j >= len(a):
				a.append(1)
				break
		else:
			a[j] = a[j] + 1
	else:
		a.append(1)
		break
#print(a)
for i in a:
	if abs(i) > 1:
		print("NO")
		break
else:
	print("YES")
