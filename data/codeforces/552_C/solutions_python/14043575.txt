w, m = map(int, raw_input().split())

def canSolve(w, m):
	if w <= 3 or m == 1 or solve(w, m - 1) or solve(w, m) or solve(w, m + 1): 
		return True
	return False

def solve(w, m):
	if m % w == 0 and canSolve(w, m / w):
		return True
	return False

if canSolve(w, m) == True:
	print "YES"
else:
	print "NO"
