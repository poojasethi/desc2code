
w, x = map(int, raw_input().split())


class Ok:
    pass

class Bad:
    pass

def get_zzz(x):
    for new_x in (x-1, x, x+1):
        if new_x == 0:
            raise Ok
        if not new_x % w:
            new_x /= w
            get_zzz(new_x)
    raise Bad

try:
    get_zzz(x)
except Ok:
    print 'YES'
except Bad:
    print 'NO'
