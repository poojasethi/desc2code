import sys


def main():
    w, m = map(int, sys.stdin.readline().split())

    arr = [m]

    while len(arr):
        cur = arr.pop()
        while cur%w==0:
            cur//=w
        if cur ==1:
            print 'YES'
            return
        if (cur-1)%w==0:
            arr.append(cur-1)
        if (cur+1)%w==0:
            arr.append(cur+1)
    print 'NO'



main()
