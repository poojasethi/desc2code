

if __name__ == "__main__":
    W, M = map(int, raw_input().split())

    num = [0 for i in range(100)]
    pos = 0
    is_ok = True    


    while(M > 0):
        num[pos] = M % W
        M /= W
        pos += 1

  
    for i in range(len(num)):
        if(num[i] <= 1):
            continue;

        else:
            num[i] -= W
            num[i+1] += 1
            
            if(num[i] < -1):
                is_ok = False

            
    print "YES" if(is_ok) else "NO"
