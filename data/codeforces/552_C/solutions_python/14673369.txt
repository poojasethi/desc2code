w, m = map(int, raw_input().split())
tmp = m
ar = []
while tmp:  
    ar.append(tmp % w)
    tmp = tmp / w
flag = True
l=len(ar)
car=0
for i in range(0,l):
    s=(ar[i]+car)%w
    car=(ar[i]+car)//w
    if s==0: 
        continue
    elif s==1:
        continue
    elif s==w-1:
        car=1 
        continue
    else:
        flag=False
        break
print "YES" if flag else "NO"
