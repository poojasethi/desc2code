import sys

def solve():
    w, m = map(int, sys.stdin.readline().split())
   
    binstr = []
    cur = m
    while cur > 0:
        binstr.append(cur % w)
        cur = cur / w
    binstr.append(0)
    binstr.append(0)

    for e in range(len(binstr)):
        if binstr[e] > 1:
            if binstr[e] - w == -1 or binstr[e] - w == 0:
                binstr[e+1] += 1
            else:
                print "NO"
                return
    print "YES"
solve()
