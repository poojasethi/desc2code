[p, m] = [int(i) for i in raw_input().split()];

Str = [];

while m > 0: Str.append(m % p); m /= p;

for i in range(len(Str)):
    if len(Str) >= 100: print "NO"; exit();
    if Str[i] >= p:
        if i == len(Str) - 1: Str.append(0);
        Str[i + 1] += Str[i] / p; Str[i] %= p;
    if Str[i] != 0 and Str[i] != 1 and Str[i] != p - 1:
        print "NO"; exit();
    if Str[i] == p - 1:
        if i == len(Str) - 1: Str.append(0);
        Str[i + 1] += 1;

print "YES";
