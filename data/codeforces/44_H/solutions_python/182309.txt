#!/usr/bin/env python

import sys

sir = sys.stdin.readline().rstrip('\n')
l = len(sir)
sir2 = [0] * l
for i in xrange(0, l):
    sir2[i] = ord(sir[i]) - ord('0')

suma = 0

for c in xrange(0, 10):
    d = [0] * 100
    d[0] = [0] * 10
    d[0][c] = 1
    for i in xrange(1, l):
        d[i] = [0] * 10
        for j in xrange(0, 10):
            cc = j + sir2[i]
            d[i][cc / 2] += d[i - 1][j]
            if cc % 2 == 1:
                d[i][(cc / 2) + 1] += d[i - 1][j]
    for i in xrange(0, 10):
        suma += d[l - 1][i]

def test (c):
    sir3 = [0] * 100
    sir3[0] = c

    for i in xrange(1, l):
        x = sir3[i - 1] + sir2[i]
        if x % 2 == 0:
            sir3[i] = x / 2
            if sir3[i] != sir2[i]:
                return 0
        else:
            if sir2[i] != x / 2 and sir2[i] != x / 2 + 1:
                return 0
            sir3[i] = sir2[i]
    return 1

suma -= test(sir2[0])
print suma
