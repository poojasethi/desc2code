#include <cstdio>
#include <set>
#include <string>
#include <iostream>
using namespace std;
#define LL unsigned long long
LL dp[60][20];

int main()
{
//	freopen("d.in","r",stdin);
	char buf[60];
	gets(buf);
	
	int len = 0;
	while(buf[len]) len++;
	for(int i=0;i<len;i++)
		buf[i] -= '0';
	for(int i=0;i<10;i++)
		dp[len-1][i] = 1;
	for(int i = len-2;i>=0;i--)
		for(int k = 0;k<10;k++)
		{
			dp[i][k] = 0;
			int tmp = k + buf[i+1];
			dp[i][k] += dp[i+1][tmp / 2];
			if(tmp & 1)
				dp[i][k] += dp[i+1][tmp / 2 + 1];
		}
	int flag = 1;
	for(int i=1;i<len && flag;i++)
	{
		char tmp = buf[i] + buf[i-1];
		if((tmp / 2) == buf[i]) continue; 
		if((tmp & 1) && (((tmp / 2) + 1) == buf[i])) continue;
		flag = 0;
	}
	LL col = 0;
	for(int i=0;i<10;i++)
		col+= dp[0][i];			
	if(flag) col--;
	cout << col;
	return 0;
}
