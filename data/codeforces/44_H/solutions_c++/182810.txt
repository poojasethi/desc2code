#include<iostream>
#include<cstring>

using namespace std;

typedef long long ll;

ll f[55][15],res=0;
int n;
string s;

ll tinh (int v,int cs) {
    if (f[v][cs]>-1) return f[v][cs];
    if (v==n-1) {
        f[v][cs]=1;
        return 1;           
    }
    int x=cs+s[v+1]-'0';
    ll sum=tinh(v+1,x/2);
    if (x%2) sum+=tinh(v+1,(x+1)/2);
    f[v][cs]=sum;
    return sum;
}

void check() {
    int k=s[0]-'0';
    for (int i=1;i<n;i++) {
        k=(k+s[i]-'0');
        if (k/2==s[i]-'0' ||  (k+1)/2==s[i]-'0') 
            k=s[i]-'0'; else return;
    }
    res--;
}
int main () {
    cin >> s;
    n=s.length();
    memset(f,255,sizeof(f));    
    for (int i=0;i<10;i++) 
        res+=tinh (0,i);
    check();
    cout << res;
//    system("pause");
    return 0;   
}
