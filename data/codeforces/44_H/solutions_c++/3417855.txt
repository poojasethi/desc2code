#include <cstdio>
#include <cstdlib>
#include <cstring>

long long f[100][10][2];
char phone[100];

int main()
{
	scanf("%s",phone);
	int n=strlen(phone);
	memset(f,0,sizeof(f));
	for (int digit=0;digit<10;++digit){
		f[0][digit][digit!=(phone[0]-'0')]=1;
	}
	
	for (int i=0;i+1<n;++i){
		for (int digit=0;digit<10;++digit){
			for (int same=0;same<2;++same){
				if (f[i][digit][same]){
					int sum=digit+phone[i+1]-'0';
					for (int nd=0;nd<10;++nd){
						if (abs(nd*2-sum)<=1){
							f[i+1][nd][same|(nd!=phone[i+1]-'0')]+=f[i][digit][same];
						}
					}
				}
			}
		}
	}
	long long ans=0;
	for (int i=0;i<10;++i){
		ans+=f[n-1][i][1];
	}
	printf("%I64d\n",ans);
	
	return 0;
}
