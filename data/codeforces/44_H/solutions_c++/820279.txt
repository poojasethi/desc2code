#include<iostream>
#include<cstring>
using namespace std;
typedef long long Long;
string S;
Long memo[100][11][2];
Long DP(int id,int p,bool f)
{
   if(id==S.size())return  f==0;
   if(memo[id][p][f]!=-1)return memo[id][p][f];
   int av = (S[id]-'0')+p;
   Long w =  DP(id+1 , av/2 , av/2 == S[id]-'0' && f);
   if(av%2 == 1)w += DP(id+1 , av/2 + 1 , av/2+1 == S[id]-'0' && f);
   return memo[id][p][f] = w;
}


int main(int argc, char *argv[]) {

   cin>>S;
   memset(memo,-1,sizeof(memo));
   Long ans = 0;
   for(int i = 0; i <= 9 ; ++i)
   {
      ans+=DP(1 , i , i==S[0]-'0');
   }
   cout<<ans<<endl;

    return 0;
}
