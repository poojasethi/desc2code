#include <iostream>
#include <cstring>
#include <cstdio>
#include <cctype>
#include <string> 
using namespace std;
typedef unsigned long long ll; 
ll Dp[1002][10];
int main()
{
	string str;
	cin>>str;
	for (int ix=0;ix<=9;ix++)
		Dp[0][ix]=1;
	for (int ix=1;ix<str.size();ix++) 
	{
		for (int i=0;i<=9;i++)
		{
			ll num=i+str[ix]-'0';
			if (num%2)
			{
				Dp[ix][num/2]+=Dp[ix-1][i];
				Dp[ix][num/2+1]+=Dp[ix-1][i];
			}
			else
				Dp[ix][num/2]+=Dp[ix-1][i];
		}
	}
	ll ans=0;
	for (int ix=0;ix<=9;ix++)
	{
		ans+=Dp[str.size()-1][ix]; 
	}
	bool ok=true; 
	for (int ix=0;ix!=str.size()-1;ix++)
	{
		if (((str[ix]+str[ix+1]-'0'*2)+1)/2!=str[ix+1]-'0'&&((str[ix]+str[ix+1]-'0'*2))/2!=str[ix+1]-'0')
		{
			ok=false;
			break;
		}
	}
	ans-=ok; 
	cout<<ans<<endl;
}
