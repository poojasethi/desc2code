#include<bits/stdc++.h>
#define LL long long
using namespace std;
LL ans=0,dp[3][11];
char s[55];
int len;
int main(){
  cin.getline(s,55);
  len=strlen(s);
  memset(dp,0,sizeof(dp));
  for(int i=0;i<10;i++) dp[0][i]=1;
  for(int i=1;i<len;i++){
    for(int j=0;j<10;j++) dp[i%3][j]=0;
    for(int j=0;j<10;j++){
      int to=j+s[i]-'0';
      dp[i%3][to/2]+=dp[(i-1)%3][j];
      if(to&1) dp[i%3][to/2+1]+=dp[(i-1)%3][j];
    }
  }
  for(int i=0;i<10;i++) ans+=dp[(len-1)%3][i];
  ans--;
  for(int i=1;i<len;i++){
    if(abs(s[i-1]-s[i])>1){
      ans++;
      break;
    }
  }
  printf("%lld\n",ans);
  return 0;
}
