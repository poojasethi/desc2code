#include<iostream>
#include<algorithm>
#include<set>
#include<cstdio>
#include<cstdlib>
#include<cmath>
using namespace std;
#define FOR(i,j,k) for(int i=j;i<k;i++)
#define FORD(i,j,k) for(int i=j;i>k;i--)
#define LL long long
string a;
LL dp[60][10];
LL ans;
int main()
{
cin>>a;
FOR(i,0,10)
dp[0][i]=1;
FOR(i,1,a.size())
{
	int t=a[i]-'0';
	FOR(j,0,10)
	{
		int l=(t+j)/2;
		int r=(t+j+1)/2;
		dp[i][l]+=dp[i-1][j];
		if (l!=r) dp[i][r]+=dp[i-1][j];
	}
}
bool ff=0;

FOR(i,0,a.size()-1)
{
	int f=(a[i]-'0');
	int s=(a[i+1]-'0');
	if((f+s)/2!=s&&(f+s+1)/2!=s) {ff=1;break;}
}
ans=-1+ff;
FOR(i,0,10)
ans+=dp[a.size()-1][i];
cout<<ans;
return 0;
}
