#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
#include<math.h>

#define LL long long
#define S system("pause")

using namespace std;

LL DP[500000][10];
string s;
LL i,j,ans;

main()
 {
  cin>>s;
  for (i=0;i<10;i++) DP[0][i]=1;
  for (i=1;i<s.size();i++)
   for (j=0;j<10;j++)
    if ((j + (int)(s[i]-'0')) % 2 == 0) DP[i][(j+((int)(s[i]-'0')))/2]+=DP[i-1][j]; else
     {
      DP[i][(j+(int)(s[i]-'0')+1)/2]+=DP[i-1][j];
      DP[i][(j+((int)(s[i]-'0')))/2]+=DP[i-1][j];
     }
  for (i=0;i<10;i++) ans+=DP[s.size()-1][i];
  for (i=1;i<s.size();i++) if (abs(s[i]-s[i-1]) > 1) break;
  if (i == s.size()) ans--;
  cout<<ans<<endl;
 }
