#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<string>
using namespace std;
#define maxn 1005
#define inf 0x3f3f3f3f
long long dp[550][100];
long long a[55];
char ch[55];
int main()
{
	int n;
	cin>>ch;
	int len=strlen(ch);
	for(int i=0;i<len;++i)
		a[i]=ch[i]-'0';
	memset(dp,0,sizeof(dp));
	for(int i=0;i<10;++i)
		dp[0][i]=1;
	for(int i=1;i<len;++i)
	{
		for(int j=0;j<10;++j)
		{
			n=j+a[i];
			int m=n/2;
			dp[i][m]+=dp[i-1][j];
			if(n%2) dp[i][m+1]+=dp[i-1][j];
		}
	}
	long long ans=0;
	for(int i=0;i<10;++i)
		ans+=dp[len-1][i];
	bool ok=1;
	for(int i=1;i<len;++i)
	{
		int t=a[i]-a[i-1];
		if(t>1||t<-1)
		  {
			  ok=0;
			  break;
		  }
	}
	if(ok) ans--;
	cout<<ans<<endl;
}
