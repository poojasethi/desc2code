using namespace std;
long long f[55][10];
string s;
int main()
{
	cin>>s;
	int len=s.size();
	for(int i=0;i<10;i++)
		f[0][i]=1;
	for(int i=1;i<len;i++)
	{
		int t=s[i]-'0';
		for(int j=0;j<10;j++)
		{
			int l=(j+t)/2,r=(j+t+1)/2;
			f[i][l]+=f[i-1][j];
			if(l!=r)f[i][r]+=f[i-1][j];
		}
	}
	long long flag=0;
	for(int i=1;i<len;i++)
	{
		int t=s[i]-'0',j=s[i-1]-'0';
		if(t!=(t+j)/2&&t!=(t+j+1)/2)flag=1;
	}
	long long ans=flag-1;
	for(int i=0;i<10;i++)
		ans+=f[len-1][i];
	cout<<ans;
	return 0;
}
