#include <iostream>
#include <vector>
#include <set>
#include <map>
#include <list>
#include <string>
#include <cmath>
#include <algorithm>

using namespace std;

int main()
{
	string num;
	cin >> num;
	int n = (int)num.length();

	vector< vector<long long> > f(n);
	for (int i = 0; i < n; ++i)
		f[i].resize(10);
	for (int j = 0; j < 10; ++j)
		f[0][j] = 1;

	for (int i = 1; i < n; ++i)
	{
		for (int j = 0; j < 10; ++j)
		{
			int sum = j + (num[i] - '0');
			f[i][sum/2] += f[i-1][j];
			if (sum & 1)
				f[i][sum/2 + 1] += f[i-1][j];
		}
	}

	long long ans = 0;
	for (int j = 0; j < 10; ++j)
		ans += f[n-1][j];

	bool self = true;
	for (int i = 1; i < n; ++i)
		self = self && (abs(num[i-1] - num[i]) <= 1);
	if (self)
		--ans;

	cout << ans << endl;

	return 0;
}
