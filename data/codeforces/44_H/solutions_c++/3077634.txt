#include<cstdio>
#include<algorithm>
#include<cstring>

typedef unsigned long long LL;
LL f[55][11];
char s[55];

int main() {
#ifndef ONLINE_JUDGE
    freopen("11.inp", "r", stdin);
    freopen("11.out", "w", stdout);
#endif
    int n; scanf("%s",s); n = strlen(s);
    for(int dg = 0; dg < 10; ++dg) f[n-1][dg] = 1;
    for(int i = n-2; i >= 0; --i)
        for(int dg = 0; dg < 10; ++dg) {
            int sum = dg + (int) s[i+1] - 0x30;
            if(sum % 2 == 0) f[i][dg] = f[i+1][sum/2];
            else f[i][dg] = f[i+1][sum/2] + f[i+1][sum/2+1];
        }
    LL res = 0;
    for(int dg = 0; dg < 10; ++dg) res += f[0][dg];
    int subtract = 1;
    for(int i = 1; i < n; ++i)
        if(std::abs(s[i] - s[i-1]) > 1) subtract = 0;
    printf("%I64d\n", res - subtract);
    return 0;
}
