#include<stdio.h>
#include<iostream>
#include<string.h>
#include<algorithm>
using namespace std;

long long dp[60][11];
char s[110];

int vis[60][11];
long long dfs(int a,int b,int now)
{
    if(vis[a][b])
	return dp[a][b];
    vis[a][b]=1;
    if(s[a+1]==0)
	return dp[a][b]=1;
    int t=b+s[a+1]-'0';
    if(t%2)
	dp[a][b]=dfs(a+1,t/2,now*10+b)+dfs(a+1,t/2+1,now*10+b);
    else
	dp[a][b]=dfs(a+1,t/2,now*10+b);
    return dp[a][b];
}

int main()
{
    scanf("%s",s);
    long long ans1=0;
    int i;
    for(i=0;i<10;i++)
	ans1+=dfs(0,i,0);
    int is=1;
    for(i=0;s[i+1];i++)
    {
	int j=(s[i]-'0'+s[i+1]-'0');
	if(s[i+1]-'0'==j/2||(j%2&&s[i+1]-'0'==j/2+1))
	    ;
	else
	    is=0;
    }
    ans1-=is;
    cout<<ans1<<endl;
}
