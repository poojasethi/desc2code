#include <stdio.h>
#include <string.h>
char arr[1000];
__int64 dp[100][30];
int main()
{
	int i,j;
	int num,k;
	while(scanf("%s",arr+1)!=EOF)
	{
		memset(dp,0,sizeof(dp));
		int len = strlen(arr+1);
		for(i=0;i<=9;i++)dp[1][i] = 1;
		for(i=2;i<=len;i++)
		{
			num = arr[i]-'0';
			for(j=0;j<=9;j++)
			{
				for(k=0;k<=9;k++)
				{
					if(num+k>(j-1)*2 && num+k<(j+1)*2)dp[i][j] += dp[i-1][k];
				}
			}
		}
		__int64 f = 1;
		for(i=2;i<=len;i++)
		{
			num = arr[i]-'0';
			k = arr[i-1]-'0';
			if(num + k>(num-1)*2 && num +k <(num+1)*2)continue;
			f = 0;
			break;
		}
		__int64 ans = 0;
		for(i=0;i<=9;i++)ans += dp[len][i];
		printf("%I64d\n",ans-f);
	}
	return 0;
}
