#include "iostream"
#include "string"
#include "cstring"
#include "cmath"
using namespace std;
string S;
long long dp[51][10];
int L;

long long solve(int idx,int pr)
{
    //cout<<idx<<" "<<L<<" "<<dp[idx][pr]<<"\n";
    if(dp[idx][pr]!=-1) return dp[idx][pr];
    if(idx==L) return dp[idx][pr]=1LL;
    int d = (int)S[idx]-'0';
    dp[idx][pr] =  solve( idx+1, (d+pr)/2);
    if( (pr+d)%2 == 1) dp[idx][pr]+=solve(idx+1, (d+pr+1)/2);
    return dp[idx][pr];
}

int main()
{
    cin>>S;
    L = (int)S.length();
    memset(dp,-1,sizeof(dp));
    long long res=0;
    for(int i=0;i<10;i++)
        res+=solve(1,i);
    // check if S is possible, and substract 1.
    bool possible=1;
    for(int i=1;i<L;i++)
        if( abs((double)S[i]-S[i-1])>1){possible=0;break;}
    if(possible) res-=1;
    cout<<res<<"\n";
    return 0;
}
