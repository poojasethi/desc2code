#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;

char cc[70];
long long dp[70][12];// dp[x][k]=sum( dp[x-1][u] , (u+aa[x]) / 2==k )
int main()
{
	int aa[70];
	while(~scanf("%s",cc))
	{
		int l=strlen(cc);
		for(int i=0;i<l;++i)
			aa[i+1]=cc[i]-'0';
		memset(dp,0,sizeof(dp));
		for(int fa=0;fa<10;++fa)
			dp[1][fa]=1;
		for(int x=2;x<=l;++x)
			for(int u=0;u<10;++u)
			{
				int k=u+aa[x];
				dp[x][k/2]+=dp[x-1][u];
				if((u+aa[x])&1)
					dp[x][k/2+1]+=dp[x-1][u];
			}
		bool yes=1;
		for(int i=2;i<=l;++i)
		{
			int u=aa[i-1]+aa[i];
			if(u&1)
			{
				if(aa[i]!=u/2&&aa[i]!=u/2+1)
					yes=0;
			}
			else if(u/2!=aa[i])
					yes=0;
		}
		long long ans=0;
		for(int i=0;i<10;++i)
			ans+=dp[l][i];
		printf("%I64d\n",ans-yes);
	}
	return 0;
}
