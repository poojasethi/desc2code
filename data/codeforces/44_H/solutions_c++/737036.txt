#include<stdio.h>
#include<string.h>
char s[55];
int a[55];
__int64 dp[55][15];
int main(){
	int len,i,j,k;
	__int64 sum;
	while(scanf("%s",s)!=EOF){
		len=strlen(s);
		for(i=0,j=1;i<len;i++)
		a[j++]=s[i]-'0';
		//for(i=1;i<=len;i++)printf("%d",a[i]);printf("\n");
		memset(dp,0,sizeof(dp));
		for(i=0;i<=9;i++)
		dp[1][i]=1;
		for(i=2;i<=len;i++){
			for(j=0;j<=9;j++){
				for(k=0;k<=9;k++){
					if(a[i]+k>(j-1)*2&&a[i]+k<(j+1)*2)dp[i][j]+=dp[i-1][k];
				}
			}
		}
		__int64 p=1;
		for(i=2;i<=len;i++){
			if(a[i]+a[i-1]>(a[i]-1)*2&&a[i]+a[i-1]<(a[i]+1)*2)continue;
			p=0;
			break;
		}
	        sum=0;
		for(i=0;i<=9;i++)
		sum+=dp[len][i];
		printf("%I64d\n",sum-p);
	}
	return 0;
}
