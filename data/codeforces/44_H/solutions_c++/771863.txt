#include<cstring>
#include<cstdio>
#include<iostream>
#include<string>

using namespace std;
int abs(int x){
    if (x < 0) return -x;
    return x;
}


char a[100];
long long f[60][60];
long long dp(int x, int y){
    if (a[x + 1] == 0) return 1;
    if (f[x][y] >= 0) return f[x][y];
    int p = a[x + 1] - '0';
    p = p + y;
    f[x][y] = dp(x + 1, p / 2);
    if (p & 1)
        f[x][y] = f[x][y] + dp(x + 1, (p + 1) / 2);
    return f[x][y];
}

int main(){
    gets(a);
    memset(f, 128, sizeof(f));
    long long ans = 0;
    for (int i = 0; i <= 9; i++)
        ans = ans + dp(0, i);
    bool yes = true;
    for (int i = 1; a[i] > 0; i++)
        if (abs(a[i] - a[i - 1]) > 1) yes = false;
    if (yes) ans --;
    cout << ans << endl;
    return 0;
}
