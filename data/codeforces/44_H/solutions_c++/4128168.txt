#include<string.h>

long long z[100][10]={0};
main()
{

  gets(ch);
  for(i=0;i<=9;i++)
  {
    memset(z,0,sizeof(z));
    z[0][i]=1;
    int bl=0;
    for(l=1;ch[l];l++)
    {
      for(j=0;j<=9;j++)
      {
        if(ch[l]%2==j%2)
        {
          z[l][(ch[l]-'0'+j)/2]+=z[l-1][j];
        }
        else
        {
          z[l][(ch[l]-'0'+j)/2]+=z[l-1][j];
          z[l][(ch[l]-'0'+j)/2+1]+=z[l-1][j];
        }
      }
    }
    for(k=0;k<l;k++)
    if(z[k][ch[k]-'0']==0)
    break;
    if(k==l)
    s--;
    for(j=0;j<=9;j++)
    s=s+z[l-1][j];
  }
  printf("%I64d\n",s);

}
