#include <iostream>
#include <cstring>
#include <string>
using namespace std;

long long dp[55][15];
string s;

void f(int x){
	int i;
	int val=int(s[x]-'0');
	for(i=0;i<10;i++){
		if((i+val)%2==1){
			dp[x][i]=dp[x+1][(val+i)/2]+dp[x+1][(val+i+1)/2];
		}
		else 
			dp[x][i]=dp[x+1][(val+i)/2];
	}
}

bool check(){
	int i,l=s.length();
	for(i=0;i<l-1;i++){
		if(int(s[i+1]-'0')!=int(s[i]-'0'+s[i+1]-'0')/2 && int(s[i+1]-'0')!=int(s[i]-'0'+s[i+1]-'0')/2+1)
			return false;
	}
	return true;
}

int main(){
	while(cin>>s){
		int i,l;
		l=s.length();
		memset(dp,0,sizeof(dp));
		for(i=0;i<10;i++)
			dp[l][i]=1;
		long long ans=0;
		for(i=l-1;i;i--)
			f(i);
		for(i=0;i<10;i++)
			ans+=dp[1][i];
		if(check())
			ans--;
		cout<<ans<<endl;
	}
}
