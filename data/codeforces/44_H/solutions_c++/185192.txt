#include <cstdio>
#include <iostream>
#include <cstring>

using namespace std;

#define llint long long

char s [ 53 ];
llint dp [ 53 ][ 12 ][ 3 ];
int n;

llint f ( int i, int d, int eq ){
  if ( i == n ) return eq ? 0 : 1;
  if ( dp[i][d][eq] != -1 ) return dp[i][d][eq];
  
  llint ret = 0;

  if ( i == 0 )
    for ( int nd = 0; nd <= 9; ++nd )
      ret += f ( i + 1, nd, nd == s[i] - '0' ? 1 : 0 );
  else {
    int sum = d + s[i] - '0';
    if ( sum % 2 == 0 )
      ret = f(i+1,sum/2,eq && s[i] - '0' == sum / 2 ? 1 : 0 );
    else 
      ret = f(i+1,sum/2,eq && s[i] - '0' == sum / 2 ? 1 : 0 ) + f(i+1,sum/2+1,eq && s[i] - '0' == sum/2+1 ? 1 : 0 );
  }

  return dp[i][d][eq] = ret;
}

int main ( void ){
  memset ( dp, -1, sizeof(dp) );
  scanf("%s",s);
  n = strlen(s);

  printf("%I64lld\n",f(0,0,1));

  return 0;
}
