#include <cstdio>
#include <cstring>

const int MAX_N = 57;

char s[MAX_N];
long long dp[MAX_N][10];

int main() {
     scanf("%s", s);
	 int n = strlen(s);
	 long long sum = 0;
     for (int i = 0; i < 10; i++) dp[0][i] = 1;
     for (int i = 0; i < n - 1; i++) {
	  for (int j = 0; j < 10; j++) {
	       int x = s[i + 1] - '0' + j;
	       dp[i + 1][x / 2] += dp[i][j];
	       if (x % 2 == 1) dp[i + 1][(x + 1) / 2] += dp[i][j];
	  }
     }
     for (int i = 0; i < 10; i++) sum += dp[n - 1][i];
	 int i;
     for (i = 1; i < n; i++) {
	  int x = s[i] + s[i - 1] - 2 * '0';
	  if (s[i] - '0' != x / 2 && s[i] - '0' != (x + 1) / 2) break;
     }
     if (i == n) sum--;
     printf("%I64d\n", sum);
     return 0;
}
