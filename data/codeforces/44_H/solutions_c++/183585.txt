#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
using namespace std;
unsigned long long a[52][12]={0};
int main()
{
	string s;
	cin>>s;
	int k=1;
	for(int j=0;j<=9;j++)
		a[1][j]=1;
	int n=s.size();
	for(int i=1;i<n;i++)
	{
		bool can=false;
		for(int j=0;j<10 && !can;j++)
		{
			double f=(s[i-1]-48+j)*1.0/2;
      if ((int)f==s[i]-48 || (int)ceil(f)==s[i]-48)
			  can=true;
		}
		if (!can)
		{
			k=0;
			break;
		}
	}
	for(int i=2;i<=n;i++)
		for(int j=0;j<10;j++)
		{
			double f=(s[i-1]-48+j)*1.0/2;
			if (f-(int)f==0)
				a[i][(int)f]+=a[i-1][j];
			else 
			{
				a[i][(int)f]+=a[i-1][j];
				a[i][(int)ceil(f)]+=a[i-1][j];
			}
		}
	unsigned long long sum=0;
	for(int j=0; j<10;j++)
		sum+=a[n][j];
	cout<<sum-k;
	return 0; 
}
