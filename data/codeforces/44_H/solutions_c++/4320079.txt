#include <iostream>
#include <string>
#include <cstdlib>
using namespace std;

long long  f[60][10];

int main(){
	string str;cin>>str;
	for(int i=0;i<str.length();++i){
		if(i==0){
			for(int j=0;j<=9;++j){
				f[i][j]=1;
			}
		}else{
			for(int j=0;j<=9;++j){
				int a=(j+(str[i]-'0'))/2;
				int b=(j+(str[i]-'0')+1)/2;
				if(a==b)f[i][a]+=f[i-1][j];
				else{
					f[i][a]+=f[i-1][j];
					f[i][b]+=f[i-1][j];
				}
			}
		}
	}
	long long ans=0;
	for(int j=0;j<=9;++j){
		ans+=f[str.length()-1][j];
	}

	int flag=1;
	for(int i=1;i<str.length();++i){
		if(abs(str[i]-str[i-1])>1)flag=0;
	}
	if(flag)ans--;

	cout<<ans<<endl;
	return 0;
}
