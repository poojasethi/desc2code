#include<iostream>
#include<string>
using namespace std;
int old[80],n,dp[80][10];
bool fromr[80][10];
int main(){
	string inp;
	cin>>inp;
	n=(int)inp.length();
	int xb,x2;
	for(xb=0;xb<n;xb++){
		old[xb+1]=inp[xb]-'0';
	}
	int num,mod;
	fromr[n][old[n]]=true;
	for(xb=0;xb<10;xb++) dp[n][xb]=1;
	for(xb=n-1;xb>0;xb--){
		for(x2=0;x2<10;x2++){
			num=x2+old[xb+1];
			mod=num % 2;
			num/=2;
			if(mod==0 || num+1>9){
				dp[xb][x2]=dp[xb+1][num];
				if(x2==old[xb] && fromr[xb+1][num]) fromr[xb][x2]=true;
			}
			else{
				dp[xb][x2]=dp[xb+1][num]+dp[xb+1][num+1];
				if(x2==old[xb] && (fromr[xb+1][num] || fromr[xb+1][num+1])) fromr[xb][x2]=true;
			}
		}
	}
	long long sum=0;
	for(xb=0;xb<10;xb++) sum+=(long long)dp[1][xb]-fromr[1][xb];
	cout<<sum<<endl;
	return 0;
}
