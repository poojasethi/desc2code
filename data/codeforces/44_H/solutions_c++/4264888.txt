#include<cstdio>
#include<string>
#include<iostream>
using namespace std;
long long f[100][20];
int main(){
  string a;
  cin>>a;
  int n=a.length(),t;
  for (int i=0;i<=9;i++) f[0][i]=1;
  for (int i=1;i<n;i++){
    for (int j=0;j<10;j++){
      t=j+a[i]-48;
	  f[i][t/2]+=f[i-1][j];
	  if (t&1) f[i][t/2+1]+=f[i-1][j];  
    }
  }
  long long ans=0;
  for (int i=0;i<=9;i++) ans+=f[n-1][i];
  bool ck=true;
  for (int i=1;i<n && ck;i++){
    int x=a[i]-48,y=a[i-1]-48;
	if ((x+y)&1){
	  if (a[i]-48==(x+y)/2 || a[i]-48==(x+y)/2+1) continue;
	}else{
	  if (a[i]-48==(x+y)/2) continue;
	} 
	ck=false; 
  }
  if (ck) --ans;
  cout<<ans<<endl;
  return 0;
}
