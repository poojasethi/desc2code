#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
char str[55];
int len;
__int64 dp[55][10];
__int64 dfs(int i,int pre,int doing)
{
	if(i==len&&doing==0)
		return 1;
	else if(i==len)
		return 0;
	if(dp[i][pre]!=-1&&doing==0)
		return dp[i][pre];
	int ch=str[i]-'0';
	__int64 ans=0;
	int num=(ch+pre)/2;
	ans+=dfs(i+1,num,ch==num&&doing);
	if((ch+pre)%2)
	{
		ans+=dfs(i+1,num+1,ch==num+1&&doing);
	}
	if(doing==0)
		dp[i][pre]=ans;
	return ans;
}
int main()
{
	while(~scanf("%s",str))
	{
		memset(dp,-1,sizeof(dp));
		len=strlen(str);
		int first=str[0]-'0';
		__int64 sum=0;
		for(int i=0; i<=9; i++)
		{
			sum+=dfs(1,i,i==first);
		}
		printf("%I64d\n",sum);
	}
}
