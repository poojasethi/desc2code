#include<iostream>
using namespace std;
typedef long long ll;
const ll MN=1100;
ll mark[MN][11],a[MN];
ll d[MN][11],n;
ll memz(ll x,ll y){
	if(mark[x][y]){
		return d[x][y];
	}else{
		if(x==n){
			return 1;
		}
		mark[x][y]=1;
		ll r=y+a[x];
		if(r%2==0){
			d[x][y]=memz(x+1,r/2);
		}else{
			d[x][y]=memz(x+1,r/2);
			d[x][y]+=memz(x+1,r/2+1);
		}
		return d[x][y];
	}
}
ll ok(){
	for(ll i=1;i<n;i++){
		if(!((a[i]+a[i-1])/2==a[i]||(a[i]+a[i-1]+1)/2==a[i])){
			return 0;
		}
	}
	return 1;
}
int main(){
	string s;
	cin>>s;
	n=s.size();
	for(ll i=0;i<n;i++){
		a[i]=s[i]-'0';
		cerr<<a[i]<<" ";
	}cerr<<endl;
	ll ans=0;
	for(ll i=0;i<=9;i++){
		ans+=memz(1,i);
		cerr<<ans<<"$"<<i<<endl;
	}
	if(ok()){
		ans--;
	}
	cout<<ans<<endl;
	return 0;
}
