using namespace std;
bool was[55][10][2];
long long res[55][10][2];
int x[55];
string s;
int go(int i,int j,int k)
{
	if(x[i]!=j)k=0;
	if(i+1==s.size())return 1-k;
	if(was[i][j][k])return res[i][j][k];
	was[i][j][k]=1;
	long long cur=0;
	cur+=go(i+1,(j+x[i+1])/2,k);
	if((j+x[i+1])&1)cur+=go(i+1,(j+x[i+1]+1)/2,k);
	res[i][j][k]=cur;
	return cur;
}
int main()
{
	cin>>s;
	int n=s.size();
	for(int i=0;i<n;i++)
		x[i]=s[i]-'0';
	long long ans=0;
	for(int i=0;i<10;i++)
		ans+=go(0,i,1);
	cout<<ans;
	return 0;
}
