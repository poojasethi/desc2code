#include <cstdio>
#include <cstring>
long long f[60][10], ans = 0;
char s[100];
int n, a[100];

long long work(int first)
{
	memset(f, 0, sizeof f);
	f[0][first] = 1;
	for (int i = 0; i < n - 1; ++i)
		for (int j = 0; j < 10; ++j) if (f[i][j] > 0){
			int k = a[i + 1] + j;
			if (k & 1)	
				f[i + 1][k / 2 + 1] += f[i][j];
			f[i + 1][k / 2] += f[i][j];
		}
	long long ret = 0;
	for (int i = 0; i < 10; ++i) ret += f[n - 1][i];
	return ret;
}

long long check()
{
	for (int i = 1; i < n; ++i){
		int k =  (a[i] - a[i - 1]);
		if (k > 1 || k < -1) return 0;
	}
	return 1;
}

int main()
{
	scanf("%s", s); n = strlen(s);
	for (int i = 0; i < n; ++i) a[i] = s[i] - '0';
	for (int i = 0; i < 10; ++i)
		ans += work(i);
	ans -= check();
	printf("%I64d\n", ans);
	return 0;
}
