#include <stdio.h>
#include <string.h>
#define NMAX 100
#define SIGMA 14
char S[NMAX];
long long v[NMAX][SIGMA];
int n;
int main(){
	//freopen("H.in", "r", stdin);
	//freopen("H.out", "w", stdout);
	fgets(S+1, NMAX, stdin);
	n = strlen(S+1);
	if(S[n]=='\n') n--;
	for(int j = 0; j  <= 9; ++j)
		v[n][j] = 1;
	for(int i = n - 1; i ; --i)
		for(int x = 0; x <= 9; ++x){
			if( (S[i+1]-'0'+x)%2 ) v[i][x] += v[i+1][(S[i+1]-'0'+x)/2+1];
			v[i][x] += v[i + 1][ (S[i+1]-'0'+x)/2];
		}
	long long s = 0;
	for(int i = 0; i <= 9; ++i)
		s += v[1][i];
	
	int ok = 1;
	for(int i = 1; i < n; ++i)
		if( (S[i]+S[i+1] -'0' - '0')/2 == S[i+1]-'0' || ( (S[i]+S[i+1]-'0'-'0')%2 && (S[i]+S[i+1] -'0'-'0') / 2 + 1 == S[i+1]-'0')) continue;
		else ok = 0;
	printf("%I64d\n", s - ok);
}
