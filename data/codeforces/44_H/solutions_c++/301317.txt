#include <iostream>
#include <string>
#include <vector>

typedef long long ll;

ll dp[51][10] = {};


using namespace std;

int main(){
  string s;cin >> s;
  int n =(int)  s.size();
  for(int i = 0;i<10;i++){
    dp[0][i] = 1;
  }
  for(int i = 1;i<n;i++){
    for(int j = 0;j<10;j++){
      if(((s[i]-'0') + j) % 2 == 1)
	dp[i][((s[i] -'0')+j)/2+1] += dp[i-1][j];
      dp[i][((s[i] -'0')+j)/2] += dp[i-1][j];
    }
  }
  


  ll ans = 0;
  for(int i = 0;i<10;i++){
    ans += dp[n-1][i];
  }

  bool flag = true;
  for(int i = 0;i<n-1;i++){
    int val = ((s[i]-'0')+(s[i+1]-'0'));
    if(val % 2 == 0 && val /2 != (s[i+1]-'0') ){
      flag = false;
    }
    else if ( val/2 != (s[i+1]-'0') 
	      &&  (val/2) +1 != (s[i+1]-'0')){
      flag = false;
    }
  }
  if(flag){
    ans--;
  }

  cout << ans << endl;
  return 0;
}
