n = int(raw_input())
l = [0] * n

for i in range(0,n):
    l[i] = int(raw_input())
max_dep = 0
temp_dep = 0
for i in range(0,n):
    idx = i
    while l[idx] != -1:
        idx = l[idx] - 1
        temp_dep +=1
    max_dep = max(max_dep , temp_dep)
    temp_dep = 0




print max_dep+1
