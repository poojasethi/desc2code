import Queue
graph = dict()
q = Queue.Queue()
q.put([-1, 0])

n = int(raw_input())
for i in range(1, n+1):
    p = int(raw_input())
    if p not in graph: graph[p] = []
    graph[p].append(i)

ans = 0
while not q.empty():
    v, ans = q.get()
    if v not in graph: continue
    for u in graph[v]: q.put([u, ans+1])
print ans
