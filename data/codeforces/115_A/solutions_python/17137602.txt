n = int(raw_input())
boss = [-1]*n

for i in range(0, n):
    b = int(raw_input())
    if b != -1:
        boss[i] = b - 1

def get_depth(vertex):
    depth = 1
    j = vertex
    while boss[j] != -1:
        j = boss[j]
        depth += 1
    return depth

print max(map(get_depth, [i for i in range(n)]))
