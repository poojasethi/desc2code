n = input()
v = [0]*(n+1); h = [0]*(n+1); ans = 0
for x in xrange(1,n+1):
    v[x] = input()
for x in xrange(1,n+1):
    rt = x; _h = 1
    while v[rt] != -1 and not h[rt]:
        rt = v[rt]
        if h[rt]:
            _h += h[rt]; break
        else:
            _h += 1
    h[x] = _h
print max(h)
