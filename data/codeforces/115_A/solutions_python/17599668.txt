#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
a = []
for i in range(n):
    a.append(max(ir()-1, -1))

ans = 0
for i in range(n):
    cur = 0; p = i
    while p != -1:
        cur = cur + 1
        p = a[p]
    if cur > ans: ans = cur
    
print ans
