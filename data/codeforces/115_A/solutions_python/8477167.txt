def solve(n, p):
  b = [False]*n
  res = 0
  for i in range(n):
    if b[i]: continue
    c = 1
    j = i
    b[j] = True
    while p[j] != -1:
      c += 1
      j = p[j]-1
      b[j] = True
    res = max(res, c)
  return res

n = int(raw_input())
p = [int(raw_input()) for i in range(n)]
print solve(n, p)
