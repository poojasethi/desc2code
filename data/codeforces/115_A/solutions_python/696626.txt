import sys

def solve():
    n = int(sys.stdin.readline())
    owner = map(int, sys.stdin.read().split())

    cnt = 0
    for i in xrange(n):
        a = i
        cur = 0
        while a >= 0:
            a = owner[a] - 1
            cur += 1
        cnt = max(cnt, cur)
    print cnt




solve()
