number = int(raw_input())
employees = []
for _ in range(number):
  employees.append(int(raw_input()))
top = 0
for i in range(number):
  depth = 0
  j = employees[i]
  while j != -1:
    depth+=1
    j = employees[j - 1]
    #print ("the current depth is " + str(depth) + " and the next employee is " + str(j))
  top = max(depth, top)
print top + 1
