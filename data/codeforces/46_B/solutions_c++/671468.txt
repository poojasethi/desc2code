#include <cstdio>
#include <cstring>
int d[5][5] = {{0,1,2,3,4},{1,2,0,3,4},{2,3,1,4,0},{3,4,2,1,0},{4,3,2,1,0}};
char n[5][5] = {"S", "M", "L", "XL", "XXL"}, s[111];
int main() {
    int k, t, c[5]; for (int i = 0;i < 5;i++) scanf("%d", &c[i]); scanf("%d", &k);
    while (k--) {
        scanf("%s", s);
        for (int i = 0;i < 5;i++) if (strcmp(n[i], s) == 0) { t = i; break; }
        for (int i = 0;i < 5;i++) if (c[d[t][i]]) { c[d[t][i]]--; puts(n[d[t][i]]); break; }
    }
}
