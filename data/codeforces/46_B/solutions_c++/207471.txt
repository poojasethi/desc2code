#include <iostream>

using namespace std;

int main()
{
  int r[5], k;
  for (int i = 0; i < 5; ++i)
    cin >> r[i];
  cin >> k;
  for (int i = 0; i < k; ++i)
  {
    string s;
    string sootv[5] = {"S", "M", "L", "XL", "XXL"};
    int q, v = -1;
    cin >> s;
    for (int j = 0; j < 5; ++j)
      if (s == sootv[j])
        q = j;
    for (int g = 0; g < 5 && v == -1; ++g)
      if (q + g < 5 && r[q+g])
        v = q + g;
      else if (q - g >= 0 && r[q-g])
        v = q - g;
    cout << sootv[v] << endl;
    r[v]--;
  }
  return 0;
}
