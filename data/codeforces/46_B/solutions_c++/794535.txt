#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main(){
	int arr[5];
	string f[5]={"S","M","L","XL","XXL"};
	for(int i=0;i<5;i++)
		cin>>arr[i];
	int t;
	cin>>t;
	while(t--){
		string a;
		cin>>a;
		int i;
		for(int p=0;p<5;p++)
			if(a==f[p])
				i=p;
		int ans;
		for(int p=4;p>=0;p--)
			if(arr[p]){
				ans=p;
				break;
			}
		for(int p=4;p>=0;p--)
			if(arr[p]&&abs(i-p)<abs(i-ans))
				ans=p;
		arr[ans]--;
		cout<<f[ans]<<endl;
	}
}
