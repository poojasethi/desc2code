#include<stdio.h>
#include<string.h>
main()
{
	int i,n,g,a[6],k,j;
	char c[6][4]={"","S","M","L","XL","XXL"},d[4];
	while(scanf("%d %d %d %d %d",&a[1],&a[2],&a[3],&a[4],&a[5])!=EOF)
	{
		scanf("%d",&k);
		while(k--)
		{
			g=1;
			scanf("%s",d);
			for(i=1;i<=5;i++)
			{
				if(strcmp(c[i],d)==0)
				{
					n=i;break;
				}
			}
			for(i=n,j=n;i<=5||j>0;i++,j--)
			{
				if(a[i]&&i<=5)
				{
					a[i]--;
					printf("%s\n",c[i]);
					break;
				}
				if(a[j]&&j>0)
				{
					a[j]--;
					printf("%s\n",c[j]);
					break;
				}
			}
		}
	}
}
