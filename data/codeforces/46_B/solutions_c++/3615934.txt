#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int have[5];
string size[5]={"S","M","L","XL","XXL"};
int n;
int main(){
#ifdef LOCAL
	freopen("b.txt","r",stdin);
#endif
	for(int i=0;i!=5;i++){
		cin>>have[i];
	}
	cin>>n;
	while(n--){
		string ts;
		cin>>ts;
		for(int i=0;i!=5;i++){
			if(size[i]==ts){
				for(int j=0;j!=5;j++){
					if(i+j<=5&&have[i+j]){
						have[i+j]--;
						cout<<size[i+j]<<endl;
						break;
					}
					if(i-j>=0&&have[i-j]){
						have[i-j]--;
						cout<<size[i-j]<<endl;
						break;
					}
				}
				break;
			}
		}
	}
}
