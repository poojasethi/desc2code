#include <cstdio>
#include <cstring>

char const * const name[] {"S", "M", "L", "XL", "XXL"};
int cnt[5];

int main() {
	for (int i = 0; i < 5; ++i) {
		scanf("%d", cnt + i);
	}
	int k;
	scanf("%d", &k);
	for (int it = 0; it < k; ++it) {
		char s[5];
		scanf("%s", s);
		int i = 0;
		while (strcmp(s, name[i])) {
			++i;
		}
		int ans = i;
		for (int it = 1; cnt[ans] == 0; ++it) {
			if (i + it < 5 && (cnt[i + it] != 0 || i - it < 0)) {
				ans = i + it;
			} else {
				ans = i - it;
			}
		}
		--cnt[ans];
		puts(name[ans]);
	}
}
