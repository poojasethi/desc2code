#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{//freopen("i.txt","r",stdin);
string r[5]={"S","M","L","XL","XXL"};
int k[5];
for (int i=0;i<5;i++) cin>>k[i];
int n; cin>>n;
for (int i=0;i<n;i++)
{string s; cin>>s;
int j;
if (s=="S") j=0;
if (s=="M") j=1;
if (s=="L") j=2;
if (s=="XL") j=3;
if (s=="XXL") j=4;
 for (int ii=0;;ii++)
 {int j1=j+ii,j2=j-ii;
  if (j1<=4)
     if (k[j1]!=0) {cout<<r[j1]<<endl; k[j1]--; break;}
  if (j2>=0)
     if (k[j2]!=0) {cout<<r[j2]<<endl; k[j2]--; break;}
 }
}
return 0;
}
