#include <iostream>
using namespace std;

int main()
{
	int n, a[5], b[9]={0,1,-1,2,-2,3,-3,4,-4}, i, x, y;
	string s, m[5]={"S", "M", "L", "XL", "XXL"};
	cin>>a[0]>>a[1]>>a[2]>>a[3]>>a[4]>>n;
	while (n--) {
		cin>>s;
		for (i=0; i<5; i++) if (m[i]==s) {
			x=i; break;
		}
		for (i=0; i<9; i++) if (x+b[i]<5&&x+b[i]>=0&&a[x+b[i]]>0) {
			a[x+b[i]]--; cout<<m[x+b[i]]<<endl; break;
		}
	}
	return 0;
}
