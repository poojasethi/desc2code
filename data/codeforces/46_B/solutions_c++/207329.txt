#include <stdio.h>
#include <string>
#include <map>
using namespace std;
#define rep(i, n) for(int i=0; i<(int)(n); i++)

int n, nt[5];
char of[5][10] = {"S", "M", "L", "XL", "XXL"};
char buf[100];

int main() {
    map<string, int> m;
    rep(i, 5) m[of[i]] = i;
    rep(i, 5) scanf("%d", nt+i);
    scanf("%d", &n);
    rep(i, n) {
        scanf("%s", buf);
        int t=m[buf], g;
        rep(d, 5) {
            if(t+d<5 && nt[t+d]) { g=t+d; break; }
            if(t-d>=0 && nt[t-d]) { g=t-d; break; }
        }
        nt[g]--;
        printf("%s\n", of[g]);
    }
    return 0;
}
