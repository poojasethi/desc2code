#include<iostream>
#include<map>
#include<string>
#include<cstdio>
#include<vector>
using namespace std;
int n[10];
int main() {
	int i;
	for(i = 0; i < 5; i++) {
		cin >> n[i];
	}
	int k;
	cin >> k;
	map<string,int>m;
	char mm[10][10] = {"S", "M", "L", "XL", "XXL"};
	m["S"] = 0;
	m["M"] = 1;
	m["L"] = 2;
	m["XL"] = 3;
	m["XXL"] = 4;
	string s;
	for(;k--; ) {
		cin >> s;
		for(i = 0; i < 5;i++) {
			if(m[s]+i < 5 && n[m[s]+i]) {
				n[m[s]+i]--;
				printf("%s\n",mm[m[s]+i]);
				break;

			}
			else if(m[s]-i >= 0 && n[m[s]-i]) {
				n[m[s]-i]--;
				printf("%s\n",mm[m[s]-i]);
				break;

			}
		}
	}
	return 0;
}
