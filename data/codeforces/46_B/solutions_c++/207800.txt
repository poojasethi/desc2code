#include <stdio.h>
#include <string.h>
int a[5];
int b[9] = {0, 1, -1, 2, -2, 3, -3, 4, -4};
char s[5][10] = {"S", "M", "L", "XL", "XXL"};
int main() {
	int n;
	for(int i = 0; i < 5; i++)
		scanf("%d", &a[i]);
	scanf("%d", &n);
	for(int i = 0; i < n; i++) {
		char size[10];
		scanf("%s", size);
		int ind;
		for(int j = 0; j < 5; j++)
			if(strcmp(size, s[j]) == 0) ind = j;
		for(int j = 0; j < 9; j++) {
			int p = ind + b[j];
			if(p >= 0 && p < 5 && a[p]) {
				a[p]--;
				printf("%s\n", s[p]);
				break;
			}
		}
	}
	return 0;
}
