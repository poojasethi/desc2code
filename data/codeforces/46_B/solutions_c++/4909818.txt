#include <iostream>
#include <string>
using namespace std;
int main(){
  string s[5] = { "S", "M", "L", "XL", "XXL" };
  string r;
  int a[5], n, p = 0, q = 0;
  for(int i = 0; i < 5; i++)
    cin >> a[i];
  cin >> n;
  for(int i = 0; i < n; i++){
    cin >> r;
    for(int j = 0; j < 5; j++){
      if(r == s[j]){
    p = j;
    q = j;
      }
    while(a[p]==0 && a[q]==0){
      if(p<4){
        p++;}
      if(q>0){
        q--;}
    }
      }
      if(a[p]){
    cout<<s[p]<<endl;
    a[p]--;
      }
      else if(a[q]){
    cout<<s[q]<<endl;
    a[q]--;
      }
    }
  return 0;
}
