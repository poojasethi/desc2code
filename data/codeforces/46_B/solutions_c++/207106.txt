#include<iostream>
using namespace std;
int sz(string s)
{
if(s=="S") return 1;
if(s=="M") return 2;
if(s=="L") return 3;
if(s=="XL") return 4;
return 5;
}
string ansz(int i)
{
if(i==1) return "S";
if(i==2) return "M";
if(i==3) return "L";
if(i==4) return "XL";
return "XXL";
}
int main()
{
int x[6],i,n,r,j;
string s;
for(i=1;i<=5;i++)
{
cin>>x[i];
}
cin>>n;
for(i=1;i<=n;i++)
{
cin>>s;
r=sz(s);
for(j=0;j<=5;j++)
{
if(r+j<=5 && x[r+j]>0)
{
x[r+j]--;
cout<<ansz(r+j)<<endl;
break;
}
if(r-j>=1 && x[r-j]>0)
{
x[r-j]--;
cout<<ansz(r-j)<<endl;
break;
}
}
}
return 0;
}
