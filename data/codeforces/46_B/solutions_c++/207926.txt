#include<cstdio>

int main(void){
	int pl,tt;
	int i,j,k;
	int sz[6];
	char size[5];
	scanf("%d %d %d %d %d",&sz[0],&sz[1],&sz[2],&sz[3],&sz[4]);
	char name[5][4]={"S","M","L","XL","XXL"};
	scanf("%d",&k);
	for(i=0;i<k;i++){
		scanf("%s%*c",size);
		if(size[0]=='S'){pl=0;}
		if(size[0]=='M'){pl=1;}
		if(size[0]=='L'){pl=2;}
		if(size[0]=='X'){
			if(size[1]=='L'){pl=3;}
			else{pl=4;}
		}
		for(j=0;j<5;j++){
			tt=-1;
			if(j<=pl)if(sz[pl-j]>0){
				tt=pl-j;
			}
			if(j<5-pl)if(sz[pl+j]>0){
				tt=pl+j;
			}
			if(tt!=-1){
				sz[tt]--;
				printf("%s\n",name[tt]);
				break;
			}
		}
	}
	return 0;
}
