#include <iostream>
#include <math.h>
#include <cmath>
#include <cstring>
#include <algorithm>
#include <cstdio>
#include <string>
using namespace std;

int mp[100][100];
string pp[6]={"","S","M","L","XL","XXL"};
int k[6];
int main()
{
	int n,x;
	string s;
	for (int i=1;i<=5;i++)
		scanf("%d",&k[i]);
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		cin>>s;
		for (int i=1;i<=5;i++) if (pp[i]==s) x=i;
		for (int sp=0;sp<=5;sp++)
		{
			if (x+sp<=5 && k[x+sp]) {cout<<pp[x+sp]<<endl;k[x+sp]--;break;}
			if (x-sp>0  && k[x-sp]) {cout<<pp[x-sp]<<endl;k[x-sp]--;break;}
		}
	}
}
