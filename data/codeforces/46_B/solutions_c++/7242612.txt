#include<iostream>
using namespace std;
const string str[5]={"S","M","L","XL","XXL"};
int main()
{
	int a[5],n;
	for(int i=0;i<5;i++)cin>>a[i];
	cin>>n;
	while(n--)
	{
		string s;
		int m,now=0;
		cin>>s;
		for(int i=0;i<5;i++)if(str[i]==s)m=i;
		while((m-now<0||a[m-now]==0)&&(m+now>=5||a[m+now]==0))now++;
		if(m+now<5&&a[m+now])
		{
			a[m+now]--;
			cout<<str[m+now]<<endl;
		}
		else
		{
			a[m-now]--;
			cout<<str[m-now]<<endl;
		}
	}
	return 0;
}
