#include <iostream>

using namespace std;

int main()
{
    int i=5,n,shirts[5],g,t;
    string j,sizes[]={"S","M","L","XL","XXL"};

    while(i--)cin>>shirts[4-i];
    cin >> n;

    for(i=0;i<n;i++)
    {
        cin >> j;
        g=0;
        while(j!=sizes[g])g++;

        for(t=0;t<=4;t++)
        {

            if(t+g<5 && shirts[t+g]) {shirts[t+g]--; cout << sizes[t+g] << endl; break;}
            else if (g-t>=0 && shirts[g-t]) {shirts[g-t]--; cout << sizes[g-t] << endl; break;}
        }


    }

    return 0;
}
