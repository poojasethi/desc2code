#include <iostream>
#include <string>

using namespace std;

const string q[] = {"S", "M", "L", "XL", "XXL"};

string t;
int a[5], k;

int main()
{
	for (int i = 0; i < 5; i++)
		cin >> a[i];
	cin >> k;
	for (int i = 0; i < k; i++)
	{
		cin >> t;
		int w = 0;
		for (; w < 5; w++)
			if (q[w] == t)
				break;
		for (int j = 0; ; j++)
			if (w + j < 5 && a[w + j])
			{
				cout << q[w + j] << endl;
				a[w + j]--;
				break;
			}
			else if (w - j > -1 && a[w - j])
			{
				cout << q[w - j] << endl;
				a[w - j]--;
				break;
			}
	}
}
