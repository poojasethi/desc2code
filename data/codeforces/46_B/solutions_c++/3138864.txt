#include<iostream>
#include<string>
using namespace std;
int main(){
	int N[5],n,P,K[5]={2,1,0,3,4},i,j,w;
	cin>>N[0]>>N[1]>>N[2]>>N[3]>>N[4]>>n;
	string R,S[5]={"S","M","L","XL","XXL"};
	for(i=0;i<n;i++){
		cin>>R;
		P=K[(R[0]-'H')/5+(R.size()==3)];
		for(j=0;1;(j>0?j=-j:j=-j+1)){
			w=P+j;
			if(w>=0&&w<5&&N[w]){
				N[w]--;
				cout<<S[w]<<endl;
				break;
			}
		}
	}
}
