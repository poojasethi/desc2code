#include<cstdio>
#include<cstring>
int main()
{
	int a[5];
	int n;
	char buf[100];
	int i,c;
	char siz[5][5]={"S","M","L","XL","XXL"};
	scanf("%d%d%d%d%d",&a[0],&a[1],&a[2],&a[3],&a[4]);
	scanf("%d",&n);
	while(n--)
	{
		scanf("%s",buf);
		for(i=0;i<4;i++)
		{
			if(strcmp(buf,siz[i])==0)
				break;
		}
		c=i;
		if(a[c]>0)
		{
			printf("%s\n",siz[c]);
			a[c]--;
		}
		else
			for(i=1;i<=5;i++)
			{
				if(c+i<5 && a[c+i]>0)
				{
					a[c+i]--;
					printf("%s\n",siz[c+i]);
					break;
				}

				if(c-i>=0 && a[c-i]>0)
				{
					a[c-i]--;
					printf("%s\n",siz[c-i]);
					break;
				}
			}
	}
	return 0;
}
