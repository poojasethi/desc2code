#include<bits/stdc++.h>

using namespace std;

int main(){
	int v[5],k;
	map<string,int>m;
	for(int i=0;i<5;++i)cin>>v[i];
	m["S"] = 0;
	m["M"] = 1;
	m["L"] = 2;
	m["XL"] = 3;
	m["XXL"] = 4;
	string t[] = {"S", "M", "L", "XL", "XXL"};
	cin>>k;
	string size;
	while(k--){
		cin>>size;
		int var = 0, sign=-1;
		string ans;
		while(true){
			int pos =  m[size] +var*sign;
			if(pos>=0 && pos<5 && v[pos]>0 ){ans = t[pos]; v[pos]--;break;}
			else if(sign == -1)var++;
			sign*=-1;
		}
		cout<<ans<<endl;
	}
	
   
	return 0;
}
