#include<cstdio>
#include<cstring>
int c[5];
char s[5][4]={"S","M","L","XL","XXL"};
int main(){
	int n,i,j,a;
	char b[4];
	while(scanf("%d%d%d%d%d",&c[0],&c[1],&c[2],&c[3],&c[4])!=EOF){
		scanf("%d",&n);
		while(n--){
			scanf("%s",b);
			for(i=0;i<5;i++)if(strcmp(b,s[i])==0){a=i;break;}
			for(i=j=a;;i++,j--){
				if(i<5&&c[i]){c[i]--;printf("%s\n",s[i]);break;}
				if(j>=0&&c[j]){c[j]--;printf("%s\n",s[j]);break;}
			}
		}
	}
	return 0;
}
