#include <iostream>
#include <string>
using namespace std;
string s[] = {"S", "M", "L", "XL" ,"XXL"};
int a[5];
string find_f(int j)
{
	for (int i = 1 ; i <20; i++)
	{
		if (j < 5 && j > -1)
			if (a[j] > 0)
			{
				a[j]--;
				return s[j];
			}
		if (i % 2 == 1)
			j += i;
		else
			j -= i;
	}
}
int main()
{
//	freopen("input.txt","r",stdin);
	for (int i = 0 ; i < 5; i++)
		cin >> a[i];
	int k;
	cin >> k;
	for (int i = 0 ; i < k ; i++)
	{
		string str;
		cin >> str;
		for (int j = 0 ; j < 5; j++)
			if (str == s[j])
				cout << find_f(j) << '\n' ;
	}
}
