#include<iostream>
#include<string>
using namespace std;

//int s,m,l,xl,xxl;
int size[5];
int k;
string cur;
string ans[] = {"S","M","L","XL","XXL"};
int main()
{
	for(int i=0;i<5;i++)
		cin >> size[i];
	cin >> k;
	for(int i=0;i<k;i++)
	{
		cin >> cur;
		int temp;
		for(int k=0;k<5;k++)
			if(ans[k] == cur)
				temp = k;
		for(int l=0;l<5;l++)
		{
			if(temp+l <5 && size[temp+l]>0)
			{
				cout << ans[temp+l] << endl;
				size[temp+l]--;
				break;
			}
			if(temp-l>=0 && size[temp-l]>0)
			{
				cout << ans[temp-l] << endl;
				size[temp-l]--;
				break;
			}
		}
	}
	return 0;
}
