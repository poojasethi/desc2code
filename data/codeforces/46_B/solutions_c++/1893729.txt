#include <map>
#include <iostream>
using namespace std;
int a[5],n;
map<string,int> m;
char s[5][11]={"S","M","L","XL","XXL"};

int find(int x){
	if(a[x]>0) {
		a[x]--;
		return x;
	}
	for(int dis=1;dis<=4;dis++){
		if(x+dis>=0&&x+dis<=4&&a[x+dis]>0) {
			a[x+dis]--;
			return x+dis;
		}
		if(x-dis>=0&&x-dis<=4&&a[x-dis]>0) {
			a[x-dis]--;
			return x-dis;
		}
	}
	return -1;
}
char ss[11];
int main() {
	for(int i=0;i<5;i++) cin>>a[i];
	for(int i=0;i<5;i++) m[s[i]]=i;
	cin>>n;
	while(n--){
		cin>>ss;
		cout<<s[find(m[ss])]<<endl;
	}

}
