#include <iostream>
#include <string>
using namespace std;
int a[][5]={{0,1,2,3,4},{1,2,0,3,4},{2,3,1,4,0},{3,4,2,1,0},{4,3,2,1,0}},b[5],k,t,i;
string s,r[]={"S","M","L","XL","XXL"};
int q(string s)
{
 t=s.size();
 if(s[t-1]=='L')
  return t+1;
 return s[0]=='M';
}
int main()
{
 for(;i<5;++i)
  cin>>b[i];
 cin>>k;
 while(k--)
 {
  cin>>s;
  t=q(s);
  i=0;
  while(!b[a[t][i]])
   ++i;
  b[a[t][i]]--;
  cout<<r[a[t][i]]<<'\n';
 }
}
