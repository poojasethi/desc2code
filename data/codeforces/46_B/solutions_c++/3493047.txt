#include <stdio.h>
#include <string.h>
int main()
{
	int n[10],i,j,k;
	int d;
	char s[5],a[10][10]={"S","M","L","XL","XXL"};
	for(i=0;i<5;i++)
	{
		scanf("%d",&n[i]);
	}
	scanf("%d",&k);
	getchar();
	for(i=0;i<k;i++)
	{
		gets(s);
		for(d=0;d<5;d++)
		{
			if(strcmp(s,a[d])==0)
				break;
		}
		//	 printf("%d\n",d);
		//	 puts(a[d]);
		for(j=0;j<5;j++)
		{
			if(n[d+j]&&d+j<5)
			{
				puts(a[d+j]);
				n[d+j]--;
				break;
			}
			if(n[d-j]&&d-j>=0)
			{
				puts(a[d-j]);
				n[d-j]--;
				break;
			}
		}
	}
	return 0;
}
