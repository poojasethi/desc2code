#include <iostream>
#include <string>
using namespace std;

int main(){
	string s[5]={"S", "M", "L", "XL", "XXL"};
	int n[5];
	for (int i=0; i<5; i++)
		cin>>n[i];
	int k;
	cin>>k;
	string t;
	for (int i=0; i<k; i++){
		cin>>t;
		int tt;
		for (int j=0; j<5; j++)
			if (t==s[j])
				tt=j;
		for (int j=0; j<6; j++){
			if (tt+j<5){
				if (n[tt+j]>0){
					n[tt+j]--;
					cout<<s[tt+j]<<endl;
					break;
				}
			}
			if (tt-j>=0){
				if (n[tt-j]>0){
					n[tt-j]--;
					cout<<s[tt-j]<<endl;
					break;
				}
			}
		}
	}
}
