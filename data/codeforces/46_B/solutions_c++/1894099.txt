#include <iostream>
#include <string>
#include <map>
using namespace std;
string s[5] = { "S","M","L","XL","XXL" };
int c[5];
map<string, int> m;
int main()
{
	for (int i=0; i<5; ++i) m[s[i]]=i;
	for (int i=0; i<5; ++i) cin >> c[i];
	int n; cin >> n;
	for (int i=0; i<n; ++i) {
		string t; cin >> t;
		int k=m[t];
		if (c[k]>0) { c[k]--; cout << s[k] << endl; }
		else for (int j=1; j<5; ++j) {
			k+=j;
			if (k<5&&c[k]>0) { c[k]--; cout << s[k] << endl; break; }
			k-=2*j;
			if (k>=0&&c[k]>0) { c[k]--; cout << s[k] << endl; break; }
			k=m[t];
		}
	}
	return 0;
}
