#include<iostream>
#include<string>

using namespace std;

string st[5]={"S","M","L","XL","XXL"};

int foot(string s)
{
	int i;

	for(i=0;i<5;i++)
		if(st[i]==s)  return i;
}

int main()
{
	int i,k,j,l,n[5];
	string s;

	for(i=0;i<5;i++)
		cin>>n[i];

	cin>>k;
	for(i=0;i<k;i++)
	{
		cin>>s;

		l=foot(s);

		j=0;
		while(true)
		{
			if((l+j<5)&&(n[l+j]>0))
			{
				l+=j;
				break;
			}

			if((l-j>=0)&&(n[l-j]>0))
			{
				l-=j;
				break;
			}			

			j++;
		}

		cout<<st[l]<<"\n";

		n[l]--;
	}

	return 0;
}
