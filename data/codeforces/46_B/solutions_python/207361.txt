
a = map(int, raw_input().split())
k = int(raw_input())

sizes = ['S', 'M', 'L', 'XL', 'XXL']
def show(size, radius):
    n = sizes.index(size)
    if n + radius < len(a) and a[n + radius] > 0:
        a[n + radius] -= 1
        return sizes[n + radius]
    elif n - radius >= 0 and a[n - radius] > 0:
        a[n - radius] -= 1
        return sizes[n - radius]
    return show(size, radius+1)

for i in xrange(k):
    print show(raw_input(), 0)
