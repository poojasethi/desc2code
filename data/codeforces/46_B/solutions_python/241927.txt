#!/usr/bin/env python
from string import split

def closest(A, size):
	low = -100
	for i in range(0, size):
		k = size - i - 1
		if A[k] != 0:
			low =k
			break

	high = 100
	for i in range(size + 1, len(A)):
		if A[i] != 0:
			high = i
			break

	if abs(high - size) <= abs(size - low):
		return high
	else:
		return low

A = map(int, split(raw_input()))
k = int(raw_input())

names = {0: 'S', 1: 'M', 2: 'L', 3: 'XL', 4: 'XXL'}
sizes = {'S': 0, 'M': 1, 'L': 2, 'XL': 3, 'XXL': 4}

for i in range(k):
	size = raw_input()

	if A[sizes[size]] > 0:
		A[sizes[size]] -= 1
		print size
	else:
		close = closest(A, sizes[size])
		A[close] -= 1
		print names[close]
