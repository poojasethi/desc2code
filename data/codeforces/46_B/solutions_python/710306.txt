def shirt(i):
	if i==0:
		return 'S'
	if i==1:
		return 'M'
	if i==2:
		return 'L'
	if i==3:
		return 'XL'
	return 'XXL'

M={'S':0,'M':1,'L':2,'XL':3,'XXL':4}
sh=[int(x) for x in raw_input().split()]
n=int(raw_input())
for i in range(n):
	s=raw_input()
	p=M[s]
	if(sh[p]>0):
		sh[p]-=1
		print s
		continue
	j=1
	while 1:
		if p+j<5 and sh[p+j]>0:
			sh[p+j]-=1
			print shirt(p+j)
			break
		if p-j>=0 and sh[p-j]>0:
			sh[p-j]-=1
			print shirt(p-j)
			break
		j+=1
