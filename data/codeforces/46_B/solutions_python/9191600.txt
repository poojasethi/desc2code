def kiesmaat(x):
    m=('S','M','L','XL','XXL')
    p=m.index(x)
    offset=0
    while offset<5:
        if p+offset<5 and inkoop[p+offset]>0:
            inkoop[p+offset]-=1
            return m[p+offset]
        if p-offset>-1 and inkoop[p-offset]>0:
            inkoop[p-offset]-=1
            return m[p-offset]
        offset+=1
    
inkoop=map(int,raw_input().split())
