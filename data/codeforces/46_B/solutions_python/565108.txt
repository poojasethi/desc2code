conv = {'S': 0, 'M': 1, 'L': 2, 'XL': 3, 'XXL': 4}
conv2 = ('S', 'M', 'L', 'XL', 'XXL')
a = map(int, raw_input().split())
n = int(raw_input())
for i in range(n):
    l = r = conv[raw_input()]
    while True:
        if r < 5 and a[r] != 0:
            print conv2[r]
            a[r] -= 1
            break
        r += 1
        if l >= 0 and a[l] != 0:
            print conv2[l]
            a[l] -= 1
            break
        l -= 1
