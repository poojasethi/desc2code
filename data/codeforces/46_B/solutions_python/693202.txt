global sz, nxl
sz = map(int, raw_input().split())
nxl = ["S", "M", "L", "XL", "XXL"]

def dec(x):
	l = r = m = x
	k = 0  
	while sz[m] == 0:
		if(not k % 2):
			r += 1
			if(r < len(nxl)): m = r
		else:
			l -= 1
			if(l >= 0): m = l
		k += 1
	sz[m] -= 1
	print nxl[m]
	return nxl[m];

k = int(raw_input())
ans = [dec(nxl.index(raw_input())) for i in range(k)]
