tshirts = {'S':0,'M':0,'L':0,'XL':0,'XXL':0}

prefs = {'S':['S','M','L','XL','XXL'],
		 'M':['M','L','S','XL','XXL'],
		 'L':['L','XL','M','XXL','S'],
		 'XL':['XL','XXL','L','M','S'],
		 'XXL':['XXL','XL','L','M','S']}

tshirts['S'],tshirts['M'],tshirts['L'],tshirts['XL'],tshirts['XXL'] = map(int,raw_input().split())
N = int(raw_input())

for i in xrange(N):
	optimal = raw_input()
	for j in prefs[optimal]:
		if tshirts[j] > 0:
			tshirts[j] -= 1
			print j
			break
