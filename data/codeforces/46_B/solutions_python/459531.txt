#!/usr/bin/env python
# -*- coding: utf-8 -*-

import sys

def test(sz):
    if sz == "S":       return [0, 1, 2, 3, 4]
    elif sz == "M":     return [1, 2, 0, 3, 4]
    elif sz == "L":     return [2, 3, 1, 4, 0]
    elif sz == "XL":    return [3, 4, 2, 1, 0]
    elif sz == "XXL":   return [4, 3, 2, 1, 0]

def main():
    ns = map(int, raw_input().split())
    for sz in sys.stdin.readlines()[1:]:
        sz = test(sz[:-1])
        for lsz in sz:
            if ns[lsz] > 0:
                if lsz == 0:    print "S"
                elif lsz == 1:  print "M"
                elif lsz == 2:  print "L"
                elif lsz == 3:  print "XL"
                elif lsz == 4:  print "XXL"
                ns[lsz] -= 1
                break
    return 0

if __name__ == '__main__':
    main()
