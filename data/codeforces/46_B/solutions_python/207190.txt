T = [0]*10+map(int,raw_input().split())+[0]*10
n = input()
sz = {"S":10,"M":11,"L":12,"XL":13,"XXL":14}
name = [0]*10+["S","M","L","XL","XXL"]
for i in range(n):
	ss = raw_input()
	s = sz[ss]
	for d in range(10):
		if T[s+d]>0:
			print name[s+d]
			T[s+d] -= 1
			break
		if T[s-d]>0:
			print name[s-d]
			T[s-d] -= 1
			break
