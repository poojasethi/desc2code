sizes = ['S','M','L','XL','XXL']
pref_sizes = {
'S':[0,1,2,3,4],
'M':[1,2,0,3,4],
'L':[2,3,1,4,0],
'XL':[3,4,2,1,0],
'XXL':[4,3,2,1,0]
}
shirts = map(int, str(raw_input()).split())
n = input()
for i in xrange(n):
    size = raw_input()
    for j in pref_sizes[size]:
        if shirts[j]>0:
            shirts[j]-=1
            print sizes[j]
            break
