#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
using namespace std;
string s;
int main()
{

	
    int a=0,b=0,c=0,t1,t2;
    cin>>s;
    for(int i=0;s[i]!='\0';i++)
    if(s[i]=='1')a++;
    else if(s[i]=='0')b++;
    else c++;
    int n=s.length()-2;
    t1=(n+1)/2;
    t2=n/2;
    if(a<=t1)cout<<"00\n";
    if(t1+1-a>=0&&t1+1-a<=c)
    {
		 if(b+c-t2-(t1+1-a)==1){
		 if(s[n+1]=='1'||s[n+1]=='?'&&t1+1-a!=0)cout<<"01\n";
	     if(s[n+1]=='0'||s[n+1]=='?'&&c-(t1+1-a)!=0)cout<<"10\n";	 }
		
	}
    if(t2>=b)cout<<"11\n";
    
    
    
    
    
    
    
    
	return 0;
}
