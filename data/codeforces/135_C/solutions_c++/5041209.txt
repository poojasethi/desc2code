#pragma GCC optimize("O2")
#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <string>

int main(){
    std::string str;
    std::cin>>str;
    int len = str.size();
    int num0 = 0, num1 = 0, num2 = 0;
    for(int i = 0; i < len; i++){
        if(str[i]=='0') ++num0;
        else if(str[i]=='1') ++num1;
        else ++num2;
    }
    int nn = (len-2)>>1;
    int mm = (len-1)>>1;
    //printf("%d %d %d %d %d\n", num0, num1, num2, nn, mm);
    if(num0+num2>num1) puts("00");
    if(num1<=mm+1 && mm+1<=num1+num2){
        if(str[len-1]=='1'||(str[len-1]=='?'&&num1<mm+1)) puts("01");
        if(str[len-1]=='0'||(str[len-1]=='?'&&num0<nn+1)) puts("10");
	}
    if(num1+num2>num0+1) puts("11");
    return 0;
}
