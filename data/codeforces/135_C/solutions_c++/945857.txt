#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <iostream>
#include <algorithm>

using namespace std;

char s[100010];
int n,num0,num1,numw;

int main(){
    scanf("%s",s);
    num0=num1=numw=0;
    n=strlen(s);
    for (int i=0;i<n;i++)
        if (s[i]=='0') num0++;
        else if (s[i]=='1') num1++;
        else numw++;
    if (num0+numw>=n/2+1) printf("00\n");
    if (num0<=n/2 && num0+numw>=n/2)
       if (s[n-1]=='1' || (s[n-1]=='?' && num1<(n+1)/2)) printf("01\n");
    if (num0<=n/2 && num0+numw>=n/2)
       if (s[n-1]=='0' || (s[n-1]=='?' && num0<n/2)) printf("10\n");
    if (num1+numw>=(n+1)/2+1) printf("11\n");
    return 0;
}
