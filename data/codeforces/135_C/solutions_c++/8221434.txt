#include<iostream>
#include<string>
#include<vector>

using namespace std;

typedef long long int LL;

int main() {
    cin.sync_with_stdio(0);
    string s;
    cin >> s;
    int length = s.length();
    int a = 0, b = 0, c = 0;
    for(int i = 0; i < length; ++i)
    {
        if(s[i]=='1') a++;
        else if(s[i]=='0') b++;
        else c++;
    }

    int x = (b+c-a+((a+b+c)%2))/2;
    int y = (a+c-b-((a+b+c)%2))/2;

    if(b+c > a) cout << "00" << "\n";
    if((s[length-1]=='1' && x >= 0 && x <= c) || (s[length-1]=='?' && x > 0 && x <= c)) cout << "01" << "\n";
    if((s[length-1]=='0' && y >= 0 && y <= c) || (s[length-1]=='?' && y > 0 && y <= c)) cout << "10" << "\n";
    if(a+c > b+1) cout << "11" << "\n";

    return 0;
}
