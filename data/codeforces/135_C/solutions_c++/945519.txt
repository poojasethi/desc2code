#include <stdio.h>
#include <string.h>
int x, y, z, n, u, v;
char str[100010];

bool test1()
{
	if (y<=v+1 && v+1<=y+z)
	{
		if (str[n-1]!='?') return str[n-1]=='1';
		else return y<v+1;
	}
	else return false;
}

bool test2()
{
	if (y<=v+1 && v+1<=y+z)
	{
		if (str[n-1]!='?') return str[n-1]=='0';
		else return x<u+1;
	}
	else return false;
}


int main()
{
	int i;
	x=y=z=0;
	scanf("%s", str);
	n=strlen(str);
	for (i=0; i<n; i++)
	  if (str[i]=='0') x++;
	  else if (str[i]=='1') y++;
	  else z++;
	  
	u=(n-2)/2;//0
	v=n-2-u;//1
//	printf("x=%d y=%d z=%d u=%d v=%d\n", x, y, z, u, v);
	if (x+z>=u+2) puts("00");
	if (test1()) puts("01");
	if (test2()) puts("10");	
	if (y+z>=v+2) puts("11");
	return 0;
}
