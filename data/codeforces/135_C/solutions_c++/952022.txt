#include <cstdio>

using namespace std;

int main(int argc, char *argv[])
{
    static char s[110000];
    
    scanf("%s", s);
    int n, c0 = 0, c1 = 0, c = 0;
    for (n = 0; s[n]; n++)
        if (s[n] == '0') c0++;
        else if (s[n] == '1') c1++;
        else c++;

    if (c + c0 > c1) printf("00\n");
    
    if (c0 <= n / 2 && c1 <= (n + 1) / 2) {
        if (s[n - 1] == '1' || s[n - 1] == '?' && c1 < (n + 1) / 2)
            printf("01\n");
        if (s[n - 1] == '0' || s[n - 1] == '?' && c0 < n / 2)
            printf("10\n");
    }
    
    if (c + c1 > c0 + 1) printf("11\n");
    
    return 0;
}
