#include <cstdio>
#include <cstring>

int n, zero, one, qm, res;
char s[100001];

int main() {
	scanf("%s", s);
	n = strlen(s);
	for (int i = zero = one = qm = res = 0; i < n; ++i) {
		if (s[i] == '0') {
			++zero;
		} else if (s[i] == '1') {
			++one;
		} else {
			++qm;
		}
	}
	for (int i = 0; i <= qm; ++i) {
		if (zero + i + 2 <= one + qm - i) {
			res |= 0x8;
		} else if (one + qm - i + 1 <= zero + i) {
			res |= 0x1;
		} else if (s[n - 1] == '0') {
			res |= 0x4;
		} else if (s[n - 1] == '1') {
			res |= 0x2;
		} else {
			if (i > 0) {
				res |= 0x4;
			}
			if (qm - i > 0) {
				res |= 0x2;
			}
		}
	}
	for (int i = 0; i < 4; ++i) {
		if ((res & 1 << i) != 0) {
			printf("%d%d\n", i / 2, i % 2);
		}
	}
	return 0;
}
// 1: 00
// 2: 01
// 4: 10
// 8: 11
