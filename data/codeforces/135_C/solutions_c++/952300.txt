#include <cstdio>

char s[100100];

int main()
{
	scanf("%s", s);
	int n, n0 = 0, n1 = 0;
	for(n = 0; s[n]; ++n){
		if(s[n] == '0')
			n0++;
		else if(s[n] == '1')
			n1++;
	}
	//00
	if((n+1)/2 > n1){
		printf("00\n");
	}
	if(n % 2 == 1)
		n1--;
	if((s[n-1] == '1' && n/2 >= n0 && n/2 >= n1) || (s[n-1] == '?' && n/2 >= n0 && n/2 >= n1+1))
		printf("01\n");
	if((s[n-1] == '0' && n/2 >= n0 && n/2 >= n1) || (s[n-1] == '?' && n/2 >= n0+1 && n/2 >= n1))
		printf("10\n");

	//11
	if(n/2 > n0){
		printf("11\n");
	}
	return 0;
}
