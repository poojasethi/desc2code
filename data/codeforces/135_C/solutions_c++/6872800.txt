#include <iostream>
#include <cmath>

using namespace std;
string s;
int n=0;
bool arr[5]; 
int main()
{
	int a=0,b=0,c=0;
	cin>>s;
	n=s.size();
	for(int i=0;i<n;i++)
	{
		if(s[i]=='1')
			a++;
		else if(s[i]=='0')
			b++;
		else
			c++;		
	}
	int y=(n-2)/2;
	int x=n-2-y;	
	if(x>=a)
		arr[0]=1;
	if(y>=b)
		arr[3]=1;
	int d=abs(a-b);
	if( a<= x+1 && b<=y+1 )
	{
		if( s[n-1] == '1' || (s[n-1]=='?' && a<=x) )
			arr[1]=1;
		if( s[n-1] == '0' || (s[n-1]=='?' && b<=y) )
			arr[2]=1;	
	}
	for(int i=0;i<4;i++)
	{
		if(arr[i]==1)
		{
			if(i==0)
				cout<<"00"<<endl;	
			if(i==1)
				cout<<"01"<<endl;	
			if(i==2)
				cout<<"10"<<endl;	
			if(i==3)
				cout<<"11"<<endl;	
		
		}	
	}		
	return 0;
}
