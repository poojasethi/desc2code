#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

char s[100005];

int main(void) {
	int a=0,b=0,c=0;
	scanf("%s", s);
	for(int i=0;s[i];i++)
	if(s[i]=='0') a++;
	else if(s[i]=='1') b++;
	else c++;
	int n=a+b+c;
	int p=n/2-1, q=(n+1)/2-1;
	if(b<=q) { puts("00"); }
	if(s[n-1]!='0' && a-1<=p && b+(s[n-1]=='?')-1<=q) puts("01");
	if(s[n-1]!='1' && a+(s[n-1]=='?')-1<=p && b-1<=q) puts("10");
	if(a<=p) puts("11");
	return 0;
}
