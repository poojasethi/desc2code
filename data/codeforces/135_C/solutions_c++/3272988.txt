#include<iostream>
#include<cstdio>
using namespace std;
#include<string>
#include<cstring>
//#include<conio.h>
int main()
{
    int l,one=0,zero=0,q=0,i,x;
    char str[100001];
    scanf("%s",str);
    l=strlen(str);;
    for(i=0;i<l;i++)
    if(str[i]=='0')
    zero++;
    else if(str[i]=='1')
    one++;
    else
    q++;
    if(str[l-1]=='?')
    x=1;
    else
    x=0;
    if(one<zero+q)
    printf("00\n");
    if((str[l-1]=='?'||str[l-1]=='1')&&zero+q-x>=one+x-1&&one+q>=zero)
    printf("01\n");
    if((str[l-1]=='?'||str[l-1]=='0')&&zero+q>=one-1&&one+q-x>=zero+x)
    printf("10\n");
    if(one+q>zero+1)
    printf("11\n");
    //getch();
    return 0;
}
