#include <iostream>
using namespace std;
int main(){
	char ch;
	int n0=0, n1=0, nU=0;
	while(cin >> ch)
		if(ch=='1')
			n1++;
		else if(ch=='0')
			n0++;
		else if(ch=='?')
			nU++;
	n1+=nU;
	bool f00=false, f01=false, f10=false, f11=false;
	for(int i=0; i<=nU; i++){
		if(n0 - n1 > 0)
			f00 = true;
		if((n0 - n1 <= 0) && (n1 - n0 <= 1) && (ch=='1' || (ch=='?' && i!=nU)))
			f01 = true;
		if((n0 - n1 <= 0) && (n1 - n0 <= 1) && (ch=='0' || (ch=='?' && i!=0 )))
			f10 = true;
		if(n1 - n0 > 1)
			f11 = true;
		n0++;
		n1--;
	}
	if(f00)
		cout << "00" << endl;
	if(f01)
		cout << "01" << endl;
	if(f10)
		cout << "10" << endl;
	if(f11)
		cout << "11" << endl;
	return 0;
}
