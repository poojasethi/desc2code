#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char s[100005];
int n;
int a,b,c;
int main()
{
    scanf("%s",s+1);
    n=strlen(s+1);
    int m1=(n-2+1)/2;
    int m2=(n-2)/2;
    for(int i=1;i<=n;i++)
    {
        if(s[i]=='0')a++;
        else if(s[i]=='1')b++;
        else c++;
    }
    //cout<<a<<" "<<b<<" "<<c<<endl;
    if(b<=m1)
    {
        puts("00");
    }
    if(b<=m1+1 && a<=m2+1){
        if((s[n]=='?' && b<m1+1) || s[n]=='1') printf("01\n");
        if((s[n]=='?' && a<m2+1) || s[n]=='0')  printf("10\n");
    }
    if(a<=m2)
    {
        puts("11");
    }
    return 0;
}
