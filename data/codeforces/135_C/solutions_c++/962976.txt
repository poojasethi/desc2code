#include<cstdio>
#include<cstring>
using namespace std;
const int N = 100005;

#define fo(i , st , en) for (int i = st; i <= en; i++)

char c[N];
int n , sum1 , sum0 , sumw;

int main(){
    scanf("%s" , &c);
    n = strlen(c);
    sum1 = sum0 = 0;
    fo (i , 0 , n - 1){
        if (c[i] == '1') sum1++;
        if (c[i] == '0') sum0++;
    }
    sumw = n - sum1 - sum0;
    if (sumw + sum0 > sum1) puts("00");
    if ((n + 1) / 2 <= sum1 + sumw && sum0 + sumw >= n / 2){
        if (c[n - 1] == '1' || c[n - 1] == '?' && sum0 + sumw > n / 2) puts("01");
        if (c[n - 1] == '0' || c[n - 1] == '?' && sum1 + sumw > (n + 1) / 2) puts("10");
    }
    if (sumw + sum1 - 1 > sum0) puts("11");
    return 0;
}
