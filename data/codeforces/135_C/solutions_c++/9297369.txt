#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,s1,s0,last0,last1,last2;
int f[4];
char p[N];
int main() {
	scanf("%s",p+1);
	n=strlen(p+1);
	For(i,1,n) {
		if (p[i]=='1') s1++,last1=i;
		else if (p[i]=='0') s0++,last0=i;
		else last2=i;
	}
	int she=n-s1-s0,g1=(n+1)/2,g0=n-g1;
	g1--,g0--;
	For(i,0,she) {
		int ss0=s0+i,ss1=s1+she-i;
		if (ss0<=g0) f[3]=1;
		else if (ss1<=g1) f[0]=1;
		else {
			if (max(i?last2:0,last0)>max(i?0:last2,last1)) f[2]=1;
			if (max(i<she?last2:0,last1)>max(i<she?0:last2,last0)) f[1]=1;
		}
	}
	For(i,0,3) if (f[i]) printf("%d%d\n",i/2,i&1);
	return 0;
}
