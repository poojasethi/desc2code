#include <iostream>
#include <cstdio>

using namespace std;

char str[200005]={0};

int main()
{
 long s0=0,s1=0;
 long i,j;
 long s=0;
 
 scanf("%s",str);
 for(i=0;str[i];i++)
   s0+=str[i]=='0',s1+=str[i]=='1',s+=str[i]=='?';
 if(s0+s+1>=s1+2)
  {
   printf("00\n");
  }
 if(str[i-1]=='1'||str[i-1]=='?')
  {
   if(str[i-1]=='?')
    s--,s1++;
   
   if(s0+s>=s1-1&&s1+s>=s0)
    printf("01\n");
   
   if(str[i-1]=='?')
    s++,s1--;
  }
 if(str[i-1]=='0'||str[i-1]=='?')
  {
   if(str[i-1]=='?')
    s--,s0++;
   
   if(s0+s>=s1-1&&s1+s>=s0)
    printf("10\n");
    
   if(str[i-1]=='?')
    s++,s0--;
  }
 if(s1+s>=s0+2)
  printf("11\n");
 
 return 0;
}
