#include <cstdio>
#include <cstring>
#define MAX_N 100000
using namespace std;
char _str[MAX_N + 4];
int _str_n;
int a, b, q;
int main(){
	char *p;
	scanf("%s", _str);
	a = 0;
	b = 0;
	q = 0;
	for(p = _str; * p; p ++){
		switch(* p){
		case '0':
			a ++;
		break;
		case '1':
			b ++;
		break;
		case '?':
			q ++;
		break;
		}
	}
	p --;
	if(a + q > b){
		printf("00\n");
	}
	if(* p == '1' && b - a + q >= 0 && b - a <= q + 1 || * p == '?' && b - a + q >= 0 && b - a <= q - 1){
		printf("01\n");
	} 
	if(* p == '0' && b - a + q >= 0 && b - a <= q + 1 || * p == '?' && b - a + q >= 2 && b - a <= q + 1){
		printf("10\n");
	} 
	if(b + q - 1 > a){
		printf("11\n");
	}
	return 0;
}
