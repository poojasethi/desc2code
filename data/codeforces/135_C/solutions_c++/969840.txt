#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

char p[256];
int z,o,q;
string str;

int main()
{

    cin>>str;

    z = count(str.begin(),str.end(),'0');
    o = count(str.begin(),str.end(),'1');
    q = str.length()-z-o;

    if(q+z>o) cout << "00\n";

    if( (z>=o&&o+q>=z) || (z<o&&z+q>=o-1) )
    {
        char l = str[str.length()-1];
        bool zero=true,one=true;
        if(o+q<=z+1) zero=false;
        if(z+q<=o) one=false;

        if(l=='1'||(l=='?'&&one)) cout << "01\n";
        if(l=='0'||(l=='?'&&zero)) cout << "10\n";
    }
        if(q+o>z+1) cout << "11\n";


    return 0;
}
