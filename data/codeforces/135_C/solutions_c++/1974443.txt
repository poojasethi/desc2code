#include <iostream>
#include <math.h>
#include <memory.h>

#include <string>
#include <string.h>

using namespace std;

int main()
{
	string s;
	cin >> s;
	int n = s.length();
	int a = 0 , b = 0;
	for(int i = 0; i < n; i++ )
		if( s[ i ] == '0' )
	        a++;
		else
		if( s[ i ] == '1')
		b++;
	if( b < n / 2 + n % 2 )
        puts( "00" );
	if( a <= n / 2 && b <= n / 2 + n % 2 )
		if( s[ n - 1 ] == '1' )
		puts( "01" );
		else
			if( s[ n - 1 ] == '0' )
			puts( "10" );
			else
			{
				if( b < n / 2 + n % 2)
				puts( "01" );
				if( a < n / 2)
				puts( "10" );
			}
	if( a < n / 2)
        puts( "11" );
        return 0;
}
