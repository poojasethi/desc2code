#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

char a[100005];

int main()
{
   int l,s=0,s0=0,s1=0;
   gets(a+1);
   l=strlen(a+1);
   for (int i=1;i<=l;i++)
   {
      s+=(a[i]=='?');
      s0+=(a[i]=='0');
      s1+=(a[i]=='1');
   }
   if (s0+s>l/2)
      cout<<"00"<<endl;
   if (s0+s>=l/2&&s1+s>=(l+1)/2)
   {
      if (a[l]=='1'||a[l]=='?'&&s0+s!=l/2)
         cout<<"01"<<endl;
      if (a[l]=='0'||a[l]=='?'&&s1+s!=(l+1)/2)
         cout<<"10"<<endl;
   }
   if (s1+s>(l+1)/2)
      cout<<"11"<<endl;
   return 0;
}
