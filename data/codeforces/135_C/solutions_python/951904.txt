s=raw_input()
c2=s.count('?')
c1=s.count('1')
c0=s.count('0')
if c0+c2>c1:
    print '00'
t = c1-c0
if (t>0 and c2>=t-1) or (t<0 and c2>=-t) or t==0:
    if s[-1]=='1':
        print '01'
    elif s[-1]=='?':
        if not (t>0 and (t-1==c2 or t==c2)):
            print '01'
        if not ((t<0 and (-t+1==c2 or -t==c2)) or (t==0 and c2==1)):
            print '10'  
    elif s[-1]=='0':
        print '10'
if c1+c2-2>=c0:
    print '11'
