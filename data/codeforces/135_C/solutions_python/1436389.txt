#!/usr/bin/env python
# coding: utf-8

seq = [2 if c == '?' else int(c) for c in raw_input()]
res = []

size = len(seq)
c0 = seq.count(0)
c1 = seq.count(1)
cq = seq.count(2)

if size % 2 == 1 and c1 > 0:
    c1 -= 1

if c1 + cq > c0:
    res.append('11')

if c0 + cq > c1:
    res.append('00')

if c1 - c0 <= cq and c0 - c1 <= cq:
    if seq[-1] == 1:
        res.append('01')
    elif seq[-1] == 0:
        res.append('10')
    else:
        if (c1 + 1) - c0 <= (cq - 1) and c0 - (c1 + 1) <= (cq - 1):
            res.append('01')
        if c1 - (c0 + 1) <= (cq - 1) and (c0 + 1) - c1 <= (cq - 1):
            res.append('10')

last = seq[-1]
res.sort()
for s in res:
    print s
