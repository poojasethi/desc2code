n = input()

zero = []
cards = 1
ans = 0

for i in range(n):
    a, b = map(int, raw_input().split(" "))
    if b == 0:
        zero.append(a)
    else:
        ans += a
        cards += b - 1

zero.sort()
zero.reverse()

for x in zero:
    if cards > 0:
        cards -= 1
        ans += x

print ans
