from sys import stdin as I

n = int(I.readline())
v = sorted((map(int, line.split())[::-1] for line in I.readlines()), reverse=True)

q = 1
i = 0
rv = 0
while i < n and q - 1 + v[i][0] > 0:
    rv += v[i][1]
    q += v[i][0] - 1
    i += 1
    
rv += max(p[1] for p in v[i:]) if i < n else 0
print rv
