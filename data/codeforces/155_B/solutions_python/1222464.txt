n = int(raw_input())


cards = []
for i in range(n):
    a, b = map(int, raw_input().split())
    cards.append((b, a))
cards.sort(reverse=True)

curr_cards = 1
curr_score = 0
for c in cards:
    if curr_cards < 1:
        break
    curr_cards -= 1
    curr_score += c[1]
    curr_cards += c[0]

print curr_score
