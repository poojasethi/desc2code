n = input()
zero = []
card, ans = (1, 0)

for _ in range(n):
    a, b = map(int, raw_input().split())
    if b==0:
        zero.append(a)
    else:
        ans += a
        card += b-1
zero.sort(reverse=True)
for x in zero:
    if card > 0:
        card -= 1
        ans += x
    else:
        break
print ans
