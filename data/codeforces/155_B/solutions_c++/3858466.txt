#include<iostream>
#include<algorithm>

using namespace std;
int main()
{
	int n,i;
	cin >> n;
	pair<int,int> v[n+1];
	for ( i = 0; i < n; i++ )
	{
		cin >> v[i].second>>v[i].first;
	}
	sort(v,v+n); 
	int s = 1;
	i=n-1;
	int su = 0;
	while ( s != 0 && i >= 0 )
	{
		su += v[i].second;
		s+=(v[i].first-1);
		i--;
	}
	cout << su << endl;
}
