#include<bits/stdc++.h>

using namespace std;

int main(){
pair<int,int> card[1002];
int n, p=0, r=1;
    cin>>n;
    for(int i=0;i<n;i++) cin>>card[i].second>>card[i].first;
    sort(card,card+n);
    for(int i=n-1;((i>=0)&&(r>0));i--)
    {
        p+=card[i].second;
        r+=(card[i].first-1);
    }

    cout<<p;

	return 0;
}
