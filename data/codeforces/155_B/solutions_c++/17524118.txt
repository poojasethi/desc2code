#include <bits/stdc++.h>
using namespace std;
#define maxn 1000

pair<int, int> p[maxn];

int main()
{
	int n, i, ret = 0, cnt = 1;
	scanf("%d", &n);
	for(i = 0; i < n; i++)
		scanf("%d %d", &p[i].second, &p[i].first);
	sort(p, p + n);
	for(i = n - 1; i >= 0 && cnt > 0; i--)
	{
		ret += p[i].second;
		cnt += p[i].first - 1;
	}
	printf("%d", ret);
	return 0;
}
