#include <vector>
#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
	int n,a,b,c=1,res=0;
	vector<int> p;
	scanf("%d",&n);
	while(n--){
		scanf("%d%d",&a,&b);
		if(b>0){
			res+=a;
			c+=b-1;
		} else p.push_back(a);
	
	}
	sort(p.begin(),p.end());
	int g = p.size();
	for(int i=0;i<c && g-1-i>=0;++i)res+=p[g-1-i];
	printf("%d\n",res);
}
