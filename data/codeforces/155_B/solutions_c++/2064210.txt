#include<cstdio>
#include<algorithm>
struct p{
int a,b;
bool operator<(const p &cmp)const{
return (b!=cmp.b)?b>cmp.b:a>cmp.a;
}
}P[1001];
int N;
int main(){
scanf("%d",&N);
for(int i=0;i<N;++i){
scanf("%d%d",&P[i].a,&P[i].b);
}
std::sort(P,P+N);
int k =0,now=1,ans=0;
while(k < N && now > 0) {
--now;
now+=P[k].b;
ans+=P[k].a;
++k;
}
printf("%d\n",ans);
}
