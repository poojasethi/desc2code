#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	long long q,w,j=-1,n,i,ans=0,m[1500];
	cin>>n;
	int e=1;
	for(i=0;i<n;i++)
	{
		cin>>q>>w;
		if(w!=0){ans+=q;e+=w;e--;}
		else {j++;m[j]=q;}
	}
	if(j!=-1)
	sort(m,m+j+1);
	
	while(e>0 && j>=0)
	{
		e--;
		ans+=m[j];
		j--;
	}
	cout<<ans;
	return 0;
}
