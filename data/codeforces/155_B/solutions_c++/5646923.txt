#include <bits/stdc++.h>

using namespace std;
#define F first
#define S second
int n,i,h=1,sum;
pair<int,int> p[1001];
int main()
{
    cin>>n;
    for(i=1;i<=n;i++)cin>>p[i].S>>p[i].F;
    sort(p+1,p+1+n);
    while (h>0 and n>0)
    {
        sum+=p[n].S;h+=(p[n--].F-1);
    }
    cout<<sum<<endl;
    return 0;
}
