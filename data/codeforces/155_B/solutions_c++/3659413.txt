#include <iostream>
#include <algorithm>

using namespace std;

int n,m,k,x=1,i;
pair <int,int> a[1005];
main () {
    cin>>n;
    for (i=0; i<n; i++) {
        cin>>a[i].second>>a[i].first;
        
    }
    sort (a,a+n);
    reverse (a,a+n);
    
    for (i=0; i<n; i++) {
        x=x+a[i].first;
        if (x>0) {k=k+a[i].second; x--;}
}
    cout<<k;
}
