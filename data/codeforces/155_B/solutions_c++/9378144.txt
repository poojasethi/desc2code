#include <iostream>
#include <algorithm>
using namespace std;
pair <int,int> arr[1000];
int n;
int ans=0,k=1;
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>arr[i].second>>arr[i].first;
    }
    sort(arr,arr+n);
    for(int i=n-1;i>=0&&k>0;i--)
    {
        ans+=arr[i].second;
        k+=arr[i].first-1;
    }
    cout<<ans<<endl;
}
