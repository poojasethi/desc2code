#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
int n,i,x,y,e=0,j=0,ans=0;
cin>>n;
vector<int>a;
for(i=0;i<n;i++)
{
cin>>x>>y;
if(y)
{
ans+=x;
e+=y;
e--;
}
else
{
a.push_back(x);
j++;
}
}
j--;
sort(a.begin(),a.end());
while(j>=0&&e>=0)
{
ans+=a[j];
j--;
e--;
}
cout<<ans;
}
