#include<cstdio>
#include<algorithm>
using namespace std;
int a1[1000];
bool cmp(int a,int b)
{return a>b;}
int main()
{
	int n,l=0,a=1,b,c,o=0;
	scanf("%d",&n);
	while(n--)
	{
		scanf("%d%d",&b,&c);
		if(c){a+=c-1;o+=b;}
		else a1[l++]=b;
	}
	sort(a1,a1+l,cmp);
	for(int i=0;i<l&&i<a;++i)
		o+=a1[i];
	printf("%d\n",o);
}
