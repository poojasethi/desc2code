#include <iostream>
#include <algorithm>
using namespace std;

pair<int, int> arr[10001];

int main(){
	long long n, i, ans=0, x=1;
	cin >> n;
	for(i=0;i<n;i++){
		cin >> arr[i].second >> arr[i].first;
	}
	sort(arr, arr+n);
	i=n-1;
	while(x>0 and i>=0){
		ans += arr[i].second;
		x += arr[i].first-1;
		i--;
	}
	cout << ans << '\n';
	return 0;
}
