#include<stdio.h>
#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
	long long int n,i,a,b,A[1010],count=1,ans=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a;
		cin>>b;
		A[i]=b*100000+a;
	}
	sort(A,A+n);
	for(i=n-1;i>=0;i--)
	{
		b=A[i]/100000;
		count+=b-1;
		ans+=A[i]-b*100000;
		if(count==0)
			break;
	}
	cout<<ans;
	return 0;
	}
