#include <bits/stdc++.h>

using namespace std;

int n, a, b, x[1005], m, ans, k = 1;

int main() {
  cin >> n;
  for (int i = 0; i < n; i++) {
    cin >> a >> b;
    if (b) ans += a, k += (b - 1);
    else x[m++] = a;
  }
  sort(x, x + m);
  for (int i = 1; i <= k && i <= m; i++) {
    ans += x[m - i];
  }
  cout << ans;
}
