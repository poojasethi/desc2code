#include <bits/stdc++.h>
using namespace std;

int main() 
{
	// your code goes here
	int n;
    cin>>n;
	pair<int,int> a[10001];
	int ans=0,c=1,i;
	for(i=0;i<n;i++)
	 {
	  cin>>a[i].second;
	  cin>>a[i].first;
	 }
	sort(a,a+n);
	n--;
	while(c>0&&n>=0)
	 {
	  ans+=a[n].second;
	  c--;
	  c+=a[n].first;
	  n--;
	 }
	cout<<ans<<endl;

    return 0;
}
