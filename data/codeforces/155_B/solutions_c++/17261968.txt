#include <iostream>
#include <algorithm>
using namespace std;

pair<int, int> p[10000];

int main()
{
	int n, i, ans=0, x=1;
	cin >> n;
	for(i=0;i<n;i++)
	{
		cin >> p[i].second >> p[i].first;
	}
	sort(p, p+n);
	i=n-1;
	while(x>0 and i>=0)
	{
		ans += p[i].second;
		x += p[i].first-1;
		i--;
	}
	cout << ans << '\n';
	return 0;
}
