#include <iostream>
using namespace std;
int main()
{
	int n,a[1000],b[1000],ans=0,c=1;
	cin>>n;
	for(int	i=0;i<n;i++)
		cin>>a[i]>>b[i];
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
		if(b[j]>b[i] || (b[j]==b[i] && a[j]>a[i]))		
		swap(b[i],b[j]),swap(a[i],a[j]);
		for(int i=0;c && i<n;i++)
		{
		ans+=a[i];
		c+=b[i]-1;
		}
		cout<<ans<<endl;
}
