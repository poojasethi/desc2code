#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n, i, a, b, c=0, x=1;
    cin >> n;
    pair<int, int>v[n];
    for (i = 0; i < n; ++i)
        cin >> v[i].second >> v[i].first;

    sort (v, v+n);
    for (i = n-1; i >=0; --i, --x)
    {
        x+=v[i].first;
        if (x>0)c+=v[i].second;
    }
    cout << c;
    return 0;
}
