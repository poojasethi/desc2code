#include <iostream>        
#include <algorithm>
using namespace std;
int k,n,m,a,b,c,d,i,f,j,p;
pair <int,int> x[1005];
main(){ 
cin>>n;
for(i=0;i<n;i++){
cin>>a>>b;
x[i].first=b;
x[i].second=a;
}
sort(x,x+n);reverse(x,x+n);
p=1;j=0;
while(p>0 && j<=n){
p=p+x[j].first;
d=d+x[j].second;
p=p-1;j=j+1;
}
cout<<d;
}
