#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int n,i,j;
int q,answ;
pair <int,int> a[1002];
int main()
{
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>a[i].second>>a[i].first;
	}
	sort(a+1,a+n+1);
	q=1;
	for(i=n;i>=1;i--)
	{
		if(q==0)
			break;
		answ+=a[i].second;
		q--;
		q+=a[i].first;
	}
	cout<<answ<<endl;
	return 0;
}
