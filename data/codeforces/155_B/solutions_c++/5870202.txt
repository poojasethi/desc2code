#include <iostream>
#include <algorithm>
using namespace std;
pair<int,int> a[1100];
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i].second>>a[i].first;
	sort(a+1,a+n+1);
	reverse(a+1,a+n+1);
	int ans=0,c=1;
	for(int i=1;i<=n && c>0;i++)
	{
		c--;
		c+=a[i].first;
		ans+=a[i].second;
	}
	cout<<ans<<endl;
	return 0;
}
