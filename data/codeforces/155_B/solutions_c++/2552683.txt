#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,k=1,count=0;
    cin>>n;
    pair<int,int> a[1001];
    for(int i=0;i<n;i++)
    cin>>a[i].second>>a[i].first;
    sort(a,a+n);
    for(int i=n-1;i>=0&&k!=0;i--)
    {
        count += a[i].second;
        k += a[i].first-1;
    }
    cout<<count<<"\n";
}
