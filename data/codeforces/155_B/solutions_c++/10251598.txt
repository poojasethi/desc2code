

#include<iostream>
#include<algorithm>
using namespace std;
pair<int ,int>a[1010];
int ans=0,k=1,i,n;
int main(){
    cin>>n;
    for(i=0;i<n;i++)
        cin>>a[i].second>>a[i].first;
    sort(a,a+n);
    for(i=n-1;i>=0&&k>0;i--)
        {
        ans+=a[i].second;
        k+=a[i].first-1;
        }
    cout<<ans<<endl;
    return 0;
}
