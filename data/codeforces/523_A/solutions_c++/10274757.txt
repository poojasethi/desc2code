#include <iostream>

using namespace std;

int main() {
  int w, h;
  cin >> w >> h;
  char a[101][101];
  
  for (int i = 0; i < h; i++)
    for (int j = 0; j < w; j++) {
      int x = h - 1 - i;
      int y = j;
      x = h -x -1;
      cin >> a[y][x];
    }
  for (int i = 0; i < w * 2; i++) {
    for (int j = 0; j < h * 2; j++)
      cout << a[i/2][j/2];
    cout << endl;
  }
  return 0;
}
