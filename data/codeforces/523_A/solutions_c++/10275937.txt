#include <iostream>

using namespace std;

int main()
{
	int w, h;
	cin >> h >> w;
	char mas[100][100];
	for (int i = 0; i < w; ++i)
	{
		for (int j = 0; j < h; ++j)
		{
			cin >> mas[i][j];
		}
	}
	for (int i = 0; i < h; ++i)
	{
		for (int h = 0; h < 2; ++h)
		{
			for (int j = 0; j < w; ++j)
			{
				cout << mas[j][i] << mas[j][i];
			}
			cout << endl;
		}
	}
	return 0;
}
