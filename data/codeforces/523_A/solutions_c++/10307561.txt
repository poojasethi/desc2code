#include <cstdio>
#include <iostream>
using namespace std;
int W,H;
string S;
char A[201][201];
int main()
{
    cin>>W>>H;
    for (int i=0;i<H;i++)
    {
	cin>>S;
	for (int j=0;j<W;j++)
	{
	    A[j*2][i*2]=A[j*2+1][i*2]=A[j*2][i*2+1]=A[j*2+1][i*2+1]=S[j];
	}
    }
    for (int i=0;i<2*W;i++)
    {
	for (int j=0;j<2*H;j++)
	    cout<<A[i][j];
	cout<<endl;
    }
    return 0;
}
