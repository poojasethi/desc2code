using namespace std;

char c[102][102];

int main() 
{
	int n, m, i, j;
	cin >> m >> n;
	for (i = 0; i < n; i++)
		for (j = 0; j < m; j++)
			cin >> c[i][j];
	for (j = 0; j < m; j++)
	{
		for (i = 0; i < n; i++)
			cout << c[i][j] << c[i][j];
		cout << endl;
		for (i = 0; i < n; i++)
			cout << c[i][j] << c[i][j];
		cout << endl;
	}
	return 0;
}
