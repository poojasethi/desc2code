#include <cstdlib>
#define N 205

{
	int n,m;
	scanf("%d%d",&m,&n);
	static char s[N],a[N][N];
	for(int i=1;i<=n;++i)
	{
		scanf("%s",s+1);
		int ii=i*2;
		for(int j=1;j<=m;++j)
			a[j][ii-1]=a[j][ii]=s[j];
	}
	for(int j=1;j<=m;++j)
	{
		a[j][n*2+1]='\0';
		puts(a[j]+1),puts(a[j]+1);
	}
	return 0;
}
