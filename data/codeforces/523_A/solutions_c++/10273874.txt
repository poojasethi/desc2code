#include <iostream>

using namespace std;

char a[200][200];

int main()
{
    int w, h;
    cin >> w >> h;
    for (int i =0; i < h; i++)
        for (int j = 0; j < w; j++ )
            cin >> a[i][j];

    for (int i = 0; i < w*2; i++)
    {
        for (int j = 0; j < h*2; j++)
            cout << a[j/2][i/2];
        cout << endl;
    }

    return 0;
}
