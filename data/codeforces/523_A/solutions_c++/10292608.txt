#include <stdio.h>

int main(){
    char img[100][100],c;
    int i,j,w,h;
    
    scanf("%d%d", &w,&h);
    
    for (i=0;i<h;i++)
        scanf("%s",&img[i]);
    for(i=0;i<2*w;i++){
        for(j=0;j<2*h;j++){
            printf("%c",img[j/2][i/2]);
        }
        printf("\n");   
    }
    
    return 0;
}
