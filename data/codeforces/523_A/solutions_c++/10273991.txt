#include <cstdio>

const int maxD = 1000;
char field[maxD][maxD];
int w, h;

int main() {
	scanf("%d%d", &w, &h);
	for (int i = 0; i < h; ++i)
		scanf("%s", field[i]);
	w <<= 1;
	h <<= 1;
	for (int j = 0; j < w; ++j)
	{
		for (int i = 0; i < h; ++i)
			putchar(field[i >> 1][j >> 1]);
		printf("\n");			
	}
	return 0;
}
