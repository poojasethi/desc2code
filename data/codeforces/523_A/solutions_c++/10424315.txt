#include<iostream>
#include<cstring>
#include<string>

using namespace std;

string s[200];

int main()
{
	int w,h;
	cin>>w>>h;
	for(int i=1;i<=h;i++)
		cin>>s[i];
	for(int i=0;i<w;i++)
		for(int j=1;j<=2;j++)
		{
			for(int x=0;x<h;x++)
				cout<<s[ x+1 ][ i ]<<s[x+1][i] ;
			cout<<endl;
		}
	return 0;
}
