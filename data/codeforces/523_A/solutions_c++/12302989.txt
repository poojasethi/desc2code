#include <stdio.h>
#include <algorithm>
#include <iostream>

#define read(x) scanf("%d",&x)

using namespace std;

int n,m;
string str[200];

int main(){
	read(n);read(m);
	for(int k=0;k<m;k++)
        cin>>str[k];

	for(int k=0;k<2*n;k++){
        for(int k2=0;k2<2*m;k2++){
            cout<<str[k2/2][k/2];
        }cout<<endl;
	}

	return 0;
}
