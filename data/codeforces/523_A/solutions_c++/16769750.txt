#include <iostream>
#include <cstdio>

using namespace std;

int main ()
{
	int h, w;
	cin >> w >> h;
	
	char a[110][110];
	for (int i = 0; i < h; i++)
		scanf ("%s", a[i]);
			
	for (int i = 0; i < 2*w; i++)
	{
		for (int j = 0; j < 2*h; j++)
			cout << a[j/2][i/2];
		cout << endl;
	}
	
	return 0;
}
