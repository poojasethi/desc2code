#include <bits/stdc++.h>
using namespace std;

char A[105][105];
int main()
{
	int w,h;
	scanf("%d %d",&w,&h);

	for (int i = 0; i < h; ++i)
		for (int j = 0; j < w; ++j)
			cin>>A[i][j];

	for (int i = 0; i < 2*w; ++i)
	{
		for (int j = 0; j < 2*h; ++j)
		{
			printf("%c",A[j/2][i/2]);
		}
		printf("\n");
	}
	return 0;
}
