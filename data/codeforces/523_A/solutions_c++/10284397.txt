	for (int i = 0; i < h; ++i)

		cin >> s[i];
	}

	for (int i = 0; i < w; ++i)
	{
		for (int j = 0; j < h; ++j)
		{
			cout << s[j][i] << s[j][i];
		}
		cout << endl;

		{
			cout << s[j][i] << s[j][i];
		}
		cout << endl;
	}
}
