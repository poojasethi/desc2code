#include "bits/stdc++.h"
using namespace std;
char mat[110][110];
int m, n;

int main()
{
	scanf("%d%d", &m, &n);
	int i, j;
	for(i = 0; i <= n - 1; ++i)
		scanf("%s", mat[i]);
	for(i = 0; i <= (m << 1) - 1; ++i) {
		for(j = 0; j <= (n << 1) - 1; ++j) {
			putchar(mat[j >> 1][i >> 1]);
		}
		puts("");
	}
}
