#include <iostream>
#include <cstdio>
using namespace std;
int w,h;
char s[110][110];
int main(){
    while(~scanf("%d%d",&w,&h)){
        for(int i=0;i<h;i++) scanf("%s",s[i]);
        for(int i=0;i<2*w;i++){
            for(int j=0;j<2*h;j++){
                printf("%c",s[j/2][i/2]);
            }
            printf("\n");
        }
    }
    return 0;
}
