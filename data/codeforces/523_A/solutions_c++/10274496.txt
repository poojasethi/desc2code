#include <stdio.h>

char img[102][102];

int main()
{
  int w,h;
  scanf("%d%d", &w, &h);
  for (int i = 0; i < h; i++)
    scanf("%s", img[i]);
  for (int i = 0; i < 2*w; i++)
  {
    for (int j = 0; j < h; j++)
      printf("%c%c", img[j][i/2], img[j][i/2]);
    if (i != 2*w-1)
      printf("\n");
  }
  return 0;
}
