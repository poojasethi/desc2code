#include <bits/stdc++.h>
using namespace std;
char a[101][101];
int main()
{
    int w, h;
    scanf("%d%d", &w, &h);
    for (int i = 0; i < h; i++)
        scanf("%s", a[i]);
    for (int i = 0; i < w + w; i++)
        for (int j = 0; j < h + h; j++)
            printf(j==h+h-1?"%c\n":"%c",a[j>>1][i>>1]);
    return 0;
}
