#include<bits/stdc++.h>
using namespace std;
const int maxn=500;
char c[maxn][maxn];
int main(){
	int n, m;
	cin >> m >> n;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			cin >> c[i][j];
	for(int j=0;j<m;j++){
		for(int i=0;i<n;i++)
			cout << c[i][j] << c[i][j];
		cout << endl;
		for(int i=0;i<n;i++)
			cout << c[i][j] << c[i][j];
		cout << endl;
	}
}
