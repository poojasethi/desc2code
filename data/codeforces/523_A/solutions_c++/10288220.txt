#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
int w,h;
char m[111][111];
int main()
{
	scanf("%d%d", &w, &h);
	for (int i = 0; i < h; i++)
	{
		scanf("%s", m[i]);
	}
	for (int i = 0; i < w; i++)
	{
		for (int z = 0; z < 2; z++)
		{
			for (int j = 0; j < h; j++)
			{
				printf("%c%c", m[j][i], m[j][i]);
			}
			printf("\n");
		}
	}
	return 0;
}
