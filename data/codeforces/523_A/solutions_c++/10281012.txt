#include <iostream>

using namespace std;

int main() {
	int w, h;
	char**img;
	
	cin >> w >> h;

	img = new char*[h];
	for(int i = 0; i < h; i ++)
		img[i] = new char[w];

	for(int i = 0; i < h; i++)
		for(int j = 0; j < w; j++)
			cin >> img[i][j];

	for(int j = 0; j < 2*w; j++) {
		for(int i = 0; i < 2*h; i++)
			cout << img[i/2][j/2];
		cout << '\n';
	}
}
