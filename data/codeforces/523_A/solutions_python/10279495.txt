#!/usr/bin/python
import sys

def main():
        params = sys.stdin.readline().split()
        w,h = int(params[0]), int(params[1])
        s = sys.stdin.read()
        matrix = s.strip('\n').split('\n')

        for i, l in enumerate(matrix):
                matrix[i] = list(l)

        permute = [[0 for x in range(2 * h)] for x in range(2 * w)]
        for i in range(h):
                for j in range(w):
                        permute[2 * j]    [2 * i]     = matrix[i][j]
                        permute[2 * j + 1][2 * i]     = matrix[i][j]
                        permute[2 * j]    [2 * i + 1] = matrix[i][j]
                        permute[2 * j + 1][2 * i + 1] = matrix[i][j]

        for line in permute:
                print(''.join(line))


if __name__ == "__main__":
    main()
