#!/usr/bin/env python

import sys

w, h = map(int, sys.stdin.readline().split())

img = []
for i in xrange(h):
	img.append(list(sys.stdin.readline()))


def makeIncreased(img):
	global h, w
	res = [ ['0'] * h*2 for i in xrange(w*2)]
	for i in xrange(h):
		for j in xrange(w):
			res[j*2][i*2] = res[j*2+1][i*2]= res[j*2][i*2+1]=  res[j*2+1][i*2+1] = img[i][j]
	return res

def print_(img):
	for i in img:
		print ''.join(i)	

img = makeIncreased(img)
print_(img)
