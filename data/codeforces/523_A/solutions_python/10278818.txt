w, h = map(int, raw_input().split())
im = [raw_input() for i in xrange(h)]
newIm = [[None] * (2*h) for i in xrange(2*w)]
for i in xrange(h):
    for j in xrange(w):
        I,J = 2*j, 2*i
        c = im[i][j]
        newIm[I][J],newIm[I+1][J],newIm[I+1][J+1], newIm[I][J+1] = c,c,c,c
print '\n'.join(''.join(s) for s in newIm)
