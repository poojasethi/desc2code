# inpute
w, h = tuple(raw_input().split(' '))
w = int(w)
h = int(h)
pic = list()
for i in xrange(h):
    pic.append(list(raw_input()))

# turn
pic = map(list, zip(*pic))

# reflection
for i in pic:
    i = i[::-1]

# doubling
temp = list()
for i in pic:
    small_temp = list()
    n = 0
    for element in i:
        small_temp.insert(n, element)
        n += 1
        small_temp.insert(n, element)
        n += 1
    temp.append(small_temp)
    temp.append(small_temp)

# output
for i in temp:
    s = str()
    for el in i:
        s += el
    print s
