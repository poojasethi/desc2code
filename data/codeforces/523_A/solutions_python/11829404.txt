w, h = map(int, raw_input().split())
board = []
for row in xrange(h):
    board.append(list(raw_input().split()[0]))
newboard = [[] for col in xrange(2*w)]
for col in xrange(w):
    for row in xrange(h):
        newboard[col*2].append(board[row][col])
        newboard[col*2+1].append(board[row][col])
        newboard[col*2].append(board[row][col])
        newboard[col*2+1].append(board[row][col])

for item in newboard:
    print "".join(item)
