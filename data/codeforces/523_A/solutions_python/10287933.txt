#!/usr/bin/python
import sys
size = sys.stdin.readline().split()
row = int(size[0])
col = int(size[1])

A = [0] * row
for i in range(row):
        A[i] = [0] * col

for i in range(col):
        line = sys.stdin.readline()
        for j in range(row):
                A[row-j-1][i]=line[j]

for i in range(row):
        S = ''
        for j in range(col):
                S+=str(A[row - i - 1][j])*2
        print S
        print S
