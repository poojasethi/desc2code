#include <bits/stdc++.h>
typedef long long LL;
using namespace std;
LL f[400000], g[400000], n, k;

LL dp(LL n){
	LL m = 0;
	for(m = 1; m*m <= n; m++) f[m] = n/m-1;
	for(LL i = 1; i <= m; i++) g[i] = i-1;
	for(LL i = 2; i <= m; i++){
		if (g[i] == g[i-1]) continue;
		for(LL j = 1; j <= min(m-1,n/i/i); j++){
			if (i*j < m) f[j] -= f[i*j]-g[i-1];
			else f[j] -= g[n/i/j]-g[i-1];
		}
		for(LL j = m; j >= i*i; j--) g[j] -= g[j/i]-g[i-1];
	}
	LL res = 0;
	for(LL i = 2; i < m; i++){
		if (g[i] == g[i-1]) continue;
		res += f[i]-g[i];
	}
	return res;
}

int main(){
	scanf("%I64d", &n);
	k = pow(n,1.0/3)+1e-9;
	printf("%I64d\n", dp(n)+g[k]);
}
