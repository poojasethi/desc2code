#define MAX 316230
#define ll long long

#define sd(t) scanf("%d",&t)
#define pd(t) printf("%d\n",t)
#define plld(t) printf("%lld\n",t)

#define pll pair<ll,ll>
#define tr(container,it) for(typeof(container.begin()) it=container.begin();it!=container.end();it++)
#define mp(a,b) make_pair(a,b)

#define pb(x) push_back(x)
#define vi vector<int>

using namespace std;

ll H[MAX],L[MAX],i,v,l,p,LIM;

ll DSUM(ll N,ll P){
	
	for(i=1;i*i<=N;i++) H[i]=(N/i)-1;
	l=N/i+1;
	for(v=1;v<=l;v++) L[v]=v-1;
	
	if((i-1)<P) LIM=i-1;					// CAPS P only used here!
	else LIM=P;
	
	for(p=2;p<=LIM;p++){
		if(L[p]==L[p-1]) continue;
		
		ll limx=N/(p*p);
		if(limx>=i-1) limx=i-1;
		
		for(ll x=1;x<=limx;x++){
			H[x]-=(((x*p<=(i-1))?H[x*p]:L[N/(x*p)])-L[p-1]);
		}
		for(ll x=l;x>=p*p;x--){
			L[x]-=(L[x/p]-L[p-1]);
		}
	}
	
	ll ans=0;
	
	for(p=2;p<i;p++){
		if(L[p]==L[p-1]) continue;
		ans+=H[p]-L[p];
	}
	
	return ans;
}

int main(){
	ll n;
	slld(n);
	
	ll ct=pow(n,(long double)1.0/3)+1e-9;
	while(ct*ct*ct<=n) ct++;
	ct--;
	
	ll ans=DSUM(n,n);
	
	ans+=L[ct];
	
	plld(ans);
}
