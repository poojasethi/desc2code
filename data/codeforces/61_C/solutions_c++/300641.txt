#include <cassert>
#include <cstdlib>
#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
#include <set>
#include <iterator>
#include <numeric>
#include <deque>
#include <stack>
#include <sstream>

int main()
{
  int a;
  std::cin >> a;
  std::string b;
  std::cin >> b;
  std::string c;
  std::cin >> c;
  long long d = strtoll(c.c_str(), 0, a);
  std::string ans;
  if(b != "R") {
    if(d == 0) {
      ans = "0";
    } else {
      int base = strtol(b.c_str(), 0, 10);
      std::stringstream ss;
      while(d > 0) {
        int r = d%base;
        if(r >= 10) {
          ss << (char)(r-10+'A');
        } else {
          ss << r;
        }
        d /= base;
      }
      ans = ss.str();
      std::reverse(ans.begin(), ans.end());
    }
  } else {
    std::string t;
    int q;
    q = d/1000;
    ans += std::string(q, 'M');
    d = d%1000;
    
    if(d >= 900) {
      ans += "CM";
      d -= 900;
    } else if(d >= 500) {
      ans += "D";
      d -= 500;
    } else if(d >= 400) {
      ans += "CD";
      d -= 400;
    }

    q = d/100;
    ans += std::string(q, 'C');
    d = d%100;

    if(d >= 90) {
      ans += "XC";
      d -= 90;
    } else if(d >= 50) {
      ans += "L";
      d -= 50;
    } else if(d >= 40) {
      ans += "XL";
      d -= 40;
    }

    q = d/10;
    ans += std::string(q, 'X');
    d = d%10;

    if(d >= 9) {
      ans += "IX";
      d -= 9;
    } else if(d >= 5) {
      ans += "V";
      d -= 5;
    } else if(d >= 4) {
      ans += "IV";
      d -= 4;
    };
    ans += std::string(d, 'I');
  }
  std::cout << ans << std::endl;
}
