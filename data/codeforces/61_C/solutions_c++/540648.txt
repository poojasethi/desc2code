#include <iostream>
#include <sstream>
#include <map>

using namespace std;

int atoi(string a)
{
	int b;
	stringstream s;
	s << a;
	s >> b;
	return b;
}

long long dec(string a, int b)
{
	long long m = 1;
	long long r = 0;

	for (int i = a.size() - 1; i >= 0; i--)
	{
		int s;
		if ((a[i] >= '0') && (a[i] <= '9'))
			s = a[i] - '0';
		if ((a[i] >= 'A') && (a[i] <= 'Z'))
			s = a[i] - 'A' + 10;

		r += m * s;
		m *= b;
	}

	return r;
}

string sys(long long a, int b)
{
	string r;
	do
	{
		char s;
		if ((a % b >= 0) && (a % b <= 9))
			s = '0' + (a % b);
		if (a % b >= 10)
			s = 'A' + (a % b - 10);

		r = s + r;
		a /= b;
	}
	while (a != 0);

	return r;
}

string rim(int a)
{
	string r;
	map<int,string> c;
	c[1000] = "M";
	c[900] = "CM";
	c[500] = "D";
	c[400] = "CD";
	c[100] = "C";
	c[90] = "XC";
	c[50] = "L";
	c[40] = "XL";
	c[10] = "X";
	c[9] = "IX";
	c[5] = "V";
	c[4] = "IV";
	c[1] = "I";

	map<int,string>::reverse_iterator p;

	for (map<int,string>::reverse_iterator i = c.rbegin(); i != c.rend(); i++)
	{
		while (a >= i->first)
		{
			r += i->second;
			a -= i->first;
		}

		p = i;
	}

	return r;
}

int main(int argc, char* argv[])
{
	int a;
	string b, c;
	cin >> a >> b >> c;

	long long d = dec(c, a);

	if (b != "R")
		cout << sys(d, atoi(b)) << endl;
	else
		cout << rim(d) << endl;

	return 0;
}
