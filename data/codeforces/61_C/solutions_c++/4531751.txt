#include<iostream>
#include<algorithm>
#include<string>

#define int long long
using namespace std;
string rom[13] = {"I","IV","V","IX","X","XL","L","XC","C","CD","D","CM","M"};
int num[13] = {1,4,5,9,10,40,50,90,100,400,500,900,1000};
string D = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
string a,b,c;

int V(char c)
{
    for (int i=0;i<D.size();i++)
        if (c == D[i])
            return i;
}

string roman(int n)
{
    string res = "";
    for (int i=12;i>=0;i--)
        if (n >= num[i])
        {
            res += rom[i];
            n -= num[i];
            i++;
        }
    return res;
}

int dec(string s,int b)
{
    reverse(s.begin(),s.end());
    int p = 1;
    int res = 0;
    for (int i=0;i<s.size();i++)   
    {
        res += V(s[i]) * p;
        p *= b;
    }
    return res;
}

string base(int n,int b)
{
    if (n == 0)
        return "0";
    string res;
    while(n)
    {
        res += char(D[n % b]);
        n /= b;
    }
    reverse(res.begin(), res.end());
    return res;
}

main()
{
    cin>>a>>b>>c;
    int vv = dec(c, atoll(a.c_str()));
    if (b == "R")
        cout<<roman(vv)<<endl;
    else
        cout<<base(vv, atoll(b.c_str()))<<endl;
    return 0;
}
