#include <iostream>
#include <vector>
#include <string>

using namespace std;

char numero[4][10][5]=
  {{"","I","II","III","IV","V","VI","VII","VIII","IX"},
   {"","X","XX","XXX","XL","L","LX","LXX","LXXX","XC"},
   {"","C","CC","CCC","CD","D","DC","DCC","DCCC","CM"},
   {"","M","MM","MMM","","","","","","" }};

string pasaaromano(int x)
{
  string s;
  int index=0;
  while (x>0) {
    s=string(numero[index][x%10])+s;
    index++;
    x/=10;
  }
  return s;
}

typedef long long int lint;

lint pasaalint(int a,string c)
{
  lint r=0;
  for (int i=0;i<int(c.size());i++) {
    lint val;
    if (c[i]>='0' and c[i]<='9')
      val=c[i]-'0';
    else
      val=10+c[i]-'A';
    r=r*a+val;
  }
  return r;
}

lint pasaaint(string b)
{
  int x=0;
  for (int i=0;i<int(b.size());i++)
    x=x*10+b[i]-'0';
  return x;
}

string pasaabase(lint b,lint c)
{
  if (c==0) return "0";
  string s;
  while (c>0) {
    lint val=c%b;
    if (val>=10)
      s=string(1,char('A'+val-10))+s;
    else
      s=string(1,char('0'+val))+s;
    c/=b;
  }
  return s;
}

int main()
{
  int a;
  cin>>a;
  string b,c;
  cin>>b>>c;
  lint val=pasaalint(a,c);
  if (b=="R")
    cout<<pasaaromano(val)<<endl;
  else
    cout<<pasaabase(pasaaint(b),val)<<endl;
}
