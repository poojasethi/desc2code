#include<iostream>
#include<string>

using namespace std;


long long convertBase10(string num, int base)
{
    long long res = 0;
    for(int i = 0 ; i < num.size() ; i++)
    {
        res *= base;
        if(num[i] >= '0' && num[i] <= '9' )
            res += (num[i] - '0');
        else if(num[i] >= 'A' && num[i] <= 'Z')
            res+= (num[i] - 'A' + 10);
    }
    return res;
}

void convertRoman(long long input)
{
    const string roman[13] = { "M", "CM", "D", "CD", "C","XC", "L", "XL", "X", "IX", "V", "IV", "I"};
    const int decimal[13] = {1000, 900, 500, 400, 100, 90,50, 40, 10, 9, 5, 4, 1};
    string romanvalue = "";
    for (int i = 0; i < 13; i++)
    {
        while (input >= decimal[i])
        {
            input -= decimal[i];
            romanvalue += roman[i];
        }
    }
    cout<<romanvalue;
}


void convertB(long long num, string b)
{
    if(b == "R")
    {
        convertRoman(num);
        return;
    }
    int base = 0;
    for(int i = 0 ; i < b.size() ; i++)
    {
        base *= 10;
        base += (b[i] - '0');
    }

    string s = "";
    if(num == 0)
        s = "0";

    while(num)  
    {
        int rem = num%base;
        char ch ;
        if(rem < 10)
            ch = '0' + rem;
        else
            ch = 'A' + rem -10;

        s = string(1,ch) + s;  

        num /= base;    
    }

    cout<<s;
    return;  
}


int main()
{
    int a;
    string b;
    string c;

    cin>>a>>b>>c;

    long long num = convertBase10(c,a);
    convertB(num,b);
    return 0;
}
