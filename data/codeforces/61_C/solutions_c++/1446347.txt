#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <string>
using namespace std;

int a, b;
long long c;
string st;
pair<int, string> f[13];

void init()
{
	cin >> a >> st;
	int n;
	if (st == "R") 
		b = -1;
	else {
		b = 0; n = st.size();
		for (int i = 0; i < n; i++) 
			b = b * 10 + st[i] - '0';
	}	
	cin >> st; n = st.size(); c = 0;
	for (int i = 0; i < n; i++) {
		if (st[i] <= '9')
			c = c * a + st[i] - '0';
		else
			c = c * a + 10 + st[i] - 'A';
	}
}

void solve()
{
	st = ""; 
	int n; 
	char z;
	if (c) {
		while (c) {
			n = c % b;
			c /= b;
			if (n < 10) z = '0' + n; else z = 'A' + n - 10;
			st += z;
		}
		n = st.size();
		for (int i = n - 1; i >= 0; i--) 
			cout << st[i];
		cout << endl;
	} else 
		cout << 0 << endl;
}

void print()
{
	f[0].first = 1; f[0].second = "I";
	f[1].first = 4; f[1].second = "IV";
	f[2].first = 5; f[2].second = "V";
	f[3].first = 9; f[3].second = "IX";
	f[4].first = 10; f[4].second = "X";
	f[5].first = 40; f[5].second = "XL";
	f[6].first = 50; f[6].second = "L";
	f[7].first = 90; f[7].second = "XC";
	f[8].first = 100; f[8].second = "C";
	f[9].first = 400; f[9].second = "CD";
	f[10].first = 500; f[10].second = "D";
	f[11].first = 900; f[11].second = "CM";
	f[12].first = 1000; f[12].second = "M";
	for (int i = 12; i >= 0; i--) {
		while (c >= f[i].first) {
			cout << f[i].second;
			c -= f[i].first;
		}
	}
	cout << endl;
}

int main(){
	init();
	if (b == -1) 
		print();
	else 
		solve();
	return 0;
}
