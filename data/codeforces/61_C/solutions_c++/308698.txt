#include <iostream>
#include <stdio.h>

using namespace std;

void Print(int res)
{
	if(res<=9)
	{
		printf("%d",res);
	}
	else
	{
		res-=10;
		printf("%c",res+'A');
	}
}

void ConvertToBase(long long num,int c)
{
	long long base = 1;
	while(base*c<=num)
	{
		base*=c;
	}
	
	
	while(base>0)
	{
		long long res = num/base;
		
		num -= res*base;
		
		Print(res);
		
		base/=c;
	}
	
	printf("\n");
	
}

void ConvertToRoman(long long num)
{
	 string res;
	
     while (num >= 1000) { num -= 1000; res += "M"; }
     if (num >= 900) { num -= 900; res += "CM"; }
     while (num >= 500) { num -= 500; res += "D"; }
     if (num >= 400) { num -= 400; res += "CD"; }
     while (num >= 100) { num -= 100; res += "C"; }
     if (num >= 90) { num -= 90; res += "XC"; }
     while (num >= 50) { num -= 50; res += "L"; }
     if (num >= 40) { num -= 40; res += "XL"; }
     while (num >= 10) { num -= 10; res += "X"; }
     if (num >= 9) { num -= 9; res += "IX"; }
     while (num >= 5) { num -= 5; res += "V"; }
     if (num >= 4) { num -= 4; res += "IV"; }
     while (num >= 1) { num--; res += "I"; }
     
     cout<<res<<endl;

}

int main()
{
	int a;
	int b;
	char ch[4];
	
	scanf("%d %s",&a,ch);
	
	if(ch[0]=='R')
	{
		b=0;
	}
	else
	{
		sscanf(ch,"%d",&b);
	}
	
	long long c=0;
	
	string str;
	cin>>str;
	
	for(unsigned int n=0;n<str.length();n++)
	{
		c*=a;
		if(str[n]>='0' && str[n]<='9')
		{
			c+=str[n]-'0';
		}
		else
		{
			c+=10+(str[n]-'A');
		}
	}
	
	if(b==(0))
	{
		ConvertToRoman(c);
	}
	else
	{
		ConvertToBase(c,b);
	}
	
	
}
