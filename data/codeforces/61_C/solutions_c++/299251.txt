using namespace std;

LL base[] = {1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1};   
string romCh[] = {"M","CM","D","CD","C","XC","L","XL","X","IX","V","IV","I"};

string roman(LL n)
{
    string ans = "";
    for (int i = 0; i < 13; i ++)
        for ( ; n >= base[i]; )
           ans += romCh[i], n -= base[i];
    return ans;
}
