#include <iostream>
#include <cstring>
#include <algorithm>
#include <cctype>
#include <vector>
using namespace std;
#define ll long long
#define pb push_back
ll a;
string c,ans;
char b[3];
ll aTodec(string t,ll a)
{
  ll st=1,ans=0;
  for(int i=t.size()-1;i>=0;i--,st*=a)
    ans+=(isdigit(t[i])?t[i]-'0':t[i]-'A'+10)*st;
  return ans;
}
string aTob(ll x,int b)
{
  string an;
  if (x==0)an='0';
  while (x)
  {
    an=char((x%b<10?x%b+'0':x%b+'A'-10))+an;
    x/=b;
  }    
  return an;
}
string digitInRome(int x)
{
  if (x==0)return "";
  int c[8]={1,5,10,50,100,500,1000,5000},i=0;
  char g[8]={'I','V','X','L','C','D','M','-'};
  while (i<7 && x>c[i+1])i++;
  int add,f1=0,f2=0;
  char addc;
  if (i%2==0)add=c[i],addc=g[i];
  else add=c[i-1],addc=g[i-1];
  int u1=c[i],u2=c[i+1];
  string t1,t2;
  t1+=g[i],t2+=g[i+1];
  while (u1!=x)u1+=add,t1=t1+addc,f1++;
  while (u2!=x)u2-=add,t2=addc+t2,f2++;
  if (f2>1)return t1;
  return t2;
}
string decTorome(ll x)
{
  vector <int> f;
  string ans;
  if (x==0)f.pb(0);
  while (x)f.pb(x%10),x/=10;
  reverse(f.begin(),f.end());
  ll st=1;
  for(int i=1;i<f.size();i++)st*=10;
  for(int i=0;i<f.size();i++,st/=10)ans=ans+digitInRome(f[i]*st);
  return ans;
}
int main()
{
  cin>>a>>b>>c;
  ll x=aTodec(c,a);
  if (b[0]!='R')ans=aTob(x,atoi(b));
  else ans=decTorome(x);
  cout<<ans;
  return 0;
}
