import string
H=string.digits+string.uppercase
R=raw_input
a,b=R().split()
c=int(R(),int(a))
s=''
def roman(d,i,v,x):
  return '' if d<1 else i*d if d<4 else i+v if d<5 else v if d<6 else v+i*(d-5) if d<9 else i+x
if 'R'==b:
  s,c=roman(c%10,'I','V','X')+s,c/10
  s,c=roman(c%10,'X','L','C')+s,c/10
  s,c=roman(c%10,'C','D','M')+s,c/10
  s='M'*c+s
else:
  b=int(b)
  while c:
    s=H[c%b]+s
    c/=b
print s or '0'
