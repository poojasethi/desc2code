#!/usr/bin/python
import math, random, re, sys

def ctod (c):
	if c >= '0' and c <= '9':
		return ord (c) - ord ('0')
	else:
		return ord (c) - ord ('A') + 10

def dtoc (d):
	if d < 10:
		return chr (d + ord ('0'))
	else:
		return chr (d + ord ('A') - 10) 

while True:
        s = sys.stdin.readline ().strip ()
        if s == '':
	        s = sys.stdin.readline ().strip ()
        	if s == '':
        		break
	a, b = s.split ()
	c = sys.stdin.readline ().strip ()
	n = int (a)
	k = 0
	for d in c:
		k = k * n + ctod (d)
	s = ''
	if b == 'R':
		p = {1: 'I', 4: 'IV', 5: 'V', 9: 'IX', 10: 'X', 40: 'XL', \
		 50: 'L', 90: 'XC', 100: 'C', 400: 'CD', 500: 'D', \
		 900: 'CM', 1000: 'M'}
		for x, y in sorted (p.items (), reverse = True):
			while k >= x:
				s += y
				k -= x
	else:
		m = int (b)
		while k > 0:
			s += dtoc (k % m)
			k /= m
		s = s[::-1]
	if s == '':
		s = '0'
	print s
