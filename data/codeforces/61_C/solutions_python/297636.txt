#! /usr/bin/env python

from sys import stdin

decimalDens=[1000,900,500,400,100,90,50,40,10,9,5,4,1]
romanDens=["M","CM","D","CD","C","XC","L","XL","X","IX","V","IV","I"]

def decToRoman(num,s,decs,romans):
	if decs:
		if (num < decs[0]):
			return decToRoman(num,s,decs[1:],romans[1:])		  
		else:
			return decToRoman(num-decs[0],s+romans[0],decs,romans)	  
	else:
		return s

def toBase(num,base):
	digits="0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ"
	if num==0:
		return "0"
	else:
		ret=""
		while num>0:
			ret+=digits[num%base]
			num/=base
		return ret[::-1]

if __name__=='__main__':
	a,b=stdin.readline().split()
	c=stdin.readline().strip()
	c=int(c,int(a))
	if b=='R':
		print decToRoman(c,"",decimalDens, romanDens)
	else:
		print toBase(c,int(b))
