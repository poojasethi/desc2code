#!/usr/bin/env python

from sys import stdin, stderr
import string

BASE = string.digits + string.ascii_uppercase
def base_encode(x, b):
   ret = ''
   while x != 0:
       ret = BASE[x % b] + ret
       x /= b
   return ret

nums = [1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1]
D = {
   1000 : 'M',
   900  : 'CM',
   500  : 'D',
   400  : 'CD',
   100  : 'C',
   90   : 'XC',
   50   : 'L',
   40   : 'XL',
   10   : 'X',
   9    : 'IX',
   5    : 'V',
   4    : 'IV',
   1    : 'I'
}
def toroman(x):
   ret = ''
   for n in nums:
      m = int(x / n)
      ret += D[n] * m
      x -= n * m
   return ret

def main():
   a, b = stdin.readline().split()
   a = int(a)
   c = stdin.readline().strip()
   num = int(c, a)
#  print >>stderr, 'num =', num
   if num == 0:
      res = '0'
   elif b == 'R':
      res = toroman(num)
   else:
      b = int(b)
      res = base_encode(num, b)
   print res
   return 0

if __name__ == '__main__': main()
