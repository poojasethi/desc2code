#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	pair<int,int> interval[100001];
	cin >> n;
	int i;
	for(i=0;i<n;i++)
		cin >> interval[i].first >> interval[i].second ;
	sort(interval,interval+n);
	int maxi = 0;
	int ans = 0;
	for(i=0;i<n;i++)
	{
		if(interval[i].second < maxi)
			ans++;
		maxi = max(interval[i].second , maxi);
	}
	cout << ans	 << endl;
	return 0;
}
