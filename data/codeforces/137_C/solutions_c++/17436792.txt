#include<bits/stdc++.h>
using namespace std;
main ()
{

int n,i,res=0,mina=0;
cin >>n;

pair <int,int>a[n];
for (i=0;i<n;i++)
{cin>>a[i].first>>a[i].second;}
sort (a,a+n);
for (i=0;i<n;i++)
{
  if (a[i].second<mina)

        res++;
        if (a[i].second>mina)
            mina=a[i].second;



}
cout <<res<<endl;

  return 0;
}
