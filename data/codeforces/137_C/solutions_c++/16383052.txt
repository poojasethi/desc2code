#include <bits/stdc++.h>
#define ii pair<int,int>
#define f first
#define s second
using namespace std;
ii a[100100];
int main(){
	int n,mx=-1,cnt=0;
	cin>>n;
	for(int i=0;i<n;i++)scanf("%d%d",&a[i].f,&a[i].s);
	sort(a,a+n);
	for(int i=0;i<n;i++){
		if(a[i].s>mx)mx=a[i].s;
		else if(a[i].s<mx)cnt++;
	}
	cout<<cnt<<endl;
}
