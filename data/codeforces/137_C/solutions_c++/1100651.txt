#include <iostream>
#include <algorithm>

using namespace std;
typedef unsigned long long ll;
pair <ll, ll> p[100010];

int main()
{int n;
cin >> n;

for (int i=0; i<n; i++)
cin >> p[i].first >> p[i].second;

sort(p,p+n);
int ans=0;
ll maxm=0;
for(int i=0;i<n;i++)
{
    if(p[i].second<=maxm)  ans++;
    maxm=max(maxm,p[i].second);
}
cout << ans;
    return 0;
}
