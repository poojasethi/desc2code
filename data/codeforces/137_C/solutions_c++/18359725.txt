#include<bits/stdc++.h>
using namespace std;
int n, mx = -1, ans = 0;;
pair<int, int> x[100010];

int main(int argc, char **argv) {

	scanf("%d", &n);
	for(int i=0 ; i<n ; ++i)
		scanf("%d%d", &x[i].first, &x[i].second);
	sort(x, x+n);
	for(int i=0 ; i<n ; ++i) 
		if(x[i].second < mx) ++ans;
		else mx = x[i].second;
	cout << ans << endl;
	return 0;
}
