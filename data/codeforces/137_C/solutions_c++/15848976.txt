#include <iostream>
#include <algorithm>
using namespace std;

const int MAXN = 1000 * 100 + 10;
pair <int, int> p[MAXN];

int main()
{
	int n;
	cin>> n;

	for(int i = 0; i < n; i ++)
		cin>> p[i].first >> p[i].second;

	sort(p, p + n);

	int ans = 0;
	int e = p[0].second;
	for(int i = 1; i < n; i ++)
		if(p[i].second < e)
			ans ++;
		else
			e = p[i].second;


	cout<< ans << endl;
}
