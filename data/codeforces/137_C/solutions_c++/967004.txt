#include <algorithm>
#include <iostream>

using namespace std;

int main(){

  int N, rg, cnt=0;
  pair<int, int> ev[100000];

  cin >> N;
  for(int i=0; i<N; i++)
    cin >> ev[i].first >> ev[i].second;

  sort(ev, ev+N);

  rg = ev[0].second;
  for(int i=1; i<N; i++){
    if(ev[i].second<rg) cnt++;
    else rg = ev[i].second;
  }

  cout << cnt << endl;

  return 0;
}
