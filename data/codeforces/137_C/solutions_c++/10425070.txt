#include <iostream>
#include <algorithm>
using namespace std;
int x,i;
int main() {
   cin>>x;
   pair <int,int> a[x];
   long long q=0,w=0;
   for(i=0;i<x;i++)
   {
       cin>>a[i].first>>a[i].second;
   }
   sort(a,a+x);
   for(i=0;i<x;i++)
   {
       if(a[i].second<q)
      w++;else q=a[i].second;
   }
   cout<<w<<endl;
}
