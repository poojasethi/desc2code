#include<bits/stdc++.h>
using namespace std;
vector<pair<int,int> >v;
int main()
{
	int i,n,x,y,c=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		v.push_back(make_pair(x,y));		
	}
	sort(v.begin(),v.end());
	y=v[0].second;
	for(i=1;i<n;i++)
	{
		if(v[i].second<y)
			c++;
		else
			y=v[i].second;	
	}
	cout<<c<<endl;
	return 0;
}
