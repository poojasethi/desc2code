#include <bits/stdc++.h>

using namespace std;

int n, m;
pair<int, int> a[100000];

int main() {
	scanf("%d", &n);
	for(int i = 0; i < n; ++i) 
		scanf("%d %d", &a[i].first, &a[i].second);

	sort(a, a+n);

	m = 0;

	int rmax = a[0].second;
	for(int i = 1; i < n; ++i)
		if (a[i].second < rmax)
			++m;
		else
			rmax = a[i].second;
	printf("%d\n", m);
	return 0;
}
