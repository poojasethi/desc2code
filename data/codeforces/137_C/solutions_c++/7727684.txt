#include <iostream>
#include <algorithm>

using namespace std;


int main() {

   int x;
   cin>>x;
   pair <int,int> b[x];
   long long q=0,w=0;
   for(int i=0;i<x;i++)
   {
       cin>>b[i].first>>b[i].second;
   }
   sort(b,b+x);
   for(int i=0;i<x;i++)
   {
       if(b[i].second<q)w++;else q=b[i].second;
   }
   cout<<w<<endl;
}
