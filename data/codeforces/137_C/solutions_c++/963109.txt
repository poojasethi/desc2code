#include <cstdio>
#include <vector>
#include <algorithm>

using namespace std;

const int N=100010;

pair<int,int> e[N];

int main(void)
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)	scanf("%d%d",&e[i].first,&e[i].second);
	sort(e,e+n);
	int right=0,res=0;
	for(int i=0;i<n;i++)
	{
		if(e[i].second<=right)	res++;
		right=max(right,e[i].second);
	}
	printf("%d\n",res);
	return 0;
}
