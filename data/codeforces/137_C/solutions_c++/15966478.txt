#include<iostream>
#include<utility>
#include<algorithm>
using namespace std;
int main()
{
	int n,max=-1,count=0,p,q;
	cin>>n;
	pair<int,int>a[n];
	for(int i=0;i<n;i++)
	{
		cin>>p>>q;
		a[i]=make_pair(p,q);
	}
	sort(a,a+n);
	for(int i=0;i<n;i++)
	{
		if(a[i].second<max)count++;
		else max=a[i].second;
	}
	cout<<count;
}
