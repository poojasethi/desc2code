#include <cstdio>
#include <utility>
#include <algorithm>
using namespace std;
typedef long long LL;
int n;
pair<int,int> a[100000];
int main() {
	scanf("%d",&n);
	for (int i=0;i<n;i++) scanf("%d%d",&a[i].first,&a[i].second);
	sort(a,a+n);
	int far=0,ans=0;
	for(int i=0;i<n;i++){
		if(a[i].second<=far) ans++;
		far=max(far,a[i].second);
	}
	printf("%d\n",ans);
}
