#include <iostream>
#include <algorithm>
using namespace std;
int main() {
pair<int,int> p[100001];
int n,ans=0,max=0;
cin>>n;
for(int i=0;i<n;i++){
    cin>>p[i].first>>p[i].second;
}
sort(p,p+n);
max=p[0].second;
for(int i=0;i<n-1;i++){
    if(max<p[i+1].second) max=p[i+1].second;
    else ans++;
}
cout<<ans;
	return 0;
}
