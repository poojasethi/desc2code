#include <cstdio>
#include <algorithm>

using namespace std; 
int m,x,n,ans;
pair <int ,int> a[100010];
int main(){
  scanf("%d",&n);
  for(int i=0;i<n;i++)
    scanf("%d%d",&a[i].first,&a[i].second);
    sort(a,a+n);
    m=a[0].second;
  for(int i=1;i<n;i++){
    if(a[i].second<m) ans++;
    else m=a[i].second;        
  }
  printf("%d",ans);
  return 0;    
}
