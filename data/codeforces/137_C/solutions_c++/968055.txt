#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
const int maxn=2*100000;
pair<int,int> arr[maxn];
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>arr[i].first>>arr[i].second;

	sort(arr,arr+n);
	int mx=arr[0].second,ans=0;
	for(int i=1;i<n;i++)
		if(arr[i].second<mx)
			ans++;
		else
			mx=arr[i].second;
	cout<<ans<<endl;
}
