#include<iostream>
#include<algorithm>
using namespace std;
long long n,ans,end,i;
pair <long long,long long> p[100005];

int main(){
	cin>>n;
	for(i=1; i<=n; i++){
		cin>>p[i].first>>p[i].second;
	}
	sort(p+1,p+n+1);
	end = p[1].second;
	for(i=2; i<=n; i++){
		if(p[i].second<end){
			ans++;
			continue;
		}
		end = p[i].second;
	}
	cout<<ans;
}
