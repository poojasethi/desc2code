#include <bits/stdc++.h>
using namespace std;

int main ()
{
	int a,ans=0;
	cin >> a;
	pair<int,int> arr[a];
	for (int i=0;i<a;i++)
	{
		cin >> arr[i].first >> arr[i].second;
	}
	sort (arr,arr+a);
	
	int max=arr[0].second;
	for (int t=1;t<a;t++)
	{
		if (arr[t].second<max)
		{	ans++;	}
		else
		{	max = arr[t].second;	}	
	}
	cout << ans <<endl;	
}
