#include<iostream>
#include<algorithm>
using namespace std;
pair <int , int > a[500002];
bool d[500002];
int main()
{
	int n,i,j,a1,a2,ans=0,mx=0;
	cin>>n;
	for (i=0;i<n;i++)
	{
		cin>>a1>>a2;
		a[i]=make_pair(a1,a2);
	}
	sort (a,a+n);
	mx=a[0].second;
	for (i=0;i<n;i++)
	{
		if (a[i].second>=mx)
		{mx=a[i].second;}
		else ans++;
	}
	cout<<ans<<endl;
	return 0;
}
