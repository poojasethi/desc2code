#include <bits/stdc++.h>
#include <stdlib.h>
using namespace std;
int i,n,mx,t[1000005],c;
pair<int,int> a[1000005];
int main () {
	cin >> n;
	for(i=1;i<=n;i++)
		cin >> a[i].first >> a[i].second;
	sort(a+1,a+n+1);
	for(i=1;i<=n;i++){
		mx=max(a[i].second,mx);
		t[i]=mx;}
	for(i=2;i<=n;i++)
		if(a[i].second<t[i])
			c++;
	cout << c << endl;
	return 0;
}
