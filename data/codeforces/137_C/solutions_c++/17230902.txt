#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int main ()
{
int n,k=0;
cin >>n;
pair <int,int> a[n];
for(int i=0;i<n;i++)
cin >>a[i].first >>a[i].second;
sort(a,a+n);
int mx=a[0].second;
for(int i=1;i<n;i++){
if(a[i].second<mx) k++;
else mx=a[i].second;
}
cout <<k;
    
    }
