#include<iostream>
#include<algorithm>
using namespace std;
pair<int,int>a[100000];
int main(){
	int n,ans = 0;
	cin >> n;
	for(int i = 0;i < n;i++)
		cin >> a[i].first >> a[i].second;
	sort(a,a + n);
	int maximom = a[0].second;
	for(int i = 1;i < n;i++){
		if(a[i].second <= maximom)
			ans++;
			else
				maximom = a[i].second;
		}
	cout << ans << endl;	
	return 0;
}
