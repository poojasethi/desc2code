#include<cstdio>
#include<algorithm>
using namespace std;
#define rep(i,n)	for(int i=0;i<(n);++i)
pair<int,int> a[102400];
int n,x,m;
int main(){
	scanf("%d",&n);
	rep(i,n)scanf("%d%d",&a[i].first,&a[i].second);
	sort(a,a+n);
	m=-10000000;
	rep(i,n){
		if(a[i].second<m)++x;
		m= max(a[i].second,m);
	}
	printf("%d\n",x);
	return 0;
}
