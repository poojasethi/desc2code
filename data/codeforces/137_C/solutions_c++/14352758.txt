#include <bits/stdc++.h>
using namespace std;

int main(){
 int a,total=0;
 cin>>a;
 pair<int,int> pol[a];
 for(int i=0; i<a; i++)
 {
  cin >> pol[i].first >> pol[i].second;
 }
 sort(pol,pol+a);

 int max = pol[0].second;
 for(int t=1;t<a;t++)
 {
  if(pol[t].second < max)
   	total++;
  else
   	max=pol[t].second;
 }
 cout<<total<<endl;
}
