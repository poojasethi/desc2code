#include<bits/stdc++.h>
using namespace std;
pair < int , int > v[100000];
int n,x,y,mx,ans;
main ()
{
    cin>>n;
     for (int i=1;i<=n;++i)
     {
        cin>>x>>y;
        v[i]=make_pair(x,y);
     }
     sort(v+1,v+n+1);
      mx=v[1].second;
   for ( int i=2;i<=n;++i)
   {
       if (v[i].second<mx) ans++;
       mx=max(mx,v[i].second);
   }
   cout<<ans;
}
