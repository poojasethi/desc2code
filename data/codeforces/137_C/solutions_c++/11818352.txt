#include<bits/stdc++.h>
using namespace std;
int a,b,n,i,dg=0,ans;
pair<int,int> p[100004];
main()
{
    scanf("%d",&n);
    for (i=0;i<n;i++)
    {
        scanf("%d%d",&a,&b);
        p[i]=make_pair(a,b);
    }
    sort(p,p+n);
    for (i=1;i<n;i++)
        if (p[i].second<p[dg].second)
            ans++;
        else dg=i;
    printf("%d",ans);
}
