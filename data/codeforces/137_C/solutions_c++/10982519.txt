#include <bits/stdc++.h>
using namespace std; 

int main() {
	int n;
	scanf("%d", &n);

	pair <int, int> a[n];
	for(int i = 0; i < n; ++i) scanf("%d%d", &a[i].first, &a[i].second);
	sort(a, a+n);

	int cnt = 0, mx = a[0].second;
	for(int i = 0; i < n; ++i) {
		if(a[i].second < mx) ++cnt;
		else mx = a[i].second;
	}

	printf("%d\n", cnt);

	return 0;
}
