#include<bits/stdc++.h>
using namespace std;

pair <int, int> arr[1000000];

int main()
{
	int n;
	cin >> n;
	for (int i=0; i<n; i++)
		cin >> arr[i].first >> arr[i].second;
	
	sort(arr, arr+n);
	
	int maks= arr[0].second, ct=0;
	for (int i=1; i<n; i++)
	{
		if (arr[i].second<maks)
			ct++;
		else
			maks=arr[i].second;
	}
	
	cout << ct << endl;
}
