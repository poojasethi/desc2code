#include <iostream>
#include <algorithm>

using namespace std;

const int Max=100*1000+10;
pair <int , int> a[Max];
int n;

int main()
{
	cin>>n;
	for(int i=0;i<n;++i)
		cin>>a[i].first>>a[i].second;
	sort(a,a+n);
	int m=a[0].second;
	int sum=0;
	for(int i=1;i<n;++i)
	{
		if(a[i].second<m)
			++sum;
		else
			m=a[i].second;			
	}
	cout<<sum<<endl;
}
