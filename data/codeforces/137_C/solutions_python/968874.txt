n = int(raw_input())
data = sorted([map(int,raw_input().split()) for i in xrange(n)])

ret = 0

i = 0
j = 1
while i < n and j < n:
 ai,bi = data[i] 

 while  j < n:
   aj,bj = data[j]
   if not (ai < aj and bj < bi):
     break

   #print "ok: (%d,%d),(%d,%d)"%(ai,bi,aj,bj)
   ret += 1
   j += 1


 i += 1
 j = max(i+1,j)
 
print ret
