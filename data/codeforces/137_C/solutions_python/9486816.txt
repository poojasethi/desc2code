n = input()
events = []
for i in range(n):
  events.append([int(_) for _ in raw_input().split()])
events = sorted(events)
# print events
ans = 0
latestEnd = events[0][1]
for i1 in range(1, n):
  newEnd = events[i1][1]
  if newEnd < latestEnd:
    ans += 1
  elif newEnd > latestEnd:
    latestEnd = newEnd
print ans
