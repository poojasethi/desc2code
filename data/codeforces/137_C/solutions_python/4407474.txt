# Codeforces Beta Round 98 (Division 2)
# Problem C: History

def containedIn(event1, event2):
    if (event1 == event2):
        return False
    if (event1[0] > event2[0] and event1[1] < event2[1]):
        return True
    return False

def numIncludedEvents(n, eventSpans):
    if (n == 1):
        return 0
    eventSpans = sorted(eventSpans)
    latest = eventSpans[0][1]
    count = 0
    i = 0
    while (i < n):
        if (eventSpans[i][1] > latest):
            latest = eventSpans[i][1]
        if (eventSpans[i][1] < latest):
            count += 1
        i += 1            
    return count

n = raw_input("")
n = int(n)
eventSpans = []
for i in xrange(n):
    line = raw_input("").split(' ')
    line = map(int, line)
    eventSpans += [line]
print numIncludedEvents(n, eventSpans)
