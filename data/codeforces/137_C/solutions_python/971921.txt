n = int(raw_input())
l = []
for i in range(n):
    args = raw_input().split()
    l.append((int(args[0]), int(args[1])))
l.sort(key=lambda tup : tup[0])
res = 0
maxFinish = l[0][1]
for t in l:
    if t[1] < maxFinish:
        res = res + 1
    elif t[1] > maxFinish:
        maxFinish = t[1]
print res
