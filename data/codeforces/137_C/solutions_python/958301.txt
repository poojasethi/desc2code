#!/usr/bin/python
import itertools, math, os, random, re, sys
random.seed (1234)

while True:
	s = sys.stdin.readline ().strip ()
	if s == '':
		s = sys.stdin.readline ().strip ()
		if s == '':
			break
	n = int (s)
	p = []
	for i in range (n):
		q = map (int, sys.stdin.readline ().split ())
		p += [q]
	p.sort ()
	hi = -1
	res = 0
	for i in range (n):
		hi = max (hi, p[i][1])
		if p[i][1] < hi:
			res += 1
	print res
