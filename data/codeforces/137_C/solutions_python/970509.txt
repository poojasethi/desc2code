n=int(raw_input())
class myClass:
    start=1
    end=1
a=[]
for i in range(n):
    cur=myClass()
    z=raw_input().split()
    cur.start=int(z[0])
    cur.end=int(z[1])
    a.append(cur)
def compare(x,y):
    return x.start-y.start
a.sort(cmp=compare)
res=0
mx=-1
for i in range(0,n):    
    if(a[i].end<=mx):
        res=res+1
    mx=max(a[i].end,mx)
print res
