def main():
    n = int(raw_input())
    events = []
    for i in range(n):
        a, b = map(int, raw_input().split())
        events.append((a, b))
    events.sort()
    max_right = events[0][1]
    count = 0
    for a, b in events[1:]:
        if b < max_right:
            count += 1
        elif b > max_right:
            max_right = b
    print count
    
main()
