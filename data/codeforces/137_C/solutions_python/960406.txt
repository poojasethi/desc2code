n = int(raw_input())
data = []
for i in range(n):
    ab = raw_input().split()
    a = int(ab[0])
    b = int(ab[1])
    pair = []
    pair.append(a)
    pair.append(b)
    data.append(pair)

data.sort()
    
Curend = data[0][1]
count = 0
for i in range(1,n):
    if data[i][1]<Curend:
        count += 1
    else:
        Curend = data[i][1]

print count
