n = int(input())

intervals = []

for _ in range(n) :
	a, b = map(int, raw_input().split())
	intervals.append([a, b])

def getStart(x) :
	return x[0]

intervals.sort(key=getStart)

maxEnd = intervals[0][1]

cnt = 0
for i in range(1, n, 1) :
	if intervals[i][1] > maxEnd :
		maxEnd = intervals[i][1]
	else :
		cnt += 1
print cnt
