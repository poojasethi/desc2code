import collections
n = int(raw_input())
k = []
for i in range(n):
    a, b = map(int, raw_input().split())
    k.append((b, a))
k.sort()
que = collections.deque()
for (b, a) in k:
    if not len(que):
        que.append(a)
    else:
        while len(que) and que[-1] > a:
            que.pop()
        que.append(a)
print n - len(que)
