using namespace std;
vector<vector<vector<int> > >dp[123];
void ot(vector<vector<int> >&a)
{
    for(int i=0;i<a.size();i++)
    {
        if(i)printf(",");
        printf("{");
        for(int q=0;q<a[i].size();q++)
        {
            if(q)printf(",");
            printf("%d",a[i][q]);
        }
        printf("}");
    }
    puts("");
}
int main()
{
    vector<int> e1;
    vector<vector<int> >e2;
    e1.push_back(1);
    e2.push_back(e1);
    dp[1].push_back(e2);
    int i,q,z;
    for(i=1;i<10;i++)
    {
        int fk=1;
        for(q=0;q<dp[i].size();q++,fk^=1)
        {
            if(fk&1)
            {
                for(z=0;z<dp[i][q].size();z++)
                {
                    e2=dp[i][q];
                    e2[z].push_back(i+1);
                    dp[i+1].push_back(e2);
                }

                e2=dp[i][q];
                e1.clear ();
                e1.push_back(i+1);
                e2.push_back(e1);
                dp[i+1].push_back(e2);
            }
            else
            {
                e2=dp[i][q];
                e1.clear ();
                e1.push_back(i+1);
                e2.push_back(e1);
                dp[i+1].push_back(e2);
                for(z=dp[i][q].size();z--;)
                {
                    e2=dp[i][q];
                    e2[z].push_back(i+1);
                    dp[i+1].push_back(e2);
                }
            }
        }
    }
    int n;
    while(cin>>n)
    {
        cout<<dp[n].size()<<endl;
        for(q=0;q<dp[n].size();q++)
            ot(dp[n][q]);
    }

}
