#include<cstring>
#include<cstdio>
#include<vector>
using namespace std;
int cnt[11], b[11], n, tot = 0;
vector<int> vec[11];
void dfs(int i, int mx, int x)
{
	if(i == n + 1)
	{
		if(x == 1)
		{
			for(int i = 0; i <= mx; i++) 
			{
				vec[i].clear();
			}
			for(int i = 1; i <= n; i++)
			{
				vec[b[i]].push_back(i);
			}
			for(int i = 0; i <= mx; i++)
			{
				printf("{");
				for(size_t j = 0; j < vec[i].size(); j++)
				{
					printf("%d%c", vec[i][j], j == vec[i].size()-1?'}':',');
				}
				printf("%c", i == mx?'\n':',');
			}
		}else
		{
			tot++;
		}
	}else
	{
		cnt[i]++;
		if(cnt[i] % 2 == 1)
		{
			for(int j = 0; j <= mx; j++)
			{
				b[i] = j;
				dfs(i + 1, mx, x);
			}
			b[i] = mx + 1;
			dfs(i + 1, mx + 1, x);
		}else
		{
			b[i] = mx + 1;
			dfs(i + 1, mx + 1, x);
			for(int j = mx; j >= 0; j--)
			{
				b[i] = j;
				dfs(i + 1, mx, x);
			}
		}
	}
}
int main()
{
	scanf("%d", &n);
	memset(cnt, 0, sizeof(cnt));
	b[1] = 0;
	dfs(2, 0, 0);
	printf("%d\n", tot);
	b[1] = 0;
	memset(cnt, 0, sizeof(cnt));
	dfs(2, 0, 1);
	fclose(stdin);
	return 0;
}
