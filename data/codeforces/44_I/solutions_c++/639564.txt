using namespace std;
#define li		 long long int
#define rep(i,to)	for(li i=0;i<((li)to);i++)
#define pb		push_back
#define sz(v)		((li)v.size())

#define MAX 12
vector<string> ans[MAX];

void recur(li pos){
	if(pos==1){
		ans[pos].pb("0");
		return ;
	}
	recur(pos-1);
	li num=0;
	rep(i,sz(ans[pos-1])){
		li maxi=0;
		while(count(ans[pos-1][i].begin(),ans[pos-1][i].end(),'0'+maxi)) maxi++;
//if(pos==2) cout<<maxi<<" "<<ans[pos-1][i]<<" "<<i<<endl;
		if(num==0){
			ans[pos].pb(ans[pos-1][i]+"0");
			for(num=maxi;0<num;num--){
				ans[pos].pb(ans[pos-1][i]+string(1,'0'+num));
			}
			num++;
		}else{
			for(num=1;num<=maxi;num++){
				ans[pos].pb(ans[pos-1][i]+string(1,'0'+num));
			}
			num=0;
			ans[pos].pb(ans[pos-1][i]+"0");
		}
	}
}

int main(){
	li n;
	cin>>n;
	recur(10);
	cout<<sz(ans[n])<<endl;
	rep(i,sz(ans[n])){
		rep(j,100){
			vector<li> vec;
			rep(k,sz(ans[n][i]))if(ans[n][i][k]=='0'+j) vec.pb(k);
			if(sz(vec)==0) break;
			cout<<(j?",{":"{");
			rep(k,sz(vec)) cout<<(k?",":"")<<vec[k]+1;
			cout<<"}";
		}
		cout<<endl;
	}
}
