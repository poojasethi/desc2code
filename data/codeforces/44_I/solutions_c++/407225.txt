#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <cassert>
#include <vector>
using namespace std;
typedef vector<int> VI;
typedef vector<VI> VVI;
typedef vector<VVI> ST;
VVI add(VVI v2, int loc, int v){
    //printf("in");
    if(loc == v2.size()) v2.push_back(VI());
    //if(loc >= v2.size() || loc < 0){
        //while(1);
    //}
    v2[loc].push_back(v);
    //printf("out");
    return v2;
}
ST a[12];
void getAns(int n){
    ST ret;
    if(n == 1){
        ret.push_back(VVI(1, VI(1, 1)));
        a[n] = ret; return;
        //return ret;
    }
    getAns(n-1);
    ST& st = a[n-1];
    //ST st = getAns(n-1);
    if(st[0].size() != 1) while(1);
    int cur = 0;
    for(int i = 0; i < (int)st.size(); i ++){
        int nc = -1;
        cur = min(cur, (int)st[i].size());
        ret.push_back(add(st[i], cur, n));
        for(int j = 0; j <= (int)st[i].size(); j ++){
            if(cur == j) continue;
            nc = j;
            ret.push_back(add(st[i], j, n));
        }
        cur = nc;
    }
    //printf("%d %d\n", n, ret.size());
    a[n] = ret; return;
    //return ret;
}
void out(VI vec){
    printf("{");
    for(int i = 0; i < (int)vec.size(); i ++){
        if(i != 0) printf(",");
        printf("%d", vec[i]);
    }
    printf("}");
}
void out(VVI v2){
    for(int i = 0; i < (int)v2.size(); i ++){
        if(i != 0) printf(",");
        out(v2[i]);
    }
}
int main() {
    int n;
    scanf("%d", &n);
    getAns(n);
    ST ans = a[n];
    printf("%d\n", ans.size());
    for(int i = 0; i < (int)ans.size(); i ++){
        out(ans[i]);
        printf("\n");
    }
    return 0;
}
