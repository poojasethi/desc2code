
#include <vector>

using namespace std;

vector<string> Ans[11];
int mark[15];
int main(int argc, char** argv) {
    int i,j,k,N,now;
    int ii;
    Ans[1].push_back("0");
    Ans[2].push_back("00");
    Ans[2].push_back("01"); 
    for (now = 3; now <= 10; now++) {
        Ans[now].clear();
        for (i = 0; i < Ans[now - 1].size(); i++) {
            string tmp = Ans[now - 1][i];
            char Max = 0;
            for (j = 0; j < tmp.size(); j++)
                Max = max(tmp[j], Max);
            Max++;
            if (i%2==0) {
                for (char j = '0'; j <= Max; j++) {
                    string s = tmp + j;
                    Ans[now].push_back(s);
                }
            }else {
            	for (char j = Max; j >='0'; j--) {
                    string s = tmp + j;
                    Ans[now].push_back(s);
                }
            }
        }
    }
    while (scanf("%d", &N) != EOF) {
        printf("%d\n", Ans[N].size());
        for (i = 0; i < Ans[N].size(); i++) {
            string tmp = Ans[N][i];
            char Max = 0;
            for (j = 0; j < tmp.size(); j++)
                Max = max(tmp[j], Max);
            for (char j = '0'; j <= Max; j++) {
                printf("{");
                bool first = true;
                for (k = 0; k < tmp.size(); k++)
                    if (tmp[k] == j) {
                        if (first) {
                            first = false;
                            printf("%d", k + 1);

                        } else printf(",%d", k + 1);
                    }
                printf("}");
                if (j != Max) printf(",");
            }
            printf("\n");
        }
    }

}
