#include<string>
#include<iostream>
#include<cstdio>
using namespace std;
int n,gc,fc;
string g[120000],f[120000];
string st,ch;
int main(){
  scanf("%d",&n);
  g[gc=1]="{1}";
  for (int test=2;test<=n;test++){
    fc=0;
    if (test<10) ch=test+48; else ch="10";
	for (int i=1,flag=1;i<=gc;i++,flag^=1){
	  st="";
	  if (flag){
	    for (int j=0;j<g[i].length();j++){
	      if (g[i][j]=='}'){
	        f[++fc]=st+','+ch;
			for (int k=j;k<g[i].length();k++) f[fc]+=g[i][k];  
	      }
	      st+=g[i][j];
	    }  
	    f[++fc]=st+','+'{'+ch+'}';
	  }else{
	     int cnt=0,c;
		 for (int j=0;j<g[i].length();j++)
		   if (g[i][j]=='}') ++cnt;
		 f[++fc]=g[i]+','+'{'+ch+'}';
		 c=cnt;
		 for (int j=0;j<g[i].length();j++){
	      if (g[i][j]=='}'){
	        f[fc+c]=st+','+ch;
			for (int k=j;k<g[i].length();k++) f[fc+c]+=g[i][k];
			--c;  
	      }
	      st+=g[i][j];
	     }
		 fc+=cnt;   
	  }  
	}
	gc=fc;
	for (int i=1;i<=fc;i++) g[i]=f[i];   
  }
  printf("%d\n",gc);
  for (int i=1;i<=gc;i++) cout<<g[i]<<endl;
  return 0;
}
