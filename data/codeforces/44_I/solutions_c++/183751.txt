#include <iostream>
#include <vector>
#include <cstdio>
using namespace std;
struct m
{
	int no;
	vector <int> n[10];
};
vector <m> find(int a)
{
	vector <m> sol;
	if (a==1)
	{
		m temp;
		temp.no=1;
		temp.n[0].push_back(1);
		sol.push_back(temp);
		return sol;
	}
	vector <m> temp=find(a-1);
	for (int i=0;i<temp.size();i++)
	{
		if (i%2==0)
		{
			m t;
			for (int j=0;j<temp[i].no;j++)
			{
				t=temp[i];
				t.n[j].push_back(a);
				sol.push_back(t);
			}
			t=temp[i];
			t.no++;
			t.n[t.no-1].push_back(a);
			sol.push_back(t);
		}
		else
		{
			m t;
			t=temp[i];
			t.no++;
			t.n[t.no-1].push_back(a);
			sol.push_back(t);
			for (int j=temp[i].no-1;j>=0;j--)
			{
				t=temp[i];
				t.n[j].push_back(a);
				sol.push_back(t);
			}
		}
	}
	return sol;
}
int main()
{
	int n;
	scanf("%d",&n);
	vector <m> sol=find(n);
	printf("%d\n",sol.size());
	for (int i=0;i<sol.size();i++)
	{
		m temp=sol[i];
		for (int j=0;j<temp.no;j++)
		{
			if (j)
				printf(",");
			printf("{");
			for (int k=0;k<temp.n[j].size();k++)
			{
				if (k)
					printf(",");
				printf("%d",temp.n[j][k]);
			}
			printf("}");
		}
		printf("\n");
	}
	return 0;
}
