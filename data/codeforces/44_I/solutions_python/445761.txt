#!/usr/bin/python
import sys

def gen (n):
	if n == 1:
		return [[0]]
	else:
		prev = gen (n - 1)
	res = []
	parity = 0
	for elem in prev:
		limit = max (elem) + 1
		if parity == 0:
			for i in range (limit + 1):
				res.append (elem + [i])
		else:
			for i in range (limit + 1) [::-1]:
				res.append (elem + [i])
		parity = parity ^ 1
	return res

n = int (sys.stdin.readline ())
res = gen (n)
print len (res)
for elem in res:
	s = ''
	limit = max (elem) + 1
	for group in range (limit):
		s += '{'
		t = ''
		for i in range (n):
			if elem[i] == group:
				t += '%d,' % (i + 1)
		s += t[:-1]
		s += '},'
	print s[:-1]
