#include <iostream>
#include <cstdio>
#include <vector>
#include <map>
#include <set>
#include <algorithm>
#include <queue>
#include <string>
#include <string.h>
#include <cmath>
using namespace std;

#define MAXN 800000
#define MAXM 20000
#define INF 1000000000
#define tl(x) (x << 1)
#define tr(x) (x << 1 | 1)

int n;
int bx[MAXN];

multiset <int> col;
map <int, int> to;
vector <int> bc;

vector <int> ans;

int main()
{

    scanf("%d", &n);
    for(int i = 1; i <= n; i++) scanf("%d", &bx[i]);

    for(int i = 1; i <= n; i++) {
        int x = bx[i];
        if(to.find(x) != to.end()) {
            ans.push_back(to[x]);
            ans.push_back(x);
            ans.push_back(to[x]);
            ans.push_back(x);
            to.clear(); col.clear(); bc.clear();
        } else {
            if(col.count(x) != 0) {
                int cnt = col.count(x) == 1 ? 0 : 1;
                while(cnt || bc[bc.size() - 1] != x) {
                    if(bc[bc.size() - 1] == x) cnt--;
                    to[bc[bc.size() - 1]] = x;
                    bc.pop_back();
                }
            }
            col.insert(x);
            bc.push_back(x);
        }
    }

    printf("%d\n", ans.size());
    for(int i = 0; i < ans.size(); i++) {
        if(i != ans.size() - 1) printf("%d ", ans[i]);
        else printf("%d\n", ans[i]);
    }

    return 0; 
}
