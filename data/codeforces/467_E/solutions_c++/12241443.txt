#include <bits/stdc++.h>
#define rep(i,a,b) for(int i=a;i<=b;i++)
#define pb push_back
#define pob pop_back

using namespace std;

const int maxn=500005;
int n,i;
int a[maxn];
vector <int> res;

bool Find()
{
    map<int,int> pre,cnt;
    vector<int> f;
    for(;i<=n;i++)
    {
        if (pre[a[i]])
        {
            res.pb(pre[a[i]]); res.pb(a[i]);
            res.pb(pre[a[i]]); res.pb(a[i]);
            i++;
            return 1;
        }
        while ( (cnt[a[i]]>1 || (cnt[a[i]]==1&&f.back()!=a[i])) && f.size() )
        {
            pre[f.back()]=a[i];
            cnt[f.back()]--;
            f.pob();
        }
        cnt[a[i]]++;
        f.pb(a[i]);
    }
    return 0;
}

int main()
{
    ios::sync_with_stdio(0);
    //freopen("in.tt","r",stdin);
    cin >> n;
    rep(i,1,n) cin >> a[i];
    i=1;
    while (Find());
    cout << res.size() << '\n';
    int x=res.size();
    rep(i,0,x-1) cout << res[i] << " ";
}
