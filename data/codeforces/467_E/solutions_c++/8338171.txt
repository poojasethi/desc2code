#define MAXN 2000050
using namespace std;
int a[MAXN],b[MAXN];
map<int,int>mp,cnt;
int st[MAXN];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
		scanf("%d",&a[i]);
	int top=0,ans=0;
	for(int i=1;i<=n;++i)
	{
		if(mp[a[i]]!=0){

			b[ans++]=mp[a[i]];
			b[ans++]=a[i];
			mp.clear();
			cnt.clear();
			top=0;
			continue;
		}
		while(top>0&&((cnt[a[i]]>0&&st[top]!=a[i])||cnt[a[i]]>1)){
			cnt[st[top]]--;
			mp[st[top]]=a[i];
			top--;
		}
		st[++top]=a[i];
		cnt[a[i]]++;
	}
	printf("%d\n",ans*2);
	for(int i=0;i<ans;i+=2)
	{
		printf("%d %d %d %d",b[i],b[i+1],b[i],b[i+1]);
		if(i==ans-1)puts("");
		else
			printf(" ");
	}
}
