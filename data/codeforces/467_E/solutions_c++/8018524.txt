#include <iostream>
#include <cstdlib>
#include <algorithm>
#include <utility>
#include <vector>
#include <map>
#define MAXN 500010
#define SQRT 710

using namespace std;

int n,a[MAXN];
map<int,int> mp,ct;
vector<pair<int,int> > ans;
int b[MAXN],c[SQRT];

void add(int x, int y) {
    ans.push_back(make_pair(x,y));
    mp.clear();
    ct.clear();
}

int main() {
    ios::sync_with_stdio(0);
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        int j = mp[a[i]];
        ct[a[i]]++;
        if (j) {
            for (int k = j+1; k < i && k % SQRT != 1; k++) b[k] = a[i];
            for (int k = 1; SQRT*k < i; k++)
                if (SQRT*(k-1) >= j) c[k] = a[i];
            for (int k = i-1; k > j && k % SQRT != 0; k--) b[k] = a[i];
            if (b[j]) add(b[j],a[i]);
            else if (c[(j-1)/SQRT+1]) add(c[(j-1)/SQRT+1],a[i]);
            else if (ct[a[i]] >= 4) add(a[i],a[i]);
            else mp[a[i]] = i;
        }
        else mp[a[i]] = i;
    }
    cout << 4*ans.size() << '\n';
    for (int i = 0; i < ans.size(); i++) {
        cout << ans[i].first << ' ' << ans[i].second << ' ';
        cout << ans[i].first << ' ' << ans[i].second << ' ';
    }
    cout << '\n';
}
