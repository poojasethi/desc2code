#include <iostream>
#include <vector>
#include <stack>

using namespace std;

const int MAXN = 1e5 + 100;
int n, m;
int a[MAXN], b[MAXN];
vector<int> edges[MAXN], ans;
stack<int> s;

int main()
{
	cin >> n >> m;
	for (int i = 0; i < m; i++)
	{
		int u, v;
		cin >> u >> v;
		u--; v--;
		edges[u].push_back(v);
		edges[v].push_back(u);
	}
	for (int i = 0; i < n; i++)
	{
		cin >> a[i];
		if (a[i] == 0)
			s.push(i);
	}
	while (!s.empty())
	{
		int v = s.top();
		s.pop();
		ans.push_back(v);
		b[v]++;
		for (int i = 0; i < edges[v].size(); i++)
		{
			int u = edges[v][i];
			b[u]++;
			if (a[u] == b[u])
				s.push(u);
		}
	}
	cout << ans.size() << endl;
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i] + 1 << " ";
	cout << endl;
	return 0;
}
