#include<bits/stdc++.h>
using namespace std;
const int N=1e5+5;

vector<int> adj[N];
int a[N];
int adad[N];
vector<int> ans;
vector<int> bfs;

int main()
{
	int n,m,u,v;
	cin >> n >> m;
	for(int i=1;i<=m;i++)
	{
		cin >> u >> v;
		adj[u].push_back(v);
		adj[v].push_back(u);
	}
	for(int i=1;i<=n;i++)
	{
		cin >> a[i];
		if(a[i]==adad[i])
			bfs.push_back(i);
	}
	while(!bfs.empty())
	{
		int b=bfs.back();
		bfs.pop_back();
		if(adad[b]==a[b])
		{
			ans.push_back(b);
			adad[b]++;
			for(int i=0;i<adj[b].size();i++)
			{
				int p=adj[b][i];
				adad[p]++;
				if(adad[p]==a[p])
					bfs.push_back(p);
			}
		}
	}
	sort(ans.begin(),ans.end());
	
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++)
		cout<<ans[i]<<" ";

	return 0;
}
