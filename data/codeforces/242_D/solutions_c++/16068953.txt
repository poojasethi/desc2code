#include <bits/stdc++.h>
using namespace std;

int vals[100010];
int x[100010] = {0};
vector<int> graph[100010];

int main(void) {
	int n, m;
	scanf("%d %d", &n, &m);
	for (int i = 0; i < m; i++) {
		int u, v;
		scanf("%d %d", &u, &v);
		graph[u].push_back(v);
		graph[v].push_back(u);
	}
	queue<int> q;
	for (int i = 1; i <= n; i++) {
		scanf("%d", &vals[i]);
		if (vals[i] == x[i])
			q.push(i);
	}
	vector<int> ans;
	while (!q.empty()) {
		int v = q.front();
		q.pop();
		if (vals[v] == x[v]) {
			ans.push_back(v);
			x[v]++;
			for (int i = 0; i < graph[v].size(); i++) {
				x[graph[v][i]]++;
				if (x[graph[v][i]] == vals[graph[v][i]])
					q.push(graph[v][i]);
			}
		}
	}
	int s = ans.size();
	printf("%d\n", s);
	for (int i = 0; i < s; i++) 
		printf("%d ", ans[i]);
	return 0;
}
