#include <algorithm>
#include <stdio.h>
#include <vector>
#define  maxn      100003
#define  pb        push_back
using    namespace std;

int n,m;
int ar[maxn];
vector<int>res;
vector<int>adj[maxn];

int main()
{
  scanf("%d%d",&n,&m);
  for(int i=0,x,y ; i<m ; i++)
  {
    scanf("%d%d",&x,&y);
    adj[x].pb(y);
    adj[y].pb(x);
  }
  for(int i=1 ; i<=n ; i++)
  {
    scanf("%d",&ar[i]);
    if(!ar[i])
      res.pb(i);
  }
  for(int i=0 ; i<res.size() ; i++)
  {
    int node=res[i];
    ar[node]--;
    for(int j=0 ; j<adj[node].size() ; j++)
    {
      ar[adj[node][j]]--;
      if(!ar[adj[node][j]])
        res.pb(adj[node][j]);
    }
  }
  int sz=res.size();
  printf("%d\n",sz);
  for(int i=0 ; i<sz ; i++)
    printf("%d ",res[i]);
  puts("");
  return 0;
}
