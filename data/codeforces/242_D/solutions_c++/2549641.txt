#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;

#define pb push_back
#define MAXN 100010

int n, m;
vector <int> g[MAXN];
int a[MAXN], b[MAXN], q[10*MAXN], l, r;
vector <int> ans;

void push(int x)
{
	if (b[x] != a[x]) return;
	b[x]++;
	ans.pb(x);
	for (int i = 0, v; i < (int) g[x].size(); ++i) {
		v = g[x][i];
		b[v]++;
		if (b[v] == a[v])
			q[r++] = v;
	}
}

int main()
{
	cin >> n >> m;
	for (int i = 0, x, y; i < m; ++i) {
		scanf("%d%d", &x, &y);
		x--, y--;
		g[x].pb(y);
		g[y].pb(x);
	}
	for (int i = 0; i < n; ++i)
		scanf("%d", &a[i]);

	for (int i = 0; i < n; ++i) 
		if (b[i] == a[i]) 
			q[r++] = i;

	while (l < r) {
		push(q[l++]);
	}
	cout << ans.size() << endl;
	for (int i = 0; i < (int) ans.size(); ++i)
		cout << ans[i]+1 << " ";	

	return 0;
}
