#include <cstdio>
#include <iostream>
#include <vector>

using namespace std;

const int MAXN = 100100 ;

int N,M;
int ar[MAXN];
int cnt[MAXN];

bool used[MAXN];

vector <int> res;
vector <int> ways[MAXN];

void f(int node){
	
	used[node]=true;

	res.push_back(node);

	cnt[node]++;
	
	for(int i=0;i<ways[node].size();i++) cnt[ways[node][i]]++;
	
	for(int i=0;i<ways[node].size();i++) 
		if(cnt[ways[node][i]]==ar[ways[node][i]] && !used[ways[node][i]])
			f(ways[node][i]);
}

int main(){
	
	ios_base::sync_with_stdio(false);
	
	cin >> N >> M ;
	
	while(M--){
		
		int a,b; cin >> a >> b ;
		ways[a].push_back(b);
		ways[b].push_back(a);
	}
	
	for(int i=1;i<=N;i++) cin >> ar[i] ;

	for(int i=1;i<=N;i++)
		if(cnt[i]==ar[i])
			f(i);
			
	cout << res.size() << endl ;
	
	for(int i=0;i<res.size();i++) cout << res[i] << " " ;
		
	return 0;
}
