#include<iostream>
#include<vector>
#include<algorithm>
#define Nmax 100010
using namespace std;

int n, m, i, y, nr, a[Nmax], c[Nmax], p, x, j, el, b[Nmax];
vector <int> muchie[Nmax], sol;

int main()
{
	cin >> n >> m;
	for (i = 1; i <= m; ++i)
	{
		cin >> x >> y;
		muchie[x].push_back(y);
		muchie[y].push_back(x);
	}
	nr = 0;
	for (i = 1; i <= n; ++i)
	{
		cin >> a[i];
		if (a[i] == 0)
		{
			nr++;
			c[nr] = i;
		}
	}
	j = 1;
	while (j <= nr)
	{
		x = c[j];
		j++;
		sol.push_back(x);
		b[x]++;
		for (i = 0; i < muchie[x].size(); ++i)
		{
			el = muchie[x][i];
			b[el]++;
			if (b[el] == a[el])
			{
				nr++;
				c[nr] = el;
			}
		}
	}
	sort (sol.begin(), sol.end());
	cout << sol.size() << '\n';
	for (i = 0; i <sol.size(); ++i)
	{
		cout << sol[i] << " ";
	}
	return 0;
}
