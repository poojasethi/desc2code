#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<vector>
#include<queue>
using namespace std;
#define maxn 100005
vector<int> edge[maxn];
vector<int> ans;
queue<int> q;
int a[maxn];
int main()
{
	int i, j, x, y, n, m;
	scanf("%d%d", &n, &m);
	while(m--)
	{
		scanf("%d%d", &x, &y);
		edge[x].push_back(y);
		edge[y].push_back(x);
	}
	for(i = 1; i <= n; i++)
	{
		scanf("%d", &a[i]);
		if(!a[i]) q.push(i);
	}
	while(!q.empty())
	{
		int u = q.front(); q.pop();
		if(a[u]) continue; 
		a[u]--;
		ans.push_back(u);
		for(i = 0; i < edge[u].size(); i++)
		{
			int v = edge[u][i];
			a[v]--;
			if(!a[v]) q.push(v);
		}
	}
	printf("%d\n", ans.size());
	for(i = 0; i < ans.size(); i++)
	{
		printf("%d", ans[i]);
		if(i == ans.size()-1) printf("\n");
		else printf(" ");
	}
	return 0;
}
