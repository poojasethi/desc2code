#include<bits/stdc++.h>

using namespace std;

#define mp make_pair
#define pb push_back 
#define nd second
#define st first

typedef long long ll;
typedef long long int lli;
typedef long long unsigned llu;
typedef pair < int,int > pii;
typedef pair < lli,lli > plli;

const int MAX = 1e5+7;

int n,m;
int a[MAX];
int b[MAX];
vector < int > v[MAX];
vector < int > ans;

int main()
{
	cin>>n>>m;
	for(int i=1; i<=m; i++)
	{
		int a,b;
		scanf("%d %d",&a,&b);
		v[a].pb(b);
		v[b].pb(a);
	}
	for(int i=1; i<=n; i++)
		scanf("%d",a+i);
	int f=1;
	while(f)
	{
		f=0;
		for(int i=1; i<=n; i++)
			if(b[i] == a[i])
			{
				ans.pb(i);
				b[i]++;
				for(int j=0; j<v[i].size(); j++)
					b[v[i][j]]++;
				f=1;
			}
	}
	cout<<ans.size()<<endl;
	for(int i=0; i<ans.size(); i++)
		cout<<ans[i]<<" "; 
}
