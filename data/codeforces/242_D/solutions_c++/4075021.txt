#include<algorithm>
#include<memory.h>
#include<iostream>
#include<iomanip>
#include<string>
#include<vector>
#include<queue>
#define CN 100001
using namespace std;

vector <int> adj[CN];
int wnt[CN], iss[CN], use[CN];
 
int main(){
	int n, m, a, b;
	cin >> n >> m;
	for (int i = 0;i < m;++i){
		cin >> a >> b;
		--a, --b;
		adj[a].push_back(b);
		adj[b].push_back(a);
	}
	
	int ans = 0;
	queue <int> q;
	for (int i = 0;i < n;++i){
		cin >> wnt[i];
		if (wnt[i] == 0)
			q.push(i);
	}
	while (!q.empty()){
		int t = q.front();
		q.pop();
		++ans;
		iss[t]++;
		use[t] = true;
		for (int i = 0;i < adj[t].size();++i){
			int v = adj[t][i];
			iss[v]++;
			if (iss[v] == wnt[v])
				q.push(v);
		}
	}
	
	cout << ans << endl;
	for (int i = 0;i < n;++i)
		if (use[i])
			cout << i + 1 << " ";
	cout << endl;
	
	return 0;
}
