//In the Name of God
#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
int n,m,V,U,a[100100],b[100100];
vector<int> g[100100],v,ans;
main(){
  scanf("%d%d",&n,&m);
  for(int i=0;i<m;i++){
    scanf("%d%d",&V,&U);
    g[V].push_back(U);
    g[U].push_back(V);
  }
  for(int i=1;i<=n;i++){
    scanf("%d",a+i);
    if(a[i]==0)
      v.push_back(i);
  }
  for(int i=0;i<v.size();i++)
    if(a[v[i]]==b[v[i]]){
      ans.push_back(v[i]);
      for(int j=0;j<g[v[i]].size();j++){
	b[g[v[i]][j]]++;
	if(b[g[v[i]][j]]==a[g[v[i]][j]])
	  v.push_back(g[v[i]][j]);
      }
    }
  printf("%d\n",ans.size());
  for(int i=0;i<ans.size();printf("%d ",ans[i]),i++);
  printf("\n");
}
