#include<stdio.h>
#include<algorithm>
#include<vector>
#include<queue>
#define N 100010
int m,n,x,y;
std::vector<int>v[N];
std::queue<int>q;
std::vector<int>ans;
int a[N],b[N];
int main()
{
	int i,j,k;
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;i++)
	{
		scanf("%d%d",&x,&y);
		v[x].push_back(y);
		v[y].push_back(x);
	}
	std::fill(b,b+N,0);
	for(i=1;i<=n;i++) scanf("%d",a+i);
	for(i=1;i<=n;i++)
		if(a[i]==b[i]) {q.push(i);ans.push_back(i);}
	while(!q.empty())
	{
		x=q.front();b[x]++;q.pop();
		for(i=0;i<v[x].size();i++)
		{
			y=v[x][i];
			b[y]++;
			if(b[y]==a[y]) {q.push(y);ans.push_back(y);}
		}
	}
	printf("%d\n",ans.size());
	for(i=0;i<ans.size();i++) printf("%d%c",ans[i],i==ans.size()-1?'\n':' ');
	getchar();getchar();
}
