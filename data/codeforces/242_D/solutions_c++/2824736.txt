#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
const int N=100100,M=200010;
int head[N],v[M],next[M],t=1;
int a[N];
int n,m;
vector<int>ans;
void add(int a,int b)
{
    next[++t]=head[a];v[t]=b;head[a]=t;
    next[++t]=head[b];v[t]=a;head[b]=t;
}
void push(int x)
{
    a[x]--;int i;ans.push_back(x);
    for(i=head[x];i;i=next[i])a[v[i]]--;
}
void dfs(int x)
{
    push(x);
    int i;
    for(i=head[x];i;i=next[i])if(!a[v[i]])dfs(v[i]);
}
int main()
{
    int i,x,y;
    scanf("%d%d",&n,&m);
    while(m--)scanf("%d%d",&x,&y),add(x,y);
    for(i=1;i<=n;i++)
    scanf("%d",&a[i]);
    for(i=1;i<=n;i++)
    if(!a[i])dfs(i);
    printf("%d\n",ans.size());
    for(i=0;i<ans.size();i++)
    printf( "%d ",ans[i]);    
}
