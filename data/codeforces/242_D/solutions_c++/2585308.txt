#include<iostream>
#include<queue>
#include<vector>
using namespace std;

#define pb push_back

vector<int> out,e[1000*100+1];
queue<int> q;
int n,m,a[1000*100+1],b[1000*100+1];
void input(){
	cin>>n>>m;
	for(int i=1;i<=m;i++){
		int v,u;
		cin>>v>>u;
		e[v].pb(u);
		e[u].pb(v);
	}
	for(int i=1;i<=n;i++)
		cin>>a[i];
}
void proses(int x){
	out.pb(x);
	b[x]++;
	for(int j=0;j<e[x].size();j++){
		int now=e[x][j];
		b[now]++;
		if(a[now]==b[now])q.push(now);
	}
}
void output(){
	cout<<out.size()<<endl;
	for(int i=0;i<out.size();i++)
		cout<<out[i]<<" ";
}
int main(){
	input();
	for(int i=1;i<=n;i++){
		if(a[i]==b[i])
		proses(i);
		while(!q.empty()){
				int now=q.front();
				if(a[now]==b[now])
				proses(now);
				q.pop();
			}
		}
	output();
}
