#include<iostream>
#include<vector>
#include<queue>
using namespace std;
vector<int> adj[100100];
queue<int> q;
int a[100100], ans[100100];
int main(){
    int n,m,i,j,x,y,tmp=0;
    cin>>n>>m;
    for(i=0;i<m;i++){
        cin>>x>>y;
        adj[x].push_back(y),adj[y].push_back(x);
    }
    for(i=1;i<=n;i++){
       cin>>a[i];
       if(a[i]==0) {q.push(i);a[i]--;} 
    }
    while(!q.empty()){
       j=q.front();
       for(i=0;i<adj[j].size();i++){
         a[adj[j][i]]--;
         if(a[adj[j][i]]==0) q.push(adj[j][i]); 
       }
       ans[tmp++]=j;
       q.pop();
    }
    cout<<tmp<<endl;
    for(i=0;i<tmp;i++) cout<<ans[i]<<" ";
    cout<<endl;
   // system("pause");
    return 0;
}
