//In the Name of God

#include<iostream>
#include<vector>

using namespace std;

int n,m,V,U,a[100100],b[100100];
vector<int> g[100100],v,ans;

int main()
{
  cin>>n>>m;
  for(int i=0;i<m;i++)
    {
      cin>>V>>U;
      g[V].push_back(U);
      g[U].push_back(V);
    }
  for(int i=1;i<=n;i++)
    cin>>a[i];
  for(int i=1;i<=n;i++)
    if(b[i]==a[i])
      v.push_back(i);
  for(int i=0;i<v.size();i++)
    if(a[v[i]]==b[v[i]])
      {
	ans.push_back(v[i]);
	b[v[i]]++;
	for(int j=0;j<g[v[i]].size();j++)
	  {
	    b[g[v[i]][j]]++;
	    if(b[g[v[i]][j]]==a[g[v[i]][j]])
	      v.push_back(g[v[i]][j]);
	  }
      }
  cout<<ans.size()<<endl;
  for(int i=0;i<ans.size();i++)
    cout<<ans[i]<<" ";
  cout<<endl;
  return 0;
}
