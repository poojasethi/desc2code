#include <iostream>
#include <queue>
#include <vector>

using namespace std;

const int MAXN = 1e5 + 10;
int a[MAXN], c[MAXN];
vector<int> ng[MAXN], ans;
queue<int> q;

int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < m; i++) {
		int x, y;
		cin >> x >> y;
		x--; y--;
		ng[x].push_back(y);
		ng[y].push_back(x);
	}
	for (int i = 0; i < n; i++)
		cin >> a[i];
	for (int i = 0; i < n; i++)
		if (a[i] == 0)
			q.push(i);
	while (!q.empty()) {
		int cur = q.front();
		ans.push_back(cur);
		q.pop();
		c[cur]++;
		for (int i = 0; i < ng[cur].size(); i++) {
			int u = ng[cur][i];
			c[u]++;
			if (c[u] == a[u])
				q.push(u);
		}
	}
	cout << ans.size() << endl;
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i] + 1 << " ";
	cout << endl;
	return 0;
}
