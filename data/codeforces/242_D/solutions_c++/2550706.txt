#include<algorithm>
#include<stdio.h>
#include<vector>
#include<queue>
#define X 100010
using namespace std;
vector<int>u[X],as;
queue<int>q;
int a[X];
int main(){
	int i,j,n,m,x,y;
	scanf("%d%d",&n,&m);
	for(i=0;i<m;i++){
	    scanf("%d%d",&x,&y);
	    u[x].push_back(y);
	    u[y].push_back(x);
	}
	for(i=1;i<=n;i++)
	    scanf("%d",&a[i]);
    while(!q.empty())q.pop();
	as.clear();
	for(i=1;i<=n;i++)
	    if(!a[i])q.push(i);
	while(!q.empty()){
		x=q.front();q.pop();
		if(a[x])continue;a[x]--;
		as.push_back(x);
		for(i=0;i<u[x].size();i++){
		    y=u[x][i];a[y]--;
		    if(!a[y])q.push(y);
		}
	}
	sort(as.begin(),as.end());
	printf("%d\n",as.size());
	for(i=0;i<as.size();i++)
	    printf("%d ",as[i]);
	return 0;
}
