#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <vector>
#include <queue>
using namespace std;

#define FOR(it, c) for(__typeof((c).begin()) it = (c).begin(); it != (c).end(); it++)
#define SZ(c) ((int)(c).size())

const int N = 100005;
int r[N], v[N];
vector<int> a[N], ret;
int main(void) {
	int x, y, n, m;
	scanf("%d%d", &n, &m);
	for(int i=0;i<m;i++) {
		scanf("%d%d", &x, &y);
		a[x].push_back(y);
		a[y].push_back(x);
	}
	queue<int> Q;
	for(int i=1;i<=n;i++) {
		scanf("%d", &r[i]);
		if(r[i]==0) Q.push(i);
	}
	while(!Q.empty()) {
		x = Q.front(); Q.pop();
		ret.push_back(x); v[x]++;
		FOR(it, a[x]) if(++v[*it] == r[*it]) Q.push(*it);
	}
	printf("%d\n", SZ(ret));
	FOR(it, ret) printf("%d ", *it);
	puts("");
	return 0;
}
