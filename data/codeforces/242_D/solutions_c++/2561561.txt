#include <iostream>
#include <vector>
#include <deque>
using namespace std;

const int MAXN=1e5+100;

vector<int> adj[MAXN];
int val[MAXN],deg[MAXN];
deque<int> fix;
vector<int> push;

int main()
{
  int n,m;
  cin>>n>>m;
  while(m--)
    {
      int a,b;
      cin>>a>>b;
      adj[a].push_back(b);
      adj[b].push_back(a);
    }
  for(int i=1;i<=n;++i)
    cin>>val[i];
  for(int i=1;i<=n;++i)
    if(val[i]==deg[i])
      fix.push_back(i);
  while(!fix.empty())
    {
      int c=fix[0];
      fix.pop_front();
      push.push_back(c);
      deg[c]++;
      for(int i=0;i<adj[c].size();++i)
	{
	  int next=adj[c][i];
	  deg[next]++;
	  if(deg[next]==val[next])
	    fix.push_back(next);
	}
    }
  cout<<push.size()<<endl;
  for(int i=0;i<push.size();++i)
    cout<<push[i]<<" ";
  cout<<endl;
  return 0;
}
