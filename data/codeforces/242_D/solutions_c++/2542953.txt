#include <iostream>
#include <fstream>
#include <queue>
#include <vector>

using namespace std;

int n,m,i,x,y,a[100999],b[100999]={0};
vector < int > g[100999] , ans ;
queue < int > q;

int main() {  
//	freopen("in","r",stdin);
//	freopen("out","w",stdout);   
	cin >> n >> m ;
	
	for(i=1;i<=m;i++){
		cin >> x >> y ;
		g[x].push_back(y);
		g[y].push_back(x);	
	}
	
	for(i=1;i<=n;i++)
		cin >> a[i] ;

	for(i=1;i<=n;i++){
		if(a[i]==b[i]){q.push(i);}
	}
	while(!q.empty()){
		int num=q.front();q.pop();
		if(a[num]!=b[num])continue;
		ans.push_back(num);
		for(i=0;i<g[num].size();i++){
			int to = g[num][i] ;
			b[to]++;
			if(b[to]==a[to]){q.push(to);}
		}
	}       
	cout<<ans.size()<<endl;
	for(i=0;i<ans.size();i++){
		cout<<ans[i]<<" ";
	}
	return 0;
}
