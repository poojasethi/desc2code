#include<bits/stdc++.h>
using namespace std;
#define ll  int
#define pb push_back
int b[100001]={0};
int a[100001]={0};
vector<int>v[100001];
vector<int>ans;
void dfs(int ind){
b[ind]++;
ans.pb(ind);
for(int i=0;i<(int)v[ind].size();i++){
  b[v[ind][i]]++;
}
for(int i=0;i<(int)v[ind].size();i++){
if(b[v[ind][i]]==a[v[ind][i]]) dfs(v[ind][i]);
}
}
int main(){
ll n,m,l,r;
cin>>n>>m;
for(int i=0;i<m;i++){
cin>>l>>r;
v[l].pb(r);
v[r].pb(l);
}
for(int i=1;i<=n;i++) cin>>a[i];
for(int i=1;i<=n;i++){
if(b[i]==a[i]) dfs(i);
}
cout<<ans.size()<<"\n";
for(int i=0;i<(int)ans.size();i++){
cout<<ans[i]<<" ";
}
}
