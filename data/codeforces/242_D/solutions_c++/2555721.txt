#include <cstdio>
#include <vector>

using namespace std;

vector<int> adj[100000];
int ai[100000];

int n, m;
int cnt[100000];

int main() {
	scanf("%d %d", &n, &m);
	for (int i = 0; i < m; ++i) {
		int a, b;
		scanf("%d %d", &a, &b);
		a--; b--;
		adj[a].push_back(b);
		adj[b].push_back(a);
	}
	
	for (int i = 0; i < n; ++i) scanf("%d", &ai[i]);

	vector<int> res;
	for (;;) {
		int flag = 0;
		for (int i = 0; i < n; ++i) {
			if (cnt[i] != ai[i]) continue;
			cnt[i]++;
			res.push_back(i);
			flag = 1;
			for (int j = 0; j < (int)adj[i].size(); ++j) {
				cnt[adj[i][j]]++;	
			}
		}
		if(!flag) break;
	}
	printf("%d\n", (int)res.size());
	for (int i = 0; i < (int)res.size(); ++i) {
		if (i != 0) printf(" ");
		printf("%d", res[i] + 1);
	}
	printf("\n");

	return 0;
}
