#include <stdio.h>
#include <vector>
#include <queue>
#include <iostream>

using namespace std;

vector<int> v[100005];
vector<int> use;

queue<int> q;
int n,m;
int a[100005];
int b[100005];

int main(){
	scanf("%d%d",&n,&m);
	while(m--){
		int s,e;
		scanf("%d%d",&s,&e);
		v[s].push_back(e);
		v[e].push_back(s);
	}
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1;i<=n;i++)if(!a[i])q.push(i);
	
	while(q.size()){
		int k=q.front();q.pop();
		if(a[k]!=b[k])continue;
		b[k]++;
		use.push_back(k);
		for(int i=0;i<v[k].size();i++){
			int p=v[k][i];
			b[p]++;
			if(b[p]==a[p])q.push(p);
		}
	}
	cout<<use.size()<<endl;
	for(int i=0;i<use.size();i++)printf("%d ",use[i]);
}
