#include<bits/stdc++.h>

using namespace std;

#define pb push_back

int A[100005];

int i,j,x,y,n,m;

vector<int> V[100005],ans;

void dfs(int x){
	
	if(A[x])
		return;

	A[x]--;

	ans.pb(x);

	int j;

	for(j=0 ; j<V[x].size() ; j++)
		A[V[x][j]]--;
	for(j=0 ; j<V[x].size() ; j++)
		dfs(V[x][j]);
}

int main(){

	cin>>n>>m;

	for(i=1 ; i<=m ; i++){
		cin>>x>>y;

		V[x].pb(y);
		V[y].pb(x);
	}

	for(i=1 ; i<=n ; i++)
		cin>>A[i];

	for(i=1 ; i<=n ; i++)
		if(!A[i]){
			A[i]--;
			ans.pb(i);
			for(j=0 ; j<V[i].size() ; j++)
				A[V[i][j]]--;
			for(j=0 ; j<V[i].size() ; j++)
				dfs(V[i][j]);
		}

	sort(ans.begin(),ans.end());

	cout<<ans.size()<<endl;

	for(i=0 ; i<ans.size() ; i++)
		cout<<ans[i]<<' ';
	
	return 0;
}
