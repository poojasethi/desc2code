#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
#include <stdio.h>
using namespace std;
int main()
{
vector <int> e[100005];
vector <int> ans;
int a[100005],b[100005]={0};
int n,m;
//freopen("i.txt","r",stdin);
cin>>n>>m;
for (int i=0;i<m;i++)
{
 int u,v;
 cin>>u>>v;
 e[u].push_back(v);
 e[v].push_back(u);
}
queue <int> q;
for (int i=1; i<=n;i++)
 {cin>>a[i];
 if (a[i]==b[i]) q.push(i);
 }
while (q.size()>0)
{ int u=q.front();
   q.pop();
  if (a[u]==b[u])
  {
   b[u]++;
   ans.push_back(u);
   for (int i=0;i<e[u].size();i++)
    {
     int it=e[u][i];
     b[it]++;
     if (b[it]==a[it]) q.push(it);
    }
  }
}

cout<<ans.size()<<endl;
for (int i=0;i<ans.size();i++) cout<<ans[i]<<" ";

return 0;
}
