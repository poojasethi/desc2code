#include <iostream>
#include <vector>
#include <queue>

using namespace std;

const int N = 1e5 + 7;

int a[N], b[N];

vector <int> adj[N];
vector <int> click;

queue <int> que;

int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < m; ++i) {
		int u, v;
		cin >> u >> v;
		u --, v --;
		adj[u].push_back(v);
		adj[v].push_back(u);
	}
	for (int i = 0; i < n; ++i) {
		cin >> a[i];
		if (a[i] == 0)
			que.push(i);
	}
	while (!que.empty()) {
		int x = que.front();
		que.pop();
		click.push_back(x);
		for (int i = 0; i < adj[x].size(); ++i) {
			int u = adj[x][i];
			b[u] ++;
			if (b[u] == a[u]) {
				que.push(u);
				b[u] ++;
			}
		}
	}
	cout << click.size() << "\n";
	for (int i = 0; i < click.size(); ++i)
		cout << click[i] + 1 << " ";
	cout << "\n";
	return 0;
}
