#include <iostream>
#include <algorithm>
#include <vector>
#include <cmath>
#define PB push_back
using namespace std;

vector<int> G[100006];
int a[100007],b[100006];
vector<int>odp;
vector<int>p;

int main()
{
 int n,m;
 cin>>n>>m;
 while(m--)
 {
          int x,y;
            cin>>x>>y;
            G[x].PB(y);
            G[y].PB(x); 
           }
 for(int i = 0 ; i <n;i++)
 cin>>b[i+1];
 int it=0;
 for(int i = 1;i<=n;i++)if(a[i]==b[i])p.PB(i);
 
 while(it!=p.size())
 {
       int v=p[it++];
       odp.PB(v);
       a[v]++;
       for(int i = 0 ; i < G[v].size();i++)
       {a[G[v][i]]++;if(a[G[v][i]]==b[G[v][i]])p.PB(G[v][i]);}             
       }
 cout<<it<<"\n";
 for(int i = 0 ; i < odp.size();i++)
 cout<<odp[i]<<" ";
 //system("PAUSE");
 return 0;
}
