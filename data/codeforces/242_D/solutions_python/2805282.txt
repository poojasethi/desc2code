R=lambda:map(int,raw_input().split())
n,m=R()
g=[[] for _ in range(n+1)]
for _ in range(m):
  a,b=R()
  g[a-1]+=[b-1]
  g[b-1]+=[a-1]
c=R()
q=[i for i in range(n) if not c[i]]
v=[x+1 for x in q]
while q:
  a=q.pop(0)
  for b in g[a]:
    c[b]-=1
    if not c[b]:
      q+=[b]
      v+=[b+1]
print len(v)
print ' '.join(map(str,v))
