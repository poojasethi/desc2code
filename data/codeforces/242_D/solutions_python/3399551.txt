r = lambda: map(int, raw_input().split())
n, m = r()
e = [[] for _ in xrange(n)]
for i in xrange(m):
    a, b = r()
    e[a - 1].append(b - 1)
    e[b - 1].append(a - 1)
a = r()
b = [0] * n
q = [i for i in xrange(n) if a[i] == b[i]]
ans = []
while q:
    x = q.pop(0)
    ans.append(x + 1)
    for y in e[x]:
        b[y] += 1
        if a[y] == b[y]: q.append(y)
print len(ans)
print ' '.join(map(str, ans))
