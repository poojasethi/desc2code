#include <stdio.h>
#include <algorithm>
using namespace std;

int a[105][2] ={}, u[105], b[105][105] ={}, flag=0;
int n, m;
void dfs(int x, int color) {
	u[x] = color;
	for(int i=0;i<m;i++) if(b[x][i]) {
		if(!u[i]) dfs(i, -color);
		else if(u[i]==color) flag = 1;
	}
}

int main(void) {
	int i, j;
	scanf("%d%d", &n, &m);
	for(i=0;i<m;i++) {
		scanf("%d%d", &a[i][0], &a[i][1]);
		if(a[i][0]>a[i][1]) swap(a[i][0], a[i][1]);
	}
	for(i=0;i<m;i++)
		for(j=0;j<m;j++)
			if(i!=j) b[i][j]=b[j][i]=1;
	for(i=0;i<m;i++)
		for(j=0;j<m;j++){
			if(i==j) continue;
			if((a[i][0]<=a[j][0] && a[j][1]<=a[i][1]) ||
			   (!(a[i][0]>a[j][0] && a[i][0]<a[j][1]) &&
				!(a[i][1]>a[j][0] && a[i][1]<a[j][1]))) {
				b[i][j]=b[j][i]=0;
			}
		}
	for(i=0;i<m;i++)
		if(!u[i]) dfs(i, 1);
	if(flag) printf("Impossible\n");
	else
	for(i=0;i<m;i++)
		if(u[i]==1) printf("i");
		else printf("o");
	return 0;
}
