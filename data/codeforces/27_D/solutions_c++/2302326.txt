#include <stdio.h>
#include <iostream>
#include <string.h>
#include <queue>
#include <stack>
#include <algorithm>
#include <math.h>
using namespace std;
int n,m;
int a[210],b[210];
int ret[210],g[210][210];

void dfs(int x)
{
	for(int i=0;i<m;i++)
	{
		if(!g[x][i]) continue;
		if(ret[i]==0) ret[i]=3-ret[x],dfs(i);
		else if(ret[i]+ret[x]!=3) ret[i]=-1;
	}
}

int main()
{
	scanf("%d%d",&n,&m);
	for(int i=0;i<m;i++)
	{
		scanf("%d%d",a+i,b+i);
		if(a[i]>b[i]) swap(a[i],b[i]);
	}
	for(int i=0;i<m;i++)
		for(int j=0;j<m;j++)
		{
			if((a[i]>a[j] && a[i]<b[j] && b[i]>b[j]))
				g[i][j]=g[j][i]=1;
		}
	for(int i=0;i<m;i++)
	{
		if(ret[i]) continue;
		ret[i]=1;
		dfs(i);
	}
	for(int i=0;i<m;i++)
	{
		if(ret[i]>=0) continue;
		cout<<"Impossible\n";
		return 0;
	}
	for(int i=0;i<m;i++)
	{
		if(ret[i]==2) cout<<"i";
		else cout<<"o";
	}
	cout<<endl;
	return 0;
}
