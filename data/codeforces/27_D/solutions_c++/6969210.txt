#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int MaxN =110;

int n,m;
int a[MaxN],b[MaxN];
int fa[MaxN << 1],flag[MaxN << 1];

int getfa(int u){return fa[u]==u ? u : fa[u]=getfa(fa[u]);}
inline void join(int u,int v){u=getfa(u),v=getfa(v);fa[v]=u;}
inline int query(int u,int v){return getfa(u)==getfa(v);}

int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++){
		scanf("%d%d",&a[i],&b[i]);
		if(a[i]>b[i]) swap(a[i],b[i]);
	}
	for(int i=1;i<=m+m;i++) fa[i]=i;
	for(int i=1;i<=m;i++)
		for(int j=i+1;j<=m;j++)
			if((a[i]<a[j] && a[j]<b[i] && b[i]<b[j])
			|| (a[j]<a[i] && a[i]<b[j] && b[j]<b[i]))
				join(i,j+m),join(i+m,j);
	for(int i=1;i<=m;i++) if(query(i,i+m)){puts("Impossible");return 0;}
	for(int i=1;i<=m;i++) if(fa[i]==i) flag[i]=1;
	for(int i=1;i<=m;i++) putchar(flag[getfa(i)] ? 'o' : 'i');
	puts("");
	return 0;
}
