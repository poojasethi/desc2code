#include <stdio.h>
#include <algorithm>
using namespace std;
#define rep(i, n) for(int i=0; i<(int)(n); i++)

int n, m;
int cr[100][100];
int a[100], b[100];
char io[101];
bool invalid;

void decide(int k, char ch) {
    if(invalid) return ;
    if(io[k]>0) {
        if(io[k]!=ch) invalid=true;
        return ;
    }
    io[k]=ch;
    char op=ch=='i'?'o':'i';
    rep(i, m) if(cr[k][i]) decide(i, op);
}

int main() {
    scanf("%d%d", &n, &m);
    rep(i, m) scanf("%d%d", a+i, b+i);
    rep(i, m) if(a[i]>b[i]) swap(a[i], b[i]);
    rep(i, m) rep(j, i) {
        if(a[i]<a[j] && a[j]<b[i] && b[i]<b[j]) cr[i][j] = cr[j][i] = 1;
        if(a[i]<b[j] && b[j]<b[i] && a[j]<a[i]) cr[i][j] = cr[j][i] = 1;
    }
    invalid = false;
    rep(i, m) if(io[i]==0) {
        decide(i, 'i');
    }
    io[m] = 0;
    if(invalid) printf("Impossible\n");
    else printf("%s\n", io);
    return 0;
}
