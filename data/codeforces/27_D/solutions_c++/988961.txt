#include<iostream>
#include<algorithm>
using namespace std;

int n,m;
int a[200],b[200];
int o[200][200];
int c[200];
bool f;
void dfs(int a) {
	for(int b=0;b<m;b++) {
		if(o[a][b]) {
			if(!c[b]) {
				c[b]=3-c[a];
				dfs(b);
			} else if(c[b]==c[a]) {
				f=true;
			}
		}
	}
}
			
int main() {
	cin>>n>>m;
	for(int i=0;i<m;i++) {
		cin>>a[i]>>b[i];
		if(a[i]>b[i])swap(a[i],b[i]);
	}
	for(int i=0;i<m;i++) {
		for(int j=0;j<m;j++) {
			o[j][i]|=o[i][j]|=(a[i]>a[j] && a[i]<b[j] && b[i]>b[j]);
		}
	}
	for(int i=0;i<m;i++) {
		if(c[i]==0) {
			c[i]=1;
			dfs(i);
		}
	}
	if(f) {
		cout<<"Impossible";
	} else {
		for(int i=0;i<m;i++) {
			cout<<"ioi"[c[i]];
		}
	}
}
