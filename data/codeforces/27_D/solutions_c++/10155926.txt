#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int ans[110],map[110][110],f[110],x[110],y[110];
int n,m,flag=0;
void Dfs(int k){
    f[k]=1;
    if(flag) return;
    for(int i=1;i<=m;i++){
        if(map[i][k]){
            if(!f[i]){
                f[i]=1;ans[i]=3-ans[k];
                Dfs(i);
            }else{
                if(ans[k]==ans[i]) flag=1;
            }
        }

    }
}
int main(){
   // freopen("test.in","r",stdin);
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;i++){
        scanf("%d%d",&x[i],&y[i]);
        if(x[i]>y[i]) swap(x[i],y[i]);
        for(int j=1;j<i;j++){
            if(y[i]<=x[j]||x[i]>=y[j]) continue;
            if(x[i]<=x[j]&&y[i]>=y[j]) continue;
            if(x[i]>=x[j]&&y[i]<=y[j]) continue;
            map[i][j]=map[j][i]=1;
        }
    }
    for(int i=1;i<=m;i++)
        if(!f[i]){
            ans[i]=1;
            Dfs(i);
        }
    if(flag){
        printf("Impossible\n");
        return 0;
    }
    for(int i=1;i<=m;i++){
        if(ans[i]==1) printf("i");
            else printf("o");
    }
    return 0;
}
