#include <iostream>
#include <stdlib.h>
using namespace std;
int n, m, i, j, sw;
int a[200], b[200], col[200];
bool u[200][200];
void dfs(int k, int cv)
{
	 col[k] = cv;
	 for (int i = 1; i <= m; i++)
	 {
		 if (!u[k][i]) continue;
		 if (col[i] == 0) dfs(i, 3 - cv);
		 else if (col[i] == col[k])
		 {
			 cout << "Impossible";
			 exit(0);
		 }
	 }
}
int main()
{
	 cin >> n >> m;
	 for (i = 1; i <= m; i++)
	 {
		 cin >> a[i] >> b[i];
		 if (a[i] > b[i])
		 {
			 sw = a[i]; a[i] = b[i]; b[i] = sw;
		 }
	 }
	 for (i = 1; i <= m; i++)
	 {
		 for (j = i + 1; j <= m; j++)
		 {
			 if ((a[j] < a[i]) && (a[i] < b[j]) && (b[j] < b[i]))
			 {
				 u[i][j] = true;
				 u[j][i] = true;
			 }
			 if ((a[i] < a[j]) && (a[j] < b[i]) && (b[j] > b[i]))
			 {
				 u[i][j] = true;
				 u[j][i] = true;
			 }
		 }
	 }
	 for (i = 1; i <= m; i++)
	 {
		 if (!col[i]) dfs(i, 1);
	 }
	 for (i = 1; i <= m; i++)
	 {
		 if (col[i] == 1) cout << "i";
		 else cout << "o";
	 }
	 //cin >> n;
	 return 0;
}
