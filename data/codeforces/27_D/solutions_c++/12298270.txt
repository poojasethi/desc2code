using namespace std;
int b[109],e[105][5];
vector<vector<int> >g;
int dfs(int u){
    for(int i=0;i<g[u].size();++i){
        if(b[g[u][i]]==0){
            b[g[u][i]]=b[u]^3;
            dfs( g[u][i]);
        }if(b[g[u][i]]==b[u]){
            cout<<"Impossible";
            exit(0);
        }
    }
}
int main(){

    ios_base::sync_with_stdio(false);
//    freopen("input.txt","r",stdin);
//    freopen("output.txt","w",stdout);
    int n,m;
    cin>>n>>m;
    g=vector<vector<int> >(m+1);
    for(int i=0;i<m;++i){
        cin>>e[i][0]>>e[i][1];
        if(e[i][1]<e[i][0])swap(e[i][0],e[i][1]);
        for(int j=0;j<i;++j){
            if((e[j][0]<e[i][1]&&e[j][0]>e[i][0]&&e[j][1]>e[i][1])||(e[j][1]<e[i][1]&&e[j][1]>e[i][0]&&e[j][0]<e[i][0])){
                g[i].pb(j);
                g[j].pb(i);
            }
        }
    }
    for(int i=0;i<m;++i)if(b[i]==0)b[i]=1,dfs(i);
    for(int i=0;i<m;++i)cout<<(b[i]==1?'i':'o');

}
