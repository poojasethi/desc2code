#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,a[1001],b[1001],tag[1001],u;
char s[1001],ss[] = {'i','o'};
bool in(int a,int b,int c){
  return (b<a && a<c);
}
bool cmp(int i,int j){
  return (a[i] != a[j] && a[i] != b[j] &&  a[j] != b[i] &&b[i] != b[j]);
}
bool dfs(int now,int xx){
  tag[now] = 1; s[now] = ss[xx];
  for(int i=0;i<m;i++){
    if(cmp(now,i) && (in(a[now],a[i],b[i]) ^ in(b[now],a[i],b[i]))){
      if((tag[i] && s[i] == s[now]) || (!tag[i] && dfs(i,xx^1)))
     return true;
    }
  }
  return false;
}

  scanf("%d%d",&n,&m);
  for(int i=0;i<m;i++){
    scanf("%d%d",&a[i],&b[i]);
    if(a[i] > b[i]) swap (a[i],b[i]);
  }
  for(int i=0;i<m && !u;i++)
    if(!tag[i])

  if(u == 1) printf("Impossible\n");
  else printf("%s\n",s);
  return 0;
}
