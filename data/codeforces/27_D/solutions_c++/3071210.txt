#include <stdio.h>
int d[1000][1000];
int c[1000][2];
int ans[1000];
int n,m;
    
int dfc(int i,int color){
    int a=0;
    for (int j=0;j<m;j++)
        if (d[i][j])
        if (ans[j]==0){
            ans[j]=color;
            a+=dfc(j,3-color);
        }else
            if (ans[j]!=color) 
                return 1;
        return a;
}
int main(){
    scanf("%d%d",&n,&m);
    for (int i=0;i<m;i++){
        int a,b;
        scanf("%d%d",&a,&b);
        if (a>b){
            c[i][0]=a;
            c[i][1]=b;
        }else{
            c[i][0]=b;
            c[i][1]=a;      
        }
        for (int j=0;j<i;j++){
            if (((a<c[j][0]&&a>c[j][1])&&!(b<=c[j][0]&&b>=c[j][1]))||
                    ((!(a<=c[j][0]&&a>=c[j][1]))&&(b<c[j][0]&&b>c[j][1]))){
                d[i][j]=1;
                d[j][i]=1;
            }
        }
        
    }
    int anss=0;
    for (int i=0;i<n;i++){
        if (ans[i]==0){
            ans[i]=1;
            anss+=dfc(i,2);
        }
    }
    if (anss)
        printf("Impossible");
    else
        for (int i=0;i<m;i++)
            if (ans[i]==1)
                printf("i");
            else
                printf("o");
    
    return 0;
}
