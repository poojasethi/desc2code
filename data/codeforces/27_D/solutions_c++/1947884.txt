#include<iostream>
#include<cstring>
using namespace std;
int w[105][105],N,M,x[105],y[105];
bool flag = 0;
int vis[105];

int dfs(int ind,int col)
{
 if(vis[ind] != -1 && vis[ind]!=col)
 {
  if(!flag)
  cout<<"Impossible"<<endl;
  flag = 1;
  return 0;
 }
 if(vis[ind] == col)return 0;
 vis[ind] = col;
 for(int i=0;i<M;i++)
 if(w[ind][i])
 dfs(i,!col);
 return 0;
}


int check(int a,int b)
{
 if(x[a]<=x[b] && y[b]<=y[a])return 0;
 if(x[b]<=x[a] && y[a]<=y[b])return 0;
 if(x[a]>=y[b] || x[b]>=y[a])return 0;
 return 1;
}


int main()
{
 int i,j;
 memset(vis,-1,sizeof vis);
 cin>>N>>M;
 for(i=0;i<M;i++)
 {
  cin>>x[i]>>y[i];
  if(x[i]>y[i])swap(x[i],y[i]);
 }
 for(i=0;i<M;i++)
  for(j=i+1;j<M;j++)
  if(check(i,j))w[i][j] = w[j][i] = 1;
  
 /*for(i=0;i<M;i++)
 {
  for(j=0;j<M;j++)
  cout<<w[i][j]<<" ";
 cout<<endl;
 }*/
  
 for(i=0;i<M;i++)
 if(vis[i]==-1)
 dfs(i,0);
 if(flag == 0)
 {
 for(i=0;i<M;i++)
 if(vis[i]==0)cout<<"i";
 else cout<<"o";
 cout<<endl;
 }
 return 0;
}
