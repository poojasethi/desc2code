#include<bits/stdc++.h>

using namespace std;

#define LL long long
#define mod 1000000007
#define N 100005

int u[100],v[100],c[100],st[101];
vector<int> mp[100];
bool pos;

void dfs(int u)
{
    for(int i=0;i<mp[u].size();i++)
    {
        int v = mp[u][i];

        if(c[v]==c[u])
        {
            cout<<"Impossible";
            exit(0);
        }
        if(c[v]!=-1)    continue;

        c[v] = (c[u]+1)%2;
        dfs(v);
    }
}

int main()
{
    int i,n,m,j,a,b;
    cin>>n>>m;
    for(i=0;i<m;i++)
    {
        cin>>a>>b;
        u[i] = min(a,b);
        v[i] =max(a,b);
        c[i] = -1;
    }


    for(i=0;i<m;i++)
        for(j=0;j<m;j++)
            if(j!=i)
                if((u[i]<u[j] && v[i]<v[j] && u[j]<v[i]) || (u[i]>u[j] && v[i]>v[j] && u[i]<v[j]))
                    mp[i].push_back(j);

    for(i=0;i<m;i++)
        if(c[i]==-1)
        {
            c[i] = 0;
            dfs(i);
        }

    for(i=0;i<m;i++)
        if(c[i])    cout<<'i';
        else    cout<<'o';
    return 0;
}
