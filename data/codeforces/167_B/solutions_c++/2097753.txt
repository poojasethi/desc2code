#include <iostream>
#include <cstdio>

using namespace std;

const int MAXN = 200;

double p[MAXN], dp[MAXN + 1][MAXN + 1][MAXN + 1];
int a[MAXN];

int main() {
	int n, l, k;
	double ans = 0;
	cin >> n >> l >> k;
	for (int i = 0; i < n; i++) {
		cin >> p[i];
		p[i] /= 100.0;
	}
	for (int i = 0; i < n; i++) {
		cin >> a[i];
		a[i]++;
	}
	dp[0][0][k] = 1.0;
	for (int i = 0; i < n; i++)
		for (int j = 0; j <= i; j++)
			for (int m = 0; m <= MAXN; m++) {
				dp[i + 1][j][m] += dp[i][j][m] * (1.0 - p[i]);
				dp[i + 1][j + 1][min(MAXN, m + a[i])] += dp[i][j][m] * p[i];
			}
	for (int i = l; i <= n; i++)
		for (int j = i; j <= MAXN; j++)
			ans += dp[n][i][j];
	printf("%.9lf", ans);
	return 0;
}
