#include <algorithm>
#include <cstdio>

using namespace std;

const int N = 200 + 11;

int n, L, K;
int P[N], a[N];
double dp[N][N][2*N];

int main() {

  scanf("%d %d %d ", &n, &L, &K);
  for (int i = 1; i <= n; ++i) {
    scanf("%d ", P+i);
  }
  for (int i = 1; i <= n; ++i) {
    scanf("%d ", a+i);
  }

  dp[0][0][min(2*n, n+K)] = 1;
  for (int i = 0; i < n; ++i) {
    double p = P[i+1] / 100.0;
    for (int j = 0; j <= i; ++j) {
      for (int k = n-i; k <= 2*n; ++k) {
        dp[i+1][j+1][min(k+a[i+1], 2*n)] += p*dp[i][j][k];
        dp[i+1][j][k] += (1-p) * dp[i][j][k];
      }
    }
  }

  double sol = 0;
  for (int i = L; i <= n; ++i) {
    for (int j = n; j <= 2*n; ++j) {
      sol += dp[n][i][j];
    }
  }

  printf("%.10lf\n", sol);

  return 0;
}
