#include <cstdio>
#include <algorithm>

using namespace std;

double dp[220][220][420];
bool z[220][220][420];
double vet[220];
int pri[220];

int n, l, k;

double top(int a, int b, int c)
{
	if (a == n)
	{
		if (b >= l && c >= 210) return 1;
		return 0;
	}
	if (z[a][b][c]) return dp[a][b][c];
	z[a][b][c] = 1;
	
	double res = 0;
	
	if (pri[a] == -1)
	{
		res = vet[a]/100.0*top(a+1, b+1, max(0,c-1));
		res += (100-vet[a])/100.0*top(a+1, b, c);
	}
	else
	{
		res = vet[a]/100.0*top(a+1, b+1, min(419, c+pri[a]));
		res += (100-vet[a])/100.0*top(a+1, b, c);
	}
	
	return dp[a][b][c] = res;
}

int main()
{
	scanf("%d%d%d", &n, &l, &k);
	for (int i = 0; i < n; ++i)
		scanf("%lf", &vet[i]);
	for (int i = 0; i < n; ++i)
		scanf("%d", &pri[i]);
		
	printf("%.7lf\n", top(0, 0, 210+k));
	
	return 0;
}
