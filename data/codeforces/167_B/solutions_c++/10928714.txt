#include<iostream>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<cstdio>
using namespace std;

double dp[204][204][404];
double p[204];
int v[204];
int zero=200;
int main()
{
	int n,l,k,temp,i,j,m;
	cin>>n>>l>>k;
	for(i=1;i<=n;i++)
		cin>>temp,p[i]=temp/100.0;
	for(i=1;i<=n;i++)
		cin>>v[i];
	dp[0][0][k+zero]=1.0;
	for(i=1;i<=n;i++)
		for(j=0;j<=i;j++)
			for(m=0;m<=zero+200;m++)
			{
				if(v[i]==-1)
				{
					dp[i][j+1][m-1]+=dp[i-1][j][m]*p[i];
					dp[i][j][m]+=dp[i-1][j][m]*(1-p[i]);
				}
				else
				{
					int x=min(zero+200,m+v[i]);
					dp[i][j+1][x]+=dp[i-1][j][m]*p[i];
					dp[i][j][m]+=dp[i-1][j][m]*(1-p[i]);
				}
			}
	double res=0;
  	for(i=l;i<=n;i++)
		for(j=zero;j<=zero+200;j++)
		{
			res+=dp[n][i][j];
		}
		printf("%.12lf",res);
	return 0;
}
