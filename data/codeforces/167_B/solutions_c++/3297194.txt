#include<cstdio>
#include<algorithm>
using namespace std;

int n,l,k,c[201];
double p[201];
double dp[201][201][501];

int main()
{
    scanf("%d%d%d",&n,&l,&k);
    for(int i=0;i<n;++i){
        scanf("%lf",&p[i]);
        p[i]/=100;
    }
    for(int i=0;i<n;++i) scanf("%d",&c[i]);
    dp[0][0][n+min(k,n)]=1;
    for(int i=0;i<n;++i)
        for(int j=0;j<=n;++j)
            for(int s=-n;s<=n;++s){
                dp[i+1][j][n+s]+=dp[i][j][n+s]*(1-p[i]);
                dp[i+1][j+1][n+min(s+c[i],n)]+=dp[i][j][n+s]*p[i];
            }
    double ans=0;
    for(int i=l;i<=n;++i)
        for(int j=0;j<=n;++j)
            ans+=dp[n][i][n+j];
    printf("%.6lf\n",ans);
    return 0;
}
