#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<vector>
using namespace std;
typedef long long LL;
LL n,l,K,k,i,j,p[205],a[205];
double dp[222][222][622],ans,tp;
int main(){
	scanf("%I64d%I64d%I64d",&n,&l,&K);
	dp[0][0][K+200]=1.0;
	for(i=0;i<n;++i)scanf("%I64d",&p[i]);
	for(i=0;i<n;++i)scanf("%I64d",&a[i]);
	for(i=0;i<n;++i){
		for(j=0;j<=i;++j){
			for(k=0;k<=600;++k){
				tp=p[i]/(100.0);
				dp[i+1][j+1][min(600LL,k+a[i])]+=dp[i][j][k]*tp;
				dp[i+1][j][k]+=dp[i][j][k]*(1-tp);
			}
		}
	}
	for(i=l;i<=n;++i){
		for(j=200;j<=600;++j){
			ans+=dp[n][i][j];
		}
	}
	printf("%.12lf\n",ans);
	return 0;
}
