#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define db double
#define N 210
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,K,nn;
int a[N];
db p[N],f[N][N][N*2],x,an;
int main() {
	scanf("%d%d%d",&n,&m,&K);
	For(i,1,n) scanf("%lf",&p[i]),p[i]/=100;
	For(i,1,n) scanf("%d",&a[i]);
	nn=N*2-1; f[0][0][min(nn,N+K)]=1;
	For(i,0,n-1)For(j,0,i)For(k,0,nn) if (x=f[i][j][k]) {
		if (a[i+1]<0) f[i+1][j+1][max(0,k-1)]+=x*p[i+1];
		else f[i+1][j+1][min(nn,k+a[i+1])]+=x*p[i+1];
		f[i+1][j][k]+=x*(1-p[i+1]);
	}
	For(j,m,n)For(i,N,nn) an+=f[n][j][i];
	printf("%.11lf\n",an);
	//for(;;);
	return 0;
}
