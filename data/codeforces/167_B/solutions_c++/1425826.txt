#include <stdio.h>
using namespace std;

#include <algorithm>
int N, L, K;
double p[200];
int c[200];
double dp[201][201][501];

int main() {
	scanf("%d %d %d", &N, &L, &K);
	for (int i= 0; i < N; i++) {
		scanf("%lf", &p[i]);
		p[i] /= 100;
	}
	for (int i = 0; i  < N; i++)
		scanf("%d", &c[i]);
	dp[0][0][N+min(K,N)] = 1;
	for (int n = 0; n < N; n++) {
		for (int w = 0; w <= n; w++) {
			for (int s = -N; s <= N; s++) {
				dp[n+1][w][N+s] += dp[n][w][N+s]*(1-p[n]);
				dp[n+1][w+1][N+min(s+c[n],N)] += dp[n][w][N+s]*p[n];
			}
		}
	}
	double erg = 0;
	for (int w = L; w <= N; w++)
		for (int s = 0; s <= N; s++)
			erg += dp[N][w][s+N];
	printf("%.12lf\n", erg);
	return 0;
}
