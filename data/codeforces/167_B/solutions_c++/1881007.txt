#include <cstdio>
#define rep(i,n) for(int i=0;i<(n);++i)

int a[210], p[210], n;
double memo[210][210][210];

double func(int i, int l, int k)
{
	if (i == n) {
		if (l > 0) return 0;
		if (k < 0) return 0;
		return 1;
	}
	int K = k < 209 ? k : 209;
	int L = l < 0 ? 0 : l;
	double& res = memo[i][L][K];
	if (res < 0) {
		res = 0.01 * p[i] * func(i+1, l-1, k+a[i]);
		res += (1 - 0.01 * p[i]) * func(i+1, l, k);
	}
	return res;
}

int main()
{
	int k, l;
	while (~scanf("%d%d%d", &n, &l, &k)) {
		rep(i, n) scanf("%d", p+i);
		rep(i, n) scanf("%d", a+i);
		rep(i, 210) rep(j, 210) rep(K, 210) memo[i][j][K] = -1;
		printf("%.10f\n", func(0, l, k));
	}
	return 0;
}
