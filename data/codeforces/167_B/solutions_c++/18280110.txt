#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

#define N 220


double dp[N][N][N * 2];
int n, l, k;
double p[N];
int a[N];

int main() {
	scanf("%d%d%d", &n, &l, &k);
	for(int i = 1; i <= n; ++i) {
		scanf("%lf", &p[i]);
		p[i] /= 100;
	}
	for(int i = 1; i <= n; ++i) scanf("%d", &a[i]);
	k = min(k, n);
	dp[1][0][k + N] = 1;
	for(int i = 1; i <= n; ++i) {
		for(int j = 0; j <= i; ++j) {
			for(int u = -i; u <= n; ++u) {
				if(dp[i][j][u+N] > 0) {
					dp[i+1][j + 1][N + min(u + a[i], n)] += dp[i][j][u + N] * p[i];
					dp[i+1][j][u + N] += dp[i][j][u + N] * (1 - p[i]);
				}
			}
		}
	}
	double ans = 0;
	for(int j = l; j <= n; ++j) for(int u = 0; u <= n; ++u) ans += dp[n+1][j][u + N];
	printf("%.12lf\n", ans);
	return 0;
}
