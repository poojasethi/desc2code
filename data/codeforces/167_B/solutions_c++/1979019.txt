#include<iostream>
#include<cstdio>
using namespace std;
double dp[202][202][401];
int n, l, k;
double p[201];
int  a[201];
int main()
{
  cin>>n>>l>>k;
  for (int i = 1; i <= n; i++)
    cin>>p[i], p[i] /= 100.0;
  for (int i = 1; i <= n; i++)
    cin>>a[i];
  dp[0][0][k + 200] = 1.0;
  for (int i = 0; i <= n; i++)
    for (int w = 0; w <= 200; w++)
      for (int c = 0; c <= 400; c++) {
	dp[i+1][w+1][min(c+a[i+1],400)] += dp[i][w][c]*p[i + 1];
	dp[i+1][w][c] += dp[i][w][c]*(1.0-p[i + 1]);
      }
  double ans = 0.0;
  for (int w = l; w <= n; w++)
    for (int c = 200; c <= 400; c++)
      ans += dp[n+1][w][c];
  printf ("%.9lf\n", ans);
  return 0;
}
