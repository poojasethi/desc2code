#include<cstdio>
#include<algorithm>
using namespace std;
int n,l,k,i,p[256],a[256],j,w;
double f[201][512][201],aa;
int main(){
	scanf("%d%d%d",&n,&l,&k);k=min(k,n);
	for(i=0;i<n;++i)scanf("%d",p+i);
	for(i=0;i<n;++i)scanf("%d",a+i);
	f[0][k+256][0]=1;
	for(i=0;i<n;++i)for(j=-i;j<=n;++j)for(w=0;w<=i;++w)
	if(f[i][j+256][w]>0){
		f[i+1][256+min(j+a[i],n)][w+1]+=f[i][j+256][w]*p[i]/100;
		f[i+1][j+256][w]+=f[i][j+256][w]*(100-p[i])/100;
	}
	for(j=0;j<=n;++j)for(w=l;w<=n;++w)aa+=f[n][j+256][w];
	printf("%.12lf\n",aa);
	return 0;
}
