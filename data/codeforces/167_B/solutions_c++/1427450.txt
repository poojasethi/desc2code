#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,a[210],p[210],sum,l,kk,pre,now;
double f[2][210][410],ans;
int main(){
    scanf("%d%d%d",&n,&l,&kk);if(kk>n+n)kk=n+n;
    f[0][0][kk]=1.0;
    for(int i=1;i<=n;++i)scanf("%d",&p[i]);
    for(int i=1;i<=n;++i){scanf("%d",&a[i]);++a[i];}
    for(int i=1;i<=n;++i){
        pre=now;now^=1;
        memset(f[now],0,sizeof(f[now]));
        for(int j=0;j<i;++j)
            for(int k=0;k<=n+n;++k){
                f[now][j+1][min(k+a[i],n+n)]+=f[pre][j][k]*p[i]/100.0;
                f[now][j][k]+=f[pre][j][k]*(100-p[i])/100.0;
            }
    }
    for(int i=l;i<=n;++i)
        for(int j=i;j<=n+n;++j)
            ans+=f[now][i][j];
    printf("%.10lf\n",ans);
    return 0;
}
