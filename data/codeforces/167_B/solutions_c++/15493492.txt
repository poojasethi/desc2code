#include<bits/stdc++.h>

using namespace std;

#define LL long long
#define mod 1000000007

double p[200],dp[201][201][501];
int a[200];

int main()
{
    int n,l,k,i,x,j;

    cin>>n>>l>>k;
    for(i=0;i<n;i++)
    {
        cin>>x;
        p[i] = 0.01*x;
    }
    for(i=0;i<n;i++)
        cin>>a[i];

    dp[0][0][200+k] = 1;
    for(i=0;i<n;i++)
        for(j=0;j<=i;j++)
            for(k=0;k<=500;k++)
            {
                if(k+a[i]>=0)
                    dp[i+1][j+1][min(500,k+a[i])]+=p[i]*dp[i][j][k];

                dp[i+1][j][k]+=(1-p[i])*dp[i][j][k];
            }

    double ans = 0;
    for(j=l;j<=n;j++)
        for(k=200;k<=500;k++)
            ans+=dp[n][j][k];
    printf("%0.9lf",ans);
    return 0;
}
