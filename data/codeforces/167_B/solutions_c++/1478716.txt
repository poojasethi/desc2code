#include <iostream>
#include <cstdio>
using namespace std;
const int N = 210;
int n, l, k;
double dp[N][N][2 * N];
int p[N], a[N];
int main()
{
  cin >> n >> l >> k;
  for(int i = 1; i <= n; i++)
    cin >> p[i];
  for(int i = 1; i <= n; i++)
    cin >> a[i];
  dp[0][0][n + k] = 1.0;
  for(int i = 1; i <= n; i++)
    for(int j = 0; j < i; j++)
      for(int q = 0; q <= 400; q++)
      {
        dp[i][j + 1][min(400, q + a[i])] += dp[i - 1][j][q] * p[i] / 100.0;
        dp[i][j][q] += dp[i - 1][j][q] * (100 - p[i]) / 100.0;
      }
  double ans = 0;
  for(int i = l; i <= n; i++)
    for(int j = n; j <= 400; j++)
      ans += dp[n][i][j];
  printf("%.7lf", ans);
  return 0;
}
