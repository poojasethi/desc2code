#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int maxn = 201;
int p[maxn], a[maxn], N, L, K;
double f[maxn][maxn][maxn * 2];

int main()
{
	scanf("%d%d%d", & N, & L, & K);
	for (int i = 0; i < N; ++i) scanf("%d", & p[i]);
	for (int i = 0; i < N; ++i) scanf("%d", & a[i]);
	f[0][0][N + min(N, K)] = 1;
	for (int i = 0; i < N; ++i)
		for (int j = 0; j <= N; ++j)
			for (int k = -N; k <= N; ++k) 
				if (f[i][j][N + k] > 0)
				{
					f[i + 1][j + 1][N + min(k + a[i], N)] += p[i] * 1.0 / 100 * f[i][j][N + k];
					f[i + 1][j][N + k] += (100 - p[i]) * 1.0 / 100 * f[i][j][N + k];
				}
	double ans = 0;
	for (int j = L; j <= N; ++j)
		for (int k = N; k <= N + N; ++k) ans += f[N][j][k];
	cout << ans << endl;
}
