#include<iostream>
#include<algorithm>
using namespace std;
double dp[205][205][405]={0};
int n,k,l;
double p[201];
int a[201];
int main(){
    cin>>n>>l>>k;
    dp[0][0][min(200+k,401)]=1;
    for(int i=1;i<=n;++i)
    {
        cin>>p[i];
        p[i]*=0.01;
    }
    for(int i=1;i<=n;++i)cin>>a[i];
    for(int i=1;i<=n;++i)
    {
        for(int won=0;won<=n;++won)
            for(int j=0;j<402;++j)
                dp[i][won][j]+=dp[i-1][won][j]*(1.0-p[i]);
        for(int won=0;won<=n;++won)
            for(int j=1;j<402;++j)
                dp[i][won+1][min(j+a[i],401)]+=dp[i-1][won][j]*(p[i]);
    }
    double ans=0;
    for(int i=l;i<=n;++i)
        for(int j=200;j<402;++j)
            ans+=dp[n][i][j];
    cout<<ans<<endl;
    //system("pause");
}
