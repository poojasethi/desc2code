#include<stdio.h>
double f[201][201][500],p[201];
int n,l,k;
int a[201],g[201][201];

int main()
{
    scanf("%d%d%d",&n,&l,&k);

    for (int i=1;i<=n;i++) {scanf("%lf",&p[i]);p[i]/=100;}
    for (int i=1;i<=n;i++) scanf("%d",&a[i]);
    f[0][0][k+n]=1;int nn=400;
    for (int i=0;i<n;i++)
     for (int j=0;j<n;j++)
      for (int m=0;m<=nn;m++)
      if (f[i][j][m]>0)
      {
         f[i+1][j][m]+=f[i][j][m]*(1-p[i+1]);
         int tmp=m-1;
         if (a[i+1]>=0) tmp=tmp+1+a[i+1];
         if (tmp>nn) tmp=nn;
         f[i+1][j+1][tmp]+=f[i][j][m]*p[i+1];
      }
    double ans=0;
    for (int i=l;i<=n;i++)
     for (int j=n;j<=nn;j++) ans+=f[n][i][j];
    printf("%.6f\n",ans);
    return 0;
}
