#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define N 205
using namespace std;
double ans,p[N],f[N][N][N];
int a[N],n,l,K;
int main(){
	scanf("%d%d%d",&n,&l,&K); K=min(K,n);
	f[0][K][0]=1; 
	
	for(int i=1;i<=n;i++) scanf("%lf",&p[i]),p[i]/=100;
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	
	for(int i=1;i<=n;i++)
	for(int j=0;j<=n;j++)
	for(int k=0;k<=n;k++)
	if(f[i-1][j][k]){
		f[i][j][k]+=f[i-1][j][k]*(1-p[i]);
		if(a[i]==-1) f[i][j][k+1]+=f[i-1][j][k]*p[i];
		else f[i][min(j+a[i]+1,n)][k+1]+=f[i-1][j][k]*p[i];
	}

	for(int j=l;j<=n;j++)
	for(int k=l;k<=j;k++) ans+=f[n][j][k];
	
	printf("%.8lf\n",ans);
}
