#include<stdio.h>
#include<string.h>
#include<string>
#include<iostream>
#include<algorithm>
#include<math.h>
using namespace std;

double dp[205][205][405];
double p[205];
int a[205];

int main()
{
	int n,l,i,j,k,m,pp;
	scanf("%d%d%d",&n,&l,&m);
	for(i=0;i<n;i++)
	{
		scanf("%d",&pp);
		p[i]=pp/100.0;
	}
	for(i=0;i<n;i++) scanf("%d",&a[i]);
	dp[0][0][200+m]=1;
	for(i=0;i<n;i++)
		for(j=0;j<n;j++)
			for(k=1;k<=400;k++)
				if(dp[i][j][k]!=0.0)
				{
					int kk=k-1;
					if(a[i]!=-1)
					{
						kk+=a[i]+1;
						if(kk>400) kk=400;
					}
					dp[i+1][j+1][kk]+=dp[i][j][k]*p[i];
					dp[i+1][j][k]+=dp[i][j][k]*(1-p[i]);
				}
	double ans=0;
	for(i=l;i<=n;i++)
		for(j=200;j<=400;j++)
			ans+=dp[n][i][j];
	printf("%.10lf\n",ans);
}
