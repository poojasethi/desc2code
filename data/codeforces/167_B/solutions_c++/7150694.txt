#include<cstdio>
#define siz 205
using namespace std;
int n,L,K,p[siz],a[siz];
double dp[siz][siz][505],ans;

int main()
{
	//freopen("1.in","r",stdin);
	scanf("%d%d%d",&n,&L,&K);
	for (int i=0;i<n;i++)scanf("%d",&p[i]);
	for (int i=0;i<n;i++)scanf("%d",&a[i]);
	if (a[0]+K>250)dp[0][1][500]=p[0]/100.0;
	dp[0][1][250+a[0]+K]=p[0]/100.0;
	dp[0][0][250+K]=1.0-p[0]/100.0;
	for (int i=0;i<n-1;i++)
	for (int j=0;j<=i+1;j++)
	for (int k=0;k<=500;k++)
		{
			dp[i+1][j][k]+=dp[i][j][k]*(1.0-p[i+1]/100.0);
			if (k+a[i+1]>500) 
			dp[i+1][j+1][500]+=dp[i][j][k]*(p[i+1]/100.0); else
			dp[i+1][j+1][k+a[i+1]]+=dp[i][j][k]*(p[i+1]/100.0);
		}
	for (int i=L;i<=n;i++)
		for (int j=250;j<=500;j++)
			ans+=dp[n-1][i][j];
	printf("%.10lf",ans);
	return 0;
}
