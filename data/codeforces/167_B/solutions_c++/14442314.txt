#include <bits/stdc++.h>
using namespace std;
double dp[201][501][201],prob[200];int prize[200];
int main(){
	int n,x,l,k,i,j,a;
	cin>>n>>l>>k;
	for(int i=0;i<n;++i){
		cin>>a;prob[i]=(double)a/100;
	}
	for(int i=0;i<n;++i)
		cin>>prize[i];
	dp[0][200+k][0]=1;
	for(i=0;i<n;++i){
		for(x=0;x<=i;++x){
			for(j=0;j<=500;++j){
				int net= min(500,j+prize[i]);
				if(j+prize[i]>=0)
				dp[i+1][net][x+1] += dp[i][j][x]*prob[i];
				dp[i+1][j][x] +=dp[i][j][x]*(1-prob[i]);
			}
		}
	}
	double ret=0;
	for(i=200;i<=500;++i)
		for(j=l;j<=n;++j)
			ret+=dp[n][i][j];
	printf("%0.9lf\n",ret);
}
