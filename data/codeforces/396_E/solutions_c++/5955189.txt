typedef long long LL;
typedef pair<int, int> PII;

#define FILE freopen("../data.in","r",stdin)
#define REP(i,n) for(int i=0; i<(n); i++)
#define foreach(it,a) for(__typeof((a).begin()) it=(a).begin(); it!=(a).end(); it++)

typedef pair<LL, LL> PLL;

const int N = 1000000;

int pdiv[N];
LL a[N];
vector<PLL> ans;

int main() {
    memset(pdiv, 0, sizeof(pdiv));
    for (int i = 2; i < N; i++)
        if (!pdiv[i])
            for (int j = i; j < N; j += i)
                pdiv[j] = i;
    
    memset(a, 0, sizeof(a));
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        int p;
        scanf("%d", &p);
        cin >> a[p];
    }
    LL k;
    cin >> k;
    
    for (int x = 0; x < 20 && k > 0; x++) {
        for (int i = 2; i < N; i++)
            if (a[i]) {
                a[i]--;
                for (int j = i - 1; j > 1; j /= pdiv[j])
                    a[pdiv[j]]++;
            }
        k--;
    }
    
    for (int i = N - 1; i > 1; i--)
        if (a[i]) {
            LL d = min(k, a[i]);
            a[i] -= d;
            for (int j = i - 1; j > 1; j /= pdiv[j])
                a[pdiv[j]] += d;
        }
    
    for (int i = 2; i < N; i++)
        if (a[i])
            ans.push_back(make_pair(i, a[i]));
    printf("%u\n", ans.size());
    foreach(it, ans)
        cout << it->first << ' ' << it->second << endl;
    return 0;
}
