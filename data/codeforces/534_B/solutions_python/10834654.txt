temp = raw_input()
temp2 = raw_input()
temp = temp.split(" ")
temp2 = temp2.split(' ')

v1,v2 = temp
t, d = temp2

v1 = int(v1)
v2 = int(v2)
t = int(t)
d = int(d)

dist = [v1]
dist2 = [v2]

for i in xrange(1, t - 1):
	dist.append(v1 + i*d)
	dist2.append(v2 + i*d)
dist.append(v2)

for i in xrange(t - 1, 1, -1):
	if dist[i - 1] - dist[i] > d:

		while dist[i - 1] - dist[i] > d:
			dist[i - 1] -= 1
print sum(dist)
