import math
def solve(start, end, time, change):
	total = 0
	for i in range(time):
		total += min(start + i * change, end + change * (time - i - 1))
	return total

start1, end1 = map(int, raw_input().split(" "))
time1, change1 = map(int, raw_input().split(" "))
print solve(start1, end1, time1, change1)
