a,b = map(int,raw_input().split())
t,d = map(int,raw_input().split())
pre = [0]*t
suf = [0]*t
pre[0] = a
suf[t-1] = b
ans = a+b
i = 1
while i < t:
    pre[i] = pre[i-1]+d
    i += 1
i = t-2
while i >= 0:
    suf[i] = suf[i+1]+d
    i -= 1
for i in range(1,t-1):
    ans += min(pre[i],suf[i])
print ans
