v1, v2 = map(int, raw_input().split())
t, d = map(int, raw_input().split())
v3 = v1
v4 = v2
m = v1 + v2
t -= 2

while v3 < v2 and abs(v3 - v2) > d:
    t -= 1
    v3 += d
    m += v3

while v4 < v1 and abs(v4 - v1) > d:
    t -= 1
    v4 += d
    m += v4

while t >= 2:
    t -= 2
    v3 += d
    v4 += d
    m += v3 + v4

if t:
    m += min(v3, v4) + d
print m
