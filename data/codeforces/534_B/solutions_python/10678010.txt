v1, v2 = [int(v) for v in raw_input().split()]
t, d = [int(v) for v in raw_input().split()]

if d == 0:
	print t * v1
	exit()

if v1 > v2:
	v1, v2 = v2, v1

speeds = [0] * t
speeds[0] = v1
for tt in xrange(1, t):
	steps_left = t - tt - 1
	need_decelerate = speeds[tt - 1] - v2
	if need_decelerate + d < steps_left * d:
		speeds[tt] = speeds[tt - 1] + d
	else:
		speeds[tt] = v2 + steps_left * d

print sum(speeds)
