d = int(a[1])

maxlen = v1+t*d+1
for i in range(t+1):
    for j in range(maxlen):
        f[i][j] = -1000000000
f[0][v1] = v1
for i in range(1,t):
    for j in range(v1+i*d+1):
	for k in range(j-d,j+d+1):
	    if k >= 0 and k < maxlen:
		f[i][j] = max(f[i][j],f[i-1][k]+j)
ans = f[t-1][v2]
