v = map(int, raw_input().split(' '))
t = map(int, raw_input().split(' '))
p = [v[1] for i in range(t[0])]
p[0] = v[0]
for i in range(len(p)-2, 0, -1):
    p[i] = p[i+1] + t[1]
for i in range(0, len(p)-1):
    if p[i]-p[i+1] < -t[1] or \
            p[i]-p[i+1] > t[1]:
        p[i+1] = p[i] + t[1]
    else:
        break
print sum(p)
