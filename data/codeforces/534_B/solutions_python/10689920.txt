v1,v2 = map(int,raw_input().split(' '))
t,d = map(int,raw_input().split(' '))
if(v1 > v2):
    tmp = v1
    v1 = v2
    v2 =tmp
nv = v1
mv = v2 + (t - 1) * d
ans = 0
for i in range(t - 1):
    if(nv <= mv):
        ans += nv
        nv += d
        mv -= d
    else:
        nv = mv
        ans += nv
        mv -= d
ans += v2
print ans
