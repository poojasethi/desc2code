v1, v2 = map(int,raw_input().split())
t, d = map(int,raw_input().split())

l1 = []
l2 = []

for _ in range(t):
    l1.append(v1 + _*d)

for _ in range(t):
    l2.append(v2 + _*d)

l2.reverse()
res = []
for _ in range(t):
    if l1[_] > l2[_]:
        res+=l2[_:]
        break
    res.append(l1[_])

print sum(res)
