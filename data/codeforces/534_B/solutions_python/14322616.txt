
def main():
    V1, V2 = map(int, raw_input().split())
    T, D = map(int, raw_input().split())

    T -= 2
    dist = V1 + V2

    for i in range(1, T+1):
        dist += min(V1 + i*D, V2 + (T-i+1)*D)
        
    print dist
    
##########################################
##########################################
##########################################
if(__name__ == "__main__"):
    main()
