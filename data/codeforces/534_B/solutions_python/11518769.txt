v_start, v_end = map(int, raw_input().split())
t, d = map(int, raw_input().split())
speed = [v_start]*t
speed[-1] = v_end
for i in xrange(1,len(speed)-1):
    speed[i] = speed[i-1] + d
speed = speed[::-1]

for i in xrange(1, len(speed)-1):
    if abs(speed[i]-speed[i-1]) > d:
        speed[i] = speed[i-1] + d
    else:
        break
print sum(speed)
