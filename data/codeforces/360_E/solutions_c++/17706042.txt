#include <bits/stdc++.h>
using namespace std;

#define fru(j,n) for(int j=0; j<(n); ++j)
#define tr(it,v) for(typeof((v).begin()) it=(v).begin(); it!=(v).end(); ++it)
#define x first
#define y second
#define pb push_back
#define ALL(V) (V).begin(),(V).end()
#if 0
	#define DEB printf
#else
	#define DEB(...)
#endif

typedef long long ll;
typedef double D;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;
typedef vector<int> vi;

const ll inft = 1LL<<60;
const int MAXN = 1000006;//10^6

vector<pii> V[MAXN];
pii E[MAXN];
int n,m,k,ANS[MAXN];
int s[2],f;
ll dist[2][MAXN];
bool play(int add){
	DEB("play %d\n",add);
	fru(h,2)fru(i,n)dist[h][i]=inft;
	fru(h,2)dist[h][s[h]]=0;
	priority_queue<pair<ll,int> > Q;
	fru(h,2)Q.push(make_pair(-dist[h][s[h]],2*s[h]+h));
	while(!Q.empty()){
		pair<ll,int> u=Q.top();Q.pop();
		int g=u.y%2,v=u.y/2;
		if(-u.x!=dist[g][v])continue;
		DEB("mam %lld %d %d\n",-u.x,g,v+1);
		if(v==f)return dist[0][f]<dist[1][f]+add;	
		tr(it,V[v]){
			int cost;
			if(dist[0][v]<dist[1][v]+add)cost=E[it->y].x;
			else cost=E[it->y].y;
			DEB("ustalam %d na %d\n",it->y,cost);
			ANS[it->y]=cost;
			if(dist[g][it->x]>cost-u.x){
				dist[g][it->x]=cost-u.x;
				Q.push(make_pair(-dist[g][it->x],2*it->x+g));
				
			}
		}
	}	
}

void solve() {
	scanf("%d%d%d",&n,&m,&k);
	scanf("%d%d%d",&s[0],&s[1],&f);
	s[0]--;s[1]--;f--;
	int h=0;
	for(;h<m;h++){
		int a,b,c;
		scanf("%d%d%d",&a,&b,&c);a--;b--;
		E[h].x=E[h].y=c;
		V[a].pb(pii(b,h));
	}
	for(;h<m+k;h++){
		int a,b,c,d;
		scanf("%d%d%d%d",&a,&b,&c,&d);a--;b--;
		E[h].x=c;E[h].y=d;
		ANS[h]=c;
		V[a].pb(pii(b,h));
	}
	if(play(0)){
		puts("WIN");
		fru(i,k)printf("%d ",ANS[m+i]);puts("");
	}
	else if(play(1)){
		puts("DRAW");
		fru(i,k)printf("%d ",ANS[m+i]);puts("");
	}
	else puts("LOSE");
}

int main(){
	//freopen("input.in", "r", stdin);
	//freopen("output.out", "w", stdout);
	int t=1;//scanf("%d",&t);
	fru(i,t) solve();
	return 0;
}
