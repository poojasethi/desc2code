#include <cstdio>
using namespace std;

int n, m, k, s1, s2, t, a, b, c, d, len, i, ans[21000], till[21000], go[21000], next[21000], f1[21000], f2[21000], n1[1000000];
long long mi[21000];

void add(int a, int b, int c, int d) {
	next[++len] = till[a];
	till[a] = len;
	go[len] = b;
	ans[len] = c;
	f1[len] = c * 2;
	f2[len] = d * 2;
}

bool ok(int s1, int s2, int a) {
	int q, h, i;
	for (i = 1; i <= n; i++)	mi[i] = 1000000000000000000LL;
	mi[s1] = 1; mi[s2] = 0;
	for (q = 1, h = 2, n1[q] = s1, n1[h] = s2; q <= h; q++)
		for (i = till[n1[q]]; i; i = next[i]) {
			int tt = ((mi[n1[q]] & 1) == a) ? f1[i] : f2[i];
			ans[i] = tt / 2;
			if (mi[n1[q]] + tt < mi[go[i]]) {
				mi[go[i]] = mi[n1[q]] + tt;
				n1[++h] = go[i];
			}
		}
	return (mi[t] & 1) == a;
}

int main() {
	scanf("%d%d%d%d%d%d", &n, &m, &k, &s1, &s2, &t);
	while (m--) {
		scanf("%d%d%d", &a, &b, &c);
		add(a, b, c, c);
	}
	for (i = 1; i <= k; i++) {
		scanf("%d%d%d%d", &a, &b, &c, &d);
		add(a, b, c, d);
	}
	if (ok(s1, s2, 1)) {
		printf("WIN\n");
		for (i = len - k + 1; i <= len; i++)	printf("%d ", ans[i]);
	}else	if (ok(s2, s1, 0)) {
		printf("DRAW\n");
		for (i = len - k + 1; i <= len; i++)	printf("%d ", ans[i]);
	}else	printf("LOSE\n");
}
