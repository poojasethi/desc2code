rs=raw_input()
s=rs.split('@')
s[0]='x'+s[0]
n=rs.count('@')
ans=[]
for i in xrange(n):
    if len(s[i])>=2 and i!=n-1 and len(s[i+1])>=2:
        ans.append(s[i][1:]+'@'+s[i+1][0])
    elif len(s[i])>=2 and i==n-1 and len(s[i+1])>=1:
        ans.append(s[i][1:]+'@'+s[i+1])
    else:
        print 'No solution'
        exit()
if ans: 
    print ','.join(ans)
else: 
    print 'No solution'
