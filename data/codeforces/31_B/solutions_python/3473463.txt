import sys
import re
addr = re.compile("[a-z]+@[a-z]")
baddr = re.compile("[a-z]+@[a-z]+")

if __name__ == "__main__":
    s = raw_input()
    l = addr.findall(s)
    if len(l) > 0:
        l[-1] += addr.split(s)[-1]
        if baddr.findall(l[-1])[0] != l[-1]:
            print "No solution"
            sys.exit(0)
    if ''.join(l) != s:
        print "No solution"
    else:
        print ','.join(l)
