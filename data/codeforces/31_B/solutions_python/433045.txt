#!/usr/bin/env python
# -*- coding: utf-8 -*-

import re

def main():
    src = raw_input()
    lst = re.findall("([a-z]{1,}@[a-z]{1,}?)", src)
    if len(lst) > 0:
        lst[-1] = lst[-1][:lst[-1].find("@")] + src[src.rfind("@"):]
    print ",".join(lst) if src == "".join(lst) else "No solution"
    return 0

if __name__ == '__main__':
    main()
