import re
def readint(): return int(raw_input())
def readfloat(): return float(raw_input())
def readarray(foo): return [foo(x) for x in raw_input().split()]

def run():
	s = raw_input()
	if not re.match('^([a-z]+@[a-z]+(?!@))+$', s):
		print "No solution"
		return
	print ','.join(re.findall('[a-z]+@[a-z]+(?!@)', s))
run()
