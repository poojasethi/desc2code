beg = num = 0
res = []
s = raw_input()
for i,c in enumerate(s):
    if c =='@':
        end = i+2
        if s[beg:i].islower() and s[i+1:end].islower():
            res.append(s[beg:end])
            beg = end
            num += 1
        else: 
            print 'No solution'
            break
else:
    if res and num == len(res):
        res[-1] = res[-1] + s[beg:]
        print ','.join(res)
    else: print 'No solution'
