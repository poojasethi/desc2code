s = raw_input()
s = s.split('@')
ans = ''
f=1
for i in range(len(s)):
	if s[i]=='':
		f=0
		break
	if i==0:
		ans += s[i]
	elif i==len(s)-1:
		ans += '@'
		ans += s[i]
	else:
		ans += '@'
		if len(s[i])<=1:
			f=0
			break
		else:
			ans += s[i][0]+','+s[i][1:]
if f:
	if '@' not in ans:
		print 'No solution'
	else:
		print ans
else:
	print 'No solution'
