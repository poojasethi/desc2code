s=raw_input()
l=[i for i in range(len(s)) if s[i]=='@']
if not len(l):
    print 'No solution'
    exit()
if l[0] and l[-1]<len(s)-1:
    if len(l)==1:
        print s
        exit()
    for i in [l[i+1]-l[i] for i in range(len(l)-1)]:
        if i<=2:
            print 'No solution'
            exit()
    res=[s[:l[0]+2]]
    for i in range(len(l)-2):
        res.append(s[l[i]+2:l[i+1]+2])
    res.append(s[l[-2]+2:])
    print ','.join(res)
else:
    print 'No solution'
