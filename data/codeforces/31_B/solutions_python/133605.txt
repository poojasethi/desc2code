#!/usr/bin/python
import sys
import os
import math

def d(x1,y1,x2,y2):
    return math.sqrt((x1-x2)**2 + (y1-y2)**2)

def main():
    s = raw_input()
    if '@' not in s or s.startswith('@') or s.endswith('@'):
        print 'No solution'
        return
    ss=s.split('@')
    x=[ss[0]]
    y=[]
    for a in ss[1:-1]:
        if not a or len(a)==1:
            print 'No solution'
            return
        y.append(a[0])
        x.append(a[1:])
    y.append(ss[-1])
    print ','.join(['%s@%s' % (a,b) for a,b in zip(x,y)])


if __name__ == '__main__':
    main()
