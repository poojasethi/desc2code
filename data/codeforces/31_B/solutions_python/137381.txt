import re,sys
s=raw_input()
a=re.findall("[^@]*@?.?",s)
#print a
a=filter(lambda x:x!="",a)
if re.search("[^@]+@[^@]+",a[-1])==None:
  a[len(a)-2]+=a[len(a)-1]
  a.pop()
#print a

if a==[]:
  print "No solution"
  sys.exit()
  
for i in a:
  if re.search("^[^@]+@[^@]+$",i)==None:
    print "No solution"
    sys.exit()

for i in range(len(a)):
  if i!=len(a)-1:
    sys.stdout.write(str(a[i])+',')
  else:
    sys.stdout.write(str(a[i]))
