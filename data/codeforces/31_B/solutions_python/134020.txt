import sys

l = raw_input()

xs = []
dx=0
for i in l:
    if i=='@':
        xs.append(dx)
    dx+=1

if len(xs)==0 or xs[0] == 0 or xs[len(xs)-1] == dx-1:
    print 'No solution'
    sys.exit(0)

newlin = l[:xs[0]+2]

if len(xs)==1:
    print l
    sys.exit(0)

for i in range(len(xs)-1):
    if xs[i]+3 > xs[i+1]:
        print 'No solution'
        sys.exit(0)
    newlin+=','
    newlin+=l[(xs[i]+2):(xs[i+1]+2)]

newlin+= l[xs[i+1]+2:]
print newlin
