def solve():
    s=raw_input().split('@')
    slen = len(s)
    if slen < 2:
        print "No solution"
        return
    if slen >= 2:
        if len(s[0]) <= 0 or len(s[slen - 1]) <= 0:
            print "No solution"
            return
        for i in xrange(1, slen - 1):
            if len(s[i]) < 2:
                print "No solution"
                return
            else:
                s[i] = s[i][0] + ',' + s[i][1:]
    print '@'.join(s)

solve()
