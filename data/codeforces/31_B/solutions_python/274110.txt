import re
s = raw_input()
res = []
beg = 0
for i,c in enumerate(s):
    if c =='@':
        res.append(s[beg:i+2])
        beg = i+2
if res:
    res[-1] = res[-1] + s[beg:]
    res2 = []
    for s in res:
        if re.match(r'^[a-z]+@[a-z]|$',s):
            res2.append(s)
        else: 
            print 'No solution'
            break
    else:
        print ','.join(res2)
else:
    print 'No solution'
