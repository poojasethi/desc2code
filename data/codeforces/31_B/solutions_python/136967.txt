def ret(str):
  print str
  exit(0)


text = raw_input()
ans = []
while True:
  if not text:
    ret(','.join(ans))
  ind = text.find('@')
  if ind == 0 or ind == len(text) - 1 or ind < 0:
    ret('No solution')
  ind2 = text.find('@', ind + 1)
  if ind2 == -1:
    ans.append(text)
    ret(','.join(ans))
  if ind2 == ind + 1:
    ret('No solution')
  if ind2 < 0:
    t = len(text)
  else:
    t = ind + 2
  ans.append(text[:t])
  text = text[t:]
