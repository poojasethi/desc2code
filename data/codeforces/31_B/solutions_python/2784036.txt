#!/usr/bin/python
import sys

s = raw_input()
l = []
while len(s) > 0:
	pos = s.find('@')
	if pos != -1 and pos != 0 and pos != len(s)-1:
		nxt = s.find('@', pos+1)
		if nxt == -1:	# last one
			l.append(s)
			s = ''
		elif nxt > pos+1:
			l.append(s[0:pos+2])
			s = s[pos+2:]
		else:
			print "No solution"
			sys.exit(0)
	else:
		print "No solution"
		sys.exit(0)
sys.stdout.write("%s" % l[0])
for i in range(1, len(l)):
	sys.stdout.write(",%s" % l[i])
print
