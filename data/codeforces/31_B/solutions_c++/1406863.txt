#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
char d[250];
int a[250];
int main(){
    int i,x=0,n,j,k=0,say=0;
    gets(d);
    n=strlen(d);
    if(d[0]=='@' || d[n-1]=='@') { printf("No solution\n"); return 0; }
    for(i=0;i<n;i++)
        if(d[i]=='@') a[x++]=i;
    if(!x){ printf("No solution\n"); return 0; }
    for(i=0;i<x-1;i++)
        if(a[i+1]-a[i]<=2){ printf("No solution\n"); return 0; }
    for(j=0;j<n;j++){
            if(d[j-1]=='@' && j!=n-1 && say!=x-1){ printf("%c,",d[j]); say++;}
            else printf("%c",d[j]);
        }

 	return 0;
}
