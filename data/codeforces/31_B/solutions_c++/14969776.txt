#include<stdio.h>
#include<cmath>
#include<cstring>
int main()
{
	int num[11]={1,2,4,8,16,32,64,128,256,512,1024};
	int a[1010]={0},b[1010],i,j,n,t,k;
	char s[1010];
	bool tt;
	scanf("%s",s);
	n=strlen(s); tt=true;
	if (s[0]=='@' || s[n-1]=='@') tt=false;
	t=2; k=0;
	for (i=1;i<=n-2;i++)
	{
		if (s[i]=='@' && t<2) tt=false;
		if (s[i]=='@') {t=0; k++;} else t++;
	}
	if (k==0) tt=false;
	if (!tt) printf("No solution\n");
	else
	{
		printf("%c",s[0]);
		for (i=1;i<=n-1;i++)
		{
			printf("%c",s[i]);
			if (s[i-1]=='@' && k>=1) printf(","); 
			if (s[i]=='@') k--;
		}
		printf("\n");
	}
}
