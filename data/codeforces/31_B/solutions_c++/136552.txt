#include<iostream>
#include<string>

using namespace std;

int main()
{
	string s;
	int i,k,g;

	getline(cin,s);

	k=1;
	g=-1;
	for(i=0;i<s.length();i++)
		if(s[i]=='@')
		{
			g=i;
			if(k>1) k=0;
			else 
			{
				cout<<"No solution";
				return 0;
			}
		}
		else k++;
	
	if((g==-1)||(g==s.length()-1))  cout<<"No solution";
	else
		for(i=0;i<s.length();i++)
		{
			cout<<s[i];
			if(s[i]=='@')
			{
				cout<<s[i+1];
				if(i!=g) cout<<',';
				i++;
			}
		}


	return 0;
}
