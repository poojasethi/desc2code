#include<cstdio>
#include<cstring>
#include<cstdlib>
char s[210];
int n,i,f1,f2,m[210],c;
int main()
{
	gets(s);
	n=strlen(s);
	for(f1=1,f2=0,i=0,c=0,m[0]=0;i<n;i++)
	{
		m[i]=0;
		if(s[i] == '@')
		{
			if(!(f1) && !(f2))
			{
				f1=1;f2=1;c++;
				if(c>1) m[i-2]=1;
			}else{puts("No solution");return 0;}
		} 
		if(s[i] != '@')
		{
			if(f2) f2=0;else
			if(f1) f1=0;
		}
	}
	if (!(c)||(f2)){puts("No solution");return 0;}
	for(i=0;i<n;i++)
	{
		printf("%c",s[i]);
		if(m[i]) printf(",");
	}
	return 0;
}
