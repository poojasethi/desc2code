#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char s[500];
int t[500];
int main(void)
{
	scanf("%s", s);
	int k=0;
	for ( int i=0; i<strlen(s); i++ )
		if ( s[i] == '@' ) t[k++]=i;
	for ( int i=0; i<k-1; i++ )
		if ( t[i+1]-t[i]<3 ){ puts("No solution"); return 0;}
	if ( !k || !t[0] || t[k-1] == strlen(s)-1 ) {puts("No solution"); return 0;}
	int j=0,i;
	for ( i=0; i<k; i++ ){
		while ( j <= t[i]+1) putchar(s[j++]) ;
		if ( i!= k-1) putchar(',');
	}
	while ( j<strlen(s) ) putchar(s[j++]);
	puts("");
	return 0;
}
