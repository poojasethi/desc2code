#include <iostream>
#include <string>
using namespace std;

int main() {
	
	int i,j,k,c,l,prev;
	string s;
	cin>>s;
	
	l=(int)s.size();
	
	c=0;
	prev=-2;
	for(i=0;i<l;i++)
	if(s[i]=='@') {
		if(i-prev<3) {
			cout<<"No solution";
			return 0;
		}
		prev=i;
		c++;
	}
		if(i-prev<2||c==0) {
			cout<<"No solution";
			return 0;
		}
	
	for(k=i=0;i<l;i++) {
		cout<<s[i];
		if(s[i]=='@') {
			k++;
			if(k==c) {
				for(j=i+1;j<l;j++) cout<<s[j];
				break;
			}
			else cout<<s[++i]<<",";
		}
	}
	
	return 0;
}
