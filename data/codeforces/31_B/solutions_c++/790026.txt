#include<cstdio>
#include<cstring>
char s[300];
int main()
{
    int i,j,pre=-2;
    bool f=1;
    scanf("%s",s);
    int len=strlen(s);
    if(s[len-1]=='@') f=0;
    for(i=0;s[i];i++)
    {
        if(s[i]=='@')
        {
            if(i-pre<=2)
            {
                f=0;break;
            }
            else pre=i;
        }
    }
    if(f==0||pre==-2) puts("No solution");
    else
    {
        for(i=0;s[i];i++)
        {
            printf("%c",s[i]);
            if(i>=2&&s[i-1]=='@'&&i-1!=pre) printf(",");
        }
        puts("");
    }
    return 0;
}
