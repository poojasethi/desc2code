#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

#define MAXN 220

char line[MAXN];

int main() {
	int n,k,cont,ok;
	scanf(" %s",line);
	n = strlen(line); 
	
	k = cont = 0;	ok = 1;
	for (int i=0; i<n; i++) {
		if (line[i] == '@') {
			if (!k || i == n-1 || line[i+1] == '@') 
				{ ok = 0;	break;	}
			cont++;	k = 0;	i++;
		}
		
		else
			k++;
	}
	
	k = 0;
	if (n == 1 || n == 2 || !cont) ok = 0;
	if (!ok) printf("No solution\n");
	else {
		for (int i=0; i<n; i++) {
			printf("%c",line[i]);
			if (line[i] == '@') {
				if (k != cont-1) { printf("%c,",line[i+1]);	i++;	}
				k++;
			}
		}
		printf("\n");
	}
	
	return 0;
}
