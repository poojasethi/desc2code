#include<stdio.h>
#include<string.h>
int main()
{
	int i=0,k=0,n,a=0,b=0;
	char dizi[250],sonuc[250],sayac=0;
	scanf(" %s",dizi);
	n=strlen(dizi);
	if(dizi[0]=='@'){ printf("No solution\n"); return 0; }
	while(i<n)
	{	
		if(dizi[i]=='@')
		{
			sayac++;
			if(dizi[i+1]=='@' || dizi[i+2]=='@' || dizi[i+1]=='\0')
			{ 
				printf("No solution\n"); 
				return 0; 
			}
		}
		if(sayac>0 && i!=0 && dizi[i+1]=='@')sonuc[a++]=',';
		sonuc[a++]=dizi[i++];
	}
	if(sayac==0){ printf("No solution\n"); return 0; }
	for(i=0;i<a;i++)
		printf("%c",sonuc[i]);
	printf("\n");
	return 0;
}
