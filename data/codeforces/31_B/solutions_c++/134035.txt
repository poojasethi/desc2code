#include <cstdio>
#include <string.h>
#include <iostream>
using namespace std;

const int N = 500;
int a[N], k=0;
char s[N];

int main ()
{
	int n;
	scanf ("%s", s);
	n = strlen(s);
	for (int i=0; i<n; i++)
	{
		if (s[i] == '@') { cout << "No solution\n"; return 0; }
		a[k++] = i;
		for (i++; i<n-1 && s[i-1]!='@'; i++);
		if (k>1 && s[i-1]!='@')
			k--; // a@aa
		else if (s[i-1]!='@' || s[i]=='@') { cout << "No solution\n"; return 0; }
	}
	a[k] = n;
	for (int i=0; i<k; i++)
	{
		for (int j=a[i]; j<a[i+1]; j++)
			cout << s[j];
		if(i!=k-1) cout << ',';
	}
	cout << endl;
	return 0;
}
