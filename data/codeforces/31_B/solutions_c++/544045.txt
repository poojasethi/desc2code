#include <cstdio>
#include <cstring>

char s[222];

int main(){
	scanf("%s", s);
	int len = strlen(s);
	int last = - 2;
	bool noans = false;
	for (int i = 0; i < len; ++i)
		if (s[i] == '@'){
			if (i - last < 3) noans = true;
			last = i;
		}
	if (last == len - 1) noans = true;
	if (last < 0) noans = true;
	if (noans) {
		puts("No solution");
		return 0;
	}
	int ll = -4;
	for (int i = 0; i < len; ++i){
		if (i - ll == 2 && i < last && ll > 0) printf(",");
		putchar(s[i]);
		if (s[i] == '@') ll = i;
	}
	puts("");
	return 0;
}
