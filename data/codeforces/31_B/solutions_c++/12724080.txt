#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef vector<int> vi;
typedef pair<int,int> pii;
typedef vector <pair <int,int> > vpii;
int main(){
    string s ; cin >> s;
    int pre=-2;
    bool f = true;
    if(s[s.size()-1]=='@') f = false;
    for(int i=0 ; i < s.size() ; i++){
        if(s[i]=='@'){
            if(i-pre<=2){f = false ; break;}
            else pre=i;
        }
    }
    if(f==0||pre==-2){cout << "No solution"; return 0;}
    for(int i=0; i < s.size();i++){
        cout << s[i];
        if(i>=2 && s[i-1]=='@' && i-1!=pre) cout << ',';
    }
}
