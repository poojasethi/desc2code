#include<cstdio>
#include<cstring>
char str[205],ans[410];
int main(){
    char *p;
    int flag=1,n,m=0;
    scanf("%s",str);n=strlen(str);
    if(str[0]=='@'||str[n-1]=='@'||strchr(str,'@')==NULL)flag=0;
    else{
        for(int i=0;i<n;i++){
            ans[m++]=str[i];p=str+i;
            if(flag==0&&i!=n-1&&strchr(p,'@')!=NULL){ans[m++]=',';flag=1;}
            if(str[i]=='@')flag=0;
        }
        p=strchr(ans,',');flag=1;
        while(p!=NULL){
        if(*(p-1)=='@'||*(p+1)=='@'){flag=0;break;}p=strchr(p+1,',');}
    }
    if(flag==0)printf("No solution\n");
    else printf("%s\n",ans);
    return 0;
}
