#include <iostream>
#include <string>

using namespace std;

int main() {
    string s, ans="";    
    cin >> s;
    int p=0,i=0;
    while (i < s.size()) {
        if (s[i]!='@') i++;
        else {            
            if ((i==p)||(i == s.size()-1) || (s[i+1]=='@')) {
                cout<<"No solution";
                return 0;
            } else {
                ans += "," + s.substr(p, i-p+2);
                i += 2;
                p = i;
            }
        }
    }
    if (ans=="") {
        cout<<"No solution";
        return 0;
    }
    ans+=s.substr(p);
    cout << ans.substr(1);
    return 0;
}
