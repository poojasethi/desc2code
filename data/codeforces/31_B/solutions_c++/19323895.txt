#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll i,j,m,n,p,q,r,l;
string s,t;
int main()
{
    cin>>s;l=s.size();
    for(i=0;i<l;i++)
    {
        if(s[i]=='@')
        {
            if(i==0||s[i+1]=='@'||s[i+2]=='@'||i==l-1)
            {
                cout<<"No solution";
                return 0;
            }
        }
    }
    for(i=0;i<l;i++)
    {
        t+=s[i];
        if(s[i]=='@')p=1;
        if(s[i+2]=='@'&&p==1)
            t+=',';
    }
    if(p==0)
        cout<<"No solution";
    else
        cout<<t;
}
