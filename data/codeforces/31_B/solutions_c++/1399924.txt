#include<stdio.h>
#include<string.h>
int main()
{
	int i,j,a,b,c,d,e,f=0,z=0;
	char A[100000];
	scanf("%s",A);
	a=strlen(A);
	for(i=0;i<a;i++)
		if(A[i]=='@')
			f++;
	if(f==1 && A[1]=='@' && a==3)
	{
		printf("%s",A);return 0;
	}
	if(A[a-1]=='@' || f>=(a/2))
	{
		printf("No solution");
		return 0;
	}
	for(i=0;i<a;i++)
	{
		if(!f || A[0]=='@' || (A[i]=='@' && A[i+2]=='@') || (A[i]=='@' && A[i+1]=='@'))
		{
			printf("No solution");
			return 0;
		}
		if(A[i]=='@' && z!=(f-1))
		{
			for(j=a;j>=i+2;j--)
				A[j]=A[j-1];
			a++;
			A[i+2]=',';
			z++;
		}

	}
	printf("%s",A);
	return 0;
}
