#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
string s;
cin>>s;
int i,pos;
int flag=0;
int flag2=0;
for(i=0;i<s.size();i++)
{
if(s[i]=='@')
{flag2=1;
if(i==0 || i==s.size()-1 || s[i+1]=='@' || s[i-2]=='@' || s[i-1]==',')
{flag=1;
break;
}
if(i+2!=s.size() )
{
pos=s.find_last_of("@");
if(pos!=i)
s.insert(i+2,",");
}
}
}
if(flag==0 && flag2==1)
cout<<s;
else
cout<<"No solution";
return 0;
}
