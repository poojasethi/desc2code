#include <iostream>
#include <string>
using namespace std;
int main()
{
	string s, t;
	int i, p, q;
	cin>>s;
    while (s.size()>0) {
		p=s.find('@');
		if (p<1||p==s.size()-1) {
			t="No solution"; break;
		}
		else {
			q = s.find('@', p+1);
			if (q==-1) {
				t+=s; break;
			}
			else if (q-p>=2){
				t+=s.substr(0,p+2)+",";
				s=s.substr(p+2,200);
			}
			else {
				t="No solution"; break;
			}
		}
    }
    cout<<t<<endl;
	return 0;
}
