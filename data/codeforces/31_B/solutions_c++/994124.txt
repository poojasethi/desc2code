#include <iostream>

#define rep(i,n) for(int i=0;i<n;i++)
#define f() cout<<"No solution"<<endl
#define pr(i) cout<<i<<endl

using namespace std;

string in;
int c[210];
int p=0,a=0,s=0;

int main(){

  cin>>in;s=in.size();
  if(in[0]=='@'||in[s-1]=='@'){f();return 0;}
  rep(i,s)if(in[i]=='@')c[p++]=i;
  if(p<=0){f();return 0;}
  rep(i,p-1)if(c[i+1]-c[i]<=2){f();return 0;}
  for(int i=0;i<p-1;i++){
    cout<<in.substr(a,c[i]+2-a)<<",";
    a=c[i]+2;
  }
  pr(in.substr(a));
  return 0;

}
