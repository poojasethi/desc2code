#include <iostream>
#include <vector>
#include <string>

using namespace std;

int main()
{
	string s;
	cin >> s;
	size_t pos = 0;
	string r;
	string comma;
	bool valid = false;

	while (pos != s.size()) {
		size_t at_pos = s.find('@', pos);
		if (at_pos == string::npos) {
			if (valid)
				r += s.substr(pos);
			else
				r = "No solution";
			break;
		}
		if (at_pos == pos || at_pos + 1 == s.size() || s[at_pos + 1] == '@') {
			r = "No solution";
			break;
		}
		r += comma + s.substr(pos, at_pos - pos + 2);
		comma = ",";
		pos = at_pos + 2;
		valid = true;
	}
	cout << r << endl;
}
