#include <iostream>
#include<string>
using namespace std;

int main()
{
    string s,s1,p=",";
    cin>>s;
    s1=s;int f=0,f1=0,f2=0;
    for(int i=0;s1[i];i++){
      if(s1[i]=='@'){f1++;
          if(i==0||i==s1.length()-1){cout<<"No solution";goto em;}
        for(int j=i+1;j<s1.length();j++){
          if(s1[j]=='@'){
            if(s1[i+2]!='@'&&s1[i+1]!='@'){s.insert(i+2+(f++),p);break;}
            else {cout<<"No solution";goto em;}
          }
        }
      }
      else f2++;
    }
    if(f1==0||f2==0){cout<<"No solution";goto em;}
    cout<<s;
    em:
    return 0;
}
