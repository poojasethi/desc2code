#include<stdio.h>
#include<string.h>
#define N 260
int n,m,t;
char s[N];
int main()
{
	int i,j,k;
	int flag=0;
	scanf("%s",s);
	if(s[0]=='@'||s[strlen(s)-1]=='@')
		flag=1;
	k=0;
	for(i=1;s[i];i++)
	{
		if(s[i]=='@')
			k=1;
		if(s[i]=='@'&&(s[i-1]=='@'||i>1&&s[i-2]=='@'))
		{
			flag=1;
			break;
		}
	}
	if(flag||k==0)
	{
		printf("No solution\n");
		return 0;
	}
	j=0;
	for(i=0;s[i];i++)
	{
		printf("%c",s[i]);
		if(s[i]=='@')
			j=1;
		if(j&&s[i+2]=='@')
			printf(",");
	}
	printf("\n");
}
