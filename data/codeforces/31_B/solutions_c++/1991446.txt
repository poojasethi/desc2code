#include <iostream>
#include<string>
#include<vector>
using namespace std;

int main()
{
  string s;cin>>s;int p=-2,l=1000000000,n=s.size(),i;

  for(i=0;i<n;i++)
  {
      if(s[i]=='@')
      {
          if(l>i-p) l=i-p;p=i;
      }
  }
  if(n+1-p<l) l=n+1-p;
  if(p==-2||l<3) {cout<<"No solution";return 0;}
  for(i=0;i<n;i++)
  {
      cout<<s[i];
      if(i>0&&i<n-1&&s[i-1]=='@'&&i-1!=p) cout<<',';
  }
    return 0;
}
