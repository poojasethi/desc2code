#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int i,l,a[200],k,j;
	char c[200],s[500];
	cin>>c;
	l=strlen(c);
	k=0;
	for(i=0;i<l;i++)
	{
		if(c[i]=='@')
		{
			a[k]=i;
			k++;
		}
	}
	int ans=1;
	
	if(a[0]==0 || a[k-1]==(l-1) || k==0)
		ans=0;
	
	for(i=0;i<k-1;i++)
	{
		if(a[i+1]-a[i]<=2)
			ans=0;
	}
	if(ans==0)
		cout<<"No solution"<<endl;
	else
	{
		j=0;
		for(i=0;i<l;i++)
		{
			cout<<c[i];
			if(i==a[j]+1 && j<k-1)
			{
				cout<<",";
				j++;
			}
		}
		cout<<endl;
	}
	return 0;
}
