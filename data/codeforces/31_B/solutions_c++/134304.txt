#include<iostream>
using namespace std;
int main()
{
string s,k[205];
int i,j,n=0;
cin>>s;
for(i=0;i<s.length();i++)
{
if(s[i]=='@')
{
cout<<"No solution";
return 0;
}
j=i;
while(s[j]!='@' && j<s.length())
{
k[n]+=s[j];
j++;
}
if(j+1>=s.length() || s[j+2]=='@')
{
cout<<"No solution";
return 0;
}
while(s[j+1]!='@' && j<s.length())
{
k[n]+=s[j];
j++;
}
i=j-1;
n++;
}
if(n>0)
{
for(i=0;i<n-1;i++)
{
cout<<k[i]<<',';
}
cout<<k[n-1];
}
return 0;
}
