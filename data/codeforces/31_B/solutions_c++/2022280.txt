#include <cstdio>
#include <cstring>


int main()
{
	int arrobas = 0;
	char st[201];
	gets(st);
	int tam = strlen(st);
	for(int i = 0; i < tam; i++)
		if(st[i] == '@')
			arrobas++;
	if(st[0] == '@' || st[tam - 1] == '@' || !arrobas)
	{
		printf("No solution");
		return 0;
	}
	for(int i = 0; i < tam - 1; i++)
	
		if(st[i] == '@' && (st[i + 1] == '@' || st[i + 2] == '@'))
		{
			printf("No solution");
			return 0;
		}
	
	int i;
	for( i = 0; st[i] != '@'; i++)
		printf("%c", st[i]);
	printf("@");
	for(i++; i < tam; i++)
		if(st[i + 1] == '@')
			printf(",%c", st[i]);
		else
			printf("%c", st[i]);
	return 0;
}
