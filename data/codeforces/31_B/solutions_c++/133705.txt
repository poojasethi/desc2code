#include<stdio.h>
#include<string.h>
#include<stdlib.h>

char str[1024];
int id[1024],cnt=0;
int solve(){
	int ptr=0;
	while(str[ptr]!=0){
		if(str[ptr]=='@')return 0;
		int p=ptr;
		while(str[p]!='@'&&str[p]!=0)++p;
		if(str[p]==0){
			if(cnt==0)return 0;
			else break;
		}
		++p;
		if(str[p]=='@'||str[p]==0)return 0;
		id[cnt++]=p;
		ptr=p+1;	
	}
	for(int i=0,j=0;str[i];++i){
		printf("%c",str[i]);
		if(i==id[j]&&j!=cnt-1){
			printf(",");
			++j;
		}
	}
	return 1;
}


int main(){
	scanf("%s",str);
	if(solve()==0)
		printf("No solution\n");
}
