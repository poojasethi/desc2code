#include<iostream>
#include<vector>
using namespace std;
int main()
{
	string s;
	vector<int> a;
	cin>>s;
	for(int i=0;i<s.size();i++)
		if(s[i]=='@')a.push_back(i);
	if(a.size()==0||a[0]==0||a[a.size()-1]==s.size()-1)
	{
		cout<<"No solution";
		return 0;
	}
	if(a.size()==1)
	{
		cout<<s;
		return 0;
	}
	for(int i=1;i<a.size();i++)if(a[i]-a[i-1]<=2)
	{
		cout<<"No solution";
		return 0;
	}
	for(int i=0,now=0;i<a.size()-1;i++)
	{
		while((now)<a[i]+2)cout<<s[now++];
		cout<<",";
	}
	for(int i=a[a.size()-2]+2;i<s.size();i++)cout<<s[i];
	return 0;
}
