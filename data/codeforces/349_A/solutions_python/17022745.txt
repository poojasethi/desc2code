n = int(raw_input())
a = map(int, raw_input().split())

x=0
y=0
for i in xrange(len(a)):
    if a[i]==50:
        if x==0:
            print "NO"
            break
        else:
            x-=1
            y+=1
    elif a[i]==100:
        if x>=1 and y>=1:
            x-=1
            y-=1
        elif x>=3:
            x-=3
        else:
            print "NO"
            break
    else:
        x+=1
    if i==n-1:
        print "YES"
