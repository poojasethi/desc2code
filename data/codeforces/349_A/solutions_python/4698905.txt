n = int(raw_input())
s =  map(int, raw_input().split())
a=b=0
for x in s:
    if x == 25:
        a += 1
    elif x == 50:
        a -= 1
        b += 1
    else:
        if a and b:
            a -= 1
            b -= 1
        else:
            a -= 3
    if a < 0:
        print "NO"
        break
else:
    print "YES"
