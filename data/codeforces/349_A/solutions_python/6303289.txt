B = map(int, raw_input().split())

left = [0, 0]
for b in B:
  if b == 25:
    left[0] += 1
  elif b == 50:
    if left[0] == 0:
      print 'NO'
      exit()
    else:
      left[0] -= 1
      left[1] += 1
  else:
    if left[1] > 0 and left[0] > 0:
      left[0] -= 1
      left[1] -= 1
    elif left[0] > 2:
      left[0] -= 3
    else:
      print 'NO'
      exit()
print 'YES'
