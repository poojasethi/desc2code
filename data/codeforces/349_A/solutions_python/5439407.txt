#!/usr/bin/python

n=int(raw_input())
l=map(int,raw_input().split())
c25,c50=0,0
for i in range(n):
	if l[i]==25:
		c25+=1
	elif l[i]==50:
		if c25>0:
			c25-=1
			c50+=1
		else:
			print 'NO'
			break
	else:
		if c50>0 and c25>0:
			c50-=1
			c25-=1
		elif c25>=3:
			c25-=3
		else:
			print 'NO'
			break
else:
	print 'YES'
