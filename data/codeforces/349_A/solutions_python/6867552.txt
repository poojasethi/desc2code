def main():
    n, a = input(), map(int, raw_input().split())
    b, c = 0, 0
    for i in a:
        if i == 25:
            b += 1
        elif i == 50 and b:
            b, c = b-1, c+1
        elif i == 100 and (b >= 3 or b and c):
            if b and c:
                b, c = b-1, c-1
            else:
                b -= 3
        else:
            return "NO"
    return "YES"
print main()
