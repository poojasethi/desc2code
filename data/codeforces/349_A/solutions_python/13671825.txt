change = 0
op = 'YES'
deno = {25 : 0, 50 : 0}
n = input()
x = raw_input()
x = x.split()

for i in range(n):
	x[i] = int(x[i])
	
	if(x[i] == 25):
		deno[25] += 1
	
	elif(x[i] == 50 and deno[25] > 0):
		deno[25] -= 1
		deno[50] += 1
	
	elif(x[i] == 100 and deno[25] > 0 and deno[50] > 0):
		deno[50] -= 1
		deno[25] -= 1
	
	elif(x[i] == 100 and deno[25] >= 3):
		deno[25] -= 3
	
	else:
		op = 'NO'
		break
		
print op
