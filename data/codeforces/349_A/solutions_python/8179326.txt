def solve(b):
  r25, r50 = 0, 0
  for x in b:
    if x == 25:
      r25 += 1
    elif x == 50:
      r50 += 1
      if r25 > 0:
        r25 -= 1
      else:
        return False
    elif x == 100:
      if r50 > 0 and r25 > 0:
        r50, r25 = r50-1, r25-1
      elif r25 > 2:
        r25 -= 3
      else:
        return False
  return True

n = int(raw_input())
b = map(int, raw_input().split())
print 'YES' if solve(b) else 'NO'
