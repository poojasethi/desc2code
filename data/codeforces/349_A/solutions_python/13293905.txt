n = input()
l = map(int, raw_input().split())
a = b = c = 0
ans = "YES"
for i in range(n):
    if(l[i] == 25):
        a += 1
    elif(l[i] == 50):
        b += 1
        if a <= 0:
            ans = "NO"
            break
        a -= 1
    else:
        c += 1
        if b > 0 and a > 0:
            a -= 1
            b -= 1
        elif a > 2:
            a -= 3
        else:
            ans = "NO"
            break
print ans
