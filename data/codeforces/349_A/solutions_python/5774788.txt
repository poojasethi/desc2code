x=int(raw_input())
money=map(int,raw_input().split(' '))
money25=0
money50=0
result='YES'
for i in range(x):
    if money[i]==25:
        money25+=1
    if money[i]==50:
        money25-=1
        money50+=1
    if money[i]==100:
        if money50==0:
            money25-=3
        else:
            money50-=1
            money25-=1
    if money25<0 or money50<0:
        result='NO'
print result
