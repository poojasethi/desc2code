import math
n = int(raw_input())
list = []
list[len(list):] = map(int, raw_input().split())

change = 0
largebills = 0
for x in range (0,n):
	if list[x] == 25:
		change += 1
	if list[x] == 50:
		change -= 1
		largebills += 1
	if list[x] == 100:
		change -= 3
		if largebills > 0:
			largebills -= 1
			change += 2
	if change < 0:
		print "NO"
		quit()
print "YES"
