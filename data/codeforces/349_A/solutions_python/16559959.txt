x = map(int, raw_input().split())
a, b = 0, 0
flag = True
for i in x:
	if i == 25:
		a = a + 1
	elif i == 50:
		if a > 0:
			a = a - 1
			b = b + 1
		else:
			print 'NO'
			flag = False
			break
	else:
		if b > 0 < a:
			a = a - 1
			b = b - 1
		elif a >= 3:
			a = a - 3
		else:
			print 'NO'
			flag = False
			break
if flag:
	print 'YES'
