n = int(raw_input())
a = map(int,raw_input().split())
r25, r50, r100 = 0,0,0
flag=0
for i in xrange(n):
	if a[i] == 25:
		r25+=1
		continue
	elif a[i] == 50:
		r50+=1
		if r25>0:
			r25-=1
		else:
			flag=1
			print "NO"
			break
	else:
		r100+=1
		if r50>0 and r25>0:
			r50-=1
			r25-=1
		elif r25>=3:
			r25-=3
		else:
			flag=1
			print "NO"
			break
if flag==0:
	print "YES"
