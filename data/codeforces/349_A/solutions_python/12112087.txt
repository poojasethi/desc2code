from collections import defaultdict
n=int(raw_input())
a=map(int, raw_input().split())
c=defaultdict(int)
hit=False
for k in a: 
	c[k]+=1
	if k==50: 
		if c[25]>0:c[25]-=1
		else:hit=True
	elif k==100:
		if c[50]>0 and c[25]>0: 
				c[50]-=1
				c[25]-=1
		elif c[25]>2:c[25]-=3
		else: hit=True
if hit:print"NO"
else:print"YES"
