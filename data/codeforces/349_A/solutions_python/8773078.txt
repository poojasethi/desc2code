s25 = s50 = 0
flag = 1
n = input()
temp = map(int, raw_input().split())
for i in temp:
	if(i==25):
		s25+=1
	elif(i==50):
		if(s25>=1):
			s25-=1
			s50+=1
		else:
			flag = 0
			break
	elif(i==100):
		if(s50>=1)and(s25>=1):
			s50-=1
			s25-=1
		elif(s50==0)and(s25>=3):	
			s25-=3
		else:
			flag = 0
			break
if(flag==1):
	print 'YES'
else:
	print 'NO'
