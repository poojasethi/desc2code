n = input()
s = map(int,raw_input().split())

d = {25:0,50:0,100:0}
ans = 'YES'
for i in range(n):
	if (s[i] == 100):
		d[100]+=1
		if ( d[50] > 0 ) and (d[25]>0):
			d[50]-=1
			d[25]-=1
		elif (d[50] == 0) and (d[25]>2):
			d[25]-=3
		else:
			ans = 'NO'
			break
	elif (s[i] == 50):
		d[50]+=1
		if (d[25]>0):
			d[25]-=1
		else:
			ans = 'NO'
	else:
		d[25]+=1
		
print ans
