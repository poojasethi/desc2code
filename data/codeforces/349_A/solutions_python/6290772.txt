n = input()
coins = [0, 0]
ok = True
for bill in map(int, raw_input().split()):
    if bill == 25:
        coins[0] += 1
    elif bill == 50:
        coins[0] -= 1
        coins[1] += 1
    else:
        if coins[1] > 0:
            coins[0] -= 1
            coins[1] -= 1
        else:
            coins[0] -= 3
    if min(coins) < 0:
        ok = False
print ['NO', 'YES'][ok]
