input()
cnt = {25 : 0, 50 : 0, 100 : 0}
for x in map(int, raw_input().split()):
    cnt[x] += 1
    if x == 50:
        cnt[25] -= 1
    elif x == 100:
        cnt[25] -= 1
        if cnt[50]:
            cnt[50] -= 1
        else:
            cnt[25] -= 2
    if min(cnt.values()) < 0:
        print 'NO'
        exit()
print 'YES'
