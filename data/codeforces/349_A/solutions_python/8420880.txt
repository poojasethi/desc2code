n = input()
line = map(int, raw_input().split())
line = [i/50 for i in line]
bank = [0]*3

ok = 1
for i in line:
    bank[i] += 1
    bank[0] -= (i>0)
    if i==2:
        if bank[1] < 1:
            bank[0] -= 2
        else:
            bank[1] -= 1
    if bank[0] < 0:
        ok = 0
print "YES" if ok else "NO"
