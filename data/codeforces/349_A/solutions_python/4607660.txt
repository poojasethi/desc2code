input()
r=map(int, raw_input().split())
q=0
p=0
for i in r:
    if i == 25:
        q += 1
    if i == 50:
        if not q:
            print "NO"
            exit()
        q -= 1
        p += 1
    if i == 100:
        if not p and q < 3 or p and q < 1:
            print "NO"
            exit()
        if p:
            p -= 1
            q -= 1
        else:
            q -= 3
print "YES"
