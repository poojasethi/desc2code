def read_ints():
    return map(int, raw_input().split(' '))


n, = read_ints()
c25 = 0
c50 = 0
fail = False
for bill in read_ints():
    if bill == 25:
        c25 += 1
    if bill == 50:
        c50 += 1
        c25 -= 1
    if bill == 100 and c50 > 0:
        c50 -= 1
        c25 -= 1
    elif bill == 100 and c50 == 0:
        c25 -= 3
    if c25 < 0:
        fail = True
print ['YES', 'NO'][fail]
