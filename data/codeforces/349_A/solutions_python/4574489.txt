def getInt():
  return int(raw_input())

def getInts():
  return map(int, raw_input().split())

n = getInt()
a = getInts()
c = [0, 0]
ans = True

for i in a:
  if i == 25:
    c[0] += 1
  elif i == 50:
    c[0] -= 1
    c[1] += 1
  else:
    if c[1] >= 1:
      c[1] -= 1
      c[0] -= 1
    else:
      c[0] -= 3
  if c[0] < 0 or c[1] < 0:
    ans = False
    break

if ans:
  print "YES"
else:
  print "NO"
