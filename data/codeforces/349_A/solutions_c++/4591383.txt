#include<iostream>
using namespace std;
int main()
{
long int n,*a,i,c=0,k=0;
cin>>n;
a=new long int[n];
for(i=0;i<n;i++)
cin>>a[i];
for(i=0;i<n;i++)
{
if(a[i]==25)
c++;
else if(a[i]==50)
{
k++;
c--;
}
else
{
if(k==0)
c-=3;
else
{
k--;
c--;
}
}
if(c<0)
break;
}
if(i==n)
cout<<"YES\n";
else
cout<<"NO\n";
return 0;
}
