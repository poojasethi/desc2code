#include <cstdio>

using namespace std;

int n,x,a[123];

int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;++i){
		scanf("%d",&x);
		int y=x-25;
		if(!y){
			a[x]++;
		}else{
			for(int j=y;j && y;--j)
			while(a[j] && y>=j)a[j]--,y-=j;
			a[x]++;
			if(y){
				puts("NO");return 0;
			}
		}
	}puts("YES");
}
