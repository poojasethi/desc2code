#include<cstdio>
int main()
{
	int n;
	scanf("%d",&n);
	int a,a50=0,a100=0,a25=0;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a);
		if(a==25) a25++;
		else if(a==50 && a25>0) {a50++;a25--;}
		else if(a==100 && a50>0 && a25>0){a50--;a25--; }
		else if(a==100 && a25>2) {a25-=3;}
		else{printf("NO\n");return 0;}
	}
	printf("YES\n");
	return 0;
}
