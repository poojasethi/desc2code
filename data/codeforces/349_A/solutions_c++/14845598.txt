#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,tf=0,ft=0,i,m,f=1;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>m;
		if(m==25)
			tf++;
        else if(m==50)
			tf?tf--,ft++:f=0;
		else
		{
			if(ft&&tf)
				tf--,ft--;
			else if(tf>=3)
				tf-=3;
			else
				f=0;
		}
	}
    f?cout<<"YES"<<endl:cout<<"NO"<<endl;
}
