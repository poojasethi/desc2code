#include <stdio.h>
int main(int argc, char *argv[])
{
	int a[100000],n,h=0,k=0,m=0;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
		if(a[i]==25)
			h++;
		if(a[i]==50){
			h--;k++;}
		if(a[i]==100){
			if(k==0)
				h-=3;
			if(k>0){
				h--;k--;}
		}
		if(h<0){
			printf("NO");
			return 0;}
			
		}
	printf("YES");
}
