#include <iostream>
using namespace std;
int n;
int a=0,b=0,q;
int main()
{
    cin >> n;
    for (int i=0;i<n;i++)
    {
    	cin >>q;
    	if (q==25) {a++;}
    	if (q==50) {a--;b++;}
    	if (q==100) {if (b!=0) {b--;a--;} else a-=3;}
    	if (a<0||b<0) {cout << "NO";return 0;}
    }
cout << "YES";
return 0;
}
