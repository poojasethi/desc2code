#include <cstdio>
int N,ok=1;
int main()
{
    int x,a=0,b=0;
    scanf("%d",&N);
    for (int i=1; ok&&i<=N; i++)
    {
        scanf("%d",&x);
        if (x==25) a++;
        else if (x==50)
            if (a) a--,b++; else ok=0;
        else
            if (a>=1&&b>=1) a--,b--; else if (a>=3) a-=3; else ok=0;
    }
    puts(ok?"YES":"NO");
}
