#include<cstdio>

using namespace std;
int a,b,c,n,d;
int main()
{
  scanf("%d",&n);
  for(int i=0;i<n;i++)
    {
      scanf("%d",&d);
      if(d==25) a++;
      else if(d==50) b++;
      else c++;
      d/=25;
      d--;
      if(d>=2 and b>0){b--;d-=2;}
      if(d>0 and d<=a ){a-=d;d=0;}
      if(d!=0) {printf("NO\n");return 0;}
    }
  printf("YES\n");
}
