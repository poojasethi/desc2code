#include<iostream>
using namespace std;
int main()
{
	int n,y25=0,y50=0,a;cin>>n;
    while(n--)
    {
    	cin>>a;if(a==50)y25--;
    	if(a==100){if(y50){y50--;y25--;}else y25-=3;}
    	if(y25<0||y50<0)
    	{
    		cout<<"NO"<<endl;
    		return 0;
    	}
    	if(a==25)y25++;if(a==50)y50++;
    }
    cout<<"YES"<<endl;
}
