#include<iostream>
#include<stdlib.h>
using namespace std;
int main(void)
{
	int a,b,i,tf,fif,n;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a;
		if(a==25)
		tf++;
		else if(a==50)
		fif++,tf--;
		else
		{
			if(fif)
			fif--,tf--;
			else
			tf-=3;
		}
		if(tf<0||fif<0)
		{
			cout<<"NO";
			exit(0);
		}
	}
	cout<<"YES";
	return 0;
}
