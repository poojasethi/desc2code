#include <iostream>
#include <string>
#include <vector> 
using namespace std;
vector<int>v;
int f(int n)
{
	if(v[n] > 8)
	return v[n]-8;
	if(n==v.size()-1)
	return v[n];
	return 1;
}
int main()
{
	string a;
	while(cin>>a)
	{
		bool b=1;
		int k=0;
		v.clear();
		int p=0;
		for(int i=0;i<a.size();i++)
		{
			if(a[i]=='.')
			{
				p++;
				if(v.size())
				if(k < 2 || k > 11)
				{
					b=0;
				}
				if(!v.size())
				if(k > 8 || k==0)
				b=0;
				v.push_back(k);
				k=0;
			}
			else
			k++;
			if(b==0)
			break;
		}
		if(k > 3 || k==0)
		b=0;
		v.push_back(k);
		if(b==0 || p==0)
		cout<<"NO"<<endl;
		else
		{
			cout<<"YES"<<endl;
			int t=0;
			int con=0;
			int g=0;
			for(int i=0;i<a.size();i++)
			{
				if(!t && a[i]!='.')
				{
					cout<<a[i];
				}
				if(a[i]=='.')
				{
					t=1;
					con++;
					cout<<a[i];
				}
				else if(t==1)
				{
					g=f(con);
					for(int j=i;j<i+g;j++)
					cout<<a[j];
					cout<<endl;
					t=0;
					i=i+g-1;
				}
			}
		}
	}
	return 0;
}
