#include<iostream>
#include <cstring>
using namespace std;

const int
	MaxN = 400005;

char S[MaxN];
bool valid[MaxN];
int lastDot;
int last[MaxN];

void FindSolution(int n)
{
	if (n)
	{
		FindSolution(last[n] - 1);
		for (int i = last[n]; i <= n; i++)
			cout << S[i];
		cout << endl;
	}
}

int main()
{

	cin >> (S + 1);

	valid[ 0 ] = true;

	int n = strlen(S + 1);
	for (int i = 1; i <= n; i++)
		if (S[ i ] == '.')
			 lastDot = i;
		else
		{
			if (!lastDot || i - lastDot > 3) continue;
			for (int j = 1; 1 <= lastDot - j && j <= 8; j++)
			{
				if (S[lastDot-j] == '.') break;
				if (valid[lastDot-j-1])
				{
					valid[i] = true; last[i] = lastDot - j;
				}
			}
		}

	if (!valid[ n ])
	{
		cout << "NO\n";
		return 0;
	}

	cout << "YES\n";
	FindSolution(n);

	return 0;
}
