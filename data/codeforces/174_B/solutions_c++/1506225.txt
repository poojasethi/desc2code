#include <iostream>
#include <vector>
#define pb push_back
using namespace std;
string s;
vector<int> v;
vector<int> ans;
bool OK = 1;
int main()
{
	cin >> s;
	for(int i=0;i<s.size();i++)
		if(s[i] == '.') v.pb(i);
	if(v.empty() == 1) OK = 0;
	else
	{
		if(v[0] > 8) OK = 0;
		if(v[v.size()-1] < s.size()-4) OK = 0;	
		if(v[v.size()-1] == s.size()-1) OK = 0;
		if(v[0] == 0) OK = 0;
		
	}	
	for(int i=1;i<v.size();i++)
	{
		int places = v[i] - v[i-1] -1;
		if(places < 2 || places > 11) OK = 0;	
	}	  	
	if(OK == 1)
	{
		cout << "YES" << endl;
		for(int i=1;i<v.size();i++)
		{
			int places = v[i] - v[i-1] -1;
			ans.pb(v[i-1]+min(places-1,3));		
		}
		int WSK=0;
		for(int i=0;i<s.size();i++)
		{
			cout << s[i];
			if(WSK < ans.size())
			{
				if(ans[WSK] < i) WSK++;
				if(i == ans[WSK]) cout << endl;
			}
		}
	}
	else cout << "NO" << endl;
	return 0;
}
