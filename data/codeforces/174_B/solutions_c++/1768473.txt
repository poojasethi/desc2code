#include <cstdio>
#include <iostream>
#include <cstring>

using namespace std;

#define openfile {freopen("a.inp","r",stdin);}
#define debug 0

string s, res[400005];
int n, m, f[400005], g[400005], last[400005], k;

int main() {
    if (debug) openfile;
    getline(cin,s);
    n = s.length();
    k = -999;
    f[0] = 1;
    for (int i=0;i<n;i++) {
        if (k>=i-3&&s[i]!='.') {
                for (int j=k-1;j>=max(last[k]+1,k-8)&&j>=0;j--) {
                    if (f[j-1+1]) {
                        f[i+1] = f[j-1+1];
                        g[i] = j-1;
                        break;
                    }
                }
        }
        last[i] = k;
        if (s[i]=='.') k = i;
    }
    if (f[n]) {
        printf("YES\n");
        int i = n-1;
        while (i>0) {
            m++;
            res[m].assign(s,g[i]+1,i-g[i]);
            i = g[i];
        }
        for (int i=m;i>0;i--) cout << res[i] << endl;
    }
    else    printf("NO\n");
}
