#include <iostream>
#include <string>
using namespace std;

int a[400000];

int main()
{
	string s;
	cin >> s;


	for(int i=0; i<400000; i++)
		a[i] =0;


	int j=1;
	for(int i=0; i<s.length(); i++)
	{
		if(s[i] == '.')
			a[j++] = i;
	}
	a[j] = s.length()-1 ;

	if(j==1 || a[1] >8 || a[1]==0 || a[j-1] == a[j] || a[j]-a[j-1] >3 )
	{
		cout <<"NO" << endl;
		j= -1;
	}

	if(j!=-1)
	{
		for(int i=1;i<j-1; i++)
		{
			if(a[i+1]-a[i]>12 || a[i+1]-a[i] < 3 )
			{
				cout <<"NO" << endl;
				j = -1;
				break;
			}
		}

	}

	if(j!=-1)
	{
		cout << "YES" << endl;
		int stat=0,enp=0;
		for(int k=1; k<j; k++)
		{
			if(k == j-1)
			{
				enp = a[j];
			}
			else if(a[k+1] - a[k] -1 <=3 )
				enp = a[k] +1;
			else
				enp = a[k] + 3;

			while(stat<=enp)
			{
				cout << s[stat];
				stat++;
			}
			cout <<endl;
		}
	}
	return 0;
}
