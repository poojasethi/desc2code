#include<cstdio>
#include<cstring>

char x[500001];
int y[500001];
int main()
{
	int k,m;
	bool flag;
	while(scanf("%s",x)==1)
	{
		flag=1;
		memset(y,0,sizeof(y));
		k=strlen(x);
		m=0;
		for(int i=0;i<k;i++)
		{
			if(x[i]!='.') y[m]++;
			else m++;
		}
		for(int i=1;i<=m-1;i++) if(y[i]<2||y[i]>11) flag=0;
		if(y[0]<1||y[0]>8) flag=0;
		if(y[m]<1||y[m]>3) flag=0;
		if(flag==0||m==0) printf("NO\n");
		else
		{
			printf("YES\n");
			int i,a,b;
			a=b=0;
			for(i=0;i<k;i++)
			{
				printf("%c",x[i]);
				b++;
				if(x[i]=='.') 
				{
					a++;
					b=0;
				}
				if(a!=0&&a!=m)
				{
					if(y[a]>3&&b==3) printf("\n");
					if(y[a]==3&&b==2) printf("\n");
					if(y[a]==2&&b==1) printf("\n");
				}
			}
			printf("\n");	
		}
	}
	return 0;
}
