#include <iostream>
#include <string>
#include <vector>

using namespace std;

string S;
vector<string> part;

void solve()
{
	int N=part.size();
	if (N<2||part[0].size()>8||part[N-1].size()>3||part[0].empty())
	{
		cout << "NO" << endl;
		return;
	}
	for (int i=1;i<N-1;i++)
		if (part[i].size()>11||part[i].size()<2)
		{
			cout << "NO" << endl;
			return;
		}
	cout << "YES" << endl << part[0];
	for (int i=1;i<N-1;i++)
	{
		string ext,fn;
		int j=0;
		for (;;j++)
		{
			ext+=part[i][j];
			if (part[i].size()-j-1<=8)
				break;
		}
		for (j++;j<part[i].size();j++)
			fn+=part[i][j];
		cout << "." << ext << endl << fn;
	}
	cout << "." << part[N-1] << endl;
}

int main()
{
	string p;
	cin >> S;
	S+='.';
	for (int i=0;i<S.size();i++)
	{
		if (S[i]!='.')
			p+=S[i];
		else
		{
			if (S[i-1]=='.')
			{
				cout << "NO" << endl;
				return 0;
			}
			part.push_back(p);
			p.clear();
		}
	}
	solve();
	return 0;
}
