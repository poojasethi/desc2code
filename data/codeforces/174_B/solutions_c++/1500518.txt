#include <cstdio>
#include <cstring>
const int maxn = 400005;

int tot;
int a[maxn];
int b[maxn];
char str[maxn];

bool check() {
	int i, len=strlen(str);
	tot = 0;
	for (i=0; i < len; i++) if (str[i] == '.') {
		a[tot++] = i;
	}
	if ( tot == 0 || a[0] == 0 || a[0] > 8 )	return false;
	if ( len - a[tot-1] - 1 == 0 || len-a[tot-1]-1 > 3 )	return false;
	for (i=0; i < tot-1; i++) if (a[i+1]-a[i] < 3 || a[i+1]-a[i] > 12) {
		return false;
	}
	return true;
}
int main() {
	int n, i, j=0, len;
	scanf("%s",str);
	if ( check() ) {
		printf("YES\n");
		len = strlen(str);
		//a[tot++] = len;
		for (i=0; i < len; i++) {
			printf("%c",str[i]);
			if ( a[j] < i )	j++;
			if ( j && j < tot && !b[j] && a[j]-i-1 <= 8 ) {
				printf("\n");
				b[j] = 1;
			}
		}
		printf("\n");
	}
	else 
		printf("NO\n");
	return 0;
}
