#include <cstdio>
#include <cstring>
using namespace std;
const int MAXN = 4*10e5+10;
char s[MAXN];
bool Is()
{
	int i,j,l=strlen(s),dot=0;
	if(s[0]=='.'||s[l-1]=='.') return false;
	for(i=0,j=0;i<l&&s[i]!='.';++i,++j);
	if(j>8||i>=l) return false;
	else j=0;
	for(dot=1,++i;i<l;++i)
		if(s[i]=='.')
		{
			if(j<2||j>11) return false;
			j=0;
		}
		else ++j;
	if(j>3) return false;
	if(!dot)
		return false;
	else
		return true;
}
int main()
{
	scanf("%s",s);
	if(!Is())	printf("NO\n");
	else
	{
		printf("YES\n");
		int i,j,l=strlen(s);
		for(i=0;i<l;)
		{
			for(;s[i]!='.';printf("%c",s[i++]));
			printf("%c",s[i++]);
			for(j=0;j<3&&i<l&&s[i+1]!='.';++j,printf("%c",s[i++]));
			printf("\n");
		}
	}
	return 0;
}
