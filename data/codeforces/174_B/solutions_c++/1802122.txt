#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main()
{
  char s[4*100001]={'\0'};
  gets(s);
  int l=strlen(s),k=-1;
  bool *arr=(bool *)calloc(l,sizeof(bool));
  int i,count=0;;
  for(i=0;i<l;i++)
   if (s[i]=='.') {k=i;break;}
  if(k==0||k==-1||k>8) printf("NO");
  else
  {
  for(i=k+1;i<l;i++)
  {
     if(s[i]=='.')
     {
       if(count>=2&&count<=11) 
       {
          if (count>8) arr[i-8-1]=1;
          else arr[i-count]=1;
          count=0;
       }
       else 
       {printf("NO");return 0;}
     } 
     else count++;
 }
  if(count>3||count<1) {printf("NO\n");return 0;}
  printf("YES\n");
  for(i=0;i<l;i++)
    if (arr[i]==0) putchar(s[i]);
    else {putchar(s[i]);putchar('\n');}
  }
  return 0;
}
