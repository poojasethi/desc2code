#include <stdio.h>

#include <math.h>
#include <iostream>

using namespace std;

string ss[200000];

int main()
{
	//freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
	string s;
	int bef,k,aft,i,j,n;
	cin >>s;
	n=(int)s.size();
	bef=aft=0;j=0;k=0;
	for (i=0;i<n;i++)
	{
		if (s[i]=='.')
		{
			if (bef==0)
			{cout <<"NO\n";return 0;}
			bef=aft=0;j=1;
		}
		else
		{
			if (j==0)
			{
				bef++;
				if (bef>8)
				{cout <<"NO\n";return 0;}
			}
			else 
			{
				if ((i+1<n && s[i+1]=='.') || aft==3)
				{
					if (aft==0)
					{cout <<"NO\n";return 0;}
					j=aft=0;bef=1;
					k++;
				}
				else aft++;
			}
		}
		ss[k]+=s[i];
	}
	if (s[n-1]=='.' || bef!=0)cout <<"NO";
	else
	{
		cout <<"YES\n";
		for (i=0;i<=k;i++)
		{
			cout <<ss[i]<<endl;
		}
	}
	return 0;
}
