#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;

const int mod = 1000000000 + 7;
const int N = 400005;
char a[N];
int stop[N], size;
int main(){
	scanf("%s", a);
	int len = strlen(a);
	bool flag = 1;
	for(int i = 0, j = 0; i < len; i = j+1) {
		for(j = i; j < len; j ++) {
			if(a[j] == '.') break;
		}
		if(j == len) break;
		if(i == 0) {
			if(j >=1 && j <= 8) stop[size ++] = j;
			else {
				flag = 0;
				break;
			}
		}
		else {
			if(j-i < 2 || j - i > 11) {
				flag = 0;
				break;
			}
			else if(j - i > 8) {
				stop[size-1] += j-i-7;
			}
			else {
				stop[size-1] += 2;
			}
			stop[size++] = j;
		}
	}
	if(len - stop[size-1] > 4 || len - stop[size-1] <= 1) flag = 0;
	else stop[size-1] = len;
	if(flag && size) {
		puts("YES");
		for(int i = 0, j = 0; i < size; i ++) {
			for( ; j < stop[i]; j ++) {
				printf("%c", a[j]);
			}
			puts("");
		}
	}
	else puts("NO");
    return 0;
}
