#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main() {
	string s;
	cin >> s;
	vector<string> ans;
	string buf;
	string end;
	bool b = false;
	for (unsigned i = 0; i < s.length(); i++) {
		if (!b && s[i] != '.') {
			buf += s[i];
			if (buf.length() > 8) {
				cout << "NO" << endl;
///				cout << buf << ' ' << end << endl;
				return 0;
			}
			if (i == s.length() - 1) {
				cout << "NO" << endl;
				return 0;
			}
		} else if (s[i] == '.' || b) {
			end += s[i];
			b = true;
			if (buf.length() == 0) {
				cout << "NO" << endl;
				return 0;
			}
///			cout << end.length() << endl;
			if (end.length() > 3 || i + 1 == s.length() || s[i + 2] == '.' || s[i + 1] == '.') {
				if (end.length() == 0 || end.length() > 4 || end == ".") {
					cout << "NO" << endl;
					return 0;
				}
				ans.push_back(buf + end);
				buf = "";
				end = "";
				b = false;
			}
		}
	}
	cout << "YES" << '\n';
	for (unsigned j = 0; j < ans.size(); j++) {
		cout << ans[j] << '\n';
	}
	cout << endl;
}
