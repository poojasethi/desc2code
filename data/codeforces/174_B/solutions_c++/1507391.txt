#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int check(char *s, int p, int l) {
	for(int i=p; i<p+l; i++) if(s[i]=='.') return 0;
	return 1;
}

int main()
{
	int i, j, p1, p2, n, name, ext, f=1, cnt=0;
	char in[400001]={0}, out[550000]={0};

	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);

	scanf("%s", in);
	name=p1=p2=0;
	n=strlen(in);
	while(p1<n) {
		if(name>8) {f=0; break; }
		if(in[p1]=='.') {
			cnt++;
			if(name==0 || in[p1+1]=='.') { f=0; break; }
			for(j=4; j>=2; j--) {
				if((p1+j<n && check(in, p1+1, j)) || (p1+j==n && check(in, p1+1, j-1))) {
					ext=j-1; j=0; break; }
			}
			if(j!=0) { f=0; break; }
			for(j=0; j<ext+1; j++) out[p2++]=in[p1++];
			out[p2++]='\n'; name=0;
		}
		else {
			out[p2++]=in[p1++]; name++;
		}
	}
	if(cnt==0 || name!=0) f=0;
	if(f) printf("YES\n%s\n", out);
	else printf("NO\n");
}
