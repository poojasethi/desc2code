#include <iostream>
//#include <conio.h>
#include <string>
#include <vector>

using namespace std;


int main() {
	string s,x;
	vector <string> s1;
	int i, n;
	bool f = true;
	cin >> s;
	x = "";
	n = 0;
	for (i = 0; i < s.length(); i++) {
		if ((s[i] == '.') && (s1.size() % 2 == 1)) f = false;
		if ((s[i] == '.') || ((x.length() == 3) && (s1.size() % 2 == 1))){
			s1.push_back(x);
			x = "";
		}
		if (!(s[i] == '.')) x += s[i];
		if ((s.length() - 1 >= i + 2) && (s1.size() % 2 == 1)) {
		    if ((s[i+2] == '.') && !((s[i] == '.') || ((x.length() == 3) && (s1.size() % 2 == 1)))) {
			    s1.push_back(x);
			    x ="";
			}
		}
        
	}
	if (x.length() > 0) s1.push_back(x);
	for (i = 0; i < s1.size(); i++) {
		if ((s1[i] == "") || (s1[i].length() > 8)) f = false;
	}
	if (s1.size() % 2 == 1) f = false;
	if (!f) cout << "NO";
	else {
		cout << "YES" << "\n";
		for (i = 0; i < s1.size(); i += 2) {
			cout << s1[i] << "." << s1[i+1] << "\n";
		}
	}
	//getch();
	return 0;
}
