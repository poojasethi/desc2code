#include <cstdio>
#include <iostream>
using namespace std;

const int Maxl=400002;
char s[Maxl];
int Num[Maxl];

int main()
{
    gets(s);
    int k=0;
    for ( int i=0;s[i]!='\0';i++ )
     if ( s[i]=='.' )  k++;
     else Num[k]++;
    bool flag=false;
    for ( int i=1;i<k;i++ )
     if ( Num[i]<2 || Num[i]>11 )
     {
         flag = true;
         break;
     }
    if ( k<1 || flag || Num[0]<1 || Num[0]>8 || Num[k]<1 || Num[k]>3 )
        puts("NO");
    else {
        puts("YES");
        for ( int i=0;i<=Num[0];i++ )  putchar(s[i]);
        int ext,p=Num[0]+1;
        for ( int i=1;i<k;i++ )
        {
            if ( Num[i]>9 )  ext = 3;
            else  ext = 1;
            for ( int j=0;j<ext;j++ )  putchar(s[p++]);
            putchar('\n');
            for ( int j=ext;j<=Num[i];j++ )  putchar(s[p++]);
        }
        for ( ;s[p]!='\0'; )  putchar(s[p++]);
    }
    return 0;
}
