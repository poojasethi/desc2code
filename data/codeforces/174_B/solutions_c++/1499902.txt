#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
#include <cmath>
using namespace std;
string s;int n,p,labl[444444],kol;
int main()
{
    cin>>s;n=s.length();
    for (int i=0;i<n;i++){
        if (kol==0 && s[i]=='.' && p==0){
                   if (i==0 || i>8){cout<<"NO";return 0;}
                   p=i;}
        else if (s[i]=='.'){
             if (i-p-1<2 || i-p-1>11){cout<<"NO";return 0;}
             if (i-p-1>3)labl[p+3]=1;
             else labl[p+1]=1;kol++;
             p=i;}
        }
    if (kol==0 && p==0){cout<<"NO";return 0;}
    if (n-p-1>3 || p==n-1){cout<<"NO";return 0;}
    cout<<"YES"<<endl;
    for (int i=0;i<n;i++){
        cout<<s[i];
        if (labl[i])cout<<endl;}
    cin>>n;
    return 0;
}
