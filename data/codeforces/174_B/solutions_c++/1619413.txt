#include <cstdio>
#include <cstring>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

char s[400005];
vector<string> a;
int main(void) {
	scanf("%s", s);
	string now = "";
	for(int i=0;s[i];) {
		if(s[i]=='.') {
			now += ".";
			++i;
			for(int j=1;j<=3 && s[i] && s[i]!='.';j++,i++)
				if(s[i+1]!='.') now += s[i];
				else break;
			a.push_back(now);
			now = "";
		} else now += s[i++];
	}
	int fail=0;
	for(int i=0;i<(int)a.size();i++) {
		for(int j=0;j<(int)a[i].size();j++)
			if(a[i][j]=='.') {
				if(j>8 || j==0 || j==(int)a[i].size()-1)
					fail=1;
			}
	}
	if(fail || now != "") puts("NO");
	else {
		puts("YES");
		for(int i=0;i<(int)a.size();i++)
			printf("%s\n", a[i].c_str());
	}
	return 0;
}
