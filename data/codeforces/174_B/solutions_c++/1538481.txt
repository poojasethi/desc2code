#include<iostream>
#include<vector>
using namespace std;

#define No { cout << "NO" << endl; return 0; }

const int MAXN = 8;
const int MAXE = 3;

vector<string> out;
string s;

int main()
{
	cin >> s;
	bool st = 0;
	int l = 0;
	out.push_back("");
	for(int i=0; i<(int)s.size(); i++)
	{
		//cerr << i << endl;
		if( s[i] == '.' )
		{
			if( l == 0 || st == 1 )
				No;
			st = 1;
			l = 0;
		}
		else
		{
			if( st == 1 && ( l == MAXE || ( i+1 < (int)s.size() && s[i+1] == '.' ) ) )
			{
				if( l == 0 )
					No;
				st = 0;
				l = 0;
				out.push_back("");
			}
			l++;
		}
		out.back() += s[i];
		if( ( st == 0 && l > MAXN ) || ( st == 2 && l > MAXE ) )
			No;
	}
	if( ( st == 0 && l > 0 ) || ( st == 1 && l == 0 ) )
		No;
	cout << "YES" << endl;
	for(int i=0; i<(int)out.size(); i++)
		if( out[i] != "" )
			cout << out[i] << endl;
	return 0;
}
