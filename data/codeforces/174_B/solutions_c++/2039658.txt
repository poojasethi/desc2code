#include <iostream>
#include <algorithm>
#include <string>
#include <cstdlib>
#include <vector>
using namespace std;

string s, rs, ns;
int n;
vector<int> a;

void no() {
  cout << "NO" << endl;
  exit(0);
}

int main() {
  cin >> s;
  n = s.length();
  rs = s;
  reverse(rs.begin(), rs.end());

  if (s.find('.') == 0 ||
      s.find('.') > 8 ||
      rs.find('.') == 0 ||
      rs.find('.') > 3)
    no();
  for (int i = 0; i < n; ++i)
    if (s[i] == '.') a.push_back(i);
  for (int i = 1; i < a.size(); ++i)
    if (a[i]-a[i-1]-1 < 2 ||
	a[i]-a[i-1]-1 > 11)
      no();

  cout << "YES" << endl;
  for (int i = 0, j = 0; i < a.size(); ++i)
    if (i < a.size()-1) {
      if (a[i+1]-a[i]-1 == 2) {
	cout << s.substr(j, a[i]+2-j) << endl;
	j = a[i]+2;
      } else if (a[i+1]-a[i]-1 == 3) {
	cout << s.substr(j, a[i]+3-j) << endl;
	j = a[i]+3;
      } else {
	cout << s.substr(j, a[i]+4-j) << endl;
	j = a[i]+4;
      }
    } else {
      cout << s.substr(j) << endl;
    }
  return 0;
}
