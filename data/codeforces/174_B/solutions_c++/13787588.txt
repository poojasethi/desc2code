//In the Name of God
#include <iostream>
#include <string>
#include <vector>

using namespace std;

const int maxn = 100000 * 4 + 10;
int n, cnt, t;
bool sw;
vector <int> dots;
string ans[maxn], s;

int main(){
	cin >> s;
	n = s.size();
	for(int i = 0 ; i < n ; i++)
		if(s[i] == '.')
			dots.push_back(i);
	dots.push_back(n);
	cnt = sw = t = 0;
	for(int i = 0 ; i < n ; i++){
		if(s[i] == '.'){
			sw = true;
			t = 0;
		}
		if(sw && (t == 4 || (dots[cnt + 1] - i == 1 && dots[cnt + 1] != n))){
			cnt++;
			sw = false;
		}
		if(sw) t++;
		ans[cnt].push_back(s[i]);
	}
	for(int i = 0 ; i <= cnt ; i++){
		int dot = ans[i].find('.');
		if(dot < 9 && dot > 0 && ans[i].size() - dot > 1 && ans[i].size() - dot < 5){}
		else{
			cout << "NO" << endl;
			return 0;
		}
	}
	cout << "YES" << endl;
	for(int i = 0 ; i <= cnt ; i++)
		cout << ans[i] << endl;
	return 0;
}
