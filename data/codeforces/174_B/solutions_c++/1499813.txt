#include <stdio.h>
#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <cstring>
#include <set>

using namespace std;

vector <int> a;
int start[1000000];
char s[1000000];

int main()
{
    //freopen("input.txt", "r", stdin); freopen("output.txt", "w", stdout);
	scanf("%s", s);
	int n=strlen(s);
	int l=0;
	a.resize(n+1);
	for (int i=0; i<n; i++)
		if (s[i]=='.') a[l++]=i;
	if (l==0)
	{
		cout << "NO";
		return 0;
	}
	start[0]=0;
	start[l]=n;
	if (a[0]>8 || a[0] < 1) {
		cout << "NO";
		return 0;
	}
	if (n-a[l-1] <= 1 || n-a[l-1]>4){
		cout << "NO";
		return 0;
	}
	for (int i=1; i<l; i++)
	{
		if (a[i]-a[i-1] <= 2 || a[i]-a[i-1] > 12)
		{
			cout << "NO";
			return 0;
		}
		if (a[i]-a[i-1] > 4 ) start[i] = a[i-1] + 4;
		else start[i] = a[i-1] + 2;
	}
	cout << "YES\n";
	for (int i=0; i<l; i++)
	{
		for (int j=start[i]; j<start[i+1]; j++)
			putc(s[j], stdout);
		putc('\n', stdout);
	}
	return 0;
}
