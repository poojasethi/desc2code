#include <iostream>
#include <string>

using namespace std;

const int Maxn = 2 * 1000 * 100 + 10;
#define NO {cout << "NO"; return 0;}

string s;
string a[Maxn];
int cnt =0 ;


int main()
{
    cin >> s;
    int n=(int) s.size();
    int i=0;
    while(i<n)
    {
        for(i;i<n && s[i]!= '.'; i++)
            a[cnt]+= s[i];
        cnt++, i++;
    }
    for(int i=1; i<cnt - 1; i++)
        if(a[i].size() < 2 || a[i].size() > 8 + 3)
            NO
    if(a[0].size() < 1 || a[0].size() > 8
    || a[cnt-1].size() < 1 || a[cnt - 1].size() >3 || cnt < 2 || s[n-1] == '.')
        NO
    cout << "YES" << endl;
    int k=0;
    for(int i=0; i<cnt-1; i++)
    {
        for(int j=k; j<a[i].size(); j++)
            cout << a[i][j];
        cout << '.';
        for(k=0; k+(i!=cnt-2)<a[i+1].size() && k<3; k++)
            cout << a[i + 1][k];
        cout << endl;
    }
    return 0;
}
