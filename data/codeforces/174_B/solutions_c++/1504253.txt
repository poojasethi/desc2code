#include<fstream>
#include<iostream>
using namespace std;

//ifstream fin("input.in");
//ofstream fout("output.out");

string Mes,Fil[1000000];

int main(){
	cin>>Mes;
	int i=0,l1=0,j=0,ok=1,q;
	if(Mes[0]=='.')
		ok=0;
	while(i<Mes.size() && ok){
		l1=0;
		while(Mes[i]!='.' && i<Mes.size()){
			Fil[j].push_back(Mes[i]);
			l1++;
			i++;
		}
		if(l1>8 || l1==0){
			ok=0;
			break;
		}
		Fil[j].push_back(Mes[i]);
		i++;
		for(q=1;Mes[i]!='.' && q<=3 && Mes[i+1]!='.' && i<Mes.size();q++,i++){
			Fil[j].push_back(Mes[i]);
		}
		if(q==1){
			ok=0;
			break;
		}
		j++;
	}
	j--;
	if(ok){
		ok=0;
		for(int i=0;i<Fil[j].size();i++){
			if(Fil[j][i]=='.' && i!=Fil[j].size()-1)
				ok=1;
		}
	}
	if(ok){
		cout<<"YES\n";
		for(int i=0;i<=j;i++){
			cout<<Fil[i]<<"\n";
		}
	}
	else
		cout<<"NO\n";
	return 0;
}
