#include<iostream>
#include<list>
#include<vector>
#include<algorithm>
#include<set>
#include<map>
#include<cstring>
using namespace std;

char str[500000];
char name[500000][10];
char ext[500000][5];
int nfiles;

int main(void)
{
	cin >> str;
	int l=strlen(str);
	int p=0, o=0;
	
	while(1)
	{
		if(str[p]=='.'&&o==0)
			{
				cout << "NO";
				return 0;
			}
		if(str[p]!='.')
		{
			name[nfiles][o++]=str[p++];
			if(o>8)
			{
				cout << "NO";
				return 0;
			}	
		}
		else
		{
			p++;
			int np=p;
			for(int x=p;x<p+3;x++)
				if(str[x]!='.'&&str[x+1]!='.'&&str[x+1]!=0)
				{
					ext[nfiles][x-p]=str[x];
					np=x+1;
				}
				else if(np==l-1&&str[np]!='.')
				{
					ext[nfiles][x-p]=str[x];
					np=x+1;
				}
				else break;
			if(np==p)
			{
				cout << "NO";
				return 0;
			}
			p=np;
		//	cout << name[nfiles] << '.' << ext[nfiles] << endl;
			nfiles++;o=0;
			if(str[p]==0)
				break;
		}
	}
	cout << "YES\n";
	for(int i=0;i<nfiles;i++)
		cout << name[i] << '.' << ext[i] << endl;
	return 0;
}
