#include<iostream>
#include<string>
#include<algorithm>

using namespace std;
int L[400000];

int main()
{
	int i,j,len,k,m;
	string s;

	cin>>s;
	len=s.size();
	j=0;
	for(i=0;i<len;i++)
	{
		if(s[i]=='.')
		{
			if(L[j]>11 || j && L[j]<=1)
			{
				cout<<"NO"<<endl;
				return 0;
			}
			j++;
			continue;
		}
		L[j]++;
	}

	if(L[0]>8 || L[j]>3 || L[0]==0 || !j || L[j]==0)
	{
		cout<<"NO"<<endl;
		return 0;
	}

	cout<<"YES"<<endl;
	k=0;
	for(i=0;i<L[k];i++)
		cout<<s[i];
	k++;

	for(;i<len-L[j];i++)
	{
		cout<<s[i];
		if(s[i]=='.')
		{
			m=min(3,L[k]/2)+i;
			if(m==i)
			{
				i++;
				break;
			}
			k++;
		}
		if(i==m)
			cout<<endl;
	}

	for(;i<len;i++)
		cout<<s[i];
	cout<<endl;

	return 0;
}
