#include<cstdio>
#include<queue>
#include<cstring>
using namespace std;
char str[400011];
int main(){
	char*cur,*next;
	char ch;
	int i;
	bool can;
	while(gets(str)){
		queue<char*>ls;
		can=false;
		for(i=1;i<9;++i)if(str[i]=='.')can=true;
		can=can&&str[0]!='.';
		next=str;
		while(can){
			cur=next;
			next=strchr(next,'.');
			if(!next){
				can=strlen(cur)<=2;
				break;
			}
			if(next-cur>8){
				if(next-cur<11)cur+=next-cur-8;
				else{
					can=false;
					break;
				}
			}
			ls.push(cur);
			++next;
			if(next[0]=='.'||!*next)can=false;
			++next;
			if(next[0]=='.')can=false;
		}
		if(can&&!ls.empty()){
			printf("YES\n");
			while(1){
				cur=ls.front();
				ls.pop();
				if(!ls.empty()){
					ch=*ls.front();
					*ls.front()=0;
					printf("%s\n",cur);
					*ls.front()=ch;
				}else{
					printf("%s\n",cur);
					break;
				}
			}
		}else printf("NO\n");
	}
	return 0;
}
