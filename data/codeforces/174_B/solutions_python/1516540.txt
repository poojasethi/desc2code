#!/usr/bin/python

import sys
import re

st = sys.stdin.readline()
ans = []
mb = re.match('([a-z]{1,8})\..+', st[0:10])
pos = 0
if mb:
	fn = mb.group(1)
	ans.append(fn + '.')
	#st = st[len(fn)+1:]
	pos = len(fn) + 1;
	while st:
		mb = re.match('([a-z]{2,11})\..+', st[pos:pos+15])
		if mb:
			cmb = mb.group(1)
			total = len(cmb)
			if (total <= 9):
				ans[-1] += cmb[0:1]
				ans.append(cmb[1:] + '.')
			else:
				ans[-1] += cmb[0:total-8]
				ans.append(cmb[total-8:] + '.')
			#st = st[total+1:]
			pos += total + 1
		else:
			mb = re.match('([a-z]{1,3})$', st[pos:])
			if mb:
				ext = mb.group(1)
				ans[-1] += ext
				st = []
			else:
				print "NO"
				sys.exit(0)
else:
	print "NO"
	sys.exit(0)

print "YES"
for s in ans:
	print s
