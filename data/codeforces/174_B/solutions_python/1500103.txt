s=raw_input().split('.')
if len(s)<=1:
    print 'NO'
    exit()
ans=[s[0]]
del s[0]
for i in s:
    ext=i[:min(len(i)-1,3)]
    ans[-1]+='.'+ext
    ans.append(i[len(ext):])
ans[-2]+=ans[-1]
del ans[-1]
for i in ans:
    c=i.split('.')
    if not (0<len(c[0])<=8 and 0<len(c[1])<=3):
        print 'NO'
        break
else:
    print 'YES'
    print '\n'.join(ans)
