#!/usr/bin/python
# -*- coding: utf-8 -*-

import sys

def main():
    file_ = sys.stdin.readline().strip().split('.')
    if (len(file_) < 2): return -1
    if((len(file_[0]) > 8) or (len(file_[-1]) > 3)):
        #print len(file_[0]), len(file_[-1])
        return -1
    if((len(file_[0]) < 1) or (len(file_[-1]) < 1)):
        #print len(file_[0]), len(file_[-1])
        return -1
    for  z in file_[1:-1]:
        if(len(z) < 2) or (len(z) > 11): 
            return -1
    sys.stdout.write("YES\n")
    sys.stdout.write(file_[0])
    for z in file_[1: -1]:
        sys.stdout.write('.')
        if len(z) > 10:
            sys.stdout.write('{}\n'.format(z[:3]))
            sys.stdout.write(z[3:])
        elif len(z) > 9:
            sys.stdout.write('{}\n'.format(z[:2]))
            sys.stdout.write(z[2:])
        else:
            sys.stdout.write('{}\n'.format(z[:1]))
            sys.stdout.write(z[1:])
    sys.stdout.write('.'+file_[-1])
    sys.stdout.write('\n')
    return 0
    
if __name__ == '__main__':
    if main() == -1:
        sys.stdout.write("NO\n")
