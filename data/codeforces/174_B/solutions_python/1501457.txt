s = raw_input()
l = s.split('.')

if len(l[-1]) > 3 or len(l[0]) > 8 or '..' in s or s[0] == '.' or s[-1] == '.' or len(l) == 1:
    print 'NO'
    exit(0)

for q in l[1:-1]:
    if len(q) > 11 or len(q) <= 1:
        print 'NO'
        exit(0)

print 'YES'
ll = [0]
for q in l[1:]:
    if len(q) > 3:
        ll.append(3)
    else:
        ll.append(1)
ll[-1] = len(l[-1])

for i in xrange(len(l)-1):
    print l[i][ ll[i] : ] + '.' + l[i+1][ : ll[i+1]]
