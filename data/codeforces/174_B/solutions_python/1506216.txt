import sys

def parse(s):
	ss = s.split('.')
	n = len(ss)
	if n <= 1 or len(ss[0]) < 1 or len(ss[0]) > 8 or len(ss[-1]) < 1 or len(ss[-1]) > 3:
		return None
	fl = list()
	name = ss[0]
	for x in ss[1:-1]:
		if len(x) < 2 or len(x) > 11:
			return None
		l = max(1,len(x)-8)
		fl.append(name+'.'+x[:l])
		name = x[l:]
	fl.append(name+'.'+ss[-1])
	return fl

s = sys.stdin.readline().strip()
fl = parse(s)
if fl:
	print 'YES'
	for f in fl:
		print f
else:
	print 'NO'
