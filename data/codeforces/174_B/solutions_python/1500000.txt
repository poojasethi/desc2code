#!/usr/bin/python2
def fail():
	print 'NO'
	exit(0)

if __name__ == '__main__':
	s = raw_input()
	if s.find('..') != -1 or s[0] == '.' or s[-1] == '.' or s.find('.') == -1:
		fail()
	s = s.split('.')

	res = [''] * len(s)
	#print s
	if not (1 <= len(s[0]) <= 8 and 1 <= len(s[-1]) <= 3):
		fail()
	ans = 'YES\n' + s[0]
	for i, val in enumerate(s[1:-1]):
		if len(val) > 8 + 3 or len(val) < 2:
			fail()

		p = min(3, len(val) - 1)
		ans += '.' + val[:p] + '\n' + val[p:]
	ans += '.' + s[-1]
	print ans
