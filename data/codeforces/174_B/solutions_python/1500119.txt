import re

str = raw_input()
r = r"^([a-z]{1,8}\.[a-z]{1,3})+$"
m = re.match(r, str)
if m == None:
	print("NO")
else:
	print("YES")
	parts = str.split('.')
	l = len(parts)
	i = 0
	while i + 1 < l:
		name = parts[i] + '.'
		if len(parts[i+1]) > 3 or i + 2 == l:
			name += parts[i+1][0:3]
			parts[i+1] = parts[i+1][3:]
		else:
			name += parts[i+1][0:1]
			parts[i+1] = parts[i+1][1:]
		print(name)
		i += 1
