#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int a[N][N];
int main() {
	scanf("%d",&n);
	if (n==4) {
		printf("-1\n"); return 0;
	}
	For(i,1,n) a[i][i%n+1]=1;
	For(i,1,n)For(j,i+2,n) if (i!=1||j!=n)
		if (j-i&1) a[j][i]=1; else a[i][j]=1;
	For(i,1,n)For(j,1,n) printf("%d%c",a[i][j],j==n?'\n':' ');
	//for(;;);
	return 0;
}
