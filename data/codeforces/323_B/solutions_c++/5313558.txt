#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cstdlib>
using namespace std;
int n,i,j,a[1200][1200];
int main(){
    scanf("%d",&n);
    if(!(n&1) && n<6)return puts("-1"),0;
    for(i=n;i>1;i-=2){
        if(i==6){
            for(j=1;j<=6;++j)a[j][j%6+1]=1;
            a[1][3]=a[4][1]=a[1][5]=a[2][4]=a[5][2]=a[2][6]=a[3][5]=a[6][3]=a[4][6]=1;
            break;
        }
        for(j=1;j<=i-2;++j)a[i][j]=1;
        for(j=1;j<=i-2;++j)a[j][i-1]=1;
        a[i-1][i]=1;
    }
    for(i=1;i<=n;++i)
        for(j=1;j<=n;++j)printf("%d%c",a[i][j],j==n?'\n':' ');
}
