#include<cstdio>
int n;
int main()
{
	int i,j;
	scanf("%d",&n);
	if((n&1)^1)
	{
		if(n==2 || n==4)
		{
			printf("-1\n");
			return 0;
		}
		for(i=1;i<n;i++)
		{
			for(j=1;j<n;j++)
			{
				if((i-j+n-1)%(n-1)>=n/2)
					printf("1 ");
				else
					printf("0 ");
			}
			if(i==1 || i==n/2)
				printf("1\n");
			else
				printf("0\n");
		}
		for(i=1;i<n;i++)
		{
			if(i==1 || i==n/2)
				printf("0 ");
			else
				printf("1 ");
		}
		printf("0\n");
		return 0;
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<n;j++)
		{
			if((i-j+n)%n>n/2)
				printf("1 ");
			else
				printf("0 ");
		}
		if(i%n>n/2)
			printf("1\n");
		else
			printf("0\n");
	}
	return 0;
}
