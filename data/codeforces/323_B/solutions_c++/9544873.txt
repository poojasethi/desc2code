#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
int n,a[1100][1100];
int main(){
    cin>>n;
    if(n==4){
      printf("-1\n");
      return 0;
    }
    for(int i=1;i*2<n;i++)
      for(int j=1;j<=n;j++){
         int t=(j+i)%n;
         if(t==0)t=n;
         a[j][t]=1;
      }
    if(n%2==0){
      for(int i=1;i<=n/2;i++)a[i][i+n/2]=1;
      swap(a[n/2-1][n/2+1],a[n/2+1][n/2-1]);
    }
    for(int i=1;i<=n;i++){
      for(int j=1;j<=n;j++)printf("%d ",a[i][j]);
      printf("\n");
      }
    return 0;
}
