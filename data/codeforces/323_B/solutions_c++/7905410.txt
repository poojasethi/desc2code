#include<cstdio>
using namespace std;
int a[1001][1001],n,i,j,x,y;
void work1()
{
   a[1][2]=1;a[2][3]=1;a[3][1]=1;
   x=n;y=n-1;
   while(x!=3)
   {
      for(i=1;i<=y-1;i++){a[i][x]=1;a[y][i]=1;}
      a[x][y]=1;x-=2;y-=2;
   }
}
void work2()
{
   a[1][2]=1;a[1][3]=1;a[1][4]=1;a[2][3]=1;a[2][5]=1;a[3][4]=1;a[3][6]=1;
   a[4][2]=1;a[4][5]=1;a[4][6]=1;a[5][1]=1;a[5][3]=1;a[5][6]=1;a[6][1]=1;a[6][2]=1;
   x=n;y=n-1;
   while(x!=6)
   {
      for(i=1;i<=y-1;i++){a[i][x]=1;a[y][i]=1;}
      a[x][y]=1;x-=2;y-=2;
   }
}
int main()
{
   scanf("%d",&n);
   if(n==4)printf("%d\n",-1);
   else{
           if(n&1)work1();else work2();
           for(i=1;i<=n;i++){for(j=1;j<=n;j++)printf("%d ",a[i][j]);printf("\n");}
       }
}
