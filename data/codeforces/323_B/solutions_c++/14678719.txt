# include <bits/stdc++.h>

using namespace std;

const int maxn = 1e3 + 100;

bool A[maxn][maxn];

void build(int n){
	if(n == 3){
		A[1][2] = A[2][3] = A[3][1] = 1;
		return;
	}
	if(n == 6){
		A[1][2] = A[2][3] = A[3][1] = 1;
		A[4][5] = A[5][6] = A[6][4] = 1;
		
		A[2][4] = A[3][6] = A[1][5] = 1;
		A[4][3] = A[5][2] = A[6][1] = 1;
		
		A[1][4] = A[2][6] = A[3][5] = 1;
		return;
	}
	build(n - 2);
	for (int i = 1; i <= n - 2; i++)
		A[n][i] = A[i][n - 1] = 1;
	A[n - 1][n] = 1;
}

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(0); cout.tie(0);
	int n;
	cin >> n;
	if(n == 4){
		cout << "-1" << endl;
		return 0;
	}

	build(n);
	for (int i = 1; i <= n; i++){
		for (int j = 1; j <= n; j++)
			cout << A[i][j] << " ";
		cout << endl;
	}

	return 0;
}
