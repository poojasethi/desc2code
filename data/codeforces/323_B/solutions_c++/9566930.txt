#include <cstdio>

const int N=1005;

int n;
int g[N][N];

int main(){
	scanf("%d",&n);
	if(n==4) return puts("-1"),0;
	for(int i=1;i<n;i++) g[i][i+1]=1;
	for(int i=1;i<=n;i++)
		for(int j=i+2;j<=n;j++)
			if((i&1)==(j&1))
				g[i][j]=1;
			else
				g[j][i]=1;
	g[n][1]=1;
	g[1][n]=0;
	for(int i=1;i<=n;puts(""),i++)
		for(int j=1;j<=n;j++) printf("%d ",g[i][j]);
}
