#include<cstdio>

const int MAXN=1000+9;

int g[MAXN][MAXN];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
#endif
	int n,i,j;
	scanf("%d",&n);
	if(n==4)
	{
		puts("-1");
		return 0;
	}

	if(n&1)
		i=2;
	else
	{
		g[1][2]=g[1][4]=g[1][5]=1;
		g[2][3]=g[2][5]=1;
		g[3][1]=g[3][6]=1;
		g[4][2]=g[4][3]=1;
		g[5][4]=g[5][6]=1;
		g[6][1]=g[6][2]=1;
		g[4][6]=g[3][5]=1;
		i=7;
	}
	for(;i<=n;i+=2)
	{
		for(j=1;j<i;++j)
			g[j][i]=1;
		g[i][i+1]=1;
		for(j=1;j<i;++j)
			g[i+1][j]=1;
	}

	for(i=1;i<=n;++i)
		for(j=1;j<=n;++j)
			printf("%d%c",g[i][j],j<n?' ':'\n');
	return 0;
}
