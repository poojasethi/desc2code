#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <string>
using namespace std;
typedef long long LL;
const int N = 1005;
int a[N][N];
int main () {
    #ifndef ONLINE_JUDGE
         freopen("input.txt","r",stdin);
    #endif
    int n ;
    cin >> n;
    if (n == 4) {
        puts("-1");
        return 0;
    }
    for (int i = 0 ; i < n ; i ++) {
        a[i][(i + 1) % n] = 1;
    }
    for (int i = 0 ; i < n ; i ++) {
        for (int j = i + 2 ; j < n ; j ++) {
            if (i == 0 && j == n - 1) continue;
            if ((j - i) & 1) a[j][i] = 1;
            else a[i][j] = 1;
        }
    }
    for (int i = 0 ; i < n ; i ++) {
        for (int j = 0 ; j < n ; j ++)
            cout << a[i][j] << " ";
        cout << endl; 
    }
    return 0;
}
