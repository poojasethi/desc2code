#include <cstdio>
#include <cstring>
#include <algorithm>

int N;
int Map[1000+7][1000+7];

int main()
{
	scanf("%d", &N);
	if (N==4) puts("-1");
	else
	{
		int st;
		if (N&1)
		{
			st=3;
			Map[1][2]=Map[2][3]=Map[3][1]=1;
		}
		else
		{
			st=6;
			Map[1][2]=Map[1][3]=Map[1][4]=1;
			Map[2][3]=Map[2][4]=Map[2][5]=1;
			Map[3][4]=Map[3][6]=1;
			Map[4][5]=Map[4][6]=1;
			Map[5][1]=Map[5][3]=Map[5][6]=1;
			Map[6][1]=Map[6][2]=1;
		}
		for (; st!=N; st+=2)
		{
			int u=st+1;
			int v=u+1;
			for (int j=1; j<u; j++) Map[u][j]=1;
			for (int j=1; j<u; j++) Map[j][v]=1;
			Map[v][u]=1;
		}
		for (int i=1; i<=N; i++){
			for (int j=1; j<=N; j++)
				printf("%d ", Map[i][j]);
			puts("");
		}
	}
	return 0;
}
