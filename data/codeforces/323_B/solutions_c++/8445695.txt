#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define N 1010
using namespace std;
int n,p[N][N];
int main()
{
    cin>>n;
    if (n!=4)
    {
        if (n&1)
        {
            for (int i=1;i<=n;i++)
            for (int j=0;j<n/2;j++) 
            p[i][(i+j*2)%n+1]=1;
        }
        else
        {
            for (int i=1;i<n;i++) p[i][i+1]=1;
            for (int i=1;i<n-1;i++)
            for (int j=i+2;j<=n;j++) 
            {
                if (!(j-i&1)) p[i][j]=1;else p[j][i]=1;
            }
        }
        for (int i=1;i<=n;i++)
        {
            for (int j=1;j<n;j++)
            printf("%d ",p[i][j]);
            printf("%d\n",p[i][n]);
        }
    }
    else printf("-1\n");
    return 0;
}
