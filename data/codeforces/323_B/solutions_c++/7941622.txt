#include <stdio.h>

#define if if (
#define then )
#define do )
#define for for (
#define while while (
#define begin {
#define end }

#define MAXN 1005

bool a[MAXN][MAXN];

int main()
begin
	int n;
	scanf("%d",&n);
	if n==3 then begin
		return puts("0 1 0\n0 0 1\n1 0 0"),0;
	end else if n==4 then begin
		return puts("-1"),0;
	end;
	int n1=n-(~n&1);
	int i,j;
	for i=1;i<=n1;i++ do begin
		a[i][i%n1+1]=1;
		for j=4;j<n1;j+=2 do begin
			a[i][(i+j-2)%n1+1]=1;
		end;
	end;
	if ~n&1 then begin
		a[n][1]=1;
		for i=5;i<=n;i+=2 do a[n][i]=1;
		a[3][n]=1;
		for i=2;i<n;i+=2 do a[i][n]=1;
	end;
	for i=1;i<=n;i++ do begin
		for j=1;j<=n;j++ do begin
			putchar(48+a[i][j]);
			putchar(32);
		end;
		putchar('\n');
	end;
end
