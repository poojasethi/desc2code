#include<cstdio>
using namespace std;
bool a[1005][1005];
int main()
{
	int i,j,n,b;
	scanf("%d",&n);
	if(n==4)return puts("-1"),0;
	b=n-(~n&1);
	for(i=0;i<b;i++)
		for(j=1;j+j<b;j++)
			a[i][(i+j)%b]=1;
	if(~n&1)
		for(i=0;i<n-1;i++)
			if(i&1)
				a[n-1][i]=1;
			else
				a[i][n-1]=1;
	for(i=0;i<n;i++,puts(""))
		for(j=0;j<n;j++)
			printf("%d ",a[i][j]);
}
