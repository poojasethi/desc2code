#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;

int n, a[1111][1111];

int main()
{
    scanf("%d", &n);
    if (n == 4) printf("-1\n");
    else {
        memset(a, 0, sizeof(a));
        for (int i = 1; i <= n; i++) a[i][i % n + 1] = 1;
        for (int i = 1; i <= n; i++) 
            for (int j = i + 2; j <= n; j++)
                if (i != 1 || j != n) 
					if ( j - i & 1) a[j][i] = 1;
					else a[i][j] = 1;
        for (int i = 1; i <= n; i++) 
            for (int j = 1; j <= n; j++) 
                printf("%d%c", a[i][j], j == n ? '\n' : ' ');   
    }
    return 0;   
}
