#include<cstdio>
int n;bool map[1001][1001];
void work(int x){
	for(int i=1;i<=x;i++)
		for(int j=1;j<=x/2;j++)
			map[i][i+j>x?i+j-x:i+j]=true;
}
int main(){
	scanf("%d",&n);
	if(n==4)return printf("-1\n"),0;
	else if(n&1)work(n);
	else work(n-1);
	if(!(n&1))
		for(int i=1;i<n;i++)
			if(i&1)map[n][i]=true;
			else map[i][n]=true;
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++)
			printf("%d ",map[i][j]);
		printf("\n");
	}
	return 0;
}
