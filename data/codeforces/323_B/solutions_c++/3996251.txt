#include <cstdio>
#include <algorithm>
#include <map>
#include <utility>
#include <vector>
using namespace std;

const int MX_N = (1e3) + 3;
int a[MX_N][MX_N];

int main()
{
	int n;
	scanf("%d", &n);
	if(n == 4) puts("-1");
	else {
		for(int i = 0; i < n; ++i) {
			for(int j = i-2; j >= 0; j -= 2)
				a[i][j] = 1;
			for(int j = i+1; j < n; j += 2)
				a[i][j] = 1;
		}
		if(!(n & 1)) {
			a[0][n-1] = 0;
			a[n-1][0] = 1;
			a[0][2] = 1;
			a[2][0] = 0;
		}
		
		for(int i = 0; i < n; ++i) {
			for(int j = 0; j < n; ++j) {
				if(j) printf(" ");
				printf("%d", a[i][j]);
			}
			printf("\n");
		}
	}
	return 0;
}
