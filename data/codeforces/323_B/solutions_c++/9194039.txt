#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n, a[1005][1005];

int main(){
	scanf("%d", &n);
	if(n == 4){
		printf("-1\n");
		return 0;
	}
	int flag = 0;
	if(~ n & 1) flag = 1, n --;
	for(int i = 1;i <= n;i ++){
		for(int j = 1;j <= n;j ++){
			int k = i + j;
			if(k <= n){
				if(j & 1) a[i][k] = 1;
				else a[k][i] = 1;
			}
		}
	}
	if(flag){
		n += flag;
		a[n][1] = a[n][2] = 1;
		for(int i = 3;i < n;i ++) a[i][n] = 1;
	}
	for(int i = 1;i <= n;i ++)
		for(int j = 1;j <= n;j ++) printf("%d%c", a[i][j], (j == n) ? '\n' : ' ');
	return 0;
}
