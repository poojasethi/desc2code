#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#define rep(i,s,t) for (int i = s; i <= t; ++i)
#define per(i,s,t) for (int i = s; i >= t; --i)
using namespace std;
int n;
bool a[1010][1010];
inline void Change(int x, int y) { a[x][y] = 0; a[y][x] = 1; }
int main()
{
	scanf("%d", &n); if (n == 4) { printf("-1\n"); return 0; }
	if (n&1)
	{
		rep(i,1,n>>1)
		{
			a[i*2][i*2+1] = 1;
			rep(j,1,i*2-1) { a[j][i*2] = 1; a[i*2+1][j] = 1; }
		}
	}
	else
	{
		rep(i,1,n)
		rep(j,i+1,n)
		a[i][j] = 1;
		int x = 2, y = n-1; Change(1,n);
		while (x < y)
		{
			Change(x-1, y);
			Change(x, y+1);
			++x; --y;
		}
	}
	rep(i,1,n)
	{
		rep(j,1,n-1) printf("%d ", a[i][j]); printf("%d\n", a[i][n]);
	}
	return 0;
}
