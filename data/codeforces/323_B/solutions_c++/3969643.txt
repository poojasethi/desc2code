int n,a[1111][1111];

  cin>>n;

  	a[i][i%n+1]=1;
  for(int i=1; i<=n; i++)
  for(int j=i+2; j<=n; j++)
  	if (i==1 && j==n) continue; else

	for(int i=1; i<=n; i++){
	for(int j=1; j<=n; j++)
		cout<<a[i][j]<<" ";

	}
