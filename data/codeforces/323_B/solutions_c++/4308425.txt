#include<stdio.h>

int g[1010][1010];
int main()
{
	int n;
	scanf("%d",&n);
	if(n == 4)
	{
		printf("-1\n");
		return 0;
	}
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
		{
			if(j == i+1)
			{
				g[i][j]=1;
				g[j][i]=0;
			}
			else if((j-i)&1)
			{
				g[i][j]=0;
				g[j][i]=1;
			}
			else
			{
				g[i][j]=1;
				g[j][i]=0;
			}
		}
	g[n-1][0]=1;
	g[0][n-1]=0;
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
		{
			printf("%d",g[i][j]);
			if(j == n-1)
				printf("\n");
			else
				printf(" ");
		}
	return 0;
}
