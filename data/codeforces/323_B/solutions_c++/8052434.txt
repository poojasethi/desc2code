#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <cctype>
#include <algorithm>
#include <iostream>
using namespace std;

int main(int argc,char **argv)
{
    #ifndef ONLINE_JUDGE
        if (argc==1) freopen("323B.in","r",stdin),freopen("323B.out","w",stdout);
        else freopen(argv[1],"r",stdin),freopen(argv[2],"w",stdout);
    #endif
    int n;
    scanf("%d",&n);
    if (n==4){puts("-1");return 0;}
    for (int i=1;i<=n;++i)
    	for (int j=1;j<=n;++j)
    		if (j==i%n+1) printf("1%c",j==n?'\n':' ');else
    		if (i==j%n+1) printf("0%c",j==n?'\n':' ');else
    		if (i<j) printf("%d%c",(i&1)==(j&1),j==n?'\n':' ');else
    		printf("%d%c",(i&1)!=(j&1),j==n?'\n':' ');
    return 0;
}
