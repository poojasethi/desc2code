#include <cstdio>

int const MaxN = 1005;
int n, a[MaxN][MaxN];

int main(void) {
    scanf("%d", &n);
    if (n == 4) {
        printf("-1\n");
    } else {
        for (int i = 0; i < n; ++i) {
            a[i][(i+1)%n] = 1;
            for (int j = i+2; j < n; ++j)
                if ((j-i)&1) a[j][i] = 1;
                else a[i][j] = 1;
        }
        a[0][n-1] = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n-1; ++j)
                printf("%d ", a[i][j]);
            printf("%d\n", a[i][n-1]);
        }
    }

    return 0;
}
