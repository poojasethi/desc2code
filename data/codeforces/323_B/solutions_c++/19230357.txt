#include<cstdio>
#include<cstdlib>
#include<iostream>
using namespace std;

int n;
int a[1005][1005];


void slove(int m)
{
	for(int i=0;i<m;i++)
		for(int j=1;j<=m/2;j++)
			a[i][(i+j)%m]=1;
}


void slove1(int m)
{
	a[m-1][0]=1;
	a[m-1][(m-1)/2]=1;
	for(int i=1;i<m-1;i++)
	{
		if(i==(m-1)/2) continue;
		a[i][m-1]=1;
	}
}

int main()
{
	cin>>n;
	if(n==2||n==4)
	{
		puts("-1");
		return 0;
	}
	if(!(n&1))
	{
		slove(n-1);
		slove1(n);
	}
	else slove(n);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
			cout<<a[i][j]<<' ';
		cout<<endl;
	}
	return 0;
}
