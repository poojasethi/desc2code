#include <iostream>

#include <string>

#include <set>
using namespace std;

const int maxn=1000+15;
int n;
int sz[maxn][maxn];
int main()
{
	scanf("%d",&n);
	if (n==4)
	{
		printf("-1\n");
		return 0;
	}
	if (n&1)
	{
		for (int i=0;i<n;i++)
			for (int j=1;j<=n/2;j++) sz[i][(i+j)%n]=1;
	}
	else
	{
		int nn=n-1;
		for (int i=0;i<nn;i++)
		 for (int j=1;j<=nn/2;j++) sz[i][(i+j)%nn]=1;
		for (int i=0;i<nn;i++)
		 if (i%2==0) sz[i][nn]=1;
		  else sz[nn][i]=1;
	}
	for (int i=0;i<n;i++)
		{
			for (int j=0;j<n-1;j++) printf("%d ",sz[i][j]);
			printf("%d\n",sz[i][n-1]);
		}
	return 0;
}
