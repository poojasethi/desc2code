#include<cstdio>
#define rep(i,x,y) for(i=x;i<=y;i++)
const int N=1010;
int n,i,j,a[N][N];
void c(int n){rep(i,1,n)rep(j,1,n/2)a[i][(i+j-1)%n+1]=1;}
int main()
{
    scanf("%d",&n);if(n==4){printf("-1\n");return 0;}
    if(n&1)c(n);else{c(n-1);rep(i,1,n-1)if(i&1)a[n][i]=1;else a[i][n]=1;}
    rep(i,1,n)
    {
        
        rep(j,1,n){if(j>1)printf(" ");printf("%d",a[i][j]);}
        printf("\n");
    }
    scanf("\n");
    return 0;
}
