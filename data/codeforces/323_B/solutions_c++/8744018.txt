#include <cstdio>
#include <algorithm>

using namespace std;

int n, e, a[1010][1010];

int main() {
	scanf("%d", &n);
	if (n == 4) {
		printf("-1\n");
		return 0;
	}
	if (!(n & 1)) {
		--n;
		++e;
	}
	for (int i = 0; i < n; ++i)
		for (int j = 1; j <= n / 2; ++j)
			a[i][(i + j) % n] = 1;
	if (e) {
		n++;
		for (int i = 0; i < n - 1; ++i)
			a[n - 1][i] = i & 1, a[i][n - 1] = !(i & 1);
	}
	for (int i = 0; i < n; ++i, puts(""))
		for (int j = 0; j < n; ++j) {
			printf("%d ", a[i][j]);
		}
}
