#http://codeforces.com/contest/606/problem/A


a, b, c = [int(n) for n in raw_input().split()]
x, y, z = [int(n) for n in raw_input().split()]
can_be_remove, need_to_create = 0, 0
for (p, q) in [(x, a), (y, b), (z, c)]:
    d = p - q
    if d > 0: need_to_create += d
    else: can_be_remove += -1 * d / 2
if can_be_remove >= need_to_create: print "Yes"
else: print "No"
