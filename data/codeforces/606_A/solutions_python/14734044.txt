a = map(int , raw_input().split())
x = map(int , raw_input().split())
leftCo = 0
extraCo = 0
for i in xrange(3):
    if a[i]>x[i] and a[i]-x[i]>=2:
        extraCo =extraCo + a[i]-x[i] - (a[i]-x[i])%2 
    elif x[i]>a[i]:
        leftCo += x[i]-a[i]

if leftCo<=extraCo/2:
    print "Yes"
else:
    print "No"
