a=map(int, raw_input().split())
b=map(int, raw_input().split())
rem=0
for i in range(3):
    a[i]-=b[i]
    if a[i]>0:
        rem+=(a[i]/2)
        a[i]=0;

for i in range(3):
    if a[i]<0:
        b=min(-a[i], rem)
        a[i]+=b;
        rem-=b;

flag=True
for i in range(3):
    if(a[i]<0):
        print 'No'
        flag=False

if flag : print 'Yes'
