a,b,c = map(int, raw_input().split())
x,y,z = map(int, raw_input().split())

ablue = aviolet = aorange = 0
rblue = rviolet = rorange = 0

if(a > x):
	ablue = a - x
else:
	rblue = x - a
if(b > y):
	aviolet = b - y
else:
	rviolet = y - b
if(c > z):	
	aorange = c - z
else:
	rorange = z - c

if(ablue/2 + aviolet/2 + aorange/2 >= rblue + rorange + rviolet):
	print "Yes"
else:
	print "No"
