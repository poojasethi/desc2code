a, b, c = map(int, raw_input().split())
x, y, z = map(int, raw_input().split())
x, y, z = (a-x, b-y,c-z)
def solve():
    global x, y, z, a, b, c
    if x>=0 and y >= 0 and z >=0:
        return True
    x, y, z = sorted([x,y,z])
    if x < 0 and y >= 0:
        if abs(x) <= (y/2  + z/2): return True
        return False
    if x < 0 and y < 0 and z > 0:
        if abs(x + y) <= (z/2): return True
        return False
    return False
print "Yes" if  solve() else "No"
