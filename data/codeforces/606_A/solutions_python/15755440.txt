
a,b,c = map(int, raw_input().split())
x,y,z = map(int, raw_input().split())

def get_needed():
    need = 0
    if a < x:
        need += x - a
    if b < y:
        need += y - b
    if c < z:
        need += z - c
    return need

def get_extra():
    extra = 0
    if a > x:
        extra += (a-x)/2
    if b > y:
        extra += (b-y)/2
    if c > z:
        extra += (c-z)/2
    return extra


if get_needed() <= get_extra():
    print 'Yes'
else:
    print 'No'
