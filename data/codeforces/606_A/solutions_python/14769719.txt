a, b, c = map(int, raw_input().split())
x, y, z = map(int, raw_input().split())

res = max(0, x-a) + max(0, y-b) + max(0, z-c)
count = 0
while a - 2 >= x:
    count += 1
    a -= 2
while b - 2 >= y:
    count += 1
    b -= 2
while c - 2 >= z:
    count += 1
    c -= 2

if count >= res:
    print "Yes"
else:
    print "No"
