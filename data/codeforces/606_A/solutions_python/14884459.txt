import io
import sys

line1 = sys.stdin.readline()
arr1 = line1.split(' ')

line2 = sys.stdin.readline()
arr2 = line2.split(' ')

sum = 0
for i in xrange(3):
    a = int(arr1[i]) - int(arr2[i])
    if a > 0:
        sum += a / 2
    else:
        sum += a

if sum >= 0:
    print 'Yes'
else:
    print 'No'
