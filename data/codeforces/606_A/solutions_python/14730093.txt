aa = (a, b, c) = map(int, raw_input().split())
ax = (x, y, z) = map(int, raw_input().split())
ad = (ax, by, cz) = a - x, b - y, c - z
cp = sum(1 for i in ad if i >= 0)
cd = sum(1 for i in ad if i >= 2)
sd = sum(i/2 for i in ad if i >= 2)
sp = sum(i for i in ad if i >= 0)
sm = sp - sum(ad)
if cp == 3: ans = "Yes"
elif cd < 1: ans = "No"
elif sd >= sm: ans = "Yes"
else: ans = "No"
print ans
