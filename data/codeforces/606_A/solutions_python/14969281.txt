a, b, c = map(int, raw_input().split())
d, e, f = map(int, raw_input().split())
x = 0
y = 0
if a > d:
    x = x + (a - d) / 2
else:
    y = y + d - a
if b > e:
    x = x + (b - e) / 2
else:
    y = y + e - b
if c > f:
    x = x + (c - f) / 2
else:
    y = y + f - c
if x - y >= 0:
    print('Yes')
else:
    print('No')
