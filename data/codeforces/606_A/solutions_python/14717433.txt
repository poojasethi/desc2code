import sys

(a, b, c) = (int(x) for x in sys.stdin.readline().strip().split(' '))
(a1, b1, c1) = (int(x) for x in sys.stdin.readline().strip().split(' '))

a2 = a - a1
b2 = b - b1
c2 = c - c1

l = sum((int(x / 2) if x > 0 else 0 for x in (a2, b2, c2)))
n = sum(x if x < 0 else 0 for x in (a2, b2, c2))
print ('Yes' if l + n >= 0 else 'No')
