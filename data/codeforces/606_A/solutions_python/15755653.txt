import sys


a=sys.stdin.readline().rstrip().split()
b=sys.stdin.readline().rstrip().split()
a = map(int, a)
b = map(int, b)
x=0
y=0

if(a[0]<b[0]):
    x=x+b[0]-a[0]
else:
    y=y+((a[0]-b[0])/2)

if(a[1]<b[1]):
    x=x+b[1]-a[1]
else:
    y=y+((a[1]-b[1])/2)

if(a[2]<b[2]):
    x=x+b[2]-a[2]
else:
    y=y+((a[2]-b[2])/2)

if x<=y:
    print "Yes"

else:
    print "No"
