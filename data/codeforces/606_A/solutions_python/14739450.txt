a,b,c = map(int, raw_input().split())
x,y,z = map(int, raw_input().split())

t = 0
if a >= x:
    a -= x
    x = 0
    if a > 1:
        t += a/2
else:
    x -= a

if b >= y:
    b -= y
    y = 0
    if b > 1:
        t += b/2
else:
    y -= b

if c >= z:
    c -= z
    z = 0
    if c > 1:
        t += c/2
else:
    z -= c

if x+y+z <= t:
    print 'Yes'
else:
    print 'No'
