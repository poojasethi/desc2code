#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

def ma0(a):
    return max(a, 0)

a, b, c = ia()
x, y, z = ia()

need = ma0(x-a) + ma0(y-b) + ma0(z-c)
have = ma0(a-x) // 2 + ma0(b-y) // 2 + ma0(c-z) // 2

print 'YES' if have >= need else 'NO'
