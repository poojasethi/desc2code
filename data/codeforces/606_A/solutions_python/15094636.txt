a,b,c=map(int,raw_input().split())
a1,b1,c1=map(int,raw_input().split())
a2 = a1-a
b2 = b1-b
c2 = c1-c
count=0
for x in [a2,b2,c2]:
    if x >0:
        count+=x*2
    else:
        if x%2==0:
            count+=x
        else:
            count+=x+1
if count>0:
    print "No"
else:
    print "Yes"
