a,b,c = [int(x) for x in raw_input().split()]
x,y,z = [int(x) for x in raw_input().split()]

ex1 = x-a
ex2 = y-b
ex3 = z-c

pos = 0
neg = 0

if ex1 < 0:
    pos += (-ex1)/2
else:
    neg += ex1

if ex2 < 0:
    pos += (-ex2)/2
else:
    neg += ex2
    
if ex3 < 0:
    pos += (-ex3)/2
else:
    neg += ex3
    
val = neg

if pos-val < 0:
    print "No"
else:
    print "Yes"
