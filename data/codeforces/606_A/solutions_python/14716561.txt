def gcd(a, b):
    if b == 0:
        return a
    else:
        return gcd(b, a % b)

a, b, c = map(int, raw_input().strip().split(' '))
x, y, z = map(int, raw_input().strip().split(' '))
q = a - x
w = b - y
e = c - z
k = 0
if q>0:
    k+=int(q/2)
else:
    k+=q
if w>0:
    k+=int(w/2)
else:
    k+=w
if e>0:
    k+=int(e/2)
else:
    k+=e
if k >= 0:
    print "Yes"
else:
    print "No"
