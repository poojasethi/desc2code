#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>

using namespace std;

int n,a[200];
int id[200],bj[200];
int dp[25][2000000];
int sum,m;
char s[10];

bool cmp(int a,int b)
{
	return a>b;
}
int main()
{
	scanf("%d",&n);
	for (int i=0;i<n;++i) scanf("%d",&a[i]);
	sort(a,a+n,cmp);
	scanf("%d",&m);
	for (int i=1;i<=m;++i)
	{
		scanf("%s",s);
		scanf("%d",&id[i]);
		if (s[0]=='b') bj[i]=0;else bj[i]=1;
	}
	n=min(n,m);
	sum=(1<<n);
	for (int i=1;i<=n+1;++i)
		for (int j=0;j<sum;++j) dp[i][j]=-1995092000;
	dp[n+1][sum-1]=0;
	for (int i=n;i>=1;--i)
		for (int j=0;j<sum;++j)
		{
			for (int k=0;k<n;++k)
			{
				if ((j&(1<<k))!=0) continue;
				dp[i][j]=max(dp[i][j],(dp[i+1][j+(1<<k)]*(id[i]==id[i+1]?1:-1))+a[k]*(bj[i]==1?1:0));
			}
		}
	if (id[1]==2) dp[1][0]=-dp[1][0];
	printf("%d\n",dp[1][0]);
	return 0;
}
