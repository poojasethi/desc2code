#include <cstdio>
#include <cstring>
#include <algorithm>
#include <functional>

using namespace std;

int n,m;
int s[110];
int f[1<<20],used[1<<20];
int op[25],T[25];

int dp(int dep,int st){
	if(dep>m) return 0;
	if(used[st]) return f[st];
	used[st]=1;
	if(T[dep]==1){
		int tmp=0x80000000;
		for(int j=0;j<m;j++) if(!(st&(1<<j)))
			tmp=max(tmp,dp(dep+1,st|(1<<j))+op[dep]*s[j+1]);
		return f[st]=tmp;
	}
	else{
		int tmp=0x7fffffff;
		for(int j=0;j<m;j++) if(!(st&(1<<j)))
			tmp=min(tmp,dp(dep+1,st|(1<<j))-op[dep]*s[j+1]);
		return f[st]=tmp;
	}
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&s[i]);
	sort(s+1,s+n+1,greater<int>());
	scanf("%d",&m);n=min(n,m);
	for(int i=1;i<=m;i++){
		char buf[5];scanf("%s%d",buf,&T[i]);
		op[i]=(buf[0]=='p');
	}
	printf("%d\n",dp(1,0));
//	for(int i=0;i<(1<<n);i++) printf("%d\n",f[i]);puts("");
	return 0;
}
