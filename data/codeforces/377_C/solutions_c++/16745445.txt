#include<cstdio>
#include<algorithm>
using namespace std;
int s[105];
int n,m;
char ss[22][2],t[22];
int inf=0x3f3f3f3f;
bool cmp(int a,int b)
{
	return a>b;
}
int used[22][1<<20];
int aans[22][1<<20];
int dp(int pos,int st)
{
	if(pos==m)return 0;
	if(used[pos][st]!=0)return aans[pos][st];
	else used[pos][st]=1;
	int ans;
	if(t[pos]==1)ans=-1*inf;
	else ans=inf;
	if(ss[pos][0]=='p')
	{
		for(int i=0;i<m;i++)
		{
			if((~st)&(1<<i))
			{
				if(t[pos]==1)ans=max(ans,dp(pos+1,st|(1<<i))+s[i]);
				else ans=min(ans,dp(pos+1,st|(1<<i))-s[i]);
			}
		}
	}
	else
	{
		for(int i=0;i<m;i++)
		{
			if((~st)&(1<<i))
			{
				if(t[pos]==1)ans=max(ans,dp(pos+1,st|(1<<i)));
				else ans=min(ans,dp(pos+1,st|(1<<i)));
			}
		}
	}
	aans[pos][st]=ans;
	return ans;
}
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)scanf("%d",&s[i]);
	sort(s,s+n,cmp);
	scanf("%d",&m);
	for(int i=0;i<m;i++)
	{
		scanf("%s%d",ss[i],&t[i]); 
	}
	printf("%d",dp(0,0));
	return 0;
}
