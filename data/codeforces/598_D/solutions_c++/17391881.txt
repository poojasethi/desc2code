#include<bits/stdc++.h>

using namespace std;

int n,m,q,cont=1,a,b;

int is[]={0, 0,1,-1};
int js[]={1,-1,0, 0};

char mapa[1005][1005];
int  comp[1005][1005];

int resp[1000005];

bool sepuede(int i,int j){
	return (i>=0 && j>=0 && i<n && j<m && mapa[i][j]=='.');
}

void dfs(int i,int j){
	if(!sepuede(i,j)) return;
	mapa[i][j]='w';
	comp[i][j]=cont;
	int ans=1;
	for(int r=0;r<4;r++)
		dfs(i+is[r],j+js[r]);
}




int main(){
	scanf("%d %d %d",&n,&m,&q);

	for(int i=0;i<n;i++)
		scanf("%s",&mapa[i]);
	
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			if(mapa[i][j]=='.'){
				dfs(i,j);
				cont++;
			}
		}
	}

	for(int i=1;i<n-1;i++){
		for(int j=1;j<m-1;j++){
			for(int r=0;r<4;r++){
				if(mapa[i+is[r]][j+js[r]]=='*'){
					resp[comp[i][j]]+=1;
				}
			}
		}
	}


	for(int i=1;i<=q;i++){
		scanf("%d %d",&a,&b);
		printf("%d\n",resp[comp[a-1][b-1]]);
	}

	return 0;
}
