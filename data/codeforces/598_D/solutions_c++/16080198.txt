#include<stdio.h>
#include<string.h>
int t,sum,row[4]={-1,0,0,1},column[4]={0,-1,1,0},a[1005][1005],ans[100005];
char s[1005][1005];
void dfs(int x,int y){
	int i;
	s[x][y]='@';
	for(i=0;i<4;++i){
		if(s[x+*(row+i)][y+*(column+i)]=='.'){
			dfs(x+*(row+i),y+*(column+i));
		}
		else if(s[x+*(row+i)][y+*(column+i)]=='*'){
			++sum;
		}
	}
	a[x][y]=t;
	return;
}
int main(){
	int i,n,m,k,x,y;
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;++i){
		scanf("%s",*(s+i)+1);
	}
	for(t=1;t<=k;++t){
		scanf("%d%d",&x,&y);
		sum=0;
		if(!a[x][y]){
			dfs(x,y);
			ans[t]=sum;
		}
		printf("%d\n",ans[a[x][y]]);
	}
	return 0;
}
