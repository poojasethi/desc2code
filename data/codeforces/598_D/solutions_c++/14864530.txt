#include<stdio.h>
#include<string.h>
#include<iostream>

using namespace std;

char arr[1001][1001];
int vis[1001][1001];
int ans[1000010];
int n,m,k;

int cal(int x, int y, int v)
{
    if(arr[x][y] == '*')
        return 1;
    if(vis[x][y] == v)
        return 0;
    vis[x][y] = v;
    return cal(x-1, y, v) + cal(x+1, y, v) + cal(x, y-1, v) + cal(x, y+1, v);
}

int main(void){

    scanf("%d%d%d",&n,&m,&k);
    memset(vis, 0, sizeof(vis));
    memset(ans, 0, sizeof(ans));
    for(int i = 0; i<n; i++)
    {
        scanf("%s", arr[i]);
    }

    for(int i = 0; i<k; i++)
    {
        int x, y;
        scanf("%d%d", &x, &y);
        if(vis[x-1][y-1] == 0)
        {
            ans[i+1] = cal(x-1, y-1, i+1);
        }
        else
            ans[i+1] = ans[vis[x-1][y-1]];

        printf("%d\n", ans[i+1]);

    }
    return 0;
}
