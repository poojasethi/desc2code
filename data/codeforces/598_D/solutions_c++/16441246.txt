#include<stdio.h>

const int maxn=1005;
int s1[4]={1,-1,0,0};
int s2[4]={0,0,1,-1};
char z[maxn][maxn];
int mark[maxn][maxn],num[maxn*maxn];

int a,b,id,count;

void dfs(int x,int y)
{
	int i,j,k,x1,y1;
	mark[x][y]=id;

	for(i=0;i<4;i++)
	{
		x1=x+s1[i];
		y1=y+s2[i];

		if(x1>=1&&x1<=a&&y1>=1&&y1<=b&&mark[x1][y1]==0)
		{
			if(z[x1][y1]=='*')
			{
				count++;
			}
			else
				dfs(x1,y1);
		}
	}

}
int main()
{
	int x,y,i,j,k,n,m;

	scanf("%d%d%d",&a,&b,&n);

	for(i=1;i<=a;i++)
		scanf("%s",z[i]+1);

	for(i=1;i<=a;i++)
	{
		for(j=1;j<=b;j++)
		{
			if(z[i][j]=='.'&&mark[i][j]==0)
			{
				count=0;
				id++;
				dfs(i,j);
				num[id]=count;
			}
		}
	}

	while(n--)
	{
		scanf("%d%d",&x,&y);

		printf("%d\n",num[mark[x][y]]);
	}

	return 0;
}
