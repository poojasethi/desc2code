#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

string a[1003];
int vis[1003][1003];
int co[1003][1003];
int comp[1000006];
int cc,csz;

int r[]={1,-1,0,0};
int c[]={0,0,1,-1};
void dfs(int i,int j){
co[i][j]=cc;
vis[i][j]=1;
for(int u=0; u<4; u++){
    csz+=(a[i+r[u]][j+c[u]]=='*');
    if(a[i+r[u]][j+c[u]]=='.' && !vis[i+r[u]][j+c[u]])
        dfs(i+r[u],j+c[u]);}


}

int main()
{
    int n,m,k;
    cin>>n>>m>>k;
    for(int i=0; i<n;i++)cin>>a[i];

    for(int i=0; i<n; i++)
    for(int j=0; j<m; j++){
        if(!vis[i][j] && a[i][j]=='.'){

        dfs(i,j);
        comp[cc]=csz;
        csz=0;
        cc++;
        }
    }

    for(int i=0; i<k; i++){
            int h,hh;
    scanf("%d %d",&h,&hh);
    h--; hh--;
        printf("%d\n",comp[co[h][hh]]);
    }

    return 0;
}
