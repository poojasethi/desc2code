#include<iostream>
#include<vector>
using namespace std;
vector< pair<int, int> > d;
bool v[1001][1001], map[1001][1001];
int n, m,k,x=1,ans=0,val[1001][1001],w[1000020];
void DFS(int i,int j,int z)
{
	if (map[i][j]) {ans++;return; }
	v[i][j] = 1;
	val[i][j] = z;
	if (!v[i+1][j])
	DFS(i + 1, j,z);
	if (!v[i-1][j])
	DFS(i - 1, j,z);
	if (!v[i][j+1])
	DFS(i , j+1,z);
	if (!v[i][j-1])
	DFS(i , j-1,z);
	return;
}
int main()
{
	int i,j,a,b;
	char q;
	cin >> n >> m >> k;
	for (i = 1;i <= n;i++)
		for (j = 1;j <= m;j++) { cin >> q;if (q == '*')map[i][j] = 1; }
	for (int f = 0;f < k;f++)
	{
		ans = 0;
		cin >> a >> b;
		if (val[a][b]) { cout << w[val[a][b]] << endl;continue; }
		DFS(a, b,x);
		w[x] = ans;
		x++;
		cout << ans << endl;
	}

	return 0;
}
