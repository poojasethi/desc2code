#include <stdio.h>
int n,m,musm[1005][1005]={0},max[100010]={0},di[4][2]={{-1,0},{0,-1},{0,1},{1,0}};
char mus[1005][1005];
void f(int,int,int);

int main()
{
	int k,i,j,l,x,y;
	scanf("%d%d%d",&n,&m,&k);
	getchar();
	for(i=0;i<n;i++) gets(mus[i]);
	for(l=1;l<=k;l++)
	{
		scanf("%d%d",&x,&y);
		x--;
		y--;
		if(musm[x][y]!=0) max[l]=max[musm[x][y]];
		else f(x,y,l);
		printf("%d\n",max[l]);
	}
	return 0;
}

void f(int x, int y, int l)
{
	int i;
	musm[x][y]=l;
	for(i=0;i<4;i++)
	{
		if(mus[x+di[i][0]][y+di[i][1]]=='.'&&musm[x+di[i][0]][y+di[i][1]]==0) f(x+di[i][0],y+di[i][1],l);
		else if(mus[x+di[i][0]][y+di[i][1]]=='*') max[l]++;
	}
	return;
}
