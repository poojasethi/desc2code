#include <stdio.h>

char m[1005][1005];
int v[1005][1005];
int rel[1005][1005][2];
int initx, inity;
int visited[1005][1005];

int dfs(int x, int y){
	if(m[x][y] == '*') return 1;
	if(visited[x][y]) return 0;
	visited[x][y] = 1;
	rel[x][y][0] = initx;
	rel[x][y][1] = inity;
	v[x][y] = dfs(x, y+1) + dfs(x, y-1) + dfs(x+1, y) + dfs(x-1, y);
	return v[x][y];
}

int main(){
	int n, b, k;
	int x, y;
	int temp;
	scanf("%d%d%d", &n, &b, &k);
	for(int i=0; i<n; i++) scanf("%s", m[i]);
	for(int i=0; i<k; i++){
		scanf("%d%d", &initx, &inity);
		initx--, inity--;
		if(rel[initx][inity][0] != 0){
			printf("%d\n", v[rel[initx][inity][0]][rel[initx][inity][1]]);
		} else printf("%d\n", dfs(initx, inity));
	}
}
