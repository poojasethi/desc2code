#include <bits/stdc++.h>
using namespace std;
string board[1001];
int dp[1001][1001];
int n,m,x,y,k,i,j;
int dx[4]={0,0,1,-1},dy[4]={1,-1,0,0};
int dfs(int y,int x){
	int ret=0;
	dp[y][x]=0;
	for(int i=0;i<4;++i){
		if(board[y+dy[i]][x+dx[i]]=='*')
			++ret;
		else{
			if(dp[y+dy[i]][x+dx[i]]==-1)
				ret+=dfs(y+dy[i],x+dx[i]);
		}
	}
	return ret;
}
void dfs2(int y,int x,int val){
	if(dp[y][x]>0)return;
	dp[y][x]=val;
	for(int i=0;i<4;++i)
		if(board[y+dy[i]][x+dx[i]]!='*')
			dfs2(y+dy[i],x+dx[i],val);
}
int main(){
	cin>>n>>m>>k;string s;
	for(i=1;i<=n;++i){
		cin>>s;s = '*'+s;
		board[i]=s;
	}
	memset(dp,-1,sizeof(dp));
	for(i=0;i<k;++i){
		cin>>y>>x;
		int ret= dp[y][x];
		if(ret==-1){
			ret= dfs(y,x);
			dfs2(y,x,ret);
		}
		cout<<ret<<endl;
	}
}
