#include<iostream>
#include<vector>
#include<cstring>
using namespace std;
char grid[1010][1010];
int vis[1010][1010] , ans[100010] , cnt = 0 , id = 0;

int solve(int i , int j){
	if(grid[i][j] == '*') return 1;
	if(!vis[i][j]){
		vis[i][j] = id;
		int sum = 0;
		if(!vis[i][j+1]) sum += solve(i,j+1);
		if(!vis[i][j-1]) sum += solve(i,j-1);
		if(!vis[i+1][j]) sum += solve(i+1,j);
		if(!vis[i-1][j]) sum += solve(i-1,j);
		return sum;
	}
	return 0;
}
int main(){
	int n , m , k; cin >> n >> m >> k;
	for(int i = 1 ; i <= n ; i++)
		for(int j = 1 ; j <= m ; j++)
			cin >> grid[i][j];
	memset(vis,0,sizeof vis);
	for(int i = 0; i < k ; i++){
		int x,y; cin >> x >> y;
		if(vis[x][y]) cout << ans[vis[x][y]] << endl;
		else{
			id++;
			ans[id] = solve(x,y);
			cout << ans[id] << endl;
		}
	}
}
