#include <iostream>
#include <cstring>
using namespace std;
char a[1002][1002];
int b[1002][1002];
int fun(int x,int y,int c)
{
    if (a[x][y]=='*') return 1;
    if (b[x][y]) return 0;
    b[x][y]=c;
    return fun(x-1,y,c)+fun(x+1,y,c)+fun(x,y-1,c)+fun(x,y+1,c);
}
int main()
{
    int n,m,k,c[100002]={0},ans;
    cin >> n >> m >> k;
    for(int i=1;i<=n;i++) for(int j=1;j<=m;j++)
        cin >> a[i][j];
    for(int i=1;i<=k;i++){
        int x,y;
        cin >> x >> y;
        if (b[x][y]) ans=c[b[x][y]]; else {ans=fun(x,y,i); c[i]=ans;}
        cout << ans << endl;
    }
    return 0;
}
