#include <cstdio>
#include <iostream>
#include <algorithm>
#include <string.h>
using namespace std;
int n,m,k,id=1,f[1000005];
char a[1005][1005];
int b[1005][1005];
int dfs(int c,int d)
{
	if(a[c][d]=='*') return 1;
	if(c<0||c>=n||d<0||d>=m)
	return 0;
	if(b[c][d]!=0) 
	return 0;
	int rt=0;
	b[c][d]=id;
	rt+=dfs(c+1,d);
	rt+=dfs(c-1,d);
	rt+=dfs(c,d+1);
	rt+=dfs(c,d-1);
	return rt;
}
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	//getchar();
	for(int i=0;i<n;i++)
		scanf("%s",a[i]);
	memset(b,0,sizeof(b));
	//for(int i=0;i<n;i++)
	//for(int j=0;j<m;j++)
	//cout<<a[i][j]<<endl;
	while(k--)
	{
		int x,y,ans=0;
		scanf("%d%d",&x,&y);
	//	cout<<x<<y;
		if(b[x-1][y-1]==0)
		{
			ans=dfs(x-1,y-1);
			f[id]=ans;
			id++;
			///cout<<ans;
			//cout<<x<<y;
		}
		else
		{
			ans=f[b[x-1][y-1]];
		}
		printf("%d\n",ans);
		
		}
	}
