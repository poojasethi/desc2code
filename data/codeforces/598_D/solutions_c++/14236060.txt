#include <cstdio>
#include <iostream>
#include <string>
#include <cstring>
#define INT long long int
using namespace std;

int n, m, k, dx[4]={0,1,0,-1}, dy[4]={1,0,-1,0}, s[100010], chk[1010][1010];
string B[1010];

int f(int a, int b, int c)
{
    int ret = 0;
    chk[a][b] = c;
    for(int i = 0 ; i < 4 ; i++ )
    {
        if( B[a+dx[i]][b+dy[i]] == '.' )
        {
            if( !chk[a+dx[i]][b+dy[i]] ) ret += f(a+dx[i], b+dy[i],c);
        }
        else ret++;
    }
    return ret;
}

int main()
{
    cin>>n>>m>>k;
    for(int i = 0 ; i < n ; i++ )
        cin>>B[i];
    while(k--)
    {
        int a, b;
        cin>>a>>b;
        a--, b--;
        if( !chk[a][b] ) s[k+1] = f(a,b,k+1), cout<<s[k+1]<<endl;
        else cout<<s[chk[a][b]]<<endl;
    }
}
