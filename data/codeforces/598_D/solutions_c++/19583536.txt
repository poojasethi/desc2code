#include <iostream>
using namespace std;

int n, m, k;
char map[1004][1004];
int answers[100005] = { 0 };
int visited[1004][1004] = { 0 };

int dfs(int i, int j, int i_ans) {
  if (map[i][j] == '*')
    return 1;
  if (visited[i][j])
    return 0;
  visited[i][j] = i_ans;
  return dfs(i-1, j, i_ans) + dfs(i+1, j, i_ans) + dfs(i, j-1, i_ans) + dfs(i, j+1, i_ans);
}

int main() {
  cin >> n >> m >> k;
  int i;
  for (i = 0; i < n; ++i) {
    cin >> map[i];
  }
  int x, y;
  for (i = 1; i <= k; ++i) {
    cin >> x >> y;
    --x; --y;
    if (visited[x][y]){
      cout << answers[visited[x][y]] << "\n";
    } else {
      answers[i] = dfs(x, y, i);
      cout << answers[i] << "\n";
    }
  }
  return 0;
}
