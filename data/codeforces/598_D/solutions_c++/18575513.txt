#include <bits/stdc++.h>
using namespace std;
int arr[20];
int visited[2000][2000];
string s[2000];
int c=0;
int n,m;
int dp[100012];
void dfs(int a,int b,int key)
{
	
	if(s[a][b]=='*') {
		c++;
		return;
	}
	if(visited[a][b]==0)
	{
		visited[a][b]=key;
		dfs(a,b+1,key);
		dfs(a,b-1,key);
		dfs(a-1,b,key);
		dfs(a+1,b,key);
		return;
	}
	else return;
}
int main()
{
	memset(visited,0,sizeof(visited));
	int k;
	cin>>n>>m>>k;
	for(int i=0;i<n;i++)
	{
		cin>>s[i];
	}
	int key=2;
	while(k--)
	{
		
		c=0;
		int a,b;
		cin>>a>>b;
		a--;
		b--;
		if(visited[a][b]==0)
           dfs(a,b,key);
       else c = dp[visited[a][b]] ;
       dp[key] = c;
	   cout<<c<<endl;
   	   key++;
	}
}
