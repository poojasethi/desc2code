#include <cstdio>
#include <cstring>
#define M 1001
int id, ans;
int n, m;
char G[M][M]; int v[M][M], r[M*M]; 
int ac[4][2]={0, 1, 0, -1, -1, 0, 1, 0};
void dfs(int x, int y)
{
	v[x][y]=id;
	for(int i=0; i<4; i++)
	{
		int nx=x+ac[i][0];
		int ny=y+ac[i][1];
		if(nx>0&&nx<=n&&ny>0&&ny<=m&&!v[nx][ny])
		{
			if(G[nx][ny]=='*')
				ans++;
			else
				dfs(nx, ny);
		}
	}
}
int main()
{
	int k;
	while(scanf("%d%d%d", &n, &m, &k)!=EOF)
	{
		for(int i=1; i<=n; i++)
			scanf("%s", G[i]+1);
		memset(v, 0, sizeof(v));
		id=0;
		for(int i=1; i<=n; i++)
		{
			for(int j=1; j<=m; j++)
			{
				if(G[i][j]=='.'&&!v[i][j])
				{
					ans=0;
					id++;
					dfs(i, j);
					r[id]=ans;	
				//	printf("%d\n", ans);				
				}
			}
		}
		while(k--)
		{
			int x, y;
			scanf("%d%d", &x, &y);
			printf("%d\n", r[v[x][y]]);
		}
	}
	return 0;
}
