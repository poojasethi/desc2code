#include <bits/stdc++.h>
using namespace std;

int N, M;

char S[1003][1003];
int color[1003][1003], ans[1000006];

int dfs(int x, int y, int col)
{
    if(x<0 || y<0 || x>=N || y>=M)
        return 0;

    if(color[x][y])
        return 0;

    if(S[x][y]=='*')
        return 1;

    color[x][y] = col;

    return dfs(x+1, y, col) + dfs(x-1, y, col) + dfs(x, y+1, col) + dfs(x, y-1, col);
}

int main()
{
    int K,l,m;
    scanf("%d %d %d", &N, &M, &K);

    for(int i=0; i<N; i++)
        scanf(" %s", S[i]);

    int k = 1;

    while(K--)
    {
        int x, y;
        scanf("%d %d", &x, &y);

        x--, y--;

        if(!color[x][y])
        {
            ans[k] = dfs(x, y, k);
            k++;
        }

        printf("%d\n", ans[color[x][y]]);
    }

    return 0;
}
