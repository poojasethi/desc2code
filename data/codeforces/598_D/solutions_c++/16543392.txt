#include<cstdio>
#include<cstring>
#include<queue>
#include<algorithm>
using namespace std;
char G[1005][1005];
int vis[1005][1005];
int d[4][2]={1,0,0,1,-1,0,0,-1};
int cnt[1005*1005];
int n,m,k;
void dfs(int x,int y,int id)
{
    if(vis[x][y])return;
    if(G[x][y]=='*'){cnt[id]++;return;}
    vis[x][y]=id;
    for(int i=0;i<4;i++)
    {
        int xx=x+d[i][0];
        int yy=y+d[i][1];
        dfs(xx,yy,id);
    }
}
int main()
{
    scanf("%d%d%d",&n,&m,&k);
    int id=0;
    for(int i=0;i<n;i++)
        scanf("%s",G[i]);
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
            if(G[i][j]=='.'&&!vis[i][j])
                dfs(i,j,++id);
    for(int i=0;i<k;i++)
    {
        int x,y;
        scanf("%d%d",&x,&y);
        x--,y--;
        printf("%d\n",cnt[vis[x][y]]);
    }
    return 0;
}
