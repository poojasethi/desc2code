#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstdio>
using namespace std;
char adj[1000][1000];
int vis[1000][1000];
int sum1[1000000];
int dfs(int x ,int y,int p)
{
    if(adj[x][y]=='*')
        {return 1;}
    if(vis[x][y])return 0;
    vis[x][y]=p;
    int sum = 0;
    sum+=dfs(x+1,y,p);
    sum+=dfs(x-1,y,p);
    sum+=dfs(x,y+1,p);
    sum+=dfs(x,y-1,p);
    sum1[p]=sum;
    return sum;
}
int main()
{
    int n , m , k;
    cin >> n >> m >> k;
    for(int i = 0 ; i < n ; i++)
        for(int j = 0 ; j < m ; j++)
        cin >> adj[i][j];
        int p = 1;
    for(int i = 0 ; i < k ; i++)
    {
        int x , y;
        cin >> x >> y;
        x--;y--;
        if(vis[x][y])cout << sum1[vis[x][y]]<<endl;
        else cout << dfs(x,y,p++)<<endl;
    }
    return 0;
 }
