#include <iostream>
#include <cstdio>

using namespace std;

int a,b,n,m,k, used[2000][2000], ans[200000];
char map[2000][2000];

int dfs(int i, int j, int k)
{
	int answer = 0;
	if (map[i][j] == '*')
		return 1;
	if (used[i][j])
		return 0;
	used[i][j] = k;
	answer += dfs(i,j+1,k);
	answer += dfs(i-1,j,k);
	answer += dfs(i,j-1,k);
	answer += dfs(i+1,j,k);
	
	return answer;
}	

int main()
{
	cin>>n>>m>>k;
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= m; j++)
			cin>>map[i][j];
	for (int i = 1; i <= k; i++)
	{
		scanf("%d%d",&a,&b);
		if (used[a][b])
			printf("%d\n",ans[used[a][b]]);
		else
		{
			ans[i] = dfs(a,b,i);
			printf("%d\n",ans[i]);
		}
	}
	return 0;
}
