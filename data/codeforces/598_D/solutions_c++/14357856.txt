#include<iostream>
#include<string.h>
using namespace std;
int n,m,k,vis[1005][1005],x,y,ans[1000005],v;
char s[1005][1005];
int dfs(int x,int y)
{
    if (s[x][y]=='*')
    {
        ans[v]++;
        return 0;
    }
    vis[x][y]=v;
    if (x+1<=n && vis[x+1][y]==0)
        dfs(x+1,y);
    if (y+1<=m && vis[x][y+1]==0)
        dfs(x,y+1);
    if (x-1>=1 && vis[x-1][y]==0)
        dfs(x-1,y);
    if (y-1>=1 && vis[x][y-1]==0)
        dfs(x,y-1);
}
int main()
{
    cin>>n>>m>>k;
    for (int i=1;i<=n;i++)
        for (int j=1;j<=m;j++)
            cin>>s[i][j];
    while (cin>>x>>y)
    {
        v++;
        if (vis[x][y]!=0)
        {
            cout<<ans[vis[x][y]]<<endl;
            continue;
        }   
        dfs(x,y);
        cout<<ans[v]<<endl;
    }   
}
