#include <iostream>
#include <cstdio>
using namespace std;

int n, m, k, d, x, y, sz, s[100005], num[1005][1005];
string a[1005];
void f(int p, int q, int r) {
	if (a[p][q] == '.') {
		a[p][q] = '-';
		num[p][q] = r;
		if (a[p - 1][q] == '*') d++;
		if (a[p + 1][q] == '*') d++;
		if (a[p][q - 1] == '*') d++;
		if (a[p][q + 1] == '*') d++;
		f(p - 1, q, r);
		f(p + 1, q, r);
		f(p, q - 1, r);
		f(p, q + 1, r);
	}
}

int main()
{
	int i, j;
//	freopen ("input.txt", "r", stdin);
	cin >> n >> m >> k;
	for (i = 1; i <= n; i++) {
		cin >> a[i];
		a[i] = " " + a[i];
	}
	for (i = 0; i < k; i++) {
		scanf ("%d %d", &x, &y);
		if (a[x][y] == '.' && num[x][y] == 0) {
			d = 0;
			sz++;
			f(x, y, sz);
			s[sz] = d;
		}
		printf("%d\n", s[num[x][y]]);
	}
    return 0;
}
