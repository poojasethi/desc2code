#include <iostream>
#include <cstdio>
using namespace std;

int n,m,k,x,y,cnt,num,ans[100005],v[1005][1005];
char a[1005][1005];

void dfs(int p, int q){
    if(a[p][q]=='*') cnt++;
    else if(!v[p][q]){
        v[p][q] = num;
        dfs(p+1,q);
        dfs(p-1,q);
        dfs(p,q+1);
        dfs(p,q-1);
    }
}

int main(){
    int i;
    cin>>n>>m>>k;
    for(i=1; i<=n; i++){
        scanf("%s",&a[i][1]);
    }
    while(k--){
        scanf("%d%d",&x,&y);
        if(!v[x][y]){
            num++;
            cnt = 0;
            dfs(x,y);
            ans[num] = cnt;
        }
        printf("%d\n",ans[v[x][y]]);
    }
    return 0;
}
