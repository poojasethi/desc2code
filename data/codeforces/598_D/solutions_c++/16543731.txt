#include<stdio.h>
#include<string.h>
char map[1005][1005];
int n,m,k,dx[]={-1,1,0,0},dy[]={0,0,-1,1},vis[1005][1005];
int cnt=0,table[1000005];
void dfs(int a,int b,int num)
{
	for(int i=0;i<4;++i)
	{
		int tx=a+dx[i],ty=b+dy[i];
		if(map[tx][ty]=='*')
		{
			++table[num];
			continue;
		}
		if(!vis[tx][ty])
		{
			vis[tx][ty]=num;
			dfs(tx,ty,num);
		}
	}
}
int main()
{
	//freopen("shuju.txt","r",stdin);
	while(~scanf("%d%d%d",&n,&m,&k))
	{
		for(int i=0;i<n;++i)
		{
			scanf("%s",map[i]);
		}
		for(int i=1;i<n;++i)
		{
			for(int j=1;j<m;++j)
			{
				if(map[i][j]=='.'&&!vis[i][j])
				{
					vis[i][j]=++cnt;
					dfs(i,j,cnt);
				}
			}
		}
		int bx,by;
		for(int i=0;i<k;++i)
		{
			scanf("%d%d",&bx,&by);
			printf("%d\n",table[vis[bx-1][by-1]]);
		}
	}	
	return 0;
}
