#include <bits/stdc++.h>
using namespace std;
string board[1001];
int id[1001][1001], n,m,x,y,k, dx[4]={0,0,1,-1},dy[4]={1,-1,0,0};
int dp[100001];
int dfs(int y,int x,int c){
	int ret=0;
	id[y][x]=c;
	for(int i=0;i<4;++i){
		if(board[y+dy[i]][x+dx[i]]=='*')
			++ret;
		else{
			if(id[y+dy[i]][x+dx[i]]==-1)
				ret+=dfs(y+dy[i],x+dx[i],c);
		}
	}
	return ret;
}
int main(){
	cin>>n>>m>>k;string s;
	for(int i=1;i<=n;++i){
		cin>>s;s = '*'+s;
		board[i]=s;
	}
	memset(id,-1,sizeof(id));
	for(int i=0;i<k;++i){
		cin>>y>>x;
		int ret;
		if(id[y][x]!=-1)
			ret= dp[id[y][x]];
		else{
			ret= dfs(y,x,i);dp[i]=ret;
		}
		cout<<ret<<endl;
	}
}
