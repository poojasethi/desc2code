#include <stdio.h>
#include <math.h>
#include <algorithm>

using namespace std;

char F[1010][1010];
int C[1010][1010];
int ans[100100];
int cl = 0;

void dfs(int x, int y)
{
    if (C[x][y])
    {
        return;
    }
    if (F[x][y] == '*')
    {
        ans[cl]++;
        return;
    }
    C[x][y] = cl;
    dfs (x + 1, y);
    dfs (x - 1, y);
    dfs (x, y + 1);
    dfs (x, y - 1);
}

int main()
{
    int n, m, k;
    scanf ("%d%d%d", &n, &m, &k);
    for (int i = 0; i < n; i++)
    {
        scanf ("%s", F[i]);
    }

    for (int i = 0; i < k; i++)
    {
        int x, y;
        scanf ("%d%d", &x, &y);
        x--;
        y--;
        if (!C[x][y])
        {
            cl++;
            dfs(x, y);
        }
        printf ("%d\n", ans[C[x][y]]);
    }
}
