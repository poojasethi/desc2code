#include <iostream>
#include <math.h>
#include <algorithm>
#include <string.h>
#include <cstdio>
#include <queue>
using namespace std;
char mp[1005][1005];
int ok[1005][1005];
int magic[1005 * 1005];
int n, m, k;
int DFS(int x, int y, int c)
{
    if(ok[x][y]) return 0;
    if(mp[x][y]== '*') return 1;
    ok[x][y] = c;
    return DFS(x - 1, y, c) + DFS(x + 1, y, c) + DFS(x, y - 1, c) + DFS(x, y + 1, c);
}

int main()
{
    int x, y, ans;
    scanf("%d%d%d", &n, &m, &k);
    for(int i = 1 ; i <= n ; i++)
        for(int j = 1 ; j <= m ; j++)
    {
        cin >> mp[i][j];
    }
    for(int i = 1 ; i <= k ;i++)
    {
        scanf("%d%d", &x, &y);
        if(ok[x][y]) ans = magic[ok[x][y]];
        else
        {
            ans = DFS(x, y, i);
            magic[i] = ans;
        }
        cout << ans << endl;
    }
    return 0;
}
