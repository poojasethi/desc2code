#include <iostream>
#include <fstream>
using namespace std;
#define MAX 1010

int a[MAX][MAX];
int b[MAX][MAX], s[MAX * MAX], dr, d[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}}, n, m;

void visit(int i, int j)
{
	if(i > n || i < 1 || j > m || j < 1)
		return;
	b[i][j] = dr;
	int k;
	for(k = 0 ; k <= 3 ; k++)
	{
		int ni = i + d[k][0];
		int nj = j + d[k][1];
		if(a[ni][nj])
		{
			s[dr]++;
		}
		else
		{
			if(!b[ni][nj])
				visit(ni, nj);
		}
	}
}

int main() {
	char c;
	int x;
	int i, j;
	cin >> n >> m;
	cin >> x;
	for(i = 1 ; i <= n ; i++)
	{
		for(j = 1 ; j <= m ; j++)
		{
			cin >> c;
			if(c == '.')
				a[i][j] = 0;
			else
				a[i][j] = 1;
		}
	}
	while(x--)
	{
		cin >> i >> j;
		if(b[i][j])
		{
			cout << s[b[i][j]] << "\n";
		}
		else
		{
			++dr;
			visit(i, j);
			cout << s[b[i][j]] << "\n";
		}
	}
	return 0;
}
