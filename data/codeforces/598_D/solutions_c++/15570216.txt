#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int map[1005][1005];
int to[1005][1005];
int ans[1000086];
void ok(int x,int y)
{
	to[x][y]=ans[0];
	for(int i=-1;i<2;i++)
		for(int j=-1;j<2;j++)
		if(i+j==1 || i+j==-1)
		{
			if(map[x+i][y+j]==-1)
			{
				map[x+i][y+j]=-2;
				ok(x+i,y+j);
			}
			if(map[x+i][y+j]==1)
			{
				ans[ans[0]]++;
			}
		}
}
int main() 
{
	int n,m,k,a,b;
	cin>>n>>m>>k;
	char c;
	c=getchar();
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			c=getchar();
			if(c=='.')
				map[i][j]=-1;
			else
				map[i][j]=1;
		}
		c=getchar();
	}
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			if(map[i][j]==-1)
			{
				ans[0]++;
				map[i][j]=-2;
				ok(i,j);
			}
		}
	}
	for(int i=0;i<k;i++)
	{
		cin>>a>>b;
		cout<<ans[to[a][b]]<<endl;
	}
	return 0;
}
