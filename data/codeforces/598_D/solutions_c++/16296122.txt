#include<stdio.h>
#include<algorithm>
#include<string.h>
using namespace std;
char mus[1001][1001];
int load[1001][1001]={0},t[1000001]={0};
int paint,ind=1;
void dfs(int x,int y)
{
	if(load[x][y]||mus[x][y]=='*') return;
	load[x][y] = ind;
	if(mus[x][y+1]=='*') paint ++;
	if(mus[x][y-1]=='*') paint ++;
	if(mus[x+1][y]=='*') paint ++;
	if(mus[x-1][y]=='*') paint ++;	
	dfs(x,y+1);
	dfs(x,y-1);
	dfs(x+1,y);
	dfs(x-1,y);
	return;
}
int main()
{
	int i,j,n,m,k,x,y;
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;i++)
		scanf("%s",mus[i]+1);
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			if(mus[i][j]=='*'||load[i][j]!=0) continue;
			ind ++;
			paint = 0;
			dfs(i,j);
			t[ind]=paint;
		}
	}
	while(k--)
	{
		scanf("%d%d",&x,&y);
		printf("%d\n",t[load[x][y]]);
	}
	return 0;
}
