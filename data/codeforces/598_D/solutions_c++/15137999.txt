using namespace std;
string str[10000];
int hit[1000000];
int visited[1000][1000]={0};
int dfs(int a,int b,int ch)
{
    if(visited[a][b]) return 0;
    if(str[a][b]=='*') return 1;
    visited[a][b]=ch;
    return dfs(a+1,b,ch)+dfs(a-1,b,ch)+dfs(a,b+1,ch)+dfs(a,b-1,ch);
}
int main()
{
    pair<int,int> p;
    int m,n,k,f=0,x,y;
    cin>>m>>n>>k;
    for(int i=0;i<m;i++)
        cin>>str[i];
    int ch=1;
    for(int i=0;i<k;i++){
        cin>>x>>y;
        if(visited[x-1][y-1]==0){
            hit[ch]=dfs(x-1,y-1,ch);
            cout<<hit[ch]<<endl;
            ch++;

        }
        else cout<<hit[visited[x-1][y-1]]<<endl;
    }
    return 0;
}
