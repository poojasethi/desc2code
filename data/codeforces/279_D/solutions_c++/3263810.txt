#include <cstdio>
#include <cstdlib>
#include <vector>
using namespace std;

const int INF = 1000000;
const int N = 23;
int a[N];
int q[1<<N];
int n, res = INF;

int calc(int k, int x)
{
    if(q[x]) return q[x];
    if(k == 0) return q[x] = 1;
    q[x] = INF;
    
    int y = x ^ (1 << k);
    for(int i = 0; i < k; ++i)
        for(int j = i; j < k; ++j)
            if(a[i] + a[j] == a[k]) {
                int p = calc(k-1, y | (1 << i) | (1 << j) | (1 << (k-1)));
                if(q[x] > p) q[x] = p;
            }
            
    int p = __builtin_popcount(x);
    if(q[x] < p) q[x] = p;
    return q[x];
}

int main()
{
    scanf("%d", &n);
    for(int i = 0; i < n; ++i)
        scanf("%d", a+i);
    calc(n-1, 1 << (n-1));
    if(q[1<<(n-1)] == INF) puts("-1");
    else printf("%d\n", q[1<<(n-1)]);
    return 0;
}
