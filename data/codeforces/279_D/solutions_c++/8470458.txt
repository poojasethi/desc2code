#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
int n,a[25],f[1<<23],stp[1<<23];
int dp(int stu,int x)
{
    if(f[stu])return f[stu];
    if(x==1)return 1;
    int ans=n+1;
    for(int i=1;i<x;i++)
        for(int j=1;j<x;j++)
            if(a[i]+a[j]==a[x])
            {
                int t=dp(stu&(~(1<<x-1))|(1<<i-1)|(1<<j-1)|(1<<x-2),x-1);
                ans=min(ans,max(t,stp[stu]));
            }
    return f[stu]=ans;
}
void Prework()
{
    for(int stu=0;stu<(1<<23);stu++)
        for(int i=0;i<23;i++)
            if(stu&(1<<i))stp[stu]++;
}
void Solve()
{
    Prework();
    while(~scanf("%d",&n))
    {
        for(int i=1;i<=n;i++)scanf("%d",a+i);
        memset(f,0,sizeof(f));
        int ans=dp(1<<n-1,n);
        if(ans==n+1)printf("-1\n");
        else printf("%d\n",ans);
    }
}
int main()
{
    Solve();
    return 0;
}
