#include<bits/stdc++.h>
#define pb push_back
#define sz(x) (int)x.size()
#define scf scanf
#define ptf printf
#define forp(i,j,k) for(int i=j;i<k;i++)
#define form(i,j,k) for(int i=j;i>k;i--)
using namespace std;
typedef long long LL;
const int oo=1e8;
int f[1<<24];
LL a[100];

inline int cc(int x)
{
    int res=0;
    for(;x;x-=x&-x)res++;
    return res;
}

int dp(int st,int x)
{
    if(f[st]!=-1)return f[st];
    if(!x)return 1;
    int res=oo,tmp=cc(st);
    forp(i,0,x)forp(j,0,x)
        if(a[i]+a[j]==a[x])
            res=min(res,max(tmp,dp((st^(1<<x))|(1<<(x-1))|(1<<i)|(1<<j),x-1)));
    return f[st]=res;
}

int main()
{
    int n,r;
    scf("%d",&n);
    forp(i,0,n)scf("%I64d",a+i);
    memset(f,-1,sizeof(f));
    r=dp(1<<(n-1),n-1);
    ptf("%d\n",r<oo?r:-1);
    return 0;
}
