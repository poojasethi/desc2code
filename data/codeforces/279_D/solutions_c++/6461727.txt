#include <iostream>
#include <algorithm>
using namespace std;

const int infi = 110;
const int N = 24;

int bits(int n)
{
    if ( n == 0 ) return 0;
    return 1 + bits(n & (n-1));
}

int a[N],ans[1<<N],n;

int df(int c,int x)
{
    if ( ans[c] ) return ans[c];
    if ( c == 1 ) return 1;
    int cnt = bits(c), mn = infi;

    for (int i=0;i<x;++i)
        for (int j=0;j<=i;++j)
            if ( a[i] + a[j] == a[x] )
            {
                int v = df( (c & ~(1<<x)) | (1<<(x-1)) | (1<<i) | (1<<j) , x-1 );
                mn = min(mn,max(v,cnt));
            }
    return ans[c] = mn;
}

int main()
{
    cin>>n;
    for (int i=0;i<n;++i)
        cin>>a[i];
    int ans = df(1<<(n-1),n-1);
    if ( ans == infi )
        cout<<"-1\n";
    else
        cout<<ans<<'\n';
}
