using namespace std;
typedef long long ll;
const int inf=1<<28;

int dp[1<<23],a[25];

int check(int x,int k)
{
	for(int i=0;i<k;i++){
		if((x>>i)&1){
			for(int j=i;j<k;j++)
				if(((x>>j)&1)&&a[i]+a[j]==a[k]) return 1;
		}
	}
	return 0;
}
int cnt(int x)
{
	int ans=0;
	while(x){
		if(x%2) ans++;
		x/=2;
	}
	return ans;
}
int main() {

	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	dp[1]=1;
	for(int i=1;i<n;i++){
		for(int j=1<<(i-1);j<(1<<i);j++){
			if(!dp[j]) continue;
			if(check(j,i)){
				dp[j^(1<<i)]=1;
				for(int k=0;k<i;k++)
					if((j>>k)&1) dp[j^(1<<i)^(1<<k)]=1;
			}
		}
	}
	int ans=inf;
	for(int i=1<<(n-1);i<(1<<n);i++){
		if(dp[i])
			ans=min(ans,cnt(i));
	}
	if(ans==inf) ans=-1;
	printf("%d\n",ans);
}
