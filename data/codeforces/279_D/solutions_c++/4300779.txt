#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#include <cstring>
#define INF 99999999
using namespace std;

int a[24], n, d[1<<23];
const int inf = (-1u>>1);

int doit(int st, int x){
	if (d[st] != -1)return d[st];
	int c = __builtin_popcount(st);
	int temp = inf;
	for (int i = 0; i < x; i++){
		for (int j = 0; j <= i; j++){
			if (a[i] + a[j] == a[x]){
				int v = doit(st & ~(1<<x) | (1<<i) | (1<<j) | (1<<(x-1)), x-1);
				temp = min(temp, max(c, v));
			}
		}
	}
	return d[st] = temp;
}





int main (){
	cin >> n;
	for(int i = 0; i < n; i++) cin >> a[i];
	memset(d, -1, sizeof(d));
	d[1] = 1;
	int ans = doit(1<<(n-1), n-1);
	if (ans == inf)ans = -1;
	cout<<ans<<endl;
	return 0;
	
}
