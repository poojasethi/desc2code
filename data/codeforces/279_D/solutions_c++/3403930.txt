#include "iostream"
#include "stdio.h"
#include "string.h"
#include "stdlib.h"
#include "algorithm"
#include "cmath"

using namespace std;

int n,a[25],dp[1<<23];

int fun(int state,int pos)
{
    if(dp[state]) return dp[state];
    int now = __builtin_popcount(state),ret = 999;
    for(int i = 0; i < pos; i ++)
    {
        for(int j = 0; j <= i; j ++)
        {
            if(a[i] + a[j] == a[pos])
            {
                int temp=(state|(1<<i)|(1<<j)|(1<<(pos-1)))-(1<<pos);
                int ans=fun(temp,pos-1);
                ret= min(ret,max(ans,now));
            }
        }
    }
    return dp[state] = ret;
}
int main()
{
    while(~scanf("%d",&n))
    {
        memset(dp,0,sizeof(dp));
        for(int i = 0; i < n; i ++)
            scanf("%d",&a[i]);
        dp[1] = 1;
        int ans = fun(1<<(n-1),n-1);
        if(ans == 999) ans = -1;
        printf("%d\n",ans);
    }
    return 0;
}
