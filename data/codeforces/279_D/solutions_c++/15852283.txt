#include<cstdio>
#include<cstring>
const int inf=0x3f3f3f3f;
int f[16777215];
int a[24];
int n;
int C(int x)
{
    int i,ans;
    ans=0;
    for (i=0;i<=n-1;i++)
        if (x&(1<<i)) ans++;
    return ans;
}
int gmin(int x,int y)
{
    return x<y?x:y;
}
int gmax(int x,int y)
{
    return x>y?x:y;
}
int dp(int s,int h)
{
    int ans,i,j;
    if (f[s]!=-1) return f[s];
    ans=inf;
    for (i=0;i<=h-1;i++)
        for (j=0;j<=i;j++)
            if (a[i]+a[j]==a[h]) ans=gmin(ans,dp(s&(~(1<<h))|(1<<i)|(1<<j)|(1<<(h-1)),h-1));
    f[s]=gmax(C(s),ans);
    return f[s];
}
int main()
{
    int i,ans;
    scanf("%d",&n);
    for (i=0;i<=n-1;i++) scanf("%d",&a[i]);
    memset(f,0xff,sizeof(f));
    f[1]=1; ans=dp(1<<(n-1),n-1);
    if (ans<inf) printf("%d\n",ans);
    else printf("-1\n");
    return 0;
}
