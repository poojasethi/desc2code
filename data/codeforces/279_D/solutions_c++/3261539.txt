#include<cstdio>
#include<cmath>
#include<iostream>
using namespace std;
#define INF 25
int n,a[25],dp[1<<23];
int fun(int m,int x){//扫描到第x位至少需要多少个中继
    if(dp[m]) return dp[m];
    int c = __builtin_popcount(m),v = 0,mn = INF;
    for(int i = 0; i < x; i ++){
        for(int j = 0; j <= i; j ++){
            if(a[i] + a[j] == a[x]){
                v = fun(m&~(1<<x)|(1<<x-1)|1<<i|1<<j,x-1);
                mn = min(mn,max(v,c));
            }
        }
    }
    return dp[m] = mn;
}
int main()
{
        scanf("%d",&n);
        for(int i = 0; i < n; i ++)
            scanf("%d",&a[i]);
        dp[1] = 1;
        int ans = fun(1<<n-1,n-1);
        if(ans == 25) ans = -1;
        printf("%d\n",ans);
return 0;
}
