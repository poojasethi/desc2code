#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

#define popcount(a) __builtin_popcount(a)

const int INF = 0x7fffffff;

int f[1<<23],a[25],n;

int F(int st,int num){
	if(f[st]) return f[st];
	if(num==1) return f[st]=1;
	int r=INF;
	for(int i=num-1;i>=1;i--)
		for(int j=i;j>=1;j--) if(a[i]+a[j]==a[num])
			r=min(r,F(st^(1<<num-1)|(1<<i-1)|(1<<j-1)|(1<<num-2),num-1));
	return f[st]=max(r,popcount(st));
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	int res=F(1<<(n-1),n);
	printf("%d\n",res<INF ? res : -1);
	return 0;
}
