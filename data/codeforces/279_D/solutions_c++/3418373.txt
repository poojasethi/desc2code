#include <cstdio>
#include <algorithm>

#define CS				const static
#define Rep(i,l,r)			for(int i = (l) ; i <= (r) ; i++)
#define RepD(i,l,r)			for(int i = (l) ; i >= (r) ; i--)

CS int N = 25 ;

using namespace std ;

int a[N] , v[1<<24] ;

int n ;

int bits(int x)	{int tot ; for(tot = 0 ; x ; x -= x & (-x) , tot++) ; return tot ;}

int Dfs(int set,int dep)
{
	if(v[set])	return v[set] ;
	if(dep==1)	return 1 ;
	int ans = 110 ;
	Rep(i,1,dep-1)
		Rep(j,1,i)
			if(a[i]+a[j]==a[dep])
			{
				int u = Dfs(set^(1<<(dep-1))|(1<<(i-1))|(1<<(j-1))|(1<<(dep-2)),dep - 1) ;
				int cnt = max(bits(set) , u) ;
				ans = min(ans , cnt) ;
			}
	return v[set] = ans ;
}

int main()
{
	scanf("%d",&n) ;
	Rep(i,1,n)	scanf("%d",&a[i]) ;

	int ans = Dfs(1<<(n-1),n) ;
	printf("%d\n",ans > 100 ? -1 : ans) ;

	return 0 ;
}
