#include<bits/stdc++.h>

using namespace std;

int rmq[6][200005][18],D[100005][4];
int d,n,s,i,j,mx,k,w;
int M[15];

int init(int x){
	int i,j;
	
	for(j=1 ; j<=17 ; j++)
		for(i=1 ; i<=d ; i++)
			rmq[x][i][j] = max(rmq[x][i][j-1],rmq[x][i+(1<<(j-1))][j-1]);	
}

int que(int x,int a,int b){
	int l=floor(log2(b-a+1));
	return max(rmq[x][a][l],rmq[x][b-(1<<l)+1][l]);
}

int main(){

	scanf("%d %d %d",&d,&n,&s);

	for(i=1 ; i<=d ; i++)
		for(j=1 ; j<=n ; j++)
		{
			scanf("%d",&D[i][j]);
			rmq[j][i][0] = D[i][j];
		}
	
	for(i=1 ; i<=n ; i++)
		init(i);

	for(i=1 , j=1 ; i<=d && j<=d ; )
	{		
		w = 0;

		for(k=1 ; k<=n ; k++)
			w += que(k,i,j);
		
		if(w <= s)
		{
			if(mx < j-i+1)
			{	
				mx = j-i+1;
				
				for(k=1 ; k<=n ; k++)
					M[k] = que(k,i,j);
			}

			j++;
		}

		else
			i++;
	}	

	if(mx == 0)
	{
		cout << s <<' ';
		for(k=2 ; k<=n ; k++)
			cout << 0<<' ';
	}
	
	else
	{
		for(k=1 ; k<=n ; k++)
			cout << M[k] << ' ';
	}
	return 0;
}
