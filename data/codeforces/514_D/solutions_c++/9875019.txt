#include<bits/stdc++.h>
using namespace std;
#define lli long long int

int main()
{
	int n,m,k;
	int arr[100001][5];
	int i,j;
	cin>>n>>m>>k;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++) cin>>arr[i][j];
	}
	deque< pair< int,int > > window[5];
	int start=0,end=0,ma[5]={0},glob[5],len=0,sum=0;
	for(i=0;i<n;i++)
	{
		sum=0;
		for(j=0;j<m;j++){ 
			ma[j]=max(ma[j],arr[i][j]); 
			sum+=ma[j];
			while(!window[j].empty()&&window[j].back().first<=arr[i][j]) window[j].pop_back();
			window[j].push_back( pair< int,int > (arr[i][j],i) );
		}
		if(sum>k){
			while(start<i&&sum>k)
			{
				start++;
				sum=0;
				for(j=0;j<m;j++){
					while(!window[j].empty()&&window[j].front().second<start) window[j].pop_front();
					sum+=window[j].front().first;
					ma[j]=window[j].front().first;
				}
			}
			if(sum>k){
				for(j=0;j<m;j++){ma[j]=0; window[j].clear();}
				start=i+1;
				continue;
			}
		}
		else
		{
			if(len<i-start+1)
			{
				len=i-start+1;
				for(j=0;j<m;j++) glob[j]=ma[j];
			}
		}
	}
	for(j=0;j<m;j++) cout<<glob[j]<<" ";
		return 0;
}
