#include <iostream>
#include <deque>
#include <algorithm>
#include <queue>
#include <cstdio>
#include <set>
using namespace std;

multiset<int,greater<int> > S[5];
//priority_queue<int> Q[6];
int n,m,k;
int a[100005][5];
int ans[5];

void input(){
  scanf("%d%d%d",&n,&m,&k);  
  for(int i=0;i<n;++i){
    for(int j=0;j<m;++j){
      scanf("%d",&a[i][j]);
    }
  }
}

int main(){
  int l=0;
  int maxlen = 0;
  input();
  for(int r=0;r<n;++r){
    for(int j=0;j<m;++j){
      S[j].insert(a[r][j]);
    }
    while(!S[0].empty()){
      int K = 0;
      for(int j=0;j<m;++j) K+=*S[j].begin();
      if(K<=k) break;
      for(int j=0;j<m;++j){
	S[j].erase(S[j].find(a[l][j]));
      }
      ++l;
    }
    if(r-l+1 > maxlen){
      maxlen = r-l+1;
      for(int j=0;j<m;++j) ans[j] = *S[j].begin();
    }
  }
  for(int i=0;i<m;++i) cout << ans[i] << " ";
  cout << endl;
  return 0;
}
