#include <bits/stdc++.h>
using namespace std;
int s, n, m, k, ans[6], a[100001][6], M, l, r;
multiset<int> ms[6];
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin >> n >> m >> k;
    for(int i=0; i<n; i++){
        for(int j=0; j<m; j++){
            cin >> a[i][j];
        }
    }
    for(;r<n;r++){
        for(int i=0; i<m; i++){
            ms[i].insert(a[r][i]);
        }
        for(;l<=r;l++){
            s=0;
            for(int i=0; i<m; i++){
                s+=*ms[i].rbegin();
            }
            if(s<=k)break;
            for(int i=0; i<m; i++){
                ms[i].erase(ms[i].find(a[l][i]));
            }
        }
        if(M<r-l+1){
            M=r-l+1;
            for(int i=0; i<m; i++){
                ans[i]=*ms[i].rbegin();
            }
        }
    }
    for(int i=0; i<m; i++){
        cout << ans[i] << ' ';
    }

}
