#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

const int N=100010;
int t[5],b[5],f[5][N][18],m,n,k;

int main()
{
   // freopen("a.txt","r",stdin);
    scanf("%d%d%d",&n,&m,&k);
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
            scanf("%d",&f[j][i][0]);
    }
    int len=int(log(n)/log(2));
    for(int j=0;j<m;j++)
        for(int q=1;q<=len;q++)
            for(int i=0;i-1+(1<<q)<n;i++)
                f[j][i][q]=max(f[j][i][q-1],f[j][i+(1<<(q-1))][q-1]);
    int tot=0,ans=0;
    for(int l=0,r=0;r<n;)
    {
        int tot=0;
        int p=(int)(log((r-l+1))/log(2));
        for(int j=0;j<m;j++)
        {
            t[j]=max(f[j][l][p],f[j][r-(1<<p)+1][p]);
            tot+=t[j];
        }
        if(tot<=k) r++;else l++;
        if(l>r) r++;
    //    cout<<l<<" "<<r<<" "<<tot<<endl;
        //ans=max(ans,r-l);
        if(ans<r-l)
        {
            ans=r-l;
            for(int j=0;j<m;j++) b[j]=t[j];
        }
    }
    for(int i=0;i<m-1;i++) printf("%d ",b[i]);
    printf("%d\n",b[m-1]);
    return 0;
}
