#include <cstdio>

const int N = 100000;

int n, m, k;
int a[5][N], b[5][N], p[5], q[5];
int ans[5], ansl;

int getsum() {
	int ans = 0;
	for (int i = 0; i < m; i++) {
		if (p[i] != q[i]) {
			ans += a[i][b[i][p[i]]];
		}
	}
	return ans;
}
int main() {
	scanf("%d%d%d", &n, &m, &k);
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			scanf("%d", &a[j][i]);
		}
	}
	for (int l = 0, r = 0, sum = 0; l < n; l++) {
		for (int i = 0; i < m; i++) {
			while (p[i] < q[i] && b[i][p[i]] < l) {
				p[i]++;
			}
		}
		while (r < n && getsum() <= k) {
			for (int i = 0; i < m; i++) {
				while (p[i] < q[i] && a[i][b[i][q[i] - 1]] < a[i][r]) {
					q[i]--;
				}
				b[i][q[i]++] = r;
			}
			r++;
			if (r - l > ansl && getsum() <= k) {
				ansl = r - l;
				for (int i = 0; i < m; i++) {
					ans[i] = a[i][b[i][p[i]]];
				}
			}
		}
	}
	for (int j = 0; j < m; j++) {
		printf("%d ", ans[j]);
	}
	printf("\n");
	return 0;
}
