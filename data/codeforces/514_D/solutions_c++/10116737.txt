#include <cstdio>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstring>
#include <queue>

typedef long long LL;
using namespace std;
deque<int> q[10];
int n,m,k,dq,l,r;
int a[200000][10];
int ans[10];
bool solve(int len)
{
	for (int i=1;i<=m;++i) 
		while (!q[i].empty()) q[i].pop_front();
	for (int i=1;i<=n;++i)
	{
		for (int j=1;j<=m;++j)
		{
			while (!q[j].empty()&&q[j].back()<a[i][j]) q[j].pop_back();
			q[j].push_back(a[i][j]);
		}
		if (i<len) continue;
		dq=0;
		for (int j=1;j<=m;++j) dq+=q[j].front();
		if (dq<=k) 
		{
			for (int j=1;j<=m;++j) ans[j]=q[j].front();
			return true;
		}
		for (int j=1;j<=m;++j)
			if (q[j].front()==a[i-len+1][j]) q[j].pop_front();
	}
	return false;
}
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for (int i=1;i<=n;++i)
		for (int j=1;j<=m;++j) scanf("%d",&a[i][j]);
	l=1;r=n;
	while (l<=r)
	{
		int mid=(l+r)/2;
		if (solve(mid)) l=mid+1;else r=mid-1;
	}
	for (int i=1;i<=m;++i) printf("%d ",ans[i]);
	return 0;
}
