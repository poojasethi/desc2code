#include <bits/stdc++.h>
#define MAX 111111

using namespace std;

int A[MAX][10];
int N, M, K;

int B[10], R[10];
multiset<int> S[10];

int update()
{
    int k = 0;
    for (int i = 1; i <= M; i++)
    {
        B[i] = *S[i].rbegin();
        k += B[i];
    }
    return k;
}

int main()
{
    cin >> N >> M >> K;
    for (int i = 1; i <= N; i++) for (int j = 1; j <= M; j++)
        cin >> A[i][j];
    for (int j = 1; j <= M; j++) S[j].insert(0);
    int l = 1, best = 0;
    for (int r = 1; r <= N; r++)
    {
        int k = 0;
        for (int j = 1; j <= M; j++) S[j].insert(A[r][j]);
        k = update();
        int cur = 0;
        while (k > K)
        {
            for (int j = 1; j <= M; j++)
                S[j].erase(S[j].find(A[l][j]));
            k = update();
            l++;
        }
        if (r - l + 1 > best)
        {
            best = r - l + 1;
            for (int j = 1; j <= M; j++) R[j] = B[j];
        }
    }
    for (int j = 1; j <= M; j++)    
        cout << R[j] << ' ';
}
