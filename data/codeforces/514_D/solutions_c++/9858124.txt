#include <stdio.h>
#include <deque>

using namespace std;

int a[100009][5];

deque<int> b[5];

int ans[5];
int ansVal;

int main()
{
    int n, m, maxSum;
    scanf("%d %d %d", &n, &m, &maxSum);

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            scanf("%d", &a[i][j]);
        }
    }

    int j = -1;
    for (int i = 0; i < n; i++) {
        for (int k = 0; k < m; k++)
            while (b[k].size() && b[k].front() < i) b[k].pop_front();

        while (j+1 < n) {
            j++;
            for (int k = 0; k < m; k++) {
                while (b[k].size() && a[b[k].back()][k] <= a[j][k]) b[k].pop_back();
                b[k].push_back(j);
            }

            int tmpSum = 0;
            for (int k = 0; k < m; k++)
                tmpSum += a[b[k].front()][k];

            if (tmpSum > maxSum) break;

            if (j-i+1 > ansVal) {
                ansVal = j-i+1;
                for (int k = 0; k < m; k++) ans[k] = a[b[k].front()][k];
            }
        }
    }

    for (int k = 0; k < m; k++) printf("%d ", ans[k]);
}
