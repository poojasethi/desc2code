#include<iostream>
#include<stdio.h>
#include<string.h>
#include<set>

using namespace std; 

int N,M,K,num[100003][5],ans[5],answer; 
multiset<int> se[5]; 
int main() 
{ 
	scanf("%d%d%d",&N,&M,&K); 
	for (int i=0;i<N;i++) 
		for (int j=0;j<M;j++) 
			scanf("%d",&num[i][j]);
	for (int i=0;i<M;i++) 
		se[i].clear(); 
	int L=0,R=-1; 
	answer=0; 
	while (1) 
	{ 
		int sum=0;  
		for (int i=0;i<M;i++) 
			if (  (int) se[i].size()>0)
				sum=sum+ ( * ( --se[i].end() ) ); 
		if ( sum<=K &&  answer<R-L+1) 
		{ 
			for (int i=0;i<M;i++) 
				ans[i]=( * ( --se[i].end() ) ) ; 
			answer=R-L+1; 
		}
		if ( R==N-1) break; 
		if ( sum<=K) 
		{ 
			R++; 
			for (int i=0;i<M;i++) 
				se[i].insert(num[R][i]); 
		}else 
		{
			for ( int i=0;i<M;i++) 
				se[i].erase( se[i].find( num[L][i]) ); 
			L++; 
		}
		
	}
	for ( int i=0;i<M;i++) 
			printf("%d ",ans[i]); 
	return 0; 
}
