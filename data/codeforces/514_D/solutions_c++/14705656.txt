#include <bits/stdc++.h>

using namespace std;

const int N = 100005;

int n, m, k;
int a[6][N], st[6][N];
int fr[6], b[6], res[6];

void Push(int x)
{
  for(int i = 1; i <= m; i++){
    while(b[i] >= fr[i] && a[i][x] >= a[i][st[i][b[i]]]) b[i]--;
    st[i][++b[i]] = x;
  }
}

bool check(int cur)
{
  int sum = 0;
  for(int i = 1; i <= m; i++){
    while(fr[i] <= b[i] && cur > st[i][fr[i]]) fr[i]++;
    if(fr[i] <= b[i]) sum += a[i][st[i][fr[i]]];
  }
  return (sum > k);
}

int main()
{
  //freopen("inp.txt", "r", stdin);
  cin >> n >> m >> k;
  for(int i = 1; i <= n; i++)
    for(int j = 1; j <= m; j++) scanf("%d", &a[j][i]);
  for(int i = 1; i <= m; i++) fr[i] = 1, b[i] = 0;
  int cur = 1;
  int ans = 0;
  for(int i = 1; i <= n; i++){
    Push(i);
    while(check(cur) && cur <= i) cur++;
    if(ans < i - cur + 1){
      ans = i - cur + 1;
      for(int j = 1; j <= m; j++) res[j] = a[j][st[j][fr[j]]];
    }
  }
  for(int i = 1; i <= m; i++) cout << res[i] << " ";
  return 0;
}
