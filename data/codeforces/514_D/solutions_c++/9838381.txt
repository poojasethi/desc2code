#include <cstdio>
#include <algorithm>
#define N 100005
#define fi(a, b, c) for(int a = (b); a < (c); a++)
#define fd(a, b, c) for(int a = (b); a > (c); a--)
#define FI(a, b, c) for(int a = (b); a <= (c); a++)
#define FD(a, b, c) for(int a = (b); a >= (c); a--)
#define fe(a, b, c) for(int a = (b); a; a = c[a])
using namespace std;

int n, m, k, x[5][N], q[5][N], l[5], h[5], a[5], left, ans;

int main(){
	scanf("%d %d %d", &n, &m, &k);
	fi(i, 0, n){
		fi(j, 0, m){
			scanf("%d", &x[j][i]);
			while(l[j] < h[j] && x[j][q[j][h[j] - 1]] < x[j][i]) h[j]--;
			q[j][h[j]++] = i;
		}
		
		while(true){
			int sum = 0;
			fi(j, 0, m) if(l[j] < h[j]) sum += x[j][q[j][l[j]]];
			if(sum <= k) break;
			left++;
			fi(j, 0, m) while(l[j] < h[j] && q[j][l[j]] < left) l[j]++;
		}
		
		if(i + 1 - left > ans){
			ans = i - left;
			fi(j, 0, m) a[j] = x[j][q[j][l[j]]];
		}
	}
	
	fi(i, 0, m) printf("%d\n", a[i]);
	scanf("\n");
}
