#include<cstdio>
#include<cmath>
#define N 100010
using namespace std;
int n,m,k,i,j,sum,s[6][N][20];
int L,R,M,S,flag;
int max(int a,int b)
{
	if (a>b) return a;return b;
}
int q(int x,int l,int r)
{
	int k;
	k=log(r-l+1)/log(2);
	return max(s[x][l][k],s[x][r-(1<<k)+1][k]);
}

int main()
{
	scanf("%d%d%d",&n,&m,&sum);
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=m;j++)
		scanf("%d",&s[j][i][0]);
	}
	for (j=1;j<=m;j++)
	{
		for (i=n;i>=1;i--)
		for (k=1;k<=log(n)/log(2);k++)
		if (i+(1<<k)-1<=n)
		s[j][i][k]=max(s[j][i][k-1],s[j][i+(1<<(k-1))][k-1]);
	}
	L=0;R=n;
	while (L<=R)
	{
		M=(L+R)/2;
		flag=0;
		for (i=1;i<=n-M+1;i++)
		{
			S=0;
			for (j=1;j<=m;j++)
			S+=q(j,i,i+M-1);
			if (S<=sum) flag=1;
		}
		if (flag)
		L=M+1;
		else
		R=M-1;
	}
	for (i=1;i<=n-R+1;i++)
	{
			S=0;
			for (j=1;j<=m;j++)
			S+=q(j,i,i+R-1);
			if (S<=sum) 
			{
				for (j=1;j<=m;j++)
					printf("%d ",q(j,i,i+R-1));
				return 0;
			}
	}
}
