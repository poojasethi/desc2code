#include<bits/stdc++.h>
using namespace std;
const int tr=524288;
long long a,s,d[5][2*tr+2],f[100002][5],g,h,j,k,l,i,n,m;
int ins(int idx){
while(idx>=1){//cout<<idx<<endl;
    if(d[a][idx]<l) d[a][idx]=l; else return 0;
    idx/=2;
}

}

int mx(int idx,int st,int end){//cout<<idx<<" "<<st<<" "<<end<<" *"<<d[j][idx]<<"* "<<i;//system("pause");
if(st>i || end<a) {/*cout<<"    brk"<<endl;*/return 0;}
if(st>=a && end<=i) {/*cout<<"    add"<<endl;*/return d[j][idx];}
if(st<=a || end>=i) {/*cout<<"    split"<<endl;*/return max(mx(idx*2,st,(st+end)/2),mx(idx*2+1,(st+end)/2+1,end));}


}
main(){
cin>>n>>m>>k;
for(i=0;i<n;i++){
    for(a=0;a<m;a++){
      cin>>l;
      ins(tr+i);
    }
}
a=0;
//cout<<k;8
g=n+1;
for(i=0;i<n;i++){
    s=0;
  for(j=0;j<m;j++){
    f[i][j]=mx(1,0,tr-1);
    s=s+f[i][j];
  }//cout<<s<<" "<<k;//cout<<" ^"<<i<<" "<<a<<" "<<s<<"^ ";system("pause");
  if(s<=k) {if(h<i-a+1){h=i-a+1;g=i;}} else {a++;i=max(i-1,a-1);}
}
for(i=0;i<m;i++){
    cout<<f[g][i]<<" ";
}
//cout<<h;
}
