#include <cstdio>
#include <iostream>
#include <string>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;

int n, m, k, a[100005][5], qx[5][100005], qy[5][100005], H[5], T[5], b[5];

int main()
{
	scanf ("%d%d%d", &n, &m, &k);
	int ans = 0, sum = 0, x = 1, y = 0;
	for (int i = 0; i < m; i++) H[i] = 1;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			scanf ("%d", &a[i][j]);
			if (H[j] <= T[j]) sum -= qy[j][H[j]];
			while ((H[j] <= T[j]) && (a[i][j] >= qy[j][T[j]])) T[j]--;
			T[j]++;
			qx[j][T[j]] = i;
			qy[j][T[j]] = a[i][j];
			sum += qy[j][H[j]];
		}
		y ++;
		while ((x <= i) && (sum > k))
		{
			for (int j = 0; j < m; j++)
				if (H[j] <= T[j] && qx[j][H[j]] == x) H[j]++;
			sum = 0;
			for (int j = 0; j < m; j++)
				if (H[j] <= T[j]) sum += qy[j][H[j]];
			x++; y--;
		}
		if (y > ans)
		{
			ans = y;
			for (int j = 0; j < m; j++) b[j] = qy[j][H[j]];
		}
	}
	for (int i = 0; i < m; i++) printf("%d ", b[i]);
	printf("\n");
	return 0;
}
