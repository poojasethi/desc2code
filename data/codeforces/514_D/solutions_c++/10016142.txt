#include <iostream>
#include <string>
#include <string.h>
#include <cstdio>
#include <deque>

using namespace std;
const int maxn = 1e5+5;

int n,m,k;
int a[maxn][10];

int ans[10] , maxlen;
int que[10][maxn] , l[10],r[10];

int main(){	

	while ( scanf("%d%d%d",&n,&m,&k)!=EOF){
		for(int i=1;i<=n;i++)
			for(int j=1;j<=m;j++) 
				scanf("%d",&a[i][j]);

		maxlen = 0;
		for(int i=1;i<=m;i++) l[i] = 1 , r[i] = 0;
		int sum , ll = 1;
		for(int i=1;i<=n;i++){
			for(int j=1;j<=m;j++){
				while ( l[j]<=r[j] && 
					a[ que[j][r[j]] ][j]<=a[i][j] )
					r[j]--;
				que[j][++r[j]] = i;
			}
			while(1){
				sum = 0;
				for(int j=1;j<=m;j++)
					sum += a[ que[j][l[j]] ][j];
				if ( sum<=k ) {
					if ( i-ll+1 > maxlen ){
						maxlen = i-ll+1;
						for(int j=1;j<=m;j++)
							ans[j] = a[ que[j][l[j]] ][j];
					}
					break;
				}
				else {
					ll++;
					for(int j=1;j<=m;j++)
						if ( que[j][l[j]]<ll ) 
							l[j]++;
				}
			}
		}
		//cout<<maxlen<<endl;
		for(int i=1;i<=m;i++)
			printf("%d ",ans[i] );
		printf("\n");
	}

	return 0;
}
