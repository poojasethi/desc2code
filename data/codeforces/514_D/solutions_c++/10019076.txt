#include<cstdio>
#include<cstdlib>
#include<cstring>

#define N 100100

int n,m,k,a[N][8],Max,h,w[8][N],l[N],r[N],tot,ans[8];

inline int value()
{
	int tot=0; for(int j=1;j<=m;j++) tot+=a[ w[j][ l[j] ] ][j]; return tot;
}
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=n;i++) 
		for(int j=1;j<=m;j++) 
			scanf("%d",&a[i][j]);
	Max=h=0;
	for(int i=1;i<=m;i++) l[i]=1;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			 while(a[i][j]>= a[ w[j][ r[j] ] ][j] && r[j]>=l[j] ) r[j]--;
			 w[j][++r[j]]=i;
		}
		//		
		while(value()>k && h<i)
		{
			h++;
			for(int j=1;j<=m;j++)
			while(w[j][ l[j] ]<= h && l[j]<=r[j]) l[j]++;	
		}
		
		if(i-h>Max)
		{
			Max=i-h;
			for(int j=1;j<=m;j++) ans[j]=a[ w[j][ l[j] ] ][j];
		}
	}
	
	printf("%d",ans[1]); 
	for(int i=2;i<=m;i++)printf(" %d",ans[i]);
	return 0;
}
