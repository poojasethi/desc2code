#include <cstdio>
#include <iostream>
#define N 100010
#define M 10
using namespace std;
int n,m,k,a[N][M],L[M],R[M],Q[M][N],ans[M];
bool check(int len){
	int minstep=k+1;
	for(int i=1;i<=m;++i)L[i]=1,R[i]=0;
	for(int i=1;i<=n;++i){
		int step=0;
		for(int j=1;j<=m;++j){
			while(L[j]<=R[j] && a[i][j]>a[Q[j][R[j]]][j])--R[j];
			Q[j][++R[j]]=i;
			while(L[j]<=R[j] && Q[j][L[j]]<=i-len)++L[j];
			step+=a[Q[j][L[j]]][j];
		}
		if(i<len)continue;
		if(step<minstep){
			minstep=step;
			for(int j=1;j<=m;++j)ans[j]=a[Q[j][L[j]]][j];
		}
	}
	return minstep<=k;
}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=n;++i)
	    for(int j=1;j<=m;++j)scanf("%d",&a[i][j]);
    int l=0,r=n;
    while(l<=r){
        int mid=(l+r)>>1;
        if(check(mid))l=mid+1;
        else r=mid-1;
    }
    for(int i=1;i<=m;++i)printf("%d ",ans[i]);
}
