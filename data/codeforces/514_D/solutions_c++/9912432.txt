#include<bits/stdc++.h>

using namespace std;

int n,m,k,i,j,l,q,bek,bas,son,mid,t;	

int rmq[6][19][400003];

int A[100003][6];

int query(int j,int x,int y){
	int l=log2(y-x+1);
	return max(rmq[j][l][x],rmq[j][l][min(n,y-(1<<l)+1)]);
}

bool ctr(int x,int y){
	int sum = 0,i;
	for(i=1 ; i<=m ; i++)
		sum+=query(i,x,y);
	return sum<=k;
}

int main(){

	scanf("%d %d %d",&n,&m,&k);

	for(i=1 ; i<=n ; i++)
		for(j=1 ; j<=m ; j++){
			scanf("%d",&A[i][j]);
			rmq[j][0][i]=A[i][j];
		}

	for(j=1 ; j<=m ; j++)
		for(l=1 ; l<=18 ; l++)
			for(i=1 ; i<=n ; i++)
				rmq[j][l][i]=max(rmq[j][l-1][i],rmq[j][l-1][min(n,i+(1<<(l-1)))]);

	int ans = 0, t = 0;

	for(i=1 ; i<=n ; i++){
	
		int bas = 0;
		int son = n - i + 1;

		while(bas/*+1*/<son){

			mid=(bas+son)/2;

			if(mid==bas)
				mid++;

			if(ctr(i,i+mid-1))
				bas=mid;
			else son=mid-1;
		}
/*
		if(ctr(i,i+son-1))
			bas=son;*/

		if(bas > ans){
			ans = bas;
			t = i;
		}

	}

	for(i = 1; i <= m; i++)
		printf("%d ",query(i,t,t+ans-1));

	return 0;
}
