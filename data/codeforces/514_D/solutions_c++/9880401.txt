#include<bits/stdc++.h>
using namespace std;

int a,s,d[524288][5][20],f[300008][5],g,h,j,k,l,i,n,m,i1;
main(){
cin>>n>>m>>h;
for(i=1;i<=n;i++){
    for(a=1;a<=m;a++){
    cin>>d[i][a][0];
    }
}


//cout<<"--"<<endl;
for(j=1;j<=log2(n);j++){
    for(i=1;i<=n;i++){
        //cout<<i+(1<<j)-1<<"* ";
        if(i+(1<<j)-1>n) break;
       for(a=1;a<=m;a++){


        d[i][a][j]=max(d[i][a][j-1],d[i+(1<<(j-1))][a][j-1]);
   // cout<<"{"<<i<<" "<<a<<" "<<j<<"} "<<d[i][a][j]<<" ("<<d[i][a][j-1]<<" "<<d[i+(1<<(j-1))][a][j-1]<<"["<<i+(1<<(j-1))<<", "<<a<<", "<<j-1<<"]"<<") ";
      }
      //cout<<endl;
    }
    //cout<<endl;
}


a=1;
g=-1;
i1=-1;
for(i=1;i<=n;i++){
    k=i-a+1;
    l=log2(k);//cout<<i<<" "<<a<<" "<<l<<endl<<":: ";
    s=0;
    for(j=1;j<=m;j++){
    f[i][j]=max(d[a][j][l],d[i-(1<<l)+1][j][l]);
    s+=f[i][j];//cout<<f[i][a]<<" ";
    }
//cout<<endl<<"&"<<i<<"&"<<endl;
    if(s<=h){if(g<k){g=k;i1=i;}} else {a++;i=max(i-1,a-1);}
}
//cout<<i1<<endl;
for(i=1;i<=m;i++){
    cout<<f[i1][i]<<" ";
}
}
