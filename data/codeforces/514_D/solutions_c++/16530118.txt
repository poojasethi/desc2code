#include<bits/stdc++.h>

using namespace std;

#define sgmn int v, int tl, int tr
#define tm  (tl + tr >> 1)
#define sol v*2, tl, tm
#define sag v*2+1, tm+1, tr
#define N 100005

int R[N][6][18],n,m,k,i,j,l,mid,r,asd,sdf,lg,t;

void f(int l, int r){

	lg = log2(r - l + 1);
	t = 0;

	for(j=1;j<=m;j++)
		t += max( R[r][j][lg] , R[l + (1 << lg) - 1][j][lg] );
}

int main(){

	scanf("%d%d%d",&n,&m,&k);

	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			scanf("%d",&R[i][j][0]);


	for(l=1;l<18;l++)
		for(i=1;i<=n;i++)
			for(j=1;j<=m;j++)
				R[i][j][l] = max( R[i][j][l-1] , R[ max( 0 , i - (1 << (l-1)) ) ][j][l-1] );

	for(i=1;i<=n;i++){

		l = i;
		r = n+1;

		while(l + 1 < r){

			mid = l + r >> 1;

			f(i , mid);

			if(t > k)  r = mid;
			else       l = mid;

		}

		f(i , l);

		if(t <= k && asd - sdf <= l - i){
			asd = l;
			sdf = i;
		}
	}

	lg = log2(asd - sdf + 1);

	for(j=1;j<=m;j++)
		printf("%d ",max( R[asd][j][lg] , R[sdf + (1 << lg) - 1][j][lg] ));
	
	return 0;
}
