#include <bits/stdc++.h>

using namespace std;

int n,m,k,l = 1,r = 1,a[100005][6],ans,out[6];
multiset <int> ms[6];
int sum()
{
    int res = 0;
    for (int i = 1; i <= m; ++i) res += *ms[i].rbegin();
    return res;
}
void del()
{
    for (int i = 1; i <= m; ++i){
        ms[i].erase(a[l][i]);
    }
    l++;
}
void add()
{
    r++;
    for (int i = 1; i <= m; ++i) ms[i].insert(a[r][i]);
}
int main()
{
    scanf("%d %d %d",&n,&m,&k);
    for (int i = 1; i <= n; ++i)
     for (int j = 1; j <= m; ++j){
         scanf("%d",&a[i][j]);
         }
    if (k == 0) return cout << 0,0;
    for (int i = 1; i <= m; ++i){ms[i].insert(0);ms[i].insert(a[1][i]);}
    while (r != n + 1)
    {

        while (sum() > k && l != r + 1)
        {

            del();


        }


        if (r - l + 1 > ans)
        {
           ans = r - l + 1;
           for (int i = 1; i <= m; ++i) out[i] = *ms[i].rbegin();
        }

     if (r <= n)  add();
    }
    for (int i = 1; i <= m; ++i) cout << out[i] << ' ' ; cout << endl;
}
