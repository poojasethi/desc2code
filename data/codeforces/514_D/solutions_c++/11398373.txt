#include <bits/stdc++.h>

#define sz(z) (ll)z.size()
#define fo(i,a,b) for (auto (i) = (a); (i) != (b); (i)++)
#define mp make_pair
#define pb push_back

using namespace std;

//#define DEBUG

#ifdef DEBUG
#define D(x...) printf(x)
#else
#define D(x...) 
#endif

typedef long long ll;
typedef pair<ll,ll> ii;

ll things[100005][5];
multiset <ll> ms[5];

ll ar[5];

int main() {
	ll n, m, k;
	cin >> n >> m >> k;
	fo(i,0,n) {
		fo(j,0,m)
			cin >> things[i][j];
	}
	ll p1 = 0, p2 = 0;
	ll ans = 0;
	while (1) {
		ll sum = 0;
		fo(i,0,m)
			sum+=things[p2][i];
		if (sum > k) {
			fo(i,0,m)
				ms[i].clear();
			p1 = p2 + 1;
			p2 = p1;
			continue;
		}
		sum = 0;
		fo(i,0,m) {
			ms[i].insert(things[p2][i]);
			sum += *(ms[i].rbegin());
		}
		p2++;	
		while (p2 > p1 && sum > k) {
			sum = 0;
			fo(i,0,m) {
				ms[i].erase(ms[i].find(things[p1][i]));
				if (sz(ms[i])) sum += *(ms[i].rbegin());
			}
			p1++;
		}
		if (p2 - p1 >= ans && p2 > p1) {
			D("%d %d\n",p1,p2);
			ans = p2 - p1;
			fo(i,0,m)
				ar[i] = *ms[i].rbegin();
		}
		if (p2 >= n)
			break;
	}
	fo(i,0,m)
		cout << ar[i] << ' ';
	puts("");
	return 0;
}
