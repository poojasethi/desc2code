#include <iostream>
#include <set>
using namespace std;
int n,m,k,data[100005][5],ml,ans[5];
multiset<int> shot[5];
int main() {
    cin>>n>>m>>k;
    for(int i=0;i<n;i++)for(int j=0;j<m;j++)cin>>data[i][j];
    for(int i=0;i<m;i++)shot[i].insert(0);
    int l=0,r=0,tol;
    while(l<n){
        tol=0;
        for(int j=0;j<m;j++){
            shot[j].insert(data[l][j]);
            tol+=*shot[j].rbegin();
        }
        while(tol>k&&r<=l){
            tol=0;
            for(int j=0;j<m;j++){
                shot[j].erase(shot[j].find(data[r][j]));
                tol+=*shot[j].rbegin();
            }
            r++;
        }
        if(l-r+1>ml){
            ml=l-r+1;
            for(int j=0;j<m;j++){
                ans[j]=*shot[j].rbegin();
            }
        }
        l++;
    }
    for(int i=0;i<m;i++){
        cout<<ans[i]<<" ";
    }
}
