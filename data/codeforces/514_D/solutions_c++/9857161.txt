#include<bits\stdc++.h>
using namespace std;
struct OrderByPair{
	bool operator()(pair<int, int> const &a, pair<int, int> const &b){
		return a.first < b.first || (a.first == b.first && a.second > b.second);
	}
};
int w[100005][5];
priority_queue < pair<int, int>, vector< pair<int, int> >, OrderByPair> PQ[5];
int main(){
	int n, m, k;
	scanf("%d%d%d", &n, &m, &k);
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			scanf("%d", &w[i][j]);
	int L = 0, R = -1, preR = -2;
	int res[5] = { 0 }, len = 0;
	while (L < n && R < n){
		for (int i = 0; i < m; i++)
			while (!PQ[i].empty() && PQ[i].top().second < L)
				PQ[i].pop();
		if (L > R)	R = L;
		for (int i = 0; i < m && preR != R; i++)
			PQ[i].push(make_pair(w[R][i], R));
		preR = R;
		int sum = 0;
		for (int i = 0; i < m; i++)
			sum += PQ[i].top().first;
		if (sum <= k){
			if (R - L + 1 > len){
				for (int i = 0; i < m; i++)
					res[i] = PQ[i].top().first;
				len = R - L + 1;
			}
			R++;
		}
		else
			L++;
	}
	printf("%d", res[0]);
	for (int i = 1; i < m; i++)
		printf(" %d", res[i]);
	putchar('\n');
	return 0;
}
