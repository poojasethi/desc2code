#include <iostream>
#include <vector>
using namespace std;
typedef long long ll;
const ll nmax = 1e5+5;
ll st[nmax][5][20];
int main(){
	ll n,m,k;
	cin >> n >> m >> k;
	for(ll i = 0; i < n; ++i)
		for(ll j = 0; j < m; ++j)
			cin >> st[i][j][0];
	for(ll s = 1; s < 20; ++s)
		for(ll i = 0; i < n; ++i)
			for(ll j = 0; j < m; ++j)
				if((1 << s) + i - 1 < n){
					st[i][j][s]=max(st[i][j][s-1],st[i+(1<<(s-1))][j][s-1]);
				}
	ll l = 0;
	ll best=0;
	vector<ll> used(m);
	for(ll r = 0; r < n; ++r){
		vector<ll> need(m);
		ll sum = 0;
		ll p = 20;
		while(l+(1<<p)-1>r)
			p--;
		for(ll i = 0; i < m; ++i){
			need[i]=max(st[l][i][p],st[r-(1<<p)+1][i][p]);
			sum+=need[i];
		}
		if(l<r&&sum>k){
			l++;
			r--;
			continue;
		}
		if(l==r&&sum>k)
			continue;
		if(r-l+1>best){
			//cout << l << ' ' << r << ' ' << sum << ' ' << k << '\n';
			best=r-l+1;
			used=need;
		}
	}
	for(ll i = 0; i < m; ++i)
		cout << used[i] << " \n"[i==m-1];
}
