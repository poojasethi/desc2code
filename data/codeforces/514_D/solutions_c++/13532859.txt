
#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<ll, ll> pii;

const ll MAXN = 1e5 + 100;

ll a[MAXN][10], n, m;

multiset<ll> cnt[10];
ll l, r, k;
ll ans[10];

inline ll getused(){
	ll outp = 0;
	for (ll i = 0; i < m; i++)
		if (cnt[i].size()) outp += *cnt[i].rbegin();
	return outp;
}

int main(){
	ios::sync_with_stdio( false );
	cin.tie(0);

	cin >> n >> m >> k;

	for (ll i = 0; i < n; i++)
		for (ll j = 0; j < m; j++)
			cin >> a[i][j];
	
	ll s = 0, e = 0;
	while(e < n){
		for (ll i = 0; i < m; i++)	
			cnt[i].insert(a[e][i]);
		e++;

		while(getused() > k){
			for (ll i = 0; i < m; i++)
				cnt[i].erase(cnt[i].lower_bound(a[s][i]));

			s++;
		}

		if (e - s > r - l) r = e, l = s;
	}


	for (ll i = 0; i < m; i++)
		for (ll j = l; j < r; j++)
			ans[i] = max(ans[i], a[j][i]);
	

	for (ll i = 0; i < m; i++)
		cout << ans[i] << " ";
}
