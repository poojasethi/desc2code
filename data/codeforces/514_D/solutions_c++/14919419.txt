#include <bits/stdc++.h>
using namespace std;
typedef pair<int,int> PII;
#define ft first
#define sc second
#define MP make_pair

const int MAX_N=100010,MAX_M=6;
int L,R,N,M,Mx,K;
int Ans[MAX_M],A[MAX_M];
struct Tqueue
{
	int l,r;
	PII q[MAX_N];
	Tqueue():l(1),r(0){}
	int max(){
		return q[l].ft;
	}
	void pop_front(){
		for(;l<=r&&q[l].sc<L;l++);
	}
	void push_back(int x,int idx){
		for(;l<=r&&q[r].ft<=x;r--);
		q[++r]=MP(x,idx);
	}
}Q[MAX_M];
int Sum()
{
	int ret=0;
	for(int j=1;j<=M;j++)
		ret+=Q[j].max();
	return ret;
}
inline void Main()
{
	scanf("%d%d%d",&N,&M,&K);
	for(L=1,R=1;R<=N;R++){
		for(int j=1;j<=M;j++){
			scanf("%d",A+j);
		}
		for(int j=1;j<=M;j++){
			Q[j].push_back(A[j],R);
		}
		for(;L<=R&&!(Sum()<=K);){
			L++;
			for(int j=1;j<=M;j++)
				Q[j].pop_front();
		}
		if(R-L+1>Mx){
			Mx=R-L+1;
			for(int j=1;j<=M;j++)
				Ans[j]=Q[j].max();
		}
	}
	for(int j=1;j<=M;j++){
		printf("%d%c",Ans[j]," \n"[j==M]);
	}
}
int main(){Main();return 0;}
