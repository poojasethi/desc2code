#include <cstdio>
#include <queue>
#define PII pair<int,int>
#define MP(x,y) make_pair(x,y)
#define N 100005
#define M 5
using namespace std;
int n, m, k;
int ar[N][M];
deque<PII> q[M];
int sum () {
    int res=0;
    for ( int i=0; i<m; i++ ) {
        res+=q[i].back().second;
    }
    return res;
}
int main ()
{
    scanf("%d%d%d",&n,&m,&k);
    for ( int i=0; i<n; i++ ) for ( int j=0; j<m; j++ ) scanf("%d",&ar[i][j]);
    int ans=0, ansk[M]={};
    for ( int l=0,r=0; r<n; r++ ) {
        for ( int j=0; j<m; j++ ) {
            while ( !q[j].empty() && q[j].front().second<ar[r][j] ) q[j].pop_front();
            q[j].push_front(MP(r,ar[r][j]));
        } 
        while ( l<=r && k-sum()<0 ) {
            for ( int j=0; j<m; j++ ) if ( q[j].back().first<=l ) q[j].pop_back();
            l++;
        }
        if ( r-l+1>ans ) {
            ans=r-l+1;
            for ( int j=0; j<m; j++ ) ansk[j]=q[j].back().second;
        }
    }
    for ( int i=0; i<m; i++ ) printf("%d%c",ansk[i],i==m-1?'\n':' ');
    return 0;
}
