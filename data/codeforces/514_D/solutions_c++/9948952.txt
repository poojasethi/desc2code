#include <iostream>
#include <vector>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <set>
using namespace std;

int MAT[100010][5];

int main() {
	
	int N , A, K;
	cin >> N >> A >> K;
	for (int i = 0; i < N; ++i) {
		for(int j = 0; j < A; ++j){
			cin >> MAT[i][j];
		}
	}
	multiset<int> MS[5];
	int t = 0, ma = 0, st = 0;
	for(int i = 0; i < N; ++i){
		for(int j = 0; j < A; ++j){
			MS[j].insert(MAT[i][j]);
		}
		while(t <= i){
			int e = 0;
			for(int j = 0; j < A; ++j){
				e += *MS[j].rbegin();
			}
			if(e <= K)break;
			for(int j = 0; j < A; ++j){
				MS[j].erase(MS[j].lower_bound(MAT[t][j]));
			}
			t++;
		}
		ma = max(ma , i-t+1);
		if(ma == i-t+1)st = t;
	}
	for(int i = 0; i < A; ++i){
		int mma = 0;
		for(int j = st; j < st + ma; ++j){
			mma = max(mma , MAT[j][i]);
		}
		cout << mma << " ";
	}
	cout << endl;
//	cout << ma << " " << st << endl;
	
	
	return 0;
}
