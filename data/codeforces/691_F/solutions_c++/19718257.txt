#include <set>
#include <map>
#include <cmath>
#include <queue>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
typedef long long  LL;
using namespace std;
const int maxn=1000010;
int num[maxn];
LL  ans[3*maxn];
LL cnt[3*maxn];
int p[maxn];
int n,m;
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&num[i]);
        cnt[num[i]]++;
    }
    int m;
    scanf("%d",&m);
    for(int i=1;i<=m;i++)
    {
        scanf("%d",&p[i]);
    }
    for(int i=1;i<3*maxn;i++)
    {
        if(!cnt[i])
            continue;
        for(int j=i;j<3*maxn;j+=i)
        {
            if(j/i!=i)
                ans[j]+=(LL)(cnt[i])*(cnt[j/i]);
            else
                ans[j]+=(LL)(cnt[i])*(cnt[i]-1);
        }
    }
    for(int i=1;i<3*maxn;i++)
    {
        ans[i]+=ans[i-1];
    }
    for(int i=1;i<=m;i++)
    {
        printf("%I64d\n",(LL)(n)*(n-1)-ans[p[i]-1]);
    }
    return 0;
}
