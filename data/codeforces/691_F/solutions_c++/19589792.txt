#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <iostream>
using namespace std;

int a[3000005], b[3000005];
long long sum[3000005], num[3000005];

int main() {
	int n;
	scanf("%d", &n);
	for(int i = 1; i <= n; i++) scanf("%d", &a[i]);
	memset(num, 0, sizeof(num));
	int m;
	scanf("%d", &m);
	sort(a+1, a+n+1);
	num[1] = 1;
	b[1] = a[1];
	int cnt = 1;
	for(int i = 2; i <= n; i++) {
		if(a[i]==a[i-1]) num[cnt]++;
		else {
			cnt++;
			b[cnt] = a[i];
			num[cnt] = 1;
		}
	}
	memset(sum, 0, sizeof(sum));
	for(int i = 1; i <= cnt; i++) {
		for(int j = 1; j <= i; j++) {
			if((long long)b[i]*b[j] > 3e6) break;
			if(j == i) sum[b[i]*b[j]] += num[i]*(num[i]-1);
			else sum[b[i]*b[j]] += 2*num[i]*num[j];
		}
	}
	for(int i = 1; i <= 3e6; i++) sum[i] += sum[i-1];
	for(int i = 1; i <= m; i++) {
		int p;
		scanf("%d", &p);
		printf("%I64d\n", (long long)n*(n-1)-sum[p-1]);
	}
	return 0;
}
