#include <bits/stdc++.h>
using namespace std;
#define ll long long
const int N = 3000010;

int n, m, a[N];
ll cnt[N], sum[N];
ll res;

int main() {
	ios_base::sync_with_stdio(false); cin.tie(0);
	cin >> n; for (int i = 1; i <= n; i++) { int x; cin >> x; a[x]++; }
	res = 1ll * n * (n-1);
	for (int i = 1; i * i < N; i++) cnt[i * i] = 1ll * a[i] * (a[i]-1);
	for (int i = 1; i < N; i++) for (int j = i; j < N; j += i) if (j / i != i) cnt[j] += 1ll * a[i] * a[j/i];
	for (int i = 1; i < N; i++) sum[i] = sum[i-1] + cnt[i];

	cin >> m;
	while(m--) {
		int p; cin >> p;
		printf("%lld\n", res - sum[p-1]);
	}
}
