using namespace std;

typedef long long ll;

const int inf = 0x3f3f3f3f;

const int MAXN = 3e6;

int d[MAXN + 10];
long long f[MAXN + 10];

int n;
int a[MAXN];
long long sum[MAXN + 10];

int main( )
{
	 scanf("%d", &n);
	for (int i = 1; i <= n; i ++)
		scanf("%d", &a[i]), d[a[i]] ++;

	for (int i = 1; i <= MAXN; i ++)
		for (int j = i; j <= MAXN; j += i)
			if (i != j / i) f[j] += 1LL * d[i] * d[j / i];
			else f[j] += 1LL * d[i] * (d[i] - 1);

	for (int i = 1; i <= MAXN; i ++)
		sum[i] = sum[i - 1] + f[i];
	int m, x;
	scanf("%d", &m);
	for (int i = 1; i <= m; i ++)
	{
		scanf("%d", &x);
		printf("%I64d\n", 1LL * n * (n - 1) - sum[x - 1]);
	}
	return 0;
}
