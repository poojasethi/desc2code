#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const int maxn = 1e6+10;
const int maxm = maxn*3; 
LL ret[maxm],preS[maxm],cnt[maxm];
int a,p,n,m;
int main() {
	
	//ios_base::sync_with_stdio(0),cin.tie(0),cout.tie(0);
	#ifndef ONLINE_JUDGE
    freopen("in", "r", stdin);
	#endif

    scanf("%d",&n);
    memset(ret,0,sizeof ret);
    memset(preS,0,sizeof preS);
    for(int i = 1; i <= n; i++)  scanf("%d",&a),cnt[a]++;
    
    for(int i = 1; i < maxm; i++) {
        if(cnt[i]!=0) {
            for(int j = i; j < maxm; j+=i) {
                if(j/i != i) ret[j] += cnt[i]*cnt[j/i];
                else ret[j] += cnt[i]*(cnt[i]-1);
            }
        }
    }
    preS[0] = 0;
    for(int i = 1; i < maxm; i++) preS[i] = preS[i-1]+ret[i]; 
    scanf("%d\n",&m);
    LL all = LL(n)*(n-1);
    for(int i = 0; i < m; i++)  {
        scanf("%d",&p);
        printf("%I64d\n",all-preS[p-1]);
    }


    
}
