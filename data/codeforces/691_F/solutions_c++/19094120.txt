#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

typedef long long int ll;

const int limite=3000005;

ll n;
ll cuantos[limite];
int raiz[limite];
int m;
int ndiv=1;
ll solucion[limite];
ll igual[limite];
ll suma[limite];

int main()
{
  scanf("%d",&n);
  for (int i=0;i<n;i++) {
    int x;
    scanf("%d",&x);
    cuantos[x]++;
  }
  for (int i=1;i*i<limite;i++)
    raiz[i*i]=i;
  for (int i=1;i<limite;i++)
    if (raiz[i]==0)
      raiz[i]=raiz[i-1];
  for (int x=1;x<=raiz[limite-1];x++) {
    if (cuantos[x]) {
      for (int p=x*x,opuesto=x;p<limite;p+=x,opuesto++) {
	if (x==opuesto) igual[p]+=cuantos[x]*(cuantos[x]-1)/2;
	else igual[p]+=cuantos[x]*cuantos[opuesto];
      }
    }
  }
  for (int p=1;p<limite;p++) {
    suma[p]=suma[p-1]+igual[p];
    solucion[p]=n*(n-1)-2*(suma[p]-igual[p]);
  }
  scanf("%d",&m);
  for (int cas=0;cas<m;cas++) {
    int p;
    scanf("%d",&p);
    //#include "alerta.h"
    //cout<<solucion[p]<<endl;
    printf("%I64d\n",solucion[p]);
  }
}
