#include <stdio.h>
#include <stdlib.h>
#include <algorithm>

using namespace std;

typedef long long LL; 
LL c[5000000],sum[5000000],ans,n,m,a;


int main(){
	scanf("%d",&n);
	
	for(int i=0;i<n;i++){
		scanf("%d",&a);
		c[a]++;
	}
	for(int i=1;i<2001;++i)
		for(int j=i;c[i]&&i*j<5000000;j++)
			if(i-j)
				sum[i*j]+=c[i]*c[j];
			else 
				sum[i*j]+=(c[i]-1)*c[i]/2;
		for(int i=1;i<5000000;i++)
			sum[i]+=sum[i-1];
		for(scanf("%d",&m);m--;){
			scanf("%d",&a);
			printf("%I64d\n",(long long)n*(n-1)-2*sum[a-1]);
		}
		
	return 0;
}
