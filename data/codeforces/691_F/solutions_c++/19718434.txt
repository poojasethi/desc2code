#include <set>
#include <map>
#include <cmath>
#include <queue>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
typedef long long  LL;
using namespace std;
const int maxn=1000010;
int a[maxn];
LL  ans[3*maxn];
LL cnt[3*maxn];
LL p[maxn];
LL n,m;
int main ()
{
    scanf("%lld",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        cnt[a[i]]++;
    }
    scanf("%lld",&m);
    for(int i=1;i<=m;i++)
    {
        scanf("%lld",&p[i]);
    }
    for(int i=1;i<3*maxn;i++)
    {
        if(!cnt[i])
            continue;
        for(int j=i;j<3*maxn;j+=i)
        {
            if(j/i!=i)
                ans[j]+=cnt[i]*cnt[j/i];
            else
                ans[j]+=cnt[i]*(cnt[i]-1);
        }
    }
    for(int i=1;i<3*maxn;i++)
        ans[i]+=ans[i-1];
    for(int i=1;i<=m;i++)
    {
        printf("%lld\n",n*(n-1)-ans[p[i]-1]);
    }
    return 0;
}
