#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define fo(i,a,b) for(int i=a;i<=b;i++)
#define fd(i,a,b) for(int i=a;i>=b;i--)
#define maxn 1000005
#define maxm 3000005
#define mx 3000000
#define ll long long
using namespace std;

int cnt[maxm];

ll f[maxm];

int n,m;

inline int read(){
	int ret=0,ff=1;
	char c=getchar();
	while (c<'0' || c>'9') {
		if (c=='-') ff=-1;
		c=getchar();
	}
	while (c>='0' && c<='9') {
		ret=ret*10+c-'0';
		c=getchar();
	}
	return ret;
}

int main(){
	n=read();
	fo(i,1,n) {
		int x=read();
		cnt[x]++;
	}
	fo(i,1,mx) {
		if (cnt[i]==0) continue;
		fo(j,i,mx / i) {
			if (cnt[j]==0) continue;
			if (i!=j) f[i*j]+=1ll*cnt[i]*cnt[j] * 2;
			else f[i*j]+=1ll*cnt[i]*(cnt[i]-1);
		}
	}
	ll tot=1ll * n*(n-1);
	fo(i,1,mx) f[i]+=f[i-1];
	m=read();
	while (m--) {
		int x=read();
		printf("%I64d\n",tot-f[x-1]);
	}
	return 0;
}
