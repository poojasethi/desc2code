#include<bits/stdc++.h>
using namespace std;
const int MAXN=3e6+5;
const int M=1e6+5;
typedef long long ll;
ll sum[MAXN]={0};
ll a[MAXN]={0};
int main()
{
	int t,Max=-1;
	ll i,j,n,m;
	scanf("%I64d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&t);
		a[t]++;
		Max=max(Max,t);
	}
	for(i=0;i<=Max;i++)
	{
		for(j=0;j<=Max;j++)
		{
			if(i*j>3e6)
			{
				break;
			}
			if(i==j)
			{
				sum[i*j]+=a[i]*(a[i]-1);
			}
			else
			{
				sum[i*j]+=a[i]*a[j];
			}
		}
	}
	for(i=1;i<MAXN;i++)
	{
		sum[i]+=sum[i-1];
	}
	scanf("%I64d",&m);
	for(i=0;i<m;i++)
	{
		ll temp=n*(n-1);
		scanf("%d",&t);
		printf("%I64d\n",temp-sum[t-1]);
	}
}
