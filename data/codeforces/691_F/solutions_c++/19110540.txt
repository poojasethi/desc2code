#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

LL cnt[3000010], num[3000010], ans[3000010];

int main () {
    LL n;
    scanf ("%I64d", &n);
    memset (cnt, 0, sizeof cnt);
    memset (num, 0, sizeof num);
    for (int i = 1; i <= n; i++) {
        int A;
        scanf ("%d", &A);
        cnt[A]++;
    }

    for (int i = 1; i <= 3000000; i++) {
        if (cnt[i] == 0)    continue;
        for (int j = 1; j * i <= 3000000; j++) {
            num[i * j] += cnt[i] * cnt[j];
            if (i == j)
                num[i * j] -= cnt[i];
        }
    }

    ans[0] = 0;
    for (int i = 1; i <= 3000000; i++) {
        ans[i] = ans[i - 1] + num[i - 1];
    }

    int m;
    scanf ("%d", &m);
    for (int i = 1; i <= m; i++) {
        int p;
        scanf ("%d", &p);
        printf ("%I64d\n", n * (n - 1) -  ans[p]);
    }
    return 0;
}
