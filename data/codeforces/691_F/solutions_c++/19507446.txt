#include<cstdio>
#include<cstring>
using namespace std;
#define N 3000005
typedef __int64 ll;
ll dp[N];
int val,n,m,a[N];
int main(){
	while(~scanf("%d",&n)){
		memset(a,0,sizeof(a));
		for(int i=0;i<n;++i){
			scanf("%d",&val);++a[val];
		}
		memset(dp,0,sizeof(dp));
		for(int i=1;i<N;++i){
			for(int j=1;j*i<N;++j){
				dp[i*j]+=(ll)a[i]*a[j];
				if(i==j) dp[i*j]-=a[i];
			}
		}
		for(int i=1;i<N;++i) dp[i]+=dp[i-1];
		scanf("%d",&m);
		for(int i=0;i<m;++i){
			scanf("%d",&val);
			printf("%I64d\n",(ll)n*(n-1)-dp[val-1]);
		}
	}
	return 0;
}
