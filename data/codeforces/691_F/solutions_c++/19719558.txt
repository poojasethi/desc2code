#include<iostream>
#include<cstdio>
using namespace std;
typedef long long LL;
#define N 3000010
int cnt[N];
LL a[N],ans[N];
int main()
{
	LL n;
	int m,x;
	scanf("%I64d", &n);
	for (int i = 1; i <= n; i++)
	{
		scanf("%d", &x);
		cnt[x]++;
	}
	for (int i = 1; i <= N - 5; i++)
	{
		if (cnt[i] == 0) continue;
		for (int j = i; j <= N-5 ; j++)
		{
			if (cnt[j] == 0)  continue;
			if (i*j > (N - 5))  break;
			if (i == j) a[i*j] += cnt[i] *(cnt[j] - 1);
			else        a[i*j] += cnt[i] * cnt[j] * 2;
		}
	}
	for (int i = 1; i <= N - 5; i++)
		ans[i] = ans[i - 1] + a[i];
	scanf("%d", &m);
	for (int i = 1; i <= m; i++)
	{  
		scanf("%d", &x);
		printf("%I64d\n", n*(n - 1) - ans[x-1]);
	}
	return 0;
}
