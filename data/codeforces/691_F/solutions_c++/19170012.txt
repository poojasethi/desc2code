#include<iostream>
#include<cstdio>
#include<cstring>
#include<queue>
#include<vector>
#include<set>
#include<map>
#include<algorithm>
using namespace std;
typedef pair<int,int>p;
typedef long long ll;
const int maxm = 3e6 + 1;
ll sum[maxm];
int cnt[maxm];
int main (void)
{
	int n;scanf("%d", &n);
	int a;
	for(int i = 0; i < n; i++){
		scanf("%d", &a);
		cnt[a]++;
	}
	int q;scanf("%d",&q);
	for(int i = 1; i <= maxm; i++){
		for(int j = 1; j * i < maxm; j++){
			sum[i * j] += cnt[i] * 1ll * (cnt[j] - (i == j));
		}
	}
	for(int i = 1; i <= maxm; i++) sum[i] += sum[i - 1];
	int b;
	for(int i = 0; i < q; i++){
		scanf("%d", &b);
		printf("%I64d\n", n * 1ll * (n - 1) - sum[b - 1]);
	}
	return 0;
}
