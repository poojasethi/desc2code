#include <iostream>
#include <fstream>
#include <algorithm>
#include <string>
#include <iomanip>
#include <vector>
#include <set>
#include <map>
#define inf 10000000000000

using namespace std;
typedef long long ll;

ll n, m;
ll a[10000000];
ll b[10000000];
ll ans[10000000];

int main()
{
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		ll x;
		scanf("%d", &x);
		a[x]++;
	}
	for (ll i = 1; i <= 3000000; i++)
	{
		for (ll j = 1; i*j <= 3000000; j++)
		{
			ll q = a[i];
			ll w = a[j];
			if (i == j)
				w--;
			b[i*j] += q * w;
		}
	}
	for (int i = 1; i <= 3000000; i++)
		ans[i] = ans[i - 1] + b[i];
	cin >> m;
	for (ll i = 0; i < m; i++)
	{
		ll x;
		scanf("%d", &x);
		ll res = (n*(n - 1)) - ans[x - 1];
		if (res<=1000000000)
		printf("%d", res);
		else
		{
			cout << res;
		}
		printf("%c", ' ');
	}
}
