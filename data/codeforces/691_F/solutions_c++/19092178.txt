#include <bits/stdc++.h>
using namespace std;

int cnt[3000005];

long long ans[3000005];

int main()
{
	memset(cnt, 0, sizeof(cnt));
	int n, m;
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
	{
		int x;
		scanf("%d", &x);
		cnt[x]++;
	}
	memset(ans, 0 , sizeof(ans));
	long long all = n*1ll*(n-1);
	for (int i = 1; i <= 3000000; i++)
	{
		if (i*1ll*i > 3000000)
			break;
		for (int j = i*i; j <= 3000000; j += i)
		{
			int p = i;
			int q = j / p;
			if (p < q)
			{
				ans[j] += 2*cnt[p] * 1ll * cnt[q];
			}
			else if (p == q)
				ans[j] += cnt[p] * 1ll * (cnt[p] - 1);
		}
	}
	for (int i = 1; i <= 3000000; i++)
		ans[i] += ans[i-1];
	scanf("%d", &m);
	for (int i = 0; i < m; i++)
	{
		int p;
		scanf("%d", &p);
		printf("%lld\n", (all - ans[p-1]));
	}
}
