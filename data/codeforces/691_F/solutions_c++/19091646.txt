#include <bits/stdc++.h>
using namespace std;

long long A[3000006], B[3000006];

int main() {
    
	int N;
	scanf("%d", &N);
	
	for(int i=1; i<=N; i++)
	{
	    int x;
	    scanf("%d", &x);
	    
	    B[x]++;
	}
	
	int M;
	scanf("%d", &M);
	
	for(int i=1; i<=3000000; i++)
	{
	    for(int j=i; j<=3000000; j+= i)
	    {
	        if(i!=j/i)
    	        A[j] += B[i]*B[j/i];
    	    else
    	        A[j] += B[i]*(B[i]-1);
	    }
	}
	
	for(int i=1; i<=3000000; i++)
	    A[i] += A[i-1];
	
	long long sum = (long long)N*(N-1);
	
	while(M--)
	{
	    int x;
	    scanf("%d", &x);
	    
	    printf("%lld\n", sum-A[x-1]);
	}
	
	return 0;
}
