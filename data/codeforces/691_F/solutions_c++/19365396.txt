#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int N,n,M;
int A[1010000];
int size[1010000],pre[1010000];
long long ans[3010000];
int main(){
	//caonima
	scanf("%d",&N);
	for (int i=1;i<=N;i++) scanf("%d",&A[i]);
	sort(A+1,A+N+1);
	for (int i=1;i<=N;i++)
		if (A[i]!=A[i-1]){
			A[++n]=A[i];
			size[n]=1;
		}else size[n]++;
	for (int i=1;i<=n;i++) pre[i]=pre[i-1]+size[i];
	for (int i=1;i<=n;i++){
		for (int j=1;j<=n;j++)
			if (i!=j){
				if (1ll*A[i]*A[j]>=3000000){
					if (i>j){
						ans[3000000]+=1ll*size[i]*(N-pre[j-1]-size[i]);
					}else{
						ans[3000000]+=1ll*size[i]*(N-pre[j-1]);
					}
					break;
				}else ans[A[i]*A[j]]+=1ll*size[i]*size[j];
			}
		if (1ll*A[i]*A[i]>=3000000) ans[3000000]+=1ll*size[i]*(size[i]-1);
		else ans[A[i]*A[i]]+=1ll*size[i]*(size[i]-1);
	}
	for (int i=3000000;i>=1;i--) ans[i]+=ans[i+1];
	scanf("%d",&M);
	for (int i=1;i<=M;i++){
		int P;
		scanf("%d",&P);
		printf("%lld\n",ans[P]);
	}
	return 0;
}
