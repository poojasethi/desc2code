#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <vector>
#include <iostream>
using namespace std;
#define LL  long long
const LL mod=1e9+7;
const int N=3e6;
int cnt[N+6];
LL num[N+6];
LL sum[N+6];
int main()
{
    //freopen("in.txt","r",stdin);
    int n,m;
    while(cin>>n)
    {
        memset(cnt,0,sizeof(cnt));
        memset(num,0,sizeof(num));
        for(int i=1;i<=n;i++)
        {
            int x;scanf("%d",&x);
            cnt[x]++;
        }

        for(int i=1;i<=6000;i++)
        {
            if(i*i<=N)num[i*i]+=(LL)cnt[i]*(cnt[i]-1);
            for(int j=i+1;i*j<=N;j++)
            {
                num[i*j]+=(LL)2*cnt[i]*cnt[j];
            }
        }
        scanf("%d",&m);
        sum[0]=0;
        for(int i=1;i<=N+1;i++)sum[i]=sum[i-1]+num[i];
        while(m--)
        {
            int p;
            scanf("%d",&p);
            printf("%lld\n",(LL)n*(n-1)-sum[p-1]);
        }
    }
    return 0;
}
