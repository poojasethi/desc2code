#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

typedef long long int ll;

const int limite=3000005;

int n;
ll cuantos[limite];
int m;
int ndiv=1;
ll solucion[limite];
ll igual[limite];

int main()
{
  scanf("%d",&n);
  for (int i=0;i<n;i++) {
    int x;
    scanf("%d",&x);
    cuantos[x]++;
  }
  for (int x=1;x*x<limite;x++)
    if (cuantos[x])
      for (int p=x*x,opuesto=x;p<limite;p+=x,opuesto++)
	if (x==opuesto) igual[p]+=cuantos[x]*(cuantos[x]-1);
	else igual[p]+=2*cuantos[x]*cuantos[opuesto];
  ll total=ll(n)*(n-1);
  ll suma=0;
  for (int p=1;p<limite;p++) {
    solucion[p]=total-suma;
    suma+=igual[p];
  }
  scanf("%d",&m);
  for (int cas=0;cas<m;cas++) {
    int p;
    scanf("%d",&p);
    //#include "alerta.h"
    //cout<<solucion[p]<<endl;
    printf("%I64d\n",solucion[p]);
  }
}
