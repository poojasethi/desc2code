#include <iostream>
#include <cstring>
#include <algorithm>
#include <cstdio>
using namespace std;
#define N 6001000
int s[N],h[N];
long long sum[N],num[N];
int main(){
    int n,m;
    memset(num,0,sizeof(num));
    scanf("%d",&n);
    int Ms=0,Mh=0;
    for(int i=0;i<n;i++){
        scanf("%d",&s[i]);
        Ms=max(Ms,s[i]);
        num[s[i]]++;
    }
    sort(s,s+n);
    scanf("%d",&m);
    for(int i=0;i<m;i++){
        scanf("%d",&h[i]);
        Mh=max(Mh,h[i]);
    }
    memset(sum,0,sizeof(sum));
    for(int i=1;i<=Ms;i++){
        for(int j=1;j*i<=Mh;j++){
            if(i!=j)
            sum[i*j]+=num[i]*num[j];
            else{
                sum[i*j]+=num[i]*(num[j]-1);
            }
        }
    }
    for(int i=1;i<=Mh;i++){
        sum[i]=sum[i]+sum[i-1];
    }
   long long ans=(long long)n*(long long)(n-1);
    for(int i=0;i<m;i++){
        printf("%I64d\n",ans-sum[h[i]-1]);
    }

}
