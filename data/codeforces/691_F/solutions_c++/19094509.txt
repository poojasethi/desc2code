#include <bits/stdc++.h>
#define NMAX 3000001
using namespace std;
int n, m;
int a[NMAX], p[NMAX];
long long cnt[NMAX];
int main() {
    int n;
    scanf("%d", &n);
    for ( int i = 1; i <= n; i++ ) {
        int x;
        scanf("%d", &x);
        a[x]++;
    }
    scanf("%d", &m);
    for (int i = 1; i <= m; i++ )
        scanf("%d", &p[i]);
    for ( int i = 1; i < NMAX; i++ )
        if ( a[i] ) {
            for ( int j = 1; j < NMAX / i; j++ ) {
                int t = i * j;
                cnt[t] = cnt[t] + (long long) a[i] * a[j];
                if ( i == j ) cnt[t] -= a[i];
            }
        }
    for ( int i = 1; i < NMAX; i++ ) cnt[i] += cnt[i-1];
    for ( int i = 1; i <= m; i++ )
        printf("%I64d\n", (long long) n * (n - 1) - cnt[p[i] - 1]);
}
