#define ll long long

using namespace std;
const ll maxn= 4e6,M= 3e6;
ll a[maxn],s[maxn];
ll n,m,i,j,p,x,N;
int main()
{
#ifndef ONLINE_JUDGE

	scanf("%lld",&n);
	N= n*(n-1);
	for (i= 1;i<=n;i++){
		scanf("%lld",&x);
		a[x]++;		
	}
	for (i= 1;i<=M;i++){
		if (!a[i]) continue;
		for (j= 1;i*j<=M;j++){
			if (i!=j)
				s[i*j]+= a[i]*a[j];
			else s[i*j]+= a[i]*(a[i]-1);
		}
	}
	for (i= 2;i<=M;i++)
		s[i]+= s[i-1];
	scanf("%lld",&m);
	for (i= 1;i<=m;i++){
		scanf("%lld",&x);
		printf("%lld\n",N-s[x-1]);
	}return 0;
}
