#include<bits/stdc++.h>//dp,数论, 预处理 
using namespace std;
typedef long long ll;
ll p[3000010];//开long long啊记得。。。 
ll k[3000010];
ll ans = 0;
int main()
{
	ll n, m;
	scanf("%I64d", &n);
	for(ll i = 1; i <= 3000000; i++)
	{
		k[i] = 0;
		p[i] = 0;
	} 
	//memset(k, 0, sizeof(k));
	//memset(p, 0, sizeof(p));
	for(ll i = 1; i <= n; i++)
	{
		ll a;
		scanf("%I64d", &a);
		k[a]++;
	}
	for(ll i = 1; i <= 3000000; i++)//这里属于调和级数，时间复杂度为nlogn 
	{
		//if(k[i])
		//{
			for(ll j = i; j <= 3000000; j += i)
		    {
		    	ll d = j/i;
			    p[j] += k[i]*k[d];
			    if(i == d)
			       p[j] -= k[i]; 
		    }
		//} 
	}
	for(ll i = 2; i <= 3000000; i++)
	   p[i] += p[i-1];
	scanf("%I64d", &m);
	ll sum = n*(n-1);
	for(ll i = 1; i <= m; i++)
	{
		ll t;
		scanf("%I64d", &t);
		ll ans = sum-p[t-1];
		printf("%I64d\n", ans);
	}
	return 0;
}
