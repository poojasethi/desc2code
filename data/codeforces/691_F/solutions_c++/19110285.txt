#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll fre[3000006],pre[3000006];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n;cin>>n;
    for(int i=0;i<n;i++)
    {
        ll val;cin>>val;
        fre[val]++;
    }
    // exploited the property of two numbers product
    for(int i=1;i<=3000000;i++)
    {
        for(int j=i;j<=3000000;j+=i)
        {
            if(i == (j/i))
                pre[j]+=(fre[i]*(fre[i]-1));
            else
                pre[j]+=(fre[i]*fre[j/i]);
        }
    }
    pre[0]=0;
    for(int i=1;i<=3000000;i++)
        pre[i]+=pre[i-1];
    ll ans=n*1ll*(n-1);
    int m;cin>>m;
    while(m--)
    {
        int num;cin>>num;
        cout<<ans-pre[num-1]<<"\n";
    }
    return 0;
}
