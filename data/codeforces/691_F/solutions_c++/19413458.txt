#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#define N 3000010
#define mod (int)(1e9+7)
using namespace std;

int n,m,p;
int a[N];
__int64 cnt[N];

int main()
{
    //freopen("in.txt","r",stdin);

    scanf("%d",&n);

    int x;
    for(int i=0;i<n;i++){
        scanf("%d",&x);
        a[x]++;
    }

    for(int i=1;i<N;i++){
        if(a[i]){
            for(int j=1;j*i<N;j++){
                if(i==j) cnt[i*j]+=(__int64)a[i]*(a[i]-1);
                else cnt[i*j]+=(__int64)a[i]*a[j];
            }
        }
    }

    for(int i=1;i<N;i++) cnt[i]+=cnt[i-1];

    scanf("%d",&m);
    for(int i=0;i<m;i++){
        scanf("%d",&p);
        __int64 ans=(__int64)n*(n-1)-cnt[p-1];
        printf("%I64d\n",ans);
    }

    return 0;
}
