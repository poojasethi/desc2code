#include <cstdio>
int c[3000003],ans,n,m,a;
long long s[3000003];
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;++i){scanf("%d",&a);++c[a];}
	for(int i=1;i<2001;++i)
	for(int j=i;c[i]&&i*j<3000001;++j)
	if(i-j)s[i*j]+=c[i]*c[j];else s[i*j]+=(c[i]-1)*c[i]/2;
	for(int i=1;i<3000001;++i)s[i]+=s[i-1];
	for(scanf("%d",&m);m--;)
	{
		scanf("%d",&a);
		printf("%I64d\n",(long long)n*(n-1)-2*s[a-1]);
	}
	return 0;
}
