#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
typedef long long LL;
const int N=3e6 + 10;
LL sum[N], a[N], n;
int main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		int x;
		scanf("%d",&x);
		a[x]++;
	}
	for(LL i=1;i<N;i++){
		if(i * i<N) sum[i * i]+=a[i] * (a[i]-1) / 2;
		for(LL j=i+1;i * j < N;j++)
			sum[i*j] += a[i] * a[j]; 
	}
	for(int i=1;i<N;i++)
		sum[i] += sum[i-1];
	int m;
	cin>>m;
	LL total = n * (n-1);
	for(int i=1;i<=m;i++){
		int p;
		scanf("%d",&p);
		printf("%I64d\n",total - sum[p-1] * 2);
	}
}
