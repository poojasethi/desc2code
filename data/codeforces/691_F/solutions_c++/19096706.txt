#include <bits/stdc++.h>
#define INF 3000001
using namespace std;

int n, m;
int a, b;
//deque<int> mydeque[INF];
int say[INF];
long long ans[INF], curans, az[INF];

int main()
{
    //freopen("out.txt", "w", stdout);


    scanf("%d", &n);
    curans = 1LL * n * (n-1);
    ans[1] = curans;
    for(int i = 1; i <= n; ++i){
        scanf("%d", &a);
        say[a]++;
    }

    for(int i = 1; i < INF; ++i)
    for(int j = i; j < INF; j += i){
        if(i != j/i)
            az[j] = az[j] + 1LL * say[i] * say[j/i];
        else
            az[j] = az[j] + 1LL * say[i] * (say[j/i]-1);
    }

    for(int i = 1; i < 3000000; ++i){
        curans -= az[i];
        ans[i+1] = curans;
    }

    scanf("%d", &m);
    for(int i = 1; i <= m; ++i){
        scanf("%d", &b);
        printf("%I64d\n", ans[b]);
        //cout<<ans[b]<<endl;
    }

    return 0;
}
