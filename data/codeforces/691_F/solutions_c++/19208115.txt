#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const int maxa = 3e6;
ll n, a, all, m, q, num[maxa], sum[maxa];

int main() {
	scanf("%I64d", &n);
	all = n * (n - 1);
	while(n--) {
		scanf("%I64d", &a);
		m = max(m, a);
		num[a]++;
	}
	for(ll i = 1; i <= m; i++) {
		for(ll j = 1; j <= m; j++) {
			if(i * j > maxa) {
				break;
			}
			if(i == j) {
				sum[i*i] += num[i] * (num[i] - 1);
			}
			else {
				sum[i*j] += num[i] * num[j];
			}
		}
	}
	for(ll i = 2; i <= maxa; i++) {
		sum[i] += sum[i-1];
	}
	scanf("%I64d", &q);
	while(q--) {
		scanf("%I64d", &a);
		printf("%I64d\n", all - sum[a-1]);
	}
	return 0;
}
