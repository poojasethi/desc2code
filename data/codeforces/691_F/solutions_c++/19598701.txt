#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
#define N 3000006
#define M 1735
#define LL long long
#define TAT "%I64d"
using namespace std;
int n,m,q,a[N]; LL ans[N];
int main()
{
	int i,j,x;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&x),a[x]++;
	m=1732;
	for(i=1;i<=m;i++){
		ans[i*i]+=1LL*a[i]*(a[i]-1);
		for(j=i+1;j<=(N-1)/i;j++)
			ans[i*j]+=2LL*a[i]*a[j];
	  }
	for(i=1;i<N;i++)
		ans[i]+=ans[i-1];
	for(i=0;i<N;i++)
		ans[i]=1LL*n*(n-1)-ans[i];
	scanf("%d",&q);
	for(i=1;i<=q;i++){
		scanf("%d",&x);
		printf(TAT"\n",ans[x-1]);
	  }
	return 0;
}
