#define N 3000010

ll cnt[N];
ll equalTo[N];

int main () {

	ll n,m;

	scanf("%lld", &n);
	ll a;
	For(i,0,n) {
		scanf("%lld", &a);
		cnt[a]++;
	}

//	int to = (int)sqrt(N) + 5;
	for(int i=1;i<N;i++) {
		if(cnt[i] == 0) continue;
		for(int j=i; j < N; j+=i) {
			if(j/i != i) {
				equalTo[j] += cnt[i] * cnt[j/i];
			} else equalTo[j] += (cnt[i]-1) * cnt[j/i];
		}
	}

	For(i,1,N) {
		equalTo[i] += equalTo[i-1];
	}

	scanf("%lld", &m);
	For(i,0,m) {
		scanf("%lld", &a);
		printf("%lld\n", n*(n-1) - equalTo[a-1]);
	}


}
