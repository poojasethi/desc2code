#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstdlib>
#include<cstring>
#include<string>
#include<algorithm>
#include<vector>
#include<map>
using namespace std;
int s[2][2];
int main(){
	//freopen("in.txt", "r", stdin);
	//freopen("out.txt", "w", stdout);
	int n;
	while (scanf("%d ", &n) != EOF){
		vector<long long> x;
		vector<long long> y;
		vector<pair<long long, long long> > p;
		for (int i = 0; i < n; ++i){
			int a, b;
			scanf("%d %d ", &a, &b);
			x.push_back(a);
			y.push_back(b);
			p.push_back(make_pair(a, b));
		}
		sort(x.begin(), x.end());
		sort(y.begin(), y.end());
		int mid = 0, m = (n - 1) / 2;
		memset(s, 0, sizeof(s));
		for (int i = 0; i < n; ++i){
			if ((p[i].first == x[m])&&(p[i].second == y[m])) ++mid;
			++s[p[i].first <= x[m]][p[i].second <= y[m]];
		}
		long long ans = 0;
		for (int i = 0, j = n - 1; i < j; ++i, --j)
			ans += y[j] + x[j] - y[i] - x[i];
		
		//cout<<ans<<endl;
		
		long long d = min(x[m + 1] - x[m], y[m + 1] - y[m]);
		if (n % 2) d = min(d, min(x[m] - x[m - 1], y[m] - y[m - 1]));
		//cout<<d<<' '<<m<<endl;
		if (((n % 2 == 0)||mid)&&(s[0][0])&&(s[1][0])) // + s[1][1]  + s[0][1]
			ans -= d;
		cout<<ans * 2<<endl;
		//printf("%d\n", ans * 2);
	}
}
