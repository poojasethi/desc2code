#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

typedef long long LL;

const int N = int(1e5 + 5);

int n, m, mx, my;
int a[N], b[N];
int x[N], y[N];

LL min(LL x, LL y){return x<y ? x:y;}

int main()
{

	scanf("%d", &n);
	for(int i=1; i<=n; i++) scanf("%d%d", &a[i], &b[i]), x[i] = a[i], y[i] = b[i];
	
	sort(x+1, x+n+1), sort(y+1, y+n+1);
	m = (n+1)>>1, mx = x[m], my = y[m];
	
	LL ans(0), d( min(x[m+1] - x[m], y[m+1] - y[m]) );
	if(n&1) d = min(d, min(x[m] - x[m-1], y[m] - y[m-1]));
	
	bool p1(0), p2(0), p3(0);
	for(int i=1; i<=n; i++)
	{
		ans += abs(a[i] - mx) + abs(b[i] - my);
		p1 |= (a[i]>mx && b[i]>my);
		p2 |= (a[i]>mx && b[i]<=my);
		p3 |= (a[i]==mx && b[i]==my);
	}
	if(p1 && p2 && (n%2==0 || p3)) ans -= d;
	printf("%I64d\n", ans * 2);
	
	return 0;
}
