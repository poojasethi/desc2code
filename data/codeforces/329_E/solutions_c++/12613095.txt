#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll N=100100;
ll x[N],y[N],xx[N],yy[N],box[5],n,ans;
double mdx,mdy;
int main()
{
    ios_base::sync_with_stdio(false);
    cin>>n;
    for (int i=1;i<=n;i++)
    {
        cin>>x[i]>>y[i];
        xx[i]=x[i];
        yy[i]=y[i];
    }
    sort(x+1,x+n+1);
    sort(y+1,y+n+1);
    int md=n>>1;
    if(n&1)
    {
        mdx=x[(n+1)/2];
        mdy=y[(n+1)/2];
    }
    else
    {
        mdx=(x[md]+x[md+1])/2.0;
        mdy=(y[md]+y[md+1])/2.0;
    }
    for(int i=1;i<=n;i++)
    {
        if(xx[i]==mdx && yy[i]==mdy)
            box[4]=1;
        if(xx[i]==mdx || yy[i]==mdy)
            continue;
        ll u=0;
        if(xx[i]>mdx)
            u+=2;
        if(yy[i]>mdy)
            u++;
        box[u]=1;
    }
    for(int i=1;i<=md;i++)ans+=1LL*(x[n-i+1]-x[i]+y[n-i+1]-y[i]);
    if(!(accumulate(box,box+4,0)<=2 || (n&1 && !box[4])) && (n&1))
        ans=max(ans-min(mdx-x[md],x[md+2]-mdx),
                  ans-min(mdy-y[md],y[md+2]-mdy));
    else if(!(accumulate(box,box+4,0)<=2 || (n&1 && !box[4])))
        ans-=min(x[md+1]-x[md],y[md+1]-y[md]);
    cout<<2*ans<<'\n';

    return 0;
}
