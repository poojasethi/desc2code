#include <bits/stdc++.h>
using namespace std;
const int maxn=111111;
int m,n,i;
long long mx,my,dx,dy,x[maxn],y[maxn],sx[maxn],sy[maxn],an;
bool CC,C1,C2;
int main(){
    //freopen("input.txt","r",stdin);
    scanf("%d",&n),m=(n+1)>>1;
    for(i=1;i<=n;++i)scanf("%lld%lld",&x[i],&y[i]),sx[i]=x[i],sy[i]=y[i];
    sort(sx+1,sx+n+1),sort(sy+1,sy+n+1),mx=sx[m],my=sy[m];
    dx=sx[m+1]-sx[m],dy=sy[m+1]-sy[m];
    if(n&1)dx=min(dx,sx[m]-sx[m-1]),dy=min(dy,sy[m]-sy[m-1]);
    an=C1=C2=CC=0;
    for(i=1;i<=n;++i){
        an+=labs(mx-x[i])+labs(my-y[i]);
        if(x[i]>mx&&y[i]<=my)C1=1;
        if(x[i]>mx&&y[i]>my)C2=1;
        if(x[i]==mx&&y[i]==my)CC=1;
    }
    if((!(n&1))||CC)if(C1&&C2)an=max(an-dx,an-dy);
    an<<=1,printf("%lld\n",an);
    return 0;
}
