#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define N 120000
using namespace std;
int i,n,m,x,y,X[N],Y[N],bo[5],px[N],py[N];
long long ans;
double xx,yy;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i)scanf("%d%d",&x,&y),X[i]=x,Y[i]=y,px[i]=x,py[i]=y;
	sort(X+1,X+n+1);
	sort(Y+1,Y+n+1);
	if(n&1)xx=X[(n+1)/2],yy=Y[(n+1)/2];
	else xx=(X[n/2]+X[n/2+1])/2.0,yy=(Y[n/2]+Y[n/2+1])/2.0;
	for(i=1;i<=n;++i){
		if(px[i]==xx && py[i]==yy)bo[4]=1;
		if(px[i]==xx || py[i]==yy)continue;
		int u=0;
		if(px[i]>xx)u+=2;
		if(py[i]>yy)u++;
		bo[u]=1;
	}
	for(i=1;i<=n/2;++i)ans+=X[n-i+1]-X[i]+Y[n-i+1]-Y[i];
	if(bo[0]+bo[1]+bo[2]+bo[3]<=2 || (n&1 && !bo[4])){}
	else if(n&1){
		ans=max(ans-min(xx-X[n/2],X[n/2+2]-xx),ans-min(yy-Y[n/2],Y[n/2+2]-yy));
	}
	else{
		ans-=min(X[n/2+1]-X[n/2],Y[n/2+1]-Y[n/2]);
	}
	printf("%I64d\n",ans*2);
}
