#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <cctype>
#include <algorithm>
#include <iostream>
#include <utility>
#define ft first
#define sd second
#define mp make_pair
using namespace std;

typedef long long LL;
typedef pair<int,int> pii;

const int N=100000+5;

int n,m,A,B,x[N],y[N];
LL ans;
pii p1[N],p2[N];

int main(int argc,char **argv)
{
	#ifndef ONLINE_JUDGE
		if (argc==1) freopen("329E.in","r",stdin),freopen("329E.out","w",stdout);
		else freopen(argv[1],"r",stdin),freopen(argv[2],"w",stdout);
	#endif
	int xx,yy;
	scanf("%d",&n);
	for (int i=1;i<=n;++i)
	{
		scanf("%d%d",&x[i],&y[i]);
		p1[i]=mp(x[i],y[i]);
		p2[i]=mp(y[i],x[i]);
	}
	sort(p1+1,p1+n+1);
	sort(p2+1,p2+n+1);
	for (int i=1;i<=n;++i)
	{
		xx=lower_bound(p1+1,p1+n+1,mp(x[i],y[i]))-p1;
		yy=lower_bound(p2+1,p2+n+1,mp(y[i],x[i]))-p2;
		if (xx*2<=n)
		{
			if (yy*2<=n) ++A;else
			if ((n-yy+1)*2<=n) ++B;
		}
	}
	m=(n+1)>>1;
	for (int i=1;i<=m;++i) ans+=p1[n-i+1].ft+p2[n-i+1].ft-p1[i].ft-p2[i].ft;
	if (A && B)
	{
		if (n&1)
		{
			if (p1[m].sd==p2[m].ft)
				ans-=min(min(p1[m+1].ft-p1[m].ft,p1[m].ft-p1[m-1].ft),min(p2[m+1].ft-p2[m].ft,p2[m].ft-p2[m-1].ft));
		}else{
			ans-=min(p1[m+1].ft-p1[m].ft,p2[m+1].ft-p2[m].ft);
		}
	}
	cout<<(ans<<1)<<endl;
	return 0;
}
