#include <cstdio>
#include <algorithm>
using namespace std;

const int N = 100010;
int x[N],y[N],cx[N],cy[N],s[2][2];
int n,m,vx,vy,dx,dy;
long long ans=0;

int main()
{
    scanf("%d",&n);
    m=(n-1)/2;
    for(int i=0;i<n;++i)
	{
	    scanf("%d%d",x+i,y+i);
		cx[i]=x[i];
		cy[i]=y[i];
	}

    sort(x,x+n); dx = x[m+1]-x[m];
    sort(y,y+n); dy = y[m+1]-y[m];
    if(n%2)
	{
        dx = min(dx,x[m]-x[m-1]);
        dy = min(dy,y[m]-y[m-1]);
    }
    vx = x[m];
	vy = y[m];

    bool exist=false;
    for(int i=0;i<n;++i)
	{
        ans += abs(x[i]-vx)+abs(y[i]-vy);
        
        int temp1=0;
        if(cx[i]<=vx) temp1=1;
        int temp2=0;
        if(cy[i]<=vy) temp2=1;
        s[temp1][temp2]=1;
        
        if(vx==cx[i]&&vy==cy[i]) exist=true;
    }

    if((n%2==0 || exist) && s[0][0]&s[0][1]) ans=max(ans-dx,ans-dy);
    printf("%I64d\n",ans*2);
    return 0;
}
