#include <iostream>
#include <algorithm>
#include <stdlib.h>
#include <math.h>
using namespace std;
typedef long long ll;
ll x[100100],y[100100],cx[100100],cy[100100];
int f[2][2];
int main()
{
    int n;
    cin >>n;
    for (int i=1; i<=n; i++)
    {
        cin >>x[i] >>y[i];
        cx[i]=x[i]; cy[i]=y[i];
    }
    int m=(n+1)/2;
    sort(x+1,x+n+1);
    sort(y+1,y+n+1);
    ll dx=x[m+1]-x[m];
    ll dy=y[m+1]-y[m];
    if (n%2==1)
    {
        dx=min(dx,x[m]-x[m-1]);
        dy=min(dy,y[m]-y[m-1]);
    }
    ll itx=x[m],ity=y[m];
    ll k=0;
    ll ans=0;
    for (int i=1; i<=n; i++)
    {
        ans+=abs(x[i]-itx)+abs(y[i]-ity);
        f[cx[i]<=itx][cy[i]<=ity]=1;
        if (cx[i]==itx && cy[i]==ity) k=1;
    }
    if ((k || n%2==0) && f[0][0]&f[0][1]) ans=max(ans-dx,ans-dy);
    cout <<ans*2 <<endl;
    return 0;
}
