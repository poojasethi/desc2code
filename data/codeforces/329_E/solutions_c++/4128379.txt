int n,x[N],y[N],p[N],cz1[N],cz2[N],typ[N],p1[N],p2[N];
bool qx(int a,int b){
	return x[a]<x[b];
}
bool qy(int a,int b){
	return y[a]<y[b];
}
ll total=0;
int main(){
	cin.sync_with_stdio(false);
	cout.sync_with_stdio(false);
	cin>>n;
	for (int i=0;i<n;i++) {
		cin>>x[i];cin>>y[i];p1[i]=i;p2[i]=i;
	}
	sort(p1,p1+n,qx);sort(p2,p2+n,qy);
	int czt1=0,czt2=0;
	for (int i=0;i<n-1;i++){
		total+=ll((x[p1[i+1]]-x[p1[i]]))*2*min(i+1,n-i-1);
		total+=ll((y[p2[i+1]]-y[p2[i]]))*2*min(i+1,n-i-1);
	}
	if (n&1&&p1[n/2]!=p2[n/2]){
		cout<<total<<endl;
		return 0;
	}
	for (int i=0;i<n/2;i++){
		cz1[p1[i]]=1;
		cz1[p2[i]]=1;
		cz2[p1[n-i-1]]=1;
		cz2[p2[n-i-1]]=1;
	}
	for (int i=0;i<n;i++){
		if ((n&1==0)||i!=p1[n/2])
			if (cz1[i]&&cz2[i]){
				typ[i]=1;
				czt1=1;
			}
			else{
				typ[i]=2;
				czt2=1;
			}
	}
	if (czt1==0||czt2==0){
		cout<<total<<endl;return 0;
	}
	sort(x,x+n);sort(y,y+n);
	if (n&1){
		cout<<total-2*min(min(x[n/2]-x[n/2-1],y[n/2]-y[n/2-1]),min(x[n/2+1] - x[n/2],y[n/2+1] - y[n/2]));
	}
	else cout<<total-2*min(x[n/2]-x[n/2-1],y[n/2]-y[n/2-1])<<endl;
	return 0;
}
