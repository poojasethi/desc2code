#include<cstdio>
#include<algorithm>
int n,m;
int x[100010],y[100010],px[100010],py[100010],sx,sy;
int flag,f[2][2];
long long ans;
int abs(int c)
{
    if(c<0)
        return -c;
    return c;
}
int main()
{
    int i,j;
    scanf("%d",&n);
    for(i=1;i<=n;++i)
    {
        scanf("%d%d",x+i,y+i);
        px[i]=x[i];
        py[i]=y[i];
    }
    std::sort(px+1,px+n+1);
    std::sort(py+1,py+n+1);
    sx=px[n+1>>1];
    sy=py[n+1>>1];
    for(i=1;i<=n;++i)
    {
        ans+=abs(x[i]-sx)+abs(y[i]-sy);
        f[x[i]<=sx][y[i]<=sy]=1;
        if(sx==x[i]&&sy==y[i])
            flag=1;
    }
    if(f[0][0]&&f[0][1]&&(flag||(n%2==0)))
    {
        if(n&1)
        {
            m=px[n/2+1]-px[n/2];
            if(m>py[n/2+1]-py[n/2])
                m=py[n/2+1]-py[n/2];
            if(m>px[n/2+2]-px[n/2+1])
                m=px[n/2+2]-px[n/2+1];
            if(m>py[n/2+2]-py[n/2+1])
                m=py[n/2+2]-py[n/2+1];
        }
        else
        {
            m=px[n/2+1]-px[n/2];
            if(m>py[n/2+1]-py[n/2])
                m=py[n/2+1]-py[n/2];
        }
        ans-=m;
    }
    printf("%I64d\n",2LL*ans);
    return 0;
}
