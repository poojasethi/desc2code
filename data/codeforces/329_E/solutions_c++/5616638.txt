#include <cstdio>
#include <algorithm>
#include <vector>
#include <iostream>
#include <map>
#include <cstdlib>
#include <set>
#include <iomanip>
#include <string.h>
#include <cmath>
#define x first
#define y second
#define pb push_back
#define mp make_pair
#define ABS(a) ((a)>0?(a):(-(a)))
using namespace std;
typedef long long LL;
typedef double ld;
typedef pair<ld,ld> pii;
const int NMax=110000;
int N;
bool cmp1(pii a,pii b) {
	return a.x<b.x;
}
bool cmp2(pii a,pii b) {
	return a.y<b.y;
}
pii A[NMax];
int main()	
{
	scanf("%d",&N);
	for(int i=1;i<=N;i++) {
		scanf("%lf%lf",&A[i].x,&A[i].y);
		A[i].x+=i/1000000.0;
		A[i].y+=i/1000000.0;
	}
	int mid=N/2+1,a=0,b=0;
	ld midx,midy;
	if(N%2){
		sort(A+1,A+N+1,cmp1);
		midx=A[mid].x;
		sort(A+1,A+N+1,cmp2);
		midy=A[mid].y;
	}else {
		sort(A+1,A+N+1,cmp1);
		midx=(A[mid].x+A[mid-1].x)/2;
		sort(A+1,A+N+1,cmp2);
		midy=(A[mid].y+A[mid-1].y)/2;
	}
	LL ret=0;
	int flag=0;
	for(int i=1;i<=N;i++) {
		if((A[i].x<midx&&A[i].y<midy)||(A[i].x>midx&&A[i].y>midy)) a++;
		else if((A[i].x<midx&&A[i].y>midy)||(A[i].x>midx&&A[i].y<midy))b++;
		if(A[i].x<midx) ret-=floor(A[i].x);
		else if(A[i].x>midx) ret+=floor(A[i].x);
		if(A[i].y<midy) ret-=floor(A[i].y);
		else if(A[i].y>midy) ret+=floor(A[i].y);
		if(A[i].x==midx&&A[i].y==midy) flag=1;
	}
	ret*=2;
	if(N%2) {
		if(a==0||b==0||!flag) printf("%lld\n",ret);
		else {
			sort(A+1,A+N+1,cmp1);
			LL a=max(ret+2*floor(A[mid].x)-2*floor(A[mid+1].x),ret-2*floor(A[mid].x)+2*floor(A[mid-1].x));
			sort(A+1,A+N+1,cmp2);
			a=max(a,(LL)max(ret+2*floor(A[mid].y)-2*floor(A[mid+1].y),ret-2*floor(A[mid].y)+2*floor(A[mid-1].y)));
			printf("%lld\n",a);
		}
	}else {
		if(a==0||b==0) printf("%lld\n",ret);
		else {
			sort(A+1,A+N+1,cmp1);
			LL a=ret-2*floor(A[mid].x)+2*floor(A[mid-1].x);
			sort(A+1,A+N+1,cmp2);
			a=max((ld)a,ret-2*floor(A[mid].y)+2*floor(A[mid-1].y));
			printf("%lld\n",a);
		}
	}
	return 0;
}
