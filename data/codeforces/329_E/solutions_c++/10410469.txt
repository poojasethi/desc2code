using namespace std;

const int maxn=100000+15;
struct Node
{
	int x,y;
}a[maxn];
int x[maxn],y[maxn];
int n;
int abs(int x)
{
	if (x<0) return -x;
	return x;
}
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d%d",&a[i].x,&a[i].y);
		x[i]=a[i].x;
		y[i]=a[i].y;
	}
	sort(x+1,x+n+1);
	sort(y+1,y+n+1);
	int qq=(n+1)/2;
	int xmid=x[qq],ymid=y[qq];
	int delx=x[qq+1]-x[qq],dely=y[qq+1]-y[qq];
	if (n&1)
	{
		delx=min(delx,x[qq]-x[qq-1]);
		dely=min(dely,y[qq]-y[qq-1]);
	}
	long long ans=0;
	bool f1=false,f2=false,f3=false;
	for (int i=1;i<=n;i++)
	{
		ans+=abs(a[i].x-xmid)+abs(a[i].y-ymid);
		if (a[i].x>xmid && a[i].y>ymid) f1=true;
		if (a[i].x>xmid && a[i].y<=ymid) f2=true;
		if (a[i].x==xmid && a[i].y==ymid) f3=true;
	}
	if (f1 && f2)
	 if (n%2==0 || f3) ans-=min(delx,dely);
	cout << ans*2 << endl;
	return 0;
}
