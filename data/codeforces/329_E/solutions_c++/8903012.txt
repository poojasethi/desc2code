#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <string>
using namespace std;
typedef long long LL;
const int Maxn=100005;
int n,px[Maxn],py[Maxn],x[Maxn],y[Maxn];
bool flag[2][2];
LL ans=0;

inline int get() {
	char ch;
	while (!isdigit(ch=getchar()));
	int v=ch-48;
	while (isdigit(ch=getchar())) v=v*10+ch-48;
	return v;
}

int main() {
	n=get();
	for (int i=1;i<=n;i++) x[i]=px[i]=get(),y[i]=py[i]=get();
	sort(x+1,x+n+1);
	sort(y+1,y+n+1);
	int mid=(n+1)/2,xm=x[mid],ym=y[mid],cost_x=x[mid+1]-x[mid],cost_y=y[mid+1]-y[mid];
	if (n&1)
		cost_x=min(cost_x,x[mid]-x[mid-1]),
		cost_y=min(cost_y,y[mid]-y[mid-1]);
	bool f=0;
	for (int i=1;i<=n;i++) {
		ans+=abs(x[i]-xm)+abs(y[i]-ym);
		flag[px[i]<=xm][py[i]<=ym]=1;
		f|=px[i]==xm&&py[i]==ym;
	}
	if ((n%2==0||f)&&flag[0][0]&&flag[0][1]) ans-=min(cost_x,cost_y);
	cout<<ans*2<<endl;
	return 0;
}
