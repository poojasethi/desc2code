#define sqr(_) ((_)*(_))

typedef long long LL;

typedef pair<int, int> Pii;
#define x first
#define y second
#define mp make_pair

const int MAXN = 100000 + 10;

int n, m, dx, dy, x[MAXN], y[MAXN];
bool area[2][2];
Pii p[MAXN];
LL Tohka;

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt", "r", stdin), freopen("output.txt", "w", stdout);
#endif
	cin >> n;
	m = (n + 1) / 2;
	for (int i = 1; i <= n; ++i)
		cin >> x[i] >> y[i], p[i] = mp(x[i], y[i]);
	sort(x + 1, x + n + 1);
	dx = x[m + 1] - x[m];
	sort(y + 1, y + n + 1);
	dy = y[m + 1] - y[m];
	if (n & 1)
		dx = min(dx, x[m] - x[m - 1]), dy = min(dy, y[m] - y[m - 1]);
	bool f = false;
	for (int i = 1; i <= n; ++i)
	{
		Tohka += abs(x[i] - x[m]) + abs(y[i] - y[m]);
		area[p[i].x <= x[m]][p[i].y <= y[m]] = true;
		if (p[i] == mp(x[m], y[m]))
			f = true;
	}
	if ((!(n & 1) || f) && area[0][0] && area[0][1])
		Tohka -= min(dx, dy);
	cout << Tohka * 2 << endl;
	fclose(stdin);
	fclose(stdout);
	return 0;
}
