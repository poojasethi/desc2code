def find_money(price, customers):
    total = 0
    sold = 0
    for y in reversed(customers):
        if y is 1: #halfplus
            if total is 0: #last one sold must be halfplus
                total = 1
                sold = 0.5
            else:
                total = (total * 2) + 1
                sold = sold + (total / 2.0)
        else: #half
            total = total * 2
            sold = sold + (total / 2)
    return sold * price


numbers = map(int, raw_input().split(' '))
n = numbers[0]
p = numbers[1]
customers = []
for x in range(0, n):
    value = raw_input()
    if value == "half":
        customers.append(0)
    else:
        customers.append(1)


print int(find_money(p, customers))
