def apples():
    n, p = map(int, raw_input().split())
    i = []
    for j in xrange(n):
        i.append(raw_input())
    apples = 1
    money = p/2.0
    for j in xrange(len(i)-2,-1,-1):
        if i[j] == 'half':
            apples *= 2
            money = money + (apples/2.0*p)
        
        else:
            apples = 2*apples + 1
            money = money + (apples/2.0*p)
    
    print int(money)        

if __name__ == "__main__":
    apples()
