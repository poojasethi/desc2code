seq = []
count=1
n , p = [int(e) for e in raw_input().split()]

for _ in range(n):
	seq.append(raw_input())

if seq[-1]=='halfplus':
	halfcount = 1
else:
	halfcount = 0

for i in range(n-1):
	if seq[-i-2]=='halfplus':
		count=count*2+1
		halfcount+=1
	else:
		count=count*2
print (p*count-p/2*halfcount)
