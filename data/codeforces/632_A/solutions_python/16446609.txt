import math

nn = raw_input().split(' ')
n = int(nn[0])
p = int(nn[1])

num = []
half_N = 0

apple_N = 0
for i in range(n):
	tmp = raw_input()
	if tmp == 'half':
		tmp = 0
	else:
		tmp = 1
		apple_N += math.pow(2,i)
		half_N += 1
	num.append(tmp)
# (reversed binary array...)

a = apple_N*p - half_N*p/2
print int(a)
