from sys import stdin

n, p = map(int, stdin.readline().strip().split(' '))
stack = []
for i in xrange(n):
    str = stdin.readline().strip()
    if str == 'half':
        stack.append(0)
    if str == 'halfplus':
        stack.append(1)

current = 0.0
sum = 0.0
for item in reversed(stack):
    if item == 0:
        sum += current * p
        current *= 2
    if item == 1:
        sum += (current + 0.5) * p
        current = current*2 + 1

print int(sum)
