(n,p) = [int(x) for x in raw_input().split()]
people = []
for i in range(n):
    people.append(raw_input())

"""n = 3
p = 10
people = ['halfplus','halfplus','halfplus']"""

j = n-1
money = 0
apples_sold = 0.0
while j >= 0:
    current = people[j]
    if current == 'halfplus':
        money+=(apples_sold + 0.5)*p
        apples_sold += apples_sold+1
    elif current == 'half':
        money+=apples_sold*p
        apples_sold+=apples_sold
    j-=1
print int(money)
