n,p = map(int, raw_input().split())
split_string = []
for i in xrange(n):
    split_string.append(raw_input())

result = p / 2
temp = 1
for i in range(len(split_string)-2, -1, -1):
    if split_string[i] == "half":
        result += temp * p
        temp *= 2
    elif split_string[i] == "halfplus":
        temp = temp * 2 + 1
        result += int(temp / 2.0 * p)

print result
