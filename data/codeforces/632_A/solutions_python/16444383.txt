if(__name__ == "__main__"):
	n, p = [int(tmp) for tmp in raw_input().split(' ')]
	l = []
	for i in range(n):
		l.append(raw_input() == "half")
	if(l[-1]):
		ans, buy = 2, 1
	else:
		ans, buy = 1, 0.5
	for i in range(n - 2, -1, -1):
		if(l[i]):
			buy += ans
			ans = ans * 2
		else:
			buy += ans + 0.5
			ans = ans * 2 + 1
	print(int(buy * p))
