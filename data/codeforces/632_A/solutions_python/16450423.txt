def main():
    n, price = map(int, raw_input().strip().split())
    currentMoney = (price / 2)
    currentApple = 1
    xs = [raw_input() for _ in xrange(n - 1)]
    for x in reversed(xs):
        if(x == "halfplus"):
            currentApple = (currentApple * 2) + 1
            additionalMoney = (currentApple * (price / 2))
            currentMoney += additionalMoney
        else:
            additionalMoney = (currentApple * 2) * (price / 2)
            currentMoney += additionalMoney
            currentApple *= 2
    print(currentMoney)

main()
