s = raw_input()
s = s.split(' ')
n = int(s[0])
p = int(s[1])

buyers = ['' for i in range(n)]

i = 0
while i < n:
    buyers[i] = raw_input()
    i += 1

i = n - 2
apples = 1
money = p / 2

while i >= 0:
    if buyers[i] == 'halfplus':
        money += apples * p + p / 2
        apples += apples + 1
    elif buyers[i] == 'half':
        money += apples * p 
        apples += apples
    i -= 1

print money
