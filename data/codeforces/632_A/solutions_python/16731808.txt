n, p = map(int, raw_input().split())
half = []
for i in range(n):

	if raw_input() == "half":

		half.append(0)

	else:

		half.append(1)


have = 0
cost = 0
for i in range(n - 1, -1, -1):

	if half[i] == 1:

		cost += have * p
		cost += p / 2
		have = 2 * have + 1

	else:

		cost += have * p
		have *= 2

print cost
