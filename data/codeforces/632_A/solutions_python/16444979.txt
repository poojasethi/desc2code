i = raw_input().split(" ")
c = long(i[1])
lis = []
for j in range(long(i[0])):
	lis.append(raw_input())
ap = long(0)
cc = long(0)
for j in lis[::-1]:
	if (j == "halfplus"):
		ap = long(ap+long(1))
		cc = long(cc+long(long(ap)*long(c)))
		ap = long(ap*long(2))
	else:
		cc = long(cc+long(long(ap)*long(c)))
		ap = long(ap*long(2))
print(long(cc)/long(2))
