#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int pr=0,k=1;
	int n,p,i;
	cin>>n>>p;
	string s[n];
	for(i=0;i<n;i++)
	{
		cin>>s[i];
	}
	for(i=n-1;i>=0;i--)
	{
		pr=pr+(p/2)*k;
		if(s[i-1]=="halfplus")
		{
			k=2*k+1;
		}
		else if(s[i-1]=="half")
		{
			k=2*k;
		}
	}
	cout<<pr;
	return 0;
}
