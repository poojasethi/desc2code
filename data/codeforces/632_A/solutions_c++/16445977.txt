#include<bits/stdc++.h>
using namespace std;
int main()
{
long long int a,p,i,j,k;
cin>>a>>p;
string s;
vector<string> v;
for(i=0;i<a;i++)
{
cin>>s;
v.push_back(s);
}
long long int c=1;
k=1;
for(i=a-2;i>=0;i--)
{
	if(v[i].size()>4)
	{
		k=k*2+1;
		c++;
	}
	else
	{
		k=k*2;
	}
}
//cout<<c;
long long int ans=k*p-c*(p/2);
cout<<ans<<endl;
return 0;
}
