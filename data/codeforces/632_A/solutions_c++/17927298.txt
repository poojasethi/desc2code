#include<iostream>
#include<string>
using namespace std;

int n, p;
string s[55];
long long num, ans;

int main() {
	cin>>n>>p;
	for(int i=0;i<n;i++) cin>>s[i];
	num=ans=0LL;
	for(int i=n-1;i>=0;i--) {
		if(s[i].length()==4) {
			ans+=num*p;
			num=num*2;
		} else {
			ans+=num*p+p/2;
			num=num*2+1;
		}
	}
	cout<<ans<<endl;
	
	return 0;
}
