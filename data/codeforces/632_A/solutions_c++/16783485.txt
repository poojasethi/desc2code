#include<iostream>
#include<string>
using namespace std;
string a[40];
int main()
{
double n,m,t=0;
cin>>n>>m;
for(int i=0;i<n;i++){
cin>>a[i];if(a[i]=="halfplus")t++;}
long long ans=0;
for(int i=n-1;i>=0;i--)
if(a[i]=="halfplus"){
ans*=2;ans++;}
else
ans*=2;
long long ans1=ans*m-t*m/2;
cout<<ans1<<endl;
}
