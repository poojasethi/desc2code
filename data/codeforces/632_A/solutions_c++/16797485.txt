#include <stdio.h>
#include <string.h>
#define LL long long
using namespace std;
int n;
LL p;
char buf[10];
int main(){
	scanf("%d %I64d", &n, &p);
	LL ans = 0LL;
	LL c = 0LL;
	for(int i=0; i<n; i++){
		scanf("%s", buf);
		if(strlen(buf)==8)ans |= (1LL<<(LL)i), c++;
	}
	printf("%I64d\n", ans*p - c*p/2LL);
	return 0;
}
