#include<bits/stdc++.h>
using namespace std;
#define LL long long

int main()
{
	string ss[50];
	int n,p;
	cin>>n>>p;
	for(int i=1;i<=n;i++) cin>>ss[i];
	LL np=0,ans=0;
	for(int i=n;i>=1;i--){
		if(ss[i]=="halfplus"){
			np<<=1;
			np++;
			ans+=np*p/2;
		}else{
			np<<=1;
			ans+=np*p/2;
		}
	}
	cout<<ans<<endl;
	return 0;
}
