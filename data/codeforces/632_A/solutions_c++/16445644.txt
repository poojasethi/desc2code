#include<cstdio>
#include<cstring>
int main()
{
	int n,p,a[45],i;
	long long ans,num;
	char s[10];
	scanf("%d %d",&n,&p);
	for(i=0;i<n;i++)
	{
		scanf("%s",s);
		if(strlen(s)==4)
			a[i]=0;
		else
			a[i]=1;
	}
	for(num=ans=0,i=n-1;i>=0;i--)
	{
		if(a[i])
			num=num*2+1,ans+=num*p/2;
		else
			num*=2,ans+=num*p/2;
	}
	printf("%I64d\n",ans);
}
