#include<iostream>
using namespace std;
typedef long long ll;
ll n,p;
const int maxn=55;
string s[maxn];
int main()
{
	ll sum=1;
	cin>>n>>p;
	for(int i=1;i<=n;i++)
		cin>>s[i];
	ll num=1,cnt=1;
	while(--n){
		if(s[n]=="half")
			num*=2;
		else
			cnt++,num=num*2+1;
	}
	ll ans=ll(num*2-cnt)*(ll)p/2;
	cout<<ans<<endl;
}
