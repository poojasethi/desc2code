#include<bits/stdc++.h>

using namespace std;

int main(){
	long long n, cost1,ans=0,cost=0;
	cin>>n>>cost1;
	char S[45][20];
	for(int i =0; i< n; i++){
		cin>>S[i];
	}
	for( int k =n-1; k>=0; k--){
		cost = cost<<1;		
		if(!strcmp(S[k], "halfplus"))
			cost += 1;
		ans += cost*cost1/2;;
	}
	cout<<ans<<endl;
	return 0;
}
