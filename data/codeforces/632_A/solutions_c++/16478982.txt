#include<bits/stdc++.h>
using namespace std;
int n,p;
string ch;
int a[50];
unsigned long long int c=0,k=0;
int main(){
	cin>>n>>p;
	for(int i=0;i<n;i++)
	{cin>>ch; if(ch.length()==4) a[i]=0; else a[i]=1;}
	for(int i=n-1;i>=0;i--)
	{
		c+=k*p + a[i]*(p/2);
		k=2*k+a[i];
	}
	cout<<c<<"\n";
	return 0;
}
