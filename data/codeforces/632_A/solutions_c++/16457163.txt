#include<iostream>
using namespace std;

int m,p;
int main(){
	cin>>m>>p;
	string *s=new string[m];
	for(int i=0;i<m;i++){
		cin>>s[i];
	}
	long long n=1;
	int count=0;
	for(int i=m-2;i>=0;i--){
		if(s[i]=="halfplus"){
			count++;
		 	n=(n+0.5)*2;
		 }
		else n*=2;
	}
	long long sum=n*p-(count+1)*0.5*p;
	cout<<sum<<endl;
	return 0;
}
