#include <iostream>
#include <cstdio>
using namespace std;
long long i,j,n,p,m,ans,a[100];
char s[10];
int main()
{
		scanf("%I64d%I64d\n",&n,&p);
		for (i=1;i<=n;++i)
		{
			for (j=0;j<8;++j) s[j]='\0';
			scanf("%s\n",s);
			if (s[5]!='\0') a[i]=1;else a[i]=0;
		}
		ans=m=0;
		for (i=n;i;i--)
		{
			m=m*2+a[i];ans+=m;
		}
		printf("%I64d\n",ans*p/2);
}
