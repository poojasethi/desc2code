#include <bits/stdc++.h>
using namespace std;
double a[45];
long long n,p,k,ans;
string s[41];
int main () {
	cin>>n>>p;
	for (int i=1;i<=n;i++)
	  cin>>s[i];	
	for (int i=n;i>=1;i--){
	  if (s[i]=="half")
	    a[i]=a[i+1]*2;
	  else{
   	    a[i]=(a[i+1]+0.5)*2;		
        k++;
	  }
 	}  
 	a[1]-=0.5*k;
	ans=a[1]*p;
	cout<<ans;
	return 0;
}
