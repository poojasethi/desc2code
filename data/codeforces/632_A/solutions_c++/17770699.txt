#include<bits/stdc++.h>
using namespace std;
#define ll long long
string st[45];
int main()
{
   ll n,p;
   cin>>n>>p;
   for(int i=0;i<n;i++)
      cin>>st[i];
   double x=1,ans=p/2;
   for(int i=n-2;i>=0;i--)
   {
      if(st[i]=="halfplus")
         x=2*x+1;
      else
         x*=2;
      ans+= double(x/2)*p;
   }
   cout<<(ll)ans;
}
