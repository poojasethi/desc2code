#include<iostream>
#include<cstring>
using namespace std;
string s[50];
int main()
{
	long long n;
	long long p;
	cin>>n>>p;
	long long now=0;
	for(int i=1;i<=n;i++)
	{
		cin>>s[i];
	}
	long long cost=0;
	for(int i=n;i>=1;i--)
	{
		if(s[i]=="half")
		{
			now*=2;
		}
		else
		now=now*2+1;
		cost+=now;
	}
	cout<<cost*p/2<<endl;
}
