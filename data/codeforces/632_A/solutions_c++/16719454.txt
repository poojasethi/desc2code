#include <bits/stdc++.h>

using namespace std;

int N,P,h;
bool B[50];
long long rs,nr;
string s;
int main(){
	cin >> N >> P;
	nr = 1;
	for(int i = N;i;i--){
		cin >> s;
		if(s == "halfplus") B[i]=1,h++;
	}
	
	for(int i = 2;i<=N;i++){
		nr*=2;
		if(B[i])nr++;
	}
	cout << nr*P-(h*P)/2;
	
	return 0;
}
