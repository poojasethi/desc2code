#include <bits/stdc++.h>
using namespace std;
int main(){
	long long n, p;
	cin>>n>>p;
	string s[n];
	for (int i = 0; i < n; i++) cin>>s[i];
	long long tot = 0;
	for (int i = n - 1; i >= 0; i--) {
		if(s[i] == "halfplus") tot *= 2, tot++;
		else tot *= 2;
	}
	long long ans = 0;
	for (int i = 0; i < n; i++) {
		ans += tot * (p / 2); 
		tot /= 2;
	}
	cout<<ans<<endl;
}
