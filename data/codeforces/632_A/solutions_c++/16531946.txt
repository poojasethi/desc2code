#include<bits/stdc++.h>
using namespace std;

int main(){
	long long n,p,a=0,sum=0;
	string s[200];
	cin>>n>>p;
	for(int i=0;i<n;i++)
	cin>>s[i];
	for(int i=n-1;i>=0;i--){
		if(s[i]=="half"){
			sum+=a*p;
			a=a*2;
		}
		else{
			sum+=p/2+a*p;
			a=a*2+1;
		}
	}
	cout<<sum<<endl;
	return 0;
}
