#include <bits/stdc++.h>
using namespace std;
int main()
{
   ios::sync_with_stdio(0);
    long long int n,p,k=0,i,app=0,ans;
    string s;
    cin>>n>>p;
    for(i=0;i<n;i++)
    {
       cin>>s;
       if(s=="halfplus")
         {
             k++;
          app+=pow(2,i);
         }
    }
    ans=p*(app-0.5*k);
    cout<<ans;
    return 0;

}
