#include <iostream>
using namespace std;
int main()
{
	long long  i,n,p,sum=0,a=0;
	string arr[45];
	cin>>n>>p;
	for( i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	sum=0;
	while(n--)
	{
		if(arr[n]==("half"))
		{
			a=a+sum*p;
			sum=sum*2;
		}
		else if(arr[n]=="halfplus")
		{
			a=a+sum*p+p/2;
			sum=sum*2+1;
			
		}
	}
	cout<<a;
	return 0;
}
