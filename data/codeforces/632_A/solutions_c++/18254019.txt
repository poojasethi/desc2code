#include <iostream>
using namespace std;
int main(){
	int n,p;
	long long  m,sum = 0;
	string a[41];
	cin>>n>>p;
	for (int i=0;i<n;i++){
		cin>>a[i];
	}
	m = 0;
	for (int i=n-1;i>=0;i--){
		if (a[i] == "halfplus"){
			sum += m*p + p/2;
			m = m*2+1;
		}
		else {
			sum += m*p;
			m = m * 2;
		}
	}
	cout<<sum<<endl;
	return 0;
}
