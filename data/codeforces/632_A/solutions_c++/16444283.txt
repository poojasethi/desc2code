#include<cstdio>
#include<iostream>
#include<string>
using namespace std;
typedef long long I64;
int main()
{
	string str[50];
	I64 n,p,now=0,s=0;
	cin>>n>>p;
	for(int i=1;i<=n;++i)
	{
		cin>>str[i];
	}
	for(int i=n;i>=1;--i)
	{
		if(str[i].length()==8)
			now=now*2+1,++s;
		else now*=2;
	}
	cout<<(I64)(((double)now-0.5*s)*p)<<endl;
}
