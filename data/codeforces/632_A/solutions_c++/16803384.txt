#include <bits/stdc++.h>

using namespace std;

#define N 45

char s[N];
bool bf[N];

main() {
	int n, p;
	scanf("%d %d", &n, &p);
	for (int i = 1; i <= n; i ++) {
		scanf("%s", s);
		bf[i] = (int)(strlen(s)) == 8;
	}
	__int64 rlt = 0, cur = 0;
	for (int i = n; i; i --) {
		cur = cur * 2 + bf[i];
		rlt += cur * p / 2;
	}
	printf("%I64d\n", rlt);
}
