#include<iostream>
#include<string>
using namespace std;
long long n,i,x,price,ans;
long double count;
string s[41];
int main()
 {
 cin>>n>>price;
 for(i=n-1;i>=0;i--)
 cin>>s[i];
 for(i=0;i<n;i++)
  {
  if(s[i]=="halfplus")
  x=2*x+1;

  else
  x=2*x;
  count+=(long double)x/2;


  }
  ans=price*count;
 cout<<ans;
 return 0;
 }
