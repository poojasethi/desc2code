#include <iostream>
#include <vector>
#include <string>

using namespace std;

typedef long long int ll;

int main()
{
  ll n,p;
  cin>>n>>p;
  p/=2;
  vector<string> v(n);
  for (int i=0;i<n;i++)
    cin>>v[i];
  ll suma=0;
  ll x=0;
  for (int i=n-1;i>=0;i--) {
    if (v[i]=="half") {
      x*=2;
    } else {
      x=2*x+1;
    }
    suma+=x;
  }
  cout<<p*suma<<endl;
}
