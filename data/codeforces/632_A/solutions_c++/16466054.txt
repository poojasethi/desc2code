#include <iostream>
#include <string>
using namespace std;
typedef long long ll;
int main(){
	int n,p;
	cin >> n >> p;
	p>>=1;
	string s[n];
	ll ans=0;
	ll apps =0;
	for(int i=0;i<n;i++)cin >> s[i];
	--n;
	while(n>=0){
		if (s[n]=="half"){
			apps *=2;
			ans+= apps*p;
		}
		else {
			apps = 2*apps+1;
			ans += apps*p;
		}
		--n;
	}
	cout << ans << endl;
	return 0;
}
