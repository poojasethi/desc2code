#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
	int n,p;
	cin>>n>>p;
	vector<string> v;
	for(int i=0;i<n;i++)
	{
		string s;
		cin>>s;
		v.push_back(s);
	}
	long long ans=0,k=0;
	reverse(v.begin(),v.end());
	for(int i=0;i<v.size();i++)
	{
		k*=2;
		if(v[i]=="halfplus")k++;
		ans+=k*p/2;
	}
	cout<<ans;
}
