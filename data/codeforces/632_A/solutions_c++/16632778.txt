#include<iostream>
#include<string>
using namespace std;
string s[45];
int main(){
	int n,p,i;
	long long sum,now;
	sum=now=0;
	cin>>n>>p;
	for(i=n-1;i>=0;i--)
		cin>>s[i];
		for(i=0;i<=n-1;i++){
			if(s[i]=="half"){
				sum+=now*p;
				now=2*now;
			}
			else{
				sum+=now*p+p/2;
				now=now*2+1;
			}
		}
		cout<<sum<<endl;
	return 0;
}
