#include <bits/stdc++.h>
using namespace std;
long long n,p,res=0,aux = 1;
string k[50];
int main(){
	cin >> n >> p;
	for(int i = n-1; i>=0; i--) cin >> k[i];
	for(int i = 1; i<n; i++){
		if(k[i] == "half") aux*=2;
		else aux = 2*aux + 1; 
	}
	while(aux>0){
		res+= (aux/2)*p + (aux%2)*(p/2);
		aux/=2;
	}
	cout << res << endl;
}
