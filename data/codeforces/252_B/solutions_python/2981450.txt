#!/usr/bin/python
N = int(raw_input())
A = map(long, raw_input().split())
for i in range(1, len(A)):
    if A[i - 1] != A[i]:
        A[i], A[i - 1] = A[i - 1], A[i]
        a, b = True, True
        for j in range(1, len(A)):
            if A[j - 1] > A[j]: a = False
            if A[j - 1] < A[j]: b = False
        if a or b: 
            A[i], A[i - 1] = A[i - 1], A[i]
            continue
        print("%d %d" % (i, i + 1))
        quit()
        
print(-1)
