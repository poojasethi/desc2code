n = input()
num = map( int, raw_input().split())
if n<3 or n==3 and num[0]==num[2] or min(num) == max(num):
  print -1
else:
  if num[0]==num[1]:
    for i, m in enumerate(num):
      if m != num[0]:
        print 2,i+1
        break
  else:
    if num[1]==num[2]:
      print 1,2
    elif num[1] <= num[0] and num[0] <= num[2] or ( num[1] >= num[0] and num[0] >= num[2] ):
      print 2,3
    else:
      print 1,2
