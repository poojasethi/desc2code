def main():
    n = int(raw_input())
    a = map(int, raw_input().split())
    for i in range(n-1):
        if a[i] != a[i+1]:
            a[i],a[i+1] = a[i+1],a[i]
            if a == list(sorted(a)) or a == list(sorted(a,reverse=True)):
               a[i],a[i+1] = a[i+1],a[i]
            else:
                print(str(i+1)+' '+str(i+2))
                return
    print(-1)
main()
