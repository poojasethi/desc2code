#include<cstdio>
#include<algorithm>
#define NN 100000
using namespace std;
int n;
int v[NN];

bool sorted(){
	bool a  =1 ,d =1 ;
	for( int i = 1 ; i  < n ; ++i ){
		if( v[i-1] > v[i] )
			a = 0;
		else if( v[i-1] < v[i]  )
			d = 0;
	}
	return a || d;
}

int main(){
	scanf("%d" , &n );
	for( int i = 0 ; i < n ;++i)
		scanf("%d" ,v+i );
	int tr = 0;
	for( int i = 1 ;  i< n;  ++i )
		tr += v[i] != v[i-1];
	if( n < 3 || !tr  ){
		puts("-1");
	}else{
		for( int i = 1;  i < n; ++i ){
			if( v[i] != v[i-1] ){
				swap( v[i] , v[i-1] );
				if( !sorted() ){
					printf("%d %d\n" , i , i+1 );
					return 0;
				}
				swap( v[i] , v[i-1] );
			}
		}
		puts("-1");
	}
	return 0;
}
