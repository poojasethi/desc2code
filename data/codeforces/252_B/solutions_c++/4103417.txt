#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
int cd[100005]={0};
int dc[100005]={0};
int n;
main(){
int a;	
int i;
int j;
scanf("%d",&n);
for(a=1;a<=n;a++){scanf("%d",&cd[a]);dc[a]=cd[a];}
if(n<=2){printf("-1\n");return 0;}
sort(dc+1,dc+n+1);
for(i=1;i<n;i++)
{
if(cd[i]==cd[i+1])continue;	
swap(cd[i],cd[i+1]);
j=1;
while(j<=n){
if(cd[j]!=dc[j])break;
j++;
}
if(j==n+1){swap(cd[i],cd[i+1]);continue;}
j=1;
while(j<=n){
if(cd[j]!=dc[n-j+1])break;
j++;
}
if(j==n+1){swap(cd[i],cd[i+1]);continue;}
printf("%d %d\n",i,i+1);return 0; 
}

printf("-1\n");
}
