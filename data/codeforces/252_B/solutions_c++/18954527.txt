#include <bits/stdc++.h>
using namespace std;
vector<int>v,a1,a2;
int main()
{
	int n;
	cin>>n;
	v.resize(n);
	for(int i=0;i<n;i++)scanf("%d",&v[i]);
	a1=a2=v;
	sort(a1.begin(),a1.end());
	sort(a2.rbegin(),a2.rend());
	for(int i=1;i<n;i++)
    {
		if(v[i]!=v[i-1])
		{
			swap(v[i],v[i-1]);
			if(v==a1 || v==a2)
			{
				swap(v[i],v[i-1]);
				continue;
			}
			cout<<i<<" "<<i+1<<endl;
			return 0 ;
		}
	}
	cout<<-1;
}
