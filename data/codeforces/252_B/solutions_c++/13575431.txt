#include <bits/stdc++.h>
#define ll long long
#define st first
#define nd second
using namespace std;


int main()
{
ll n;
cin>>n;

vector <ll>a(n);
vector <ll>b(n);
vector <ll>c(n);
for(int i=0;i<n;i++)
        cin>>a[i];
b=a;
c=a;
sort(b.begin(),b.end());
sort(c.rbegin(),c.rend());

for(int i=0;i<n-1;i++)
{
    if(a[i]!=a[i+1])
    {
        swap(a[i],a[i+1]);
        if(a!=b && a!=c)
            return cout<<i+1<<" "<<i+2<<endl,0;
        swap(a[i],a[i+1]);
    }
}

cout<<-1<<endl;

return 0;
}
