#include <iostream>
#include <algorithm>

using namespace std;
const int MAXN=100009;

int main()
{
    int n, x;
    int P[MAXN];

    cin>>n;

    for (int i=0; i<n; i++)
        cin>>P[i];

    for (int i=1; i<n; i++)
    {
	if (P[i] != P[i-1])
	{
        swap(P[i], P[i-1]);
        bool sorted1=false;
        bool sorted2=false;

        for (int j=1; j<n; j++)
        {
            if(P[j-1] > P[j]) sorted1=true;
            if(P[j-1] < P[j]) sorted2=true;

         if (!sorted1 || !sorted2) continue;
         cout<<i<<' '<<i+1;
         return 0;
        }
        swap(P[i], P[i-1]);
	}
    }

    cout<<-1;

    return 0;


}
