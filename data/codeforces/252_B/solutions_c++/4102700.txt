#include<stdio.h>
#include<string.h>
#include<math.h>
#include<iostream>
#include<algorithm>
using namespace std;
int a[100005],n;
void solve()
{
	int i,j,x,y;
	for(i=1;i<n;i++)
	{
		if(a[i]!=a[i-1])
		{
			swap(a[i],a[i-1]);
			x=y=0;
			for(j=1;j<n;j++)
			{
				if(a[j]<a[j-1]) x=1;
				if(a[j]>a[j-1]) y=1;
				if(x&&y)
				{
					printf("%d %d",i,i+1);
					return;	
				}
			}
			swap(a[i],a[i-1]);
		}
	}
	printf("-1\n");
}
int main()
{
	int i;
	while(~scanf("%d",&n))
	{
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
		solve();
	}
}
