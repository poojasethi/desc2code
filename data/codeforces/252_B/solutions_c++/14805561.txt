#include<stdio.h>
#define swap(a,b,c) (c=a,a=b,b=c)
int f[111111],n,i,tmp;
main(){
while(~scanf("%d",&n)){
	int flag=0;
for(i=1;i<=n;i++)scanf("%d",&f[i]);
   for(i=2;i<=n&&flag==0;i++)
 	  if(f[i]!= f[i-1]){
   		swap(f[i],f[i-1],tmp);
     bool ok1 = true,ok2 = true;
  	for(int j=2;j<=n;j++){
  	 if(f[j-1]>f[j])ok1 = false;
   	 if(f[j-1]<f[j])ok2 = false;
  	 if(ok1 || ok2) continue;
  flag=1;
  break;
  }
  if(flag) break;
  swap(f[i],f[i-1],tmp);
 }
   if(flag)  printf("%d %d\n",i-1,i);
   else  puts("-1");
 }
return 0;
}
