#include <stdio.h>
#include <algorithm>
using namespace std;
int n,arr[100001];
bool si[2];
int main()
{
    scanf("%d",&n);
    for (int i=0; i<n; i++) scanf("%d",&arr[i]);
    for (int i=1; i<n; i++){
        if (arr[i-1]!=arr[i]){
            swap(arr[i-1],arr[i]);
            si[0]=si[1]=true;
            for (int j=1; j<n; j++){
                if (arr[j-1]>arr[j]) si[0]=false;
                if (arr[j-1]<arr[j]) si[1]=false;
            }
            swap(arr[i-1],arr[i]);
            if (si[0]||si[1]) continue;
            printf("%d %d",i,i+1);
            return 0;
        }
    }
    printf("-1");
}
