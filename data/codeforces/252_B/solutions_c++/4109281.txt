#include<stdio.h>
int a[100005];
int main()
{
	int n,i,z,j,l,m;
	
	while(scanf("%d",&n)!=EOF)
	{
		for(i=1;i<=n;i++)
			scanf("%d",&a[i]);
		for(i=1;i<n;i++)	
		{
			m=0;
			if(a[i]!=a[i+1])
			{
				int k=a[i];
				a[i]=a[i+1];
				a[i+1]=k;
				l=z=0;
				for(j=1;j<n;j++)
				{
					if(a[j]>a[j+1]) z=1;
					if(a[j]<a[j+1]) l=1;
				}
				if(z==1 && l==1) {printf("%d %d\n",i,i+1); m=1;break;}	
				k=a[i];
				a[i]=a[i+1];
				a[i+1]=k;
			}
		}
		if(m==0)printf("-1\n");
	}
}
