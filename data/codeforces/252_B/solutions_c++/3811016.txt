#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
int i,j,k,l,n,m,a[120000],flag,tmp=-1;
bool check(){
	int i;
	for(i=2;i<=n;++i)if(a[i]>a[i-1])break;
	if(i==n+1)return 0;
	for(i=2;i<=n;++i)if(a[i]<a[i-1])break;
	if(i==n+1)return 0;
	return 1;
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i){
		scanf("%d",&a[i]);
		if(tmp==-1)tmp=a[i];
		else if(tmp!=a[i])flag=1;
	}
	if(!flag){
		printf("-1\n");
		return 0;
	}
	for(i=1;i<=n;++i)
		for(j=i+1;j<=n;++j)if(a[i]!=a[j]){
			swap(a[i],a[j]);
			if(check()){
				printf("%d %d\n",i,j);
				return 0;
			}
			swap(a[i],a[j]);
		}

}
