#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
#include <queue>

using namespace std;
const int MN=1e5+5;
int a[MN],b[MN],c[MN];
int main()
{
    int n;
    cin >>n;
    for(int i=0;i<n;i++) 
    {
        cin >>a[i];
        b[i]=a[i];
        c[i]=a[i];
    }
    sort(b,b+n);
    sort(c,c+n,greater<int>());
    for(int i=0;i<n-1;i++)
        if(((a[i]!=b[i+1])||(a[i+1]!=b[i]))&&((a[i]!=c[i+1])||(a[i+1]!=c[i]))&&(a[i]!=a[i+1])){cout <<i+1<<" "<<i+2<<endl;return 0;}
    cout <<"-1"<<endl;
    return 0;
}
