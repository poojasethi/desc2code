#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
using namespace std;
const int M = 100010;
int s[M],n;

bool check(){
	bool f1=1,f2=1;
	for(int i=2;i<=n;i++){
		if(s[i]>s[i-1])	f1=0;
		if(s[i]<s[i-1])	f2=0;
		if(!f1&&!f2)	return 1;
	}
	return 0;	
}
int main(){
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>s[i];
	for(int i=2;i<=n;i++){
		if(s[i]==s[i-1])	continue;
		swap(s[i],s[i-1]);
		if(check()){
			cout<<i-1<<" "<<i<<endl;
			return 0;
		}
		swap(s[i],s[i-1]);
	}
	cout<<"-1"<<endl;
	return 0;
}
