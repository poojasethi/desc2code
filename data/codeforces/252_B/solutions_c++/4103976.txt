#include<iostream>

int a[100005],n;
int is()
{
    int f1=0,f2=0;
    for(int i=2;i<=n;i++)
    {
        if(a[i-1]<a[i]) 
		f1=1;
        if(a[i-1]>a[i])    
		f2=1;
        if(f1&&f2)
        return 0;
    }
    return 1;
}
