#include<bits/stdc++.h>
using namespace std;
#define all(v)  ((v).begin()), ((v).end())
int main ()
{
int n,i,f=0;
cin>>n;
vector<int> a(n),s1,s2;
for(int i=0;i<n;i++) scanf("%d",&a[i]);
s1=a;
sort(all(s1));
s2 = s1;
reverse(all(s2));
for(i=0;i<n-1;i++)
    if(a[i]!=a[i+1])
    {
    swap(a[i],a[i+1]);
    if(a!=s1&&a!=s2)  { f=1; break; }
    swap(a[i],a[i+1]);
    }
if(f) cout<<i+1<<" "<<i+2;
else cout<<-1;
}
