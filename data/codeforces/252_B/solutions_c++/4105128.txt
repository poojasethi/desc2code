#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
using namespace std;
int a[100005];
int main()
{
	int n,m,i,j,t,s,flag1,flag2;
	while(scanf("%d",&n)!=EOF)
	{
		for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
		if(n==1||n==2) printf("-1\n"); 
		else 
		{
			for(i=1;i<n;i++)
			{
				flag1=flag2=0;
				if(a[i]!=a[i+1])
				{
					swap(a[i],a[i+1]);
					for(j=1;j<n;j++)
					{
						if(a[j]>a[j+1]) flag1=1;
						if(a[j]<a[j+1]) flag2=1;
					}
					if(flag1+flag2==2)  {printf("%d %d\n",i,i+1);break;}
				}
				swap(a[i],a[i+1]);
			}
			if(flag1+flag2<2) printf("-1\n");
		}
	}
}
