#include <iostream>
#include<cstdio>
using namespace std;
int a[100005];
int main()
{
    int n;
    int i,j;
    while(cin>>n){
    for(i=0;i<n;i++)
        cin>>a[i];
    for(i=1;i<n;i++)
    {
        if(a[i]!=a[i-1])
        {
            swap(a[i],a[i-1]);
            int f1=0,f2=0;
            for(j=1;j<n;j++)
            {
                if(a[j]>a[j-1])f1=1;
                if(a[j]<a[j-1])f2=1;
                if(f1&&f2)
                {
                    cout<<i<<" "<<i+1<<endl;
                    goto ed;
                }
            }
            swap(a[i],a[i-1]);
        }
    }
    cout<<"-1"<<endl;
    ed:;
    }
    return 0;
}
