#include<cstdio>
#include<algorithm>
using namespace std;
const int M1 = 110000;

int N;
int arr[M1];
int main()
{
    scanf("%d",&N);
	for(int i=0;i!=N;++i) scanf("%d",arr+i);
	for(int i=1;i!=N;++i)
	{   if(arr[i]!=arr[i-1])
	    {   swap(arr[i],arr[i-1]);
		    bool ok1=true,ok2=true;
		    for(int j=1;j!=N;++j)
			{   if(arr[j]<arr[j-1]) ok1 = false;
			    if(arr[j]>arr[j-1]) ok2 = false;
				if(!ok1&&!ok2) break;
			}
			if(!ok1&&!ok2)
			{   printf("%d %d\n",i,i+1);
			    return 0;
			}
			swap(arr[i],arr[i-1]);
		}
	}
	printf("%d\n",-1);
	return 0;
}
