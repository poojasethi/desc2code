#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
using namespace std;
int a[100005];
int main()
{
	int i,j,k,t,n,m,flag1,flag2;
	while(scanf("%d",&n)!=EOF)
	{
		for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
		for(i=1;i<n;i++)
		{
			if(a[i]!=a[i+1])
			{
				swap(a[i],a[i+1]);
				flag1=0,flag2=0;
				for(j=1;j<=n-1;j++)
				{
					if(a[j]>a[j+1])flag1=1;
					if(a[j]<a[j+1])flag2=1;
				}
				if(flag1+flag2==2)
				{
					printf("%d %d\n",i,i+1);break;
				}
				swap(a[i],a[i+1]);
			}
		}
		if(flag1+flag2!=2)printf("-1\n");
	}
}
