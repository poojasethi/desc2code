#include<stdio.h>
#include<algorithm>
using namespace std;
#define N 100010

int buf[N],n;

void solve()
{
	int i,j;
	bool legal1,legal2;
	for(i=0;i<n-1;i++){
		if(buf[i]!=buf[i+1]){
			swap(buf[i],buf[i+1]);
			for(j=0,legal1=false,legal2=false;j<n-1;j++){
				if(buf[j]<buf[j+1]) legal1=true;
				if(buf[j]>buf[j+1]) legal2=true;
				if(legal1 && legal2){
					printf("%d %d\n",i+1,i+2);
					return ;
				}
			}
			swap(buf[i],buf[i+1]);
		}
	}
	printf("-1\n");
}
int main()
{
	int i;
	scanf("%d",&n);
	for(i=0;i<n;i++) scanf("%d",&buf[i]);
	solve();
	return 0;
}
