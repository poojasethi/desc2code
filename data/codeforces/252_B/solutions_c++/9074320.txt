#include <bits\stdc++.h>


using namespace std;


int main(void)
{
 int n;
 cin >> n;
    vector<int>a(n + 1, 0);
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
    }
    for (int i = 2; i <= n; ++i) {
        if (a[i] != a[i - 1]) {
            swap(a[i], a[i - 1]);
            bool asc(0), desc(0);
            for (int i = 1; i + 1 <= n; ++i) {
                if (a[i] < a[i + 1]) asc = 1;
                if (a[i] > a[i + 1]) desc = 1;
            }
            if (asc && desc) return cout << i << " " << i - 1 << endl, 0;
            swap(a[i], a[i - 1]);
        }
    }
    return cout << -1 << endl, 0;
}
