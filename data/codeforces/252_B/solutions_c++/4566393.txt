#include <bits/stdc++.h>
using namespace std;

vector<int> a;
int main() {
  ios_base::sync_with_stdio(0);
  int n;
  cin>>n;
  a.resize(n);
  for(int i=0; i<n; i++) cin>>a[i];
  vector<int> b(a);
  sort(b.begin(),b.end());
  vector<int> c(b);
  reverse(c.begin(),c.end());
  if(b[0]==b[n-1]) goto print;
  for(int i=0; i<n; i++) {
    for(int j=i+1; j<n; j++) {
      if(a[i]==a[j]) continue;
      swap(a[i],a[j]);
      if(a!=b && a!=c) {
	cout<<i+1<<" "<<j+1<<endl;
	return 0;
      }
      swap(a[i],a[j]);
    }
  }
 print:
  cout<<"-1"<<endl;
  return 0;
}
