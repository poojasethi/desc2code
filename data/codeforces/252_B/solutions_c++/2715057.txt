#include<cstdio>

int f[111111],n,i;
main(){
 //freopen("input.txt","r",stdin);

scanf("%d",&n);
for(i=1;i<=n;i++)scanf("%d",&f[i]);
 for(i=2;i<=n;i++)if(f[i] != f[i-1]){
  swap(f[i],f[i-1]);
  bool ok1 = true,ok2 = true;
  for(int j=2;j<=n;j++){
   if(f[j-1]>f[j])ok1 = false;
   if(f[j-1]<f[j])ok2 = false;
  if(ok1 || ok2)continue;
  printf("%d %d",i-1,i);
  return 0;
  }
  swap(f[i],f[i-1]);
 }
puts("-1");

}
