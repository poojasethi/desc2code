#include <stdio.h>
#include <algorithm>
#define for_up(i,n) for(int i=0;i<n;i++)
int n,i,m[100400]={0};
bool answer(){
  bool flag1=true,flag2=true;
  for_up(i,n-1){
    if (m[i]>m[i+1])flag1=false;
    if (m[i]<m[i+1])flag2=false;
  }
  if (flag1 || flag2)
     return false;
  else 
     return true;
}
