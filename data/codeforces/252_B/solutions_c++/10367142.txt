#include<cstdio>
#include<algorithm>
using namespace std;
int a[100005],n,i,j;
int main()
{  
   scanf("%d",&n);
   for(i=0;i<n;i++)
      scanf("%d",&a[i]);
   for(i=0;i<n-1;i++)
     if(a[i]!=a[i+1])
     {
        swap(a[i],a[i+1]);
        int ok1=1,ok2=1;
        for(j=0;j<n-1;j++)
        {
          if(a[j]>a[j+1])
              ok1=0;
          if(a[j]<a[j+1])
              ok2=0;
          if(ok1||ok2)continue;
          printf("%d %d\n",i+1,i+2);
          return 0;
        }
        swap(a[i],a[i+1]);
     }
    puts("-1");
  return 0;
}
