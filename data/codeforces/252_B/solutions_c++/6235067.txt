#include <cstdio>

#define N 100000

int a[N], n, i, j, z;
bool f1, f2;

int main()
{
	scanf("%d", &n);

	for (i = 0; i < n; i++)
		scanf("%d", &a[i]);

	for (i = 0; i < n - 1; i++)
		if (a[i] != a[i + 1])
		{
			z = a[i], a[i] = a[i + 1], a[i + 1] = z;
			f1 = 0, f2 = 0;
			for (j = 0; j < n - 1; j++)
				f1 |= a[j] < a[j + 1], f2 |= a[j] > a[j + 1];
			if (f1 & f2)
			{
				printf("%d %d\n", i + 1, i + 2);
				return 0;
			}
			z = a[i], a[i] = a[i + 1], a[i + 1] = z;
		}
	puts("-1");
	return 0;
}
