#include<iostream>
#include<cstdio>
using namespace std;
int data[100006];
int main()
{
  int n;
  bool d1,d2;
  cin>>n;
  for(int i=0;i<n;i++)cin>>data[i];
  for(int i=0;i<n-1;i++)if(data[i]!=data[i+1])
  {
    swap(data[i],data[i+1]);
    d1=true;
    d2=true;
    for(int j=0;j<n-1;j++)
    {
      if(data[j]>data[j+1])d1=false;
      if(data[j]<data[j+1])d2=false;
     if(!d1&&!d2)
      {
        cout<<i+1<<" "<<i+2<<endl;
        return 0;
      }

    
    }
    swap(data[i],data[i+1]);
  }
  cout<<-1<<endl;
  return 0;
}
