#include "iostream"
using namespace std;

int a[100000];
int main()
{
    int n;
    cin>>n;
    for (int i = 0; i < n; i++) cin>>a[i];
    
    for (int i = 0; i+1 < n; i++) {
        if (a[i] == a[i+1]) continue;
        bool x = false, y = false;
        swap(a[i], a[i+1]);
        for (int j = 0; j + 1 < n; j++)
        {
            if (a[j] > a[j+1]) x = true;
            if (a[j] < a[j+1]) y = true;
            if (x && y) break;
        }
        if (x && y) {
             cout<<i+1<<" "<<i+2<<endl;
             return 0;
        }
        swap(a[i], a[i+1]);
    }
    cout<<-1<<endl;
    return 0;
}
