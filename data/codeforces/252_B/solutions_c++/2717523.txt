#include<stdio.h>
#include<algorithm>
#include<stdlib.h>
#include<iostream>
using namespace std;
int main ()
{
    int n;
    int a[100000];
    int sortorg[100000] ,sortinv[100000];
    cin>>n;
    for (int i=0;i<n;i++)
    {
        cin>>a[i];
        sortorg[i]=a[i],sortinv[i]=a[i];
    }
    sort(sortorg,sortorg+n);
    sort(sortinv,sortinv+n,greater<int>());
    for (int i=1;i<n;i++)
    {
        if (!((a[i]==sortinv[i-1]&&a[i-1]==sortinv[i])||(a[i]==sortorg[i-1]&&a[i-1]==sortorg[i]))&&a[i]!=a[i-1])
        {
            printf("%d %d\n",i,i+1);
            return 0;
        }       
    }
    printf("-1\n");
}
