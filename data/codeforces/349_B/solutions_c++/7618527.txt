#include <cstdio>
int A[10], n, pos = 1;
int main()
{
	scanf("%d", &n);
	for (int i = 1; i < 10; i++)
	{
		scanf("%d", &A[i]);
		if (A[i] <= A[pos])
			pos = i;
	}
	int l = n / A[pos];
	if (l)
	{
		for (int i = l; i >= 0; i--)
			for (int j = 9; j > 0; j--)
				if (A[j] <= n - i * A[pos])
				{
					printf("%d", j);
					n -= A[j];
					break;
				}
	}
	else
		printf("-1");
   return 0;
}
