#include<iostream>
#include<algorithm>
using namespace std;

int a[12], l, s=1000000000, i;

int main()
{
	int v;
	cin >> v;
	for (i=1; i<=9; ++i)
	{
		cin >> a[i];
		s = min(s, a[i]);
	}
	l=v/s;
	if (l)
	  {for (i=9; i>=1; --i)
	  while ((v-a[i])/s==l-1 && l && v>=a[i])
	  {
	  	cout << i;
	  	l--;
	  	v -= a[i];
	  }
   }
   else cout << -1;
   return 0;
}
