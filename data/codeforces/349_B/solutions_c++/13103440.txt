#include<bits/stdc++.h>
using namespace std;
int main()
{
	int v;
	cin>>v;
	int arr[9];
	int r=100000000;
	for(int i=0;i<9;i++)
	{
		scanf("%d",&arr[i]);
		r=min(r,arr[i]);
	}	
	int m=v/r;
	if(m==0)
	{
		cout<<"-1";
	}
	else
	{
		while(m--)
		{
			for(int i=9;i>0;i--)
			{
				if((v-arr[i-1])/r ==m and arr[i-1]<=v)
				{
					cout<<i;
					v-=arr[i-1];
				}
			}
		}
	}
	return 0;	
}
