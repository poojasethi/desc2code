#include <stdio.h>
int v, mn = 1;
int a[10];
int main(){
	scanf("%d",&v);
	for(int i = 1; i <= 9; i ++){
		scanf("%d", a+i);
		if(a[i] < a[mn]) mn = i;
	}
	if(a[mn] > v){
		puts("-1");
		return 0;
	}
	int n = v / a[mn];
	while(n --){
		for(int i = 9; i ; i --){
			if(n * a[mn] + a[i] <= v){
				putchar(i + '0');
				v -= a[i];
				break;
			}
		}
	}
	return 0;
}
