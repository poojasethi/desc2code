#include <iostream>
using namespace std;
int main() {
	int v;cin>>v;
	int as[9];
	int low=1e9;
	for(int i=0; i<9; ++i) cin>>as[i], low=min(low,as[i]);
	if (v<low) {
		cout<<"-1\n";
		return 0;
	}
	string res;
	while(v>=low) {
		for(int i=8; i>=0; --i)
			if (as[i]<=v && (v-as[i])/low+1 == v/low) {
				res+='1'+i;
				v -= as[i];
				break;
			}
	}
	cout<<res<<'\n';
}
