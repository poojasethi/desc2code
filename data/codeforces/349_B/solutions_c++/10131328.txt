#include <iostream>

using namespace std;

int main()
{
	int n;
	cin>>n;
	int num,min=0,a[10]={100001};
	for(int i=1;i<=9;i++){
		cin>>a[i];
		if(a[i]<a[min]){
			min=i;
		}
	}
	num=n/a[min];
	if(n<a[min])
		cout<<"-1"<<endl;
	else{
		while(num--){
			for(int i=9;i>0;i--){
				if(((n-a[i])/a[min]==num)&&(n>=a[i])) {
					cout<<i;
					n-=a[i];
					break;
				}
			}
		}
	}
	return 0;
}
