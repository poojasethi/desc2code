#include <iostream>
#include <string>
using namespace std;
int v,a[20],m;
int main(){
 	cin>>v;
 	a[0]=1000000;
 	for(int i=1;i<10;i++) {
 		cin>>a[i];
   		if(a[i]<=a[m]) m=i;
   	}
   	if(v<a[m]) {
   		cout<<-1; 
   		return 0; 
   	}
 	int l=v/a[m];
 	while(l) {
 		int mm;
 		for(int i=m;i<10;i++) {
 			if((l-1)*a[m]+a[i]<=v) mm=i;
	 	}	
	 	cout<<mm;
	 	l--;
	 	v-=a[mm];
 	}
 	return 0;     
}
