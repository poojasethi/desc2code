#include<iostream>
using namespace std;

int n,m,a[10],i;
int main(){
	cin>>m;
	int min=1000000;
	for (i=1;i<=9;i++){
		cin>>a[i];
		min=min<a[i]?min:a[i];
	}
	if (m<min){
		cout<<-1<<endl;
		return 0;
	}
	n=m/min;
	while (n--){
		for (i=9;i>=1;i--)
			if (((m-a[i])/min>=n) && (m-a[i]>=0)) break;
		m-=a[i];
		cout<<i;
	}
	cout<<endl;
	return 0;
}
