#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int f[9];
int main(){
	int n,i;
	while(scanf("%d",&n)!=EOF){
		int s=1e8;
		for(i=1;i<10;i++) scanf("%d",&f[i]),s=min(s,f[i]);
		if(!(n/s)) printf("-1\n");
		else {
			int m=n/s;
			while(m--){
				for(i=9;i>=1;i--) if((n-f[i])/s==m&&n>=f[i]) break;
				printf("%d",i); 
				n-=f[i];
			}
			printf("\n");
		}
	}
	return 0;
}
