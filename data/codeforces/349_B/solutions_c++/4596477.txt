#include <cstdio>
int v,a[9];
int main(){
	int i, j,min=0,n;
	scanf("%d", &v);
	for(i = 0; i < 9; i++){
		scanf("%d", &a[i]);
		if(a[i] <= a[min])min = i;
	}
	n = v / a[min];
	if(n == 0) printf("-1");else{
		v %= a[min];
		for(i = 0; i < n; i++){
			for(j = 8; j > min; j--)
				if(a[j] - a[min] <= v){v -= a[j] - a[min];break;}
			putchar('1' + j);
		}
	}
	printf("\n");
	return 0;
}
