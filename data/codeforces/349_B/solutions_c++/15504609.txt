#include<stdio.h>
#include<string.h>
int n,i,j,min=101000;
int a[101000];
int main(){
	scanf("%d",&n);
	for(i=1;i<=9;i++){
		scanf("%d",&a[i]);
		if(min>a[i]){
			min=a[i];
		}
	}
	if(n<min){
		printf("-1\n");
		return 0;
	}
	else{
		for(i=(n/min);i>0;i--){
			for(j=9;j>0;j--){
				if(n-a[j]>=0&&(n-a[j])/min>=i-1){
					printf("%d",j);
					n=n-a[j];
					break;
				}
			}
		}
	}
	return 0;
}
