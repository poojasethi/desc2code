#include<bits/stdc++.h>
using namespace std;
int k=1000000,a[12],i,v,l;
main()
{
    cin>>v;
    for (i=1;i<=9;i++)
        cin>>a[i],k=min(k,a[i]);
    l=v/k;
    if (!l) cout<<-1;
    while (l--)
        for (i=9;i>=1;i--)
            if ((v-a[i])/k==l && v>=a[i])
            {
                cout<<i;
                v-=a[i];
                break;
            }
}
