#include<stdio.h>
int main(){
	int a[10],j,l,x,y,i,n,m,v;
	while(~scanf("%d",&v)){
		for(x=1000001,i=1;i<10;i++){
			scanf("%d",&a[i]);
			if(a[i]<x||(a[i]==x&&i>y)) x=a[i],y=i;
		}
		if(v<x){
			printf("-1\n");
			continue;
		}
		for(n=v/x,v=v%x,i=0;i<n;i++){
			for(j=9;j>y;j--){
				if(a[j]-a[y]<=v){
					v-=a[j]-a[y];
					break;
				}
			}
			printf("%d",j);
		}
		printf("\n");
	}
	return 0;
}
