#include <iostream>

using namespace std;

int n,a[10],s=10000000,l;

int main(void)
{
	cin>>n;
	for(int i=1;i<10;i++)
	{
		cin>>a[i];
		s = min(s,a[i]);
	}
	l = n / s;

	if(!l) cout<<-1<<endl;

	while(l--)
	{
		for(int i=9;i>0;i--)
		{
			if((n-a[i])/s==l && n>=a[i])
			{
				cout<<i;
				n -= a[i];
				break;
			}
		}
	}
	return 0;

}
