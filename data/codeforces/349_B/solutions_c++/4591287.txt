#include <iostream>
#include <stdio.h>
#include <algorithm>

using namespace std;

int main(){
	int v,ma=100001,m;
	scanf("%d",&v);
	int ta[9];
	for(int i=0;i<9;i++){
		scanf("%d",&ta[i]);
		if(ta[i]<ma) ma=ta[i];
	}
	if(v<ma) cout << -1 << endl;
	m = v/ma;
	v%=ma;
	for(int i=0;i<m;i++){
		for(int j=9;j>=1;j--){
			if(v+ma-ta[j-1]>=0){
				v = v+ma-ta[j-1];
				cout << j;
				break;
			}
		}
	}
	cout << endl;
	return 0;
}
