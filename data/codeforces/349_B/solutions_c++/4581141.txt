#include <iostream>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int b[10];
	for (int i=0; i<9; i++) cin >> b[i+1];
	int p=9;
	for (int i=8; i>0; i--)
		if (b[i]<b[p]) p=i;
	int o=n / b[p],last=n % b[p];
	if (o==0) cout << -1;
	else 
		for (int i=0; i<o; i++)
			for (int j=9; j>=p; j--)
				if (last>=b[j]-b[p])
				{
					cout << j;
					last-=(b[j]-b[p]);
					break;
				}
	return 0;
}
