# include <bits/stdc++.h>
using namespace std;
# define fi cin
# define fo cout
int s[105];
int main(void)
{
    int n,m = INT_MAX;
    fi>>n;
    for (int i = 1;i <= 9;++i) fi>>s[i],m = min(m,s[i]);
    int p = n / m;
    if (!p) return fo << "-1\n",0;
    for (int i = 1;i <= p;++i)
    {
        for (int j = 9;j;--j) if (n - s[j] >= (p - i) * m) {fo << j;n -= s[j];break;}
    }
    return 0;
}
