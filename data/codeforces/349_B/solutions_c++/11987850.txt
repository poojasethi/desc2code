#include<bits/stdc++.h>
using namespace std;
int v,i,k=1e9;
int a[12];
main()
{
    cin>>v;
    for (i=1;i<=9;i++)
        cin>>a[i],k=min(k,a[i]);
    if (v<k) return cout<<-1,0;
    int l=v/k;
    while (l--)
    {
        for (i=9;i>=1;i--)
            if ((v-a[i])/k==l&&v>=a[i])
                printf("%d",i),v-=a[i];
    }
}
