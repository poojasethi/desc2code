p=input()
a=map(int,raw_input().split())
m=a[0]
ind=0
for i in xrange(len(a)):
     if a[i]<=m:
         m=a[i]
         ind=i
cnt=[0 for _ in range(9)]
if p<m:
    print -1
else:
    cnt[ind]=p/a[ind]
    r=p%a[ind]
    t=8
    while r and t>ind:
        d=a[t]-a[ind]
        rep=r/d
        if rep>0:
            cnt[t]=rep
            cnt[ind]-=rep
            r-=(r/d)*d
        t-=1
    s=''
    for i in range(9):
        s+=cnt[i]*str(i+1)    
    print s[::-1]
