#!/usr/bin/env python2
# -*- coding : utf-8 -*-

def main():
    v = int(raw_input())
    a = map(int, raw_input().split())
    cnt = [0] * 10
    length = -1

    for i, x in enumerate(a):
        if v / x > length or (v / x == length and x == a[j - 1]):
            length, j = v / x, i + 1

    if length:
        res = v - length * a[j - 1]
        for i in xrange(9, j, -1):
            if res / (a[i - 1] - a[j - 1]) > 0:
                cnt[i] = res / (a[i - 1] - a[j - 1])
                res -= cnt[i] * (a[i - 1] - a[j - 1])

        cnt[j] = length - sum(cnt)
        print ''.join([x * str(i) for i, x in enumerate(cnt)])[::-1]
    else:
        print -1

if __name__ == '__main__':
    main()
