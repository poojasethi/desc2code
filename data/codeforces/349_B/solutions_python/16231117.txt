v=input()
a=map(int,raw_input().split())
x,y=-1,int(1e6)
for idx,item in enumerate(a):
    if item <= y:
        x,y=idx,item
if v<y:print -1;exit(0)
tmp=str(x+1)*(v/y)
left=v-y*(v/y)
i,j=0,9
while left>0 and j>x+1:
    if left+y>=a[j-1]:
        tmp=tmp[:i]+str(j)+tmp[i+1:]
        left-=a[j-1]-y
        i+=1
    else: j-=1
print tmp
