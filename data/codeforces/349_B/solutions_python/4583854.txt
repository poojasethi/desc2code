

mybudge = int(raw_input())

mycosts = map(int, raw_input().split())

mymin = 1000001
myplace = 0
cando = False

for i in mycosts:
    if i <= mybudge:
        cando = True

for i in range(len(mycosts)):
    if mycosts[i] <= mymin:
        myplace = i + 1
        mymin = mycosts[i]


out = []
while mybudge >= mymin:
    mybudge -= mymin
    out.append(myplace)

at = 0

for i in range(9, myplace, -1):
    while mycosts[i - 1] - mymin <= mybudge:
        mybudge -= mycosts[i - 1] - mymin
        out[at] = i
        at +=1

if cando == True:
    print ''.join(map(str, out))
else:
    print -1
