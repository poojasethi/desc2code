n = int(raw_input())
a = map(int,raw_input().split())
b = []
for i in range(9):
	b.append([a[i],9-i])
b.sort()
c = []
while n >= b[0][0]:
	n -= b[0][0]
	c.append(10 - b[0][1])
for j in range(len(c)):
	flag = 0
	for i in range(9):
		if b[i][0] - a[c[j]-1] <= n and 10 - b[i][1] > c[j]:
			n -= b[i][0] - a[c[j]-1]
			c[j] = 10 - b[i][1]
			flag = 1
	if flag == 0:
		break
s = ""
for i in c:
	s = s + (str(i))
if s == "":
	s = "-1"
print s
