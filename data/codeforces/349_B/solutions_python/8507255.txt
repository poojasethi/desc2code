tot=input()
arr=map(int,raw_input().split())
l=[]
if min(arr)>tot:
	print -1
	exit(0)
for i in range(len(arr)):
	l.append([arr[i],i+1])
l.sort()
amount=l[0][0]
val=l[0][1]
for i in l:
	if i[0]==amount and i[1]>val:
		val=i[1]
s=""
total=tot/amount
left=tot%amount
for i in range(8,-1,-1):
	if arr[i]-amount<=left and i+1>val and (arr[i]-amount):
		for j in range(left/(arr[i]-amount)):
			s+=str(i+1)
			left-=(arr[i]-amount)
			total-=1
for i in range(total):
	s+=str(val)
print s
