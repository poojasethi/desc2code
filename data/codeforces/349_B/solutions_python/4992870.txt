V = int(input())

weights = map(int, raw_input().split())

cnts = [V / weights[i] for i in range(9)]

maxCnt = max(cnts)
maxNum = cnts.index(maxCnt) + 1
        
if maxCnt == 0 :
    print -1

else :
    m = maxNum
    for i in range(m, 9) :
        if weights[i] == weights[m-1] :
            maxNum = i + 1
            
    resultList = [maxNum for _ in range(maxCnt)]

    reminder = V - weights[maxNum-1] * maxCnt
    
    for cur in range(len(resultList)) :
        flag = False
        for i in range(8, maxNum-1, -1) :
            if weights[maxNum-1] + reminder >= weights[i] :
                reminder -= (weights[i] - weights[maxNum-1])
                resultList[cur] = i + 1
                flag = True
                break
            else :
                continue
        if flag == False :
            break
            
    print "".join(map(str, resultList))
