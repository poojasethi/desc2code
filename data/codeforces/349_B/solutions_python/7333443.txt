get_ints = lambda: map(int, raw_input().split())

v = input()
seq = get_ints()

l = min(seq)
ldigit = 9 - seq[::-1].index(l)

if v < l:
    print -1
else:
    n = v / l
    res = ""
    for ni in xrange(n, 0, -1):
        room = v - l * (ni - 1)
        if room == l:
            res += ni * str(ldigit)
            break
        for i, cost in enumerate(seq):
            if cost <= room:
                digit = i + 1
        v -= seq[digit-1]
        res += str(digit)
    print res
