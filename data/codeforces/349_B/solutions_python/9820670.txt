v = input()
a = map(int,raw_input().split())
d,p = min(enumerate(a),key=lambda x:(x[1],-x[0]))
ans = list(str(d+1)*(v/p))
if p > v:
    print -1
else:   
    r = v % p
    i = 0
    while r > 0:
        n,c = [x for x in enumerate(a) if x[1] <= (r+p)][-1]
        r -= c-p
        ans[i] = str(n+1)
        i += 1
        if i == len(ans):
            break
    print ''.join(ans)
