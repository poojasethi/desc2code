#!/usr/bin/env python

import sys


v = int(raw_input())
costs = map(int, raw_input().split())

min_a = min(costs)
length = v / min_a

if length == 0:
    print -1
    exit()

while length > 0:
    cost_left = v - length * min_a
    if cost_left == 0:
        min_digit = 9 - list(reversed(costs)).index(min_a)
        sys.stdout.write(str(min_digit) * length)
        break
    cost_left += min_a
    for d, a in enumerate(costs, start=1):
        if a <= cost_left:
            max_d = d
            max_a = a
    sys.stdout.write(str(max_d))
    if max_a == min_a:
        v -= cost_left - max_a
    v -= max_a
    length -= 1
