import sys
v = int(raw_input())
a = map(int, raw_input().split())

smallesta = min(a)

if v < smallesta:
    print -1
else:
    equalSmallest = [i for i in range(9) if a[i] == smallesta]
    numberToBeRepeated = equalSmallest[-1]+1 #get the largest number
    numberOfTimesToRepeat = v/smallesta
    if v%smallesta ==0:
        print str(numberToBeRepeated)*numberOfTimesToRepeat
    else: 
        tmpSolution=[numberToBeRepeated]*numberOfTimesToRepeat #list of number that has repeated values throughout
        moneyUsed = smallesta*numberOfTimesToRepeat
        for i in range(numberOfTimesToRepeat):
            for j in range(8,numberToBeRepeated-1,-1): #try to replace
                if moneyUsed-a[tmpSolution[i]-1]+a[j] <= v:
                    moneyUsed += (a[j]-a[tmpSolution[i]-1])
                    tmpSolution[i] = j+1
                    break
            if moneyUsed == v:
                break
        for i in range(numberOfTimesToRepeat):
            sys.stdout.write(str(tmpSolution[i]))
