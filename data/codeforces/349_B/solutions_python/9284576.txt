v=int(input())
ar=map(int,raw_input().split())
mi=10**9+7
mid=-1
for i in range(0,9):
    if ar[i]<=mi:
        mi=ar[i]
        mid=i+1
dig=v//mi
mod=v%mi
if dig==0:
    print "-1"
else:
    ans=""
    for j in range(0,dig):
        got=-1
        for i in range(mid,9):
            if mi+mod>=ar[i]:
                got=i
        if got==-1:
            break
        else:
            ans+=str(got+1)
            mod-=(ar[got]-mi)
    ans+=((dig-len(ans))*str(mid))
    print ans
