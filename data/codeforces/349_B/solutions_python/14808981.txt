v = int(raw_input())
a = map(int, raw_input().split())[::-1]
m = min(a)
x = 9 - a.index(m)
d, r = divmod(v, m)
if d < 1:
    print -1 
else:
    c = d * [x]
    i = 0
    while i < d:
        t = m + r
        j = 0
        while a[j] > t: j += 1
        c[i] = 9 - j
        i += 1
        r = t - a[j]
    print "".join(map(str, c))
