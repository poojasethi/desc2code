I = lambda:map(int, raw_input().split())

value = int(raw_input())
L = map(int,raw_input().split())
cost = min(L)
n = value/cost
m=0
ans=""
if n<=0:
    ans=str(-1)
for i in range(9,0,-1):
    if L[i-1]==cost:
        m=i
        break
while n>0:
    if n*cost == value:
        ans = ans + str(m)*n
        break
    n-=1
    for i in range(9,0,-1):
        if value - L[i-1] >= cost*n:
            value -=L[i-1]
            ans = ans + str(i)
print ans
