#miaomiao = []

v = int(raw_input())

miao = raw_input()
amiao = map(int, miao.split(' '))

#amiaomiao = list(enumerate(amiao))
amiaomiao = []
for i in range(len(amiao)):
    amiaomiao.append((amiao[i], i + 1))

#print amiaomiao

amiaomiao = sorted(amiaomiao, cmp=lambda x, y: y[1] - x[1] if x[0] == y[0] else x[0] - y[0])

meow = [amiaomiao[0]]

for i in range(1, len(amiaomiao)):
    if amiaomiao[i][0] != amiaomiao[i - 1][0]:
        meow.append(amiaomiao[i])


def main(v, moew):
    a, moe = meow[0]

    if v < a:
        return "-1"
    elif v % a == 0:
        return str(moe) * (v / a)
    else:
        n = v / a
        ans = [moe] * n
        v %= a
        k, b = 0, 0
        for x in range(n):

            feasible = sorted(filter(lambda x: x[0] - a <= v, meow[1:]),
                              key=lambda x: x[1], reverse=True)
            if not feasible:
                break
            else:
                k, b = feasible[0]
                if b > moe:
                    ans[x] = b
                    v -= (k - a)
        return ''.join(map(str, ans))


print main(v, meow)
