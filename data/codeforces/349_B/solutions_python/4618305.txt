v = int(input())
a = list(map(int,raw_input().split()))
j = 0
for i in range(9):
    if a[i]<=a[j]: 
        j = i
n = v / a[j]
s = v - n * a[j]
out = [j+1] * n
for k in range(len(out)):
    i = 8
    while i>j:
        if a[i] - a[j] <= s:
            out[k] = i+1
            s -= a[i] - a[j]
            break
        i = i - 1

if len(out)>0:
    print(''.join(map(str,out)))
else:
    print(-1)
