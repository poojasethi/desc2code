#include<iostream>
#include<cmath>
using namespace std;

int k, a[101][101], n;

int main() {
	int i, k,kk, l, j;
	
	cin >> kk;
	
	a[1][2] = a[2][1] = 1;
	
	for(i = 3; i<=100; ++i) {
		for(j = i - 1; j >= 2 && (j * (j - 1)/2) > kk; --j);
		
		kk-= (j * (j - 1)/2);
		
		for(k = 1; k<=j; ++k)
			a[i][k] = a[k][i] = 1;
		
		if(!kk)
			break;
	}
	
	cout << i << "\n";
	
	for(int j = 1; j<=i; ++j) {
		for(int k = 1; k<=i; ++k)
			cout << a[j][k];
		cout << "\n";
	}
	
	return 0;
}
