#include <cstdio>
#include <algorithm>

using namespace std;

int a[105][105];

int main( void ){
  int k;
  scanf( "%d", &k );
  for( int i = 0; i < 100; ++i ){
    for( int j = i-1; j >= 0; --j )
    {
      int pp = 0;
      for( int k = 0; k < 100; ++k )
	if( a[i][k] && a[k][j] ) ++pp;

      if( k >= pp ){
	a[i][j] = a[j][i] = 1;
	k -= pp;
      }
    }
  }

  printf( "%d\n", 100 );
  for( int i = 0; i < 100; ++i, puts( "" ) )
    for( int j = 0; j < 100; ++j )
      printf( "%d", a[i][j] );

  return 0;
}
