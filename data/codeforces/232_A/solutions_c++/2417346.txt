#include <stdio.h>
int k;
int s[101][101];
int i,j;
int x,y;
int main(void)
{
	scanf("%d",&k);
	s[1][2]=1;
	s[2][1]=1;
	x=3;
	y=1;
	while(k>0)
	{
		if(y==x)
		{
			y=1;
			x++;
		}
		if(y-1>k)
		{
			y=1;
			x++;
		}
		k=k-(y-1);
		s[x][y]=1;
		s[y][x]=1;
		y++;
	}
	printf("%d\n",x);
	for(i=1;i<=x;i++)
	{
		for(j=1;j<=x;j++)
		{
			printf("%d",s[i][j]);
		}
		printf("\n");
	}
	return 0;
}
