#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int k; scanf("%d",&k);

	bool A[100][100]={};
	int cnt=0;
	rep(u,100) rep(v,u) if(!A[u][v]) { // 辺 (u,v) を追加する
		int tmp=0;
		rep(w,u) if(A[u][w] && A[v][w]) tmp++;
		if(cnt+tmp<=k){
			A[u][v]=A[v][u]=true;
			cnt+=tmp;
		}
	}

	puts("100");
	rep(u,100){ rep(v,100) printf("%d",A[u][v]?1:0); puts(""); }

	return 0;
}
