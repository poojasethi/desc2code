#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int a[105][105];
int main()
{
	int k;
	scanf("%d",&k);
	a[0][1] = a[1][0] = 1;
	int j,p;
	for(j=2;j<100;j++)
	{
		for(p=2; (p*(p-1))/2 <= k && p <=j ; p++);
			p--;
		k-= (p*(p-1))/2;
		int l;
		for(l=0;l<p;l++)
		{
			a[l][j]  = a[j][l] = 1;
		}
		if(k==0)break;
	}
	printf("%d\n",j+1);
	int i;
	for(i=0;i<j+1;i++){
		for(k=0;k<j+1;k++)
			printf("%d",a[i][k]);
		printf("\n");
}
		return 0;
}
