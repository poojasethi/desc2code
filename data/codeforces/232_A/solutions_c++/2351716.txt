#include<cstdio>
#include<algorithm>
#define N 110
using namespace std;

int n = 1, k, tbl[N][N];

int main(){
	scanf("%d", &k);
	while(k > 0){
		int m;
		for(m = n; m*(m-1)/2 > k; m--);
		for(int i = 0; i < m; i++) tbl[n][i] = tbl[i][n] = 1;
		k -= m*(m-1)/2;
		n++;
	}
	printf("%d\n", n);
	for(int i = 0; i < n; i++){
		for(int j = 0; j < n; j++){
			printf("%d", tbl[i][j]);
		}
		puts("");
	}
}
