#include <cstdio>

int mat[100][100];

int main()
{
	int k;
	scanf("%d", &k);
	
	int n = 3;
	mat[0][1] = mat[1][0] = 1;
	while (1)
	{
		mat[n-2][n-1] = mat[n-1][n-2] = 1;
		for (int i = 0; i < n-2; ++i)
		{
			if (k-i-1 < 0) break;
			mat[n-1][i] = mat[i][n-1] = 1;
			k -= i+1;
		}
		if (k) ++n;
		else break;
	}
	
	printf("%d\n", n);
	for (int i = 0; i < n; ++i)
	{
		for (int j = 0; j < n; ++j)
			printf("%d", mat[i][j]);
		printf("\n");
	}
	
	return 0;
}
