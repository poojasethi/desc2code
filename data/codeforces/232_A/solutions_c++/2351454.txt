#include<iostream>
#include<stdio.h>
using namespace std;
int A,n,l,m,i,j,k,v[101][101];
main()
{
 cin>>k;
 v[1][2]=1;
 v[2][1]=1;
 v[1][3]=1;
 v[3][1]=1;
 v[3][2]=1;
 v[2][3]=1;
 k--;
 n=3;
 while  (k)
 {
     A=n+1;
    m=0;
    while (m*(m-1)/2<k && m<n) m++;
    if (m*(m-1)/2>k) m--;
    for (i=1;i<=m;i++)
    {
        v[i][A]=1;
        v[A][i]=1;
    }
    k-=m*(m-1)/2;
    n++;
    }
cout<<n<<endl;
for (i=1;i<=n;i++)
{
    for (j=1;j<=n;j++)
    cout<<v[i][j];cout<<endl;

 }
