#include<cstdio>
int n,m,a,st;
bool b[105][105];
void work(int m)
{
	for (a=n;a;a--) if (a*(a-1)/2<=m) break; m-=a*(a-1)/2;
	for (int i=0;i<a;i++) b[i][st]=b[st][i]=1; st++;
	if (m) work(m);
}
int main()
{
	scanf("%d",&m); st=0;
	for (n=100;n;n--) if (n*(n-1)*(n-2)/6<=m) break;
	m-=n*(n-1)*(n-2)/6; st=n;
	for (int i=0;i<n;i++)
		for (int j=i+1;j<n;j++) b[i][j]=b[j][i]=1;
	work(m); n=st; printf("%d\n",n);
	for (int i=0;i<n;i++){
		for (int j=0;j<n;j++) printf("%d",b[i][j]);
		printf("\n");
		}
	return 0;
}
