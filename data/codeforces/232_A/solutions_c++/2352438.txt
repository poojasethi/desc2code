#include <cstdio>

using namespace std;

int a[101][101];

int main(void)
{
  int s;
  scanf("%d",&s);
  int n=1;
  while(s>0)
  {
     int m=n;
     while(m*(m-1)/2>s)
       --m;
     s-=m*(m-1)/2;
     for(int i=1;i<=m;++i)
     {
       a[i][n+1]=1;
       a[n+1][i]=1;
     }
     ++n;
  }
  printf("%d\n",n);
  for(int i=1;i<=n;++i)
  {
    for(int j=1;j<=n;++j)
      printf("%d",a[i][j]);
    printf("\n");
  }
  return 0;
}
