#include<cstdio>
int n,k,i,j,now;
int a[110][110];
int main()
{
    scanf("%d",&k);
    n=0;while(n*(n-1)*(n-2)/6<=k)n++;
    n--;now=n*(n-1)*(n-2)/6;
    for(i=1;i<=n;i++)for(j=1;j<=n;j++)if(i!=j)a[i][j]=1;
    while(now!=k)
    {
        n++;
        for(i=1;i<n;i++)
        {
            if(now+i-1>k)break;
            now+=i-1;a[n][i]=1;a[i][n]=1;
        }
    }
    printf("%d\n",n);
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)printf("%d",a[i][j]);
        putchar('\n');
    }
    return 0;
}
