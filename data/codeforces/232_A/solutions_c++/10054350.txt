#include <bits/stdc++.h>
using namespace std;

bool mat[101][101];

int main(){
    int k;
    cin>>k;
    int sz ;
    for(sz = 1 ; k ; sz++){
        for(int i=0;(i<sz) && (k) ; i++){
            if(i <= k){
                mat[i][sz] = mat[sz][i] = 1;
                k -= i;
            }
        }
    }
    cout<<sz<<endl;
    for(int i=0;i<sz;i++){
        for(int j=0;j<sz;j++){
            cout<<char(mat[i][j]+'0');
        }
        cout<<endl;
    }

}
