#include <cstdio>

int n, t;
bool a[150][150];

int main ()
{
    scanf ("%d", &n);
    for (int i = 2; i <= 100; i ++)
        for (int j = 1; j < i; j ++)
        {
            t = 0;
            for (int k = 1; k < j; k ++)
                if (a[i][k] && a[k][j])
                    t ++;
            if (t <= n) a[i][j] = a[j][i] = 1, n -= t;
        }
    puts ("100");
    for (int i = 1; i <= 100; i ++, puts (""))
        for (int j = 1; j <= 100; j ++)
            printf ("%d", a[i][j]);
    return 0;
}
