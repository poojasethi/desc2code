#include <cstdio>

int n,k,b,a[110][110];

int main() {
	scanf("%d",&n);
	for (k=1;(k+1)*k*(k-1)/6<=n;++k);
	for (int i=1;i<=k;++i)
		for (int j=1;j<=k;++j)
			if (i!=j) a[i][j]=a[j][i]=1;
	n-=k*(k-1)*(k-2)/6;
	while (n) {
		++k;
		for (b=1;b*(b+1)/2<=n;++b);
		n-=b*(b-1)/2;
		for (int i=1;i<=b;++i) a[i][k]=a[k][i]=1;
	}
	printf("%d\n",k);
	for (int i=1;i<=k;++i) {
		for (int j=1;j<=k;++j) printf("%d",a[i][j]);
		printf("\n");
	}
	return 0;
}
