#include <iostream>
#include <vector>
#include <map>
using namespace std;
int i,wynik,n;
map<int, map<int,int> > tab;
int main(){
	cin>>n;
	i=0;
	while(wynik<n){
		for(int j=0;j<i && wynik<n;j++){
			if(wynik+j>n)break;
			wynik+=j;
			tab[i][j]=1;
			tab[j][i]=1;
		}
		i++;
	}
	cout<<i<<endl;
	for(int a=0;a<i;a++){
		for(int b=0;b<i;b++)
			cout<<tab[a][b];
		cout<<endl;
	}
	return 0;
}
