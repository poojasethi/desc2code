#include<cstdio>
int d[101][101],n;
int main(){
    scanf("%d",&n);
    for(int i=1;i<=100;i++)
        for(int j=1;j<i;j++){
            int res=0;
            for(int k=1;k<=100;k++)
                if(d[i][k]&&d[j][k])res++;
            if(res<=n)n-=res,d[i][j]=d[j][i]=1;
        }
    printf("100\n");
    for(int i=1;i<=100;i++){
        for(int j=1;j<=100;j++)printf("%d",d[i][j]);printf("\n");
    }
    getchar(),getchar();
}
