k = int(raw_input())

n = 0
a = [[0 for x in xrange(100)] for x in xrange(100)] 

while(k > 0):
	m = n
	while(m * (m - 1) / 2 > k):
		m = m - 1
	for i in range(0, m):
		a[n][i] = 1
		a[i][n] = 1
	k = k - m * (m - 1) / 2
	n = n + 1

print(n)
for i in range(0, n):
	ans = ''
	for j in range(0, n):
		ans = ans + str(a[i][j])
	print(ans)
