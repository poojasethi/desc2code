#include<stdio.h>
#include<string.h>
#define INF 10000000

int min(int a,int b){return a<b?a:b;}

char S[1000001];

bool solve(int n,int t){
    int p=0;
    int m=0;
    for(int i=0;i<n-1;i++){
        if(S[i]=='L')p--;
        else p++;
        if(p==t)p--;
        m=min(m,p);
    }
    return p-1<m;
}

int main(){
    scanf(" %s",S);
    int n=strlen(S);
    int i;
    if(S[n-1]=='R'){
        for(i=0;i<n;i++){
            if(S[i]=='L')S[i]='R';
            else S[i]='L';
        }
    }
    int l=0,r=INF+1;
    while(r-l>1){
        int mid=(l+r)/2;
        if(solve(n,mid))l=mid;
        else r=mid;
    }
    if(l==INF)puts("1");
    else printf("%d\n",l);
    return 0;
}
