#include<stdio.h>
#include<string.h>
int f[2000005],len;
char str[1000005];
bool judge(int x){
	int i,pos=0;
	for(i=-len;i<len;i++)f[i+len]=0;
	f[len]=1;
	for(i=0;i<len;i++){
		pos+=str[i]=='R'?1:-1;
		if(pos>=x)pos=x-1;
		f[pos+len]++;
	}
	if(f[pos+len]>1)return false;
	else return true;
}
int i,l,r;
int main(){
	scanf("%s",str);
	len=strlen(str);
	if(str[len-1]=='R'){
		for(i=0;i<len;i++)str[i]=str[i]=='R'?'L':'R';
	}
	if(judge(len))printf("1\n");
	else {
		l=1;r=len;
		while(l<=r){
			int mid=l+r>>1;
			if(!judge(mid))r=mid-1;
			else l=mid+1;
		}
		printf("%d\n",l-1);
	}
}
