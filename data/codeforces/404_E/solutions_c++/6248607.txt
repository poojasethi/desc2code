#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>

using namespace std;

const int st=1000000,M=2000000;
int T[M+3],n,now;
char s[st+3];

bool check0()
{
now=st;T[now]++;
for (int i=1;i<=n;i++)
  if (s[i]=='L') T[--now]++;
    else T[++now]++;
return T[now]==1;
}

bool check(int cut)
{
memset(T,0,sizeof(T));
now=st;T[now]++;
for (int i=1;i<=n;i++)
  {
  if (s[i]=='L') now--;
    else now++;
  now=max(now,cut+1);
  T[now]++;
  }
return T[now]==1;
}

int main()
{
scanf("%s",s+1);n=strlen(s+1);
if (s[n]=='L')
  for (int i=1;i<=n;i++)
    if (s[i]=='L') s[i]='R';else s[i]='L';
if (check0()) {puts("1");return 0;}
int l=0,r=st-1;
for (;l<=r;)
  {
  int mid=(l+r)>>1;
  if (check(mid)) r=mid-1;
    else l=mid+1;
  }
printf("%d",st-l);
}
