#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=1000005;
char s[N];
int n,vis[N+N],opt[N];
int exe(int x)
{
    memset(vis,0,sizeof(vis));
    int p=0;++vis[p+N];
    for(int i=0;i<n;++i)
    {
        p+=opt[i];
        if(x&&p==x) p-=opt[i];
        else ++vis[p+N];
    }
    if(vis[p+N]==1) return 1;
    return 0;
}
void solve()
{
    int l=-N,r=0;
    while(l<r)
    {
        int m=l+r>>1;
        if(exe(m)) r=m;
        else l=m+1;
    }
    cout<<-r<<endl;
}
int main()
{
    scanf("%s",s);
    n=strlen(s);
    int flag=1;
    if(s[n-1]=='L') flag=-1;
    for(int i=0;i<n;++i) opt[i]=(s[i]=='R'?1:-1)*flag;
//    for(int i=0;i<n;++i) cout<<opt[i];
    if(exe(0))cout<<1<<endl;
    else solve();
    return 0;
}
