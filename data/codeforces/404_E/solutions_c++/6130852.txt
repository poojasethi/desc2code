#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

const int MAXN = 1000010;
char s[MAXN];
int L;

bool check(int D){
    int mi = 0,now = 0;
    for (int i=1;i<=L;i++){
        if (s[i]=='L') now--; else now++;
        if (now>D) now=D;
        if (i!=L) mi = min(mi,now);
    }
    return now<mi;
}

int main(){
  //  freopen("in.txt","r",stdin);
    gets(s+1);
    L = strlen(s+1);
    if (s[L]=='R')
        for (int i=1;i<=L;i++)
            s[i] = (s[i]=='L')?'R':'L';
    int lef = 0,rig = L+1;
    while(lef!=rig){
        int mid = (lef+rig)/2+1;
        if (check(mid)) lef = mid; else rig = mid-1;
    }
    if (!check(lef)){
        printf("0\n");
        return 0;
    }
    if (rig == L+1) printf("1\n"); else printf("%d\n",lef+1);



}
