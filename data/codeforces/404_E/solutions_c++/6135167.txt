#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char str[1111111];
char last;
int len;

bool check(int n)
{
    int pos=0;
    int minp=0;
    for(int i=0;i<len;i++)
    {
        minp = min(minp,pos);
        if(str[i]==last)
            pos--;
        else
            pos++;
        if(pos==n) pos--;
    }
    return pos<minp;
}

int main()
{
    scanf("%s",str);
    len = strlen(str);
    last = str[len-1];

    int l=1,r=len;
    while(l<=r)
    {
        int m = (l+r)>>1;
        if(!check(m))
            r = m-1;
        else
            l = m+1;
    }

    printf("%d\n", r==len?1:r);
}
