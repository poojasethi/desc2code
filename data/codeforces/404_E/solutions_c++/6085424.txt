#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxn = 1000000 + 5;
int s[maxn], n;
bool work(int lim)
{
	int now = 0, mn = 0;
	for(int i = 1; i < n; i ++)
	{
		if(!s[i]) now --;
		else if(now + 1 < lim)
			now ++;
		mn = min(mn, now);
	}
	return now - 1 < mn;
}
int main()
{
	while(char c = getchar())
	{
		if(c != 'L' && c != 'R') break;
		s[++ n] = c == 'R';
	}
	if(s[n] == 1)
		for(int i = 1; i <= n; i ++)
			s[i] ^= 1;
	if(work(n + 1))
		printf("1\n");
	else
	{
		int l = 0, r = n, mid;
		while(l < r)
		{
			mid = (l + r + 1) >> 1;
			if(work(mid)) l = mid;
			else r = mid - 1; 
		}
		printf("%d\n", l);
	}
	return 0;
}
