#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <vector>
#include <string>
#include <set>
#include <map>
#include <cmath>
#include <memory.h>
using namespace std;
typedef long long ll;

const int N = 1e6+6;

char s[N];

int main(){
   // freopen("input.txt","r",stdin);// freopen("output.txt","w",stdout);
    
    int n, m;
    
    gets(s);
    n=strlen(s);
    if(s[n-1]=='R') for(int i=0;i<n;++i) s[i]^=('R'^'L');
    
    int lx = 1, rx = 2e6;
    while(lx<rx){
        int px = (lx+rx)>>1;
        
        int x = 0, mx = 0;
        for(int i=0;i<n-1;++i){
            if(s[i]=='L') --x; else if(x+1<px) ++x;
            mx = min(mx, x);
        }
        
        bool ok = (x<=mx);
        
        if(ok) lx = px+1; else rx = px;
    }
    
    
    cout<<(lx>n ? 1 : lx-1)<<endl;
    
    
    
    return 0;
}
