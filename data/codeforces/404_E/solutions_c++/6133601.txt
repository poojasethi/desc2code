bool check(int o) {
  int p=0;
  int mx=0;
  bool uni=true;
  for (int i=0; i<n; i++) {
    if (s[i] == 'L') p--;
    else { if (p<o-1) p++; }
    if (p<mx) { mx=p; uni=true; }
    else { uni=false; }
  }
  return uni;
}
