#include <iostream>
#include <cstdio>

using namespace std;

char b[1001100];

int main()
{
    gets(b);
    int n=0;
    while(b[n]!='\0') n++;
    int pos=0;
    int m=0;
    for(int i=0;i<n-1;i++)
    {
        if(b[i]==b[n-1])
        {
            pos--;
            if(pos<m)m=pos;
        }else pos++;
    }
    if(pos==m){cout <<1; return 0;}
    int l=-1;
    int r=n;
    while(r-l>1)
    {
        int z = (l+r)/2;
        int pos = 0;
        int m = 0;
        for(int i=0;i<n-1;i++)
        {
            if(b[i]==b[n-1]) pos--; else pos++;
            if(pos>z) pos=z;
            if(m>pos) m=pos;
        }
        if(m==pos){l=z;} else r=z;
    }
    cout <<l+1;
    return 0;
}
