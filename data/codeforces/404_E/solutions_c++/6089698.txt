#define LEN 1000005

char c[LEN];
int n;

bool check(int x)
{
    int now=0,Min=0;
    for(int i=1;i<n;i++)
    {
        if(c[i]=='R') 
        {
            if(now+1<x)
                now++;
        }
        else now--; 
        Min=min(Min,now);
    }
    if(now-1<Min) return true;
    else return false;
}

        for(int i=1;i<=n;i++)
            c[i]='L'+'R'-c[i];
    if(check(n+1))

}
