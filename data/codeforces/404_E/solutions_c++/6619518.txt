#include <bits/stdc++.h>
using namespace std;
string str;
int Simulate(int idx)
{
	int place = 0;
	std::vector<int> V;
	for(int i = 0;i<str.length();i++)
	{
		V.push_back(place);
		place += (str[i]=='R'?1:-1);
		if(idx<0)
		{
			if(place<=idx)
				place = idx+1;
		}
	}
	for(int i = 0;i<V.size();i++)
	{
		if(place==V[i])
			return false;
	}
	return true;
}
int main()
{
	cin>>str;
	int N = str.length();
	if(str[N-1]=='L')
	{
		for(int i = 0;i<N;i++)
		{
			if(str[i]=='L')
				str[i] = 'R';
			else
				str[i] = 'L';
		}
	}
	if(Simulate(1))
	{
		cout<<1<<endl;
		return 0;
	}
	int l = -1e9;
	int r = -1;
	int res = 4;
	while(l<=r)
	{
		int m = (l+r)/2;
		if(Simulate(m))
		{
			res = m;
			r = m-1;
		}
		else
			l = m+1;
	}
	if(res==4)
		cout<<0<<endl;
	else
		cout<<abs(res)<<endl;
	return 0;
}
