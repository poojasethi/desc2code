#include <cstdio>
#include <cstring>

using namespace std;

int n;
char st[1000010];

void init() {
	scanf("%s",st+1);
	n = strlen(st+1);
	if (st[n] == 'R') 
		for (int i = 1; i <= n; i++)
			if (st[i] == 'L')
				st[i] = 'R';
			else
				st[i] = 'L';
}

bool check(int x) {
	int now = 0, left = 0;

	for (int i = 1; i <= n; i++) {
		if (st[i] == 'L') {
			now--;

			if (i == n) {
				if (now < left)
					return true;
				else
					return false;
			}

			if (now < left)  left = now;
		}
		else if (now+1 < x)
			now++;

	}

	return(false);
}

void solve() {
	if (check(n)) {
		printf("1\n");
		return;
	}

	int l = 0, r = n;
	while (l < r) {
		int m = (l+r+1)/2;

		if (check(m))
			l = m;
		else
			r = m-1;
	}
	
	printf("%d\n",l);
}

int main() {
	init();
	solve();
}
