#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#define MAXN 1100000
using namespace std ;

int n ; 
char s[MAXN] ;

int main()
{
	//freopen( "E.in" , "r" , stdin );
	//freopen( "E.out" , "w" , stdout );
	
	scanf( "%s" , s ); n = strlen( s ) ;
	
	int l = 1 , r = MAXN , mid , ret = 0 , DIS , MAX ;
	while ( l <= r )
	{
		mid = (l+r) / 2;
		MAX = DIS = mid - 1 ;
		for ( int i = 0 ; i < n-1 ; i++ ) 
		{	
			if ( s[i] == s[n-1] ) DIS ++ ;
			if ( s[i] != s[n-1] && DIS ) DIS -- ;
			MAX = max( MAX , DIS ) ;
		}
		if ( DIS + 1  > MAX ) ret = mid , l = mid + 1 ;
			else r = mid - 1 ;
	}
	
	if ( ret == MAXN ) ret = 1 ;
	printf( "%d\n" , ret );
	return 0 ;
}
