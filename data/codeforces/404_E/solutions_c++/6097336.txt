#include<stdio.h>
#include<string.h>
char s[1000005];
int n,a[1000005]={0},b[1000005];
bool check(int k)
{
	bool ans=true;
	int p=0,Min=0;
	for (int i=0;i<n;i++)
		{
			if (p+a[i]<k) p+=a[i];
			if (a[i]==1) ans=false;
			if (p<Min)
				{
					Min=p;
					ans=true;
				}
		}
	return ans;
}
int main()
{
	scanf("%s",s);n=strlen(s);
	for (int i=0;i<n;i++)
		if (s[i]=='L') a[i]=-1;
			else a[i]=1;
	if (a[n-1]==1)
		for (int i=0;i<n;i++) a[i]=-a[i];
	if (check(n+2))
		{
			printf("1\n");
			return 0;
		}
	int l=0,r=n+1,mid;
	while (l+1<r)
		{
			mid=(l+r)/2;	
			if (check(mid)) l=mid;
				else r=mid;
		}
	printf("%d\n",l);
}
