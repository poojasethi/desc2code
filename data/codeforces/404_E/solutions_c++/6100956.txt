//E
#include <cstdio>

#define N 1000000

int  n, i, m, ll, rr, t, p, r, x;
char c;
bool a[N], f;

int main()
{
    while ((c = getchar()) != '\n')
        a[n++] = c == 'R';

    if (!a[n - 1])
        for (i = 0; i < n; i++)
            a[i] ^= 1;

    ll = 0, rr = n + 1;
    while (ll < rr)
    {
        t = (ll + rr + 1) >> 1;
        m = 0, x = 0;
        for (i = 0; i < n; i++)
        {
            f = 0;
            if (a[i])
                ++m;
            else
                if (m - 1 > -t)
                    --m;
            if (m > x)
                x = m, f = 1;
        }
        if (f)
            ll = t;
        else
            rr = t - 1;
    }

    printf("%d\n", ll == n + 1 ? 1 : ll);

    return 0;
}
