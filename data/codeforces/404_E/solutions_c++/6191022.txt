#include<stdio.h>
#include<string.h>
#define N 1000001
char str[1000005];
int flag[2000005],len;
int judge(int x)
{
	int t=0;
	memset(flag,0,sizeof(flag));
	flag[0+N]=1;
	for(int i=0;i<len;i++)
	{
		if(str[i]=='L')t--;
		else t++;
		if(t>=x)t--;
		flag[t+N]++;
	}
	if(flag[t+N]>1)return 0;
	else return 1;
}
int main()
{
	int n,i,j,k,t,l,r,mid,lmin,rmax;
	while(scanf("%s",str)!=EOF)
	{
		len=strlen(str);
		if(str[len-1]=='R')
		for(i=0;i<len;i++)
		{
			if(str[i]=='L')str[i]='R';
			else str[i]='L';
		}
		if(judge(len)){
			printf("1\n");continue;
		}
		l=1;r=len;
		while(l<=r)
		{
			mid=(l+r)/2;
			if(!judge(mid))r=mid-1;
			else l=mid+1;	
		}
		printf("%d\n",l-1);
	}
}
