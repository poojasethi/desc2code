#include <cstring>
#include <cstdio>
#include <algorithm>

using namespace std ;

const int MAXN = 1000000 + 10 ;

char str[MAXN] ;

bool ok( int t ) {
    int pos = 0 ; int minpos = 0 ;
    for( int i = 0 ; str[i+1] != '\0' ; i ++ ) {
        if( str[i] == 'L' ) minpos = min( minpos , --pos ) ;
        if( str[i] == 'R' && pos < t - 1 ) pos ++ ;
    }
    return --pos < minpos ;
}

int main() {

    scanf("%s" , str ) ;
    int len = strlen( str ) ;

    if( str[len-1] == 'R' ) for( int i = 0 ; i < len ; i ++ ) str[i] = str[i] == 'R' ? 'L' : 'R' ;

    int l = 0 , r = len ;

    while( l < r ) {
        int m = l + ( r - l + 1 ) / 2 ;
        if( ok( m ) ) l = m ; else r = m - 1 ;
    }

    printf("%d\n" , l == len ? 1 : l ) ;

    return 0 ;
}
