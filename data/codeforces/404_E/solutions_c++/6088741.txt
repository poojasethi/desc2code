#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
const int N = 1000050;
int n,val;
char s[N];
inline int get(char c){ return c=='L'?-val:val; }
int calc(int x,int k)
{
    int mx=0,num=2;
    for(int i=0;i<n;++i)
    {
        if(x+get(s[i])!=k)
            x+=get(s[i]);
        if(mx<x)
            mx=x,num=1;
        else if(mx==x) ++num;
    }
    if(num==1&&x==mx) return mx;
    return N;
}
int pro()
{
    int l=1,r=N-1;
    while(l<=r){
        int mid=l+r>>1;
        if(calc(0,-mid)==N)
            r=mid-1;
        else
            l=mid+1;
    }
    return r;
}
int solve()
{
    if(calc(0,N)<N) return 1;
    return pro();
}
int main()
{
    while(~scanf("%s",s))
    {
        n=strlen(s);
        val=(s[n-1]=='R'?1:-1);
        printf("%d\n",solve());
    }
    return 0;
}
