#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
#define M 1001001
#define MIN(a,b) (a<b?a:b)
#define MAX(a,b) (a>b?a:b)
char s[M];
int n;

bool able(int m){
	int mn=0, mx=0, cur=0;
	for(int i=0; i<n; i++){
		mn=MIN(mn, cur);
		mx=MAX(mx, cur);
		if(s[i]=='R')cur++; else cur--;
		if(cur==m)cur--;
	}
	return cur<mn;
}

int main(){
	int i, j, k;
	scanf("%s", s);
	n = strlen(s);
	if(s[n-1]=='R'){
		for(i=0; i<n; i++)s[i]='L'+'R'-s[i];
	}
	int cur=0;
	int mn=0, mx=0;
	for(i=0; i<n; i++){
		mn=MIN(mn, cur);
		mx=MAX(mx, cur);
		if(s[i]=='R')cur++;	else cur--;
	}
	if(cur<mn){
		puts("1");
		return 0;
	}
	if(!able(1)){
		puts("0");
		return 0;
	}
	int l=1, r=mx+1, m;
	while(l<r-1){
		m=(l+r)/2;
		if(able(m))l=m;
		else r=m;
	}
	printf("%d\n", l);
	return 0;
}
