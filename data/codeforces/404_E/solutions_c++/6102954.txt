#include <cstdio>

#define min(x,y) (((x)<(y))?(x):(y))
#define max(x,y) (((x)>(y))?(x):(y))

int a[1000005], n;

bool pos(int bl){
	int c, sr;
	c = sr = 0;
	for(int i = 0; i < n-1; i++){
		c += a[i];
		if(c >= bl) c--;
		sr = min(c,sr);
	}
	return c == sr;
}

int main(){
	char in;
	while((in = getchar())!='\n'){
		a[n++] = (in=='L')?-1:1;
	}
	if(a[n-1] == 1)
		for(int i = 0; i < n; i++)
			a[i] *= -1;

	if(pos(2000000)){
		printf("%d\n", 1);
		return 0;
	}

	int s = 1, e = 2000000, m, b = 0;
	while(s <= e){
		m = (s+e)/2;
		if(pos(m)){
			b = max(b,m);
			s = m+1;
		}
		else
			e = m-1;
	}

	printf("%d\n", b);
	return 0;
}
