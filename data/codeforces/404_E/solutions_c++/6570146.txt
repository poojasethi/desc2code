#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

char a[1000006];
char last;
int l,r,mid;
int len;

bool check(int n)
{
    int now=0;
    int __min=0;
    for(int i=0;i<len;i++)
    {
        __min=min(now,__min);
        if(a[i]==last)
            now--;
        else
            now++;
        if(now==n) now--;
    }
    if(now<__min) return true;
    return false;
}

int main()
{
    scanf("%s",a);
    l=1,r=len=strlen(a);
    last=a[r-1];
    while(l<=r)
    {
        mid=(l+r)>>1;
        if(check(mid))
            l=mid+1;
        else r=mid-1;
    }
    if(r==len) printf("1\n");
    else
        printf("%d\n",r);
    return 0;
}
