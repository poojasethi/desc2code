#include <cstdio>
char s[1001001];
int l;
int go(int b) {
	int t = 0, x = 0;
	for (int i = 0; i < l; i++) {
		t += s[i] == 'L' ? -1 : 1;
		if (s[l] == 'L' && t == b) t--;
		if (s[l] == 'R' && t == -b) t++;
		if (s[l] == 'L' && x > t) x = t;
		if (s[l] == 'R' && x < t) x = t;
	}
	return x == t;
}
int main() {
	scanf("%s", s);
	while (s[l]) l++;
	--l;
	if (go(10000000)) {
		printf("1");
		return 0;
	}
	if (!go(1)) {
		printf("0");
		return 0;
	}
	int lo = 1, hi = 10000000, mid;
	while (lo + 1 < hi) {
		mid = (lo + hi) / 2;
		if (go(mid)) {
			lo = mid;
		} else {
			hi = mid;
		}
	}
	printf("%d", lo);
	return 0;
}
