#include <iostream>
#include <string>
#include <cstring>
#include <cstdio>

using namespace std;

int n;
string str;

bool judge(int obs)
{
	int now=0,l=0;
	for(int i=0;i<n;++i)
	{
		if(str[i]=='L')
		{
			--now;
			if(i==n-1)
			{
				if(now<l)
					return true;
				return false;
			}
			if(now<l)
				l=now;
		}
		else
		{
			if(now+1<obs) ++now;
		}
	}
}

int main()
{
	cin>>str;
	n=str.size();
	if(str[n-1]=='R')
		for(int i=0;i<n;++i)
			str[i]=(str[i]=='R'?'L':'R');
	if(judge(n))
	{
		cout<<1<<endl;
		return 0;
	}
	int l=1,r=n-1,ans=-1;
	while(l<=r)
	{
		int mid=l+r>>1;
		if(judge(mid)) l=mid+1,ans=mid;
		else r=mid-1;
	}
	if(ans!=-1) cout<<ans<<endl;
	else cout<<0<<endl;
	return 0;
}
