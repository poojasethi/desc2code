#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxn=1000050;
int n,s[maxn];
bool judg(int lim)
{
	int i,cur=0,d=0;
	for(i=0;i<n;i++)
		d=min(cur,d),cur+=s[i],cur=min(cur,lim);
	//printf("%d %d\n",lim,cur);
	return cur<d;
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j;
	char ci;
	while((ci=getchar())!='\n')
		s[n++]=(ci=='L'?-1:1);
	if(s[n-1]==1)
		for(i=0;i<n;i++)
			s[i]=-s[i];
	//printf("%d\n",n);
	int l=-1,r=n,mid;
	while(l<r)
	{
		mid=l+(r-l+1)/2;
		if(judg(mid)) l=mid;
		else r=mid-1;
	}
	printf("%d\n",l==n?1:l+1);
	return 0;
}
