#include<iostream>
#include<cstdio>
#include<cstring>
#define N 1000009
using namespace std;
bool s[N];
int n;
bool can(int lim){
    int pos=0,L=0,c=1;
    for(int i=0;i<n;i++){
        if(s[i])pos--;
        else pos++;
        if(pos>=lim)pos=lim-1;
        if(pos<L){L=pos;c=1;}
        else if(pos==L)c++;
    }
    return (pos==L && c==1);
}
int main(){
    char ch;
    while((ch=getchar())!='\n'){
        if(ch=='L')s[n++]=true;
        else s[n++]=false;
    }
    if(!s[n-1]){for(int i=0;i<n;i++)s[i]=!s[i];}
    int l=0,r=N;
    while(l<r){
        int mid=(l+r+1)>>1;
        if(can(mid))l=mid;
        else r=mid-1;
    }
    if(l==N)cout<<"1"<<endl;
    else cout<<l<<endl;
    return 0;
}
