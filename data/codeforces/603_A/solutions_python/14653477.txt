n = input()
s = raw_input()
cnt, parts, flag = 0, 0, False
i = 0
while i < n:
    parts += 1
    if i < n - 1 and s[i] == s[i+1]:
        cnt += 1
        acc = 1
        while i < n - 1 and s[i] == s[i+1]:
            i += 1
            acc += 1
        if acc >= 3:
            flag = True
    i += 1

if cnt == 1 and flag:
    cnt = 2
print parts + min(2, cnt)
