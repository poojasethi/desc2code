n=int(raw_input())
string=str(raw_input())
ans=1
prev=string[0]
for j in range(1,n):
    if(string[j]!=prev):
        ans+=1
        prev=string[j]
temp=0
for j in range(1,n):
    if(string[j]==string[j-1]):
        temp+=1
        for k in range(j+1,n):
            if(string[k]==string[k-1]):
               temp+=1
               break
        break
print ans+temp
