using namespace std;
int n,k,digit[10],u[10];
char s[111],t[111],tmp[111];
int a[4];
vector<string> ans;
void add(int i,int j,int len)
{
	if(i==len&&j==4)
	{
		sprintf(tmp,"%d.%d.%d.%d",a[0],a[1],a[2],a[3]);
		ans.push_back(tmp);
	}
	if(i==len||j==4)return;
	if(i+(4-j)>len)return;
	if(i+3*(4-j)<len)return;
	if(i+1<=len)
	{	
		a[j]=t[i];
		add(i+1,j+1,len);
	}
	if(i+2<=len&&t[i]!=0)
	{
		a[j]=(int)(t[i]*10+t[i+1]);
		add(i+2,j+1,len);
	}
	if(i+3<=len&&t[i]!=0)
	{
		a[j]=((int)(t[i]*10+t[i+1]))*10+t[i+2];
		if(a[j]<256)add(i+3,j+1,len);
	}
}
void go(int i)
{
	int good=1;
	for(int j=0;j<10;j++)
		if(digit[j]&&!u[j])good=0;
	if(i&&good)
	{
		memcpy(t,s,i);
		memcpy(t+i,s,i);
		reverse(t+i,t+2*i);
		add(0,0,2*i);
		memcpy(t+i-1,s,i);
		reverse(t+i-1,t+2*i-1);
		add(0,0,2*i-1);
	}
	if(i==6)return;
	for(int j=0;j<10;j++)
	{
		if(digit[j])
		{
			s[i]=j;
			u[j]++;
			go(i+1);
			u[j]--;
		}
	}
}
int main()
{
	int n,k;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>k;
		digit[k]=1;
	}
	go(0);
	cout<<ans.size()<<endl;
	for(int i=0;i<(int)ans.size();i++)
	cout<<ans[i]<<endl;
	return 0;
}
