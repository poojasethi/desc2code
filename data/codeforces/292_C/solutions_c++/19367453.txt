#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<memory.h>
using namespace std;
int n,ip[15];
vector<string> ans;
bool a[11];
bool pal(int l,int r){
	bool f=true;
	for(int i=l;i<=(l+r)/2+1;i++){
		if (ip[i]!=ip[r+l-i]){
			f=false;break;
			
		}
	}
	return f;
}
int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		int x;
		cin>>x;
		a[x]=true;
	}
	for(int x1=0;x1<256;x1++){
		for(int x2=0;x2<256;x2++){
			for(int x3=0;x3<256;x3++){
				bool b[11],f=true;
				int cnt=3;
				memset(b,false,sizeof(b));
				int x[3]={x3,x2,x1};
				for(int i=0;i<3;i++){
					int k=0,tmp[3];
					while (true){
						b[x[i]%10]=true;
						tmp[k++]=x[i]%10;
						x[i]/=10;
						if (x[i]==0) break;
					}
					while(k--){
						ip[cnt++]=tmp[k];
					}
				}
				for(int i=0;i<10;i++)
				if ((a[i] && !b[i])||(b[i] && !a[i])){
					f=false;break;
				}
				if (!f) continue;
				for(int i=1;i<=3;i++){
					int s=0;
					for(int j=3-i;j<3;j++)ip[j]=ip[cnt+2-i-j],s=s*10+ip[j];
					if (s<256 && (i==1 || ip[3-i]>0) && pal(3-i,cnt-1)){
						char tmp[100];
						sprintf(tmp,"%d.%d.%d.%d",s,x3,x2,x1);
						ans.push_back(tmp);
					}
				}
			}
		}
	}
	printf("%d\n",ans.size());
	for(int i=0;i<ans.size();i++)printf("%s\n",ans[i].c_str());
	return 0;
}
