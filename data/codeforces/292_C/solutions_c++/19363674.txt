#include <cstdio>
#include <iostream>
#include <cstring>
#include <string>
using namespace std;

int out[100001][5];
int ans,f[13];
int n,t,num[5],s;
bool a[10],have[10];

void dfs(int p)
{
	if (p==s+1)
	{
		int pos=0;
		for(int i=1;i<=4;i++)
		{
			int temp=0;
			for(int j=1;j<=num[i];j++)
			{
				pos++;
				if (j==1 && f[pos]==0 && num[i]!=1) return;
				temp=temp*10+f[pos];
			}
			if (temp>255) return;
		}   
		
		for(int i=0;i<=9;i++) have[i]=false;
		for(int i=1;i<p;i++) have[f[i]]=true;
		for(int i=0;i<=9;i++)
			if (a[i] && !have[i]) return;
		
		ans++;
		pos=0;
		for(int i=1;i<=4;i++)
		{
			int temp=0;
			for(int j=1;j<=num[i];j++)
			    pos++,temp=temp*10+f[pos];
			out[ans][i]=temp;
		} 
	}
	
	else if (s%2==0 && p>s/2 || s%2==1 && p>s/2+1)
	{
		f[p]=f[s+1-p];
		dfs(p+1);
	}
	
	else
	{
		for(int i=0;i<=9;i++)
		    if (a[i])
		    {
		    	f[p]=i;
		    	dfs(p+1);
		    }
	}
	return;
}

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&t);
		a[t]=true;
	}
	
	for(int i=0;i<=80;i++)
	{
		int temp=i;s=0;
		for(int j=1;j<=4;j++)
		    num[j]=temp%3+1,temp/=3,s+=num[j];
		
		dfs(1);
	}
	
	printf("%d\n",ans);
	for(int i=1;i<=ans;i++)
	    for(int j=1;j<=4;j++)
	    {
	    	printf("%d",out[i][j]);
	    	if (j<=3) printf("."); else printf("\n");
	    }
	return 0;
}
