#include <iostream>

using namespace std;

int main()
{
    int n;
    cin>>n;

    string s,t;
    cin>>s>>t;
    int first =0 ,second = 0,both=0;
    for(int i = 0 ; i<2*n ; i++)
    {
      if(s[i]=='1' && t[i]=='1')  both++;
       if(s[i]=='1') first++;
       if(t[i]=='1') second++;
    }


     if(both%2==1) second--;

    if(first>second) cout<<"First"<<endl;
    else if(second-first >1)  cout<<"Second"<<endl;
    else cout<<"Draw"<<endl;
    return 0;
}
