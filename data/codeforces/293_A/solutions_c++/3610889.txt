#include<stdio.h>
#include<string.h>

int amm;
char s[5000000],t[5000000];

int main()
{
    scanf("%d%s%s",&amm,s,t);
    int a=0,b=0,c=0;
    for (int i=0;i<amm*2;i++)
    {
        if (s[i]=='1'&&t[i]=='1')c++;
        else if (s[i]=='1')a++;
        else if (t[i]=='1')b++;
    }
    a+=(c+1)/2;
    b+=c/2;
    if (a>b)puts("First");
    else if (a+1<b)puts("Second");
    else puts("Draw");
    return 0;
}
