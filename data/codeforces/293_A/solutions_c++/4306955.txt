#include <iostream>
#include <string>

using namespace std;

int main(){
	int n,k1=0,k2=0,k3=0;
	cin>>n;
	string s1,s2;
	cin>>s1>>s2;
	for (int i=0; i<2*n; i++){
		if (s1[i]=='0' & s2[i]=='1') k2++;
		if (s1[i]=='1' & s2[i]=='0') k1++;
		if (s1[i]=='1' & s2[i]=='1') k3++;
	}
	if (k3%2==1) k1++;
	if (k2-k1==1) k1=k2;
	if (k1>k2) cout<<"First";
	if (k1==k2) cout<<"Draw";
	if (k1<k2) cout<<"Second";
}
