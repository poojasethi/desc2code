#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	ll n;
	string s1,s2;
	cin>>n;
	cin>>s1>>s2;
	ll f=0,s=0,d=0;

	for(int i=0;i<s1.length();i++)
	{

		{if(s1[i]=='1'&&s2[i]=='0')
			f+=1;
	    if(s1[i]=='0'&&s2[i]=='1')
	        s+=1;
	    if(s1[i]=='1'&&s2[i]=='1')
             d++;
		}

	}
	d%=2;
	f+=d;

//	cout<<f<<" "<<s<<endl;
	if(f==s||f+1==s)
		cout<<"Draw"<<endl;
	else
		if(f<s)
		cout<<"Second"<<endl;
	else
		cout<<"First"<<endl;
}
