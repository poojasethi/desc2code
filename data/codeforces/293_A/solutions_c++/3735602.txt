#include <iostream>
using namespace std;

int a[3], n, scr[2];
string s, t;

int main() {
	cin.sync_with_stdio(false);
	cin >> n >> s >> t;
	for (int i = 0; i < 2 * n; i++) {
		int x = s[i] - '0', y = t[i] - '0';
		scr[0] += x;
		scr[1] += y;
		a[x + y]++;
	}
	if (a[2] % 2) {
		scr[0] += 2;
		scr[1] += (a[1] % 2);
	}
	else 
		scr[0] += a[1] % 2;
	if (scr[0] > scr[1])
		cout << "First\n";
	else if (scr[0] == scr[1])
		cout << "Draw\n";
	else
		cout << "Second\n";
	return 0;
}
