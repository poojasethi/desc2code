#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	string s1, s2;
	
	cin>>n;
	cin>>s1;
	cin>>s2;
	
	int cont1 = 0, cont2 = 0, cont3 =0;
	for(int i = 0; i < 2*n; i++){
		if(s1[i] - '0') cont1++;
		if(s2[i] - '0') cont2++;
		if(s1[i] == s2[i] && (s1[i] - '0')) cont3++;
	}
	
	if(cont3 % 2 == 1) cont2--;
	if(cont1 > cont2) cout<<"First"<<endl;
	else{
		if(cont2 - cont1 > 1)cout<<"Second"<<endl;
		else cout<<"Draw"<<endl;
	}
	return 0;
}
