#include<cstdio>

char a[2001000]={0},b[2001000]={0};

int main()
{
	int n;
	scanf("%d",&n);
	scanf("%c",&a[0]);
	for (int i=1; i<=n*2; i++)
		scanf("%c",&a[i]);
	scanf("%c",&b[0]);
	for (int i=1; i<=n*2; i++)
		scanf("%c",&b[i]);
	int same=0,x1=0,x2=0;
	for (int i=1; i<=n*2; i++)
	{
		if (a[i]=='1'&&b[i]=='1') same++; else
		if (a[i]=='1') x1++; else
		if (b[i]=='1') x2++;
	}
	if (same%2) x1++;
	if (x1>x2) printf("First\n"); else
	if (x1<x2-1) printf("Second\n"); else
	printf("Draw\n"); return 0;
}
