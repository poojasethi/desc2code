#include <bits/stdc++.h>
using namespace std;

char a[2000005];
char b[2000005];

int main(){
	
	
    int i,n,coun=0,l=0,r=0;
    
	scanf("%d%s%s",&n,a,b);
    
	for(i=0;i<n+n;i++){
    
	              if(a[i]=='1' && b[i]=='1') coun +=1;
                 else if(a[i]=='1' && b[i]=='0') l++;
                  else if(a[i]=='0' && b[i]=='1') r++;
    }
    coun=coun%2;
    
    l+=coun;
    if(l>r) puts("First"); 
	else if(l<r-1) puts("Second");
	 else puts("Draw");
	 cout<<endl;
	 
	 return 0;
}
