#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
char s[3000010], t[3000010];
int n, a[10][10];

int main()
{
	while (scanf("%d", &n) != EOF)
	{
		memset(a, 0, sizeof(a));
		scanf("%s%s", s, t);
		for (int i = 0; i < 2 * n; i++)
			a[s[i] - '0'][t[i] - '0']++;
		if (a[1][1] % 2 == 1)
			a[0][1]--;
		if (a[1][0] > a[0][1])
			puts("First");
		else if (a[1][0] < a[0][1] - 1)
			puts("Second");
		else puts("Draw");
	}
	return 0;
}
