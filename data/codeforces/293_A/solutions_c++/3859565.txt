#include<iostream>
using namespace std;
//char a[1000001],b[1000001];
string a,b;
int n,s1,st1,s2;
int main()
{
	cin>>n;
	cin>>a>>b;
	for(int i=0;i<a.length();i++)
	{
		if(a[i]=='1'&&a[i]==b[i])
		st1++;
		if(a[i]=='1')
		s1++;
		if(b[i]=='1')
		s2++;
	}
	if((st1%2&&s1>=s2)||(st1%2==0 && s1>s2))
	cout<<"First";
	else if((st1%2==0 && s1==s2)||(st1%2&&s1+2==s2)||(s1+1==s2))
	cout<<"Draw";
	else 
	cout<<"Second";
	return 0;
	
}
