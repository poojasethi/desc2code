#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char a[2000010],b[2000010];
struct rec{int x,y;}c[2000010];
int n,i,x,y;
bool operator <(rec a,rec b)
{
	return a.x>b.x||a.x==b.x&&a.y>b.y;
}
int main()
{
	cin>>n;
	n*=2;
	scanf("%s%s",a+1,b+1);
	for(i=1;i<=n;i++)
		c[i].x=a[i]-'0',c[i].y=b[i]-'0';
	sort(c+1,c+n+1);
	for(i=1;i<=n;i++)
		if(i&1) x+=c[i].x; else y+=c[i].y;
	if(x>y) puts("First"); else if(x<y) puts("Second"); else puts("Draw");
}
