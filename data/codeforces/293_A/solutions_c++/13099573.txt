#include "bits/stdc++.h"
using namespace std;
int main()
{
    int a,b,n,t=0,aa=0,bb=0;
    string f,s;
    cin>>n;
    cin>>f>>s;
    n=2*n;
    for (int i=0;i<n;i++)
    {
        a=f[i];
        b=s[i];
        if (a==b&&a=='1') t++;
        if (a=='1') aa++;
        if (b=='1') bb++;
    }
    if (t%2!=0) aa++;
    if (aa>bb) cout<<"First";
    else if (bb>aa+1) cout<<"Second";
    else cout<<"Draw";
}
