#include <iostream>
#include <cmath>
#include <cstdio>
#include <vector>
using namespace std;
int main () {
	int n, ss=0, tt=0, st=0;
	string s, t;
	cin >> n >> s >> t;
	for (int i=0; i<2*n; i++){
		if (s[i]=='1' && t[i]=='0') ss++;
		if (s[i]=='0' && t[i]=='1') tt++;
		if (s[i]=='1' && t[i]=='1') st++;
	} 
	if ((ss==tt && st%2==0) || (ss+2==tt && st%2==1) || 
	(tt==ss+1 && (st==0 || st%2==1))) cout << "Draw";
	else if (ss>tt || (ss==tt && st%2==1)) cout << "First";
	else cout << "Second";
	return 0;
}
