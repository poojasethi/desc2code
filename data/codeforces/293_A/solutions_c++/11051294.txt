#include <iostream>
#include <string>
using namespace std;
int main()
{
	int n;
	string st1,st2;
	while(cin>>n>>st1>>st2)
	{
		int a=0,b=0,c=0;
		for(int i=0;i<n*2;i++)
			if(st1[i]=='1' && st2[i]=='1')
				c++;
			else if(st1[i]=='1' && st2[i]=='0')
				a++;
			else if(st1[i]=='0' && st2[i]=='1')
				b++;
		if(c%2!=0)
			a++;
		if(a<b)
			b--;
		if(a==b)
			cout<<"Draw"<<endl;
		else if(a>b)
			cout<<"First"<<endl;
		else
			cout<<"Second"<<endl;
	}
	return 0;
}
