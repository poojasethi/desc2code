from sys import stdin
def main():
    n = int(stdin.readline())
    s = stdin.readline().strip()
    t = stdin.readline().strip()
    a = b = c = 0
    for x, y in zip(s, t):
        if x == y == '1':
            a = a + 1
        elif x == '1':
            b = b + 1
        elif y == '1':
            c = c + 1
    if a % 2:
        b = b + 1
    if b > c:
        print "First"
    elif b >= c - 1:
        print "Draw"
    else:
        print "Second"
main()
