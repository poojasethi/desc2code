#!/usr/bin/env python
# -*- coding: UTF-8 -*-

import sys

def ReadIn():
    data = sys.stdin.read().split()
    offset = 0
    while offset < len(data):
        n = int(data[offset])
        offset += 1
        s, t = data[offset : offset + 2]
        offset += 2
        yield s, t

def Solve(s, t):
    cnt = {'00' : 0, '01' : 0, '10' : 0, '11' : 0}
    for x, y in zip(s, t):
        cnt[x + y] += 1
    ans = 0
    cnt['10'] += cnt['11'] & 1
    if cnt['10'] >= cnt['01']:
        ans += (cnt['10'] - cnt['01'] + 1) >> 1
    else:
        ans -= (cnt['01'] - cnt['10']) >> 1
    if ans > 0:
        print 'First'
    elif ans < 0:
        print 'Second'
    else:
        print 'Draw'

if __name__ == '__main__':
    for s, t in ReadIn():
        Solve(s, t)
