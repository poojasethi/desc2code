n=input()
s=raw_input()
t=raw_input()
a,b,c=0,0,0
for i in xrange(2*n):
    if s[i] == t[i] and s[i] == '1':
        c += 1
    elif s[i] == '1':
        a += 1
    elif t[i] == '1':
        b += 1
if c % 2:
    if a >= b: print "First"
    else:
        if b-a<=2: print "Draw"
        else: print "Second"
else:
    if a>b: print "First"
    else:
        if a>=b-1: print "Draw"
        else: print "Second"
