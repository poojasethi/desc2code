import sys

n = int(sys.stdin.readline())
a = sys.stdin.readline()
b = sys.stdin.readline()
a1, b1, ab1 = 0, 0, 0
for i in xrange(2 * n):
    if a[i] == '1' and b[i] == '1':
        ab1 += 1
    elif a[i] == '1':
        a1 += 1
    elif b[i] == '1':
        b1 += 1
am = (ab1 + 1) // 2 + a1
bm = ab1 // 2 + b1
print "First" if am - bm >= 1 else "Second" if bm - am >= 2 else "Draw"
