#include <bits/stdC++.h>

using namespace std;
#define PB push_back
const int N=(5e+2)+10;
vector<int> a[N];
int d[N][N][3]={0};
pair<int,int> p[N][N][3];
int main(){
    ios::sync_with_stdio(false);cin.tie(0);
    int n,m,u,v;
    cin>>n>>m;
    for(int i=0;i<m;i++){
        cin>>u>>v;
        u--;v--;
        a[u].PB(v);
        a[v].PB(u);
    }
    queue<int> q;
    q.push(0);q.push(n-1);q.push(0);
    d[0][n-1][0]=1;
    while(q.size()){
        int x=q.front();q.pop();
        int y=q.front();q.pop();
        int r=q.front();q.pop();
        if(r==0){
            for(int i=0;i<a[x].size();i++){
                if(d[a[x][i]][y][1-r]==0){
                    q.push(a[x][i]);q.push(y);q.push(1-r);
                    p[a[x][i]][y][1-r]={x,y};
                    d[a[x][i]][y][1-r]=d[x][y][r]+1;
                }
            }
        }else{
            for(int i=0;i<a[y].size();i++){
                if(d[x][a[y][i]][1-r]==0 && a[y][i]!=x){
                    q.push(x);q.push(a[y][i]);q.push(1-r);
                    p[x][a[y][i]][1-r]={x,y};
                    d[x][a[y][i]][1-r]=d[x][y][r]+1;
                }
            }
        }
    }
    vector<pair<int,int> > ans;
    if((d[n-1][0][0]-1)/2==0){
        cout<<-1;
        return 0;
    }
    cout<<(d[n-1][0][0]-1)/2<<"\n";
    pair<int,int> d;
    d={n-1,0};
    ans.PB(d);
    int r=0;
    while(!(d==make_pair(0,n-1) && r==0)){
        d=p[d.first][d.second][r];
        r=1-r;
        ans.PB(d);
    }
    for(int i=ans.size()-1;i>=0;i-=2)
        cout<<ans[i].first+1<<" ";

    cout<<"\n";
    for(int i=ans.size()-1;i>=0;i-=2)
        cout<<ans[i].second+1<<" ";

    return 0;
}
