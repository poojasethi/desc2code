#include <cstdio>
#include <iostream>
#include <vector>
#include <queue>
#define MAXN 500
#define mp(a,b,c) make_pair(make_pair(a,b),c)
#define fi first.first
#define se first.second
#define th second
using namespace std;
typedef pair<int,int> ii;
typedef pair<ii,int> iii;
queue <iii> q;
int N,m;
int H[MAXN+1][MAXN+1][2];
iii yol[MAXN+1][MAXN+1][2];
vector <int> A[MAXN+1];
vector <ii> res;
int solve()
{
	iii t;
	int i,j;
	int a,b;
	yol[1][N][0]=mp(0,0,0);
	while(!q.empty())
	{
		t=q.front();
		q.pop();
		a=t.fi; b=t.se;
		//printf("%d %d %d %d\n",a,b,yol[a][b].fi,yol[a][b].se);
		
		if(t == mp(N,1,0)) return 0;
		
		if(!t.th)
		{
			for(i=0;i<A[a].size();i++)
			{
				if(H[A[a][i]][b][1]) continue;
				H[A[a][i]][b][1]=1;
				yol[A[a][i]][b][1]=t;
				q.push(mp(A[a][i],b,1));
			}
		}
		else
		{
			for(i=0;i<A[b].size();i++)
			{
				if(H[a][A[b][i]][0] || a==A[b][i]) continue;
				H[a][A[b][i]][0]=1;
				yol[a][A[b][i]][0]=t;
				q.push(mp(a,A[b][i],0));
			}
		}
	}
	return 1;
}
int main()
{
	iii t;
	int i;
	int k=0;
	int a,b;
	scanf(" %d %d",&N,&m);
	while(m--)
	{
		scanf(" %d %d",&a,&b);
		A[a].push_back(b);
		A[b].push_back(a);
	}
	H[1][N][0]=1;
	q.push(mp(1,N,0));
	if(solve())
	{
		printf("-1\n");
		return 0;
	}
	t=mp(N,1,0);
	while(t!=mp(0,0,0))
	{
		if(!t.th) res.push_back(make_pair(t.fi,t.se));
		t=yol[t.fi][t.se][t.th];
	}
	printf("%d\n",res.size()-1);
	for(i=res.size()-1;i>=0;i--)
		printf("%d ",res[i].first);
	printf("\n");
	for(i=res.size()-1;i>=0;i--)
		printf("%d ",res[i].second);
	printf("\n");
	return 0;
}
