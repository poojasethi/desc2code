#include<bits/stdc++.h>
#define ll long long 

using namespace std;

const int MAXN=500+10;
int dis[MAXN][MAXN][2];
vector<int>a[MAXN];
int p[MAXN][MAXN][2][2];

int main(){
	int n,m;
	cin>>n>>m;
	int x,y;
	for(int i=0;i<m;i++)
		cin>>x>>y,a[x].push_back(y),a[y].push_back(x);
	for(int i=1;i<n+1;i++)
		for(int j=1;j<n+1;j++)
			dis[i][j][0]=dis[i][j][1]=-1;
	queue<int>q;
	q.push(1);
	q.push(n);
	q.push(0);
	dis[1][n][0]=0;
	while(q.size()){
		int x=q.front();
		q.pop();
		int y=q.front();
		q.pop();
		int z=q.front();
		q.pop();
		if(z==0)
			for(int i=0;i<a[x].size();i++)
				if(dis[a[x][i]][y][1]==-1)
					dis[a[x][i]][y][1]=dis[x][y][0],q.push(a[x][i]),q.push(y),q.push(1),p[a[x][i]][y][1][0]=x,p[a[x][i]][y][1][1]=y;
		if(z==1)
			for(int i=0;i<a[y].size();i++)
				if(a[y][i]!=x and dis[x][a[y][i]][0]==-1)
					dis[x][a[y][i]][0]=dis[x][y][1]+1,q.push(x),q.push(a[y][i]),q.push(0),p[x][a[y][i]][0][0]=x,p[x][a[y][i]][0][1]=y;
	}
	cout<<dis[n][1][0]<<endl;
	vector<int>ans1,ans2;
	if(dis[n][1][0]!=-1){
		int x=n,y=1,c=0;
		bool flag=false;
		while(!(x==1 and y==n and c==0)){
			ans1.push_back(x);
			ans2.push_back(y);
			flag=true;
			int u=x,v=y;
			x=p[u][v][c][0];
			y=p[u][v][c][1];
			c=1-c;
			u=x,v=y;
			x=p[u][v][c][0];
			y=p[u][v][c][1];
			c=1-c;
		}
		cout<<1<<' ';
		for(int i=ans1.size()-1;i>-1;i--)
			cout<<ans1[i]<<' ';
		cout<<endl;
		cout<<n<<' ';
		for(int i=ans2.size()-1;i>-1;i--)
			cout<<ans2[i]<<' ';
	}
	return 0;
}
