#include<iostream>
#include<algorithm>
#include<queue>
#include<vector>
using namespace std;

#define N 505

typedef pair<int,int>P2;

vector<int>ans[2];

int p[N][N][2];

queue<P2>q;

bool vis[N][N];

int n,m;

vector<int>g[N];

void bfs(){


    q.push(P2(1,n));

    int cas=1;

    while(!q.empty())
    {
        P2 t=q.front();
        q.pop();
        int x=t.first,y=t.second;

        vis[x][y]=1;
        for(int i=0;i<g[x].size();i++)
           for(int j=0;j<g[y].size();j++)
           {
               int a=g[x][i],b=g[y][j];
               if(vis[a][b]||a==b) continue;

               p[a][b][0]=x;
               p[a][b][1]=y;
               vis[a][b]=1;
               if(a==n&&b==1) return ;
               q.push(P2(a,b));

           }

    }

}
int main()
{


    cin>>n>>m;

    for(int i=0;i<m;i++)
    {
        int a,b;
        cin>>a>>b;
        g[a].push_back(b);
        g[b].push_back(a);
    }

    bfs();

    if(!p[n][1][0])  {cout<<-1<<endl;return 0;}

    int x,y,z;
    x=n,y=1;

    while(p[x][y][0])
    {
        ans[0].push_back(x);
        ans[1].push_back(y);
        z=y;
        y=p[x][y][1];
        x=p[x][z][0];

    }

    cout<<ans[0].size()<<endl;

    for(int i=0;i<ans[0].size();i++) cout<<ans[1][i]<<" ";
    cout<<n;
    cout<<endl;
    for(int i=0;i<ans[1].size();i++) cout<<ans[0][i]<<" ";
    cout<<1;
    cout<<endl;



}
