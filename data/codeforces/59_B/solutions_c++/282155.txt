#include <cstdio>

int main (int argc, char const* argv[]) {
	
	int n, tot = 0, k, minodd = 1<<30;
	
	scanf("%d",&n);
	while(n--) {
		scanf("%d",&k);
		tot += k;
		if(k&1&&k<minodd)
			minodd = k;
	}
	
	if(!(tot&1)&&minodd==1<<30)
		printf("0\n");
	else
		printf("%d\n",tot&1?tot:tot-minodd);
	
	return 0;
}
