count = int(raw_input(''))
strCounts = raw_input('')

lst = strCounts.split(' ')
lst = [int(lst[x]) for x in range(count)]

summ = 0

for x in range(0,count):
    summ += lst[x]

lst.sort()

if summ % 2 == 0:
    for x in range(0,count):
        if lst[x] % 2 != 0:
            summ -= lst[x]
            break

if summ % 2 != 0:
    print summ
else:
    print '0'
