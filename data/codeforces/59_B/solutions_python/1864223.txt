import sys
import math

file = sys.stdin
cnt = int(file.readline().rstrip())
p = map(int, file.readline().rstrip().split())
even = [x for x in p if x%2 == 0]
odd = [x for x in p if x%2 == 1]
if len(odd) == 0:
    print "0"
else:
    if len(odd)%2 == 0:
        odd.remove(min(odd))
    print sum(even) + sum(odd)
