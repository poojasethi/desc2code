def main():
    x = int(input())
    line = list(map(int, raw_input().split()))
    result = sum(line)
    if result % 2 == 0:
        line = list(filter(filt, line))
        line.sort()
        if len(line) == 0:
            result = 0
        else:
            result = result - line[0]
    print(result)

def filt(x):
    return x % 2 == 1

if __name__ == '__main__':
    main()
