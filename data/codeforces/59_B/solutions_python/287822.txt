#!/usr/bin/python

import sys, os
sys.setrecursionlimit(10000)

def readrow():
	return sys.stdin.readline().strip().split(' ')

(n) = map(int, readrow())
lb = 0
ps = []
for p in map(int, readrow()):
	if (p & 1) == 0:
		lb += p
	else:
		ps.append(p)
ps.sort(reverse=True)
lb += sum(ps) if (len(ps) % 2) == 1 else sum(ps[:-1])
if lb % 2 == 0: lb = 0
print(lb)
