from __future__ import division

def read(func=None):
    return func(raw_input()) if func else raw_input()

def reads(func=None):
    return map(func, raw_input().split()) if func else raw_input().split()

################################

n = read(int)
a = reads(int)

odd = [val for val in a if val % 2 == 1]
even = [val for val in a if val % 2 == 0]

if not odd:
    print 0
else:
    s = sum(odd)
    if len(odd) % 2 == 0:
        s -= min(odd)
    print s + sum(even)
