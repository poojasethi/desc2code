
def calc(xs):
    ans = 0
    odd = []
    for x in xs:
        if x % 2 == 0:
            ans += x
        else:
            odd.append(x)
            
    if len(odd) == 0: return 0
    odd.sort()
    ans += odd.pop()
    while len(odd) >= 2:
        ans += odd.pop() + odd.pop()
    return ans

def main():
    raw_input()
    print calc(map(int, raw_input().split()))

if __name__ == "__main__":
    main()
