from itertools import *

a = raw_input()
a = map(int, raw_input().split())
counter = 0

cht = sorted([i for i in a if i % 2 == 0])
ncht = sorted([i for i in a if i % 2 == 1])
max_ncht = len(ncht) - (1 - len(ncht) % 2)
if max_ncht != len(ncht) and max_ncht > 0:
    counter += sum(cht) + sum(ncht[1:])
elif max_ncht > 0:
    counter += sum(cht) + sum(ncht)
print counter
