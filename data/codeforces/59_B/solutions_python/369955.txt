import sys

sys.stdin.readline()

numbers = [int(x) for x in sys.stdin.readline().strip("\n").split(" ")]

s = sum(numbers)

even = [x for x in numbers if x % 2 == 0]
odd = [x for x in numbers if x % 2 == 1]

even.sort()
odd.sort()

if s % 2 == 1:
    print s
else:
    if len(odd) == 0:
        print 0
    else:
        print s - odd[0]
