
def b(petals):
  res = 0
  petals = list(reversed(sorted(petals)))
  evens = [p for p in petals if p % 2 == 0]
  odds = [p for p in petals if p % 2 == 1]
  if len(odds) == 0:
    return 0
  ans = sum(evens)
  for x in odds:
    ans += x
    if ans % 2 == 1:
      res = ans

  return res
