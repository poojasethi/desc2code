import sys

def main():
    sys.stdin.readline()

    even = []
    odd = []
    for x in map(int, sys.stdin.readline().split()):
        if x % 2 == 0:
            even.append(x)
        else:
            odd.append(x)

    if len(odd) == 0:
        print (0)
        return

    ans = sum(even)
    odd.sort()
    ans += odd.pop()
    while len(odd) >= 2:
        ans += odd.pop() + odd.pop()

    print(ans)

main()
