#!/usr/bin/python

from collections import deque

def ia():
    line = raw_input()
    return line.split()

n, m, p = ia(); n, m = int(n), int(m)

M = []
for i in range(n):
    M.append(raw_input())

ddx = [0, 0, 1, -1]
ddy = [1, -1, 0, 0]

def valid(i, j):
    return i>=0 and i<n and j>=0 and j<m

s = set()    
for i in range(n):
    for j in range(m):
        if M[i][j]!=p: continue
        for dx, dy in zip(ddx, ddy):
            ip = i + dx; jp = j + dy
            if not valid(ip, jp): continue
            if M[ip][jp]==p     : continue
            if M[ip][jp]=='.'   : continue
            s.add(M[ip][jp])
print len(s)
