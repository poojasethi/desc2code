n, m, p = raw_input().split()
n = int(n)
m = int(m)

office = [raw_input().strip() for i in range(n)]

i,j = 0,0
deputies = []

while i < n:
    while j < m:
        if office[i][j] != p and office[i][j] != '.':
            if (i != 0 and office[i-1][j] == p) or (j != 0 and office[i][j-1] == p) or (i != n-1 and office[i+1][j] == p) or (j != m-1 and office[i][j+1] == p):
                deputies.append(office[i][j])
        j += 1
    i += 1
    j = 0

print len(set(deputies))
