#!/usr/bin/env python
# -*- coding: utf-8 -*-

import sys

def main():
    n, m, p = sys.stdin.readline().split()
    n, m = int(n), int(m)
    f = [[e for e in s[:-1]] for s in sys.stdin.readlines()]
    c = set()
    for i in xrange(n):
        for j in xrange(m):
            if f[i][j] == p:
                if i > 0 and f[i-1][j] != '.':
                    c.add(f[i-1][j])
                if i < n-1 and f[i+1][j] != '.':
                    c.add(f[i+1][j])
                if j > 0 and f[i][j-1] != '.':
                    c.add(f[i][j-1])
                if j < m-1 and f[i][j+1] != '.':
                    c.add(f[i][j+1])
    print len(c)-1 if p in c else len(c)
    return 0

if __name__ == '__main__':
    main()
