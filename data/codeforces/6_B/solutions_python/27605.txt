#!/usr/bin/env python

from sys import stdin

N, M, C = stdin.readline().split()
N = int(N)
M = int(M)

B = [ ]
B.append( '.' * (M+2) )
for r in xrange(N):
   B.append('.' + stdin.readline().strip() + '.')
B.append( '.' * (M+2) )


dir = [(-1,0), (0,-1), (0,+1), (+1,0)]

S = ''
for r in xrange(1, N+1):
   for c in xrange(1, M+1):
      if B[r][c] != C: continue
      for d in dir:
         nr = r+d[0]
         nc = c+d[1]
         ch = B[nr][nc]
         if ch.isalpha() and ch != C and ch not in S:
            S += ch

print len(S)
