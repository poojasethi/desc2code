#!/usr/bin/env python
import string

input = string.split(raw_input())
m = int(input[0])
n = int(input[1])
color = input[2]

field = []
for i in range(m):
	field.append(raw_input())

tables = []
for i in range(m):
	for j in range(n):
		if field[i][j] == color:
			# Vverh
			if i + 1 < m:
				if (field[i + 1][j] != '.') and (field[i + 1][j] != color) and ((field[i + 1][j] in tables) == False):
					tables.append(field[i + 1][j])
			# Niz
			if i - 1 >= 0:
				if (field[i - 1][j] != '.') and (field[i - 1][j] != color) and ((field[i - 1][j] in tables) == False):
					tables.append(field[i - 1][j])

			# Levo
			if j - 1 >= 0:
				if (field[i][j - 1] != '.') and (field[i][j - 1] != color) and ((field[i][j - 1] in tables) == False):
					tables.append(field[i][j - 1])

			# Pravo
			if j + 1 < n:
				if (field[i][j + 1] != '.') and (field[i][j + 1] != color) and ((field[i][j + 1] in tables) == False):
					tables.append(field[i][j + 1])

print len(tables)
