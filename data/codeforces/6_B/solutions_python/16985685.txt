n, m, c = raw_input().split()
n, m = int(n), int(m)
grid = []
for i in xrange(n): grid.append(raw_input())
vis = [0] * 26
dx = [0, 1, 0, -1]
dy = [-1, 0, 1, 0]
for i in xrange(n):
	for j in xrange(m):
		if c == grid[i][j]:
			for k in xrange(4):
				iNext = i + dy[k]
				jNext = j + dx[k]
				if 0 <= iNext < n and 0 <= jNext < m and grid[iNext][jNext] != '.' and grid[iNext][jNext] != c:
					vis[ord(grid[iNext][jNext]) - ord('A')] = 1
print sum(vis)
