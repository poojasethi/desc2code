d=[(0,1),(1,0),(0,-1),(-1,0)]
inside=lambda x,y: 0<=x<n and 0<=y<m

n,m,c=raw_input().split()
n=int(n)
m=int(m)
buf=[raw_input() for _ in xrange(n)]
s=set()
s.add(c)
for x in xrange(n):
	for y in xrange(m):
		if buf[x][y]==c:
			for dx,dy in d:
				nx, ny = x+dx, y+dy
				if inside(nx,ny) and buf[nx][ny]!='.':
					s.add(buf[nx][ny])
print len(s)-1
