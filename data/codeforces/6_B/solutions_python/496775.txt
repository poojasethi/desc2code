import sys
def OK(r,c):
	global row,col
	if (r>=0 and r<row and c>=0 and c<col):return True
	return False
row,col,ch=raw_input().split()
row=int(row)
col=int(col)
G=[]
for i in range(row):
	G.append(raw_input())
Dir=[(0,1),(1,0),(0,-1),(-1,0)]
vis=[]
for r in range(row):
	for c in range(col):
		if(G[r][c]!=ch):continue
		for d in Dir:
			x=c+d[0]
			y=r+d[1]
			if OK(y,x) and G[y][x].isalpha() and G[y][x]!=ch and G[y][x] not in vis:
				vis.append(G[y][x])
print len(vis)
