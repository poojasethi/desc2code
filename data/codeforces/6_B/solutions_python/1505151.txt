n, m, c = raw_input().split()
n, m = int(n), int(m)
room = [raw_input() for _ in xrange(n)]

def get_color(y, x):
    if 0 <= x < m and 0 <= y < n:
        return room[y][x]
    return '.'

was = set()
for y in xrange(n):
    for x in xrange(m):
        if room[y][x] == c:
            was.add(get_color(y - 1, x))
            was.add(get_color(y + 1, x))
            was.add(get_color(y, x - 1))
            was.add(get_color(y, x + 1))

was -= set(('.', c))
print len(was)
