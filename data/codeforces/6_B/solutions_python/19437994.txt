R,C,Co=raw_input().strip().split()
R,C=int(R),int(C)
Inp=[list(raw_input().strip()) for i in xrange(R)]
S=set()
for i in xrange(R):
	for j in xrange(C):
		if Inp[i][j]==Co:
			for x,y in [(-1,0),(1,0),(0,-1),(0,1)]:
				if -1<i+x<R and -1<j+y<C and Inp[i+x][j+y]!='.' and Inp[i+x][j+y]!=Co:
					S.add(Inp[i+x][j+y])
print len(S)
