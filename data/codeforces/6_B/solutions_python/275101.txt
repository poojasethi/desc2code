n, m, color = map (lambda x: int (x) if x.isdigit () else x, raw_input ().split ())
colors, ans = [raw_input () for _ in xrange (n)], set ()
for i in xrange (n):
    for j in xrange (m):
        if colors[i][j] == color:
            if i + 1 < n: ans.add (colors[i + 1][j])
            if i - 1 >= 0: ans.add (colors[i - 1][j])
            if j + 1 < m: ans.add (colors[i][j + 1])
            if j - 1 >= 0: ans.add (colors[i][j - 1])
print len (ans - set (['.', color]))
