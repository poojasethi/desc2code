def main():
  m, n, c = raw_input().split()
  m, n = int(m), int(n)
  index = set()
  index.add(c)
  index.add('.')
  last = ['.'] * n
  for x in xrange(0, m):
	cur = raw_input().strip()
	for i in xrange(0, n):
	  if cur[i] == c:
		index.add(last[i])
		if i+1<n:
		  index.add(cur[i+1])
		if i > 0:
		  index.add(cur[i-1])
	  elif last[i] == c:
		index.add(cur[i])
	last = cur
  print len(index) - 2

if __name__ == '__main__':
  main()
