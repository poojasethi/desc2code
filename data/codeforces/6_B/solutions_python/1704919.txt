#!/usr/bin/python
import sys

num = ''

s = sys.stdin.readline()
ar = s.split()
max_n = ar[0]
max_m = ar[1]
owner= ar[2].strip()
table=[]

ho = []

for n,s in enumerate(sys.stdin):
    s.strip()
    table.append(s.strip())
    if owner in s:
        l = s.count(owner)
        for t in range(l):
            tm = s.index(owner) +t
            ho.append( (n,tm) )

            if n>0:
                num += table[n-1][tm].strip('.')
            if tm>0:
                num += s[tm-1].strip('.')
            if tm<max_m:
                num += s[tm+1].strip('.')
    if ho:
        for tu in ho:
            if tu[0]+1 == n:
                num += s[tu[1]].strip('.')

num = set(num.strip(''))
if '\n' in num:
    num.remove('\n')
if owner in num:
    num.remove(owner)

print '%d'%( len(num) )
