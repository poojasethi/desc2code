#!/usr/bin/env python
n, m, c = raw_input().split()
n = int(n)
m = int(m)
a = [' '*(m+2)]
f = set()
for i in range(1, n+1):
    a.append(' ' + raw_input() + ' ')
a.append(' '*(m+2))
for i in range(1, n+1):
    k = -1
    while c in a[i][k+1:]:
        k = a[i].find(c, k+1)
        f.add(a[i][k-1])
        f.add(a[i][k+1])
        f.add(a[i-1][k])
        f.add(a[i+1][k])
f.discard('.')
f.discard(c)
f.discard(' ')
print len(f)
