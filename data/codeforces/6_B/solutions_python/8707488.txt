n, m, c = raw_input().split()
n, m = map(int, [n, m])
a = [raw_input() for i in xrange(n)]

s = set(['.', c])
for i in xrange(n):
    for j in xrange(m):
        if a[i][j] == c:
            for x, y in [(i - 1, j), (i + 1, j), (i, j - 1), (i, j + 1)]:
                if x >= 0 and x < n and y >= 0 and y < m: s.add(a[x][y])

print len(s) - 2
