
n,m,c = raw_input().split()

a=[]
for i in range(int(n)):
	a.append(raw_input())

i=0
while a[i].find(c) == -1:
	i+=1
j=int(n)-1
while a[j].find(c) ==-1:
	j-=1
r={}
e1=a[i].find(c)
e2=a[i].rfind(c)
if i!=0:
	for k in a[i-1][e1:e2+1]:
		if k!='.':
			r[k]=0
if j!=int(n)-1:
	for k in a[j+1][e1:e2+1]:
		if k!='.':
			r[k]=0
if e1>0:
	for k in range(i,j+1):
		p = a[k][e1-1]
		if p!='.':
			r[p]=0
if e2< int(m)-1:
	for k in range(i,j+1):
		p = a[k][e2+1]
		if p!='.':
			r[p]=0
print len(r)
