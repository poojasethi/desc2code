def addCol(y,x):
 if y<0 or x<0 or y>=n or x>=m: return
 rez[pl[y][x]]=1

(sn,sm,c)=raw_input().split()
n=int(sn)
m=int(sm)
pl=[]
for i in range(n):
 pl.append(raw_input())

rez={}
for i in range(n):
 for j in range(m):
  if pl[i][j]==c:
   addCol(i-1,j) 
   addCol(i+1,j) 
   addCol(i,j-1) 
   addCol(i,j+1) 

otv=0
for tm in rez.keys():
 if tm!=c and tm!=".": otv+=1

print otv
