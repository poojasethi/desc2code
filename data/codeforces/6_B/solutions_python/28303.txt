n, m, ch = raw_input().split()
n = int(n)
m = int(m)

dat = []
for i in range(n):
    dat.append(raw_input())

#print dat

dist = [(-1,0),(1,0),(0,-1),(0,1)]
dep = []


for i in range(n):
    for j in range(m):
        #print i, j, dat[i][j]
        if dat[i][j] == ch:
            for dx, dy in dist:
                if i+dx in range(n) and j+dy in range(m):
                    dep.append(dat[i+dx][j+dy])
dep = set(dep)
for c in [ch, "."]:
    if c in dep:
        dep.remove(c)

print len(dep)
