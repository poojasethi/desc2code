dx = [1, 0,  0, -1]
dy = [0, 1, -1,  0] 
   
n, m, c = raw_input().split()
n, m = int(n), int(m)

se = set()
ma = [raw_input() for i in range(n)]

for i in range(n):
   for j in range(m):
      for k in range(4):
         x = i + dx[k]
         y = j + dy[k]
         if(x < 0 or x >= n or y < 0 or y >= m or ma[x][y] == '.'):
            continue
         if(ma[i][j] == c and ma[x][y] != c):  
            se.add(ma[x][y])
            
print len(se)
