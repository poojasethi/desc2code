#include <iostream>
using namespace std;
int n,m,a[256],i,j,s;
char q,b[111][111];
int main()
{
	cin>>n>>m>>q;
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			cin>>b[i][j];
		}
	}
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			if (b[i][j]==q){
				a[b[i-1][j]]=1;
				a[b[i+1][j]]=1;
				a[b[i][j-1]]=1;
				a[b[i][j+1]]=1;
			}
		}
	}
	for(i='A';i<='Z';i++)
		s=s+a[i];
	cout<<s-a[q];
	return 0;
}
