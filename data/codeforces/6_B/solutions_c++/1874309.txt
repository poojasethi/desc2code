#include <iostream>
#include <set>

using namespace std;

const int MAXN = 100+10;

char table[MAXN][MAXN], ch;
int n, m;
set <char> s;

int main()
{
	cin >> n >> m >> ch;
	for (int i = 1 ; i <= n ; i++)
		for (int j = 1 ; j <= m ; j++)
			cin >> table[i][j];
	for (int i = 1 ; i <= n ; i++)
		for (int j = 1 ; j <= m ; j++)
			if (table[i][j] != '.' && (table[i-1][j] == ch || table[i+1][j] == ch || table[i][j-1] == ch || table[i][j+1] == ch) && table[i][j] != ch)
				s.insert(table[i][j]);
	cout << s.size() << endl;
	return 0;
}
