#include<iostream>
using namespace std;
int n,m,a[256],sum,i,j;
char c,b[110][110];
int main()
{
    cin>>n>>m>>c;
    for(i=1;i<=n;i++)
    for(j=1;j<=m;j++)
    cin>>b[i][j];
    for(i=1;i<=n;i++)
    for(j=1;j<=m;j++)
    if(b[i][j]==c)a[b[i-1][j]]=a[b[i+1][j]]=a[b[i][j-1]]=a[b[i][j+1]]=1;
    for(i='A';i<='Z';i++)sum+=a[i];
    cout<<sum-a[c];
    return 0;
}
