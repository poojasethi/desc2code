#include <iostream>
using namespace std;
int n,m,b[256],s;
char a[110][110],c;
int main()
{
    cin>>n>>m>>c;
    for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
    cin>>a[i][j];
    for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
    if(a[i][j]==c)b[a[i][j-1]]=b[a[i][j+1]]=b[a[i-1][j]]=b[a[i+1][j]]=1;
    for(int i='A';i<='Z';i++)
    s+=b[i];
    cout<<s-b[c];
    return 0;
}
