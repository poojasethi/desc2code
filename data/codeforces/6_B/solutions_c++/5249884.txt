#include <iostream>
//#include <string>
using namespace std;
char a[201][201];
bool b[201]={false};
int main ( ) {
	int n,m,i,j,s=0;
	char t;
	cin>>n>>m>>t;
	for (i=1;i<=n;i++) {
		for (j=1;j<=m;j++) cin>>a[i][j];
	}
	for (i=1;i<=n;i++) {
		for (j=1;j<=m;j++) {
			if (a[i][j]==t) b[a[i-1][j]]=b[a[i][j-1]]=b[a[i+1][j]]=b[a[i][j+1]]=true;
		}
	}
	for (i=65;i<=90;i++) {
		if (b[i]==true && i!=t) s++;
	}
	cout<<s<<endl;
	return 0;
}
