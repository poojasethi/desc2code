#include<iostream>
#include<cstring>
#include<string>
#include<set>
using namespace std;
string dat[100];
int n,m;char a;
set<char> se;
int main()
{
	cin>>n>>m;
	cin>>a;if (a==' ') cin>>a;
	for(int i=0;i<n;i++) cin>>dat[i];
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if (dat[i][j]==a || dat[i][j]=='.') continue;
			else if (dat[max(i-1,0)][j]==a || dat[min(i+1,n-1)][j]==a || dat[i][max(j-1,0)]==a || dat[i][min(j+1,m-1)]==a) se.insert(dat[i][j]);
		}
	}
	cout<<se.size()<<"\n";
	return 0;
}
