#include<iostream>
using namespace std;
int n,m,b[256],s,i,j;
char q,a[110][110];
int main()
{
	cin>>n>>m>>q;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			cin>>a[i][j];
	for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
            if(a[i][j]==q)
                b[a[i-1][j]]=b[a[i+1][j]]=b[a[i][j-1]]=b[a[i][j+1]]=1;
    for(i='A';i<='Z';i++)
        s+=b[i];
        cout<<s-b[q]<<endl;
    return 0;
}
