#include <iostream>
using namespace std;
int N,M;
char T;
char A[200][200];
int K[200];
int yon[4][2]={{0,1},{-1,0},{1,0},{0,-1}};
void read()
{
	cin >> N >> M >> T;
	for(int i=1;i<=N;i++)
		for(int j=1;j<=M;j++)
			cin>>A[i][j];
}
void solve()
{
	for(int i=1;i<=N;i++)
		for(int j=1;j<=M;j++)
			if(A[i][j]==T)
				for(int k=0;k<4;k++) K[A[i+yon[k][0]][j+yon[k][1]]]++;
	K[T]=0;
	int s=0;
	for(int i='A';i<='Z';i++)
		if(K[i]) s++;
	cout << s << endl;
}
int main()
{
	read();
	solve();
	return 0;
}
