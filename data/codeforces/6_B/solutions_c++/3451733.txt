#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int n,m,ans,mark[300];
char p,a[200][200];

int main()
	{
	cin>>n>>m>>p;
	for(int i=1;i<=n;i++) for(int j=1;j<=m;j++) cin>>a[i][j];
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			if(a[i][j]==p) mark[a[i+1][j]]=mark[a[i][j+1]]=mark[a[i-1][j]]=mark[a[i][j-1]]=1;
	for(int i='A';i<='Z';i++) ans+=mark[i];
	cout<<ans-mark[p];
	return 0;
	}
