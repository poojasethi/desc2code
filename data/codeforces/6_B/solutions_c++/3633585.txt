#include <iostream>
#include <string>
using namespace std;
long n,m,i,j,ans=0,b[300]={};
string s[110]={};
char ch;
int main() {
	cin >> n >> m >> ch;
	s[0]=s[n+1]=string(m+2,'.');
	for (i=1;i<=n;i++) {
		cin >> s[i];s[i]="."+s[i]+".";
	}
	for (i=1;i<=n;i++)
		for (j=1;j<=m;j++) 
			if (s[i][j]==ch) {
				b[s[i-1][j]]=b[s[i+1][j]]=b[s[i][j-1]]=b[s[i][j+1]]=1;
			}
	for (i='A';i<='Z';i++) if (i!=ch) ans+=b[i];
	cout << ans << endl;
}
