#include<set>
#include<iostream>
using namespace std;
char a[150][150];
int main()
{
	set<char> s;
	int n,m,i,j;
	char w;
	cin>>n>>m>>w;
	for (i=0;i<n;i++)
	 for (j=0;j<m;j++)
	  cin>>a[i][j];
	s.insert(w);
	s.insert('.');
	for (i=0;i<n;i++)
	 for (j=0;j<m;j++)
	  if (a[i][j]==w)
	  {
	  	if (i!=0)
	  		s.insert(a[i-1][j]);
	  	if (i!=n-1)
	  		s.insert(a[i+1][j]);
	  	if (j!=0)
	  		s.insert(a[i][j-1]);
	  	if (j!=m-1)
	  		s.insert(a[i][j+1]);
	  }
	cout<<(int)s.size()-2;
	return 0;
}
