#include<bits/stdc++.h>
using namespace std;

char s[111][111];
int a[256];
int m,n;
char p;
int ans;

int main()
{
	cin>>n>>m>>p;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			cin>>s[i][j];
		}
	}
	
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			if(s[i][j]==p)
			{
				a[s[i-1][j]]=a[s[i+1][j]]=a[s[i][j-1]]=a[s[i][j+1]]=1;
			}
		}
	}																											
	
	for(int i='A';i<='Z';i++)
	{
		ans+=a[i];
	}
	
	cout<<ans-a[p]<<endl;
	return 0;
}
