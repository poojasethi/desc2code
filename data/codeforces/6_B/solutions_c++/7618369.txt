#include <iostream>
using namespace std;
int adj[128][128];
int dx[4]={1,0,-1,0};
int dy[4]={0,1,0,-1};
char a[150][150];
int main()
{
	int n,m,i,j,k;
	char w;
	cin>>n>>m>>w;
	for (i=0;i<n;i++)
	{
		for (j=0;j<m;j++)
		cin>>a[i][j];
	}
	for (i=0;i<n;i++)
	{
		for (j=0;j<m;j++)
		{
			for (k=0;k<4;k++)
			adj[a[i][j]][a[i+dx[k]][j+dy[k]]]=1;
		}
	}
	int r=0;
	for (i=1;i<128;i++)
	{
	if (adj[w][i]==1 && (char)i!='.' && i!=w)
	r++;
	}
	cout<<r;
	return 0;
}
