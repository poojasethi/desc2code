#include <stdio.h>
#define sz 105
char mat[sz][sz];
bool flag[130];
int main()
{
	int r,c,i,j,k=0;
	char ch;
	scanf(" %d %d %c",&r,&c,&ch);
	for(i=1;i<=r;i++)
		for(j=1;j<=c;j++)
			scanf(" %c",&mat[i][j]);

	for(i=1;i<=r;i++)
		for(j=1;j<=c;j++)
			if(mat[i][j]==ch)
			{
				flag[mat[i-1][j]]=true;
				flag[mat[i+1][j]]=true;
				flag[mat[i][j-1]]=true;
				flag[mat[i][j+1]]=true;
			}

	for(i='A';i<='Z';i++)if(i!=ch)
		if(flag[i])
			k++;

	printf("%d",k);
	return 0;
}
