#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int n,m;
char map[105][105],c;
int vis[300];
int main()
{
	int i,j,cnt=0;
	cin>>n>>m>>c;
	memset(vis,0,sizeof(vis));
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			cin>>map[i][j];
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			if(map[i][j]==c)
			{
				vis[map[i+1][j]]=1;
				vis[map[i-1][j]]=1;
				vis[map[i][j+1]]=1;
				vis[map[i][j-1]]=1;
			}
	vis[c]=0;
	for(i='A';i<='Z';i++)
		if(vis[i])
			cnt+=vis[i];
	printf("%d\n",cnt);
}
