#include <stdio.h>

int adj[128][128];

int dx[4]={0, 1, 0, -1}, dy[4] = {1, 0, -1, 0};

char a[150][150];

int main(void) {
	int m, n, i, j, k;
	char w[10];
	scanf("%d%d%s", &m, &n, w);
	for(i=1;i<=m;i++)
		scanf("%s", a[i]+1);
	for(i=1;i<=m;i++){
		for(j=1;j<=n;j++)
			for(k=0;k<4;k++)
				adj[a[i][j]][a[i+dx[k]][j+dy[k]]]=1;
	}
	int cnt=0;
	for(i=1;i<128;i++)
		if(adj[w[0]][i] && (char)i!='.' && i != w[0]) {
			++cnt;
		}
	printf("%d\n", cnt);
}
