#include<iostream>
using namespace std;

int main()
{
	int n,m,v[30];
	char x,y[100][100];
	cin>>n>>m;
	cin>>x;
	v[x-'A']=1;
	for(int i=0;i<n;i++)
	   //for(int j=0;j<m;j++)
	    cin>>y[i];
	for(int i=0;i<n;i++)
	   for(int j=0;j<m;j++)
	      if(y[i][j]==x)
	      {
	      	if(i>0) v[y[i-1][j]-'A']=1;
	      	if(j>0) v[y[i][j-1]-'A']=1;
	      	if(i<n-1) v[y[i+1][j]-'A']=1;
	      	if(j<m-1) v[y[i][j+1]-'A']=1;
	      }
	int ans=-1;
	for(int i=0;i<26;i++)
	   ans+=v[i];
	cout<<ans;
	return 0;
}
