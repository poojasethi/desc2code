#include <iostream>
#include <map>
#include <cctype>

#define Ctrl(x,y) if(isalpha(Off[x][y]) and Off[x][y]!=c and !M[Off[x][y]])  M[Off[x][y]]=1,res++; 

using namespace std;

map < char , int > M;

char c,Off[105][105];

int main ()
{
	int m,n,i,j,res=0;
	cin >> m >> n >> c;
	for(i=1; i<=m; i++)
		for(j=1; j<=n; j++)
			cin >> Off[i][j];
	for(i=1; i<=m; i++)
		for(j=1; j<=n; j++)
		{
			if(Off[i][j] == c)
			{
				Ctrl(i+1,j)
				Ctrl(i-1,j)
				Ctrl(i,j+1)
				Ctrl(i,j-1)
			}
		}
	cout << res << endl;
}
