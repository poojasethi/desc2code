#include<iostream>
#include<set>
using namespace std;
set<char>s;
int main()
{
	int n,m;
	char c,a[105][105];
	cin>>n>>m>>c;
	for (int i=0;i<n;i++)
	for (int j=0;j<m;j++)
	cin>>a[i][j];
	for (int i=0;i<n;i++)
	for (int j=0;j<m;j++)
	if (a[i][j]==c)
	for (int mp=-1;mp<=1;mp++)
	for (int ad=-1;ad<=1;ad++)
	{
		if (mp && ad) continue;
		int x=i+mp;
		int y=j+ad;
		if (x<0 || y<0 || x>=n || y>=m)
		continue;
		if (a[x][y]!=c && a[x][y]!='.')
		s.insert(a[x][y]);
	}
	cout<<(int)s.size();
	return 0;
}
