#include <iostream>
#include <algorithm>
using namespace std;
const int N=100100;
typedef pair<int,int> PII;
#define A first
#define B second
PII a[N],b[N],h[N];
int n,m,s,c[N],f[N];
char go(int t){
  int p=0;
  for(int i=m,j=n,r=s;i>0;){
    while(j>0&&b[j-1].A>=a[i-1].A)--j,h[p++]=PII(-c[b[j].B],b[j].B),push_heap(h,h+p);
    if(p<1)return 0;
    if((r+=h[0].A)<0)return 0;else for(int u=t;u-->0&&i-->0;)f[a[i].B]=h[0].B;
    pop_heap(h,h+p--);
  }
  return 1;
}
int main(){
  cin>>n>>m>>s;
  for(int i=0;i<m;++i)cin>>a[i].A,a[i].B=i;
  for(int i=0;i<n;++i)cin>>b[i].A,b[i].B=i;
  for(int i=0;i<n;++i)cin>>c[i];
  sort(a,a+m),sort(b,b+n);
  int lo=0,hi=m+1;
  while(lo<hi){
    int md=(lo+hi)/2;
    if(go(md))hi=md;else lo=md+1;
  }
  if(lo>m)cout<<"NO"<<endl;
  else{
    cout<<"YES"<<endl;
    go(lo);
    for(int i=0;i<m;++i)cout<<(f[i]+1)<<" ";cout<<endl;
  }
}
