#include <bits/stdc++.h>
using namespace std;

const int maxn=200010;

struct NN
{
    int ab,pri,ind;

    bool operator <(const NN &n)const
    {
	return pri>n.pri;
    }
}nn[maxn];

int cmp(NN a,NN b)
{
    if(a.ab==b.ab)
	return a.pri>b.pri;
    return a.ab>b.ab;
}
priority_queue<NN>q;
int out[maxn];
long long jud(int n,int mid)
{
    while(!q.empty())
	q.pop();
    long long ans=0;
    int i,j,now=0,ind;

    for(i=1;i<=n;i++)
    {
	if(nn[i].pri<0)
	{
	    if(now)
		now--,out[nn[i].ind]=ind;
	    else if(q.empty())
		return 2100000000;
	    else
		now=mid-1,ans+=q.top().pri,ind=q.top().ind,out[nn[i].ind]=ind,q.pop();
	}
	else
	    q.push(nn[i]);
    }
    return ans;
}
int main()
{
    int n,m,i,j,k,s;

    cin>>n>>m>>s;

    for(i=1;i<=m;i++)
	scanf("%d",&nn[i].ab),nn[i].pri=-1,nn[i].ind=i;
    for(i=1;i<=n;i++)
	scanf("%d",&nn[i+m].ab),nn[i+m].ind=i;
    for(i=1;i<=n;i++)
	scanf("%d",&nn[i+m].pri);
    sort(nn+1,nn+1+m+n,cmp);

    int l=1,r=m,mid;

    while(l<r)
    {
	mid=(l+r)/2;
	if(jud(n+m,mid)>s)
	    l=mid+1;
	else
	    r=mid;
    }
    if(l<=m&&jud(n+m,l)<=s)
    {
	jud(n+m,l);
	puts("YES");
	for(i=1;i<=m;i++)
	    printf("%d ",out[i]);
	puts("");
    }
    else
	puts("NO");
}
