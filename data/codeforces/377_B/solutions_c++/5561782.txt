#include <bits/stdc++.h>
using namespace std;

#define ll long long

pair<pair<ll,ll>,ll> a[200005];
pair<ll,ll> b[200005];
ll n,m;
ll s;
ll tm1[200005];

bool go(ll mi) {
  ll p = 0;  
  ll an = n-1, bn = m-1;
  set<pair<ll,ll> > d;
  while(bn>=0) {
    while(an>=0 && a[an].first.first>=b[bn].first) {
      d.insert(make_pair(a[an].first.second,a[an].second));
      an--;
    }
    if(d.empty()) return false;
    set<pair<ll,ll> >::iterator it = d.begin();
    p+=it->first;
    for(int i=0; i<min(mi,bn+1); i++) {
      tm1[b[bn-i].second] = it->second;
    }
    bn -= min(bn+1,mi);
    d.erase(it);
  }
  if(p>s) return false;
  return true;
}


int main(){ 
  ios_base::sync_with_stdio(0);
  cin>>n;
  cin>>m;
  cin>>s;
  for(int i=0; i<m; i++) {
    cin>>b[i].first;
    b[i].second = i;
  }
  for(int i=0; i<n; i++) {
    cin>>a[i].first.first;
  }
  for(int i=0; i<n; i++) {
    cin>>a[i].first.second;
    a[i].second = i+1;
  }
  sort(a,a+n);
  sort(b,b+m);
  ll lo = 0, hi = m,mi;
  while(hi>lo) {
    mi = (lo+hi)/2;
    if(go(mi)) hi = mi;
    else lo = mi+1;
  }
  if(!go(lo)) {
    cout<<"NO"<<endl;
  }
  else {
    cout<<"YES"<<endl;
    for(int i=0; i<m; i++) {
      cout<<tm1[i]<<" ";
    }
    cout<<endl;
  }
  return 0;
}
