#include<bits/stdc++.h>
#define skill first
#define cost second.first
#define num second.second
using namespace std;
typedef pair<int, pair<int,int> > tt;
int n,m,s,ans[111111];
pair<int,int> p[111111];
tt a[111111];
bool f(int t);
bool f(int& l,int &r){
    int z=(l+r)/2;
    if(f(z)) r=z;
        else l=z;
}

bool f(int t){
    if(!t) return false;
    int all=0,j=n-1;
    priority_queue<pair<int,int> > q;
    for(int i=m-1;i+1;){
        while(j+1){
            if(a[j].skill<p[i].first) break;
            q.push(make_pair( -a[j].cost,a[j].num ));
            j--;
        }
        if(q.empty()) return false;
        all-=q.top().first;
        if(all>s) return false;
        int T=t;
        while(T && i+1){
            ans[ p[i].second ]=q.top().second;
            i--; T--;
        }
        q.pop();
    }
    return true;
}

int main(){
    //freopen("1.txt","r",stdin);
    scanf("%d%d%d",&n,&m,&s);
    for(int i=0;i<m;i++) scanf("%d",&p[i].first), p[i].second=i;
    for(int i=0;i<n;i++) scanf("%d",&a[i].skill);
    for(int i=0;i<n;i++) scanf("%d",&a[i].cost), a[i].num=i+1;
    sort(p,p+m); sort(a,a+n);
    int l=1,r=m;
    while(l+1<r) f(l,r);
    if(!f(r)){
        printf("NO");
        return 0;
    }
    while(!f(l)) l++;
    printf("YES\n");
    for(int i=0;i<m;i++) printf("%d ",ans[i]);
    return 0;
}
