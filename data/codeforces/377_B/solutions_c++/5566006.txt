#include<iostream>
#include<set>
#include<algorithm>
#define SIZE 100100
using namespace std;
int ans[SIZE];
pair<int,int> bug[SIZE];
pair<int, pair<int,int> > stud[SIZE];
int n,m,s;
bool C(int x);
int main()
{
	cin>>n>>m>>s;
	for(int i=0;i<m;i++)
	{
		cin>>bug[i].first; bug[i].second = i;
	}
	for(int i=0;i<n;i++)
	{
		cin>>stud[i].first;
	}
	for(int i=0;i<n;i++)
	{
		cin>>stud[i].second.first;
		stud[i].second.second = i;
	}
	sort(bug,bug+m);
	sort(stud,stud+n);
	int low = 1,high = m;
	int sol = 1e9;
	while(low<=high)
	{
		
		int mid = (low+high)>>1;
		if(C(mid)){high = mid-1; sol = min(sol,mid);}
		else low = mid+1;
	}
	if(sol==1e9) cout<<"NO"<<endl;
	else
	{
		C(sol);
		cout<<"YES"<<endl;
		for(int i=0;i<m;i++)
		{
		  cout<<ans[i]; if(i==m-1)cout<<endl; else cout<<' ';
		}
	}
	return 0;
}

bool C(int x)
{
	long long sum = 0;
	set< pair<int,int> > ss;	
	for(int i=m-1,j=n-1;i>-1;i-=x)
	{
		while(j>=0 && stud[j].first >= bug[i].first)
		{
			ss.insert(stud[j].second);
			j--;
		}
		if(ss.empty())return 0;
		int t =ss.begin()->second;
		sum+= ss.begin()->first;
		ss.erase(ss.begin());
		for(int k=0 ;k<x ;k++)
		{
			if(i-k>=0)
			ans[ bug[i-k].second] =t+1; 
		}
	}
	return sum<= (long long)s;
}
