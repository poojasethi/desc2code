#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
#include<queue>
using namespace std;
struct data{
	int a,b,c;
}a[210001],b[210001];
int i,j,k,n,m,l,r,mid,ans[210001],lim;
struct hcmp{
	bool operator() (int x,int y){
		return b[x].b>b[y].b;
	}
};
priority_queue<int,vector<int>,hcmp> q;
bool check(int x){
	int i,j,k,t,tot=0;
	while (!q.empty()) q.pop();
	for (i=j=1;i<=m;i+=x){
		for (;j<=n&&b[j].a>=a[i].a;j++) q.push(j);
		if (q.empty()) return 0;
		t=q.top();q.pop();tot+=b[t].b;
		if (tot>lim) return 0;
		for (k=i;k<i+x;k++) ans[a[k].b]=b[t].c;
	}
	return 1;
}
bool cmp(data a,data b){
	return a.a>b.a;
}
int main(){
	scanf("%d%d%d",&n,&m,&lim);
	for (i=1;i<=m;i++) scanf("%d",&a[i].a),a[i].b=i;
	for (i=1;i<=n;i++) scanf("%d",&b[i].a),b[i].c=i;
	for (i=1;i<=n;i++) scanf("%d",&b[i].b);
	sort(a+1,a+1+m,cmp);sort(b+1,b+1+n,cmp);
	if (!check(m)){
		puts("NO");return 0;
	}
	puts("YES");
	for (l=1,r=m;l<r-1;){
		mid=(l+r)>>1;
		if (check(mid)) r=mid;else l=mid+1;
	}
	if (!check(l)) check(r);
	for (i=1;i<=m;i++) printf("%d ",ans[i]);
	puts("");
	return 0;
}
