#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <queue>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int b,c,id;
} a[N],b[N];
int i,j,k,n,m,s;
int num[N];
inline bool cc1(const ww &a,const ww &b) {
	return a.b<b.b;
}
inline bool operator < (const ww &a,const ww &b) {
	return a.c>b.c;
}
priority_queue <ww> q;
inline void fail() {
	printf("NO\n");
	//for(;;);
	exit(0);
}
inline bool judge(int x) {
	int i,j,k=s,u;
	for(;!q.empty();) q.pop();
	for (i=m,j=n;i;) {
		for (;j&&b[j].b>=a[i].b;j--) q.push(b[j]);
		if (q.empty()) return 0;
		ww A=q.top();
		if (k>=A.c) {
			k-=A.c;
			for (u=1;u<=x&&i;u++,i--) num[a[i].id]=A.id;
			q.pop();
		} else return 0;
	} return 1;
}
int main() {
	scanf("%d%d%d",&n,&m,&s);
	For(i,1,m) scanf("%d",&a[i].b),a[i].id=i;
	For(i,1,n) scanf("%d",&b[i].b),b[i].id=i;
	For(i,1,n) scanf("%d",&b[i].c);
	sort(a+1,a+m+1,cc1);
	sort(b+1,b+n+1,cc1);
	int F=0;
	For(i,1,n) if (b[i].b>=a[m].b&&b[i].c<=s) F=1;
	if (!F) fail();
	int l=1,r=m,mid;
	for (;l<=r;) {
		mid=(l+r)/2;
		if (judge(mid)) r=mid-1;
		else l=mid+1;
	}
	judge(r+1);
	printf("YES\n");
	For(i,1,m) printf("%d ",num[i]);
	printf("\n");
	//for(;;);
	return 0;
}
