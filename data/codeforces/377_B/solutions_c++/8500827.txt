#include <bits/stdc++.h>
using namespace std;
const int maxn=1111111;
struct ww{
    int v1,v2,t;
    friend bool operator<(ww a,ww b){
        return(a.v1<b.v1);
    }
}a[maxn],b[maxn];
multimap<int,int>M;
int n,m,s,L,R,mi,i,an[maxn];
int check(int q){
    int ss=s,nw=m,nw2=n;
    M.clear();
    while(nw>0){
        while(nw2>=1&&a[nw2].v1>=b[nw].v1)
            M.insert(make_pair(a[nw2].v2,a[nw2].t)),--nw2;
        if(M.begin()==M.end()||M.begin()->first>ss)return 0;
        ss-=M.begin()->first;
        int sl=M.begin()->second;
        for(int i=nw;i>=1&&i>=nw-q+1;--i)an[b[i].t]=sl;
        nw-=q,M.erase(M.begin());
    }
}
int main(){
    //freopen("input.txt","r",stdin);
    scanf("%d%d%d",&n,&m,&s);
    for(i=1;i<=m;++i)scanf("%d",&b[i].v1),b[i].t=i;
    for(i=1;i<=n;++i)scanf("%d",&a[i].v1);
    for(i=1;i<=n;++i)scanf("%d",&a[i].v2),a[i].t=i;
    sort(b+1,b+m+1),sort(a+1,a+n+1);
    L=1,R=m+1;
    while(L<R){
        mi=(L+R)>>1;
        if(check(mi))R=mi;else L=mi+1;
    }
    if(L>m)puts("NO");else{
        check(L);
        puts("YES");
        for(i=1;i<=m;++i)printf("%d ",an[i]);
    }
    return 0;
}
