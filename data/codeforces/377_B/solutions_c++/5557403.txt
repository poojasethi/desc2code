#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
#include<queue>
using namespace std;
#define inf 1000000001

struct student{int b,c,id;} s[100100];
struct bug{int d,id;} a[100100];
bool operator <(student a,student b){return a.c>b.c;}
bool cpr(student a,student b) {return a.b>b.b;}
bool operator <(bug a,bug b){return a.d>b.d;}
int n,m,S;
int q[100010],op;

priority_queue<student> h;

int tp[100100];

int count(int d)
{
	while (!h.empty()) h.pop();
	int ret=0,j=0;

	for (int i=0;i<m;i+=d)
	{
		while (j<n && s[j].b>=a[i].d) h.push(s[j]),j++;
		if (h.empty()) return 0;
		student tmp=h.top();
		ret+=tmp.c;
		for (int k=i;k<min(i+d,m);k++) tp[a[k].id]=tmp.id+1;
		h.pop();
		if (ret>S)	return 0;
	}
	return 1;
}

int bs()
{
	int l=0,r=m+1,mid;
	while (l<r)
	{
		mid=(l+r)/2;
		if (count(mid)) r=mid;else l=mid+1;
	}
	return l;
}

int main()
{
	scanf("%d%d%d",&n,&m,&S);
	for (int i=0;i<m;i++)	scanf("%d",&a[i].d),a[i].id=i;
	for (int i=0;i<n;i++)	scanf("%d",&s[i].b);
	for (int i=0;i<n;i++)	scanf("%d",&s[i].c),s[i].id=i;
	sort(a,a+m);
	sort(s,s+n,cpr);
	int ans=bs();
	if (ans>m){puts("NO");return 0;}
	puts("YES");
	count(ans);
	for (int i=0;i<m;i++) printf("%d ",tp[i]);
	puts("");
	return 0;
}
