n, k  = map(int, raw_input().split())
s = raw_input()
if k > n / 2:
	while k < n:
		print 'RIGHT'
		k += 1
	while k >= 1:
		print 'PRINT %s' % s[k - 1]
		k -= 1
		if k >= 1:
			print 'LEFT'
else:
	while k > 1:
		print 'LEFT'
		k -= 1
	while k <= n:
		print 'PRINT %s' % s[k - 1]
		k += 1
		if k <= n:
			print 'RIGHT'
