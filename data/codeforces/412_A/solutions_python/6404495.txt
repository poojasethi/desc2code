n, k = map(int, raw_input().split(" "))
S = raw_input()
if k <= n/2:
    for i in range(k - 1):
        print "LEFT"
    for i in range(n):
        print "PRINT " + S[i]
        if i != n - 1: print "RIGHT"
else:
    for i in range(n - k):
        print "RIGHT"
    for i in range(n):
        print "PRINT " + S[n - i - 1]
        if i != n - 1: print "LEFT"
