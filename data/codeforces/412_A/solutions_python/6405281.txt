stringa = raw_input()
message = raw_input()
lista = stringa.split()
numLetters = int(lista[0])
startPos = int(lista[1])

numHalf = float(numLetters+1)/2

if startPos >= numHalf:
    while(startPos != numLetters):
        print "RIGHT"
        startPos+=1
else:
    while(startPos != 1):
        print "LEFT"
        startPos -=1

if startPos == numLetters:
    while(startPos != 0):
        print "PRINT" + " " + message[startPos -1]
        if(startPos != 1):
            print "LEFT"
        startPos-=1
else:
    while(startPos != numLetters + 1):
        print "PRINT" + " " + message[startPos -1]
        if(startPos != numLetters):
            print "RIGHT"
        startPos+=1
