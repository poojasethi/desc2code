n,k=[int(x) for x in raw_input().split()]
w=raw_input()
l=['LEFT','RIGHT','PRINT']
chk=0
if k!=1 or k!=n:
    if n-k>=k:
        for i in range(k-1):
            print l[0]
    else:
        chk=1
        for i in range(n-k):
            print l[1]
if chk:
    for i in range(-1,-n-1,-1):
        print ' '.join([l[2],w[i]])
        if i!=-n:
            print l[0]
else:
    for i in range(n):
        print ' '.join([l[2],w[i]])
        if i!=n-1:
            print l[1]
