#!/usr/bin/python
import sys
n , k = map(int , raw_input().split())
a = raw_input()

if n == 1:
	print "PRINT",a[0]
	sys.exit()

if k <= n / 2:
	k -= 1
	for i in range(k,0,-1):
		print "PRINT",a[i]
		print "LEFT"
	print "PRINT",a[0]
	for i in range(0 , k + 1):
		print "RIGHT"
	for i in range(k + 1,n - 1):
		print "PRINT",a[i]
		print "RIGHT"
	print "PRINT",a[n - 1]
else:
	k -= 1
	for i in range(k,n - 1):
		print "PRINT",a[i]
		print "RIGHT"
	print "PRINT",a[n - 1]
	for i in range(n - 1,k - 1,-1):
		print "LEFT"
	for i in range(k - 1,0,-1):
		print "PRINT",a[i]
		print "LEFT"
	print "PRINT",a[0]
