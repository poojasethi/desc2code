n, k = map(int, raw_input().split())
s = raw_input()

r = len(s)
k -= 1

if k < r - k:
	while k != 0:
		print "LEFT"
		k -= 1

	for i, c in enumerate(s):
		print "PRINT", c
		if i != r - 1:
			print "RIGHT"
else:
	while k != r - 1:
		print "RIGHT"
		k += 1
		
	for i, c in enumerate(s[::-1]):
		print "PRINT", c
		if i != r - 1:
			print "LEFT"
