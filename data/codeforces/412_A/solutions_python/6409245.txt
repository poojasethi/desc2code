n,k=raw_input().split(" ")
s=raw_input()
k=int(k)
n=int(n)
if k<=n/2:
    while k<>1:
        print 'LEFT'
        k-=1
    
    for i in range(n):
        print 'PRINT '+ s[i]
        if i<>n-1 : print 'RIGHT'
else:
    while k<>n :
        print 'RIGHT'
        k+=1
    
    for i in (range(n-1,-1,-1)):

        print 'PRINT '+s[i]
        if i<>0 : print 'LEFT'
