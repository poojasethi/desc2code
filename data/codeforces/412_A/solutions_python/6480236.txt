n, m = map(int, raw_input().split())
s = str(raw_input())
out = ["LEFT", "RIGHT", "PRINT"]
k = min(m-1, n-m)
if k == m-1: t = 0
else: t = 1
if n==1: print out[2], s[0]
else:
    for i in range(0, k):
        print out[t]
    t = 1-t
    if t==0:
        for i in range(n-1, -1, -1):
            print out[2], s[i]
            if i!=0: print out[t]
    else:
        for i in range(0, n):
            print out[2], s[i]
            if (i!=n-1): print out[t]
