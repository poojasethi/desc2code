n,k = map(int,raw_input().split())
s = raw_input()
k -= 1
if k < len(s)/2:
    for _ in xrange(k):
        print "LEFT"
    for i in xrange(len(s)):
        print "PRINT ",s[i]
        if i < len(s)-1:
            print "RIGHT"
else:
    for _ in xrange(len(s)-k-1):
        print "RIGHT"
    for i in xrange(len(s)-1,-1,-1):
        print "PRINT", s[i]
        if i > 0:
            print "LEFT"
