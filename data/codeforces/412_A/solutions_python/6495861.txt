n,k=map(int,raw_input().split())
s=list(raw_input().strip())
k-=1
if k<n/2:
    for i in range(k,0,-1):
        print 'LEFT'
    for i in range(n-1):
        print 'PRINT '+s[i]
        print 'RIGHT'
    print 'PRINT '+s[n-1]
else:
    for i in range(k,n-1):
        print 'RIGHT'
    for i in range(n-1,0,-1):
        print 'PRINT '+s[i]
        print 'LEFT'
    print 'PRINT '+s[0]
