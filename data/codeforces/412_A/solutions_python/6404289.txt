n, k = map(int, raw_input().split())
s = raw_input().strip()
k -= 1
if k < n / 2:
    while k > 0:
        print "LEFT"
        k -= 1
    print "PRINT", s[0]
    for c in s[1:]:
        print "RIGHT"
        print "PRINT", c
else:
    while k < n-1:
        print "RIGHT"
        k += 1
    print "PRINT", s[-1]
    for c in reversed(s[:-1]):
        print "LEFT"
        print "PRINT", c
