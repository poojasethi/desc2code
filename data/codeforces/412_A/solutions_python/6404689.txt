#!/usr/bin/env python

import sys
import logging

n, k = map(int, raw_input().split())
pattern = raw_input()

if k - 1 < n - k: 
    for _ in range(k-1): 
        print 'LEFT'

    for i in range(len(pattern)): 
        print 'PRINT %s' %pattern[i]
        if i != len(pattern)-1: 
            print 'RIGHT'

else: 
    for _ in range(n-k): 
        print 'RIGHT'

    for i in range(len(pattern)-1, -1, -1): 
        print 'PRINT %s' %pattern[i]
        if i != 0: 
            print 'LEFT'
