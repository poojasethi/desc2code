#include<cstdio>
#include<iostream>
#include<algorithm>

using namespace std;
int n,k;
char s[101];
int main()
{
	scanf("%d%d",&n,&k);
	scanf("%s",s);
	int mid=n/2;
	if(k>mid)
	{		
	while(k<n){printf("RIGHT\n");k++;}
	for(int i=n-1;i>=0;i--)
	{
		printf("PRINT %c\n",s[i]); 
		if(i>0)printf("LEFT\n");
	}	
	}
	else 
	{
	while(k>1){printf("LEFT\n");k--;}
	for(int i=0;i<n;i++)
	{
		printf("PRINT %c\n",s[i]);
		if(i<n-1)printf("RIGHT\n");
	}
	}	
	return 0;
}
