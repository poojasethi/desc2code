#include<iostream>
using namespace std;
int main() {
	char s[101];
	int n,k;
	cin>>n>>k>>s;
	if(n-k>k-1) {
		while(k!=1) {
			cout<<"LEFT"<<endl;
			k--;
		}
	}
	else {
		while(k!=n) {
			cout<<"RIGHT"<<endl;
			k++;
		}
	}
	if(k==1) {
		while(k!=n) {
			cout<<"PRINT "<<s[k-1]<<endl<<"RIGHT"<<endl;
			k++;
		}
	}
	else if(k==n) {
		while(k!=1) {
			cout<<"PRINT "<<s[k-1]<<endl<<"LEFT"<<endl;
			k--;
		}
	}
	cout<<"PRINT "<<s[k-1]<<endl;
	return 0;
}
