#include <cstdio>

int main()
{
	char sl[101];
	int n, k;
	scanf("%d %d", &n, &k);
	scanf("%s", sl+1);

	if (n - k + 1 <= k) {
		for (int i = k; i < n; i++) {
			printf("RIGHT\n");
		}
		for (int i = n; i > 0; i--) {
			printf("PRINT %c\n", sl[i]);
			if (i != 1) printf("LEFT\n");
		}
	} else {
		for (int i = k; i >1; i--) {
			printf("LEFT\n");
		}
		for (int i = 1; i <= n; i++) {
			printf("PRINT %c\n", sl[i]);
			if (i != n) printf("RIGHT\n");
		}
	}

	return 0;
}
