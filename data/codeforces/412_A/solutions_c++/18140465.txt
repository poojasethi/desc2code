#include<stdio.h>
#include<string.h>
char s[105];
int n,k;
int main()
{
	scanf("%d%d",&n,&k);
	scanf("%s",s);
	if(k<=n/2)
	{
		for(int i=1;i<k;i++)
		{
			printf("LEFT\n");
		}
		for(int i=0;i<n;i++)
		{
			printf("PRINT %c\n",s[i]);
			if(i!=n-1)	printf("RIGHT\n");
		}
	}
	else
	{
		for(int i=k;i<n;i++)
		{
			printf("RIGHT\n");
		}
		for(int i=n-1;i>=0;i--)
		{
			printf("PRINT %c\n",s[i]);
			if(i!=0)	printf("LEFT\n");
		}
	}
	return 0;
}
