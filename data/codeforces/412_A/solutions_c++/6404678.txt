#include<iostream>
#include<cstdio>
#include<string>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
	int i,j,c,d,n,k;
	char st[111];
	cin>>n>>k>>st;
	if(k*2>n)
	{
		for(i=k;i<n;i++)
			puts("RIGHT");
		for(i=n-1;i>=0;i--)
		{
			printf("PRINT %c\n",st[i]);
			if(i!=0) puts("LEFT");
		}	
	}
	else
	{
		for(i=1;i<k;i++)
			puts("LEFT");
		for(i=0;i<n;i++)
		{
			printf("PRINT %c\n",st[i]);
			if(i!=n-1) puts("RIGHT");
		}
	}
}
