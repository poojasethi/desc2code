
#include <iostream>
using namespace std;
char ch[110];
int main()
{
	int n,k;
	cin>>k>>n;
	for(int i=1;i<=k;i++)
		cin>>ch[i];
	int l=n-1,r=k-n;
	if(l<=r)
	{
		while(l--)
			cout<<"LEFT\n";
		for(int i=1;i<=k;i++)
		{
			cout<<"PRINT "<<ch[i]<<endl;
					if(i!=k)
				cout<<"RIGHT\n";
		}
	}
	else
	{
		while(r--)
			cout<<"RIGHT\n";
		for(int i=k;i>=1;i--)
		{
			cout<<"PRINT "<<ch[i]<<endl;
			if(i!=1)
				cout<<"LEFT\n";
		}
	}
	return 0;
}
