#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n,k; char s[101];
int main()
{
	scanf("%d%d%s",&n,&k,&s);k--;
	if(k<n/2){while(k-->0)printf("LEFT\n");for(k=0;k<n;k++){if(k)printf("RIGHT\n");printf("PRINT %c\n",s[k]);}}
    else{while(++k<n)printf("RIGHT\n");for(k=n-1;k>=0;k--){printf("PRINT %c\n",s[k]);if(k!=0)printf("LEFT\n");}}
	return 0;
}
