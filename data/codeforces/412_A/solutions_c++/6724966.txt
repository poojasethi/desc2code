#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int n,k;
char str[1000];

int main(){
	scanf("%d%d",&n,&k);
	scanf("%s",str+1);
	if (k-1<n-k){
		while (k-->1) puts("LEFT");
		++k;
	}
	else {
		while (k++<n) puts("RIGHT");
		--k;
	}
	if (k==1){
		printf("PRINT %c\n",str[k]);
		while (++k<=n) {
			puts("RIGHT");
			printf("PRINT %c\n",str[k]);
		}
	}
	else {
		printf("PRINT %c\n",str[n]);
		while (--k>=1){
			puts("LEFT");
			printf("PRINT %c\n",str[k]);
		}
	}
	return 0;
}
