#include <iostream>
#include <cstdio>
using namespace std;

int main() {
	int n, k; cin >> n >> k; --k;
	string s; cin >> s;
	if(k < n - k - 1) {
		for(int i = 0; i < k; ++i) puts("LEFT");
		for(int i = 0; i < n; ++i) {
			if(i > 0) puts("RIGHT");
			printf("PRINT %c\n", s[i]);
		}
	} else {
		for(int i = 0; i < n - k - 1; ++i) puts("RIGHT");
		for(int i = n - 1; i >= 0; --i) {
			if(i < n - 1) puts("LEFT");
			printf("PRINT %c\n", s[i]);
		}
	}
	return 0;
}
