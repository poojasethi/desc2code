#include<iostream>
using namespace std;
int main()
{
	int n,k,i;
	string s;
	cin>>n>>k>>s;
	if(n-k<k-1)
	{
		for(i=k-1;i<n-1;i++)
			cout<<"RIGHT\n";
		for(i;i>=0;i--)
		{
			cout<<"PRINT "<<s[i]<<endl;
			if(i!=0)cout<<"LEFT\n";
		}
	}
	else
	{
		for(i=k-1;i>0;i--)
			cout<<"LEFT\n";
		for(i;i<n;i++)
		{
			cout<<"PRINT "<<s[i]<<endl;
			if(i!=n-1)cout<<"RIGHT\n";
		}
	}
}
