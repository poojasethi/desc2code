#include <cstdio>
#include <cstring>
const int N = 105;
char a[N];
int main() {
	int n, m;
	scanf("%d%d%s", &n, &m, a);
	if(m > n - m) {
		for(int i = m;i < n; i ++) puts("RIGHT");

		for(int i = n-1; i >= 0; i --) {
			printf("PRINT %c\n", a[i]);
			if(i != 0)
				puts("LEFT");
		}
	} else {
		for(int i = 1;i < m; i ++) puts("LEFT");

		for(int i = 0; i < n; i ++) {
			printf("PRINT %c\n", a[i]);
			if(i != n-1)
				puts("RIGHT");
		}
	}
	return 0;
}
