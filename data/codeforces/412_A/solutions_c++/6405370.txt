#include<iostream>
#include <cstdio>
#include <string>
using namespace std;

int main(){
	int n,k;
	cin>>n>>k;
	string s;
	cin>>s;
	if(k>n/2) {
		while(k<n) {
			printf("RIGHT\n");
			k++;
		}
		for(int i=n-1;i>=0;i--) {
			printf("PRINT %c\n",s[i]);
			if(i) printf("LEFT\n");
		}
	} else {
		while(k>1) {
			printf("LEFT\n");
			k--;
		}
		for(int i=0;i<n;i++) {
			printf("PRINT %c\n",s[i]);
			if(i<n-1) printf("RIGHT\n");
		}
	}
	return 0;
}
