#include<stdio.h>
char a[1000]={0};
int i,k,n;
int main()
{
    scanf("%d %d %s",&n,&k,a+1);
    if(k-1<n-k)
    {
        while(k>1)
        {printf("LEFT\n");k--;}
        while(k<=n)
        {printf("PRINT %c\n",a[k++]);if(k<=n) printf("RIGHT\n");}
    }
    else
    {
        while(k<n)
        {printf("RIGHT\n");k++;}
        while(k>0)
        {printf("PRINT %c\n",a[k--]);if(k) printf("LEFT\n");}
    }
//  scanf("%d",&i);
    return 0;
}
