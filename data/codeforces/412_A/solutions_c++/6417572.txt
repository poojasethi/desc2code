#include <iostream>
using namespace std;

int main()
{
  int n,k;
  cin>>n>>k;
  string s;
  cin>>s;
  if(k<=n-k)
    {
      for(int i=k-2;i>=0;i--)
	cout<<"LEFT"<<endl;
      for(int i=0;i<n;i++)
	{
	  cout<<"PRINT "<<s[i]<<endl;
	  if(i+1<n)
	    cout<<"RIGHT"<<endl;
	}
    }
  else
    {
      for(int i=k;i<n;i++)
	cout<<"RIGHT"<<endl;
      for(int i=n-1;i>=0;i--)
	{
	  cout<<"PRINT "<<s[i]<<endl;
	  if(i-1>=0)
	    cout<<"LEFT"<<endl;
	}
    }
  return 0;
}
