#include<iostream>
#include<string>
using namespace std;

int n,k;
string s;

int main(){
  cin>>n>>k>>s;
  if(k>n/2){
    for(int i=0;i<n-k;i++)
      cout<<"RIGHT"<<endl;
    cout<<"PRINT "<<s[n-1]<<endl;
    for(int i=n-2;i>=0;i--)
      cout<<"LEFT"<<endl<<"PRINT "<<s[i]<<endl;
    return 0;
  }
  for(int i=1;i<k;i++)
    cout<<"LEFT"<<endl;
  for(int i=0;i<n-1;i++)
    cout<<"PRINT "<<s[i]<<endl<<"RIGHT"<<endl;
  cout<<"PRINT "<<s[n-1]<<endl;
  return 0;
}
