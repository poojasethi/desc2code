#include<iostream>
using namespace std;
int main()
{
int n,k;
cin>>n>>k;
char *x=new char [n+1];
cin>>x;
if(k>n/2)
{
for(--k; k<n-1; k++)
cout<<"RIGHT\n";
for(k; k>=0; k--)
if(k>0)
cout<<"PRINT "<<x[k]<<"\nLEFT\n";
else
cout<<"PRINT "<<x[k];
}
else
{
//cout<<k<<"\n";
for(k; k>1; k--)
cout<<"LEFT\n";
//cout<<k<<"\n";
for(--k; k<n; k++)
if(k<n-1)
cout<<"PRINT "<<x[k]<<"\nRIGHT\n";
else
cout<<"PRINT "<<x[k];
}
delete x;
return 0;
}
