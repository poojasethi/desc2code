#include <iostream>
using namespace std;

int main(){
	cin.tie(NULL);
	ios_base::sync_with_stdio(false);
	int n, k;
	char str[101];
	cin>>n>>k>>str;
	if(k<=n/2){
		for(int i=k; i>1; --i){
			cout<<"LEFT\n";
		}
		for(int i=0; i<n; ++i){
			cout<<"PRINT "<<str[i]<<"\n";
			if(str[i+1])cout<<"RIGHT\n";
		}
	}else{
		for(int i=k; i<n; ++i){
			cout<<"RIGHT\n";
		}
		for(int i=n-1; i>=0; --i){
			cout<<"PRINT "<<str[i]<<"\n";
			if(i)cout<<"LEFT\n";
		}
	}
	return 0;
}
