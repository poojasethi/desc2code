#include<stdio.h>

int n,k,i,bekci;
char dizi[101];

int main(){
	
	scanf("%d %d",&n,&k);

	scanf("%s",dizi);
	
	if(k-1>n-k){
		for(i=0 ; i<n-k ; i++)
			printf("RIGHT\n");
		
		bekci=1;
	}
	else
		for(i=0 ; i<k-1 ; i++)
			printf("LEFT\n");
	if(bekci){
		for(i=n-1 ; i>=0 ; i--){
			printf("PRINT %c\n",dizi[i]);
			if(i!=0)
				printf("LEFT\n");
			
		}
	}
	else  {
		for(i=0 ; i<n ; i++){
			printf("PRINT %c\n",dizi[i]);
			if(i!=n-1){
				printf("RIGHT\n");
			}
		}
	}
	
	return 0;
}
