#include <stdio.h>

int main()
{
	int n = 0, k = 0;
	scanf("%d %d\n", &n, &k);
	char p[100];
	gets(p);

	printf("\n");
	if (k > n / 2)
	{
		for (int i = k; i < n; ++i)
			printf("RIGHT\n");
		printf("PRINT %c\n", p[n - 1]);
		for (int i = n; i > 1; --i)
			printf("LEFT\nPRINT %c\n", p[i - 2]);
	}
	else
	{
		for (int i = k; i > 1; --i)
			printf("LEFT\n");
		printf("PRINT %c\n", p[0]);
		for (int i = 1; i < n; ++i)
			printf("RIGHT\nPRINT %c\n", p[i]);
	}

	return 0;
}
