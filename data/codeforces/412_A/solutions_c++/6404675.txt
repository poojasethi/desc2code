#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
char ar[1010];
int n,k;
int main()
{
	scanf(" %d %d %s",&n,&k,ar+1);
	int uz=strlen(ar+1);
	if(n-k>=k){
		for(;k>1;k--) printf("LEFT\n");
		printf("PRINT %c\n",ar[k++]);
		for(;k<=uz;k++) printf("RIGHT\nPRINT %c\n",ar[k]);
	}
	else{
		for(;k<uz;k++) printf("RIGHT\n");
		printf("PRINT %c\n",ar[k--]);
		for(;k>0;k--) printf("LEFT\nPRINT %c\n",ar[k]);
	}
	return 0;
}
