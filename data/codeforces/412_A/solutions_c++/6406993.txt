#include<bits/stdc++.h>
using namespace std;
int n,k;
string st;
int main(){
	cin>>n>>k>>st;
	int i,j,pos=k,u=st.size();
	if(k!=1 && k!=n)
	{
		if(n-k>k-1)	
			while(pos!=1) cout<<"LEFT\n",pos--;
		else 
			while(pos!=n) cout<<"RIGHT\n",pos++;
	}
	if(pos==1)
	{
		for(i=0;i<n;i++)
		{
			printf("PRINT %c\n",st[i]);
			if(i!=n-1) cout<<"RIGHT\n";
		}
	}
	else 
		for(i=n-1;i>-1;i--){
			printf("PRINT %c\n",st[i]);
			if(i) printf("LEFT\n");
		}
	return 0;
}
