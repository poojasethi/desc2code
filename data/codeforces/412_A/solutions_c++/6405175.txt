#include <iostream>
#include <string>

using namespace std;

int main()
{
    int n,k;
    cin >> n >> k;
    k--;
    string s;
    cin >> s;
    for (;k!=0 && k!=n-1; k+=k<n-k?-1:1)
        cout << (k<n-k?"LEFT":"RIGHT") << endl;
    cout << "PRINT " << s[k] << endl;
    if (k==0)
        for (;k<n-1;)
            cout << "RIGHT" << endl << "PRINT " << s[++k] << endl;
    else
        for (;k>0;)
            cout << "LEFT" << endl << "PRINT " << s[--k] << endl;
    return 0;
}
