#include<cstdio>
#include<cstring>
int main(){
	int n,k;
	char x[105];
	scanf("%d %d",&n,&k);
	scanf("%s",x);
	int l=strlen(x);
	if(k-1<=n-k){
		for(int i=1;i<=k-1;i++){
			printf("LEFT\n");
		}
		for(int i=1;i<=l;i++){
			printf("PRINT %c\n",x[i-1]);
			if(i!=l)
			printf("RIGHT\n");
		}
	}
	else{
		for(int i=1;i<=n-k;i++){
			printf("RIGHT\n");
		}
		for(int i=l;i>=1;i--){
			printf("PRINT %c\n",x[i-1]);
			if(i!=1)
			printf("LEFT\n");
		}
	}
	return 0;
}
