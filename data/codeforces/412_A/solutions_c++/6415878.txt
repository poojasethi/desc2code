#include <cstdio>
#include <cstdlib>


char s[111];
int main(){
	int n,m;
	scanf("%d %d",&n,&m);
	scanf("%s",s+1);
	int is_right = (n-m)<(m-1)?1:0;
	if(is_right){
		for(int i = 0;i < n-m;i++)
			puts("RIGHT");
		for(int i = n;i > 0;i--){
			printf("PRINT %c\n",s[i]);
			if(i > 1)
				puts("LEFT");
		}
	}else{
		for(int i = 0;i < m-1;i++)
			puts("LEFT");
		for(int i = 1;i <= n;i++){
			printf("PRINT %c\n",s[i]);
			if(i < n)
				puts("RIGHT");
		}
	}
	return 0;
}
