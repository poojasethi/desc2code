#include <bits/stdc++.h>
#define maxn 120

using namespace std;

char ar[maxn];

int main()
{
	int a,b;
	scanf("%d %d",&a,&b);
	scanf(" %s",ar+1);
	if( a/2 < b ) {
		for(int i=b;i<a;i++)
			printf("RIGHT\n");
		for(int i=a;i>0;i--) {
			printf("PRINT %c\n",ar[i]);
			if( i == 1 ) break;
			printf("LEFT\n");
		}
		
	}
	else {
		for(int i=b;i>1;i--)
			printf("LEFT\n");
		for(int i=1;i<=a;i++) {
			printf("PRINT %c\n",ar[i]);
			if( i == a ) break;
			printf("RIGHT\n");
		}
	}
	
	return 0;
}
