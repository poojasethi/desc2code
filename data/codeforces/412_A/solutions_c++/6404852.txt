#include <stdio.h>

int main (void) {
	int n, k;
	char s[101];
	scanf ("%d%d%s", &n, &k, s);
	k--;
	if (k >= n/2) {
		while (++k < n)	printf ("RIGHT\n");
		for (k = n-1; k >= 0; k--) {
			printf ("PRINT %c\n", s[k]);
			if (k)	printf ("LEFT\n");
		}
	} else {
		while (k-- > 0)	printf ("LEFT\n");
		for (k = 0; k < n; k++) {
			if (k)	printf ("RIGHT\n");
			printf ("PRINT %c\n", s[k]);
		}
	}
	return 0;
}
