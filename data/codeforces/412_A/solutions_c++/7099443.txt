#include<iostream>
using namespace std;

int main()
{
	int n,k;
	cin>>n>>k;
	string str;
	cin>>str;
	
	if((k-1)<(n-k))
	{
		for(int i=k-1 ; i>=1 ; i--)
			cout<<"LEFT\n";
		
		for(int i=0 ; i<n ; i++)
		{
			cout<<"PRINT "<<str[i]<<endl;
			if(i<n-1)
				cout<<"RIGHT\n";
		}	
	}
	
	else
	{
		for(int i=k+1 ; i<=n ; i++)
			cout<<"RIGHT\n";
		
		for(int i=n-1 ; i>=0 ; i--)
		{
			cout<<"PRINT "<<str[i]<<endl;
			if(i>0)
				cout<<"LEFT\n";
		}	
	}
}
