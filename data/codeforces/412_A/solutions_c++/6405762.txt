#include<iostream>
#include<string>
using namespace std;
int main()
{
	int n,k;
	char s[101];
	cin>>n>>k;
	cin>>s;
	if(k>n/2)
	{
		while(k!=n)
		{
			cout<<"RIGHT"<<endl;
			k++;
		}
		for(int i=n-1;i>=0;i--)
		{
			cout<<"PRINT "<<s[i]<<endl;
			if(i!=0)
			cout<<"LEFT"<<endl;
		}
	}
		else
		{
			while(k!=1)
			{
				cout<<"LEFT"<<endl;
				k--;
			}
			for(int i=0;i<n;i++)
			{
				cout<<"PRINT "<<s[i]<<endl;
				if(i!=n-1)
				cout<<"RIGHT"<<endl;
			}
		}
}
