#include<iostream>
using namespace std;

int main()
{ int n,k,i;
  char s[100];
  cin>>n>>k;
  cin>>s;
  i=k;
  if(k>n/2)
  { while(i!=n) {i++; cout<<"RIGHT\n"; }
    cout<<"PRINT "<<s[i-1]<<endl;
    while(i>1) {i--; cout<<"LEFT\nPRINT "<<s[i-1]<<endl;}
  }
  else
  { while(i>1) {i--; cout<<"LEFT\n"; }
    cout<<"PRINT "<<s[i-1]<<endl;
    while(i!=n) {i++; cout<<"RIGHT\nPRINT "<<s[i-1]<<endl;}
  }
  return 0;
}
