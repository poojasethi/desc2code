#include <iostream>

using namespace std;

int main()
{
	int k,n;
	cin>>n>>k;
	string s;
	cin>>s;
	if ((n-k)<(k-1))
	{
		for (int i=k;i<n;i++)
			cout<<"RIGHT"<<endl;
		for (int i=n;i>=1;i--){
			cout<<"PRINT "<<s[i-1]<<endl;
			if (i!=1)
				cout<<"LEFT"<<endl;
		}
		int _;cin>>_;
		return 0;
	}
	for (int i=k;i>1;i--)
		cout<<"LEFT"<<endl;
	for (int i=1;i<=n;i++)
	{
		cout<<"PRINT "<<s[i-1]<<endl;
		if (i!=n)
			cout<<"RIGHT"<<endl;
	}
	int _;cin>>_;
	
}
