#include <iostream>
#include <algorithm>
using namespace std;

int main()
{int n,k; cin>>n>>k;
string s; cin>>s;
if (k>n-k)
{ int QQ=n-k;
    while (QQ--)
  cout<<"RIGHT\n";
  for (int i=n-1;i>=0;i--)
    {cout<<"PRINT "<<s[i]<<"\n";
  if (i>0)cout<<"LEFT\n";}

}
else if (k<=n-k)
{ int QQ=k-1;
    while (QQ--)
  cout<<"LEFT\n";
  for (int i=0;i<n;i++)
    {cout<<"PRINT "<<s[i]<<"\n";
  if (i<n-1)cout<<"RIGHT\n";}
}
}
