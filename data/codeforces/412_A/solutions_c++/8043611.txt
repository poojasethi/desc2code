
#include <iostream>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);

	int n,k;
	cin>>n>>k;
	
	string s;
	cin>>s;
	
	if (k>n/2)
	{
		while (k!=n) { cout<<"RIGHT\n"; k++; }
		for (int i=k-1 ; i>=0 ; i--)
			cout<<"PRINT "<<s[i]<<"\n"<<(i!=0 ? "LEFT" : " ")<<"\n";
	}
	else
	{
		while (k!=1) { cout<<"LEFT\n"; k--; }
		for (int i=0 ; i<s.size() ; i++)
			cout<<"PRINT "<<s[i]<<"\n"<<(i!=n-1 ? "RIGHT" : " ")<<"\n";
	}
}
