#include <iostream>
using namespace std;

int main()
{
    int n,k,i;
    char sir[110];

    cin>>n>>k;
    cin>>sir;

    if(k<=n-k)
    {
        for(i=1;i<k;i++) cout<<"LEFT\n";
        for(i=0;i<n-1;i++) cout<<"PRINT "<<sir[i]<<"\nRIGHT\n";
        cout<<"PRINT "<<sir[n-1]<<"\n";
    }
    else
    {
        for(i=1;i<=n-k;i++) cout<<"RIGHT\n";
        for(i=n-1;i>0;i--) cout<<"PRINT "<<sir[i]<<"\nLEFT\n";
        cout<<"PRINT "<<sir[0]<<"\n";
    }

    return 0;
}
