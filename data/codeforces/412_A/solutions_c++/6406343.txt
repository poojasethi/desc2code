#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	int i,j,n,k;
	char s[200];
	cin>>n>>k;
	scanf("%s",s);
	if(k<=n/2)
	{
		for(i=1;i<k;i++)
		cout<<"LEFT\n";
		for(i=0;i<n-1;i++)
		cout<<"PRINT "<<s[i]<<endl<<"RIGHT"<<endl;
		cout<<"PRINT "<<s[n-1]<<endl;
	}
	else
	{
		for(i=k+1;i<=n;i++)
		cout<<"RIGHT\n";
		for(i=n-1;i>0;i--)
		cout<<"PRINT "<<s[i]<<endl<<"LEFT"<<endl;
		cout<<"PRINT "<<s[0]<<endl;
	}
	return 0;
}
