#include <cstdio>

char s[200];
int n, k;
int main(){
    scanf("%d%d", &n, &k);
    scanf("%s", s);
    if(k*2 <= n){//idem lijevo
        for(int i = 1;i < k;++i)printf("LEFT\n");
        printf("PRINT %c\n", s[0]);
        for(int i = 1;i < n;++i){
            printf("RIGHT\nPRINT %c\n", s[i]);
        }
    }
    else {
        for(int i = k;i < n;++i)printf("RIGHT\n");
        printf("PRINT %c\n", s[n-1]);
        for(int i = n - 2;i >= 0;--i)printf("LEFT\nPRINT %c\n", s[i]);
    }

}
