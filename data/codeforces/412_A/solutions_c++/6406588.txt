#include<bits/stdc++.h>
using namespace std;
char arr[10000];
int n,q,m;
int i,j,k;
int main(){
	scanf("%d %d",&n,&m);
	scanf("%s",arr);
	if(abs(n-m)<abs(m))
		q=1;
	if(q==0){
		for(i=0;i<m-1;i++)
			printf("LEFT\n");
		for(i=0;i<n;i++){
			printf("PRINT %c\n",arr[i]);
			if(i!=n-1)
				printf("RIGHT\n");}
	}
	if(q==1){
		for(i=0;i<n-m;i++)
			printf("RIGHT\n");
		for(i=n-1;i>=0;i--){
			printf("PRINT %c\n",arr[i]);
			if(i!=0)
				printf("LEFT\n");
		}
	}
	return 0;
}
