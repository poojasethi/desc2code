#include<stdio.h>
#include<string.h> 
int main()
{
	int m,t,i,j,d[105];
	char s[105];
	scanf("%d%d",&m,&t);
	getchar();
	for(i=1;i<=m;i++)
	{
		scanf("%c",&s[i]);
	}
	if(m-t<t)
	{
		
		for(i=1;i<=m-t;i++)
		{
			printf("RIGHT\n");
		}
		for(j=m;j>=1;j--)
		{
			printf("PRINT %c\n",s[j]);
			if(j!=1) printf("LEFT\n");
		}
	}
	else
	{
		
		for(i=1;i<t;i++)
		printf("LEFT\n");
		for(j=1;j<=m;j++)
		{
			printf("PRINT %c\n",s[j]);
			if(j!=m) printf("RIGHT\n");
		}
	}
}
