#include<stdio.h>
#include<string.h>
char a[105];
int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	scanf("%s",a+1);
	if(k>n/2){
		while(k<n){
			printf("RIGHT\n");
			k++;
		}
	}
	else {
		while(k>1){
			printf("LEFT\n");
			k--;
		}
	}
	if(k==1){
		while(k<n){
			printf("PRINT %c\n",a[k]);
			printf("RIGHT\n");
			k++;
		}
	}
	else {
		while(k>1){
			printf("PRINT %c\n",a[k]);
			printf("LEFT\n");
			k--;
		}
	}
	printf("PRINT %c\n",a[k]);
	return 0;
}
