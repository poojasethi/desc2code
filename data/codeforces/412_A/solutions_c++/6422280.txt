#include <iostream>
using namespace std;
int main()
{
	int n, k, i;
	string s;
	cin>>n>>k>>s;
	if (k>n/2) {
		for (i=k; i<n; i++) cout<<"RIGHT"<<endl;
		for (; i>1; i--) cout<<"PRINT "<<s[i-1]<<endl<<"LEFT"<<endl;
		cout<<"PRINT "<<s[0]<<endl;
	}
	else {
		for (i=k; i>1; i--) cout<<"LEFT"<<endl;
		for (; i<n; i++) cout<<"PRINT "<<s[i-1]<<endl<<"RIGHT"<<endl;
		cout<<"PRINT "<<s[n-1]<<endl; 
	}
	return 0;
}
