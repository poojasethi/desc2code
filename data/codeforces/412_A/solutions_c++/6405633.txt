#include<cstdio>
using namespace std;
int main()
{
	int n,k,j,i;
	char str[100];
	scanf("%d%d",&n,&k);
	scanf("%s",str);
	if(k>n-k)
	{
	    for(i=0; i<n-k; i++) puts("RIGHT");
	    for(j=n; j>=1; j--)
	    {
	    	printf("PRINT %c\n",str[j-1]);
	    	if(j>1) puts("LEFT");
	    }
	}
	else
	{
	    for(i=1; i<k; i++) puts("LEFT");
		for(j=0; j<n; j++)	
		{
			printf("PRINT %c\n",str[j]);
			if(j!=n-1) puts("RIGHT");
		}
	}
	return 0;
}
