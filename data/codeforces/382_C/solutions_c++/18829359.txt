#include<bits/stdc++.h>
using namespace std;
const int N=1e5+6;
long long n,a[N],diff[N],d,cnt,td,tcnt,ps,i;
int main()
{
	ios::sync_with_stdio(0);cin.tie(0);
    cin>>n;
    for(i=0 ;i < n ;i++)cin>>a[i];
    sort( a,a+n );
    for( i=1 ; i<n ; i++)
    diff[i] = a[i]-a[i-1];
    sort( diff+1,diff+n );
    if(n==1)
    {cout<<-1; return 0;}
    if( n == 2) 
    {
    	if(diff[1] == 0)
    		cout<< 1 << endl << a[0];
    	else if( diff[1]%2 == 0 )
    		cout<<3<<endl<<a[0]-diff[1]<<" "<<a[0]+diff[1]/2<<" "<<a[1]+diff[1];
    	else
    		cout<<2<<endl<<a[0]-diff[1]<<" "<<a[1]+diff[1];
    }
    else
    {
    	  if( diff[1]!=diff[n-1]  && ( diff[1]!= diff[n-2] || 2*diff[1] != diff[n-1] ) )
    	  { cout<<0;return 0; }
    	  d=diff[1];
    	  for(i=1 ; i<n ;i++)
    	  	if(a[i]!=a[i-1]+d)
    	  		{cout<<1<<endl<<a[i-1]+d; return 0;}
    	  if(diff[1] == 0)
    	  	cout<<1<<endl<<a[0];
    	  else
    	  	cout<<2<<endl<<a[0]-d<<" "<<a[ n - 1] +d ;
    }
	return 0;
}
