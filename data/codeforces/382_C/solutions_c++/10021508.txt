#include <bits/stdc++.h>
using namespace std;
const int N=100010;
int a[N];
int d[N];
set<int> ans;
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",a+i);
    sort(a,a+n);
    if(n==1)
    {
        puts("-1");
        return 0;
    }
    if(n==2)
    {
        int dist=a[1]-a[0];
        ans.insert(a[0]-dist);
        ans.insert(a[1]+dist);
        if(dist%2==0) ans.insert(a[0]+dist/2);
    }
    else
    {
        for(int i=0;i<n-1;i++)
        {
            d[i]=a[i+1]-a[i];
        }
        sort(d,d+n-1);
        if(d[n-2]==d[0])
        {
            ans.insert(a[0]-d[0]);
            ans.insert(a[n-1]+d[0]);
        }
        else if(d[0]==d[n-3])
        {
            int dis=d[0];
            for(int i=0;i<n-1;i++)
                if(a[i+1]-a[i]==2*dis&&a[i+1]-a[i]!=dis) ans.insert(a[i]+dis);
        }
    }
    printf("%d\n",ans.size());
    for(set<int>::iterator it=ans.begin();it!=ans.end();it++) printf("%d ",*it);
    return 0;
}
