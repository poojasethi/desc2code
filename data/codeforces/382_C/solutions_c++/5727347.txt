#include <iostream>
#include <algorithm>

using namespace std;

long long arr[100010];

int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;++i)
		cin>>arr[i];
	sort(arr,arr+n);
	if(n==1) cout<<-1<<endl;
	else if(n==2)
	{
		long long d=arr[1]-arr[0];
		if((arr[1]+arr[0])%2LL==0 && arr[0]!=arr[1])
		{
			cout<<3<<endl;
			cout<<arr[0]-d<<" "<<arr[0]+d/2<<" "<<arr[1]+d<<endl;
		}
		else if(arr[0]==arr[1])
		{
			cout<<1<<endl;
			cout<<arr[0]<<endl;
		}
		else
		{
			cout<<2<<endl;
			cout<<arr[0]-d<<" "<<arr[1]+d<<endl;
		}
	}
	else
	{
		int d=arr[1]-arr[0];
		int cnt=0;
		int id;
		for(int i=2;i<n;++i)
			if(d ^ (arr[i]-arr[i-1])) id=i,cnt++;
		if(cnt==n-2 && arr[2]-arr[1]<d) cnt=1,id=1,d=arr[2]-arr[1];
		if(cnt==0)
		{
			if(arr[0]!=arr[1])
			{
				cout<<2<<endl;
				cout<<arr[0]-d<<" "<<arr[n-1]+d<<endl;
			}
			else
			{
				cout<<1<<endl;
				cout<<arr[0]<<endl;
			}
		}
		else if(cnt==1)
		{
			if((arr[id]-arr[id-1])==2*d)
			{
				cout<<1<<endl;
				cout<<arr[id]-d<<endl;
			}
			else
				cout<<0<<endl;
		}
		else cout<<0<<endl;
	}
}
