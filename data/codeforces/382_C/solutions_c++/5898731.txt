#include <iostream>
#include <algorithm>
using namespace std;
int v[100001];
int r[100000];

int main()
{
    int i,n,mx=0,rmax=0;
    cin>>n;
    for(i = 1; i <= n; i++) cin>>v[i];
    sort(v+1,v+n+1);
    if(n == 1) cout<<-1;
    else if(v[1] == v[n]) cout<<1<<'\n'<<v[1];
    else if(n == 2)
    {
        if((v[2] - v[1]) % 2 == 0) cout<<3<<'\n'<<2*v[1] - v[2]<<" "<<v[1] + (v[2] - v[1])/2<<" "<<2*v[2] - v[1];
        else cout<<2<<'\n'<<2*v[1] - v[2]<<" "<<2*v[2] - v[1];
    }
    else
    {
        for(i = 1; i < n; i++)
        {
            r[i] = v[i + 1] - v[i];
            if(r[i]>rmax)
            {
                rmax=r[i];
                mx=i;
            }
        }
        sort(r+1,r+n);
        if(r[1]==r[n-1]) cout<<2<<'\n'<<v[1]-r[1]<<" "<<v[n]+r[1];
        else if(r[1]==r[n-2] && r[n-1]==2*r[1]) cout<<1<<'\n'<<v[mx]+r[1];
        else cout<<0;
    }
    return 0;
}
