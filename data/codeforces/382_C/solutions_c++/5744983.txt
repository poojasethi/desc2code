#include<iostream>
#include<cstdio>
#include<string>
#include<algorithm>
#include<cmath>
#include<vector>
using namespace std;
int main(){
int a,c,d=0,e,f,g,h,j=0,i,l=100000000,r=0;
vector<int > b;
cin>>a;
for(i=1;i<=a;i++){
cin>>c;
b.push_back(c);
}

sort(b.begin(),b.end());
if(a==1){cout<<-1;return 0;}
if(b[0]==b[a-1]){cout<<1<<endl<<b[0];return 0;}
for(i=1;i<a;i++){
if(l>b[i]-b[i-1])l=b[i]-b[i-1];
}
if(a==2 && l%2==0){cout<<3<<endl<<b[0]-l<<" "<<b[0]+(l/2)<<" "<<b[a-1]+l;return 0;}
for(i=1;i<a;i++){
if(b[i]-b[i-1]>l && ((b[i]-b[i-1])/2!=l || (b[i]-b[i-1])%2==1)){cout<<0;return 0;}
if(b[i]-b[i-1]<l){cout<<0;return 0;}
if(b[i]-b[i-1]!=l && r>0){cout<<0;return 0;}
if(b[i]-b[i-1]!=l)r=(b[i]+b[i-1])/2;
}
if(r==0){cout<<2<<endl<<b[0]-l<<" "<<b[a-1]+l;return 0;}
else cout<<1<<endl<<r;
return 0;
}
