#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,s,v;
int a[N],f1[N],f2[N],b[N];
int main() {
	scanf("%d",&n);
	if (n==1) {
		printf("-1\n"); return 0;
	}
	For(i,1,n) scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	f1[2]=a[2]-a[1];
	For(i,3,n) if (f1[i-1]>=0&&a[i]-a[i-1]==f1[i-1]) f1[i]=f1[i-1];
		else f1[i]=-1;
	f2[n-1]=a[n]-a[n-1];
	for (i=n-2;i;i--) if (f2[i+1]>=0&&a[i+1]-a[i]==f2[i+1]) f2[i]=f2[i+1];
		else f2[i]=-1;
	if (f2[1]>=0) b[++s]=a[1]-f2[1];
	if (f1[n]>=0) b[++s]=a[n]+f1[n];
	For(i,1,n-1) {
		int A=a[i+1]-a[i];
		if (A%2==0&&(f1[i]==A/2||i==1)&&(f2[i+1]==A/2||i==n-1)) b[++s]=a[i]+A/2;
	}
	sort(b+1,b+s+1);
	For(i,1,s) if (i==1||b[i]>b[v]) b[++v]=b[i];
	printf("%d\n",v);
	For(i,1,v) printf("%d ",b[i]);
	printf("\n");
	return 0;
}
