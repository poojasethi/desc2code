#include<stdio.h>
#include<algorithm>
using namespace std;
int n;
int a[200001];
bool check(int d){
	for(int i=1;i<n;++i) if(a[i]+d!=a[i+1]) return false;
	return true;
}
bool answer(int d){
	int rec=-1;
	for(int i=1;i<n;++i) if(a[i]+d!=a[i+1]){
		if(a[i]+2*d!=a[i+1]) return false;
		else{
			if(rec!=-1) return false;
			rec=a[i]+d;
		}
	}
	printf("1\n%d\n",rec);
	return true;
}
int main(){
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif
		
	scanf("%d",&n);
	for(int i=1;i<=n;++i){
		scanf("%d",a+i);
	}
	if(n==1){
		puts("-1");
		return 0;
	}
	sort(a+1,a+n+1);
	if(a[1]==a[n]){
		printf("1\n%d\n",a[1]);
		return 0;
	}
	int d=min(a[2]-a[1],a[n]-a[n-1]);
	if(check(d)){
		if(n!=2||(a[1]+a[n])%2!=0)printf("2\n%d %d\n",a[1]-d,a[n]+d);
		else printf("3\n%d %d %d\n",a[1]-d,(a[1]+a[n])/2,a[n]+d);
		return 0;
	}
	else{
		if(!answer(d)) printf("0\n");
	}
	return 0;
}
