#include<bits/stdc++.h>
using namespace std;
int n, a[100000], d=-1, b[100000];
int main(){
  cin>>n;
  for(int i=0;i<n;i++){
	 cin >> a[i];
  }
  sort(a,a+n);
	for(int i=0;i<n-1;i++){
		b[i] = a[i+1]-a[i];
	}
	sort(b,b+n-1);
	if(n==1){cout << "-1"<<endl;}
	else if(n==2) {d = a[1]-a[0];
	if(d == 0) {
		cout <<"1"<<endl<<a[0]<<endl;
		return 0;
	}
	if( d % 2 == 0 ){cout << "3"<<endl;}
	else {cout << "2"<<endl;}
    cout << a[0]-d <<" ";
	if( d % 2 == 0 )
		cout <<a[0] + d/2 << " ";
		cout << a[1] + d << endl;
	}
	else {
		if(b[n-2] != b[0] && (b[n-3] != b[0] || b[n-2] != b[0]*2)) {
			cout <<"0"<<endl;
			return 0;
		}
		d = b[0];
		for(int i=1;i<n;i++)
			if(a[i] - a[i-1] != d) {
				cout <<1<< endl<< a[i-1] + d<<endl;;
				return 0;
			}
		if(d == 0)
			cout << "1"<<endl<<a[0] <<endl;
		else
			cout << 2 <<endl<< a[0] - d <<" "<< a[n-1]+d << endl;
	}
	return 0;
}
