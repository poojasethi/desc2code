#include<iostream>
#include<algorithm>

using namespace std;

#define f(i,a,b) for(i=a;i<b;++i)
#define rep(i,a) for(i=0;i<a;++i)

int a[100005],diff[100005],n,ans[20],as,i,j;

int main()
{
	cin>>n;
	rep(i,n)
		cin>>a[i];
	if(n==1){cout<<-1;return 0;}
	sort(a,a+n);
	if(a[0]==a[n-1]){cout<<1<<endl<<a[0];return 0;}
	if(n==2)
	{
		diff[0]=a[1]-a[0];
		ans[as++]=a[0]-diff[0];
		ans[as++]=a[1]+diff[0];
		if(diff[0]%2==0)
			ans[as++]=(a[0]+a[1])/2;
		sort(ans,ans+as);
		cout<<as<<endl;
		rep(i,as)
			cout<<ans[i]<<" ";
		return 0;
	}
	rep(i,n-1)
		diff[i]=a[i+1]-a[i];
	sort(diff,diff+n-1);
	if(diff[0]==diff[n-2])
	{
		ans[as++]=a[0]-diff[0];
		ans[as++]=a[n-1]+diff[0];
	}
	else if(diff[0]==diff[n-3] && diff[n-2]%2==0 && diff[n-2]/2==diff[0])
	{
		rep(i,n-1)
			if(diff[n-2]==a[i+1]-a[i])
				ans[as++]=(a[i+1]+a[i])/2;
	}
	else
	{
		cout<<0;
		return 0;
	}
	sort(ans,ans+as);
	cout<<as<<endl;
	rep(i,as)
		cout<<ans[i]<<" ";
	return 0;	
}
