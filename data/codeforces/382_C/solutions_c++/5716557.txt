#include <cstdio>
#include <algorithm>
using namespace std;
int a[100100],n,a1=-1,a2=-1,b2,t;

bool cmp(int xx,int yy){
	return xx<yy;
}

int main()
{
	int i,j;
	//freopen("C.in","r",stdin);
	//freopen("C.out","w",stdout);
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	sort(a+1,a+n+1,cmp);
	a1=a[n]-a[n-1];
	for(i=n-1;i>1;i--)
	{
		if(a[i]-a[i-1]!=a1)
		{
			if(a2==-1) {a2=a[i]-a[i-1];b2=i;}
			else if(a2==a[i]-a[i-1] && b2==n-1)
			{
				t=a1;a1=a2;a2=t;b2=n;
			}
			else
			{
				printf("0\n"); return 0;
			}
		}
	}
	if(a2!=-1)
	{
		if(a2!=0 && a1==0) printf("0\n");
		else if(n==3 && a2==0 && a1!=0) printf("0\n");
		else if(n==3 && a1==2*a2) printf("1\n%d\n",a[n]-a2);
		else if(a2==2*a1) printf("1\n%d\n",a[b2]-a1);
		else printf("0\n");
		return 0;
	}
	else
	{
		if(n==1) printf("-1\n");
		else if(a1==0)  printf("1\n%d",a[1]);
		else if(n==2 && a1%2==0) printf("3\n%d %d %d\n",a[1]-a1,a[1]+a1/2,a[2]+a1);
		else printf("2\n%d %d\n",a[1]-a1,a[n]+a1);
		return 0;
	}
	return 0;
}
