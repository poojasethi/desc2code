#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=100050;
int n,se[maxn],d,s=0,ans[maxn];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&se[i]);
	sort(se,se+n);
	if(n==1) printf("-1\n");
	else if(n==2) 
	{
		if(se[0]==se[1])
			printf("1\n%d\n",se[0]);
		else if((se[0]+se[1])&1)
			printf("2\n%d %d\n",se[0]*2-se[1],se[1]*2-se[0]);
		else 
			printf("3\n%d %d %d\n",se[0]*2-se[1],(se[0]+se[1])/2,se[1]*2-se[0]);
	}
	else
	{
		int d1=se[1]-se[0],d2=se[2]-se[1],z=0;
		if(d1==d2) d=d1;
		else if(d1*2==d2) d=d1;
		else if(d1==d2*2) d=d2;
		else {printf("0\n");return 0;}
		for(i=0;i<n-1;i++)
			if(!z&&se[i+1]-se[i]==d*2) z=i+1;
			else if(se[i+1]-se[i]!=d) {printf("0\n");return 0;}
		if(!z)
			printf("2\n%d %d\n",se[0]-d,se[n-1]+d);
		else if((se[z-1]+se[z])&1) 
			printf("0\n");
		else
			printf("1\n%d\n",(se[z-1]+se[z])/2);
	}
	return 0;
}
