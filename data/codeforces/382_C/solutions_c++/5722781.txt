#include <iostream>
#include <string>
#include <cstdio>
#include <sstream>
#include <fstream>
#include <algorithm>

using namespace std;

int main()
{
  long long int A[110000];
	long long int keep;
	bool check=false;
  int n;
	cin>>n;
	for( int i=0; i<n; i++) {
	  cin>>A[i];
	}
	if(n==1) { cout<<-1<<endl; return 0;}
	sort(A, A+n);
	int diff=A[1]-A[0];
	if(n>2) {
	int diff2=A[2]-A[1];
	diff= diff<diff2? diff:diff2;
	}
	for( int i=0; i<n-1; i++) {
	  if(A[i+1]-A[i]!=diff) {
		  if(check || A[i+1]-A[i]!=2*diff) {
		    cout<<0<<endl; return 0;
		  }
			check=true;
			keep= (A[i+1]+A[i])/2;
	  }
	}
	if(diff==0) {
	  cout<<1<<endl;
	  cout<<A[0]<<endl;
    return 0;
  }
	if(check)
	{
	  cout<<1<<endl;
		cout<<keep<<endl;
		return 0;
  }
	int ans=2;
	if(n==2 && (A[1]-A[0])%2==0) {
	  ans++;
		cout<<ans<<endl;
		cout<<A[0]-diff<<" "<<(A[1]+A[0])/2<<" "<<A[1]+diff<<endl;
	}
	else {
	  cout<<2<<endl;
		cout<<A[0]-diff<<" "<<A[n-1]+diff<<endl;
	}
}
