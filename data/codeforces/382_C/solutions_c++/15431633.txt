#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;

int n,m,s;
__int64 x[100005],ans[100005],q,Q;
map<__int64,int> mp;

int main()
{
	int i,yes;
	scanf("%d",&n);
	for(i=0;i<n;i++) scanf("%I64d",&x[i]);
	sort(x,x+n);
	if(n==1)
	{
		printf("-1\n");
		return 0;
	}
	else if(n==2)
	{
		q=x[1]-x[0];
		ans[m++]=x[0]-q;
		ans[m++]=x[1]+q;
		if(q%2==0) ans[m++]=x[0]+q/2;
	}
	else
	{
		mp.clear();
		q=x[1]-x[0];
		for(i=1;i<n;i++)
		{
			mp[x[i]-x[i-1]]++;
			if(x[i]-x[i-1]!=q) Q=x[i]-x[i-1],s=i;
			if(mp.size()>2) break;
		}
		if(mp.size()==1)
		{
			ans[m++]=x[0]-q;
			ans[m++]=x[n-1]+q;
		}
		else if(mp.size()==2)
		{
			if(mp[q]==1&&q==Q*2)
			{
				ans[m++]=x[1]-Q;
			}
			if(mp[Q]==1&&Q==q*2)
			{
				ans[m++]=x[s]-q;
			}
		}
	}
	sort(ans,ans+m);
	m=unique(ans,ans+m)-ans;
	printf("%d\n",m);
	for(i=0;i<m;i++) printf("%I64d%s",ans[i],(i<m-1)?" ":"\n");
	return 0;
}
