#include<iostream>
#include<algorithm>
using namespace std;
int n, num[100000], diff = -1, dev=-1, df[100000];
int main() {
	cin >> n;
	for(int i=0;i<n;i++)
		cin >> num[i];
	sort(num, num+n);
	for(int i=0;i<n-1;i++)
		df[i] = num[i+1]-num[i];
	sort(df, df+n-1);
	if(n==1)
		cout << "-1\n";
	else if(n==2) {
		diff = num[1]-num[0];
		if(diff == 0) {
			cout << "1\n" << num[0] << '\n';
			return 0;
		}
		if(diff%2 == 0)
			cout << "3\n";
		else
			cout << "2\n";
		cout << num[0]-diff << ' ';
		if(diff%2 == 0)
			cout << num[0] + diff/2 << ' ';
		cout << num[1] + diff << '\n'; 
	}
	else {
		if(df[n-2] != df[0] && (df[n-3] != df[0] || df[n-2] != df[0]*2)) {
			cout << "0\n";
			return 0;
		}
		diff = df[0];
		for(int i=1;i<n;i++)
			if(num[i] - num[i-1] != diff) {
				cout << 1 << '\n' << num[i-1] + diff << '\n';
				return 0;
			}
		if(diff == 0)
			cout << "1\n" << num[0] << '\n';
		else
			cout << 2 << '\n' << num[0] - diff << ' ' << num[n-1]+diff << '\n';
	}
	return 0;
}
