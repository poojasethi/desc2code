#include <bits/stdc++.h>

using namespace std;

const int N = 100001;
int a[N], b[N];

int main()
{
    int i,n;
    cin >> n;
    for ( i = 0; i < n; i++)
        cin >> a[i];
    sort(a, a + n);
    for ( i = 0; i < n - 1; i++)
        b[i] = a[i + 1] - a[i];

    int mx=distance(b, max_element(b, b+n-1));
    sort(b, b + n - 1);

    if (n == 1)
        cout << "-1\n";
    else if (a[0] == a[n - 1])
        cout << "1\n" << a[0] << '\n';
    else if (n == 2)
        if (a[1] - a[0] & 1)
            cout << "2\n" << 2 * a[0] - a[1] << ' ' << 2 * a[1] - a[0] << '\n';
        else
            cout << "3\n" << 2 * a[0] - a[1] << ' ' << (a[0] + a[1]) / 2 << ' ' << 2 * a[1] - a[0] << '\n';
    else if (b[0] == b[n - 2])
        cout << "2\n" << a[0] - b[0] << ' ' << a[n - 1] + b[0] << '\n';
    else if (b[0] == b[n - 3] && b[n - 2] == 2 * b[0])
        cout << "1\n" << a[mx] + b[0] << '\n';
    else
        cout << "0\n";
    return 0;
}
