#include <bits/stdc++.h>
#define maxn 100020

using namespace std;

map <int,int> used;
vector <int> w;
int a,ar[maxn];

int main()
{
	scanf("%d",&a);
	if( a==1 ) {
		printf("-1\n");
		return 0;
	}
	for(int i=1;i<=a;i++)
		scanf("%d",&ar[i]);
	
	sort( ar+1 , ar+1+a );
	
	int m=0 , g = ar[2] - ar[1];
	
	for(int i=1,k=ar[1];i<=a;i++,k+=g)
		if( ar[i]!=k ) {
			if( ar[i] == k+g && !m ) {
				m=1;
				i--;
				used[k]=1;
				w.push_back( k );
			}
			else {
				w.clear();
				m=1;
				break;
			}
		}
	
	if( !m ) {
		if( !used[ ar[1]-g ] )
			w.push_back( ar[1] - g ),used[ ar[1]-g ]=1;
		if( !used[ ar[a]+g ] )
			w.push_back( ar[a] + g ),used[ ar[a]+g ]=1;
	}
	
	if( g%2==0 ) {
		int c=0;
		for(int i=2,k=ar[2];i<=a;i++,k+=g/2)
			if( ar[i]!=k ) { c=1; break; }
		if( !c && !used[ ar[1]+g/2 ] ) w.push_back( ar[1]+g/2 ),used[ ar[1]+g/2 ]=1;
	}
	sort( w.begin() , w.end() );
	printf("%d\n",(int)w.size());
	for(int i=0;i<w.size();i++)
		printf("%d ",w[i]);
	if( w.size() )
		printf("\n");
	return 0;
}
