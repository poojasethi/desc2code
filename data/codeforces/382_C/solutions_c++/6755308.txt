#include<bits/stdc++.h>

using namespace std;

int a,ar[100005],diff[100005],dif,apa;


int main(){
	bool cek1=true;
	scanf("%d",&a);
	scanf("%d",&ar[0]);
	for(int x=1;x<a;x++) {
		scanf("%d",&ar[x]);
		if(ar[x]!=ar[x-1]) cek1=false;
	}
	sort(ar,ar+a);
	if(a==1 ){
		cout<<"-1\n";
		return 0;
	}
	if(cek1){
		cout<<1<<endl;
		cout<<ar[0]<<endl;
		return 0;
	}
	dif=ar[1]-ar[0];
	if(a==2 && dif%2==0){
		cout<<3<<endl;
		cout<<ar[0]-dif<<' '<<ar[0]+dif/2<<' '<<ar[1]+dif<<endl;
		return 0;
	}
	if(a==2 && dif%2!=0){
		cout<<2<<endl;
		cout<<ar[0]-dif<<' '<<ar[1]+dif<<endl;
		return 0;
	}
	dif=0;
	for(int x=0;x<a-1;x++){
			diff[x] = ar[x+1] - ar[x];
            if(diff[x]>dif){
                dif=diff[x];
                apa=x;
            }
	}
	sort(diff,diff+a-1);
    if(diff[0]==diff[a-2]){
		cout<<2<<endl;
		cout<<ar[0]-diff[0]<<' '<<ar[a-1]+diff[0]<<endl;
    	return 0;
	} 
    if(diff[0]==diff[a-3] && diff[a-2]==2*diff[0]){
		cout<<1<<endl;
		cout<<ar[apa]+diff[0]<<endl;
		return 0;
    }
    cout<<0<<endl;
}
