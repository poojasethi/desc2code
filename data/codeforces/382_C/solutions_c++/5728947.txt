#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cstdlib>
#define N 100010
using namespace std;
int sum[10],tot,n,num[N],d[N],ans,way[2*N];
int main()
{
 scanf("%d",&n);
 for (int i=1;i<=n;i++)
  scanf("%d",&num[i]);
 if (n==1) {printf("-1\n");return 0;} 
 sort(num+1,num+1+n);
 for (int i=1;i<n;i++)
  d[i]=num[i+1]-num[i];
 sort(d+1,d+n);
 tot=unique(d+1,d+n)-(d+1);
 if (tot>2) {printf("0\n");return 0;}
 for (int i=1;i<n;i++)
  if (num[i+1]-num[i]==d[1]) sum[1]++;
   else sum[2]++;
 if (sum[2]==0) {ans+=2;way[1]=num[1]-d[1],way[2]=num[n]+d[1];}
 for (int i=1;i<n;i++)
  if (tot==2)
  {
   if (d[1]*2==d[2]&&num[i+1]-num[i]==2*d[1]&&sum[2]==1) way[++ans]=num[i]+d[1];
   if (d[2]*2==d[1]&&num[i+1]-num[i]==2*d[2]&&sum[1]==1) way[++ans]=num[i]+d[2];
  }else if (sum[1]==1&&d[1]%2==0)way[++ans]=num[1]+d[1]/2;
  
 sort(way+1,way+1+ans);
 ans=unique(way+1,way+1+ans)-(way+1);
 printf("%d\n",ans);
 for (int i=1;i<=ans;i++)
  printf("%d ",way[i]);
 
 return 0;
}
