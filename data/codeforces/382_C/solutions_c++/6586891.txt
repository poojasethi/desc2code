#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int x[1000005];
int main()
{
    int n,m,ans;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    scanf("%d",&x[i]);
    sort(x,x+n);
    m=x[1]-x[0];
    int flag=0;
    for(int i=0;i<n-1;i++)
    {
        if(x[i]+m!=x[i+1])
        {
            if(!flag&&i==1&&x[i+1]-x[i]==m/2)
            {
                flag=1;
                m/=2;
                ans=x[0]+m;
            }
            else if(!flag&&x[i+1]-x[i]==m*2)
            {
                flag=1;
                ans=x[i]+m;
            }
            else
            flag=2;
        }
    }
    if(n==1)
    puts("-1");
    else if(flag==2)
    puts("0");
    else if(flag==1)
    printf("1\n%d\n",ans);
    else
    {
        if(m==0)
        printf("1\n%d\n",x[0]);
        else if(n==2&&m%2==0){
        printf("3\n");
        printf("%d %d %d\n",x[0]-m,x[0]+m/2,x[1]+m);
        }
        else
        printf("2\n%d %d\n",x[0]-m,x[n-1]+m);
    }
}
