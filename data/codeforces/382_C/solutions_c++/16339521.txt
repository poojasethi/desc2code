#include<cstdio>
#include<algorithm>
using namespace std;

#define ran 111111

int n;
int a[ran];
int ans[ran], tot;

int main() {
	scanf("%d", &n);
	for(int i=1;i<=n;i++) scanf("%d", &a[i]);
	if(n==1) {
		printf("-1\n");
		return 0;
	}
	sort(a+1,a+n+1);
	tot=0;
	if(n==2) {
		ans[tot++] = 2*a[1]-a[2];
		ans[tot++] = 2*a[2]-a[1];
		if((a[2]-a[1])%2==0)
			ans[tot++] = (a[1]+a[2])/2;
	} else {
		int min_step = 111111111, max_step = -111111111;
		for(int i=2;i<=n;i++) {
			min_step = min(min_step, a[i]-a[i-1]);
			max_step = max(max_step, a[i]-a[i-1]);
		}
		if(min_step == max_step) {
			ans[tot++] = 2*a[1]-a[2];
			ans[tot++] = 2*a[n]-a[n-1];
		} else if(min_step*2==max_step) {
			for(int i=2;i<=n;i++)
				if(a[i]-a[i-1]!=min_step)
					ans[tot++] = (a[i]+a[i-1])/2;
			if(tot>1) tot=0;
		}
	}
	sort(ans, ans+tot);
	tot = unique(ans, ans+tot) - ans;
	printf("%d\n", tot);
	for(int i=0;i<tot;i++)
		printf("%d%c", ans[i], i<tot-1 ? ' ' : '\n');
	
	return 0;
}
