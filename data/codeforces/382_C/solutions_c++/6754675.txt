#include <bits/stdc++.h>
using namespace std;
long long n,arr[100005],beda,temp,hasil,arr1[100005],mx,maks;
vector<long long> v;
bool cek;
int main(){
	cin>>n;
	for(int i=0;i<n;i++) cin>>arr[i];
	sort(arr,arr+n);
	hasil=0;
	if(n==1) cout<<-1<<endl;else
	if(arr[0]==arr[n-1]){
		cout<<1<<endl;
		cout<<arr[0]<<endl;
	}else
	if(n==2){
		if((arr[1]-arr[0])%2==0){
			cout<<3<<endl;
			beda=arr[1]-arr[0];
			cout<<arr[0]-beda<<" "<<arr[0]+beda/2<<" "<<arr[1]+beda<<endl;
		}else{
			cout<<2<<endl;
			beda=arr[1]-arr[0];
			cout<<arr[0]-beda<<" "<<arr[1]+beda<<endl;
		}
	}else{
		for(int i=0;i<n-1;i++){
			arr1[i] = arr[i + 1] - arr[i];
            if(arr1[i]>maks){
                maks=arr1[i];
                mx=i;
            }
		}
		sort(arr1,arr1+n-1);
        if(arr1[0]==arr1[n-2]) cout<<2<<'\n'<<arr[0]-arr1[0]<<" "<<arr[n-1]+arr1[0];
        else if(arr1[0]==arr1[n-3] && arr1[n-2]==2*arr1[0]) cout<<1<<'\n'<<arr[mx]+arr1[0];
        else cout<<0;
	}
}
