#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>
using namespace std;
int a[100000];
int main()
{
	int n;
	bool f=1;
	int t;
	vector <int> ans;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	if(n==1)
	{
		cout<<-1<<endl;
		return 0;
	}
	sort(a,a+n);
	if(a[n-1]-a[0]==0)
	{
		cout<<1<<endl;
		cout<<a[0]<<endl;
		return 0;
	}
	t=a[1]-a[0];
	for(int i=2;i<n;i++)
	{
		if(a[i]-a[i-1]!=t)
		{
				f=0;
				break;
		}
	}
	if(f)
	{
		ans.push_back(a[0]-t);
		ans.push_back(a[n-1]+t);
	}
	if((a[n-1]-a[0])%n==0)
	{
		int d=(a[n-1]-a[0])/n;
		bool s=0;
		int p;
		f=1;
		for(int i=1;i<n;i++)
		{
			if(a[i-1]+d!=a[i])
			{
				if(s)
				{
					f=0;
					break;
				}
				if(a[i-1]+d==a[i]-d)
					p=a[i-1]+d;
				s=1;
			}
		}
		if(f)
			ans.push_back(p);
	}
	cout<<ans.size()<<endl;
	if(ans.size()==0)
	{
		return 0;
	}
	sort(ans.begin(),ans.end());
	for(int i=0;i<ans.size()-1;i++)
		cout<<ans[i]<<' ';
	cout<<ans[ans.size()-1]<<endl;
	return 0;
}
