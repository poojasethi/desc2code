import sys
import os

def main():
    N = int(sys.stdin.readline())
    data = map(int, sys.stdin.readline().strip().split())
    data.sort()
    if len(data) == 1:
        print -1
        return

    if data[0] == data[-1]:
        print 1
        print data[0]
        return

    if len(data) == 2:
        diff = data[1] - data[0]
        if (data[0] + data[1]) & 1:
            print 2
            print data[0] - diff, data[1] + diff
        else:
            print 3
            print data[0] - diff, sum(data)/2, data[1] + diff
        return

    d = dict()
    for i in xrange(1, len(data)):
        diff = data[i] - data[i-1]
        if diff in d:
            d[diff][1] += 1
        else:
            d[diff] = [i, 1]
            if len(d) > 2:
                print 0
                return

    if len(d) == 1:
        diff = d.keys()[0]
        print 2
        print data[0] - diff, data[-1] + diff
        return

    diff0, diff1 = d.items()
    if diff0[0] > diff1[0]:
        diff0, diff1 = diff1, diff0
    if diff1[1][1] > 1 or diff0[0] * 2 != diff1[0]:
        print 0
        return
    print 1
    print (data[diff1[1][0]] + data[diff1[1][0] - 1]) / 2


if __name__ == '__main__':
    main()
