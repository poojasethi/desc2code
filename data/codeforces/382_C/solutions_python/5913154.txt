#!/usr/bin/env python

import sys

sys.stdin.readline()
ns = map(int, sys.stdin.readline().strip().split())
ns.sort()

ans = set()
if len(ns) == 1:
    print -1
elif len(ns) == 2:
    ans = set()
    a, b = tuple(ns)
    ans.add(b + b - a)
    ans.add(a - (b - a))
    if (b - a) % 2 == 0:
        ans.add(a + (b - a) / 2)
    ans = sorted(ans)
    print len(ans)
    print " ".join(map(str, ans))
else:
    n = len(ns)
    def candidate(diff):
        ret = None
        cnt = 0
        for i in range(n - 1):
            if ns[i] + diff != ns[i + 1]:
                if ns[i + 1] - ns[i] != 2 * diff:
                    return None
                cnt += 1
                ret = ns[i] + diff
        if cnt == 1:
            return ret
        elif cnt == 0:
            return True
        else:
            return None
    ans = set()
    e = candidate(ns[1] - ns[0])
    if e is not None:
        if isinstance(e, bool):
            ans.add(ns[0] - ns[1] + ns[0])
            ans.add(ns[-1] + ns[-1] - ns[-2])
        else:
            ans.add(e)
    e = candidate(ns[-1] - ns[-2])
    if e is not None:
        if isinstance(e, bool):
            ans.add(ns[0] - ns[1] + ns[0])
            ans.add(ns[-1] + ns[-1] - ns[-2])
        else:
            ans.add(e)
    ans = sorted(ans)
    print len(ans)
    print " ".join(map(str, ans))
