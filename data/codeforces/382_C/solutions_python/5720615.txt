#!/usr/bin/env python2
n = int(raw_input())
a = sorted(map(int, raw_input().split()))
D, T, ans = {}, {}, []
if n == 1: print -1
elif n == 2: 
	d = a[-1]-a[0]
	ans.append(a[0]-d)
	ans.append(a[-1]+d)
	if d&1 == 0:
		ans.append(a[0]+d/2)
	ans = list(set(ans))
	print len(ans)
	print ' '.join(map(str, sorted(ans)))
else:
	y, d = a[0], 0
	for x in a[1:]:
		d, y = x-y, x
		D[d] = D.get(d, 0)+1
		T[d] = x-d
		if len(D) > 2:
			break
	if len(D) > 2: print 0
	elif len(D) == 2:
		if 1 not in D.values():
			print 0
		else:
			d = min(D.keys())
			if max(D.keys()) == d*2 and D[d*2] == 1:
				ans.append(T[d*2]+d)
			ans = list(set(ans))
			print len(ans)
			if len(ans):
				print ' '.join(map(str, sorted(ans)))
	elif len(D) == 1:
		ans.append(a[0]-d)
		ans.append(a[-1]+d)
		ans = list(set(ans))
		print len(ans)
		print ' '.join(map(str, sorted(ans)))
