#include<iostream>
#include<cstdio>
using namespace std;

double pr[1001][1001];
int a[1001];

int n, m;

int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++)
			{
				if(a[i]>a[j]) pr[i][j]=1.0;
			}	
	}
	for(int i=0;i<m;i++){
		int u, v;
		cin>>u>>v;
		
		for(int j=1;j<=n;j++){
			if(j!=u && j!=v){
				pr[j][v]=pr[j][u]=  (pr[j][u]+pr[j][v])/2.0;
				pr[v][j]=pr[u][j]=  1.0-pr[j][u];
				
			}	
		}
		
		pr[u][v]=pr[v][u]=0.5;
	}
	double ans=0.0;
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
			ans += pr[i][j];
	printf("%.7lf\n",ans);
//	system("PAUSE");
}
