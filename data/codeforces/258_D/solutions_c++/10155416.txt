#include <cstdio>
#include <iostream>
#define N 1001
using namespace std;
int n,m,p[N];
double f[N][N];
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;++i)scanf("%d",&p[i]);
	//f[i][j]��ʾp[i]>p[j]�ĸ��� 
	for(int i=1;i<=n;++i)
	    for(int j=1;j<=n;++j)
		    if(i!=j)f[i][j]=p[i]>p[j] ? 1:0;
	for(int i=1,a,b;i<=m;++i){
		scanf("%d%d",&a,&b);
		f[a][b]=f[b][a]=0.5;
		for(int j=1;j<=n;++j)
		    if(j!=a && j!=b){
			    f[j][a]=f[j][b]=f[j][a]*0.5+f[j][b]*0.5;
			    f[a][j]=f[b][j]=1-f[j][a];
		    }
	}
	double ans=0;
	for(int i=1;i<n;++i)
	    for(int j=i+1;j<=n;++j)ans+=f[i][j];
	printf("%.9lf",ans);
}
