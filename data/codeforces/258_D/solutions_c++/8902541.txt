#include<stdio.h>
double p[1001][1001],ans;
int n,m,a,b,x[1001];
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)scanf("%d",&x[i]);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
			p[i][j]=x[i]>x[j];
	while(m--)
	{
		scanf("%d%d",&a,&b);
		for(int i=1;i<=n;i++)
		if(i!=a&&i!=b)
		{
			p[a][i]=p[b][i]=(p[a][i]+p[b][i])/2;
			p[i][a]=p[i][b]=1-p[a][i];
		}
		p[a][b]=p[b][a]=0.5;
	}
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
			ans+=p[i][j];
	printf("%.9lf\n",ans);
}
