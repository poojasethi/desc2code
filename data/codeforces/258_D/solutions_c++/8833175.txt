#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define N 2333
using namespace std;
int n,m;
int d[N];
double f[N][N],ans;
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) scanf("%d",&d[i]);
	
	
	for(int i=1;i<=n;i++)
	for(int j=1;j<=n;j++)
	if(d[i]>d[j]) f[i][j]=1;
	
	
	while(m--){
		int x,y; scanf("%d%d",&x,&y);
		
		f[x][y]=f[y][x]=0.5;
		
		for(int i=1;i<=n;i++)
		if(i!=x&&i!=y){
			f[i][x]=f[i][y]=(f[i][x]+f[i][y])/2;
			f[x][i]=f[y][i]=(f[x][i]+f[y][i])/2;
		}
		
	}
	
	for(int i=1;i<=n;i++)
	for(int j=i+1;j<=n;j++) ans+=f[i][j];
	printf("%.8lf\n",ans);
	
}
