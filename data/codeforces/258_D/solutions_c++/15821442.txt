#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <string>
#include <cmath>
#define MAXN 1011
using namespace std;
int n,m;
double a[MAXN],p[MAXN][MAXN],ans;
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++)
		scanf("%lf",&a[i]);
	for (int i=1;i<=n;i++)
		for (int j=1;j<=n;j++)
			p[i][j]=a[i]>a[j];
	int x,y;
	while(m--)
	{
		scanf("%d%d",&x,&y);
		p[x][y]=p[y][x]=0.5;
		for (int i=1;i<=n;i++)
			if(i!=x&&i!=y)
			{
				p[i][x]=p[i][y]=(p[i][x]+p[i][y])/2;
				p[x][i]=p[y][i]=(p[x][i]+p[y][i])/2;
			}
	}
	for (int i=1;i<=n;i++)
		for (int j=i+1;j<=n;j++)
			ans+=p[i][j];
	printf("%lf",ans);
	return 0;
}
