#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
#define db double
#define N 2100
#define foru(i,a,b) for(int i=a;i<=b;++i)

using namespace std;
int n,m,a[N],x,y;
db p[N][N],ans;
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	cin>>n>>m;
	foru(i,1,n) cin>>a[i];
	foru(i,1,n)
	  foru(j,1,n)
	    p[i][j]=a[i]>a[j];
	while(m--)
	{
		cin>>x>>y;
		p[x][y]=p[y][x]=0.5;
		foru(i,1,n)
		  if(x!=i&&y!=i)
		    p[x][i]=p[y][i]=(p[x][i]+p[y][i])/2,
		    p[i][x]=p[i][y]=(p[i][x]+p[i][y])/2;
	}
	foru(i,1,n-1)
	  foru(j,i+1,n)
	    ans+=p[i][j];
	printf("%.10lf\n",ans);
	return 0;
}
