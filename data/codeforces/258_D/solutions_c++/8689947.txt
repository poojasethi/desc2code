#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <string>
using namespace std;
const int Maxn=1005;
int n,m,a[Maxn];
double p[Maxn][Maxn];

int main() {
	cin>>n>>m;
	for (int i=1;i<=n;i++) cin>>a[i];
	for (int i=1;i<=n;i++)
		for (int j=1;j<=n;j++) p[i][j]=a[i]>a[j];
	while (m--) {
		int a,b;
		double t;
		cin>>a>>b;
		p[a][b]=p[b][a]=0.5;
		for (int i=1;i<=n;i++)
			if (i!=a&&i!=b) {
				t=(p[a][i]+p[b][i])*0.5;
				p[a][i]=p[b][i]=t;
				p[i][a]=p[i][b]=1-t;
			}
	}
	double ans=0;
	for (int i=1;i<n;i++)
		for (int j=i+1;j<=n;j++) ans+=p[i][j];
	printf("%.8lf\n",ans);
	return 0;
}
