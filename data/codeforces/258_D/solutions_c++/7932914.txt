#include<cstdio>
#include<algorithm>
#include<iostream>
#include<vector>
#include<cstring>
using namespace std;
double pro[1001][1001];
int n,m,a[1001];
double ans;
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
			pro[i][j]=(a[i]>a[j]),pro[j][i]=1-pro[i][j];
	while(m--){
		int x,y;
		scanf("%d%d",&x,&y);
		pro[x][y]=pro[y][x]=0.5;
		for(int i=1;i<=n;i++)
			if(i!=x&&i!=y){
				pro[i][x]=pro[i][y]=(pro[i][x]+pro[i][y])/2.0;
				pro[x][i]=pro[y][i]=(pro[x][i]+pro[y][i])/2.0;
			}
	}for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
			ans+=pro[i][j];
	printf("%.9f\n",ans);
}
