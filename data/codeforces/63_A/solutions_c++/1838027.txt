using namespace std;

{
  int n,i;
  string a[101],b[101];
  cin>>n;
  for(i=0;i<n;i++) cin>>a[i]>>b[i];

  for(i=0;i<n;i++) if (b[i]=="man") cout<<a[i]<<endl;
  for(i=0;i<n;i++) if (b[i]=="captain") cout<<a[i]<<endl;
//  system("pause");
  return 0;
}
