#include <stdio.h>

int n;
int a[1000];
char b[1000][100];

int main()
{
scanf("%d",&n);
int i, k;
char j[1000];
for(i=1; i<=n; i++)
{
scanf("%s",b[i]);
scanf("%s",j);
if(j[0]=='r')
a[i]=1;
else if(j[0]=='w' || j[1] == 'h')
a[i]=2;
else if(j[0]=='m')
a[i]=3;
else if(j[0]=='c')
a[i]=4;
}

for(k=1; k<=4; k++)
{
for(i=1; i<=n; i++)
{
if(a[i]==k)
printf("%s\n",b[i]);
}
}

return 0;

}
