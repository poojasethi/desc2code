#include<iostream>
#include<vector>
using namespace std;

int n;
vector<string> ary[4];

int main(){
	cin >> n;
	for(int i=0;i<n;i++){
		string s, s2;
		cin >> s >> s2;
		if(s2=="rat")
			ary[0].push_back(s);
		if(s2=="woman" || s2=="child")
			ary[1].push_back(s);
		if(s2=="man")
			ary[2].push_back(s);
		if(s2=="captain")
			ary[3].push_back(s);
	}
	for(int i=0;i<4;i++)
		for(int j=0;j<ary[i].size();j++)
			cout << ary[i][j] << endl;
	return 0;
}
