#include<iostream>
#include<map>
using namespace std;
int main()
{
    int n,i,j;
    cin>>n;
    char s[n][2][11];
    for(i=0;i<n;i++)cin>>s[i][0]>>s[i][1];

    map<string,int>mp;

    mp["rat"]=0,mp["child"]=1,mp["woman"]=5,mp["man"]=2;mp["captain"]=3;
    for(i=0;i<4;i++){
        for(j=0;j<n;j++){
            if(mp[s[j][1]]%4==i)cout<<s[j][0]<<endl;
        }
    }

    return 0;

    }
