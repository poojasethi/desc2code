#include <bits/stdc++.h>
using namespace std;

int main()
{
	vector<string> v[4];
	string s,t;
	int n;
	cin >> n;
	while(n--)
	{
		cin >> s >> t;
		if(t=="rat")
			v[0].push_back(s);
		else if(t=="woman"||t=="child")
			v[1].push_back(s);
		else if(t=="man")
			v[2].push_back(s);
		else
			v[3].push_back(s);
	}
	for(int i=0;i<4;++i)
		for(int j=0;j<v[i].size();++j)
			cout << v[i][j] << endl;
}
