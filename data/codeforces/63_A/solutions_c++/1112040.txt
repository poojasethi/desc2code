#include <iostream>
#include <string>

using namespace std;

int a, i, z[11111], j;
string s[11111], q;

int main ()
{
	cin >> a;
		for (i = 1; i <= a; i++)
			{
			cin >> s[i] >> q;
			if (q == "rat") z[i] = 0;
			else if (q == "woman" || q == "child") z[i] = 1;
			else if (q == "captain") z[i] = 3;
			else z[i] = 2;
            }
   for (j = 0; j <= 3; j++)
    	for (i = 1; i <= a; i++)
    		if (j == z[i]) cout << s[i] << endl;
}
