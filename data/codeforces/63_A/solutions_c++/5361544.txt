#include <string>
#include <iostream>
using namespace std;
int i,j,n;
int b[102];
string a[102],x;
int main()
{ cin>>n;
  for (i=1;i<=n;i++)
  { cin>>a[i]>>x;
    if (x=="rat") b[i]=1;
    else if ((x=="woman")||(x=="child")) b[i]=2;
         else if (x=="man") b[i]=3;
              else b[i]=4;
  }
  for (i=1;i<=4;i++)
   for (j=1;j<=n;j++)
   if (b[j]==i) cout<<a[j]<<endl;
  return 0;
}
