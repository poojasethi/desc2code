#include<cstdio>

int main(){
    char a[101][15],b[101][15];
    int n;
    scanf("%d ",&n);
    for(int i=0;i<n;i++) scanf("%s%s",a[i],b[i]);
    for(int i=0;i<n;i++) if(b[i][2]=='t') puts(a[i]);
    for(int i=0;i<n;i++) if(b[i][2]=='m'||b[i][2]=='i') puts(a[i]);
    for(int i=0;i<n;i++) if(b[i][2]=='n') puts(a[i]);
    for(int i=0;i<n;i++) if(b[i][2]=='p') puts(a[i]);
    return 0;
}
