#include <iostream>
#include <string>
#include <map>
using namespace std;
int main ()
{
	int n;
	string name,type;
	map <string , string>a;
	cin>>n;
		a["rat"]="";a["woman"]="";a["man"]=""; a["captain"]="";
	while (n--)
	{
	
		cin>>name>>type;
		
		if (type=="child")
		type="woman";	
			
		a[type]+=name+"\n";
	
	}
	cout<<a["rat"]<<a["woman"]<<a["man"]<<a["captain"];
	
}
