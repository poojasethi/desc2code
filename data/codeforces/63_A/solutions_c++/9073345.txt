#include <bits/stdc++.h>
using namespace std;
string s,wc[100],m[100],r[100],c,e,t; int n,i,wci,mi,ri;
main(){
    for(cin>>n;i<n;i++){cin>>t>>s;
    if(!s.compare("rat"))r[ri++]=t;
    else if(!s.compare("woman")|!s.compare("child"))wc[wci++]=t;
    else if(!s.compare("man"))m[mi++]=t; else c=t;
    }
    for(i=0;i<ri;i++)cout<<r[i]<<'\n';
    for(i=0;i<wci;i++)cout<<wc[i]<<'\n';
    for(i=0;i<mi;i++)cout<<m[i]<<'\n';
    if(c!=e)cout<<c;
}
