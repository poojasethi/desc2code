#include <cstdio>
char s[100][11],t[100][8];
int n,i;
int main (){
    scanf("%d",&n);
    for (i=0;i<n;i++) scanf("%s %s",s[i],t[i]);
    for (i=0;i<n;i++) if (t[i][2]=='t') printf("%s\n",s[i]);
    for (i=0;i<n;i++) if (t[i][2]=='i' || t[i][2]=='m') printf("%s\n",s[i]);
    for (i=0;i<n;i++) if (t[i][2]=='n') printf("%s\n",s[i]);
    for (i=0;i<n;i++) if (t[i][2]=='p') printf("%s\n",s[i]);
    return 0;
}
