#include<iostream>
using namespace std;
int main()
{
int n,i,a[105],j;
string s[105],t;
cin>>n;
for(i=1;i<=n;i++)
{
cin>>s[i]>>t;
if(t=="rat") a[i]=0;
else if(t=="child" || t=="woman") a[i]=1;
else if(t=="captain") a[i]=3;
else a[i]=2;
}
for(j=0;j<=3;j++)
{
for(i=1;i<=n;i++)
{
if(a[i]==j) cout<<s[i]<<endl;
}
}
return 0;
}
