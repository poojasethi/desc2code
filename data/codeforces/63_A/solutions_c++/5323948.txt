#include <iostream>
#include <cstring>
using namespace std;
string s[200],c[200];
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>s[i]>>c[i];
	for(int i=1;i<=n;i++)
		if(c[i]=="rat")
			cout<<s[i]<<endl;
	for(int i=1;i<=n;i++)
		if(c[i]=="woman" || c[i]=="child")
			cout<<s[i]<<endl;
	for(int i=1;i<=n;i++)
		if(c[i]=="man")
			cout<<s[i]<<endl;
	for(int i=1;i<=n;i++)
		if(c[i]=="captain")
			cout<<s[i]<<endl;
	return 0;
}
