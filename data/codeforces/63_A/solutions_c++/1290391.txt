#include <iostream>
#include <string>
#include <vector>

using namespace std;

vector<string> a[4];

int main() {
	int n;
	cin >> n;
	for(int i=0;i<n;i++) {
		string s,t;
		cin >> s >> t;
		if (t=="rat") a[0].push_back(s);
		if (t=="child" || t=="woman") a[1].push_back(s);
		if (t=="man") a[2].push_back(s);
		if (t=="captain") a[3].push_back(s);
	}
	for(int i=0;i<4;i++) for(int j=0;j<a[i].size();j++) cout << a[i][j] << endl;
	return 0;
}
