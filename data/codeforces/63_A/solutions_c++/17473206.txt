#include <bits/stdc++.h>

using namespace std;

int main()
{
    int x;
    cin>>x;
    string s1,s2,sr="",sm="",sw="",sc="";
    for(int i=0;i<x;i++)
    {
        cin>>s1>>s2;
        if(s2=="rat")
            sr+=s1+"\n";
        else if (s2=="man")
            sm+=s1+"\n";
        else if(s2=="woman"||s2=="child")
            sw+=s1+"\n";
        else
            sc+=s1+"\n";
    }
    cout<<sr<<sw<<sm<<sc<<endl;
    return 0;
}
