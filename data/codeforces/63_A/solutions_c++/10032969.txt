#include<bits/stdc++.h>
using namespace std;
typedef long long int lld;
int main()
{
	lld i,j,n;
	cin>>n;
	lld a[n];
	string str,s[n];
	for(i=0;i<n;i++)
	{
		cin>>s[i]>>str;
		if(str=="rat")
			a[i]=0;
		else if(str=="child" || str=="woman")
			a[i]=1;
		else if(str=="man")
			a[i]=2;
		else
			a[i]=3;
	}
	for(j=0;j<=3;j++)
	{
		for(i=0;i<n;i++)
		{
			if(a[i]==j)
				cout<<s[i]<<endl;
		}
	}
	return 0;
}
