#include <cstdio>
int main() {
    int n;
    char na[110][20], s[110][20];
    scanf("%d", &n);
    for (int i = 0;i < n;i++) scanf("%s%s", na[i], s[i]);
    for (int i = 0;i < n;i++) if (s[i][0] == 'r') puts(na[i]);
    for (int i = 0;i < n;i++) if (s[i][1] == 'o' || s[i][1] == 'h') puts(na[i]);
    for (int i = 0;i < n;i++) if (s[i][0] == 'm') puts(na[i]);
    for (int i = 0;i < n;i++) if (s[i][2] == 'p') puts(na[i]);
}
