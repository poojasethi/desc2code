n = int(raw_input())
crew = {
    'captain':[],
    'woman':[],
    'man':[],
    'rat':[],
    }
while n>0:
    name, cat = raw_input().split()
    if cat=='child':
        cat = 'woman'
    crew[cat].append(name)
    n-=1

for cat in ['rat', 'woman', 'man', 'captain']:
    for n in crew[cat]:
        print n
