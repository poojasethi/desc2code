#include <iostream>
#include <set>
#include <cstdio>
using namespace std;
int a[55][55];
int ans;
int sum(int aa,int b,int c,int d){
	int res=0;
	for (int i=aa;i<=c;i++){
		for (int j=b;j<=d;j++){
			res+=a[i][j];
		}
	}
	return res;
}
int main(){
	int n,m;
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	cin>>n>>m;
	for (int i=1;i<=n;i++){
		for (int j=1;j<=m;j++){
			cin>>a[i][j];
		}
	}
	multiset<int> A;
	for (int i=1;i<=3;i++){
		int t;
		cin>>t;
		A.insert(t);
	}
	
	for (int i=1;i<=n-1;i++){
		for (int j=i+1;j<=n-1;j++){
		    multiset<int> S;
			S.insert(sum(1,1,i,m));
			S.insert(sum(i+1,1,j,m));
			S.insert(sum(j+1,1,n,m));
			if (S==A){
				ans++;
			}
		}
	}
	for (int i=1;i<=m-1;i++){
		for (int j=i+1;j<=m-1;j++){
			multiset<int> S;
			S.insert(sum(1,1,n,i));
			S.insert(sum(1,i+1,n,j));
			S.insert(sum(1,j+1,n,m));
			if (S==A){
				ans++;
			}
		}
	}
	cout<<ans;
	return 0;
}
