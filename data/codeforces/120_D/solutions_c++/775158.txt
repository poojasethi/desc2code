#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;

int n, m, p[3], a[55][55], s[55], b[4], pa, pb, pc, ret;

void work(int x){
	s[0] = 0;
	for (int i=1; i<=n; i++){
		s[i] = s[i-1];
		for (int j=1; j<=m; j++){
			if (!x) s[i] += a[i][j];
			else s[i] += a[j][i];
		}
	}
	
	for (int i=1; i<n; i++){
		for (int j=i+2; j<=n; j++){
			b[1] = s[j-1] - s[i];
			b[2] = s[n] - s[j-1];
			b[0] = s[n] - b[1] - b[2];
			p[0] = 0;
			p[1] = 1;
			p[2] = 2;
			int ok = 0;
			do{
				if (b[p[0]]==pa && b[p[1]]==pb && b[p[2]]==pc) ok = 1;
			}   while (next_permutation(p, p+3));
			ret += ok;
		}
	}
}

int main(){
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

	scanf("%d%d", &n, &m);
	for (int i=1; i<=n; i++){
		for (int j=1; j<=m; j++) scanf("%d", &a[i][j]);
	}
	scanf("%d%d%d", &pa, &pb, &pc);
	
	work(0);
	swap(n, m);
	work(1);
	printf("%d\n", ret);
	return 0;
}
