#include<cstdio>
#include<algorithm>
using namespace std;

int n,m,a[55][55],s[55][55],A,B,C,ans;
int main()
{
	freopen("input.txt","r",stdin);	freopen("output.txt","w",stdout);
	int i,j;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{
			scanf("%d",&a[i][j]);
			s[i][j]=s[i][j-1]+s[i-1][j]+a[i][j]-s[i-1][j-1];
		}
	scanf("%d%d%d",&A,&B,&C);
	int x[3]={A,B,C};
	sort(x,x+3);
	do
	{
		for(i=1;i<=n;i++)
			if(s[i][m]==x[0])
				for(j=i+1;j<n;j++)
					if(s[j][m]-s[i][m]==x[1]&&s[n][m]-s[j][m]==x[2])
					{	
						//printf("%d %d\n",i,j);
						ans++;
					}
	}while(next_permutation(x,x+3));
	do
	{
		for(i=1;i<=m;i++)
			if(s[n][i]==x[0])
				for(j=i+1;j<m;j++)
					if(s[n][j]-s[n][i]==x[1]&&s[n][m]-s[n][j]==x[2])
						ans++;
	}while(next_permutation(x,x+3));
	printf("%d\n",ans);
}
