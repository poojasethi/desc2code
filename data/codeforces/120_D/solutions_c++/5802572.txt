#include <cstdio>
#include <iostream>
#include <set>
using namespace std;
int w[55][55];
int sum (int t,int l,int b,int r) {
	int i,j,s=0;
	for (i=t;i<b;i++) {
		for (j=l;j<r;j++) s+=w[i][j];
	}
	return s;
}
int main ( ) {
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int m,n,i,j,k,l,x,ans=0;
	scanf("%d%d",&m,&n);
	for (i=0;i<m;i++) {
		for (j=0;j<n;j++) scanf("%d",w[i]+j);
	}
	multiset <int> a;
	for (i=0;i<=2;i++) {
		scanf("%d",&x);
		a.insert(x);
	}
	for (i=1;i<m;i++) {
		for (k=i+1;k<m;k++) {
			multiset <int> s;
			s.insert(sum(0,0,i,n));
			s.insert(sum(i,0,k,n));
			s.insert(sum(k,0,m,n));
			if (s==a) ans++;
		}
	}
	for (j=1;j<n;j++) {
		for (l=j+1;l<n;l++) {
			multiset <int> s;
			s.insert(sum(0,0,m,j));
			s.insert(sum(0,j,m,l));
			s.insert(sum(0,l,m,n));
			if (s==a) ans++;
		}
	}
	printf("%d\n",ans);
	return 0;
}
