#include<bits/stdc++.h>
using namespace std;

int a,b,c,n,m,s,arr[100][100],satir[100],sutun[100];

bool kontrol(int x, int y, int z){
	if(x==a){
		if(y==b && z==c)
			return true;
		
		if(y==c && z==b)
			return true;
	}	
	if(x==b){
		if(y==a && z==c)
			return true;

		if(y==c && z==a)
			return true;
	}
	if(x==c){
		if(y>=a && z==b)
			return true;

		if(y==b && z==a)
			return true;
	}
//	if(x>=a && y>=b && z>=c)
//		return true;

	return false;
}

int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);


	cin >> n >> m;

	for(int i=1;i<=n;i++)
		for(int j=1 ;j<=m;j++){
			cin >> a;
			satir[i]+=a;	
			sutun[j]+=a;
		}	
	cin >> a >> b >> c;
	
	for(int i=2;i<=n;i++)
		satir[i] += satir[i-1];
	
	for(int i=2;i<=m;i++)
		sutun[i] += sutun[i-1];

	for(int i=1;i<=n-1;i++)
		for(int j=i+1;j<=n-1;j++)
			if(kontrol(satir[i],satir[j]-satir[i],satir[n]-satir[j]))
				s++;

	for(int i=1;i<=m-1;i++)
		for(int j=i+1;j<=m-1;j++)
			if(kontrol(sutun[i],sutun[j]-sutun[i],sutun[m]-sutun[j]))
				s++;



	cout << s << endl;
}
