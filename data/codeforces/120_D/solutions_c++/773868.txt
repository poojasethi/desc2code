#include <stdio.h>
#include <string.h>
#include <algorithm>

using namespace std;

int a[3], b[3];
int mp[60][60], sum[60][60];

int main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	int n, m;
	scanf("%d%d", &n, &m);
	for(int i = 1; i <= n; ++i)
		for(int j = 1; j <= m; ++j) {
			scanf("%d", &mp[i][j]);
			sum[i][j] = sum[i][j - 1] + sum[i - 1][j] - sum[i - 1][j - 1] + mp[i][j];
		}
	for(int i = 0; i < 3; ++i) scanf("%d", a + i);
	sort(a, a + 3);
	int res = 0;
	if(n >= 3) {
		for(int i = 1; i <= n; ++i)
			for(int j = i + 1; j < n; ++j) {
				b[0] = sum[i][m];
				b[1] = sum[j][m] - sum[i][m];
				b[2] = sum[n][m] - sum[j][m];
				sort(b, b + 3);
				if(a[0] == b[0] && a[1] == b[1] && a[2] == b[2]) ++res;
			}
	}
	if(m >= 3) {
		for(int i = 1; i <= m; ++i)
			for(int j = i + 1; j < m; ++j) {
				b[0] = sum[n][i];
				b[1] = sum[n][j] - sum[n][i];
				b[2] = sum[n][m] - sum[n][j];
				sort(b, b + 3);
				if(a[0] == b[0] && a[1] == b[1] && a[2] == b[2]) ++res;
			}
	}
	printf("%d\n", res);
	return 0;
}
