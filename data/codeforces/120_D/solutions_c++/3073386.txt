#include <iostream>
#include <stdio.h>
using namespace std;
int n,m,a[55][55],s[55][55]={0},A,B,C;
bool chek(int a, int b,int c)
{
  if (a==A&&b==B&&c==C) return 1;
  if (a==A&&b==C&&c==B) return 1;
  if (a==B&&b==C&&c==A) return 1;
  if (a==B&&b==A&&c==C) return 1;
  if (a==C&&b==B&&c==A) return 1;
  if (a==C&&b==A&&c==B) return 1;
 return 0;
}
int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
 cin>>n>>m;
 for (int i=1;i<=n;i++)
  for (int j=1;j<=m;j++)
   {
    cin>>a[i][j]; s[i][j]=a[i][j]+s[i-1][j]+s[i][j-1]-s[i-1][j-1];
   }
cin>>A>>B>>C;
int ans=0;
for (int i1=1;i1<n;i1++)
 for (int i2=i1+1;i2<n;i2++)
 {
 int a=s[i1][m], b=s[i2][m]-s[i1][m], c=s[n][m]-s[i2][m];
 if (chek(a,b,c)) ans++;
 }
for (int j1=1;j1<m;j1++)
 for (int j2=j1+1;j2<m;j2++)
 {
  int a=s[n][j1], b=s[n][j2]-s[n][j1],c=s[n][m]-s[n][j2];
  if (chek(a,b,c)) ans++;
 }
 cout<<ans;
 return 0;
}
