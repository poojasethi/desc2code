#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int row[55],line[55],num,A[3],ot[5][2],lvl;
void op(int ii[])
{
	int a[3];
	memset(a,0,sizeof(a));
	for(int i=0;i<3;i++)
	{
		for(int j=ot[i][0];j<=ot[i][1];j++)
			a[i]+=ii[j];
	}
	sort(a,a+3);			
	for(int i=0;i<3;i++)
		if(a[i]!=A[i])return;
	num++;
	return ;
}

void dfs(int now,int lvl,int len,int ii[])
{
	if(lvl==2&&now<len)
	{
		ot[lvl][0]=now;
		ot[lvl][1]=len-1;
		op(ii);
		return;
	}
	for(int i=now;i<len;i++)
	{
		ot[lvl][0]=now;
		ot[lvl][1]=i;
		dfs(i+1,lvl+1,len,ii);
	}
}
int main()
{
	freopen("input.txt","r",stdin);		
	freopen("output.txt","w",stdout);
	int m,n;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		memset(row,0,sizeof(row));
		memset(line,0,sizeof(line));
		num=0;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				int p;
				scanf("%d",&p);
				row[i]+=p;
				line[j]+=p;
			}
		}
		for(int i=0;i<3;i++)
			scanf("%d",&A[i]);
		sort(A,A+3);
		dfs(0,0,n,row);
		dfs(0,0,m,line);
		printf("%d\n",num);
	}
	return 0;
}
