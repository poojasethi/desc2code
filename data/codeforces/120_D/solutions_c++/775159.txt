#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int v[60][60];
int n,m;
int a,b,c;


int sum( int l,int r, int x, int y ){
	int res = 0;
	for( int i = l; i < r; i++ )
		for( int j = x; j < y;j++ )
			res += v[i][j];
	return res;
}

int h(){
	int q[3] = {a,b,c};
	sort(q,q+3);
	int res = 0;
	for( int i = 1; i < n; i++ ){
		for( int j = i+1; j < n; j++ ){
			int s[3];
			s[0] = sum(0,i,0,m);
			s[1] = sum(i,j,0,m);
			s[2] = sum(j,n,0,m);

			sort(s,s+3);
			bool ok = true;
			for( int k = 0; k < 3; k++ )
				if ( q[k]!=s[k] ) ok=false;
			if (ok)res++;
		}
	}
	return res;
}

int v2[60][60];

int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);

	cin>>n>>m;
	for( int i =0; i < n; i++ )
		for( int j =0; j < m; j++ )
			cin>>v[i][j];
	cin>>a>>b>>c;
	
	int res = h();
	for( int i = 0; i < n; i++ )
		for( int j = 0; j < m; j++ )
			v2[j][i] = v[i][j];
	swap(n,m);
	for( int i = 0; i < n; i++ )
		for( int j = 0; j < m; j++ )
			v[i][j] = v2[i][j];
	
	res += h();
	cout<<res<<endl;

	fclose(stdin); fclose(stdout);
	return 0;
}
