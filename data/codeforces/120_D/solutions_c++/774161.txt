#include <stdio.h>
#include <cstdlib>
#include <iostream>
using namespace std;

FILE *fin=freopen("input.txt","r",stdin);
FILE *fout=freopen("output.txt","w",stdout);

int n,m,a[2000][2000],aa,bb,cc,r[2000],c[2000];

bool check(int x,int y,int z)
{
 if ( x==aa ) if ( ( y==bb && z==cc ) || ( z==bb && y==cc ) ) return true;
 if ( x==bb ) if ( ( y==aa && z==cc ) || ( z==aa && y==cc ) ) return true;
 if ( x==cc ) if ( ( y==bb && z==aa ) || ( z==bb && y==aa ) ) return true;
 return false;
}

main()
{
 cin>>n>>m;
 for ( int k=1;k<=n;k++ )
  for ( int i=1;i<=m;i++ )
   {
    cin>>a[k][i];
    r[k]+=a[k][i];
    c[i]+=a[k][i];
   }
 for ( int k=2;k<=n;k++)
  r[k]+=r[k-1];
 for ( int i=2;i<=m;i++)
  c[i]+=c[i-1];
 
 cin>>aa>>bb>>cc;
 
 int ans=0;
 for ( int i=1;i<=n-2;i++ )
  for ( int j=i+1;j<=n-1;j++ )
   if ( check(r[i],r[j]-r[i],r[n]-r[j]) ) ans++;
 
 for ( int i=1;i<=m-2;i++ )
  for ( int j=i+1;j<=m-1;j++ )
   if ( check(c[i],c[j]-c[i],c[m]-c[j]) ) ans++;
   
 cout<<ans<<endl;
}
