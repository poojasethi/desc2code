#include		<iostream>
#include		<cstdio>
#include		<cstdlib>
#include		<cstring>
#include		<algorithm>
#define         Maxn 101
using namespace std;
typedef long long  ll;
ll s[Maxn][Maxn],A,B,C,SA,SB,SC,i,j,n,m,ans;
void		swap(ll &a,ll &b,ll &c){
	int t;
	if (a<b) {t=a;a=b;b=t;}
	if (a<c) {t=a;a=c;c=t;}
	if (b<c) {t=b;b=c;c=t;}
}
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d %d\n",&n,&m);
	int ans=0;
	for (i=1;i<=n;i++)
	for (j=1;j<=m;j++)
	{
		scanf("%d",&s[i][j]);
		s[i][j]+=s[i-1][j]+s[i][j-1]-s[i-1][j-1];
	}
	scanf("%d %d %d\n",&A,&B,&C);
	swap(A,B,C);
	for (i=1;i<m;i++)
	for (j=i+1;j<m;j++)
	{
		SA=s[n][i];SB=s[n][j]-s[n][i];SC=s[n][m]-s[n][j];
		swap(SA,SB,SC);
		if (SA==A && SB==B && SC==C) ans++;
	}
	for (i=1;i<n;i++)
	for (j=i+1;j<n;j++)
	{
		SA=s[i][m];SB=s[j][m]-s[i][m];SC=s[n][m]-s[j][m];
		swap(SA,SB,SC);
		if (SA==A && SB==B && SC==C) ans++;
	}
	printf("%d\n",ans);
	fclose(stdout);
	return 0;
}
