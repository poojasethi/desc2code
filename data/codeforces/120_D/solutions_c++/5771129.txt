#include <cstdio>
#include <set>
using namespace std;
int n,m;
int field[55][55];
inline int GetSum(int bl,int bc,int el,int ec)
{
	int res=0;
	for(int i=bl;i<=el;i++)
		for(int j=bc;j<=ec;j++)
			res+=field[i][j];
	return res;
}
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			scanf("%d",&field[i][j]);
	int num;
	set<int> ans;
	for(int i=1;i<=3;i++)
	{
		scanf("%d",&num);
		ans.insert(num);
	}
	set<int> tmp;
	int cnt=0;
	for(int i=1;i<m-1;i++)
		for(int j=i+1;j<m;j++)
		{
			tmp.insert(GetSum(1,1,n,i));
			tmp.insert(GetSum(1,i+1,n,j));
			tmp.insert(GetSum(1,j+1,n,m));
			if(tmp==ans)
				cnt++;
			tmp.clear();
		}
	for(int i=1;i<n-1;i++)
		for(int j=i+1;j<n;j++)
		{
			tmp.insert(GetSum(1,1,i,m));
			tmp.insert(GetSum(i+1,1,j,m));
			tmp.insert(GetSum(j+1,1,n,m));
			if(tmp==ans)
				cnt++;
			tmp.clear();
		}
	printf("%d\n",cnt);
	return 0;
}
