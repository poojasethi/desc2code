{

	int n, x, m;
	scanf("%d%d", &n, &m);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++) {
			scanf("%d", &x);
			rs[i] += x;
			cs[j] += x;
		}
	for(int i=1;i<=n;i++)
		rs[i] += rs[i-1];
	for(int j=1;j<=m;j++)
		cs[j] += cs[j-1];
	int a[3];
	for(int i=0;i<3;i++) scanf("%d", &a[i]);
	sort(a, a+3);
	int cnt=0;
	do{
		for(int i=1;i<=n;i++)
			for(int j=i+1;j<=n-1;j++)
				if(rs[i]==a[0] && rs[j]-rs[i]==a[1] && rs[n]-rs[j]==a[2])
					++cnt;
		for(int i=1;i<=m;i++)
			for(int j=i+1;j<=m-1;j++)
				if(cs[i]==a[0] && cs[j]-cs[i]==a[1] && cs[m]-cs[j]==a[2])
					++cnt;
	}while(next_permutation(a, a+3));
	
	printf("%d\n", cnt);
	return 0;
}
