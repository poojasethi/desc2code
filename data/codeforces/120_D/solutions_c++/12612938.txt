#include<bits/stdc++.h>

using namespace std;

int N[3],A[3],SAT[55],SUT[55],m,n,i,j,r,r2,top,ans;

int main(){

	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);

	cin >> m >> n;

	for(i=0;i<m;i++)
		for(j=0;j<n;j++){
			cin >> r;
			top += r;
			SAT[i] += r;
			SUT[j] += r;
		}
   cin >> A[0] >> A[1] >> A[2];

	sort(A, A + 3);

	for(r = SAT[i=0]; i < m-2;r += SAT[++i])
		for(r2 = SAT[j=i+1]; j<m-1; r2 += SAT[++j]){
			N[0] = r;
	      N[1] = r2;
			N[2] = top - r - r2;
			sort(N, N + 3);
			if(A[0] == N[0] && A[1] == N[1] && A[2] == N[2]) ans++;
		}
	for(r = SUT[i=0]; i < n-2;r += SUT[++i])
		for(r2 = SUT[j=i+1]; j<n-1; r2 += SUT[++j]){
			N[0] = r;
			N[1] = r2;
			N[2] = top - r - r2;
			sort(N, N + 3);
			if(A[0] == N[0] && A[1] == N[1] && A[2] == N[2]) ans++;
		}
	cout << ans;
	return 0;
}
