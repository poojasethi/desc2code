using namespace std;
int lsum[55], tsum[55], A[55][55], X[3];
bool ok(int a,int b,int c) {
	int B[3] = {a,b,c};
	sort(B,B+3);
	return (B[0] == X[0] && B[1] == X[1] && B[2] == X[2]);
}
int main() {
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,m,ans = 0;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;++i) {
		tsum[i] = tsum[i-1];
		for(int j=1;j<=m;++j) {
			scanf("%d",&A[i][j]);
			tsum[i] += A[i][j];
		}
	}
	scanf("%d%d%d",&X[0],&X[1],&X[2]);
	sort(X,X+3);
	for(int j=1;j<=m;++j) {
		lsum[j] = lsum[j-1];
		for(int i=1;i<=n;++i)
			lsum[j] += A[i][j];
	}
	// cut
	for(int a=1;a<n;++a)
		for(int b=a+1;b<n;++b)
			if(ok(tsum[a],tsum[b]-tsum[a],tsum[n]-tsum[b]))
				++ans;
	for(int a=1;a<m;++a)
		for(int b=a+1;b<m;++b)
			if(ok(lsum[a],lsum[b]-lsum[a],lsum[m]-lsum[b]))
				++ans;
	printf("%d\n",ans);
}
