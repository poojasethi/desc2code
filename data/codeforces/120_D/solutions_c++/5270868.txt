#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=55;
int n,m,ans,c[maxn][maxn],s[maxn],num[4],q[4];
void work(int n)
{
     for(int i=1;i<=n-2;i++)
       for(int j=i+1;j<n;j++)
       {
         q[1]=s[i];
         q[2]=s[j]-s[i];
         q[3]=s[n]-s[j];
         sort(q+1,q+4);
         if(q[1]==num[1]&&q[2]==num[2]&&q[3]==num[3])ans++;
       }
}
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)for(int j=1;j<=m;j++)scanf("%d",&c[i][j]);
    for(int i=1;i<=3;i++)scanf("%d",&num[i]);
    sort(num+1,num+4);
    for(int i=1;i<=n;i++)
    {
      for(int j=1;j<=m;j++)s[i]+=c[i][j];
      s[i]+=s[i-1];
    }
    work(n);
    memset(s,0,sizeof(s));
    for(int i=1;i<=m;i++)
    {
      for(int j=1;j<=n;j++)s[i]+=c[j][i];
      s[i]+=s[i-1];
    }
    work(m);
    printf("%d\n",ans);
    return 0;
}
