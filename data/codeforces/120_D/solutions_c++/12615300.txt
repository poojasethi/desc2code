#include<bits/stdc++.h>
using namespace std;
int h[55][55],n,satir[55],sutun[55],k,l,i,j,ii,jj,ll,a,b,c,t,m,s;
int f(int x,int y,int z)
{
	if(x==a and y==b and z==c)return 1;
	if(x==a and y==c and z==b)return 1;
	if(x==b and y==a and z==c)return 1;
	if(x==b and y==c and z==a)return 1;
	if(x==c and y==a and z==b)return 1;
	if(x==c and y==b and z==a)return 1;
	return 0;
}
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	cin>>n>>m;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
		{
			cin>>h[i][j];
			satir[i]+=h[i][j];
			sutun[j]+=h[i][j];
		}
	cin>>a>>b>>c;
	for(i=ii=0;i<n-2;i++)
	{
		ii+=satir[i];
		jj=0;
		for(j=i+1;j<n-1;j++)
		{
			jj+=satir[j];
			ll=0;
			for(l=j+1;l<n;l++)
				ll+=satir[l];
			if( f(ii,jj,ll))t++;
		}
	}
	for(i=ii=0;i<m-2;i++)
	{
		ii+=sutun[i];
		jj=0;
		for(j=i+1;j<m-1;j++)
		{
			jj+=sutun[j];
			ll=0;
			for(l=j+1;l<m;l++)
				ll+=sutun[l];
			if(f(ii,jj,ll))t++;
		}
	}
	cout<<t<<endl;
}
