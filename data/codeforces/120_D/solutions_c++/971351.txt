#include<iostream>
#include<fstream>
#include<algorithm>
using namespace std;
const int maxn=60;
int arr[maxn][maxn],c[maxn],r[maxn];
int main(){
	ifstream cin("input.txt");
	ofstream cout("output.txt");
	int m,n,a[3],ans=0;
	cin>>m>>n;
	for(int i=0;i<m;i++)
		for(int j=0;j<n;j++){
			cin>>arr[i][j];
			r[i]+=arr[i][j];
			c[j]+=arr[i][j];
		}
	cin>>a[0]>>a[1]>>a[2];
	for(int i=1;i<m;i++)
		r[i]+=r[i-1];
	for(int i=1;i<n;i++)
		c[i]+=c[i-1];
	for(int i=0;i<m;i++)
		for(int j=i+1;j<m-1;j++)
			for(int p=0;p<6;p++){
				if(r[i]==a[0]&&r[j]-r[i]==a[1]&&r[m-1]-r[j]==a[2]){
					ans++;
					break;
				}
				next_permutation(a,a+3);
			}
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n-1;j++)
			for(int p=0;p<6;p++){
				if(c[i]==a[0]&&c[j]-c[i]==a[1]&&c[n-1]-c[j]==a[2]){
					ans++;
					break;
				}
				next_permutation(a,a+3);
			}
	cout<<ans<<endl;
}
