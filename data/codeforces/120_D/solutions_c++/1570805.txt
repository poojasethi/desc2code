#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int s[51][51];
int map[51][51];
int a[3];
int sum[3];
int main(){
		int m,n;
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
		scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
			for(int j=1;j<=m;j++){
					scanf("%d",&map[i][j]);
					s[i][j]=s[i-1][j]+s[i][j-1]-s[i-1][j-1]+map[i][j];
			}
	scanf("%d%d%d",&a[0],&a[1],&a[2]);
	sort(a,a+3);
	int cnt=0;
	for(int i=1;i<n;i++)
			for(int j=i+1;j<n;j++)
			{sum[0]=s[i][m];
			 sum[1]=s[j][m]-s[i][m];
			 sum[2]=s[n][m]-s[j][m];
			 sort(sum,sum+3);
			 if(sum[0]==a[0]&&sum[1]==a[1]&&sum[2]==a[2]) cnt++;
			}
	for(int i=1;i<m;i++)
			for(int j=i+1;j<m;j++)
			{sum[0]=s[n][i];
			 sum[1]=s[n][j]-s[n][i];
			 sum[2]=s[n][m]-s[n][j];
			 sort(sum,sum+3);
			 if(sum[0]==a[0]&&sum[1]==a[1]&&sum[2]==a[2]) cnt++;
			}
	printf("%d\n",cnt);
	return 0;
}
