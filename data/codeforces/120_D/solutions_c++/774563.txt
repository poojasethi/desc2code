#include <cstdio>
#include <algorithm>
using namespace std;

const int N=105;
int s[N],ss[N][N],b[N];

bool ok(int x,int y,int z)
{
  int a[10];
  a[0]=x; a[1]=y; a[2]=z;
  sort(a,a+3);
  if (a[0]==b[0] && a[1]==b[1] && a[2]==b[2]) return 1;
  else return 0;
}

int main()
{
  freopen("input.txt","r",stdin);
  freopen("output.txt","w",stdout);
  int n,m,ans=0;
  scanf("%d%d",&n,&m);
  for (int i=1;i<=n;i++)
    for (int j=1,k;j<=m;j++) scanf("%d",&k),s[j]=s[j-1]+k,ss[i][j]=ss[i-1][j]+s[j];
  scanf("%d%d%d",&b[0],&b[1],&b[2]);
  sort(b,b+3);
  for (int i=1;i<=n;i++)
    for (int j=i+1;j<=n-1;j++)
      if (ok(ss[i][m],ss[j][m]-ss[i][m],ss[n][m]-ss[j][m])) ans++;
  for (int i=1;i<=m;i++)
    for (int j=i+1;j<=m-1;j++)
      if (ok(ss[n][i],ss[n][j]-ss[n][i],ss[n][m]-ss[n][j])) ans++;
  printf("%d",ans);
  return 0;
}
