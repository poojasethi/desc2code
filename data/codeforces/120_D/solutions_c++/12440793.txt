#include<iostream>
#include<string>
#include<set>
#include<cstdio>
using namespace std;
int w[55][55];
int sum(int t,int l,int b,int r)
{
	int res=0;
	for (int i=t;i<b;i++)
	for (int j=l;j<r;j++)
	res=res+w[i][j];
	return res;
}
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int m,n;
	cin>>m>>n;
	for (int i=0;i<=m-1;i++)
	for (int j=0;j<=n-1;j++)
	{
		cin>>w[i][j];
	}
	multiset<int>A;
	for (int i=1;i<=3;i++)
	{
		int a;
		cin>>a;
		A.insert(a);
	}
	int ans=0;
	for (int i=1;i<m;i++)
	{
		for (int j=i+1;j<m;j++)
		{
			multiset<int>S;
			S.insert(sum(0,0,i,n));
			S.insert(sum(i,0,j,n));
			S.insert(sum(j,0,m,n));
			if(S==A) ++ans;
		}
	}
	for (int k=1;k<n;k++)
	{
		for(int l=k+1;l<n;l++)
		{
			multiset<int>S;
			S.insert(sum(0,0,m,k));
			S.insert(sum(0,k,m,l));
			S.insert(sum(0,l,m,n));
			if(S==A) ++ans;
		}
	}
	cout<<ans;
	return 0;
}
