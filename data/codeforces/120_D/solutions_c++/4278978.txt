#include <iostream>
#include <fstream>
#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
int t,n,m,i,j,x,y,k,a[111111],b[111111],c[4];

int po(int n)
{
 int t=0;
for (i=0;i<n-2;i++)
 for (j=i+1;j<n-1;j++)
 {
   if (a[i]==c[1] && a[j]-a[i]==c[2] && a[n-1]-a[j]==c[3]) t++;
   if (a[i]==c[1] && a[j]-a[i]==c[3] && a[n-1]-a[j]==c[2]) t++;
   if (a[i]==c[2] && a[j]-a[i]==c[1] && a[n-1]-a[j]==c[3]) t++;
   if (a[i]==c[2] && a[j]-a[i]==c[3] && a[n-1]-a[j]==c[1]) t++;
   if (a[i]==c[3] && a[j]-a[i]==c[2] && a[n-1]-a[j]==c[1]) t++;
   if (a[i]==c[3] && a[j]-a[i]==c[1] && a[n-1]-a[j]==c[2]) t++;

   }
 return t;
 }

int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
cin>>n>>m;
for (i=0;i<n;i++)
 for (j=0;j<m;j++)
  {
    cin>>x;
    a[i]+=x;
    b[j]+=x;
  }
int ans=0;
cin>>c[1]>>c[2]>>c[3];

for (i=1;i<n;i++) a[i]+=a[i-1];
ans+=po(n);
a[0]=b[0];
for (i=1;i<m;i++) a[i]=a[i-1]+b[i];
ans+=po(m);

if (c[1]==c[2] && c[1]==c[3]) ans/=6;else
 if (c[1]==c[2] || c[1]==c[3] || c[2]==c[3]) ans/=2;
cout<<ans;
}
