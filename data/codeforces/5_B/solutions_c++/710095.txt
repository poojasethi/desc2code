#include <iostream>
#include <vector>
#define pb push_back
using namespace std;
vector <string> v;
string s;
int ma, x, y = 1, z;
int main(){
    while(getline(cin,s)){v.pb(s);
        ma=(ma<s.size()?s.size():ma);}
    for(int i=0;i<ma+2;i++)cout<<'*';
    cout<<"\n";
    for(int i=0;i<v.size();i++){
        x=ma-v[i].size();
        if(x%2)y++;
        cout<<'*';
        for(int j=0;j<(z=x/2+(x%2)*(y%2));j++)cout<<' ';
        cout<<v[i];
        for(int j=0;j<x-z;j++)cout<<' ';
        cout<<"*\n";
    }
    for(int i=0;i<ma+2;i++)cout<<'*';
}
