#include <stdio.h>
#include <string.h>

char g[1002][1002];

int main()
{
	//freopen("out.txt","w",stdout);
	int i=0,l,t,q=-1,k,m,n;
	int f;
	while(gets(g[i])) {
		t=strlen(g[i++]);
		if(t>q) q=t;
	}
	for(m=0;m<(q+2);m++) putchar('*');
	putchar('\n');
	f=0;
	for(l=i,i=0;i<l;i++) {
		putchar('*');
		k=q-strlen(g[i]);
		t=k>>1;
		n=k-t;
		if(t!=n) {
			if(f) {
				t=t^n;
				n=t^n;
				t=t^n;
				f=0;
			}
			else f=1;
		}
		for(m=0;m<t;m++) putchar(' ');
		printf("%s",g[i]);
		for(m=0;m<n;m++) putchar(' ');
		puts("*");
	}
	for(m=0;m<(q+2);m++) putchar('*');
	putchar('\n');
	return 0;
}
