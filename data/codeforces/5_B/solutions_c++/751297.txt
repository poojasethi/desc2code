#include<iostream>
#include<vector>

using namespace std;

string rep(char c,int n)
{
	string r;
	for(int i=0;i<n;i++)
		r.push_back(c);
	return r;
}

int main()
{
	vector<string> v;
	string s;
	int tam=0;
	while(getline(cin,s))
	{
		v.push_back(s);
		tam=max(tam,(int)s.size());
	}
	cout<<rep('*',tam+2)<<endl;
	int ini;
	bool left=0;
	for(int i=0;i<v.size();i++)
	{
		ini=tam-v[i].size();
		if(ini%2)
		{
			ini/=2;
			if(left)
				ini++;
			left=not left;
		}
		else
			ini/=2;
		cout<<"*"<<rep(' ',ini)<<v[i]<<rep(' ',tam-v[i].size()-ini)<<"*"<<endl;
	}
	cout<<rep('*',tam+2)<<endl;
}
