#include<iostream>

using namespace std;

string s[1005];

int main()
{
	int n = 0, MAX = 0;
	while(getline(cin, s[n]))
		MAX = max(MAX, int(s[n++].length()));
	for(int i = 0; i < MAX + 2; i++)
		cout << '*';
	cout << endl;
	bool fl = 0;
	for(int j = 0; j < n; j++)
	{
		int t = (MAX - s[j].length()), x = 0;
		if(t %  2 == 1)
		{
			x = t / 2 + fl;
			fl = !fl;
		}
		else
			x = t / 2;
		cout << '*';
		for(int i = 0; i < x; i++)
			cout << ' ';
		cout << s[j];
		for(int i = 0; i < t - x; i++)
			cout << ' ';
		cout << '*';
		cout << endl;
	}
	for(int i = 0; i < MAX + 2; i++)
		cout << '*';
	cout << endl;	
	return 0;
}
