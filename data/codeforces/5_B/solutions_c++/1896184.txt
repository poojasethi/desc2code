#include<iostream>
#include<string>
#include<vector>
using namespace std;
int main()
{
	int m=0;
	string s;
	vector<string>v;
	
	while(getline(cin,s))
	{
		m=m<s.length() ? s.length() : m;
		v.push_back(s);
	};

	cout<<string(m+2,'*')<<endl;

	for(int i=0,j=1;i<v.size();++i)
	{

		int w=m-v[i].length();
		if(w%2)
		{
			++j;
		};
		cout<<'*'<<string(w/2+(w%2)*(j%2),' ')<<v[i]<<string(m-w/2-v[i].length()-(w%2)*(j%2),' ')<<"*\n";
		
	};

	cout<<string(m+2,'*')<<endl;

	return 0;
};
