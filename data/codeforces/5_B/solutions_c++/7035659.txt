#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int main(){
string str[1500];
int i=0, maxLen=0, spc;
bool b=0;
while(getline(cin,str[i])){
    maxLen=max(maxLen,(int)str[i].size());
    i++;
}
cout<<string(maxLen+2, '*')<<endl;
for (int j=0; j<i; j++){
    spc=maxLen-str[j].size();
    if (spc%2==0){
        cout<<"*"<<string(spc/2, ' ')<<str[j]<<string(spc/2, ' ')<<"*"<<endl;
        }
    else{
        cout<<"*"<<string((spc/2)+b, ' ')<<str[j]<<string((spc/2)+!b, ' ')<<"*"<<endl;
        b=!b;
    }
  }
cout<<string(maxLen+2, '*')<<endl;
return 0;
}
