#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
const int mm=1111;
char s[mm][mm];
int i,n,m,ml,w,a,b;
void outc(int m,char c)
{
     while(m--)printf("%c",c);
}
int main()
{
    n=0;
    while(gets(s[n]))++n;
    for(ml=i=0;i<n;++i)
        ml=max(ml,(int)strlen(s[i]));
    outc(ml+2,'*');
    puts("");
    for(w=i=0;i<n;++i)
    {
        printf("*");
        m=ml-strlen(s[i]);
        a=m/2;
        if(m&1)a+=w,w=w^1;
        b=m-a;
        outc(a,' ');
        printf("%s",s[i]);
        outc(b,' ');
        printf("*\n");
    }
    outc(ml+2,'*');
    puts("");
    return 0;
}
