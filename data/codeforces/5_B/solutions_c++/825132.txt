#include <iostream>
#include <vector>
#define pb push_back
using namespace std;
vector <string> v;
string a;
int mx, x, y = 1, z;
int main(){
    while(getline(cin,a)){
	v.pb(a);
        if(a.length()>mx)
		mx=a.length();
    }
    for(int i=0;i<mx+2;i++,cout<<'*');
    cout<<endl;
    for(int i=0;i<v.size();i++){
        x=mx-v[i].size();
        if(x%2)
		y++;
        cout<<'*';
        for(int j=0;j<x/2+(x%2)*(y%2);j++,cout<<' ');
        cout<<v[i];
        for(int j=0;j<x-(x/2+(x%2)*(y%2));j++,cout<<' ');
        cout<<'*'<<endl;
    }
    for(int i=0;i<mx+2;i++)cout<<'*';
}
