#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main()
{
    vector <string> v;
    string str;
    unsigned int m=0,n;
    while (getline(cin,str))
    {
        m=m>str.length() ? m : str.length();
        v.push_back(str);
    }
    cout<<string(m+2,'*')<<endl;
    for (unsigned int i=0,j=1;i<v.size();i++)
    {
        n=m-v[i].length();
        if (n%2) j++;
        cout<<'*'<<string(n/2+(n%2)*(j%2),' ')<<v[i]<<string(m-n/2-v[i].length()-(n%2)*(j%2),' ')<<"*\n";
    }
    cout<<string(m+2,'*')<<endl;
    return 0;
}
