#include<iostream>
#include<string>
#include<algorithm>
#include<vector>
#include<cmath>
using namespace std;
vector<string>v;
vector<string>::iterator it;
string s;
int main()
{
    v.clear();
    int w=0,l,r,c=0;
    while(getline(cin,s))
         v.push_back(s),w=max(w,(int)s.size());
    cout<<string(w+2,'*')<<endl;
    for(it=v.begin();it!=v.end();it++)
    {
          l=(w-(*it).size())/2;
          r=w-(*it).size()-l;
          if(l!=r)if(c++%2)swap(l,r);
          cout<<'*'<<string(l,' ')<<*it<<string(r,' ')<<'*'<<endl;
    }
    cout<<string(w+2,'*')<<endl;
    return 0;
}
