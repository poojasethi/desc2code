#include<iostream>
#include<string>

string s[100005];
int k,i,j,m,n,f1,f2=1;

{
    while(getline(cin,s[j++]));
    for(i=0;i<j;i++)if(s[i].size()>m)m=s[i].size();
    for(i=0;i<m+2;i++)cout<<"*";cout<<endl;
    for(i=0;i<j-1;i++)
    {
        cout<<"*";
        for(int q=0;q<(m-s[i].size()+f1)/2;q++)cout<<" ";
        cout<<s[i];
        for(int q=0;q<(m-s[i].size()+f2)/2;q++)cout<<" ";if(m%2!=s[i].size()%2)swap(f1,f2);

    }
