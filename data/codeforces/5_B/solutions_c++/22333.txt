#include <stdio.h>
#include <string.h>
char s[1002][1010];
int main()
{
    int w = 0, n, l = 1, len;
    for (n = 0; gets(s[n]); ++n) if (strlen(s[n]) > w) w = strlen(s[n]);
    for (int i = 0; i < w+2; ++i) putchar('*');
    puts("");
    for (int i = 0; i < n; ++i) {
        putchar('*');
        if ((len=w-strlen(s[i]))&1) ++l;
        for (int j = 0; j < (len+(l&1))/2; ++j) putchar(' ');
        printf("%s", s[i]);
        for (int j = 0; j < (len+1-(l&1))/2; ++j) putchar(' ');
        puts("*");
    }
    for (int i = 0; i < w+2; ++i) putchar('*');
    puts("");
    return 0;
}
