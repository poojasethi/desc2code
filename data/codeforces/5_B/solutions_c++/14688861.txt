#include <stdio.h>
#include <string.h>
char a[1009][1009];
int t,ma,w,m,e;
int max(int a,int b)
{
    return a>b?a:b;
}
void z(int x,char a)
{
    while(x--)putchar(a);
}
int main()
{
    while(gets(a[t]))ma=max(ma,strlen(a[t++]));
    z(ma+2,'*');
    puts("");
    for(int i=0;i<t;i++)
    {
        m=ma-strlen(a[i]);
        e=m/2;
        if(m&1)e+=w,w^=1;
        putchar('*');
        z(e,' ');
        printf("%s",a[i]);
        z(m-e,' ');
        puts("*");
    }
    z(ma+2,'*');
    puts("");
    return 0;
}
