#include <iostream>
#include <string>
#include <vector>
using namespace std;
vector<string> a;
string s;
long m,i,j,x;
int main(){
    while (getline(cin,s)){
        if (s.length()>m) m=s.length();
        a.push_back(s);
    }
    cout << string(m+2,'*') << endl;
    for (i=0,j=1;i<a.size();i++){
        x=m-a[i].length();
        if (x%2==1) j++;
        cout << '*' << string(x/2+(x%2)*(j%2),' ')<< a[i] 
            << string(x-x/2-(x%2)*(j%2),' ') << '*' << endl;
    }
    cout << string(m+2,'*') << endl;
    return 0;
}
