#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
#define rep(i, n) for(int i=0; i<(int)(n); i++)

int main() {
    vector<string> v;
    string s;
    int w=0;
    while(getline(cin, s)) { v.push_back(s); w=max(w, (int)s.size()); }
    cout << string(w+2, '*') << endl;
    int c=0;
    rep(i, v.size()) {
        int l = (w-v[i].size())/2;
        int r = w-l-v[i].size();
        if(l!=r) if(c++%2) swap(l, r);
        cout << '*' << string(l, ' ') << v[i] << string(r, ' ') << '*' << endl;
    }
    cout << string(w+2, '*') << endl;
}
