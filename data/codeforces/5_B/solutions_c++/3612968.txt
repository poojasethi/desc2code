#include <iostream>
#include <string>
#include <vector>
using namespace std;

vector< string > a;
string s;
long m,i,j,x;

int main(){
	
	while (getline(cin,s)){
		(int)s.size()>m?m=s.size():0;
		a.push_back(s);
	}
	cout << string(m+2,'*') << endl;
	
	for (i=0,j=1; i<(long)a.size(); i++){
		x=m-a[i].length();
		x%2?j++:0;
		cout << '*' << string(x/2+(x%2)*(j%2),' ')<< a[i] << string(x-x/2-(x%2)*(j%2),' ') << "*\n";
	}
	
	cout << string(m+2,'*') << endl;
	return 0;
}
