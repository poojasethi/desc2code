#include <iostream>
#include <fstream>
#include <string>
using namespace std;
string str[1003];
int K[1003];
int N,Max;
void solve()
{
	int k,r,t=0,l;
	for(int i=1;i<=N;i++)
	{
		k=(Max-str[i].length()-2);
		r=k/2;
		k=Max-2-r-str[i].length();
		if(k!=r && (t++%2)) {l=k; k=r; r=l;}
		cout<<"*";
		for(int j=1;j<=r;j++) cout<<" ";
		cout<<str[i];
		for(int j=1;j<=k;j++) cout<<" ";
		cout<<"*";
		cout << endl;
	}
}
int main()
{
	int i=0;
	while(i++,getline(cin,str[i])){K[i]=str[i].length(); N++; Max=max(K[i],Max);}
	Max+=2;
	for(i=1;i<=Max;i++) cout<<'*';
	cout<<'\n';
	solve();
	for(i=1;i<=Max;i++) cout<<'*';
	cout<<'\n';
	return 0;
}
