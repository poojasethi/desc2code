#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i=0,j,k;
	size_t m=0;
	bool f=0;
	string s[1010];
	while(getline(cin,s[i]))
	{
		m=max(m,s[i].size());
		i++;
	}
	n=i;
	for(i=1;i<=m+2;i++)
		putchar('*');
	putchar('\n');
	for(i=0;i<n;i++)
	{
		putchar('*');
		k=m-s[i].size();
		if(k&1 && f)
			putchar(' ');
		for(j=1;j<=k/2;j++)
			putchar(' ');
		cout<<s[i];
		for(j=1;j<=k/2;j++)
			putchar(' ');
		if(k&1 && !f)
			putchar(' ');
		if(k&1)
			f=1-f;
		putchar('*');
		putchar('\n');
	}
	for(i=1;i<=m+2;i++)
		putchar('*');
	putchar('\n');
	return 0;
}
