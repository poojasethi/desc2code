#include<bits/stdc++.h>
using namespace std;
int main()
{
    char input[1000];
    int MAX=0;
    vector<string>v;
    while(gets(input))
    {
        v.push_back(input);
        MAX=max(MAX,(int)strlen(input));
    }
    cout<<string(MAX+2,'*')<<endl;
    int cnt=1;
    for(int i=0;i<v.size();i++)
    {
        int x=MAX-v[i].size();
        if(x&1)cnt++;
        cout<<"*"<<string(x/2+(x%2)*(cnt%2),' ')<<v[i];
        cout<<string(x-x/2-(x%2)*(cnt%2),' ')<<"*"<<endl;
    }
    cout<<string(MAX+2,'*')<<endl;
    return 0;
}
