#include<stdio.h>
#include<string.h>
int main()
{
	char str[1005][1005];
	int len[1005],i=0,max=0,j,k,flag=0;
	while(gets(str[i]))
	{
		len[i]=strlen(str[i]);
		if(max<len[i])max=len[i];
		i++;
	}
	for(j=0;j<max+2;j++)
		printf("*");
	printf("\n");
	for(j=0;j<i;j++)
	{
		printf("*");
		for(k=0;k<(max-len[j]+flag)/2;k++)
			printf(" ");
		printf("%s",str[j]);
		for(k=0;k<(max-len[j]+1-flag)/2;k++)
			printf(" ");
		printf("*\n");
		if(max%2!=len[j]%2)flag=1-flag;
	}
	for(j=0;j<max+2;j++)
	printf("*");
	printf("\n");
	return 0;
}
