#include <iostream>
#include <string>
#include <vector>
using namespace std;
vector<string> all;
string str;
int main()
{
	int mx=0;
	while(getline(cin,str)){
		all.push_back(str);
		mx=max(mx,(signed)str.size());
	}
	bool v=0;
	str=string(mx+2,'*');
	cout<<str<<endl;
	for(int i=0;i<all.size();++i){
		cout<<'*';
		int len=mx-all[i].size();
		str=string((len+v)/2,' ');
		cout<<str<<all[i];
		str=string((len+!v)/2,' ');
		if(len&1)
			v=!v;
		cout<<str<<'*'<<endl;
	}
	str=string(mx+2,'*');
	cout<<str<<endl;
	return 0;
}
