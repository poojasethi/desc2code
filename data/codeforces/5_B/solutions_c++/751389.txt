#include <cstdio>
#include <cstring>
char data[1000][10000];
void put(char c,int times) {
	for (int i=0;i<times;i++)
		putchar(c);
}
int main() {
	int n=0;
	int max=0;
	while (gets(data[n])) {
		n++;
		if (strlen(data[n-1])>max)
			max=strlen(data[n-1]);
	}
	put('*',max+2);
	putchar('\n');
	bool s=false;
	for (int i=0;i<n;i++) {
		putchar('*');
		int left=(max-strlen(data[i]));
		if (left%2) {
			put(' ',left/2+s)	;
			printf("%s",data[i]);
			put(' ',left/2+1-s);
			s=!s;
		} else {
			put(' ',left/2)	;
			printf("%s",data[i]);
			put(' ',left/2);			
		}
		putchar('*');
		printf("\n");
	}	
	put('*',max+2);
	putchar('\n');
}
