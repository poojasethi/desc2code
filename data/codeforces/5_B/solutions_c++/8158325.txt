#include <bits/stdc++.h>

using namespace std;

vector<string> lines;

int main()
{
	int glob = 0, flag = 0;
	string line;
	while (getline(cin, line)) {
		int loc = line.length();
		glob = max(loc, glob);
		lines.push_back(line);
	}
	lines.insert(lines.begin(), string(glob, '*'));
	lines.insert(lines.end(), string(glob, '*'));

	for (int i = 0; i < lines.size(); i++) {
		int b = glob - lines[i].length();
		int l = b/2, r = (b+1)/2;
		if (b % 2 != 0) {
			if (flag) swap(l, r);
			flag = (flag+1)%2;
		}

		cout << '*' << string(l, ' ') << lines[i] << string(r, ' ') << '*' << endl;
	}

	return 0;
}
