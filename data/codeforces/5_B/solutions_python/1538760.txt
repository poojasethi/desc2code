#!/usr/bin/python
import sys

lines = sys.stdin.read().splitlines()

maxLen = 0
for i in lines:
	#i = i.rstrip()
	l = len(i)
	if l > maxLen:
		maxLen = l

shift = 0
print '*'*(maxLen + 2)
for i in lines:
	l = len(i)
	paddingLeft  = (maxLen - l + shift) / 2
	paddingRight = maxLen - l - paddingLeft
	if l > 0 and (maxLen - l) % 2 != 0:
		shift = 1 - shift
	print '*' + ' '*paddingLeft + i + ' '*paddingRight + '*'

print '*'*(maxLen + 2)
