data = []
maxLen = 0
while True:
    try:
        line = raw_input()
        data.append(line)
        if len(line) > maxLen:
            maxLen = len(line)
    except EOFError:
        break;
print '*'*(maxLen+2)
flag = 1
for line in data:
    l = len(line)
    if l%2 != maxLen%2:
        if flag == 1:
            line = line + ' '
            flag = 2
        else:
            line = ' ' + line
            flag = 1
    print '*' + line.center(maxLen) + '*'
print '*'*(maxLen+2)
