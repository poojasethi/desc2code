import sys

l = []
w = 0
while True:
	s = sys.stdin.readline()
	if not s: break
	s = s.strip()
	l.append(s)
	w = max(w, len(s))

print '*' * (w + 2)
flag = False
for s in l:
	dw = w - len(s) 
	w1 = w2 = dw / 2
	if dw % 2 == 1:
		if flag: w1 = w1 + 1
		else: w2 = w2 + 1
		flag = not flag
	print '*' + (' ' * w1) + s + (' ' * w2) + '*'
	

print '*' * (w + 2)
