l = []
try:
    while True:
        l.append(raw_input())
except EOFError:
    pass
m = 0
for s in l:
    m = max(m,len(s))

print "*"*(m+2)
x = 0
for s in l:
    d = m-(len(s))
    a = (d-(d%2))/2
    b = d - a
    if a!=b:
        if x==1:
            a,b=b,a
        x=1-x
    print "*"+(" "*a)+s+(" "*b)+"*"
print "*"*(m+2)
