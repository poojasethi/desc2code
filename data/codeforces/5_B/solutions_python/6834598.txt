import sys
words = []
lengths = []
flip = 1
for text in sys.stdin:
    text = text.strip()
    words.append(text)
    lengths.append(len(text))

ma = max(lengths)
border = ma + 2
print "*" * border
for i in range(len(words)):
    space = ma - len(words[i])
    if space % 2 == 0:
        print "*" + (" " * (space/2)) + words[i] + (" " * (space/2)) + "*"
    else:

        print "*" + (" " * ((space-flip)/2)) + words[i] + (" " * ((space+flip)/2)) + "*"
        flip *= -1
print "*" * border
