import sys

text = []

for line in sys.stdin:
	line = line[:len(line)-1]
	text.append(line)

def findMax(text):
	localMax = 0
	for line in text: localMax = max(len(line),localMax)
	return localMax

maxVal = findMax(text)

print (maxVal + 2) * "*"

padding = 0

for line in text:
	diff = maxVal - len(line)
	numOfSp = diff / 2
	extSpPred = diff % 2
	extSpL = extSpR = numOfSp * " "
	if extSpPred:
		if padding: extSpL += " "
		else: extSpR += " "
		padding = 1 - padding
	print (("*" + extSpL +  "%s" + extSpR + "*") % line)

print (maxVal + 2) * "*"
