from itertools import *
import sys
words = [line[:-1] for line in sys.stdin]
wide = len (max (words, key=lambda x: len (x)))
print '*' * (wide + 2)
left = True
for word in words:
    n = len (word)
    print '*' + ' ' * ((wide - n + (1 if not left and (wide - n) % 2 else 0)) / 2) + word + ' ' * ((wide - n + (1 if left and (wide - n) % 2 else 0)) / 2) + '*'
    if (wide - n) % 2:
        left ^= 1
print '*' * (wide + 2)
