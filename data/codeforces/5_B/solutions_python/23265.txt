import sys

lines = sys.stdin.read().splitlines()
w = max(map(len, lines))
print '*'*(w+2)
left=True
for l in lines:
    n = w - len(l)
    if (n & 1) == 1:
        if left:
            print '*'+' '*(n/2)+l+' '*((n+1)/2)+'*'
        else:
            print '*'+' '*((n+1)/2)+l+' '*(n/2)+'*'
        left = not left
    else:
        print '*'+' '*(n/2)+l+' '*(n/2)+'*'
print '*'*(w+2)
