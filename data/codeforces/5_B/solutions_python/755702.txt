import sys

lines = [line.strip() for line in sys.stdin.readlines()]
length = max(map(len, lines))

l, r = 0, 1

print "*" * (length + 2)

for line in lines:
    n = length - len(line)

    print "*%s*" % (" " * (n / 2 + l * (n & 1)) + line + " " * (n / 2 + r * (n & 1)))

    if n & 1 is not 0:
        l, r = r, l

print "*" * (length + 2)
