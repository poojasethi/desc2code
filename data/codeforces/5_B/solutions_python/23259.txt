#!/bin/env python

import sys

lines = sys.stdin.read().split('\n')[:-1]
maxLength = 0
for line in lines:
    maxLength = max(maxLength, len(line))

f = 0
print '*' * (maxLength+2)
for line in lines:
    l = (maxLength-len(line))/2
    if (maxLength-len(line)) % 2 != 0:
        if f==0:
            f = 1
        else:
            f = 0
            l += 1
    out = '*'
    out += ' ' * l
    out += line
    out += ' ' * (maxLength-l-len(line))
    out += '*'

    print out
print '*' * (maxLength+2)
