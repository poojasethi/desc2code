import sys;
s = [ x.strip()  for x in sys.stdin ]
L = 0
for A in s:
	L = max(L, len(A));
print '*' * (L + 2)
swp = 0;
for A in s:
	le = (L - len(A)) / 2;
	ri = (L - len(A) + 1) / 2;
	if (le != ri) :
		if not swp : print '*' + ' ' * le + A + ' ' * ri + '*';
		else : print '*' + ' ' * ri + A + ' ' * le + '*'; 
		swp ^= 1;
	else :
		print '*' + ' ' * le + A + ' ' * ri + '*';
print '*' * (L + 2)
