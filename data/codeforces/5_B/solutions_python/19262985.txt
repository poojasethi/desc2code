import sys
lines=[]
m=0
for line in sys.stdin:
    lines.append(line.strip())
    m=max(m,len(lines[-1]))
print '*'*(m+2)
l=1
for i in lines:
    u=m-len(i)
    if u%2:
        if l:
            u/=2
        else:
            u+=1
            u/=2
        l^=1
    else:u/=2
    print '*'+u*' '+i+(m-u-len(i))*' '+'*'
print '*'*(m+2)
