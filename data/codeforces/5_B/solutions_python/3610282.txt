import sys
L = [line.strip() for line in sys.stdin]
mx = max(map(len,L))
print '*'*(mx+2)

flag=0
for line in L:
    sp = mx-len(line)
    if sp%2!=0:
        spl = (sp+flag)/2
        flag = 1 - flag
    else : spl = sp/2
    spr = sp-spl
    print '*' + ' '*spl + line + ' '*spr + '*'
print '*'*(mx+2)
