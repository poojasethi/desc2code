l=[]
chk=0
while 1:
    try:
        w=raw_input()
        l.append(w)
        chk=max(chk,len(w))
    except:
        break
print '**'+'*'*chk
f=0
for i in range(len(l)):
    p=chk-len(l[i])
    if p%2==0:
        print '*'+' '*(p/2)+l[i]+' '*(p/2)+'*'
    else:
        if f%2==0:
            print '*'+' '*(p/2)+l[i]+' '*((p/2)+1)+'*'
        else:
            print '*'+' '*((p/2)+1)+l[i]+' '*(p/2)+'*'
        f+=1
print '**'+'*'*chk
