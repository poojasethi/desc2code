import sys

def main():
	ss = []
	width = 0
	isLeft = True
	for s in sys.stdin:
		s = s.strip()
		width = max(width, len(s)) 
		ss.append(s)
	print "*" * (width + 2)
	for s in ss:
		left = 0
		if 0 == (width - len(s)) % 2: left = (width - len(s)) / 2
		else:
			if isLeft: left = (width - len(s)) / 2
			else: left = (width - len(s)) / 2 + 1
			isLeft = not isLeft
		print "*" + " " * left + s + " " * (width - len(s) - left) + "*"
	print "*" * (width + 2)
 
if __name__ == "__main__": main()
