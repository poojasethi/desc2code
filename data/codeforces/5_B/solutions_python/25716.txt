import sys


l = 0
lines = [ ll[:-1] for ll in sys.stdin.readlines()]

for ll in lines:
    if len(ll)> l:
        l = len(ll)

print '*'*(l+2)
al = True
for ll in lines:
    d= l - len(ll)
    p = ' '
    pp = ' '
    if (d%2) ==1:
        p = p*(d/2+1)
        al = not al
    else:
        p = p*(d/2)
    pp = pp*(d/2)
    if al:
        print "*%s%s%s*"%(p,ll,pp)
    else:
        print "*%s%s%s*"%(pp,ll,p)
        

print '*'*(l+2)
