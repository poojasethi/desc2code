line = []
try:
	while True:
		line.append(raw_input())
except EOFError:
	pass
maxlen = 0
for i in line:
	maxlen = max(maxlen, len(i))
print '*'*(maxlen+2)
turn = 0
for i in line:
	spa = maxlen - len(i)
	a = (spa-spa%2)/2
	b = spa - a
	if not a == b:
		if turn == 1:
			a,b = b,a
		turn = 1 - turn 
	print '*'+' '*a+i+' '*b+'*'

print '*'*(maxlen+2)
