import sys                                                                      
left = 1
lines = []
max_l = 0
for line in sys.stdin:
  l = len(line)-1
  if l > max_l:
    max_l = l
  lines.append(line[:-1])
print '*'*(max_l+2)
for i in lines:
  sp = max_l - len(i)
  if sp%2:
    if left:
      sp1 = sp/2
      left = 0
    else:
      sp1 = sp/2+1
      left = 1
  else:
    sp1 = sp/2
  sp2 = sp-sp1
  print '*'+' '*sp1+i+' '*sp2+'*'
print '*'*(max_l+2)
