import sys

f = sys.stdin
width = 0
l = list()
for line in f:
        l.append(line[:-1])
        width = max(width, len(line[:-1]))

print '*' * (width + 2)

flag = 0

for s in l:
        indent = width - len(s)
        if indent % 2:
                indent = indent / 2 + flag
                flag = 1 - flag
        else:
                indent = indent / 2
        print '*' + ' ' * indent + s + ' ' * (width - indent - len(s)) + '*'

print '*' * (width + 2)
