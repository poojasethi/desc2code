#!/usr/bin/python

import sys

list = []
mx = -1
for i in sys.stdin:
    i = i.strip()
    list.append(i)
    mx = max(mx, len(i))

print '*' * (mx + 2)
flag = False
for i in list:
    l = mx - len(i)
    r = l // 2
    l = l - r

    if l != r:
        if flag == False:
            flag = True
            l, r = r, l
        else:
            flag = False
            
    print '*' + ' ' * l + i + ' ' * r + '*'
print '*' * (mx + 2)
