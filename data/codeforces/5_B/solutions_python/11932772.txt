#!/usr/bin/env python
import sys
s = []
for line in sys.stdin:
    s.append(line[:-1])
maxl = max(map(len, s))
flag = False
print '*'*(maxl+2)
for a in s:
    k = maxl - len(a)
    if (k & 1) == 0:
        k >>= 1
        a = '*' + ' '*k + a + ' '*k + '*'
    else:
        k >>= 1
        if flag:
            a = '*' + ' '*(k+1) + a + ' '*k + '*'
        else:
            a = '*' + ' '*k + a + ' '*(k+1) + '*'
        flag = not flag
    print a
print '*'*(maxl+2)
