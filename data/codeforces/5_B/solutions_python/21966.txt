from sys import stdin

lines = [x.strip() for x in stdin]

width = max ([len(x) for x in lines])


rem = 0

def adjust(x):
    global rem
    w = len(x)
    left = (width - w) / 2
    if (width - w) % 2 != 0:
        left += rem
        rem = 1 - rem
    return (' ' * left) + x + (' ' * (width - left - w))

print ('*' * (width + 2))

for x in lines:
    print('*' + adjust(x) + '*')

print ('*' * (width + 2))
