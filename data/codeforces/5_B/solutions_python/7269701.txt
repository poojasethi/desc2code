import sys

#sys.stdin = open('input.txt', 'r')
#sys.stdout = open('output.txt', 'w')
#om te testen want de interactive shell heeft geen eof
regels = [ x.strip()  for x in sys.stdin ]
lengte=max(len(x) for x in regels)

print '*'*(lengte+2)
links=True
for x in regels:
    t=lengte-len(x)
    sl=sr=0
    if t%2:  #oneven dan wisselen
        links=not links
        sl=links
        sr=not sl
    print '*'+' '*(t/2+sl)+x+' '*(t/2+sr)+'*'
    
            
print '*'*(lengte+2)
