N = input()
Array = map(int,raw_input().split())
global Array1
Array1 = [int(0)]*10001
C1=int(0)
C2=int(0)
C3=int(0)
for i in range(0,N) :
    if Array[i]%3==0 and Array1[i]==0:
        C3=C3+ int(1)
        Array1[i]=1
    elif Array[i]%3==1 and Array1[i]==0:
        C1=C1+ int(1)
        Array1[i]=1
    else :
        C2=C2+ int(1)
        Array1[i]=1
print (C3/2) + min(C1,C2)
