#!/usr/bin/env python
# -*- coding: utf-8 -*-

from collections import defaultdict

def main():
    raw_input()
    n = map(lambda x: sum(map(int, list(x))) % 3, raw_input().split())
    ns = defaultdict(int)
    for e in n:
        ns[e] += 1
    print ns[0]/2 + min(ns[1], ns[2])
    return 0

if __name__ == '__main__':
    main()
