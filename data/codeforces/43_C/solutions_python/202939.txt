import sys
import math
from collections import defaultdict


input = sys.stdin
n = int(input.readline().strip()) 
a = map(lambda x: int(x), input.readline().strip().split(' '))

divisable = [ b for b in a if b % 3 == 0 ]
div = len(divisable) / 2

non_divisable = [ b for b in a if b % 3 != 0 ]
selected = [False] * len(non_divisable)

non = 0

for i in xrange(len(non_divisable)):
    if not selected[i]:
      selected[i] = True

      for j in xrange(i+1, len(non_divisable)):
        if not selected[j] and ( int(str(non_divisable[i]) + str(non_divisable[j]))%3 == 0 or int(str(non_divisable[j]) + str(non_divisable[i]))%3 == 0 ):
          selected[j] = True
          non += 1
          break
 
print div + non
