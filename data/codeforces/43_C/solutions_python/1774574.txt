import sys
readin=sys.stdin.readline
def readnums():
	return map(int,readin().split())

n=readnums()[0]
a=readnums()
mod0,mod1,mod2=[],[],[]
for piece in a:
	if piece%3==0:
		mod0.append(piece)
	elif piece%3==1:
		mod1.append(piece)
	else:
		mod2.append(piece)

print len(mod0)/2 + min(len(mod1),len(mod2))
