n = raw_input()
tickets = map(int, raw_input().split())

def sum_digits(s):
	return reduce(lambda x, a: int(a) + int(x), str(s), 0)
		
tickets = map(sum_digits, tickets)
tickets = map(lambda x: x % 3, tickets)
# need two == 0 tickets or (one == 1 and one == 2)
print tickets.count(0) / 2 + min(tickets.count(1), tickets.count(2))
