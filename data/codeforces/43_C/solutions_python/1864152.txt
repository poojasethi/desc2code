import sys
import math

def f(t):
    sum = 0
    while t != 0:
        sum += t%10
        t = t/10
    return sum%3

    
file = sys.stdin
cnt = int(file.readline().rstrip())
p = map(int, file.readline().rstrip().split())
s = []
for t in p:
    s.append(f(t))

a0 = s.count(0)
a1 = s.count(1)
a2 = s.count(2)

ans = a0/2 + min(a1, a2)
print ans
