import sys

sample = sys.stdin
output = sys.stdout

def toOutput(line):
	output.write("%s" % line)

if __name__ == "__main__":
	n = int(sample.readline().strip())

	numbers = sample.readline().strip().split(" ")
	res = []
	
	for num in numbers:
		res.append(sum([int(d) for d in num]) % 3)

	zeroMod = res.count(0)
	oneMod = res.count(1)
	twoMod = res.count(2)
	
	toOutput("%d" % ((zeroMod / 2) + min(oneMod, twoMod)))
