#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
from operator import getitem
input = sys.stdin
output = sys.stdout

import itertools

WIN = 'WIN'
FAIL = 'FAIL'

def all_know(t,R):
    a,b,c = t
    return (b in R[a]) and (c in R[a]) and (c in R[b])

def nobody_know(t,R):
    a,b,c = t
    return (b not in R[a]) and (c not in R[a]) and (c not in R[b])

def solve(R):
    has_all_know = False
    has_nobody_know = False
    for t in itertools.combinations(range(1,5+1), 3):
        if not has_all_know:
            has_all_know = all_know(t,R)
        if not has_nobody_know:
            has_nobody_know = nobody_know(t,R)
    if not has_all_know and not has_nobody_know:
        return FAIL
    return WIN
    
m = int(input.readline())
assert 0<=m and m<=10

R = {}
for i in range(1,5+1):
    R[i] = set([])

for i in range(m):
    S = input.readline().split(' ')
    assert len(S)==2
    a = int(S[0])
    b = int(S[1])
    assert 1<=a and a<=5
    assert 1<=b and b<=5
    R[a].add(b)
    R[b].add(a)

n = solve(R)
output.write('%s\n' % n)
