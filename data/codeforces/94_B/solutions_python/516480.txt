#! /usr/bin/env python

from sys import stdin

def all_in(i,j,k,edges):
	my=[(i,j),(j,k),(i,k)]
	return all(x in edges for x in my)

def all_out(i,j,k,edges):
	my=[(i,j),(j,k),(i,k)]
	return all(not (x in edges) for x in my)

def fail(edges):
	contains=0
	out=0
	for i in xrange(1,6):
		for j in xrange(i+1,6):
			for k in xrange(j+1,6):
				if all_in(i,j,k,edges):
					contains+=1
				elif all_out(i,j,k,edges):
					out+=1

	return contains==0 and out==0


if __name__=='__main__':
	m=int(stdin.readline())
	edges=set()
	for i in xrange(m):
		x,y=map(int,stdin.readline().split())
		edges.add((x,y))
		edges.add((y,x))
	#print edges
	#print all_in(5,2,3,edges)
	#print all_out(1,5,2,edges)
	if not fail(edges):
		print "WIN"
	else:
		print "FAIL"
