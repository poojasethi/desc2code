import itertools

n = input()
mp = [([0] * 5) for i in xrange(5)]
for i in range(n):
    a,b = map(int,raw_input().split())
    a,b = a-1,b-1
    mp[a][b] = mp[b][a] = 1

ret = 'FAIL'
for i,j,k in itertools.combinations(range(5),3):
    if mp[i][j] == mp[j][k] == mp[i][k]:
        ret = 'WIN'
        break

print ret
