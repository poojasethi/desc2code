def solve():
    n = int(raw_input())
    adj = [[], [], [], [], []]
    while n:
        n-=1
        line = map(int, str(raw_input()).split())
        adj[line[0]-1] += [line[1]-1]
        adj[line[1]-1] += [line[0]-1]
    for i in adj:
        if len(i) != 2:
            print 'WIN'
            return
    print 'FAIL'

solve()
