#!/usr/bin/env python
# -*- coding: utf-8 -*-

import itertools, collections

m = collections.defaultdict(list)
for _ in xrange(int(raw_input())):
    a, b = map(int, raw_input().split())
    m[a].append(b)
    m[b].append(a)
ans = False
def test(i, j):
    if m[i].count(j) == 1 and m[j].count(i) == 1:
        return 1
    elif m[i].count(j) == 0 and m[j].count(i) == 0:
        return -1
    return 0
for i, j, k in itertools.combinations(xrange(1,6), 3):
    s = test(i, j) + test(j, k) + test(k, i)
    if abs(s) == 3:
        ans = True
        break
print "WIN" if ans else "FAIL"
