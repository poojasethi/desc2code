import itertools

m = int(raw_input())
q = [[0] * 5 for i in range(5)]
for i in range(m):
	a, b = map(lambda x: int(x) - 1, raw_input().split())
	q[a][b] = 1
	q[b][a] = 1
for i in itertools.combinations([0, 1, 2, 3, 4], 3):
	x = q[i[0]][i[1]] + q[i[0]][i[2]] + q[i[1]][i[2]]
	if x == 3 or x == 0:
		print 'WIN'
		break
else:
	print 'FAIL'
