import sys

pairs = []



for i in range(1, 6):
  for j in range(i + 1, 6):
    pairs.append((i, j))


    
n = int(raw_input())

inputs = []

for i in range(n):
  a,b = raw_input().split()
  a = int(a)
  b = int(b)
  inputs.append(tuple(sorted((a, b))))
  
hasRelations = set()
noRelations = set()



for p in pairs:
  if p in inputs:
    hasRelations.add(p)
  else:
    noRelations.add(p)  

  
  
found = False

for i in range(1, 6): 
  for j in range(i + 1, 6):
    for k in range(j + 1, 6):
      if (i, j) in hasRelations and (j, k) in hasRelations and (i, k) in hasRelations:
        found = True
        break
      elif (i, j) in noRelations and (j, k) in noRelations and (i, k) in noRelations:
        found = True
        break       
                

if found:
  print "WIN"
else:
  print "FAIL"
