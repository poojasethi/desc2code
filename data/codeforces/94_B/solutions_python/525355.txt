from itertools import permutations
import sys
T = input()
j = [ map(int, raw_input().split()) for i in xrange(T)]
graph = [ [0]*7 for i in xrange(7) ]
for one, two in j:
    graph[one][two] = graph[two][one] = 1
    
for one, two, three in permutations([1, 2, 3, 4, 5], 3):
    if graph[one][two] == graph[two][three] == graph[three][one] == graph[one][three]:
        print "WIN"
        exit()
print "FAIL"
