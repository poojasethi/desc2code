edge = {1:[], 2:[], 3:[], 4:[], 5:[]}

for _ in range(input()):
    l = raw_input()
    x = int(l[0])
    y = int(l[2])
    edge[x].append(y)
    edge[y].append(x)

f = False

for i in range(1,6):
    for j in range(i+1,6):
        for k in range(j+1,6):
            if i in edge[j] and j in edge[k] and k in edge[i]:
                f = True
            if (not i in edge[j]) and (not j in edge[k]) and (not k in edge[i]):
                f = True

if f:
    print 'WIN'
else:
    print 'FAIL'
