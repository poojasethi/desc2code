
from itertools import combinations

A = [[False]*5 for i in range(5)]
for _d in xrange(input()):
   i, j = map(int, raw_input().split())
   A[i-1][j-1] = A[j-1][i-1] = True

def cliq(A, v):
   for c in combinations(range(5), 3):
      V = [A[i][j] == v for i in c for j in c if i != j]
      if all(V): return True
   return False

print ("FAIL", "WIN")[cliq(A, True) or cliq(A, False)]
