#!/usr/bin/env python
# -*- coding: utf-8 -*-
m = int(raw_input())
g = [[]for i in range(5)]
for i in range(m):
    t, f = map(int, raw_input().split(" "))
    t -= 1
    f -= 1
    g[t].append(f)
    g[f].append(t)
    
flag = True
for i in range(5):
    if len(g[i]) >= 3 or len(g[i]) <= 1:flag = False
        
if flag == False: print "WIN"
else: print "FAIL"
