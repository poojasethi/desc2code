pole=[[0]*5 for r in xrange(5)]

komb = ((1,2,3),

        (1,3,5),
        (1,4,5),
        (2,3,4),

        (2,4,5),
        (3,4,5))
        
win = 0
for a,b,c in komb :
    if pole[a-1][b-1] == pole[b-1][c-1] == pole[c-1][a-1] :
        print "WIN"
        win = 1
        break
if win != 1 :
