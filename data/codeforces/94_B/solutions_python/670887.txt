import sys
def read_values():
  return map(int,raw_input().split())
n=5
m=input()
a = [[0]*n for _ in range(n)]
for i in range(m):
  u,v=read_values()
  u-=1
  v-=1
  a[u][v]=1
  a[v][u]=1

res='FAIL'
for i in range(n):
  for j in range(i):
    for k in range(j):
      if a[i][j] and a[i][k] and a[j][k]:
        res = 'WIN'
      if a[i][j]==0 and a[i][k]==0 and a[j][k]==0:
        res = 'WIN'
print res
