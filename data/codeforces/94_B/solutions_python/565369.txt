n = int(raw_input())
if n == 0:
    print "WIN"
    exit()
a = []
for i in range(n):
    a.append(map(int, raw_input().split()))
    a.append([a[-1][1], a[-1][0]])
for i in range(5):
    t = 0
    for j in range(5):
        if [i + 1, j + 1] in a:
            t += 1
    if t != 2:
        print "WIN"
        break
else:
    print "FAIL"
