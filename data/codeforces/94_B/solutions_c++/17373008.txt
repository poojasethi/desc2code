#include<bits/stdc++.h>
#define pb push_back
using namespace std;
int gir,x,y;
vector<int>v[100];
int main()
{
	scanf("%d",&gir);
	for(int i=1;i<=gir;i++)
	{
		scanf("%d %d",&x,&y);
		v[x].pb(y);
		v[y].pb(x);
	}
	if(!gir)
	{
		printf("WIN");
		return 0;
	}
	for(int i=1;i<=5;i++)
	{
		if((int)v[i].size()!=2)
		{
			printf("WIN");
			return 0;
		}
	}
	printf("FAIL");
	
}
