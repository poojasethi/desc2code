#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int m; scanf("%d",&m);
	bool adj[5][5]={};
	rep(i,m){
		int u,v; scanf("%d%d",&u,&v);
		u--, v--;
		adj[u][v]=adj[v][u]=true;
	}

	bool ok=false;
	rep(i,5) rep(j,i) rep(k,j) {
		if(( adj[i][j] &&  adj[j][k] &&  adj[k][i])
		|| (!adj[i][j] && !adj[j][k] && !adj[k][i])) ok=true;
	}
	puts(ok?"WIN":"FAIL");

	return 0;
}
