#include<cstdio>
#include<iostream>
using namespace std;
int a[50][50];
int main()
{
	int i,j,k,x,m,y,t,c,f=0;
	cin>>m;
	for(i=0;i<m;i++)
	{
		cin>>x>>y;
                a[x][y]=1;
		a[y][x]=1;
	}
	for(i=1;i<=5;i++)
	{
		for(j=i+1;j<=5;j++)
		{
			c=a[i][j];
			for(k=j+1;k<=5;k++)
			{
				if(a[j][k]==c && a[k][i]==c)
					f=1;

			}
		}
	}
	if(f==1)
		cout<<"WIN"<<endl;
	else
		cout<<"FAIL"<<endl;
	return 0;
}
