#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int maxn=6;

int num[maxn];
int m,i;
int a,b;
bool flag=true;

int main()
{
	cin >> m;
	while(m--)
	{
		cin >> a >> b;
		num[a]++;
		num[b]++;
	}
	for(i=1;i<=5;i++)
	{
		if(num[i]!=2)
		{
			flag=false;
			break;
		}
	}
	if(!flag)
		cout << "WIN" << endl;
	else
		cout << "FAIL" << endl;
	return 0;
}
