#include <stdio.h>
#include <stdlib.h>

int main()
{
    int n,a,b[5]={0},i;
    scanf("%d",&n);
    for(i=0;i<2*n;i++)
    {
        scanf("%d",&a);
        b[a-1]++;
    }
    for(i=0;i<5;i++)
    {
        if(b[i]!=2)
            break;
    }
    if(5-i!=0)
        printf("WIN");
    else
    printf("FAIL");
    return 0;
}
