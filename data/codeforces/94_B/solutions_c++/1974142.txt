#include <iostream>
#include <vector>

using namespace std;

int main() {
  vector <vector<int> > a(5);
  int m;
  cin >> m;
  
  for (int i = 0; i < m; i++) {
    int k, w;
    cin >> k >> w;
    k--;
    w--;
    a[k].push_back(w);
    a[w].push_back(k);
  }
  
  cout << ((a[0].size() == 2 && a[1].size() == 2 && a[2].size() == 2 && a[3].size() == 2 && a[4].size() == 2)? "FAIL" : "WIN");
  return 0;
}
