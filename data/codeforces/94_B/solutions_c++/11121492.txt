#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int n,i,a,b,knows[5]={0};


    cin >> n;
    for(i=0;i<n;i++)
    {
        cin >> a >> b;
        knows[a-1]++;
        knows[b-1]++;
    }

    for(i=0;i<5;i++) if(knows[i]!=2) {cout << "WIN" << endl; return 0;}
    cout << "FAIL" << endl;

    return 0;
}
