#include <iostream>
using namespace std;
int t[6];
int main() {
    int m,a,b;
    cin >> m;
    for(int i=0;i<m;i++) {
        cin >> a >> b;
        t[a]++;
        t[b]++;
    }
    for(int i=1;i<=5;i++)
        if(t[i]>=3 || 4-t[i]>=3) {
            cout << "WIN\n";
            return 0;
        }
    cout << "FAIL\n";
    return 0;
}
