#include <stdio.h>

int main(void) {
	int m, a[10][10]={};
	scanf("%d", &m);
	int x, y;
	while(m--) {
		scanf("%d%d", &x, &y);
		--x;--y;
		a[x][y] = a[y][x]=1;
	}
	for(int i=0;i<5;i++)
		for(int j=i+1;j<5;j++)
			for(int k=j+1;k<5;k++)
				if(a[i][j]==a[j][k]&&a[j][k]==a[k][i]) {
					puts("WIN");
					return 0;
				}
	puts("FAIL");
	return 0;
}
