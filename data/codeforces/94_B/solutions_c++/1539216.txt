#include <iostream>

using namespace std;

int M,A[10][10];

int main()
{
	cin >> M;
	for (int i=0;i<M;i++)
	{
		int a,b;
		cin >> a >> b;
		A[a][b]=1;
		A[b][a]=1;
	}
	for (int i=1;i<4;i++)
		for (int j=i+1;j<5;j++)
			for (int k=j+1;k<6;k++)
			{
				if ((A[i][j]&&A[i][k]&&A[j][k])||(!A[i][j]&&!A[i][k]&&!A[j][k]))
				{
					cout << "WIN" << endl;
					return 0;
				}
			}
	cout << "FAIL" << endl;
	return 0;
}
