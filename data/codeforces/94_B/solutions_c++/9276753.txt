#include <cstdio>

int main(){
	int a[6] = {0};
	int n, u, v;
	
	scanf("%d", &n);
	for (int i = 0; i < n; i++){
		scanf("%d%d", &u, &v);
		a[u]++;
		a[v]++;
	}
	
	for (int i = 1; i < 6; i++)
		if (a[i] >= 3 || a[i] < 2){
			printf("WIN");
			return 0;
		}
	printf("FAIL");
	
	return 0;
}
