#include <cstdio>
int a[]={0,0,0,0,0,0};
int main()
{
    int i,m,t1,t2;
    scanf("%d",&m);
    for (i=0;i!=m;i++)
    {
        scanf("%d%d",&t1,&t2);
        a[t1]++; a[t2]++;
    }
    for (i=1;i!=6;i++)
    {
        if (a[i]<=1 || a[i]>=3)
        {
            printf("WIN\n");
            return 0;
        }
    }
    printf("FAIL\n");
    return 0;
}
