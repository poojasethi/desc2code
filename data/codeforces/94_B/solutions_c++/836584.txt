#include <iostream>

using namespace std;

int main(){  
    int people[6]={0};
    int m, a, b;
    cin >> m;
    for ( int i=1; i <= m; i++) {
        cin >> a;
        ++people[a];
        cin >> b;
        ++people[b];
    }
    for ( int i=1 ; i <= 5; i++)
        if (people[i] != 2) {
            cout << "WIN";
            return 0;
        }

    cout << "FAIL";

    return 0;
}
