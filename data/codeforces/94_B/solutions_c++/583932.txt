#include<iostream>

using namespace std;

int main()
{
	int a[6]={},m,l,r;
	bool f = false;;

	cin>>m;

	for(int i = 0 ; i < m ; ++i )
	{
		cin>>l>>r;

		a[l]++;
		a[r]++;
	}

	for(int i = 1 ; i < 6 ; ++i )
	{
		if(a[i] >=3 || a[i]<=1)
		{
			f = true;
			break;
		}
	}

	if(f)	cout<<"WIN"<<endl;
	else	cout<<"FAIL"<<endl;
}
