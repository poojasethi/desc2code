#include<stdio.h>
int a[15][15];
int main()
{
int n,m,x,y,i,j,k;
int found = 0;
scanf("%d",&m);
n=5;
for(i=0;i<m;i++)
{
scanf("%d%d",&x,&y);
a[x][y] = a[y][x] = 1;
}
for(i=1;i<=5;i++)
for(j=i+1;j<=5;j++)
for(k=j+1;k<=5;k++)
if(a[i][j] && a[j][k] && a[k][i])found = 1;
else if(a[i][j]==0 && a[j][k]==0 && a[k][i]==0)found = 1;
if(found)printf("WIN\n");
else printf("FAIL\n");
return 0;
}
