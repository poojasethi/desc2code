#include<stdio.h>
int main()
{
	int i,n,count[6]={0,0,0,0,0,0},a,b,p=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&a,&b);
		count[a]++;
		count[b]++;
	}
	for(i=1;i<=5;i++)
	{
		if(count[i]>=3 || count[i]<=1)
		{
			p=1;
			break;
		}
	}
	if(p==1)
		printf("WIN\n");
	else
		printf("FAIL\n");
	return 0;
}
