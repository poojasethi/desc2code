#include<cstdio>
int cnt[10];
int main()
{
	int m,from,to;
	scanf("%d", &m);
	while (m--)
	{
		scanf("%d %d", &from, &to);
		cnt[from]++;
		cnt[to]++;
	}
	bool find = false;
	for (int i = 1; i <= 5; i++)
	{
		if (cnt[i] >= 3 || cnt[i] <= 1)
		{
			find = true;
			break;
		}
	}
	if (find)
		puts("WIN");
	else
		puts("FAIL");
	return 0;
}
