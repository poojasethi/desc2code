#include <iostream>
using namespace std;

int main() {
    int n,count[6] = {0};
    cin >> n;
    for(int i = 0 ; i < n ; i++){
        int a , b;
        cin >> a >> b;
        count[a]++;
        count[b]++;
    }
    for(int i = 1 ; i < 6 ; i++){
        if( count[i] !=2){
            cout << "WIN" << endl;
            return 0;
        }
    }
    cout << "FAIL" << endl;
    return 0;
}
