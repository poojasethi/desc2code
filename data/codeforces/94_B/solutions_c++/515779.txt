#include<stdio.h>
int a[100][100]={};
int main(){
    int m,i,x,y,ans=0,z;
    scanf("%d",&m);
    for(i=0;i<m;i++){scanf("%d%d",&x,&y);a[x][y]=1;a[y][x]=1;}
    for(x=1;x<=5;x++){
        for(y=x+1;y<=5;y++){
            for(z=y+1;z<=5;z++){
                if(a[x][y]==a[y][z]&&a[y][z]==a[x][z])ans=1;
            }
        }
    }
    if(ans)puts("WIN");
    else puts("FAIL");
}
