#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int n, c, d;
int main()
{
	cin>>n;
	vector <int> a(5);
	for (int j=0; j<n; j++)
	{
		cin>>c>>d;
		a[c-1]++;
		a[d-1]++;
	}
	for (int i=0; i<5; i++)
	{
		if (a[i]!=2)
		{
			cout<<"WIN";
			exit(0);
		}
	}
	cout<<"FAIL";
	return 0;
}
