#include <iostream>
using namespace std;
int main(int argc, char* argv[])
{
	int mass[5] = {0};
	int a, b;
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
	{
		cin >> a >> b;
		mass[a - 1]++;
		mass[b - 1]++;
	}
	for(int i = 0; i < 5; i++)
	{
		if(mass[i] >= 3 || mass[i] <= 1)
		{
			cout << "WIN";
			return 0;
		}
	}
	cout << "FAIL";
	return 0;
}
