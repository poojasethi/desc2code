#include <iostream>
#include <string.h>

using namespace std;

int main(void){

	int a[6]={0}, n, i, j, pair[2]={0}, fail=0;
	
	cin >>n;

	for(i=0;i<n;i++){
		cin >>pair[0] >>pair[1];
		a[pair[0]]++;
		a[pair[1]]++;
	}

	for(i=1;i<6;i++){
		if(a[i]>=3 || a[i]==1 || a[i]==0){
			fail++;
		}
	}
	if(fail==0)
		cout <<"FAIL" <<endl;
	else
		cout <<"WIN" <<endl;

	return 0;
}
