#include<cstdio>

int m;
bool edge[5][5];

int main(){
 int i,j,k;
 int x,y;

 scanf("%d",&m);
 for(i=0;i<m;i++){
  scanf("%d%d",&x,&y);
  x--;y--;
  edge[x][y]=1;
  edge[y][x]=1;
 }

 for(i=0;i<5;i++){
  for(j=i+1;j<5;j++){
   for(k=j+1;k<5;k++){
    if(edge[i][j]==edge[j][k] && edge[j][k]==edge[k][i]){
     printf("WIN\n");
     return 0;
    }
   }
  }
 }

 printf("FAIL\n");
 return 0;
}
