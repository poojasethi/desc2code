#include <cstdio>
int n, a, b, m[6][6];
int main() {
    scanf("%d", &n); while (n--) { scanf("%d%d", &a, &b); m[a-1][b-1]=m[b-1][a-1]=1; }
    for (int i = 0;i < 5;i++)
        for (int j = i + 1;j < 5;j++)
            for (int k = j + 1;k < 5;k++)
                if (m[i][j] == m[i][k] && m[i][j] == m[j][k]) {
                    puts("WIN"); return 0;
                }
    puts("FAIL");
}
