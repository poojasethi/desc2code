#include<cstdio>
bool mat[6][6];
int main(){
int N,a,b;
scanf("%d",&N);
for(int i=0;i<N;i++){
scanf("%d%d",&a,&b);
mat[a-1][b-1]=mat[b-1][a-1]=true;
}
bool flg=false;
for(int i =0;i<5;i++)
for(int j=0;j<5;j++)
for(int k=0;k<5;k++)
if(i!=j&&j!=k&&i!=k&&((mat[i][j]&&mat[k][j]&&mat[i][k])||(!mat[i][j]&&!mat[k][j]&&!mat[i][k])))flg=true;
puts(flg?"WIN":"FAIL");
}
