#include <iostream>
#include <algorithm>

using namespace std;

int n,a[15];

int main()
{
	cin>>n;
	for(int i = 0 ; i<n ; i++)
	{
		int a1,b1;
		cin>>a1>>b1;
		a[a1]++;
		a[b1]++;
	}
	bool flag = false;	
	for(int i = 1 ; i<6 ; i++)
	{
		if(a[i] <= 1 || a[i] >= 3)
			flag = true;
	}
	if(flag)
		cout<<"WIN"<<endl;
	else
		cout<<"FAIL"<<endl;
	return 0;

}
