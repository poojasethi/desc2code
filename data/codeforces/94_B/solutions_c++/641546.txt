#include<cstdio>
using namespace std;
int main(){
    int f[6]={0, 0, 0, 0, 0, 0}, n;
    scanf("%d", &n);
    for(int i=0; i<n; i++){
        int k, j;
        scanf("%d %d", &k, &j);
        f[k]++;
        f[j]++;
    }
    for(int i=1; i<=5; i++){
        if(f[i]>=3||f[i]<=1){
            printf("WIN\n");
            return 0;
        }
    }
    printf("FAIL\n");
return 0;}
