#include <iostream>
using namespace std;
int a[6];
int main(){
	int m;
	for(int i = 1; i <= 5; i++){
		a[i] = 0;
	}
	cin >> m;
	for(int i = 1; i <= m; i++){
		int x, y;
		cin >> x >> y;
		a[x]++;
		a[y]++;
	}
	int tr = 1;
	for(int i = 1; i <= 5; i++){
		if(a[i] != 2){
			tr = 0;
		}
	}
	if(tr == 1){
		cout << "FAIL" << endl;
	}
	else{
		cout << "WIN" << endl;
	}
	return 0;
}
