#include <iostream>

using namespace std;

int main()
{
    int s[5];
    for (int i=0; i<5; i++) s[i]=0;
    int m,n;
    cin >> m;
    for (int i=0; i<2*m; i++)
    {
        cin >> n;
        s[n-1]++;
    }
    for (int i=0; i<5; i++)
    {
        if(s[i]!=2) {cout << "WIN"; return 0;}
    }

    cout << "FAIL";
    return 0;
}
