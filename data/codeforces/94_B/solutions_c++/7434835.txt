#include<stdio.h>
#include<string.h>
int main()
{
	int m,i,a,b,flag[10];
	while(scanf("%d",&m) != EOF)
	{
		memset(flag,0,sizeof(flag));
		for(i=0;i<m;i++)
		{
			scanf("%d%d",&a,&b);
			flag[a]++;flag[b]++;
		}
		for(i=1;i<=5;i++)
		{
			if(flag[i]!=2)
			{
				break;
			}
		}
		if(i>5)
		{
			printf("FAIL\n");
		}
		else
		{
			printf("WIN\n");
		}
	}
	return 0;
}
