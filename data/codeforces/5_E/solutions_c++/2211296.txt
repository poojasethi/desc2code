#include<stdio.h>
#define ll long long
#define N 1000005
int c[N],q[N],s[N*2];
int main()
{
   // freopen("in.txt","r",stdin);
    int n,i;
    while(~scanf("%d",&n))
    {
     int m=0;
     ll ans=0;
     for(i=0;i<n;i++)
     {
      scanf("%d",&s[i]);
      if(s[i]>s[m])m=i;
      s[i+n]=s[i];
     }
     int r=1;
     q[0]=s[m];c[0]=1;
     for(i=1;i<n;i++)
     {
      int y=s[i+m];
      while(q[r-1]<y)r--,ans+=c[r];
      if(q[r-1]>y)ans++,q[r]=y,c[r]=1,r++;
      else
      {
       ans+=c[r-1];
       if(r>1)ans++;
       c[r-1]++;
      }
     }
     while(r>=3)ans+=c[r-1],r--;
     if(r>=2)ans+=(c[0]>1)*c[r-1];
     printf("%I64d\n",ans);
    }
    return 0;
}
