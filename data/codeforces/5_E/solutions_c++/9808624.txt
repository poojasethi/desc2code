#include<bits/stdc++.h>
#define M 1000000007
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
ll r[1000010],l[1000010],c[1000010],h[1000010];
// r[i] ==> element >= h[i] to right
// l[i] ==> element > h[i] to left
// c[i] ==> #repeated elements between i and r[i]
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int n,i;
	ll ans=0;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>h[i];
	rotate(h,max_element(h,h+n),h+n);
	h[n]=h[0];
	for(i=n-1;i>=0;i--)
	{
		r[i]=i+1;
		while(r[i]<n && h[i] > h[r[i]])
			r[i]=r[r[i]];
		if(r[i]<n && h[i]==h[r[i]])
		{
			c[i]=c[r[i]]+1;
			r[i]=r[r[i]];
		}
	}
	for(i=0;i<n;i++)
	{
		ans+=c[i];
		if(h[i]!=h[0])
		{
			l[i]=i-1;
			while(l[i]>0 && h[i] >= h[l[i]])
				l[i]=l[l[i]];
			ans+=2;
			if(l[i]==0 && r[i]==n)
				ans--;
		}
	}
	cout<<ans<<'\n';
	return 0;
}
