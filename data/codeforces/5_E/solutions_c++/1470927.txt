#include <cstdio>
#include <algorithm>

using namespace std;

int n,a[1000001],l[1000001],r[1000001],c[1000001];

int main(void)
{
  scanf("%d",&n);
  for(int i=0;i<n;++i)
    scanf("%d",&a[i]);
  rotate(a,max_element(a,a+n),a+n);
  a[n]=a[0];
  for(int i=n-1;i>=0;--i)
  {
    r[i]=i+1;
    while((r[i]<n)&&(a[r[i]]<a[i]))
      r[i]=r[r[i]];
    if((r[i]<n)&&(a[r[i]]==a[i]))
    {
      c[i]=c[r[i]]+1;
      r[i]=r[r[i]];
    }
  }
  long long ans=0;
  for(int i=0;i<n;++i)
  {
    ans+=c[i];
    if(a[i]==a[0])
      continue;
    ans+=2;
    l[i]=i-1;
    while((l[i]>0)&&(a[i]>=a[l[i]]))
      l[i]=l[l[i]];
    if((l[i]==0)&&(r[i]==n))
      --ans;
  }
  printf("%I64d\n",ans);
  return 0;
}
