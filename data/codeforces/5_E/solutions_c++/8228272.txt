#include <bits/stdc++.h>

using namespace std;

#define N 1000010
int n, m, p;
int a[N], t[N], l[N], r[N], c[N];

int main()
{
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> t[i];
		if (m < t[i]) { m = t[i]; p = i; }
	}
	memcpy(a, t+p, sizeof(int)*(n-p));
	memcpy(a+(n-p), t, sizeof(int)*p);
	a[n] = m;

	l[0] = -1;
	for (int i = 1; i < n; i++) {
		l[i] = i-1;
		while (l[i] >= 0 && a[i] >= a[l[i]])
			l[i] = l[l[i]];
	}

	c[n] = 0;
	for (int i = n-1; i >= 0; i--) {
		r[i] = i+1;
		while (r[i] < n && a[i] > a[r[i]])
			r[i] = r[r[i]];
		if (r[i] < n && a[i] == a[r[i]]) {
			c[i] = c[r[i]] + 1;
			r[i] = r[r[i]];
		}
	}

	long long int total = 0;
	for (int i = 0; i < n; i++) {
		total += c[i];
		if (a[i] < m) {
			if (l[i] == 0 && r[i] == n)
				total += 1;
			else
				total += 2;
		}
	}

	cout << total << endl;
}
