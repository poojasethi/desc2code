#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
char s[10010];
int n, m, g[10010], x[10010];

int main()
{
	memset(g, 0, sizeof(g));
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; i++)
	{
		scanf("%s", s);
		int k = 0;
		for (int j = 0; j < m; j++)
			if (s[j] == '1')
			{
				x[k] = j;
				k++;
			}
		if (k == 0)
		{
			puts("-1");
			return 0;
		}
		for (int j = 0; j < k; j++)
		{
			int tmp1 = 1, tmp2 = x[(j + 1) % k] - x[j];
			if (tmp2 <= 0)
				tmp2 += m;
			tmp2--;
			for (int l = (x[j] + 1) % m; l != x[(j + 1) % k]; l = (l + 1) % m)
			{
				g[l] += (tmp1 < tmp2) ? tmp1 : tmp2;
				tmp1++;
				tmp2--;
			}
		}
	}
	int ans = g[0];
	for (int i = 1; i < m; i++)
		if (g[i] < ans) ans = g[i];
	printf("%d\n", ans);
	return 0;
}
