#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

#define rep(i, n) rep2(i, 0, n)
#define rep2(i, m, n) for(int i = m; i < (int)(n); ++i)
#define pb(e) push_back(e)

int n, m, ans[10010];
string str;
vector<int> v;

int main()
{
    cin >> n >> m;
    rep(i, n) {
        v.clear();
        cin >> str;
        rep(j, m) if(str[j] == '1') v.pb(j);
        if(v.empty()) {
            cout << -1 << endl;
            return 0;
        }
        v.pb(v[0] + m);
        for(int k = 0, l = v[0]; k < v.size() - 1; ++l) {
            ans[l%m] += min(l - v[k], v[k+1] - l);
            if(l + 1 == v[k+1]) ++k;
        }
    }
    int a = 1000000;
    rep(i, m) a = min(a, ans[i]);
    cout << a << endl;
}
