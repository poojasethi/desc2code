#include<cstdio>
#include<cstring>
#include<algorithm>
const int N=105,M=10005;
int a[N][M*3],l[N],r[N],ans=10000000,n,m;
char s[M];
inline int min(int a,int b){return a<b?a:b;}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++){
		int cnt=0; scanf("%s",s+1);
		for (int j=1;j<=m;j++)
			a[i][j]=a[i][m+j]=a[i][m+m+j]=s[j]-48,cnt+=a[i][j];
		if (!cnt) {printf("-1\n"); return 0;}
		}
	for (int i=1;i<=n;i++){
		for (l[i]=m;!a[i][l[i]];l[i]--);
		for (r[i]=m;!a[i][r[i]];r[i]++);	
		}
	for (int j=m+1;j<=m+m;j++){
		int res=0;
		for (int i=1;i<=n;i++){
			if (a[i][j]) l[i]=j;
			if (r[i]<j) for (r[i]++;!a[i][r[i]];r[i]++);
			res+=min(j-l[i],r[i]-j);
			}
		if (res<ans) ans=res;
		}
	printf("%d\n",ans); return 0;
}
