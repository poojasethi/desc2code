#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
#define INF  1<<29
#define nMax 105
#define mMax 20005
int dp[mMax];
int f[mMax];
char str[mMax];

int main()
{
	int i,j,n,m;
	scanf("%d%d",&n,&m);
	for(i=0; i<n; i++){
		scanf("%s",str);
		for(j=0; j<2*m; j++)	dp[j] = INF;
		for(j=0; j<2*m; j++)	if(str[j%m] == '1')	dp[j] = 0;
		for(j=0; j<2*m; j++)	dp[(j+1)%m] = min(dp[j%m]+1, dp[(j+1)%m]);
		for(j=2*m; j>=1; j--)	dp[(j-1)%m] = min(dp[j%m]+1, dp[(j-1)%m]);
		for(j=0; j<m; j++)
			if(dp[j] == INF)	{puts("-1");	return 0;}
			else f[j] += dp[j];
	}
	int Min = INF;
	for(i=0; i<m; i++)	Min = min(Min, f[i]);
	printf("%d\n",Min);
	return 0;
}
