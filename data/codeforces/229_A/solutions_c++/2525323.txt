#include<stdio.h>
#include<string.h>
#include<algorithm>
#define min(a,b) ((a)<(b)?(a):(b))
long long dp[10010],f[10010];
int n,m;
using namespace std;
void solve()
{
	int i;
	for(i=0;i<2*m;i++)
		dp[(i+1)%m]=min(dp[(i+1)%m],dp[i%m]+1);
	for(i=2*m;i>=1;i--)
		dp[(i-1)%m]=min(dp[(i-1)%m],dp[i%m]+1);
	for(i=0;i<m;i++)
		f[i]+=dp[i];
}
int main()
{
	int x,i,j;
	while(scanf("%d%d",&n,&m)!=EOF){
		memset(f,0,m*sizeof(f[0]));
		for(j=0;j<n;j++){
			for(i=0;i<m;i++)
				dp[i]=0xfffffff;
			for(i=0;i<m;i++){
				scanf("%1d",&x);
				x?dp[i]=0:1;
			}
			solve();
		}
		if(f[0]>=0xfffffff)
			printf("-1\n");
		else
		printf("%I64d\n",*min_element(f,f+m));
	}
	return 0;
}
