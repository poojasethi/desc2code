#include <iostream>
using namespace std;

int n, m, cl[10005], l, r[10005] = {}, res = 999999999;
string s, z;

int main(){

	cin >> n >> m;
	for(int i = 0; i<m; i++) z.push_back('0');
	for(int i = 0; i<n; i++){
		cin >> s;
		if(s==z){
			cout << -1;
			return 0;
		}
		for(int j = 0; j<m; j++) cl[j] = 999999999;
		l = -999999999;
		for(int j = 0; j<2*m; j++){
			if(s[j%m]=='1') l = j;
			cl[j%m] = min(cl[j%m],j-l);
		}
		l = 999999999;
		for(int j = 2*m-1; j>=0; j--){
			if(s[j%m]=='1') l = j;
			cl[j%m] = min(cl[j%m], l-j);
		}
		for(int j = 0; j<m; j++) r[j]+=cl[j];
	}
	for(int i = 0; i<m; i++) res = min(res, r[i]);
	cout << res;

}
