#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;


int main()
{
    int sum[10000]={0};    
    int ans[10000]={0};
    bool f[10000]={0};
    int n,m,i,j,k;
    cin>>n>>m;
    while(n--){
        getchar();
        for(i=0;i<m;i++)
            f[i]=getchar()-48;
        ans[0]=f[0]?0:1000000;
        for(i=1;i<2*m;i++)
           ans[i%m]=(f[i%m]?0:ans[(i-1)%m]+1);
        for(i=2*m-2;i>=0;i--)
           ans[i%m]= ans[(i+1)%m]+1<ans[i%m] ? ans[(i+1)%m]+1:ans[i%m];
        if(ans[0]>=1000000)
        {
           puts("-1");
           return 0;
        }
        for(i=0;i<m;i++)
          sum[i]+=ans[i];
    }
    
    cout<<(*min_element(sum,sum+m))<<endl;
   // cout<<endl;  system("pause");
    return 0;
}
