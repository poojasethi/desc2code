#include <stdio.h>

const int N=110,M=10010;
int n,m,x;
char a[M+1];
int l[N][M],r[N][M],T;

int min(int p,int q)
{
	return p<q?p:q;
}

int Ans()
{
	int i,j,t;
	for(i=0;i<n;++i)
	{
		scanf("%s",a);
		x=-1;
		for(j=0;j<m;++j)
		{
			if(a[j]=='1') x=j;
			l[i][j]=j-x;
		}
		if(x==-1) return -1;
		for(j=0;j<m && a[j]=='0';++j)
			l[i][j]=(m-x)+j;

		x=-1;
		for(j=m-1;j>=0;--j)
		{
			if(a[j]=='1') x=j;
			r[i][j]=x-j;
		}
		if(x==-1) return -1;
		for(j=m-1;j>=0 && a[j]=='0';--j)
			r[i][j]=x+(m-j);
	}
	T=n*m;
	for(j=0;j<m;++j)
	{
		t=0;
		for(i=0;i<n;++i)
			t+=min(l[i][j],r[i][j]);
		if(t<T)
			T=t;
	}
	return T;
}

int main()
{
	int i;
	scanf("%d%d",&n,&m);
	printf("%d\n",Ans());
	return 0;
}
