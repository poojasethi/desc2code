#include<cstdio>
const int N=10005;
int a[N],b[N];
char s[N];
int min(int a,int b){return a<b?a:b;}
int main()
{
	int i,j,x,y,n,m,ans,flag=1;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
	{
		scanf("%s",s);
		if(!flag)continue;
		for(x=0;x<m;x++)
		{
			if(s[x]=='1')
				break;
		}
		if(x==m)flag=0;
		else
		{
			for(j=0;j<m;j++)
			{
				y=x+j>=m?x+j-m:x+j;
				if(s[y]=='1')
					b[y]=0;
				else
					b[y]=b[(y-1+m)%m]+1;
			}
			for(j=0;j<m;j++)
			{
				y=x-j<0?x-j+m:x-j;
				b[y]=min(b[y],b[(y+1)%m]+1);
				a[y]+=b[y];
			}
		}
	}
	if(!flag)ans=-1;
	else
	{
		ans=0x3f3f3f3f;
		for(i=0;i<m;i++)
			ans=min(ans,a[i]);
	}
	printf("%d\n",ans);
	return 0;
}
