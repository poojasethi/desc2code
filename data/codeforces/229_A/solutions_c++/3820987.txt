#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=101;
const int maxm=10001;
int l[maxn][maxm],r[maxn][maxm];
char a[maxm];
int n,m,s;

int solve(){
  int i,j,k;
  for (i=0;i<n;i++){
    scanf("%s",a);
    k=-1;
    for (j=0;j<m;j++){
      if (a[j]=='1') k=j;
      l[i][j]=j-k;
    }
    if (k<0) return -1;
    for (j=0;j<m && a[j]=='0';j++) l[i][j]=m-k+j;
    k=-1;
    for (j=m-1;j>=0;j--){
      if (a[j]=='1') k=j;
      r[i][j]=k-j;
    }
    if (k<0) return -1;
    for (j=m-1;j>=0 && a[j]=='0';j--) r[i][j]=k+m-j;
  }
  s=n*m;
  for (j=0;j<m;j++){
    for (k=i=0;i<n;i++) k+=min(l[i][j],r[i][j]);
    s=min(s,k);
  }
  return s;
}

int main(){
  scanf("%d%d",&n,&m);
  printf("%d\n",solve());
  return 0;
}
