#include <iostream>
#include <string>
using namespace std;

string s[100];
int dist[100][10000];

const int INF = 1<<22;

int main(){
	int n,m;
	cin>>n>>m;
	for(int i = 0; i < n; i++)cin>>s[i];
	
	for(int i = 0; i < n; i++){
		for(int j = 0; j < m; j++){
			if(s[i][j]=='0')dist[i][j] = INF;
		}
	}
	
	for(int i = 0; i < n; i++){
		for(int j = 0; j < 2*m; j++){
			dist[i][(j+1)%m] = min(dist[i][(j+1)%m],dist[i][j%m]+1);
		}
		for(int j = 2*m; j > 0; j--){
			dist[i][(j-1)%m] = min(dist[i][(j-1)%m],dist[i][j%m]+1);
		}
	}
	
	int ans = INF;
	for(int j = 0; j < m; j++){
		int t = 0;
		for(int i = 0; i < n; i++){
			t += dist[i][j];
		}
		ans = min(ans,t);
	}
	cout<<(ans==INF?-1:ans)<<endl;
	return 0;
}
