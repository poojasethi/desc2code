#include<iostream>
#include<algorithm>
#include<vector>
#include<iomanip>
#include<string>
#include<cmath>
#include<cstdio>

using namespace std;

int n,m,k,ans=1000000000;
string s;
int a[102][10010];

int main()
{
	cin>>n>>m;
	for(int i=0;i<n;i++)
	{
		k=10000000;
		cin>>s;
		for(int j=0;j<2*m;j++)
		{
			k++;
			if(s[j%m]=='1')
				k=0;
			a[i][j%m]=k;
		}
		k=10000000;
		for(int j=2*m-1;j>=0;j--)
		{
			k++;
			if(s[j%m]=='1')
				k=0;
			a[i][j%m]=min(k,a[i][j%m]);
		}
	}
	for(int i=0;i<m;i++)
	{
		k=0;
		for(int j=0;j<100;j++)
			k+=a[j][i];
		if(k<ans)
			ans=k;
	}
	if(ans>10000000)
		cout<<-1<<endl;
	else
		cout<<ans<<endl;
	return 0;
}
