#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

int ans[1000*10+10];
vector <int> v;

int main(){
    int n, m;
    cin>>n>>m;
    for (int i=0; i<n; i++){
        v.clear();
        string s;
        cin>>s;
        for (int j=0; j<m; j++)
            if (s[j]=='1')
                v.push_back(j);
        if (v.empty()){
            cout<<-1<<endl;
            return 0;
        }
        v.push_back(v[0]+m);
        for (int k=0, l=v[0]; k<int(v.size())-1; l++){
            ans[l%m]+=min(l-v[k], v[k+1]-l);
            if (l+1==v[k+1])
                k++;
        }
    }
    int x=1000*1000+10;
    for (int i=0; i<m; i++)
        x=min(x, ans[i]);
    cout<<x<<endl;
    return 0;
}
