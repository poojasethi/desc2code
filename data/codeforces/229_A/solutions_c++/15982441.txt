#include<iostream>
using namespace std;
int main()
{
	int n,m,count,k,i,j,max=100000000;
	cin>>n>>m;
	char a[n][m];int b[n][m];
	for(i=0;i<n;i++)
	{  
		cin>>a[i];count=0;k=0;
        while(a[i][k]=='0'&&k<m)k++;k--;
        if(k==m-1){cout<<-1;return 0;}
        for(j=0;j<m;j++)
        {   if(k==-1)k=m-1;
        	if(a[i][k]=='0')count++;
        	else count=0;
        	b[i][k--]=count;
		}
		k=m-1;while(a[i][k]=='0'&&k>=0)k--;k++;count=0;
		for(j=0;j<m;j++)
        {   if(k==m)k=0;
        	if(a[i][k]=='0')count++;
        	else count=0;
        	b[i][k]=min(count,b[i][k]);k++;
		}
	}	
	for(i=0;i<m;i++)
	{   count=0;
		for(j=0;j<n;j++)count+=a[j][i]=='1'?0:b[j][i];
		max=min(max,count);
	}cout<<max;
}
