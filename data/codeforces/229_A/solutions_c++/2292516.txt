#include <iostream>
#include <cstring>
using namespace std;
const int inf=1000000009;
int n,m,l[110][20200],r[110][20200];
char g[110][20200];
int main(void) {
  cin>>n>>m;
  for(int i=0;i<n&&cin>>g[i];++i)for(int j=0;j<m;++j)g[i][j+m]=g[i][j];
  memset(l,0x3f,sizeof(l)),memset(r,0x3f,sizeof(r));
  for(int i=0;i<n;++i){
    for(int j=1;j<(m<<1);++j)l[i][j]=('1'==g[i][j])?0:(l[i][j-1]+1);
    for(int j=(m<<1);j-->1;)r[i][j-1]=('1'==g[i][j-1])?0:(r[i][j]+1);
  }
  int t=inf;
  for(int i=0;i<m;++i){
    int s=0;
    for(int j=0;j<n;++j)s=min(inf,s+min(min(l[j][i],l[j][i+m]),min(r[j][i],r[j][i+m])));
    t=min(t,s);
  }
  cout<<(t<inf?t:-1)<<endl;
    return 0;
}
