#include <bits/stdc++.h>
using namespace std;

int a[105][10005], n, m, d[105][10005];
char s[10005];
int main() {
	scanf("%d%d", &n, &m);
	for(int i=1; i<=n; ++i) {
		scanf("%s", s+1);
		int flag=0;
		for(int j=1; j<=m; ++j) {
			a[i][j]=s[j]-'0';
			if(a[i][j]) flag=1;
		}
		if(!flag) { puts("-1"); return 0; }
	}
	for(int i=1; i<=n; ++i) {
		int l=0, r=0;
		for(int j=m; j>=1; --j) { if(a[i][j]) break; ++l; }
		for(int j=1; j<=m; ++j) { if(a[i][j]) break; ++r; }
		for(int j=1; j<=m; ++j) { if(a[i][j]) l=0; else ++l; d[i][j]=d[i-1][j]+l; }
		for(int j=m; j>=1; --j) { if(a[i][j]) r=0; else ++r; d[i][j]=min(d[i][j], d[i-1][j]+r); }
	}
	int ans=~0u>>1;
	for(int i=1; i<=m; ++i) ans=min(ans, d[n][i]);
	printf("%d\n", ans);
	return 0;
}
