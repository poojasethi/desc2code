#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;
const int MAXM= 1000 * 10 + 10; 
vector <int> v;
int d[MAXM];

int main(){
	int n , m;
	cin >> n >> m;
	for(int i= 0; i< n; i++){
		v.clear();
		string s;
		cin >> s;
		for(int j= 0; j< m; j++)
			if(s[j]== '1')
				v.push_back(j);
		if(v.size()== 0){
			cout << -1 << endl;
			return 0;
		}
		v.push_back(v[0]+ m);
		for(int l= v[0], k= 0; l< m+v[0]; l++){
			d[l%m]+= min(abs(l-v[k]), abs(v[k+1]-l));
			if(l== v[k+1])
				k++;
		}
	}
	int minimum= 1000* 1000+ 10;
	for(int i= 0; i< m; i++){
		minimum= min(minimum, d[i]);
	}
	cout << minimum << endl;
	return 0;
}
