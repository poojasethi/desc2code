#include<cstdio>
#include<deque>
#include<algorithm>
#define N 110
using namespace std;
deque<int> t[N];
char s[N][10010];
int main(){
	int n, m, ans = 999999999;
	scanf("%d%d", &n, &m);
	for(int i = 0, fl; i < n; i++){
		scanf("%s", s[i]); fl = 0;
		for(int j = 0; j < m; j++){
			if(s[i][j] == '1')
				t[i].push_back(j), fl = 1;
		}
		if(fl == 0) return puts("-1"), 0;
		t[i].push_front(t[i][t[i].size()-1]-m);
		t[i].push_back(t[i][1]+m);
	}
	for(int i = 0; i < m; i++){
		int cnt = 0;
		for(int j = 0; j < n; j++){
			int p = lower_bound(t[j].begin(), t[j].end(), i)-t[j].begin();
			cnt += min(abs(t[j][p]-i), abs(t[j][p-1]-i));
		}
		ans = min(cnt, ans);
	}
	printf("%d\n", ans);
}
