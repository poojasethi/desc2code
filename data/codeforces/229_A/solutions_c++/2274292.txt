#include<cstdio>
#include<vector>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

const int INF=1<<29;

int main(){
	int h,w; scanf("%d%d ",&h,&w);
	vector<int> one[100];
	rep(i,h){
		char s[10001]; gets(s);
		rep(j,w) if(s[j]=='1') {
			one[i].push_back(j-w);
			one[i].push_back(j);
			one[i].push_back(j+w);
		}
		sort(one[i].begin(),one[i].end());
		if(one[i].empty()) return puts("-1"),0;
	}

	int ans=INF;
	rep(j,w){
		int cnt=0;
		rep(i,h){
			int idx=lower_bound(one[i].begin(),one[i].end(),j)-one[i].begin();
			cnt+=min(one[i][idx]-j,j-one[i][idx-1]);
		}
		ans=min(ans,cnt);
	}
	printf("%d\n",ans);

	return 0;
}
