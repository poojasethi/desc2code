#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

const int maxn = 10100;

int n, m, opt[maxn];
char line[maxn * 3];


int main() {
	cin >> n >> m;
	while (n--) {
		cin >> line;
		bool all0 = true;
		for (int i = 0; i < m; ++i) {
			if (line[i] == '1') {
				all0 = false;
			}
			line[i + 2 * m] = line[i + m] = line[i];
		}
		if (all0) {
			printf("-1\n");
			return 0;
		}

		int l = m, r = m;
		while (line[l] != '1') {
			--l;
		}
		for (int i = m; i < 2 * m; ++i) {
			if (line[i] == '1') {
				l = r = i;
			} else if (line[i] == '0' && r <= i) {
				r = i;
				while (line[r] == '0') {
					++r;
				}
			}
			opt[i - m] += min(i - l, r - i);
		}
	}
	cout << *min_element(opt, opt + m) << endl;
	return 0;
}
