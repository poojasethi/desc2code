#include <iostream>
#include <string.h>
#include <string>
#include <math.h>
using namespace std;
int f[10005];
int l[10005],r[10005];
int main()
{
	int n,m,i,j,k;
	cin>>n>>m;
	for (i=0;i<m;i++) {f[i]=0;}
	bool ans=1;
	for (i=1;i<=n;i++)
	{
		string s;
		cin>>s;
		ans=0;
		for (j=0;j<m;j++) {if (s[j]=='1') {ans=1; break;}}
		if (!ans) {cout<<-1<<endl; ans=0; break;}
		s=s+s+s;
		j=m;
		while (s[j]=='0') {j--;}
		for (k=m;k<=2*m-1;k++)
		{
			if (s[k]=='1') {j=k;}
			l[k-m]=k-j;
		}
		j=2*m-1;
		while (s[j]=='0') {j++;}
		for (k=2*m-1;k>=m;k--)
		{
			if (s[k]=='1') {j=k;}
			r[k-m]=j-k;
		}
		for (j=0;j<m;j++) {f[j]=f[j]+min(l[j],r[j]);}
	}
	if (ans)
	{
		j=10000000;
		for (i=0;i<m;i++) {j=min(j,f[i]);}
		cout<<j<<endl;
	}
}
