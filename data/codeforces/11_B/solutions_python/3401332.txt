import math

x = abs(input())
r = int(math.sqrt(2 * x))
#find first r such that S = 1+2+...+r >= x
while (r + 1) * r / 2 < x:
    r += 1
#then find a r such that diff = S - x is even
#this r is the result because we can get S - 2, S - 4 ... S - 2*r using r jumps
#to get S - 2k, just jump to the other direction on k-th step
while ((r + 1) * r / 2 - x) & 1:
    r += 1
print r
