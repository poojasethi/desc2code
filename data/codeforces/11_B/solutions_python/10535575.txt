def main():
    x = abs(int(raw_input()))

    jump_distance = 1
    jump_count = 0
    distance = 0;

    while distance < x:
        distance += jump_distance
        jump_distance += 1
        jump_count += 1

    if distance == x:
        print jump_count
    else:
        while (distance - x) % 2 != 0:
            distance += jump_distance
            jump_distance += 1
            jump_count += 1
        print jump_count


if __name__ == '__main__':
    main()
