#include<cstdio>
#include<algorithm>
using namespace std;
int i,j,n,m,a[1111][1111],f[5][1111][1111],ans;
int main(){
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;i++)for(j=1;j<=m;j++)scanf("%d",&a[i][j]);
    for(i=1;i<=n;i++)for(j=1;j<=m;j++)f[3][i][j]=a[i][j]+max(f[3][i-1][j],f[3][i][j-1]);
    for(i=n;i;i--)for(j=1;j<=m;j++)f[4][i][j]=a[i][j]+max(f[4][i+1][j],f[4][i][j-1]);
    for(i=n;i;i--)for(j=m;j;j--)f[1][i][j]=a[i][j]+max(f[1][i+1][j],f[1][i][j+1]);
    for(i=1;i<=n;i++)for(j=m;j;j--)f[2][i][j]=a[i][j]+max(f[2][i-1][j],f[2][i][j+1]);
    for(i=2;i<n;i++)for(j=2;j<m;j++)
    ans=max(ans,max(f[1][i+1][j]+f[3][i-1][j]+f[2][i][j+1]+f[4][i][j-1],f[2][i-1][j]+f[4][i+1][j]+f[1][i][j+1]+f[3][i][j-1]));
    printf("%d\n",ans);
}
