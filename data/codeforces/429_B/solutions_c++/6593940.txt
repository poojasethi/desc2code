#include<iostream>
using namespace std;
int n,m;
long long  a[1100][1100];
long long  d1[1100][1100];
long long  d2[1100][1100];
long long  d3[1100][1100];
long long  d4[1100][1100];
int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	for(int j=1;j<=m;j++)
	cin>>a[i][j];
	for(int i=1;i<=n;i++)
	for(int j=1;j<=m;j++)
	d1[i][j]=max(d1[i-1][j],d1[i][j-1])+a[i][j];
	for(int i=n;i>=1;i--)
	for(int j=m;j>=1;j--)
	d2[i][j]=max(d2[i+1][j],d2[i][j+1])+a[i][j];
	for(int i=n;i>=1;i--)
	for(int j=1;j<=m;j++)
	d3[i][j]=max(d3[i+1][j],d3[i][j-1])+a[i][j];
	for(int i=1;i<=n;i++)
	for(int j=m;j>=1;j--)
	d4[i][j]=max(d4[i-1][j],d4[i][j+1])+a[i][j];
	long long sum=0;
	for(int i=2;i<n;i++)
	for(int j=2;j<m;j++)
	{
		sum=max(sum,d1[i-1][j]+d2[i+1][j]+d3[i][j-1]+d4[i][j+1]);
		sum=max(sum,d1[i][j-1]+d2[i][j+1]+d3[i+1][j]+d4[i-1][j]);
	}
	cout<<sum<<endl;
	//cin>>n;
}
