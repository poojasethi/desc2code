#include<bits/stdc++.h>
using namespace std;
int i,j,n,m,a[1111][1111], f[5][1111][1111], ans;
int main() {
	scanf("%d %d",&n,&m);
	for(int i=1;i<=n;i++) 
		for(int j=1;j<=m;j++) 
			scanf("%d",&a[i][j]);
	for(int i=1;i<=n;i++) {
		for(int j=1;j<=m;j++) {
			f[3][i][j] = a[i][j] + max(f[3][i-1][j],f[3][i][j-1]);
		}
	}
	for(int i=n;i>0;i--) {
		for(int j=1;j<=m;j++) {
			f[4][i][j] = a[i][j] + max(f[4][i+1][j],f[4][i][j-1]);
		}
	}
	for(int i=n;i>0;i--) {
		for(int j=m;j>0;j--) {
			f[1][i][j] = a[i][j] + max(f[1][i+1][j],f[1][i][j+1]);
		}
	}
	for(int i=1;i<=n;i++)for(int j=m;j;j--)f[2][i][j]=a[i][j]+max(f[2][i-1][j],f[2][i][j+1]);
    for(int i=2;i<n;i++)for(int j=2;j<m;j++)
    ans=max(ans,max(f[1][i+1][j]+f[3][i-1][j]+f[2][i][j+1]+f[4][i][j-1],f[2][i-1][j]+f[4][i+1][j]+f[1][i][j+1]+f[3][i][j-1]));
    printf("%d\n",ans);
	return 0;

}
