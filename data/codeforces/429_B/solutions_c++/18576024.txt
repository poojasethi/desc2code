#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll a[1002][1002];
ll dp1[1004][1004],dp2[1004][1004],dp3[1004][1004],dp4[1004][1004];
int main()
{
	ll i,j,k,l,m,n;
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n>>m;
	for(i=1;i<=n;i++)for(j=1;j<=m;j++)cin>>a[i][j];
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
				dp1[i][j]=max(dp1[i][j-1],dp1[i-1][j])+a[i][j];
	for(i=n;i>0;i--)
		for(j=m;j>0;j--)
			dp2[i][j]=max(dp2[i+1][j],dp2[i][j+1])+a[i][j];
	for(i=1;i<=n;i++)
		for(j=m;j>0;j--)
			dp3[i][j]=max(dp3[i-1][j],dp3[i][j+1])+a[i][j];
	for(i=n;i>0;i--)
		for(j=1;j<=m;j++)
			dp4[i][j]=max(dp4[i+1][j],dp4[i][j-1])+a[i][j];
	ll ans=0;
	for(i=2;i<n;i++)
		for(j=2;j<m;j++)
		{
			ans=max(ans,dp1[i-1][j]+dp2[i+1][j]+dp3[i][j+1]+dp4[i][j-1]);
			ans=max(ans,dp1[i][j-1]+dp2[i][j+1]+dp3[i-1][j]+dp4[i+1][j]);
		}
	cout<<ans<<endl;
}
