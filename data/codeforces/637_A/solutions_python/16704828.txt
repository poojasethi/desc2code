d = {}
e = {}
n = input()

nums = [int(x) for x in raw_input().split()]

i = 0
num_val_max = 1
f = nums[0]
for num in nums:
    num_val = d.get(num)
    if num_val is None:
        d.update({num: 1})
        e.update({num: i})
    else:
        d.update({num: num_val+1})
        e.update({num: i})
        if num_val_max < num_val+1:
            num_val_max = num_val+1
            f = num
    i += 1
print(f)
