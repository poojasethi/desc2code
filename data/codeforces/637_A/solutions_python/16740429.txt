from collections import Counter

def vote_pics(pics):
    if len(pics) == 1:
        return pics[0]
    c = Counter(pics)
    results = dict(c)
    top_vote = c.most_common(1)[0][1]
    winners = [result for result in results if results[result] == top_vote]
    if len(winners) == 0:
        return winners[0]
    else:
        pos = []
        for winner in winners:
            i = len(pics) - 1
            while i >= 0:
                if pics[i] == winner:
                    pos.append(i)
                    break
                i -= 1
        return pics[min(pos)]
