
x = int(raw_input())
mylist = map(int,raw_input().strip().split())

cnt_d = dict()
cur_max = 0
max_index = 0

for i in xrange(len(mylist)):
	identy = mylist[i]
	cnt_d[identy] = cnt_d[identy]+1 if identy in cnt_d else 1
	cur_cnt = cnt_d[identy]

	if cur_cnt>cur_max:
		cur_max = cur_cnt
		max_index = i

print mylist[max_index]
