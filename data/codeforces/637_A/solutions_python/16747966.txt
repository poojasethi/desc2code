# -*- coding: utf-8 -*-

class Photos(object):
	def __init__(self, n, L):
		self.n = n
		self.L = L
		self.d = {}
	def solve(self):
		u = -1
		for id in self.L:
			if self.d.has_key(id):
				self.d[id] = self.d[id] + 1
			else:
				self.d[id] = 1
			if u == -1 or self.d[id] > self.d[u]:
				u = id
		return u

if __name__ == '__main__':
	n = input()
	L = map(int, raw_input().split())
	photos = Photos(n, L)
	print photos.solve()
