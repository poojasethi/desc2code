n = int(raw_input())
l = list(map(int,raw_input().split(' ')))
points = []
ids = []
index = []
e = 0
for i in l:
	if (i in ids)==0:
		ids.append(i)
		points.append(0)
		index.append(e)
	else:
		a = ids.index(i)
		points[a] += 1
		index[a] = e
	e += 1
m = 100000000
j = max(points)
e = 0
for i in points:
	if i==j:
		if m>index[e]:
			m = index[e]
	e += 1
