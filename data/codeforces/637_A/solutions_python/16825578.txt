n =int(raw_input())
a = []
b=map(int,raw_input().split())
for x in b:
    ind = -1
    for i in range(len(a)):
        if a[i][0]==x:
            ind=i
    if ind == -1:
        a.append([x,1])
    else:
        c = [x,a[ind][1]+1]
        a.remove([x,a[ind][1]])
        a.append(c)
max = 0
for i in a:
    if max<i[1]:
        max = i[1]
        
for i in a:
    if i[1] == max:
        print(i[0])
        break;
