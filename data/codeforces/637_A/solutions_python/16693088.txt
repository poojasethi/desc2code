import collections


def main():
    input()
    likes = map(int, raw_input().split())
    best_picture = -1
    count = collections.Counter()
    for like in likes:
        count[like] += 1
        if best_picture == -1 or count[best_picture] < count[like]:
            best_picture = like
    # and the oscar goes to:
    print(best_picture)


if __name__ == '__main__':
    main()
