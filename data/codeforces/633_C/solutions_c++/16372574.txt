#include <bits/stdc++.h>
//Damned if I don't take what I want.

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

const int A = 1e4 + 100, BASE = 1e6 + 3, MOD = 98765431;

ll par[A];
map<ll, string> htos;

void print(int n){
	
	if (n == 0) return;
	string t = htos[ par[n] ];
	print(n - t.size() );
	cout << t << " ";
}

int main(){
	ios::sync_with_stdio( false );
	cout.tie(0); cin.tie(0);

	string s;
	int n; cin >> n >> s;

	memset(par, -1, sizeof par);
	par[0] = 0;

	int m; cin >> m;
	while(m--){
		string t; cin >> t;
		
		ll hash = 0;
		for (int i = 0; i < t.size(); i++)
			hash = hash * BASE +  tolower(t[i]);

		htos[ hash ] = t;

	}

	for (int i = 1; i <= n; i++){
		ll hash = 0;
		for (int j = i - 1; j >= 0; j--){
			hash = hash * BASE + tolower(s[j]);

			if (par[j] != -1 && htos.count(hash)){
				par[i] = hash;
				break;
			}
		}
	}
		
	print(n);
}
