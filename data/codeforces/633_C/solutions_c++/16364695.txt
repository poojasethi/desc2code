#include <cstdio>
#include <queue>
using namespace std;
char s[10010];
char w[2000010];
int st[100010];
inline char lower(char c) {
	return c <= 'Z' ? c - 'A' + 'a' : c;
}
int e[1000010][26];
int nd[1000010];
int len[1000010];
int prev[10010], dp[10010];
int l;
int main() {
	int n, m, p = 0;
	scanf("%d", &n);
	scanf("%s", s);
	scanf("%d", &m);
	l = 1;
	for (int i = 0; i < m; i++) {
		scanf("%s", w + p);
		int k = 0;
		for (; w[p]; p++) {
			int c = lower(w[p]) - 'a';
			if (!e[k][c]) {
				e[k][c] = l++;
			}
			k = e[k][c];
		}
		nd[k] = i + 1;
		len[i] = p - st[i];
		p++;
		st[i+1] = p;
	}
	dp[n] = 1;
	for (int i = n; i > 0; i--) {
		if (!dp[i]) continue;
		int k = 0;
		for (int j = i - 1; j >= 0; j--) {
			k = e[k][s[j]-'a'];
			if (!k) break;
			if (nd[k]) {
				dp[j] = 1;
				prev[j] = nd[k] - 1;
			}
		}
	}
	int j = 0;
	while (j < n) {
		printf("%s ", w + st[prev[j]]);
		j += len[prev[j]];
	}
	return 0;
}
