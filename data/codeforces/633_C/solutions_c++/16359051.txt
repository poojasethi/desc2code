#include<cstdio>
#include<cstring>
using namespace std;
const int N=100010,M=1000010;
int n,m,i,j,k,x,len,st[N],en[N],id[M],son[M][26],tot,cnt;
int ans[N],fin;
char a[N],s[N],q[M];
int f[N],g[N],w[N];
int main(){
  scanf("%d%s%d",&n,a+1,&m);
  for(i=1;i<=m;i++){
    scanf("%s",s);
    len=strlen(s);
    st[i]=cnt+1;
    for(j=x=0;j<len;j++){
      q[++cnt]=s[j];
      k=s[j]-'a';
      if(s[j]>='A'&&s[j]<='Z')k=s[j]-'A';
      if(!son[x][k])son[x][k]=++tot;
      x=son[x][k];
    }
    id[x]=i;
    en[i]=cnt;
  }
  for(f[0]=i=1;i<=n;i++){
    for(j=i,x=0;j;j--){
      if(i-j>1010)break;
      x=son[x][a[j]-'a'];
      if(!x)break;
      if(id[x]&&f[j-1]){
        f[i]=1;
        g[i]=j-1;
        w[i]=id[x];
        break;
      }
    }
  }
  for(i=n;i;i=g[i])ans[++fin]=w[i];
  for(i=fin;i;i--){
    for(j=st[ans[i]];j<=en[ans[i]];j++)putchar(q[j]);
    if(i>1)putchar(' ');
  }
  return 0;
}
