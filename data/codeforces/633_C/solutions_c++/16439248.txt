#include<bits/stdc++.h>
using namespace std;
map<string,string> dict;
vector<string> res;
bool rec(string s,int cur)
{
	if(s.size()==cur)
	return 1;
	string t="";
	for(int i=cur;i<s.length();i++)
	{
		t+=s[i];
		if(dict.find(t)!=dict.end()){
		res.push_back(dict[t]);
		if(rec(s,i+1))
		return 1;
		else
		res.pop_back();
	}
	}
	return 0;
}
int main()
{
	int n,m;
	string s;
	string key,value;
	cin>>n;
	cin>>s;
	cin>>m;
	for(int i=0;i<m;i++)
	{
		cin>>value;
		key=value;
		reverse(key.begin(),key.end());
		transform(key.begin(),key.end(),key.begin(),::tolower);
		dict[key]=value;
	}
	rec(s,0);

	for(int i=0;i<res.size();i++)
	cout<<res[i]<<" ";
	cout<<"\n";
	return 0;
}
