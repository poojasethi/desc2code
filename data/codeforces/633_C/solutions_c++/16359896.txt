#include<bits/stdc++.h>
#include<stdlib.h>
using namespace std;
#define pb push_back
#define mod 1000000007
map<string,bool> mp;
map<string,string> mp2;
int l;
string s;
list<string> ans;

bool rec(int i,vector<bool>& dp)
{
	if(i==l)
		return true;
	if(dp[i]==false)
		return 0;

	int j=i;
	bool b;
	string p;
	for(j;j<l;j++)
	{
		p+=s[j];
		if(mp.find(p)!=mp.end())
		{
			b=rec(j+1,dp);
			if(b==true)
			{
				ans.push_front(p);
				return true;
			}
		}
	}
	dp[i]=false;
	return 0;
}


int main()
{
	int i,m,k=0,n,a,j;
	bool b;
	cin>>l;
	string p,q;
	cin>>s;
	cin>>n;

	for(i=0;i<n;i++)
	{
		cin>>p;
		q=p;
		for(j=0;j<p.length();j++)
		{
			if(p[j]>='A' && p[j]<='Z')
				p[j]+='a'-'A';
		}
		reverse(p.begin(),p.end());
		mp[p]=1;
		mp2[p]=q;
	}
	vector<bool> dp(l,true);
	b=rec(0,dp);
	if(b==true)
	{
		while(!ans.empty())
		{
			p=ans.front();
			cout<<mp2[p]<<" ";
			ans.pop_front();
		}
	}
}
