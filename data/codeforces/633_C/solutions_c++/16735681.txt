using namespace std;
ll read()
{
    ll x=0,f=1;char ch=getchar();
    while(ch<'0'||ch>'9'){if(ch=='-')f=-1;ch=getchar();}
    while(ch>='0'&&ch<='9'){x=x*10+ch-'0';ch=getchar();}
    return x*f;
}
int n,m;
char a[10005],ch[1005];
string b;
ll f[10005];
map<ll,string> mp;
int main()
{
    n=read();
    scanf("%s",a);
    m=read();
    for(int i=1;i<=m;i++)
    {
        scanf("%s",ch);b="";
		int l=strlen(ch);
        for(int j=0;j<l;j++)b+=ch[j];
        ll h=0;
        for(int j=l-1;j>=0;j--)
        	h=h*31+tolower(b[j]);
        mp[h]=b;
    }
    f[n]=1;
    for(int i=n-1;i>=0;i--)
    {
    	ll h=0;
    	for(int j=1;j<=1000&&i+j<=n;j++)
    	{
    		h=h*31+a[i+j-1];
    		if(mp.count(h)&&f[i+j])
    		{
    			f[i]=h;
    			break;
    		}
    	}
    }
    for(int i=0;i<n;i+=mp[f[i]].length())
    	cout<<mp[f[i]]<<' ';
    return 0;
}
