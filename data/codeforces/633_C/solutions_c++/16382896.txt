void add(int x,int ind)
{
	if(ind==s.size())
	{
		f[x]=IND;
		return;
	}
	char c=tolower(s[ind])-'a';
	
	if(!tr[x][c])
	{
		tr[x][c]=++N;
	}
	add(tr[x][c],ind+1);
}
void go(int x,int ind,int st)
{
	if(f[x])
	{
		d[ind-1]=st;
		p[ind-1]=f[x]-1;
	}
	char c=tolower(S[ind])-'a';
	if(ind>=S.size())return ;
	if(tr[x][c])go(tr[x][c],ind+1,st);
}
