#include<stdio.h>
#include<iostream>
#include<string.h>
#include<map>
#include<vector>
using namespace std;
vector <string> wlist;
vector <string>::iterator it;
map <string,string> dict;
char key[10005]={'\0'},wrd[1005]={'\0'},tmp[1005]={'\0'};
int n,i,m,j,len;
int tour(int k)
{
   if(k==n) return 1;
   string t2;
   for(int i=k;i<n;i++)
   {
      t2+=key[i];
      //cout << t2 << "\n";
      if(dict.find(t2)!=dict.end())
      {
         wlist.push_back(dict[t2]);
         //printf("added word\n");
         if(tour(i+1)==1) return 1;
         else wlist.pop_back();
      }
   }
   return 0;
}
int main()
{
  scanf("%d",&n);
  scanf("%s",key);
  scanf("%d",&m);
  for(i=1;i<=m;i++)
  {
     scanf("%s",wrd);
     strcpy(tmp,wrd);
     strrev(tmp);
     len=strlen(tmp);
     for(j=0;j<len;j++) tmp[j]=tolower(tmp[j]);
     dict[tmp]=wrd;
     //cout << tmp << "\n";
  }
  tour(0);
  for(it=wlist.begin();it!=wlist.end();it++)
  {
     cout << *it << " ";
  }
return 0;
}
