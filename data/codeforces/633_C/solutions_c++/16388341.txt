#include <bits/stdc++.h>
using namespace std;
const int N = 1e6+10;
int trie[N][26], mark[N][26];
string ar[(int)1e5+2];
int pos=1,rpos=0;
string s,a;
const int diff = abs('a'-'A');
bool dp[10005];
int ret[(int)1e5+5];
void sol(int n){
	if(n==s.size()){
		int i;
		for(i=rpos-1;i>=0;--i)
			cout<<ar[ret[i]]<<" ";
		cout<<endl;exit(0);
	}
	if(dp[n])return;
	dp[n]=1;
	int i,next=1;char c;
	for(i=n;i<s.size();++i){
		c= s[i]-'a';
		if(!trie[next][c])break;
		if(mark[next][c]){
			ret[rpos++]=mark[next][c];
			sol(i+1);
			--rpos;
		}
		next= trie[next][c];
	}
}
int main(){
	int n,m,i,j;scanf("%d",&n);
	cin>>s;
	char c;
	cin>>m;
	reverse(s.begin(),s.end());
	for(i=1;i<=m;++i){
		cin>>a;
		ar[i]=a;
		int next=1;
		for(j=0;j<a.size();++j){
			c= a[j];
			if('A'<=c && c<='Z')c+=diff;
			if(trie[next][c-'a']==0){
				trie[next][c-'a']= pos+1;++pos;
			}
			if(j==a.size()-1){
				mark[next][c-'a']=i;
			}
			next= trie[next][c-'a'];
		}
	}
	sol(0);
}
