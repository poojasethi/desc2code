# include <cstdio>
# include <iostream>
# include <cstring>
# include <map>
# include <string>
typedef long long LL;
using namespace std;
# define p 111
# define MAX 100009
# define inf 99999999
LL str[MAX];
map<LL, string> hs;
string s;
void print(int n) 
{
  if ( n==0 ) 
    return;
  string t = hs[str[n]];
  int len = t.size();
  print( n - len );
  cout<<t<<" ";
}
int main(void)
{
  int n, m;
  cin >> n >> s >> m;
  for ( int i = 0; i < m; i++ )  
  {
    string ss;
    cin >> ss;
    LL tmp = 0;
    for ( int j = 0; j < ss.size(); j++ ) 
    {
      tmp = tmp * p + tolower(ss[j]);
    }
    hs[tmp] = ss;
  }
  for ( int i = 0;i < MAX;i++ )
  {
    str[i] = -inf;
  }
  str[0] = 0;
  for (int i = 0; i < n; i++) 
  {
    LL tmp = 0;
    for (int j = i; j >= 0; j--) 
    {
      tmp = tmp*p + s[j];
      if (str[j] != -inf && hs.count(tmp)>=1 ) 
      {
        str[i + 1] = tmp;
        break;
      }
    }
  }
  print(n);
}
