#include <bits/stdc++.h>
using namespace std;
#define ll unsigned long long
int n,m;
string s,p,ans;
ll h[10100],M[10100];
map<ll,string> q;
int memo[10100];
void prepare(){
	M[0] = 1;
	for(int i = 1; i <= 10000; i++) M[i] = M[i-1] * 2309;
	for(int i = 1; i <= n; i++) h[i] = h[i-1]*M[1] + s[i-1];
}
int solve(int in){
	if(in==s.size()+1) return 1;
	if(memo[in]!=-1) return memo[in];
	for(int i = in; i <= n; i++){
		ll ch = h[i] - h[in-1]*M[i-in+1];
		if(q[ch]!=""&&solve(i+1)==1) return memo[in] = 1;
	}
	return memo[in] = 0;
}
int main(){
	//freopen("in.in","r",stdin);
	cin >> n;
	cin >> s;
	prepare();
	cin >> m;
	for(int i = 1; i <= m; i++){
		cin >> p;
		ll cur = 0;
		for(int i = p.size()-1; i >= 0; i--){
			ll k = p[i];
			if('A'<=p[i]&&p[i]<='Z') k = k - 'A' + 'a';
			cur = cur*M[1] + k;
		}
		q[cur] = p;
	}
	memset(memo,-1,sizeof(memo));
	int in = 1;
	while(in<=s.size()){
		//cout << in << "\n";
		for(int i = in; i <= n; i++){
			ll ch = h[i] - h[in-1]*M[i-in+1];
			if(q[ch]!=""&&solve(i+1)==1){
				ans += q[ch] + " ";
				in = i+1;
				break;
			}
		}
	}
	cout << ans << "\n";
}
