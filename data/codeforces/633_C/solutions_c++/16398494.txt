#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long LL;
const int maxn = 10010;
const int maxm = 2000010;

char s[maxn], str[maxm];
char *t[maxm];
int gao[maxn], n, m, len[maxm];
map<LL, int> H;

int main()
{
	while(cin >> n)
	{
		scanf("%s", s);
		scanf("%d", &m);
		H.clear();
		t[1] = str;
		for(int i = 1; i <= m; i++) {
			scanf("%s", t[i]);
			LL h = 0;
			len[i] = strlen(t[i]);
			reverse(t[i], t[i]+len[i]);
			for(int j = 0; j < len[i]; j++)
				h = h*123 + tolower(t[i][j]);
			H[h] = i;
			t[i+1] = t[i] + len[i] + 1;
		}
		memset(gao, 0, sizeof gao);
		gao[n] = 1;
		for(int i = n-1; i >= 0; i--) {
			LL h = 0;
			for(int j = i; j < n; j++) {
				h = h*123 + s[j];
				if(gao[j+1] && H.find(h) != H.end()) {
					gao[i] = H[h];
					break;
				}
			}
		}
		for(int i = 1; i <= m; i++) {
			reverse(t[i], t[i]+len[i]);
		}
		for(int i = 0; i < n;) {
			printf("%s", t[gao[i]]);
			i += len[gao[i]];
			printf("%c", i==n ? '\n':' ');
		}
	}
	return 0;
}
