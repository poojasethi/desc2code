#include<bits/stdc++.h>
#define N 100005
#define M 1005
using namespace std;
char a[N],b[N][M];
int id[1000005],son[1000005][26],now[N],pre[N],f[N];
int n,m,i,tot,k,p,c;
void Insert(int &p,int ch){
	int &v=son[p][ch];
	if (!v) v=++tot;p=v;
}
void Print(int k){
	if (pre[k]) Print(pre[k]);
	printf("%s",b[now[k]]);
	if (k<n) putchar(' ');
}
int main(){
	scanf("%d",&n);
  scanf("%s",a+1);
  scanf("%d",&m);tot=1;
  for (i=1;i<=m;i++){
  	scanf("%s",b[i]);p=1;
  	for (k=0;b[i][k];k++)
  		Insert(p,(b[i][k]<'a')?b[i][k]-'A':b[i][k]-'a');
  	id[p]=i;
  }f[0]=1;
  for (i=1;i<=n;i++){
  	p=1;
  	for (k=i,c=0;k>0&&c<=1000;k--,c++){
  		p=son[p][a[k]-'a'];
  		if (!p) break;
  		if (f[k-1]&&id[p]){
  			f[i]=1,pre[i]=k-1;
				now[i]=id[p];break;
		  }
    }
  }Print(n);
}
