#include<bits/stdc++.h>

using namespace std;
const int base =143;
#define MAX 10006
map<long long,string>m;
int id[MAX];
long long f[MAX];

void print(int num){
    if(num==0)return;
    print(id[num]);
    string ans=m[f[num]];
    cout<<ans<<" ";
}
int main(void)
{
    int num,n;
    string str;
    cin>>num>>str>>n;
    for(int i=0;i<n;i++){
        string mc;
        cin>>mc;
        long long hashval=0;
        for(int j=0;mc[j];j++){
            hashval=hashval*base+tolower(mc[j]);
        }
        m[hashval]=mc;
    }
    memset(f,-1,sizeof(f));
    f[0]=0;
    for(int i=0;i<num;i++){
        long long hashval=0;
        for(int j=i;j>=0;j--){
            hashval=hashval*base +str[j];
            if(f[j]!=-1 && m.count(hashval)){
                f[i+1]=hashval;
                id[i+1]=j;
                break;
            }
        }
    }
    print(num);
    return 0;
}
