#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
#include <string>
#define pi acos(-1)
#define MAX 26

using namespace std;

const int M = 100010;
const int N = 1010;
const int O = 10010;
char s[O];
char str[M][N];
int len[M];
int trie[1000010][MAX], mk[1000010];
int ans[O], tot, cnt, m, n;
void Build(int x)
{
	int now = 0;
	for (int i = len[x] - 1; i >= 0; i--)
	{
		char ch = str[x][i];
		if (ch < 'a' || ch > 'z')
			ch += 'a' - 'A';
		int p = ch - 'a';
		if (!trie[now][p])
			trie[now][p] = ++cnt;
		now = trie[now][p];
	}
	mk[now] = x;
}
void solve(int pos)
{
	if (pos == n)
	{
		for (int i = 1; i < tot; i++)
			printf("%s ", str[ans[i]]);
		printf("%s\n", str[ans[tot]]);
		exit(0);
	}
	int now = 0;
	for (int i = pos + 1; i <= n; i++)
	{
		int p = s[i] - 'a';
		now = trie[now][p];
		if (!now)
			break;
		if (mk[now])
		{
			ans[++tot] = mk[now];
			solve(pos + len[mk[now]]);
			--tot;
		}
	}
}
int main()
{
	scanf("%d%s", &n, s + 1);
	scanf("%d", &m);
	for (int i = 1; i <= m; i++)
	{
		scanf("%s", str[i]);
		len[i] = strlen(str[i]);
		Build(i);
	}
	solve(0);
	return 0;
}
