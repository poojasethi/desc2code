#include<bits/stdc++.h>
#define  mx           100010
#define  base         10000007
#define  clr(a)       memset(a, -1, sizeof(a))
using namespace std;

typedef long long int ll;
map<ll, string> mp;
ll dp[mx];
int sz;

void call(int i)
{
    if(i<=0) return;
    string str = mp[dp[i]];
    int len = str.size();
    call(i - len);
    cout<<str<<" ";
    //if(i != sz)cout<<" ";
    return;
}

int main()
{
    int cas, t, n, m;
    string s, a;
    scanf("%d", &n);
    cin>>s;
    cin>>m;
    mp.clear();
    clr(dp);
    for(int j=0; j<m; j++)
    {
        cin>>a;
        int len = a.size();
        ll h = 0;
        for(int i=0; i<len; i++)
        {
            h = h*base + tolower(a[i]);
        }
        mp[h] = a;
    }
    int len = s.size();
    dp[0] = 0;
    for(int i=0; i<len; i++)
    {
        ll h = 0;
        for(int j=i; j>=0; j--)
        {
            h = h*base + s[j];
            if(dp[j] != -1 && mp.count(h))
            {
                dp[i+1] = h;
                break;
            }
        }
    }
    sz = s.size();
    call(sz);
    cout<<endl;
    return 0;
}
