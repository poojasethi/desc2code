#include <bits/stdc++.h>

using namespace std;

typedef unsigned long long ull;

const int MAX_N = 1000000 + 10;
const ull seeds = 233;

string s;
map<ull, string> rec;
int n, m;
ull vis[MAX_N];

void output(int x) {
	if(x == 0)
		return ;
	string fafafa = rec[vis[x]];
	output(x - fafafa.size());
	cout << fafafa << " ";
}

int main() {
	cin >> n >> s >> m;
	for(int i = 0; i < m; i++) {
		string x; cin >> x;
		ull h = 0;
		for(int j = 0; j < x.size(); j++)
			h = h * seeds + tolower(x[j]);
		rec[h] = x;
	}
	for(int i = 0; i < n; i++)
		vis[i] = -1;
	vis[0] = 0;
	for(int i = 0; i < n; i++) {
		ull h = 0;
		for(int j = i; j >= 0; j--) {
			h = h * seeds + s[j];
			if(vis[j] != -1 && rec.count(h)) {
				vis[i + 1] = h;
				break;
			}
		}
	}
	output(n);
	return 0;
}
