#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <vector>
#include <map>
using namespace std;
typedef long long ll;
const int N=1E6+5;
const int MOD=2E7+117;
struct data{
	int x,p;
}f[N];
int n,m,len,u,tot,h[N],x,c[N][27],sum=1;
char s[N],st[N],S[N],ch;
int o[N];

void dfs(int x)
{
	if (!x) return;
	dfs(f[x].x);
	int p=f[x].p;
	for (int i=h[p];i<h[p+1];i++) printf("%c",S[i]);
	printf(" ");
}	

int main()
{
	cin>>n;
	scanf("%s",s);
	cin>>m;
	for (int i=1;i<=m;i++)
	{
		scanf("%s",st);
		len=strlen(st); u=1; h[i]=tot+1;
		for (int j=0;j<len;j++)
		{
			ch=st[j]; S[++tot]=ch;
			if (ch<'a'||ch>'z') x=ch-'A'+1; else x=ch-'a'+1;
			if (!c[u][x]) c[u][x]=++sum;
			u=c[u][x];
		}
		o[u]=i;
	}
	h[m+1]=tot+1;
	for (int i=0;i<n;i++)
	{
		u=1;
		for (int j=i;j>=0;j--)
		{
			u=c[u][s[j]-'a'+1];
			if (!u) break;
			if (o[u]&&(!j||f[j].p)) 
			{
				f[i+1].p=o[u]; f[i+1].x=j; break;
			}
		}
	}
	dfs(n);
}
