#include<cstdio>
#include<cstring>
#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
#define MAXN 1111111
int tn,ch[MAXN][26],flag[MAXN];
void insert(const char *s,int k){
    int x=0,y;
    for(int i=0; s[i]; ++i){
        if(s[i]>='A' && s[i]<='Z') y=s[i]-'A';
        else y=s[i]-'a';
        if(ch[x][y]==0) ch[x][y]=++tn;
        x=ch[x][y];
    }
    flag[x]=k;
}

int d[11111];
char str[11111];
string word[110000];
void output(int n){
    if(n<0) return;
    output(n-word[d[n]].length());
    cout<<word[d[n]]<<' ';
}
int main(){
    int n,m;
    scanf("%d%s%d",&n,str,&m);
    for(int i=1; i<=m; ++i){
        cin>>word[i];
        insert(word[i].c_str(),i);
    }
    for(int i=0; i<n; ++i){
        int x=0;
        for(int j=i; j>=0; --j){
            int y=str[j]-'a';
            if(ch[x][y]==0) break;
            x=ch[x][y];
            if(flag[x] && (i-(i-j+1)<0 || d[i-(i-j+1)])){
                d[i]=flag[x];
                break;
            }
        }
    }
    output(n-1);
    return 0;
}
