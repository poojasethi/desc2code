#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

vector<string> s[30];
string aa;
int n , m ;
string text;
vector<string> v;
bool vis[2000200];
bool sol=0;

bool comp(int i,string a){
int l=a.length();
if( i+l>n )return 0;

for(int j=i,d=0;j<i+l;j++,d++)
if( text[j] != tolower( a[d] ) )
return 0;

return 1;
}

void f(int idx){
if( vis[idx] )return;
vis[idx]=1;
if(idx==n){
sol=1;
for(int i=0;i<v.size();i++){
if(i)cout<<" ";
string w=v[i];
reverse(w.begin(),w.end() );
cout<< w ;
}
cout<<endl;
return;
}

for(int i=0;!sol&&i<s [ text[idx]-'a' ].size();i++  ){

 if( comp( idx , s[ text[idx]-'a' ][i]  ) ){
        v.push_back( s[text[idx]-'a'][i] );
       f( idx+s[ text[idx]-'a' ][i].length() );
 v.pop_back();
 }

}


}

int main()
{
cin>>n;
cin>>text;
cin>>m;

for(int i=0;i<m;i++){
    cin>>aa;
reverse(aa.begin(),aa.end());

if( aa[0]<'a' )
s[ aa[0]-'A' ].push_back(aa);
else
s[ aa[0]-'a' ].push_back(aa);

}

f(0);


    return 0;
}
