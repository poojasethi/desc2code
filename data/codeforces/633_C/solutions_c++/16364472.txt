#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
using namespace std;

int n, s, m, v[10005];
char a[10005], b[100005][1005];
vector<int> ans;
int abs(int p) {return p > 0 ? p : -p;}
void f(int p) {
	int i, j, sz, t;
	if (v[p] == -1) {
		for (i = 1; i <= m; i++) {
			sz = strlen(b[i]);
			if (p - sz + 1 < 0) continue;
			for (j = 0; j < sz; j++) {
 				t = abs(a[p - j] - b[i][j]);
				if (t != 32 && t != 0) break;
			}
			if (j == sz) {
				if (p - sz >= 0) f(p - sz);
				if (p - sz == -1 || v[p - sz] > 0) {
					v[p] = i;
					break;
				}
			}
		}
		if (v[p] == -1) v[p] = 0;
	}
}

int main()
{
	int i;
//	freopen ("input.txt", "r", stdin);
	cin >> n;
	scanf ("%s", a);
	cin >> m;
	for (i = 1; i <= m; i++) {
		scanf ("%s", b[i]);
	}
	for (i = 0; i < n; i++) {
		v[i] = -1;
	}
	f(n - 1);
	for (i = n - 1; i >= 0;) {
		ans.push_back(v[i]);
		i -= strlen(b[v[i]]);
	}
	for (i = ans.size() - 1; i >= 0; i--) {
		printf("%s ", b[ans[i]]);
	}
    return 0;
}
