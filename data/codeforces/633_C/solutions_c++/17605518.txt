#include<bits/stdc++.h>
using namespace std;
vector<string> vv;
string s,st[1000001],suit[1000001];
map<string,int> mp2;
map<string,string> mp;
int len,num,dp[1000001];
char c;
void dfs(int place)
{
    string a;
    if(dp[place]) return ;
    dp[place]=1;
    if(place==len)
    {
        int k=(int)vv.size();
        for(int i=0;i<k;i++)
            cout<<vv[i]<<" ";
        exit(0);
    }
    for(int i=place;i<len;i++)
    {
        a.push_back(s[i]);
        string l=a;
        reverse(l.begin(),l.end());
        if(mp2[l])
        {
            vv.push_back(mp[l]);
            dfs(i+1);
            vv.pop_back();
        }
    }
}
int main()
{
    ios::sync_with_stdio(false);
    cin>>len;
    for(int i=1;i<=len;i++)
    {
        cin>>c;
        s.push_back(c);
    }
    cin>>num;
    for(int i=1;i<=num;i++)
    {
        cin>>st[i];
        suit[i]=st[i];
        for(int j=0;j<(int)st[i].length();j++)
        if(suit[i][j]>='A' && suit[i][j]<='Z')
        suit[i][j]+='a'-'A';
        mp[suit[i]]=st[i];
        mp2[suit[i]]=1;
    }
    dfs(0);


}
