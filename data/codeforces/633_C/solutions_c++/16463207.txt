#include<bits/stdc++.h>
using namespace std;
#define maxn 10005
#define maxnd 1000005
char s[maxn],haha[100005][1005];
int n,m,nd,trie[maxnd][26],mrk[maxnd],len[100005],sta[maxn],top;
void insert(int x){
	int p=0;
	for(int i=len[x];i;i--){
		char tmp=haha[x][i];
		if(tmp>='A'&&tmp<='Z')tmp=tmp-'A'+'a';
		if(!trie[p][tmp-'a'])trie[p][tmp-'a']=++nd;
		p=trie[p][tmp-'a'];
	}
	mrk[p]=x;
}
void dfs(int dep){
	if(dep==n){
		for(int i=1;i<=top;i++){
			printf("%s",haha[sta[i]]+1);
			printf(i==top?"\n":" ");
		}
		exit(0);
	}
	int p=0;
	for(int i=dep+1;i<=n;i++){
		if(!trie[p][s[i]-'a'])break;
		p=trie[p][s[i]-'a'];
		if(mrk[p]){
			sta[++top]=mrk[p],dfs(dep+len[mrk[p]]);
			top--;
		}
	}
}
int main(){
	scanf("%d%s",&n,s+1);
	scanf("%d",&m);
	for(int i=1;i<=m;i++){
		scanf("%s",haha[i]+1);
		len[i]=strlen(haha[i]+1);
		insert(i);
	}
	dfs(0);
	return 0;
}
