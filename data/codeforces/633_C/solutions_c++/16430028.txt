#include<bits/stdc++.h>
using namespace std;
int n,pd,SIZE;
char w[100010][1010],t[1010],s[10010];
int top,trie[1000010][31],vis[1000010],cc[100010];
void Insert(char t[],int ii)
{
    int now=0,lt=strlen(t+1),k,i;
    for(i=lt;i>=1;i--)
    {
        if(t[i]>='a'&&t[i]<='z')k=t[i]-'a'+1;
        else k=t[i]-'A'+1;
        if(trie[now][k]==0)trie[now][k]=++SIZE;
        now=trie[now][k];
    }
    vis[now]=ii;
}
void DFS(int deep)
{
    int i,now;
    if(pd==1)return;
    if(deep==n)
    {
        for(i=1;i<=top;i++)printf("%s ",w[cc[i]]+1);
        pd=1;
        exit(0);
        //return;
    }
    now=0;
    for(i=deep+1;i<=n;i++)
    {
        if(trie[now][s[i]-'a'+1]==0)break;
        now=trie[now][s[i]-'a'+1];
        if(vis[now]!=0)
        {
            cc[++top]=vis[now];
            DFS(deep+strlen(w[vis[now]]+1));
            top--;
        }
    }
}
int main()
{
    int m,i;
    scanf("%d",&n);
    scanf("\n%s",s+1);
    scanf("%d",&m);
    for(i=1;i<=m;i++)
    {
        scanf("\n%s",w[i]+1);
        Insert(w[i],i);
    }
    pd=0;
    DFS(0);
    return 0;
}
