#include<bits/stdc++.h>
#include<cstring>
using namespace std;

int n,m;
string s;
map<long long,string>MAP;
long long dp[1000005];

void print(int pos)
{
    if(pos<=0) return ;
    string s=MAP[dp[pos]];
    print(pos-s.size());
    cout<<s<<" ";
}

int main()
{
    cin>>n>>s>>m;
    for(int i=0;i<m;i++)
    {
        string temp;
        cin>>temp;
        int l=temp.size();
        long long ago=0;
        for(int j=0;j<l;j++)
            ago=ago*27+tolower(temp[j])-'a'+1;
        MAP[ago]=temp;
    }
    memset(dp,-1,sizeof(dp));
    dp[0]=0;
    int l=s.size();
    for(int i=0;i<l;i++)
    {
        long long temp=0;
        for(int j=i;j>=0;j--)
        {
            temp=temp*27+s[j]-'a'+1;
            if(dp[j]!=-1 && MAP.count(temp))
            {
                //if(MAP.count(temp)) cout<<MAP[temp]<<endl;
                //cout<<"                   "<<temp<<" "<<MAP[temp]<<endl;
                dp[i+1]=temp;
                break;
            }
        }
    }
    print(l);
    return 0;
}
