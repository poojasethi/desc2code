#include <iostream>
#include <vector>
#include <set>
#include <map>
#include <queue>
#include <algorithm>
#include <cmath>
#include <iomanip>

//#define int long long

using namespace std;

int n;
string t;
int m;
set <string> w;
set <string> :: iterator par[10 * 1000 + 100];
bool b[10 * 1000 + 100];
map <string, string> f;
vector <string> ans;

main()
{
	ios_base::sync_with_stdio(false);
	
	cin >> n;
	cin >> t;
	
	cin >> m;
	for (int i = 0; i < m; i++)
	{
		string word;
		cin >> word;
		string tmp = word;
		for (int j = 0; j < tmp.size(); j++)
		{
			if ('A' <= tmp[j]  &&  tmp[j] <= 'Z')
				tmp[j] += 'a' - 'A';
		}
		w.insert(tmp);
		f[tmp] = word;
	}

	for (int i = 0; i < n; i++)
	{
		string s = "";
		for (int j = i; j >= 0; j--)
		{
			s += t[j];
			if (j == 0  ||  b[j - 1] == true)
			{
				par[i] = w.find(s);
				if (par[i] != w.end())
				{
					b[i] = true;
					break;
				}
			}
		}
	}
	
	int i = n - 1;
	while (i >= 0)
	{
		ans.push_back(f[*par[i]]);
		i -= (*par[i]).size();
	}
	
	for (int i = ans.size() - 1; i >= 0; i--)
		cout << ans[i] << " ";
	cout << endl;
	
	return 0;
}
