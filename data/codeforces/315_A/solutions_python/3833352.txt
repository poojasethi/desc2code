n = int(raw_input())
bot = []
can_open = []
for i in xrange(n):
    a, b = map(int, raw_input().split())
    bot.append(a)
    can_open.append(b)
for i in xrange(n):
    op = can_open[i]
    for j in xrange(n):
        if bot[j] == op and j != i:
            bot[j] = -1
count = 0
#print bot
for i in bot:
    if i != -1:
        count += 1
print count
