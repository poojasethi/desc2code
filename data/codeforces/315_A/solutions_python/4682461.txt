N=input()
o=set()
n=set()
c=[0]*1001
C=[0]*1001
for e in [0]*N:
    a,b=map(int,raw_input().split())
    if a == b:
        o.add(b)
        C[b]+=1
    else:
        n.add(b)
    c[a]+=1
r=0
for i,v in enumerate(c):
    if not v: continue
    if (v==1 or C[i]==1) and i not in n:
        r+=1
    elif i not in n and i not in o:
        r+=v
print r
