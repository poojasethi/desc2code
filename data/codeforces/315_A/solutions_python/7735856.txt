#!/usr/bin/env python
n=input()
l=[]
count=n
may_be_opened=[]
for i in range(n):
    l.append(map(int,raw_input().split(' ')))
    may_be_opened.append(l[i][1])
for i in range(n):
    if l[i][0]!=l[i][1] and l[i][0] in may_be_opened:
        count-=1
    elif l[i][0]==l[i][1] and may_be_opened.count(l[i][1])>=2:
        count-=1
    else:
        continue

print count
