from collections import defaultdict

ri = lambda: map(int, raw_input().split())

i = input()
bs = [None]*i
ts = defaultdict(list)

for j in xrange(i):
    a,b = ri()
    bs[j] = a
    ts[b].append(j)
    
t = 0
for k, kp in enumerate(bs):
    if len(ts[kp]) == 0 or ts[kp] == [k]:
        t += 1

print t
