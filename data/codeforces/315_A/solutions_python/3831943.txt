rd = lambda : map(int, raw_input().split())
n = input()
a, b = [0] * n, [0] * n
for i in range(n):
  a[i], b[i] = rd()
res = 0
for i in range(n):
  opened = False
  for j in range(n):
    if i == j: continue
    if a[i] == b[j]:
      opened = True
      break
  if not opened:
    res += 1
print res
