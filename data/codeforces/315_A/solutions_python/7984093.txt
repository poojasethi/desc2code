n = int(raw_input())
ai = list()
bi = list()
for i in xrange(0, n):
    a, b = raw_input().split()
    ai.append(int(a))
    bi.append(int(b))

res = [False]*n
for i in xrange(0, len(ai)):
    for j in xrange(0, len(bi)):
        if i != j and ai[i] == bi[j] and res[i] is False:
            n -= 1
            res[i] = True
print n
