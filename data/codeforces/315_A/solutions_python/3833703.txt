n = int(input());
x = [0]*n;
y = [0]*n;
bb = [0]*n;
for i in range(n):
    a,b = map(int, raw_input().split());
    #print a, b;
    x[i] = a;
    y[i] = b;

for i in range(n):
    for j in range(n):
        if i != j:
            if y[i] == x[j]:
                bb[j] = 1;
ans = 0;
for i in range(n):
    if bb[i] == 0:
        ans += 1;
print ans;
