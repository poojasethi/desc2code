import sys
#br = open('a.in')
br = sys.stdin
n = int(br.readline())
v = [0 for i in range(n)]
o = [0 for i in range(n)]
f = [1 for i in range(n)]
for i in range(n):
  a, b = map(int, br.readline().split())
  v[i] = a
  o[i] = b
for i in range(n):
  for j in range(n):
    if o[i] == v[j] and i != j:
      f[j] = 0
print sum(f)
