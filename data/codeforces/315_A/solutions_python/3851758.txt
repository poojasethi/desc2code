n = input()
a, b, c = ([0] * n, [0] * n, [False] * n)
for i in range(n):
  a[i], b[i] = map(int, raw_input().split())

s = 0

while True:
  before = s
  for i in range(n):
    for j in range(n):
      if i != j and b[i] == a[j]:
        c[j] = True
  after = sum(1 for x in c if x)
  if before == after:
    break
  s = after

print n - s
