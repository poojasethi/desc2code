n = input()
ls = [0 for i in xrange(n)]
a = []
b = []
for i in xrange(n):
    ai, bi= map(int, raw_input().split())
    a.append(ai)
    b.append(bi)
for i in xrange(n):
    for j in xrange(n):
        if i == j:
            continue
        else:
            if b[i] == a[j]:
                ls[j] = 1
print n-sum(ls)
