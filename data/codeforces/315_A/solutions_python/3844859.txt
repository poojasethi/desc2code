import sys

n = int(sys.stdin.readline())

to_open = []
can_open = []

count = n

for i in xrange(n):
    a, b = map(int, sys.stdin.readline().split(" "))
    can_open.append(b)
    to_open.append(a)

for i in xrange(n):
    for j in xrange(n):
        if to_open[i] == can_open[j] and i != j:
            count = count - 1
            break

print count
