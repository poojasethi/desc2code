n = int(raw_input())
a = []
b = []
status = []
for idx in range(n):
    x, y = map(int, raw_input().split(" "))
    a.append(x-1)
    b.append(y-1)
    status.append(0)
cnt = n
for source in range(n):
    for target in range(n):
        if source == target:
            continue
        if b[source] == a[target] and status[target] == 0:
            cnt-=1
            status[target] = 1
print cnt
