n=int(raw_input())
ans=0
a,b,chk=[],[],[]
for i in xrange(n):
    x,y=map(int,raw_input().split())
    a.append(x)
    b.append(y)
for j in xrange(n):
    if b[j] in a:
        if (a.index(b[j])==j and a.count(b[j])==1) or b[j] in chk:
            pass
        else:
            ans+=a.count(b[j])
            if b.count(b[j])==1 and a[j]==b[j]:
                    ans-=1
            chk.append(b[j])
print n-ans
