n = int(raw_input())
a = []
b = []
for i in xrange(0, n):
	tmp = raw_input().split()
	a.append(int(tmp[0]))
	b.append(int(tmp[1]))
res = 0

for i in xrange(0, n):
	openable = False
	for j in xrange(0, n):
		if (a[i] == b[j]) and not ( i == j):
			openable = True
	if not openable:
		res = res + 1

print res
