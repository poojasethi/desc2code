n = input()

bottles = {}
openables = {}
dupes = {}
for i in range(n):
  a,b = map(int, raw_input().split())
  if a not in bottles:
    bottles[a] = 1
  else:
    bottles[a] += 1

  if b not in openables:
    openables[b] = 1
  else:
    openables[b] += 1

  if a == b:
    if a not in dupes:
      dupes[a] = 1
    else:
      dupes[a] += 1 

print sum([bottles[bot]*(bot not in dupes) + (bot in dupes) for bot in bottles if (bot not in openables or (bot in dupes and openables[bot] < 2))])
