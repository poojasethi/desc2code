n = int(raw_input())
lst = []
for i in range(n):
    row = raw_input().split(" ")
    lst.append(row)
count = 0
for i in range(n):
    flag = 1
    for j in range(n):
        if i != j and int(lst[j][1]) == int(lst[i][0]):
            flag = 0
            break
    if flag == 1:
        count += 1
print count
