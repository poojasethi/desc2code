#include <stdio.h>
#include <string.h>
int main(){
	int n,i,j,h[101],g[101],ans=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		scanf("%d%d",&h[i],&g[i]);
	}
	for(i=1;i<=n;i++){
		for(j=1;j<=n;j++){
			if(i!=j&&h[i]==g[j]){ans++;break;}
		}
	}
	printf("%d\n",n-ans);
	return 0;
}
