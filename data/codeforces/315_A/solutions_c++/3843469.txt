#include<cstdio>
#define N 100
int a[N],b[N],o[N];
int n;

int main(){
	scanf("%d",&n);
	for(int i=0;i<n;++i){
		scanf("%d%d",a+i,b+i);
	}	
	for(int i=0;i<n;++i)
		for(int j=0;j<n;++j) 
			if(i!=j && a[j]==b[i])
				o[j]=1;

	int c=0;
	for(int i=0;i<N;++i)
		c+=o[i];
	printf("%d",n-c);
	return 0;
}
