#include<stdio.h>
int ch[105];
int main()
{
	int a,b,n,p[105],q[105];
	scanf("%d",&n);
	for(a=0;a<n;a++)
		scanf("%d%d",&p[a],&q[a]);
	for(a=0;a<n;a++)
		for(b=0;b<n;b++)
			if(b!=a && q[a]==p[b])
				ch[b]=1;
	for(a=0,b=0;a<n;a++)
		if(!ch[a])	b++;
	printf("%d\n",b);
	return 0;
}
