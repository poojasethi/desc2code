#include<stdio.h>
int main()
{
	int n,i,j,k=0;
	int A[100][2];
	scanf("%d",&n);
    for(i=0;i<n;i++)
		scanf("%d %d",&A[i][0],&A[i][1]);
    for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			if(A[i][0]==A[j][1]&&i!=j)
				break;
		}
		if(j==n)
			k++;
	}
	printf("%d",k);
	return 0;
}
