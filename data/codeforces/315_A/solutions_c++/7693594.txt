#include <iostream>

using namespace std;

int n;
int a[1000],b[1000],c[1000];

int main()
{
  cin>>n;
  for (int i=0;i<n;i++)
    cin>>a[i]>>b[i];
  for (int i=0;i<n;i++)
    for (int j=0;j<n;j++)
      if (i!=j and a[j]==b[i])
	c[j]=1;
  int s=0;
  for (int i=0;i<n;i++)
    s+=c[i]==0;
  cout<<s<<endl;
}
