# include<iostream>
using namespace std;
int main()
{
	int n,a[1000],b[1000],num=0,i,j;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i]>>b[i];
	}
	for( i=0;i<n;i++)
		for(j=0;j<n;j++)
		{
			if(i!=j&&a[i]==b[j])
			{
				num++;
				break;
			}
		}
		cout<<n-num<<endl;
  return 0;
}
