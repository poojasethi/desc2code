#include <iostream>

using namespace std;

int main(){
	int n,m,x,c=0;
	cin>>n;
	bool T[n]={0};
	cin>>m;
	while(m--){
		cin>>x;
		if(!T[x-1]){
			T[x-1]=true;
			c++;
		}
	}
	cin>>m;
	while(m--){
		cin>>x;
		if(!T[x-1]){
			T[x-1]=true;
			c++;
		}
	}
	cout<<(c==n?"I become the guy.":"Oh, my keyboard!");
	return 0;
}
