#include <bits/stdc++.h>
using namespace std;
int n, a, b, cnt = 0;
bool lol[105];
int main(){
	cin >> n;
	for(int z = 0; z < 2; z++){
		cin >> a;
		while(a--){
			cin >> b;
			if(!lol[b]) cnt++;
			lol[b] = 1;
		}
	}
	if(cnt == n) puts("I become the guy.");
	else puts("Oh, my keyboard!");
	return 0;
}
