#include<iostream>
#include<map>

using namespace std;

map <string,int> m;
int n,imax=1;
char who[250],reposted[10],whom[250];

int main()
{
	ios_base::sync_with_stdio(0);
	
	cin>>n;
	
	m["polycarp"]=1;
	
	for(int i=0;i<n;i++)
	{
		cin>>who>>reposted>>whom;
		for(int i=0;who[i] ;i++) who[i] =tolower(who[i] );
		for(int i=0;whom[i];i++) whom[i]=tolower(whom[i]);
		
		m[who]=m[whom]+1;
		if(m[who]>imax) imax=m[who];
	}
	
	cout<<imax;
	
	return 0;
}
