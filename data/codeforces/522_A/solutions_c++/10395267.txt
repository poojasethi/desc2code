#include<iostream>
#include<cstdio>
#include<cstring>
#include<map>
using namespace std;

int main()
{
    map<string,int>A;
    string a, b, c;
    int i, j, n, result=1;
    cin>>n;
    A["polycarp"]=1;

    for(i=0;i<n;i++)
    {
        cin>>a>>b>>c;

        for(j=0;j<a.length();j++)a[j]=tolower(a[j]);

        for(j=0;j<c.length();j++)c[j]=tolower(c[j]);

        A[a]=A[c]+1;
        if(result<A[a])result=A[a];
    }

    cout<<result<<endl;
    return 0;
}
