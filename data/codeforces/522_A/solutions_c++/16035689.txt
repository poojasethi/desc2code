#include<bits/stdc++.h>
using namespace std;
map<string,int> mp;
int mx;
int main(){
	int n; cin >> n;
	for(int i = 0 ; i < n ; i++){
		string n1 , s , n2; cin >> n1 >> s >> n2;
		transform(n1.begin(),n1.end(),n1.begin(),::tolower);
		transform(n2.begin(),n2.end(),n2.begin(),::tolower);
		mp[n1] = mp[n2]+1;
		mx = max(mx,mp[n1]);
	}
	cout << mx+1 << endl;
}
