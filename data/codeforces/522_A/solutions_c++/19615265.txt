#include<bits/stdc++.h>
using namespace std;
map<string,int> mp;
int main(){
	string a,b;
	int n ,answer=0;
	cin>>n;
	while(n--){
		cin>>a>>b>>b;
		transform(a.begin(),a.end(),a.begin(), :: tolower);
		transform(b.begin(),b.end(),b.begin(), :: tolower);
		mp[a]=mp[b]+1;
		answer=max(answer,mp[a]);
	}
	cout<<answer+1<<endl;
}
