#include<bits/stdc++.h>

using namespace std;

string s1, s2, s3;
map<string, int> s;
int n, res = 0;

void chuanhoa()
{
	for (int j=0;j<s1.length();++j)
		s1[j]=tolower(s1[j]);
	for (int j=0;j<s2.length();++j)
		s2[j]=tolower(s2[j]);
}

int main()
{
  	cin>>n;
  	s["polycarp"] = 1;
  	while(n--)
  	{
  		cin>>s1>>s3>>s2;
  		chuanhoa();
		s[s1]=s[s2]+1;
		res=max(res,s[s1]);
	}
	cout<<res;
}
