#include <iostream>
#include <map>
#include <string>
using namespace std;
int main(){
	int n,e,max=1;
	map<string, int> l;
	string a,b,c;
	l["polycarp"]=1;
	cin >> n;
 	while(n>0){
 		cin >> a >> b >> c;
 		for(int i=0;i<a.length();i++)
 			a[i]=tolower(a[i]);
 		for (int i = 0; i < c.length(); ++i)
 		{
 			c[i]=tolower(c[i]);
 		}
 		e=l[c]+1;
 		if(e>max)
 			max=e;
 		l[a]=e;
 		n--;
 	}
	cout << max;

	return 0;

}
