#include "bits/stdc++.h"
using namespace std;
map<string, int> m;
int main() {
	int i, n, an = 1;
	string a, b;
	cin >> n;
	for (i = 0; i < n; ++i) {
		cin >> a >> b >> b;
		transform(a.begin(), a.end(), a.begin(), ::tolower);
		transform(b.begin(), b.end(), b.begin(), ::tolower);
		m[a] = m[b] + 1;
		an = max(an, m[a]);
	}
	cout << an + 1 << endl;
    return 0;
}
