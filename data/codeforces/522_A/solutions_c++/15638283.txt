#include "bits/stdc++.h"
using namespace std;
map<string, int> mp;
int main() {
	int i, n, ans = 1;
	string a, b;
	cin >> n;
	for (i = 0; i < n; ++i) {
		cin >> a >> b >> b;
		transform(a.begin(), a.end(), a.begin(), ::tolower);
		transform(b.begin(), b.end(), b.begin(), ::tolower);
		mp[a] = mp[b] + 1;
		ans = max(ans, mp[a]);
	}
	cout << ans + 1 << endl;
    return 0;
}
