#include<map>
#include<iostream>
#include<ctype.h>
using namespace std;
int main()
{
    map<string,int> mp;
    string n1,rep,n2;
    int n,maxi=-1;cin>>n;
    mp["polycarp"]=1;
    while(n--)
    {
    cin>>n1>>rep>>n2;
    for(int i=0;i<n1.length();i++) n1[i]=tolower(n1[i]);
    for(int i=0;i<n2.length();i++) n2[i]=tolower(n2[i]);
    mp[n1]=mp[n2]+1;
    maxi=max(maxi,mp[n1]);
    }
    cout<<maxi;
    return 0;
}
