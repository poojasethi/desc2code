#include <iostream>
#include <map>
#include <algorithm>
#include <cctype>

using namespace std;

int n, mx;
string a, r, b;
map<string, int> repost;

int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);
	cin >> n;
	while (n--) {
		cin >> a >> r >> b;
		transform(a.begin(), a.end(), a.begin(), ::tolower);
		transform(b.begin(), b.end(), b.begin(), ::tolower);
		mx = max(mx, repost[a] = max(repost[a], repost[b] + 1));
	}
	cout << mx+1 << endl;
}
