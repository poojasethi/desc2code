#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,ans;
    ans=1;
    cin>>n;
    string a,b,c;
    map<string,int> m;
    m["polycarp"] = 1;
    while (n--)
    {
        cin>>a>>b>>c;
        for (int i=0;i<a.size();i++)
            a[i]=tolower(a[i]);
        for (int i=0;i<c.size();i++)
            c[i]=tolower(c[i]);
        int e=m[c]+1;
        m[a]=e;
        if (e>ans)
            ans=e;
    }
    cout<<ans<<endl;
    return 0;
}
