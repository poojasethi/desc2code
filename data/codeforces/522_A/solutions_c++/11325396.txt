#include<bits/stdc++.h>
#include<string>
#include<ctype.h>
#include<map>
using namespace std;
int main()
{
    int n,i,j;
    cin>>n;
    string n1,z,n2;
    map<string,int> mp;
    mp["polycarp"]=1;
    int max=0;
    for(i=0;i<n;i++)
    {
    cin>>n1>>z>>n2;
    for(j=0;j<n1.size();j++)
    n1[j]=tolower(n1[j]);
    for(j=0;j<n2.size();j++)
    n2[j]=tolower(n2[j]);
    mp[n1]=mp[n2]+1;
    if(max<mp[n1])
    max=mp[n1];
    }
    cout<<max;
    return 0;
}
