#include<iostream>
#include<map>
#include<algorithm>
#include<string>
using namespace std;

map<string, int> m;

int n, ans;
string a, b, s;

int main() {
	m["polycarp"] = 1;
	ans = 1;
	cin>>n;
	for(int i=0;i<n;i++) {
		cin>>a>>s>>b;
		for(int j=0;j<a.size();j++)
			a[j] = tolower(a[j]);
		for(int j=0;j<b.size();j++)
			b[j] = tolower(b[j]);
		int x = m[b] + 1;
		m[a] = x;
		ans = max(x, ans);
	}
	cout<<ans<<endl;
	
	return 0;
}
