#include <iostream>
#include <string>
#include <algorithm>
#include <map>
using namespace std;
int main()
{
	map<string,int> s;
	string n1,n2;
	int n,res=1;
	s["polycarp"]=1;
	cin>>n;
	for (int i=1;i<=n;++i){
		cin>>n1>>n2>>n2;
		for (int j=0;j<n1.length();++j)
			n1[j]=tolower(n1[j]);
		for (int j=0;j<n2.length();++j)
			n2[j]=tolower(n2[j]);
		s[n1]=s[n2]+1;
		res=max(res,s[n1]);
	}
	cout<<res;
	return 0;
}
