#include <vector>
#include <algorithm>
using namespace std;

#include <cstdio>

struct st {
	int a,b,c,i;
	st(){}
};

int n;
int f[5*100*100*10];
st t[5*100*100*10];

void change(int i, int x)
{
	for(;i<n;i=(i|(i+1)) )
		f[i]=max(f[i],x);
}

int fmax(int i)
{
	int r=0;
	for(;i>=0;i=(i&(i+1))-1 )
		r=max(f[i],r);
	return r;
}

bool cmpb(st a,st b)
{
	return a.b>b.b ? true : a.b==b.b ? a.c<b.c : false;
}

bool cmpa(st a,st b)
{
	return a.a>b.a ? true : a.a==b.a ? a.b<b.b : false; 
}

int getint()
{
    char c = getchar();
    for (;'0'>c || c>'9';c=getchar());

    int res=0;
    for (;'0'<=c && c<='9';c=getchar())
      res=res*10+int(c)- int('0');
    return res;
}

int main()
{
	n=getint();
	for(int i=0;i<n;i++) t[i].a=getint();
	for(int i=0;i<n;i++) t[i].b=getint();
	for(int i=0;i<n;i++) t[i].c=getint();

	sort(t,t+n,cmpb);
	for(int i=0;i<n;i++)
		t[i].i=i;
	sort(t,t+n,cmpa);

	int res=0;
	for(int i=0;i<n;i++)
	{
		if(fmax(t[i].i)>t[i].c)
			res++;
		change(t[i].i,t[i].c);
	}

	printf("%d",res);
	return 0;
}
