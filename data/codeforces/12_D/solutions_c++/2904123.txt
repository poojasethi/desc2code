#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
const int mm=501000;
const int oo=1e9+2;
class node
{
public:
  int x,y,z;
}l[mm];
int n;
bool cmp(node a,node b)
{
  return a.x!=b.x? a.x<b.x:a.y!=b.y?a.y>b.y:a.z<b.z;
}
map<int,int>mp;
map<int,int >::iterator it;
int main()
{
  while(cin>>n)
  {
    for(int i=0;i<n;i++)cin>>l[i].x;
    for(int i=0;i<n;i++)cin>>l[i].y;
    for(int i=0;i<n;i++)cin>>l[i].z;
    sort(l,l+n,cmp);

    mp[-oo]=oo;mp[oo]=-oo;
    //cout<<mp[-oo]<<" "<<mp[oo];
    int ans=0;
    for(int i=n-1;i>=0;i--)
    {
      it=mp.upper_bound(l[i].y);
      if(it->second>l[i].z)ans++;
      else if(mp[l[i].y]<l[i].z)
      {
        mp[l[i].y]=l[i].z;
        for(it=--mp.lower_bound(l[i].y);it->second<l[i].z;)
          mp.erase(it--);
      }
    }
    cout<<ans<<"\n";
  }
}
