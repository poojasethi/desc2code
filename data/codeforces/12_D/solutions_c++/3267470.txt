#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 1010000
struct node {
    int a, b, c;
    friend bool operator < (node a, node b)
    {
        return a.a!=b.a?a.a>b.a:a.b<b.b;
    }
} rec[N];
int h[N], n;

int main()
{
    scanf("%d",&n);
    for (int i=0; i<n; i++) scanf("%d",&rec[i].a);
    for (int i=0; i<n; i++) scanf("%d",&rec[i].b), h[i]=rec[i].b;
    for (int i=0; i<n; i++) scanf("%d",&rec[i].c);
    sort(rec,rec+n);
    sort(h,h+n);
    for (int i=0; i<n; i++) rec[i].b=n-(lower_bound(h,h+n,rec[i].b)-h);
    memset(h,128,sizeof h);
    int ans=0;
    for (int i=0; i<n; i++) 
    {
        int temp=-0x7fffffff;
        for (int j=rec[i].b-1;j;j-=j&-j) temp=max(temp,h[j]);
        if (temp>rec[i].c) ans++;
        for (int j=rec[i].b;j<=n;j+=j&-j) h[j]=max(h[j],rec[i].c);
    }
    printf("%d\n",ans);
    return 0;
}
