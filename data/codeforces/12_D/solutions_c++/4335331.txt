#include <bits/stdc++.h>
using namespace std;
typedef pair<int, int> pii;
const int maxn = 500005, oo = (int)2e+9;
struct point
{
	int x, y, z;
	friend bool operator<(const point &A, const point &B)
	{
		return A.x!=B.x?A.x>B.x:A.y!=B.y?A.y<B.y:A.z<B.z;
	}
}a[maxn];

int main(int argc, char const *argv[])
{
	// freopen("in", "r", stdin);
	int n;	scanf("%d", &n);
	for(int i=0; i<n; ++i)	scanf("%d", &a[i].x);
	for(int i=0; i<n; ++i)	scanf("%d", &a[i].y);		
	for(int i=0; i<n; ++i)	scanf("%d", &a[i].z);	
	sort(a, a+n);
	map<int, int> dic;		
	dic[oo] = -oo;	dic[-oo] = oo;
	int res = 0;	
	for(int i=0; i<n; ++i)
	{		
		map<int, int>::iterator f = dic.upper_bound(a[i].y);
		if(f==dic.end())	continue;
		if(f->second > a[i].z)	++res;
		else if(!dic.count(a[i].y) || dic[a[i].y]<a[i].z)
		{
			dic[a[i].y] = a[i].z;
			map<int, int>::iterator ff = dic.lower_bound(a[i].y);	--ff;			
			for(; ff->second < a[i].z; )	dic.erase(ff--);
		}
	}
	printf("%d\n", res);
	return 0;
}
