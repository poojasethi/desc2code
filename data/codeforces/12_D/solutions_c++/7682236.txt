#include <cstdio>
#include <algorithm>

using namespace std;

int n,i,j,L,cnt,ans,t[500005];

struct bg{int x,y,z;}a[500005];

inline bool cmpx(const bg &i,const bg &j){return i.x>j.x;}

inline bool cmpy(const bg &i,const bg &j){return i.y>j.y;}

inline void add(int x,int y){for(;x<=cnt;x+=x&-x)if(y>t[x])t[x]=y;else return;}

inline int Qs(int x){int ans=0;for(;x;x-=x&-x)if(t[x]>ans)ans=t[x];return ans;}

int main(){
    scanf("%d",&n);
    for(i=1;i<=n;++i)scanf("%d",&a[i].x);
    for(i=1;i<=n;++i)scanf("%d",&a[i].y);
    for(i=1;i<=n;++i)scanf("%d",&a[i].z);
    sort(a+1,a+n+1,cmpx);L=-55555;
    for(i=1;i<=n;++i)if(a[i].x==L)a[i].x=cnt;else L=a[i].x,a[i].x=++cnt;
    sort(a+1,a+n+1,cmpy);
    j=1;
    for(i=1;i<=n;++i){
        if(a[i].y^a[i-1].y)for(;j<i;++j)add(a[j].x,a[j].z);
        if(Qs(a[i].x-1)>a[i].z)++ans;
    }
    printf("%d\n",ans);
}
