#include <cstdio>

#define M 9999
#define N 100000

typedef long long ll;

ll a[M], x, k, m, i, n;

void go(ll x, ll k)
{
    if (n < N)
    {
        if (!k || x == 1)
            printf("%I64d ", x), ++n;
        else
            for (int i = 0; i < m && a[i] <= x; i++)
                if (x % a[i] == 0)
                    go(a[i], k - 1);
    }
}

int main()
{
    scanf("%I64d%I64d", &x, &k);

    for (i = 1; i * i <= x; i++)
        if (x % i == 0)
            a[m++] = i;

    for (i = m - (a[m - 1] * a[m - 1] == x ? 2 : 1); ~i; i--)
        a[m++] += x / a[i];

    go(x, k);

    return 0;
}
