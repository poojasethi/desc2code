#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
ll num[1000010];
int m=0,n=0;
void dfs(ll x,ll k)
{ if(n>=100000)
   return;
  if(k==0 || x==1)
  { printf("%I64d ",x);
    n++;
    return;
  }
  for(int i=1;i<=m && x>=num[i];i++)
    if(x%num[i]==0)
   dfs(num[i],k-1);
}
int main()
{ ll i,j,x,k;
  scanf("%I64d%I64d",&x,&k);
  for(i=1;i*i<x;i++)
   if(x%i==0)
   { num[++m]=i;
     num[++m]=x/i;
   }
  if(x/i==i)
   num[++m]=i;
  sort(num+1,num+1+m);
  dfs(x,k);
}
