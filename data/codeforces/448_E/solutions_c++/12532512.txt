#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
ll a1[10000];
int aa=100000;bool bl;
void f(int a,int b)
{
    if(!a||!b)
    {
        if(bl)putchar(' ');else bl=1;
        --aa;printf("%I64d",a1[a]);
        return;
    }
    for(int i=0;aa&&i<=a;++i)
    if(!(a1[a]%a1[i]))
        f(i,b-1);
}
int main()
{
    int n=0,k;ll a,b;
    scanf("%I64d%I64d",&a,&b);
    k=min((ll)aa,b);
    for(int i=1;(ll)i*i<=a;++i)
    if(!(a%i))a1[n++]=i;
    for(int i=n-1-(a1[n-1]*a1[n-1]==a);~i;--i)
        a1[n++]=a/a1[i];
    f(n-1,k);
}
