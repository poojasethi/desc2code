#include<cstring>
#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
ll x,k,d[1000010];int t,cnt;
void dfs(ll x,ll k) {
    if (t>=100000) return;
    if (k==0||x==1) {
        printf("%I64d ",x);
        ++t;return;
    }
    for (int i=1;i<=cnt&&d[i]<=x;++i) if (x%d[i]==0) dfs(d[i],k-1);
}
int main() {
    scanf("%I64d%I64d",&x,&k);
    ll i;
    for (i=1;i*i<x;++i) if (x%i==0) {
        d[++cnt]=i;
        d[++cnt]=x/i;   
    }
    if (i*i==x) d[++cnt]=i;sort(d+1,d+cnt+1);
    dfs(x,k);puts("");
    return 0;
}
