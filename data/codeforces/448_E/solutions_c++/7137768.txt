#include <algorithm>
#include<cstring>
#include<cstdlib>
#include<cstdio>
#define N 8500
#define inf 100000
#define rep(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,L;
long long n,m,a[N];
short b[N][N];
void dfs(int x,int cen) {if (!L) return;
	if (x==1||cen==m) {
		printf("%I64d ",a[x]);
		L--;
		return;
	}
	int i;
	for (i=1;i<=*b[x]&&L;i++) dfs(b[x][i],cen+1);
}
int main() {
	scanf("%I64d%I64d",&n,&m);
	for (i=1;1ll*i*i<=n;i++) if (n%i==0) a[++k]=i;
	for (;i;i--) if (n%i==0&&1ll*i*i<n) a[++k]=n/i;
	rep(i,1,k) {
		rep(j,1,i) if (a[i]%a[j]==0) b[i][++*b[i]]=j;
	}
	L=inf;
	dfs(k,0);
}
