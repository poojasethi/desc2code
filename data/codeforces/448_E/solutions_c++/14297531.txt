#include <bits/stdc++.h>
using namespace std;
int n,cnt;
long long d[100000],x,k;
void dfs(int x,long long k){
    if (k==0){
        cout<<d[x]<<" ";
        cnt++;
        if (cnt==100000) exit(0);
        return ;
    }
    if (d[x]==1){
        cout<<d[x]<<" ";
        cnt++;
        if (cnt==100000) exit(0);
        return ;
    }
    for (int i=1; i<=x; i++){
        if (d[x]%d[i]==0){
            dfs(i,k-1);
        }
    }
}
int main(){
    ios::sync_with_stdio(0);
    cin>>x>>k;
    for (int i=1; 1LL*i*i<=x; i++)
        if (x%i==0){
            d[++n]=i;
            if (1LL*i*i!=x) d[++n]=x/i;
        }
    sort(d+1,d+n+1);
    dfs(n,k);
}
