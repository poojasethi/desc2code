#include <bits/stdc++.h>
using namespace std;
#define LL __int64
LL  sum; 
LL  t;
LL  a[100000];
void dfs(LL n,LL k)
{
    if (sum>=1e5)
        return;
    if (k==0 || n==1)
    {
        sum++;
        printf("%I64d ",n);
        return;
    }
    for (LL i=0;i<t && n>=a[i];i++)
        if (n % a[i]==0)
            dfs(a[i],k-1);
    return;
}
int main(){
    LL  x, k; 
    scanf("%I64d%I64d",&x,&k);
    t=0;
    for(LL i =1;i * i<= x; i ++)
        if(x % i == 0 )
        {
            a[t++] = i ; 
        if(i != x/i)
            a[t++] = x/i;
        }
    sum=0;
    sort(a,a+t);
    dfs(x,k);
    return 0;
}
