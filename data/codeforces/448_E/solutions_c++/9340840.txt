#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define int64 long long
#define N 8500
#define inf 100000
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,len;
short b[N][N];
int64 n,m,a[N];
void dfs(int x,int cen) {
if (!len) return;
if (x==1||cen==m) {
printf("%I64d ",a[x]);
len--;
return;
}
int i;
for (i=1;i<=*b[x]&&len;i++) dfs(b[x][i],cen+1);
}
int main() {
scanf("%I64d%I64d",&n,&m);
for (i=1;1ll*i*i<=n;i++) if (n%i==0) a[++k]=i;
for (;i;i--) if (n%i==0&&1ll*i*i<n) a[++k]=n/i;
For(i,1,k) {
For(j,1,i) if (a[i]%a[j]==0) b[i][++*b[i]]=j;
}
len=inf;
dfs(k,0);
return 0;
}
