#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;

const int N=8500;
typedef long long ll;

ll x,k,a[N];
short next[N][N];
int n,m=1e5;

void dfs(int depth,int i){
	if(!i||depth==k){
		printf("%I64d ",a[i]);
		if(!--m){
			puts("");
			exit(0);
		}
		return;
	}
	for(int j=0;j!=-1;j=next[i][j])
		dfs(depth+1,j);
}

int main(){
	scanf("%I64d%I64d",&x,&k);
	for(int i=1;(ll)i*i<=x;i++) if(x%i==0){
		a[n++]=i;
		if((ll)i*i!=x){
			a[n++]=x/i;
		}
	}
	sort(a,a+n);
	for(int i=0;i<n;i++){
		int c=-1;
		for(int j=i;j>=0;j--) if(a[i]%a[j]==0)
			next[i][j]=c,c=j;
	}
	dfs(0,n-1);
	puts("");
}
