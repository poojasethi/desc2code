#include <bits/stdc++.h>
using namespace std;

#define N 112345
#define LL long long

LL divi[N], m = 0, cnt = 0;

inline void Solve(LL n, LL k){
    if(cnt >= 1e5){
        return;
    }
    if(n == 1 or k == 0){
        printf("%I64d ", n);
        cnt++;
        return;
    }
    for(int i = 1; i <= m and divi[i] <= n; i++){
        if(n % divi[i]) continue;
        Solve(divi[i], k - 1);
    }
    return;
}

int main(){
    LL n, k;
    cin>>n>>k;
    for(LL i = 1; i * i <= n; i++){
        if(n % i) continue;
        divi[++m] = i;
        if(i * i != n) divi[++m] = n / i;
    }
    sort(divi + 1, divi + m + 1);
    Solve(n, k);
    return 0;
}
