#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long ll;
ll a[1000000],co,t;
void dfs(ll x,ll k)
{
    if (co>=1e5)
        return;
    if (k==0||x==1)
       {
        printf("%I64d ",x);
         co++;
         return;
       }
     for (int i=0;i<t&&a[i]<=x;i++)
        if (x%a[i]==0)
           dfs(a[i],k-1);
}
int main ()
{
    ll n,m;
      t=0;
    scanf("%I64d %I64d",&n,&m);
    for (ll i=1;i*i<=n;i++)
        if (n % i==0)
        {
            a[t++]=i;
            if (i!=n/i)
            a[t++]=n/i;
        }
        sort(a,a+t);
        co=0;
        dfs(n,m);
       return 0;
}
