#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
#define N 100000
ll a[1000000];
int m,s;

void factor(ll n){
  for (ll i=1;i*i<=n;++i)
    if (n%i==0){
      a[m++]=i;
      if (i!=n/i) a[m++]=n/i;
    }
  sort(a,a+m);
}

void dfs(ll n,ll k){
  if (k<0 || s>=N) return ;
  if (k==0 || n==1){
    printf("%I64d ",n);
    ++s;
    return ;
  }
  for (int i=0;i<m && a[i]<=n && s<=N;++i)
    if (n%a[i]==0) dfs(a[i],k-1);
}

int main(){
  ll n,k;
  scanf("%I64d%I64d",&n,&k);
  factor(n);
  dfs(n,k);
  return 0;
}
