#include<cstdio>
int i,j,k,len=100000,c[8192];
short b[8192][8192];
__int64 n,m,a[8192];
void dfs(int x,int y)
{
	if(!len)return;
	if(x==0||y==m){
		--len;
		printf("%I64d ",a[x]);
		return;
	}
	for(int i=0;i<c[x];++i)dfs(b[x][i],y+1);
}
int main()
{
	scanf("%I64d%I64d",&n,&m);
	for(i=1;1ll*i*i<n;++i)if(n%i==0)a[k++]=i;
	for(i-=i*i>n;i;--i)if(n%i==0)a[k++]=n/i;
	for(i=0;i<k;++i)
		for(j=0;j<=i;++j)
			if(a[i]%a[j]==0)b[i][c[i]++]=j;
	dfs(k-1,0);		
	return 0;
}
