#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 100000;
typedef long long LL;
LL a[N * 11], b[N + 1];
int m, n;

void dfs(LL x,LL k){
	if (m == N) return ;
	if (x == 1 || k == 0){
		b[++m] = x;
		return ;
	}
	for (int i = 1; i <= n && a[i] <= x; ++i)
		if (x % a[i] == 0) dfs(a[i], k - 1);
}

int main(){
	LL x, k;
	scanf("%I64d%I64d", &x, &k);
	for (int i = 1; (LL)i * i <= x; ++i)
		if (x % i == 0){
			a[++n] = i;
			if (x != (LL)i * i) a[++n] = x / i; 
		}
	sort(a + 1, a + 1 + n);
	if (n > N) n = N;
	dfs(x, k);
	for (int i = 1; i <= m; ++i)
		printf("%I64d%c", b[i], " \n"[i == m]);
	return 0;
}
