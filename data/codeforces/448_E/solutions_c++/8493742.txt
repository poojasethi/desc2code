#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
long long x,k,gs,xx,i,l,p[500010],q[500010],ln,dn[500010],K;
void make(long long x,long long k)
{
	if(ln>=1e5)return;
	if(k<=0&&ln<1e5||x==1){dn[++ln]=p[x];return;}
	for(int i=1;i<=x&&ln<1e5;i++)
		if(p[x]%p[i]==0)make(i,k-1);
}
int main()
{
	scanf("%I64d%I64d",&x,&k);gs=0;xx=x;if(k>1e5)k=1e5;K=k;
	for(i=1;i<=sqrt(xx);i++)
		if(xx%i==0)p[++l]=i,p[++l]=xx/i;
	if(p[l]==p[l-1])l--;sort(p+1,p+l+1);
	make(l,k);for(i=1;i<=ln;i++)printf("%I64d ",dn[i]);
}
