#include<cstdio>
int i,k,len=100000;
__int64 n,m,a[8192];
void dfs(int x,int y)
{
	if(!len)return;
	if(x==0||y==m){
		--len;
		printf("%I64d ",a[x]);
		return;
	}
	for(int i=0;i<=x;++i)if(a[x]%a[i]==0)dfs(i,y+1);
}
int main()
{
	scanf("%I64d%I64d",&n,&m);
	for(i=1;1ll*i*i<n;++i)if(n%i==0)a[k++]=i;
	for(i-=i*i>n;i;--i)if(n%i==0)a[k++]=n/i;
	dfs(k-1,0);		
	return 0;
}
