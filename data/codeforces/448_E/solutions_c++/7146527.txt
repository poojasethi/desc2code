#include<iostream>
#include<string.h>
#include<stdio.h>
#include<algorithm>
#define N 100000
typedef long long ll;
using namespace std;
ll hh[5000000],t;
ll k;
ll ci;
void dfs(ll x,int y)
{
    if (ci==N) return;
    if (y==k||x==1)
    {
        printf("%I64d ",x);
        ci++;
        return;
    }
    for (int i=1;i<=t&&hh[i]<=x;i++)
        if (x%hh[i]==0)
        dfs(hh[i],y+1);
}

int main()
{
    ll x;
    scanf("%I64d%d",&x,&k);
    if (k>=N) k=N;
    for (ll i=1;i*i<=x;i++)
    {
        if (x%i==0) hh[++t]=i;
        if (x%i==0&&i*i!=x) hh[++t]=x/i;
        }
        sort(hh+1,hh+t+1);
        dfs(x,0);
    return 0;
}
