#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<cmath>
using namespace std;
long long x,k,a[200000],divi[100000],ind=0,i=0;
void fil(long long k1,long long n)
{
    if(k1==0){a[ind++]=divi[n];return;}
    a[ind++]=1;
    if(ind==100000)return ;
    for(int j=1;j<=n;++j)
        if(divi[n]%divi[j]==0&&ind<100000)fil(k1-1,j);
}


int main()
{
    cin>>x>>k;
    long long l=(int)sqrt(x);
    if(x==1){cout<<"1";return 0;}
    for(long long j=1;j<=l;++j)
    if(x%j==0){divi[i]=j;if(x/j!=j)divi[++i]=x/j;++i;}
    sort(divi,divi+i);
    if(k<100000)fil(k,i-1);
    else {fill(a,a+100000,1);ind=100000;}
    for(int j=0;j<ind;++j)printf("%I64d ",a[j]);

}
