#include<cstdio>
#include<algorithm>
int g,k;
const int N=5000000;
long long sta[N];
int top;
void dfs(long long v,int d){
  if(g==100000)return;
  if(d==k || v==1){
    printf("%I64d ",v);
    g++;
    return;
  }
  for(int i=1;i<=top&&sta[i]<=v;i++)
    if(v%sta[i]==0)
      dfs(sta[i],d+1);
}

int main(){
  long long x;long long t;
  scanf("%I64d%I64d",&x,&t);
  if(t>=100000)k=100000;
  else k=t;
  for(long long i=1;i*i<=x;i++)
    if(x%i==0){
      sta[++top]=i;
      if(i*i!=x) sta[++top]=x/i;
    }
  std::sort(sta+1,sta+top+1);
  dfs(x,0);
}
