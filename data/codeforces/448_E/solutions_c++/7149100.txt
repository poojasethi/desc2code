#include <bits/stdc++.h>
using namespace std;
long long n,k,sum;
vector<long long>ans;
void solve(long long x,int QAQ)
{
    if(sum >= 100000) return ;
    if(x == 1 || QAQ == k)
    {
        cout << x << " ";
        ++ sum;
        return ;
    }
    for(int i = 0; i < ans.size() && ans[i] <= x && sum < 100000; ++i)
        if(x % ans[i] == 0) solve(ans[i],QAQ+1);
}
int main()
{
    cin >> n >> k;
    for(long long i = 1; i * i <= n; ++i)
        if(n%i == 0)
        {
            ans.push_back(i);
            if(n/i!=i) ans.push_back(n/i);
        }
    sort(ans.begin(),ans.end());
    sum = 0;
    solve(n,0);
    return 0;
}
