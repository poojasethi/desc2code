#include <string.h>
#include <cstdio>
#include <algorithm>
#include <map>
#include <math.h>
#define maxn 100000-1
using namespace std;
long long n,m,a[100000],h,cnt;
void dfs(long long u,long long v)
{
    if(cnt>maxn)return;
    if(u==1 || v==m)
    {
        cnt++;
        printf("%I64d ",a[u]);
        return;
    }
    for(long long i=1;i<=u;i++)if(a[u]%a[i]==0)dfs(i,v+1);
}
int main()
{
    scanf("%I64d%I64d",&n,&m);
    long long u=sqrt(n);
    for(long long i=1;i<=u;i++)if(n%i==0)a[++h]=i;
    if(u*u==n)u--;
    for(long long i=u;i>=1;i--)if(n%i==0)a[++h]=n/i;
    dfs(h,0);
    return 0;
}
