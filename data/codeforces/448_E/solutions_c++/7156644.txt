#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
long long x,k;
vector <long long> v;
int lim = 100000;
void dfs(long long a,long long b){
  if(!lim) return ;
  if(a == 1ll || b == 0){
    printf("%I64d ",a);
    lim--;
    return ;
  }
  for(int I=0;I<v.size() && v[I]<=a ;I++)
    if(a%v[I] == 0)
      dfs(v[I],b-1);
}
int main(){
  scanf("%I64d %I64d",&x,&k);
  for(long long I = 1ll;I*I<=x;I++){
    if(x%I == 0){
     v.push_back(I); 
      if(x/I != I)
    v.push_back(x/I);
    }
  }
  sort(v.begin(),v.end());
  dfs(x,k);
  return 0;
}
