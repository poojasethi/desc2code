import re
s = raw_input()
li, p, pi = [], re.compile(r'(\d+)|(,)|(\.\.\.)'), re.compile(r'(\d+)')
while s:
    s = s.strip()
    m = p.match(s)
    i, c, d = m.groups()
    if c:
        li.append(', ')
    elif d:
        li.append(' ...')
    else:
        try:
            if pi.match(li[-1]):
                i = ' '.join([li.pop(), i])
        except IndexError:
            pass
        finally:
            li.append(i)
    s = s[m.end():]
print re.sub(r' +', ' ', ''.join(li).strip())
