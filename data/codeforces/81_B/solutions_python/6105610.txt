I=lambda:map(int, raw_input().split())
s = raw_input()
i, n = 0, len(s)

ans = ''
while i < n:
    if s[i] == '.':
        if i != 0 and ans[-1] != ' ': ans += ' ...'
        else: ans += '...'
        i += 2
    elif s[i] == ',' and i != n-1: ans += ', '
    elif s[i] == ' ':
        while i < n and s[i] == ' ': i += 1
        i -= 1
        if '0'<=ans[-1]<='9' and i+1 <= n-1 and '0'<=s[i+1]<='9': ans += ' '
    else: ans += s[i]
    i += 1
print ans
