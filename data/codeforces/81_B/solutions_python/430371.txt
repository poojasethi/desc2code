s = raw_input()
lex = ' ... '.join(' , '.join(s.split(',')).split('...')).split()
ans = lex[0]
for i in xrange(1, len(lex)):
    if lex[i] == ',' and lex[i-1].isdigit() or lex[i] == ',' and lex[i-1] == '...' or lex[i].isdigit() and lex[i-1] == '...':
        ans += lex[i]
    else:
        ans += ' ' + lex[i]
print ans
