#!/usr/bin/env python

from sys import stdin
import re

line = stdin.readline().strip()

B = []
for s in line.split(','):
   ss = s.strip()
   C = re.split('( +|\.\.\.\s*\d*)', ss)
   for i in xrange(len(C)):
      C[i] = C[i].strip()
      if C[i].startswith('...'):
         C[i] = C[i].replace(' ', '')
   D = [ c for c in C if c != '' ]
   B.append( ' '.join(D) )

print ', '.join(B).strip()
