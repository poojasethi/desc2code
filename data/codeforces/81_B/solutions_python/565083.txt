s = raw_input()
while s.find(' ,') != -1:
    s = s.replace(' ,', ',')
while s.find('... ') != -1:
    s = s.replace('... ', '...')
s = s.replace(',', ', ')
s = s.replace('...', ' ...')
while s.find('  ') != -1:
    s = s.replace('  ', ' ')
while s[0] == ' ':
    s = s[1:]

    s = s[:-1]
