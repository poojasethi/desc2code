from re import sub
s=raw_input()
s=sub(r",\s*",", ",s)
s=sub(r"\s*\.{3}"," ...",s)
for i in range(10):
    s=sub(r"(\d+)\s{2,}(\d+)",lambda x:x.group(1)+' '+x.group(2),s)
s=sub(r"^\s+","",s)
s=sub(r"\s+$","",s)
s=sub(r"(\d+|\.{3})\s+,",lambda x: x.group(1)+',',s)
s=sub(r"\.{3}\s+([^.])",lambda x:'...'+x.group(1),s)
print s
