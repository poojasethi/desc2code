from operator import add
from sys import stdout

s = raw_input().replace(',', ' , ').replace('...', ' ... ').split()
isnumber = False
hasspace = True
cnt = 0
l = len(s)

for x in s:
  if x.isdigit():
    if isnumber: stdout.write(' ')
    stdout.write(x)
    hasspace = False
    isnumber = True
  else:
    isnumber = False
    if x == ',':
      stdout.write(x)
      if cnt < l - 1: stdout.write(' ')
      hasspace = True
    else:
      if not hasspace: stdout.write(' ')
      stdout.write(x)
      hasspace = False
  cnt += 1
