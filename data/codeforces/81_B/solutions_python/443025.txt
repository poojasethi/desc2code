#!/usr/bin/python

import sys

line = sys.stdin.readline()
ans = ""
last_not_space = -1
c = 0
last_put = ""
for i in xrange(len(line)):
	if line[i] >= "0" and line[i] <= "9":
		if last_not_space != i - 1\
		and (last_not_space == -1\
		or (line[last_not_space] >= "0"\
		and line[last_not_space] <= "9")):
			ans += " "
		ans += line[i]
		last_put = line[i]
	elif line[i] == ",":
		ans += line[i]
		if i < (len(line) - 2):
			ans += " "
			last_put = " "
	elif line[i] == ".":
		c = c + 1
		if c % 3 == 1 and i != 0 and last_put != " ":
			ans += " "
		ans += line[i]
		last_put = line[i]
	if line[i] != " ":
		last_not_space = i
print ans
