import re
s = raw_input()
rules = ((r'\.\.\.'        ,'dots' ),
         (r'(?<!\d)\s+' ,''     ),#rule 4
         (r'\s+(?!\d)'  ,''     ),#rule 4
         (r',(?!$)'     ,', '   ),#rule 1
         (r'(?<!^)dots' ,' dots'),#rule 2
         (r'\s+'        ,' '    ),#rule 3
         (r'dots'       ,'...'  ))
for r in rules: s = re.sub(r[0],r[1],s)
print s
