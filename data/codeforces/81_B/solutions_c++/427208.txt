#include <iostream>
#include <cstdio>
#include <string>

using namespace std;

string s;
string ans="";

int main()
{
	//freopen("in.txt","rt",stdin);
	getline(cin,s);
	
	int ch=0;
	for(int i=0; i<(int)s.size(); i++)
	{
		if(s[i]=='.')
		{
			i+=2;
			if((i!=2)&&(ch!=3))
				ans+=" ...";
			else
				ans+="...";
			ch=2;
		}
		if(s[i]==',')
		{
			ch=3;
			if(i!=(int)s.size()-1)
				ans+=", ";
			else
				ans+=",";
		}
		if((s[i]>='0')&&(s[i]<='9'))
		{
			if(ch==1)
				ans+=" ";
			while((s[i]>='0')&&(s[i]<='9')&&(i<(int)s.size()))
			{
				ans+=s[i];
				i++;
			}
			i--;
			ch=1;
		}
	}
	cout <<ans;
	return 0;
}
