#include<cstdio>
char line[1001];
int main(){
	gets(line);
	int typ = 0;
	for(int i = 0; line[i]; i++){
		if(line[i] == ' '){
		}else if(line[i] == '.'){
			if(i && typ != 0)putchar(' ');
			printf("...");
			i += 2;
			typ = 2;
		}else if(line[i] == ','){
			putchar(',');
			if(line[i + 1])putchar(' ');
			typ = 0;
		}else{
			if(i && typ != 0 && typ != 2)putchar(' ');
			while(line[i] >= '0' && line[i] <= '9')putchar(line[i++]);
			--i;
			typ = 1;
		}
	}
	putchar('\n');
	//scanf(" ");
}
