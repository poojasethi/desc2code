#include <cstdio>
#include <string>
#include <iostream>
using namespace std;
string s;
int x;
int main(){
    getline(cin,s);
    x=-1;
    for(int i=0; i<s.length(); ++i)
        if (s[i]=='.'){
            if (i&&x!=0) cout<<" ";
            cout<<"...";
            i+=2;
            x=1;
        }
        else if (s[i]==','){
            cout<<",";
            if (i<s.length()-1) cout<<" ";
            x=0;
        }
        else if (s[i]>='0'&&s[i]<='9'){
            if (x==2) cout<<" ";
            while(s[i]>='0'&&s[i]<='9') cout<<s[i],++i;
            --i;
            x=2;
        }
    cout<<"\n";
    return 0;
}
