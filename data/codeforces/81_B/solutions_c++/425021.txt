#include<string>
#include<iostream>
#include<cstdio>
#include<cstdlib>

using namespace std;

int main() {
	string ans, s, cur, last;
	char str[300];
	gets(str);
	
	for(int i = 0; str[i]; s = " ") {
		cur = "";
		if(isdigit(str[i])) {
			for(; isdigit(str[i]); ++i)
				cur += str[i];
			if(last != "...")
				ans += s + cur;
			else
				ans += cur;
		} else if(str[i] == '.') {
			i += 3;
			cur = "...";
			ans += s + cur;
		} else if(str[i] == ',') {
			++i;
			cur = ",";
			
			if(last == ",")
				ans += s + cur;
			else
				ans += cur;
		} else {
			++i;
			continue;
		}
		last = cur;
	}
	cout << ans;
}
