char s[260];

    char ch;int k=0;
    gets(s);//p=s;//printf("%c",*p);
    ch=s[0];printf("%c",s[0]);if(ch=='.')k=1;
    for(int i=1;i<strlen(s);i++){
        if(s[i]==' '){if(ch!=','&&ch!='.')ch=' ';}
        else if (isdigit(s[i])){
            if(ch==' '||ch==',')printf(" ");printf("%c",s[i]);
            ch=s[i];
        }
        else if(s[i]==','){
            if(ch==',')printf(" ");printf("%c",s[i]);ch=',';
        }
        else {
            k++;if(k==3)k=0;
            if(ch!='.'||k==1)printf(" ");printf("%c",s[i]);ch='.';
        }
    }

}
