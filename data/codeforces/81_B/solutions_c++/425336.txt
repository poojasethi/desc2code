#include <stdio.h>
#include <string.h>

int n, i, j;
char str[300], ans[300];

int main()
{
	gets(str);
	i=0;
	n=strlen(str);
	j=0;
	while (i<n)
	{
		while (str[i]==' ' && i<n) i++;
		if (i==n) break;
		if (str[i]==',')
		{
			ans[j++]=str[i++];
			ans[j++]=' ';
		}
		else if (str[i]=='.')
		{
			if (j>0 && ans[j-1]!=' ') ans[j++]=' ';
			for (int k=0; k<3; k++)
				ans[j++]=str[i++];
		}
		else
		{
			if (j>0 && '0'<=ans[j-1] && ans[j-1]<='9')
				ans[j++]=' ';
			while ('0'<=str[i] && str[i]<='9')
				ans[j++]=str[i++];
		}
	}
	while (j>0 && ans[j-1]==' ') j--;
	ans[j]=0;
	printf("%s\n", ans);
	return 0;
}
