#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<cstring>
using namespace std;
char s[500],answer[10000];
int top=-1,asp=-1;

int main(void){int i,j,m,flag=-1,l;
gets(s);
l=strlen(s);
for(i=0;i<l;i++){
if(s[i]==','){
flag=3;
answer[++asp]=s[i];
if(i!=l-1) answer[++asp]=' ';
}
else if(s[i]==' '){
if(flag==0){flag=7;}
}
else if(s[i]=='.'){
flag=2;
if(answer[asp]!=' '&&i!=0) answer[++asp]=' ';
for(j=i;j<i+3;j++) answer[++asp]='.';
i+=2;
}
else{
if(flag==7) answer[++asp]=' ';
answer[++asp]=s[i];
flag=0;
}
}
answer[++asp]=0;
printf("%s\n",answer);



return 0;
}
