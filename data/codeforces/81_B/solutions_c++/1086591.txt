#include <stdio.h>

int main()
{
    int sign=2;
    char temp;
    while((temp=getchar())!='\n')
    {
        if(temp=='.')
        {
            if(sign!=2)
                putchar(' ');
            sign=0;
            putchar(getchar());
            putchar(getchar());
        }
        if(sign==1)
        {
            putchar(' ');
            sign=2;
        }
        if(temp>='0'&&temp<='9')
        {
            if(sign==4)
                putchar(' ');
            sign=3;
        }
        if(temp!=' ')
            putchar(temp);
        else    if(sign==3)
            sign=4;

        if(temp==',')
            sign=1;
    }
    return 0;
}
