#include<cstdio>
char s[1000];
int i,S,W,O;
main(){
	gets(s);
	for(i=0;s[i];i++){
		if(s[i]==' ')continue;		
		if((s[i]>='0'&&s[i]<='9'))O=1;
		else if(s[i]==',')O=2;
		else{
			O=3;
			W=(W%3)+1;
		}
		if((S&&O==3&&W==1)||(i&&S==1&&O==1&&s[i-1]==' ')||S==2)printf(" ");
		printf("%c",s[i]);		
		S=O;
	}
}
