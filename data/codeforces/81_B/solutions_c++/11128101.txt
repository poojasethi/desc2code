#include <iostream>

using namespace std;

int main()
{
    string j,last="";
    getline(cin,j);
    int i=0;

    while(i<j.length())
    {
        while(i<j.length()&&j[i]==' ') i++;
        if(j[i]==','){cout << ","; if(i!=j.length()-1) cout <<" "; last="comma"; i++;}
        else if (j[i]=='.') {if (i==0 or last =="comma") cout << "..."; else cout <<" ..."; i+=3; last="dots";}
        else
        {
          if(last=="number") cout << " ";
            while(i<j.length() && j[i]<='9' && j[i]>='0') {cout << j[i]; i++;}
            last = "number";
        }
    }

    return 0;
}
