#include <stdio.h>
#define TT 255+2
char a[TT],b[TT];
void solve ( ) {
	int i,j,flag=0;
	for (i=0,j=0;a[i]!='\0';i++) {
		if (a[i]==' ')
		flag=1;
		else {
			if (b[j-1]>='0'&&b[j-1]<='9'&&a[i]>='0'&&a[i]<='9'&&flag)
			b[j++]=' ';
			b[j++]=a[i];
			flag=0;
		}
	}
	flag=0;
	for (int k=0;k<j;k++) {
		if (k&&b[k-1]==',') {
		printf(" ");
		flag=1;
	}
		if (k+2<j&&b[k]=='.'&&b[k+1]=='.'&&b[k+2]=='.') {
			if (k&&!flag)
			printf(" ..");
			else
			printf("..");
			k+=2;
		}
		printf("%c",b[k]);
		flag=0;
	}
	printf("\n");
}
int main ( ) {
	while (gets(a)) {
		solve ( );
	}
	return 0;
}
