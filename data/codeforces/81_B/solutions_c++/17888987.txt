#include <string>
#include <iostream>
using namespace std;
string s;
bool num(char c)
{
	return (c>='0'&&c<='9');
}
int main()
{
	getline(cin,s);
	for (int i = 0;i < (int)s.size();i++) if (s[i]==' '&&!(num(s[i-1])&&num(s[i+1])))
	{
		s = s.substr(0,i)+s.substr(i+1);
		i--;
	}
	for (int i = 0;i < (int)s.size()-1;i++) if (s[i]==','&&s[i+1]!=' ')
	{
		s = s.substr(0,i+1)+' '+s.substr(i+1);
		i++;
	}
	for (int i = 0;i < (int)s.size();i++)
		if (((s[i-1]=='.'&&s[i-2]=='.'&&s[i-3]=='.')
			||s[i-1]!='.')&&s[i-1]!=' '&&s[i]=='.'&&s[i+1]=='.'&&s[i+2]=='.')
		{
			s = s.substr(0,i)+' '+s.substr(i);
			i+=3;
		}
	if (s[0]==' ')
		s=s.substr(1);
	cout << s << endl;
	return 0;
}
