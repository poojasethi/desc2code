#include <iostream>
#include <vector>
#include <cstring>
#include <algorithm>
using namespace std;

int main(){
__int64 n,m,d;
string b;
getline(cin,b);
for(int i=0;i<b.length();i++){
 if(b[i]==' ')
  if(i>0 && i<b.length() && ((b[i+1]>'9' || b[i+1]<'0') || (b[i-1]<'0' || b[i-1]>'9'))){
  b.erase(b.begin()+i);
  i--;
 }
}
for(int i=0;i<b.length();i++){
 if(b[i]=='.'){
  if(i>0 && b[i-1]!=' '){
   b.insert(b.begin()+i,' ');i+=3;
  }else
  i+=2;
 }
 else if(b[i]==','){
  if(i<b.length()-1 && b[i+1]!=' '){
   b.insert(b.begin()+i+1,' ');i++; 
  }
 }
}
cout<<b;
//system("pause");
return 0;
}
