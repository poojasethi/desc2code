#include<iostream>
#include<string>

using namespace std;

void replace(string& s, const string& from, const string& to)
{
	int t = 0;
	while((t=s.find(from,t))!=string::npos)
	{
		s.erase(t,from.length());
		s.insert(t,to);
		t+=to.length();
	}
}

int main()
{
	string s;
	getline(cin,s);
	for(int i=0;i<255;i++) replace(s,"  "," ");
	for(int i=1;i<s.length()-1;i++) if (isdigit(s[i-1])&&s[i]==' '&&isdigit(s[i+1])) s[i] = '?';

	replace(s," ","");
	replace(s,",",", ");
	replace(s,"..."," ...");
	replace(s,"  "," ");
	replace(s,"?"," ");
	while(s[0]==' ') s.erase(0,1);
	while(s[s.length()-1]==' ') s.erase(s.length()-1,1);
	cout<<s<<endl;

	return 0;
}
