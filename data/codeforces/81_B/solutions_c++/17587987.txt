#include<iostream>
#include<cstring>
using namespace std;
string s;
int main(){
	getline(cin,s);
	int i=0,l=s.length();s+=' ';
	bool spc=false;
	while(i<l){
		while (i<l && s[i]==' ')i++;
		if (i==l) break;
		if ('0'<=s[i]&&s[i]<='9'){
			spc=true;
			while('0'<=s[i] && s[i]<='9'){
				cout<<s[i];
				i++;
			}
			while (i<l && s[i]==' ')i++;
			if (i<l && '0'<=s[i]&&s[i]<='9')cout<<' ';
		}else if(s[i]=='.'){
			if (i && spc) cout<<' ';
			cout<<"...";
			i+=3;
			spc=true;
		}else{
			cout<<',';
			if (i<l-1) cout<<' ';
			i++;
			spc=false;
		}
	}
	return 0;
}
