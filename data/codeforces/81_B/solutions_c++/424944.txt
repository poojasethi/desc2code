#include <cstdio>
#include <vector>
#include <cstring>
#include <cctype>

using namespace std;

char buf[20008];

int main() {
	int pre;
	// 1 num, 2 , 3.
	while (gets(buf)) {
		pre = -1;
		int len = strlen(buf);
		for (int i = 0; i < len; ++i) {
			if (isdigit(buf[i])) {
				if (pre == 1 || pre == 2) {
					putchar(' ');
				}
				while (isdigit(buf[i])) {
					putchar(buf[i]);
					i++;
				}
				i--;
				pre = 1;
			} else if (buf[i] == ',') {
				if (pre == 2) {
					putchar(' ');
				}
				putchar(buf[i]);
				pre = 2;
			} else if (buf[i] == '.') {
				if (pre != -1) {
					putchar(' ');
				}
				i += 2;
				printf("...");
				pre = 3;
			}
		}
		puts("");
	}
}
