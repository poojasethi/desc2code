#include <iostream>

using namespace std;

int main() {
  string a;
  bool z = true, x = false;
  getline(cin, a);
  int n = a.size();
  for (int i = 0; i < n; i++) {
    if (a[i] == ' ')
      continue;
    if (a[i] == ',') {
      cout << "," << (i + 1 < n? " ": "\n");
      z = true;
      x = false;
      continue;
    }
    if (a[i] == '.' && a[i + 1] == '.' && a[i + 2] == '.') {
      cout << (z? "...": " ...");
      i += 2;
      z = false;
      x = false;
      continue;
    }
    cout << (x? " ": "") << a[i];
    x = false;
    if (a[i + 1] == ' ')
      x = true;
    z = false;
  }
  return 0;
}
