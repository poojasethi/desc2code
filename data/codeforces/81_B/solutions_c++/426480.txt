#include <stdio.h>

#define N 1
#define C 2
#define D 3

char s[1000],dc;

int main(){
	int i,state=0,dc=0,newstate=0;
	gets(s);
	for (i=0; s[i]; i++){
		if (s[i]==' ') continue;
		
		if ((s[i]>='0' && s[i]<='9')) newstate=N;
		else if (s[i]==',') newstate=C;
		else{
			newstate=D;
			dc=(dc%3)+1;
		}
		if ((state && newstate==D && dc==1)
			||(i&&state==N&&newstate==N&&s[i-1]==' ')
			||state==C) printf(" ");
		printf("%c",s[i]);
		
		state=newstate;
	}
	printf("\n");
	return 0;
}
