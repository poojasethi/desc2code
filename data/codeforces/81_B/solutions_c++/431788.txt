#include <stdio.h>

char in[256],out[256];
int outlen;

int main()
{
	int j;
	gets(in);
	for(int i=0;in[i];i++)
	{
		if(in[i]=='.')
		{
			if(outlen!=0&&out[outlen-1]!=' ')
			{
				out[outlen++]=' ';
			}
			for(int j=0;j<3;j++)
			{
				out[outlen++]='.';
			}
			i+=2;
		}
		else if(in[i]==',')
		{
			out[outlen++]=',';
			out[outlen++]=' ';
		}
		else if(in[i]==' ')
		{
		}
		else
		{
			while('0'<=in[i]&&in[i]<='9')
				out[outlen++]=in[i++];
			for(j=i;in[j]==' ';j++);
			if('0'<=in[j]&&in[j]<='9')
				out[outlen++]=' ';
			i--;
		}
	}
	if(out[outlen-1]==' ')
		out[outlen-1]='\0';
	else
		out[outlen]='\0';
	puts(out);
}
