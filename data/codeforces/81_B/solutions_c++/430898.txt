#include <iostream>
#include <vector>
#include <string>
#include <cctype>


using namespace std;

int main(){
  string s;
  while(getline(cin, s)){
    for(int i = s.size() -1;i>= 0;i--){
      if(s[i] == ' '){
	if( i -1 >= 0 && i + 1 < s.size() && isdigit(s[i-1] ) && isdigit(s[i+1]))
	  continue;
	s.erase(i, 1);
      }
    }
  
    for(int i = s.size();i>=0;i--){
      if(s[i] == ',' &&  i!= s.size()-1){
	s.insert(s.begin () + i + 1,1, ' ');      
      }

    }

    int cnt = 0;
    for(int i = s.size();i>=0;i--){
      if(s[i] == '.'){
	cnt ++;
	if(cnt % 3 == 0 && s[i-1] != ' ' && i != 0){
	  s.insert(s.begin () + i ,1, ' ');
	}      
      }

    }
    cout << s  <<endl;
  }

  return 0;

}
