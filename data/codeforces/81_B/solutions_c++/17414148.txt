#include<cstdio>
//last: start:0 number:1 comma:2 space:3 dot:4
using namespace std;
int last=0;
void Put(char c){
	putchar(c);
	if(c>='0'&&c<='9') last=1;
	if(c==',') last=2;
	if(c==' ') last=3;
	if(c=='.') last=4;
}
int main(){
	char c=getchar();
	while(c!='\n'){
		if(last==2) Put(' ');
		if(c>='0'&&c<='9'){
			if(last==1) Put(' ');
			do{
				Put(c);
				c=getchar();
			}while(c>='0'&&c<='9');
		}else if(c==','){
			c=getchar();
			Put(',');
		}
		else if(c==' ') c=getchar();
		else if(c=='.'){
			if(last&&last!=3) Put(' ');
			for(int i=0;i<3;i++){
				Put('.');
				c=getchar();
			} 
		}
	}
	return 0;
}
