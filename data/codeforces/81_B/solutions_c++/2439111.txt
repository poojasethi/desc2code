#include<iostream>
#include<string>
using namespace std;
int main()
{
	string a,s="",p="";
	int m=0;
	bool f=false;  
	getline(cin,a);
	for(int i=0;i<a.size();i++)
	{
		if(a[i]==' '){f=true;continue;}
		if(a[i]=='.' && a[i+1]=='.' && a[i+2]=='.')
		{
			int len=s.size();
			if(i!=0 && s[len-1]!=' '){s+=" ...";m++;f=true;}
			else s+="..."; 
			i+=2;
			
		}
		if(a[i]>='0' && a[i]<='9'){if(m==0 && i!=0 && (a[i-1]<'0' || a[i-1]>'9')&& f==true){s+=" ";}s+=a[i];m=0;}
		if(a[i]==',')
		{
			if(i!=a.size()-1){s+=", ";m++;f=true;}
			else s+=",";
	    } 
	}
	cout<<s;
   	return 0;
}
