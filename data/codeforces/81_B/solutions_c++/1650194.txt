# include <iostream>
# include <string>
# include <ctype.h>

using namespace std;

int str,z,z1;
string a;

int main()
{
	getline(cin,a);
	
	str=a.length();
	
	for(int i=0;i<str;i++)
	{
		if(a[i]==',')
		{
			cout<<",";
			
			if(i!=str-1)
			{cout<<" ";z1=1;}
			
			z=0;
		}
		
		if(a[i]=='.')
		{
			if(i==0 || a[i-1]==',' || z1==1)
			{cout<<"...";z1=0;}
			
			else
			cout<<" ...";
			
			i+=2;z=0;z1=0;
		}
		
		if(isdigit(a[i]))
		{
			if(z==1 && isdigit(a[i]))
			{cout << " ";z=0;}
			
			cout<<a[i];
			z1=0;
			
			if(a[i+1]==' ')
			z=1;
		}
	}
	
	cout<<endl;
}
