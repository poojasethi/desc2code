#include <cstdio>
#include <cstring>

using namespace std;

int main()
{
	char buf[300];
	gets(buf);
	int l = strlen(buf), left = 0, right;
	printf("%c", buf[0]);
	if (buf[0] == '.')
	{
		printf("..");
		left = 2;
	}
	while (left + 1 < l)
	{
		right = left + 1;
		while (buf[right] == ' ')
			++right;

		if (buf[left] == ',' || buf[right] == '.' ||
			(right - left > 1 && buf[left] >= '0' && buf[right] >= '0'))
				printf(" ");
		printf("%c", buf[right]);
		if (buf[right] == '.')
		{
			printf("..");
			right += 2;
		}
		left = right;
	}

	printf("\n");
	return 0;
}
