#include <cstdio>

char s[256],dc;
int main()
{
    int i,state=0,dc=0,newstate=0;
    gets(s);
    for (i=0; s[i]; i++)
    {
        if (s[i]==' ') continue;
        if ((s[i]>='0' && s[i]<='9')) newstate=1;
        else if (s[i]==',') newstate=2;
        else
        {
            newstate=3;
            dc=(dc%3)+1;
        }
        if ((state && newstate==3 && dc==1)
            ||(i&&state==1&&newstate==1&&s[i-1]==' ')
            ||state==2) printf(" ");
        printf("%c",s[i]);
        state=newstate;
    }
    printf("\n");
    return 0;
}
