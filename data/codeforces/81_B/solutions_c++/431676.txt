#include <iostream>
#include <string>
using namespace std;

string s, res;
bool is = false;

int main() {
	getline(cin, s);
	for (int i = 0; i < s.size(); i++) {
		switch(s[i]) {
			case ' ': {
				is = true;
				break;
			}
			case ',': {
				res += ',';
				res += ' ';
				break;
			}
			case '.': {
				if (res.size() && res[res.size() - 1] != ' ') res += ' ';
				res += "...";
				i += 2;
				break;
			}
			default: {
				if (res.size() && res[res.size() - 1] >= '0' && res[res.size() - 1] <= '9' && is) res += ' ';
				is = false;
				res += s[i];
				break;
			}
		}
	}
	while (res.size() && res[res.size() - 1] == ' ') res.erase(res.size() - 1);
	cout << res;
}
