#include<stdio.h>
#include<ctype.h>
int l;
char s[9999],o[9999];
int main(){
	int TYPE=1;
	gets(s);
	for(int i=0; s[i];)
		if(isdigit(s[i])){
			if(TYPE==0)o[l++]=' ';
			while(isdigit(s[i]))
				o[l++]=s[i++];
			TYPE=0;//0 - number;
		}else
		if(s[i]=='.'){
			if(TYPE!=1)o[l++]=' ';//1 - space
			i+=3;
			o[l++]='.';
			o[l++]='.';
			o[l++]='.';
			TYPE=2;//2 - other
		}else
		if(s[i]==','){
			o[l++]=',';
			o[l++]=' ';
			i++;
			TYPE=1;//1 - space
		}
		else
			i++;
	while(l>0 && o[l-1]==' ')l--;
	o[l]=0;
	puts(o);
	return 0;
}
