#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

int main()
{
	char c;
	string s, o;
	while(cin.get(c))
		s += c;
	for(int i=0; i<s.size(); i++)
		if(s[i] == '.')
		{
			i += 2;
			if(o.size() && o[o.size()-1] != ' ')
				o += " ";
			o += "...";
		}
		else if(isdigit(s[i]))
		{
			int j;
			string d;
			for(j=i; isdigit(s[j]); j++)
				d += s[j];
			i = j-1;
			if(o.size() && isdigit(o[o.size()-1]))
				o += " ";
			o += d;
		}
		else if(s[i] == ',')
			o += ", ";
	while(o.size() && o[o.size()-1] == ' ')
		o = o.substr(0, o.size()-1);
	cout << o << endl;
	return 0;
}
