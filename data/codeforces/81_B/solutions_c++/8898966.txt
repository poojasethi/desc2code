#include <iostream>
#include <cstdio>
#include <string>

using namespace std;

string s,r;

int main()
{
    
getline(cin,s);

for (int i=0;i<s.size();i++)
{
    if(s[i]==' ')
        continue;

    else if (s[i]==',')
        {
        r+=s[i];
        if (i!=s.size()-1)
        r+=' ';
        }

    else if (s[i]=='.' && s[i+1]=='.' && s[i+2]=='.')
        {
        if (i!=0 && r[r.size()-1]!=' ')
            r+=' ';
        r+=s[i++],r+=s[i++],r+=s[i];
        }

    else
    {
        if (r[r.size()-1]>='0' && r[r.size()-1]<='9' && s[i-1]==' ')
        r+=' ';
        r+=s[i];
    }

}
cout<<r;

return 0;

}
