#include<iostream>
#include<cstring>
#include<cstdio>
#include<queue>
#include<string>
using namespace std;

int main() {
	char n[300];
	gets(n);
	int i,len=strlen(n);
	char pre='!';
	for(i=0; i<len; i++) {
		if(n[i]==' ') {
			int l=i,r=i;
			while(l-1>=0 && n[l]==' ') l--;
			while(r+1<len && n[r]==' ') r++;
			if(n[l]>='0' && n[l]<='9' && n[r]>='0' && n[r]<='9') {
				cout<<" ";
				pre=' ';
			}
			i=r;
		}
		if(n[i]==',' && i+1<len) {
			printf(", ");
			pre=' ';
		}
		else if(n[i]=='.') {
			if(pre==' ' || i==0) printf("...");
			else printf(" ...");
			pre='.';
			i+=2;
		}
		else {
			cout<<n[i];
			pre='!';
		}
	}
	cout<<endl;
	return 0;
}
