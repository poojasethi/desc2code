#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
char s[333];
string r;
int l, cnt, sp;
int main() {
    gets(s); l = strlen(s); r = "";
    for (int i = 0;s[i];i++)
        if (s[i] == '.') r += " ...", i += 2;
        else if (s[i] == ',') r += ", ";
        else if (isdigit(s[i])) {
            if (i && s[i - 1] == ' ' && r.size() && isdigit(r[r.size() - 1])) r += ' ';
            r += s[i];
        }
    sp = true; cnt = 0;
    for (int i = 0;i < r.size();i++)
        if (r[i] == ' ') if (!sp) { sp = true; s[cnt++] = ' '; } else {}
        else s[cnt++] = r[i], sp = false;
    while (cnt && s[cnt - 1] == ' ') cnt--; s[cnt] = 0;
    puts(s);
}
