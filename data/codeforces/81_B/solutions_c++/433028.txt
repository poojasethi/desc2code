#include <iostream>
#include <string>
#include <vector>
using namespace std;
bool isNum(char c) {return ('0' <= c && c <= '9');}
int main(int argc, char *argv[]) {
	string str;
	getline(cin,str);
	char bef='x';
	for (int i = 0; i < str.size(); i++) {
		if (isNum(str[i])) {
			if (i-1>=0 && str[i-1] == ',' || str[i-1] == ' ' && bef!='.') {cout<<' ';bef=' ';}
			cout<<str[i]; bef=str[i];
		}
		else if (str[i] == ',') {
			if (i-1>=0 && str[i-1] == ',') {cout<<' ';bef=' ';}
			if (bef == ',') cout <<' ';
			cout<<','; bef=str[i];
		}
		else if (str[i] == '.') {
			if (i-1>=0) cout <<' ';
			cout<<"..."; i+=2; bef='.';
		}
	}
	cout<<endl;
	return 0;
}
