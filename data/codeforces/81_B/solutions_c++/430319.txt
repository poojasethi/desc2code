#include <iostream>

using namespace std;
long i;
string str;
string rez;


int main()
{
    getline(cin,str);
    while(i<str.size())
    {
    if(str[i]<='9'&&str[i]>='0')while(str[i]<='9'&&str[i]>='0'){rez+=str[i];i++;continue;}
    if(str[i]==','){if(rez[rez.size()-1]==' '&&rez[rez.size()-2]!=','&&rez[rez.size()-2]!='.')rez[rez.size()-1]=str[i];else rez+=str[i];if(i!=str.size()-1)rez+=" ";i++;continue;}
    if(str[i]=='.'){if(rez[rez.size()-1]!=' '&&i!=0)rez+=" ";{rez+="...";i+=3;continue;}}
    if(str[i]==' ')if(rez[rez.size()-1]!=' '&&i!=0&&rez[rez.size()-1]!='.')rez+=" ";
    i++;    
    }
    cout<<rez;
    
    return 0;
}
