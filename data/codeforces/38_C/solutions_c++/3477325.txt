#include <stdio.h>
int a[1000];
int main(){
    int n,l;
    scanf("%d%d",&n,&l);
    for (int i=0;i<n;i++)
        scanf("%d",&a[i]);
    int max=0;
    for (int m=l;m<200;m++){
        int num=0;
        for (int i=0;i<n;i++)
                num+=a[i]/m;
        if (max<m*num)
            max=m*num;
    }
    printf("%d",max);;
}
