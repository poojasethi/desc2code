#include <stdio.h>
int main(){
	int n,l;
	while(scanf("%d%d",&n,&l)!=EOF){
		int a,i,j,sum[109]={0};
		for(i=0;i<n;i++){
			scanf("%d",&a);
			if(a>=l){
				for(j=l;j<=a;j++)
					sum[j]+=a/j*j;
			}
		}
		int max=-1;
		for(j=l;j<=100;j++)
			if(max<sum[j]) max=sum[j];
		printf("%d\n",max);
	}
	return 0;
}
