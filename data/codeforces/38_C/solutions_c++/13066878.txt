#include<bits/stdc++.h>
using namespace std;
int a[1005];
int main()
{
int n;
scanf("%d",&n);
int i;
int l;
scanf("%d",&l);
for(i=1;i<=n;i++)
scanf("%d",&a[i]);
int ans = 0;
int d;
for(d=l;d<=105;d++)
{
int cur = 0;
for(i=1;i<=n;i++)
cur+=(a[i]/d)*d;
ans = max(ans,cur);
}
printf("%d\n",ans);
return 0;
}
