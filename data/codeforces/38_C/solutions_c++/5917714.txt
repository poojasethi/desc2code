#include <iostream>
#include <algorithm>
using namespace std;
int N,L,A[110];
int main(){
  int i,j;
  cin>>N>>L;
  for(i=0;i<N;++i)cin>>A[i];
  int ans=0;
  for(i=L;i<=100;++i){
   j=0;
   for(int k=0;k<N;++k)j+=A[k]/i;
   ans=max(ans,j*i);
                     }
  cout<<ans<<'\n';
  //system("pause");
  return 0;
}
