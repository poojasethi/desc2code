#include<stdio.h>
int main()
{
    int n,m,sum=0,a[105],i,max,j;
    scanf("%d%d",&n,&m);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    max=0;
    for(i=m;i<=100;i++)
    {
        sum=0;
    for(j=0;j<n;j++)
    sum+=a[j]/i;
    if(max<sum*i)max=sum*i;
    }
    printf("%d\n",max);
    return 0;
}
