#include<iostream>
#include<algorithm>
using namespace std;

int a[1000], n, l;

int main(){
	cin >> n >> l;
	for(int i = 0; i < n; i++)
		cin >> a[i];

	int ans = 0;
	for(int k = l; k < 110; k++){
		int d = 0;
		for(int i = 0; i < n; i++)
			d += a[i]/k;
		ans = max(ans, d*k);
	}
	cout << ans << endl;		
	return 0;
}
