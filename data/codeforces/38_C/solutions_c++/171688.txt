#include <cstdio>
#include <algorithm>
int a[300];
int main()
{
	int n,i,j,sum,L,maxv;
	scanf("%d%d",&n,&L);
	for (i=1;i<=n;i++)
		scanf("%d",&a[i]);
	maxv=-1;
	for (i=L;i<=100;i++)
	{
		sum=0;
		for (j=1;j<=n;j++)
			sum+=a[j]/i;
		if (sum*i>maxv)
			maxv=sum*i;
	}
	printf("%d\n",maxv);
	return 0;
}
