#include<iostream>
using namespace std;

int main(){
	int n,l,max=0,ans=0,sum,i,j,a[100];
	cin>>n>>l;
	for(i=0;i<n;i++){
		cin>>a[i];
		if(a[i]>max) max=a[i];
	}
	for(i=l;i<=max;i++){
		for(sum=0,j=0;j<n;j++) sum+=a[j]/i;
		if(sum*i>ans) ans=sum*i;
	}
	cout<<ans<<endl;
	return 0;
}
