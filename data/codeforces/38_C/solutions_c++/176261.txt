#include <iostream>

using namespace std;

int a[101];

int main(){
  int n,l;
  cin >> n >> l;
  for(int i=0;i<n;i++){
    cin >>a[i] ;
  }
  
  int ans(0);

  for(int i=l;i<100;i++){
    int count(0);
    for(int j = 0;j<n;j++){
      count += a[j]/i;
    } 
    ans = ans >= count*i ? ans : count*i;
  }

  cout << ans <<endl;

}
