#include<cstdio>
int a[100];
int main(){
	int n,l,r,s,ans,i,j;
	while(~scanf("%d%d",&n,&l)){
		r=ans=0;
		for(i=0;i<n;i++){scanf("%d",a+i);if(a[i]>r)r=a[i];}
		for(j=l;j<=r;j++){
			s=0;
			for(i=0;i<n;i++)s+=a[i]/j;
			if(s*j>ans)ans=s*j;
		}
		printf("%d\n",ans);
	}
	return 0;
}
