#include <iostream>

using namespace std;

const int Maxn = 1000;

int n, l;
int mx, k, a[Maxn];

int main()
{
	cin >> n >> l;

	for(int i = 0;i < n;i++)
		cin >> a[i];

	for(int i = l;i <= 100;i++)
	{
		k = 0;
		for(int j = 0;j < n;j++)
		{
			k += a[j] / i;
		}

		if(mx < k * i)
			mx = k * i;
	}

	cout << mx << endl;
}
