#include<stdio.h>
int main()
{
	int i,j,n,l,m[101],sum,max=0,ans=0;
	scanf("%d%d",&n,&l);
	for(i=0;i<n;++i){
		scanf("%d",&m[i]);
		if(max<m[i])
			max=m[i];
	}
	for(i=l;i<=max;++i){
		sum=0;
		for(j=0;j<n;++j)
			sum+=(m[j]/i);
		if(sum*i>ans)
			ans=sum*i;
	}
	printf("%d\n",ans);
	return 0;
}
