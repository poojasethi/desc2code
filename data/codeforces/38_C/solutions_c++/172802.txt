#include <stdio.h>

int main()
{
	int n, l, a[101], max=0, Pl=0, pl, i, j;
	scanf("%d %d", &n, &l);
	for (i=1; i<n+1; i++)
	{
		scanf("%d", &a[i]);
		if (a[i]>max) max=a[i];
	}
	for (i=l; i<max+1; i++)
	{
		pl=0;
		for (j=1; j<n+1; j++)
		{
			pl+=a[j]/i*i;
		}
		if (pl>Pl) Pl=pl;
	}
	printf("%d\n", Pl);
	return 0;
}
