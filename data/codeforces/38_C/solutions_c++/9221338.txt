#include<cstdio>
using namespace std;
int aa[101];
int main(){
	int n,l,a,ans=0,m=0,area=0;
	scanf("%d%d",&n,&l);
	for(int i=1;i<=n;i++){
		scanf("%d",&aa[i]);
		if(aa[i]>m)m=aa[i];
	}
	for(int i=l;i<=m;i++){
		ans=0;
		for(int j=1;j<=n;j++){
			ans+=aa[j]/i;
		}if(ans*i>area)area=ans*i;
	}
	printf("%d",area);
	return 0;
}
