#include <iostream>

using namespace std;

const int Gmax = 100;

int ans;
int g[Gmax+10];

int l, n, cur;

int main(){
	cin >> n >> l;

	for (int counter = 0; counter < n; counter++){
		cin >> cur;
		for (int i = l; i <= Gmax; i++)
			g[i] += cur/i;
	}

	for (int i = l; i <= Gmax; i++)
		ans = max(ans, i*g[i]);

	cout << ans;

	return 0;
}
