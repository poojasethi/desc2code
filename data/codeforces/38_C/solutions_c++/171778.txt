#include<cstdio>
#include<algorithm>
#define MAX 105
using namespace std;

int a[MAX],n,l,i,j,res,now;

int main()
{
scanf("%d%d",&n,&l);
for(i=0; i<n; ++i) scanf("%d",&a[i]);
for(i=l; i<=100; ++i)
   {
   now=0;
   for(j=0; j<n; ++j) now+=a[j]/i;
   res=max(res,now*i);
   }
printf("%d\n",res);
return 0;
}
