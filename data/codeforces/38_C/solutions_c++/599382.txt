#include<iostream>
using namespace std;
const int maxn = 100+10;
int a[maxn];
int n,l;
int ans=0;
int main()
{
	cin >> n >> l;
	for(int i=0;i<n;i++)
		cin >> a[i];
	for(int cur = l;cur <=maxn;cur++)
	{
		int temp=0;
		for(int i=0;i<n;i++)
			temp+= a[i]/cur;
		if(temp * cur > ans)
			ans = temp *cur;
	}
	cout << ans << endl;
	return 0;
}
