#include <iostream>
using namespace std;
int main()
{
    int i,j,n,k,l,a[102],max = 0;
    cin >> n >> l;
    for (i=0; i<n; i++) cin >> a[i];
    for (i=l; i<=100; i++)
    {
        k = 0;
        for (j=0; j<n; j++) k += a[j] / i;
        if (k*i > max) max = k*i;
    }
    cout << max << endl;
    return 0;
}
