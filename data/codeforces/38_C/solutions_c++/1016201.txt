#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int n,a[110],l,ans=0;
    cin>>n>>l;
    for(int i=0;i<n;i++)cin>>a[i];
    for(int i=l;i<110;i++)
    {
        int sum=0;
        for(int j=0;j<n;j++)
            sum+=a[j]/i;
        ans=max(ans,sum*i);
    }
    cout<<ans;
}
