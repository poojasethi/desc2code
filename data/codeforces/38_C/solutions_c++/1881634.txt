#include<iostream>
using namespace std;

int a[101];
int main()
{
	int n,l;
	cin>>n>>l;
	int i;
	for(i=0;i<n;i++)cin>>a[i];
	int max=0;
	for(i=l;i<101;i++)
	{
		int tmax=0;
		for(int j=0;j<n;j++)
			tmax+=a[j]/i;
		tmax*=i;
		if(tmax>max)max=tmax;
	}
	cout<<max<<endl;
	return 0;
}
