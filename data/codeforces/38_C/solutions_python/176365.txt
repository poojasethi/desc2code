import sys

def main():
	n, minl = [int(it) for it in sys.stdin.readline().split()]
	a = [int(it) for it in sys.stdin.readline().split()]
	maxl = max (a)
	best_area = 0
	for l in xrange (minl, maxl + 1):
		cnt = 0
		for d in a:
			cnt += d // l
		best_area = max (best_area, cnt * l)
	print best_area

if len (sys.argv) > 1:
	sys.stdin = open (sys.argv[1], 'r')
main()
