#include <bits/stdc++.h>
using namespace std;
int n,m,a[200200],b[200200],l,r,x;
int main(){
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		(a[i]==a[i-1])?b[i]=b[i-1]:b[i]=i-1;
	}
	for(int i=0;i<m;i++){
		scanf("%d%d%d",&l,&r,&x);
		if(a[r]!=x)printf("%i\n",r);
		else if(b[r]>=l&&b[r])printf("%i\n",b[r]);
		else printf("-1\n",r);
	}	
}
