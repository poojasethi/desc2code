#include <stdio.h>
#include <iostream>
#include <string.h>
using namespace std;
int main()
{
	int n,m,a[200005],l,next[200005],r,x;
	cin>>n>>m;
	int i,j;
	for(i=1;i<=n;i++)
	cin>>a[i];
	for(i=1;i<=n;i++)
	if(a[i-1]!=a[i]||i==1)	next[i]=i-1;
	else
		next[i]=next[i-1];	
	for(i=0;i<m;i++)
	{
		scanf("%d%d%d",&l,&r,&x);
		if(a[r]==x&&next[r]<l)		printf("-1\n");
		else if(a[r]==x)	printf("%d\n",next[r]);
		else printf("%d\n",r);
	 } 
}
