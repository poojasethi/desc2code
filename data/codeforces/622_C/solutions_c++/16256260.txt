#include<bits/stdc++.h>
using namespace std;
int a[200009],b[200009];
int main()
{
    int n,q,m,k,c,i,j,l,r,x;
    cin>>n>>m;
    for(i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        (a[i]==a[i-1])?b[i]=b[i-1]:b[i]=i-1;
    }
    for(i=0;i<m;i++)
    {
        scanf("%d %d %d",&l,&r,&x);
        if(a[r]!=x) cout<<r<<endl;
        else if(b[r]>=l) cout<<b[r]<<endl;
        else cout<<-1<<endl;
    }
}
