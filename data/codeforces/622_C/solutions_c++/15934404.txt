#include <cstdio>
#define MAXN 200010
int n,m,a[MAXN],f[MAXN];
int main()
{
	scanf("%d %d",&n,&m);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	for(int i=n;i>=1;i--)
	{
		if(a[i]==a[i+1]) f[i]=f[i+1];
		else f[i]=i;
	}
	for(int i=1,l,r,x;i<=m;i++)
	{
		int next;
		scanf("%d %d %d",&l,&r,&x);
		if(a[l]!=x) printf("%d\n",l);
		else
		{
			next=f[l]+1;
			if(next>r) printf("-1\n");
			else printf("%d\n",next);
		}
	}
	return 0;
}
