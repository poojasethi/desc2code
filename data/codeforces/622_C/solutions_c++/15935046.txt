#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int N=2e5+32;
int n,m,l,r,x;
int A[N],nex[N];
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;++i)
		scanf("%d",&A[i]);
	nex[n]=n+1;
	for(int i=n;i>1;--i)
		if(A[i]==A[i-1])nex[i-1]=nex[i];
		else nex[i-1]=i;
	for(int i=1;i<=m;++i)
	{
		scanf("%d%d%d",&l,&r,&x);
		if(A[l]==x){
			if(nex[l]>r)printf("-1\n");
			else printf("%d\n",nex[l]);
		}
		else printf("%d\n",l);
	}
	return 0;
}
