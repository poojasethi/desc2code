#include<bits/stdc++.h>
using namespace std;
int a[200020],b[200020],l,r,x,n,m;
int main(){
scanf("%d%d",&n,&m);
for(int i=1;i<=n;i++)
{scanf("%d",&a[i]); 
	b[i]=((a[i]==a[i-1])?b[i-1]:(i-1));
}
for(int i=0;i<m;i++)
{
	scanf("%d%d%d",&l,&r,&x);
	if(a[r]!=x) printf("%d \n",r);
	else if(b[r]>=l && b[r]) printf("%d \n",b[r]);
	else printf("-1\n");
}
return 0;
}
