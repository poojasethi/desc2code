#include<cstdio>
int a[200000+5];
int pre[200000+5];
int main(void){
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
	}
	for(int i=2;i<=n;i++){
		if(a[i]==a[i-1]) pre[i]=pre[i-1];
		else pre[i]=i-1;
	}
	int l,r,x;
	
	while(m--){
		scanf("%d%d%d",&l,&r,&x);
		if(a[r]!=x) printf("%d\n",r);
		else if(pre[r]<l) printf("-1\n");
		else printf("%d\n",pre[r]);
	}
	return 0;
}
