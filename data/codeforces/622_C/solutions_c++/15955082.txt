#include<iostream>
#include<cstdio>
using namespace std;

int a[200001],b[200001],m,n,l,r,x,ans[200001];

int main(){
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		if(a[i-1]==a[i])
			b[i]=b[i-1];
		else
			b[i]=i-1;
	}	
	for(int i=1;i<=m;i++){
		scanf("%d%d%d",&l,&r,&x);
		if(a[r]!=x)
			ans[i]=r;
		else{
			if(b[r]>=l)
				ans[i]=b[r];
			else
				ans[i]=-1;
		}
	}	
	for(int i=1;i<=m;i++)
		printf("%d\n", ans[i]);
return 0;
}
