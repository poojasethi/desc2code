#include <stdio.h>
int a[200010],b[200010];
int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		if (a[i]==a[i-1]) b[i]=b[i-1];
		else b[i]=i;
	}
	int l,r,x;
	while (m--)
	{
		int ans=0;
		scanf("%d %d %d",&l,&r,&x);
		if (a[r]!=x) ans=r;
		else if (b[r]>l) ans=b[r]-1;
		if (!ans) printf("-1\n");
		else printf("%d\n",ans);
	}
}
