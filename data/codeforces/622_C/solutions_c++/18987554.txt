#include <iostream>
#include <cstdio>
int a[2*100000+10];
int pos[2*100000+10];
int n, query, l, r, x;
void find()
{
	for(int i=r; i>=l;)
	{
		if(a[i]!=x)
		{
			printf("%d\n", i);
			return;
		}
		i=pos[i];
	}
	printf("-1\n");
	return;
}
int main()
{
	scanf("%d%d", &n, &query);
	for(int i=1;i<=n;i++)
	{
		scanf("%d", &a[i]);
		if(a[i]==a[i-1])
			pos[i]=pos[i-1];
		else
			pos[i]=i-1;
	}
	while(query--)
	{
		scanf("%d%d%d", &l, &r, &x);
		find();
	}
	return 0;
}
