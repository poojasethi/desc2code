#include <iostream>
#include <cstdio>
using namespace std;

int n, m, l, r, x, a[200005], b[200005];
int main()
{
	int i, t1, t2 = 0;;
//	freopen ("input.txt", "r", stdin);
	cin >> n >> m;
	for (i = 1; i <= n; i++) {
		scanf("%d", &a[i]);
		if (a[i] == a[i - 1]) b[i] = b[i - 1];
		else b[i] = i - 1;
	}
	while(m--) {
		scanf("%d %d %d", &l, &r, &x);
		if (a[r] != x) printf("%d", r);
		else {
			if (b[r] >= l) printf("%d", b[r]);
			else printf("-1");
		}
		puts("");
	}
    return 0;
}
