#include <stdio.h>
#define ll int

int main()
{
    ll n,m;
    scanf("%d %d",&n,&m);
    ll a[n+1];
    ll i;
    ll c[n+1];
    for (i=1; i<=n; i++)
    {
        scanf("%d",a+i);
        if (a[i]==a[i-1]) c[i]=c[i-1]; else c[i]=i-1;
    }
    for (i=1; i<=m; i++)
    {
        ll l,r,x;
        scanf("%d %d %d",&l,&r,&x);
        ll res=-1;
        if (a[r]!=x) res=r; else
        if (c[r]>=l) res=c[r];
        printf("%d\n",res);
    }
    return 0;
}
