#include <bits/stdc++.h>
#define MAX 200005
using namespace std;
int n, m, l, r, x;
int a[MAX], p[MAX];

int main()
{
	p[0] = -1;
	scanf("%d%d",&n,&m);
	for(int i = 0; i < n; i++) scanf("%d", &a[i]);
	for(int i = 1; i < n; i++)
		if(a[i] != a[i - 1]) p[i] = i - 1;
		else p[i] = p[i - 1];

	while(m--)
	{
		scanf("%d%d%d",&l,&r,&x);
		l--, r--;
		if(a[r] != x) printf("%d", r + 1);
		else if(p[r] >= l) printf("%d", p[r] + 1);
		else printf("-1");
		printf("\n");
	}
}
