#include <stdio.h>
#include <iostream>
#include <string.h>
using namespace std;
int main()
{
	int n,m,a[200005],l,next[200005],r,x;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
   scanf("%d",&a[i]);

	for(int i=n;i>=1;i--)
	if(a[i+1]!=a[i]||i==n)	next[i]=i+1;
	else
		next[i]=next[i+1];	
	for(int i=0;i<m;i++)
	{
		 scanf("%d%d%d",&l,&r,&x);
		if(a[l]==x&&next[l]>r)		printf("-1\n");
		else if(a[l]==x)	printf("%d\n",next[l]);
		else printf("%d\n",l);
	 } 
}
