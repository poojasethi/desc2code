#include<iostream>
#include<cstdio>
using namespace std;

int main()
{
	int n,m,a[200005],nxt[200005],i,l,r,x,num;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++) scanf("%d",&a[i]);
	for(i=n;i>0;i--)
	{
		if(i==n||a[i]!=a[i+1]) nxt[i]=i+1;
		else nxt[i]=nxt[i+1];
	}
	while(m--)
	{
		scanf("%d%d%d",&l,&r,&x);
		if(a[l]==x&&nxt[l]>r) num=-1;
		else if(a[l]==x) num=nxt[l];
		else num=l;
		printf("%d\n",num);
	}
	return 0;
}
