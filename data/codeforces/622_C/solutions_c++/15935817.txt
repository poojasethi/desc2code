#include<cstdio>

int n, m, l, r, x, a[200005], p[200005];

int main() {
	scanf("%d %d", &n, &m);
	for (int i = 1; i <= n; i++) {
		scanf("%d", &a[i]);
		p[i] = a[i] == a[i - 1] ? p[i - 1] : i - 1;
	}
	for (int i = 0; i < m; i++) {
		scanf("%d %d %d", &l, &r, &x);
		printf("%d\n", a[r] == x ? p[r] >= l ? p[r] : -1 : r);
	}
	scanf("\n");
	return 0;
}
