#include <bits/stdc++.h>
using namespace std;

int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	int arr[n+5];
	int before[n];
	for(int i = 1 ; i <= n ; i++){
		scanf("%d",&arr[i]);
		if(arr[i]!=arr[i-1]){
			before[i]=i-1;
		}
		else {
			before[i]=before[i-1];
		}
	}
	while(m--){
		int l,r,x;
		scanf("%d %d %d",&l,&r,&x);
		if(arr[r]!=x){
			printf("%d\n",r);
		}
		else if(before[r]<l){
			printf("-1\n");
		}
		else{
			printf("%d\n",before[r]);
		}
	}
}
