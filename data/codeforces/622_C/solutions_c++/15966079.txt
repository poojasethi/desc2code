#include <bits/stdc++.h>
using namespace std;
int n,m;
int a[200009],p[200009];
int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1;i<=n;)
	{
		int j=i;
		while(j<=n && a[i]==a[j])j++;
		for(;i<j;i++)p[i]=j;
	}
	while(m--)
	{
		int l,r,x;
		scanf("%d%d%d",&l,&r,&x);
		if(a[l]!=x)printf("%d\n",l);
		else if(p[l]>r)printf("-1\n");
		else printf("%d\n",p[l]);
		//printf("%d\n",query(1,l,r,1,n,x));
	}
	return 0;
}
