#include<iostream>
#include<cstdio>
#include<set>
using namespace std;
int n,ans;
int a[100010],b[100010];
multiset<int> an;
multiset<int>::iterator it;
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&ans);
		a[ans]=i;
	}
	for(int i=0;i<n;i++){
		scanf("%d",&b[i]);
		an.insert(i-a[b[i]]);
	}
	for(int i=0;i<n;i++){
		it=an.lower_bound(i);
        ans=100000;
        if(it!=an.end())   ans=min(ans,(*it)-i);
        if(it!=an.begin()) ans=min(ans,i-(*(--it)));
        printf("%d\n",ans);
        an.erase(an.find(i-a[b[i]]));
        an.insert((i+n)-a[b[i]]);
	}
	return 0;
}
