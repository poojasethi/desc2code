#include <iostream>
#include <set>
using namespace std;

int pos[100000];
int a[100000];
multiset<int> s;

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) {
        int num;
        cin >> num;
        pos[num] = i;
    }
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        s.insert(i - pos[a[i]]);
    }
    for (int i = 0; i < n; i++) {
        multiset<int>::iterator it = s.lower_bound(i);
        int k = 1 << 29;
        if (it != s.end()) k = min(k, *it - i);
        if (it != s.begin()) k = min(k, i - *(--it));
        cout << k << endl;
        s.erase(s.find(i - pos[a[i]]));
        s.insert(i + n - pos[a[i]]);
    }
    return 0;
}
