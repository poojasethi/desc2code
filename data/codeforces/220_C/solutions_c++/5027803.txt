#include <iostream>
#include <set>

using namespace std;

typedef pair<int, int> pii;

const int MAXN = 1000 * 100 + 10;
int a[MAXN], b[MAXN];
multiset<int> que;

int main()
{
	int n, x;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> x, a[x] = i;
	for (int i = 0; i < n; i++)
	{
		cin >> b[i];
		que.insert(i - a[b[i]]);
	}
	for (int i = 0; i < n; i++)
	{
		int ans = MAXN;
		multiset<int>::iterator it = que.lower_bound(i);
//		cout << *it << endl;
		if (it != que.end())
			ans = min(ans, *it - i);
		if (it != que.begin())
		{
			it--;
			ans = min(ans, i - *it);
		}
		cout << ans << endl;
		que.erase(que.find(i - a[b[i]]));
		que.insert(n + i - a[b[i]]);
	}
	return 0;
}
