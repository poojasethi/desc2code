#include <iostream>
#include <algorithm>
#include <set>
using namespace std;
const int M=100100l;
int n,a[M],c[M];
main(){
  multiset<int> v;
  cin>>n;
  for(int i=0,x;i<n&&cin>>x;a[x]=i++);
  for(int i=0;i<n&&cin>>c[i];++i)v.insert(i-a[c[i]]);
  for(int i=0,t;i<n;++i){
    multiset<int>::iterator it=v.lower_bound(i);
    int u=M;
    if(it!=v.end())u=min(u,*it-i);
    if(it!=v.begin())u=min(u,i-*(--it));
    cout<<u<<endl;
    v.erase(v.find(t=i-a[c[i]]));
    v.insert(t+n);
  }
}
