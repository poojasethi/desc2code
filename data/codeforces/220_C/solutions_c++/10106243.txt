#include <cstdio>
#include <cstdlib>
#include <set>
#include <algorithm>
#include <ctime>
#define N 100005
#define LIMIT 100000005
using namespace std;
typedef multiset<int>::iterator mst;
multiset<int> a;

int get_ans(int offset)
{
	mst t=a.lower_bound(offset);
	int delta=*t-offset;
	return delta?min(delta,offset-(*--t)):0;
}

int main()
{
	a.insert(-LIMIT),a.insert(LIMIT);
	static int p[N],b[N],n;
	scanf("%d",&n);
	for(int i=0,x;i<n;++i)
		scanf("%d",&x),p[x]=i;
	for(int i=0;i<n;++i)
		scanf("%d",&b[i]),a.insert(p[b[i]]-i);
	for(int i=0;i<n;++i)
	{
		printf("%d\n",get_ans(-i));
		a.erase(a.find(p[b[i]]-i));
		a.insert(p[b[i]]-n-i);
	}
	return 0;
}
