#include<set>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=1000*100+100;
multiset<int> s;
int n, a[maxn], b[maxn], c[maxn], i, j;
int main(){
    cin>>n;
    for(i = 0; i < n; i++) cin>>j, c[j] = i;
    for(i = 0; i < n; i++) cin>>b[i], s.insert(i - c[b[i]]);
    for(i = 0; i < n; i++){
        multiset<int>::iterator it = s.lower_bound(i);
        int ans = 999999999;
        if(it != s.end())
            ans = min(ans, *it-i);
        if(it != s.begin())
            ans = min(ans, i-(*(--it)));
        cout<<ans<<endl;
        s.erase(s.find(i-c[b[i]]));
        s.insert((i+n)-c[b[i]]);
    }
    return 0;
}
