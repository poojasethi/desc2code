#include<stdio.h>
#include<set>
#include<algorithm>
#define N 1000000
using namespace std;
int a[N],b[N];
multiset<int> mm;
multiset<int>::iterator it;
int main()
{
    int i,j,n,ans;
    scanf("%d",&n);
    for(i=0;i<n;i++) scanf("%d",&j),a[j-1]=i;
    for(i=0;i<n;i++) scanf("%d",b+i),b[i]--,mm.insert(i-a[b[i]]);
    for(i=0;i<n;i++)
    {
        it=mm.lower_bound(i),ans=N;
        if(it!=mm.end()) ans=min(ans,(*it)-i);
        if(it!=mm.begin()) ans=min(ans,i-*(--it));
        printf("%d\n",ans);
        mm.erase(mm.find(i-a[b[i]]));
        mm.insert(i-a[b[i]]+n);
    }
}
