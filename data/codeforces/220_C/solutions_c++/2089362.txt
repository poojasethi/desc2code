#include<cstdio>
#include<set>
#include<climits>
#include<algorithm>
using namespace std;
#define MAX_N 100005
int a[MAX_N],b[MAX_N],pos[MAX_N];
int main(){
	int n;
	scanf("%d",&n);
	multiset <int> S;
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
		pos[a[i]]=i;
	}
	for(int i=0;i<n;i++){
		scanf("%d",&b[i]);
		S.insert(i-pos[b[i]]);
	}
	for(int i=0;i<n;i++){
		multiset <int>::iterator itr=S.lower_bound(i);
		int answer=INT_MAX;
		if(itr!=S.end())
			answer=min(answer,*itr-i);
		if(itr!=S.begin())
			answer=min(answer,i-*(--itr));
		printf("%d\n",answer);
		S.erase(S.find(i-pos[b[i]]));
		S.insert(i-pos[b[i]]+n);
	}
	return 0;
}
