#include<cstdio>
#include<iostream>
#include<set>

#define FOR(a,b) for(int a=0;a<b;a++)

#define MAXN  100002
#define INF 1<<28

using namespace::std;

int pos[MAXN],b[MAXN],n,a;
multiset<int> s;

int main(){
    scanf("%d",&n);
    FOR(i,n) scanf("%d",&a), pos[a] = i;
    FOR(i,n) scanf("%d",&b[i]), s.insert(i-pos[b[i]]);
    FOR(i,n){
        int mini = INF;
        multiset<int>::iterator it = s.upper_bound(i);
        if(it != s.end()) mini = min(mini,*it-i);
        if(it != s.begin()) mini = min(mini, i-*(--it)); 
        printf("%d\n",mini);
        s.erase(s.find(i-pos[b[i]]));
        s.insert(i+n-pos[b[i]]);
    }
    return 0;
}
