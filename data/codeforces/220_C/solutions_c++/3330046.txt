#include <iostream>
#include <set>
#include <algorithm>

#define rep(i,n) for(int i=0;i<n;i++)
#define INF      100000000

using namespace std;

multiset<int> ms;
int n,a[100010],b[100010],t1;

int main(){
  cin>>n;
  rep(i,n){cin>>t1;t1--;a[t1] = i; }
  rep(i,n){
    cin>>b[i];b[i]--;
    ms.insert(i-a[b[i]]);
  }
  rep(i,n){
    int res = INF;
    multiset<int>::iterator it = ms.lower_bound(i);
    if(it!=ms.end())res = min(res,*it-i);
    if(it!=ms.begin())res = min(res,i-(*(--it)));
    ms.erase(ms.find(i-a[b[i]]));
    ms.insert(n+i-a[b[i]]);
    cout<<res<<endl;
  }


}
