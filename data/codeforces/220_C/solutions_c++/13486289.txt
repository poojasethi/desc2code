#include <bits/stdc++.h>
using namespace std;

int N, a[100010], b[100010], posA[100010];
multiset <int> S;
multiset <int> :: iterator it, jt;

main()
{
    scanf("%d", &N);
    for (int i = 1; i <= N; i++) scanf("%d", &a[i]), posA[a[i]] = i;
    for (int i = 1; i <= N; i++) scanf("%d", &b[i]), S.insert(i - posA[b[i]]);
    S.insert(-1e9);
    for (int x = 1; x <= N; x++)
    {
        it = jt = S.lower_bound(x - 1);
        printf("%d\n", min(abs(*it - x + 1), abs(*--jt - x + 1)));
        S.erase(S.find(x - posA[b[x]]));
        S.insert(x - posA[b[x]] + N);
    }
}
