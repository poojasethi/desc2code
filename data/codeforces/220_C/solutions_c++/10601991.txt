#include <stdio.h>
#include <iostream>
#include <set>
#include <algorithm>
using namespace std;
#define INF 99999999
multiset <int> dist;
int n;
int hash[100010];
int b[100010];
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
		{
			int x;
			cin>>x;
			hash[x]=i;
		} 
	for(int i=0;i<n;i++)
		{
			cin>>b[i];
			dist.insert(i-hash[b[i]]);
		}
	int ans;
	for(int i=0;i<n;i++)
		{
			ans=INF;
			multiset<int>::iterator it=dist.lower_bound(i);
			if(it!=dist.end())
				{
					ans=min(ans,*it-i);
				}
			if(it!=dist.begin())
				{
					ans=min(ans,i-*(--it));
				}
			cout<<ans<<endl;
			dist.erase(dist.find(i-hash[b[i]]));
			dist.insert((i+n)-hash[b[i]]);
			
		}
}
