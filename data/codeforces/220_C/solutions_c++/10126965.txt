#include<cstdio>
#include<algorithm>
#include<set>
#define nn 100010
using namespace std;
multiset<int>M;
int n,i,l,r,h;
int a[nn],f[nn],b[nn],d[nn];

int main()
{
 scanf("%d",&n);
 for (i=1;i<=n;i++)scanf("%d",&a[i]),f[a[i]]=i;
 for (i=1;i<=n;i++)scanf("%d",&b[i]);
 for (i=1;i<=n;i++)d[i]=f[b[i]]-i,M.insert(d[i]);
 M.insert(-n*2);M.insert(n*2);
 for (i=0;i<n;i++){
   l=*(--M.upper_bound(-i));
   r=*M.upper_bound(-i);
   h=min(abs(l+i),abs(r+i));
   printf("%d\n",h);
   M.erase(M.find(d[i+1]));d[i+1]=d[i+1]-n;M.insert(d[i+1]);
                   }
 return 0;
}
