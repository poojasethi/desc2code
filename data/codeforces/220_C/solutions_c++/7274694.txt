#include<cstdio>
#include<cstdlib>
#include<set>
using namespace std;
multiset <int> S;
int n,a[100005],b[100005];
int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;i++){
		int x;
		scanf("%d",&x);
		a[x]=i;
	}
	for (int i=1;i<=n;i++){
		scanf("%d",&b[i]);
		S.insert(i-a[b[i]]);
	}
	for (int i=0;i<n;i++){
		int ans=0x7f7f7f7f;
		set<int> :: iterator it=S.lower_bound(i);
		if(it!=S.end()){ans=min(ans,(*it)-i);}
		if(it!=S.begin()){ans=min(ans,i-(*(--it)));}
		S.erase(S.find(i+1-a[b[i+1]]));
		S.insert(i+1-a[b[i+1]]+n);
		printf("%d\n",ans);
	}
}
