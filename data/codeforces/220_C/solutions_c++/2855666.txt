#include<set>
#include<cstdio>
#include<algorithm>
#define N 100010
using namespace std;
multiset<int> s;
int n, a[N], b[N], i, j, ans;
int main() {
	scanf("%d", &n);
	for (i = 0; i < n; i++) {
		scanf("%d", &j);
		a[j] = i;
	}
	for (i = 0; i < n; i++) {
		scanf("%d", &b[i]);
		s.insert(i - a[b[i]]);
	}
	for (i = 0; i < n; i++) {
		ans = n + 1;
		multiset<int>::iterator it = s.lower_bound(i);
		if (it != s.end())
			ans = min(ans, (*it - i));
		if (it != s.begin())
			ans = min(ans, (i - *(--it)));
		printf("%d\n", ans);
		s.erase(s.find(i - a[b[i]]));
		s.insert((n + i - a[b[i]]));
	}
}
