#include <bits/stdc++.h>
using namespace std;
int n,a,pos[100010],b[100010];
multiset<int>s;
multiset<int>::iterator x;
int main()
{
   //freopen("r.txt","r",stdin);
   scanf("%d",&n);
   for(int i=0;i<n;i++)
   {
       scanf("%d",&a);
       pos[a]=i;
   }
   for(int i=0;i<n;i++)
   {
       scanf("%d",&b[i]);
       s.insert(i-pos[b[i]]);
   }
   for(int i=0;i<n;i++)
   {
       x=s.lower_bound(i);
       int pr=n;
       if(x!=s.end())
          pr=min(pr, (*x)-i);
       if(x!=s.begin())
          x--, pr=min(pr, i-(*x));
       x=s.find(i-pos[b[i]]);
       s.erase( x );
       s.insert(n-pos[b[i]]+i);
       printf("%d\n",pr);
   }

    return 0;
}
