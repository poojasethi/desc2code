#include<cstdio>
#include<algorithm>
#include<set>

using namespace std;

const int nmax=100100;

int abs(int x){
  return x>0?x:-x;
}

int n;
int pos[nmax],razlika[nmax];
int a[nmax],b[nmax];
multiset<int> S;
multiset<int>::iterator it;

int main(){
scanf("%d",&n);
for(int i=0;i<n;++i){ scanf("%d",&a[i]); pos[a[i]]=i; }
for(int i=0;i<n;++i){
  scanf("%d",&b[i]);
  S.insert(i-pos[b[i]] );
  razlika[i]=i-pos[b[i]];
}

for(int i=0;i<n;++i){
  it=S.lower_bound(i);
  it--;
  int lo=abs(i-(*it));
  it++;
  int hi=abs(i-(*it));
  printf("%d\n",min(lo,hi));
  it=S.lower_bound(razlika[i]);
  S.erase(it);
  S.insert(razlika[i]+n);
}


return 0;
}
