#include <iostream>
using namespace std;
int main() {
int n,m;
char k,l,c='a';
cin>>n>>m;
for(int i=0;i<n;i++) { 
    cin>>k;
    if(c!=k) {
        for(int j=1;j<m;j++) {
            cin>>l;
            if(k!=l) {cout<<"NO"; return 0;}
            }
        c=k; }
    else {cout<<"NO"; return 0;}}
cout<<"YES";
return 0;
}
