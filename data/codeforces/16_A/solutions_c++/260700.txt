#include <iostream>
using namespace std;

int n,m;
char c;
string s;
bool ok = true;

int main()
{
	cin >> n >> m;
	c = 'z';
	while(n--)
	{
		cin >> s;
		if(c == s[0])
			ok = false;
		c = s[0];
		for(int i = 1; i < m; i++)
			if(s[i] != c)
				ok = false;
	}
	if(ok)
		cout << "YES" << endl;
	else
		cout << "NO" << endl;
	return 0;
}
