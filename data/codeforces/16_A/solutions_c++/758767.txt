#include<iostream>
using namespace std;
int main(){
  int n,m,f=0;
  cin>>n>>m;
  string t[111];
  for(int i=0;i<n;i++)
    cin>>t[i];
  for(int i=0;i<n-1;i++)
    if(t[i]==t[i+1])f=1;
  for(int i=0;i<n;i++)
    for(int j=0;j<m-1;j++)
      if(t[i][j]!=t[i][j+1])f=1;
  if(f)cout<<"NO"<<endl;
  else cout<<"YES"<<endl;
  return 0;
}
