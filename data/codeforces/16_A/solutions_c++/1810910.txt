// Codeforces 16A
#include <cstdio>

int main() {
	int n, m; scanf("%d %d", &n, &m);
	char pre=0;
	bool fail=false;
	for(int i=0; i<n; i++) {
		char x; scanf(" %c", &x);
		for(int j=1; j<m; j++) {
			char y; scanf(" %c", &y);
			if(y!=x) fail=true;
		}
		if(x==pre) fail=true;
		pre=x;
	}
	if(fail==true) printf("NO\n");
	else printf("YES\n");
}
