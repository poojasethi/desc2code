#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,i,j;
	char c,p='a';
	string s;
	cin>>n>>m;
	for(i=0;i<n;i++)
	{
		cin>>s;
		c=s[0];
		if(p==c)
		{
			cout<<"NO\n";
			return 0;
		}
		for(j=1;j<m;j++)
		{
			if(s[j]!=c)
			{
				cout<<"NO\n";
				return 0;
			}
		}
		p=c;
	}
	cout<<"YES\n";
	return 0;
}
