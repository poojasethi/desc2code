#include<iostream>
using namespace std;
int n,m,p=1;
string a,b;
int main(){
	cin>>n>>m;
	for(int i=0;i<n;i++){
		cin>>a;
		if(b[0]==a[0]){
			p=0;
		}
		else{
			b[0]=a[i];
		}
		for(int j=0;j<m;j++){
			if(a[0]!=a[j]){
				p=0;
			}
		}
	}
	if(p==1){
		cout<<"YES"<<endl;
	}
	else{
		cout<<"NO"<<endl;
	}
	return 0;
}
