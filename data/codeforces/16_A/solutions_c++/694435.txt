#include<iostream>
using namespace std;

int m, n,i,j;
char ch, d1, d0 = '?';
int main(){
    cin>>m>>n;
    for(i=1;i<=m;i++){
      cin>>d1;
      if(d1==d0){cout<<"NO";return 0;}      
      d0 = d1;
      for(j=2;j<=n;j++){
        cin>>ch;
        if(ch!=d1) {cout<<"NO";return 0;}
      }     
    }
    cout<<"YES";
    return 0;
}
