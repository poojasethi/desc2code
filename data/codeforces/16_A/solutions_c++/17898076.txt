#include<stdio.h>
int main(){
	int n,m;
	while(scanf("%d%d",&n,&m)!=EOF){
		int flag=0;
		char ch;
		for(int i=0;i<n;i++){
			char s[110];
			scanf("%s",s);
			if(i&&ch==s[0]) flag=1;
			ch=s[0];
			for(int i=1;i<m;i++){
				if(ch!=s[i]) flag=1;
			}
		}
		if(flag) printf("NO\n");
		else printf("YES\n");
	}
	return 0;
}
