#include <cstdio>

char tmp[1024];

bool check() {
	int n, m;
	char prev = 0;
	scanf("%d %d", &n, &m); gets(tmp);
	for (int i = 0; i < n; i++) {
		gets(tmp);
		for (int j = 1; j < m; j++) if (tmp[j] != tmp[0]) return false;
		if (tmp[0] == prev) return false;
		prev = tmp[0];
	}
	return true;
}

int main() {
	printf("%s\n", (check() ? "YES" : "NO"));
}
