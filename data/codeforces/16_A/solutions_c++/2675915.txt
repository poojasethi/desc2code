#include <iostream>
using namespace std;

int main(){
string a="YES";
int temp=-1;
string num;
int r,c;
cin>>r>>c;
while(r--){
cin>>num;
if(temp==(num[0]-48))
{
a="NO";
continue;
}
temp=(num[0]-48);
for(int i=0;i<c;i++)
{

if(temp!=(num[i]-48))
{
a="NO";
break;
}
temp=(num[i]-48);
}
}
cout<<a;
return 0;}
