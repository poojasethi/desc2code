#include<cstdio>
int main()
{
	char cch,prch,pch;
	int n,m,i,j;
	scanf("%d %d%*c",&n,&m);
	for (i=0;i<n;i++)
	{
		for (j=0;j<m;j++)
		{
			scanf("%c",&cch);
			if (j&&(cch!=pch))
			{
				puts("NO");
				return 0;
			}
			if (i&&(cch==prch))
			{
				puts("NO");
				return 0;
			}
			pch=cch;
		}
		prch=cch;
		scanf("%*c");
	}
	puts("YES");
	return 0;
}
