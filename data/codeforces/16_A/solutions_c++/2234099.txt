#include <iostream>
using namespace std;
int main(){
	int n; cin >> n;
	int m; cin >> m;
	char c[n][m];
	int a,i,j; a=1;
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			cin >> c[i][j];
			if (j>0 && c[i][j]!=c[i][j-1]){a=0; break;}
			if (i>0 && c[i][j]==c[i-1][j]){a=0; break;}
		}
	}
	if(a){cout << "YES";}else{cout << "NO";}
return 0;}
