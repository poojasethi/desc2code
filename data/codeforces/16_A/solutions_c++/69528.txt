#include <cstdio>

int n,m,now,pre;

bool can;

char buf[200];

int main(){
	while(~scanf("%d%d\n", &n, &m)){
		pre=-1;
		can=1;
		while(n--){
			gets(buf);
			now= buf[0];
			if(now == pre)can=0;
			for(int i=0; buf[i]; i++)
				if(buf[i] != now)
					can=0;
			pre=now;
		}
		puts(can ? "YES" : "NO");
	}
}
