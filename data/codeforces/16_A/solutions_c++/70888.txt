#include <cstdio>

int main() {
	int n, m, i, j;
	char maj[100][100];
	scanf("%d%d", &n, &m);
	for(i = 0; i < n; ++i) {
		getchar();
		for(j = 0; j < m; ++j) {
			maj[i][j] = getchar();
			if(j > 0 && maj[i][j - 1] != maj[i][j] || i > 0 && maj[i - 1][j] == maj[i][j]) {
				puts("NO");
				return 0;
			}
		}
	}
	puts("YES");
	return 0; }
