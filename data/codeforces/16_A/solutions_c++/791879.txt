#include<iostream>
using namespace std;
int main(){
	int m,n;
	cin>>m>>n;
	char arr[m][n];
	bool k=true;
	for(int i=0;i<m;i++){
		cin>>arr[i];
		if(i&&arr[i-1][n-1]==arr[i][1])
			k=false;
	}
	for(int i=0;i<m;i++)
		for(int j=1;j<n;j++)
			if(arr[i][j]!=arr[i][j-1]){
				k=false;
				break;
			}
	if(k)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;
}
