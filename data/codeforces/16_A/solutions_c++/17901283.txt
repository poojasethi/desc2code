#include <iostream>
#include <cstring>
using namespace std;
char a[101][101];
int main(){
	int n,m;
	cin>>n>>m;
	string c="YES";
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
		{	
			cin>>a[i][j];
				if(i!=0&&a[i][j]==a[i-1][j])
				c="NO";
				if(j!=0&&a[i][j]!=a[i][j-1])
				c="NO";
			}
		
	cout<<c;
	return 0;
}
