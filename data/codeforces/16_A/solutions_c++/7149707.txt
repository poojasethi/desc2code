#include <iostream>
using namespace std;
int main()
{
	int n, m, i, j;
	cin >> n >> m;
	char a[101][101];
	for(i=1; i<=n; i++){
		cin >> a[i];
		for (j=1;j<m;j++){
			if(a[i][j]!=a[i][j-1])
			{
				cout<<"NO";
				return 0;
			}
    	}
    	if(a[i][0]==a[i-1][0])
		{
			cout<<"NO";
			return 0;
		}
	}
	cout<<"YES";
	return 0;
}
