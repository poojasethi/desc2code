#include <iostream>
using namespace std;
int main()
{
    //input;output;
    int n,m,i,j;char r='A';string s;bool ok=true;
    cin>>n>>m;
    for (i=0;i<n;i++)
    {
        cin>>s;
        for (j=0;j<m;j++) if ((s[j]==r)||(s[j]!=s[0])) ok=false;
        r=s[0];
    }
    if (ok) cout<<"YES"; else cout<<"NO";
    return 0;
}
