#include <iostream>
#include <cstdlib>
using namespace std;
string s;
int main() {
	int n,m;
	cin >> n >> m;
	char c = '*';
	for(int i=0;i<n;i++) {
		cin >> s;
		for(int j=0;j<s.size();j++)
			if((s[j] != s[0]) || (c != '*' && c == s[j])) {cout << "NO" << endl; exit(0);}
		c = s[0];
	}
	cout << "YES" << endl;
	return 0;
}
