#include <iostream>
using namespace std;
int main()
{	
	int n, m, i;
	string s, t=".";
	cin>>n>>m;
	while (n--) {
		cin>>s;
		for (i=1; i<m; i++) if (s[i]!=s[i-1]) {
			cout<<"NO"<<endl; return 0;
		}
		if (s[0]==t[0]) {
			cout<<"NO"<<endl; return 0;
		}
		t[0]=s[0];
	}
	cout<<"YES"<<endl;
	return 0;
}
