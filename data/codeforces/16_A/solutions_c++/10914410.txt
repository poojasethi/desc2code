#include <iostream>
#include<string>
using namespace std;
int main()
{int n,m;cin>>n>>m;string x[n];bool q=1;
    for(int i=0;i<n;i++){cin>>x[i];if (i!=0 && x[i][0]== x[i-1][0]){q=0;}
    else{for (int j=0 ; j<m ; j++){if (j!=m-1 && x[i][j]!= x[i][j+1]){q=0;}}}}
    if(q){cout<<"YES";}else{cout<<"NO";}
    return 0;}
