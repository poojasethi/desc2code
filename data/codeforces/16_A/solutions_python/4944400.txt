n, m = map(int, raw_input().split())

preColor = -1
flag = True

for _ in range(n) :
	colors = raw_input()
	rowColor = colors[0]
	if int(rowColor) == preColor :
		flag = False
		break
	else :
		preColor = int(rowColor)

	for c in colors :
		if c != rowColor :
			flag = False
			break
if flag :
	print 'YES'
else :
	print "NO"
