n, m = [int(x) for x in raw_input().split()]
flag = []
for x in xrange(n):
    tp = raw_input()
    a = tp[0]
    for y in tp:
        if y != a:
            print 'NO'
            exit()
    flag += [tp]
for x in xrange(n - 1):
    if flag[x][0] == flag[x+1][0]:
        print 'NO'
        exit()
print 'YES'
