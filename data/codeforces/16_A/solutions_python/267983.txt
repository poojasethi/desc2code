def main():
    n,m = map(int,raw_input().split(' '))
    flag = [raw_input() for i in xrange(n)]
    old = -1
    for line in flag:
        new = line[0]
        if new == old or line != new*m:
            print 'NO'
            return
        old = new
    print 'YES'
if __name__ == "__main__":
	main()
