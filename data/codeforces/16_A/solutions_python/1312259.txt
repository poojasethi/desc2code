def solve():
    n,m = map(int, str(raw_input()).split())
    previous = -1
    for j in range(n):
        i = raw_input().strip()
        if len(set(list(i)))!=1:
            print 'NO'
            return
        if i[0]==previous:
            print 'NO'
            return
        previous = i[0]
    print 'YES'
