N, M = [int(x) for x in raw_input().split()]

S = []
for i in range(N):
    S.append(raw_input())

ok = True
prev = ''
for line in S:
    if line[0] * M != line:
        ok = False
    if prev and prev[0] == line[0]:        
        ok = False
    prev = line

if ok :
    print 'YES'
else:
    print 'NO'
