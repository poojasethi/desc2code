n,m=map(int,raw_input().split(" "))
flag=[]
a=0
for x in range(0,n):
    flag.append(raw_input())
if flag[0]==flag[0][0]*m:
    for x in range(1,n):
        if flag[x]!=flag[x][0]*m or flag[x]==flag[x-1]:
            a=1
            break
    if a==1:
        print "NO"
    else:
        print "YES"
else:
    print "NO"
