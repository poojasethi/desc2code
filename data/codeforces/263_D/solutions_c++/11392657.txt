using namespace std;
vector<int> e[100010];
int ans[100010],n,m,k,top,st,ed,Dfs[100010];
bool dfs(int u)
{
	Dfs[u]=++top;
	ans[top]=u;
	for(int i=0;i<e[u].size();i++)
	{
		int v=e[u][i];
		if(Dfs[v])
		{
			if(top-Dfs[v]<k) continue;
			st=Dfs[v];
			ed=Dfs[u];
			return true;
		}
		if(dfs(v))
			return true;
	}
	return false;
}

int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<m;i++)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		e[x].push_back(y);
		e[y].push_back(x);
	}
	dfs(1);
	printf("%d\n",ed-st+1);
	for(int i=st;i<=ed;i++)
		printf("%d ",ans[i]);
	printf("\n");
	return 0;
}
