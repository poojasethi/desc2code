#include <iostream>
#include <vector>
#include <cstdio>
using namespace std;

const int oo = 1e8;

vector<int> adj[100 * 1000], p;
int n, m, h[100 * 1000], k;

void dfs(int v, int c);

int main() {
	scanf("%d%d%d", &n, &m, &k);
	for (int i = 0; i < m; i++) {
		int u, v;
		scanf("%d%d", &u, &v);
		u--, v--;
		adj[u].push_back(v);
		adj[v].push_back(u);
	}
	fill(h, h + n, oo);
	dfs(0, 0);
	return 0;
}
void dfs(int v, int c) {
	h[v] = c;
	p.push_back(v);
	for (int i = 0; i < adj[v].size(); i++) {
		int u = adj[v][i];
		if (h[v] - h[u] >= k) {
			printf("%d\n", h[v] - h[u] + 1);
			for (int i = h[u]; i < p.size(); i++)
				printf("%d ", p[i] + 1);
			printf("\n");
			break;
		}
		else if (h[u] == oo) {
			dfs(u, c + 1);
			break;
		}
	}
}
