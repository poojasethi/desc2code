#include <iostream>
#include <vector>
#include <cstdio>
using namespace std;

int n,m,K,x,y,v[101000],en,node,len,Q[101000],pre[101000];
vector <int> e[101000];

int main()
{
	//freopen("in.txt","r",stdin);

	scanf("%d%d%d",&n,&m,&K);
	for (int i=1;i<=m;i++)
	{
		scanf("%d%d",&x,&y);
		e[x].push_back(y);e[y].push_back(x);
	}
	v[1]=1;Q[en=1]=1;
	for (int st=1;st<=en;st++)
	{
		for (int i=0;i<e[Q[st]].size();i++)
		{
			node=e[Q[st]][i];
			if (v[node])
			{
				len=v[Q[st]]-v[node];
				if (len>=K)
				{
					printf("%d\n",len+1);
					for (int j=Q[st];j!=node;j=pre[j]) printf("%d ",j);
					printf("%d\n",node);
					return 0;
				}
			}else { Q[++en]=node;v[node]=v[Q[st]]+1;pre[node]=Q[st];break; }
		}
	}
	
	return 0;
}
