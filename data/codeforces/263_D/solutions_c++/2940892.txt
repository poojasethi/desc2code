#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
#include<vector>
#include<math.h>
#include<queue>
#include<set>
using namespace std;
typedef long long ll;

#define N 100010

vector<int> e[N],qu,ans;

int vis[N],tt=0;

void dfs(int u)
{
	qu.push_back(u);
	vis[u]=++tt;
	int t=N;
	for(int i=0;i<(int)e[u].size();i++){
		int v=e[u][i];
		if(vis[v]){
			t=min(vis[v],t);
			continue;
		}
		else{
			dfs(v);
			return;
		}
	}
	int i;
	for(i=0;i<(int)qu.size();i++){
		if(vis[qu[i]]==t)
			break;
	}
	printf("%d\n",qu.size()-i);
	for(;i<(int)qu.size();i++){
		printf("%d ",qu[i]);
	}
}
int main()
{
	int n,m,k,x,y;
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<m;i++){
		scanf("%d%d",&x,&y);
		e[x].push_back(y);
		e[y].push_back(x);
	}
	dfs(1);
}
