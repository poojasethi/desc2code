#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;

vector<int>adj[100001];
int d[100001],par[100001],ans,md;
void dfs(int node,int deg){
	d[node]=deg;
	for(int i=0;i<adj[node].size();i++)
		if(!d[adj[node][i]]){
			par[adj[node][i]]=node;
			dfs(adj[node][i],deg+1);
		}
	if(deg>md)
		md=deg,ans=node;
	return;
}
int main() {
	int n,m,k,a,b,s;
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<m;i++){
		scanf("%d%d",&a,&b);
		adj[a].push_back(b);
		adj[b].push_back(a);
	}
	dfs(1,1);
	for(int i=0;i<adj[ans].size();i++)
		if(d[ans]-d[adj[ans][i]]>=k){
			printf("%d\n",d[ans]-d[adj[ans][i]]+1);
			s=adj[ans][i];
			break;
		}
	while(ans!=s){
		printf("%d ",ans);
		ans=par[ans];
	}
	printf("%d",s);
	return 0;
}
