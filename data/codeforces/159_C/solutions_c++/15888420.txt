#include <bits/stdc++.h>
#define p push_back
using namespace std;
vector < int > v[27];
int main()
{
int k;
while(cin>>k)
{
    string s;
    cin>>s;
    int sz=s.size();
    for(int j=0;j<k;++j)
    for(int i=0;i<sz;++i)
    {
        v[s[i]-'a'].p(i+j*sz);
    }
    int n;
    cin>>n;
    bool vis[200005]={0};
    while(n--)
    {
        int x;char c;
        cin>>x>>c;
        vis[v[c-'a'][x-1]]=1;
        v[c-'a'].erase(v[c-'a'].begin()+x-1);

    }
    for(int i=0;i<k*sz;++i)
        if(!vis[i])cout<<s[i%sz];

    cout<<endl;


}


}
