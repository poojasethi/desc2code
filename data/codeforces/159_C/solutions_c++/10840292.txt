# include <bits/stdc++.h>
using namespace std;
# define fi cin
# define fo cout
vector < int > dp[26];
bitset < 200005 > b;
string c,p;
int main(void)
{//
    ios_base :: sync_with_stdio(0);
    int a,t,x;
    char y;
    fi>>a>>c>>t;p = "";
    int n = c.length();
    for (int i = 1;i <= a;++i) p += c;
    n *= a;
    for (int i = 0;i < n;++i) dp[p[i] - 'a'].push_back(i);
    while (t --)
    {
        fi>>x>>y;
        y -= 'a';--x;
        b[dp[y][x]] = 1;
        dp[y].erase(dp[y].begin()+x);
    }
    for (int i = 0;i < n;++i) if (!b[i]) fo << p[i];
    return fo << '\n',0;
}
