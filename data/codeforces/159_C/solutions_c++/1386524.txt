#include <iostream>
#include <string>
#include <vector>

using namespace std;

int K,N,check[200010];
string S;
vector<int> C[30];

int main()
{
	string s;
	cin >> K >> s;
	for (int i=0;i<K;i++)
		S+=s;
	for (int i=0;i<S.size();i++)
		C[S[i]-'a'].push_back(i);
	cin >> N;
	for (int i=0;i<N;i++)
	{
		int p;
		char c;
		vector<int>::iterator it;
		cin >> p >> c;
		it=C[c-'a'].begin();
		check[*(it+p-1)]=1;
		C[c-'a'].erase(it+p-1);
	}
	for (int i=0;i<S.size();i++)
		if (!check[i])
			cout << S[i];
	cout << endl;
	return 0;
}
