#include <bits/stdc++.h>
using namespace std;
int main ()
{
int n;cin>>n;n=n-1;
map<char,vector<int> > m;
string str,s;

cin>>str; s=str;
char y;
int x;
for(int i=0;i<n;i++)
{
    str+=s;
}
for(int i=0;i<str.size();i++)
{
 m[str[i]].push_back(i);
}

int t;
cin>>t;
while(t--)
{
    cin>>x>>y;
    int u=m[y][x-1];
    str[u]='#';
    m[y].erase(m[y].begin()+ x-1 );
}
for(int i=str.size()-1;i>=0;i--)
{
    if(str[i]=='#')
        str.erase(i,1);
}
cout<<str;
}
