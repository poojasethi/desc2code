#include<bits/stdc++.h>
using namespace std;

vector<vector<int> > vec(26);
vector<int>::iterator it;
int main() {
    int k;
    string s,str="";
    cin >> k >> s;
    for(int i=0;i<k;i++)str+=s;
    int str_size = str.size();
    for(int i=0;i<str_size;i++)vec[str[i]-'a'].push_back(i);
    int q;
    cin>>q;
    while(q--){
        int a;
        char c;
        cin>>a>>c;
        str[vec[c-'a'][a-1]] = ' ';
        it = vec[c-'a'].begin()+a-1;
        vec[c-'a'].erase(it);
    }
    for(int i=0;i<str_size;i++)if(str[i]-' ')cout<<str[i];
}
