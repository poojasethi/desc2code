#include <iostream>
#include <set>
#include <vector>
using namespace std;

int main(){
	int n,m,k,d;
	vector <vector <int> > v(26, vector <int>(0,0));
	string c,s,ss="";
	cin>>k>>s;
	while(k--)	ss=ss+s;
	for(int i=0;i<ss.size();i++)
		v[int(ss[i]-'a')].push_back(i);
	cin>>n;
	set<int> st;
	for(int i=0;i<n;i++){
		cin>>d>>c;
		st.insert(v[c[0]-'a'][d-1]);
		v[c[0]-'a'].erase(v[c[0]-'a'].begin()+d-1);
	}
	for(int i=0;i<ss.size();i++)
		if (!st.count(i)) cout<<ss[i];
	cout<<endl;
	return 0;
}
