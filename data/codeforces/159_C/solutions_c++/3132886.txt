#include<iostream>
#include<string>
#include<vector>
using namespace std;
int K,N,h[200010],i,p;
string S;
vector<int>C[30];
main(){
	string s;
	cin>>K>>s;
	for(i=0;i<K;i++)
		S+=s;
	for(i=0;i<S.size();i++)
		C[S[i]-'a'].push_back(i);
	cin>>N;
	for(i=0;i<N;i++){
		char c;
		vector<int>::iterator it;
		cin>>p>>c;
		it=C[c-'a'].begin();
		h[*(it+p-1)]=1;
		C[c-'a'].erase(it+p-1);
	}
	for(i=0;i<S.size();i++)
		if(!h[i])
			cout<<S[i];
}
