#include <iostream>
#include <algorithm>
#include <vector>
#include <set>
#include <queue>
#define jyn cin
#define pb push_back
using namespace std;
priority_queue<int> pq;
vector < int > vec[50];
string S,s2;
int N;
int main()
{
	int x;
	char t;
	cin>>N>>s2;
	S="";
	while(N--)S+=s2;
	for(int i=0;i<S.size(); i++)
	vec[S[i]-'a'].pb(i);
	cin>>N;
	while(N--)
	{
		jyn>>x>>t;
		t-='a';
		vec[t].erase(vec[t].begin()+x-1);
	}
	for(int i=0;i<45;i++)
		for(int j=0;j<vec[i].size();j++)
			pq.push(-1*vec[i][j]);
	while(!pq.empty()){ cout<< S[-1*pq.top()]; pq.pop();} cout<<endl;
	return 0;
}
