#include <bits/stdc++.h>
using namespace std;
string s, t;
int k, n, p, used[200001];
char c;
map<char, vector<int> > m;

int change()
{
	int idx = m[c][p - 1];
	m[c].erase(m[c].begin() + p - 1);
	return idx;
}

void solve()
{
	for(int i = 0; i < t.size(); i++)
	{
		m[t[i]].push_back(i);
	}

	cin >> n;

	while(n--)
	{
		cin >> p >> c;
		int idx = change();
		used[idx] = 1;
	}

	for(int i = 0; i < t.size(); i++)
	{
		if(!used[i]) cout << t[i];
	}

}

void read()
{
	cin >> k;
	cin >> s;
	for(int i = 0; i < k; i++) t += s;

}

int main()
{
	read();
	solve();
}
