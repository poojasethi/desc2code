{int k,N,i,j,n,L;
 cin>>k;
 string S;
 cin>>S;
 vector<vector<int> >V(26);
 set<int>Di;
 for(L=j=0;j<k;j++)
 for(i=0,N=S.size();i<N;i++,L++)
 	V[S[i]-'a'].push_back(L);
 cin>>n;
 while(n--)
 	{int p,q; char c;
	 cin>>p>>c;
	 q=V[c-'a'][p-1];
	 Di.insert(q);
	 V[c-'a'].erase(V[c-'a'].begin()+p-1);
	}
 for(L=j=0;j<k;j++)
 for(i=0,N=S.size();i<N;i++,L++)
