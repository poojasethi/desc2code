#include <iostream>
#include <vector>

using namespace std;

int main() {
	int k;
	vector <int> v[30];
	int a[2000000];
	cin >> k;
	string s, str;
	cin >> str;
	s = "";
	for (int i = 0; i < k; i++) {
		s += str;
	}

	for (int i = 0; i < s.size(); i++) {
		v[s[i]-'a'].push_back(i+1);
	}
	
	int n, p;
	cin >> n;

	for (int i = 0; i < n; i++) {
		char c;
		cin >> p >> c;

		vector <int> :: iterator it;
		it = v[c - 'a'].begin();
		a[*(it+p-1) - 1] = 1;
//		cout << *(it+p-1) << endl;
		v[c - 'a'].erase(it+p-1);
	}
	for (int i = 0; i < s.size(); i++) {
		if (a[i] != 1) {
			cout << s[i];
		}
	}
	cout << endl;
	return 0;	
}
