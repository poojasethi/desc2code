#include<bits/stdc++.h>
using namespace std;

vector <int> vec[300];
int k,n,a;
char c;
string str,str2;

int main(){
	cin >> k >> str >> n;
	str2=str;
	for(int i=2; i<=k ; i++)
		str+=str2;

	for(int i=0 ; i<str.size() ; i++)
		vec[str[i]].push_back(i);

	for(int i=1;i<=n;i++){
		cin >> a >> c ;
		str[vec[c][a-1]]='.';
		vec[c].erase(vec[c].begin()+a-1);
	}
	for(int i=0;i<str.size();i++)
		if(str[i]!='.')
			cout << str[i];

	cout <<endl;
}
