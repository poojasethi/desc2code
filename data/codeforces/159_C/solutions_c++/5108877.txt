#include <iostream>
#include <cstdio>
#include <string>
#include <vector>
using namespace std;
bool checked[200005];
vector<int> occ[26];
int main()
{
    int k,n;
    string ss,s="";
    cin>>k>>ss>>n;
    for (int i=0; i<k; i++)
        s+=ss;
    for (int i=0; i<s.size(); i++)
        occ[s[i]-97].push_back(i);
    for (int i=0; i<n; i++)
    {
        int num;
        char c;
        cin>>num>>c;
        checked[occ[c-97][num-1]]=1;
        occ[c-97].erase(occ[c-97].begin()+num-1);
    }
    for (int i=0; i<s.size(); i++)
        if(!checked[i]) cout<<s[i];
    cout<<endl;
    return 0;
}
