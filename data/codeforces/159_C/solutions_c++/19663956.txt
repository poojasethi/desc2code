#include <iostream>
#include <string>
#include <vector>
using namespace std;
char ch;
int k,n,i;
string s,p;
vector<int> a[26];
int main()
{
    cin>>k>>s>>n;
    for (i=0; i<k; i++) p+=s;
    for (i=0; i<p.size(); i++) a[p[i]-97].push_back(i);
    for (i=0; i<n; i++) {
        cin>>k>>ch;
        p[a[ch-97][k-1]]='#';
        a[ch-97].erase(a[ch-97].begin()+k-1);
    }
    for(i=0;i<p.size();i++) if(p[i]!='#') cout<<p[i];
}
