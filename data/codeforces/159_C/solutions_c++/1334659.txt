#include <cstdio>
#include <cstring>
#include <iostream>
#include <vector>

using namespace std;


int i, j, k, n, t, p;
char s[200], c[4];
bool o[200000];

int main(){
	vector<int> v[26];
	
	scanf("%d", &k);
	scanf("%s", s);
	
	n=strlen(s);
	j=n*k;
	for(i=0;i<j;++i){
		v[s[i%n]-'a'].push_back(i);
	}
	

	memset(o,1,sizeof o);
	scanf("%d", &t);
	while(t--){
		scanf("%d%s", &p, c);
		o[v[c[0]-'a'][p-1]]=0;
		v[c[0]-'a'].erase(v[c[0]-'a'].begin()+p-1);
	}
	for(i=0;i<j;++i)if(o[i])printf("%c", s[i%n]);
	printf("\n");
	
	
	
	return 0;
}
