#include<bits/stdc++.h>
using namespace std;
int main()
{
    //freopen("a.txt","r",stdin);
    int d,n,i,pos;
    vector<int>v[1000];
    string s,p;
    char c;
    cin>>d>>s>>n;
    for(i=1;i<=d;i++)
        p+=s;
    for(i=0;p[i];i++)
        {
            v[p[i]].push_back(i);
        }

    for(i=1;i<=n;i++)
    {
        cin>>pos>>c;
        p[v[c][pos-1]]='#';
        v[c].erase(v[c].begin()+pos-1);
    }
     for(i=0;p[i];i++)
        {
            if(p[i]!='#')
                cout<<p[i];
        }

return 0;
}
