#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

vector<int> vec[150];
vector < pair<int,char> > res;
int N,M;
string str;

int main()
{
	char c;
	int x,i,j,n;
	cin >> N >> str >> M;
	n=str.size();
	for(i=0;i<N;i++)
		for(j=0;j<n;j++)
			vec[str[j]].push_back(j+i*n);
	for(i=0;i<M;i++) 
	{
		cin >> x >> c;
		vec[c].erase(vec[c].begin()+x-1);
	}
	for(i='a';i<='z';i++)
	{
		n=vec[i].size();
		for(j=0;j<n;j++)
			res.push_back(make_pair(vec[i][j],(char)i));
	}
	sort(res.begin(),res.end());
	n=res.size();
	for(i=0;i<n;i++) cout << res[i].second; cout << endl;
	return 0;
}
