#include <iostream>
#include <bits/stdc++.h>
using namespace std;
string s,sc;
map< char,vector<int> > m;
int main()
{
   int k,n,p;
   char c;
   scanf("%d",&k);
   cin >> s;
   sc=s;
   for(int i=1; i<k; i++) s+=sc;
   for(int i=0; i<s.size(); i++)
   {
       m[s[i]].push_back(i);
   }
   scanf("%d",&n);
   for(int i=0; i<n; i++)
   {
       scanf("%d %c",&p,&c);
       s[m[c][p-1]]='#';
       m[c].erase(m[c].begin()+p-1);
   }
  for(int i=0; i<s.size(); i++)
  {
      if(s[i]!='#') cout << s[i];
  }
    return 0;
}
