#include <cstdlib>
#include <iostream>
#include <string>
#include <set>
#include <vector>

using namespace std;

int main(int argc, char *argv[])
{int k,N,n,i,j,L; char c;
 string S;
 cin>>k>>S>>n;
 N=S.size();
 vector<vector<int> >V(26);
 for(L=i=0;i<k;i++)
    for(j=0;j<N;j++,L++)
    V[S[j]-'a'].push_back(L);
 set<int>P;
 for(i=0;i<n;i++)
    {cin>>j>>c;
     c-='a';
     P.insert(V[c][j-1]);
     V[c].erase(V[c].begin()+j-1);
    }
 for(L=i=0;i<k;i++)
    for(j=0;j<N;j++,L++)
    if(!P.count(L))cout<<S[j];
 
    
   
    return EXIT_SUCCESS;
}
