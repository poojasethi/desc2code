#include <bits/stdc++.h>
using namespace std;
vector<long long> v[200];
string s,final;
int main(){
    int k,n;
    char ci;
    long long pi;
    cin>>k>>s>>n;
    for(int i=0;i<k;i++) final=final+s;
    for(int i=0;final[i];i++) v[final[i]].push_back(i);
    for(int i=0;i<n;i++){
        cin>>pi>>ci;
        final[v[ci][pi-1]]='#';
        v[ci].erase(v[ci].begin()+pi-1);
    }
    for(int i=0;final[i];i++)
        if(final[i]!='#') cout<<final[i];
    return 0;
}
