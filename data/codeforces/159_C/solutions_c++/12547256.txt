#include<bits/stdc++.h>
using namespace std;
int n,i,j,k,l,s,u;
char c;
string str;
vector<int>h[1000];
vector<char>a;
int main()
{
	cin>>n>>str>>k;
	u=str.size();
	for(i=0;i<n;i++)
		for(j=0;j<u;j++)
		{
			a.push_back(str[j]);
			h[str[j]].push_back(i*u+j);
		}
	u*=n;
	while(k--)
	{
		cin>>l>>c;
		a[h[c][l-1]]='0';
		h[c].erase(h[c].begin()+l-1);
	}
	for(i=0;i<u;i++)
		if(a[i]!='0')
			cout<<a[i];
	cout<<endl;
}
