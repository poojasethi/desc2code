#include<stdio.h>
#include<vector>
#include<string.h>
using namespace std;

vector <int> pos[400];
char s[200000], orig[100];

int main()
{
	int k, i, n;
	scanf("%d%s", &k, orig);
	for(i=0;i<k;i++)strcat(s,orig);
	int l = strlen(s);
	for(i=0;i<l;i++)pos[s[i]].push_back(i);
	scanf("%d", &n);
	while(n--)
	{
		int p;
		char c;
		scanf("%d %c", &p, &c);
		p--;
		s[pos[c][p]]='\0';
		pos[c].erase(pos[c].begin()+p);
	}
	for(i=0;i<l;i++)if(s[i]!='\0')printf("%c", s[i]);
	printf("\n");
	return 0;
}
