#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;
vector<int> c[26];
bool f[200006];
char s[110];
int main(){
	int k,n,l,i,j;
	char ch[2];
	while(~scanf("%d",&k)){
		scanf("%s",s);
		l=strlen(s);
		for(j=0;j<k;j++){
			i=0;
			while(s[i]){
				c[s[i]-'a'].push_back(i+j*l);
				i++;
			}
		}
		memset(f,0,sizeof(bool)*l*k);
		scanf("%d",&n);
		while(n--){
			scanf("%d%s",&i,ch);
			i--,j=ch[0]-'a';
			f[c[j][i]]=1;
			c[j].erase(c[j].begin()+i);
		}
		for(j=0;j<k;j++){
			i=0;
			while(s[i]){
				if(!f[i+j*l])printf("%c",s[i]);
				i++;
			}
		}
		puts("");
	}
	return 0;
}
