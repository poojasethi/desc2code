#include <iostream>
#include <string>
#include <stdlib.h>
#include <vector>
int mark[1000000]={0};
using namespace std;
int main()
{
	int k,n;
	string S,str;
	cin >> k;
	cin >> str;
	for(int i=0 ; i < k ;++i)
		S = S + str;
	vector<int> v[30];
	for(int i=0 ; i < S.size() ; ++i)
		v[S[i] - 'a'].push_back(i);
	cin >> n;
	int p;
	char ch;
	int size = S.size();
	char output[size];
	for(int i=0 ; i < n ; ++i)
	{
		cin >> p >> ch;
		vector<int>::iterator it;
		it = v[ch-'a'].begin() + p -1;
		mark[*it] = 1;
		v[ch-'a'].erase(it);
	}
	for(int i=0 ; i < size ; ++i)
	{
		if(mark[i]!=1)
			cout << S[i];
	}
	cout << endl;
	return 0;
}
