#include <iostream>
#include<vector>
#include<string>
using namespace std;
vector<int>c[30];
int main()
{
    int n,k,i,p;
    string s1,s;
    char ch;
    cin>>k>>s1;
    for(i=0;i<k;i++)
    s=s+s1;
    for(i=0;i<s.size();i++)
    c[s[i]-'a'].push_back(i);
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>p>>ch;
        vector<int>::iterator it;
        it=c[ch-'a'].begin();
        s[*(it+p-1)]='1';
        c[ch-'a'].erase(it+p-1);
    }
    for(i=0;i<s.size();i++)
    {
        if(s[i]!='1')
        cout<<s[i];
    }
    return 0;
}
