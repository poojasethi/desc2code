#include<bits/stdc++.h>
using namespace std;
vector <int> v[1000];
int n,k;
string str,arr;
int main ()
{
	int len,p,i,j;
	char c;
	cin>>k>>str>>n;
	len=str.size();
	arr=str;
	for(i=2; i<=k; i++)
		str+=arr;

	for(i=0; i<str.size(); i++)
		v[str[i]].push_back(i);

	while(n--)	
	{
	 	cin>>p>>c;
		str[v[c][p-1]]=0;
		v[c].erase(v[c].begin()+p-1);
	}
	for(i=0; i<str.size(); i++)
		 if(str[i])	cout<<str[i];
}
