#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

vector<int> vec[199];
vector < pair<int,char> > r;
int N,M;

int main()
{
	int x,i,j,n;
	char c;
	string S;
	
	cin >> N >> S >> M;
	
	n=S.size();
	for(i=0;i<N;i++) for(j=0;j<n;j++) vec[S[j]].push_back(j+i*n);
	for(i=0;i<M;i++) 
	{
		cin >> x >> c;
		vec[c].erase(vec[c].begin()+x-1);
	}
	for(i='a';i<='z';i++)
	{
		n=vec[i].size();
		for(j=0;j<n;j++) r.push_back(make_pair(vec[i][j],(char)i));
	}
	sort(r.begin(),r.end());
	n=r.size();
	for(i=0;i<n;i++) cout << r[i].second; 
	cout << endl;
	return 0;
}
