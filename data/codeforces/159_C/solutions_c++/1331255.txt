#include <cstdio>
#include <iostream>
#include <vector>
using namespace std;
vector<int> C[26];
char MS[2000*100+1];
int main(){
    int k,n,o,i,j;
    string s;
    cin>>k>>s>>n;
    char c;
    for (i = 0, o = s.size(); i < o*k; i++)
        C[s[i%o]-'a'].push_back(i+1), MS[i] = s[i%o];
    for (i = 0; i < n; i++)
        scanf("%d %c",&o,&c), c-='a', MS[C[c][o-1]-1] = ' ', C[c].erase(C[c].begin()+(o-1));
    for (i = 0, o = s.size(); i < o*k; i++)
        if (MS[i]!=' ')
            printf("%c",MS[i]);
    cout<<endl;
    //system("PAUSE");
    return 0;
}
