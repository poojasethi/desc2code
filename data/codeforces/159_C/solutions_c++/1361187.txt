#include <iostream>
#include <string>
#include <vector>

using namespace std;

int k, n;
vector <int> a[27];
string s1, s = "";
bool p[200010] = {};

int main(){
    cin >> k >> s1 >> n;
    for (; k; k--) s += s1;

    for (int i = 0; i < s.size(); i++)
        a[s[i] - 'a'].push_back(i);

    for (int i = 0; i < n; i++){
        int h, j = 0;
        string x;
        cin >> h >> x;

        p[a[x[0] - 'a'][h - 1]] = true;
        a[x[0] - 'a'].erase(a[x[0] - 'a'].begin() + h - 1);
    }

    for (int i = 0; i < s.size(); i++)
        if (!p[i]) cout << s[i];
}
