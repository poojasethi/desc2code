# include <iostream>
# include <vector>
# include <string>
# include <stdio.h>
# include <algorithm>

using namespace std;

int a,e;

vector< int > d[26];

string b,c;

int v[200009];

int main()
{
	scanf("%d",&a);
	cin>>b;
	
	for(int i=0;i<a;i++)
	c+=b;
	
	scanf("%d",&e);
	
	for(int i=0;i<c.length();i++)
	d[c[i]-'a'].push_back(i);
	
	for(int i=0;i<e;i++)
	{
		int m;
		char n;
		
		scanf("%d %c",&m,&n);
		
		d[n-'a'].erase(d[n-'a'].begin()+m-1);
	}
	
	for(int i=0;i<26;i++)
	for(int j=0;j<d[i].size();j++)
	v[d[i][j]]=1;
	
	for(int i=0;i<c.length();i++)
	if(v[i]) printf("%c",c[i]);
}
