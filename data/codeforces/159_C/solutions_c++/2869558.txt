#include <iostream>
#include <vector>
#define FOR(i,a,b) for(i=a; i<=b; i++)
#define FOR2(i,n) FOR(i,0,n-1)
#define MAXN 200000
using namespace std;
string str,temp;
vector <int> v[500];
int A[500];
bool kont[MAXN+5];
void solve()
{
	int N,a,i,s;
	char c;
	cin >> N >> temp;
	FOR2(i,N)
		str+=temp;
	FOR(c,'a','z')
		FOR2(i,str.size())
			if(str[i] == c)
				v[c].push_back(i);
	cin >> N;
	FOR(i,1,N)
	{
		cin >> a >> c;
		kont[v[c][a-1]] = true;
		v[c].erase(v[c].begin()+a-1);
	}
	FOR2(i,str.size())
		if(!kont[i])
			cout << str[i];
}
int main()
{
	solve();
	return 0;
}
