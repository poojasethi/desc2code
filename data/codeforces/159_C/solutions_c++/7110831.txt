#include <iostream>
#include <string>
#include <vector>
using namespace std;
char c;
int k,n,i;
string s,p;
vector<int> a[26];
int main()
{
    cin>>k>>s>>n;
    for (i=0; i<k; i++) p+=s;
    for (i=0; i<p.size(); i++) a[p[i]-97].push_back(i);
    for (i=0; i<n; i++) {
        cin>>k>>c;
        p[a[c-97][k-1]]='#';
        a[c-97].erase(a[c-97].begin()+k-1);
    }
    for(i=0;i<p.size();i++) if(p[i]!='#') cout<<p[i];
}
