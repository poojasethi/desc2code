#include<stdio.h>
#include<iostream>
#include<string>
#include<vector>
using namespace std;

int k,n,i;
string s="",s1;
vector<int>a[27];
bool p[200005];

int main()
{
    scanf("%d",&k);
    cin>>s1;
    scanf("%d",&n);
    for(i=0;i<k;i++)
    s+=s1;
    for(i=0;i<s.size();i++)
    a[s[i]-'a'].push_back(i);
    while(n--)
    {
        int q;
        string x;
        scanf("%d",&q);
        cin>>x;
        p[a[x[0]-'a'][q-1]]=true;
        a[x[0]-'a'].erase(a[x[0]-'a'].begin()+q-1);
    }
    for(i=0;i<s.size();i++)
    if(!p[i])
    printf("%c",s[i]);
}
