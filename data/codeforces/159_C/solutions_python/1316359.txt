k = int(raw_input())

name = {}

i = 0
for it in raw_input() * k:
    if it in name:
        name[it].append(i)
    else:
        name[it] = [i]
    i += 1

n = int(raw_input())
for _ in range(0, n):
    count, letter = raw_input().split()
    count = int(count) - 1

    name[letter].pop(count)

result = list(' ' * i)
for letter in name:
    for pos in name[letter]:
        result[pos] = letter

result = filter(lambda x: x != ' ', result)
print (''.join(result))
