n = input()
s = list(raw_input()) * n
a = [[] for _ in xrange(128)]
for i, x in enumerate(s):  
    a[ord(x)].append(i)
    
for _ in xrange(input()):
    x, y = raw_input().split()
    a[ord(y)].pop(int(x)-1)
    
c = [0] * 200001
for i in xrange(26):
    for y in a[ord('a')+i]:
        c[y] = chr(ord('a')+i)
ans = [x for x in c if x]
print ''.join(ans)
