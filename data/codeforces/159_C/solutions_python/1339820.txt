import string

def alpha_index(s): return ord(s)-ord('a')

k = input()
s = raw_input().strip()
source = list(s)*k
L = [[] for i in xrange(26)]
for i,ch in enumerate(source):
    L[alpha_index(ch)].append(i)
n = input()
for i in xrange(n):
    j,a = raw_input().strip().split()
    j= int(j)-1
    a = alpha_index(a)
    source[L[a][j]] = ''
    del L[a][j]
print ''.join(source)
