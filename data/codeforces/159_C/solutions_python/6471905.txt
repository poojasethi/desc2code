from collections import defaultdict

k = int(raw_input())
username = list(raw_input()) * k

ch = defaultdict(list)
for i, c in enumerate(username):
    ch[c].append(i)

n = int(raw_input())
for _ in range(n):
    p, c = raw_input().split()
    p = int(p)
    index = ch[c].pop(p - 1)
    username[index] = ''

print ''.join(username)
