k = int(raw_input())
s_repeat = raw_input()
s = ''
for i in range(k):
    s += s_repeat
n = int(raw_input())
operations = {}
for i in range(n):
    input = raw_input().split()
    remove_index = int(input[0])
    letter = input[1]
    if not letter in operations:
        operations[letter] = []
    operations[letter].append(remove_index)


    

letters = {}
i = 0
for letter in s:
    if not letter in letters:
        letters[letter] = []
    letters[letter].append(i)
    i += 1

global_indexes = []
i = 0


for letter in operations:
    for index in operations[letter]:
        remove_index = index - 1
        global_index = letters[letter][remove_index]
        letters[letter].pop(remove_index)
        global_indexes.append(global_index)

global_indexes.sort()

result = ''
last_index = 0
for remove_index in global_indexes:
    result += s[last_index:remove_index]
    last_index = remove_index + 1
    
try:
    result += s[last_index:]
except:
    pass
    
print result
