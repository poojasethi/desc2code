#!/usr/bin/env python
from sys import stdin
from UserString import MutableString

mystr = int(raw_input()) * raw_input()
temp = set(mystr)
mydict = { x: [ i for i, m in enumerate(mystr) if m == x] for x in temp }
mystr = MutableString(mystr)
for n in xrange(int(raw_input())):
    delt, char = stdin.readline().split()
    delt = int(delt)-1
    mystr[mydict[char][delt]] = ' '
    del mydict[char][delt]
print mystr.replace(' ', '')
