k=input()
s=raw_input()
n=input()
d={}
C={}
for i,e in reversed([raw_input().split() for i in [0]*n]):
    l=d.setdefault(e,[0]*20)
    L=len(l)
    i=int(i)
    if (i >= L):
        l += [0]*(i-L+1)
    l.insert(i-1,1)

r=''
for e in s*k:
    c=C.setdefault(e,0)
    if e not in d or len(d[e])<=c or not d[e][c]:
        r += e
    C[e] = c+1
print r
