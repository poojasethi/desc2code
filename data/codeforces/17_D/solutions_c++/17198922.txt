#include <stdio.h>
#include <string.h>
#define N 1000001
char b[N], n[N];
int e[10];
int main() {
    int c;
    scanf("%s%s%d", b, n, &c);
    int len = strlen(n), i;
    for (i = len - 1; i >= 0; --i) {
        if (n[i] == '0') n[i] = '9';
        else {
            --n[i];
            break;
        }
    }
    long long d = 0, t;
    i = 0;
    while (b[i]) {
        d = d*10 + (b[i++]-'0');
        if (d >= c) d %= c;
    }
    e[0] = 1, t = 1;
    for (i = 1; i <= 9; ++i) {
        t = t * d % c;
        e[i] = t;
    }
    i = 0;
    while (n[i] == '0') ++i;
    long long re = 1, t2, t4;
    while (n[i]) {
        t2 = re * re % c, t4 = t2 * t2 % c;
        re = t4 * t4 % c * t2 % c;
        re = re * e[n[i] - '0'] % c;
        ++i;
    }
    re = (d+c-1)%c*re%c;
    if (re == 0) re = c;
    printf("%I64d\n", re);
    return 0;
}
