#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
int n2;
long long sv=0ll,mod,ans,sv2;
char s1[1000002],s2[1000002];
int main(){
  scanf("%s%s%I64d",s1,s2,&mod);
  for(int i=0;s1[i]!='\0';i++){
    sv *= 10; sv += s1[i]-'0';
    sv %= mod;
  }
  ans = 1ll;
  n2 = strlen(s2); s2[n2-1]--;
  for(int i=n2-1;i>=0 && s2[i]<'0';i--)
    s2[i] = '9', s2[i-1]--;
  if(s2[0] == '0')
    for(int i=0;s2[i] != '\0';i++)
      s2[i] = s2[i+1];
  for(int i=0;s2[i] != '\0';i++){
    sv2 = ans;
    for(int j=0;j<9;j++) ans *= sv2, ans %= mod;
    for(int j=0;j<s2[i]-'0';j++) ans *= sv, ans %= mod;
  }
  ans *= (sv-1ll+mod)%mod, ans %= mod; 
  if(ans == 0) ans = mod;
  printf("%I64d\n",ans);
  return 0;
}
