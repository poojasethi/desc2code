#define REP(i,n) for(int i = 0; i < (n); ++i)
#define REV(i,n) for(int i = (n)-1; i >= 0; --i)

using namespace std;

typedef long long ll;

ll pow(int a, int b, int c)
{
	if(b == 0)
		return 1;
	ll d = pow(a, b/2, c);
	d = d * d % c;
	return b % 2 == 0 ? d : d * a % c;
}

int main()
{
	string b, n;
	int c;
	cin >> b >> n >> c;
	ll bmc = 0;
	REP(i,b.size())
		bmc = (bmc * 10 + b[i] - '0') % c;
	REV(i,n.size())
	{
		if(n[i] != '0')
		{
			n[i]--;
			break;
		}
		n[i] = '9';
	}
	int ans = 1;
	REP(i,n.size())
		ans = pow(ans, 10, c) * pow(bmc, n[i]-'0', c) % c;
	ans = ans * (bmc + c - 1) % c;
	
	cout << (ans == 0 ? c : ans) << endl;
	
	return 0;
}
