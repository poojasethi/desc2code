#include <iostream>
using namespace std;

typedef long long ll;

ll mypow(ll a, ll b, ll c){
  ll ret = 1;
  for(ll i=0; i<b; i++){
    ret *= a;
    ret %= c;
  }
  return ret;
}

int main(){
  ll b = 0, n = 0;
  string bs, ns;
  int c;

  cin >> bs >> ns >> c;
  for(int i=0; i<bs.length(); i++){
    b = b*10 + (bs[i]-'0');
    b %= c;
  }
  for(int i=ns.length()-1;; i--){
    if(ns[i] != '0'){ ns[i]--; break; }
    else ns[i] = '9';
  }

  ll ret = 1;
  for(int i=0; i<ns.length(); i++){
    ret = mypow(ret,10,c) * mypow(b, ns[i]-'0', c);
    ret %= c;
  }
  ret = (ret * (b-1+c))%c;

  if(ret == 0) ret = c;
  cout << ret << endl;

  return 0;
}
