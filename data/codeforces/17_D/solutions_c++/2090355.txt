#include<iostream>
#include<cstring>
#include<cstdio>
#include<cctype>

using namespace std;

const int maxn = 1000010;

char str1[maxn] , str2[maxn];
long long b , c;

int main() {
	scanf( "%s" , str1 );
	scanf( "%s" , str2 );
	cin >> c;
	
	for ( int i = 0 , len = strlen( str1 ) ; i < len ; i++ )
		b = ( b * 10 + str1[i] - '0' ) % c;
	
	int len = strlen( str2 );
	str2[len - 1]--;
	for ( int i = len - 1 ; i >= 0 ; i-- )
		if ( str2[i] >= '0' ) break; else {
			str2[i] = '9';
			str2[i - 1]--;
		}
	
	long long p = b , d = 0 , ans = 1;
	for ( int i = len - 1 ; i >= 0 ; i-- ) {
		for ( int j = 1 , end = ( int ) str2[i] - '0' ; j <= end ; j++ )
			ans = ans * p % c;
		d = p;
		for ( int j = 2 ; j <= 10 ; j++ )
			p = p * d % c;
	}
	ans = ans * ( b + c - 1 ) % c;
	
	if ( ans == 0 ) ans = c;
	cout << ans << endl;
	
	return 0;
}
