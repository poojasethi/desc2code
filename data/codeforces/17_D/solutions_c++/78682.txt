#define LL __int64
//#define LL (long long)

using namespace std;

LL myPow(LL b,LL n,LL c)
{
	if(n == 0)return 1;
	else if(n%2==0)
	{
		LL tmp = myPow(b,n/2,c);
		return (tmp*tmp)%c;
	}
	else
	{
		LL tmp = b*myPow(b,n-1,c);
		return tmp%c;
	}
}

int main()
{
	string bs,ns;
	LL b=0,c;
	cin >> bs >> ns >> c;
	
	for(unsigned int i = 0 ; i < bs.size() ; i++)
	{
		b = (b*10 + (bs[i]-'0')) % c;
	}
	
	for(int i = ns.size() - 1 ; i >= 0 ; i--)
	{
		if(ns[i]!='0')
		{
			ns[i]--;
			break;
		}
		ns[i]='9';
	}
	
	LL bn = 1;
	for(unsigned int i = 0 ; i < ns.size() ; i++)
	{
		bn = myPow(bn,10,c)*myPow(b,ns[i]-'0',c)%c;
	}
	
	LL res = (b+c-1)*bn%c;
	cout << (res == 0 ? c : res) << endl;
	
}
