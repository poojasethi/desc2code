#include <iostream>
#include <cstring>
using namespace std;
int c,i,j,m,r=1;
long long b,t[10];
char s[10000000],n[10000000];
int main()
{
     cin>>s>>n>>c;
     m=strlen(s);
     for (i=0;i<m;i++) b=(b*10+s[i]-'0')%c;
     t[0]=1LL%c;
     for (i=1;i<10;i++) t[i]=(t[i-1]*b)%c;
     m=strlen(n);
     for (i=m-1;i>=0;i--)
     {
          n[i]--;
          if (n[i]>='0') break; else n[i]='9';
     }
     for (i=0;i<m;i++)
     {
          long long ro=r;
          for (j=1;j<10;j++) r=(r*ro)%c;
          r=(r*t[n[i]-'0'])%c;
     }
     r=(r*(b+c-1))%c;
     if (r==0) cout<<c; else cout<<r;
     return 0;
}
