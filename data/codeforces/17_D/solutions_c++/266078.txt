#include <cstdio>
#include <cstring>

const int maxn = 1000005;

long long power( long long a , long long b , long long c ) {
    long long ret = 1;
    while ( b-- ) ret = ret * a % c;
    return ret;
}

char t[maxn] , str[maxn];
long long base;
int c;

int main() {
    scanf( "%s%s%d" , t , str , &c );
    int len = strlen(t);
    for ( int i = 0; i < len; i++ ) base = ( ( base * 10 ) + t[i] - '0' ) % c;
    len = strlen(str);
    for ( int i = len; i > 0; i-- ) {
        if ( str[i-1] != '0' ) {  str[i-1]--;  break;  }
        else str[i-1] = '9';
    }
    long long ans = 1;
    for ( int i = 0; i < len; i++ ) {
        ans = power( ans , 10 , c ) * power( base , str[i]-'0' , c ) % c;
    }
    ans = ans * ( base + c - 1 ) % c;
    if ( !ans ) ans = c;
    printf( "%lld\n" , ans );
    return 0;
}
