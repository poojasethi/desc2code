#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
#define MAXN 1000006
char B[MAXN],N[MAXN];
__int64 c,b,n;

int main()
{
    int i,j;
    __int64 k,t,ans;
    scanf("%s%s%I64d",B,N,&c);
    for (i=0;B[i];i++) b=(b*10+B[i]-'0')%c;
    //printf("%d",b);
    if (b==1)
    {
        printf("%I64d\n",c);
        return 0;
    }
    t=b;
    ans=(b-1+c)%c;
    for (j=strlen(N)-1 , N[j]--;j>0;j--)
    {
    //printf("%s\n",N);
        if (N[j]>='0') break;
        N[j]+=10;
        N[j-1]--;
    //printf("%s\n",N);
    }
    for (j=strlen(N)-1;j>=0;j--)
    {
        for (i='1';i<=N[j];i++) ans=(ans*t)%c;
        k=t;
        for (i=0;i<9;i++) t=(t*k)%c;
    }
    if (ans==0) printf("%I64d\n",c);
    else printf("%I64d\n",ans);
}
