#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>

#define LL long long 

using namespace std;

const int Max = 1000009;
char n[Max],b[Max];
LL p[11];

int main() {
	LL c;
	scanf("%s%s%lld",b,n,&c);
	int len = strlen(b);
	LL r = 0;
	for(int i = 0; i < len; i++) {//b%c
		r = (r * 10 + b[i] - '0') % c;
	}
	LL b_1 = (r + c - 1) % c;
	LL b_0 = r;
	p[0] = 1;
	for(int i = 1; i < 10; i++) {//b^i
		p[i] = p[i-1] * b_0 % c;
	}
	len = strlen(n);
	for(int i = len - 1; i >= 0; i--) {//n-1
		if(n[i] == '0')
			n[i] = '9';
		else {
			n[i]--;
			break;
		}
	}
	r = 1;
	for(int i = 0; i < len; i++) {//b^(n-1)%c
		LL t = r;
		for(int j = 1; j < 10; j++) {
			r = r * t % c;
		}
		r = r * p[n[i]-'0'] % c;
	}
	r = b_1 * r % c;
	if(r == 0)
		printf("%lld\n",c);
	else
		printf("%lld\n",r);
	return 0;
}
