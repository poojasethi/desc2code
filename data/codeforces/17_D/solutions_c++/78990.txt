#include <stdio.h>
#include <iostream>
#include <string.h>
#include <algorithm>
using namespace std;

typedef long long LL;

char B[1000005];
char N[1000005];
int lb, lN;

LL b, c;
LL pk[1000005];

int main()
{
	scanf("%s", B);
	lb = strlen( B );
	
	scanf("%s", N);
	lN = strlen( N );
	
	scanf("%d", &c);
	
	b = 0;
	int i;
	for (i=0; i<lb; i++)
	{
		b = b * 10 + (B[i]-'0');
		b %= c;
	}
	
	int j;
	pk[0] = b;
	for (i=1; i<lN; i++)
	{
		pk[i] = 1;
		for (j=0; j<10; j++)
			pk[i] = (pk[i] * pk[i-1]) % c;
	}
	
	reverse(N, N + lN);	
	for (i=0; i<lN; i++)
	{
		if (N[i] == '0') N[i] = '9';
		else
		{
			N[i]--;
			break;
		}
	}
		
	
	//for (i=0; i<lN; i++) printf("%c", N[i]); printf("\n");
	//printf("%lld\n", pk[0]);
	
	LL res = 1;
	for (i=0; i<lN; i++)
	{
		LL tmp = 1;
		for (j=0; j<N[i]-'0'; j++)
			tmp = tmp * pk[i] % c;
		(res *= tmp) %= c;
	}
	
	//cout << res << endl;
	
	(res *= b-1) %= c;
	if (res <= 0) res += c;
	
	cout << res << endl;
}
