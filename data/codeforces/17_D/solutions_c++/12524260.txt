#include <cstdio>
#include <cstring>
#include <cmath>
const int MAXN=1000005;
char a[MAXN], b[MAXN];
inline int qpow(int x, int y, int mod){
    int ans=1;
    for (;y;y>>=1, x=1ll*x*x%mod) if (y&1) ans=1ll*ans*x%mod;
    return ans;
}
int main(){
    //freopen("ex.in", "r", stdin);
    int n=0, k=0, mod;
    scanf("%s%s%d", a+1, b+1, &mod);
    for (int i=1;a[i];i++) n=(1ll*n*10+a[i]-'0')%mod;
    for (int l=strlen(b+1);;l--) if (b[l]!='0') {b[l]--;for (int i=l+1;b[i];i++) b[i]='9';break;}
    int ans=1;
    for (int i=1;b[i];i++) ans=1ll*qpow(ans, 10, mod)*qpow(n, b[i]-'0', mod)%mod;
    ans=1ll*ans*(n+mod-1)%mod;
    printf("%d\n", ans?ans:mod);
    return 0;
}
