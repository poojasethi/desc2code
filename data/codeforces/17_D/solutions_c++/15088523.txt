ll MOD;
ll sq(ll a){return a*a;}
ll modPot(ll a, ll b){return b ? sq(modPot(a, b/2))%MOD*(b&1 ? a : 1)%MOD : 1;}


int main(){
	string B, N;
	ll b=0, bn=1, c, pot10=1, potb, res;
	int k;
	cin>>B>>N>>c; MOD=c;
	k=B.length();
	FOR(i, 0, k){
		b+=(B[k-i-1]-'0')*pot10;
		b%=c;
		pot10=pot10*10%c;
	}
	k=N.length();
	int j;
	for(j=k-1; N[j]=='0'; j--) N[j]='9';
	N[j]--;
	potb=b;
	FOR(i, 0, k){
		bn*=modPot(potb, N[k-i-1]-'0');
		bn%=c;
		potb=modPot(potb, 10);
	}
	res = ((b-1)*bn%c+c-1)%c+1;
	cout << res << endl;
}
