#include<iostream>
#include<string>

using namespace std;

string b,n;
int c;

long long int pow(int b,int p)
{
    if (p == 0)
        return 1;
    if (p == 1)
        return b%c;
        
    long long int d = pow(b,p/2);
    return ((d*d)%c* (p % 2 ? b : 1))%c;
}

int main()
{
    cin>>b>>n>>c;
    
    long long int bm = 0;
    int i;
    for (i=0;i<b.size();i++)
        bm = ((bm * 10)%c + (b[i]-'0'))%c;


    if (n == "1")
    {
        cout<<(bm+c-1)%c<<endl;
        return 0;
    }

    for (i=n.size()-1;i>=0 && n[i] == '0';i--)
        n[i] = '9';
    n[i]--;


    long long int t = bm;
    long long int ans = (bm-1);
    
    for (i=n.size()-1;i>=0;i--)
    {
        ans = (ans * pow(t,n[i]-'0'))%c;
        t = pow(t,10);
    }
    
    if (ans == 0)
        cout<<c<<endl;
    else
        cout<<ans<<endl;
    return 0;
}
