#include <iostream>
#include <string>
using namespace std;
string b,n,m;
long long int c;
long long int a,d,e,f;

int main()
{
        a=0;
        d=0;
        e=1;
        cin >> b >> n >> c;
        for (int i=0;i<b.size();i++)
        {
                a=(a*10+b[i]-'0')%c;
        }
        if (a-1==0) cout << c;
        else 
        {
                if (a!=0) d=a-1;
                else d=c-1;
                m=n;
                int t;
                for (t=m.size()-1;t>=0 && m[t]=='0';t--) m[t]='9';
                m[t]--;
                for (int i=0;i<m.size();i++)
                {
                        f=e;
                        for (int j=0;j<9;j++) e=(e*f)%c;
                        for (int j=0;j<m[i]-'0';j++) e=(e*a)%c;
                }
                e=(e*d)%c;
                if (e==0) cout << c;
                else cout << e;
        }
        return 0;
}
