#include<bits/stdc++.h>
using namespace std;

char A[1000005],B[1000005];
long long c,b;
int at,bt;
int main()
{
    scanf("%s",A+1);
    at=strlen(A+1);
    
    scanf("%s",B+1);
    bt=strlen(B+1);
    
    bool tw=false;
    for(int i=bt;i>=1;i--)
    {
        B[i]--;
        tw=false;
        if(B[i]<'0')
        {
            tw=true;
            B[i]='9';
        }
        if(tw==false) break;
    }
    cin>>c;
    
    for(int i=1;i<=at;i++)
    {
        b=(b*10+1*(A[i]-'0'))%c;
    }
    
    long long nc=1;
    for(int i=1;i<=bt;i++)
    {
        long long br=nc;
        for(int j=0;j<9;j++)
        {
            nc=nc*br%c;
        }
        
        for(int k=0;k<int(B[i])-48;k++)
        {
            nc=nc*b%c;
        }
    }
    
    long long ans=nc*(b-1)%c;
    if(ans<=0)
    {
        ans=ans+c;
    } 
    
    cout<<ans<<endl;
    return 0;
}
