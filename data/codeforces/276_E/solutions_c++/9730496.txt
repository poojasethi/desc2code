#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
class sp{
  public:
  int q;
  sp *w;
  sp(int q1,sp *w1){
   q=q1;w=w1;}
};
sp *a[100000];
int l1(int q,int w){
  sp *c;
  int t=1;
  for(c=a[q];c;c=c->w)
   if(c->q!=w)
    t+=l1(c->q,q);
  return t;
}
int t=0,a1[100000],a2[100000],*s[100000];
int *d[100000],d1[100000];
void l(int q,int w,int r){
  sp *c;
  s[t][r]=0;
  a1[q]=t;
  a2[q]=r;
  for(c=a[q];c;c=c->w)
   if(c->q!=w)
    l(c->q,q,r+1);
}
void k(int *a,int q,int t,int e){
  int w=1;
  for(t>>=2;t;t>>=1){
   w<<=1;
   if(q>=t){
    q-=t;
    a[w]+=e;
    w++;}}
}
int m(int *a,int q,int t){
  int r=0;
  for(q+=t>>1;q;q>>=1)
   r+=a[q];
  return r;
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,o,s1[100000];
  cin>>q>>o;
  for(w=0;w<q;w++)
   a[w]=0;
  for(w=1;w<q;w++){
   scanf("%d%d",&e,&r);
   e--;r--;
   a[e]=new sp(r,a[e]);
   a[r]=new sp(e,a[r]);}
  sp *c;
  for(c=a[0];c;c=c->w){
   s1[t]=l1(c->q,0);
   s[t]=new int[s1[t]];
   l(c->q,0,0);
   for(d1[t]=4;s1[t]*2>=d1[t];d1[t]<<=1);
   d[t]=new int[d1[t]];
   for(w=0;w<d1[t];w++)
    d[t][w]=0;
   t++;}
  d1[t]=1<<18;
  d[t]=new int[d1[t]];
  for(w=0;w<d1[t];w++)
   d[t][w]=0;
  for(;o;o--){
   scanf("%d%d",&e,&w);
   w--;
   if(e==0){
    scanf("%d%d",&e,&r);
    if(w==0){
     k(d[t],r+1,d1[t],e);
     continue;}
    k(d[a1[w]],a2[w]+r+1,d1[a1[w]],e);
    k(d[a1[w]],a2[w]-r,d1[a1[w]],-e);
    if(a2[w]<r){
     k(d[t],r-a2[w],d1[t],e);
     k(d[a1[w]],r-a2[w]-1,d1[a1[w]],-e);}}
   else
    if(w==0)
     printf("%d\n",m(d[t],0,d1[t]));
    else
     printf("%d\n",m(d[a1[w]],a2[w],d1[a1[w]])+m(d[t],a2[w]+1,d1[t]));}
  return 0;}
