#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
const int NM=(int)1e5+5;
int n,m,I,f[NM],cnt[NM],q[NM],head=0,tail=0;
vector<int> a[NM],b[NM];
bool cmp(const int j,const int jj){
	return a[I][j]<a[I][jj];
}
void move(int i){
	if(f[i]==m) return;
	if(!f[i]||a[i][b[i][f[i]]]!=a[i][b[i][f[i]-1]])
		for(int j=f[i];j<m&&a[i][b[i][f[i]]]==a[i][b[i][j]];j++)
			if(!--cnt[b[i][j]]) q[tail++]=b[i][j];
	f[i]++;
}
int main(){
	scanf("%d%d",&n,&m);
	for(int j=0;j<m;j++) cnt[j]=n;
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			int x; scanf("%d",&x);
			a[i].push_back(x);
			b[i].push_back(j);
		}
		I=i;
		sort(b[i].begin(),b[i].end(),cmp);
		for(f[i]=0;f[i]<m&&a[i][b[i][f[i]]]==-1;f[i]++)
			if(!--cnt[b[i][f[i]]]) q[tail++]=b[i][f[i]];
		move(i);
	}
	while(head<tail){
		for(int i=0;i<n;i++) if(a[i][q[head]]!=-1) move(i);
		head++;
	}
	if(tail!=m) return puts("-1"),0;
	for(int i=0;i<tail;i++) printf("%d ",q[i]+1); puts("");
}
