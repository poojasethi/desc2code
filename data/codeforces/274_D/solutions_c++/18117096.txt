#include<iostream>
#include<algorithm>
#include<cstring>
#include<string>
#include<cstdlib>
#include<cstdio>
#include<ctime>
#include<cmath>
#define N 200005
using namespace std;

int n,m,cnt=1,tot,siz,head,tail;
int first[N],next[N],to[N],r[N],p[N],ans[N];
struct node{int w,id;}e[N];

void inser(int x,int y)
{
	next[++siz]=first[x];
	first[x]=siz;
	to[siz]=y;
	r[y]++;
}

bool cmp(const node &a,const node &b)
{
	return a.w<b.w;
}

int main()
{
	scanf("%d%d",&n,&m);e[0].w=-2;
	for (int i=1;i<=n;i++)
	{
		for (int j=1;j<=m;j++)
		{
			scanf("%d",&e[j].w);
			e[j].id=j;
		}
		sort(e+1,e+m+1,cmp);   
		for (int j=1;j<=m;j++)
		{
			if (e[j].w==-1) continue;
			if (e[j].w^e[j-1].w) cnt++;
			inser(e[j].id,m+cnt+1);
			inser(m+cnt,e[j].id);
		}
		cnt++;
	}
	for (int i=1;i<m+cnt;i++)
		if (!r[i]) p[++tail]=i;
	while (head^tail)
	{
		int x=p[++head];
		if (x<=m) ans[++tot]=x;
		for (int i=first[x];i;i=next[i])
			if (!(--r[to[i]])) p[++tail]=to[i];
	}
	if (tot<m) puts("-1");
		else for (int i=1;i<=m;i++) printf("%d ",ans[i]);
	
	return 0;
}
