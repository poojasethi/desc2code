#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
#include<algorithm>
#include<vector>
using namespace std;
int n,m,a[100010],p[100010],cnt=0,num[100010];
vector <int> g[100010];
int vis[100010];
bool cmp(const int A,const int B)
{
	return a[A]<a[B]; 
}
bool dfs(int x)
{
//	cout<<x<<endl;
	if(vis[x]==1)return 1;
	vis[x]=1;
	for(int i=0;i<g[x].size();i++)
	{
		int y=g[x][i];
		if(vis[y]!=-1&&dfs(y))return 1; 
	}
	vis[x]=-1;p[++cnt]=x;
	return 0;
} 
int main()
{
	cin>>n>>m;
	int t=m;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		scanf("%d",&a[j]);
		for(int j=1;j<=m;j++)num[j]=j;
		sort(num+1,num+1+m,cmp);
		int j=1;
//		for(int j=1;j<=m;j++)
//		cout<<num[j]<<endl; 
		while(a[num[j]]==-1)j++;
		while(j<=m)
		{
			int k=j;t++;
//			cout<<k<<" **"<<t<<endl;
			while(a[num[j]]==a[num[k]])k++;
			k--;
			for(int l=j;l<=k;l++)
			{
				if(k!=m) 
				g[num[l]].push_back(t);
				if(a[num[j-1]]!=-1&&j>1)
				g[t-1].push_back(num[l]);
			}
			j=k+1;
		}
	}
//	cout<<"******"<<t<<endl;
	memset(vis,0,sizeof(vis));
	for(int i=1;i<=t;i++)
	if(!vis[i]&&dfs(i))
	{
		cout<<"-1"<<endl;
		return 0;
	}
	for(int i=t;i>=1;i--)
	if(p[i]<=m)printf("%d ",p[i]);
	cout<<endl;
	return 0;
}
