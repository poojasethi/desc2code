#include<bits/stdc++.h>
using namespace std;

const int N=1100000;
int in[N],add;
vector<int> g[N];
void tpsort(int limit){
    queue<int> que;
    for(int i=0;i<add;i++){
        if(0==in[i])que.push(i);
    }
    vector<int> ans;
    while(!que.empty()){
        int u=que.front();
        que.pop();
        if(u<limit)ans.push_back(u);
        for(int i=0;i<g[u].size();i++){
            int v=g[u][i];
            in[v]--;
            if(0==in[v])que.push(v);
        }
    }
    if(ans.size()<limit){
        puts("-1");
        return ;
    }
    for(int i=0;i<limit;i++){
        printf("%d ",ans[i]+1);
    }
}

pair<int,int> a[N];
int main(){
    int n,m;
    scanf("%d%d",&n,&m);
    add=m;
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            scanf("%d",&a[j].first);
            a[j].second=j;
        }
        sort(a,a+m);
        add++;
        for(int j=0;j<m;j++){
            if(0==j||a[j].first!=a[j-1].first){
                add++;
            }
            if(-1==a[j].first)continue;
            g[add-1].push_back(a[j].second);
            in[a[j].second]++;
            g[a[j].second].push_back(add);
            in[add]++;
        }
    }
    tpsort(m);
    return 0;
}
