#include<iostream>
#include<stdlib.h>
#include<string.h>
#include<stdio.h>
#include<math.h>
#include<map>
#include<algorithm>
using namespace std;

int now,x,dui[700000],n,m,l,r,j,i,v,e,bian[1000000],hj[1000000],tou[700000],du[700000];
pair<int,int> a[700000];

void insert(int i,int j)
{
    e++;
    bian[e]=i;hj[e]=tou[j];tou[j]=e;
    du[i]++;
}

int main()
{
scanf("%d%d",&n,&m);
v=n*m;
for (i=1;i<=n;i++)
{
    for (j=1;j<=m;j++)
    {
      scanf("%d",&a[j].first);
      a[j].second=j;
    }
 sort(a+1,a+1+m);a[0].first=-1;
 for (j=1;j<=m;j++)
 {
   if (a[j].first==-1) continue;
   if (a[j].first!=a[j-1].first)
   {
       v++;insert(a[j].second,v);
       if (a[j-1].first!=-1) insert(v,v-1);
       v++;insert(v,a[j].second);
   }else
   {
       insert(v,a[j].second);insert(a[j].second,v-1);
   }
 }
}
l=1;r=0;
for (i=1;i<=v;i++) if (du[i]==0) {r++;dui[r]=i;}
while (l<=r)
{
    x=dui[l];
    l++;
    now=tou[x];
    while (now!=0)
    {
        du[bian[now]]--;
        if (du[bian[now]]==0)
        {
            r++;dui[r]=bian[now];
        }
        now=hj[now];
    }

}
if (r<v) printf("-1");else for (i=1;i<=v;i++) if (dui[i]<=m) printf("%d ",dui[i]);
return 0;
}
