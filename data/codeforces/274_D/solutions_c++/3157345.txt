#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
const int mm=220000;
int i,j,k,l,n,m,id[mm],bfs[mm],du[mm],e[mm*6],next[mm*6],tot,map[mm],len,head[mm],r,x;
bool cmp(const int &x,const int &y){
	return map[x]<map[y];
}
void addedge(int x,int y){
	next[++len]=head[x]; head[x]=len; e[len]=y; du[y]++;
//	printf("%d %d\n",x,y);
}
int main(){
	scanf("%d%d",&n,&m);
	tot=m;
	for(i=1;i<=n;++i){
		for(j=1;j<=m;++j)scanf("%d",&map[j]);
		map[m+1]=-1;
		for(j=1;j<=m;++j)id[j]=j;
		sort(id+1,id+m+1,cmp);
		j=1;
		while(map[id[j]]==-1)j++;
		for(;j<=m;j=k){
			for(k=j;map[id[j]]==map[id[k]];++k);
			if(k==m+1)break;
			if(k==j+1){
				for(l=k;map[id[k]]==map[id[l]];++l)addedge(id[j],id[l]);
			}else{
				++tot;
				for(l=j;l<k;++l)addedge(id[l],tot);
				for(l=k;map[id[k]]==map[id[l]];++l)addedge(tot,id[l]);
			}
		}
	}
	for(i=1;i<=tot;++i)if(!du[i])bfs[++r]=i;
	l=1;
	while(l<=r){
		x=bfs[l++];
		for(i=head[x];i;i=next[i]){
			du[e[i]]--;
			if(!du[e[i]])bfs[++r]=e[i];
		}
	}
	if(r!=tot)printf("-1\n");
	else{
		for(i=1;i<=tot;++i)if(bfs[i]<=m)printf("%d ",bfs[i]);
	}
//	system("pause");
}
