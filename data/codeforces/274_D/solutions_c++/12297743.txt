#include<cstdio>
#include<algorithm>
#define N 2000010
using namespace std;
int n,m,tot,i,j,b[N],rd[N],a[N],dp,o,w,t,z[N];
int pre[N],p[N],tt[N],ans[N];
bool cmp(int x,int y)
{
	return a[x]<a[y];
}
void link(int x,int y)
{
	dp++;pre[dp]=p[x];p[x]=dp;tt[dp]=y;
}
int main()
{
	scanf("%d%d",&n,&m);
	tot=m;
	for (i=1;i<=n;i++)
	{
	for (j=1;j<=m;j++)
	{
		scanf("%d",&a[j]);
		b[j]=j;
	}
	sort(b+1,b+1+m,cmp);
	tot++;
	for (j=1;j<=m;j++)
	if (a[b[j]]!=-1)
	{
		link(tot,b[j]);rd[b[j]]++;
		link(b[j],tot+1);rd[tot+1]++;
		if (a[b[j]]!=a[b[j+1]]) tot++;
	}
	}
	
	for (i=1;i<=tot;i++)
	if (rd[i]==0)
	{
		w++;z[w]=i;
	}
	
	while (t!=w)
	{
		t++;
		if ((1<=z[t])&&(z[t]<=m))
		{
			o++;ans[o]=z[t];
		}
		i=p[z[t]];
		while (i)
		{
			rd[tt[i]]--;
			if (rd[tt[i]]==0)
			{
				w++;z[w]=tt[i];
			}
			i=pre[i];
		}
	}
	if (o==m)
	for (i=1;i<=o;i++)
	printf("%d ",ans[i]);
	else
	printf("-1");

}
