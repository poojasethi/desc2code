#include <cstring>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;
const int inf = 0x7f7f7f7f;
int n,m,k;
int a[110][110];
int ans;
int Get(int z){
  int ans = 0;
  for(int j = 1;j<=m;j++){
   int sum = 0;
   for(int i = 1;i<=n;i++)sum+=(a[i][z]==a[i][j]);
   sum = min(sum,n-sum);
   ans += sum;
 }
  return ans;
}
int main()
{
	ans = inf;
   #ifndef ONLINE_JUDGE
	freopen("425problemB.in","r",stdin);freopen("425problemB.out","w",stdout);
	#endif
   scanf("%d%d%d",&n,&m,&k);
	for(int i = 1;i<=n;i++)
	 for(int j = 1;j<=m;j++)
		scanf("%d",&a[i][j]);
	if(n>k){
	 for(int i = 1;i<=m;i++)
		 ans = min(ans,Get(i));
    }else{
	  int tt = 1<<n;
	 for(int i = 0;i<tt;i++){
	  for(int j = 1;j<=n;j++)
		 a[j][0] = (i>>(j-1))&1;
	  ans = min(ans,Get(0));
	 }
	}
	if(ans <= k)
	printf("%d",ans);
    else
	 puts("-1");
	#ifndef ONLINE_JUDGE
	fclose(stdout);fclose(stdin);
	#endif
}
