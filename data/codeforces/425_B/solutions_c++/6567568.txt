#include<bits/stdc++.h>
using namespace std;
int n,m,k,a[105][105],ans,num[12];

int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<n;i++)
	  for(int j=0;j<m;j++)
		scanf("%d",&a[i][j]);
	ans=100;
	if(n>k)
	{
		for(int i=0;i<n;i++)
		{
			int sum=0;
			for(int j=0;j<n;j++)
			{
				int tmp=0;
				for(int p=0;p<m;p++)
				  if(a[i][p]==a[j][p])
					tmp++;
				sum+=min(tmp,m-tmp);
			}
			if(ans>sum) ans=sum;
		}
	}
	else
	{
		for(int x=0;x<(1<<n);x++)
		{
			int sum=0;
			for(int j=0;j<m;j++)
			{
				int tmp=0;
				for(int i=0;i<n;i++)
				  if(a[i][j]==((x>>i)&1))
					tmp++;
				sum+=min(tmp,n-tmp);
			}
			if(ans>sum) ans=sum;
		}
	}
	if(ans>k) puts("-1");
	else printf("%d\n",ans);
	return 0;
}
