#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;
#define MAXN 300300
int a[110][110];
int main(){
	int n,m,k,ans;
	while(cin>>n>>m>>k){
		if(n<m){
			swap(n,m);
			for(int i=1;i<=m;++i)
				for(int j=1;j<=n;++j)
					scanf(" %d",&a[j][i]);
		}
		else{
			for(int i=1;i<=n;++i)
				for(int j=1;j<=m;++j)
					scanf("%d",&a[i][j]);
		}
		if(n>k){
			ans=1<<20;
			for(int i=1;i<=n;++i){
				int now=0;
				for(int j=1;j<=n;++j){
					int tmp=0;
					for(int t=1;t<=m;++t)
						tmp+=a[i][t]^a[j][t];
					now+=min(tmp,m-tmp);
				}
				ans=min(now,ans);
			}
		}
		else{
			ans=1<<20;
			for(int i=0;i<(1<<m);++i){
				int now=0;
				for(int j=1;j<=n;++j){
					int tmp=0;
					for(int t=0;t<m;++t)
						tmp+=a[j][t+1]^((i>>t)&1);
					now+=min(tmp,m-tmp);
				}
				ans=min(ans,now);
			}
		}
		if(ans<=k)
			cout<<ans<<endl;
		else
			cout<<-1<<endl;
	}
}
