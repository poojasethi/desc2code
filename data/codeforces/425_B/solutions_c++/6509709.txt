#include <cstdio>
#include <algorithm>
using namespace std;
const int N = 101;
int num[N][N];
int main()
{
	int n, m, k; scanf("%d%d%d", &n, &m, &k);
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			scanf("%d", num[i] + j);
	int minval = k + 1;
	if (n <= 10)
	{
		for (int val = 0; val < 1024; val++)
		{
			int sum = 0;
			for (int j = 0; j < m; j++)
			{
				int temp = 0;
				for (int i = 0; i < n; i++)
					if (num[i][j] != ((val & (1 << i)) ? 1 : 0)) temp++;
				sum += min(temp, n - temp);
			}
			if (sum < minval) minval = sum;
		}
	}
	else
	{
		for (int i = 0; i < n; i++)
		{
			int val = 0;
			for (int j = 0; j < n; j++)
			{
				int temp = 0;
				for (int k = 0; k < m; k++)
					if (num[j][k] != num[i][k]) temp++;
				val += min(temp, m - temp);
			}
			if (val < minval) minval = val;
		}
	}
	printf("%d\n", minval > k ? -1 : minval);
	return 0;
}
