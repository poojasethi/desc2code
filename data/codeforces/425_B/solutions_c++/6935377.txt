#include <cstdio>
inline int min(int x, int y) { return x < y ? x : y; }
int a[110][110];
int main() {
	int n, m, k;
	scanf("%d%d%d", &n, &m, &k);
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			scanf("%d", &a[i][j]);
		}
	}
	int ans = k + 1;
	if (m > k) {
		for (int i = 0; i < m; i++) {
			int t = 0;
			for (int j = 0; j < m; j++) {
				int b = 0;
				for (int l = 0; l < n; l++) {
					if (a[l][i] != a[l][j]) {
						b++;
					}
				}
				t += min(b, n - b);
			}
			ans = min(ans, t);
		}
	} else {
		for (int i = 0; i < 1 << m; i++) {
			int t = 0;
			for (int j = 0; j < n; j++) {
				int b = 0;
				for (int l = 0; l < m; l++) {
					if ((i >> l & 1) != a[j][l]) {
						b++;
					}
				}
				t += min(b, m - b);
			}
			ans = min(ans, t);
		}
	}
	printf("%d\n", ans > k ? -1 : ans);
	return 0;
}
