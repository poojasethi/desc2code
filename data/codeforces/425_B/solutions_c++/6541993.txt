
#include <cstdio>


#define FOR(i,n) for(int i=0;i<n;++i)

const int N = 110;
int d[N][N];


int main()
{
	int n,m,k;
	scanf("%d%d%d",&n,&m,&k);
	FOR(i,n)FOR(j,m)scanf("%d",d[i]+j);

	int res = 1<<30;
	if(m>k){
		// some column is not changed
		FOR(i,m){
			int sum=0;
			FOR(j,m){
				if(j==i)continue;	
				int t=0;
				FOR(r,n){
					if( d[r][i] != d[r][j] )++t;
				}
				if( t > n - t) t = n-t;
				sum += t;
			}	

			if(sum < res) res = sum;
		}
	}
	else{
		FOR(s,(1<<m)){
			int sum = 0;
			FOR(r,n){
				int t = 0;
				FOR(c,m) if( ((s>>c)&1) != d[r][c] )++t;
				if( t > m-t) t=m-t;
				sum += t;
			}
			if(sum < res) res = sum;
		}
	}


	if(res > k) puts("-1");
	else printf("%d\n",res);


	
	
	
	return 0;	
}
