#include <bits/stdc++.h>
using namespace std;

const int N = 105;

int n, m, k, a[N][N];

int main() {
	cin >> n >> m >> k;
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < m; ++j) {
			cin >> a[i][j];
		}
	}
	if (n < m) {
		for (int i = 0; i < n; ++i) {
			for (int j = i + 1; j < m; ++j) {
				swap(a[i][j], a[j][i]);
			}
		}
		swap(n, m);
	}
	int ans = k + 1;
	if (n <= k) {
		for (int msk = 0; msk < 1 << m; ++msk) {
			int sum = 0;
			for (int i = 0; i < n; ++i) {
				int cnt = 0;
				for (int j = 0; j < m; ++j) {
					cnt += a[i][j] ^ (msk >> j & 1);
				}
				sum += min(cnt, m - cnt);
			}
			ans = min(ans, sum);
		}
	} else {
		for (int r = 0; r < n; ++r) {
			int sum = 0;
			for (int i = 0; i < n; ++i) {
				int cnt = 0;
				for (int j = 0; j < m; ++j) {
					cnt += a[i][j] != a[r][j];
				}
				sum += min(cnt, m - cnt);
			}
			ans = min(ans, sum);
		}
	}
	cout << (ans > k ? -1 : ans);
}
