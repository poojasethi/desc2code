#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<cstdio>
using namespace std;
#define N 110
int n,m,k,ans=1000000,a[N][N];
int check(int z){
	int ans=0;
	for(int j=1;j<=m;j++){
	int sum=0;
	for(int i=1;i<=n;i++)sum+=a[i][z]==a[i][j];
	sum=min(sum,n-sum);
	ans+=sum;
	}
	return ans;
}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=n;i++)
	  for(int j=1;j<=m;j++)scanf("%d",&a[i][j]);
	if(n>k){
	  for(int i=1;i<=m;i++)ans=min(ans,check(i));
	}
	else{
		int tt=1<<n;
	  for(int i=0;i<tt;i++){
		for(int j=1;j<=n;j++)a[j][0]=(i>>(j-1))&1;
		ans=min(ans,check(0));
		}
	}	
	if(ans>k)cout<<-1<<endl;
	else cout<<ans<<endl;
	return 0;
}
