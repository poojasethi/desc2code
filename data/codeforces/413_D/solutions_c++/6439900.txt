#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const long mod=1000000007;
long long f[2001][2049][2];
long long ans=0;
void add(long long &x,long long y) {x=(x+y)%mod;}
int main()
{
    long i,j,n,k,x;
    scanf("%ld%ld",&n,&k);
    memset(f,0,sizeof(f));f[0][0][0]=1;
    for (i=1;i<=n;i++)
    {
        scanf("%ld",&x);
        for (j=0;j<=(1<<(k-1))-2;j+=2)
        {
            if (x==0||x==2) 
            {
                add(f[i][j][1],f[i-1][j][0]);add(f[i][j+2][0],f[i-1][j][1]);
            }
            if (x==0||x==4)
            {
                add(f[i][j+2][0],f[i-1][j][0]);add(f[i][2][0],f[i-1][j][1]);
            }
        }
        if (x==0) add(ans,ans);add(ans,f[i][1<<(k-1)][0]);
    }
    printf("%I64d\n",ans);
    return 0;
}
