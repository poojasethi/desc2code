#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define FOR(i,a,b) for(i=a; i<b; i++)
#define rep(i,n) FOR(i,0,n)
#define ff first
#define ss second
#define mod 1000000007

using namespace std;
typedef long long LL;
typedef vector<int> VI;
typedef vector< vector<int> > VVI;
typedef pair< int, int > PII;

int n,k,a[2010],dp[2010][2100];
LL ans;

int main()
{
	cin>>n>>k;
	memset(dp,0,sizeof(dp));
	dp[0][0]=1;
	for(int i=1; i<=n; i++)
	{
		cin>>a[i];
		for(int j=0; j<=(1<<k); j++)
		{
			if(dp[i-1][j]==0)continue;
			if(j==(1<<k))
			{
				if(a[i]==0)dp[i][j] = (dp[i][j]+dp[i-1][j])%mod;
				dp[i][j] = (dp[i][j]+dp[i-1][j])%mod;
				continue;
			}
			if(a[i]==0 || a[i]==2)
				dp[i][j+2] = (dp[i][j+2]+dp[i-1][j])%mod;
			if(a[i]==0 || a[i]==4)
			{
				if(j&2)dp[i][4] = (dp[i][4]+dp[i-1][j])%mod;
				else dp[i][j+4] = (dp[i][j+4]+dp[i-1][j])%mod;
			}
		}
	}
	cout<<dp[n][1<<k]<<endl;
	return 0;
}
