#include <cstdio>
#define LL long long

const LL MO=1000000007;
LL ans,a[1000010],b[1000010];
int x,y,n;

LL Pow(LL a,LL b) {
	LL ret=1;
	for (;b;b>>=1ll) {if (b&1ll) ret=ret*a%MO;a=a*a%MO;}
	return ret;
}

bool Check(int k) {
	while (k) {
		if ((k%10!=x)&&(k%10!=y)) return false;
		k/=10;
	}
	return true;
}

LL C(LL n,LL k) {
	return a[n]*b[k]%MO*b[n-k]%MO;
}

int main() {
	scanf("%d%d%d",&x,&y,&n);
	a[0]=b[0]=1;
	for (LL i=1;i<=n;++i) {
		a[i]=a[i-1]*i%MO;
		b[i]=Pow(a[i],MO-2);
	}
	for (int i=0;i<=n;++i) {
		int sum=i*x+(n-i)*y;
		if (Check(sum)) ans=(ans+C(n,i))%MO;
	}
	printf("%I64d\n",ans);
	return 0;
}
