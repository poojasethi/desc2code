#include <iostream>

using namespace std;
#define mod 1000000007
int a,b,n;
int test (int s)
{
 while (s)
 {
  int x=s%10;
  if (!(x==a||x==b)) return 0;
  s/=10;
 }
 return 1;
}
long long pows(long long a, long long b)
{long long ans=1;
 while (b)
 {if (b%2) {ans*=a; ans%=mod;}
  b=b/2;
  a*=a;
  a%=mod;
 }
 return ans;
}
int main()
{cin>>a>>b>>n;
 long long c=1,ans=0;
 if (test(n*b)) ans++;
for (int i=1;i<=n;i++)
 {
  c*=(n-i+1); c%=mod;
  c*=pows(i,mod-2); c%=mod;
  //cout<<c<<"   ";
  int s=i*a+(n-i)*b;
  if (test(s)) {ans+=c; ans%=mod;}
 }
cout<<ans;
return 0;
}
