#include <iostream>
using namespace std;
#define ll long long 
#define mod 1000000007
#define maxn 1000005

ll f[maxn];
ll a,b,n;

ll mypow(ll a,ll n)
{
	ll ans=1;
	while(n)
	{
		if(n&1) ans=(ans*a)%mod;
		a=(a*a)%mod;
		n>>=1;
	}
	return ans;
}
bool check(ll s)
{
	ll t;
	while(s)
	{
		t=s%10;
		if(t==a||t==b) s=s/10;
		else return false;	
	}
	return true;
}
int main()
{
	f[0]=1;
	f[1]=1;
	for(ll i=2;i<maxn;++i) f[i]=(i*f[i-1])%mod;
	ll ret;
	while(cin>>a>>b>>n)
	{
		ret=0;
		for(ll i=0;i<=n;++i)
		{
			if(check(a*i+b*n-b*i))
			{
			 	ret=(((mypow(f[n-i],mod-2)*mypow(f[i],mod-2))%mod*f[n])%mod+ret)%mod;
			}
		}
		cout<<ret<<endl;
	}
	return 0;
}
