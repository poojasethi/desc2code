#include <iostream>
using namespace std;
typedef long long ll;

const ll m(1000000007);
ll ans, a, b, n, f[1000001] = {1}, g[1000001] = {1};

ll h(ll x) {
  ll y(1);
  for (ll i(m - 2); i; i >>= 1, x = x * x % m)
    if (i & 1)
      y = y * x % m;
  return y;
}

bool c(ll x) {
  for ( ; x; x /= 10)
    if (x % 10 != a && x % 10 != b)
      return false;
  return true;
}

int main() {
  cin >> a >> b >> n;
  for (ll i(1); i <= n; ++i) {
    f[i] = f[i - 1] * i % m;
    g[i] = h(f[i]);
  }
  for (ll i(0); i <= n; ++i)
    if (c(i * a + (n - i) * b))
      ans = (ans + g[i] * g[n - i]) % m;
  cout << ans * f[n] % m << endl;
}
