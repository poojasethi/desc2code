#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
const ll mod = (ll)1e9+7;
int a,b,n;
bool check(int s){
	while(s){
		if(s%10!=a&&s%10!=b)return 0;
		s/=10;
	}
	return 1;
}
ll Pow(ll a,ll b){
	ll ans=1;
	while(b){
		if(b&1)ans=ans*a%mod;
		a=a*a%mod;
		b>>=1;
	}
	return ans;
}
int main(){
	while(cin>>a>>b>>n){
		ll ans=0;
		ll result=1;
		for(int i=0;i<=n;i++){
			int sum=i*a+(n-i)*b;
			if(check(sum)){
				ans+=result;
				ans%=mod;
			}
			result=result*(n-i)%mod;
			result=result*Pow(i+1,mod-2)%mod;
		}
		cout<<ans<<endl;
	}
	return 0;
}
