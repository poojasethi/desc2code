#include <iostream>

using namespace std;

const int mod = 1000*1000*1000+7;

int binpow(long long a, long long n)
{
	a%=mod;
	long long res = 1;
	while (n)
	{
		if (n&1)
			res = (res*a)%mod;
		a = (a*a)%mod;
		n>>=1;
	}
	return res;
}

int main()
{
	int a,b,n;
	cin>>a>>b>>n;
	int * fact = new int[n+1];
	fact[0] = 1;
	for (long long i=1; i<=n; i++)
		fact[i] = (i*fact[i-1]) % mod;
	long long res = 0;
	for (int i=0; i<=n; i++)
	{
		int t = a*i + b*(n-i);
		while (t)
		{
			if (t%10 != a && t%10 != b)
				break;
			t/=10;
		}
		if (!t)
			res = (res + fact[n]*(long long)binpow(fact[i]*(long long)fact[n-i],mod-2))%mod;
	}
	cout<<res<<endl;
	return 0;
}
