#include<stdio.h>
#define LL __int64
const int mod=1000000007;
    
int a,b,n;
int judge(int x){
    for(;x%10==a||x%10==b;x/=10);
    return x?0:1;
}   
int Power(int x,int cf)
{   
    LL r=1,y=x;
    for(;cf;cf=cf>>1,y=y*y%mod) (cf&1)&&(r=r*y%mod);
    return r;
}   
int main(){
    int i,x;
    LL s,c,up,dn;
    while(~scanf("%d%d%d",&a,&b,&n)){
        for(s=i=0,c=up=dn=1,x=a*n;i<=n;i++,x+=b-a){
            if(judge(x)){
                s=(s+c)%mod;
//              printf("x=%d i=%d c=%I64d s=%I64d\n",x,i,c,s);
            }
            up=up*(n-i)%mod;dn=dn*(i+1)%mod;
            c=up*Power(dn,mod-2)%mod;
        }
        printf("%I64d\n",s);
    }
    return 0;
}
