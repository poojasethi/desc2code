#include<stdio.h>
#include<math.h>
#define MOD 1000000007
typedef long long int LL;
LL check(LL a,LL b,LL q){
	while(q){
	if(q%10!=a && q%10!=b){
		return 0;
	}
		q=q/10;
	}
	return 1;
}

LL power(LL a,LL b){
	if(b==0)
		return 1;
	else if(b==1)
		return a;

	else if(b%2==0){
		LL p = power(a,b/2);
		return (p*p)%MOD;
	}
	else{
		LL p = power(a,b/2);
		return ((p*p)%MOD*a)%MOD;
	}
}
int main(){
int a,b,n,i;
LL p,sum,ans=0,q;
	scanf("%d %d %d",&a,&b,&n);
sum=a*n;
	for(i=0;i<=n;i++){
		q = sum + i*(b-a);
		if(i==0)
			p=1;
		else{
			p = (p*(n-i+1))%MOD;
			p = (p * power(i,MOD-2))%MOD;
		}
		if(check(a,b,q))
			ans = (ans+p)%MOD;
	}
printf("%I64d\n",ans);
return 0;
}
