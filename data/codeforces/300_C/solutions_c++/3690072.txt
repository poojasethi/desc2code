#include <iostream>
#include <vector>
#include <stdio.h>
using namespace std;
#define mod 1000000007
int n,a,b;
int test(int s)
 { while (s)
  {int x=(s%10);
   if (!(x==a||x==b)) return 0;
   s/=10;
  }
 return 1;
 }
long long pow(long long a, long long b)
{long long ans=1;
 while (b)
 {if (b%2==1) {ans*=a; ans%=mod;}
   b/=2;
   a*=a; a%=mod;
  }
 return ans;
}
int main()
{// freopen("i.txt","r",stdin);
cin>>a>>b>>n;
long long ans=0;
long long c=1;
for (int i=0;i<=n;i++)
 {//cout<<c<<"  ";
  int s=i*a+b*(n-i);
  if (test(s)) {ans+=c; ans%=mod;}
  c*=(n-i); c%=mod;
  //cout<<pow(i+1,mod-2)<<"  ";
  c*=pow(i+1,mod-2); c%=mod;
 }
cout<<ans;
return 0;
}
