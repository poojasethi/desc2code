# include <stdio.h>
typedef long long ll ;
ll fac[1000010]={1};
int n,a,b,M=1000000007;
int Good(int x)
{
	for(;x%10==a||x%10==b;x/=10);
	return !x;
}
ll inv(ll x)
{
	ll r,y;
	for(r=1,y=M-2;y;x=x*x%M,y>>=1)(y&1)&&(r=r*x%M);
	return r;
}
int C(int n, int m)
{
	return fac[n]*inv(fac[m])%M*inv(fac[n-m])%M ;
}
int main ()
{
	int i, ans ;
	while (~scanf("%d%d%d",&a,&b,&n))
	{
		for (i=1;i<=n;i++)fac[i]=fac[i-1]*i%M;
		for (i=0,ans=0;i<=n;i++)Good(a*i+(n-i)*b)&&(ans=(ans+C(n,i))%M) ;
		printf ("%d\n", ans) ;
	}
	return 0;
}
