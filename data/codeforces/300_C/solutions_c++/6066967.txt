#include"stdio.h"
#define mod 1000000007 
__int64 a,b,n,f[1111111],ans,i;
int main() {
	int pan(__int64 x);
	__int64 h(__int64 x);
  	scanf("%I64d%I64d%I64d",&a,&b,&n);
  	f[0]=1;
  	for(i=1;i<=n;i++)f[i]=f[i-1]*i%mod;
  	for(i=0;i<=n;i++)
    	if(pan(i*a+(n-i)*b))ans=(ans+h(f[i])*h(f[n-i]))%mod;
  	printf("%I64d\n",ans*f[n]%mod);
}
int pan(__int64 x)
{
	__int64 i,k;
	k=1;
	while(x>0)
	{
		i=x%10;
		x/=10;
		if(i!=a&&i!=b){
			k=0;break;
		}
	}
	return k;
}
__int64 h(__int64 x)
{
	__int64 i,y=1;;
	i=mod-2;
	while(i)
	{
		if(i&1)
		{
			y=y*x%mod;
		}
		x=x*x%mod;
		i/=2;
	}
	return y;
}
