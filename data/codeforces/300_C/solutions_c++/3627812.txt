#include<iostream>
#define mod 1000000007
using namespace std;
int n,a,b,i;
long long f[1000005],p,s=0;
bool is(long long x)
{
while(x)
{
if(x%10!=a && x%10!=b) return 0;
x/=10;
}
return 1;
}
long long r(long long x)
{
int z=mod-2;
long long q=1;
while(z)
{
if(z&1) q=(q*x)%mod;
x=(x*x)%mod;
z>>=1;
}
return q;
}
int main()
{
cin>>a>>b>>n;
f[0]=f[1]=1;
for(i=2;i<=1000000;i++)
{
f[i]=(f[i-1]*i)%mod;
}
p=a*n;
for(i=0;i<=n;i++)
{
if(is(p)) s=(s+((((f[n]*r(f[i]))%mod)*r(f[n-i]))%mod))%mod;
p+=b-a;
}
cout<<s;
return 0;
}
