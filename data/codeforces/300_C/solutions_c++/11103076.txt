#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long 
int a,b;
ll power(ll a,ll b)
{
	if(b==0)
		return 1;
	ll temp=power(a,b/2);
	temp=(temp*temp)%mod;
	if(b&1)
		temp=(temp*a)%mod;
	return temp;		
}
bool sum(int x)
{
	int c;
	while(x!=0)
	{
		c=x%10;
		if(c!=a&&c!=b)
			return false;
		x=x/10;		
	}	
	return true;	
}
