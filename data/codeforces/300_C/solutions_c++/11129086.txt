#include<bits/stdc++.h>
using namespace std;

int a , b , n , ans , fact[1000001] = {1} ;
#define M 1000000007

bool isGood(int num){
  while(num){
    if(num%10 != a && num%10 != b) return false ; 
    num /= 10 ; 
  }   
  return true ; 
}

int power(int a){
  int ret = 1 , b = M-2 ;
  while(b){
    if(b&1) ret = (1LL*ret*a)%M ;
    a = (1LL*a*a)%M ; b >>= 1 ; 
  }  
  return ret ;
}

int main(){
  cin >> a >> b >> n ;
  for(int i=1 ; i <= n ; i++) fact[i] = (1LL*fact[i-1]*i)%M ; 
  for(int i=0 ; i <= n ; i++) if( isGood(a*i + b*(n-i)) ) 
    ans = ( ans + ( (1LL*fact[n] * power(fact[i]) )%M * power(fact[n-i]) )%M )%M ;  
  
  cout << ans ;

  return 0;
}
