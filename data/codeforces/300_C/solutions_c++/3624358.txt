#include<iostream>
using namespace std;
typedef long long ll;
const int mod = 1e9+7;
inline ll pow(int b,int n){
	ll ans = 1, t = b;
	while(n){
		if(n&1) ans = (ans * t) % mod;
		t = (t * t) % mod; n >>= 1;
	}
	return ans;
}
int a,b,n;
bool chk(int sum){
	while(sum){
		if(sum % 10 != a && sum % 10 != b)
			return 0;
		sum /= 10;
	}
	return 1;
}
int main(){
	cin >>a >> b >> n;
	ll C = 1;
	long long ans = 0;
	for(int i = 0; i <= n; i++){
		int sum = a * i + b * (n - i);
		if(chk(sum)) {
			ans += C;
			ans %= mod;
		}
		C = (C * (n - i)) % mod;
		C = (C * pow(i+1,mod - 2)) % mod;
	}
	cout<< ans <<endl;
}
