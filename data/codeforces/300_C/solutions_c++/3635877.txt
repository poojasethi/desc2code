#include<iostream>
#define MAXN 1000000007
using namespace std;

long long a,b,n;

long long div( long long o ){
  if ( o<=1 ){
    return o;
  }else{
    return div( MAXN % o )*( MAXN - MAXN/o ) % MAXN;
  }
}

bool istrue( long long o ){
  int p;
  while( o!=0 ){
    p=o % 10;
    o/=10;
    if( p!=a && p!=b ) return false;
  }
  return true;
}

int main(  ){
  cin>>a>>b>>n;
  long long ans=0;
  long long tmp=1;

  long long tt;
  for( long long i=0;i<=n;i++ ){
    tt=a*i+b*( n-i );
    if( istrue( tt ) ) ans=( ans+ tmp ) % MAXN;
    tmp=tmp*( n-i ) % MAXN;
    tmp=tmp*div( i+1 ) % MAXN;
    
  }
  cout<<ans<<endl;
}
