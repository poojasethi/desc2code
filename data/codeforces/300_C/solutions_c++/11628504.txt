#include<bits/stdc++.h>
using namespace std;
#define ll long long 
#define MAX 1000005
#define N 1000000007
ll i,j,k,l,m,n,t;
ll a,b,c;
ll inv[MAX];
int main()
{
	cin>>a>>b>>n;
	inv[1]=1;
	for(i=2;i<MAX;i++)
	{
		inv[i]=N-(((N/i)*(inv[N%i]))%N);
		//cout<<inv[i]<<endl;
	}
	ll msd=0;
	ll val=1;
	for(i=0;i<=n;i++)
	{
		ll sum=a*i+(n-i)*b;
		while(sum)
		{
			if((sum%10!=a) && (sum%10!=b))
			break;
			sum/=10;
		}
		//cout<<sum<<endl;
		if(sum==0)
		{
			//cout<<sum<<endl;
			msd=(msd+val)%N;
		}
		//cout<<val<<(n-i)<<" "<<inv[i+1]<<endl;
		val=val*(n-i)%N*inv[i+1]%N;
	}
	cout<<(msd)%N<<endl;
	return 0;
}
