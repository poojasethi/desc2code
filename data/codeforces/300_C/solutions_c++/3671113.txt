#include <cstring>
#include <cstdio>
#include <iostream>
using namespace std;

const int mod=1000000007;
int a,n,b;

long long pow(long long x, long long p)
{
	long long res=1;
	for (;p;p>>=1)
	{
		if (p&1) res=res*x%mod;
		x=x*x%mod;
	}
	return res;
}

bool check(long long x)
{
	while (x)
	{
		if (x%10!=a && x%10!=b) return 0;
		x/=10;
	}
	return 1;
}

int main()
{
		scanf("%d%d%d", &a,&b,&n);
		long long ans=0, x=1;
		for (int i=0; i<=n; i++)
		{
				long long sum=(a*i+b*(n-i));
				if (check(sum)) ans+=x, ans%=mod;
				x=x*(n-i)%mod*pow(i+1, mod-2)%mod;
		}
		cout<<ans<<endl;
		//for (;;);
}
