#include <iostream>
using namespace std;
const int m=1000000007;
int n,a,b,fact[1000001]={1},inv[1000001];
bool is(int s){
	if(!s)
		return false;
	int x;
	while(s){
		x=s%10;
		if(x!=a && x!=b)
			return false;
		s/=10;
	}
	return true;
}
int Pow(int &n,int p){
	if(!p)
		return 1;
	long long r=Pow(n,p/2);
	return r*r%m*(p&1?n:1)%m;
}
int main()
{
	cin>>a>>b>>n;
	for(long long i=1;i<=n;++i)
		fact[i]=i*fact[i-1]%m;
	long long r=0;
	int x;
	for(int i=0;i<=n;++i)
		if(is(i*a+b*(n-i))){
			x=(long long)fact[n-i]*fact[i]%m;
			r=(r+(long long)fact[n]*Pow(x,m-2)%m)%m;
		}
	cout<<r;
	return 0;
}
