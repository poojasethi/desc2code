#include <iostream>
#define MOD 1000000007
using namespace std;
int a,b,n;
long long p[1000005];
long long s[1000005];
bool check(int x)
{
	while (x>0)
	{
		if (x%10!=a && x%10!=b) {return 0;}
		x/=10;
	}
	return 1;
}
long long niyuan(long long x)
{
	long long t=x,s=1;
	int n=MOD-2;
	while (n)
	{
		if (n%2==1)
		{
			s=(s*t)%MOD;
		}
		t=(t*t)%MOD;
		n/=2;
	}
	return s;
}
int main()
{
	int i;
	long long ans=0;
	cin>>a>>b>>n;
	p[1]=1;
	for (i=2;i<=n;i++) {p[i]=(p[i-1]*i)%MOD;}
	for (i=1;i<=n;i++) {s[i]=niyuan(p[i]);}
	s[0]=1;
	for (i=0;i<=n;i++)
	{
		if (check(a*i+b*(n-i)))
		{
			ans=ans+(((p[n]*s[n-i])%MOD)*s[i])%MOD; ans%=MOD;
		}
	}
	cout<<ans<<endl;
}
