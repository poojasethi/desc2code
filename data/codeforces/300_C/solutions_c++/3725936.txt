#include <iostream>
using namespace std;

const int m(1000000007);
int ans, a, b, n, f[1000001] = {1}, g[1000001] = {1};

int h(int x) {
  int y(1);
  for (int i(m - 2); i; i >>= 1, x = (long long)x * x % m)
    if (i & 1)
      y = (long long)y * x % m;
  return y;
}

bool c(int x) {
  for ( ; x; x /= 10)
    if (x % 10 != a && x % 10 != b)
      return false;
  return true;
}

int main() {
  cin >> a >> b >> n;
  for (int i(1); i <= n; ++i) {
    f[i] = (long long)f[i - 1] * i % m;
    g[i] = h(f[i]);
  }
  for (int i(0); i <= n; ++i)
    if (c(i * a + (n - i) * b))
      ans = (ans + (long long)g[i] * g[n - i]) % m;
  cout << (long long)ans * f[n] % m << endl;
}
