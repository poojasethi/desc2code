#include<cstdio>
#include<iostream>
using namespace std;
typedef long long ll;
const ll mod = 1e9+7;
ll p[1000001],r[1000001];
ll pow_mod(ll a,ll x){
	ll ans=1;
	while(x){
		if(x%2)ans=ans*a%mod;
		a=a*a%mod;
		x=x/2;
	}
	return ans;
}
int main(){
	int a,b,n;
	scanf("%d%d%d",&a,&b,&n);
	ll ans=0;
	p[0]=r[0]=1;
	for(int i=1;i<=n;i++){
		p[i]=p[i-1]*i%mod;
		r[i]=pow_mod(p[i],mod-2);
	}
	for(int i=0;i<=n;i++){
		int sum=a*i+(n-i)*b;
		bool pass=true;
		while(sum){
			if(sum%10==a||sum%10==b){sum/=10;continue;}
			else {pass=false;break;}
		}
		if(pass){
			ans=(ans+p[n]*r[i]%mod*r[n-i])%mod;
		}
	}
	cout<<ans<<endl;

}
