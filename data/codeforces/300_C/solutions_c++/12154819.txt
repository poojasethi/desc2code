#include <bits/stdc++.h>

using namespace std;
#define f(i, n) for (int i = 0; i < (int)(n); i++)
typedef long long ll;
ll fact[1000006],sum=0,j,mod=1e9+7;
int a,b,n;

ll powMOD(ll p,ll m){
if(m==1)return p%mod;
if(m&1)return (powMOD(p,m-1)*p)%mod;
return powMOD((p*p)%mod,m/2);
}

bool ch(int s){
if(s==0)return 1;
if(s%10!=a && s%10!=b)return 0;
return ch(s/10);
}


int main()
{
cin>>a>>b>>n;
fact[0]=1;
for(int i=1; i<1000001; i++)
    fact[i]=(i*fact[i-1])%mod;

for(int i=0; i<=n; i++){
if(ch(i*a+(n-i)*b)){
sum+=fact[n]*powMOD((fact[n-i]*fact[i])%mod,mod-2);
sum%=mod;
}
}
cout<<sum<<endl;
    return 0;
}
