#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#define ll long long
const ll mo=1000000007;
using namespace std;
ll a,b,n,ans,sum;
ll fgm(ll b,ll e,ll mo)
{
    ll ans=1;
    for (;e;e>>=1,b=(b*b)%mo)
        if (e&1) ans=(ans*b)%mo;
    return ans;
}
bool check(ll c)
{
    for (;c;c/=10)
        if ((c%10!=a) && (c%10!=b)) return 0;
    return 1;
}
int main()
{
    cin>>a>>b>>n;
    ans=0,sum=1;
    for (int i=0;i<=n;i++) {
        if (check(i*a+(n-i)*b)) ans=(ans+sum)%mo;
        sum=(sum*(n-i)%mo)*fgm(i+1,mo-2,mo)%mo;
    }
    cout<<ans;
    return 0;
}
