#include<cstdio>

const long long MOD=1000000007;

long long inv(long long a)
{
	long long res=1;
	for(int b=MOD-2;b;b>>=1)
	{
		if(b&1)
			res=res*a%MOD;
		a=a*a%MOD;
	}
	return res;
}
