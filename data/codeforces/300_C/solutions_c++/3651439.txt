#include <iostream>

#define ff(i,a,b) for ((i)=(a);(i)<=(b);(i)++)
#define fd(i,a,b) for ((i)=(a);(i)>=(b);(i)--)
#define rep(i,n) for ((i)=0;(i)<(n);i++)
#define max(a,b) (a)>(b)?(a):(b)
using namespace std;

int a,b,i,n;
long long ans,ss,c[1000010];
const int p=1000000007;

void work(int w)
{
    while(w) if (w%10!=a && w%10!=b) return; else w/=10;
    ans= (ans+ss*(c[i]*c[n-i]%p)) % p;
}

int main()
{
    c[0]=c[1]=ss=1;
    cin>>a>>b>>n;
    ff(i,2,n) c[i]= c[p%i]*(p-p/i) % p;
    ff(i,2,n) c[i]= c[i-1]*c[i] %p;
    ff(i,2,n) ss = ss*i % p;
    ff(i,0,n) work(a*i+b*(n-i));
    cout<<ans<<endl;
}
