#include <stdio.h>
const int M=1000000007;
int fact[1000001]={1};
int a,b,n;
bool check(int n){
    while(n){
        if(n%10!=a && n%10!=b)
            return false;
        n/=10;
    }
    return true;
}
int POW(int b,int p){
    if(!p)
        return 1;
    long long x=POW(b,p/2);
    if(p&1)
        return x*x%M*b%M;
    return x*x%M;
}
int main()
{
    for(int i=1;i<1000001;++i)
        fact[i]=(long long)i*fact[i-1]%M;
    scanf("%d%d%d",&a,&b,&n);
    long long sum=0;
    for(int n1=0,n2;n1<=n;++n1){
        n2=n-n1;
        if(check(n1*a+n2*b))
            sum=(sum+(long long)fact[n]*POW(fact[n1],M-2)%M*POW(fact[n2],M-2))%M;
    }
    printf("%lld",sum);
    return 0;
}
