#include<cstdio>
const int MOD = 1e9 + 7;
typedef long long LL;
const int maxn = 1e6 + 5;
LL inv[maxn];
int n, a, b;
bool check(int x, int y)
{
	int sum = x*a + y*b;
	while (sum)
	{
		int tmp = sum % 10;
		sum /= 10;
		if (tmp != a&&tmp != b) return 0;
	}
	return 1;
}
int main()
{

	LL ans = 0;
	scanf("%d %d %d", &a, &b, &n);
	inv[1] = 1;
	for (int i = 2; i <= n + 1; i++) inv[i] = (MOD - MOD / i) * inv[MOD%i] % MOD;
	LL c = 1;
	for (int i = 0; i <= n; i++)
	{
		if (i) c = c*(n - i + 1) % MOD*inv[i] % MOD;
		if (check(i, n - i)) ans += c;
	}
	ans %= MOD;
	printf("%lld\n", ans);
	return 0;
}
