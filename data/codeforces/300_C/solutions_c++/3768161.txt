#include<cstdio>
#include<cstring>
#include<cstdlib>
using namespace std;
int mod=1000000007;
int C[1100000],n,a,b,ans;
inline int mul(int a,int b)
{
	int res=1;
	for (;b;b/=2,a=a*(long long)a%mod)
		if (b%2)
			res=res*(long long)a%mod;
	return res;
}
bool check(int u)
{
	int F=1;
	for (;u&&F;u/=10)
		if (u%10!=a&&u%10!=b)
			F=0;
	return F;
}
int main()
{
	scanf("%d%d%d",&a,&b,&n);
	C[0]=1;
	for (int i=1;i<=n;++i)
		C[i]=(C[i-1]*(long long)mul(i,mod-2))%mod*(long long)(n-i+1)%mod;
	for (int i=0;i<=n;++i)
		if (check(i*a+(n-i)*b))
			ans=(ans+C[i])%mod;
	printf("%d\n",ans);
}
