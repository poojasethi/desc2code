#include<bits/stdc++.h>
#define int long long
using namespace std;
const int maxn=1000*1000+10,mod=1000*1000*1000+7;
int a,b,n,ans,f[maxn];
bool check(int x)
{
    while(x>0)
    {
        if(x%10!=a && x%10!=b) return 0;
        x/=10;
    }
    return 1;
}
int power(int x,int y)
{
    if(y==0) return 1;
    int z=power(x,y/2);
    if(y%2==0) return (z*z)%mod;
    return ((z*z)%mod*x)%mod;
}
main()
{
    cin>>a>>b>>n;
    f[0]=1;
    for(int i=1; i<=n; i++) f[i]=(f[i-1]*i)%mod;
    for(int i=0; i<=n; i++)
        if(check((n-i)*a+i*b))
            ans=(ans+(f[n]*power((f[i]*f[n-i])%mod,mod-2))%mod)%mod;
    cout<<ans<<'\n';
    return 0;
}
