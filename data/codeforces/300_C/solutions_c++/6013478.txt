#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
#define mod 1000000007
__int64 ans,a,b,n,f[1000001];
__int64 h(__int64 x) {
	__int64 y=1,i;
	for (i=mod-2;i;i=i/2,x=x*x%mod)
		if(i&1)y=y*x%mod;
	return y;
}
bool c(int x) {
  	while(x>0){
	  	if(x%10!=a&&x%10!=b)return false;
		x/=10;
	}
  	return true;
}
__int64 i;
int main() {
  	scanf("%I64d%I64d%I64d",&a,&b,&n);
  	f[0]=1;
  	for(i=1;i<=n;i++)f[i]=f[i-1]*i%mod;
  	for(i=0;i<=n;i++)
    	if(c(i*a+(n-i)*b))ans=(ans+h(f[i])*h(f[n-i]))%mod;
  	printf("%I64d\n",ans*f[n]%mod);
}
