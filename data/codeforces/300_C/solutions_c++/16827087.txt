#include<bits/stdc++.h>
#define ll long long int
#define M 1000000007
using namespace std;
ll n,a,b,c[1000100],res;
bool is_ex(ll i){
	bool f=1;
	ll j=i*a+(n-i)*b;
	while(j){
		if(j%10!=a&&j%10!=b){
			f=0;
			break;
		}
		j/=10;
	}
	return f;
}
ll pmod(ll aa,ll bb,ll nn){
	ll rs=1;
	while (bb)
	{
 		if(bb%2)rs=(rs*aa)%nn; 
	    aa=(aa*aa)%nn;
		bb/=2;
	}
	return rs;
}
ll comb(ll k){
	ll x=(c[k]*c[n-k])%M;
	ll p=pmod(x,M-2,M);
	return (c[n]*p)%M;
}
int main(){
	cin>>a>>b>>n;
	c[0]=c[1]=1;
	for(int i=2;i<=1000000;i++){
		c[i]=(i*c[i-1])%M;
	}
	for(int i=0;i<=n;i++){
		if(is_ex(i)){
			res+=comb(i);
			res%=M;
		}
	}
	cout<<res<<endl;
}
