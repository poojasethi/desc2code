#!/usr/bin/python

def perfect(x):
	global a, b
	'''
	if all(int(i) == a or int(i) == b for i in str(x)):
		return True
	return False
	'''
	while x > 0:
		k = x % 10
		if k !=a and k != b:
			return False
		x/=10
	return True

def deg(a, p):
	global MOD
	ret = 1
	while p > 0:
		if p % 2 == 1:
			ret*=a
		a*=a
		p/=2
		a%=MOD
		ret%=MOD

	return ret

def C(x, y):
	global f, MOD, n;
	r = f[x] * f[y] % MOD
	r = deg(r, MOD-2)
	r*=f[n]
	return r%MOD

a, b, n = map(int, raw_input().split())
MOD = 10**9 + 7
f = [1 for _ in xrange(n+1)]

for i in xrange(2, n+1, 1):
	f[i] = f[i-1] * i % MOD

ret = 0
for i in xrange(n+1):
	if perfect(i * a + (n-i)*b):
		ret+=C(i, n-i)
		ret = (ret - MOD) if ret >= MOD else ret

print ret
