a,b,n=map(int,raw_input().split());
md=1000000007;
fact=[1]*(n+1)
for i in xrange(1,n+1):
	fact[i]=fact[i-1]*i%md;

digi=[a,b];

def check(n):
	while(n>0):
		if(n%10 not in digi):
			return 0;
		n=n/10;
	return 1;

ans=0
for i in xrange(n+1):
	if(check(a*i+b*(n-i))):
		ans=(ans+pow(fact[i]*fact[n-i],md-2,md))%md;
print (ans*fact[n])%md
