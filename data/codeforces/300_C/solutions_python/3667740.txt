ans = 0
mod = 1000000007
a, b, n = map(int, raw_input().split())
def chk(x):
    while x:
        if x % 10 != a and x % 10 != b:
            return 0
        x /= 10
    return 1
f = [1]
for i in range(1, n + 1):
    f.append(i * f[-1] % mod)
for x in range(n + 1):
    if chk(x * a + (n - x) * b):
        ans += pow(f[x] * f[n - x], mod - 2, mod)
        ans %= mod
print(ans * f[n] % mod)
