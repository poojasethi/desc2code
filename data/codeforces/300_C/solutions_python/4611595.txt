from sys import stdin


def main():
    mod = 10 ** 9 + 7
    a, b, n = map(int, stdin.read().split())
    f = [1] * (n + 1)
    for i in xrange(1, n + 1):
        f[i] = i * f[i - 1] % mod

    def is_good(s):
        while s > 0:
            s, x = divmod(s, 10)
            if x != a and x != b:
                return False
        return True

    print f[n] * sum(pow(f[n - i] * f[i], mod - 2, mod) for i in xrange(n + 1) if is_good(i * a + (n - i) * b)) % mod

main()
