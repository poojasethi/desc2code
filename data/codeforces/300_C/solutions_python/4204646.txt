[a,b,n]=[int(x) for x in raw_input().split()]
mod=1000000007
fact=[1]*(n+1)
for i in xrange(1,n+1):
	fact[i]=fact[i-1]*i%mod

digit=[a,b]

def check(n):
	while n>0:
		if n%10 not in digit:
			return 0
		n/=10
	return 1

ans=0

for i in xrange(n+1):
    if check(a*i+(n-i)*b):
        ans=(ans+pow(fact[i]*fact[n-i],mod-2,mod))%mod
print fact[n]*ans%mod
