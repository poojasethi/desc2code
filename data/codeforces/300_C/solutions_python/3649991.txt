a, b, n = map(int, raw_input().split())
mod = 10 ** 9 + 7
fac = [1] * (n + 1)
for i in xrange(1, n + 1): fac[i] = fac[i - 1] * i % mod
v = (a, b)
ans = 0
def c(x):
    while x:
        if x % 10 not in v: return 0
        x /= 10
    return 1
for i in xrange(n + 1):
    if c(i * a + (n - i) * b):
        ans = (ans + pow(fac[i] * fac[n - i], mod - 2, mod)) % mod
print ans * fac[n] % mod
