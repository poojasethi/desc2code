def main():
    a, b, n = map(int, raw_input().split())
    mod = 10 ** 9 + 7
    f = [1] * (n + 1)
    for i in xrange(1, n+1):
        f[i] = f[i-1] * i % mod
    def ok(i):
        while i > 0:
            t = i % 10
            if t != a and t != b: return False
            i /= 10
        return True
    print f[n] * sum(pow(f[i] * f[n-i], mod - 2, mod) for i in xrange(n+1) if ok(a*i + b*(n-i))) % mod
main()
