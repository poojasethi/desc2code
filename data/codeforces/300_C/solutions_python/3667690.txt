ans = 0
mod = 1000000007
a, b, n = map(int, raw_input().split())
def chk(x):
    while x:
        if x % 10 not in (a, b):
            return 0
        x /= 10
    return 1
f = [1] * (n + 1)
for i in xrange(1, n + 1):
    f[i] = f[i - 1] * i % mod
for x in xrange(n + 1):
    if chk(x * a + (n - x) * b):
        ans = (ans + pow(f[x] * f[n - x], mod - 2, mod)) % mod
print(ans * f[n] % mod)
