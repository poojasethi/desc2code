def main():
    a,b,n=map(int,raw_input().split())
    r=0
    f=[1]*(n+1)
    M=10**9+7
    for i in xrange(n):f[i+1]=f[i]*(i+1)%M

    v=(a,b)
    for k in xrange(n+1):
        s=a*k+b*(n-k)
        B=0
        while s: 
            if s%10 not in v:B=1;break
            s/=10
        if B:continue
        r=(r+pow(f[n-k]*f[k],M-2,M))%M
    print r*f[n]%M
main()
