a, b, n = map(int, raw_input().split())
f = [1]*(n+1)
mod = 10**9+7
for i in xrange(1, n+1):  f[i] = f[i-1]*i%mod

v = [a, b]
def OK(x):
    while x > 0:
        if x%10 not in v:
            return 0
        x /= 10
    return 1

ans = 0
for i in xrange(n+1):
    if OK(a*i+(n-i)*b):
        ans = (ans+ pow(f[i]*f[n-i], mod-2, mod) )%mod
print f[n]*ans%mod
