def main():
    mod = 10 ** 9 + 7
    a, b, n = map(int, raw_input().split())
    f = [1] * (n + 1)
    for i in xrange(1, n + 1):
        f[i] = i * f[i - 1] % mod

    def ok(s):
        while s > 0:
            x = s % 10
            if x != a and x != b: return False
            s /= 10
        return True

    print f[n] * sum(pow(f[n - i] * f[i], mod - 2, mod) for i in xrange(n + 1) if ok(a*i + b*(n - i))) % mod

main()
