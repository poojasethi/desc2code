#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
struct stt{int a,b;}st[100000];
bool cmp(stt st1,stt st2)
{return st1.a<st2.a;}
priority_queue<int>qe1,qe2;
int main()
{
    int n,t=0,a=0,o;
    scanf("%d",&n);
    for(int i=0;i<n;++i)scanf("%d",&st[i].a);
    for(int i=0;i<n;++i)
    {scanf("%d",&st[i].b);t+=st[i].b;}
    sort(st,st+n,cmp);
    o=a=st[0].b;qe2.push(st[0].b);
    for(int i=1;i<n;++i)
    {
        if(st[i].a!=st[i-1].a)
        {
            while(!qe2.empty())
            {qe1.push(qe2.top());qe2.pop();}
            a=0;
        }
        else if(!qe1.empty())
        {
            a+=qe1.top();
            qe2.push(qe1.top());
            qe1.pop();
        }
        a+=st[i].b;o=max(o,a);
        qe2.push(st[i].b);
    }
    printf("%d\n",t-o);
}
