#include<bits/stdc++.h>

using namespace std;

class abc{public: int l,c;}A[100005];

int i,n,h,f[205],k,d,g,s[100005],j,l;

bool cmp(abc a,abc b){return a.l<b.l;}

main(){

    cin>>n;

    for(i=1 ; i<=n ; i++)
        cin>>A[i].l;
    for(i=1 ; i<=n ; i++)
        cin>>A[i].c;

    sort(A+1,A+n+1,cmp);

    for(i=1 ; i<=n ; i++)
        s[i]=s[i-1]+A[i].c;

    h=s[n];

    for(i=1 ; i<=n ; i=j+1){

        for(j=i ; A[j+1].l==A[i].l ; j++);

        k=j-i+1;
        d=i-k;
        g=s[n]-s[j];

        for(l=1 ; l<=200 and d>0 ; l++)
            if(d>f[l]){
                d-=f[l];
                g+=f[l]*l;
            }
            else {
                g+=d*l;
                d=0;
            }
        h=min(h,g);
        for(l=i ; l<=j ; l++)
            f[A[l].c]++;
    }

    cout<<h;

}
