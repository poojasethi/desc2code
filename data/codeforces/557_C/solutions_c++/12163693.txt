#include <bits/stdc++.h>
using namespace std;

const int MAX = 1e5 + 5;
int cnt[205], rep[MAX];
pair<int,int> p[MAX];
int main () {
	
	int n, sum = 0;
	cin >> n;

	for (int i = 0; i < n; ++i) {

		cin >> p[i].first;
		rep[p[i].first]++;
	}

	for (int i = 0; i < n; ++i) {

		cin >> p[i].second;
		sum += p[i].second;
	}
	
	sort(p, p + n);

	int mini = 1e8, ans, add;
	for (int i = 0; i < n; ++i) {

		ans = sum;
		add = rep[p[i].first] - 1;
		for (int j = 200; j > 0 && add > 0; --j) {

			int t = cnt[j];
			while (add > 0 && t != 0) {

				ans -= j;
				t--;
				add--;
			}
		}

		ans -= p[i].second;
		cnt[p[i].second]++;
		while (i < n - 1 && p[i].first == p[i + 1].first) {

			++i;
			ans -= p[i].second;
			cnt[p[i].second]++;
		}
       //cout << "ans"<<ans << "mini"<<mini;
		mini = min(mini, ans);
	}

	cout << mini;
return 0;	
}
