#include<iostream>
#include<utility>
#include<algorithm>
using namespace std;
int main()
{
	int n,sum=0,max=2000000000,count=0,s;
	cin>>n;
	int l[n],d[n],f[100001]={0};
	for(int i=0;i<n;i++)cin>>l[i];
	for(int i=0;i<n;i++)cin>>d[i],sum+=d[i],f[l[i]]+=d[i];
	int k[100001]={0};pair<int,int>a[n];
	for(int i=0;i<n;i++)
	{
		k[l[i]]++;
		a[i]=make_pair(d[i],l[i]);
	}
	sort(a,a+n);
	for(int i=0;i<=100000;i++)
	{
		if(k[i])
		{
			count=0;s=0;
			for(int j=n-1;j>=0;j--)
			{
				if(count==k[i]-1)break;
				if(a[j].second<i)s+=a[j].first,count++;
			}
			if(sum-s-f[i]<max)max=sum-s-f[i];
		}
	}
	cout<<max;
}
