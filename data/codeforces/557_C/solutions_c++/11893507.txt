#include<bits/stdc++.h>
using namespace std;
#define N 100004
int a[N],b[N],c[N][204],cgt[N],n,i,s[N];
int tgt,ans;
main()
{
    scanf("%d",&n);
    for (i=0;i<n;i++) scanf("%d",&a[i]),s[a[i]]++;
    for (i=0;i<n;i++) scanf("%d",&b[i]),c[a[i]][b[i]]++,tgt+=b[i];
    for (i=1;i<=1e5;i++)
        for (int j=1;j<=200;j++)
            c[i][j]+=c[i-1][j];
    ans=tgt;
    for (i=0;i<n;i++)
        cgt[a[i]]+=b[i];
    for (i=1;i<=1e5;i++)
        if (s[i]!=0)
    {
        int sc=s[i]*2-1;
        int ct=sc-s[i],tonggt=cgt[i];
        for (int j=200;j>=1;j--)
        {
            if (ct<=0) break;
            tonggt+=min(c[i-1][j],ct)*j;
            ct-=c[i-1][j];
        }
        ans=min(ans,tgt-tonggt);
    }
    printf("%d",ans);
}
