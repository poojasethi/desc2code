#include<stdio.h>
#include<algorithm>
int n,i,j,l[100001],d[100001],rd[100001],cnt[201],t,ans[100001],min=1999999999,x[100001];
int main(){
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&l[i]);
		x[l[i]]++;
	}
	for(i=0;i<n;i++)
		scanf("%d",&d[i]);
	for(i=0;i<n;i++)
		l[i]=l[i]*1000+d[i];
	std::sort(l,l+n);
	for(i=0;i<n;i++){
		d[i]=l[i]%1000;
		l[i]/=1000;
	}
	for(i=n-1;i>=0;i--)
		rd[i]=rd[i+1]+d[i];
	for(i=0;i<n;i++){
		if(i==0||l[i]!=l[i-1]){
			t=x[l[i]];
			int sum=0;
			sum+=rd[i+t];
			int z=i-t+1;
			if(z>0)
				for(j=1;j<=200;j++){
					if(z>cnt[j]){sum+=cnt[j]*j;z-=cnt[j];}
					else {sum+=j*z;break;}
				}
			ans[l[i]]=sum;
		}
		cnt[d[i]]++;
	}
	for(i=1;i<=100000;i++)
		if(x[i]>0)min=min>ans[i]?ans[i]:min;
	printf("%d",min);
}
