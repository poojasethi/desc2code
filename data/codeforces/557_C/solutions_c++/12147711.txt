#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int f[210];
struct node 
{
	int l,d;
}a[100005];

bool cmp(node x,node y)
{
	return x.l>y.l;
}
int main()
{
	int n,sum=0;	
	memset(f,0,sizeof(f)); 
	cin>>n;
	for(int i=0;i<n;i++) cin>>a[i].l;
	for(int i=0;i<n;i++) 
	{
		cin>>a[i].d; f[a[i].d]++; sum+=a[i].d;
	}
	sort(a,a+n,cmp);
	int ct=0,sm=0,ans=0;
	for(int i=0;i<=n;i++)
	{
		if(i==0 || a[i].l==a[i-1].l)
		{
			ct++; sm+=a[i].d; f[a[i].d]--;
		}
		else 
		{
			for(int j=200;j>0;j--)
			{
				if(f[j]>0) 
				{
					if(f[j]<=ct-1)  sm+=(j*f[j]),ct-=f[j];
					else sm+=j*(ct-1), ct=1;					
				}
				if(ct<=1) break;
			}
			ans=max(ans,sm);
			sm=a[i].d; ct=1; f[a[i].d]--;
		}				
	}
	printf("%d\n",sum-ans);
	return 0;
}
