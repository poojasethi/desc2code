#include <bits/stdc++.h>
#define ll long long 
using namespace std;


ll h[201+14];
ll k[201+14];
ll sum[1000000+50];
pair < ll, ll > q[1000000+50];
ll ans = 1000000000;

int main(){
	int n;
	cin >> n;	
	for (int i = 0; i < n; i++)
		cin >> q[i].first;
	for (int i = 0; i < n; i++)
		cin >> q[i].second;
		
	sort(q, q + n);
	
	for (int i = n - 1; i > -1; i--)
		sum[i] = sum[i + 1] + q[i].second;
	

	
	for (int i = 0; i < n; i++){
		ll t = 1;
		while (i + 1 < n && q[i].first == q[i + 1].first)
			i++, t++;
		ll tmp = sum[i + 1];
		t = i + 1 - (2 * t - 1);
		for (int j = 1; j < 201 && t > 0; j++){
			ll mini = min(t, h[j]);
			t -= mini;
			tmp += mini * j;
		}
		for (int j = i; j > -1 && q[j].first == q[i].first; j--)
			h[q[j].second]++;
		ans = min(ans, tmp);
	}
	cout << ans << endl;
	
	return 0;  
		
}
