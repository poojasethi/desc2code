#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define MAX 100021
int main()
{
	int n,l[MAX],maxx=0,cnt[MAX]={0},soll=2e9,temp,sum[MAX]={0},total=0;
	std::vector<int> diff[211],d[MAX];
	cin>>n;
	for (int i = 0; i < n; ++i) cin>>l[i],maxx=max(maxx,l[i]);
	for (int i = 0; i < n; ++i) cin>>temp, d[l[i]].push_back(temp),sum[l[i]]+=temp;
	for (int i = maxx; i >= 0; --i) sum[i] = sum[i] + sum[i+1];
	for (int i = 1; i <= maxx; ++i)
	{
		if(d[i].size()==0) continue;
		int cou = d[i].size(), sol = sum[i+1], local = total;
		for (int j = 1; j < 201; ++j)
		{
			if(local >= cou) sol += min(cnt[j],local - cou + 1) * j, local -= min(cnt[j],local - cou + 1);
			else break;
		}
		for (int j = 0; j<d[i].size(); j++) cnt[d[i][j]]++,total++;
		soll = min(soll,sol);
	}	
	cout<<soll<<endl;

	return 0; 

}
