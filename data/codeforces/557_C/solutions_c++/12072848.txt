#include<bits/stdc++.h>
#include<algorithm>
#include<vector>
using namespace std;
typedef struct node
{
	int l;
	int d;
}node;
bool sortbyl(node &left,node &right)
{
	return left.l<right.l;
}
int main()
{
	int n,j,i,sum=0;
	node a[100001];
	priority_queue<int>q1,q2;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&a[i].l);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i].d);
		sum+=a[i].d;
	}
	sort(a,a+n,sortbyl);
	q2.push(a[0].d);
	int temp=a[0].d,ans=a[0].d;
	for(i=1;i<n;i++)
	{
		if(a[i].l!=a[i-1].l)
		{
			temp=0;
			while(!q2.empty())
			{
				q1.push(q2.top());
				q2.pop();
			}
		}
		else if(!q1.empty())
		{
			temp+=q1.top();
			q2.push(q1.top());
			q1.pop();
		}
		q2.push(a[i].d);
		temp+=a[i].d;
		ans=temp>ans?temp:ans;
	}
	ans=sum-ans;
	printf("%d\n",ans);
return 0;
}
