#include<cstdio>
#include<iostream>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<string>
#include<map>
#include<set>
#include<vector>
#include<queue>
#include<stack>
#include<algorithm>
using namespace std;
int n,ans=0x3f3f3f3f,cnt[201],m[100001];
pair<int,int> ld[100001];
int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>ld[i].first;
		m[ld[i].first]++;
	}
	for(int i=0;i<n;i++){
		cin>>ld[i].second;
		cnt[ld[i].second]++;
	}
	sort(ld,ld+n);
	int j=n-1,sum=0;
	for(int i=n-1;i>=0;i--){
		while(j>=0 && ld[i].first==ld[j].first){
			cnt[ld[j].second]--;
			j--;
		}
		int num=i-j,del=max(0,i+1-2*num+1),t=sum;
		for(int e=1;e<=200;e++){
			t+=min(del,cnt[e])*e;
			del-=min(del,cnt[e]);
		}
		ans=min(ans,t);
		sum+=ld[j+m[ld[i].first]-num+1].second;
	}
	cout<<ans<<endl;
}
