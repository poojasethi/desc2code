#include<bits/stdc++.h>
using namespace std;
pair<int,int>p[100005];
int n,q[100005],h[100005],r=1e9;
multiset<int>s;
int main(){
	cin>>n;
	for(int i=0;i<n;i++)cin>>p[i].first;
	for(int i=0;i<n;i++)cin>>p[i].second;
	sort(p,p+n);
	for (int i=1;i<=n;i++)q[i]=q[i-1]+p[i-1].second;
	for(int i=n-1;i>=0;i--)h[i]=h[i+1]+p[i].second;
	for(int i=0;i<n;i++){
		int j=i+1;
		while (j<n&&p[j].first==p[j-1].first)
			j++;
		int e=h[j]+q[i],k=j-i-1;
		for (multiset<int>::iterator it=s.begin();it!=s.end()&&k;it++){
			k--;
			e+=*it;
		}
		r=min(r,e);
		while(i!=j)
			s.insert(-p[i++].second);
		i--;
	}
	cout<<r;
}
