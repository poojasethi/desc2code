#include<cstdio>
#include<algorithm>
using namespace std;

const int maxn = 100005, INF = 20000000;
int cnt[205], sum[maxn];
pair<int, int> a[maxn];

int main() {
	int n, ans=INF;
	scanf("%d", &n);
	for(int i = 0; i < n; i++) scanf("%d", &a[i].first);
	for(int i = 0; i < n; i++) scanf("%d", &a[i].second);
	sort(a,a+n);
	for(int i=n-1; i >= 0; i--) sum[i]= sum[i+1]+a[i].second;
	for(int i=0, j=0; i<n; i=j) {
		while(a[j].first == a[i].first) j++;
		int num=j-i, cost=sum[j], left=j-2*num+1;
		if(left>0)
			for(int k=1; k <= 200; k++) {
				int d=min(left, cnt[k]);
				cost += d*k;
				left -= d;
			}
		for(int k=i; k<j; k++) cnt[a[k].second]++;
		ans = min(ans, cost);
	}
	printf("%d\n", ans);
	return 0;
}
