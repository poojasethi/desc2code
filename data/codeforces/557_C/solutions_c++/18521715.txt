#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
#define MAXN 100001
#define MAXV 201
struct Leg{
	int l,d;
}lg[MAXN];
int cmp(Leg l1,Leg l2){return l1.l>l2.l;}

int cnt[MAXV];
int main(){
	int n;scanf("%d",&n);
	for(int i=1;i<=n;++i)scanf("%d",&lg[i].l);
	for(int i=1;i<=n;++i)scanf("%d",&lg[i].d);
	for(int i=1;i<=n;++i)++cnt[lg[i].d];
	sort(lg+1,lg+1+n,cmp);
	int sum=0;
	int ans=1000000000;
	for(int i=1;i<=n;++i){
		int t=i;
		while(i<n&&lg[i].l==lg[i+1].l)--cnt[lg[i++].d];
		--cnt[lg[i].d];
		int tp=i-t+1,lst=tp*2-1;
		int tmp=0;
		if(lst<n-t+1){
			int qmp=n-t+1-lst;
			for(int j=1;j<MAXV;++j)
				if(cnt[j]>=qmp){
					tmp+=j*qmp;
					break;
				}else{
					tmp+=j*cnt[j];
					qmp-=cnt[j];
				}
		}
		ans=min(ans,sum+tmp);
		for(int j=t;j<=i;++j)sum+=lg[j].d;
	}
	printf("%d\n",ans);
}
