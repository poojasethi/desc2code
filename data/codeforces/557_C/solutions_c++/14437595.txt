#include <bits/stdc++.h>
using namespace std;

int main() {
int i,j,k,l,r,t,o,z,m=1000000000,n;
cin>>n;
pair<int,int> a[n];
int c[300]={0};
for(i=0;i<n;i++)  cin>>a[i].first;
for(i=0;i<n;i++)  cin>>a[i].second;
sort(a,a+n);
long long sum[n];
sum[n-1]=a[n-1].second;
for(i=n-2;i>=0;i--)   sum[i]=sum[i+1]+a[i].second;
i=0;
while(i<n){
   j=i+1;
   while(a[i].first==a[j].first && j<n) j++;
   int cn=i+1-(j-i),rem;
   if(j>=n)
   rem=0;
   else rem=sum[j];
   if(cn<0) cn=0;
   for(int d=0;d<=200;d++) {
    if(cn>c[d]) {cn-=c[d]; rem+=d*c[d];}
    else {rem+=d*cn; break;}
   }
   for(int hi=i;hi<j;hi++)  c[a[hi].second]++;
   m=min(m,rem);
   i=j;
}
cout<<m;
return 0;
}
