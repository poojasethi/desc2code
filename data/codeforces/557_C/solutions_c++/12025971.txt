#include <iostream>
#include <algorithm>
#include <utility>
using namespace std;
int n, c[256], t = 2000020000, s = 0;
pair<int, int> v[100100];
int main() {
  cin >> n;
  for (int i = 0; i < n; ++i) cin >> v[i].first;
  for (int i = 0; i < n; ++i) cin >> v[i].second, ++c[v[i].second];
  sort(v, v + n);
  for (int i = n, j; i > 0; i = j) {
    int d = 0;
    for (j = i; j > 0 && v[j - 1].first == v[i - 1].first; --j) {
      d += v[j - 1].second;
      --c[v[j - 1].second];
    }
    int k = j - (i - j - 1), w = s;
    for (int u = 0; u < 256 && k > 0; ++u) {
      int x = min(c[u], k);
      w += x * u;
      k -= x;
    }
    t = min(t, w);
    s += d;
  }
  cout << t << endl;
}
