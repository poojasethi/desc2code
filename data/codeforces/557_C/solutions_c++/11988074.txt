#include<algorithm>
#include<iostream>
#include<cstdio>
using namespace std;
const int N=100005;
int a[N][205],L[N],D[N],n,k[205];
int main(){
	ios::sync_with_stdio(0);
	cin>>n;
	int tot=0;
	for(int i=1 ;i<=n;i++)cin>>L[i];
	for(int i=1 ;i<=n;i++)cin>>D[i];
	for (int i=1,l,d;i<=n;i++){
		l=L[i];d=D[i];tot+=d;
		a[l][0]++,a[l][d]++;
	}
	int ans=0,mm;
	for(int i=1;i<N;i++){
		if(!a[i][0])continue;
		--a[i][0];
		mm=0;
		for(int d=200;d>=1&&a[i][0];d--){
			if(a[i][0]>=k[d]){a[i][0]-=k[d];mm+=k[d]*d;}
			else {mm+=a[i][0]*d,a[i][0]=0;}
		}
		for(int d=200;d>=1;d--){
			k[d]+=a[i][d];mm+=d*a[i][d];
		}
		ans=max(ans,mm);
	}
	cout<<tot-ans<<endl;
}
