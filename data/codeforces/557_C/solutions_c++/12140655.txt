#include<bits/stdc++.h>
using namespace std;
int tonggt,n,l[100005],c[100005][205],cgt[100005]={0},d[100005],i,s[100005],j;
main()
{
    scanf("%d",&n);
    for (i=1;i<=n;i++)
        scanf("%d",&l[i]),s[l[i]]++;
    for (i=1;i<=n;i++)
        scanf("%d",&d[i]),c[l[i]][d[i]]++,tonggt+=d[i];
    for (i=1;i<=n;i++)
        cgt[l[i]]+=d[i];
    for (i=1;i<=1e5;i++)
        for (j=1;j<=200;j++)
            c[i][j]+=c[i-1][j];
    int ans=1e9;
    for (i=1;i<=1e5;i++)
        if (s[i]!=0)
    {
        int tgt=cgt[i];
        int sc=s[i]*2-1,ct=sc-s[i];
        for (j=200;j>=1;j--)
        {
            if (ct<=0) break;
            tgt+=min(c[i-1][j],ct)*j;
            ct-=c[i-1][j];
        }
        ans=min(ans,tonggt-tgt);
    }
    cout<<ans;
}
