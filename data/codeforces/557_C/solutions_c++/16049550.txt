#include <bits/stdc++.h>
using namespace std;
const int N = 1e5+5;
int a[N],sum[N],cc[N][201],d[N],C[N],csum[N];
int main(){
	int n,i,j,x;
	cin>>n;
	for(i=0;i<n;++i)
		cin>>a[i];
	for(i=0;i<n;++i)
		cin>>d[i];
	for(i=0;i<n;++i)
		cc[a[i]][d[i]]++,sum[a[i]]+=d[i],C[a[i]]+=1;
	for(i=1;i<=1e5;++i)
		for(j=1;j<=200;++j)
			cc[i][j]+=cc[i-1][j];
	for(i=1e5;i>=0;--i)
		sum[i] += sum[i+1],csum[i]= csum[i+1]+C[i];
	int ret=1e9+2;
	for(i=1e5;i>=0;--i){
		if(C[i]>0){
			int k = n+1-C[i]*2-csum[i+1];
			int temp = sum[i+1];
			for(j=1;j<=200&&k>0;++j){
				int mi = min(k,cc[i-1][j]);
				k-=mi;temp+=mi*j;
			}
			ret= min(ret,temp);
		}
	}
	cout<<ret<<endl;
}
