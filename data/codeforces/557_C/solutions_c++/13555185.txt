#include<bits/stdc++.h>

using namespace std;

#define N 100005

int M[N][200],T[N][200],M2[N],Q[N],C[N],M3[N],A[N],n,i,j,c,t,x,mn = N*204;
	
int main(){

	cin >> n;

	for(i=1;i<=n;i++)
		cin >> A[i];

	for(i=1;i<=n;i++){
		cin >> x;
		T[ A[i] ][x]++;
		Q[ A[i] ] += x;
		C[ A[i] ]++;
	}

	for(i=2;i<=N-4;i++)
		for(j=1;j<=200;j++)
			M[i][j] = M[i-1][j] + T[i-1][j];
	
	for(i=N-4; i ; i--)
	{
		M2[i] = M2[i+1] + Q[i+1];
	}

	for(i=2;i<=N-3;i++)
		M3[i] = M3[i-1] + C[i-1]; 


	for(i=1;i<=N-2;i++)
		if(C[i]){
			t = M2[i]; c = M3[i] - C[i] + 1;
	//cout << t << " " << c << " " << j << endl;
			for(j=1;j<=200;j++){
				if(c <= 0) break;
				c -= M[i][j];
				t += j * M[i][j];
			}
			t -= (-c) * (j-1);
			mn = min(mn , t);
		}

	cout << mn;
	return 0;
}
