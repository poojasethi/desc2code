#include<iostream>
#include<vector>

using namespace std;

const int maxn=100100;
int n,l[maxn],d,a[maxn],b[maxn],ans=1000010000,c,x,g[300];
vector<int> v[maxn];

int main()
{
  cin>>n;
  for(int i=0;i<n;i++)
    {
      cin>>l[i];
      b[l[i]]++;
    }
  for(int i=0;i<n;i++)
    {
      cin>>d;
      a[l[i]]+=d;
      v[l[i]].push_back(d);
    }
  for(int i=100000;i>=0;i--)
    {
      a[i]+=a[i+1];
      b[i]+=b[i+1];
    }
  for(int i=1;i<=100000;i++)
    {
      c=a[i+1];
      x=max(0,n-b[i+1]-(int)(v[i].size()*2)+1);
      for(int j=1;j<=200;j++)
	{
	  c+=(min(g[j],x)*j);
	  x-=min(g[j],x);
	}
      ans=min(ans,c);
      for(int j=0;j<v[i].size();j++)
	g[v[i][j]]++;
    }
  cout<<ans<<endl;
  //cin>>ans;
  return 0;
}
