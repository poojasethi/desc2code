#include <iostream>
#include <stdio.h>
#include <map>
#include <string.h>
#include <string>
#include <bitset>
#include <algorithm>
using namespace std;
struct Leg{
int l,d;
bool operator< (Leg t){
return l<t.l;
}
};

int main (){
Leg a[100010];
int cnt[100010],t,num;
memset(cnt,0,sizeof(cnt));
int n,i,j,k,total=0,maxlen,cost,ans=1e+5*201;
scanf("%d",&n);
for(i=0;i<n;i++)
scanf("%d",&a[i].l);
for(i=0;i<n;i++){
scanf("%d",&a[i].d);
total+=a[i].d;
}
sort(a,a+n);
i=0;
while(i<n){
maxlen=a[i].l;
num=0;
cost=total;
for(t=i;a[t].l==maxlen;t++){
cost-=a[t].d;
num++;
}
num--;
for(j=200;j>=1&&num>0;j--){
cost-=min(num,cnt[j])*j;
num-=cnt[j];
}
ans=min(ans,cost);
for(t=i;a[t].l==maxlen;t++)
cnt[a[t].d]++;
i=t;
}
printf("%d\n",ans);
  return 0;
}
