#include<bits/stdc++.h>
using namespace std;
#define For(i,n) for(int i=1;i<=n;i++)
#define Dor(i,n) for(int i=n;i>=1;i--)
#define Mem(a) memset(a,0,sizeof(a))
#define s(a) scanf("%d",&a)
#define p(a) printf("%d\n",a)
#define mod 1e9+9
#define maxn 100000+10
vector<int>v[maxn];
int cnt[maxn];
int ans=999999999;
int l[maxn];
int main()
{
	int n,x;s(n);int cost=0;
	For(i,n)
	{
		s(l[i]);
	}
	For(i,n)
	{
		s(x);v[l[i]].push_back(x);
		cnt[x]++;cost+=x;
	}
	Dor(i,1e5)
	{
		if(v[i].size()==0) continue;
		int need=v[i].size()-1;
		int cur=0;cur+=cost;
		for(int j=0;j<v[i].size();j++)
		{
			cnt[v[i][j]]--;
			cur-=v[i][j];
		}
		for(int j=200;j>0&&need;j--)
		{
			int tmp=min(need,cnt[j]);
			cur-=tmp*j;
			need-=tmp;
		}
		ans=min(ans,cur);
	}
	p(ans);
	return 0;
}
