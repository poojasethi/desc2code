#include <bits/stdc++.h>
using namespace std;
#define int long long

int h[201];
int k[201];
int sum[1000002];
pair < int, int > v[1000001];
int ans = 123412341234;

main()
{
	int n;
	cin >> n;	
	for (int i = 0; i < n; i++)
		cin >> v[i].first;
	for (int i = 0; i < n; i++)
		cin >> v[i].second;
		
	sort(v, v + n);
	
	for (int i = n - 1; i > -1; i--)
		sum[i] = sum[i + 1] + v[i].second;
	
	fill(h, h + 201, 0);
	
	for (int i = 0; i < n; i++)
	{
		int t = 1;
		while (i + 1 < n && v[i].first == v[i + 1].first)
			i++, t++;
		int tmp = sum[i + 1];
		t = i + 1 - (2 * t - 1);
		for (int j = 1; j < 201 && t > 0; j++)
		{
			int mini = min(t, h[j]);
			t -= mini;
			tmp += mini * j;
		}
		for (int j = i; j > -1 && v[j].first == v[i].first; j--)
			h[v[j].second]++;
		ans = min(ans, tmp);
	}
	cout << ans << endl;
}
