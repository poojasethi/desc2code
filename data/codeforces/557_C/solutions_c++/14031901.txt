#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int n;
	cin>>n;
	int l[100001],e[201]={0};
	for(int i=0;i<n;i++){
		cin>>l[i];
	}
	for(int i=0;i<n;i++){
		int x;
		cin>>x;
		l[i]=l[i]*1000+x;
		e[x]++;
	}
	sort(l,l+n);
	int y=n,x=y-1,sum=0,min=20000000;
	while(y!=0){
		int temp=l[y-1]/1000,ans=0;
		while(1){
			if(l[x]/1000==temp){
				x--;
			}else{
				break;
			}
		}
		ans+=sum;
		for(int i=x+1;i<y;i++){
			sum+=l[i]%1000;
			e[l[i]%1000]--;
		}
		int term=0;
		if((y-x-1)*2>y){
			term=0;
		}else{
			term=y-(y-x-1)*2+1;
		}
		int t=1;
		while(term>0){
			if(e[t]>0){
				if(term>e[t]){
					term-=e[t];
					ans+=e[t]*t;
				}else{
					ans+=term*t;
					term-=term;
				}
			}
			t++;
		}
		if(min>ans){
			min=ans;
		}
		y=x+1;
	}
	cout<<min<<endl;
	return 0;
}
