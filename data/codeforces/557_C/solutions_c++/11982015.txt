#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
#include<string>
#include<cstdlib>
using namespace std;

pair<int, int> a[100005];
int cnt[205], ps[100005];

int main()
{
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
		scanf("%d",&a[i].first);
	for(int i = 0; i < n; i++)
		scanf("%d",&a[i].second);
	sort(a,a+n);
	for(int i=n-1;i>=0;i--)
		ps[i] = ps[i + 1] + a[i].second;
	int ans = 1000000000;
	for(int i = 0; i < n; i++)
	{
		int j = i;
		while(a[j].first== a[i].first)
			j++;
		int t = j - i;
		i = j - 1;
		int sum = ps[j];
		int w = j - (2 * t - 1);
		if(w > 0)
		for(int j = 1; j <= 200; j++)
		{
			sum += min(w, cnt[j]) * j;
			w -= min(w, cnt[j]);
		}
		for(int k = j - t; k < j; k++)
			cnt[a[k].second]++;
		ans = min(ans, sum);
	}
	printf("%d\n",ans);
	return 0;
}
