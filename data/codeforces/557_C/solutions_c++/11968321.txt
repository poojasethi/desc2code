#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mp(x,y) make_pair(x,y)
#define pr pair<int,int>
pair<ll,ll> l[100005];
ll cnt[205],suf[100005];
ll i,j,k,ans,sum,n,c,rem;
int main()
{
	std::ios_base::sync_with_stdio(false);
	cin>>n;
	for(i=0;i<n;i++)
		cin>>l[i].first;
	for(i=0;i<n;i++)
		cin>>l[i].second;
	sort(l,l+n);
	for(i=n-1;i>=0;i--)
		suf[i]=suf[i+1]+l[i].second;
	ans=1000000000;
	for(i=0;i<n;){
		c=1,sum=0,k=i;
		i++;
		while(l[i].first==l[i-1].first) c++,i++;
		sum=suf[i];
		rem=i-2*c+1;
		if(rem>0)
			for(j=1;j<=200;j++){
				sum+=min(rem,cnt[j])*j;
				rem-=min(rem,cnt[j]);
			}
		for(j=k;j<i;j++)
			cnt[l[j].second]++;
		ans=min(ans,sum);
	}
	cout<<ans<<endl;

	return 0;
}
