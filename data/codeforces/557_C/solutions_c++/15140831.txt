#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=1e5+10;
pair<int,int>lp[mx];
int main(){
   int n,tot=0;
   scanf("%d",&n);
   for(int i=0;i<n;i++)scanf("%d",&lp[i].first);
   for(int i=0;i<n;i++){
      scanf("%d",&lp[i].second);
      tot+=lp[i].second;
   }
   sort(lp,lp+n);
   set<pair<int,int> >s;
   int best=tot;
   for(int i=0;i<n;){
      int l=i;
      int bet=0;
      while(i<n && lp[i].first==lp[l].first)bet+=lp[i++].second;
      int q=min(int(s.size()),i-l-1);
      set<pair<int,int> >::iterator it=s.begin();
      for(int j=0; j<q; j++,it++) bet+=-it->first;
      best=min(best,tot-bet);
      for(int j=l;j<i;j++){
         s.insert(make_pair(-lp[j].second,j));
      }
   }
   printf("%d\n",best);
   return 0;
}
