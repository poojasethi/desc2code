#include <stdio.h>
#include <algorithm>
#include <memory.h>
using namespace std;
pair<int,int> a[100005];
int n,e[205];

int main()
{
    int curl,cure;
    scanf("%d",&n);
    for(int i=0;i<n;i++)scanf("%d",&a[i].first);
    for(int i=0;i<n;i++){
        scanf("%d",&a[i].second);
        e[a[i].second]++;
    }
    sort(a,a+n);
    int j=n-1,cur=0,res=n*200;
    for(int i=n-1;i>=0;i--){
        while(j>=0 && a[i].first==a[j].first){
            e[a[j].second]--;
            j--;
        }
        int have=i-j;
        int del=max(0,i+1-2*have+1);
        int tmp=cur;
        for(int p=1;p<=200;p++){
            tmp+=min(del,e[p])*p;
            del-=min(del,e[p]);
        }
        res=min(res,tmp);
        cur+=a[i].second;
    }
    printf("%d",res);
    return 0;
}
