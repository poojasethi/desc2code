#include<cstdio>
#include<algorithm>
#include<vector>

using namespace std;

vector<int> v[100100];
int l[100100], n, x, cnt[201], cost = 0;
int ans = 1e9, cur;

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; ++i)
		scanf("%d", &l[i]);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &x);
		v[l[i]].push_back(x);
		++cnt[x], cost += x;
	}
	for (int i = 1e5; i > 0; --i) {
		if (v[i].size() == 0)
			continue;
		cur = 0;
		for (int j = 0; j < (int) v[i].size(); ++j) {
			--cnt[v[i][j]];
			cur -= v[i][j];
		}
		int need = v[i].size() - 1;
		cur += cost;
		for (int j = 200; j > 0 && need > 0; --j) {
			int tmp = min(cnt[j], need);
			need -= tmp;
			cur -= tmp * j;
		}
		ans = min(ans, cur);
	}
	printf("%d", ans);
	return 0;
}
