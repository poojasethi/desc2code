#include <algorithm>
#include <cstdio>

using namespace std;

struct tb {
	int l, d;
} t[100000];

bool operator<(tb a, tb b) {
	return a.l < b.l;
}

int cnt[200 + 2];

int main() {
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &t[i].l);
	}
	for (int i = 0; i < n; ++i) {
		scanf("%d", &t[i].d);
	}
	sort(t, t + n);
	int ans = 0;
	for (int i = 0; i < n;) {
		int j = upper_bound(t, t + n, t[i]) - t;
		int get = 0;
		int tot = j - i - 1;
		for (int k = 200; k >= 1; --k) {
			if (tot > 0) {
				int del = min(tot, cnt[k]);
				get -= del * k;
				tot -= del;
			} else {
				break;
			}
		}
		for (int k = i; k < j; ++k) {
			cnt[t[k].d]++;
			get -= t[k].d;
		}
		if (ans > get) {
			ans = get;
		}
		i = j;
	}
	for (int i = 0; i < n; ++i) {
		ans += t[i].d;
	}
	printf("%d\n", ans);
}
