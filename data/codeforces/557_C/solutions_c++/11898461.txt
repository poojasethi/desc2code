# include <iostream>
# include <cstdio>
# include <algorithm>
using namespace std;
int cnt[201]={0};

struct node
{
	int l,d;
}leg[100000+10];

bool cmp(node x,node y)
{
	return x.l>y.l;
}

int main(void)
{
	int n; cin>>n;
	for(int i=0;i<n;i++) scanf("%d",&leg[i].l);
	for(int i=0;i<n;i++) {scanf("%d",&leg[i].d);cnt[leg[i].d]++;}
	sort(leg,leg+n,cmp);
	int m=n,tot=0,sum=0,cur=0,ans=2147483647;
	for(int i=0;i<n;i++)
	{
		tot++; cnt[leg[i].d]--; sum+=leg[i].d;
		if(i==n-1||leg[i+1].l!=leg[i].l)
		{
			if(tot<=m/2)
			{
				int k=m+1-2*tot;
				for(int j=1;j<=200;j++)
					if(cnt[j])
						if(cnt[j]<k) {cur+=cnt[j]*j; k-=cnt[j];}
						else {cur+=k*j;break;}
			}
			ans=min(ans,cur);
			m-=tot; tot=0; cur=sum;
		}
	}
	printf("%d\n",ans);
	return 0;
}
