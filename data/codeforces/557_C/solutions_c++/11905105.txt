#include<cstdio>
#include<algorithm>
using namespace std;

int n, cur, nxt, ans;
pair<int, int> v[111111];
int sum[111111], cnt[222];

int main() {
	scanf("%d", &n);
	for(int i=1;i<=n;i++) scanf("%d", &v[i].first);
	for(int i=1;i<=n;i++) scanf("%d", &v[i].second);
	
	sort(v+1,v+n+1);
	
	for(int i=n;i>0;i--)
		sum[i] = sum[i+1] + v[i].second;
	
	ans = sum[1];
	cur = 0, nxt = 1;
	while(nxt <= n) {
		cur = nxt;
		while(v[nxt].first == v[cur].first) nxt ++; //这里本来可以用lower_bound求的，但是懒得了。。
		
		int tmp = sum[nxt], sz = nxt - cur;
		int rem = max((cur-1) - (sz-1), 0);
		for(int i=1;i<=200 && rem;i++) {
			int dec = min(cnt[i], rem);
			tmp += i * dec;
			rem -= dec;
		}
		ans = min(ans, tmp);
		
		for(int i=cur;i<nxt;i++)
			cnt[v[i].second] ++;
	}
	printf("%d\n", ans);
	
	return 0;
}
