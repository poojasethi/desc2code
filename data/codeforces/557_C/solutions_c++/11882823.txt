#include <stdio.h>
#include <vector>
#include <utility>
#include <algorithm>

using namespace std;

int main(){
	int n,i,j,k,d[201],b,x,y,z,m,s;

	scanf("%d", &n);

	vector< pair<int,int> > v;
	v.resize(n);

	for(i=0;i<n;i++)
		scanf("%d", &v[i].first);

	for(i=0,s=0;i<n;i++){
		scanf("%d", &v[i].second);
		s+=v[i].second;
	}

	//printf("%d\n", s);

	sort(v.begin(),v.end());

	for(i=0;i<201;i++)
		d[i]=0;

	for(i=0,m=s;i<n;){
		for(j=i+1,y=v[i].second;j<n && v[j].first==v[i].first;j++)
			y+=v[j].second;

		b=j-i;

		for(k=200,x=0;k>0 && b>0;k--)
			if(d[k]==0);
			else
				if(d[k]<b){
					x+=(k*d[k]);
					b-=d[k];
				}
				else{
					x+=(k*(b-1));
					b=0;
				}

		z=s-x-y;

		//printf("%d %d %d %d %d %d %d\n", i,j,k,x,y,z,m);

		if(z<m)
			m=z;

		for(;i<j;i++)
			d[v[i].second]++;
	}

	printf("%d\n", m);

	return 0;
}
