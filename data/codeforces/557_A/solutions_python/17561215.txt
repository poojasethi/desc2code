#!/usr/bin/python

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
l1, h1 = ia()
l2, h2 = ia()
l3, h3 = ia()

d1 = n - l2 - l3
d1 = min(d1, h1)
n = n - d1

d2 = n - l3
d2 = min(d2, h2)
n = n - d2

d3 = n
d3 = min(d3, h3)

print d1, d2, d3
