n=int(raw_input())
a,b=map(int, raw_input().split(' '))
c,d=map(int, raw_input().split(' '))
e,f=map(int, raw_input().split(' '))
p=n-a-c-e
while p>0:
    if b>a:
        a+=1
        p-=1
    elif d>c:
        c+=1
        p-=1
    elif f>e:
        e+=1
        p-=1
print str(a)+' '+str(c)+' '+str(e)
