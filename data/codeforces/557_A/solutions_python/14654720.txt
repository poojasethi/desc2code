n=input()
min1,max1=map(int,raw_input().split())
min2,max2=map(int,raw_input().split())
min3,max3=map(int,raw_input().split())

a1=min1
a2=min2
a3=min3

while a1<max1 and a1+a2+a3<n:
    a1+=1
while a2<max2 and a1+a2+a3<n:
    a2+=1    
while a3<max3 and a1+a2+a3<n:
    a3+=1
        
print a1,a2,a3
