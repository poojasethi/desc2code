n = raw_input()
n = int(n)

m = []
ans = []
for i in xrange(3):
	temp = map(int, raw_input().split(' '))
	m.append(temp)
	a, b = temp
	n -= a
	ans.append(a)
for i in xrange(3):
	if n-(m[i][1] - m[i][0])>=0:
		ans[i] += m[i][1] - m[i][0]
		n -= m[i][1] - m[i][0]
	else:
		ans[i] += n
		n = 0
		break
