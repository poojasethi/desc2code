n=int(raw_input())
min1,max1=map(int,raw_input().split())
min2,max2=map(int,raw_input().split())
min3,max3=map(int,raw_input().split())
a,b,c=0,0,0
a=min(max1,n)
b=min(max2,n-a)
if(b<min2):
	a-=min2-b
	b=min2
c=min(max3,n-a-b)
if(c<min3):
	d=min3-c
	if(b-min2>=d):
		b-=d
	else:
		d-=b-min2
		b=min2
		a-=d
	c=min3
print a,b,c
