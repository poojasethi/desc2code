n = int(raw_input())

limit = []
ans = []

for i in range(3):
	[a,b] = map(int,raw_input().split(' '))
	n -= a
	ans.append(a)
	limit.append(b-a)

# print n
# print ans
# print limit

for i in range(3):
	if limit[i] > n:
		ans[i] += n
		break
	else:
		ans[i] += limit[i]
		n -= limit[i]

print ' '.join(map(str,ans))
