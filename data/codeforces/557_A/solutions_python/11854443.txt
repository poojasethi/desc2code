n = int(raw_input())
p1 = map(int, raw_input().split(" "))
p2 = map(int, raw_input().split(" "))
p3 = map(int, raw_input().split(" "))
min1 = p1[0]
max1 = p1[1]
min2 = p2[0]
max2 = p2[1]
min3 = p3[0]
max3 = p3[1]
a = min(n - min2 - min3, max1)
n -= a
b = min(n - min3, max2)
c = n - b 
print a, b, c
