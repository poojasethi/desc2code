n = int(raw_input())
limits = [map(int, raw_input().split()) for _ in range(3)]
cur = [val[0] for val in limits]
more = n - sum(cur)
aval = [limits[i][1] - cur[i] for i in range(3)]
for i in range(3):
    moreofthis = min(aval[i], more)
    cur[i] += moreofthis
    more -= moreofthis
print(' '.join(map(str, cur)))
