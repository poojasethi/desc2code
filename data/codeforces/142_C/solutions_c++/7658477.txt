#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <vector>
#define maxn 10
using namespace std;
int n,m,ans;
char s[maxn][maxn],a[maxn][maxn];
int d[4][5][2]=
{
	0,0,0,1,0,2,1,1,2,1,
	0,0,1,0,2,0,1,1,1,2,
	0,2,1,0,1,1,1,2,2,2,
	0,1,1,1,2,0,2,1,2,2,
};
bool check(int x,int y)
{
	if(x<1||x>n||y<1||y>m)
		return 0;
	else
		return a[x][y]=='.';
}
void dfs(int x,int y,int cur,int rest)
{
	if(rest/5+cur<=ans)
		return;
	if(y==m+1)
		x++,y=1;
	if(x==n+1)
	{
		if(cur>ans)
		{
			ans=cur;
			memcpy(s,a,sizeof(s));
		}
		return;
	}
	for(int i=0;i<4;i++)
	{
		bool ok=1;
		for(int j=0;j<5;j++)
		{
			if(!check(x+d[i][j][0],y+d[i][j][1]))
			{
				ok=0;break;
			}
		}
		if(ok)
		{
			for(int j=0;j<5;j++)
				a[x+d[i][j][0]][y+d[i][j][1]]=cur+'A';
			if(i<2)
				dfs(x,y+1,cur+1,rest-5);
			else
				dfs(x,y+1,cur+1,rest-5-(a[x][y]=='.'));
			for(int j=0;j<5;j++)
				a[x+d[i][j][0]][y+d[i][j][1]]='.';
		}
	}
	dfs(x,y+1,cur,rest-(a[x][y]=='.'));
}
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) for(int j=1;j<=m;j++) s[i][j]=a[i][j]='.';
	ans=0;
	dfs(1,1,0,m*n);
	printf("%d\n",ans);
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			printf("%c",s[i][j]);
		}
		puts("");
	}
	//system("pause");
	return 0;
}
