n = int(raw_input())

values = map(int, raw_input().split())

m = max(values)
counts = [0] * m

for v in values:
	counts[v-1] += 1

for i in xrange(m-1):
	if counts[i] < counts[i+1]:
		print -1
		exit(0)

print counts[0]

counts2 = [0] * m
perm = [0] * n
for i, v in enumerate(values):
	counts2[v-1] += 1
	perm[i] = counts2[v-1]

print ' '.join(str(i) for i in perm)
