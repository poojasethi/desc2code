#include <iostream>
using namespace std;

int t[100001];
int c[100001];

int main(void){
	int n;
	cin>>n;

	for(int i=0;i<n;i++){
		cin>>t[i];
		c[t[i]]++;
	}

	int sum = 0;
	if(c[1] == 0){
		cout<<"-1\n";
		return 0;
	}
	for(int i=1;;i++){
		sum += c[i];
		if(sum == n) break;
		if(c[i] < c[i+1]){
			cout<<"-1\n";
			return 0;
		}
	}

	cout<<c[1]<<endl;
	for(int i=0;i<n-1;i++){
		cout<<c[t[i]]<<" ";
		c[t[i]]--;
	}
	cout<<c[t[n-1]]<<endl;

	return 0;
}
