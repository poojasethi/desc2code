#include <cstdio>
#define INF 100000+5

int N,nset[INF];
int seq[INF];

int main(){
	scanf("%d",&N);
	for (int i=0;i<INF;++i) nset[i] = 0;
	for (int i=1;i<=N;++i){
		scanf("%d",&seq[i]);
		++nset[seq[i]];
	}
	for (int i=1;i<=N;++i){
		if (!(nset[i] >= nset[i+1]) || seq[i] > N){
			printf("-1\n");
			return 0;
		}
	}
	printf("%d\n",nset[1]);
	for (int i=1;i<=N;++i){
		printf("%d ",nset[seq[i]]);
		--nset[seq[i]];
	}
	printf("\n");
	return 0;
}
