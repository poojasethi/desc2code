#include "iostream"
using namespace std;
int main()
{
	int v[ 100001 ]={};
	int c[ 100001 ];
	int n;
	cin >> n;
	for( int i = 0 ; i < n ; i++ ) { cin >> c[ i ]; v[ c[ i ] ]++; }
	for( int i = 1 ; i < 100000 ; i++ )
	if( v[ i ] < v[ i + 1 ] ) { cout <<-1<<endl; return 0; }
	cout << v[ 1 ] << endl;
	for (int i = 0; i < n; ++i)
		cout << v[ c[ i ] ]-- <<" ";
	cout<<endl;
}
