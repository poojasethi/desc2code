#include <iostream>
#include <cstdio>
using namespace std;
int n,c[100001],y[100001],b,m,i;
int main()
{
cin>>n;
for (i=0;i<n;i++)
{
cin>>c[i];
y[c[i]]++;
if (c[i]>m)
m=c[i];
}
b=y[1];
for (i=2;i<=m;i++)
{
	if (y[i]>y[i-1])
		{
		 cout<<-1;
		 return 0;
		}
	if (y[i]>b)
		b=y[i];				
} 
cout<<b<<endl;
for (i=0;i<n;i++)
{
	cout<<y[c[i]]<<" ";
	y[c[i]]--;
}



return 0;
}
