#include <stdio.h>
#include <stdlib.h>

int main()
{
	int a[100000], p[100002]={0}, n, i, max=0;

	scanf("%d", &n);

	for(i=0; i<n; i++) { scanf("%d", &a[i]); p[a[i]]++; if(a[i]>max) max=a[i]; }
	for(i=1; i<=max; i++) if((p[i]==0 && p[i+1]!=0) || p[i]<p[i+1]) { printf("-1\n"); return 0; } 
	printf("%d\n", p[1]);
	for(i=0; i<n; i++) p[i]=1;
	for(i=0; i<n; i++) {
		printf("%d ", p[a[i]]++);
	}
	printf("\n");
}
