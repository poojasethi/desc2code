#include <stdio.h>
int a[100010];
int main() {
	int n;
	scanf("%d", &n);
	int b[100010];
	for(int i = 0; i < n; i++) {
		scanf("%d", &b[i]);
		a[b[i]]++;
	}
	for(int i = 1; i < 100000; i++)
		if(a[i] < a[i + 1]) {
			printf("-1\n");
			return 0;
		} 
	printf("%d\n", a[1]);
	for(int i = 0; i < n; i++)
		if(a[b[i]]) {
			printf("%d ", a[b[i]]);
			a[b[i]]--;
		}
	printf("\n");
	return 0;
}
