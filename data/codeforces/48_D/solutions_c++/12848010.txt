#include<cstdio>
#include<algorithm>
#include<cstring>
#include<iostream>
using namespace std;
int n,i,f[200000],a[200000];
int main(){
	scanf("%d",&n);
	for(i=0;i<n;++i){
		scanf("%d",&a[i]);
		f[a[i]]++;
	}
	for(i=2;i<=100000;++i){
		if(f[i]>f[i-1]){
			printf("-1");
			return 0;
		}
	}
	printf("%d\n",f[1]);
	for(i=0;i<n;++i)a[i]=f[a[i]]--;
	printf("%d",a[0]);
	for(i=1;i<n;++i){
		printf(" %d",a[i]);
	}
	return 0;	
}
