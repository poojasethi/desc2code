#include <iostream>
using namespace std;

int C[100001] = { 0 };
int ans[100001] = { 0 };
int v[100001] = { 0 };

int main() {
	int n,val; cin>>n;
	for(int i=0;i<n;i++)
		cin>>v[i];

	for(int i=0;i<n;i++) {
		C[v[i]]++;
		ans[i] = C[v[i]];
	}

	for(int i=1;i<100000;i++) {
		if(C[i] < C[i+1])
		{
			cout<<-1<<endl;
			return 0;
		}
	}

	cout<<C[1]<<endl;
	int i;
	for(i=0;i<n-1;i++) cout<<ans[i]<<' ';
	cout<<ans[i]<<endl;
	return 0;
}
