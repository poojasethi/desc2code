#include <iostream>
using namespace std;
int a[100001], b[100001], c[100001], n, i, m;
int main()
{
	cin>>n;
	for (b[0]=999999,i=1; i<=n; i++) {
		cin>>a[i];
		if (a[i]<=n && a[i]>0) c[i]=++b[a[i]];
		else b[0]=-1;
	}
	for (i=1; i<=n; i++) if (b[i]>b[i-1]) {
		cout<<-1<<endl; return 0;
	} else m=max(m, b[i]);
	cout<<m<<endl;
	for (i=1; i<=n; i++) cout<<c[i]<<" ";
	cout<<endl;
	return 0;
}
