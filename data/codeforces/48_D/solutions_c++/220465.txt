#include<cstdio>

int main(void){
	int i;
	int n,p[100001],r[100001]={0};
	int c,f,max=0;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&p[i]);
		r[p[i]-1]++;
		if(p[i]-1>max)max=p[i]-1;
	}
	int pre=r[0];
	for(i=0;i<=max;i++){
		if(r[i]>pre||r[i]==0)break;
		pre=r[i];
	}
	if(i<max||r[i]>pre)printf("-1\n");
	else{
		printf("%d\n",r[0]);
		for(i=0;i<n-1;i++)printf("%d ",r[p[i]-1]--);
		printf("%d\n",r[p[n-1]-1]--);
	}
	return 0;
}
