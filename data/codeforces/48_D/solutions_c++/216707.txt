#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
int n,a[100005],b[100005]={0},i,k=0;
cin>>n;
for(i=1;i<=n;i++)
{
cin>>a[i];
b[a[i]]++;
k=max(k,a[i]);
}
for(i=2;i<=k;i++)
{
if(b[i]>b[i-1])
{
cout<<-1;
return 0;
}
}
cout<<b[1]<<endl;
for(i=1;i<=n;i++)
{
cout<<b[a[i]]<<" ";
b[a[i]]--;
}
return 0;
}
