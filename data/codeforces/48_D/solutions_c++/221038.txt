#include <cstdio>
int a[100010], c[100010];
int main()
{
	int n; scanf("%d",&n);
	int m=0;
	for (int i=0; i<n; ++i) {
		scanf("%d", a+i);
		c[a[i]]++;
		if (m < a[i]) m=a[i];
	}
	for (int i=2; i<=m; ++i) if (c[i] > c[i-1]) { puts("-1"); return 0; }
	printf("%d\n", c[1]);
	for (int i=0; i<n-1; ++i) {
		printf("%d ", c[a[i]]);
		c[a[i]]--;
	}
	printf("%d\n", c[a[n-1]]);
	return 0;
}
