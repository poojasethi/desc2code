#include <iostream>
using namespace std;
int main()
{
	long long int n;
	cin>>n;
	int a[100002]={0},b[100002]={0};
	for  (int i=0; i<n; i++)
	{
		cin>>a[i];
		b[a[i]]++;
	}
	for (int i=2; i<=100000; i++) if(b[i]>b[i-1]) 
	{
		cout<<-1;
		return 0;
	}
	cout<<b[1]<<endl;
	for (int i=0; i<n; i++)
	{
		b[a[i]]--;
		a[i]=b[a[i]]+1;
	}
	for (int i=0; i<n; i++) cout<<a[i]<<' ';
}
