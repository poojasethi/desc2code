#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
int n,i,a[120000],cnt[120000],ans[120000];
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i){
		scanf("%d",&a[i]);
		ans[i]=++cnt[a[i]];
	}
	for(i=1;i<=100000;++i)if(cnt[i+1]>cnt[i])return printf("-1"),0;
	printf("%d\n",cnt[1]);
	for(i=1;i<=n;++i)printf("%d ",ans[i]);
}
