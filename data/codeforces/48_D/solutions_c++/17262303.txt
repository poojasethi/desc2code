#include<cstdio>
#include<iostream>
using std::max;
const int MAXN = 2e5;
int a[MAXN], n;
int cnt[MAXN];
int main() {
	scanf("%d", &n);
	for(int i=0; i<n; ++i) {
		scanf("%d", a+i);
		++cnt[a[i]];
	}
	int ans = cnt[1];
	for(int i=1; i<1e5; ++i)
		if(cnt[i]<cnt[i+1]) {
			puts("-1");
			return 0;
		}
	printf("%d\n", ans);
	for(int i=0; i<n; ++i)
	{
		if(i) printf(" ");
		printf("%d", cnt[a[i]]--);
	}
	puts("");
	return 0;
}
