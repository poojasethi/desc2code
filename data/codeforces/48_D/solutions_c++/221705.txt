#include <cstdio>
int main() {
    int n;
    int a[100100], c[100100];
    scanf("%d", &n);

    for (int i = 1; i <= n; ++i) 
    {
        scanf("%d", &a[i]);
        ++c[a[i]];
    }
    for (int i = 2; i < 100100; ++i) 
    {
        if (c[i] > c[i - 1]) 
        {
            printf("-1");
            return 0;
        }
    }
    printf("%d\n", c[1]);
    for (int i = 1; i <= n; ++i) printf("%d%c", c[a[i]]--, i == n ? '\n' : ' ');

}
