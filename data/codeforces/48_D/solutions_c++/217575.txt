#include <cstdio>
#include <algorithm>

using namespace std;

int n,a[100000],b[100000]={},t;

int main()
{
    scanf("%d",&n);
    for (int i=0;i<n;++i)
    {
        scanf("%d",&t);
        b[t-1]++;
        a[i]=b[t-1];
    }
    for (int i=1;i<100000;++i)
        if (b[i]>b[i-1])
        {
            printf("-1");
            return 0;
        }
    printf("%d\n",b[0]);
    for (int i=0;i<n;++i)
        printf("%d ",a[i]);
    return 0;
}
