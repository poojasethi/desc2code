#include <stdio.h>
#define nmax 100100
int h[nmax];
int x[nmax];
int n;
int main()
{
	scanf("%d", &n);
	for(int i=0; i<nmax; i++) h[i] = 0;
	for(int i=0; i<n; i++)
	{
		scanf("%d", &x[i]);
		h[x[i]] ++;
	}
	for(int i=2; i<nmax; i++)
	{
		if(h[i] > h[i-1])
		{
			printf("-1\n");
			return 0;
		}
	}
	printf("%d\n", h[1]);
	for(int i=0; i<n; i++)
	{
		printf("%d ", h[x[i]] --);
	}
	printf("\n");
	return 0;
}
