#include <stdio.h>

int main(void) {
	int a[100005]={}, b[100005]={};
	int n;
	int s[100005];
	scanf("%d", &n);
	int flag=0;
	for(int i=0;i<n;i++) {
		scanf("%d", &s[i]);
		if(s[i] > n) flag=1;
		a[s[i]]++;
	}
	for(int i=2;i<=n;i++)
		if(a[i] > a[i-1]) flag=1;
	if(flag) puts("-1");
	else {
		printf("%d\n", a[1]);
		for(int i=0;i<n;i++) {
			printf("%d ", ++b[s[i]]);
		}
	}
	return 0;
}
