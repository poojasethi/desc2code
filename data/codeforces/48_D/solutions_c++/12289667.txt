#include<iostream>

using namespace std ;
int MX = 100000 ;
int n ; 
int cnt[100000+50], arr[100000+50]  ; 


int main(){
	cin>> n ;
	for ( int i = 0  ; i < n ; i++ ) {
		cin>>arr[i]; 
		
		cnt[arr[i]]++ ; 
	}
	
	for ( int i = 1 ;i < MX ; i++ ) 
		if ( cnt[i] < cnt[i+1]  ){ 
			cout<<-1 ;
			return 0; 
		}
	cout<<cnt[1]<<endl ;
	for ( int i = 0 ; i < n ; i++ ){
		cout<<cnt[arr[i]]<< " "  ; 
		cnt[arr[i]]--; 
	}
	cout<<endl ;
	return 0;
}
