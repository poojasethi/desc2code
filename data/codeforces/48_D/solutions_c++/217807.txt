#include<cstdio>

int N;
int perm[100000];
int count[100001];
int k;

int main(){
 int i;
 scanf("%d",&N);
 for(i=0;i<N;i++){
  scanf("%d",&k);
  if(k>N){ printf("-1\n"); return 0; }
  perm[i]=++count[k];
 }
 for(i=1;i<N;i++) if(count[i]<count[i+1]) break;
 if(i<N) printf("-1\n");
 else{
  printf("%d\n",count[1]);
  for(i=0;i<N-1;i++) printf("%d ",perm[i]);
  printf("%d\n",perm[i]);
 }

 return 0;
}
