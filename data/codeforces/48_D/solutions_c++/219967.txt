int main ()
{
 cin>>n;
 For(i,1,n)
  { 
   cin>>a[i];
   s[a[i]]++;
   u[i]=s[a[i]];
   if (ma<u[i]) ma=u[i];
   if (m<a[i]) m=a[i];
  }
 For(i,2,m)
 if (s[i]>s[i-1]) 
  {
   cout<<-1;
   exit(0);
  }
 cout<<ma<<endl;
 For(i,1,n) cout<<u[i]<<" ";
 return 0; 
}
