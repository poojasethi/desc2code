//BE NAMASH



#include<iostream>
using namespace std;
const int MN=110000;
int a[MN],n;
int nm[MN],v[MN];
int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];
		nm[a[i]]++;
		v[i]=nm[a[i]];
	}
	for(int i=1;i<MN-1;i++){
		if(nm[i]<nm[i+1]){
			cout<<"-1"<<endl;
			return 0;
		}
	}
	cout<<nm[1]<<endl;
	for(int i=0;i<n;i++){
		cout<<v[i]<<" ";
	}cout<<endl;
	return 0;
}
