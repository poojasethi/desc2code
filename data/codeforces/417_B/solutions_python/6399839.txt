input = raw_input
n = int(input())

user = [-1 for i in range(100001)]
for i in range(n):
    x, k = map(int, input().split())
    if user[k] != -1:
        if (user[k] + 1) == x:
            user[k] = x
        else:
            if user[k] < x:
                print('NO')
                exit()
    else:
        if x != 0:
            print('NO')
            exit()
        else:
            user[k] = 0

print('YES')
