from sys import stdin

n = int(stdin.readline())
l = map(int,stdin.read().split())
s = zip(l[::2],l[1::2])

d = {}

for x,k in s:
    zn = d.get(k,-1)
#    print zn
    if zn +1 == x :
        d[k] = x
    elif zn < x:
        print 'NO'
        exit()

print 'YES'
