#!/usr/bin/env python

import sys
import logging

n = input()

K = {}
flag = True
for _ in range(n): 
    x, k = map(int, raw_input().split())
    if k in K: 
        if K[k] < x - 1: 
            flag = False
        else: 
            K[k] = max(K[k], x)
    elif x > 0 and k not in K: 
        flag = False
    else:
        K[k] = 0

if flag: 
    print 'YES'
else: 
    print 'NO'
