from collections import defaultdict
def main():
    n = int(raw_input())
    D = defaultdict(lambda: 0)

    for i in xrange(n):
        x,k = map(int, raw_input().split())
        if x > D[k]:
            print "NO"
            return
        if x == D[k]:
            D[k] += 1

    print "YES"

if __name__ == "__main__":
    main();
