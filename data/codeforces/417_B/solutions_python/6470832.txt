#!/usr/bin/env python

__author__ = 'firago'

from sys import stdin
from itertools import izip
n = int(stdin.readline())
l = map(int,stdin.read().split())
#s = izip(islice(l,[::2],l[1::2])
it = iter(l)
iS = izip(it,it)

d = {}

for x,k in iS:
    zn = d.get(k)
#    print zn
    if zn is None :
        if  x==0:
            d[k] = x
        else:
            print 'NO'
            exit()
    elif zn +1 == x :
        d[k] = x
    elif zn >= x:
        continue
    else:
        print 'NO'
        exit()

print 'YES'
