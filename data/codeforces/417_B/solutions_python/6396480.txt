D = {}

def solve():
    N = input()
    for i in xrange(N):
        x, k = map(int, raw_input().split(" "))
        if k in D:
            if x > D[k] + 1:
                return False
            D[k] = max(x, D[k])
        else:
            if x != 0:
                return False
            D[k] = 0
    return True

if solve():
    print "YES"
else:
    print "NO"
