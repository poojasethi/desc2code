import sys, re

#f = open('..\\input.txt','r')
f = sys.stdin

correct = True
d = {}
n = int(f.readline())
for i in xrange(n):
    x, k = map(int, f.readline().strip().split())

    if k not in d:
        if x != 0:
            correct = False
            break
        d[k] = x
    else:
        if x > d[k] + 1:
            correct = False
            break
        d[k] = max(d[k], x)

print 'YES' if correct else 'NO'
