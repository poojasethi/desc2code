import sys,math

if __name__ == '__main__':
	input = int(raw_input())
	highest, found = {}, True
	for i in range(input):
		data = map(int,raw_input().split())
		if not found:
			continue
		if data[1] not in highest:
			if data[0] != 0:
				found = False
			highest[data[1]] = 0
		elif highest[data[1]] < data[0]-1:
			found = False
		else:
			highest[data[1]] = max(highest[data[1]],data[0])
	print 'NO' if not found else 'YES'
