from sys import stdin

def read_int():
    return [int(x) for x in stdin.readline().split()][0]
def read_ints():
    return [int(x) for x in stdin.readline().split()]
def char_list():
    return [c for c in stdin.readline()][:-1]
def toss():
    stdin.readline()

def compute(pairs):
    next_number = {}
    for x, k in pairs:
        if k not in next_number:
            next_number[k] = 0
        if x > next_number[k]:
            #print 'Wrong: X=%s, next=%s' % (x, next_number[k])
            return False
        if x == next_number[k]:
            next_number[k] += 1
    return True

n = read_int()
pairs = []
for _ in range(n):
    x, k = read_ints()
    pairs.append((x, k))

if compute(pairs):
    print 'YES'
else:
    print 'NO'
