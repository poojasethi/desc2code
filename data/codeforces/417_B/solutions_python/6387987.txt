import os, sys

if __name__ == "__main__" :
  n = int(raw_input())
  maxc = {}
  okay = True
  for i in range(n) :
    x, k = map(int, raw_input().split(' '))
    if not k in maxc :
      if x != 0 :
        okay = False
        break
      maxc[k] = 0
    xx = maxc[k]
    if xx + 1 < x :
      okay = False
      break
    if x == xx + 1 :
      xx += 1
    maxc[k] = xx

  if okay :
    print "YES"
  else :
    print "NO"
