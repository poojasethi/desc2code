from sys import stdin
from itertools import izip
n = int(stdin.readline())
l = map(int,stdin.read().split())
# s = izip(l[::2],l[1::2])
def iS():
    iterator = iter(l)
    try:
        while True:
            a = next(iterator)
            b = next(iterator)
            yield (a,b)
    except  StopIteration:
        pass
    del iterator



d = {}

for x,k in iS():
    zn = d.get(k)
#    print zn
    if zn is None :
        if  x==0:
            d[k] = x
        else:
            print 'NO'
            exit()
    elif zn +1 == x :
        d[k] = x
    elif zn >= x:
        continue
    else:
        print 'NO'
        exit()

print 'YES'
