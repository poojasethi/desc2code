def main():
	n = int(raw_input())
	flag = True
	dict = {}
	for i in range(n):
		ss = raw_input().split()
		x = int(ss[0])
		k = int(ss[1])
		if dict.has_key(k) == False:
			if x == 0:
				dict[k] = x
			else:
				flag = False
		else:
			if x > dict[k] + 1:
				flag = False
			elif x == dict[k] + 1:
				dict[k] = x

	if flag:
		print 'YES'
	else:
		print 'NO'


if __name__ == '__main__':
	main()
