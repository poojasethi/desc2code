#!/usr/bin/env python
# -*- coding: UTF-8 -*-
__author__ = 'firago'

from sys import stdin

n = int(stdin.readline())
l = map(int,stdin.read().split())
ks = l[1::2]
vl = l[::2]
st = [-1] * (n)
s = zip(vl,ks)

d = dict(zip(ks,st))
#print d
#print ks
#print st
for x,k in s:
    zn = d.get(k)
#    print zn, k
    if zn +1 == x :
        d[k] = x
    elif zn >= x:
        continue
    else:
        print 'NO'
        exit()

print 'YES'
