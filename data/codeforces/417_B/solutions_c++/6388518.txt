#include<iostream>
#include<vector>
#include<map>
#include<cstdio>
#include<algorithm>
using namespace std;
main()
{
	int n;
	cin>>n;
	map <int, map<int,bool> > res;
	int f=1;
	for(int i=0;i<n;i++)
	{
		int a,b;
		cin>>a>>b;
		res[b][a]=1;
		if(a>0 && res[b][a-1]==0)
			f=0;
	}
	if(f) cout<<"YES\n";
	else cout<<"NO\n";
}
