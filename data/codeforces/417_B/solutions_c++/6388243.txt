#include <cstring>
#include <cstdio>

int n,x,k,p[100009];

int max(int a,int b)
{
	return a>b?a:b;
}
int main()
{
	memset(p,-1,sizeof(p));
	scanf("%d",&n);
	for (int i=1;i<=n;++i)
	{
		scanf("%d%d",&x,&k);
		if (x<=p[k]+1) p[k]=max(p[k],x);
		else goto NO;
	}
	puts("YES");
	goto EXIT;
	NO:puts("NO");
	EXIT:return 0;
}
