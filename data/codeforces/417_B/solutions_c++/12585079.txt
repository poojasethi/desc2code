#include <bits/stdc++.h>

using namespace std;

const int N = 5e5 + 9;

int n, t, x;

int hash[N];

int main() {

	cin >> n;

	for (int i = 1; i <= n; i++) {
		cin >> t >> x;
		if (t > hash[x]) { cout << "NO" << endl; return 0; }
		hash[x] += (hash[x] == t);
	}

	cout << "YES" << endl;

	return 0;
}
