#include <bits/stdc++.h>
using namespace std;
int slv[100100]={0};
int main()
{
    bool f = 0;
    int n,i,x,k;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d%d",&x,&k);
        x++;
        if(x>slv[k]+1)f=1;
        slv[k]=max(slv[k],x);
    }
    cout<<(f?"NO":"YES");
    return 0;
}
