#include<cstdio>
#include<iostream>
#include<stdlib.h>
#include<cmath>
#include<cstring>
#include<algorithm>
using namespace std;

int main(){
	int i,m,x,k,a[120000];
	for (i=0;i<=110000;i++) a[i]=-1;
	cin>>m;
	for (i=1;i<=m;i++){
		cin>>x>>k;
		if (x>a[k]+1) {cout<<"NO"; return 0;}
		if (x==a[k]+1) a[k]++;
	}
	cout<<"YES";
}
