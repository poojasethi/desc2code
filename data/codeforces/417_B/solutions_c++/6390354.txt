#include <iostream>
#include <algorithm>
using namespace std;
int main()
{int n; cin>>n; int res[100111]; for(int i=0;i<100111;i++) res[i]=-1; bool ok=true;
for(int i=1;i<=n;i++)
{int x; int k; 
cin>>x>>k;
if(res[k]+1<x) ok=false;
else 
{if(x>res[k]) res[k]=x;
} 
}
if(ok) cout<<"YES";
else cout<<"NO";
return 0;
}
