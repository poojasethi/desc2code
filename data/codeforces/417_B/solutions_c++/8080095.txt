#include <iostream>
#define MAXN 100009
using namespace std;

int n;
int x,k;
int tab[MAXN];
int main() {
cin >> n;
for(int i=1;i<=100005;i++) tab[i] = -1;
for(int i=0;i<n;i++) {
cin >> x >> k;
if(x > tab[k]+1) {cout << "NO" << endl; return 0;}
if(x > tab[k]) tab[k] = x;
}
cout << "YES" << endl;
return 0;
}
