using namespace std;

int main()
{
	int t;

	int a[100005];
	for(int i=0;i<100005;i++)
		a[i]=-1;
	while(t--)
	{
		int q,p;
		cin>>p>>q;
		
		if((p==a[q]+1))
		{
			a[q]=p;
		}
		else if(p<=a[q]){}

		else
		{
			cout<<"NO"<<endl;
			return 0;
		}
	}
		cout<<"YES"<<endl;

		return 0;
}
