#include <iostream>
using namespace std;
#define MAXN 100010
int conta[MAXN];
int N;
int main() {
	cin >> N;
	int a, b;
	bool p=true;
	for(int i=0; i < N; i++) {
		cin >> a >> b;
		if(conta[b] > a) continue;
		else if(conta[b] == a) conta[b]++;
		else p=false;
	}
	if(!p) cout<<"NO\n";
	else cout<<"YES\n";
	return 0;
}
