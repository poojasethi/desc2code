#include <iostream>

using namespace std;

const int MAXN = 1e5 + 10;
int cnt[MAXN];

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++) {
		int x, k;
		cin >> x >> k;
		if (cnt[k] == x)
			cnt[k]++;
		else if (cnt[k] < x) {
			cout << "NO" << endl;
			return 0;
		}
	}
	cout << "YES" << endl;
	return 0;
}
