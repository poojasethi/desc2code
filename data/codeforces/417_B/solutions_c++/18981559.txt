#include<cstdio>
int a[100005];
int main()
{
   int n,flag=1;
   scanf("%d",&n);
   for (int i=0;i<n;i++)
   {
   	 int x,k;
   	 scanf("%d%d",&x,&k);
   	 if (x>a[k])
   	 {
 	   	flag=0;
 	   }
 	   else if (x==a[k]) a[k]++;
   }   
   if (flag) printf("YES\n");
   else printf("NO\n");
}
