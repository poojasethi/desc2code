#include <stdio.h>
#include <string.h>
int main()
{   int n,a[100005],flag=0,x,k;
	scanf("%d",&n);
	memset(a, 0, sizeof(a));
	while(n--)
	{
		scanf("%d%d",&x,&k);
		if(x<a[k])
		continue;
		else if(x==a[k])
		 a[k]++;
		else
		 {flag=-1;
		 }
	}
	if(flag==-1)
	printf("NO\n");
	else
	printf("YES\n");
	
}
