#include <stdio.h>
#include <memory.h>

int tab[100010];
int n;

int main() {
	memset(tab,0,sizeof tab);
	scanf("%d",&n);
	for(int i = 0; i < n;i++) {
		int x,k;
		scanf("%d %d",&x,&k);
		if(tab[k] == x) {
			tab[k]++;
			continue;
		}
		if(tab[k] > x) continue;
		printf("NO"); 
		return 0;
	}
	printf("YES");
	return 0;
}
