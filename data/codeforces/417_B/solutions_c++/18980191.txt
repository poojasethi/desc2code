#include<cstdio>
#include<cstring>
int p[100005];
int main()
{
	int n,i,x,y;
	scanf("%d",&n);
	memset(p,-1,sizeof(p));
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&x,&y);
		if(x>p[y]+1)
			break;
		if(x>p[y])
			p[y]=x;
	}
	if(i==n+1)
		puts("YES");
	else
		puts("NO");
	return 0;
}
