#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int u=1000010;
int du[u],fa[u],v[u],n,m,i,x,y,ans;
int get(int x)
{
	if(x==fa[x]) return x;
	return fa[x]=get(fa[x]);
}
int main()
{
	cin>>n>>m;
	for(i=1;i<=n;i++) fa[i]=i;
	du[1]=2;
	for(i=1;i<=m;i++)
	{
		scanf("%d%d",&x,&y);
		du[x]++,du[y]++,fa[get(x)]=get(y);
	}
	for(i=1;i<=n;i++)
		if(du[i]&1) v[get(i)]|=1,ans++;
	for(i=1;i<=n;i++)
		if(get(i)==i&&du[i]&&!v[i]) ans+=2;
	if(!v[get(1)]&&ans==2) ans=0;
	cout<<ans/2<<endl; 
}
