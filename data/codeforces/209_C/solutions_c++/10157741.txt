#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
int num,n,m,z,x,y,fa[1000005],ru[1000005],color[1000005],p,q;
int rn[1000005],ans;
bool c[1000005];
inline int findfather(int l){
	if (fa[l]==l) return l;
	else return fa[l]=findfather(fa[l]);
}
int main(){
	//freopen("cf209C.in","r",stdin);freopen("cf209C.out","w",stdout);
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i) fa[i]=i;
	for (int i=1;i<=m;++i){
		scanf("%d%d",&x,&y);
		++ru[x];++ru[y];
		p=findfather(x);q=findfather(y);
		if (p!=q) fa[p]=q;
	}
	z=0;
	for (int i=1;i<=n;++i)
	if (ru[i]!=0){
		c[findfather(i)]=true;
		if (ru[i]%2==1) ++rn[fa[i]];
	}
	y=0;
	for (int i=1;i<=n;++i)
	if (c[i]==true) {
		++z,y+=rn[i];
		if (rn[i]!=0) ans-=1;
	}
	if (ru[1]==0) ++z;
	ans+=y/2;
	if (z!=1 || y!=0) ans+=z;
	printf("%d",ans);
}
