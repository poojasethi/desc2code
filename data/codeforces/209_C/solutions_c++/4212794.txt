#include<iostream>
#include<cstdio>
using namespace std;
const int N = 1000006;
bool du[N],flag[N];
int cnt[N];
int p[N];
int find(int u) { return u == p[u] ? u : p[u] = find(p[u]);}
int main(){
	int n,m;
	cin >> n >> m;
	for(int i=1;i<=n;i++) p[i] = i;
	for(int i=0;i<m;i++){
		int x,y;
		cin >>x >> y;
		du[x]^=1; du[y]^=1;
		if(find(x)!=find(y)){
			p[p[x]] = y;
		}
		cnt[x] ++;
		cnt[y] ++;
	}
	int ans = 0,k = 0;
	for(int i=1;i<=n;i++){
		ans +=du[i];
		if(du[i]) flag[find(i)] = 1;
		cnt[find(i)] ++;
	}
	int s = 0;
	for(int i=1;i<=n;i++) {
		s+= (p[i] == i && (cnt[i] >1||i==1) );
		if(p[i] == i && flag[i]) k++;
	}
	if(s == 1 && k ==0) {cout<<0<<endl; return 0;}
	ans -= k*2;
	cout<<ans/2 + s <<endl;
}
