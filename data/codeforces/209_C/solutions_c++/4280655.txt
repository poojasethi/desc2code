#include<cstdio>
using namespace std;
int deg[1000005];
int oc[1000005];
int f[1000005];
int n,m;
int find(int x)
{   if (f[x]==0)
        return x;
    f[x]=find(f[x]);
    return f[x];
}
void joint(int x,int y)
{   int fx,fy;
    fx=find(x),fy=find(y);
    if (fx!=fy)
        f[fx]=fy;
}
int work()
{   int i,ans=0;
    for (i=1;i<=n;++i)
        if (deg[i]&1)
        {   oc[find(i)]=1;
            ++ans;
        }
    for (i=1;i<=n;++i)
        if (find(i)==i&&deg[i]>0&&oc[i]==0)
            ans+=2;
    if (oc[find(1)]==0&&ans==2)
        return 0;
    else return ans/2;
}
int main()
{   int i,x,y;
    scanf("%d%d",&n,&m);
    deg[1]=2;
    for (i=1;i<=m;++i)
    {   scanf("%d%d",&x,&y);
        ++deg[x],++deg[y];
        joint(x,y);
    }
    printf("%d\n",work());
    return 0;
}
