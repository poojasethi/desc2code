#include<cstdio>
int n,m,fa[1001111],u,v,d[1001111],f[1001111],h[1001111],cnt,sum,tmp;

int getfa(int x)
{
	if (fa[x]==x)
		return x;
	return fa[x]=getfa(fa[x]);
}

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1; i<=n; i++)
		fa[i]=i;
	for (int i=1; i<=m; i++)
	{
		scanf("%d%d",&u,&v);
		d[u]++; d[v]++;
		fa[getfa(u)]=getfa(v);
	}
	for (int i=1; i<=n; i++)
	{
		if (d[i])
			f[getfa(i)]++;
		if (d[i]%2)
			h[getfa(i)]++,sum++;
	}
	for (int i=1; i<=n; i++)
		if (getfa(i)==i)
		{
			if (f[i] || i==1)
				cnt++;
			if (h[i])
				tmp++;
		}
	printf("%d\n",(cnt>1)*cnt+sum/2-(cnt>1)*tmp);
	//while (1);
	return 0;
}
