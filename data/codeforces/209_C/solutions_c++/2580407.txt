#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#include<vector>
#define pb push_back
using namespace std;
const int maxn = 1000010;
vector<int> A[maxn];
static int n,m,ct,ans,num;
static bool vis[maxn];
void dfs(int x){
	if (vis[x]) return;
	vis[x]=true;num++;
	for (int i=0;i<A[x].size();i++)
		dfs(A[x][i]);
	if (A[x].size()%2!=0) ct++;
}
int main(){
//	freopen("209C.in","r",stdin);
//	freopen("209C.out","w",stdout);
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;i++){
		int x,y;scanf("%d%d",&x,&y);A[x].pb(y);A[y].pb(x);
	}
	memset(vis,false,sizeof(vis));
	ans=0;int pd=0;
	for (int i=1;i<=n;i++)
	if (vis[i]==false&&A[i].size()!=0||i==1){
		ct=0;
		dfs(i);
		pd++;
		if (ct==0) ans+=1;
		else ans+=ct/2;
	}
	if (pd==1) printf("%d\n",ct/2);else printf("%d\n",ans);
	return 0;	
}
