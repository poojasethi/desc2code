#include<stdio.h>
int i,n,m,u,v,g,r[1000010],d[1000010],c[1000010];
int fa(int x){return r[x]==x?x:r[x]=fa(r[x]);}
int get()
{
    int ret=0,c;
    bool ok=0;
    while(1)
    {
        c=getchar();
        if(c>='0'&&c<='9'){ret=(ret<<3)+ret+ret+c-'0';ok=true;}
        else if(ok)return ret;
    }
}
int main()
{
    n=get();m=get();
    for(i=1;i<=n;i++)r[i]=i;
    for(i=1;i<=m;i++)
    {
        u=get(),v=get();
        d[u]++,d[v]++;
        r[fa(u)]=fa(v);
    }
    for(i=1;i<=n;i++){g+=d[i]&1;c[fa(i)]+=d[i]&1;}
    for(i=1;i<=n;i++)if(r[i]==i&&(d[i]||i==1)&&!c[i])g+=2;
    if(c[fa(1)]==0&&g==2)g=0;
    printf("%d\n",(g>>1));
    return 0;
}
