#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <vector>
#include <algorithm>
using namespace std;

typedef long long LL;
const int N = 1000005;
vector<int> a[N];
int u[N];
int nc=0, odd=0, nce=0;
void dfs(int x) {
	u[x] = 1;
	if(a[x].size()%2==1) ++odd;
	nce += (int)a[x].size();
	for(int i=0;i<(int)a[x].size();i++) {
		int y = a[x][i];
		if(!u[y]) dfs(y);
	}
	return;
}

int main(void) {
	int n, m;
	scanf("%d%d", &n, &m);
	for(int i=0;i<m;i++) {
		int x, y;
		scanf("%d%d", &x, &y);
		a[x].push_back(y);
		a[y].push_back(x);
	}
	int odc=0, ans = 0;
	for(int i=1;i<=n;i++)
		if(!u[i]) {
			nce=0;
			odd=0;
			dfs(i);
			if(nce > 0 || i==1) {
				if(odd) ans += odd/2;
				else ans++;
				++nc;
				if(odd) ++odc;
			}
		}
	if(nc == 1 && odc == 0) ans--;
	printf("%d\n", ans);
	return 0;
}
