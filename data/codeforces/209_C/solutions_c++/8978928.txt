#include<bits/stdc++.h>
using namespace std;
const int maxn=1000000+10;
vector<int> g[maxn];
int n,m;
int vis[maxn];
int odV,odC,evC;
void dfs(int u){
	vis[u]=1;
	if(g[u].size()&1)
		++odV;
	for(int i=0;i<g[u].size();i++)
		if(!vis[g[u][i]])
			dfs(g[u][i]);
}
int main(){
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
#endif
	scanf("%d%d",&n,&m);
	if(m==0){
		puts("0");
		return 0;
	}
	while(m--){
		int u,v;
		scanf("%d%d",&u,&v);
		g[u].push_back(v);
		g[v].push_back(u);
	}
	for(int i=1;i<=n;i++)
		if(!vis[i]){
			int tmp=odV;
			dfs(i);
			if(odV-tmp)
				++odC;
			else if(g[i].size())
				++evC;
		}
	if(g[1].size()==0)
		++evC;
	if(evC==1 && odV==0)
		cout<<0<<endl;
	else
		cout<<evC+(odV>>1)<<endl;
	return 0;
}
