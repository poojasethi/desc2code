#include<cstdio>
#include<iostream>
#include<cstring>

const int N = 2000012;

using namespace std;

int n, m;
int fa[N], deg[N], num[N], s[N], l[N];

int get(int x) {
	int v = x, top = 0;
	while (fa[v] != v) s[top++] = v, v = fa[v];
	while (top) fa[s[--top]] = v;
	return v;
}

int main() {
	scanf("%d %d", &n, &m);
	for (int i = 1; i <= n; i++) fa[i] = i;
	while (m--) {
		int u, v;
		scanf("%d %d", &u, &v);
		deg[u]++; deg[v]++;
		u = get(u); v = get(v);
		if (u != v) fa[u] = v;
	}
	for (int i = 1; i <= n; i++) {
		if (deg[i] == 0 && i != 0) continue;
		if (deg[i] & 1) num[get(i)]++;
	}
	int ans = 0, cnt = 0;
	for (int i = 1; i <= n; i++)
		if (fa[i] == i && (deg[i] || i == 1)) {
			cnt++; if (num[i]) ans += num[i] - 2;
		}
	if (cnt == 1 && !num[get(1)]) { printf("0\n"); return 0; }
	ans /= 2; ans += cnt;
	printf("%d\n", ans);
}
