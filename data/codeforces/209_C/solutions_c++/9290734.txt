#include<cstdio>
using namespace std;
int n,m,dad[1000010],x,y,gs[1000010],b[1000010],ans,i;
int find(int x){return dad[x]==x?x:dad[x]=find(dad[x]);}
int main()
{
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;i++)dad[i]=i;gs[1]=2;
    for(i=1;i<=m;i++)
    {
        scanf("%d%d",&x,&y);gs[x]++;gs[y]++;
        if(find(x)!=find(y))dad[dad[x]]=dad[y];
    }for(i=1;i<=n;i++)ans+=gs[i]&1,b[find(i)]|=gs[i]&1;
    for(x=0,i=1;i<=n;i++)if(dad[i]==i&&gs[i])ans+=2-2*b[i],x++;
    if(x==1&&(!b[find(1)]))ans-=2;printf("%d\n",ans/2);
}
