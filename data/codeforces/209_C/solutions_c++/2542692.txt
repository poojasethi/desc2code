#include <cstdio>
const int ww=1000000;
int n,m,i,x,y,k,tot,ans,q[ww+1],e[ww+ww+1],r[ww+ww+ww+1],s[ww+1],l,c[ww+1];
bool p[ww+1];
void read(int &x){
char c;
for (c = getchar();c < '0' || c > '9';c = getchar());
x = c - '0';
for (c = getchar();c >= '0' && c <= '9';c = getchar()) x = x * 10 + c - '0';
}
void gc(int t)
{
	int v=r[m+m+t];p[t]=true;
	if (s[t]%2==1) ++k;
	while (v!=0)
	{
		if (!p[e[v]]) gc(e[v]);v=r[v];
	}
}
int main()
{
	read(n);read(m);
	for (i=1;i<=n;++i) q[i]=m+m+i;
	for (i=1;i<=m;++i)
	{
		read(x);read(y);e[++l]=y;r[q[x]]=l;q[x]=l;e[++l]=x;r[q[y]]=l;q[y]=l;++s[x];++s[y];
	}
	for (i=1;i<=n;++i) if ((!p[i])&&((r[m+m+i]!=0)||(i==1)))
	{
		k=0;++tot;gc(i);c[tot]=k;
	}
	if (tot>=2)
	{
		ans+=tot;for (i=1;i<=tot;++i) if (c[i]>=2) c[i]-=2;
	}
	for (i=1;i<=tot;++i) ans+=c[i]/2;
	printf("%d\n",ans);
}
