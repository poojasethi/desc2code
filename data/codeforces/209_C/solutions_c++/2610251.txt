#include <cstdio>
#include <iostream>
#include <cstring>
#include <cmath>

using namespace std;

const int N=1000012;
int du[N],n,m,x,y,ans=0,fa[N],a[N];

int Find(int x)
  {
    return fa[x]!=x?fa[x]=Find(fa[x]):x;
  }
int main()
  {
    scanf("%d%d",&n,&m);
    for (int i=1;i<=n;++i) fa[i]=i;
    for (int i=1;i<=m;++i) scanf("%d%d",&x,&y),++du[x],++du[y],fa[Find(x)]=Find(y);
    if (m==0) {puts("0");return 0;}
    for (int i=1;i<=n;++i) ans+=du[i]&1,a[Find(i)]|=du[i]&1;
    for (int i=2;i<=n;++i) 
      if (Find(i)==i && Find(i)!=Find(1) && du[i] && !a[Find(i)]) ans+=2;
    bool found=false;for (int i=2;i<=n;++i) if (Find(i)!=Find(1) && du[i]) found=true;
    if (found && !a[Find(1)]) ans+=2;
    printf("%d\n",ans/2);
  }
