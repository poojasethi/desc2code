#include <cstdio>

const int maxn = 1000010;
int n, m, x, y, fa[maxn], d[maxn];

int getanc(int x){ return fa[x] <= 0 ? x : fa[x] = getanc(fa[x]); }

int main(){
	for (scanf("%d%d", &n, &m); m--;){
		scanf("%d%d", &x, &y);
		d[x]++; d[y]++;
		x = getanc(x); y = getanc(y);
		if (x != y) fa[x] = y;
	}
	int ans = 0, cnt = 0, zero = 0;
	for (int i = 1; i <= n; i++) if (d[i] & 1){
		x = getanc(i); fa[x]--;
	}
	for (int i = 1; i <= n; i++) if (fa[i] <= 0){
		if (!d[i] && i > 1) continue;
		cnt++;
		if (fa[i] == 0) zero++;
		ans += -fa[i];
	}
	ans /= 2;
	if (cnt > 1) ans += zero;
	printf("%d\n", ans);
}
