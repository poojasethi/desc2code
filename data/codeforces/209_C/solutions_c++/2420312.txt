#include <cstdio>
#define rep(i,n) for (int i=0;i<n;i++)
int n,m,cnt,Cnt,x,y,fg,d[1010000],f[1010000],id[1010000];
int find(int x) {return f[x]==x?x:f[x]=find(f[x]);}
int main() {
	scanf("%d%d",&n,&m);
	rep(i,n) f[i]=i;
	rep(i,m) {
		scanf("%d%d",&x,&y);x--;y--;
		d[x]++;d[y]++;
		f[find(x)]=find(y);
	}
	rep(i,n) id[find(i)]+=d[i]&1;
	if (!id[f[0]]) fg=1;
	rep(i,n) if (f[i]==i&&(d[i]||!i)) {cnt++;if (!id[i]) id[i]=2;Cnt+=id[i];}
	if (cnt==1 && fg) Cnt=0;
	printf("%d",Cnt/2);
}
