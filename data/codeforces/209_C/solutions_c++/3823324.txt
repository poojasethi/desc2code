#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define N 2200000
using namespace std;
int x,y,i,n,len,m,du[N],res,next[N],e[N],head[N],sum,ans,M,K;
bool bo[N];
void dfs(int x){
	if(bo[x])return;
	bo[x]=1;
	res+=du[x]&1;
	for(int i=head[x];i;i=next[i])dfs(e[i]);
}
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;++i){
		scanf("%d%d",&x,&y);
		next[++len]=head[x]; head[x]=len; e[len]=y;
		next[++len]=head[y]; head[y]=len; e[len]=x;
		du[x]++; du[y]++;
	}
	for(i=1;i<=n;++i)if(i==1 || du[i]!=0 && !bo[i]){
		res=0;
		dfs(i);
		sum+=res;
		if(res==0)M++;
		else K++;
	}
	if(!K){
		if(M==1)ans=0;
		else ans=M;
	}
	else {
		ans=K-1; sum-=(K-1)*2;
		ans+=sum/2+M;
	}
	printf("%d\n",ans);
}
