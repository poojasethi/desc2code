#include<stdio.h>
#include<stdlib.h>
long n,m,i,x,y,ans;
bool b[1000010];
long f[1000010];
long tot[1000010];
long find(long x){return f[x]==x?x:f[x]=find(f[x]);}
int main(){
    scanf("%ld%ld",&n,&m);
    for (i=1;i<=n;i++)
        f[i]=i;
    for (tot[1]=2,i=1;i<=m;i++){
        scanf("%ld%ld",&x,&y);
        tot[x]++;tot[y]++;
        f[find(x)]=find(y);
        }
    for (i=1;i<=n;i++)
        if (tot[i]%2){
            ans-=b[find(i)]?0:2;
            b[f[i]]=1;
            ans++;
            }
    for (i=1;i<=n;i++)
        if (find(i)==i && tot[i])
            ans+=2;
    if (!b[find(1)] && ans==2) ans=0;
    printf("%ld\n",ans/2);
    return 0;
}
