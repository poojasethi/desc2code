#include <cstdio>
#include <iostream>
using namespace std;
long long ans;
int i,j,k,x,y,n,m,d[1000011],fa[1000011],odd[1000011],tot,dl[1000011],lef;
int root(int now)
{
        if (fa[now]==0) return now;
        fa[now]=root(fa[now]);
        return fa[now];
}
int main()
{
        scanf("%d%d",&n,&m);
        for (i=1;i<=m;i++)
        {
                scanf("%d%d",&x,&y);
                d[x]++,d[y]++;
                if (root(x)!=root(y)) fa[root(x)]=root(y);
        }
        for (i=1;i<=n;i++)
        {
                if (d[i]&1) odd[root(i)]++;
                if ((i==1 || d[i]>0) && root(i)==i) dl[++tot]=i;
        }
        for (i=1;i<=tot;i++)
                if (odd[dl[i]]) ans+=odd[dl[i]]/2;
                else                        ans+=1;
        if (tot==1 && odd[dl[1]]==0) ans=0;
        cout << ans;
        return 0;
}
