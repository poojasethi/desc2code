#include<cstdio>
#define MAXN 1010000

int Ans,n,m,x[MAXN],y[MAXN],f[MAXN],o[MAXN],Cnt[MAXN],bl[MAXN];

int Find(int a){return f[a]==a?a:f[a]=Find(f[a]);}

int main()
{	int i,j;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)f[i]=i;
	for(i=0;i<m;i++)
	{	scanf("%d%d",x+i,y+i);
		o[x[i]]++;
		o[y[i]]++;
		if(Find(x[i])!=Find(y[i]))
			f[f[x[i]]]=f[y[i]];
	}
	for(i=0;i<m;i++)bl[Find(x[i])]=1;
	bl[Find(1)]=1;
	//for(i=1;i<=n;i++)printf("%d %d\n",o[i],f[i]);
	for(i=1;i<=n;i++)if(o[i]&1)Cnt[Find(i)]++;
	for(i=1,j=0;i<=n;i++)if(f[i]==i&&bl[i])j++;
	if(j==1&&!Cnt[Find(1)])Ans=-2;
	for(i=1;i<=n;i++)
	if(f[i]==i&&bl[i])
	{	if(!Cnt[i])Cnt[i]=2;
		Ans+=Cnt[i];
	}
	printf("%d\n",Ans/2);
	return 0;
}
