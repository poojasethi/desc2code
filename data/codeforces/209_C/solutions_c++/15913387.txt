#include<bits/stdc++.h>
using namespace std;

int n,m,ans;
int fa[1000010],d[1000010],s[1000010];

int find(int i)
{return fa[i]==i?i:fa[i]=find(fa[i]);
}

int main()
{int i,j,x,y,fx,fy,flag;
 
 scanf("%d%d",&n,&m);
 for(i=1;i<=n;i++) fa[i]=i;
 for(i=1;i<=m;i++)
 {	scanf("%d%d",&x,&y);
 	d[x]++;
 	d[y]++;
 	fx=find(x);
 	fy=find(y);
 	if(fx!=fy) fa[fx]=fy;
 }
 for(i=1;i<=n;i++)
 {	s[find(i)]+=(d[i]&1);
 	ans+=(d[i]&1);
 }
 ans/=2;
 flag=1;
 for(i=2;i<=n;i++) 
 	if((i==1 || d[i]) && find(i)!=find(1)) 
	{	flag=0;
		break;
	}
 if(!flag)	
 {	for(i=1;i<=n;i++)
 		if(find(i)==i && (i==1 || d[i]) && s[i]==0) ans++;	
 }
 printf("%d\n",ans);
 return 0;
}
