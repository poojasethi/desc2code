#include<stdio.h>
int getint()
{
	int ret=0;bool ok=false;
	for(;;)
	{
		int c=getchar();
		if(c>='0'&&c<='9')ret=(ret<<3)+ret+ret+c-'0',ok=true;
		else if(ok)return ret;
	}
}
int n,m;
int r[1000010],d[1000010];
int fa(int x)
{
	return r[x]==x?x:r[x]=fa(r[x]);
}
int c[1000010];
int main()
{
	n=getint(),m=getint();
	for(int i=1;i<=n;i++)r[i]=i;
	for(int i=1;i<=m;i++)
	{
		int u=getint(),v=getint();
		d[u]++,d[v]++;
		r[fa(u)]=fa(v);
	}
	int cnt=0;
	for(int i=1;i<=n;i++)cnt+=d[i]&1,c[fa(i)]+=d[i]&1;
	for(int i=1;i<=n;i++)if(r[i]==i&&(d[i]||i==1)&&!c[i])cnt+=2;
	if(c[fa(1)]==0&&cnt==2)cnt=0;
	printf("%d\n",cnt/2);

	return 0;
}
