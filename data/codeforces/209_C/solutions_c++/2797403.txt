#include <cstring>
#include <cstdio>
using namespace std;
int f[1000010],d[1000010],flag[1000010],n,m,ans,cnt;

int findset(int x)
{
	if (f[x]==x) return x;
	return f[x]=findset(f[x]);
}

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1; i<=n; i++) f[i]=i;
	
	for (int i=1; i<=m; i++)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		d[x]++; d[y]++; 
		int f1=findset(x), f2=findset(y);
		f[f1]=f2;
	}
	
	for (int i=1; i<=n; i++)
		if (d[i]&1) ans+=1, flag[findset(i)]=1;
	for (int i=1; i<=n; i++)
	{
		if (d[i] && f[i]==i) cnt++;
		if (d[i] && f[i]==i && !flag[i]) ans+=2;
	}
	
	if (d[1]==0 && ans!=0) ans+=2;
	if (!flag[findset(1)] && cnt==1 && d[1]!=0) ans=0;
	printf("%d\n",ans/2);
	//for (;;);
}
