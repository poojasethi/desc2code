#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define N 1000001

int n,m;
int num,ans;
int du[N],f[N],sum[N],vis[N];

int find(int x) { return f[x] == x ? x:(f[x] = find(f[x]));}

int main()
{
    int i,j,k;

    scanf("%d %d",&n,&m);
    for (i = 1; i <= n; ++i) f[i] = i;
    for (i = 1; i <= m; ++i)
    {
        scanf("%d %d",&j,&k);
        ++du[j],++du[k];
        f[find(j)] = find(k);
    }
    vis[find(1)] = true;
    for (i = 1; i <= n; ++i)
        if (du[i]) vis[find(i)] = true,sum[find(i)] += du[i] & 1;
    for (i = 1; i <= n; ++i)
        if (vis[i]) ++num;
    if (num > 1) ans = num;
    for (i = 1; i <= n; ++i)
        if (sum[i] >= 2 && num > 1) ans += (sum[i]-2)/2;
        else ans += sum[i]/2;
    printf("%d",ans);

    return 0;
}
