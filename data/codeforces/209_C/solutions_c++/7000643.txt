#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>

using namespace std;

int f[1000100] , x , y , du[1000100] , n , m , num[1000100] , tot , rt ; 
int ans ;

int find(int x){ return f[x] == x ? x : f[x] = find(f[x]) ; }

int main(){

  
  cin >> n >> m ; 
  for(int i = 1 ; i <= n ; i ++ ) f[i] = i ; 
  for(int i = 1 ; i <= m ; i ++ ){
    scanf("%d %d",&x,&y) ; 
    du[x] ++ , du[y] ++ ;  
    int fx = find(x) , fy = find(y) ;
    if(fx != fy) f[fy] = fx ; 
  }
  
  for(int i = 1 ; i <= n ; i ++ ) 
    if(du[i] & 1) num[find(i)] ++ ; 
  
  for(int i = 1 ; i <= n ; i ++ )
    if(f[i] == i && du[i]){
      tot ++ ; 
      if(num[i]) ans += (num[i] - 2) / 2 , num[i] = 2 , rt ++ ; 
    }  
  
  tot += (du[1] == 0) ; 
  if(tot != 1 || rt ) ans += tot ; 
  cout << ans << endl; 
}
