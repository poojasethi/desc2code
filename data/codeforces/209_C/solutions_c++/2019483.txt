#include <cstdio>
#include <cstdlib>
#include <cstring>
int ans,n,m,cnt[2000000],b[2000000],f[2000000],g[2000000],tot;
int find(int x) {if (b[x]!=x) b[x]=find(b[x]);return b[x];}
void init()
{
  int i,x,y;
  scanf("%d%d\n",&n,&m);
  for (i=1;i<=n;i++) b[i]=i;
  for (i=1;i<=m;i++) {
    scanf("%d%d\n",&x,&y);
    cnt[x]++,cnt[y]++;
    x=find(x),y=find(y);
    b[x]=y;
  }
  for (tot=0,i=1;i<=n;i++) x=find(i),f[x]+=cnt[i],g[x]+=(cnt[i]&1);
  for (i=1;i<=n;i++) tot+=(b[i]==i && (f[i] || i==1));
  if (1==tot) {
    ans=0;
    for (i=1;i<=n;i++) ans+=(cnt[i]&1);
    printf("%d\n",ans/2);
  }
  else {
    ans=0;
    for (i=1;i<=n;i++) {
      if (b[i]!=i) continue;
      if (g[i]) ans+=((g[i]-2)>>1);
    }
    ans+=tot;
    printf("%d\n",ans);
  }
}
int main()
{
    init();
  return 0;
}
