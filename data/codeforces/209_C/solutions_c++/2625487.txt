#include<cstdio>
#include<iostream>
#include<cstring>
const int maxn=1000001;
using namespace std;
int f[maxn],q[maxn],p[maxn],n,m;
int getf(int x){
    if(f[x]==x)return x;
    else{
        f[x]=getf(f[x]);
        return f[x];
    }
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i=0;i<=n;i++)
        f[i]=i;
    for(int i=0;i<m;i++){
        int x,y;
        scanf("%d%d",&x,&y);
        p[x]++;
        p[y]++;
        x=getf(x);
        y=getf(y);
        f[x]=y;
    }
    int t=0;
    p[1]+=2;
	for(int i=1;i<=n;i++)
        if(p[i]%2==1){
            t++;
            int j=getf(i);
            if(q[j]%2==0)q[j]++;
        }
	for(int i=1;i<=n;i++)
		if(f[i]==i)
            if(p[i]!=0)
                if(q[i]==0)
                    t=2+t;
	t/=2;
    if(q[getf(1)]==0&&t==1)puts("0");
	else printf("%d",t);
	return 0;
}
