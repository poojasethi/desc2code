#include<cstdio>
#define N 2000005
int n,m,x,y,ans,l,t,cnt,d[N],son[N],next[N],ed[N]; bool b[N];
int dfs(int x)
{
	if (b[x]) return 0; b[x]=1; int res=d[x]&1;
	for (int p=son[x];p;p=next[p]) res+=dfs(ed[p]); return res;
}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;i++)
		scanf("%d%d",&x,&y),d[x]++,d[y]++,
		ed[++l]=y,next[l]=son[x],son[x]=l,
		ed[++l]=x,next[l]=son[y],son[y]=l;
	for (int i=1;i<=n;i++) if ((i<2)+d[i] && !b[i]) t=dfs(i),ans+=t/2+(t==0),cnt++;
	printf("%d\n",ans-(cnt<2 && t==0)); return 0;
}
