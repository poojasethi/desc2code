#include<iostream>
#include<cstdio>

using namespace std;

const int maxn = 1000010;

int n , m;
int par[maxn] , lev[maxn];
int odd[maxn] , size[maxn];
bool self[maxn];

int anc( int t ) {
	if ( par[t] != t ) par[t] = anc( par[t] );
	return par[t];
}

int main() {
	scanf( "%d%d" , &n , &m );
	for ( int i = 1 ; i <= n ; i++ ) par[i] = i;
	for ( int i = 1 ; i <= m ; i++ ) {
		int u , v; scanf( "%d%d" , &u , &v );
		if ( u == v ) self[u] = true;
		lev[u]++ , lev[v]++;
		par[anc( u )] = anc( v );
	}
	int ans = 0 , cnt = 0;
	for ( int i = 1 ; i <= n ; i++ ) {
		if ( lev[i] % 2 == 1 ) odd[anc( i )]++;
		size[anc( i )]++;
	}
	for ( int i = 1 ; i <= n ; i++ )
		if ( anc( i ) == i ) {
			if ( i != 1 && size[i] == 1 && !self[i] ) continue;
			cnt++;
			if ( odd[i] > 0 ) cnt = 2;
			ans += max( 1 , odd[i] / 2 );
		}
	if ( cnt == 1 ) ans = 0;
	printf( "%d\n" , ans );
	return 0;
}
