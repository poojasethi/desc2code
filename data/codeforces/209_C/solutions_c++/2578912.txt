#include<cstdio>
int N,M,i,X,Y,flag,res,cnt;
int Cnt[1000005],f[1000005],size[1000005];
int getfather(int i)
{
	if(f[i]==i) return i;
	f[i]=getfather(f[i]);
	return f[i];
}
int main()
{
	scanf("%d%d",&N,&M);
	for(i=1;i<=N;++i) f[i]=i;
	for(i=1;i<=M;++i)
	{
		scanf("%d%d",&X,&Y);
		++Cnt[X],++Cnt[Y];
		X=getfather(X),Y=getfather(Y);
		if(X!=Y) f[X]=Y;
	}
	for(i=1;i<=N;++i) f[i]=getfather(i);
	for(i=1;i<=N;++i) size[f[i]]+=Cnt[i]&1;
	if(!size[f[1]]) flag=1;
	for(i=1;i<=N;++i)
		if(f[i]==i&&(Cnt[i]||i==1))
		{
			++cnt;
			if(!size[i]) size[i]=2;
			res+=size[i];
		}
	if(cnt==1&&flag) res=0;
	printf("%d\n",res/2);
	return 0;
}
