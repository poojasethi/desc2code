#include <cstdio>
int prt[1000005],deg[1000005],read[1000005],sing[1000005],root[1000005];
int find(int x)
{
  if(prt[x] == 0)
    return x;
  prt[x] = find(prt[x]);
  return prt[x];
}
int main()
{
  //freopen("c.in","r",stdin);
  //freopen("c.out","w",stdout);
  int n,m,ans=0,i,u,v,mark=0;
  scanf("%d%d",&n,&m);
  for(i=1;i<=m;i++)
    {
      scanf("%d %d",&u,&v);
      deg[u]++; deg[v]++;
      if(find(u) != find(v))
	prt[find(u)] = find(v);
    }
  for(i=1;i<=n;i++)
    {
      if(!read[find(i)] && (deg[find(i)] != 0 || find(i) == find(1)))
	{
	  read[find(i)] = ++mark;
	  root[mark] = find(i);
	}
      if(deg[i]%2)
	sing[read[find(i)]]++;
    }
  for(i=1;i<=mark;i++)
    {
      if(sing[i] > 2)
	ans += (sing[i]-2)/2;
    }
  ans += (mark==1) ? 0 : mark;
  if(mark == 1 && sing[read[find(1)]] > 0)
    ans++;
  printf("%d\n",ans);
}
