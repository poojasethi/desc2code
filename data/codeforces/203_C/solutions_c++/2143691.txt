#include <iostream>
#include <algorithm>
#include <map>
using namespace std;
int main(){
    int n,d,a,b,ans=0;
    pair<int, int> foto[100010];
    cin>>n>>d>>a>>b;
    for(int i=0;i<n;i++){
        int x,y;
        cin>>x>>y;
        foto[i].first=x*a+y*b;
        foto[i].second=i+1;
    }
    sort(foto,foto+n);
    for(ans=0;ans<n;ans++){
        d-=foto[ans].first;
        if(d<0) break;
       }
    cout<<ans<<endl;
    for(int i=0;i<ans;i++)
            cout<<foto[i].second<<' ';
}
