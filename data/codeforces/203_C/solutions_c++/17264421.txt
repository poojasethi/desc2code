#include<iostream>
#include<algorithm>
using namespace std;
int a,s,f,g,h,j,k,l,i,n,o;
pair<int,int>M[100002];
main(){
	cin>>n>>k;
	cin>>g>>h;
	for(i=0;i<n;i++){
		cin>>j>>l;
		M[i].first=j*g+h*l;
		M[i].second=i+1;
	}
	sort(M,M+n);
	for(i=0;i<n;i++){
		k=k-M[i].first;
		if(k<0){
			cout<<i<<endl;
			break;
		}
	}
	if(i==n)cout<<i<<endl;
	for(o=0;o<i;o++)
		cout<<M[o].second<<" ";
}
