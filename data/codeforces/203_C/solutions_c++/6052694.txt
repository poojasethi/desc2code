#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int i,x,y,n,d,a,b;
    cin>>n>>d;
    cin>>a>>b;
    vector< pair<int,int> > v;
    for(i=0;i<n;i++)
    {
         cin>>x>>y;
         v.push_back(make_pair(a*x+b*y,i));
    }
    sort(v.begin(),v.end());
    i=0;
    while(d&&(i<n))
    {
        if(d<v[i].first) break;
        d-=v[i].first;
    i++;
    }
    cout<<i<<"\n";
    for(x=0;x<i;x++)
        cout<<v[x].second+1<<" ";
    return 0;
}
