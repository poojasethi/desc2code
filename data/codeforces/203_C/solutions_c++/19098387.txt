#include<cstdio>
#include<algorithm>
using namespace std;
int ans[100000];
struct node
{
	int v,id;
	const operator<(const node x){return v<x.v;}
}p[100000];
int main()
{
	int i,n,a,b,x,y,d;
	scanf("%d%d%d%d",&n,&d,&a,&b);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		p[i].v=a*x+b*y;
		p[i].id=i+1;
	}
	sort(p,p+n);
	for(x=0;x<n&&p[x].v<=d;x++)
	{
		ans[x]=p[x].id;
		d-=p[x].v;
	}
	printf("%d\n",x);
	for(i=0;i<x;i++)
		printf("%d ",p[i].id);
	return 0;
}
