#include <bits/stdc++.h>
#define F first
#define S second

using namespace std;
typedef long long ll;

ll n, d, a, b, ans, p = 1;
pair <ll, ll> meg[100001];

 int main() {
	cin >> n >> d >> a >> b;
	for (ll x, y, i = 1; i <= n; i++) {
		cin >> x >> y;
		meg[i].S = i;
		meg[i].F = a * x + b * y;
	}
	sort (meg + 1, meg + n  + 1);
	for (; p <= n && ans + meg[p].F <= d; p++) {
		ans += meg[p].F;
	}
	cout << p - 1 << endl;
	for (ll i = 1; i < p; i++)
		cout << meg[i].S << " ";
	return 0;
}
