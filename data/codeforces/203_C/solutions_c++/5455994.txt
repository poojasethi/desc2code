#include <bits/stdc++.h>

using namespace std;
#define F first
#define S second
#define ll long long
pair <ll,ll> p[100005];
ll n,d,a,b,x,y,i,kil=0,t;
int main()
{
    cin>>n>>d>>a>>b;
    for(i=1;i<=n;i++)
    {
        cin>>x>>y;
        p[i].F=x*a+y*b;p[i].S=i;
    }
    sort(p+1,p+1+n);
    for(i=1;i<=n;i++)
    {
        if (d==0) break;
        d-=p[i].F;
        if (d>=0) kil++;
    }
    cout<<kil<<endl;
    for(i=1;i<=kil;i++)cout<<p[i].S<<' ';
    return 0;
}
