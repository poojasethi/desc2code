#include <bits/stdc++.h>

using namespace std;

const int N = 100001;
int n, d, a, b;

pair < long, long > p[N];
vector < int > ans;

main() {
	cin >> n >> d >> a >> b;
	for (int i = 0; i < n; i++) {
		int x, y;
		cin >> x >> y;
		p[i] = make_pair(1LL * x * a + 1LL * y * b, i);
	}
	sort(p, p + n);
	for (int i = 0; i < n; i++) {
		if (d >= p[i].first)
			d -= p[i].first, ans.push_back(p[i].second);	
	}
	cout << ans.size() << endl;
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i] + 1 << ' ';
	return 0;
}
