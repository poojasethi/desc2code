#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
	long long i,j,n,d,a,b,x[100005],y[100005],p,q,sum=0;
	pair<long long ,long long >pa[100005];
int main()
{
	cin>>n>>d;
	cin>>a>>b;
	for(i=0;i<n;i++)
	{
		cin>>x[i]>>y[i];
		pa[i].first=x[i]*a+y[i]*b;
		pa[i].second=i;
	}
	sort(pa,pa+n);
	for(i=0;i<n;i++)
	{
		sum+=pa[i].first;
		if(sum>d)
			break;
	}
	cout<<i<<endl;
	for(j=0;j<i;j++)
	{
		cout<<pa[j].second+1<<" ";
	}
	return 0;
}
