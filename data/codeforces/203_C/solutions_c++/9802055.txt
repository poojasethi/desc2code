#include <bits/stdc++.h>
using namespace std;

int main () {
	int n, d;
	cin >> n >> d;
	int a, b;
	cin >> a >> b;
	vector < pair < int, int > > v; 
	int t1; int t2;
	for (int i=0; i<n; i++) {
		scanf("%d%d",&t1,&t2);
		v.push_back(make_pair(a*t1+b*t2,i));
	}
	sort(v.begin(),v.end());
	int cnt = 0;
	vector < int > ans;
	while (d>0 and (cnt<n)) {
		if (d<v[cnt].first)
			break;
		d = d - v[cnt].first;
		cnt++;
	}
	printf("%d\n",cnt);
	for (int j=0; j<cnt; j++) {
		printf("%d ",v[j].second+1);
	}

	return 0;
}
