#include <iostream>
#include <algorithm>
using namespace std;

long long n,m,i,j,k,a,b,x,y,ra,r,ans[100005];
pair <long long,long long> q[100005];
main()
{
cin>>n>>m;
 cin>>a>>b;

for (i=1;i<=n;i++)
{
 cin>>x>>y;
 ra=x*a+y*b;
 
 q[i].first=ra;
 q[i].second=i;
}

sort (q,q+n+1);


while (m>0)
  {
  if (m>=q[r+1].first)      
   {
    r++;
    ans[r]=q[r].second;
    m-=q[r].first;
    if (r>=n) break;
   } else break; 
  }
 cout<<r<<endl;
  for (i=1;i<=r;i++)
   cout<<ans[i]<<" ";
}
