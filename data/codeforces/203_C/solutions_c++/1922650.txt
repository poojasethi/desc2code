#include <iostream>
#include <algorithm>
#define N 100010
using namespace std;
int main()
{
  int n,l,a,b,x,y,tot=-1,sol[N];
  pair < int , int > g[N];
  cin>>n>>l;
  cin>>a>>b;
  for (int i=0; i<n; i++)
    {
      cin>>x>>y;
      g[i].first=x*a+y*b;
      g[i].second=i;
    }
  sort(g,g+n);
  while (l>=0 && tot<n)
    {
      tot++;
      sol[tot]=g[tot].second;
      l-=g[tot].first;
    }
  cout<<tot<<endl;
  for (int i=0; i<tot; i++) cout<<sol[i]+1<<" ";
}
