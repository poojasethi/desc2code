#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
long long int i,k,m,l,n;
cin>>n>>m;
long long int a,b;
cin>>a>>b;
long long int w,q;
vector<pair<long long,long long > >z;
for(i=0;i<n;i++){
	cin>>q>>w;
	z.push_back(make_pair(q*a+w*b,i));
}
sort(z.begin(),z.end());
int s=0;
vector<long long int>ss;
for(i=0;i<n and m>-1;i++){
m-=z[i].first;
if(m>-1)	{s++;ss.push_back(z[i].second);}
}
cout<<s<<'\n';
for(i=0;i<s;i++)	cout<<ss[i]+1<<' ';

return 0;}
