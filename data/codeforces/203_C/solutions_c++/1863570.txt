#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
pair<int, int> p[100002];
int n, d, a, b, x, y;
int main()
{
    cin >> n >> d >> a >> b;
    for(int i = 1; i <= n; ++i)
    {
        cin >> x >> y;
        p[i - 1] = make_pair(x * a + y * b, i);
    }
    sort(p, p + n);
    int res = 0;
    while(res < n && d >= 0) d -= p[res++].first;
    if(d < 0) --res;
    cout << res << endl;
    for(int i = 0; i < res; ++i)
        cout << p[i].second << ' ';
    return 0;
}
