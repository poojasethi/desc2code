#include<bits/stdc++.h>
using namespace std;
int n,d,x,y,i,a,b,v[100003],l=1;
pair <int,int> p[100003];
main()
{
    scanf("%d%d%d%d",&n,&d,&a,&b);
    for (i=1;i<=n;i++)
    {scanf("%d%d",&x,&y);
    p[i]=make_pair(x*a+y*b,i);}
    sort(p+1,p+1+n);
    while (d>=p[l].first&&l<=n)
    {
        d-=p[l].first;
        v[l]=p[l].second;
        l++;
    }
    printf("%d\n",l-1);
    for (i=1;i<l;i++)
        printf("%d ",v[i]);
}
