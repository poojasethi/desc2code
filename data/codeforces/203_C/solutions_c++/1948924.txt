#include <iostream>
#include <algorithm>
#include <stdio.h>
using namespace std;
pair <int,int> f[100005];
int main()
{
long long d;
int a,b,n;
//freopen("i.txt","r",stdin);
cin>>n>>d;
cin>>a>>b;
int x,y;
for (int i=0;i<n;i++)
 {
  cin>>x>>y;
  f[i].first=x*a+y*b;
  f[i].second=i;
  }
sort(f,f+n);
int i=0;
for (;i<n;i++)
 {
  if (d>=f[i].first) d-=f[i].first;
    else break;
 }
cout<<i<<endl;
 for (int i1=0;i1<i;i1++) cout<<f[i1].second+1<<" ";
return 0;
}
