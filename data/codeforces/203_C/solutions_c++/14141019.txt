#include<cstdio>
#include<algorithm>
#define N 100010
using namespace std;
struct cus{int c,i;}c[N];
bool comp(cus a,cus b){return a.c<b.c;}
int main(){
	int n,d,a,b,x,y,i,j; long long sum;
	scanf("%d%d%d%d",&n,&d,&a,&b);
	for(i=0;i<n;i++) scanf("%d%d",&x,&y),c[i].c=x*a+y*b,c[i].i=i+1;
	sort(c,c+n,comp);
	for(i=sum=0;i<n;i++){
		sum=0ll+sum+c[i].c;
		if(sum>d) break;
	}
	printf("%d\n",i);
	for(j=0;j<i;j++) printf("%d%c",c[j].i,j==i-1?'\n':' ');
}
