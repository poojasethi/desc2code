#include <iostream>
#include <vector>
#include <algorithm>
#define For(i,n) for(int i=1;i<=n;i++)

using namespace std;
long long int n,i,d,b,a,x,y;
vector< pair <int,int> > m;

main()
{
cin>>n>>d>>a>>b;
For(i,n){ 
         cin>>x>>y;
         m.push_back(make_pair(x*a+y*b,i));   
        }
sort(m.begin(),m.end());

for(i=0;i<n && m[i].first<=d;i++){ d-=m[i].first;}

cout<<i<<endl;
while(i--)
     {
       cout<<m[i].second<<' ';
     }
}
