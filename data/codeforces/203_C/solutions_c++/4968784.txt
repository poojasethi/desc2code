#include<iostream>
#include<algorithm>
using namespace std;
pair<int,int> x[100001];
int n,d,a,b,i,q,w,j;
long long m;
main()
{
      cin>>n>>d;
      cin>>a>>b;
      for(i=1;i<=n;i++){
       cin>>q>>w;
       x[i].first=q*a+w*b;
       x[i].second=i;
      }
      sort(x+1,x+n+1);
      i=1;
      for(i=1;i<=n;i++){
       m=m+x[i].first;
       if(m>d) break;
      }
      cout<<i-1<<endl;
      for(j=1;j<=i-1;j++)
       cout<<x[j].second<<" ";
      //system("pause");
}
