#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
struct rec{int x,y;}a[100010];
int n,m,p,q,i,j,x,y;
long long now;
bool operator <(rec a,rec b)
{
	return a.x<b.x;
}
int main()
{
	cin>>n>>m>>p>>q;
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&x,&y);
		a[i].x=p*x+q*y,a[i].y=i;
	}
	sort(a+1,a+n+1);
	for(i=1;i<=n;i++)
		if(now+a[i].x<=m) now+=a[i].x; else break;
	cout<<i-1<<endl;
	for(j=1;j<i;j++) printf("%d ",a[j].y);
	cout<<endl;
	return 0;
}
