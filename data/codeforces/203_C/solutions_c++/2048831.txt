#include <vector>
#include <iostream>
#include <algorithm>
#include <iomanip>
#include <cmath>
using namespace std;
int n,i,d,a,b,dd;
vector < pair <int, int> > v;
int main()
{
	cin>>n>>d>>a>>b;
	for(i=0;i<n;i++)
	{
		int x,y;
		cin>>x>>y;
		v.push_back(make_pair(x*a+y*b,i+1));
	}
	sort(v.begin(),v.end());
	dd=d;
	for(i=0;i<n&&dd>=v[i].first;i++)
		dd-=v[i].first;
	cout<<i<<endl;
	for(i--;i>=0;i--)
		cout<<v[i].second<<" ";
	return 0;
}
