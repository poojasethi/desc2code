using namespace std;

int main()
{
   int n,m;
   int a,b;
   cin>>n>>m>>a>>b;
   pair<int,int> p[100002];
   int x,y;
   for(int i=0;i<n;i++)
   {
       cin>>x>>y;
       x=x*a+y*b;
       p[i]=make_pair(x,i+1);
   }
   sort(p,p+n); long long s=0; y=0;
   for(int i=0;i<n;i++)
   {
       if(s+p[i].first<=m) {s+=p[i].first;y=i+1;}
       else {y=i; break;}
   }
   cout<<y<<endl;
   for(int i=0;i<y;i++)
   cout<<p[i].second<<" ";
   cout<<endl;
    return 0;
}
