#include<bits/stdc++.h>

using namespace std;

pair <long long,long long> q[1000000];
long long n,d,al,bh,s,x,y,ct,i;

int main () {
cin>>n>>d>>al>>bh;
long long z[n];
for(long long i= 0 ; i<n ; i++){
	cin>>x>>y;
	z[i]=al*x+bh*y;
	q[i].first=z[i];
	q[i].second=i+1;
}
sort(q,q+n);
i=0;
while(s+q[i].first<=d && i<n){
	s+=q[i].first;
	ct++;
	i++;
}
cout<<ct<<endl;
for(long long j=0 ; j<ct; j++){
	cout<<q[j].second<<" ";
}



	return 0;
}
