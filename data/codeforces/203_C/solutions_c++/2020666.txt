#include<cstdio>
#include<algorithm>
#include<iostream>
using namespace std;
typedef long long LL;
LL n,d,a,b,ans[100005],tot;
pair<LL,LL>s[100005];

int main()
{
	cin>>n>>d>>a>>b;
	LL x,y;
	for (int i=0;i<n;i++)
	{
		cin>>x>>y;
		s[i].first=x*a+b*y;
		s[i].second=i;
	}
	sort(s,s+n);
	for (int i=0;i<n;i++)
		if (d>=s[i].first)
		{
			d-=s[i].first;
			ans[tot++]=s[i].second;
		}
	cout<<tot<<endl;
	for (int i=0;i<tot;i++)
		cout<<ans[i]+1<<' ';
	return 0;
}
