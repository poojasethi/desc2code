#include<cstdio>
#include<algorithm>
using namespace std;
typedef struct _{
	int v;
	int i;
	bool operator <(const _ &x)const{
		return v<x.v;
		} 
	}V;
V v[100005];
int main(){
	int n,d,i,a,b,x,y;
	scanf("%d%d",&n,&d);
	scanf("%d%d",&a,&b);
	for(i=0;i<n;i++){
		scanf("%d%d",&x,&y);
		v[i].v=x*a+y*b;
		v[i].i=i+1;
		}
	sort(v,v+n);
	for(i=0;i<n;i++){
		if(v[i].v>d)break;
		d-=v[i].v;
		}
	printf("%d\n",i);
	while(--i>=0)printf("%d ",v[i].i);
	puts("");
}
