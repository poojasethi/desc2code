#include <iostream>
#include <algorithm>
#include <stdio.h>
using namespace std;
pair <int,int> f[100005];
int main()
{
long long d;
int a,b,n;
scanf("%d %I64d",&n,&d);
scanf("%d %d",&a,&b);
int x,y;
for (int i=0;i<n;i++)
 {
  scanf("%d %d",&x,&y);
  f[i].first=x*a+y*b;
  f[i].second=i;
  }
sort(f,f+n);
int i;
for (i=0;i<n;i++)
 {
  if (d>=f[i].first) d-=f[i].first;
    else break;
 }
printf("%d\n",i);
 for (int j=0;j<i;j++)
    printf("%d ",f[j].second+1);
 printf("\n");
return 0;
}
