#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <math.h>
#include <cstdio>
using namespace std;
	
int main() {
	int n,d,a,b,x,y;
	cin>>n>>d>>a>>b;
	vector < pair<int, int> > cl;
	for(int i=1; i<=n; i++){
		cin>>x>>y;
		cl.push_back(make_pair((a*x+b*y), i));
		}
	sort(cl.begin(), cl.end());
	int count=0;
	for(int i=0; i<n; i++){
		if(d-cl[i].first>=0) {
			count++; 
			d=d-cl[i].first;
			}
		}
	cout<<count<<endl;
	for(int i=0; i<count; i++)
		cout<<cl[i].second<<" ";
	return 0;
}
