#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    pair<long, int> aa[100005];
    long n,d,a,b,i,j,x,y;
    cin>>n>>d>>a>>b;
    for(i=0;i<n;i++){
        cin>>x>>y;
        aa[i].first=x*a+y*b;
        aa[i].second=i+1;
    }
    sort(aa,aa+n);
    for(i=0;i<n;i++){
        if(aa[i].first<=d){
            d-=aa[i].first;
        }
        else
        break;
    }
    cout<<i<<endl;;
    for(j=0;j<i;j++)
    cout<<aa[j].second<<" ";
    return 0;
}
