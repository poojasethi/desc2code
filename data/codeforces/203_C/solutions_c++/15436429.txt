#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
pair < long long , long long > p[100005];
long long n,m,a,b,i,cnt=0,l=0,x,y;
cin>>n>>m;
cin>>a>>b;
for(i=1;i<=n;i++)
{
cin>>x>>y;
p[i].first=x*a+y*b;
p[i].second=i;
}
sort(p+1,p+n+1);
for(i=1;i<=n;i++)
if(l+p[i].first<=m)
{
cnt++;
l+=p[i].first;
}
cout<<cnt<<"\n";
for(i=1;i<=cnt;i++)
cout<<p[i].second<<" ";
}
