#include <iostream>
#include <algorithm>
#define N 100010
using namespace std;
int main()
{
  int n,l,a,b,x,y,k=-1,sol[N];

  pair < int , int > g[N];

  cin>>n>>l;
  cin>>a>>b;

  for (int i=0; i<n; i++)
    {
      cin>>x>>y;
      g[i].first=x*a+y*b;
      g[i].second=i;
    }
  sort(g,g+n);
  while (l>=0 && k<n)
    {
      k++;
      sol[k]=g[k].second;
      l-=g[k].first;
    }
  cout<<k<<endl;
  for (int i=0; i<k; i++) cout<<sol[i]+1<<" ";
}
