#include <iostream>
#include <algorithm>
using namespace std;
long long n, x, y, a, d, b, i, h, ans;
pair<long long, long long>arr[100007];
main() 
{  
  cin>>n>>d>>a>>b;h=d;
  for(i=1;i<=n;i++){
    arr[i].second=i;
    cin>>x>>y;
    x*=a;
    y*=b;
    arr[i].first=x+y;
  }
  sort(arr+1,arr+n+1);
  i=1;
  while(d>=arr[i].first){
  if(arr[i].second==0)break;
  d-=arr[i].first;
  ans++;
  i++;
  }
  cout<<ans<<endl;
  for(i=0;i<ans;i++){if(arr[i+1].second==0)break;cout<<arr[i+1].second<<" ";}
}
