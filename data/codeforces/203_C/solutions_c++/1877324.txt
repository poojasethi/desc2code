#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

pair<int,int> v[100100];
int n,d,a,b,x,y,i,j;

int main()
{
	cin>>n>>d>>a>>b;
	for(i=0;i<n;++i)
	{
		scanf("%d%d",&x,&y);
		v[i].first=x*a+b*y;
		v[i].second=i+1;
	}
	sort(v,v+n);
	for(i=0;i<n;++i)
	{
		d-=v[i].first;
		if(d<0)
		{
			cout<<i<<endl;
			break;
		}
		if(i==n-1) cout<<n<<endl;
	}
	for(j=0;j<i;++j)
		cout<<v[j].second<<' ';
	return 0;
}
