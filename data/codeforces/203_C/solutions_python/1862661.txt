n,d=[int(w) for w in raw_input().split()]
ls,hs=[int(w) for w in raw_input().split()]
c=[]
for i in xrange(n):
    ln,hn=[int(w) for w in raw_input().split()]
    c.append((ln*ls+hn*hs,i))
c.sort()
ans=[]
for i in c:
    d-=i[0]
    if d<0:
        break
    ans.append(i[1]+1)
print len(ans)
print " ".join(str(x) for x in ans)
