n,d=map(int, raw_input().split())
a,b=map(int, raw_input().split())
s=[ [0,0] for i in range(n)]
for i in range(n):
    x,y=map(int,raw_input().split())
    s[i][0]=a*x+b*y
    s[i][1]=i+1

s.sort()

res=0
index=n
for i in range(n):
    res+=s[i][0]
    if res > d :
        index=i
        break;

print index
print ' '.join(map(str, [ s[i][1] for i in range(index)] ) )
