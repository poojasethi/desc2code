#!/usr/bin/env python
# -*- coding: UTF-8 -*-
__author__ = 'firago'
from sys import stdin, stdout
from itertools import izip
n, d = map(int, stdin.readline().split())
a, b = map(int, stdin.readline().split())

_XY = map(int, stdin.read().split())
xy = [(i, x, y, x*a+y*b) for i,(x, y) in enumerate(izip(_XY[::2],_XY[1::2]))]

xy_on_mem = sorted(xy, key = lambda x: x[3])
res = []
sum = 0
for item in xy_on_mem:
    if sum + item[3] > d :
        break
    sum +=item[3]
    res.append(str(item[0]+1))

print len(res)
print ' '.join(res)
