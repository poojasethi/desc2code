n, d = map(int, raw_input().split())
a ,b = map(int, raw_input().split())
l = []
#w = 0
count = 0
res = ""
for i in xrange(1, n+1):
    z, x = map(int, raw_input().split())
    z *= a
    x *= b
    #q = (i, z + x)
    l.append((i, z + x))
#l.sort(key = lambda x: x[1])
l.sort(key = lambda x: x[1])
for c in l:
    if d - c[1] >= 0:
        count += 1
        res += str(c[0]) + " "
        d -= c[1]
    else:
        break
print count
print res.strip()
