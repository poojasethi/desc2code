klanten=[]

n,d=map(int,raw_input().split())
a,b=map(int,raw_input().split())

for i in range(n):
    hulp=map(int,raw_input().split())
    tot=hulp[0]*a+hulp[1]*b
    klanten.append([tot,i])
klanten.sort()
som,i=0,0
while i<n and som<=d:
    som+=klanten[i][0]
    i+=1
    
if som<=d:
    print i
    for i in range(i):
        print klanten[i][1]+1,
else:    
    print i-1
    for i in range(i-1):
        print klanten[i][1]+1,
