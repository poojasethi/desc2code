def stripped_line():
    return raw_input().strip()


def read_ints():
    return map(int, stripped_line().split())

c, m = read_ints()
lqs, hqs = read_ints()
o = []

for i in xrange(c):
    l, h = read_ints()
    o.append((i + 1, (l, h)))

so = sorted(o, key=lambda x: (x[1][0] * lqs) + (x[1][1] * hqs))

um = 0
r = []
for oi in so:
    um += (oi[1][0] * lqs) + (oi[1][1] * hqs)
    if um <= m:
        r.append(oi[0])
    else:
        break

print len(r)
print " ".join([str(x) for x in r])
