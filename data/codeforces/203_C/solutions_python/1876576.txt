#!/usr/bin/env python
n,d = map(int,raw_input().split())
a,b = map(int,raw_input().split())
l=[]
for i in xrange(1,n+1):
	x,y = map(int,raw_input().split())
	l.append((x*a+y*b,i))
l=sorted(l)
res=[]
for x,y in l:
	if d-x < 0:
		break
	else:
		d -= x
		res.append(y)
print len(res)
print ' '.join(map(str, res))
