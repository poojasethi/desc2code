#!/usr/bin/env python

n, d = map(int, raw_input().split())
a, b = map(int, raw_input().split())
T = []
for i in xrange(n):
    x, y = map(int, raw_input().split())
    T.append( (x * a + y * b, i + 1) )
T = sorted(T)
ret = []
i = 0
while d > 0 and i < len(T):
    t = T[i]
    if d >= t[0]:
        d -= t[0]
        ret.append(t[1])
    else:
        break 
    i += 1
print(len(ret))
