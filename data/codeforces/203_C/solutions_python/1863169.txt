s,clients,m="",[],[]
n,d=raw_input().split(' ')
a,b=raw_input().split(' ')
n,d,a,b=int(n),int(d),int(a),int(b)
for i in range (n):
    x,y=raw_input().split(' ')
    clients.append([i+1,int(x)*a+int(y)*b])
clients=sorted(clients,key=lambda o:o[1])
for element in clients:
    d-=element[1]
    if d<0:
        break
    m.append(element[0])
print len(m)
print " ".join(str(x) for x in m)
