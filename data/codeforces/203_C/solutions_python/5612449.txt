n, d = map(int, raw_input().split())
a, b = map(int, raw_input().split())

vals = {}

for i in range(n):
    x, y = map(int, raw_input().split())
    cost = a * x + b * y
    vals[i+1] = cost

k = vals.keys()
k.sort(key=lambda x: vals[x])

sols = []
c = 0
for v in k:
    if c + vals[v] <= d:
        sols.append(v)
        c += vals[v]
    else:
        break

print len(sols)
print ' '.join([str(x) for x in sols])
