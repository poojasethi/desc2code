#!/usr/bin/python
from operator import itemgetter
n,m = map(int,raw_input().split())
a,b = map(int,raw_input().split())
c = [() for i in range(n)]

def cmp(x,y):
	global a
	global b
	return a*x[0]+b*x[1] - a*y[0]-b*y[1]

for i in range(n):
    	x,y = map(int,raw_input().split())
	c[i] = (x,y,i + 1)
c = sorted(c,cmp)
ans = 0
d = []
j = 0
for i in range(n):
	if (m >= c[i][0] * a + c[i][1] * b):
		ans += 1
		m -= c[i][0] * a + c[i][1] * b
		d.append(c[i][2]) 
		j += 1
	else:
		break
print ans
