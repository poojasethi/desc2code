

n, d = [int(w) for w in raw_input().split(' ')]
a, b = [int(w) for w in raw_input().split(' ')]

ns = []

for i in range(n):
    x, y = [int(w) for w in raw_input().split(' ')]

    ns.append( (i+1, x*a + y*b) )

ns.sort(key=lambda a: a[1])

d1 = 0

r = 0
rs = []

for n in ns:
    d1 += n[1]

    if d1 <= d:
        r +=1
        rs.append(n[0])
    else:
        break

print r
print ' '.join([str(a) for a in rs])
