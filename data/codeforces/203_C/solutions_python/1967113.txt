#!/usr/bin/python     
import sys

def main():
  lines = sys.stdin.readlines()
  [num_clients, memory] = map(int, lines.pop(0).split(' '))
  [lsize, hsize] = map(int, lines.pop(0).split(' '))

  costs = []
  for i, l in enumerate(lines):
    request = (map(int, l.split(' ')))
    costs.append([request[0]*lsize + request[1]*hsize, i+1])

  costs.sort()
  currentCost = 0
  jobs = []
  for c in costs:
    currentCost += c[0]
    if currentCost <= memory:
      jobs.append(c[1])
  print len(jobs)
  print ' '.join(str(x) for x in jobs)

  return

if __name__ == "__main__":
  main()
