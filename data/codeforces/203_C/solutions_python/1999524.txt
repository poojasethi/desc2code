n, m = map(int, raw_input().split(" "));
a, b = map(int, raw_input().split(" "))
v = [];
for i in range(0, n):
    x, y = map(int, raw_input().split(" "));
    v.append((a * x + b * y, i + 1));
v.sort();
#print v;
ans = [];
res = 0;
for w, n in v:
    if w <= m:
        ans.append(n);
        res += 1;
        m -= w;
    else:
        break;
print res;
print " ".join(str(x) for x in ans);
