debug = False
n,d = map(int, raw_input('').split(' '))
a,b = map(int, raw_input('').split(' '))
s = [tuple(map(int, raw_input('').split(' ')))+(i,) for i in range(n)]
f = lambda x: a*x[0]+b*x[1]
s.sort(key=f)
clients = []
i = 0
while d >= 0 and i < n:
    d -= (a*s[i][0]+b*s[i][1])
    if d >= 0:
        clients.append(s[i][2]+1)
        i += 1

print(len(clients))
print(' '.join(map(str, clients)))
