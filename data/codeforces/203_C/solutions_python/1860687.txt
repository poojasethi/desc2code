n, d = [int(i) for i in raw_input().split()]
a, b = [int(i) for i in raw_input().split()]
clients = [(a*int(x)+b*int(y), i) for i, (x, y) in enumerate(raw_input().split() for i in xrange(n))]
clients.sort()
ok = []
for cost, no in clients:
    if d >= cost:
        d -= cost
        ok.append(no)
    else:
        break
print len(ok)
for c in ok:
    print c+1,
