n, d = [int(j) for j in raw_input().split()]
a, b = [int(j) for j in raw_input().split()]

spaceReq = []
for j in range(n):
    x, y = [int(k) for k in raw_input().split()]
##    print [repr(j) for k in [a, b, x, y, k]]
    spaceReq.append( (x*a+y*b, j+1) )

spaceReq = sorted(spaceReq)
sol = []
##memCount = 0
for byt, cNum in spaceReq:
    d -= byt
    if not d<0:
        sol.append(str(cNum))
    else:
        break
print len(sol)
print ' '.join(sol)
