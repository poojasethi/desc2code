import sys

n = int(sys.stdin.readline())
s = sys.stdin.readline()
mn = 0
a = 1
sol = 0
A = [1]
for c in s[:n - 1]:
  if c == 'L':
    if a != 1:
      a = 1
    else:
      a = 0
  elif c == 'R':
    a += 1
  A.append(a)
  if a == 0:
    a += 1
    for j in xrange(len(A) - 1, -1 ,-1):
      A[j] += 1
      if A[j - 1] < A[j] and s[j - 1] == 'R':
        break
      if A[j - 1] > A[j] and s[j - 1] == 'L':
        break
for i in xrange(n):
  A[i] = '%s' % (A[i] + mn)
print ' '.join(A)
