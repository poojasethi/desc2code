n=input()
s=raw_input()

ans=list()
for i in xrange(0,n,1):
    l=r=1
    for j in xrange(i-1,-1,-1):
        if s[j]=='L' : break
        if s[j]!='=' : r+=1
    for j in xrange(i,n-1,1):
        if s[j]=='R' : break
        if s[j]!='=' : l+=1
    ans.append(max(l,r))
print ' '.join(map(str,ans))
