n = int(raw_input())
s = raw_input()

res = [0] * n
cur = 0
for i in xrange(n):
	l = 1
	r = 1
	for j in xrange(i, n - 1):
		if s[j] == '=': continue
		if s[j] == 'R': break
		l += 1
	for j in xrange(i - 1, -1, -1):
		if s[j] == '=': continue
		if s[j] == 'L': break
		r += 1
	res[i] = max(l, r)

print " ".join([str(x) for x in res])
