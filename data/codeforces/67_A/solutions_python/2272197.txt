n = int(raw_input())
rate = [0] * n
s = raw_input()
rate[0] = 1
for i in xrange(n - 1):
    if s[i] == 'L':
        rate[i + 1] = 1
        for j in xrange(i, -1, -1):
            if s[j] == 'L' and rate[j] == rate[j + 1] or s[j] == '=':
                rate[j] += 1
            else:
                break
    elif s[i] == 'R':
        rate[i + 1] = rate[i] + 1
    else:
        rate[i + 1] = rate[i]
print(' '.join(map(str, rate)))
