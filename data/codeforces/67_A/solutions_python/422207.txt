#!/usr/bin/env python
# -*- coding: utf-8 -*-

def main():
    l = [1 for i in xrange(0, int(raw_input()))]
    d = [i for i in raw_input()]
    f = True
    while (f):
        f = False
        for i in xrange(0, len(d)):
            if d[i] == '=' and l[i] != l[i+1]:
                l[i] = l[i+1] = max(l[i], l[i+1])
                f = True
            elif d[i] == 'L' and l[i] <= l[i+1]:
                l[i] = l[i+1] + 1
                f = True
            elif d[i] == 'R' and l[i] >= l[i+1]:
                l[i+1] = l[i] + 1
                f = True
    for n in l:
        print n,
    print ""
    return 0

if __name__ == '__main__':
    main()
