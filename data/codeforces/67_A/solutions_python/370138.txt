def r(a,b): return 0 if a>b else (1 if a==b else 2)
n=input()
l=[1]*n
s=raw_input()
for i in range(1,n):
    if s[i-1]=='=':
        l[i]=l[i-1]
    elif s[i-1]=='R':
        l[i]=l[i-1]+1
    else:
        l[i]=0
        j=i
        while True:
            l[j]+=1
            if j==0 or r(l[j],l[j-1])==r(l[j]-1,l[j-1]):
                break
            j-=1
print ' '.join(map(str,l))
