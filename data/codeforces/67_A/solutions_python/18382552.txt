ans, n, a = [1], input(), raw_input()
for i in xrange(0, n-1):
    if a[i] == 'R':
        ans.append(ans[-1] + 1)
    elif a[i] == '=':
        ans.append(ans[-1])
    else:
        ans.append(1)

for i in xrange(n-2, -1, -1):
    if a[i] == 'L':
        ans[i] = max(ans[i], ans[i+1] + 1)
    elif a[i] == '=':
        ans[i] = ans[i+1] = max(ans[i], ans[i+1])
print " ".join(map(str, ans))
