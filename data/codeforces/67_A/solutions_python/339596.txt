import sys

sys.stdin.next()
s = sys.stdin.next().strip()

cnts = [1]
for c in s:
  if c == '=':
    cnts[-1] += 1
  else:
    cnts.append(1)

s = filter(lambda c: c != '=', s)

res = [1]
for c in s:
  if c == 'L':
    res.append(1)
  else:
    res.append(res[-1]+1)

for i in xrange(len(res)-2, -1, -1):
  if s[i] == 'L':
    res[i] = max(res[i], res[i+1]+1)

res2 = []
for i, c in zip(res, cnts):
  res2.extend([i]*c)

print " ".join(map(str, res2))
