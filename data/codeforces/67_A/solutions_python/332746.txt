n = input()
s = raw_input()
a = [1]*n
for j in xrange(n):
    for i in xrange(n-1):
        if s[i]=='L':
            if a[i]<=a[i+1]:                
                a[i]=a[i+1]+1
        elif s[i]=='R':
            if a[i]>=a[i+1]:                
                a[i+1]=a[i]+1
        elif s[i]=='=' and a[i]!=a[i+1]:
            a[i+1]=a[i]=max(a[i],a[i+1])
print ' '.join(map(str,a))
