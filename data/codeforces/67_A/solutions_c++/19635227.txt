#include<bits/stdc++.h>
using namespace std;
string s;

int main()
{
	int n;
        cin>>n>>s;
	for(int i=0;i<n;++i)
	{
		int l,r,ll,rr;
		l=i-1,r=i,ll=rr=1;
		while(s[l]!='L' && l>=0)ll+=(s[l]=='R'),--l;
		while(s[r]!='R' && r<n) rr+=(s[r]=='L'),++r;
                cout<<max(ll,rr)<<' ';
	}
	return 0;
}
