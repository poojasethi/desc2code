#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{  //freopen("i.txt","r",stdin);
int n; cin>>n;
string s; cin>>s;
int a[1005],m=1;
a[0]=1;
for (int i=0;i<s.size();i++)
 {
  if (s[i]=='=') a[i+1]=a[i];
  if (s[i]=='L') {a[i+1]=1;}
  if (s[i]=='R') a[i+1]=a[i]+1;
 }

for (int i=s.size()-1;i>=0;i--)
 {
  if (s[i]=='=') a[i]=a[i+1];
  if (s[i]=='L') {a[i]=max(a[i],a[i+1]+1);}
 }

for (int i=0;i<n;i++) cout<<a[i]<<' ';
return 0;
}
