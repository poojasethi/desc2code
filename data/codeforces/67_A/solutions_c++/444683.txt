#include <iostream>
#include <string>

using namespace std;

int main()
{
	int irr[1010];
	string row;
	int n;
	
	cin>>n;
	cin>>row;
		
	for(int i=0;i<1010;i++)
		irr[i]=1;
		
	for(int i=0;i<n-1;i++)
	{
		if (row[i]=='=') irr[i+1]=irr[i];
		if (row[i]=='R' && irr[i+1]<=irr[i]) irr[i+1]=irr[i]+1;
	}
	
	for(int i=n;i>0;i--)
	{
		if (row[i-1]=='=') irr[i-1]=irr[i];
		if (row[i-1]=='L' && irr[i-1]<=irr[i]) irr[i-1]=irr[i]+1;
	}	
	
	for(int i=0;i<n;i++)
	cout<<irr[i]<<" ";		
}
