#include <iostream>
#include <cstdio>
#include <string>
using namespace std;


string s;
int n, a[1000];


int main()
{
	cin >> n >> s;
	for(int i = 0; i < n; i++)	a[i] = 1;
	for(int i = 0; i < n - 1; i++)
	{
		if(s[i] == 'R')
			a[i + 1] = a[i] + 1;
		if(s[i] == '=')
			a[i + 1] = a[i];
	}
	for(int i = n - 1 - 1; i >= 0; i--)
	{
		if(s[i] == 'L'  &&  a[i] <= a[i + 1])
			a[i] = a[i + 1] + 1;
		if(s[i] == '=')
			a[i] = a[i + 1];
	}
	for(int i = 0; i < n; i++)
		cout << a[i] << ' ';
}
