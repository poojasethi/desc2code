#include <iostream>
#include <algorithm>

using namespace std;

const int MAX_N = 1000 + 10;

int a[MAX_N];

int main()
{
	int n;
	string s;
	cin >> n >> s;
	for(int i = 0; i < n; i++)
	{
		int l = i, r = i - 1, tedadl = 1, tedadr = 1;
		while(l < n - 1 && s[l] != 'R')
		{
			if(s[l] == 'L')
				tedadl = tedadl + 1;
			l = l + 1;
		}
		while(r >= 0 && s[r] != 'L')
		{
			if(s[r] == 'R')
				tedadr = tedadr + 1;
			r = r - 1;
		}
		cout << max(tedadr , tedadl) << " ";
	}
	cout << endl;
	return 0;
}
