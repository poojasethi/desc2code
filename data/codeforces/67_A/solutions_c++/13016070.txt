#include<bits/stdc++.h>
using namespace std;
int main()
{
   
    char a[1010];
    int n,i,j,r,lt,rt;
    scanf("%d%s",&n,a+1);
    for(i=1;i<=n;i++)
    {
        int j=i-1;
        int r=i;
        int lt=1;
        int rt=1;
        while(a[j]!='L'&&j>0)
        {
            lt+=(a[j]=='R'?1:0);
            j--;
        }
        while(a[r]!='R'&&r<=n)
        {
            rt+=(a[r]=='L'?1:0);
            r++;
        }
        cout<<max(rt,lt)<<" ";
    }
    return 0;
        
    }
