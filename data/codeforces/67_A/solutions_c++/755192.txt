#include <stdio.h>

int a[1005];
char s[1005];

int main(void) {
	int n;
	scanf("%d", &n);
	scanf("%s", s);
	a[n-1] = 1;
	for(int i=n-2;i>=0;--i) {
		if(s[i]=='L') a[i] = a[i+1] + 1;
		else if(s[i]=='R') a[i] = 1;
		else if(s[i]=='=') a[i] = a[i+1];

		if(s[i]=='R' && a[i] >= a[i+1]) {
			for(int j=i;j<n-1;j++) {
				if(s[j]=='L' || s[j]=='R' && a[j] < a[j+1]) break;
				++a[j+1];
			}
		}
	}
	for(int i=0;i<n;i++)
		printf("%d ", a[i]);
	puts("");
	return 0;
}
