#include <iostream>
#include <cstdio>
#include <string>
using namespace std;
int n,i,r,l,rr,ll;
string s;
int main()
{
cin>>n>>s;
for (i=0;i<n;i++){
r=i,rr=0,ll=0,l=i-1;
while (r<n-1 && (s[r]=='L' || s[r]=='=' ))
	{
	if (s[r]=='L') rr++;
	r++;
	}
while (l>=0 && (s[l]=='R' || s[l]=='='))
	{
	if (s[l]=='R') ll++;
	l--;
	}
	cout<<max(ll,rr)+1<<" ";
}



return 0;
}
