#include <iostream>
using namespace std;

const int N = 1005;
char a[N];

int main()
{
    int n, i;
    while( cin>>n )
    {
        cin>>a;
        int l, r, ll, rr;
        for( i = 0; i < n; i++)
        {
            l = i - 1, r = i, ll = rr = 1;
            while(a[l] != 'L' && l >= 0 ) ll += (a[l] == 'R'), -- l;
            while(a[r] != 'R' && r < n ) rr += (a[r] == 'L' ), ++ r;
            cout<< max(ll, rr)<<endl;
        }
        
    }
    return 0;
}
