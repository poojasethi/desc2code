#include <iostream>
#include <string>
#include <cstring>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n,m,c,l,r,cp,cm;
string s;

int main(){
  cin>>n>>s;
  rep(i,n){
    l=i-1;r=i;cp=1;cm=1;
    while(l>=0 && s[l]!='L'){
      cp+=(s[l]=='R');
      l--;
    }
    while(r<n-1 && s[r]!='R'){
      cm+=(s[r]=='L');
      r++;
    }
    cout<<max(cp,cm)<<endl;
  }
  
}
