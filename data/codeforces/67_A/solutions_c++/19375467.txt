#include<cstdio>
char s[1001];
int main()
{
	int n;
	scanf("%d%s",&n,s);
	for(int i = 0 ; i < n ; ++i)
	{
		if(i) putchar(' ');
		int n1 = 0,n2 = 0;
		int j;
		for(j = i - 1 ; j >= 0 ; j--)
            if(s[j] == 'R') n1++;
            else if(s[j] == 'L') break;
		for(j = i ; j < n - 1 ; j++)
            if(s[j] == 'L') n2++;
            else if(s[j] == 'R') break;
        printf("%d",1 + (n1 > n2 ? n1 : n2));
	}
	return 0;
}
