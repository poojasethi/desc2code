#include <cstdio>
#include <algorithm>
char str[10011];
int main(){
    int n,l,r,t,t2;
    scanf("%d",&n);
    scanf(" %s",str);
    for(int i=0;i<n;i++){
        t=t2=1;
        l=i,r=i-1;
     //   printf("r%d ",r);
        for(;str[l]!='R'&&l<=n-2;l++)
            if(str[l]=='L')
               t++;
        for(;str[r]!='L'&&r>=0;r--)
            if(str[r]=='R')
               t2++;
        printf("%d ",std::max(t2,t));
    }
   return 0;
}
