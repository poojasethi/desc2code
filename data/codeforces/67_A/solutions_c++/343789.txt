#include<cstdio>
char s[1001];
//        Last Change:  2011-03-19 17:31:56
int main()
{
	int n;
	scanf("%d%s",&n,s);
	for(int i=0;i<n;++i)
	{
		int l,r,ll,rr;
		l=i-1,r=i,ll=rr=1;
		while(s[l]!='L'&&l>=0)ll+=(s[l]=='R'),--l;
		while(s[r]!='R'&&r<n)rr+=(s[r]=='L'),++r;
		printf("%d\n",ll>rr?ll:rr);
	}
	return 0;
}
