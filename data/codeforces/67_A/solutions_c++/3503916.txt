#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
const int mm=1100;
const int oo=1e9;
int ll,rr,n;
char s[mm];
int main()
{
  while(cin>>n)
  { cin>>s;
    for(int i=0;i<n;++i)
    {
      ll=1;rr=1;
      int l=i-1,r=i;
      while(l>=0&&s[l]!='L')ll+=s[l]=='R',--l;
      while(r<n&&s[r]!='R')rr+=s[r]=='L',++r;
      cout<<(ll<rr?rr:ll)<<"\n";
    }

  }
}
