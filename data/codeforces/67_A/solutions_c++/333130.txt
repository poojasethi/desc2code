#include<iostream>
#include<string>
using namespace std;
int main()
{
	int n,r,rr,l,ll;
	string s;
	cin>>n>>s;
	for(int i = 0; i < n; i++)
	{
		r = i;
		rr = 0;
		l = i - 1;
		ll = 0;
		while(r < n - 1 && s[r] == 'L' || s[r] == '=') 
		{
			if (s[r] == 'L') rr++;
			r++;
		}
		while(l >= 0 && s[l] == 'R' || s[l] == '=')
		{
			if (s[l] == 'R') ll++;
			l--;
		}
		cout<<max(ll,rr) + 1<<" ";
	}
	return 0;
}
