#include <cmath>
#include <iostream>

using namespace std;

const int MAX_N = 1000 + 10;
int a[MAX_N];

int main()
{
	int n;
	string s;
	cin >> n >> s;
	for (int i = 0; i < n; i++)
	{
		int left = i - 1, right = i, ans1 = 1, ans2 = 1;
		while (left >= 0 && s[left] != 'L')
		{
			if (s[left] == 'R')
				ans1++;
			left--;
		}
		while (right < n - 1 && s[right] != 'R')
		{
			if (s[right] == 'L')
				ans2++;
			right++;
		}
		cout << max(ans1, ans2) << " ";
	}
	cout << endl;
	return 0;
}
