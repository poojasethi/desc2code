#include<iostream>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<cstdio>
using namespace std;

int n;
char s[1004];
//int num[1004]={0};

int main()
{
	int i;
	cin>>n;
	cin>>(s+1);
	int l,r,l_sum,r_sum;
	for(i=1;i<=n;i++)
	{
		l=i-1,r=i;
		l_sum=1,r_sum=1;
		while(l>=1)
		{
			l_sum+=(s[l]=='R');
			if(s[l]=='L')break;
			l--;
		}
		while(r<=n-1)
		{
			r_sum+=(s[r]=='L');
			if(s[r]=='R')break;
			r++;
		}
		cout<<max(l_sum,r_sum)<<' ';
	}
	cout<<endl;	
}
