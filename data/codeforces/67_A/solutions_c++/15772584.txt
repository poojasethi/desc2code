#include<bits/stdc++.h>

using namespace std;

int S[1003],n,i,asd;

string s;

int main(){

    cin >> n >> s;

    s = 'R' + s;

    S[n-1] = 1;

    for(i=n-2; i>=0; i--)
        S[i] = 1 + (s[i+1] != 'R' ? S[i+1] - (s[i+1] == '=') : 0);

    for(i=0;i<n;i++){

        if(s[i] == 'R')
            asd = max(asd + 1 , S[i]);

        else if(s[i] == 'L')
            asd = min(asd - 1 , S[i]);

        cout << asd << " ";
    }

    return 0;
}
