#include <bits/stdc++.h>

using namespace std; 
const int MX = 100000 ;
string s ; 
int n ;
int main(){
	cin >> n >> s; 
	for ( int i  = 0 ; i < n ; i++ ) { 
		int l = i-1 ;
		int r = i ; 
	
		int cr = 1 , cl = 1 ;
		while ( s[l] != 'L'&& l >= 0 ) { 
			if ( s[l] == 'R' ){ cl++ ; }
			l--;
		}
		while ( s[r] != 'R' && r < n-1 ) {
			if ( s[r] == 'L' ){ cr++ ;}
			r++ ;
		}
		
		cout << max ( cr , cl ) << " " ;
	}
	
		
	return 0 ; 
}
