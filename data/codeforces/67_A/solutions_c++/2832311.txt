#include<cstdio>
char s[1001];
//只要两边可以对接
int main(void)
{
    int n,i,l,r,ll,rr;
    scanf("%d%s",&n,&s[1]);
    s[0]='X';
    for(i=0;i<n;i++)
    {
        l=i;r=i+1;ll=rr=1;
        while(l>0&&s[l]!='L')
        {
            ll+=(s[l]=='R');
            l--;
        }
        while(r<n&&s[r]!='R')
        {
            rr+=(s[r]=='L');
            r++;
        }
        printf("%d ",rr>ll?rr:ll);
    }
    printf("\n");
    return 0;
}
