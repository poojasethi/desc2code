#include <iostream>
#include <cstdio>
using namespace std;
int ans[5008];
int main()
{
	int n,j=-1;char ch[1005];
	scanf("%d",&n);
	for(int i=0;i<n+5;i++) ans[i]=1;
	scanf("%s",ch);
	for(int k=0;k<1001;k++){
	j=-1;
	for(int i=0;i<n-1;i++) {
		j++;
		if(ch[i]=='=')
			ans[j]=ans[j+1]=max(ans[j],ans[j+1]);
		if(ch[i]=='L')
			if(ans[j+1]>=ans[j])
				ans[j]=ans[j+1]+1;
		if(ch[i]=='R'){
			if(ans[j]>=ans[j+1])
				ans[j+1]=ans[j]+1;
		}
	}
	}
	for(int i=0;i<n;i++) printf("%d ",ans[i]);
	printf("\n");
	return 0;
}
