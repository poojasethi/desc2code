#include<stdio.h>
#include<algorithm>
#include<iostream>
using namespace std;
int a[1001];
char s[1001];
int main(void)
{
    int n;
        scanf("%d%s",&n,s);
        for(int i = 0;i < n;i++){
            int l,r,ll,rr;
            l = i-1,r = i,ll = rr = 1;
            while(s[l]!='L'&&l>=0)ll+=(s[l]=='R'),--l;
            while(s[r]!='R'&&r<n)rr+=(s[r]=='L'),++r;
            printf("%d ",ll>rr?ll:rr);
        }
        puts("");
        return 0;
}
