#include <cstdio>
#include <cstring>

char s[1000];

int main() {
    int n;
    scanf("%d %s",&n,s);
    for (int i=0; i<n; i++) {
        int x=1,y=1;
        for (int j=i-1; j>=0&&s[j]!='L'; j--)
            if (s[j]=='R')
                x++;
        for (int j=i; j<(n-1)&&s[j]!='R'; j++)
            if (s[j]=='L')
                y++;
        if (i>0)
            printf(" ");
        printf("%d",x>y?x:y);
    }
    return 0;
}
