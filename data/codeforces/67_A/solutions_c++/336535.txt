#include <iostream>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    int n;
    cin>>n;
    string s;
    cin>>s;
    for (int i=0; i<n; i++) {
        int x=1,y=1;
        for (int j=i-1; j>=0&&s[j]!='L'; j--)
            if (s[j]=='R')
                x++;
        for (int j=i; j<(n-1)&&s[j]!='R'; j++)
            if (s[j]=='L')
                y++;
        if (i>0)
            cout<<" ";
        cout<<max(x,y);
    }
    return 0;
}
