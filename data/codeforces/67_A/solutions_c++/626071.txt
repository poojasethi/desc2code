#include<iostream>
using namespace std;

int n;
string inp;


int main()
{
	cin >> n >> inp;
	for(int i=0;i<n;i++)
	{
		int a=1,b=1,l=i-1,r=i;
		while(l>=0 && inp[l]!='L')
		{
			a+=(inp[l]!='=');
			--l;
		}
		while(r<n-1 && inp[r]!='R')
		{
			b+= (inp[r]!='=');
			r++;
		}
		cout << max(a,b) << " ";
	}
	cout << endl;
	return 0;
}
