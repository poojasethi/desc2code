#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n;
char s[1010];
int num[1010];
int main()
{
	scanf("%d%s",&n,s+1);
	s[0]='L';
	s[n+1]='R';
	for(int i=1;i<=n;i++)
	{
		int j=i-1;
		while(s[j]!='L' && j>=0)
		{
			num[i]+= (s[j]=='R');
			j--;
		}
		j=i;
		int cnt=0;
		while(s[j]!='R' && j<=n+1)
		{
			cnt+= (s[j]=='L');
			j++;
		}
		if(cnt>num[i]) num[i]=cnt;
		num[i]++;
	}
	for(int i=1;i<=n;i++)
	{
		printf("%d ",num[i]);
	}
	puts("");
	return 0;
}
