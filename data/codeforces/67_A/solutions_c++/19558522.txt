#include<iostream>
#include<queue>
using namespace std;
long b[100003];long a[100004];
char c[10003];
int main(){
long n;
cin>>n;
string s;
cin>>s;
a[n-1]=0;
for(long i=n-2;i>=0;i--){
    if(s[i]=='R')a[i]=0;
    else {
    if(s[i]=='=')a[i]=a[i+1];
    else a[i]=1+a[i+1];
    }
}
b[0]=0;
for(long j=1;j<n;j++){
    if(s[j-1]=='R')b[j]=1+b[j-1];
    else  {
         if(s[j-1]=='=')b[j]=b[j-1];
    else b[j]=0;
    }
}
for(long i=0;i<n;i++){
    long h=max(b[i],a[i]);
    cout<<h+1<<" ";
}
cout<<endl;}
