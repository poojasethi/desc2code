#include <iostream>
#include <cstdio>
using namespace std;
static int use[100005],ans[100005];
int main()
{
    int n;
    cin>>n;
    if(n&1)
    {
        printf("-1\n");
        return 0;
    }
    ans[n]=0;
    use[0]=1;
    for(int i=n;i>0;i--)
    {
        if(use[(ans[i]+n)/2])
            ans[i-1]=ans[i]/2;
        else
            ans[i-1]=(ans[i]+n)/2;
        use[ans[i-1]]=1;
    }
    for(int i=0;i<n;i++)
        cout<<ans[i]<<" ";
    printf("0\n");
    return 0;
}
