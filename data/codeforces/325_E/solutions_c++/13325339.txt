# include <cstdio>
# include <cstring>
# include <cassert>

# define NR 100100

int p[NR], n;

void dfs (int x)
{
	if (p[x] != -1) return ;
	int y = (x + n / 2) % n;
	if (p[y] != -1) dfs (p[x] = (2 * x + (p[y] == 2 * x % n)) % n);
	else 
	{
		dfs (p[x] = 2 * x % n);
		if (p[y] == -1) p[y] = p[x], dfs (p[x] = (2 * x + 1) % n);
	}
}

int main ()
{
	//freopen ("325E.in", "r", stdin);
	scanf ("%d", &n);
	if (n % 2) return puts ("-1"), 0;
	memset (p, -1, sizeof (p));
	dfs (0), printf ("0 ");
	for (int i = p[0]; i; i = p[i]) printf ("%d ", i);
	printf ("0\n");
	//while (1) ;
	return 0;
}
