#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<deque>
#include<vector>
using namespace std;
const int SIZEN=100010;
vector<int> ans;
int N;
bool vis[SIZEN]={0};
void DFS(int x){
	vis[x]=true;
	if(!vis[(x*2)%N]) DFS((x*2)%N);
	if(!vis[(x*2+1)%N]) DFS((x*2+1)%N);
	ans.push_back(x);
}
int main(){
	scanf("%d",&N);
	if(N&1){
		printf("-1\n");
	}
	else{
		DFS(0);
		reverse(ans.begin(),ans.end());
		ans.push_back(0);
		for(int i=0;i<ans.size();i++) printf("%d ",ans[i]);
		printf("\n");
	}
	return 0;
}
