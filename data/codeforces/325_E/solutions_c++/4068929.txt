#include<cstdio>

int n;
int u[100010];

void DFS(int x)
{
  u[x]=1;
  if (x==1)
    printf("0 1");
  else
  {
    if (u[(n+x)/2])
      DFS(x/2);
    else
      DFS((n+x)/2);
    printf(" %d",x);
  }
}

int main()
{
  scanf ("%d", &n);
  if (n%2)
  {
    puts("-1");
    return 0;
  }
  DFS(0);
  return 0;
}
