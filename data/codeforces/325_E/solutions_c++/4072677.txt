#include <iostream>
using namespace std;
int n;
bool vis[100000];
void dfs(int x){
    vis[x]=1;
    if(x==1){cout<<"0 1";return;}
    if(!vis[x+n>>1])
        dfs(x+n>>1);
    else
        dfs(x>>1);
    cout<<' '<<x;
}
int main()
{
    cin>>n;
    if(n&1)cout<<-1<<endl;
    else dfs(0);
    return 0;
}
