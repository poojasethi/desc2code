#include <cstdio>
inline unsigned int getuint(){
    char w=getchar();
    while (w<'0'||'9'<w) w=getchar();
    unsigned int ans=0;
    for (;'0'<=w&&w<='9';w=getchar()) ans=ans*10+w-'0';
    return ans;
}
int n;
int ans[100005];
void dfs(int i){
    static int vis[100005];
    if (vis[i%=n]) return;
    vis[i]=1, dfs(i<<1), dfs(i<<1|1);
    ans[++*ans]=i;
}
int main(){
    //freopen("ex.in", "r", stdin);
    n=getuint();
    if (n&1) return puts("-1"), 0;
    dfs(0);
    for (int i=*ans;i;i--) printf("%d ", ans[i]);puts("0");
    return 0;
}
