#include <bits/stdc++.h>
using namespace std;

const int N=100000+5;

int n,cnt,ans[N];
bool vis[N];

inline void dfs(int now)
{
	vis[now]=1;
	if (!vis[now*2%n]) dfs(now*2%n);
	if (!vis[(now*2+1)%n]) dfs((now*2+1)%n);
	ans[++cnt]=now;
}

int main()
{
	scanf("%d",&n);
	if (n&1){puts("-1");return 0;}
	dfs(0);
	for (int i=n;i;--i) printf("%d ",ans[i]);
	puts("0");
	return 0;
}
