#include <stdio.h>

#define if if (

#define do )
#define for for (
#define while while (
#define begin {
#define end }

#define MAXN 100005

int n;
bool ok;
bool v[MAXN];
int stack[MAXN];

inline void dfs(int x,int dep)
begin
	if dep==n+1 && x==0 then ok=true;
	if v[x] then return;
	v[x]=true;
	stack[dep]=x;
	dfs(x/2,dep+1);
	if ok then return;
	dfs(x/2+n/2,dep+1);
	v[x]=false;
end;

int main()
begin
	scanf("%d",&n);
	if n&1 then begin
		puts("-1");
	end else begin
		dfs(0,1);
		printf("0");
		int i;
		for i=n;i>=1;i-- do begin
			printf(" %d",stack[i]);
		end;
	end;
end
