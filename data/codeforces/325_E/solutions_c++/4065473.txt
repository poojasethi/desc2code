#include <iostream>
#include <cstring>
#include <cstdio>
#include <vector>
using namespace std;
int n, sum;
bool b[50010][2];
vector<int> v;

void dfs(int k)
{
	sum++;
	if (b[k][1])
	{
		b[k][1] = false;
		dfs((k * 2 + 1) % n);
		v.push_back(1);
	}
	if (b[k][0])
	{
		b[k][0] = false;
		dfs(k * 2 % n);
		v.push_back(0);
	}
}

int main()
{
	sum = 0;
	scanf("%d", &n);
	if (n % 2)
		puts("-1");
	else
	{
		n /= 2;
		memset(b, true, sizeof(b));
		v.clear();
		dfs(0);
		printf("0");
		int k = 0;
		for (int i = v.size() - 1; i >= 0; i--)
		{
			k = (k * 2 + v[i]) % (n * 2);
			printf(" %d", k);
		}
		puts("");
	}
	return 0;
}
