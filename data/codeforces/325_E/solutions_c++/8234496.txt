#include<cstdio>
using namespace std;int C[1000005],ans[1000005],n,top;
void dfs(int x)
{
  C[x]=1;
  if (!C[x*2%n]) dfs((x*2)%n);
  if (!C[(x*2+1)%n]) dfs((x*2+1)%n);
  ans[++top]=x;
}
int main()
{
  scanf("%d",&n);if (n&1) return puts("-1"),0;
  dfs(0);
  while (top--) printf("%d ",ans[top+1]);puts("0");
}
