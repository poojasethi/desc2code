#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
bool vis[100010];
int n,ans[100010];
void dfs(int x)
{
	vis[x]=1;
	if(!vis[2*x%n])dfs(2*x%n);
	if(!vis[(2*x+1)%n])dfs((2*x+1)%n);
	ans[++ans[0]]=x;
}
int main()
{
	scanf("%d",&n);
	if(n&1){puts("-1");return 0;}
	dfs(0);
	for(int i=ans[0];i>=1;--i)printf("%d ",ans[i]);
	puts("0");
}
