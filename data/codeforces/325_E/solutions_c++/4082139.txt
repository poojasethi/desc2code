#include <iostream>
#include <cstdlib>
#include <cstdio>
#define MAXN 100005

using namespace std;

int n, mark[MAXN];

int dfs(int i)
{
  mark[i] = 1;
  if (i)
  {
    int x, y;
    x = i / 2;
    y = (i + n) / 2;
    if (mark[y])
      dfs(x);
    else
      dfs(y);
  }
  cout << i << " ";
}

main()
{
  cin >> n; 
  if (n & 1) 
    cout << -1;
  else
  {
    dfs(n / 2);
    cout << 0;
  }
  
  //system("pause");
}
