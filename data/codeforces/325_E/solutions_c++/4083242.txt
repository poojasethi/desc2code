#include <iostream>
#include <cstdio>
using namespace std;
#define MAXN 111111
int N;
bool Vis[MAXN];
void DFS(int m)
{
    Vis[m]=true;
    if(m==1)
    {
        printf("0 1");
        return;
    }
    if(!Vis[(m+N)>>1])
        DFS((m+N)>>1);
    else
        DFS(m>>1);
    printf(" %d",m);
}
int main()
{
    scanf("%d",&N);
    if(N&1)
        puts("-1");
    else
    {
        DFS(0);
        puts("");
    }
    return 0;
}
