#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
#define N 100050
bool vis[N];int n,ans[N],st;
void DFS(int x)
 {
 	vis[x]=1;
 	if (!vis[2*x%n]) DFS(2*x%n);
 	if (!vis[(2*x+1)%n]) DFS((2*x+1)%n);
 	ans[++st]=x;
 }
int main()
 {
 	cin >>n;
 	if (n&1) puts("-1");else
 	 {
 	 	DFS(0);
 	 	for (int i=st;i>=1;i--) printf("%d ",ans[i]);
 	 	puts("0");
 	 }
 	return 0;
 }
