#include<cstdio>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
using namespace std;
bool used[100009];int n;
void dfs(int u)
{
    if(!used[(u+n)/2])
    {
        used[(u+n)/2]=1;
        dfs((u+n)/2);
        printf(" %d",(u+n)/2);
    }
    else if(!used[u/2])
    {
        used[u/2]=1;
        dfs(u/2);
        printf(" %d",u/2);
    }
    return ;
}

int main()
{
    cin>>n;
    memset(used,false,sizeof(used));
    used[1]=true;used[0]=true;
    if(n&1)puts("-1");
    else
    {
        printf("0 1");
        dfs(1);
        puts(" 0");
    }
    return 0;
}
