#include<bits/stdc++.h>
using namespace std;
const int N=100010;
int n,a[N]={};
bool v[N]={};
void dfs(int s)
{
	if(v[s])
		return;
	v[s]=true;
	int t[2]={s*2%n,(s*2+1)%n};
	if(a[s]!=-1)
		dfs(a[s]);
	else
	{
		int side=v[t[0]];
		a[s]=t[side];
		a[(s+n/2)%n]=t[!side];
		dfs(t[side]);
	}
	if(!v[(s+n/2)%n])
		swap(a[s],a[(s+n/2)%n]),dfs(a[s]);
}
int main()
{
	cin>>n;
	if(n%2)
		puts("-1");
	else
	{
		fill(a,a+n,-1);
		dfs(0);
		printf("0");
		for(int now=a[0]; now; now=a[now])
			printf(" %d",now);
		printf(" 0\n");
	}
	return 0;
}
