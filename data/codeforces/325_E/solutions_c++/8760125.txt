#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <string>
using namespace std;
const int Maxn=100005;
int n;
bool f[Maxn];

void dfs(int x) {
	f[x]=true;
	if (x!=1) {
		if (!f[(x+n)/2]) dfs((x+n)/2);
		else dfs(x/2);
	}
	printf(" %d",x);
}

int main() {
	cin>>n;
	if (n&1) puts("-1");
	else printf("0"),dfs(0);
	return 0;
}
