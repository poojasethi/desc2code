#include <cstdio>
bool v[1<<20];
int n,a[1<<20];
int main()
{
    scanf("%d",&n);
    if(n%2)puts("-1");
    else{
	int t,i=n;
	n/=2;
	while(i){
	    v[t=a[i--]]=1;
	    t=t/2+n;
	    a[i]=t;
	    if(v[t])a[i]-=n;
	}
	printf("0");
	while(a[++i]) printf(" %d",a[i]);
	puts(" 0");
    }
    return 0;
}
