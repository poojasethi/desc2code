#include <cstdio>
using namespace std;
const int N=100004;
int n;
bool vis[N];
void dfs(int x)
{
	vis[x]=1;
	if (x==1) {printf("0 1"); return;}
	dfs(vis[(x+n)/2]?(x/2):(x+n)/2);
	printf(" %d",x);
}
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++) vis[i]=0;
	if (n%2) printf("-1"); else dfs(0);
	putchar('\n');
	return 0;
}
