def main():
    n = input()
    if n&1:
        print "-1"
        return

    seen, r, pos, cur = [False]*(10**6), [0]*(10**6), 0, 0
    while 1:
        seen[cur] = True
        r[pos] = cur
        pos += 1
        if not seen[(cur+n)>>1]:
            cur = (cur+n)>>1
        elif not seen[cur>>1]:
            cur = cur>>1
        else:
            break
    print " ".join(str(r[i]) for i in xrange(n,-1,-1))

main()
