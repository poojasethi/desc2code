import sys
def main():
    x, t, a, b, da, db = map(int,sys.stdin.readline().strip().split())

    if x==0:
        return "YES"
    for q in range(t):
        for w in range(t):
            if x == (max(0,a - q*da) + max(0,b - w*db)) or x==(max(a - q*da,0)) or (x == max(0,b-w*db)):
                return 'YES'
    return 'NO'
if __name__ == "__main__":
    print main()
