x, t, a, b, da, db = map(int, raw_input().split())
ok = 0
for i in xrange(t):
  for j in xrange(t):
    if x == a - (da*i) + b - (db*j):
      ok = 1
      break
    if x == a - (da*i):
      ok = 1
      break
    if x == b - (db*j):
      ok= 1
      break
    if x == 0:
      ok = 1
      break
print 'YES' if ok else 'NO'
