def __main__():
    a = [int(i) for i in raw_input().split(' ') ]

    res = False
    for i in range(0, a[1]):
        for j in range(0, a[1]):
            res |= (a[0] == a[2] - i*a[4]) or (a[0] == a[3] - j*a[5]) or (a[0] == a[2] + a[3] - i*a[4] - j*a[5])

    if res or a[0] == 0:
        print 'YES'
    else:
        print 'NO'
        

__main__()
