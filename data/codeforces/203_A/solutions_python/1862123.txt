
x,t,a,b,da,db = map(int,raw_input().split())
def solve():
    if x == 0: return True;

    for i in range(t):
        if x == a-i*da or x == b - i*db: return True;
    for i in range(t):
        for j in range(t):
            if x == a+b-i*da - j*db : return True;


if solve() :
    print "YES"
else:
    print "NO"
