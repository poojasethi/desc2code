import sys
from itertools import product

file = sys.stdin
#file = open("test", "r")
x, t, a, b, da, db =  map(int, file.readline().rstrip().split())
if x == 0: print "YES"; exit()
for i, j in [(i, j) for i in range(t) for j in range(t)]:
    if a - i*da + b - j*db == x or a - i*da == x or b - j*db == x:
        print "YES"
        break
else:
    print "NO"
