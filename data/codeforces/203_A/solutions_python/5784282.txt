x, t, a, b, da, db = map(int, raw_input().split())
af = a - (t - 1) * da
bf = b - (t - 1) * db;

res = False
if (x == 0):
	res = True
	
for i in xrange(a, af - 1, -da):
	for j in xrange(b, bf - 1, -db):
		#print i, j
		if (i + j == x) or (i == x) or (j == x):
			res = True
			
if res:
	print "YES"
else:
	print "NO"
