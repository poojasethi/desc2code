x, t, a, b, da, db = [int(i) for i in raw_input().split()]
pas = [a-da*i for i in xrange(t) if 0 < a-da*i <= x]
pbs = [b-db*i for i in xrange(t) if 0 < b-db*i <= x]
pas += [0]
pbs += [0]
for pa in pas:
    for pb in pbs:
        if pa+pb == x:
            print 'YES'
            exit()
print 'NO'
