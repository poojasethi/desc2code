import sys

(x, t, a, b, ta, tb) = [int(x) for x in sys.stdin.readline().strip().split()]

aa = [max(a - ta*i, 0) for i in range(0, t)]
bb = [max(b - tb*i, 0) for i in range(0, t)]

aa.append(0)
bb.append(0)

yes = False

for xa in aa:
    for xb in bb:
        if xa + xb == x:
            yes = True
if yes:
    print('YES')
else:
    print('NO')
