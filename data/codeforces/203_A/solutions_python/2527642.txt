def stripped_line():
    return raw_input().strip()


def read_ints():
    return map(int, stripped_line().split())

x, t, a, b, da, db = read_ints()

ar, br = [0, a], [0, b]

for i in xrange(1, t):
    ar.append(a - (i * da))
    br.append(b - (i * db))

for am in ar:
    for bm in br:
        if (am + bm) == x:
            print 'YES'
            exit(0)
print 'NO'
