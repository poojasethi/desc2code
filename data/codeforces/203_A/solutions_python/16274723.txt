


x, t, a, b, da, db = [int(x) for x in raw_input().split()]

possible = False

for i in range(t):
  if x == (a - i*da) or x == (b - i*db):
    possible = True
    break	
    
  for j in range(t):
    if x == (a - i * da) + (b - j * db):
      possible = True
      break
  if possible:
    break
	
if possible or x == 0:
  print "YES"
else:
  print "NO"
