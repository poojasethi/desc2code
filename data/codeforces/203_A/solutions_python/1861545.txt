import sys

def main():
	x,t,a,b,ta,tb = map(int, sys.stdin.readline().strip().split())

	if (x==0): return 'YES'
	for i in range(t):
		for j in range(t):
			if (x == a-i*ta): return 'YES'
			if (x == b-i*tb): return 'YES'
			if (x == a-i*ta + b - j*tb): return 'YES'
	return 'NO'


if __name__ == "__main__":
 print main()
