[x,tt,a,b,da,db] = map(int, raw_input().split())
res = "NO"
for t in xrange(tt):
  for s in xrange(tt):
    if x == (max(0, a - da * t) + max(0, b - db * s)):
      res =  "YES"
    if x == (max(0, a - da * t)):
      res = "YES"
    if x == (max(0, b - db * s)):
      res = "YES"
if x == 0:
  res = "YES"

print res
