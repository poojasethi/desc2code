#!/usr/bin/env python
x,t,a,b,da,db = map(int,raw_input().split())
la = set()
lb = set()
for i in xrange(t):
	la.add(a-i*da)
	lb.add(b-i*db)
for i in xrange(x+1):
	if i in la and x-i in lb:
		print "YES"
		break
else:
	if x in la or x in lb:
		print "YES"
	elif x == 0:
		print "YES"
	else:
		print "NO"
