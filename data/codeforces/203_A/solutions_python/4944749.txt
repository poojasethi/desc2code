x, t, a, b, da, db = map(int, raw_input().strip().split())
aa = [0, a]
bb = [0, b]

f = False
for i in range(t-1):
    a = a-da
    b = b-db
    aa.append(a)
    bb.append(b)

for i in aa:
    for j in bb:
        if i+j == x or x == 0:
            f = True

if f:
    print 'YES'
else:
    print 'NO'
