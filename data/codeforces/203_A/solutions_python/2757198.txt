import sys
x,t,a,b,da,db=map(int,raw_input().split())
if x==0 :print "YES" 
else :
	for i in range (t):
		if(x==a-da*i):
			print "YES"
			sys.exit()
	for i in range (t):
		if(x==b-db*i):
			print "YES"
			sys.exit()
	for i in range (t):
		for j in range (t):
			if(x==a-da*i + b-db*j):
				print "YES"
				sys.exit()
	print "NO"
