import sys

debug = False
x,t,a,b,da,db = map(int, raw_input('').split(' '))
try:
    for i in range(a, a-(t-1)*da-1, -da)+[0]: 
        for j in range(b, b-(t-1)*db-1, -db)+[0]:
            c = i+j
            if debug: print(c)
            if c==x:
                print('YES')
                sys.exit(0)
except SystemExit:
    pass
else:
    print('NO')
