#include<iostream>
#include<cstdio>
using namespace std;

int T;

int val (int x,int dx,int t){
	if(t==T)return 0;
	return x-dx*t;
}

int main(){
	int x,a,b,da,db;
	cin>>x>>T>>a>>b>>da>>db;
	for(int i=0;i<=T;++i){
		for(int j=0;j<=T;++j){
			if(val(a,da,i)+val(b,db,j)==x){
				puts("YES");
				return 0;
			}
		}
	}
	puts("NO");
}
