#include<cstdio>
char* yn[2]={"NO","YES"};
int main(){
	   int x,t,a,b,da,db,flag=0;
	   int i,j;
	   scanf("%d%d%d%d%d%d",&x,&t,&a,&b,&da,&db);
	   for(i=0;i<t;i++)
			 for(j=0;j<t;j++)
				    if((x==(a+b-da*i-db*j))||(x==a-i*da)||(x==b-i*db)||(x==0))
						  flag=1;
	   printf("%s",yn[flag]);
	   return 0;
}
