# include <stdio.h>

int a,b,c,d,e,f;

int main()
{
	scanf("%d %d %d %d %d %d",&a,&b,&c,&d,&e,&f);
	
	if(a==0)
	{printf("YES");return 0;}
	
	for(int i=0;i<b;i++)
	for(int j=0;j<b;j++)
	if(c-e*i+d-f*j==a || c-e*i==a || d-f*j==a)
	{printf("YES");return 0;}
	
	printf("NO");
	
	getchar();
	getchar();
}
