#include <stdio.h>
int main()
{
    int s,t,a,b,A,B;
    scanf("%d%d%d%d%d%d",&s,&t,&a,&b,&A,&B);
    bool f=!s;
    for(int i=0;!f && i<t;++i)
        for(int j=0;!f && j<t;++j)
            f=s==a-i*A+b-j*B || s==a-j*A || s==b-j*B;
    if(f)
        printf("YES");
    else
        printf("NO");
    return 0;
}
