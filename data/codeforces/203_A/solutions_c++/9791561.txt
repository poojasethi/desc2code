#include <bits/stdc++.h>

using namespace std;

int main () {
	int x, t, a, b, da, db;
	cin >> x >> t >> a >> b>> da >> db;

	for (int i=0; i<t; i++) {
		for (int j=0; j<t; j++) {
			if(x==a-i*da+b-j*db || x==a-i*da || x==b-j*db || x==0) {
				printf("YES");
				return 0;
			}
		}
	}
	printf("NO");
	return 0;
}
