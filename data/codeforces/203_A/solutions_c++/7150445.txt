#include<bits/stdc++.h>
using namespace std;
int main()
{
	int x,t,a,b,da,db,i,j,n,m,o;
	cin>>x>>t>>a>>b>>da>>db;
	for(i=0;i<t;i++)
	{
		for(j=0;j<t;j++)
		{
			n=a-da*i+b-db*j;
			m=a-da*i;
			o=b-db*j;
			if(n==x || m==x || o==x || !x)	
			{
				cout<<"YES\n";
				return 0;
			}
		}
	}
	cout<<"NO\n";
	return 0;
}
