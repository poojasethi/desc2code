#include <cstdio>
#include <iostream>

using namespace std;

int x,t,A,B,da,db;

int main(){
	
	cin >> x >> t >> A >> B >>  da >>db;
	
	for(int i=0;i<t;i++)
		for(int j=0;j<t;j++)
			if(A+B-da*i-db*j==x || A-da*i==x || B-db*j==x || x==0){
				
				puts("YES");
				return 0;
			}
			
	puts("NO");

	return 0;
}
