#include<cstdio>
int x,t,a,b,_a,_b;
int main(){
scanf("%d%d%d%d%d%d", &x,&t,&a,&b,&_a,&_b);
bool f=false;
for(int i = 0; i < t;++i)
for(int j=0;j<t;++j){
if(x==a+b-i*_a-j*_b)f=true;
}
if(x==0)f=true;
for(int i = 0; i < t;++i)
if(x==a-i*_a)f=true;
for(int i = 0; i < t;++i)
if(x==b-i*_b)f=true;
puts(f?"YES":"NO");
}
