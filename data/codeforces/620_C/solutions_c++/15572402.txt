#include<cstdio>
#include<cstring>
#include<set>
using namespace std;
const int N=300005;
int pos[N];
int main()
{
	set<int>st;
	int n;
	scanf("%d",&n);
	int x;
	pos[1]=1;
	int op=1;
	for(int i=1;i<=n;i++){
		scanf("%d",&x);
		if(!st.count(x)){
			st.insert(x);
		}
		else{
			st.clear();
			pos[++op]=i;
			pos[++op]=i+1;
		}
	}
	pos[op/2*2]=n;
	if(op==1)
		printf("-1\n");
	else{
		printf("%d\n",op/2);
		for(int i=1;i<=op/2;i++)
			printf("%d %d\n",pos[2*i-1],pos[2*i]);
	}
	return 0;
}
