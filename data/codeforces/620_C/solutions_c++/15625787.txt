#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

using namespace std;

int n,s[300005],pv=1;
int cnt=0,l[151515],r[151515];
map<int,bool> M;
int main(){
  scanf("%d",&n);
  FI(i,1,n) scanf("%d",&s[i]);
  FI(i,1,n){
    if(M.count(s[i])){
      l[++cnt]=pv;
      r[cnt]=i;
      pv=i+1;
      M.clear();
    }else M[s[i]]=true;
  }
  if(pv!=n+1){
    if(cnt==0) cnt--;
    else r[cnt]=n;
  }
  printf("%d\n",cnt);
  FI(i,1,cnt) printf("%d %d\n",l[i],r[i]);
  return 0;
}
