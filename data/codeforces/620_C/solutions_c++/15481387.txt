#include<iostream>
#include<set> 	 	
using namespace std;
set <long long> s;
long long n,a[400000][3],x,l,r,z,dp;
int main()
{
	z=1;
	cin>>n;
	l=1;
	for (int i=1;i<=n;++i)
	{
		cin>>x;
		if (s.find(x)!=s.end())
		{
			r=i;
			a[z][1]=l;
			a[z][2]=r;
			l=i+1;
			z++;
			dp=1;
			s.clear();
		}
		else
		{
			
		s.insert(x);	
			
			
		}
		
		
		
		
	}
	if (dp==0 or n==1 )
	{
		cout<<-1;
		return 0;
	}
	cout<<z-1<<endl;
	a[z-1][2]=n;
	for (int i=1;i<z;++i)
	cout<<a[i][1]<<" "<< a[i][2]<<endl;
	
}
