#include <bits/stdc++.h>
using namespace std;
int ans,res[300005],prv;
map<int,int> occ;
int main() {
	int n,i,v;scanf("%d",&n);
	for(i=1;i<=n;i++) {
		scanf("%d",&v);
		if(occ[v]>prv) {
			res[++ans]=i;
			prv=i;
		}
		else
			occ[v]=i;
	}

	if(!ans)
		puts("-1");
	else {
		res[ans]=n;res[0]=0;
		printf("%d\n",ans);
		for(i=1;i<=ans;i++)
			printf("%d %d\n",res[i-1]+1,res[i]);
	}
}
