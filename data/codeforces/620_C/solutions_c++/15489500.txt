#include<bits/stdc++.h>
using namespace std;
const int MAXN = 3e5 + 5;
int n, pos[MAXN], k;
set<int> s;
int main()
{
	scanf("%d", &n);
	for(int i = 1; i <= n; ++i) {
		int x;
		scanf("%d", &x);
		if(s.count(x) == 0) s.insert(x);
		else {
			pos[k++] = i;
			s.clear();
		}
	}
	if(k == 0) printf("-1\n");
	else {
		pos[k - 1] = n;
		printf("%d\n", k);
		printf("1 %d\n", pos[0]);
		for(int i = 0; i < k - 1; ++i)
			printf("%d %d\n", pos[i] + 1, pos[i + 1]);
	}
	return 0;
}
