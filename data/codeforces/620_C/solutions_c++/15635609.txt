#include <bits/stdc++.h>
using namespace std;

const int N=3e5+5;
map<long long int,int> mymap;

vector<long int>a,b; 

int main() {
	int t;
	scanf("%d",&t);
	long long int n;
	a.push_back(1);
	for(int i=0;i<t;i++){
		cin>>n;
		if(mymap[n]==0){
			mymap[n]=1;
		}
		else{
			mymap.clear();
			b.push_back(i+1);
			a.push_back(i+2);
		}
	}
	if(b.size()==0){
		printf("-1\n");
	}
	else{
		printf("%d\n",b.size());
		for(int i=0;i<b.size();i++){
			printf("%d %d\n",a[i],(i==b.size()-1)?t:b[i]);
		}
	}
	return 0;
}
