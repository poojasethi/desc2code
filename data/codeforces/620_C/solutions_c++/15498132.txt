#include<iostream>
#include<stdio.h>
#include<map>
int c[300005];
using namespace std;
int main()
{
	map<int,int>mp;
	int x,l=0;
	scanf("%d",&x);
	mp.clear();
	for(int i=0;i<x;i++)
	{
		int k;
		scanf("%d",&k);
		if(mp[k]==0)mp[k]=1;
		else 
		{
			c[l++]=i+1;
			mp.clear();
		}
	}
	int k=1;
	if(l==0)printf("-1\n");
	else 
	{
		printf("%d\n",l);
		for(int i=0;i<l;i++)
		{
			if(i==l-1)printf("%d %d\n",k,x);
			else printf("%d %d\n",k,c[i]);
			k=c[i]+1;
		}
	}
}
