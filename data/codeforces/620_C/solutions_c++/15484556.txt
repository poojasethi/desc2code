#include <bits/stdc++.h>
using namespace std;
const int N = 3e5+5;
map<int,int> hash;
int ret[N];
int main(){
	int n,i,j,x,last=0,cc=0;cin>>n;
	for(i=1;i<=n;++i){cin>>x;
		if(hash[x]>last){
			ret[cc++]= i;
			hash[x]= i;
			last= i;
		}
		else
			hash[x]= i;
	}
	if(!cc)cout<<-1<<endl;
	else{
		cout<<cc<<endl;
		ret[cc-1]= n;
		printf("1 %d\n",ret[0]);
		for(i=1;i<cc;++i)
			printf("%d %d\n",ret[i-1]+1,ret[i]);
	}
}
