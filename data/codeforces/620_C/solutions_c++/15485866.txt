#include<bits/stdc++.h>
using namespace std;
int main(){

	int n,i,j,a[300005],l,count,x[300005],y[300005];
	set<int> s;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&a[i]);
	}
	l = 0;
	count = 0;
	for(i=0;i<n;i++){
		if(s.find(a[i]) != s.end()){
			x[count] = l;
			y[count] = i;
			l = i+1;
			count++;
			s.clear();
		}
		else
			s.insert(a[i]);
	}
	if(count == 0)
		printf("-1\n");
	else{
	y[count-1] = n-1;
	printf("%d\n",count);
	for(i=0;i<count;i++){
		printf("%d %d\n",x[i]+1,y[i]+1);
	}
	}
	return 0;
}
