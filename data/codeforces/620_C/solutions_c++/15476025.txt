#include <cstdio>
#include <utility>
#include <vector>
#include <map>

using namespace std;


int main() {
	int n;
	scanf("%d",&n);
	int a,b=1;
	int m=1;
	map<int,int> seen;
	vector<pair<int,int> > ans;
	for(int i=0;i<n;i++) {
		scanf("%d",&a);
		if (seen[a] == m) {
			ans.push_back(make_pair(b,i+1));
			b=i+2;
			m++;
		} else {
			seen[a]=m;
		}
	}
	if (ans.empty()) printf("-1\n");
	else {
		printf("%d\n",(int)ans.size());
		ans[ans.size()-1].second = n;
		for(int i=0;i<ans.size();i++) {
			printf("%d %d\n",ans[i].first,ans[i].second);
		}
	}
	return 0;
}
