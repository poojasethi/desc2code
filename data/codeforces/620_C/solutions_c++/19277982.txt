#include<cstdio>
#include<algorithm>
#include<set>
using namespace std;
int b[1000000];
int main()
{
	set<int> a;
	int  i, sum = 0, k = 0, t;
	int n;
	scanf("%d",&n);
	for( i = 0 ; i < n ; i++ )
	{
		scanf("%d",&t);
		
		if( a.count(t) == 0 )
		a.insert(t);
		else {
			sum++;
			b[k++] = i;
			a.clear();
			
		}
										
	}
	b[k-1] = n-1;
	
	if( sum == 0 ) printf("-1\n");
	else 
	{
	printf("%d\n",sum);
	printf("1 %d\n",b[0]+1);
	
	for( i = 0 ; i < sum -1 ; i++)	
	printf("%d %d\n",b[i] + 2 , b[i+1] +1);
			
    }
	return 0;	
}
