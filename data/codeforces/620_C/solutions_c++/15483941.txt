#include <cstdio>
#include <set>
using namespace std;

set<int> s;
int n, a, ansL[300005], ansR[300005], ans = 0, start = 0;

int main(){
	scanf("%d", &n);
	for(int i = 0; i < n; i++){
		scanf("%d", &a);
		if(s.find(a) != s.end()){
			ansL[ans] = start;
			ansR[ans++] = i;
			start = i + 1;
			s.clear();
		}else s.insert(a);
	}
	if(start < n) ansR[ans - 1] = n - 1;
	s.clear();
	printf("%d\n", ans > 0 ? ans : -1);
	for(int i = 0; i < ans; i++) printf("%d %d\n", ansL[i] + 1, ansR[i] + 1);
	return 0;
}
