#include<iostream>
#include<map>
using namespace std;
map<long long ,long>m;
long long v[300003];
struct sajal{long i,f;
};
sajal a[300004];
int main(){
	long n;
	cin>>n;
	for(long i=1;i<=n;i++){
		cin>>v[i];
	}
	long u=1;long k=0;
	for(long i=1;i<=n;i++){
		long long h=v[i];
		if(m[h]!=0){
			if(m[h]>=u){k=k+1;a[k].i=u;a[k].f=i;m[h]=i;u=i+1;
			}
			else{m[h]=i;
			}
		}
		else m[h]=i;
	}
	if(k==0)cout<<"-1"<<endl;
	else{
	cout<<k<<endl;
	for(long i=1;i<k;i++)cout<<a[i].i<<" "<<a[i].f<<endl;
	cout<<a[k].i<<" "<<n<<endl;
	
}}
