#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int a[300005];
int main(){
	int n,i,x,left,right;
	scanf("%d",&n);
	vector< pair<int,int> >v;
	for(i=0;i<n;i++){
		scanf("%d",&a[i]);
	}
	map<int,int> m;
	left=0;
	for(i=0;i<n;i++){
		x=a[i];
		m[x]+=1;
		if(m[x]==2){
			v.push_back(make_pair(left+1,i+1));
			left=i+1;
			m.clear();
		}
	}
	ll size=v.size();
	if(size==0){
		cout<<-1<<endl;
		return 0;
	}
	cout<<size<<endl;
	v[size-1].second=n;
	for(i=0;i<size;i++){
		cout<<v[i].first<<" "<<v[i].second<<endl;
	}
}
