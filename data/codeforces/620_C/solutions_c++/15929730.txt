#include <stdio.h>
#include <set>
using namespace std;

int N, S[300010], SC = 1;
set<int> D;

int main() {
  scanf("%d", &N);
  for (int i = 1; i <= N; i++) {
    int v;
    scanf("%d", &v);
    if (D.find(v) == D.end()) {
      D.insert(v);
    } else {
      S[SC++] = i;
      D.clear();
    }
  }
  if (SC > 1) {
    S[0] = 0;
    S[SC-1] = N;
    printf("%d\n", SC-1);
    for (int i = 1; i < SC; i++)
      printf("%d %d\n", S[i-1]+1, S[i]);
  } else {
    printf("-1\n");
  }

  return 0;
}
