#include<iostream>
#include<set>
const int qq=3e5+10;
using namespace std;
set<int>p;
int x[qq];
int main()
{
	int n,m,c;	cin >> n;
	c=0;
	for(int i=1;i<=n;++i){
		cin >> m;
		if(p.count(m)==0)	p.insert(m);
		else{
			x[c++]=i;
			p.clear();
		}
	}
	if(!c)	cout << "-1" << endl;
	else{
		x[c-1]=n;
		cout << c << endl;
		cout << "1" << " " << x[0] << endl;
		for(int i=0;i<c-1;++i)
			cout << x[i]+1 << " " << x[i+1] << endl; 
	}
	return 0;
}
