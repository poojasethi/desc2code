#include<bits/stdc++.h>

using namespace std;

long long x,n,m,k,l,p,o,i,j,s;

int main(){
	queue<int> b,c;
	set<int> a;
	set<int>::iterator it;
	
	cin>>n;
	p=1;
	k=0;
	for(i=1;i<=n;i++){
	  cin>>x;
	  if(a.find(x)!=a.end()){
  	  b.push(p);
  	  c.push(i);
  	  p=i+1;
  	  a.clear();
  	  k++;
		}
		else{
		  a.insert(x);
		}
	}
	if(k==0){
	  cout<<-1<<endl;
	  return 0;
	}
	cout<<k<<endl;
	for(i=1;i<k;i++){
	 cout<<b.front()<<" "<<c.front()<<endl;
	 b.pop();
	 c.pop();
	}
	cout<<b.front()<<" "<<n<<endl;
}
