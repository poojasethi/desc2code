#include <bits/stdc++.h>
using namespace std;
int n, a;
map<int, int> m;
vector< pair<int, int> > v;
bool has;

int main()
{
	cin >> n;
	int i = 0, j = 0;
	while(i < n && j < n)
	{
		cin >> a;
		if(m[a])
		{
			has = 1;
			v.push_back(make_pair(i + 1, j + 1));
			m.clear();
			i = j + 1;
			j = i;
		}
		else
		{
			has = 0;
			m[a]++;
			j++;
		}
	}

	if(v.size())
	{
		if(!has) v[v.size() - 1].second = n;
		cout << v.size() << endl;
 		for(int i = 0; i < v.size(); i++)
		{
			cout << v[i].first << " " << v[i].second << endl;
		}
	}
	else cout << -1;

}
