#include <bits/stdc++.h>

using namespace std;

#define fi first
#define se second

typedef long long ll;
typedef pair<ll,ll> ii;

ll N, x;

map<ll,ll> m;
vector<ii> v;

int main(){

	ios::sync_with_stdio(false);

	cin >> N;

	ll u = 1, cnt = 0;
	for (int i=1; i<=N; i++){
		cin >> x;

		if (m[x] == 0){
			m[x] = 1;
		}
		else{
			cnt++;
			m.clear();
			v.push_back(ii(u,i));
			u = i + 1;
		}
	}

	cout << (cnt == 0 ? -1 : cnt) << endl;
	for (int i=0; i<v.size(); i++) cout << v[i].fi <<  " " << ( i == v.size() - 1 ? N : v[i].se) << endl;

	return 0;
}
