#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	
	scanf("%d", &n);

	vector<int> ans;
	set<int> s;
	for (int i = 0; i < n; i++)
	{
		int x;
		scanf("%d", &x);
		if (s.count(x))
		{
			ans.push_back(i);
			s.clear();
		}
		else
		{
			s.insert(x);
		}
	}
	if (ans.size() == 0)
	{
		printf("-1\n");
		return 0;
	}
	ans.back() = n-1;

	printf("%d\n", (int)ans.size());
	int first = 0;
	for (int i = 0; i < ans.size(); i++)
	{
		printf("%d %d\n", first+1, ans[i]+1);
		first = ans[i]+1;
	}
	return 0;
}
