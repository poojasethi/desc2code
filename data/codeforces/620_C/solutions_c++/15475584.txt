#include <cstdio>
#include <algorithm>
#include <cstring>
#include <map>
using namespace std;

map <int,int> Map;

const int maxn = 300000+100;
int n, a[maxn], p[maxn], cnt;

int main() {
	scanf("%d",&n);
	for(int i = 1; i <= n; i++) {
		scanf("%d",&a[i]);
		if(Map.count(a[i])) {
			p[++cnt] = i;
			for(int j = p[cnt-1]+1; j < i; j++) Map.erase(a[j]);
		} else Map[a[i]] = 1;
	}
	p[cnt] = n;
	if(!cnt) puts("-1");
	else {
		printf("%d\n",cnt);
		for(int i = 1; i <= cnt; i++) printf("%d %d\n",p[i-1]+1,p[i]);
	}
	return 0;
}
