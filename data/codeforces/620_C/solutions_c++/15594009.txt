#include<cstdio>
#include<map>
#include<algorithm>
using namespace std;
	int a[300005],ans[300005][2];
main(){
	int n;
	while(scanf("%d",&n)!=EOF){
		int i,begin=1,t=0;
        map<int,int>q;
		for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
		for(i=1;i<=n;i++){
			if(q[a[i]]==1){
				ans[t][0]=begin;
				ans[t++][1]=i;
				begin=i+1;
				q.clear();
			}
			else
			q[a[i]]=1;
		}
		if(t==0)
		printf("-1");
		else{
			printf("%d\n",t);
			ans[t-1][1]=n;
			for(i=0;i<=t-1;i++)
			printf("%d %d\n",ans[i][0],ans[i][1]);
		}
		
	}
	return 0;
}
