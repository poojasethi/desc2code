#include<bits/stdc++.h>
using namespace std;
struct pairs
{
    int x; int y;
} p[151000];
int a[300005];
int main()
{
ios::sync_with_stdio(0);
int n,i,k=0,j;
cin>>n;
for(int i=0;i<n;i++)cin>>a[i];
for(i=0,j=0;i<n;i=j)
{
     set <int> s;
     while(j<n && !s.count(a[j]))
     s.insert(a[j++]);
     if(j==n) break;
     else { p[k].x=i; p[k++].y=j;}
     j++;
}
p[k-1].y=max(p[k-1].y,n-1);
if(k==0) { cout<<"-1\n"; return 0;}
cout<<k<<"\n";
for(i=0;i<k;i++){
    cout<<p[i].x+1<<" "<<p[i].y+1<<"\n";
}
return 0;
}
