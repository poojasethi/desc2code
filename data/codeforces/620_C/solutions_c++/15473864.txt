#include <stdio.h>
#include <map>
using namespace std;

typedef pair <int, int> pii;
const int MAX = 3e5 + 1;
int n, res, a, pos;
pii seg[MAX];
map <int, int> cnt;

int main(){
	scanf("%d", &n);
	pos = 1;
	for (int i = 1; i <= n; ++i){
		scanf("%d", &a);
		if (++cnt[a] == 2){
			seg[++res] = pii(pos, i);
			pos = i + 1;
			cnt.clear();
		}
	}

	if (res){
		printf("%d\n", res);
		seg[res].second = n;
		for (int i = 1; i <= res; ++i)
			printf("%d %d\n", seg[i].first, seg[i].second);
	}
	else
		printf("-1");

	return 0;
}
