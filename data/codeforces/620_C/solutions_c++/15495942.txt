#include<bits/stdc++.h>
using namespace std;

map<int,int>mera;
int ara[300005];
int main(){
    int n,cnt=0,i,j,k,p,first=1;
    scanf("%d",&n);

    for(i=1;i<=n;i++){
        scanf("%d",&p);
        mera[p]++;
        if(mera[p]==2){
            ara[2*cnt]=first;
            ara[2*cnt+1]=i;
            cnt++;
            first=i+1;
            mera.clear();}
    }

    if(cnt==0){
        printf("-1");
        return 0;}
    ara[2*cnt-1]=n;
    cout<<cnt<<endl;
    for(int i=0;i<cnt;i++) printf("%d %d\n",ara[2*i],ara[2*i+1]);
}
