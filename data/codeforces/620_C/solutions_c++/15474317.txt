#include <iostream>
#include<stdio.h>
#include<map>
#include<vector>
using namespace std;
int n,x,r;
vector<pair<int,int> >v;
map<int,int> mp;

int main() {
	scanf("%d",&n);
	int l=1;
	for(int i=1;i<=n;i++){
		scanf("%d",&x);
		if(mp[x]==1){
			v.push_back(make_pair(l,i));
			mp.clear();
			l=i+1;
			r=i;
		}
		else
			mp[x]++;
	}
	if(v.size()<1){
		printf("%d\n",-1);
		return 0;
	}
	cout<<v.size()<<endl;
	v[v.size()-1].second=n;
	for(int i=0;i<v.size();i++)
	printf("%d %d\n",v[i].first,v[i].second);
		
	return 0;
}
