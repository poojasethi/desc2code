#include <iostream>
#include <bits/stdc++.h>
#define ll long long int
#define mod 1000000009

using namespace std;

int main()
{
    ll n,i,val;
    cin>>n;
    set<ll>st;
    ll pos[n+1],k=0;
    for(i=1;i<=n;i++){
      cin>>val;
      if(st.count(val)==0)st.insert(val);
      else{
        pos[k++]=i;
        st.clear();
      }
    }

    if(k==0){cout<<"-1"<<endl;return 0;}
    pos[k-1]=n;
    cout<<k<<endl;
    cout<<"1"<<" "<<pos[0]<<endl;
    for(i=0;i<k-1;i++){
      cout<<pos[i]+1<<" "<<pos[i+1]<<endl;
    }
}
