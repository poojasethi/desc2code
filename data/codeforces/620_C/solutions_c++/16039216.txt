#include<iostream>
#include<map>
#include<vector>
#include<cstdio>
using namespace std;

int main(){
	int n,x,c[300001];
	vector<int>a;
	cin>>n;
	map<int, int>b;
	a.push_back(1);
	for(int i=1;i<=n;i++){
		scanf("%d", &c[i]);
		b[c[i]]++;
		if(b[c[i]]>1){
			for(int j=a[a.size()-1];j<=i;j++)
				b[c[j]]=0;
			a.push_back(i);
			a.push_back(i+1);
		}
	}
	if(a.size()<2){
		cout<<-1;
		return 0;
	}
	a[a.size()-2]=n;
	cout<<a.size()/2<<endl;
	for(int i=0;i<a.size()-1;i+=2){
		printf("%d %d\n", a[i],a[i+1]);
	}
return 0;
}
