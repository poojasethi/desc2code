
#include<bits/stdc++.h>

typedef int LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,i;
cin>>n;

vector <LL> v(n);
vector <pair<LL,LL> > p;

for(i=0;i<n;i++)
	scanf("%d",&v[i]);

map <LL,LL> mp;

LL part=1,st=0;


for(i=0;i<n;i++)
{
if(mp[v[i]]==part)
	{p.push_back(make_pair(st+1,i+1));
	st=i+1;
	part++;
	}
else
mp[v[i]]=part;
}


if(!p.empty() && st!=n)
	p[p.size()-1].second=n;
if(p.empty())
	{cout<<-1<<endl;return 0;}


cout<<p.size()<<endl;
for(i=0;i<p.size();i++)
	printf("%d %d\n",p[i].first,p[i].second);

}
