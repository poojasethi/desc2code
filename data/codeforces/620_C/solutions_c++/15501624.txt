#include<bits/stdc++.h>

using namespace std;

int main(){
	long long n,m,k,l,p,i,o,x;
  queue<int> b,c;
  set<int> a;
  cin>>n;
  p=1;
  k=0;
  for(i=1;i<=n;i++){
 	  cin>>x;
 	  if(a.find(x)!=a.end()){
	  	b.push(p);
	  	c.push(i);
	  	p=i+1;
	  	k++;
	  	a.clear();
	  }
	  else a.insert(x);
	}
	if(k==0){
	  cout<<-1<<endl;
	  return 0;
	}
	cout<<k<<endl;
	for(i=1;i<k;i++){
	  cout<<b.front()<<" "<<c.front()<<endl;
	  b.pop();
	  c.pop();
	}
	cout<<b.front()<<" "<<n<<endl;
}
