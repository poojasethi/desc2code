#include <bits/stdc++.h>
using namespace std;
#define maxn 300001

int p[maxn];
set<int> se;
vector<int> v;

int main()
{
	int n, i, x;
	scanf("%d", &n);
	for(i = 1; i <= n; i++)
		scanf("%d", p + i);
	for(i = 1; i <= n; i++)
		if(se.find(p[i]) == se.end()) se.insert(p[i]);
		else
		{
			v.push_back(i);
			se.clear();
		}
	if(v.size() == 0) printf("-1");
	else
	{
		printf("%u\n", v.size());
		v.back() = n;
		x = 1;
		for(i = 0; i < v.size(); i++)
		{
			printf("%d %d\n", x, v[i]);
			x = v[i] + 1;
		}
	}
	return 0;
}
