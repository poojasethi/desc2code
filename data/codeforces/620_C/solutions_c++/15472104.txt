#include<bits/stdc++.h>
#define N 300005
using namespace std;
map<int,int>F;int a[N],L[N],R[N];
int n,i,j,ans;
int main(){
	scanf("%d",&n);
	for (i=1;i<=n;i++){
		scanf("%d",&a[i]);
		if (F.find(a[i])!=F.end()) ans=1;
		F[a[i]]=1;
  }if (!ans) return puts("-1"),0;
	ans=0;
  for (i=1;i<=n;i=j+1){
  	F.clear();
  	for (j=i;j<=n;j++){
  		if (F.find(a[j])!=F.end()) break;
  		F[a[j]]=1;
    }if (j>n) R[ans]=n;
		else L[++ans]=i,R[ans]=j;
  }printf("%d\n",ans);
  for (i=1;i<=ans;i++)
  	printf("%d %d\n",L[i],R[i]);
}
