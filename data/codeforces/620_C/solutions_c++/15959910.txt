#include <cstdio>
#include <map>
using namespace std;
map<int,int> k;
int a[300010];
struct
{
	int l;
	int r;
}ans[150005];
int main()
{
	int n,m=0,l=1,r;
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	for (int i=1;i<=n;i++)
	{
		if (k[a[i]]==1)
		{
			k.clear();
			ans[m].l=l;
			ans[m].r=i;
			l=i+1;
			m++;
		}
		else
		{
			k[a[i]]=1;
		}
	}
	if (ans[m-1].r!=n) ans[m-1].r=n;
	if (m)
	{
		printf("%d\n",m);
		for (int i=0;i<m;i++) printf("%d %d\n",ans[i].l,ans[i].r);
	}
	else
	{
		printf("-1");
	}
}
