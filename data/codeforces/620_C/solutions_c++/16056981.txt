#include <bits/stdc++.h>

using namespace std;
vector<long long> V;
set<long long> S;
int main()
{
    int n,q;
    cin>>n;
    V.push_back(0);
    for(int i=0 ; i<n ; i++)
    {
        cin>>q;
        if(S.count(q))
        {
            S.clear();
            V.push_back(i+1);
            continue;
        }
        S.insert(q);
    }
    V.back()=n;
    if(V.size()<2) {cout<<-1<<endl; return 0;}
    cout<<V.size()-1<<endl;
    for(int i=1 ; i<V.size() ; i++)
        cout<<V[i-1]+1<<" "<<V[i]<<endl;

    return 0;
}
