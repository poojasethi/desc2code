#include <bits/stdc++.h>
using namespace std;
	
int main() {
	int n, a, w = 0;
	cin >> n;	
		
	set<int> S;
	vector<int> v;
	v.push_back(0);
	for(int i = 0; i < n; i++) {
		scanf("%d", &a);
		if(S.find(a) == S.end())
			S.insert(a);
		else {
			S.clear();
			v.push_back(i+1);
			w++;
		}
	}
	if(w == 0) cout << -1;
	else {
		v.back() = n;
		cout << w << endl;
		for(int i = 0; i < v.size()-1; i++)
			cout << v[i]+1 << " " << v[i+1] << endl;

	}
}
