#include<bits/stdc++.h>
using namespace std;
set<int> s;
int n,x,ans=0,a[300010],b[300010];
int main(){
    scanf("%d",&n);
	for (int i=1; i<=n; i++){
	    scanf("%d",&x);
	    if (s.count(x)){
	    	ans++;
	    	a[ans]=b[ans-1]+1;
	    	b[ans]=i;
	    	s.clear();
	    }
	    else s.insert(x);
	}
	if (ans==0){
		printf("-1");
		return 0;
	}
	if (b[ans]!=n) b[ans]=n;
	printf("%d\n",ans); 
	for (int i=1; i<=ans; i++) printf("%d %d\n",a[i],b[i]);
}
