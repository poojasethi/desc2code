#include <iostream>
#include <cstdio>
#include <set>
#include <vector>
using namespace std;

int n, st, a, sz;
set<int> v;
vector<int> ans;
int main()
{
	int i;
//	freopen ("input.txt", "r", stdin);
	cin >> n;
	for (i = 1; i <= n; i++) {
		scanf ("%d", &a);
		if (v.count(a) == 0) v.insert(a);
		else {
			ans.push_back(i);
			v.clear();
		}
	}
	sz = ans.size();
	if (sz == 0) {cout << -1; return 0;}
	ans[sz - 1] = n;
	cout << sz << endl;
	st = 1;
	for (i = 0; i < sz; i++) {
		printf("%d %d\n", st, ans[i]);
		st = ans[i] + 1;
	}
    return 0;
}
