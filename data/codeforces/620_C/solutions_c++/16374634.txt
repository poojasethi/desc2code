#include<bits/stdc++.h>
using namespace std;
map<int,int> gr;
int a[300005];
vector< pair<int,int> > v;
int main()
{
int n;
scanf("%d",&n);
int i;
for(i=1;i<=n;i++)
scanf("%d",&a[i]);
for(i=1;i<=n;i++)
if(gr.find(a[i])==gr.end())gr[a[i]] = i;
else 
{
int x = gr[a[i]];
gr.clear();
v.push_back(make_pair(x,i));
}
if(v.size()==0)return 0*printf("-1\n");
printf("%d\n",v.size());
int pr = 1;
for(i=0;i<v.size()-1;i++){
printf("%d %d\n",pr,v[i].second);
pr = v[i].second+1;
}
printf("%d %d\n",pr,n);


return 0;
}
