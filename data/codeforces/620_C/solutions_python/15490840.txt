def solve():
    n = input()
    A = map(int,raw_input().split())
    if n == len(set(A)):
        print -1
        return

    ans = []
    l = 0
    r = 0
    i = 0
    visited = {}
    while i < n:
        if A[i] in visited:
            ans.append([l+1,r+1])
            l = r+1
            visited = {}
        else:
            visited[A[i]] = 1
        r += 1
        i+=1
    ans[-1][1] = n
    print len(ans)
    for tup in ans:
        print tup[0], tup[1]

solve()
