n=raw_input()
a=map(int,raw_input().split())
index = 0
dic = {}
res = []
for i in range(len(a)):
    if a[i] in dic:
        res.append([index+1,i+1])
        index = i+1
        dic={}
    
    else:
        dic[a[i]]=1

if(len(res)==0):
    print "-1"

else:
    print len(res)
    res[-1][1]=n # -1 means the last element  ->  res[len(res)-1][1]
    for i in res :
        print i[0],i[1]
