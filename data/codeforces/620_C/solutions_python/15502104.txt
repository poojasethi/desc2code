
n = input()

arr = map(int, raw_input().split())
result = []
temp = set()
c = 0
for i in range(n):
	if arr[i] in temp:
		result.append([c + 1, i + 1])
		temp = set()
		c = i + 1
	else:
		temp.add(arr[i])

if len(result) == 0:
	print "-1"
else:
	result[-1][1] = len(arr)
	print len(result)
