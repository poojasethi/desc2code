n = int(raw_input())
v = map(int, raw_input().split())

start = 0
s = set()
ans = []

for i,j in enumerate(v):
	if not(j in s):
		s.add(j)
	else:
		ans.append([start+1, i+1])
		s = set()
		start = i+1
		
if len(ans) == 0:
	print -1
else:
	print len(ans)
	ans[-1][1] = len(v)
	for i in ans:
		print i[0], i[1]
