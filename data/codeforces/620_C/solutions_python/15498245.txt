n = int( raw_input() )
A = map(int, raw_input().split())

M = {}
left = 1
IP = []
# traverse A
SZ = len(A)
for i in range(1, SZ + 1):
	if A[i-1] in M:
		M.clear()
		IP.append([left, i])
		left = i + 1
	else:
		M[ A[i-1] ] = 1

if IP:
	L = len(IP)
	# R for last seg
	IP[-1][1] = SZ
	print L
	for i in range(L):
		print IP[i][0], IP[i][1]
else:
	print -1
