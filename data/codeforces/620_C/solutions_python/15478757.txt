n = input()
arr = map(int,raw_input().split())
ans = []
d = {}
d[arr[0]] = 1
pre = 1
for inn in range(2,n+1):
	i = inn - 1
	if arr[i] in d:
		d = {}
		ans.append([pre,inn])
		pre = inn + 1
	else:
		d[arr[i]] = 1
l = len(ans)
if l==0:
	print -1
else:
	print len(ans)
	ans[-1][1]=n
	for item in ans:
		print item[0],item[1]
