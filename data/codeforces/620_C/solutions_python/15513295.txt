def main():
	n = input()
	a = map(int,raw_input().split())
	ans = 0
	ans_list=[]
	s = set()
	pre = 0
	for (i,ai) in enumerate(a):
		if ai in s:
			ans += 1
			ans_list.append([pre+1,i+1])
			pre = i+1
			s.clear()
		else:
			s.add(ai)
	if ans == 0:
		print -1
		return
	print ans
	for [l,r] in ans_list:
		ans -= 1
		if ans==0:
			print l,n
		else:	print l,r	
if __name__ == '__main__':
	main()
