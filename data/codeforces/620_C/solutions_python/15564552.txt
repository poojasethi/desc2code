#! /usr/bin/python

n = int(raw_input())
row = [int(x) for x in raw_input().split()]
result = []
l = 0
r = 0
currSet = set(row[l:r])
while r<n:
    if row[r] in currSet:
        result.append([l+1, r+1])
        l = r+1
        r = r+1
        currSet.clear()
    else:
        currSet.add(row[r])
        r = r+1
if len(result) == 0:
    print -1
else:
    print len(result)
    result[-1][1] = n
    for r in result:
        print " ".join(str(i) for i in r)
