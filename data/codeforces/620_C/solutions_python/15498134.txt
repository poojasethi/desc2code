#!/usr/bin/env python
# -*- coding:utf-8 -*-



def main():
    n = input()
    a = map(int, raw_input().split())
    s = set()
    last_pos = -1
    ans_list = []
    ans_cnt = 0
    for (i, ai) in enumerate(a):
        if ai in s:
            ans_cnt += 1
            ans_list.append([last_pos + 2, i + 1])
            last_pos = i
            s.clear()
        else:
            s.add(ai)

    if ans_cnt == 0:
        print -1
        return 

    print ans_cnt
    for (i, (l, r)) in enumerate(ans_list):
        if i == ans_cnt - 1:
            print l, n
        else:
            print l, r



if __name__ == '__main__':
    main()
