#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<vector>
#include<algorithm>
using namespace std;
typedef long long ll;
const int maxn=100002;
struct arr{
  int x,y;
  arr(){}
  arr(int _x,int _y):x(_x),y(_y){}
  bool operator<(const arr &i)const{
    return x<i.x || (x==i.x && y<i.y);
  }
}a[maxn],b[maxn];
int n,s;

int main(){
  int i,j,k,dx,dy;
  scanf("%d",&n);
  for (i=0;i<n;++i){
    scanf("%d%d",&a[i].x,&a[i].y);
    b[i].x=a[i].y,b[i].y=a[i].x;
  }
  sort(a,a+n);
  sort(b,b+n);
  for (i=0;i<n;++i){
    k=lower_bound(b,b+n,arr(a[i].y,a[i].x))-b;
    for (j=i+1;j<n && a[i].x==a[j].x && k<n && a[i].y==b[k].x;){
      dx=b[k].y-a[i].x;
      dy=a[j].y-a[i].y;
      if (dx<dy) ++k;
      else if (dx>dy) ++j;
      else {
        if (binary_search(a,a+n,arr(b[k].y,a[j].y))) ++s;
        ++j,++k;
      }
    }
  }
  printf("%d\n",s);
  return 0;
}
