#include <bits/stdc++.h>
using namespace std;

const int maxn = 222222;
int n, x, y;
set<int> pt[maxn];

int main() {
	for (int i = 0; i < maxn; ++i) pt[i].clear();
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) {
		scanf("%d%d", &x, &y);
		pt[x].insert(y);
	}
	
	int ans = 0, S = sqrt(n);
	for (int i = 0; i < maxn; ++i) {
		if (pt[i].size() == 0) continue;
		if (pt[i].size() > S) {
			for (int j = i + 1; j < maxn; ++j) {
				if (pt[j].size() < 2) continue;
				for (set<int>::iterator it = pt[j].begin(); it != pt[j].end(); it++) {
					if (pt[i].find(*it) != pt[i].end() && pt[i].find(*it + j - i) != pt[i].end() && pt[j].find(*it + j - i) != pt[j].end()) ans++;
				}
			}
		} else {
			for (set<int>::iterator it1 = pt[i].begin(); it1 != pt[i].end(); ++it1) {
				set<int>::iterator it2 = it1;
				it2++;
				for (; it2 != pt[i].end(); it2++) {
					int d = *it2 - *it1;
					if (pt[i + d].find(*it1) != pt[i + d].end() && pt[i + d].find(*it2) != pt[i + d].end()) ans++;
				}
			}
		}
	}
	cout << ans << endl;
}
