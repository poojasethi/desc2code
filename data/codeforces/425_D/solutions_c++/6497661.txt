#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <map>
#include <vector>
#define X first
#define Y second
using namespace std;
typedef pair<int,int> pii;
typedef vector<pii> vp;
typedef vp::iterator itor;
vp a,b;
int main()
{
    //freopen("problem425D.in","r",stdin);freopen("problem425D.out","w",stdout);
	int n;scanf("%d",&n);
	for(int i = 1;i<=n;i++)
	{ 
	 int x,y;scanf("%d%d",&x,&y);a.push_back(pii(x,y));b.push_back(pii(y,x));
	}
	int ans= 0;
	sort(a.begin(),a.end());sort(b.begin(),b.end());
	for(int i = 1;i<=n;i++)
	{
	 int x = a[i-1].X;
	 int y = a[i-1].Y;
	 itor it = lower_bound(b.begin(),b.end(),pii(y,x));
	 for(int j = i+1;j-1<a.size()&&it != b.end()&&a[j-1].X == x &&it->X == y;)
	 {
		 int xx= it->Y;
		 int yy = a[j-1].Y;
		 if(xx - x > yy - y)
			 j++;
		 else if(xx-x<yy-y)
			 it++;
		 else
		 {
			 j++;it++;
			 itor ni = lower_bound(a.begin(),a.end(),pii(xx,yy));
			 if(ni == a.end())
				 continue;
			 if(ni->X != xx || ni->Y != yy)
		        continue;
			ans++;
		  }
	 }
	}
	printf("%d\n",ans);
	//fclose(stdin);fclose(stdout);
	return 0;
}
