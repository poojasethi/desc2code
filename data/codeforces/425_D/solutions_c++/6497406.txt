#include<stdio.h>
#include<algorithm>
#include<vector>
#define mp(x,y) make_pair(x,y)
#define REP(i,n) for(int i=0;i<(int)(n);++i)
using namespace std;

inline int scan(){char c;int r;while(c=getchar())if('0'<=c&&c<='9')break;r=c-'0';while(c=getchar()){if(c<'0'||c>'9')break;r*=10;r+=c-'0';}return r;}
typedef pair<int,int> pi;
int n;

int main(){
    n=scan();
    vector<pi> u(n),v(n);
    for(int i=0;i<n;++i){
        u[i].first=v[i].second=scan();
        u[i].second=v[i].first=scan();
    }
    sort(u.begin(),u.end());
    sort(v.begin(),v.end());
    int ans=0;
    REP(i,n){
        int x=u[i].first,y=u[i].second;
        vector<pi>::iterator it=lower_bound(v.begin(),v.end(),mp(y,x));
        int j=i+1;
        while(j<n && u[j].first==x && it!=v.end() && it->first==y){
            int X=it->second,Y=u[j].second;
            if(X-x>Y-y)++j;
            else if(X-x<Y-y)++it;
            else{
                ++j;++it;
                if(binary_search(u.begin(),u.end(),mp(X,Y)))ans++;
            }
        }
    }
    printf("%d",ans);
    return 0;
}
