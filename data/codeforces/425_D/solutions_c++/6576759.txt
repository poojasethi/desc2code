#include <bits/stdc++.h>

#define FI(ii,aa,bb) for(int ii=aa;ii<=bb;ii++)
#define F(ii,aa,bb) for(int ii=aa;ii<bb;ii++)
#define TF(ii,aa,bb) for(int ii=aa;ii>=bb;ii--)
#define pb push_back
#define pdd pair<double,double>
#define ll long long
#define inf (INT_MAX)

using namespace std;

int n;
vector<int> X[100005],Y[100005];

int main()
{
	int x,y,sz,k,l,sz2,ans=0;
	scanf("%d",&n);
	FI(i,1,n){
		scanf("%d %d",&x,&y);
		X[x].pb(y); Y[y].pb(x);
	}
	FI(i,0,100000) sort(X[i].begin(),X[i].end()),sort(Y[i].begin(),Y[i].end());
	FI(i,0,100000){
		sz=X[i].size();
		F(j,0,sz){
			x=i,y=X[i][j],l=j+1,sz2=Y[y].size();
			k=upper_bound(Y[y].begin(),Y[y].end(),x)-Y[y].begin();
			while(l<sz && k<sz2){
				if(X[i][l]-y>Y[y][k]-x) k++;
				else if(X[i][l]-y<Y[y][k]-x) l++;
				else{
					ans+=binary_search(X[Y[y][k]].begin(),X[Y[y][k]].end(),X[i][l]);
					k++; l++;
				}
			}
		}
	}
	cout << ans << endl;
	return 0;
}
