#include<cstdio>
#include<algorithm>
#include<cstring>
#include<map>
#include<set>
using namespace std;

const int N=100050;
set<int>s[N];
set<int>::iterator it1,it2,it3;

bool ask(int x,int y){
  if(x<0) return false;if(x>100000)return false;
  if(y<0) return false;if(y>100000)return false;
  if(s[x].size()==0)return false;
  it3=s[x].lower_bound(y);
  if(it3==s[x].end())return false;
  return *it3==y;
}

int main(){
   int n;scanf("%d",&n);
   int M=0;
   for(int i=1;i<=n;i++){
     int x,y;
     scanf("%d%d",&x,&y);
     s[x].insert(y);
     if(x>M)M=x;
   }
   int ans=0;
   for(int i=0;i<=M;i++){  
     if(s[i].size()<=330){                
       for(it1=s[i].begin();it1!=s[i].end();it1++){
         for(it2=s[i].begin();it2!=it1;it2++){
           int y1=*it2,y2=*it1;
           if(ask(i+y2-y1,y1)&&ask(i+y2-y1,y2))ans++;
         }
       }  
     }
     else{
       for(int j=i+1;j<=M;j++)
         for(it1=s[j].begin();it1!=s[j].end();it1++){
           int y1=*it1;
           if(ask(i,y1)&&ask(i,y1+j-i)&&ask(j,y1+j-i))ans++;
         }
     }
   }
   printf("%d\n",ans);
}
