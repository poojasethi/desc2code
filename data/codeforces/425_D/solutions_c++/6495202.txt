#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <set>

using namespace std;

struct point{
	int x, y, dif;
};


struct point a[111111];

int cmp(point p1, point p2)
{
	if (p1.dif > p2.dif)
		return 1;
	return 0;
}

set <int> wasX[111111];
set <int> wasY[111111];
set <int> wasD[222222];


int main()
{
//	freopen("file.txt", "r", stdin);
	int n;

	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> a[i].x >> a[i].y;
		a[i].dif = a[i].x - a[i].y;
	}

	sort(a, a+n, cmp);
	
	int ans = 0;
	for (int i = 0; i < n; i++)
	{
		int d = a[i].x + a[i].y;
		for (set<int>::iterator it = wasD[d].begin(); it != wasD[d].end(); it++)
		{
			int x1 = *it;
			int y1 = d - x1;
			if (wasX[x1].find(a[i].y) != wasX[x1].end() && 
				wasY[y1].find(a[i].x) != wasY[y1].end())
				ans++;//, cout << x1 << '#' << y1 << endl;
		}
		wasX[a[i].x].insert(a[i].y);
		wasY[a[i].y].insert(a[i].x);
		wasD[a[i].x+a[i].y].insert(a[i].x);
		
//		cout << a[i].x << ' ' << a[i].y << endl;
	}
	cout << ans << endl;
	return 0;
}
