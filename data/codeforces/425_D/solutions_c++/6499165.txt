#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>

using namespace std;

const int maxn = 100001;

vector<int>  vx[maxn];
int x[maxn],y[maxn],cnt[maxn];
int main()
{
    int n,a,b,res=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d%d",x+i,y+i);
        cnt[y[i]]++;
    }
    for(int i=0;i<n;i++)
    {
        if(cnt[y[i]]>=2)
            vx[x[i]].push_back(y[i]);
    }
    for(int i=0;i<maxn;i++)
        sort(vx[i].begin(),vx[i].end());
    for(int i=0;i<maxn;i++)
    {
        for(int j=0;j<vx[i].size();j++)
        {
            for(int k=j+1;k<vx[i].size();k++)
            {
                int px = i + vx[i][k]-vx[i][j];
                if(px<maxn)
                {
                    if(binary_search(vx[px].begin(),vx[px].end(),vx[i][j])&&binary_search(vx[px].begin(),vx[px].end(),vx[i][k]))
                        res++;
                }
                else
                    break;
            }
        }
    }
    printf("%d\n",res);
}
