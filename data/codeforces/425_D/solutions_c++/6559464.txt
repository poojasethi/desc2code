#define rep(i,n) for(i=1; i<=n; i++)
#define pb push_back
using namespace std;
vector <int>xs[100005];
int i,n,a[100005],b[100005],j,k,ans;
int Find(int x,int y)
{
    if (x>100000) return 0; else
      return binary_search(xs[x].begin(),xs[x].end(),y);
}
int main()
{
    scanf("%d",&n);
    for (i=1; i<=n; i++) {scanf("%d%d",&a[i],&b[i]); xs[a[i]].pb(b[i]);}
    for (i=0; i<=100000; i++) sort(xs[i].begin(),xs[i].end());
    for (i=0; i<=100000; i++)
      if (xs[i].size()<=330)
      {
          for (j=0; j<xs[i].size(); j++)
            for (k=j+1; k<xs[i].size(); k++)
              if (Find(i+abs(xs[i][j]-xs[i][k]),xs[i][j]) && Find(i+abs(xs[i][j]-xs[i][k]),xs[i][k])) ans++;
      } else
        for (j=i+1; j<=100000; j++)
          for (k=0; k<xs[j].size(); k++)
            if (Find(i,xs[j][k]) && Find(i,xs[j][k]+j-i) && Find(j,xs[j][k]+j-i)) ans++;
    cout<<ans;
    return 0;
}
