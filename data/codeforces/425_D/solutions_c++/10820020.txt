#include<cstdio>
#include<vector>
#include<algorithm>
#define all(_v) _v.begin(),_v.end()
const int N=100000;
std::vector<int> X[N+5],Y[N+5];
int n,ans;

int main()
{
    scanf("%d",&n);
    for(int i=1,x,y;i<=n;i++)
    {
        scanf("%d%d",&x,&y);
        X[x].push_back(y);
        Y[y].push_back(x);
    }
    for(int i=0;i<=N;i++)
        sort(all(X[i])),
        sort(all(Y[i]));
    for(int i=0,j,x,y,px,py;i<=N;i++)
        for(j=0;j<X[i].size();j++)
        {
            x=i,y=X[i][j];
            px=j+1,py=upper_bound(all(Y[y]),x)-Y[y].begin();
            while(px<X[x].size()&&py<Y[y].size())
            {
                if(X[x][px]-y<Y[y][py]-x){px++;continue;}
                if(X[x][px]-y>Y[y][py]-x){py++;continue;}
                ans+=binary_search(all(X[Y[y][py]]),X[x][px]);
                px++,py++;
            }
        }
    printf("%d\n",ans);
    return 0;
}
