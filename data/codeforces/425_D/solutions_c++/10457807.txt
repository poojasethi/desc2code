#include <bits/stdc++.h>
#define fi first
#define se second
#define endl '\n'

using namespace std;

typedef long long lint;
typedef pair<int,int> ii;
const int MAXN = 2e5+222;
const int mod = 1e9+7;

vector<int> vx[MAXN],vy[MAXN];
map<int,int> mp[MAXN];
ii ar[MAXN];
int N;

int main(){
	
	scanf(" %d",&N);
	for(int i=1;i<=N;i++){
		scanf(" %d %d",&ar[i].fi,&ar[i].se);
		vx[ar[i].fi].push_back(ar[i].se);
		vy[ar[i].se].push_back(ar[i].fi);
		mp[ar[i].fi][ar[i].se]=1;
	}
	
	int ans=0;
	for(int i=1;i<=N;i++){
		int x=ar[i].fi;
		int y=ar[i].se;
		if(vx[x].size()<vy[y].size()){
			for(int j=0;j<(int)vx[x].size();j++)
				if(vx[x][j] > y && mp[x+vx[x][j]-y].count(y) && mp[x+vx[x][j]-y].count(vx[x][j]))
					ans++;
		}
		else{
			for(int j=0;j<(int)vy[y].size();j++)
				if(vy[y][j] > x && mp[x].count(y+vy[y][j]-x) && mp[vy[y][j]].count(y+vy[y][j]-x))
					ans++;
		}
	}
	
	printf("%d\n",ans);
	
	return 0;
}
