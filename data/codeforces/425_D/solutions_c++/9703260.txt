#include <cstdio>
#include <cstring>
#include <string>
#include <map>
#include <vector>
#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;
vector<int> V[112345];
map<int,bool> mp[112345];
int main()
{
	int n;

	cin>>n;
	for(int i=0;i<n;i++){
		int x,y;
		scanf("%d%d",&x,&y);
		V[x].push_back(y);
		mp[x][y]=1;
	}
	for(int i=0;i<112345;i++)sort(V[i].begin(),V[i].end());
	int sq=sqrt(100000)+2,ans=0;
	for(int i=0;i<=100000;i++){
		if(V[i].size()<sq){
			for(int k=0;k<V[i].size();k++){
				for(int kk=k+1;kk<V[i].size();kk++){
					int len=V[i][kk]-V[i][k];
					if(i+len<=100000&&mp[i+len][V[i][k]]&&mp[i+len][V[i][kk]])
						ans++;
				}
			}
		}
		else{
			for(int k=i+1;k<=100000;k++){
				for(int kk=0;kk<V[k].size();kk++){
					int len=k-i;
					if(V[k][kk]+len<=100000&&mp[i][V[k][kk]]&&mp[i][V[k][kk]+len]&&mp[k][V[k][kk]+len])
						ans++;
				}
			}
		}
	}
	printf("%d\n",ans);
	return 0;
}
