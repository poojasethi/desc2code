#!/usr/bin/env python

from sys import stdin

s = stdin.readline().strip()
rook_pos = ( ord(s[1])-ord('1'), ord(s[0])-ord('a') )

s = stdin.readline().strip()
knight_pos = ( ord(s[1])-ord('1'), ord(s[0])-ord('a') )

knight_moves = [ (-2, -1), (-2, +1), (-1, -2), (-1, +2), (+1, -2), (+1, +2), (+2, -1), (+2, +1)  ]

res = 0
for r in xrange(8):
   if r == rook_pos[0]: continue
   for c in xrange(8):
      if c == rook_pos[1]: continue
      if (r, c) == rook_pos or (r, c) == knight_pos: continue
      for d in knight_moves:
         pos = (r + d[0], c + d[1])
         if pos[0] < 0 or pos[0] >= 8 or pos[1] < 0 or pos[1] >= 8:
            continue
         if pos == rook_pos or pos == knight_pos:
            break
      else:
         res += 1

print res
