def read():
    s=raw_input()
    return ord(s[0])-97,int(s[1])-1
def canK(a,b):
    return sorted([abs(a[0]-b[0]),abs(a[1]-b[1])])==[1,2]
def canR(a,b):
    return min(abs(a[0]-b[0]),abs(a[1]-b[1]))==0

a=read()
b=read()
ans=0
for c in ((i,j) for i in xrange(8) for j in xrange(8)):
    if c!=a and c!=b and not canR(a,b) and not canR(a,c)\
        and not canK(b,a) and not canK(b,c)\
        and not canK(c,a) and not canK(c,b):
        ans+=1
print ans
