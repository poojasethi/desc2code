a=raw_input()
b=raw_input()
a=ord(a[0])-ord('a')+1,int(a[1])
b=ord(b[0])-ord('a')+1,int(b[1])
n=0
for x in xrange(1,9):
	for y in xrange(1,9):
		if (x,y)==b or x==a[0] or y==a[1]:
			continue
		for dx in (-2,-1,1,2):
			for dy in (3-abs(dx),abs(dx)-3):
				if (x+dx,y+dy) in (a,b) or (b[0]+dx,b[1]+dy)==(x,y):
					break
			else:
				continue
			break
		else:
			n+=1
print n
