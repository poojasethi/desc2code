#!/usr/bin/env python
chars = ["a", "b", "c", "d", "e", "f", "g", "h"]
l1 = raw_input()
l1x = l1[0]
l1y = int(l1[1]) - 1

k2 = raw_input()
k2x = k2[0]
k2y = int(k2[1]) - 1

for i in range(len(chars)):
    if chars[i] == l1x:
        l1x = i
    if chars[i] == k2x:
        k2x = i
        
m = []

for i in range(8):
    m.append([])
    for j in range(8):
        m[i].append(0)

for i in range(8):
    m[l1x][i] = 1

for i in range(8):
    m[i][l1y] = 1

m[l1x][l1y] = 2    
m[k2x][k2y] = 2

if k2x+1<8 and k2y+2<8:
    m[k2x+1][k2y+2] = 1

if k2x+1<8 and k2y-2>=0:
    m[k2x+1][k2y-2] = 1
    
if k2x-1>=0 and k2y+2<8:
    m[k2x-1][k2y+2] = 1
    
if k2x-1>=0 and k2y-2>=0:    
    m[k2x-1][k2y-2] = 1

if k2x+2<8 and k2y+1<8:
    m[k2x+2][k2y+1] = 1
    
if k2x+2<8 and k2y-1>=0:
    m[k2x+2][k2y-1] = 1
    
if k2x-2>=0 and k2y+1<8:
    m[k2x-2][k2y+1] = 1
    
if k2x-2>=0 and k2y-1>=0:
    m[k2x-2][k2y-1] = 1

o=0

for i in range(8):
    for j in range(8):
        if m[i][j] == 0:
            
            b = 0
            
            if i+1<8 and j+2<8 and m[i+1][j+2] == 2:
                b = 1
            
            if i+1<8 and j-2>=0 and m[i+1][j-2] == 2:
                b = 1
                
            if i-1>=0 and j+2<8 and m[i-1][j+2] == 2:
                b = 1
                
            if i-1>=0 and j-2>=0 and m[i-1][j-2] == 2:
                b = 1
            
            if i+2<8 and j+1<8 and m[i+2][j+1] == 2:
                b = 1
                
            if i+2<8 and j-1>=0 and m[i+2][j-1] == 2:
                b = 1
                
            if i-2>=0 and j+1<8 and m[i-2][j+1] == 2:
                b = 1
                
            if i-2>=0 and j-1>=0 and m[i-2][j-1] == 2:
                b = 1
                
            if b == 0:
                o += 1   
       
print(o)
