#!/usr/bin/env python

def check(x, y):
	if x >= 0 and x <= 7 and y >=0 and y <= 7:
		return True
	else:
		return False

ld = raw_input()
kn = raw_input()

desk = [0] * 8
for i in range(8):
	desk[i] = [0] * 8

ld_x = ord(ld[0]) - ord('a')
ld_y = int(ld[1]) - 1

kn_x = ord(kn[0]) - ord('a')
kn_y = int(kn[1]) - 1

for i in range(8):
	desk[ld_x][i] = 1
	desk[i][ld_y] = 1

if check(kn_x + 2, kn_y + 1): desk[kn_x + 2][kn_y + 1] = 1
if check(kn_x - 2, kn_y - 1): desk[kn_x - 2][kn_y - 1] = 1
if check(kn_x + 2, kn_y - 1): desk[kn_x + 2][kn_y - 1] = 1
if check(kn_x - 2, kn_y + 1): desk[kn_x - 2][kn_y + 1] = 1

if check(kn_x + 1, kn_y + 2): desk[kn_x + 1][kn_y + 2] = 1
if check(kn_x - 1, kn_y - 2): desk[kn_x - 1][kn_y - 2] = 1
if check(kn_x + 1, kn_y - 2): desk[kn_x + 1][kn_y - 2] = 1
if check(kn_x - 1, kn_y + 2): desk[kn_x - 1][kn_y + 2] = 1

desk[ld_x][ld_y] = 2
desk[kn_x][kn_y] = 2

r = 0
for i in range(8):
	for j in range(8):
		if desk[i][j] == 0:
			if not check(i + 2, j + 1) or desk[i + 2][j + 1] != 2:
				if not check(i - 2, j - 1) or desk[i - 2][j - 1] != 2:
					if not check(i + 2, j - 1) or desk[i + 2][j - 1] != 2:
						if not check(i - 2, j + 1) or desk[i - 2][j + 1] != 2:
							if not check(i + 1, j + 2) or desk[i + 1][j + 2] != 2:
								if not check(i - 1, j - 2) or desk[i - 1][j - 2] != 2:
									if not check(i + 1, j - 2) or desk[i + 1][j - 2] != 2:
										if not check(i - 1, j + 2) or desk[i - 1][j + 2] != 2:
											r += 1
							

print r
