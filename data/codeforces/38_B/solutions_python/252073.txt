s1 = raw_input()
s2 = raw_input()

(x1,y1) = ord(s1[0])-ord('a'), ord(s1[1])-ord('1')
(x2,y2) = ord(s2[0])-ord('a'), ord(s2[1])-ord('1')

cnt = 8*8
for x in range(0,8):
    for y in range(0,8):
        if (x == x1 or y == y1 or abs(x-x1)*abs(y-y1) == 2 or abs(x-x2)*abs(y-y2)==2 or (x == x2 and y == y2)):
            cnt -= 1
        
print cnt
