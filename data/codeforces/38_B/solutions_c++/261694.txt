#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
	string r,k;
	cin>>r>>k;
	int rx,ry,kx,ky;
	rx=r[0]-'a';
	ry=r[1]-'1';
	kx=k[0]-'a';
	ky=k[1]-'1';
	
	int answer=0;
	
	for(int i=0;i<8;i++)
		for(int j=0;j<8;j++)
		{
			if(i==rx||j==ry)
				continue;
			else if(abs(i-rx)+abs(j-ry)==3&&i!=rx&&j!=ry)
				continue;
			else if(abs(i-kx)+abs(j-ky)==3&&i!=kx&&j!=ky)
				continue;
			else if(i==kx&&j==ky)
				continue;
			else
				answer++;
		}
		
	cout<<answer<<endl;
	
	return 0;
}
