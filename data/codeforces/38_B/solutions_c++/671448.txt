#include <cstdio>
int main() {
    int c, d, r = 0; char a[5], b[5]; scanf("%s%s", a, b);
    for (int i = 0;i < 8;i++) if (i != a[0] - 'a')
        for (int j = 0;j < 8;j++) if (j != a[1] - '1') {
            c = i - (b[0] - 'a'); d = j - (b[1] - '1');
            if ((c == 0 && d == 0) || (c * c + d * d == 5)) continue;
            c = i - (a[0] - 'a'); d = j - (a[1] - '1');
            if (c * c + d * d == 5) continue;
            r++;
        }
    printf("%d\n", r);
}
