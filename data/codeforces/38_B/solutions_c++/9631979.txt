#include <bits/stdc++.h>

using namespace std;


int ans = 0 ;
int main() {
    int i,j ;
    char a1,b1,a2,b2 ;
    
    cin >> a1 >> b1>>a2>>b2 ;
    a1-='a';
    a2-='a';
    b1-='1';
    b2-='1';
    for(i=0;i<8;i++){
        for(j=0;j<8;j++){
            if(i!=a1&&j!=b1&&(abs(i-a1)*abs(j-b1)!=2)&&(abs(i-a2)*abs(j-b2)!=2)&&!(i==a2&&j==b2))ans++;
        }
        }  
        cout << ans ; 
}
