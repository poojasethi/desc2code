#include<iostream>
#include<cmath>
using  namespace std;

int main()
{
    char c1,c2;
    int x1,x2,y1,y2;
    
    cin>>c1>>y1>>c2>>y2;
    
    x1=c1-'a';
    x2=c2-'a';
    y1--;
    y2--;
    
    int ans=0;
    for(int i=0;i<8;i++)
        for(int j=0;j<8;j++)
        {
            if(i==x1||j==y1) continue;
            if((abs(i-x1)==1&&abs(j-y1)==2)||(abs(i-x1)==2&&abs(j-y1)==1)) continue;
            if((abs(i-x2)==1&&abs(j-y2)==2)||(abs(i-x2)==2&&abs(j-y2)==1))  continue;
            else ans++;
        }
    
    cout<<ans-1<<endl;

}
