#include<iostream>
using namespace std;
int a,b,c,d;
int v[100][100];
long ans(int x,int y){
	if(x<=0||y<=0||x>8||y>8)return 0;
	if(x==a||y==b)return 0;
	if(v[x][y]==1)return 0;
	v[x][y]=1;
	return 1;
}
int main(){
	string s1,s2;
	cin>>s1>>s2;
	a=s1[0]-'a'+1;
	b=s1[1]-'0';
	c=s2[0]-'a'+1;
	d=s2[1]-'0';
	int g=ans(c,d)+ans(c+2,d+1)+ans(c-2,d+1)+ans(c+1,d+2)+ans(c+1,d-2)+ans(c+2,d-1)+ans(c-2,d-1)+ans(c-1,d+2)+ans(c-1,d-2);
	c=a;d=b;
	g=g+ans(c,d)+ans(c+2,d+1)+ans(c-2,d+1)+ans(c+1,d+2)+ans(c+1,d-2)+ans(c+2,d-1)+ans(c-2,d-1)+ans(c-1,d+2)+ans(c-1,d-2);
	cout<<49-g<<endl;
	
}
