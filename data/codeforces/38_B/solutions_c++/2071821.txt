#include <stdio.h>

int main() {
	char a,c;
	int b,d,ans=48,q[8]={-2,-2,-1,-1,1,1,2,2},w[8]={-1,1,-2,2,-2,2,-1,1};
	scanf("%c%d%*c%c%d",&a,&b,&c,&d);
	a-='a'-1;
	c-='a'-1;
	for(int i=0;i<8;++i)
		if(c+q[i]>0&&c+q[i]<=8&&d+w[i]>0&&d+w[i]<=8&&a!=c+q[i]&&b!=d+w[i])
			--ans;
	for(int i=0;i<8;++i)
		if(a+q[i]>0&&a+q[i]<=8&&b+w[i]>0&&b+w[i]<=8){
			int j=0;
			for(;j<8;++j)
				if(a+q[i]==c+q[j]&&b+w[i]==d+w[j]) break;
			if(j==8) --ans;
		}
	printf("%d",ans);
	return 0;
}
