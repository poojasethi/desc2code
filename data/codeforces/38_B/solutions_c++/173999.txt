#include<stdio.h>
#include<iostream>
using namespace std;
int x,y,xx,yy,cnt,i,j;
string str;
int main()
{
    cin>>str;
    x = str[0]-'a'+1;
    y = str[1]-48;
    cin>>str;
    xx = str[0]-'a'+1;
    yy = str[1]-48;
    for(i=1;i<=8;i++)
    {
        for(j=1;j<=8;j++)
        {
            if(i==x || j==y)continue;
            if(i==xx && j==yy)continue;
            if((i-xx)*(i-xx)+(j-yy)*(j-yy)==5)continue;
            if((i-x)*(i-x)+(j-y)*(j-y)==5)continue;
            cnt++;
        }
    }
    cout<<cnt<<endl;
    return 0;
}
