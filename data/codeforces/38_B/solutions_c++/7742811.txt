#include <iostream>
#include <cmath>
using namespace std;
int main()
{
    char x,y,xx,yy;
    cin>>x>>y>>xx>>yy;
    x-=96;
    y-=48;
    xx-=96;
    yy-=48;
    int n=8,sum=0;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            if(i==x || j==y)continue;
            if(i==xx && j==yy)continue;
            if((i-xx)*(i-xx)+(j-yy)*(j-yy)==5)continue;
            if((i-x)*(i-x)+(j-y)*(j-y)==5)continue;
            sum++;;
        }
    }
    cout<<sum;
    return 0;
}
