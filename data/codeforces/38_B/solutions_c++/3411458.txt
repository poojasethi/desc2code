#include<iostream>
#include<cstdio>
#include<cmath>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
	string s1,s2;
	int i,j;
	int sum=0;
	cin>>s1>>s2;
	for(i=0;i<8;i++)
		for(j=0;j<8;j++)
		   if(i!=s1[0]-97&&j!=s1[1]-'1')
			if(abs(s1[0]-97-i)!=1||abs(s1[1]-'1'-j)!=2)
			if(abs(s1[0]-97-i)!=2||abs(s1[1]-'1'-j)!=1)
			if(abs(s2[0]-97-i)!=1||abs(s2[1]-'1'-j)!=2)
		    if(abs(s2[0]-97-i)!=2||abs(s2[1]-'1'-j)!=1) 
				sum++;
		cout<<sum-1<<endl;
	return 0;
}
