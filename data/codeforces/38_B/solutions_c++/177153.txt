#include <cstdio>
#define abs(a) (a > 0 ? a : (-a))

int x1, y1, x2, y2;

int main()
{
//    freopen("B", "r", stdin);
    char i1, j1, i2, j2;
    int ans = 0;
    scanf("%c%c", &i1, &j1);
    char ch;
    scanf("%c", &ch);
    x1 = i1 - 'a' + 1;
    y1 = j1 - '0';
    scanf("%c%c", &i2, &j2);
    x2 = i2 - 'a' + 1;
    y2 = j2 - '0';
    for (int i = 1; i <= 8; i++)
    for (int j = 1; j <= 8; j++)
    if ((i == x1) || (j == y1) || (i == x2 && j == y2) || (abs((i - x1)*(j - y1)) == 2) || (abs((i - x2)*(j - y2)) == 2)) 
        continue;
    else ans++;
    printf("%d", ans);
    return 0;
}
