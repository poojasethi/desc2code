#include<cstdio>
#include<string.h>
int abs(int x){	 return x>0?x:-x;}
int main()
{
	int x1,y1,x2,y2,i,j,sum=0,x,y;
	char c[10];
	scanf("%s",c);
	x1=c[0]-'a'+1;
	y1=c[1]-'0';
	scanf("%s",c);
	x2=c[0]-'a'+1;
	y2=c[1]-'0';
    for(i=1;i<9;i++)
	{
		if(i==x1)continue;
		for(j=1;j<9;j++)
		{
			if(j==y1)continue;
			x=abs(x1-i);y=abs(y1-j);
            if((x==1&&y==2)||(x==2&&y==1))continue;
			x=abs(x2-i);y=abs(y2-j);
            if((x==1&&y==2)||(x==2&&y==1))continue;
            if(i==x2&&j==y2)continue;
            sum++;
		}
	}
	printf("%d\n",sum);
	return 0;
}
