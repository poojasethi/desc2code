#include <iostream>
using namespace std;
#define abs(X) (((X)<0)?-(X):(X))
int main(){
    char K[3],R[3];
    cin>>R>>K;
    int P = 0;
    for (char i = 'a'; i <= 'h'; i++)
        for (char j = '1'; j <= '8'; j++)
            if (i!=R[0] && j!=R[1] && (abs(K[0]-i)!=2 || abs(K[1]-j)!=1) && (abs(K[0]-i)!=1 || abs(K[1]-j)!=2) && (i!=K[0] || j!=K[1]) && (abs(R[0]-i)!=2 || abs(R[1]-j)!=1) && (abs(R[0]-i)!=1 || abs(R[1]-j)!=2))
                P++;
    cout<<P<<endl;
    //system("PAUSE");
    return 0;
}
