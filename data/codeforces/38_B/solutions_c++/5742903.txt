#include<bits/stdc++.h>
using namespace std;
int main()
{
	char fs,sd;
	int fss,sdd;
	cin>>fs>>fss;
	getchar();
	cin>>sd>>sdd;
	int fn=fs-'a'+1;
	int sn=sd-'a'+1;
	int cnt=0;
	for(int i=1;i<=8;i++)
	{
		for(int j=1;j<=8;j++)
		{
			if(fn==i||fss==j) cnt++;
			else if(abs(sdd-j)==1&&abs(sn-i)==2) cnt++;
			else if(abs(sdd-j)==2&&abs(sn-i)==1) cnt++;
			else if(abs(fss-j)==2&&abs(fn-i)==1) cnt++;
			else if(abs(fss-j)==1&&abs(fn-i)==2) cnt++;
		}
	}
	cout<<63-cnt<<endl;
	return 0;
}
