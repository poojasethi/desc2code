#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

char r1, r2;
int c1, c2, ans;

bool check(int r, int c)
{
	if (r == r1 || c == c1) return false;
	if (abs(r - r2) * abs(c - c2) == 2) return false;
	if (abs(r - r1) * abs(c - c1) == 2) return false;
	if (r == r2 && c == c2) return false;
	return true;
}

int main() {
	scanf("%c%d\n%c%d", &r1, &c1, &r2, &c2);

	c1--;
	r1 -= 'a';
	c2--;
	r2 -= 'a';
	
	for (int i = 0; i < 8; ++i)
		for (int j = 0; j < 8; ++j)
			if (check(i, j))
				++ans;

				
	cout << ans << endl;
	return 0;
}
