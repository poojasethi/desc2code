#include <iostream>
using namespace std;
int i,x,y,x1,y1,o=0,j;
char c1,c2,c3,c4;
int abs(int z){
	if(z<0)z*=-1;
	return z;
}
int main(){
    cin>>c1>>c2>>c3>>c4;
    y=c1-97;y1=c3-97;x=c2-49;x1=c4-49;
	for(i=0;i<8;i++){
		for(j=0;j<8;j++){
			if((x-i)*(y-j)!=0 && abs((x-i)*(y-j))!=2 && abs((x1-i)*(y1-j))!=2 && (i!=x1 || j!=y1))o++;
	    }	
	}
	cout<<o;
    return 0;
}
