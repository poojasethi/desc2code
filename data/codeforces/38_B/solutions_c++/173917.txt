#include <iostream>
using namespace std;

int mood(int q)
{
	if(q >= 0)
		return q;
	return -q;
}

int main()
{
	char row='a',q1,q2;
	int col=1,b1,b2,c1,c2,i,j,res=0;
	bool t;

	cin>>q1>>c1>>q2>>c2;
	c1--;
	c2--;
	b1=q1-'a';
	b2=q2-'a';

	for(i=7;i>=0;i--)
	{
		for(j=7;j>=0;j--)
		{
			t=true;	
			if(i == c1)
				t=false;
			if(j == b1)

			if(mood((i-c2)*(j-b2)) == 2)
