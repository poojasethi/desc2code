#include <stdio.h>
int main(){
	char str[10];
	while(~scanf("%s",str)){
		int i,j,sum=0,a1,b1;
		char a,b;
		a=str[0];a1=str[1]-'0';
		scanf("%s",str);
		b=str[0];b1=str[1]-'0';
		for(i=1;i<=8;i++)
			for(j=1;j<=8;j++)
				if( i!=a-'a'+1 && j!=a1
					&&!((i==a-'a' ||i==a-'a'+2)&&(j==a1-2||j==a1+2))
					&&!((i==a-'a'-1 ||i==a-'a'+3)&&(j==a1-1||j==a1+1))

					&&!((i==b-'a'||i==b-'a'+2)&&(j==b1-2||j==b1+2))
					&&!((i==b-'a'-1||i==b-'a'+3)&&(j==b1-1||j==b1+1))
				)
					sum++;
		printf("%d\n",sum-1);
		
	}
	return 0;
}
