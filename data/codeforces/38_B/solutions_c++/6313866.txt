#include<iostream>
using namespace std;
int abs(int a){
    if (a<0) return -a; else return a;
}
int main(){
    string s1,s2;
    int a,b,c,d;
    cin>>s1;
    cin>>s2;
    a = s1[0] - 96;
    b = s1[1] - 48;
    c = s2[0] - 96;
    d = s2[1] - 48;
    int ans=0;
    for (int j=1;j<=8;j++){
        for (int i=1;i<=8;i++){
            if ( (a==i) || (b==j) || abs(i-a)*abs(j-b)==2 || abs(i-c)*abs(j-d)==2 || (i==c &&j==d) ) ans=ans + 0; else ans=ans+1;
        }
    }
    cout<<ans<<endl;    
    return 0;
}
