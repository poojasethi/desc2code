#include <iostream>
#include <math.h>

using namespace std;

int iabs(int a)
{
	return (a<0)?(-a):a;

}

int main()
{
	char a,b;
	cin >> a >> b;;
	int x1 = a - 'a' + 1;
	int y1 = b - '0';
	cin >> a >> b;
	int x2 = a - 'a' + 1;
	int y2 = b - '0';
	int ans = 0;
	for (int i = 1; i <= 8; i++)
		for (int j = 1; j <= 8; j++)
			{
				if (x1==i || y1==j)
					continue;
				if (iabs(x2-i)*iabs(y2-j)==2 || (x2==i && y2==j))
					continue;
				if (iabs(x1-i)*iabs(y1-j)==2 || (x1==i && y1==i))
					continue;
				ans++;
			}
	cout << ans;
	return 0;
}
