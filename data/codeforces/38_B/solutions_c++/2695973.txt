#include <cstdio>
#include <cstring>
#define abs(a) (((a) > 0)? (a) : (-(a)))

int main (void)
{
  int i, j, res = 0;
  char i1, j1, i2, j2;
  scanf (" %c %c %c %c", &i1, &j1, &i2, &j2);
  i1 -= 'a', i2 -= 'a';
  j1 -= '1', j2 -= '1';
  for (i = 0; i < 8; i++)
    for (j = 0; j < 8; j++)
      if ((i == i1) || (j == j1) || ((i == i2) && (j == j2)) ||
        (abs ((i - i2) * (j - j2)) == 2) || (abs ((i - i1) * (j - j1)) == 2))
          ;
      else
        res++;
  printf ("%d\n", res);
  return 0;
}
