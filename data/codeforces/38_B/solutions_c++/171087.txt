#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cmath>

using namespace std;

int main() {
	int i, j, x1, y1, x2, y2, ans;
	char c;

	cin >> c;
	x1 = c - 'a' + 1;
	cin >> y1;
	cin >> c;
	x2 = c - 'a' + 1;
	cin >> y2;
	ans = 0;
	for (i = 1; i <= 8; ++i)
		for (j = 1; j <= 8; ++j)
			if (i != x1 && j != y1 && (abs(x1 - i) != 2 || abs(y1 - j) != 1) && (abs(x1 - i) != 1 || abs(y1 - j) != 2) && (abs(x2 - i) != 2 || abs(y2 - j) != 1) && (abs(x2 - i) != 1 || abs(y2 - j) != 2) && (i != x2 || j != y2) )
				++ans;
	cout << ans << endl;

	return 0;
}
