#define X first

const int inf = ~0U >> 1;
map<pii, int> mp;
const int N=4e5+1;
int a[N],b[N];
int main() {
  //  freopen("c:\\data.txt", "r", stdin);
    int n,m;cin>>n>>m;
    rep(i,n) {
        int x,y;cin>>x>>y;
        if(x>y) swap(x,y);
        ++mp[pii(x,y)];
        a[x]++;b[x]++;
        a[y]++;b[y]++;
    }
    sort(a+1,a+n+1);
    ll ans=0;
    for(int i=n;i>=1;i--) {
        int x=lower_bound(a+1,a+i,m-a[i])-a;
        ans+=i-x;
       //cout<<ans<<endl;
    }
    for(map<pii,int>::iterator it=mp.begin();it!=mp.end();it++) {
        int temp=b[it->X.X]+b[it->X.Y];
        if(temp>=m&&(temp-(it->Y)<m)) ans--;
    }
    cout<<ans<<endl;
}
