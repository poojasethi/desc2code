#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(false);
    long long n,p;
    cin>>n>>p;
    map<pair<long long,long long>,long long>m;
    long long arr[n+1];
    fill(arr,arr+n+1,0);
    for(long long i=0;i<n;i++)
    {
        long long a,b;
        cin>>a>>b;
        if(a>b)
            swap(a,b);
        arr[a]++;
        arr[b]++;
        m[make_pair(a,b)]++;
    }
    long long ans=0;
    map<pair<long long,long long>,long long>::iterator it;
    for(it=m.begin();it!=m.end();it++)
    {
        long long a=(it->first).first;
        long long b=(it->first).second;
        long long c=it->second;
        if(((arr[a]+arr[b])>=p)&&((arr[a]+arr[b]-c)<p))
            ans--;
    }
    sort(arr+1,arr+n+1);
    for(long long i=n;i>0;i--)
    {
        long long ind=lower_bound(arr+1, arr+n+1, p-arr[i])-arr;
        if(ind<i)
            ans += i-ind;
    }
    cout<<ans<<"\n";
}
