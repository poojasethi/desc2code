#include <stdio.h>
#include <iostream>
#include <string>
#include <math.h>
 
using namespace std;
 
string f(string s)
{
    string res = "";
    for (int i=0; i<s.size(); i++)
        if (s[i] == 'H')
            res += 'H';
        else
            res += s;
    return res;
}
 
int main()
{
    int h = 0, q = 0;
    string s;
    cin >> s;
    for (int i=0; i<s.size(); i++)
    {
        h += s[i] == 'H';
        q += s[i] == 'Q';
    }
    
    if (q == 0)
    {
        cout << "Yes";
        return 0;
    }
    double qq = sqrt((double)q);
    q = int(qq);
    if (double(q) - qq != 0.0 || h % (q+1) != 0)
    {
        cout << "No";
        return 0;
    }

    h /= q+1;
    int i=0;
    string res = "";

    for(; s[i]!='Q'; i++);
    if (i & 1)
    {
        cout << "No";
        return 0;
    }
    
    res = s.substr(i/2, h+q);   
    if (s == f(res))
        cout << "Yes";
    else
        cout << "No";
    return 0;
}
