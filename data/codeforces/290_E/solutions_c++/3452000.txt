#include <iostream>
#include <stdio.h>
#include <cstring>
#include <cmath>

using namespace std;
char str[1000005],my[1000005];
int main()
{
    //cout <<int(4.9)<<endl;
    while (scanf("%s",str)!=EOF){
        bool got=true;
       int l=strlen(str);
       int h=0,q=0,h0=0;
       for (int i=0;i<l;i++){
         if (str[i]=='H') h++; else q++;
       }
       int qq=sqrt(double(q));
       if (qq*qq!=q) {printf("No\n");continue;}
       q=qq;
       if (h%(q+1)!=0) {printf("No\n");continue;}
       if (q==0) {printf("Yes\n");continue;}
       h=h/(q+1);
       for (h0=0;h0<l;h0++){
         if (str[h0]=='Q') break;
       }
       if (h0%2==1) {printf("No\n");continue;}
       h=h-h0/2;q=q-1; int t=h0+1;
      while (h>0||q>0){
          if (t>=l) {printf("No\n");got=false;break;}
          if (h<0||q<0) {printf("No\n");got=false;break;}
          if (str[t]=='H') h--;else q--;
          t++;
      }
      if (!got) continue;
      if (h<0||q<0) {printf("No\n");continue;}
      if (h==0&&q==0){
          h0=h0/2;
         for (int i=h0;i<t;i++){
           my[i-h0]=str[i];
         }
        int  myl=t-h0; int now=0;
         for (int i=0;i<myl;i++){
           if (my[i]=='H') {
             if (str[now]=='H') now++;
             else {printf("No\n");got=false;break;}

           }
           else {
              for (int j=0;j<myl;j++){
                 if (str[now]!=my[j]) {printf("No\n");got=false;break;}
                 now++;
              }
              if(!got) break;
           }
         }
         if (got) {printf("Yes\n");}

      }
    }
    return 0;
}
