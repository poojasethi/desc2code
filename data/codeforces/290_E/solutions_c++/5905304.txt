#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>

using namespace std;

int main()
{
    int count1, count2, ct1, ct2, ct3, counter, pos;
    string s, s1, s2;

    cin >> s;

    if (s.find('Q') == -1) {
        cout << "Yes" << endl;
        return 0;
    }
    count1 = count(s.begin(), s.end(), 'Q');
    count2 = s.size() - count1;

    ct1 = sqrt(count1);

    if (ct1 * ct1 != count1) {
        cout << "No" << endl;
        return 0;
    } else {
        ct2 = count2 % (ct1 + 1);

        if (ct2 != 0) {
            cout << "No" << endl;
            return 0;
        } else {
            ct3 = count2 / (ct1 + 1);

            pos = s.find('Q');
            if (pos == -1) {
                cout << "No" << endl;
                return 0;
            } else {
                pos /= 2;
                s1 = s.substr(pos, ct3+ct1);

                for (int i = 0; i < s1.size(); i++) {
                    if (s1[i] == 'H') {
                        s2 += "H";
                    } else {
                        s2 += s1;
                    }
                }
                if (s2 == s) {
                    cout <<  "Yes" << endl;
                } else {
                    cout << "No" << endl;
                }
            }

        }
    }
    return 0;
}
