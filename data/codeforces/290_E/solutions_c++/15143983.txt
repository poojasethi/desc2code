#include <iostream>
#include <fstream>
#include <algorithm>
#include <cmath>
using namespace std;

 main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	string s;
	cin >> s;
	size_t fq = s.find("Q");
	if (fq == string::npos) {
		cout << "Yes";
		return 0;
	}
	fq/=2;
	int noq = count(s.begin(), s.end(), 'Q');
	int nq = sqrt(noq);
	int nh = (s.size()-noq) / (nq+1);

	string a = s.substr(fq, nh+nq);
	string t;
	for (int i=0; i<int(a.size()); ++i) {
		if (a[i] == 'H') t += "H";
		else t += a;
	}
	cout << (s == t ? "Yes" : "No");;
}
