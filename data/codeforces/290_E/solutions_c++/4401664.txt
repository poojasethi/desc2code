#include <cstdio>

char nap[1000005];
char kod[1000005];

int n,m,o;

int main()
{
	scanf("%s",nap);
	while(nap[++n]);
	int q=0;
	for(int i=0;i<n;i++) if(nap[i]=='Q') q++;
	for(int i=0;i<=q;i++)
	{
		if(i*i==q) { q=i; break; }
		if(i*i>q) { printf("No\n"); return 0; }
	}
	if(q==0) { printf("Yes\n"); return 0; }
	int hp=0,hk=0;
	for(int i=0;nap[i]!='Q';i++) hp++;
	for(int i=n-1;nap[i]!='Q';i--) hk++;
	if((hp%2!=0)||(hk%2!=0)) { printf("No\n"); return 0;}
	hp/=2; hk/=2;
	while(hp--) kod[m++]='H';
	bool dod=false;
	for(int i=0;i<n;i++)
	{
		if(nap[i]=='Q' && q--) dod=true;
		if(dod) kod[m++]=nap[i];
		if(q==0) break;
	}
	while(hk--) kod[m++]='H';
	for(int i=0;i<m;i++)
	{
		if(kod[i]=='H') { if(nap[o++]!='H') { printf("No\n"); return 0; } }
		else { for(int j=0;j<m;j++) if(nap[o++]!=kod[j]) { printf("No\n"); return 0; } }
	}
	if(o==n) printf("Yes\n");
	else printf("No\n");
	return 0;
}
