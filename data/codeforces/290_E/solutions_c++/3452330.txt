#include <cstdio>

int TUX, FOO, BAR, BAZ, QUZ;

int main() 
{
	string s;
	cin >> s;
	int q1 = 0, h1 = 0;
	for (int i = 0; i < s.length(); ++i)
		if (s[i] == 'Q')
			++q1;
		else
			++h1;
	int q = sqrt(double(q1)), h;
	if (q*q != q1)
		cout << "No";
	else
	{
		h = h1/(q + 1);
		if (h*(q + 1) != h1)
			cout << "No";
		else
		{
			if (q == 0)
				cout << "Yes";
			else
			{
				int l = h + q;
				int ii = 0;
				while (s[ii] == 'H')
					++ii;
				string ss = s.substr(ii/2, l), s1;
				for (int i = 0; i < l; ++i)
					if (ss[i] == 'H')
						s1 += 'H';
					else
						s1 += ss; 
				if (s1 == s)
					cout << "Yes";
				else
					cout << "No";
			}

		}
	}
	return 0;
}
