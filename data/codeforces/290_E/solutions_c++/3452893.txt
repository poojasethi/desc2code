#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
string e(string HQ){
    string N;
    for(int i=0;i<HQ.length();i++){
        if (HQ[i]=='H') N+='H';
        else N+=HQ;
    }return N;
}
bool valid(string HQ){
    int h = 0, q = 0, a, b;
    for (int i=0; i<HQ.length();i++){
        if (HQ[i] == 'H') h++;
        else q ++;
    }
    if (q == 0) return true;
    for (b=0;b*b<q;b++);
    if (b * b != q) return false;
    for (a=0;a*b+a<h;a++);
    if (a * b + a != h) return false;
    int len = a + b;
    for (a=0;HQ[a]!='Q';a++);
    if (a&1) return false;
    string x = HQ.substr(a/2, len);
    return e(x) == HQ;
}
int main(){
    string s;
    cin>>s;
    puts(valid(s)?"Yes":"No");
    return 0;
}
