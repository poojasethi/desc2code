int a=0,b=0;
string s,s1,s2;
int main(){
    ios::sync_with_stdio(0);
    cin>>s;
    bool jud=1;
    for(int i=0;s[i];i++){
        if(s[i]=='H') a++;
        else b++;
    }
    int q=sqrt(b);
    int len=s.size();
    if(q*q!=b) jud=0;
    if(a%(q+1)) jud=0;
    if(jud){
        int loc=0,loc2=0;
        for(;s[loc]=='H';loc++);
        if(s[loc]){
            for(;s[len-loc2-1]=='H';loc2++);
            if(loc%2||loc2%2) {
                jud=0;
            }
            for(int i=0;i<loc/2;i++){
                s1+='H';
            }
            int cnt=0;
            for(;cnt!=q&&s[loc];loc++){
                if(s[loc]=='Q') cnt++;
                s1+=s[loc];
            }
            for(int i=0;i<loc2/2;i++){
                s1+='H';
            }
            for(int i=0;i<s1.size();i++){
                if(s1[i]=='H') s2+='H';
                else s2+=s1;
            }
            if(s2!=s) jud=0;
        }
    }
    puts(jud?"Yes":"No");
    return 0;
}
