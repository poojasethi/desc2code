#include <cstdio>
char s[2020202];
char t[2020202];
int l, q, n, i;
typedef long long int ll;

int main(){
  scanf("%s", s);
  for (n = 0; s[n]; n++);
  for (i = 0; s[i]; i++) if (s[i] == 'Q') break; 
  if (!s[i]){printf("Yes"); return 0;}
  if (i & 1){printf("No"); return 0;}
  q = 0;
  for (i >>= 1, i--, l = 1; s[i + l]; l++){
    q += (s[i + l] == 'Q' ? 1 : 0);
    if (ll(q) * ll(l) + l - q > n) {printf("No"); return 0;}
    if (ll(q) * ll(l) + l - q == n) break;
  }
  q = 0;
  for (int j = 1; j <= l; j++)
    if (s[i + j] == 'Q'){
      for (int w = 1; w <= l; w++)
    t[++q] = s[i + w];
    } else t[++q] = 'H';
  for (int i = 1; i <= n; i++) if (t[i] != s[i - 1]){printf("No"); return 0;}
  printf("Yes");
  return 0;
}
