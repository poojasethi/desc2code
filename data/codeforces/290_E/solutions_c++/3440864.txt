#include<stdio.h>
#include<string.h>
#include<math.h>
int l,h,q,n,nh,nq;
char s[1111111];
bool chk(int st,int en){
	int lt=0;
	for(int i=st; i<en; i++)
		if(s[i]=='H'){
			if(s[lt]!='H')return false;
			lt++;
		}else{
			if(memcmp(s+lt,s+st,sizeof(char)*(en-st))!=0)
				return false;
			lt+=(en-st);
		}
	return true;
}
int main(){
	scanf("%s",s);
	l=strlen(s);
	h=0,q=0;
	for(int i=0; i<l; i++)
		if(s[i]=='H')
			h++;
		else
			q++;
	nq=(int)(sqrt(q+.5));
	if(nq*nq!=q){puts("No");return 0;}
	nh=h/(nq+1);
	if(nh*(nq+1)!=h){puts("No");return 0;}
	n=nq+nh;
	
	h=q=0;
	for(int i=0; i<n; i++)
		if(s[i]=='H')
			h++;
		else
			q++;

	for(int i=n-1;;){
		if(h==nh && q==nq && chk(i-n+1,i+1)){
			puts("Yes");
			return 0;
		}
		i++;
		if(i==l)break;
		if(s[i-n]=='H')h--;else q--;
		if(s[i]=='H')h++;else q++;
	}

	puts("No");
	return 0;
}
