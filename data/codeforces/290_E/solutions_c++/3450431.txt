string F(string a) {
    string ans = "" ;
    for (int i = 0; i < Size(a) ; ++i)
        ans += ((a[i]=='H')?"H":a) ;
    return ans ;
}

int main() {
    ios_base::sync_with_stdio(false) ;
    cin >> s ;
    int q = 0 , h = 0 ;
    for (int i = 0 ; i < Size(s) ; ++i)
        if (s[i] == 'Q')
            ++q ;
        else
            ++h ;
    int d = sqrt(q) ;
    if (d*d != q) no() ;
    q = d ;
    h /= q+1 ;
    int l = 0 ;
    for ( ; l < Size(s) && s[l] == 'H' ; ++l) ;
    if (l == Size(s)) yes() ;
    if (l%2) no() ;
    string ans = "" ;
    for (int i = l/2 , j = 0 ; j < q+h ; ++j , ++i)
        ans += s[i] ;
    if (F(ans) == s) yes() ;
    else no() ;
    return 0 ;
}
