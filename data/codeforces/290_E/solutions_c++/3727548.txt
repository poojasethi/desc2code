#include <iostream>

#include <string>

#include <algorithm>
#define MAXLEN 1000005
#define sqr(x)   ( (x) * (x)  )
using namespace std;

char st[MAXLEN], ori[MAXLEN];

bool judge(void)
{
    int i,j,stlen=strlen(st), qcnt, nq, nh, nqh;
    char *ps0,*ps1;
    ps0=strchr(st, 'Q');
    if(ps0==NULL) return true;
    ps1=strrchr(st, 'Q');
    if( (ps0-st)%2==1 || (st+stlen-1-ps1)%2==1  ) return false;
    for(i=qcnt=0;i<stlen;i++) if(st[i]=='Q') qcnt++;
    if( sqr((int)sqrt(qcnt))!=qcnt  ) return false;
    nq=(int)sqrt(qcnt);
    if((stlen - sqr(nq)) % (nq+1)!=0) return false;
    nh=(stlen - sqr(nq)) / (nq+1);
    nqh=nq+nh;
    strncpy( ori, st + (ps0-st)/2, nqh );
    ori[nqh]='\0';
    for(i=j=0;i<nqh;i++)
    if(ori[i]=='H')
    {
        if(st[j]!='H') return false;
        j++;
    }
    else
    {
        if(strncmp( ori, st+j, nqh )!=0) return false;
        j+=nqh;
    }
    return true;
}

int main()
{
    /*freopen("d:\\1.in","r",stdin);
    freopen("d:\\1.out","w",stdout);*/
    scanf("%s", st);
    if(judge()) printf("Yes\n");
            else  printf("No\n");
    return 0;
}
