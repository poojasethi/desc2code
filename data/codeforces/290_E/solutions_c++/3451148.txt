     string temp;

     int len=temp.size();
     int h=0;

        if(temp[i]=='H')
           h++;
        else q++;
     if((int)sqrt(q)!=sqrt(q))
        puts("No");
     else if(h%((int)sqrt(q)+1)!=0)
        puts("No");
     else
     {
         int cnt=0;
         for(int i=0;i<len;i++)
           if(temp[i]=='Q')
             break;
           else cnt++;
         if(cnt%2==1)
            puts("No");
         else if(cnt==len)
            puts("Yes");
         else
         {
            int H=h/(int)(sqrt(q)+1);
            int Q=(int)sqrt(q);
            int L=H+Q;

               if(origin[i]=='H')
                  out+="H";

         }
     }
