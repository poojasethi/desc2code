#include <iostream>
#include <cstdio>
#include <math.h>
using namespace std;
string s;
string F(string a) 
{
    string ans="";
    for (int i=0; i<a.size(); i++)
        ans+=((a[i]=='H')?"H":a);
    return ans;
}

int main() 
{
    cin >>s;
    int q=0, h=0;
    for (int i=0; i<s.size(); i++)
    {
        if (s[i]=='Q') q++; else h++;
    }
    int d=sqrt(q);
    if (d*d!=q) {cout <<"No" <<endl; return 0;}
    q=d;
    h/=q+1;
    int l=0;
    for (; l<s.size() && s[l]=='H'; l++);
    if (l==s.size()) {cout <<"Yes" <<endl; return 0;}
    if (l%2==1) {cout <<"No" <<endl; return 0;}
    string temp="";
    for (int i=l/2, j=0; j<q+h; ++j, ++i)
    {
        temp+=s[i];
    }
    if (F(temp)==s) cout <<"Yes" <<endl;
    else cout <<"No" <<endl;
    return 0;
}
