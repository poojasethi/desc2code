#include<iostream>
#include<algorithm>
#include<math.h>

using namespace std;

string s,kl,rec;
int i,Q,ansQ,j,ansH;

main()
 {
  cin>>s;
  for (i=0;i<s.size();i++)
   if (s[i] == 'Q') ansQ++;else
   if (s[i] == 'Q') ansH++;
  if (ansQ == 0)
   {
    cout<<"Yes"<<endl;
    return 0;
   }
  if ((int)(sqrt(ansQ))*(int)(sqrt(ansQ)) != ansQ)
   {
    cout<<"No"<<endl;
    return 0;
   }
  Q=(int)(sqrt(ansQ));
  if (ansH % (Q+1) != 0)
   {
    cout<<"No"<<endl;
    return 0;
   }
  for (i=0;i<s.size();i++)
   if (s[i] == 'Q') break;
  if (i % 2 != 0)
   {
    cout<<"No"<<endl;
    return 0;
   }
  for (j=0;j<i/2;j++) kl+='H';
  for (;i<=s.size();i++)
   {
    kl+=s[i];
    if (s[i] == 'Q') Q--;
    if (Q == 0) break;
   }
  for (i=s.size()-1;i>=0;i--)
   if (s[i] == 'Q') break;
  if ((s.size()-1-i) % 2 != 0)
   {
    cout<<"No"<<endl;
    return 0;
   }
  for (j=s.size()-((s.size()-1-i)/2);j<s.size();j++)
   kl+='H';
  for (j=0;j<kl.size();j++)
   if (kl[j] == 'H') rec+='H'; else
                     rec+=kl;
  if (rec == s) cout<<"Yes"<<endl; else
                cout<<"No"<<endl;
 }
