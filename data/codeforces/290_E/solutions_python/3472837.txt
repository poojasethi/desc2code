def F():
	q = s.count('Q')
	Q = int(q ** .5)
	h = s.count('H')
	if Q * Q != q or h % (Q + 1):
		return 0
	H = h / (Q + 1)
	if not q:
		return 1
	p = s.index('Q')
	if p & 1:
		return 0
	t = s[p / 2 : p / 2 + Q + H]
	return ''.join([t, 'H'][x == 'H'] for x in t) == s

s = raw_input()
print ['No', 'Yes'][F()]
