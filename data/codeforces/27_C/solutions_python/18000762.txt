def main():
	n = input()
	a = map(int, raw_input().split())
	s = [(1, a[0])]
	for i in xrange(1, n):
		if a[i] != s[-1][1]: s.append((i + 1, a[i]))
	for i in xrange(1, len(s) - 1):
		if s[i - 1][1] < s[i][1] > s[i + 1][1] or s[i - 1][1] > s[i][1] < s[i + 1][1]:
			print 3
			print s[i - 1][0], s[i][0], s[i + 1][0]
			return 
	print 0
 
if __name__ == "__main__": main()
