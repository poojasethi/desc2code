#!/usr/bin/python
import sys
from operator import ge,le
def dbg(a): sys.stderr.write(str(a) + "\n")
def readint(): return int(raw_input())
def readfloat(): return float(raw_input())
def readarray(f): return map(f,raw_input().split())
def alloc(size, default = 0): return [default] * size
def dig(c): return ord(c) - 48

n = readint()
vs = readarray(int)
s = 3*[-1]
r = 0
op = le
if n >= 3:
    s[0] = 0
    i = 1
    while i < len(vs) \
              and vs[i] == vs[s[0]]:
        i += 1
    if i < len(vs):
        s[1] = i
        inc = vs[s[0]] < vs[s[1]]
        i += 1
        if inc: op = ge
        while i < len(vs) and op(vs[i], vs[s[1]]):
            s[1] = i
            i += 1
        if i < len(vs):
            s[2] = i
            s = [v+1 for v in s]
            r = len(s)

if r != 0:
    print r
