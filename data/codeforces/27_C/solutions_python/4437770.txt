#!/usr/bin/python
n = int(raw_input())
a = map(lambda x: int(x), raw_input().split())
prevSign = 0
for i in range(1, n):
    if a[i] == a[i-1]:
        continue
    if prevSign == 0 or cmp(a[i], a[i-1]) == prevSign:
        prevSign = cmp(a[i], a[i-1])
        lastPos = i
    else:
        print 3
        print lastPos, lastPos+1, i+1
        exit(0)
print 0
