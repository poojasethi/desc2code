n = int(raw_input())
a = map(int, raw_input().split())
greaterLeft = n*[None]
lessLeft = n*[None]
minIndex, maxIndex = 0, 0
for i in range(1, n):
    if a[i] <= a[minIndex]:
        minIndex = i
    else:
        lessLeft[i] = minIndex
    if a[i] >= a[maxIndex]:
        maxIndex = i
    else:
        greaterLeft[i] = maxIndex

found = False
minIndex, maxIndex = n-1, n-1
for i in range(n-2, -1, -1):
    if a[i] <= a[minIndex]:
        minIndex = i
    elif lessLeft[i] is not None:
        print 3
        print lessLeft[i]+1, i+1, minIndex+1
        found = True
        break
    if a[i] >= a[maxIndex]:
        maxIndex = i
    elif greaterLeft[i] is not None:
        print 3
        print greaterLeft[i]+1, i+1, maxIndex+1
        found = True
        break
if not found:
    print 0
