from operator import ge, le
n = input()
vs = map(int, raw_input().split())
s = 3*[-1]
r=0
op=le
if n >= 3:
	s[0] = 0
	i=1
	while i < len(vs) and vs[i] == vs[s[0]]: i += 1
	if i < len(vs):
		s[1] = i
		inc = vs[s[0]] < vs[s[1]]
		i += 1
		if inc: op = ge
		while i < len(vs) and op(vs[i], vs[s[1]]):
			s[1] = i
			i += 1
		if i < len(vs):
			s[2] = i
			s = [v+1 for v in s]
			r = len(s)
if r != 0:
	print r
	r = ' '.join(map(str, s))
print r
