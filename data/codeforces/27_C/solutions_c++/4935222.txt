#include<stdio.h>
int a[100001];
int main()
{
	int n,i,j,f;
	while(scanf("%d",&n)!=EOF)
	{
		f=0;
		for(i=1;i<=n;i++)
			scanf("%d",&a[i]);
		for(j=2;a[j]==a[1];j++){}
		if(a[j]>a[1])
		{
			for(i=j+1;i<=n;i++)
			{
				if(a[i]<a[i-1])
				{
					f=1;
					printf("3\n1 %d %d\n",i-1,i);
					break;
				}
			}
		}
		if(a[j]<a[1])
		{
			for(i=j+1;i<=n;i++)
			{
				if(a[i]>a[i-1])
				{
					f=1;
					printf("3\n1 %d %d\n",i-1,i);
					break;
				}
			}
		}
		if(f==0)
			printf("0\n");
	}
	return 0;
}
