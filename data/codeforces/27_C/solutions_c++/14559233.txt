#include <iostream>

using namespace std;

int n,in[100001],cnt;

int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>in[i];
    }
    int i=2;
    for(int j=1;j<=n-2;j++){
        if(in[i]>in[1] && in[i]>in[i+1]){
            goto finish;
        }
        if(in[i]<in[1] && in[i]<in[i+1]){
            goto finish;
        }
        i++;
    }
    finish:
    if(i+1<=n){
        cout<<3<<'\n';
        cout<<1<<" "<<i<<" "<<i+1;
    }
    else{
        cout<<0;
    }
}
