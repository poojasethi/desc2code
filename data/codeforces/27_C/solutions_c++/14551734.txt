#include<bits/stdc++.h>
using namespace std;

int main(){
	int n,a[100000],x;
	scanf("%d",&n);
	if(n>2){
		scanf("%d",&x);
	}
	for(int i=0;i<n-1;i++){
		scanf("%d",&a[i]);
	}
	for(int i=0;i<n-2;i++){
		if((x>a[i] && a[i]<a[i+1]) || (x<a[i] && a[i]>a[i+1])){
			printf("3\n1 %d %d",i+2,i+3);
			return 0;
		}
	}
	printf("0");
	return 0;
}
