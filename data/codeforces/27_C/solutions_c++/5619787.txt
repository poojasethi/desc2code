#include<stdio.h>

int main(){
    int A[100009],N;

    scanf("%d",&N);
    for (int i=1;i<=N;i++) scanf("%d", &A[i]);

    for (int i=2;i<=N-1;i++){
        if ( (A[1]>A[i] && A[i]<A[i+1]) || (A[1]<A[i] && A[i]>A[i+1]) ){
            printf("3\n");
            printf("1 %d %d\n", i, i+1);
            return 0;
        }
    }
    printf("0\n");
    return 0;
}
