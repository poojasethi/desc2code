#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n; scanf("%d",&n);
	int a[100000];
	rep(i,n) scanf("%d",a+i);

	int p_lt=-1,p_gt=-1;
	rep(i,n-1){
		if(p_lt==-1 && a[i]>a[i+1]) p_lt=i;
		if(p_gt==-1 && a[i]<a[i+1]) p_gt=i;

		if(~p_lt && ~p_gt){
			puts("3");
			if(p_lt<p_gt) printf("%d %d %d\n",p_lt+1,p_gt+1,p_gt+2);
			else          printf("%d %d %d\n",p_gt+1,p_lt+1,p_lt+2);
			return 0;
		}
	}

	puts("0");

	return 0;
}
