#include<cstdio>
#include<algorithm>
using namespace std;
int n,a[100001],u,a2[100001],d2[100001],n2;
int main(){
  scanf("%d",&n);
  for(int i=0;i<n;i++) scanf("%d",&a[i]);
  for(int i=0;i<n;i++){
      if(n2 == 0 || (a2[n2-1] != a[i]))
          a2[n2] = a[i], d2[n2++] = i+1;
  }
  for(int i=2;i<n2;i++){
    if((a2[i-2] > a2[i-1] && a2[i-1] < a2[i]) || (a2[i-2] < a2[i-1] && a2[i-1] > a2[i])){
      printf("3\n%d %d %d\n",d2[i-2],d2[i-1],d2[i]); u = 1;break;}
  }
  if(u == 0) printf("0");
  return 0;
}
