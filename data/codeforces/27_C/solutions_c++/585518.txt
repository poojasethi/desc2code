#include <cstdio>
#include <vector>
#include <algorithm>
#include <set>

using namespace std;

int main() {

    int N;
    int S[101010];

    scanf("%d", &N);
    for (int i = 0; i < N; i++) scanf("%d", S+i);

    if (N <= 2) { puts("0"); return 0; }

    for (int i = 1; i < N-1; i++) {
        if ((S[i+1]-S[i]) * (long long)(S[i]-S[0]) < 0) {
            printf("3\n%d %d %d\n", 1, i+1, i+2);
            return 0;
        }
    }

    puts("0");
    return 0;
}
