#include<iostream>
using namespace std;

int main()
{
    long int a[100000],n;
    
    cin>>n;
    for (int i=1;i<=n;i++) cin>>a[i];
    
    for (int i=2;i<=n-1;i++)
    {
        if ( (a[1]>a[i] && a[i]<a[i+1]) || (a[1]<a[i] && a[i]>a[i+1]) )
        {
             cout<<"3\n";
             cout<<"1 "<<i<<" "<<i+1<<endl;
             return 0;
        }
    }
    
    cout<<"0\n";
    return 0;
}
