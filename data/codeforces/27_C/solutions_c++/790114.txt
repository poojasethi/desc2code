#include<iostream>

using namespace std;

int main()
{
	int n;
	cin>>n;
	int res=0;
	long long arrey[n];
	for(int i=0 ; i<n ; i++)
		cin>>arrey[i];
	for(int i=1 ; i<n-1 ; i++)
	{
		if((arrey[i]-arrey[0])*(arrey[i+1]-arrey[i])<0)
		{
			res=1;
			cout<< 3 << endl;
			cout<< 1 << " " << i+1 << " " << i+2 << endl;
			return 0;
		}
	}
	if(res==0)
		cout<< 0 << endl;
}
