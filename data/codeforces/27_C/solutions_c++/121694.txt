#include <iostream>
using namespace std;
int main()
{
    long mass[100000],m[100000],n,i,j,k=0;
    cin>>n;
    cin>>mass[0]; m[0]=0;
    for(i=1;i<n;i++){cin>>mass[i-k]; m[i-k]=i; if(mass[i-k]==mass[i-k-1])k++;}
    for(i=1;i<n-1-k;i++)
    {
        if(mass[i-1]<mass[i]!=mass[i]<mass[i+1])
        {
            cout<<3<<endl<<m[i-1]+1<<' '<<m[i]+1<<' '<<m[i+1]+1;
            cin>>i;
            return 0;
        }
    }
        cout<<0;
        cin>>i;
        return 0;
}
