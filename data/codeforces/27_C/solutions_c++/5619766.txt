#include<stdio.h>

int main(){
    int a[100009],n;

    scanf("%d",&n);
    for (int i=1;i<=n;i++) scanf("%d", &a[i]);

    for (int i=2;i<=n-1;i++){
        if ( (a[1]>a[i] && a[i]<a[i+1]) || (a[1]<a[i] && a[i]>a[i+1]) ){
            printf("3\n");
            printf("1 %d %d\n", i, i+1);
            return 0;
        }
    }
    printf("0\n");
    return 0;
}
