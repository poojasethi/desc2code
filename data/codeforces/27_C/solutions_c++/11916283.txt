#include <bits/stdc++.h>
using namespace std;

const int N=100005;
int arr[N];
int main(){
    int n;
    cin>>n;

    for(int i=0;i<n;i++)cin>>arr[i];


    int ind;
    for(ind=1;ind<n;ind++)if(arr[ind-1]!=arr[ind])break;
    for(int i=ind;i<n-1;i++){
        if(arr[ind-1]<arr[i] && arr[i]>arr[i+1]  ||
           arr[ind-1]>arr[i] && arr[i]<arr[i+1]  ){
                cout<<"3\n";
                cout<<ind<<" "<<i+1<<" "<<i+2<<"\n";
                return 0;
           }
    }

    cout<<"0\n";
}
