#include<stdio.h>
#include<string.h>
int a[100001];
int main()
{
	int n,i,ok;
    while(scanf("%d",&n)!=-1)
	{
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		if(n<3)
		{printf("0\n");continue;}
		ok=0;
		for(i=1,ok=0;i<n-1;i++)
		{
			if(a[0]<a[i] && a[i]>a[i+1])
			{printf("3\n1 %d %d\n",i+1,i+2);ok=1;break;}
		}
		if(ok==0  )
		{
			for(i=1;i<n-1;i++)
			{
				if(a[0]>a[i] && a[i]<a[i+1])
				{printf("3\n1 %d %d\n",i+1,i+2);ok=1;break;}
			}
			if(ok==0) printf("0\n");
		}
		
		
		
	}
	return 0;
}
