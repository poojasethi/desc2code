#include<cstdio>
int a[100002];
int main(){
	int n,i,j,x,y,z;
	while(~scanf("%d",&n)){
		for(i=1;i<=n;i++)scanf("%d",&a[i]);
		int s=-1;
		for(i=2;i<=n;i++){
			if(a[i]!=a[i-1]){
				if(a[i]>a[i-1])s=0;
				else s=1;
				break;
			}
		}
		bool f=1;
		for(j=i+1;j<=n;j++){
			if(a[j]!=a[j-1]){
				if(a[j]>a[j-1]&&s==1){x=1,y=j-1,z=j;f=0;break;}
				if(a[j]<a[j-1]&&s==0){x=1,y=j-1,z=j;f=0;break;}
			}
		}
		if(f)puts("0");
		else printf("3\n%d %d %d\n",x,y,z);
	}
	return 0;
}
