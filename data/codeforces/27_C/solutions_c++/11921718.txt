#include<iostream>
using namespace std;

int main()
{
    int n, a[1000005], ind, f=0;

    cin>>n;
    for(int i=0;i<n;i++) cin>>a[i];

    for(ind=1;ind<n;ind++)
        if(a[ind-1]!=a[ind]) break;

    for(int i=ind;i<n-1;i++)
    {
        if((a[ind-1]<a[i] && a[i]>a[i+1]) ||
            (a[ind-1]>a[i] && a[i]<a[i+1]))
        {
            cout<<"3"<<endl;
            cout<<ind<<" "<<i+1<<" "<<i+2<<endl;
            return 0;
        }
    }
    cout<<"0"<<endl;
    return 0;
}
