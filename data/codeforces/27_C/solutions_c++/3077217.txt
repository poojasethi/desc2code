#include <iostream>

using namespace std;

long long a[100005];

int main()
{
    int i,n;
    while(cin >> n)
    {
        for(i=0;i<n;i++)
        cin >> a[i];
        for(i=1;i<n-1;i++)
        if((a[i]-a[0])*(a[i+1]-a[i])<0)
        {
            cout << 3 << endl << 1 << " " << i+1 << " " << i+2 << endl;
            break;
        }
        if(i>=n-1)
        cout << 0 << endl;
    }
    return 0;
}
