#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int n, x, y, i, s, a, b, c=0;
	cin>>n;
	cin>>x;
	for (i=2,s=0; i<=n; i++)  {
		cin>>y;
		if (s==0) {
			if (y>x) {
				s=1;
				a=i-1;
			}
			else if (y<x) {
				s=2;
				a=i-1;
			}
		}
		else if (s==1 && y<x || s==2 && y>x) {
			b=i-1;
			c=i;
			break;
		}
		x=y;   
	}
	if (c==0) cout<<0<<endl;
	else cout<<3<<endl<<a<<" "<<b<<" "<<c<<endl;
	return 0;
}
