#include <iostream>

using namespace std;

int main(int argc, char* argv[])
{
	int n;
	cin >> n;
	
	int f, d = 0, p;
	
	for (int i = 0; i < n; ++i)
	{
		int a;
		cin >> a;
		
		if (i == 0)
			f = a;
		else if (d == 0)
		{
			if (f < a)
				d = 1;
			else if (f > a)
				d = -1;
		}
		else
		{
			if (((p < a) && (d == -1)) ||
				((p > a) && (d == 1)))
			{
				cout << 3 << endl;
				cout << "1 " << i << " " << i + 1 << endl;
				return 0;
			}
		}
		
		p = a;
	}
	
	cout << 0 << endl;
	
	return 0;
}
