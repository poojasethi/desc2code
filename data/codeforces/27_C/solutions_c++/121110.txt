#include<iostream>
typedef long long ll;
using namespace std;
int main()
{
    ll n,arr[100001];
    cin >> n;
    for(ll i=1;i<=n;i++) cin >> arr[i];
    ll i,j,k;
    i=1;j=2;k=3;
    for(ll u=1;u<=n-2;u++){
        if(arr[j]>arr[i] && arr[j]>arr[k]) break;
        if(arr[j]<arr[i] && arr[j]<arr[k]) break;
        k++;j++;
    }
    if(k<=n) cout << "3\n1 " << j << " " << k << endl;
    else cout << "0\n";
    return 0;
}
