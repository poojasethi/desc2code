#include <iostream>
using namespace std;

int main(){
    int n;
    cin>>n;
    int a[n];
    for (int i=0;i<n;i++){
        cin>>a[i];
    }
    for (int i=1;i<n-1;i++){
        if ( (a[0]>a[i] && a[i]<a[i+1]) || (a[0]<a[i] && a[i]>a[i+1]) ){
            cout<<"3"<<endl;
            cout<<"1 "<<i+1<<" "<<i+2;
            return 0;
        }
    }
    cout<<"0";
    return 0;
}
