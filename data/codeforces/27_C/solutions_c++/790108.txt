#include<iostream>
using namespace std;
const int maxn=100100;
long long arr[maxn];
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>arr[i];
	for(int i=1;i<n-1;i++)
		if((arr[i]-arr[0])*(arr[i+1]-arr[i])<0){
			cout<<3<<endl<<1<<' '<<i+1<<' '<<i+2<<endl;
			return 0;
		}
	cout<<0<<endl;
}
