#include<cstdio>
#include<iostream>
using namespace std;
int n, m, i, j, sn, ans, a[510][510], f[510][510];
char str[100010];
struct rec1{
	int x, y;
	char c;
	rec1(){};
	rec1(char _c, int _x, int _y){
		c=_c;x=_x, y=_y;
	}
}s[1000010];
int dfs(int x, int y, int p){
	f[x][y] = 1;s[++sn]=rec1('B', x, y);
	if(x<n && a[x+1][y] && (!f[x+1][y]))dfs(x+1, y, 0);
	if(x>1 && a[x-1][y] && (!f[x-1][y]))dfs(x-1, y, 0);
	if(y<m && a[x][y+1] && (!f[x][y+1]))dfs(x, y+1, 0);
	if(y>1 && a[x][y-1] && (!f[x][y-1]))dfs(x, y-1, 0);
	if(!p)s[++sn]=rec1('D', x, y), s[++sn]=rec1('R', x, y);
}
int main(){
	scanf("%d%d", &n, &m);
	for(i=1;i<=n;i++){
		scanf("%s", str+1);
		for(j=1;j<=m;j++){
			a[i][j] = (str[j]=='.');
			if(a[i][j])ans+=2;
		}
	}
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			if((!f[i][j]) && a[i][j])ans--, dfs(i, j, 1);
		}
	}
	printf("%d\n", sn);
	for(i=1;i<=sn;i++)printf("%c %d %d\n", s[i].c, s[i].x, s[i].y);
	return 0;
}
