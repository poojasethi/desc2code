#include<stdio.h>
	
int dr[4][2]={0,1,0,-1,1,0,-1,0},n,m;
int op[1000000][3],s[3]={'B','R','D'},cnt=0;
char mp[501][501];
	
void dfs(int x,int y){
	for(int k=0;k<4;k++){
		int tx=x+dr[k][0],ty=y+dr[k][1];
		if(tx<0||ty<0||tx>=n||ty>=m||mp[tx][ty]!='.')
			continue;
		mp[tx][ty]='#';
		op[cnt][0]=tx;
		op[cnt][1]=ty;
		op[cnt++][2]=0;
		dfs(tx,ty);
		op[cnt][0]=tx;
		op[cnt][1]=ty;
		op[cnt++][2]=2;
		op[cnt][0]=tx;
		op[cnt][1]=ty;
		op[cnt++][2]=1;

	}
}	
int main(){
	int i,j;
	scanf("%d%d",&n,&m);
	for(cnt=i=0;i<n;i++)
		scanf("%s",mp[i]);
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			if(mp[i][j]=='.'){
				mp[i][j]='#';
				op[cnt][0]=i;
				op[cnt][1]=j;
				op[cnt++][2]=0;
				dfs(i,j);
			}
		}
	}
	printf("%d\n",cnt);
	for(i=0;i<cnt;i++)
		printf("%c %d %d\n",s[op[i][2]],op[i][0]+1,op[i][1]+1);
	return 0;
}
