#include<stdio.h>
#define X 510
char s[X][X];
bool vis[X][X],flag;
int xx[1000010],yy[1000010],c[1000010],top;
void dfs(int x,int y){
   vis[x][y]=1;
   xx[top]=x;yy[top]=y;c[top]=1;top++;
   if(!vis[x+1][y]&&s[x+1][y]=='.')
      dfs(x+1,y);
   if(!vis[x-1][y]&&s[x-1][y]=='.')
      dfs(x-1,y);
   if(!vis[x][y-1]&&s[x][y-1]=='.')
      dfs(x,y-1);
   if(!vis[x][y+1]&&s[x][y+1]=='.')
      dfs(x,y+1);
   xx[top]=x;yy[top]=y;c[top]=2;top++;
   xx[top]=x;yy[top]=y;c[top]=3;top++;
}
int main(){
   int i,j,n,m;
   scanf("%d%d",&n,&m);
   for(i=1;i<=n;i++)
      scanf("%s",s[i]+1);
   for(i=1;i<=n;i++)
      for(j=1;j<=m;j++)
         if(!vis[i][j]&&s[i][j]=='.')
            dfs(i,j),top-=2;
   printf("%d\n",top);
   for(i=0;i<top;i++)
      printf("%c %d %d\n",c[i]==1?'B':c[i]==2?'D':'R',xx[i],yy[i]);
   return 0;
}
