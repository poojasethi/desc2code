#include<iostream>
using namespace std;
bool f[500][500]={false};
char stk[1000005];
int stki[1000005][2],top=-1;
void dfs(char ch[][500],int i,int j,int n,int m,int val)
{
	if(ch[i][j]=='#'||f[i][j])
	return;
	stk[++top]='B';
	stki[top][0]=i+1;stki[top][1]=j+1;
	//cout<<"B "<<i+1<<" "<<j+1<<endl;
	f[i][j]=true;
	if(i!=0)
	dfs(ch,i-1,j,n,m,val+1);
	if(i<n-1)
	dfs(ch,i+1,j,n,m,val+1);
	if(j!=0)
	dfs(ch,i,j-1,n,m,val+1);
	if(j<m-1)
	dfs(ch,i,j+1,n,m,val+1);
	if(val)
	{
	stk[++top]='D';
	stki[top][0]=i+1;stki[top][1]=j+1;
	stk[++top]='R';
	stki[top][0]=i+1;stki[top][1]=j+1;}
	//cout<<"R "<<i+1<<" "<<j+1<<endl;
}
	
int main()
{
	int n,m,i,j;
	cin>>n>>m;
	char ch[n][500];
	for(i=0;i<n;i++)
	for(j=0;j<m;j++)
	cin>>ch[i][j];
	for(i=0;i<n;i++)
	for(j=0;j<m;j++)
	dfs(ch,i,j,n,m,0);
	cout<<top+1;
	for(i=0;i<=top;i++)
	cout<<endl<<stk[i]<<" "<<stki[i][0]<<" "<<stki[i][1];
}
