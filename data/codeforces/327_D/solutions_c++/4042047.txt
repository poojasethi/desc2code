#include <iostream>
#include <vector>
using namespace std;

char inp[1000][1000];
struct triple
{
  int x,y; char oper;
  triple(char a,int b,int c)
  {
    oper=a; x=b; y=c;
  }
};
vector<triple> v;

int n,m;

void dfs(int x,int y,bool red=true)
{
  if(x<1 or y<1 or x>n or y>m)
    return;
  if(inp[x][y]!='.')
    return;
  inp[x][y]='#';
  triple add('B',x,y);
  v.push_back(add);
  dfs(x+1,y);
  dfs(x-1,y);
  dfs(x,y+1);
  dfs(x,y-1);
  if(red)
    {
      triple add('D',x,y);
      v.push_back(add);
      triple add2('R',x,y);
      v.push_back(add2);
    }
}

int main()
{
  cin>>n>>m;
  for(int i=1;i<=n;++i)
    for(int j=1;j<=m;++j)
	cin>>inp[i][j];
  for(int i=1;i<=n;++i)
    for(int j=1;j<=m;++j)
      if(inp[i][j]=='.')
	dfs(i,j,false);
  cout<<v.size()<<endl;
  for(int i=0;i<v.size();++i)
    cout<<v[i].oper<<" "<<v[i].x<<" "<<v[i].y<<endl;
  return 0;
}
