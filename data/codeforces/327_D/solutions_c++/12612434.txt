#include<bits/stdc++.h>

using namespace std;

int ANI[1000006][2],m,n,i,j,d;

char A[555][555],ANS[1000006];

void f(int x, int y){	

	A[x][y] = '#';

	ANS[d] = 'B';
	ANI[d][0] = x;
	ANI[d++][1] = y; 

	if(x + 1 < m && A[x+1][y] != '#')  f(x+1,y);
	if(y + 1 < n && A[x][y+1] != '#')  f(x,y+1);
	if(x && A[x-1][y] != '#')          f(x-1,y);
	if(y && A[x][y-1] != '#')          f(x,y-1);

	ANI[d][0] = ANI[d+1][0] = x;
	ANI[d][1] = ANI[d+1][1] = y;
   ANS[d++] = 'D';
	ANS[d++] = 'R';
}

int main(){

	cin >> m >> n;

	for(i = 0; i < m; i++)
		scanf("%s",A[i]);

	for(i = 0; i < m; i++)
		for(j = 0; j < n; j++)
			if(A[i][j] != '#')
				f(i,j), d -= 2;
			
	cout << d << endl;

	for(i=0;i<d;i++)
		printf("%c %d %d\n",ANS[i], ANI[i][0] + 1, ANI[i][1] + 1);

	return 0;
}
