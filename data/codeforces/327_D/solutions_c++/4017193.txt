#include <vector>
#include <string>
#include <stdio.h>
#include <string.h>
using namespace std;

char v[555][555];
char s[555];
vector< vector<string> > f;

void fill(int x, int y) {
	if (v[x][y]) return ;
	v[x][y] = true;
	sprintf(s, "B %d %d", x, y);
	f.back().push_back(string(s));
	fill(x - 1, y);
	fill(x + 1, y);
	fill(x, y - 1);
	fill(x, y + 1);
	sprintf(s, "D %d %d", x, y);
	f.back().push_back(string(s));
	sprintf(s, "R %d %d", x, y);
	f.back().push_back(string(s));
}

#include<stdlib.h>
int main() {
	int n, m;
	int i, j, t = 0;
	scanf("%d%d", &n, &m);
	memset(v, 1, sizeof v);
	for (i = 1; i <= n; ++i) {
		scanf("%s", s + 1);
		for (j = 1; j <= m; ++j) v[i][j] = s[j] == '#';
	}

	for (i = 1; i <= n; ++i)
		for (j = 1; j <= m; ++j)
			if (!v[i][j]) {
				f.push_back(vector<string>());
				fill(i, j);
				t += f.back().size() - 2;
			}

	printf("%d\n", t);
	for (i = 0; i < f.size(); ++i)
		for (j = 0; j < f[i].size() - 2; ++j) puts(f[i][j].c_str());
	return 0;
}
