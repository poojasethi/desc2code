#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
int n,m,c1[4] = {-1,1,0,0}, c2[4]={0,0,-1,1};
vector < pair<char,pair<int,int> > > V;
char s[503][503];
void dfs(int x,int y,int ct){
    s[x][y] = '#';
    V.push_back(make_pair('B', make_pair(x,y)));
    for(int i=0;i<4;i++)
        if(s[x+c1[i]][y+c2[i]] == '.')
            dfs(x+c1[i],y+c2[i],ct+1);
    if(ct){
        V.push_back(make_pair('D', make_pair(x,y)));
        V.push_back(make_pair('R', make_pair(x,y)));
    }
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%s",s[i]+1);
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            if(s[i][j] == '.')
                dfs(i,j,0);
    printf("%d\n",V.size());
    for(int i=0;i<V.size();i++)
        printf("%c %d %d\n",V[i].first,V[i].second.first,V[i].second.second);
    return 0;
}
