#include <stdio.h>
int n,m,v[1<<20],k;
char g[512][512];
void go(int r,int c,int d=1){
  if('.'==g[r][c]){
    int w=(r<<12)+(c<<2);
    v[k++]=w;
    g[r][c]=1;
    go(r-1,c),go(r,c+1),go(r+1,c),go(r,c-1);
    if(d)v[k++]=w+1,v[k++]=w+2;
  }
}
main(){
  scanf("%d%d",&n,&m);
  for(int r=1;r<=n;++r)scanf("%s",g[r]+1);
  for(int r=1;r<=n;++r)for(int c=1;c<=m;++c)go(r,c,0);
  printf("%d\n",k);
  for(int i=0;i<k;++i)printf("%c %d %d\n","BDR"[v[i]&3],v[i]>>12,v[i]/4&0x3ff);
}
