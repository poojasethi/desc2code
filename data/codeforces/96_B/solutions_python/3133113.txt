luckys = [0]
def rec(length, str_num):
    if length > 0:
        for i in [4,7]:
            rec(length-1, str_num+str(i))
    else:
        if str_num.count("4") == str_num.count("7"):
            luckys.append(int(str_num))
            
for length in xrange(1, 11):
    rec(length, "")

n = input()
for i in xrange(len(luckys)):
    if luckys[i] < n and n <= luckys[i+1]:
        print luckys[i+1]
        break
