import sys
from itertools import product

file = sys.stdin
#file = open("test", "r")
num = long(file.readline().rstrip())

for l in [i for i in range(2, 11) if i%2 == 0]:
    p = []
    for t in [x for x in product([4, 7], repeat=l) if x.count(4) == l/2]:
        lv = long("".join(map(str, list(t))))
        p.append(lv)
    p.sort()

    for t in p:
        if t >= num:
            print t
            exit()
