#! /usr/bin/env python

from sys import stdin

def generate_super(size):
	return "4"*(size/2)+"7"*(size/2)

def smallest_lucky(n):
	s=str(n)
	sz=len(s)
	if sz&1:
		sz+=1
	x=generate_super(sz)
	while True:
		x=generate_super(sz)
		if int(x)<n:
			sz+=2
		else:
			break
	return x

def generate(prefix,lst, step,fs,ss):
	if step>10:
		return
	else:
		if (step&1)==0 and fs==ss:

			lst.append(prefix)
		generate(prefix*10+4,lst,step+1,fs+1,ss)
		generate(prefix*10+7,lst,step+1,fs,ss+1)

if __name__=='__main__':
	n=int(stdin.readline().strip())
