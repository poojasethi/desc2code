import itertools
n=raw_input()
d=len(n)
n=int(n)
if(d%2!=0):
	print str(4)*(d/2+1)+str(7)*(d/2+1)
else:
	x=int(str(7)*(d/2)+str(4)*(d/2))
	if(n>x):
		print str(4)*(d/2+1)+str(7)*(d/2+1)
	else:
		permutation=itertools.permutations(['4']*(d/2)+['7']*(d/2))
		for i in permutation:
			j=int(''.join(i))
			if(j>=n):
				print j
				break
