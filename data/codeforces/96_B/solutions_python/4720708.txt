n=input()
c=len(str(n))
if c%2:
    print ('4'*((c+1)/2))+('7'*((c+1)/2))
    exit()
s=lambda x,n:[int(e) for e in x if e.count('4') == e.count('7') and int(e) >= n]
l=['4','7']
for i in range(c-1):
    r=[]
    for e in l:
        r+=[e+'4',e+'7']
    l=r
l2=['4','7']
for i in range(c+1):
    r=[]
    for e in l2:
        r+=[e+'4',e+'7']
    l2=r
print min(s(l+l2,n))
