from itertools import *
res = []
for L in range(2,12,2):
    x = L>>1
    stuff = range(L)
    for sub in combinations(stuff,x):
        num = ["4"]*L
        for i in sub:
            num[i] = "7"
        res+= [int("".join(num))]
res.sort()
t = input()
for i in res:
    if(i>=t):
        print i
        break
