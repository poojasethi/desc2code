import sys
import itertools
import collections
n = int(sys.stdin.readline())
def luckies(digs):
    for i in range(1<<digs):
        s = bin(i)[2:].zfill(digs)
        if s.count('1') == digs/2:
            s = s.replace('1','7').replace('0','4')
            yield int(s)

m = len(str(n))
if m % 2 == 1:
    lucks = luckies(m+1)
elif m % 2 == 0 and n <= int("7"*(m/2)+"4"*(m/2)):
    lucks = luckies(m)
else:
    lucks = luckies(m+2)

for l in lucks:
    if l >= n:
        print l
        break
