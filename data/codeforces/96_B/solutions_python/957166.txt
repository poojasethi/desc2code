n = input()
w = len(str(n))

class X():
  x = 1000000000000L

def dfs(w4, w7, t=0):
  if w4 > 0 or w7 > 0:
    if w4 > 0: dfs(w4 - 1, w7, t * 10 + 4)
    if w7 > 0: dfs(w4, w7 - 1, t * 10 + 7)
  else:
    if t >= n: X.x = min(X.x, t)

if w & 0x01: w += 1
dfs(w >> 1, w >> 1)
dfs((w >> 1) + 1, (w >> 1) + 1)
print X.x
