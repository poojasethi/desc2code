
def binary(n):
	if (n==0):	
		yield ''
	else:
		for i in binary(n-1):
			yield i + '4'
			yield i + '7'



def gen():
	for i in range(2,12,2):
		for code in binary(i):
			if (code.count('4') == code.count('7')):
				yield int(code)





def main():
	num = input()
	xs = sorted(list(gen()) + [num])
	print xs[xs.index(num) + 1]




main()
