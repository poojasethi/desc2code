n = input()
from itertools import  permutations
import  bisect
for i in range(1,5):
    nums = sorted([int("".join(per)) for per in permutations('4'*i + '7'*i)])
    pos = bisect.bisect_left(nums, n)
    if pos != len(nums) and nums[pos] >= n:
        print nums[pos]
        exit(0)
