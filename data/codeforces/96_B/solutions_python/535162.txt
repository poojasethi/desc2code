#!/usr/bin/env python
import sys
n = raw_input()

def is_super_lucky(n):
    four = 0
    seven = 0
    for d in n:
        if d == '4':
            four += 1
        elif d == '7':
            seven += 1
        else:
            return False
    if four == seven:
        return True
    return False

l = nl = len(n)
if l %2 == 0:
    if is_super_lucky(n):
        print n
        sys.exit(0)

if n[0] == '8' or n[0] == '9':
    l += 1

if l%2:
    l += 1

for i in range(2**l):
    b = bin(i)[2:]
    num = '0'*(l-len(b)) + b
    lnum = ''.join(map(lambda(a): '4' if a == '0' else '7',num))
    if nl == l:
        if lnum <= n:
            continue
    if is_super_lucky(lnum):
        print lnum
        break
else:
    l += 1
    if l%2:
        l += 1
    for i in range(2**l):
        b = bin(i)[2:]
        num = '0'*(l-len(b)) + b
        lnum = ''.join(map(lambda(a): '4' if a == '0' else '7',num))
        if is_super_lucky(lnum):
            print lnum
            break
