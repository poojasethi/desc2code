n = int(raw_input())
luck = []
for length in range(1,11):
 for mask in range(0,2**length):
  cur = 0
  four=0
  seven=0
  for i in range(length):
   cur*=10
   if mask & (2**i):
    cur+=4
    four+=1
   else:
    cur+=7
    seven=seven+1
  if four==seven:
   luck += [cur]


res=-1
for lucky in luck:
 if lucky>=n:
  if res==-1 or res>lucky:
   res=lucky

print res
