n = raw_input()
ans = ['4','7']
l = len(ans)
n = long(n)
while ans.count('4') != ans.count('7') or long("".join(ans)) < n:
    for i in range(l):
        if ans[-1-i] == '4':
            ans[-1-i] = '7'
            break
        else:
            ans[-1-i] = '4'
            if i == l-1:
                ans.insert(0,'4')
                l += 1
print long("".join(ans))
