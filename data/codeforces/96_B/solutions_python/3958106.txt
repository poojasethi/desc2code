import sys

def appendLuckyNums(index, vals, sofar, s):
    if index == 0:
        if s == 0:
            vals.append(sofar)
        return

    appendLuckyNums(index-1, vals, 
                    sofar * 10 + 4, s+1)
    appendLuckyNums(index-1, vals, 
                    sofar * 10 + 7, s-1)

lucky_lucky_nums = []

for i in range(2, 12):
    appendLuckyNums(i, lucky_lucky_nums, 0, 0)

lucky_lucky_nums = sorted(lucky_lucky_nums)
#print lucky_lucky_nums

num = int(sys.stdin.readline())

start, end = 0, len(lucky_lucky_nums)-1

while (end >= start):
    mid = (start + end) / 2

    if lucky_lucky_nums[mid] > num:
        end = mid-1
    elif lucky_lucky_nums[mid] < num:
        start = mid+1
    else:
        start = mid
        end = mid
        break

print lucky_lucky_nums[start]
