LIM = 10

nums = [4, 7]
start = 0
for _ in range(LIM-1):
	l = len(nums)
	for x in xrange(start, l):
		nums.append(nums[x] * 10 + 4)
		nums.append(nums[x] * 10 + 7)
	start = l
	#print start, len(nums)

super_lucky = []
for num in nums:
	s = str(num)
	if s.count('4') == s.count('7'):
		super_lucky.append(num)

# print len(nums)
# print len(super_lucky)


def b(n):
	for sl in super_lucky:
		if sl >= n:
			return sl
	return -1
