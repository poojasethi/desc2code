
all47 = set([])


def dfs(x, v):
  if x > 10**10:
    return

  v.add(x)
  dfs(x * 10 + 4, v)
  dfs(x * 10 + 7, v)


def eq(x):
  c4 = 0
  c7 = 0
  for c in str(x):
    if c == '4':
      c4 += 1
    if c == '7':
      c7 += 1
  return c4 == c7


dfs(0, all47)
all47 = sorted(list(all47))
all47 = filter(eq, all47)


n = input()
for x in all47:
  if x >= n:
    print x
    break
