#include <iostream>
using namespace std;
long long int res,n;
void calc(long long int m,int k){
  	if(m>=n && k==0){
    		res=min(res,m);
  	}
  	if(m>=n*1000){ 
		return;
	}
  	calc(m*10+4,k-1);
  	calc(m*10+7,k+1);
}
int main(){
  	res=10E12;;
  	cin>>n;
  	calc(0,0);
  	cout<<res<<endl;
  	return 0;
}
