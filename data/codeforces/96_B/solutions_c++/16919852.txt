#include<iostream>
using namespace std;
long long n,ans=1LL<<60;
void DFS(long long x,int y,int z)
{
    if (x>=n)
        if (y==z)
            ans=min(ans,x);
    if (x>n*100)
        return ;
    DFS(x*10+4,y+1,z);
    DFS(x*10+7,y,z+1);
}
int main()
{
    cin>>n;
    DFS(0,0,0);
    cout<<ans<<endl;
    return 0;
}
