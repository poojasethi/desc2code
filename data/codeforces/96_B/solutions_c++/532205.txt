#include <iostream>
#include <string>
using namespace std;

long long A;
long long ans;

void recur (long long n, int f, int s)
{
	if (n > 1000000000000ll) return;
	if (n >= A && f == s)
	{
		if (ans == -1) ans = n;
		ans = min(ans,n);
		return;
	}
	recur(n*10+4,f+1,s);
	recur(n*10+7,f,s+1);
}

main ()
{
	cin >> A;

	ans = -1;
	recur(0,0,0);
	cout << ans << "\n";
}
