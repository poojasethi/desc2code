#include <iostream>
#include <conio.h>
using namespace std;
long long value = 7777777777ll,n;
void dfs(long long v,int x,int y)
{
//	cout << v << endl;getch();
	if (v>=n && x==y)
		value = min(value,v);
	if (v>=n*100) return;
	dfs(v*10+4,x+1,y);
	dfs(v*10+7,x,y+1);
}
int main()
{
	cin >> n;
//	cout << n << endl;
	dfs(0,0,0);
	cout << value << endl;
	return 0;
}
