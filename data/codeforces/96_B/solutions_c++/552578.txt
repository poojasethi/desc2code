#include <cstdio>
#include <algorithm>

using namespace std;

typedef long long ll;

ll res[1<<11], nres;

void dfs(ll x, int q, int s) {
	if (x > 10000000000LL) return;
	if (q == s) res[nres++] = x;
	dfs(x*10+4, q+1, s);
	dfs(x*10+7, q, s+1);
}

int main() {
	nres = 0;
	dfs(0, 0, 0);
	sort(res, res+nres);
	int n;
	scanf("%d", &n);
	printf("%I64d\n", *lower_bound(res, res+nres, n));
}
