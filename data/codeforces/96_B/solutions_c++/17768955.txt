#include<iostream>
using namespace std;
long long t,c1,c2,a,i;
int main(){
	cin>>i;
	if(i>=77774445){cout<<4444477777;return 0;}
	while(1){
		t=i;c1=0;c2=0;
		while(t!=0){
			a=t%10;
			if(a!=4&&a!=7)break;//
			if(a==4)c1++;
			if(a==7)c2++;
			t/=10;
		}
		//cout<<i;system("pause");

		if(c1==c2){cout<<i;return 0;}

	}
