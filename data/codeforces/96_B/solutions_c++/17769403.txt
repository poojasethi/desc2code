#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
vector < long long > l;
long long n;
void rec(long long a,int b,int c)
{
	if(a>0&&b==c) l.push_back(a);
	if(a>n*10) return;
	rec(a*10+4,b+1,c);
	rec(a*10+7,b,c+1);
}
int main()
{
	cin>>n;
	rec(0,0,0);
	sort(l.begin(),l.end());
	cout<<*lower_bound(l.begin(),l.end(),n);
}
