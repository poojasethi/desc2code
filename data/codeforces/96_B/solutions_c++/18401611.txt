#include<bits/stdc++.h>
using namespace std;

int main(){
	
	unsigned long long int x;
	cin>>x;
	for(int k=1;k<=10;k++)
	for(int i=0;i<(1<<k);i++){
		unsigned long long int  a=0,b=0,t=0;
		for(int j=k-1;j>=0;j--){
			if(i&(1<<j))t=t*10+7,a++;
			else t=t*10+4,b++;
		}
		if(a==b&&t>=x){
				cout<<t<<endl;
				return 0;
		}
	}

	return 0;
}
