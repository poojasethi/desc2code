#include<bits/stdc++.h>
using namespace std;
long long n,s;
void fun(long long x,int c4,int c7)
{
    if(x>=n&&c4==c7&&(!s||s>x))s=x;
    if(x<n*100){
        fun(x*10+4,c4+1,c7);
        fun(x*10+7,c4,c7+1);
    }
}
int main()
{
    cin>>n;
    fun(0,0,0);
    if(n==0){
        cout<<47;
    }
    else
    cout<<s;
}
