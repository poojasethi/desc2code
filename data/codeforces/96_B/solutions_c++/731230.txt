#include <iostream>
#include <algorithm>

using namespace std;

typedef long long ll;

ll n, ans = 1e10;

void recur(ll num, ll num4, ll num7){
	if (num >= 1e10) return;
	if (num >= n && num4 == num7){
		ans = min(ans, num);
		return;	
	}
	recur(num*10+4, num4+1, num7);
	recur(num*10+7, num4, num7+1);
}

int main(){

	cin >> n;
	recur(0,0,0);
	cout << ans << endl;

	return 0;
}
