#include <iostream>
using namespace std;
long long maxs = 9999999999, ans, n;
void fun(long long  m, int n4, int n7)
{
	if (m >= maxs)
		return;
	if (n4 == n7 && m>=n)
	{
		if (ans > m)
			ans = m;
		return;
	}

	fun(10 * m + 4, n4 + 1, n7);
	fun(10 * m + 7, n4, n7 + 1);

}
int main()
{
	cin >> n;
	ans = maxs;
	fun(0, 0, 0);
	cout << ans << endl;

}
