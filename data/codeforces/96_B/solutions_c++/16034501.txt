#include<bits/stdc++.h>
using namespace std;
int q=0;
long long n,ans=LONG_LONG_MAX;
void rec(long long x,int a,int b,int t)
{
	if(t>=q)
	{
		if(x>=n&&a==b)
		ans=min(ans,x);
		if(t-2>=q)
		return;
	}
	rec(x*10+4,a+1,b,t+1);
	rec(x*10+7,a,b+1,t+1);
}
int main()
{
	cin>>n;
	long long k=n;
	while(k)
	{
		k/=10;
		q++;
	}
	rec(0,0,0,0);
	cout<<ans<<endl;
}
