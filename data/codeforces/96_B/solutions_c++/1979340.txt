#include<iostream>
using namespace std;
#define LL long long
LL n,ans;

void dfs(LL x,LL a,LL b)
{
	if(a!=0 && a==b && x>=n &&(ans==0 || x<ans))
		ans=x;
	if(x<=100*(n+1)) //（n+1）防止 n==0  控制DFS 
	{
		dfs(x*10+4,a+1,b);
		dfs(x*10+7,a,b+1);
	}
}
int main()
{
	while(cin>>n)
	{
		ans=0;
		dfs(0,0,0);
		cout<<ans<<endl;
	}
	return 0;
}
