#include<iostream>

using namespace std;

long long maximum = 9999999999;
long long n,ans=9999999999;

void f(long long m,int n4,int n7)
{
	if(m>=maximum)	return;

	if(m>=n && n4 == n7)
	{
		ans = min(ans,m);
		return;
	}
	
	
	f(10*m+4,n4+1,n7);
	f(10*m+7,n4,n7+1);
}

int main()
{
	cin>>n;
	f(0,0,0);

	cout<<ans<<endl;
}
