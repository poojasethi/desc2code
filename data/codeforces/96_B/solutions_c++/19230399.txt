#include <iostream>
#include <cstdio>
long long n,ans=-1;
void dfs(long long now,int a,int b){
	if (now/n>100) return;
	if (now>=n&&a==b){
		if (ans==-1||now<ans) ans=now;
	}
	dfs(now*10+4,a+1,b);
	dfs(now*10+7,a,b+1);
}
int main(){
	scanf("%I64d",&n);
	dfs(0,0,0);
	printf("%I64d",ans);
	return 0;
}
