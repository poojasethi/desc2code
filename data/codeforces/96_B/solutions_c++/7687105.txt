#include <iostream>
#define f cin
#define g cout
using namespace std;
long long n,s;
void rez(long long x,int nr7,int nr4)
{
	if(x>=n && nr7==nr4 && (x<s || !s))
	{
	    s=x;
	}
	if(x<n*100)
	{
		rez(x*10+4,nr7,nr4+1);
		rez(x*10+7,nr7+1,nr4);
	}
}
int main()
{
	f>>n;
	rez(0,0,0);
	if(n==0)
	{
	    s=47;
	}
	g<<s;
	return 0;
}
