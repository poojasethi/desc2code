#include <iostream>
#include <climits>
using namespace std;
int n;
long long int mn=LONG_LONG_MAX;
void bas(int k,long long int sayi,int s1,int s2)
{
	if(k>10) return;
	if(s1==s2&&sayi>=n&&sayi<mn)
		mn=sayi;
	bas(k+1,sayi*10+4,s1,s2+1);
	bas(k+1,sayi*10+7,s1+1,s2);
	return;
}
int main()
{
	cin>>n;
	bas(0,0,0,0);
	cout<<mn;
}
