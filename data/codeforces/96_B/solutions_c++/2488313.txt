#include <cstdio>
using namespace std;

typedef long long ll;

ll n, res = 7777777777ll;
void find(ll a, ll c4, ll c7) {
  if (a < res) {
    if (c4 == c7 && a >= n && a < res) res = a;
    find(a * 10 + 4, c4 + 1, c7);
    find(a * 10 + 7, c4, c7 + 1);
  }
}

int main() {
  scanf("%I64d", &n);
  find(0, 0, 0);
  printf("%I64d\n", res);
  return 0;
}
