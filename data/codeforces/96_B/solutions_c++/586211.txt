#include<iostream>

using namespace std;
const long long MAX =5*1000*1000*1000LL;

long long ans=-1;
int n;
void make(long long x,int a4,int a7)
{
	if(x>MAX)
		return;
	if(x>=n && a4==a7)
	{
		if(ans==-1) 
			ans = x;
		ans= min(ans,x);
	}
	make(x*10+4,a4+1,a7);
	make(x*10+7,a4,a7+1);
}
int main()
{
	cin >> n;
	make(0,0,0);
	cout << ans << endl;
	
	return 0;
}
