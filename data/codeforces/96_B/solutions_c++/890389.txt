#include <iostream>
using namespace std;
#define INF 10E12;

typedef long long int lli;

void calc(lli,int);

lli N;
lli res;

int main(){
  res=INF;
  cin >> N;
  calc(0,0);
  cout << res << endl;
  return 0;
}



void calc(lli m,int k){
  if(m>=N && k==0){
    res=min(res,m);
  }
  if(m>=N*1000) return;
  calc(m*10+4,k-1);
  calc(m*10+7,k+1);
}
