#include <iostream>
using namespace std;

long long res, n;

void solve(long long num, int c)
{
	if(num >= n && (res == -1 || res > num))
		if(c == 0)
			res = num;
	if(num >= 1000 * n)
		return;
	solve(num * 10 + 7, c + 1);
	solve(num * 10 + 4, c - 1);
}

int main()
{
	cin >> n;
	res = -1;
	solve(0, 0);
	cout << res << endl;
	return 0;
}
