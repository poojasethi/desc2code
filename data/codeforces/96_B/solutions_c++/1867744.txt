#include<stdio.h>
#include<stdlib.h>
__int64 n,res;
void dfs(__int64 x,int f,int s){
if(x>100000000000)return;
if(x>=n&&f==s)if(res==0||res>x)res=x;
dfs(x*10+4,f+1,s);
dfs(x*10+7,f,s+1);
}
int main(){
	
	scanf("%I64d",&n);
	dfs(4,1,0);
	dfs(7,0,1);
	printf("%I64d\n",res);
	return 0;

}
