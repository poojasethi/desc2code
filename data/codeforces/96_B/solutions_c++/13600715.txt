#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll n;
vector<ll> v;
void f(ll x,int sv,int fr)
{
   if(x>4444477777) return ;
   if(x&&sv==fr) v.push_back(x);
   f(x*10+7,sv+1,fr);
   f(x*10+4,sv,fr+1);
}
int main ()
{
    cin>>n ;
    f(0,0,0);
    sort(v.begin(),v.end());
    for(int i=0;i<v.size();i++) if(v[i]>=n) {cout<<v[i];break;}
}
