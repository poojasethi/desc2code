#include <iostream>
#include <cmath>
#define MAXN 1LL<<62
using namespace std;

long long n;
long long mn = MAXN;

void ans(long long x,int cont4,int cont7){
    if(x >= n && (cont4 == cont7)) mn = min(mn,x);
    if(x > n*100) return;
    ans(x*10+4, cont4+1, cont7);
    ans(x*10+7, cont4, cont7+1);
}

int main(){
    cin >> n;
    ans(0, 0, 0);
    cout << mn << endl;
}
