#include <iostream>

using namespace std;
long long int n,re;
void fun(long long int x,int c4,int c7)
{
    if(x>=n&&c4==c7&&(x<re||re==0))
    {
        re=x;
    }
    if(x<n*100)
    {
    fun(x*10+4,c4+1,c7);
    fun(x*10+7,c4,c7+1);
    }
}
int main()
{
    cin>>n;
    fun(0,0,0);
    cout<<re;
    return 0;
}
