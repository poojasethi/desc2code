#include <cstdio>
#include <iostream>

using namespace std;

typedef long long int Lint;

const int MAXN=512;

Lint res;

int N,M,K;
int sum[MAXN][MAXN];

char ar[MAXN][MAXN];

bool star[MAXN][MAXN];

int main(){
	
	scanf(" %d %d %d",&N,&M,&K);
	
	for( int i=1 ; i<=N ; i++ )
		for( int j=1 ; j<=M ; j++ )
			scanf(" %c",&ar[i][j]);
	
	for( int i=1 ; i<=N ; i++ )
		for( int j=1 ; j<=M ; j++ )
			if( ar[i][j]=='1' && ar[i-1][j]=='1' && ar[i+1][j]=='1' && ar[i][j-1]=='1' && ar[i][j+1]=='1' )
				star[i][j]=1;
	
	for( int i=1 ; i<=N ; i++ )
		for( int j=1 ; j<=M ; j++ )
			sum[i][j]=star[i][j]+sum[i-1][j]+sum[i][j-1]-sum[i-1][j-1];
	
	for( int i=1 ; i<=N ; i++ )
		for( int j=1 ; j<=M ; j++ )
			for( int l=1,k=j-2 ; k>=1 ; k-- ){
				for( ; l<i-1 && sum[i-1][j-1]-sum[l][j-1]-sum[i-1][k]+sum[l][k]>=K ; l++ );
				res+=(Lint)l-1;
			}
	
	printf("%I64d\n",res);
	return 0;
}
