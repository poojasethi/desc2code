#include<iostream>
#include<cstdio>
#include<algorithm>

using namespace std;

#define forange(i, a, b) for(int i(a); i < (b); i++)
#define rep(i, n) forange(i, 1, n + 1)

#define SUM (q[i][j] - q[r - 1][j] - q[i][c - 1] + q[r - 1][c - 1])

#define long long long

const int MAX_N = 505;

int n, m, k;
long con;
char s[MAX_N][MAX_N];
int q[MAX_N][MAX_N];

int main()
{
    cin >> n >> m >> k;
    rep(i, n)
        scanf("%s", &s[i][1]);
    rep(i, n - 2)
    {
        int sum_row = 0;
        rep(j, m - 2)
        {
            if(s[i][j + 1] == '1'and s[i + 1][j] == '1' and s[i + 1][j + 1] == '1' and s[i + 1][j + 2] == '1' and s[i + 2][j + 1] == '1')
                sum_row++;
            q[i][j] = q[i - 1][j] + sum_row;
        }
    }
    n -= 2, m -= 2;
    rep(i, n)
    {
        rep(j, m)
        {
            int r = 1, c = 1;
            for(; SUM >= k and r <= i; r++)
                con++;
            r--;
            for(c = 2; c <= j; c++)
            {
                for(; SUM < k and r >= 1; r--);
                con += r;
            }
        }
    }
    cout << con;
}
