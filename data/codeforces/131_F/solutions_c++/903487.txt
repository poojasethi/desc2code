#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

typedef long long ll;

const int dx[]={0,1,0,-1,0},dy[]={0,0,-1,0,1};

int main(){
	int m,n,k; scanf("%d%d%d",&m,&n,&k);
	static char sky[500][501];
	rep(i,m) scanf("%s",sky[i]);

	static int dp[501][500];
	for(int i=1;i<m-1;i++) for(int j=1;j<n-1;j++) {
		bool ok=true;
		rep(l,5){
			int y=i+dy[l],x=j+dx[l];
			if(sky[y][x]=='0') ok=false;
		}
		if(ok) dp[i][j]=1;
	}

	rep(i,m) rep(j,n) dp[i+1][j]+=dp[i][j];

	ll ans=0;
	for(int i1=1;i1<m-1;i1++) for(int i2=i1+1;i2<m;i2++) {
		int star=0;
		for(int j1=1,j2=1;j1<n-1;){
			for(;j2<n;j2++){
				if(star>=k) break;
				star+=dp[i2-1][j2]-dp[i1-1][j2];
			}

			ans+=n-j2;

			star-=dp[i2-1][j1]-dp[i1-1][j1];
			j1++;
		}
	}

	printf("%I64d\n",ans);

	return 0;
}
