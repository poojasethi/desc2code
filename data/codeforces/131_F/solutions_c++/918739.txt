#include <iostream>
using namespace std;
int a[500][500];
int s[500][500];
int calc(int u, int d, int l, int r){
	return s[d-1][r-1]+s[u][l]-s[d-1][l]-s[u][r-1];
}
int main(){
	int n, m, k;
	cin >> n >> m >> k;
	for(int i=0; i<n; i++){
		for(int j=0; j<m; j++){
			while(1){
				char c;
				cin >> c;
				if(c=='0' || c=='1'){
					a[i][j]=c-'0';
					break;
				}
			}
		}
	}
	for(int i=1; i<n-1; i++)
	for(int j=1; j<m-1; j++){
		s[i][j]-=s[i-1][j-1];
		s[i][j]+=s[i-1][j];
		s[i][j]+=s[i][j-1];
		s[i][j]+=(a[i][j]&a[i-1][j]&a[i][j-1]&a[i+1][j]&a[i][j+1]);
		//cerr<< s[i][j]<< " ";
	}
	long long ans=0;
	for(int u=0; u<n; u++)
	for(int d=u+1; d<n; d++){
		int r=0;
		for(int l=0; l<m; l++){
			while(r<m  && calc(u, d, l, r) < k){
				r++;
			}
			if(r==m)
				break;
			else
				ans+=m-r;
			//cerr<< u << " " << d << " "<<  l << " " << r<< endl;
		}
	}
	cout<<ans<<endl;
}
