#include <cstdio>
#include <cstdlib>
#include <iostream>

using namespace std;

bool a[500][500];
bool b[500][500];
int c[501][501];

int main(){
  int n,m,K; 
  scanf("%d%d%d", &n, &m, &K);
  for(int i = 0; i < n; i++){
    string s; cin >> s;
    for(int j = 0; j < m; j++)
      a[i][j]=s[j]=='1';
  }
  for(int i = 1; i+1 < n; i++)
    for(int j = 1; j+1 < m; j++)
      b[i-1][j-1] = a[i-1][j] && a[i][j-1] 
                    && a[i][j] && a[i][j+1] && a[i+1][j];
  for(int i = 1; i <= n-2; i++)
    for(int j = 1; j <= m-2; j++)
      c[i][j] = c[i][j-1]+c[i-1][j]-c[i-1][j-1]+b[i-1][j-1];

  long long res=0;
  for(int i = 1; i <= n-2; i++){
    for(int j = 1; j <= m-2; j++){
      int l=j;
      for(int k = 0; k < i; k++){
        while(l>=0 and c[i][j]-c[k][j]-c[i][l]+c[k][l]<K)
          l--;
        if(l<0) break;
        res += l+1;
      }
    }
  }
  cout << res << endl;
  return 0;
}
