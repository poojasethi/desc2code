#include<cstdio>
using namespace std;

char g[505][505];
int sum[505][505] = {0};
int n, m, k;
long long ans;

int main()
{
    scanf("%d%d%d", &n, &m, &k);
    for (int i = 0; i < n; i ++)
        scanf("%s", g[i]);
    for (int i = 0; i < n; i ++)
        for (int j = 0; j < m; j ++)
        {
            sum[i][j] = sum[i][j - 1];
            if (g[i][j] == '1' && 
                g[i + 1][j] == '1' && 
                g[i - 1][j] == '1' &&
                g[i][j + 1] == '1' &&
                g[i][j - 1] == '1')
                sum[i][j] ++;
        }
    ans = 0;
    for (int r = 1; r < m - 1; r ++)
        for (int l = 0; l < r; l ++)
        {
            int t = 0, s = 0;
            for (int b = 1; b < n - 1; b ++)
            {
                ans += t;
                s += sum[b][r] - sum[b][l];
                while (s >= k)
                {
                    ans ++;
                    t ++;
                    s -= sum[t][r] - sum[t][l];
                }
            }
        }
    printf("%I64d\n", ans);
    return 0;
}
