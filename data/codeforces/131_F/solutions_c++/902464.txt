#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char s[510][510];
int dir[4][2]={0,1,0,-1,1,0,-1,0};
int a[510][510],n,m;
int sure(int x,int y)
{
	return x>=1&&x<=n&&y>=1&&y<=m;
}
int main()
{
	int i,j,k,r;
	scanf("%d%d%d",&n,&m,&r);
	memset(s,0,sizeof(s));
	for(i=1;i<=n;i++)
		scanf("%s",s[i]+1);
	__int64 ans=0;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			if(s[i][j]=='1'&&s[i][j-1]=='1'&&s[i-1][j]=='1'&&s[i][j+1]=='1'&&s[i+1][j]=='1') a[i][j]=a[i-1][j]+a[i][j-1]+1-a[i-1][j-1];
			else a[i][j]=a[i-1][j]+a[i][j-1]-a[i-1][j-1];
	int lt,rt;
	int x;
	for(i=1;i<=n;i++){
		for(j=i+2;j<=n;j++){
			lt=1,rt=3;
			while(lt<=rt&&rt<=m){
				x=a[j-1][rt-1]-a[i][rt-1]-a[j-1][lt]+a[i][lt];
				if(x<r) rt++;
				else{
					ans+=(m-rt+1);
					lt++;
				}
			}

		}
	}
	printf("%I64d\n",ans);
	//cin>>ans;
	//return 0;

	return 0;
}
