#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<vector>
#include<string>
#include<queue>
#define X first
#define Y second
#define pb push_back
#define mp make_pair
using namespace std;
const int N=555;
int n,m;
int a[N][N],s[N],sum[N][N];
int main()
{
	long long ans=0;
	int k,i,j,x,y,l,r;
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;i++)
	for(j=1;j<=m;j++)
	scanf("%1d",&a[i][j]);
	for(i=1;i<=n;i++)
	for(j=1;j<=m;j++)
	sum[i][j]=a[i-1][j]&&a[i][j-1]&&a[i][j]&&a[i][j+1]&&a[i+1][j];
	for(i=1;i<=n;i++)
	for(j=1;j<=m;j++)
	sum[i][j]+=sum[i][j-1];
	for(l=2;l<m;l++)
	for(r=l;r<m;r++)
	{
		memset(s,0,sizeof(s));
		for(i=2,j=0;i<n;i++)
		{
			s[i]=s[i-1]+sum[i][r]-sum[i][l-1];
			while(s[i]-s[j]>=k)j++;
			if(j)ans+=j-1;
		}
	}
	cout<<ans;
}
