#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;

const int N = 505;

char g[N][N];
int f[N][N];

int main() {
	//freopen("in.txt", "r", stdin);
	//freopen("out.txt", "w", stdout);

	int n, m, K;
	cin >> n >> m >> K;
	for (int i = 1; i <= n; i++) {
		scanf("%s", g[i] + 1);
	}
	for (int i = 2; i < n; i++) {
		for (int j = 2; j < m; j++) {
			if (g[i][j] == '1' && g[i - 1][j] == '1' && g[i + 1][j] == '1' && g[i][j - 1] == '1' && g[i][j + 1] == '1') {
				f[i][j]++;
			}
			f[i][j] += f[i - 1][j] + f[i][j - 1] - f[i - 1][j - 1];
		}
	}
	long long ret = 0;
	for (int i = 2; i < n; i++) {
		for (int j = i; j < n; j++) {
			int u = 1;
			for (int k = 2; k < m; k++) {
				while (u < k && f[j][k] - f[j][u] - (f[i - 1][k] - f[i - 1][u]) >= K) {
					u++;
				}
				ret += (u - 1);
			}
		}
	}
	cout << ret << endl;
}
