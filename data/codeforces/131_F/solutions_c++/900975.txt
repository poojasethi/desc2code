#include<iostream>
#include<cstring>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
#define rp(i,l,r) for ( int i=(l); i<=(r); ++i)
#define dp(i,l,r) for ( int i=(l); i>=(r); --i)
#define mn 510

int n,m,k;
long long ans;
char s[mn];
bool map[mn][mn],v[mn][mn];
int sv[mn][mn],h[mn];

int main()
{
    ans=0;
    scanf("%d%d%d%*c",&n,&m,&k);
    rp( i,1,n )
    {
        gets(s);
        rp( j,1,m ) map[i][j]=(s[j-1]=='1');
        } // 0 - 0
    rp( i,2,n-1 ) rp( j,2,m-1 ) if ( map[i][j] && map[i][j-1] && map[i][j+1] && map[i-1][j] && map[i+1][j] )
    {
        v[i-1][j-1]=1;
        } // 0 - 0
    n-=2;  m-=2;
    rp( i,1,n ) rp( j,1,m ) sv[i][j]=sv[i][j-1]+v[i][j];
    rp( i,1,m ) rp( j,i,m ) 
    {
        h[0]=0; int now=n;
        rp( o,1,n ) h[o]=sv[o][j]-sv[o][i-1]+h[o-1];
        dp( o,n,1 ) 
        {
            while ( now>=0 && h[o]-h[now]<k ) --now;
            if ( now<0 ) break;
            ans+=now+1;
            } // 0 - 0
    } // 0 - 0
    cout << ans << endl;
   // system("pause");
}
