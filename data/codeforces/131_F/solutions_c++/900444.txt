#include<cstdio>
#include<cstring>
#include<queue>
#include<algorithm>
using namespace std;

int n,m,req;
char g[505][505];
int s[505][505];
int sum[505][505];
int dx[]={1,0,-1,0,0},dy[]={0,1,0,-1,0};

int main() {
	scanf("%d%d%d",&n,&m,&req);
	for (int i=0; i<n; i++)
		scanf("%s",g[i]);
	memset(s,0,sizeof(s));
	for (int i=0; i<n; i++)
		for (int j=0; j<m; j++) {
			int ok=1,nx,ny;
			for (int k=0; k<5; k++) { 
				nx=i+dx[k]; ny=j+dy[k];
				if (nx>=0 && ny>=0 && nx<n && ny<m) {
					if (g[nx][ny]=='0') ok=0;
				}
				else ok=0;
			}
			s[i+1][j+1]=ok;
		}
	
	for (int i=1; i<=m; i++) {
		sum[i][0]=0;
		for (int j=1; j<=n; j++)
			sum[i][j]=sum[i][j-1]+s[j][i];
	}
	
	long long ret=0;
	for (int i=1; i<=n; i++)
		for (int j=i+2; j<=n; j++) {
			int st=2,cnt=0;
			for (int k=3; k<=m; k++) {
				cnt+=sum[k-1][j-1]-sum[k-1][i];
				while (cnt>=req) {
					ret+=m-k+1;
					cnt-=sum[st][j-1]-sum[st][i];
					st++;
				}
			}
			//printf("%d %d: %d\n",i,j,ret);
		}
	printf("%I64d\n",ret);
}
