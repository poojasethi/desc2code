#include<cstdio>
#include<iostream>
using namespace std; 
int n,m,p,sum[510][510],map[510][510],dx[]={0,0,0,-1,1},dy[]={0,1,-1,0,0};
long long ans;
char s[510][510];
int main(){
    scanf("%d%d%d\n",&n,&m,&p);
    for(int i=1;i<=n;++i)
        gets(s[i]+1);
    for(int i=1;i<=n;++i)
        for(int j=1;j<=m;++j){
            int res=0;
            for(int k=0;k<5;++k)
                if(s[i+dx[k]][j+dy[k]]=='1')
                    ++res;
            if(res==5)map[i][j]=1;
        }
    for(int i=2;i<n;++i)
        for(int j=2;j<m;++j)
            sum[i][j]=sum[i][j-1]+sum[i-1][j]-sum[i-1][j-1]+map[i][j];
    for(int i=2;i<n;++i)
        for(int j=i;j<n;++j)
            for(int k=2,l=2;k<m&&l<m;++k){
                while(l<m&&sum[j][l]-sum[i-1][l]-sum[j][k-1]+sum[i-1][k-1]<p)++l;
                ans+=m-l;
            }
    cout<<ans<<endl;
    return 0;
}
