#include <iostream>
using namespace std;
#define MAXN 510

typedef long long int lli;

int N,M,K;
char map[MAXN][MAXN];
int sm[MAXN][MAXN];
int str[MAXN][MAXN];
lli res;

int main(){
  cin >> N >> M >> K;
  for(int i=1;i<=N;i++){
    for(int j=1;j<=M;j++){
      cin >> map[i][j];
      map[i][j]-='0';
    }
  }

  for(int i=1;i<N;i++)
    for(int j=1;j<M;j++){
      if(map[i][j] && map[i-1][j] && map[i+1][j] && map[i][j-1] && map[i][j+1]) str[i][j]=1;
    }

    for(int i=1;i<=N;i++)
      for(int j=1;j<=M;j++)
        sm[i][j]=str[i][j]+sm[i-1][j]+sm[i][j-1]-sm[i-1][j-1];

    int tmp;

    for(int i=1;i<=N;i++)
      for(int j=1;j<=M;j++){
        tmp=1;
        for(int k=j-1;k>=1;k--){
            for(;tmp<i-1 && sm[i-1][j-1]-sm[tmp][j-1]-sm[i-1][k]+sm[tmp][k]>=K;tmp++);
            res+=(lli)tmp-1;
        }
      }

  cout << res << endl;


  return 0;
}
