#include <iostream>
using namespace std;

const int MAXn = 500 + 10;
char s[MAXn][MAXn];
int is[MAXn][MAXn], is2[MAXn][MAXn];

int main()
{
	ios::sync_with_stdio(false);
	int rn, cn, t;
	cin >> rn >> cn >> t;
	if(rn <= 2 || cn <= 2)
	{
		cout << 0 << endl;
		return 0;
	}
	for(int i = 0; i < rn; i++)
		cin >> s[i];
	for(int i = 1; i < rn - 1; i++)
		for(int j = 1; j < cn - 1; j++)
			if(s[i][j]=='1'&&s[i-1][j]=='1'&&s[i+1][j]=='1'&&s[i][j+1]=='1'&&s[i][j-1]=='1')
				is2[i][j] = is[i][j] = true;
	long long count = 0;
	for(int l = 1; l <= cn-2; l++)
	{
		/*for(int i = 1; i < rn-1; i++,cerr<<endl)
			for(int j = 1; j < cn-l;j++)
				cout << is2[i][j];*/
		for(int i = 1; i < cn - l; i++)
		{
			int q = 0;
			int beg = 1;
			for(int j = 1; j < rn-1; j++)
			{
				q += is2[j][i];
				while(q >= t)
				{
					//cerr << "YA" << beg << " " << j << " " << q << " " << beg << " ";//<< endl;
					q -= is2[beg][i];
					beg++;
					count += rn - 1 - j;
					//cerr << count << endl;

				}
			}
		}
		for(int i = 1; i < cn-l;i++)
			for(int j = 1; j < rn-1; j++)
				is2[j][i]+=is[j][i+l];
	}
	cout << count << endl;
	return 0;
}
