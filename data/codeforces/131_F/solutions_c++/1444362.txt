#include <stdio.h>
#include <queue>
#include <string.h>
#include <stack>
#include <algorithm>
#include <map>
using namespace std;
char s[600][600];
long long mtr[600][600];
int main(){
    int n,m,k,c1,c2,sum;
    long long cont;
    scanf("%d%d%d",&n,&m,&k);
    for(int i=1;i<=n;i++)
        scanf("%s",s[i]+1);
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            mtr[i][j]=mtr[i-1][j]+mtr[i][j-1]-mtr[i-1][j-1];
            if(s[i][j]=='1' && s[i-1][j]=='1' && s[i+1][j]=='1'
                    && s[i][j-1]=='1' && s[i][j+1]=='1'){
                mtr[i][j]++;
            }
        }
    }    
    cont=0;
    for(int i=1;i<=n-2;i++){
        for(int j=i+2;j<=n;j++){
            c1=1;
            c2=3;
            while(c1<=c2 && c2<=m){
                sum=mtr[j-1][c2-1]-mtr[i][c2-1]-mtr[j-1][c1]+mtr[i][c1];
                
                if(sum>=k){
                    cont+=m-c2+1;
                    c1++;
                }else
                    c2++;

            }
        }
    }
    printf("%I64d\n",cont);
    return 0;
}
