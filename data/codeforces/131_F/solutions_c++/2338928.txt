#include <algorithm>
#include <stdio.h>
#define  MAXN      504
using    namespace std;
typedef  long long llint;

llint n,m,k,ans;
llint DP[MAXN][MAXN];
char S[MAXN][MAXN];

int main()
{
  scanf("%I64d%I64d%I64d",&n,&m,&k);
  for(llint i=1 ; i<=n ; i++)
    scanf("%s",S[i]+1);
  for(llint i=1 ; i<=n ; i++)
    for(llint j=1 ; j<=m ; j++)
      if(S[i][j]=='1' && S[i][j+1]=='1' && S[i][j-1]=='1' && S[i+1][j]=='1' && S[i-1][j]=='1')
        DP[i][j]=1;
  for(llint i=1 ; i<=n ; i++)
    for(llint j=1 ; j<=m ; j++)
    {
      DP[i][j]+=DP[i-1][j]+DP[i][j-1]-DP[i-1][j-1];
      //printf("DP[%d][%d]=%d\n",i,j,DP[i][j]);
    }
  for(llint i=1 ; i<=n ; i++)
    for(llint j=1 ; j<=m ; j++)
    {
      llint tmp=1;
      for(llint l=j-2 ; l>=1 ; l--)
      {
        for( ; tmp<i-1 && DP[i-1][j-1]-DP[tmp][j-1]-DP[i-1][l]+DP[tmp][l]>=k ; tmp++);
		ans+=tmp-1;
      }
    }
  printf("%I64d\n",ans);
  return 0;
}
