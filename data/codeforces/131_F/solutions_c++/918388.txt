#include <iostream>
#include <algorithm>
#include <string>
#include <cstring>

#define FOR(i,s,n) for ( int i=(s); i<(n); ++i )
#define N 500
#define M 500

using namespace std;
typedef long long int ll;

int n,m,k;
char sky[N][M+1];
int stars[N][M];

int dy[] = { 0, -1,  1,  0,  0 };
int dx[] = { 0,  0,  0,  1, -1 };

int main()
{
	cin >> n >> m >> k;
	FOR(i,0,n) cin >> sky[i];

	FOR(i,1,n-1) FOR(j,1,m-1)
	{
		bool yes = true;
		FOR(k,0,5) if ( !(yes=sky[i+dy[k]][j+dx[k]]=='1') ) break;
		if ( yes ) stars[i][j] = 1;
	}

	FOR(i,1,n) FOR(j,0,m) stars[i][j] += stars[i-1][j];

	ll res = 0;

	FOR(i1,0,n) FOR(i2,i1+2,n)
	{
		ll sum = 0;

		for ( int j1=0,j2=1; j1<m; ++j1 )
		{
			while ( sum<k )
			{
				++j2;
				if ( j2>=m ) break;
				sum += stars[i2-1][j2-1] - stars[i1][j2-1];
			}

			if ( j2>=m ) break;

			res += m-j2;
			sum -= stars[i2-1][j1+1] - stars[i1][j1+1];
		}
	}

	cout << res << endl;

	return 0;
}
