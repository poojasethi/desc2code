#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int c[610][610],sum[610][610],w[610];

int main()
{
	int n,m,limit; scanf("%d%d%d",&n,&m,&limit);
	for (int i=1; i<=n; i++)
		for (int j=1; j<=m; j++)
		{
			char ch; do ch=getchar(); while (ch!='0' && ch!='1');
			if (ch=='1') c[i][j]=1; else c[i][j]=0;
		} 
	for (int i=2; i<=n-1; i++)
		for (int j=2; j<=m-1; j++)
			if (c[i][j] && c[i-1][j] && c[i][j-1] && c[i+1][j] && c[i][j+1])
				sum[i][j]=1;
	for (int i=1; i<=n; i++)
	{
		sum[i][0]=0;
		for (int j=1; j<=m; j++) sum[i][j]+=sum[i][j-1];
	}
	long long final=0;
	for (int l=1; l<=m; l++)
		for (int r=l+2; r<=m; r++)
		{
			w[0]=0; for (int i=1; i<=n; i++) w[i]=w[i-1]+sum[i][r-1]-sum[i][l];
			int up=2;
			for (int i=2; i<=n-1; i++)
			{
				while (w[i]-w[up-1]>=limit) up++;
				final+=up-2;
			}
		}
	printf("%I64d\n",final);
	return 0;
}
