#include <cstdio>
#include <iostream>
#define maxn 510
using namespace std;
typedef long long ll;

int N,M,K;
int ar[maxn][maxn];
int stars[maxn][maxn];
int sum[maxn][maxn];
int sum1[maxn][maxn];
ll res;

int main(){
	scanf("%d%d%d",&N,&M,&K);
	
	char ch;
	
	for(int i=1;i<=N;i++)
		for(int j=1;j<=M;j++){
			scanf(" %c",&ch);
			
			ar[i][j]=ch-'0';
		}
		
	for(int i=2;i<N;i++)
		for(int j=2;j<M;j++)
			if(ar[i-1][j] && ar[i+1][j] && ar[i][j-1] && ar[i][j+1] && ar[i][j])
				stars[i][j]=1;
	
	for(int i=1;i<=N;i++)
		for(int j=1;j<=M;j++)
			sum[i][j]=stars[i][j]+sum[i-1][j]+sum[i][j-1]-sum[i-1][j-1];
	
	int tmp;
	
	for(int i=1;i<=N;i++)
		for(int j=1;j<=M;j++){
			tmp=1;
			for(int k=j-2;k>=1;k--){
				for(;tmp<i-1 && sum[i-1][j-1]-sum[tmp][j-1]-sum[i-1][k]+sum[tmp][k]>=K;tmp++);
				
				res+=(ll)tmp-1;
			}
		}
	
	cout << res << endl;
	return 0;
}
