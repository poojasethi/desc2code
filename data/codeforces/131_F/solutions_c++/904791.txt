#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define ll long long
#define lim 501

ll n,m,k,i,j,K,p,s,res;
char A[lim][lim];
ll M[lim][lim];

int main()
{
  scanf("%I64d%I64d%I64d",&n,&m,&K);
  for(i=0;i<n;i++)
    scanf("%s",A[i]);
  for(i=1;i+1<n;i++)
    for(j=1;j+1<m;j++)
	  if(A[i][j]=='1' && A[i][j+1]=='1' && A[i][j-1]=='1' && A[i+1][j]=='1' && A[i-1][j]=='1')
	    M[i][j]=1;
  for(i=1;i+1<n;i++)
    for(j=0;j<m;j++)
	  M[i][j]+=M[i-1][j];
  for(i=1;i+1<n;i++)
    for(j=i;j+1<n;j++)
	  for(k=1,p=1,s=0;k+1<m;k++)
	  {
	    while(p+1<m && s<K)
		{
	      s+=M[j][p]-M[i-1][p];
	      p++;
		}
		if(s>=K)
		  res+=m-p;
		//printf("%I64d %I64d %I64d %I64d %I64d %I64d \n",i,j,k,p,s,res);
	    s-=M[j][k]-M[i-1][k];
	  }
  printf("%I64d\n",res);
  return 0;
}
