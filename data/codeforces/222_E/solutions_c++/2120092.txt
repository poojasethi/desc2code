using namespace std;

#define VI vector<int>
#define pb push_back
#define debug(x) cout<<#x<<"="<<x<<endl;
#define rep(i,n) for(int i=0;i<(n);++i)
typedef long long ll;

ll n,m,k;
const ll mod=1000000007;
ll tmp[55][55];
void mul(ll a[][55] ,ll b[][55]){
    rep(i,m)rep(j,m){
        tmp[i][j]=0;
        rep(k,m)tmp[i][j]=(tmp[i][j] + a[i][k] * b[k][j])%mod;
    }
    rep(i,m)rep(j,m)a[i][j]=tmp[i][j];
}
int main()
{
    ll ans=0 , a[55][55] , A[55][55];
    cin>>n>>m>>k;
    rep(i,m)rep(j,m)a[i][j]=1;
    rep(i,m)rep(j,m)
        if(i==j)A[i][j]=1;
        else    A[i][j]=0;
    while(k--){
        char s,t;
        cin>>s>>t;
        int x,y ;
        if(s<='z' && s>='a') x=s-'a';
        else                 x=s-'A'+26;
        if(t<='z' && t>='a') y=t-'a';
        else                 y=t-'A'+26;
        a[x][y]=0;
    }
    n--;
    while(n){
        if(n&1) mul(A,a);
        n>>=1;  mul(a,a);
    }
    rep(i,m)rep(j,m)ans=(ans+A[i][j])%mod;
    cout<<ans<<endl;
    return 0;
}
