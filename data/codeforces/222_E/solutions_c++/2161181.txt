#include<stdio.h>
#include<iostream>
#define LL long long
#define P 1000000007
using namespace std;
struct Mat{
	int ma[53][53];
}bit,ans,C;
int m;
int get(char c){
	if(c>='a'&&c<='z')
		return c-'a'+1;
	return c-'A'+27;
}
void Mul(Mat &A,Mat &B){
	int i,j,k;
	for(i=1;i<=m;i++)
		for(j=1;j<=m;j++){
			C.ma[i][j]=0;
			for(k=1;k<=m;k++)
				C.ma[i][j]=(C.ma[i][j]+(1LL)*A.ma[i][k]*B.ma[k][j])%P;
		}
	for(i=1;i<=m;i++)
		for(j=1;j<=m;j++)
			A.ma[i][j]=C.ma[i][j];
}
int main(){
	LL n;
	int k,i,j;
	while(cin>>n>>m>>k){
		for(i=1;i<=m;i++)
			for(j=1;j<=m;j++){
				bit.ma[i][j]=1;
				ans.ma[i][j]=(i==j);
			}
		char str[10];
		for(i=0;i<k;i++){
			scanf("%s",str);
			int x=get(str[0]);
			int y=get(str[1]);
			bit.ma[y][x]=0;
		}
		n--;
		while(n){
			if(n&1)Mul(ans,bit);
			Mul(bit,bit);
			n>>=1;
		}
		LL res=0;
		for(j=1;j<=m;j++)
			for(i=1;i<=m;i++)
				res+=ans.ma[i][j];
		cout<<res%P<<endl;
	}
	return 0;
}
