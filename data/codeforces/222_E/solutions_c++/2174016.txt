#include<iostream>
#include<algorithm>
#define M 1000000007
#define LL long long
using namespace std;

LL m;

LL tran(char c){
	if(c >= 'a') return c-'a';
	else return c-'A'+26;
}

LL c[52][52];
void mul(LL a[52][52], LL b[52][52]){
	LL i, j, k;
	for(i = 0; i < m; i++) for(j = 0; j < m; j++) c[i][j] = 0;
	for(i = 0; i < m; i++) for(j = 0; j < m; j++) for(k = 0; k < m; k++)
		c[i][j] = (c[i][j] + (a[i][k] * b[k][j]) % M) % M;
	for(i = 0; i < m; i++) for(j = 0; j < m; j++) a[i][j] = c[i][j];
}

LL a[52][52], b[52][52];
void mpow(LL n){
	if(n == 0) return;
	mpow(n/2);
	if(n & 1){ mul(a, a); mul(a, b); }
	else mul(a, a);
}

void init(){
	LL i, j;
	for(i = 0; i < m; i++) for(j = 0; j < m; j++) b[i][j] = 1;
	for(i = 0; i < m; i++) for(j = 0; j < m; j++) a[i][j] = (i == j);
}

int main(){
	
	LL n, k, i, j, sum = 0;
	char s[3];
	
	cin >> n >> m >> k;
	init();

	for(i = 0; i < k; i++){
		cin >> s;
		b[tran(s[0])][tran(s[1])] = 0;
	}
	
	mpow(n-1);

	for(i = 0; i < m; i++) for(j = 0; j < m; j++) sum = (sum+a[i][j]) % M;
	cout << sum << endl;
}
