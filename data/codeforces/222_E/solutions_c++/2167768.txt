#include<cstdio>
#include<cstring> 
typedef long long ll;
typedef int arr[52][52];
const int mo=1000000007; 
const ll INF=(ll)1e18; 
arr a,b,c;
ll n,p[52][52]; 
int m,t;
char s[3];
int ch(char x){
    if (x>='a') return x-'a';
    else return x-'A'+26;
}
void Mul(arr a,arr b,arr c) {
    memset(p,0,sizeof(p)); 
    for(int k=0;k<m;k++) 
        for(int i=0;i<m;i++) 
            if (a[i][k]) 
                for(int j=0;j<m;j++) { 
                    p[i][j]+=(ll)a[i][k]*b[k][j]; 
                    if (p[i][j]>INF) p[i][j]%=mo; 
                } 
    for(int i=0;i<m;i++) 
        for(int j=0;j<m;j++) 
            if (p[i][j]>=mo) c[i][j]=p[i][j]%mo;
            else c[i][j]=p[i][j]; 
             
} 
  
int main(){
    scanf("%I64d%d%d",&n,&m,&t);
    for(int i=0;i<m;i++) a[0][i]=1; 
    for(int i=0;i<m;i++) 
        for(int j=0;j<m;j++)
            b[i][j]=1;
    for(int i=0;i<t;i++){
        scanf("%s",s);
        b[ch(s[0])][ch(s[1])]=0;
    }
    for(n--;n;n>>=1){
        if(n&1) Mul(a,b,a); 
        Mul(b,b,b); 
    } 
    for(int i=0;i<m;i++) n+=a[0][i];
    printf("%I64d\n",n%mo);
    return 0; 
}
