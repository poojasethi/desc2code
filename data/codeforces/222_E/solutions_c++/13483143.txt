#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#define mod 1000000007
#define ll long long 
using namespace std;
int d[52][52],e[52][52],mt[52][52],m,k,res,a,b,i,j;
ll n;
char str[2];
void mrg(){
	 int i,j,k,tmp[52][52]={0};
	 for(i=0;i<m;i++)
	 for(j=0;j<m;j++){
		 int res=0;
		 for(k=0;k<m;k++)
		 res=(res+1ll*d[i][k]*d[k][j])%mod;
	     tmp[i][j]=res;
	 }
	 memcpy(d,tmp,sizeof tmp);
}
void crp(){
	 int i,j,k,tmp[52][52]={0};
	 for(i=0;i<m;i++)
	 for(j=0;j<m;j++){
		 int res=0;
		 for(k=0;k<m;k++)
		 res=(res+1ll*d[i][k]*e[k][j])%mod;
	     tmp[i][j]=res;
	 }
	 memcpy(d,tmp,sizeof tmp);
}
void carp(ll x){
	 if(x==1)
	 return ;
	 carp(x/2);
	 mrg();
	 if(x%2)
	 crp();
}
int main(){
	scanf("%lld %d %d",&n,&m,&k);
	if(n==1){
		printf("%d\n",m);
		return 0;
	}
	for(i=1;i<=k;i++){
		scanf("%s",str);
		a=(str[0]>='a' ? str[0]-'a' : str[0]-'A'+26);
		b=(str[1]>='a' ? str[1]-'a' : str[1]-'A'+26);		
		mt[a][b]=true;
	}
	for(i=0;i<m;i++)
	for(j=0;j<m;j++)
	d[i][j]=!mt[j][i];    
	memcpy(e,d,sizeof d);
	carp(n-1);
	for(i=0;i<m;i++)
	for(j=0;j<m;j++)
	res=(res+d[i][j])%mod;
	printf("%d\n",res);
	return 0;
}
