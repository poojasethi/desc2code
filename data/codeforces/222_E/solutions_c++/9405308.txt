#include<bits/stdc++.h>
using namespace std;
typedef long long i64;
#define rep(i,a,b) for(i=a;i<=b;i++)
#define mod 1000000007LL
i64 n,m,k;
struct matrix
{
  i64 val[55][55];
};
matrix mat;
matrix mul(matrix a,matrix b)
{
   int i,j,k;
   matrix c;
   rep(i,1,m)rep(j,1,m)c.val[i][j]=0;
   rep(i,1,m)
    rep(j,1,m)
      rep(k,1,m)
      {
        c.val[i][j]+=(a.val[i][k]*b.val[k][j])%mod;
        c.val[i][j]%=mod;
      }
      return c;
}
matrix power(i64 N)
{
  if(N==1)return mat;
  matrix tmp;
  tmp=power(N/2);
  tmp=mul(tmp,tmp);
  if(N & 1)tmp=mul(tmp,mat);
  return tmp;
}
int main()
{
  cin>>n>>m>>k;
  int i,j;
  rep(i,1,m)rep(j,1,m)mat.val[i][j]=1;
  char ch[500];
  for(char i='a';i<='z';i++)ch[i]=i-'a'+1;
  for(char i='A';i<='Z';i++)ch[i]=i-'A'+27;
  rep(i,1,k)
  {
    string a;
    cin>>a;
    int u=ch[a[0]];
    int v=ch[a[1]];
    mat.val[u][v]=0;
  }
  if(n==1)
  {
     cout<<m;return 0;
  }
  else
  {
    matrix tmp=power(n-1);
    i64 res=0;
    rep(i,1,m)
    {
      rep(j,1,m)res=(res+tmp.val[i][j])%mod;
    }
    cout<<res<<endl;
  }
  return 0;
}
