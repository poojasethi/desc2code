#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
typedef long long ll;
const ll mod= int(1e9+7);
ll N;
int M, K;
char s[3];
struct __data{
	ll a[52][52];
	__data(){
		memset(a, 0, sizeof a);
	}
	ll&operator() (int x, int y){
		return a[x][y];
	}
	ll ans(){
		ll ret= 0ll;
		for(int i= 0;i< M;++ i)
			for(int j= 0;j< M;++ j)
				(ret+= a[i][j])%= mod;
		return ret;
	}
	void show(){
		for(int i= 0;i< M;++ i){
			for(int j= 0;j< M;++ j)
				cout<< a[i][j]<< " ";
			cout<< endl;
		}
		cout<< endl;
	}
}one, ans;
int getnum(char ch){
	if(ch>='a' && ch<='z')
		return ch-'a';
	return ch-'A'+26;
}
__data operator* (__data&a, __data&b){
	__data ret;
	for(int k= 0;k< M;++ k)
		for(int i= 0;i< M;++ i)
			for(int j= 0;j< M;++ j)
				(ret(i, j)+= a(i, k) * b(k, j))%= mod;
	return ret;
}
int main(){
	cin>> N>> M>> K;
	for(int i= 0;i< M;++ i){
		for(int j= 0;j< M;++ j)
			one(i, j)= 1ll;
		ans(i, i)= 1ll;
	}
	while(K --){
		scanf("%s", s);
		one(getnum(*s), getnum(s[1]))= 0;
	}
	-- N;
	for(;N;N>>= 1ll){
//		ans.show();
		if(N&1ll)
			ans= ans*one;
//		ans.show();
		one= one*one;
	}
	cout<< ans.ans()<< endl;
	return 0;
}
