#include <cstring>
#include <iostream>

using namespace std;

typedef long long LL;

const LL MOD = 1000000007;

LL n;
int m,k;
LL a[54][54],c[54][54];

int MAP(char a){return a>='a' ? a-'a'+1 : a-'A'+27;}
void mul(LL a[][54],LL b[][54],LL c[][54]){
	static LL t[54][54];memset(t,0,sizeof(t));
	for(int i=0;i<=m+1;i++)
		for(int j=0;j<=m+1;j++)
			for(int k=0;k<=m+1;k++)
				(t[i][j]+=a[i][k]*b[k][j])%=MOD;
	memcpy(c,t,sizeof(t));
}
void pow(LL t){for(;t;t>>=1,mul(a,a,a)) if(t&1) mul(c,a,c);}
int main()
{
	cin >> n >> m >> k;
	for(int i=1;i<=m;i++) a[0][i]=1,a[i][m+1]=1;
	for(int i=1;i<=m;i++) for(int j=1;j<=m;j++) a[i][j]=1;
	for(int i=0;i<=m+1;i++) c[i][i]=1;
	for(int i=1;i<=k;i++){
		char st[5];cin >> st;
		a[MAP(st[0])][MAP(st[1])]=0;
	}
	pow(n+1);
	cout << c[0][m+1] << endl;
	return 0;
}
