#include<iostream>
#include<cstring>
using namespace std;
typedef long long ll;

#define mod 1000000007

ll a[55][55],b[55][55],n,m;

int val( char c)
{
  if( c>='a' && c<='z' )
   return c-'a';

  return c-'A'+26;
}


void mult( ll a[][55] , ll b[][55] )
{
ll c[55][55];

memset(c,0,sizeof(c) );

for(int i=0;i<m;i++)
 for(int j=0;j<m;j++)
   for(int k=0;k<m;k++)
     c[i][j]=(c[i][j]+a[i][k]*b[k][j] )%mod;

  for(int i=0;i<m;i++)
   for(int j=0;j<m;j++)
    a[i][j]=c[i][j];
}





int main()
{


for(int i=0;i<55;i++)
{ 

   a[0][i]=1;

  for(int j=0;j<55;j++)
    b[i][j]=1;

   
}

int k;
string st;



cin>>n>>m>>k;


for(int i=1;i<=k;i++)
{  cin>>st;

    b[ val(st[0]) ][ val(st[1]) ]=0;
  
}




for(n-- ;n;n=n/2)
 { if( n%2==1 )
   mult(a,b);
 
   mult(b,b);
}

ll tt=0;

for(int i=0;i<m;i++)
  for(int j=0;j<m;j++)
    tt=(tt+a[i][j] )%mod;

cout<<tt<<endl;

return 0;
}
