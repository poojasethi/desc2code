#include <cstdio>
#define N 53
#define B 1000000007
#define LL long long
#define fi(a, b, c) for(int a = (b); a < (c); a++)

int m, k, A;
LL n, a[N][N], b[N][N];

int cc(char &x){
	if(x < 'a') return x -= 'A' - 26;
	return x -= 'a';
}

void mul(LL a[][N], LL b[][N]){
	LL c[N][N] = {};
	fi(i, 0, m) fi(j, 0, m) fi(k, 0, m) c[i][j] = (c[i][j] + a[i][k] * b[k][j]) % B;
	fi(i, 0, m) fi(j, 0, m) a[i][j] = c[i][j];
}

int main(){
	scanf("%I64d %d %d", &n, &m, &k);
	fi(i, 0, m) fi(j, 0, m) b[i][j] = 1;
    fi(i, 0, m) a[0][i] = 1;
	
	while(k--){
		char x, y;
		scanf("\n%c%c", &x, &y);
		b[cc(x)][cc(y)] = 0;
	}
	
	for(n--; n; n >>= 1){
		if(n & 1) mul(a, b);
		mul(b, b);
	}
	
	fi(i, 0, m) fi(j, 0, m) A = (A + a[i][j]) % B;
	printf("%d\n", A);
}
