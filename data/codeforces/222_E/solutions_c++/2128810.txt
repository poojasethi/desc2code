#include <cstdio>
#include <cctype>
#include <cstring>
const long long MOD = 1000000007;
int m, k, ans;
char s[2];
long long M[55][55], n, X[55][55], cnt;
int Get_index ( char x ) {
        if ( islower ( x ) ) return x - 'a';
        else return x - 'A' + 26;
}
void Mul ( long long a[55][55], long long b[55][55] ) {
        long long c[55][55] = { };
        int i, j, k;
        for ( i = 0; i < m; ++i ) for ( j = 0; j < m; ++j ) for ( k = 0; k < m; ++k ) c[i][j] = ( c[i][j] + a[i][k] * b[k][j] ) % MOD;
        for ( i = 0; i < m; ++i ) for ( j = 0; j < m; ++j ) a[i][j] = c[i][j];
}
int main ( ) {
        int i, j, x, y;
        scanf ( "%I64d%d%d", &n, &m, &k );
        for ( i = 0; i < m; ++i ) for ( j = 0; j < m; ++j ) M[i][j] = 1;
        for ( i = 0; i < k; ++i ) scanf ( "%s", s ), M[Get_index ( s[0] )][Get_index ( s[1] )] = 0;
        for ( i = 0; i < m; ++i ) X[i][i] = 1;
        for ( n--; n; n >>= 1 ) {
                if ( n & 1 ) Mul ( X, M );
                Mul ( M, M );
        }
        for ( i = 0; i < m; ++i ) for ( j = 0; j < m; ++j ) ans = ( ans + X[i][j] ) % MOD;
        printf ( "%d\n", ans );
}
