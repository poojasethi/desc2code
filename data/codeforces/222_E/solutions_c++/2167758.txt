#include<cstdio>
#include<cstring> 
const int mo=1000000007; 
typedef long long ll;
typedef int arr[52][52];
arr a,b,c,p;
ll n;
int m,t;
char s[3];
int ch(char x){
    if (x>='a') return x-'a';
    else return x-'A'+26;
}
void Mul(arr a,arr b,arr c) {
    memset(p,0,sizeof(p)); 
    for(int k=0;k<m;k++) 
        for(int i=0;i<m;i++) 
            if (a[i][k]) 
                for(int j=0;j<m;j++) 
                    p[i][j]=(p[i][j]+(ll)a[i][k]*b[k][j])%mo; 
    memcpy(c,p,sizeof(p)); 
} 
  
int main(){
    scanf("%I64d%d%d",&n,&m,&t);
    for(int i=0;i<m;i++) a[0][i]=1; 
    for(int i=0;i<m;i++) 
        for(int j=0;j<m;j++)
            b[i][j]=1;
    for(int i=0;i<t;i++){
        scanf("%s",s);
        b[ch(s[0])][ch(s[1])]=0;
    }
    for(n--;n;n>>=1){
        if(n&1) Mul(a,b,a); 
        Mul(b,b,b); 
    } 
    for(int i=0;i<m;i++) n+=a[0][i];
    printf("%I64d\n",n%mo);
    return 0; 
}
