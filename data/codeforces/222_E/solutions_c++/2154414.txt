#include <iostream>
#include <string>
using namespace std;

#define fr(i,t) for(i = 1; i <= t; i++)
#define nmax 60
#define ll long long
#define mod 1000000007 

ll mat[nmax][nmax],pwr[nmax][nmax],m,k;
int let[256];

void mult(ll a[nmax][nmax],ll b[nmax][nmax])
{
	ll c[nmax][nmax];
	int i,j,p;
	fr(i,m) fr(j,m) c[i][j] = 0;
	fr(i,m)fr(j,m){
		fr(p,m)
			c[i][j] = (c[i][j] + a[i][p] * b[p][j])%mod;
	}
	fr(i,m) fr(j,m) a[i][j] = c[i][j];
}

int main()
{
	int i,j;
	string s;
	ll n;
	cin>>n>>m>>k;
	n--;
	for(i = 'a'; i <= 'z'; i++)
		let[i] = i - 'a' + 1;
	for(i = 'A'; i <= 'Z'; i++)
		let[i] = 27 + i - 'A';
	fr(i,m)
		mat[i][i] = 1;
	fr(i,m) fr(j,m) pwr[i][j] = 1;
	fr(i,k)
		{
			cin>>s;
			pwr[let[s[0]]][let[s[1]]] = 0;
		}
	while(n)
	{
		if(n & 1)
			mult(mat,pwr);
		n >>= 1;
		mult(pwr,pwr);
	}	
	ll ans = 0;
	fr(i,m)fr(j,m)
		ans = (ans + mat[i][j])%mod;
	cout<<ans<<'\n';
	return 0;
}
