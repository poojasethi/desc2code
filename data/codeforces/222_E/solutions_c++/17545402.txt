#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
typedef long long ll;
typedef vector<vector<int> > mat;

ll n;
int m, k;
char s[3];

mat mult(mat a, mat b) {
	mat c(m, vector<int>(m, 0));
	for(int i=0 ; i<m ; ++i)
		for(int j=0 ; j<m ; ++j)
			for(int k=0 ; k<m ; ++k)
				c[i][j] = (c[i][j] + (ll)a[i][k] * b[k][j])%mod;
	return c;
}

mat power(mat a, ll p) {
	if(p == 1) return a;
	mat c = power(a, p/2);
	c = mult(c, c);
	if(p & 1)
		c = mult(c, a);
	return c;
}


int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	map<char, int> mp;
	int idx = 0;
	for(char ch = 'a' ; ch <= 'z' ; ++ch)
		mp[ch] = idx++;
	for(char ch = 'A' ; ch <= 'Z' ; ++ch)
		mp[ch] = idx++;
	cin >> n >> m >> k;
	mat x(m, vector<int>(m, 1));
	while(k--) {
		scanf("%s", s);
		x[mp[s[0]]][mp[s[1]]] = 0;
	}
	if(n == 1) return printf("%d\n", m), 0;
	--n;
	x = power(x, n);
	ll ans = 0;
	for(int i=0 ; i<m ; ++i)
		for(int j=0 ; j<m ; ++j)
			ans = (ans + x[i][j])%mod;
	cout << ans << endl;
	return 0;
}
