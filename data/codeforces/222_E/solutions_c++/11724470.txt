#include <iostream>
using namespace std;
#define MOD 1000000007

struct mat{ long long a[52][52];};
long long n,m,k;
mat mult(mat a, mat b)
{
	mat c;
	for(int i=0; i<m; i++)
		for(int j=0; j<m; j++)
		{
			c.a[i][j]=0;
			for(int k=0; k<m; k++)
				c.a[i][j]=(c.a[i][j]+a.a[i][k]*b.a[k][j])%MOD;
		}
	return c;
}

int conv(char a)
{
	int x;
	if(a>='a' && a<='z') x=a-'a';
	if(a>='A' && a<='Z') x=a-'A'+26;
	return x;
}

int main() {
	cin>>n>>m>>k;
	mat id,dp[64];
	for(int i=0; i<m; i++)
	{
		for(int j=0; j<m; j++)
		{
			if(i==j) id.a[i][j]=1;
			else id.a[i][j]=0;
			dp[0].a[i][j]=1;
		}
	}
	while(k--)
	{
		char a,b;
		cin>>a>>b;
		int x=conv(a),y=conv(b);
		dp[0].a[x][y]=0;
	}
	n--;
	if(n&1) id=mult(id,dp[0]);
	for(int i=1; i<64; i++)
	{
		dp[i]=mult(dp[i-1],dp[i-1]);
		if(n&(1LL<<i)) id=mult(id,dp[i]);
	}
	long long ans=0;
	for(int i=0; i<m; i++)
	{
		for(int j=0; j<m; j++)
		{
			ans+=id.a[i][j];
			ans%=MOD;
		}
	}
	cout<<ans<<endl;
}
