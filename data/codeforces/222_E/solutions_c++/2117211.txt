#include<stdio.h>
#include<string.h>

long long ans[55][55],a[55][55],tmp[55][55];
int m;
long long mod=1000000007;

void ppp(long long b)
{
	int i,j,k;
	for(i=1;i<=m;i++)
		ans[i][i]=1;
	while(b)
	{
		if(b&1)
		{
			memset(tmp,0,sizeof(tmp));
			for(i=1;i<=m;i++)
				for(j=1;j<=m;j++)
					for(k=1;k<=m;k++)
						tmp[i][j]+=ans[i][k]*a[k][j],tmp[i][j]%=mod;
			memcpy(ans,tmp,sizeof(tmp));
		}
		b>>=1;
		memset(tmp,0,sizeof(tmp));
		for(i=1;i<=m;i++)
				for(j=1;j<=m;j++)
					for(k=1;k<=m;k++)
						tmp[i][j]+=a[i][k]*a[k][j],tmp[i][j]%=mod;
			memcpy(a,tmp,sizeof(tmp));
	}
}
char s[20];
int main()
{
	long long n;
	int i,j,k;

	scanf("%I64d%d%d",&n,&m,&k);
	for(i=1;i<=52;i++)
		for(j=1;j<=52;j++)
			a[i][j]=1;
	while(k--)
	{
		scanf("%s",s);
		if(s[0]>='a'&&s[0]<='z')
			i=s[0]-'a'+1;
		else
			i=s[0]-'A'+1+26;
		if(s[1]>='a'&&s[1]<='z')
			j=s[1]-'a'+1;
		else
			j=s[1]-'A'+27;
		a[i][j]=0;
	}
	ppp(n-1);
	int aa=0;
	for(i=1;i<=m;i++)
		for(j=1;j<=m;j++)
		{
			aa+=ans[i][j];
			if(aa>=mod)
				aa-=mod;
		}
	printf("%d\n",aa);
}
