#!/usr/bin/env python

import sys

string = sys.stdin.readline().rstrip('\n')
count = 0
skipped = 0
size = len(string)

for char in string:
    if char == ')' and count <= 0:
        skipped += 1
        continue
    if char == ')':
        count -= 1
    else:
        count += 1

print size - count - skipped
