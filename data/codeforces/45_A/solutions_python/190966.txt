#!/usr/bin/python
import sys

a = ['January', 'February', 'March', 'April', 'May', 'June', 'July', \
 'August', 'September', 'October', 'November', 'December']
MaxL = len (a)

s = sys.stdin.readline ().strip ()
n = int (sys.stdin.readline ())
for i in range (MaxL):
	if s == a[i]:
		print a[(i + n) % 12]
