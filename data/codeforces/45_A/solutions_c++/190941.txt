#include <iostream>

using namespace std;

int main()
{
    int k;
    string A[12] = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};
    string s;
    cin >> s;
    cin >> k;
    int i;
    for (i = 0; i < 12 && s != A[i]; ++i);
    cout << A[(i + k % 12) % 12] << endl;
    return 0;
}
