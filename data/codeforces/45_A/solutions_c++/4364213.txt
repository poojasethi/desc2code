#include <iostream>

using namespace std;

int main()
{
    string m,s[]={"January","February","March","April","May",
    "June","July","August","September","October","November","December"};
    int n,t;
    cin >> m >> n;
    for (int i=0;i<12;i++){if (s[i]==m){t=i;break;}}
    t+=n;
    t%=12;
    cout << s[t];
    return 0;
}
