#include <iostream>
#include <cstdio>

using namespace std;

string s;
string t[12] = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};
int k;

int main() {
    cin >> s >> k;
    int a = 0;
    while (t[a] != s) {
        ++a;
    }
    cout << t[(a + k) % 12];
    return 0;
}
