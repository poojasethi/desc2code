#include<iostream>
#include<string>
using namespace std;
int main()
{
	string a,month[13]={"December","January","February","March","April","May","June","July","August","September","October","November","December"};
	cin>>a;
	int n,i;
	cin>>n;
	for(i=1;i<=12;i++)
	{
		if(a==month[i])
		{
			break;
		}
	}
	cout<<month[(i+n)%12];
}
