#include <iostream>
#include <cstdio>

using namespace std;

string month[12]={"January", "February", "March", 
"April", "May", "June", "July", "August", "September", "October", 
"November", "December"};


int main()
{	
	string m;
	int k;
	cin >>m>>k;
	int i=0;
	while (month[i] !=m)
		i++;
	cout << month[(i+k)%12];
	return 0;
}
