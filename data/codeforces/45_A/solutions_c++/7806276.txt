#include <stdio.h>
#include <string.h>

const char *M [] = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};

int main() {
 char s[111];
 int n;
 scanf("%s%d", s, &n);
 int i = 0;
 while (strcmp(M[i], s)) ++i;
 puts(M[(i + n) % 12]);
 return 0;
}
