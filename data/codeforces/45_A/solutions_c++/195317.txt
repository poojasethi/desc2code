#include <iostream>
using namespace std;
int main(){
    #define n 12
    string a[12]={"January", "February", "March", "April", "May", "June",
    "July", "August", "September", "October", "November", "December"};
    int i,d; string s; cin>>s>>d; for(i=0; i<n; i++)if(a[i]==s)break;
    cout<<a[(i+d)%n];
    return 0;
}
