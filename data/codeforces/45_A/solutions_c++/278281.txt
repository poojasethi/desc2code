#include <iostream>
#include <cstdio>
using namespace std;


string year[12] = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};
string s;
int k;

int main()
{
    cin >> s >> k;
    int i = 0;
    while(s != year[i]) i++;
    i += k;
    i = i % 12;
    cout << year[i];
}
