#include <iostream>

using namespace std;

int main ()
{
	int x,n,i;
	string Str,Months[12]={"January","February","March","April","May","June","July","August","September","October","November","December"};
	cin >> Str >> n;
	for(i=0; i<12; i++)
		if(Months[i]==Str) x=i;
	cout << Months[(x+n)%12];
	return 0;
}
