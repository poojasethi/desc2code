#include <string>
#include <iostream>
using namespace std;
string month[] = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};
string a;
int main(){
	int i = 0, n;
	cin >> a >> n;
	while(month[i] != a) i ++;
	cout << month[(i + n) % 12] << endl;
	return 0;
}
