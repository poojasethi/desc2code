#include <iostream>
#include <string>

using namespace std;

string months[12] = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};

int main () {
	string month;
	int n;
	cin >> month >> n;
	for(int i = 0; i < 12; i++) if(months[i]==month) {
		cout << months[(i+n)%12] << endl;
	}
	return 0;
}
