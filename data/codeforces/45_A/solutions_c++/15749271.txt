#include <iostream>

using namespace std;

string month[12] = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};

int32_t main()
{
	string s;
	int k;
	cin >> s >> k;
	int pt;
	int n = 12;
	for (int i = 0; i < n; i++)
		if (month[i] == s)
			pt = i;
	cout << month[(pt + k) % 12] << endl;
}
