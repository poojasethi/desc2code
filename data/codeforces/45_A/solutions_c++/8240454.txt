#include <iostream>
using namespace std;
main(){
    string a[]={"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};
    int i,k;
    string s;
    cin>>s>>k;
    for(i=0;i<12;i++)
        if(s==a[i]) break;
    k%=12;
    k+=i;
    k%=12;
    cout<<a[k];
}
