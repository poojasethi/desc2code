#include<bits/stdc++.h>

using namespace std;

string st[12] = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};
string s;
int n, r;

int main()
{
    cin >> s >> n;
    for(int i = 0; i < 12; i++)
        if(st[i] == s) r = i;
    cout << st[(r + n) % 12];
    return 0;
}
