#include <string.h>
#include <iostream>

using namespace std;

int main()
{
	string m[15] = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"}, curr;
	int k, i;
	cin >> curr >> k;
	for (i = 0; i < 12; i++)
	{
		if (m[i] == curr) break;
	}
	cout << m[(k + i) % 12];
	return 0;
}
