#include<bits/stdc++.h>
using namespace std;
int main(){
	string s[12] = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};
	string t;
	cin >> t;
	int ind = 0,k;
	while (s[ind] != t)
        ind++;
    cin >> k;
	k %= 12;
    ind=(ind+k)%12;
    t=s[ind];
    cout<<t;
}
