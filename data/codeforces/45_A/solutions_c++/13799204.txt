#include <bits/stdc++.h>
using namespace std;
string months[] = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};
int main(){
    string s; cin >> s;
    int n, i = 0; cin >> n;
    while(months[i] != s) i++;
    cout << months[(i+n)%12];
}
