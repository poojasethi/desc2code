#include <iostream>

using namespace std;

int main() {
  string arr[] = {
      "January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December" };
  string s;
  int n;
  cin >> s >> n;
  
  for(int i=0; i<12; ++i) {
  if(arr[i] == s) {
  cout << arr[(i+n)%12];
  }
  }
  
  return 0;
}
