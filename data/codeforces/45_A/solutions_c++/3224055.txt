#include<cstdio>
#include<cstring>
char s[12][20]={"January","February","March","April","May","June","July","August", "September", "October", "November", "December"};
char str[20];
int main()
{
    int k,i;
    scanf("%s",str);
    for(i=0;i<20;i++)if(strcmp(s[i],str)==0)break;
    scanf("%d",&k);
    printf("%s\n",s[(i+k)%12]);
    return 0;
}
