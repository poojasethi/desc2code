#include <iostream>
using namespace std;

int main()
{
	string s[] = { "January", "February", "March" , "April", "May", "June", "July", "August", "September", "October", "November", "December" };
	string in;
	cin >> in;
	int n;
	cin >> n;
	
	int i = 0;
	
	while ( s[i] != in ) {
		i++;
	}
	
	cout << s[(i+n)%12] << endl;
	return 0;
}
