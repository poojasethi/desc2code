#include<iostream>
#include<string>
using namespace std;
int main()
{
      string p,s[12]={"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};
      int n,i;
      cin>>p>>n;
      for(i=0;i<12;i++)if(p==s[i])break;
      n=n+i;
      n=n%12;
      cout<<s[n];
      return 0;
      }
