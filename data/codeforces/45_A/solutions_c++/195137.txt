#include <iostream>
using namespace std;

string months[12] = {"December","January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November"};

int main()
{
	string m;
	int c;
	string result;
	cin >> m >> c;
	for(int i = 0; i < 12; i++)
		if(months[i] == m)
			result = months[(i+c)%12];
	cout <<  result << endl;
}
