#!/usr/bin/python

from collections import deque
from collections import defaultdict as dd

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
a = ia()
d = dd(int)

s = 0
for e in a:
    s += e
    d[s] += 1

ma = -1    
for k in d:
    if d[k]>ma: ma = d[k]

print n - ma
