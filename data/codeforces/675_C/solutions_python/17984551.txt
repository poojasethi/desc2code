import sys
from collections import defaultdict

if __name__ == "__main__":
    sys.stdin.readline()
    line = sys.stdin.readline().strip().split()
    partial_sums = defaultdict(int)
    total = 0
    for element in line:
        total += int(element)
        partial_sums[total] += 1
    print len(line) - max(partial_sums.values())
