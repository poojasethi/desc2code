from collections import defaultdict

def solve(accounts):
	
	n = len(accounts)
	ans = n - 1
	s = 0
	c = defaultdict(int)

	for i in range(0, len(accounts)):
		s += accounts[i]
		c[s] += 1
		ans = min(ans, n - c[s])

	return ans

n = input()
accounts = map(int, raw_input().split())
print solve(accounts)
