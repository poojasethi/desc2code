import sys

if __name__ == "__main__":
    n = int(sys.stdin.readline())
    costs = [int(x) for x in sys.stdin.readline().split()]
    diffs = []

    current = 0
    for i in xrange(0, n):
        current += costs[i]
        diffs.append(current)
    diffs.sort()

    max_count = 0

    prev = None
    current_count = 1

    for i in xrange(0, len(diffs)):
        if diffs[i] == prev:
            current_count += 1
        else:
            current_count = 1
        if current_count > max_count:
            max_count = current_count
        prev = diffs[i]
    print n - max_count
