#include <bits/stdc++.h>

using namespace std;
int n, t;
map<long long, int> d;
int main() {
    cin >> n;
    long long sum = 0;
    int ans = n - 1;
    for (int i = 0; i < n; i++) {
        cin >> t;
        sum += t;
        d[sum]++;
        ans = min(ans, n - d[sum]);
    }
    cout << ans << endl;
}
