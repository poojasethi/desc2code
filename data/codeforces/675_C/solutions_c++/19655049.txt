#include <iostream>
#include <map>
using namespace std;

long long x, s = 0;
map<long long, int> occ;
int n, ans = 1e9;

int main() {
    cin >> n;
    for(int i = 0; i < n; i++) {
        cin >> x;
        s += x;
        occ[s]++;
        ans = min(ans, n - occ[s]);
    }
    cout << ans << endl;
}
