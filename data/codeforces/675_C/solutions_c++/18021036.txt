#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int n, x, ans;
ll s = 0;
map<ll, int> mp;

int main(int argc, char **argv) {
	scanf("%d", &n);
	ans = n;
	for(int i=1 ; i<=n ; ++i){
		scanf("%d", &x);
		s += x;
		++mp[s];
		ans = min(ans, n - mp[s]);
	}
	cout<<ans<<endl;
	return 0;
}
