#include<iostream>
#include<vector>
#include<algorithm>
#include<map>

int main()
{
	std::map<long long,int> A;
	long long n,f,sum;
	std::cin>>n;
	long long ans=n;
	sum=0;
	for(long long i=0;i<n;i++)
	{
		std::cin>>f;
		sum=sum+f;
		A[sum]++;
		ans=std::min(ans,n-A[sum]);
	}
	std::cout<<ans;
	return 0;
}
