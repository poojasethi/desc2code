#include<bits/stdc++.h>

using namespace std;
long long s,n,x;
map<long long,int> a;

int main(){
    ios_base::sync_with_stdio(0); cin.tie(0);
    cin>>n;
    long long mn=n-1;
    for(int i=1;i<=n;++i){
        cin>>x;
        s+=x;
        a[s]++;
        mn=min(mn,n-a[s]);
    }
    cout<<mn;
}
