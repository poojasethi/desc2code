#include <bits/stdc++.h>
using namespace std;

int num;
int n;
long long sum;
int ans;

int main() {
	scanf("%d", &n);
	map<long long, int> d;
	
	ans = n -1;
	for (int i = 0; i < n; i++) {
		scanf("%d", &num);
		sum += num;
		d[sum]++;
		ans = min(ans, n - d[sum]);
	}
	
	printf("%d\n", ans);
	return 0;
}
