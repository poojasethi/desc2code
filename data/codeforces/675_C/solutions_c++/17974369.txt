#define N (1<<17)
#define LL long long
#include <bits/stdc++.h>
using namespace std;

LL a[N];
int n,res;
map<LL,int> mp;

int main()
{
	cin>>n,res=n;
	for(int i=1;i<=n;i++)
		scanf("%I64d",a+i),a[i]+=a[i-1],mp[a[i]]++;
	for(int i=1;i<=n;i++)
		res=min(res,n-mp[a[i]]);
	cout<<res<<endl;
	return 0;
}
