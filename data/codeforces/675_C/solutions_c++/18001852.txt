#include <bits/stdc++.h>
using namespace std;
map<long long, int> Map;
int main()
{
    int n, a, ans = 0;
    long long s = 0;
    cin >> n;
    for(int i = 1; i <= n; i++)
    {
        cin >> a;
        ans = max(ans, ++Map[s += a]);
    }
    cout << n - ans << endl;
    return 0;
}
