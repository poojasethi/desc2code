#include <iostream>
#define forn(i,n) for(int i=0;i<n;++i)
int n,m,c[20][20],a,b;
long long ans[19][1<<19],ret;
int main()
{
   std::cin>>n>>m;
   forn(i,m)    
   {
     std::cin>>a>>b;
     c[--a][--b]=c[b][a]=true;
   }
   ret-=m;
   forn(i,n)
     {
       forn(j,i)
         ans[j][(1<<i)|(1<<j)]=c[i][j];
       for(m=(1<<i)|1;m<(1<<(i+1));++m)
         forn(j,i)
         {
           if(!(m>>j&1) || ans[j][m]==0)
             continue;
           forn(k,i)
             if (!(m>>k&1) && c[j][k])
               ans[k][m|(1<<k)]+=ans[j][m];
           if(c[i][j])
             ret+=ans[j][m];
         }
     }
   std::cout<<ret/2;
}
