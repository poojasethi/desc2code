#include <cstdio>
#include <algorithm>

using namespace std;

bool a[19][19];
long long f[1<<19][19];

int main(void)
{
  int n=0,m=0;
  scanf("%d %d",&n,&m);
  for(int i=1;i<=m;++i)
  {
    int j=0,k=0;
    scanf("%d %d",&j,&k);
    --j;
    --k;
    a[j][k]=true;
    a[k][j]=true;
    ++f[(1<<j)|(1<<k)][max(j,k)];
  }
  long long ans=0;
  for(int i=0;i<1<<n;++i)
  {
    int k=0;
    while(k<n)
    {
      if(((1<<k)&i)>0)
        break;
      ++k;
    }
    for(int j=k+1;j<n;++j)
    {
      if(f[i][j]==0)
        continue;
      if(a[j][k])
        ans+=f[i][j];
      for(int p=k+1;p<n;++p)
        if((((1<<p)&i)==0)&&(a[j][p]))
          f[i|(1<<p)][p]+=f[i][j];
    }
  }
  printf("%I64d\n",(ans-m)/2);
  return 0;
}
