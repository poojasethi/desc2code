#include <stdio.h>
#include <iostream>
using namespace std;
#define rep(i, n) for(int i=0; i<(int)(n); i++)

int n, m;
int g[20][20];
long long dp[1<<19][20];

int main() {
    scanf("%d%d", &n, &m);
    rep(i, m) {
        int a, b;
        scanf("%d%d", &a, &b);
        a--; b--;
        g[a][b] = g[b][a] = 1;
        if(a<b) dp[(1<<a)+(1<<b)][b] = 1;
        else dp[(1<<a)+(1<<b)][a] = 1;
    }
    long long ans = 0;
    int nn=1<<n;
    rep(d, nn) {
        int s=-1;
        rep(e, n) if((d&(1<<e)) && dp[d][e]==0) {
            if(s==-1) { s=e; continue; }
            for(int i=s+1; i<n; i++) if((d&(1<<i)) && g[i][e]) {
                dp[d][e] += dp[d-(1<<e)][i];
            }
            if(g[e][s]) ans += dp[d][e];
        }
    }
    cout << ans/2 << endl;
    return 0;
}
