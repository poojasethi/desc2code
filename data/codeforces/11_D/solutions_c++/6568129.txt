#include<stdio.h>
#include<string.h>
long long f[1<<19][20];
int G[20][20];
int main()
{
    int n,m,i,j,k,l,u,v;
    long long ans=0;
    scanf("%d%d",&n,&m);
    for(i=0;i<m;i++)
    {
        scanf("%d%d",&u,&v);
        u--;v--;
        G[u][v]=G[v][u]=1;
        f[(1<<u)|(1<<v)][u>v?u:v]=1;
    }
    for(i=1;i<(1<<n);i++)
    {
        for(k=0;!(i&(1<<k));k++);
        for(j=k+1;j<n;j++) if(f[i][j])
        {
            if(G[j][k]) ans+=f[i][j];
            for(l=k+1;l<n;l++)
                if(G[l][j] && !(i&(1<<l)))
                    f[i|(1<<l)][l]+=f[i][j];
        }
    }
    printf("%I64d\n",(ans-m)/2);
    return 0;
}
