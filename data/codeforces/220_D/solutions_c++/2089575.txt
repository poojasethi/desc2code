#include<cstdio>
#include<iostream>
#include<cstring>
#define mod 1000000007ll
#define ll long long

using namespace std;

int n,m,c[4];
ll f[4009][4009];

bool check(int a,int b){
    int xx1=a>>1,yy1=a&1,xx2=b>>1,yy2=b&1;    
    return ((xx1&yy2)^(xx2&yy1))==0;    
}
int gcd(int a,int b){
    while(b) b^=a^=b^=a%=b;
    return a;
}
int main(){
    ll res=0;
    cin>>n>>m;
    c[0]=(n/2+1)*(m/2+1);
    c[1]=(n/2+1)*((m+1)/2);
    c[2]=((n+1)/2)*(m/2+1);
    c[3]=((n+1)/2)*((m+1)/2);    
    for(int i=0;i<4;i++)
        for(int j=0;j<4;j++)
            for(int k=0;k<4;k++)
                if(check(i^j,i^k)){
                    int t[4];
                    ll d=1;
                    memset(t,0,sizeof(t));
                    if(++t[i]>c[i]) continue;
                    if(++t[j]>c[j]) continue;
                    if(++t[k]>c[k]) continue;
                    for(int x=0;x<4;x++)for(int y=0;y<t[x];y++) d=d*(c[x]-y)%mod;                    
                    res+=d;
                }
    res%=mod;    
    for(int i=1;i<=n+1;i++)
        for(int j=1;j<=m+1;j++){
            f[i][j]=f[i-1][j]+f[i][j-1]-f[i-1][j-1];
            if(i==1&&j==1) continue;
            f[i][j]+=gcd(i-1,j-1)-1;
            f[i][j]%=mod;
            ll t=f[i][j]*2;
            if(i>2) t-=(i-2)*(i-1)/2;
            if(j>2) t-=(j-2)*(j-1)/2;
            res=(res-t*6)%mod;
        }
    cout<<(res+mod)%mod<<endl;
}
