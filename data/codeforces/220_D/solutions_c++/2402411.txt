#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

const long long mod = 1000000007;
long long ans;
long long n, m;
long long c[2][2];
#define rep(i, n) for (long long i = 0; i < n; ++i)

long long gcd(long long a, long long b) {
    if (a == 0 || b == 0) return a + b;
    while ((a %= b) && (b %= a));
    return a + b;
}

int main() {
  //  freopen("in.txt", "r", stdin);
    scanf("%d%d", &n, &m);
    c[0][0] = (n / 2 + 1) * (m / 2 + 1);
    c[0][1] = (n / 2 + 1) *((m + 1) / 2);
    c[1][0] = ((n + 1) / 2 ) * (m / 2 + 1);
    c[1][1] = ((n + 1) / 2) * ((m + 1) / 2);
    long long area;
    rep(x1, 2) rep(y1, 2) rep(x2, 2) rep(y2, 2) rep(x3, 2) rep(y3, 2) {
        area = (x2 - x1) * (y3 - y1) - (x3 - x1) * (y2 - y1);
        if (area % 2 == 0) {
            ans = ans + (long long) c[x1][y1] * c[x2][y2] % mod * c[x3][y3] % mod;
            if (ans >= mod) ans -= mod;
        }
    }
    long long tmp = (n + 1) * (m + 1);
    ans = ans - (long long) tmp % mod;
    if (ans < 0) ans += mod;
    ans = ans - (long long) 3 * tmp * (tmp - 1) % mod;
    if (ans < 0) ans += mod;
    for (long long i = 0; i <= n; ++i)
    for (long long j = 0; j <= m; ++j) {
        if (i == 0 && j == 0) continue;
        tmp = gcd(i, j);
        ans = ans - (long long)6 * (n - i + 1) * (m - j + 1) * (tmp - 1);
        if (ans < 0) ans += mod;
        if (i == 0 || j == 0) continue;
        ans = ans - (long long)6 * (n - i + 1) * (m - j + 1) * (tmp - 1);
        if (ans < 0) ans += mod;
    }
    printf("%d\n", ans);
    return 0;
}
