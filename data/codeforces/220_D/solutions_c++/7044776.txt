#include <cstdio>
#include <cstring>

using namespace std;

typedef long long LL;

const LL MOD = 1000000007LL;

int w,h;
int x[3],y[3];
LL ans=0;

int gcd(int a,int b){return b ? gcd(b,a%b) : a;}
int main()
{
	scanf("%d%d",&w,&h);
	w++,h++;
	for(int i=0;i<8;i++){
		for(int j=0;j<8;j++){
			LL sx=1,sy=1;
			for(int k=0;k<3;k++) {
				x[k]=(i&(1<<k))!=0;
				y[k]=(j&(1<<k))!=0;
				sx=sx*((w+1-x[k])/2)%MOD;
				sy=sy*((h+1-y[k])/2)%MOD;
			}
			int t=x[0]*(y[1]-y[2])+x[1]*(y[2]-y[0])+x[2]*(y[0]-y[1]);
			if(t%2 == 0) ans=(ans+sx*sy)%MOD;
		}
	}
	LL t=w*h;
	ans=(ans-t-t*(t-1)*3)%MOD;
	ans=(ans-t*(h-1)*(h-2)%MOD-t*(w-1)*(w-2)%MOD)%MOD;
	for(int i=1;i<w;i++)
		for(int j=1;j<h;j++)
			ans-=(LL)(gcd(i,j)-1)*(w-i)*(h-j)*12%MOD;
	ans=(ans%MOD+MOD)%MOD;
	printf("%I64d\n",ans);
	return 0;
}
