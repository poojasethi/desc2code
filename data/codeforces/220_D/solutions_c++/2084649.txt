#include <iostream>
#include <cmath>

using namespace std;

typedef long long int lint;

lint modulo=1000000007;

lint mcd(lint a,lint b)
{
  if (a<b) return mcd(b,a);
  if (b==0) return a;
  return mcd(a%b,b);
}

lint cuantos(int p,int tope)
{
  if (p==1) return (tope+1)/2;
  return (tope+2)/2;
}

int main()
{
  lint w,h;
  cin>>w>>h;
  lint count=0;
  for (int c=0;c<16;c++) {
    int x1=c&1;
    int x2=(c&2)/2;
    int x3=(c&4)/4;
    int x4=(c&8)/8;
    int p=(max(x1,x2)+max(x3,x4)+max((x1+x3)%2,(x2+x4)%2))%2;
    if (p==0) {
      for (lint pi=0;pi<2;pi++) {
	for (lint pj=0;pj<2;pj++) {
	  lint cp1=cuantos((x1+pi)%2,h)*cuantos((x2+pj)%2,w);
	  if (x1==0 and x2==0) cp1--;
	  lint cp2=cuantos((x3+pi)%2,h)*cuantos((x4+pj)%2,w);
	  if (x3==0 and x4==0) cp2--;
	  lint adicion;
	  if (x1==x3 and x2==x4)
	    adicion=(cp1*cp2-cp1)%modulo;	    
	  else
	    adicion=(cp1*cp2)%modulo;
	  count+=cuantos(pi,h)*cuantos(pj,w)*adicion;
	}
      }
    }
  }
  count%=modulo;
  lint degen=0;
  for (lint i=1;i<=h;i++)
    for (lint j=1;j<=w;j++)
      degen+=12*(mcd(i,j)-1)*(h-i+1)*(w-j+1);
  degen=(degen+(w+1)*w*(w-1)*(h+1)+(h+1)*h*(h-1)*(w+1))%modulo;

  cout<<(count+modulo-degen)%modulo<<endl;
}
