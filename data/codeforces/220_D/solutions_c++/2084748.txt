
#include <cstdio>
#include <iostream>

typedef long long LL;
const LL Mod = 1000000007LL;

LL a[2][2];

int gcd(int a,int b){return b==0?a:gcd(b,a%b); }

LL C2(LL n)
{
    LL res=n*(n-1)/2;
    return res%Mod;
}

LL C3(LL n)
{
    LL a=n,b=n-1,c=n-2;
    if(a%2==0)
        a/=2;
    else
        b/=2;

    if(a%3==0)
        a/=3;
    else if(b%3==0)
        b/=3;
    else
        c/=3;
    LL res = (a*b)%Mod; // integer overflow
    res = (res*c)%Mod;
    return res;
}

int main()
{
    int w,h;
    LL zero = 0;

    scanf("%d%d",&w,&h);
    for(int i=0;i<=w;++i){
        for(int j=0;j<=h;++j){
            a[i%2][j%2]++;

            LL t = gcd(i,j);
            if(t>1){
                zero += (LL)(w-i+1)*(h-j+1)*(t-1) * (i>0 && j>0 ? 2 : 1);
                zero %= Mod;
            }
        }
    }

    LL ans=0 ;
    for(int i=0;i<2;++i)for(int j=0;j<2;++j) ans += C3(a[i][j]);

    //printf("single point: ans=%I64d\n",ans);

    for(int i=0;i<2;++i)for(int j=0;j<2;++j)
    for(int ii=0;ii<2;++ii)for(int jj=0;jj<2;++jj){
        if(i!=ii || j!=jj){
            LL t = a[i][j]*C2(a[ii][jj]);
            ans += t%Mod;

            //printf("1-(%d,%d) 2-(%d,%d) -->%I64d\n",i,j,ii,jj,t);
        }
    }

    //printf("res=%I64d zero=%I64d\n",ans,zero);

    ans %= Mod;
    ans = (ans-zero+Mod)%Mod;
    ans = (6*ans)%Mod;


    std::cout<<ans<<std::endl;

    return 0;
}
