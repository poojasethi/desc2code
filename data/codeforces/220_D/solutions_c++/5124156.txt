#include<cstdio>
#include<algorithm>
using namespace std;
int n,m;
long long ans,mo = 1e9+7;
int main(){
    scanf("%d%d",&n,&m);
    n++,m++;
    ans = 1ll*n*m;
    ans = ans*(ans-1)%mo*(ans-2)%mo;
    long long a,b,c,d;
    a = 1ll*(n/2)*(m/2);
    b = 1ll*((n+1)/2)*(m/2);
    c = 1ll*(n/2)*((m+1)/2);
    d = 1ll*((n+1)/2)*((m+1)/2);
    ans -= a*b%mo*c%mo*6; ans %= mo;
    ans -= d*b%mo*c%mo*6; ans %= mo;
    ans -= a*b%mo*d%mo*6; ans %= mo;
    ans -= a*c%mo*d%mo*6; ans %= mo;
    ans -= 1ll*n*m*(n-1)%mo*(n-2)%mo;
    ans -= 1ll*m*n*(m-1)%mo*(m-2)%mo;
    for(int i=1;i<n;i++)
        for(int j=1;j<m;j++)
            ans -= 1ll*(__gcd(i,j)-1)*(n-i)%mo*(m-j)*12%mo, ans %= mo;
    printf("%I64d\n",(ans+mo)%mo);
    return 0;
}
