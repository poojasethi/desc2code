#define MAX(X,Y) ((X)>(Y)?(X):(Y))
#define MIN(X,Y) ((X)<(Y)?(X):(Y))

#define MOD 1000000007
using namespace std;

int gcd(int a,int b){
    return (b==0?a:gcd(b,a%b));
}

int main(){
    int w,h;
    long long ans=0,buf,sub;

    cin>>w>>h;

    for(int i=1;i<=w;i++){
        for(int j=1;j<=h;j++){
    //      cout<<i<<" "<<j<<endl;
            buf=0;
            sub=(i/2)*(j/2);
            if(i*j%2==0) sub=(i-1)*(j-1)-sub;

            buf+=sub*4;// cout<<sub*4<<" ";

            if(i*j%2==0)
                buf+=(i+j-2)*2; //cout<<(i+j-2)*2<<" ";
            
            if(i%2&&j%2){
                sub=((i+1)/2)*((j+1)/2)+(i/2+1)*(j/2+1);

            }else if(!(i%2)&&!(j%2)){
                sub=(i+1)*(j+1);
            }else if(j%2){
                sub=(j+1)*(i/2+1);
            }else{//i%2
                sub=(i+1)*(j/2+1);
            }
            sub=sub-gcd(i,j)-1;

        //  cout<<sub*2<<endl;

            buf+=sub*2;

            buf%=MOD;
            ans+=buf*(w-i+1)*(h-j+1);
            ans%=MOD;
        }
    }
    ans=ans*6%MOD;

    cout<<ans<<endl;
    return 0;
}
