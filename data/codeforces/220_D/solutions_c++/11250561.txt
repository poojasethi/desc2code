#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#define LL long long
#define mod 1000000007
using namespace std;
LL ans=0;
int n,m,s[6];
int gcd(int i,int j){return j?gcd(j,i%j):i;}
int main(){
  scanf("%d %d",&n,&m);
  for(int i=0;i<(1<<6);i++){
    LL cur=1;
    for(int j=0;j<6;j++) s[j]=(i&(1<<j))>0;
    if((s[0]*s[3]+s[0]*s[5]+s[2]*s[1]+s[2]*s[5]+s[4]*s[1]+s[4]*s[3])&1) continue;
    for(int j=0;j<6;j+=2){
      if((s[j]%2==0) && (n%2==0)) cur=(cur*(n/2+1))%mod;
      else cur=(cur*((n+1)/2))%mod;
    }
    for(int j=1;j<6;j+=2){
      if((s[j]%2==0) && (m%2==0)) cur=(cur*(m/2+1))%mod;
      else cur=(cur*((m+1)/2))%mod;
    }
    ans=(ans+cur)%mod;
  }
  ans=(ans-(n+1)*(m+1))%mod;
  for(int i=0;i<=n;i++){
    for(int j=0;j<=m;j++){
      LL cur=gcd(i,j);
      LL minus=(n+1-i)*(m+1-j)*cur*6%mod;
      if(i*j) minus*=2;
      ans=(ans-minus)%mod;
    }
  }
  if(ans<0) ans+=mod;
  printf("%I64d\n",ans);
  return 0;
}
