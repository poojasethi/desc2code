#include<stdio.h>
#include<string.h>
const long long mod=1000000007;
long long n,m;
long long ans;
int gcd(int x,int y){
    if (y==0) return x;
    else return gcd(y,x%y);
}
void pre(){
    int x1,x2,x3,y1,y2,y3;
    long long sx1,sx2,sx3,sy1,sy2,sy3;
    ans=0;
    for (x1=0;x1<2;x1++){
        if (x1==0) sx1=m/2+1;else sx1=(m+1)/2;
    for (x2=0;x2<2;x2++){
        if (x2==0) sx2=m/2+1;else sx2=(m+1)/2;
    for (x3=0;x3<2;x3++){
        if (x3==0) sx3=m/2+1;else sx3=(m+1)/2;
    for (y1=0;y1<2;y1++){
        if (y1==0) sy1=n/2+1;else sy1=(n+1)/2;
    for (y2=0;y2<2;y2++){
        if (y2==0) sy2=n/2+1;else sy2=(n+1)/2;
    for (y3=0;y3<2;y3++){
        if (y3==0) sy3=n/2+1;else sy3=(n+1)/2;
        int dx1=x2-x1,dx2=x3-x1;
        int dy1=y2-y1,dy2=y3-y1;
        int dd=dx1*dy2-dx2*dy1;
        if ((dx1*dy2-dx2*dy1)%2==0){
            ans=ans+sx1*sx2%mod*sx3%mod*sy1%mod*sy2%mod*sy3%mod;
           // printf("%d %d %d %d %d %d\n",x1,y1,x2,y2,x3,y3);
            ans%=mod;
        }
    }}}}}}
}
int main(){
    long long t;
    int i,j;
    scanf("%I64d%I64d",&n,&m);
    pre();
    for (i=0;i<=n;i++){
        for (j=0;j<=m;j++){
            long long d=gcd(i,j)+1;
            if (d==1){
                t=(n+1)*(m+1)%mod;
            }else{
                t=(n-i+1)*(m-j+1)*(d-2)*6%mod;
                t=(t+(n-i+1)*(m-j+1)*6)%mod;
                if (i>0&&j>0) t=(t+t)%mod;
            }
            ans=(ans-t)%mod;
            if (ans<0) ans+=mod;
        }
    }
    printf("%I64d\n",ans);
    return 0;
}
