#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
using namespace std;
int n,m,i,j;
long long ans;
const int MOD=1000000007;
long long work(int x,int y)
{
    long long ans=0;
    if (x%2==0 || y%2==0) ans=x+x+y+y;
    if (x%2==1 && y%2==1) ans+=(x/2)*(y/2)*4;
    if (x%2==0 && y%2==0) ans+=((x-1)*(y-1)-(x/2)*(y/2))*4;
    if (x%2+y%2==1) ans+=((x-1)*(y-1)-(x/2)*(y/2))*4;
    if (x%2==0) ans+=(y-1)*4; else ans+=(y-1)/2*4;
    if (y%2==0) ans+=(x-1)*4; else ans+=(x-1)/2*4;
    ans-=2*(__gcd(x,y)-1);
    if (x%2==1 && y%2==1) ans+=2*(x/2)*(y/2)+2*((x-1)/2)*((y-1)/2);
    if (x%2==0 && y%2==1) ans+=2*(y-1)*((x-1)/2);
    if (x%2==0 && y%2==0) ans+=2*(y-1)*(x-1);
    if (x%2==1 && y%2==0) ans+=2*(x-1)*((y-1)/2);
    return ans;
}
int main()
{
    scanf("%d%d",&n,&m);
    for (i=1; i<=n; i++)
      for (j=1; j<=m; j++)
        ans=(ans+work(i,j)*(n-i+1)%MOD*(m-j+1))%MOD;
    cout<<ans*6%MOD;
    return 0;
}
