#include<iostream>
#include<algorithm>
#include<queue>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<vector>
#define N 4003
#define hash 100003
#define ls p<<1
#define rs p<<1|1
#define lst p<<1,l,mid
#define ll long long
#define mod 1000000007ll
using namespace std;
ll p[2][2];
int g[N][N];
int main()
{
 // freopen("in.txt","r",stdin);
  int x,y,i,j;
  while(~scanf("%d%d",&x,&y))
  {
   x++,y++;
   for(i=0;i<2;i++)
   for(j=0;j<2;j++)
   p[i][j]=((i==1)?x/2:x-x/2)*((j==1)?y/2:y-y/2);
   int a,b,c,d,e,f;
   ll ans=0,zero=0;
   for(a=0;a<2;a++)for(b=0;b<2;b++)
   for(c=0;c<2;c++)for(d=0;d<2;d++)
   for(e=0;e<2;e++)for(f=0;f<2;f++)
   {
    int x1=c-a,y1=d-b,x2=e-a,y2=f-b;
    int s=x1*y2-x2*y1;
    if(s<0)s=-s;
    if(s%2==0)
    {
     ll tmp=(p[a][b]*p[c][d]%mod)*p[e][f]%mod;
     ans=(ans+tmp)%mod;
    }
   }
   ll s=x*y;
   zero+=s;
   zero=(zero+s*(s-1)%mod*3%mod)%mod;
   zero=(zero+x*y%mod*(y-1)%mod*(y-2)%mod)%mod;
   zero=(zero+y*x%mod*(x-1)%mod*(x-2)%mod)%mod;
   for(i=0;i<x;i++)
   for(j=0;j<y;j++)
   {
    if(i==0)g[i][j]=j;
    else if(j==0)g[i][j]=i;
    else if(i>j)g[i][j]=g[i%j][j];
    else g[i][j]=g[i][j%i];
   }
   for(i=1;i<x;i++)
   for(j=1;j<y;j++)
   if(g[i][j]>=2)
   {
    ll tmp=(ll)((g[i][j]-1)*(x-i)%mod*(y-j)*12%mod);
    zero=(zero+tmp)%mod;
   }
   ans=(ans-zero+mod)%mod;
   printf("%lld\n",ans);
  }
  return 0;
}
