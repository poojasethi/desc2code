#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
#define mod 1000000007
LL W,H,w[2],h[2],c[4001][4001];
int main(){
    cin>>W>>H;
    w[1]=(W-1)/2+1,h[1]=(H-1)/2+1;
    w[0]=W-w[1]+1,h[0]=H-h[1]+1;
    for(int i=0;i<=4000;i++) c[0][i]=c[i][0]=i;
    for(int i=1;i<=4000;i++)
       for(int j=1;j<=4000;j++)
       {
           if(i<=j) c[i][j]=c[i][j-i];
           else c[i][j]=c[i-j][j];
       }
   LL ret=0;
   for(int x=0;x<2;x++)
   for(int y=0;y<2;y++)
   for(int x0=0;x0<2;x0++)
   for(int y0=0;y0<2;y0++)
   for(int x1=0;x1<2;x1++)
   for(int y1=0;y1<2;y1++)
   {
       int t=(x0-x)*(y1-y)-(y0-y)*(x1-x);
       if(t%2==0){
           ret=ret+w[x]*w[x0]*w[x1]%mod*(h[y]*h[y0]*h[y1]%mod)%mod;
           if(ret>=mod) ret%=mod;
       }
   }
   LL t=(W+1)*(H+1);
   ret=(ret-t)%mod;
   if(ret<0) ret+=mod;
   ret=(ret-t*(t-1)*3%mod)%mod;
   if(ret<0) ret+=mod;

   for(int i=0;i<=W;i++)
     for(int j=0;j<=H;j++)
     {
         if(i==0&&j==0) continue;
         t=c[i][j];
         t=(W-i+1)*(H-j+1)*6*(t-1)*((i&&j)?2:1)%mod;

         ret=(ret-t)%mod;
         if(ret<0) ret+=mod;
     }
     cout<<ret<<endl;
}
