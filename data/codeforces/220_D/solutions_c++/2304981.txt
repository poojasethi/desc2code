#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

typedef long long ll;

const ll M=1000000007;

int memo[4001][4001];
int gcd(int a,int b){
	if(memo[a][b]) return memo[a][b];
	return memo[a][b]=b?gcd(b,a%b):a;
}

int main(){
	int w,h; scanf("%d%d",&w,&h); w++; h++;

	ll cnt[2][2][2][2]={};
	rep(y,h) rep(x,w) {
		rep(y1,2) rep(x1,2) rep(y2,2) rep(x2,2) {
			int area=(x1-x)*(y2-y)-(x2-x)*(y1-y);
			if(area%2==0) cnt[y1][x1][y2][x2]++;
		}
	}
	ll even=0;
	rep(y1,2) rep(x1,2) rep(y2,2) rep(x2,2) {
		even+=cnt[y1][x1][y2][x2]*((h+1-y1)/2)*((w+1-x1)/2)%M*((h+1-y2)/2)*((w+1-x2)/2);
		even%=M;
	}

	ll bad=0;
	rep(y,h) rep(x,w) if(y>0 || x>0) {
		ll c=(y==0||x==0?1:2); // 軸に平行な線分は一回だけ数える
		bad+=c*6*(h-y)*(w-x)*(gcd(x,y)-1); // 三点が異なるとき
		bad+=c*3*(h-y)*(w-x)*2; // 三つめの点が端点のどちらかに一致するとき
		bad%=M;
	}
	bad+=h*w; // 三点がすべて一致するとき
	bad%=M;

	printf("%d\n",(int)((even-bad+M)%M));

	return 0;
}
