#include <iostream>
#define maxn 4001
#define MOD  1000000007
typedef long long LL;

using namespace std;

int gcd(int a, int b) {
    return b? gcd(b, a % b): a;
}

LL function(int n, int m)
{
    int g = gcd(n, m), ans;
    if(n % 2 && m % 2) {
        ans = 4 * (n / 2) * (m / 2);
    } else {
        ans = 4 * ((n - 1) * (m - 1) - (n / 2) * (m / 2));
        ans += 2 * (n - 1) + 2 * (m - 1);
    }
    int diag = 0;
    if(n % 2 && m % 2) {
        diag = ((n + 1) / 2) * ((m + 1) / 2) + (n / 2 + 1) * (m / 2 + 1);
    } else if(n % 2 && !(m % 2)) {
        diag = (n + 1) * (m / 2 + 1);
    } else if(!(n % 2) && m % 2) {
        diag = (n / 2 + 1) * (m + 1);
    } else {
        diag = (n + 1) * (m + 1);
    }
    ans += (diag - g - 1) * 2;
    return ans;
}

int main()
{
    int n, m;
    cin >> n >> m;
    long long ans = 0;
    for(int i = 1; i <= n; ++i) {
        for(int j = 1; j <= m; ++j) {
            ans = (ans + (n - i + 1) * (m - j + 1) * function(i, j)) % MOD;
        }
    }
    cout << ans * 6 % MOD << endl;
}
