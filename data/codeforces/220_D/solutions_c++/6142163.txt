#include <cstdio>
#include <iostream>
#define Rep(i,j,k) for (int i=(j); i<=(k); i++)
using namespace std;
typedef long long LL;

const int MOD=1000000007;

int n,m,Gcd[4100][4100];

LL C(int n, int d){
	return d?((n+1)/2):(n/2+1);
}

int gcd(int x, int y){
	if (y==0) return x;
	if (Gcd[x][y]) return Gcd[x][y];
	return Gcd[x][y]=gcd(y,x%y);
}

int main(){
	cin>>n>>m;
	
	LL ans=0;
	
	Rep(x,0,1)
		Rep(y,0,1)
			Rep(da,0,1)
				Rep(db,0,1)
					Rep(dc,0,1)
						Rep(dd,0,1)
							if ( ( (x+da)*(y+dd)+(x+dc)*(y+db) ) %2 ==0)
								ans=(C(n,da)*C(m,db)*C(n,dc)*C(m,dd)%MOD*C(n,x)*C(m,y)+ans)%MOD;
	
	for(LL dx=0; dx<=n; dx++)
		for(LL dy=0; dy<=m; dy++){
			LL ret=0;
			if (dx||dy) ret=6; else ret=1;
			ret+=max(0,gcd(dx,dy)-1)*6;
			ret=ret*((n-dx+1)*(m-dy+1))%MOD;
			if (dx && dy) ret<<=1;
			if (ret>=MOD) ret-=MOD;
			ans=ans-ret;
			if (ans<0) ans+=MOD;
		}
	
	cout<<(ans+MOD)%MOD<<endl;

    return 0;
}
