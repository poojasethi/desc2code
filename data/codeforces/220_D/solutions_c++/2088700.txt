#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int N,M;
long long mod=1000000007;
int gcd(int a,int b)
{
    if(b==0) return a;
    else return gcd(b,a%b);
}
int main()
{
    int i,j,k;
    long long a,b,s,u,v;
    scanf("%d%d",&N,&M);
    s=0;
    for(i=1;i<=N;i++)
    {
        for(j=1;j<=M;j++)
        {
            a=i/2;b=j/2;
            if(i%2==0&&j%2==0)
            {
                u=12*a*b+4*a+4*b-4;
                v=(2*a-1)*(2*b-1)-(gcd(i,j)-1);
                u+=2*v;
                s+=u*(N+1-i)*(M+1-j);
            }
            else if(i%2==0&&j%2==1)
            {
                u=(12*a*b+4*a+4*b+4*a-2);
                v=(a-1)*(2*b)-(gcd(i,j)-1);
                u+=2*v;
                s+=u*(N+1-i)*(M+1-j);
            }
            else if(i%2==1&&j%2==0)
            {
                u=(12*a*b+4*b+4*a+4*b-2);
                v=(2*a)*(b-1)-(gcd(i,j)-1);
                u+=2*v;
                s+=u*(N+1-i)*(M+1-j);
            }
            else
            {
                u=4*a*b+4*a+4*b;
                v=2*a*b-(gcd(i,j)-1);
                u+=2*v;
                s+=u*(N+1-i)*(M+1-j);
            }
            s%=mod;
        }
    }s*=6;s%=mod;
    printf("%I64d\n",s);
    return 0;
}
