#!/usr/bin/python

from sys import stdin as cin
from itertools import permutations as perm

l = cin.readlines()
ans = False
for p in perm("ABC"):
	ok = True
	for e in l:
		for i in xrange(3):
			for j in xrange(3):
				if i < j and e[0] == p[i] and e[2] == p[j] and e[1] == ">":
					ok = False	
				if i > j and e[0] == p[i] and e[2] == p[j] and e[1] == "<":
					ok = False	
	if ok:
		ans = True
		print ("%c" * 3) % p

if not ans:
	print "Impossible"
