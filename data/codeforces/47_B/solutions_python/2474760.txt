def main():
    index = {'A': 0, 'B': 0, 'C': 0}
    for i in xrange(3):
        a, sign, b = raw_input()
        key = a if sign == '>' else b
        index[key] += 1

    index = dict([(v, k) for k, v in index.iteritems()])
    try:
        return ''.join([index[0], index[1], index[2]])
    except KeyError:
        return 'Impossible'


if __name__ == '__main__':
    print main()
