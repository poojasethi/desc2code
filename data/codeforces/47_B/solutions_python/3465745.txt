import re

ineq = re.compile("([A-C])([<>])([A-C])")

if __name__ == "__main__":
    coins = {'A': 0, 'B': 0, 'C': 0}
    for i in range(3):
        m = ineq.match(raw_input())
        v1 = m.group(1)
        v2 = m.group(3)
        sign = 1 if m.group(2) == ">" else -1
        coins[v1] += sign
        coins[v2] -= sign
    res = [None] * 3
    for c in coins:
        res[coins[c] / 2 + 1] = c
    if None in res:
        print "Impossible"
    else:
        print ''.join(res)
