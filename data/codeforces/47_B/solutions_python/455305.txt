#!/usr/bin/env python
# -*- coding: utf-8 -*-

def read():
    s = raw_input()
    return s[0]+s[2] if s[1] == '<' else s[2]+s[0]

def test(s, d):
    for t in d:
        if s.find(t[0]) > s.find(t[1]):
            return False
    return True

def main():
    d = [read() for i in xrange(3)]
    t = d[d.index('AB') if d.count('AB') == 1 else d.index('BA')]
    if test('C' + t, d):
        print 'C' + t
    elif test(t[0] + 'C' + t[1], d):
        print t[0] + 'C' + t[1]
    elif test(t + 'C', d):
        print t + 'C'
    else:
        print "Impossible"
    return 0

if __name__ == '__main__':
    main()
