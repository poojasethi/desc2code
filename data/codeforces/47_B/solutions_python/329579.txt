from itertools import *

o = 'Impossible'
L = [raw_input() for i in range(0,3)]
for c in permutations('ABC'):
	l = list(c)
	cnt = 0
	for p in product('ABC',repeat=2):
		s = p[0]+('<' if l.index(p[0]) < l.index(p[1]) else '>')+p[1]
		try:
			L.index(s)
			cnt+=1
		except:
			pass
	if cnt==3:
		o = ''.join(l)
		break
print o
