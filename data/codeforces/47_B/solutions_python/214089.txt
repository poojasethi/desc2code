import sys

dic = {'A':0, 'B':0, 'C':0}
for i in range(0, 3):
	arg = raw_input()
	if arg[1] == '>':
		dic[arg[0]] += 1;
	else:
		dic[arg[2]] += 1;
result = ''
if dic['A'] == dic['B'] or dic['B'] == dic['C'] or dic['A'] == dic['C']:
	print 'Impossible'
	sys.exit(0)
for i in range(0, 3):
	for key in dic.keys():
		if dic[key] == i:
			result += key
print result
