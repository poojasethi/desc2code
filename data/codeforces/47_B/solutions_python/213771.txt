dic = {'A':0, 'B':0, 'C':0}
for i in range(3):
    ex = str(raw_input())
    if ex[1] == '<':
        dic[ex[2]] += 1
    else:
        dic[ex[0]] += 1

res = ""
if dic['A'] == dic['B'] == dic['C'] == 1:
    res = "Impossible"
else:
    for val in range(3):
        for k in dic.keys():
            if dic[k] == val:
                res += k
                break

print res
