ans = ""
dic = {'A':0, 'B':0, 'C':0}
for _ in range(3):
    s = raw_input()
    if s[1] == '>':
        dic[s[0]] += 1
    else:
        dic[s[2]] += 1
if dic['A']==dic['B'] or dic['A']==dic['C'] or dic['B']==dic['C']:
    print "Impossible"
else:
    for i in range(3):
        for key in dic.keys():
            if dic[key] == i:
                ans += key
    print ans
