#include<iostream>
using namespace std;
int main(){
	char a[3]={'A','B','C'};
	int b[3]={0,0,0};
	char arr[4][4];
	for(int i=0;i<3;i++){
			cin>>arr[i];
	}
	for(int i=0;i<3;i++)
		if(arr[i][1]=='<')
			swap(arr[i][0],arr[i][2]);
	for(int i=0;i<3;i++) b[arr[i][0]-'A']++;
	int x[3] = {-1,-1,-1};
	for(int i=2;i>=0;i--) x[b[i]] = i;
//	cout << x[0] << " " << x[1] << " " << x[2] << endl;
	if(x[0]<0||x[1]<0||x[2]<0){
		cout<<"Impossible"<<endl;
		return 0;
	}
	for(int i=0;i<3;i++){
		cout<<a[x[i]];
	}
	cout<<endl;
}
