#include<iostream>
using namespace std;
int main()
{

{cin>>s;

f[s[0]-65]++;
else
f[s[2]-65]++;
}
if(f[0]==f[1]&&f[1]==f[2])
cout<<"Impossible";
else
{
for(int i=0;i<3;i++)
{

}

{
if(f[i]==1)
cout<<(char)(65+i);
}
for(int i=0;i<3;i++)
{

}}

}
