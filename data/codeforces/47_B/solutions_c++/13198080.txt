#include<bits/stdc++.h>
using namespace std;
char s[5][55];
int main()
{
scanf("%s%s%s",&s[0],&s[1],&s[2]);
char b[] = "ABC";
int valid = 0;
while(1)
{
map<int,int> gr;
for(int i = 0;b[i];i++)gr[b[i]] = i;
int i;
int ff = 1;
for(i=0;i<3;i++)
{
int aa,bb,cc;
aa = s[i][0]; bb = s[i][1]; cc = s[i][2];
if(bb == '>' && gr[aa] < gr[cc])ff = 0;
if(bb=='<' && gr[aa] > gr[cc])ff = 0;
}
if(ff){
valid = 1 ;
break;
}
if(!next_permutation(b,b+3))break;
}

if(valid)printf("%s\n",b);
else printf("Impossible\n");

return 0;
}
