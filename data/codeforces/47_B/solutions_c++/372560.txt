#include <iostream>
#include <string>

using namespace std;

int main()
{
    char X, Y, Z;
    int A[3] = {};
    int i = 0;
    string R = "XXX";
    while ( i < 3 )
    {
        cin >> X >> Z >> Y;
        ++A[( ( Z == '>' ) ? ( X ) : ( Y ) ) - 'A'];
        ++i;
    }
    for ( i = 0; i < 3; ++i )
    {
        R[ A[i] ] = i + 'A';
    }
    for ( i = 0; i < 3; ++i )
    {
        if (R[i] == 'X')
        {
            cout << "Impossible";
            return 0;
        };
    }
    cout << R;
    return 0;
}
