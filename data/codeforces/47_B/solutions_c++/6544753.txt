#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int main(){
	char a,b,f,i,j;
	int c[11];
	memset(c,0,sizeof(c));
	for(i = 0; i < 3; i++){
	scanf("%c%c%c",&a,&f,&b);
	getchar();
	if(f == '>')
         c[a - 'A']++;
	else
		c[b - 'A']++;
	}
	if(c[0] == c[1] || c[0] == c[2] || c[1] == c[2])
		printf("Impossible\n");
	else{
		for(i = 0;i < 3;i++)
			for(j = 0; j < 3;j++)
				if(c[j] == i)
					printf("%c",'A' + j);
				printf("\n");
	}
return 0;
}
