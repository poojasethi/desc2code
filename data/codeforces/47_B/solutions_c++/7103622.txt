#include <iostream>
#include <algorithm>
using namespace std;

int main ()
{
    int c[3]={};
    char q[3];
    string z[3];
    cin >> z[0] >> z[1] >> z[2];
        for (int w=0;w<3;w++){
        if ( z[w][1]=='>')     c[z[w][0]-'A']++;
        if ( z[w][1]=='<')     c[z[w][2]-'A']++;
        }
        if (c[0]==c[1] ) {
            cout << "Impossible";
            return 0;
        }
        for (int i=0;i<3;i++){
            q[c[i]]='A'+i;
        }
        cout << q[0] << q[1] << q[2];
}
