#include <iostream>

using namespace std;
int main()
{
	char s[4];
	int sum[4]={0};
  
    for(int i=0;i<3;i++)
    {
        cin>>s;
        if(s[1]=='<') 
		{
			sum[s[2] - 'A']++;
		}
        else 
		{
			sum[s[0] - 'A']++;
		}
    }
    if(sum[0] && sum[1] && sum[2])
    {
        cout<<"Impossible"<<endl;
        return 0;
    }
    for(int i = 0; i < 3; i++)
        for(int j = 0; j < 3; j++)
            if(sum[j] == i) cout<<(char)('A'+j);
    cout<<endl;
    return 0;
}
