#include <iostream>
#include <string>

using namespace std;

int cnt[3],ans[3];

void solve()
{
	if (cnt[0]==cnt[1]||cnt[0]==cnt[2]||cnt[1]==cnt[2])
	{
		cout << "Impossible" << endl;
		return;
	}
	for (int i=0;i<3;i++)
		for (int j=0;j<3;j++)
			if (cnt[j]==i)
				ans[i]=j;
	for (int i=0;i<3;i++)
		cout << char(ans[i]+'A');
}

int main()
{
	for (int i=0;i<3;i++)
	{
		string s;
		cin >> s;
		if (s[1]=='>')
			cnt[s[0]-'A']++;
		else
			cnt[s[2]-'A']++;
	}
	solve();
	return 0;
}
