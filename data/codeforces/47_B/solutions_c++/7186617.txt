#include <iostream>
using namespace std;
bool L2(string &x)
{
	char t;
	if (x[0]==x[2]) t=x[1], x[1]=x[2], x[2]=t;
	else if (x[1]==x[2]) t=x[0], x[0]=x[2], x[2]=t;
	else if (x[0]!=x[1]) return false;
	return true;
}
int main()
{
	string s, a, b;
	int i;
	char t;
	for (i=0; i<3; i++) {
		cin>>s;
		if (s[1]=='<') a+=s[2], b+=s[0];
		else a+=s[0], b+=s[2];
	}
	if (L2(a) && L2(b) && a[2]==b[2]) cout<<b[0]<<a[2]<<a[0]<<endl;
	else cout<<"Impossible"<<endl;
	return 0;
}
