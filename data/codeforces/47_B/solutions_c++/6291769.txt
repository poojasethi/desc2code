#include <iostream>
#include <algorithm>
#include <cstring>
#include <cstdio>
using namespace std;
int cnt[3],ord[3]={0,1,2};
bool cp(int x,int y){
	return cnt[x]<cnt[y];
}
int main(){
	char x,op,y;
	for(int i=0;i<3;i++){
		scanf(" %c%c%c",&x,&op,&y);
		if (op=='>') cnt[x-'A']++; else cnt[y-'A']++;
	}
	sort(ord,ord+3,cp);
	if (cnt[ord[0]]!=0){
		printf("Impossible");
		return 0;
	}
	else
		for(int i=0;i<3;i++)
			putchar('A'+ord[i]);
	return 0;
}
