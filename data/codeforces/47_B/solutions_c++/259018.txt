#include <iostream>
using namespace std;

int main(void){
    string s;
    int t[3] = {0,0,0};

    for(int i=0;i<3;i++){
        cin>>s;
        if(s[1] == '<'){
            t[s[2]-'A']++;
        }
        else{
            t[s[0]-'A']++;
        }
    }

    if(t[0] != t[1] && t[0] != t[2] && t[1] != t[2]){
        char s[4];
        s[t[0]] = 'A';
        s[t[1]] = 'B';
        s[t[2]] = 'C';
        s[3] = 0;
        cout<<s<<endl;
    }
    else{
        cout<<"Impossible\n";
    }

    return 0;
}
