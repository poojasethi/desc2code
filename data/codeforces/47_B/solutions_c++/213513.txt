#include <iostream>
#include <string>
using namespace std;

int a[3];

int main() {

	for (int i=0;i<3;++i)
	{
		string s;
		cin >> s;
		if (s[1] == '>')
			swap(s[0], s[2]);
		++a[s[2] - 'A'];
	}
	int maxt = 0;

	for (int i=1;i<3;++i)
		if (a[i - 1] == a[i])
		{
			cout << "Impossible";
			return 0;
		}

	for (int i=0;i<3;++i)
	{

		for (int j=0;j<3;++j)
		{
			if (a[j] < a[maxt])
				maxt = j;
		}
		cout << (char)(maxt + 'A');
		a[maxt] = 10;
	}

	return 0;
}
