#include <stdio.h>
#include <string.h>

using namespace std;
int main()
{
	int n = 0,score[3];
	char x,y,z,enter;
	memset(score,0,sizeof(score));
	while(n++ < 3)
	{
		scanf("%c%c%c%c",&x,&y,&z,&enter);
		if(y == '>')score[x - 'A']++;
		else score[z - 'A']++;
	}
	if(score[0] == score[1] || score[1] == score[2] || score[2] == score[0])printf("Impossible");
	else
	for(int i = 0,j = 0;j < 3;i++)
	if(score[i] == j){printf("%c",i + 'A');i = -1;j++;}
}
