#include<cstdio>
#include<vector>
#include<queue>
#include<string>
#include<iostream>
using namespace std;
vector<int> order;
int main()
{
  string s;
  order.resize(3);
  for(int i=0;i<3;i++)
  {
  cin>>s;
  if(s[1]=='>') 
  {
   order[s[0]-65]++;
    order[s[2]-65]--;
  }
  else
  {
order[s[0]-65]--;
    order[s[2]-65]++;
  }
  }
  
  if(order[0]==0 && order[1]==0&&order[2]==0)
    cout<<"Impossible\n";
  else
  {
    for(int i=-2;i<=2;i++)
      for(int j=0;j<3;j++)
	if(order[j]==i)
	  cout<<char(j+65);
	
   cout<<endl;	
  }

  return 0;
}
