#include<iostream>
#include<cstdio>

using namespace std;

string str;
int data[3];

int main() {
	for ( int i = 0 ; i < 3 ; i++ ) {
		cin >> str;
		if ( str[1] == '<' ) data[str[2] - 'A']++;
			else data[str[0] - 'A']++;
	}
	
	if ( data[0] == 1 && data[1] == 1 && data[2] == 1 ) {
		printf( "Impossible\n" );
		return 0;
	}
	
	for ( int i = 0 ; i < 3 ; i++ )
		for ( int j = 0 ; j < 3 ; j++ )
			if ( data[j] == i ) printf( "%c" , 'A' + j );
	printf( "\n" );
	return 0;
}
