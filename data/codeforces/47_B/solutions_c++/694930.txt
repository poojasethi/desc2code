#include <cstdio>
#include <iostream>
#include <string>

using namespace std;

int main() {

    int V[3] = {0};
    string s;
    for (int i = 0; i < 3; i++) {
        cin >> s;
        if (s[1]=='>') V[s[0]-'A']++, V[s[2]-'A']--;
        else V[s[0]-'A']--, V[s[2]-'A']++;
    }
    string ret;
    for (int v = -2; v <= 2; v+=2) for (int i = 0; i < 3; i++) if (v == V[i]) {
        ret += string(1,'A'+i);
        break;
    }
    if (ret.size() != 3) puts("Impossible");
    else cout << ret << endl;

    return 0;
}
