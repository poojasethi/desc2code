#include <iostream>
#include <string>
using namespace std;

int main(){
    string s;
    int abc[120] = { 0 };
    for(int x = 0;x<3;x++){
        cin >> s;
        if(s[1] == '<') abc[s[2]]++;
        else abc[s[0]]++;
    }
    if(abc['A'] and abc['B'] and abc['C']){
        cout << "Impossible" << endl;
        return 0;
    }
    for(int y = 0;y < 3; y++) for(int w = 0;w < 3;w++)
        if(abc[w + 'A'] == y) cout << char(w + 'A');
    cout << endl;
    return 0;
}
