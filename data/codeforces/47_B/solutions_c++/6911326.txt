#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
struct node {
	int sum;
	char c;
}f[3];
int cmp (struct node f1,struct node f2) {
	return f1.sum<f2.sum;
}
int main ( ) {
	char c,c1,c2;
	for (int i=0;i<3;i++) {
		f[i].c='A'+i;
		scanf("%c%c%c",&c1,&c,&c2);
		if (c=='>')
			f[c1-'A'].sum++;
		else
			f[c2-'A'].sum++;
			getchar ( );
	}
	sort (f,f+3,cmp);
	if (f[0].sum==f[1].sum||f[1].sum==f[2].sum)
		printf("Impossible\n");
	else
		printf("%c%c%c\n",f[0].c,f[1].c,f[2].c);
	return 0;
}
