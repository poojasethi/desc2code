#include<cstdio>
#include<cstring>
using namespace std;
char a[4];
int v[3];

int main(){
  int i,j;
  for (i=0;i<3;i++){
    scanf("%s",a);
    if (a[1]=='<') v[a[2]-'A']++;
    else v[a[0]-'A']++;
  }
  if (v[0] && v[1] && v[2]) puts("Impossible");
  else {
    for (i=0;i<3;i++)
      for (j=0;j<3;j++)
        if (v[j]==i) putchar('A'+j);
  }
  puts("");
  return 0;
}
