#include<bits/stdc++.h>
using namespace std;
char s[5];
int b[3];
int main()
{
    memset(b,0,sizeof(b));
    for(int i=0;i<3;i++)
    {
        scanf("%s",s);
        if(s[1]=='>')   b[s[0]-'A']++;
        else    b[s[2]-'A']++;
    }
    if(b[0]==1&&b[1]==1&&b[2]==1)
    {
        printf("Impossible\n");
        return 0;
    }
    for(int i=0;i<3;i++)
    {
        for(int j=0;j<3;j++)
        {
            if(b[j]==i)
                printf("%c",j+'A');
        }
    }
    printf(" \n");
    return 0;
}
