#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;
int cnt[3],ord[3]={0,1,2};
bool cp(int a,int b){
	return cnt[a]<cnt[b];
}
int main(){
    char x,op,y;
	for (int i=0;i<3;i++){
		cin>>x>>op>>y; 
		if (op=='>')cnt[x-'A']++;
		if (op=='<')cnt[y-'A']++;
	}
	
	sort(ord,ord+3,cp);
	if (cnt[ord[0]]!=0 || cnt[ord[1]]!=1 || cnt[ord[2]]!=2){printf("Impossible");return 0;}
	for (int i=0;i<3;i++){
		putchar(ord[i]+'A');
	}
	return 0;
}
