#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

string s;
int a[3];
string ans = "ABC";

int findd(int x){
	for(int i = 0; i < 3; i++)
		if(a[i] == x)
			return i;
}

int main(){
	for(int i = 0; i < 3; i++){
		cin >> s;
		if(s[1] == '>')
			swap(s[0], s[2]);
		a[s[2]-'A']++;
	}
	if((a[0] == a[1])&&(a[1] == a[2])){
		cout << "Impossible";
		return 0;
	}

	for(int i = 0; i < 3; i++)
		cout << ans[findd(i)];

	return 0;
}
