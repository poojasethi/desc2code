#include <cstdio>

char line[10];
int mat[3];

int conv(char a){
	return a-'A';
}


int main(){
	mat[0] = mat[1] = mat[2] = 0;
	for (int i = 0; i < 3; i++) 
	{	
		scanf("%s", line);
		if (line[1] == '>') mat[conv(line[0])]++;
		else mat[conv(line[2])]++;
	}
	line[0] = line[1] = line[2] = line[3] = '\0';
	for (int i = 0; i < 3; i++){
		for (int j = 0; j < 3; j++){
			if (mat[j] == i){
				if (line[i] != '\0'){
					printf("Impossible\n");
					return 0;
				}
				line[i] = 'A'+j;
			}
		}
	}
	printf("%s\n", line);
}
