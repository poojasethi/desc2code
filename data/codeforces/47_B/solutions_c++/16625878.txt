#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long

using namespace std;

char s[10];
int cnt[3];

int main(){
  FI(i,1,3){
    cin.getline(s,10);
    if(s[1]=='<') cnt[s[2]-'A']++;
    else cnt[s[0]-'A']++;
  }
  int m=0;
  FI(i,0,2) m=max(m,cnt[i]);
  if(m==2){
    FI(j,0,2) FI(i,0,2) if(cnt[i]==j) printf("%c",i+'A');
    printf("\n");
  }
  else printf("Impossible\n");
  return 0;
}
