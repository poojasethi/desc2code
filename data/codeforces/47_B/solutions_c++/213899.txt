#include <iostream>
#include <string>
using namespace std;

int main()
{
	string str;
	int num[5] ={0};
	for( int i=0 ; i<3 ; i++ )
	{
		cin >> str;
		if( str[1] == '<' )
			num[str[2]-'A']++;
		else
			num[str[0]-'A']++;
	}
	bool print = true;
	int t = 0;
	while(print && t < 3 )
	{
		print = false;
		for( int i=0 ; i<3 ; i++ )
			if( num[i] == t )
			{
				cout << char(i+'A');
				print = true;
				t++;
				break;
			}
	}
	if( t == 3 )
		cout << endl;
	else
		cout << "Impossible" << endl;
	return 0;
}
