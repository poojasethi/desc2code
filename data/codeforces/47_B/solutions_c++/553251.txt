#include <iostream>

using namespace std;

int main(int argc, char* argv[])
{
	int r[3] = {};

	for (int i = 0; i < 3; i++)
	{
		string s;
		cin >> s;
		r[(s[1] == '<' ? s[2] : s[0]) - 'A']++;
	}

	if ((r[0] == 1) && (r[1] == 1) && (r[2] == 1))
		cout << "Impossible" << endl;
	else
	{
		for (int i = 0; i < 3; i++)
		{
			for (int k = 0; k < 3; k++)
			{
				if (r[k] == i)
					cout << (char)('A' + k);
			}
		}
	}

	cout << endl;

	return 0;
}
