#include<stdio.h>
int b[5];
char a[5];
int main()
{
	char q,w,e;
	for(int i=0;i<3;i++)
	{
		scanf("%c%c%c",&q,&w,&e);
		getchar();
		if(w=='<')
		{
			b[e-'A']+=2;
			b[q-'A']++;
		}
		else
		{
			b[q-'A']+=2;
			b[e-'A']++;
		}
	}
	int biao=0;
	for(int i=0;i<3;i++)
	{
		if(b[i]==4&&!a[2])
		a[2]=i+'A';
		else if(b[i]==3&&!a[1])
		a[1]=i+'A';
		else if(b[i]==2&&!a[0])
		a[0]=i+'A';
		else
		biao=1;
	}
	if(biao)
	printf("Impossible");
	else
	{
		for(int i=0;i<3;i++)
		printf("%c",a[i]);
	}
	return 0;
}
