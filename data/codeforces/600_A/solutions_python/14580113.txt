import re

st = raw_input()
tokens = re.split(r"[\,\;]",st)

def is_num(s):
    m  =re.match("([1-9][0-9]*)",s)
    return  (m!= None and len(m.group(0)) == len(s)) or s == "0" 

nums = filter(is_num,tokens)
others = filter(lambda x: not is_num(x),tokens)

print "\"" + ",".join(nums) + "\"" if len(nums) > 0 else "-"
print "\""+ ",".join(others) + "\"" if len(others) > 0 else "-"
