from re import split

def main():
    """TODO: Docstring for main.
    :returns: TODO

    """
    s = split(",|;", raw_input())
    a = []
    b = []
    for token in s:
        if token.isdigit() and str(int(token)) == token:
            a.append(token)
        else:
            b.append(token)

    if a:
        print "\"" + ",".join(a) + "\""
    else:
        print "-"

    if b:
        print "\"" + ",".join(b) + "\""
    else:
        print "-"


if __name__ == "__main__":
    main()
