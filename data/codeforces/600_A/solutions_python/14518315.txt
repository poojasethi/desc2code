import sys

line = sys.stdin.readline().rstrip('\n')
line = line.replace(';', ',')
#line = line.replace('.', ',')
tokens = line.split(',')

a = []
b = []

for token in tokens:
    if token.isdigit() and (token[0] != '0' or len(token) == 1):
        a.append(token)
    else:
        b.append(token)

if len(a) == 0:
    print "-"
else:
    print "\"%s\"" % (",".join(a))

if len(b) == 0:
    print "-"
else:
    print "\"%s\"" % (",".join(b))
