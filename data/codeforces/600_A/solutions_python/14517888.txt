from itertools import chain

s = chain(*map(lambda x:x.split(','), raw_input().split(';')))
a = ''
b = ''

for token in s:
    try:
        x = int(token)
        if token[0] == '0' and len(token) > 1:
            raise Exception()
        a += token + ','
    except:
        b += token + ','

if len(a) == 0:
    print '-'
else:
    print '"' + a[:-1] + '"'

if len(b) == 0:
    print '-'
else:
    print '"' + b[:-1] + '"'
