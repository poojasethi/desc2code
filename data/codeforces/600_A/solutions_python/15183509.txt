import re
if __name__ == "__main__":
  words = re.split("[;,]", raw_input())
  
  a = []
  b = []
  for x in words:
    if re.match("(^[1-9]([0-9])*$)|^0$",x) is not None:
      a.append(x)
    else:
      b.append(x)

  if len(a) is not 0:
    print "\"" +  ",".join(a) + "\""
  else: print "-"
  if len(b) is not 0:
    print "\"" + ",".join(b) + "\""
  else: print "-"
