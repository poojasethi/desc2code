import re

ls = re.split(',|;', raw_input())
ds = []
ss = []

for s in ls:
	if s.isdigit():
		if ord(s[0]) == 48 and len(s) > 1:
			ss.append(s)
		else:
			ds.append(s)
	else:
		ss.append(s)

if len(ds) > 0:
	print '"%s"' % (','.join(ds))
else:
	print '-'

if len(ss) > 0:
	print '"%s"' % (','.join(ss))
else:
	print '-'
