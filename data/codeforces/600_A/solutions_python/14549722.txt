s = str(raw_input())
s = s.replace(";",",")

l = s.split(',')

a = []
b = []
dig = []
for i in range(ord('0'), ord('9') + 1):
	dig.append(i)

for word in l:
	where = 0
	if not len(word):
		b.append(word)
		continue
	for char in word:
		if not ord(char) in dig:
			where = 1
	if where or word != str(int(word)):
		b.append(word)
	else:
		a.append(word)
	
if len(a):
	print '"'+",".join(a)+'"'
else:
	print "-"

if len(b):
	print '"'+",".join(b)+'"'
else:
	print "-"
