import re

s = raw_input()
s = re.split(',|;', s)

num = []
letters = []

for i in s:
	if(len(i) > 1 and (i[0] == '0' or i[0] == '.')):
		letters.append(i)
	elif(not i.isdigit()):
		letters.append(i)
	else:
		num.append(i)

if(len(num) == 0):
	print '-'
else:
	print '\"' + ','.join(num) + '\"'

if(len(letters) == 0):
	print '-'
else:
	print '\"' + ','.join(letters) + '\"'
