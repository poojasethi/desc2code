s = raw_input()
a = []
b = []

import re
s2 = re.split(',|;',s)

for i in range(len(s2)):
    if s2[i].isdigit() and s2[i][0] != "0" or s2[i] == "0":
        a.append(s2[i])
    else:
        b.append(s2[i])
        
print '"%s"'%(','.join(a)) if len(a) else "-"
print '"%s"'%(','.join(b)) if len(b) else "-"
