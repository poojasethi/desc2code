s=raw_input()

s = s.replace(';', ',').split(',')
a = []
b = []

for word in s:
    if not word:
        b.append(word)
    elif word == '0':
        a.append(word)
    elif word[0] == '0':
        b.append(word)
    elif word.isdigit() == True:
        a.append(word)
    else: 
        b.append(word)

if not a:
    print('-')
else:
    a = ','.join(map(str, a)) 
    print('"' + a + '"')    

if not b:
    print('-')
else:
    b = ','.join(map(str, b)) 
    print('"' + b + '"')
