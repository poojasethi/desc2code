import re

__author__ = 'aste'


def main():
    s = raw_input()
    a = []
    b = []
    num_pattern = re.compile("^(0|[1-9]([0-9]+)?)$")
    for token in re.split("[,;]", s):
        if num_pattern.match(token):
            a.append(token)
        else:
            b.append(token)
    print a and "\"%s\"" % ",".join(a) or "-"
    print b and "\"%s\"" % ",".join(b) or "-"

main()
