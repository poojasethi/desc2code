if __file__.endswith('code.py'):
    import sys
    sys.stdin = open('input.txt')

import re

s = re.split('[;,]', raw_input())

a = []
b = []
for t in s:
    if t.isdigit() and str(int(t)) == t:
        a.append(t)
    else:
        b.append(t)

print ('"%s"' % ','.join(a)) if a else '-'
print ('"%s"' % ','.join(b)) if b else '-'
