def str_num(x):
    try:
        return str(int(x))
    except:
        return False
        
w = raw_input().replace(",",";")
w = w.split(";")

a = filter(lambda x: str_num(x) == x, w)
b = filter(lambda x: x not in a, w)

print '"'+",".join(a)+'"' if len(a) > 0 else "-"
print '"'+",".join(b)+'"' if len(b) > 0 else "-"
