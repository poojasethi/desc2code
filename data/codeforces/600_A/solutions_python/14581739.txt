import sys

a = raw_input().replace(';', ',').split(",")
# print a
outa = []
outb = []
for i in a:
    flag = i.isdigit() and i[0] != '0' or i == "0"
    if flag :
        outa.append(i)
    else :
        outb.append(i)
print '"%s"' % (','.join(outa)) if len(outa) else "-"
print '"%s"' % (','.join(outb)) if len(outb) else "-"
