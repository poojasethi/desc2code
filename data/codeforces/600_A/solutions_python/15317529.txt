string = raw_input()
substrs = string.split(',')
subwords = []
for i in substrs:
    subwords.extend(i.split(';'))
dsw = filter(lambda k: k.isdigit() and (k[0] != '0' or k == '0'), subwords)
ndsw = filter(lambda k: not k.isdigit() or (k[0] == '0' and k != '0'), subwords)
print '"%s"' % ','.join(dsw) if len(dsw) else '-'
print '"%s"' % ','.join(ndsw) if len(ndsw) else '-'
