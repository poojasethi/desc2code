def ok(x):
    return x.isdigit() and (x[0] != '0' or len(x) == 1)

str = raw_input().replace(';', ',').split(',')
a = filter( ok, str)
b = filter( lambda x: not ok(x), str)


if len(a) == 0:
    print "-"
else:
    print '"' + ",".join(a) + '"'

if len(b) == 0:
    print "-"
else:
    print '"' + ",".join(b) + '"'
