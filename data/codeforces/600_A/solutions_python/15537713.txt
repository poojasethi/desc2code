def parse(a):
    aux = a.replace(",", ";")
    aux = aux.split(";")
    b = []
    c = []
    for item in aux:
      if not item.isdigit() or (item[0] == '0' and len(item) > 1):
        c.append(item)
      else:
        b.append(item)

    if not b:
      print '-'
    else:
      print "\"" + ",".join(b) + "\""

    if not c:
      print '-'
    else:
      print "\"" + ",".join(c) + "\""

if __name__ == '__main__':
    a = raw_input()
    parse(a)
