import re
s = raw_input()
strings = re.split('[,;]', s)

a = []
b = []

for s in strings:
	try:
		if not str(int(s)) == s:
			raise ValueError()
		a.append(s)
	except ValueError:
		b.append(s)

if not a:
	print '-'
else:
	print '"%s"' % ','.join(a)

if not b:
	print '-'
else:
	print '"%s"' % ','.join(b)
