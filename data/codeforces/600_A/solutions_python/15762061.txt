import re

words = re.split('[,;]', raw_input())
ints = filter(lambda word: word.isdigit() and (word is '0' or word[0] is not '0'), words)
others = filter(lambda word: not (word.isdigit() and (word is '0' or word[0] is not '0')), words)

print '"%s"' % ','.join(ints) if ints else '-'
print '"%s"' % ','.join(others) if others else '-'
