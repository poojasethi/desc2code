import sys

import re

s = sys.stdin.next().strip('\n')
s = re.split("[;,]", s)

l1 = []
l2 = []

for e in s:
    is_int = 1
    try:
        int(e)
    except:
        is_int = 0
    if e and is_int and ((e[0] != '0') or (e == '0')):
        l1.append(e)
    else:
        l2.append(e)

if l1:
    print('"' + ','.join(l1) + '"')
else:
    print('-')

if l2:
    print('"' + ','.join(l2) + '"')
else:
    print('-')
