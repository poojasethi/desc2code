import fileinput
import re

n = list()
o = list()

line = raw_input()
line = line.replace(',', ';')
lines = line.split(';')

for s in lines:
        if s == '0':
                n.append(s)
        elif '.' in s or re.search('[a-zA-Z]', s) is not None or s == '' or s[0] == '0':
                o.append(s)
        else:
                n.append(s)

res1 = '-' if len(n) == 0 else '"' + ','.join(n) + '"'
res2 = '-' if len(o) == 0 else '"' + ','.join(o) + '"'

print(res1)
print(res2)
