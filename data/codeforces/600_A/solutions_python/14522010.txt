a = raw_input()
a = a.replace(';',',')
b = a.split(',')
c = []
d = []
for e in b:
    try:
        tmp = int(e)
        tmpp = str(tmp)
        if tmpp == e:
            c.append(e)
        else:
            d.append(e)
    except (Exception):
        d.append(e)

if len(c) == 0:
    print "-"
else:
    print '"' + ",".join(c) + '"'

if len(d) == 0:
    print "-"
else: 
    print '"' + ",".join(d) + '"'
