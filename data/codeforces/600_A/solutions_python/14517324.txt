ls = s.split(',')
lls = [a.split(';') for a in ls]
res1 = []

for a in lls:
	for b in a:
		try:
			val = int(b)
			if len(b) > 1 and b[0] == '0':
				res2.append(b)
			else:
				res1.append(b)
		except:
			res2.append(b)

	print '"' + ','.join(res1) + '"'
