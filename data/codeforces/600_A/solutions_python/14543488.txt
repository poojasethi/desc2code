import re

words = raw_input()
words = re.split(',|;', words)

A = [word for word in words if word == '0' or (word.isdigit() and str(word)[0]!='0')]
B = [word for word in words if word not in A]
#print words

if len(A) == 0:
    A = '-'
else:
    A = '"' + ','.join(map(str, A)) + '"'
if len(B) == 0:
    B = '-'
else:
    B = '"' + ','.join(B) + '"'

print A
print B
