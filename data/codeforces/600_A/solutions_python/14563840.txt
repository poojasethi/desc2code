import re

def pr(lst):
    if not lst:
        print '-'
        return
    print '"' + ','.join(lst) + '"'

line = raw_input()
line = re.split(',|;', line.strip('\n'))

numbers = []
non_numbers = []
number_regex = re.compile('^[1-9]\d*$')
for word in line:
    if word == '0' or number_regex.match(word):
        numbers.append(word)
    else:
        non_numbers.append(word)

pr(numbers)
pr(non_numbers)
