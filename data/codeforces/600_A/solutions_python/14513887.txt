st = raw_input()
#a = tsplit(st,(';',','))
a = [i.split(';') for i in st.split(',')]
ans = [[],[]]
for j in a:
    for i in j: 
        if i.isdigit() and str(int(i)) == i:
            ans[0].append(str(int(i)))
        else:
            ans[1].append(i)

for i in ans:
    if i == []:
        print '-'
    else:
        print '"%s"' % (','.join(i))
