#include <iostream>
#include <string>
using namespace std;

string s1, s2;

int main() {
	string s;
	cin >> s;
	
	string current;
	bool isNumber = true;
	
	for (size_t i = 0; i < s.size(); i++) {
		char c = s[i];
		if (c == ';' || c == ',') {
			if (isNumber && current.size() > 0)
				s1 += "," + current;
			else
				s2 += "," + current;
			current = "";
			isNumber = true;
		} else {
			if (!isdigit(c))
				isNumber = false;
			if (isdigit(c) && current.size() == 1 && current[0] == '0')
				isNumber = false;
			current += c;
		}
	}
	if (isNumber && current.size() > 0)
		s1 += "," + current;
	else
		s2 += "," + current;
		
	if (s1.size())
		s1[0] = '"';
	if (s2.size())
		s2[0] = '"';
	
	if (s1.size() == 0)
		cout << "-" << endl;
	else
		cout << s1 << '"' << endl;
	
	if (s2.size() == 0)
		cout << "-" << endl;
	else
		cout << s2 << '"' << endl;
}
