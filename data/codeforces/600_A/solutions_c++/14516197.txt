#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
string s, a, b, c;
int main(){
    cin >> s;
    s+=',';
    int flag = 1;
    for( int i=0 ; i<s.size() ; i++ ){
        if( s[i]==','||s[i]==';'){
            if( c==""||c.size()>1&&c[0]=='0' ) flag = 0;
            if( flag == 1 ) a += c+',';
            else b += c+',';
            c = "";
            flag = 1;
        }
        else{
            c += s[i];
            if( !isdigit(s[i]) ) flag = 0;
        }
    }
    if( a!="" ){
        a.erase(--a.end());
        cout << "\"" << a  << "\"" << endl;
    }
    else cout << "-" << endl;
    if( b!="" ){
        b.erase(--b.end());
        cout << "\"" << b  << "\"" << endl;
    }
    else cout << "-" << endl;
    return 0;
}
