#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>

using namespace std;

string s;
vector<string> a, b;

int main() {
	cin >> s;
	s.push_back(',');
	for (int i = 0; i < s.length();) {
		int j;
		bool flag = true;
		for (j = i; j < s.length(); ++j) {
			if (s[j] == ';' || s[j] == ',') break;
			if (!(s[j] >= '0' && s[j] <= '9')) {
				flag = false;
			}
		}
		if (j > i && flag && (i + 1 == j || s[i] != '0')) {
			a.push_back(s.substr(i, j - i));
		} else {
			b.push_back(s.substr(i, j - i));
		}
		i = j + 1;
	}
	if (a.size()) {
		cout << '"';
		for (int i = 0; i < a.size(); ++i) {
			cout << a[i] << (i + 1 == a.size() ? '"' : ',');
		}
		cout << endl;
	} else
		cout << "-\n";
	if (b.size()) {
		cout << '"';
		for (int i = 0; i < b.size(); ++i) {
			cout << b[i] << (i + 1 == b.size() ? '"' : ',');
		}
		cout << endl;
	} else
		cout << "-\n";
}
