#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;
const int N = 2e5 + 5;
char tmp[N], ans1[N], ans2[N];
int main()
{
	int len1 = 0, len2 = 0, len = 0;
	bool all_num = 1, flen1 = 1, flen2 = 1;
	char c;
	do
	{
		c = getchar();
		if (c == ',' || c == ';' || c == '\n' || c == EOF)
		{
			if (len && all_num && !(len > 1 && tmp[0] == '0'))
			{
				if (!flen1) ans1[len1 ++] = ',';
				flen1 = 0;
				for (int i = 0; i < len; ++ i) ans1[len1 ++] = tmp[i];
			}
			else
			{
				if (!flen2) ans2[len2 ++] = ',';
				flen2 = 0;
				for (int i = 0; i < len; ++ i) ans2[len2 ++] = tmp[i];
			}
			len = 0;
			all_num = 1;
		}
		else
		{
			if (c < '0' || c > '9') all_num = 0;
			tmp[len ++] = c;
		}
	}
	while (c != EOF && c != '\n');
	if (!flen1) printf("\"%s\"\n", ans1); else puts("-");
	if (!flen2) printf("\"%s\"\n", ans2); else puts("-");
}
