#include <iostream>

using namespace std;

int main()
{
    string s;
    cin >> s;
    string nbr;
    string word;
    string l;
    bool isnbre=true;
    nbr+="\"";
    word+="\"";
    for (int i=0;i<=(int)s.length();i++)
    {
        if (s[i]==','||s[i]==';'||s[i]=='\0')
        {
            if (l.length()==0||(l[0]=='0' && l.length()>1))
                isnbre=false;
            if (!isnbre)
                word+=l+",";
            else
                nbr+=l+",";
            l.clear();
            isnbre=true;
        }
        else
        {
            if (s[i]>'9'||s[i]<'0')
                isnbre=false;
            l+=s[i];
        }
    }
    nbr[nbr.length()-1]='\"';
    word[word.length()-1]='\"';
    nbr.length() == 1 ? cout << "-\n" : cout <<nbr << endl;
    word.length() == 1 ? cout << "-\n" : cout << word << endl;
    return 0;
}
