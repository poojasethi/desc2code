#include <bits/stdc++.h>

using namespace std;
bool is_number(string s)
{
    if(s.size()==0)return false;
    if(s.size()>=2&&s[0]=='0')return false;
    for(int i=0;i<s.size();i++)
        if(s[i]>'9'||s[i]<'0')return false;
    return true;


}
int main()
{
   string s1="",s2="",s;
   cin>>s;
   string cur="";
   for(int i=0;i<s.size();i++)
   {
      if(s[i]==','||s[i]==';')
      {
          if(is_number(cur))s1+=cur+',';
          else s2+=cur+',';
          cur="";
      }
      else cur+=s[i];
      if(i==s.size()-1)
      {
           if(is_number(cur))s1+=cur+',';
          else s2+=cur+',';

      }

   }
   if(s1.size()==0)cout<<"-"<<endl;
   else cout<<char(34)<<s1.substr(0,s1.size()-1)<<char(34)<<endl;
   if(s2.size()==0)cout<<"-"<<endl;
   else cout<<char(34)<<s2.substr(0,s2.size()-1)<<char(34)<<endl;

    return 0;
}
