#include<iostream>
#include<string>
using namespace std;
string a,b,str,s;
int i,start,ca,cb;
bool good;
bool is_end(int j)
{
	return (s[j]==';' || s[j]==',' || j==s.size());
}
int main()
{
	cin>>s;
	a=a+"\"";
	b=b+"\"";
	while(start<s.size())
	{
		i=start;
		good=!is_end(i);
		str="";
		while (!is_end(i))
		{
			if (s[i]<48 || s[i]>57) good=0;
			else if (s[i]=='0' && i==start && !is_end(i+1)) good=0;
			str+=s[i];
			i++;
		} 
		if (good)
		{
			if (ca) a+=',';
			a+=str;
			ca++;
		}
		else
		{
			if (cb) b+=',';
			b+=str;
			cb++;
		}
		start=i+1;
	}
	if (s[s.size()-1]==',' || s[s.size()-1]==';')
	{
		if (cb) b+=',';
		else cb++;
	} 
	a+="\"";
	b+="\"";
	if (ca) cout<<a<<endl;
	else cout<<"-"<<endl;
	if (cb) cout<<b<<endl;
	else cout<<"-"<<endl;
	return 0;
}
