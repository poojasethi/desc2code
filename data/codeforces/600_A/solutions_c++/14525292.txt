#include <cstdio>
#define M 300000

char s[M],a[M],b[M],c[M];
int x,y,z;

bool lead () {
	for (int i = 0; i < z; i++) {
		if (c[i] == '.' || ('a' <= c[i] && c[i] <= 'z') || ('A' <= c[i] && c[i] <= 'Z')) return 1;
	}
	if (z == 1) return 0;
	return c[0] == '0';
}

int main () {
	scanf("%s", s);
	x = y = z = 0;
	for (int i = 0; ; i++) {
		if (s[i] == ';' || s[i] == ',' || !s[i]) {
			if (z == 0 || lead()) {
				for (int j = 0; j < z; j++) {
					b[y+j] = c[j];
				}
				y += z;
				z = 0;
				b[y++] = ',';
			} else {
				for (int j = 0; j < z; j++) {
					a[x+j] = c[j];
				}
				x += z;
				z = 0;
				a[x++] = ',';
			}
		} else {
			c[z++] = s[i];
		}
		if (!s[i]) break;
	}
	if (x) {
		a[--x] = 0;
		printf("\"%s\"\n", a);
	} else printf("-\n");
	if (y) {
		b[--y] = 0;
		printf("\"%s\"\n", b);
	} else printf("-\n");
	return 0;
}
