#include <bits/stdc++.h>
using namespace std;

bool isNumber(string &s) {
	if (s.length() == 0)
		return false;
	if (s == "0")
		return true;
	if (!('1' <= s[0] && s[0] <= '9'))
		return false;
	for (int i = 0; i < (int)s.length(); i++) {
		if (!('0' <= s[i] && s[i] <= '9'))
			return false;
	}
	return true;
}

int main() {
	string S;
	cin >> S;
	S += ',';
	for (int i = 0; i < (int)S.length(); i++) {
		if (S[i] == ';')
			S[i] = ',';
	}
	stringstream ss(S), a, b;
	
	a << '\"';
	b << '\"';
	
	bool emptyA = true, emptyB = true;
	
	string word;
	while (getline(ss, word, ',')) {
		if (isNumber(word)) {
			if (!emptyA) a << ",";
			emptyA = false;
			a << word;
		} else {
			if (!emptyB) b << ",";
			emptyB = false;
			b << word;
		}
	}
	
	a << '\"';
	b << '\"';
	
	cout << ((emptyA) ? "-" : a.str()) << endl;
	cout << ((emptyB) ? "-" : b.str()) << endl;
	return 0;
}
