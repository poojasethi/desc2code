#include<iostream>
#include<string>
using namespace std;
string a = "", b = "";
bool chkNum(string s)
{
	for (int i = 0;i < s.length();i++)
	{
		if (s[i] < '0' || s[i] > '9') return 0;
	}
	return 1;
}
void d(string s)
{
	if (s == "") { b += s + ','; return; }
	if (s[0] == '0')
	{
		if (s.length() == 1 )a += s + ',';
		else b += s + ',';
		return;
	}
	if (chkNum(s))a += s + ',';
	else b += s + ',';
	return;
}
int main()
{
	string s;
	int i, j,r=0;
	cin >> s;
	for (i = 0;i < s.length();i++)
	{
		if ( s[i] == '\,' || s[i] == '\;') { d(s.substr(r, i - r)); r = i + 1; }
	}
	d(s.substr(r));
	if (a!="") {
		a = a.substr(0, a.length() - 1);
		cout << '\"' << a << '\"' << '\n';
	}
	else cout << '-' << '\n';
	if (b!="") {
		b = b.substr(0, b.length() - 1);
		cout << '\"' << b << '\"' << '\n';
	}
	else cout << '-' << '\n';
	//system("pause");
	return 0;
}
