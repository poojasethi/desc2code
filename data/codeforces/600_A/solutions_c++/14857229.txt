#include <iostream>
#include <string>
#include <vector>
using namespace std;

string S;
int N;
vector<string> A[2];

int main()
{
	cin >> S;
	N = S.size();

	for (int i = 0; i < N;)
	{
		int j = i;
		bool num = true;
		for (; j < N; j++)
		{
			if (S[j] == ';' || S[j] == ',')
				break;
			if (!('0' <= S[j] && S[j] <= '9'))
				num = false;
		}
		if (j == i)
			num = false;

		if (num && !(j > i + 1 && S[i] == '0'))
			A[0].push_back(S.substr(i, j - i));
		else
			A[1].push_back(S.substr(i, j - i));
		i = j + 1;
	}

	if (S[N - 1] == ',' || S[N - 1] == ';')
		A[1].push_back("");

	for (int i = 0; i < 2; i++)
	{
		if (A[i].empty())
			cout << "-\n";
		else
		{
			cout << "\"";
			for (int j = 0; j < (int)A[i].size() - 1; j++)
				cout << A[i][j] << ",";
			cout << A[i][(int)A[i].size() - 1] << "\"\n";
		}
	}

	return 0;
}
