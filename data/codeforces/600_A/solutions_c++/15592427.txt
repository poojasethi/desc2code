#include <iostream>
#include <string> 
using namespace std;
string s,a,b,str[100005];
int len;
bool isdigit(char x)
{
	return ('0'<=x && x<='9');
}
int main()
{
	ios::sync_with_stdio(false);
	cin>>s;
	s+=',';
	len=(int)s.size();
	a+="\"";
	b+="\"";
	int num;
	num=1;
	for (int i=0;i<len;i++)
		if (s[i]==',' || s[i]==';')
			s[i]=',';
	for (int i=0;i<len;i++)
	{
		if (s[i]==',')
			num++;
		else
			str[num]+=s[i];
	}
	num--;
	bool T1=0,T2=0;
	for (int i=1;i<=num;i++)
	{
		bool f=0;
		for (int j=0;j<(int)str[i].size();j++)
			if (!isdigit(str[i][j])) f=1;
		if (str[i].size()==0) f=1;
		if (str[i].size()>1 && str[i][0]=='0') f=1;
		if (!f)
		{
			if (T1) a+=",";
			T1=1;
			a+=str[i];
		}else
		{
			if (T2) b+=",";
			T2=1;
			b+=str[i];
		}
	}
	a+="\"";
	b+="\"";
	if (!T1) a="-";
	if (!T2) b="-";
	cout<<a<<endl<<b;
	return 0;
}
