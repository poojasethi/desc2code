#include<cstdio>
#include<iostream>
#include<string>

using namespace std;

int main(){
	string s;
	cin >> s;
	string a(""), b(""), t("");
	int i;
	bool isnum = true;
	for(i=0;i<s.size();i++)
	{
		if(s[i]==',' || s[i] ==';')
		{
			if(t.size()==0 || (t.size()>1 && t[0]=='0')) isnum = false;
			//cout << s[i] << t << endl;
			if(isnum) a = a + "," + t;
			else  b = b + "," + t;
			isnum = true;
			t.clear();
		}
		else{
			t += s[i];
			if(!(s[i]>='0' && s[i]<='9'))
				isnum = false;
		}
	}
	if(s[i-1]==',' || s[i-1]==';')
		b += ',';
	else if(t.size() > 0){
		if(t[0]=='0' && t.size()>1) isnum = false;
		if(isnum) a = a + "," + t;
		else  b = b + "," + t;
	}

	if(a.size() == 0)
		cout << "-" << endl;
	else
		cout << "\"" << a.substr(1) << "\""<< endl;
	if(b.size() == 0)
		cout << "-" << endl;
	else
		cout << "\""<< b.substr(1) << "\""<< endl;
	return 0;
}
