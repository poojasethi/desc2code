#include <bits/stdc++.h>
using namespace std;

int main()
{
	string s,temp,num="",ch="";
	cin >> s;
	s+=",";
	int start=0,end;
	for(int i=0;i<s.length();i++)
	{
		if(s[i]==',' || s[i]==';')
		{
			temp=s.substr(start,i-start);
			start=i+1;	
			int flag=0;
			
			if(temp.length()==0)
			{
				ch+=",";flag=1;
			}
			else
			for(int j=0;j<temp.length();j++)
			{
				if(j==0 && temp[j]=='0' && temp[j+1]!='\0')
				{
					ch=ch+temp+",";
					flag=1;
					break;
				}
				else if(((char)temp[j]>='a' && (char)temp[j]<='z')|| ((char)temp[j]>='A' && (char)temp[j]<='Z')|| (char)temp[j]=='.')
				{
					ch+=temp+",";
					flag=1;
					break;
				}
			
			}
			if(flag==0)
			{
				num+=temp+",";
			}	
		}
	}
	if(num=="")
		cout << "-" << endl;
	else
		cout << "\"" <<num.substr(0,num.length()-1)<< "\"" << endl;
	if(ch=="")
		cout << "-" << endl;
	else
		cout << "\"" << ch.substr(0,ch.length()-1) << "\"" << endl;

	return 0;
}
