#include <bits/stdc++.h>
using namespace std;
int main(){
	ios_base::sync_with_stdio(0); cin.tie(0);
	string s; cin >> s; s += ',';
	string cur = "", a = "", b = ""; int yes = 1, cnta= 0, cntb = 0;
	for(int i=0;i<(int)s.length();i++){
		if(s[i] == ',' || s[i] == ';'){
			if(yes && cur.length()>0 && !(cur.length() > 1 && cur[0] == '0')){
				if(cnta) a += ',';
				a += cur; cnta++;
			}
			else{
				if(cntb) b += ',';
				b += cur; cntb++;
			}
			cur = ""; yes = 1;
		} else {
			cur += s[i]; if(s[i] < '0' || s[i] > '9') yes = 0; 
		}
	}
	a = '"' + a + '"'; b = '"' + b + '"';
	if(cnta == 0) a = "-";
	if(cntb == 0) b = "-";
	cout << a << "\n" << b << "\n";
}
