#include <stdio.h>
#include <string.h>

int main (void) {
	char s[100005], ints[100005], nints[100005], curr[100005];
	scanf ("%s", s);
	int len = strlen(s);
	s[len++] = ';';
	int isz = 0, nsz = 0, sz = 0;
	bool isInt = true;

	for (int i = 0; i < len; i++) {
		if (s[i] == ';' || s[i] == ',') {
			if (isInt && sz > 0 && (sz == 1 || curr[0] != '0')) {
				ints[isz++] = ',';
				for (int j = 0; j < sz; j++)	ints[isz++] = curr[j];
			} else {
				nints[nsz++] = ',';
				for (int j = 0; j < sz; j++)	nints[nsz++] = curr[j];
			}
			isInt = true;
			sz = 0;
		} else {
			curr[sz++] = s[i];
			if (s[i] < '0' || s[i] > '9')	isInt = false;
		}
	}
	if (isz == 0) {
		ints[0] = '-';
		isz = 1;
	} else {
		ints[0] = ints[isz++] =  '"';
	}
	if (nsz == 0) {
		nints[0] = '-';
		nsz = 1;
	} else {
		nints[0] = nints[nsz++] = '"';
	}
	ints[isz] = nints[nsz] = '\0';
	printf ("%s\n%s\n", ints, nints);
}
