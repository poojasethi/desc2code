#include<iostream>
#include<vector>
#include<string>
#include<algorithm>
#include<stdio.h>
#include<set>
#include<math.h>
using namespace std;
string s,a,b;
vector<string > c;
int judge(string x) {
    if(x.length()==0) return 0;
    if(x[0]=='0'&&x.length()>1) return 0;
    for(int i=0;i<x.length();i++)
        if(x[i]<'0'||x[i]>'9') return 0;
    return 1;   
}
int main()
{
    cin>>s;
    int pre=0; s+=";"; a="\""; b="\"";
    for(int i=0;i<s.length();i++) {
        //if(s[i]==';') cout<<i<<endl;
        if(s[i]==','||s[i]==';') {
            c.push_back(s.substr(pre,i-pre)),pre=i+1;
        }
    }
    for(int i=0;i<c.size();i++) {
        if(judge(c[i])) a+=c[i],a+=",";
        else b+=c[i],b+=",";
    }
    a[a.length()-1]=a[0];   
    b[b.length()-1]=b[0];
    if(a.length()==1) puts("-");
    else cout<<a<<endl;
    if(b.length()==1) puts("-");
    else cout<<b<<endl;
}
