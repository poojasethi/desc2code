#include <iostream>
#include <vector>
using namespace std;

vector <string> num, oth;

int main()
{
	string s;
	cin>> s;

	for(int i = 0; i < s.size(); i ++)
	{
		string t;
		bool ch = true;
		while(s[i] != ';' && s[i] != ',' && i < s.size())
		{
			t += s[i];
			if('0' > s[i] || s[i] > '9')
				ch = false;
			i ++;
		}
		
		if((t.size() > 1 && t[0] == '0') || (t.size() == 0))
			ch = false;

		if(ch)
			num.push_back(t);
		else
			oth.push_back(t);
	}

	if(s[s.size() - 1] == ';' || s[s.size() - 1] == ',')
		oth.push_back("");

	if(num.size() == 0)
		cout<< "-" << endl;
	else
	{
		cout<< '"';
		for(int i = 0; i < num.size() - 1; i ++)
			cout<< num[i] << ",";
		cout<< num[num.size() - 1] << '"' << endl;
	}

	if(oth.size() == 0)
		cout<< "-" << endl;
	else
	{
		cout<< '"';
		for(int i = 0; i < oth.size() - 1; i ++)
			cout<< oth[i] << ",";
		cout<< oth[oth.size() - 1] << '"' << endl;
	}
}
