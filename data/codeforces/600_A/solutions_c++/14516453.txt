#include<bits/stdc++.h>
using namespace std;
char st[2000100],opt1[2000100],opt2[2000100];
int top,t1,t2;
bool get()
{
	top=0;
	while (1)
	{
		st[++top]=getchar();
		if (st[top]==EOF) return false;
		if ((st[top]==';')||(st[top]==',')||(st[top]=='\n')) {top--;break;}
	}
	return true;
}
bool pend()
{
	if ((top==1)&&(st[1]=='0')) return true;
	if ((st[1]<='0')||(st[1]>'9')) return false;
	for (int i=2;i<=top;i++) if ((st[i]<'0')||(st[i]>'9')) return false;
	return true;
}
int main()
{
	 while (get())
	 {
	 	if (pend()) {for (int i=1;i<=top;i++) opt1[++t1]=st[i];opt1[++t1]=',';}
	 	else {for (int i=1;i<=top;i++) opt2[++t2]=st[i];opt2[++t2]=',';}
	}
	if (t1) {putchar('"');for (int i=1;i<t1;i++) putchar(opt1[i]);putchar('"');}else putchar('-');putchar('\n');
	if (t2) {putchar('"');for (int i=1;i<t2;i++) putchar(opt2[i]);putchar('"');}else putchar('-');putchar('\n');
}
