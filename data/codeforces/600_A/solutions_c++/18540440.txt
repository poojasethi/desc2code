#include <stdio.h>
#include <algorithm>
#include <string.h>
#include <memory.h>

#define NN 100010

using namespace std;

char str[NN], A[NN], B[NN], tmp[NN];

int main (){
	
	scanf("%s",str);
	int N=strlen(str), i;
	int p=0,pa=0,pb=0,fl=0;
	
	for(i=0; i<N; i++){
		while(str[i]!=';' && str[i]!=',' && i<N){
			if(str[i]>'9' || str[i]<'0') fl=1;
			tmp[p++]=str[i++];
		}
		if(!p){
			fl=0;
			B[pb++]=',';
			continue;
		}
		if(fl || tmp[0]=='0' && p>1){
			strncpy(B+pb,tmp,p);
			pb+=p;
			B[pb++]=',';
		}
		else{
			strncpy(A+pa,tmp,p);
			pa+=p;
			A[pa++]=',';
		}
		p=0;
		fl=0;
	}
	
	if(pa) A[pa-1]=0;
	if(pb && str[N-1]!=',' && str[N-1]!=';') B[pb-1]=0;
	if(!pb && (str[N-1]==',' || str[N-1]==';')) pb++;
	
	if(pa) printf("%c%s%c\n",'"',A,'"');
	else puts("-");
	if(pb) printf("%c%s%c\n",'"',B,'"');
	else puts("-");
	
	return 0;
}
