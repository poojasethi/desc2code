#include <iostream>
#include <string>
#include <vector>
using namespace std;
int len;
string s,a,b;
bool ca,cb;
vector<string> vec;
int strpos(int p){
	int res=p;
	while(s[res]!=',' && s[res]!=';' && res<len)res++;
	return res;
}
bool isnum(string& str){
	int l=str.size();if(str[0]=='0' && l>1 || l==0)return false;
	for(int i=0;i<l;i++)if(str[i]<'0' || str[i]>'9')return false;
	return true;
}
int main(){
	getline(cin,s);
	int p=0,q;
	len=s.size();
	q=strpos(p);
	while(q<len){
		vec.push_back(s.substr(p,q-p));
		p=q+1;q=strpos(p);
	}
	vec.push_back(s.substr(p,len-p));
	len=vec.size();
	for(int i=0;i<len;i++){
		if(isnum(vec[i]))a+=vec[i]+',',ca=true;
		else b+=vec[i]+',',cb=true;
	}
	if(ca)cout<<"\""<<a.substr(0,a.size()-1)<<"\""<<endl;else cout<<'-'<<endl;
	if(cb)cout<<"\""<<b.substr(0,b.size()-1)<<"\""<<endl;else cout<<'-'<<endl;
	return 0;
}
