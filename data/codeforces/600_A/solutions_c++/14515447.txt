#include<cstdio>
#include<cstring>
#define MAXN 100000
char s[MAXN+10],a[MAXN+10],b[MAXN+10],c[MAXN+10];
int lena,lenb,lens;
int main()
{
	scanf("%s",s);
	lens=strlen(s);
	for(int i=0;i<lens;i++){
		//memset(c,sizeof c);
		int lenc=0;
		bool flag=true;
		for(;i<lens;i++){
			if(s[i]==','||s[i]==';')
				break;
			c[lenc++]=s[i];
			if(!(s[i]>='0'&&s[i]<='9'))
				flag=false;
		}
		if(!lenc) flag=false;
		if(!flag||(c[0]=='0'&&lenc>1)){
			for(int j=0;j<lenc;j++)
				b[lenb++]=c[j];
			b[lenb++]=',';
		}
		else{
			for(int j=0;j<lenc;j++)
				a[lena++]=c[j];
			a[lena++]=',';
		}
	}
	if(s[lens-1]==','||s[lens-1]==';')
		b[lenb++]=',';
	if(!lena) printf("-\n");
	else{
		printf("\"");
		for(int i=0;i<lena-1;i++)
			printf("%c",a[i]);
		printf("\"\n");
	}
	if(!lenb) printf("-\n");
	else{
		printf("\"");
		for(int i=0;i<lenb-1;i++)
			printf("%c",b[i]);
		printf("\"\n");
	}
}
