#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char s[200500];
string a,b,tmp;
bool bo;
int main()
{
	scanf("%s",s+1);
	int len=strlen(s+1);
	a="";	b="";	tmp="";		bo=0;
	for (int i=1;i<=len+1;++i)
		if (i>len||s[i]==','||s[i]==';')
		{
			if (i==1||s[i-1]==','||s[i-1]==';') bo=1;
			if (!bo) a=a+","+tmp;
			else b=b+","+tmp;
			tmp="";	bo=0;
		}
		else {
			if (s[i]<'0'||s[i]>'9') bo=1;
			if (tmp==""&&s[i]=='0'&&(i<len&&(s[i+1]!=','&&s[i+1]!=';'))) bo=1;
			tmp=tmp+s[i];
		}	
	char I='"';
	if (a=="") printf("-\n");
	else {
		a[0]='"';
	
		cout<<a;
		printf("%c\n",I);
	}
	if (b=="") printf("-\n");
	else {
		
		b[0]='"';
		cout<<b;
		printf("%c\n",I);
	}
}
