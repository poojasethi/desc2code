#include <iostream>
#include <string>
using namespace std;

bool checknum(string s){
	//cout<<"in "<<s<<endl;
	for (int i = 0; i < s.size(); ++i){
		if(s[i]=='.' || (s[i]>='a' && s[i] <='z') || (s[i]>='A' && s[i] <='Z')) return false;
	}
	if(s.size()>1 && s[0] == '0') return false;
	if(s.size()==0) return false;
	return true;
}

int main(){
	string s;
	cin>>s;
	string first = "",second = "";
	string curr = "";
	int ff = 0,ss = 0;
	for (int i = 0; i < s.size()+1; ++i){
		if(i==s.size() || s[i]==';' || s[i] == ','){
			if(checknum(curr)){
				if(ff!=0) first = first + ",";
				first += curr;
				ff++;
			}
			else{
				if(ss!=0) second = second + ",";
				second += curr;
				ss++;
			}
			curr="";
		}
		else curr +=s[i];
	}
	if(ff==0) cout<<"-"<<endl;
	else cout<<"\""<<first<<"\""<<endl;
	if(ss==0) cout<<"-"<<endl;
	else cout<<"\""<<second<<"\""<<endl;
}
