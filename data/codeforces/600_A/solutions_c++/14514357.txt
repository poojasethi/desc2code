#include <bits/stdc++.h>

using namespace std;

int main( void )
{
	string s;
	string a,b;
	string buf;
	bool isdigit = true;
	a += "\"";
	b += "\"";

	cin >> s;
	for( int i = 0; i <= s.length(); i++ ) {
		if( s[i] == ',' || s[i] == ';' || s[i] == '\0' ) {
			if( buf.length() == 0 || (buf[0] == '0' && buf.length() > 1) )
				isdigit = false;
			if( !isdigit )
				b += buf+",";
			else
				a += buf+",";
			buf.clear();
			isdigit = true;
		}
		else {
			if( s[i] > '9' || s[i] < '0' )
				isdigit = false;
			buf += s[i];
		}

	}
	a[a.length()-1] = '\"';
	b[b.length()-1] = '\"';

	a.length() == 1 ? cout << "-\n" : cout << a << endl;
	b.length() == 1 ? cout << "-\n" : cout << b << endl;

	return 0;
}
