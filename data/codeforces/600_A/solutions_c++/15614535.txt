#include<iostream>
#include<string>
#include<vector>
#define pb push_back
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;
string str,cur;
vector<string> a,b;
bool f;
int main(){
	cin>>str;
	str=str+';';
	rep(i,(int)str.size())
		if(str[i]==',' || str[i]==';'){
			if(!(int)cur.size() || f || cur[0]=='0' && (int)cur.size()>1)
				b.pb(cur);
			else
				a.pb(cur);
			cur="";
			f=0;
		}
		else{
			cur=cur+str[i];
			(str[i]<'0' || str[i]>'9') && (f=1);
		}
	if(!a.size())
		cout<<"-\n";
	else{
		cout<<'\"';
		rep(i,a.size())
			if(i!=a.size()-1)
				cout<<a[i]<<',';
			else
				cout<<a[i]<<'\"';
		cout<<endl;
	}
	if(!b.size())
		cout<<"-\n";
	else{
		cout<<'\"';
		rep(i,b.size())
			if(i!=b.size()-1)
				cout<<b[i]<<',';
			else
				cout<<b[i]<<'\"';
		cout<<endl;
	}
}
