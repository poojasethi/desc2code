#include <bits/stdc++.h>

using namespace std;

string s;
string res1,res2;

bool isnum(const string &t)
{
	if(t.size() == 0)
		return false;
	if(t[0] == '0' && t.size() != 1)
		return false;
	for(int i=0;i<t.size();++i)
		if(t[i] > '9' || t[i] < '0')
			return false;
	return true;
}

void solve()
{
	stack<char>S;
	string tmp;
	for(int i=0;i<=s.size();++i){
		if(i == s.size() || s[i] == ';' || s[i] == ','){
			while(!S.empty()){
				tmp = S.top() + tmp;
				S.pop();
			}
			if(isnum(tmp))
				res1 += tmp + ',';
			else
				res2 += tmp + ',';
		}
		else
			S.push(s[i]);
		tmp.clear();
	}
	if(res1.size() == 0)
		res1 = "-";
	else{
		res1 = '"' + res1;
		res1[res1.size()-1] = '"';
	}
	if(res2.size() == 0)
		res2 = "-";
	else{
		res2 = '"' + res2;
		res2[res2.size()-1] = '"';
	}
	cout << res1 << endl << res2 << endl;
}

int main()
{
	cin >> s;
	solve();
	return 0;
}
