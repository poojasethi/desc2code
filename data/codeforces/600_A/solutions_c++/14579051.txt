#include <stdio.h>

const int N = 100001;
char s[N];
char a[N];
char b[N];

void copyS(char *s, char *t, int *tidx, int sidx, int start) {
  if ((*tidx) == 0) t[(*tidx)++] = '"';
  else t[(*tidx)++] = ',';
  for (int i = start; i < sidx; ++i) t[(*tidx)++] = s[i];
}

int main(void) {
  int sidx, aidx, bidx, start;
  int isA = 1;
  start = sidx = aidx = bidx = 0;
  scanf("%s", s);
  while (1) {
    if (s[sidx] == ',' || s[sidx] == ';' || s[sidx] == 0) {
      if (isA && sidx > start) {
        copyS(s, a, &aidx, sidx, start);
      } else {
        copyS(s, b, &bidx, sidx, start);
      }
      isA = 1;
      start = sidx + 1;
      if (s[sidx] == 0) break;
    } else {
      isA = isA && s[sidx] >= '0' && s[sidx] <= '9' && (s[start] != '0' || sidx <= start);
    }
    ++sidx;
  }
  a[aidx] = aidx == 0 ? '-' : '"';
  b[bidx] = bidx == 0 ? '-' : '"';
  printf("%s\n%s\n", a, b);
  return 0;
}
