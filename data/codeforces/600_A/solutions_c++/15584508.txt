#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
using namespace std;

int main(){
	int l,temp=0;
	char c;
	string s,t,ans2,ans1;
	bool f=false;
	cin>>s;
	s+=';';
	l=s.size();
	for (int i=0;i<l;i++){
		temp++;
		c=s[i];
		if (c==',' || c==';'){
			if (t==""){
				ans2+=',';
			}else if (t=="0" && temp==2){
				ans1=ans1+t+',';
			}else if (f) ans2=ans2+t+','; else ans1=ans1+t+',';
			temp=0;
			f=false;
			t="";
			continue;
		}
		if (f==false){
			if (temp==1 && c=='0') f=true;
			if (c=='.') f=true;
			if (!(c>='0' && c<='9')) f=true;
		}
		t+=c;
	}
	if (ans1!=""){
		cout<<"\""<<ans1.substr(0,ans1.size()-1)<<"\""<<endl;
	}else cout<<"-"<<endl;
	if (ans2!=""){
		cout<<"\""<<ans2.substr(0,ans2.size()-1)<<"\"";
	}else cout<<"-";
	return 0;
}
