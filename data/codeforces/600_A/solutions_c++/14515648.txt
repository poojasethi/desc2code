#include <iostream>
#include <string>
using namespace std;
int n;
string s,a,b;


int main()
{
    cin>>s;
    n=s.length();
    s=" "+s;
    s+=',';
    ++n;
    for (int i=1;i<=n;++i)
    {
        bool chu=false;
        string tmp="";
        int j=i;
        while (j<=n && s[j]!=',' && s[j]!=';')
        {
            if (s[j]<'0' || s[j]>'9') chu=true;
            tmp+=s[j];
            ++j;
        }
        if (chu || (tmp[0]=='0' && tmp.length()!=1) || tmp.length()==0) b+=','+tmp;
        else a+=','+tmp;
        i=j;
    }
    if (a.length()==0) a="-";
    else
    {
        a[0]='"';
        a+='"';
    }
    if (b.length()==0) b="-";
    else
    {
        b[0]='"';
        b+='"';
    }
    cout<<a<<"\n"<<b;
    return 0;
}
