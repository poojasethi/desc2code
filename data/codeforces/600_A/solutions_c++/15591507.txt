    #include <iostream>
    #include <string>
    #include <set>  
    #include <algorithm>
    #include <vector>
    #include <cmath>
    #include <queue>
    #include <cstring>
    #include <cstdio>
    #include <map>
using namespace std;
string s,a,b;
int e;
bool ta,tb,flag;
int main()
{
	cin>>s;
	a+='"';b+='"';
	s+=';';
	
	for(int i=0;i<s.size();i++)
	{
		e=i;
		flag=0;
		while(e<s.size()&&s[e]!=','&&s[e]!=';')
		{
			if(s[e]<'0'||s[e]>'9')
				flag=1;
			e++;
		}
		if(e==i)flag=1;
		if(e>i+1&&s[i]=='0')flag=1;
		if(!flag)
		{
			if(ta)a+=',';
			ta=1;
			for(int j=i;j<e;j++)
				a+=s[j];
		}
		else
		{
			if(tb)b+=',';
			tb=1;
			for(int j=i;j<e;j++)
				b+=s[j];
		}
		i=e;
	}
	a+='"';
	b+='"';
	if(!ta)a="-";
	if(!tb)b="-";
	cout<<a<<endl;
	cout<<b<<endl;
	return 0;
}
