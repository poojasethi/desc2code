#include <iostream>
#include <cstdio>
using namespace std;

string s,t,a,b;
bool chk(string p){
    int i;
    if(p.size()==0) return false;
    for(i=0; i<p.size(); i++){
        if(p[i]<'0' || p[i]>'9') return false;
    }
    if(p[0]=='0' && p.size()>1) return false;
    return true;
}

int main(){
    int i;
    cin>>s;
    s += ';';
    a += '"';
    b += '"';
    for(i=0; i<s.size(); i++){
        if(s[i]==',' || s[i]==';'){
            chk(t) ? a += t+',' : b += t+',';
            t.clear();
        } else t += s[i];
    }
    a[a.size()-1] = '"';
    b[b.size()-1] = '"';
    a.size()==1 ? cout<<'-' : cout<<a;
    puts("");
    b.size()==1 ? cout<<'-' : cout<<b;
    return 0;
}
