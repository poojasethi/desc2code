#include <bits/stdc++.h>
using namespace std;
const int MAXN=200010;
string s;
bool number(string t)
{
    if(t.length()==0) return false;
    if(t[0]=='0'&&t.length()>1) return false;
    for(int i=0;i<t.length();i++)
        if(!(t[i]>='0'&&t[i]<='9')) return false;
    return true;
}
bool sep(char c)
{
    return c==','||c==';';
}
int main()
{
    cin >> s;
    s+=",";
    int n=s.length();
    string cur="";
    string line1="";
    string line2="";
    for(int i=0;i<n;i++)
    {
        if(sep(s[i]))
        {
            if(number(cur)) line1+=cur+",";
            else line2+=cur+",";
            cur="";
        }
        else cur+=s[i];
    }
    if(line1=="") line1="-";
    else line1="\""+line1.substr(0,line1.length()-1)+"\"";
    if(line2=="") line2="-";
    else line2="\""+line2.substr(0,line2.length()-1)+"\"";
    cout << line1 << endl << line2 << endl;
    return 0;
}
