#include<iostream>
#include<string>
#include<vector>
using namespace std;

char s[111111];
string v;
vector<string>vec[2];

bool chk(string &v) {
	if (!v.length()) return true;
	for(int i=0;i<v.length();i++)
		if(v[i] > '9'||v[i]<'0') return true;
	if(v.length()>1 && v[0]=='0') return true;
	return false;
}

void prt(int id) {
	if (vec[id].size()) {
		cout<<"\"";
		for(int i=0;i<vec[id].size();i++) {
			if(i)cout<<",";
			cout<<vec[id][i];
		}
		cout<<"\""<<endl;
	} else cout<<"-"<<endl;
}

int main() {
	cin>>s;
	v = "";
	vec[0].clear();vec[1].clear();
	for(int i=0;s[i];i++) {
		if(s[i] == ',' || s[i] == ';') {
			vec[chk(v)].push_back(v);
			v = "";
		} else v+=s[i];
	}
	vec[chk(v)].push_back(v);
	
	prt(0);
	prt(1);
	
	return 0;
}
