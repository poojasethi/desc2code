#include<iostream>
#include<string>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;
string str,a,b,cur;
bool f;
int main(){
	ios_base::sync_with_stdio(0);
	cin>>str;
	str=str+';';
	rep(i,(int)str.size())
		if(str[i]==',' || str[i]==';'){
			if(!(int)cur.size() || f || cur[0]=='0' && (int)cur.size()>1)
				b+=cur+',';
			else
				a+=cur+',';
			cur="";
			f=0;
		}
		else{
			cur+=str[i];
			(str[i]<'0' || str[i]>'9') && (f=1);
		}
	if(!a.size())
		cout<<"-\n";
	else{
		a[(int)a.size()-1]='\"';
		cout<<'\"'<<a<<endl;
	}
	if(!b.size())
		cout<<"-\n";
	else{
		b[(int)b.size()-1]='\"';
		cout<<'\"'<<b<<endl;
	}
}
