#include<iostream>
#include<string>
#include<vector>
using namespace std;
int main() {
	string s;
	string a, b;
	cin >> s;
	s+= ";";
	string aux;
	aux = "";
	for(int i = 0;i < s.size(); i++) {
		if(s[i] == ';' or s[i] == ',') {
			bool flag = true;
			if(aux[0] == '0' and aux.size() > 1)
				flag = false;
			for(int j = 0; j < aux.size(); j++){
				if(aux[j] < '0' or aux[j] > '9')
					flag = false;
			}
			if(aux == "")
				flag = false;
			if (flag){
				a += aux;
				a += ",";
			} else {
				b += aux;
				b += ",";
			}

			aux = "";
		}
		else
			aux += s[i];
	}
	
	if(a == "")
		cout << "-" << endl;
	else {
		cout << "\"";
		for(int i = 0; i < ((int)a.size() - 1); i++)
			cout << a[i];
		cout << "\"" << endl;
	}
	if(b == "")
		cout << "-" << endl;
	else { 
		cout << "\"";
		for(int i = 0; i < ((int)b.size() - 1); i++)
			cout << b[i];
		cout << "\"" << endl;
	}
	return 0;
}
