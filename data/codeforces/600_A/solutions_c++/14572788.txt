#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <string>

using namespace std;

string tmp;

int check() {
	if (tmp.length() == 0)	return 0;
	if (tmp[0] == '0' && tmp.length() > 1)	return 0;
	for (int i = 0; i < tmp.length(); i++) {
		if (tmp[i] < '0' || tmp[i] > '9')	return 0;
	}
	return 1;
}

int main() {
	int c;
	bool p2 = false;
	string str1, str2;
	while (c = getchar()) {
		if (c != ',' && c != ';' && c != '\n') {
			tmp += c;
		} else {
			if (check()) {
				if (str1.length() == 0)	str1 += tmp;
				else  str1 += ',' + tmp;
			} else {
				if (!p2)	str2 += tmp, p2 = true;
				else  str2 += ',' + tmp;
			}
			tmp = "";
		}
		if (c == '\n')	break;
	}

	if (str1.length() == 0)	cout << "-\n";
	else cout << '"' << str1 << '"' << endl;
	if (!p2)	cout << "-\n";
	else cout << '"' << str2 << '"' << endl;

	return 0;
}
