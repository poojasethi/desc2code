n,m,k = map(int,raw_input().split())
ins = []
for _ in range(n):
  ins.append(map(int,raw_input().split()))
lc = [0]*n
lm = [False]*(k+1)
for i in range(m):
  wm = [-1]*(k+1)
  for j in range(n):
    if lc[j] == 0 and ins[j][i] > 0:
      if lm[ins[j][i]]:
        lc[j] = i+1
      elif wm[ins[j][i]] >= 0:
        lc[wm[ins[j][i]]] = i+1
        lc[j] = i+1
        lm[ins[j][i]] = True
      else:
        wm[ins[j][i]] = j
for x in lc:
  print x
