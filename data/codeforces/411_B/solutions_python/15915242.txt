R=lambda:map(int,raw_input().split())
n,m,k=R()
x=[0]*n
y=set()
t=zip(*(R() for _ in range(n)))
for c in range(m):
  z=[[] for _ in range(k+1)]
  a=t[c]
  for i in range(n):
    if not x[i]:
      z[a[i]]+=[i]
  for i in range(1,k+1):
    if i in y or len(z[i])>1:
      y.add(i)
      for j in z[i]:
        if x[j]<1:
          x[j]=c+1
print '\n'.join(map(str,x))
