n, m, k = map(int, raw_input().split())

X = []
for i in range(n):
  X.append(map(int, raw_input().split()))

cores = [-1]*n
for j in range(m):
  accessed = [0]*(k+1)
  for i in range(n):
    if cores[i] >= 0:
      accessed[X[i][cores[i]]] += 1
    elif X[i][j] != 0:
      accessed[X[i][j]] += 1
  for i in range(n):
    if cores[i] == -1 and X[i][j] != 0 and accessed[X[i][j]] > 1:
      cores[i] = j

for c in cores:
  print c+1
