#include <iostream>
using namespace std;
int a[110][110],ans[110];
bool mark[110],mark2[110];
int main()
{
  int n,m,k;
  cin>>n>>m>>k;
  for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
      cin>>a[i][j];
  for(int i=1;i<=m;i++)
    for(int j=1;j<=n;j++)
      {
	if(mark2[j]==true || a[j][i]==0)
	  continue;
	if(mark[a[j][i]]==true)
	  {
	    ans[j]=i;
	    mark2[j]=true;
	    continue;
	  }
	for(int o=1;o<=n;o++)
	  if(a[o][i]==a[j][i] && mark2[o]==false && o!=j)
	    {
	      ans[o]=ans[j]=i;
	      mark2[o]=mark2[j]=true;
	      mark[a[j][i]]=true;
	      break;
	    }
      }
  for(int i=1;i<=n;i++)
    cout<<ans[i]<<endl;
  return 0;
}
