#include <cstdio>

int n, m, k;
int x[123][123];
int p[123];
int locked[123];
int l[123];

int main()
{
	scanf("%d%d%d", &n, &m, &k);

	for(int i=1; i<=n; i++) {
		for(int j=1; j<=m; j++) scanf("%d", &x[i][j]);
	}

	for(int ins=1; ins<=m; ins++) {
		for(int i=1; i<=k; i++) p[i]=0;
		for(int i=1; i<=n; i++) {
			if(l[i] || !x[i][ins]) continue;
			if(locked[x[i][ins]]) {
				if(!l[i]) {
					l[i]=ins;
				}
			}
			else {
				p[x[i][ins]]++;
			}
		}
		for(int i=1; i<=n; i++) {
			if(p[x[i][ins]]>1) {
				locked[x[i][ins]]=1;
				if(!l[i]) {
					l[i]=ins;
				}
			}
		}
	}

	for(int i=1; i<=n; i++) printf("%d\n", l[i]);

	return 0;
}
