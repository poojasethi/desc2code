#include <iostream>
//#include <conio.h>
using namespace std;

int n, m, k, res[109], x[109][109], d[109];

int main ()
{
    cin >> n >> m >> k;
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++)
        {
			cin >> x[i][j];
		}
	}
	for (int j = 0; j < m; j++)
	{
		for (int i = 0; i < n; i++)
		{
			if ((d[x[i][j]]) && (res[i] == 0)) res[i] = j + 1;
			for (int h = i + 1; h < n; h++)
			{
				if ((x[i][j] == x[h][j]) && (x[i][j] != 0) && (res[i] == 0) && (res[h] == 0))
				{
					res[i] = res[h] = j + 1;
					d[x[i][j]] = 1;
				}
			}
		}
	}
	for (int i = 0; i < n; i++) cout << res[i] << endl;
	//getch();
    return 0;
}
