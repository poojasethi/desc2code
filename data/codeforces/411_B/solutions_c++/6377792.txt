#include<iostream>
using namespace std;
int a[101][101],etapi[101],n,m,k,cnt[101];
bool blokirebuli[101];
int main()
{
    cin>>n>>m>>k;
 for(int i=1;i<=n;i++)
  for(int j=1;j<=m;j++)
   cin>>a[i][j];
 for(int j=1;j<=m;j++)
 {
  for(int i=1;i<=k;i++)
   cnt[i]=0;
  for(int i=1;i<=n;i++)
   if(etapi[i]==0 && a[i][j]>0 && !blokirebuli[a[i][j]])
    cnt[a[i][j]]++;
  for(int i=1;i<=k;i++)
   if(cnt[i]>1) 
   blokirebuli[i]=1;
  for(int i=1;i<=n;i++)
   if(etapi[i]==0 && a[i][j]>0 && blokirebuli[a[i][j]])
    etapi[i]=j;
 }
 for(int i=1;i<=n;i++)
  cout<<etapi[i]<<endl;
 //system("pause");
   return 0;
    
}
