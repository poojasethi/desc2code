#include<stdio.h>
#include<string.h>
    
const int N=105;
int ans[N],c[N],a[N][N],b[N],d[N];
int main(){
    int n,j,i,k,t,m,x,y;
    scanf("%d%d%d",&n,&m,&k);
    for(i=0;i<n;i++)
        for(j=0;j<m;j++) scanf("%d",&a[i][j]);
    for(j=0;j<m;j++){
        memset(c,0,sizeof(c));
        for(i=0;i<n;i++){
            if(!a[i][j]||d[i]) continue;
            if(ans[a[i][j]]) d[i]=j+1;
            else if(!c[a[i][j]]) c[a[i][j]]=i+1;
            else if(c[a[i][j]]) ans[a[i][j]]=1,d[i]=d[c[a[i][j]]-1]=j+1;
        }
    }
    for(i=0;i<n;i++) printf("%d\n",d[i]);
    return 0;
}
