#include <iostream>
using namespace std;

int main(){
	cin.tie(NULL);
	int nlocked[102]={};
	bool klocked[102]={};
	int com[101][101];
	int n, m, k;
	cin>>n>>m>>k;
	for(int i=0; i<n; ++i){
		for(int j=0; j<m; ++j){
			cin>>com[i][j];
		}
	}
	for(int j=0; j<m; ++j){
		int t[102]={};
		for(int i=0; i<n; ++i){
			int nu=com[i][j];
			if(nlocked[i+1]||(!nu))continue;
			if((!t[nu])&&(!klocked[nu]))t[nu]=i+1;
			else if(klocked[nu]){
				nlocked[i+1]=j+1;
			}
			else{
				nlocked[t[nu]]=nlocked[i+1]=j+1;
				klocked[nu]=true;
			}
		}
	}
	for(int i=1; i<=n; ++i){
		cout<<nlocked[i]<<"\n";
	}
	return 0;
}
