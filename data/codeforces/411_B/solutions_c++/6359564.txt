#include <iostream>
#include <cstring>
using namespace std;
int sol[150];
int A[150][150];
int caract[150];
int n,m,k;
int main()
{
    int i,j;
    cin>>n>>m>>k;
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
            cin>>A[i][j];
     for(i=1;i<=m;i++)
     {
         for(j=1;j<=n;j++) if(sol[j]==0) caract[A[j][i]]++;
         for(j=1;j<=n;j++) if(caract[A[j][i]]>1 &&sol[j]==0 &&A[j][i]!=0) sol[j]=i;
         for(j=1;j<=k;j++) if(caract[j]==1) caract[j]=0;
     }
     for(i=1;i<=n;i++) cout<<sol[i]<<"\n";

}
