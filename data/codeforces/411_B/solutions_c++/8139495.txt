# include <bits/stdc++.h>
using namespace std;
# define fi cin
# define fo cout
int S[105][105],x[105],y[105];
int main(void)
{
    int n,m,k;
    fi>>n>>m>>k;
    for (int i=1;i<=n;++i) for (int j=1;j<=m;++j) fi>>S[i][j];
    for (int j=1;j<=m;++j)
    {
        int p=j;
        for (int i=1;i<=n;++i)
        {
            if (x[S[i][j]] && !y[i]) y[i]=j;
            for (int t=i+1;t<=n;++t)
                if (S[i][j]==S[t][p] && S[i][j] && !y[i] && !y[t])
                  x[S[i][j]]=1,y[i]=j,y[t]=j;
        }
    }
    for (int i=1;i<=n;++i) fo<<y[i]<<'\n';
}
