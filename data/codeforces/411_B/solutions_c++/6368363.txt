#include <iostream>
using namespace std;
int main()
{
  int n,m,k,i,j;
  cin>>n>>m>>k;
  int x[n][m];
  int b[k];
  for (i=0;i<k;++i)
  {
    b[i]=0;
  }
  int c[n];
  for (i=0;i<n;++i)
  {
    c[i]=0;
  }
  for (i=0;i<n;++i)
    for (j=0;j<m;++j)
      cin>>x[i][j];
  for (j=0;j<m;++j)
  {
    for (i=0;i<k;++i)
        if (b[i]<2) b[i]=0;
    for (i=0;i<n;++i)
        if ((x[i][j]!=0)&&(c[i]==0)) b[x[i][j]-1]=b[x[i][j]-1]+1;
    //cout<<b[0]<<endl;
    for (i=0;i<n;++i)
        if (x[i][j]!=0)
          if ((b[x[i][j]-1]>1)&&(c[i]==0)) c[i]=j+1;
  }
  for (i=0;i<n;++i)
    cout<<c[i]<<endl;
  return 0;
}
