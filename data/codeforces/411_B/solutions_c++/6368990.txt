#include <cstdlib>
#include <cstring>
#include <iostream>

using namespace std;

int n,m,k,map[101][101],vis[101][101],vislen[101],flag[101],ans[101];

int main(int argc, char *argv[])
{
    cin>>n>>m>>k;
    for(int i=1;i<=n;i++)for(int j=1;j<=m;j++)cin>>map[i][j];
    for(int i=1;i<=m;i++)
    {
      memset(vislen,0,sizeof(vislen));
      for(int j=1;j<=n;j++)if(!ans[j])vis[map[j][i]][vislen[map[j][i]]++]=j;
      for(int I=1;I<=k;I++)
      {
        if(vislen[I]>1 || flag[I])
        {
          for(int j=0;j<vislen[I];j++)ans[vis[I][j]]=i;
          flag[I]=1;
        }
      }
    }
    for(int i=1;i<=n;i++)cout<<ans[i]<<endl;
    return 0;
}
