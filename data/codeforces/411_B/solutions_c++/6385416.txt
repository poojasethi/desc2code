#include <cstdio>
#include <algorithm>
using namespace std;

int a[105][105],u[105],v[105];//u记下core跪的时间。v记录cell在什么时候跪的
//v !!well done!!多记了一步！！

int main(){
    int n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    for(int i=0;i<n;i++)
    for(int j=0;j<m;j++) scanf("%d",&a[i][j]);
    for(int j=0;j<m;j++)
    {
        int tmp[105]={0};//ji lu shi fou gui le
        for(int i=0;i<n;i++) if(!u[i]) tmp[a[i][j]]++;
        for(int i=0;i<n;i++) if(!u[i]&&a[i][j])  {
         if(v[a[i][j]]) u[i]=j+1;
        else   if(tmp[a[i][j]]>=2) {v[a[i][j]]=j+1;u[i]=j+1;}
    }}
    for(int i=0;i<n;i++) printf("%d\n",u[i]);
    return 0;
}
