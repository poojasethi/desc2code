#include <iostream>
#define f cin
#define g cout
#include <cstring>
using namespace std;
int X[101][101],LOCK[101],USED[101],i,k,j,N,M,K;
bool OK[101];
int main()
{
    f>>N>>M>>K;
    for (i=1;i<=N;i++)
       for (j=1;j<=M;j++) f>>X[i][j];
    for (i=1;i<=M;i++)
    {
       memset(USED,0,sizeof(USED));
       for (j=1;j<=N;j++)
        if (LOCK[j]==0)
        { if (OK[X[j][i]]) {LOCK[j]=i;continue;}
        if (X[j][i]!=0) ++USED[X[j][i]];}
       for (k=1;k<=K;k++)
          if (USED[k]>1 && !OK[k])
               {for (j=1;j<=N;j++) if (X[j][i]==k && LOCK[j]==0) LOCK[j]=i;OK[k]=true;}
    }
    for (i=1;i<=N;i++) g<<LOCK[i]<<'\n';
    return 0;
}
