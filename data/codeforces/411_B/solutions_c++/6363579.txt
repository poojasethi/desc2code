#include <cstdio>

int n, m, k, a[101][101], core[101], cell[101];

int main() {
    scanf("%d%d%d", &n, &m, &k);
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            scanf("%d", &a[i][j]);
        }
    }
    for (int i = 0; i < m; ++i) {
        int cnt[101] = {};
        for (int j = 0; j < n; ++j) if (!core[j]) {
            cnt[a[j][i]]++;
        }
        for (int j = 1; j <= k; ++j) {
            if (cnt[j] > 1) cell[j] = 1;
        }
        for (int j = 0; j < n; ++j) if (!core[j]) {
            if (cell[a[j][i]]) core[j] = i + 1;
        }
    }
    for (int i = 0; i < n; ++i) {
        printf("%d\n", core[i]);
    }
}
