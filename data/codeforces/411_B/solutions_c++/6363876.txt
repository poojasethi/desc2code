#include <cstdio>
int n,m,k,a[101][101],re[101],cell[101];
int main(void){
//	freopen("input.txt","r",stdin);
	scanf("%d %d %d",&n,&m,&k);
	for(int i=0; i<n; i++){
		for(int j=0; j<m; j++)
			scanf("%d",&a[i][j]);
	}
	for(int i=0; i<m; i++){
		for(int j=0; j<n; j++){
			for(int k=j+1; k<n; k++){
				if(a[j][i] && a[k][i] && (!re[j] || re[j]==i+1) && (!re[k] || re[k]==i+1) && a[j][i] == a[k][i]){
					cell[ a[j][i] ]=1;
					re[j]=i+1;
					re[k]=i+1;
				}
			}
		}
		for(int j=0; j<n; j++){
			if( a[j][i] && !re[j] && cell[ a[j][i] ] )
				re[j]=i+1;
		}
	}
	for(int i=0; i<n; i++)
		printf("%d\n",re[i]);
	return 0;
}
