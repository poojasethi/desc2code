#include<iostream>
using namespace std;
bool yad[110],mem[110];
int ans[110];
int cnt[110][110];
main()
{
	int n,m,k,i,j;
	cin>>n>>m>>k;
	int a[n+5][m+5];
	for(i=1;i<=n;++i)for(j=1;j<=m;++j)cin>>a[i][j];
		
	for(j=1;j<=m;++j)
		{
			for(int o=1;o<=n;++o)if(yad[o]==false)cnt[a[o][j]][j]++;
			for(i=1;i<=n;++i)
				{	
				if(a[i][j]!=0 && cnt[a[i][j]][j]>1 && yad[i]==false)
					{
						mem[a[i][j]]=true;
						yad[i]=true;
						ans[i]=j;
					}
				else if(cnt[a[i][j]][j]==1)
					{
						if(mem[a[i][j]]==true && yad[i]==false)
						{
							yad[i]=true;
							ans[i]=j;
						}
					}
				}
		}
	for(i=1;i<=n;++i)cout<<ans[i]<<endl;
}
