#include <iostream>
#include <string.h>
using namespace std;
int s[105][105];
int main()
{
	bool f[105];
	int ans[105],n,m,i,j,k,l;
	cin>>n>>m>>k;
	for (i=1;i<=n;i++) for (j=1;j<=m;j++) cin>>s[i][j];
	memset(ans,0,sizeof(ans));
	memset(f,0,sizeof(f));
	for (i=1;i<=m;i++)
	{
		int v[105];
		memset(v,0,sizeof(v));
		for (j=1;j<=n;j++) if (ans[j]==0)
		{
			if (s[j][i]!=0 && (v[s[j][i]]>0 || f[s[j][i]]))
			{
				ans[j]=i;
				ans[v[s[j][i]]]=i;
				f[s[j][i]]=1;
			}
			else
			{
				v[s[j][i]]=j;
			}
		}
	}
	for (i=1;i<=n;i++) cout<<ans[i]<<endl;
}
