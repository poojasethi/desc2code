#include <cstdio>
#include <cstring>

const int N = 101;
int n, m, k;
int a[N][N], b[N], ans[N];
bool x[N];

int main() {
  scanf("%d%d%d", &n, &m, &k);
  for(int i = 0; i < n; ++i)
    for(int j = 0; j < m; ++j)
      scanf("%d", &a[i][j]);
  for(int j = 0; j < m; ++j) {
    memset(b, 0, sizeof(b));
    for(int i = 0; i < n; ++i)
      if(!ans[i+1] && a[i][j]) {
        if(b[a[i][j]] || x[a[i][j]]) {
          ans[b[a[i][j]]] = ans[i+1] = j+1;
          x[a[i][j]] = true;
        }else
          b[a[i][j]] = i+1;
      }
  }
  for(int i = 1; i <= n; ++i)
    printf("%d\n", ans[i]);
}
