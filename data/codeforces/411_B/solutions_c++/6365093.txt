#include<cstdio>
int a[105][105],core[105]={0},lock[105]={0};
int main()
{
	int n,m,k,i,j;
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;++i)
	for(j=1;j<=m;++j)
	scanf("%d",a[i]+j);
	for(j=1;j<=m;++j)
	for(int z=1;z<=k;++z){
		int fir=0,xx=0;
		for(i=1;i<=n;++i){
			if(a[i][j]==z&&!core[i])
			if(!fir){
				core[fir=i]=j;
				if(lock[z])xx=1;
			}
			else {
				core[i]=j;
				if(!xx){xx=1;lock[z]=1;}
			}
		}
		if(!xx)core[fir]=0;
	}
	for(i=1;i<=n;++i)printf("%d\n",core[i]);
	return 0;
}
