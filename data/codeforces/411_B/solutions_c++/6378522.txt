#include <stdio.h>
#include <stdlib.h>

int main()
{
	int a[101][101], k0[101] = { 0 }, n0[101] = { 0 }, i, j, k, n, m, p, o, q;
	scanf("%d %d %d", &n, &m, &k);
	for (i = 1; i <= n; i++)
	for (j = 1; j <= m; j++)
		scanf("%d", &a[i][j]);
	for (j = 1; j <= m; j++)
	{
		for (i = 1; i<=n; i++)
		if (!n0[i]&&a[i][j])
			k0[a[i][j]]++;
		for (o = 1; o <= n; o++)
		if (!n0[o] && k0[a[o][j]]>1)
			n0[o] = j;
		for (p = 1; p <= k; p++)
		if (k0[p] == 1)
			k0[p] = 0;
	}
	for (q = 1; q <= n; q++)
		printf("%d\n", n0[q]);
	return 0;
}
