#include <iostream>
#include <cstring>
using namespace std;

const int N = 100;
int x[N][N], lock[N], cnt[N], n, m, k;

int main() {
	cin >> n >> m >> k;
	for(int i = 0; i < n; ++i)
		for(int j = 0; j < m; ++j)
			cin >> x[i][j], --x[i][j];
	memset(lock, -1, sizeof lock);
	for(int j = 0; j < m; ++j) {
		for(int i = 0; i < n; ++i)
			if(lock[i] == -1 && x[i][j] != -1) ++cnt[x[i][j]];
		for(int i = 0; i < n; ++i)
			if(x[i][j] != -1 && cnt[x[i][j]] > 1 && lock[i] == -1)
				lock[i] = j;
		for(int i = 0; i < k; ++i)
			if(cnt[i] == 1) cnt[i] = 0;
	}
	for(int i = 0; i < n; ++i) cout << lock[i] + 1 << '\n';
	return 0;
}
