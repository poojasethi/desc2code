#include<iostream>
#include<string>
using namespace std;
int n,m,k;
int g[110][110];
int ans[110];
int cel[110];
int main()
{
	cin>>n>>m>>k;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			cin>>g[i][j];
	for(int i=1;i<=m;i++)
	{
		for(int j=1;j<=k;j++)
			if(cel[j]==1) cel[j]=0;
		for(int j=1;j<=n;j++)
		{
			if(!g[j][i]) continue;
			if(!ans[j]&&!cel[g[j][i]])
				cel[g[j][i]]=1;
			else if(!ans[j])
				cel[g[j][i]]=2;
		}
		for(int j=1;j<=n;j++)
			if(!ans[j]&&cel[g[j][i]]==2) ans[j]=i;
	}
	for(int i=1;i<=n;i++) cout<<ans[i]<<endl;
	return 0;
}
