#include <cstdio>
int n,m,k,a[200][200],r[200][200],bef[200][200],mem[200],brc[200];


int main()
{
	//freopen("B.in","r",stdin);
	//freopen("B.out","w",stdout);
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			scanf("%d",&a[i][j]);
	for(int j=1;j<=m;j++)
		for(int i=1;i<=n;i++)
		if(brc[i]==0 && a[i][j]>0)
		{
			if(mem[a[i][j]]==1 || r[a[i][j]][j]==1)
			{
				brc[bef[a[i][j]][j]]=j;
				brc[i]=j;
				mem[a[i][j]]=1;
			}
			else{
				r[a[i][j]][j]=1;
				bef[a[i][j]][j]=i;
			}
		}
	for(int i=1;i<=n;i++)
	{
		printf("%d\n",brc[i]);
	}
	
	
	return 0;
}
