#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n , m , k , x[101][101] , cnt[101] , ans[101] , y[101] ;

int main() 
{
  scanf("%d%d%d",&n,&m,&k) ;
  for( int i=1 ; i<=n ; i++) for( int j=1 ; j<=m ; j++) scanf("%d",&x[j][i]) ;

  for( int i=1 ; i<=m ; i++) 
  {
    memset(cnt,0,sizeof(cnt)) ;
    for( int j=1 ; j<=n ; j++) if(!ans[j] && x[i][j])
    { 
      cnt[x[i][j]]++ ;
      if(y[x[i][j]])  ans[j]=i ;  
    }      
    for( int j=1 ; j<=n ; j++) if(!ans[j] && cnt[x[i][j]]>1) ans[j]=i , y[x[i][j]]=1 ;
  }
  
  for( int j=1 ; j<=n ; j++) printf("%d\n",ans[j]) ; 

 return 0;
}
