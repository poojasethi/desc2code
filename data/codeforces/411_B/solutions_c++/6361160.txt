#include <iostream>
using namespace std;

int n, m, k, cl[105] = {}, pl[105] = {}, x[105][105];

int main(){
  cin >> n >> m >> k;
  for(int i = 0; i < n; i++){
    for(int j = 1; j <= m; j++) cin >> x[j][i];
  }
  for(int i = 1; i <= m; i++){
    int acc[105] = {};
    for(int j = 0; j < n; j++){
      if(pl[j] || x[i][j] == 0) continue;
      acc[x[i][j]]++;
    }
    for(int j = 0; j < n; j++){
      if(pl[j] || x[i][j] == 0) continue;
      if(acc[x[i][j]] > 1 || cl[x[i][j]]){
        pl[j] = i;
        cl[x[i][j]] = 1;
      }
    }
  }
  for(int i = 0; i < n; i++){
    cout << pl[i] << '\n';
  }
}
