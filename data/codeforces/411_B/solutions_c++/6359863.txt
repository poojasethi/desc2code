#include <iostream>
using namespace std;

int n,s,k;

int ist[101][101];

int m[101];
int l[101]; //cosa si è bloccato qua
int o[101]; //cosa c'è qua

int main(){
	cin >> n >> s >> k;
	for(int i=1; i<=n; i++) for(int j=1; j<=s; j++) cin >> ist[i][j];
	
	for(int i=1; i<=s; i++){
		for(int x=1; x<101; x++) o[x] = 0;
		for(int x=1; x<=n; x++) if(m[x]==0 && ist[x][i] > 0){
			if(o[ist[x][i]]) {l[ist[x][i]] = o[ist[x][i]]; m[l[ist[x][i]]] = i;}
			if(l[ist[x][i]]) m[x] = i;
			o[ist[x][i]] = x;
		}
	}
	
	for(int i=1; i<=n; i++) cout << m[i] << endl;
	return 0;
}
