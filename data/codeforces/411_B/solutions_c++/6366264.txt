#include<iostream>
#include<cstdio>
using namespace std;
int n, m, k;
int m1[101], m2[101];
int mark[101];
int a[101][101];
int main()
{
	scanf("%d%d%d", &n, &m, &k);
	for(int i = 1; i <= n; i++)
		for(int j = 1; j <= m; j++)
			scanf("%d", &a[i][j]);
	for(int i = 1; i <= m; i++)
	{
		for(int j = 1; j <= k; j++)
			mark[j] = 0;
		for(int j = 1; j <= n; j++)
			if(!m1[j] && a[j][i] != 0)
			{
				if(mark[a[j][i]])
					m2[a[j][i]] = 1;
				mark[a[j][i]] = j;
			}
		for(int j = 1; j <= n; j++)
			if(!m1[j] && m2[a[j][i]])
				m1[j] = i;
	}
	for(int i = 1; i <= n; i++)
		printf("%d\n", m1[i]);
	return 0;
}
