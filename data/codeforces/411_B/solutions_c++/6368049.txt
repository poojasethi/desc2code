#include<iostream>
using namespace std;
int n,m,i,j,k,l,a[105][105],vp[105]={0},vc[105]={0},re[105]={0},te[105];
int main()
{
    cin>>n>>m>>k;
    for(i=0;i<n;i++) for(j=0;j<m;j++) cin>>a[i][j];
    for(j=0;j<m;j++)
    {
        for(i=0;i<=k;i++) te[i]=0;
        for(i=0;i<n;i++)
        {
            if(vc[i]||!a[i][j]) continue;
            if(vp[a[i][j]]) {vc[i]=1;re[i]=j+1;continue;}
            te[a[i][j]]++;
        }
        for(i=0;i<n;i++)
        {
            if(vc[i]||!a[i][j]) continue;
            if(te[a[i][j]]>1) {vc[i]=1;vp[a[i][j]]=1;re[i]=j+1;}
        }
    }
    for(i=0;i<n;i++) cout<<re[i]<<endl;
    //cin>>i;
    return 0;
}
