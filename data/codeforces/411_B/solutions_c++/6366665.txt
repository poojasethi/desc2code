#include <iostream>
#include <vector>

using namespace std;

int n,m,k,t[101][101],fp[101];
vector <int> f(101),fm(101);

int main()
{
   cin>>n>>m>>k;
   for(int i=1; i<=n;i++) for(int j=1; j<=m; j++) cin>>t[i][j];

   for(int j=1; j<=m; j++) { f=fm;
    for(int i=1; i<=n;i++) if (fp[i]) continue; else f[t[i][j]]++; f[0]=0;
    for(int i=1; i<=n;i++) if (fp[i]) continue; else if (f[t[i][j]]>1) {fp[i]=j; fm[t[i][j]]=f[t[i][j]];}
   }

   for(int i=1; i<=n;i++) cout<<fp[i]<<endl;
}
