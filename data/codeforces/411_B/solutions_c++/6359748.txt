#include <cstdio>

#define N 100

int a[N][N], b[N][N + 1], c[N + 1], d[N], n, m, i, j, k;

int main()
{
    scanf("%d%d%d", &n, &m, &k);

    for (i = 0; i < n; i++)
        for (j = 0; j < m; j++)
            scanf("%d", &a[i][j]);

    for (j = 0; j < m; j++)
    {
        for (i = 0; i < n; i++)
            if (!d[i] && a[i][j])
                ++b[j][a[i][j]];
        for (i = 0; i < n; i++)
            if (!d[i] && a[i][j] && ((c[a[i][j]] && c[a[i][j]] <= j + 1) || b[j][a[i][j]] > 1))
                d[i] = j + 1, c[a[i][j]] = j + 1;
    }

    for (i = 0; i < n; i++)
        printf("%d\n", d[i]);

    return 0;
}
