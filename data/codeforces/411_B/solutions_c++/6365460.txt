#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxn=120;
int n,m,k,x[maxn][maxn],ans[maxn];
int lc[maxn]={0};
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j,cnt[maxn];
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			scanf("%d",&x[i][j]);
	for(i=1;i<=m;i++)
	{
		memset(cnt,0,sizeof(cnt));
		for(j=1;j<=n;j++)
			if(!ans[j]&&!lc[x[j][i]])
				cnt[x[j][i]]++;
		for(j=1;j<=k;j++)
			if(cnt[j]>1)
				lc[j]=1;
		for(j=1;j<=n;j++)
			if(!ans[j]&&lc[x[j][i]])
				ans[j]=i;
	}
	for(i=1;i<=n;i++)
		printf("%d\n",ans[i]);
	return 0;
}
