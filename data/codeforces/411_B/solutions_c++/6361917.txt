#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;


int n,m,k,a[505][505],tab[505][505];
int lock[505],Belocked[505];


int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=n;i++)
	for(int j=1;j<=m;j++)scanf("%d",&a[i][j]);
	for(int x=1;x<=m;x++)
	{
		memset(tab,0,sizeof(tab));
		for(int i=1;i<=n;i++)
		if(!lock[i]&&a[i][x])
		{
			int t=a[i][x];
			tab[t][++tab[t][0]]=i;
		}
		for(int i=1;i<=k;i++)
		if(tab[i][0]>1||Belocked[i])
		{
			for(int j=1;j<=tab[i][0];j++)lock[tab[i][j]]=x;
			Belocked[i]=1;
		}
	}
	for(int i=1;i<=n;i++)printf("%d\n",lock[i]);
	return 0;
}
