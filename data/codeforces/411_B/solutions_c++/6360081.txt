#include <iostream>
#include <vector>
using namespace std;
int i,j,h,t,n,m,k,core[101],a[101][101];
bool celula[101];
vector<int> v[101];

void verif(){
	for(h=1;h<=k;h++){
		if(v[h].size()>1){
			for(t=0;t<int(v[h].size());t++)
				core[v[h][t]]=j;
				celula[h]=1;
		}
	}
}


int main(){
	cin>>n>>m>>k;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			cin>>a[i][j];

	for(j=1;j<=m;j++){
		for(i=1;i<=n;i++){
			if(!core[i]){
				if(celula[a[i][j]]) core[i]=j;
				else if(a[i][j]) v[a[i][j]].push_back(i);
			}
		}
		verif();
		for(i=1;i<=k;i++)
			v[i].clear();
	}

	for(i=1;i<=n;i++)
		cout<<core[i]<<"\n";
}
