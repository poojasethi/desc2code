#include <iostream>
#include <cstring>
using namespace std;
const int MAX = 105;
int val[MAX][MAX], ans[MAX], cnt[MAX];
bool islock[MAX];
int main()
{
	ios::sync_with_stdio(false);
	int n, m, k;
	cin >> n >> m >> k;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			cin >> val[i][j];
	for (int j = 0; j < m; j++)
	{
		memset(cnt, 0, sizeof(cnt));
		for (int i = 0; i < n; i++)
			if (!ans[i])
				cnt[val[i][j]]++;
		for (int i = 1; i <= k; i++)
			islock[i] |= (cnt[i] > 1);
		for (int i = 0; i < n; i++)
			if (islock[val[i][j]] && !ans[i])
				ans[i] = j + 1;
	}
	for (int i = 0; i < n; i++)
		cout << ans[i] << endl;
	return 0;
}
