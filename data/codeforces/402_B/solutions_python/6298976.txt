

n, k = [int(_) for _ in raw_input().split()]
a = [int(_) for _ in raw_input().split()]


ans = 10**10
res = []


for v in range(1, 1001):
    t_res = [v+k*i for i in range(n)]
    t_res = [v-ai for v, ai in zip(t_res, a)]
    t_ans = sum(1 for _ in t_res if _ != 0)
    if t_ans < ans:
        res = t_res
        ans = t_ans

print ans
for i, v in enumerate(res, 1):
    if v != 0:
        if v > 0:
            print '+', i, v
        else:
            print '-', i, -v
