n,k = map(int,raw_input().split())
a = map(int,raw_input().split())
b = list(a)
c = {}
for i in range(n):
  b[i] -= i*k
  if b[i] > 0:
    if b[i] not in c:
      c[b[i]] = 0
    c[b[i]] += 1
m = 0
M = 0
for key in c:
  if c[key] > M:
    M = c[key]
    m = key

p=0
for i in range(n):
  p += a[i] != m + k*i
print p
for i in range(n):
  if a[i] < m + k*i:
    print '+ %d %d' % (i+1, - a[i] + m + k*i)
  if a[i] > m + k*i:
    print '- %d %d' % (i+1, a[i] - (m + k*i) )
