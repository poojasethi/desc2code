#!/usr/bin/env python

import sys
import logging
import traceback

logging.basicConfig(level = logging.INFO)

n, k = map(int, raw_input().split())

checked = [0] * n

nums = map(int, raw_input().split())

m = n + 1
ms = -1

for i in range(n) : 
    if checked[i] == 1 : 
        continue
    s = nums[i] - (i*k)
    if s <= 0 : 
        continue
    target = [s + j*k for j in range(n)]
    diff = 0
    for j in range(n) : 
        if target[j] != nums[j] : 
            diff += 1
        else : 
            checked[j] = 1
    if diff < m : 
        m = diff
        ms = s

print m 

target = [ms + j*k for j in range(n)]

for i in range(n) : 
    if target[i] != nums[i] : 
        if nums[i] < target[i] : 
            print "+ %d %d" %(i+1, abs(nums[i] - target[i]))
        else : 
            print "- %d %d" %(i+1, abs(nums[i] - target[i]))
