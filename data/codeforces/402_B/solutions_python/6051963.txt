[n,k] = map(int, raw_input().split())
T = map(int, raw_input().split())

best = n
bestt = -1
for t in xrange(n):
    count = 0
    if T[t] + -t*k < 1: continue
    for l in xrange(n):
        if T[l] != T[t] + (l-t)*k: count += 1
    if count < best:
        best = count
        bestt = t

print best
t = bestt
for l in xrange(n):
    temp = T[t] + (l-t)*k
    if T[l] < temp:
        print "+ %d %d" % (l+1, temp - T[l])
    elif T[l] > temp:
        print "- %d %d" % (l+1, T[l] - temp)
