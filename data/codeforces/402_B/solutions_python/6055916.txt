I=lambda:map(int, raw_input().split())
n, k = I()
a = I()
minCnt, ans = 1000000, []
for i in xrange(1, 1001):
    start = i
    tmp = []
    for j in xrange(n):
        if a[j] < start:
            tmp.append((1, j+1, start-a[j]))
        elif a[j] > start:
            tmp.append((0, j+1, a[j]-start))
        start += k
    if minCnt > len(tmp):
        ans, minCnt = tmp, len(tmp)
print minCnt
for x, y, z in ans:
    if x == 1:
        print '+', y, z
    else:
        print '-', y, z
