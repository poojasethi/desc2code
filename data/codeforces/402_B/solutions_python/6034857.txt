[n, k] = map(int, raw_input().split())
a = map(int, raw_input().split())
d = {}
maxd = 0
maxx = -1
for i in xrange(n):
    x = a[i] - k * i
    if x > 0:
        d[x] = d.get(x, 0) + 1
        if d[x] > maxd:
            maxd = d[x]
            maxx = x
print n - maxd
for i in xrange(n):
    dif = maxx + k * i - a[i]
    if dif > 0:
        print "+", i + 1, dif
    elif dif < 0:
        print "-", i + 1, -dif
