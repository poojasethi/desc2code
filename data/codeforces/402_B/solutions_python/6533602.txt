n,k=map(int,raw_input().split())
a=map(int,raw_input().split())

res,index=1000,0
for i in range(1,1001):
    m=0
    for j in range(n):
        if a[j]!=i+j*k:
            m+=1
    if m<res:
        res,index=m,i
        
print res
for i in range(n):
    if a[i]<index+i*k:
        print '+',i+1,(index+i*k)-a[i]
    if a[i]>index+i*k:
        print '-',i+1,a[i]-(index+i*k)
