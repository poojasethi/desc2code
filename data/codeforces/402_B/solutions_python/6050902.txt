n, k = map(int, raw_input().split())
a, m, mi = map(int, raw_input().split()), 1000, 0
for i in range(n):
	c = 0
	for j in range(n):
		x, y = a[i] + k * (j - i), a[j]
		if x < 1:
			break
		c += (x != y)
	else:
		if c < m:
			m, mi = c, i
print m
for i in range(n):
	x, y = a[mi] + k * (i - mi), a[i]
	if x < y:
		print '-', i + 1, y - x
	elif x > y:
		print '+', i + 1, x - y
