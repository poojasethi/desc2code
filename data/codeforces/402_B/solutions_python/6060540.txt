n,k=map(int,raw_input().split())
a=map(int,raw_input().split())
ans=0
s=0
for i in xrange(1,1001):
    cc=0
    for j in xrange(0,n):
        cc+=int(a[j]==i+j*k)
    if cc>ans:
        ans=cc
        s=i
print n-ans
to=s
for j in xrange(0,n):
    if a[j]>to:
        print '-',j+1,a[j]-to
    if a[j]<to:
        print '+',j+1,-a[j]+to
    to+=k
