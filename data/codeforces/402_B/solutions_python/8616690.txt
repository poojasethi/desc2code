import sys

entrada1 = [int(x) for x in raw_input().split()]

n = entrada1[0]
k = entrada1[1]
cnt = 0
resp = sys.maxint

comeco = 0

arvores = [int(x) for x in raw_input().split()]

for i in range(1,1005):
	cnt=0
	for j in range(n):
		if arvores[j]!=i+k*j:
			cnt+=1
	if cnt<resp:
		resp = cnt
		comeco = i
print resp

for i in range(n):
	if arvores[i]>comeco+k*i:
		print "- %d %d" % (i+1, arvores[i]-(comeco+k*i))
	elif arvores[i]<comeco+k*i:
		print "+ %d %d" % (i+1, (comeco+k*i)-arvores[i])
