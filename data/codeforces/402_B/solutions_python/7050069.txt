n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
b = []
for i in range(n):
	matches = 0
	if a[i] > k * i:
		for j in range(n):
			if a[i] == a[j] - k * (j - i):
				matches += 1
	b += [matches]
print n - max(b)
p = b.index(max(b))
for i in range(n):
 	dest = a[p] - k * (p - i)
	if a[i] > dest:
		print '-' , i + 1 , a[i] - dest
	elif a[i] < dest:
		print '+' , i + 1 , dest - a[i]
