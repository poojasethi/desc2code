#include<stdio.h>

int a[1010], b[1010];

int main ()
{
	int n, k, m=0, t;
	scanf("%d %d", &n, &k);
	for(int i=0; i<n; i++){
		scanf("%d", a+i);
		if((a[i]-=k*i)>0 && (++b[a[i]])>m)
			m=b[a[i]], t=a[i];
	}
	
	printf("%d\n", n-m);
	for(int i=0; i<n; i++){
		a[i]-=t;
		if(a[i]>0){
			printf("- %d %d\n", i+1, a[i]);
		}else if(a[i]<0){
			printf("+ %d %d\n", i+1, -a[i]);
		}
	}
	return 0;
}
