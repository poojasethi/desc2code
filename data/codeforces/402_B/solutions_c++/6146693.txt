#include <iostream>
#include<algorithm>
using namespace std;
int n, k, ans=1005, i, t, mx ,x;
int a[1005];
int main()
	{
	cin>>n>>k;
	for(i=0;i<n;++i)
		cin>>a[i];
	for(x=1;x<=1000;++x)
		{
		t = 0;
		for(i=0;i<n;++i)
			if(x+i*k!=a[i]) 
            t++;
		if(t<ans)
			{
			ans = t;
			mx = x;
			}
		}
	cout<<ans<<endl;
	for(i=0;i<n;++i)
		if(mx+i*k!=a[i])
			{
			if(mx+i*k>a[i]) cout<<"+ ";
			else cout<<"- ";
			cout<<i+1<<" "<<abs(mx+i*k-a[i])<<endl;
			}
		//	system("pause");
			return 0;
	}
