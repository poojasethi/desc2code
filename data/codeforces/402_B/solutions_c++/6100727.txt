#include<stdio.h>
#include<cstdlib>
long a,b,c,arr[1001],n,k,m,i,j;
int main(){
    scanf("%ld%ld",&n,&k);
    for(i=0;i<n;i++) scanf("%ld",&arr[i]);
    m=999999;
    for(i=1;i<=1000;i++){
        for(j=0;j<n;j++){
            if(i+k*j!=arr[j]) c++;
        }
        if(m>=c){
            m=c;
            a=i;
        }
        c=0;
    }
    printf("%ld\n",m);
    for(i=0;i<n;i++){
        if(arr[i]<a+k*i) printf("+ %ld %ld\n",i+1,(a+k*i)-arr[i]);
        else if(arr[i]>a+k*i) printf("- %ld %ld\n",i+1,arr[i]-(a+k*i));
    }
}
