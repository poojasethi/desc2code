#include<cstdio>

int tab[1010];
int base[1010];

int abs(int a){return (a<0)?-a:a; }

int main()
{
	int n, k; scanf("%d%d", &n, &k);
	for(int i = 0; i < n; i++)
	{
		scanf("%d",tab+i);
		if(tab[i]-i*k > 0)
			base[tab[i]-i*k]++;
	}
	int h = 1, c = 0;
	for(int i = 1; i <= 1000; i++)
		if(base[i] > c)
		{
			h = i;
			c = base[i];
		}
	printf("%d\n",n-c);
	for(int i = 0; i < n; i++)
	{
		if(tab[i] == h+i*k) continue;
		printf("%c %d %d\n",
			(tab[i] < h+i*k)?'+':'-',
			i+1, abs(tab[i]-(h+i*k)));
	}
	return 0;
}
