#include <cstdio>
#include <algorithm>
#define rep(i,l,r) for (int i=l;i<=r;++i)
int n,k;
int a[1050];
int Min=~0U>>1,pos;
int main(){
	scanf("%d%d",&n,&k);
	rep(i,1,n) scanf("%d",&a[i]);
	rep(i,1,n){
		int res=0;
		rep(j,1,n){
			if (a[i]-(i-j)*k<=0){res=-1;break;}
			if (a[j]!=a[i]-(i-j)*k) res++;
			}
		if (res>=0) Min=Min>res?pos=i,res:Min;
		}
	printf("%d\n",Min);
	rep(i,1,n){
		int tp=a[pos]-(pos-i)*k;
		if (a[i]!=tp) printf("%c %d %d\n",a[i]>tp?'-':'+',i,abs(a[i]-tp));
	}
}
