#include<cstdio>
int a[10000],b[10000],k,t,d[10000000],i,n,ans=0;
bool f;
int main(){
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=1;i<=n;i++){b[i]=a[i]-i*k+k;if(b[i]>0)d[b[i]]++,t=b[i];}
	for(i=1;i<=5000000;i++)if(d[t]<d[i])t=i;
	for(i=1;i<=n;i++)b[i]=t+i*k-k;for(i=1;i<=n;i++)ans+=(a[i]!=b[i]);printf("%d\n",ans);
	for(i=1;i<=n;i++)if(a[i]-b[i]){
		if(a[i]>b[i])printf("- %d %d\n",i,a[i]-b[i]);else printf("+ %d %d\n",i,b[i]-a[i]);}
}
