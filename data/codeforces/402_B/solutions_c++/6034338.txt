#include <iostream>

using namespace std;

int arr[1055];

int main()
{
	int n,k;
	cin>>n>>k;
	for(int i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	int a=0;
	int b=9999;
	for(int i=1;i<1055;i++)
	{
		int c=0;
		for(int j=0;j<n;j++)
		{
			if(arr[j]!=i+(j*k))
			{
				c++;	
			}
		}
		if(c<b)
		{
			b=c;
			a=i;	
		}
	}
	cout<<b<<endl;
	for(int i=0;i<n;i++)
	{
		if(arr[i]-a>0)
		{
			cout<<"- "<<i+1<<" "<<arr[i]-a<<endl;	
		}
		else if(arr[i]-a<0)
		{
			cout<<"+ "<<i+1<<" "<<a-arr[i]<<endl;		
		}
		a+=k;		
	}
	return 0;
}
