#include <cstdio>

int n,k;
int optv,optc;
int as[1002];
int main() {
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++) {
		scanf("%d",&as[i]);
		as[i]-=i*k;
	}
	optc=-1;
	for(int i=0;i<n;i++) {
		int nwc=0;
		if(as[i]<=0) continue;
		for(int j=0;j<n;j++) if(as[j]==as[i]) nwc++;
		if(optc<nwc) {
			optv=as[i];
			optc=nwc;
		}
	}
	printf("%d\n",n-optc);
	for(int i=0;i<n;i++) {
		if(as[i]!=optv) {
			if(as[i]<optv) {
				printf("+ %d %d\n",i+1,optv-as[i]);
			} else {
				printf("- %d %d\n",i+1,as[i]-optv);
			}
		}
	}
	return 0;
}
