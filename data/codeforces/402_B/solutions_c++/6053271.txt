#include<iostream>
using namespace std;

const int maxn=1000+100;
int a[maxn];
int b[maxn];

main()
{
	int n,k;
	cin >> n >> k;
	for(int i=0;i<n;i++)
	{
		cin >> a[i];
		if(a[i]-(i*k)>0)
			b[a[i]-(i*k)]++;
	}
	int m=0;
	int an=0;
	for(int i=0;i<maxn;i++)
	{
		if(m<b[i])
		{
			m=b[i];
			an=i;
		}
	}
	cout << n-m << endl;
	for(int i=0;i<n;i++)
	{
		if(a[i]<an+(k*i))
			cout << "+ " << i+1 << " " << an + (k*i)-a[i] << endl;
		if(a[i]>an+(k*i))
			cout << "- " << i+1 << " " <<a[i]- an - (k*i) << endl;
	}
	return 0;
}
