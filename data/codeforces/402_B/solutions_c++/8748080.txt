#include<iostream>
using namespace std;
int n,x,mx;
int a[1010],b[1010];
int main()
{
    cin>>n>>x;
    for (int i=0;i<n;i++)
        {
            cin>>a[i];
            if (a[i]-x*i>0) b[a[i]-x*i]++;
        }
    mx=1;
    for (int i=1;i<1010;i++)
        if (b[i]>b[mx]) mx=i;
    cout<<n-b[mx]<<endl;
    for (int i=0;i<n;i++)
    {
        if (mx+x*i-a[i]>0) cout<<"+ "<<i+1<<" "<<mx+x*i-a[i]<<endl; else
        if (mx+x*i-a[i]<0) cout<<"- "<<i+1<<" "<<-(mx+x*i-a[i])<<endl;
    }
    return 0;
}
