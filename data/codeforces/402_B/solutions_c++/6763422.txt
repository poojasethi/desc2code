#include <iostream>
using namespace std;

int a[1001], b[1001];
int main()
{
	int n, k, i, y, m=0;
	cin>>n>>k;
	for (i=1; i<=n; i++) cin>>a[i];
	for (i=1; i<=n; i++) {
		y=a[i]-k*(i-1);
		if (y>0) {
			b[y]++;
			if (y>m) m=y;
		}
	}
	for (y=1,i=2; i<=m; i++) if (b[i]>b[y]) y=i;
	cout<<n-b[y]<<endl;
	for (i=1; i<=n; i++,y+=k) {
		if (y>a[i]) cout<<"+ "<<i<<" "<<y-a[i]<<endl;
		else if (y<a[i]) cout<<"- "<<i<<" "<<a[i]-y<<endl;
	}
	return 0;
}
