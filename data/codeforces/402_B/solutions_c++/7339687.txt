#include <iostream>

using namespace std;
int n,m,q,z,f=1000000;
int a[1005];
main(){
cin>>n;cin>>m;
for(int k=1;k<=n;k++){
cin>>a[k];
}
for(int i=1;i<=1000;i++){
q=0; 
 for(int j=1;j<=n;j++){
if(a[j]!=i+(j-1)*m){
q++;
}

}
if(q<f){
f=q;z=i;
}

}
cout<<f<<endl;
for(int i=1;i<=n;i++){
if(a[i]>z+(i-1)*m){
cout<<"- "<<i<<" "<<a[i]-z-(i-1)*m<<endl;
}
if(a[i]<z+(i-1)*m){
cout<<"+ "<<i<<" "<<z+(i-1)*m-a[i]<<endl;
}

}

}
