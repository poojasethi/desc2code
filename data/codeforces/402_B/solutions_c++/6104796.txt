#include <string.h>
#include <iostream>
#include <stdio.h>

using namespace std;
int n,k,a[1001];
int main()
{
	cin>>n>>k;
	for(int i=0;i<n;i++)cin>>a[i];
	int num=1001,st=-1;
	for(int i=1;i<=1000;i++)
	{
		int tn=0;
		for(int j=0;j<n;j++)
		if(i+j*k!=a[j]) tn++;
		if(tn<num) num=tn,st=i;
	}
	cout<<num<<endl;
	
	for(int i=0;i<n;i++)
	{
		int nd=st+i*k;
		if(nd>a[i]) printf("+ %d %d\n",i+1,nd-a[i]);
		if(nd<a[i]) printf("- %d %d\n",i+1,a[i]-nd);
	}
	return 0;
}
