#include<stdio.h>
int main()
{ int h[1010],i,j,n,k,p,a;
  scanf("%d %d",&n,&k);
  for(i=0;i<n;i++)
  scanf("%d",&h[i]);
  int minci,mintou;
  for(a=1;a<=1000;a++)  
  { p=0;
  for(i=0;i<n;i++)
  if(h[i]!=a+k*i)p++;
  if(a==1||p<minci)minci=p,mintou=a;    
  }
  printf("%d\n",minci);
  for(i=0;i<n;i++)
  { if(h[i]>mintou+k*i)printf("- %d %d\n",i+1,h[i]-(mintou+k*i));
    else if(h[i]<mintou+k*i)printf("+ %d %d\n",i+1,(mintou+k*i)-h[i]);
  } 
  return 0;
}
