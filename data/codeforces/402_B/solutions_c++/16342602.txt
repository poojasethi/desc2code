#include<cstdio>

int n, k;
int a[1111];
int s[1111];

int main() {
	scanf("%d%d", &n, &k);
	for(int i=1;i<=n;i++) {
		scanf("%d", &a[i]);
		a[i] -= k*(i-1);
		if(a[i]>0) s[a[i]] ++;
	}
	int ma = a[1];
	for(int i=1;i<=1000;i++)
		if(s[i]>s[ma]) ma = i;
	printf("%d\n", n - s[ma]);
	for(int i=1;i<=n;i++)
		if(a[i]<ma) printf("+ %d %d\n", i, ma-a[i]);
		else if(a[i]>ma) printf("- %d %d\n", i, a[i]-ma);
	
	return 0;
}
