# include <stdio.h>
# include <algorithm>

int a,mn,an,z,k;
int b[1009];

using namespace std;

int main()
{
	scanf("%d %d",&a,&k);
	mn=9999999;
	for(int i=1;i<=a;i++)
	scanf("%d",&b[i]);
	
	for(int i=1;i<=1000;i++)
	{
		z=0;
		
		for(int j=1;j<=a;j++)
		if(b[j]!=i+(j-1)*k)
		z++;
		
		if(z<mn)
		{mn=z;an=i;}
	}
	
	printf("%d\n",mn);
	
	for(int i=1;i<=a;i++)
	{
		if(b[i]>an+(i-1)*k)
		printf("- %d %d\n",i,b[i]-an-(i-1)*k);
		if(b[i]<an+(i-1)*k)
		printf("+ %d %d\n",i,an+(i-1)*k-b[i]);
	}
}
