#include<cstdio>
int i,j,k,n,m,a[1000],b[1001];
int main()
{
	scanf("%d%d",&n,&k);
	for(i=0;i<n;++i){
		scanf("%d",a+i);
		if((a[i]-=i*k)>0&&++b[a[i]]>m)m=b[j=a[i]];
	}
	printf("%d\n",n-m);
	if(n!=m)
		for(i=0;i<n;++i)
			if(a[i]!=j)
				if(a[i]>j)
					printf("- %d %d\n",i+1,a[i]-j);
				else
					printf("+ %d %d\n",i+1,j-a[i]);
	return 0;
}
