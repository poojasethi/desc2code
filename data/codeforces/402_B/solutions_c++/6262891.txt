#include<cstdio>
int n,K;
int a[1100];
int main()
{
    scanf("%d%d",&n,&K);
    for (int i=1;i<=n;++i) scanf("%d",&a[i]);
    int res=1000000,ans;
    for (int i=1;i<=1000;++i)
    {
        int cnt=0;
        for (int j=1,k=i;j<=n;++j,k+=K)
            if (a[j]!=k)
                ++cnt;
        if (cnt<res) res=cnt,ans=i;
    }
    printf("%d\n",res);
    for (int j=1,k=ans;j<=n;++j,k+=K)
        if (a[j]>k)
            printf("- %d %d\n",j,a[j]-k);
        else if (a[j]<k)
            printf("+ %d %d\n",j,k-a[j]);

}
