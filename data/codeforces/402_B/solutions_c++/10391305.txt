#include <stdio.h>
int n,d,mn,z,i,k,l,arr[1002],pos[1001];
int main(){
 scanf("%d%d",&n,&k);
 for(i=1;i<=n;i++){
  scanf("%d",&arr[i]);
   mn=arr[i]-(i-1)*k;
    if(mn>0)pos[mn]++;
    if(mn>z)z=mn;
  }
    for(i=1;i<=z;i++)
      if(pos[i]>d){d=pos[i];l=i;}

  printf("%d\n",n-d);
  for(i=1;i<=n;i++){
    if((i-1)*k+l>arr[i])
     printf("+ %d %d\n",i,((i-1)*k+l)-arr[i]);
     else if((i-1)*k+l<arr[i])
     printf("- %d %d\n",i,arr[i]-((i-1)*k+l));
  }
}
