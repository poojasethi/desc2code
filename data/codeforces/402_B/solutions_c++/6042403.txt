#include <stdio.h>
int main() {
	int n,k,i,min,mina,c,cost,j;
	scanf("%d %d",&n,&k);
	int a=1,h[n];
	for(i=0;i<n;i++)
	scanf("%d",&h[i]);
	min=1001,mina=1;
	for(a=1;a<=1000;a++)
	{
		cost=0;
		for(j=1;j<=n;j++)
		if((a+(j-1)*k)!=h[j-1])
		cost++;
		if(min>cost)
		{
		min=cost;
		mina=a;
		}
	}
	printf("%d\n",min);
	for(i=0;i<n;i++)
	if((c=(mina+i*k-h[i]))!=0)
		{
		if(c>0)
		printf("+ ");
		else
		{
			printf("- ");
			c=-c;
		}
		printf("%d %d\n",i+1,c);
		}
	return 0;
}
