#include<cstdio>
#include<algorithm>
int i,j,k,t,n,m,a[1000],b[1000];
int main()
{
	scanf("%d%d",&n,&k);
	for(i=0;i<n;++i){
		scanf("%d",a+i);
		b[i]=a[i]-=i*k;
	}
	std::sort(a,a+n);
	for(i=0;i<n;++i)if(a[i]>0)break;
	for(i+=t=1;i<n;++i)
		if(a[i]==a[i-1])
			++t;
		else{
			if(t>m)m=t,j=a[i-1];
			t=1;
		}
	if(t>m)m=t,j=a[i-1];
	printf("%d\n",n-m);
	if(n!=m)
		for(i=0;i<n;++i)
			if(b[i]!=j)
				if(b[i]>j)
					printf("- %d %d\n",i+1,b[i]-j);
				else
					printf("+ %d %d\n",i+1,j-b[i]);
	return 0;
}
