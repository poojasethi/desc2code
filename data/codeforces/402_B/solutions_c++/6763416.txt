#include<iostream>
using namespace std;
int s[10000];
int main()
{
	int n,k;
	cin>>n>>k;
	int a[n],min=1;
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=1;i<10000;i++)
	{
		for(int j=0;j<n;j++)if(a[j]!=k*j+i)s[i]++;
		if(s[min]>s[i])min=i;
	}
	cout<<s[min]<<endl;
	for(int i=0;i<n;i++)if(a[i]!=k*i+min)
	{
		if(a[i]>k*i+min)cout<<"- "<<i+1<<" "<<a[i]-k*i-min<<endl;
		else cout<<"+ "<<i+1<<" "<<k*i+min-a[i]<<endl;
	}
}
