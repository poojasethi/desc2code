#include <iostream>

using namespace std;

int a[1005];

int main()
{
	int i,j;
	int n,k,b,s;
	int ans=1000000000,t;
	cin>>n>>k;
	for (i=0;i<n;i++) cin>>a[i];
	for (i=1;i<=1000;i++)
	{
		s=0;
		for (j=0;j<n;j++)
		{
			b=a[j]-i-k*j;
			if (b) s++;
		}
		if (s<ans)
		{
			ans=s;
			t=i;
		}
	}
	cout<<ans<<endl;
	for (i=0;i<n;i++)
	{
		if (a[i]==t+k*i) continue;
		
		if (a[i]>t+k*i) cout<<"- "<<i+1<<" "<<a[i]-(t+k*i)<<endl;
		else cout<<"+ "<<i+1<<" "<<t+k*i-a[i]<<endl;
	}
	return 0;
}
