#include<iostream>
using namespace std;
int main()
{
    int n,k,mx=0,mn=123456,c;cin>>n>>k;
    int a[n],b[n];for(int i=0;i<n;++i) {cin>>a[i];if(a[i]>mx)mx=a[i];}
    for(int i=1;i<=mx;++i)
    {
        c=0;
        for(int j=0;j<n;++j) if((i+j*k)!=a[j]) c++;
        if(c<mn) {mn=c;
        for(int j=0;j<n;++j) b[j]=i+j*k;}
    }
    cout<<mn<<endl;
    for(int i=0;i<n;++i)
    {
        if(b[i]!=a[i]) (b[i]>a[i]) ? (cout<<"+ "<<i+1<<" "<<b[i]-a[i]<<"\n") : (cout<<"- "<<i+1<<" "<<a[i]-b[i]<<"\n");
    }
}
