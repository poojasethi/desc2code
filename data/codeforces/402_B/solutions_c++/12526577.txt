#include <iostream>
#include <stdio.h>
#define N 2002 
using namespace std;

int main(void){
	int y[N],b[N]={0},num[2*N]={0},n,k,max = 0;
	cin >> n >> k;
	for(int i = 1;i <= n;i++){
		scanf("%d",&y[i]);
		b[i] = y[i] - k*i;
		if(k+b[i] > 0) num[b[i]+N]++;
	}
	for(int i = 1;i < 2*N;i++){
		if(num[max+N] < num[i]) max = i-N;
	}
	cout << n - num[max+N] << endl;
	for(int i = 1;i <=n;i++){
		if(y[i] > k*i+max) printf("- %d %d\n",i,y[i]-k*i-max);
		if(y[i] < k*i+max) printf("+ %d %d\n",i,k*i+max-y[i]);
	}
	return 0; 
}
