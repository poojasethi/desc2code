#include <iostream>
using namespace std;
int n,k;
int a[1005];
int main() {
    cin>>n>>k;
    for(int i=0;i<n;i++) cin>>a[i];
    int ans=~0u>>1;
    int pos=1;
    for(int i=1;i<=1000;i++) {
        int tmp=0;
        for(int j=0;j<n;j++) {
            if(a[j]!=k*j+i) tmp++;
        }
        if(tmp<ans) ans=tmp,pos=i;
    }
    cout<<ans<<endl;
    for(int j=0;j<n;j++) {
        if(a[j]<k*j+pos) cout<<"+ "<<j+1<<' '<<k*j+pos-a[j]<<endl;
        else if(a[j]>k*j+pos) cout<<"- "<<j+1<<' '<<a[j]-k*j-pos<<endl;
    }
    return 0;
}
