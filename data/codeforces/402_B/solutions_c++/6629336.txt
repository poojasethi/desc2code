#include<stdio.h>
int main(){
	int x,y,z,i,j;
	int n,m,k;
	int s=0;
	int a,b,c,min=99999;
	int arr[100000];
	scanf("%d %d",&n,&k);
	for(i=1 ; i<=n ; i++)
		scanf("%d",&arr[i]);
	for(i=1 ; i<=n ; i++){
		s=0;
		a=arr[i]-i*k;
		if(a+k<1)
			continue;
		for(j=1 ; j<=n ; j++){
			if(arr[j]!=a+k*j)
				s++;
		}
		if(min>s){
			min=s;
			b=a;
		}
	}
	printf("%d\n",min);
	for(i=1 ; i<=n ; i++){
		c=b+k*i;
		if(arr[i]>c){
			printf("- %d %d\n",i,arr[i]-c);
		}
		else if(arr[i]<c){
			printf("+ %d %d\n",i,c-arr[i]);
		}
	}
	return 0;
}
