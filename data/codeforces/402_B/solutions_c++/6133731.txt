#include <iostream>
#include <climits>
#include <algorithm>
using namespace std;

int main()
{
int i,q,b,j,c,n,k,d[1000000],ans=INT_MAX;
cin>>n>>k;
for(i=1;i<=n;i++)
{
cin>>d[i];
}
j=1;
while(j<1001)
{
b=j;
c=0;
for(i=1;i<=n;i++)
{
if(b!=d[i])
c++;
b+=k;
}
if(c<ans)
{
ans=c;
q=j;
}
j++;
}
cout<<ans<<endl;
for(i=1;i<=n;i++)
{
if(d[i]>q)
cout<<"-"<<" "<<i<<" "<<d[i]-q<<endl;
else if(d[i]<q)
cout<<"+"<<" "<<i<<" "<<q-d[i]<<endl;
q+=k;
}

    return 0;
}
