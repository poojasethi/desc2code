#include <stdio.h>
#include <math.h>
int n, k, x, I, min = 10000;
int a[1001];
int main(){
	scanf("%i %i", &n, &k);
	for(int i = 0; i < n; ++ i) scanf("%i", a + i);
	for(int i = 1; i <= 1000; ++ i){
		x = 0;
		for(int j = 0; j < n; ++ j){
			if(a[j] != (i + j * k)) x ++;
		}
		if(x < min){
			min = x;
			I = i;
		}
	}
	printf("%i\n", min);
	for(int i = 0; i < n; ++ i){
		if(I + i*k > a[i]) printf("+ %i %i\n", i + 1, I + i*k - a[i]);
		if(I + i*k < a[i]) printf("- %i %i\n", i + 1, - I - i*k + a[i]);
	}
	return 0;
}
