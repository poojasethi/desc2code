#include<iostream>
using namespace std;
int n,k,i,j=1,t[1001],a[1001001];
int main()
{cin>>n>>k;
    for(i=0;i<n;i++)
    {cin>>t[i];
	if(t[i]-i*k>0) a[t[i]-i*k]++;}
	for(i=0;i<1001000;i++) if(a[i]>a[j]) j=i;
	cout<<n-a[j];
	for(i=0;i<n;i++)
	{if(j!=t[i])
	    {if(j>t[i]) cout<<"\n+ "<<i+1<<' '<<j-t[i];
		else if(j<t[i]) cout<<"\n- "<<i+1<<' '<<t[i]-j;}
		j+=k;}
		return 0;
}
