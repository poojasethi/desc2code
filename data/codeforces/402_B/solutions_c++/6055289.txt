#include <cstdio>
#include <algorithm>
#define rep(i,l,r) for (int i=l;i<=r;++i)
int n,k,a[1050];
int tp,Min=~0U>>1,pos;
int main(){
	scanf("%d%d",&n,&k);
	rep(i,1,n) scanf("%d",&a[i]);
	rep(i,1,n){
		int res=0;
		rep(j,1,n){
			if (a[i]<=(i-j)*k) res=-99999;
			if (a[j]!=a[i]-(i-j)*k) res++;
			}
		if (res>=0) Min=Min>res?pos=i,res:Min;
		}
	printf("%d\n",Min);
	rep(i,1,n) if (a[i]!=(tp=a[pos]-(pos-i)*k)) printf("%c %d %d\n",a[i]>tp?'-':'+',i,abs(a[i]-tp));
}
