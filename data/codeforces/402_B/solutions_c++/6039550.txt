#include <stdio.h>
#define N 1010
int a[N],b[N],c[N];
int main()
{
	int n,k,i,ans,t,j,x;
	scanf("%d %d",&n,&k);
	for(i=0;i<n;i++)scanf("%d",&a[i]);
	t=n;
	for(i=0;i<n;i++)
	{
		x=a[i]-i*k;
		for(j=0;j<n;j++)b[j]=x+j*k;
		for(j=0;j<n;j++)if(b[j]<=0)break;
		if(j<n)continue;
		ans=0;
		for(j=0;j<n;j++)if(b[j]-a[j])ans++;
		if(ans<t)
		{
			t=ans;
			for(j=0;j<n;j++)c[j]=b[j];
		}
	}
	printf("%d\n",t);
	for(i=0;i<n;i++)if(c[i]>a[i])printf("+ %d %d\n",i+1,c[i]-a[i]);
		else if(c[i]<a[i])printf("- %d %d\n",i+1,a[i]-c[i]);
	return 0;
}
