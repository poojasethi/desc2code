#include <iostream>
#include <vector>
using namespace std;
int n,r[6005],ans,dp[6005][6005];
vector<int> adj[6005];
int rec(int cur,int prev, int pval){
	if(dp[cur][pval])return dp[cur][pval];
	int mx=r[cur]>r[pval]?1:0;
	for(int i=0;i<adj[cur].size();i++){
		if(adj[cur][i]==prev)continue;
		if(r[cur]>r[pval]){
			mx=max(mx,1+rec(adj[cur][i],cur,cur));
		}
		if(pval!=6000)mx=max(mx,rec(adj[cur][i],cur,pval));
	}
	dp[cur][pval]=mx;
	return mx;
}
int main(){
	cin>>n;
	for(int i=0;i<n;i++)cin>>r[i];
	for(int i=1;i<n;i++){
		int a,b;
		cin>>a>>b;
		adj[a-1].push_back(b-1);
		adj[b-1].push_back(a-1);
	}
	for(int i=0;i<n;i++){
		ans=max(ans,rec(i,6000,6000));
	}
	cout<<ans<<"\n";
	return 0;
}
