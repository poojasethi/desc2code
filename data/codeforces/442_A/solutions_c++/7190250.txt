#include<bits/stdc++.h>

using namespace std;

string st[105];
int a,ans;
map <char, bool> bol;
string cod="12345BGYRW";

int main(){
	ans=102;
	scanf("%d",&a);
	for(int x=0;x<a;x++) cin>>st[x];
	for(int x=0;x<(1<<10);x++){
		int tmp=x;
		bool can=true;
		int used=0;
		for(int y=0;y<10;y++) {
			if(tmp%2==1) bol[cod[y]]=true; else bol[cod[y]]=false;  
			if(tmp%2==1) used++;
			tmp/=2;
		}
		for(int y=0;y<a;y++)
			for(int z=y+1;z<a;z++) if(st[y]!=st[z])
				if (!((st[y][1]!=st[z][1] &&(bol[st[y][1]] || bol[st[z][1]])) || (st[y][0]!=st[z][0] && (bol[st[y][0]] || bol[st[z][0]])))) can=false; 
		if(can) ans=min(ans,used);
	}
	cout<<ans<<endl;
}
