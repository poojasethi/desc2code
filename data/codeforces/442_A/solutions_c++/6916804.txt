#include<stdio.h>
#define M 110

int f[26],a[M],b[M],g[10][10],cnt,ans;
char s[10];


void run(){
	ans=11;
	int i,j,k,r;
	for(i=0;i<32;i++)for(j=0;j<32;j++){
		for(k=0;k<5;k++){
			cnt=0;
			for(r=0;r<5;r++)if(g[k][r]){
				if(((1<<k)&i) && !((1<<r)&j))cnt++;
			}
			if(cnt>1)break;
		}
		if(k<5)continue;
		for(r=0;r<5;r++){
			cnt=0;
			for(k=0;k<5;k++)if(g[k][r]){
				if(!((1<<k)&i) && ((1<<r)&j))cnt++;
			}
			if(cnt>1)break;
		}
		if(r<5)continue;
		cnt=0;
		for(k=0;k<5;k++)for(r=0;r<5;r++)if(g[k][r]){
			if(!((1<<k)&i) && !((1<<r)&j))cnt++;
		}
		if(cnt>1)continue;
		cnt=0;
		for(k=0;k<5;k++)if((1<<k)&i)cnt++;
		for(k=0;k<5;k++)if((1<<k)&j)cnt++;
		if(ans>cnt)ans=cnt;
	}
	return;
}

int main(){
	//freopen("A.in","r",stdin);
	f['R']=0;f['G']=1;f['B']=2;f['Y']=3;f['W']=4;
	int n,i;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%s",s);
		a[i]=f[s[0]];b[i]=s[1]-'0'-1;
		g[a[i]][b[i]]++;
	}
	run();
	printf("%d\n",ans);
	return 0;
}
