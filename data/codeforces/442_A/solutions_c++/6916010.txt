#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <set>
#include <algorithm>

using namespace std;

int f[256];
char s[10];
int n;
int g[1000];

int main(){
    f[(int)'R']=0;
    f[(int)'G']=1;
    f[(int)'B']=2;
    f[(int)'Y']=3;
    f[(int)'W']=4;
    f[(int)'1']=5;
    f[(int)'2']=6;
    f[(int)'3']=7;
    f[(int)'4']=8;
    f[(int)'5']=9;
    set<int> S;
    S.clear();
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%s",s);
        g[i]=((1<<f[(int)s[0]])|(1<<f[(int)s[1]]));
        S.insert(g[i]);
    }
    int ans=10;
    int m=S.size();
    int x;
    for(int i=0;i<(1<<10);i++){
        x=0;
        for(int j=0;j<10;j++)
            if(i&(1<<j))x++;
        if(x>=ans)continue;
        S.clear();
        for(int j=0;j<n;j++)
            S.insert(g[j]&i);
        if((int)S.size()==m)ans=x;
    }
    printf("%d\n",ans);
    return 0;
}
