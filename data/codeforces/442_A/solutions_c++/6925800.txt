#include <iostream>
using namespace std;


int n;
int t[5][5];

int solve(int mask){
	for(int i=0; i<25; i++) for(int j=0; j<25; j++) if(i!=j && t[i/5][i%5] && t[j/5][j%5]){
		bool a = !(mask & (1 << (5+i%5))) && !(mask & (1 << (5+j%5)));
		bool b = !(mask & (1 << (i/5))) && !(mask & (1 << (j/5)));
		if(i/5 == j/5){ if(a) return 10; } else
		if(i%5 == j%5){ if(b) return 10; } else
		if(a && b) return 10;		
	}
	
	int ans = 0;
	while(mask > 0) {mask -= mask&(-mask); ans++;}
	return ans;
}

int main(){
	cin >> n;
	for(int i=0; i<n; i++){
		char c,v; cin >> c >> v;
		switch(c){
			case 'R': c = 0; break;
			case 'G': c = 1; break;
			case 'B': c = 2; break;
			case 'Y': c = 3; break;
			case 'W': c = 4; break;
		}
		switch(v){
			case '1': v = 0; break;
			case '2': v = 1; break;
			case '3': v = 2; break;
			case '4': v = 3; break;
			case '5': v = 4; break;
		}
		t[c][v] = 1;
	}
	
	int ans = 10;
	for(int i=0; i<(1<<10); i++) ans = min(ans, solve(i));
	
	cout << ans;
	return 0;
}
