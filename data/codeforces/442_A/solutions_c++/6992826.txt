#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int n;
int a[20],b[20],c[6][6];
char op[10];
int change(char s){
	if(s=='R') return 1;else if(s=='G') return 2;else if(s=='B') return 3;else if(s=='Y') return 4;else return 5;
}
int main(){
	scanf("%d",&n);
	int ans=1010;
	for(int i=1;i<=n;i++) scanf("%s",op),c[change(op[0])][op[1]-'0']++;
	for(int st=0;st<(1<<10);st++){
		memset(b,0,sizeof(b));int cnt=0;
		for(int i=1;i<=10;i++) if(st&(1<<(i-1))) a[i]=1,cnt++;else a[i]=0;
		for(int i=1;i<=5;i++) for(int j=1;j<=5;j++) if(c[i][j]){
			if(a[i]&&a[j+5]);
				else if(a[i]) ++b[i];
					else if(a[j+5]) ++b[j+5];
						else ++b[11];
		}
		bool flag=true;
		for(int i=1;i<=11;i++) if(b[i]>1) flag=false;
		if(flag) ans=min(ans,cnt);
	}printf("%d\n",ans);
	return 0;
}
