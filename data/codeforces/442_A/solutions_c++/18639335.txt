#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

char a[110][3];
int c[120];

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
	#endif
	c['R']=0;c['G']=1;c['B']=2;c['Y']=3;c['W']=4;
	int n;scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%s",a[i]);
	int ans=10;
	for (int i=0;i<32;i++)
		for (int j=0;j<32;j++)
		{
			bool ok=1;
			for (int p=1;p<n;p++)
				for (int q=p+1;q<=n;q++)
				{
					if (a[p][0]==a[q][0]&&a[p][1]==a[q][1]) continue;
					if (!((a[p][0]!=a[q][0]&&((i&(1<<c[a[p][0]]))||(i&(1<<c[a[q][0]]))))||(a[p][1]!=a[q][1]&&((j&(1<<(a[p][1]-49)))||(j&(1<<(a[q][1]-49))))))) ok=0;
				}
			if (ok)
			{
				int s=0;
				for (int k=0;k<5;k++) if (i&(1<<k)) s++;
				for (int k=0;k<5;k++) if (j&(1<<k)) s++;
				ans=min(ans,s);
			}	
		}
	printf("%d\n",ans);
	return 0;
}
