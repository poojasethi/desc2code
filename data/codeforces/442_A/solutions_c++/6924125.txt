#include <bits/stdc++.h>
using namespace std;

int n;
map<char,int> m;
bool use[5][5] = {};

int main(){
  m['R'] = 0;
  m['G'] = 1;
  m['B'] = 2;
  m['Y'] = 3;
  m['W'] = 4;
  cin >> n;
  for(int i = 0; i < n; i++){
    string s;
    cin >> s;
    int col = m[s[0]], num = s[1] - '1';
    use[col][num] = 1;
  }
  int res = 10;
  for(int i = 0; i < (1 << 5); i++){
    for(int j = 0; j < (1 << 5); j++){
      int cost = __builtin_popcount(i) + __builtin_popcount(j);
      int freq[6][6] = {}, ok = 1;
      for(int k = 0; k < 5; k++){
        int a = (i & (1 << k)) ? k : 5;
        for(int l = 0; l < 5; l++){
          if(!use[k][l]) continue;
          int b = (j & (1 << l)) ? l : 5;
          freq[a][b]++;
          if(freq[a][b] > 1) ok = 0;
        }
      }
      if(ok) res = min(res, cost);
    }
  }
  cout << res << '\n';
}
