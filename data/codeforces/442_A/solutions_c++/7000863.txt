#include <cstdio>
#include <cstring>
#include <iostream>
#define rep(i,j,k)	for (int i = j; i <= k; i++)
using namespace std;
int n, mat[15][15];
char s[3], ch[6] = {'A','R','G','Y','W','B'};
int bit(int x){	return x?bit(x>>1)+(x&1):0;}
bool judge(int x)
{
	int tot = 0, num[20] = {0};
	rep(i,1,5)	rep(j,6,10)	if (mat[i][j])
	{
		if (((1<<(i-1))&x) && (!((1<<(j-1))&x)))	num[i]++;
		if ((!((1<<(i-1))&x)) && ((1<<(j-1))&x))	num[j]++;
		if ((!((1<<(i-1))&x)) && (!((1<<(j-1))&x)))	tot++;
	}
	if (tot > 1)	return 0;
	rep(i,1,10)	if (num[i] > 1)	return 0;
	return 1;
}
int main()
{
	while (scanf("%d", &n) != EOF)
	{
		memset(mat,0,sizeof(mat));
		rep(i,0,n-1)
		{
			scanf("%s", s);
			int a = 0;
			rep(j,1,5)	if (s[0] == ch[j]) {a =j; break;}
			int b = s[1]-'0'+5;
			mat[a][b] = 1;
		}
		int ans = 100;
		rep(i,0,(1<<11)-1)	if (judge(i))	ans = min(ans,bit(i));
		cout << ans << endl;
	}
	return 0;
}
