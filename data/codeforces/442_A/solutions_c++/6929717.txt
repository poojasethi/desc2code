#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<map>
#include<vector>
#include<set>
#include<iostream>
using namespace std;
#define FOR(i,a,b) for (int i = (a); i < (b); i++)
#define FOE(i,a,b) for (int i = (a); i <= (b); i++)
#define PB push_back
#define X first
#define Y second
#define INF 1 << 27
#define EPS 1e-9
int n, m, a[100];
char s[100][3], c[10] = {'R', 'G', 'B', 'Y', 'W', '1', '2', '3', '4', '5'};
int ok(){
	FOR(i,0,n) FOR(j,0,i) if (a[i] == a[j] && strcmp(s[i], s[j])) return 0;
	return 1;
}
int f(int x){
	int ret = 0;
	while( x & -x ){
		ret++;
		x -= x & -x;
	}
	return ret;
}
int main(){
	int ans = 10;
	scanf("%d", &n);
	FOR(i,0,n) scanf("%s", s[i]);
	FOR(k,0,1 << 10){
		memset(a, 0, sizeof(a));
		FOR(i,0,10){
			if (!(k & (1 << i))) continue;
			FOR(j,0,n){
				if (s[j][0] == c[i] || s[j][1] == c[i]) a[j] |= 1 << i;
			}
		}
		if (ok()) ans = min(ans, f(k));
	}
	printf("%d\n", ans);
	return 0;
}
