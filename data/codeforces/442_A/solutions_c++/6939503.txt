#include<bits/stdc++.h>
#define s(a) scanf("%d",&a)
#define ss(a) scanf("%s",a)
using namespace std;
inline int chr(char a)
{
	if(a=='R')return 0;
	if(a=='G')return 1;
	if(a=='B')return 2;
	if(a=='Y')return 3;
	if(a=='W')return 4;
}
bool inp[5][5];
int main()
{
	int i,n,j,k;
	vector<pair<int,int> > v;
	char str[5];
	s(n);
	memset(inp,0,sizeof(inp));
	for(i=0;i<n;i++)
	{
		ss(str);
		inp[chr(str[0])][str[1]-'1']=1;
	}
	int res=10;
	for(i=0;i<5;i++)
		for(j=0;j<5;j++)
			if(inp[i][j])
				v.push_back(make_pair(i,j));
	for(i=0;i<(1<<10);i++)
	{
		for(j=0;j<v.size();j++)
		{
			for(k=j+1;k<v.size();k++)
			{
				int l1=v[j].first,r1=v[j].second;		
				int l2=v[k].first,r2=v[k].second;
				if( ((i>>(l1+5))&1) || ((i>>(l2+5))&1))
					if(l1!=l2)
						continue;
				if( ((i>>r1)&1) || ((i>>r2)&1))
					if(r1!=r2)
						continue;
				break;
			}
			if(k!=v.size())break;
		}
		if(j==v.size())
			res = min(res,__builtin_popcount(i));
	}
	cout<<res<<endl;
	return 0;
}
