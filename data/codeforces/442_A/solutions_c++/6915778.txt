#include<iostream>
#include<string>
#include<algorithm>
using namespace std;

int n, p1, result = 10;
char c;
bool exists[25], choice[10];
string col = "RGBYW";

void solve(int pos, int chosen) {
	if(pos == 10) {
		int valid = true;
		
		for(int i=0;i<25;i++)
			if(exists[i])
				for(int j=i+1;j<25;j++)
					if(exists[j]) {
						int c1 = i/5, c2 = j/5, n1 = i%5, n2 = j%5;
						
						if(c1 != c2 && (choice[c1] || choice[c2]) )
							continue;
						if(n1 != n2 && (choice[5+n1] || choice[5 + n2]) )
							continue;
						
						valid = false;
					}
		
		if(valid)
			result = min(result, chosen);
		return;
	}
	
	choice[pos] = false;
	solve(pos+1, chosen);
	choice[pos] = true;
	solve(pos+1, chosen + 1);
}

int main() {
	cin >> n;
	
	for(int i=0;i<n;i++) {
		cin >> c >> p1, p1--;
		
		exists[col.find(c) * 5 + p1] = true;
	}
	
	solve(0, 0);
	
	cout << result << '\n';
	return 0;
}
