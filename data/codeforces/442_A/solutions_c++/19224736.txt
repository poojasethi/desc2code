#include <iostream>
#include <map>
using namespace std;

int n;
string s[105], cards="RGBYW12345";
map<char, bool> m;

int main() {
	cin>>n;
	for (int i=0; i<n; i++)
		cin>>s[i];
	int ans=10;
	for (int i=0; i < (1<<10); i++) {
		int cnt=0;
		for (int j=0; j<10; j++) {
			m[cards[j]] = (i>>j)&1;
			cnt+=m[cards[j]];
			//cout<<i<<' '<<m[cards[j]]<<'\n';
		}
		bool poss=true;
		for (int j=0; j<n; j++) {
			for (int k=j+1; k<n; k++) {
				if (s[j]!=s[k]) {
					poss &= ((s[j][0]!=s[k][0] && (m[s[j][0]]||m[s[k][0]])) || 
							(s[j][1]!=s[k][1] && (m[s[j][1]]||m[s[k][1]])));
				}
			}
		}
		if (poss) {
			ans=min(ans, cnt);
		//	cout<<i<<'\n';
		//	for (int j=0; j<10; j++) {
		//		cout<<' '<<cards[j]<<' '<<m[cards[j]]<<'\n';
		//	}
		}
	}
	cout<<ans<<'\n';
	return 0;
}
