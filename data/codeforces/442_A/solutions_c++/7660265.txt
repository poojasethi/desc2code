#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=120,maxc=12;
int n,A[maxn],ans=0x7fffffff;
bool whe[maxc];
inline int Chg(char c) { return c=='R'?0:(c=='G'?1:(c=='B'?2:c=='Y'?3:4)); }
bool judg()
{
	int i,j;
	for(i=0;i<n;i++)
		for(j=0;j<i;j++)
			if(A[i]!=A[j]&&(A[i]/5==A[j]/5||(!whe[A[i]/5+5]&&!whe[A[j]/5+5]))&&(A[i]%5==A[j]%5||(!whe[A[i]%5]&&!whe[A[j]%5])))
				return 0;
	return 1;
}
void dfs(int x,int s)
{
	if(s>=ans) return ;
	if(x==10)
	{
		if(judg()) ans=s;
		return ;
	}
	whe[x]=0,dfs(x+1,s);
	whe[x]=1,dfs(x+1,s+1);
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i;
	char ci;
	scanf("%d\n",&n);
	for(i=0;i<n;i++)
	{
		ci=getchar();
		while(ci<'A'||ci>'Z') ci=getchar();
		A[i]=Chg(ci)*5+getchar()-'1';
	}
	//for(i=0;i<n;i++) printf("%d ",A[i]); printf("\n");
	dfs(0,0);
	printf("%d\n",ans);
	return 0;
}
