#include <iostream>
#include <map>

using namespace std;

const int MAXN = 101;
string s[MAXN], cols = "RGBYW12345";
map<char, bool> mp;

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> s[i];
	int ans = 1e9;
	for (int i = 0; i < (1 << 10); i++) {
		int cnt = 0;
		for (int j = 0; j < 10; j++)
			mp[cols[j]] = (i >> j) & 1, cnt += mp[cols[j]];
		bool check = true;
		for (int j = 0; j < n; j++)
			for (int k = j + 1; k < n; k++) {
				if (s[j] != s[k])
					check &= (s[j][0] != s[k][0] && (mp[s[j][0]] || mp[s[k][0]])) ||
							(s[j][1] != s[k][1] && (mp[s[j][1]] || mp[s[k][1]]));
			}
		if (check)
			ans = min(ans, cnt);
	}
	cout << ans << endl;
	return 0;
}
