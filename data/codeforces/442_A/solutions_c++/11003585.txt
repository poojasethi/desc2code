#include<iostream>
#include<cstring>
#include<string>
#include<vector>
#include<map>
using namespace std;
int n;
int f[6][6],mn=-1,have[6][6];
string s[105];
map<char,int> m;
int er(int x)
{
	int sum=0;
	while(x!=0)
	{
		sum++;
		x&=(x-1);
	}
	return sum;
}
int main()
{
	m['R']=0;m['G']=1;m['B']=2;m['Y']=3;m['W']=4;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>s[i];
		have[m[s[i][0]]][int(s[i][1])-'1']=1;
	}
	for(int i=0;i<(1<<5);i++)
	for(int j=0;j<(1<<5);j++)
	{
		memset(f,0,sizeof(f));
		int erge=er(i)+er(j);
		int x,can=1;
		if(erge<mn || mn==-1)
		{
			for(int k=0;k<5;k++)
			{
				x=i&(1<<k)?k:5;
				for(int l=0;l<5;l++)
				if(have[k][l]==1)
				{
					int y;
					y=j&(1<<l)?l:5;
					f[x][y]++;
					if(f[x][y]>1)
					can=0;
				}
			}
			if(can==1)mn=erge;
		}
	}
	cout<<mn;
	return 0;
}
