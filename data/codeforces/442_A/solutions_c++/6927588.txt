#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define rep(i,n) for (int i=0;i<n;++i)
typedef long long LL;
int n,ans,c[6][6],a[15],b[15]; char s[6];
int chg(char x){return x=='R'?0:x=='G'?1:x=='B'?2:x=='Y'?3:4;}
int main()
{
	scanf("%d",&n),ans=10;
	rep(i,n) scanf("%s",s),++c[chg(*s)][s[1]-49];
	rep(i,1<<10){
		int cnt=0,flag=1;
		rep(j,10) cnt+=a[j]=i>>j&1; memset(b,0,sizeof(b));
		rep(i,5) rep(j,5) if (c[i][j])
			if (a[i] && a[j+5]);
			else if (a[j+5]) ++b[j+5];
			else if (a[i]) ++b[i];
			else ++b[10];
		rep(j,11) if (b[j]>1) flag=0;
		if (flag) ans=min(ans,cnt);
	}
	printf("%d\n",ans);
	return 0;
}
