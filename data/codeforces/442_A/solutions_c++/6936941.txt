#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

char s[10];
int n, col[105], num[105], val[105];
int get(char c)  {
	if (c == 'R') return 0;
	if (c == 'G') return 1;
	if (c == 'B') return 2;
	if (c == 'Y') return 3;
	return 4;
}
int main()   {
	scanf("%d", &n);
	for (int i = 0; i < n; i ++)   {
		scanf("%s", s);
		col[i] = get(s[0]);
		num[i] = s[1] - '1' + 5;
	}
	int ans = 10;
	for (int i = 0; i < 1 << 10; i ++)   {
		int cnt = 0;
		for (int j = 0; j < 10; j ++)
			if ((i >> j) & 1) cnt ++;
		bool check = true;
		for (int j = 0; j < n; j ++)   {
			val[j] = 0;
			if ((i >> col[j]) & 1) val[j] += 1 << col[j];
			if ((i >> num[j]) & 1) val[j] += 1 << num[j];
			for (int k = 0; k < j; k ++)
				if (val[j] == val[k])  {
					if (col[j] == col[k] && num[j] == num[k])  continue;
					check = false;
				}
		}
		if (check) ans = min(ans, cnt);
	}
	printf("%d\n", ans);
	return 0;
}
