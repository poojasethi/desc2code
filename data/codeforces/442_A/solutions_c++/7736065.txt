#include <bits/stdc++.h>
using namespace std;

int n, ans, id[105];
char c[105][5], tp[] = "RGBYW";

int main(){

  scanf("%d", &n);

  for(int i = 0; i < n; i ++)
    scanf(" %s", c[i]);

  ans = 1000000073;
  for(int i = 0; i < (1 << 10); i ++){

    if(__builtin_popcount(i) >= ans)
      continue;
    
    memset(id, 0, sizeof id);

    // letter
    for(int j = 0; j < 5; j ++)
      if((i & (1 << j)) != 0)
         for(int k = 0; k < n; k ++)
           if(c[k][0] == tp[j])
             id[k] = (1 << j);

    // digit
    for(int j = 5; j < 10; j ++)
      if((i & (1 << j)) != 0)
         for(int k = 0; k < n; k ++)
           if(c[k][1] == (char)((j - 5 + 1) + '0'))
             id[k] += (1 << j);
    
    bool ok = true;
    for(int j = 0; j < n; j ++)
      for(int k = 0; k < j; k ++)
        if(id[j] == id[k] && (c[j][0] != c[k][0] || c[j][1] != c[k][1]))
          ok = false;

    if(ok)
      ans = __builtin_popcount(i);
  }

  printf("%d\n", ans);

  return 0;
}
