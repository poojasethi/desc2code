#include <iostream>
#include <string>
#include <algorithm>
#include <set>

using namespace std;

set<string> S, T;
int main(){
    int n; 
    cin >> n; 
    string x; 
    while(n--){
        cin >> x; 
        S.insert(x); 
    }
    string H = "RGBYW12345"; 
    
    int ans = 10; 
    for(int mask=0;mask<(1<<10); mask++){
        T.clear(); 
        for(set<string>::iterator it = S.begin(); it != S.end(); it ++){
            x = string(""); 
            for(int i=0;i<5; i++) if (((1<<i)&mask)  && H[i] == (*it)[0]) x += H[i]; 
            for(int i=5;i<10; i++) if (((1<<i)&mask)  && H[i] == (*it)[1]) x += H[i]; 
            T.insert(x); 
        }
        if(S.size() == T.size()) 
      ans = min(ans, __builtin_popcount(mask));
    }
    cout << ans <<endl;
}
