#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 25
bool a[N],v[N];
int n,ans;

int change(char x){
  if (x=='R') return 0;
  if (x=='G') return 1;
  if (x=='B') return 2;
  if (x=='Y') return 3;
  return 4;
}

void dfs(int k,int s){
  int i,j;
  if (k>=10){
    bool flag=1;
    for (i=0;i<N;++i)
      if (a[i])
        for (j=i+1;j<N;++j)
          if (a[j]){
            int x1=i/5,y1=i%5;
            int x2=j/5,y2=j%5;
            if (x1!=x2 && (v[x1] || v[x2])) continue;
            if (y1!=y2 && (v[y1+5] || v[y2+5])) continue;
            flag=0;
          }
    if (flag) ans=min(ans,s);
    return ;
  }
  v[k]=0;
  dfs(k+1,s);
  v[k]=1;
  dfs(k+1,s+1);
}

int main(){
  char s[3];
  int i;
  // freopen("1.txt","r",stdin);
  scanf("%d",&n);
  for (i=0;i<n;++i){
    scanf("%s",s);
    a[(s[1]-'1')*5+change(s[0])]=1;
  }
  ans=10;
  dfs(0,0);
  printf("%d\n",ans);
  return 0;
}
