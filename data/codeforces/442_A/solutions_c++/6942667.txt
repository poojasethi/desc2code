#include <algorithm>
#include <cstdio>
#include <vector>
using namespace std;

char tc[] = {'R', 'G', 'B', 'Y', 'W'};

int main() {
	int n;
	scanf("%d", &n);
	vector<int> c(n), v(n);
	for (int i = 0; i < n; ++i) {
		char a;
		scanf(" %c%d", &a, &v[i]);
		--v[i];
		if (a == 'R') c[i] = 0;
		if (a == 'G') c[i] = 1;
		if (a == 'B') c[i] = 2;
		if (a == 'Y') c[i] = 3;
		if (a == 'W') c[i] = 4;
	}
	int ans = 1 << 20;
	for (int i = 0; i <= 1 << 5; ++i)
		for (int j = 0; j <= 1 << 5; ++j) {
			bool dis = true;
			for (int k = 0; k < n; ++k)
				for (int l = 0; l < n; ++l) {
					if (k == l) continue;
					if (c[k] == c[l] && v[k] == v[l]) continue;
					if (c[k] != c[l] && (((i >> c[k]) & 1) || ((i >> c[l]) & 1))) continue;
					if (v[k] != v[l] && (((j >> v[k]) & 1) || ((j >> v[l]) & 1))) continue;
					dis = false;
				}
			if (dis)
				ans = min(ans, __builtin_popcount(i) + __builtin_popcount(j));
		}
	printf("%d\n", ans);
	return 0;
}
