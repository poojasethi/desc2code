#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long

using namespace std;

int n,x,s[5][5],us[10];
int res=10;
char c;

int main(){
  scanf("%d",&n);
  FI(i,1,n){
    scanf("%c%c%d",&c,&c,&x);
    if(c=='B') s[0][x-1]=1;
    if(c=='Y') s[1][x-1]=1;
    if(c=='W') s[2][x-1]=1;
    if(c=='G') s[3][x-1]=1;
    if(c=='R') s[4][x-1]=1;
  }
  FI(i,0,1023){
    int cnt=0;
    FI(j,0,9){
      if(i&(1<<j)) us[j]=1,cnt++;
      else us[j]=0;
    }
    FI(j,0,4) FI(k,0,4) FI(l,0,4) FI(m,0,4){
      if(j==l && k==m) continue;
      if(!s[j][k] || !s[l][m]) continue;
      if(j!=l && (us[j] || us[l])) continue;
      if(k!=m && (us[k+5] || us[m+5])) continue;
      goto gg;
    }
    res=min(res,cnt);
    gg:;
  }
  printf("%d\n",res);
  return 0;
}
