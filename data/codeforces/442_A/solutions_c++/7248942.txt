#include<cstdio>
#include<cstdlib>
int n;
int use[100001],opp[10000],ans;
struct rec{int a,b;}E[1001];
void check(){
	for (int i=1;i<=n;i++)
		for (int j=1;j<=n;j++)
		if(E[i].a!=E[j].a||E[i].b!=E[j].b){
			if(E[i].a==E[j].a){
				if(!use[E[i].b]&&!use[E[j].b]) return;
			}
			else
			if(E[i].b==E[j].b){
				if(!use[E[i].a]&&!use[E[j].a]) return;
			}
			if(!use[E[i].a]&&!use[E[j].a]&&!use[E[i].b]&&!use[E[j].b]) return;
		}
	int cnt=0;
	for (int i=1;i<=10;i++) if(use[opp[i]]) cnt++;
	if(cnt<ans) ans=cnt;
}
void dfs(int x){
	if(x>10){check(); return;}
	use[opp[x]]=1;
	dfs(x+1);
	use[opp[x]]=0;
	dfs(x+1);
}
char ch;
int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;i++){
		for (ch=getchar();ch<'A'||ch>'Z';ch=getchar());
		E[i].a=ch;
		ch=getchar();
		E[i].b=ch-'0';
	}
	opp[1]=1;
	opp[2]=2;
	opp[3]=3;
	opp[4]=4;
	opp[5]=5;
	opp[6]='R';
	opp[7]='G';
	opp[8]='B';
	opp[9]='Y';
	opp[10]='W';
	ans=0x7f7f7f7f;
	dfs(1);
	printf("%d\n",ans);
}
