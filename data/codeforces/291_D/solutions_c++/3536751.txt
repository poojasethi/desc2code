#include <iostream>
#include <cmath>
using namespace std;

int main(int argc, char* argv[])
{
	int n,k;
	cin>>n>>k;
	for (int i = 0; i<k; i++)
	{
		int p = pow(2, i);
		for (int j=1; j<=n; j++)
		{
			if (j<=n-2*p) cout<<j<<' ';
			else if (j<n-p) cout<<(p+j)<<' ';
			else cout<<n<<' ';
		}
		cout<<endl;
	}
	return 0;
}
