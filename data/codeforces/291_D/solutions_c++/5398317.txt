#include <iostream>
using namespace std;
int main ( ) {
	int n,k,i,j,t,p;
	cin>>n>>k;
	for (i=1;i<=k;i++) {
		for (j=1;j<=n;j++) {
			t=n-j;p=1<<(i-1);
			if (t>=2*p) cout<<1<<' ';
			else if (t>p) cout<<p+n-t<<' ';
			else cout<<n<<' ';
		}
		cout<<endl;
	}
	return 0;
}
