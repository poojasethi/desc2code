#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int a[1000001],n,k,i,j,t=1;
int main()
{
	cin>>n>>k;
	for(i=1;i<n;i++) a[i]=1;
	for(i=1;i<=k;i++)
	{
		for(j=1;j<=n;j++)
			if(n-j-a[j]<=t) printf("%d ",j+a[j]),a[j]=n-j;
			else printf("%d ",n-t),a[j]+=t;
		puts("");
		for(j=1,t=0;j<=n;j++) t=max(t,a[j]);
	}
}
