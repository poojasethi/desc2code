#include <algorithm>
#include <cstdio>
#include <iostream>

using namespace std;

main() {
    int n, k;
    cin >> n >> k;

    for (int t=0, p=1; t<k; t++, p*=2) {
        for (int i=1; i<n; i++)
            if (p < n-i)
                cout << max(n-p, i+p) << " ";
            else
                cout << n << " ";
        cout << n << endl;
    }
}
