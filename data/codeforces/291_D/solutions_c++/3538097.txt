#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    int no,k,c,d;
    cin>>no>>k;
    for(int i=0;i<k;i++)
    {
        d=pow(2,i);
        for(int k=1;k<=no-1-d;k++)
        cout<<k+d<<" ";
        if(no-d<=0)
        {c=1;}
        else
        {c=no-d;}
        for(int j=c;j<=no;j++)
        {
            cout<<no<<" ";
        }cout<<endl;
    }
}
