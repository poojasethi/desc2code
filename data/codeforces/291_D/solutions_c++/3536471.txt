#include <iostream>

using namespace std;

int main()
{
	int n,k,t,d,z;
	cin>>n>>k;
	z=0;
	
	for(t=2;t<n;t=(t-1)*2+1){
		for(d=t;d<=n-1;++d)
			cout<<d<<" ";
		for(int i=1;i<t;++i)
			cout<<n<<" ";
		cout<<n<<endl;
		++z;
	}
	
	k-=z;
	for(int i=1;i<k;++i){
		for(int j=1;j<n;++j)
			cout<<n<<" ";
		cout<<n<<endl;
	}
	if(k>0){
		for(int j=1;j<n;++j)
			cout<<n<<" ";
		cout<<n;
	}
	
	return 0;
}
