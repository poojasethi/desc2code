#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int a[10005];
int main()
{
    int n,k; 
    scanf("%d%d",&n,&k);
    for(int i=1;i<n;i++) a[i]=1;
    for(int i=0;i<k;i++) 
	{
        for(int j=1;j<=n;j++) 
		{
            int x=j+a[j];
        	printf("%d ",x);
            a[j]+=a[x];
        }
        printf("\n");
    }
    return 0;
}
