#include <cstdio>

const int MAX_N = 10000;

int a[MAX_N+3], p[MAX_N+3];
int n, k;

int main()
{
	int i, t;
	
	scanf("%d%d", &n, &k);
	for (i = 1; i < n; ++i)
		a[i] = 1;
	for (t = 0; t < k; ++t) {
		for (i = 1; i <= n; ++i) {
			if (i < n && a[i] + a[i+1] <= n-i)
				p[i] = i+1;
			else
				p[i] = i+a[i];
			a[i] += a[p[i]];
			printf("%d%c", p[i], (i==n)?'\n':' ');
		}
	}
	return 0;
}
