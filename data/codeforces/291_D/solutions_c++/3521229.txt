#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
	int n,k,r,q;
	scanf("%d%d",&n,&k);
	
	for(int i=0;i<k;i++)
	{
		r=pow(2,i);
		for(int j=1;j<n-r;j++)
			printf("%d ",j+r);
		if((n-r)<1)
			q=1;
		else
			q=n-r;
		for(int j=q;j<n;j++)
			printf("%d ",n);
		printf("%d\n",n);
	}
	
	return 0;
}
