#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    #ifdef Debug
    freopen("test.in", "r", stdin);
    freopen("test.out", "w", stdout);
    #endif
    int n, k, m = 1;
    scanf("%d%d", &n, &k);
    while (k--) {
        for (int i = n-1; i >=0; i--)
            printf("%d%c", i<m ? n : n-(i-m), i ? ' ' : '\n');
        m <<= 1;
    }
    return 0;
}
