#include <cstdio>

int main()
{
  int n, k;
  scanf("%d%d", &n, &k);

  for (int i = 0, j = 1; i < k; ++i, j <<= 1)
  {
    if (j >= n) j = n - 1;
    printf("%d", 1 + j);
    for (int x = 2; x + j <= n; ++x)
      printf(" %d", x + j);
    for (int x = 0; x < j; ++x)
      printf(" %d", n);
    putchar('\n');
  }

  return 0;
}
