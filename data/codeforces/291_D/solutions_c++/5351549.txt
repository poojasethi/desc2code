#include <stdio.h>
#include <iostream>
using namespace std;
int main( ){
	int n,m,p=1;
	cin>>n>>m;
	if (n<2){
		for (int j=1;j<=m;j++)
			cout<<1<<endl;
	}
	else{
		for (int i=1;i<=m;i++){
			for (int j=0;j<n-1;j++)
				if (1+j+p<n) cout<<1+j+p<<' '; else cout<<n<<' ';
			cout<<n<<endl;
			if (p<n) p*=2;
		}
	}
	return 0;
}
