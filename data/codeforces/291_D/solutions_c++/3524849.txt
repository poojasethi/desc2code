#include <iostream>
#include <algorithm>
using namespace std;
const int N=10005;
int c[N],q[N],n,k;
int main(){
    cin>>n>>k;
    for(int i=1;i<n;i++) q[i]=1;
    while(k--){
        int i,j;
        for(i=1;i<=n;i++){
            if(q[i]&&q[i]*2<=n-i) c[i]=i,q[i]*=2;
            else c[i]=i+q[i],q[i]=n-i;
            if(i<n) cout<<c[i]<<" ";
            else cout<<c[i]<<endl;
        }
    }

}
