#include<cstdio>
#define rep(i,n) for (int i=0;i<n;i++)
int n,k,a[100005];
int main()
{	
	scanf("%d%d",&n,&k);
	rep(j,k){
		rep(i,n) a[i]=0;
		rep(i,n) if ((i&-i)!=1<<j)
			if ((i&-i)<1<<j) if (i&1<<j) a[i]=1<<j; else;
			else if (i&1<<j); else a[i]=1<<j;
		rep(i,n-1) printf("%d ",n-a[n-1-i]); printf("%d\n",n-a[0]);
		}
	return 0;
}
