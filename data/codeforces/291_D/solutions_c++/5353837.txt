#include<cstdio>
#include<iostream>
using namespace std;
int n,k;
int main(){
    cin>>n>>k;
    for(int i=0;i<k;i++)
    {
        for(int j=0;j<n-1;j++){
            if((n-j-2)&(1<<i)) printf("%d ",n-(1<<i)); 
            else 
            printf("%d ",n);
        }
        printf("%d\n",n);
    }
    //system("pause");
    return 0;
}
