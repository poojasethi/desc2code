#include<cstdio>
#include<cstring>

int n,k;
int ans[22][22222];

int main(){
	scanf("%d%d",&n,&k);
	memset(ans,0,sizeof ans);
	for(int i=0;i<k;i++){
		for(int j=1;j<n;j++){
			int rem=(j&((1<<(i+1))-1));
			if(rem==0||rem>(1<<i))
				ans[i][j]=(1<<i);
		}
		for(int j=n-1;j>=0;j--){
			printf("%d",n-ans[i][j]);
			if(j>0)printf(" ");
			else printf("\n");
		}
	}

	return 0;
}
