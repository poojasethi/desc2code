
#include <cstdio>
#include <algorithm>
using namespace std;

typedef long long i8;

int n, k;
int re[10111];

main() {
	scanf("%d%d", &n, &k);
	int ok=1;
	for (int i=0; i<k; i++) {
		for (int j=0; j<n; j++) {
			if (j<=ok) re[j]=0;
			else if (j<=ok*2) re[j]=j-ok;
			else re[j]=j;
		}
		
		for (int j=n-1; j>=0; j--) printf("%d ", n-re[j]);
		printf("\n");
		
		if (ok<=n) ok*=2;
	}
}
