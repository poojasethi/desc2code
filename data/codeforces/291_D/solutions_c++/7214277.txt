#include <iostream>

using namespace std;

int main() {
	int n, k;
	cin >> n >> k;
	for (int i = 0; i < k; i++) {
		for (int j = 0; j < n; j++) {
			if (j >= n - (1 << i) - 1)
				cout << n << " ";
			else if (j >= n - (1 << i) - 1 - (1 << i))
				cout << j + (1 << i) + 1 << " ";
			else
				cout << j + 1 << " ";
		}
		cout << endl;
	}
	return 0;
}
