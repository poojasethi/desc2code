#include <iostream>
using namespace std;
long i,j,n,k,a[20000]={},f[20000]={};
int main() {
    cin >> n >> k;
    for (i=1;i<n;a[i++]=1) f[i]=i;
    f[n]=n;
    for (i=1;i<=k;i++,cout << n << endl)
        for (j=1;j<n;j++) {
            while (a[j]+a[f[j]]>n-j) f[j]++;
            a[j]+=a[f[j]];
            cout << f[j] << ' ';
        }
    return 0;
}
