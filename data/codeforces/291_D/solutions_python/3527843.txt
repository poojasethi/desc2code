if __name__ == "__main__":
    n, k = map(int, raw_input().split())
    a = [1] * (n - 1) + [0]
    for j in range(k):
        c = []
        for i in range(n):
            diff = (n - i - 1) - a[i]
            j = n - diff
            a[i] += a[j - 1]
            c.append(str(j))
        print " ".join(c)
