def get_steps(process_count, step_count):
    steps = []
    last_index = process_count - 1
    shift = 1
    while last_index > 1:
        steps.append(map(lambda index: max(1, min(process_count, shift + index)),
                         range(1, process_count + 1)))
        last_index -= shift
        shift *= 2
    for unused in range(step_count - len(steps)):
        steps.append([process_count] * process_count)    
    return steps

def main():
    process_count, step_count = map(int, raw_input().split())
    for step in get_steps(process_count, step_count):
        print ' '.join(map(str, step))

if __name__ == '__main__':
    main()
