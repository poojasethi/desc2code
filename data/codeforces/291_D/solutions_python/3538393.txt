import sys

n, k = [int(x) for x in raw_input().split()]

values = [1] * (n - 1) + [0]

def out(i, ci):
  sys.stdout.write("%d" % (ci + 1))
  values[i] += values[ci]
  if i + 1 != n:
    sys.stdout.write(' ')

for it in xrange(k):
  for i in xrange(n):
    if 2 * values[i] <= n - i - 1:
      out(i, i)
    else:
      out(i, i + values[i])
  print ''
