n, k = map(int, raw_input().split())

if n == 1:
    for i in xrange(k):
        print 1
    exit(0)

if n == 2:
    for i in xrange(k):
        print 2, 2
    exit(0)
        
from math import log
st = int(log(n - 2, 2) + 0.0001) + 1

for i in xrange(st):
    t = (1 << i) + 1
    for j in xrange(t, n):
        print j,
    for j in xrange(t):
        print n,
    print    

for i in xrange(st, k):
    for j in xrange(n):
        print n,
    print
