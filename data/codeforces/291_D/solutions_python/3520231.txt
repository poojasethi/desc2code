import sys

inp = [[int(j) for j in i.split(' ')] for i in sys.stdin.read().strip().splitlines()]

n, k = inp[0]

res = []
for i in xrange(0, k):
    r = [n] * (2**i+1)
    r = range(n - 2**i, n) + r
    r = [r[0]]*n + r
    res += [r[-n:]]

sys.stdout.write('\n'.join([' '.join([str(j) for j in i]) for i in res]))
