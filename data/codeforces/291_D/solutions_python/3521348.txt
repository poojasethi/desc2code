from __future__ import print_function
import sys


(n, k) = map(int, sys.stdin.readline().split())

for i in xrange(k):
    minV = 2 ** i
    maxV = minV * 2
    for i in xrange(1, n):
        v = n - i         
        if v >= maxV:
            print(i, end=' ')
        elif (v > minV) and (v <= maxV):
            print(n - (v - minV), end=' ')
        else:
            print(n, end=' ')
    print(n)
