import string
n, k = map(int, raw_input().strip().split())
a = [1] * n;
a[n - 1] = 0
b = [0] * n
for i in xrange(n):
    b[i] = i + 1
for i in xrange(k):
    for j in xrange(n):
        r = n - j - 1
        if a[j] * 2 >= r:
            b[j] = n - (r - a[j])
            a[j] += a[n - r + a[j] - 1]
        else :
            b[j] = j + 1
            a[j] += a[j]
    print string.join(map(str, b), ' ')
