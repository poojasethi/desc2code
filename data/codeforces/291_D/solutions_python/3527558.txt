#!/usr/bin/python
# -*- coding: utf-8 -*-

from sys import stdin

DEBUG = False
# DEBUG = True

if DEBUG:
	stdin = open('D.in')

tmp = stdin.readline()
n, n_steps = map(int, tmp.split())

mem = [1 if i < n-1 else 0 for i in range(n)]
good_mem = [n-i-1 for i in range(n)]

# print mem
# print good_mem
# print 

for i_step in range(n_steps):
	C = []
	for proc_i in range(n):
		if mem[proc_i] != good_mem[proc_i]:
			diff = good_mem[proc_i] - mem[proc_i]
			if diff >= mem[proc_i]:
				C.append(proc_i)
			else:
				C.append(n-1-diff)
		else:
			C.append(n-1)

	new_mem = mem
	for mem_i, c_mem_i in enumerate(C):
		new_mem[mem_i] += mem[c_mem_i]
	
	mem = new_mem

	C = [x+1 for x in C]
	print " ".join(map(str, C))
	# print "C  : %s" % ", ".join(map(str, C))
