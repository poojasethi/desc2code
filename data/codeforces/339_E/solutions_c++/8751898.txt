#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <cctype>
#include <algorithm>
#include <iostream>
using namespace std;

const int N=1000+5;

int n,a[N],ans[11][2];

inline void check(int dep)
{
	bool flag=1;
	for (int i=1;i<=n && flag;++i) flag&=(a[i]==i);
	if (flag)
	{
		printf("%d\n",dep);
		for (int i=dep;i;--i) printf("%d %d\n",ans[i][0],ans[i][1]);
		exit(0);
	}
}

inline void dfs(int dep)
{
	check(dep-1);
	if (dep>3) return;
	for (int i=1;i<=n;++i)
		for (int j=i+1;j<=n;++j)
			if (abs(a[j]-a[i-1])==1 || abs(a[i]-a[j+1])==1)
				ans[dep][0]=i,ans[dep][1]=j,
				reverse(a+i,a+j+1),dfs(dep+1),reverse(a+i,a+j+1);
}

int main(int argc,char **argv)
{
	#ifndef ONLINE_JUDGE
		if (argc==1) freopen("339E.in","r",stdin),freopen("339E.out","w",stdout);
		else freopen(argv[1],"r",stdin),freopen(argv[2],"w",stdout);
	#endif
	scanf("%d",&n);a[n+1]=n+1;
	for (int i=1;i<=n;++i) scanf("%d",&a[i]);
	dfs(1);
	return 0;
}
