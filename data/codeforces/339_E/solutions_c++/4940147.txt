#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>

using namespace std;

const int MAXN = 1000 + 10;

int n, a[MAXN], ans[5][2];

inline bool Check()
{
	for (int i = 1; i <= n; ++i) if (a[i] != i) return 1;
	return 0;
}

inline void Reverse(int l, int r)
{
	for (int i = l; i <= (l + r) / 2; ++i) swap(a[i], a[r - i + l]);
}

void dfs(int deep)
{
	if (!Check())
	{
		printf("%d\n", deep - 1);
		for (int i = deep - 1; i; --i) printf("%d %d\n", ans[i][0], ans[i][1]);
		exit(0);
	}
	if (deep > 3) return;
	for (int l = 1; l <= n; ++l)
		for (int r = l + 1; r <= n; ++r) 
			if (abs(a[l - 1] - a[r]) == 1 || abs(a[l] - a[r + 1]) == 1)
			{
				Reverse(l, r);
				ans[deep][0] = l; ans[deep][1] = r;
				dfs(deep + 1);
				Reverse(l, r);
			}
}

int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) scanf("%d", &a[i]);
	a[0] = 0; a[n + 1] = n + 1;
	dfs(1);
	
	return 0;
}
