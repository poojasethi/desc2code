typedef long long LL;
int get()
{
	int f=0,v=0;char ch;
	while(!isdigit(ch=getchar()))if(ch=='-')break;
	if(ch=='-')f=1;else v=ch-'0';
	while(isdigit(ch=getchar()))v=v*10+ch-'0';
	if(f)return -v;else return v;
}
const int maxn=1003;
int a[maxn],pos[maxn],l[5],r[5],n;
bool dfs(int x)
{
	bool flag=1;
	for(int i=1;i<=n&&flag;i++)flag&=a[i]==i;
	if(flag)return printf("%d\n",x-1),1;
	if(x>3)return 0;
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
			if(abs(a[i]-a[j+1])==1||abs(a[j]-a[i-1])==1)
			{
				reverse(a+i,a+j+1);
				if(dfs(x+1))return printf("%d %d\n",i,j),1;
				reverse(a+i,a+j+1);
			}
	return 0;
}
int main()
{
	n=get();a[n+1]=n+1;
	for(int i=1;i<=n;i++)a[i]=get();
	dfs(1);
	return 0;
}
