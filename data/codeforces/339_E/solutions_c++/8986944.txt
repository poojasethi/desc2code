#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<vector>
using namespace std;
int n;
int a[1002];
vector< pair<int, int> > ans;
int findl()
{
	for(int i = 1; i <= n; i++)
		if(a[i] != i)
			return i;
	return 0;
}
bool solve(int d)
{
	int l = findl();
	if(l == 0)
		return true;
	if(d == 3)
		return false;
	for(int i = 1; i <= n; i++)
		for(int j = i+1; j <= n; j++)
			if(abs(a[i]-a[j+1])==1||abs(a[j]-a[i-1])==1)
			{
				reverse(a+i,a+j+1);
				if(solve(d+1))
				{
					ans.push_back(make_pair(i, j));
					return true;
				}
				reverse(a+i,a+j+1);
			}
	return false;
}
int main()
{
	scanf("%d", &n);
	for(int i = 1; i <= n; i++)
		scanf("%d", &a[i]);
	a[n+1] = n+1;
	solve(0);
	if(ans.empty())
	{
		printf("2\n1 2\n1 2\n");
		return 0;
	}
	int l = ans.size();
	printf("%d\n", l);
	for(int i = 0; i < l; i++)
		printf("%d %d\n", ans[i].first, ans[i].second);
	return 0;
}
