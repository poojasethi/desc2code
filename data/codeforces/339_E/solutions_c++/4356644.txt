#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int a,b;
} pa[4];
int i,j,k,n,m,a[1010];
inline bool judge() {
	int i;
	For(i,1,n) if (a[i]!=i) break;
	return i>n;
}
void dfs(int x) {
	int i,j;
	if (judge()) {
		printf("%d\n",x-1);
		for (i=x-1;i;i--) printf("%d %d\n",pa[i].a,pa[i].b);
		exit(0);
	}
	if (x>3) return;
	For(i,1,n) if (a[i]!=i&&(abs(a[i]-a[i-1])>1||abs(a[i]-a[i+1])>1))
	For(j,i+1,n) if (a[j]!=j&&(abs(a[j]-a[j-1])>1||abs(a[j]-a[j+1])>1)) {
		reverse(a+i,a+j+1);
		pa[x].a=i,pa[x].b=j;
		dfs(x+1);
		reverse(a+i,a+j+1);
	}
}
int main() {
	scanf("%d",&n); a[n+1]=-1;
	For(i,1,n) scanf("%d",&a[i]);
	dfs(1);
	return 0;
}
