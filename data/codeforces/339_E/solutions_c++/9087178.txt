#include<cstdio>
#include<algorithm>
#include<vector>
#include<iostream>
#include<cstring>
using namespace std;
int n,a[2000];
vector<pair<int,int> >ans;
bool dfs(int step){
	int flag=1;
	for(int i=1;i<=n;i++)
		if(a[i]!=i){
			flag=0;
			break;
		}
	if(flag)return 1;
	if(step==3)return 0;
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++){
			if(abs(a[i]-a[j+1])==1||abs(a[j]-a[i-1])==1){
				reverse(a+i,a+j+1);
				if(dfs(step+1)){
					ans.push_back(make_pair(i,j));
					return 1;
				}
				reverse(a+i,a+j+1);
			}
		}
	return 0;
}
int main(){
	scanf("%d",&n);a[n+1]=n+1;
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	/*
	reverse(a+22,a+98);
	reverse(a+27,a+75);
	reverse(a+47,a+70);
	for(int i=1;i<=n;i++)cout<<a[i]<<" ";
	*/
	dfs(0);
	if(ans.size()==0)puts("2\n1 2\n1 2");
	else{
		printf("%d\n",(int)ans.size());
		for(int i=0;i<ans.size();i++)printf("%d %d\n",ans[i].first,ans[i].second);
	}
}
