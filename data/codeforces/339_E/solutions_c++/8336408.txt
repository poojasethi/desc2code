#include<cstdio>
#include<algorithm>
using namespace std;
int a[1005],n;
bool Check()
{
	for(int i=1;i<=n;i++)
		if(a[i]!=i)
			return 0;
	return 1;
}
bool Dfs(int x)
{
	if(Check())
		return printf("%d\n",x),1;
	if(x>2)return 0;
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
			if(abs(a[i-1]-a[j])==1||abs(a[i]-a[j+1])==1)
			{
				for(int k=0;i+k<j-k;k++)
					swap(a[i+k],a[j-k]);
				if(Dfs(x+1))
					return printf("%d %d\n",i,j),1;
				for(int k=0;i+k<j-k;k++)
					swap(a[i+k],a[j-k]);

			}
	return 0;
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	a[n+1]=n+1;
	Dfs(0);
}
