#include<cstdio>
#include<algorithm>
using namespace std;

int n, m, a[1004], ans[5][2];

int left()
{
	for(int i = 1; i <= n; i++)
		if(a[i] != i)
			return i;
	return 0;
}

int right()
{
	for(int i = n; i >= 1; i--)
		if(a[i] != i)
			return i;
	return 0;
}

int find(int x)
{
	for(int i = 1; i <= n; i++)
		if(a[i] == x)
			return i;
	return 0;
}

bool dfs(int s)
{
	int l, r;
	l = left();
	if(!l)
		return 1;
	if(s == 3)
		return 0;
	r = find(l);
	reverse(a+l, a+r+1);
	if(dfs(s+1))
	{
		ans[m][0] = l;
		ans[m++][1] = r;
		return 1;
	}
	reverse(a+l, a+r+1);
	r = right();
	l = find(r);
	reverse(a+l, a+r+1);
	if(dfs(s+1))
	{
		ans[m][0] = l;
		ans[m++][1] = r;
		return 1;
	}
	reverse(a+l, a+r+1);
	return 0;
}



int main()
{
	scanf("%d", &n);
	for(int i = 1; i <= n; i++)
		scanf("%d", &a[i]);
	dfs(0);
	
	printf("%d\n", m);
	for(int i = 0; i < m; i++)
		printf("%d %d\n", ans[i][0], ans[i][1]);
	
	return 0;
}
