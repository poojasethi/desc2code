#include<stdio.h>
#include<string.h>
#include<algorithm>
#define N 1100
using namespace std;
int a[N];
int L[5],R[5],top,n;
void check()
{
	for(int i=1;i<=n;i++)
	{
		if(a[i]!=i)
			return ;
	}
	printf("%d\n",top);
	for(int i=top;i>=1;i--) 
	{
		printf("%d %d\n",L[i],R[i]);
	}
	exit(0);
}
void dfs(int deep)
{
	check();
	if(deep==3)	return ;
	for(int i=1;i<=n;i++)
	{
		for(int j=n;j>=i+1;j--)
		{
			if(abs(a[i]-a[j+1])==1||abs(a[j]-a[i-1])==1)
			{
				L[++top]=i,R[top]=j;
				reverse(a+i,a+j+1);
				dfs(deep+1);
				top--;
				reverse(a+i,a+j+1);
			}
		}
	}
}
int main()
{
	int i,j,k;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",a+i);
	dfs(0);
	return 0;
}
