#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <string>
using namespace std;
const int Maxn=1005;
int n,a[Maxn];

bool check() {
    for (int i=1;i<=n;i++)
        if (a[i]!=i) return false;
    return true;
}

bool dfs(int cnt) {
    if (check()) {
        cout<<cnt-1<<endl;
        return true;
    }
    if (cnt>3) return false;
    for (int i=1;i<n;i++)
        for (int j=i+1;j<=n;j++)
            if (abs(a[i]-a[j+1])==1||abs(a[i-1]-a[j]==1)) {
                reverse(a+i,a+j+1);
                if (dfs(cnt+1)) {
                    cout<<i<<' '<<j<<endl;
                    return true;
                }
                reverse(a+i,a+j+1);
            }
    return false;
}

int main() {
    cin>>n;
    for (int i=1;i<=n;i++) cin>>a[i];
    a[n+1]=n+1;
    dfs(1);
    return 0;
}
