# include <cstdio>
# include <algorithm>
# include <cmath>

using namespace std;

# define REP(i, n) for (int i = 1; i <= n; ++ i)
# define FOR(i, a, b) for (int i = a; i <= b; ++ i)

int n, a[2010], lM[10], rM[10];

bool dfs (int lv) {
    bool succ = true;
    REP (i, n) if (a[i] != i) { succ = false; break; }
    if (succ) {
        printf ("%d\n", lv - 1);
        REP (i, lv - 1) printf ("%d %d\n", lM[lv - i], rM[lv - i]); return true; 
    }
    if (lv == 4) return false;

    REP (i, n) if (a[i] != i && max (abs (a[i - 1] - a[i]), abs (a[i + 1] - a[i])) > 1) {
        FOR (j, i + 1, n) if (a[j] != j && max (abs (a[j - 1] - a[j]), abs (a[j + 1] - a[j])) > 1) {
            reverse (a + i, a + j + 1);
            lM[lv] = i, rM[lv] = j;
            if (dfs (lv + 1)) return true;
            reverse (a + i, a + j + 1);
        }
    }
    return false;
}

int main () {
    scanf ("%d", &n), a[0] = a[n + 1] = -10;
    REP (i, n) scanf ("%d", &a[i]);
    dfs (1);
    return 0;
}
