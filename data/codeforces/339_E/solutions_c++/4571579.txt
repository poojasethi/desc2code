#include <cstdio>
#include <algorithm>
using namespace std;

int N,a[1005],l[5],r[5];

void dfs(int t)
{
	bool ok=1;
	for (int i=1; ok&&i<=N; i++) if (a[i]!=i) ok=0;
	if (ok)
	{
		printf("%d\n",t-1);
		for (int i=t-1; i; i--) printf("%d %d\n",l[i],r[i]);
		exit(0);
	}
	if (t>3) return;
	for (int i=1; i<=N; i++)
		for (int j=i+1; j<=N; j++) if (abs(a[i-1]-a[j])==1||abs(a[i]-a[j+1])==1) l[t]=i,r[t]=j,reverse(a+i,a+j+1),dfs(t+1),reverse(a+i,a+j+1);
}

void doit()
{
	scanf("%d",&N),a[0]=0,a[N+1]=N+1;
	for (int i=1; i<=N; i++) scanf("%d",&a[i]);
	dfs(1);
}

int main()
{
	doit();
	return 0;
}
