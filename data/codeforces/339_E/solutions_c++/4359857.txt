#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

int n,a[1005],l[10],r[10];

int findl()
{
	for(int i=1;i<=n;i++)
	  if(a[i]!=i)
		return i;
	return 0;
}

int findr()
{
	for(int i=n;i>=1;i--)
	  if(a[i]!=i)
		return i;
	return 0;
}

int findp(int x)
{
	for(int i=1;i<=n;i++)
	  if(a[i]==x)
		return i;
}

void rev(int l,int r)
{
	if(l>r)
	{
		int t=l;
		l=r,r=t;
	}
	for(int p=0;p<(r-l+1)/2;p++)
	{
		int t=a[l+p];
		a[l+p]=a[r-p];
		a[r-p]=t;
	}
}

bool dfs(int num)
{
	if(num>4) return false;
	int ll,rr;
	ll=findl();
	if(ll==0)
	{
		printf("%d\n",num-1);
		for(int i=num-1;i>=1;i--)
		  printf("%d %d\n",min(l[i],r[i]),max(l[i],r[i]));
		return true;
	}
	rr=findp(ll);
	l[num]=ll,r[num]=rr;
	rev(ll,rr);
	if(dfs(num+1)) return true;
	rev(ll,rr);
	
	rr=findr();
	ll=findp(rr);
	l[num]=ll,r[num]=rr;
	rev(ll,rr);
	if(dfs(num+1)) return true;
	rev(ll,rr);
	return false;
}

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	  scanf("%d",&a[i]);
	dfs(1);
	return 0;
}
