#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <cstdlib>
using namespace std;
const int N(1e3+200);
int n,a[N],cnt,ans[3][2];
inline bool dfs(int c){
    bool flag=0;
    for(int i(1);i<=n;i++)if(a[i]!=i){flag=1;break;}
    if(!flag){
        cout<<c<<endl;
        for(int i(c-1);i>=0;i--)printf("%d %d\n",ans[i][0],ans[i][1]);
        exit(0);
    }
    if(c>=3)return 0;
    for(int i(1);i<=n;i++)
        if(i!=a[i]&&(abs(a[i]-a[i-1])!=1||abs(a[i+1]-a[i])!=1))
            for(int j(i+1);j<=n;j++)
                if(j!=a[j]&&(abs(a[j]-a[j-1])!=1||abs(a[j+1]-a[j])!=1)){
                    reverse(a+i,a+j+1);
                    ans[c][0]=i;ans[c][1]=j;
                    dfs(c+1);
                    reverse(a+i,a+j+1);
                }
    return 0;
}
int main(){
    scanf("%d",&n);
    for(int i(1);i<=n;i++)scanf("%d",a+i);a[0]=a[n+1]=10086;
//  a[0]=0;a[n+1]=n+1;
    dfs(0);
    return 0;
