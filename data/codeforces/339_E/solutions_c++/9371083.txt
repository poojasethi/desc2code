#include <cstdio>
#include <algorithm>
#include <cstdlib>
int a[1009],sz[10],l[4],r[4],n,m;
bool check()
{
	for (int i = 1; i <= n; ++i)
		if (a[i] != i) return false;
	return true;
}
void dfs(int step)
{
	if (check()) {
		printf("%d\n",step - 1);
		for (int i = 1; i < step; ++i) printf("%d %d\n",l[step - i],r[step - i]);
		exit(0);
	}
	if (step > 3) return;
	for (int i = 1; i <= n; ++i)
		for (int j = i + 1; j <= n; ++j) {
			if (abs(a[i] - a[j + 1]) != 1 && abs(a[i - 1] - a[j]) != 1) continue;
			std::reverse(a + i,a + j + 1);
			l[step] = i; r[step] = j;
			dfs(step + 1);
			std::reverse(a + i,a + j + 1);
		}
}
int main()
{
	#ifndef ONLINE_JUDGE

	freopen("339E.out","w",stdout);
	#endif
	scanf("%d",&n);
	for (int i = 1; i <= n; ++i) scanf("%d",&a[i]);
	dfs(1);
}
