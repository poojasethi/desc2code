#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 1111;

int a[N], n;
int ans[N][2];

bool check() {
    for (int i = 1; i <= n; ++ i)
        if (a[i] != i) return false;
    return true;
}

void dfs(int u) {
    if (check()) {
        printf("%d\n", u);
        for (int i = u; i >= 1; -- i)
            printf("%d %d\n", ans[i][0], ans[i][1]);
        exit(0);
    }
    if (u == 3) return;
    for (int i = 1; i <= n; ++ i)
        if (a[i] != i && (abs(a[i] - a[i-1]) > 1 || abs(a[i] - a[i+1]) > 1)) 
        	for (int j = i+1; j <= n; ++j)
        		if (a[j] != j && (abs(a[j] - a[j-1]) > 1 || abs(a[j] - a[j+1]) > 1)) {
					ans[u + 1][0] = i;
		            ans[u + 1][1] = j;
		            reverse(a + i, a + j + 1);
		            dfs(u + 1);
		            reverse(a + i, a + j + 1);
		        }
}

int main() {
    scanf("%d", &n);
    a[0] = -9999; a[n+1] = 9999;
    for (int i = 1; i <= n; ++ i) scanf("%d", a + i);
    dfs(0);
    return 0;
}
