#include <iostream>
#include <algorithm>
#include <stdlib.h>
using namespace std;
int n,a[1005];
int h[10][5];

inline bool check(){
	for(int i=0;i<n;i++)if(a[i]!=i)return false;
	return true;
}

void dfs(int x){
	if(check()){
		cout<<x<<endl;
		for(int i=x-1;i>=0;i--)cout<<h[i][0]<<" "<<h[i][1]<<endl;
		exit(0);
	}
	if(x>=3)return;
	for(int i=0;i<n;i++){
		if(a[i]!=i&&(i<=0||abs(a[i]-a[i-1]>1)||i>=n-1||abs(a[i]-a[i+1])>1)){
			for(int j=i+1;j<n;j++){
				if(a[j]!=j&&(j<=0||abs(a[j]-a[j-1])>1||j>=n-1||abs(a[j]-a[j+1])>1)){
					reverse(a+i,a+j+1);
					h[x][0]=i+1;h[x][1]=j+1;
					dfs(x+1);
					reverse(a+i,a+j+1);
				}
			}
		}
	}
}

int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];a[i]--;
	}
	dfs(0);
	return 0;
}
