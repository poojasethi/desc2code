#!/usr/bin/python
n = int(raw_input());
a = map(int, raw_input().split(' '));
a.sort(reverse=True);
for i in range(1, n): 
    a[i] = a[i] + a[i - 1];
m = int(raw_input());
ans = [0 for i in range(100005)];
w = map(int, raw_input().split(' '));
pr = ''
out = []
for i in range(m):
        if (ans[w[i]] == 0 ) :
            j = 0
            k = w[i]
            q = 1
            ans[w[i]] = 0;
            while (True) :
                if (j + k >= n) : 
                    ans[w[i]] = ans[w[i]] + (a[n - 1] - a[j]) * q
                    break
                else: 
                    ans[w[i]] = ans[w[i]] + (a[j + k] - a[j]) * q
                    q = q + 1
                    j = j + k
                    k = k * w[i]
        out.append(str(ans[w[i]]));
print ' '.join(out)
