R=lambda:map(int,raw_input().split())
n=input()
a=sorted(R(),reverse=True)
for i in xrange(1,n):a[i]+=a[i-1]
t=[a[n-1]-a[0]]*100100
for k in xrange(1,n+1):
  s,u,p,d=0,0,1,0
  while u<n:
    p*=k
    d+=1
    s+=(a[min(u+p,n-1)]-a[u])*d
    u+=p
  t[k]=s
q=input()
print ' '.join(map(lambda x:str(t[x]),R()))
