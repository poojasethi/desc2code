#include<iostream>
#include<algorithm>
using namespace std;
__int64 s[100001]={0};
__int64 a[100001]={0};
__int64 ans[100001]={0};
int main()
{
    __int64 n,i,k,m;
    __int64 sum,kk;
    cin>>n;
    for(i=0;i<n;i++)cin>>a[i];
    sort(a,a+n);    s[0]=a[0];
    for(i=1;i<n;i++)s[i]=s[i-1]+a[i];
    cin>>m;
    while(m--){
        cin>>k;
        if(ans[k]==0)
        {
          kk=1;
          sum=0;
          i=n-2;
          while(i>=0){
            kk*=k;
            sum+=s[i];
            i-=kk;
          }
          ans[k]=sum;
        }
        cout<<ans[k]<<" ";
    }
    return 0;
}
