#include <bits/stdc++.h>

using namespace std;

typedef long long int ll;

const int maxl = 1000 * 100 + 5;

ll a [ maxl ];
int n, q;

ll ans [ maxl ];

int main (){
	ios_base::sync_with_stdio ( false );

	cin >> n;
	for ( int i = 0 ; i < n ; i++ )
		cin >> a [ i + 1 ];

	sort ( a + 1, a + n + 1 );

	a [ 0 ] = 0;
	for ( int i = 0 ; i < n ; i++ )
		a [ i + 1 ] += a [ i ];

	cin >> q;
	for ( int i = 0 ; i < q ; i++ ){
		int k;
		cin >> k;

		ll stp = 1;
		if ( !ans [ k ] )
			for ( ll j = n - 1 ; j >= 0 ; j -= stp ){
				ans [ k ] += a [ j ];
				stp *= k;
			}

		cout << ans [ k ] << " ";
	}
}
