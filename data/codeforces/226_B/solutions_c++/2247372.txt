#include <stdio.h>
#define F(I,B,E) for(int I=B; I<E; I++)
#define S "%I64d"
typedef long long _ll;

void qrs(_ll *a, int n, _ll b) {
  if(n>1 && b) {
    int k=0;
    _ll t;
    for(int i=0; i<n; i++)
      if(!(a[i]&b))
        t=a[i], a[i]=a[k], a[k++]=t;
    qrs(a,k,b>>=1);
    qrs(a+k,n-k,b);
  }
}

main() {
  _ll n,a[100100],m[100100]={},q,k,p,d;
  scanf(S,&n);
  F(i,0,n)
    scanf(S,a+i);
  qrs(a,n,1LL<<33);
  F(i,1,n)
    a[i]+=a[i-1];
  scanf(S,&q);
  F(i,0,q) {
    scanf(S,&k);
    if(!m[k]) {
      p=n-2;
      d=1;
      while(p>=0)
        m[k]+=a[p],
        p-=(d*=k);
    }
    printf("%I64d ",m[k]);
  }
  return 0;
}
