#include <algorithm>
#include <iostream>
using namespace std;
const int MaxN= 100010;
typedef long long ll;
int N, Q, K;
ll a[MaxN], s[MaxN], ans[MaxN], r;
int cmp(int a, int b){
    return a>b;
}
int main(){
    cin>> N;
    for(int i= 1;i<= N;++ i) cin>> a[i]; sort(a+1, a+1+N, cmp);
    for(int i= 1;i<= N;++ i) s[i]= s[i-1] + a[i];
    for(int i= 1;i<= N;++ i)
        for(ll l= 1ll, b= 1ll, c= 0ll;l< N;){
            b*= i;
            ++ c;
            r= l+b>N ? N : l+b;
            ans[i]+= c*(s[r]-s[l]);
            l= r;
        }
    cin>> Q;
    while(Q --){
        cin>> K;
        cout<< ans[min(K, N)]<< " \n"[Q==0];
    }
    return 0;
}
