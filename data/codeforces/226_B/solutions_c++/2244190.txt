#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef __int64 lld;
const lld maxn = 100003;

lld a[maxn];
lld b[maxn];
lld c[maxn];

int main() {
	lld n, m, i, k, t;
	scanf("%I64d",&n);
	for (i=1; i <= n; i++) {
		scanf("%I64d",&a[i]);
	}
	sort(a+1, a+n+1);
	for (i=1; i <= n; i++) {
		b[i] = b[i-1] + a[i];
	}
	for (k=1; k <= n; k++) {
		lld t = k;
		for (i=n-1; i > 0; i-=t, t*=k) {
			c[k] += b[i];
		}
	}
	scanf("%I64d",&m);
	while (m--) {
		scanf("%I64d",&k);
		if (k >= n-1) 
			printf("%I64d ",b[n-1]);
		else
			printf("%I64d ",c[k]);
	}
	printf("\n");
	return 0;
}
