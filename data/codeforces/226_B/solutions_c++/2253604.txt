#include<iostream>
#include<algorithm>
#include<cstdlib>
#include<cstdio>
#include<cstring>
using namespace std;
long long a[100009];
long long sum[100009];
long long n;



main(){
	long long i,t;
	scanf("%I64d",&n);
	
	for (i=0;i<n;i++)
		scanf("%I64d",&a[i]);
	sort(a,a+n);

	long long t1=0;
	sum[0]=0;
	for (i=1;i<=n;i++)
	t1+=sum[i]=sum[i-1]+a[i-1];
	scanf("%I64d",&t);
	t1-=sum[n];	
	//for (i=0;i<n;i++) cout<<a[i]<<endl;
	
	while (t--){
		long long i,j,k;
		scanf("%I64d",&k);
		long long ans;
		if (k==1) {printf("%I64d ",t1);}else{
			i=n-1;
			j=1;
			for (ans=0;i>=0;i-=j*=k) ans+=sum[i];
		
			printf("%I64d ",ans);
		}
	}
}
