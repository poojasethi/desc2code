#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
long long a[100010],b[100010],j,k,s,c,ans,temp;
int n,m,i; 
int main()
{
	cin>>n;
	for(i=1;i<=n;i++) scanf("%I64d",&a[i]);
	sort(a+1,a+n+1);
	reverse(a+1,a+n+1);
	for(i=1;i<=n;i++) b[i]=b[i-1]+a[i],temp+=a[i]*(i-1);
	cin>>m;
	for(i=1;i<=m;i++)
	{
		scanf("%I64d",&k);
		if(k==1) ans=temp; else 
		for(j=1,s=0,c=0,ans=0;s<n;j*=k,c++)
			if(s+j>=n) ans+=(b[n]-b[s])*c,s=n;
			else ans+=(b[s+j]-b[s])*c,s+=j; 
		printf("%I64d\n",ans); 
	}
	return 0;
}
