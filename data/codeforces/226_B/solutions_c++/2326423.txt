#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define NN 100000
#define ll long long
ll v[NN];

int main(){

	int n ;  scanf("%d" , &n );
	for( int i = 0 , t ; i < n ; ++i ){
		scanf("%d" , &t  );
		v[i] = t;
	}
	sort( v , v+ n , greater<ll>() );
	ll onea = 0;
	for( int i = n-2, t = v[n-1] ;  i >= 0 ;  --i){
		 v[i] += v[i+1]; 
		 onea += v[i+1];
	} 

	int q; scanf("%d" , &q);
	for( int i = 0 ; i < q ; ++i ){
		int k ;
		scanf("%d" , &k );
		ll ans = 0;
		
		if( k == 1 ){
			ans = onea;
		}else{
			for( ll acc = 1 , kp = k ; acc < n  ; acc += kp , kp *= k  )
				ans +=  v[acc];
		}	
		if( i ) cout << " ";
		cout << ans;
	}
	puts("");
	return 0;
}
