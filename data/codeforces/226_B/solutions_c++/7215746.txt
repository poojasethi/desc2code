#include <iostream>
#include <algorithm>

using namespace std;

typedef long long ll;

const int MAXN = 1e5 + 10;
ll a[MAXN], ps[MAXN], ans[MAXN];

int main() {
	ll n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	sort(a, a + n);
	reverse(a, a + n);
	for (int i = 0; i < n; i++)
		ps[i + 1] = ps[i] + a[i];
	for (int i = 1; i < MAXN; i++) {
		ll cur = i, cnt = 1, last = 1, left = n - 1;
		while (left > 0) {
			ans[i] += (ps[min(n, last + cur)] - ps[last]) * cnt;
			left -= cur;
			last += cur;
			cur *= i;
			cnt++;
		}
	}
	int q;
	cin >> q;
	for (int i = 0; i < q; i++) {
		int k;
		cin >> k;
		cout << ans[k] << " ";
	}
}
