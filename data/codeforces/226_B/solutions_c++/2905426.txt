#include <algorithm>
#include <stdio.h>
#define  maxn      100002
using    namespace std;
typedef  long long llint;

llint n,m,q,tot;
llint ar[maxn];
llint sum[maxn];

int main()
{
  scanf("%I64d",&n);
  for(llint i=1 ; i<=n ; i++)
    scanf("%I64d\n",&ar[i]);
  sort(ar+1,ar+n+1);
  for(llint i=1 ; i<=n ; i++)
  {
    sum[i]=sum[i-1]+ar[i];
    tot+=sum[i];
  }
  scanf("%I64d",&m);
  while(m--)
  {
    scanf("%I64d",&q);
    llint ans=0;
    if(q==1) ans=tot-sum[n];
    else
      for(llint i=n-1,tmp=1 ; i>=1 ; tmp*=q,i-=tmp)
        ans+=sum[i];
    printf("%I64d\n",ans);
  }
  return 0;
}
