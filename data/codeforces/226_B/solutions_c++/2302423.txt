#include<stdio.h>
#include <algorithm>
#define N 100010
using namespace std;
int main(){
	 long n, m[N], q, k,i,j;
	 long long t[N], ans[N],s,x;
	 scanf("%ld",&n);
	 for(i=0;i<n;i++)
	  scanf("%ld",&m[i]);
	 sort(m,m+n);
	 t[0] = m [0];
	 for(i=1;i<n;i++)
		t[i] = t[i-1]+m[i];
	  scanf("%ld",&q);
		for(j=0;j<q;j++)
		{
			scanf("%ld",&k);
		  if(ans[k]==0)
		  {
				long long cnt = 0;
				for(x=n-2, s=1;x>=0;s=s*k, x=x-s)
				  {
					 cnt += t[x];
				  }
				ans[k] = cnt;
		  }
		  if(j==q-1)
		  printf("%I64d",ans[k]);
		  else
		  printf("%I64d ",ans[k]);
	   }
	return 0;
}
