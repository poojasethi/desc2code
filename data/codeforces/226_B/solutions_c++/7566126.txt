#include <bits/stdc++.h>
using namespace std;

#define LL long long int 
#define N 100001
LL a[N], sum[N];

int main() {
	int n, i, k, q, l, r;
	LL ans1 = 0, ans, y, z, x;
	cin>>n;
	for(i = 1; i <= n; i++){
		cin>>a[i];
	}
	sort(a + 1, a + n + 1);
	sum[0] = 0;
	for(i = 1; i <= n; i++){
		sum[i] = sum[i - 1] + a[i];
		ans1 += sum[i];
	}
	ans1 -= sum[n];
	cin>>q;
	while(q--){
		cin>>k;
		if(k != 1){
			r = n - 1;
			y = k;
			ans = 0;
			z = 1;
			while(r >= 1){
				l = max(0LL, r - y);
				ans += z * (sum[r] - sum[l]);
				z++;
				y *= k;
				r = l;
			}
		}
		else{
			ans = ans1;
		}
		printf("%I64d\n", ans);
		
	}
	return 0;
}
