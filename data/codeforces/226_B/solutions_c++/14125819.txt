#include <cstdio>
#include <algorithm>
using namespace std;
int main()
{
	int n,q,k,a[100000];
	long long x,shift,ans,total=0,sum[1+100000]={0};
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	sort(a,a+n);
	sum[0]=a[0];
	for(int i=1;i<n;i++)
	{
		sum[i]=sum[i-1]+a[i];
		total+=sum[i-1];
	}
	scanf("%d",&q);
	for(int i=0;i<q;i++)
	{
		scanf("%d",&k);
		if(k==1)
			ans=total;
		else
		{
			x=k;
			ans=0;
			shift=n-2;
			while(shift>=0)
			{
				ans+=sum[shift];
				shift-=x;
				x*=k;
			}
		}
		printf("%I64d%c",ans,i!=q-1?' ':'\n');
	}
	return 0;
}
