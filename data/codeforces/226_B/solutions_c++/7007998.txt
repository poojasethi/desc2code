#include<cstdio>
#include<algorithm>
#include<queue>

using namespace std;
typedef long long ll;
#define N 100010
ll a[N], sum[N], res[N];

int main() {
	int n, Q, k;
	scanf("%d", &n);
	for(int i = 1; i <= n; i ++)		scanf("%I64d", a + i);
	sort(a + 1, a + n + 1);
	for(int i = 1; i <= n; i ++)
		sum[i] = sum[i - 1] + a[i];
	for(int k = 1; k < n; k ++) {
		for(ll i = n - 1, j = k; i >= 1; i -= j, j *= k) {
			res[k] += sum[i];
		}
	}
	scanf("%d", &Q);
	while(Q --) {
		scanf("%d", &k);
		if(k > n - 1)	
			k = n - 1;
		printf("%I64d ", res[k]);
	}
	puts("");
	return 0;
}
