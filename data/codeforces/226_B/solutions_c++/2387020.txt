#include <algorithm>
#include <stdio.h>
#define  maxn      100002
using    namespace std;
typedef  long long llint;

llint n,m,tot;
llint a[maxn];
llint sum[maxn];

int main()
{
  scanf("%I64d",&n);
  for(llint i=1 ; i<=n ; i++)
    scanf("%I64d",&a[i]);
  sort(a+1,a+n+1);
  for(llint i=1 ; i<=n ; i++)
  {
    sum[i]=sum[i-1]+a[i];
    tot+=sum[i];
  }
  scanf("%I64d",&m);
  while(m--)
  {
    llint q;
    scanf("%I64d",&q);
    if(q==1)
      printf("%I64d ",tot-sum[n]);
    else
    {
      llint ans=0;
      for(llint i=n-1,tmp=1 ; i>=1 ; tmp=tmp*q,i-=tmp)
        ans+=sum[i];
      printf("%I64d ",ans);
    }
  }
  printf("\n");
  return 0;
}
