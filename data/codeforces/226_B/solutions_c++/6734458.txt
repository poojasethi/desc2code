#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,cen;
int64 a[N],b[N],an,s,sum;
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%I64d",&a[i]);
	sort(a+1,a+n+1); reverse(a+1,a+n+1);
	For(i,1,n) b[i]=b[i-1]+a[i],sum+=a[i]*(i-1);
	scanf("%d",&m);
	For(i,1,m) {
		scanf("%d",&k);
		k=min(k,n-1);
		if (k==1) an=sum;
		else {
			an=0;
			for (j=2,s=k,cen=1;j<=n;cen++) {
				int A=min(0ll+n,j+s-1);
				an+=(b[A]-b[j-1])*cen;
				j=A+1;
				s*=k;
			}
		}
		printf("%I64d ",an);
	}
	printf("\n");
	return 0;
}
