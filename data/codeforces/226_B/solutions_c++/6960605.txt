#include <cstdio>
#include <algorithm>
using namespace std;
#define N 123456

int n, q, k; long long res[N], cum[N];
int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) scanf("%lld", cum+i);
	sort(cum+1, cum+n+1);
	for (int i = 2; i <= n; i++) cum[i] += cum[i-1];
	for (int i = 1; i < n; i++) {
		long long st = n, end = n, sz = 1, mult = 0;
		while (1) {
            res[i] += mult*(cum[end] - cum[st-1]);
            if (st==1) break;
            end = st-1; sz *= i; mult++;
            st = max(1LL, end-sz+1);
		}
	}
	scanf("%d", &q);
	for (int i = 0; i < q; i++) {
		scanf("%d", &k);
		printf("%lld ", k >= n ? res[n-1] : res[k]);
	}

	return 0;
}
