#include<cstdio>
#include<algorithm>
using namespace std;
#define ll __int64
const int N=100005;
int a[N];
ll s[N];
int main()
{
	ll ans,a1;
	int i,k,n,m;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	a1=0;
	for(i=1;i<n;i++)
	{
		a1+=(ll)(n-i)*a[i];
		s[i]=s[i-1]+a[i];
	}
	scanf("%d",&m);
	while(m--)
	{
		scanf("%d",&k);
		if(k==1)printf("%I64d ",a1);
		else
		{
			ans=0;
			ll b=k;
			int p=n-1,c=1;
			while(p>=b)
			{
				ans+=(s[p]-s[p-b])*c;
				p-=b;
				b*=k;
				c++;
			}
			ans+=s[p]*c;
			printf("%I64d ",ans);
		}
	}
	return 0;
}
