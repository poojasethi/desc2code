#include<stdio.h>
#include<algorithm>
using namespace std;

long long sum[200001]={0},pile[200001]={0},m,p;

int main(){
    int i,n,q,k;
    scanf("%d",&n);
    for(i=1;i<=n;i++){
        scanf("%I64d",&pile[i]);
    }
    sort(pile+1,pile+n+1);
    for(i=1;i<=n;i++){
        pile[i]+=pile[i-1];
    }
    for(i=1;i<=n;i++){
        p=n-1;
        m=i;
        while(p>=1){
            sum[i]+=pile[p];
            p-=m;
            m*=i;
        }
    }
    scanf("%d",&q);
    for(i=1;i<=q;i++){
        scanf("%d",&k);
        printf("%I64d ",sum[min(k,n)]);
    }
    return 0;
}
