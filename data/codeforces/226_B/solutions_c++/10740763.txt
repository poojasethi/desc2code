#include <iostream>
#include <algorithm>
using namespace std;

const int MAXN = 100000+5;

int a[MAXN];
long long l, t, x;
long long ans;
long long s[MAXN];
long long one = 0;
int n, q;

int main (int argc, char* argv[]) {
    cin >> n;
    for (int i = 0; i < n; ++i) 
        cin >> a[i];
    sort(a, a+n);
    for (int i = 0; i < n; ++i) 
        one += s[i+1] = s[i] + a[i];
    cin >> q;
    while (q--) {
        cin >> x;
        if (x == 1) ans = one - s[n];
        else 
            for (l = 1, ans = 0, t = n-1; t >= 0; t -= l *= x)
                ans += s[t];
        cout << ans << " ";
    }
    return 0;
}
