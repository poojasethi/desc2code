#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
const int maxn = 100010;
int dat[maxn], datq[maxn], n, q;
long long ans[maxn], sum[maxn];
int main()
{
	scanf("%d", &n);
	for (int i = 0; i < n; i++) scanf("%d", &dat[i]);
	scanf("%d", &q);
	for (int i = 0; i < q; i++) scanf("%d", &datq[i]);
	sort(dat, dat + n);
	sum[0] = dat[0];
	for (int i = 1; i < n; i++) sum[i] = sum[i - 1] + dat[i];
	for (int i = 1; i <= n; i++)
	{
		long long now = 1;
		for (long long j = n - 2; j >= 0; j -= now *= i) ans[i] += sum[j];
	}
	printf("%I64d", ans[min(datq[0], n)]);
	for (int i = 1; i < q; i++) printf(" %I64d", ans[min(datq[i], n)]);
	printf("\n");
	return 0;
}
