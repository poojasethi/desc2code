#include <cstdio>
#include <algorithm>
#define N 100000
using std::sort;

typedef long long LL;
LL pre[N+1];
LL ans[N+1];

int main(){
    int n;
    scanf("%d",&n);

    for(int i=0;i<n;++i)
        scanf("%I64d",&pre[i]);
    sort(pre,pre+n);
    
    for(int i=1;i<n;++i)
        pre[i] += pre[i-1];
    
    for(LL i=1;i<=n;++i){
        for(LL j=n-2,k=i;j>=0;j-=k,k*=i)
            ans[i] += pre[j];   
    }
    int q;
    scanf("%d",&q);
    for(int i=0;i<q;++i){
        int tmp;
        scanf("%d",&tmp);
        if( tmp > n ) tmp = n;
        printf("%I64d ",ans[tmp]);
    }
    printf("\n");
    return 0;   
}
