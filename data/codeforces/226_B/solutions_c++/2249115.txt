#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;

typedef __int64 lld;
const int maxn=100005;

lld a[maxn];
lld b[maxn];
lld z[maxn];

int main(){
	lld n,m,i,f,j;
	scanf("%I64d",&n);
	for(i=0;i<n;i++)
		scanf("%I64d",&a[i]);
	sort(a,a+n);
	b[0]=a[0];
	for(i=1;i<n;i++)
		b[i]=b[i-1]+a[i];
	for(i=1;i<n;i++){
		lld t=1;
		for(j=n-2;j>=0;j-=t){
			z[i]+=b[j];
			t*=i;
		}
	}
	scanf("%I64d\n",&m);
	for(i=0;i<m;i++){
		scanf("%I64d",&f);
		if(f>n-1)
			printf("%I64d ",z[n-1]);
		else
			printf("%I64d ",z[f]);
	}
	puts("");
	return 0;
}
