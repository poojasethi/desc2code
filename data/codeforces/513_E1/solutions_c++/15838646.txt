#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
const int N=10000;
const int INF=1e9;
int a[N],s1[N][300],s2[N][300],n,m,s[N];
int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		s[i]=s[i-1]+a[i];
	}
	for(int i=0;i<=n;i++)
		for(int j=0;j<=m;j++)
		{
			s1[i][j]=-INF;
			s2[i][j]=-INF;
		}
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=i;j++)
		{
			int ss=s[i]-s[i-j];
			s1[i][1]=max(s1[i][1],ss);
			s2[i][1]=max(s2[i][1],-ss);
		}
	}
	for(int j=2;j<=m;j++)
		for(int i=1;i<=n;i++)
			if (j<=i)
			{
				for(int k=1;k<i;k++)
				{
					int ss=s[i]-s[k];
					s1[i][j]=max(s1[i][j],s1[k][j-1]);
					s1[i][j]=max(s1[i][j],s2[k][j-1] + 2*ss);
					s2[i][j]=max(s2[i][j],s2[k][j-1]);
					s2[i][j]=max(s2[i][j],s1[k][j-1] -2*ss);
				}
			}
	int ans=-INF;
	for(int i=1;i<=n;i++)
		for(int j=i;j<=n;j++)
		{
			int ss=s[j]-s[i-1];
			ans=max(ans,s1[i-1][m-1] - ss);
			ans=max(ans,s2[i-1][m-1] + ss);
		}
	cout<<ans<<endl;
}
