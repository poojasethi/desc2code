#include<cstdio>
#include<cstring>
#define MAXM 55
#define MAXN 405
#define max(x,y) x>y?x:y
int n,m;
int f[MAXN][MAXN][MAXM],s[MAXN];
int G(int x,int y)
{
    return x>y?x-y:y-x;
}
void DP()
{
    int i,j,k,l,L,ans=0;
    memset(f,0,sizeof(f));
    for(i=1,s[0]=0;i<=n;i++)
    {
        scanf("%d",&s[i]);
        s[i]+=s[i-1];
    }
    for(k=2;k<=m;k++)
        for(i=2;i<=n;i++)
            for(j=i-1;j>=k-1;j--)
                for(l=j-1;l>=k-2;l--)
                    f[i][j][k]=max(f[i][j][k],f[j][l][k-1]+G(s[i]-s[j],s[j]-s[l]));
    for(i=1;i<=n;i++)
        for(j=1;j<=n;j++)
            ans=max(ans,f[i][j][m]);
    printf("%d\n",ans);
}

int main()
{
//  freopen("zin1.txt","r",stdin);
    while(~scanf("%d%d",&n,&m)) DP();
    return 0;
}
