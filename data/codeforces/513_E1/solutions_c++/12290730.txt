#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
#include<queue>
using namespace std;
const int maxn=508;
const int INF=0x7f7f7f7f;
int dp[maxn][maxn][68];
int a[maxn];
int absolut(int x)
{
    if(x<0)
        return -x;
    return x;
}
int main()
{
    int i,j,n,k;
    scanf("%d%d",&n,&k);
    for(i=1;i<=n;i++)
        scanf("%d",&a[i]);
    int ans=0;
    int cas=1;
    for(i=1;i<=n;i++)
    {
        int si=0;
        for(int ii=i;ii>1;ii--)
        {
            si+=a[ii];
            j=ii-1;
            int sj=0;
            for(int jj=j;jj>=0;jj--)
            {
                sj+=a[jj];
                for(int d=1;d<=jj and d+1<=k;d++)
                {
                    dp[ii][i][d+1]=max(dp[ii][i][d+1],dp[jj][j][d]+absolut(si-sj));
                }
            }
            ans=max(ans,dp[ii][i][k]);
        }
    }
    printf("%d\n",ans);
    //printf("%d\n",cas);
    return 0;
}
