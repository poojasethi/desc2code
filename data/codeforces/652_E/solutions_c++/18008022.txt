#include<bits/stdc++.h>

using namespace std;

#define mp make_pair
#define mp3(x,y,z) mp(x,mp(y,z))
#define st first
#define nd second
#define N 300005

vector < pair < int , pair < int , int > > > V[N];

stack < int > s; 

int low[N],D[N],Bcc[N],BT[N],T,bcc,n,w,x,y,z;

void g(int x){
	D[x] = 1;
	BT[ Bcc[x] ]++;
	for(int i=0; i<V[x].size(); i++)
		if(!D[ V[x][i].st ])
			g(V[x][i].st);

	if(x == y){
		for(int i=1;i<=n;i++)
			for(z=0;z<V[i].size();z++)
				if(V[i][z].nd.st && BT[ Bcc[ i ] ] && BT[ Bcc[ V[i][z].st ] ]){
					puts("YES");
					exit(0);
				}
		puts("NO");
		exit(0);
	}

	BT[ Bcc[x] ]--;
}

void f(int x, int b){
	D[x] = low[x] = ++T;
	s.push(x);
	for(int i=0; i<V[x].size(); i++){
		if(b == V[x][i].st)
			continue;
		if(!low[ V[x][i].st ])
			f(V[x][i].st,x);
		low[x] = min(low[x] , low[ V[x][i].st ]);
	}
	if(low[x] == D[x]){
		bcc++;
		for(z=0; z != x; ){
			z = s.top();
//			cout << bcc << " --> " << z << endl;
			Bcc[z] = bcc;
			s.pop();
		}
	}
}
int main(){

	cin >> n >> w;

	while(w--){
		cin >> x >> y >> z;
		V[x].push_back( mp3(y,z,w) );
		V[y].push_back( mp3(x,z,w) );
	}

	cin >> x >> y;

	f(1,-1);

	memset(D , 0 , sizeof D);

	g(x);

	return 0;
}
