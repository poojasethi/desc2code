#include <bits/stdc++.h>
using namespace std;
multiset <long long> S;
int main() {
	ios::sync_with_stdio(0);
	int n, k;
	cin >> n >> k;
	for(int i = 0; i < k; i++) S.insert(0LL);
	long long s, t;
	for(int i = 0; i < n; i++) {
		cin >> s >> t;
		multiset <long long> :: iterator it = S.begin();
		long long x = *it;
		S.erase(it);
		if(x < s) {
			cout << s + t << endl;
			S.insert(s + t);
		} else {
			cout << x + t << endl;
			S.insert(x + t);
		}
	}
	return 0;
}
