#include <iostream>
#include <cstdio>
#include <queue>

using namespace std;

int main()
{
	long long n(0), k(0);
	scanf("%I64d%I64d", &n, &k);

	priority_queue<long long> ends;

	for(long long i = 0; i < n; ++i) {
		long long t, e;

		scanf("%I64d%I64d", &t, &e);
		if(ends.size() < k) {
			printf("%I64d\n", t + e);
			ends.push(-(t + e));
			continue;
		}

		long long new_end = (-ends.top() > t ? -ends.top() : t) + e; 
		printf("%I64d\n", new_end);
		ends.pop();
		ends.push(-new_end);
	}
	return 0;
}
