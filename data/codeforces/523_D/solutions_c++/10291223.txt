#include <queue>
#include <stdio.h>
#include <algorithm>
using namespace std;
priority_queue <__int64>a;
__int64 n, k, l, r;
int main(){
	scanf("%I64d %I64d", &n, &k);
	for(int i = 0; i < k; ++ i){
		scanf("%I64d %I64d", &l, &r);
		printf("%I64d\n", l + r);
		a.push(-(l + r));
	}
	for(int i = k; i < n; ++ i){
		scanf("%I64d %I64d", &l, &r);
		__int64 x = max(l, -a.top());
		printf("%I64d\n", x + r);
		a.pop();
		a.push(-(x + r));
	}
	return 0;
}
