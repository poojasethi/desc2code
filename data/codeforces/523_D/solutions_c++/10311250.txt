#include<iostream>
#include<algorithm>
#include<queue>
#include<stdio.h>
using namespace std;
priority_queue<long long>q;
long long x,n,k,a,b;
main()
{
      cin>>n>>k;
      
      for(int i=1;i<=k;i++)q.push(0);
      for(int i=1;i<=n;i++)
      { 
              scanf("%d%d",&a,&b);
              x=-q.top();q.pop();
              x=max(a,x);
              printf("%I64d\n",b+x);
              x+=b;
              q.push(-x);       
      }
      cin>>n;
}
