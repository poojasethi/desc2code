#include <iostream>
#include <queue>
#include <stdio.h>
#define ll long long
using namespace std;
priority_queue<ll> q;
int main()
{
    ll n, k, i, j, x, y;
    cin >> n >> k;
    for(i = 1; i <= k; i ++)
    {
        q.push(0);
    }
    for(i = 1; i <= n; i ++)
    {
        scanf("%I64d %I64d", &x, &y);
        ll d = max(-1 * q.top(), x);
        q.pop();
        d += y;
        printf("%I64d\n", d);
        q.push(-d);
    }
    return 0;
}
