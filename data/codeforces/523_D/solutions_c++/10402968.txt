#include <bits/stdc++.h>
#include <queue>

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    long long n, k, s, m, real_start;
    cin >> n >> k;
    priority_queue <long long> q;
    for (int j=0; j<k; j++)
        q.push(0);
    for (int i=0; i<n; i++)
    {
        cin >> s >> m;
        real_start=max(s,-1*q.top())+m;
        q.pop();
        printf("%I64d\n",real_start);
        q.push(-1*real_start);
    }
    return 0;
}
