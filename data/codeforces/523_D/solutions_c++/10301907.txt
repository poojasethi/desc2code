#include <stdio.h>
#include <algorithm>
#include <queue>
using namespace std;
priority_queue <__int64> p;
int n, k;
__int64 s, m, t;
int main(){
	scanf("%d %d", &n, &k);
	for(int i = 0; i < n; ++ i){
		scanf("%I64d %I64d", &s, &m);
		if(i < k){
			p.push(-(s + m));
			printf("%I64d\n", s + m);
		}
		else{
			t = -p.top();
			p.pop();
			t = max(s + m, m + t);
			printf("%I64d\n", t);
			p.push(-t);
		}
	}
	return 0;
}
