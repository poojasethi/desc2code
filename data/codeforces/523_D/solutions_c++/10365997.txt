typedef long long lint;

const lint MAXN = 1e6;

lint N,K,ans;
priority_queue < lint > pq;

int main(){
	
	scanf("%lld%lld",&N,&K);
	
	for( lint i=1 ; i <= K ; i++ )
		pq.push(0);
	
	for( lint i=1,s,m,x ; i <= N ; i++ ){
		scanf("%lld%lld",&s,&m);
		x = -pq.top();
		pq.pop();
		printf("%lld\n",max(x,s) + m);
		pq.push(-(max(x,s) + m));
	}
	
	return 0;
}
