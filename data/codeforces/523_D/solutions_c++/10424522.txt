#include<cstdio>
#include<set>
using namespace std;
#define maa(a,b) ((a)>(b)?(a):(b))
typedef long long LL;
multiset<LL> S;
int n,m;
int main()
{
	//freopen("1.in","r",stdin);freopen("1.out","w",stdout);
	scanf("%d%d",&n,&m);
	int i,x,y;
	for (i=1;i<=m;i++) S.insert(0);
	for (i=1;i<=n;i++)
	{
		scanf("%d%d",&x,&y);
		multiset<LL>::iterator it=S.begin();
		LL tmp=*it;
		LL res=maa(tmp,x)+y;
		printf("%I64d\n",res);
		S.erase(it);
		S.insert(res);
	}
	return 0;
}
