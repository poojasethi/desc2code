#include<bits/stdc++.h>
using namespace std;
typedef __int64 ll;
int n,k;

struct cmp{
	bool operator () (ll x,ll y){
		return x>y;
	}
};

int main(void){
	int i,j;
	while(~scanf("%d%d",&n,&k)){
		priority_queue<ll,vector<ll>,cmp> q;
		//priority_queue<ll,vector<ll>,greater<ll> > q;
		ll st,en,la;
		for(i=0;i<k;i++) q.push(0);
		for(i=0;i<n;i++){
			scanf("%I64d%I64d",&st,&la);
			ll en=q.top(); q.pop();
			en=max(en,st);
			en+=la;
			printf("%I64d\n",en);
			q.push(en);
		}
	}
	return 0;
}
