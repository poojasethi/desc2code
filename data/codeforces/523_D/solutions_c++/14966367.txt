#include <stdio.h>
#include <queue>
#include <iostream>
using namespace std;
int n,k;
#define ll long long
int main(){
	int i;cin>>n>>k;
	priority_queue<pair<ll,int> >pq;
	for(i=0;i<k;++i)
		pq.push(make_pair(0,i));
	int s,d;
	pair<ll,int> t;
	for(i=0;i<n;++i){
		scanf("%d%d",&s,&d);
		t = pq.top();pq.pop();
		if(-t.first<s)
			cout<<s+d<<endl,t.first=-( s+d);
		else
			cout<<-t.first+d<<endl,t.first-=d;
		pq.push(t);
	}
}
