#include<bits/stdc++.h>
using namespace std;
multiset<long long>mm;
int main()
{
    long long n,k;
    scanf("%lld%lld",&n,&k);
    for(int i=1;i<=n;i++){
        long long a,b;
        scanf("%lld%lld",&a,&b);
        if(mm.size()<k){
            mm.insert(a+b);
            printf("%lld\n",a+b);
        }
        else{
            long long c=*mm.begin();
            mm.erase(mm.begin());
            mm.insert(max(a+b,c+b));
            printf("%lld\n",max(a+b,c+b));
        }
    }
}
