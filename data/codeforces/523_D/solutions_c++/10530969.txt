#include<cstdio>
#include<algorithm>
#include<cstring>
#include<set>
using namespace std;

const int maxn = 500005;

typedef long long ll;
multiset<ll> s;

multiset<ll> :: iterator it;
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=0;i<m;i++)
        s.insert(0);
    int t,l;
    for(int i=0;i<n;i++)
    {
        scanf("%d%d",&t,&l);
        it = s.begin();
        s.insert(max((*it),(ll)t)+l);
        printf("%lld\n",max((*it),(ll)t)+l);
        s.erase(s.begin());
    }
}
