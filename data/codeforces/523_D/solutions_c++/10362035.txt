#include <bits/stdc++.h>
using namespace std;
priority_queue <long long, vector<long long>, greater<long long> > Q;
int main() {
	int n, k;
	scanf("%d%d", &n, &k);
	for(int i = 0; i < k; i++) Q.push(0LL);
	long long s, t;
	for(int i = 0; i < n; i++) {
		scanf("%I64d%I64d", &s, &t);
		long long x = Q.top();
		Q.pop();
		if(x < s) {
			printf("%I64d\n", s + t);
			Q.push(s + t);
		} else {
			printf("%I64d\n", x + t);
			Q.push(x + t);
		}
	}
	return 0;
}
