#include<queue>
#include<cstdio>
#include<algorithm>
using namespace std;
priority_queue<long long,vector<long long>,greater<long long> > q;
long long n,m,s,t;
int main(){
    scanf("%d%d",&n,&m);
    for(int i = 1; i <= m; i++) q.push(0);  
    for(int i = 1; i <= n; i++){
        scanf("%I64d%I64d", &s, &t);  
        long long td=q.top();  
        q.pop();
        td=max(td,s)+t;
        printf("%I64d\n",td);
        q.push(td);
    }
}
