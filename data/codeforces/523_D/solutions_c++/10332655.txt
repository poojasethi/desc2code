#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <queue>
using namespace std;

typedef long long LL;
LL m,n,s,x,dq;
priority_queue<LL> q;
int main()
{
	scanf("%I64d%I64d",&m,&n);
	for (int i=1;i<=n;++i) q.push(0);
	for (int i=1;i<=m;++i)
	{
		scanf("%I64d%I64d",&s,&x);
		dq=q.top();
		q.pop();
		dq=-dq;
		dq=max(dq,s);
		dq+=x;
		printf("%I64d\n",dq);
		q.push(-dq);
	}
	return 0;
}
