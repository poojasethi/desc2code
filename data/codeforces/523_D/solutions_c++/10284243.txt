#include <iostream>
#include <cstdio>
#include <set>

using namespace std;

typedef long long ll;

int main() {
    int n, k;
    multiset<ll> q;
    scanf("%d %d", &n, &k);
    for (int i = 0; i < k; i++)
        q.insert(0);
    for (int i = 0; i < n; i++) {
        int s, m;
        scanf("%d %d", &s, &m);
        ll best = *(q.begin());
        q.erase(q.begin());
        ll ans = max(best, (ll)s) + m;
        q.insert(ans);
        printf("%I64d\n", ans);
    }
    return 0;
}
