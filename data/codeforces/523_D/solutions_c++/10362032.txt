#include <bits/stdc++.h>
using namespace std;
multiset <long long> S;
int main() {
	int n, k;
	scanf("%d%d", &n, &k);
	for(int i = 0; i < k; i++) S.insert(0LL);
	long long s, t;
	for(int i = 0; i < n; i++) {
		scanf("%I64d%I64d", &s, &t);
		multiset <long long> :: iterator it = S.begin();
		long long x = *it;
		S.erase(it);
		if(x < s) {
			printf("%I64d\n", s + t);
			S.insert(s + t);
		} else {
			printf("%I64d\n", x + t);
			S.insert(x + t);
		}
	}
	return 0;
}
