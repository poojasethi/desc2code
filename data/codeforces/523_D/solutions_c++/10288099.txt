#include <bits/stdc++.h>

using namespace std;

int main()
{
    cin.tie(0);
    ios_base::sync_with_stdio(0);
    long long n,m;
    cin>>n>>m;
    priority_queue <long long> serv;
    for (int i=0; i<m; i++)
        serv.push(0);
    long long a,t,ans=0;
    for (int i=0; i<n; i++)
    {
        cin>>a>>t;
        ans=max(a,-1*serv.top())+t;
        serv.pop();
        serv.push(-ans);
        cout<<ans<<endl;
    }
}
