#include<bits/stdc++.h>
using namespace std;
priority_queue<long long> h;
long long s,m,t;
int i,k,n;
int main()
{
    cin.tie(0);
    ios_base::sync_with_stdio(0);
    cin>>n>>k;
     for(i=1; i<=k; ++i) h.push(0);
     for(i=1; i<=n; ++i)
     {
       cin>>s>>m;
       t=max(s,h.top()*(-1))+m;
        cout<<t<<"\n";
        h.pop();
        h.push(-t);
     }
    return 0;
}
