#include <cstdio>
#include <queue>
using namespace std;

int n, k;
priority_queue<long long> q;

int main () {
    scanf("%d%d", &n, &k);
    for (int i = 1; i <= k; i++) q.push(0);
    for (int i = 1; i <= n; i++) {
        int s, m;
        scanf("%d%d", &s, &m);
        long long t = max((long long)s, -q.top()) + m;
        printf("%lld\n", t);
        q.pop();
        q.push(-t);
    }
    return 0;
}
