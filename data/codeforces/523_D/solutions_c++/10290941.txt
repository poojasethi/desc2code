#include <stdio.h>
#include <queue>
#include <math.h>

using namespace std;

priority_queue < long long > q;
int n, m;
long long res, l, r;

int main()
{
    scanf( "%d%d", &n, &m );
    
    while ( m-- )
        q.push( 0 );

    while ( n-- )
    {
        scanf( "%I64d%I64d", &l, &r );
        res = max( -q.top(), l ) + r;
        printf( "%I64d\n", res );
        q.pop();
        q.push( -res );
    }

    return 0;
}
