#include<stdio.h>
#include<queue>
#include<vector>
#include<algorithm>
using namespace std;
#define ll long long int
typedef bool (*comp)(ll,ll);
bool compare(ll a, ll b)
{
	if(a<b)return 0;
	else return 1;
}
int main()
{
	int n, k;
	scanf("%d%d", &n, &k);
	priority_queue<ll,std::vector<ll>, comp> pq(compare);
	for(int i=0;i<k;i++)pq.push(0);
	for(int i=0;i<n;i++)
	{
		ll s, m;
		scanf("%lld%lld", &s, &m);
		ll f=pq.top();
		pq.pop();
		printf("%lld\n", max(s,f)+m);
		pq.push(max(s,f)+m);
	}
	return 0;
}
