#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	ll n, k;
	cin >> n >> k;
	priority_queue<ll> a;
	for(int i=0;i<k;i++)
    {
        a.push(0);
    }
    for(int i=0;i<n;i++)
    {
        ll b=-a.top();
        a.pop();
        ll x,y;
        cin >> x >> y;
        x=max(b,x);
        b=x+y;
        cout <<  b << endl;
        a.push(-b);
    }
    return 0;

}
