#include <bits/stdc++.h>

using namespace std;

#define ll long long

int main(){
    int n, k;
    ll s, m;
    scanf("%d %d", &n, &k);

    priority_queue<ll> pqr;
    for(int i=0 ; i<k ; i++)
        pqr.push(0);
    ll temp;
    for(int i=0 ; i<n ; i++){
        scanf("%I64d %I64d", &s, &m);
        temp = max(s, -pqr.top()) + m;
        printf("%I64d\n", temp);
        pqr.pop();
        pqr.push(-temp);
    }

    return 0;
}
