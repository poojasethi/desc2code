#include <cstdio>
#include <algorithm>
#include <iostream>
#include <string>
#include<set>
using namespace std;
multiset<long long> q;
long long n,k,x,y;
int main(){
    scanf("%lld%lld",&n,&k);
    for(int i=0; i<k; i++)
        q.insert(0);
    for(int i=0;i<n;i++){
        scanf("%lld%lld",&x,&y);
        x=max(x,*q.begin())+y;
        printf("%lld\n",x);
        q.erase(q.begin());
        q.insert(x);
    }
    return 0;
}
