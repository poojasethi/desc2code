#include <bits/stdc++.h>
typedef __int64 ull;
using namespace std;

int main() {
    ios_base::sync_with_stdio(0);
    ull a, b, n, k;
    priority_queue<ull> pq;
    cin >> n >> k;
    while(k--) pq.push(0);
    while(n--) {
        cin >> a >> b;
        a = max(a, -pq.top()) + b;
        printf("%I64d\n", a);
        pq.pop();
        pq.push(-a);        
    }   
}
