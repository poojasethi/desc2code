#include<cstdio>
#include<queue>

using namespace std;

int main()
{
    priority_queue<long long> q;
    long long n,k,x,y;
    scanf("%lld%lld",&n,&k);
    for(int i=0; i<k; i++)q.push(0);
    while(n--)
    {
        scanf("%lld%lld",&x,&y);
        x=max(x,-q.top())+y;
        printf("%lld\n",x);
        q.pop();
        q.push(-x);
    }
}
