#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <set>
using namespace std;

multiset <long long> servers;

int n, k, t, len;

int main()
{
	scanf("%d%d", &n, &k);

	for (int i = 0; i < k; i++)
	{
		servers.insert(0);
	}

	for (int i = 0; i < n; i++)
	{
		scanf("%d%d", &t, &len);
		long long best = *servers.begin();
		servers.erase(servers.begin());
		best = max((long long)t, best);
		servers.insert(best + len);
		printf("%I64d\n", best + len);
	}


	return 0;
}
