#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <queue>

using namespace std;


long long a,b,i,n,m,tmp;
priority_queue <long long >q;

main()
{

  cin>>n>>m;
   
    for (i=1;i<=m;i++)
     q.push(0);
     
     
     while (n--)
      {
   scanf("%d %d",&a,&b);
       tmp=q.top();
       q.pop();
       tmp=min(tmp,-a);
      printf ("%I64d\n",b-tmp);
      tmp-=b;
      q.push(tmp);
      
      }
     

     
}
