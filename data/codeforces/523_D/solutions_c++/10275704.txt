#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

struct cmp {
	bool operator() (ll a, ll b) {
		return a > b;
	}
};

int main() {
	int n, k;
	scanf("%d%d", &n, &k);

	priority_queue<ll, vector<ll>, cmp> q;
	for (int i = 0; i < n; i++) {
		ll s, m;
		scanf("%I64d%I64d", &s, &m);

		while (!q.empty() && q.top() <= s) {
			q.pop();
		}

		if ((int)q.size() == k) {
			s = q.top();
			q.pop();
		}
		q.push(s + m);
		printf("%I64d\n", s + m);
	}
}
