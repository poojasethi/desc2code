#include <bits/stdc++.h>
using namespace std;
priority_queue<long long, vector<long long>, greater<long long> > servers;
int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    for(int i=0;i<k;i++)servers.push(0LL);
    for(int i=0;i<n;i++)
    {
        long long s,d;
        scanf("%lld%lld",&s,&d);
        long long finishtime=servers.top();
        servers.pop();
        long long nexttime=max(finishtime,s)+d;
        printf("%lld\n",nexttime);
        servers.push(nexttime);
    }
    return 0;
}
