#include <iostream>
#include <algorithm>
using namespace std;
 int n;
int a;
pair<int,int> tab[102];
int main()
{
 cin>>n;

 for (int i=0;i<n;i++)
 {
     cin>>a;
     tab[i]=make_pair(a,i+1);
 }
 sort(tab,tab+n);
for (int j=0;j<n/2;j++)
 {
   cout<<tab[j].second<<" "<<tab[n-j-1].second<<endl;
 }
    return 0;
}
