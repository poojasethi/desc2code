#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	int n;
	pair <int, int> a[101];

	cin >> n;

	for (int i = 0; i < n; i++)
	{
		cin >> a[i].first;
		a[i].second = i+1;
	}

	sort(a, a+n);

	for (int i = 0; i < n/2; i++)
	{
		cout << a[i].second << " " << a[n-i-1].second << endl;
	}

	return 0;
}
