#include <bits/stdc++.h>
using namespace std;
int main()
{
  int n,a[102];
  vector< pair<int,int> > ap;
  cin>>n;
  for (int i = 0; i < n; ++i)
  {
    cin>>a[i];
    ap.push_back(make_pair(a[i],i+1));
  }
  sort(ap.begin(),ap.end());
  for (int i = 0; i < n/2; ++i)
  {
    cout<<ap[i].second<<'\t'<<ap[n-i-1].second<<endl;
  }

}
