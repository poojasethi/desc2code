#include<bits/stdc++.h>

using namespace std;
typedef long long ll;
#define N 100005

int main(){
	int n;
	cin>>n;
	pair< int,int> a[n];
	for(int i = 0;i<n;i++){
		int x;
		cin>>x;
		a[i] = make_pair(x,i+1);
	}
	sort(a,a+n);
	for(int i = 0,j = n-1;i<j;i++,j--){
		cout<<a[i].second<<" "<<a[j].second<<endl;
	}
	return 0;
}
