#include<iostream>
using namespace std;
int main()
{
long long n,a[105],i,j,k=0,l;
cin>>n;
for(i=1;i<=n;i++)
{
cin>>a[i];
}
for(i=1;i<=n;i++)
{
  k+=a[i];
}
j=k/(n/2);
for(i=1;i<=n;i++)
{
  for(l=i+1;l<=n;l++)
  {
if(a[i]+a[l]==j&&i!=l)
{
  a[i]=-100;
  a[l]=-100;
cout<<i<<" "<<l<<endl;
break;
}
}
}
}
