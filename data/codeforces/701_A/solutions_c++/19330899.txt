#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
vector <pair<int,int> > a;
int main()
{
int n,i,d;
scanf("%d",&n);
for(i=0;i<n;i++)
{
scanf("%d",&d);
a.push_back(make_pair(d,i+1));
}

sort(a.begin(),a.end());

i=0;
int j=n-1;
while(i<j)
{
printf("%d %d\n",a[i].second,a[j].second);
i++;
j--;
}
return 0;
}
