#include<iostream>
#include<utility>
#include<algorithm>
using namespace std;
int main()
{
	pair<int,int>mp[100000];
	int i,n;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>mp[i].first;	
		mp[i].second = i+1;
	}
	sort(mp,mp+n);
	for(i=0;i<n/2;i++)
	{
		cout<<mp[i].second<<" "<<mp[n-1-i].second<<endl;
	}
	return 0;
}
