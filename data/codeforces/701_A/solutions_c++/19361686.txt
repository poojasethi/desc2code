#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	int n,a[105],i,j=0,k,u[105]={0};
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		j+=a[i];
	}
	j/=(n/2);
	for(i=0;i<n;i++)
		for(k=0;k<n,k!=i;k++)
			if(!u[i]&&!u[k]&&a[i]+a[k]==j)
			{
				cout<<i+1<<" "<<k+1<<endl;
				u[i]=1;
				u[k]=1;
			}
	return 0;
}
