#include<bits/stdc++.h>
using namespace std;
int main(){
    int t,n;
    string s;
    cin>>n;
    pair<int,int> mp[n];
    for(int i=0;i<n;++i){

        cin>>t;
        mp[i]=make_pair(t,i+1);
    }
    sort(mp,mp+n);
    for(int i=0;i<n/2;++i){
        cout<<mp[i].second<<" "<<mp[n-i-1].second<<endl;
    }
return 0;
}
