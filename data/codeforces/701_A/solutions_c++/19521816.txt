#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    pair < int, int > ii[100];
    int n;
    cin >> n;
    for(int i = 0; i < n; ++i) cin >> ii[i].first, ii[i].second = i+1;
    sort(ii, ii + n);
    for(int i = 0; i < n/2; ++i) cout << ii[i].second << " " << ii[n - i - 1].second << endl;
    return 0;
}
