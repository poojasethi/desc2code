#include<bits/stdc++.h>
#define f first
#define s second
using namespace std;
int main()
{
    int n;cin>>n;
    pair<int,int> a[101];
    for(int i=0;i<n;i++)
    {
        cin>>a[i].f;
        a[i].second=i+1;
    }
    sort(a,a+n);
    for(int i=0;i<n/2;i++)
        cout<<a[i].s<<" "<<a[n-i-1].s<<endl;
}
