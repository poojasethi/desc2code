#include<bits/stdc++.h>
using namespace std;
int a[105];
vector<pair<int,int> >V;
main()
{int N;
 scanf("%d",&N);
 int i;
 for (i=0;i<N;i++)
  {scanf("%d",&a[i]);
   V.push_back(make_pair(a[i],i));
  }
  sort(V.begin(),V.end());
  for (i=0;i<N/2;i++)
   {printf("%d %d\n",V[i].second+1,V[N-1-i].second+1);
   }
}
