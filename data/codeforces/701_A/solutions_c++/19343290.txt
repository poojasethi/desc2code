#include <bits/stdc++.h>
using namespace std;
int v[102];
bool m[102];
int main(){
	int n,i,j,k,x=0;
	cin>>n;
	for(i=1;i<=n;++i)
		cin>>v[i],x+=v[i];
	x/= (n/2);
	for(i=0;i<n/2;++i){
		for(j=1;j<=n;++j)
			for(k=j+1;k<=n;++k){
				if( !m[j] && !m[k] && v[j]+ v[k]== x){
					m[j]=m[k]=1;
					printf("%d %d\n",j,k);
				}
			}
	}
}
