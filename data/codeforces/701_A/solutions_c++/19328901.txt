#include <iostream>
#include <algorithm>
#include <utility>
using namespace std;
pair<int,int> arr[101];
int main()
{
	int n;
	cin>>n;
	
	for(int i=0;i<n;i++)
	{
		int x;
		cin>>x;
		arr[i]=make_pair(x,i+1);
	}
	sort(arr,arr+n);
	int i=0,j=n-1;
	while(i<j)
	{
		cout<<arr[i].second<<" "<<arr[j].second<<endl;
		i++;j--;
	}
}
