#include <bits/stdc++.h>

using namespace std;

int main() {
	int n;
	scanf("%d",&n);
	int x;
	pair<int, int> arr[1000];
	for(int i = 1; i <= n; i++) {
		scanf("%d",&x);
		arr[i] = make_pair(x, i);
	}
	sort(arr+1, arr+n+1);
	for(int i = 1; i <= n/2; i++) {
		printf("%d %d\n", arr[i].second, arr[n-i+1].second);
	}
	return 0;
}
