#include<bits/stdc++.h>
using namespace std;

int main(){

int n,i;
cin>>n;
vector<pair<int,int> >v(n+1);
for(i=1;i<=n;i++)
{
cin>>v[i].first;
v[i].second=i;
}
sort(v.begin(),v.end());
for(i=1;i<=n/2;i++)
{
	cout<<v[i].second<<" "<<v[n-i+1].second;
	cout<<"\n";
}
return 0;
}
