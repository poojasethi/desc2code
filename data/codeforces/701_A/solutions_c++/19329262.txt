#include<bits/stdc++.h>
using namespace std;
# define l long long int
# define rep(i,a,b) for(l i=a;i<b;i++)
int a[105];
bool cmpr(int x,int y){
	return(a[x]<a[y]);
}
int main(){
	int n;
	cin>>n;
	int idx[105];
	rep(i,0,n) {cin>>a[i]; idx[i]=i;}
	sort(idx,idx+n,cmpr);
	rep(i,0,n/2){
		cout<<idx[i]+1<<" "<<idx[n-i-1]+1<<"\n";
	}
	return 0;
}
