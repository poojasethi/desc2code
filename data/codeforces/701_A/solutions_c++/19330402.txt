#include<bits/stdc++.h>
using namespace std;

const int maxn = 111;
int n;
typedef pair<int,int>ii;
ii a[maxn];
int main(){
	scanf("%d",&n);
	for(int i = 0; i < n; ++i) {
			scanf("%d",&a[i].first);
			a[i].second = i+1;
	}
	sort(a,a+n);
	for(int i = 0; i < n/2; ++i) printf("%d %d\n",a[i].second,a[n-1-i].second);
}
