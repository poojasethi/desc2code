#include<algorithm>
#include<cstdio>
using namespace std;
pair<int,int> a[1000];
int n;
int main() {
    scanf("%d",&n);
    for(int i=1; i<=n; i++) {
        scanf("%d",&a[i].first);
        a[i].second=i;
    }
    sort(a+1,a+n+1);
    for(int i=1; i<=n/2; i++)
        printf("%d %d\n",a[i].second,a[n-i+1].second);
    return 0;
}
