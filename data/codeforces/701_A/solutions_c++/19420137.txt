#include <algorithm>
#include <cstdio>
using namespace std;
int n;
pair<int, int> a[100];
int main (){
	scanf("%d", &n);
	for (int i = 0; i < n; i++){
		scanf("%d", &a[i].first);
		a[i].second = i + 1;
	}
	sort(a, a + n);
	for (int i = 0; i < n / 2; i++)
		printf("%d %d\n", a[i].second, a[n - 1 - i].second);
	return 0;
}
