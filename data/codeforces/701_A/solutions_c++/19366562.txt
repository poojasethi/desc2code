#include <cstdio>
#include <algorithm>
using namespace std;
pair<int,int> s[105];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&s[i].first),s[i].second=i;
    sort(s+1,s+n+1);
    for(int i=1;i<=n/2;i++)printf("%d %d\n",s[i].second,s[n-i+1].second);
    return 0;
}
