#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,i,temp;
	scanf("%d",&n);
	vector < pair<int,int> > v;
	for(i=0;i<n;i++)
	{
		scanf("%d",&temp);
		v.push_back(make_pair(temp,i+1));
	}
	sort(v.begin(),v.end());
	for(i=0;i<n/2;i++)
	printf("%d %d\n",v[i].second,v[n-1-i].second);
	return 0;
}
