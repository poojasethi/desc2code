#include<cstdio>
int sum,n,a,p[110][110];
int main(){
	scanf("%d",&n);
	for (int i=1; i<=n; i++) scanf("%d",&a),p[a][0]++,p[a][p[a][0]]=i,sum+=a;
	sum=sum*2/n;
	for (int i=1; i<=(sum>>1); i++)
		while(p[i][0]){
			printf("%d ",p[i][p[i][0]]);p[i][0]--;
			printf("%d\n",p[sum-i][p[sum-i][0]]);p[sum-i][0]--;
		}
	return 0;
}
