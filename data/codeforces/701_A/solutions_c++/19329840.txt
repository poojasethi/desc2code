#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,i;
    cin>>n;
    pair<int,int> a[n];
    for(i=0;i<n;i++)
        {cin>>a[i].first;
        a[i].second=i;}
    sort(a,a+n);
    for(i=0;i<n/2;i++)
        cout<<a[i].second+1<<' '<<a[n-i-1].second+1<<endl;
    return 0;
}
