n=input()
a=map(int,raw_input().split())
s=sum(a)/(n/2)
check=[0]*(n+1)
for i in range(n):
    for j in range(n):
        if(check[i]== 0 and check[j]==0 and i!=j):
            if(a[i]+a[j]==s):
                check[i]=1
                check[j]=1
                print i+1,j+1
                break
