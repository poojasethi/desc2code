from sys import stdin

n = int(stdin.readline())
a = map(int, stdin.readline().split(' '))
sum = sum(a) * 2 / n
used = [False] * n
for i in range(n):
  if used[i]:
    continue
  for j in range(i + 1, n):
    if not used[j] and a[i] + a[j] == sum:
      used[j] = True
      print i + 1, j + 1
      break
