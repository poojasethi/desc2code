#include <stdio.h>
#include<memory.h>
#include<algorithm>
using namespace std;
#define module 9901
#define maxn 100005
#define max(x,y) x>y?x:y
int fa[maxn];
int s[maxn];
int getfather(int x){
	return fa[x]==x?x:(fa[x]=getfather(fa[x]));
}
int main(void){
	int n,m,k,x,y;
	while(scanf("%d%d%d",&n,&m,&k)==3){
		memset(s,0,sizeof(s));
		memset(fa,0,sizeof(fa));
		long long int ans=1;
		int num=0;
		for(int i=1;i<=n;i++) fa[i]=i;
		for(int i=0;i<m;i++){
			scanf("%d%d",&x,&y);
			fa[getfather(x)]=getfather(y);
		}for(int i=1;i<=n;i++){
			s[getfather(i)]++;
		}for(int i=1;i<=n;i++){
			if(s[i]!=0){
				num++;ans=(ans*s[i])%k;
			}
		}for(int i=0;i<num-2;i++){
			ans=(ans*n)%k;
		}
		if(num==1) printf("%d\n",1%k);
		else printf("%lld\n",ans);
	}
	return 0;
}
