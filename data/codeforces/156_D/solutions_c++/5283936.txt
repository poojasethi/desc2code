#include<cstring>
#include<cstdio>
#include<algorithm>
#include<cstdlib>
using namespace std;
int n,m,x,y,fa[120000],ans,p,i,s[120000],num;
int get(int x){
	return fa[x]==x?x:(fa[x]=get(fa[x]));
}
int main(){
	scanf("%d%d%d",&n,&m,&p);
	for(i=1;i<=n;++i)fa[i]=i;
	for(i=1;i<=m;++i){
		scanf("%d%d",&x,&y);
		fa[get(x)]=get(y);
	}
	ans=1;
	for(i=1;i<=n;++i)s[get(i)]++;
	for(i=1;i<=n;++i)if(s[i])num++,ans=(long long)ans*s[i]%p;
	for(i=1;i<=num-2;++i)ans=(long long)ans*n%p;
	if(num==1)printf("%d\n",1%p);
	else printf("%d\n",ans);
}
