#include <stdio.h>
#define MaxN 100010

int par[MaxN], v[MaxN];

int root(int a)
{
    if (par[a] == a) return a;
    return par[a] = root(par[a]);
}

int main()
{
    int n, m, k, a, b;
    while (scanf("%d%d%d", &n, &m, &k) != EOF)
    {
        for (int i = 1; i <= n; ++i) par[i] = i, v[i] = 0;
        while (m--)
        {
            scanf("%d%d", &a, &b);
            a = root(a), b = root(b);
            if (a != b) par[a] = b;
        }
        for (int i = 1; i <= n; ++i) v[root(i)]++;
        long long ans = 1;
        int cnt = 0;
        for (int i = 1; i <= n; ++i) if (v[i]) ans = ans * v[i] % k, ++cnt;
        if (cnt == 1) ans = 1 % k;
        for (int i = 0; i < cnt-2; ++i) ans = ans * n % k;
        printf("%d\n", (int)ans);
    }
    return 0;
}
