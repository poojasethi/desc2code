import sys

def flip(bit, num):
	try:
		if (num[bit] == 0):
			num[bit] = 1
		else:
			num[bit] = 0
			flip(bit + 1, num)
	except:
		num[bit] = 1

def main():
	sys.stdin.readline()
	nums = map(int, sys.stdin.readline().split(" "))

	bin = {}
	for num in nums:
		flip(num, bin)

	num_ones = 0
	max_ones = 0

	for i in bin.keys():
		if bin[i] == 1:
			num_ones = num_ones + 1
			if i > max_ones:
				max_ones = i

	print max_ones + 1 - num_ones

if __name__ == '__main__':
	main()
