
N = int(raw_input())
A = [int(x) for x in raw_input().split()]

m = {}

def inc(m, k, v=1):
	if not m.has_key(k):
		m[k] = v
	else:
		m[k] += v

hi = 0

for x in A:
	inc(m, x)
	i = x
	hi = max(hi, i)
	while m[i]>1:
		inc(m, i+1, m[i]>>1)
		m[i] %= 2
		i += 1
		hi = max(hi, i)

print hi+1-len(filter(lambda x:x==1, m.values()))
