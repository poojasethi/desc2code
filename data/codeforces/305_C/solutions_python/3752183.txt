#!/usr/bin/env python
# -*- coding: UTF-8 -*-

import sys
import itertools

DEBUG = 0

def ReadIn():
    data = [int(x) for x in sys.stdin.read().split()]
    offset = 0
    while offset < len(data):
        n = data[offset]
        offset += 1
        yield data[offset : offset + n]
        offset += n

def Solve(a):
    if DEBUG: print 'a =', a
    idx = 0
    now = 0
    ans = 0
    for key in a:
        while idx < key and now:
            if ~now & 1: ans += 1
            now >>= 1
            idx += 1
        if DEBUG: print idx, now, ans
        now += 1
        ans += key - idx
        idx = key
    while now:
        if ~now & 1: ans += 1
        now >>= 1
        idx += 1

    print ans

if __name__ == '__main__':
    for a in ReadIn():
        Solve(a)
