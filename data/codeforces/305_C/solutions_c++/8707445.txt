#include <bits/stdc++.h>
using namespace std;

bitset<2000000001> bs;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int n;
	cin >> n;
	int maxi = 0, count = 0;
	for(int i = 0; i < n; ++i)
	{
		int a;
		cin >> a;
		while(bs[a]) bs[a++] = 0, --count;
		bs[a] = 1;
		++count;
		maxi = max(maxi, a);
	}
	cout << maxi - count + 1 << endl;
	return 0;
}
