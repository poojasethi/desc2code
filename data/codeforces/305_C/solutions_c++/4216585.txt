#include <iostream>  
#include <set>  
using namespace std;  
set<int>Set;  
int main()  
{  
  
    int n , x, Max=0;  
    cin>>n;  
    for(int i=0;i<n;i++)  
    {  
        cin>>x;  
        while(Set.count(x)) Set.erase(x),x++;  
        Max=max(Max,x);  
        Set.insert(x);  
    }  
    cout<<Max+1-Set.size()<<endl;  
    return 0;  
}
