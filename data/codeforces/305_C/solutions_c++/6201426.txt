#include <cstdio>
#include <set>
using namespace std;
set<int>Set;
int main(){
    int n,x,Max=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&x);
        while(Set.count(x)) Set.erase(x),x++;
        Max=max(Max,x);
        Set.insert(x);
    }
    printf("%I64d\n",(long long)(Max+1-Set.size()));
    return 0;
}
