#include <iostream>
#include <set>
using namespace std;
set <int>s;
int max(int a,int b) {return a>b?a:b;}
int main() {
  int n,m=-1; cin >> n;
  for (int i=1; i<=n; i++) {
    int a; cin >> a;
    while (s.find(a)!=s.end()) s.erase(s.find(a++));
    s.insert(a); m=max(m,a);
  }
  cout << m-s.size()+1 << endl;
  //system("pause");
  return 0;
}
