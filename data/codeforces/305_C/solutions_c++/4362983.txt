#include<stdio.h>
#include <set>
using namespace std;
#define Max(a,b) a>b?a:b
set<int>Set;
int main()
{

    int n,x,max=0;
    scanf("%d",&n);;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&x);
        while(Set.count(x)) Set.erase(x),x++;
        max=Max(max,x);
        Set.insert(x);
    }
    printf("%d\n",max+1-Set.size());
    return 0;
}
