#include<stdio.h>
#include<set>
using namespace std;
set<__int64>s;
int main()
{
	__int64 n,i,a,max;
	scanf("%I64d",&n);
	max=0;
	for(i=0;i<n;i++)
	{
		scanf("%I64d",&a);
		while(s.count(a))
		{
			s.erase(a);
			a++;
		}
		if(max<a)
			max=a;
		s.insert(a);
	}
	printf("%I64d\n",max+1-s.size());
	return 0;
}
