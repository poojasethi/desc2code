#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	while(~scanf("%d",&n)) {
		map<int,int>M;
		int tmp,time=1-n,ans=0;
		while(n--) {
			scanf("%d",&tmp);
			M[tmp]++;
			ans=max(ans,tmp);
			while(M[tmp]==2) {
				M[tmp]-=2;
				M[++tmp]++;
				time++;
				ans=max(ans,tmp);
			}
		}
		printf("%d\n",ans+time);
	}
	return 0;
}
