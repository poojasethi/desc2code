#include<iostream>
#include<set>
using namespace std;
multiset<int> ms;
int main()
{
	int n;
	cin>>n;
	int maxi=-1;
	for(int i=0;i<n;i++)
	{
		int x;cin>>x;
		ms.insert(x);
		int j=x;
		while(ms.count(j)>1)
		{
			ms.erase(j);ms.erase(j);ms.insert(++j);
		}
		maxi=max(j,maxi);
	}
	cout<<maxi-ms.size()+1<<endl;
}
