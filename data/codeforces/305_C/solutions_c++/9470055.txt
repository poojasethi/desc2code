#include<stdio.h>
#include<set>
using namespace std;
set<int>s;
int main()
{
    int n,a,m=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a);
        while(s.count(a))
        {
            s.erase(a);
            a++;
        }
        m=(m>a)?m:a;
        s.insert(a);
    }
    printf("%d\n",m+1-s.size());
    return 0;
}
