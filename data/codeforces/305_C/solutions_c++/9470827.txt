#include<iostream>
#include<set>
using namespace std;
int main()
{
	set<int> s;
	int n,a,max=0;
	int i;
	cin >>n;
	for(i=0;i<n;i++)
	{
		cin >>a;
		while(s.count(a))
		{
			s.erase(a);
			a++;
		}
		s.insert(a);
		if(a>max)
		{
			max=a;
		}
	}
	cout <<max-s.size()+1<<endl;
	return 0;
}
