#include<cstdio>
#include<iostream>
#include<map>
using namespace std;
map<int, int>sta;
int n, i;
int main(){
	scanf("%d", &n);
	for(i=1;i<=n;i++){
		int x;scanf("%d", &x);
		while(sta[x] == 1)sta.erase(sta.find(x++));
		sta[x]=1;
	}
	int num = sta.size(), num1=(--sta.end())->first;
	printf("%d\n", num1-num+1);
	return 0;
}
