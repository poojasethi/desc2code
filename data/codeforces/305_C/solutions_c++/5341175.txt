#include<stdio.h>
#include<map>
using namespace std;
map<int,int>M;
int main()
{
	int n,t,ma = 0;
	scanf("%d",&n);
	M.clear();
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&t);
		while(M[t]) M.erase(t++);
		M[t]++;
		if(t > ma) ma = t;
	}
	printf("%d\n",ma - M.size() + 1);
	return 0;
}
