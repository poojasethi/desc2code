#include <iostream>
#include <set>
using namespace std;
set<long long> s;
void insert(long long x)
{
	if (s.count(x))
	{
		s.erase(x++);
		insert(x);
	}
	else
		s.insert(x);
}
int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		long long val;
		cin >> val;
		insert(val);
	}
	cout << *s.rbegin() + 1 - s.size() << endl;
	return 0;
}
