#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n, a, m = 0;
    set<int> s;
    scanf("%d", &n);
    while(n--)
    {
        scanf("%d", &a);
        while(s.count(a))
        {
            s.erase(a);
            ++a;
        }
        s.insert(a);
        m = max(m, a + 1);
    }
    printf("%d\n", m - s.size());
    return 0;
}
