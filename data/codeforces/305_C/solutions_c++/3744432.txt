#include<cstdio>
#include<iostream>
#include<map>
using namespace std;
map<int,bool> mymap;
int main()
{
	int n,i;
	scanf("%d",&n);
	int cnt=0,mm=0,t;
	for(i=0;i<n;i++)
	{
		scanf("%d",&t);
		while(mymap[t])
		{
			mymap[t]=0;
			t++;
			cnt--;
		}
		mymap[t]=1;
		mm=max(mm,t);
		cnt++;
	}
	//printf("%d %d\n",mm,cnt);
	printf("%d\n",1+mm-cnt);
	return 0;
}
