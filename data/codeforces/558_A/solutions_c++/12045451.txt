#include <iostream>
#include <algorithm>
using namespace std;

pair<int,int> t[100];

int main(){
    int n;
    cin>>n;
    for(int a=0;a<n;++a){
        cin>>t[a].first>>t[a].second;
    }

    sort(t,t+n);
    int i=0;
    while(i<n && t[i].first<0)++i;


    int ret=0;

    int j=0;
    int p=i-1,q=i;
    while(p>=0 && q<n){
        ret+=t[p].second;
        ret+=t[q].second;
        --p; ++q;
    }
    if(p<0 && q<n)ret+=t[q].second;
    if(p>=0 && q>=n)ret+=t[p].second;

    cout<<ret;



    return 0;
}
