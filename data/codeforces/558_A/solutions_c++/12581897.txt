#include <bits/stdc++.h>
using namespace std;
int n, x , ne=0, po=0 , a, b , ans=0;
int main(){
    cin >> n;
    pair<int , int> p[101];
    for(int i = 0 ; i < n ; i++){
        cin >> a >> b;
        p[i]=make_pair(a , b);
        if(a<0)ne++; else po++;
    }sort(p , p+n);
    for(int i = max(0 , ne-po-1) ; i <= min(n , ne*2);i++)
        ans += p[i].second;
    cout << ans << endl ;
}
