#include <iostream>
#include <algorithm>
using namespace std;

pair<int, int> a[111];

int main()
{
	int n, p=0;
	cin >> n;
	for(int i=0; i<n; i++)
	{
		cin >> a[i].first >> a[i].second;
		if(a[i].first > 0)
			p++;
	}
	sort(a, a+n);

	n-=p;

	int res = 0;
	if(p == n)
	{
		for(int i=0; i<n+p; i++)
			res += a[i].second;
	}
	else if(p > n)
	{
		for(int i=0; i<2*n+1; i++)
			res += a[i].second;
	}
	else
	{
		for(int i=0; i<2*p+1; i++)
			res += a[n+p-1-i].second;
	}

	cout << res << endl;

	return 0;
}
