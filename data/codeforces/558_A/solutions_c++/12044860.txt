#include<stdio.h>
int n,rx[100001],lx[100001],ra,la,r,l,a,i,s,min,k,t;
int main(){
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d%d",&k,&t);
		if(k>0){ra++;rx[k]=t;}
		else {la++;lx[-k]=t;}
	}
	if(ra==la){r=l=ra;}
	else if(ra<la){r=ra;l=ra+1;}
	else {l=la;r=la+1;}
	t=0;
	for(i=1;i<=100000;i++){
		if(rx[i]>0){s+=rx[i];t++;}
		if(t==r)break;
	}
	t=0;
	for(i=1;i<=100000;i++){
		if(lx[i]>0){s+=lx[i];t++;}
		if(t==l)break;
	}
	printf("%d",s);
}
