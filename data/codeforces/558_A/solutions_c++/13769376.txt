#include<bits/stdc++.h>
using namespace std;
int main()
{
    pair<int ,int> pos[100],neg[100];
    long r,i,j,k=0,l=0,m,n;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>j>>r;
    	if(j>0)
    	pos[k++]=make_pair(j,r);
    	else
    	neg[l++]=make_pair(-1*j,r);
    }
    sort(pos,pos+k);
    sort(neg,neg+l);
    long ans=0;
    for(i=0;i<min(k,l);i++)
    ans+=(pos[i].second+neg[i].second);
    
    if(k>l)
    ans+=pos[i].second;
    if(l>k)
    ans+=neg[i].second;
	cout<<ans;
}
