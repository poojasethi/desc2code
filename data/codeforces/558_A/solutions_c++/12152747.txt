#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin >> n;
    pair<int,int>p[101];
    int pp=0,nn=0;
    for(int i=0;i<n;i++)  {
        cin >> p[i].first >> p[i].second;
        if(p[i].first > 0)
            pp++;
        else
            nn++;
    }
    sort(p,p+n);
    long long int cnt=0;
    for(int i=max(0,nn-pp-1);i<=min(n,2*nn);i++)
        cnt+=p[i].second;
    cout<<cnt<<endl;
}
