#include <bits/stdc++.h>
#define maxn 150
#define el endl

using namespace std;

pair <int, int> a[maxn];
int n, am = 0, duong = 0;
long long tong = 0;

int main()
{
    cin >> n;
    for(int i = 0;i < n; i++)
    {
        cin >>a[i].first >> a[i].second;
        if(a[i].first < 0)
            am ++;
        else duong ++;
    }
    sort(a, a +n);
    for(int i = max(0, am - duong - 1); i <= min(n, am * 2); i++)
        tong += a[i].second;
    cout << tong;
    return 0;
}
