#include <bits/stdc++.h>
using namespace std;
pair <int,int> p[150];
int neg,pos;
int main()
{
	int n;
	cin>>n;
	for (int i=0; i<n; i++)
	{
		cin>>p[i].first>>p[i].second;
		if (p[i].first<0)
			neg++;
		else
			pos++;
	}
	sort(p,p+n);
	int t=0;
	for (int i=max(0,neg-pos-1); i<=min(n,neg*2); i++)
		t+=p[i].second;
	cout<<t;
	return 0;
}
