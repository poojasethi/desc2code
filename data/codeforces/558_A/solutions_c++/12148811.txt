#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
struct Node{
	int x,y;
}a[101];
bool tmp1(Node a,Node b)
{return a.x<b.x;}
bool tmp2(Node a,Node b)
{return a.x>b.x;}
int s,i,n,num;
int main()
{
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&a[i].x,&a[i].y);
		if(a[i].x<0) num++;
	}
	if(num<=n-num)
	sort(a,a+n,tmp1);
	else {sort(a,a+n,tmp2);num=n-num;}
	for(i=0;i<2*num;i++)
	s+=a[i].y;
	if(2*num<n) s+=a[2*num].y;
	printf("%d\n",s);
	//system("pause");
}
