#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	pair<int,int> a[101];
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i].first>>a[i].second;
	}
	sort(a,a+n);
	int ind=0;
	for(int i=0;i<n;i++) if(a[i].first>0){
		ind=i;
		break;
	}
	int p,q;
	p=q=upper_bound(a,a+n,make_pair(0,0))-a;
	p--;
	int ans=0;
	while(p >= 0 && q < n)
    {
	ans += a[p].second + a[q].second;
	p--;
	q++;
    }
    if(p>=0) ans+=a[p].second;
    if(q<n) ans+=a[q].second;
	cout<<ans;
}
