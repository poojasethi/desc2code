#define N (1<<17)
#define LL unsigned long long
#include <bits/stdc++.h>

using namespace std;

int n,x,p,l,r,a[N<<1],ls[N],rs[N];

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>x>>p,a[x+N]=p;
	for(int i=N-1;i>=0;i--)
		if(a[i]) ls[l+1]=ls[l]+a[i],l++;
	for(int i=N+1;i<N+N;i++)
		if(a[i]) rs[r+1]=rs[r]+a[i],r++;

	if(l==r)
		cout<<ls[l]+rs[r]<<endl;
	else if(l<r)
		cout<<ls[l]+rs[l+1]<<endl;
	else
		cout<<rs[r]+ls[r+1]<<endl;
}
