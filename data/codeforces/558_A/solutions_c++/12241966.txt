#include<cstdio>
#include<algorithm>
using namespace std;
struct stt{int a,b;}st[100];
bool cmp(stt st1,stt st2)
{return st1.a<st2.a;}
int main()
{
    int n,a=0,b,c,o=0;
    scanf("%d",&n);
    for(int i=0;i<n;++i)
    {
        scanf("%d%d",&st[i].a,&st[i].b);
        a+=st[i].a<0;
    }
    sort(st,st+n,cmp);
    b=a,c=n-a;if(b>c)b=c+1;if(c>b)c=b+1;
    for(int i=a-1;i>a-b-1;--i)o+=st[i].b;
    for(int i=a;i<a+c;++i)o+=st[i].b;
    printf("%d\n",o);
}
