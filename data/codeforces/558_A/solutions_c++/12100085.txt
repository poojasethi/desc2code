#include <bits/stdc++.h>

using namespace std;

int n, s[110], res = 0;
pair<int, int> a[110];

int main()
{
  cin >> n;
  for(int i = 1; i <= n; i++) cin >> a[i].first >> a[i].second;
  sort(a + 1, a + n + 1);
  a[n + 1] = make_pair(200000, 1);
  int p1 = -1;
  for(int i = 1; i <= n + 1; i++){
    if(a[i].first > 0 && p1 < 0) p1 = i - 1;
    s[i] = s[i - 1] + a[i].second;
  }
  int p2 = n - p1;
  if(p2 == p1) cout << s[n];
  if(p2 > p1) cout << s[2 * p1 + 1];
  if(p2 < p1) cout << s[n] - s[2 * p1 - n - 1];
}
