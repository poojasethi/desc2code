#include<iostream>
#include<algorithm>
using namespace std;
#define c 100005
#define Int long long
Int Hash[300002]={0};
int main()
{
    Int N,X[102],sum=0,i,j;
    cin>>N;
    for (i=0;i<N;++i)
    {
        cin>>X[i]>>j;
        Hash[X[i]+c]=j;
    }
    sort(X,X+N);
    for (i=0;i<N&&X[i]<0;++i);
    for (j=i-1;j>=0&&i<N;++i,--j)
        sum+=Hash[X[i]+c]+Hash[X[j]+c];
    if (i<N)
        sum+=Hash[X[i]+c];
    else if (j>=0)
        sum+=Hash[X[j]+c];
    cout<<sum;
    return 0;
}
