#include <iostream>
#include <algorithm>
using namespace std;
int main() {
	int n, st, ans = 0;
	cin >> n;
	pair<int, int> A[n + 1];
	for(int i = 0; i < n; ++i)
		cin >> A[i].first >> A[i].second;
	A[n] = make_pair(0, 0);
	sort(A, A + n + 1);
	for(int i = 0; i < n + 1; ++i)
		if(A[i].first == 0) {
			st = i; break;
		}
	int i, j;
	for(i = st - 1, j = st + 1; i >= 0 && j <= n; --i, ++j)
		ans += A[i].second + A[j].second;
	if(i >= 0) ans += A[i].second;
	if(j <= n) ans += A[j].second;
	cout << ans << endl;
}
