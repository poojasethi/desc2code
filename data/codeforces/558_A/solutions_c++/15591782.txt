#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
struct node
{
	int x,a;
}sz[1005];
bool cmp(node n,node m)
{
	return n.x<m.x;
}
int main()
{
	int n,i,j,k,sum=0,s,a,b;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d%d",&sz[i].x,&sz[i].a);
	sort(sz,sz+n,cmp);
	for(i=0;i<n;i++)
		if(sz[i].x>0)
			break;
	for(a=i-1,b=i;a>=0&&b<n;a--,b++)
		sum+=sz[a].a+sz[b].a;
	if(a>=0)
		sum+=sz[a].a;
	if(b<n)
		sum+=sz[b].a;
	printf("%d\n",sum);
	return 0;
}
