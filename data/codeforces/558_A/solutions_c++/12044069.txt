#include <bits/stdc++.h>
using namespace std;

const int maxN=1e5+100;
int n,a[maxN],b[maxN],na,nb;

int main() {
    cin>>n;
    int res=0;
    for(int i=0;i<n;i++) {
        int x,y;
        scanf("%i%i",&x,&y);
        if (x<0) a[-x]=y,++na; else b[x]=y,++nb;
    }
    int m=min(na,nb)+1;
    for(int i=0;i<=100000;i++) if (m>0) {
        if (a[i]>0) res+=a[i], --m;
    }
    m=min(na,nb)+1;
    for(int i=0;i<=100000;i++) if (m>0) {
        if (b[i]>0) res+=b[i], --m;
    }
    cout<<res;
}
