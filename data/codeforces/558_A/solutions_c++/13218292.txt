#include<bits/stdc++.h>
using namespace std;
int n,negative,positive;
long long ans;
pair < int , int > P[150] ;
int main(){
	cin >> n ;
	for(int i=0;i<n;i++){
		cin >> P[i].first >> P[i].second ;
		if(P[i].first < 0)
			negative++;
		else
			positive++;
	}
	sort(P,P+n);
	for(int i=max(0,negative-positive-1) ; i<=min(n,2*negative);i++)
		ans += P[i].second;	

	cout << ans << endl;
}
