#include<bits/stdc++.h>
using namespace std;
int main(){
   int n,a=0,kol=0;
   cin>>n;
   pair<int,int>p[100005];
   for( int i=1;i<=n;i++ ){
      cin>>p[i].first>>p[i].second;
      if( p[i].first < 0 ){
         a++;
      }
   }
   sort(p+1,p+n+1);
   int k = a+1;
   int b = a;
   while( k <= n && b >= 1 ){
      kol += p[k].second;
      k++;
      kol += p[b].second;
      b--;
   }
   if( a > (n-a) ){
     kol += p[b].second;
   }
   else {
     kol += p[k].second;
   }
   cout<<kol<<endl;
}
