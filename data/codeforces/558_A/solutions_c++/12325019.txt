#include <bits/stdc++.h>
#define ll long long
#define fi first
#define se second
using namespace std;
typedef pair<int,int> ii;
int n,ps,ns;
ii p[110];
ll snc;
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d %d",&p[i].fi,&p[i].se);
		if(p[i].fi<0)
		{
			ns++;
		}
		else 
		{
			ps++;
		}
	}
	sort(p,p+n);
	for(int i=max(0,ns-ps-1);i<=min(n,2*ns);i++)
	{
		snc+=p[i].se;
	}
	printf("%lld\n",snc);
	return 0;
}
