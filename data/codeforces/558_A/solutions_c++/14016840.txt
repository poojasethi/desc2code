#include <iostream>
#include <algorithm>
using namespace std;

int main(){
    pair<int,int> a[101];
    int n, po=0, ne=0, s[101];
    cin >> n;
    for(int i=1; i<=n; i++){
        cin >> a[i].first >> a[i].second;
        if(a[i].first<0)ne++;
        else po++;
    }
    sort(a+1,a+n+1);
    s[0]=0;
    for(int i=1; i<=n; i++){
        s[i]=s[i-1]+a[i].second;
    }

    int z = po-ne, sum=0;
    if(z==0)sum=s[n];
    else if(z>0)sum=s[2*ne+1];
    else sum=(s[n]-s[-z-1]);
    cout << sum;
}
