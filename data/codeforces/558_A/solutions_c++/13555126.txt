#include<bits/stdc++.h>
#define f(i,a,b) for(i=a;i<b;i++)
using namespace std;
int main()
{
    int n,i,x,a,len;
    vector<int>r,l;map<int,int>m;
    cin>>n;
    f(i,0,n){
        cin>>x>>a;if(x<0)l.push_back(-x);else r.push_back(x);
        m[x]=a;
    }
    long long ans=0;sort(r.begin(),r.end()),sort(l.begin(),l.end());
    len=min(l.size(),r.size());
    f(i,0,len)ans+=m[r[i]]+m[-l[i]];
    if(l.size()>len)ans+=m[-l[i]];
    else if(r.size()>len)ans+=m[r[i]];
    cout<<ans;
}
