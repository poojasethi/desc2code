#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t,n=0,p=0,a,b,d=0,k;
	cin>>t;
	int l[t];
	map<int,int> m;
	for(int i=0;i<t;i++)
	{
		cin>>a>>b;
		l[i]=a;
		m[l[i]]=b;
		if(a<0)
		n++;
		else if(a>0)
		p++;
	}
	sort(l,l+t);
	k=n-p;
	if(k==1 || k==-1 || k==0)
	{
		for(int i=0;i<t;i++)
		d+=m[l[i]];
	}
	else if(k>1)
	{
		for(int i=k-1;i<n+p;i++)
		d+=m[l[i]];
	}
	else if(k<-1)
	{
		for(int i=0;i<=n+p+k;i++)
		d+=m[l[i]];
	}
	cout<<d;
	return 0;
}
