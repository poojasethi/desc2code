#include <stdio.h>
#include <algorithm>
using namespace std;
struct node{
	int a,b;
}que[105];
int cmp(node m,node n){
	return m.a<n.a;
}
int main(){
	int n;
	scanf("%d",&n);
	int i,l=0,r=0;
	for(i=0;i<n;i++){
		scanf("%d%d",&que[i].a,&que[i].b);
		if(que[i].a<0) l++;
		else r++;
	}
	sort(que,que+n,cmp);
	int begin,end;
	if(r>l){
		begin=0,end=l*2+1;
	}
	else if(r==l) {
		begin=0,end=l+r;
	}
	else {
		begin=n-(2*r+1),end=n;
	}
	int ans=0;
	for(i=begin;i<end;i++)
		ans+=que[i].b;
	printf("%d\n",ans);
}
