#include <bits/stdc++.h>
using namespace std;
#define forn(i, n) for(i = 0; i < n; ++i)
#define N 102
int main() {
	pair<int, int> v[N];
	int n, x, a, i, l, r, ans = 0;
	cin >> n;
	forn(i, n) {
		cin >> x >> a;
		v[i] = make_pair(x, a);
	}
	sort(v, v + n);
	for(i = 0; i < n; i++)
		if(v[i].first > 0) break;
	l = i - 1;
	r = i;
	while(l >= 0 && r < n) {
		ans += v[l].second + v[r].second;
		l--;
		r++;
	}
	if(l >= 0) ans += v[l].second;
	if(r < n) ans += v[r].second;
	cout << ans << endl;
	return 0;
}
