#include <bits/stdc++.h>
#define ll long long
using namespace std;
pair< ll, ll>m[100005];
int n,l,r,a,b,ans;
int main(){
    cin>>n;
    for(ll i = 0; i < n; i++){
        cin>>a>>b;
        m[i] = make_pair(a,b);
        if(a<0)l++;
        else r++;
    }
    sort(m,m+n);
    for(int i = max(0,l-(r+1));i <= min(n,2*l);i++)
        ans += m[i].second;
    cout<<ans<<endl;
}
