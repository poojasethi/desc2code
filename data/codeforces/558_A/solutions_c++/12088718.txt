#include <bits/stdc++.h>
using namespace std;

int main ()
{
  int n, neg=0;
  cin>>n;
  pair <int, int> a[n];
  for(int i=0;i<n;i++)
  {
        cin>>a[i].first>>a[i].second;
        if(a[i].first<0)
            neg++;
    }
    sort(a, a+n);
    int sol=0;
    if(neg+neg<n)
        for(int i=0;i<min(neg*2+1, n);i++)
            sol+=a[i].second;
    else
        for(int i=max(neg*2-n-1, 0);i<n;i++)
            sol+=a[i].second;
    cout<<sol<<endl;
  return 0;
}
