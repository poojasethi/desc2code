#include <bits/stdc++.h>
using namespace std;
int main()
{
int n;
while(cin>>n)
{
    long long j=0,k=0;
    pair <int , int > a[105],b[105];
    long long x,p;
    for(int i=0;i<n;++i)
    {
        cin>>x>>p;
        if(x>0) a[j++]=make_pair(x,p);
        else b[k++]=make_pair(-x,p);
    }
    a[j]=make_pair(10001,0);
    b[k]=make_pair(10001,0);
    sort(a,a+j);
    sort(b,b+k);
    long long sum=0;
    for(int i=min(j,k);i>-1;--i)
        sum+=b[i].second+a[i].second;

    cout<<sum<<endl;
}





}
