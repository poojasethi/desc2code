#include<iostream>
#include<cstdio>
#include<algorithm>
#include<utility>
using namespace std;
pair< int, int>mypair[100005];
int main(){
    int t;
    int l = 0,r = 0;
    scanf("%d",&t);
    for(int i = 0; i < t; i++){
        int a,b;
        scanf("%d%d",&a,&b);
        mypair[i] = make_pair(a,b);
        if(a<0)
            l++;
        else r++;
    }
    int ans = 0;
    sort(mypair,mypair+t);
    for(int i = max(0,l-(r+1));i <= min(t-1,2*l);i++)
        ans += mypair[i].second;
    cout<<ans<<endl;;
}
