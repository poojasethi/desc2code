#include<bits/stdc++.h>
using namespace std;
vector<pair<int,int> >v;
vector<pair<int,int> >v1;
int main()
{
	int i,n,x,y,z=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>x>>y;
		if(x>0)
			v.push_back(make_pair(x,y));
		else
			v1.push_back(make_pair(-1*x,y));	
	}
	sort(v.begin(),v.end());
	sort(v1.begin(),v1.end());
	i=0;
	while(i<min(v.size(),v1.size()))
	{
		z+=v1[i].second;
		z+=v[i].second;
		i++;
	}
	if(i<v.size())
		z+=v[i].second;
	if(i<v1.size())
		z+=v1[i].second;
	cout<<z<<endl;		
	return 0;
}
