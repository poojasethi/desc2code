#include <bits/stdc++.h>
using namespace std;
bool arr[1000][1000];
bool p[1000];
int main()
{
    int n,x=0,z=0,k=0;
    cin >>n;
    pair<int,int>ar[150];
    for(int i=0;i<n;i++){
        cin >>ar[i].first>>ar[i].second;
        if(ar[i].first<0)x++;
        else z++;
    }
    sort(ar,ar+n);
    for(int i=max(0,x-z-1);i<=min(n,2*x);i++){
        k+=ar[i].second;
    }
    cout <<k;
    return 0;
}
