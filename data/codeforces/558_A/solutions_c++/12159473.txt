#include <iostream>
#include <algorithm>

using namespace std;
int n,i,sum=0,nr;
pair<int,int> a[105];
int main()
{
    cin>>n;
    for(i=1;i<=n;++i) cin>>a[i].first>>a[i].second;
    sort(a+1,a+n+1);
    for(i=1;i<=n&&a[i].first<0;++i);--i;
    nr=i;
    if(nr*2==n) for(i=1;i<=n;++i) sum+=a[i].second;
    else if(nr<n-nr) for(i=1;i<=2*nr+1;++i) sum+=a[i].second;
    else for(nr=n-nr,i=n-2*nr;i<=n;++i) sum+=a[i].second;
    cout<<sum<<endl;
    return 0;
}
