n = int(raw_input())
p = []
m = []
for i in xrange(n):
    x, a = map(int, raw_input().split())
    if x > 0:
        p.append((x, a))
    else:
        m.append((x, a))
p.sort()
m.sort(reverse=True)
p = [x[1] for x in p]
m = [x[1] for x in m]
t = min(len(p), len(m))
a = sum(p[:t+1]) + sum(m[:t])
b = sum(p[:t]) + sum(m[:t+1])
print max(a, b)
