L, R = [], []
for i in xrange(input()):
    x, y = map(int, raw_input().split())
    if x > 0: R.append((x, y))
    else: L.append((x, y));

R.sort()
L.sort(reverse = True)

ans = 0
while L and R:
    ans += L.pop(0)[1] + R.pop(0)[1]
    
    
if L:
    ans += L[0][1]
if R:
    ans += R[0][1]
print ans
