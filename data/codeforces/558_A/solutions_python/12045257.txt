n = input()

left = []
right = []

l = 0
r = 0

for i in xrange(n):
    x,a = map(int,raw_input().split())
    if x < 0:
        left.append((x,a))
        l += 1
    else:
        right.append((x,a))
        r += 1

left = sorted(left, key=lambda t: t[0])
left.reverse()
right = sorted(right, key=lambda t: t[0])

def s(l):
    def f(t):
        return t[1]
    return sum(map(f,l))

if l > r:
    ans = s(left[:r+1]) + s(right)
else:
    ans = s(right[:l+1]) + s(left)

print ans
