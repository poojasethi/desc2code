n=input()
p=[]
m=[]
for i in xrange(n):
    t=[map(int,raw_input().split())]
    if(t[0][0]>0):
        p+=t
    else:
        m+=t
p.sort()
m.sort()
p1=[]
m1=[]
p1+=[i[1] for i in p]
m1+=[i[1] for i in m]
l1=len(p)
l2=len(m)
if(l1==l2):
    print sum(p1)+sum(m1)
elif(l1>l2):
    print sum(p1[:l2+1])+sum(m1)
else:
    m1.reverse()
    print sum(p1)+sum(m1[:l1+1])
