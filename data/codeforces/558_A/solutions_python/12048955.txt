t=int(raw_input())
x=[]
a=[]
r=0
l=0
end=t-1
start=0
for i in range(t):
	t,b= map(int, raw_input().split())
	x.append(t)
	a.append(b)
	if x[i]>0:
		r+=1
	else:
		l+=1
new=zip(x,a)
new.sort()
if r>l:
	end-=(r-l-1)
elif (l>r):
	start+=(l-r-1)	
sums=0
#print range(start,end+1)
#print new[0]
for i in range(start,end+1):
	sums+=new[i][1]
print sums
