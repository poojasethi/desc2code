n = input()
l,r = [],[]
for _ in xrange(n):
    x,a = map(int,raw_input().split())
    if x > 0:
        r.append((x,a))
    else:
        l.append((x,a))
l = sorted(l,reverse=1); r = sorted(r)
if len(l) == len(r):
    print sum(a for x,a in l) + sum(a for x,a in r)
elif len(l) > len(r):
    print sum(a for x,a in l[:len(r)+1]) + sum(a for x,a in r)
elif len(l) < len(r):
    print sum(a for x,a in r[:len(l)+1]) + sum(a for x,a in l)
