n=input()
a=[[],[]]
for _ in range(n):
    k=map(int,raw_input().split())
    a[k[0]>0].append((abs(k[0]),k[1]))
l=min(len(a[0]),len(a[1]))
x=y=z=w=0
a[0].sort()
a[1].sort()
for i in range(l):
    x+=a[0][i][1]
    y+=a[1][i][1]
z=x if len(a[0])==l else x+a[0][l][1]
w=y if len(a[1])==l else y+a[1][l][1]
print max(x+w,y+z)
