n = int(raw_input())
left = []
right = []
for i in xrange(n):
    x, a = map(int, raw_input().split())
    if x > 0:
        right.append([x, a])
    else:
        left.append([x, a])
left.sort()
left.reverse()
right.sort()
x = min(len(left), len(right))
print sum((d[i][1] if i < len(d) else 0)  for i in xrange(x + 1) for d in (left, right))
