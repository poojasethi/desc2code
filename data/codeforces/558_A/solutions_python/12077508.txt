n=input()
ar=[ map(int,raw_input().split()) for i in range(n)]
pos=0
neg=0

ar.sort()

for i in range(n):
    if ar[i][0]<0 :
        neg+=1
    else:
        break

pos=n-neg

if pos==neg:
    be=0
    en=n-1
elif pos>neg:
    be=0
    en=neg+neg
else:
    be=neg-(pos+1)
    en=n-1

ans=0
for i in range(be,en+1):
    ans+=ar[i][1]


print ans
