n = int(input())
L = []
R = []
for i in xrange(n):
    x, a = map(int, raw_input().split())
    if x < 0:
        L.append((x, a))
    else:
        R.append((x, a))
L.sort()
L.reverse()
R.sort()
ans = 0
if len(L) < len(R):
    ans = sum(a for _, a in L)
    ans += sum(R[i][1] for i in xrange(len(L) + 1))
elif len(L) > len(R):
    ans = sum(a for _, a in R)
    ans += sum(L[i][1] for i in xrange(len(R) + 1))
else:
    ans = sum(a for _, a in L)
    ans += sum(a for _, a in R)
print ans
