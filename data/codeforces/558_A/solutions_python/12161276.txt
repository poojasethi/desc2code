n = int(input())
left , right = [] , []

for i in range(0,n):
    x , a = map(int,raw_input().split())
    if x > 0 :
        right.append([x,a])
    else :
        left.append([x,a])

left.sort(reverse = True)
right.sort()

if len(right) > len(left) :
    right = right[:len(left)+1]
elif len(right) < len(left) :
    left = left[:len(right)+1]
print sum(i[1] for i in right + left)
