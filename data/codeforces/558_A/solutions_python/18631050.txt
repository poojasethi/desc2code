n = input()

left = []
right = []

for _ in range(n):
    x, y = map(int, raw_input().split())
    pair = [x, y]

    if x < 0:
        left.append(pair)
    else:
        right.append(pair)

left = sorted(left, reverse=True)
right = sorted(right)

minlr = min(len(left), len(right))
count = 0

for i in range(minlr):
    count += left[i][1] + right[i][1]

if len(left) > len(right):
    count += left[len(right)][1]
elif len(left) < len(right):
    count += right[len(left)][1]

print count
