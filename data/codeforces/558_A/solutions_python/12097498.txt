n=int(raw_input())
p,m=[],[]

for i in xrange(n):
    x,a=map(int,raw_input().split())
    if x>=0:
        p.append((x,a))
    else:
        m.append((-x,a))
p.sort()
m.sort()
p=[x[1] for x in p]
m=[x[1] for x in m]
t=min(len(p),len(m))
p.append(0)
m.append(0)
ans=sum(p[:t])+sum(m[:t])
ans += max(p[t],m[t])
print ans
