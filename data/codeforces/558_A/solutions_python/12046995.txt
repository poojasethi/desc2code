n = int(raw_input())
l, r = [], []
for x in range(n):
    a, b = map(int, raw_input().split())
    if a < 0:
        l.append((a, b))
    else:
        r.append((a, b))


l.sort()
r.sort()
l.reverse()
csum = lambda a: sum([y for (x, y) in a])

if len(l) == len(r):
    print csum(l) + csum(r)
elif len(l) < len(r):
    print csum(l) + csum(r[:len(l)+1])
else:
    print csum(l[:len(r)+1]) + csum(r)
