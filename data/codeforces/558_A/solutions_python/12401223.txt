n = int(raw_input())
left = []
right = []
for i in xrange(n):
  [a, b] = map(int, raw_input().split())
  if a < 0:
    left.append((a, b))
  else:
    right.append((a, b))

left.sort(reverse=True)
right.sort()

left.append((0, 0))
right.append((0, 0))
res = 0
for ((i1, v1), (i2, v2)) in zip(left, right):
  res += v1 + v2

print res
