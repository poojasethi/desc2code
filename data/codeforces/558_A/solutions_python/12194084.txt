n = int(raw_input())
neg = []
pos = []
for _ in range(n):
    x, a = map(int, raw_input().split())
    if x < 0: neg.append([-1*x, a])
    else: pos.append([x, a])
neg.sort()
pos.sort()
posi = [t[1] for t in pos]
negi = [t[1] for t in neg]
if len(posi) < len(negi): print sum(posi) + sum(negi[:len(posi)+1])
elif len(posi) > len(negi): print sum(posi[:len(negi) + 1]) + sum(negi)
else: print sum(posi) + sum(negi)
