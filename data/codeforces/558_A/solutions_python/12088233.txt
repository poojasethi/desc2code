n = int(raw_input())
pos = []
neg = []
for i in xrange(n):
    x, a = map(int, raw_input().split())
    if x > 0:
        pos.append((x, a))
    else:
        neg.append((-x, a))
t = min(len(pos), len(neg))
pos.sort()
neg.sort()
pos.append((100009, 0))
pos.append((100010, 0))
neg.append((100009, 0))
neg.append((100010, 0))
pos = [x[1] for x in pos]
neg = [x[1] for x in neg]
print max(sum(pos[:t]) + sum(neg[:t + 1]), sum(pos[:t + 1]) + sum(neg[:t]))
