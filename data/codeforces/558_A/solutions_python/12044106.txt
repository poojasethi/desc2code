n = int(raw_input())

r = []
left, right = 0, 0

for i in xrange(n):
    x, a = map(int, raw_input().split())
    r.append((x, a))
    if x > 0:
        right += 1
    else:
        left += 1

r.sort()

s = 0

if left == right:
    s = sum(a for (x, a) in r)
elif left < right:
    s = sum(a for (x, a) in r[: left + left + 1])
else:
    r.reverse()
    s = sum(a for (x, a) in r[: right + right + 1])

print s
