n = int(raw_input())

l, r = [[0, 0]], [[0, 0]]

for i in range(n):
    x, y = map(int, raw_input().rstrip().split())
    if x < 0:
        l += [[-x, y]]
    else:
        r += [[x, y]]

l = sorted(l, key=lambda x: x[0])
r = sorted(r, key=lambda x: x[0])

Len = min(len(l), len(r))

ans = sum(zip(*l[:Len])[1]) + sum(zip(*r[:Len])[1])

if len(l) > Len:
    ans += l[Len][1]

if len(r) > Len:
    ans += r[Len][1]
print ans
