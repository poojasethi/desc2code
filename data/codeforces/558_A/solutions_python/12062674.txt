N = int(raw_input())
neg, pos = [], []
for i in xrange(N):
    a, b = map(int, raw_input().split())
    if a < 0: neg.append( (-a,b) )
    else: pos.append( (a, b) )

neg.sort()
pos.sort()
m = min(len(neg), len(pos))
ans = sum([x[1] for x in neg[:min(m+1, len(neg))]]) + sum([x[1] for x in pos[:min(m+1, len(pos))]])
print ans
