n = input()

l = []
r = []
for x in range(n):
    a, b = map(int, raw_input().split())
    if a < 0:
        l.append((a, b))
    else:
        r.append((a, b))

l.sort()
r.sort()

sum_second = lambda ll: sum([b for (a, b) in ll])

if len(l) == len(r):
    print sum_second(l) + sum_second(r)

if len(l) > len(r):
    print sum_second(r) + sum_second(l[len(l)-len(r)-1:])

if len(l) < len(r):
    print sum_second(l) + sum_second(r[:len(l)+1])
