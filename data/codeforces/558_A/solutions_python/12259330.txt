#! /usr/bin/python
__author__ = 'Acesine'


n = int(raw_input())
t = {}
neg = 0
pos = 0
for _ in range(n):
    [x, a] = [int(i) for i in raw_input().split(" ")]
    t[x] = a
    if x < 0:
        neg += 1
    else:
        pos += 1
keys = t.keys()
keys.sort()
m = 0
if neg == pos:
    m = sum(t.values())
else:
    if neg < pos:
        for i in keys[:2*neg+1]:
            m += t[i]
    else:
        for i in keys[::-1][:2*pos+1]:
            m += t[i]
print m
