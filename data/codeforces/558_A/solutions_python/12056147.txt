x=input()
neg = 0
pos = 0
tot = 0
arrayneg=[]
arraypos=[]
for y in range(x):
	a,b = map(int, raw_input().split())
	if a < 0:
		neg+=1
		arrayneg.append((a,b))
	else: 	
		pos+=1
		arraypos.append((a,b))
arrayneg.sort(reverse=True)
arrayneg = [i[1] for i in arrayneg]
arraypos.sort()
arraypos = [i[1] for i in arraypos]
tot = 0
if neg==pos: 
	tot=sum(arrayneg)+sum(arraypos)
elif neg>pos:
	tot=sum(arrayneg[0:pos+1])+sum(arraypos)
else:
	tot=sum(arrayneg)+sum(arraypos[0:neg+1])

print tot
