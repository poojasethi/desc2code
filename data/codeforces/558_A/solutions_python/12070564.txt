n = int(input(""))
cnt = 0
la=0
ra=0
a=[]
for i in range(0,n):
    x1,x2=map(int,raw_input("").split())
    a.append([x1,x2])
    if x1 < 0:
        la=la+1
    elif x2 >0:
        ra=ra+1
    cnt=cnt+x2
a.sort()
t=1
if la <ra :
    while la<ra-t:
        cnt =cnt -a[n-t][1]
        t=t+1
if la >ra:
    while la>ra+t:
        cnt =cnt- a[t-1][1]
        t=t+1
print cnt
