n = input()
P = []
N = []
for i in xrange(n):
    a = map(int, raw_input().split())
    if a[0] > 0:
        P.append(a)
    else:
        N.append(a)
        
P.sort()
N.sort()
N.reverse()

lp, ln= len(P), len(N)
k = min(ln, lp)
ans = 0
for i in xrange(k):
    ans += (P[i][1] + N[i][1])
P.append([0,0])
N.append([0,0])
ans += max(P[k][1], N[k][1])

print ans
