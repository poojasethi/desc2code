n = int(raw_input())
vs = [map(int, raw_input().split()) for _ in range(n)]

a = sorted([v for v in vs if v[0] > 0])
b = sorted([v for v in vs if v[0] < 0], reverse=True)

l = min(len(a), len(b))

ans1 = sum(y for x, y in a[:l]) + sum(y for x, y in b[:l + 1])
ans2 = sum(y for x, y in a[:l + 1]) + sum(y for x, y in b[:l])

print(max(ans1, ans2))
