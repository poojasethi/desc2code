N = input()

Left, Right = [], []

for i in range(1, N + 1):
	X, A = map(int, raw_input().split())

	if X < 0:
		Left.append((X, A))
	else:
		Right.append((X, A))

Left.sort(reverse = True)
Right.sort()

minLen = min(len(Left), len(Right))
ans = sum([i for _, i in Left[0 : minLen]]) + sum([i for _, i in Right[0 : minLen]])

if len(Left) > minLen:
	ans += Left[minLen][1]

if len(Right) > minLen:
	ans += Right[minLen][1]

print ans
