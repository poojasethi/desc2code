a=int(raw_input())
neg=[]
pos=[]
ans1=0
ans2=0
for i in xrange(a):
    b,c=map(int,raw_input().split())
    if b<0:neg.append((b,c))
    else:pos.append((b,c))
neg.sort()
neg.reverse()
pos.sort()
neg2=neg[:]
pos2=pos[:]
while 1:
    if not pos:break
    ans1+=pos.pop(0)[1]
    if not neg:break
    ans1+=neg.pop(0)[1]
while 1:
    if not neg2:break
    ans2+=neg2.pop(0)[1]
    if not pos2:break
    ans2+=pos2.pop(0)[1]
print max(ans1,ans2)
