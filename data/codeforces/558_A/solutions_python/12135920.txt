nc = pc = 0
na = []
pa = []
for _ in range(int(raw_input())):
    x, a = map(int, raw_input().split())
    if x < 0:
        nc += 1
        na.append((x, a))
    else:
        pc += 1
        pa.append((x, a))
if nc == pc:
    print sum(j for i, j in na) + sum(j for i, j in pa)
elif nc > pc:
    print sum(j for i, j in pa) + sum(j for i, j in sorted(na, reverse=True)[:pc + 1])
else:
    print sum(j for i, j in na) + sum(j for i, j in sorted(pa)[:nc + 1])
