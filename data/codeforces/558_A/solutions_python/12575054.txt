n=int(raw_input())
l,r=[],[]
ans=chk=0
while n:
    n-=1
    x,a=map(int,raw_input().split())
    if x<0:
        l.append((-x,a))
    else:
        r.append((x,a))
l.sort()
r.sort()
def apple(p,q):
    s=d=0
    while 1:
        try:
            if d%2:
                s+=p[d/2][1]
            else:
                s+=q[d/2][1]
        except IndexError:
            break
        d+=1
    return s

print max(apple(l,r),apple(r,l))
