from sys import stdin
n = int(stdin.readline())
neg = []
pos = []
for _ in xrange(n):
    line = map(int, stdin.readline().split())
    if line[0]>0:
        pos.append((line[0], line[1]))
    else:
        neg.append((line[0], line[1]))
pos.sort()
neg.sort()
ans = 0
i = -1
for i in xrange(min(len(neg), len(pos))):
    ans += pos[i][1]+neg[-i-1][1]
if len(neg)>len(pos):
    ans += neg[-i-2][1]
elif len(pos)>len(neg):
    ans += pos[i+1][1]
print ans
