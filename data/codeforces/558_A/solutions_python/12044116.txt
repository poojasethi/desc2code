n = int(raw_input())
xn = []
xp = []
for _ in xrange(n):
	xi, ai = map(int, raw_input().split())
	if xi < 0:
		xn.append((xi, ai))
	else:
		xp.append((xi, ai))
xn.sort(reverse=True)
xp.sort()
n = len(xn)
p = len(xp)
if n < p:
	p = n + 1
	print sum(i for _, i in xn) + sum(i for _, i in xp[: p])
else:
	n = min(n, p + 1)
	print sum(i for _, i in xp) + sum(i for _, i in xn[: n])
