n = input()
Left, Right = [], []
for i in xrange(n):
    a, b = map(int, raw_input().split())
    if a < 0:
        Left.append((a, b))
    else :
        Right.append((a, b))
Right.sort()
Left.sort(reverse = True)
llen = len(Left)
rlen = len(Right)
if llen < rlen:
    rlen = llen + 1
elif rlen < llen:
    llen = rlen + 1
print sum(p[1] for p in Left[:llen] + Right[:rlen])
