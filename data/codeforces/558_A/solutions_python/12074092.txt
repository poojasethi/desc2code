N = input()

Left, Right = [], []

for i in range(1, N + 1):
	X, A = map(int, raw_input().split())

	if X < 0:
		Left.append((X, A))
	else:
		Right.append((X, A))

Left.sort(reverse = True)
Right.sort()

lenLeft = len(Left)
lenRight = len(Right)

if lenLeft > lenRight: 
	lenLeft = lenRight + 1
elif lenRight > lenLeft:
	lenRight = lenLeft + 1

print sum(p[1] for p in Left[: lenLeft] + Right[: lenRight])
