lList=[]
rList=[]

n=int(raw_input())
for i in range(n):
 a,b=raw_input().split()
 a=int(a)
 b=int(b)
 if a<0:
  lList.append((-a,b))
 else:
  rList.append((a,b))
lList.sort(None,lambda d:d[0],False)
rList.sort(None,lambda d:d[0],False)

answer=0
lLen=len(lList)
rLen=len(rList)
iLen=min(lLen, rLen)
for i in xrange(iLen):
 answer+=lList[i][1]+rList[i][1]
if lLen>rLen:
 answer+=lList[iLen][1]
if rLen>lLen:
 answer+=rList[iLen][1]
print answer
