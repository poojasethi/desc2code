n = input()

left = []
right = []

for _ in xrange(n):
    t = x, y = map(int, raw_input().split())

    if x < 0:
        left.append(t)
    else:
        right.append(t)

left.sort(reverse=1)
right.sort()

l_left = len(left)
l_right = len(right)

if l_left > l_right:
    l_left = l_right + 1
elif l_right > l_left:
    l_right = l_left + 1

print sum(t[1] for t in left[:l_left] + right[:l_right])
