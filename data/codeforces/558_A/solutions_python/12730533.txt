N,L,R=input(),[],[]
for i in xrange(N):
    x,a=map(int, raw_input().split())
    if x<0: L.append((x,a))
    else: R.append((x,a))
L=sorted(L)
R=sorted(R)
A=lambda x: x[1]
L=map(A,L)
R=map(A,R)
if len(L) > len(R):
    print sum(L[-len(R)-1:])+sum(R)
elif len(L) < len(R):
    print sum(L)+sum(R[:len(L)+1])
else:
    print sum(L)+sum(R)
