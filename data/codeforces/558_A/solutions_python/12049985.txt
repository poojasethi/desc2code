n=int(raw_input())
ans=chk=0
l,r=[],[]
for i in range(n):
    x,a=map(int,raw_input().split())
    if x<0:
        l.append((-x,a))
    else:
        r.append((x,a))
l.sort()
r.sort()

def sol(a,b):
    chk=cnt=0
    while 1:
        try:
            if chk%2==0:
                cnt+=a[chk/2][1]
            else:
                cnt+=b[chk/2][1]
        except IndexError:
            return cnt
        chk+=1


print max(sol(l,r),sol(r,l))
