n = int(raw_input())
vn=[]
vp=[]
for i in range(n):
	a,b = map(int,raw_input().split(" "))
	if(a<0):
		t = []
		t.append(a)
		t.append(b)
		vn.append(t)
	else:
		t = []
		t.append(a)
		t.append(b)
		vp.append(t)
sump=0
vp.sort()
vn.sort()
vn.reverse()
minp = min(len(vn),len(vp))
for i in range(minp):
	sump+=vn[i][1]+vp[i][1]
if(len(vn)<len(vp)):
	sump+=vp[minp][1]
elif(len(vn)>len(vp)):
	sump+=vn[minp][1]
print sump
