#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
#define LL long long
const LL mod=1000000007ll;

LL n,k,l,C[100][100],E[100],V[100],ans=0;

struct mt
{
	LL s[100][100];

}A,I;

mt operator *(mt A,mt B)
{
	mt C1;
	for (int i=0;i<l;i++) 
		for (int j=0;j<l;j++)
		{
			C1.s[i][j]=0;
			for (int k=0;k<l;k++)
				(C1.s[i][j]+=A.s[i][k]*B.s[k][j])%=mod;
		}
	return C1;
}

int main()
{
	cin>>n>>k;l=k*2+3;
	for (int i=0;i<=k;i++) C[i][0]=C[i][i]=1;
	for (int i=1;i<=k;i++) 
		for (int j=1;j<i;j++) 
			C[i][j]=(C[i-1][j-1]+C[i-1][j])%mod;
	E[0]=1;
	for (int i=1;i<=k+2;i++) E[i]=(E[i-1]<<1)%mod;
	for (int i=0;i<l;i++) for (int j=0;j<l;j++) A.s[i][j]=0;
	A.s[0][0]=1;
	for (int i=0;i<=k;i++)
	{
		for (int j=0;j<=i;j++) A.s[i+1][j+1]=C[i][j];
		for (int j=0;j<=i;j++) A.s[i+1][j+k+2]=(C[i][j]*E[i-j])%mod;
	}
	for (int i=0;i<=k;i++) A.s[i+k+2][i+1]=1;
	for (int i=1;i<l;i++) A.s[0][i]=A.s[k+1][i];
	V[0]=E[k+1]+1;
	for (int i=0;i<=k;i++) V[i+1]=E[i+1];
	for (int i=k+2;i<=2*k+2;i++) V[i]=1;
	for (int i=0;i<l;i++) for (int j=0;j<l;j++) I.s[i][j]=0;
	for (int i=0;i<l;i++) I.s[i][i]=1;
	if (n==1) {cout<<1;return 0;}
	n-=2;
	while(n)
	{
		if (n&1) I=I*A;
		A=A*A;
		n>>=1;
	}
	for (int i=0;i<l;i++) (ans+=I.s[0][i]*V[i])%=mod;
	cout<<ans;
	return 0;
}
