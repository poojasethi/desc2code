#include <iostream>
#include <cstdio>
#include <cstdlib>
#define N 1000000007
#define M
#define LL long long
using namespace std;

LL t,k,n;
LL c[41][41];

struct mat
{
	LL a[90][90];
	struct mat operator *(const struct mat& b) const
	{
		struct mat ans;
		int i,j,y;
		for(i=0;i<n;i++)
		  for(j=0;j<n;j++)
		    ans.a[i][j]=0;
		
		for(i=0;i<n;i++)
		  for(j=0;j<n;j++)  
		  	for(y=0;y<n;y++)
		    {
		    	ans.a[i][j]+=a[i][y]*b.a[y][j];
		    	ans.a[i][j]%=N;
		    }
		return ans;
	}
};
struct mat qpow(struct mat a,LL asd)
{
	int i,j;
	struct mat ans;
	for(i=0;i<n;i++)
	  for(j=0;j<n;j++)
	    if(i==j) ans.a[i][j]=1;
	    else ans.a[i][j]=0;
	while(asd)
	{
		if(asd&1)
		  ans=ans*a;
		asd=asd>>1;
		a=a*a;
	}
	return ans;
}

void initialize()
{
	int i,j;
	cin>>t>>k;
	n=2*k+3;
	for(i=0;i<=k;i++)
	{
		c[i][0]=c[i][i]=1;
		for(j=1;j<i;j++)
		  c[i][j]=c[i-1][j-1]+c[i-1][j];
	}
}

int main()
{
	LL ans=0;
	int i,j;
	initialize();
	struct mat p;
	for(i=0;i<n;i++)
	  for(j=0;j<n;j++)
	    p.a[i][j]=0;
	
	
	for(i=0;i<=k;i++)
	{
		for(j=0;j<=i;j++)
		{
			p.a[i][j]=c[i][j];
			p.a[i+1+k][j]=c[i][j];
			p.a[i][j+1+k]=c[i][j];
		}
	}
	
	p.a[2*k+2][k]=1;
	p.a[2*k+2][2*k+2]=1;
	p=qpow(p,t);
	for(i=0;i<2*k+2;i++)
	{
		ans+=p.a[2*k+2][i];
		ans%=N;
	}
	cout<<ans<<endl;
}
