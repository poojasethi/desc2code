#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;
#define MAXN 85
#define MOD 1000000007

int n,k;
long long N;
int po[MAXN];
int init[MAXN][MAXN],tmp[MAXN][MAXN],ans[MAXN][MAXN],c[MAXN][MAXN];
void MUL(int a[MAXN][MAXN])
{
    memset(tmp,0,sizeof(tmp));
    for (int i=0;i<=n;i++)
        for (int j=0;j<=n;j++)
            for (int k=0;k<=n;k++)
                if (a[k][j])
                    tmp[i][j]=(tmp[i][j]+(long long)ans[i][k]*a[k][j])%MOD;
    memcpy(ans,tmp,sizeof(tmp));
}
void POW(long long p)
{
    if (p==1)
    {
        memcpy(ans,init,sizeof(ans));
        return;
    }
    POW(p/2);
    MUL(ans);
    if (p&1) MUL(init);
}
int main()
{
    //freopen("in.txt","r",stdin);
    po[0]=1;
    for (int i=1;i<=45;i++) po[i]=po[i-1]*2%MOD;
    c[0][0]=1;
    for (int i=1;i<=45;i++)
    {
        c[i][0]=1;
        for (int j=1;j<=i;j++)
            c[i][j]=(c[i-1][j-1]+c[i-1][j])%MOD;
    }
    while (scanf("%I64d%d",&N,&k)!=-1)
    {
        memset(init,0,sizeof(init));
        for (int i=0;i<=k;i++)
            for (int j=0;j<=i;j++)
                init[j][i]=c[i][j];
        for (int i=0;i<=k;i++)
            for (int j=0;j<=i;j++)
                init[j+k+1][i]=(long long)c[i][j]*po[i-j]%MOD;
        for (int i=0;i<=k;i++)
            init[i][i+k+1]=1;
        n=2*k+2;
        init[k][n]=init[n][n]=1;
        POW(N+1);
        printf("%d\n",ans[0][n]);
    }
    return 0;
}
