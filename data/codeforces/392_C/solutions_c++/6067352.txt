#include<cstdio>
#include<cstdlib>
#define P 1000000007
#define LL long long
#define rep(i,n) for (int i=0;i<=n;i++)
int c[51][51];
LL N;
int K,i,j,ans;
int top,dd[105][105],f[105][105],tmp[105][105];
int power(LL b){
	rep(i,top) rep(j,top) dd[i][j]=f[i][j];
	while (b){
		if(b&1){
			rep(i,top) rep(j,top) tmp[i][j]=0;
			rep(i,top) rep(j,top) rep(k,top) tmp[i][j]=(tmp[i][j]+(LL)dd[i][k]*f[k][j])%P;
			rep(i,top) rep(j,top) dd[i][j]=tmp[i][j];
		}
		b>>=1;
		rep(i,top) rep(j,top) tmp[i][j]=0;
		rep(i,top) rep(j,top) rep(k,top) tmp[i][j]=(tmp[i][j]+(LL)f[i][k]*f[k][j])%P;
		rep(i,top) rep(j,top) f[i][j]=tmp[i][j];
		
		
		
	}
/*	rep(i,top){
		printf("\n");
		rep(j,top) printf("%d ",dd[i][j]);
		}printf("---------------\n");*/
	return dd[top-1][top];
}
#undef int
int main(){
	scanf("%I64d%d",&N,&K);
	c[0][0]=1;
	for (i=1;i<=K;i++){
		c[i][0]=1;
		for (j=1;j<=i;j++)
		c[i][j]=(c[i-1][j-1]+c[i-1][j])%P;
	}
//	for (i=0;i<=K;i++,printf("\n"))
//		for (j=0;j<=i;j++) printf("%d ",c[i][j]);
	for (i=0;i<=2*K+1;i++)
		for (j=0;j<=2*K+1;j++){
			if(i<=K&&j<=K) continue;
			if(i<=K&&j>K) f[i][j]=c[2*K+1-j][K-i];
			if(i>K&&j<=K) f[i][j]=c[K-j][2*K+1-i];
			if(i>K&&j>K) f[i][j]=c[2*K+1-j][2*K+1-i];
		}
	f[2*K+2][2*K+2]=1;
	f[K+1][2*K+2]=1;
//	for (i=0;i<=2*K+2;i++,printf("\n"))
//		for (j=0;j<=2*K+2;j++) printf("%3d ",f[i][j]);
	top=2*K+2;
	ans=power(N);
	printf("%d\n",ans);
}
