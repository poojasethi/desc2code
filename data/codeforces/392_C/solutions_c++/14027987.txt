#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
typedef long long ll;
#define rep(i,n) for (int i=0;i<n;++i)
const int P=1000000007;
ll T,a[85][85],b[85][85],f[85],g[85]; int n;
void mul()
{
	rep(i,n) rep(j,n) b[i][j]=0;
	rep(k,n) rep(i,n) rep(j,n) b[i][j]=(b[i][j]+a[i][k]*a[k][j])%P;
	rep(i,n) rep(j,n) a[i][j]=b[i][j];
}
void mul1()
{
	rep(i,n) g[i]=0;
	rep(i,n) rep(j,n) g[j]=(g[j]+f[i]*a[j][i])%P;
	rep(i,n) f[i]=g[i];
}
int main()
{
	scanf("%I64d%d",&T,&n),++n;
	rep(i,n){
		rep(j,i+1) a[i][j]=i&&j?(a[i-1][j]+a[i-1][j-1])%P:1;
		rep(j,n) a[i][j+n]=a[i+n][j]=a[i][j];
	}
	a[n*2][n-1]=a[n*2][n*2]=1; n*=2; rep(i,n) f[i]=1; f[n++]=0;
	for (;T;T>>=1,mul()) if (T&1) mul1();
	printf("%I64d\n",f[n-1]);
	return 0;
}
