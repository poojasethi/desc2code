#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
#define mod 1000000007
using namespace std;

typedef long long int num;
typedef num arr[88];
typedef arr arr2[88];

arr R0,R1;
num *ret=R0,*ret1=R1;
arr2 M0,M1;
arr *mat=M0,*mat1=M1;

num n;
int k,i,j,t,k1,siz;
arr2 c,c2;

int main()
{
	scanf("%I64d%d",&n,&k);
	for (i=0;i<=k;i++)
	{
		c[i][0]=1;
		c2[i][0]=(1ll<<i)%mod;
	}
	for (i=1;i<=k;i++)
		for (j=i;j<=k;j++)
		{
			c[j][i]=(c[j-1][i]+c[j-1][i-1])%mod;
			c2[j][i]=(c[j][i]<<j-i)%mod;
		}
	k1=k+1;
	siz=(k1<<1)+1;
	for (i=0;i<k1;i++)
	{
		for (j=0;j<k1;j++)
		{
			mat[i][j]=c[j][i];
			mat[i+k1][j]=c2[j][i];
		}
		mat[i][i+k1]=1;
		mat[i][siz-1]=mat[i][k1-1];
		mat[i+k1][siz-1]=mat[i+k1][k1-1];
	}
	mat[siz-1][siz-1]=1;
	for (i=0;i<=k1;i++)
		ret[i]=1;
	ret[siz-1]=1;
	for (n--;n>0;n>>=1)
	{
		if (n&1)
		{
			for (i=0;i<siz;i++)
			{
				ret1[i]=0;
				for (j=0;j<siz;j++)
					ret1[i]=(ret1[i]+ret[j]*mat[j][i])%mod;
			}
			swap(ret,ret1);
		}
		for (i=0;i<siz;i++)
			for (j=0;j<siz;j++)
			{
				mat1[i][j]=0;
				for (t=0;t<siz;t++)
					mat1[i][j]=(mat1[i][j]+mat[i][t]*mat[t][j])%mod;
			}
		swap(mat,mat1);
	}
	printf("%I64d\n",ret[siz-1]);
}
