#include<iostream>
#include<stdio.h>
#include<string.h>
#define mod 1000000007
using namespace std;
struct mat
{
    long long a[105][105];
    mat(){memset(a,0,sizeof(a));}
};
int n,k;long long s;
mat mul(mat p, mat q) {
    mat ans;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {

            for (int t = 0; t < n; t++) {
                ans.a[i][j] += p.a[i][t]*q.a[t][j];
                ans.a[i][j] %= mod;
            }
        }
    }
    return ans;
}
mat pow(mat x,long long nn)
{
    mat temp;
    for(int i=0;i<n;i++)
         temp.a[i][i]=1;
    while(nn)
    {
        if(nn&1) temp=mul(temp,x);
        x=mul(x,x);
        nn/=2;
    }
    return temp;
}
long long c[105][105];
int main()
{
    cin>>s>>k;
     n=k*2+3;
    for(int i=0;i<=k;i++)
    {
        c[i][0]=c[i][i]=1;
        for(int j=1;j<i;j++)
            c[i][j]=c[i-1][j-1]+c[i-1][j];
    }
    mat ls;
    for(int i=0;i<=k;i++)
        for(int j=0;j<=i;j++)
    {
        ls.a[i][j]=ls.a[i+k+1][j]=ls.a[i][j+k+1]=c[i][j];
    }
    ls.a[2*k+2][k]=ls.a[2*k+2][2*k+2]=1;
    mat ret=pow(ls,s);
    long long ans=0;
    for(int i=0;i<2*k+2;i++)
        ans=(ans+ret.a[2*k+2][i])%mod;
    cout<<ans<<endl;
}
