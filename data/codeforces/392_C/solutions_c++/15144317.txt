#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#define rep(i,k) for (i=1; i<=k; i++)
using namespace std;
long long a[88][88],b[88][88],c[88][88],n,ans;
int k,i,j;
const int MOD=1000000007;
void ksm(long long x)
{
    if (x==1) return;
    ksm(x/2);
    int i,j,l;
    rep(i,k) rep(j,k) {c[i][j]=0; rep(l,k) c[i][j]=(c[i][j]+a[i][l]*a[l][j])%MOD;}
    rep(i,k) rep(j,k) a[i][j]=c[i][j];
    if (x%2==1)
    {
        rep(i,k) rep(j,k) {c[i][j]=0; rep(l,k) c[i][j]=(c[i][j]+a[i][l]*b[l][j])%MOD;}
        rep(i,k) rep(j,k) a[i][j]=c[i][j];
    }
}
int main()
{
    cin>>n>>k; k++;a[k+2][2]=a[2][k+2]=a[k+2][k+2]=1;
    for (i=k+2; i<=2*k+1; i++)
      for (j=i-k; j<=k+1; j++)
        if (i!=k+2 || j!=i-k)
          a[i][j]=(a[i][j-1]+a[i-1][j-1])%MOD;
    for (i=k+2; i<=2*k+1; i++)
      for (j=i; j<=2*k+1; j++)
        if (i!=k+2 || j!=i)
          a[i][j]=(a[i][j-1]+a[i-1][j-1])%MOD;
    for (i=2; i<=k+1; i++)
      for (j=i+k; j<=2*k+1; j++)
        if (i!=2 || j!=i+k)
          a[i][j]=(a[i][j-1]+a[i-1][j-1])%MOD;
    a[1][1]=a[2*k+1][1]=1;
    k=2*k+1;
    rep(i,k) rep(j,k) b[i][j]=a[i][j];
    ksm(n);
    ans=-a[1][1];
    rep (i,k) ans=(ans+a[i][1])%MOD;cout<<ans;
    return 0;
}
