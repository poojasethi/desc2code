#include <stdio.h>
#include <iostream>
#include <cstring>
using namespace std;
#define ll long long
const ll MOD=1000000007;
struct node{
	ll a[100][100];
	node(){
		memset(a,0,sizeof(a));
	}
}p,v;
ll n,pre[60][60]={1},t,k,s=0;
node operator * (node a,node b){
	node t;
	for (int i=0;i<n;i++)
		for (int j=0;j<n;j++)
			for (int k=0;k<n;k++)
				t.a[i][j]=(t.a[i][j]+a.a[i][k]*b.a[k][j])%MOD;
	return t;
}
node pow(node a,ll b){
	node t;
	for (int i=0;i<n;i++)
		t.a[i][i]=1;
	while (b>0){
		if (b&1) t=t*a;
		b>>=1;
		a=a*a;
	}
	return t;
}
int main( ){
	cin>>t>>k;
	n=k+k+3;
	for (int i=0;i<=k;i++){
		pre[i][0]=1;
		for (int j=1;j<=i;j++)
			pre[i][j]=(pre[i-1][j-1]+pre[i-1][j])%MOD;
	}
	for (int i=0;i<=k;i++)
		for (int j=0;j<=i;j++){
			p.a[i][j]=pre[i][j];
			p.a[i+k+1][j]=pre[i][j];
			p.a[i][j+k+1]=pre[i][j];
		}	
	p.a[n-1][k]=p.a[n-1][n-1]=1;
	v=pow(p,t);
	for (int i=0;i<n-1;i++)
		s=(s+v.a[n-1][i])%MOD;
	cout<<s<<endl;
	return 0;
}
