#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define int64 long long
#define mo 1000000007
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int64 i,j,k,n,m,mm;
int64 C[41][41],a[90],b[90][90];
inline void mul(int64 a[90],int64 b[90][90]) {
	int i,j; int64 c[90]={0};
	For(i,0,mm)For(j,0,mm) c[j]=(a[i]*b[i][j]+c[j])%mo;
	For(i,0,mm) a[i]=c[i];
}
inline void multi(int64 a[90][90],int64 b[90][90]) {
	int i,j,k; int64 c[90][90]={0};
	For(i,0,mm)For(j,0,mm)For(k,0,mm) c[i][k]=(a[i][j]*b[j][k]+c[i][k])%mo;
	For(i,0,mm)For(j,0,mm) a[i][j]=c[i][j];
}
int main() {
	scanf("%I64d%I64d",&n,&m); mm=m+m+2;
	For(i,0,m) {
		C[i][0]=1;
		For(j,1,i) C[i][j]=(C[i-1][j-1]+C[i-1][j])%mo;
	}
	For(i,0,mm-1) a[i]=1;
	b[mm-1][mm]=b[mm][mm]=1;
	For(i,0,m)For(j,i,m) b[i][j+m+1]=b[i+m+1][j+m+1]=b[i+m+1][j]=C[j][i];
	for (;n;n>>=1) {
		if (n&1) mul(a,b);
		multi(b,b);
	}
	printf("%I64d\n",a[mm]);
	return 0;
}
