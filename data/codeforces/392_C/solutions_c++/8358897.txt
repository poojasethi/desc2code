using namespace std;
typedef long long ll;
typedef double D;
const int mod=1000000007;
const int N=210;
ll n,a[N][N],b[N][N],f[N],g[N];
int k;
void mul(int n) {
    int i,j,k;
    for (i=1;i<=n;i++) for (j=1;j<=n;j++) b[i][j]=0;
    for (i=1;i<=n;i++) for (j=1;j<=n;j++) for (k=1;k<=n;k++) 
        b[i][k]=(b[i][k]+a[i][j]*a[j][k])%mod;
    for (i=1;i<=n;i++) for (j=1;j<=n;j++) a[i][j]=b[i][j];
}
void mul1(int n) {
    int i,j,k;
    for (i=1;i<=n;i++) g[i]=0;
    for (i=1;i<=n;i++) for (j=1;j<=n;j++) g[i]=(g[i]+f[j]*a[i][j])%mod;
    for (i=1;i<=n;i++) f[i]=g[i];
}
int main(){
    int i,j;
    cin>>n>>k;
    k++;
    a[1][1]=a[1][k+1]=a[k+1][1]=1;
    for (i=1;i<=k<<1;i++) f[i]=1;
    f[k<<1|1]=0;
    for (i=2;i<=k;i++) for (j=1;j<=i;j++) {
        a[i][j]=(a[i-1][j-1]+a[i-1][j])%mod;
        a[i+k][j]=a[i][k+j]=a[i][j];
    }
    a[k<<1|1][k<<1|1]=1;
    a[k<<1|1][k]=1;
    k=k<<1|1;
    for (;n;n>>=1LL,mul(k)) if (n&1LL) mul1(k);
    cout<<f[k]<<endl;
    return 0;
}
