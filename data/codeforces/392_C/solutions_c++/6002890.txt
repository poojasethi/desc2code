#include<iostream>
using namespace std;
#define N 85
#define Mod 1000000007
long long a[N][N],c[N][N];
struct ss
{
	long long a[N][N];
};
ss mul(ss a,ss b)
{
	ss c={0};
	int i,j,k;
	for(i=0;i<=82;i++)
	for(j=0;j<=82;j++)
	for(k=0;k<=82;k++)
	{
		c.a[i][j]+=a.a[i][k]*b.a[k][j];
		c.a[i][j]%=Mod;
	}
	return c;
}
int main()
{
	long long  i,j,n,k,l;
	for(i=0;i<41;i++)
	c[i][i]=c[i][0]=1;
	for(i=2;i<41;i++)
	for(j=1;j<i;j++)
	c[i][j]=(c[i-1][j-1]+c[i-1][j])%Mod;
	for(i=0;i<41;i++)
	for(j=0;j<=i;j++)
	a[i][j]=a[i][j+41]=a[i+41][j]=c[i][j];
	while(cin>>n>>k)
	{
		a[82][82]=a[82][k]=1;
		ss x;
		for(i=0;i<=82;i++)
		for(j=0;j<=82;j++)
		x.a[i][j]=a[i][j];
		ss ans={0};
		for(i=0;i<=82;i++)
		ans.a[i][i]=1;
		while(n)
		{
			if(n%2)
			ans=mul(ans,x);
			x=mul(x,x);
			n/=2;
		}
		long long s=0;
		for(i=0;i<82;i++)
		{
			s+=ans.a[82][i];
			s%=Mod;
		}
		cout<<s<<endl;
		a[82][k]=0;
	}
}
