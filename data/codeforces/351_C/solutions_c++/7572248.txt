#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
#define rep(i,n) for(int i=0;i<=n;i++)
long long Stilwell[100],f[101][101],g[101][101],ans[101][101],tmp[101][101];
int n,m;
int a[101],b[101];
void power(int m){
    rep(i,n) rep(j,n) ans[i][j]=0x7f7f7f7f; ans[0][0]=0;
    while (m){
        if(m&1){
            rep(i,n) rep(j,n) tmp[i][j]=0x7f7f7f7f;
            rep(i,n) rep(j,n) rep(k,n) tmp[i][j]=min(tmp[i][j],ans[i][k]+f[k][j]);
            rep(i,n) rep(j,n) ans[i][j]=tmp[i][j];
        }
        rep(i,n) rep(j,n) tmp[i][j]=0x7f7f7f7f;
        rep(i,n) rep(j,n) rep(k,n) tmp[i][j]=min(tmp[i][j],f[i][k]+f[k][j]);
        rep(i,n) rep(j,n) f[i][j]=tmp[i][j];
        m>>=1;
    }
}
int main(){
    scanf("%d%d",&n,&m);
    for (int i=0;i<n;i++) scanf("%d",&a[i]);
    for (int i=0;i<n;i++) scanf("%d",&b[i]);
    n*=2;
    rep(i,n) rep(j,n) f[i][j]=0x7f7f7f7f;
    rep(i,n) f[i][i]=0;
    for (int i=0;i<n/2;i++){
        rep(j,n) rep(k,n) g[j][k]=f[j][k],f[j][k]=0x7f7f7f7f;
        rep(j,n) rep(k,n){
            if(k!=0) f[j][k]=min(f[j][k],g[j][k-1]+a[i]);
            if(k!=n) f[j][k]=min(f[j][k],g[j][k+1]+b[i]);
        }
    }
//  rep(i,n) rep(j,n) printf("%d->%d cost %I64d\n",i,j,f[i][j]);
    power(m);
    printf("%I64d\n",ans[0][0]);
//  
}
