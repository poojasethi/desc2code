#include<cstdio>
#include<cstring>

const int MAXN=20+3;

void up(unsigned &a, unsigned b) { if(a>b) a=b; }

struct matrix
{
	unsigned a[MAXN*2][MAXN*2];
}tr;

int n;
matrix operator * (const matrix &a, const matrix &b)
{
	matrix ans;
	memset(ans.a, 127, sizeof ans.a);
	int i,j,k;
	for(i=0;i<2*n;++i)
		for(j=0;j<2*n;++j)
			for(k=0;k<2*n;++k)
				up(ans.a[i][j], a.a[i][k]+b.a[k][j]);
	return ans;
}
matrix pow(matrix a, int b)
{
	matrix c;
	memset(c.a, 127, sizeof c.a);
	for(int i=0;i<2*n;++i)
		c.a[i][i]=0;
	for(;b;b>>=1)
	{
		if(b&1)
			c=c*a;
		a=a*a;
	}
	return c;
}

unsigned a[MAXN], b[MAXN], f[MAXN][MAXN*2];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
#endif
	int m,i,j,k;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;++i)
		scanf("%u",a+i);
	for(i=0;i<n;++i)
		scanf("%u",b+i);
	for(k=0;k<2*n;++k)
	{
		memset(f, 127, sizeof f);
		f[0][k]=0;
		for(i=0;i<n;++i)
			for(j=0;j<2*n;++j)
			{
				up(f[i+1][j+1], f[i][j]+a[i]);
				if(j)
					up(f[i+1][j-1], f[i][j]+b[i]);
			}
		memcpy(tr.a[k], f[n], sizeof f[n]);
	}
	printf("%u\n",pow(tr,m).a[0][0]);
	return 0;
}
