#include <iostream>
using namespace std;
#define oo 0x3F3F3F3F
#define FOR(i, a, b) for (int _end_ = (b), i = (a); i <= _end_; ++i)
inline void chkmin(int &a,int b) {if (a>b)a=b;return;}
const int N = 22;
int g[N * 100][N * 100],n,m,f[N * 2][N * 2],a[N],b[N];
int main() {
	cin >> n >> m;
	FOR (i, 0, n - 1) cin >> a[i];
	FOR (i, 0, n - 1) cin >> b[i];
	int LMT = min(m, 100) * n;
	fill(g[0], g[LMT + 1], oo), g[0][0] = 0;
	FOR (i, 0, LMT - 1) 
		FOR (j, 0, i) {
			chkmin(g[i + 1][j + 1], g[i][j] + a[i % n]);
			if (j) chkmin(g[i + 1][j - 1], g[i][j] + b[i % n]);
		}
	cout << g[LMT][0] + (m - min(m, 100)) / 2 * (g[LMT][0] - g[LMT - n * 2][0]) << endl;
	return 0; 
}
