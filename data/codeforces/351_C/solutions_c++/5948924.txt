#include<stdio.h>
#include<string.h>
int dp[415][415];
int a[415],b[415];
int min(int a,int b){
	return a<b?a:b;
}
int main (){
	int i,j,n,m,p,q,s,t,v,k;
	while(scanf("%d%d",&n,&m)!=EOF){
		for(i=1;i<=n;i++){
			scanf("%d",&a[i]);
			a[i+n*2]=a[i+n]=a[i];	
		}
		for(i=1;i<=n;i++){
			scanf("%d",&b[i]);
			b[i+n*2]=b[i+n]=b[i];	
		}
		dp[0][0]=0;
		dp[1][1]=a[1];
		for(i=2;i<=2*n;i++){
		//		dp[i][i/2+1]=dp[i-1][i/2]+b[i];
			if(i%2==0) dp[i][i/2]=dp[i-1][i/2]+b[i];
			for(j=i/2+1;j<=i;j++){
			//	if(i%2==1) continue;
				if(i!=j)
					dp[i][j]=min(dp[i-1][j]+b[i],dp[i-1][j-1]+a[i]);
				else dp[i][j]=dp[i-1][j-1]+a[i];
			}
		}
		v=s=dp[2*n][n];
		for(k=1;k<n;k++){
			dp[0][0]=0;
			dp[1][1]=a[k+1];
			for(i=2;i<=2*n;i++){
			//	{dp[i][i/2+1]=dp[i-1][i/2]+b[i+k];printf("%d %d %d\n",i,i/2+1,dp[i][i/2+1]);}
				if(i%2==0) {dp[i][i/2]=dp[i-1][i/2]+b[i+k];}
				for(j=i/2+1;j<=i;j++){
				//	if(i%2==1&&j==i/2+1) continue;
					if(i!=j)
						dp[i][j]=min(dp[i-1][j]+b[i+k],dp[i-1][j-1]+a[i+k]);
					else dp[i][j]=dp[i-1][j-1]+a[i+k];
				//	printf("%d %d %d\n",i,j,dp[i][j]);
				}
			}
		//	printf("\n\n");
			s=min(s,dp[2*n][n]);
		}
	//	printf("%d %d\n",s,v);
		printf("%d\n",v+s*(m/2-1));
	}
}
