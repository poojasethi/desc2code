#include <stdio.h>
#include <iostream>
#include <cstring>
using namespace std;
int p[50][50],d[50][50],a[50],b[50],f[50][50],c[50][50],n,m;
void back(int a[50][50],int b[50][50]){
	memset(c,120,sizeof(c));
	for (int i=0;i<=2*n;i++)
		for (int j=0;j<=2*n;j++)
			for (int k=0;k<=2*n;k++)
				if ((long long)c[i][k]>(long long)a[i][j]+b[j][k])
					c[i][k]=a[i][j]+b[j][k];
	memcpy(a,c,sizeof(c));
}
int main( ){
	cin>>n>>m;
	for (int i=1;i<=n;i++)
		cin>>a[i];
	for (int i=1;i<=n;i++)
		cin>>b[i];
	memset(d,120,sizeof(d));
	for (int i=0;i<=2*n;i++)
		d[i][i]=0;
	for (int i=1;i<=n;i++){
		memcpy(p,d,sizeof(d));
		memset(d,120,sizeof(d));
		for (int j=0;j<=2*n;j++)
			for (int k=0;k<=2*n;k++){
				if (k!=0 && d[j][k-1]>p[j][k]+b[i])
					d[j][k-1]=p[j][k]+b[i];
				if (k!=2*n && d[j][k+1]>p[j][k]+a[i])
					d[j][k+1]=p[j][k]+a[i];
			}
	}
	memset(f,120,sizeof(f));
	f[0][0]=0;
	while (m){
		if (m&1) back(f,d);
		back(d,d);
		m/=2;
	}
	cout<<f[0][0]<<endl;
	return 0;
}
