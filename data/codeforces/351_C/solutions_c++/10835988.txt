#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
const int N=105;
int n,m,a[N],b[N],f[N][N];
template<class T> inline bool ten(T &x,const T &y){return y<x?(x=y,1):0;}
void init()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	a[0]=a[n];
	for(int i=1;i<=n;i++)
		scanf("%d",&b[i]);
	b[0]=b[n];
	memset(f,0x7f,sizeof(f));
}

void work()
{
	f[0][0]=0;
	for(int i=0;i<4*n;i++)
		for(int j=0;j<n+n;j++)
		{
			if(j>0)ten(f[i+1][j-1],f[i][j]+b[(i+1)%n]);
			if(j<n+n)ten(f[i+1][j+1],f[i][j]+a[(i+1)%n]);
		}
	if(m<4)printf("%d\n",f[n*m][0]);
	else printf("%I64d\n",f[n*4][0]+(long long)(m-4)/2*(f[n*4][0]-f[n*2][0]));
}

int main()
{
	init();
	work();
	return 0;
}
