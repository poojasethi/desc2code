#include <cstring>
#include <iostream>
#include <algorithm>

using namespace std;

typedef long long LL;

const LL INF = 0x3f3f3f3f3f3f3f3fLL;

#define MS(a) memset(a,0x3f,sizeof(A))
#define FOR(i,l,r) for(int i=l;i<=r;i++)

int n,m;
LL a[25],b[25];
LL A[25][25],B[25][25],f[25][25];

void init(LL a[25][25]){MS(a);FOR(i,0,n) a[i][i]=0;}

void mul(LL a[][25],LL b[][25],LL c[][25]){
	static LL tmp[25][25];MS(tmp);
	FOR(i,0,n) FOR(j,0,n) FOR(k,0,n) tmp[i][j]=min(tmp[i][j],a[i][k]+b[k][j]);
	memcpy(c,tmp,sizeof(tmp));
}

int main()
{
	cin>>n>>m;
	FOR(i,1,n) cin>>a[i];
	FOR(i,1,n) cin>>b[i];
	init(A),init(f);
	FOR(k,1,n){
		init(B);
		FOR(i,0,n) FOR(j,0,n) i+1==j ? B[j][i]=a[k] : j+1==i ? B[j][i]=b[k] : B[j][i]=INF;
		mul(B,A,A);
	}
	for(;m;m>>=1,mul(A,A,A)) if(m&1) mul(f,A,f);
	cout << f[0][0] << endl;
	return 0;
}
