#include <iostream>
#include <vector>
using namespace std;

typedef long long ll;
typedef vector<ll> VI;
typedef vector<VI> VVI;
typedef vector<VVI> VVVI;

VI a, b;
VVVI d;
int n, m;

ll Bt(int p, int k, int obj) {
  ll& r = d[p][k][obj];
  if (r != -1) return r;
  if (p == n) {
    if (k == obj) return r = 0;
    return r = 1e11;
  }
  r = Bt(p + 1, k + 1, obj) + a[p];
  if (k) r = min(r, Bt(p + 1, k - 1, obj) + b[p]);
  return r;
}

int main() {
  cin >> n >> m;
  a = b = VI(n);
  for (int i = 0; i < n; ++i) cin >> a[i];
  for (int i = 0; i < n; ++i) cin >> b[i];
  d = VVVI(n + 1, VVI(3 * n + 1, VI(2 * n + 1, -1)));
  VVI v(2 * n + 1, VI(2 * n + 1));
  for (int i = 0; i <= 2 * n; ++i) {
    for (int j = 0; j <= 2 * n; ++j) v[i][j] = Bt(0, i, j);
  }
  
  ll res = 1e11;
  for (int i = 0; i <= n; ++i) {
    for (int j = 0; j <= 2 * n; ++j) {
      res = min(res, v[0][i] + v[i][0] + ((m - 2) / 2) * (v[i][j] + v[j][i]));
    }
  }
  
  cout << res << endl;
}
