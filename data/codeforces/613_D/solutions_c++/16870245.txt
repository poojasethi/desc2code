int i,j,k,n,m,x,y,z,ans[N],a;
vector<int> arr[N],q2[N];
set<int> st[N],q[N];
map<int,int> H[N];
map<int,bool> H2[N];

int f(int x,int root){
	set< int >::iterator it,it2;
	set< int > st2;
	for(int i=0 ; i<arr[x].size() ; i++){
		if(arr[x][i] != root){
			int y = arr[x][i];
			f(y,x);

			if(st[y].size() > st[x].size())
				swap(st[y],st[x]);

			for(it=st[y].begin() ; it!=st[y].end() ; it++){
				if(q[x].find(*it) != q[x].end())
					ans[*it]++;
				else if(st[x].find(*it) != st[x].end())
					st2.insert(*it);
				else
					st[x].insert(*it);
			}
		}
	}
	
	for(it=st2.begin() ; it!=st2.end() ; it++){
		ans[*it]++;
		st[x].erase(*it);
	}

	for(it=q[x].begin() ; it!=q[x].end() ; it++){
		if(st[x].find(*it) != st[x].end())
			ans[*it]++;
		st[x].insert(*it);
	}
}

int main(){
	cin >> n;

	for(i=1 ; i<n ; i++){
		scanf("%d %d",&x,&y);
		arr[x].pb(y);
		arr[y].pb(x);
	}

	cin >> m;

	for(i=1 ; i<=m ; i++){
		scanf("%d",&a);

		for(j=1 ; j<=a ; j++){
			scanf("%d",&x);
			q[x].insert(i);
			q2[i].pb(x);
		}
	}

	f(1,0);

	for(i=1 ; i<=m ; i++){
		for(j=0 ; ans[i]>=0 and j<q2[i].size() ; j++){
			int x = q2[i][j];
			for(k=0 ; k<arr[x].size() ; k++){
				if(q[arr[x][k]].find(i) != q[arr[x][k]].end()){
					ans[i]=-1;
					break;
				}
			}
		}
		printf("%d\n",ans[i]);
	}
}
