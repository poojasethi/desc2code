#include<iostream>
#include<stdio.h>
#include<string.h>
#define LL long long
using namespace std;
LL MOD = 1000000007;
LL d[1111][2222][2];
int used[1111][2222][2];
int k;
LL calc(int len, int last, int was)
{
	if (len == 0) return was;
	if (used[len][last][was]) return d[len][last][was];
	used[len][last][was] = 1;
	LL &res = d[len][last][was];
	for(int i = 0; i < 10; i++)
		if (i == 4 || i == 7) res = (res + calc(len - 1,1,was | (last <= k))) % MOD; else
		res = (res + calc(len - 1,last + 1,was)) % MOD;
	return res;
}
LL solve(char *a)
{
	int n = strlen(a), d = k + 1, was = 0;
	LL res = 0;
	for(int i = n - 1; i >= 0; i--)
	{
		for(int j = 0; j < a[n - 1 - i] - '0'; j++)
			if (j == 4 || j == 7) res = (res + calc(i, 1, was | (d <= k))) % MOD; else
			res = (res + calc(i, d + 1, was)) % MOD;
		if (a[n - 1 - i] == '4' || a[n - 1 - i] == '7')
		{
			if (d <= k) was = 1;
			d = 0;
		}
		d++;
	}
	res = (res + was) % MOD;
	return res;
}
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	#endif
	char l[1111],r[1111];
	int t,n,last,lucky;
	scanf("%d%d",&t,&k);
	for(int tt = 0; tt < t; tt++)
	{
		scanf("%s%s",&l,&r);
		n = strlen(l);
		lucky = 0;
		last = -k - 1;
		for(int i = 0; i < n; i++)
			if (l[i] == '4' || l[i] == '7')
			{
				if (i - last <= k) lucky = 1;
				last = i;
			}
		//cout<<solve(r)<<" "<<solve(l) - lucky<<endl;
		printf("%I64d\n",(solve(r) - solve(l) + lucky + MOD) % MOD);
	}
	return 0;
}
