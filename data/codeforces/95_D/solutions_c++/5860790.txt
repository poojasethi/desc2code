#include<cstdio>
#include<cstring>
using namespace std;
#define LL long long
#define N 1010
#define MOD 1000000007
char l[N],r[N];
LL f[N][N*2],a[N];
int t,k,len;
LL dfs(int pos,int dist,bool t){
    if(pos==-1)return dist==0;
    if(!t&&f[pos][dist]!=-1)return f[pos][dist];
    LL ans=0,end=t?a[pos]:9;
    for(int i=0;i<=end;i++){
        int nd=dist==0?0:dist+1;
        if(i==4||i==7)nd=dist<=k?0:1;
        ans=(ans+dfs(pos-1,nd,t&&i==end))%MOD;
    }
    if(!t)f[pos][dist]=ans;
    return ans;
}
int cal(char s[]){
    len=strlen(s);
    for(int i=0;i<len;i++)a[i]=s[len-1-i]-48;
    return dfs(len-1,k+2,1);
}
int main(){
    memset(f,-1,sizeof(f));
    scanf("%d%d",&t,&k);
    while(t--){
        scanf("%s%s",l,r);
        len=strlen(l);
        int j=len-1;l[j]--;
        while(l[j]<'0'){l[j]='9';l[--j]--;}
        if((len>1)&&(l[0]=='0'))for(int j=0;j<len;j++)l[j]=l[j+1];
        printf("%d\n",(cal(r)-cal(l)+MOD)%MOD);
    }
    return 0;
}
