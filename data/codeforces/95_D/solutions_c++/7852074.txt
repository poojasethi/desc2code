#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1010
#define mo 1000000007
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
const int a1[2]={8,2};
int i,j,k,n,m,K,x;
int f[N][N],h[N][N],w[N];
char p[N];
inline void M(int &x,int y) {
	x=(x+y)%mo;
}
inline void pre() {
	f[0][0]=1;
	For(i,0,N-5)For(j,0,K+1) if (x=f[i][j]) {
		For(k,0,9) {
			int A=k==4||k==7;
			if (j>K) M(f[i+1][K+1],x);
			else if (!j) {
				if (!A) M(f[i+1][0],x);
				else M(f[i+1][1],x);
			} else {
				if (A) M(f[i+1][K+1],x);
				else if (j<K) M(f[i+1][j+1],x);
				else M(f[i+1][0],x);
			}
		}
	}
	For(i,0,N-5) {
		h[i][0]=f[i][0];
		For(j,1,K+1) h[i][j]=(h[i][j-1]+f[i][j])%mo;
	}
	w[0]=1; For(i,1,N-5) w[i]=10ll*w[i-1]%mo;
}
inline int cal(int ty) {
	scanf("%s",p+1);
	int n=strlen(p+1),i,j,an=f[n-1][K+1],dis=K+1;
	for (i=n;i;i--) {
		int l=i==n?1:0,r=(i==1?p[n-i+1]-ty:p[n-i+1]-1)-'0';
		For(j,l,r) {
			int A=j==4||j==7;
			if (A) {
				if (dis==K+1) M(an,(h[i-1][K+1]-h[i-1][0])%mo);
				else M(an,w[i-1]);
			} else {
				if (!dis) M(an,w[i-1]);
				else {
					M(an,f[i-1][K+1]);
					if (dis<=K) M(an,(h[i-1][dis-1]-h[i-1][0])%mo);
				}
			}
		}
		int w=p[n-i+1]-'0';
		if (w==4||w==7) {
			if (dis!=K+1) dis=0;
			else dis=K;
		} else {
			if (dis&&dis<=K) {
				if (dis==1) dis=K+1;
				else dis--;
			}
		}
	}
	return an;
}
int main() {
	//freopen("95d.in","r",stdin);
	scanf("%d%d",&n,&K);
	pre();
	For(i,1,n) {
		int A=-cal(1);
		A=(A+cal(0))%mo;
		printf("%d\n",(A+mo)%mo);
	}
	return 0;
}
