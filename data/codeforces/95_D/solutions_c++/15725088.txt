#include<stdio.h>
#include<cstring>
#define N 1005
using namespace std;
int T,D,l,r,n,ans,i,P=1e9+7;
int a[N],f[N][N][2];char s[N];
inline void up(int &x){if (x>=P) x-=P;}
int calc(int k,int pre,int small,int ok){
  if (f[k][pre][ok]>=0&&small) return f[k][pre][ok];
  if (k==0) return ok;int ret=0;
  for (int c=small?9:a[n-k+1];c>=0;c--,ret%=P)
    if (c==4||c==7)
      ret+=calc(k-1,k,small||c<a[n-k+1],ok|(pre&&pre-k<=D));
    else
      ret+=calc(k-1,pre,small||c<a[n-k+1],ok);
  if (small) f[k][pre][ok]=ret;
  return ret;
}
int ok(){
  int k=0;
  for (int i=1;i<=n;i++)
    if (a[i]==4||a[i]==7){
      if (k&&i-k<=D) return 1;k=i;
    }return 0;
}
int main(){
  memset(f,-1,sizeof(f));
  scanf("%d%d",&T,&D);
  while (T--){
    scanf("%s",s+1);n=strlen(s+1);
    for (i=1;i<=n;i++) a[i]=s[i]-'0';
    ans=P-calc(n,0,0,0);ans+=ok();
    scanf("%s",s+1);n=strlen(s+1);
    for (i=1;i<=n;i++) a[i]=s[i]-'0';
    ans+=calc(n,0,0,0);ans%=P;
    printf("%d\n",ans);
  }
}
