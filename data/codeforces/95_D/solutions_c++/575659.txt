#include <stdio.h>
#include <string>
#include <iostream>
#include <algorithm>
using namespace std;
const int mol=1000000007;
const int maxn=1010;
string h,l,s;
int t,k;
long long dp[maxn][maxn];
inline int next(int p,int i)
{
    if(p==maxn-2)p=i;
    else if(p<maxn-2){
        if(p-i<=k)p=maxn-1;
        else p=i;
    }
    return p;
}
long long dfs(int i,int j)
{
    if(dp[i][j]!=-1)return dp[i][j];
    dp[i][j]=0;
    for(int c=0;c<=9;c++){      
        if(c==4||c==7)dp[i][j]+=dfs(i-1,next(j,i));
        else dp[i][j]+=dfs(i-1,j);
    }
    dp[i][j]%=mol;
    return dp[i][j];
}
int check(string s)
{
    int p=maxn-2;
    for(int i=s.length()-1;i>=0;i--)
    if(s[i]=='4'||s[i]=='7')p=next(p,i);
    return p==maxn-1;
}
long long gao(string s)
{
    long long ans=0;
    int p=maxn-2;
    reverse(s.begin(),s.end());
    for(int i=s.length()-1;i>=0;i--){
        for(int j=0;j<s[i]-'0';j++){
            if(j==4||j==7)ans+=dfs(i,next(p,i+1));
            else ans+=dfs(i,p);
        }
        if(s[i]=='4'||s[i]=='7')p=next(p,i+1);
    }
    return (ans+check(s))%mol;
}
int main()
{
    while(scanf("%d%d",&t,&k)!=EOF){
        for(int i=0;i<maxn;i++)
            for(int j=0;j<maxn;j++)
                dp[i][j]=-1;
        for(int i=0;i<maxn;i++)
            dp[0][i]=0;
        dp[0][maxn-1]=1;
        for(int i=0;i<t;i++){
            cin>>l>>h;
            cout<<(gao(h)-gao(l)+check(l)+mol)%mol<<endl;
        }
    }
}
