#include <cstdio>
inline int dist(int x,int y)
{
  return (x-200)*(x-200)+(y-200)*(y-200);
}
int a[500][500];
int x[500],y[500];
int main()
{
  int X,Y,d,n,i,j,k;
  scanf("%d%d%d%d",&X,&Y,&n,&d);
  for (i=1;i<=n;i++)
    scanf("%d%d",&x[i],&y[i]);
  for (i=400;i>=0;i--)
    for (j=400;j>=0;j--)
    {
      if (dist(i,j) > d*d)
      {
        a[i][j]=1;
        continue;
      }
      a[i][j]=0;
      for (k=1;k<=n;k++)
      {
        if (i+x[k]>400 || j+y[k] >400)
          continue;
        if (!a[i+x[k]][j+y[k]])
          a[i][j]=1;
      }
    }
  if (a[X+200][Y+200])
    printf("Anton\n");
  else
    printf("Dasha\n");
  return 0;

}
