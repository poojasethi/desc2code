#include <iostream>
#include <cstdio>
#include <string>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

int x,y,n,d;
int win[500][500];
int dir[25][2];

int dfs(int dx,int dy)
{
    if(win[dx+250][dy+250]!=-1) return win[dx+250][dy+250];
    for(int i=0;i<n;i++)
    {
        int nextx=dx+dir[i][0],nexty=dy+dir[i][1];
        if(nextx*nextx+nexty*nexty>d*d) continue;
        if(dfs(nextx,nexty)==0) return win[dx+250][dy+250]=1;
    }
    return win[dx+250][dy+250]=0;
}
int main()
{
    fill(win[0],win[0]+500*500,-1);
    cin>>x>>y>>n>>d;
    for(int i=0;i<n;i++) scanf("%d%d",dir[i],dir[i]+1);
    puts(dfs(x,y)?"Anton":"Dasha");
	return 0;
}
