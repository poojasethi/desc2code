#include <cstdio>

#define C 220
#define N 22

int x,y,d,n,i,j,k;
int vx[N],vy[N];
int f[2*C][2*C];

int calc(int x,int y) // 1 - lose pos, 2 - win pos
{
	if(x*x + y*y > d*d) return 1;
	if(f[x+C][y+C]) return f[x+C][y+C];
	int i;
	bool flag=1;
	for(i=0;i<n;i++)
		if(calc(x+vx[i],y+vy[i])==2)
		{
			flag = 0;
			break;
		}
	if(flag)
		f[x+C][y+C] = 2;else
		f[x+C][y+C] = 1;
	return f[x+C][y+C];
}

int main()
{
	scanf("%d%d%d%d",&x,&y,&n,&d);
	for(i=0;i<n;i++)
		scanf("%d%d",&vx[i],&vy[i]);
	if(calc(x,y)==1)
		printf("Anton\n");else
		printf("Dasha\n");
	return 0;
}
