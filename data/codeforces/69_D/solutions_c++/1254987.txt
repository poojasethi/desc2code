#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int x[25], y[25];
int sg[405][405];
inline int dist(int a, int b) {
	return a*a+b*b;
}
int main(void) {
	int sx, sy, n, d;
	scanf("%d%d%d%d", &sx, &sy, &n, &d);
	for(int i=0;i<n;i++)
		scanf("%d%d", &x[i], &y[i]);
	int OX=200, OY=200;
	for(int i=200;i>=sx;--i)
		for(int j=200;j>=sy;--j) {
			int v[25]={};
			for(int k=0;k<n;k++) {
				if(dist(i+x[k], j+y[k]) > d*d) v[1]=1;
				else v[sg[OX+i+x[k]][OY+j+y[k]]]=1;
			}
			for(sg[OX+i][OY+j]=0;v[sg[OX+i][OY+j]]==1;++sg[OX+i][OY+j]);
		}
	if(sg[OX+sx][OY+sy]==0) printf("Dasha\n");
	else puts("Anton");
	return 0;
}
