#include <cstdio>
using namespace std;
int f[500][500]={};
int d,n;
int a[100],b[100];
bool solve(int y,int x)
{
	if(x*x+y*y>d*d) return true;
	int &ret=f[y+250][x+250];
	if(ret==0)
	{
		//ret=1;
		//if(!solve(x,y)) ret=1;
		//else
		{
			ret=-1;
			for(int i=0;i<n;i++)
				if(!solve(y+b[i],x+a[i]))
				{
					ret=1;
					break;
				}
		}
	}
	return ret>0;
}
int main()
{
	int x,y;
	scanf("%d%d%d%d",&x,&y,&n,&d);
	for(int i=0;i<n;i++) scanf("%d%d",&a[i],&b[i]);
	puts(solve(y,x)?"Anton":"Dasha");
	return 0;
}
