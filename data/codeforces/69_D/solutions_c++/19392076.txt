#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int x0,y0,n,d;
int dp[405][405],x[405],y[405];
int check(int tx,int ty){
	if((tx-200)*(tx-200)+(ty-200)*(ty-200)<=d*d)
		return 1;
	return 0;
}
int dfs(int tx,int ty){
	if(dp[tx][ty]!=-1)
		return dp[tx][ty];
	for(int i=1; i<=n; i++){
		if(check(tx+x[i],ty+y[i])&&dfs(tx+x[i],ty+y[i])==0)
			return dp[tx][ty]=1;
	}
		return dp[tx][ty]=0;
}
int main(){
	cin>>x0>>y0>>n>>d;
	for(int i=1; i<=n; i++)
		cin>>x[i]>>y[i];
	memset(dp,-1,sizeof dp);
	if(dfs(x0+200,y0+200))
		cout<<"Anton"<<endl;
	else
		cout<<"Dasha"<<endl;
	return 0;
}
