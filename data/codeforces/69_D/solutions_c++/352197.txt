#include <cstdio>
#include <cstring>
using namespace std;
int d,n;
int dp[500][500];
int v[21][2];

int dfs(int x,int y)
{	
	if(x*x+y*y>d*d) return 1;
	if(dp[x+200][y+200]!=-1)	return dp[x+200][y+200];

	for(int i=0;i<n;i++)
		if(dfs(x+v[i][0],y+v[i][1])==0)
		{
			dp[x+200][y+200]=1;
			return 1;
		}
	dp[x+200][y+200]=0;
	return 0;
}
int main()
{
	int x,y;
	memset(dp,-1,sizeof(dp));
	scanf("%d%d%d%d",&x,&y,&n,&d);
	for(int i=0;i<n;i++)
		scanf("%d%d",&v[i][0],&v[i][1]);
	if(dfs(x,y))	printf("Anton\n");
	else printf("Dasha\n");
	return 0;
}
