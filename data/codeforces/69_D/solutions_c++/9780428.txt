#include "iostream"
#include "cstdio"
#include "cstring"
#include "algorithm"
using namespace std;
int vis[444][444];
int _x,_y,n,d;
int dx[22],dy[22];
int dfs(int x,int y)
{
    if((x-200)*(x-200)+(y-200)*(y-200)>d*d)
        return 1;
    if(vis[x][y])
        return vis[x][y];
    int i;
    for(i = 1; i<=n; ++i)
        if(dfs(x+dx[i],y+dy[i])==2)
            return vis[x][y] = 1;
    return vis[x][y] = 2;
}

int main()
{
    int i;
    scanf("%d%d%d%d",&_x,&_y,&n,&d);
    _x += 200;
    _y += 200;
    for(i = 1; i<=n; ++i)
        scanf("%d%d",&dx[i],&dy[i]);
    if(dfs(_x,_y)==1)
        printf("Anton\n");
    else
        printf("Dasha\n");
}
