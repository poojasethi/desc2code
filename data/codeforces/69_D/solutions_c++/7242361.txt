#include <iostream>
#include <cstring>

using namespace std;

const int MAXD = 500, SHIFT = 250, MAXN = 51;
int dx[MAXN], dy[MAXN], mxDist, dp[MAXD][MAXD], n;

bool okDist(int x, int y) {
	return x * x + y * y <= mxDist * mxDist;
}

int bt(int x, int y) {
	int& ret = dp[x + SHIFT][y + SHIFT];
	if (ret != -1)
		return ret;
	ret = 0;
	for (int i = 0; i < n; i++) {
		int nx = x + dx[i], ny = y + dy[i];
		if (okDist(nx, ny) && !bt(nx, ny))
				ret = 1;
	}
	return ret;
}

int main() {
	int x, y;
	cin >> x >> y >> n >> mxDist;
	for (int i = 0; i < n; i++)
		cin >> dx[i] >> dy[i];
	memset(dp, -1, sizeof dp);
	cout << (bt(x, y) ? "Anton" : "Dasha") << endl;
	return 0;
}
