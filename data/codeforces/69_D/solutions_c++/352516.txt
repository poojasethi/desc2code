#include <cstdio>
#define sqr(a) a*a
using namespace std;

int a[1000][1000];
int x[30], y[30];
int mx, my;
int n, d;

bool funct(int q, int r)
{
    if(sqr(q)+sqr(r)>sqr(d))return 0;
    if(a[q+300][r+300]!=-1)return a[q+300][r+300];
    a[q+300][r+300]=1;
    for(int i=0; i<n; i++)
    {
        if(funct(q+x[i], r+y[i]))return a[q+300][r+300]=0;
    }
    return 1;
}

int main()
{
    scanf("%d%d", &mx, &my);
    scanf("%d%d", &n, &d);
    for(int i=0; i<1000; i++)
    {
        for(int j=0; j<1000; j++)a[i][j]=-1;
    }
    for(int i=0; i<n; i++)scanf("%d%d", &x[i], &y[i]);
    if(funct(mx, my))printf("Dasha");
    else printf("Anton");
}
