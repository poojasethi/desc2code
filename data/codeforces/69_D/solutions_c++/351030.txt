#include<iostream>
#include<cstdio>
#include<map>
using namespace std;
map<pair<int,int>,int> Winner;
int Vec[25][2],d,n;
int dist(int a,int b){
    return a*a+b*b;
}
int solve(int x,int y){
    pair<int,int> p=pair<int,int>(x,y);
    if(Winner.find(p)!=Winner.end())return Winner[p];

    for(int i=0;i<n;i++){
        if(dist(x+Vec[i][0],y+Vec[i][1])>d*d){
            continue;
        }
        if(solve(x+Vec[i][0],y+Vec[i][1])==0)return Winner[p]=1;
    }
    return Winner[p]=0;
}

int main(){
    int x,y;
    scanf("%d%d%d%d",&x,&y,&n,&d);
    for(int i=0;i<n;i++){
        scanf("%d%d",&Vec[i][0],&Vec[i][1]);
    }
    printf("%s\n",(solve(x,y)?"Anton":"Dasha"));
    return 0;
}
