#include <cstdio>

int vx[22], vy[22], n, d;
int dp[405][405];

int dfs(int x, int y) {
    if(x*x + y*y > d*d) return 1;
    if(dp[x+200][y+200]) return dp[x+200][y+200];
    for(int i = 0; i < n; ++i) {
        if(dfs(x+vx[i], y+vy[i]) == -1) return dp[x+200][y+200] = 1;
    }
    return dp[x+200][y+200] = -1;
}

int main() {
    int x, y; scanf("%d%d%d%d", &x, &y, &n, &d);
    for(int i = 0; i < n; ++i) scanf("%d%d", &vx[i], &vy[i]);
    if(dfs(x, y) == 1) puts("Anton");
    else puts("Dasha");
}
