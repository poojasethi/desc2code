#include <cstdio>
#include <cstring>
#include <algorithm>
#define RETURN(x) {depth--;return x;}
using namespace std;

int n,d,f[401][401],dx[20],dy[20],depth;

int gao(int x, int y){
    if(x*x+y*y>d*d) return 1;
    if(f[x+200][y+200]>=0) return f[x+200][y+200];
    for(int i=0;i<n;i++) if(!gao(x+dx[i],y+dy[i])) return f[x+200][y+200]=1;
    return f[x+200][y+200]=0;
}

int main(){
    int x,y;
    memset(f,255,sizeof(f));
    scanf("%d%d%d%d",&x,&y,&n,&d);
    for(int i=0;i<n;i++) scanf("%d%d",&dx[i],&dy[i]);
    puts(gao(x,y)?"Anton":"Dasha");
}
