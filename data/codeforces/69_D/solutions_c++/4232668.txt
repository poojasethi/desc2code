#include<stdio.h>
int dp[500][500];
int dis(int x,int y)
{
    return x*x+y*y;
}
int main()
{
    int x0,y0,n,d,i,j,k,x[24],y[24];
    bool sg[24];
    scanf("%d%d%d%d",&x0,&y0,&n,&d);
    for(i=1;i<=n;i++)
       scanf("%d%d",&x[i],&y[i]);
    for(i=d-x0;i>=0;i--)
       for(j=d-y0;j>=0;j--)
       {
          if(dis(i+x0,j+y0)>d*d)continue;
          for(k=0;k<=n;k++)sg[k]=false;
          for(k=1;k<=n;k++)
          {
             if(dis(i+x[k]+x0,j+y[k]+y0)>d*d)continue;
             sg[dp[i+x[k]][j+y[k]]]=true;
          }
          for(k=0;k<=n;k++)
             if(!sg[k])break;
          dp[i][j]=k;
       }
    printf("%s\n",dp[0][0]==0?"Dasha":"Anton");
    return 0;
}
