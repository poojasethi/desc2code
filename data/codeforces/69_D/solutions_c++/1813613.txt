#include <iostream>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;

int n,d,b[500][500],vx[20],vy[20],x,y;

int f(int x,int y){

  int xx=x+250,yy=y+250;
  if(x*x+y*y>=d*d)return 1;
  if(b[xx][yy]>=0)return b[xx][yy];

  rep(i,n)if(!f(x+vx[i],y+vy[i]))return b[xx][yy]=1;
  return b[xx][yy]=0;

}


int main(){

  cin>>x>>y>>n>>d;
  rep(i,n)cin>>vx[i]>>vy[i];
  rep(i,500)rep(j,500)b[i][j]=-1;

  cout<<(f(x,y)?"Anton":"Dasha")<<endl;

}
