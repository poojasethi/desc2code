#include<iostream>

using namespace std;

int dx[30],dy[30];
int field[500][500]={0,};
int n,d;
const int add=200;


int rec(int x,int y){
    if(field[x][y]!=0) return field[x][y];

    for(int i=0;i<n;i++){
        if((x+dx[i]-add)*(x+dx[i]-add) + (y+dy[i]-add)*(y+dy[i]-add)<= d*d ){
            if(rec(x+dx[i],y+dy[i])!=1){
                field[x][y]=1;
                return 1;
            }
        }
    }
    field[x][y]=-1;
    return -1;



}

int main(){
    int x,y;
    cin>>x>>y>>n>>d;
    for(int i=0;i<n;i++) cin>>dx[i]>>dy[i];

    if(rec(x+add,y+add)==1) cout<<"Anton\n";
    else cout<<"Dasha\n";

    return 0;
}
