#include <cstdio>
#include <iostream>
#include <algorithm>
#include <map>
using namespace std;
int sx,sy,d,n;
int mx[20];
int my[20];
map < pair<int,int>,int > F;

int Gao(int x,int y){
    if (x*x+y*y>d) return 1;
    if (F.count(make_pair(x,y))) return F[make_pair(x,y)];
    F[make_pair(x,y)]=0;
    for (int i=0;i<n;i++)
        if (!Gao(x+mx[i],y+my[i])){
           F[make_pair(x,y)]=1;
           return 1;
        }
    return 0;
}

int main(){
    scanf("%d%d%d%d",&sx,&sy,&n,&d);
    d*=d;
    for (int i=0;i<n;i++)
        scanf("%d%d",&mx[i],&my[i]);
    F.clear();
    int ret=Gao(sx,sy);
    if (ret) puts("Anton");
        else puts("Dasha");
}
