#include <cstdio>
#include <cstring>

int n,d2,dx[20],dy[20],w[401][401];

int win(int x0,int y0) {
    if (w[x0+200][y0+200]==-1) {
        for (int i=0; i<n; i++) {
            int x=x0+dx[i],y=y0+dy[i];
            if ((x*x+y*y)<=d2)
                if (!win(x,y)) {
                    w[x0+200][y0+200]=1;
                    return 1;
                }
        }
        w[x0+200][y0+200]=0;
        return 0;
    }
    return w[x0+200][y0+200];
}

int main() {
    int x,y,d;
    scanf("%d %d %d %d",&x,&y,&n,&d);
    d2=d*d;
    for (int i=0; i<n; i++)
        scanf("%d %d",&dx[i],&dy[i]);
    memset(w,-1,sizeof(w));
    printf(win(x,y)?"Anton\n":"Dasha\n");
    return 0;
}
