#include <cstdio>
#include <cstring>

int v[30][2];
int m[500][500];
#define delta 250

    int x, y, n, d;
bool rec(int i, int j) {
    if (i*i+j*j > d*d) return true;
    if (m[i+delta][j+delta] >= 0) return m[i+delta][j+delta];

    for (int a = 0; a < n; a++)
        if (!rec(i+v[a][0], j+v[a][1]))
            return m[i+delta][j+delta] = 1;
    return m[i+delta][j+delta] = 0;
}

int main() {
    scanf("%d%d%d%d", &x, &y, &n, &d);
    for (int i = 0; i < d; i++)
        scanf("%d%d", &v[i][0], &v[i][1]);

    for (int i = 0; i < 500; i++)
        for (int j = 0; j < 500; j++)
            m[i][j] = -1;

    if (rec(x,y)) printf("Anton\n");
    else printf("Dasha\n");
    return 0;
}
