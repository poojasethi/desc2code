#include <iostream>
#include <cstring>
using namespace std;
#define D 500
int x,y,n,d,dp[D][D],v[22][2];
int f(int x,int y){
   if(x*x+y*y>d*d)return 1;
   int a=x+D/2, b=y+D/2;
   if(dp[a][b]!=-1)return dp[a][b];
   for(int i=0; i<n; i++)
      if(!f(x+v[i][0],y+v[i][1]))
         return dp[a][b]=1;
   return dp[a][b]=0;
}
int main(){
   cin>>x>>y>>n>>d;
   for(int i=0; i<n; i++)cin>>v[i][0]>>v[i][1];
   memset(dp,-1,sizeof(dp));
   cout<< ((f(x,y))?"Anton":"Dasha") <<endl;
}
