#include<cstdio>
#include<iostream>
#include<cstring>
#define maxn 1000
using namespace std;
int x[maxn],y[maxn],d,n,xx,yy,dp[500][500];
int dfs(int a,int b)
{
	if(dp[a+200][b+200]!=-1)return dp[a+200][b+200];
	if(a*a+b*b>d*d)return 1;
	for(int i=1;i<=n;i++)
	{
		int xx=a+x[i],yy=b+y[i];
		int u=dfs(xx,yy);
		if(u==0)return dp[a+200][b+200]=1;
	}
	return dp[a+200][b+200]=0;
}
int main()
{
	scanf("%d%d%d%d",&xx,&yy,&n,&d);
	for(int i=1;i<=n;i++)scanf("%d%d",&x[i],&y[i]);
	memset(dp,-1,sizeof(dp));
	if(dfs(xx,yy))puts("Anton");else puts("Dasha");
	return 0;
}
