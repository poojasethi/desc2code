#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 200+5;
int dp[N<<1][N<<1], n, dx[N], dy[N], d;

int dfs(int x, int y) {
    if((x-200)*(x-200) + (y-200)*(y-200) > d*d)   return 1;
    if(dp[x][y] != -1)  return dp[x][y];
    dp[x][y] = 0;
    for(int i = 0;i < n; i++) if(!dfs(x+dx[i], y+dy[i]))
        return dp[x][y] = 1;
    return dp[x][y];
}

int main() {
    int x, y;
    scanf("%d%d%d%d", &x, &y, &n, &d);
    x += 200; y += 200;
    for(int i = 0;i < n; i++) {
        scanf("%d%d", &dx[i], &dy[i]);
    }
    memset(dp, -1, sizeof(dp));
    puts(dfs(x, y) ? "Anton" : "Dasha");
    return 0;
}
