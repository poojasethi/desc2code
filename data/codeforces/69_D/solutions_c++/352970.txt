#include <cstdio>
#include <cstring>
int f[500][500], x[22], y[22];
int sx, sy, n, d;
int gao(int xx, int yy) {
     if (xx*xx+yy*yy > d*d) return 1;
     if (f[xx+222][yy+222] >= 0) return f[xx+222][yy+222];
     for(int j = 0; j < n; ++j) if (gao(xx+x[j], yy+y[j]) == 0) return f[xx+222][yy+222] = 1;
     return f[xx+222][yy+222] = 0;
     
}
int main() {
    scanf("%d%d%d%d", &sx, &sy, &n, &d);
    for (int j = 0 ; j < n; ++j) scanf("%d%d", &x[j], &y[j]);
    memset(f, -1, sizeof(f));  
    int flag = gao(sx, sy);
    if (flag) puts("Anton"); else puts("Dasha");
    return 0;
}
