#include <algorithm>
#include <iostream>
#include <map>
using namespace std;

#define REP(i,a,b) for(int i=(a);i<(b);++i)
#define mp make_pair
#define sqr(a) ((a)*(a))
int X[20],Y[20],n,D,x,y;
map<pair<int,pair<int,int> >,int > m;
int brute(int t, int x, int y) {
    if (m.count(mp(t,mp(x,y))))return m[mp(t,mp(x,y))];
    REP(i, 0, n) {
        if (sqr(x+X[i])+sqr(y+Y[i]) > D * D) continue;
        if (brute(t ^ 1, x+X[i],y+Y[i])==0) return m[mp(t,mp(x,y))]=1;
    }
    return m[mp(t,mp(x,y))]=0;
}
int main() {
    cin>>x>>y>>n>>D;
    REP(i, 0, n) cin >> X[i] >> Y[i];
    cout << (brute(0, x, y) == 1 ? "Anton" : "Dasha");    
}
