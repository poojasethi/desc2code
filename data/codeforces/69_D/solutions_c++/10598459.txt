#include<cstdio>
#include<iostream>
#define maxn 1000
using namespace std;
int x[maxn],y[maxn],d,n,xx,yy,dp[500][500];
int dfs(int a,int b)
{
	if(a*a+b*b>d*d)return 1;
        int &u=dp[a+201][b+201];
	if(u!=0)return u;
	for(int i=1;i<=n;i++)if(dfs(a+x[i],b+y[i])==2)return u=1;
	return u=2;
}
int main()
{
	scanf("%d%d%d%d",&xx,&yy,&n,&d);
	for(int i=1;i<=n;i++)scanf("%d%d",&x[i],&y[i]);
	if(dfs(xx,yy)==1)puts("Anton");else puts("Dasha");
	return 0;
}
