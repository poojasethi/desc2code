#include<cstdio>

int f[405][405][2][2];
int x[25],y[25];
int n,d,sx,sy;

int dfs(int sx,int sy,int a,int b)
{
    if (sx*sx+sy*sy>d*d) return 1;
    if (f[sx+d][sy+d][a][b]) return f[sx+d][sy+d][a][b];
    for (int i=0;i<n;i++)
    if (dfs(sx+x[i],sy+y[i],b,a)==2)
        return f[sx+d][sy+d][a][b]=1;
    if (!a && dfs(sy,sx,b,1)==2)
        return f[sx+d][sy+d][a][b]=1;
    return f[sx+d][sy+d][a][b]=2;
}

int main()
{
    scanf("%d%d%d%d",&sx,&sy,&n,&d);
    for (int i=0;i<n;i++)
        scanf("%d%d",&x[i],&y[i]);
    if (dfs(sx,sy,0,0)==1) printf("Anton\n");
    else printf("Dasha\n");
}
