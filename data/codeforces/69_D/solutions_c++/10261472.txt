#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int x0,y0,dp[405][405],n,d;
int x[405],y[405];
int check(int tx,int ty)
{
  if ((tx-200)*(tx-200)+(ty-200)*(ty-200)<=d*d) return 1;
  return 0; 
}
int dfs(int tx,int ty)
{
  if (dp[tx][ty]!=-1) return dp[tx][ty];
  for (int i=1;i<=n;i++)
	if (check(tx+x[i],ty+y[i])&&dfs(tx+x[i],ty+y[i])==0) 
	  return dp[tx][ty]=1;
  return dp[tx][ty]=0;
}
int main()
{
  scanf("%d%d%d%d",&x0,&y0,&n,&d);
  for (int i=1;i<=n;i++)
	scanf("%d%d",&x[i],&y[i]);
  memset(dp,-1,sizeof(dp));
  if (dfs(x0+200,y0+200)) printf("Anton\n");
  else printf("Dasha\n");
}
