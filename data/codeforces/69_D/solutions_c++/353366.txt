#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>

using namespace std;

const int maxn=20,cc=200;
int f[maxn][2],n,d,u[401][401];

bool solve(int a,int b)
{
    bool p=false;
    int x,y;
    for (int i=0; i<n; i++){
        x=a+f[i][0];y=b+f[i][1];
        if ((x*x+y*y)<=d){
            if (u[x+cc][y+cc]==0) solve(x,y);
            if (u[x+cc][y+cc]<0){p=true; break;}
        }
    }
    if (p) u[a+cc][b+cc]=1; else u[a+cc][b+cc]=-1;
    return p;
}

int main()
{
    int x,y;
    cin>>x>>y>>n>>d;
    d*=d;
    for (int i=0; i<n; i++) scanf("%d%d",&f[i][0],&f[i][1]);
    if (solve(x,y)) printf("Anton"); else printf("Dasha");
}
