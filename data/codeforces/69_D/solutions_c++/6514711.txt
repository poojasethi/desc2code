#include <cstdio>
#include <cstring>

const int MAX_N = 21;
const int MAX_S = 1001; 

int x, y, n, d;
int vx[MAX_N], vy[MAX_N];
int dp[MAX_S][MAX_S];

int dfs(int x, int y) {
	int nx = x + 500;
	int ny = y + 500;
	if (~dp[nx][ny]) return dp[nx][ny];
	if (x * x + y * y >= d * d) return 1;
	for (int i = 1; i <= n; ++i) {
		if (!dfs(x + vx[i], y + vy[i])) return dp[nx][ny] = 1;
	}
	return dp[nx][ny] = 0;
}

int main() {
	scanf("%d%d%d%d", &x, &y, &n, &d);
	memset(dp, -1, sizeof dp);
	for (int i = 1; i <= n; ++i) {
		scanf("%d%d", &vx[i], &vy[i]);
	}
	if (dfs(x, y)) puts("Anton");
	else puts("Dasha");
	return 0;
}
