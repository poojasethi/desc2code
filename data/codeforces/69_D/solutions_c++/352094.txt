#include <iostream>
#include <cstdio>

#define N 404

using namespace std;
int x0, y0, n, d;
int p[20][2];
int w[N][N];

int calc(int x, int y) {
  if (x*x+y*y > d)
    return 1;
  int &win = w[x-x0][y-y0];
  if (win > -1) return win;
  for (int i = 0; i < n; i++)
    if (!calc(x+p[i][0], y+p[i][1]))
      return win = 1;
  return win = 0;
}

int main() {
  cin >> x0 >> y0 >> n >> d;
  d *= d;
  for (int i = 0; i < 401; i++)
  for (int j = 0; j < 401; j++)
    w[i][j] = -1;
  for (int i = 0; i < n; i++)
    cin >> p[i][0] >> p[i][1];
  if (calc(x0, y0))
    cout << "Anton";
  else
    cout << "Dasha";
  return 0;
}
