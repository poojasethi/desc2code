#include <cstdio>
#include <map>
using namespace std;

int X, Y, N, d;
int dx[30], dy[30];
map<int, map<int, bool> > visited;
map<int, map<int, bool> > win;

bool solve(int x, int y){
    if(x*x + y*y > d*d){
        return true;
    }
    if(visited[x][y]){
        return win[x][y];
    }
    visited[x][y] = true;
    for(int i=0; i<N; i++){
        int ny = y + dy[i], nx = x + dx[i];
        if(!solve(nx,ny)){
            return win[x][y] = true;
        }
    }
    return win[x][y] = false;
}

int main(){
    scanf("%d%d%d%d",&X,&Y,&N,&d);
    for(int i=0; i<N; i++){
        scanf("%d%d",dx+i,dy+i);
    }
    puts(solve(X,Y)?"Anton":"Dasha");
    return 0;
}
