#include <stdio.h>

#define N 202

int x, y, n, d, vx[20], vy[20], mem[N*2][N*2];

int rec( int a, int b )
{
	if( a*a + b*b > d*d ) return 1;
	int & ans = mem[a+N][b+N];
	if( ans > -1 ) return ans;
	int i;
	for( i = 0; i < n; i++ ) if( !rec( a+vx[i], b+vy[i] ) ) return ans = 1;
	return ans = 0;
}

int main()
{
	int i, j;
	for( i = 0; i < N*2; i++ ) for( j = 0; j < N*2; j++ ) mem[i][j] = -1;
	scanf( "%d%d%d%d", &x, &y, &n, &d );
	for( i = 0; i < n; i++ ) scanf( "%d%d", &vx[i], &vy[i] );
	printf( rec(x,y) ? "Anton\n" : "Dasha\n" );
	return 0;
}
