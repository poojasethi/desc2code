#include <cstdio>
#include <map>

using namespace std;

map<pair<int,int>,bool> f;
int n,d,dx[21],dy[21];

bool dfs(int x,int y)
{
  pair<int,int> p=make_pair(x,y);
  if(f.count(p)>0)
    return f[p];
  if(x*x+y*y>d*d)
  {
    f[p]=true;
    return true;
  }
  for(int i=1;i<=n;++i)
  {
    if(!dfs(x+dx[i],y+dy[i]))
    {
      f[p]=true;
      return true;
    }
  }
  f[p]=false;
  return false;
}

int main(void)
{
  int x0,y0;
  scanf("%d %d %d %d",&x0,&y0,&n,&d);
  for(int i=1;i<=n;++i)
    scanf("%d %d",&dx[i],&dy[i]);
  if(dfs(x0,y0))
    printf("Anton\n");
  else
    printf("Dasha\n");
  return 0;
}
