#include<cstdio>
#include<cstring>
#define f(x,y) dp[x+200][y+200]
int dp[405][405],vx[25],vy[25],x,y,d,n;
int saya(int x,int y)
{
    if (x*x+y*y>d*d) return 1;
    int &ret=f(x,y);
    if (ret!=-1) return ret;
    for(int i=0;i<n;i++)
        if (!saya(x+vx[i],y+vy[i])) return ret=1;
    return ret=0;
}
int main()
{
    scanf("%d%d%d%d",&x,&y,&n,&d);
    for(int i=0;i<n;i++) scanf("%d%d",&vx[i],&vy[i]);
    memset(dp,-1,sizeof(dp));
    if (saya(x,y)) puts("Anton");
    else puts("Dasha");
}
