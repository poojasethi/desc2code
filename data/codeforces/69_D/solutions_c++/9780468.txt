#include<bits/stdc++.h>
using namespace std;
int n,d;
struct Vector
{
    int x,y;
}V[25];
int f[500][500];
bool dfs(int x,int y)
{
    if(~f[x][y]) return f[x][y];
	if((x-200)*(x-200)+(y-200)*(y-200)>=d*d) return 1;
	for (int i=0;i<n;++i)
        if(!dfs(x+V[i].x,y+V[i].y)) return f[x][y]=1;
	return f[x][y]=0;
}
int main()
{
	int x, y;
	cin>>x>>y>>n>>d;
	memset(f,-1,sizeof(f));
	for (int i=0;i<n;++i) scanf("%d%d",&V[i].x,&V[i].y);
	if (dfs(x+200,y+200)) puts("Anton");
	else puts("Dasha\n");
	return 0;
}
