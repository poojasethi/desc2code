#include <stdio.h>
#define F(I,B,E) for(int I=B; I<E; I++)
main() {
  int a[100100],n,t,m;
  long long v=0,p=0;
  scanf("%d",&n);
  n++;
  F(i,1,n)
    scanf("%d",&t),
    a[t]=i;
  scanf("%d",&m);
  F(i,0,m)
    scanf("%d",&t),
    v+=a[t],
    p+=n-a[t];
  printf("%I64d %I64d",v,p);
  return 0;
}
