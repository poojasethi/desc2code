#include<iostream>
#include<cstdlib>
using namespace std;
int main()
{int *a,n,m,i,x;
 static long long u,v;
 cin>>n;
 a=(int*)calloc((n+1),sizeof(int));
 for(i=1;i<=n;i++)
   {cin>>x;a[x]=i;}
 cin>>m;
  for(i=0;i<m;i++)
    {cin>>x;
      u+=a[x];v+=(n-a[x]+1);
    }
 cout<<u<<" "<<v;
 return 0;
}
