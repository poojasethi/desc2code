#include<iostream>
using namespace std;
long long n,m;
long long j=0,k,a[100005];
int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>k;
        a[k]=i;
    }
    cin>>m;
    for(int i=1;i<=m;i++){
        cin>>k;
        j+=a[k];
    }
    cout<<j<<" "<<(n+1)*m-j<<endl;

    return 0;
}
