#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	int val,b[100005];
	int n,i,j;
	cin >> n;
	for(i=0;i<n;i++)
	{
		cin >> val;
		b[val]=i+1;
	}
	int m;
	long long int a1=0,a2=0;
	cin >> m;
	for(i=0;i<m;i++)
	{
		cin >> val;
		a1+=b[val];
		a2+=(n-b[val]+1);
	}
	cout << a1 << ' ' << a2 << endl;
	return 0;
}
