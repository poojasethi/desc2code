#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
long long int  n,sum1=0,sum2=0,k,m; cin>>n;
pair<int,int>p[n];
for(int i=0;i<n;i++)
{cin>>k; p[k].first=i+1; p[k].second=n-i;}
cin>>m;
for(int i=0;i<m;i++)
{cin>>k; sum1+=p[k].first; sum2+=p[k].second;}
cout<<sum1<< " "<<sum2;

}
