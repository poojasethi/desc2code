#include<stdio.h>
int main()
{
	int n,a[100000],m;
	int i;
	int c,d;
	__int64 s1=0,s2=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&c);
		a[c]=i;
	}
	scanf("%d",&m);
	for(i=0;i<m;i++)
	{
		scanf("%d",&d);
		s1=s1+a[d];
		s2=s2+n-a[d]+1;
	}
	printf("%I64d %I64d\n",s1,s2);
	return 0;
}
