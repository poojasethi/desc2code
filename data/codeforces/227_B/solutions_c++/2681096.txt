#include <iostream>
#include <cmath>
#include <string>
using namespace std;
long long n,i,a[111111],p[111111],x,m,a1,a2;
int main()
{
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		p[a[i]]=i+1;
	}
	cin>>m;
	for(i=0;i<m;i++)
	{
		cin>>x;
		a1+=p[x];
		a2+=n-p[x]+1;
	}
	cout<<a1<<" "<<a2<<endl;
	return 0;
}
