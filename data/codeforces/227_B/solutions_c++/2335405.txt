#include <iostream>

using namespace std;

int a[100000];
int n,m,tmp;
long long l=0, r=0;

int main(){
	cin >> n;
	for (int i=0; i<n; ++i) {
		cin >> tmp;
		a[tmp]=i;
	}
	cin >> m;
	for (int i=0;i<m;++i){
		cin >> tmp;
		l += a[tmp] + 1;
		r += n - a[tmp];
	}
	cout << l << ' ' << r << endl;
	return 0;
}
