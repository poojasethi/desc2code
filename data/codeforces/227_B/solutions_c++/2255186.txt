#include<iostream>
using namespace std;

int A[100006];
int main(){
		long long a = 0, b = 0;
		int n, m, v, q;
		cin >> n;
		for(int i = 0; i < n; ++i){
				cin >> v;
				A[v] = i;
		}
		cin >> m;
		for(int i = 0; i < m; ++i){
				cin >> q;
				a += A[q] + 1;
				b += n - A[q];
		}
		cout << a << " " << b << endl;

}
