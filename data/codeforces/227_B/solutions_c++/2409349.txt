#include<stdio.h>
int n,m,a,used[100010];
__int64 sum1,sum2;
int main()
{
	int i;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a);
		used[a]=i;
	}
	scanf("%d",&m);
	for(i=1;i<=m;i++)
	{
		scanf("%d",&a);
		sum1+=used[a];
		sum2+=n-used[a]+1;
	}
	printf("%I64d %I64d\n",sum1,sum2);
	return 0;
}
