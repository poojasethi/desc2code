#include <iostream>

using namespace std;

int main(){
	int n, A[100100], pos[100100];
	cin>>n;
	for (int i = 0; i < n; i++){
		cin>>A[i];
		pos[A[i]] = i;
	}
	long long m, q, c1 = 0, c2 = 0;
	cin>>m;
	while(m--){
		cin>>q;
		c1 += pos[q]+1;
		c2 += n-pos[q];
	}
	cout<<c1<<" "<<c2<<endl;
	return 0;
}
