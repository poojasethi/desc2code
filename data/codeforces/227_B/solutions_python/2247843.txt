n = int(raw_input())
a = map(int, raw_input().split())
hash = {}

for i in xrange(len(a)):
    hash[a[i]] = i + 1

m = int(raw_input())
b = map(int, raw_input().split())
ansA, ansB = 0, 0
for i in xrange(m):
    ansA += hash[b[i]]
    ansB += n - hash[b[i]] + 1
print ansA, ansB
