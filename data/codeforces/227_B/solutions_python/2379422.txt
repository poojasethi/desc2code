from sys import stdin
n = int(stdin.readline().split()[0])
a = map(int, stdin.readline().split())
m = int(stdin.readline().split()[0])
b = map(int, stdin.readline().split())
p = [n] * 100001
v = [n] * 100001
for i in xrange(n):
    if p[a[i]] == n:
        p[a[i]] = i + 1
    v[a[i]] = n - i
sp = 0
sv = 0
for x in b:
    sv += v[x]
    sp += p[x]
print str(sp) + ' ' + str(sv) + '\n'
