def main():
	import sys
	n, aa, m, bb = sys.stdin.readlines()
	n = convert(n)[0]
	m = convert(m)[0]
	aa = convert(aa)[:n]
	bb = convert(bb)[:m]

	mm = dict((a, index) for index, a in enumerate(aa))

	fcount, scount = 0, 0
	for b in bb:
		index = mm[b]
		fcount += index + 1
		scount += n - index

	print "%d %d" % (fcount, scount)

def convert(s):
	return tuple(int(x) for x in s.strip().split())

if __name__ == "__main__":
	main()
