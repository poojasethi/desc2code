
def main():
    cin = lambda: map(int, raw_input().split())
    [n] = cin();  a = cin()
    [m] = cin();  b = cin()

    cnt1 = [0]*(n+10)
    cnt2 = [0]*(n+10)
    for i in xrange(n):
        cnt1[a[i]] += i+1
        cnt2[a[i]] += (n-i)
    ans1 = ans2 = 0
    for i in xrange(m):
        ans1 += cnt1[b[i]]
        ans2 += cnt2[b[i]]
    print ans1, ans2

if __name__ == '__main__':
    main()
