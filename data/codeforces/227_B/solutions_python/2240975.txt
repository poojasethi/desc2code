def start():
    from sys import stdin
    n = input()
    inp = stdin.read().split('\n')
    arr = map(int, inp[0].split())
    revArr = [0]*(n+1)
    for x in xrange(n):
        revArr[arr[x]] = x
    q = int(inp[1])
    queries = map(int, inp[2].split())
    vasyaCount, petyaCount = 0, 0
    for el in queries:
         vasyaCount += revArr[el]+1
         petyaCount += (n-revArr[el])
    print vasyaCount, petyaCount

start()
