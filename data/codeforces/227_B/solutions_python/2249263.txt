def main():
    n = int(raw_input())
    data = map(int, raw_input().split())
    _index = {}
    for i, x in enumerate(data):
        if x not in _index:
            _index[x] = [i + 1, i + 1]
        else:
            _index[x][1] = i + 1

    int(raw_input())
    requests = map(int, raw_input().split())
    a, b = 0, 0
    for x in requests:
        a += _index[x][0]
        b += n - _index[x][1] + 1
    print '%s %s' % (a, b)

if __name__ == '__main__':
    main()
