from __future__ import print_function
from collections import defaultdict
n = input()
mas = raw_input().split()
d = defaultdict(list)
for ind, elem in enumerate(mas, start=1):
    d[elem].append(ind)

for i in d:
    d[i] = [d[i][0], d[i][-1]]

m = input()
vasya = petya = 0
elems = raw_input().split()
for elem in elems:
    if elem in d:
        vasya += d[elem][0]
        petya += n + 1 - d[elem][1]
    else:
        vasya += n
        petya += n

print(vasya, petya)
