NAME = "effective_approach"
try:
    inFile = open(NAME+".txt")
except:
    pass

def read():
    try:
        return inFile.readline().strip()
    except:
        return raw_input().strip()

n = int(read())
a = map(int,read().split())
pos = [0]*n
for i in xrange(n):
    pos[a[i]-1] = i
    
x = y = 0
read()
q = map(lambda x: int(x)-1, read().split())

for z in q:
    x += pos[z]+1
    y += n-pos[z]
print x,y
