import sys
def main():
    RL = sys.stdin.readline
    n = int(RL())
    a = map(int, RL().split())
    d = dict()
    d2 = dict()
    for i, x in enumerate(a):
        if x not in d:
            d[x] = i+1
        d2[x] = n-i
    m = int(RL())
    s = t = 0
    for x in map(int, RL().split()):
        s += d[x]
        t += d2[x]
    print s, t
main()
