#!/usr/bin/python
n = input()
src = map(int, raw_input().split())
m = input()
qry = map(int, raw_input().split())

def num(lst):
    cp = [n+1]*(10**5+1)
    for i in range(n):
        x = lst[i]
        cp[x] = min(cp[x], i+1)
    return sum([cp[x] for x in qry])

print num(src), num(list(reversed(src)))
