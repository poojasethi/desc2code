
mysize = int(raw_input())

myarr = map(int, raw_input().split())

my1 = [0,] * mysize

for i in range(len(myarr)):
    my1[myarr[i] - 1] += i + 1

mychecks = int(raw_input())

tocheck = map(int, raw_input().split())

myp1 = 0

for i in tocheck:
    myp1 += my1[i - 1]

out = str(myp1) + ' ' + str(mychecks * (mysize + 1) - myp1)

print out
