def main():
	n = input()
	a = map(int, raw_input().split())
	for i in xrange(30, -1, -1):
		f = []
		for j in a:
			if j & (1 << i) > 0: f.append(j)
		if f:
			ans = f[0]
			for j in xrange(1, len(f)): ans &= f[j]
			if ans != 0 and 0 == ans % (1 << i):
				print len(f)
				print " ".join(map(str, f))
				break

if __name__ == "__main__": main()
