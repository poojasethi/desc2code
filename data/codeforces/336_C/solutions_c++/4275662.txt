#include<iostream>
#include<cstdio>
#define MAXN 100100
using namespace std;
int n,tot=0;
int ans[MAXN],a[MAXN];
int path[MAXN];
int main()
{
	cin>>n;
	for (int i=1;i<=n;++i)
		scanf("%d",&a[i]);
	for (int i=1;i<=30;++i)
	{
		int tmp=(1<<(i-1))-1,cnt=0;
		int full=tmp;
		for (int j=1;j<=n;++j)
			if (a[j]&(1<<(i-1)))
			{
				++cnt;
				tmp=tmp&(a[j]&full);
				path[cnt]=a[j];
			}
		if (tmp!=0) continue;
		tot=cnt;
		for (int i=1;i<=cnt;++i)
			ans[i]=path[i];
	}
	cout<<tot<<endl;
	for (int i=1;i<=tot;++i)
		printf("%d ",ans[i]);
	cout<<endl;
	return 0;
}
