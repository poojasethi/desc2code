#include <iostream>
#include <algorithm>
#include <utility>
#include <vector>
using namespace std;
int a[100010];
int main()
{
    ios_base::sync_with_stdio(0);
    int n; cin>>n;
    for(int i=0;i<n;++i) cin>>a[i];

    int v=-1, k=0;
    for(int j=0;j<32;++j) {
        int t=-1, c=0;
        for(int i=0;i<n;++i) if (a[i]&(1<<j)) { t&=a[i]; ++c; }
        if (c && !(t&(1<<j)-1)) { v=j; k=c; }
    }
    cout<<k<<endl;
    for(int i=0;i<n;++i) if (a[i]&(1<<v)) cout<<a[i]<<' '; cout<<endl;
    return 0;
}
