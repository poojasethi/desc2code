#include<cstdio>
#include<iostream>
#include<cstring>
#include<cstdlib>
using namespace std;

int a[100100],p[100100];

int main()
{
	int n;
	scanf("%d",&n);
	for (int i=1; i<=n; i++)
		scanf("%d",&a[i]);
	for (int i=30; i+1; i--)
	{
		int tmp=1<<i,now=(1<<30)-1,num=0;
		memset(p,0,sizeof(p));
		for (int j=1; j<=n; j++)
			if (a[j]&tmp) now&=a[j],p[++num]=a[j];
		if (now%tmp) continue; else
		{
			printf("%d\n",num);
			for (int i=1; i<=num; i++)
				printf("%d ",p[i]);
			return 0;
		}
	} printf("-1"); return 0; 
}
