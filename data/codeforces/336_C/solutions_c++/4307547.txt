#include<cstdio>
int a[100005];
int main(){
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    for(int v=30;v>=0;v--){
        int t=(1<<v)-1,ans=0,x=1<<v;
        for(int i=0;i<n;i++){
            if(a[i]&x){
                t&=a[i];
                ans++;
            }
        }
        if(t==0){
            printf("%d\n",ans);
            for(int i=0;i<n;i++)
                if(a[i]&x) printf("%d ",a[i]);
            return 0;
        }
    }
}
