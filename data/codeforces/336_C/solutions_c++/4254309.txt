#include<iostream>
#include<vector>
using namespace std;
int a[100000];
int mi;
int mcnt;
int mmax=-1;
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++) cin>>a[i];
	for(int i=0;i<31;i++){
		int mm=0;
		int toand=0x7fffffff;
		for(int j=0;j<n;j++){
			if((a[j]&(1<<i))!=0){
				toand&=a[j];
				mm++;
			}
		}
		int bt=0;
		while(toand%2!=1){
			bt++;
			toand/=2;
		}
		if(mmax<bt){
			mmax=bt;
			mi=i;
			mcnt=mm;
		}
	}
	cout<<mcnt<<endl;
	for(int i=0;i<n;i++){
		if((a[i]&(1<<mi))!=0){
			cout<<a[i]<<" ";
		}
	}
	return 0;
}
