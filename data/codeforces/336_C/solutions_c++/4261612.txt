#include<cstdio>
#include<vector>
using namespace std;
int arr[100000];
vector<int>ans;
int main(){
	int n,mask;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",&arr[i]);
	for(int i=29;i>=0;i--){
		mask=-1;
		ans.clear();
		for(int j=0;j<n;j++)
			if(arr[j]&(1<<i)){
				if(mask==-1)
					mask=arr[j];
				else
					mask&=arr[j];
				 ans.push_back(arr[j]);
			}
		if((mask%(1<<i))==0)
			break;
	}
	printf("%d\n",(int)ans.size());
	for(int i=0;i<(int)ans.size();i++){
		if(i)printf(" ");
		printf("%d",ans[i]);
	}
}
