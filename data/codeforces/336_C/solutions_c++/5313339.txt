#include<iostream>
using namespace std;
int main()
{
int n,a[100005],i,k=-1,j,l,p,r=0,o;
cin>>n;
for(i=1;i<=n;i++) cin>>a[i];
for(i=1;i<=1000000000;i*=2)
{
l=p=0;
for(j=1;j<=n;j++)
{
if(a[j]&i)
{
p++;
if(!l) l=a[j];
else l&=a[j];
}
}
j=0;
while(l>0 && l%2==0) j++,l/=2;
if(j>k || j==k && p>r)
{
k=j;
r=p;
o=i;
}
}
cout<<r<<endl;
for(i=1;i<=n;i++)
{
if(a[i]&o) cout<<a[i]<<" ";
}
return 0;
}
