#include <stdio.h>

int seq[100005];

int main()
{
	int n;
	scanf("%d", &n);

	for(int i = 0; i < n; i++)
		scanf("%d", seq+i);

	for(int i = 31; i >= 0; i--) {
		int sf = -1, cur = 0;
		for(int j = 0; j < n; j++) {
			if(seq[j] & (1 << i)) {
				if(sf == -1) sf = seq[j];
				else sf = sf & seq[j];
				cur++;
			}
		}
		if((sf & ((1 << i)-1)) == 0) {
			printf("%d\n", cur);
			for(int j = 0; j < n; j++)
				if(seq[j] & (1 << i))
					printf("%d ", seq[j]);
			printf("\n");
			return 0;
		}
	}
}
