#include <cstdio>
#include <iostream>

using namespace std;

int list[35][100010];
int size[35];
int a[100010];

int main(){
	int n, ind = 0; cin >> n;
	for(int i = 0; i < n; i++) cin >> a[i];
	for(int j = 0; j < 31; j++){
		int res = -1 + (1<<30);
		//cout << res << endl;
		for(int i = 0; i < n; i++){
			if(a[i] & (1 << j)){
				list[j][size[j]++] = a[i];
				res &= a[i];
			}
		}
		if(res % (1 << j) == 0){
			ind = j;
		}
	}
	cout << size[ind] << endl;
	for(int i = 0; i < size[ind]; i++)
				printf("%d ", list[ind][i]);
	return 0;
}
