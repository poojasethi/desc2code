#include <cstdio>
#include <bitset>
#include <iostream>

using namespace std;

int v[100100];
#define forn(i,n) for(int i=0;i<n;i++)

int main(int argc, char const *argv[])
{
	int N;
	scanf("%d",&N);
	forn(i,N)
		scanf("%d",&v[i]);
	int b = 29;
	for (;b>=0;b--) {
		int acum =  int(1 << b)-1;
		int count = 0;
		forn (i,N) {
			if (v[i] & int(1<<b)) {
				count++;
				acum &= v[i];
			}
		}
		if (!acum) {
			printf("%d\n", count);
			forn (i,N)
				if (v[i] & int(1<<b))
					printf("%d ", v[i]);
			printf("\n");
			return 0;
		}
	}
	return 0;
}
