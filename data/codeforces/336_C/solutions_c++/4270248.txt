#include <iostream>
#include <vector>
using namespace std;
int main()
{int n;cin>>n;
vector <int> a(n);
for (int i=0;i<n;i++) cin>>a[i];
vector <vector <int> > ans(31);
int pos = 0;
for (int j=0;j<31;j++)
{int res=(1<<30)-1;  // макссимальное число типа int
  for (int i=0;i<n;i++)
   if (a[i]&(1<<j))
   {ans[j].push_back(a[i]); res&=a[i];}
    if (res%(1<<j)==0)  pos=j;
}
cout<<ans[pos].size()<<endl;
for (int i=0;i<ans[pos].size();i++) cout<<ans[pos][i]<<" ";
return 0;
}
