#include <cstdio>
#define f(a, b, c) for(int a = b; a < c; a++)

int n, l[100009], x, q[100004], e;

int main(){
	scanf("%d", &n);
	f(i, 0, n) scanf("%d", &l[i]);
	
	for(int i = 1 << 30; i; i = i >> 1){
		e = 0;
		x = (1 << 30) - 1 + (1 << 30);
		f(j, 0, n) if(i & l[j]){
			q[e++] = l[j];
			x &= l[j];
		}
		
		if(i == (x & -x)){
			e = 0;
			f(j, 0, n) if(i & l[j]) q[e++] = l[j];
			printf("%d\n", e);
			f(i, 0, e) printf("%d%c", q[i], i == e - 1 ? '\n' : ' ');
			break;
		}
	}
	
	scanf("\n");
}
