#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;

int a[110000];
int main()
{
    int n,M,i,j,k;

    scanf("%d",&n);
    for(i=1;i<=n;i++)
	scanf("%d",&a[i]);
    M=0;
    int ps;
    for(i=30;i>=0;i--)
    {
	k=0;
	int now=(1<<31)-1;
	for(j=1;j<=n;j++)if(a[j]&(1<<i))
	{
	    now&=a[j];
	    k++;
	}
	if(now&&now%(1<<i)==0)
	{
	    M=k;
	    ps=i;
	    break;
	}
    }
    printf("%d\n",M);
    for(i=1;i<=n;i++)if(a[i]&(1<<ps))
	printf("%d ",a[i]);
    puts("");
}
