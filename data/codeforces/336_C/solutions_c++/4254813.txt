#include <iostream>
#include <algorithm>
#include <cstdio>
#include <iomanip>
#include <cmath>
using namespace std;

int n;
int a[100000];
int p;
int mas; int mc;
int k;

int main(){
	cin >> n;
	for(int i=0; i<n; i++) cin >> a[i];
	mas = -1;
	for(int c=1; c<=a[n-1]; c<<=1){
		p=(c<<1)-1;
		for(int i=0; i<n; i++) if(c&a[i]) p&=a[i];
		if(p)if(__builtin_ctz(p)>mas){mas=__builtin_ctz(p); mc=c;}
	}
	for(int i=0; i<n; i++) if(mc&a[i])k++;
	cout << k << '\n';
	for(int i=0; i<n; i++) if(mc&a[i]) cout << a[i] << ' ';
	
	return 0;
}
