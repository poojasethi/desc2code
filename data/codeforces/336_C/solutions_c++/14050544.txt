#include <cstdio>
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

const int MAXN=100010;
int n;
int a[MAXN];
int b[MAXN];

int main() {
	scanf("%d",&n);
	REP(i,n) scanf("%d",&a[i]);
	for (int i=30;i>=0;--i) {
		int tmp=(1<<i)-1;
		b[0]=0;
		REP(j,n)
			if (a[j]&(1<<i)) {
				tmp&=a[j];
				b[++b[0]]=a[j];
			}
		if (!tmp) {
			printf("%d\n",b[0]);
			REP(j,b[0])
				printf("%d ",b[j]);
			return 0;
		}
	}
	return 0;
}
