#include<cstdio>
#include<cstdlib>
#include<cstring>

using namespace std;

const int maxn=100010;

int n,z[maxn];

int main()
{
	scanf("%d",&n);
	for (int a=1;a<=n;a++)
		scanf("%d",&z[a]);
	for (int a=31;a>=0;a--)
	{
		int v=-1;
		int cnt=0;
		for (int b=1;b<=n;b++)
			if ((z[b]>>a)&1)
			{
				cnt++;
				if (v==-1) v=z[b];
				else v&=z[b];
			}
		if (v!=-1 && !(v%(1<<a)))
		{
			printf("%d\n",cnt);
			for (int b=1;b<=n;b++)
				if ((z[b]>>a)&1) printf("%d ",z[b]);
			printf("\n");
			return 0;
		}
	}
	printf("-1\n");

	return 0;
}
