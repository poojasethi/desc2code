#include <cstdio>
#include <vector>

using namespace std;

vector<int> a, cur, ans;

int main()
{
	int bit,i,n,t;
	scanf("%d", &n); 
	a.resize(n);
	for(i = 0; i < n; i++) 
	scanf("%d", &a[i]);
	for(bit = 0; bit < 31; bit++) {
		t = (1<<30) - 1;
		cur.clear();
		for(i = 0; i < n; i++) {
			if ((a[i] & (1<<bit))) cur.push_back(a[i]), t&= a[i]; 
		}
		if ((t % (1<<bit)) == 0) ans = cur; 
	}
	printf("%d\n", (int)ans.size());
	for(i = 0; i < (int)ans.size(); i++) 
		printf("%d\n", ans[i]);
}
