#include <cstdio>

int N,a[100005],q[100005];

void doit()
{
	scanf("%d",&N);
	for (int i=1,x; i<=N; i++) scanf("%d",&a[i]);
	for (int t=29; t>=0; t--)
	{
		int x=(1<<t+1)-1; q[0]=0;
		for (int i=1; i<=N; i++) if ((a[i]>>t)&1) x&=a[i],q[++q[0]]=a[i];
		if (x==1<<t)
		{
			printf("%d\n",q[0]);
			for (int i=1; i<=q[0]; i++) printf("%d%c",q[i],i==q[0]?'\n':' ');
			return;
		}
	}
	puts("-1");
}

int main()
{
	doit();
	return 0;
}
