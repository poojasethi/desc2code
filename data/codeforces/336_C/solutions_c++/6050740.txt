#include <stdio.h>
#include <memory.h>

int main()
{
	int n,M[100010]={0};
	scanf("%d",&n);
	for (int i=1; i<=n; i++) { scanf("%d",M+i); }

	int Ret[100010]={0}, ret=-1;
	for (int v=0; (1<<v)<=M[n]; v++) {
		int sum=0x7fffffff, cnt=0;
		bool R[100010]={0};
		for (int i=1; i<=n; i++) {
			if (M[i] & (1<<v)) { R[i]=1; cnt++; sum &= M[i]; }
		}
		if (sum % (1<<v) == 0) {
			for (int i=1; i<=n; i++) { Ret[i] = R[i]; }
			ret = cnt;
		}
	}
	printf("%d\n", ret);
	for (int i=1; i<=n; i++) {
		if (Ret[i]) { printf("%d ",M[i]); }
	}
}
