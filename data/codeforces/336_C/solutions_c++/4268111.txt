#include <iostream>
#include <stdio.h>
#include <vector>

#define pb push_back

using namespace std;

int n,a[111111];
vector < int > ans,cur;

int main()
{
	scanf("%d",&n);
	for ( int i = 1; i <= n; i++ )
		scanf("%d",&a[i]);

	for ( int bit = 0; bit <= 30; bit++ ) {
		int t = (1 << 30) - 1;
		cur.clear();
		for ( int i = 1; i <= n; i++ ) {
			if  ( (a[i] >> bit ) & 1 ) cur.pb(a[i]),t&=a[i];
		}
		if ( (t % (1 << bit)  == 0) ) ans = cur;
	}
	printf("%d\n",ans.size());
	for ( int i = 0;i < ans.size(); i++ )
		printf("%d ",ans[i]);

}
