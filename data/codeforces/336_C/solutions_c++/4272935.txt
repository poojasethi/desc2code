#include <iostream>
#include <vector>
#include <cstdlib>

using namespace std;

int main()
{
  int n;
  cin>>n;
  vector<int> v(n);
  for (int i=0;i<n;i++)
    cin>>v[i];
  for (int b=29;b>=0;b--) {
    int x=1<<b;
    int r=(1<<(b+1))-1;
    vector<int> sol;
    for (int i=0;i<n;i++) {
      if (v[i]&x) {
	r=v[i]&r;
	sol.push_back(v[i]);
      }
    }
    if (r==x) {
      cout<<int(sol.size())<<endl;
      for (int i=0;i<int(sol.size());i++) {
	if (i>0) cout<<" ";
	cout<<sol[i];
      }
      cout<<endl;
      exit(0);
    }
  }
  cout<<-1<<endl;
}
