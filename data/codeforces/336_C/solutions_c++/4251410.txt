#include<stdio.h>
#include<vector>
using namespace std;
int a[100000];
vector<int> ans;
int main(){
    int n,i,j;
    scanf("%d",&n);
    for(i=0;i<n;i++)scanf("%d",&a[i]);
    for(i=30;i>=0;i--){
        int res=(1<<(i+1))-1;
        for(j=0;j<n;j++)if(a[j]>>i&1)res&=a[j];
        if(res==1<<i)break;
    }
    int k=i;
    for(i=0;i<n;i++)if(k==-1||a[i]>>k&1)ans.push_back(a[i]);
    printf("%d\n",ans.size());
    for(i=0;i<ans.size();i++)printf("%d ",ans[i]);
    puts("");
    return 0;
}
