#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;

const int MAXN=1e5+9;

vector <int> ans;
int a[MAXN];
int n,i,k,cnt,tmp;

int main()
{
	cin >> n;
	for(i=1;i<=n;i++)
		cin >> a[i];
	
	for(k=30;k>=0;k--)
	{
		tmp=(1<<(k+1))-1;
		ans.clear();
		for(i=1;i<=n;i++)
		{
			if(a[i]&(1<<k))
			{
				tmp&=a[i];
				ans.push_back(a[i]);
			}
		}
		if(tmp==(1<<k))
		{
			cnt=ans.size();
			printf("%d\n",cnt);
			for(i=0;i<cnt;i++)
				printf("%d ",ans[i]);
			printf("\n");
			break;
		}
	}
    return 0;
}
