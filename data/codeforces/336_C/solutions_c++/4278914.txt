#include<iostream>
#include<cstdio>
using namespace std;

int a[100000];
int b[100001];
int main(){
    int n,i,j,t,k;
    scanf("%d",&n);

    for(i=0;i<n;i++) scanf("%d",&a[i]);

    for(i=30;i>=0;i--){
        k=1<<i;
        t=k-1;
        b[0]=0;
        for(j=0;j<n;j++)
            if(a[j] & k){
                t&=a[j];
                b[++b[0]]=a[j];
            }
        if(!t){
            printf("%d\n",b[0]);
            for(j=1;j<=b[0];j++)printf("%d ",b[j]);
            break;
        }
    }
    return 0;
}
