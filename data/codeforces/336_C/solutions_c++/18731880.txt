#include<bits/stdc++.h>
#define newl '\n'
using namespace std;
const int N=1e5+5;
int a[N],n,i,j,b[N],id,nd;
int main()
{
    ios::sync_with_stdio(false);cin.tie(0);
    cin>>n;
    for(i=0;i<n;i++)cin>>a[i];
    for(j=30 ; j>=0 ; j--)
    {
    	id=0;
    	for(i=0 ; i<n ;i++)
    		if( a[i] & (1<<j) )
    			b[id++]=a[i];
    	nd=b[0];
    	for(i=1;i<id;i++)
    		nd&=b[i];
    	if(id && nd%(1<<j)==0 )
    	{
    	    cout<<id<<newl;
            for(i=0; i<id; i++)
    	 	 cout<<b[i]<<" ";
            break;
        }
    }
	return 0;
}
