#include <iostream>
#include <cstdio>
using namespace std;

const int N=121111;
int num[N];

int main()
{
	ios::sync_with_stdio(false);
	int n; cin>>n;
	for(int i=0;i<n;i++)
		cin>>num[i];
	for(int k=30;k>=0;k--)
	{
		int x=(~0);
		for(int i=0;i<n;i++)
			if((num[i]>>k)&1)
				x&=num[i];
		if((x&((1<<k)-1))==0)
		{
			int ans=0;
			for(int i=0;i<n;i++)
				if((num[i]>>k)&1) ans++;
			cout<<ans<<'\n';
			for(int i=0;i<n;i++)
				if((num[i]>>k)&1) cout<<num[i]<<' ';
			cout<<'\n';
			break;
		}
	}
	return 0;
}
