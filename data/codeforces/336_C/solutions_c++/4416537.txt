#include <cstdio>
#include <algorithm>
#include <bitset>
using namespace std;

int seq[100001];

int main(void){
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",&seq[i]);

	for(int i=30;i>=0;i--){
		int j=(1<<i)-1;
		int num=0;
		for(int k=0;k<n;k++){
			if(seq[k]&(1<<i)){
				j&=seq[k];
				num++;
			}
		}
		if(j)
			continue;
		printf("%d\n",num);
		for(int k=0; k<n; ++k)
            if(seq[k]&(1<<i))
                printf("%d ",seq[k]);
		return 0;
	}
	return 0;
}
