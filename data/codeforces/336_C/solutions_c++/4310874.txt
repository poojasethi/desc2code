#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
int n,a[100005];
vector<int> t,ans;
int main()
{
    int i,j;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    scanf("%d",&a[i]);
    for(i=0;i<31;i++)
    {
        int tmp=(1<<30)-1;
        t.clear();
        for(j=1;j<=n;j++)
        if((a[j])&(1<<i)){t.push_back(a[j]);tmp&=a[j];}
        if(tmp%(1<<i)==0)ans=t;
    }
    printf("%d\n",ans.size());
    for(i=0;i<ans.size();i++)
    printf("%d ",ans[i]);
    printf("\n");
    return 0;
}
