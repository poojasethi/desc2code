#include<cstdio>
#define N 100000
int v[N];
int main(){
	int n;scanf("%d",&n);
	for(int i=0;i<n;++i)
		scanf("%d",v+i);
	int mx=v[n-1];

	int msk=1,ans=-1;
	for(;msk<=mx;msk<<=1){
		int x=-1;
		for(int i=0;i<n;++i)
			if(msk&v[i])
				x&=v[i];
		if( __builtin_ctz(x)==__builtin_ctz(msk) )
			ans=msk;
	}
	msk=ans;
	if(msk==-1){
		printf("%d\n",n);
		for(int i=0;i<n;++i)
			printf("%d",v[i]);
	}else{
		int k=0;
		for(int i=0;i<n;++i)
			if(msk&v[i])
				++k;
		printf("%d\n",k);
		for(int i=0;i<n;++i)
			if(msk&v[i])
				printf("%d ",v[i]);
	}
	puts("");
	
	
	return 0;
}
