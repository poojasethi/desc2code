#include <iostream>

using namespace std;

int main() {
  int n;
  cin >> n;
  int a[n];
  for (int i = 0; i < n; i++)
    cin >> a[i];
  for (int i = 29; i >= 0; i--) {
    int k = 0;
    int p = 1<<i;
    unsigned long long d = 1073741823;
    for (int j = 0; j < n; j++)
      if (a[j] % (p * 2) / p) {
        k++;
        d = d & a[j];
      }
    if ((d >= p) && !(d % p)) {
      cout << k << "\n";
      for (int j = 0; j < n; j++)
        if (a[j] % (p * 2) / p) {
          cout << a[j] << " ";
        }
      break;
    }
  }/**/
  return 0;
}
