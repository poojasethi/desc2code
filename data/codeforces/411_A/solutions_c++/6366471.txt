#include <iostream>
#include <string>

using namespace std;

string s; int f1,f2,f3;

int main()
{
   cin>>s;
   for(int i=0; i<s.size();i++) {
    if (s[i]>64 && s[i]<91) f1=1;
    if (s[i]>96 && s[i]<123) f2=1;
    if (s[i]>47 && s[i]<58) f3=1;}
   if (s.size()>4 && f1+f2+f3==3) cout<<"Correct"; else cout<<"Too weak";
}
