#include<bits/stdc++.h>
using namespace std;
int main()
{
    string s;
    cin>>s;
    int x=0,y=0,z=0;
    for(int i=0;i<s.size();i++)
    {
        if(isdigit(s[i])){x=1;}
        if(isupper(s[i])){y=1;}
        if(islower(s[i])){z=1;}
    }
    cout<<(x&&y&&z&&s.size()>=5?"Correct\n":"Too weak\n");
    return 0;
}
