#include<stdio.h>
#include<string.h>
char s[10000];
int main()
{
	int m,n,j,k,l,t,a,b,c,d;
	a=b=c=d=0;
	scanf("%s",s);
	n=strlen(s);
	if(n>=5)d++;
	for(j=0;j<n;j++)
	{
		if(s[j]>='a'&&s[j]<='z')a++;
		if(s[j]>='A'&&s[j]<='Z')b++;
		if(s[j]>='0'&&s[j]<='9')c++;
	}
	if(a&&b&&c&&d)printf("Correct\n");
	else printf("Too weak\n");
	return 0;
}
