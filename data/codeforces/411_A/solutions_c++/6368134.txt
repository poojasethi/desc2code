#include <bits/stdc++.h>
using namespace std;

int main()
{
	int i,a=0,b=0,c=0;
	string str;
	cin >> str;
	i=0;
	if( str.size() < 5 ) {printf("Too weak\n"); return 0;}
	while( str[i] )
	{
		if( islower(str[i]) ) a=1;
		if( isupper(str[i]) ) b=1;
		if( isdigit(str[i]) ) c=1;
			
		i++;
	}
	if( a && b && c) printf("Correct\n");
	else printf("Too weak\n"); 
return 0;
}
