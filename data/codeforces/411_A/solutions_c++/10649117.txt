#include <iostream>

using namespace std;

int main()
{
   string x;
   int c=0,s=0,d=0,k=0;
   cin >>x;
   for(int i=0;i<x.size();i++)
   {
       if(isupper(x[i])) c++;
       if(islower(x[i])) s++;
       if(x.size()>4) d++;
       if (isdigit(x[i])) k++;
   }
    if(c>0&&s>0&&d>0&&k>0) cout <<"Correct";
    else cout <<"Too weak" ;
return 0;




}
