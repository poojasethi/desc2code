#include <iostream>
#include <string>

using namespace std;

int main() {
	string s;
	int i;
	bool a=0,b=0,c=0;
	
	cin>>s;

	for(i=0;i<s.size();++i) {
		if (s[i]>='A'&&s[i]<='Z') a=1;
		if (s[i]>='a'&&s[i]<='z') b=1;
		if (s[i]>='0'&&s[i]<='9') c=1;
	}

	if (s.size()>=5&&a&&b&&c) cout<<"Correct"<<endl; else cout<<"Too weak"<<endl;

	return 0;
}
