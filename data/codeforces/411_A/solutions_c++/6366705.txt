#include<iostream>
#include<string>
using namespace std;

main()
{
	int a=0,b=0,c=0;
	string s;
	cin>>s;
	for(int i=0;i<s.size();i++)
	{
		if(s[i]>=97 && s[i]<=122)a++;
		if(s[i]>=48 && s[i]<=57) b++;
		if(s[i]>=65 && s[i]<=90) c++;
	}
	
	if(a!=0 && b!=0 && c!=0 && s.size()>=5)cout<<"Correct";
	else cout<<"Too weak";
}
