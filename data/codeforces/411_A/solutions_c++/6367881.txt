#include <iostream>
#include <cctype>
using namespace std;

int main() {
	string s; cin >> s;
	bool hasSmall = false, hasBig = false, hasDigit = false;
	for(unsigned i = 0; i < s.size(); ++i) {
		hasSmall |= islower(s[i]);
		hasBig |= isupper(s[i]);
		hasDigit |= isdigit(s[i]);
	}
	cout << (s.size() >= 5 && hasSmall && hasBig && hasDigit ? "Correct" : "Too weak");
	return 0;
}
