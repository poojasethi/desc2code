#include <bits/stdc++.h>

using namespace std;

char A[105];

int main(){
	int bh=0,kh=0,s=0,i,j,n;
	scanf("%s",A);
	n=strlen(A);
	for(i=0;i<n;i++){
		if(isdigit(A[i]))s=1;	
		if(A[i]-'A'>=0 && A[i]-'A'<=26) bh=1;
		if(A[i]-'a'>=0 && A[i]-'a'<=26) kh=1;
	}
	if(bh+kh+s==3 && n>=5)
		printf("Correct");
	else
		printf("Too weak");
return 0;
}
