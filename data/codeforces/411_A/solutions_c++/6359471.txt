#include <cstdio>
#include <algorithm>
using namespace std;

char a[101] ;
int x , y , z , i ;

int main() 
{
   scanf("%s",&a);
   for( i=0 ; a[i] ; i++) 
   {
      if(a[i]>='A' && a[i]<='Z') x=1 ;
      if(a[i]>='a' && a[i]<='z') y=1 ;
      if(a[i]>='0' && a[i]<='9') z=1 ;  
   }
   printf(((i>=5) && x && y && z)?"Correct":"Too weak") ;

 return 0;
}
