#include<stdio.h>
#include<string.h>
#include<ctype.h>
int a,b,c,d,i;
char A[105];
int main(){
	int n;
	scanf("%s",A);
	n=strlen(A);
	if(strlen(A)<5){printf("Too weak");return 0;}
	for(i=0;i<n;i++){
		if(isdigit(A[i]))a++;
		if(islower(A[i]))b++;
		if(isupper(A[i]))c++;
	}
	if( a && b && c){printf("Correct");return 0;}
	printf("Too weak");
	return 0;
}
