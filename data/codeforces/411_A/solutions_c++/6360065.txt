#include<stdio.h>
#include<string.h>
#include<ctype.h>
int main()
{
int l,t=0,n=0,m=0,i;
char s[102],c;
scanf("%s",s);
l=strlen(s);
if(l<5)
{
printf("Too weak");
}
else
{
for(i=0;i<l;i++)
{
c=s[i];
if(isdigit(c))
t=1;
if(c>='A'&&c<='Z')
n=1;
if(c>='a'&&c<='z')
m=1;
}
if(t==1&&n==1&m==1)
printf("Correct");
else
printf("Too weak");
}
return 0;
}
