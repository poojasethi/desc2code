#include<stdio.h>
int main()
{
    char ch;
    int a=0,b=0,c=0,d=0;
    while((ch=getchar())!='\n')
    {
       a++;
       if(ch>='A'&&ch<='Z') b++;
       if(ch>='a'&&ch<='z') c++;
       if(ch>='0'&&ch<='9') d++;
    }
    if(a>=5&&c>=1&&d>=1&&b>=1)  printf("Correct\n");
      else printf("Too weak\n");
    return 0;
}
