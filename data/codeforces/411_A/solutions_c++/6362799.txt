#include <iostream>
#include <cctype>

using namespace std;

string s;

int a,b,c;

int main()
{
	cin >> s;
	
	for (int i = 0; i < s.length(); i++)
	{
		if (isupper(s[i])) a = 1;
		if (isdigit(s[i])) b = 1;
		if (islower(s[i])) c = 1;
	}
	
	if (a+b+c!=3 || s.length() < 5)
		cout << "Too weak";
	else
		cout << "Correct";
}
