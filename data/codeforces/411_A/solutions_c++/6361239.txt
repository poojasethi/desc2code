#include<iostream>
#include<string>
using namespace std;
int main()
{
	string s;
	int c=0,c1=0,c2=0,c3=0;
	cin>>s;
	if(s.length()>=5)c=1;
	for(int i=0;i<s.length();i++){
		if(s[i]>='a'&&s[i]<='z')c1=1;
		if(s[i]>='A'&&s[i]<='Z')c2=1;
		if(s[i]>='1'&&s[i]<='9')c3=1;
	}
	if(c1*c2*c3*c==1)cout<<"Correct"<<endl;
	else cout<<"Too weak"<<endl;
	return 0;
}
