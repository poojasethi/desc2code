#include <iostream>
using namespace std;
char ch[1020];
int b[5];
int main(){
    cin >> ch;
    int tot=0;
    for(int &i=tot;ch[i];i++){
        if(ch[i]<='9'&&ch[i]>='0') b[0]=1;
        if(ch[i]<='Z'&&ch[i]>='A') b[1]=1;
        if(ch[i]<='z'&&ch[i]>='a') b[2]=1;
    }
    if(tot>=5&&b[0]&&b[1]&&b[2]) cout <<"Correct";
    else cout <<"Too weak";
}
