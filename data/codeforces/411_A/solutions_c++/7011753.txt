#include<cstdio>
#include<cstring>
using namespace std;
char s[101];
int main()
{
	bool b1=0,b2=0,b3=0;
	scanf("%s",s);
	if(strlen(s)<5)
	{puts("Too weak");return 0;}
	for(int i=0;s[i];++i)
	if(s[i]>47&&s[i]<58)
		b1=1;
	else if(s[i]>64&&s[i]<91)
		b2=1;
	else if(s[i]>96&&s[i]<123)
		b3=1;
	puts(b1&&b2&&b3?"Correct":"Too weak");
}
