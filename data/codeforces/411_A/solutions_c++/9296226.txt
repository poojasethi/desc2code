#include<iostream>
#include<string>
using namespace std;
int main()
{
string s;
cin>>s;
int a=0,b=0,c=0,i;
if(s.size()>=5)
{
for(i=0;i<s.size();i++)
{
if(s[i]>=48 && s[i]<=57)
a=1;
else if(s[i]>=65 && s[i]<=90)
b=1;
else if(s[i]>=97 && s[i]<=122)
c=1;
}
if(a==0 || b==0 || c==0)
cout<<"Too weak";
else
cout<<"Correct";
}
else
cout<<"Too weak";
}
