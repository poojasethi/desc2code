#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i;
	bool flag=0,a=0,b=0,c=0;
	string s;
	cin>>s;
	if(s.size()<5)
		flag=1;
	for(i=0;i<s.size();i++)
	{
		if(s[i]>='a' && s[i]<='z')
			a=1;
		if(s[i]>='A' && s[i]<='Z')
			b=1;
		if(s[i]>='0' && s[i]<='9')
			c=1;
	}
	if(!(a && b && c))
		flag=1;
	(flag) ? cout<<"Too weak\n" : cout<<"Correct\n";	
	return 0;
}
