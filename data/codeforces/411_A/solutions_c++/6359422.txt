#include <iostream>
#include <string>
using namespace std;

string s;
bool A, a, d;

int main(){
	cin >> s;
	if(s.size() < 5){cout << "Too weak"; return 0;}
	for(int i=0; i<(int)s.size(); i++){
		if(s[i]>='A' && s[i]<='Z') A = true;
		if(s[i]>='a' && s[i]<='z') a = true;
		if(s[i]>='0' && s[i]<='9') d = true;
	}
	if(a&A&d) cout << "Correct"; else cout << "Too weak";
	return 0;
}
