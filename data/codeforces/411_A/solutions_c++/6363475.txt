#include <cstdio>
#include <cstring>
#include <cctype>

bool check(const char* s) {
	int n = strlen(s);
	int a = 0, A = 0, d = 0;
	for (int i = 0; i < n; ++i) {
		if (isupper(s[i])) ++A;
		if (islower(s[i])) ++a;
		if (isdigit(s[i])) ++d;
	}
	return n >= 5 && A && a && d;
}

int main() {
	char s[128];
	scanf("%s", s);
	puts(check(s) ? "Correct" : "Too weak");
}
