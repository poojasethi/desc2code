#include<iostream>
using namespace std;
int main()
{
	bool sm=0,l=0,d=0;
	string s;
	int n,i;
	cin>>s;
	n=s.size();
	for(i=0;i<n;i++)
		if(s[i]>='a' && s[i]<='z')sm=1;
		else if(s[i]>='A' && s[i]<='Z')l=1;
		else if(s[i]>='0' && s[i]<='9')d=1;
	if(d+l+sm==3 && n>=5)cout<<"Correct\n";
	else cout<<"Too weak\n";
}
