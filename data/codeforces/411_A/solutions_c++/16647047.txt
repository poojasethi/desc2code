#include<stdio.h>
#include<ctype.h>
int main(void)
{
	char str[105];
	int a, b, c, i;
	a = b = c = 0;
	scanf("%s", str);
	for(i=0;str[i]!='\0';i++)
	{
		if(isdigit(str[i]))
			a = 1;
		if(islower(str[i]))
			b = 1;
		if(isupper(str[i]))
			c = 1;
	}
	if(a==1 && b==1 && c==1 && i>=5)
		printf("Correct\n");
	else
		printf("Too weak\n");
	return 0;
}
