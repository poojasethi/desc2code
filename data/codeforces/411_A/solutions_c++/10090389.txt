#include <iostream>
#include <string>
using namespace std;
int main(){
    string s; cin >> s; 
    bool up = 0, lo = 0, dig = 0;
    for (int i = 0; i < s.size(); i++){
        if (isupper(s[i]))up = 1;
        if (islower(s[i]))lo = 1;
        if (isdigit(s[i]))dig = 1;
    }
    cout << (up&&lo&&dig&&s.size()>=5 ? "Correct\n" : "Too weak\n");
    return 0;
}
