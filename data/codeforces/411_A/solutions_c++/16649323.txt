#include<stdio.h>
#include<string.h>
int main()
{char p[100];
int a=0,b=0,c=0,i,len;
gets(p);
len=strlen(p);
if(len<5)
printf("Too weak\n");
else
{for(i=0;i<len;i++)
{if(p[i]>='a'&&p[i]<='z')
a++;
else if(p[i]>='A'&&p[i]<='Z')
b++;
else if(p[i]>='0'&&p[i]<='9')
c++;}
if(a>0&&b>0&&c>0)
printf("Correct\n");
else
printf("Too weak\n");}
return 0;}
