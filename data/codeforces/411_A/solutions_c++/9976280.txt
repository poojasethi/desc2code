#include<iostream>
#include<cctype>
#include<string>
using namespace std;

int main(){
	bool a=0,b=0,c=0;
	string s;
	cin>>s;
	for(int i=0;i<s.size();i++){
		if(islower(s[i])) a=1;
		if(isupper(s[i])) b=1;
		if(isdigit(s[i])) c=1;
	}
	if(a&&b&&c&&s.size()>=5){
		cout<<"Correct";
	}
	else cout<<"Too weak";
	return 0;
}
