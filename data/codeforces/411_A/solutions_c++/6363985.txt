#include <iostream>
#include <string>
using namespace std;
int main()
{
	int i,j;
	bool q,w,e;
	string s;
	q=w=e=0;
	cin>>s;
	for (i=0;i<s.length();i++)
	if (s[i]>='a'&&s[i]<='z') q=1;
	else if (s[i]>='A'&&s[i]<='Z') w=1;
	else if (s[i]>='0'&&s[i]<='9') e=1;
	if (q&&w&&e&&s.length()>=5) cout<<"Correct"<<endl; else cout<<"Too weak"<<endl;
}
