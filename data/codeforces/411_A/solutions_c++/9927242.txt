#include <bits/stdc++.h>

int main()
{
	std::string s;
	std::cin >> s;
	
	bool hasL = 0, hasU = 0, hasN = 0;
	for(int i = 0; i < s.size(); i++)
	{
		if(isupper(s[i])) hasU = 1;
		if(islower(s[i])) hasL = 1;
		if(isdigit(s[i])) hasN = 1;
	}
	
	if(hasL && hasU && hasN && (int)s.size() >= 5)
	{
		puts("Correct");
	}
	else puts("Too weak");
	
	return 0;
}
