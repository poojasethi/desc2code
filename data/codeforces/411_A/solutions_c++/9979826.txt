#include<stdio.h>
int main()
{
	char c;
	int a=0,A=0,n=0,k=0;
	c=getchar();
	while(c!='\n')
	{
		if(c>='a'&&c<='z')
			a++;
		else if(c>='A'&&c<='Z')
			A++;
		else if(c>='0'&&c<='9')
			k++;
		else;
		n++;
		c=getchar();
	}
	if(a>=1&&A>=1&&k>=1&&n>=5)  printf("Correct");
	else printf("Too weak");
	return 0;
}
