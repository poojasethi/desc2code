#include <iostream>
using namespace std;
string s;

int i,a1,a3,a2;

main()
{

  cin>>s;
   
   for (i=0;i<s.size();i++)
    {
      if (s[i]>='A' && s[i]<='Z') a1=1;
      if (s[i]>='a' && s[i]<='z') a2=1;
      if (s[i]>='0' && s[i]<='9') a3=1;
    }
    
    if (a1+a2+a3==3 && s.size()>=5) cout<<"Correct"<<endl; else
    cout<<"Too weak"<<endl;
}
