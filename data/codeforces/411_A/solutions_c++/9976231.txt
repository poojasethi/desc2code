#include<iostream>
#include<string>
using namespace std;
string s;
int a,b,c,flag;
int main()
{
	cin>>s;
	if(s.length()>=5) flag=1;
	for(int i=0;i<s.length();i++)
	{
		if(s[i]>='a'&&s[i]<='z') a=1;
		if(s[i]>='A'&&s[i]<='Z') b=1;
		if(s[i]>='0'&&s[i]<='9') c=1;
	}
	if(a&&b&&c&&flag) cout<<"Correct"<<endl;
	else cout<<"Too weak"<<endl;
	return 0;
}
