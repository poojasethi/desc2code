#include<iostream>
using namespace std;
bool f,la,sm,d;
int main() {
	char s[101];
	cin>>s;
	for(int i=0;s[i]!='\0';i++) {
		if(i==4)f=true;
		if('A'<=s[i]&&s[i]<='Z')la=true;
		if('a'<=s[i]&&s[i]<='z')sm=true;
		if('0'<=s[i]&&s[i]<='9')d=true;
	}
	if(f&&la&&sm&&d)cout<<"Correct";
	else cout<<"Too weak";
	return 0;
}
