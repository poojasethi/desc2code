#include <iostream>
#include <string>
using namespace std; 

string s;
bool L, S, D, Z;

int main ( ) {
cin >>s;
L=S=D=Z=1;
if(s.length()>=5)
Z=0;
for(int i = 0; i < s.length(); i++)
if(s[i]<='Z' && s[i]>='A')
L=0;
else if(s[i]<='z' && s[i]>='a')
S=0;
else if(s[i]<='9' && s[i]>='0')
D=0;
if(L+D+S+Z> 0)
cout <<"Too weak";
else
cout <<"Correct";
}
