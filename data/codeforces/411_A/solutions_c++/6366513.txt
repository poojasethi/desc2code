#include<iostream>
#include<string>
#include<cstdio>
using namespace std;
int main()
{
	string s;
	cin>>s;
	bool f=false,f1=false,f2=false;
	for(int i=0;i<s.size();i++)
	if(s[i]>='a' && s[i]<='z') f=true; else if(s[i]>='A' && s[i]<='Z') f1=true; else if(s[i]>='0' && s[i]<='9') f2=true;
	if(f && f1  && f2 && s.size()>=5) cout<<"Correct"; else cout<<"Too weak";
	}
