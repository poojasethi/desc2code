#include <bits/stdc++.h>
using namespace std;
bool a=0,b=0,c=0,d=0;
int cnt=0;
void visit(char k)
{
    if('a'<=k&&k<='z') a=1;
    if('A'<=k&&k<='Z') b=1;
    if('0'<=k&&k<='9') c=1;
    cnt++;
    if(cnt>=5)d=1;
}
int main()
{
    string s;
    cin>>s;
    for_each(s.begin(),s.end(),visit);
    if(a&&b&&c&&d) cout<<"Correct"; else cout<<"Too weak";
}
