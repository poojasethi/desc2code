#include<cstdio>
#include<cstring>
int main()
{
    int n,i;
    char s[105];
    int y1=0,y2=0,y3=0;
    scanf("%s",s);
    n=strlen(s);
    for(i=0;i<n;++i){
        if(s[i]>='a'&&s[i]<='z')y1=1;
        if(s[i]>='A'&&s[i]<='Z')y2=1;
        if(s[i]>='0'&&s[i]<='9')y3=1;
    }
    if(y1&&y2&&y3&&n>=5)puts("Correct");
    else puts("Too weak");
    return 0;
}
