#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    int x=0;
    char c;
    int u=0,v=0,w=0;
    while((c=getchar())!='\n' && c!=EOF){
    x++;
    if(c>='a' && c<='z')u=1;
    if(c>='A' && c<='Z')v=1;
    if(c>='0' && c<='9')w=1;
    }
    if(x>=5 && u && v && w)cout<<"Correct\n";
    else cout<<"Too weak\n";
    return 0;
}
