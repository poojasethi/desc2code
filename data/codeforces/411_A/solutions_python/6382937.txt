line = raw_input()

digit = 0
lower = 0
upper = 0

for i in range(len(line)):
    if line[i].isdigit():
        digit += 1
    elif line[i].islower():
        lower += 1
    elif line[i].isupper():
        upper += 1

if digit and lower and upper and len(line) >= 5:
    print "Correct"
else:
    print "Too weak"
