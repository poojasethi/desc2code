a = raw_input()
L,S,D,F=[0,0,0,0]
m = ['!', '?', '.', ',', '_']
for t in a:
	if t >= '0' and t<= '9':
		D = D + 1
	if t >= 'a' and t<= 'z':
		S = S + 1
	if t >= 'A' and t<= 'Z':
		L = L + 1
	if t in m:
		F = F + 1
if D + S + L + F >= 5 and D > 0 and S > 0 and L > 0:
	print "Correct"
else:
	print "Too weak"
