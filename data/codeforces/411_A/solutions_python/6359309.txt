pw = raw_input()

small, large, digit = (False,)*3
if len(pw)>=5:
    for x in pw:
        if x>='0' and x<='9': 
            digit=True
        elif x>='A' and x<='Z': 
            large=True
        elif x>='a' and x<='z':
            small=True
            
if (small, large, digit)==(True,)*3:
    print "Correct"
else :
    print "Too weak"
