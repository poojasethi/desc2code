def solve():
	pwd = raw_input()
	if len(pwd) >= 5:
		large = False
		small = False
		digit = False
		for ch in pwd:
			if ch.islower():
				small = True
			if ch.isupper():
				large = True
			if ch.isdigit():
				digit = True
			if small and large and digit:
				print "Correct"
				return
	print "Too weak"

if __name__ == "__main__":
	solve()
