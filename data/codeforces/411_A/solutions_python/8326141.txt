def main():
	line = raw_input()
	a = [False, False, False]
	if len(line) < 5:
		print "Too weak"
		return 
	for ch in line:
		if ch>='1' and ch<='9':
			a[0] = True
		if ch>='a' and ch<='z':
			a[1] = True
		if ch>='A' and ch<='Z':
			a[2] = True
	if a[0] and a[1] and a[2]:
		print "Correct"
	else:
		print "Too weak"
main()
