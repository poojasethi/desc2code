def solve():
    line = raw_input()
    upper = False
    lower = False
    digit = False
    for i in line:
        upper |= i.isupper()
        lower |= i.islower()
        digit |= i.isdigit()
    if len(line) >= 5 and upper and lower and digit:
        print "Correct"
    else:
        print "Too weak"

solve()
