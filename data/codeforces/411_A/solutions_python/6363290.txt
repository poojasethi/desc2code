def correct(s):
  if len(s) < 5:
    return False
  d = lo = ca = False
  for c in s:
    if c in '0123456789':
      d = True
    elif ord(c) >= 65 and ord(c) <= 90:
      ca = True
    elif ord(c) >= 97 and ord(c) <= 122:
      lo = True
  return (d and lo and ca)

s = raw_input().strip()
if correct(s):
  print 'Correct'
else:
  print 'Too weak'
