import sys

s = str(raw_input())
q = [0] * 3

if len(s) < 5:
    print "Too weak"
    sys.exit()
else:
    for i in range(len(s)):
        if s[i].isupper():
            q[0] = 1
        elif s[i].islower():
            q[1] = 1
        elif s[i].isdigit():
            q[2] = 1

if sum(q) == 3:
    print "Correct"
else:
    print "Too weak"
