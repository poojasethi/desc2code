s = raw_input()

def good():
  print 'Correct'
  exit()
def bad():
  print 'Too weak'
  exit()

if len(s) < 5:
  bad()

import string

if all(map(lambda c: c not in s, string.ascii_lowercase)):
  bad()
if all(map(lambda c: c not in s, string.ascii_uppercase)):
  bad()
if all(map(lambda c: c not in s, string.digits)):
  bad()
good()
