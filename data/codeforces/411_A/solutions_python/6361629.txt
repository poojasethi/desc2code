
s = raw_input()

big = False
little = False
digit = False
l = len(s) >= 5

for si in s:
    if (si.isalpha() and si.upper() == si):
        big = True
    if (si.isalpha() and si.lower() == si):
        little = True
    if (si.isdigit()):
        digit = True

if (big and little and digit and l):
    print 'Correct'
else:
    print 'Too weak'
