password = raw_input().strip()
have_uppercase = False
have_lowercase = False
have_digits = False
for ch in password:
	if (ch.isalpha()):
		if (ch.isupper()):
			have_uppercase = True
		if (ch.islower()):
			have_lowercase = True
	elif (ch.isdigit()):
		have_digits = True
if (len(password) >= 5) and (have_uppercase) and (have_lowercase) and(have_digits):
	print "Correct"
else:
	print "Too weak"
