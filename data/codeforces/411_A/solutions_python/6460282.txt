string = raw_input()
lists = list(string)
f1=False
f2=False
f3=False
for i in lists:
    if i.isdigit() == True:
        f1 = True
    if i.isupper() == True:
        f2 = True
    if i.islower() == True:
        f3 = True
if len(lists) >= 5 and f1 and f2 and f3:
    print "Correct"
else :
    print "Too weak"
