s = raw_input()
f1 = True if len(s) >= 5 else False
f2 = any(map(lambda a: ord('a') <= ord(a) <= ord('z'), s))
f3 = any(map(lambda a: ord('A') <= ord(a) <= ord('Z'), s))
f4 = any(map(lambda a: ord('0') <= ord(a) <= ord('9'), s))

if all([f1, f2, f3, f4]):
    print "Correct"
else:
    print "Too weak"
