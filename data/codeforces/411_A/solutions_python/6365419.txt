def isCorrect(password):
    if len(password) < 5:
        return False
    for seg in [('A', 'Z'), ('a', 'z'), ('1', '9')]:
        if sum([1 if ord(c) >= ord(seg[0]) and ord(c) <= ord(seg[1]) else 0 for c in password]) == 0:
            return False
    return True

print 'Correct' if isCorrect(raw_input()) else 'Too weak'
