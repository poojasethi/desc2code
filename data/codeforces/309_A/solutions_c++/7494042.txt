#include<iostream>
#include<iomanip>
using namespace std;
double y;
int n,l,t,x,j,i,a[30000000];

int main() 
{
  cin>>n>>l>>t;
  for (i=0; i<n; i++) 
  {
    cin>>a[i];
    a[n+i]=a[i]+l;
  }
  t=t*2; 
  x=t/l; 
  t=t%l;
  for (i=0; i<n; i++) 
  {
    for (; j<2*n && a[j]-a[i]<=t; j++);
    {
    y+=j-i-1;
    }
  }
  cout<<fixed;
  cout<<setprecision(8)<<0.25*(y+double(x)*double(n)*(n-1));
  return 0;
  
}
