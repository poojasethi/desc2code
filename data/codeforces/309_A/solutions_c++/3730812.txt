#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,a[2100000];
int len,tim;
int main()
{
    scanf("%d %d %d",&n,&len,&tim);
    for(int i=0;i<n;++i)
    {
        scanf("%d",&a[i]);
        a[i+n]=a[i]+len;
    }
    int k=2*tim%len;
    double ans=0;
    for(int i=0;i<n;++i)
    {
        int tmp=a[i]+k+1;
        int pos=lower_bound(a,a+2*n,tmp)-a;
        ans+=(pos-i-1.0)*0.25;
    }
    int kk=tim*2/len;
    ans+=(kk+0.0)*(n+0.0)*(n-1.0)*0.25;
    printf("%.10lf\n",ans);
    return 0;
}
