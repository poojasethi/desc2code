#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m;
int loop=0;
int par[55];
int deg[55];
int maxd=0;
int find(int x)
{
    if(par[x]==x)
        return x;
    else    par[x]=find(par[x]);
}
int main()
{
    memset(deg,0,sizeof(deg));
    scanf("%d %d",&n,&m);
    for(int i=1;i<=n;i++)
        par[i]=i;
    for(int i=1;i<=m;i++)
    {
        int a,b;
        scanf("%d %d",&a,&b);
        if(find(a)==find(b))    loop++;
        else    par[find(a)]=find(b);
        deg[a]++;
        deg[b]++;
        maxd=max(maxd,max(deg[a],deg[b]));
    }
    if(maxd>2)
    {
        printf("NO\n");
        return 0;
    }
    if(loop>0)
    {
        if(loop==1&&maxd==2&&n==m)
            printf("YES\n0\n");
        else 
        {
            printf("NO\n");

        }
        return 0;
    }
    printf("YES\n");
    printf("%d\n",n-m);
    for (int i=1;i<=n-1;i++)
        for (int j=i+1;j<=n;j++)
            if (deg[i]<2 && deg[j]<2 && find(i)!=find(j))
            {
                deg[i]++;
                deg[j]++;
                par[find(i)]=find(j);
                printf("%d %d\n",i,j);
            }
    for (int i=1;i<=n;i++)
        for (int j=n;j>=i;j--)
            if (deg[i]<2 && deg[j]<2)
            {
                deg[i]=deg[j]=2;
                par[find(i)]=find(j);
                printf("%d %d\n",i,j);
            }
    return 0;
}
