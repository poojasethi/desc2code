#include <bits/stdc++.h>
using namespace std;
int n,m;
int root[55],deg[55];
int getroot(int u){
    if(u == root[u]) return u;
    return root[u] = getroot(root[u]);
}
int main(){
    //freopen("A.inp","r",stdin);
    scanf("%d%d",&n,&m);
    for(int i = 0;i < n;++i) root[i] = i;
    bool check = 0;
    for(int i = 1;i <= m;++i){
        int u,v;
        scanf("%d%d",&u,&v);
        u--;v--;
        deg[u]++;deg[v]++;
        int ru = getroot(u),rv =getroot(v);
        if((ru == rv) && ( (i != n ) || (i != m) )) check = 1;
        else root[ru] = rv;
        if(deg[u] > 2 || deg[v] > 2) check = 1;
    }
    if(check){
        puts("NO");
    }
    else{
        puts("YES");
        printf("%d\n",n-m);
        for(int i = 0;i < n;++i)
            for(int j = i+1;j < n;++j)
        if(deg[i] < 2 && deg[j] < 2){
            int ru = getroot(i),rv = getroot(j);
            if(ru != rv){
                printf("%d %d\n",i+1,j+1);
                deg[i]++;deg[j]++;
                root[ru] = rv;
            }
        }
        for(int i = 0;i < n;++i)
            while(deg[i] < 2) printf("%d ",i+1),deg[i]++;
    }
return 0;
}
