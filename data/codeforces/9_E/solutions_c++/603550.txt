#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>

using namespace std;

int n,m,d[100],x,y,ans,res[200];
bool flag;
int fa[100];

int gf(int x){return fa[x]<0?x:fa[x]=gf(fa[x]);}

int main(){
	cin>>n>>m;
	memset(fa, 255, sizeof(fa));
	for (int i=0; i<m; i++){
		cin>>x>>y;
		d[x]++; d[y]++;
		if (d[x]>2 || d[y]>2) flag=true;
		if (x==y && n!=1)	flag=true;
		int a=gf(x), b=gf(y);
		if (n!=m && a==b)	flag=true;
		if (a!=b)	fa[a]=b;
	}
	for (int i=1; i<=n; i++) ans+=2-d[i];
	if (flag || ans%2==1 || m>n){cout<<"NO"<<endl; return 0;}
	int all=ans; ans=0;
	bool tot=false;
	for (int i=1; i<=n; i++){
		for (;d[i]<2;){
			res[++ans]=i; d[i]++; flag=false;
			for (int j=i; j<=n; j++){
				if (d[j]==2)	continue;
				int a=gf(i), b=gf(j);
				if (a!=b){
					fa[a]=b;
					res[++ans]=j; d[j]++;
					flag=true;
					break;
				}
			}
			if (!flag && tot){cout<<"NO"<<endl; return 0;}
			if (flag)	continue;
			tot=true;
			for (int j=i; j<=n; j++){
				if (d[j]!=2){
					res[++ans]=j; d[j]++;
					break;
				}
			}
		}
	}
	cout<<"YES"<<endl<<ans/2<<endl;
	for (int i=1; i<=ans; i+=2)
		cout<<res[i]<<' '<<res[i+1]<<endl;
}
