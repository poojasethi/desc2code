#include<cstdio>
#include<algorithm>
#include<iostream>
#define fr(i,n) for (int i=0;i<n;i++)
using namespace std;
#define N 50
#define x first
#define y second
int a,b,n,m,f[N],d[N],t,p;
pair<int,int> c[N],e[N];
int F(int x){return f[x]==x?x:f[x]=F(f[x]);}
int main(){
cin>>n>>m;
fr(i,n)f[i]=i;
fr(i,m)cin>>a>>b,a--,b--,d[a]++,d[b]++,f[F(a)]=F(b);
if(n==1||m>n)return puts((m==1)?"YES\n0":!m?"YES\n1\n1 1":"NO"),0;
fr(i,n)if(d[i]>2)return puts("NO"),0;
fr(i,n)if(f[i]==i)fr(j,n)if(F(j)==i&&d[j]==1)
{fr(k,n)if(k!=j&&F(k)==i&&d[k]==1) c[t++]=make_pair(j,k);break;}
fr(i,n)if(!d[i])c[t++]=make_pair(i,i);
while(t>1){sort(c,c+t),e[p++]=make_pair(c[0].x,c[1].x);f[F(c[0].x)]=c[1].x,c[0].x=c[1].y,c[1]=c[--t];if (c[0].x>c[0].y) swap(c[0].x,c[0].y);}
if(t)e[p++]=make_pair(c[0].x,c[0].y),f[F(c[0].x)]=F(c[0].y);
fr(i,n)if(F(i)!=F(0))return puts("NO"),0;
printf("YES\n%d\n",p);
fr(i,p)printf("%d %d\n",e[i].x+1,e[i].y+1);
}
