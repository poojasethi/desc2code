#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<iostream>
#define MAXN 2550
using namespace std;
int n,m,deg[MAXN],uu[MAXN],vv[MAXN];
int cnt,a[MAXN],b[MAXN];
int pre[MAXN];
int find(int u)
{
	if(pre[u]^u)
		pre[u]=find(pre[u]);
	return pre[u];
}
void solve()
{
	int lv=n;
	cnt=0;
	for(int i=1;i<=n;i++)
		pre[i]=i;
	if(n==1){
		if(m>1){
			puts("NO");
			return;
		}
		if(!m){
			a[cnt]=b[cnt]=1;
			cnt++;
		}
		puts("YES");
		printf("%d\n",cnt);
		for(int i=0;i<cnt;i++)
			printf("%d %d\n",a[i],b[i]);
		return;
	}
	memset(deg,0,sizeof(deg));
	for(int i=0;i<m;i++){
		if(uu[i]==vv[i]){
			puts("NO");
			return;
		}
		deg[uu[i]]++;
		deg[vv[i]]++;
		if(deg[uu[i]]>2||deg[vv[i]]>2){
			puts("NO");
			return;
		}
		int u=find(uu[i]);
		int v=find(vv[i]);
		if(u^v){
			pre[u]=v;
			lv--;
		}
	}
	while(1){
		int u=-1,v=-1;
		for(int i=1;i<=n;i++)
			if(deg[i]<2){
				u=i;
				break;
			}
		if(u==-1)
			break;
		for(int i=u+1;i<=n;i++){
			if(deg[i]<2){
				if(lv==1||find(i)!=find(u)){
					v=i;
					break;
				}
			}
		}
		if(v==-1){
			puts("NO");
			return;
		}
		deg[u]++;
		deg[v]++;
		a[cnt]=u;
		b[cnt++]=v;
		if(find(u)!=find(v)){
			lv--;
			pre[find(u)]=find(v);
		}
	}
	if(lv>1){
		puts("NO");
		return;
	}
	puts("YES");
	printf("%d\n",cnt);
	for(int i=0;i<cnt;i++)
		printf("%d %d\n",a[i],b[i]);
}
int main()
{
	//freopen("input.txt","r",stdin);
	while(scanf("%d%d",&n,&m)==2){
		for(int i=0;i<m;i++)
			scanf("%d%d",uu+i,vv+i);
		solve();
	}
	return 0;
}
