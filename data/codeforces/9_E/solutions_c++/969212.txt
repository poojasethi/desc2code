#include <iostream>
#include <queue>
using namespace std;
#define FOR(i,a,b) for(i=a;i<=b;i++)

int f[60],d[60];

int find(int x){
        if (f[x]==x) return x;return f[x]=find(f[x]);
}

int main(){
        int n,m,i,j;
        bool fail;
        cin>>n>>m;
        FOR(i,1,n) f[i]=i;
        fail=false;
        FOR(i,1,m){
                int x,y;
                cin>>x>>y;
                if (++d[x]>2) fail=true;
                if (++d[y]>2) fail=true;
                if (x==y&&n!=1) fail=true;
                if (find(x)==find(y)&&n!=m) fail=true;
                f[find(x)]=find(y);
        }
        if (fail) {
                cout<<"NO"<<endl;
                return 0;
        }
        if (n==1){
                cout<<"YES"<<endl<<1-m<<endl;
                if (m==0) cout<<"1 1"<<endl;
                return 0;
        }
        queue<int> ans;
        FOR(i,1,n){
                //cout<<d[i]<<endl;
                while(d[i]<2){
                        FOR(j,i+1,n) if (d[j]<2&&(find(i)!=find(j)||ans.size()/2+m+1==n)){
                                ans.push(i);ans.push(j);
                                d[i]++;d[j]++;
                                f[find(i)]=find(j);
                                break;
                        }
                }
        }
        cout<<"YES"<<endl<<ans.size()/2<<endl;
        while (!ans.empty()){
                cout<<ans.front()<<' ';ans.pop();
                cout<<ans.front()<<endl;ans.pop();
        }
}
