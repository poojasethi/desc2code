#include<iostream>
#define ll long long
using namespace std;
ll sum1,sum2,H,d[500][500],N,p[100005],n,m,a[100005];
int pa(int x)
{
    return (p[x]==x?x:p[x]=pa(p[x]));
}
main()
{
    cin>>n>>m;
    if(m>n){cout<<"NO";return 0;}
    for(int i=1;i<=n;i++)p[i]=i;
    for(int i=0;i<m;i++)
    {
        int x,y;
        cin>>x>>y;
        a[x]++;
        a[y]++;
        if(pa(x)==pa(y))
        {
            if(i==n-1)
            {
                cout<<"YES"<<endl<<0;
                return 0;               
            }else
            {cout<<"NO";return 0;}
        }
        p[pa(y)]=x;
    }
    for(int i=1;i<=n;i++)if(a[i]>2){cout<<"NO";return 0;}
    
     cout<<"YES"<<endl;
     cout<<n-m<<endl;
     
     for(int i=1;i<=n;i++)
     {
        if(a[i]<2)
        {
            for(int j=i+1-(n==1);j<=n;j++)
            {
                if(a[j]<2)
                {
                    if((pa(i)!=pa(j)) || (m+1==n))
                    {
                        cout<<i<<" "<<j<<endl;
                        a[i]++;
                        a[j]++;
                        m++;
                        p[pa(j)]=i;
                        i--;
                        break;
                    }
                }
            }
        }
     }
}
