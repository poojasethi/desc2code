#include<cstdio>
using namespace std;
const int N = 100010;
int n,pos,a[N];
char s[N];
bool vis[N];
int main(){
	scanf("%d",&n);
	scanf("%s",s+1);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	pos=1;
	while(1){
		vis[pos]=1;
		if(s[pos]=='>') pos=pos+a[pos];
		else pos=pos-a[pos];
		if(pos<1||pos>n) {printf("FINITE\n");return 0;}
		else if(vis[pos]) {printf("INFINITE\n");return 0;}
	}
	return 0;
}
