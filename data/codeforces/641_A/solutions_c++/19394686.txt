#include<cstdio>
int n,i,a[110000],f[110000];
char s[110000];
int main(){
	scanf("%d%s",&n,s+1);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(i=1;i>0&&i<=n;){
		if(f[i]){
			printf("INFINITE");
			return 0;
		}
		f[i]=1;
		if(s[i]=='>') i+=a[i];
		else          i-=a[i];
	}
	printf("FINITE");
	return 0;
}
