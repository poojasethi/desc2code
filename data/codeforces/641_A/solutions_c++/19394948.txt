#include<cstdio>
#define N 100005
int n,a[N],v[N];
char s[N];
int main(){
    scanf("%d%s",&n,s);
    for(int i=0;i<n;i++)scanf("%d",&a[i]);
    for(int i=0;i<n&&i>=0;)
    {
        if(!v[i]){
            v[i]=1;
            if(s[i]=='>')
            i+=a[i];
            else i-=a[i];
        }else{
            puts("INFINITE");
            return 0;
        }
    }
    puts("FINITE");

}
