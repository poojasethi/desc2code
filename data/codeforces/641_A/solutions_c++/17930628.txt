#include<bits/stdc++.h>
using namespace std;
long long n,i,j,x[1000009],ans,sum;
string s;
main()
{
	cin>>n;
	cin>>s;
	ans=0;
	for(i=0;i<=n;i++)
	{
		cin>>x[i];
	}
	for(i=0;i<=n;i++)
{
	
		if(s[ans]=='>')ans+=x[ans];
        else
        ans-=x[ans];
        if(ans<0 || ans>=n){
        	cout<<"FINITE"; return 0;
		}
	}
	cout<<"INFINITE";
	return 0;	
}
