#include <iostream>
#include <string>
using namespace std;
int x,a[200000],i,b[200000];
string s;
int main(){
    cin>>x>>s;
	a[1]=1;
	for(i=1;i<=x;i++){
		cin>>b[i];
		if(s[i-1]=='<')b[i]*=-1;
		if(i+b[i]<=x && i+b[i]>0){
			a[i+b[i]]=1;
		}
	}
	for(i=1;i<=x;i++){
		if(a[i]==1 && (i+b[i]>x || i+b[i]<=0)){
			cout<<"FINITE";
			return 0;
		}
	}
	cout<<"INFINITE";
    return 0;
}
