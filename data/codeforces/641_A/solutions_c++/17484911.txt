#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
  int n;
  string s;
  cin >> n >> s;
  vector<int> a(n), used(n, 0);
  for (int i = 0; i < n; ++i) cin >> a[i];
  int cur = 0;
  while (cur >= 0 && cur < n && !used[cur]) {
    used[cur] = 1;
    cur += (s[cur] == '>' ? 1 : -1) * a[cur];
  }
  if (cur < n && cur >= 0 && used[cur]) cout << "INFINITE" << endl;
  else cout << "FINITE" << endl;
  return 0;
}
