#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
int n,cur=1,step[100010],op[100010];bool vis[100010];char s[100010];
int main(){
	scanf("%d%s",&n,s+1);for(int i=1;i<=n;++i)op[i]=s[i]=='>'?1:-1;
	for(int i=1;i<=n;++i)scanf("%d",&step[i]);
	while(1){
		vis[cur]=1;
		cur+=op[cur]*step[cur];
		if(cur<1||cur>n)return puts("FINITE"),0;
		else if(vis[cur])return puts("INFINITE"),0;
	}
}
