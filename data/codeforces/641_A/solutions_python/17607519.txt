n = int(raw_input())
direction = raw_input()
distance = map(int,raw_input().split())

trace = []

for i in range(0,n):
    trace.append(0)

current = 0

def move(current):
    trace[current] = 1
    if (direction[current]=="<") :
        return current - distance[current]
    else:
        return current + distance[current]
while(True):
    current = move(current)
    if ((current<0) or (current >=n)):
        print "FINITE"
        break
    if ((trace[current]==1)):
        print "INFINITE"
        break
