n = int(raw_input())
dir = raw_input()
dis = map(int, raw_input().split())

vis = set([0])
cur = 0

while(1) :
	if (dir[cur] == '<'):
		cur -= dis[cur]
	else:
		cur += dis[cur]
	if (cur < 0 or cur >= n):
		print 'FINITE'
		break
	else:
		if (cur in vis):
			print 'INFINITE'
			break
		else:
			vis.add(cur)
