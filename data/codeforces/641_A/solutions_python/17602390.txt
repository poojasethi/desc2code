n = int(raw_input())
direction = raw_input()
distance = map(int,raw_input().split())

left = []
right = []

for i in range(0,n):
    left.append(0)
    right.append(0)

check = True
current = 0

def move(current):
    if (direction[current]=="<") :
        left[current] = 1
        return current - distance[current]
    else:
        right[current] = 1
        return current + distance[current]
while(check):
    current = move(current)
    if ((current<0) or (current >=n)):
        print "FINITE"
        break
    if ((left[current]==1) or (right[current]==1)):
        print "INFINITE"
        break
