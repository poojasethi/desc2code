#include<cstdio>
#include<vector>
using namespace std;

const int MAX_N = 1e5 + 5;

int na, nb, nc;
int a[MAX_N];
int b[MAX_N];
int c[MAX_N << 1];

inline int solve(int x, bool print = false){
  nc = 0;
  for(int ia = 0, ib = 0; ia < na or ib < nb; x ^= 1){
    for(; ia < na and x == a[ia]; ia++){
      c[nc++] = a[ia];
      print? printf("%d ", ia + 1): int();
    }
    for(; ib < nb and x == b[ib]; ib++){
      c[nc++] = b[ib];
      print? printf("%d ", ib + na + 1): int();
    }
  }
  int cnt = 0;
  for(int i = 0; i < nc; i++){
    cnt += c[i] != c[i + 1];
  }
  return cnt;
}

int main(){
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);

  scanf("%d", &na);
  for(int i = 0; i < na; scanf("%d", &a[i++]));
  scanf("%d", &nb);
  for(int i = 0; i < nb; scanf("%d", &b[i++]));
  
  printf("\n%d\n", solve(0) < solve(1)? solve(0, true): solve(1, true));
  for(int i = 0; i < nc; i++){
    if(c[i] != c[i + 1]){
      printf("%d ", i + 1);
    }
  }
}
