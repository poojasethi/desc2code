#include<cstdio>

int n, m, id;
int a[222222];
int ans[2], ret[2][222222], op[2][2222222];

void work(int cur, int p1, int p2, int f) {
	int p = 1, v = a[cur];
	ans[f] = 0;
	ret[f][p] = cur;
	while(p1 <= n || p2 <= n+m) {
		while(p1 <= n && a[p1] == v) ret[f][++p] = p1++;
		while(p2 <= n+m && a[p2] == v) ret[f][++p] = p2++;
		v = 1-v;
		ans[f] ++;
		op[f][ans[f]] = p;
	}
	if(v==1) ans[f] --;
}

int main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	
	scanf("%d", &n);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	scanf("%d", &m);
	for(int i=1;i<=m;i++) scanf("%d", &a[i+n]);
	
	work(1, 2, n+1, 0);
	work(n+1, 1, n+2, 1);
	
	if(ans[0] > ans[1]) id = 1;
	else id=0;
	for(int i=1;i<=n+m;i++)
		printf("%d%c", ret[id][i], i<n+m ? ' ' : '\n');
	printf("%d\n", ans[id]);
	for(int i=1;i<=ans[id];i++)
		printf("%d%c", op[id][i], i<ans[id] ? ' ' : '\n');
	
	return 0;
}
