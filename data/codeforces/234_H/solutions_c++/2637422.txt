#include<cstdio>
#include<cstring>
#include<vector>
#define maxn 100005
using namespace std;

int     n,m,data[2][maxn];
vector<int> que[2],ans[2];

void    solve(int k ) 
{
    int i = 0 , j = 0, now = k;
    while ( i<n || j<m ){
        while ( i<n && data[0][i] == now ) {
            que[k].push_back( (i++)+1 );
        }
        while ( j<m && data[1][j] == now ) que[k].push_back( n+(j++)+1 );
        if ( i==n && j == m && now == 0 ) break;
        ans[k].push_back( que[k].size() );
        now ^= 1;
    }
}

int     main(){
    freopen( "input.txt" , "r" , stdin );
    freopen( "output.txt" , "w" , stdout );
    scanf("%d", &n );
    for (int i = 0; i<n; i++) scanf("%d", &data[0][i] );
    scanf("%d", &m );
    for (int i = 0; i<m; i++) scanf("%d", &data[1][i] );
    solve( 0 );
    solve( 1 );
    int k;
    if ( ans[0].size() <= ans[1].size() ) k = 0;
    else k = 1; 
    for (int i = 0; i<n+m; i++) printf("%d%c",  que[k][i] , i==n+m-1?'\n':' ');
    printf("%d\n" , (int)ans[k].size() );
    for (int i = 0; i<ans[k].size(); i++)
        printf("%d%c", ans[k][i] , i == ans[k].size()-1?'\n':' ' );
    return 0;
}
