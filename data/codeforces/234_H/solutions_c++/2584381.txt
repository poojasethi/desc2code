#include <cstdio>
using namespace std;

int m, n, mn, a[100005], b[100005], s[200005][2], sol[200005], ans;

void pre() {
	int i = m - 1, j = n - 1;
	for (int k=0, l=mn-1; (i>=0 || j>=0); k^=1) {
		while (i >= 0 && a[i] == k) {
			s[l][0] = i + 1;
			s[l][1] = k;
			--l;
			--i;
		}
		while (j >= 0 && b[j] == k) {
			s[l][0] = m + j + 1;
			s[l][1] = k;
			--l;
			--j;
		}
	}
	s[mn][1] = 0;
}

int main() {
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d", &m);
	for (int i=0; i<m; ++i) scanf("%d", &a[i]);
	scanf("%d", &n);
	for (int i=0; i<n; ++i) scanf("%d", &b[i]);
	mn = m + n;
	pre();
	for (int i=0; i<mn; ++i) printf("%d%c", s[i][0], i==mn-1? '\n': ' ');
	ans = 0;
	for (int i=0; i<mn; ++i) {
		if (s[i][1] != s[i+1][1]) {
			sol[ans++] = i+1;
		}
	}
	printf("%d\n", ans);
	for (int i=0; i<ans; ++i) {
		printf("%d%c", sol[i], i==ans-1? '\n': ' ');
	}
	return 0;
}
