#include <iostream>
#include <cstring>
#include <cstdio>

using namespace std;

bool deck1[100001],deck2[100001];

int ans1[100001 << 1];
int ans2[100001];

int n,m;

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int cnt1,cnt2=1;
//	memset(ans1,-1,sizeof(ans1));
//	memset(ans2,-1,sizeof(ans2));
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>deck1[i];
	cin>>m;
	cnt1=n+m;
	for(int i=1;i<=m;i++)
		cin>>deck2[i];
	int i;
	for(i=n;i>=1;i--)
	{
		if(deck1[i]==0) ans1[cnt1--]=i;
		else break;
	}
	int j;
	for(j=m;j>=1;j--)
	{
		if(deck2[j]==0) ans1[cnt1--]=j+n;
		else break;
	}
//	for(int k=1;k<=n+m;k++)
//		cout<<ans1[k]<<" ";
//	cout<<endl;
	int k;
	int tmp=1;
	while(cnt1>0)
	{
		if(i>0 && deck1[i]==tmp)
		{
			ans1[cnt1--]=i--;
			ans2[cnt2]++;
		}
		else if(j>0 && deck2[j]==tmp)
		{
			ans1[cnt1--]=j+n,j--;
			ans2[cnt2]++;
		}
		else
		{
			tmp=(tmp+1)%2;
			cnt2++;
		}
	}

	for(int k=1;k<=n+m;k++)
		cout<<ans1[k]<<" ";
	cout<<endl<<cnt2<<endl;
	int t=0;
	for(int k=cnt2;k>=1;k--)
	{
		t+=ans2[k];
		cout<<t<<" ";
	}
	cout<<endl;
	return 0;
}
