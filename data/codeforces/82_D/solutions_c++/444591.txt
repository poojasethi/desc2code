#include <iostream>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[1000];
    for(int i=0;i<n;i++)cin>>a[i];
    bool added=false;
    if(n%2){a[n++]=0;added=true;}
    int dp[1010][1010];
    for(int i=0;i<n-1;i++)
    {
        dp[1][i]=a[n-1]>a[i]?a[n-1]:a[i];
    }

    for(int i=3;i<n;i+=2)
    {
        for(int j=0;j<n-i;j++)
        {
            dp[i][j]=(a[j]>a[n-i]?a[j]:a[n-i])+dp[i-2][n-i+1];
            int tmp=(a[j]>a[n-i+1]?a[j]:a[n-i+1])+dp[i-2][n-i];
            if(tmp<dp[i][j])dp[i][j]=tmp;
            tmp=(a[n-i+1]>a[n-i]?a[n-i+1]:a[n-i])+dp[i-2][j];
            if(tmp<dp[i][j])dp[i][j]=tmp;

        }
    }
    cout<<dp[n-1][0]<<endl;
    int j=0;
    for(int i=n-1;i>=3;i-=2)
    {
        int tmp=a[n-i]>a[n-i+1]?a[n-i]:a[n-i+1];
        if(dp[i][j]==dp[i-2][j]+tmp)
        {
            cout<<n-i+1<<' '<<n-i+1+1<<endl;
        }
        else
        {
            tmp=a[n-i+1]>a[j]?a[n-i+1]:a[j];
            if(dp[i][j]==dp[i-2][n-i]+tmp)
            {
                cout<<j+1<<' '<<n-i+1+1<<endl;
                j=n-i;
            }
            else
            {
                cout<<j+1<<' '<<n-i+1<<endl;
                j=n-i+1;

            }
        }

    }
    cout<<j+1;
    if(!added)cout<<' '<<n;
    return 0;
}
