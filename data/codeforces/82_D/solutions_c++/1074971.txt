#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int dp[1001][1001],pre[1001][1001];
int tim[1000];
int n;
int calc(int a,int b)
{
    if (dp[a][b])
        return dp[a][b];
    int &ret=dp[a][b];
    if (b==n)
        return ret=tim[a];
    if (b==n-1)
        return ret=max(tim[a],tim[n-1]);
    ret=calc(a,b+2)+max(tim[b],tim[b+1]);
    pre[a][b]=a;
    if (ret>calc(b+1,b+2)+max(tim[a],tim[b]))
    {
        ret=calc(b+1,b+2)+max(tim[a],tim[b]);
        pre[a][b]=b+1;
    }
    if (ret>calc(b,b+2)+max(tim[a],tim[b+1]))
    {
        ret=calc(b,b+2)+max(tim[a],tim[b+1]);
        pre[a][b]=b;
    }
    return ret;
}
void print(int a,int b)
{
    if (b==n)
        printf("%d\n",a+1);
    else if (pre[a][b]==-1)
    {
        printf("%d %d\n",a+1,b+1);
    }
    else if (pre[a][b]==a)
    {
        printf("%d %d\n",b+1,b+2);
        print(a,b+2);
    }
    else if (pre[a][b]==b+1)
    {
        printf("%d %d\n",a+1,b+1);
        print(b+1,b+2);
    }
    else
    {
        printf("%d %d\n",a+1,b+2);
        print(b,b+2);
    }
}
int main()
{
    memset(dp,0,sizeof(dp));
    memset(pre,-1,sizeof(pre));
    scanf("%d",&n);
    for (int i=0;i<n;i++)
        scanf("%d",&tim[i]);
    printf("%d\n",calc(0,1));
    print(0,1);
    //system("pause");
    return 0;
}
