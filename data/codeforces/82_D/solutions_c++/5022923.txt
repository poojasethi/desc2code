#include <cstdio>
#include <climits>
#include <algorithm>
using namespace std;
int N,d[1005],ff[1105/2][10005];
int p[1005][1005];
int f(int nw,int kl)
{
	int a;
	if(ff[nw][kl])	return ff[nw][kl];
	if(nw > N/2+N%2)	return 0;
	if(N%2 && nw == N/2+N%2)
		return ff[nw][kl]=d[kl];
	int mn=INT_MAX;
	if(nw*2+1 > N)
		mn=max(d[nw*2],d[kl]);
	if(nw*2+1 <= N)
		if(mn > f(nw+1,nw*2)+max(d[nw*2+1],d[kl]))
		{
			mn=f(nw+1,nw*2)+max(d[nw*2+1],d[kl]);
			a=nw*2;

		}
	if(nw*2+1 <= N)
		if(mn >f(nw+1,nw*2+1)+max(d[nw*2],d[kl]))
		{
			mn=f(nw+1,nw*2+1)+max(d[nw*2],d[kl]);
			a=nw*2+1;
	
		}	
	if(nw*2+1 <= N)
		if(mn > f(nw+1,kl)+max(d[nw*2],d[nw*2+1]))
		{
			mn=f(nw+1,kl)+max(d[nw*2],d[nw*2+1]);
			a=kl;
		}
	p[nw][kl]=a;
	return ff[nw][kl]=mn;
}
int main()
{
	scanf("%d",&N);
	for (int i = 1; i <= N; i++)
		scanf("%d",&d[i]);
	printf("%d\n",f(1,1));
	int i=1;
	int j=1;
	while(j <= N/2+N%2)
	{
		if(N%2 && j== N/2+N%2)
		{
			printf("%d\n",i);
			break;
		}
		if(j*2+1 > N)
		{
			printf("%d %d",j*2,i);
			break;
		}

		if(p[j][i] == i)
		{
			printf("%d %d\n",j*2,j*2+1);
			j++;
		}
		else if(p[j][i] == j*2)
		{
			printf("%d %d\n",i,j*2+1);
			i=j*2;
			j++;
		}
		else if(p[j][i] == j*2+1)
		{
			printf("%d %d\n",i,j*2);
			i=j*2+1;
			j++;
		}
	}
	
	return 0;
}
