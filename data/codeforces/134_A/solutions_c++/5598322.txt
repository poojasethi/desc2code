#include <iostream>
using namespace std;
int a[200100],ans[200100],ansn=1;
int main()
{	
	int n,sum=0;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		sum+=a[i];
	}
	for(int i=1;i<=n;i++)
	{
		if((sum-a[i])/(n-1)==a[i] && (sum-a[i])%(n-1)==0)
			ans[ansn++]=i;
	}
	cout<<ansn-1<<endl;
	for(int i=1;i<ansn;i++)
		cout<<ans[i]<<" ";
	cout<<endl;
	return 0;
}
