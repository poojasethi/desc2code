#include<bits/stdc++.h>
using namespace std;
vector<int> v;
int a[1000005];
int main()
{
int n;
scanf("%d",&n);
int i,total = 0;
for(i=0;i<n;i++)
{
scanf("%d",&a[i]);
total+=a[i];
}
for(i=0;i<n;i++)
{
int cur  = total - a[i];
if((n-1)*a[i]==cur)v.push_back(i+1);
}
printf("%d\n",v.size());
for(i=0;i<v.size();i++)
printf("%d ",v[i]);
}
