#include <iostream>
using namespace std;
int main()
{
	int n,sum=0,ans=0;
	cin>>n;
	int a[n+1],b[n+1];
	for(int i=1;i<=n;i++){
		cin>>a[i];
		sum=sum+a[i];
	}
	for(int i=1;i<=n;i++){
		if(a[i]*n==sum){
			b[ans++]=i;
		}
	}
	cout<<ans<<endl;
	for(int i=0;i<ans;i++){
		cout<<b[i]<<' ';
	}
	return 0;
}
