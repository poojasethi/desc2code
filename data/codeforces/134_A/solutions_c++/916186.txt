#include <stdio.h>
int main()
{
  int m[200000],i,n,kol=0;
  __int64 s=0;
  scanf ("%i",&n);
  for (i=0;i<n;i++){
    scanf("%i",m+i);
    s+=m[i];
  }
  for (i=0;i<n;i++){
    if ((s-m[i])/(double)(n-1)==m[i]) {kol++;m[i]=1;} else {m[i]=0;};
  }
  printf("%i\n",kol);
  for (i=0;i<n;i++){
    if (m[i]) printf("%i ",i+1);
  }
  return 0;
}
