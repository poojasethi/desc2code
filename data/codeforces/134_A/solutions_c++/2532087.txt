#include <cstdio>
int n,a[2000005],ans[2000005];
int main(){
    scanf("%d",&n);
    int sum=0;
    for(int i=1; i<=n; ++i) scanf("%d",&a[i]),sum+=a[i];
    int m=0;
    for(int i=1; i<=n; ++i)
        if (a[i]*n==sum) ans[++m]=i;
    printf("%d\n",m);
    for(int i=1; i<=m; ++i) printf("%d%c",ans[i],i<m?' ':'\n');
    return 0;
}
