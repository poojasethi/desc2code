#include<stdio.h>
#include <iostream>
#include<string.h>
using namespace std;
int d[300000],e[300000];
int main(){
	int t=0,n,i,s=0,x=0;
	double o=0;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>d[i];
		t+=d[i];
	}
	o=(double)t/n;
	for(i=0;i<n;i++)
		if((double)d[i]==o){ 
			s++;
			e[x++]=i+1;
		}

	cout<<s<<endl;
	for(i=0;i<x;i++)
		cout<<e[i]<<" ";
	cout<<endl;
	return 0;
}
