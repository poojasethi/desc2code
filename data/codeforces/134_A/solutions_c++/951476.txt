#include <cstdio>
int table[300000];
int num[300000];
int main(){
  int n;
  scanf("%d",&n);
  int s = 0;
  for(int i=0;i<n;++i){
    scanf("%d",&table[i]);
    s+=table[i];
  }
  int cnt = 0;
  for(int i=0;i<n;++i){
    if(table[i]*(n-1)==s-table[i]){
      num[cnt++]=i+1;
    }
  }
  printf("%d\n",cnt);
  for(int i=0;i<cnt;++i) printf("%d ",num[i]);
  printf("\n");
  return 0;
}
