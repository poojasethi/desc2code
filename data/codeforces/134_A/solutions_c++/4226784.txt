#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int a[200005];
int main()
{
    int n,sum=0;
    cin>>n;
    for(int i=1;i<=n;++i)
        cin>>a[i],sum+=a[i];
    int ans=0;
    for(int i=1;i<=n;++i)
        if(sum==a[i]*n)++ans;
    cout<<ans<<endl;
    for(int i=1;i<=n;++i)
        if(sum==a[i]*n)cout<<i<<' ';
    cout<<endl;
}
