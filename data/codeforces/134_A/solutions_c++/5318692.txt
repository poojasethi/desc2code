#include<stdio.h>
int a[1000000],b[1000000];

int main()
{
	int n,i,s=0,m=0,z=0;
	double v;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{scanf("%d",&a[i]);
	s+=a[i];}
	for(i=0;i<n;i++)
	{
		v=1.0*(s-a[i])/(n-1);
		if(v==a[i])
		{
			m++;
			b[z]=i+1;
			z++;
		}
	}
	printf("%d\n",m);
	for(i=0;i<m;i++)
		printf("%d ",b[i]);
	printf("\n");
	return 0;
}
