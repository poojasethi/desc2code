#include <iostream>
#include <vector>
using namespace std;

int a[200001],n,i,s;
vector<int> v;

int main(){
	cin >> n;
	for(i = 0;i < n;i++){
		cin >> a[i];
		s += a[i];
	}

	for(i = 0;i < n;i++)
		if((s-a[i]) == a[i] * (n-1))
			v.push_back(i+1);

	cout << (int)v.size() << endl;
	for(i = 0;i < (int)v.size();i++)
		cout << v[i] << " ";
	return 0;
}
