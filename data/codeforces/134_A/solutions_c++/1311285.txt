#include <stdio.h>
main()
{
  int a[200100],q[200100],k=0,s=0;
  scanf("%d",&a[0]);
  for (int i=1; i<=a[0]; i++)
   {
     scanf("%d",&a[i]);
     s+=a[i];
   }
  for (int i=1; i<=a[0]; i++)
    if (a[i]==(s-a[i])*1.0/(a[0]-1))
     {
       q[k]=i;
       k++;
     }
  printf("%d\n",k);
  for (int i=0; i<k; i++)
    printf("%d ",q[i]);
  return 0;
}
