#include<iostream>
using namespace std;
int data[200010];
int main()
{
    int n,sum=0,k=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>data[i];sum+=data[i];
    }
    for(int i=0;i<n;i++)
    {
        if(data[i]*n==sum)  k++;
    }
    cout<<k<<endl;
    if(k!=0)
    for(int i=0;i<n;i++)
        if(data[i]*n==sum)  cout<<i+1<<' ';
    cout<<endl;
    return 0;
}
