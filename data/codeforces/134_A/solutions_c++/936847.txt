#include<cstdio>
#include<cstring>
using namespace std;

int val[200005];
int main(){
	int n;
	scanf("%d",&n);
	int sum=0;
	for(int i=0;i<n;i++){
		scanf("%d",&val[i]);
		sum+=val[i];
	}
	int cnt=0;
	for(int i=0;i<n;i++){
		if(val[i]*n==sum) cnt++;
	}
	printf("%d\n",cnt);
	for(int i=0;i<n;i++){
		if(val[i]*n==sum) printf("%d ",i+1);
	}
}
