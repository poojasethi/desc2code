#include <iostream>
#include <stdio.h>

using namespace std;

double a[1000000], b[1000000];
int main()
{
	int n, i, j = 0;
	cin >> n;
	double S = 0;
	for ( i = 0; i < n; i++ )
	{
		cin >> a[i];
		S += a[i];
	}
	for ( i = 0; i < n; i++ )
		if ( a[i] == ( S - a[i] ) / ( n - 1 ) )	b[j++] = i + 1;
	cout << j << endl;
	for ( i = 0; i < j; i++ )
		cout << b[i] << " "; 
	return 0;
}
