#include <stdio.h>
#include <stdlib.h>

int main()
{
	int a[200000], b[200000];
	int n, i, j=0;
	long long sum=0, t;

	scanf("%d", &n);
	for(i=0; i<n; i++) {
		scanf("%d", &a[i]);
		sum+=a[i];
	}

	for(i=0; i<n; i++) {
		t=sum-a[i];
		if(t%(n-1)==0 && t/(n-1)==a[i]) b[j++]=i+1;
	}
		printf("%d\n", j);
		for(i=0; i<j; i++) printf("%d ", b[i]);

	printf("\n");
	
	return 0;
}
