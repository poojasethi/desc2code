#include <iostream>
using namespace std;

int n,mid,s;
int a[200005];

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		mid+=a[i];
	}

	for(int i=0;i<n;i++)
		if( a[i]*n==mid )
			s++;

	cout << s << '\n';
	
	for(int i=0;i<n;i++)
		if( a[i]*n==mid )
			cout << i+1 << ' ';
	cout << '\n';

	return 0;
}
