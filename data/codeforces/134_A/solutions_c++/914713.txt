#include <iostream>
using namespace std;

int a[200000], b[200000];
double x, t;
int k, n;

int main() {
    cin >> n;
    for (int i=0; i<n; i++) { cin >> a[i]; x+=a[i]; }
    for (int i=0; i<n; i++)
        if (((x-a[i])/(n-1)) == a[i])
            b[k++]=i;
    cout << k <<endl;
    for (int i=0; i<k; i++) cout << b[i]+1 << " ";
    return 0;
}
