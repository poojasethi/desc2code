#include<cstdio>
#include<vector>
#include<numeric>
int n,a[200010],i,s;
std::vector<int> o;
int main() {
    scanf("%d",&n);
    for(i=0;i<n;scanf("%d",a+i++));
    s=std::accumulate(a,a+n,0);
    for(i=0;i<n;i++)
        if(float(s-a[i])/(n-1)==a[i])
            o.push_back(i+1);
    for(printf("%d\n",o.size()),i=0;i<o.size();printf("%d ",o[i++]));
}
