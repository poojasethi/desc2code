#include <vector>
#include <iostream>
using namespace std;

int a[200000];

int main() {
	int n;
	int s = 0;
	vector<int> b;

	cin >> n;
	for (int i = 0; i < n; ++i) {
		cin >> a[i];
		s += a[i];
	}
	for (int i = 0; i < n; ++i)
		if (a[i]*(n - 1) == s - a[i])
			b.push_back(i + 1);
	cout << b.size() << endl;
	for (size_t i = 0; i < b.size(); ++i)
		cout << b[i] << " ";
}
