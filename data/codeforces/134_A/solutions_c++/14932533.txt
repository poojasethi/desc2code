#include <iostream>
using namespace std;
int s[211111],f[211111];
int main()
{
	int a,d=0,q=0;
	cin>>a;
	for(int i=1;i<=a;i++)
	{
		cin>>s[i];
		d+=s[i];
	}
	for(int i=0;i<=a;i++)
	{
		if(d/a==s[i] && d%a==0)
		{
			f[q]=i;
			q++;
		}
	}
	cout<<q<<endl;
	for(int i=0;i<q;i++)
	{
		cout<<f[i]<<" ";
	}
	
}
