#include<stdio.h>
int n,i,j,k,ort,A[1000000],syc=0;
double top;
int main()
{
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&A[i]);
		top+=A[i];
	}
	ort=(int)(top/=n);
	for(i=0;i<n;i++)
		if(A[i]==top) syc++;
	printf("%d\n",syc);
	for(i=0;i<n;i++)
		if(A[i]==top)
			printf("%d ",i+1);
	return 0;
}
