#include<cstdio>
#include<vector>
int n,a[200010],i,s;
std::vector<int> o;
int main() {
    for(scanf("%d",&n),i=0;i<n;scanf("%d",a+i),s+=a[i++]);
    for(i=0;i<n;i++)
        if((s-a[i])==a[i]*(n-1))
            o.push_back(i+1);
    for(printf("%d\n",o.size()),i=0;i<o.size();printf("%d ",o[i++]));
}
