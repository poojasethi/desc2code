#include <stdio.h>
int a[200002], ans[200002];
int main()
{
    int n, i, c=0;
    double sum=0;
    scanf("%d", &n); for(i=0; i<n; i++) {scanf("%d", &a[i]); sum=sum+a[i];}
    for(i=0;i<n;i++) if(a[i]==sum/n) ans[c++]=i+1;
    printf("%d\n", c);
    for(i=0; i<c; i++) printf("%d ", ans[i]);
    return 0;
}
