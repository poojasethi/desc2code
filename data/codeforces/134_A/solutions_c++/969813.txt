#include<iostream>

using namespace std;

int arr[200001];

int main()
{
	int n,s = 0;
	cin>>n;
	for (int i=0;i<n;i++)
	{
		cin>>arr[i];
		s += arr[i];
	}
	int ans = 0;
	for (int i=0;i<n;i++)
		if (n * arr[i] == s)
			ans++;
	cout<<ans<<endl;
	for (int i=0;i<n;i++)
		if (n * arr[i] == s)
			cout<<i+1<<" ";
	cout<<endl;
	
	return 0;
}
