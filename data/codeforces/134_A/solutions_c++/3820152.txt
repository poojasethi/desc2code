#include <stdio.h>
int v[200000],res[200000],r;
int main()
{
	int n,sum=0;
	scanf("%d",&n);
	for(int i=0;i<n;++i){
		scanf("%d",&v[i]);
		sum+=v[i];
	}
	r=-1;
	for(int i=0;i<n;++i)
		if(double(sum-v[i])/(n-1)==v[i])
			res[++r]=i+1;
	++r;
	printf("%d\n",r);
	if(r){
		printf("%d",res[0]);
		for(int i=1;i<r;++i)
			printf(" %d",res[i]);
	}
	return 0;
}
