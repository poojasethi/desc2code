#include<stdio.h>
int main()
{
	double ort=0;
	int n,i,dizi[200005],sonuc[200005],sayac=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&dizi[i]);
		ort+=dizi[i];
	}
	ort/=n;
	for(i=0;i<n;i++)
		if(dizi[i]==ort)
			sonuc[sayac++]=i+1;
	printf("%d\n",sayac);
	for(i=0;i<sayac;i++)
		printf("%d ",sonuc[i]);
	if(sayac>0)printf("\n");
	return 0;
}
