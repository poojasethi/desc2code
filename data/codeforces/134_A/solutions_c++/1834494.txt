#include <iostream>
using namespace std;
int main(){
double n, mas[200200], sum=0, ans=0,otv[200200];
int j=0;
cin>>n;
for(int i=0;i<n;i++){
    cin>>mas[i];
    sum+=mas[i];
}
for(int i=0;i<n;i++){
    if(mas[i]==((sum-mas[i])/(n-1))){
      ans++;
       otv[j]=i+1;
       j++;
    }
} 
cout<<ans<<'\n';
for(int i=0;i<ans;i++)
    cout<<otv[i]<<' ';
}
