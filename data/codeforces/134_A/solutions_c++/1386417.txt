#include<iostream>
#include<cstdio>
using namespace std;
int main(){
	int i,n,dizi[1000000],dizi2[1000000],say=1,say2=0;
	double ort=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		scanf("%d",&dizi[i]);
		ort+=dizi[i];
	}
	ort/=n;
	for(i=1;i<=n;i++){
		if(dizi[i]==ort){
			dizi2[say++]=i;
			say2++;
		}
	}
	cout <<say2<<endl;
	for(i=1;i<say;i++)
		cout <<dizi2[i]<<" ";
	return 0;
}
