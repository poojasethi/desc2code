#include<iostream>
#include<cstdio>
using namespace std;
long long n,a[1000000],ans,s;
int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        s+=a[i];
        }
    for(int i=1;i<=n;i++)
        if(a[i]*n==s)ans++;
    cout<<ans<<endl;
    for(int i=1;i<=n;i++){
        if(a[i]*n==s)cout<<i<<" ";
    }
    return 0;
}
