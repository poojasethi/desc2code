#include<iostream>
using namespace std;
int a[200010];
int main(){
	int n,c=0,t=0,i;
	cin>>n;
	for(i=1;i<=n;i++)
	{cin>>a[i];c+=a[i];}
	for(i=1;i<=n;i++)
		if(double(c-a[i])/(n-1)==a[i]){
			t++;
			a[t]=i;
		}
		cout<<t<<endl;
		for(i=1;i<t;i++)
			cout<<a[i]<<' ';
		if(t!=0)cout<<a[i];
		cout<<endl;
	
}
