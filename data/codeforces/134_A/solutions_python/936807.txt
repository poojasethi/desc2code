#!/usr/bin/env python
# coding: utf-8

n = raw_input()
a = map(int, raw_input().split())
total = sum(a)
length = len(a)
res = []
for i in xrange(length):
    x = a[i]
    if x * (length - 1) == (total - x):
        res.append(str(i + 1))
if len(res) == 0:
    print 0
else:
    print len(res)
    print ' '.join(res)
