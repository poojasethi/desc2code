import sys 
n=input()
l=[]
l=map(int,raw_input().split())
x=sum(l)
m=[]
count=0
for i in range(0,n):
        y=x-l[i]
        if(l[i]*(n-1)==y):
            count+=1
            m.append(i+1)
print count
for i in range(0,len(m)):
    sys.stdout.write(str(m[i]))
    sys.stdout.write(" ")
print
