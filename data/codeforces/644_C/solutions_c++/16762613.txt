#include <bits/stdc++.h>
#include <string>
using namespace std;

int n;
string s;

map<string,vector<string> >m;
map<string,vector<string> >::iterator it;

map< vector<string> , vector<string> >ans;
map< vector<string> , vector<string> >::iterator it2;


int main()
{
cin>>n;
while(n--){
    cin>>s;

   string  hostname="" , path="" ;

   for(int i=0;i<s.length();i++){
    if( i>=7 && s[i]=='/' ){

        for(;i<s.length();i++)
            path+=s[i];

    }
    else hostname+=s[i];
   }

   m[ hostname ].push_back(path);

}

for(it=m.begin();it!=m.end();it++){
vector<string> v=it->second;
sort(v.begin(),v.end());

vector<string> v2;
if( v.size()>0 )v2.push_back(v[0]);

for(int i=1;i<v.size();i++){
    if(v[i]==v[i-1])continue;
    else v2.push_back(v[i]);
}

ans[v2].push_back( it->first );

}

int k=0;
for(it2=ans.begin();it2!=ans.end();it2++){
vector<string> v=it2->second;
sort(v.begin(),v.end());

vector<string> v2;
if( v.size()>0 )v2.push_back(v[0]);
for(int i=1;i<v.size();i++)
    if( v[i]!=v[i-1] )
    v2.push_back(v[i]);

it2->second=v2;

   if( v2.size()>=2 )
    k++;
}
cout<<k<<endl;

if(k>0)
for(it2=ans.begin();it2!=ans.end()&&k;it2++){
vector<string> v=it2->second;

   if( v.size()>=2 ) {
        k--;
        for(int i=0;i<v.size();i++)
            cout<<v[i]<<((i+1==v.size())?'\n':' ');
    }
}

    return 0;
}
