#include<bits/stdc++.h>
using namespace std;

map<set<int>, int> m1;
int c1;
map<string, int> m2;
int c2;
map<string, int> m3;
int c3;
set<int> a[100000 + 10];
vector<int> ans[100000 + 10];
string hostname[100000 + 10];
int n;

int main()
{
      ios::sync_with_stdio(false);
      cin >> n;
      for(int i = 0; i < n; i++)
      {
            string str;
            cin >> str;
            int pt;
            for(pt = 7; pt < str.length(); pt++) if(str[pt] == '/') break;
            string tmp = str.substr(pt, str.length() - pt);
            string host = str.substr(7, pt - 7);
            if(!m2[tmp]) m2[tmp] = ++c2;
            if(!m3[host]) m3[host] = ++c3, hostname[c3] = host;
            a[m3[host]].insert(m2[tmp]);
      }
      for(int i = 1; i <= c3; i++)
      {
            if(!m1[a[i]]) m1[a[i]] = ++c1;
            int t = m1[a[i]];
            ans[t].push_back(i);
      }
      int cnt = 0;
      for(int i = 1; i <= c1; i++) if(ans[i].size() > 1) cnt++;
      cout << cnt << endl;
      for(int i = 1; i <= c1; i++)
      {
            if(ans[i].size() > 1)
            {

                  for(int j = 0; j < ans[i].size(); j++)
                        cout << "http://" << hostname[ans[i][j]] << " ";
                  cout << endl;
            }
      }
      return 0;
}
