#include <bits/stdc++.h>
using namespace std;
#define fs  first
#define sc  second
#define lb  lower_bound
#define pb  push_back
#define mp  make_pair
#define ll  long long
#define vi  vector<ll >
#define vvi vector<vi >
#define mit map<ll, int>::iterator
#define sit set<ll>::iterator
#define all(x) x.begin(), x.end()

int n;
string s;
map<string, set<string> > M;
map<string, set<string> >::iterator itm;
map<vector<string>, vector<string> > N;
map<vector<string>, vector<string> >::iterator itn;

int main() {
    ios_base::sync_with_stdio(false);
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> s;
        int en = 7;
        while (en < s.size() && s[en] != '/')
            ++en;
        M[s.substr(0, en)].insert(s.substr(en));
    }
    for (itm = M.begin(); itm != M.end(); ++itm) {
        vector<string> w;
        for (set<string>::iterator it = itm->sc.begin(); it != itm->sc.end(); ++it)
            w.pb(*it);
        N[w].pb(itm->fs);
    }
    int num = 0;
    for (itn = N.begin(); itn != N.end(); ++itn) {
        if (itn->sc.size() > 1)
            ++num;
    }
    cout << num << "\n";
    for (itn = N.begin(); itn != N.end(); ++itn) {
        if (itn->sc.size() > 1) {
            for (int i = 0; i < itn->sc.size(); ++i)
                cout << itn->sc[i] << " ";
            cout << "\n";
        }
    }

    return 0;
}
