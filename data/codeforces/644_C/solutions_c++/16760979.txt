#include <bits/stdc++.h>

using namespace std;

#define fr first
#define sc second
#define pb push_back

map<string, set<string> > a;
map<set<string>, vector<string> > ans;
vector<vector<string> > answer(51000);

int main()
{
    //freopen("input.txt", "r", stdin);
    ios_base::sync_with_stdio(0);
    int n;
    cin >> n;
    string s;
    getline(cin, s);
    int j;
    for(int i = 0; i < n; i++){
        getline(cin, s);
        s = s.substr(7, s.size() - 7);
        j = s.size();
        for(int g = 0; g < j; g++){
            if(s[g] == '/'){
                j = g;
            }
        }
        a[s.substr(0,j)].insert(s.substr(j, s.size() - j));
        //cout << s.substr(0,j) << "    " << s.substr(j, s.size() - j) << "\n";
    }
    map<string, set<string> >::iterator h;
    for(h = a.begin(); h!=a.end(); ++h){
        ans[(*h).sc].pb((*h).fr);
        /*cout << (*h).fr << "\n";
        for(j = 0; j < (*h).sc.size(); j++){
            cout << "           " << (*h).sc[j] << "\n";
        }*/
    }
    map<set<string>, vector<string> >::iterator f;
    int k = 0;
    for(f = ans.begin(); f!=ans.end(); ++f){
        if((*f).sc.size() > 1){
            for(j = 0; j < (*f).sc.size(); j++){
                answer[k].pb((*f).sc[j]);
            }
            k++;
        }
    }
    cout << k << "\n";
    for(int i = 0; i < k; i++){
        for(j = 0; j < answer[i].size(); j++){
            cout << "http://" << answer[i][j] << " ";
        }
        cout << "\n";
    }
    return 0;
}
