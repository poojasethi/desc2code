#include <iostream>
#include <map>
#include <string>
#include <vector>
#include <set>

using namespace std;

int main() 
{
	int N;
	cin>>N;
	string x;
	map<string,set<string> >M;
	while(N--)
	{
		cin>>x;
		string S,SD;
		int i;
		set<string>D;
		i=x.find("/",7);
		if(i<0)
		{
			S=x;
			SD="";
		}
		else
		{
			S=x.substr(0,i);
			SD=x.substr(i);
		}
		D.insert(SD);
		if(!M.count(S))M[S]=D;
		else M[S].insert(SD);
	}
	map<string,set<string> >::iterator p;
	map<set<string>,vector<string> >MD;
	for(p=M.begin();p!=M.end();p++)
	{
		MD[p->second].push_back(p->first);
	}
	map<set<string>,vector<string> >::iterator q;
	N=0;
	for(q=MD.begin();q!=MD.end();q++)
	{
		N+=MD[q->first].size()>1;
	}
	cout<<N<<endl;
	if(N)
	{
		for(q=MD.begin();q!=MD.end();q++)
		{
			vector<string>V(MD[q->first]);
			if(V.size()==1) continue;
			cout<<V[0];
			for(int i=1;i<V.size();i++)
			{
				cout<<' '<<V[i];
			}
			cout<<endl;
		}
	}
	return 0;
}
