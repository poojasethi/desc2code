#include<iostream>
#include<algorithm>
#define f first
#define s second
using namespace std;
int n,i,k,j,w,z;
string q;
pair<string,string>p[100011];
pair<string,string>x[100011];
main()
{
      cin>>n;
      for(i=0;i<n;i++){
         cin>>q;
         j=7;
         while(q[j]!='/' && j<q.size()){
              p[i].f+=q[j];
              j++;
         }
         if(j==q.size())
           p[i].s=" ";
         for(;j<q.size();j++)
            p[i].s+=q[j];
      }
      sort(p,p+n);
      k=1;
      x[k].f+=p[0].s;
      for(i=1;i<n;i++){
         if(p[i].f==p[i-1].f){
           if(p[i].s!=p[i-1].s){
             x[k].f+="$";
             x[k].f+=p[i].s;
             }
         }else{
           x[k].s=p[i-1].f;
           k++;
           x[k].f+=p[i].s;
         }
      }
      x[k].s=p[i-1].f;
      k++;
      x[k].f+=p[i].s;
      sort(x+1,x+k);
      for(i=1;i<k;i++)
         if(x[i].f!=x[i+1].f && x[i].f==x[i-1].f)
           z++;
      cout<<z<<endl;
      for(i=1;i<k;i++){
         if(x[i].f==x[i+1].f)
           cout<<"http://"<<x[i].s<<" ";
         else{
           if(x[i].f==x[i-1].f){
             cout<<"http://"<<x[i].s<<" ";
             cout<<endl;
           }
         }
      }
      //system("pause");
}
