#include <iostream>
#include <stdio.h>
#include <set>
#include <string.h>
#include <map>
using namespace std;
string s,s1,s2;
map<string,set<string> >mp1;
map<set<string>,set<string> >mp2;
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>s;
        s1.clear();
        s2.clear();
        bool ok=false;
        for(int j=0;j<s.size();j++)
        {
            if(j>7&&s[j]=='/')
                ok=true;
            if(ok)
                s2.push_back(s[j]);
            else s1.push_back(s[j]);
        }
        mp1[s1].insert(s2);
    }
    for(map<string,set<string> >::iterator it=mp1.begin();it!=mp1.end();it++)
        mp2[it->second].insert(it->first);
    int ans=0;
    for(map<set<string>,set<string> >::iterator it=mp2.begin();it!=mp2.end();it++)
        if(it->second.size()>1)
            ans++;
    cout<<ans<<endl;
    for(map<set<string>,set<string> >::iterator it=mp2.begin();it!=mp2.end();it++)
    {
        if(it->second.size()>1)
        {
            for(set<string>::iterator its=it->second.begin();its!=it->second.end();its++)
                cout<<(*its)<<" ";
            cout<<endl;
        }
    }
    return 0;
}
