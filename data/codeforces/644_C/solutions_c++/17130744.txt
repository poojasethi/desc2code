#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
#include <set>
#include <map>
using namespace std;

int main()
{
	map <string,set<string> > hq;
	map <set<string>,vector<string> > sg;
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		string q;
		cin >> q;
		int t = q.find('/',7);
		string p = " ";
		if (t != -1) p = q.substr(t);
		hq[q.substr(0,t)].insert(p);
	}
	for (map <string,set<string> > :: iterator it = hq.begin(); it != hq.end(); it++) sg[it->second].push_back(it->first);
	vector <vector<string> > anw (0,vector <string> (0));
	for (map <set<string>,vector<string> > :: iterator it = sg.begin(); it != sg.end(); it++)
	{
		vector <string> t = it->second;
		if (t.size() == 1) continue;
		anw.push_back(t);
	}
	cout << anw.size() << endl;
	for (int i = 0; i < anw.size(); i++)
	{
		for (int j = 0; j < anw[i].size(); j++)
		{
			if (j != 0) cout << " ";
			cout << anw[i][j];
		}
		cout << endl;
	}
}
