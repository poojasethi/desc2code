from itertools import groupby
def parse(s):
    x=s.find('/',7)
    return (s[:x],s[x:])
a=(parse(s) for s in sorted(set(raw_input()+'/' for _ in range(input()))))
b=((map(lambda x: x[1],list(v)), k)for k,v in groupby(a, lambda x:x[0]))
ans=[]
for k, g in groupby(sorted(b), lambda x:x[0]):
    v=list(g)
    if len(v)>1:ans.append(x[1] for x in v)
print len(ans)
print '\n'.join(' '.join(x) for x in ans)
