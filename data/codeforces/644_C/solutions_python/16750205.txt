def main():
	
	import sys
	import bisect
	#import math
	#from Queue import PriorityQueue
	#from operator import itemgetter
	#from sets import Set
 
	n = int(sys.stdin.readline().strip())

	domains = {}
	queries = {}
	index_query = 0
	for i in xrange(n):
		url = sys.stdin.readline().strip()
		parts = url.split('/',3)
		
		domain = parts[0]+"//"+parts[2]
		
		if (len(parts) < 4):
			query = '*'
		else:
			query = '/'+parts[3]
		
		if query not in queries:
			queries[query] = index_query
			index_query += 1
		
		if domain not in domains:
			domains[domain] = []
		_l = bisect.bisect_left(domains[domain],queries[query])
		if (_l >= len(domains[domain])) or  (domains[domain][_l] != queries[query]):
			domains[domain].insert(_l, queries[query])
	
	hashes = {}
	for do in domains:
		
		hash = "-".join(map(str,domains[do]))
		if hash not in hashes:
			hashes[hash] = []
		hashes[hash].append(do)

		
	total = 0
	_str = ""
	for ha in hashes:
		if len(hashes[ha])> 1:
			total += 1
			_str += " ".join(hashes[ha]) + "\n"
	
	print total
	sys.stdout.write(_str)
	
if __name__ == "__main__":

	main()
