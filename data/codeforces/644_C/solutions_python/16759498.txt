from urlparse import urlparse
from collections import defaultdict
n = int(raw_input())
d = defaultdict(set)
for _ in xrange(n):
    l = urlparse(raw_input())
    d[l.netloc].add(l.path)
iv = defaultdict(list)
for k, v in d.viewitems():
    iv[frozenset(v)].append(k)
ans = []
for k, v in iv.viewitems():
    if len(v) > 1:
        ans.append(v)
print len(ans)
for v in ans:
    for x in v:
        print "http://" + x,
    print
