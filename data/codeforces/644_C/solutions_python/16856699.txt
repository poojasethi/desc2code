def splitdomainpath(q):
	cnt = 0
	for i in xrange(len(q)):
		if q[i] == '/':
			cnt += 1
		if cnt == 3:
			break
	return (q[:i], q[i:]) if cnt == 3 else (q, "")

n = int(raw_input())
queries = sorted([raw_input() for i in xrange(n)])

domain_paths = dict()
for q in queries:
	domain, path = splitdomainpath(q)
	if domain not in domain_paths:
		domain_paths[domain] = list()
	domain_paths[domain].append(path)

paths_domains = dict()
for domain, paths in domain_paths.iteritems():
	key = tuple(sorted(list(set(paths))))
	if key not in paths_domains:
		paths_domains[key] = list()
	paths_domains[key].append(domain)

result = [" ".join(domains)
		  for _, domains in paths_domains.iteritems() if len(domains) > 1]
print len(result)
if len(result) > 0: print "\n".join(result)
