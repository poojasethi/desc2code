from collections import defaultdict


def solution():
    n = int(raw_input())
    paths = {}
    addrs = defaultdict(set)
    for _ in xrange(n):
        addr = raw_input().strip().replace('http://', '')
        addr, path = (addr, None) if addr.find('/') < 0 else addr.split('/', 1)
        if path not in paths:
            paths[path] = len(paths)
        addrs[addr].add(paths[path])
    grouping = defaultdict(list)
    groups = 0
    for addr, paths in addrs.iteritems():
        fr = frozenset(paths)
        grouping[fr].append(addr)
        if len(grouping[fr]) == 2:
            groups += 1
    print groups
    for group in grouping.itervalues():
        if len(group) > 1:
            print ' '.join('http://' + o for o in group)


if __name__ == '__main__':
	solution()
