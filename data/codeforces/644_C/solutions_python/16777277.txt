import json
import re
n = int(raw_input())
t = [re.match(r'(http://.*?)(/.*)',raw_input() + r'/').groups() for i in xrange(n)]
d = {}
for k, v in t:
    d[k] = d.get(k, set([]))
    d[k].add(v)
dt = {}
for v, k in d.items():
    k1 = ','.join(sorted(list(k)))
    dt[k1] = dt.get(k1, set([]))
    dt[k1].add(v)
ans = 0
for k, v in dt.items():
    if len(v) > 1:
        ans += 1
print ans
for k, v in dt.items():
    if len(v) > 1:
        print ' '.join(list(v))
