#include <iostream>
#include <stdio.h>

using namespace std;

int n, m, k;
int a[102][102][102];

int check(int i, int j, int l)
{
    int cnt = 0;
    if(a[i-1][j][l] && a[i+1][j][l]) cnt++;
    if(a[i][j-1][l] && a[i][j+1][l]) cnt++;
    if(a[i][j][l-1] && a[i][j][l+1]) cnt++;

    if(a[i-1][j][l] && a[i][j+1][l] && !a[i-1][j+1][l]) cnt++;
    if(a[i-1][j][l] && a[i][j][l+1] && !a[i-1][j][l+1]) cnt++;
    if(a[i][j-1][l] && a[i+1][j][l] && !a[i+1][j-1][l]) cnt++;
    if(a[i][j-1][l] && a[i][j][l+1] && !a[i][j-1][l+1]) cnt++;
    if(a[i][j][l-1] && a[i+1][j][l] && !a[i+1][j][l-1]) cnt++;
    if(a[i][j][l-1] && a[i][j+1][l] && !a[i][j+1][l-1]) cnt++;

    return cnt;
}

int main()
{
    // freopen("input.txt", "r", stdin);
    int i, j, l;
    char ch;
    for(i = 0; i < 102; i++)
        for(j = 0; j < 102; j++)
            for(l = 0; l < 102; l++)
                a[i][j][l] = 0;
    cin >> n >> m >> k;
    for(i = 1; i <= n; i++)
        for(j = 1; j <= m; j++)
            for(l = 1; l <= k; l++)
            {
                cin >> ch;
                if(ch == '1')
                    a[i][j][l] = 1;
                else
                    a[i][j][l] = 0;
            }
    int res = 0;
    for(i = 1; i <= n; i++)
        for(j = 1; j <= m; j++)
            for(l = 1; l <= k; l++)
                if(a[i][j][l] && check(i, j, l))
                {
                    res += 1;
                }
    cout << res << endl;

    return 0;
}
