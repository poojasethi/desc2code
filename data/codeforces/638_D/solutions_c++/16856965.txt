#include <vector>
#include <iostream>

using namespace std;

int main()
{
    int n,m,k,kol=0;
    char z;
    cin>>n>>m>>k;
    int qr[n+2][m+2][k+2];

    for (int j=0; j<=m+1; j++)
        for (int q=0; q<=k+1; q++)
        {
             qr[0][j][q]=0;
             qr[n+1][j][q]=0;
        }
     for (int i=0; i<=n+1; i++)
        for (int q=0; q<=k+1; q++)
        {
             qr[i][0][q]=0;
             qr[i][m+1][q]=0;
        }
     for (int j=0; j<=m+1; j++)
        for (int i=0; i<=n+1; i++)
        {
             qr[i][j][0]=0;
             qr[i][j][k+1]=0;
        }
    for (int i=1; i<=n; i++)
    {
        for (int j=1; j<=m; j++)
        {
            for (int q=1; q<=k; q++)
            {
                cin>>z;
                while ((z!='0') && (z!='1'))
                {
                    cin>>z;
                }
               // cout<<z<<endl;
                if (z=='1')
                {
                    qr[i][j][q]=1;
                }
                else
                    qr[i][j][q]=0;
                //cout<<i<<" "<<j<<" "<<q<<" "<<qr[i][j][q]<<endl;
            }
        }
    }
    for (int i=1; i<=n; i++)
    {
        for (int j=1;j<=m;j++)
        {
            for (int q=1; q<=k; q++)
            {
                if (qr[i][j][q]==0)
                    continue;

                if (((qr[i+1][j][q]==1) &&(qr[i-1][j][q]==1)) || ((qr[i][j-1][q]==1) && (qr[i][j+1][q]==1)) || ((qr[i][j][q-1]==1)&&(qr[i][j][q+1]==1)) || ((qr[i-1][j][q]==1) && (qr[i][j+1][q]==1) && (qr[i-1][j+1][q]==0)) || ((qr[i-1][j][q]==1) && (qr[i][j][q+1]==1) && (qr[i-1][j][q+1]==0)) || ((qr[i][j-1][q]==1) && (qr[i+1][j][q]==1) && (qr[i+1][j-1][q]==0)) || ((qr[i][j-1][q]==1) && (qr[i][j][q+1]==1) && (qr[i][j-1][q+1]==0))|| ((qr[i][j][q-1]==1) && (qr[i+1][j][q]==1) && (qr[i+1][j][q-1]==0)) || ((qr[i][j][q-1]==1) && (qr[i][j+1][q]==1) && (qr[i][j+1][q-1]==0)))
                {
                    kol++;
                   // cout<<i<<" "<<j<<" "<<q<<" "<<qr[i][j][q]<<endl;
                }
            }
        }
    }
    cout<<kol<<endl;
    return 0;
}
