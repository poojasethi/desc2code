using namespace std;




int main() {
//    ifstream fin("input.txt");
//    ofstream fout("output.txt");
	int n, m, k, r, f;
	bool a[101][101][101];
	string s;

	cin >> n >> m >> k;
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < m; ++j) {
			cin >> s;
			for (int t = 0; t < k; ++t) {
				a[i][j][t] = (s[t] == '1');
			}
		}
	}

	r = 0;
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < m; ++j) {
			for (int t = 0; t < k; ++t) {
				if (a[i][j][t]) {
					f = 0;

					if (i > 0 && a[i - 1][j][t]) {
						if ((i < (n - 1) && a[i + 1][j][t]) ||
								(j < (m - 1) && a[i][j + 1][t] && !a[i - 1][j + 1][t]) ||
								(t < (k - 1) && a[i][j][t + 1] && !a[i - 1][j][t + 1])) {
							f = 1;
						}
					}

					if (j > 0 && a[i][j - 1][t]) {
						if ((j < (m - 1) && a[i][j + 1][t]) ||
								(i < (n - 1) && a[i + 1][j][t] && !a[i + 1][j - 1][t]) ||
								(t < (k - 1) && a[i][j][t + 1] && !a[i][j - 1][t + 1])) {
							f = 1;
						}
					}

					if (t > 0 && a[i][j][t - 1]) {
						if ((t < (k - 1) && a[i][j][t + 1]) ||
								(i < (n - 1) && a[i + 1][j][t] && !a[i + 1][j][t - 1]) ||
								(j < (m - 1) && a[i][j + 1][t] && !a[i][j + 1][t - 1])) {
							f = 1;
						}
					}

					r += f;
				}
			}
		}
	}
	cout << r;

    return 0;
}
