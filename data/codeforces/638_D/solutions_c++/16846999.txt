#include <iostream>
#include <vector>

using namespace std;

int comp[100][100][100];

int main()
{
    int n, m, k;
    cin >> n >> m >> k;
    string s;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            cin >> s;
            for (int t = 0; t < k; ++t)
                comp[i][j][t] = (s[t] == '1') ? 1 : 0;
        }
    }
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < m; ++j)
            for (int t = 0; t < k; ++t) {
                if (i > 0 && j > 0 && comp[i][j][t] && comp[i - 1][j][t] && comp[i - 1][j - 1][t] && !comp[i][j - 1][t])
                    comp[i - 1][j][t] = 2;
                if (i > 0 && j > 0 && comp[i][j][t] && !comp[i - 1][j][t] && comp[i - 1][j - 1][t] && comp[i][j - 1][t])
                    comp[i][j - 1][t] = 2;

                if (j > 0 && t > 0 && comp[i][j][t] && comp[i][j - 1][t] && comp[i][j - 1][t - 1] && !comp[i][j][t - 1])
                    comp[i][j - 1][t] = 2;
                if (j > 0 && t > 0 && comp[i][j][t] && !comp[i][j - 1][t] && comp[i][j - 1][t - 1] && comp[i][j][t - 1])
                    comp[i][j][t - 1] = 2;

                if (i > 0 && t > 0 && comp[i][j][t] && comp[i - 1][j][t] && comp[i - 1][j][t - 1] && !comp[i][j][t - 1])
                    comp[i - 1][j][t] = 2;
                if (i > 0 && t > 0 && comp[i][j][t] && !comp[i - 1][j][t] && comp[i - 1][j][t - 1] && comp[i][j][t - 1])
                    comp[i][j][t - 1] = 2;

                if (i > 1 && comp[i][j][t] && comp[i - 1][j][t] && comp[i - 2][j][t])
                    comp[i - 1][j][t] = 2;
                if (j > 1 && comp[i][j][t] && comp[i][j - 1][t] && comp[i][j - 2][t])
                    comp[i][j - 1][t] = 2;
                if (t > 1 && comp[i][j][t] && comp[i][j][t - 1] && comp[i][j][t - 2])
                    comp[i][j][t - 1] = 2;
            }

    int cnt = 0;
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < m; ++j)
            for (int t = 0; t < k; ++t)
                if (comp[i][j][t] == 2)
                    cnt++;
    cout << cnt;
    return 0;
}
