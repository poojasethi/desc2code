#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

using namespace std;

int a,b,c,res;

char s[105][105][105];

int dx[3]={1,0,0};
int dy[3]={0,1,0};
int dz[3]={0,0,1};

bool gg(int x,int y,int z){
  if(x<=0 || y<=0 || z<=0) return true;
  if(x>a || y>b || z>c) return true;
  if(s[x][y][z]=='0') return true;
  return false;
}

bool tr(int x,int y,int z,int ma,int mb){
  if(gg(x-dx[ma],y-dy[ma],z-dz[ma])) return false;
  if(gg(x+dx[mb],y+dy[mb],z+dz[mb])) return false;
  if(gg(x,y,z)) return false;
  
  if(ma==mb) return true;
  else return gg(x-dx[ma]+dx[mb],y-dy[ma]+dy[mb],z-dz[ma]+dz[mb]);
}
int main(){
  scanf("%d %d %d",&a,&b,&c);
  FI(i,1,a){
    FI(j,1,b) scanf(" %s",s[i][j]+1);
    scanf(" ");
  }
  FI(i,1,a) FI(j,1,b) FI(k,1,c){
    bool ok=false;
    FI(y,0,2) FI(z,0,2){
      ok|=tr(i,j,k,y,z);
    }
    if(ok) res++;
  }
  printf("%d\n",res);
  return 0;
}
