#include <iostream>

using namespace std;

int main() {
    int n, m, k, ans = 0;
    bool a[101][101][101] = {0};
    string s[101][101];
    cin >> n >> m >> k;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            cin >> s[i][j];

    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            for (int q = 0; q < k; q++)
            {
                if (s[i][j][q] == '0')
                    continue;

                if (j > 0 && j < m - 1)
                    if (s[i][j + 1][q] == '1' && s[i][j - 1][q] == '1')
                        a[i][j][q] = 1;

                if (q > 0 && q < k - 1)
                    if (s[i][j][q + 1] == '1' && s[i][j][q - 1] == '1')
                        a[i][j][q] = 1;

                if (i > 0 && i < n - 1)
                    if (s[i - 1][j][q] == '1' && s[i + 1][j][q] == '1')
                        a[i][j][q] = 1;

                if (q > 0 && j < m - 1)
                    if (s[i][j][q - 1] == '1' && s[i][j + 1][q] == '1' && s[i][j + 1][q - 1] == '0')
                        a[i][j][q] = 1;
                if (j > 0 && q < k - 1)
                    if (s[i][j][q + 1] == '1' && s[i][j - 1][q] == '1' && s[i][j - 1][q + 1] == '0')
                        a[i][j][q] = 1;
//-------
                if (q > 0 && i < n - 1)
                    if (s[i][j][q - 1] == '1' && s[i + 1][j][q] == '1' && s[i + 1][j][q - 1] == '0')
                        a[i][j][q] = 1;
                if (i > 0 && q < k - 1)
                    if (s[i][j][q + 1] == '1' && s[i - 1][j][q] == '1' && s[i - 1][j][q + 1] == '0')
                        a[i][j][q] = 1;
//-------
                if (i > 0 && j < m - 1)
                    if (s[i - 1][j][q] == '1' && s[i][j + 1][q] == '1' && s[i - 1][j + 1][q] == '0')
                        a[i][j][q] = 1;
                if (j > 0 && i < n - 1)
                    if (s[i + 1][j][q] == '1' && s[i][j - 1][q] == '1' && s[i + 1][j - 1][q] == '0')
                        a[i][j][q] = 1;

                if (a[i][j][q])
                    ans++;
            }
    cout << ans;
}
