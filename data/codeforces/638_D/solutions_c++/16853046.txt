inline ll mult(ll a, ll b, ll mod){
    if(b == 0)
        return 0;
    if(b % 2 == 1)
        return (mult(a, b - 1, mod) + a) % mod;
    return (mult(a, b / 2, mod) * 2) % mod;
}

inline ll BinPow(ll a, ll b, ll mod){
    if(b == 0)
        return 1;
    if(b == 1)
        return a;
    if(b % 2 == 1)
        return (mult(BinPow(a, b - 1, mod), a, mod)) % mod;
    ll res = BinPow(a, b / 2, mod);
    return mult(res, res, mod);
}
