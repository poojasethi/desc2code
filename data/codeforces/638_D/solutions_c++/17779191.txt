#include <iostream>
#include <string.h>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;
const int N = 110;
int v[N][N][N];
int n, m, k, ans;
const int dx[3] = {1, 0, 0};
const int dy[3] = {0, 1, 0};
const int dz[3] = {0, 0, 1};
bool judge(int x, int y, int z)
{
	for (int i = 0; i < 3; i++)
	{
		if (v[x-dx[i]][y-dy[i]][z-dz[i]])
		{
			for (int j = 0; j < 3; j++)
			{
				if (v[x+dx[j]][y+dy[j]][z+dz[j]])
				{
					if (i == j)
					  return true;
					else if (!v[x-dx[i]+dx[j]][y-dy[i]+dy[j]][z-dz[i]+dz[j]])
					  return true;
				}
			}
		}
	}
	return false;
}
int main()
{
	scanf("%d%d%d", &n, &m, &k);
	for (int i = 1; i <= n; i++)
	  for (int j = 1; j <= m; j++)
		for (int l = 1; l <= k; l++)
		  scanf("%1d", &v[i][j][l]);
	ans = 0;
	for (int i = 1; i <= n; i++)
	  for (int j = 1; j <= m; j++)
		for (int l = 1; l <= k; l++)
		  if (v[i][j][l] && judge(i, j, l))
			ans++;
	cout<<ans<<endl;
	return 0;
}
