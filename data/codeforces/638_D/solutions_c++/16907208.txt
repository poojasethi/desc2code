#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <vector>
#include <string.h>
using namespace std;
char mp[101][101][101];
int dir[3][3]={{1,0,0},{0,1,0},{0,0,1}};
int n,m,k;
bool ck(int x,int y,int z){
    if(x<0||x>=n)
        return false;
    if(y<0||y>=m)
        return false;
    if(z<0||z>=k)
        return false;
    return mp[x][y][z]=='1';
}
int main()
{
    //freopen("dd.txt","r",stdin);

    scanf("%d%d%d",&n,&m,&k);
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            scanf("%s",mp[i][j]);
        }
    }
    int ans=0;
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            for(int x=0;x<k;x++){
                if(mp[i][j][x]=='1'){
                    int tru=0;
                    for(int st=0;st<3;st++){
                        int xx=i-dir[st][0],yy=j-dir[st][1],zz=x-dir[st][2];
                        if(ck(xx,yy,zz)){
                            for(int ed=0;ed<3;ed++){
                                int x2=i+dir[ed][0],y2=j+dir[ed][1],z2=x+dir[ed][2];
                                if(ck(x2,y2,z2)){
                                    //printf("%d %d %d %d %d %d %d %d %d\n",xx,yy,zz,i,j,x,x2,y2,z2);
                                    x2-=xx,y2-=yy,z2-=zz;
                                    if(x2==2||y2==2||z2==2){
                                        tru=1;
                                        break;
                                    }
                                    if(x2==1&&xx+x2!=i&&mp[xx+x2][yy][zz]=='1')
                                        continue;
                                    if(y2==1&&yy+y2!=j&&mp[xx][yy+y2][zz]=='1')
                                        continue;
                                    if(z2==1&&zz+z2!=x&&mp[xx][yy][zz+z2]=='1')
                                        continue;
                                    tru=1;
                                }
                                if(tru)
                                    break;

                            }
                        }
                        if(tru)
                            break;
                    }
                    //if(tru)
                    //    printf("%d %d %d\n",i,j,x);
                    ans+=tru;
                }
            }
        }
    }
    printf("%d\n",ans);
    return 0;
}
