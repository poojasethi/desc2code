#include <iostream>

#include <string.h>

#pragma warning (disable : 4996)

using namespace std;

#define MP make_pair
#define lli long long int

const int N = 111;
const lli M = 1e9 + 7;

bool f[N][N][N];

int main() {
	ios_base::sync_with_stdio(0);
#ifdef FILE_IO
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
#endif	
	int n, m, k;
	cin >> n >> m >> k;
	for (int i = 1; i <= n; ++i) {
		for (int j = 1; j <= m; ++j) {
			for (int t = 1; t <= k; ++t) {
				char c;
				cin >> c;
				f[i][j][t] = c == '1';
			}
		}
	}
	int ans = 0;
	for (int x = 1; x <= n; ++x) {
		for (int y = 1; y <= m; ++y) {
			for (int z = 1; z <= k; ++z) {
				if (!f[x][y][z]) continue;
				bool ok = 0;

				if (f[x - 1][y][z] && f[x + 1][y][z]) ok = 1;
				if (f[x][y - 1][z] && f[x][y + 1][z]) ok = 1;
				if (f[x][y][z - 1] && f[x][y][z + 1]) ok = 1;

				if (f[x - 1][y][z] && f[x][y + 1][z]) ok |= !f[x - 1][y + 1][z];
				if (f[x - 1][y][z] && f[x][y][z + 1]) ok |= !f[x - 1][y][z + 1];

				if (f[x][y - 1][z] && f[x + 1][y][z]) ok |= !f[x + 1][y - 1][z];
				if (f[x][y - 1][z] && f[x][y][z + 1]) ok |= !f[x][y - 1][z + 1];

				if (f[x][y][z - 1] && f[x][y + 1][z]) ok |= !f[x][y + 1][z - 1];
				if (f[x][y][z - 1] && f[x + 1][y][z]) ok |= !f[x + 1][y][z - 1];

				ans += ok;
			}
		}
	}
	cout << ans;
	return 0;
}
