#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back

using namespace std;

int n, m, k, ans = 0;

char p[105][105][105];

int main() {
    //freopen("input.txt", "r", stdin);
    //freopen("output.txt", "w", stdout);
    scanf("%d%d%d\n", &n, &m, &k);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            for(int l = 0; l < k; l++)
                scanf("%c", &p[i][j][l]);
            scanf("\n");
        }
        scanf("\n");
    }
    bool f;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            for(int l = 0; l < k; l++)
                if (p[i][j][l] == '1') {
                    f = 0;
                    if (i+1 < n && p[i+1][j][l] == '1') {
                        if (i-1 >= 0 && p[i-1][j][l] == '1') f = 1;
                        if ((j-1 >= 0 && p[i][j-1][l] == '1') && p[i+1][j-1][l] == '0') f = 1;
                        if ((l-1 >= 0 && p[i][j][l-1] == '1') && p[i+1][j][l-1] == '0') f = 1;
                    }
                    if (j+1 < m && p[i][j+1][l] == '1') {
                        if (j-1 >= 0 && p[i][j-1][l] == '1') f = 1;
                        if ((l-1 >= 0 && p[i][j][l-1] == '1') && p[i][j+1][l-1] == '0') f = 1;
                        if ((i-1 >= 0 && p[i-1][j][l] == '1') && p[i-1][j+1][l] == '0') f = 1;
                    }
                    if (l+1 < k && p[i][j][l+1] == '1') {
                        if (l-1 >= 0 && p[i][j][l-1] == '1') f = 1;
                        if ((j-1 >= 0 && p[i][j-1][l] == '1') && p[i][j-1][l+1] == '0') f = 1;
                        if ((i-1 >= 0 && p[i-1][j][l] == '1') && p[i-1][j][l+1] == '0') f = 1;
                    }
                    if (f) ans++;
                }
    printf("%d\n", ans);
    return 0;
}
