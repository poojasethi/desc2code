#include <bits/stdc++.h>

#define mp make_pair
#define pb push_back
#define fi first
#define se second

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;
typedef unsigned long long ull;
typedef long double ld;
typedef pair<ll, ll> pll;

const int N = 1e6 + 7;
const ll INF = 1e9 + 7;
const ld EPS = 1e-9;
const ll P = 485437;
const int C = 2016;

int dx[] = {1, 0, 0};
int dy[] = {0, 1, 0};
int dz[] = {0, 0, 1};

bool a[107][107][107];

int main()
{
    ios_base::sync_with_stdio(0);
    int n, m, k, ans = 0;
    cin >> n >> m >> k;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            for (int c = 1; c <= k; c++){
                char cc;
                cin >> cc;
                if (cc == '1')
                    a[i][j][c] = 1;
            }
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            for (int c = 1; c <= k; c++){
                if (!a[i][j][c])
                    continue;
                a[i][j][c] = 0;
                bool f = false;
                for (int x = 0; x < 3; x++)
                    for (int y = 0; y < 3; y++){
                        if (!a[i + dx[x]][j + dy[x]][c + dz[x]] ||
                            !a[i - dx[y]][j - dy[y]][c - dz[y]])
                                continue;
                        if (!a[i + dx[x] - dx[y]][j + dy[x] - dy[y]][c + dz[x] - dz[y]])
                            f = true;
                    }
                if (f){
                    ans++;
                }
                a[i][j][c] = 1;
            }
    cout << ans;
    return 0;
}
