#include<bits/stdc++.h>
#define L 100005

using namespace std;
int i, j, f,n, m, k, a[200][200][200], ok, p[4][4], u[4][4], nr=0;
string s;

bool Dafq(int px, int py, int pz, int x, int y, int z, int ux, int uy, int uz){
	int q, e;
	
	if((z==pz && z==uz && x==ux && x==px) ||
		(z==pz && z==uz && y==uy && y==py) ||
		(y==py && y==uy && x==ux && x==px)) return 1;
		
	
	if (z==pz && z==uz){
		if (px!=x) q=px; else q=ux;
		if (py!=y) e=py; else e=uy;
		
		if (a[q][e][z]==0) return 1;
	}
	if (y==py && y==uy){
		if (px!=x) q=px; else q=ux;
		if (pz!=z) e=pz; else e=uz;
		
		if (a[q][y][e]==0) return 1;
	}
	if (x==px && x==ux){
		if (pz!=z) q=pz; else q=uz;
		if (py!=y) e=py; else e=uy;
		
		if (a[x][e][q]==0) return 1;
	}
	return 0;
}

bool Check(int x, int y, int z){
	int i, j, f;
	bool ok;
		
	if (a[x][y][z]){
		for(i=1; i<=3; ++i)
			for(j=1; j<=3; ++j){ 
				if( a[x+p[i][1]][y+p[i][2]][z+p[i][3]] &&
					a[x+u[j][1]][y+u[j][2]][z+u[j][3]]){
						
						if(Dafq( x+p[i][1], y+p[i][2], z+p[i][3], 
								 x, y, z, x+u[j][1], y+u[j][2], z+u[j][3]) )
								 
								 	return 1;
						
					}
					
			}		
	}
	return 0;
}

int main(){
	
	cin>>n>>m>>k;
	for(i=1; i<=n; ++i){
		
		for(j=1; j<=m; ++j){
			cin>>s;
			for(f=1; f<=k; ++f)
				a[i][j][f]=s[f-1]-48;
		}
		getline(cin, s);
	}
	
	for(i=1; i<=3; ++i){
		p[i][i]=-1;
		u[i][i]=1;
	}
	
	for(i=1; i<=n; ++i)
		for(j=1; j<=m; ++j)
			for(f=1; f<=k; ++f)
				if (Check(i, j, f)) ++nr;
				
	cout<<nr;
	
	return 0;
}
