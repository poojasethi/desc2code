#include <cstdio>

int main() {
	bool ar[100][100][100];
	int n, m, k;
	scanf("%d %d %d",&n, &m, &k);
	int ans = 0;
	for (int i = 0; i < n; ++i) {
		getchar();
		for (int j = 0; j < m; ++j) {
			for (int l = 0; l < k; ++l) {
				ar[i][j][l] = getchar() - '0';
			}
			getchar();
		}
	}

	// for (int i = 0; i < n; ++i) {
	// 	for (int j = 0; j < m; ++j) {
	// 		for (int l = 0; l < k; ++l) {
	// 			printf("%d ",ar[i][j][l]);
	// 		}
	// 		printf("\n");
	// 	}
	// 	printf("\n");
	// }

	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < m; ++j) {
			for (int l = 0; l < k; ++l) {
				if (!ar[i][j][l]) continue;
				else if (i && (i != n - 1) && ar[i - 1][j][l] && ar[i + 1][j][l]) {
					ans++;
				}
				else if (j && (j != m - 1) && ar[i][j - 1][l] && ar[i][j + 1][l]) {
					ans++;
				}
				else if (l && (l != k - 1) && ar[i][j][l - 1] && ar[i][j][l + 1]) {
					ans++;
				}
				else if (j && (l != k - 1) && ar[i][j - 1][l] && ar[i][j][l + 1] && !ar[i][j - 1][l + 1]) {
					ans++;
				}
				else if (l && (j != m - 1) && ar[i][j][l - 1] && ar[i][j + 1][l] && !ar[i][j + 1][l - 1]) {
					ans++;
				}
				else if (i && (j != m - 1) && ar[i - 1][j][l] && ar[i][j + 1][l] && !ar[i - 1][j + 1][l]) {
					ans++;
				}
				else if (i && (l != k - 1) && ar[i - 1][j][l] && ar[i][j][l + 1] && !ar[i - 1][j][l + 1]) {
					ans++;
				}
				else if (j && (i != n - 1) && ar[i + 1][j][l] && ar[i][j - 1][l] && !ar[i + 1][j - 1][l]) {
					ans++;
				}
				else if (l && (i != n - 1) && ar[i + 1][j][l] && ar[i][j][l - 1] && !ar[i + 1][j][l - 1]) {
					ans++;
				}
//				else printf("!! %d %d %d\n",i, j, l);
			}
		}
	}
	printf("%d\n",ans);
}
