#include<bits/stdc++.h>

using namespace std;
int const M=1e6+5;
int md=1e9+7;

long long x,y,ans;

int i,j,n,m,N,t,k;
int a[111][111][111];


int main()
{
ios_base::sync_with_stdio(0);
cin>>n>>m>>k;
for (int i=1;i<=n;i++)
 for (int j=1;j<=m;j++)
  for (int z=1;z<=k;z++)
   {
       char c;
       cin>>c;
       a[i][j][z]=c-'0';
   }
int ans=0;
for (int i=1;i<=n;i++)
 for (int j=1;j<=m;j++)
  for (int z=1;z<=k;z++)
    if (a[i][j][z]>0) {

     if (a[i-1][j][z]==1 && a[i+1][j][z]==1 ) ans++; else
     if (a[i][j-1][z]==1 && a[i][j+1][z]==1 ) ans++; else
     if (a[i][j][z-1]==1 && a[i][j][z+1]==1 ) ans++; else
     if (a[i-1][j][z]==1 && a[i][j+1][z]==1 && a[i-1][j+1][z]==0  ) ans++; else
     if (a[i-1][j][z]==1 && a[i][j][z+1]==1 && a[i-1][j][z+1]==0  ) ans++; else
     if (a[i][j-1][z]==1 && a[i][j][z+1]==1 && a[i][j-1][z+1]==0  ) ans++; else
     if (a[i][j-1][z]==1 && a[i+1][j][z]==1 && a[i+1][j-1][z]==0  ) ans++; else
     if (a[i][j][z-1]==1 && a[i+1][j][z]==1 && a[i+1][j][z-1]==0  ) ans++; else
     if (a[i][j][z-1]==1 && a[i][j+1][z]==1 && a[i][j+1][z-1]==0  ) ans++;
    }
cout<<ans;
}
