
#include <iostream>
#include <list>

using namespace std;

int p[102][102][102];
int p2[102][102][102];

int main()
{
    int n, m, k;
    cin >> n >> m >> k;

    for ( int i = 0 ; i < n + 2 ; i++ ) {
        for ( int i2 = 0 ; i2 < m + 2; i2++ ) {
            for ( int i3 = 0 ; i3 < k + 2 ; i3++ ) {
                p[i][i2][i3] = 0;
                p2[i][i2][i3] = 0;
            }
        }
    }

    for ( int i = 0 ; i < n ; i++ ) {
        for ( int i2 = 0 ; i2 < m ; i2++ ) {
            for ( int i3 = 0 ; i3 < k ; i3++ ) {
                char inp = 0;
                cin >> inp;
                p[i][i2][i3] = inp == '1';
            }
        }
    }

    int answ = 0;

    for ( int i = 0 ; i < n ; i++ ) {
        for ( int i2 = 0 ; i2 < m ; i2++ ) {
            for ( int i3 = 0 ; i3 < k ; i3++ ) {

                if (!p[i][i2][i3])
                    continue;

                if (!p2[i + 1][i2][i3] && p[i + 1][i2][i3] && (p[i + 2][i2][i3] || p[i + 1][i2 + 1][i3] && !p[i][i2 + 1][i3] || p[i + 1][i2][i3 + 1] && !p[i][i2][i3 + 1])) {
                    p2[i + 1][i2][i3] = 1;
                    answ++;
                }
                if (!p2[i][i2 + 1][i3] && p[i][i2 + 1][i3] && (p[i][i2 + 2][i3] || p[i + 1][i2 + 1][i3] && !p[i + 1][i2][i3] || p[i][i2 + 1][i3 + 1] && !p[i][i2][i3 + 1])) {
                    p2[i][i2 + 1][i3] = 1;
                    answ++;
                }
                if (!p2[i][i2][i3 + 1] && p[i][i2][i3 + 1] && (p[i][i2][i3 + 2] || p[i + 1][i2][i3 + 1] && !p[i + 1][i2][i3] || p[i][i2 + 1][i3 + 1] && !p[i][i2 + 1][i3])) {
                    p2[i][i2][i3 + 1] = 1;
                    answ++;
                }


            }
        }
    }

    cout << answ << endl;

    return 0;
}
