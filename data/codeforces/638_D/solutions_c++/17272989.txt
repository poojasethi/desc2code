#include <iostream>
#include <vector>
using namespace std;

int main(){
	int n, m, K;
	cin >> n >> m >> K;
	
	vector<vector<vector<bool> > > s(n+2, vector< vector<bool> >(m+2, vector<bool>(K+2, false) ));
	
	string in;
	for (int i = 1; i <= n; ++i){
		for (int j = 1; j <= m; ++j){
			cin >> in;
			for(int k = 1; k <= in.size(); ++k){
				s[i][j][k] = in[k-1] - '0';
			}
		}
	}
    
	int ans = 0;
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j)
			for (int k = 1; k <= K; ++k)
				if (s[i][j][k]){
					ans += s[i - 1][j][k] && (s[i + 1][j][k] || s[i][j + 1][k] && !s[i - 1][j + 1][k] || s[i][j][k + 1] && !s[i - 1][j][k + 1])
						|| s[i][j - 1][k] && (s[i][j + 1][k] || s[i + 1][j][k] && !s[i + 1][j - 1][k] || s[i][j][k + 1] && !s[i][j - 1][k + 1]) 
						|| s[i][j][k - 1] && (s[i][j][k + 1] || s[i + 1][j][k] && !s[i + 1][j][k - 1] || s[i][j + 1][k] && !s[i][j + 1][k - 1]);
				}
	cout << ans;
	return 0;
}
