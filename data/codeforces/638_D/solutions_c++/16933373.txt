#include <iostream>
#include <set>
#include <algorithm>
#include <map>
#include <cstdio>
#include <vector>

using namespace std;

char s[120][120][120];
int res = 0;

int main()
{
    int r[3];
    int i[3];
    gets(s[0][0]);
    sscanf(s[0][0],"%d%d%d",r,r+1,r+2);
    for(int i=0;i<r[0];i++)
    {
        for(int j=0;j<r[1];j++)
            gets(s[i][j]);
        gets(s[119][0]);
    }
    int tot = 0;
    for(i[0]=0;i[0]<r[0];i[0]++)
    for(i[1]=0;i[1]<r[1];i[1]++)
    for(i[2]=0;i[2]<r[2];i[2]++)
    if(s[i[0]][i[1]][i[2]]=='1')
    {
        tot++;
        bool ok = false;
        for(int p=0;p<3;p++)
        if(i[p]>0)
        {
            ok = true;
            i[p]--;
            if(s[i[0]][i[1]][i[2]]!='1') ok = false;
            i[p]+=2;
            if(s[i[0]][i[1]][i[2]]!='1') ok = false;
            i[p]--;
            if(ok) break;
        }
        if(ok){res++;continue;}

        for(int p=0;p<3;p++)
        for(int q=0;q<3;q++)
        if(i[p]>0 && i[q]<r[q]-1)
        {
            bool tok = true;
            i[p]--;
            if(s[i[0]][i[1]][i[2]]=='0') tok = false;
            i[p]++; i[q]++;
            if(s[i[0]][i[1]][i[2]]=='0') tok = false;
            i[q]--;
            i[p]--;
            i[q]++;
            if(s[i[0]][i[1]][i[2]]=='1') tok = false;
            i[p]++;
            i[q]--;
            if(tok) ok = true;
        }
        if(ok){res++;continue;}

    }
    cerr<<tot<<endl;
    cout <<res <<endl;
    return 0;
}
