#include <iostream>
#include <cstdio>
#include <string>

using namespace std;

const int N = 100 + 2;

bool a[N][N][N];

bool check(int x, int y, int z) {
    if (a[x - 1][y][z] && a[x + 1][y][z]) return 1;
    if (a[x][y - 1][z] && a[x][y + 1][z]) return 1;
    if (a[x][y][z - 1] && a[x][y][z + 1]) return 1;
    if (a[x - 1][y][z] && a[x][y + 1][z] && !a[x - 1][y + 1][z]) return 1;
    if (a[x - 1][y][z] && a[x][y][z + 1] && !a[x - 1][y][z + 1]) return 1;
    if (a[x][y - 1][z] && a[x + 1][y][z] && !a[x + 1][y - 1][z]) return 1;
    if (a[x][y - 1][z] && a[x][y][z + 1] && !a[x][y - 1][z + 1]) return 1;
    if (a[x][y][z - 1] && a[x + 1][y][z] && !a[x + 1][y][z - 1]) return 1;
    if (a[x][y][z - 1] && a[x][y + 1][z] && !a[x][y + 1][z - 1]) return 1;
    return 0;
}

int main() {
    // freopen("input.txt", "r", stdin);
    int n, m, k;
    cin >> n >> m >> k;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++) {
            string s;
            cin >> s;
            for (int l = 1; l <= k; l++)
                a[i][j][l] = s[l - 1] == '1';
        }
    int ans = 0;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            for (int l = 1; l <= k; l++)
                if (a[i][j][l])
                    ans += check(i, j, l);
    cout << ans << endl;
    return 0;
}
