#include <deque>
#include <cmath>
#include <string>
#include <cstdio>
#include <fstream>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
const int maxn = 100002;
deque<int> a[355];
int c[355][maxn], lastans, n, B;
int deal(int x) {
      return (x + lastans - 1) % n + 1;
}
int main(void)
{
    int t, L, R, k, x, ans, l, r, q;
    scanf("%d", &n); B = (int)sqrt(n+0.0);
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &x);
        a[i/B].push_back(x);
        c[i/B][x]++;
    }
    scanf("%d", &q);
    lastans = 0;
    while (q--)
    {
        scanf("%d %d %d", &t, &L, &R);
        L = deal(L); R = deal(R);
        if (L > R)
            swap(L, R);
        L--;
        l = L / B, r = R / B;
        if (t == 2) {
            scanf("%d", &k);
            k = deal(k);
            ans = 0;
            if (l == r) {
                  for (int i = L%B; i < R%B; i++) ans += a[l][i]==k;
            } else {
                for (int i = l+1; i < r; i++) ans += c[i][k];
                for (int i = L%B; i < a[l].size(); i++) ans += a[l][i]==k;
                for (int i = 0; i < R%B; i++) ans += a[r][i]==k;
            }
            printf("%d\n", ans);
            lastans = ans;
        } else {
              if (l == r) {
                x = a[l][R%B-1];
                a[l].erase(a[l].begin()+R%B-1);
                a[l].insert(a[l].begin()+L%B, x);
            } else {
                  for (int i = l; i < r; i++) {
                    x = a[i].back(); a[i].pop_back(); c[i][x]--;
                    a[i+1].push_front(x); c[i+1][x]++;
                }
                x = a[r][R%B]; a[r].erase(a[r].begin()+R%B); c[r][x]--;
                a[l].insert(a[l].begin()+L%B, x); c[l][x]++;
            }
        }
     }
     return 0;
}
