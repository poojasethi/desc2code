#include"stdio.h"
#include"string.h"
#include"algorithm"
using namespace std;
struct pi{
  int x,y,r,d;
}q[10004],a[300005];
int out[290000];
bool cp1(pi x,pi y)
{
   return x.x<y.x;
}
int main()
{
    int n,m;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
    	scanf("%d %d",&q[i].x,&q[i].r);
	}
	scanf("%d",&m);
	for(int i=0;i<m;i++)
	{
        scanf("%d %d",&a[i].x,&a[i].y);
        a[i].d=i;
	}
	sort(a,a+m,cp1);
	int su=0;
	for(int i=0;i<n;i++)
	{
	    out[i]=10000000;
	    int l,r; 
		if(a[0].x>q[i].x+q[i].r||a[m-1].x<q[i].x-q[i].r)
	    continue;
	    int he=-1,ta=m-1;
	   
	    while(he+1<ta)
	    {
		    int mid=he+ta>>1;
		    if(a[mid].x>=q[i].x-q[i].r)
		    ta=mid;
		    else he=mid;
		}
		l=ta;
		he=0;
		ta=m;
		 while(he+1<ta)
	    {
		    int mid=he+ta>>1;
		    if(a[mid].x<=q[i].x+q[i].r)
		    he=mid;
		    else ta=mid;
		}
		r=he;
		int ju=0;
	//printf("%d %d\n",l,r);
		for(int j=l;j<=r;j++)
		{
		
		  if((a[j].x-q[i].x)*(a[j].x-q[i].x)+a[j].y*a[j].y<=q[i].r*q[i].r)
		  {
		  out[i]=min(a[j].d,out[i]);
		   ju=1;
		  }
		}
		su+=ju;
	}	
	printf("%d\n",su);
	for(int i=0;i<n;i++)
	{
	  if(i)putchar(' ');
	  printf("%d",out[i]==10000000?-1:out[i]+1);
	}
	puts("");
}
