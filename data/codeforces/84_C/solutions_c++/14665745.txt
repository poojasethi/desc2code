# include <bits/stdc++.h>

using namespace std;

# define L first
# define R second

const int maxn = 1e4 + 100;

pair<pair<int,int>,int> a[maxn];

int ans[maxn], X[maxn];

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(0); cout.tie(0);
	
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i].L.L >> a[i].L.R, a[i].R = i;
	sort(a, a + n);

	for (int i = 0; i < n; i++)
		X[i] = a[i].L.L;

	int m;
	cin >> m;
	memset (ans, -1, sizeof ans);
	for (int i = 0; i < m; i++){
		int x, y;
		cin >> x >> y;

		int idx = lower_bound(X, X + n, x) - X;
		if(idx < n && y * y + (a[idx].L.L - x) * (a[idx].L.L - x) <= a[idx].L.R * a[idx].L.R && ans[a[idx].R] == -1)
			ans[a[idx].R] = i;
		
		idx--;
		if(idx >= 0 && y * y + (a[idx].L.L - x) * (a[idx].L.L - x) <= a[idx].L.R * a[idx].L.R && ans[a[idx].R] == -1)
			ans[a[idx].R] = i;
	}
	
	int number = 0;
	for (int i = 0; i < n; i++)
		if(ans[i] != -1)
			number++;

	cout << number << endl;
	for (int i = 0; i < n; i++)
		cout << (ans[i] == -1 ? -1 : ans[i] + 1) << " ";
	cout << endl;

	return 0;
}
