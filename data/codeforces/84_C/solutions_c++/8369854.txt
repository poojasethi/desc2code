#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
int a[10000],s[10000],d[10000];
void qs(int q,int w){
  int e=q,r=w,t=a[q+(rand()%(w-q+1))],y;
  do{
  while(a[e]<t)e++;
  while(a[r]>t)r--;
  if(e<=r){
   y=a[e];a[e]=a[r];a[r]=y;
   y=s[e];s[e]=s[r];s[r]=y;
   y=d[e];d[e]=d[r];d[r]=y;
   e++;r--;}
  }while(e<=r);
  if(q<r)qs(q,r);
  if(e<w)qs(e,w);
}
bool l(int q,int w,int e,int r,int t){
  q=(q-e)*(q-e)+(w-r)*(w-r)-t;
  if(q<=0)
   return 1;
  return 0;
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t,c,v,f[10000];
  cin>>q;
  for(w=0;w<q;w++){
   cin>>a[w]>>s[w];
   d[w]=w;
   f[w]=900000;}
  qs(0,q-1);
  cin>>t;
  for(w=0;w<t;w++){
   cin>>c>>v;
   e=0;
   for(r=8192;r;r>>=1)
    if(e+r<q)
     if(a[e+r]-s[e+r]<=c)
      e+=r;
   if(l(c,v,a[e],0,s[e]*s[e]))
    f[d[e]]=min(f[d[e]],w);
   if((e)&&(l(c,v,a[e-1],0,s[e-1]*s[e-1])))
    f[d[e-1]]=min(f[d[e-1]],w);}
  t=0;
  for(w=0;w<q;w++)
   if(f[w]<900000)
    t++;
  cout<<t<<"\n";
  for(w=0;w<q;w++)
   if(f[w]==900000)
    cout<<"-1 ";
   else
    cout<<f[w]+1<<" ";
  return 0;}
