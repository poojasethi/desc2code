#include <cstdio>
#include <algorithm>

using namespace std;

typedef pair<int,int> pnt;

int cir[10005][2];
pnt tar[10005];
int hit[10005];
int n,m;

int mn(int a,int b) { return (a<b&&a!=-1) ? a : b; }
int isin(int a,int b,int c,int d) { return ((a-c)*(a-c))+(b*b) <= (d*d); }
int comp(pnt a,pnt b) { return a.first <= b.first; }

int main()
{
	int i,j;
	int a,b,c,d;
	scanf("%d",&n);
	for(i=0;i<n;i++) {
		scanf("%d %d",&cir[i][0],&cir[i][1]);
		tar[i] = pnt(cir[i][0],i);
		hit[i] = -1;
	}
	sort(tar,tar+n);
	scanf("%d",&m);
	for(i=0;i<m;i++) {
		scanf("%d %d",&a,&b);
		j = lower_bound(tar,tar+n,pnt(a,0),comp) - tar;
		d = -1;
		//printf("J = %d\n",j);
		if(isin(a,b,cir[tar[j].second][0],cir[tar[j].second][1])) {
			hit[tar[j].second] = mn(hit[tar[j].second],i+1);
			//printf("HIT %d\n",j);
		}
		if(j>0&&isin(a,b,cir[tar[j-1].second][0],cir[tar[j-1].second][1])) {
			hit[tar[j-1].second] = mn(hit[tar[j-1].second],i+1);
			//printf("HIT %d\n",j-1);
		}
	}
	for(i=0,d=0;i<n;i++) d += (hit[i]!=-1);
	printf("%d\n",d);
	for(i=0;i<n;i++) printf("%d ",hit[i]);
	printf("\n");
	return 0;
}
