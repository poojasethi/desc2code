#include <cmath>
#include <cstdio>
#include <algorithm>

using namespace std;

const int MAXM = (int)1e4 + 5;

struct Mish {
 int num;
 double x, r;
} m[MAXM];

int n, k, ind, a[MAXM], kol;
double x, y;

bool cmp(Mish a, Mish b) {
 return a.x < b.x;
}

double dist(Mish a, double x, double y) {
 return sqrt((a.x - x) * (a.x - x) + y * y);
}

int binarySearch(int x) {
 int l = 0, r = n, mid;

 while (l < r) {
  mid = (l + r) / 2;
  if (m[mid].x < x)
   l = mid + 1;
  else
   r = mid;
 }

 return l;
}

void check(int ind, int num) {
 if (!a[m[ind].num])
  if (dist(m[ind], x, y) <= m[ind].r)
   a[m[ind].num] = num + 1, kol++;
}

int main() {
 #ifndef ONLINE_JUDGE
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
 #endif

 scanf("%d", &n);
 for (int i = 0; i < n; i++)
  scanf("%lf%lf", &m[i].x, &m[i].r), m[i].num = i;
 sort(m, m + n, cmp);

 scanf("%d", &k);
 for (int i = 0; i < k; i++) {
  scanf("%lf%lf", &x, &y);
  ind = binarySearch((int)x);
  check(ind, i), check(ind - 1, i);
 } 

 printf("%d\n", kol);
 for (int i = 0; i < n; i++)
  printf("%d ", a[i] ? a[i] : -1);

 return 0;
}
