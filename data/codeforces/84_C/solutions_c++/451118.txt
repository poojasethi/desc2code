#include<stdio.h>
#include<algorithm>
using namespace std;
struct A{
 long c;
 long r;
 long in;
}a[10004];
long p[10004],n,m,i,j,k,s,r,x,y,ini,fi,mi;
bool f(A xx, A yy){
 return (xx.c<yy.c);
}
int main(){
 scanf("%ld",&n);
 for(i=1;i<=n;i++){
  scanf("%ld %ld",&a[i].c,&a[i].r);
  a[i].r=a[i].r*a[i].r;
  a[i].in=i;
 }
 sort(a+1,a+n+1,f);
 a[0].c=-22000;
 a[0].r=1;
 a[n+1].c=22000;
 a[n+1].r=1;
 a[n+1].in=n+1;
 scanf("%ld",&m);
 for(i=1;i<=m;i++){
  scanf("%ld %ld",&x,&y);
  ini=0;
  fi=n+1;
  uno:
  mi=((ini+fi)/2);
  if(a[mi].c>=x&&a[mi-1].c<x){
    if(p[a[mi-1].in]==0&&(((x-a[mi-1].c)*(x-a[mi-1].c)+y*y)<=a[mi-1].r)){
         p[a[mi-1].in]=i+1;
         s=s+1;
    }
    if(p[a[mi].in]==0&&(((x-a[mi].c)*(x-a[mi].c)+y*y)<=a[mi].r)){
        p[a[mi].in]=i+1;
        s=s+1;
    }
  }
  else{
   if(a[mi].c<x){
     ini=mi+1;
    }
    else{
     fi=mi-1;
    }
   goto uno;
  }
 }
 printf("%ld\n%ld",s,p[1]-1);
 for(i=2;i<=n;i++)
    printf(" %ld",p[i]-1);
 printf("\n");
}
