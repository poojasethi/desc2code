#include<iostream>
#include<cmath>
#include <string>
#include <map>
#include <cstdio>
using namespace std;
typedef long long LL;
const int fix=30000;
int n,m;
int c[100005];
int coo[10005],ra[10005];
int res[10005];
int num;
void add(int k,int v)
{
    for(int i=k;i<=50000;i+=i&-i)
        c[i]+=v;
}
int sum(int k)
{
    int ans=0;
    for(int i=k;i>0;i-=i&-i)
        ans+=c[i];
    return ans;
}
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        int tp1,tp2;
        scanf("%d %d",&tp1,&tp2);
        add(tp1-tp2+fix,i);
        add(tp1+tp2+fix,-i);
        coo[i]=tp1+fix;
        ra[i]=tp2;
    }
    scanf("%d",&m);
    for(int i=1;i<=m;i++)
    {
        int x,y;
        scanf("%d %d",&x,&y);
        int a=sum(x+fix);
        if(a&&!res[a]&&(x+fix-coo[a])*(x+fix-coo[a])+y*y<=ra[a]*ra[a])
        {
            res[a]=i;
            num++;
        }
         a=sum(x+fix-1);
        if(a&&!res[a]&&(x+fix-coo[a])*(x+fix-coo[a])+y*y<=ra[a]*ra[a])
        {
            res[a]=i;
            num++;
        }
    }
    printf("%d\n",num);
    for(int i=1;i<=n;i++)
    {
        if(res[i])
            printf("%d ",res[i]);
        else
            printf("-1 ");
    }
}
