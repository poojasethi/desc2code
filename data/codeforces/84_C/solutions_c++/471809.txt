#include<iostream>
#include<algorithm>
using namespace std;
const int MAXN=10000+10;
int n, k, x[MAXN], r[MAXN], a[MAXN], ans[MAXN], cnt;
bool cmp(int a, int b){
	return x[a]<x[b];
}
bool cmp2(int a, int b){
	return x[a]<b;
}
int main(){
	cin>>n;
	for(int i=0; i<n; i++){
		cin>>x[i]>>r[i];
		a[i]=i;
	}	
	sort(a, a+n, cmp);
	cin>>k;
	for(int i=0, z, y; i<k; i++){
		cin>>z>>y;
		int ind=lower_bound(a, a+n, z, cmp2)-a;
		if(ind<n &&  (x[a[ind]]-z)*(x[a[ind]]-z)+y*y<=r[a[ind]]*r[a[ind]] && ans[a[ind]]==0)	ans[a[ind]]=i+1;
		ind--;
		if(ind>=0 && (x[a[ind]]-z)*(x[a[ind]]-z)+y*y<=r[a[ind]]*r[a[ind]] && ans[a[ind]]==0)	ans[a[ind]]=i+1;
	}
	for(int i=0; i<n; i++)
		if(ans[i]!=0)	
			cnt++;
	cout<<cnt<<endl;
	for(int i=0; i<n; i++)	
		if(ans[i]==0)	cout<<-1<<" ";
		else cout<<ans[i]<<" ";
	return 0;
}
