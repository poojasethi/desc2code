#include<bits/stdc++.h>
using namespace std;
const int N = 1e4 + 9;
typedef long long ll;
map< ll  , int > id;
ll  n , m , x, y ,h[N],c[N],r[N],th[N],ans;
void solve( ll a, ll b ,int ids ,int t)
{
     if((a - th[ids])*(a - th[ids]) + b*b - r[ids]*r[ids] <= 0)
     {
          if( c[ids] == -1)
           c[ids] = t , ans++;
     }      
}
int main()
{
	int i ,p,f;
    ios::sync_with_stdio(false);
    cin >> n;
    memset(c, -1 ,sizeof(c) );
    for( i = 0  ; i < n ; i++ )
    {
         cin >> h[i] >> r[i];
         id[h[i]] = i;
         th[i] = h[i];	
    } 
    sort( h ,h + n);
    cin >> m;
    for( i = 1 ; i <= m ; i++)
    {
    	 cin >> x >> y;
    	 f = 0;
    	 p = lower_bound(h ,h + n ,x ) - h;
         if( p >= n )
         	p--;
         solve(x , y, id[h[p]] ,i);
         if( p > 0 )
         {
         	p--;
           solve(x , y, id[h[p]] ,i);
         }
    }
    cout << ans << "\n";
    for( i = 0 ; i  < n ; i++ )
    cout << c[i] << " "; 
	return 0;
}
