#include <iostream>
#include <map>
#include <cmath>
using namespace std;

struct krug
{
    int x,r,k;
    krug()
    {
        k=-1;
    }
    int inmyrange(int ex, int ey, int n)
    {
        if(k==-1)
        {
            if(sqrt((double)(x-ex)*(x-ex)+ey*ey)<=r)
            {
                k=n;
                return 1;
            }
        }
        return 0;
    }
};

int main()
{
    map<int, krug*> ks;
    int n;
    cin>>n;
    krug* ar= new krug[n];
    for(int i=0; i<n; i++)
    {
        cin>>ar[i].x>>ar[i].r;
        for(int j=ar[i].x-ar[i].r; j<=ar[i].x+ar[i].r; j++)
            ks[j]=&ar[i];

    }
    int m,x,y;
    cin>>m;
    int v=0;
    for(int i=1; i<=m; i++)
    {
        cin>>x>>y;
        if(ks.count(x-1))
            v+=ks[x-1]->inmyrange(x,y,i);
        if(ks.count(x))
            v+=ks[x]->inmyrange(x,y,i);
        if(ks.count(x+1))
            v+=ks[x+1]->inmyrange(x,y,i);
    }
    cout<<v<<endl;
    for(int i=0; i<n; i++)
        cout<<ar[i].k<<" ";
    return 0;
}
