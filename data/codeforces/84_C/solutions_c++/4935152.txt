#include <iostream>
#include <set>

using namespace std;

int n,m;
int c[1000000],r[1000000],x[1000000],y[1000000];
int sol[1000000];

int cuad(int a)
{
  return a*a;
}

bool interior(int ic,int ip)
{
  return cuad(r[ic])>=cuad(x[ip]-c[ic])+cuad(y[ip]);
}

int main()
{
  cin>>n;
  set<pair<int,int> > s;
  for (int i=1;i<=n;i++) {
    cin>>c[i]>>r[i];
    s.insert(pair<int,int> (c[i],i));
  }
  for (int i=1;i<=n;i++)
    sol[i]=-1;
  cin>>m;
  for (int i=1;i<=m;i++)
    cin>>x[i]>>y[i];
  for (int i=m;i>=1;i--) {
    set<pair<int,int> >::iterator it=s.upper_bound(pair<int,int> (x[i],0));
    if (it!=s.end()) {
      int ic=(*it).second;
      if (interior(ic,i))
	sol[ic]=i;
    }
    if (it!=s.begin()) {
      it--;
      int ic=(*it).second;
      if (interior(ic,i))
	sol[ic]=i;
    }
  }
  int count=0;
  for (int i=1;i<=n;i++)
    count+=sol[i]!=-1;
  cout<<count<<endl;
  for (int i=1;i<=n;i++) {
    if (i>1) cout<<" ";
    cout<<sol[i];
  }
  cout<<endl;
}
