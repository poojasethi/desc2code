#include<iostream>
#include<algorithm>
using namespace std;
typedef pair<int,int> ii;
#define fi first.first
#define se first.second
#define id second

int sq(int a) {
	return a*a;
}
pair<ii,int> circ[10010];
int hit[10010];
int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++) {
		cin>>circ[i].fi>>circ[i].se;
		circ[i].id=i;
	}
	circ[n].fi=20999;
	circ[n+1].fi=-20999;
	n+=2;
	sort(circ,circ+n);
	int m;
	cin>>m;
	int ans=0;
	for(int i=1;i<=m;i++) {
		int x,y;
		cin>>x>>y;
		int t=upper_bound(circ,circ+n,make_pair(ii(x,y),0))-circ;
		if(sq(x-circ[t].fi)+sq(y)<=sq(circ[t].se)) {
			if(!hit[circ[t].id]) {
				ans++;
				hit[circ[t].id]=i+1;
			}
		}
		t--;
		if(sq(x-circ[t].fi)+sq(y)<=sq(circ[t].se)) {
			if(!hit[circ[t].id]) {
				ans++;
				hit[circ[t].id]=i+1;
			}
		}
	}
	cout<<ans<<endl;
	for(int i=0;i<n-2;i++) {
		cout<<hit[i]-1<<' ';
	}
}
