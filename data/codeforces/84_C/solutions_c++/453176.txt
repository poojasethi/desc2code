#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
struct circle
{
    long long x, r,ind;
};
circle c[10000];
int ans[10000],res;
long long  n,t,i,m,x,y;

bool cmp (circle a, circle b)
{
    return a.x<b.x;
}
void bs (long long  x, long long  y)
{
    int l,r,m;
    l=0;
    r=n-1;
    while (l!=r)
    {
        m=(l+r)/2;
        if (x<=c[m].x)
            r=m;
        else
            l=m+1;
    }
    if (y*y+(x-c[l].x)*(x-c[l].x)<=c[l].r*c[l].r)
    {
     if (ans[c[l].ind]==0) ans[c[l].ind]=t;
    }
    if (l>0)
    if (y*y+(x-c[l-1].x)*(x-c[l-1].x)<=c[l-1].r*c[l-1].r)
    {
     if (ans[c[l-1].ind]==0)
     ans[c[l-1].ind]=t;
    }

}


int main()
{

    //freopen("input.txt","r",stdin);
    cin>>n;
    for (i=0; i<n; i++)
    {
     cin>>c[i].x>>c[i].r;
     c[i].ind=i;
    }
    cin>>m;
    sort(c,c+n,cmp);
    for (t=1; t<=m; t++)
    {
        cin>>x>>y;
        bs (x,y);
    }

    for (i=0; i<n; i++)
     if (ans[i]!=0) res++;
    cout<<res<<endl;
    for (i=0; i<n; i++)
     if (ans[i]==0) cout<<-1<<" ";
      else cout<<ans[i]<<" ";
    return 0;
}
