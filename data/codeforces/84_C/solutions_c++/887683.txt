#include <cstdio>
#include <cstring>
#include <algorithm>
#include <map>
using namespace std;

int a[10005], r[10005];
int h[10005], v[10005];
bool cmp(int x, int y) {
	return a[x] < a[y];
}
map<int, int> id;

int win=0, mm;
void test(int i, int x, int y) {
	if(h[i]>=0) return;
	if((a[i]-x)*(a[i]-x)+y*y <= r[i]*r[i]) {
		h[i] = mm;
		++win;
	}
}
int srt[100005];
int main(void) {
	int n, m;
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%d%d", &a[i], &r[i]);
		srt[i] = i;
		v[i]=a[i];
		id[v[i]] = i;
	}
	sort(v, v+n);
	sort(srt, srt+n, cmp);
	scanf("%d", &m);
	memset(h, -1, sizeof(h));
	for(mm=1;mm<=m;mm++) {
		int x, y;
		scanf("%d%d", &x, &y);
		int p = lower_bound(v, v+n, x) - v;
		if(p>=n) --p;
		int i = id[v[p]];
		test(i, x, y);
		if(p>0) {
			--p;
			test(id[v[p]], x, y);
		}
	}
	printf("%d\n", win);
	for(int i=0;i<n;i++)
		printf("%d ", h[i]);
	puts("");
	return 0;
}
