#include <iostream>
#include <fstream>
#include <string>

using namespace std;

const int MAX = 2 * 10 * 1000 + 1;

bool th[MAX + 10];
int tx[MAX + 10];
int tr[MAX + 10];
int tf[MAX + 10];

int x[2 * MAX + 2000][2];

int main()
{
	int n, m, c = 0;
	cin >> n;
	for(int i = 1; i <= n; i++)
	{
		cin >> tx[i] >> tr[i];
		for(int j = -1 * tr[i]; j <= tr[i]; j++)
		{
			if(!x[tx[i] + j + MAX + 1000][0])
				x[tx[i] + j + MAX + 1000][0] = i;
			else
				x[tx[i] + j + MAX + 1000][1] = i;
		}
	}
	cin >> m;
	for(int i = 0, a, b; i < m; i++)
	{
		cin >> a >> b;
		int t[2] = {x[a + MAX + 1000][0], x[a + MAX + 1000][1]};
		for(int j = 0; j < 2; j++)
			if(b * b <= tr[t[j]] * tr[t[j]] - (a - tx[t[j]]) * (a - tx[t[j]]) && t[j])
			{
				th[t[j]] = true;
				if(!tf[t[j]])
					tf[t[j]] = i + 1;
			}
	}
//	cerr << x[3 + MAX + 1000][0] << ' ' << x[3 + MAX + 1000][1] << endl;
	for(int i = 1; i <= n; i++)
		if(th[i])
			c++;
	cout << c << endl;
	for(int i = 1; i <= n; i++)
		cout << (tf[i]?tf[i]:-1)  << ' ';
	cout << endl;
	return 0;
}
