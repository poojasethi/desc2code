#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
struct C{
  int x,w,r;
};
struct CC{
  int x,y,w;
};
C a[10010];
CC b[200010];
int n,m;
int ans[10010],t;
int cmpa(C x,C y){
  if (x.x<y.x) return 1;
  return 0;
}
int cmpb(CC x,CC y){
  if (x.x<y.x) return 1;
  return 0;
}
int main(){
  int i,j,k;
  scanf("%d",&n);
  for (i=0;i<n;i++){
    scanf("%d%d",&a[i].x,&a[i].r);
    a[i].w=i+1;
  }
  scanf("%d",&m);
  for (i=0;i<m;i++){
    scanf("%d%d",&b[i].x,&b[i].y);
    b[i].w=i+1;
  }
  sort(a,a+n,cmpa);
  sort(b,b+m,cmpb);
  j=0;
  memset(ans,-1,sizeof(ans));
  for (i=0;i<n;i++){
    if (i>0)
      if (a[i-1].x+a[i-1].r==a[i].x-a[i].r){
        j--;
        while (j>=0&&b[j].x==a[i].x-a[i].r) j--;
        j++;
      }
    while (j<m){
      if (b[j].x>a[i].x+a[i].r) break;
      if ((b[j].x-a[i].x)*(b[j].x-a[i].x)+b[j].y*b[j].y<=a[i].r*a[i].r)
        if (ans[a[i].w]==-1||ans[a[i].w]>b[j].w){
          if (ans[a[i].w]==-1) t++;
          ans[a[i].w]=b[j].w;
        }
      j++;
    }
  }
  printf("%d\n",t);
  for (i=1;i<n;i++) printf("%d ",ans[i]);
  printf("%d\n",ans[i]);
  return 0;
}
