#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>
using namespace std;
#define f first
#define s second
#define mp make_pair
#define pb push_back
#define ll long long
#define INF 1<<30
#define PI 3.1415926535897932
#define EPS 1e-9
#define sqr(x) ((x)*(x))
vector <pair <pair<int,int>,int> > sht,a;
vector <int> ans;
int main()
{
  int n,m,x,y;
  cin>>n;
  ans.resize(n+1,INF);
  for(int i=0;i<n;i++)scanf("%d%d",&x,&y),a.pb(mp(mp(x-y,y+x),i));
  cin>>m;
  for(int i=0;i<m;i++)scanf("%d%d",&x,&y),sht.pb(mp(mp(x,y),i+1));
  sort(a.begin(),a.end());
  sort(sht.begin(),sht.end());
  int j=0,answ=0;
  for(int i=0;i<n;i++)
    for(;j<m && sht[j].f.f<=a[i].f.s;j++)
    {
      y=(a[i].f.s-a[i].f.f)/2;
      x=a[i].f.f+y;
      if (sqr(x-sht[j].f.f)+sqr(sht[j].f.s)<=y*y)ans[a[i].s]=min(ans[a[i].s],sht[j].s);
      if (i<n && a[i].f.s==a[i+1].f.f && a[i].f.s==sht[j].f.f && sht[j].f.s==0)ans[a[i+1].s]=min(ans[a[i+1].s],sht[j].s);
    }
  for(int i=0;i<n;i++)
    if (ans[i]!=INF)answ++;
  cout<<answ<<endl;
  for(int i=0;i<n;i++)
    if (ans[i]==INF)printf("-1 ");
    else printf("%d ",ans[i]);
  return 0;
}
