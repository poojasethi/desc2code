#include <iostream>
#include <vector>
#include <cstdlib>
#include <algorithm>
#include <cstdio>
using namespace std;
int z[10010],r[10010],d[10010],ans[10010];

bool ls(int i,int j){
	return z[i]<z[j];
}
bool comp(int i,int j){
	//cerr<<'!'<<i<<' '<<j<<endl;
	//cerr<<'?'<<z[i]-r[i]<<' '<<z[i]+r[i]<<endl;
	return (z[i]-r[i]<j);
}
int main(){
	int n;
	cin>>n;
	for (int i=0;i<n;i++){
		cin>>z[i]>>r[i];
		d[i]=i;
	}
	int m,x,y,cnt=0,k;
	sort(d,d+n,ls);
	cin>>m;
	for (int i=1;i<=m;i++){
		cin>>x>>y;
		k=lower_bound(d,d+n,x,comp)-d;
		if (ans[d[k]]==0 && (x-z[d[k]])*(x-z[d[k]])+y*y<=r[d[k]]*r[d[k]]) ans[d[k]]=i,cnt++;
		k--;
		if (k<0) continue;
		if (ans[d[k]]==0 && (x-z[d[k]])*(x-z[d[k]])+y*y<=r[d[k]]*r[d[k]]) ans[d[k]]=i,cnt++;
	}
	cout<<cnt<<endl;
	for (int i=0;i<n;i++)
		if (ans[i]==0) cout<<-1<<' '; else cout<<ans[i]<<' ';
	return 0;
}
