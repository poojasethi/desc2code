from bisect import bisect_left as bl
r=lambda:map(int,raw_input().split())
n=input()
a=[r()+[i] for i in range(n)]
f=[-1]*n
a.sort()
t=0
for i in range(input()):
	x,y=r()
	j=bl(a,[x])
	for k in (j-1,j):
		if 0<=k<n and (x-a[k][0])**2+y**2<=a[k][1]**2:
			m=a[k][2]
			if f[m]==-1:
				f[m]=i+1
				t+=1
print t
print ' '.join(map(str,f))
