from bisect import bisect_left as bl
n = input()
a = [map(int,raw_input().split())+[i] for i in xrange(n)]
v = [-1]*n
a.sort()
r = 0
for i in xrange(input()):
    x,y = map(int,raw_input().split())
    k = bl(a,[x,0])
    for k in (z for z in (k-1,k) if 0<=z<n and (x-a[z][0])**2+y**2<=a[z][1]**2 and v[a[z][2]]==-1):
        v[a[k][2]]=i+1
        r+=1
print r
print ' '.join(map(str,v))
