from bisect import bisect_left as BL
xr=[map(int,raw_input().split())+[i] for i in range(input())]
xr.sort()
h=[-1]*len(xr)
for i in range(input()):
  x,y=map(int,raw_input().split())
  k=BL(xr,[x,0])
  for c in xr[max(0,k-1):k+1]:
    if h[c[2]]<0 and (x-c[0])**2+y**2<=c[1]**2:h[c[2]]=i+1
print len(xr)-h.count(-1)
print ' '.join(map(str,h))
