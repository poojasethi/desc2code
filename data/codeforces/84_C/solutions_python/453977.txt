import bisect
get=lambda: map(int, raw_input().split())
ok =lambda c,r,x,y: (x-c)**2+y**2 <= r**2
N=input(); p=[-1]*N; k=0
t=sorted([get()+[n] for n in xrange(N)]+[[-22000,0,0],[22000,0,0]], key=lambda z:z[0])
for m in xrange(input()):
    x,y=get(); i=bisect.bisect(t,[x,0,0])
    c,r,n=t[i]
    if p[n]==-1 and ok(c,r,x,y):
        p[n]=m+1; k+=1
    c,r,n=t[i-1]
    if p[n]==-1 and ok(c,r,x,y):
        p[n]=m+1; k+=1
print k
print ' '.join(map(str,p))
