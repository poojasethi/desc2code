#include <iostream>
#include <algorithm>

using namespace std;

const int maxN = 1000 * 1000 + 10;

int b[maxN], cnt[50], ans;

int main()
{
   ios::sync_with_stdio(false);
   int n, mx = 0, m;
   cin >> n >> m;
   for (int i = 0; i < n; i++)
   {
      int a, ind = 0;
      cin >> a;
      while (a)
      {
         if (a % 2)
            cnt[ind]++;
         ind++;
         a /= 2;
      }
      mx = max(mx, ind);
   }
   for (int i = 0; i < m; i++)
      cin >> b[i];
   sort(b, b + m);
   for (int i = 0; i < m; i++)
   {
      int a = b[i];
      for (int j = a; j < mx; j++)
         if (cnt[j] > 0)
         {
            cnt[j]--;
            for (int k = a; k < j; k++)
               cnt[k]++;
            ans++;
            break;
         }
   }
   cout << ans << endl;
   return 0;
}
