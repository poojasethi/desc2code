#include<stdio.h>
#include<algorithm>

#define maxdim 1000005

using namespace std;

int n,m;
int a[maxdim],B[35];

int main () {
	
	#ifndef ONLINE_JUDGE
	freopen("croc_final.in","r",stdin);
	freopen("croc_final.out","w",stdout);
	#endif
	
	scanf("%d %d",&n,&m);
	int x;
	for ( int i = 1 ; i <= n ; ++i ){
		scanf("%d",&x);
		
		for ( int b = 0 ; b <= 30 ; ++b ){
			if ( x & (1<<b) ){
				++B[b];
			}
		}
	}
	
	for ( int i = 1 ; i <= m ; ++i ){
		scanf("%d",&a[i]);
	}
	sort(a+1,a+m+1);
	
	int sol = 0;
	for ( int i = 1 ; i <= m ; ++i ){
		int now = a[i];
		int where = now;
		while ( where <= 30 && !B[where] )	++where;
		if ( where == 31 )	continue ;
		
		++sol;
		for ( int j = now ; j < where ; ++j ){
			++B[j];
		}
		--B[where];
	}
	
	printf("%d\n",sol);
	
	return 0;
}
