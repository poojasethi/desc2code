#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#define MAXL 31
#define MAXN 1000005
using namespace std ;

int N, M ; 
int a[MAXN], p[MAXN] ;
int sum[MAXL] ;

int main()
{
    int i, j ;
    scanf("%d %d", &N, &M) ;
    for(i = 1; i <= N; i ++)
    {
        scanf("%d", &a[i]) ;
        for(j = 0; j <= 29; j ++)
            if((a[i]>>j)&1) sum[j] ++ ;
    }
    for(i = 1; i <= M; i ++)
        scanf("%d", &p[i]) ;
    sort(p+1, p+1+M) ;
    for(i = 1; i <= M; i ++)
    {
        for(j = p[i]; j <= 29; j ++)
            if(sum[j]) break ;
        if(j > 29) break ;
        sum[j] -- ;
        for(j --; j >= p[i]; j --)
            sum[j] ++ ;
    }
    printf("%d\n", i-1) ;
    //system("pause") ;
    return 0 ;
}
