#include<stdio.h>
#include<string.h>
#include<functional>
#include<algorithm>
using namespace std;

int a[35]={0};
int b[35]={0};
int main()
{
    int n,m;
    int i,j,k;scanf("%d%d",&n,&m);
    for(i=1;i<=n;i++)
    {
        scanf("%d",&k);
        for(j=0;j<32;j++)
        {
            if((1<<j)&k) a[j]++;
        }
    }
    for(i=1;i<=m;i++)
    {
        scanf("%d",&k);b[k]++;
    }
    int ans=0;
    while(1)
    {
        i=0;
        while(i<32&&b[i]==0) i++;
        b[i]--;
        j=i;
        while(j<32&&a[j]==0) j++;
        if(i>=32||j>=32) break;
        a[j]--;ans++;for(k=j-1;k>=i;k--) a[k]++;
    }
    printf("%d\n",ans);
    return 0;
}
