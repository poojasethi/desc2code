#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1000010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,s;
int g[30],a[30],b[N];
int main() {
	scanf("%d%d",&n,&m);
	g[0]=1; For(i,1,29) g[i]=g[i-1]*2;
	For(i,1,n) {
		scanf("%d",&k);
		For(j,0,29) if (k/g[j]%2) a[j]++;
	}
	For(i,1,m) scanf("%d",&b[i]);
	sort(b+1,b+m+1);
	For(i,1,m) {
		For(j,b[i],29) if (a[j]) break;
		if (j==30) break;
		s++;
		a[j]--;
		For(k,b[i],j-1) a[k]++;
	}
	printf("%d\n",s);
	//for(;;);
	return 0;
}
