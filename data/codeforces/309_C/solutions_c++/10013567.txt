#include <iostream>
#include <cstdio>
#include <fstream>
#include <algorithm>
#include <cmath>
#include <set>
#include <iomanip>

using namespace std;

long long a[31];
int b[1000000],ans,n,m,d;

int main()
{
    scanf("%d%d",&n,&m);
    while(n--)
    {
        scanf("%d",&d);
    	for(int j=29;j>=0;j--)
    		if(d>=(1<<j))
    		{
    			a[j]++;
    			d-=(1<<j);
    		}
    }
    for(int i=0;i<30;i++)
    	a[i]*=(1<<i);
    for(int i=0;i<m;i++)
    {
        scanf("%d",&b[i]);
    	b[i]=(1<<b[i]);
    }
    sort(b,b+m);
    int j=0,i=0;
    while(i<30 && j<m)
    {
        if(a[i]>=b[j] && b[j]<=(1<<i))
        {
            ans++;
            a[i]-=b[j];
            j++;
        }
        else
        	i++;
    }
    cout<<ans;
    return 0;
}
