#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxn = 1000003;

int a[maxn];
int b[maxn];
int c[31];

int main() {
	int n, m, i, j, k;
	scanf("%d%d",&n,&m);
	for (i=0; i < n; i++) {
		scanf("%d",&a[i]);
		for (j=0; j < 31; j++) if (a[i]&(1<<j)) {
			c[j]++;
		}
	}
	for (i=0; i < m; i++) {
		scanf("%d",&b[i]);
	}
	sort(b, b+m);
	int ans = 0;
	for (i=0; i < m; i++) {
		for (j=b[i]; j < 31; j++) if (c[j]) {
			ans++;
			c[j]--; 
			for (k=b[i]; k < j; k++) {
				c[k]++;
			}
			break;
		}
	}
	printf("%d\n",ans);
	return 0; 
}
