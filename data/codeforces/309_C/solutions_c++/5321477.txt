#include <cstdio>
using namespace std;

int a[32], b[32];

int main() {
    int n, m;

    scanf("%d%d", &n, &m);
    for (int i=0,t; i<n; ++i) {
        scanf("%d", &t);
        for (int k=0; k<31; ++k) a[k] += t>>k&1;
    }
    for (int i=0,t; i<m; ++i) scanf("%d", &t), ++b[t];
    int tot = 0;
    for (int i=0,j; ;) {
        for (; i<31&&b[i]==0; ++i);
        for (j=i; j<31&&a[j]==0; ++a[j++]);
        if (j >= 31) break;
        --b[i], --a[j], ++tot;
    }
    printf("%d\n", tot);

    return 0;
}
