#include <cstdio>

int a[30], b[30], c;

int main() {
  int n, m;
  scanf("%d%d", &n, &m);
  while (n--) {
    int x;
    scanf("%d", &x);
    for (int j(0); x; ++j, x >>= 1)
      a[j] += x & 1;
  }
  while (m--) {
    int x;
    scanf("%d", &x);
    ++b[x];
  }
  for (int i(0), j(0); i < 30 && j < 30; ++i)
    while (b[i]) {
      for (j = i; j < 30 && !a[j]; ++j)
        ++a[j];
      if (j == 30)
        break;
      --a[j];
      --b[i];
      ++c;
    }
  printf("%d\n", c);
}
