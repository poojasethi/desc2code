#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <string>
using namespace std;
typedef __int64 Uint;
#define N 1000050
int bit[35];
int b[N];
int main()
{
    int n,m,a;
    scanf("%d%d",&n,&m);
    memset(bit,0,sizeof(bit));
    for(int i=0; i<n; i++)
    {
        scanf("%d",&a);
        for(int j=0;j<32;j++)
        if(a&(1<<j))bit[j]++;
    }
    for(int i=0; i<m;i++)
        scanf("%d",&b[i]);
    sort(b,b+m);

    int cnt=0;
    int j;
    for(int i=0; i<m; i++)
    {
        for(j=b[i];j<32 && bit[j]==0;j++);
        if(j==32)break;
        bit[j]--;
        cnt++;
        for(j--;j>=b[i];j--)bit[j]++;
    }
    printf("%d\n",cnt);
    return 0;
}
