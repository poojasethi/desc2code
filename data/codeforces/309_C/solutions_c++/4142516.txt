#include <iostream>
#include <algorithm>

using namespace std;

const int MAX_N = 100 * 100 * 100 + 10;

int a[MAX_N] , b[MAX_N] , bin[31+5];

int main()
{
	ios::sync_with_stdio(false);
	int n , m;
	int ans = 0;
	cin >> n >> m;
	for(int i = 0; i < n; i++)
	{
		cin >> a[i];
		for(int j = 0; j < 31; j++)
			bin[j] += (a[i]>>j)&1;
	}
	for(int i = 0; i < m; i++)
	{
		cin >> b[i];
	}
	sort(b , b + m);
	for(int i = 0; i < m; i++)
	{
		if(bin[b[i]] != 0)	
			bin[b[i]]--;
		else
		{
			int k = b[i];
			while(k < 32 && bin[k] == 0)
			{
				bin[k]++;
				k++;
			}
			if (bin[k] > 0)
				bin[k]--;
			else
				break;	
		}		
		ans += 1;
	}
	cout << ans;
	return 0;
}
