#include <iostream>
#include <vector>
#include <set>
#include <cstdio>
#include <algorithm>
using namespace std;
const int N=1e6+9;
int n,m,t[N],ans,B[31];
int main()
{
	scanf("%d%d",&n,&m);
	for( int i=0 ; i<n ; i++ )
	{
		int a;
		scanf("%d", &a);
		for(int x=0;x<31;x++)
			if(a>>x & 1) B[x]++;
	}
	for( int i=0 ; i<m ; i++ )
		scanf("%d", &t[i]);
	sort( t,t+m );
	for( int i=0 ; i<m ; i++ )
	{
		for(int x=t[i];x<31;x++)
		{
			if(B[x])
			{
				ans++;
				B[x]--;
				for(int k=t[i];k<x;k++)
					B[k]++;
				break;
			}
		}
	}
	printf("%d\n",ans);
	return 0;
}
