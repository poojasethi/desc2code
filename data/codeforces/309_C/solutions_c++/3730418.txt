#include <cstdio>
#include <algorithm>
#define MAXN 1010000
using namespace std;
int n,m;
int K[30];
int ar[MAXN];
int res;
void f(int x)
{
	int j;
	for(j=0;j<30;j++)
		if((1<<j) & x)
			K[j]++;
}
int main()
{
	int i,j;
	int x;
	scanf(" %d %d",&n,&m);
	for(i=1;i<=n;i++)
	{
		scanf(" %d",&x);
		f(x);
	}
	for(i=1;i<=m;i++) scanf(" %d",&ar[i]);
	sort(ar+1,ar+m+1);
	for(i=1;i<=m;i++)
	{
		for(j=ar[i];j<30;j++)
		{
			if(K[j])
			{
				res++;
				K[j]--;
				f((1<<j) - (1<<ar[i]));
				break;
			}
		}
		if(j==31) break;
	}
	printf("%d\n",res);
	return 0;
}
