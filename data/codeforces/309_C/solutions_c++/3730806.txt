#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int a[35], b[1000005];

void ins(int k) {
  for (int sz = 0; k; sz++, k >>= 1)
    if (k & 1)
      a[sz]++;
}

int main() {
  int ac, k, n, m, sz;
  
  scanf("%d%d", &n, &m);
  memset(a, 0, sizeof a);
  for (int i = 0; i < n; i++) {
    scanf("%d", &k);
    ins(k);
  }
  for (int i = 0; i < m; i++)
    scanf("%d", &b[i]);
  sort(b, b + m);
  for (ac = 0; ac < m;) {
    for (sz = 0; sz < 33 && a[sz] == 0; sz++);
    if (sz == 33)
      break;
    a[sz]--;
    k = b[ac];
    if (k < sz) {
      ac++;
      ins((1<<sz)-(1<<k));
    }
    else if (k == sz)
      ac++;
  }
  printf("%d\n", ac);
  
  return 0;
}
