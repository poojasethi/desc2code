#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
int a[40],b[40];
int main()
{
    int n,m,i,j,x;
    while(scanf("%d%d",&n,&m)!=EOF)
    {
        int ans = 0;
        memset(a,0,sizeof(a));
        memset(b,0,sizeof(b));
        for(i = 1;i<=n;i++)
        {
            scanf("%d",&x);
            for(j = 0;j<=30;j++)a[j]+=(x&(1<<j))?1:0;
        }
        for(i = 1;i<=m;i++)
        {
            scanf("%d",&x);
            b[x]++;
        }
        while(true)
        {
            for(i = 0;i<=30&&b[i] == 0;i++);
            for(j = i;j<=30&&a[j] == 0;j++)a[j]++;
            if(i>=31||j>=31)break;
            a[j]--,b[i]--,ans++;
        }
        printf("%d\n",ans);
    }
    return 0;
}
