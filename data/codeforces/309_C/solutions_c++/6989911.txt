#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int n,m;
int cnt[32],b[1000010];

int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		int a;scanf("%d",&a);
		for(int j=0;j<32;j++) if(a&(1<<j)) cnt[j]++;
	}
	
	for(int i=1;i<=m;i++) scanf("%d",&b[i]);
	sort(b+1,b+m+1);int ans=0;
	for(int i=1;i<=m;i++){
		if(cnt[b[i]]) ans++,cnt[b[i]]--;
		else{
			int j;
			for(j=b[i];j<32;j++) if(cnt[j]) break;
			if(j>=32) break;
			//printf("%d %d\n",b[i],j);
			cnt[j]--;
			for(int p=j-1;p>=b[i];p--) cnt[p]++;
			ans++;
		}
	//	for(int j=31;j>=0;j--) printf("%d ",cnt[j]);puts("");
	}
	printf("%d\n",ans);
	return 0;
}
