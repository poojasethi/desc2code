#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <set>
using namespace std;
int a[100], b[100];
int main()
{
     int n, m, i, j, k;
     cin >> n >> m;
	 for (i = 1; i <= n; i ++) {
	   scanf("%d", &j); k = 0;
	   while (j) { a[k ++] += j % 2; j /= 2; }
	  }
	 for (i = 1; i <= m; i ++) {
	   scanf("%d", &j); b[j] ++;
	   }
	 int ans = 0;
	 for (i = 0; i < 32; i ++)
       while (b[i]) {
         for (j = i; j < 32; j ++)
		   if (a[j]) break;
		 if (a[j]) {b[i] --; a[j] --; while (j > i) a[--j] ++; ans ++; }
		  else break;
       }
	 cout << ans << endl;
    return 0;
}
