#include <iostream>
#include <stdio.h>
#define FOR(I,N) for (I=0;I<(N);++I)
#define sci(d) scanf("%d",&d)
using namespace std;

int b[32],a[32];

int main ()
{
	int i,m,n,ans=0;
	sci(n),sci(m);
	FOR(i,n)
	{
		int t,j;sci(t);
		FOR(j,31) if ((t>>j)&1) ++a[j];
	}
	FOR(i,m) sci(n),++b[n];
	FOR(i,31)
	{
		if (b[i]<=a[i])
		{
			a[i]-=b[i];
			ans+=b[i];
		}
		else while (b[i])
		{
			if (a[i]==0)
			{
				int j;
				for (j=i+1;j<31;++j)
					if (a[j])
					{
						--a[j];
						for (--j;j>i;--j) ++a[j];
						a[i]+=2;
						break;
					}
				if (a[i]==0) break;
				if (a[i]>b[i]) a[i]=b[i];
			}
			b[i]-=a[i];
			ans+=a[i];
			a[i]=0;
		}
	}
	cout<<ans<<endl;
	return 0;
}
