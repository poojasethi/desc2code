#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int N = 1000005;
int bit[32], b[N];
int main() {
	int n, m;
	scanf("%d%d", &n, &m);
	for(int i = 0, x; i < n; i ++) {
		scanf("%d", &x);
		for(int j = 0; j < 31; j ++) {
			if((x >> j) & 1) bit[j] ++;
		}
	}
	for(int i = 0; i < m; i ++) {
		scanf("%d", &b[i]);
	}
	sort(b, b + m);
	int ans = 0;
	for(int i = 0; i < m; i ++) {
		int x = b[i];
		if(x > 31) break;
		else if(bit[ x ]) {
			bit[ x ] --;
			ans ++;
		}
		else {
			int j;
			for(j = x; j < 32; j ++) {
				if( bit[j] ) {
					ans ++;
					bit[j] --;
					for(int k = x; k < j; k ++) bit[k] ++;
					break;
				}
			}
			if(j >= 32) break;
		}
	}
	printf("%d\n", ans);
	return 0;
}
