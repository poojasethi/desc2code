#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

const int N = 30+10;

int a[N],b[N];

int main(){
  int n,m,x;
  scanf("%d%d",&n,&m);
  for (int i=0;i<n;i++){
    scanf("%d",&x);
    for (int j=0;(1<<j)<=x;j++)
      if (x>>j&1)
	a[j]++;
  }
  for (int i=0;i<m;i++){
    scanf("%d",&x);
    b[x]++;
  }

  int ans = 0;
  for (int t=0;t<30;t++){
    while (b[t]){
      int z = min(a[t],b[t]);
      a[t]-=z;
      b[t]-=z;
      ans += z;
      if (b[t]){
	bool found = 0;
	for (int j=t+1;j<30;j++)
	  if (a[j]){
	    a[j]--;
	    a[j-1]+=2;
	    found = 1;
	    break;
	  }
	if (!found)
	  break;
      }
    }
  }
  printf("%d\n",ans);

  return 0;
}
