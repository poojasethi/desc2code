#include <cstdio>
#include <algorithm>
using namespace std;
int a[1010000];
int b[1010000];
int cnt[40];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++) scanf("%d",&a[i]);
	for(int i=0;i<m;i++) scanf("%d",&b[i]);
	for(int i=0;i<=30;i++){
		cnt[i]=0;
		for(int j=0;j<n;j++){
			if(a[j]&(1<<i)) cnt[i]++;
		}
	}
//	for(int i=0;i<10;i++) printf("%d\n",cnt[i]);
	sort(b,b+m);
	int ans=0;
	for(int i=0;i<m;i++){
		if(cnt[b[i]]){
			cnt[b[i]]--;
			ans++;
		}
		else {
			for(int j=b[i]+1;j<=30;j++){
				if(cnt[j]){
					ans++;
					cnt[j]--;
					for(int k=b[i];k<j;k++) cnt[k]++;
					break;
				}
			}
		}
	}
	printf("%d\n",ans);
}
