//In the name of God
#include <iostream>
#include <algorithm>
using namespace std;

typedef long long ll;
const int M = 1e6 + 2;

int b[M], m, n, fix[40];

int main() {
	cin.sync_with_stdio(false);
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		ll t;
		cin >> t;
		for (int j = 0; t; j++, t >>= 1)
			if (t & 1)
				fix[j]++;
	}
	for (int i = 0; i < m; i++)
		cin >> b[i];
	sort(b, b + m);
	int p;
	for (int i = 0; i < m; i++) {
		p = b[i];
		while (p < 40 && !fix[p])
			fix[p++] = 1;
		if (p == 40) {
			cout << i << '\n';
			return 0;
		}
		fix[p]--;
	}
	cout << m << '\n';
	return 0;
}
