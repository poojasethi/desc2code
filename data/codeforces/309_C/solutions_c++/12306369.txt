#include<bits/stdc++.h>
using namespace std;
int a[1000100];
int b[1000100];
int main()
{
    int n,m,cnt=0,i,x,j,k;
    //freopen("test.inp","r",stdin);
    scanf("%d %d",&n,&m);
    for(i=1;i<=n;i++) {
            scanf("%d",&x);
            for(j=30;j>=0;j--)
            {
                if(x&(1<<j)) a[j]++;
            }
    }
    for(i=1;i<=m;i++) {
            scanf("%d",&b[i]);

    }
    sort(b+1,b+1+m);
    for(i=1;i<=m;i++)
    {
         for(j=b[i];j<=30;j++) if(a[j]) {
                a[j]--;
                for(k=b[i];k<j;k++) a[k]++;
                cnt++; break;
        }
    }
    cout<<cnt;
}
