#include <cstdio>
#include <cstring>

#define MAX  1000000

int n,m;
int a[32];
int b[32];

int main(){
    scanf("%d",&n);
    scanf("%d",&m);
    int x;
    for(int i = 0; i < n; i++){
        scanf("%d",&x);
        for(int j=0;j<32;j++) 
            if(x&(1<<j))
                a[j]++; 
    }
    for(int i = 0; i < m; i++){
        scanf("%d",&x);
        b[x]++;
    }
    int count = 0;

    int i = 0, j = 0;
    while(true){
        while(j<32 && b[j] == 0)
            j++;
        while(i<32 && (a[i] == 0|| i < j))
            i++;
        if(i == 32 || j == 32)
            break;
        a[i]--;
        b[j]--;
        while(i > j)
            a[--i]++;
        count++;
    
    }
    printf("%d",count);
    return 0;
}
