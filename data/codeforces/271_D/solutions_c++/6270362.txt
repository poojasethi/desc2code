#include <iostream>
#include <set>
#include <cstring>
#include <algorithm>
using namespace std;

int k,bad[26],sum[1501],n,m=0;
set<long long>arr;
int main() {
	string s,x;
	cin>>s>>x>>k;
	n=s.size();
	for(int i=0;i<26;i++)
		bad[i]=(x[i]=='0');
	for(int i=0;i<n;i++)
		sum[i+1]=sum[i]+bad[s[i]-'a'];
	for(int i=0;i<n;i++){
		long long d=1;
		for(int j=i;j<n;j++){
			if(sum[j+1]-sum[i]>k)
				break;
			d=d*131+(s[j]-'a');
			arr.insert(d);
		}
	}
	cout<<arr.size();
	return 0;
}
