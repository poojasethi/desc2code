#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll p=1000000007;
ll cum[1600],al[30];
int main()
{
	set<ll>see;
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll k,i,j;
	string s,t;
	cin>>s>>t>>k;
	for(i=0;i<26;i++)
	al[i]='1'-t[i];
	for(i=1;i<=s.size();i++)
		cum[i]=cum[i-1]+al[s[i-1]-'a'];
	for(i=0;i<s.size();i++)
	{
		ll d=1;
		for(j=i;j<s.size();j++)
		{
			if(cum[j+1]-cum[i]>k)
				break;
			d=(d*131+(s[j]-'a'));
			see.insert(d);
		}
	}
	cout<<see.size()<<endl;
}
