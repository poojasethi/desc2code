#include <bits/stdc++.h>
#define rssi(a,b,c) scanf("%s\n%s\n%d",a,b,&c)
#define FOR(i,l,r) for(int i=l;i<r;i++)
using namespace std;

int K, N, V=1, trie[1200000][26];
char str[1600], A[30];

void insert(int ptr, int i, int k) {
	if(i==N) return;
	k+=A[str[i]-'a']=='0';
	if(k>K) return;
	if(trie[ptr][str[i]-'a']==-1) trie[ptr][str[i]-'a']=V++;
	insert(trie[ptr][str[i]-'a'], i+1, k);
}

int main() {
	memset(trie,-1,sizeof(trie));
	rssi(str,A,K); N=strlen(str);
	FOR(i,0,N) insert(0,i,0);
	printf("%d\n",V-1);
}
