#include <bits/stdc++.h>

using namespace std;
#define HODOR unsigned long long
set<HODOR> S;
string ip, H;
int k;

int main()
{
	cin >> ip >> H >> k;
	for (int i = 0; i < ip.length(); ++i) {
		for (HODOR bad = k, hash = 0, j = i; j < ip.length(); ++j) {
			if( H[ip[j] -'a'] == '0' && !(bad--) )
				break;
			hash = (hash*131)^ip[j];
			S.insert(hash);
		}
	}
	cout << S.size() << "\n";
	return 0;
}
