s=raw_input()
a=map(int, raw_input())
k=input()
S=sorted(s[i:] for i in xrange(len(s)))
p=''
r=0
X=lambda x:not a[ord(x)-ord('a')]
for e in S:

    t=0
    h=0
    l=-1
    for i in xrange(min(len(e),len(p))):
        if e[i] != p[i]: break
        h=i
        t+=X(e[i])
        l=i
    for j in xrange(h,len(e)):
        if j == l: continue 
        t+=X(e[j])
        if t > k:break
        r+=1
    p = e

print r
