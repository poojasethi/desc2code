s = raw_input()
d = sorted(list(s[i : ]) for i in range(len(s)))
g = list(raw_input())
k = input()
r = 0
for i, u in enumerate(d):
	j = 0
	while i and j < len(d[i - 1]) and d[i - 1][j] == u[j]:
		j += 1
	c = 0
	for o, x in enumerate(u):
		c += g[ord(x) - ord('a')] == '0'
		if c > k:
			break
		r += o >= j
print r
