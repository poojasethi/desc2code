s, t = raw_input(), raw_input()
k = input()
l = len(s)
sub = [s[i:] for i in range(l)]
sub.sort()
res = 0
pre = ''
for ss in sub:
	l1 = len(ss)
	l2 = len(pre)
	ct = 0
	i = 0
	while i < l1 and i < l2:
		if pre[i] != ss[i]:
			break
		if t[ord(ss[i]) - ord('a')] == '0':
			ct += 1
		i += 1
	while i < l1:
		if t[ord(ss[i]) - ord('a')] == '0':
			ct += 1
		if ct > k:
			break
		res += 1
		i += 1
	pre = ss
print res
