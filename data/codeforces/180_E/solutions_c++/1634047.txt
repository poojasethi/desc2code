#include <iostream>
#include <fstream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <vector>
using namespace std;
const int N=100003;
int main()
{
	int n,m,k;
	vector<int> col[N];
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0,a;i<n;++i)
	{
		scanf("%d",&a);
		col[a].push_back(i);
	}
	int s,res=0;
	for(int i=1;i<=m;++i)
	{
		s=0;
	//	printf("%d\n",col[i].size());
		for(int j=0;j<col[i].size();++j)
		{
			while(col[i][j]-col[i][s]>j-s+k)s++;
			res=res<j-s+1?j-s+1:res;
		}
	}
	printf("%d\n",res);
	return 0;
}
