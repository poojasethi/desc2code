#include<iostream>

using namespace std;

const int N = 200010;

int n, m, K, a[N], t[N], z[N];
int b[N], w;

int main() {
	cin >> n >> m >> K;
	for (int i = 1; i <=m; i++)
		z[i] = -1;
	for (int i = 0; i < n; i++) {
		cin >> a[i];
		t[i] = z[a[i]];
		z[a[i]] = i;
	}
	int ans = 1;
	for (int i = 1; i <=m; i++) {
		w = 0;
		for (int j = z[i]; j != -1; j = t[j])
			b[++w] = j;
		for (int l = 1, r = 1; r <= w; r++) {
			while( b[l] - b[r] - (r - l ) > K) l++;
			if (r - l + 1 > ans)
				ans = r - l + 1;
		}
	}
	cout << ans << endl;
}
