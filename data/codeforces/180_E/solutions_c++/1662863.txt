#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;

vector<int> a[100005];

int main()
{
	int n,m,k,t;
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&t);
		a[t].push_back(i);
	}
	int ans=0,p;
	for(int i=1;i<=m;i++)
	{
		int j=0;
		for(int p=0;p<a[i].size();p++)
		{
			while(a[i][p]-a[i][j]-p+j>k)
				j++;
			ans=max(ans,p-j+1);
		}
	}
	printf("%d\n",ans);
}
