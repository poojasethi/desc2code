#include<bits/stdc++.h>
using namespace std;

int k;
vector<int> v[100005];

inline int cal(vector<int>& v)
{
	int i,j,c,s,ss,n=v.size();
	for(c=j=0,s=ss=i=1;i<n;i++)
	{
		c+=v[i]-v[i-1]-1,ss++;
		while(c>k)
			j++,c-=v[j]-v[j-1]-1,ss--;
		s=max(s,ss);
	}
	return s;
}

int main()
{
	int n,m,i,a,s;
	for(scanf("%d%d%d",&n,&m,&k),i=0;i<n;i++)
	{
		scanf("%d",&a);
		v[a].push_back(i);
	}
	for(s=0,i=1;i<=m;i++)
		s=max(s,cal(v[i]));
	printf("%d\n",s);
	return 0;
}
