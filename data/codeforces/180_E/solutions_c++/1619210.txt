#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

vector<int> a[100005];

int main(void) {
	int n, m, k;
	scanf("%d%d%d", &n, &m, &k);
	int ans=0;
	for(int i=0;i<n;i++) {
		int x;
		scanf("%d", &x);
		a[x].push_back(i);
		int L=0,R=(int)a[x].size()-1,A=a[x].size()-1;
		while(L<=R) {
			int M=(L+R)/2;
			if(a[x].back() - a[x][M]+1 - (int)a[x].size() + M <= k) {
				A = M; R=M-1;
			} else L=M+1;
		}
		ans = max(ans, (int)a[x].size()-A);
	}
	printf("%d\n", ans);
	return 0;
}
