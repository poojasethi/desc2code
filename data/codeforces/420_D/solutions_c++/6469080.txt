using namespace std;

const int maxn=(1<<21);
int md=(1<<20);
int n, m;
vector<int>v;
int r[maxn-1000];
int a[maxn-1000], c[maxn-1000];
int us[maxn], act[maxn-1000];
bool was[maxn];
void modify(int x, int y)
{
    for (; x<maxn; x=(x|(x+1)))
        r[x]+=y;
}
int sum(int x)
{
    int res=0;
    for (; x>=0; x=((x&(x+1))-1))
        res+=r[x];
    return res;
}
int bin(int st, int fin, int nc)
{
    if (st==fin) return st;
    int med=(st+fin)/2;
    if ( sum(med)<nc ) 
        return bin(med+1, fin, nc);
    else
        return bin(st, med, nc);
}
int main ()
{
    scanf("%d%d", &n, &m);
    for (int i=md; i<md+n; i++)
        modify(i, 1), a[i]=i-md+1;
    for (int i=1; i<=m; i++)
    {
        int aa, bb;
        scanf("%d%d", &aa, &bb);
        int f=bin(1, maxn-1000, bb);
        if ( (c[f]!=0 && c[f]!=aa)||(act[aa]!=0 && act[aa]!=f) )
        {
            cout<<-1<<"\n";
            return 0;
        }
        modify(f, -1);
        md--;
        modify(md, 1);
        c[md]=us[a[f]]=aa;
        a[md]=a[f];
        act[aa]=md;
        was[aa]=1;      
    }
    
    for (int i=1; i<=n; i++) 
        if ( !was[i] ) 
            v.push_back(i);
    int uf=0;
    for (int i=1; i<=n; i++)
        if ( us[i] )
            printf("%d ", us[i]);
        else
            printf("%d ", v[uf]), uf++;
    printf("\n");
    return 0;    
}
