using namespace std;
typedef long long LL;
int bit[2000005];
int n,m,all;

inline void add(int x,int v)
{
    for(;x<=all;x+=x&-x) bit[x]+=v;
}

inline int read(int x)
{
    int res=0;
    for(;x;x^=x&-x) res+=bit[x];
    return res;
}

inline int get_kth(int k)
{
    int l=1,r=all,mid,cnt,res;
    while(l<=r)
    {
        mid=(l+r)>>1;
        cnt=read(mid);
        if(cnt>=k)
        {
            res=mid;
            r=mid-1;
        }
        else l=mid+1;
    }
    return res;
}

int pos[1000005],ans[1000005],num[2000005];
int main()
{
    scanf("%d%d",&n,&m);
    all=n+m;
    for(int i=1;i<=n;++i)
    {
        add(m+i,1);
        num[m+i]=i;
    }
    int top=m;
    while(m--)
    {
        int x,y;
        scanf("%d%d",&x,&y);
        int t=get_kth(y);
        if(pos[x]&&pos[x]!=num[t])
        {
            puts("-1");
            return 0;
        }
        if(ans[num[t]]&&ans[num[t]]!=x)
        {
            puts("-1");
            return 0;
        }
        pos[x]=num[t];
        ans[num[t]]=x;
        add(t,-1);
        add(top,1);
        num[top--]=num[t];
    }
    for(int j=1,i=1;i<=n;++i)
    {
        if(ans[i]) printf("%d ",ans[i]);
        else
        {
            for(;j<=n&&pos[j];++j);
            printf("%d ",j++);
        }
    }
    return 0;
}
