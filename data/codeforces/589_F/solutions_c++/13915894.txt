#include <bits/stdc++.h>

using namespace std;

int n,ans;
int x[105],y[105];

bool check(int o)
{
    int m,i,id,count[105]={0},t=-1;
    while (++t<=10000)
    {
        m=100000;
        id=0;
        for (i=1;i<=n;++i)
          if (x[i]<=t && y[i]>t && y[i]<m && count[i]<o)
              m=y[i],id=i;
        if (id!=0) ++count[id];
    }
    for (i=1;i<=n;++i)
      if (count[i]<o) return false;
    return true;
}

int main()
{
    int i,l,r;
    scanf("%d",&n);
    for (i=1;i<=n;++i)
        scanf("%d%d",x+i,y+i);
    l=0,r=10000;
    while (l<=r)
    {
        int mid=(l+r) >>1;
        if (check(mid)) l=mid+1,ans=mid;
          else r=mid-1;
    }
    cout <<ans*n <<endl;
    return 0;
}
