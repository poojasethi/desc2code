#include<iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>
using namespace std;
struct haha
{
    int a,b;
}t[110];
bool cmp(haha x,haha y)
{
    if (x.b==y.b) return x.a<y.a;
    else return x.b<y.b;
}
int n;
bool judge(int k)
{
    bool f[10010];
    int cot,i,j;
    memset(f,false,sizeof(f));
    for(i=1;i<=n;i++)
    {
          cot=0;
          for(j=t[i].a+1;j<=t[i].b;j++)
          {
              if (f[j]) continue;
              cot++;f[j]=true;
              if (cot==k) break;
          }
          if (cot<k) return false;
    }
    return true;
}
int main()
{
    int i,l,r;
    cin>>n;
    for(i=1;i<=n;i++)
        scanf("%d%d",&t[i].a,&t[i].b);
    sort(t+1,t+1+n,cmp);
    l=0;r=10001;
    while(l+1<r)
    {
        int mid=(l+r)/2;
        if (judge(mid)) l=mid;
        else r=mid;
    }
    cout<<l*n<<endl;
    return 0;
}
