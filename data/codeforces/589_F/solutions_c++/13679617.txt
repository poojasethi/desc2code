#include<bits/stdc++.h>
using namespace std;
struct t{int x,y;};
t a[105];
int n;
int main(){
    scanf("%d",&n);
    for (int i=1; i<=n; i++) scanf("%d%d",&a[i].x,&a[i].y);
    int l=0,r=10000,m,mn,z,cnt,d[105];
    while (l<r){
        m=(l+r+1)/2;
        memset(d,0,sizeof(d));
        cnt=0;
        for (int i=0; i<=10000; i++){
            mn=10010;
            for (int j=1; j<=n; j++) if ((d[j]<m) && a[j].x<=i && a[j].y>i && a[j].y<mn){
                mn=a[j].y;
                z=j;
            }
            if (mn<10010){
                d[z]++;
                cnt++;
            }
        }
        if (cnt==m*n) l=m;
        else r=m-1;
    }
    printf("%d",l*n);
}
