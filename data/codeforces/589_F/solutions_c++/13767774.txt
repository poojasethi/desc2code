#include <bits/stdc++.h>

#define ll long long
#define fi freopen("", "r", stdin)
#define fo freopen("", "w", stdout)

using namespace std;

int n;
vector<pair<int, int> > a;
vector<int> v;

int main()
{
  fi;fo;
  scanf("%d", &n);
  int x, y;
  for(int i = 1; i <= n; i++){
    scanf("%d%d", &x, &y);
    a.push_back(make_pair(x, y));
    v.push_back(x);
    v.push_back(y);
  }
  sort(v.begin(), v.end());
  int ans = 1e9;
  for(int i = 0; i < v.size(); i++)
    for(int j = i + 1; j < v.size(); j++){
      int cnt = 0;
      for(int k = 0; k < a.size(); k++)
        if(a[k].first >= v[i] && a[k].second <= v[j]) cnt++;
      if(cnt) ans = min(ans, (v[j] - v[i]) / cnt);
    }
  cout << ans * n;
  return 0;
}
