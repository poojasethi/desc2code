#include <bits/stdc++.h>

using namespace std;

struct point{
	int a, b;
}d[108];

int vis[10008], n;

bool cmp(point p1, point p2){
	return p1.b < p2.b;
}

int check(int mid){
	memset(vis, 0, sizeof(vis));
	for(int i = 0; i < n; i++){
		int cou = 0;
		for(int j = d[i].a; j < d[i].b; j++){
			if(vis[j]) continue;
			vis[j] = 1;
			cou++;
			if(cou >= mid) break;
		}
		if(cou < mid) return 0;
	}
	return 1;
}

int main(){

	while(scanf("%d", &n) > 0){
		for(int i = 0; i < n; i++){
			scanf("%d%d", &d[i].a, &d[i].b);
		}
		sort(d, d + n, cmp);
		int res = 0, l = 0, r = 10000;
		while(l <= r){
			int mid = (l + r) >> 1;
			if(check(mid)){
				res = mid;
				l = mid + 1;
			}
			else{
				r = mid - 1;
			}
		}
		printf("%d\n", res * n);
	}

	return 0;
}
