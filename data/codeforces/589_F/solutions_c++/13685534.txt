#include<bits/stdc++.h>
using namespace std;
int a[110],b[110];
int cnt[110];
int n;
bool ok(int mid)
{
    memset(cnt,0,sizeof cnt);
    for(int i=0;i<=10000;i++)
    {
        int e=-1;
        for(int j=0;j<n;j++)
        {
            if(a[j]<=i && i<b[j] && cnt[j]<mid)
            {
                if(e==-1 || b[j]<b[e])
                    e=j;
            }
        }
        if(e+1)
            cnt[e]++;
    }
    for(int i=0;i<n;i++)
        if(cnt[i]<mid)
            return 0;
    return 1;
}
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
        scanf("%d %d",&a[i],&b[i]);
    int l=1,r=10000;
    while(l<r)
    {
        int mid=(l+r)>>1;
        if(ok(mid))
            l=mid+1;
        else
            r=mid;
    }
    printf("%d\n",n*(r-1));
    return 0;
}
