#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <iostream>

using namespace std;

int N;
int G[110][2]={{0}};
int ans[110]={0};
int hash[110]={0};

bool check(int mid)
{
	memset(ans,0,sizeof(ans));
	for(int i=0;i<=10000;i++)
	{
		int cnt=0;
		for(int j=1;j<=N;j++)
		{
			if(G[j][0]<=i && G[j][1]>i && ans[j]<mid)
			{
				if(cnt==0 || G[cnt][1]>G[j][1])
					cnt=j;
			}
		}
		ans[cnt]++;
	}
	for(int i=1;i<=N;i++)
		if(ans[i]<mid) return false;
	return true;
}

int main()
{
	cin>>N;
	for(int i=1;i<=N;i++)
		scanf("%d%d",&G[i][0],&G[i][1]);
	int l=0,r=10000;
	int Out=0;
	for(;l<=r;)
	{
		int mid=(l+r)>>1;
		if(check(mid)==true)
		{
			Out=mid;
			l=mid+1;
		}
		else r=mid-1;
	}
	cout<<Out*N<<endl;
	return 0;
}
