#include<cstdio>
#include<cmath>
#include<algorithm>
#include<cstring>
#include<iostream>
using namespace std;
struct node
{
	int s,e;
}p[105];
int n,vis[10005];
bool cmp(node a,node b)
{
	return a.e<b.e;
}
int f(int k)
{
	int i,flag,j;
	memset(vis,0,sizeof(vis));
	if(k==0)
		return 1;
	for(i=1;i<=n;i++)
	{
		int cnt=0;
		flag=0;
		for(j=p[i].s;j<p[i].e;j++)
		{
			if(vis[j]==0)
			{
				vis[j]=1;
				cnt++;
			}
			if(cnt==k)
			{
				flag=1;
				break;
			}
		}
		if(flag==0)
			return 0;
	}
	return 1;
}
int main()
{
	int i,l,r,mid;
	scanf("%d",&n);
	l=0;r=0;
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&p[i].s,&p[i].e);
		if(p[i].e-p[i].s>r)
			r=p[i].e-p[i].s;
	}
	sort(p+1,p+1+n,cmp);
	while(l<=r)
	{
		mid=(l+r)/2;
		if(f(mid))
			l=mid+1;
		else
			r=mid-1;
	}
	printf("%d\n",r*n);
	return 0;
}
