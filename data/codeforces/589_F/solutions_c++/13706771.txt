#include<bits/stdc++.h>
using namespace std;
int a[101],b[101],n;
bool check(int x){
    //int p=1;
    int count[109]={0};
    for(int t=0;t<=10000;t++){
        int p=0;
        for(int i=1;i<=n;i++){
            if(count[i]<x&&b[i]==t) return false;
            if(a[i]<=t&&t<b[i]&&count[i]<x&&(p==0||b[i]<b[p])) p=i;
        }
        ++count[p];
    }
    return true;
}
int main(){
    //int n;
    cin>>n;
    for(int i=1;i<=n;i++) cin>>a[i]>>b[i];
    int l=0,r=10000;
    while(l<r){
        int mid=(l+r)/2;
        if(check(mid)) l=mid;
        else r=mid-1;
        if(l==r-1){
            if(check(r))l=r;
            else r=l;
        }
        //cout<<l<<" "<<r<<" "<<mid<<endl;
    }
    cout<<l*n<<endl;
}
