#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <ctime>
#include <cmath>
#include <iostream>
#include <algorithm>

using namespace std;

int n, a[101][2];
bool b[10001];

bool check(int now) {
	memset(b, false, sizeof(b));
	for (int i = 1; i <= n; i++) {
		int cnt = 0;
		for (int j = a[i][0], k = 1; j < a[i][1] && k <= now; j++)
			if (!b[j]) b[j] = true, ++k, ++cnt;
		if (cnt != now) return false;
	}
	return true;
}
 
int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) scanf("%d%d", &a[i][0], &a[i][1]);
	for (int i = 1; i <= n; i++)
		for (int j = i + 1; j <= n; j++)
			if (a[i][1] > a[j][1]) swap(a[i][0], a[j][0]), swap(a[i][1], a[j][1]);
		int Left = 0, Right = 10001, Mid = (Left + Right) >> 1;
		for (; Left + 1 < Right; Mid = (Left + Right) >> 1) 
			if (check(Mid)) Left = Mid;
			else Right = Mid;
		printf("%d\n", Left * n);
}
