#include<bits/stdc++.h>
using namespace std;
const int maxn=100500;
struct node{
	int l,r;
	bool operator<( const node& ths )const{
		return r<ths.r||((r==ths.r)&&(l<ths.l));
	}
}a[maxn];
int n;
bool vis[maxn];
bool is(int x){
	memset(vis,0,sizeof(vis));
	for(int i=1;i<=n;i++){
		int cnt=0;
		for(int j=a[i].l+1;j<=a[i].r;j++){
			if(!vis[j]){
			    vis[j]=1;
				if(++cnt==x){
					break;
				}
			}
		}
		if(cnt<x)return false;
	}
	return true;
}
int main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i].l>>a[i].r;
	}
	sort(a+1,a+1+n);
	int l=1,r=maxn;
	while(l<r-1){
		int mid=(l+r)/2;
		if(is(mid)){
			l=mid;
		}
		else r=mid;
	}
	if(is(l))cout<<l*n<<endl;
	else cout<<0<<endl;
}
