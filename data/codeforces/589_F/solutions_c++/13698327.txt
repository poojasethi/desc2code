#include<algorithm>
#include<iostream>
#include<utility>
using namespace std;

pair<int,int> a[105];
int n,ans,vis[10005],mi=1e9;

int main(){
	cin >> n;
	for(int i=0; i<n; i++)cin >> a[i].first >> a[i].second,mi=min(mi,a[i].second-a[i].first);
	int l=0, r=mi;
	while(r>=l){
		int mid = (l+r)/2;
	//	cout << "###########\n";
	//	cout << l << " " << r << " " << mid << endl;
		for(int i=0; i<n; i++)vis[i]=mid;
		for(int k=0; k<10000; k++){
			int cur=-1;
			for(int i=0; i<n; i++){
				if(vis[i]==0||a[i].first>k||a[i].second<k+1)continue; //no need
				if(cur==-1||a[cur].second>a[i].second)cur=i;
			}
			if(cur!=-1)vis[cur]--;
		}
	//	for(int i=0; i<n; i++)cout << vis[i] << " "; cout << endl;
		bool fail=0;
		for(int i=0; i<n; i++)if(vis[i]!=0)fail=1;
		if(fail)r=mid-1; else ans=mid,l=mid+1;
	}
	cout << n*ans << endl;
	return 0;
}
