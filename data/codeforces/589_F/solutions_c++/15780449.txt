#include <bits/stdc++.h>

int n;
int s[100], t[100];

int check(int consume) {
  static int cnt[100];
  std::fill(cnt, cnt + n, 0);
  for (int i = 0; i < 10000; i++) {
    int k = -1;
    for (int j = 0; j < n; j++)
      if (s[j] <= i && i < t[j] && cnt[j] < consume)
        if (k == -1 || t[j] < t[k]) 
          k = j;
    cnt[k]++;
  }
  for (int i = 0; i < n; i++) if (cnt[i] < consume) return 0;
  return 1;
}

int main() {
  scanf("%d", &n);
  for (int i = 0; i < n; i++) 
    scanf("%d%d", s + i, t + i);

  int low = 0, high = 10000 / n + 1;
  while (low + 1 < high) {
    int mid = (low + high) >> 1;
    if (check(mid)) low = mid;
    else high = mid;
  }
  printf("%d\n", low * n);

  return 0;
}
