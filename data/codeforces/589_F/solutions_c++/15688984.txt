#include <cstdio>
#include <algorithm>
#include <cstring> 
using namespace std;
const int N=105;
struct node{int l,r;}a[N];
int n,l=0,r=0,ans=0;
bool v[100*N];
bool cmp(node a,node b){return a.r<b.r;}
bool check(int x){
	  memset(v,false,sizeof(v));
	  for(int i=1;i<=n;i++){
		    if(a[i].r-a[i].l<x)return 0;
		    int sum=0;
		    for(int j=a[i].l;j<a[i].r;j++)if(!v[j]){sum++;v[j]=1;if(sum==x)break;}
		    if(sum<x)return 0;
	  }return 1;
}
int main(){
	  scanf("%d",&n);
	  for(int i=1;i<=n;i++){
		    scanf("%d%d",&a[i].l,&a[i].r);
		    r=max(r,a[i].r-a[i].l);
	  }sort(a+1,a+1+n,cmp);
	  while(l<=r){
		    int mid=(l+r)>>1;
		    if(check(mid)){l=mid+1;ans=max(mid,ans);}
		    else r=mid-1;
	  }printf("%d\n",ans*n);
	  return 0;
}
