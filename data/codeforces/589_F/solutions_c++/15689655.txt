#include <bits/stdc++.h>
using namespace std;

struct dish
{
	int a,b;
}a[105];
int vis[10005],n;
bool com(dish i,dish j)
{
	return i.b < j.b;
}
bool check(int mid)
{
	memset(vis,0,sizeof(vis));
	for(int i = 1; i <= n; i++)
	{
		int cont = 0;
		for(int j = a[i].a; j < a[i].b; j++)
		{
			if(vis[j]) continue;
			vis[j] = 1;
			cont++;
			if(cont >= mid) break;
		}
		if(cont < mid) return false;
	}
	return true;
}
int main()
{
	while(cin>>n)
	{
		for(int i = 1; i <= n; i++)
		{
			scanf("%d%d",&a[i].a,&a[i].b);
		}
		sort(a+1,a+1+n,com);
		int l = 0, r = 10000,mid,ans = 0;
		while(l <= r)
		{
			mid = (l+r)>>1;
			if(check(mid))
			{
				ans = mid;
				l = mid + 1;
			}
			else
			{
				r = mid - 1;
			}
		}
		printf("%d",ans*n);
	}
}
