#include <cstdio>
#include <algorithm>
#define snd first
#define fst second
#define MAXN 110
#define MAX 10010
using namespace std;

int n;
pair<int,int> v[MAXN+10];
bool mark[MAX+10];

bool pode( int x )
{
	int sum = 0;
	for ( int i = 0; i <= 10000; i++ )
		mark[i] = true;
	for ( int i = 1; i <= n; i++ ) {
		sum = 0;
		for ( int j = v[i].fst; j < v[i].snd; j++ ) {
			if ( mark[j] ) {
				sum++;
				mark[j] = false;
				if ( sum == x ) break;
			}
		}
		if ( sum < x ) return false;
	}
	return true;
}

int bs(int ini, int fim)
{
	while ( ini < fim ) {
		int mid = (ini+fim+1)/2;
		if ( pode(mid) ) 
			ini = mid;
		else
			fim = mid - 1;
	}
	return ini;
}

int main()
{
	scanf("%d", &n);
	for ( int i = 1; i <= n; i++ ) {
		scanf("%d %d", &v[i].fst, &v[i].snd);
	}
	sort(v+1, v+n+1);
	int resp = bs(0,10000);
	printf("%d\n", n*resp);
	return 0;
}
