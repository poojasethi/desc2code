#include <bits/stdc++.h>

using namespace std;

int n, ck, ate[10005];
pair <int, int> a[105];

bool eatin(int x)
{
  for (int i = 1; i <= n; ++i)
  {
    int add = 0;
    for (int j = a[i].first; j < a[i].second; ++j)
    {
      if (add == x) break;
      if (ate[j] == ck) continue;
      ate[j] = ck;
      ++add;
    }
    if (add < x) return 0;
  }
  return 1;
}

bool cmp(pair <int, int> u, pair <int, int> v)
{
  return u.second < v.second || (u.second == v.second && u.first < v.first);
}

int main()
{
  #ifndef ONLINE_JUDGE
  freopen("input", "r", stdin);
  //freopen("output", "w", stdout);
  #endif
  ios_base::sync_with_stdio(0);
  cin >> n;
  for (int i = 1; i <= n; ++i) cin >> a[i].first >> a[i].second;
  sort (a + 1, a + n + 1, cmp);
  int l = 0, r = 10005;
  while (r - l > 1)
  {
    ++ck;
    int m = (r + l) /2;
    if (eatin(m)) l = m;
    else r = m;
  }
  cout << l*n;
}
