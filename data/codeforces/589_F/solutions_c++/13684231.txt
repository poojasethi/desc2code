#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

pair<int, int> A[200];
int p, r, mid, N;
int T[20000];

int check(int mid){
	memset(T, 0, sizeof(T));
	int counter;
	for (int i = 0; i < N; i++){
		if (A[i].first + A[i].second < mid) return 0;
		counter = 0;
		for (int j = -A[i].second; j < A[i].first; j++)
		   if (T[j] == 0 && counter < mid) T[j] = 1, counter++;
		if (counter < mid) return 0;
	}
	return 1;
}

int main(void){
	scanf("%d", &N);
	for (int i = 0; i < N; i++){
		scanf("%d%d", &A[i].second, &A[i].first);
		 A[i].second *= -1;
	}
	sort(A, A + N);
	p = 0;
	r = 10000+1;
	while (r - p > 1){
		mid = (p + r) / 2;
		if (check(mid)) p = mid;
		else r = mid;
	}
	if (check(p + 1)) p++;
	printf("%d", p * N);
	return 0;
}
