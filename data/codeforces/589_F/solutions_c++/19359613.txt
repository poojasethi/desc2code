#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long ll;
typedef struct{
	int b;
	int e;
}food;

const int maxn = 105;
const ll INF = 10000000;


food dish[maxn];
int eat[10005],cnt;
int n;

int cmp(food a,food b)
{
	return a.e < b.e;
}

bool judge(int t)
{
	cnt++;
	for(int i = 0;i < n;i++){
		int cur = 0;
		for(int j = dish[i].b;j < dish[i].e;j++){
			if(cur == t)
				break;
			if(eat[j] != cnt){
				eat[j] = cnt;
				cur++;
			}	
		}
		if(cur < t)
			return false;
	}
	return true;
}

int main()
{
	int i,j,k;
	int lb,ub,mid;
	while(scanf("%d",&n) != EOF){
		cnt = 0;
		for(i = 0;i < n;i++){
			scanf("%d %d",&dish[i].b,&dish[i].e);
		}
		sort(dish,dish + n,cmp);
		lb = 0,ub = 10002;
		while(1 < ub - lb){
			mid = (lb + ub) / 2;
			//printf("mid = %d\n",mid);
			if(judge(mid))
				lb = mid;
			else
				ub = mid;
		}
		printf("%d\n",lb * n);
	}
	return 0;
}
