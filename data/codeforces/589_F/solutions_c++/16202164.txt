#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
struct seg
{
	int a,b;
	bool operator <(const seg&segm)const{
		if(b!=segm.b)
		return b<segm.b;
		return a<segm.a;
	}
}seg[110];
int cunt[10010];
bool check(int u,int n)
{
	int l,r,ct=0;
	memset(cunt,0,sizeof(cunt));
	for(int i=1;i<=n;i++)
	{
		l=seg[i].a+1;
		r=seg[i].b;
		ct=0;
		for(int j=l;j<=r;j++)
		{
			if(!cunt[j])
			{
				ct++;cunt[j]=1;
			}
			if(ct==u) break;
		}
		if(ct<u) return false;
	}
	return true;
}
int main()
{
	int n,i,j,l,r,maxb=1e4,mid,ans=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&l,&r);
		seg[i].a=l;
		seg[i].b=r;
		maxb=min(maxb,r-l);
	}
	sort(seg+1,seg+n+1);
	l=1;
	r=maxb;
	while(l<=r)
	{
		mid=(l+r)>>1;
		if(check(mid,n))
		{
			ans=mid;
			l=mid+1;
		}
		else r=mid-1;
	}
	printf("%d",ans*n);
	return 0;
}
