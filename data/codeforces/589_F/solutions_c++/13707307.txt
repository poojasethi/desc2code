#include<bits/stdc++.h>
using namespace std;

pair<int, int>pii[110];
int arr[110];
vector<int>vv;

int main()
{
    int n, a, b, mini=100000000;

    scanf("%d", &n);
    for(int i=0; i<n; i++)
    {
        scanf("%d %d", &a, &b);
        pii[i].first=a+1;
        pii[i].second=b+1;
        vv.push_back(a+1);
        vv.push_back(b+1);
    }

    sort(vv.begin(), vv.end());

    for(int i=0; i<vv.size(); i++)
    {
        for(int j=i+1; j<vv.size(); j++)
        {
            int cnt=0;
            for(int k=0; k<n; k++)
            {
                if(pii[k].first>=vv[i] && pii[k].second<=vv[j])
                {
                    cnt++;
                }
            }
            if(cnt>0)
                mini=min(mini, (vv[j]-vv[i])/cnt);
        }
    }

    printf("%d\n", mini*n);

    return 0;
}
