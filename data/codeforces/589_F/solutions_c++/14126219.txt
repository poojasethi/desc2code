#include<cstdio>
#include<vector>
#include<set>
#include<string>
#include<cstring>
#include<queue>
#include<algorithm>

using namespace std;

vector<int> a[11111];
priority_queue<int> PQ;

int main(){
	int l,r,mid;
	int i,j,k,n;
	scanf("%d", &n);
	for(i=0;i<n;i++){
		scanf("%d%d", &l, &r);
		a[l].push_back(r);
	}
	l=0;
	r=11111;
	while(l<r){
		mid = (l+r+1)/2;
		while(!PQ.empty()) PQ.pop();
		for(i=0;i<11111;i++){
			for(j=0;j<a[i].size();j++) for(k=0;k<mid;k++) PQ.push(-a[i][j]);
			if(PQ.empty())continue;
			if(-PQ.top()<=i)break;	
			PQ.pop();
		}
		if(i<11111)r=mid-1;
		else l=mid;
	}
	printf("%d",n*l);
}
