#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
struct pp{int x,y;} p[1000005];
bool cmpy(const pp &i,const pp &j){return i.y<j.y;}
int n,hc[10005];
int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;i++){scanf("%d%d",&p[i].x,&p[i].y);p[i].y--;}
	sort(p+1,p+n+1,cmpy);
	int l=0,r=10001;
	while (l+1<r){
		memset(hc,0,sizeof(hc));
		int mid=(l+r)/2,can=1;
		for (int i=1;i<=n;i++){
			int need=mid;
			for (int j=p[i].x;j<=p[i].y&&need;j++)
				if (!hc[j]){hc[j]=1;need--;}
			if (need){can=0;break;}
		}
		if (can) l=mid;
		else r=mid;
	}
	printf("%d",l*n);
}
