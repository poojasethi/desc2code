#include<bits/stdc++.h>
using namespace std;

int n;
pair<int,int> dish[110];

bool isvalid(int x){
    bool f[10011] = {0};
    for(int i = 1;i <= n;++i){
        int cnt = 0;
        for(int j = dish[i].second + 1;j <= dish[i].first;++j){
            if(f[j]) continue;
            f[j] = 1;
            cnt++;
            if(cnt == x) break;
        }
        if(cnt < x) return false;
    }
    return true;
}

int main(){
    scanf("%d",&n);
    for(int i = 1;i <= n;++i)
        scanf("%d%d",&dish[i].second,&dish[i].first);
    sort(dish + 1,dish + 1 + n);
    int l = 0,r = 10001;  //×ó±ÕÓÒ¿ª
    while(l + 1 < r){
        int mid = (l + r) >> 1;
        if(isvalid(mid)) l = mid;
        else r = mid;
    }
    printf("%d\n",l * n);

    return 0;
}
