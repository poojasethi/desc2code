#include <bits/stdc++.h>

using namespace std;

int n, a[101], b[101];
vector<int> v[10002];

bool solvable(int w) {
    priority_queue<int, vector<int>, greater<int> > q;
    for(int i=0;i<10002;i++) {
        for(int j=0; j<v[i].size();j++) {
            for(int k=0;k<w;k++) {
                q.push(v[i][j]);
            }
        }
        if(q.empty()) continue;
        if(q.top()<=i) return false;
        q.pop();
    }
    return true;
}

int main()
{
    cin >> n;
    for(int i=0;i<n;i++) {
        cin >> a[i] >> b[i];
        v[a[i]].push_back(b[i]);
    }
    int f=0;
    int l=0, h=10000/n;
    while(l<h) {
        int m=(l+h+1)/2;
        if(solvable(m)) l=m;
        else h=m-1;
    }
    cout << l*n << endl;
    return 0;
}
