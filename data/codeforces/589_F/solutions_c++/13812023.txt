#include<bits/stdc++.h>
using namespace std;

struct par
{
    int x, y;
};

par p[1000005];

bool comp(const par &a, const par &b)
{
    return a.y < b.y;
}

int n,hc[10005];

int main()
{
	scanf("%d",&n);
	for(int i = 1; i <= n; i++){
        scanf("%d%d", &p[i].x, &p[i].y);
        p[i].y --;
    }
	sort(p + 1, p + n + 1, comp);

	int l = 0, r = 10001;
	while(l + 1 < r){
		fill(hc, hc + 10005, 0);
		int mid = (l + r)/2;
		bool can = 1;
		for(int i = 1; i <= n; i++){
			int nec = mid;
			for(int j = p[i].x; j <= p[i].y && nec; j++)
				if(!hc[j]){
                    hc[j] = 1;
                    nec--;
                }
			if(nec){
                can = 0;
                break;
            }
		}
		if(can)
            l = mid;
		else r = mid;
	}
	printf("%d", l * n);

	return 0;
}
