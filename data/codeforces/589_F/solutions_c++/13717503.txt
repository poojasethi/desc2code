#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int, int> ii;

int n;
ii interval[128];

bool possible(int x) {
    set<int> S;
    set<int>::iterator it;
    for(int i=0; i<10001; i++) {
        S.insert(i);
    }

    for(int i=0; i<n; i++) {
        for(int j=0; j<x; j++) {
            it = S.lower_bound(interval[i].second);
            if (*it >= interval[i].first) return false;
            S.erase(it);
        }
    }

    return true;
}

int main() {
    scanf("%d", &n);

    for(int i=0; i<n; i++) {
        scanf("%d%d", &interval[i].second, &interval[i].first);
    }

    sort(interval, interval+n);

    int lo=0, mid, hi=10000;
    while(hi-lo > 1) {
        mid = (lo+hi)/2;
        if (possible(mid)) lo = mid;
        else hi = mid;
    }

    if (possible(hi)) printf("%d\n", hi*n);
    else printf("%d\n", lo*n);

    return 0;
}
