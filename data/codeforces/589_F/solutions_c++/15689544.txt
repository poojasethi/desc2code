#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
using namespace std;
struct dish
{
	int a,b,len;
}store[105];
bool cmp(dish a,dish b)
{ 
  return a.b<b.b;
}
int n;
bool vis[10010];
bool check(int x)
{
	memset(vis,0,sizeof(vis));
	int cnt;
	for(int i=0;i<n;i++)
	{
		cnt=0;
		for(int j=store[i].a;j<store[i].b;j++)
		{
			if(!vis[j])
			{
               vis[j]=1;
			   cnt++;
			}
			if(cnt==x)break;
		}
		if(cnt<x)return false;
	}
	return true;
}
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d%d",&store[i].a,&store[i].b);
		store[i].len=store[i].b-store[i].a;
	}
	sort(store,store+n,cmp);
	int le=0,ri=10000,ans=0;
	while(le<=ri)
	{
		int mid=(le+ri)>>1;
		if(check(mid))
		{
			if(mid>ans)
				ans=mid;
			le=mid+1;
		}
		else
			ri=mid-1;
	}
	printf("%d\n",(ans)*n);
	return 0;
}
