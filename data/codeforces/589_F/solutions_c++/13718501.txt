#define N (1<<14)
#include <bits/stdc++.h>

using namespace std;

int n,x,y;
vector<int> f[N];

struct Node
{
    int lf,ed;
    Node(int lf,int ed):lf(lf),ed(ed){}
    bool operator<(const Node f) const
    {
        return ed>f.ed;
    }
};

bool judge(int x)
{
    priority_queue<Node> q;
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<f[i].size();j++)
            q.push(Node(x,f[i][j]));

        if(!q.empty())
        {
            Node u=q.top(); q.pop();
            if(i+u.lf>u.ed) return 0;
            if(u.lf>1)
                q.push(Node(u.lf-1,u.ed));
        }
    }
    return 1;
}

int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>x>>y,f[x].push_back(y);

    int Left=0,Right=N-1;
    for(;Left<Right;)
    {
        int mid=((Left+Right)>>1)+1;
        if(judge(mid)) Left=mid;
        else Right=mid-1;
    }
    cout<<Left*n<<endl;
    return 0;
}
