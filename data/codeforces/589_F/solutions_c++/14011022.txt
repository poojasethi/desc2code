#include <bits/stdc++.h>
using namespace std;

typedef pair<int, int> pii;
const int N = 100, M = 10000;
pii a[N];
int b[N], n;

bool f(int m) {
	fill(b, b + n, 0);
	for (int i = 0; i <= M; i++) {
		int x = -1;
		for (int j = 0; j < n; j++) {
			if (b[j] < m && a[j].first < i && a[j].second >= i && (x == -1 || a[j].second < a[x].second)) {
				x = j;
			}
		}
		if (x != -1) {
			b[x]++;
		}
	}
	for (int i = 0; i < n; i++) {
		if (b[i] < m) {
			return false;
		}
	}
	return true;
}

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d%d", &a[i].first, &a[i].second);
	}
	int l = 0, r = M;
	while (l < r) {
		int m = (l + r + 1) / 2;
		if (f(m)) {
			l = m;
		}
		else {
			r = m - 1;
		}
	}
	cout << l * n;
}
