#include<bits/stdc++.h>

using namespace std;
int n,a[110],b[110];

bool check(int x)
{
    int cnt[110];
    memset(cnt,0,sizeof(cnt));
    for(int i=0;i<=10000;i++)
    {
        int now=0;
        for(int j=1;j<=n;j++)
        {
            if(i>=a[j] and i<b[j] and cnt[j]<x and (now==0 or b[j]<b[now]))
            now=j;
            if(i==b[j] and cnt[j]<x)return 0;
        }
        cnt[now]++;

    }
    return true;
}


int main()
{
 //   freopen("a.txt","r",stdin);
    cin>>n;
    for(int i=1;i<=n;i++)
    cin>>a[i]>>b[i];
    int ans=0;
    int l=0,r=10000,mid;
    while(l<=r)
    {
        mid=(l+r)/2;
        if(check(mid))
        {
            ans=mid;
            l=mid+1;
        }
        else r=mid-1;
 //       cout<<l<<" "<<r<<endl;
    }
 // cout<<ans<<endl;
    cout<<ans*n<<endl;

}
