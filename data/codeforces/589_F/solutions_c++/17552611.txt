#include<bits/stdc++.h>
using namespace std;
multiset<pair<int,int> >s[10050];
int ans[200];
bool can(int num,int n){
    memset(ans,0,sizeof(ans));
    for(int i=0;i<=10000;i++){
        multiset<pair<int,int> > ::iterator it=s[i].begin();
        while(ans[it->second]>=num&&it!=s[i].end()) it++;
        if(it!=s[i].end())ans[it->second]++;
    }
    for(int i=0;i<n;i++){
        if(ans[i]<num)return 0;
    }
    return 1;
}
int main()
{
    int n;
    scanf("%d",&n);
    int l=0,r=0;
    for(int i=0;i<n;i++){
        int a,b;
        scanf("%d%d",&a,&b);
        r=max(r,b-a);
        for(int j=a;j<b;j++){
            s[j].insert(make_pair(b-j,i));
        }
    }
    while(l<=r){
        int mid=(l+r)>>1;
        if(can(mid,n))l=mid+1;
        else r=mid-1;
    }
    printf("%d\n",r*n);
}
