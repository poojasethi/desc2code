#include <bits/stdc++.h>

using namespace std;

int n;
int f[105];
vector<pair<int, int> > v;

bool works(int val) {
	for(int i = 0; i < n; i++) f[i] = 0;
	for(int t = 0; t < 11000; t++) {
		for(int i = 0; i < n; i++) {
			if(f[i] >= val) continue;
			if(v[i].second > t) continue;
			if(v[i].first <= t) continue;
			f[i]++;
			break;
		}
	}
	//cout << val << " "; for(int i = 0; i < n; i++) cout << f[i] << " "; cout << endl;
	bool ret = true;
	for(int i = 0; i < n; i++) if(f[i] < val) ret = false;
	return ret;
}

int main () {
	ios_base::sync_with_stdio(0);

	int a,b;
	cin >> n;

	for(int i = 0; i < n; i++) {
		cin >> a >> b;
		v.push_back(make_pair(b,a));
	}
	sort(v.begin(), v.end());

	int low = 0, high = 16000;
	while(high - low > 1) {
		int mid = (high+low)/2;
		if(works(mid)) low = mid;
		else high = mid;
	}
	if(works(high)) cout << n*high << endl;
	else cout << n*low << endl;

	return 0;
}
