# include <stdio.h>
# include <string.h>

using namespace std;

int l=0,r=10000;
int ans,n,a[110],b[110];
int t[110];

inline bool check(int k) {
    memset(t,0,sizeof(t));
    for (int i=0;i<=10000;++i) {
        int id=0;
        for (int j=1;j<=n;++j)
            if(a[j]<=i&&b[j]>i&&t[j]<k) {
                if(id==0||b[id]>b[j]) id=j;
            }
        t[id]++;
    }
    for (int j=1;j<=n;++j) if(t[j]<k) return false;
    return true;
}

int main() {
    scanf("%d",&n);
    for (int i=1;i<=n;++i) scanf("%d%d",&a[i],&b[i]);
    
    while(l<=r) {
        int mid=l+r>>1;
        if(check(mid)) {
            ans=mid;
            l=mid+1;
        }
        else r=mid-1;
    }
    printf("%d\n",ans*n);
    
    return 0;
}
