#include<bits/stdc++.h>

using namespace std;
const int maxn = 11000;
typedef pair<int, int> P;

P t[maxn];
int n;
int cnt = 0, cur = 0;
int flag[maxn];

bool cmp(P a, P b)
{
	return a.second < b.second;
}

bool Judge(int val)
{
	cnt++;
	for(int i = 0;i<n;i++)
	{
		int cur = 0;
		for(int k = t[i].first;k<t[i].second;k++)
		{
			if(cur == val)	break;
			if(flag[k] != cnt)
			{
				cur++;
				flag[k] = cnt;
			}
		}
		if(cur < val)	return false;
	}
	return true;
}

int main()
{
	scanf("%d", &n);
	for(int i = 0;i<n;i++)
		scanf("%d%d", &t[i].first, &t[i].second);
	sort(t, t+n, cmp);
	int l = 0, r = 10000;
	while(l < r-1)
	{
		int mid = (l+r)/2;
		if(Judge(mid))	l = mid;	
		else r = mid;
	}
	printf("%d\n", l*n);
	return 0;
}
