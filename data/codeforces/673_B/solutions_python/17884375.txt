#! /usr/bin/env python
# http://www.codeforces.com/problemset/problem/673/B

import sys

line = [int(x) for x in sys.stdin.readline().split()]
m = line[1]
d2 = 1
d1 = line[0]
for i in xrange(m):
    line = [int(x) for x in sys.stdin.readline().split()]
    line.sort()
    if line[0] >= d1 or line[1] <= d2:
        #print d1, d2
        d1 = d2
        break
    d2 = max(d2, line[0])
    d1 = min(d1, line[1])
print d1 - d2
