import math
import sys

def num_splits(num_probs, sim_pairs):

	startnode = 1
	endnode = num_probs - 1

	for p,q in sim_pairs:
		p,q = min(p,q), max(p,q)-1
		if p > startnode:
			startnode = p
		if q < endnode:
			endnode = q

	return max(0, endnode-startnode+1)

lines = sys.stdin.readlines()
num_probs = int(lines[0].split()[0])
sim_pairs = [map(int,line.split()) for line in lines[1:]]
print num_splits(num_probs, sim_pairs)
