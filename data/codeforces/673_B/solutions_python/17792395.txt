import bisect
n, m = map(int, raw_input().split())
set1 = 1
set2 = n
error = False
for i in range(m):
    a, b = map(int, raw_input().split())
    if a > b:
        a, b = b, a
    if set2 <= a or set1 >= b:
        error = True
        break
    set1 = max(a, set1)
    set2 = min(b, set2)
res = 0
if error:
    res = 0
else:
    res = set2 - set1
print res
