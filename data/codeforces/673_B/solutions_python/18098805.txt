def solve(n, pairs):
    lo, hi = 1, n

    for pair_lo, pair_hi in pairs:
        lo = max(lo, pair_lo)
        hi = min(hi, pair_hi)
        if hi <= lo:
            return 0

    return hi - lo

n, m = map(int, raw_input().split())
pairs = [sorted(map(int, raw_input().split())) for _ in xrange(m)]
print solve(n, pairs)
