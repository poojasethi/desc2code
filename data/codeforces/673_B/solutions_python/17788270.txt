#!/usr/bin/python

from collections import deque
import math

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n, m = ia()

d1max = -1
d2min = 1e20
for i in range(m):
    d1, d2 = ia()
    if d1>d2:
        d1, d2 = d2, d1
    if d1>d1max:
        d1max = d1
    if d2<d2min:
        d2min = d2

#
# [1, d1max]   [d2min n]
#

ans = 0
for i in range(1, n+1):
    if i < d1max: continue
    if i >=d2min: continue
    d1 = i
    d2 = n - i
    if d1<1: continue
    if d2<1: continue
    ans = ans + 1

print ans
