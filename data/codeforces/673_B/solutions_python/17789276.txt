def solve():
    if max(left) > min(right):
	return 0
    if len(left & right) > 0:
	return 0
    
    return min(right) - max(left)

n, m = map(int, raw_input().split(' '))
if m == 0:
    print n-1
else:
    left = set()
    right = set()
    for i in range(m):
	l, r = sorted(map(int, raw_input().split(' ')))
	left.add(l)
	right.add(r)

    print solve()
