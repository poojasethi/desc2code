#include <cstdio>
#include <algorithm>
using namespace std;

int main() {
	int n, m,a,b,wmin,wmax;
	scanf("%d%d",&n,&m);
	wmax = 1;
	wmin = n;
	for (int i = 0; i < m; i++) {
		scanf("%d%d",&a,&b);
		wmax = max(min(a, b), wmax);
		wmin = min(max(a, b), wmin);
	}
	a = wmin - wmax;
	printf("%d\n",max(a,0));
}
