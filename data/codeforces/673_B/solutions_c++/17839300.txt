#include <iostream>
#include <algorithm>
using namespace std;

int main(){
     int n, m;
     cin >> n >> m;
     int a,b;
     int r = n, l = 1;
     for(int i = 0; i < m; i++){
	  cin >> a >> b;
	  l = max(l,min(a,b));
	  r = min(r,max(a,b));
     }
     if(r-l < 0)
	  cout << 0 << endl;
     else
	  cout << r-l << endl;
     return 0;
}
