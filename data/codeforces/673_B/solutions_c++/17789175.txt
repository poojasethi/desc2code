#include<bits/stdc++.h>
using namespace std;
int zb = 0;
int main() {
	int n, m;
	scanf("%d%d", &n, &m);
	int md1 = 1, md2 = n;
	int a, b;
	for(int i = 0; i < m; i++) {
		scanf("%d%d", &a, &b);
		if(a>b) swap(a, b);
		md1 = max(md1, a);
		md2 = min(md2, b);
		}
	int c = md2-md1;
	if(md2<=md1) c = 0;
	printf("%d\n", c);
}
