#include <iostream>
using namespace std;

int main(){
	int n,m;
	cin>>n>>m;
	int m1=1;
	int m2=n;
	for(int i=0;i<m;i++){
		int a,b;
		cin>>a>>b;
		if(a>b){
			int temp=a;
			a=b;
			b=temp;
		}
		if(a>m1)m1=a;
		if(b<m2)m2=b;
	}
	if(m1>=m2){
		cout<<0<<endl;
	}
	else{
		cout<<(m2-m1)<<endl;
	}
	return 0;
}
