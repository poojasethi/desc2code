#include<bits/stdc++.h>

using namespace std;

main(){
    int n,m;
    cin >> n >> m;
    int l = 1;
    int r = n - 1;
    for (int i=1; i<=m; i++){
        int u,v;
        cin >> u >> v;
        if (u > v)
            swap(u,v);
        l = max(l,u);
        r = min(r,v-1);
    }
    cout << max(r-l+1,0);
}
