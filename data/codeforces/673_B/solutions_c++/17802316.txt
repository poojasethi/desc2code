#include<iostream>
//#define swap(x,y) (x)+=(y) , (y)=(x)-(y) , (x)=(x)-(y);
using namespace std;
int main()
{
    int n,m,a,b,x=2,y;
    cin>>n>>m;
    y=n;
    for(int i=0;i<m;i++)
    {
        cin>>a>>b;
        if(a>b)
            swap(a,b);
        x=max(a+1,x);
        y=min(b,y);
    }
    cout<<max(y-x+1,0);
}
