#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,m;
    cin>>n>>m;
    int x=1,y=n-1;
    for(int i=1;i<=m;i++)
    {
        int d,c;
        cin>>d>>c;
        if(d>c)
         swap(d,c);
        x=max(x,d);
        y=min(y,c-1);
    }
    cout<<max(y-x+1,0)<<' ';
    return 0;
}
