#include <bits/stdc++.h>
using namespace std;

int n, m, ma, mi;

int main () {
	scanf ("%d %d", &n, &m);
	ma = 0, mi = n - 1;
	for (int i = 0; i < m; i++) {
		int u, v;
		scanf ("%d %d", &u, &v);
		u--, v--;
		if (v < u) swap (u, v);
		ma = max (ma, u);
		mi = min (mi, v);
	}
	printf ("%d\n", max (0, mi - ma));
	return 0;
}
