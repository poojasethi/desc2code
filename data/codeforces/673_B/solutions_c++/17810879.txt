#include <bits/stdc++.h>
using namespace std;

int main()
{
   int n, m;
   cin>>n>>m;
   int a, b, p=1, q=n;
   for(int i=0; i<m; i++){
      cin>>a>>b;
      if(a>b) swap(a, b);
      //if(i==0) p=a, q=b;
      p=max(p, a); q=min(q, b);
   }
   if(q>p) cout<<q-p<<endl;
   else cout<<0<<endl;
   return 0;
}
