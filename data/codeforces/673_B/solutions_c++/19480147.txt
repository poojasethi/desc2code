#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int n,m;
	while(cin>>n>>m){
		int l=1,r=n;
		for(int i=0;i<m;i++){
			int a,b;
			cin>>a>>b;
			if(a>b) swap(a,b);
			l=max(l,a);
			r=min(r,b);
		}
		if(r-l<0) printf("0\n");
		else printf("%d\n",r-l);
	}
	return 0;
}
