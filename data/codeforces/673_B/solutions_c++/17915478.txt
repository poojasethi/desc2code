#include<bits/stdc++.h>
using namespace std;

const int N = (int)1e5+11;

int main()
{
	int n,m;
	cin>>n>>m;
	if(m==0){
		cout<<n-1<<endl;
		return 0;
	}
	int l=-1;
	int r= INT_MAX;
	for(int i=0;i<m;i++){
		int a,b;
		cin>>a>>b;
		if(a>b)
			swap(a,b);
		l=max(l,a);
		r=min(r,b);
	}
	cout<<max(r-l,0)<<endl;















	return 0;
}
