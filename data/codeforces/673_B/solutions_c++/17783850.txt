#include<cstdio>

int n,m;
int u,v,t;
int a,b;

int main()
{
    scanf("%d%d",&n,&m);
    
    a=1;
    b=n;
    
    for(;m>0;m--)
    {
        scanf("%d%d",&u,&v);
        if(u>v) t=u,u=v,v=t;
        
        if(u>a) a=u;
        if(v<b) b=v;
    }
    t=b-a; if(t<0) t=0;
    printf("%d\n",t);
    return 0;
}
