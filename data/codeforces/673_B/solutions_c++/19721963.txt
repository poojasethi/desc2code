#include<cstdio>
#include<algorithm>
using namespace std;
int n,m;
int main(void){
	scanf("%d%d",&n,&m);
	int mi=1;
	int ma=n;
	for(int i=0;i<m;i++){
		int x,y;
		scanf("%d%d",&x,&y);
		if(x>y)swap(x,y);
		mi=max(mi,x);
		ma=min(ma,y);
	}
	printf("%d\n",max(ma-mi,0));
	
	return 0;
}
