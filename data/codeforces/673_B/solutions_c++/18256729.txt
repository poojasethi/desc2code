#include <bits/stdc++.h>

using namespace std;

int main() {
	ios_base::sync_with_stdio(false);
	int n, m, X, Y;
	cin >> n >> m;
	X = 1, Y = n;
	for (int i = 0; i < m; i++) {
		int x, y;
		cin >> x >> y;
		if (x > y)
			swap(x, y);
		X = max(x, X);
		Y = min(y, Y);
	}
	cout << max(Y - X, 0) << endl;
	
	return 0;
}
