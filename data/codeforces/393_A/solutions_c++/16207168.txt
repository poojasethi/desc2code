#include <iostream>
#include <string>
#include <algorithm>
using  namespace std;

long long arr[26];

int main(){
	string s;
	cin >> s;
	long long i, j, ans=0;
	for(i=0;i<s.size();i++){
		arr[s[i]-'a']++;
	}
	ans = arr['i'-'a'];
	ans = min(ans, arr['t'-'a']);
	ans = min(ans, (arr['n'-'a']-1)/2);
	ans = min(ans, arr['e'-'a']/3);
	cout << ans << '\n';
	return 0;
}
