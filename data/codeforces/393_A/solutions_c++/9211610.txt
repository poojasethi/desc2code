#include <iostream>
using namespace std;

string s;
int f[100];
int ans = 1000000;

int main() {
    cin >> s;
    for (int i = 0; s[i]; ++i)
        ++f[s[i] - 'a'];
    ans = min(ans, f['i' - 'a']);
    ans = min(ans, f['t' - 'a']);
    ans = min(ans, (f['n' - 'a'] - 1) / 2);
    ans = min(ans, f['e' - 'a'] / 3);
    cout << ans;
    return 0;
}
