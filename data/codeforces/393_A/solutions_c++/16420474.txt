#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main()
{
	int c[26] = {0};
	string s;
	cin >> s;
	for (int i = 0; i < s.length(); i++)
		c[s[i] - 'a']++;
	int n = c[13];
	int i = c[8];
	int e = c[4];
	int t = c[19];
	cout << min((n - 1) / 2, min(i, min(e / 3, t))) << endl;
	return 0;
}
