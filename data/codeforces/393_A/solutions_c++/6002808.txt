#include<stdio.h>

int a[30];
char s[105];

int min(int x,int y){
	return x>y?y:x;
}

int main(){
	scanf("%s",s);
	int i;
	for(i=0;i<26;i++)
		a[i]=0;
	for(i=0;s[i]!='\0';i++)
		a[s[i]-'a']++;
	int m=min(a['e'-'a']/3,(a['n'-'a']-1)/2);
	m=min(m,a['i'-'a']);
	m=min(m,a['t'-'a']);
	printf("%d\n",m);
	return 0;
}
