#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
	int n=0,i=0,e=0,t=0;
	char c;
	int ans = 0,min1 = 0,min2 = 0;
	while(scanf("%c",&c)&&c!='\n')

	{
		if(c == 'n') n++;
		if(c == 'i') i++;
		if(c == 'e') e++;
		if(c == 't') t++;
	}
	ans = min(min(min((n-1)/2,i),e/3),t);
	printf("%d\n",ans);
}
