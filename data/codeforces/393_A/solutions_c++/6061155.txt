#include <cstdio>

int count['z'+1];

int main(){
	char s[110];
	scanf("%s", s);
	for(int i = 0; s[i]; i++)
		count[s[i]]++;
	int min = count['i'];
	if(min > count['t'])
		min = count['t'];
	if(count['e']/3 < min)
		min = count['e']/3;
	if(min*3 - (min-1) > count['n']){
		min = (count['n'] - 1)/2;
	}
	printf("%d\n",min);
	return 0;
}
