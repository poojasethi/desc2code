#include<cstdio>
#include<algorithm>

using namespace std;

char s[200];
int c[26];

int main()
{
	scanf("%s", s);
	for (int i = 0; s[i]; i++) {
		c[s[i] - 'a']++;
	}
	int k = min(c['i'-'a'], min(c['t'-'a'], c['e'-'a']/3));
	if (c['n' - 'a'] >= 2*k + 1) {
		printf("%d\n", k);
	}
	else {
		printf("%d\n", (c['n' - 'a']-1)/2);
	}
	return 0;
}
