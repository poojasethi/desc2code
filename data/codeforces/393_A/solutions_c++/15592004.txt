#include<stdio.h>
int main()
{
	char A[100];
	gets(A);
	int i,a=0,b=0,c=0,d=0,max;
	for(i=0;A[i]!='\0';i++){
		if(A[i]=='n') a++;
		else if(A[i]=='i') b++;
		else if(A[i]=='e') c++;
		else if(A[i]=='t') d++;
	}
	c=c/3;
	a=(a-1)/2;
	max=a;
	if(max>b) max=b;
	if(max>c) max=c;
	if(max>d) max=d;
	printf("%d",max);
	return 0;
}
