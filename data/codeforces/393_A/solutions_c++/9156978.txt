#include <iostream>
#include <cstdio>
#include <string>
#include <algorithm>
using namespace std;

int arr[26];

int main()
{
    string s;
    cin >> s;
    int i;
    for (i = 0; i < s.length(); i++) arr[s[i]-'a']++;
    cout << min((arr[13]-1)/2,min(arr[4]/3,min(arr[19],arr[8]))) << endl;
    //system("pause");
    return 0;
}
