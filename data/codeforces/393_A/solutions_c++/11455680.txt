#include <bits/stdc++.h>
using namespace std;
#define int long long

main()
{
	string s;
	cin >> s;
	int n = 0, i = 0, e = 0, t = 0;
	for (int k = 0; k < s.size(); k++)
	{	
		if (s[k] == 'n') n++;
		if (s[k] == 'i') i++;
		if (s[k] == 'e') e++;
		if (s[k] == 't') t++;
	}
	cout << min(i, min((n - 1) / 2, min(t, e / 3)));
}
