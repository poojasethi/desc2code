#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    string s;
    cin>>s;
    int a,b,c,d;
    a=count(s.begin(),s.end(),'e');
    b=count(s.begin(),s.end(),'i');
    c=count(s.begin(),s.end(),'n');
    d=count(s.begin(),s.end(),'t');
    c=max(((c-1)/2),0);
    a/=3;
    cout<<min(a,min(b,min(c,d)));
    return 0;
}
