#include <stdio.h>
#include <algorithm>

using namespace std;

int k[228];
int ans;

int main ()
{
    char c = getchar();
    while (c != '\n') {
        ++k[c];
        c = getchar();
    }
    
    ans = min(k['e']/3, min(k['i'], min(max((k['n']+1)/2 - 1, 0), k['t'])));
    
    printf("%d\n", ans);
    
    return 0;
}
