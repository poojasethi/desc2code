#include<iostream>
using namespace std;
int main()
{
	char x[100];
	while(cin>>x)
	{
		int a=0,b=0,c=0,i,d=0;
		for(i=0;x[i];i++)
		if(x[i]=='n')a++;
		else if(x[i]=='i')b++;
		else if(x[i]=='e')c++;
		else if(x[i]=='t')d++;
		a=(a-1)/2;
		c/=3;
		if(a>b)a=b;
		if(a>c)a=c;
		if(a>d)a=d;
		cout<<a<<endl;
	}
}
