#include<iostream>
using namespace std;
int main ()
{ ios_base::sync_with_stdio(0);
  string s;  cin>>s;
  int n=0, e=0, i=0, t=0;
  for(int j=0;j<s.size();j++)
  { if(s[j]=='n')n++;  if(s[j]=='i')i++;
    if(s[j]=='e')e++;  if(s[j]=='t')t++;
  }
  if(n>0)n--;
 cout<<min(min((n/2),(i)),min((e/3),(t)));
  return 0;
}
