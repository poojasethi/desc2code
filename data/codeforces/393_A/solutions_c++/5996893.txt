#include<stdio.h>
int main()
{
	char str[110];
	int i=0,n=0,e=0,t=0,k=0,q;
	gets(str);
	for(q=0;str[q];q++)
	{
		if(str[q]=='i')i++;
		else if(str[q]=='n')n++;
		else if(str[q]=='t')t++;
		else if(str[q]=='e')e++;
	}
	while(n>=3&&i>=1&&e>=3&&t>=1)
	{
		n-=2;
		i--;
		e-=3;
		t--;
		k++;
	}
	printf("%d\n",k);
	return 0;
}
