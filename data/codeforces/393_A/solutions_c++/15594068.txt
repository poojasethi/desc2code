#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
	int t=0,i=0,n=0,e=0;
	char c;
	while((c=getchar())!='\n')
	{
		if(c=='t')t++;
		if(c=='i')i++;
		if(c=='n')n++;
		if(c=='e')e++;
	}
	e=e/3;
	n=(n-1)/2;
	cout<<min(min(t,i),min(n,e))<<endl;
	return 0;
}
