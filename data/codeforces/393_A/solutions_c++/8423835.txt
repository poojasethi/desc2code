#include<iostream>
#include<algorithm>
using namespace std;
int h2[26];
int main(){
	string s;cin>>s;
	int l=s.length();
	for(int i=0;i<l;i++) h2[s[i]-'a']++;
	
	int a;
	if(h2['n'-'a']>=3)
	a=((h2['n'-'a']-3)/2)+1;
	else a=0;
	a=min(a,h2['e'-'a']/3);
	a=min(a,h2['i'-'a']/1);
	a=min(a,h2['t'-'a']/1);
	
	cout<<a<<endl ;
	return 0;
}
