#include <stdio.h>
#include <algorithm>
using namespace std;

char str[111];
int counts[256];
int main(){
	gets(str);
	for(char*p=str;*p;++p)
		counts[*p]++;

	int maxi = max((counts['n'] - 1)/2, 0);
	maxi = min(maxi, counts['e']/3);
	maxi = min(maxi, counts['i']);
	maxi = min(maxi, counts['t']);

	printf("%d\n", maxi);
}
