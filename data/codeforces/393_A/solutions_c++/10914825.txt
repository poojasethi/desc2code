#include <iostream>
#include<cmath>
using namespace std;
int main()
{
    int num[26] = {0}, result;
    char c;
    while(cin >> c) num[c - 97]++;
    num['n' - 97] = (num['n' - 97] - 1) / 2;
    num['e' - 97] /= 3;
    result = min(num['n' - 97], min(num['e' - 97], min(num['i' - 97], num['t' - 97])));
    cout << result;
    return 0;
}
