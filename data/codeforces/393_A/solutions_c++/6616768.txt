#include <bits/stdc++.h>

using namespace std;

const int maxN = 1000*1000 + 321;
map < char , int> m;
int main()
{
  string s;
  cin >> s;
  for(int i = 0 ; i < s.size() ; i ++)
    m[s[i]] ++;
  int ans;
  ans = min((m['n'] - 1) / 2, m['e'] / 3);
  ans = min(ans , m['i']);
  ans = min(ans , m['t']);
  cout << ans << endl;
  return 0;
}
