#include<stdio.h>
#include<string.h>
int main()
{
	char a[101];
	scanf("%s",a);
	int n=0,i=0,t=0,e=0,j;
	for(j=0;j<=strlen(a)-1;j++)
	{
		if(a[j]=='n')
			n++;
		if(a[j]=='i')
			i++;
		if(a[j]=='t')
			t++;
		if(a[j]=='e')
			e++;
	}
	int k;
	k=(n-1)/2;
	if(i<k)
		k=i;
	if(t<k)
		k=t;
	if(e/3<k)
		k=e/3;
	printf("%d",k);
	return 0;
}
