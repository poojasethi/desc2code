#include <stdio.h>
#include <iostream>
#include <string.h>

using namespace std;

int cnt[26];

int main (void) {
	char s[101];
	int i, n;
	scanf ("%s", s);
	int len = strlen(s);
	for (i = 0; i < len; i++) {
		cnt[s[i]-'a']++;
	}
	n = min((cnt['n'-'a']-1)/2, min(cnt['i'-'a'], min(cnt['e'-'a']/3, cnt['t'-'a'])));
	printf ("%d\n", n);
	
	return 0;
}
