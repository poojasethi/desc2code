#include<stdio.h>
#include<algorithm>

using namespace std;

int ch[30];

int main() {
  for (char c='a'; c<='z'; c++) { ch[c-'a'] = 0; }
  char c;
  while ((c = getchar())!=EOF) {
    if (c<'a') {break;}
    ch[c-'a']++;
  }
  printf("%d", min(min(min(ch['e'-'a']/3, (ch['n'-'a']-1)/2), ch['i'-'a']), ch['t'-'a']));

}
