#include<bits/stdc++.h>
using namespace std;
string s;
int hsh[26],i,mn=1e9,f;
int main()
{
     cin >> s;
     for( i = 0 ; i<(int)s.length() ; i++ )hsh[s[i]-'a']++;
     if(hsh['n'-'a'] > 3)f=1;
     mn=min(f==0?hsh['n'-'a']/3:1+(hsh['n'-'a']-3)/2,min(hsh['e'-'a']/3,min(hsh['i'-'a'],min(hsh['t'-'a'],mn))));
     cout << mn;  
	 return 0;
}
