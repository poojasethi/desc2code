#include<cstdio>
#include<algorithm>

using namespace std;

char s[111];
int a[26];

int main() {
	scanf("%s", s);
	for (int i = 0; s[i]; ++i)
		++a[s[i] - 'a'];
	--a['n' - 'a'];
	int nn = a['n' - 'a'] / 2;
	int ne = a['e' - 'a'] / 3;
	int ni = a['i' - 'a'];
	int nt = a['t' - 'a'];
	printf("%d\n", min(min(nn, ne), min(ni, nt)));
	return 0;
}
