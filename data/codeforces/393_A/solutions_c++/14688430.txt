#include<iostream>
using namespace std;
int main()
{
    string s;
    cin>>s;
    long long a[28]={0},i,ans=0,x=0;
    for(i=0;i<s.length();i++)
    {
        a[s[i]-'a']++;
    }
    if(a['n'-'a']>=3)
    x=((a['n'-'a']-3)/2)+1;
    else
    x=0;
    ans=min(x,min(a['e'-'a']/3,min(a['i'-'a'],a['t'-'a'])));
    cout<<ans;
    return 0;
}
