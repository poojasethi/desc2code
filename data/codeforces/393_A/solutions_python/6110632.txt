import sys
import math
from collections import defaultdict

def main():
  n = sys.stdin.readline()
  letter_map = defaultdict(int)

  for c in n:
    letter_map[c] += 1

  ns = 0
  if letter_map['n'] < 3:
    ns = 0
  elif letter_map['n'] == 3:
    ns = 1
  else:
    ns = (letter_map['n'] - 1) / 2
  es = letter_map['e']/3
  i = letter_map['i']
  t = letter_map['t']
  print min(ns, es, i, t)

if __name__ == '__main__':
  main()
