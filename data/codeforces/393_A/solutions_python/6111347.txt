#nineteen.py
import sys

def main():
    word = sys.stdin.readline()
    n = 0
    i = 0
    e = 0
    t = 0

    for letter in word:
        if letter == "n": n += 1
        elif letter == "i": i += 1
        elif letter == "e": e += 1
        elif letter == "t": t += 1
    c = 0
    if n >= 3:
        c = 1
        n -= 3
    else:
        n = 0
    while n >= 2:
        n -= 2
        c += 1
    e /= 3

    m = min(min(e,i),min(c,t))
    print m

main()
