#!/usr/bin/env python

if __name__ == '__main__':
    string = raw_input()
    count = {}
    for w in 'abcdefghijklmnopqrstuvwxyz':
        count[w] = 0
    for s in string:
        count[s] += 1
    count['e'] /= 3
    count['n'] = (count['n'] - 1) / 2
    min_char_count = min(count['n'], min(count['e'], min(count['i'], count['t'])))
    print min_char_count if min_char_count >= 0 else 0
