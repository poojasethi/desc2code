"""http://codeforces.com/problemset/problem/39/B"""
from __future__ import print_function

def easy(n, A):
    to_find = 1
    length = 0
    answer = []
    for x in xrange(n):
        if A[x] == to_find:
            length +=1
            to_find+=1
            answer.append(x)
    print(length)
    for x in answer:
        print(x+1+2000, end = ' ')

if __name__ == '__main__':
    n = int(raw_input())
    easy(n, map(int, raw_input().strip().split()))
