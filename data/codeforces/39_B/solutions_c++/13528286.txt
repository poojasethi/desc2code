#include <bits/stdc++.h>
using namespace std;

int main() {
    int n,a[100],u[100],f=1;
    cin >> n;
    for (int i=0; i<n; i++) cin >> a[i]; memset(u,0,sizeof(u));
    for (int i=0; i<n; i++)
        if (a[i]==f) { u[i]++; f++; }
    cout << f-1 << endl;
    for (int i=0; i<n; i++)
        if (u[i]==1) cout << 2001+i << " ";
}
