#include <iostream>
#include <vector>
#define		pb		push_back
using namespace std;
vector <int> v;
int yr=1,a=0,arr[1001],n;
int main()
{
	int N;
	cin>>N;
	for(int i=0;i<N;i++)	cin>>arr[i];
	for(int i=0;i<N;i++){	
		if(arr[i]==yr){
			v.pb(2000+i+1);
			yr++;
			a++;
		}
	}
	int sz = v.size();
	cout<<a<<endl;
	for(int i=0;i<sz;i++){
			cout<<v[i]<<" " ;
	}
}
