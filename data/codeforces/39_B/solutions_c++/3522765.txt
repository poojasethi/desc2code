#include <stdio.h>
int main(){
    int  n;
    int a[1000];
    int m=0;
    scanf("%d",&n);
    for (int i=0;i<n;i++){
        int b;
        scanf("%d",&b);
        if (b==m+1){
            a[m]=2000+i;
            m++;
        
        }
    }
    printf("%d\n",m);
    for (int i=0;i<m;i++)
        printf("%d ",a[i]+1);
   
}
