#include<stdio.h>

main()
{
      int n,k,i,x;
      int a[101];
      
      scanf("%d",&n);
      k=1;
      for(i=1; i<=n; i++)
      {
               scanf("%d",&x);
               if(x==k) { a[k]=i+2000; k++; }
      }
      printf("%d\n",k-1);
      for(i=1; i<k; i++)
        printf("%d ",a[i]);
      return 0;
}
