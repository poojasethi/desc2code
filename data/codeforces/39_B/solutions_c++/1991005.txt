#include<iostream>
using namespace std;
int main()
{
	int n;
	cin >> n;
	int i;
	int A[n],B[n],top1=0;
	int cur=1;
	for(i=0;i<n;i++)
	{	
		cin >> A[i];
		if(A[i]==cur)
		{
			B[top1]=2001+i;
			cur++;
			top1++;
		}
	}
	if(top1==0)
		cout << "0" << endl;
	else
	{
		cout << top1 << endl;
	for(i=0;i<top1;i++)
	{
		cout << B[i] << " ";
	}
	cout << endl;
	}
}
