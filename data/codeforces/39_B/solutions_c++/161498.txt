#include<stdio.h>
int rez[1000];
int p;
int main()
{
	//freopen("in","r",stdin);
	int n,k;
	k=1;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		int a;
		scanf("%d",&a);
		if(a==k)
		{
			rez[p++]=2000+i+1;
			k++;
		}
	}
	if(k==1)
		printf("0");
	else
	{
		printf("%d\n",p);
		for(int i=0;i<p;i++)
			printf("%d ",rez[i]);
	}
	return 0;
}
