#include <iostream>
using namespace std;
int a1[150], b1[150];
int main(){
	int n;
	cin >> n;
	for(int i = 1; i <= n; i ++){
		cin >> a1[i];
	} 
	int cnt = 1;
	for(int i = 1; i <= n; i ++)
		if(a1[i] == cnt)
			b1[cnt++] = i + 2000;
	cout << cnt - 1 << endl;
	for(int i = 1; i < cnt; i ++)
		cout << b1[i] << " ";
	cout << endl;
	return 0;
}
