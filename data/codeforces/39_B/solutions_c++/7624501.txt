#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,ind=1,tmp;
    cin>>t;
    vector<int>ans;
    for(int i=0;i<t;i++){
        cin>>tmp;
        if(tmp==ind){
            ind++;
            ans.push_back(i+2001);
        }
    }
    cout<<ans.size()<<endl;
    for(int i=0;i<ans.size();i++)
        cout<<ans[i]<<" ";
    return 0;
}
