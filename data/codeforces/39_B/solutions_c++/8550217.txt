#include<iostream>
using namespace std;
int main()
{
    int n,i,t=1,p=0;
    cin>>n;
    int a[n+10],dd[n+10];
    for(i=0;i<n;i++)
        cin>>a[i];
    for(i=0;i<n;i++)
    {
        if(a[i]==t)
        {
            dd[p]=i; p++; t++;
        }
    }
    cout<<p<<endl;
    for(i=0;i<p;i++)
        cout<<2001+dd[i]<<" ";




}
