#include<stdio.h>
int main()
{
	int t;
	scanf("%d",&t);
	
	int a[300],i,y[300];
	
	for(i=0;i<t;i++)
		scanf("%d",&a[i]);
	
	int count=1,j=0;
	
	for(i=0;i<t;i++)
	{
		if(a[i]==count)
		{
			y[j]=2000+i+1;
			j++;
			count++;
		}
	}
	printf("%d\n",count-1);
	for(i=0;i<j;i++)
		printf("%d ",y[i]);
	printf("\n");
		
		
	return 0;
}
