#include<stdio.h>
#include<stdlib.h>
int main(){
	int i,j,ans[1001],A[1001],n,temp;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&A[i]);
	temp = 1;
	int index = 0;
	for(i=1;i<=n;i++){
		if(A[i] == temp){
			temp++;
			ans[++index] = i;
		}
	}
	printf("%d\n",index);
	if(index){
		for(i=1;i<index;i++)
			printf("%d ",2000+ans[i]);
		printf("%d\n",2000+ans[index]);
	}
	return 0;
}
