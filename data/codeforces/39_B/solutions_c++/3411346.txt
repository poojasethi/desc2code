#include<stdio.h>
int main()
{
	int n,year[101],i,m,t=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&m);
		if(m==t+1)
		{
			year[t]=2001+i;
			t++;
		}
	}
	if(t)
	{
		printf("%d\n",t);
		for(i=0;i<t-1;i++)
			printf("%d ",year[i]);
		printf("%d\n",year[i]);
	}
	else
		printf("0\n");
	return 0;
}
