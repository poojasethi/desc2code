#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n, a[100], step = 1;
	scanf("%d", &n);
	for( int i = 1; i <= n; i ++){
		scanf("%d", &a[i]);
		if(a[i] == step) a[i] = 2000 + i, step ++;
	}
	printf("%d\n",step - 1);
	for( int i = 1; i <= n; i ++)
		if(a[i] > 2000)
		printf("%d ",a[i]);
	printf("\n");
	return 0;
}
