#include<cstdio>
int a[100],b[100];
int main(){
	int n,i,j,p,f;
	while(~scanf("%d",&n)){
		for(i=0;i<n;i++)scanf("%d",a+i);
		f=p=0;
		for(j=1;j<=100;j++){
			for(i=p;i<n;i++)if(a[i]==j){b[f++]=i;break;}
			if(i==n)break;p=i+1;
		}
		printf("%d\n",f);
		for(i=0;i<f;i++)printf("%d ",2001+b[i]);
	}
	return 0;
}
