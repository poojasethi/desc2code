#include <stdio.h>
int main(){
	int n;
	while(~scanf("%d",&n)){
		int i,a[109],b[1000],l=1;
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		for(i=0;i<n;i++)
			if(a[i]==l) b[l]=i+1+2000,l++;
		if(l>1){
			printf("%d\n",l-1);
			for(i=1;i<l-1;i++)
				printf("%d ",b[i]);
			printf("%d\n",b[i]);
		}
		else printf("0\n");
	}
	return 0;
}
