#include<stdio.h>
int main(){
	int n,a[1000],i,ans[1000];
	while(~scanf("%d",&n)){
		for(i=1;i<=n;i++){
			scanf("%d",&a[i]);
		}
		int k=1;
		for(i=1;i<=n;i++){
			if(a[i]==k)ans[k++]=i+2000;
		}
		printf("%d\n",k-1);
		for(i=1;i<k-1;i++)
			printf("%d ",ans[i]);
		if(k>1)
		printf("%d\n",ans[k-1]);
	}
	return 0;
}
