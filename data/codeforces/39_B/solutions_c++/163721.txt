#include<iostream>
#include<vector>
using namespace std;
int main()
{
	int n,r = 1;
	cin>>n;
	vector<int> a(n);
	vector<int> ans;
	for(int i = 0; i < n; i++)
	{
		cin>>a[i];
		if (a[i] == r)
		{
			ans.push_back(i + 2001);
			r++;
		}
	}
	cout<<ans.size()<<endl;
	for(int i = 0; i < (int)ans.size(); i++)
		cout<<ans[i]<<" ";
	return 0;   
}
