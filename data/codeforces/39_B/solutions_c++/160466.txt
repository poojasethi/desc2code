#include <iostream>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int ans[100];
    int k=0;
    int exp=1;
    for(int i=0;i<n;i++)
    {
        int a;
        cin>>a;
        if(a==exp)
        {
            ans[k++]=2000+i+1;
            exp++;
        }
    }
    cout<<k<<endl;
    for(int i=0;i<k;i++)
        cout<<ans[i]<<' ';
    return 0;
}
