#include <iostream>
#include <vector>
using namespace std;

vector<int> ans;

int main()
{
	int n;
	cin >> n;
	for(int i=1; i<=n; i++)
	{
		int d;
		cin >> d;
		if(d == ans.size() + 1)
			ans.push_back(2000 + i);
	}
	cout << ans.size() << endl;
	for(int i=0; i<ans.size(); i++)
		cout << ans[i] << " ";
	cout << endl;
	return 0;
}
