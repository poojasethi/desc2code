#include<stdio.h>
int a[102],b[102];
int main()
{
	int n,sum=0;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		if(a[i]==sum+1)
		{b[sum]=2000+i;
		sum++;}
	}
	printf("%d\n",sum);
	if(sum)
	{
		for(int i=0;i<sum-1;i++)
		printf("%d ",b[i]);
		printf("%d",b[sum-1]);
	}
	return 0;
}
