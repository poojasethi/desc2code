#include <iostream>
#include <cstdio>
using namespace std;
#define MAX 101

int n, a, cnt, A[MAX];

int main() {
	scanf("%d", &n);
	cnt = 1;
	for (int i = 0; i < n; i++) {
		scanf("%d", &a);
		if (a == cnt) {
			A[cnt++] = 2001 + i;
		}
	}
	printf("%d\n", cnt - 1);
	for (int i = 1; i < cnt; i++) {
		printf("%d ", A[i]);
	}
	return 0;
}
