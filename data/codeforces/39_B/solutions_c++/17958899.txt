#include<stdio.h>
#define N 111
int p[N];
int n, x;
int main(){
    scanf("%d", &n);
    int now = 1;
    int cnt = 0;
    for(int i =  0; i < n; i ++) {
        scanf("%d", &x);
        if(x == now) now ++, p[cnt ++] = 2000 + i + 1;
    }
    printf("%d\n", cnt);
    for(int i = 0; i < cnt; i ++) printf("%d ", p[i]);
    puts("");
}
