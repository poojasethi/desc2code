#include<iostream>
#include<vector>
using namespace std;

int a[110];
int n;
vector<int> ans;

void calc()
{
	int cur=1;
	for(int i=0;i<n;i++)
		if(a[i]==cur)
		{
			ans.push_back(i);
			cur++;
		}
}

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i];
	calc();
	cout<<(int)ans.size()<<endl;
	for(int i=0;i<(int)ans.size();i++)
		cout<<ans[i]+2001<<' ';
	cout<<endl;
	return 0;
}
