#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

int a[100];

int main()
{
int n;
cin >> n;
int mx=-1;
int i=0;
while(n>0)
{
i++;
a[i]=n%10;
mx=max(mx,n%10);
n/=10;

}
cout<<mx<<endl;

for(int j=1;j<=mx;j++)
{
int p=0;

for(int o=i;o>=1;o--)
if(a[o]>=1)
{
a[o]--;
p=p*10+1;
}
else
p=p*10;

cout<<p<<" ";
}

return 0;
}
