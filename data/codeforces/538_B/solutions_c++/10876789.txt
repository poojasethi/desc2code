#include <bits/stdc++.h>

using namespace std;

int n, out[10], w;

int main() {
	scanf("%d", &n);
	int pot = 1;
	while (n > 0) {
		w = max(w, n%10);
		for (int i = 0; i < n%10; i++) out[i] += pot;
		n /= 10;
		pot *= 10;
	}
	printf("%d\n", w);
	for (int i = 0; i < w; i++) printf("%d ", out[i]);
	printf("\n");
	return 0;
}
