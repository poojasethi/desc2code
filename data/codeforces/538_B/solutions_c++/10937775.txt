#include <iostream>
using namespace std;
int a[]={1,10,100,1000,10000,100000,1000000},c[7];
int main()
{
	int n,k=0,i=0,out;
	cin>>n;
	while (n)
	{
		c[i]=n%10;
		k=max(k,c[i]);
		n/=10;
		i++;
	}
	cout<<k<<endl;
	while (k--)
	{
		out=0;
		for (i=0;i<7;i++) if (c[i]) {out+=a[i];c[i]--;}
		cout<<out;
		if (k) cout<<" ";
	}
	cout<<endl;
	return 0;
}
