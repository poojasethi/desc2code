#include <cstdio>

int main(){
	char s[10];
	scanf("%s",s);
	int n=0,m=0;
	for(;s[n];n++){
		if(s[n]-'0'>m)m=s[n]-'0';
	}
	printf("%d\n",m);
	for(int i=0;i<m;i++){
		bool f=0;
		for(int j=0;j<n;j++){
			if(s[j]>'0'){
				printf("1");
				s[j]--;
				f=1;
			}else if(f){
				printf("0");
			}
		}
		printf("%c",i+1<m?' ':'\n');
	}
	return 0;
}
