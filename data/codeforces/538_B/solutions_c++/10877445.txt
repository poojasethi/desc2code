#include<cstdio>
#define max(a,b) (a)>(b)?(a):(b)
int n,cnt,a[10],ma=0;
int main(){
	scanf("%d",&n);
	while(n)a[++cnt]=n%10,n/=10;
	for(int i=1;i<=cnt;i++) ma=max(ma,a[i]);
	printf("%d\n",ma);
	for(int i=1;i<=ma;i++){
		int ans=0;
		for(int j=cnt;j>=1;j--) 
			if(a[j])ans*=10,ans++,a[j]--;
			else ans*=10;
		printf("%d ",ans);
	}
	return 0;
}
