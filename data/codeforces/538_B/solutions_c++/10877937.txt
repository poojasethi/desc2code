#include <iostream>

using namespace std;

int main()
{
	int n, a[9]={0}, r = 0;
	cin >> n;
	int t = 1000000;
	while (t > 0)
	{
		int i = 0;
		while (n >= t)
		{
			a[i] += t;
			n -= t;
			i++;
			r = r < i ? i : r;
		}
		t /= 10;
	}
	cout << r << endl;
	for (int i = 0; i < r; i++)
		cout << a[i] << " ";
	cout << endl;
	return 0;
}
