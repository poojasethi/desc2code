#include<stdio.h>
int n,max=-1,t,i,r,e,us=1,arr[1000000];
int main ()
{
	scanf("%d",&n);
	t=n;
	while(t)
	{r++;t/=10;}
	for(i=0; i<r-1; i++)
		us*=10;
	while(n>0){
		t=n;
	for(i=0; i<t/us; i++)
	{arr[i]+=us;n-=us;if (i>max)max=i;}
	r--;
	us/=10;
	if (!us)
		us=1;
	}
	printf("%d\n",max+1);
	for(i=0; i<max+1; i++)
		printf("%d ",arr[i]);
	return 0;
}
