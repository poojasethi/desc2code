#include<bits/stdc++.h>
using namespace std;

vector<int> res;

int main(){
	int n;
	cin >> n;
	while(n>0){
		int cur=0, p=1;
		while(n/p>0){
			if(n/p%10>0)
				cur+=p;
			p*=10;
		}
		res.push_back(cur);
		n-=cur;
	}
	cout << res.size() << endl;
	for(int i=0;i<res.size();i++)
		cout << res[i] << " ";
	cout << endl;
}
