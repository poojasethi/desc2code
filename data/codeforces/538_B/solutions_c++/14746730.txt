#include <iostream>

using namespace std;

int r[999];

int main()
{
	int n, k;
	cin >> n;
	int ans = 0;
	while (n > 0)
	{
		int num = 1;
		while (num <= n)
			num *= 10;
		k = 0;
		while (num /= 10)
		{
			if (n / num % 10)
				k += num;
		}
		r[ans] = k;
		n -= k;
		ans++;
	}
	cout << ans << endl;
	for (int i = ans - 1; i >= 0; i--)
		cout << r[i] << ' ';
}
