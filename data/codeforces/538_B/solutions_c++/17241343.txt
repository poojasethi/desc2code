#include <bits/stdc++.h>
#define ri(n) scanf("%d",&n)
#define FOR(i,l,r) for(int i=l;i<r;i++)
using namespace std;

int N, A[10], d, cnt=1, ans=0;

int main() {
	ri(N); while(N>0) {
		d=N%10; ans=max(ans,d);
		FOR(i,0,d) A[i]+=cnt;
		cnt*=10; N/=10;
	}
	printf("%d\n",ans);
	FOR(i,0,ans) printf("%d ",A[i]); printf("\n");
}
