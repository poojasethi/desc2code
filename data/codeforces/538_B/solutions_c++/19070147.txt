#include<bits/stdc++.h>
using namespace std;
long long i,j,m,n,a[1000005],p,t=1;
int main()
{
    cin>>n;
    while(n)
    {
        p=n%10;
        m=max(m,p);
        for(i=0;i<p;i++)
            a[i]+=t;
        t*=10;
        n/=10;
    }
    cout<<m<<"\n";
    for(i=0;i<m;i++)
        cout<<a[i]<<" ";
}
