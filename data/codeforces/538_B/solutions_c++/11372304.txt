#include <stdio.h>
#define max(a,b) (a>b?a:b)
int a[10],n,p=1;
main() {
	//freopen("in.txt","r",stdin);
	scanf("%d",&n);
	while (n>=p) p*=10;
	while (n) {
		for (int i=0;i<10;i++) if (n/p) a[i]+=p,n-=p;
		p/=10;
	}
	int cou = 0;
	for(int i=0; i<10; i++) if(a[i]) cou++;
	printf("%d\n", cou);
	for (int i=0;i<10;i++) if (a[i]) printf("%d ",a[i]);
}
