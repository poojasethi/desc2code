#include<bits/stdc++.h>
using namespace std;

int main()
{
	int i, j, k, n, r[999];
	cin >> n;
	for (i = 0; n; ++i)
	{
		for (j = 1; j <= n; j*= 10);
		k = 0;
		while(j /= 10)
		{
			if((n / j) % 10)
				k += j;
		}
		r[i] = k;
		n -= k;
	}
	cout << i << "\n";
	while(i--)
		cout << r[i] << " ";
	return 0;
}
