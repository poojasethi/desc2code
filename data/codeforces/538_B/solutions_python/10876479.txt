n = raw_input()
dig = map(int, list(n))
nums = []
while sum(dig) > 0:
    s2 = []
    for i,d in enumerate(dig):
        if d > 0:
            dig[i] -= 1
            s2.append('1')
        else:
            s2.append('0')
    nums.append(''.join(s2[s2.index('1'):]))
print len(nums)
print ' '.join(nums)
