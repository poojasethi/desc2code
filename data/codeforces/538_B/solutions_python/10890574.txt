def process(num):
  result = []
  while num > 0:
    strNum = 0
    for ch in list(str(num)):
      strNum *= 10
      strNum += 0 if ch == '0' else 1
    result.append(strNum)
    num -= strNum
  return result

inp = int(input())
result = process(inp)
print len(result)
print ' '.join(map(str, result))
