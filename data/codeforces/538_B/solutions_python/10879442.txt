n = int(raw_input())

def max_bin(n): #return max quasi-binary number smaller than n
    m = int( '1'*len(str(n)) )
    m = ''.join( [ '1' if int(c)>=1 else '0' for c in str(n) ] )
    return int(m)

res = []
while n!=0:
    m = max_bin(n)
    res.append(m)
    n-=m
