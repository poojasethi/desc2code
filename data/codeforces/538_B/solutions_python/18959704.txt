n = map(int, (list(raw_input())))
res = []
while sum(n) > 0:
    temp = ''
    for i in range(len(n)):
        if n[i] > 0:
            temp += '1'
            n[i] -= 1
        else:
            if len(temp) > 0:
                temp += '0'

    res.append(temp)

print len(res)
print ' '.join(res)
