#include <cstdio>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <queue>

using namespace std;

char s[2005][2005];
int minx,maxx,miny,maxy;
int n,m;
int sum,l;
bool check(int x,int y)
{
	int dqsum=0;
	if (x<=0) return false;
	if (y<=0) return false;
	if (x+l-1>n) return false;
	if (y+l-1>m) return false;
	for (int i=x;i<=x+l-1;++i) if (s[i][y]=='w') dqsum++;
	for (int i=x;i<=x+l-1;++i) if (s[i][y+l-1]=='w') dqsum++;
	for (int i=y;i<=y+l-1;++i) if (s[x][i]=='w') dqsum++;
	for (int i=y;i<=y+l-1;++i) if (s[x+l-1][i]=='w') dqsum++;
	if (s[x][y]=='w') dqsum--;
	if (s[x+l-1][y]=='w') dqsum--;
	if (s[x][y+l-1]=='w') dqsum--;
	if (s[x+l-1][y+l-1]=='w') dqsum--;
	if (l==1)
	{
		if (s[x][y]=='w') dqsum=1;
		else dqsum=0;
	}
	if (dqsum==sum)
	{
		for (int i=x;i<=x+l-1;++i) if (s[i][y]!='w') s[i][y]='+';
		for (int i=x;i<=x+l-1;++i) if (s[i][y+l-1]!='w') s[i][y+l-1]='+';
		for (int i=y;i<=y+l-1;++i) if (s[x][i]!='w') s[x][i]='+';
		for (int i=y;i<=y+l-1;++i) if (s[x+l-1][i]!='w') s[x+l-1][i]='+';
		for (int i=1;i<=n;++i) printf("%s\n",s[i]+1);
		return true;
	}
	return false;
}
int main()
{
	minx=2009;maxx=0;
	miny=2009;maxy=0;
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i) scanf("%s",s[i]+1);
	for (int i=1;i<=n;++i)
		for (int j=1;j<=m;++j)
		{
			if (s[i][j]=='.') continue;
			maxx=max(maxx,i);
			minx=min(minx,i);
			maxy=max(maxy,j);
			miny=min(miny,j);
			sum++;
		}
	l=max(maxx-minx+1,maxy-miny+1);
	if (check(minx,1)) return 0;
	if (check(minx,miny)) return 0;
	if (check(minx,maxy-l+1)) return 0;
	if (check(maxx-l+1,miny)) return 0;
	if (check(maxx-l+1,maxy-l+1)) return 0;
	if (check(maxx-l+1,1)) return 0;
	if (check(1,1)) return 0;
	if (check(1,miny)) return 0;
	if (check(1,maxy-l+1)) return 0;
	printf("-1\n");
	return 0;
}
