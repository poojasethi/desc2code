#include<cstdio>
#include<cmath>
#include<algorithm>
#include<vector>
#include <queue>
#define N 2007
#define MP make_pair
#define PII pair<int,int>
using namespace std;
char s[N][N];
int n,m;
int x[N*N],y[N*N],cnt;
bool check(int a,int b,int d)
{
    if (a<0||b<0||a+d>n||b+d>m) return false;
    for (int i=0;i<cnt;i++)
    {
        if (x[i]==a||x[i]==a+d-1)
        {
            if (y[i]<b||y[i]>=b+d) return false;
        }
        else if (y[i]==b||y[i]==b+d-1)
        {
            if (x[i]<a||x[i]>=a+d) return false;
        }
        else
            return false;
    }
    for (int i=a;i<a+d;i++)
    {
        if (s[i][b]=='.') s[i][b]='+';
        if (s[i][b+d-1]=='.') s[i][b+d-1]='+';
    }
    for (int i=b;i<b+d;i++)
    {
        if (s[a][i]=='.') s[a][i]='+';
        if (s[a+d-1][i]=='.') s[a+d-1][i]='+';
    }
    for (int i=0;i<n;i++)
        puts(s[i]);
    return true;
}
int main()
{
    int minx=N,miny=N,maxx=0,maxy=0,d;
    scanf("%d%d",&n,&m);
    getchar();
    for (int i=0;i<n;i++)
    {
        gets(s[i]);
        for (int j=0;j<m;j++)
            if (s[i][j]=='w')
            {
                minx=min(minx,i);
                maxx=max(maxx,i);
                miny=min(miny,j);
                maxy=max(maxy,j);
                x[cnt]=i;
                y[cnt++]=j;
            }
    }
    d=max(maxx-minx,maxy-miny)+1;
    bool ok=false;
    for (int i=maxx-d+1;i<=minx;i++)
    {
        for (int j=maxy-d+1;j<=miny;j++)
            if (check(i,j,d))
            {
                ok=true;
                break;
            }
        if (ok) break;
    }
    if (!ok) printf("-1\n");
    return 0;
}
