#include<iostream>
#include<stdio.h>
#include<string.h>
#include<vector>
#define INF 1000000000
using namespace std;

struct node{
	int x,y;
};
vector<node> map;

char s[2005];

int x[4],y[4];
bool vv[2005][2005];
int x1,y1,x2,y2;

bool online(int p,int q){
    if((p==x1||p==x2)){
        if(q>=y1&&q<=y2) return true;
    }
    if(q==y1||q==y2){
        if(p>=x1&&p<=x2) return true;
    }
    return false;
}

int main(){
    //freopen("c.in","r",stdin);
	int n,m;
	int i,j;
	int max_x=0,min_x=INF,max_y=0,min_y=INF;
	node t;
	memset(vv,false,sizeof(vv));
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++){
		scanf("%s",s+1);
		for(j=1;j<=m;j++){
			if(s[j]=='w'){
				t.x=i;t.y=j;
				vv[i][j]=true;
				map.push_back(t);
				max_x=max(max_x,i);
				min_x=min(min_x,i);
				max_y=max(max_y,j);
				min_y=min(min_y,j);
			}
		}
	}
	int ok=0;
	int len=max(max_x-min_x,max_y-min_y);
	x[1]=1;y[1]=1;
	x[2]=min_x;y[2]=min_y;
	x[3]=max(1,max_x-len);y[3]=max(1,max_y-len);
	for(i=1;i<=3;i++){
		for(j=1;j<=3;j++){
			x1=x[i];y1=y[j];x2=x[i]+len;y2=y[j]+len;
			//printf("%d %d %d %d\n",x1,x2,y1,y2);
			if(x2>n||y2>m) continue;
			int flag=1;
			for(int k=0;k<map.size();k++){
				int px,py;
				px=map[k].x;py=map[k].y;
				if(!online(px,py)){
					flag=0;
					break;
				}
			}
			if(flag){
				ok=1;
				break;
			}
		}
		if(ok) break;
	}
	if(ok){
	    //printf("%d %d %d %d\n",x1,x2,y1,y2);
		for(i=1;i<=n;i++){
			for(j=1;j<=m;j++){
				if(vv[i][j]){
					printf("w");
				}
				else if(online(i,j)){
					printf("+");
				}
				else{
					printf(".");
				}
			}
			printf("\n");
		}
	}
	else{
		printf("-1\n");
	}
	return 0;
}
