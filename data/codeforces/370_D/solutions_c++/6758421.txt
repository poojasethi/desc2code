#include <bits/stdc++.h>
using namespace std;
#define FREP(b) for(int i=0;i<b;i++)
#define REP(a,b,c) for(int a=b;a<c;a++)
#define asd(x) cout << "Here for " << #x << " :" <<x << endl;
#define asdf(x, y)  cout << "Here FOR " << #x << " :" << x << " " << #y  << " :" << y << endl;
typedef pair<int,int> ii;
typedef long long LL;
string G[2008];
int total = 0,n,m;
bool valid(int x, int y){
	return x >=0 && x < n && y >= 0 && y < m;
}
void go(int sx, int sy, int sz){
	int here = 0;
	for(int i = sx; i < sx + sz; i++){
		for(int j = sy; j < sy + sz; j++){
			if(!valid(i, j)) return;
			if(i == sx || j == sy || i == sx + sz - 1 || j == sy + sz - 1){
				here += (G[i][j] == 'w');
			}
		}
	}
	if(here != total) return;
	for(int i = sx; i < sx + sz; i++){
		for(int j = sy; j < sy + sz; j++){
			if(!valid(i, j)) return;
			if(i == sx || j == sy || i == sx + sz - 1 || j == sy + sz - 1){
				(G[i][j] == 'w') ? 0 : G[i][j] = '+';
			}
		}
	}
	REP(i,0,n) cout << G[i] << endl;
	exit(0);
}
int main(){
	cin >> n >> m;
	int maxi,maxj,minj,mini;
	maxi = maxj = INT_MIN;
	mini = minj = INT_MAX;
	REP(i,0,n){
		cin >> G[i];
		REP(j,0,m){
			if(G[i][j] == 'w'){
				total++;
				maxi = max(maxi, i), mini = min(mini, i);
				maxj = max(maxj, j), minj = min(minj, j);
			}
		}
	}
	int sz = max(maxi - mini + 1, maxj - minj + 1);
	int A[3] = {0, mini, maxi - sz + 1};
	int B[3] = {0, minj, maxj - sz + 1};
	REP(i,0,3) REP(j,0,3) go(A[i], B[j], sz);
	cout << -1 << endl;
    return 0;
}
