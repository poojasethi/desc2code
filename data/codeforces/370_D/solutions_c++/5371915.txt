#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
const int N=2100;
int s[N][N],n,m,i,j,x,y,len,ans,tx,ty;
bool p[N][N];
char str[N][N];
int calc(int x1,int y1,int x2,int y2){
    return s[x2][y2]+s[x1-1][y1-1]-s[x1-1][y2]-s[x2][y1-1];
}
void ff(int x,int y){
     if(str[x][y]=='.')str[x][y]='+';
     }
int main(){
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;i++){
      scanf("%s",str[i]+1);
      for(j=1;j<=m;j++)
        if(str[i][j]=='w')p[i][j]=1;
        else p[i][j]=0;
    }
    memset(s,0,sizeof s);
    x=n+1; y=m+1;
    for(i=1;i<=n;i++)
      for(j=1;j<=m;j++){
                        s[i][j]+=p[i][j]+s[i][j-1]+s[i-1][j]-s[i-1][j-1];
                        if(p[i][j]){
                                    x=min(x,i);
                                    y=min(y,j);
                                    }
                        }
    ans=n*2;
    for(i=1;i<=n;i++)
      for(j=1;j<=m;j++){
        if(i<n && calc(i+1,1,n,m)>0)continue;
        if(j<m && calc(1,j+1,n,m)>0)continue;
        len=max(i-x+1,j-y+1);
        if(len>i || len>j)continue;
        if(len>2 && calc(i-len+2,j-len+2,i-1,j-1)>0)continue;
        ans=min(ans,len);
        if(ans==len){
                     tx=i;
                     ty=j;
                     }
      }
    if(ans==n*2)printf("-1\n");
    else{
        // cout<<tx<<" "<<ty<<endl;
         for(i=1;i<=ans;i++)ff(tx-i+1,ty),ff(tx,ty-i+1),ff(tx-ans+i,ty-ans+1),ff(tx-ans+1,ty-ans+i);
         for(i=1;i<=n;i++){
           for(j=1;j<=m;j++)printf("%c",str[i][j]);printf("\n");
         }
         }
    return 0;
}
