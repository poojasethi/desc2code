#include<string>

for(a=0;a<m;a++){
if(x[i][a]=='w'){s++;
  if(min1>i) min1=i;
  if(min2>a) min2=a;
  if(max1<i) max1=i;
  if(max2<a) max2=a;
  }
}
}
if(s==1){
for(i=0;i<n;i++){
cout<<x[i]<<endl;
}
return 0;}

if(x[i+min1][l+k]!='.') j++;
if(x[min1][l+i]!='.' && i!=0 && i!=k) j++;
if(x[max1][l+i]!='.' && i!=0 && i!=k) j++;
}
if(j==s) {
for(i=0;i<=k;i++){

}}
else{
