typedef long long LL;

int a[2100][2100];
char s[2100][2100];

int calc(int i,int j,int k,int l){
    return i<=k && j<=l?a[k][l]-a[i-1][l]-a[k][j-1]+a[i-1][j-1]:0;
}

int main(){
    int N,M,i,j,k,l;
    cin >> N >> M;
    for (i=1;i<=N;i++){
        scanf("%s",s[i]+1);
        for (j=1;j<=M;j++) a[i][j]=s[i][j]=='w';
    }
    int mx=INT_MAX,my=INT_MAX,Mx=INT_MIN,My=INT_MIN;
    for (i=1;i<=N;i++)
        for (j=1;j<=M;j++){
            if (a[i][j]){
                mx=min(mx,i); Mx=max(Mx,i);
                my=min(my,j); My=max(My,j);          
            }
            a[i][j]+=a[i-1][j]+a[i][j-1]-a[i-1][j-1];

        }   
    int d=max(Mx-mx,My-my);
    for (i=1;(k=i+d)<=N;i++)
        for (j=1;(l=j+d)<=M;j++){
            if (calc(i,j,k,l)-calc(i+1,j+1,k-1,l-1)==a[N][M]){
                mx=i; Mx=i+d; my=j; My=j+d;
                for (i=1;i<=N;i++){
                    for (j=1;j<=M;j++) putchar( ((i==mx || i==Mx) && my<=j && j<=My ||  (j==my || j==My) && mx<=i && i<=Mx) && s[i][j]=='.'?'+':s[i][j]);
                    puts("");
                }
                return 0;
            }
        }
    puts("-1");

}
