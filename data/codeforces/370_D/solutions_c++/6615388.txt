#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
const int N = 2005;
char a[N][N], b[N][N];
int sx, sy;
int n, m, l, cnt;
bool check(int x, int y) {
	if(x + l >= n || y + l >= m) return false;

	int t = cnt;
	for(int i = 0; i < n; i ++) {
		for(int j = 0; j <= m; j ++) {
			b[i][j] = a[i][j];
		}
	}
	for(int j = y; j <= y + l; j ++) {
		if(b[x][j] != 'w') b[x][j] = '+';
		else t--;

		if(b[x + l][j] != 'w') b[x + l][j] = '+';
		else t--;
	}
	for(int i = x + 1; i < x + l; i ++) {
		if(b[i][y] != 'w') b[i][y] = '+';
		else t--;

		if(b[i][y + l] != 'w') b[i][y + l] = '+';
		else t--;
	}
	if(t == 0) {
		for(int i = 0; i < n; i ++) {
			printf("%s\n", b[i]);
		}
		return true;
	} else
		return false;
}
int main() {
	int x1 = -1, x2 = -1, y1 = -1, y2 = -1;
	scanf("%d%d", &n, &m);
	for(int i = 0; i < n; i ++) {
		scanf("%s", &a[i]);
		for(int j = 0; j < m; j ++) {
			if(a[i][j] == 'w') {
				cnt++;
//				if(cnt == 1) sx = i, sy = j;
				if(x1 == -1 || i < x1 ) x1 = i;
				if(y1 == -1 || j < y1 ) y1 = j;
				if(x2 == -1 || i > x2 ) x2 = i;
				if(y2 == -1 || j > y2 ) y2 = j;
			}
		}
	}
//	printf("x = %d, y = %d\n", sx, sy);
	sx = x1, sy = y1;
	if(cnt == 1) {
		for(int i = 0; i < n; i ++) {
			printf("%s\n", a[i]);
		}
		return 0;
	}
	l = max(x2 - x1, y2 - y1);

	for(int i = sx; i >= 0; i --) {
		if(check(i, sy)) {
			return 0;
		}
	}
	for(int j = sy; j >= 0; j --) {
		if(check(sx, j)) {
			return 0;
		}
	}
	puts("-1");

	return 0;
}
