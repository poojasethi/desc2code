using namespace std;

typedef long long LL;
typedef double ld;

const int NUM=2000+10;

int n,m;
char g[NUM][NUM];
int sum[NUM][NUM];

int get(int a,int b,int c,int d)
{
	--a;
	--c;
	return sum[b][d]+sum[a][c]-sum[a][d]-sum[b][c];
}

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
#endif
	int i,j;
	scanf("%d%d",&n,&m);
	int lx=n+1,rx=0;
	int ly=m+1,ry=0;
	REP(i,1,n)
	{
		scanf("%s",g[i]+1);
		REP(j,1,m)
		{
			sum[i][j]=sum[i-1][j]+sum[i][j-1]-sum[i-1][j-1];
			if(g[i][j]=='w')
			{
				lx=min(lx,i);
				rx=max(rx,i);

				ly=min(ly,j);
				ry=max(ry,j);

				++sum[i][j];
			}
		}
	}
	int px=0,py=0,area=max(n,m)+1;
	REP(i,1,lx)
		REP(j,1,ly)
		{
			int d=max(rx-i+1,ry-j+1);
			if(i+d-1<=n && j+d-1<=m && d<area && ( d==1 || get(i+1,i+d-2,j+1,j+d-2)==0 ) )
			{
				px=i;
				py=j;
				area=d;
			}
		}
	if(!px)
	{
		printf("-1\n");
		return 0;
	}
	int d=area;
	REP(i,1,n)
		REP(j,1,m)
			if(i==px || i==px+d-1 || j==py || j==py+d-1)
			{
				if(i>=px && i<=px+d-1 && j>=py && j<=py+d-1 && g[i][j]!='w')
					g[i][j]='+';
			}
	REP(i,1,n)
		printf("%s\n",g[i]+1);
	return 0;
}
