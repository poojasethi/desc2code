#include <iostream>
#include <cstdio>

using namespace std;

char mo[2005][2005];

int n,m;

bool judge(int x,int y,int s)
{
	//cout<<s<<" "<<x<<" "<<y<<endl;
	if(x<0 || x>=m || y<0 || y>=n) return false;
	if(s>n || s>m) return false;
	for(int i=1;i<s-1;++i)
		for(int j=1;j<s-1;++j)
		{
			if(y+i>=n || x+j>=m) return false;
			if(mo[y+i][x+j]=='w') return false;
		}
	return true;
}

void print(int x,int y,int s)
{
//	cout<<y<<" "<<x<<" "<<s<<endl;
	for(int i=0;i<n;++i)
	{
		for(int j=0;j<m;++j)
		{
			if(mo[i][j]=='w') cout<<"w";
			else if(((i==y || i==y+s-1) && (j<=x+s-1 && j>=x)) || ((j==x+s-1 || j==x) && (i<=y+s-1 && i>=y))) {cout<<"+";}
			else cout<<".";
		}
		cout<<endl;
	}
}

int main()
{
	int l=2005,r=0,t=2005,b=0;
	int s;
	cin>>n>>m;
	for(int i=0;i<n;++i)
	{
		bool f=0,ff=0;
		for(int j=0;j<m;++j)
		{
			cin>>mo[i][j];
			if(mo[i][j]=='w') ff=1;
			if(mo[i][j]=='w' && !f && l>j) l=j,f=1;
		}
		if(ff)
		{
			b=i;
			if(t>i) t=i;
		}
		f=0;
		for(int j=m-1;j>=0;--j)
		{
			if(mo[i][j]=='w' && !f && r<j) r=j,f=1;
		}
	}
	s=max(r-l+1,b-t+1);
	int minx=max(0,r-s+1),maxx=l;
	int miny=max(0,b-s+1),maxy=t;
	bool flag=0;
	for(int i=minx;i<=maxx;++i)
		for(int j=miny;j<=maxy;++j)
			if(judge(i,j,s)) {flag=1;print(i,j,s);return 0;}
	if(!flag) puts("-1");
	return 0;
}
