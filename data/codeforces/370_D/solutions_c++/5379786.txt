#include <string>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;
string maze[2222];
int counter[2222][2222];
int main() {
	int n , m;
	cin >> n >> m;
	int min_x = 2222;
	int min_y = 2222;
	int max_x = 0;
	int max_y = 0;
	int total = 0;
	for (int i = 0; i < n; i ++) {
		cin >> maze[i];
		for (int j = 0 ; j < m; j ++) {
			if (maze[i][j] == 'w') {
				min_y = min(min_y, j);
				min_x = min(min_x, i);
				max_y = max(max_y, j);
				max_x = max(max_x, i);
				total ++;
			}
			counter[i + 1][j + 1] = counter[i][j + 1] + counter[i + 1][j] - counter[i][j] + (maze[i][j] == 'w');
		}
	}
	int l = max(max_x - min_x, max_y - min_y);
	for (int i = l; i < n; i ++) {
		for (int j = l; j < m; j ++) {
			if (counter[i + 1][j + 1] - counter[i + 1][j - l] - counter[i - l][j + 1] + counter[i - l][j - l] != total) {
				continue;
			}
			if (l >= 2 && counter[i][j] - counter[i][j + 1 - l] - counter[i + 1 - l][j] + counter[i + 1 - l][j + 1 - l] != 0) {
				continue;
			}
			for (int x = i - l; x <= i; x ++) {
				for (int y = j - l; y <= j; y ++) {
					if (x == i - l || x == i || y == j - l || y == j) {
						if (maze[x][y] == '.') {
							maze[x][y] = '+';
						}
					}
				}
			}
			for (int x = 0; x < n; x ++) {
				cout << maze[x] << endl;
			}
			return 0;
		}
	}
	cout << -1 << endl;
	return 0;
}
