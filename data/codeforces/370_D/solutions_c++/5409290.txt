#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 2010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,li,lj,ri,rj,f[N][N],h[N][N],g[N][N],s,A;
char p[N][N];
inline bool judge() {
    int B=s-(h[i][j+A]-h[i][j])-(h[i+A][j+A-1]-h[i+A][j-1])-
        (g[i+A-1][j]-g[i-1][j])-(g[i+A][j+A]-g[i][j+A]);
    return B==0;
}
inline void tu(int x,int y) {
    if (p[x][y]!='w') p[x][y]='+';
}
inline void print(int x,int y,int xx,int yy) {
    int i;
    For(i,x,xx) tu(i,y),tu(i,yy);
    For(i,y,yy) tu(x,i),tu(xx,i);
    For(i,1,n) printf("%s\n",p[i]+1);
    //for(;;);
    exit(0);
}
int main() {
    scanf("%d%d",&n,&m);
    li=n,lj=m;
    For(i,1,n) {
        scanf("%s",p[i]+1);
        For(j,1,m) if (p[i][j]=='w') {
            li=min(li,i); lj=min(lj,j);
            ri=max(ri,i); rj=max(rj,j);
            f[i][j]=1; s++;
        }
        For(j,1,m) h[i][j]=h[i][j-1]+f[i][j],g[i][j]=g[i-1][j]+f[i][j];
    }
    A=max(ri-li,rj-lj);
    if (!A) print(li,lj,li,lj);
    For(i,1,n-A)For(j,1,m-A) if (judge()) print(i,j,i+A,j+A);
    printf("-1\n");
    //for(;;);
    return 0;
}
