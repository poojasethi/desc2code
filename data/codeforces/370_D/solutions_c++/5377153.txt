#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

const int N=2100;
char Map[N][N];

bool Solve(int minx,int maxx,int miny,int maxy)
{
    int i,j;
    for(i=minx+1;i<maxx;i++)
    {
        for(j=miny+1;j<maxy;j++)
        {
            if(Map[i][j]=='w')
            {
                return false;
            }
        }
    }
    return true;
}

int x[4],y[4];

int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    
    int i,j;
    for(i=0;i<n;i++)scanf("%s",Map[i]);
    
    int minx=n,maxx=0,miny=m,maxy=0;
    for(i=0;i<n;i++)
    {
        for(j=0;j<m;j++)
        {
            if(Map[i][j]=='w')
            {
                minx=min(minx,i);
                miny=min(miny,j);
                maxx=max(maxx,i);
                maxy=max(maxy,j);
            }
        }
    }
    
    bool Judge=false;
    int len=max(maxx-minx,maxy-miny);
    x[0]=minx;x[1]=maxx-len;x[2]=0;
    y[0]=miny;y[1]=maxy-len;y[2]=0;
    for(i=0;i<3;i++)
    {
    	for(j=0;j<3;j++)
    	{
    		if(x[i]<0||x[i]+len>=n||y[j]<0||y[j]+len>=m)continue;
    		if(x[i]<=minx&&x[i]+len>=maxx&&y[j]<=miny&&y[j]+len>=maxy&&Solve(x[i],x[i]+len,y[j],y[j]+len))
	    	{
	    		Judge=true;
	    		minx=x[i];maxx=x[i]+len;
	    		miny=y[j];maxy=y[j]+len;
	    		break;
	    	}
	    }
	    if(Judge)break;
    }
    if(Judge)
    {
    	for(i=minx;i<=maxx;i++)
    	{
	    	if(Map[i][miny]=='.')Map[i][miny]='+';
	    	if(Map[i][maxy]=='.')Map[i][maxy]='+';
	    }
	    for(i=miny;i<=maxy;i++)
	    {
    		if(Map[minx][i]=='.')Map[minx][i]='+';
    		if(Map[maxx][i]=='.')Map[maxx][i]='+';
    	}
	    for(i=0;i<n;i++)printf("%s\n",Map[i]);
    }
    else printf("-1\n");
    
    return 0;
}
