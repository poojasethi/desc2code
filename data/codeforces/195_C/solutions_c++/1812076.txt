#include <iostream>
#include <stack>
#include <sstream>

using namespace std;

int n;

int main()
{
	ios::sync_with_stdio(0);
	cin >> n >> ws;
	stack<int> st;
	int head = -1;
	string temp;
	for (int i = 0 ; i < n ; i++)
	{
		string line;
		getline(cin, line);
		for (int j = 0 ; j < line.size() ; j++)
			if (line[j] == '(' || line[j] == ')' || line[j] == ',')
				line[j] = ' '; 
		istringstream fin(line);
		string command;
		fin >> command;
		if (command == "try")
			st.push(i);
		else	if (command == "throw")
		{
			fin >> temp;
			head = i;
		}
		else	if (command == "catch")
		{
			string temp1;
			fin >> temp1;
			if (temp1 == temp && st.top() < head)
			{
				string message;
				getline(fin, message);
				message.erase(0, message.find('"')+1);
				message.erase(message.find('"'), message.size());
				cout << message << endl;
				return 0;
			}
			st.pop();
		}
	}
	cout << "Unhandled Exception" << endl;
	return 0;
}
