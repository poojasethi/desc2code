#include<stdio.h>
#include<string.h>
char s[100010][100];
int stack[100010]={0};
int top=-1;
int main()
{
	int n,i,j,k,loc=-1,x,y,ans=-1,len=-1;
	char pao[100];
	scanf("%d",&n);
	getchar();
	for(i=0;i<n;i++)
	{
		gets(s[i]);
	}
	for(i=0;i<n;i++)
	{
		for(j=0;s[i][j]==' ';j++);
		if(s[i][j]=='t'&&s[i][j+1]=='r')
			stack[++top]=i;
		else if(s[i][j]=='t'&&s[i][j+1]=='h')
		{
			loc=i;
			for(j=0;s[i][j]!='(';j++);
			j++;
			for(;s[i][j]==' ';j++);
			for(k=0;s[i][j+k]!=')'&&s[i][j+k]!=' ';k++)
				pao[k]=s[i][j+k];
			len=k;
			pao[k]='\0';
		}
		else if(s[i][j]=='c')
		{
			x=stack[top--];
			y=i;
			if(top<-1)
				break;
			for(j=0;s[i][j]!='(';j++);
			j++;
			for(;s[i][j]==' ';j++);
			for(k=0;s[i][j+k]!=','&&s[i][j+k]!=',';k++)
			{
				if(s[i][j+k]!=pao[k])
					break;
			}
			//printf("%d %d\n",len,k);
			if(k==len&&len>0)
			{
			
			if(x<loc&&loc<y)
			{
				ans=y;
				break;
			}
			}
		}
	}
	if(ans==-1)
	{
		printf("Unhandled Exception\n");
	}
	else
	{
		//puts(s[ans]);
		for(j=0;s[ans][j]!='"';j++);
		j++;
		//printf("%d %c\n",j,s[ans][j]);
		for(;s[ans][j]!='"';j++)
		{
			printf("%c",s[ans][j]);
		}
		printf("\n");
	}
	return 0;
}
