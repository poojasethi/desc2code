#include<iostream>
#include<cstdio>
#include<cstring>
#define SIZE 1024
// learn from others, cool scanf parsing style
using namespace std;
char str[SIZE];
char th[SIZE];
char cat[SIZE];
char msg[SIZE];
char sol[SIZE];
bool ans ;
int solve()
{
    int ret = false;
    while(scanf("%*[^a-z]%[a-z]",str) && str[1]!='a')
    {
        if(str[1]=='r') // try
        {
            ret|=solve();
        }
        else if(str[1]=='h') // throw
        {
            scanf("%*[^A-Za-z]%[A-Za-z]",th);
            ret= true;
        }
    }
    scanf("%*[^A-Za-z]%[A-Za-z]",cat);
    scanf("%*[^\"]\"%[^\"]\"",msg);
    if(ret && !ans && !strcmp(cat,th))
    {
       strcpy(sol,msg);
       ans = true;
    }
    return ret;
}
int main()
{
    int n;
    ans =false;
    scanf("%d",&n);
    while(scanf("%*[^A-Za-z]%[A-Za-z]",str)==1) // must be try
    {
        solve();
    }
    if(ans)cout<<sol<<endl;
    else cout<<"Unhandled Exception"<<endl;

    return 0;
}
