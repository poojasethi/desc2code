#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;

bool ans;
char str[1000], output[1000];
char s1[1000], s2[1000], s3[1000];

bool solve()
{
    bool ret = 0;
    while(scanf("%*[^a-z]%[a-z]", str), str[1] != 'a')
    {
        if(str[1] == 'r') ret |= solve();
        if(str[1] == 'h')
        {
            scanf("%*[^A-Za-z]%[A-Za-z]", s1);
            ret = true;
        }
    }
    scanf("%*[^A-Za-z]%[A-Za-z]", s2);
    scanf("%*[^\"]\"%[^\"]", s3);
    if(ret && !ans && strcmp(s1, s2) == 0)
    {
        ans = true;
        strcpy(output, s3);
    }
    return ret;
}

int main()
{
    int n;
    scanf("%d", &n);
    while(~scanf("%*[^a-z]%[a-z]", str) && str[1] == 'r') solve();
    if(ans) puts(output);
    else puts("Unhandled Exception");

    return 0;
}
