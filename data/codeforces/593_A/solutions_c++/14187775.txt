#include <iostream>
using namespace::std;

string s[101];

bool isAllThese(string s, char a, char b) {
	for (int i = 0; i < s.size(); i ++)
		if (s[i] != a && s[i] != b)
			return 0;
	return 1;
}

int main() {
	int n, maxx = 0; cin >> n;
	for (int i = 0; i < n; i ++)
		cin >> s[i];

	for (int i = 97; i < 123; i ++) {
		for (int j = 97; j < 123; j ++) {
			char c1 = i, c2 = j;
			int maxtmp = 0;
			for (int k = 0; k < n; k ++)
				if (isAllThese(s[k], c1, c2))
					maxtmp += s[k].size();
			maxx = max(maxx, maxtmp);
		}
	}

	cout << maxx << endl;

	return 0;
}
