#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
#define Max 1010
char s[110][Max];
int main()
{
	int n;
	while(scanf("%d",&n)!=EOF){
		int sum=0;
		for(int i=0;i<n;i++){
			scanf("%s",&s[i]);
		}
		int res=0;
		for(int i=0;i<26;i++)
			for(int j=i+1;j<26;j++){
				sum = 0;
				for(int k=0;k<n;k++){
					int t=0;
					while(s[k][t]){
						if(s[k][t]!=i+'a'&&s[k][t]!=j+'a'){
							t=0;
							break;
						}
						t++;
					}	
					sum+=t;	
				}
				res=max(sum,res);
			}
		printf("%d\n",res);
	}
}
