#include <iostream>

using namespace std;

int n, x;
string s[200];

int main(){
	cin >> n;
	
	for(int i = 0; i < n; i++)
		cin >> s[i];
	
	for(int i = 'a'; i <= 'z'; i++){
		for(int j = 'a'; j <= 'z'; j++){
			int x1 = 0;
			for(int k = 0; k < n; k++){
				bool Ok = true;
				for(int l = 0; l < s[k].length(); l++){
					if(s[k][l] != i && s[k][l] != j){
						Ok = false;
						break;
					}
				}
				if(Ok){
					x1 += s[k].size();
				}
			}
			x = (x > x1 ? x : x1);
		}
	}
	cout << x;
	
	return 0;
}
