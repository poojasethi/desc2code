#include<bits/stdc++.h>
using namespace std;

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(0);
  cout.tie(0);

  map<set<char>, int> mp;
  int n; cin>>n;
  while(n--) {
    string x; cin>>x;
    set<char> st(x.begin(), x.end());
    if(st.size()<=2) mp[st]+=x.size();
  }

  int ret=0;
  for (char a='a'; a<='z'; a++)
    for (char b=a+1; b<='z'; b++) {
      set<char> x,y,z;
      x.insert(a); y.insert(b);
      z.insert(a); z.insert(b);
      ret=max(ret,mp[x]+mp[y]+mp[z]);
    }
  cout << ret << endl;
  return 0;
}
