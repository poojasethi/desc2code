#include<stdio.h>
#include<iostream>
#include<vector>
#include<string>
using namespace std;
int main()
{
	int n;
	cin>>n;
	vector<string> arr(n);
	for(int i=0;i<n;i++)
		cin>>arr[i];
	int ans = 0;
	for(char a = 'a';a<='z';a++)
	{
		for(char b='a';b<='z';b++) {
			int cur = 0;
			for(int i=0;i<arr.size();i++) {
				string s = arr[i];
				bool flag=1;
				for(int j=0;j<s.size();j++) {
					char c = s[j];
					if(c!=a && c!=b)
						flag=0;
				}
				if(flag)
					cur = cur+s.size();
			}
			ans = max(ans, cur);
		}
	}
	cout<<ans<<endl;
	return 0;
}
