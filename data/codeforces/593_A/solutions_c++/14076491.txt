#include<bits/stdc++.h>
int main()
{
	int hash[102][30]={0};
	int n,i;
	scanf("%d",&n);
	char str[1003];
	for(i=0;i<n;i++)
	{
		scanf("%s",str);
		int l=strlen(str);
		hash[i][28]=l;
		for(int j=0;j<l;j++)
		{
			if(hash[i][str[j]-'a']==0)
				hash[i][29]++;
				hash[i][str[j]-'a']++;
		}
	}
	int max=0;
	int j,k;
	for(i=0;i<27;i++)
	{
		for(j=i+1;j<27;j++)
		{
			int s=0;
			for(k=0;k<n;k++)
			{
				if(hash[k][i]+hash[k][j]==hash[k][28]&&hash[k][29]<3)
				{
					s+=hash[k][28];
				}
			}
			if(max<s)
				max=s;
		}
	}
	printf("%d\n",max);
	return 0;
}
