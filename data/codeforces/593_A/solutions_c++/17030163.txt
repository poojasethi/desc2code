#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;
char a[105][1004];
int n;
int hh(char a[],char x,char y)
{
    int l=strlen(a);
    for(int i=0;i<l;i++)
    {
        if(a[i]!=x&&a[i]!=y)
            return 0;
    }
    return l;
}
int main()
{
    int Max=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
        scanf("%s",a[i]);
    for(char x='a';x<='z';x++)
        for(char y=x;y<='z';y++)
    {
        int s=0;
        for(int i=0;i<n;i++)
            s+=hh(a[i],x,y);
        Max=max(s,Max);
    }
    printf("%d\n",Max);
}
