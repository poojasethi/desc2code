#include <bits/stdc++.h>
using namespace std;

long long n,i,j,u,y,s,maxl;

string t[111];

int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	cin>>n;
	for(i=0;i<n;i++)
		cin>>t[i];
	for(i=0;i<26;i++)
		for(j=i+1;j<26;j++)
		{
			s=0;
			for(u=0;u<n;u++)
				{
					long q=0;
					for(y=0;y<t[u].size();y++)
						if (t[u][y]!=('a'+i)&&t[u][y]!=('a'+j))
							q=1;
					if (q==0)
					{
						s=s+t[u].size();
					}
				}
			if (s>maxl)maxl=s;
		}
		cout<<maxl;
}
