#include <stdio.h>
#include<string.h>
int main()
{
	int n;
	while(~scanf("%d",&n))
	{
		char c[110][1001];
		for(int i=0;i<n;i++)
		scanf("%s",&c[i]);
		int sum=0,count=0,max=0;
		for(int i='a';i<='z';i++)
		for(int j='a';j<='z';j++)
		{
			for(int a=0;a<n;a++)
			{
				int s=strlen(c[a]);	
				for(int b=0;b<s;b++)
				{
				  if(c[a][b]==i||c[a][b]==j)
				  sum++;
				  else
				  {sum=0;break;}	
				}
				count+=sum;
				sum=0;
			}
			if(max<count)
			max=count;
			count=0;
		}
		printf("%d\n",max);
	}
	return 0;
}
