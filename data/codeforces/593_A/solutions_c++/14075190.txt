#include <bits/stdc++.h>
using namespace std;

const int maxN=1e5;

string s[maxN];
int n;

int cnt(int k,char c1,char c2) {
    for(int i=0;i<s[k].length();i++)
        if (s[k][i]!=c1 && s[k][i]!=c2) return 0;
    return s[k].length();
}

int main() {
    scanf("%i\n",&n);
    for(int i=0;i<n;i++) getline(cin,s[i]);
    int res=0;
    for(char c1='a';c1<='z';c1++) for(int c2='a';c2<='z';c2++) {
        int ans=0;
        for(int j=0;j<n;j++) ans+=cnt(j,c1,c2);
        res=max(res,ans);
    }
    cout<<res;
    return 0;
}
