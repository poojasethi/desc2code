#include <iostream>
#include <stdio.h>

using namespace std;

int n,mx=0;
string s[107];

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif
	
	cin>>n;
	for(int i=0;i<n;i++) cin>>s[i];
	for(int i=0;i<26;i++)
	{
		for(int j=0;j<26;j++)
		{
			int r = 0;
			for(int k=0;k<n;k++)
			{
				bool ok = true;
				for(int l=0;s[k][l];l++) if(s[k][l] != ('a' + i) and s[k][l] != ('a'+j)) ok = false;
				if(ok) r += s[k].size();
			}
			mx = max(mx,r);
		}
	}
	cout<<mx<<endl;
	return 0;
}
