#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
int n,tem,ans;
string s[105];
int len[105];
int main(){
    int i,j,k,l;
    cin>>n;
    for(i=1;i<=n;i++)
    cin>>s[i],len[i]=s[i].length();
    for(i=0+'a';i<26+'a';i++)
        for(j=i;j<26+'a';j++)
        {
            tem=0;
            for(k=1;k<=n;k++)
            {
                for(l=0;l<len[k];l++)
                if(s[k][l]!=i && s[k][l]!=j) break;
                if(l==len[k]) tem+=len[k];
            }
            if(tem>ans) ans=tem;
        }
    cout<<ans<<endl;
}
