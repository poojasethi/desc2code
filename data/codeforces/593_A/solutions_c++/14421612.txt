#include<cstdio>
#include<algorithm>
#include<cstring>
#include<climits>
using namespace std;
#define N 110
#define M 1010
char s[N][M];
int n;
#define For(i,a,b) for(int i=(a);i<=(b);i++)
int main(){
    scanf("%d",&n);For(i,1,n)scanf("%s",s[i]+1);
    int ans=0;
    For(i,0,25)For(j,0,25){
        int tmp=0;
        For(l,1,n){
            int flag=1,len=strlen(s[l]+1);
            For(p,1,len)if(s[l][p]!=(i+'a')&&s[l][p]!=(j+'a')){
                flag=0;break;
            }if(flag)tmp+=len;
        }
        ans=max(ans,tmp);
    }printf("%d\n",ans);
}
