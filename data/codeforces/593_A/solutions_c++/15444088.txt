#include <bits/stdc++.h>
using namespace std;

char s[100][1001];

int main()
{
    int n, i, k, h, t, j, ret = 0;
    bool ok;
    scanf("%d\n", &n);
    for(i = 0; i < n; i++)
        gets(s[i]);
    for(i = 0; i < 26; i++)
    for(k = i + 1; k < 26; k++)
    {
        h = 0;
        for(t = 0; t < n; t++)
        {
            ok = true;
            for(j = 0; s[t][j]; j++)
                ok &= (s[t][j] - 'a' == i || s[t][j] - 'a' == k);
            h += j*ok;
        }
        ret = max(ret, h);
    }
    printf("%d", ret);
    return 0;
}
