#include<bits/stdc++.h>
//#define maxi 100005
#define f first
#define s second
#define mp make_pair
#define pb push_back
#define ll long long
#define si(n) scanf("%d",&n)
#define sll(n) scanf("%lld",&n)
using namespace std;
int main(){
	int maxi=0,n,a[105][30]={0},p[105];
	char s[1005];
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>s;
		p[i]=strlen(s);
		for(int j=0;j<p[i];j++)
			a[i][s[j]-'a']++;
	}
	for(int j=0;j<26;j++)
		for(int k=j+1;k<26;k++){
			int ans=0;
			for(int i=0;i<n;i++)
				if(a[i][j]+a[i][k]!=p[i]);
				else ans+=p[i];
			maxi=max(ans,maxi);
		}
	cout<<maxi<<endl;
	return 0;
}
