#include<bits/stdc++.h>
using namespace std;
int n,mx=0;
char s[105][1005];
int main(){
    scanf("%d",&n);
    for (int i=1; i<=n; i++) scanf("%s",s[i]);
    for (int j=1; j<26; j++)
        for (char k=j+1; k<=26; k++){
            int ans=0;
            for (int i=1; i<=n; i++){
                bool b=1;
                for (int z=0; z<strlen(s[i]); z++) if ((s[i][z]!=j+'a'-1) && (s[i][z]!=k+'a'-1)) b=0;
                if (b) ans+=strlen(s[i]);
            }
            if (ans>mx) mx=ans;
        }
    printf("%d",mx);
}
