#include<iostream>
#include<string>
using namespace std;
string s[100000];
int i,n,j,ans,p,mx,t,l;
main()
{
      cin>>n;
      for(i=1;i<=n;i++)
      cin>>s[i];
      for(i=1;i<26;i++)
      for(j=i+1;j<=26;j++)
      {
                          ans=0;
                          for(t=1;t<=n;t++)
  {                        p=1;   
  for(l=0;l<s[t].size();l++)
  if(s[t][l]!=i+'a'-1 && s[t][l]!=j+'a'-1) p=0;

if(p==1) ans+=s[t].size();
}
mx=max(ans,mx);
}
cout<<mx;
//system("pause");
}
