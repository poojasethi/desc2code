#include <iostream>
#include <string>

using namespace std;
string a[110];
int ans,n;


bool check(string s,char a,char b)
{
    for (int i=0;i<s.length();++i)
    {
        if (s[i]!=a && s[i]!=b) return false;
    }
    return true;
}

int main()
{
    cin>>n;
    for (int i=1;i<=n;++i) cin>>a[i];
    for (int i=0;i<26;++i)
        for (int j=i;j<26;++j)
        {
            int tmp=0;
            for (int k=1;k<=n;++k)
                if (check(a[k],i+'a',j+'a')) tmp+=a[k].length();
            ans=max(ans,tmp);
        }
    cout<<ans;
    return 0;
}
