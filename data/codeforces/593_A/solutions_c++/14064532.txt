#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

const int N = 200;

string S[N];

int main(){
	int n;
	int i,j,k,l;
	int p,wynik,a_wynik,suma;
	cin >> n;
	suma = 0;
	for(i=0; i<n; i++){
		cin >> S[i];
		suma += S[i].size();
	}
	wynik = 0;
	for(i=0; i<28; i++){
		for(j=0; j<28; j++){
			a_wynik = 0;
			for(k=0; k<n; k++){
				for(l=0; l<S[k].size(); l++){
					p = (int)S[k][l]-(int)'a';
					if( p!=i && p!=j ){
						a_wynik += S[k].size();
						break;
					}
				}
			}
			wynik = max(wynik,suma-a_wynik);
		}
	}
	cout << wynik << "\n";
	return 0;
}
