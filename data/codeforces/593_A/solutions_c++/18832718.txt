#include <bits/stdc++.h>
using namespace std;
int main(){
	string cad;
	int n,sum=0,Max=0;
	cin>>n;
	string M[n];
	for (int i = 0; i < n; ++i)
		cin>>M[i];
	for (char a = 'a'; a <= 'z'; ++a){
		for (char b = 'a'; b <= 'z'; ++b){
			sum=0;
			for (int i = 0; i < n; ++i){
				string aux=M[i];
				sum+=aux.size();
				for (int j = 0; j < aux.size(); ++j){
					if(aux[j]!=a&&aux[j]!=b){
						sum-=aux.size();
						break;
					}
				}
			}
			Max=max(sum,Max);
		}
	}
	cout<<Max<<endl;
	//system("pause");
	return 0;
}
