#include <iostream>
#include <string>
using namespace std;

int main() {
    int n;
    string s[109];
    cin>>n;
    for(int i=0;i<n;i++)
    cin>>s[i];
    int res=0;
    for(int i=0;i<26;i++)
    {
    for(int j=i+1;j<26;j++)
    {
    int sum=0;
    for(int k=0;k<n;k++)
    {
    bool flag=true;
    for(int l=0;l<s[k].length();l++)
    if((s[k][l]-'a')!=i && (s[k][l]-'a')!=j)
    flag=false;
    if(flag)
    sum=sum+s[k].length();
    }
    res=max(res,sum);
    }
    }
    cout<<res<<"\n";
    return 0;
}
