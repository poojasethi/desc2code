#include<bits/stdc++.h>

using namespace std;

char C[105][1005];

bool comp(int k,int l,int m)
{
 int t=strlen(C[k]);
 for(int i=0;i<t;i++)
  if(C[k][i]-'a'!=l && C[k][i]-'a'!=m)
   return 0;
 return 1;
}

int main()
{
 int N;
 scanf("%d",&N);


 for(int i=1;i<=N;i++)
  scanf("%s",C[i]);

 int sol=0;
 for(int i='a'-'a';i<='z'-'a';i++)
  for(int j='a'-'a';j<='z'-'a';j++)
  {
   int sol1=0;
   for(int k=1;k<=N;k++)
   {
    if(comp(k,i,j))
     sol1+=strlen(C[k]);
   }
   sol=max(sol,sol1);
  }

  printf("%d\n",sol);
    return 0;
}
