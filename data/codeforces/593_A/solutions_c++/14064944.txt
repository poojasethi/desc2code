#include <bits/stdc++.h>
using namespace std;

#define int long long

string s[101];

main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)	
		cin >> s[i];
	
	int ans = 0;
	
	for (int i = 0; i < 26; i++)	
		for (int j = 0; j < 26; j++)
			if (i - j)
			{
				int sum = 0;
				for (int l = 0; l < n; l++)
				{
					bool can = true;
					for (int k = 0; k < s[l].size(); k++)
						if (s[l][k] - 'a' != i && s[l][k] - 'a' != j)
							can = false;
					if (can)
						sum += s[l].size();
				}
				ans = max(ans, sum);
			}
	cout << ans << endl;
}
