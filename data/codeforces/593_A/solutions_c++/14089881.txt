#include <iostream>
#include <string>

using namespace std;

int n;
string s[120];

int main()
{
	int ans = 0;
	cin >> n;
	for(int i = 0; i < n; i++)
		cin >> s[i];
	for(int i = 0; i <= ('z'-'a'); i++)
		for(int j = 0; j <= ('z'-'a'); j++)
		{
			int cnt = 0;
			for(int k = 0; k < n; k++)
			{
				int t = 0;
				for(int l = 0; l < s[k].length(); l++)
					if(s[k][l] != (char)(i+'a') && s[k][l] != (char)(j+'a'))
					{
						t = 0;
						break;
					}	 
					else
						t++;
				cnt += t;
			}
			ans = max(cnt, ans);
		}
	cout << ans;
	return 0;
}
