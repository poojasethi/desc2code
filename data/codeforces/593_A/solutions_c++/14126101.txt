#include <iostream>
#include <algorithm>
using namespace std;
#define N 111
string s[N],b,v;int p[N],n,i,j,m,o,z;char c;pair<int,string>a[N];

for(i=0;i<n;i++){for(j=0;j<n;j++)if(s[i].find(s[j])!=-1)m+=p[j];
if(s[i].size()<2)a[o++]=make_pair(m,s[i]);if(m>z)z=m;m=0;}sort(a,a+n);
