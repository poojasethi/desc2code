#include<stdio.h>
#include<string.h>
char str[110][1100];
int len[110];
int n,m;
char c1,c2;
bool doo(int i){
	for(int j=0;j<len[i];j++){
		if(str[i][j]!=c1&&str[i][j]!=c2)return false;
	}
	return true;
	
}
int main(){
	int i,j,k;
	while(~scanf("%d",&n)){
		for(i=0;i<n;i++){
			scanf("%s",&str[i]);
			len[i]=strlen(str[i]);
		}
		int res=0,t;
		for(i=0;i<26;i++){
			for(j=i+1;j<26;j++){
				t=0;
				c1='a'+i;
				c2='a'+j;
				for(k=0;k<n;k++){
					if(doo(k))t+=len[k];
				}
				if(t>res)res=t;
			}
		}
		printf("%d\n",res);
		
	}
	return 0;
}
