#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	string s[105];

	cin >> n;
	for(int i = 0; i < n; i++)
		cin >> s[i];
	int res = 0;
	for(int i = 0; i < 26; i++)
		for(int j = i; j < 26; j++)
		{
			int temp = 0;
			for(int k = 0; k < n; k++)
			{
				bool flag = true;
				char a = 'a' + i, b = 'a' + j;
				for(int l = 0; l < s[k].length(); l++)
					if(s[k][l] != a && s[k][l] != b)
						flag = false;
				if(flag)
					temp += s[k].length();
			}
			res = max(res, temp);
		}
	cout << res << endl;
	return 0;
}
