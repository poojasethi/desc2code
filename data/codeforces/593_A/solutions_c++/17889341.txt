#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	ios::sync_with_stdio(0);
	int n;
	cin>>n;
	vector<string>s;
	string str;
	for(int i=0;i<n;i++)
	{
		cin>>str;
		s.push_back(str);
	}
	ll len=0,c1=0,c2=0;
	for(int i='a';i<='z';i++)
	{
		for(int j='a';j<='z';j++)
		{
			for(int k=0;k<n;k++)
			{
				for(int d=0;s[k][d];d++)
				{
					if(s[k][d]!=i&&s[k][d]!=j)
					{
						c1=0;
						break;
					}
					c1++;
				}
				c2+=c1;
				c1=0;
			}
			len=max(len,c2);
			c2=0;
		}
	}
	cout<<len;
	return 0;
}
