#include <bits/stdc++.h>
using namespace std;

int n;
string a[1005];
int main() {

	ios::sync_with_stdio(0);
	cin >> n;
	for(int i=0;i<n; i++) { cin >> a[i]; }


	int ans = -1;
	for(int i=0; i<26; i++) {
		for(int j=i+1; j<26; j++) {


			int cur = 0;
			for(int k=0; k<n; k++) {
				bool ok = 1;
				for(int l=0; l<(int)a[k].size(); l++) {
					if ((int)a[k][l] != i+'a' && (int)a[k][l] != j+'a') { ok = 0; break; }
				}

				if (ok) cur+=a[k].size();

			}

			ans = max(ans, cur);

		}
	}


	cout << ans << "\n";

	return 0; 
}
