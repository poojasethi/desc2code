#include <bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	string s[n];
	int maxi = 0;
	for (int i = 0; i < n; i++) cin>>s[i];
	for (int i = 0; i < 26; i++) {
		for (int j = i + 1; j < 26; j++) {
			int ans = 0;
			for (int k = 0; k < n; k++) {
				bool ok = true;
				for (int w = 0; w < (int)s[k].length(); w++) {
					if(s[k][w] != char(i + 'a') && s[k][w] != char(j + 'a')) ok = false;
				}
				if(ok) ans += (int)s[k].length();
			}
			maxi = max(maxi, ans);
		}
	}
	cout<<maxi<<endl;
}
