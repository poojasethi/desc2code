#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n, i, j, maxlen, m, k, len, Count;
	char s[101][1001], char1, char2;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>s[i];
	}
	maxlen=0;
	Count=0;
	for(i=0;i<26;i++)
	{
		char1='a'+i;
		for(j=i+1;j<26;j++)
		{
			char2='a'+j;
			Count=0;
			for(m=0;m<n;m++)
			{
				
				len=strlen(s[m]);
				for(k=0;k<len;k++)
				{
					if(s[m][k]==char1 || s[m][k]==char2)
						continue;
					else
						break;
				}
				if(k==len)
					Count+=len;
			}
			if(maxlen<Count)
				maxlen=Count;
		}
		
	}
	cout<<maxlen<<endl;
	return 0;
}
