#include <iostream>
#include <string>

using namespace std;

string s[200];
int n;

int main()
{
    cin >> n;
    for (int i=1;i<=n;i++)
        cin >> s[i];
    int res=0;
    for (char i='a';i<='z';i++)
        for (char j=i+1;j<='z';j++)
    {
        int tmp=0;
        for (int o=1;o<=n;o++)
        {
            bool kt=true;
            for (int u=0;u<s[o].length();u++)
                if (s[o][u]!=i and s[o][u]!=j) kt=false;
            if (kt) tmp+=s[o].size();
        }
        res=max(res,tmp);
    }
    cout << res;
}
