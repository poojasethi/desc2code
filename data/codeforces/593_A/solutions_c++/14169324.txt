#include<cstdio>
#include<cstring>
#define N 110
int n,i,j,k,l,len,tmp,ans;
char s[N][N];
int main()
{
    scanf("%d",&n);
    for (i=1;i<=n;i++)
        scanf("%s",&s[i]);
    for (i=97;i<=122;i++)
    for (j=i+1;j<=122;j++)
    {
        tmp=0;
        for (k=1;k<=n;k++)
        {
            len=strlen(s[k]);
            for (l=0;l<len;l++)
            if ((s[k][l]!=j)&&(s[k][l]!=i)) break;
            if (l==len) tmp+=len;
        }
        if (tmp>ans) ans=tmp;
    }
    printf("%d\n",ans);
}
