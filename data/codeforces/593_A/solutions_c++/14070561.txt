#include <bits/stdc++.h>

using namespace std;

string s[111];
int n;

long getLen(char x, char y) {
	long sum = 0;
	for(int i = 0; i < n; i++) {
		bool ok = true;
		for (int j = 0; j < s[i].length(); j++) {
			if (s[i][j] != x && s[i][j] != y) {
				ok = false;
				break;
			}
		}
		if (ok) sum += s[i].length();
	}
	return sum;
}

int main() {
	long ans = 0;
	cin >> n;
	for(int i = 0; i < n; i++) cin >> s[i];
	for(char x = 'a'; x <= 'z'; x++) {
		for(char y = x + 1; y <= 'z'; y++) {
			ans = max(ans, getLen(x, y));
		}
	}
	cout << ans;
	return 0;
}
