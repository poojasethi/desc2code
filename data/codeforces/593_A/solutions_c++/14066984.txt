#include<bits/stdc++.h>
using namespace std;
int n,uz[10000],res,k=-1e9,cnt;
char ar[1000][10000];
int main(){
	scanf("%d",&n);
	for(int i=1 ; i<=n ; i++){
		scanf(" %s",ar[i]+1);
		uz[i]=strlen(ar[i]+1);
	}
	for(int i=1 ; i<26 ; i++){
		for(int j=1 ; j<=26 ; j++){
			for(int q=1 ;q<=n ; q++){
				for(int r=1 ;r<=uz[q] ; r++){
					if(ar[q][r]!=i+'a'-1 && ar[q][r]!=j+'a'-1){
						cnt=1;
						break;
					}
				}
				if(!cnt){
					res+=uz[q];
				}
				cnt=0;
			}
			if(res>k){
				k=res;
			}
			res=0;
		}
	}
	printf("%d\n",k);
	return 0;
}
