#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,k,l,mx;
    string s[1000];
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>s[i];
    }
    mx=0;
    for(int i=0;i<26;i++){
        for(int j=0;j<26;j++){
            int c=0;
            for(k=0;k<n;k++){
                int flag=1;
                for(l=0;s[k][l];l++)
                    if(s[k][l]!=i+'a' && s[k][l]!=j+'a') flag=0;
                    c+=flag*s[k].size();
            }
            mx=max(mx,c);
        }
    }
    cout<<mx<<endl;
}
