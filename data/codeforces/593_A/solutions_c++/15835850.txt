#include<bits/stdc++.h>
using namespace std;
const int MAX=100;
string w[MAX+10];
int main ()
{
	int n ;
	cin>>n;
	for (int i=0;i<n;i++)
		cin>>w[i];
	int res=0;
	for (char c1='a';c1<='z';c1++){
		for (char c2=c1+1;c2<='z';c2++){
			int tot=0;
			for (int i=0;i<n;i++){
				bool f=true;
				for (int j=0;j<w[i].size();j++)
					if (w[i][j]!=c1&&w[i][j]!=c2)
						f=false;
				tot+= f*w[i].size();
			}
			res=max(res,tot);
		}
	}
	cout<<res<<endl;
}
