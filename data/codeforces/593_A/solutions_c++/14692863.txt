#include <bits/stdc++.h>
using namespace std;
char s[105][1005];
int main()
{
    int n, ans = 0;
    scanf("%d", &n);
    for(int i = 1; i <= n; i++)
        scanf("%s", s[i]);
    for(int i = 0; i < 26; i++)
        for(int j = 0; j < 26; j++)
        {
            int k, l, sum = 0;
            for(k = 1; k <= n; k++)
            {
                for(l = 0; s[k][l]; l++)
                    if(s[k][l] != i + 'a' && s[k][l] != j + 'a') break;
                sum += s[k][l] ? 0 : l;
            }
            ans = max(sum, ans);
        }
    printf("%d\n", ans);
    return 0;
}
