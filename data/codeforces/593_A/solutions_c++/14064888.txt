#include <iostream>

using namespace std;

string s[101];

bool check(string s, char a, char b)
{
	for (int i = 0; i < s.length(); i++)
		if (s[i] != a && s[i] != b)
			return 0;
	return 1;
}

int main()
{
	int n;
	cin>>n;
	for (int i = 0; i < n; i++)
		cin>>s[i];
	int count = 0, max = 0;
	for (int i = 'a'; i < 'z'; i++)
	{
		for (int j = i+1; j <= 'z'; j++)
		{
			count = 0;
			for (int l = 0; l < n; l++)
			{
				if (check(s[l],(char)i,(char)j))
					count += s[l].length();
			}
			if (count > max)
				max = count;
		}	
	}
	cout<<max;
	return 0;
}
