#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
char a[555][555];
int valid(int i,int j,int k)
{
//printf("check string %s\n",a[i]);
int l;
for(l=0;a[i][l];l++)
if(a[i][l]!=j && a[i][l]!=k)return 0;
//printf("valid string %s\n",a[i]);
return l;
}
int main()
{
int n;
scanf("%d",&n);
int i;
for(i=0;i<n;i++)
scanf("%s",&a[i]);
int ma = 0;
int j,k,l;
for(i='a';i<='z';i++)
for(j=i;j<='z';j++)
{
int cur = 0;
for(l=0;l<n;l++)
cur+=valid(l,i,j);
if(cur > ma)ma = cur;
}
	printf("%d\n",ma);
	return 0;
	}
