#include<string>
#include<iostream>
using namespace std;

int main()
{
	int n,length,ans=0;
	string word[100];
	cin >> n;
	for(int i=0; i<n; i++)
		cin >> word[i];
	for(int i=0; i<26;i++)
		for(int j=i+1; j<26; j++)
		{
			length = 0;
			for(int k=0; k<n; k++)
			{
				bool flag=true; 
				for(int l=0; l<word[k].size(); l++)
				{
					if(word[k][l]==i+'a' || word[k][l]==j+'a')continue;
					flag=false;
					break;
				}
				if(!flag)continue;
				length+=word[k].size();
			}
			if(length>ans)ans=length;
		}
	cout << ans << endl;
	return 0;
}
