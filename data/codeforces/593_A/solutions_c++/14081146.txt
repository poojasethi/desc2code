#include<bits/stdc++.h>
#define lld long long int

using namespace std;


int i, n, j, res, k, l, x, y;
string in[103];
bool chk;
int main(void)
{

cin>>n;

for(i=0; i<n; i++){
    cin>>in[i];

}

res=0;

for(i=0; i<26; i++){
    j=i;
  for(k=i+1; k<26; k++){
   l=0;
   for(x=0; x<n; x++){
        chk=true;
    for(y=0; in[x][y]; y++){

     if( (in[x][y]-'a')!=j && (in[x][y]-'a')!=k){
        chk=false;
        break;
     }

    }
    if(chk){
        l+=in[x].size();
    }
   }

   res=max(l, res);
}

}
cout<<res<<endl;
    return 0;
}
