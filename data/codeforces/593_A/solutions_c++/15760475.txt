#include<cstdio>
using namespace std;
int cnt[26][26];
int main(){
	int n;
	scanf("%d",&n);
	char s[2010];
	for(int i=0;i<n;i++){
		int tcnt[26]={},l=0;
		scanf("%s",s);
		while(s[l]){
			tcnt[s[l]-'a']++;
			l++;
		}
		for(int j=0;j<26;j++){
			for(int k=j+1;k<26;k++){
				if(tcnt[j]+tcnt[k]==l){
					cnt[j][k]+=l;
				}
			}
		}
	}
	int ans=0;
	for(int i=0;i<26;i++){
		for(int j=i+1;j<26;j++){
			if(cnt[i][j]>ans) ans=cnt[i][j];
		}
	}
	printf("%d\n",ans);
	return 0;
}
