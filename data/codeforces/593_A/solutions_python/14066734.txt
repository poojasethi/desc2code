n = input()
s = [''] * n
w = [[0 for j in range(26)] for i in range(n)]
q = [0] * n

for i in range(n):
  s[i] = raw_input()
  for c in s[i]:
    w[i][ord(c) - 97] = 1
  q[i] = sum(w[i])

m = 0
for i in range(26):
  for j in range(i + 1, 26):
    x = 0
    for k in range(n):
      if q[k] == 1 and (w[k][i] == 1 or w[k][j] == 1):
        x += len(s[k])
      if q[k] == 2 and (w[k][i] == 1 and w[k][j] == 1):
        x += len(s[k])

    if x > m: m = x

print m
