import string

def match(s, c1, c2):
    for c in s:
        if c != c1 and c != c2:
            return False
    return True
    

n = int(raw_input())

ss = []
for i in xrange(n):
    ss.append(raw_input())

ans = 0
for c1 in string.lowercase:
    for c2 in string.lowercase:
        ans = max(ans, sum([len(x) for x in  [y for y in ss if all([c == c1 or c == c2 for c in y])]]))
        
print(ans);
