n = input()
words = []
for _ in xrange(n):
    inp = raw_input()
    words+=[inp]
temp = 0
for i in xrange(26):
    for j in xrange(i+1,26):
        x = chr(i+97)
        y = chr(j+97)
        t1=0
        for ij in words:
            if(ij.count(x)+ij.count(y)==len(ij)):
                t1+=len(ij)
        temp = max(t1,temp)
print temp
