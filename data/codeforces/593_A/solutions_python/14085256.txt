from itertools import combinations
from string import ascii_lowercase


n = int(raw_input())
l = []
for i in range(n):
    l.append(raw_input())

res = map(lambda x: (set(x),len(x)), l)
l = [(tuple(k),v) for (k,v) in filter(lambda (x,y): len(x) <= 2, res)]
resd = {k:sum([i[1] for i in l if not set(i[0]) - set(k)]) for k in combinations(ascii_lowercase, 2)}
print max(resd.values() or [0])
