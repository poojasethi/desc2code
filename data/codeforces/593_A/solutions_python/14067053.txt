n = int(raw_input())
s= []
for i in range(0,n):
   s.append(raw_input())
maxx = 0


for i in range(0,26):
    for j in range(0,26):
        count =0
        for k in s:
            bad = 0
            for l in k:
                if l!=chr(i+ord('a')) and l!=chr(j+ord('a')):
                    bad = 1
            if not bad:
                count += len(k)
        if maxx < count:
            maxx =count

print maxx
