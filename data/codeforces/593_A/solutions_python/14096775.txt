
n_words = int(raw_input())
words = []
for i in range(n_words):
    words.append(raw_input())

max_c = 0
for i1 in range(ord('a'), ord('z')+1):
    for i2 in range(ord('a'), ord('z')+1):
        c1 = chr(i1)
        c2 = chr(i2)
        count = 0
        for w in words:
            w1 = w.replace(c1,'')
            w1 = w1.replace(c2,'')
            if not w1:
                count += len(w)
        if count > max_c:
            max_c = count

print max_c
