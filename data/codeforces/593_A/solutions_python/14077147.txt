n = input()
words = []
for i in xrange(n):
	words.append(raw_input())

ans = 0

for i in range(26):
	for j in range(i+1, 26):
		f = chr(i + 97)
		l = chr(j + 97)
		total = 0
		for word in words:
			if(word.count(f) + word.count(l) == len(word)):
				total += len(word)
		
		ans = max(total, ans)
		
print ans
