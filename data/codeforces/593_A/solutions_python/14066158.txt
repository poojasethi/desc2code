n = int(raw_input(""))
wordsLensDict = {}

for i in range(n):
    theWord = raw_input("").rstrip()
    s = frozenset(list(theWord))
    if len(s) < 3:
        if s in wordsLensDict:
            wordsLensDict[s].append(len(theWord))
        else:
            wordsLensDict[s] = [len(theWord)]
alphabet = map(chr, range(97, 123))
max_len = 0

# print wordsLensDict
for a in alphabet:
    for b in alphabet:
        if a == b:
            continue
        cur_len = sum(wordsLensDict.setdefault(frozenset([a,b]), [0])) + sum(wordsLensDict.setdefault(frozenset([a]), [0])) + sum(wordsLensDict.setdefault(frozenset([b]), [0]))
        max_len = max(max_len, cur_len)

print max_len
