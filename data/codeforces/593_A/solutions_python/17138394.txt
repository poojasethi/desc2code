from string import ascii_lowercase
lets='abcdefghijklmnopqrstuvwxyz'
n=int(raw_input())
wrds=[raw_input().strip() for i in xrange(n)]
ans=0
for i in lets:
    for j in lets:
        co=0
        for k in wrds:
            if not filter(lambda x:not(x==i or x==j),list(k)):
                co+=len(k)
        ans=max(ans,co)
print ans
