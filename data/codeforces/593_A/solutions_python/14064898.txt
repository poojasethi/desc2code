import string

n = int(input())
words = []
for i in range(n):
    w = raw_input()
    words.append(w)

max_len = 0
for i in range(len(string.ascii_lowercase)):
    for j in range(i+1, len(string.ascii_lowercase)):
        tot_len = 0
        for w in words:
            valid = True
            for l in w:
                if l not in [string.ascii_lowercase[i],string.ascii_lowercase[j]]:
                    valid = False
            if valid:
                tot_len += len(w)
        max_len = max(max_len, tot_len)

print max_len
