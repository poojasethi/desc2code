#! /usr/bin/env python
n=input()
lis=[]
cnt=[]
for i in range(26):
    temp=[]
    for j in range(26):
        temp.append(0)
    cnt.append(temp)

for i in range(n):
    lis.append(raw_input())
for i in lis:
    x=list(set(i))
    if len(x)>2:
        continue
    if len(x)==1:
        val=ord(x[0])-ord('a')
        cnt[val][val]+=len(i)
    else:
        val1=ord(x[0])-ord('a')
        val2=ord(x[1])-ord('a')
        cnt[val1][val2]+=len(i)
        cnt[val2][val1]+=len(i)
res=0
for i in range(26):
    for j in range(i+1,26):
        res=max(res,cnt[i][j]+cnt[i][i]+cnt[j][j])
print res
