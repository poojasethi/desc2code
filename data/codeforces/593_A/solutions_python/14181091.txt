import string

n = int(raw_input())
s = []
for i in range(n) :
	s.append(raw_input())

ans = 0
for c1 in string.lowercase:
	for c2 in string.lowercase :
		ans = max(ans, sum([len(x) for x in [y for y in s if all([t == c1 or t == c2 for t in y])]]))

print ans
