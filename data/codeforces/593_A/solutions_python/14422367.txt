#!/usr/bin/env python2

import sys

def main():
    words = sys.stdin.read().split()[1:]
    alpha = "abcdefghijklmnopqrstuvwxyz"
    max_len = 0
    for a in alpha:
        for b in alpha:
            t = 0
            if a == b:
                continue
            for w in words:
                if set(w) <= set([a, b]):
                    t += len(w)
            max_len = max(max_len, t)
    print max_len
                    
                

main()
