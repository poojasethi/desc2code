import sys
w=[sys.stdin.readline()[:-1] for i in range(int(sys.stdin.readline()))]
res=0

def only(c1, c2):
  def f(s):
    for c in s:
      if c != c1 and c != c2:
        return False
    return True
  return f


for i in range(26):
  for j in range(i, 26):
    res = max(res,sum(map(len,filter(only(chr(i+ord('a')),chr(j+ord('a'))), w))))
print res
