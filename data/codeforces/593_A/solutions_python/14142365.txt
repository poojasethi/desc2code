import string

def match(s, c1, c2):
    for c in s:
        if c != c1 and c != c2:
            return False
    return True
    

n = int(raw_input())

ss = []
for i in xrange(n):
    ss.append(raw_input())

ans = 0
for c1 in string.lowercase:
    for c2 in string.lowercase:
        if c1 == c2:  
            continue
        sl = [x for x in ss if match(x, c1, c2)]
        ans = max(ans, sum([len(x) for x in sl]))
        
print(ans);
