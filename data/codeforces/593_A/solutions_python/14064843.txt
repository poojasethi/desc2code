n = int(raw_input())
a = [raw_input().strip() for _ in xrange(n)]
alpha = 'qwertyuiopasdfghjklzxcvbnm'
t = [sum((c in s) << i for i, c in enumerate(alpha)) for s in a]
ans = 0
for i in xrange(26):
    for j in xrange(26):
        m = (1 << 26) - 1
        m ^= 1 << i
        m ^= 1 << j
        c = sum(len(a[k]) for k in xrange(n) if (t[k] & m) == 0)
        if ans < c:
            ans = c
print ans
