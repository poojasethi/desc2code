def hasonly(w,c1,c2):
	alpha = "abcdefghijklmnopqrstuvwxyz"
	for c in alpha:
		if (c == c1) or (c == c2):
			continue
		if c in w:
			return False
	return True 

def main():
	alpha = "abcdefghijklmnopqrstuvwxyz"
	n = int(raw_input(''))
	words = []
	for i in xrange(n):
		w   = raw_input('').strip()
		words.append(w)
	maxlength = 0
	for c1 in alpha:
		for c2 in alpha:
			length = 0
			for word in words:
				if hasonly(word, c1, c2):
					length += len(word)
			if length >= maxlength:
				maxlength = length
	print maxlength

main()
