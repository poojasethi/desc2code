n = int(raw_input())
from collections import defaultdict
import string
d = defaultdict(int)
for i in xrange(n):
    wordlist = raw_input()
    word = set(wordlist)
    if len(word) > 2:
        continue
    else:
        chars = sorted(list(word))
    if len(chars) == 2:
        d[(chars[0], chars[1])] += len(wordlist)
    else:
        for c in string.lowercase:
            if chars[0] < c:
                d[(chars[0], c)] += len(wordlist)
            else:
                d[(c, chars[0])] += len(wordlist)
ans = 0
for key in d:
    if d[key] > ans:
        ans = d[key]
print ans
