n=int(raw_input())
l=[]
ans=0
for i in range(n):
    w=list(raw_input())
    if len(set(w))<3:
        w.sort()
        l.append(w)
for i in range(26):
    for j in range(i,26):
        tmp=0
        x,y=chr(i+97),chr(j+97)
        for k in l:
            if x==y:
                if k[0]==k[-1]==x:
                    tmp+=len(k)
            else:
                chk=[x,y]
                if k[0] in chk and k[-1] in chk:
                    tmp+=len(k)
        ans=max(ans,tmp)
print ans
