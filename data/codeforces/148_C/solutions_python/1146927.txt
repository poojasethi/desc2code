#!/usr/bin/env python
n, a, b = map(int, raw_input().split())

x = [1]
if a != 0 and b == 0:
    x.append(1)

for i in xrange(b):
    x.append(sum(x)+1)
for i in xrange(a):
    x.append(x[-1]+1)
x.extend([1 for i in xrange(n-len(x))])

if len(x) > n:
    print -1
else:
    print ' '.join([str(i) for i in x])
