import sys
n,a,b = map(int,raw_input().split())

s = range(0,n)
s[0] = 1
for i in range(1,n):
    if b>0:
        b -= 1
        s[i] = sum(s[0:i]) + 1
        if s[i] >= 50000:
            print -1
            exit(0)
        continue
    if i>1 and a>0:
        a -= 1
        s[i] = s[i-1] + 1
        continue
    s[i] = 1

if a>0 or b>0:
    print -1
    exit(0)

for x in s:
    sys.stdout.write('%d ' % x)
print
