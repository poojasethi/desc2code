#!/usr/bin/env python

if __name__ == "__main__":
    n, b, a = map(int, raw_input().split())
    if a != 0:
        if 2 ** a + b <= 50000:
            price = [1]
            for i in range(1, a + 1):
                price.append(2 ** i)
            for i in range(price[-1] + 1, price[-1] + 1 + b):
                price.append(i)
            for i in range(len(price), n):
                price.append(1)
            print " ".join(map(str, price))
        else:
            print -1
    else:
        if b == 0:
            print "1 " * n
        elif 2 + b <= n:
            price = [1, 1]
            for i in range(2, 2 + b):
                price.append(i)
            for i in range(len(price), n):
                price.append(1)
            print " ".join(map(str, price))
        else:
            print -1
