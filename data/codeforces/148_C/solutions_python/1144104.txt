#!/usr/bin/env python

n, a, b = tuple(map (int, raw_input().split(" ")))

w0=1

if n==1:
    print 1
    exit()


if w0*2**b+a>50000 or (b==0 and a==n-1):
    print -1
    exit()

w=[1]*n

if b==0:
    for i in xrange(2,n):
        if b>0:
            w[i]=w[i-1]*2
            b-=1
        elif a>0:
            w[i]=w[i-1]+1
            a-=1
        else:
            break
else:
    for i in xrange(1,n):
        if b>0:
            w[i]=w[i-1]*2
            b-=1
        elif a>0:
            w[i]=w[i-1]+1
            a-=1
        else:
            break

print ' '.join(map(str,w))
