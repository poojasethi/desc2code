#!/usr/bin/env python

from sys import stdin

N, A, B = map(int, stdin.readline().split())

T = []
if B > 0:
   T.append(1)
   for i in xrange(B):
      last = T[-1]
      T.append( last*2 )
elif A > 0:
   if A == N-1:
      print -1
      exit(0)
   T += [1, 1]

for i in xrange(A):
   last = T[-1]
   T.append( last + 1 )

T += [1] * (N-len(T))
print ' '.join(map(str, T))
