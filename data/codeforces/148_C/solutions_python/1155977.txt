n, a, b = map(int, raw_input().split())
ans = [1]
if n == 1:
	ans = [1]
elif b == 0 and a == n - 1:
	ans = [-1]
elif b == 0:
	ans = [1] * (n - a)
	for i in range(a):
		ans.append(ans[-1] + 1)
else:
	for i in range(b):
		ans.append(sum(ans) + 1)
	ans += range(ans[-1] + 1, ans[-1] + a + 1)
	ans += [1] * (n - a - b - 1)
print ' '.join(map(str, ans))
