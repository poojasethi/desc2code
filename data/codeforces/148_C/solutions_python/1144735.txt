#!/usr/bin/env python
from sys import stdin as cin

def main():
    n, a, b = map(int, next(cin).split())
    r = [1]
    if b == 0:
        if a + 1 == n:
            return 1 if a == 0 else -1
        r.append(1)
        b = 1
    else:
        for i in range(b):
            r.append(sum(r) + 1)
    for i in range(a):
        r.append(r[-1] + 1)
    if r[-1] > 50000:
        return -1
    for i in range(n - a - b - 1):
        r.append(1)
    return ' '.join(map(str, r))

print main()
