n, b, a = map(int, raw_input().split())
c = n - a - b - 1
if c < 0: print '-1'
elif a:
    d = (1 << a) + 1
    print ' '.join(str(1 << i) for i in range(a + 1)), ' '.join(map(str, range(d, d + b))), '1 ' * c
elif b < n - 1: print '1 ' * (c + 1), ' '.join(map(str, range(2, 2 + b)))
elif b: print '-1'
else: print '1'
