import sys
n,a,b = map(int,raw_input().split())
m=n-a-b

if n==1:
    print 1
    sys.exit()

if m==1 and b==0:
    print -1
    sys.exit()
s = [1]
add = 1

if b>0:
    for i in xrange(b):
        s.append(add+1)
        add += s[-1]
    for i in xrange(a):
        s.append(s[-1]+1)
        add += s[-1]
    s += [1]*(m-1)

else:
    s = [1,1]
    for i in xrange(a):
        s.append(s[-1]+1)
    s += [1]*(m-2)

print ' '.join(map(str,s))
