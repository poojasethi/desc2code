n, a, b = map(int, raw_input().split())
if n == 1:
    print 1
    exit()
if b == 0 and n - 1 == a:
    print -1
    exit()
res = [2 ** i for i in xrange(b + 1)]
if b == 0:
    res = [2, 1]
    b += 1
res.extend([max(res) + x + 1 for x in xrange(a)])
res.extend([1 for i in range(n - a - b - 1)])
print ' '.join(str(i) for i in res)
