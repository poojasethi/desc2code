n, a, b = map(int, raw_input().split())
v = [1] * n
if b == 0:
    for i in range(2, n):
        if a > 0:
            v[i] = v[i-1] + 1
            a -= 1
        else:
            v[i] = v[i-1]
else:
    for i in range(1, n):
        if b > 0:
            v[i] = v[i-1] * 2
            b -= 1
        elif a > 0:
            v[i] = v[i-1] + 1
            a -= 1
        else:
            v[i] = v[i-1]
