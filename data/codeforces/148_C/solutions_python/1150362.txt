import sys

n, a, b = [int(s) for s in sys.stdin.readline().split()]

grooms = [1]

for i in range(b):
    grooms.append(sum(grooms)+1)

for i in range(a):
    if(sum(grooms) == max(grooms)):
        grooms.append(1)
    grooms.append(max(grooms)+1)

for i in range(n-len(grooms)):
    grooms.append(1)

if(max(grooms) > 50000 or len(grooms) != n):
    sys.stdout.write("-1\n")
else:
    for g in grooms:
        sys.stdout.write("{0} ".format(g))
    sys.stdout.write("\n")
