#include <cstdio>
int sum=1,d=1,ans[101]={1,0};
int main(){
    int n,a,b;
    scanf("%d%d%d",&n,&a,&b);
    for(int i=1;i<n;i++){
        if(b>0){
            d=sum+1;
            b--;
        }
        else if(a>0&&i!=1){
            d++;
            a--;
        }
        ans[i]=d;
        sum+=d;
    }
    if(a>0||b>0) printf("-1");
    else
    for(int i=0;i<n;i++) printf("%d ",ans[i]);
   return 0;
}
