int t[10050],n,a,b;
int main()
{
	scanf("%d%d%d",&n,&a,&b);
	if (!b && n-a==1 && a) {printf("-1\n"); return 0;} 
	int sum=1,j=1; t[1]=1;
	for (int i=1;i<=b;i++) t[++j]=sum+1,sum+=t[j];
	if (!b) t[1]=t[++j]=1000;
	for (int i=1;i<=a;i++) t[++j]=t[j-1]+1;
	for (int i=1;i<n-a-b;i++) t[++j]=1;
	for (int i=1;i<n;i++) printf("%d ",t[i]);
	printf("%d\n",t[n]);
	//for(;;);

}
