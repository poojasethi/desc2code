#include<cstdio>
using namespace std;
int main()
{
	int n,a,b,c=3,d=2;
	scanf("%d%d%d",&n,&a,&b);
	if(n==1)
	{puts("1");return 0;}
	if(a==n-1)
	{puts("-1");return 0;}
	if(b)
	{printf("1 2");--b;}
	else{printf("1 1");c=1;d=1;}
	for(int i=2;i<n;++i)
	{
		if(b)
		{
			printf(" %d",c+1);
			d=c+1;c=2*c+1;
			--b;
		}
		else if(a)
		{
			printf(" %d",++d);
			--a;
		}
		else printf(" 1");
	}
	puts("");
}
