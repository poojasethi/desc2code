#include <cstdio>
int n,a,b;
int main(){
    scanf("%d%d%d",&n,&a,&b);
    if (b==0&&a>0){
        if (n==a+b+1){
            printf("-1\n");
            return 0;
        }
        printf("1 ");
        --n;
    }
    printf("1 ");
    for(int i=1; i<=b; ++i) printf("%d ",1<<i);
    for(int i=1; i<=a; ++i) printf("%d ",(1<<b)+i);
    for(int i=a+b+2; i<=n; ++i) printf("1%c",i<n?' ':'\n');
    return 0;
}
