#include<cstdio>
int main()
{
	int i,a,b,n,s;
	scanf("%d%d%d",&n,&a,&b);
	if(b==0)
	{
		if(n==1)
			printf("1");
		else if(a==n-1)
			printf("-1\n");
		else
		{
			printf("1 1 ");
			for(i=3;i<=a+2;i++)
				printf("%d ",i-1);
			for(;i<=n;i++)
				printf("1 ");
		}
	}
	else
	{
		for(s=i=1;i<=b+1;s<<=1,i++)
			printf("%d ",s);
		s/=2;
		for(;i<=b+a+1;i++,s++)
			printf("%d ",s+1);
		for(;i<=n;i++)
			printf("%d ",1);
	}
	return 0;
}
