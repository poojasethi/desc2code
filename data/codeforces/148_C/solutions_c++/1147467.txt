#include <iostream>
#include <vector>

using namespace std;

int main()
{
	int n,a,b;
	int index=1;
	cin >> n >> a >> b;
	vector<int> res(n,1);
	if (b==0) {
		index++;
	} else {
		for (int i=0; i<b; i++,index++) {
			res[index] = res[index-1]*2;
		}
	}
	for (int i=0; i<a; i++,index++) {
		if (index == n) {
			cout << "-1\n";
			return 0;
		}
		res[index] = res[index-1] + 1;
	}
	for (int i=0; i<n; i++) {
		cout << res[i] << " ";
	}
	cout << endl;
	return 0;
}
