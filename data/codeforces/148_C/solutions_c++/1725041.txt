#include<cstdio>
int f[105];
int main()
{
    int n,a,b;
    scanf("%d%d%d",&n,&a,&b);
    if (n>1&&b==0&&a==n-1) return 0&puts("-1");
    f[0]=1;
    int sum=1;
    for(int i=1;i<=b;i++)
    {
        f[i]=sum+1;
        sum+=f[i];
    }
    if (b==0) f[1]=1;
    for(int i=b+1+(b==0);i<=a+b+(b==0);i++)
    {
        //puts("!");
        f[i]=f[i-1]+1;
    }
    for(int i=a+b+1+(b==0);i<n;i++) f[i]=f[i-1];
    for(int i=0;i<n;i++) printf("%d ",f[i]);
    puts("");
}
