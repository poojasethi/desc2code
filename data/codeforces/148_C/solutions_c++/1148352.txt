#include <iostream>

using namespace std;

int n, a, b;
int seq[100];

int main()
{
	cin >> n >> a >> b;
	seq[0]=1;
	int i=1;
	while(b)
		seq[i++]=seq[i-1]*2, b--;		
	while(i<n-a)
		seq[i++]=seq[i-1];
	if(i==1 && a!=0)
	{
		cout << -1 << endl;
		return 0;
	}
	while(a)
		seq[i++]=seq[i-1]+1, a--;	
	for(int i=0; i<n; i++)
		cout << seq[i] << " ";
	return 0;
}
