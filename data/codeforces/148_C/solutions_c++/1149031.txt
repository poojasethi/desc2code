#include <math.h>
#include <stdio.h>
#include <stdlib.h>

int main(){
    int n,a,b,a1[101],s=1,k=0;
    scanf("%i%i%i",&n,&a,&b);
    a1[0]=1;
    for(int i=0; i<b; i++){
		k++;
		a1[k]=s+1;
		s+=a1[k];
	}
	for(int i=0; i<n-a-b-1; i++){
		k++;
		a1[k]=a1[k-1];
	}
	for(int i=0; i<a; i++){
		k++;
		a1[k]=a1[k-1]+1;
	}
	if(!b&&n-a==1&&a) printf("-1"); else for(int i=0; i<n; i++)printf("%i ",a1[i]);
}
