#include <cstdio>
using namespace std;
int s[103],n,a,b,cur,sum;
int main()
{
    scanf("%d%d%d",&n,&a,&b);
    s[1]=cur=sum=1;
    for(int i=2;i<=n;i++)
    {
        if(b>0)cur=sum+1,b--;else
        if(a>0&&i!=2)cur++,a--;
        s[i]=cur,sum+=cur;
    }
    if(a>0||b>0)puts("-1");else 
        for(int i=1;i<=n;i++)printf("%d ",s[i]);
    return 0;    
}
