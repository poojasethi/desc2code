#include <iostream>
#include <vector>

using namespace std;

int N,A,B,groom[110],cnt=1;

int main()
{
	cin >> N >> A >> B;
	int fortune=16-B,sum;
	groom[0]=sum=fortune;
	for (int i=0;i<B;i++)
	{
		groom[cnt++]=fortune=sum+1;
		sum+=fortune;
	}
	if (cnt==1&&A&&N>1)
		groom[cnt++]=fortune;
	for (int i=0;i<A;i++)
		groom[cnt++]=++fortune;
	for (;cnt<N;cnt++)
		groom[cnt]=fortune;
	if (cnt!=N)
		cout << -1;
	else
		for (int i=0;i<cnt;i++)
			cout << groom[i] << " ";
	cout << endl;
	return 0;
}
