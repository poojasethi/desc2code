#include <stdio.h>

int main(){
	int n,a,b,r=1,sum=0,i;
	scanf("%d%d%d",&n,&a,&b);
	if(b>0){
		printf("%d",r); 
		sum=1;
		for(i=0;i<b;i++){
			r*=2;
			printf(" %d",r);
			sum++;
		}
	}
	else{
		if(n>1&&a==n-1){
			printf("-1\n");
			return 0;
		}
		r=2;
		printf("%d",r);
		sum=1;
		if(a>0){
			printf(" %d",r);
			sum++;
		}
	}
	for(i=0;i<a;i++){
		r++;
		printf(" %d",r);
		sum++;
	}
	for(i=sum;i<n;i++){
		printf(" 1");
	}
	printf("\n");
	return 0;
}
