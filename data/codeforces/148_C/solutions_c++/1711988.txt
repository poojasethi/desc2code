#include <iostream>

using namespace std;

int n, a, b, sum;

int ans[101];

int main()
{
    cin >> n >> a >> b;
    ans[1] = sum = 1;
    for (int i = 2; i <= n; i++)
    {
        if (b) ans[i] = sum + 1, sum += sum + 1, b--;
        else if (a && i > 2) ans[i] = ans[i - 1] + 1, a--;
        else ans[i] = 1;
    }
    if (a || b) cout << -1;
    else
    {
        for (int i = 1; i <= n; i++) cout << ans[i] << ' ';
    }
    return 0;
}
