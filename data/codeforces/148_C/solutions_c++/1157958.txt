#include <stdio.h>

int main()
{
	long T[100]={1};
	int n,a,b;
	scanf("%d %d %d",&n,&a,&b);
	if (b==0 && a+1==n && n!=1)
	{
		printf("-1");
		return 0;
	}
	for (int k=1; k<=b; k++)
		T[k]=T[k-1]*2;
	if (b==0) {b=b+1; T[b]=T[b-1];};
	for (int k=b+1; k<=b+a; k++)
		T[k]=T[k-1]+1;
	for (int k=b+a+1; k<n; k++)
		T[k]=T[k-1];
	for (int k=0; k<n; k++)
		printf("%d ",T[k]);

	return 0;
}
