#include<cstdio>
int n,a,b,c[110],now,sum;
int main(){
    scanf("%d%d%d",&n,&a,&b);
    c[1]=now=sum=1;
    for(int i=2;i<=n;++i){
        if(b)now=sum+1,--b;
        else if(a){
		if(i!=2){
			now++;
			a--;
		}	
	}
        c[i]=now;
	sum+=now;
    }
    if(a||b)printf("-1");
    else for(int i=1;i<=n;++i)printf("%d ",c[i]);
    return 0; 
}
