#include<iostream>
using namespace std;
int main()
{
int i,n,a,b,mx=1,x[105],cm=1;
cin>>n>>a>>b;
x[1]=1;
for(i=2;i<=n;i++)
{
if(i==2 && b)
{
b--;
x[i]=cm+1;
}
else if(i==2)
{
x[1]=cm=mx=50;
x[2]=1;
}
else if(b)
{
b--;
x[i]=cm+1;
}
else if(a)
{
a--;
x[i]=mx+1;
}
else x[i]=1;
cm+=x[i];
mx=max(mx,x[i]);
}
if(!a && !b && mx<=50000)
{
for(i=1;i<=n;i++)
{
cout<<x[i]<<" ";
}
}
else cout<<-1;
return 0;
}
