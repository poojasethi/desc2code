#include <iostream>
using namespace std;
int main ( ) {
        int n, oh, wow, sum, i, ans[128], now;
        cin >> n >> oh >> wow;
        ans[1] = now = sum = 1;
        for ( i = 2; i <= n; ++i ) {
                if ( wow ) now = sum + 1, --wow;
                else if ( oh ) if ( i != 2 ) ++now, --oh;
                ans[i] = now, sum += now;
        }
        if ( oh ) cout << "-1\n";
        else for ( i = 1; i <= n; ++i ) cout << ans[i] << ' ';
}
