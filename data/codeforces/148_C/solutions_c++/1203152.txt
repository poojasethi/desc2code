#include <iostream>
using namespace std;

int main()
{
  int i,n,a,b,p=1;
  cin >> n >> a >> b;
  if(n==1)
  {
    cout << "1";
    return 0;
  }
  if(b==0&&a==n-1)
  {
    cout << "-1";
    return 0;
  }
  cout << "1 ";
  for(i=1;i<=b;i++)
  {
    p*=2;
    cout << p << " ";
  }
  if(b==0) 
  {
    p--;
    a++;
  }
  for(i=1;i<=a;i++)
  {
    p++;
    cout << p << " ";
  }
  for(i=a+b+2;i<=n;i++)
  {
    cout << "1 ";
  }
  return 0;
}
