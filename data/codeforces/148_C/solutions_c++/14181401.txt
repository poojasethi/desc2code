#include <iostream>
using namespace std;
int A[200];
int main() {
    int n,a,b;
    cin >> n >> b >> a;
    if(b==n-1&&b!=0)return cout << -1,0;
    A[0]=1;
    for(int i=1;i<=a;i++)A[i]=A[i-1]<<1;
    if(a==0)A[++a]=1;
    for(int i=1;i<=b;i++)A[i+a]=A[i+a-1]+1;
    for(int i=a+b+1;i<n;i++)A[i]=1;
    for(int i=0;i<n;i++)cout << A[i]<<' ';
    return 0;
}
