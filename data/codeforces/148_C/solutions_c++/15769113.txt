#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,a,b;
    scanf("%d%d%d",&n,&a,&b);
    int s[105];
    s[1]=1;
    for (int i=2;i<=n;i++){
        if (b) s[i]=1<<(i-1),b--;
        else {
            s[i]=s[i-1];
            if (a&&i>2) s[i]++,a--;
        }
    }
    if (a||b) puts("-1");
    else {
        for (int i=1;i<=n;i++){
            if (i!=1) printf(" ");
            printf("%d",s[i]);
        }
        puts("");
    }
}
