#include<iostream>
#include<cstdio>
using namespace std;
int ans[105];
int main()
{
	int a,b,n,cur=1;
	scanf("%d%d%d",&n,&a,&b);
	ans[1]=1;
	for(int i=2;i<=n;i++)
	{
		if(b)
		{
			ans[i]=1<<(i-1);
			b--;
		}
		else if(a&&i>2)
		{
			ans[i]=ans[i-1]+1;
			a--;
		}
		else ans[i]=ans[i-1];
	}
	if(a||b)
	{
		printf("-1\n");
	}
	else 
	{
		for(int i=1;i<n;i++)
			printf("%d ",ans[i]);
		printf("%d\n",ans[n]);
	}
	return 0;
}
