#include <iostream>

using namespace std;

int main()
{
    long long int n,a,b,i,d=1;
    cin >> n >> a >> b;

    if(a+b>n-1 || (a==n-1 && a>0)) {cout << -1 << endl; return 0;}
    cout << 1 << " ";
    for(i=0;i<b;i++) {cout << d+1 << " "; d+=d+1;}
    d++;
    d/=2;
    if(b==0 && a>0) {cout << 1 << " "; b++;}
    for(i=1;i<=a;i++) cout << d+i << " ";
    for(i=a+b+1;i<n;i++) cout << 1 << " ";
    return 0;
}
