#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int n,a,b,sum=2,maxn=2;
    cin>>n>>a>>b;
    if(a==n-1&&a){printf("-1\n");return 0;}
    cout<<2;
    for(int i=1;i<n;++i)
    {
        if(b)cout<<' '<<sum+1,--b,maxn=sum+1,sum+=sum+1;
        else if(i!=1&&a)cout<<' '<<maxn+1,--a,++maxn;
        else cout<<' '<<2,sum+=2;
    }
    cout<<endl;
}
