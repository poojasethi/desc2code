#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	int n,a,b;
	cin>>n>>a>>b;
	if(a==n-1&&n!=1)
		printf("-1\n");
	else
	{
		ll cur=1;
		n--;
		cout<<cur;
		for(int i=0;i<b;i++)
		{

			cout<<" "<<cur+1;
			if(i+1!=b)

			{
				cur*=2;
				cur+=1;
			}
			else
				cur++;


		}

		n-=b;
		int r=n-a;
		for(int i=0;i<r;i++)
			cout<<" "<<cur;
			for(int i=0;i<a;i++)
				cout<<" "<<++cur;
			cout<<endl;
	}

}
