#include <cstdio>
#include <cstring>
int t[105];
int main(void)
{
	int n,a,b;
	scanf("%d%d%d",&n,&a,&b);
	if(!b&&a&&n-a==1)
	{
		printf("-1\n");
		return 0;
	}
	printf("1");
	t[0]=1;
	for(int i=1;i<n;i++)
	{
		if(!b&&i==1)
		{
			t[1]=1;
			printf(" 1");
			a++;
			continue;
		}
		if(i<=b)
			t[i]=1<<i;
		else if(i<=a+b)
			t[i]=t[i-1]+1;
		else
			t[i]=t[i-1];
		printf(" %d",t[i] );
	}
	printf("\n");
}
