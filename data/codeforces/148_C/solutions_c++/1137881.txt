#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n, a, b, t, s, r[105];

int main(){
	scanf("%d%d%d", &n, &a, &b);
	
	t = r[0] = 1;
	for (int i=0; i<b; i++) r[t++] = (1<<(i+1));
	
	for (int i=0; i<n-a-b-1; i++) r[t] = r[t-1], t++;
	
	for (int i=0; i<a; i++){
		r[t] = r[t-1] + 1;
		t++;
	}

	if (n - 1 == a && n > 1) printf("-1\n");
	else{
		for (int i=0; i<n; i++) printf("%d ", r[i]);
		printf("\n");
	}
	return 0;
}
