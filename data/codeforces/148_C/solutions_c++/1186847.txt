#include <iostream>
using namespace std;

int n, a, b;

int main() {
	cin >> n >> a >> b;
	int cur = 1;
	if (b > 0) {
		n = n - 1 - a - b;
		cout << cur;
		while (b--) {
			cur <<= 1;
			cout << " " << cur;
		}
	} else if (n > a + 1) {
		n = n - 2 - a;
		cout << "2 1";
		cur = 2;
	} else if (n == 1) {
		n--;
		cout << "1";
	} else {
		cout << -1 << endl;
		return 0;
	}
	while (a--) {
		cur++;
		cout << " " << cur;
	}
	while (n--) 
		cout << " 1";
	cout << endl;
	return 0;
}
