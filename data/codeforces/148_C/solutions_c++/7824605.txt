#include<cstdio>

int main()
{
	int n, a, b; scanf("%d%d%d", &n, &a, &b);
	int prev = 1;
	if(a > 0 && a > n-2)
	{
		printf("-1\n");
		return 0;
	}
	printf("%d ", prev);
	for(int i = 1; i < n; i++)
	{
		if(b)
		{
			prev *= 2;
			printf("%d ", prev);
			b--;
			continue;
		}
		if(a && i >= 2)
		{
			prev++;
			printf("%d ", prev);
			a--;
			continue;
		}
		printf("1 ");
	}
	printf("\n");
	return 0;
}
