#include <cstdio>

int main(){
    int n, a, b, s, i, c[100];
    scanf("%d%d%d", &n, &a, &b);
    for(c[0] = 1, s = i = 1; i < n; s += c[i++]){
        if(b) {c[i] = s + 1; b--;}
        else if(a && i != 1) {c[i] = c[i-1] + 1; a--;}
        else c[i] = 1;
        if(c[i] > 50000) break;
    }
    if(i >= n && !a && !b)
        for(i = 0; i < n ; i++)
            printf("%d ", c[i]);
    else
        printf("-1");
    puts("");
    return 0;
}
