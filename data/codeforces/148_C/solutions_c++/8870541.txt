#include <iostream>
using namespace std;

int n, a, b, now, sum, c[110];

int main()
{
    cin >> n >> a >> b;
    now = sum = c[1] = 1;

    for (int i = 2; i <= n; ++i)
    {
        if (b)  now = sum+1, --b;
        else if (a && i != 2)   --a, ++now;
        c[i] = now;
        sum += now;
    }
    if (a || b)
    {
        cout << -1 << endl; return 0;
    }
    for (int i = 1; i <= n; ++i)
        cout << c[i] << " ";
    return 0;
}
