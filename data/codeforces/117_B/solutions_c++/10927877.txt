#include <iostream>
#include <cstdio>
using namespace std;
long long a,b,mod;
int p;
int main()
{
	cin>>a>>b>>mod;
	if (b>=mod)
	{
		cout<<2<<endl;
		return 0;
	}
	p=1000000000%mod;
	a=min(a, mod);
	long long i;
	for (i=1; i<=a; i++)
		if (i*p%mod+b<mod&&i*p%mod!=0)
		{
			printf("1 %09d\n", (int)i);
			return 0;
		}
	cout<<2<<endl;
	return 0;
}
