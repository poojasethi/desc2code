#include<cstdio>
main() {
  int i, j, a, b, m;
  scanf("%d %d %d", &a, &b, &m);
  if (b >= m-1) {
    puts("2");
    return 0;
  }
  for (i = 1; i < m && i <= a; ++i) {
    j = i*1000000000ll%m;
    if (j > 0 && j < m - b) break;
  }
  if (i < m && i <= a) printf("1 %09d\n", i);
  else puts("2");
}
