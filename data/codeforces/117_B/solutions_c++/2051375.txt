#include <cstdio>
int main ( ) {
        int a, b, mod, i;
        scanf ( "%d%d%d", &a, &b, &mod );
        for ( i = 0; i <= a && i < mod; ++i )
                if ( ( mod - (long long)i * 1000000000 % mod ) % mod > b ) {
                        printf ( "1 %09d\n", i );
                        return 0;
                }
        puts ( "2" );
}
