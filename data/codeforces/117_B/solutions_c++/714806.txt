#include <cstdio>
int main() {
	int a, b, mod, t, ret;
	scanf("%d%d%d", &a, &b, &mod);
	if (b + 1 >= mod) puts("2");
	else {
		ret = -1;
		for (int i = 1;i <= a && i <= mod;i++) {
			t = (i * 1000000000LL) % mod;
			if (1 <= t && t <= mod - b - 1)
				if (ret < 0 || i < ret)
					ret = i;
		}
		if (ret < 0) puts("2");
		else printf("1 %09d\n", ret);
	}
	return 0;
}
