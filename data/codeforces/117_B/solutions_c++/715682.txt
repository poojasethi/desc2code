#include <cstdio>

int main() {
  long long a,b,mo;
  scanf("%I64d%I64d%I64d",&a,&b,&mo);
  long long c=1000000000ll%mo;
  long long s=0,t=0;
  if (a>mo) a=mo;
  for (;s<=a;s++,t+=c) {
    if (t>=mo) t-=mo;
    if (t+b<mo&&t) break;
  };
  if (s==a+1) printf("2\n");
        else printf("1 %09I64d\n",s);
}
