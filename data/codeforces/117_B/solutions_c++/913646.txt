#include <iostream>
#include <algorithm>
#include <cstdio>

using namespace std;

int a;
int b;
int mod;
int m;

int main(){
    cin>>a;
    cin>>b;
    cin>>mod;
    m=min(a,mod);
    for(int i=0;i<=m;i++){
        if((mod-(i*1000000000LL)%mod)%mod>b){
            printf("1 %09d",i);
            return 0;
        }
    }
    cout<<"2";
}
