#include<cstdio>
using namespace std;
const int p=1e9;
int a,b,mod;
int main(){
	scanf("%d%d%d", &a, &b, &mod);
	if (b>=mod-1){puts("2");return 0;}
	for(int i=1;i<mod && i<=a;i++)
		if (((long long)i*(p%mod)+b)%(long long)mod>b){printf("1 %09d\n", i);return 0;}
	puts("2");
	return 0;
}
