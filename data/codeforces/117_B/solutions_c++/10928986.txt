#include<cstdio>
using namespace std;

long long p=1000000000;
int main(){
	int a,b,mod;
	scanf("%d%d%d",&a,&b,&mod);
	if(b>=mod-1){printf("2\n");return 0;}
	for(long long i=1;i<mod && i<=a;i++)
		if((i*p+b)%(long long)mod>b){printf("1 %09d\n", i);return 0;}
	printf("2\n");
	return 0;
}
