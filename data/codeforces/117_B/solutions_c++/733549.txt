#include<stdio.h>
int i , x , y , z ;
int main( ) 
{
  	while( scanf( "%d%d%d" , &x , &y , &z ) != EOF )
  	{
		bool ans = true ;
	  	for( i = 1 ; i < z && i <= x ; i++ ) 
		  	if( ( z - ( i * 1000000000LL ) % z ) % z > y ) 
			{ 
				printf( "1 %09d\n" , i ) ; 
				ans = false ; 
				break ;
			}
	  	if( ans )
			puts( "2" ) ;
	}
  	return 0 ;
}
