#include <iostream>
#include <iomanip>
using namespace std;

typedef long long ll;

int main() {
  ll a, b, m;
  cin >> a >> b >> m;
  for (ll i = 0; i < m and i <= a; ++i) {
    ll t = ((-i*1000000000LL)%m + m)%m;
    if (t > b) {
      cout << "1 " << setfill('0') << setw(9) << i << endl;
      return 0;
    }
  }
  cout << 2 << endl;
}
