
//E - Very Interesting Game
#include<iostream>
#include<cstdio>
#define R(a,b) ((a)%b)
#define ll long long
using namespace std;
int main()
{
    ll a,b,mod;
    cin>>a>>b>>mod;
    for(ll i=1; i<=a&&i<mod; i++)
    {
        if(R(mod-R(i*1000000000L, mod), mod) >b) {printf("1 %.9d\n",i);return 0;}
    }
    puts("2");
    return 0;
}
