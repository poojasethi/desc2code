#include<stdio.h>
int min(int a,int b)
{
	return a<b?a:b;
}
int main()
{	
	__int64 a,b,m,i,d,t;
	scanf("%I64d%I64d%I64d",&a,&b,&m);
	a=min(a,m-1);
	t=1000000000%m;
	for(i=1;i<=a;i++)
	{
		d=t*i%m;
		if(b<m-d&&d)
		{
			printf("1 %09I64d\n",i);
			return 0;
		}
	}
	printf("2\n");
	return 0;
}
