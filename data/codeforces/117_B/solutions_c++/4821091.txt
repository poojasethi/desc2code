#include <bits/stdc++.h>

using namespace std;

typedef long long LL;

LL a, b, m;

int main()
{
    cin >> a >> b >> m;
    for (LL i = 0; i < min(m, a+1); i++)
    {
        if ((m-((1000000000*i)%m))%m > b)
        {
            cout << 1 << " " << setfill('0') << setw(9) << i << endl;
            return 0;
        }
    }
    cout << 2 << endl;
}
