#include <cstdio>
typedef long long ll;
int main()
{
    int a, b, c;
    ll j;
    scanf("%d%d%d", &a, &b, &c);
    for (int i = 1; i < c && i <= a; i++)
    {
        j = ll(1e9 * i) / c;
        j = (ll(1e9*i) % c == 0) ? 0 : j * c + c - ll(1e9*i);
        if (j > b) { printf("1 %09d\n", i); return 0;}
    }
    printf("2\n");
}
