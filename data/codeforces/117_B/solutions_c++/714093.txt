#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
#define LL long long
const LL cc = 1000000000;
int main()
{
	LL a,b,mod;
	cin >> a >> b >> mod;
	for (int f=0;f<mod && f<=a;f++)
	{
		LL t=(mod - (f*cc % mod)) % mod;
		if (t>b){
			printf("1\n");
			printf("%09d\n",f);
			return 0;
		}
	}
	printf("2\n");
	return 0;
}
