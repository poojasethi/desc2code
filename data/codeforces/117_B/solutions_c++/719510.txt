#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
  long long a, b, m;
  cin >> a >> b >> m;
  for (long long i = 0; i <= a && i < m; i++) {
    long long x = -1000000000LL * i;
    const long long r = ((x % m) + m) % m;
    if (r > b) {
      printf("1 %09lld\n", i);
      return 0;
    }
  }
  puts("2");
  return 0;
}
