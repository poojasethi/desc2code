#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

const long long k = 1000000000LL;
long long a, b, mod;

int main() {
	cin >> a >> b >> mod;
	long long m = min(a, mod);
	for (int i = 1; i <= m; ++i) {
		long long r = i * k % mod;
		if (r && mod - r > b) {
			printf("1 %09d\n", i);
			return 0;
		}
	}
	puts("2");
}
