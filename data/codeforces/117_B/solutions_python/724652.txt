import itertools as it
from sys import stdin

def solve(a, b, mod):
    if b >= mod: return 2

    m = 10**9 % mod
    s = 0
    for i in xrange(1, a+1):
        s += m
        if s >= mod: s -= mod
        if s == 0: break
        if s and s + b < mod:
            return 1, i
    return 2

def main():
    for l in stdin:
        a, b, mod = map(int, l.split())
        winner = solve(a, b, mod)
        if winner == 2:
            print 2
        else:
            print 1, '%09d' % winner[1]
