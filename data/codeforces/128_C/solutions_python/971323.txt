n,m,k=map(int,raw_input().split())
n-=1
m-=1
tot=max(m,n)+2
C=[]
for i in range (0,tot):
    C.append([0]*tot)
mod=1000000007
for i in range (0,tot):
    C[i][0]=1
    for j in range(1,i+1):
        C[i][j]=C[i-1][j-1]+C[i-1][j]
        if(C[i][j]>=mod):
            C[i][j]-=mod
res=1
if(2*k<=n):
    res=res*C[n][2*k]
    res=res%mod
else:
    res=0
if(2*k<=m):
    res=res*C[m][2*k]
    res=res%mod
else:
    res=0
print res
