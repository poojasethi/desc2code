#include <iostream>
using namespace std;

int n, m, k;
long long choose[1100][4100];
int main() {
	choose[0][0]=1;
	for (int i=1; i<1100; i++) {
		choose[i][0]=1;
		for (int j=1; j<=i; j++) 
			choose[i][j]=(choose[i-1][j-1]+choose[i-1][j])%1000000007;
	}
	cin>>n>>m>>k;
	cout<<(choose[n-1][2*k]*choose[m-1][2*k])%1000000007<<'\n';
	return 0;
}
