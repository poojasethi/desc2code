#include<cmath>
#include<iostream>

using namespace std;

long long int b[2020][2020];

int main()
{

	for(int i=0;i<2013;i++)
		b[i][0]=1;
    for(int i=1;i<=2013;i++)
	{
		for(int j=1;j<=2013;j++)
		{
			b[i][j]=(b[i-1][j]+b[i-1][j-1])%(1000000007);
		}
	}
	int n,m,k;
	cin >> n >> m >> k;

	cout << (b[n-1][2*k] * b[m-1][2*k])%1000000007;

	cin >> n;
	return 0;
}
