#include <bits/stdc++.h>
using namespace std;


const int mod = 1000000007;

long long C[1001][1001];

int main(){
	C[0][0] = 1;
	for(int i = 1;i <= 1000;i++){
		C[i][0] = 1;
		for(int j = 1;j <= i;j++)
			C[i][j] = (C[i - 1][j] + C[i - 1][j - 1])%mod;
	}
	long long n,m,k,ans;
	cin >> n >> m >> k;
	n--; m--;
	k <<= 1;
	if(k <= n && k <= m)
		ans = (C[n][k]*C[m][k])%mod;
	else
		ans = 0;
	cout << ans << "\n";
}
