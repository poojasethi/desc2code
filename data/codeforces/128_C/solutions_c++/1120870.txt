#include<cstdio>
#include<cstring>
int F[1024][1024],G[1024][1024],n,m,k,s;
#define P	1000000007
#define go(i,a,b) for(int i=a;i<=(b);++i)
#define DO(n,F) \
go(i,1,n)F[0][i]=1;\
go(i,1,k){s=0;\
	go(j,i*2+1,n){\
		(s+=F[i-1][j-2])%=P;\
		F[i][j]=(F[i][j-1]+s)%P;\
	}}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	DO(n,F);DO(m,G);
	printf("%d\n",(int)((long long)F[k][n]*G[k][m]%P));
	return 0;
}
