#include <iostream>
using namespace std;
typedef long long ll;

const int MOD = 1000000007;
const int MAXN = 1005;
int N, M, K;
int C[MAXN][2*MAXN];

int main()
{
	cin >> N >> M >> K;

	for (int i = 0; i < MAXN; i++)
		C[i][0] = C[i][i] = 1;
	for (int i = 1; i < MAXN; i++)
		for (int j = 1; j < i; j++)
			C[i][j] = (C[i-1][j] + C[i-1][j-1]) % MOD;

	cout << (ll)C[N-1][2*K]*C[M-1][2*K] % MOD << "\n";

	return 0;
}
