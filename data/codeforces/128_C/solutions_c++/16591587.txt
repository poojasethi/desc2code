#include <iostream>
using namespace std;

const int MAXN = 1000 + 10, MOD = 1000000007;
long long dp[MAXN][MAXN];

int main()
{
	int n, m, k;
	cin>> n >> m >> k;

	for(int i = 0; i < max(n, m); i ++)
		for(int j = 0; j <= i; j ++)
			if(j == 0)
				dp[i][j] = 1;
			else
				dp[i][j] = (dp[i - 1][j - 1] + dp[i - 1][j]) % MOD;

	long long ans = dp[n - 1][k * 2] * dp[m - 1][k * 2];
	cout<< ans % MOD << endl;

}
