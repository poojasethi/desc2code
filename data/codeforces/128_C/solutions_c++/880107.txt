#include<iostream>
#include<cstdio>
#include<cstring>

using namespace std;
const long long mod=1000000007ll;
const long long N=1000;
long long c[N*2+10][N*2+10];

int main()
{
    int i,j,n,m,k;
    for (i=0; i<=N; i++)
    {
        c[i][0]=1;
        for (j=1; j<=i; j++)
         c[i][j]=(c[i-1][j-1]+c[i-1][j])%mod;
    }
    cin>>n>>m>>k;
    cout<<(c[n-1][k*2]*c[m-1][k*2])%mod<<endl;
}
