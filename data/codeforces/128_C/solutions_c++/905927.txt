#include<cstdio>
int c[1010][1010],n,m,k,p=1000000007;
int main(){
    scanf("%d%d%d",&n,&m,&k);
    if(n-1<2*k||m-1<2*k){
        printf("0\n");
        return 0;
    }
    for(int i=1;i<=1000;++i){
        c[i][0]=c[i][i]=1; 
        for(int j=1;j<i;++j)
            c[i][j]=(c[i-1][j]+c[i-1][j-1])%p;
    }
    printf("%I64d\n",(long long)c[n-1][2*k]*c[m-1][2*k]%p); 
    return 0;
}
