#include<iostream>
using namespace std;
#define FOR(i,j,k) for(i=j;i<=k;i++)
#define N 1002
#define M 1000000007
int c[N][N];
int main()
{
    int i,j,n,m,k;
    c[0][0]=1;
    FOR(i,1,1000)
    {
        c[i][0]=1;c[i][i]=1;
        FOR(j,1,i-1)c[i][j]=(c[i-1][j]+c[i-1][j-1])%M;
    }
    cin>>n>>m>>k;
    if(2*k>n-1 || 2*k>m-1)cout<<0;else cout<<(c[n-1][2*k]*(unsigned long long)c[m-1][2*k])%M;
}
