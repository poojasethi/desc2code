#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;

const int N=1e3+10;
int dp[N][2*N];

int main() {
	int n, m, k;
	for(int i=0; i<N; i++) {
		dp[i][0]=1;
		for(int j=1; j<=i; j++) dp[i][j]=(dp[i-1][j]+dp[i-1][j-1])%MOD;
	}
	scanf("%d %d %d", &n, &m, &k);
	printf("%I64d\n", (1ll*dp[n-1][2*k]*dp[m-1][2*k])%MOD);
	return 0;
}
