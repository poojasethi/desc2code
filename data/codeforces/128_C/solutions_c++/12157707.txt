#include <bits/stdc++.h>
using namespace std;
int main(){
	long long int w[1005][1005],n,m,k,s;
    for(int i = 1 ; i < 1001 ; i++ ) w[i][0]=1;
    for(int j = 1 ; j < 1001 ; j++){
		s = 0 ;
		for( int i = 3 ; i < 1001 ; i++){
			s +=( w[i - 2 ][ j - 1 ]) % 1000000007 ;
			w[i][j] = ( w[i - 1][j] + s )  % 1000000007;
			}
	}
    cin>>n>>m>>k;
    cout<<(w[n][k]*w[m][k]) % 1000000007;
}
