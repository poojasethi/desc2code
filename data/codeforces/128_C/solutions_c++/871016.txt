# include <stdio.h>


# define MOD 1000000007 


long long dp[1010][2020] ;
int n, m, k ;


void fun()
{
	int i, j ; 
	dp[1][0] = 1 ;
	for (i = 2 ; i <= 1000 ; i++)
	{
		dp[i][0] = 1 ;
		for (j = 1 ; j <= 2000 ; j++)
			dp[i][j] = (dp[i-1][j] + dp[i-1][j-1]) % MOD ;
	}
}


int main ()
{
	fun() ;
	scanf ("%d%d%d", &n, &m, &k) ;
	printf ("%d\n", dp[n][k*2] * dp[m][k*2] % MOD) ;
	return 0 ;
}
