#include<iostream>
using namespace std;
const int maxn=3000,MOD=1000000000+7;
int c[maxn][maxn];
int main(){
	fill(c[0],c[0]+maxn,1);
	for(int i=0;i<maxn;i++)
		c[i][i]=1;
	for(int i=1;i<maxn;i++)
		for(int j=i+1;j<maxn;j++)
			c[i][j]=(c[i][j-1]+c[i-1][j-1])%MOD;
	int n,m,k;
	cin>>n>>m>>k;
	k*=2;
	cout<<((long long)c[k][n-1]*c[k][m-1])%MOD<<endl;
}
