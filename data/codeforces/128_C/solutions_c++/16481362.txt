#include <bits/stdc++.h>
using namespace std;

const int xn = 1e3 + 20, mod = 1e9 + 7;
long long n, m, k, co[xn][xn];

int main()
{
    cin >> n >> m >> k;
    for (int i = 0; i < max(n, m); i++)
        for (int j = 0; j <= i; j++)
            co[i][j] = (j ? (co[i - 1][j] + co[i - 1][j - 1]) % mod : 1);
    cout << (co[n - 1][2 * k] * co[m - 1][2 * k]) % mod;
    return 0;
}
