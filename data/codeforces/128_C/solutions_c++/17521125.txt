#include<cstdio>
#include<algorithm>
#include<iostream>
using namespace std;
int f[1100][2100];
int M=1000000007;
int main()
{
	for(int i=0;i<=1000;i++)
	{
		f[i][0]=1;
		for(int j=1;j<=i;j++)
		f[i][j]=(f[i-1][j]+f[i-1][j-1])%M;
	}
	int a,b,c;
	cin>>a>>b>>c;
	cout<<(1ll*f[a-1][2*c]*f[b-1][2*c])%M<<"\n";
	return 0;
}
