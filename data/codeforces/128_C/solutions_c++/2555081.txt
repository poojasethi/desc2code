#include<cstdio>
#include<set>
#include<vector>
#include<algorithm>
#include<iostream>
using namespace std;
int c[1010][2010];
int main()
{
	for(int i=0;i<=1000;i++)
	{
		c[i][0] = 1;
		for(int j=1;j<=i;j++)
		{
			c[i][j] = (c[i-1][j] + c[i-1][j-1])%1000000007;
		}
	}
	int n,m,k;
	cin>>n>>m>>k;
	printf("%I64d\n",1LL*c[n-1][2*k]*c[m-1][2*k]%1000000007);
	return 0;
}
