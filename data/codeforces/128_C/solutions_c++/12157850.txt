#include <bits/stdc++.h>
using namespace std;
int mod = 1e9+7;
int comb[2100][2100] = {};
int main(){
	for(int i = 0 ; i < 2100 ; i++){
		comb[0][i] = 1;
		comb[i][i] = 1;
	}
	for(int i = 1 ; i < 2100 ; i++)
		for(int j = i + 1 ; j < 2100 ; j++)
			comb[i][j] = (comb[i][j-1]+comb[i-1][j-1]) % mod;
	int n , m , k;
	cin >> n >> m >> k;
	cout<<((long long)comb[k*2][n-1] * comb[k*2][m-1]) % mod;
}
