#include <iostream>

using namespace std;

long C[2005][2005]={0};

int main()
{
 long n,m,k;
 long i,j;
 
 cin>>n>>m>>k;
 
 if(2*k>n-1||2*k>m-1)
  {
   cout<<0<<endl;
   return 0;
  }
 for(i=0;i<=2000;i++)
   for(j=1,C[i][0]=1;j<=i;j++)
     {
      C[i][j]=(C[i-1][j-1]+C[i-1][j])%1000000007;
     }
 cout<<(long long)C[n-1][2*k]*C[m-1][2*k]%1000000007<<endl;
 
 return 0;
}
