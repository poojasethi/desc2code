#include<iostream>
using namespace std;
const int mod = 1000000007;
int dp[1002][1002];
int main()
{
    int n,m,k;
    cin>>n>>m>>k;
    for(int i = 1;i < n || i < m; ++i)
    {
        dp[i][0] = dp[i][i] = 1;
        for(int j = 1; j < i; ++j)
            dp[i][j] = (dp[i-1][j-1] + dp[i-1][j]) % mod;
    }
    cout<< ((long long)dp[n-1][2*k] * dp[m-1][2*k]) % mod <<endl;
    return 0;
}
