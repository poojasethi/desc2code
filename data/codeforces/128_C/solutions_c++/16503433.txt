#include<iostream>

using namespace std;

const int maxn = 1000+10;
const long long mod = 1000000007;

long long dp[maxn][maxn];

int main(){
	int m,n,k,s;
	for(int i=1;i<maxn;i++)
		dp[i][0]=1;
	for(int j=1;j<maxn;j++){
		s=0;
		for(int i=3;i<maxn;i++){
			s=(s+dp[i-2][j-1])%mod;
			dp[i][j]=(dp[i-1][j]+s)%mod;
		}
	}
    cin >> n >> m >> k;
	cout<<(dp[n][k]*dp[m][k])%mod;
}
