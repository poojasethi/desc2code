#include<iostream>
#include<cstring>
#include<cstdlib>
#include<cstdio>

using namespace std;

const int mod=1000000007;
int c[2010][2010],n,m,k;

int main()
{
	c[0][1]=c[1][1]=1;
	for(int i=2;i<=2000;i++)
	{
		c[0][i]=1;
		for(int j=1;j<=i;j++)
			c[j][i]=(c[j][i-1]+c[j-1][i-1])%mod;
	}
	scanf("%d%d%d",&n,&m,&k);
	cout<<(long long)c[2*k][n-1]*c[2*k][m-1]%mod<<endl;
	return 0;
}
