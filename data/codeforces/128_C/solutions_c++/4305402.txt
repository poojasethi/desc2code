#include <iostream>
using namespace std;
const int MOD=1000*1000*1000+7;
const int N = 2001;
const int M = 2001;

int c[M][N];
int main()
{
	int n,m,mx,k;
	cin>>n>>m>>k;
	mx=m;
	if(m<n)mx=n;
	for(int i=0 ; i<=mx ; i++) c[i][0]=1;
	for(int i = 1 ; i<=mx ; i++)
		for(int j = 1 ; j<=i ; j++)
            c[i][j]=(c[i-1][j-1]+c[i-1][j])%MOD;
    cout << 1ull*c[m-1][2*k]*c[n-1][2*k] %MOD;
}
