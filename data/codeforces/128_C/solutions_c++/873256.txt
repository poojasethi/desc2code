#include<stdio.h>
#include<string.h>
#define ll long long
const int N=1010;
const int M=1000000007;
ll f[N][N],s,ans;
int n,m,k;
int main(){
    scanf("%d%d%d",&n,&m,&k);
    int i,j;
    for (i=1;i<N;i++) f[0][i]=1;
    for (i=1;i<N;i++){
        s=0;
        for (j=3;j<N;j++){
            s=(s+f[i-1][j-2])%M;
            f[i][j]=(s+f[i][j-1])%M;
        }
    }
    ans=f[k][n]*f[k][m]%M;
    printf("%lld\n",ans);
    return 0;
}
