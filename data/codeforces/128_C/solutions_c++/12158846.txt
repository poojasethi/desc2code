#include<bits/stdc++.h>
using namespace std;
const int mx=1000;
long long int dp[2*mx+10][2*mx+10];
int main(){
	int n,m,k;
	cin>>n>>m>>k;
	k*=2;
	for(int i=0;i<=mx;i++){
		dp[i][0]=1;
		for(int j=1;j<=i;j++){
			dp[i][j]=(dp[i-1][j]+dp[i-1][j-1])%1000000007;
		}
	}
	long long int ans;
	ans=(dp[n-1][k]*dp[m-1][k])%1000000007;
	cout<<ans;
}
