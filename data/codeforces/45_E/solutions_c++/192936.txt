#include <iostream>
#include <fstream>
#include <algorithm>

#define nmax 1100

using namespace std;

bool used[nmax];

string s1[nmax];
string s2[nmax];

int n;

int cnt1[256] = { };
int cnt2[256] = { };

int main()
{
	scanf("%d\n",&n);
	for (int i = 0; i < n; i ++)
	{
		cin >> s1[i];
		cnt1[s1[i][0]] ++;
	}
	for (int i = 0; i < n; i ++)
	{
		cin >> s2[i];
		cnt2[s2[i][0]] ++;
	}
	sort(s1,s1 + n);
	sort(s2,s2 + n);
	for (int i = 0; i < n; i ++)
	{
		bool found = false;
		for (int j = 0; j < n; j ++)
			if (!used[j])
				if((cnt2[s1[i][0]] > 0 && s2[j][0] == s1[i][0]) || (cnt1[s1[i][0]] > cnt2[s1[i][0]] && cnt2[s2[j][0]] > cnt1[s2[j][0]]))
				{
					used[j] = true;
					cnt2[s2[j][0]] --;
					if (i != 0) cout << ", "; 
					cout << s1[i] << " " << s2[j];
					found = true;
					break;
				}
		cnt1[s1[i][0]] --;
	}
	cout << endl;
	return 0;	
}
