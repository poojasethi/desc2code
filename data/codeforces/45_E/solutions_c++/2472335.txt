#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

int n, opt, cn[26], cs[26];
string name[101], surname[101];
bool used[101];

int main(){
	scanf("%d", &n);
	for (int i = 0; i < n; i++) cin >> name[i];
	for (int i = 0; i < n; i++) cin >> surname[i];
	sort(name, name + n); sort(surname, surname + n);
	for (int i = 0; i < n; i++) cn[name[i][0] - 'A']++, cs[surname[i][0] - 'A']++;
	for (int i = 0; i < 26; i++) opt += min(cn[i], cs[i]);
	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++) if (!used[j]){
			int cnt = 0; cn[name[i][0] - 'A']--; cs[surname[j][0] - 'A']--;
			for (int k = 0; k < 26; k++) cnt += min(cn[k], cs[k]);
			if (cnt != opt - (name[i][0] == surname[j][0])) cn[name[i][0] - 'A']++, cs[surname[j][0] - 'A']++;
			else{
				used[j] = true;
				if (name[i][0] == surname[j][0]) opt--;
				if (i) printf(", ");
				printf("%s %s", name[i].c_str(), surname[j].c_str());
				break;
			}
		}
}
