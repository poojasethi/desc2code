#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

int main()
{
  int N;scanf("%d",&N);
  static string A[100],B[100],C[100];
  for(int i=0;i<N;i++)cin>>A[i];
  for(int i=0;i<N;i++)cin>>B[i];
  sort(A,A+N);sort(B,B+N);
  static int X[100],Y[100];
  memset(X,0,sizeof(X));
  memset(Y,0,sizeof(Y));
  for(int i=0;i<N;i++)X[A[i][0]]++;
  for(int i=0;i<N;i++)Y[B[i][0]]++;
  static bool Flag[100];
  memset(Flag,0,sizeof(Flag));
  for(int i=0;i<N;i++)
    {
      int Ans=0;
      for(int j=65;j<=90;j++)
        Ans+=min(X[j],Y[j]);
      for(int j=0;j<N;j++)
        if(!Flag[j])
          {
            int New=(A[i][0]==B[j][0]);
            X[A[i][0]]--;Y[B[j][0]]--;
            for(int k=65;k<=90;k++)
              New+=min(X[k],Y[k]);
            if(New==Ans){C[i]=B[j];Flag[j]=true;break;}
            X[A[i][0]]++;Y[B[j][0]]++;
          }
    }
  for(int i=0;i<N;i++)
    {
      cout<<A[i]<<' '<<C[i];
      if(i==N-1)putchar('\n');else printf(", ");
    }
  return 0;
}
