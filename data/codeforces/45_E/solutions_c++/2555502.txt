#include<cstdio>
#include<string>
#include<algorithm>
#include<iostream>

using namespace std;

int i,j,k,n,bl[110],Cnt1[110],Cnt2[110];

string Name[110],SurName[110];

int main()
{
	cin>>n;
	for(i=0;i<n;i++){cin>>Name[i];Cnt1[Name[i][0]]++;}
	for(i=0;i<n;i++){cin>>SurName[i];Cnt2[SurName[i][0]]++;}
	sort(Name,Name+n);
	sort(SurName,SurName+n);
	for(i=0;i<n;i++)
	{
		if(Cnt1[Name[i][0]]>Cnt2[Name[i][0]])
		{
			for(j=0;j<n;j++)
				if(!bl[j]&&(SurName[j][0]==Name[i][0]||Cnt1[SurName[j][0]]<Cnt2[SurName[j][0]]))break;
		}
		else for(j=0;j<n;j++)if(!bl[j]&&SurName[j][0]==Name[i][0])break;
		bl[j]=1;
		Cnt1[Name[i][0]]--;
		Cnt2[SurName[j][0]]--;
		cout<<Name[i]<<' '<<SurName[j];
		if(i==n-1)cout<<endl;
			else cout<<", ";
	}
	return 0;
}
