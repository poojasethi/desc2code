#include<iostream>
#include<algorithm>
#include<queue>
#include<string>
#include<cstdio>
#include<cstdlib>
#include<cstring>

using namespace std;

const int fin=0,maxn=103;
int n;
string a[maxn],b[maxn];
queue<int>Q[26];
int c[26],d[26],e[26],f[maxn];

int main(){
  if(fin){
    freopen("t.in","r",stdin);
    freopen("t.out","w",stdout);
  }
  int i,t,j;
  scanf("%d\n",&n);
  for(i=0;i<n;++i)cin>>a[i],++c[a[i][0]-'A'];
  for(i=0;i<n;++i)cin>>b[i],++d[b[i][0]-'A'];
  sort(a,a+n),sort(b,b+n);
  //for(i=0;i<26;++i)cout<<a[i]<<" ";cout<<endl;
  //for(i=0;i<26;++i)cout<<b[i]<<" ";cout<<endl;
  for(i=0;i<n;++i)
    Q[b[i][0]-'A'].push(i);
  for(i=0;i<n;++i){
    if(t=a[i][0]-'A',c[t]>d[t]){
      for(j=0;j<26;++j)
        if(d[j]!=0&&(t==j||c[j]<d[j]))
          break;
    }else j=t;
    f[i]=Q[j].front(),Q[j].pop();
    --c[t],--d[j];
  }
  cout<<a[0]<<" "<<b[f[0]];
  for(i=1;i<n;++i)
    cout<<", "<<a[i]<<" "<<b[f[i]];
  return 0;
}
