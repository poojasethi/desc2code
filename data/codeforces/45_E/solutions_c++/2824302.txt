#include <cstring>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <ctime>
using namespace std;
class str{public:char s[200];}s1[200],s2[200];
int v1[200],v2[200],p[200];
bool b[200];
int N,l1,l2,k,i,j;
bool cmp(const str &a,const str &b){
    l1=strlen(a.s),l2=strlen(b.s);
    for (i=0;i<min(l1,l2);i++)if (a.s[i]!=b.s[i])return a.s[i]<b.s[i];
    return l1<l2;}
int main(){
    scanf("%d\n",&N);
    for (i=1;i<=N;i++)scanf("%s",s1[i].s),v1[s1[i].s[0]]++;
    for (i=1;i<=N;i++)scanf("%s",s2[i].s),v2[s2[i].s[0]]++;
    sort(s1+1,s1+N+1,cmp);
    sort(s2+1,s2+N+1,cmp);
    for (i=1;i<=N;i++){
        for (j=1;j<=N;j++)
        if (!b[j])
        if (((v2[s1[i].s[0]]<v1[s1[i].s[0]])&&(v2[s2[j].s[0]]>v1[s2[j].s[0]]))||(s1[i].s[0]==s2[j].s[0]))
        {k=j;break;}
        b[k]=true;p[i]=k;v2[s2[k].s[0]]--;v1[s1[i].s[0]]--;}
    for (i=1;i<=N;i++){
        l1=strlen(s1[i].s);l2=strlen(s2[p[i]].s);
        for (j=0;j<l1;j++)printf("%c",s1[i].s[j]);printf(" ");
        for (j=0;j<l2;j++)printf("%c",s2[p[i]].s[j]);if (i<N)printf(", ");}
}
