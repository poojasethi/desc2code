#include <iostream>
#include <algorithm>
using namespace std;

int N,sa[256],sb[256];
bool f[105];
string a[105],b[105];

int main()
{
	cin>>N;
	for (int i=1; i<=N; i++) cin>>a[i],sa[a[i][0]]++;
	for (int i=1; i<=N; i++) cin>>b[i],sb[b[i][0]]++;
	sort(a+1,a+N+1),sort(b+1,b+N+1);
	for (int i=1; i<=N; i++)
	{
		for (int j=1; j<=N; j++) if (!f[j]&&(a[i][0]==b[j][0]||(sa[a[i][0]]>sb[a[i][0]]&&sb[b[j][0]]>sa[b[j][0]])))
		{
			cout<<a[i]<<' '<<b[j],sa[a[i][0]]--,sb[b[j][0]]--,f[j]=1;
			break;
		}
		if (i<N) cout<<", ";
	}
}
