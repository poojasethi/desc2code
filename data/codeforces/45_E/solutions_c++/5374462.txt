#include <cstdio>
#include <algorithm>
#include <string>
#include <iostream>
using namespace std;
string a[102],b[102];
int nu1[102],nu2[102],n,k,j;
bool v[102];
int main()
{
    scanf("%d\n",&n);
    for (int i=1;i<=n;i++) cin>>a[i];
    for (int i=1;i<=n;i++) cin>>b[i];
    sort(b+1,b+n+1);sort(a+1,a+n+1);
    for (int i=1;i<=n;i++) nu1[a[i][0]-'A'+1]++;
    for (int i=1;i<=n;i++) nu2[b[i][0]-'A'+1]++;
    for (int i=1;i<=n;i++)
    for (int j=1;j<=n;j++)
    {
        if (!v[j]&&(a[i][0]==b[j][0]||
            (nu1[a[i][0]-'A'+1]>nu2[a[i][0]-'A'+1]&&nu1[b[j][0]-'A'+1]<nu2[b[j][0]-'A'+1])))
        {
            v[j]=1;nu1[a[i][0]-'A'+1]--;nu2[b[j][0]-'A'+1]--;
            if (i<n)cout << a[i] << " " << b[j] << ", ";
                else cout << a[i] << " " << b[j] <<endl;
            break;
        }
    }
    return 0;
}
