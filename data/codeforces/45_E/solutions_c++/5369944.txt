#include <cstdio>
#include <string>
#include <algorithm>
#include <iostream>

using namespace std;
string nam[101],sur[101];
int n,i,j,mat[101],a[101],node[101],next[101],t,s[101],S[101];

void add(int x,int y){
  node[++t]=y;
  next[t]=a[x];
  a[x]=t;
  s[x]++;
}

int main(){
  //freopen("45E.in","r",stdin);
  //freopen("45E.out","w",stdout);
  scanf("%d\n",&n);
  for (i=1;i<=n;i++) cin >> nam[i];
  for (i=1;i<=n;i++) cin >> sur[i];
  sort(nam+1,nam+n+1);
  sort(sur+1,sur+n+1);
  
  for (i=1;i<=n;i++) S[nam[i][0]-'A']++;
  for (i=n;i>0;i--) add(sur[i][0]-'A',i);
  for (i=1,j=0;i<=n;i++){
    while (!a[j] || S[j]>=s[j]) j++;
    if (a[nam[i][0]-'A'] && !(S[nam[i][0]-'A']>s[nam[i][0]-'A'] && j<nam[i][0]-'A')){
      mat[i]=node[a[nam[i][0]-'A']];
      a[nam[i][0]-'A']=next[a[nam[i][0]-'A']];
    } else
    {
      s[j]--;
      mat[i]=node[a[j]];
      a[j]=next[a[j]];
    }
    S[nam[i][0]-'A']--;
  }
  for (i=1;i<n;i++) cout << nam[i] << " " << sur[mat[i]] << ", ";
  cout << nam[n] << " " << sur[mat[n]] << endl;
  return 0;
}
