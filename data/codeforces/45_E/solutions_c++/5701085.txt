#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<iostream>
using namespace std;
string a[120],b[120];
int cnt[440],cnt1[440],cnt2[440],use[120],i,j,n,k;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i)cin>>a[i],cnt1[a[i][0]]++;
	for(i=1;i<=n;++i)cin>>b[i],cnt2[b[i][0]]++;
	for(i=1;i<=300;++i)cnt[i]=cnt2[i]-min(cnt1[i],cnt2[i]),cnt1[i]-=min(cnt1[i],cnt2[i]);
	sort(a+1,a+n+1);
	sort(b+1,b+n+1);
	for(i=1;i<=n;++i){
		for(j=1;j<=n;++j)if(!use[j] && a[i][0]==b[j][0])break;
		for(k=1;k<=n;++k)if(!use[k] && cnt[b[k][0]] && cnt1[a[i][0]])break;
		if(k<j){
			j=k;
			cnt1[a[i][0]]--;
			cnt[b[j][0]]--;
		}
		use[j]=1;
		if(i!=1)printf(", ");
		cout<<a[i]<<' '<<b[j];
	}
}
