#include<stdio.h>
#include<string>
#include<algorithm>
using namespace std;
int n;
int ca[200],cb[200];
int la[200];
bool vb[200];
string a[200],b[200];
int main()
{
	scanf("%d",&n);
	char buf[30];
	for(int i=1;i<=n;i++)scanf("%s",buf),a[i]=string(buf);
	for(int i=1;i<=n;i++)scanf("%s",buf),b[i]=string(buf);
	sort(a+1,a+n+1);
	sort(b+1,b+n+1);
	for(int i=1;i<=n;i++)ca[a[i][0]]++,cb[b[i][0]]++;
	for(int i=1;i<=n;i++)
	{
		int j;
		for(j=1;j<=n;j++)if(!vb[j]&&(a[i][0]==b[j][0]||(ca[a[i][0]]>cb[a[i][0]]&&ca[b[j][0]]<cb[b[j][0]])))break;
		la[i]=j;
		vb[j]=1;
		ca[a[i][0]]--;
		cb[b[j][0]]--;
	}
	printf("%s %s",a[1].c_str(),b[la[1]].c_str());
	for(int i=2;i<=n;i++)printf(", %s %s",a[i].c_str(),b[la[i]].c_str());
	putchar('\n');
	return 0;
}
