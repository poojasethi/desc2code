#include<cstdio>
#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
const int maxn=205;
string a[maxn],b[maxn];
int n,as[maxn],bs[maxn],v[maxn];
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
      cin>>a[i];
      as[a[i][0]]++;
    }
    for(int i=1;i<=n;i++)
    {
      cin>>b[i];
      bs[b[i][0]]++;
    }
    sort(a+1,a+n+1);
    sort(b+1,b+n+1);
    for(int i=1;i<=n;i++)
    {
      int j;
      for(j=1;j<=n;j++)
        if(!v[j]&&(bs[b[j][0]]>as[b[j][0]]||a[i][0]==b[j][0])&&(b[j][0]>=a[i][0]||as[a[i][0]]>bs[a[i][0]]))break;
      v[j]=1;
      as[a[i][0]]--;
      bs[b[j][0]]--;
      if(i>1)printf(", ");
      cout<<a[i]<<" "<<b[j];
    }
    return 0;
}
