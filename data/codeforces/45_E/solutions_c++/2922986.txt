#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <string>
using namespace std;

const int maxn = 105;
string a[maxn], b[maxn];
int sa[maxn], sb[maxn];
int match[maxn], used[maxn];
int n;

int main() {
    int i, j;
    cin >> n;
    for (i = 1; i <= n; ++i)
        cin >> a[i], ++sa[a[i][0]];
    for (i = 1; i <= n; ++i)
        cin >> b[i], ++sb[b[i][0]];
    sort(a + 1, a + n + 1); sort(b + 1, b + n + 1);
    for (i = 1; i <= n; ++i) {
        for (j = 1; j <= n; ++j)
            if (!used[j]) {
                if (a[i][0] == b[j][0]) break;
                if (sa[a[i][0]] > sb[a[i][0]] && sb[b[j][0]] > sa[b[j][0]])
                    break;
            }
        match[i] = j; used[j] = true; --sa[a[i][0]]; --sb[b[j][0]];
    }
    for (i = 1; i < n; ++i)
        cout << a[i] << " " << b[match[i]] << ", ";
    cout << a[n] << " " << b[match[n]] << endl;
    return 0;
}
