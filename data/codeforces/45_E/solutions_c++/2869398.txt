#include<cstdio>
#include<cstring>
#define Rep(i,n) for(int i = 0; i<n; i++)
using namespace std;

bool    mark[2][105];
char    s[2][105][20];
int     n,num[2][26];

int     get(int x,int y){
    int ret = -1;
    Rep(i,n) if ( !mark[1][i] ){
        int k = s[1][i][0]-'A';
        if ( y==k || x==-1&&num[1][k]>num[0][k] )
            if ( ret == -1 || strcmp(s[1][i],s[1][ret])<0 ) ret = i;
    }
    return ret;
}

    scanf("%d", &n );
    Rep(k,2) Rep(i,n) {

        num[k][s[k][i][0]-'A']++;
    }
    Rep(tt,n) {
        int i = -1;
        Rep(j,n) 
            if ( !mark[0][j] && ( i == -1 || strcmp(s[0][j],s[0][i])<0 ) ) i = j;
        int k = s[0][i][0]-'A';
        printf("%s ", s[0][i] );
        int j;
//      printf("%d %d %d\n" , k , num[1][k] , num[0][k] );
        if ( num[1][k] >= num[0][k] ) j = get( 0,k );
        else j = get(-1,k);
        printf("%s", s[1][j] );
        if ( tt < n-1 ) printf(", "); else printf("\n");
        mark[1][j] = true;
        mark[0][i] = true;
        num[1][s[1][j][0]-'A']--;
        num[0][k]--;
    }
}
