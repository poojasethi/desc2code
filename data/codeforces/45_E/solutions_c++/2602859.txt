#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
string a[111],b[111];
int sa[111],sb[111],c[111],n;
bool v[111];
int main(){
    //freopen("AC.in","r",stdin);
    //freopen("AC.out","w",stdout);
	cin>>n;
	for(int i=1;i<=n;i++){
        cin>>a[i];
        sa[a[i][0]]++;
    }
	for(int i=1;i<=n;i++){
        cin>>b[i];
        sb[b[i][0]]++;
    }
	sort(a+1,a+n+1);
	sort(b+1,b+n+1);
	for(int i=1;i<=n;i++){
        //int j;
        for(int j=1;j<=n;j++)
            if(!v[j]&&(a[i][0]==b[j][0]||sa[a[i][0]]>sb[a[i][0]]&&sb[b[j][0]]>sa[b[j][0]])){
                c[i]=j;
                v[j]=1;
                sa[a[i][0]]--;
                sb[b[j][0]]--;
                break;
            }
    }
	for(int i=1;i<n;i++)
        cout<<a[i]<<" "<<b[c[i]]<<", ";
	cout<<a[n]<<" "<<b[c[n]]<<endl;
	return 0;
}
