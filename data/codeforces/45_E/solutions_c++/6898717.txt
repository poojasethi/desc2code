#include <algorithm>
#include <iostream>
#include <cstdio>
#define N 120
#define K 500

using namespace std;

int n;
string ans;
string a[N], b[N];
int ac[N], bc[N];
bool used[N];

int main() {
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        ++ac[a[i][0]];
    }
    for (int i = 0; i < n; ++i) {
        cin >> b[i];
        ++bc[b[i][0]];
    }
    sort(a, a + n);
    sort(b, b + n);
    for (int i = 0; i < n; ++i) {
        int f = (int)a[i][0];
        if (bc[f] < ac[f]) {
            int r = 0;
            while (bc[r] <= ac[r] && (r < f || bc[f] == 0)) {
                ++r;
            }
            f = r;
        }
        --ac[a[i][0]];
        --bc[f];
        int t = 0;
        while (used[t] || b[t][0] != (char)f) {
            ++t;
            if (t >= n) {
                cerr << "t >= n\n";
                return 0;
            }
        }
        used[t] = true;
        ans += a[i] + " " + b[t];
        if (i < n - 1) {
            ans += ", ";
        }
    }
    cout << ans;
    return 0;
}
