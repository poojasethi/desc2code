#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
#define REP(i,a,b) for(i=a;i<b;i++)

vector< string > a,b,ans;
int ca[255],cb[255],vis[110];

int count(){
        int ret=0,i;
        REP(i,0,26) ret+=min(ca[i+'A'],cb[i+'A']);
        return ret;
}

int main(){
        int n,i,j;
        string s;
        cin>>n;
        a.resize(n);b.resize(n);ans.resize(n);
        REP(i,0,n) cin>>a[i];
        REP(i,0,n) cin>>b[i];
        sort(a.begin(),a.end());
        sort(b.begin(),b.end());
        REP(i,0,n) ca[a[i][0]]++,cb[b[i][0]]++;
        REP(i,0,n){
                int best=count();
                REP(j,0,n)if(!vis[j]){
                        ca[a[i][0]]--;cb[b[j][0]]--;
                        if(count()+(a[i][0]==b[j][0])==best) {
                                s=a[i]+' '+b[j];
                                vis[j]=1;
                                ans[i]=s;
                                break;
                        }
                        ca[a[i][0]]++;cb[b[j][0]]++;
                }
        }
        REP(i,0,n-1)cout<<ans[i]<<", ";
        cout<<ans[n-1]<<endl;
}
