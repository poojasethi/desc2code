#include<cstdio>
#include<string>
#include<algorithm>
#include<iostream>
using namespace std;
string s1[111],s2[111];
bool vis[111],p1[111],p2[111];
int cnt1[111],cnt2[111];
int n;
int getans(){
    int i,ans=0;
    for (i=0;i<26;i++) cnt1[i]=cnt2[i]=0;
    for (i=0;i<n;i++) if (p1[i]) cnt1[(*s1[i].begin())-'A']++;
    for (i=0;i<n;i++) if (p2[i]) cnt2[(*s2[i].begin())-'A']++;
    for (i=0;i<26;i++) ans+=min(cnt1[i],cnt2[i]);
    return ans;
}
int main(){
//  freopen("45E.in","r",stdin);
//  freopen("45E.out","w",stdout);
    int i,j,ans;
    scanf("%d",&n);
    for (i=0;i<n;i++) cin>>s1[i];
    for (i=0;i<n;i++) cin>>s2[i];
    sort(s1,s1+n);
    sort(s2,s2+n);
    for (i=0;i<n;i++) p1[i]=p2[i]=true;
    ans=getans();
    for (i=0;i<n;i++){
        p1[i]=false;
        for (j=0;j<n;j++) if (p2[j]){
            p2[j]=false;
            if (getans()==ans-(*s1[i].begin()==*s2[j].begin())) break;
            p2[j]=true;
        }
        ans-=(*s1[i].begin()==*s2[j].begin());
        cout<<s1[i]<<' '<<s2[j];
        if (i!=n-1) printf(", "); else puts("");
    }
    return 0;
}
