#include <bits/stdc++.h>
#define rep(i,j,k)	for (int i = (j); i <= (k); i++)
#define dep(i,j,k)	for (int i = (j); i >= (k); i--)
using namespace std;
const int N = 111;
typedef long long ll;
string s1[N], s2[N];
int n, v[N], t1[N], t2[N];
int main()
{
	cin >> n;
	rep(i,1,n)	cin >> s1[i], t1[s1[i][0]]++;
	rep(i,1,n)	cin >> s2[i], t2[s2[i][0]]++;
	sort(s1+1,s1+1+n), sort(s2+1,s2+1+n);
	rep(i,1,n)	rep(j,1,n)	if (!v[j] && (s1[i][0]==s2[j][0] || (t1[s1[i][0]]>t2[s1[i][0]] && t1[s2[j][0]]<t2[s2[j][0]])))
	{
		v[j] = 1, t1[s1[i][0]]--, t2[s2[j][0]]--;
		if (i > 1)	cout << ", ";	cout << s1[i] << " " << s2[j];
		break;
	}
}
