#include<cstring>
#include<iostream>
#include<algorithm>
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=300;
string a[mn],b[mn];
bool vis[mn];
int x[mn],y[mn],g[mn];
int n,m,i,j,ans;
int calc()
{
    int i,res=0;
    fo(i,1,26)res+=min(x[i],y[i]);
    return res;
}
int main()
{
    cin>>n;
    fo(i,1,n)cin>>a[i];
    fo(i,1,n)cin>>b[i];
    sort(a+1,a+1+n);
    sort(b+1,b+1+n);
    fo(i,1,n)++x[a[i][0]-64];
    fo(i,1,n)++y[b[i][0]-64];
    fo(i,1,n)
    {
        ans=calc();
        --x[a[i][0]-64];
        fo(j,1,n)
        {
            if(vis[j])continue;
            --y[b[j][0]-64];
            if((calc()+(a[i][0]==b[j][0]))==ans)
            {
                vis[j]=1,g[i]=j;
                break;
            }
            ++y[b[j][0]-64];
        }
    }
    fo(i,1,n-1)cout<<a[i]<<' '<<b[g[i]]<<", ";
    cout<<a[n]<<' '<<b[g[n]]<<endl;
    return 0;
}
