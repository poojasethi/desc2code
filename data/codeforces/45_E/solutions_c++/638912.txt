#define li		 int
#define rep(i,to)	for(li i=0;i<((li)to);i++)
#define pb		push_back
#define sz(v)		((li)v.size())

#define MAX 105
li used[MAX];
int main(){
	rep(i,MAX) used[i]=false;
	li n;
	cin>>n;
	vector<string> a(n),b(n);
	rep(i,n) cin>>a[i];
	rep(i,n) cin>>b[i];
	sort(a.begin(),a.end());
	sort(b.begin(),b.end());
	li cnt[2][26];
	rep(i,2)rep(j,26) cnt[i][j]=0;
	rep(i,n) cnt[0][a[i][0]-'A']++;
	rep(i,n) cnt[1][b[i][0]-'A']++;
	rep(i,n){
		if(i) cout<<", ";
		cout<<a[i]<<" ";
		rep(j,n)if(a[i][0]==b[j][0] || (cnt[0][a[i][0]-'A']>cnt[1][a[i][0]-'A'] && cnt[1][b[j][0]-'A']>cnt[0][b[j][0]-'A'])){
			if(used[j]) continue;
			used[j]=true;
			cout<<b[j];
			cnt[0][a[i][0]-'A']--;
			cnt[1][b[j][0]-'A']--;
			break;
			
		}
	}
}
