#include <cstdio>

int build(){
	int ans=0;
	char c=getchar();
	if(c=='0')
		ans=1<<0;
	else
	if(c=='1')
		ans=1<<1;
	else
	if(c=='?')
		ans=1<<2;
	else
	if(c=='('){
		int l=build();
		char opt=getchar();
		int r=build();
		getchar();
		if(opt=='|')
			ans=( (l>>0&1)&(r>>0&1) )<<0|( (l>>1&1)|(r>>1&1)|(l>>2&1)&(r>>2&1) )<<1|( (l>>0&1)&(r>>2&1)|(l>>2&1)&(r>>0&1)|(l>>2&1)&(r>>2&1) )<<2;
		else
		if(opt=='&')
			ans=( (l>>0&1)|(r>>0&1)|(l>>2&1)&(r>>2&1) )<<0|( (l>>1&1)&(r>>1&1) )<<1|( (l>>1&1)&(r>>2&1)|(l>>2&1)&(r>>1&1)|(l>>2&1)&(r>>2&1) )<<2;
		else
		if(opt=='^')
			ans=( (l>>0&1)&(r>>0&1)|(l>>1&1)&(r>>1&1)|(l>>2&1)&(r>>2&1) )<<0|( (l>>0&1)&(r>>1&1)|(l>>1&1)&(r>>0&1)|(l>>2&1)&(r>>2&1) )<<1|( ((l>>0&1)|(l>>1&1))&(r>>2&1)|(l>>2&1)&((r>>0&1)|(r>>1&1)) )<<2;
		else
			fprintf(stderr,"err opt=%c\n",opt);
	}
	else
		fprintf(stderr,"err c=%c\n",c);
	return ans;
}

int main(){
	scanf("%*d\n");
	puts(build()>>2&1?"YES":"NO");
}
