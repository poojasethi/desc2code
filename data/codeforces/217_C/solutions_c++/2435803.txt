#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int work()
{
	char c;
	c = getchar();
	if (c == '(')
	{
		int _ = 0;
		int __ = work();
		c = getchar();
		int ___ = work();
		for (int i = 0; i < 16; ++i)
			for (int j = 0; j < 16; ++j)
				if (((__ >> i) & 1) && ((___ >> j) & 1))
					switch (c)
					{
						case '&': _ |= 1 << (i & j); break;
						case '|': _ |= 1 << (i | j); break;
						case '^': _ |= 1 << (i ^ j); break;
					}
		getchar();
		return _;
	}
	else if (c == '?')
		return (1 << 3) | (1 << 5);
	else if (c == '1')
		return 1 << 15;
	else if (c == '0')
		return 1;
	return -1;
}

bool check(int _)
{
	for (int i = 0; i < 16; ++i)
		if (((_ >> i) & 1) && (((i >> 1) & 3) == 1 || ((i >> 1) & 3) == 2))
			return 1;
	return 0;
}

int main()
{
	char debuff[99];
	gets(debuff);
	int _ = work();
	//~ printf("%d\n", _);
	puts(check(_) ? "YES" : "NO");
	return 0;
}
