#include <cstdio>

using namespace std;

#define gch() getchar()

int S(int x, int y, char ch){
	return ch == '&' ? (x & y) : (ch == '|' ? (x | y) : (x ^ y));
}

int solve()
{
	char ch = gch();
	if (ch == '0') return 1;//0001
	if (ch == '1') return 8;//1000
	if (ch == '?') return 6;//0110
	int x = solve(); ch = gch(); int y = solve(); gch();
	int ret = 0;
	for (int i = 0; i < 4; i ++) if (x & (1 << i))
		for (int j = 0; j < 4; j ++) if (y & (1 << j))
			ret |= 1 << S(i, j, ch);
	return ret;
}

int main()
{
	scanf("%*d"); gch();
	puts(solve() & 6 ? "YES" : "NO");
	return 0;
}
