int rec() {
	char c = getchar();
	if (c == '0')
		return  1 << 0;
	if (c == '1')
		return 1 << 3;
	if (c == '?')
		return (1 << 1) | (1 << 2);
	int x = rec();
	char op = getchar();
	int y = rec();
	getchar();

	int ret = 0;
	for (int i = 0; i < 4; i++) if (x & (1 << i))
	for (int j = 0; j < 4; j++) if (y & (1 << j))
		if (op == '&') ret |= (1 << (i&j));
		else if (op == '^') ret |= (1 << (i^j));
		else if (op == '|') ret |= (1 << (i|j));
	return ret;
}
