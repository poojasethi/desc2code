#include<stdio.h>
int f(int a,int b,int c)
{
	if(c=='&')return a&b;
	if(c=='|')return a|b;
	return a^b;
}
int calc()
{
	int c=getchar();
	if(c=='0')return 1;
	if(c=='1')return 8;
	if(c=='?')return 6;
	int l=calc(),o=getchar(),r=calc();
	getchar();
	int ret=0;
	for(int i=0;i<4;i++)if(l&(1<<i))
		for(int j=0;j<4;j++)if(r&(1<<j))ret|=1<<f(i,j,o);
	return ret;
}
int main()
{
	scanf("%*d"),getchar();
	puts(calc()&6?"YES":"NO");
	return 0;
}
