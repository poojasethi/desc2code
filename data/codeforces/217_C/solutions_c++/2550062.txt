#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

const int A[3][3][3]={
{{1,1,1},{1,2,4},{1,4,5}},
{{1,2,4},{2,2,2},{4,2,6}},
{{1,2,4},{2,1,4},{4,4,3}}};

bool OK(char c)
{
  if(c=='0'||c=='1'||c=='?')return true;
  if(c=='&'||c=='|'||c=='^')return true;
  return c=='('||c==')';
}

int DP()
{
  char c;while(c=getchar(),!OK(c));
  if(c=='0')return 1;
  if(c=='1')return 2;
  if(c=='?')return 4;
  int L=DP();
  c=getchar();
  int R=DP();
  int Ans=0,Type;
  if(c=='&')Type=0;
  if(c=='|')Type=1;
  if(c=='^')Type=2;
  for(int i=0;i<3;i++)
    for(int j=0;j<3;j++)
      if((L&(1<<i))&&(R&(1<<j)))
        Ans|=A[Type][i][j];
  getchar();return Ans;
}

int main()
{
  int N;scanf("%d",&N);
  printf((DP()&4)?"YES\n":"NO\n");
  return 0;
}
