int opr(char ch, int a, int b)
{
	if (ch == '&')
		return a & b;
	if (ch == '|')
		return a | b;
	return a ^ b;
}

int solve()
{
	char ch;
	scanf("%c", &ch);
	if (ch == '0')
		return 1;
	if (ch == '1')
		return 1 << 3;
	if (ch == '?')
		return (1 << 1) | (1 << 2);
	int a = solve();
	scanf("%c", &ch);
	int b = solve();
	int CC = 0;
	for (int i = 0; i < 4; ++i)
		if (a & (1 << i))
			for (int j = 0; j < 4; ++j)
				if (b & (1 << j))
					CC |= 1 << opr(ch, i, j);
	getchar();
	return CC;
}
