#include <cstring>
using namespace std;
char s[1000005];
int n;
int g(int a,int b,char c){return c=='&'?(a&b):(c=='|'?(a|b):(a^b));}
int work()
{
    if (s[n]=='(')
        {
            n++;
            int L=work(); char p=s[n++]; int R=work(); n++;int ans=0;
            for (int i=0; i<4; i++) if (L&(1<<i)) for (int j=0; j<4; j++)
              if (R&(1<<j)) ans|=(1<<(g(i,j,p)));
            return ans;
        } else
          if (s[n]=='1') {n++; return 8;} else
            if (s[n]=='?') {n++; return 6;} else
            {
                n++;
                return 1;
            }
}
int main()
{
    scanf("%d",&n);
    n=0; scanf("%s",s);
    cout<<(work()&6?"YES":"NO");
    return 0;
}
