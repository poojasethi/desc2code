int op(int i,int j,char c)
{
    if(c=='&')
        return (i & j);
    if(c=='|')
        return (i | j);
    return (i ^ j);
}

    if(c=='?')
        return 6;
    l=calc();
    o=getchar();
    r=calc();
    getchar();
    for(i=0;i<4;i++)
        if(l&(1<<i))
            for(j=0;j<4;j++)
                if(r&(1<<j))

    return z;
}
