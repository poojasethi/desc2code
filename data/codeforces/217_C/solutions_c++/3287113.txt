#include<stdio.h>
#include<stdlib.h>
#include<string.h>
long work(){
     char c=getchar();
     long x,y,i,j,z=0;
     if (c=='0') return 1;
     if (c=='1') return 8;
     if (c=='?') return 6;
     x=work();
     c=getchar();
     y=work();
     for (i=0;i<4;i++)
         if (x&(1<<i))
             for (j=0;j<4;j++)
                 if (y&(1<<j))
                     if (c=='&') z|=1<<(i&j);
                     else if (c=='|') z|=1<<(i|j);
                     else if (c=='^') z|=1<<(i^j);
     getchar();
     return z;
}
int main(){
    scanf("%*d ");
    puts(work()&6?"YES":"NO");
    return 0;
}
