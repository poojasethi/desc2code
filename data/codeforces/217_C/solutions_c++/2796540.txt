#include <cstdio>
#include <cstring>

const int maxn = 1000010;
int n;

inline int eval(int a, char op, int b){
	if (op == '&') return a & b;
	if (op == '|') return a | b;
	return a ^ b;
}

int parse(){
	char c = getchar();
	switch (c){
		case '0' :{ return 1; break; }
		case '1' :{ return 8; break; }
		case '?' :{ return 6; break; }
	}
	int L, R, ret = 0;
	L = parse();
	c = getchar();
	R = parse();
	for (int i = 0; i < 4; i++)
		for (int j = 0; j < 4; j++)
			if ((L & (1 << i)) && (R & (1 << j))) ret |= 1 << eval(i, c, j);
	getchar();
	return ret;
}

int main(){
	scanf("%d\n", &n);
	if (parse() & 6) printf("YES"); else printf("NO");
}
