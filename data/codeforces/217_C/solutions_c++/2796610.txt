#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

int calc(int a, int b, char c) {
    if (c == '^') return a ^ b;
    else if (c == '&') return a & b;
    else return a | b;
}

int work() {
    int a, b, x, i, j; x = 0;
    char c = getchar();
    if (c == '?') return 6;
    else if (c == '0') return 1;
    else if (c == '1') return 8;
    a = work(); c = getchar();
    b = work(); getchar();
    for (i = 0; i < 4; ++i)
        for (j = 0; j < 4; ++j)
            if (((a >> i) & 1) && ((b >> j) & 1))
                x |= 1 << calc(i, j, c);
    return x;
}

int main() {
    scanf("%*d\n");
    puts(work() & 6 ? "YES" : "NO");
    return 0;
}
