int merge(int l,char o,int r) {
	int s=0;
	if(o=='|') {
		if( (l&1)&&(r&1) ) s|=1;
		if( ((l&2)||(r&2)) || ((l&4)&&(r&4)) ) s|=2;
		if( ((l&4)&&(r&5)) || ((l&5)&&(r&4)) ) s|=4;
	} else if(o=='&') {
		if( ((l&1)||(r&1)) || ((l&4)&&(r&4)) ) s|=1;
		if( (l&2)&&(r&2) ) s|=2;
		if( ((l&4)&&(r&6)) || ((l&6)&&(r&4)) ) s|=4;
	} else {
		if( ((l&4)&&(r&4)) || ((l&1)&&(r&1)) || ((l&2)&&(r&2)) ) s|=1;
		if( ((l&1)&&(r&2)) || ((l&2)&&(r&1)) || ((l&4)&&(r&4)) ) s|=2;
		if( ((l&4)&&(r&3)) || ((l&3)&&(r&4)) ) s|=4;
	}
	return s;
}

int get() {
	char c=getchar();
	if(c=='(') {
		int l=get();
		char op=getchar();
		int r=get();
		getchar();
		return merge(l,op,r);
	}
	return c=='0'?1 : c=='1'?2 : 4;
}
