def main():
    a, b, c = [int(x) for x in raw_input().split()]
    y = 0
    while True:
        temp = c - b * y
        if temp < 0:
            break
        if temp % a == 0:
            print "Yes"
            return
        y += 1
    print "No"

if __name__ == '__main__':
    main()
