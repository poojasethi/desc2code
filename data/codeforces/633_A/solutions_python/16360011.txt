from fractions import gcd
a,b,c=[int(x) for x in raw_input().split()]
tt=0
for i in range((c/a)+1):
    for j in range((c/b)+1):
        if i*a+j*b==c:
            print("Yes")
            tt=1
            break
        if i*a+j*b>c:
            break
    if tt==1:
        break
if tt==0:
    print("No")
