def gcd(a,b):
    if (a==0):
        return b
    else:
        return gcd(b%a,a)
a,b,c=map(int,raw_input().split())
k=0
if c%gcd(a,b)==0:
    for i in range(c+1):
        for j in range(c+1):
            if a*i+b*j==c:
                k+=1
                break
        else:
            continue
        break
    if (k==0):
        print "NO"
    else:
        print"YES"
else:
    print "NO"
