#! /usr/bin/python

[a, b, c] = [int(x) for x in raw_input().split(" ")]

dp = [0]*10001
dp[a] = 1
dp[b] = 1

yes = False
for i in range(10001):
    if i > c:
        break;
    if dp[i] == 1:
        if i+a < 10001:
            dp[i+a] = 1
        if i+b < 10001:
            dp[i+b] = 1
        if i==c or i+a==c or i+b==c:
            yes = True
            break
if yes:
    print("Yes")
else:
    print("No")
