line = raw_input().split()
d1 = int(line[0])
d2 = int(line[1])
s = int(line[2])
k = [None]*(s+1) 
k[0] = True

for i in range(len(k)):
	if(k[i]) == True:
		if i+d1 <= s and k[i+d1] == None:
			k[i+d1] = True
		if i+d2 <= s and k[i+d2] == None:
			k[i+d2] = True

if(k[s]):
	print 'Yes'
else:
	print 'No'
