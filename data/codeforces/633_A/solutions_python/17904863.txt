#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

a, b, c = ia()

while True:
    if c < 0:
        print 'No'
        break
    elif c % b == 0:
        print 'Yes'
        break
    c = c - a
