# -*- coding: utf-8 -*-

import itertools


if __name__ == '__main__':
    a, b, c = map(int, raw_input().split())

    max_x = c / a
    max_y = c / b
    possible = any(
        a*x + b*y == c
        for x, y
        in itertools.product(range(max_x+1), range(max_y+1))
    )
    if possible:
        result = 'Yes'
    else:
        result = 'No'
    print(str(result))
