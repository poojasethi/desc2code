s = raw_input()
s = s.split(' ')
a = int(s[0])
b = int(s[1])
c = int(s[2])

def find(a, b, c):
    i = 0
    while i * a <= c:
        j = 0
        while i * a + j * b <= c:
            if i * a + j * b == c:
                return 'Yes'
            j += 1
        i += 1
    return 'No'

print find(a, b, c)
