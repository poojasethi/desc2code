from time import time


a, b, c = map(int, raw_input().split())


def timeit(f):
    def wrapped(*args, **kwargs):
        start = time()
        result = f(*args, **kwargs)
        print '{0} s'.format(time() - start)
        return result
    return wrapped


def solve():
    summ = 0
    while summ <= c:
        if (c - summ) % b == 0:
            return 'Yes'
        summ += a
    return 'No'

print solve()
