#!/usr/bin/python

import sys
line = sys.stdin.readline()

numbers = line.split()

eb = int(numbers[0])
iv = int(numbers[1])
dm = int(numbers[2])
flag = 0
for numEb in range(0,dm/eb+1):
        if (dm-numEb*eb)%iv == 0:
                print "Yes",
                flag = 1
                break
for numIv in range(0,dm/iv+1):
        if flag == 1:
                break
        if (dm-numIv*iv)%eb == 0:
                print "Yes",
                flag = 1
                break
if flag == 0:
        print "No",
