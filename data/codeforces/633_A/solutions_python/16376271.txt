import sys

sys.setrecursionlimit(9999999)

a, b, c = map(int, raw_input().split())


class Memoize(object):
    cache = {}

    @staticmethod
    def memoizer(func):
        def memoizedFunc(*args):
            if args not in Memoize.cache:
                Memoize.cache[args] = func(*args)
            return Memoize.cache[args]

        return memoizedFunc

    @staticmethod
    def clear():
        Memoize.cache = {}


@Memoize.memoizer
def solve(a, b, c):
    if c < 0:
        return False
    if c % a == 0 or c % b == 0:
        return True
    return solve(a, b, c - a) or solve(a, b, c - b)


solvable = solve(a, b, c)
if solvable:
    print "Yes"
else:
    print "No"
