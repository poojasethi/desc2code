#include<cstdio>
#include<cstdlib>

using namespace std;

int A,B,N;

bool Check(int x)
{
	int left=A;
	for(int i=1;i<=N;i++)
	{
		left*=x;
		if(abs(left)>abs(B)) return 0;
	}
	if(left==B)return 1;
}

int main()
{
	scanf("%d%d%d",&A,&B,&N);
	for(int i=-20000;i<=20000;i++)
		if(Check(i)) {printf("%d",i); return 0;}
	printf("No solution");
	return 0;
}
