#include <iostream>
using namespace std;

int main() {

	int a,b,n,x,xs;
	cin >> a >> b >> n;

	for (int x = -1000;x<=1000;++x)
	{
		xs = 1;
		for (int i=1;i<=n;++i)
		{
			xs *= x;
			if (xs > 1000 || xs < -1000)
				break;
		}
		if (xs * a == b)
		{
			cout << x;
			return 0;
		}
	}

	cout << "No solution";

	return 0;
}
