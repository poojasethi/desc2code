#include <iostream>
#include <cmath>
#include <string>
using namespace std;
int main ()
{
    double a,b,c,d,i,j,n,m;
    cin>>a>>b>>n;
    m=0;
    for (i=2000;i>-2000;i--)
    {
        d=abs((a*pow(i,n)) - b);
        if (d<0.0000001){m++;c=i;break;}
    }
    if (m==0){cout<<"No solution"<<endl;}
    else if (m==1){cout<<c<<endl;}
    return 0;
}
