#include <stdio.h>
int main(){
    int A,B,n;
    scanf("%d%d%d",&A,&B,&n);
    for (int k=-2000;k<=2000;k++){
        int b=1;
        for (int j=1;j<=n;j++){
            if (b<10000)
                b*=k;
        }
        if (b*A==B){
            printf("%d",k);
            return 0;
        }
    }
    printf("No solution");
    return 0;
}
