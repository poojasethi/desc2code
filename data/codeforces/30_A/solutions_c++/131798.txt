#include<iostream>
#include<fstream>
using namespace std;
int main(){
	typedef long long ll;
	ll a,b,n;
	cin>>a>>b>>n;
	
	for(int i=-2000;i<=2000;i++){
		ll ans=1;
		for(int j=0;j<n;j++){
			if(ans*i>30000){
				goto hell;
			}
			ans*=i;
		}
		ans*=a;
		if(ans==b){
			cout<<i<<endl;
			return 0;
		}
hell:
		ans++;
	}
	cout<<"No solution"<<endl;
	return 0;
}
