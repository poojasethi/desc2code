#include<iostream>
#include<cmath>

using namespace std;

int main()
{
    double a, b, n;

    cin >> a >> b >> n;

    double i = 0;

    for(i = -2000; i < 2000; ++i)
        if(abs(a*pow(i, n) - b) < 0.00001)
           break;

    if(abs(a*pow(i, n) - b) < 0.00001)
       cout << i;
    else
       cout << "No solution";

    return 0;
}
