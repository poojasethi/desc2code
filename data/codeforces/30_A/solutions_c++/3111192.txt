#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>

using namespace std;

int main()
{
	int A,B,n;
	scanf("%d%d%d",&A,&B,&n);
	for(int X = -1000;X<=1000;X++)
	{
		if (fabs(A*pow((double)X,(double)n)-B)<1e-9)
		{
			printf("%d\n",X);
			return 0;
		}
	}
	printf("No solution\n");
	return 0;
}
