#include <iostream>
#include <cmath>
using namespace std;
int cifang(int x,int n)
{
	int k=1;
	for (int i=0;i<n;i++)
	{
		k*=x;
	}
	return k;
}

int main()
{
    int a,b,n;
    cin>>a>>b>>n;
    for (int i=(-abs(b));i<=(abs(b));i++)
    {
    	if (a*(cifang(i,n))==b) 
		{
			cout<<i;
			return 0;
		}
    }
    cout<<"No solution";
}
