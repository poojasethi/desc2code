#include <cmath>
#include <iostream>
using namespace std;

#define EPS 1e-09

int main(void)
{
	int A, B, n;
	bool flg=false;

	cin >> A >> B >> n;
	for (int X=-1000; X<=1000; ++X)
		if (fabs( A*pow((double)X,(double)n)-B ) < EPS)	{
			flg=true;
			cout << X << endl;
			break;
		}
	if (!flg)	cout << "No solution" << endl;
	return 0;
}
