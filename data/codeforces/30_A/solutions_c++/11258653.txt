#include <iostream>
#include <cmath>

using namespace std;

int main() {
    int a,b,n;
    cin >> a >> b >> n;
    for(int i = -1000;i <= 1000 ; i++){
        if(fabs(a*pow(i,n)-b)<0.01)
        {
            cout << i << endl;
            return 0;
        }
    }
    cout << "No solution" << endl;
    return 0;
}
