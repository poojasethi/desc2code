#include <iostream>
#include <cmath>
#include <cstdlib>
using namespace std;

int main()
{
	int A, B, n;
	cin >> A >> B >> n;
	/*
	A*X^n=B
	X^n = B/A
	n log X = log(B/A)
	logX = log(B/A) / n
	X=e^(log(B/A)/n)
	*/

	for(int x = -1000; x <= 1000; x++)
		if(abs(A * pow(x, n) - B) < 0.0001)
		{
			cout << x << endl;
			return 0;
		}
	cout << "No solution" << endl;
}
