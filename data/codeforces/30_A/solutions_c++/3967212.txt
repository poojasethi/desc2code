# include <stdio.h>

typedef long long llint;

llint a,b,n,k;

int main()
{
	scanf("%I64d %I64d %I64d",&a,&b,&n);
	
	if(b == 0.0)	{printf("0");return 0;}
	
	for(llint h=-1000; h<1001; h++)
	{
		llint k = h;
		
		for(int j=0; j<n-1; j++)
		{
			if(k*h < -1000  ||  k*h > 1000)	{k =-10001 ;break;}
			k *= h;
		}
		
		if(a*k == b)	{printf("%I64d",h); return 0;}
	}
	
	printf("No solution");
}
