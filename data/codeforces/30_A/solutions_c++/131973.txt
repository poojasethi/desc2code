#include<iostream>
#include<cmath>
#define ll long long
using namespace std;

int main()
{
   int A,B,n;
   cin>>A>>B>>n;
   
   for(int i=-1000;i<1001;++i)
   {
     if(fabs(A*pow((double)i,(double)n)-B)<1e-9)
     {
        cout<<i<<endl;
        return 0;
     }

   }
   cout<<"No solution"<<endl;          
}
