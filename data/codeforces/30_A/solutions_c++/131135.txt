#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int A, B, n;
	cin >> A >> B >> n;
	bool find = false;
	for (int x = -1000; !find && x <= 1000; x++)
	{
		if (fabs(A * pow((double)x, (double)n) - B) < 1e-9)
		{
			find = true;
			cout << x << endl;
		}
	}
	if (!find) cout << "No solution" << endl;
}
