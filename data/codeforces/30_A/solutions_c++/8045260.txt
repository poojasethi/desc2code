#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
    int a,b,n;
    while(scanf("%d%d%d",&a,&b,&n)==3)
    {
        int flag=0;
        for(int i=-1000;i<=1000;i++) if(fabs(a*pow(i,n)-b)<1e-6) {flag=1;printf("%d\n",i);break;}
        if(!flag) printf("No solution\n");
    }
    return 0;
}
