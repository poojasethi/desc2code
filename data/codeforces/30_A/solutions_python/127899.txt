#!/usr/bin/env python

from sys import stdin

def main():
   A, B, n = map(int, stdin.readline().split())
   for x in xrange(-10000, 100001):
      xn = x ** n
      if A * xn == B:
         print x
         return 0
      x += 1

   print 'No solution'   
   return 0

if __name__ == '__main__': main()
