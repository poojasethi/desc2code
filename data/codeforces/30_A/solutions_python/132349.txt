#!/usr/bin/env python

def main():
	a, b, n = [int(x) for x in raw_input().split()]
	flag = ans = 0

	for i in xrange(-1000,1001):
		lhs = a*(i**n)	
		rhs = b
		
		if not a and not b:
			ans = 5
		if lhs == rhs:
			flag = 1
			if a*b > 0:
				ans = abs(i)
			elif a*b < 0:
				ans = -abs(i)
				
	if flag:
		print ans
	else:
		print "No solution"
		
if __name__ == "__main__":
	main()
