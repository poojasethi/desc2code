A, B, n = [ int(x) for x in raw_input().split() ]
if B == 0:
    print "0"
elif A == 0:
    print "No solution"
elif B % A != 0:
    print "No solution"
else:
    D = abs(B / A) ** (1.0 / n)
    if abs(round(D) - D) < .000001:
        E = int(round(D))
        if B * A < 0:
            if n % 2 == 0: print "No solution"
            else: print -E
        else:
            print E
    else:
        print "No solution"
