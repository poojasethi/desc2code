a, b, n = map(int, raw_input().split())
# if n is even, only need to check +
# break early if we've overshot B
xs = range(1001)
for x in xs:
	if a * pow(x, n) == b:
		print x; exit()
	if n % 2 == 1 and a * pow(-x, n) == b:
		print -x; exit()
	elif abs(a * pow(x, n)) > abs(b):
		print 'No solution'; exit()
# got to the end
print 'No solution'; exit()
