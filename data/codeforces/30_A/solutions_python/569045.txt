#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout

NO = 'No solution'

def sign(x):
    return 1 if x>0 else -1 if x<0 else 0

def solve(a,b,n):
    if a == 0:
        if b == 0:
            # any number
            return 1
        else:
            return NO
    else:
        d,r = divmod(b,a)
        if r != 0:
            return NO
        if d<0 and n%2==0:
            return NO
        if d==0:
            return 0
        sign_d = sign(d)
        # need n-th root from d
        for i in range(1,abs(d)+1):
            x = sign_d*i
            if x ** n == d:
                return x
        return NO

def numbers_from_line(d=' '):
    return [int(s) for s in input.readline().strip().split(d) if len(s.strip())>0]

A = numbers_from_line()
a = A[0]
b = A[1]
n = A[2]

x = solve(a,b,n)
output.write('%s\n' % x)
