import math
A, B, n = map(float, raw_input().split())
try:
    X = math.copysign(math.exp(math.log(abs(B / A)) / n), B / A) if B / A > 0 or not int(n) % 2 is 0 else 1 / 0
    if abs(X - round(X) * 1.0) < 1e-10:
        print int(round(X))
    else:
        print 'No solution'
except:
    if B == 0.0:
        print 0
    else:
        print 'No solution'
