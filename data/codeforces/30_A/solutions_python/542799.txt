import math

s = raw_input().split()
a = int(s[0])
b = int(s[1])
x = int(s[2])

if a == 0:
	if b == 0:
		print 5
	else:
		print 'No solution'
	quit()
elif b == 0:
	print '0'
	quit()

y = b/a

if b%a != 0:
	print 'No solution'
	quit()

if y < 0 and x % 2 == 0:
	print 'No solution'
	quit()

n = math.exp((math.log(abs(b))-math.log(abs(a)))/x)

if y < 0:
	ans = -int(n+.5)
else:
	ans = int(n+.5)

if a*pow(ans, x) != b:
	print 'No solution'
else:
	print ans
