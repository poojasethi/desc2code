A, B, n = map(int, raw_input().split())
if A == 0 and B != 0:
    print "No solution"
    exit()
elif A == 0 and B == 0:
    print "1"
    exit()
elif B == 0 and A != 0:
    print "0"
    exit()
xn = B/A
for i in xrange(min(-xn, xn), max(-xn+1, xn+1)):
    if i**n == xn:
        print i
        exit()
