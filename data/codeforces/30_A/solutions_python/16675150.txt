a,b,n=map(int,raw_input().split(' '))
if((a==b==0)&(n>0)):
    print '0'
elif((a==0)&(b!=0)):
    print 'No solution'
elif((b==0)&(a!=0)&(n==0)):
    print 'No solution'
elif((b==0)&(a!=0)&(n!=0)):
    print '0'
else:
    t=b/a
    c=False
    for i in range(2001):
        if(((i-1000)**n)==t):
            c=True
            ans=i-1000
            break
    print ans if(c==True) else 'No solution'
