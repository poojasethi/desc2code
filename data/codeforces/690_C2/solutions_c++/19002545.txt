#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
int s;
vector<int> v[100005];
int Sech(int x, int p);
int main(void)
{
	int i, a, b, n, m;
	scanf("%d%d", &n, &m);
	for(i=1;i<=m;i++)
	{
		scanf("%d%d", &a, &b);
		v[a].push_back(b);
		v[b].push_back(a);
	}
	s = 0;
	printf("%d\n", Sech(1,0));
	return 0;
}

int Sech(int x, int p)
{
	int i, sum, max1, max2;
	sum = max1 = max2 = 0;
	for(i=0;i<v[x].size();i++)
	{
		if(v[x][i]==p)
			continue;
		sum = max(sum, Sech(v[x][i], x));
		if(s>max1)
		{
			max2 = max1;
			max1 = s;
		}
		else
			max2 = max(max2, s);
	}
	s = max1+1;
	sum = max(max1+max2, sum);
	return sum;
}
