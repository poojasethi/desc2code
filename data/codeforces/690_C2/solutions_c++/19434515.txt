#include<iostream>
#include<cstdio>
using namespace std;
struct node {int to,next;} a[200001];
int h[100001],dis[100001],cnt=0;
void dfs(int x,int y)
{
	int i,to;
	for(i=h[x];i;i=a[i].next)
	{
		to=a[i].to;
		if(to==y) continue;
		dis[to]=dis[x]+1;
		dfs(to,x);
	}
}
void add(int x,int y) {a[++cnt]=(node){y,h[x]};h[x]=cnt;}
int main()
{
	int n,m;
	int i;
	int x,y,maxx=0;
	
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;i++) {scanf("%d%d",&x,&y);add(x,y);add(y,x);}
	dfs(1,0);y=0;
	for(i=1;i<=n;i++) if(dis[i]>dis[y]) y=i;
	for(i=1;i<=n;i++) dis[i]=0;
	dfs(y,0);
	for(i=1;i<=n;i++) maxx=max(maxx,dis[i]);
	printf("%d",maxx);
}
