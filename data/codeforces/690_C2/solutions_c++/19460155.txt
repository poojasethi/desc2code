#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;
vector<int> G[100010];
int de,ma;
void dfs(int u,int fa,int deep){
	for(int i=0;i<G[u].size();i++){
		if(G[u][i]==fa)
			continue;
		if(deep+1>ma){
			ma=deep+1;
			de=G[u][i];
		}
		dfs(G[u][i],u,deep+1);
	}
}
int main(){
	int n,m,q,p;
	while(scanf("%d %d",&n,&m)!=EOF){
		for(int i=0;i<100010;i++){
			G[i].clear();
		}
		de=0;
		ma=0;
		for(int i=1;i<=m;i++){
			scanf("%d %d",&p,&q);
			G[p].push_back(q);
			G[q].push_back(p);
		}
		dfs(1,-1,0);
		ma=0;
		dfs(de,-1,0);
		printf("%d\n",ma);
	}
	return 0;
}
