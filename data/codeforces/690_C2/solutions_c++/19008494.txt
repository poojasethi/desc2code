const int maxn= 5e5;
vector<int> g[maxn];
int n,m,i,j,k,x,y,z,now,ans,p;
void dfs(int o,int fa,int s){
	if (s>=now){
		now= s;
		p= o;
	}
	int len= g[o].size(),i;
	for (i= 0;i<len;i++){
		int v= g[o][i];
		if (v==fa) continue;
		dfs(v,o,s+1);
	}
}
