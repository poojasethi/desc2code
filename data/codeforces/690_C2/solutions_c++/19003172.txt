#include <cstdio>
using namespace std;
int e[200050][2],h[100050],dis[100050],tot;

void ins(int a,int b) {
	tot++;
	e[tot][0]=b;
	e[tot][1]=h[a];
	h[a]=tot;
}
void dfs(int u) {
	//printf("%d\n",u);
	for (int p=h[u];p;p=e[p][1]) if (!dis[e[p][0]]) {
		dis[e[p][0]]=dis[u]+1;
		dfs(e[p][0]);
	}
}
int main() {
	int n,m,i,a,b,maxx,maxw;
	scanf("%d%d",&n,&m);
	for (i=1;i<=m;i++) {
		scanf("%d%d",&a,&b);
		ins(a,b); ins(b,a);
	}
	for (i=1;i<=n;i++) dis[i]=0; dis[1]=1;
	dfs(1);
	//for (i=1;i<=n;i++) printf("%d",dis[i]);
	for (i=1,maxx=0;i<=n;i++) if (dis[i]>maxx) maxx=dis[i],maxw=i;
	for (i=1;i<=n;i++) dis[i]=0; dis[maxw]=1;
	//printf("mw:%d\n",maxw); 
	dfs(maxw);
	for (i=1,maxx=0;i<=n;i++) if (dis[i]>maxx) maxx=dis[i],maxw=i;
	printf("%d",maxx-1);
}
