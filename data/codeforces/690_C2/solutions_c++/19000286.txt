#include<cstdio>
#include<vector>
using namespace std;

vector<int> e[100100];
int m,n,a,b,i,j,mxa,t;

int dfs(int n,int f){
    int mx=0;
    for(int i=0;i<e[n].size();i++){
        if(e[n][i]!=f){
            t=dfs(e[n][i],n);
            if(t+mx>mxa)mxa=t+mx;
            if(t>mx)mx=t;
        }
    }
    return mx+1;
}

int main(){
    scanf("%d %d",&m,&n);
    while(n--){
        scanf("%d %d",&a,&b);
        e[a].push_back(b);
        e[b].push_back(a);
    }
    mxa=0;
    dfs(1,1);
    printf("%d\n",mxa);
}
