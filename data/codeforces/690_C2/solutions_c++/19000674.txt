#include <bits/stdc++.h>
#define LL long long
#define pb push_back
using namespace std;
const int N = 100010;
int n, m, ans, pos;
vector<int> edge[N];

void dfs(int u, int fa, int dis) {
    if (dis > ans) {
        ans = dis; pos = u;
    }
    for (int i = 0; i < edge[u].size(); ++i) {
        int v = edge[u][i];
        if (v == fa) continue;
        dfs(v, u, dis + 1);
    }
}

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= m; ++i) {
        int u, v;
        scanf("%d%d", &u, &v);
        edge[u].pb(v);
        edge[v].pb(u);
    }
    ans = 0;
    dfs(1, -1, 0);
    dfs(pos, -1, 0);
    printf("%d\n", ans);
    return 0;
}
