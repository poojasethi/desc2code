#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <vector>
using namespace std;
int n,m;
vector<int>son[100010];
int MAX,p1,p2;
void dfs(int x,int fa,int d){
	if (MAX<d){
		MAX=d;
		p2=x;
	}
	for (int i=0;i<son[x].size();i++)
		if (son[x][i]!=fa)
			dfs(son[x][i],x,d+1);
}
int main(int argc, char const *argv[])
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++) son[i].clear();
	for (int i=0;i<m;i++){
		int u,v;
		scanf("%d%d",&u,&v);
		son[u].push_back(v);
		son[v].push_back(u);
	}
	MAX=0;
	dfs(1,0,0);
	MAX=0;
	p1=p2;
	dfs(p1,0,0);
	printf("%d\n",MAX );
	return 0;
}
