
#include"stdio.h"
#include"iostream"
#include "string.h"
#include"algorithm"
#include"math.h"
#include"vector"
using namespace std;
const int N = 100005;
vector <int> q[N];
int vis[N];
int ans,pos;
void dfs(int v,int sum)
{
	if(ans<sum)
	{
		ans=sum;
		pos=v;
	}
	if(vis[v])
	return ;
	vis[v]=1;
	for(int i=0;i<q[v].size();i++)
		if(!vis[q[v][i]])
			dfs(q[v][i],sum+1);
}
int main()
{
	int u,v;
	int n,m;
	ans=0;
	memset(vis,0,sizeof(vis));
	scanf("%d %d",&n,&m);
	for(int i=1;i<=m;i++)
	{
		scanf("%d %d",&u,&v);
		q[u].push_back(v);
		q[v].push_back(u);
	}
	dfs(2,0);
	memset(vis,0,sizeof(vis));
	dfs(pos,0);
	printf("%d\n",ans);
	return 0;
}
