#include <bits/stdc++.h>
using namespace std;


int n, a, b, maxd, fnode,m;
vector<int> G[112345];

void dfs(int c, int p, int d) {
    if (d > maxd) {
        maxd = d;
        fnode = c;
    }
    for(int i = 0; i < G[c].size(); i += 1) {
        if (G[c][i] != p) {
            dfs(G[c][i], c, d + 1);
        }
    }
}

int main() {
    cin >> n>>m;
    for(int i = 0; i < m; i += 1) {
        cin >> a >> b;
        G[a].push_back(b);
        G[b].push_back(a);
    }
    dfs(1, -1, 0);
    dfs(fnode, -1, 0);
    cout << maxd << endl;
    return 0;
}
