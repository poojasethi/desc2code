#include<bits/stdc++.h>
using namespace std;

typedef struct lnod {
     int info;
     lnod *next;   
}*nod;
 
int i,n,m,x,y,s,rs,a[100005];
nod lda[100005];
 
void add(int x,nod &y)
{
  nod p=new lnod; 
  p->info=x;
  p->next=y;
  y=p;  
}
 
void dfs(int x,int tata)
{
  for(nod p=lda[x];p;p=p->next)
  if(a[p->info]==0 && p->info!=tata) {a[p->info]=a[x]+1; dfs(p->info,x);  }
}
int main()
{
  ios_base::sync_with_stdio(0);

  cin>>n>>m;
   
  for(i=1;i<=m;i++)
  {
    cin>>x>>y;
    add(x,lda[y]);
    add(y,lda[x]);        
  }
   
  dfs(1,1); rs=0;
  for(i=1;i<=n;i++)
  if(a[i]>a[rs]) rs=i;
   
  memset(a,0,sizeof(a));
  dfs(rs,rs); rs=0;
  for(i=1;i<=n;i++) 
  if(a[i]>rs) rs=a[i];
     
  cout<<rs;  

 return 0;
}
