#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

typedef struct{
	int to,next;
}node;

bool vis[200200];
node nodes[200200];
int edges=0,head[200200],ans,a;

void add(int u,int v)
{
	nodes[edges].to=v;
	nodes[edges].next=head[u];
	head[u]=edges++;
}

void dfs(int k,int len)
{
	vis[k]=true;
	if(ans<len)
	{
		ans=len;
		a=k;
	}
	for(int i=head[k];i!=-1;i=nodes[i].next)
		if(!vis[nodes[i].to])
			dfs(nodes[i].to,len+1); 
}

int main()
{
	memset(head,-1,sizeof head);
	int n,m;
	scanf("%d %d",&n,&m);
	for(int i=1;i<=m;i++)
	{
		int u,v;
		scanf("%d %d",&u,&v);
		add(u,v);	add(v,u);
	}
	memset(vis,false,sizeof vis);
	ans=0;	dfs(1,0);
	memset(vis,false,sizeof vis);
	ans=0;	dfs(a,0);
	printf("%d",ans);
}
