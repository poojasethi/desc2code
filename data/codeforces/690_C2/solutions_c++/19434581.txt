#include<cstdio>
#include<cstring>
#define N 100005
int n,m,head[N],cnt,v[N],ta,ans;
struct edge
{
    int to,next;
}e[N<<1];
void add(int u,int v)
{
    e[++cnt]=(edge){v,head[u]};
    head[u]=cnt;
}
void dfs(int x,int s){
        if(v[x])return;
        v[x]=1;
        if(s>=ans){
            ans=s;
            ta=x;
        }
        for(int i=head[x];i;i=e[i].next)
                dfs(e[i].to,s+1);
}
int main(){
        scanf("%d%d",&n,&m);
        for(int i=1;i<=m;i++){
                int a,b;
                scanf("%d%d",&a,&b);
                add(a,b);
                add(b,a);
        }
        dfs(1,0);
        memset(v,0,sizeof v);
        dfs(ta,0);
        printf("%d",ans);
}
