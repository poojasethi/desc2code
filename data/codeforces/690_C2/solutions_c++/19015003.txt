#include<bits/stdc++.h>
using namespace std;

int n,m;
int x,y;
vector<int> v[500000];
int d[500000];
int d2[500000];
int mx,mx2;

void dfs(int x,int y,int z)
{
    d[x]=z;
    mx=max(mx,z);
    for(int i=0;i<v[x].size();i++)
        if(v[x][i]!=y)
            dfs(v[x][i],x,z+1);
}

void dfs2(int x,int y,int z)
{
    d2[x]=z;
    mx2=max(mx2,z);
    for(int i=0;i<v[x].size();i++)
        if(v[x][i]!=y)
            dfs2(v[x][i],x,z+1);
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cin>>n>>m;
    for(int i=1;i<n;i++)
    {
        cin>>x>>y;
        v[x].push_back(y);
        v[y].push_back(x);
    }
    dfs(1,1,0);
    for(int i=1;i<=n;i++)
        if(mx==d[i])
        {
            dfs2(i,i,0);
            break;
        }
    cout<<mx2<<"\n";
}
