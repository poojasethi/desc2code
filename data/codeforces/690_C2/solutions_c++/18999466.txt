#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;

struct edge{
	int x,nex;
}e[200010];
int first[200010],d[200010],mx,x,y,n,tot,m;

void add(int x,int y){
	e[++tot]=(edge){y,first[x]};
	first[x]=tot;
}
void dfs(int x,int y){
	d[x]=d[y]+1;
	for(int i=first[x];i;i=e[i].nex)
	if(e[i].x!=y) dfs(e[i].x,x);
}
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++){
		scanf("%d%d",&x,&y);
		add(x,y); add(y,x);
	}
	dfs(1,0);
	for(int i=1;i<=n;i++)
	if(d[i]>mx){
		mx=d[i]; x=i;
	}
	dfs(x,0);
	for(int i=1;i<=n;i++)
	mx=max(mx,d[i]);
	printf("%d\n",mx-1);
	return 0;
}
