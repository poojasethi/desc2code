#include <bits/stdc++.h>
using namespace std;
int n,m;
vector<int>G[105000];
bool vis[105000]; 
int len,tag;
void dfs(int dis,int now)
{
	vis[now]=true;
	bool ok=false;
	for(int i=0;i<G[now].size();i++)
	{
		if(!vis[G[now][i]]){dfs(dis+1,G[now][i]);ok=true;}
	}
	if(!ok)
	{
		if(dis>len)
		{
			len=dis;
			tag=now;
		}
		return;
	}
	return;
}

int main()
{
	while(~scanf("%d%d",&n,&m))
	{
		//printf("%d %d\n",n,m);
		for(int i=1;i<=n;i++)G[i].clear();
		int x,y;
		for(int i=1;i<n;i++)
		{
			scanf("%d%d",&x,&y);
			G[x].push_back(y);
			G[y].push_back(x);
		}
		memset(vis,false,sizeof(vis));
		len=0;tag=1;
		dfs(0,1);
		//printf("%d %d\n",tag,len);
		len=0;
		memset(vis,false,sizeof(vis));
		dfs(0,tag);
		printf("%d\n",len);
	}
	return 0;
}
