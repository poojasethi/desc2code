#include <cstdio>
#include <vector>
#include <cstring>
using namespace std;
int d[100050];
vector<int> g[100050];
int fa[100050];
int n,i,root,m;
void dfs(int now){
	if (d[now]>d[root]) root=now;
	for (int ii=0;ii<g[now].size();ii++) if (g[now][ii]!=fa[now]){
		fa[g[now][ii]]=now;
		d[g[now][ii]]=d[now]+1;
		dfs(g[now][ii]);
	}
}
int main()
{
	scanf("%d%d",&n,&m);
	while (m--){
		int x,y;scanf("%d%d",&x,&y);
		g[x].push_back(y);g[y].push_back(x);
	}
	memset(fa,0,sizeof(fa));
	dfs(1);
	memset(fa,0,sizeof(fa));
	d[root]=0;
	dfs(root);
	printf("%d\n",d[root]);
	return 0;
}
