#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int n,m,f[200000];
vector<int> a[200000];
queue<int> q;

int bfs(int u)
{
    for (int i=1;i<=n;i++)
        f[i]=0;
    f[u]=1;
    int ret=u;
    q.push(u);
    while (q.size())
    {
        int u=q.front();
        q.pop();
        for (int i=0;i<a[u].size();i++)
        {
            int v=a[u][i];
            if (f[v]) continue;
            f[v]=f[u]+1;
            if (f[v]>f[ret]) ret=v;
            q.push(v);
        }
    }
    return ret;
}

int main()
{
    cin >> n >> m;
    for (int i=1;i<=m;i++)
    {
        int u,v;
        cin >> u >> v;
        a[u].push_back(v);
        a[v].push_back(u);
    }
    int u=bfs(1);
    int v=bfs(u);
    cout << f[v]-1;
}
