#include <bits/stdc++.h>
using namespace std;
#define ii pair<int,int>
int n,m;
vector<int> a[100010];
bool c[100010];
ii bfs(int x){
	memset(c,0,sizeof(c));
	queue<ii> q;
	q.push(ii(0,x));
	int ans = 0,nd;
	while(!q.empty()){
		int mv = q.front().first;
		int u = q.front().second;
		q.pop();
		if(c[u]==1) continue;
		c[u] = 1;
		if(mv>ans){
			ans = mv;
			nd = u;
		}
		for(int i = 0 ; i < a[u].size(); i++) q.push(ii(mv+1,a[u][i]));
	}
	return ii(ans,nd);
}
int main(){
	//freopen("in.in","r",stdin);
	cin >> n >> m;
	for(int i = 1; i <= m; i++){
		int u,v; cin >> u >> v;
		a[u].push_back(v);
		a[v].push_back(u);
	}
	int s = bfs(1).second;
	cout << bfs(s).first;
}
