#include<stdio.h>
#include<string.h>
#include<vector>
using namespace std;
#define MAX 100005
bool vis[MAX];
int index,Max;
vector<int>v[MAX];
void solve(int x,int t)
{
	vis[x]=1;
	if(Max<t)Max=t,index=x;
	for(int i=0;i<v[x].size();i++)
	{
		if(!vis[v[x][i]])solve(v[x][i],t+1);
	}
}
int main()
{
	int m,n,a,b;
	while(~scanf("%d%d",&m,&n))
	{
		for(int i=0;i<=n;i++)v[i].clear();
		for(int i=0;i<n;i++)
		{
			scanf("%d%d",&a,&b);
			v[a].push_back(b);
			v[b].push_back(a);
		}
		Max=0;
		memset(vis,0,sizeof(vis));
		solve(1,0);
		Max=0;
		memset(vis,0,sizeof(vis));
		solve(index,0);
		printf("%d\n",Max);
	}
	return 0;
}
