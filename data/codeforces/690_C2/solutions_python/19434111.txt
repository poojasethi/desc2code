#include <cstdio>
#include <cmath>
#include <algorithm>
#include <cstring>
using namespace std;
const int maxn=200010;
const int mod=1000000007;
int ne[maxn],head[maxn],to[maxn],vis[maxn],cnt,rec,n,maxd,m;
void dfs(int x,int d)
{
    if(vis[x])return;
    if(d>=maxd)rec=x,maxd=d;
    vis[x]=1;
    for(int i=head[x];i;i=ne[i])
        dfs(to[i],d+1);
}
int main(void)
{
    scanf("%d%d",&n,&m);
    int a,b;
    for(int i=0;i<m;i++)
    {
        scanf("%d%d",&a,&b);
        to[++cnt]=b;
        ne[cnt]=head[a];
        head[a]=cnt;
        to[++cnt]=a;
        ne[cnt]=head[b];
        head[b]=cnt;
    }
    dfs(1,0);
    memset(vis,0,sizeof(vis));
    dfs(rec,0);
    printf("%d\n",maxd);
    return 0;
}
