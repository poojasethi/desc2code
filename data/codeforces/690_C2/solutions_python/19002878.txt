#include <bits/stdc++.h>
using namespace std;
#define N 100100
vector<int> g[N];
int maxx,k;
void dfs(int u,int deep,int fa){
	bool f=false;
	for(int i=0;i<g[u].size();i++){
		if(g[u][i]==fa) continue;
		dfs(g[u][i],deep+1,u);
		f=true;
	}
	if(f==false){
		if(maxx<deep){
			maxx=deep;
			k=u;
		}
	}
}
int main(){
	//freopen("in.txt","r",stdin);
	int n,m,a,b;
	while(~scanf("%d%d",&n,&m)){
		int ans=0;
		for(int i=1;i<n;i++){
			scanf("%d%d",&a,&b);
			g[a].push_back(b);
			g[b].push_back(a);
		}
		maxx=0;
		k=0;
		dfs(1,0,1);
		dfs(k,0,k);
		printf("%d\n",maxx);
		for(int i=1;i<=n;i++) g[i].clear();
	}
	return 0;
}
