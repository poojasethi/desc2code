#include<queue>
#include<cstdio>
#include<vector>
using namespace std;
bool f[100010];
int n,m,k,a,d[100010];
queue  <int> q;
vector <int> g[100010];
void bfs(int x){
	for(int i=1;i<=n;i++) d[i]=666666;
	q.push(x),d[x]=0;
	while(!q.empty()){
		int u=q.front();
		q.pop(),f[u]=0;
		for(int i=g[u].size()-1;i>=0;i--){
			int v=g[u][i];
			if(d[v]>d[u]+1){
				d[v]=d[u]+1;
				if(!f[v]){
					q.push(v);
					f[v]=1;
				}
			}
		}
	}
	for(int i=1;i<=n;i++)
		if(d[i]>a) a=d[i],k=i;
}
int main(){
	scanf("%d%d",&n,&m);
	for(int u,v,i=1;i<=m;i++){
		scanf("%d%d",&u,&v);
		g[u].push_back(v);
		g[v].push_back(u);
	}
	bfs(1);
	bfs(k);
	printf("%d",a);
	return 0;
}
