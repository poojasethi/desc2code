#include <stdio.h>
#include <iostream>
#include <cstring>
#include <vector>
using namespace std;

int n, m, mx = 0, p = -1;
vector<int> Tree[100005];

void dfs(int x, int d, int fa){
	if (d > mx){
		mx = d;
		p = x;
	}
	for (int i = 0; i < Tree[x].size(); ++i){

		if (Tree[x][i] != fa)
		{
			dfs(Tree[x][i], d + 1, x);
		}
	}
}

int main(){
	scanf("%d%d", &n, &m);
	for (int i = 1; i < n; ++i){
		int x, y; scanf("%d%d", &x, &y);
		Tree[x].push_back(y);
		Tree[y].push_back(x);
	}
	dfs(1, 0, 0);
	dfs(p, 0, 0);
	printf("%d\n", mx);
	return 0;
}
