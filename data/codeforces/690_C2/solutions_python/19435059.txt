#include <cstdio>
#include <string.h>
#include <string>
#include <vector>
#define maxn 100001
using namespace std;
vector<int> a[maxn];
int ans=0;
int last=0;
void DFS(int cur,int pre,int len){
    if(len>ans){
        ans=len;
        last=cur;
    }
    for(int i=0;i<a[cur].size();i++){
        int to=a[cur][i];
        if(to!=pre)
            DFS(to,cur,len+1);
    }
}
int branch(int n,int m){
    DFS(1,0,0);
    ans=0;
    DFS(last,0,0);
    return ans;
}
int main(){
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=0;i<m;++i){
        int x, y;
        scanf("%d%d",&x,&y);
        a[x].push_back(y);
        a[y].push_back(x);
    }
    printf("%d\n",branch(n,m));
    return 0;
}
