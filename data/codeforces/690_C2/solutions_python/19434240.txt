#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;
const int N=100001;
int n,m,ans,num;
vector<int> g[N];
int visit[N];
void dfs(int k,int dis)
{
	if(dis>ans){ans=dis;num=k;}
	for(int i=0;i<g[k].size();i++)
	    if(!visit[g[k][i]])
	    {
	    	visit[g[k][i]]=1;
	    	dfs(g[k][i],dis+1);
	    }
	return;
}
int main()
{
	scanf("%d%d",&n,&m);
	int u,v;
	for(int i=1;i<=m;i++)
	{
		scanf("%d%d",&u,&v);
		g[u].push_back(v);
		g[v].push_back(u);
	}
	num=1;ans=0;
	visit[1]=1;
	dfs(1,0);
	memset(visit,0,sizeof(visit));
	visit[num]=1;
	ans=0;
	dfs(num,0);
	printf("%d",ans);
	return 0;
}
