#include <iostream>
#include <cstdio>
#include <vector>
#include <cstring>

#define Max 100001

using namespace std;


int n,pos,ans;
int visited[Max];

vector<int>g[Max];

void dfs(int v,int c)
{
    int i;
    if(ans<c)
    {
        ans=c;
        pos=v;
    }
    if(visited[v] != 0)
        return;
    
    visited[v]=1;
    
    for(i=0;i<g[v].size();i++)
        if(visited[g[v][i]] == 0)
            dfs(g[v][i],c+1);
}


int main()
{
    int n,m,i;
    int u,v;
    
    ans=0;

    scanf("%d %d",&n,&m);
    
    for(i=1;i<=m;i++)
    {
        scanf("%d %d",&u,&v);
        g[u].push_back(v);
        g[v].push_back(u);
    }
    
    memset(visited,0,sizeof(visited));
    dfs(2,0);
    
    memset(visited,0,sizeof(visited));
    dfs(pos,0);
    
    printf("%d\n",ans);
    return 0;
}
