#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define fo(i,a,b) for(int i=a;i<=b;i++)
#define fd(i,a,b) for(int i=a;i>=b;i--)
#define maxn 100005
using namespace std;

int head[maxn],t[maxn * 2],next[maxn * 2],sum;

int n,m;

int ma,w;

void insert(int x,int y){
	t[++sum]=y;
	next[sum]=head[x];
	head[x]=sum;
}

void dfs(int x,int father,int dis) {
	if (dis>ma) {
		ma=dis;
		w=x;
	}
	for(int tmp=head[x];tmp;tmp=next[tmp]) {
		if (t[tmp]==father) continue;
		dfs(t[tmp],x,dis+1);
	}
}
