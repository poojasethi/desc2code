#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int n;
int arr[103];
int main()
{
	cin>>n;

	for(int i=0 ; i < n ;i++){
		cin>>arr[i];
	}

	for(int i=1; i<n ; i++){
		int j=i;
		while(j>0 && arr[j-1]>arr[j]){
			cout<<j-1+1<<" "<<j+1<<endl;
			swap(arr[j-1],arr[j]);
			j--;
		}
	}
}
