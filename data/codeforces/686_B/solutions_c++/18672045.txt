#include <iostream>
using namespace std;
#define ll long long
ll arr[200];
int main() {
	ll n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>arr[i];
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n-1;j++)
		{
			if(arr[j]>arr[j+1])
			{
				swap(arr[j],arr[j+1]);
				cout<<j+1<<" "<<j+2<<endl;
			}
		}
	}
	return 0;
}
