#include<bits/stdc++.h>
using namespace std;
long A[105];
int main()
{
  int N;
  cin>>N;
  for(int i=0;i<N;i++)scanf("%ld",&A[i]);

  for(int i=0;i<N;i++)
  {
    for(int j=0;j<N-1;j++)
    {
      if(A[j]>A[j+1])
      {
        printf("%d %d\n",j+1,j+2);
        A[j]=A[j]+A[j+1]-(A[j+1]=A[j]);
      }
    }
  }
  return 0;
}
