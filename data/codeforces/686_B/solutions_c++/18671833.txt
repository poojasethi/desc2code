#include <bits/stdc++.h>
using namespace std;

const int N = 102;
int n;
int v[N];

int main(void){
	scanf("%d", &n);

	for(int i = 0; i < n; i++){
		scanf("%d", &v[i]);
	}

	for(int i = 0; i < n; i++){
		for(int j = 0; j < n-1; j++){
			if(v[j] > v[j+1]){
				swap(v[j], v[j+1]);
				printf("%d %d\n", j+1, j+2);
			}
		}
	}
}
