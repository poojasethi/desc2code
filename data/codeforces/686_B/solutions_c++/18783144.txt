#include<iostream>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int i;
	int j;
	int input[103];
	for(i=1;i<=n;i++)
		cin >> input[i];

	for(i=1;i<=n;i++)
		for(j=1;j<=n-1;j++)
			if(input[j] > input[j+1])
			{
				cout << j << " " << j+1 << endl;
				swap(input[j],input[j+1]);
			}

	return 0;
}
