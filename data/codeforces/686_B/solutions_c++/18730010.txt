#include<iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,i,j,l=0,r=0,a[102]={},cnt=0;
	cin>>n;
	for(i=1;i<=n;i++) cin>>a[i];
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n-i;j++)
		{
			if(a[j]>a[j+1])
			{
				swap(a[j],a[j+1]);
				cout<<j<<" "<<j+1<<"\n";
			}
		}
	}
	return 0;
}
