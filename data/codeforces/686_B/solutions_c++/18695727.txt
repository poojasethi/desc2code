#include <bits/stdc++.h>
using namespace std;

int N;
int a[1007];

int main() {

	scanf("%d" ,&N);
	for (int i = 1; i <= N; i++)
		scanf("%d", &a[i]);

	for (int i = 1; i <= N; i++)
		for (int j = i-1; j > 0; j--) {
			if (a[j] > a[j + 1]) {
				swap(a[j], a[j+1]);
				printf("%d %d\n", j, j+1);
			}
		}

}
