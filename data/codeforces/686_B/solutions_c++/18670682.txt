#include <iostream>
#include <cstdio>
using namespace std;

int main(){
	int n;
	cin>>n;
	int height[n];
	for(int i=0;i<n;i++)
		cin>>height[i];
	for(int i=n-1;i>0;i--){
		for(int j=0;j<i;j++){
			if(height[j]>height[j+1]){
				printf("%d %d\n",j+1,j+2);
				swap(height[j],height[j+1]);
			}
		}
	}
}
