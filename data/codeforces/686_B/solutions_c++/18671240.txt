#include<iostream>
#include<algorithm>
using namespace std;
int n;
int pin[101];
int main()
{
	cin>>n;
	int i, j;
	for(i=0; i<n; i++)
		cin>>pin[i];
	for(i=n-1; i>0; i--)
	{
		for(j=0; j<i; j++)
		{
			if(pin[j]>pin[j+1])
			{
				cout<<(j+1)<<" "<<(j+2)<<endl;
				swap(pin[j], pin[j+1]);
			}
		}
	}
}
