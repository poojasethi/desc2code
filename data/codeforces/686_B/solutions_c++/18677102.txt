#include<bits/stdc++.h>
using namespace std;



int main()
{
	long long int n,temp;
	cin>>n;
	long long int ar[n];
	for(int i=0;i<n;i++)
	{
		cin>>ar[i];
	}
	for(int i=1;i<n;i++)
	{
		if(ar[i]<ar[i-1])
		{
			temp=ar[i];
			ar[i]=ar[i-1];
			ar[i-1]=temp;
			cout<<i<<" "<<i+1<<endl;
			i=0;
		}
	}
}
