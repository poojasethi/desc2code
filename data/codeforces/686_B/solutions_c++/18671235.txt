#include <stdio.h>

int n, A[110];

int main() {
	scanf("%d", &n);
	for (int i = 0 ; i < n ; ++i) {
		scanf("%d", &A[i]);
	}

	for (int i = 0 ; i < n ; ++i) {
		for (int j = 0 ; j < n - 1 ; ++j) {
			if (A[j] > A[j + 1]) {
				printf("%d %d\n", j + 1, j + 2);
				A[j] ^= A[j + 1] ^= A[j] ^= A[j + 1];
			}
		}
	}
}
