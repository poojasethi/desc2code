#include <iostream>
using namespace std;
int n, a[105];
int main(int argc, char const *argv[])
{
	cin >> n;
	for(int i = 1; i <= n; i++) cin >> a[i];
	for(int i = 1; i <= n; i++) for(int j = i + 1; j <= n; j++) if(a[i] > a[j]) for(int k = j; k > i; k--) cout << k - 1 << " " << k << endl, swap(a[k], a[k - 1]);
}
