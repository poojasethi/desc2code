#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,j;
	cin>>n;
	vector<int>v(n+1);
	for(i=1;i<=n;i++)
	cin>>v[i];
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n-i;j++)
		{
			if(v[j]>v[j+1]){
				swap(v[j],v[j+1]);
				cout<<j<<" "<<j+1<<"\n";
			}
			
		}
	}
	/*for(i=1;i<=n;i++)
	cout<<v[i];*/
}
