#include <bits/stdc++.h>
using namespace std;
#define ri(x) scanf("%d",&x)
#define FOR(i,j,k) for(int i = j; i<k; i++)
int A[101];
int main(){
	int N; ri(N);
	FOR(i,0,N) ri(A[i]);
	FOR(i,0,N){
		FOR(j,0,N-i-1){
			if(A[j] > A[j+1]){
				printf("%d %d\n",j+1,j+2);
				swap(A[j],A[j+1]);
			}
		}
	}

}
