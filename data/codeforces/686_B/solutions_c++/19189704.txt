#include<cstdio>
#include<algorithm>
int data[105];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d",&data[i]);
	for(int i=n;i>=1;i--)for(int j=1;j<i;j++)
	{
		if(data[j]>data[j+1]){std::swap(data[j],data[j+1]);printf("%d %d\n",j,j+1);}
	}
	return 0;
}
