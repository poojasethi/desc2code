#include<iostream>
using namespace std;
typedef long long int ll;
int main()
{ll n,i;
cin>>n;
ll arr[n];
for(i=0;i<n;i++)
cin>>arr[i];
ll sum=0;
while(sum!=n-1)
{sum=0;
for(i=1;i<n;i++)
 {if(arr[i]<arr[i-1])
   {swap(arr[i],arr[i-1]);
    cout<<i<<" "<<i+1<<endl;
   }
  else
  sum++;
 }
}


return 0;
}
