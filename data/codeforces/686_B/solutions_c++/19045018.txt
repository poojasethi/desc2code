#include <bits/stdc++.h>
using namespace std;

int a[200];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d",a+i);
    for(int i=1;i<=n;i++)
        for(int k=2;k<=n;k++)
            if(a[k]<a[k-1])
                swap(a[k],a[k-1]),printf("%d %d\n",k-1,k);
    return 0;
}
