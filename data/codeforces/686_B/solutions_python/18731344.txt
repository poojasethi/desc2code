#!/usr/bin/env python

def main():
    n = int(raw_input())
    numbers = map(int, raw_input().split())
    for i in range(n - 1):
        for j in range(n - 1 - i):
            if numbers[j] > numbers[j + 1]:
                print j + 1, j + 2
                numbers[j], numbers[j + 1] = numbers[j + 1], numbers[j]

if __name__ == '__main__':
    main()
