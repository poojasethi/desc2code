n = int(raw_input())
ls = map(int, raw_input().split())
boo = True

while True:
	boo = True
	for i in range(n-1):
		if ls[i] > ls[i+1]:
			boo = False
			l = i
			j = 0
			while True:
				if l+j+1>=n:
					print l+1,j+l
					break
				elif ls[l+j]<ls[l+j+1]:
					print l+1, j+l
					break
				ls[l+j], ls[l+j+1] = ls[l+j+1], ls[l+j]
				j+=2
	if boo == True:
		break
