n = input()
a = [int(i) for i in raw_input().split(" ")]
s = sorted(a)
x = 0

while True:
    if x == n:
        break
    if a[x] == s[x]:
        x+=1
    else:
        t = s[x]
        r = a[x:].index(t)+x
        z = a.pop(r)
        a.insert(x,z)
        for i in range(r,x,-1):
            print i,i+1
            pass
