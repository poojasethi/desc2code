n = input()
a = map(int, raw_input().split())

sort_success = False
while not sort_success:
    sort_success = True
    for i in range(1, n):
        if a[i] < a[i-1]:
            sort_success = False
            a[i], a[i-1] = a[i-1], a[i]
            print (i-1)+1, (i)+1
