#include <iostream>
#include <cstdio>

using namespace std;

int a[105];

int main(){
	int n,d,l;
	while(~scanf("%d%d%d",&n,&d,&l)){
		for(int i=1;i<n;i++){
			if(d>0) a[i]=l;
			else a[i]=1;
			d=a[i]-d;
		}
		if(1<=d&&d<=l){
			for(int i=1;i<n;i++)
				printf("%d ",a[i]);
			printf("%d\n",d);
		}
		else
			printf("-1\n");
	}
	return 0;
}
