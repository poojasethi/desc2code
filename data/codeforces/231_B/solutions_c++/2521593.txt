#include<stdio.h>
int main()
{
	int i, j;
	int n, l, d;
	scanf("%d%d%d", &n, &d, &l);
	int a[222];
	for(i = 1; i < n; i++)
	{
		if(d <= 0) a[i] = 1, d = 1 - d;
		else a[i] = l, d = l - d;
	}
	a[n] = d;
	if(d >= 1 && d <= l)
	{
		for(i = 1; i < n; i++)
			printf("%d ", a[i]);
		printf("%d\n", a[n]);
	}
	else puts("-1");
	return 0;
}
