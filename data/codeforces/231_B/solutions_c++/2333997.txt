#include<stdio.h>
int a[103];
int main()
{
    int d, l, n, i;
    scanf("%d%d%d", &n, &d, &l);
    for(i = 0; i < n-1; i++)
    {
        if(d <= 0) a[i] = 1;
        else a[i] = l;
        d = a[i] - d;
    }
    a[n-1] = d;
    if(a[n-1] <= 0 || a[n-1] > l) { puts("-1"); return 0; }
    for(i = 0; i < n-1; i++)
        printf("%d ", a[i]);
    printf("%d\n", a[n-1]);
    return 0;
}
