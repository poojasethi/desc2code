#include<iostream>
using namespace std;

int main(){
	int n, d, l;
	cin>>n>>d>>l;
	int array[n];
	int i;
	for(i = 1; i < n; ++i){
		if(d < 1){
			d = 1 - d;
			array[i] = 1;
		}
		else{
			d = l - d;
			array[i] = l;
		}
	}
	if(d < 1 || d > l)
		cout<<"-1\n";
	else{
		for(i = 1; i < n; ++i)
			cout<<array[i]<<" ";
		cout<<d<<endl;
	}
	return 0;
}
