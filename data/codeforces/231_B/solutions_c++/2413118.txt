#include <cstdio>
#include <iostream>

using namespace std;

int a[200], i, n, l, d;

int main() {
    scanf("%d%d%d", &n, &d, &l);
    for(i = 1; i <= n; i++) {
        if(d < 1) a[i] = 1;
        else a[i] = l;
        d = a[i] - d;
    }
    a[n] -= d;
    if(a[n] < 1 || a[n] > l) printf("-1");
    else
        for(i = 1; i <= n; i++)
            printf("%d ", a[i]);
}
