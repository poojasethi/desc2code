#include<cstdio>
#include<cstring>
using namespace std;
const int maxn=102;
int a[maxn];
int n,d,l;

int main(){
  int i;
  scanf("%d%d%d",&n,&d,&l);
  for (i=1;i<=n;i++){
    if (d>0) a[i]=l;
    else a[i]=1;
    d=a[i]-d;
  }
  a[n]-=d;
  if (a[n]<1 || a[n]>l) printf("-1");
  else {
    for (i=1;i<n;i++) printf("%d ",a[i]);
    printf("%d\n",a[i]);
  }
  return 0;
}
