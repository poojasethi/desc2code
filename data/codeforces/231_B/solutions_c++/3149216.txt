#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<stdio.h>
using namespace std;
int n,k,m,a[111];
int main()
{
  scanf("%d%d%d",&n,&k,&m);
  for(int i=1;i<=n;i++)
   {
     if(k>0) a[i]=m;
     else a[i]=1;
     k=a[i]-k;
   }
  a[n]-=k;
  if(a[n]<=0 || a[n]>m) { puts("-1"); return 0; }
  for(int i=1;i<=n;i++)
   printf("%d ",a[i]);
  return 0;
}
