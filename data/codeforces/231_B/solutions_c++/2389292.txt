#include"stdio.h"
#include"string.h"
int num[1000];
int main(){
	int n,d,l,i;
	while(scanf("%d%d%d",&n,&d,&l)!=EOF)
	{
		memset(num,0,sizeof(0));
		for(i=0;i<n-1;i++)
		{
			if(d>0)
				num[i]=l;
			else
				num[i]=1;
			d=num[i]-d;
		}
		if(d<1||d>l)
			printf("-1");
		else
		{
			for(i=0;i<n-1;i++)
				printf("%d ",num[i]);
			printf("%d",d);
		}
		printf("\n");
	}
	return 0;
}
