#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int i,n,d,l;
    int arr[101];
    scanf("%d%d%d",&n,&d,&l);
    for(i=0;i<n-1;i++)
    {
        arr[i]=d>0?l:1;
        d=arr[i]-d;
    }
    arr[n-1]=d;
    if(arr[n-1]<1 || arr[n-1]>l) printf("-1\n");
    else for(i=0;i<n;i++) printf("%d ",arr[i]);
    //system("PAUSE");
}
