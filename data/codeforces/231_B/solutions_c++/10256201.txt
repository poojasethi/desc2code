#include<bits/stdc++.h>
using namespace std;

const int N=105;

int a[N];

int main() {
  int n,k,m;
  scanf("%d%d%d",&n,&k,&m);
  for(int i=1;i<=n;i++) {
    if(k>0) {
      a[i]=m;
    } else {
      a[i]=1;
    }
    k=a[i]-k;
  }
  a[n]-=k;
  if(a[n]<=0||a[n]>m) {
    puts("-1");
  } else {
    for(int i=1;i<=n;i++) {
      printf("%d ",a[i]);
    }
  }
  return 0;
}
