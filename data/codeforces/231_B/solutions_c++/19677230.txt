#include <bits/stdc++.h>

using namespace std;

int main()
{
    int i , j , n , d , l , a[110];
    cin >> n >> d>>l;
    for(i=0;i<n-1;i++){
        if(d>0)a[i]=l;
        else a[i]=1;

        d=a[i]-d;
    }
    a[n-1]=d;
    if(a[n-1]<1||a[n-1]>l){cout<<-1<<endl ; return 0;}

    for(i=0;i<n;i++)cout<<a[i]<<" ";
    cout<<endl;




    return 0;

}
