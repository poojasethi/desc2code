#include <iostream>

using namespace std;
int a[150],i=0;
int main()
{
	int n, l, d;
	cin >> n >> d >> l;

	
	for (i = 0; i < n; i++)
	{
		if (d>0) a[i] = l;
		else a[i] = 1;
		d = a[i] - d;
	}
	a[n - 1] -= d;
	if (a[n - 1]<1 || a[n - 1]>l) cout << "-1" << endl;
	else
	{
		for (int i = 0; i<n; i++) cout << a[i] << " ";
	}
	return 0;
}
