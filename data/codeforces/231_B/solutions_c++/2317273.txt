#include<iostream>
using namespace std;

int a[1000000];

int main()
{
	int n, d, l;
	cin>>n>>d>>l;
	a[0]=d;
	for(int i=1; i<n; i++)
	{
		if(a[i-1]>0)
		{
			a[i]=l-a[i-1];
			a[i-1]=l;
		}
		else
		{
			a[i]=a[i-1]*-1+1;
			a[i-1]=1;
		}
	}
	for(int i=0; i<n; i++)
	if(a[i]<1 || a[i]>l)
	{
		cout<<"-1";
		return 0;
	}
	for(int i=0; i<n; i++)
	cout<<a[i]<<" ";
	return 0;
}
