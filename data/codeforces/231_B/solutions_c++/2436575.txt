#include<iostream>
#include<vector>
using namespace std;

int main() {
	int n, d, l;
	cin>>n>>d>>l;
	vector<int> A(n, 0);
	for (int i = 0;i < n - 1;++i) {
		if (d <= 0)
			A[i] = 1;
		else
			A[i] = l;
		d = A[i] - d;
	}
	if (d > 0&&d <= l) {
		A[n - 1] = d;
		for (int i = 0;i < n;++i) {
			cout<<A[i]<<" ";
		}
	}
	else
		cout<<-1;
	return 0;
}
