#include <iostream>

using namespace std;

int main()
{
  int n, d, l;
  int a[101];
  int i;
  cin >> n >> d >> l;
  for (i=1;i<n;i++) {
    if (d <= 0)
      a[i] = 1;
    else
      a[i] = l;
    d = a[i] - d;
  }
  a[n] = d;
  if (a[n] >= 1 && a[n] <= l)
    for (i=1;i<=n;i++)
      cout << a[i] << " ";
  else
    cout << "-1";
  return 0;
}
