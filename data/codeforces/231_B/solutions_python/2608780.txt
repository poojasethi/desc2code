r = [(1, l)]
for i in range(n):
	r.append((1 - r[i][1], l - r[i][0]))
if r[n - 1][0] <= d <= r[n - 1][1]:
	for i in range(n - 2, -1, -1):
		for j in range(1, l + 1):
			if r[i][0] <= j - d <= r[i][1]:
				print j, 
				d = j - d;
				break
	print d
else:
	print -1
