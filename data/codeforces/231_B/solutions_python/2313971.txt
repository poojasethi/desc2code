n, d, l = map(int, raw_input().split())
i0 = (n + 1) / 2
i1 = n / 2
d = d - i0 + i1
ans = []
for i in range(0, n):
	if i % 2 == 0 and d >= 0:
			ans.append(min(d, l - 1))
			d = d - min(d, l - 1)
	else:
		if i % 2 == 1 and d < 0:
			ans.append(min(abs(d), l - 1))
			d = d + min(abs(d), l - 1)
		else:
			ans.append(0)
if d == 0:
	for i in ans: print (i + 1),
else:
	print -1
