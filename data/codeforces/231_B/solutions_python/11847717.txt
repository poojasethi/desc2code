n , d , l = raw_input().split()
n = int(n)
d = int(d)
l = int(l)
board = []
while n > 1:
    if d > 0 :
        board.append(l) 
        d = l-d 
    else :
        board.append(1) 
        d = 1-d
    n-=1
board.append(d)
if d >= 1 and d <= l :
    for i in board :
        print i,
else :
    print "-1"
