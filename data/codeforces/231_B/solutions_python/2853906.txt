#! /usr/bin/env python2

n, d, l = map(int, raw_input().strip().split(' '))

u = (n + 1) // 2
v = n - u
d -= u - v
l -= 1

ans = [[], []]

s = 0
if d < 0:
    u, v = v, u
    d = -d
    s = 1

if u * l >= d:
    while d > 0:
        if d >= l:
            ans[0].append(l)
            d -= l
        else:
            ans[0].append(d)
            d = 0
    while len(ans[0]) < u:
        ans[0].append(0)
    ans[1] = [0] * v
    if s:
        ans[0], ans[1] = ans[1], ans[0]
    cur = 0
    for i in xrange(n):
        print ans[cur][i / 2] + 1, 
        cur ^= 1
else:
    print -1
