#! /usr/bin/python

x,y = map(int,raw_input().split())

nx = 0
ny = 0
step = 1
ans = 0
while True:
    newx = nx + step
    if y == ny and x >= nx and x <= newx:
       break
    ans += 1
    nx = newx
    newy = ny + step
    if x == nx and y >= ny and y <= newy:
       break
    ans += 1
    ny = newy
    step += 1

    newx = nx - step;
    if y == ny and x >= newx and x <= nx:
       break
    ans += 1
    nx = newx
    newy = ny - step
    if x == nx and y >= newy and y <= ny:
        break
    ans += 1
    ny = newy
    step += 1

print ans
