#!/usr/bin/env python
import sys

RIGHT = 0
UP    = 1
LEFT  = 2
DOWN  = 3

if __name__ == '__main__':
    x, y = map(int, sys.stdin.readline().strip().split())

    turns = 0 
    i, j  = 0, 0
    count = 0
    while True:
        s_i, s_j = i, j
        if count % 4 == RIGHT:
            i += (count / 2) + 1
        elif count % 4 == UP:
            j += (count / 2) + 1
        elif count % 4 == LEFT:
            i -= (count / 2) + 1
        elif count % 4 == DOWN:
            j -= (count / 2) + 1

        if min(s_i, i) <= x <= max(s_i, i) and min(s_j, j) <= y <= max(s_j, j):
            print turns
            break

        turns += 1
        count += 1
