xy = raw_input().split()

x = int(xy[0])
y = int(xy[1])

def getAns(x,y):
	d1 = x + y
	d2 = y - x

	if ((d1 == 0) & (d2 == 0)):
		return 0

	if ((d1 >= 0) & (d2 > 0)):
		return 4*y - 2

	if ((d2 >= 0) & (d1 < 0)):
		return -4*x - 1

	if ((d1 <= 1) & (d2 < 0)):
		return -4*y

	if ((d2 <= 0) & (d1 > 1)):
		return 4*x - 3

print getAns(x,y)
