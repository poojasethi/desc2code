

m, n = map(int, raw_input().split())


x, y = 0, 0
steps = 0
length = 1

d = [(1,0), (0,1), (-1,0), (0,-1)]

while True:
  nx = x + d[steps % 4][0] * length
  ny = y + d[steps % 4][1] * length

  if x == nx == m:
    if y <= n <= ny or ny <= n <= y:
      break
  elif y == ny == n:
    if x <= m <= nx or nx <= m <= x:
      break
  x, y = nx, ny

  steps += 1
  if steps % 2 == 0:
    length += 1

print steps
