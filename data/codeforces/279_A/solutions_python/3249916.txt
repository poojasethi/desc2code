#!/usr/bin/env/ python
x,y = map(int,raw_input().split())
k = max(abs(x),abs(y))
if (k==0):
    print 0
elif (x==1 and y==0):
    print 0
else:
    turn = k
    if x == turn and x>0 and y==-(turn-1) : print (turn-1)*4
    elif x == turn and y>-(turn-1): print (turn-1)*4+1
    elif y == turn : print (turn-1)*4+2
    elif x == -turn : print (turn-1)*4+3
    elif y == -turn : print (turn-1)*4+4
