n, m = map(int, raw_input().split())

def main():
	if n == 0 and m == 0: return 0
	dir = [(1, 0), (0, 1), (-1, 0), (0, -1)]
	x, y, k, ans = 0, 0, 0, 0
	for i in xrange(1, 501):
		for j in xrange(1, (i + 1) / 2 + 1):
			x, y = x + dir[k][0], y + dir[k][1]
#			print 'x= ', x, 'y= ', y
			if x == n and y == m:
				return ans
		ans += 1
		k = (k + 1) % 4

print main()
