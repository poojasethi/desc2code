def travel(x, y) : 
	dx = [1, 0, -1, 0]
	dy = [0, 1, 0, -1]
	sx, sy = [0, 0]
	nx, ny = [1, 0]
	turn = 0
	while not (min(sx, nx) <= x <= max(sx, nx) and min(sy, ny) <= y <= max(sy, ny)): 
		sx, sy = nx, ny
		turn += 1
		nx += dx[turn % 4] * (turn // 2 + 1)
		ny += dy[turn % 4] * (turn // 2 + 1)
	return turn
x, y = map(int, raw_input().split())
print travel(x, y)
