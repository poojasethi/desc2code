x0, y0 = map(int, raw_input().split())
x = y = ans = 0
d = 1
while True:
    if y == y0 and min(x, x + d) <= x0 <= max(x, x + d):
        break
    else:
        x += d
        ans += 1
        if x == x0 and min(y, y + d) <= y0 <= max(y, y + d):
            break
        else:
            y += d
            ans += 1
            if d > 0:
                d = -d - 1
            else:
                d = -d + 1
print ans
