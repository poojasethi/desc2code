#!/usr/bin/env python
from sys import stdin as cin

def main():
    x,y = map(int, next(cin).split())

    l = max(abs(x), abs(y))

    if l == 0:
        return 0

    if x == l and y == 1 - l:
        x = x - 1
        l = l - 1

    if x >= y and x > -y:
        cor = 1
    elif x < y and x >= -y:
        cor = 2
    elif x <= y and x < -y:
        cor = 3
    else:
        cor = 4

    return (l-1) * 4 + cor

print main()
