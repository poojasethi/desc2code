#include <bits/stdc++.h>

using namespace std;

int main()
{
    int x, y, a = 0;
    cin >> x >> y;
    if (y > x && x >= -y) a = -2 + y * 4;
    if (y < x && x <= -y + 1) a = -y * 4;
    if (y <= x && x > -y + 1) a = -3 + x * 4;
    if (y >= x && x < -y) a = -1 - 4 * x;
    cout << a;
    return 0;
}
