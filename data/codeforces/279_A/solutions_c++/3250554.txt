#include <iostream>
using namespace std;


int main() {
    int x, y;
    int cnt = 0;
    cin>>x>>y;
    if ((y>x) && (x>=-y))
        cnt=-2+y*4;
    if((y>=x) && (x<-y))
        cnt=-1+4*-x;
    if ((x>y) && (-x+1>=y))
        cnt=-y*4;
    if ((x>=y) && (-x+1<y))
        cnt=-3+x*4;
    cout<<cnt;
    return 0;
}
