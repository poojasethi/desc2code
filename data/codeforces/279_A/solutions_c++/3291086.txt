#include <iostream>
#include <stdio.h>
using namespace std; 
int abs(int x){ return x>0?x:-x; }
int x, y; 
int main()
{
	cin >> x >> y; 
	if (x==0&&y==0) cout << 0;
	if (-y+1>=x&&-y>-x) cout << -y*4;
	if (x>=y&&x-1>-y) cout << x*4-3;
	if (y>x&&y>=-x) cout << y*4-2;
	if (-x>=-y&&-x>y) cout << -x*4-1;
}
