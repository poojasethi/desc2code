#include <stdio.h>
int main()
{
	int x,y;
	scanf("%d%d",&x,&y);
	if(y==1-x&&y<=0) x=x-1;
	if(x==0&&y==0)  printf("0\n"); 
	if(x>(-y)&&x>=y) printf("%d\n",4*x-3);
	else if(x>=(-y)&&x<y)  printf("%d\n",4*y-2);
	else if(x<=(-y)&&x>y)  printf("%d\n",-4*y);
	else if(x<(-y)&&x<=y)  printf("%d\n",-4*x-1);
}
