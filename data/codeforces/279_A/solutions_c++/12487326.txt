#include <stdio.h>
int x,y,m;
int main()
{
    while(~scanf("%d%d",&x,&y))
    {
        if(y > x && x >= -y) m = -2+y*4;
        if(y < x && x <= -y+1) m = -y*4;
        if(y <= x && x > -y+1) m = -3+x*4;
        if(y >= x && x < -y) m = -1-4*x;
        printf("%d\n",m);
    }
    return 0;
}
