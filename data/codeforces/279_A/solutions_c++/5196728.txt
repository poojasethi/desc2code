#include <stdio.h>
#include <string.h>

int main(){
    int x, y;
    scanf("%d %d", &x, &y);
    int ans = 0;
    if(x>=y && (x+y)>1) ans = 4*x-3;
    else if(x<y && y>=-x) ans = 4*y-2;
    else if(y<-x && y>=x) ans = 4*(-x)-1;
    else ans = 4*(-y);
    printf("%d\n", ans);
    return 0;
}
