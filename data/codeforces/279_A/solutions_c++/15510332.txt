#include<bits/stdc++.h>

using namespace std;

int main()
{
    int x,y;
    cin>>x>>y;
    int m;
    
    if(y>x&&x>=-y)m=4*y-2;
    if(y<x&&x<=-y+1)m=-y*4;
    if(y<=x&&x>-y+1)m=4*x-3;
    if(y>=x&&x<-y)m=-4*x-1;
    if(x==0&&y==0)m=0;
    cout<<m<<endl;
    return 0;
}
