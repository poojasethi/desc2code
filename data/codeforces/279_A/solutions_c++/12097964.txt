#include <bits//stdc++.h>
using namespace std;
int x,y,ans;
int main(){
    scanf("%d %d",&x,&y);
    if(y>x && x>=-y)ans=-2+y*4;
    if(y<x && x<=-y+1)ans=-y*4;
    if(y<=x && x>-y+1)ans=-3+x*4;
    if(y>=x && x<-y)ans=-1-4*x;
    printf("%d\n",ans);
    return 0;
}
