#include<stdio.h>
#include<math.h>
int main()
{
	int a,b;

	scanf("%d%d",&a,&b);
	if(a>=b&&a+b-1>0)printf("%d\n",4*a-3);
	else if(a<b&&a+b>=0)printf("%d\n",4*b-2);
	else if(a<=b&&a+b<0)printf("%d\n",-4*a-1);
	else if(a>b&&a+b-1<=0)printf("%d\n",-4*b);
	else printf("0\n");
	return 0;
}
