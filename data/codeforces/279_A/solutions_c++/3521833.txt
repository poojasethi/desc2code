#include<stdio.h>

int main()
{
	int x,y,time;
	while(scanf("%d%d",&x,&y)!=EOF)
	{
		if(x==1 && y==0)
			time=0;
		else if(y<=x && y>-x+1)
			time=x*4-3;
		else if(y>x && y>=-x)
			time=y*4-2;
		else if(y>=x && y<-x)
			time=(-x)*4-1;
		else time=(-y)*4;
		printf("%d\n",time);
	}
	return 0;
}
