#include <iostream>

using namespace std;

int main(  ) {
	int x, y, sol= 0;
	cin>>x>>y;

	if ( x<y && x>=-y ) {
        sol= y*4-2;
	}
	if ( x>y && x<=1-y ) {
        sol= -y*4;
	}
	if ( x>=y && x>1-y ) {
        sol= x*4-3;
	}
	if ( x<=y && x<-y ) {
        sol= -(x*4+1);
	}

	cout<<sol<<"\n";

	return 0;
}
