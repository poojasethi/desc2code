#include<stdio.h>
#include<stdlib.h>
int main()
{   int x,y,ans;
    scanf("%d%d",&x,&y);
    if(x==0&&y==0)     ans=0;
    if(x-y>=0&&x+y>1)  ans=4*x-3;
    if(x-y<0&&x+y>=0)  ans=4*y-2;
    if(x-y<=0&&x+y<0)  ans=4*(-x)-1;
    if(x-y>0&&x+y<=1)  ans=4*(-y);
    printf("%d",ans);
    return 0;
}
