#include<iostream>
using namespace std;

int main()
{
    int x,y;
    cin>>x>>y;
    if(y<=x&&y>1-x)
       cout<<(4*(x-1)+1)<<endl;
    else if(y>=-x&&y>x)
       cout<<(4*y-2)<<endl;
    else if(y<-x&&y>=x)
       cout<<(4*(-x)-1)<<endl;
    else
       cout<<(4*(-y))<<endl;
    return 0;
}
