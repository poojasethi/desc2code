#include <iostream>
#include <stdio.h>
using namespace std;

int main(){
  int x, y;
  cin >> x >> y;
  if(x==0 && y==0) cout << 0;
  else if(x >= y && x > 1-y) cout << 4*x-3;
  else if(y > x && y >= -x) cout << 4*y-2;
  else if(-x > y && -x >= -y) cout << 4*(-x)-1;
  else if(-y > -x && -y >= x-1) cout << 4*(-y);
}
