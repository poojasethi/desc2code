#include "bits/stdc++.h"
using namespace std;

int main()
{
	int x, y;
	cin >> x >> y;

	int s = 0; 
	
	if(y > x && x >= -y) s = 4*y - 2;
	else if(y >= x && x <-y) s = -4*x - 1;
	else if(y <= x && x > -y+1) s = -3 + 4*x;
	else if(y < x && x <= -y+1) s = -y*4;

	cout << s << endl;
    return 0;
}
