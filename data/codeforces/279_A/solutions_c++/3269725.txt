#include <iostream>
#include<stdio.h>
#include<cstring>
using namespace std;

int main()
{
    int x,y,ans=0;
    scanf("%d%d",&x,&y);
    if(y<=x && y>1-x) ans=4*x-3;
    if(y>=x && y<-x) ans=-4*x-1;
    if(x<y && x>=-y) ans=4*y-2;
    if(y<x && x<=-y+1) ans=-4*y;
    printf("%d\n",ans);
    return 0;
}
