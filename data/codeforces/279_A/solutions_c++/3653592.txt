#include<stdio.h>
int main()
{
    int x,y;
    scanf("%d %d",&x,&y);
    if(x==0&&y==0)printf("0");
    else if(x<y&&x>=-1*y)printf("%d",4*(y-1)+2);
    else if(y<-1*x&&y>=x)printf("%d",4*((-1*x)-1)+3);
    else if(-1*y+1>=x&&x>y)printf("%d",4*-1*y);
    else printf("%d",4*(x-1)+1);
}
