#include <iostream>

using namespace std;
int a[60],b[60],c[60],d[60];
int main()
{
    int p,q,l,r;
    cin>>p>>q>>l>>r;
    for(int i=0;i<p;i++)cin>>a[i]>>b[i];
    for(int i=0;i<q;i++)cin>>c[i]>>d[i];
    int h=0;
    int t,k,j;
    for(t=l;t<=r;t++)
    {
        j=0;k=0;
        while(j<p&&k<q)
        {
            if(b[j]<c[k]+t)j++;
            else if(d[k]+t<a[j])k++;
            else{h++;break;}
        }
    }
    cout<<h;


}
