#include <bits/stdc++.h>

using namespace std;

int xx[1001],yy[1001];
int main(){
	int p,q,l,r;
	
	scanf("%d%d%d%d",&p,&q,&l,&r);
	int x,y;
	for(int i=0;i<p;i++){
		scanf("%d%d",&x,&y);
		for(int j=x;j<=y;j++)
			xx[j] = 1;
	}
	
	for(int i=0;i<q;i++){
		scanf("%d%d",&x,&y);
		for(int j=x;j<=y;j++)
			yy[j] = 1;
	}
	
	int ans = 0;
	
	for(int i=l;i<=r;i++){
		for(int j=0;j+i<=1000;j++)
			if(yy[j]==1 && xx[j+i]==1){
				ans++;
				break;
			}
	}
	
	cout<<ans<<endl;
}
