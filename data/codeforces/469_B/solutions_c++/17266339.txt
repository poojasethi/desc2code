#include<bits/stdc++.h>
using namespace std;
int p,q,l,r,ans,tl,tr;
int v[5000],a[5000],b[5000];
int main() {
	cin>>p>>q>>l>>r;
	for(int i=0;i<p;i++){
		cin>>tl>>tr;
		for(int j=tl;j<tr;j++) a[j]=1;
	}
	for(int i=0;i<q;i++){
		cin>>tl>>tr;
		for(int j=tl;j<tr;j++) b[j]=1;
	}
	for(int i=0;i<1001;i++) 
		for(int j=0;j<1001;j++)
			if(a[i]&&b[j]) 
				v[i-j]=v[i-j-1]=1;
	for(int k=l;k<r+1;k++) 
		ans+=v[k]|v[k-1];
	cout<<ans<<endl;
}
