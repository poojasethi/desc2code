#include <iostream>
using namespace std;

long long arr[1001], ans[1001];

int main(){
	long long p, q, l, r, a, b, c, d, i, t;
	cin >> p >> q >> l >> r;
	while(p--){
		cin >> a >> b;
		for(i=a;i<=b;i++) arr[i]=1;
	}
	while(q--){
		cin >> a >> b;
		for(t=l;t<=r;t++){
			for(i=a+t;i<=b+t;i++){
				if(i<1001 and arr[i]){
					ans[t] = 1;
					break;
				}
			}
		}
	}
	t=0;
	for(i=0;i<1001;i++){
		if(ans[i]) t++;
	}
	cout << t << '\n';
	return 0;
}
