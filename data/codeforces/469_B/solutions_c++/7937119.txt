#include<cstdio>
using namespace std;
bool b1[1001],b2[1001];
int main()
{
	int n,m,l,r,a,b,oa=0,o;
    scanf("%d%d%d%d",&n,&m,&l,&r);
	while(n--)
	{
		scanf("%d%d",&a,&b);
		for(int i=a;i<=b;++i)
			b1[i]=1;
	}
	while(m--)
	{
		scanf("%d%d",&a,&b);
		for(int i=a;i<=b;++i)
			b2[i]=2;
	}
	for(int i=l;i<=r;++i)
	for(int j=i;j<1001;++j)
	if(b1[j]&&b2[j-i])
	{++o;break;}
	printf("%d\n",o);
}
