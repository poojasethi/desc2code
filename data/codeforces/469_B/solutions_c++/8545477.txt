#include<iostream>
using namespace std;
pair<int,int> z[51], x[51];
int g;
int main()
{
    int n,m,t,r,dem=0,kt=0;
    cin>>n>>m>>t>>r;
    for(int i=0;i<n;i++)
        cin>>z[i].first>>z[i].second;
    for(int i=0;i<m;i++)
        cin>>x[i].first>>x[i].second;
    int i,j;
    for(;t<=r;t++)
        for(g=i=0;i<n && !g;i++)
            for(j=0;j<m && !g;j++)
                if(z[i].first<=x[j].second+t && z[i].second>=x[j].first+t) g=++dem;

    cout<<dem;
}
