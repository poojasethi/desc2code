#include<iostream>
using namespace std;

int n,m,l,r,ans,ll,rr;
bool a[1010],b[1010];

int main()
{
	ios_base::sync_with_stdio(0);
	
	cin>>n>>m>>ll>>rr;
	for(int i=0;i<n;i++)
	{
		cin>>l>>r;
		for(int j=l;j<=r;j++)
			a[j]=true;
	}
	for(int i=0;i<m;i++)
	{
		cin>>l>>r;
		for(int j=l;j<=r;j++)
			b[j]=true;
	}
	
	for(int i=ll;i<=rr;i++)
		for(int j=0;j<=1000-i;j++)
			if(a[j+i]&b[j])
			{
				ans++;
				break;
			}
			
	cout<<ans;
	
	return 0;
}
