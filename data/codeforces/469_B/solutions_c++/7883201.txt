#include <bits/stdc++.h>
using namespace std;

int p, q, l, r, ok[2005], a, b, x[2005], res;

int main(){
cin >> p >> q >> l >> r;
for(int i = 0; i < p; i++){
cin >> a >> b;
for(int j = a; j <= b; j++) ok[j] = 1;
}
for(int i = 0; i < q; i++){
cin >> a >> b;
for(int j = a; j <= b; j++) x[j] = 1;
}
for(int i = l; i <= r; i++){
int r = 0;
for(int j = i; j <= 1000; j++){
if(ok[j] && x[j - i]) r = 1;
}
res += r;
}
cout << res;
}
