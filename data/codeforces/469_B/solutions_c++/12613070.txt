#include <bits/stdc++.h>
using namespace std;
int main(){
int p,q,l,r,ans=0,f=0;
cin>>p>>q>>l>>r;
int a[p][2],b[q][2];
for(int i=0;i<p;i++){
cin>>a[i][0]>>a[i][1];
}
for(int i=0;i<q;i++){
cin>>b[i][0]>>b[i][1];
}
for(int i=l;i<=r;i++){
for(int j=0;j<q;j++){
    for(int k=0;k<p;k++){
       if(b[j][0]+i<=a[k][1] && b[j][1]+i>=a[k][0]){
       f=1;
        }
     }
}
if(f!=0){
ans++;
}
f=0;
}
cout<<ans<<endl;
}
