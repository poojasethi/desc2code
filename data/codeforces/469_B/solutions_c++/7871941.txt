#include<cstdio>
#include<cstdlib>
int a[58],b[58],c[58],d[58];
int main()
{
	int p,q,l,r;
	scanf("%d%d%d%d",&p,&q,&l,&r);
	for(int i=1;i<=p;i++)
		scanf("%d%d",&a[i],&b[i]);
	for(int i=1;i<=q;i++)
		scanf("%d%d",&c[i],&d[i]);
		
	int ans=0;
	for(int i=l;i<=r;i++){
		int j=1,t1,t2,k=1;
		while(j<=q && k<=p){
			t1=c[j]+i;t2=d[j]+i;
			if(t1>b[k])
					k++;
			 else if(t2<a[k])
			 		j++;
			 else {ans++;break;}
		}
	}
	printf("%d\n",ans);
	return 0;
}
