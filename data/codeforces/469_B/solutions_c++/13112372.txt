#include<bits/stdc++.h>
using namespace std;
set<int>s;
int main()
{
	int p,q,l,r,a1[1000],b1[1000],a2[1000],b2[1000];
	cin>>p>>q>>l>>r;
	int k,i,j;
	for(i=0;i<p;i++)
	{
		cin>>a1[i]>>b1[i];
	}
	for(i=0;i<q;i++)
		cin>>a2[i]>>b2[i];
	for(i=l;i<=r;i++)
	{
		for(j=0;j<q;j++)
		{
			for(k=0;k<p;k++)
			{
				if(a2[j]+i>b1[k] || b2[j]+i<a1[k])
					;
				else
					s.insert(i);
			}
		}
	}
	cout<<s.size();
	return 0;

}
