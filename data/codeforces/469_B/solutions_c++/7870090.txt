#include<stdio.h>
int main()
{
	int a[55],b[55],c[55],d[55];
	int q,p,l,r;
	scanf("%d%d%d%d",&p,&q,&l,&r);
	int i,j,k,f;
	for(i=0;i<p;i++)
	scanf("%d%d",&a[i],&b[i]);
	for(i=0;i<q;i++)
	scanf("%d%d",&c[i],&d[i]);
	int sum=0;
	for(i=l;i<=r;i++)
	{
		f=0;
		for(j=0;j<q;j++)
		{
		 for(k=0;k<p;k++)
		 if(c[j]+i>b[k]||d[j]+i<a[k])
		 continue;
		 else f=1;	
		}
		if(f)sum++;
		
	}
	printf("%d\n",sum);
}
