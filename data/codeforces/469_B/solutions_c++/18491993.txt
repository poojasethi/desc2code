#include<bits/stdc++.h>
using namespace std;
set<int>s;
int main()
{
    int a[100],b[100],c[100],d[100],i,j,k,n;
	int q,p,l,r;
	cin>>p>>q>>l>>r;
	for(i=0;i<p;i++)
	cin>>a[i]>>b[i];
	for(i=0;i<q;i++)
	cin>>c[i]>>d[i];
	for(i=l;i<=r;i++)
	{
	    n=0;
	    for(j=0;j<q;j++)
	    for(k=0;k<p;k++)
	    if(c[j]+i>b[k]||d[j]+i<a[k])
	      ;
	    else
	    s.insert(i);
	}
	cout<<s.size();
}
