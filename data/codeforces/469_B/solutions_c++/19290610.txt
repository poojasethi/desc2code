#include<bits/stdc++.h>
using namespace std;
int arr[1001];
int main()
{
	int i,j,k,p,q,l,r,x,y,cnt=0;
	cin>>p>>q>>l>>r;
	int a[p],b[p],c[q],d[q];
	for(i=0;i<p;i++)
		cin>>a[i]>>b[i];
	for(i=0;i<q;i++)
		cin>>c[i]>>d[i];
	for(i=0;i<q;i++)
	{
		for(j=0;j<p;j++)
		{
			x=a[j]-d[i];// d+x >= a
			y=b[j]-c[i];// c+y <= b
			if(x<0) x=0;
			for(k=x;k<=y;k++)
				arr[k]=1;
		}
	}
	for(i=l;i<=r;i++)
		if(arr[i])
			cnt++;
	cout<<cnt<<endl;
}
