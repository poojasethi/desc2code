#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
	int p,q,l,r;
	int i,j,k;
	cin>>p>>q>>l>>r;
	int z[p][2],x[q][2];
	for(i=0;i<p;i++)
		cin>>z[i][0]>>z[i][1];
	for(i=0;i<q;i++)
		cin>>x[i][0]>>x[i][1];

	int win=0,count=0;
	for(i=l;i<=r;i++)
	{	
		win=0;
		for(j=0;j<q &&win==0 ;j++)
		{
			for(k=0;k<p;k++)
			{
				if( !(z[k][0]>x[j][1]+i || x[j][0]+i>z[k][1] ))
				{
					count++;win=1;break;
				}
			}
		}
	}
	cout<<count<<endl;
}
