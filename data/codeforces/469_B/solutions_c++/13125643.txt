#include<stdio.h>
#include<string.h>
#include<set>
using namespace std;
int main()
{
	set<int> se;
	int vis[2005];
	int j,i,n,m,t_begin,t_end,b,e;
	scanf("%d %d %d %d",&n,&m,&t_begin,&t_end);
	memset(vis,0,sizeof(vis));
	while(n--)
	{
		scanf("%d %d",&b,&e);
		for(i=b;i<=e;i++)
			vis[i]=1;
	}
	while(m--)
	{
		scanf("%d %d",&b,&e);
		for(j=b;j<=e;j++)
			for(i=t_begin;i<=t_end;i++)
				if(vis[j+i])
					se.insert(i);
	}
	printf("%d\n",se.size());
	return 0;
}
