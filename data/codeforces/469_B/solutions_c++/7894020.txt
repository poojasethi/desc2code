#include<iostream>
using namespace std;

int a[1002],b[1000],c[1000],d[1000];
int main()
{
	int i,j,p,q,l,r,z,f=0;
	cin>>p>>q>>l>>r;
	
	
	for(i=1;i<=p;i++)
		{
		cin>>a[i]>>b[i];
}
	for(i=1;i<=q;i++)
		{
		cin>>c[i]>>d[i];
}
	for(z=l;z<=r;z++)
	{
		for(i=1,j=1;i<=p&&j<=q;)
		{
			if(b[i]<c[j]+z)
			i++;
			else
			if(d[j]+z<a[i])
			j++;
			else
			{
				f++;
				break;
			}
		}
	
	}
	cout<<f;
	
	cin.get();
	return 0;
}
