#include<bits/stdc++.h>
using namespace std;
int main()
{
	int p,q,l,r,i,j,k,ps[53][2],qs[52][2],c=0,flag;
	//freopen("in.txt","r",stdin);
	cin>>p>>q>>l>>r;
	for(i=0;i<p;i++)
		cin>>ps[i][0]>>ps[i][1];
	for(i=0;i<q;i++)
		cin>>qs[i][0]>>qs[i][1];
	for(i=l;i<=r;i++)
	{
		flag=0;
		for(j=0;j<p;j++)
		{
			for(k=0;k<q;k++)
			{
				if(((qs[k][1]+i)>=ps[j][0]) && ((qs[k][0]+i)<=ps[j][1]))
					flag=1;
			}
		}
		if(flag==1)
		c++;
	}
	cout<<c<<endl;
	return 0;
}
