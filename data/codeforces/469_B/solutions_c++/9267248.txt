#include<iostream>
#include<set>
using namespace std;
set<int> s;
int main()
{
	int p, q, l, r, a[52], b[52], c[52], d[52];
	cin>>p>>q>>l>>r;
	for(int i = 0; i < p; i++)
		cin>>a[i]>>b[i];
	for(int i = 0; i < q; i++)
		cin>>c[i]>>d[i];
	for(int i = l; i <= r; i++){
		for(int j = 0; j < q; j++){
			for(int k = 0; k < p; k++){
				if(c[j]+i > b[k] || d[j]+i < a[k])
					continue;
				else
					s.insert(i);
			}
		}
	}
	cout<<s.size()<<endl;
	return 0;
}
