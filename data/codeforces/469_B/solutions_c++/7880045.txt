#include<bits/stdc++.h>
using namespace std;
int main()
{
	int p,q,l,r,i,j,ans,fla,k;
	scanf("%d%d%d%d",&p,&q,&l,&r);
	int a[2*p];
	for(i=0;i<2*p;i++)
	scanf("%d",&a[i]);
	int b[2*q];
	for(i=0;i<2*q;i++)
	scanf("%d",&b[i]);
	ans=0;
	fla=0;
	for(i=l;i<=r;i++)
	{
		for(k=0;k<2*q;)
	{	
		for(j=0;j<2*p;)
		{
			
			if((b[k]+i<=a[j+1])&&(b[k+1]+i>=a[j]))
			{ans++;
				fla=1;
			break;}
			j=j+2;
		}
		if(fla==1)
		{
			fla=0;
			break;
		}
		k=k+2;
	}
	}
	printf("%d",ans);
	return 0;
}
