#include<cstdio>
using namespace std;
int a1[2];
bool b1[2][1001];
int main()
{
    int l,r,a,b,o=0;
    scanf("%d%d%d%d",a1,a1+1,&l,&r);
    for(int i=0;i<2;++i)
    while(a1[i]--)
    {
        scanf("%d%d",&a,&b);
        for(int j=a;j<=b;++j)
            b1[i][j]=1;
    }
    for(int i=l;i<=r;++i)
    for(int j=i;j<1001;++j)
    if(b1[0][j]&&b1[1][j-i])
    {++o;break;}
    printf("%d\n",o);
}
