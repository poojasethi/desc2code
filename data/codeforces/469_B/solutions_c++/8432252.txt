int ans,q,p,l,r,i,f,j,k,a[nn],b[nn],c[nn],d[nn];

{

 for (i=1;i<=q;i++)scanf("%d%d",&c[i],&d[i]);
 for (i=l;i<=r;i++){
 f=0;
 for (j=1;j<=q;j++)
 for (k=1;k<=p;k++)

 &&d[j]+i<=b[k])||(c[j]+i<=a[k]&&d[j]+i>=b[k]))f=1;

                   }

}
