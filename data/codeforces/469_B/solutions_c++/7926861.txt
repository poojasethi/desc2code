#include<stdio.h>
#include<stdlib.h>
int p,q,l,r,i,j,k,cnt;
int a,b,c[101],d[101],bj[2069],flag;;
int main(){
	scanf("%d%d%d%d",&p,&q,&l,&r);
	for (i=1;i<=p;i++){
		scanf("%d%d",&a,&b);
		for (j=a;j<=b;j++)
			bj[j]=1;
	}
	for (i=1;i<=q;i++)
		scanf("%d%d",&c[i],&d[i]);
	for (i=l;i<=r;i++){
		flag=0;
		for (j=1;j<=q;j++){
			for (k=c[j]+i;k<=d[j]+i;k++)
				if (bj[k]){
					cnt++;
					flag=1;
					break;
				}
			if (flag) break;
		}
	}
	printf("%d\n",cnt);
  return 0;
}
