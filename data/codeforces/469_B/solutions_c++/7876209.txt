#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;

int h,g[10005],n,m,j,k,l,i,a,e,r,res,b,d[10005][2];

main(){

cin>>n>>m>>a>>b;

for(i=0;i<n;i++){
cin>>e>>r;
for(j=e;j<=r;j++){
    g[j]=1;
}
}


for(i=0;i<m;i++){
cin>>d[i][0]>>d[i][1];
}

for(i=a;i<=b;i++){
    for(j=0;j<m;j++){
        for(h=d[j][0]+i;h<=d[j][1]+i;h++){
            if(g[h]==1) {l=1;res++;break;}
        }
        if(l==1) {l=0;break;}
    }
}

cout<<res;
}
