#include<iostream>
using namespace std;
int main()
{

int p,q,r,a[60],b[60],c[60],d[60],i,j,k,l;
cin>>p>>q>>l>>r;
for(i=0;i<p;i++)cin>>a[i]>>b[i];
for(i=0;i<q;i++)cin>>c[i]>>d[i];
int h=0;
//checking
for(i=l;i<=r;i++)
{bool r;
j=0;k=0;r=0;
while(r==0&&j<p&&k<q)
{
if(b[j]<c[k]+i){j++;}
else if(d[k]+i<a[j]){k++;}
else {r=1;h++;break;}


}

}

return 0;
}
