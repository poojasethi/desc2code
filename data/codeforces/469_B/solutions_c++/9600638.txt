#include <bits/stdc++.h>
using namespace std;

int main() 
{
	int p,q,l,r;
	cin>>p>>q>>l>>r;
	int a,b;
	vector <int> v(1001,0),w(1001,0);
	for(int i=0;i<p;i++)
	{
		cin>>a>>b;
		for(int j=a;j<=b;j++)
			v[j]=1;
	}
	for(int i=0;i<q;i++)
	{
		cin>>a>>b;
		for(int j=a;j<=b;j++)
			w[j]=1;
	}
	int ans=0;
	for(int i=l;i<=r;i++)
	{
		for(int j=0;j<1001&&j+i<1001;j++)
		{
			if(w[j]&&v[j+i])
			{
				ans++;
				break;
			}
		}
	}
	cout<<ans;
	return 0;
}
