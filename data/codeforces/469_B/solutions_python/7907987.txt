p, q, l, r = map(int, raw_input().split())

A = []
B = []

for i in xrange(p):
    x, y = map(int, raw_input().split())
    A += range(x, y + 1)

for i in xrange(q):
    x, y = map(int, raw_input().split())
    B += range(x, y + 1)

A = set(A)
B = set(B)
sol = 0

for i in xrange(l, r + 1):
    C = []
    for item in B:
        C.append(item + i)
    C = set(C)
    if len(A & C):
        sol += 1

print sol
