from collections import defaultdict, Counter
import random

import itertools
from functools import partial
from math import factorial
import math

def Ints():
    return map(int, raw_input().split())

p,q,l,r = Ints()

X = []
Z = []
for _ in range(p):
    X.append(Ints())
for _ in range(q):
    Z.append(Ints())
    
def overlap(a, b, t):
    x, y = a
    i, j = b
    i += t
    j += t

    return x <= j and y >= i

def issuitable(t):
    for x in X:
        for z in Z:
            if overlap(x, z, t):
                return True
    return False
s = 0
for t in range(l, r+1):
    if issuitable(t):
        s += 1
print s
