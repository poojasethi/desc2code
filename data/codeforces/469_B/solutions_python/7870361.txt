p,q,l,r=map(int,raw_input().split())
rr=[0 for i in xrange(2000)]
s=[0 for i in xrange(2000)]
mm=0
for i in xrange(p):
	a,b=map(int,raw_input().split())
	for j in xrange(a,b+1):
		rr[j]=1
		mm=max(mm,b)
for i in xrange(q):
	a,b=map(int,raw_input().split())
	for j in xrange(a,b+1):
		s[j]=1

ans=0
for j in xrange(l,r+1):
	ch=0
	t=j
	for k in xrange(t,mm+1):
		if s[k-t]==1:
			if rr[k]==1:
				ch=1
				break
	if ch==1:
		ans+=1
print ans
