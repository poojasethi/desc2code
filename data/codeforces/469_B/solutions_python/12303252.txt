L = lambda: map(int,raw_input().split())
p,q,l,r = L()
X = [0]*1001
for _ in xrange(p):
    a,b = L()
    for i in xrange(a,b+1):
        X[i] = 1
Z = [L() for _ in xrange(q)]
ans = 0
for k in xrange(l,r+1):
    flag = False
    for a,b in Z:
        for j in xrange(a+k,b+k+1):
            if 0 <= j <= 1000:
                if X[j]:
                    ans += 1
                    flag = True
                    break
        if flag:
            break

print ans
