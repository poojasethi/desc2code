p, q, l, r = [int(x) for x in raw_input().split(' ')]
ax = []
ay = []
for i in range(p):
    a, b = raw_input().split(' ')
    ax += range(int(a), int(b) + 1)

for i in range(q):
    a, b = raw_input().split(' ')
    ay.append((int(a), int(b)))

cnt = 0
for t in range(l, r+1):
    c = []
    for y in ay:
        c += range(y[0]+t, y[1]+t+1)
    cnt += bool(len(set(ax) & set(c)))

print cnt
