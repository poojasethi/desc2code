p, q, l, r = map(int, raw_input().split())

X, Z = 0, 0
for i in range(0, p):
    a, b = map(int, raw_input().split())
    for j in range(a, b+1): X |= (1<<j)

for i in range(0, q):
    a, b = map(int, raw_input().split())
    for j in range(a, b+1): Z |= (1<<j)

sol = 0
Z <<= l 
for shift in range(l, r+1):
    if X & Z: sol += 1
    Z <<= 1

print sol
