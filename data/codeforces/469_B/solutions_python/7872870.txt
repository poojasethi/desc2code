m=map(int,raw_input().split())
p=m[0]
q=m[1]
l=m[2]
r=m[3]
cm_p=[0]*1001
cm_q=[0]*1001
for i in xrange(p):
    m=map(int,raw_input().split())
    for k in xrange(m[0],m[1]+1):
	cm_p[k]=1

for i in xrange(q):
    m=map(int,raw_input().split())
    for k in xrange(m[0],m[1]+1):
	cm_q[k]=1

ans=0
for i in xrange(l,r+1):
    #check
    check=0
    for j in xrange(0,1001):
	if j+i>1000:
	    break
	if cm_p[j+i]==cm_q[j] and cm_q[j]==1:
	    check=1
	    break
    ans+=check

print ans
