n = map(int, raw_input().split())
A = [0] * 1001
B = [0] * 1001

for i in xrange(0, n[0]):
	k = map(int, raw_input().split())
	for j in xrange(k[0], k[1] + 1):
		A[j] = 1
		
for i in xrange(0, n[1]):
	k = map(int, raw_input().split())
	for j in xrange(k[0], k[1] + 1):
		B[j] = 1

ans = 0
for i in xrange(n[2], n[3] + 1):
	for j in xrange(0, 1001):
		if B[j] == 1 and i + j <= 1000 and A[i + j] == 1:
			ans += 1
			break
print ans
