p,q,l,r=map(int,raw_input().split())
ab=[]
co=0
cd=[]
for g in xrange(p):
    x=map(int,raw_input().split())
    for ff in xrange(x[0],x[1]+1):
        ab.append(ff)
for gg in xrange(q):
    y=map(int,raw_input().split())
    for ff in xrange(y[0],y[1]+1):
        cd.append(ff+l)
for xx in xrange(0,r-l+1):
    a=set(cd+ab)
    if len(ab)+len(cd)>len(a):
        co+=1
    for x in xrange(len(cd)):
        cd[x]=cd[x]+1
print co
