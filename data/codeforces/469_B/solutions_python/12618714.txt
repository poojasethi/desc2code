ntimes_z, ntimes_x, wake_i, wake_f = map(int, raw_input().split())

times_z = set()
times_x = set()

for _ in range(ntimes_z):
	initial,final = map(int, raw_input().split())
	for i in range(initial,final+1):
		times_z.add(i)

for _ in range(ntimes_x):
	initial,final = map(int, raw_input().split())
	for i in range(initial,final+1):
		times_x.add(i)

times = 0
for i in range(wake_i, wake_f+1):
	if not times_z.isdisjoint(set([x+i for x in times_x])):
		times += 1

print times
