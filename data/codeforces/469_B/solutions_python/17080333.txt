I=lambda:map(int, raw_input().split())
p,q,l,r=I()
fixed = []
flexible = []
for _ in xrange(p):
	a, b = I()
	fixed += [x for x in range(a,b+1)]
for _ in xrange(q):
	a, b = I()
	flexible += [x for x in range(a,b+1)]

fixed = set(fixed) 
count = 0
for t in range(l, r+1):
	flexiblet = set([x+t for x in flexible])
	if len(fixed & flexiblet) > 0: count += 1
print count
