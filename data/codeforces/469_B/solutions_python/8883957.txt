p,q,l,r = map(int,raw_input().split())
a = []
b = []
for i in range(p):
    a.append(map(int,raw_input().split()))
for i in range(q):
    b.append(map(int,raw_input().split()))

sol = 0
for t in range(l, r + 1):

    i = 0
    j = 0
    while i < p and j < q:
        if a[i][1] < b[j][0] + t:
            i += 1
        elif b[j][1] + t < a[i][0]:
            j += 1
        else:
            sol += 1
            break

print sol
