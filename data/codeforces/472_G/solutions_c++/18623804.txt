#define forn(i, n) for(int i = 0; i < n; ++i)

const int maxlen = 200010;

uint a[33][maxlen];
uint b[33][maxlen];

int LUT[1<<16];

int bits(uint x) {
	return LUT[x&((1<<16)-1)]+LUT[x>>16];
}

	forn(i, 32) for (int j = 0; j + i < s2.size(); ++j) if (s2[j+i] == '1') b[i][j>>5] |= (1<<(j&31));

		int ans = 0;
		int i;
