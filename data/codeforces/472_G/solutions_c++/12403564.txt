#include<cstdio>
#include<cstring>
#define REP(i,N) for(int i=0;i<N;i++)
#define ford(i,a,b) for(int i=a;i>=b;i--)
using namespace std;
const int MAXN=200001;
typedef unsigned long long ull;
char a[MAXN],b[MAXN];
int m;
int x, y, len;
int cnt[65536];
ull abit[MAXN],bbit[MAXN];
ull sabit[64][MAXN/63],sbbit[64][MAXN/63];
int main() {
    REP(i,65536) cnt[i]=__builtin_popcount(i);
//    freopen("test.inp","r",stdin);
    scanf("%s %s %d",a,b,&m);
    int n1=strlen(a);
    int n2=strlen(b);
    ford(i,n1-1,0) abit[i]=(abit[i+1]<<1)+a[i]-'0';
    ford(i,n2-1,0) bbit[i]=(bbit[i+1]<<1)+b[i]-'0';
    ford(i,n1-1,0) sabit[i&63][i>>6]=abit[i];
    ford(i,n2-1,0) sbbit[i&63][i>>6]=bbit[i];
    ull *p,*q;
    int num,res,tmp2;
    ull tmp;
    REP(i,m) {
        scanf("%d %d %d", &x, &y, &len);
        p=&sabit[x&63][x>>6];
        q=&sbbit[y&63][y>>6];
        num=len>>6;
        res=0;
        REP(j,num) {
            tmp=*p^*q;
            res+=cnt[tmp&0xffff]+cnt[tmp>>16&0xffff]+cnt[tmp>>32&0xffff]+cnt[tmp>>48];
            ++p;
            ++q;
        }
        ford(j,len-1,len-(len&63)) res+=(a[x+j]^b[y+j])&1;
        printf("%d\n",res);
    }
}
