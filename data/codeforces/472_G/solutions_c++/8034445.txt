#include <cstdio>
#include <algorithm>
#include <vector>
#include <cstring>

using namespace std;

#define all(a) (a).begin(),(a).end()
#define pb push_back
#define sz(a) ((int)(a).size())
#define mp make_pair
#define fi first
#define se second

typedef pair<int,int> pint;
typedef long long ll;
typedef unsigned long long ull;
typedef vector<int> vi;


#define MAX_N 200005

int n,m;
char sa[MAX_N],sb[MAX_N];
unsigned int a[32][6255],b[32][6255];

int query(int p1, int p2, int len)
{
    int pf1=p1&31,pf2=p2&31;
    int full=len/32,half=len&31;
    int ans=0;
    int i=(p1-pf1)/32,j=(p2-pf2)/32;
    for (; full; full--,i++,j++)
        ans+=__builtin_popcount(a[pf1][i]^b[pf2][j]);
    unsigned int x=1;
    for (; half; half--,x*=2)
        ans+=bool(a[pf1][i]&x)^bool(b[pf2][j]&x);
    return ans;
}

int main()
{
    scanf("%s",sa);
    n=strlen(sa);
    for (int pf=0; pf<32; pf++)
    {
        unsigned int x=1;
        int j=-1;
        for (int i=pf; i<n; i++)
        {
            if (!((i-pf)&31))
                j++,x=1;
            if (sa[i]=='1')
                a[pf][j]|=x;
            x*=2;
        }
    }
    scanf("%s",sb);
    m=strlen(sb);
    for (int pf=0; pf<32; pf++)
    {
        unsigned int x=1;
        int j=-1;
        for (int i=pf; i<m; i++)
        {
            if (!((i-pf)&31))
                j++,x=1;
            if (sb[i]=='1')
                b[pf][j]|=x;
            x*=2;
        }
    }
    int q,p1,p2,len;
    scanf("%d",&q);
    while (q--)
    {
        scanf("%d %d %d",&p1,&p2,&len);
        printf("%d\n",query(p1,p2,len));
    }
    return 0;
}
