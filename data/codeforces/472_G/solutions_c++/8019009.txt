#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define if if (
#define then )
#define do )
#define for for (
#define while while (
#define begin {
#define end }

#define MAXN 200005

#define u32 unsigned int

char ch;
inline void read(int &x)
begin
	x=0;ch=getchar();
	while ch<=32 do ch=getchar();
	while ch>32 do begin
		x=x*10+ch-48;ch=getchar();
	end;
end;

int cnt_table[1<<16];

inline void pre()
begin
	int i;
	for i=1;i<(1<<16);i++ do begin
		cnt_table[i]=cnt_table[i>>1]+(i&1);
	end;
end;

char a[MAXN],b[MAXN];
int n,m,q;

inline void main1()
begin
	read(q);
	while q-- do begin
		int p1,p2,len;
		read(p1);read(p2);read(len);
		int ans=0;
		char *A=a+p1,*B=b+p2;
		while len-- do begin
			ans+=(*A++)^(*B++);
		end;
		printf("%d\n",ans);
	end;
end;

u32 aa[32][MAXN/32+3],bb[32][MAXN/32+3];

inline void build(char *a,u32 *A,int n)
begin
	int i;
	for i=0;i<n;i++ do begin
		A[i>>5]|=a[i]<<(i&31);
	end;
end;

inline void init()
begin
	int i;
	for i=0;i<32;i++ do begin
		build(a+i,aa[i],n-i);
	end;
	for i=0;i<32;i++ do begin
		build(b+i,bb[i],m-i);
	end;
end;

inline int getcnt(u32 x)
begin
	return cnt_table[x>>16]+cnt_table[x&65535];
end;

inline int calc(u32 *A,u32 *B,int len)
begin
	int ret=0;
	int l=len>>5;
	while l-- do begin
		ret+=getcnt((*A++)^(*B++));
	end;
	return ret+getcnt(((1u<<(len&31))-1u)&(*A^*B));
end;

int main()
begin
	scanf("%s%s",a,b);
	n=strlen(a);m=strlen(b);
	int i;
	for i=0;i<n;i++ do a[i]-=48;
	for i=0;i<m;i++ do b[i]-=48;
	pre();
	if n<=35 || m<=35 then begin
		main1();
		return 0;
	end;
	init();
	read(q);
	while q-- do begin
		int p1,p2,len;
		read(p1);read(p2);read(len);
		u32 *A,*B;
		A=aa[p1&31]+(p1>>5);
		B=bb[p2&31]+(p2>>5);
		printf("%d\n",calc(A,B,len));
	end;
end
