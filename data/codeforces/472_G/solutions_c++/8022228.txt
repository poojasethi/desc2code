using namespace std;

const int N = 200010;

char a[N], b[N];
unsigned cc[1 << 16];
unsigned long long aa[66][3333], bb[66][3333];

inline unsigned count(unsigned long long x) {
  return cc[x & 65535] + cc[(x >> 16) & 65535] + cc[(x >> 32) & 65535] + cc[(x >> 48) & 65535];
}

inline int query(int pa, int pb, int len) {
  int r = 0;
  int end = len + pa;
  int xa = pa & 63, ya = pa >> 6;
  int xb = pb & 63, yb = pb >> 6;
  for (int i = 0; i + 63 < len; i += 64, pa += 64, pb += 64, ++ya, ++yb) {
    r += count(aa[xa][ya] ^ bb[xb][yb]);
  }
  for (; pa < end; ++pa, ++pb) {
    r += (a[pa] - '0') ^ (b[pb] - '0');
  }
  return r;
}

int main() {
  //assert(freopen("input.txt", "r", stdin));
  //assert(freopen("output.txt", "w", stdout));
  //std::ios::sync_with_stdio(0);
  //std::cin.tie(0);
  scanf("%s %s", a, b);
  int na = strlen(a);
  int nb = strlen(b);
  for (int i = 0; i < 65536; ++i) {
    cc[i] = __builtin_popcount(i);
  }
  for (int i = 0; i < 64; ++i) {
    for (int j = 0, k = i; k + 63 < na; ++j, k += 64)
      for (int z = 0; z < 64; ++z)
        if (a[k + z] == '1')
          aa[i][j] |= 1ULL << z;
    for (int j = 0, k = i; k + 63 < nb; ++j, k += 64)
      for (int z = 0; z < 64; ++z)
        if (b[k + z] == '1')
          bb[i][j] |= 1ULL << z;
  }
  int m;
  scanf("%d", &m);
  while (m--) {
    int pa, pb, len;
    scanf("%d %d %d", &pa, &pb, &len);
    printf("%d\n", query(pa, pb, len));
  }
}
