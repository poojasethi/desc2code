#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define unt	unsigned int
#define CH	(ch=getchar())
#define For(i,a,b)	for(int i=a;i<b;i++)

int		IN(){
		int x=0,ch;
		for	(;CH<'0'||ch>'9';);
		for	(;ch>='0'&&ch<='9';CH)	(x*=10)+=ch-'0';
		return	x;
}

char	s[200005];
int		cnt[1<<16];
unt		A[32][7005],B[32][7005];

int		Num(unt x)	{return	cnt[x>>16]+cnt[x&((1<<16)-1)];}

int		main(){
		scanf("%s",s);
		For(i,0,32)	for (int j=0;s[i+j];j++)
			if	(s[i+j]=='1')	A[i][j>>5]|=1<<(j&31);
		scanf("%s",s);
		For(i,0,32)	for (int j=0;s[i+j];j++)
			if	(s[i+j]=='1')	B[i][j>>5]|=1<<(j&31);

		For(i,1,1<<16)	cnt[i]=cnt[i>>1]+(i&1);
		for	(int Q=IN();Q--;){
			int l=IN(),r=IN(),len=IN(),Ans=0,i,j;
			for (i=l>>5,j=r>>5;len>=32;len-=32,i++,j++)
				Ans+=Num(A[l&31][i]^B[r&31][j]);
			Ans+=Num((A[l&31][i]^B[r&31][j])&((1<<len)-1));
			printf("%d\n",Ans);
		}
}
