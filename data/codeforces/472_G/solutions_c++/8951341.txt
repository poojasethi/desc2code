

#include <bits/stdc++.h>
using namespace std;
const int N = 200100;
const int M = 1<<16;
unsigned int a[32][N>>5],b[32][N>>5];
int h[M];
char sa[N],sb[N];
inline int H(const unsigned int &x){
    return h[x>>16]+h[(x&(M-1))];
}
int main(){
    scanf("%s%s",sa,sb);
    int n=strlen(sa);
    int m=strlen(sb);
    for(int i=0;i<32;++i)
        for(int j=0;i+j<n;++j){
            if(sa[i+j]=='1') 
                a[i][j>>5]|=(1U<<(j&31));
        }
    for(int i=0;i<32;++i)
        for(int j=0;i+j<m;++j){
            if(sb[i+j]=='1')
                b[i][j>>5]|= (1U<<(j&31));
        }
    for(int i=1;i<M;++i)
        h[i]=h[i>>1]+(i&1);
    int q;
    scanf("%d",&q);
    while(q--){
        int l,r,len;
        scanf("%d%d%d",&l,&r,&len);
        int i=l&31,j=r&31;
        int ca=l>>5,cb=r>>5;
        int ans=0;
        while(len>=32){
            ans+=H(a[i][ca++]^b[j][cb++]);
            len-=32;
        }
        ans+=H((a[i][ca]^b[j][cb])&((1U<<len)-1));
        printf("%d\n",ans);
    }
    return 0;
}
