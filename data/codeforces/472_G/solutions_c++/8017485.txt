#define pb push_back

#define REP(i, n) for (int i = 0; i < (int)(n); ++i)
#define foreach(e, x) for (__typeof(x.begin()) e = x.begin(); e != x.end(); ++e)
typedef long long LL;
typedef pair<int, int> PII;

string s, t;
unsigned a[32 << 13] = {}, b[32 << 13] = {};
int n, m, N, M;
int pre[1 << 16];

void go(string s, unsigned (&a)[32 << 13]) {
    REP(i, s.length()) if (s[i] == '1')
        a[i >> 5] |= (unsigned)1 << (i & 31);
    for (unsigned i = 1; i < 32; ++i) {
        REP(j, 6281) a[(i << 13) | j] = a[((i - 1) << 13) | j];
        unsigned mem = 0, tmp;
        for (int j = 6280; j >= 0; --j) {
            tmp = a[(i << 13) | j] & 1;
            a[(i << 13) | j] >>= 1;
            a[(i << 13) | j] |= mem << 31;
            mem = tmp;
        }
    }
}

inline int countBits(unsigned x) {
    return pre[x & 65535] + pre[x >> 16];
}

int q;

int main() {
    REP(i, 1 << 16) pre[i] = __builtin_popcount(i);
    ios_base::sync_with_stdio(false);
    cin >> s >> t;
    go(s, a);
    go(t, b);
    n = (int)s.length();
    N = (n + 100) / 32;
    m = (int)t.length();
    M = (m + 100) / 32;
    cin >> q;
    REP(query, q) {
        int x, y, len;
        cin >> x >> y >> len;
        int apos = (x & 31) << 13;
        int bpos = (y & 31) << 13;
        int as = x >> 5;
        int bs = y >> 5;
        apos += as;
        bpos += bs;
        int cnt = len >> 5;
        int ans = 0;
        for (int i = 0; i < cnt; ++i)
            ans += countBits(a[apos + i] ^ b[bpos + i]);
        int rem = len & 31;
        if (rem > 0) {
            ans += countBits((a[apos + cnt] ^ b[bpos + cnt]) & (((unsigned)1 << rem) - 1));
        }
        cout << ans << '\n';
    }
    return 0;
}
