#include<cmath>
#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

void open()
{
	freopen("constraints.in","r",stdin);
	freopen("constraints.out","w",stdout);
}

class character
{
	public:
		char ori[200010],end[200010];
		unsigned int a[32][200010/32],b[32][200010/32],ask,n,m,v[1<<16];
		void work()
		{
			read();
			process();
		}
	private:
		inline int val(unsigned int x)
		{
			return v[x>>16]+v[x&((1<<16)-1)];
		}
		void read()
		{
			scanf("%s%s",ori,end);
			n=strlen(ori);m=strlen(end);
			for(int i=0;i<32;++i)
				for(int j=0;i+j<n;++j)
					a[i][j>>5]|=((ori[i+j]-'0')<<(j&31));
			for(int i=0;i<32;++i)
				for(int j=0;i+j<m;++j)
					b[i][j>>5]|=((end[i+j]-'0')<<(j&31));
			for(int i=0;i<(1<<16);++i)
				v[i]=v[i>>1]+(i&1);
		}
		void process()
		{
			int A,B,L,t,x,y,u,v,ans=0;
			scanf("%d",&ask);
			for(int i=1;i<=ask;++i,ans=0)
			{
				scanf("%d%d%d",&A,&B,&L);
				x=A&31;y=B&31;u=A>>5;v=B>>5;
				while(L>=32)
				{
					ans+=val(a[x][u++]^b[y][v++]);
					L-=32;
				}
				//cout<<x<<" "<<u<<" "<<y<<" "<<v<<endl;
				ans+=val((a[x][u]^b[y][v])&((1<<L)-1));
				printf("%d\n",ans);
			}
		}
}will;

int main()
{
	//open();
	will.work();
	return 0;
}
