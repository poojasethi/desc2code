#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
using namespace std;
#define fo for
#define sf scanf
#define pf printf


char s[200005],S[3000005];
unsigned int Mask1[32][10000],Mask2[32][10000],a,b,N;
int q,g[1<<20],A[20],sn;


void precalc() {
	gets(s);
	int len = strlen(s);
	fo(int i = 0;i < len;++i) if(s[i] == '1') {
		fo(int j = 0;j < 32;++j) if(i >= j) 
			Mask1[j][(i - j)>>5] |= (1u<<((i - j)&31));
	}	
	

	gets(s);
	len = strlen(s);
	fo(int i = 0;i < len;++i) if(s[i] == '1') {
		fo(int j = 0;j < 32;++j) if(i >= j) 
			Mask2[j][(i - j)>>5] |= (1u<<((i - j)&31));
	}	

	N = (1<<16) - 1;
	fo(int i = 1;i <= N;++i) 
		g[i] = g[i>>1] + (i&1);
}


inline void add(int x) {
	if(x == 0) {
		S[sn++] = '0';
		S[sn++] = '\n';
		return;
	}
	int an = 0;
	while(x > 0) {
		A[an++] = x%10;
		x /= 10;
	}
	fo(int i = an - 1;i >= 0;--i)
		S[sn++] = char(A[i] + '0');
	S[sn++] = '\n';
}


int main () {
	#ifdef LOCAL
	freopen("in.txt","r",stdin);
	freopen("out.txt","w",stdout);
	#endif

	precalc();
	sf("%d",&q);
	int j1,j2,i1,i2,len,ans,k;
	while(q--) {
		sf("%d%d%d",&j1,&j2,&len);
		k = (len>>5);
		i1 = j1>>5;
		i2 = j2>>5;
		j1 &= 31;
		j2 &= 31;
		len &= 31;
		ans = 0;
		fo(int i = 0;i < k;++i) {
			a = (Mask1[j1][i1]^Mask2[j2][i2]);
			ans += g[a&N] + g[a>>16];
			++i1, ++i2;
		}
		a = (Mask1[j1][i1]^Mask2[j2][i2])&((1<<len) - 1);
		ans += g[a&N] + g[a>>16];
		add(ans);
//		pf("%d\n",ans);
	}

	puts(S);


	return 0;
}
