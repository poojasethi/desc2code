

#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long ULL;
const int N = 200200;
const ULL M = (1ULL<<16)-1;

ULL  a[64][N>>6],b[64][N>>6];
int h[1<<16];
char sa[N],sb[N];
inline int H(const ULL &x){
    return h[x&M]+h[(x>>16)&M]+h[(x>>32)&M]+h[(x>>48)&M];
}
int main(){
    scanf("%s%s",sa,sb);
    int n=strlen(sa);
    int m=strlen(sb);
    for(int i=0;i<64;++i)
        for(int j=0;i+j<n;++j){
            if(sa[i+j]=='1') 
                a[i][j>>6]|=(1ULL<<(j&63));
        }
    for(int i=0;i<64;++i)
        for(int j=0;i+j<m;++j){
            if(sb[i+j]=='1')
                b[i][j>>6]|= (1ULL<<(j&63));
        }
    for(int i=1;i<=M;++i)
        h[i]=h[i>>1]+(i&1);
    int q;
    scanf("%d",&q);
    while(q--){
        int l,r,len;
        scanf("%d%d%d",&l,&r,&len);
        int i=l&63,j=r&63;
        int ca=l>>6,cb=r>>6;
        int ans=0;
        while(len>=64){
            ans+=H(a[i][ca++]^b[j][cb++]);
            len-=64;
        }
        ans+=H((a[i][ca]^b[j][cb])&((1ULL<<len)-1));
        printf("%d\n",ans);
    }
    return 0;
}
