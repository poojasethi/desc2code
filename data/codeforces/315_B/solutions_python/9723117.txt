n, m = map(int, raw_input().split(' '))
x = list(map(int, raw_input().split(' ')))
inc = 0
for i in range(m):
    bad = list(map(int, raw_input().split(' ')))
    if len(bad) == 3:
        a, b, c = bad[0], bad[1], bad[2]
    else:
        a, b = bad[0], bad[1]
        
    if a == 1:
        x[b-1] = c - inc
    elif a == 2:
        inc += b
    else:
        print(x[b-1] + inc)
