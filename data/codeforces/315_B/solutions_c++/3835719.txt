#include<iostream>
#include<cstdio>
using namespace std;
int arr[100002];
int main(){
	int n,m,count=0,a,b,c,i;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		scanf("%d",&arr[i]);
	while(m--){
		scanf("%d%d",&a,&b);
		if(a==3)
			printf("%d\n",count+arr[b]);
		if(a==2)
			count+=b;
		if(a==1){
			scanf("%d",&c);
			arr[b]=c-count;
		}
	}
	return 0;
}
