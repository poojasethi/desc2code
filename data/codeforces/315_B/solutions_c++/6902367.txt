#include<cstdio>
using namespace std;
int a1[100001];
int main()
{
	int n,m,a,b,d=0,op;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;++i)
		scanf("%d",a1+i);
	while(m--)
	{
		scanf("%d%d",&op,&a);
		if(op==1)
		{
			scanf("%d",&b);
			a1[a]=b-d;
		}
		else if(op==2)d+=a;
		else printf("%d\n",a1[a]+d);
	}
}
