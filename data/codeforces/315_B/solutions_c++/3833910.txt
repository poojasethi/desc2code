#include<iostream>

using namespace std;
int a[100010];
int main(){
	int n,m,offset=0;
	cin>>n>>m;
	int p,q,r;
	for (int i=0; i<n; i++)	cin>>a[i];
	for(int i=0; i<m; i++){
		cin>>p;
		if(p==1){
			cin>>q>>r;
			a[q-1]=r-offset;
		}
		if(p==2){
			cin>>q;
			offset+=q;
		}
		if(p==3){
			cin>>q;
			cout<<(a[q-1]+offset)<<endl;
		}
		
	}
}
