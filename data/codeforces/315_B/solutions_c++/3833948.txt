#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
int inc, n, m, a[100010], i;
int main(){
	scanf("%d%d", &n, &m);
	for(i=1;i<=n;i++)scanf("%d", &a[i]);
	for(i=1;i<=m;i++){
		int x, y, t;
		scanf("%d%d", &t, &x);
		if(t == 1)scanf("%d", &y), a[x]=y-inc;
		if(t == 2)inc+=x;
		if(t == 3)printf("%d\n", a[x]+inc);
	}
	return 0;
}
