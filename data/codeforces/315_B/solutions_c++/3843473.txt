#include<iostream>
#define N 100001
#define LL long long
using namespace std;
LL a[N];
int main(){
	int n,m;cin >> n >> m;
	for(int i=1;i<=n;++i)
		cin >> a[i];
	LL acc=0;
	for(int i=0;i<m;++i){
		int t,v,x;
		cin >> t >> v;
		if(t==1)
			cin>>x, a[v]=x-acc;
		if(t==2)
			acc+=v;
		if(t==3)
			cout << a[v]+acc << endl;
	}
	return 0;
}
