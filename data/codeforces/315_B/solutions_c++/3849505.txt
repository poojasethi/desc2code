# include <iostream>
using namespace std;

	long int n,m,j,a[100009],t,k,sum;

int main()
{
	cin>>n>>m;
	
	for(int i=1; i<=n; i++)
		cin>>a[i];
	
	for(int i=0;i<m;i++)
	{
		cin>>t;
		if(t==3)
		{
			cin>>j;
			cout<<a[j]+sum<<"\n";
		}
		else if(t==2)
		{
			cin>>k;
			sum+=k;
		}
		else
		{
			cin>>j>>k;
			a[j]=k-sum;
		}
	}
}
