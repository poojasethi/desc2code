d = {}
x, y = 0, 0
for i in range(4):
    x1, y1, x2, y2 = map(int, raw_input().split());
    x += (x1 == x2 and y1 != y2)
    y += (y1 == y2 and x1 != x2)
    if (x1, y1) not in d:
        d[(x1, y1)] = 0
    if (x2, y2) not in d:
        d[(x2, y2)] = 0
    d[(x1, y1)] += 1
    d[(x2, y2)] += 1
b = d.values();
f = (x == 2 and y == 2)
for i in b:
    if(i != 2):
        f = 0;
print ("NO", "YES")[f]
