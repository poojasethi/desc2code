import sys

i = sorted([[[min(a,c),min(b,d)],[max(a,c),max(b,d)]] for a,b,c,d in [l.strip().split(" ") for l in sys.stdin.readlines()]])
v = [e for e in i if (e[0][0] == e[1][0] and e[0][1] != e[1][1])]
h = [e for e in i if (e[0][1] == e[1][1] and e[0][0] != e[1][0])]
if len(v) != 2 or len(h) != 2 or v[0][0] != h[0][0] or v[0][1] != h[1][0] or v[1][0] != h[0][1] or v[1][1] != h[1][1]: print "NO"
else: print "YES"
