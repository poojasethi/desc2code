I=lambda:map(int, raw_input().split())
d = {}
x, y = 0, 0
for i in range(4):
    x1, y1, x2, y2 = I()
    x += (x1 == x2 and y1 != y2)
    y += (y1 == y2 and x1 != x2)
    d[(x1,y1)]=d.get((x1,y1),0)+1
    d[(x2,y2)]=d.get((x2,y2),0)+1

if any(z != 2 for z in d.values()): print 'NO';exit()
print 'YES' if (x == 2 and y == 2) else 'NO'
