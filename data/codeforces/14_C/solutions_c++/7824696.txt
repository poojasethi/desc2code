#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
vector<pair<int,int> > v;
int main()
{
    int x1,y1,x2,y2;
    int a=0,b=0;
    for(int i=0;i<4;i++)
    {
        cin>>x1>>y1>>x2>>y2;
        if(x1==x2&&y1!=y2)  a++;
        if(y1==y2&&x1!=x2)  b++;
        v.push_back(make_pair(x1,y1));
        v.push_back(make_pair(x2,y2));
    }
    sort(v.begin(),v.end());
    int len = unique(v.begin(), v.end()) - v.begin();
    if (len != 4 || a != 2 || b != 2) cout << "NO\n";
   else if (v[0].first == v[1].first && v[2].first == v[3].first &&
             v[0].second == v[2].second && v[1].second == v[3].second) cout << "YES\n";
    else cout<<"No\n";
    return 0;
}
