#include<stdio.h>
#include<math.h>
#include<string.h>
int x[7],y[7],a[6];

int main()
{
	int i,j,x1,x2,y1,y2,t=0,h=0,s=0;
	memset(a,0,sizeof(a));
	for (i=1;i<=4;i++)
	{
		scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
		if ((x1==x2 && y1!=y2) || (y1==y2 && x1!=x2))
		{
			if (x1==x2) h++;else s++;
			for (j=1;j<=t;j++)
			if (x[j]==x1 && y[j]==y1) 
			{
				a[j]++;
				break;
			}
			if (j==t+1 && t<4)
			{
				x[j]=x1;
				y[j]=y1;
				t++;
			}
			
			for (j=1;j<=t;j++)
			if (x[j]==x2 && y[j]==y2) 
			{
				a[j]++;
				break;
			}
			if (j==t+1 && t<4)
			{
				x[j]=x2;
				y[j]=y2; 
				t++;
			}
		}
	}

	
	int ok=1;
	for (i=1;i<=4;i++) if (a[i]!=1)
	{
		ok=0;
		break;
	}
	if (t=4 && ok==1 && h==2 && s==2) printf("YES");
	else printf("NO");
	return 0;
}
