#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
map<pair<long,long>,int>mp;
int main()
{
    int n=4,p=0,q=0;
    while(n--)
    {
        
    long x1,y1,x2,y2;cin>>x1>>y1>>x2>>y2;
    mp[make_pair(x1,y1)]++;mp[make_pair(x2,y2)]++;
    p+=(x1==x2&&y1!=y2);q+=(x1!=x2&&y1==y2);
    //cout<<p<<' '<<q<<endl;
    }
    int ans=0,c=1;
    if(p==2&&q==2)ans=1;
    for(map<pair<long,long>,int>::iterator it=mp.begin();it!=mp.end();it++)
    {
        if(it->second!=2)
        {
            c=0;break;
        }
    }
    (ans&&c)?cout<<"YES"<<endl:cout<<"NO"<<endl;
}
