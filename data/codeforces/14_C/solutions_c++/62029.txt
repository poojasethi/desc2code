#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<set>
using namespace std;

bool init()
{
	set< pair<int,int> > s;
	int x1,y1,x2,y2,vx=0,vy=0,yl,yr,xl,xr,xx,yy;
	for(int i=0;i<4;i++)
	{
		scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
		s.insert(make_pair(x1,y1));
		s.insert(make_pair(x2,y2));
		if(x1==x2)
		{
			if(y1==y2) return false;
			if(y1>y2) swap(y1,y2);
			if(vx)
			{
				if(x1==xx || y1!=yl || y2!=yr) return false;
			}
			else yl=y1,yr=y2,xx=x1;
			vx++;
		}
		if(y1==y2)
		{
			if(x1==x2) return false;
			if(x1>x2) swap(x1,x2);
			if(vy)
			{
				if(y1==yy || x1!=xl || x2!=xr) return false;
			}
			else xl=x1,xr=x2,yy=y1;
			vy++;
		}
	}
	return vx==2 && vy==2 && s.size()==4;
}

int main()  
{
	if(init()) printf("YES\n");
	else printf("NO\n");
	return 0;
}
