#include<iostream>
#include<set>
using namespace std;

int main()
{
    int pts[4][4],i,j,k;
    for(i=0;i<4;i++)
    {
        for(j=0;j<4;j++) cin>>pts[i][j];
    }
    int a=0;
    set<int> x,y;
    for(i=0;i<4;i++)
    {
        x.insert(pts[i][0]);
        x.insert(pts[i][2]);
        y.insert(pts[i][1]);
        y.insert(pts[i][3]);
    }
    if(x.size()!=2 || y.size()!=2) cout<<"NO";
    else
    {
        set<int> px,py;
        for(i=0;i<4;i++)
        {
            if(pts[i][0]==pts[i][2] && pts[i][1]==pts[i][3]) continue;
            if(pts[i][0]==pts[i][2]) px.insert(pts[i][0]);
            if(pts[i][1]==pts[i][3]) py.insert(pts[i][1]);
        }
        if(px.size()!=2 || py.size()!=2) cout<<"NO";
        else cout<<"YES";
    }
    return 0;
}
