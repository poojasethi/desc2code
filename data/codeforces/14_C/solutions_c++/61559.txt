#include<cstdio>
#include<cstring>
#include<cmath>
#include<cstdlib>
#include<algorithm>
#include<map>
#include<functional>

using namespace std;

int main(){
	map<pair<int, int>, int> pp;
	int P[4][2][2];
	for (int i = 0 ; i < 4; ++i){
		for (int j = 0 ; j < 2; ++j){
			scanf("%d%d", &P[i][j][0], &P[i][j][1]);
			pp[make_pair(P[i][j][0], P[i][j][1])]++;
		}
	}
	bool ok = 1;
	for (map<pair<int, int>, int>::iterator i = pp.begin(); i != pp.end(); i++)
		ok &= i->second == 2;
	if (!ok) puts("NO");
	else{
		for (int i = 0 ; i < 4; ++i)
			ok &= (P[i][0][0] == P[i][1][0]) || (P[i][0][1] == P[i][1][1]);
		if (!ok) puts("NO");
		else{
		long long h = 0, w = 0;
		for(int i = 0 ; i < 4; ++i)
			if (P[i][0][0] == P[i][1][0])
				h = abs(P[i][0][1] - P[i][1][1]);
			else w = abs(P[i][0][0] - P[i][1][0]);
		puts((h&&w)?"YES":"NO");
		}
	}
	return 0;
}
