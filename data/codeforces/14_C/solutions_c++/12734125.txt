#include <bits/stdc++.h>

using namespace std;

typedef long long LL;

bool isPx(LL a, LL b, LL c, LL d) {
  return (a != c && b == d);
}

bool isPy(LL a, LL b, LL c, LL d) {
  return (a == c && b != d);
}

int main() {
  // freopen("input.in", "r", stdin);
  // freopen("output.out", "w", stdout);
  LL a, b, c, d, i;
  set<pair<LL, LL> > S, W;
  for (i = 0; i < 4; ++i) {
    cin >> a >> b >> c >> d;
    if (isPx(a, b, c, d)) {
      S.insert(make_pair(a, b));
      S.insert(make_pair(c, d));
    } else if (isPy(a, b, c, d)) {
      W.insert(make_pair(a, b));
      W.insert(make_pair(c, d));
    } else {
      puts("NO");
      return 0;
    }
  }
  if (S.size() == 4 && S == W) {
    puts("YES");
  } else {
    puts("NO");
  }
  return 0;
}
