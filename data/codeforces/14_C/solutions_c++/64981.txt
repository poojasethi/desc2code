#include <iostream>
#include <cstring>
#include <cstdio>
#include <vector>
#include <algorithm>
#include <map>
using namespace std;

#define pb push_back
#define sz size()
#define x first
#define y second
#define MAXN 
#define REP(i,n) for ((i)=1;(i)<(n);(i)++)
typedef pair<int,int>pii;
const int oo=(int)1e9;

map<pair<int,int>,int>mp;
static int x1[5],y1[5],x2[5],y2[5],nx,ny;

int main(){
    bool valid=1;
	for (int i=0;i<4;i++){
		scanf("%d%d%d%d",&x1[i],&y1[i],&x2[i],&y2[i]);
		mp[pii(x1[i],y1[i])]++;
		mp[pii(x2[i],y2[i])]++;
		if (x1[i]!=x2[i] && y1[i]!=y2[i]){
			valid=0;
		}
		if (x1[i]==x2[i]) nx++;
		if (y1[i]==y2[i]) ny++;
	}
	for (int i=0;i<4;i++){
		if (mp[pii(x1[i],y1[i])]!=2) valid=0;
		if (mp[pii(x2[i],y2[i])]!=2) valid=0;
	}
	if (nx!=2 || ny!=2) valid=0;
	if (valid) printf("YES\n");
	else printf("NO\n");
	return 0;
}
