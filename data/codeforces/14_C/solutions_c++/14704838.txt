#include <bits/stdc++.h>
using namespace std;

typedef pair<int,int> ii;

int main(){
    ios_base::sync_with_stdio(false);
    int hor = 0, ver = 0; set<ii> P;
    set<pair<ii,ii> > L;
    for(int i=0; i<4; i++){
        int x1, y1, x2, y2; 
        cin >> x1 >> y1 >> x2 >> y2;
        if((x1==x2 || y1==y2) && (x1!=x2 || y1!=y2)){
            ii p1(x1,y1), p2(x2,y2);            
            P.insert(p1);
            P.insert(p2);
            if(p1 < p2) swap(p1,p2);
            L.insert(make_pair(p1,p2));
            if(x1==x2) ver++;
            if(y1==y2) hor++;
        }   
        else{cout << "NO\n"; return 0;}
    }
    if(P.size()==4 && L.size()==4 && hor==2 && ver==2)
        cout << "YES\n";        
    else cout << "NO\n";
    return 0;
}
