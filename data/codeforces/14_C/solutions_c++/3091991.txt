#include <iostream>
#define for_up(i,n) for(int i=0;i<n;i++)
using namespace std;
int x[90],y[90];
typedef pair<int,int> point;
main()
{
  int x1,y1,x2,y2,i,j,tmp=0,tmp0=0,tmp1=0;
  point A[5],B[5];
  for_up(i,4){
                cin>>x1>>y1>>x2>>y2;
                x[i]=x2-x1;
                y[i]=y2-y1;      
                A[i]=point(x1,y1);
                B[i]=point(x2,y2);
                if(y[i]==0 || x[i]==0) tmp0++;
  }
  for_up(i,4)
    for_up(j,4){
       if (i!=j && (x[i]*x[j]+y[i]*y[j]!=0))tmp++; 
       if (i!=j && ( A[i]==A[j] || A[i]==B[j])) tmp1++;          
    }                    
  if (tmp0==4 && tmp==4 && tmp1==4) cout<<"YES";else cout<<"NO";
}
