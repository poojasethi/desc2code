#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
int main()
{
    int x1,y1,x2,y2,x=0,y=0;
    map<pair<int,int>,int>a;
    for(int i=0;i<4;i++)
    {
        cin>>x1>>y1>>x2>>y2;
        x+=(x1==x2&&y1!=y2),y+=(y1==y2&&x1!=x2);
        a[make_pair(x1,y1)]++;
        a[make_pair(x2,y2)]++;
    }
    bool f=(x==2&&y==2);
    for(map<pair<int,int>,int>::iterator i=a.begin();i!=a.end();f&=(i->second==2),i++);
    cout<<(f?"YES":"NO");
    return 0;
}
