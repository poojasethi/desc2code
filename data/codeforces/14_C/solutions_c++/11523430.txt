#include <stdio.h>

struct Point
{
    int x;
    int y;
    int num;
}ans[8];
int k = 0;

void judge(Point a)
{
    for(int j = 0; j < k; j++)
        if(a.x == ans[j].x && a.y == ans[j].y)
        {
            ans[j].num++;
            return;
        }
    ans[k].num = 1; ans[k].x = a.x; ans[k++].y = a.y;
}

int main()
{
    Point a, b;
    int r = 0;   //横边的条数

    for(int i = 0; i < 4; i++)
    {
        scanf("%d%d%d%d", &a.x, &a.y, &b.x, &b.y);
        if( (a.x != b.x && a.y != b.y) || (a.x == b.x && a.y == b.y) )
        {
            printf("NO");
            return 0;
        }
        if(a.y == b.y)
            r++;
        judge(a);
        judge(b);
    }

    if(k == 4 && ans[0].num == 2 && r == 2)
        printf("YES");
    else
        printf("NO");

    return 0;
}
