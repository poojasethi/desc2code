#include<iostream>
using namespace std;
struct point
{
    int x,y;
} p[10];
int main()
{
    int i,j,t,f;
       int a[10],count=0;
        for(i=0;i<8;i+=2){
           cin>>p[i].x>>p[i].y>>p[i+1].x>>p[i+1].y;
           f=1;
            if(p[i].x==p[i+1].x)
                count+=1;
            else if(p[i].y==p[i+1].y)
                count+=10;
            else
                f=0;
            a[i]=a[i+1]=0;
        }
        if(count!=22)
            f=0;
        for(i=0;i<8;++i)
            for(j=0;j<8;j++)  {
                if(p[i].x==p[j].x&&p[i].y==p[j].y)
                    a[i]++;
            }
        for(i=0;i<8;++i)
            if(a[i]!=2)
                f=0;
        if(f)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    return 0;
}
