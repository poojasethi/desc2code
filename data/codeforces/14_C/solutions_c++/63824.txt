#include <cstdio>
#include <map>
#include <algorithm>
using namespace std;

int main() {
#ifndef ONLINE_JUDGE
    freopen("date.in", "r", stdin);
    freopen("date.out", "w", stdout);
#endif

    int horiz = 0, vert = 0;
    map<pair<int, int>, int> points;

    for (int i = 0; i < 4; ++i) {
        int x1, x2, y1, y2;
        scanf("%d %d %d %d", &x1, &y1, &x2, &y2);

        if (x1 == x2 && y1 != y2) ++vert;
        if (y1 == y2 && x1 != x2) ++horiz;
        ++points[make_pair(x1, y1)];
        ++points[make_pair(x2, y2)];
    }

    int ok = horiz == 2 && vert == 2 && points.size() == 4;
    for (map<pair<int, int>, int>::iterator it = points.begin(); it != points.end(); ++it)
        ok &= it->second == 2;

    printf("%s\n", ok ? "YES" : "NO");
}
