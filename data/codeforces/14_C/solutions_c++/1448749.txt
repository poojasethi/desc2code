
#include <iostream>
#include <cstdio>
#include <cstring>
#include <set>

using namespace std;

set< pair<int, int> > s1, s2, s3;

int main() {
	int cnt1 = 0, cnt2 = 0, x1, y1, x2, y2, flag = true;
	for (int i = 1; i <= 4; ++i) {
		cin >> x1 >> y1 >> x2 >> y2;
		pair<int, int> p1 = make_pair(x1, y1);
		pair<int, int> p2 = make_pair(x2, y2);
		if (p1 == p2) flag = false;
		if (x1 == x2) {
			++cnt1;
			s2.insert(p1), s2.insert(p2);
		} else if (y1 == y2) {
			++cnt2;
			s3.insert(p1), s3.insert(p2);
		} else flag = false;
		s1.insert(p1), s1.insert(p2);
	}
	if (flag && cnt1 == 2 && cnt2 == 2 && s1.size() == 4 && s2.size() == 4 && s3.size() == 4) {
		cout << "YES" << endl;
	} else {
		cout << "NO" << endl;
	}
	return 0;
}
