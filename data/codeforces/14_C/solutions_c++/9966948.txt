#include <iostream>
#include <cstdio>
#include <map>
using namespace std;

int x1, y1, x2, y2, ngang = 0, doc = 0;
map<pair<int, int>, int> a;

int main () {
    //freopen("test02.in", "r", stdin);
    //freopen("", "w", stdout);
    for (int i = 1; i <= 4; i++) {
        cin >> x1 >> y1 >> x2 >> y2;
        if (x1 == x2 && y1 != y2) doc++;
        if (x1 != x2 && y1 == y2) ngang++;
        a[make_pair(x1, y1)]++;
        a[make_pair(x2, y2)]++;
    }
    if (doc != 2 || ngang != 2) {
        cout << "NO\n";
        return 0;
    }
    for (map<pair<int, int>, int>::iterator i = a.begin(); i != a.end(); i++)
        if (i->second != 2) {
            cout << "NO\n";
            return 0;
        }
    cout << "YES\n";
    return 0;
}
