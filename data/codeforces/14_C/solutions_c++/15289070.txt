#include<bits/stdc++.h>

using namespace std;

#define mod 1000000007
#define LL long long

int main()
{
    int x1,y1,x2,y2,i;
    map<pair<int,int>,int> mp;
    map<pair<int,int>,int>::iterator it;
    int x,y;

    x=y=0;
    for(i=0;i<4;i++)
    {
        cin>>x1>>y1>>x2>>y2;
        x+=(x1==x2 && y1!=y2);
        y+=(y1==y2 && x1!=x2);

        mp[make_pair(x1,y1)]++;
        mp[make_pair(x2,y2)]++;
    }

    bool rec = (x==2 && y==2);
    for(it=mp.begin();it!=mp.end();it++)
        rec&=(it->second==2);

    if(rec)
        cout<<"YES";
    else
        cout<<"NO";
    return 0;
}
