#include <bits/stdc++.h>

using namespace std;

typedef long long LL;

int main() {
  // freopen("input.in", "r", stdin);
  // freopen("output.out", "w", stdout);
  LL a, b, c, d, i;
  set<pair<LL, LL> > S, W;
  for (i = 0; i < 4; ++i) {
    cin >> a >> b >> c >> d;
    if (b == d) {
      S.insert(make_pair(a, b));
      S.insert(make_pair(c, d));
    } else if (a == c) {
      W.insert(make_pair(a, b));
      W.insert(make_pair(c, d));
    }
  }
  if (S.size() == 4 && S == W) {
    puts("YES");
  } else {
    puts("NO");
  }
  return 0;
}
