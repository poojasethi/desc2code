#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

struct point{int  x,y; void input(){cin>>x>>y;}};


int main()
{
    point p[8];
    int count,cx=0,cy=0,sx=0,sy=0;
    bool flag=true;
    for(int i=0;i<8;i++)p[i].input();
    for(int i=0;i<8;i++)
    {
        count=0;
      for(int j=0;j<8;j++)
        if(i!=j and p[i].x==p[j].x and p[i].y==p[j].y)count++;
        if(count!=1)flag=false;
    }
    for(int i=0;i<7;i+=2)
      if(p[i].x==p[i+1].x) sx++;
      else if(p[i].y==p[i+1].y) sy++;
    if(sx!=2 or sy!=2) flag =false;
    for(int i=1;i<8;i++)
    {
        if(p[0].x==p[i].x)cx++;
        if(p[0].y==p[i].y)cy++;
    }
    if(cx>=7 or cy>=7)flag=false;
    if(flag)cout<<"YES"<<endl;
    else cout<<"NO"<<endl;


}
