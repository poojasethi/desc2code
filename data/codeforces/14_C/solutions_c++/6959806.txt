#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
struct node {
	int x1,y1,x2,y2;
};
typedef struct node node;
vector<int> xval,yval;
int main()
{
	node a[4];
	int i,j,k;
	int cy=0,cx=0;
	
	for(i=0;i<4;i++) {
		scanf("%d %d %d %d",&a[i].x1,&a[i].y1,&a[i].x2,&a[i].y2);
		if(a[i].x1==a[i].x2)
			cx++;
		else if(a[i].y1==a[i].y2)
			cy++;
		xval.push_back(a[i].x1);
		xval.push_back(a[i].x2);
		yval.push_back(a[i].y1);
		yval.push_back(a[i].y2);
	}
	sort(xval.begin(),xval.end());
	sort(yval.begin(),yval.end());
	int dx=1,dy=1;
	int px=0,py=0;
	for(i=1;i<8;i++)
	{
		if(xval[i]!=xval[i-1])
		{
			dx++;
			px=i;
		}
		if(yval[i]!=yval[i-1])
		{
			dy++;
			py=i;
		}
	}
//	printf("%d %d %d %d\n",cx,cy,dx,dy);
	if(dx==2&&dy==2&&cx==2&&cy==2&&px==4&&py==4)
		printf("YES\n");
	else
		printf("NO\n");
}
