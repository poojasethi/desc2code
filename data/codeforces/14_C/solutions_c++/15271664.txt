#include <bits/stdc++.h>
#define st first
#define nd second
using namespace std;

typedef pair<int,int> ii;

map<ii,int> m;
ii x,y;
int a,b;

int main() {
    for (int i=0; i<4; i++) {
        cin >> x.st >> x.nd >> y.st >> y.nd;
        if ((x.st!=y.st)&&(x.nd!=y.nd)) { cout << "NO\n"; return 0; }
        if (x.nd==y.nd) a=abs(x.st-y.st); else b=abs(x.nd-y.nd);
        m[x]++; m[y]++;
    }
    if ((m.size()!=4)||(a==0)||(b==0)) cout << "NO\n"; else {
        for (map<ii,int>::iterator it=m.begin(); it!=m.end(); it++) 
            if (it->second!=2) { cout << "NO\n"; return 0; }
        cout << "YES\n";
    }
}
