#include <cstdio>
#include <map>
#include <algorithm>

using namespace std;


int main() {

//	freopen("C.in","r",stdin);

    map<pair<int,int>, int> M;
    int H = 0, V = 0;
    for (int i = 0; i < 4; i++) {
        int x1, y1, x2, y2;
        scanf("%d %d %d %d", &x1, &y1, &x2, &y2);
        if (x1==x2 && y1!=y2) V++;
        if (y1==y2 && x1!=x2) H++;
        M[make_pair(x1,y1)]++;
        M[make_pair(x2,y2)]++;
    }
    bool ok = H==2&&V==2;
    for (map<pair<int,int>, int>::iterator it = M.begin(); it != M.end(); it++)
        if (it->second != 2) ok = 0;

    if (ok) puts("YES");
    else puts("NO");

    return 0;
}
