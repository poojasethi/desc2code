#include <iostream>
using namespace std;

int main(){
    pair <int, int> one[4];
    pair <int, int> two[4];

    for (int i = 0; i < 4; i++){
        cin >> one[i].first >> one[i].second >> two[i].first >> two[i].second;
        if (one[i].first != two[i].first && one[i].second != two[i].second){
            cout << "NO";
            return 0;
        }
    }
    int FIRST = 0, SECOND = 0;
    for (int i = 0; i < 4; i++){
        for (int w = 0; w < 4; w++){
            if (one[i].first == one[w].first || one[i].first == two[w].first)FIRST++;

            if (one[i].second == one[w].second || one[i].second == two[w].second)SECOND++;
        }
    }
    if (FIRST == SECOND && FIRST == 12){
        cout << "YES";
    }
    else{
        cout << "NO";
    }
}
