#include <iostream>
#include <utility>
#include <map>
using namespace std;

int main()
{
	map<pair<int, int>, int> p;
	int x1, y1, x2, y2, x = 0, y = 0;
	for(int i = 0; i < 4; i++)
	{
		cin >> x1 >> y1 >> x2 >> y2;
		x += (y1 == y2) && (x1 != x2);
		y += (x1 == x2) && (y1 != y2);
		p[make_pair(x1, y1)]++;
		p[make_pair(x2, y2)]++;
	}
	bool flag = (x == 2 && y == 2);
	for(map<pair<int, int>, int>::iterator it = p.begin(); it != p.end(); it++)
	{
		flag = flag && it->second == 2;
	}
	if(flag)	cout << "YES" << endl;
	else	cout << "NO" << endl;
	return 0;
}
