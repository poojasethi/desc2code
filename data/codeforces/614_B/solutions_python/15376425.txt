n = int(raw_input())
a = raw_input().split()

def isbeautiful(i):
    return i[0] == "1" and (set(i[1:]) == set(["0"]) or len(i) == 1)

zero = False
nb = "1"
b = 0
for i in a:
    if i == "0":
        zero = True
        break
    else:
        if isbeautiful(i):
            b += len(i)-1
        else:
            nb = i

print 0 if zero else (nb + ("0"*b))
