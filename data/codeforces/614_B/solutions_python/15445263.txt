n=input()
lst = raw_input().split()
ans = '1'
zeros = ''
for i in xrange(n):
   tmp = ''
   if lst[i][0]=='0':
      ans='0'
      zeros = ''
      break
   elif lst[i][0]!='1':
      ans=lst[i]
      continue
   for x in xrange(1,len(lst[i])):
      if lst[i][x]=='0':
	     tmp += '0'
      else:
         ans = lst[i]
         break
   if ans!=lst[i]:
      zeros += tmp
print ans+zeros
