n = input()
arr = raw_input().split()
se = False
tot = 0
fl = False
for item in arr:
	ze = 0
	on = 0
	l = len(item)
	for it in item:
		if it=='1':
			on += 1
		elif it=='0':
			ze += 1
	if l==ze+on and on==1:
		tot += ze
	elif l==ze+on and on==0:
		ans = 0
		fl = True
		break
	else:
		non = item
		se = True
if fl:
	print ans
else:
	if se:
		ans = non + '0'*tot
	else:
		ans = '1'+'0'*tot
	print ans
