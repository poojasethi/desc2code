def main():
	n = int(raw_input())
	tanks = raw_input().split()
	ugly = ''
	for i in tanks:
		if (i == '0'):
			print '0'
			return
		elif (i.count('1') > 1):
			ugly = i
		elif (i != '1'):
			for j in range(2,10):
				if str(j) in i:
					ugly = i
	zeros = 0
	if (ugly != ''):
		tanks.remove(ugly)
	else:
		ugly = '1'
	for i in tanks:
		zeros += i.count('0')
	solution = ''
	for i in range(zeros):
		solution += '0'
	print ugly + solution

main()
