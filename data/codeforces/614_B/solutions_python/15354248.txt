n = input()
nums = raw_input().split()
zcnt,prod = 0,''
for num in nums:
    tzcnt = num.count('0')
    toncnt = num.count('1')
    cnt = len(num)
    if tzcnt + toncnt == cnt and toncnt <= 1: 
        if toncnt == 0:         
            prod = '0'
            break
        else:
            zcnt += tzcnt
    else:
        prod = num 
if prod != '0':
    if prod == '':
        prod = '1'
    prod = prod + '0'*zcnt


print prod
