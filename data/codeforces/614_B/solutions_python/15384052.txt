n = input()
ans = '1'
cnt = 0 #num of zero
a = map(str, raw_input().split())
for ax in a:
    if ax == '0':
        #zero
        print '0'
        break
    elif ax.count('1') == 1 and ax.count('0') == len(ax) - 1:
        #beautiful
        cnt += len(ax) - 1
    else:
        #not beautiful
        ans = str(ax)
else:
    print ans + '0' * cnt
