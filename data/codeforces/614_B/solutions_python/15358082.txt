n = int(raw_input())


def is_beauty(a):
    if a[0]!='1':
        return False

    for c in a[1:]:
        if c != '0':
            return False

    return True


def solve():
    nonb = 1
    zeros = 0
    for x in raw_input().split():

        if x == '0':
            return '0'

        elif nonb == 1 and not is_beauty(x):
            nonb = x
        else:
            zeros += len(x) - 1


    return str(nonb) + '0'*zeros

print solve()
