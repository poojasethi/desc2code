n = input()
num = raw_input().split()
noz = 0
ans_z = False
pr = "1"
for i in num:
    x = i.count('1')
    y = i.count('0')
    l = len(i)
    if(y==l):
        ans_z=True
        break
    if((x+y)!=l ):
        pr = i
    elif(x!=1):
        pr = i
    elif(l == (x+y)):
        noz+=y
if(ans_z==True):
    print 0
else:
    print pr+"0"*noz
