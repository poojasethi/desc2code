n = input()
a = raw_input().split()
p = None
zero_cnt = 0
for x in a:
    if x == '0':
        print 0
        exit()
    if x == '1':
        continue
    if x[0] == '1':
        for i in xrange(1, len(x)):
            if x[i] != '0':
                p = x
                break
        else:
            zero_cnt += len(x)-1
    else:
        p = x

if p:
    print p + ''.join(['0' for _ in xrange(zero_cnt)])
else:
    print '1' + ''.join(['0' for _ in xrange(zero_cnt)])
