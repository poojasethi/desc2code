#!/usr/bin/python
import sys
n = int(sys.stdin.readline().strip())
nn = sys.stdin.readline().split()[:n]
m = '1'
zeroes = 0
for n in nn:
  n = n.lstrip('0')
  dig = n.rstrip('0')
  if not dig:
    print 0
    break
  elif dig!='1': m = dig
  zeroes += len(n) - len(dig)
else:
  print ''.join([m]+ ['0']*zeroes)
