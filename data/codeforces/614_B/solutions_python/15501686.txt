import sys


def check(s):
    if s[0] != '1':
        return False
    else:
        for ch in s[1:]:
            if ch != '0':
                return False
    return True

n = int(sys.stdin.next())
a = map(lambda x: x.strip(), sys.stdin.next().split(' '))


p = '1'
z = 0
stop = False

for d in a[:n]:
    if d == '0':
        print(0)
        stop = True
        break
    elif check(d):
        z += len(d) - 1
    else:
        p = d

if not stop:
    print(p + '0' * z)
