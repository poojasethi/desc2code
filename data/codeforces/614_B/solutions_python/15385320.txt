def solve():
   n = input()
   a = raw_input().split()
   count = 0
   temp=''
   for x in a:
      if x[0]=='1':
         if x.count('0')==len(x)-1:
            count+=len(x)-1
         else:
            temp=x
      else:
         if x[0]=='0':
            return 0
         temp=x
   if temp=='':
      return '1'+'0'*count
   return temp+'0'*count
print solve()
