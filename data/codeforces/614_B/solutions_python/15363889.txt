#! /usr/bin/env python

def main():
  n = int(raw_input())
  s = raw_input()
  cnt = 0
  save = ""
  for num in s.split(" "):
    if num[0] == "0":
      print "0"
      return
    if num[0] == "1":
      ok = True
      add = 0
      for i in range(1, len(num)):
        if num[i] != "0":
          ok = False
          break
        else:
          add += 1

      if not ok:
        save = num
      else:
        cnt += add
    else:
      save = num
  if len(save) == 0:
    save = "1"
  print save + ("0" * cnt)

main()
