n=input()
count = 0
diff= "1"
strings = raw_input().split(' ')
flag = True
for i in xrange(n):
	string = strings[i]
	j=0
	if(string.count('0')==len(string)):
		flag = False
		continue
	while(string[j]=='0' and j<len(string)):
		j+=1
	string = string[j:]
	zeros = string.count('0')
	ones = string.count('1')
	l=len(string)
	if ones>1 or ones+zeros<l:
	 	diff=string
	else:
	 	count+=zeros

if flag:
	print diff+'0'*count
else:
	print 0
