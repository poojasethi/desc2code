n = int(input())
tanks = raw_input().split(" ")
ans = ""
isZero = False
hasUgly = False
for tank in tanks:
	if tank == "0" or (tank[0] == "1" and tank.count("0") == len(tank) - 1):
		if tank == "0":
			isZero = True
		ans += (len(tank) - 1) * "0"
	else:
		hasUgly = True
		ans = tank + ans
if isZero:
	print 0
else:
	if hasUgly:
		print ans
	else:
		print "1" + ans
