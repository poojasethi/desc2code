a = raw_input().split()

def func(x):
  ret = 0
  for i in x:
    if(i=="0"):
      ret+=1
  if(ret==len(x)-1 and x[0]=="1"):
    return True
  return False

cnt = 0
temp="1"
for x in a:
  if(x=="0"):
    print 0
    exit()
  if(func(str(x))):
    cnt+=len(x)-1
  else:
    temp = str(x)
ans=temp
while(cnt>0):
  ans=ans+"0"
  cnt-=1
