n = input()
a = raw_input().split()
nz = 0
nb = "1"
for i in xrange(n):
    co = a[i].count("1")
    cz = a[i].count("0")
#    print "x ",co,cz
    if a[i] == "0":
        print 0
        exit()
    elif co <= 1 and co + cz == len(a[i]):
        nz += cz
    else :
        nb = a[i]
print nb + "0"*nz
