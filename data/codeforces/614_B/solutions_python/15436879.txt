def multiply(l):
	lead, zeroCnt = None, 0

	for t in l:
		cnt = 0

		if(t == "0"):
			return 0
		elif(lead != None or t == "1" + "0" * (len(t) - 1) ):
			zeroCnt += len(t) - 1
		else:
			lead = t

	if(lead == None):
		lead = "1"
	lead += "0"*zeroCnt	
	return lead
