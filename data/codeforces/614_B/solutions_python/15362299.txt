n = int(raw_input())
a = raw_input().split()
for v in a:
	if v == "0":
		print 0
		exit(0)

bad = "1"
zeros = 0

def is_good(v):
	if v[0] != '1':
		return False
	for i in range(1, len(v)):
		if v[i] != '0':
			return False
	return True

for v in a:
	if is_good(v):
		zeros = zeros + len(v) - 1
	else:
		bad = v
ans = bad + '0' * zeros
print ans
