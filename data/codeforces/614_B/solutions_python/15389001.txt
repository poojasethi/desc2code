def isBeautiful(a):
    if a[0]=='1':
        for ii in range(1,len(a)):
            if a[ii]!='0':
                return False
    else: return False
    return True

n = input()
a = raw_input().split()
ans = '1'
zeros = 0
for ii in a:
    if ii=='0':
        ans = 0
        break
    if isBeautiful(ii): zeros+=len(ii)-1
    else: ans = ii
if ans==0: print ans
else:
    ans+='0'*zeros
    print ans
