n = int(raw_input())
lst = raw_input().split()
c = "1"
ans = 0
f = 1
for i in range(n):
	if len(lst[i]) == 1 and lst[i][0] == '0':
		print 0
		f = 0
		break
	if lst[i].count('1') + lst[i].count('0') != len(lst[i]) or lst[i].count('1') > 1:
		c = lst[i]
	else:
		ans += lst[i].count('0')
if f:
	print c + '0' * ans
