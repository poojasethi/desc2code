def is_bad(s):
    res = False
    if s[0] != '1':
        res = True
    if len(s) > 1 and int(s[1:]) != 0:
        res = True
    return res

n = int(raw_input())
a = raw_input().split()
if '0' in a:
    print 0

    s += len(a[i])-1
    if is_bad(a[i]):
        bad = i
if bad == -1:
