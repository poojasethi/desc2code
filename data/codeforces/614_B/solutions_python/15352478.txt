n = int(raw_input())
a = raw_input().split(' ')
if a.__contains__('0'):
    print 0
else:
    ans = '1'
    for i in a:
        t = i.split('0')
        if t.count('') == len(i) - 1 and t.__contains__('1'):
            ans += '0' * (len(i) - 1)
        else:
            q = len(i)
            while i[q - 1] == '0':
                q -= 1
            ans = ''.join(i[:q]) + ''.join(ans[1:]) + '0' * (len(i) - q)

    print ''.join(ans)
