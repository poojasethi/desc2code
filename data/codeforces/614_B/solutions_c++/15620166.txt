#include <iostream>

using namespace std;

int main()
{
    int n,pro=0,zero;
    string s,c;
    cin>>n;

    for(int i=0;i<n;i++){
        cin>>s;
        zero=0;
        for(int i=0;i<(int)s.size();i++)
            zero+=s[i]-'0';

        if(zero==0){
             cout<<0; return 0;}

        else if(zero==1)
            pro+=s.size()-1;
        else
            c=s;
    }
    cout<<c;
    if(c.size()==0)
        cout<<1;
    for(int i=0;i<pro;i++)
        cout<<0;
    return 0;
}
