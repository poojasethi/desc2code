#include <bits/stdc++.h>
using namespace std;

int main(){
	int n;
	cin>>n;
	string s;
	int zero=0;
	bool flag=0;
	string ss="1";
	while(n--){
		cin>>s;
		if(s.length()==1 && s[0]=='0'){
			puts("0");
			return 0;
		}
		bool ok=s[0]=='1';
		for(int i=1;i<s.length();i++)
			if(s[i]!='0'){
				ok=0;
				break;
			}
		if(ok) zero+=s.length()-1; else ss=s;
	}
	cout<<ss;
	while(zero--) cout<<0;
	puts("");
	return 0;
}
