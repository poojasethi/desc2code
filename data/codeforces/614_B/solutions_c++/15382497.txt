#include<iostream>

using namespace std;

int main(){
    int n,i,j,ans=0,tmp;
    string s,m="1";
    cin>>n;
    for(i=0;i<n;i++){
        cin>>s;
        tmp=0;
        for(j=0;j<s.length();j++){
            tmp+=s[j]-'0';
        }
        if(tmp==0){
            cout<<0;
            return 0;
        }
        if(tmp==1){
            ans+=s.length()-1;
        }
        else{
           m=s;
        }
    }
    cout<<m;
    for(i=0;i<ans;i++){
        cout<<0;
    }

    return 0;
}
