#include <iostream>
using namespace std;
int n, zero;
string s, beg;

int beautiful()
{
	int one = 0;
	for(int i = 0; i < s.size(); i++)
	{
		if(s[i] == '1') one++;
		else if(s[i] != '0') return 0;
		if(one == 2) return 0;
	}

	return 1;
}

int main()
{
	cin >> n;
	while(n--)
	{
		cin >> s;
		if(s == "0") { cout << 0; return 0; }
		if(beautiful()) zero += s.size() - 1;
		else beg = s;
	}

	if(beg == "") beg = "1";
	cout << beg;
	while(zero--) cout << 0;
}
