#include <bits/stdc++.h>

using namespace std;
#define ll long long

int main(){
    int n;cin>>n;
    string a,t="1";
    int p=0,k=0;
    while(n--){
        cin>>a;int r = 0,y=0;
        if(a.size()==1&&a[0]=='0'){t="0";k=1;p=0;}
        for(int i=0;i<a.size()&&!k;i++){
            if(a[i]=='0'&&r>0)y++;
            else if(a[i]=='1'&&r==0)r++;
            else{
                t=a;y=0;break;
            }
        }
        p+=y;
    }
    while(p--)t+='0';
    cout<<t<<"\n";
}
