#include <bits/stdc++.h>
using namespace std;
string a[100005];
int n;
long long z;
int main(){
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    sort(a,a+n);
    for(int i=0;i<n-1;i++){
        if(a[i]=="0"){
            cout<<"0";
            return 0;
        }
        else {
            z+=a[i].size()-1;
        } 
    }
    cout<<a[n-1];
    for(int i=0;i<z;i++){
        cout<<"0";
    }
    return 0;
}
