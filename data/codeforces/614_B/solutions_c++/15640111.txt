#include<cstdio>
#include<cstring>
using namespace std;
char s[100010],now[100010];
int n,tot,i,j,l;
int main()
{
	scanf("%d\n",&n);
	tot=0;
	for (i=1;i<=n;i++)
	{
		scanf("%s",s);
		if (s[0]=='0') {printf("0\n");return 0;}
		l=strlen(s);
		bool flag=1;
		if (s[0]=='1')
		{
			for (j=1;j<l;j++) if (s[j]!='0') flag=0;
			if (flag) tot+=l-1;
		}
		else flag=0;
		if (!flag) strcpy(now,s);
    }
    if (strlen(now)!=0)printf("%s",now);else printf("1");
    for (i=1;i<=tot;i++) printf("0");
    printf("\n");
	return 0;
}
