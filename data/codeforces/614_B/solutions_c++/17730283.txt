#include <bits/stdc++.h>
using namespace std;
int main(){
	int n,tam,zeros=0;
	string cad,aux="1";
	cin>>n;
	bool cd=true;
	for (int i = 0; i < n; ++i){
		cin>>cad;
		if(cad[0]!='0'){
			while(cad[cad.size()-1]=='0'){
				cad.erase(cad.size()-1);
				zeros++;
			}
			if(aux=="1")
				aux=cad;
		}
		else{
			aux="0";cd=false;
		}
	}
	cout<<aux;
	if(cd){	
		for (int i = 0; i < zeros; ++i){
				cout<<"0";
		}
	}
//	system("pause");	
	return 0;
}
