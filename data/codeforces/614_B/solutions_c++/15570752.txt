#include<bits/stdc++.h>
using namespace std;

#define ll unsigned long long  
int main()
{
	int n,c=0;
	string ans="1";
	cin>>n;
	for(int i=0;i<n;i++)
	{
		int d=0;
		string s;
		cin>>s;
		for(int j=0;j<s.size();j++)
		{
			d+=int(s[j])-48;
		}
		if(d==0)
		{
			cout<<0;
			return 0;
		}
		if(d==1)
		{
			c+=(s.size()-1);
		}
		else ans=s;
	}
	cout<<ans;
	for(int i=0;i<c;i++)cout<<0;
	return 0;
}
