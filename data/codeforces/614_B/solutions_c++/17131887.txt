#include<iostream>
#include<string>
using namespace std;

int main(){
    string s, ans = "1";
    int n, count=0;
    cin >> n;
    for(int i=0 ; i< n ; i++){
        cin >> s;
        if(s == "0"){
            cout << "0";
            return 0;
        }
        while(s[s.size()-1] == '0'){
            count++;
            s.erase(s.size()-1);
        }
        if(ans == "1"){
            ans = s;
        }
    }
    cout << ans;
    for(int i=0 ; i<count ; i++){
        cout << "0";
    }
    return 0;
}
