#include<iostream>
#include<string>
using namespace std;

bool check(string s){
	if(s[0]!='1')
		return 0;
	for(int i=1;i<s.length();i++){
		if(s[i]!='0')
			return 0;
	}
	return 1;
}

int main(){
	int n,k=0,t=0;
	string a[100001];
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
		if(a[i][0]=='0'){
			cout<<0;
			return 0;
		}
		if(!check(a[i]))
			k=i;
		else
			t+=(a[i].length()-1);
	}
	if(k==0)
		cout<<1;
	else
		cout<<a[k];
	for(int i=1;i<=t;i++)
		cout<<0;

return 0;
}
