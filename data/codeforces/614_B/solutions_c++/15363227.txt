#include<bits/stdc++.h>
#define LL long long
using namespace std;
int main()
{

    LL n;
    cin>>n;
    string s[n];
    for(LL i=0;i<n;i++)
        cin>>s[i];
        
    sort(s,s+n);

     
     LL cnt=0;
     for(LL int i=n-2;i>=0;i--)
     {
        if(s[i]=="0")
        {
            cout<<"0";return 0;
        }
        else
            cnt+=(s[i].size()-1);
     }
     cout<<s[n-1];
     for(LL i=0;i<cnt;i++)cout<<"0";

















    return 0;
}
