#include<iostream>

using namespace std;



int main() {
  int n, c = 0, o = 0, z = 0, zs = 0;
  string s, nb = "1";
  cin>>n;
  for (int i=0;i<n;i++) {
    z = o = 0;
    cin>>s;
    if (s == "0")
      zs = 1;
    for (int j=0;j<s.size();j++)
      if (s[j] == '1')
	o++;
      else if (s[j] == '0')
	z++;
    if (o + z == s.size() && o <= 1)
      c += z;
    else
      nb = s;
  }
  
  if (zs)
    cout<<0<<endl;
  else {
    cout<<nb;
    for (int i=0;i<c;i++)
      cout<<0;
    cout<<endl;
  }
  return 0;
}
