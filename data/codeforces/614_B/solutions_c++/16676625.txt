#include<bits/stdc++.h>
using namespace std;
string ans="1",ar;
int main(){
	int n,ss=0;
	scanf("%d",&n);
	for(int i=1 ; i<=n ; i++){
		cin >> ar;
		int cnt=1;
		for(int j=1 ; j<ar.size() ; j++){
			if(ar[j]!= '0')
				cnt=0;
		}
		if(ar[0]>'1')
			cnt=0;
		if(ar.size()==1 && ar[0]=='0'){
			printf("0\n");
			return 0;
		}
		if(cnt)
			ss+=ar.size()-1;
		else
			ans=ar;
	}
	cout << ans;
	for(int i=1 ; i<=ss ; i++){
		printf("0");
	}
	return 0;
}
