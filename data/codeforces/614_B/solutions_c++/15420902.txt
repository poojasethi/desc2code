#include <iostream>
#include <string>

using namespace std;

int main()
{
    int n, x=0;
    string s, ans="1";
    cin >> n;
    while (n-->0) {
        cin >> s;
        int c=0;
        for (int j=0; j<s.size(); j++)
            c+=s[j]-'0';
        if (c==0) {
            cout << '0';
            return 0;
        }
        else if (c==1) {
            x+=s.size()-1;
        }
        else {
            ans=s;
        }
    }
    cout << ans;
    for (int i=0; i<x; i++) cout << '0';
    return 0;
}
