#include<cstdio>
#include<string>
#include<iostream>
using namespace std;
int main(){
	int n;
	bool zero=false;
	int zc=0;
	string nb="1";
	string tmp;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		cin>>tmp;
		if(tmp=="0") zero=true;
		else{
			bool b=(tmp[0]=='1');
			for(int i=1;i<tmp.size();i++){
				if(tmp[i]!='0'){
					b=false;
					break;
				}
			}
			if(b) zc+=tmp.size()-1;
			else{
				nb=tmp;	
			}
		}
	}
	if(zero) puts("0");
	else{
		cout<<nb;
		for(int i=0;i<zc;i++) putchar('0');
		puts("");
	}
	return 0;
}
