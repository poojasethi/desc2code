#include <iostream>
using namespace std;

int main(){
    int n;  cin>>n;
    string r1 = "",s,r2="1";
    while(n--){
        cin>>s;
        int n1=0,n2=0;
        if(s[0]=='0')   {   cout<<0;    return 0;   }

        for(int i=0;i<s.length();i++){
            if(s[i]=='1')   n1++;
            else if(s[i]>'1')   n2++;
        }
        if(n1>1||n2>0)  r2 = s;
        else for(int i=1;i<s.length();i++)  r1+="0";
    }
    cout<<r2<<r1;
}
