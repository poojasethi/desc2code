#include <bits/stdc++.h>
using namespace std;
char s[100005],ret[100005];
int main(){
	int l=0,n,i,j;cin>>n;
	bool zero=false;
	for(i=0;i<n;++i){
		scanf("%s",s);
		int cc=0;
		for(j=0;j<strlen(s);++j){
			if(s[j]=='0' || s[j]=='1'){
				if(s[j]=='1')++cc;
			}
			else
				cc=10000;
		}
		if(cc==1)
			l+= strlen(s)-1;
		else{
			if(cc==0)zero=true;
			else
				strcpy(ret,s);
		}
	}
	if(zero){
		cout<<0<<endl;return 0;
	}
	else{
		if(strlen(ret)!=0)
		printf("%s",ret);
		else
			putchar('1');
		for(i=0;i<l;++i)putchar('0');
	}
}
