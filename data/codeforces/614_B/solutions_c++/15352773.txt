#include <iostream>
#include <string>

using namespace std;
int is_b(string& x) {
  if (x[0] != '1') return 0;
  for (int i = 1; i < x.size(); ++i) if (x[i] != '0') return 0;
  return 1;
}

int main() {
  int n; cin >> n;
  string num = "1";
  int total = 0;
  int zero = 0;
  while (n--) {
    string x; cin >> x;
    if (x == "0") zero = 1;
    else if (is_b(x)) total += x.size() - 1; 
    else num = x;
  }
  if (zero) cout << "0\n";
  else {
    cout << num;
    while (total--) cout << "0";
    cout << '\n';
  }
  return 0;
}
