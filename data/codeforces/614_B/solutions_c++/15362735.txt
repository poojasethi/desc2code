#include <cstdio>
#include <cstring>

char s[111111], t[111111];

bool judge() {
	if(s[0] != '1') return false;
	for(int i = 1; i < strlen(s); ++ i) {
		if(s[i] != '0') return false;
	}
	return true;
}

int main () {
	int n; scanf("%d", &n);
	int cnt = 0;
	strcpy(t, "1");
	while(n --) {
		scanf("%s", s);
		if(!strcmp(s, "0")) {
			while(n --) scanf("%s", s);
			printf("0\n");
			return 0;
		} else {
			if(!judge()) strcpy(t, s);
			else cnt += strlen(s) - 1;
		}
	}
	printf("%s", t);
	for(int i = 1; i <= cnt; ++ i) printf("0");
	printf("\n");
	return 0;
}
