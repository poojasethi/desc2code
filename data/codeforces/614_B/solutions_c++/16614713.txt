int main() 
{
	int n,c=0,o=0,z=0;
	string s,ans="-1";
	cin>>n;
	while(n--)
	{
		cin>>s;
		z=count(s.begin(),s.end(),'0');
		o=count(s.begin(),s.end(),'1');
		if(z==s.length())
		{
			cout<<"0";
			return 0;
		}
		else if(o+z==s.length()&&o==1)
		c+=z;
		else
		ans=s;
		//cout<<z<<"\t"<<o<<endl;
	}
	if(ans=="-1")
	ans="1";
	cout<<ans;
	while(c--)
	cout<<"0";
	// your code goes here
	return 0;
}
