#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define llu unsigned long long

int main() {
	int n;
	scanf("%d", &n);

	string s, pref, cur, suf;

	while (n--) {
		cin >> s;

		if (s == "0") {
			cout << "0" << endl;
			return 0;
		}

		cur = "";
		bool flag = true;
		for (int i = s.length()-1; i >= 0; i--) {
			if (s[i] == '0' && flag)
				suf += '0';
			else {
				flag = false;
				cur += s[i];
			}
		}

		reverse(cur.begin(), cur.end());
		if (cur > pref) pref = cur;
	}

	cout << pref << suf << endl;
}
