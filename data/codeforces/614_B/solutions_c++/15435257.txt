#include <bits/stdc++.h>
using namespace std;
bool beaufitul(string s){
	bool one = false;
	for(int i=0; i<s.size(); i++){
		if(s[i] != '0' && s[i] != '1') return false;
		if(s[i] == '1'){
			if(one) return false;
			one = true;
		}
	}
	return true;
}
int main()
{
	int n,ceros=0;
	string s,y="1";
	cin>>n;
	while(n--)
	{
		cin>>s;
		if(s=="0"){cout<<0;return 0;}
		if(beaufitul(s))ceros+=s.size()-1;
		else y=s;
	}
	cout<<y;
	for(int i=0;i<ceros;i++)
		cout<<0;
	return 0;
}
