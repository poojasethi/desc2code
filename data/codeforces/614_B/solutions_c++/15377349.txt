#include<iostream>
#include<cstdlib>
#include<cmath>
using namespace std;

int main(){
	long long a, n = 0;
	string b, c = "1";
	cin >> a;
	while(a--){
		cin >> b;
		if(atoll(b.c_str()) == 0){
			cout << 0 << endl;
			return 0;
		}
		bool bo = false;
		for(int i = 0; i < b.length(); i++){
			if(b[i] != '0'){
				if(bo || b[i] != '1'){
					c = b;
					bo = false;
					break;
				} else {
					bo = true;
				}
			}
		}
		if(bo){
			n += b.length() - 1;
		}			
	}
	cout << c;
	for(long long i = 0; i < n; i++){
		cout << 0;
	}
	cout << endl;
	return 0;
}
