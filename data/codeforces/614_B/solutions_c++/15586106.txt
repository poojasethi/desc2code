#include <iostream>
#include <string>
using namespace std;

string a, b = "1";
int main()
{
	int n, zer = 0;
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> a;
		if (a == "0") { b = a; break; }
		while (a[a.size() - 1] == '0')
			a.erase(a.end() - 1), zer++;
		if (b == "1") b = a;
	}
	cout << b;
	if (b != "0")
		for (int i = 0; i < zer; i++)
			cout << '0';
	return 0;
}
