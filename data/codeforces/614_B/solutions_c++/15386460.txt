#include <iostream>

using namespace std;


int main()
{
	long long n,c=0;
	long long i,j;
	long long check;
	cin>>n;
	string x[n],k="1";
	
	for (i=0;i<n;i++)
	{
		cin >> x[i];
	}
	
	for (i=0;i<n;i++)
	{
		check=0;
		for (j=0;j<x[i].size();j++)
		{
			check = check + x[i][j]-'0';
		}
		if (check ==0)
		{
			cout<<"0";
			return 0;
		}
		else if (check ==1)
		{
			c=c+x[i].size()-1;
		}
		else
		{
			k=x[i];
		}
	}
	cout<<k;
	for (i=0;i<c;i++)
	{
		cout<<"0";
	}
	return 0;
}
