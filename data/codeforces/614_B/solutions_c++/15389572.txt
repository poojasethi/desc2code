#include<iostream>
#include<cstdio>
using namespace std;
string st[500005];
int main()
{   int len,n,bj=0,tot=0,i,j;
    cin>>n;
    for(i=1;i<=n;i++)
      {cin>>st[i];
       len=st[i].length();
       if(st[i][0]=='0'){cout<<0;return 0;}
       if(st[i][0]=='1')
         {for(j=1;j<len;j++)if(st[i][j]!='0')bj=i;}
       else bj=i;
       if(bj!=i)tot+=(len-1);
      }
    if(bj)cout<<st[bj];else cout<<1;while(tot--)cout<<0;
    return 0;
}
