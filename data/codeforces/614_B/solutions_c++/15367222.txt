#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n, z = 0;
    string s, s1="1";

    cin >> n;
    while(n--) {
        cin >> s;
        int a = 0;
        for( int i = 0; i < s.size(); i++ ) a += s[i] - '0';
        if( a == 0 ) {
            cout << 0;
            return 0;
        }
        else if( a == 1 ) z += s.size() - 1;
        else s1 = s;
    }
    cout << s1;
    for( int i = 0; i < z; i++ ) cout << 0;
}
