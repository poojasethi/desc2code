#include<cstdio>
#include<cstring>

int n, tot;
bool zero;
char s[111111];
char t[111111];

int main() {
	scanf("%d", &n);
	zero = false;
	t[0] = '1'; t[1] = 0;
	tot=0;
	for(int i=0;i<n;i++) {
		scanf("%s", s);
		if(s[0]=='0') zero = true;
		else {
			bool ok = (s[0] == '1');
			for(int i=1;s[i] && ok;i++)
				if(s[i]!='0') ok = false;
			if(!ok) memcpy(t, s, sizeof s);
			else tot+=strlen(s)-1;
		}
	}
	if (zero) printf("0\n");
	else {
		printf("%s", t);
		for(int i=0;i<tot;i++) printf("0");
		printf("\n");
	}
	
	return 0;
}
