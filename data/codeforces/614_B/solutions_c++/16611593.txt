#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	string head("1");
	int zeros=0;
	while(n--)
	{
		string s;
		cin>>s;
		int j=s.length()-1;
		while(s[j]=='0'&&j>0)
			j--;
		if(j==0&&s[0]=='1')
			zeros+=s.length()-1;

		else if(j==0&&s[0]=='0')
		{
			cout<<0<<endl;
			return 0;
		}
		else head=s;
	}
		cout<<head;
		while(zeros--)
			putchar('0');
		return 0;
}
