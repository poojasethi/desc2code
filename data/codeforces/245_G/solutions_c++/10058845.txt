#include<cstdio>
#include<iostream>
#include<vector>
#include<map>
#include<algorithm>
#include<cstring>
using namespace std;
string s,t;
map<string,int> mm;
string na[10005];
vector<int> ve[10005];
int ans[10005],temp[10005],tot;
bool mp[10005][10005];
main()
{
    int m;scanf("%d",&m);
    for(int i=1;i<=m;i++)
    {
        cin>>s>>t;
        if(mm[s]==0) mm[s]=++tot,na[tot]=s;
        if(mm[t]==0) mm[t]=++tot,na[tot]=t;
        int d1=mm[s],d2=mm[t];
        ve[d1].push_back(d2);
        ve[d2].push_back(d1);mp[d1][d2]=mp[d2][d1]=1;
    }
    printf("%d\n",tot);
    for(int i=1;i<=tot;i++)
    {
        int al=0,ma=0;
        for(int j=1;j<=tot;j++) if(i!=j&&!mp[i][j])
        {
            int to=0;
            for(int k=0;k<ve[i].size();k++)
            {
                int u=ve[i][k];
                if(mp[u][j]&&j!=u)
                to++;
            }
            temp[al++]=to;ma=max(ma,to);
        }
        int ans=0;
        for(int j=0;j<al;j++) if(temp[j]==ma) ans++;
        cout<<na[i]<<" "<<ans<<endl;

    }

}
