#include <iostream>
#include<map>
#include<vector>
#include<string>
using namespace std;
int n=0;
vector<string> names;
map<string,int> num;
vector<vector<int> > a;
int getnum(string&s) {
    if (num.find(s)==num.end()) {
        num[s]=n++;
        names.push_back(s);
        a.push_back(vector<int>());
    }
    return num[s];
}
int main()
{
    int m;
    cin >> m;
    for (int i=0;i<m;++i) {
            string s1,s2;
            cin >> s1 >> s2;
            int t1=getnum(s1),t2=getnum(s2);
            a[t1].push_back(t2);
            a[t2].push_back(t1);
    }
    cout << n << endl;
    for (int v=0;v<n;++v) {
        vector<int> frc(n);
        for (int i=0;i<a[v].size();++i) {
            for (int j=0;j<a[a[v][i]].size();++j) {
                ++frc[a[a[v][i]][j]];
            }
        }
        for (int i=0;i<a[v].size();++i) {
            frc[a[v][i]]=-1;
        }
        frc[v]=-1;
        int mx=0,mxc=0;
        for (int i=0;i<n;++i) {
            if (frc[i]>mx) {
                mx=frc[i];
            }
        }
        for (int i=0;i<n;++i) {
            if (frc[i]==mx) {
                ++mxc;
            }
        }
        cout << names[v] << " " << mxc << endl;
    }
    return 0;
}
