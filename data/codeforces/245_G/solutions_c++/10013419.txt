#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<map>
#include<vector>
#include<queue>
using namespace std;

const int N=11000;
map<string,int> mp;
int idx;
vector<int> g[N];
int f[N],ans[N];

int main(){
    int m;
    scanf("%d",&m);
    for(int i=0;i<m;i++){
        string a,b;
        cin>>a>>b;
        int u=mp[a];
        if(0==u){
            mp[a]=++idx;
            u=idx;
        }
        int v=mp[b];
        if(0==v){
            mp[b]=++idx;
            v=idx;
        }
        g[u].push_back(v);
        g[v].push_back(u);
    }
    for(int u=1;u<=idx;u++){
        for(int i=0;i<g[u].size();i++)f[g[u][i]]=1;
        int mx=0;
        for(int v=1;v<=idx;v++){
            if(f[v]||v==u)continue;
            int tmp=0;
            for(int i=0;i<g[v].size();i++){
                tmp+=f[g[v][i]];
            }
            if(tmp>mx){
                mx=tmp;
                ans[u]=1;
            }
            else if(tmp==mx)ans[u]++;
        }
        for(int i=0;i<g[u].size();i++)f[g[u][i]]=0;
    }
    cout<<idx<<endl;
    map<string,int> ::iterator it=mp.begin();
    for(int i=1;i<=idx;i++){
        cout<<it->first<<" "<<ans[it->second]<<endl;
        it++;
    }
    return 0;
}
