int a, b, n;
char used[maxn][maxn];
ii ar[maxn], bs[maxn];
vector <iii> v, h;
vector <int> w[maxn], w2[maxn];

bool f( int x ) {
	
	x <<= 1;
	memset( used, 0, sizeof( used ) );
	v.clear();
	h.clear();
	for(int i=0;i<=a;i++) w[i].clear();
	for(int i=0;i<=b;i++) w2[i].clear();

	for(int i=1;i<=a;) {
		int j = i, end = ar[i].se + x;
		while( j <= a && ar[i].fi == ar[j].fi && ar[j].se - x <= end ) end = ar[j].se + x, j++;
		v.pb( iii( ar[i].fi, ii( ar[i].se-x, end ) ) );
		//~ printf("%d iii( %d, ii( %d, %d ) )\n",i-1,ar[i].fi,ar[i].se-x,end);
		i = j;
	}
	
	for(int i=1;i<=b;) {
		int j = i, end = bs[i].se + x;
		while( j <= b && bs[i].fi == bs[j].fi && bs[j].se - x <= end ) end = bs[j].se + x, j++;
		h.pb( iii( bs[i].fi, ii( bs[i].se-x, end ) ) );
		i = j;
	}
	
	for(int i=0;i<v.size();i++)
		for(int j=0;j<h.size();j++)
			if( v[i].se.fi <= h[j].fi && h[j].fi <= v[i].se.se && h[j].se.fi <= v[i].fi && v[i].fi <= h[j].se.se ) {
				w[i].pb( j );
				w2[j].pb( i );
			}
	
	for(int i=0;i<v.size();i++)
		for(int j=0;j<w[i].size();j++)
			for(int k=0;k<w2[w[i][j]].size();k++)
				if( w2[w[i][j]][k] != i ) {
					used[i][w2[w[i][j]][k]]++;
					used[w2[w[i][j]][k]][i]++;
					if( used[i][w2[w[i][j]][k]] == 4 || used[w2[w[i][j]][k]][i] == 4 ) return true;
				}
	
	return false;
}

int main() {
	
	scanf("%d %d",&a,&b);

	for(int i=1,j,k;i<=a;i++) {
		scanf("%d %d",&j,&k);
		ar[i].fi = j+k;
		ar[i].se = j-k;
	}
	
	for(int i=1,j,k;i<=b;i++) {
		scanf("%d %d",&j,&k);
		bs[i].se = j+k;
		bs[i].fi = j-k;
	}
	
	sort( ar+1, ar+1+a );
	sort( bs+1, bs+1+b );
	int t = 0;

	for(int k=25;k>=0;k--)
		if( !f( t+(1<<k) ) ) t += 1<<k;
	t++;
	if( t > 1e7 ) printf("Poor Sereja!\n");
	else printf("%d\n",t);
	
	return 0;
}
