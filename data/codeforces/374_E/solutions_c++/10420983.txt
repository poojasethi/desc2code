#include <bits/stdc++.h>

#define fi first
#define se second
#define o ((b+e)/2)

using namespace std;

typedef pair <int,int> ii;
typedef pair <int,ii> iii;
ii red[2100],blue[2100];
int N,M,cnt,K=15000,res=1e9,usd[2100][2100];
vector <iii> b,r,tt1;
vector <int> v,b1[2100],r1[2100];
map <int,int> used;

int cmp( const iii &a , const iii &b ){ a.se.fi < b.se.fi; }

vector <iii> fix( vector <iii> b ){
	tt1.clear();
	for( int i=0 ; i<(int)b.size() ; i++ ){
		int j=i;
		int l=-1e9,r=-1e9;
		while( j<(int)b.size() ){
			if( b[j].se.fi > r ){
				if( l != -1e9 ) tt1.push_back( iii( b[j].fi , ii( l , r ) ) );
				l=b[j].se.fi; r=b[j].se.se;
			}else	r=b[j].se.se;
			if( b[j+1].fi != b[j].fi || j == b.size()-1 ) break;
			j++;
		}
		if( l != -1e9 ) tt1.push_back( iii( b[j].fi , ii( l , r ) ) );
		i=j;
	}
	return tt1;
}

bool check( int t ){
	cnt=0;
	v.clear(); b.clear(); r.clear();
	memset( usd , 0 , sizeof usd );
	for( int i=1 ; i<=N ; i++ ){
		int x=blue[i].fi+blue[i].se,y=blue[i].fi-blue[i].se;
		b.push_back( iii ( x , ii( y-t*2  , y+t*2 )) );
		v.push_back( x ); 	v.push_back( y-t*2 ); v.push_back( y+t*2 );
	}
	for( int i=1 ; i<=M ; i++ ){
		int x=red[i].fi+red[i].se,y=red[i].fi-red[i].se;
		r.push_back( iii( y , ii( x-t*2  , x+t*2 ) ));
		v.push_back( x-t*2 ); v.push_back( x+t*2 );	v.push_back( y ); 
	}
	
	sort( v.begin() , v.end() );
	for( int i=0 , prev=-1e9 ; i<(int)v.size() ; i++ ) if( v[i] != prev ){ prev=v[i]; used[v[i]]=++cnt; }
	for( int i=0 ; i<N ; i++ ) b[i].fi=used[b[i].fi],b[i].se.fi=used[b[i].se.fi],b[i].se.se=used[b[i].se.se];
	for( int i=0 ; i<M ; i++ ) r[i].fi=used[r[i].fi],r[i].se.fi=used[r[i].se.fi],r[i].se.se=used[r[i].se.se];
	sort( b.begin() , b.end() );
	sort( r.begin() , r.end() );
	b=fix( b );
	r=fix( r );
	
	for( int i=0 ; i<(int)b.size() ; i++ ) b1[i].clear();
	for( int i=0 ; i<(int)r.size() ; i++ ) r1[i].clear();
	for( int i=0 ; i<(int)b.size() ; i++ )
		for( int j=0 ; j<(int)r.size() ; j++ ) 
			if( b[i].se.fi <= r[j].fi && r[j].fi <= b[i].se.se && r[j].se.fi <= b[i].fi && b[i].fi <= r[j].se.se ){
				b1[i].push_back( j );
				r1[j].push_back( i );
			}
	
	for( int i=0 ; i<(int)b.size() ; i++ )
		for( int j=0 ; j<(int)b1[i].size() ; j++ )
			for( int k=0 ; k<(int)r1[b1[i][j]].size() ; k++ ) if( i != r1[b1[i][j]][k] ){
				usd[i][r1[b1[i][j]][k]]++;
				usd[r1[b1[i][j]][k]][i]++;
				if( usd[i][r1[b1[i][j]][k]] == 4 || usd[r1[b1[i][j]][k]][i] == 4 ) return true;
			}
	return false;

}

int main(){
	
	cin >> N >> M;
	
	for( int i=1,a,b ; i<=N ; i++ ){
		scanf(" %d %d",&a,&b);
		blue[i]=ii( a , b );
	}
	for( int i=1,a,b ; i<=M ; i++ ){
		scanf(" %d %d",&a,&b);
		red[i]=ii( a , b );
	}
	
	int b=1,e=1e6;
	while( b <= e ){
		if( check( o ) ){ res=o; e=o-1; }
		else b=o+1;
	}
	
	if( res == 1e9 ) puts("Poor Sereja!");
	else cout << res << endl;
	
	return 0;
	
}
