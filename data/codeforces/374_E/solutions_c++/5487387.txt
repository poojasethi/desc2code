int rd() { return RAND_MAX == 32767 ? ((rand() << 15) ^ rand()) : rand(); }

const int maxn = 2010;

int n, m;
int ins[maxn][maxn];

struct Point
{
	int x, y;

	void read()
	{
		int X, Y;
		cin >> X >> Y;
		x = X-Y;
		y = X+Y;
	}
} a[maxn], b[maxn];
struct LineX
{
	int x0, x1, y;

	LineX() {}
	LineX(int x0_, int x1_, int y_) : x0(x0_), x1(x1_), y(y_) {}
};
struct LineY
{
	int y0, y1, x;

	LineY() {}
	LineY(int y0_, int y1_, int x_) : y0(y0_), y1(y1_), x(x_) {}
};
bool cmpX(Point u, Point v)
{
	return u.x < v.x || (u.x == v.x && u.y < v.y);
}
bool cmpY(Point u, Point v)
{
	return u.y < v.y || (u.y == v.y && u.x < v.x);
}
int check(int L)
{
	vector<LineX> lx;
	vector<LineY> ly;
	for (int i = 0, j; i < n; i = j) {
		for (j = i+1; j < n && a[i].y == a[j].y && a[j-1].x+L*4 >= a[j].x; j++);
		lx.pb(LineX(a[i].x-L*2, a[j-1].x+L*2, a[i].y));
	}
	for (int i = 0, j; i < m; i = j) {
		for (j = i+1; j < m && b[i].x == b[j].x && b[j-1].y+L*4 >= b[j].y; j++);
		ly.pb(LineY(b[i].y-L*2, b[j-1].y+L*2, b[i].x));
	}
	for (int i = 0; i < ly.size(); i++)
		for (int j = i+1; j < ly.size(); j++)
			ins[i][j] = 0;
	for (int i = 0; i < lx.size(); i++) {
		vector<int> cnt;
		for (int j = 0; j < ly.size(); j++)
			if (ly[j].x >= lx[i].x0 && ly[j].x <= lx[i].x1 && lx[i].y >= ly[j].y0 && lx[i].y <= ly[j].y1)
				cnt.pb(j);
		for (int j = 0; j < cnt.size(); j++)
			for (int k = j+1; k < cnt.size(); k++) {
				int u = cnt[j], v = cnt[k];
				if (u > v) swap(u, v);
				if (ly[u].x != ly[v].x) {
					if (ins[u][v]) return 1;
					ins[u][v] = 1;
				}
			}
	}
	return 0;
}
int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	cin >> n >> m;
	for (int i = 0; i < n; i++) a[i].read();
	for (int i = 0; i < m; i++) b[i].read();
	sort(a, a+n, cmpY);
	sort(b, b+m, cmpX);
	int lt, rt, mid;
	for (lt = 1, rt = 2100000; lt < rt; ) {
		mid = (lt+rt) >> 1;
		check(mid) ? rt = mid : lt = mid+1;
	}
	lt < 2100000 ? cout << lt << endl : cout << "Poor Sereja!" << endl;
	return 0;
}
