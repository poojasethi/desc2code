#include <iostream>
#include <cstdio>
#include <string>
#include <algorithm>
#include <cstring>
#include <vector>
#define N 4004
#define INF (2e6+9)
using namespace std;
int n,m;
int x[N],y[N],f[2*N];
struct edge
{
    int s;
    int e;
}e[N*N];
struct seg
{
    int w;
    int x1,x2;
    bool operator < (const struct seg &b)const{
        if(w!=b.w)return w<b.w;
        else return x1<b.x1;
    }
}cur;
bool xj(struct seg a,struct seg b)
{
    if(a.x1<=b.w&&a.x2>=b.w&&b.x1<=a.w&&b.x2>=a.w)
        return true;
    else return false;
}
int refa(int k)
{
    if(f[k]==k)return k;
    else return f[k]=refa(f[k]);
}
void du(vector<seg> &v)
{
    vector<seg> w;
    sort(v.begin(),v.end());
    for(int i=0;i<(int)v.size();i++)
    {
        cur.w=v[i].w;
        cur.x1=v[i].x1;
        cur.x2=v[i].x2;
        while(i+1<(int)v.size()&&v[i].w==v[i+1].w&&v[i].x2>=v[i+1].x1)
        {
            cur.x2=v[i+1].x2;
            i++;
        }
        w.push_back(cur);
    }
    v=w;
    return;
}
bool check(int t)
{
    vector<seg> blue;
    for(int i=0;i<n;i++)
    {
        cur.w=x[i];
        cur.x1=y[i]-2*t;
        cur.x2=y[i]+2*t;
        blue.push_back(cur);
    }
    du(blue);
    vector<seg> red;
    for(int i=n;i<n+m;i++)
    {
        cur.w=y[i];
        cur.x1=x[i]-2*t;
        cur.x2=x[i]+2*t;
        red.push_back(cur);
    }
    du(red);
    int cnt=0;
    for(int i=0;i<2*N;i++)
        f[i]=i;
    for(vector<seg>::iterator i=blue.begin();i!=blue.end();i++)
    {
        for(vector<seg>::iterator j=red.begin();j!=red.end();j++)
        {
            if(xj(*i,*j)){
                e[cnt].s=i-blue.begin();
                e[cnt++].e=(j-red.begin())+n;
            }
        }
    }
    int fa,fb;
    for(int i=0;i<cnt;i++)
    {
        fa=refa(e[i].s);
        fb=refa(e[i].e);
        if(fa==fb) return true;
        else f[fa]=fb;
    }
    return false;
}
int main()
{
//    freopen("out.txt","r",stdin);
    scanf("%d%d",&n,&m);
    for(int i=0,a,b;i<n+m;i++)
    {
        scanf("%d%d",&a,&b);
        x[i]=a+b;
        y[i]=b-a;
    }
    int l=1,r=INF,m;
    while(l<r)
    {
        m=(l+r)>>1;
        if(check(m))r=m;
        else l=m+1;
    }
    if(r==INF)cout<<"Poor Sereja!"<<endl;
    else cout<<r<<endl;
    return 0;
}
