#include<cstdio>
#include<set> 
#include<algorithm>
#define MAXN 100010

using namespace std;
typedef long long int ll;

struct node
{
	node* to[2];
	node()
	{
		to[0] = to[1] = NULL;
	}
};

int n;
ll in[MAXN];

int main()
{
	ll left = 0, right = 0, ac = 0;
	scanf("%d", &n);
	for(int i = 0; i < n; i++)
		scanf("%I64d", in + i), ac ^= in[i];
	left = ac;
	node* root = new node();
	for(int i = n - 1; i >= -1; i--)
	{
		node* p = root;
		for(int j = 40; j >= 0; j--)
		{
			int t = right >> j & 1;
			if(!p->to[t])
				p->to[t] = new node();
			p = p->to[t];
		}
		ll sum = 0;
		p = root;
		for(int j = 40; j >= 0; j--)
		{
			int t = left >> j & 1;
			if(p->to[t ^ 1])
			{
				t ^= 1;
				sum |= 1ll << j;
			}
			p = p->to[t];
		}
		ac = max(ac, sum);
		if(i >= 0)
			left ^= in[i], right ^= in[i];
	}
	printf("%I64d\n", ac);
}
