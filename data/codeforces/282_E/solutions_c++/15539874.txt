#include<iostream>
typedef long long int ll;
using namespace std;
ll a,b,tot,y[100000],u,v,w[45],i;

struct node
{
	ll n,son[2];
}z[10000000];

void insert(ll o)
{
	int i=44;
	ll j=0;
	while(i>-1)
	{
		ll x=o&w[i];
		x=(x!=0?1:0);
		if(z[j].son[x]==0)
		    z[j].son[x]=++tot;
		j=z[j].son[x];
		i--;
	}
	z[j].n=o;
}

ll find(ll p)
{
	int i=44;
	ll j=0;
	while(i>-1)
	{
		ll x=p&w[i];
		x=(x!=0?0:1);
		if(z[j].son[x]!=0)
		    j=z[j].son[x];
		else
		    j=z[j].son[1-x];
		i--;
	}
	return z[j].n;
}

int main()
{
    ios::sync_with_stdio(false);
	w[0]=1;
	i=1;
	while(i<=44)
	{
		w[i]=w[i-1]<<1;
		i++;
	}
	cin>>a;
	i=0;
	while(i<a)
	{
		cin>>y[i];
		v^=y[i++];
	}
	b=v;
	insert(0);
	i=0;
	while(i<a)
	{
		u^=y[i];
		v^=y[i];
		insert(u);
		b=max(b,find(v)^v);
		i++;
	}
	cout<<b;
	return 0;
}
