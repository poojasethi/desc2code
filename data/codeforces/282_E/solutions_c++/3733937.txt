#include <iostream>
using namespace std;
long long x[8003006];
int main() {
	int n, k = 100;
	long long a[100100], s = 0, t = 0, ans;
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> a[i];
		s ^= a[i];
	}
	for (int i = 0; i < 41; i++) {
		x[i*2] = i*2 + 2;
	}
	ans = s;
	for (int i = 0; i < n; i++) {
		int c = 0;
		t ^= a[i];
		for (int j = 40; j >= 0; j--) {
			if (t >> j & 1) c++;
			if (!x[c]) {
				x[c] = k;
				k += 2;
			}
			c = x[c];
		}
		x[c] = t;
		s ^= a[i];
		c = 0;
		for (int j = 40; j >= 0; j--) {
			if (!x[c+1] || (x[c] && s >> j & 1)) {
				c = x[c];
			} else {
				c = x[c+1];
			}
		}
		if (ans < (x[c] ^ s)) ans = x[c] ^ s;
	}
	cout << ans << endl;
	return 0;
}
