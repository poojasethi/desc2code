#include <iostream>
#include<cstdio>
using namespace std;

typedef long long LL;

const int N=100010;
int son[N*40][2];
LL a[N],tot;

void add(LL x)
{
    int i;
    int p=0;
    for(i=40;i>=0;--i)
    {
        int num=(x>>i)&1;
        if(!son[p][num])son[p][num]=++tot;
        p=son[p][num];
    }
}

LL get(LL x)
{
    int i;
    int p=0;
    LL tmp=0ll;
    for(i=40;i>=0;--i)
    {
        int num=(x>>i)&1;
        if(son[p][!num])
        {
            p=son[p][!num];
            tmp+=(LL)1<<i;
        }
        else p=son[p][num];
    }
    return tmp;
}

int main()
{
    int i,n;
    scanf("%d",&n);
    LL s=0ll,t=0ll,ans=0ll;;
    for(i=1;i<=n;++i)
    {
        scanf("%I64d",&a[i]);
        t^=a[i];
        add(t);
    }
    ans=get(0);
    for(i=n;i>=1;--i)
    {
        s=s^a[i];
        ans=max(ans,get(s));
    }
    printf("%I64d",ans);
    return 0;
}
