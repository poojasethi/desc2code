#include <cstdio>
#include <algorithm>
using namespace std;
long long f[7000005][2],n,i;
long long a[100005],h,g,ans,tot;
inline void add(long long x){
	long long k=0;
	for(long long i=50;i>=0;i--){
		long long s=(1ll<<i)&x;
		if (s)s=1ll;
		if (f[k][s])k=f[k][s];
		else k=f[k][s]=++tot;
	}
}
inline void sum(long long x){
	long long k=0;long long S=0;
	for(long long i=50;i>=0;i--){
		long long s=(1ll<<i)&x;
		if (s)s=1ll;
		if (f[k][s^1]){k=f[k][s^1];S+=(1ll<<i);}
		else k=f[k][s];
	}
	ans=max(ans,S);
}
int main(){
	scanf("%I64d",&n);
	for(i=1;i<=n;i++)scanf("%I64d",&a[i]);
	for(i=1;i<=n;i++)h=h^a[i];
	ans=h;add(0);
	for(i=1;i<=n;i++){
		h=h^a[i];
		g=g^a[i];
		add(g);
		sum(h);
	}
	printf("%I64d",ans);
	return 0;
}
