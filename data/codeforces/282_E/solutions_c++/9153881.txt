#include <cstdio>
#include <algorithm>

using namespace std;

typedef long long ll;

const int N = 100010, D = 45;
int n, nc, g[N * D][2];
ll a[N];
ll x, y, res;

int main()
{
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) scanf("%I64d", &a[i]), x ^= a[i];
    for (int i = n; i >= 0; x ^= a[i], y ^= a[i], i--)
    {
        int c = 0;
        for (int j = D; j >= 0; j--)
        {
            ll d = 1ll & (y >> (ll) j);
            if (!g[c][d]) g[c][d] = ++nc;
            c = g[c][d];
        }
        ll r = 0ll; c = 0;
        for (int j = D; j >= 0; j--)
        {
            ll d = 1ll & (x >> (ll) j);
            d = 1ll - d;
            if (g[c][d]) r |= 1ll << (ll) j, c = g[c][d]; else
            c = g[c][1ll - d];
        }
        res = max(res, r);
    }
    printf("%I64d\n", res);
    return 0;
}
