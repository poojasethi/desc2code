#include <cstdio>
#include <iostream>
using namespace std;
#define ll long long

int trie[8000000][2], cnt;
ll ans, tot, z;
ll pre[100010], a[100010];

void insert(ll x){
    int p = 0;
    for(int i=42; i>=0; --i){
        if(! trie[p][x>>i&1])
            trie[p][x>>i&1] = ++cnt;
        p = trie[p][x>>i&1];
    }
}

ll query(ll x){
    int p = 0;
    for(int i=42; i>=0; --i){
        if(trie[p][(x>>i&1)^(tot>>i&1)^1])
            z += 1ll<<i,p=trie[p][(x>>i&1)^(tot>>i&1)^1];
        else
            p = trie[p][(x>>i&1)^(tot>>i&1)];
    }
}

int main(){
    int n; cin>>n;
    insert(0);
    for(int i=1; i<=n; ++i)
        scanf("%I64d", &a[i]), pre[i] = pre[i-1]^a[i];
    for(int i=1; i<=n; ++i)
        insert(pre[i]);
    ans = tot = pre[n];
    for(int i=0; i<=n; ++i)
        {z = 0; query(pre[i]); ans = max(ans, z);}
    cout<<ans<<endl;

    return 0;
}
