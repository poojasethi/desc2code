#include <cstdio>
#include <algorithm>
typedef unsigned long long ULL;
const int ml=41,ma=111111;

int n,o[ma][ml];
ULL a[ma],f[ma],p[ml];

ULL cc(ULL x)
{
	int l=0,r=n+1;
	for (int i=ml;i>=0;i--)
		if (x&p[i])
		{
			if (l<o[l][i]) r=std::min(r,o[l][i]);
		}
		else
			if (r>o[l][i]) l=o[l][i];
	return x^f[l];
}

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
		scanf("%I64d",&a[i]);
	f[0]=0;
	for (int i=1;i<=n;i++)
		 f[i]=f[i-1]^a[n-i+1];
	std::sort(f,f+n+1);
	for (int i=0;i<=ml;i++)
	{
		int l=n+1;
		p[i]=1LL<<i;
		for (int j=n;j>=0;j--)
		{
			if (f[j]&p[i])l=j;
			o[j][i]=l;
		}
	}	
	ULL t=0,ans=cc(0);

	for (int i=1;i<=n;i++)
		ans=std::max(ans,cc(t)),t^=a[i];
	printf("%I64d\n",ans);
	return 0;
}
