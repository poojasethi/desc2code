#include <iostream>
#include <algorithm>
#include <cstdio>
#define MAXN 100005

using namespace std;

typedef long long ll;

class trie
{
	public:
	int next[3];

	trie() { next[0] = next[1] = -1; }
};

trie T[50*MAXN];
ll N,S[MAXN];
int ind = 1;
ll bas[MAXN],son[MAXN];
ll ans;

void ekle(ll x)
{
	ll i,k,t=1;

	for(i=40;i>=0;i--)
	{
		k = !!(x & (1LL << i));
		if(T[t].next[k] == -1) 
			T[t].next[k] = ++ind;
		t = T[t].next[k];
	}
}

ll bul(ll x)
{
	ll i,k,t=1,ret=0;

	for(i=40;i>=0;i--)
	{
		k = !(x & (1LL << i));
		if(T[t].next[k] != -1)
		{
			ret += (1LL << i);
			t = T[t].next[k];
		}
		else
			t = T[t].next[!k];

	}
	return ret;
}

int main()
{
	ll i,j;
	
	cin >> N;
	for(i=2;i<=N+1;i++)
		scanf("%I64d",&S[i]);
	N+=2;

	for(i=1;i<=N;i++) 
		bas[i] = bas[i-1] ^ S[i];
	for(i=N;i>=1;i--)
		son[i] = son[i+1] ^ S[i];

	for(i=N-1;i>=1;i--)
	{
		ekle( son[i+1] );
		ans = max( ans , bul(bas[i]) );
	}

	cout << ans << endl;

	return 0;
}
