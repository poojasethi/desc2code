#include<bits/stdc++.h>

using namespace std;

long long Ar[100005],n,r,q,z,i,j,s,sz,pos,mx,pre,x,t;

int A[10000007][2];

main(){

	scanf("%lld",&n);

	for(i=1;i<=n;i++){
		scanf("%lld",&Ar[i]);
		t ^= Ar[i];
	}

	for(j=40; j>=0 ; j--)
		pos = A[pos][0] = ++sz;

	for(i=1;i<=n;i++){

		pre ^= Ar[i];

		s = pos = 0;

		for(j=40; j>=0 ; j--){

			q = ((t & (1ll << j)) > 0);
			r = ((pre & (1ll << j)) > 0);

			z = !((q+r) & 1);

			if(A[pos][z]){
				s += (1ll << j);
				pos = A[pos][z];
			}

			else
				pos = A[pos][!z];
		}

		mx = max(mx , max(pre,s));

		pos = 0;

		for(j=40; j>=0 ; j--){

			q = ((pre & (1ll << j)) > 0);

			if(A[pos][q])
				pos = A[pos][q];

			else
				pos = A[pos][q] = ++sz;
		}
	}

	printf("%lld\n",mx);

	return 0;
}
