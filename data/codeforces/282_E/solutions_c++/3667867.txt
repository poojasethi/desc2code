#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <string>
#include <map>
#include <set>
#include <algorithm>

using namespace std;
#define forn(i,a,b) for(i=(a);i<=(b);++i)
#define ford(i,a,b) for(i=(b);i>=(a);--i)
#define ll long long
ll ans;
int i,j,n;
int sz=0;
int go[5000000][3];
ll x,y,a[1000000];
void add(ll ee){
 int i,rt=0,r;
 ford(i,0,60){
    r=((ee&(1LL<<i))>0);
    if (go[rt][r]==0) go[rt][r]=++sz;
    rt=go[rt][r]; 
 }
}
void find(ll ee){
  int i,r,rt=0;
  ll res=0;
  ford(i,0,60){
    r=1-((ee&(1LL<<i))>0);
    if (go[rt][r]==0) rt=go[rt][1-r];
    else {
     res+=(1LL<<i);
     rt=go[rt][r];
    }
  }
 ans=max(ans,res); 
}
int main(){
  cin>>n;
  forn(i,1,n)cin>>a[i];
  forn(i,1,n) {
    x^=a[i];
  }
  ans=x;
  y=0;
  add(0);
  ford(i,1,n){
     y=y^a[i];
     x=x^a[i];
     add(y);
     find(x);
  }
 find(0); 
 cout<<ans<<endl; 
}
