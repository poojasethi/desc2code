#include <stdio.h>
#include <string.h>
char s[1048576];
int i,j,l,p;
int main()
{
    p=0;
    while(gets(s))
    {
        l=strlen(s);
        for(i=0;i<l&&s[i]==' ';++i);
        if(s[i]=='#')
        {
            if(p) puts("");
            puts(s);
            p=0;
        }
        else
        {
            for(;i<l;++i)
                if(s[i]!=' ')
                    putchar(s[i]);
            p=1;
        }
    }
    if(p) puts("");
    return 0;
}
