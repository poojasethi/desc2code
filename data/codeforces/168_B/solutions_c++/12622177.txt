#include <stdio.h>
#include<string.h>
int n,m;	int t=0;
int len;
char s[1148580];
int is_amp(){
	for(int i=0;i<len;i++){
		if(s[i]!=' '){
			if(s[i]=='#') return 1;
			else return 0;
		}
	}return 0;
}
int main(void){
	int k=0;int tt=1;
    while(gets(s)){
        len=strlen(s);
    	int ok=is_amp();
    	if((k||ok)){
    		if(tt!=1) printf("\n");
    	}
		if(ok==0) for(int i=0;i<len;i++){
			if(s[i]!=' ') printf("%c",s[i]);
		}
    	else printf("%s",s);
    	k=ok;
		tt++;
    }
	printf("\n");
	return 0;
}
