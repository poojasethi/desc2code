#include<iostream>
#include<vector>
#include<string>
#define n_max 1048580
using namespace std;

string s;


int main()
{
	
	unsigned int i, spell = 1;
	
	while(getline(cin, s)){
		for(i=0; i<s.size() && s[i] == ' '; ++i);
		
		if(s[i] == '#'){
			if(!spell)
				cout <<'\n';
			
			cout << s << '\n';
			
			spell = 1;
		}
		
		else{
			for(i=0 ;i<s.size(); ++i)
				if(s[i] != ' ')
					cout << s[i];
			spell = 0;
		}
	}
	
	if (!spell)
		cout << '\n';
	
	return 0;
}
