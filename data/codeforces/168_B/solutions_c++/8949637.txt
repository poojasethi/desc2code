#include <bits/stdc++.h>
using namespace std;


int main()
{
	string a;

	int f = 0;
	while(getline(cin, a)) {
		int i = 0;
		int l = a.length();
		while(i < l && a[i] == ' ') i++;
		if(i < l && a[i] == '#') {
			if(f) printf("\n");//cout << endl;
			cout << a << endl;
			f = 0;
			continue;
		}

//		for(i = 0; i < a.length(); i++) if(a[i] == ' ') a.erase(a.begin()+i);
		for(; i < l; i++) if(a[i] != ' ') printf("%c", a[i]);


		f = 1;
	}
	if(f) printf("\n");//cout << endl;

	return 0;
}
