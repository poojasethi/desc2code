#include <bits/stdc++.h>
using namespace std;
vector <string> v;
string s;
int main() {
	while(getline(cin, s)) {
		if(s[s.find_first_not_of(" ")] == '#') {
			v.push_back(s);
			for(int i=0; i<v.size(); i++) cout << v[i] << endl;
			v.clear();
		}	
		else {
			stringstream ss(s);
			if(v.empty()) v.push_back("");
			while(ss >> s) *(v.rbegin()) += s;
		}
	}
	for(int i=0; i<v.size(); i++) cout << v[i] << endl;	
}
