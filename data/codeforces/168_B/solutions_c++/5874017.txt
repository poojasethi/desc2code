#include <cstdio>
#include <cstring>
#include <ctype.h>

char buff[1024*1024+1],*ch;

int main()
{
#ifdef local
 freopen("a.in", "r", stdin);
#endif
 bool prevReg = false;
 while(gets(buff))
 {
  buff[strlen(buff)] = '\0';
  for(ch=buff;isspace(*ch);ch++);
  if(*ch=='#')
  {
   if(prevReg) printf("\n");
   prevReg = false;
   printf("%s\n",buff);
  }
  else
  {
   for(ch=buff;*ch;ch++)
    if(!isspace(*ch)) putchar(*ch);
   prevReg = true;
  }
 }
 if(prevReg)printf("\n");
 return 0;
}
