#include<cstdio>
#include<cstring>

using 	namespace std; 

char	s[1<<20];
bool	pan()
{
	for (int i = 0; s[i]; i++)
	{
		if ( s[i] == ' ' ) continue; 
		if ( s[i] != '#' ) return false; 
		else return true; 
	}
	return false; 
}
int		main()
{
	bool flag = false; 
	while ( gets(s) )
	{
		if ( pan() ) 
		{
			if ( flag ) printf("\n");
			flag = false; 
			printf("%s\n" , s );
		}
		else
		{
			for (int i = 0 ; s[i]; i++) 
				if ( s[i] != ' ' ) printf("%c" , s[i] );
			flag = true;
		}
	}
	if ( flag ) printf("\n");
	return 0;
}
