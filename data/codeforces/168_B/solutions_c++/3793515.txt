#include<iostream>
using namespace std;
int main()
{
    int i,lm,k=0;
    string s;
    while(getline(cin,s))
    {
        lm=s.length();
        for(i=0; i<lm; i++)
            if(!isspace(s[i]))break;

        if(i==lm || s[i]!='#')
        {
            if(k==2) cout<<endl;
            for(i=0; i<lm ; i++) if(s[i]!=' ')
                cout<<s[i];
            k=1;
        }

        else
        {
            if(k) cout<<endl;
            cout<<s;
            k=2;
        }
    }
    cout<<endl;
    return 0;
}
