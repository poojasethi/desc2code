#include<iostream>
using namespace std;
int main()
{
    int i,Hech=0;
    string s;
    while(getline(cin,s))
    {
        for(i=0;i<s.length();i++) if(s[i]!=' ') break;
            if(i==s.length() || s[i]!='#')
               {
                if(Hech==2)cout<<endl;
                for(i=0;i<s.length();i++) 
                    if(s[i]!=' ') cout<<s[i];
                Hech=1;
            }
            else
            {
                if(Hech)cout<<endl;
                cout<<s;
                Hech=2;
            }
    }
    cout<<endl;
    return 0;
}
