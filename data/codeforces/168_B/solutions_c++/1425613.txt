#include<iostream>
#include<string>

using namespace std;

bool is_ampli(const string & s){
  int curr=0;
  while(curr<s.size() && s[curr]==' ') curr++;
  return s[curr]=='#';
}

void prints(const string & s){
  for(int i=0;i<s.size();++i)
    if(s[i]!=' ') cout << s[i];
}

int main(){
  string line;
  bool onsaute=false;
  while(getline(cin, line)){
    if(is_ampli(line)){
      if(onsaute)
	cout << endl;
      cout << line << endl;
      onsaute=false;
    }
    else{
      prints(line);
      onsaute=true;
    }
  }
  if(onsaute)
    cout << endl;
}
