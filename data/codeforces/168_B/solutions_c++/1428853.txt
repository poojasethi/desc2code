#include <cstdio>
#include <cstring>
#define N 2000000

char s[N];

int main(void)
{
  //freopen("test.in","r",stdin);
  //freopen("test.out","w",stdout);
  int n,i;
  bool bla=0;
  while(gets(s))
  {
    n=strlen(s);
    for (i=0;i<n;i++)
      if (s[i]!=' ')
        break;
    if (s[i]=='#')
    {
      if (bla)puts("");
      puts(s);
      bla=0;
    }
    else
    {
      bla=1;
      for (i=0;i<n;i++)
        if (s[i]!=' '&&s[i]!='\n')
          putchar(s[i]);
    }
  }
  if (bla)puts("");
}
