#include<iostream>
#include<cstdio>
using namespace std;

int i,k=0;
string s;

int main()
{
	while(getline(cin,s))
	{
		for(i=0;i<s.size();++i)
			if (s[i]!=' ') break;
		if (i==s.size() || s[i]!='#')
		{
			if (k==2) cout << endl;
			for(i=0;i<s.size();++i)
				if (s[i]!=' ') cout << s[i];
			k=1;
		}
		else
		{
			if (k) cout << endl;
			cout << s;
			k=2;
		}
	}
	cout << endl;
	return 0;
}
