#include <iostream>
#include <cstring>
using namespace std;
#include <stdio.h>

#define maxn 1048579
char s[maxn];

int main(){
    bool  yes=false, j=false;
    int i;
    while(gets(s)){
        int len = strlen(s);
        for(i = 0; i < len; i ++)
            if(s[i] != ' ') break;
        if(s[i] == '#') yes = true;
        else yes = false;
        if(yes) { if(j) puts(""); puts(s); j=false; }
        else {
           for(i = i; i<len;i++)
             if(s[i] != ' ' && s[i] != '\n')  putchar(s[i]);
           j = true;
        }
    }
    if(j) puts("");
}
