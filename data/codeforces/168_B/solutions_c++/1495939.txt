#include <cstdio>
#include <cstring>
char s[1048577];

bool check() {
    int len = strlen(s), i = 0;
    while (s[i] == ' ' && i < len) ++i;
    return s[i] == '#';
}

int main() {
    bool br = 1;
    while (gets(s)) {
        int l = strlen(s);
        if (check()) {
            if (!br) puts("");
            for (int i = 0; i < l; ++i) putchar(s[i]);
            puts("");
            br = 1;
        } else {
            br = 0;
            for (int i = 0; i < l; ++i)
                if (s[i] != ' ') putchar(s[i]);
        }
    }
    if (!br) puts("");
}
