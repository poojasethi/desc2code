#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
using namespace std;
string s;

int main(){
  int i;
  bool flag;
  while (getline(cin,s)){
    for (i=0;i<s.length();i++)
      if (s[i]!=' ') break;
    if (s[i]=='#'){
      if (flag) printf("\n");
      flag=0;
	  cout<<s<<endl;
    }
    else {
      for (i=0;i<s.length();i++)
        if (s[i]!=' ') printf("%c",s[i]);
      flag=1;
    }
  }
  if (flag) printf("\n");
  return 0;
}
