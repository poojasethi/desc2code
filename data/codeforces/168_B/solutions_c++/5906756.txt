#include <iostream>
#include <string>

using namespace std;

int main(){

    string linea;
    int i;
    bool concat = false;
    
    while(getline(cin,linea)){
    
        for(i = 0;isspace(linea[i]);i++);
        
        if(linea[i] == '#'){
        
            if(concat) cout << endl;
            cout << linea << endl;
            concat = false;
        }else{
        
            for(i = 0;i < linea.size();i++) if(!isspace(linea[i])) cout << linea[i];
            concat = true;
        }
    }
    
    if(concat) cout << endl;

    return 0;
}
