#include<iostream>
#include<string>
#include<cstring>
#include<cstdio>
using namespace std;
int main() {
	string str="", s;
	bool last=false;
	while ( getline(cin,s) ) {
		bool flag=false;
		for (int i=0;i<s.length();i++)
			if (s[i]!=' ') {
				flag=(s[i]=='#');
				break;
			}
		if (flag) {
			if (last) cout<<str<<endl;
			str="";
			cout<<s<<endl;
			last=false;
		} else {
			last=true;
			string h="";
			for (int i=0;i<s.length();i++)
				if (s[i]!=' ') h+=s[i];
			str+=h;
		}	
	}
	if (last) cout<<str<<endl;
	return 0;
}
