#include<cstdio>
#include<cstring>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

bool is_amplified(const char *s){
	for(int i=0;s[i];i++) if(s[i]!=' ') return s[i]=='#';
	return false;
}

int main(){
	static char s[1<<21],t[1<<21];
	bool pre=false;
	for(int c=0;gets(s);c++){
		int m=strlen(s);

		bool b=is_amplified(s);
		// del spaces
		if(!b){
			int n=0;
			rep(i,m) if(s[i]!=' ') t[n++]=s[i];
			t[n]='\0';
			strcpy(s,t);
		}

		if(c>0 && (pre || b)) puts("");
		printf("%s",s);

		pre=b;
	}

	puts("");

	return 0;
}
