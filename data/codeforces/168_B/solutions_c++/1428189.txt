#include<stdio.h>
#include<string.h>
char str[1048600];
int main()
{
    int len,i,pre=0;
    while(gets(str))
    {
        len=strlen(str);
        for(i=0;i<len&&str[i]==' ';i++);
        if(str[i]=='#')
        {
            if(pre) puts("");
            puts(str);
            pre=0;
        }
        else
        {
            pre=1;
            for(i=0;i<len;i++)
            {
                if(str[i]==' ') continue;
                printf("%c",str[i]);
            }
        }
    }
    if(pre) puts("");
    return 0;
}
