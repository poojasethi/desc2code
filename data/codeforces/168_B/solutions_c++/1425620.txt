#include <stdio.h>
char s[1500000];
bool isamp(char* s)
{
for(int i=0;s[i];i++)
   if(s[i] != ' ')
      return s[i] == '#';
return 0;
}
int main()
{
//freopen("input.txt","r",stdin);
//freopen("output.txt","w",stdout);
bool iyo = 0;
while(gets(s) != NULL)
   {
   if(isamp(s))
      {
      if(iyo)
         printf("\n");
      printf("%s\n",s);
      iyo = 0;
      }
   else
      {
      iyo = 1;
      for(int i=0;s[i];i++)
         if(s[i] != ' ')
            putchar(s[i]);
      }
   }
if(iyo)
   printf("\n");
return 0;
}
