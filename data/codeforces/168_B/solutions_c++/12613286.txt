#include <bits/stdc++.h>
using namespace std;

string s1, s2;
vector <string> v;

int main() {
	while(getline(cin, s1)) {
		bool ok=false;
		for(int i=0; i<s1.size(); i++) 
			if(s1[i] != ' ') {
				if(s1[i] == '#') ok=true;
				break;
			}
		if(ok) {
			v.push_back(s1);
			for(int i=0; i<v.size(); i++) cout << v[i] << endl;
			v.clear();
		}	
		else {
			stringstream ss(s1);
			if(v.empty()) v.push_back("");
			while(ss >> s2) v[v.size()-1] += s2;
		}
	}
	for(int i=0; i<v.size(); i++) cout << v[i] << endl;	
	return 0;
}
