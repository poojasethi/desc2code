#include <bits/stdc++.h>
using namespace std;

int main()
{
    string str,res;
    bool b=0;
    while(getline(cin,str)){

        int i=0;
        while(i<str.size()&&str[i]==' ')i++;

        if(i!=str.size()&&str[i]=='#'){if(b)res+='\n';res+=str; res+='\n'; b=0;}
        else{
            string add;
            for(;i<str.size();i++)
                if(str[i]!=' ')add+=str[i];
            res+=add;
            b=1;
        }

    }
    if(b)res+='\n';
    cout<<res;
    return 0;
}
