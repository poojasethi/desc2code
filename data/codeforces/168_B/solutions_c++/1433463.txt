#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <string>
#include <cstring>
using namespace std;

int main() {
	ios_base::sync_with_stdio(false);
	char line[2000000];
	bool flag = false;
	while(gets(line)) {
		int i=0;
		while(line[i]!='\0' && line[i] == ' ') i++;
		if(line[i] == '#') {
			if(flag) puts("");
			puts(line);
			flag = false;
		}
		else {
			for(int k=0; line[k]!='\0'; k++)
				if(line[k] != ' ') printf("%c", line[k]);
			flag = true;
		}
	}
	if(flag) puts("");
	return 0;
}
