#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;

char c[2000000];
int main(){
	int p=1;
	while(gets(c)!=0){
		int i;
		for(i=0;c[i]!=0;i++)
			if(c[i] != ' ') break;
		if(c[i] == '#'){
			if(p==0) puts("");
			puts(c);p=1;
		}
		else{
			for(i=0;c[i]!=0;i++)
				if(c[i] != ' ') printf("%c",c[i]); 
			p=0;
		}
	}
	if(p==0)puts("");
	return 0;
}
