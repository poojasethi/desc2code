#include<iostream>
using namespace std;
int main(){
char a[10000000];
long long int  k=0;
long long int  cur;
while(cin.getline(a,10000000)){
cur=0;
while(a[cur]==' ') cur++;
if(a[cur]=='#'){
if(k){
    cout<<"\n";
    k=0;
}

}
else{
cur=0;
k=1;
while(a[cur]){
    if(a[cur]!=' '&&a[cur]!='\n'){
        cout<<a[cur];
    }
    cur++;
}   
}

}

}
