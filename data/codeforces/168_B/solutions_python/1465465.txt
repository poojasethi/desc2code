sp = []
while 1:
    try: 
        t = raw_input()
    except:
        break
    tt = t.split()
    
    
    if len(tt) == 0 or tt[0][0] != '#':
        if len(tt) == 0: sp += [""]
        else: sp += tt
        continue
    if len(sp) > 0:
        print("".join(sp))
    sp = []
    print(t)
if sp: print("".join(sp))
