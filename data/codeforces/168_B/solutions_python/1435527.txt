import sys, math

def readln():
    return raw_input()

def parse(s, fun=int, sep=" "):
    return map(fun, s.split(sep))

def write(obj):
    sys.stdout.write(str(obj))

def writeln(obj):
    sys.stdout.write(str(obj) + "\n")

def cool(s):
    return s.lstrip() and s.lstrip()[0] == '#'

sPrev = readln()
if not cool(sPrev):
    sPrev = "".join(sPrev.split(" "))
while True:
    try:
        s = readln()
        if not cool(s):
            s = "".join(s.split(" "))
        if cool(sPrev) or cool(s):
            writeln(sPrev)
        else:
            write(sPrev)
        sPrev = s
    except EOFError:
        writeln(sPrev)
        break
