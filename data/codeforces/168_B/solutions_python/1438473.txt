import sys

prevdiez = False
not_first = False
try:
    while True:
        line = raw_input()
        st = line.lstrip()
        wasdiez = st and st[0] == '#'
        if not wasdiez:
            new = line.replace(' ', "")
            if prevdiez and not_first:
                sys.stdout.write('\n')
            sys.stdout.write(new)
        else:
            if not_first:
                sys.stdout.write('\n')
            sys.stdout.write(line)
        prevdiez = wasdiez
        not_first = True
except:
    sys.stdout.write('\n')
