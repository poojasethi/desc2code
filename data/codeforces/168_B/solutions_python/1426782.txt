def wrt(s):
    print ''.join(''.join(s.split('\n')).split(' '))

res = ''
while True:
    
    try:
        s = raw_input()+'\n'
        if len(s.strip())!=0:
            if s.strip()[0] == '#':
                if res!='':
                    wrt(res)
                    res = ''            
                print s[:-1]
            else:
                res+=s
        else:
            res+=s
            
    except EOFError:
        if res!='':
            wrt(res)
        break
