
import sys

lf=True
for line in sys.stdin:
    if(len(line.strip())==0):
        lf=False
    elif(line.lstrip()[0]=="#"):
        if(lf==False):
            lf=True
            print
        sys.stdout.write(line)
    else:
        sys.stdout.write(line.lstrip().replace(" ","").replace("\n",""))
        lf=False

if lf==False:
    print
