import sys;

flag = False;

while True:
    try:
        s = raw_input();
    except:
        break;
    if len(s.strip()) and s.strip()[0] == "#":
        if flag: sys.stdout.write("\n");
        sys.stdout.write(s + "\n");
        flag = False;
    else:
        sys.stdout.write(s.replace(" ", ""));
        flag = True;
if flag: sys.stdout.write("\n");
