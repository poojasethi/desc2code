import re
first = True
flag = False
ans = ''
while True:
    try:
        s = raw_input()
    except EOFError:
        break
    if re.match('\s*#', s): 
        if not first:
            print ans
        ans = s
        flag = True
    else:
        tmp = ''.join(s.split('\n'))
        tmp = ''.join(tmp.split(' '))
        if flag:
            ans += '\n'
        ans += tmp
        flag = False
    first = False
print ans
