import sys
nl = False
while True:
    try:
        s = raw_input()
    except:
        break;
    if len(s.strip()) and s.strip()[0] == '#':
        if nl: sys.stdout.write("\n")
        sys.stdout.write(s + "\n")
        nl = False
    else:
        sys.stdout.write(s.replace(' ', ''))
        nl = True
if nl: print ""
