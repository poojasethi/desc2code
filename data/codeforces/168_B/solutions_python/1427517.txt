#!/usr/bin/python

import sys

if __name__=='__main__':
    out = sys.stdout
    line = sys.stdin.readline()
    trans = line.translate(None,' ')
    nonspec = trans[0]!='#'
    if nonspec:
        out.write(trans[:-1])
    else:
        out.write(line)
    for line in sys.stdin.readlines():
        trans = line.translate(None,' ')
        t = trans[0]!='#'
        if nonspec:
            if t:
                out.write(trans[:-1])
            else:
                out.write('\n%s'%line)
        else:
            if t:
                out.write(trans[:-1])
            else:
                out.write(line)
        nonspec = t
    if nonspec:
        out.write('\n')
