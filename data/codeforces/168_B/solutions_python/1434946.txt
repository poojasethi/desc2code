import re

    isampre = re.compile("^ *\#")
    bufflg = False
    buf = ""
    trans = string.maketrans('','')

    for line in sys.stdin:
        line = line.rstrip("\r\f\n")
        linewk = line.lstrip()
        if linewk[:1] == '#':
            if bufflg: print buf
            print line
            bufflg = False
            buf = ""
        else:
            bufflg = True
            buf += line.translate(trans,' ')

    if bufflg: print buf
