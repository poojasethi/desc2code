import sys
s=sys.stdin.readlines();
cur=""
tag=False
for l in s:
    if l.strip() and l.strip()[0]=='#':
        if tag: print cur
        print l[:-1]
        cur=""
        tag=False
    else:
        tag=True
        for ch in l:
            if not ch.isspace():
                cur+=ch
if tag: print cur
