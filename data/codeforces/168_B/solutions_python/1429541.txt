import sys


temp = ""
flag = False
for line in sys.stdin:
    a = line.strip()
    if a and a[0] == '#':
        if flag:
            print temp
        temp = ""
        flag = False
        print line[:-1]
    else:
        temp = temp + filter(lambda x: x not in [' ', '\x0D', '\x0A'], line)
        flag = True

if flag:
    print(temp)
