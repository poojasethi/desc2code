#include <cstdio>

using namespace std;
#define FOR(x,b,e)  for(int x = b;x <= (e);x++)
#define REP(x,n)    for(int x = 0;x < (n);x++)
#define MAX_SIZE    2005

char map[MAX_SIZE][MAX_SIZE];

int n,m;

void fun(int x,int y){
    if(x < 0 || y < 0 || x == n - 1 || y == m - 1) return;

    int cnt = 0;
    FOR(i,0,1)
        FOR(j,0,1)
            if(map[i + x][j + y] == '.') cnt++;
    if(cnt == 3){
        FOR(i,0,1)
            FOR(j,0,1)
                map[x + i][y + j] = '.';
        FOR(i,-1,1)
            FOR(j,-1,1)
                fun(x + i,y + j);
    }
}
int main(){
    while(scanf("%d %d",&n,&m) != EOF){
        for(int i = 0;i < n;i++){
            scanf("%s",map[i]);
        }
        REP(i,n)
            REP(j,m)
                fun(i,j);
        REP(i,n)
            printf("%s\n",map[i]);
    }
    return 0;
}
