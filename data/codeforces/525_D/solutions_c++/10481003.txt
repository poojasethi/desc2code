#include<cstdio>
using namespace std;
int n,m;
bool a[2011][2011];
void check(int x,int y)
{
	if (x<1||x>n||y<1||y>m-1) return;
	int tmp=0,i,j;
	for (i=0;i<2;i++)
		for (j=0;j<2;j++) tmp+=a[x+i][y+j];
	if (tmp==3)
	{
		for (i=0;i<2;i++)
			for (j=0;j<2;j++) a[x+i][y+j]=true;
		for (i=-1;i<=2;i++)
			for (j=-1;j<=2;j++) check(x+i,y+j);
	}
}
int main()
{
	//freopen("1.in","r",stdin);freopen("1.out","w",stdout);
	scanf("%d%d\n",&n,&m);
	int i,j;
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=m;j++) a[i][j]=(getchar()=='.');
		getchar();
	}
	for (i=1;i<=n;i++)
		for (j=1;j<=m;j++) check(i,j);
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=m;j++) printf("%c",a[i][j]?'.':'*');
		printf("\n");
	}
	return 0;
}
