#include <bits\stdc++.h>
#include <string.h>

using namespace std;

int n,m;
char arr[2009][2009];

int dfs(int x,int y)
{
    if(x<0||y<0||x>n||y>m)
        return 0;
    int temp=0;
    for(int f=0;f<2;f++)
        for(int f1=0;f1<2;f1++)
            temp+=(arr[f+x][f1+y]=='*');
    if(temp!=1)
        return 0;
    for(int f=0;f<2;f++)
        for(int f1=0;f1<2;f1++)
            arr[f+x][f1+y]='.';
    int ret=0;
    for(int f=-1;f<=1;f++)
        for(int f1=-1;f1<=1;f1++)
            ret+=dfs(x+f,y+f1);
    return ret;
}

int main()
{
    cin>>n>>m;
    for(int f=1;f<=n;f++){
        arr[f][0]=arr[f][m+1]='*';
        for(int f1=1;f1<=m;f1++)
            scanf(" %c",&arr[f][f1]);
    }
    for(int f=0;f<m;f++)
        arr[n+1][f]=arr[0][f]='*';
    for(int f=0;f<n;f++)
        for(int f1=0;f1<m;f1++)
            dfs(f,f1);
    for(int f=1;f<=n;f++){
        for(int f1=1;f1<=m;f1++)
            printf("%c",arr[f][f1]);
        printf("\n");
    }
    return 0;
}
