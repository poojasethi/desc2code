#include <cstdio>

using namespace std;

const int auxx[8]={-1,-1,0,1,1,1,0,-1},auxy[8]={0,1,1,1,0,-1,-1,-1};
int n,m;
char v[2010][2010];

int bun(int x,int y)
{
    if(x>1 && y<m && v[x-1][y]=='.' && v[x-1][y+1]=='.' && v[x][y+1]=='.') return 0;
    if(x<n && y<m && v[x][y+1]=='.' && v[x+1][y+1]=='.' && v[x+1][y]=='.') return 0;
    if(x<n && y>1 && v[x][y-1]=='.' && v[x+1][y-1]=='.' && v[x+1][y]=='.') return 0;
    if(x>1 && y>1 && v[x][y-1]=='.' && v[x-1][y-1]=='.' && v[x-1][y]=='.') return 0;
    return 1;
}

void fill(int x,int y)
{
    if(bun(x,y)) return;
    v[x][y]='.';
    for(int i=0;i<8;i++)
    {
        int a=x+auxx[i],b=y+auxy[i];
        if(a<1 || a>n || b<1 || b>m) continue;
        if(v[a][b]=='*') fill(a,b);
    }
}

int main()
{
    //freopen("file.in", "r", stdin);
    scanf("%d %d",&n,&m);
    for(int i=1;i<=n;i++) scanf("\n%s",v[i]+1);
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            if(v[i][j]=='*') fill(i,j);
    for(int i=1;i<=n;i++) printf("%s\n",v[i]+1);
    return 0;
}
