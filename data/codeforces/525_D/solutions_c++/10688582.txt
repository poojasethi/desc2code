#include<iostream>
using namespace std;
#include<cstring>
#include<cstdio>
char a[2000+2][2000+2];
int n,m;
//int vis[2000+2][2000+2];
void gao(int x,int y){
	if(x==-1 || y==-1 || x==n-1 || y==m-1) return;
	int cnt=0;
	for(int i=0;i<=1;i++){
		for(int j=0;j<=1;j++){
			if(a[x+i][y+j]=='*') cnt++;
		}
	}
	if(cnt==1){
		for(int i=0;i<=1;i++){
			for(int j=0;j<=1;j++){
				a[x+i][y+j]='.';
				//gao(x+i,y+j);
			}
		}
		
		for(int i=-1;i<=1;i++){
			for(int j=-1;j<=1;j++){
				gao(x+i,y+j);
			}
		}
		
	} else {
		//vis[x][y]=1;
	}
}
int main()
{
	//memset(vis,0,sizeof(vis));
	scanf("%d%d",&n,&m); //cin>>n>>m;
	
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			scanf(" %c",&a[i][j]); //cin>>a[i][j];
		}
	}
	for(int i=0;i<n-1;i++){
		for(int j=0;j<m-1;j++){
			gao(i,j); //}
		}
	}
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			printf("%c",a[i][j]);//cout<<a[i][j];
		}
		printf("\n");//cout<<endl;
	}
	
	return 0;
}
