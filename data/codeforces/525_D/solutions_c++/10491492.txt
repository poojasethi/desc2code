#include <iostream>

using namespace std;


int a[2005][2005];

int rooms (int i, int j){
	return a[i][j] + a[i+1][j] + a[i][j+1] + a[i+1][j+1];
}

void initialize(int m){
	for (int i=0; i<2005; i++){
		for (int j=0; j<2005; j++){
			a[i][j]= m;
		}
	}
}	

void replace(int x, int y){
	if (rooms(x, y)==3){
		for (int i=0; i<2; i++){
			for (int j=0; j<2; j++){
				a[x+i][y+j]= 1;
			}
		}

		for (int i=-1; i<=1; i++){
			for (int j=-1; j<=1; j++){
				replace(x+i, y+j);
			}
		}
	}
	
}

int main() {
	ios::sync_with_stdio(false);
	initialize(100);
	char x;
	int n, m;
	cin>>n;
	cin>>m;
	for (int i=1; i<=n; i++){
		for (int j=1; j<=m; j++){
			cin>>x;
			if (x=='.'){
				a[i][j]= 1;
			}
			else{
				a[i][j]= 0;
			}
		}
	}
	for (int i=1; i<=n; i++){
		for (int j=1; j<=m; j++){
			if (rooms(i, j)==3){
				replace(i, j);
			}
		}
	}

	
	for (int i=1; i<=n; i++){
		for (int j=1; j<=m; j++){
			if (a[i][j]==0){
				cout<<'*';
			}
			else{
				cout<<'.';
			}
		}
		cout<<endl;
	}
	
	return 0;
}
