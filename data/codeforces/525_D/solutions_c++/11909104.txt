#include<stdio.h>
#define F(i,m,n) for(i=m;i<=n;i++)
char A[2002][2002];
int n, m;
void f(int i, int j){

    int x,y,s=0;

    if(i < 1 || j < 1 || i == m  || j == n) return ;

    F(x,0,1)
        F(y,0,1)
            s += (A[i+x][y+j] == '*');

    if(s == 1){
        
        F(x,0,1)
            F(y,0,1)
                A[i+x][y+j] = '.';

        F(x,-1,1)
            F(y,-1,1)
               f(i+x,j+y);
    }
}
int main(){
    int i,j;
    scanf("%d %d",&m,&n);
    F(i,1,m)
        scanf("%s",A[i]+1);
    F(i,1,m-1) F(j,1,n-1) f(i,j);
    F(i,1,m)
        printf("%s\n",A[i]+1);
}
