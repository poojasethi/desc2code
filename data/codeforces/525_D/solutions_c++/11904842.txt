#include<bits/stdc++.h>
using namespace std;
string grid[20005];
int n,m;
int out[2005][2005];
int a_x[]={1,-1,0,0,1,-1,-1,1};
int a_y[]={0,0,1,-1,1,-1,1,-1};
void dfs(int x,int y)
{
  if(x<0 || y<0)return;
  if(y>m || x>n)return;
  int cnt=0,fx,fy;
   for(int i=0;i<2;i++)
      for(int j=0;j<2;j++)
       {
          if(out[x+i][y+j])cnt++;
          else fx=x+i,fy=y+j;
        }
    if(cnt!=3)return;
    out[fx][fy]=true;
    for(int i=0;i<8;i++)
         dfs(x+a_x[i],y+a_y[i]);
}
int main()
{
 scanf("%d %d",&n,&m);
  for(int i=0;i<n;i++)cin>>grid[i];

   for(int i=0;i<n;i++)
     for(int j=0;j<m;j++)out[i][j]=(grid[i][j]=='.');


   for(int i=0;i<n;i++)
    for(int j=0;j<m;j++)
       dfs(i,j);

    for(int i=0;i<n;i++){
    for(int j=0;j<m;j++){
         cout<<((out[i][j]==1)?'.':'*');
    }cout<<endl;}
 return 0;
}
