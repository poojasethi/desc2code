int n,m;
char arr[2010][2010],brr[2010][2010];
void dfs(int x,int y){
	int s=0;
	if(x < 1 || x >= n || y < 1 || y >= m) return;
	if(arr[x][y] == '*') s++;
	if(arr[x+1][y] == '*') s++;
	if(arr[x][y+1] == '*') s++;
	if(arr[x+1][y+1] == '*') s++;
	if(s == 1){
		arr[x][y] = arr[x][y+1] = arr[x+1][y] = arr[x+1][y+1] = '.';
		FOR(i,-1,1)
			FOR(j,-1,1)
				dfs(x+i,y+j);
	}
}	
int main(){
	cin >> n >> m;
	FOR(i,1,n)
		scanf("%s",arr[i]+1);

	FOR(i,1,n-1)
		FOR(j,1,m-1)
			dfs(i,j);
	FOR(i,1,n)
		printf("%s\n",arr[i]+1);
}
