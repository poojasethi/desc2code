#include<bits/stdc++.h>
using namespace std;
char s[2005][2005];
int vis[2005][2005];
int n,m;

void solve(int x,int y){
	int i,j;
	if(s[x][y]=='.') return;
	if(s[x-1][y-1]=='.' && s[x-1][y]=='.' && s[x][y-1]=='.') s[x][y]='.';  
    else if(s[x-1][y+1]=='.' && s[x-1][y]=='.' && s[x][y+1]=='.') s[x][y]='.';  
    else if(s[x+1][y+1]=='.' && s[x+1][y]=='.' && s[x][y+1]=='.') s[x][y]='.';  
    else if(s[x+1][y-1]=='.' && s[x+1][y]=='.' && s[x][y-1]=='.') s[x][y]='.';
	if(s[x][y]=='.'){
		for(i=-1;i<=1;i++){
			for(j=-1;j<=1;j++){
				if(!i && !j) continue;
				solve(x+i,y+j);
			}
		}
	} 
}

int main(void){
	int i,j;
	while(~scanf("%d%d",&n,&m)){
		for(i=1;i<=n;i++) scanf("%s",s[i]+1);
		for(i=1;i<=n;i++)
			for(j=1;j<=m;j++)
				solve(i,j);
		for(i=1;i<=n;i++){
			for(j=1;j<=m;j++)
				printf("%c",s[i][j]);
			puts("");
		}
	}
	return 0;
}
