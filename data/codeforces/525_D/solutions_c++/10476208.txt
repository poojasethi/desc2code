#include<cstdio>
#include<vector>
#include<queue>

using namespace std;

typedef pair<int,int> pii;

const int SIZE=2005;
char S[2005][2005];
queue<pii> Q;

int main(void)
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=2;i<=n+1;i++)
		scanf("%s",S[i]+2);
	for(int i=2;i<=n+1;i++)
		for(int j=2;j<=m+1;j++)
			if(S[i][j]=='.')
				Q.push(pii(i,j));
	for(int i=0;i<SIZE;i++)
		for(int j=0;j<SIZE;j++)
			if(S[i][j]!='.')
				S[i][j]='*';
	while(!Q.empty())
	{
		pii k=Q.front();Q.pop();
		int x=k.first,y=k.second;
		for(int i=0;i<2;i++)
			for(int j=0;j<2;j++)
			{
				pii __;
				int star=0;
				for(int k=0;k>=-1;k--)
					for(int l=0;l>=-1;l--)
						if(S[x+i+k][y+j+l]=='*')
							__=pii(x+i+k,y+j+l),star++;
				if(star==1)
				{
					//printf("%d %d\n",__.first,__.second);

					Q.push(__);
					S[__.first][__.second]='.';
				}
			}
	}
	for(int i=2;i<=n+1;i++)
	{
		for(int j=2;j<=m+1;j++)
			putchar(S[i][j]);
		puts("");
	}
	return 0;
}
