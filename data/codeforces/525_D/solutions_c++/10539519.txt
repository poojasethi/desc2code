using namespace std;
char mp[2003][2003];
int n,m;
void f(int x,int y)
{
	if(x<0 || y<0 || x>=n-1 || y>=m-1)return ;
	
	int c=0,i,j;
	for(i=0;i<2;i++)for(j=0;j<2;j++)if(mp[x+i][y+j]=='*')c++;
	
	if(c==1)
	{
		for(i=0;i<2;i++)for(j=0;j<2;j++)mp[x+i][y+j]='.';
		for(i=-1;i<2;i++)for(j=-1;j<2;j++)f(x+i,y+j);
	}
}

int main()
{
	while(~scanf("%d%d",&n,&m))
	{
		int i,j;
		for(i=0;i<n;i++)
		cin>>mp[i];
		
		for(i=0;i<n;i++)for(j=0;j<m;j++)f(i,j);
		for(i=0;i<n;i++)printf("%s\n",mp[i]);
	}
}
