#include <bits/stdc++.h>
#define N 2100

using namespace std;

int n, m;
string b[N];

void Solve_(int x, int y)
{
  int c = 0;
  if (x == n-1 || y == m-1 || x < 0 || y < 0) return;
  for (int i = x; i < x + 2; i++)
    for (int j = y; j < y + 2; j++) if (b[i][j] == '*') ++c;
  if (c == 1)
  {
    for (int i = x; i < x + 2; i++)
      for (int j = y; j < y + 2; j++) b[i][j] = '.';
    for (int i = -1; i <= 1; i++)
      for (int j = -1; j <= 1; j++)
        Solve_(x + i, y + j);
  }
}

int main()
{
  ios_base::sync_with_stdio(false);
  //freopen("d.in","r",stdin);
  //freopen("d.out","w",stdout);
  cin >> n >> m;
  for (int i = 0; i < n; i++) cin >> b[i];
  for (int i = 0; i < n; i++)
    for (int j = 0; j < m; j++)
      Solve_(i, j);
  for (int i = 0; i < n; i++, cout << endl)
    for (int j = 0; j < m; j++)
      cout << b[i][j];
}
