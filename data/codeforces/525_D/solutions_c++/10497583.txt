#include <bits/stdc++.h>
using namespace std;

#define fi first
#define se second
#define mp make_pair
typedef pair<int,int> pii;
int n,m;
char inp[2005][2005];
queue<pair<int,int> > que;
void bfs(int r,int c){
	que.push(mp(r,c));

	while(!que.empty()){
		pii f = que.front();
		que.pop();


		if(!(0<=f.fi && f.fi<n && 0<=f.se && f.se<m))continue;

		int cnt=0;
		for(int q=0;q<2;q++){
			for(int w=0;w<2;w++){
				if(inp[f.fi+q][f.se+w]=='.')cnt++;
			}
		}
		if(cnt==3){
			for(int q=0;q<2;q++){
				for(int w=0;w<2;w++){
					inp[f.fi+q][f.se+w]='.';
				}
			}
			for(int q=-1;q<2;q++){
				for(int w=-1;w<2;w++){
					que.push(mp(f.fi+q,f.se+w));
				}
			}
		}
	}
}
int main(){
	cin>>n>>m;
	for(int q=0;q<n;q++){
		cin>>inp[q];
	}

	for(int q=0;q<n;q++){
		for(int w=0;w<m;w++){
			bfs(q,w);
		}
	}

	for(int q=0;q<n;q++){
		for(int w=0;w<m;w++){
			cout<<inp[q][w];
		}
		cout<<"\n";
	}

	return 0;
}
