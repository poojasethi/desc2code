const long long A=100000000000000LL,N=2280,M=(int)1e6;

char a[N][N];
long long q1,q2,xx[M],yy[M],i,j,n,m,x,y;

void rec(long long x,long long y){
	if(x<0 || y<0 || x>=n-1 || y>=m-1)return;
	long long i,j,k=0;
	for(i=0;i<2;i++)for(j=0;j<2;j++)if(a[x+i][y+j]=='*')k++;
	if(k==1){
		for(i=0;i<2;i++)for(j=0;j<2;j++)a[x+i][y+j]='.';
		for(i=-1;i<2;i++)for(j=-1;j<2;j++)rec(x+i,y+j);
	}
}

int main(){
	cin>>n>>m;
	for(i=0;i<n;i++)scanf("%s",a[i]);
	for(i=0;i<n;i++)for(j=0;j<m;j++)rec(i,j);
	for(i=0;i<n;i++,cout<<"\n")for(j=0;j<m;j++)cout<<a[i][j];
}
