#include <bits/stdc++.h>

using namespace std;

#define mp make_pair
#define pb push_back
#define INF 99999999999
#define NMAX 2000+50
#define MOD 1e9+7


char t[NMAX][NMAX];
long long n, m, i_[]={0,1,0,-1,1,1,-1,-1}, j_[]={1,0,-1,0,-1,1,1,-1};

void sol(long long i, long long j)
{
    if((t[i][j]=='.')+(t[i+1][j]=='.')+(t[i][j+1]=='.')+(t[i+1][j+1]=='.')==3)
    {
        t[i][j]=t[i+1][j]=t[i][j+1]=t[i+1][j+1]='.';
        for(int d=0;d<8;d++)
            if(i+i_[d]>=0 and j+j_[d]>=0 and i+i_[d]<n-1 and j+j_[d]<m-1)
                sol(i+i_[d], j+j_[d]);
    }
}

int main()
{
    cin>>n>>m;
    for(int i=0;i<n;i++)
        scanf("%s", t[i]);
    for(int i=0;i<n-1;i++)
        for(int j=0;j<m-1;j++)
            sol(i, j);
    for(int i=0;i<n;i++)
        printf("%s\n", t[i]);
}
