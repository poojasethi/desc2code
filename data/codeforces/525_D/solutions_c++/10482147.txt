#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>

using namespace std;

char s[2500][2500];
int n,m;
int check(int x,int y)
{
	if (x>1&&y<m&&s[x-1][y]=='.'&&s[x-1][y+1]=='.'&&s[x][y+1]=='.') return 0;
	if (x<n&&y<m&&s[x+1][y]=='.'&&s[x+1][y+1]=='.'&&s[x][y+1]=='.') return 0;
	if (x<n&&y>1&&s[x+1][y-1]=='.'&&s[x+1][y]=='.'&&s[x][y-1]=='.') return 0;
	if (x>1&&y>1&&s[x-1][y-1]=='.'&&s[x-1][y]=='.'&&s[x][y-1]=='.') return 0;
	return 1;
}

void work(int x,int y)
{
	int dqx,dqy;
	if (check(x,y)) return;
	s[x][y]='.';
	for (int i=-1;i<=1;++i)
		for (int j=-1;j<=1;++j)
		{
			if (i==0&&j==0) continue;
			dqx=x+i,dqy=y+j;
			if (dqx<1||dqx>n||dqy<1||dqy>m) continue;
			if (s[dqx][dqy]=='*') work(dqx,dqy);
		}
}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i) scanf("%s",s[i]+1);
	for (int i=1;i<=n;++i)
		for (int j=1;j<=m;++j) work(i,j);
	for (int i=1;i<=n;++i) printf("%s\n",s[i]+1);
	return 0;
}
