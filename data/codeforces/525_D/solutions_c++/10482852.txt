#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int xx[]={-1,-1,0,0};
int yy[]={0,-1,0,-1};
int n,m;
char s[2005][2005];
int check(int x,int y)
{
  int i,j,cnt=0;
  if (x<0||y<0||x>=n-1||y>=m-1) return 0;
  for (i=x;i<=x+1;i++)
	for (j=y;j<=y+1;j++)
	  if (s[i][j]=='*') cnt++;
  return cnt==1;
}
int dfs(int x,int y)
{
  for (int tx=x;tx<=x+1;tx++)
	for (int ty=y;ty<=y+1;ty++)
	  if (s[tx][ty]=='*'){
		s[tx][ty]='.';
		for (int i=0;i<4;i++)
		  if (check(tx+xx[i],ty+yy[i])){
			dfs(tx+xx[i],ty+yy[i]);
		  }
		return 0;
	  }
}
int main()
{
  int i,j;
  scanf("%d%d",&n,&m);
  for (i=0;i<n;i++)
	scanf("%s",s[i]);
  for (i=0;i<=n-2;i++)
	for (j=0;j<=m-2;j++)
	  if (check(i,j)) dfs(i,j);
  for (i=0;i<n;i++) printf("%s\n",s[i]);
  return 0;
}
