typedef long long ll;
using namespace std;

int n,m,i,j;
string g[20010];
void check(int x,int y)
{
	 if(x<0||x>=n||y<0||y>=m)
        return;
    int cnt=0;
    for(int i=0;i<2;i++)
        for(int j=0;j<2;j++)
            if(g[x+i][y+j]=='.')
                cnt++;
    if(cnt==3)
    {
        for(int i=0;i<2;i++)
            for(int j=0;j<2;j++)
                g[x+i][y+j]='.';
        for(int i=-1;i<2;i++)
            for(int j=-1;j<2;j++)
                check(x+i,y+j);
    }
}
int main()
{
	cin>>n;
	cin>>m;
    for(i=0;i<n;i++)
    {
            cin>>g[i];
    }
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            check(i,j);
        }
    }
    for(int i=0;i<n;i++)
    {
        cout<<g[i]<<endl;
    }
//	system("pause");
}
