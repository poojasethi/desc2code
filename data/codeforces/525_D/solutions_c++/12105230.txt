#include <cstdio>
using namespace std;
int n, m;
char a[2005][2005];

void dfs(int x, int y){
    if (x < 0 || y < 0 || x >= n - 1 || y >= m - 1) return;
    int cnt = 0;
    for (int i = 0; i < 2; ++ i){
        for (int j = 0; j < 2; ++ j){
            cnt += a[x+i][y+j] == '*';
        }   
    }
    if (cnt != 1) return;
    for (int i = 0; i < 2; ++ i){
        for (int j = 0; j < 2; ++ j){
            a[x+i][y+j] = '.';
        }
    }
    for (int i = -1; i < 2; ++ i){
        for (int j = -1; j < 2; ++ j){
            dfs(x+i, y+j);
        }
    }
}

int main(){
    scanf("%d%d", &n, &m);
    for (int i = 0; i < n; ++ i){
        scanf("%s", a[i]);
    }
    for (int i = 0; i < n; ++ i){
        for (int j = 0; j < m; ++ j){
            dfs(i, j);
        }
    }
    for (int i = 0; i < n; ++ i){
        printf("%s\n", a[i]);
    }
    return 0;
}
