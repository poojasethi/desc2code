#include <bits/stdc++.h>

using namespace std;

char a[2005][2005];
int n,m;

void doit(int i, int j) {
    if (i<0 || j<0 || i==n-1 || j==m-1) return;
    int s=0;
    for (int x=0;x<=1;x++)
    for (int y=0;y<=1;y++)
        s+=(a[i+x][j+y]=='*');
    if (s==1) {
        for (int x=0;x<=1;x++)
            for (int y=0;y<=1;y++)
                a[x+i][y+j]='.';
        for (int x=-1;x<=1;x++)
            for (int y=-1;y<=1;y++) doit(x+i,y+j);
    }
}

int main() {
    //freopen("A.inp","rt",stdin);
    //freopen("A.out","wt",stdout);
    scanf("%d%d\n",&n,&m);
    for (int i=0;i<n;i++) scanf("%s\n",a[i]);
    for (int i=0;i<n-1;i++)
        for (int j=0;j<m-1;j++) doit(i,j);
    for (int i=0;i<n;i++) {
        for (int j=0;j<m;j++)
            cout << a[i][j];
        cout << endl;
    }
    return 0;
}
