#include <bits/stdc++.h>
#define LL long long
using namespace std;
const int maxn = 2222;
int n, m;
char g[maxn][maxn];

void dfs(int x, int y)
{
    if (x < 0 || x >= n || y < 0 || y >= m)
        return;
    if ((g[x][y] == '.') + (g[x+1][y] == '.') + (g[x][y+1] == '.') + (g[x+1][y+1] == '.') == 3)
    {
        g[x][y] = g[x][y+1] = g[x+1][y] = g[x+1][y+1] = '.';
        for(int i = -1; i <= 1; ++i)
            for(int j = -1; j <= 1; ++j)
                dfs(x + i, y + j);
    }
}

int main()
{
    cin >> n >> m;
    for(int i = 0; i < n; ++i)
        scanf("%s", g[i]);
    for(int i = 0; i < n; ++i)
        for(int j = 0; j < m; ++j)
            dfs(i, j);
    for(int i = 0; i < n; ++i)
        puts(g[i]);
    return 0;
}
