n=int(raw_input())
a=map(int,raw_input().split())
a.sort()
m=int(raw_input())
b=map(int,raw_input().split())
b.sort()
ans=0
for i in range(n):
    for j in range(m):
        if abs(a[i]-b[j])<=1:
            ans+=1
            m-=1
            b.pop(j)
            break
        elif b[j]>a[i]:
            break
print ans
