n = input()
a = [int(i) for i in raw_input().split()]
m = input()
b = [int(i) for i in raw_input().split()]
a.sort()
b.sort()
i = j = ans = 0
while (i < n and j < m):
    if abs(a[i]-b[j]) <= 1:
        ans += 1
        i += 1
        j += 1
    elif a[i] < b[j]:
        i += 1
    else:
        j += 1
print ans
