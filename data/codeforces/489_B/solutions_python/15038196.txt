n = int(raw_input())
boys = [int(i) for i in raw_input().split()]
m = int(raw_input())
girls = [int(i) for i in raw_input().split()]

boys.sort()
girls.sort()
count = 0
for i in range(n):
    for j in range(m):
        if abs(boys[i] - girls[j]) <= 1:
            girls[j] = 200
            count += 1
            break
print count
