v=input()
lv=map(int,raw_input().split())
m=input()
lm=map(int,raw_input().split())
lv.sort()
lm.sort()
x,y=0,0
ans=0
ulty=0
while x<v and y<m:
    
    if abs(lv[x]-lm[y])<2:
        ans+=1
        x+=1
        ulty=y+1
        y+=1
    else:
        y+=1
        if y==m:
            y=ulty
            x+=1
print ans
