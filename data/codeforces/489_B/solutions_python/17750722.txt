n = int(raw_input())
boys = sorted(map(int, raw_input().split()))
m = int(raw_input())
girls = sorted(map(int, raw_input().split()))
b, g, matches = 0, 0, 0
while b < n and g < m:
  if boys[b] - girls[g] > 1: g += 1
  elif boys[b] - girls[g] < -1: b += 1
  else: b += 1; g += 1; matches += 1
print matches
