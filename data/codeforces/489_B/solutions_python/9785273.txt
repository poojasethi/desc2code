n = int(raw_input())
boys = map(int,raw_input().split())
m = int(raw_input())
girls = map(int,raw_input().split())
count = 0
boys.sort()
girls.sort()

for i in range(n):
    for j in range(m):
        if(abs(boys[i] - girls[j]) <= 1):
            girls[j] = 99999
            count += 1
            break
    
print count
