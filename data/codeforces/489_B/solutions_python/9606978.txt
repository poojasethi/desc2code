a = sorted(map(int, raw_input().split()))
m = input()
b = map(int, raw_input().split())
occur = [b.count(i) for i in range(102)]

ans = 0
for i in a:
    if sum(occur[i-1:i+2]):
        ans += 1
    if occur[i-1]:
        occur[i-1] -= 1
    elif occur[i]:
        occur[i] -= 1
    elif occur[i+1]:
        occur[i+1] -= 1

print ans
