n = int(raw_input())
arn = map(int,raw_input().split())

m = int(raw_input())
arm = map(int,raw_input().split())

arn.sort()
arm.sort()
p = 0
i,j = 0,0
while i<n and j<m:
    if abs(arn[i]-arm[j]) <= 1:
        p += 1
        i+=1
        j+=1
    elif arn[i]>arm[j]:
        j+=1
    else:
        i+=1
        
print p
