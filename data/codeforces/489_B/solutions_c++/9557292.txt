#include<cstdio>
#include<algorithm>
using namespace std;
int a1[2][102];
int main()
{
	int n,a,o=0;
	for(int i=0;i<2;++i)
	{
		scanf("%d",&n);
		while(n--)
		{scanf("%d",&a);++a1[i][a];}
	}
	for(int i=1;i<101;++i)
	for(int j=-1;j<2;++j)
	{
		a=min(a1[0][i],a1[1][i+j]);
		o+=a;a1[0][i]-=a;a1[1][i+j]-=a;
	}
	printf("%d\n",o);
}
