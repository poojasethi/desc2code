#include <bits/stdc++.h>
using namespace std;
int a[5005];
int b[5005];
int main()
{
	int n,m;
	cin>>n;
	for(int i=0;i<n;i++)cin>>a[i];
	cin>>m;
	for(int i=0;i<m;i++)cin>>b[i];
	sort(a,a+n);sort(b,b+m);
	int ind=0;
	int ans=0;
	for(int i=0;i<n;i++)
	{
		while(ind<m && b[ind]<a[i]-1)ind++;
		if(ind==m)break;
		if(b[ind]<=a[i]+1)ans++,ind++;
	}
	cout<<ans<<endl;
	return 0;
}
