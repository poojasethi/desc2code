#include<stdio.h>
#include<algorithm>
using namespace std;
int n,m,a[1000],b[1000],c,i,j;
main ()
{
scanf("%d",&n);
for(i=0;i<n;i++)
scanf("%d",&a[i]);
scanf("%d",&m);
for(i=0;i<m;i++)
scanf("%d",&b[i]);
sort(a,a+n);
sort(b,b+m);
for(i=0;i<n;i++)
{
for(j=0;j<m;j++)
    if((b[j]>=a[i]-1)&&(b[j]<=a[i]+1))
{
    c++;
    b[j]=-1;
   break;
}
}

printf("%d",c);
}
