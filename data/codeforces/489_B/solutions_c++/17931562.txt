#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,m,i,j,ans=0;
	int b[100],g[100];
	cin>>n;
	for(i=0;i<n;i++)cin>>b[i];
	cin>>m;
	for(i=0;i<m;i++)cin>>g[i];
	sort(b,b+n);
	sort(g,g+m);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			if(abs(b[i]-g[j])<=1)
			{
				ans++;
				//cout<<i<<' '<<j<<"  "<<b[i]<<' '<<g[j]<<endl;
				g[j]=b[i]=102;
			}
		}
	}
	cout<<ans;
}
