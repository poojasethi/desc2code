#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	long long int n,i,a[105],l=0,r=0,c=0,m,b[105];
	cin>>n;
	for(i=0;i<n;i++)
	cin>>a[i];
	cin>>m;
	for(i=0;i<m;i++)
	cin>>b[i];
	sort(a,a+n);
	sort(b,b+m);
	while(l<n && r<m)
	{
		if(abs(a[l]-b[r])<=1)
		{
			l++;
			r++;
			c++;
		}
		else if(a[l]>b[r])
		r++;
		else
		l++;
	}
	cout<<c;
	return 0;
}
