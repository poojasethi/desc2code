

#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,i,z,j,a[101],b[101],cnt;
	cnt=0;z=1;
	cin>>n;
	for(i=1;i<=n;i++) cin>>a[i];
	cin>>m;
	for(j=1;j<=m;j++) cin>>b[j];
	sort(a+1,a+n+1);
	sort(b+1,b+m+1);
	for(i=1;i<=n;i++)
	for(j=z;j<=m;j++)
	{
		if(abs(a[i]-b[j])<=1) {
			cnt++;
			z=j;
			a[i]=-1;
			b[j]=-2;
			break;
		}
	}
	cout<<cnt;
}
