#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
int a[105],b[105];
int main()
{int n; cin>>n;
int i=0,j=0,k=0;
 for(int i=0; i<n; i++) cin>>a[i];
 int m; cin>>m;
 for(int j=0; j<m; j++) cin>>b[j];
sort(a,a+n);
sort(b,b+m);
i=0; j=0;
while (i!=n&&j!=m)
{if (abs(a[i]-b[j])<=1) {k++; i++; j++;}
else if (a[i]<b[j]) i++; else j++;}
    cout <<k;
    return 0;
}
