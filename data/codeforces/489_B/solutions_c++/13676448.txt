#include<bits/stdc++.h>
using namespace std;
int n, m, i, j, a[100], b[100], ans;
int main(){
  cin>>n;
  for(i=0;i<n;i++)cin>>a[i];
  cin>>m;
  for(j=0;j<m;j++)cin>>b[j];
  sort(a,a+n);
  sort(b,b+m);
  for(i=0;i<n;i++){
    for(j=0;j<m;j++){
      if(abs(a[i]-b[j])<2){
        ans++;
        b[j]=-10;
        break;
      }
    }
  }
  cout<<ans;
}
