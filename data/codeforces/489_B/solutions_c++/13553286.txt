#include <bits/stdc++.h>
using namespace std;
int main() {
	int f[105]={0},s1=0,s2=0,i,j,k,l,n,m,a[105],b[105];
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	cin>>m;
	for(i=0;i<m;i++)
		cin>>b[i];
	sort(a,a+n);
	sort(b,b+m);
	for(i=0;i<n;i++)
		for(j=0;j<m;j++) 
			if(!f[j] && abs(a[i]-b[j])<2) {
				f[j] = 1;
				break;
			}
	for(i=0;i<m;i++) {
		s1+=f[i];
	}
	cout<<s1<<endl;
	return 0;
}
