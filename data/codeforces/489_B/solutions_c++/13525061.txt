#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
int i,j,m,n,count=0;

cin>>n;
int a[n];
for(i=0;i<n;i++)
cin>>a[i];
sort(a,a+n);
cin>>m;
int b[m];
for(i=0;i<m;i++)
cin>>b[i];
sort(b,b+m);
i=0;
j=0;
while(i<n && j<m)
{

if(abs(a[i]-b[j])>1)
{
if(a[i]>b[j])
j++;
else
i++;
}
else
{
count++;
i++;
j++;
}
}
cout<<count;
return 0;

}
