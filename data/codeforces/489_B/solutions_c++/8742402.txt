#include <iostream>
#include <algorithm>
using namespace std;
int main(){
int m,n;
int O[200],G[200];

cin>>n;
for(int i=0;i<n;i++)
cin>>O[i];

cin>>m;
for(int i=0;i<m;i++)
cin>>G[i];

sort(O, O+n);
sort(G, G+m);

int a=0,b=0;
int ans=0;

while(a<n && b<m)
{

if(abs(O[a]-G[b])<2)
{
a++;
b++;
ans++;
}

else if(O[a]<G[b])
a++;

else
b++;

}

cout<<ans;

}
