#import<bits/stdc++.h>
using namespace std;
int n,m,s,a[105],b[105];
int main() {
    int i,j;
    cin>>n;
    for (i=0; i<n; i++) cin>>a[i];
    cin>>m;
    for (i=0; i<m; i++) cin>>b[i];
    sort(a, a+n);
    sort(b, b+m);
    for (i=0,j=0; i<n && j<m;)
        if (abs(a[i] - b[j]) <= 1) s++, i++, j++;
        else if (a[i] < b[j]) i++;
        else j++;
    cout<<s;
}
