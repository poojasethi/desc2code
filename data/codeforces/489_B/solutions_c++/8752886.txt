#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
int a[103],b[103];
int main()
{int n,m,i0=0,j0=0,k=0;cin>>n;
    for (int i=0;i<n;i++) cin>>a[i];
cin>>m;
   for (int i=0;i<m;i++) cin>>b[i];
sort(a,a+n);
sort(b,b+m);
while(i0!=n&&j0!=m) {
    if(abs(a[i0]-b[j0])<=1) 
        {k++;j0++;i0++;}
    else if(a[i0]<b[j0])i0++;
    else j0++;}

cout<<k;
    return 0;
}
