#include <bits/stdc++.h>
using namespace std;

int main() {
int n,m,a[1002],b[1002],i,j;
cin>>n;
for(i=0;i<n;i++)
cin>>a[i];
cin>>m;
for(i=0;i<m;i++)
cin>>b[i];
sort(a,a+n);
sort(b,b+m);
int c=0;
for(i=0,j=0;i<n&&j<m;)
{
    if(abs(a[i]-b[j])<=1)
    {
        c++;
        i++;
        j++;
    }
    else
    if(a[i]<b[j])
    i++;
    else
    j++;
}
cout<<c;
    return 0;
}
