#include<bits/stdc++.h>

using namespace std;

int n,m,i,j,ans,a[101],b[101];

int main()
{
    cin>>n;
    for(i=0;i<n;++i)
        cin>>a[i];
    cin>>m;
    for(i=0;i<m;++i)
        cin>>b[i];
    sort(a,a+n);
    sort(b,b+m);
    i=0; j=0;
    while(i<n&&j<m)
    {
        if(abs(a[i]-b[j])<=1){++i; ++j; ++ans;}
            else a[i]>b[j]?++j:++i;
    }
    cout<<ans;
}
