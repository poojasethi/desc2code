#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin >> n;
	int b[n];
	for(int i=0;i<n;++i)
		cin >> b[i];
	int m;
	cin >> m;
	int g[m];
	for(int i=0;i<m;++i)
		cin >> g[i];
	sort(b,b+n);
	sort(g,g+m);
	int i=0;
	int j=0;
	int count=0;
	while(i<n&&j<m)
	{
		
		if(abs(b[i]-g[j])<=1)
		{
			
			++count;
			++i;
			++j;
		}
		else if(b[i]>g[j])
				++j;
				else
				++i;
	}
		cout << count;
}
