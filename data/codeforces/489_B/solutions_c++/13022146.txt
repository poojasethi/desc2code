#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,i,j,count=0;
	int a[1000],b[1000];
	cin>>n;
	for(i=0;i<n;i++)
	cin>>b[i];

	cin>>m;
	for(i=0;i<m;i++)
	cin>>a[i];
	sort(b,b+n);
	sort(a,a+m);

	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if(abs(b[i]-a[j]) <= 1)
			{
				count++;
				a[j]=55555;
				break;
			}
		}
	}
	cout<<count;
	return 0;
}
