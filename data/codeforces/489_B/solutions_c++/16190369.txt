#include <bits/stdc++.h>
using namespace std;

int n,m,b[101],g[101];

int main(){
	cin >> n;
	for(int i=0;i<n;i++) cin >> b[i];
	cin >> m;
	for(int i=0;i<m;i++) cin >> g[i];
	sort(b,b+n);
	sort(g,g+m);
	
	int res=0;
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			if(abs(b[i]-g[j])<=1){
				res++;
				g[j]=1001;
				break;
			}
		}
	}
	cout << res;
	
}
