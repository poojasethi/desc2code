#include <iostream>
#include <algorithm>
using namespace std;

int n, m, A[200], B[200];

main(){
	cin >> n;
	for(int i=0; i<n; ++i) cin >> A[i];
	cin >> m;
	for(int i=0; i<m; ++i) cin >> B[i];
	
	sort(A, A+n);
	sort(B, B+m);

	int ans = 0;
	for(int i=0, j=0; i<n && j < m;){
		if(abs(A[i] - B[j]) <=1 ) {
			ans++;
			i++;
			j++;
		}
		else if(A[i] < B[j]) i++;
		else j++;
	}

	cout << ans << endl;
}
