#include<bits/stdc++.h>
using namespace std;
long long n[1001],m[1001],a,b,d=0;
int main(){
	
	cin>>a;
	
	for(int i=0;i<a;i++)
	cin>>n[i];
	
	sort(n,n+a);
	
	cin>>b;
	
	for(int i=0;i<b;i++)
	cin>>m[i];
		
	sort(m,m+b);	
		
	
	for(int i=0;i<a;i++){
		for(int j=0;j<b;j++){
		if(abs(n[i]-m[j])<=1){
		d++;
		m[j]=-1001;
		j=b;
		}
	
	}
		
	}
	
	cout<<d<<endl;
	return 0;
}
