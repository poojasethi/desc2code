#include<iostream>
#include<algorithm>
using namespace std;
int n,m,k,a[3010],b[3010];
bool c[3010];
int main()
{
	cin>>n;
	for (int i=0;i<n;i++)
		cin>>a[i];
	sort(a,a+n);
	cin>>m;
	for (int i=0;i<m;i++)
		cin>>b[i];
	sort(b,b+m);
	for (int i=0;i<n;i++)
	{
		for (int j=0;j<m;j++) 
		if (!c[j]&&abs(a[i]-b[j])<2) 
		{
			k++;
			c[j]=1;
			break;
		}
	}
	cout<<k;
	return 0;
}
