#include<iostream>
#include<algorithm>
#include<math.h>
using namespace std;
int main()
{
	int m,n;
	cin>>n;
	int b[n];
	for(int i=0;i<n;i++)
		cin>>b[i];
	cin>>m;
	int g[m];
	for(int i=0;i<m;i++)
		cin>>g[i];
	sort(b,b+n);
	sort(g,g+m);
	int count =0;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			if(fabs(b[i]-g[j])<=1)
			{
				g[j]=110;
				count++;
				break;
			}
	cout<<count;
	return 0;
}
