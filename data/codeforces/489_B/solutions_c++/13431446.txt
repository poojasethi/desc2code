#include<bits/stdc++.h>
using namespace std;
int a[104],b[104],i,j,n,m,s;
int main()
{
	for(cin>>n;i<n;i++)
		cin>>a[i];
	for(cin>>m;j<m;j++)
		cin>>b[j];
	sort(a,a+n);
	sort(b,b+m);
	i=j=0;
	while(i<n and j<m)
	{
		if(abs(a[i]-b[j])<2){s++;j++;i++;}
		else if(a[i]>b[j])j++;
		else i++;
		//cout<<a[i]<<' '<<b[i]<<"****   "<<i<<' '<<j<<endl;
	}
	cout<<s<<endl;
}
