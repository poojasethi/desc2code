#include <iostream>
#include <algorithm>

int a[500], b[500], n, m;

using namespace std;

int pos, r;

int main(){
	cin >> n;
	for(int i = 0;i < n;++i)cin >> a[i];
	cin >> m;
	for(int i = 0;i < m;++i)cin >> b[i];
	sort(a, a + n);
	sort(b, b + m);

	for(int i = 0;i < n;++i){
		for(int j = pos;j < m;++j){
			if(abs(a[i] - b[j]) <= 1){
				++r;
				pos = j + 1;
				break;
			}
		}
	}
	cout << r << "\n";
}
