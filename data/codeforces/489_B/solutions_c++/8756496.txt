#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
const int md=1000000007;
int a[1111],b[1110];
int i,t,j,n,m,ans;

int main() {

cin>>n;
for (i=0;i<n;i++) cin>>a[i];
sort(a,a+n);
cin>>m;
for (i=0;i<m;i++) cin>>b[i];
sort(b,b+m);
for (i=j=0;i<n && j<m;)
{
  if (abs(a[i]-b[j])<2) {i++;j++;ans++;} else
   if (a[i]<b[j]) i++; else j++;
}

 cout<<ans;
}
