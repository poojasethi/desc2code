#include<iostream>
#include<algorithm>

using namespace std;
int main()
{
	int i,c=0,j,n,m,k;
	cin >> n;
	int b[n];
	for(i=0;i<n;i++)
		cin >> b[i];
	cin >> m;
	int g[m];
	for(i=0;i<m;i++)
		cin >> g[i];
	sort(b,b+n);
	sort(g,g+m);
	i=0;j=0;
	while(i<n&&j<m)
	{
		k=b[i]-g[j];
		if(k==0||k==1||k==-1)
		{
			i++;
			j++;
			c++;
		}
		else if(b[i]>g[j])
			j++;
		else
			i++;
	}
	cout << c <<endl;
	return 0;
}
