#include <bits/stdc++.h>
#define rep(i,s,n) for(int i=(int)s;i<=(int)n;i++)

using namespace std;

int n,m,b[102],g[102];

int main() 
{	
	scanf("%d",&n);
	rep(i,1,n)scanf("%d",&b[i]);
	scanf("%d",&m);
	rep(i,1,m)scanf("%d",&g[i]);
	sort(b+1,b+1+n); sort(g+1,g+1+m);
	int ans=0;
	rep(i,1,n)rep(j,1,m)if(abs(b[i]-g[j])<=1){++ans;g[j]=1000;break;}
	printf("%d",ans);
	return 0;
}
