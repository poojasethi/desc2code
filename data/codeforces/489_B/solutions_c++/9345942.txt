#include <bits/stdc++.h>
using namespace std;
int n,m,s,a[105],b[105];
int main()
{
	int i,j;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	scanf("%d",&m);
	for(int i=0;i<m;i++)
	{
		scanf("%d",&b[i]);
	}
	sort(a,a+n);
	sort(b,b+m);
	for(i=0,j=0;i<n&&j<m;)
	{
		if(abs(a[i]-b[j])<=1)
		{
			s++,i++,j++;
		}
		else if(a[i]<b[j])
		{
			i++;
		}
		else
		{
			j++;
		}
	}
	printf("%d\n",s);
}
