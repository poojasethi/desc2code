#include <bits/stdc++.h>
using namespace std;
const int maxn=100 + 10;

int N,M;
int a[maxn],b[maxn];



int main() 
{
  cin>>N;
  for(int i=0; i<N; i++)cin>>a[i];
  cin>>M;
  for(int i=0; i<M; i++)cin>>b[i];
  sort( a , a+N);
  sort( b , b+M);
  int ans=0;
  for(int i=0; i<N; i++)
    {
      for(int j=0; j<M; j++)
	{
	  if( fabs(a[i] - b[j] ) < 2 ){ans++ , b[j] = 102; break;}
	}
    }
  cout<<ans<<endl;
}
