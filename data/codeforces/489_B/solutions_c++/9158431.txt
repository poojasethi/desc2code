#include <bits/stdc++.h>
using namespace std;
int main ()
{
int long long a,c,b[100001],g[100001],i,f=0;
cin>>a;
for (i=0; i<a; i++)
{
cin>>b[i];
}
cin>>c;
for (i=0; i<c; i++)
{
cin>>g[i];
}
sort(b, b+a);
sort(g, g+c);

for (int i = 0; i < a; i++)
    for (int j = 0; j < c; j++)
        if (abs(b[i] - g[j]) <= 1)
        {
            g[j] = 1000;
            f++;
            break;
        }
cout<<f<<endl;
}
