#include<bits/stdc++.h>
using namespace std;
main()
{
	int a,b,c=0,d=0;
	vector<int>i(100),j(100);
	cin>>a; for(int x=0;x<a;x++) cin>>i[x];
	cin>>b; for(int x=0;x<b;x++) cin>>j[x];
	sort(i.begin(),i.begin()+a); sort(j.begin(),j.begin()+b);
	for(int x=0;x<a;++x) for(int y=c;y<b;++y) if(abs(i[x]-j[y])==1||abs(i[x]-j[y])==0) { d++; c=y+1; break;
	}
	cout<<d;
}
