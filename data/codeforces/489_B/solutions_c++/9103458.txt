#include <algorithm>
using namespace std;
int a[100],n,i,k,o,l,r,op,m,c,b[100],j;
	
int abs(int vol)
{
return (vol<0 ? -vol : vol);
}
int main(){ 
cin>>n;
for(i=0;i<n;i++){
cin>>a[i];}
sort(a,a+n);

for(j=0;j<k;j++){

}
sort(b,b+k);
for(i=0;i<n;i++){

break;} }}

}
