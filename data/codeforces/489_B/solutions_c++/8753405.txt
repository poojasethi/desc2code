#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
int a[105],b[105];
int main()
{int n,m;cin>>n;
int i0=0,j0=0,k=0;
for(int i=0;i<n;i++)
    cin>>a[i];
cin>>m;
for(int i=0;i<m;i++)
cin>>b[i];
sort(a,a+n);
sort(b,b+m);
while(i0!=n&&j0!=m)
{if(abs(a[i0]-b[j0])<=1){k++;i0++;j0++;}
 else if(a[i0]<b[j0])i0++;
   else j0++;}
cout<<k;
    return 0;
}
