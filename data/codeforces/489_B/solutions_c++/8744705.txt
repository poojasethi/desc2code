#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,m,a[1000],b[1000],dem=0;
cin>>n;
for(int i=0;i<n;i++)
cin>>a[i];
cin>>m;
for(int i=0;i<m;i++)
cin>>b[i];
sort(a,a+n);
sort(b,b+m);
for(int i=0,j=0;i<n && j<m;){
    if(abs(a[i]-b[j])<=1)
    dem++,i++,j++;
    else if(a[i]<b[j])
    i++;
    else j++;
}
cout<<dem;
}
