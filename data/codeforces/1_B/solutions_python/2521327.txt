import re

for i in range(input()):
	s = raw_input()
	if (re.match('R[0-9]+C[0-9]+', s)):
		c = int(s[s.index('C') + 1 : ])
		l = 0
		while c >= 26 ** l:
			c -= 26 ** l
			l += 1
		print ''.join([chr(ord('A') + c / 26 ** i % 26) for i in range(l)][ : : -1]) + s[1 : s.index('C')]
	else:
		t = filter(lambda x : x.isupper(), s)
		l = len(t)
		print 'R{0}C{1}'.format(filter(lambda x : x.isdigit(), s), (26 ** l - 1) / 25 + sum((ord(t[i]) - ord('A')) * 26 ** (l - i - 1) for i in range(l)))
