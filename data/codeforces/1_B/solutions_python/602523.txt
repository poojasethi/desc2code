import re

for i in range(input()):
	text = raw_input()
	match = re.search(r'R(\d+)C(\d+)', text)
	if match:
		x = int(match.group(2))
		res = ''
		while x:
			res = chr(ord('A') + (x - 1) % 26) + res
			x = (x - 1) / 26
		print(res + match.group(1))
	else:
		t = re.search(r'(\D+)(\d+)', text)
		x = 0
		for letter in t.group(1):
			x = x * 26 + (ord(letter) - ord('A') + 1)
		print('R%dC%d' % (int(t.group(2)), x))
