import re

def RCtoAZ(x, y):
    while y:
        x = chr(ord('A') + (y-1)%26) + x
        y = (y-1)/26
    return x

def AZtoRC(st, i):
    x = 'R' + i + 'C'
    n = 0
    for s in st:
        n = n*26 + ord(s) - ord('A') + 1
    x += str(n)
    return x

m = input()
while m:
    m -= 1
    s = raw_input()
    p = re.match(r'R(\d+)C(\d+)', s)
    q = re.match(r'(\D+)(\d+)', s)
    if p: print RCtoAZ(p.group(1), int(p.group(2)))
    elif q: print AZtoRC(q.group(1), q.group(2))
