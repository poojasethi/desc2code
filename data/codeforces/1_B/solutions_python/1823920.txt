import re
def turn(str):
    m = re.match(r'R(\d+)C(\d+)',str)
    if m:
        x = int (m.group(2))
        res = ''
        while x:
            res = chr(ord('A')+(x-1)%26)+res
            x=(x-1)/26
        print res+m.group(1)
    else :
        t = re.match(r'(\D+)(\d+)',str)
        x = 0
        for s in t.group(1):
            x = ord(s)-ord('A')+1+x*26
        print 'R%dC%d'%(int(t.group(2)),x)
n = input()
for i in range(0,n):
    str = raw_input()
    turn(str)
