n = int(raw_input())

def correct(s):
    if s[1].isdigit() and s.find('C',2)>=0:
        a, b = map(int,s[1:].split('C'))
        h=''
        while b:
            h=chr((b-1)%26 + ord('A'))+h
            b=b/26-(b%26==0)
        print h+str(a)
    else:
        c, t = 0,''
        for i in s:
            if i.isalpha():
                c=c*26+ord(i)-ord('A')+1
            else :
                t+=i
        print 'R'+t+'C'+str(c)

for i in range(n):
    s=raw_input()
    correct(s)
