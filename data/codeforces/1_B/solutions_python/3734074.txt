n=input()
S=map(raw_input,['']*n)

for s in S:
    ans=''
    if s[0]=='R' and s[1].isdigit() and 'C' in s:
        i=s.index('C')
        m=int(s[i+1:])
        while m:
            ans=chr(ord('A')+(m-1)%26)+ans;
            m=(m-1)/26
        ans+=s[1:i]
    else:
        i=m=0
        while s[i].isupper():
            m=m*26+(ord(s[i])-ord('A')+1);
            i+=1
        ans='R'+s[i:]+'C'+str(m)
    print ans
