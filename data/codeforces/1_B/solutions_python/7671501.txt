for i in range(input()):
	s=raw_input()
	r,c=s.find('R'),s.find('C')
	def f(x,s):return s if x == 0 else f((x-1)/26,chr((x-1)%26+65)+s)
	def g(x,s):return 'R'+s+'C'+str(x) if s[0].isdigit() else g(x*26+ord(s[0])-64, s[1:])
	print f(int(s[c+1:]),'')+s[1:c] if r==0 and s[1].isdigit() and c>r else g(0,s)
