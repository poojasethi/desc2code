import re

n=input()
for i in range(n):
	s = raw_input()
	p = re.match('R(\d+)C(\d+)', s)
	if p:
		res = p.group(1)
		c = int(p.group(2))
		while c:
			res = chr(ord('A') + (c - 1) % 26) + res
			c = (c - 1) / 26
	else:
		p = re.match('(\D+)(\d+)', s)
		res = 'R' + p.group(2) + 'C'
		t = 0
		for x in p.group(1):
			t = t * 26 + ord(x) - ord('A') + 1
		res += str(t)
	print res
