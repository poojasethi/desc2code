for _ in range(input()):
    I=raw_input()
    if I[0]=='R' and I[1].isdigit() and 'C' in I[2:]:
        r,c=map(int,I[1:].split('C'))
        s=''
        while c>0:
            c,p=divmod(c-1,26)
            s=chr(p+ord('A'))+s
        print s+str(r)
    else:
        r=c=0
        for i,a in enumerate(I):
            if a.isdigit():
                r=I[i:]
                break
            c=c*26+ord(a)-ord('A')+1
        print 'R'+r+'C'+str(c)
