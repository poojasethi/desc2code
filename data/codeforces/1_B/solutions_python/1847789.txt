import re
n=input()
for _ in xrange(n):
  s=raw_input()
  m=re.match('^R(\d+)C(\d+)$', s)
  if m is None:
    m=re.match('^([A-Z]+)(\d+)$', s)
    p=1
    n=0
    for c in reversed(m.group(1)):
      n+=(ord(c)-ord('A')+1)*p
      p*=26
    print 'R'+m.group(2)+'C'+str(n)
  else:
    p=26
    n=''
    t=int(m.group(2))
    while t>0:
      r=t%p
      if r==0:r=p
      t-=r
      r=r/(p/26)
      n=chr(r+ord('A')-1)+n
      p*=26
    print n+m.group(1)
