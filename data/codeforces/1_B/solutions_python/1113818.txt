for t in range(input()):
    s = raw_input()
    for i in range(len(s)):
        if s[i].isdigit():
            a, b = s[:i], s[i:]
            break
    if b.isdigit():
        n = 0
        for i in a:
            n = n * 26 + int(ord(i) - 64)
        print 'R' + b + 'C' + str(n)
    else:
        c, b = b.split('C')
        a = ''
        b = int(b)
        while b:
            a = chr(65 + (b - 1) % 26) + a
            b = int((b - 1) / 26)
        print a + c
