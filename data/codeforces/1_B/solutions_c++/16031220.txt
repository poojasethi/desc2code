#include<cstdio>
void functions(int y){
    if(y){
        functions((y-1)/26);
        putchar(65+(y-1)%26);
    }
}
int main(){
	int n,num,y;
	char s[64],*p;
	for(scanf("%d ",&n);n--;){
		gets(s);
		if(sscanf(s,"%*c%d%*c%d",&num,&y)==2){
			functions(y);
			printf("%d\n",num);
		}else{
			for(num=0,p=s;*p>64;++p)
				num=num*26+*p-64;
			printf("R%sC%d\n",p,num);
		}
	}
	return 0;
}
