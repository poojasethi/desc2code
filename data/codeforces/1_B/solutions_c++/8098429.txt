#include<cstdio>
 
 void print(int t) { if(t--) { print(t/26); putchar(65+t%26); } }
 
 int main()
 {
 char s[10000],*p;
 int n,x;
 scanf("%d",&n);
 for(int i=0;i<n;i++)
 {
 scanf("%s",s);
 int r=0,c=0;
 if(sscanf(s,"R%dC%d",&r,&c)==2)
 {
 print(c);
 printf("%d\n",r);
 }
 else
 {
 for(x=0,p=s;*p>64;++p)
 {
 x=x*26+*p-64;
 }
 printf("R%sC%d\n",p,x);
 }
 }
 
 return 0; 
 }
