def ret(str):
  print str
  exit(0)

vv = [int(x) for x in raw_input().split()]

N = vv[0]
W = vv[1]
jump = vv[2]

def reserve(size, j):
  if size % j:
    return size % j
  else:
    return j

canN = reserve(N, jump)
canW = reserve(W, jump)

stripes = (W - 1) // jump
cols = (N - 1) // jump

ret((stripes + 1)* (cols + 1) * canN * canW)
