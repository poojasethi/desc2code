#include <cstdio>
const int N=5005,mo=1e9+7;
int n;
char s[N];
int next[N],lcp[N][N],f[N][N],sf[N][N],g[N][N],sg[N][N];
int main(){
	scanf("%s",s);
	for(n=0;s[n];n++);
	for(int i=0;i<n;i++) s[i]-='0';
	s[n]=1;
	for(int i=n-1;i>=0;i--) next[i]=s[i]?i:next[i+1];
	for(int i=n-1;i>=0;i--)
		for(int j=i;j<n;j++)
			lcp[i][j]=s[i]==s[j]?lcp[i+1][j+1]+1:0;
	f[0][0]=sf[0][0]=1;
	for(int i=1;i<=n;i++) if(s[i]){
		for(int j=0;j<i;j++) if(s[j]){
			int k=j*2-i;
			if(k<0) k=0; else
				if(!s[k]||lcp[k][j]<i-j&&s[k+lcp[k][j]]>s[j+lcp[k][j]])
					k=next[k+1];
			f[i][j]=sf[j][k];
			g[i][j]=sg[j][k]+1;
		}
		else g[i][j]=mo;
		sg[i][i]=mo;
		for(int j=i;j;j--){
			if((sf[i][j-1]=sf[i][j]+f[i][j-1])>=mo) sf[i][j-1]-=mo;
			sg[i][j-1]=sg[i][j]<g[i][j-1]?sg[i][j]:g[i][j-1];
		}
	}
	int ans=mo;
	for(int t=n-1,now=0,w=1;t>=0;t--,w=(w<<1)%mo) if(s[t]){
		(now+=w)%=mo;
		if(g[n][t]<mo){
			if(n-t>20){
				if(ans==mo)
					ans=(g[n][t]+now)%mo;
				break;
			}
			else
			if(g[n][t]+now<ans) ans=g[n][t]+now;
		}
	}
	printf("%d\n%d\n",sf[n][0],ans);
}
