#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
#include <cstring>
using namespace std;
class bor{
  public:
  int q;
  bor *a[2];
  bor(){
   a[0]=a[1]=0;}
};
const int n=5001;
char a[n];
int q,*s[n],*d[n],f[n][n],i=0,p=1000000007;
void ini(bor *c){
  if(!c)
   return;
  c->q=i;
  i++;
  ini(c->a[0]);
  ini(c->a[1]);
}
void del(bor *c){
  if(!c)
   return;
  del(c->a[0]);
  del(c->a[1]);
  delete(c);
}
void upd(int e,int r,int t,int y){
  if(e<=q){
   s[e][r]+=t;
   if(s[e][r]>=p)
    s[e][r]-=p;
   d[e][r]=min(d[e][r],y);}
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int w,e,r,t,ans;
  bor *c=new bor(),*v,*v1;
  scanf("%s",a);
  q=strlen(a);
  for(w=0;w<q;w++)
   a[w]-=48;
  for(w=0;w<q;w++){
   v=c;
   for(e=w;e<q;e++){
    if(!v->a[a[e]])
     v->a[a[e]]=new bor();
    v=v->a[a[e]];}}
  ini(c);
  for(w=0;w<q;w++){
   v=c;
   for(e=w;e<q;e++){
    v=v->a[a[e]];
    f[w][e]=v->q;}}
  del(c);
  for(w=0;w<=q;w++){
   s[w]=new int[w+1];
   d[w]=new int[w+1];
   for(e=0;e<=w;e++){
    s[w][e]=0;
    d[w][e]=1<<30;}}
  upd(1,0,1,1);
  for(w=0;w<q;w++)
   for(e=0;e<w;e++){
    upd(w+1,e,s[w][e],d[w][e]);
    if((w+w-e<=q)&&(a[w]==1))
     if(f[e][w-1]<=f[w][w+w-e-1])
      upd(w+w-e,w,s[w][e],d[w][e]+1);
     else
      upd(w+w-e+1,w,s[w][e],d[w][e]+1);}
  ans=0;
  for(e=0;e<q;e++){
   ans+=s[q][e];
   if(ans>=p)
    ans-=p;}
  cout<<ans<<"\n";
  ans=1<<30;
  r=0;
  t=1;
  for(w=0;(w<20)&&(w<q);w++){
   r=(r+t*a[q-w-1])%p;
   t=(t*2)%p;
   if(a[q-w-1]==0)
    continue;
   ans=min(ans,r+d[q][q-w-1]);}
  if(ans<1<<30){
   cout<<ans;
   return 0;}
  for(;w<q;w++){
   r=(r+t*a[q-w-1])%p;
   t=(t*2)%p;
   if((a[q-w-1]==0)||(d[q][q-w-1]==1<<30))
    continue;
   ans=(r+d[q][q-w-1])%p;
   break;}
  cout<<ans;
  return 0;}
