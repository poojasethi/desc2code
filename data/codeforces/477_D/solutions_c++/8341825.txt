#include<bits/stdc++.h>
#define rep(i,j,k)	for(int i=j;i<=k;i++)
#define dep(i,j,k)	for(int i=j;i>=k;i--)
using namespace std;
typedef long long ll;
const int N=5010,mod=1e9+7;
char s[N];
int l,nxt[N],f[N][N],g[N][N],tf[N][N],tg[N][N],lcp[N][N];
int main()
{
scanf("%s",s);l=strlen(s);rep(i,0,l-1)s[i]-='0';s[l]=1;
dep(i,l-1,0)nxt[i]=s[i]?i:nxt[i+1];dep(i,l-1,0)rep(j,i,l-1)lcp[i][j]=(s[i]==s[j])?lcp[i+1][j+1]+1:0;
f[0][0]=tf[0][0]=1;rep(i,1,l)if(s[i]){rep(j,0,i-1)if(s[j]){int k=(j<<1)-i;
if(k<0)k=0;else if(!s[k]||lcp[k][j]<i-j&&s[k+lcp[k][j]]>s[j+lcp[k][j]])k=nxt[k+1];
f[i][j]=tf[j][k],g[i][j]=tg[j][k]+1;}else g[i][j]=mod;tg[i][i]=mod;dep(j,i,1)
{if((tf[i][j-1]=tf[i][j]+f[i][j-1])>=mod)tf[i][j-1]-=mod;
tg[i][j-1]=(tg[i][j]<g[i][j-1])?tg[i][j]:g[i][j-1];}}int ans=mod;
for(int t=l-1,now=0,w=1;t+1;t--,w=(w<<1)%mod)if(s[t]){now+=w,now%=mod;
if(g[l][t]<mod){if(l-t>20){if(ans==mod)ans=(g[l][t]+now)%mod;break;}
else if(g[l][t]+now<ans)ans=g[l][t]+now;}}cout<<tf[l][0]<<endl<<ans<<endl;
}
