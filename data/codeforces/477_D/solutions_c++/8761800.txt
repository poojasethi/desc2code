#include<stdio.h>
#include<string.h>
#include<algorithm>
#define mp make_pair
#define ft first
#define sd second
using namespace std;
typedef pair<int,int> pii;
const int maxl=5050,mod=1e9+7,big=0x3f3f3f3f;
char s[maxl];
int l,ans1=0,ans2=big;
pii qu[maxl];
int st,en;
bool mat[maxl][maxl];
int num[maxl];
int f1[maxl][maxl],f2[maxl][maxl];
int Add(int x,int y) { return (x+=y)>=mod?x-=mod:x; }
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
	freopen("data.out","w",stdout);
#endif
	int i,j,k;
	scanf("%s",s+1);
	l=strlen(s+1);
	for(i=l,j=1;i;i--,j=Add(j,j))
		num[i]=Add(num[i+1],(s[i]-'0')*j);
	for(i=1;i<=l;i++)
		mat[i][l]=1;
	for(i=1;i+i<=l;i++)
	{
		st=en=0;
		for(j=1;j<i;j++)
			if(s[j]!=s[j+i])
				qu[en++]=mp(j,s[j+i]>s[j]);
		for(j=i;j+i<=l;j++)
		{
			if(s[j]!=s[j+i])
				qu[en++]=mp(j,s[j+i]>s[j]);
			while(st<en&&qu[st].ft<=j-i) st++;
			mat[j-i][j]=(st==en||qu[st].ft>j||qu[st].sd);
		}
	}
	//for(i=0;i<=l;i++) { for(j=0;j<l;j++) printf("%d ",mat[i][j]); printf("\n"); }
	memset(f2,big,sizeof(f2));
	f1[0][1]=1,f2[0][1]=1;
	for(i=0;i<l;i++)
		if(s[i+1]!='0')
			for(j=i+1;j<l;j++)
			{
				f1[i][j+1]=Add(f1[i][j+1],f1[i][j]);
				f2[i][j+1]=min(f2[i][j+1],f2[i][j]);
				if(s[j+1]!='0')
				{
					k=j*2-i+!mat[i][j];
					f1[j][k]=Add(f1[j][k],f1[i][j]);
					f2[j][k]=min(f2[j][k],f2[i][j]+1);
				}
			}
	//for(i=0;i<l;i++) { for(j=0;j<=l;j++) printf("%d ",f1[i][j]); printf("\n"); }
	//for(i=0;i<l;i++) { for(j=0;j<=l;j++) printf("%d ",f2[i][j]); printf("\n"); }
	for(i=0;i<l;i++)
		if(s[i+1]!='0')
			ans1=Add(ans1,f1[i][l]);
	for(i=max(0,l-15);i<l;i++)
		if(f2[i][l]!=big)
			ans2=min(ans2,Add(f2[i][l],num[i+1]));
	if(ans2==big)
		for(i=l-1;i>=0;i--)
			if(f2[i][l]!=big)
			{ ans2=Add(f2[i][l],num[i+1]);break; }
	printf("%d\n%d\n",ans1,ans2);
	return 0;
}
