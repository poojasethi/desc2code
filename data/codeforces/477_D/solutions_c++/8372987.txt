#include <cstdio>
#include <cstdlib>
#include <cstring>
#define P 1000000007
using namespace std;

int n,lcp[5151][5151];
long long f[5151][5151],ans1,ans2;
int g[5151][5151],anst;
char s[5151];

bool cmp(int a,int b)
{
    // return a-b<0
    if (!b) return 1;
    long long cha=0;
    for (int i=a;i<b;i++) 
    {
        cha=(cha*2+(s[i]=='1'));
        if (cha>=g[b][n]-g[a][n]) return 0;
    }
    for (int i=b;i<=n;i++) 
    {
        cha=(cha*2);
        if (cha>=g[b][n]-g[a][n]) return 0;
    }
    return 1;
}

int main()
{
    scanf("%s",s+1);
    n=strlen(s+1);
    for (int i=n;i;i--)
        for (int j=n;j;j--)
            lcp[i][j]=s[i]==s[j]?lcp[i+1][j+1]+1:0;
    /*
        f[i][j]-> f[j+1][k]
        k=j-i+1+j  ||  j-i+1+j+1
    */
    f[1][1]=1;
    for (int i=1;i<=n;i++)
        for (int j=i;j<=n;j++) g[i][j]=n+n;
    g[1][1]=1;
    for (int j=1;j<=n;j++)
    {
        if (s[j+1]=='1')
        {
            for (int i=1;i<=j;i++)
            {
                int k;
                int tmp=lcp[i][j+1];
                if (tmp>=j-i+1||s[i+tmp]<s[j+1+tmp]) k=j-i+1+j;
                else k=j-i+1+j+1;
                if (k>n) continue;
                f[j+1][k]=(f[j+1][k]+f[i][j])%P;
                if (g[i][j]+1<g[j+1][k]) g[j+1][k]=g[i][j]+1;
            }
        }
        for (int i=1;i<=j;i++) 
        {
            f[i][j+1]=(f[i][j+1]+f[i][j])%P;
            if (g[i][j]<g[i][j+1]) g[i][j+1]=g[i][j];
        }
    }
    for (int i=1;i<=n;i++) ans1=(ans1+f[i][n])%P;
    printf("%I64d\n",ans1);
    for (int i=n;i;i--) 
        if (g[i][n]!=n+n) if (cmp(i,anst)) anst=i;
    for (int i=anst;i<=n;i++)
        ans2=(ans2*2+(s[i]=='1'))%P;
    ans2=(ans2+g[anst][n])%P;
    printf("%I64d\n",ans2);
}
