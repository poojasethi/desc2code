#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#define N 1200
int a[N];
int main()
{
	int n,k,p,x;
	scanf("%d%d%d",&n,&k,&p);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	int num, ans = 0,l,r;
	for (int i=1;i<=n;i++)
	{
		x = p;
		if (a[i]!=x&&a[i+1]!=x) continue;
		num = 1;
		l = i; r = i + 1;
		int tmp = 0 ;
		while (l>0||r<=n)
		{
			while (l>0&&a[l]==x) {l--;num++;}
			while (r<=n&&a[r]==x) {r++;num++;}
			if (num<=2) break;
			else 
			{
				tmp = tmp + num;
				num = 0;
				if (l>0) x = a[l]; else x = a[r];
			}
		}
		if (tmp-1>ans) ans = tmp-1;
	}
	printf("%d\n",ans);
	return 0;
}
