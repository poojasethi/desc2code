#include <cstdio>
#include <algorithm>
using namespace std;
int n,k,x,C[200],mx;
int fd(int a,int b,int v){
    int ac = 0,bc = 0,ca = C[a-1],cb = C[b+1];
    if(ca == cb){
        while(C[a-1] == ca)
            ac++,a--;
        while(C[b+1] == cb)
            bc++,b++;
        if((ac+bc) > 2)
            return fd(a,b,v+ac+bc);
    }
    return v;
}
int main()
{
    scanf("%d %d %d",&n,&k,&x);
    for(int i = 1;i <= n;i++)
        scanf("%d",&C[i]);
    C[0] = k+1,C[n+1] = k+2;
    for(int i = 2;i <= n;i++){
        if(C[i] == x && C[i-1] == x){
            mx = max(mx,fd(i-1,i,2));
        }
    }
    printf("%d\n",mx);
    return 0;
}
