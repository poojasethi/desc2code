#include <cstdio>
#include <algorithm>
using namespace std;
int a[105],b[105],c,d,e,f,g,h,i,j,k,l,m,n;
int main(){
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=1;i<=n;i++)if(a[i]==a[i-1])b[l]++;
	else{a[++l]=a[i];b[l]=1;}
	n=l;
	for(i=1;i<=n;i++)if(a[i]==k&&b[i]>=2){
		g=b[i];
		for(j=i-1,l=i+1;j&&l<=n&&a[j]==a[l]&&b[j]+b[l]>=3;j--,l++)g+=b[j]+b[l];
		h=max(h,g);
	}
	printf("%d",h);
	return 0;
}
