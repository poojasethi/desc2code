#include<stdio.h>
#include<stdlib.h>

int n;
int a[120];
int num,sum;
//int num,sum;
void find(int l,int r)
{
	if(l<1||r>n)
	{
		num=r-l-1;
		return;
	}
	if(a[l]==a[r])
	{
	    sum=2;
		if(a[l-1]==a[l])
		    l=l-1,sum++;
	    if(a[r]==a[r+1])
		    r=r+1,sum++;
		if(sum>=3)
			find(l-1,r+1);
		else
		{
			num=r-l-1;
			return;
		}
	}
	else
	{
		num=r-l-1;
		return;
	}
}
int main()
{
	int max=0;
	int i,j;
	int k,x;
	scanf("%d%d%d",&n,&k,&x);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(i=1;i<=n;i++)
	{
		if(a[i]==x&&x==a[i+1])
		{
			find(i-1,i+1+1);
			if(num>max)
				max=num;
		}
	}
	printf("%d\n",max);
}
