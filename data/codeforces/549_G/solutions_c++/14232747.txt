#include<algorithm>
#include<iostream>
using namespace std;

int main (void){
	int n;
	cin>>n;
	int line[n];
	for(int i=0;i<n;i++){
		int temp;
		cin>>temp;
		line[i]=temp+i;
	}
	sort(line,line+n);
	for(int i=0;i<n-1;i++){
		if(line[i]==line[i+1]){
			cout<<":(";
			return 0;
		}
	}
	for(int i=0;i<n;i++){
		cout<<line[i]-i<<" ";
	}	
}
