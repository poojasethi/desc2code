#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n;
	int a[200001];
	cin>>n;
	for(int i=0;i<n;i++)
	{
        cin>>a[i];
        a[i]+=i;
    }
    sort(a,a+n);
    for(int i=1;i<n;i++)
    {
    	a[i]-=i;
    	if(a[i]<a[i-1])
    	{
    		cout<<":(";
    		return 0;
    	}
    }
	for(int i=0;i<n;i++)
	{
		cout<<a[i]<<" ";
	}
	return 0;
}
