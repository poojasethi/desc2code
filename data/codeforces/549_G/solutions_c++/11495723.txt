#include<stdio.h>
#include<string.h>
#include<algorithm>
#define M 222222
using namespace std;
int a[M];
int n;
int main(){
	scanf("%d",&n);
	for(int i = 1 ; i <= n ; i++){
		scanf("%d",&a[i]);
		a[i]+=i;
	}
	sort(a + 1 , a + 1 + n);
	for(int i = 1 ; i < n ; i++)	if(a[i] == a[i + 1]){
		printf(":(");
		return 0;
	}
	for(int i = 1 ; i <= n ; i++)	printf("%d ",a[i] - i);
}
