#include <stdio.h>
#include <algorithm>

int n;
int a[200005];

int main()
{
	scanf("%d",&n);
	int i;
	for (i=1;i<=n;i++) {
		scanf("%d",a+i);
		a[i]+=i-1;
	}
	std::sort(a+1,a+n+1);
	for (i=1;i<=n;i++) {
		a[i]-=i-1;
	}
	bool f=1;
	for (i=2;i<=n;i++) {
		if (a[i]<a[i-1]) f=0;
	}
	if (!f) {
		puts(":(");
	} else {
		for (i=1;i<=n;i++) {
			printf("%d ",a[i]);
		}
	}
}
