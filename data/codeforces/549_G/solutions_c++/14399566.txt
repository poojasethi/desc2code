#include <cstdio>
#include <algorithm>
using namespace std;
int n, a[200005], flag;
int main() {
    scanf("%d", &n);
    for (int i=0; i<n; i++) {scanf("%d", &a[i]); a[i]+=i;}
    sort(a, a+n);
    for (int i=0; i<n-1; i++) if (a[i]==a[i+1]) flag=1;
    if (flag) puts(":(");
    else {
        for (int i=0; i<n-1; i++) printf("%d ", a[i]-i);
        printf("%d\n", a[n-1]-n+1);
    }
    return 0;
}
