#include<iostream>
#include<algorithm>
using namespace std;
#define f(i,a,b) for(int i=a;i<b;i++)
int main(){
	int n,m[200000];
	cin>>n;
	f(i,0,n){
		cin>>m[i];
		m[i] += i;
	}
	sort(m,m+n);
	f(i,1,n)if(m[i]==m[i-1]){
		cout<<":("<<endl;
		return 0;
	}
	f(i,0,n)cout<<m[i]-i<<" ";
	cout<<endl;
	return 0;
}
