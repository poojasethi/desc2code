#include <iostream>
#include<algorithm>
using namespace std;
int main() {
	int i, j, n;
	cin >> n;
	int a[n];
	for (i = 0; i < n; i++) {
		cin >> a[i];
		a[i] += i;
	}
	std::sort(a, a + n);
	for (i = 0; i < n - 1; i++) {
		if (a[i] == a[i + 1]) {
			cout << ":(";
			return 0;
		}
	}
	for (i = 0; i < n; i++) {
		cout << a[i] - i<<" ";
	}
}
