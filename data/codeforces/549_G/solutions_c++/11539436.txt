#include<bits/stdc++.h>

using namespace std;

int n,s,a[400000];

int main(){
    cin>>n;
    for (int i=1;i<=n;++i){
        cin>>s;
        a[i]=s+i;
    }
    sort (a+1,a+n+1);
    for (int i=1;i<n;++i){
        if (a[i]==a[i+1])   return cout<<":(", 0;
    }
    for (int i=1;i<=n;++i){
        cout<<a[i]-i<<" ";
    }
    cout<<endl;
    return 0;
}
