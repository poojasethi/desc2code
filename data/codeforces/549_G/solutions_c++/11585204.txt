#include <bits/stdc++.h>
using namespace std;

const int nmax = 2e5 + 10;

int n, a[nmax], prev;

int main() {
	scanf("%d", &n);
	for(int i = 0; i<n; i++)
		scanf("%d", a+i), a[i] -= n-i;
	sort(a, a+n);
	prev = 0;
	for(int i = 0; i<n; i++) {
		a[i] += (n-i);
		if(prev > a[i]) {
			printf(":(\n");
			return 0;
		}
		prev = a[i];
	}
	for(int i = 0; i<n; i++)
		printf("%d ", a[i]);
	return 0;
}
