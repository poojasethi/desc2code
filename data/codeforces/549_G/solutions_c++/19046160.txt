#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int n,i,a[200000],d,s;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&d);
		a[i]=i+d;
	}
	sort(a,a+n);
	s=1;
	for(i=1;i<n;i++)
		if(a[i-1]==a[i]) s=0;
    if(s)
        for(i=0;i<n;i++) printf("%d ",a[i]-i);
    else printf(":(");    
	return 0;
}
