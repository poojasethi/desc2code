#include<stdio.h>
#include<algorithm>

using namespace std;
int n;
int h[200200];

int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++) {scanf("%d",&h[i]); h[i]-= (n-1-i);}
    sort(&h[0],&h[n]);
    for(int i=0;i<n-1;i++) if(h[i]>=h[i+1]) {printf(":("); return 0;}
    for(int i=0;i<n;i++) printf("%d ",h[i]+(n-1-i));
    return 0;
}
