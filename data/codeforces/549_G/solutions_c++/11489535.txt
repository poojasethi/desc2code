#include <bits/stdc++.h>
using namespace std;
int a[200010];
int main()
{
	int n;
	cin>>n;
	for(int i=0; i<n; i++)
	{
		scanf("%d", &a[i]);
		a[i] += i;
	}
	sort(a, a+n);
	bool good = true;
	for(int i=0; i<n-1; i++)
		if(a[i] == a[i+1])
			good = false;
	if(!good)
		cout<<":("<<endl;
	else
	{
		for(int i=0; i<n; i++)
			printf("%d ", a[i]-i);
	}
	return 0;
}
