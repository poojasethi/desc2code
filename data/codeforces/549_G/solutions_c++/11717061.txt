#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	long long a[200010];
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
		a[i]=a[i]+i;
	}
	sort(a,a+n);
	for(int i=0;i<n-1;i++)
		if(a[i]==a[i+1])
		{
			printf(":(\n");
			return 0;
		}
	for(int i=0;i<n;i++)
		printf("%lld ",a[i]-i);
	return 0;
}
