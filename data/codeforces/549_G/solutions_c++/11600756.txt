#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=200000+10;
int a[maxn];
int main()
{
	int n;
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		a[i]+=i;
	}
	sort(a+1,a+n+1);
	for (int i=1;i<=n-1;i++)
	if (a[i]==a[i+1])
	{
		printf(":(\n");
		return 0;
	}
	for (int i=1;i<=n;i++)
	  printf("%d ",a[i]-i);
	return 0;
}
