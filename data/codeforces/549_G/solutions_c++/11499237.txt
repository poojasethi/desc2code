#include <bits/stdc++.h>

using namespace std;

int N,ar[200200];

int main(){
	scanf("%d",&N);
	
	for(int i=1;i<=N;i++){
		scanf("%d",&ar[i]);
		ar[i]-=N-i;
	}
	
	sort(ar+1,ar+N+1);
	
	for(int i=1;i<N;i++)
		if(ar[i]==ar[i+1]){
			printf(":(\n");
			return 0;
		}
	
	for(int i=1;i<=N;i++)
		printf("%d ",ar[i]+N-i);
	puts("");	
	
	return 0;
}
