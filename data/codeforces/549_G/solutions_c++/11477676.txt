#include<stdio.h>
#include<iostream>
#include<algorithm>
int n,i,s,a[200001];
int main(){
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(i=0;i<n;i++)
		a[i]=a[i]-n+i+1;
	std::sort(a,a+n);
	if(a[n-1]<0){printf(":(");return 0;}
	for(i=0;i<n-1;i++)
		if(a[i+1]-a[i]==0){printf(":(");return 0;}
	for(i=0;i<n;i++)
		printf("%d ",a[i]+n-i-1);
}
