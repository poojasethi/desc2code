#include <iostream>
#include <algorithm>
using namespace std;

int ll  [200000+100];

int main()
{
    int n;
	cin>>n;
	for(int i=0;i<n;++i)
    {
		cin>>ll[i];
		ll[i]+=i;
	}
	sort(ll,ll+n);
	for(int i=1;i<n;++i)
	{
		if(ll[i]==ll[i-1])
		{
			cout<<":(";
			return 0;
		}
	}
	for(int i=0;i<n;++i)
		cout<<ll[i]-i<<" ";
    return 0;
}
