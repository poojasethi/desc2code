#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int peo[200001];
int n;
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		cin>>peo[i];
		peo[i]+=i;
	}
	sort(peo,peo+n);
	for(int i=0;i<n-1;i++){
		if(peo[i]==peo[i+1]){
			cout<<":("<<endl;
			return 0;
		}
	}
	for(int i=0;i<n;i++)
	cout<<peo[i]-i<<" ";
	cout<<endl;
	return 0; 
}
