#include <bits/stdc++.h>
using namespace std;

int N;
int p[200005];

int main() {
	cin >> N;
	for (int i = 0; i < N; i++) {
		cin >> p[i];
		p[i] += i;
	}
	sort(p, p+N);
	for (int i = 0; i < N; i++)
		p[i] -= i;
	for (int i = 1; i < N; i++) {
		if (p[i] < p[i-1]) {
			cout << ":(" << endl;
			return 0;
		}
	}
	
	for (int i = 0; i < N; i++)
		cout << p[i] << " ";
	cout << endl;
	return 0;
}
