#define N (1<<18)
#include <bits/stdc++.h>
using namespace std;

int n,a[N];
bool flag=1;

int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
		scanf("%d",a+i),a[i]+=i;
	sort(a+1,a+n+1);

	for(int i=2;i<=n;i++)
		if(a[i]==a[i-1])
		{
			flag=0;
			break;
		}
	if(!flag)
		puts(":(");
	else
	{
		for(int i=1;i<=n;i++)
			printf("%d ",a[i]-i);
	}
	return 0;
}
