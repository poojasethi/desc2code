#include <iostream>
#include <set>
#include <stdio.h>

using namespace std;

int main()
{
	set<int> S;
	int N;
	cin >> N;

	int x;
	for (int i=0;i<N;i++)
	{
		cin >> x;
		S.insert(x+i);
	}

	if (S.size() == N){
		int p = 0;
		for (set<int>::iterator it = S.begin();it != S.end();it++,p++)
			printf("%d ", *it - p);
		cout << endl;
	}
	else
		cout << ":(\n";
}
