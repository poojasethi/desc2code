{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	cin>>a[i],a[i]+=i;
	sort(a,a+n);
	for(int i=0;i<n;i++)
		a[i]-=i;
	for(int i=0;i<n-1;i++)
	{
		if(a[i]>a[i+1])
		{
			cout<<":(\n";
			return 0;
		}
	}
	for(int i=0;i<n;i++)
	cout<<a[i]<<" ";
}
