#include<iostream>
#include<algorithm>
using namespace std;
int x[200020];
int main()
{
	int n,i;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>x[i];
		x[i]+=i;
	}
	sort(x,x+n);
	for(i=1;i<n;i++)
	{
		if(x[i]==x[i-1])
		{
			cout<<":(";
		    return 0;
		}
	}
	for(i=0;i<n;i++)
	cout<<x[i]-i<<" ";
	return 0;
}
