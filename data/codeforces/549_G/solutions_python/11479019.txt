#!/usr/bin/python

import sys
import math
import string

def solve(n, nums):
	scores = set()
	for i in range(n):
		score = i+nums[i]
		if score in scores:
			return None
		scores.add(score)

	sortedScores = sorted(list(scores))
	return [sortedScores[i] - i for i in range(n)]

if __name__ == "__main__":
	n = int(sys.stdin.readline())
	nums = map(int, sys.stdin.readline().split())
	ans = solve(n, nums)
	print ' '.join([str(s) for s in ans]) if ans else ':('
