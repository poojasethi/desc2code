n = input()
store = map(int,raw_input().split())

dup = set()

pos = []

for index,value in enumerate(store):
    k = index + value
    if k in dup:
        print ":("
        break
    else:
        dup.update([k])
        pos.append(k)
else: # possible
    pos = sorted(pos)
    print ' '.join(str(pos[i]-i) for i in range(len(pos)))
