from sys import stdin
n = int(stdin.readline())
a = map(int, stdin.readline().split())

sub = range(n, 0, -1)
a = map(lambda x, y: x-y, a, sub)
a.sort()
a = map(lambda x, y: x+y, a, sub)

flag = 0
for i in xrange(n-1):
    if not a[i+1]>=a[i]>=0:
        flag = 1
if flag:
    print ":("
else:
    print " ".join(map(str, a))
