n=int(raw_input())
res=map(int,raw_input().split())

pos=[];
dup=set()
for i,v in enumerate(res):
    k=i+v
    if k not in dup:
        pos.append(k)
        dup.update([k])
    else:
        print ":("
        break
else:
    pos=sorted(pos)
    print " ".join(str(pos[i]-i) for i in range (len(pos)))
