#!/usr/bin/python

def solve(xs):
    if len(xs) == 1:
        print xs[0]
        return
    for i in range(len(xs)):
        xs[-1-i] -= i
    xs = sorted(xs)
    for i in range(len(xs)-1):
        if xs[i] == xs[i+1]:
            print ':('
            return
    for i in range(len(xs)):
        xs[-1-i] += i
    print ' '.join(map(str, xs))

if __name__ == '__main__':
    raw_input()
    xs = [int(x) for x in raw_input().split()]
    solve(xs)
