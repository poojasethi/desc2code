//Educational Codeforces Round 13
#include <cstdio>
#include <cstring>
#include <iostream>
#include <cctype>
#include <cmath>
using namespace std;
const int N = 20;
int n;
double p[N][N];
double dp[1 << N][N];

int main(){
    int i, j, k;
    scanf("%d", &n);
    for (i = 0; i < n; ++ i){
        for (j = 0; j < n; ++ j)
            scanf("%lf", &p[i][j]);
    }
    dp[1][0] = 1.0;
    for (i = 0; i < (1 << n); ++ i){
        for (j = 0; j < n; ++ j) if (i & (1 << j)) {
            for (k = 0; k < n; ++ k)
            if (j != k &&  (i & (1 << k)))
                dp[i][j] = max(dp[i][j], dp[i - (1 << k)][j] * p[j][k] + \
                    dp[i - (1 << j)][k] * p[k][j]);
        }
    }
    double res = 0;
    for (i = 0; i < n; ++  i)  res = max(res, dp[(1 << n) - 1][i]);
    printf("%lf\n", res);
    return 0;
}
