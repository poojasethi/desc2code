#include <bits/stdc++.h>

using namespace std;
#define MAX 20
double dp[1<<MAX][MAX];
double rate[MAX][MAX];
int main()
{
    int n;
    cin >> n;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            cin >> rate[i][j];
    memset(dp,0,sizeof(dp));
    dp[1][0] = 1;
    for (int i = 0; i < (1 << n); i++)
    {
        for (int j = 0; j < n; j++)
        {
            if (i & (1 << j) == 0) continue;
            for (int k = 0; k < n; k++)
            {
                if (j == k || i & (1 << k) == 0) continue;
                dp[i][j] = max(dp[i][j],dp[i^(1 << k)][j]*rate[j][k]+rate[k][j]*dp[i^(1 << j)][k]);
            }
        }
    }
    double ans = 0;
    for (int i = 0; i < n; i++)
        ans = max(ans, dp[(1<<n)-1][i]);
    cout << setprecision(12) << ans << endl;
}
