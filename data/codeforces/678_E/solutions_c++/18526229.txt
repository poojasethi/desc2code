#include <bits/stdc++.h>
#define FN(i, n) for (int i = 0; i < (int)(n); ++i)
#define FEN(i,n) for(int i=1;i<=(int)(n); ++i)
#define FA(i, a) for (__typeof((a).begin()) i = (a).begin(); i != (a).end(); i++)
#define pb push_back
#define mp make_pair
#define sz(a) (int)(a).size()
#define f first
#define s second
#define pii pair<int,int>
#define vi vector<int>
#define ll long long
#define db double
using namespace std ;
db P[20][20] ,dp[20][20] ;
int E[20] ; bool V[20] ;
int main()
{
    std::ios::sync_with_stdio(false);
    int N ; cin>>N ;
    FN(i,N)FN(j,N) cin>>P[i][j] ;
    memset(E,-1,sizeof(E)) ;
    dp[0][0]=1 ;V[0]=1 ;E[0]=0 ;
    FEN(i,N-1)
    {
        FEN(j,N-1) if(!V[j])
        {
            dp[i][j] = dp[i-1][E[i-1]]*P[E[i-1]][j] + dp[i-1][j]*P[j][E[i-1]] ;
            if(E[i]==-1 || dp[i][j] > dp[i][E[i]]) E[i]=j ;
        }
        V[E[i]]=true ;
    }
    cout<<setprecision(15)<<dp[N-1][E[N-1]]<<endl ;
    return 0 ;
}
