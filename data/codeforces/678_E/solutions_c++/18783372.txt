#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>

#define ll long long
#define inf 1e9
#define md
using namespace std;
double f[300010][20];
double p[20][20];
int main()
{
	int n; scanf("%d",&n);
	for (int i=0;i<n;i++)
	  for (int j=0;j<n;j++)
	    scanf("%lf",&p[i][j]);
	f[1][0]=1;
	for (int S=2;S<(1<<n);S++)
	{
		if (!(S&1)) continue;
		for (int i=0;i<n;i++)
		  if (S&(1<<i))
		  {
		  	for (int j=0;j<n;j++)
		  	  if (S&(1<<j)) f[S][i]=max(f[S][i],f[S^(1<<i)][j]*p[j][i]+f[S^(1<<j)][i]*p[i][j]);
		  }
	}
	double ans=0;
	for (int i=0;i<n;i++) ans=max(ans,f[(1<<n)-1][i]);
	printf("%.10lf\n",ans);
	return 0;
}
