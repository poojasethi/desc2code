#include <cstdio>
#include <map>
#include <algorithm>
using namespace std;
double dp[1<<18][18];
double m[20][20];

int main()
{
	int n;
	double ans=0;
	scanf("%d",&n);
	
	
	for (int i=0;i<n;i++)
	{
		for (int j=0;j<n;j++)
		{
			scanf("%lf",&m[i][j]);
		}
	}
	dp[1][0]=1.0;
	for (int i=3;i<(1<<n);i+=2)
	{
		for (int j=0;j<n;j++)
		{
			if ( i&(1<<j) )
			{
				for (int k=0;k<n;k++)
				{
					if ( i&(1<<k) && j!=k )
					{
						dp[i][j]=max(dp[i][j],m[j][k]*dp[i^(1<<k)][j]+m[k][j]*dp[i^(1<<j)][k]);
					}
				}
			}
		}
	}
	for (int i=0;i<n;i++) ans=max(ans,dp[(1<<n)-1][i]);
	
	
	
	
	
	
	
	printf("%lf",ans);
}
