#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const int maxn = 18;
double p[maxn][maxn],dp[1<<maxn][maxn];
int main() {
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            cin>>p[i][j];
    dp[1][0]=1;
    for(int i=0;i<(1<<n);i++){
        for(int j=0;j<n;j++)
        if(i&(1<<j)){
            for(int k=0;k<n;k++)
            if(i&(1<<k)&&(k!=j))
                dp[i][j]=max(dp[i][j],p[j][k]*dp[i^(1<<k)][j]+p[k][j]*dp[i^(1<<j)][k]);
        }
    }
    double ans = 0;
    for(int i=0;i<n;i++) ans=max(ans,dp[(1<<n)-1][i]);
    printf("%f\n",ans);
    return 0;
}
