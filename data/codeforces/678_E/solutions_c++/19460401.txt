
#include<stdio.h>
#include<bits/stdc++.h>

using namespace std;

double rec[18][18];
double dp[1<<18][18];

int main()
{
    int n;
    scanf("%d",&n);
    
    for(int i=0;i<n;i++)
    for(int j=0;j<n;j++)
    scanf("%lf",&rec[i][j]);
    
    dp[1][0]=1;
    
    for(int i=0;i<(1<<n);i++)
    {
        for(int j=0;j<n;j++)
        {
	
		if (i&(1<<j)==0) continue;
        
        for(int k=0;k<n;k++){
        	
			if( i&(1<<k) && (k!=j) )
            
			dp[i][j] = max(dp[i][j],rec[j][k]*dp[i^(1<<k)][j]+rec[k][j]*dp[i^(1<<j)][k]);
        
			}
		}
    }
    double ans = 0;
    for(int i=0;i<n;i++)
        ans = max( ans, dp[(1<<n)-1][i]);
    printf("%.12f\n",ans);
}
