#include <cstdio>
#include <map>
#define FOR(i,a,b) for(i=a; i<=b; i++)
using namespace std;
map < long long , long long > mp,mp2;
int main()
{
	long long res(0);
	int K,N,a;
	scanf("%d %d" , &N , &K );
	while(N--)
	{
		scanf("%d" , &a );
		if( !(a%K) )
		{
			res += mp2[ a/K ];
			mp2[a] += mp[a/K];
		}
		mp[a]++;
	}

	printf("%lld\n" , res );

	return 0;
}
