#include <iostream>
#include <map>

using namespace std;

int main() {
	int n, k;
	long long ans=0;
	map<long long, long long> a, a1;
	cin>>n>>k;
	for (int i=0; i<n; i++){
		int x;
		cin>>x;
		if (x % (k*k)==0){
			ans += a1[x / k];
		}
		if (x % k ==0){
			a1[x] += a[x / k];
		}		
		a[x]++;
	}
	cout<<ans;
}
