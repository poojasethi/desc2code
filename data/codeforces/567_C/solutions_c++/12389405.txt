#include <bits/stdc++.h>
using namespace std;
map<long long int, long long int> m,m1;
int main() {
    ios::sync_with_stdio(false);
    int n, k,i;
    cin>>n>>k;
    long long int a,c=0;
    for(i=0;i<n;i++)
    {
        cin>>a;
        c+=m[a];
        if(a%k == 0)
            m[a*k]+=m1[a/k];
        m1[a]++;
    }
    cout<<c;
}
