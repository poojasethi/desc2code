#include<cstdio>
#include<map>
using namespace std;
int main()
{
	int n,k;
	long long ans=0;
	scanf("%d%d",&n,&k);
	map<int,long long> a,b;
	for(int i=0;i<n;i++)
	{
		int p;
		scanf("%d",&p);
		if(p%k==0)
		{
			ans+=b[p/k];
			b[p]+=a[p/k];
		}
		a[p]++;
	}
	printf("%I64d\n",ans);
}
