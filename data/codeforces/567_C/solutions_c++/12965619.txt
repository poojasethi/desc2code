#include<bits/stdc++.h>
using namespace std;
#define ll long long 
ll a;
map<ll,ll>lt,rt;
int main()
{
	ll i,j,n,k;
	ll ans=0,l,r;
	scanf("%lld%lld",&n,&k);
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a);
		if(a%k==0)
		{
			ans+=rt[a/k];
			rt[a]+=lt[a/k];
		}
		lt[a]++;
	}
	printf("%lld\n",ans);
	return 0;
}
