#include<iostream>
#include<map>
using namespace std;
int main()
{
	int n,k;
	long long a[200005];
	cin>>n>>k;
	map<long long,long long> r,l;
	int i,j;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		r[a[i]]++;
	}
	long long count=0;
	for(i=0;i<n;i++)
	{
		r[a[i]]--;
		if(a[i]%k==0)
		count+= (l[a[i]/k]*r[a[i]*k]);
		l[a[i]]++;
	}
	cout<<count;
}
