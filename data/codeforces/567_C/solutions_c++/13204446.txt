#include<bits/stdc++.h>
using namespace std;
#define ll long long
map <ll, ll> o,m;
int main()
{
    int n, k;
    cin>>n>>k;
    ll sum = 0;
    int tmp = 0;
    for(int i = 0; i<n; i++) {
        cin>>tmp;
        if(tmp%k==0) {
            sum+=m[tmp/k];
            m[tmp]+=o[tmp/k];
        }
        o[tmp]++;
    }
    cout<<sum<<endl;
}
