#include <iostream>
#include <cstdio>
#include <map>
using namespace std;

typedef long long LL;
LL n, k, a, ans;
map<int, LL> s1, s2;
int main()
{
	cin >> n >> k;
	while (n--) {
		scanf ("%I64d", &a);
		if (a % k == 0) {
			ans += s2[a / k];
			s2[a] += s1[a / k];
		}
		s1[a]++;
	}
	cout << ans << endl;
    return 0;
}
