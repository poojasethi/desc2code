#include<stdio.h>
#include<iostream>
#include<map>
using namespace std;
map<int,__int64>mp;
map<int,__int64>mp2;
int main()
{
	int n,k,x;
	mp.clear(),mp2.clear();
	__int64 ans=0;
	scanf("%d%d",&n,&k);
	while(n--){
		scanf("%d",&x);
		if(x%k==0)ans+=mp2[x/k],mp2[x]+=mp[x/k];
		mp[x]++;
	}
	printf("%I64d\n",ans);
	return 0;
}
