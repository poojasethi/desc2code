#include<iostream>
#include<cstdio>
#include<map>
using namespace std;
map<__int64,__int64>sp,sp1;
int main(){
	__int64 n,k,i,x,num=0;
	cin>>n>>k;
	for(i=1;i<=n;i++){
		scanf("%I64d",&x);
		if(x%k==0){
			num=num+sp1[x/k];
			sp1[x]+=sp[x/k];	
		}
		sp[x]++;
	}
	cout<<num<<endl;
	return 0;
}
