#include <stdio.h>
#include <map>

using namespace std;

int main (void) {
	int n, k;
	scanf ("%d%d", &n, &k);
	long long ans = 0;
	map<int, long long> sum, cnt;
	for (int i = 0; i < n; i++) {
		int x;
		scanf ("%d", &x);
		if (x%k == 0) {
			int y = x/k;
			ans += sum[y];
			sum[x] += cnt[y];
		}
		cnt[x]++;
	}
	printf ("%lld\n", ans);
}
