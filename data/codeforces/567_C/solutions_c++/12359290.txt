#include<bits/stdc++.h>
using namespace std;
#define ll long long
map<int,ll> m,dp;
int n,k,x;
ll ans=0;
int main(){
    scanf("%d%d",&n,&k);
    for (int i=1; i<=n; i++){
        scanf("%d",&x); 
        if (!(x%k)){
            ans+=dp[x/k]; 
            dp[x]+=m[x/k];
        }
        m[x]++;
    }
    printf("%I64d",ans);
}
