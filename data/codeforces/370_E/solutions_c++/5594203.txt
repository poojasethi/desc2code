#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <sstream>
#include <map>
#include <queue>
#include <vector>

using namespace std;

#define FOR(prom, a, b) for(int prom = (a); prom < (b); prom++)
#define FORD(prom, a, b) for(int prom = (a); prom > (b); prom--)
#define FORDE(prom, a, b) for(int prom = (a); prom >= (b); prom--)

#define PB push_back
#define MP make_pair

#define MM(co, cim) memset((co), (cim), sizeof((co)))

#define DEB(x) cerr << ">>> " << #x << " : " << x << endl;

#define MX 200007
int main ()
{
  int N;
  cin >> N;
  int a[MX];
  FOR(i,0,N) cin >> a[i];
  int mnb[MX], mnd[MX], mxb[MX], mxd[MX];
  int rb[MX], rd[MX];
  mnb[0] = mnd[0] = mxb[0] = mxd[0] = 1;
  if(a[0] > 1) {cout << -1 << endl; return 0;}
  FOR(i,1,N) {
  	mnb[i] = mnb[i-1];
  	mxb[i] = mxb[i-1];
  	mnd[i] = mnd[i-1]+1;
  	mxd[i] = mxd[i-1]+1;
  	if(mnd[i] > 5) {
  		mnb[i]++;
  		mnd[i] = 1;
  	}
  	if(mxd[i] > 2) {
  		mxb[i]++;
  		mxd[i] = 1;
  	}
  	if(a[i] != 0) {
  		if(a[i] < mnb[i]) {cout << -1 << endl; return 0;}
  		if(a[i] > mxb[i]) {cout << -1 << endl; return 0;}
  		if(a[i] > mnb[i]) {
  			mnb[i] = a[i];
  			mnd[i] = 1;
  		}
  		if(a[i] < mxb[i]) {
  			mxb[i] = a[i];
  			mxd[i] = 5;
  		}
  	}
  }
  if(mnb[N-1] == mxb[N-1] && mnd[N-1] == 1 && mxd[N-1] == 1) {cout << -1 << endl; return 0;}
  if(mxd[N-1] == 1) {
  	rb[N-1] = mxb[N-1]-1;
  	rd[N-1] = 2;
  	if(rb[N-1] == mnb[N-1] && mnd[N-1] > rd[N-1]) rd[N-1] = mnd[N-1];
  } else {
  	rb[N-1] = mxb[N-1];
  	rd[N-1] = mxd[N-1];
  }
	FORDE(i,N-2,0) {
		rb[i] = rb[i+1];
		rd[i] = rd[i+1]-1;
		if(rd[i] == 0) {
			rb[i]--;
			rd[i] = 5;
		}
		if(mxb[i] == rb[i] && mxd[i] < rd[i]) rd[i] = mxd[i];
	}
	cout << rb[N-1] << endl;
	FOR(i,0,N) cout << rb[i] << " ";
	cout << endl;
  return 0;
}
