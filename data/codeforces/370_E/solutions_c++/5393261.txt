#include <iostream>
#define pii pair<int, int>
using namespace std;
int a[200005], cnt[200005];
pii maxVal[200005], minVal[200005];
int main() {
  int n;
  cin >> n;
  for (int i = 1; i <= n; ++i) cin >> a[i];
  if (a[1] > 1) {
    cout << -1;
  } else {
    int flag = 1;
    maxVal[1] = minVal[1] = make_pair(1, 1);
    for (int i = 2; i <= n && flag; ++i) {
      if (maxVal[i - 1].second >= 2) {
        maxVal[i] = make_pair(maxVal[i - 1].first + 1, 1);        
      } else {
        maxVal[i] = make_pair(maxVal[i - 1].first, maxVal[i - 1].second + 1);
      }
      if (minVal[i - 1].second == 5) {
        minVal[i] = make_pair(minVal[i - 1].first + 1, 1);        
      } else {
        minVal[i] = make_pair(minVal[i - 1].first, minVal[i - 1].second + 1);
      }
      if (a[i]) {
        if (maxVal[i].first < a[i] || minVal[i].first > a[i]) {
          flag = 0;
        } else {
          maxVal[i] = min(maxVal[i], make_pair(a[i], 5));
          minVal[i] = max(minVal[i], make_pair(a[i], 1));
        }
      }
    }
    if (flag) {
      pii ans = maxVal[n];
      if (ans.second == 1) ans = make_pair(maxVal[n].first - 1, 5);
      if (ans < minVal[n]) {
        cout << -1;
      } else {
        cout << ans.first << '\n';
        a[n] = ans.first;
        cnt[a[n]] = 1;
        for (int i = n - 1; i; --i) {
          a[i] = min(a[i + 1], maxVal[i].first);
          if (cnt[a[i]] == 5) a[i]--;
          cnt[a[i]]++;
        }
        for (int i = 1; i <= n; ++i) cout << a[i] << ' ';        
      }
    } else {
      cout << -1;
    }
  }
}
