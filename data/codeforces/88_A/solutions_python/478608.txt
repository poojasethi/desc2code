#!/usr/bin/env python


stuff = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'B', 'H']

def det(X, Y, Z):
    X = stuff.index(X)
    Y = stuff.index(Y)
    Z = stuff.index(Z)
    if Y < X: Y += len(stuff)
    if Z < Y: Z += len(stuff)
    if (Y - X, Z - Y) == (4, 3):
        return "major"
    elif (Y - X, Z - Y) == (3, 4):
        return "minor"

def main():
    X, Y, Z = raw_input().split()
    print (
        det(X, Y, Z) or
        det(X, Z, Y) or
        det(Y, X, Z) or
        det(Y, Z, X) or
        det(Z, X, Y) or
        det(Z, Y, X) or
        "strange")


if __name__ == "__main__":
    main()
