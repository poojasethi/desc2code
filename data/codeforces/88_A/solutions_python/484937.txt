from itertools import *
ar = 'C C# D D# E F F# G G# A B H'.split() * 2
ins = raw_input().split()
per = [ x for x in permutations(ins) ]

def cal(s, t):
    d = ar.index(s)
    return ar[d:].index(t)

for x in per:
    if cal(x[0], x[1]) == 4 and cal(x[1], x[2]) == 3:
        print 'major'
        break
    if cal(x[0], x[1]) == 3 and cal(x[1], x[2]) == 4:
        print 'minor'
        break
else:
    print 'strange'
