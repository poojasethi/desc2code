#!/usr/local/bin/python

# 

import math
import sys

table = {
	"C": 0,
	"C#": 1,
	"D": 2,
	"D#": 3,
	"E": 4,
	"F": 5,
	"F#": 6,
	"G": 7,
	"G#": 8, 
	"A" : 9, 
	"B" : 10, 
	"H" : 11
	}


input = sys.stdin.readline().split()
input.sort()
chord = [table[e] for e in input]

if ((chord[1]-chord[0])%12 == 4 and (chord[2]-chord[1])%12 == 3) or ((chord[0]-chord[2])%12 == 4 and (chord[1]-chord[0])%12 == 3) or ((chord[2]-chord[1])%12 == 4 and (chord[0]-chord[2])%12 == 3):
	print 'major'
elif ((chord[1]-chord[0])%12 == 3 and (chord[2]-chord[1])%12 == 4) or ((chord[0]-chord[2])%12 == 3 and (chord[1]-chord[0])%12 == 4) or ((chord[2]-chord[1])%12 == 3 and (chord[0]-chord[2])%12 == 4):
	print 'minor'
else:
	print 'strange'
