#! /usr/bin/env python

tones=['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'B', "H" ]
tones.extend(tones)

from sys import stdin

def dist(x,y):
	#print tones
	first=tones.index(x)
	second=tones.index(y,first)
	return second-first

def is_major(x,y,z):
	#print dist(x,y),dist(y,z)
	return dist(x,y)==4 and dist(y,z)==3

def is_minor(x,y,z):
	#print dist(x,y),dist(y,z)
	return dist(x,y)==3 and dist(y,z)==4

if __name__=='__main__':
	X,Y,Z=stdin.readline().split()
	#print dist('C','E'), dist('E','C')
	found = False
	for (x,y,z) in [(X,Y,Z),(X,Z,Y),(Y,X,Z),(Y,Z,X),(Z,X,Y),(Z,Y,X)]:
		#print x,y,z
		if is_major(x,y,z):
			print "major"
			found=True
			break
		elif is_minor(x,y,z):
			print 'minor'
			found=True
			break
	
	if not found:
		print 'strange'
