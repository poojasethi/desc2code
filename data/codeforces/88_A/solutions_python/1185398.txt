N = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'B', 'H']
ch = map(lambda x: N.index(x), raw_input().split())
ch.sort()
ch.extend([ch[0] + 12, ch[1] + 12])
D = ''.join([str(min(ch[i] - ch[i-1], 9)) for i in range(1, 5)])
if '34' in D:
   print "minor"
elif '43' in D:
   print "major"
else:
   print "strange"
