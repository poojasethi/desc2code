from itertools import *
l=['C','C#','D','D#','E','F','F#','G','G#','A','B','H']
def i(s): return l.index(s)
def d(a,b):
    v=i(b)-i(a)
    if v<0: v+=len(l)
    return v
s=raw_input().split()
a='strange'
for c in permutations(s):
    ls=[d(c[0],c[1]),d(c[1],c[2])]
    if ls==[3,4]: a='minor'
    if ls==[4,3]: a='major'
print a
