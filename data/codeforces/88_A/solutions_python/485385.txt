from itertools import *
key = 'C C# D D# E F F# G G# A B H'.split()
s = map(key.index, raw_input().split())
ans = ('major', 'minor', 'strange')
t = 2
g = [(4, 3), (3, 4)]
for x in permutations(s):
    for i in range(2):
        if (x[1] - x[0] + 12) % 12 == g[i][0] and (x[2] - x[1] + 12) % 12 == g[i][1]:
            t = i
print ans[t]
