import sys

x,y,z = sys.stdin.readline().strip().split(' ')

list = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'B', 'H']

l = list + list


a = [(x,y,z), (x,z,y), (y,x,z), (y,z,x), (z,x,y), (z,y,x)]

for xx,yy,zz in a:
  b = l.index(xx)
  ll = l[b:]
  if ll.index(yy) - ll.index(xx) == 4 and ll.index(zz) - ll.index(yy) == 3:
    print 'major'
    break
  elif ll.index(yy) - ll.index(xx) == 3 and ll.index(zz) - ll.index(yy) == 4:
    print 'minor'
    break
else:
  print 'strange'
