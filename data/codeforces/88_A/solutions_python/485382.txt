key = ["C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "B", "H"]
ans = "strange"
s = map(key.index, raw_input().split())
for i in range(3):
	for j in range(3):
		if j != i:
			for k in range(3):
				if j != k and i != k:
					if (s[j] - s[i] + 12) % 12 == 4 and (s[k] - s[j] + 12) % 12 == 3:
						ans = "major"
					if (s[j] - s[i] + 12) % 12 == 3 and (s[k] - s[j] + 12) % 12 == 4:
						ans = "minor"
print ans
