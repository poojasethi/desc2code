#!/usr/bin/env python
import itertools
NOTES = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'B', 'H']
NOTES +=NOTES
def get_dist(t1, t2):
    global NOTES
    d = NOTES[NOTES.index(t1):].index(t2)
    return d
    
t = raw_input().split(' ')

for x in  list(itertools.permutations(t)):
    d1= get_dist(x[0], x[1]);d2= get_dist(x[1], x[2])
    if d1 == 4 and d2 ==3:
        print "major"
        exit()
    elif d1==3 and d2==4:
        print "minor"
        exit()
print 'strange'
