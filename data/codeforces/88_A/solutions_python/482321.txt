#!/usr/env	python
import sys;

def translate (a):
	if a == 'C': return 0
	elif a == 'C#': return 1;
	elif a == 'D': return 2;
   	elif a == 'D#': return 3;
   	elif a == 'E': return 4;
   	elif a == 'F': return 5;
   	elif a == 'F#': return 6;
   	elif a == 'G': return 7;
   	elif a == 'G#': return 8;
   	elif a == 'A': return 9;
   	elif a == 'B': return 10;
	elif a == 'H': return 11;
	else: return -1;

def calculate(a, b, c):
	if((translate(a) + 4)%12==translate(b) and (translate(b)+3)%12==translate(c)):
		sys.stdout.write('major\n');
		return True;
	if((translate(a) + 3)%12==translate(b) and (translate(b)+4)%12==translate(c)):
		sys.stdout.write('minor\n');
		return True;
	return False;


def test(a, b, c):
	if(calculate(a, b, c)):
		return
	elif(calculate(a, c, b)):
		return
	elif(calculate(b, a, c)):
		return
	elif(calculate(b, c, a)):
		return
	elif(calculate(c, b, a)):
		return
	elif(calculate(c, a, b)):
		return
	else:
		sys.stdout.write('strange\n');

if __name__ == '__main__':
	a, b, c = sys.stdin.readline().split();
	test(a, b, c)
