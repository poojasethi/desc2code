#!/usr/bin/env python
# -*- coding: utf-8 -*-

import itertools

def test(l, s):
    nl = l[l.index(s[0]):]
    if nl.index(s[2]) == 7:
        if nl.index(s[1]) == 4: return 'major'
        elif nl.index(s[1]) == 3: return 'minor'
    return 'strange'

l = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'B', 'H']*2
s = raw_input().split()

ans = 'strange'
for ts in itertools.permutations(s):
    ans = test(l, ts)
    if ans != 'strange': break
print ans
