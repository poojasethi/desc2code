from itertools import *
f = lambda(x): ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'B', 'H'].index(x)
def g(x, y): return ((y - x + 12) if (y < x) else (y - x))
s = 'strange'
for p in permutations( map(f, raw_input().split())):
  if [4, 3] == [g(p[0], p[1]), g(p[1], p[2])]: s = 'major'
  elif [3, 4] == [g(p[0], p[1]), g(p[1], p[2])]: s = 'minor'
print s
