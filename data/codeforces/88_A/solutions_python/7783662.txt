#!/usr/bin/env python
l=['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'B', 'H', 'C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'B', 'H']
m=raw_input().split(' ')
if 7==l[l.index(m[0]):].index(m[2]):
    mod_m=[m[0],m[1],m[2]]
elif 7==l[l.index(m[2]):].index(m[0]):
    mod_m=[m[2],m[1],m[0]]
elif 7==l[l.index(m[0]):].index(m[1]):
    mod_m=[m[0],m[2],m[1]]
elif 7==l[l.index(m[1]):].index(m[0]):
    mod_m=[m[1],m[2],m[0]]
elif 7==l[l.index(m[1]):].index(m[2]):
    mod_m=[m[1],m[0],m[2]]
elif 7==l[l.index(m[2]):].index(m[1]):
    mod_m=[m[2],m[0],m[1]]
else:
    print 'strange'
    exit()
z=[]
start=l.index(mod_m[0])
for i in range(2):
    z.append(l[start:].index(mod_m[i+1])-l[start:].index(mod_m[i]))
    start=l.index(mod_m[i])
if z==[4,3]:
    print 'major'
elif z==[3,4]:
    print 'minor'
else:
    print 'strange'
