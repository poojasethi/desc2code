s = ['C','C#','D','D#','E','F','F#','G','G#','A','B','H']
n = len(s)

p = [s.index(x) for x in raw_input().split()]
p.sort()
p = p+p
for x in range(3):
	if (p[x+1]-p[x]+n)%n == 4 and (p[x+2]-p[x+1]+n)%n == 3:
		print 'major'
		exit()
	elif (p[x+1]-p[x]+n)%n == 3 and (p[x+2]-p[x+1]+n)%n == 4:
		print 'minor'
		exit()
print 'strange'
