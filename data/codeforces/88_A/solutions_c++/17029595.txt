#include<bits/stdc++.h>

using namespace std;
int main()
{	
	string str[12]={"C","C#","D","D#","E","F","F#","G","G#","A","B","H"};
	int arr[12],i,j,k,l;
	string a,b,c;
	cin>>a>>b>>c;
	for(i=0;i<12;i++)
	{
		arr[i]=0;
		if(a==str[i] || b==str[i]||c==str[i])
		{
			//cout<<1;
			arr[i]=1;
		}
	}
	for(i=0;i<12;i++)
	{
		if(arr[i]==1 &&arr[(i+4)%12]==1 &&arr[(i+7)%12]==1)
		{
			cout<<"major";
			return 0;
		}
		if(arr[i]==1 &&arr[(i+3)%12]==1 && arr[(i+7)%12]==1)
		{
			cout<<"minor";
			return 0;
		}
	}
	cout<<"strange";
	return 0;
}
