#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
string s[12] = {"C","C#","D","D#","E","F","F#","G","G#","A","B","H"};
int v[3];
int check() {
  if(v[1]-v[0]==4 && v[2]-v[1]==3) { cout << "major"; return 1; }
  else if(v[1]-v[0]==3 && v[2]-v[1]==4) { cout << "minor"; return 1; }
  return 0;
}
int main() {
  string vs[3];

  for(int i=0; i<3; ++i) {
    cin >> vs[i];
    for(int j=0; j<12; ++j) if(vs[i]==s[j]) v[i]=j;
  }
  sort(v,v+3);
  for(int i=0; i<3; ++i) {
    if(check()) return 0;
    v[0]+=12;
    sort(v, v+3);
  }
  cout << "strange";
  return 0;
}
