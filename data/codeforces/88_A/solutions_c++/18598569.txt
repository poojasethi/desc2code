int main() {
	/*freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);*/
	cin>>a>>b>>c;
	for(int i=0;i<12;++i){
		if(a==s[i])x[i]=1;
		if(b==s[i])x[i]=1;
		if(c==s[i])x[i]=1;
	}
	for(int i=0;i<12;++i)
		if(x[i]){
			if(x[(i+4)%12]&&x[(i+7)%12]){ printf("major"); return 0;}

		}
	printf("strange");

}
