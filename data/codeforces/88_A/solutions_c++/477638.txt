#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

string note[12]={"C","C#","D","D#","E", "F", "F#", "G", "G#", "A", "B", "H"};

int main(int argc, char* argv[])
{
   string a[3];
   int i,j,b[3];
   cin>>a[0]>>a[1]>>a[2];
   for (i=0;i<3;i++)
      for (j=0;j<12;j++)
         if (a[i]==note[j])
            b[i]=j;
   sort(b,b+3);
   if (b[1]-b[0]==4 && b[2]-b[1]==3 || b[2]-b[1]==4 && b[0]+12-b[2]==3 ||
      b[0]+12-b[2]==4 && b[1]-b[0]==3)
      cout<<"major";
   else if (b[1]-b[0]==3 && b[2]-b[1]==4 || b[2]-b[1]==3 && b[0]+12-b[2]==4 ||
      b[0]+12-b[2]==3 && b[1]-b[0]==4)
      cout<<"minor";
   else
      cout<<"strange";
   return 0;
}
