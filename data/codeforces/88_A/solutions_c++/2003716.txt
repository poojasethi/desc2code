#include <iostream>

using namespace std;
string m[12] = {"C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "B", "H"};

bool f(string& a, string& b, string& c) {
  int i = 0;
  for (; i < 12; i++)
    if (a == m[i])
      break;
  
  if (m[(i + 3) % 12] == b && m[(i + 7) % 12] == c) {
    cout << "minor";
    return 1;
  }
  
  if (m[(i + 4) % 12] == b && m[(i + 7) % 12] == c) {
    cout << "major";
    return 1;
  }
  return 0;
}

int main() {
  string a, b, c;
  cin >> a >> b >> c;
  if (!(f(a, b, c) || f(a, c, b) ||f(b, a, c) || f(b, c, a) || f(c, a, b) || f(c, b, a)))
    cout << "strange";
  return 0;
}
