#include<iostream>
#include<string>
using namespace std;
int i,x[12];
int main(){
	string s[12]={"C","C#","D","D#","E","F","F#","G","G#","A","B","H"},a,b,c;
	cin>>a>>b>>c;
	for(i=0;i<12;i++)
		if(a==s[i]||b==s[i]||c==s[i])x[i]=1;
	for(i=0;i<12;i++)
		if(x[i]){
			if(x[(i+4)%12]&&x[(i+7)%12]){
				cout<<"major";
				return 0;
			}
			if(x[(i+3)%12]&&x[(i+7)%12]){
				cout<<"minor";
				return 0;
			}
		}
	cout<<"strange";
}
