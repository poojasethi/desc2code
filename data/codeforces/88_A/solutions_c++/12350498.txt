#include<iostream>
#include<string>
#include<algorithm>
using namespace std;

string notes[12]={"C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "B", "H"};
int find(string s)
{
	return find(notes,notes+12,s)-notes;
}
int main()
{
	string s[3];
	string res="strange";
	cin>>s[0]>>s[1]>>s[2];
	sort(s,s+3);
	do
	{
		int d1=(find(s[1])-find(s[0])+12)%12;
		int d2=(find(s[2])-find(s[1])+12)%12;
		if(d1==4&&d2==3)
		res="major";
		if(d2==4&&d1==3)
		res="minor";
	}while(next_permutation(s,s+3));
	cout<<res<<endl;
return 0;
}
