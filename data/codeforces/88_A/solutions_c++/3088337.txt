int tab[]={9,10,0,2,4,5,7,11};
int s[3];

void input(int a){
	char c[5];scanf("%s",c);
	s[a] = tab[c[0]-'A'] + (c[1]=='#'?1:0);
}

int main(){
	for(int i=0;i<3;i++)input(i);
	sort(s,s+3);
	for(int i=0;i<3;i++){
		if(s[1]-s[0]==4 && s[2]-s[1]==3){puts("major"); return 0;}
		if(s[1]-s[0]==3 && s[2]-s[1]==4){puts("minor"); return 0;}
		s[0]+=12;
		sort(s,s+3);
	}

}
