using namespace std;

string notes[12]={"C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "B", "H"};

int findn(string s)
{
//  return find(notes,notes+12,s)-notes;
    for(int i=0;i<12;i++)
    {
        if(s==notes[i]) return i;
    }
}

{
    string s[3],res="strange";
    cin>>s[0]>>s[1]>>s[2];
    sort(s,s+3);
    do{
        int d1=((findn(s[1])-findn(s[0])+12)%12);
        int d2=((findn(s[2])-findn(s[1])+12)%12);
        if(d1==4&&d2==3)
            res="major";
        if(d1==3&&d2==4)
            res="minor";
    }

    cout<<res;
}
