#include <iostream>
#include <map>
#include <algorithm>
using namespace std;

string arr[] = {"C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "B", "H"};
map <string, int> M;

int main() {

	for(int i=0; i<12; i++)
		M[arr[i]] = i;

	string a, b, c;
	cin >> a >> b >> c;
	string sss[] = {a, b, c};
	sort(sss, sss+3);
	do {
		int x = M[sss[0]], y = M[sss[1]], z = M[sss[2]];
		if((x + 3) % 12 == y && (x+7) % 12 == z) {
			cout << "minor" << endl;
			return 0;
		} else if((x + 4) % 12 == y && (x+7) % 12 == z) {
			cout << "major" << endl;
			return 0;
		}

	} while(next_permutation(sss, sss+3));

	cout << "strange" << endl;

	return 0;
}
