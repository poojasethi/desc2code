#include <bits/stdc++.h>
using namespace std;

int main(){ 
    int ch[12]={0};
    string arr[]={"C","C#","D","D#","E","F","F#","G","G#","A","B","H"};
    string s1,s2,s3;
    cin>>s1>>s2>>s3;
    for(int i=0;i<12;i++)
    {
        if(s1==arr[i])ch[i]=1;
        else if(s2==arr[i])ch[i]=1;
        else if(s3==arr[i])ch[i]=1;
    }
    for(int i=0;i<12;i++)
    {
        if(ch[i] && ch[(i+4)%12] && ch[(i+7)%12])
        {
            cout<<"major"<<endl;return 0;
        }
        else if(ch[i] && ch[(i+3)%12] && ch[(i+7)%12])
        {
            cout<<"minor"<<endl;return 0;
        }
    }
    cout<<"strange" <<endl;
    
}
