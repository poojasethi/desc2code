#include <iostream>
#include <algorithm>
using namespace std;
int main(){
	string x,y,z,s[12]={"C","C#","D","D#","E","F","F#","G","G#","A","B","H"};
	cin>>x>>y>>z;
	int a[3];
	for(int i=0;i<12;i++)
	{
		if(x==s[i])
			a[0]=i;
			
		if(y==s[i])
			a[1]=i;
			
		if(z==s[i])
			a[2]=i;
	
	}
	sort(a,a+3);
	if(a[1]-a[0]==4&&a[2]-a[1]==3||a[2]-a[1]==4&&a[0]+12-a[2]==3||a[0]+12-a[2]==4&&a[1]-a[0]==3){
		cout<<"major";
		return 0;
	}
	else
	if(a[1]-a[0]==3&&a[2]-a[1]==4||a[2]-a[1]==3&&a[0]+12-a[2]==4||a[0]+12-a[2]==3&&a[1]-a[0]==4){
		cout<<"minor";
		return 0;
	}
	else{	
		cout<<"strange";
		return 0;
	}
}
