#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
char a[12][5]={"C", "C#", "D", "D#", "E", "F",
	"F#", "G", "G#", "A", "B", "H"};

int main(void) {
	char x[3][10];
	int p[3];
	scanf("%s%s%s", x[0], x[1], x[2]);
	for(int i=0;i<3;i++) {
		for(p[i]=0;p[i]<12;p[i]++)
			if(strcmp(x[i], a[p[i]])==0) break;
		p[i] %= 12;
	}
	sort(p, p+3);
	int flag=0;
	do {
		//printf("%d %d %d\n", p[0], p[1], p[2]);
		if(p[1]==(p[0]+4)%12 && p[2]==(p[1]+3)%12) {flag=1;}
		else if(p[1]==(p[0]+3)%12 && p[2]==(p[1]+4)%12) {flag=-1;}
	} while(next_permutation(p, p+3));
	if(!flag) puts("strange");
	else if(flag==1) puts("major");
	else puts("minor");
	return 0;
}
