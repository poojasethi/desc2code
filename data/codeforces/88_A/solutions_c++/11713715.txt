#include<iostream>
using namespace std;
bool check [12];
int main()
{ string s[]={"C","C#","D","D#","E","F","F#","G","G#","A","B","H"};
  string x,y,z;
  cin>>x>>y>>z;
  for(int i=0;i<12;i++)
  {
      if(s[i]==x||s[i]==y||s[i]==z)
        check[i]=true;
  }
  for(int i=0;i<12;i++)
  {
      if(check[i]==true)
      {
          if(check[(i+4)%12]==true&&check[(i+7)%12]==true)
          {
              cout<<"major";
              return 0;
          }
        if(check[(i+3)%12]==true&&check[(i+7)%12]==true)
          {
              cout<<"minor";
              return 0;
          }
      }
     // cout<<check[i]<<endl;
  }
  cout<<"strange";

    return 0;
}
