#include <bits/stdc++.h>
using namespace std;


string note[12]{"C","C#","D","D#","E","F","F#","G","G#","A","B","H"};
int main(){

    string X,Y,Z;
    cin>>X>>Y>>Z;
    int a[3];
    for(int i=0;i<12;i++){
        if(note[i]==X)a[0]=i;
        if(note[i]==Y)a[1]=i;
        if(note[i]==Z)a[2]=i;
    }

    sort(a,a+3);

    int d1=a[1]-a[0];
    int d2=a[2]-a[1];
    int d3=12-(a[2]-a[0]);


    if(d1==4 && d2==3 &&d3==5 || d1==5 && d2==4 &&d3==3 || d1==3 && d2==5 &&d3==4 )cout<<"major\n";
    else if(d1==3 && d2==4 &&d3==5 || d1==5 && d2==3 &&d3==4 || d1==4 && d2==5 &&d3==3 )  cout<<"minor\n";
    else cout<<"strange\n";
}
