#include<cstdio>
#include<iostream>
#include<string>
#include<map>

using namespace std;

char s[12][3] = {"C", "C#", "D", "D#", "E", "F", 
				"F#", "G", "G#", "A", "B", "H"};
map<string, int> m;
char ts[10];
bool vis[12];

int main()
{
	for(int i=0;i<12;i++)
		m[ s[i] ] = i;

	scanf("%s", ts);
	vis[ m[ts] ] = 1;
	scanf("%s", ts);
	vis[ m[ts] ] = 1;
	scanf("%s", ts);
	vis[ m[ts] ] = 1;
	
	int ans = 0;
	for(int i=0;i<12;i++)
	{
		if(vis[i] && vis[(i+4)%12] && vis[(i+7)%12])
			ans = 1;

		if(vis[i] && vis[(i+3)%12] && vis[(i+7)%12])
			ans = 2;
	}

	if(!ans)
		printf("strange\n");
	else if(ans == 1)
		printf("major\n");
	else
		printf("minor\n");
}
