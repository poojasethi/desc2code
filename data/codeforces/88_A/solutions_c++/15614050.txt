#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	string w[12] = {"C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "B", "H"};
	string s;
	int arr[3];
	
	for(int i = 0; i < 3; i ++)
	{
		cin>> s;
		for(int j = 0; j < 12; j ++)
			if(w[j] == s)
				arr[i] = j;
	}

	sort(arr, arr + 3);
	
	do
	{
		if((arr[0] + 4) % 12 == arr[1] && (arr[1] + 3) % 12 == arr[2])
		{
			cout<< "major\n";
			return 0;
		}
		if((arr[0] + 3) % 12 == arr[1] && (arr[1] + 4) % 12 == arr[2])
		{
			cout<< "minor\n";
			return 0;
		}
	}while(next_permutation(arr, arr + 3));

	cout<< "strange\n";
}
