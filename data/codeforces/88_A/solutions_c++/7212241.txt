#include<iostream>
#include<string>
int a[12],i;
using namespace std;
int main()
{
    string notes[]={"C","C#","D","D#","E","F","F#","G","G#","A","B","H"},x,y,z;
    cin>>x>>y>>z;
    for(i=0;i<12;i++)
        if(x==notes[i] || y==notes[i] || z==notes[i])
            a[i]=1;
    for(i=0;i<12;i++)
        if(a[i])
        {
            if(a[(i+3)%12] && a[(i+7)%12])
            {
                cout<<"minor"<<endl;
                return 0;
            }
            else if(a[(i+4)%12] && a[(i+7)%12])
            {
                cout<<"major"<<endl;
                return 0;
            }
        }
    cout<<"strange"<<endl;
}
