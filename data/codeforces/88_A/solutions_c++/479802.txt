#include<map>
#include<vector>
#include<algorithm>
#include<string>
#include<iostream>
using namespace std;

int main() {
	string list[12]={"C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "B", "H"};
	map<string, int> m;
	for (int i=0; i<12; i++) m[list[i]]=i;

	string s;
	vector<int> v;
	while(cin>>s) {
		int curr=m[s];
		v.push_back(curr);
		v.push_back(curr+12);
	}
	sort(v.begin(), v.end());

	int sz=v.size();
	bool flag=false;
	for (int i=0; i<sz-2; i++) {
		int p=v[i+1]-v[i], q=v[i+2]-v[i+1];
		if(p==3 && q==4) { flag=true; cout << "minor" << endl; break;}
		if(p==4 && q==3) { flag=true; cout << "major" << endl; break;}
	}

	if(!flag) cout << "strange" << endl;

	return 0;
}
