#include <bits/stdc++.h>
using namespace std;

string n [] ={"C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "B", "H"};
bool a[12];
string x,y,z;

int main() {
    
    cin>>x>>y>>z;
    for(int i=0; i<12; ++i){
        if(n[i] == x || n[i] == y || n[i] == z)
           a[i] = 1;
    }
    for(int i=0; i<12; ++i){
        if(a[i]){
            if(a[(i+4)%12] && a[(i+7)%12]){
                puts("major");
                return 0;
            }
            if(a[(i+3)%12] && a[(i+7)%12]){
                puts("minor");
                return 0;
            }
        }
    }
    puts("strange");
    
    return 0;
}
