#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
   string arr[3];
   string notemusic[12]={"C","C#","D","D#","E", "F", "F#", "G", "G#", "A", "B", "H"};
   int i,j,b[3];
   cin>>arr[0]>>arr[1]>>arr[2];
   for (i=0;i<3;i++)
      for (j=0;j<12;j++)
         if (arr[i]==notemusic[j])
            b[i]=j;
   sort(b,b+3);
   if (b[1]-b[0]==4 && b[2]-b[1]==3 || b[2]-b[1]==4 && b[0]+12-b[2]==3 ||
      b[0]+12-b[2]==4 && b[1]-b[0]==3)
      cout<<"major"<<endl;
   else if (b[1]-b[0]==3 && b[2]-b[1]==4 || b[2]-b[1]==3 && b[0]+12-b[2]==4 ||
      b[0]+12-b[2]==3 && b[1]-b[0]==4)
      cout<<"minor"<<endl;
   else
      cout<<"strange"<<endl;
   return 0;
}
