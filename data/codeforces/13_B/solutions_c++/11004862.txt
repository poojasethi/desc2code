#include <iostream>
#include <fstream>
using namespace std;

typedef long long int LL;

struct coor{
	LL x_a, y_a, x_b, y_b;
};


coor toch[5];

bool common(coor &s, coor &t) {
	if (s.x_a == t.x_a && s.y_a == t.y_a) return 1;
	swap(t.x_a, t.x_b);
	swap(t.y_a, t.y_b);
	if (s.x_a == t.x_a && s.y_a == t.y_a) return 1;
	swap(s.x_a, s.x_b);
	swap(s.y_a, s.y_b);
	if (s.x_a == t.x_a && s.y_a == t.y_a) return 1;
	swap(t.x_a, t.x_b);
	swap(t.y_a, t.y_b);
	if (s.x_a == t.x_a && s.y_a == t.y_a) return 1;
	swap(s.x_a, s.x_b);
	swap(s.y_a, s.y_b);
	return 0;
}

int main()
{
	/*ifstream in("input.txt");*/
	LL a, b, c, d, n;
	cin >> n;
	while (n--){
		for (int i = 0; i < 3; i++){
			cin >> toch[i].x_a >> toch[i].y_a >> toch[i].x_b >> toch[i].y_b;
		}
		for (int i = 0; i < 3; ++i){
			for (int j = i + 1; j < 3; ++j){
				if (common(toch[i], toch[j])){
					swap(toch[0], toch[i]);
					swap(toch[j], toch[1]);
					goto found;
				}
			}
		}
		goto failed;
	found:;
		for (int z = 3; z--; ){
			toch[z].x_b -= toch[0].x_a;
			toch[z].y_b -= toch[0].y_a;
			toch[z].x_a -= toch[0].x_a;
			toch[z].y_a -= toch[0].y_a;
		}
		if (toch[0].x_b * toch[1].x_b + toch[0].y_b*toch[1].y_b < 0)goto failed;
		if (toch[0].x_b*toch[1].y_b - toch[1].x_b*toch[0].y_b == 0)goto failed;
		if (toch[0].x_b*toch[2].y_a - toch[2].x_a*toch[0].y_b != 0){
			swap(toch[2].x_a, toch[2].x_b);
			swap(toch[2].y_a, toch[2].y_b);
		}
		if (toch[0].x_b*toch[2].y_a - toch[2].x_a*toch[0].y_b != 0)goto failed;
		if (toch[1].x_b*toch[2].y_b - toch[2].x_b * toch[1].y_b != 0)goto failed;
		if (toch[0].x_b*toch[2].x_a + toch[0].y_b*toch[2].y_a < 0)goto failed;
		if (toch[1].x_b*toch[2].x_b + toch[1].y_b*toch[2].y_b < 0)goto failed;
		a = toch[0].x_b*toch[0].x_b + toch[0].y_b*toch[0].y_b;
		b = toch[1].x_b*toch[1].x_b + toch[1].y_b*toch[1].y_b;
		c = toch[2].x_a*toch[2].x_a + toch[2].y_a*toch[2].y_a;
		d = toch[2].x_b*toch[2].x_b + toch[2].y_b*toch[2].y_b;
		if (c * 25 < a || a * 16 < c * 25)goto failed;
		if (d * 25 < b || b * 16 < d * 25)goto failed;
		cout << "YES\n";
		continue;
	failed:;
		cout << "NO\n";
	}
	/*system("pause>>null");*/
}
