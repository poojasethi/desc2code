#include <iostream>
#include <vector>
#include <cmath>
using namespace std;
class pt{
public:
    long long x;
    long long y;
};
double dot(pt p1, pt p2,pt p3, pt p4){
     double A=sqrt((p1.x-p2.x)*(p1.x-p2.x)+(p1.y-p2.y)*(p1.y-p2.y)+0.0);
     double B=sqrt((p3.x-p4.x)*(p3.x-p4.x)+(p3.y-p4.y)*(p3.y-p4.y)+0.0);
    return (((p1.x-p2.x)*(p3.x-p4.x)+(p1.y-p2.y)*(p3.y-p4.y)+0.0)/(A*B));
}

bool ptos(pt s, pt e, pt p){
    double a1=0;
    if(s.y!=p.y) a1= (s.x-p.x+0.0)/(s.y-p.y);
    double a2=0;
    if(e.y!=p.y) a2= (p.x-e.x+0.0)/(p.y-e.y);

    return (a1==a2)&&(p.x>=min(s.x,e.x) && p.x<=max(e.x,s.x) && p.y>=min(s.y,e.y) && p.y<=max(e.y,s.y));
}
bool ratio(pt s, pt e, pt p){
    double a=sqrt((e.x-p.x)*(e.x-p.x)+(e.y-p.y)*(e.y-p.y)+0.0);
    double b=sqrt((s.x-p.x)*(s.x-p.x)+(s.y-p.y)*(s.y-p.y)+0.0);
    if(a==0 || b==0 ) return 0;
    else if(a/b>4 || b/a>4) return 0;

    return true;
}

bool check(pt s1, pt e1, pt s2, pt e2, pt s3, pt e3){
    bool b1=false, b2=false, b3=false;
    if(s1.x==s2.x && s1.y==s2.y) b1 = true;
    //cout<<dot(s1,e1,s2,e2)<<endl;
    if(dot(s1,e1,s2,e2)>=0 && dot(s1,e1,s2,e2)<1) b2=true;
    if(ptos(s1,e1,s3) && ptos(s2,e2,e3)){
        if(ratio(s1,e1,s3) && ratio(s2,e2,e3)) b3=true;
    }
    if(ptos(s1,e1,e3) && ptos(s2,e2,s3)){
        if(ratio(s1,e1,e3) && ratio(s2,e2,s3)) b3=true;
    }

    return b1&&b2&&b3;

}
bool checkall(pt s1, pt e1, pt s2, pt e2, pt s3, pt e3, long long i){

    if (i == 0) {

        return checkall( s1 , e1 , s2 , e2 , s3 , e3 , 1 ) || checkall( e1 , s1 , s2 , e2 , s3 , e3 , 1 );

    }
    if ( i == 1 ) {

        return check( s1 , e1 , s2 , e2 , s3 , e3 ) || check( s1 , e1 , e2 , s2 , s3 , e3 );
    }


}

int main(){

    long long T;
    cin>>T;
    for(long long k=0;k<T;k++){
        vector<pt> v;
        pt p;
        for(long long i=0;i<6;i++){
            cin>>p.x>>p.y;
            v.push_back(p);
        }
        bool b=false;
        b=checkall(v[0],v[1],v[2],v[3],v[4],v[5],0)||checkall(v[0],v[1],v[4],v[5],v[2],v[3],0)||checkall(v[2],v[3],v[4],v[5],v[0],v[1],0);
    
        if(b) cout<<"YES";
        else cout<<"NO";
        cout<<endl;
    }
    

    return 0;
}
