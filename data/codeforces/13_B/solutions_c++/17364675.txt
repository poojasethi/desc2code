#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
typedef long long ll;
struct po{
	ll x[2],y[2];
}p[3];
ll a[3];
bool onseg(ll x1,ll y1, ll x2,ll y2,ll x, ll y){
	if((x-x1)*(x-x2)+(y-y1)*(y-y2)<=0){
		if((x-x1)*(y-y2)==(y-y1)*(x-x2)){
			ll lx=abs(x1-x2),ly=abs(y1-y2),tx=abs(x-x1),ty=abs(y-y1);
			if(5*tx>=lx&&5*tx<=4*lx&&5*ty>=ly&&5*ty<=4*ly)
				return true;
		}
	}
	return false;
}
bool judge(){
	for(int i=0;i<2;i++){
		if(onseg(p[a[0]].x[0],p[a[0]].y[0],p[a[0]].x[1],p[a[0]].y[1],p[a[2]].x[i],p[a[2]].y[i])){
			if(onseg(p[a[1]].x[0],p[a[1]].y[0],p[a[1]].x[1],p[a[1]].y[1],p[a[2]].x[!i],p[a[2]].y[!i]))
				return true;
		}
	}
	return false;
}
int main(){
	int T;
	scanf("%d",&T);
	while(T--){
		for(int i=0;i<3;i++){
			scanf("%I64d%I64d%I64d%I64d",&p[i].x[0],&p[i].y[0],&p[i].x[1],&p[i].y[1]);
			a[i]=i;
		}
		do{
			for(int i=0;i<2;i++)
				for(int j=0;j<2;j++)
					if(p[a[0]].x[i]==p[a[1]].x[j]&&p[a[0]].y[i]==p[a[1]].y[j]){
						ll px=p[a[0]].x[!i]-p[a[0]].x[i],py=p[a[0]].y[!i]-p[a[0]].y[i];
						ll qx=p[a[1]].x[!j]-p[a[1]].x[j],qy=p[a[1]].y[!j]-p[a[1]].y[j];
						if(px*qx+py*qy>=0&&px*qy!=py*qx)
							if(judge()){
								puts("YES");
								goto E;
							}
					}
		}while(next_permutation(a,a+3));
		puts("NO");
		E:;
	}
	return 0;
}
