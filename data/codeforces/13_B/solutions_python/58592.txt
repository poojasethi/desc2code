#!/usr/bin/python

n = int(raw_input())
for t in xrange(n):
    p1 = p2 = p3 = None
    p = [map(int,raw_input().split()),map(int,raw_input().split()),map(int,raw_input().split())]
    for i in xrange(2):
        for j in xrange(i+1,3):
            for k in xrange(2):
                for l in xrange(2):
                    if p[i][k*2]==p[j][l*2] and p[i][k*2+1]==p[j][l*2+1]:
                        p1 = [p[i][k*2],p[i][k*2+1],p[i][2-k*2],p[i][3-k*2]]
                        p2 = [p[j][l*2],p[j][l*2+1],p[j][2-l*2],p[j][3-l*2]]
                        p3 = p[3-i-j]
    if p1 is None:
        print "NO"
        continue
    v1 = [p1[2]-p1[0],p1[3]-p1[1]]
    v2 = [p2[2]-p2[0],p2[3]-p2[1]]
    sp = v1[0]*v2[0]+v1[1]*v2[1]
    l1 = v1[0]**2+v1[1]**2
    l2 = v2[0]**2+v2[1]**2
    if sp<0 or l1*l2==sp**2:
        print "NO"
        continue
    vt = [p1[0]-p3[0],p1[1]-p3[1]]
    if vt[0]*v1[1]-vt[1]*v1[0]!=0:
        p3 = [p3[2],p3[3],p3[0],p3[1]]
    vt = [p1[0]-p3[0],p1[1]-p3[1]]
    if vt[0]*v1[1]-vt[1]*v1[0]!=0 or not((p1[0]<=p3[0]<=p1[2] or p1[2]<=p3[0]<=p1[0]) and (p1[1]<=p3[1]<=p1[3] or p1[3]<=p3[1]<=p1[1])):
        print "NO"
        continue
    lt = vt[0]**2+vt[1]**2
    lt2 = (p1[2]-p3[0])**2+(p1[3]-p3[1])**2
    if lt>lt2:
        lt,lt2 = lt2,lt
    if 16*lt < lt2:
        print "NO"
        continue
    vt = [p2[0]-p3[2],p2[1]-p3[3]]
    if vt[0]*v2[1]-vt[1]*v2[0]!=0 or not((p2[0]<=p3[2]<=p2[2] or p2[2]<=p3[2]<=p2[0]) and (p2[1]<=p3[3]<=p2[3] or p2[3]<=p3[3]<=p2[1])):
        print "NO"
        continue
    lt = vt[0]**2+vt[1]**2
    lt2 = (p2[2]-p3[2])**2+(p2[3]-p3[3])**2
    if lt>lt2:
        lt,lt2 = lt2,lt
    if 16*lt < lt2:
        print "NO"
        continue
    print "YES"
