#include<iostream>
#include<cstring>
#include<cstdlib>
#include<cstdio>
using namespace std;
#define LL long long
const int mod=1000000007;
const int N=100010;
int n,m,L,tot;
LL sum;
int lnk[N];
int q[N],h,t;
int a[N],b[N],in[N],f[N];
int main() {
    scanf("%d%d%d",&n,&m,&L);
    for(int i=1;i<=n;i++) scanf("%d",&a[i]);
    for(int i=1,x,y;i<=m;i++) {
        scanf("%d%d",&x,&y);
        lnk[x]=y;
        in[y]++;
    }
    for(int i=1;i<=n;i++)
        if(!in[i]) {
            int s=0;
            for(int j=i;j;j=lnk[j]) {
                sum+=s;
                s+=a[j];
                b[++tot]=s;
            }
        }
    if(sum>L||tot<n) {printf("0\n");return 0;}
    f[sum]=1;
    for(int i=1;i<=tot;i++)
        for(int j=b[i];j<=L;j++) {
            f[j]+=f[j-b[i]];
            if(f[j]>=mod) f[j]-=mod;
        }
    printf("%d\n",f[L]);
    return 0;
}
