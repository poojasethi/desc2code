#include<cstdio>
#include<cstring>
#define N 100007
#define MOD 1000000007
int t,cnt;
int a[N];
int f[N];
int son[N],far[N];
void dfs(int p,int h){
	++cnt;
	t-=a[p]*h;
	if(t<0) return ;
	if(far[p]!=0){
		dfs(far[p],h+1);
		a[p]+=a[far[p]];
	}
}
int main(){
	int n,q,i,j,b,c;
	scanf("%d%d%d",&n,&q,&t);
	for(i=1;i<=n;++i) scanf("%d",&a[i]);
	for(i=0;i<q;++i){
		scanf("%d%d",&b,&c);
		son[b]=c,far[c]=b;
	}
	cnt=0;
	for(i=1;i<=n;++i)
	if(son[i]==0) dfs(i,0);
	if(cnt==n && t>=0){
		f[0]=1;
		for(i=1;i<=n;++i)
		for(j=a[i];j<=t;++j){
			f[j]=(f[j]+f[j-a[i]])%MOD;
		}
		printf("%d",f[t]);
	}
	else{
		printf("0");
	}
	return 0;
}
