#include <iostream>
using namespace std;
typedef long long int ll;
ll dp[100001]={};			//dp[j]:(i�Ԗڂ܂łŁ��ė��p)j�����g�ݍ��킹�̐�
int main(){
	int n,q,a[301]={},Gto[301]={},Gfrom[301]={};
	ll t;
	cin >> n >> q >> t;
	for(int i=1;i<=n;i++) cin >> a[i];			//1-indexed
	for(int i=0;i<q;i++){
		int b,c;
		cin >> b >> c;
		Gto[b]=c;
		Gfrom[c]=b;
	}
	for(int i=1;i<=n;i++){
		int hoge=i;
		while(Gto[hoge]!=0){
			hoge=Gto[hoge];
			if(hoge==i){
				cout << 0;
				return 0;
			}
		}
	}
	for(int i=1;i<=n;i++){
		if(Gfrom[i]!=0) continue;
		int hoge=i;
		int sum=0;
		while(Gto[hoge]!=0){
			sum+=a[hoge];
			a[hoge]=sum;
			t-=sum;
			hoge=Gto[hoge];
		}
		a[hoge]+=sum;
	}
	if(t<0){
		cout << 0;
		return 0;
	}
	dp[0]=1;
	for(int i=1;i<=n;i++){
		for(int j=0;j<=t;j++){
			if(j-a[i]>=0) dp[j]=(dp[j]+dp[j-a[i]])%1000000007;
		}
	}
	cout << dp[t];
	return 0;
}
