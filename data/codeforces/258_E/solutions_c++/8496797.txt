#include<cstdio>
#include<cstring>
#include<vector>
#define Lsn (x<<1)
#define Rsn (x<<1|1)
#define Mid (((L)+(R))>>1)
#define pb push_back
using namespace std;

const int Maxn=1e5+19;
typedef pair<int,int> PII;
struct Edge {int y,nxt;} E[Maxn*2];
vector<PII> In[Maxn],Out[Maxn];
int Last[Maxn],Ecnt;
int n,Q,x,y,tot,v,Ans[Maxn];
int in[Maxn],out[Maxn],Dis[Maxn];
int cnt[Maxn*4],add[Maxn*4],Ql,Qr;

inline void Link(int x,int y)
{
	E[Ecnt]=(Edge){y,Last[x]};Last[x]=Ecnt++;
	E[Ecnt]=(Edge){x,Last[y]};Last[y]=Ecnt++;
}
inline void DFS(int x,int Fa)
{
	tot++;Dis[tot]=x;in[x]=tot;
	for (int i=Last[x];i!=-1;i=E[i].nxt) if (E[i].y!=Fa) DFS(E[i].y,x);
	out[x]=tot;
}

inline void Change(int x,int L,int R) 
{
	if (add[x]) cnt[x]=R-L+1;
		else cnt[x]=(L==R?0:cnt[Lsn]+cnt[Rsn]);
}
inline void Update(int x,int L,int R)
{
	if (Ql<=L&&R<=Qr) {add[x]+=v;Change(x,L,R);return;}
	if (Ql<=Mid) Update(Lsn,L,Mid);
	if (Qr>Mid) Update(Rsn,Mid+1,R);
	Change(x,L,R);
}

int main()
{
	memset(Last,-1,sizeof(Last));
	scanf("%d%d",&n,&Q);
	for (int i=1;i<n;i++) scanf("%d%d",&x,&y),Link(x,y);
	DFS(1,0);
	while (Q--)
	{
		scanf("%d%d",&x,&y);
		int x1=in[x],y1=out[x];
		int x2=in[y],y2=out[y];
		PII a=make_pair(x1,y1),b=make_pair(x2,y2);
		In[x1].pb(a);In[x1].pb(b);Out[y1].pb(a);Out[y1].pb(b);
		In[x2].pb(a);In[x2].pb(b);Out[y2].pb(a);Out[y2].pb(b);
	}
	for (int i=1;i<=tot;i++)
	{
		for (int j=0;j<In[i].size();j++) 
		{
			Ql=In[i][j].first;
			Qr=In[i][j].second;v=1;
			Update(1,1,tot);
		}
		Ans[Dis[i]]=max(cnt[1]-1,0);
		for (int j=0;j<Out[i].size();j++)
		{
			Ql=Out[i][j].first;
			Qr=Out[i][j].second;v=-1;
			Update(1,1,tot);
		}
	}
	for (int i=1;i<=n;i++) printf("%d ",Ans[i]);puts("");
}
