#include<cstdio>
#include<algorithm>
#include<iostream>
#include<vector>
using namespace std;
typedef long long LL;
int n,m,p,tot,Div[100000],f[100000],ans,b,a[100000];
LL modpow(LL x,LL y){
	LL res=1;
	for(;y;y>>=1,(x*=x)%=p)
		if(y&1)(res*=x)%=p;
	return res;
}
int main(){
	cin>>n>>m>>p;
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1,x;i<=m;i++)scanf("%d",&x),b=b?__gcd(b,x):x;
	for(int i=1;i*i<p;i++)
		if((p-1)%i==0){
			Div[tot++]=i;
			if(i*i!=p-1)
				Div[tot++]=(p-1)/i;
		}
	sort(Div,Div+tot);
	for(int i=1;i<=n;i++){
		a[i]=modpow(a[i],b);
		for(int j=0;j<tot;j++)
			if(modpow(a[i],Div[j])==1){
				a[i]=(p-1)/Div[j];break;
			}
	}sort(a+1,a+1+n);
	for(int i=n;i>=1;i--)
		if(a[i]!=a[i+1]){
			f[i]=(p-1)/a[i];
			for(int j=i+1;j<=n;j++)
				if(a[j]!=a[j-1]&&a[j]%a[i]==0)
					f[i]-=f[j];
			ans+=f[i];
		}else f[i]=f[i+1];
	cout<<ans<<endl;
}
