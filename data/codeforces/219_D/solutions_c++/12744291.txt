#include<bits/stdc++.h>

using namespace std;

vector < pair < int , bool > > V[200005];

int badways[200005],n,u,v,i,mn = 200005;

bool H[200005],h;

void f(int x){

    H[x] = 1 - h;

    int i = 0;

    for(;i < V[x].size(); i++)
        if(H[ V[x][i].first ] == h){

            if(h) badways[ V[x][i].first ] = badways[x] - ( V[x][i].second ? 1 : -1 );

            else  badways[1] += V[x][i].second;

            f(V[x][i].first);

        }
    if(h && mn > badways[x])
        mn = badways[x];
}
int main(){

    cin >> n;

    for(i=n;--i;){
        cin >> u >> v;
        V[u].push_back( make_pair(v,0) );
        V[v].push_back( make_pair(u,1) );
    }

    f(1);
    h=1;
    f(1);

    cout << mn << endl;

    for(i=1;i<=n;i++)
        if(badways[i] == mn)
           cout << i << " ";

    return 0;
}
