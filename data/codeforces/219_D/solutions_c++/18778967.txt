#include <iostream>
#include <cstdio>
const int N=200010;
int f[N],g[N],to[N<<1],v[N<<1],ne[N<<1],fi[N],
	i,n,x,y,ans,tot=0;
void add(int x,int y,int z)
{
	to[++tot]=y;v[tot]=z;ne[tot]=fi[x];fi[x]=tot;
}
void dfs(int x)
{
	for (int i=fi[x];i;i=ne[i])
		if (to[i]!=f[x])
		{
			f[to[i]]=x;
			dfs(to[i]);
			g[x]+=g[to[i]]+v[i];
		}
}
void dfs2(int x)
{
	ans=std::min(ans,g[x]);
	for (int i=fi[x];i;i=ne[i])
		if (to[i]!=f[x]) g[to[i]]=g[x]+(v[i]?-1:1),dfs2(to[i]);
}
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%d\n",&n);ans=n;
	for (i=1;i<n;++i)
	{
		scanf("%d%d\n",&x,&y);
		add(x,y,0);add(y,x,1);
	}
	dfs(1);
	dfs2(1);
	printf("%d\n",ans);
	for (i=1;i<=n;++i)
		if (g[i]==ans) printf("%d ",i);
}
