#include<bits/stdc++.h>
using namespace std;

#define pb push_back

int n,s,mn=INT_MAX;

int DP[200005];

vector<int> V[200005],TV[200005];

void dfs(int x,int root){

	int i;

	for(i=0 ; i<V[x].size() ; i++)
		if(V[x][i]!=root)
			dfs(V[x][i],x);

	for(i=0 ; i<TV[x].size() ; i++)
		if(TV[x][i]!=root){
			dfs(TV[x][i],x);
			s++;
		}

}

void dfs2(int x,int val,int root){

	int i;

	DP[x]=val;

	mn=min(mn,val);

	for(i=0 ; i<V[x].size() ; i++)
		if(V[x][i]!=root)
			dfs2(V[x][i],val+1,x);

	for(i=0 ; i<TV[x].size() ; i++)
		if(TV[x][i]!=root)
			dfs2(TV[x][i],val-1,x);

}

int main(){
	
	int a,b,i;

	scanf("%d",&n);

	for(i=1 ; i<=n-1 ; i++){

		scanf("%d %d",&a,&b);

		V[a].pb(b);
		TV[b].pb(a);

	}

	dfs(1,0);

	dfs2(1,s,0);

	cout<<mn<<endl;

	for(i=1 ; i<=n ; i++)
		if(DP[i]==mn)
			cout<<i<<' ';
	
	return 0;
}
