#include<cstdio>
#include<algorithm>
#include<cstring>
#include<vector>
using namespace std;
#define MAXN 200010
int n;
#define ii pair<int,int>
#define mp make_pair
#define fr first
#define se second
#define it vector<ii>::iterator
vector<ii>g[MAXN];
int a[MAXN];
inline int Find(int x,int y){
	int nowans=0;
	for(it i=g[x].begin();i!=g[x].end();i++)
		if(i->fr!=y)
			nowans+=Find(i->fr,x)+i->se;
	return nowans;
}
inline void Go(int x,int y){
	for(it i=g[x].begin();i!=g[x].end();i++)
		if((*i).fr!=y)
			a[(*i).fr]=a[x]+(i->se?-1:1),Go(i->fr,x);
}
int main(){
	scanf("%d",&n);
	for(int i=1,x,y;i<=n-1;i++){
		scanf("%d%d",&x,&y);
		g[x].push_back(mp(y,0));
		g[y].push_back(mp(x,1));
	}a[1]=Find(1,0);Go(1,0);
	int x=*min_element(a+1,a+n+1);
	printf("%d\n",x);
	for(int i=1;i<=n;i++)
		if(a[i]==x)printf("%d ",i);
}
