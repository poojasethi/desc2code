def spinning_arrow():
    with open('input.txt') as f:
        n, k = [int(i) for i in f.readline().split()]
        a = [int(i) for i in f.readline().split()]
    while True:
        if a[k-1] == 1:
            yield k
        k = k+1 if k != n else 1

def main():
    with open('output.txt', 'w') as f:
        f.write(str(spinning_arrow().next()))

if __name__ == '__main__':
    main()
