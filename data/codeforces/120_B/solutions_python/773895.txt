def main():
	fi = open("input.txt", 'r')
	fo = open("output.txt", 'w')
	[n,k] = [int(x) for x in fi.readline().split()]
	m = [int(x) for x in fi.readline().split()]
	for i in xrange(k-1, n):
		if m[i] == 1:
			return i+1
	for i in xrange(0,k-1):
		if m[i] == 1:
			return i+1
			
fo.write(str(main()))
