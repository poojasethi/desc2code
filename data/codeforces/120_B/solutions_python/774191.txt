f = open('input.txt', 'rb')
data = f.readlines()
f.close()

z = data[0]
z = z.split()
n = int(z[0])
k = int(z[1])
z = data[1]
ttt = z.split()
mtx = []
for i in ttt:
	mtx.append(int(i))

if mtx[k-1] == 1:
	res = k
else:
	cnt = k
	while True:
		if cnt == n:
			cnt = 0
		if mtx[cnt] == 1:
			res = cnt+1
			break
		else:
			cnt += 1



#print res

f = open('output.txt', 'wb')
f.write(str(res))
f.close()
