source = open('input.txt', 'r')
result = open('output.txt', 'w')
n, k = [int(i) for i in source.readline().split()]
questions = [int(i) for i in source.readline().split()]
answer = False
for i in range(k - 1, n):
	if questions[i]:
		result.write(str(i + 1))
		answer = True
		break
if not answer:
	for i in range(k):
		if questions[i]:
			result.write(str(i + 1))
			break
result.close()
source.close()
