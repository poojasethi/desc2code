#include<iostream>
#include<stdio.h>
using namespace std;

int main()
{
	int a[10000];
	int n,m;
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d%d",&n,&m);
	m--;
	for (int i=0;i<n;i++)
		scanf("%d",&a[i]);
	while (!a[m])
	{
		m++;
		if (m>=n)m%=n;
	}
	printf("%d",m+1);
	return 0;
}
