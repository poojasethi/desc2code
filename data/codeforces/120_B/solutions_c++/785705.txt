#include<stdio.h>
#include<string.h>


int main()
{

	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);

	char cad[1010];
	int n, k, i;

	scanf("%d%d\n",&n, &k);

	for( i=0; i<n; i++ )		scanf("%d", &cad[i]);

	k--;
	
	while( cad[k]==0 )			k = (k+1)%n;
		
	printf("%d\n",k+1);
	
	

	return 0;
}
