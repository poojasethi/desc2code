#include <fstream>
#include <iostream>
#include <vector>

using namespace std;

int main() {
  ifstream fin("input.txt");
  ofstream fout("output.txt");

  int N, K;
  fin >> N >> K;
  vector<int> V(N);
  for (int i = 0; i < N; ++i) fin >> V[i];

  for (--K; V[K] == 0; (K += 1) %= N);
  fout << (K + 1) << endl;
}
