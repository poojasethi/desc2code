#include<cstdio>
using namespace std;
int a1[1000];
int main()
{
    int n,a=0,o=0,d=1;
    scanf("%d",&n);
    for(int i=0;i<n;++i)scanf("%d",a1+i);
    for(int i=0;1;i+=d)
    {
        if(a1[i]<=a)a1[i]=9999,++a;
        if(a==n)break;
        if(i+d<0||i+d==n)d=-d,++o;
    }
    printf("%d\n",o);
}
