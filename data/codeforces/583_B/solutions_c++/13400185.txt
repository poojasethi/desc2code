#include <stdio.h>
#include <algorithm>
using namespace std;
int main(){
	int A[1000],n,hv=0;
	scanf("%d",&n);
	for(int i=0;i<n;++i)
		scanf("%d",&A[i]);
	int pos = 0,dir=1;bool visit[1000];
	int ret= 0;
	for(int i=0;i<n;++i)visit[i]=false;
	while(true){
		int k = pos;
		while(k<n&&k>=0){
			if(hv>=A[k]&& !visit[k]){++hv;visit[k]=true;pos = k;}
			k+=dir;
		}
		dir*=-1;
		if(hv==n)break;

		++ret;
	}
	printf("%d\n",ret);
}
