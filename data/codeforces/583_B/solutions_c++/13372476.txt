#include <iostream>
using namespace std;
int n, a[1001], b[1001], p=1, d=1, ans, m;
int main()
{
	cin>>n;
	for (int i=1; i<=n; i++) cin>>a[i];
	while (m<n) {
		for (int i=p; i<=n&&i>=1; i+=d) {
			if (b[i]==0&&a[i]<=m) {
				b[i]=1; p=i; m++;
			}
		}
		d=-d;
		if (m<n) ans++;
	}
	cout<<ans<<endl;
	return 0;
}
