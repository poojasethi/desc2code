#include<cstdio>
int v[1005],a[1005];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
         scanf("%d",&a[i]);

    int c=0,i=-1,j=0;
    while(c<n)
    {
        j++;
        for((j%2)?i++:i-- ;(j%2)?i<n:i>=0 ;(j%2)?i++:i--)
            if(!v[i] && a[i]<=c)
            {
                v[i]=1;
                c++;
            }
    }
    printf("%d\n",j-1);
}
