#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i;
	cin>>n;
	int a[n+1];
	for(i=1;i<=n;i++) cin>>a[i];
	int ans,cnt;
	ans=cnt=0;
	while(1)
	{
		for(i=1;i<=n;i++)
		{
			if(a[i]<=cnt) { a[i]=n+1; cnt++; }
		}
		if(cnt<n)
		{
			ans++;
			for(i=n;i>0;i--)
			{
				if(a[i]<=cnt) { a[i]=n+1;cnt++; }
			}
		}
		if(cnt==n) break;
		else ans++;
	}
	cout<<ans;
	return 0;
}
