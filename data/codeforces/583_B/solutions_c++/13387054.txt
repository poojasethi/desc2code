#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n], i;
	for(i=0;i<n;i++)
		cin>>a[i];
	int cnt=0, turns=0;
	while(true){
		for(i=0;i<n;i++)
			if(a[i]<=cnt)
				a[i]=n+1, ++cnt;
		if(cnt==n)
			break;
		++turns;
		for(i=n-1;i>=0;i--)
            if(a[i]<=cnt)
				a[i]=n+1, ++cnt;
		if(cnt==n)
			break;
		++turns;
	}
	cout<<turns;
	return 0;
}
