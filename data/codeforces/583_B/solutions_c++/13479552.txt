#include<iostream>
using namespace std;
const int N=1005;
int a[N],hacked[N],cnt=0,n,ans=0;
int main(){
	cin>>n;
	for(int i=1;i<=n;i++)cin>>a[i];
	for(int p=1,di=1;cnt<n;di=-di,ans++){
		for(;p<=n&&p>=1;p+=di){
			if(!hacked[p]&&cnt>=a[p]){
				hacked[p]=1;
				cnt++;
			}
		}
		p-=di;
	}
	cout<<ans-1<<endl;
}
