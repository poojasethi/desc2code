#include<bits/stdc++.h>

using namespace std;

int i,j,k,a[1001];

int main(void)
{
	int n,s=0,ans=0;
	cin>>n;
	for(i=0;i<n;i++)cin>>a[i];
	while(1)
	{
		for(i=0;i<n;i++)
		{
			if(a[i]<=s)a[i]=n+1,s++;
		}
		ans++;
		if(s==n)break;
		for(i=n-1;i>=0;i--)
		{
			if(a[i]<=s)a[i]=n+1,s++;
		}
		ans++;
		if(s==n)break;
	}
	cout<<ans-1;

	return 0;
}
