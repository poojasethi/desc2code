#include<bits/stdc++.h>
using namespace std;
int main()
{
int n;
scanf("%d",&n);
int a[n+1];
for(int i=1;i<=n;i++)
scanf("%d",&a[i]);
int d=1,c=0,move=0;
for(int i=1;;i=i+d)
{
if(a[i]<=c)
{
a[i]=n+1;
c++;
}
if(c==n)
break;
else if(i==n)
{
move++;
d=-1;
}
else if(i==1&&d==-1)
{
move++;
d=1;
}
}
cout<<move;
return 0;
}
