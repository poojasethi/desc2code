#include <iostream>
#include <vector>
#include <algorithm>
#include <utility>
#include <cmath>

using namespace std;

int main(){

	int n;
	cin >> n;
	int a[n], ans=0, dir=1, u=0;
	for (int i = 0 ; i < n ; i++){
		cin >> a[i];
	}
	for (int i = 0 ; u < n ;){
		if (a[i] <= u) {
			a[i] = 1e9;
			u++;
		}
		if (i+dir >= 0 && i+dir < n) {
			i+=dir;
		} else if (u < n){
			ans++;
			dir = -dir;
		}
	}
	cout << ans << endl;
	return 0;
}
