#include<iostream>
using namespace std;
int a[1001];
int main(){
    int n,i,m=0,count=0;cin>>n;
    for(i=0;i<n;i++)cin>>a[i];
    while(1){
        for(i=0;i<n;i++)if(m>=a[i]){m++;a[i]=n+1;}
        if(m==n)break;count++;
        for(i=n-1;i>=0;i--)if(m>=a[i]){m++;a[i]=n+1;}
        if(m==n)break;count++;
    }
    cout<<count<<endl;
    return 0;
}
