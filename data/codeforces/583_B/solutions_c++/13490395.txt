#include <bits/stdc++.h>
using namespace std;

int n,ct,now,a[1005],turn,all;
bool dir,ok[1005];

int main(){
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> n;
	for (int i=1;i<=n;i++) cin >> a[i];
	ct=1; now=0; dir=1; all=0;
	while (true){
		if (!ok[ct]&&now>=a[ct]){
			ok[ct]=true; now++; all++;
		}
		if (dir&&ct==n||!dir&&ct==1){
			if (all==n) break;
			dir=!dir; turn++;
		}
		if (dir) ct++; else ct--;
	}
	cout << turn << "\n";
}
