#include <bits/stdc++.h>
using namespace std;
int main()
{
	int i,n;
	cin >> n;
	int a[n];
	for(i=0;i<n;i++) scanf("%d",&a[i]);
	int ans = 0;
	int j = 0; //index
	int k = 1; // adder
	int count = 0 ;
	while(count!=n)
	{
		if(a[j]<= count)
		{
			count++;
			a[j]=1001;
			if(count==n) break;
		}
		if((j==n-1 and k==1) || (j==0 and k==-1))
		{
			ans++;
			k*=-1;
		}
		j+=k;
	}
	cout << ans << endl;
	return 0;
}
