#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n],now=0,ans=0;
	for(int i=0;i<n;i++)cin>>a[i];
	while(1)
	{
		for(int i=0;i<n;i++)if(a[i]<=now&&a[i]!=-1)a[i]=-1,now++;
		if(now==n)break;
		ans++;
		for(int i=n-1;i>=0;i--)if(a[i]<=now&&a[i]!=-1)a[i]=-1,now++;
		if(now==n)break;
		ans++;
	}
	cout<<ans<<endl;
}
