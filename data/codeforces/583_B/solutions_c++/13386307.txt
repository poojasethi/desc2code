#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
int a[1100];
bool b[1100];
int main(){
int n;cin>>n;
for (int i=0;i<n;i++) {cin>>a[i]; b[i]=true;}
int c=0, dir=1, i=0, k=0;
while (c<n)
{
if (a[i]<=c&&b[i]==true) {b[i]=false; c++;  }
if (c!=n){
	
	if (i==n-1) {k++; dir=-1; }
	
	i+=dir;
	if (a[i]<=c&&b[i]==true) {b[i]=false; c++;  }
	if (i==0&&c!=n) {dir=1; k++;}
	}  
}
cout<<k;}
