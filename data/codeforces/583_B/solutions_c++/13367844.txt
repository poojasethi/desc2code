#include<bits/stdc++.h>
#define maxi (int)1e3+9
using namespace std;
int main(){
	int n,dir=-1,a[1005],ans=0,c=0,r=0;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i];
	while(c<n){
		dir++;
		if(dir%2==0)
			for(int i=0;i<n;i++){
				if(a[i]<=c && a[i]!=-1){
					c++;
					a[i]=-1;
				}
			}
		else
			for(int i=n-1;i>=0;i--)
				if(a[i]<=c && a[i]!=-1){
					c++;
					a[i]=-1;
				}
	}
	cout<<dir<<endl;
	return 0;
}
