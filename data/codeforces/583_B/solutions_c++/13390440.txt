#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin>>n;
	vector<int> a(n);
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int cnt=0;
	int res=0;
	while(cnt<n)
	{
		res++;
		for(int i=0;i<n;i++)
		{
			if(a[i]>=0 && a[i]<=cnt)
			{
				cnt++;
				a[i]=-1;
			}
		}
		reverse(a.begin(),a.end());
	}
	cout<<res-1<<endl;
}
