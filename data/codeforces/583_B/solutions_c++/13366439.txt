#include <bits/stdc++.h>

using namespace std;

int	n,t,Ans,a[1100];
bool	visited[1100];

int main()
{
	int	i;

	cin >> n;
	for(i=1;i<=n;++i)
	{
		cin >> a[i];
	}

	while(t<=n)
	{
		for(i=1;i<=n;++i)
		{
			if(!visited[i]&&a[i]<=t)
				t++,visited[i]=true;
		}
		if(t>=n)break;
		Ans++;
		for(i=n;i>=1;--i)
		{
			if(!visited[i]&&a[i]<=t)
				t++,visited[i]=true;
		}
		if(t>=n)break;
		Ans++;
	}

	cout << Ans << endl;

	return 0;
}
