#include <iostream>
using namespace std;

int main(){
	short n,a[1001];
	cin >> n;
	for (int i=1; i<=n; i++)
		cin >> a[i];
	a[0]=a[n+1]=-1;
	short i=1,r=1,d=1,change=0,have=0;
	while (r<=n){
		if (i>n){
			d=-1;
			change++;
		}
		else
		if (i<1){
			d=1;
			change++;
		}
		
		if (a[i]<=have && a[i]!=-1){
			have++;
			a[i]=-1;
			r++;
		}		
		i += d;	
	}
	
	cout << change;

	return 0;
}
