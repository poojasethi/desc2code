#include <stdio.h>
#include<iostream>
using namespace std;
int a[10001];
int main(){
	int n,p=0,c=0;
		cin>>n;
	a[-1] = 999999;
	a[n] = 999999;

	for(int i = 0 ; i < n ; i ++) cin>>a[i];
	int i = -1;
	while(p<n){
		if(c%2==0) i ++;
		else i--;
		if(p>=a[i]){
			p++;
			a[i] = 99999;
		}
		if(i==-1||i==n) c++;
	}
	cout<<c;
}
