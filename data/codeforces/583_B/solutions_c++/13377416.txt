#include<iostream>
using namespace std;

int main(){
	int n=0;
	cin>>n;
	int info_count=0,turn_count=0;
	int a[n],tmp=0;
	for(int i=0;i<n;++i){cin>>tmp;a[i]=tmp;}
	int i=0,dir=1;
	while(info_count!=n){
		if(i==n || i==-1){
			++turn_count;
			dir*=-1;
			i+=dir;
		}
		if(a[i]!=-1 && a[i]<=info_count){
			a[i]=-1;
			++info_count;
			i+=dir;
		}else{
			i+=dir;
		}
	}
	cout<<turn_count;
	return 0;
}
