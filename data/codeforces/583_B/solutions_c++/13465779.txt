#include<iostream>
using namespace std;
int main()
{
int n,i,m=0,count=0;
cin>>n;
int a[n];
for(i=0;i<n;i++)
cin>>a[i];
while(1)
{
for(i=0;i<n;i++)
if(m>=a[i])
{
m++;
a[i]=n+1;


}
if(m==n)
break;
count++;
for(i=n-1;i>=0;i--)
if(m>=a[i])
{
m++;
a[i]=n+1;
}
if(m==n)
break;
count++;
}
cout<<count;
return 0;
}
