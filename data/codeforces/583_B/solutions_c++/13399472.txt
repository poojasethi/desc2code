#include <iostream>
using namespace std;

int main(){
	int n; cin >>n; int a[n];
	for (int i=0;i<n;i++) cin >> a[i];
	int cn = 0; int ans = 0;
	while (cn != n){
		ans++;
		for (int i=0;i<n;i++)
			if (a[i] <= cn){
				cn++;
				a[i] = n+1;
			}
		if (cn == n) break;
		ans++;
		for (int i=n-1;i>=0;i--)
			if (a[i] <= cn){
				cn++;
				a[i] = n+1;
			}
	}
	cout << --ans << endl;
}
