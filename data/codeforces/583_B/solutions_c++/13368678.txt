#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int a[1100];
int main()
{
	int n,i,j,ans=0,dir=1;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	scanf("%d",&a[i]);
	i=1;
	j=0;
	while(j<n)
	{
		if(a[i]!=-1&&a[i]<=j)
		{
			j++;
			a[i]=-1;
			
		}
		if(j==n)
		break;
		if(i+dir>n||i+dir<1)
		{
			dir=-dir;
			ans++;
		}
		i+=dir;
	}
	printf("%d",ans);
	return 0;
 }
