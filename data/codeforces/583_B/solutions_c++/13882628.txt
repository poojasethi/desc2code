#include <bits/stdc++.h>

using namespace std;

int main ()
{
    int i, j, c, n;
    cin >> n;
    int v[1009];
    c = 0;
    for (i = 0; i < n; ++i)
        scanf("%d", &v[i]);
    for (i = 0; ; ++i)
    {
        j = i % (n*2);
        if (j >= n)j = (2*n) - j - 1;
        if (v[j] > -1 && c >= v[j])++c , v[j] = -1;
        if (c == n)break;
    }
    printf("%d", i/n);
    return 0;
}
