#include<iostream>
using namespace std;

int a[1000];
bool b[1000];
int n,ans,parts;
bool r;

int main()
{
	ios_base::sync_with_stdio(0);

	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i];

	while(parts<n)
	{
		for(int i=r?n-1:0;(r)?(i>=0):(i<n);r?i--:i++)
			if(!b[i]&&a[i]<=parts)
			{
				parts++;
				b[i]=true;
			}
		if(parts<n) ans++;
		r=!r;
	}

	cout<<ans<<endl;

	return 0;
}
