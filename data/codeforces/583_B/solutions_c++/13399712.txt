#include<bits/stdc++.h>
int a[1001],n,s,res,i;             
void go(){
  if(s==n){printf("%d",res); exit(0);}
  res++;
}
int main(){
  scanf("%d",&n);
  for(i=1;i<=n;i++) scanf("%d",a+i);
  while(1){        
    for(i=1;i<=n;i++) if(a[i]<=s) a[i]=n+1,s++; go();
    for(i=n;i>=1;i--) if(a[i]<=s) a[i]=n+1,s++; go();
  }
}
