#include <iostream>
#include<algorithm>
using namespace std;
#define ll long long
#define MAX 200005
ll b[MAX];
ll i,p,n,ans,r;

int main()
{
      ll n;cin>>n;
      ll d=1;
      ll a[n];
      for(i=0;i<n;i++)cin>>a[i];
      while(r<n){
          for(i=p;i<n && i>=0; i+=d){if(b[i]==0 && r>=a[i]) b[i]=1,r++,p=i;}
          d=-d;
          if(r<n) ans++;
      }
      cout<<ans<<endl;
      return 0;
}
