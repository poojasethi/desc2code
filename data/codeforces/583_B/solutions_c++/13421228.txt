#include <stdio.h>
int sec[1100],n;
main()
{
	int data,ans=0,i,j,k;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&sec[i]);
	data=0;
	i=0;
	k=1;
	while(data<n)
	{
		while(data<n&&i<n&&i>=0)
		{
			// printf("%d: %d\n",i,sec[i]);
			if(data>=sec[i])
			{
				// printf("get\n");
				sec[i]=1000000;
				data++;
			}
			i+=k;
		}
		ans++;
		k*=-1;
		i+=k;
	}
	printf("%d",ans-1);
	return 0;
}
