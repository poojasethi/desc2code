#include<iostream>
#include<cstdio>
using namespace std;

const int maxn = 1005;
int a[maxn],v[maxn],cur,ans,d=1,n;
int main(){
	scanf("%d", &n);
	for(int i=0; i<n; i++) scanf("%d", a+i);
	for(int i=0; cur<n; i+=d){
		if(i==-1 || i==n){
			d = -d; ans++;
		}else if(a[i]<=cur && !v[i]){
			cur++; v[i]=1;
		}
	}
	printf("%d\n", ans);	
	return 0;
}
