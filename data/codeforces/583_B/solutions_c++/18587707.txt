#include<bits/stdc++.h>

using namespace std;

int a[1000];

int main()
{
	int n,s=0,k=-1;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	while(s<n)
	{
		k++;
		for(int i=1;i<=n;i++)
		{
			if(s>=a[i] && a[i]>=0)
			{
				s++;
				a[i]=-1;
			}
		}
		if(s>=n)
		{
			break;
		}
		k++;
		for(int i=n;i>=1;i--)
		{
			if(s>=a[i] && a[i]>=0)
			{
				s++;
				a[i]=-1;
			}
		}
	}
	cout<<k;
}
