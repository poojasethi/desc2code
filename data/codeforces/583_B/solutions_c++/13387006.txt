#include <cstdio>

using namespace std;

int a[1001],done[1001];
int numdone,numdir;

int main(void) {
	int n,t;
	scanf("%d",&n);
	
	for(int i=0;i<n;++i)
		scanf("%d",&a[i]);
	
	numdir=-1;
	while(numdone<n) {
		numdir++;
		for(int i = (numdir%2)?n-1:0; i>=0 && i<n; (numdir%2)?--i:++i) {
			if(done[i]) continue;
			if(a[i]<=numdone) {
				numdone++;
				done[i]=1;
			}
		}
	}
	printf("%d",numdir);
	return 0;
}
