#include <algorithm>
#include <stdio.h>
#define	 maxn	   1003
using	 namespace std;

int n,h,ans;
int ar[maxn];
bool ok[maxn];

int main(){
	scanf("%d",&n);
	for(int i=1 ; i<=n ; i++)
		scanf("%d",&ar[i]);
	for(int i=1,dir=1 ; h<n ; i+=dir){
		if(!i || i==n+1)
			ans++,dir*=-1;
		if(i>=1 && i<=n && h>=ar[i] && !ok[i])
			ok[i]=true,h++;
	}
	printf("%d\n",ans);
	return 0;
}
