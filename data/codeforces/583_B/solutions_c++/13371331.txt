#include<iostream>
using namespace std;
int main(){
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	int done=0;
	int turn=0;
	int i=0;
	while(done!=n){
		//cout<<i<<endl;
		if(a[i]!=-1&&a[i]<=done){
			done++;
			a[i]=-1;
		}
		if(turn%2==0){
			i++;
		}else{
			i--;
		}
		if(i>=n&&done<n){
			i--;
			turn++;
		}
		if(i<0&&done<n){
			i++;
			turn++;
		}
	}
	cout<<turn<<endl;
}
