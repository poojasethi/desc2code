n = input()
nums = [int(i) for i in raw_input().split()]
n1 = nums[:]
no = 0
dr = 1
ans1=0
x = 0
tot = 0
while(tot!=n):
        if(x==n):
                x-=1
                dr*=-1
                ans1+=1
        if(x==-1):
                x=0
                dr*=-1
                ans1+=1
        if(nums[x]<=no and nums[x]!=-1):
                tot+=1
                no+=1
                nums[x]=-1
        x+=dr
print ans1
