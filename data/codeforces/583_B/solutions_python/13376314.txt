from sys import stdin, stdout

n = int(stdin.readline())
a = [int(s) for s in stdin.readline().split(" ")]

turns = 0
parts = 0
x = 1

while parts < n:
    if x > 0:
        for i in range(0, n, x):
            if a[i] <= parts:
                parts += 1
                a[i] = 999999
    else:
        for i in range(n-1, -1, x):
            if a[i] <= parts:
                parts += 1
                a[i] = 999999
    x = -x
    turns += 1
print turns - 1
