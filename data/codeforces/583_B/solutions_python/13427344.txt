n = input()
A = map(int, raw_input().split())
B = []
for k in range(n):
    B.append(0)
t = 0
T = 0
while sum(B)!= n :
    for i in range(0,n):
        if A[i] <= t and B[i] == 0:
            t = t + 1
            B[i] = 1
    if sum(B)!= n:
        A.reverse()
        B.reverse()
        T = T + 1
        
print T
