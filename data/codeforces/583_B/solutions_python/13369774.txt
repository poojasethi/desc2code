n=input()
l=map(int, raw_input().split())
c=0
m=[0 for i in xrange(n)]
z=-1
while c<n:
    z+=1
    for i in xrange(n):
        if l[i]<=c and m[i]==0:
            c+=1
            m[i]=1
    if c<n:
        z+=1
        for i in xrange(n-1,-1,-1):
            if l[i]<=c and m[i]==0:
                c+=1
                m[i]=1
print z
