n = input()
A = map(int, raw_input().split())

i = 0
collected = 0
direction = 1
change = 0
while True:
	if A[i] <= collected and A[i] != -1:
		collected += 1
		#print A[i], "collected"
		A[i] = -1
		if collected == n:
			break

	i += direction

	if i >= n:
		#print "change"
		direction = -1
		change += 1
		i -= 1
	if i < 0:
		#print "change"
		direction = 1
		change += 1
		i += 1

print change
