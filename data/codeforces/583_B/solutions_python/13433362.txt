n = input()
A = map(int, raw_input().split())
cracked = [False] * n
infos = 0
dir_changes = 0
direction = 1
i = 0
while infos < n:
  if infos >= A[i] and not cracked[i]:
    cracked[i] = True
    infos += 1
    continue
  i += direction
  if i == n or i < 0:
    i = i - (2 * direction)
    direction = -direction
    dir_changes += 1
print dir_changes
