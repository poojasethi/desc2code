n = int(raw_input())

a = map(int, raw_input().split())
ar = a[::-1]
col = [False]*n

collected = 0
forward = True
passes = 0
while collected < n:
    if forward:
        for i, j in enumerate(a):
            if collected >= j and not col[i]:
                col[i] = True
                collected += 1
    else:
        for i, j in enumerate(ar):
            if collected >= j and not col[n-i-1]:
                col[n-i-1] = True
                collected += 1
    passes += 1
    forward = not forward

print passes - 1
