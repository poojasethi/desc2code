n=int(raw_input())
a=map(int,raw_input().split(' '))
ans=0
x=0
while(x<n):
    ans+=1
    for i in range(n):
        if(a[i]<=x)&(a[i]!=-1):
            a[i]=-1
            x+=1
    if(x==n):
        break
    else:
        ans+=1
        i=n-1
        while(i>=0):
            if(a[i]<=x)&(a[i]!=-1):
                a[i]=-1
                x+=1
            i-=1
print ans-1
