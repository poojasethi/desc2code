n = input()
a = map(int, raw_input().split())
c = 0
count = 0
while c != n:
	for i in range(n):
		if a[i] != -1 and a[i] <= c:
			a[i] = -1
			c += 1
	if c == n:
		break
	count+=1
	for i in range(n-1, -1, -1):
		if a[i] != -1 and a[i] <= c:
			a[i] = -1
			c += 1 
	if c == n:
		break
	count += 1
print count
