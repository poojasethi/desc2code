n=int(raw_input())
ar=map(int,raw_input().split())
vis=[0]*n
ctr=0
cur=0
pos=0
dir=1
while sum(vis)!=n:
	if dir==1:
		while(pos!=n):
			if(cur>=ar[pos] and vis[pos]==0):
				cur+=1
				vis[pos]=1
			pos+=1
		pos=n-1
		dir=-1
		ctr+=1
	else:
		while(pos!=-1):
			if(cur>=ar[pos] and vis[pos]==0):
				cur+=1
				vis[pos]=1
			pos-=1
		pos=0
		dir=1
		ctr+=1
print ctr-1
