take=lambda:raw_input().strip()
N=int(take())
arr=list(map(int,take().split()))

collected,turns,falgs=0,0,[True for i in xrange(N)]
while collected<N:
	turns+=1
	for i in xrange(0,N):
		if arr[i]<=collected and falgs[i]:
			collected+=1
			falgs[i]=False
	if collected<N:
		turns+=1
		for i in xrange(0,N):
			if arr[N-1-i]<=collected and falgs[N-1-i]:
				collected+=1
				falgs[N-1-i]=False
print turns-1
