n=input()
a=map(int,raw_input().split())
b=[1 for i in range(n)]
info=0
pos=0
change=0
forw=True
while info<n:
    if pos<0:
        forw=True
        change+=1
        pos=1
    if pos==(n):
        forw=False
        change+=1
        pos=n-2
    if info>=a[pos] and b[pos]:
        info+=1
        b[pos]=0
    if forw:
        pos+=1
        
    else:
        pos-=1
        
                
print change
