import sys

n=int(sys.stdin.readline());
visited=[0]*1101
a=map(int,sys.stdin.readline().split())
i=0;
s=0;
while(s<n):
    i+=1
    for j in range(0,n):
        if(a[j]<=s and visited[j]==0):
            s+=1
            visited[j]=1
    if(s>=n):
        break
    i+=1
    for j in range(n-1,-1,-1):
        if(a[j]<=s and visited[j]==0):
            s+=1
            visited[j]=1
    if(s>=n):
        break
