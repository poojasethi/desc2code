from math import *
from Queue import *
from sys import *
from datetime import *






n = int(raw_input())
a = map(int, raw_input().split())
c = 0
res = 0
d = 1
r = 0
pos = 0
while c < n:
    while pos < n and pos > -1:
        if a[pos] <= r:
            a[pos] = n+1
            r += 1
            c += 1
        pos += d
    d = -d
    pos += d
    res += 1
print(res-1)
