n = input()
arr = map(int,raw_input().split())
collected = [False for i in range(n)]
num = 0
change = 0
while 1:
    for i in range(n):
        if arr[i]<=num  and collected[i]==False:
            num += 1
            collected[i] = True
    if num==n:
        break
    change += 1
    for i in range(n-1,-1,-1):
        if arr[i]<=num  and collected[i]==False:
            num += 1
            collected[i] = True
    if num==n:
        break
    change += 1
print change
