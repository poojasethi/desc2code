n = int(raw_input())
ans = 0
nums = map(lambda x: int(x), list(raw_input().split(" ")))
bits = 0
got = n * [False]
x = 0
i = 0
step = 1
while (True):
  if i == -1 or i == n: i = i + step
  if got[i] == False and bits >= nums[i]:
    got[i] = True
    bits = bits + 1
    x = x + 1
  if x == n : break
  i = i + step
  if i == -1 or i == n: 
    step = step * -1
    ans = ans + 1

print ans
