n=input()
l=map(int,raw_input().split())

ans=0
amount=0

while amount!=n:
       ans+=1
       for i in xrange(n):
           if l[i]<=amount:
               amount+=1
               l[i]=9999999999
       if amount!=n:
           ans+=1
           for i in xrange(n-1,-1,-1):
               if l[i]<=amount:
                   amount+=1
                   l[i]=999999999999
print ans-1
