n = input()
computers = map(int,raw_input().split())
if n==1:
    print 0
else:
    j = 0
    i = 0
    changes = -1
    no_information = 0
    x = -1
    while no_information!=n:
            
        if no_information>=computers[i]:
            no_information+=1
            computers[i]=2000
            j = i
        if i==n-1 or i==0:
            x*=-1
            if i!=j:
                i = j-x
            if no_information!=n:
                changes+=1
        i+=x

    print changes
