def GI(): return int(raw_input())
def GIS(): return map(int, raw_input().split())

def main():
    def collect(ps):
        for i, req in enumerate(seq):
            if req is None:
                continue
            if req <= ps:
                ps += 1
                seq[i] = None
        return ps
    n = GI()
    seq = GIS()
    ps = 0
    reversals = 0
    while True:
        ps = collect(ps)
        if ps == n:
            return reversals
        reversals += 1
        seq.reverse()

print main()
