n=int(input())
s=raw_input()
a=s.split(' ')
i=0
flag=[]
while i<n:
    a[i]=int(a[i])
    flag.insert(i,0)
    i+=1
result=0
i=0
t=0
k=0
while result<n:
    if (a[i]<=result) & (flag[i]==0):
        result+=1
        flag[i]=1
    if t==0:
        i+=1
    else:
        i-=1
    if i==n:
        t=1
        i=n-2
        if result<n:
            k+=1
    if i==-1:
        t=0
        i=1
        if result<n:
            k+=1
print k
