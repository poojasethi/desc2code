n=input()
l=map(int,raw_input().strip().split())
#z=[False for i in range(n)]
l=[[i,False] for i in l]
obtained=0
turned=-1
while obtained<n:
    turned+=1
    for i in l:
        if i[0]<=obtained and not i[1]:
            obtained+=1
            i[1]=True
    if obtained<n:
        turned+=1
        for i in range(n-1,-1,-1):
            if l[i][0]<=obtained and not l[i][1]:
                obtained+=1
                l[i][1]=True

print turned
