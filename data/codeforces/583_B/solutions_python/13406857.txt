n = int(raw_input())
a = tuple(int(u) for u in raw_input().split())
b = [False for i in a]
pieces = 0
turns = 0
i = None
while pieces<n:
    if i == None:
        i = 0
    elif i == n-1:
        turns+=1
        i = -2
    elif i == -n:
        turns+=1
        i = 1
    elif i<0:
        i-=1
    elif i>=0:
        i+=1
    if not b[i]:
        if pieces>=a[i]:
            pieces+=1
            b[i] = True
print turns
