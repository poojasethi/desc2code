n=int(raw_input())
a=map(int,raw_input().split(' '))
count=0
x=0

while True:
	
	if(count%2==0):
		for i in xrange(0,len(a)):
			if(a[i]<=x and a[i]!=-1):
				x+=1
				a[i]=-1
	else:
		for i in xrange(len(a)-1,-1,-1):
			if(a[i]<=x and a[i]!=-1):
				x+=1
				a[i]=-1
				
	
	if(max(a)==-1):
		break
	
	
	count+=1
print count
