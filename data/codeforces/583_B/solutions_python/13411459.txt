n = input()
l = map(int, raw_input().split())
a = 0
c = 0
d = 0
while(a != n):
    for i in range(n):
        if(l[i] <= a):
            if d == 1:
                c += 1
            d = 0
            a += 1
            l[i] = 1001
    for i in range(n - 1, -1, -1):
        if(l[i] <= a):
            if d == 0:
                c += 1
            d = 1
            a += 1
            l[i] = 1001
print c
