n = input()

cracked = [False] * n

a = map(int, raw_input().split())

count = 0
idx = 0
delta = 1
nTurn = 0

while count < n:
    if a[idx] <= count and not cracked[idx]:
        cracked[idx] = True
        count += 1
        if count >= n:
            break

    if idx == n - 1:
        nTurn += 1
        delta = -1

    idx += delta

    if idx == -1:
        idx = 0
        nTurn += 1
        delta = 1

print nTurn
