#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
a = ia()
u = [False for i in range(n)]

c, p = 1, -1
k, ans, w = 0, 0, 0
while True:
    if u[k]==False and w>=a[k]: u[k] = True; w   += 1
    if w==n: break
    
    k += c
    
    if u[k]==False and w>=a[k]: u[k] = True; w   += 1
    if w==n: break
    
    if k==0 or k==n-1:
        c, p = p, c
        ans += 1
    
print ans
