n=input()
a=map(int,raw_input().split())
ans=0
number=0
d="r"
visited=[0]*n
while number<n:
	if d=="r":
		for i in range(n):
			if a[i]<=number and not visited[i]:
				number+=1
				visited[i]=1
		if number==n:break
		d="l"
		ans+=1
	elif d=="l":
		for i in range(n-1,-1,-1):
			if a[i]<=number and not visited[i]:
				number+=1
				visited[i]=1
		if number==n:break
		d="r"
		ans+=1
print ans
