n=input()
a=map(int,raw_input().split())
cl = [True]*n
c = 0
tc = 0
inc = 1
cr = 0
while(True):
    if cl[cr] and a[cr] <= c:
        c+=1
        cl[cr]=False
    if c == n: break
    if cr == 0 and inc == -1:
        tc +=1
        inc = 1
    if cr == n-1 and inc == 1:
        tc+=1
        inc = -1
    cr += inc
print tc
