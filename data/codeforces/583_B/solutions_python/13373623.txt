n = int(raw_input().strip())
computers = map(int, raw_input().strip().split(' '))
has = 0
dir = 0
count = -1
while has < n:
    count += 1
    for i in (xrange(len(computers)) if dir == 0 else xrange(len(computers) - 1, -1, -1)):
        if computers[i] != -1 and computers[i] <= has:
            has += 1
            computers[i] = -1
    dir = 1 - dir
print count
