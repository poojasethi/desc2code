from sys import stdin
inp=stdin.readlines()
inp=iter(inp)


def solve(arr):
	uptoNow=0
	count=-1
	while True:
		if uptoNow==len(arr):
			return count
		count=count+1

		for i in range(len(arr)):
			if not arr[i]==None and arr[i] <= uptoNow:
				arr[i]=None
				uptoNow=uptoNow+1
	
		if uptoNow==len(arr):
			return count

		count=count+1
		for i in range(len(arr)-1,-1,-1):
			if not arr[i]==None and arr[i] <= uptoNow:
				arr[i]=None
				uptoNow=uptoNow+1


def main():
	num=int(inp.next())
	arr=map(int,inp.next().split())
	print solve(arr)
		
main()
