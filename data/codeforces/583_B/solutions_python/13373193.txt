n = input()
a = [int(i) for i in raw_input().split()]
done = [0]*n
ans = d = cnt = 0
i = -1
while cnt < n:
    if d == 0:
        if i == n-1:
            d = 1
            ans += 1
            i = n-2
        else:
            i += 1
    else:
        if i == 0:
            d = 0
            ans += 1
            i = 1
        else:
            i -= 1
    if done[i] == 0 and cnt >= a[i]:
        cnt += 1
        done[i] = 1
print ans
