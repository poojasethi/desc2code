n = int(raw_input())
l = raw_input().split()
inform = {}
for k in range(n):
    l[k] = int(l[k])
    inform[k] = l[k]
checkList = []
for k in range(n):
    checkList.append(k)
turns = 0
unlocked = 0
while unlocked < n:
    for i in checkList:
        if inform[i] <= unlocked:
            unlocked += 1
            inform[i] = n+10
    if unlocked < n:
        checkList.reverse()
        turns += 1
print turns
