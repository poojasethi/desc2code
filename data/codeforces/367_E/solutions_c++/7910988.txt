#include <cstdio>
#include <cstdlib>
#define P 1000000007
using namespace std;

int n,m,x;
long long f[401][401];

int main()
{
    scanf("%d%d%d",&n,&m,&x);
    if (n>m) { printf("0\n"); exit(0); }
    f[0][0]=1;
    for (int i=1;i<=m;i++)
    {
        for (int j=n;j;j--)
            for (int k=j;k>=0;k--)
                if (i==x) f[j][k]=(f[j-1][k-1]+f[j-1][k])%P;
                else f[j][k]=(f[j][k]+f[j-1][k]+(k-1>=0?f[j][k-1]+f[j-1][k-1]:0))%P;
        if (i==x) f[0][0]=0;
    }
    for (int i=1;i<=n;i++) f[n][n]=(f[n][n]*i)%P;
    printf("%I64d\n",f[n][n]);
}
