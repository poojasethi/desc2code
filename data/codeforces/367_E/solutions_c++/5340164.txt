#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 400
#define mo 1000000007
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,a,b,x;
int f[2][N][N];
inline void M(int &a,int b) {
	a+=b;
	if (a>=mo) a-=mo;
}
int main() {
	scanf("%d%d%d",&n,&m,&x);
	if (n>m) {
		printf("0\n"); return 0;
	}
	f[a=1][0][0]=1;
	For(i,1,m) {
		swap(a,b);
		For(j,0,n)For(k,0,j) f[a][j][k]=0;
		For(j,0,n)For(k,0,j) if (f[b][j][k]) {
			M(f[a][j+1][k+1],f[b][j][k]);
			if (i!=x) M(f[a][j][k],f[b][j][k]);
		}
		
		swap(a,b);
		For(j,0,n)For(k,0,j) f[a][j][k]=0;
		For(j,0,n)For(k,0,j) if (f[b][j][k]) {
			M(f[a][j][k],f[b][j][k]);
			if (k) M(f[a][j][k-1],f[b][j][k]);
		}
	}
	int64 an=f[a][n][0];
	For(i,1,n) an=an*i%mo;
	printf("%I64d\n",an);
	//for(;;);
	return 0;
}
