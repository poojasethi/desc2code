#include<cstdio>
#include<cstring>
const int maxn=333,M=1000000007;
typedef int Ar[maxn][maxn];
int n,m,x;
Ar F[2];
inline void up(int &x,const int &y)
{x+=y;if(x>=M)x-=M;}
int main()
{
	scanf("%d%d%d",&n,&m,&x);
	if(n>m)return puts("0"),0;
	F[0][0][0]=1;
	for(int i=1;i<=m;i++)
	{
		Ar &f=F[i-1&1],&g=F[i&1];
		for(int j=0;j<=n;j++)
			memset(g[j],0,n+1<<2);
		for(int j=0,k,t;j<=n;j++)
			for(k=0;k<=n;k++)
			if(t=f[j][k])
			{
				up(g[j+1][k],t);
				up(g[j][k+1],t);
				if(i!=x)
				{
					if(k)up(g[j+1][k-1],t);
					up(g[j][k],t);
				}
			}
	}
	int ans=F[m&1][n][0];
	for(int i=2;i<=n;i++)ans=(long long)ans*i%M;
	printf("%d\n",ans);
	return 0;
}
