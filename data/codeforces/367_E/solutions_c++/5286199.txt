#include<cstdio>
int dp[2][333][333]={1};
const int mod=1000000007;
void add(int& a,int& b){
    a+=b;
    if(a>=mod)a-=mod;
}
int main(){
    int i,j,n,m,k,l,x;
    scanf("%d%d%d",&n,&m,&l);
    if(n>m)return puts("0"),0;
    for(k=1;k<=m;k++)
        for(i=0;i<=n;i++)
            for(j=0;j<=i;j++){
                x=0;
                if(k!=l){
                    add(x,dp[k-1&1][i][j]);
                    if(j)add(x,dp[k-1&1][i][j-1]);
                }
                if(i){
                    add(x,dp[k-1&1][i-1][j]);
                    add(x,dp[k-1&1][i-1][j-1]);
                }
                dp[k&1][i][j]=x;
            }
    long long as=dp[k-1&1][n][n];
    for(i=1;i<=n;i++)
        as=as*i%mod;
    printf("%I64d\n",as);
    return 0;
}
