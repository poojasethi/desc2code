void add(int &x,int y)
{

    memset(f,0,sizeof(f));

    for (int i=1;i<=m;i++)
        for (int j=0;j<=n;j++)
        for (int k=0;k<=n;k++)
        {
            if (i!=K)
            add(f[i][j][k],f[i-1][j][k]);//NO
            if (j&&k)
            add(f[i][j][k],f[i-1][j-1][k-1]);//(
            if (j)
            add(f[i][j][k],f[i-1][j-1][k]);//()
            if ((i!=K))
            add(f[i][j][k],f[i-1][j][k+1]);//)
        }
    an=f[m][n][0];
    for (int i=1;i<=n;i++)
    an=(long long)an*i%pp;
    printf("%d\n",an);
    //system("pause");
    return 0;
}
