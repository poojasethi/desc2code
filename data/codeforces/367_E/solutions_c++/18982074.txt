#include <bits/stdc++.h>
using namespace std;
vector < vector < vector <long long> > > dp;
const int mod = 1000000007;

int main(int argc, char const *argv[]) {
  int n, m, x;
  cin >> n >> m >> x;
  if(n > m) {
    cout << 0 << endl;
  } else {
    dp.resize(m + 10);
    for(int i = 0; i < m+10; i++) dp[i].resize(n+10);
    for(int i = 0; i < m+10; i++) {
      for(int j = 0; j < n+10; j++) {
        dp[i][j].resize(n+10);
      }
    }
    dp[m+1][n][0] = 1;
    for(int i = m; i >= 1; i--) {
      for(int j = n; j >= 0; j--) {
        for(int k = n; k >= 0; k--) {
          if(i != x) dp[i][j][k] += dp[i+1][j][k];
          dp[i][j][k] += dp[i+1][j+1][k];
          dp[i][j][k] += dp[i+1][j+1][k+1];
          if(k > 0 and i != x) {
            dp[i][j][k] += dp[i+1][j][k-1];
          }
          dp[i][j][k] %= mod;
        }
      }
    }
    long long ans = dp[1][0][0];
    for(int i = 1; i <= n; i++) {
      ans *= i;
      ans %= mod;
    }
    cout << ans << endl;
  }
  return 0;
}
