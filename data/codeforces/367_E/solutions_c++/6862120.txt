#include<cstdio>
#include<cstdlib>
#include <cmath>
#include <cstring>
#include <algorithm>
#include<climits>
#include<utility>

using namespace std;
const int MOD=1000000007,maxn=500;
long long f[2][maxn][maxn];
int main()
{
    int n,m,x;
    scanf("%d%d%d",&n,&m,&x);
    if(n>m){puts("0");return 0;}
    f[0][0][0]=1;
    for (int i=1;i<=m;++i)
    {
        int I=i%2;
        for (int j=0;j<=i&&j<=n;++j)
            for (int k=0;k<=j;++k)
            {
                f[I][j][k]=0;
                   if(j)f[I][j][k]+=f[!I][j-1][k];
                if(j&&k)f[I][j][k]+=f[!I][j-1][k-1];
                if(i!=x)
                {
                         f[I][j][k]+=f[!I][j][k];
                    if(k)f[I][j][k]+=f[!I][j][k-1];
                }
                f[I][j][k]%=MOD;
            }
    }
    long long ans=f[m%2][n][n];
    for (int i=2;i<=n;++i)
        ans=(ans*i)%MOD;
    printf("%d\n",(int)ans);
}
