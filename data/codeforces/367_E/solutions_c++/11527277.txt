#include<cstdio>
using namespace std;
typedef long long LL;
#define mo 1000000007
LL f[2][555][555];
int n,m,X,now;
inline void renew(LL &x,LL &y)
{
	x+=y;
	if (x>=mo) x-=mo;
}
int main()
{
	//、、freopen("1.in","r",stdin);freopen("1.out","w",stdout);
	scanf("%d%d%d",&n,&m,&X);
	if (n>m) {printf("0\n");return 0;}
	now=0;
	f[0][0][0]=1;
	int i,j,k;
	for (i=0;i<m;i++)
	{
		for (j=0;j<=n;j++)
			for (k=0;k<=j;k++) f[now^1][j][k]=0;
		for (j=0;j<=n;j++)
			for (k=0;k<=j;k++) if (f[now][j][k])
			{
				renew(f[now^1][j+1][k+1],f[now][j][k]);
				renew(f[now^1][j+1][k],f[now][j][k]);
				if (i+1!=X)
				{
					renew(f[now^1][j][k],f[now][j][k]);
					renew(f[now^1][j][k+1],f[now][j][k]);
				}
			}
		now^=1;
		//for (j=0;j<=n;j++)for(k=0;k<=j;k++)printf("f[%d][%d][%d]=%I64d\n",i+1,j,k,f[now][j][k]);
	}
	LL ans=f[now][n][n];
	for (i=1;i<=n;i++) ans=ans*i%mo;
	printf("%I64d\n",ans);
	return 0;
}
