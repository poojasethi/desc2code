#include<cstdio>
#include<cstdlib>
#define mo 1000000007
#define rep(i,n) for(int i=1;i<=n;i++)
#define REP(i,n) for(int i=0;i<=n;i++)
int n,m,x,f[2][336][336],now,pre;
void update(int &a,int &b){a=(a+b)%mo;}
int main(){
	scanf("%d%d%d",&n,&m,&x);
	if(n>m){printf("0\n"); return 0;}
	f[0][0][0]=1;
	rep(i,m){
		pre=now; now^=1;
		REP(j,n) REP(k,j) f[now][j][k]=0;
		if(i==x){
			rep(j,n) REP(k,j-1) update(f[now][j][k],f[pre][j-1][k]);
			rep(j,n) rep(k,j)   update(f[now][j][k],f[pre][j-1][k-1]);
			continue;
		}
		rep(j,n) REP(k,j-1) update(f[now][j][k],f[pre][j-1][k]);
		rep(j,n) rep(k,j)   update(f[now][j][k],f[pre][j][k-1]);
		rep(j,n) rep(k,j)   update(f[now][j][k],f[pre][j-1][k-1]);
		REP(j,n) REP(k,j)   update(f[now][j][k],f[pre][j][k]);
	}
	rep(i,n) f[now][n][n]=1LL*f[now][n][n]*i%mo;
	printf("%d\n",f[now][n][n]);
}
