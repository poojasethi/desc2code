#include <bits/stdc++.h>
using namespace std;

const int mod = 1000000007;
const int N = 350;
int dp[2][N][N];

void add(int &x,int y) { x += y; if(x >= mod) x -= mod; }

int main()
{
    int n,m,r;
    cin >> n >> m >> r;
    if(n > m) {
        cout << 0 << endl;
        return 0;
    }
    dp[0][0][0] = 1;
    for(int i = 1; i <= m; i ++) {
        int flag = i & 1;
        for(int j = 0; j <= n; j ++)
            for(int k = 0; k <= n; k ++)
                dp[flag][j][k] = 0;
        for(int j = 0; j <= n; j ++) {
            for(int k = 0; k <= j; k ++) {
                add(dp[flag][j + 1][k],dp[flag ^ 1][j][k]);
                add(dp[flag][j + 1][k + 1],dp[flag ^ 1][j][k]);
                if(i != r) {
                    add(dp[flag][j][k],dp[flag ^ 1][j][k]);
                    if(k + 1 <= j) add(dp[flag][j][k + 1],dp[flag ^ 1][j][k]);
                }
            }
        }
    }
    long long ans = dp[m & 1][n][n];
    for(int i = 2; i <= n; i ++) ans = ans * i % mod;
    cout << ans << endl;
    return 0;
}
