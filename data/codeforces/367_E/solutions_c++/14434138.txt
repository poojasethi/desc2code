#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
#define V 200005
#define N 320
#define mo 1000000007
#define LL long long
#define oo (1<<30)
using namespace std;
int n,m,x,f[V][N];
int gd(int x,int y){ return x*(n+1)+y;}
int main()
{
    int i,j,k;
    scanf("%d %d %d",&n,&m,&x);
    if(n>m){
        printf("0");
        return 0;
      }
    // m,n,n
    f[0][0]=1;
    for(i=1;i<=m;i++)
        for(j=0;j<=n;j++)
            for(k=0;k<=j;k++){
                if(i!=x){
                //Nothing
                f[gd(i,j)][k]=f[gd(i-1,j)][k];
                //No add but minus
                (f[gd(i,j)][k]+=f[gd(i-1,j)][k+1])%=mo;
                  }
                //add but no minus
                if(k) (f[gd(i,j)][k]+=f[gd(i-1,j-1)][k-1])%=mo;
                //add and minus
                if(j) (f[gd(i,j)][k]+=f[gd(i-1,j-1)][k])%=mo;
              }
    for(i=1;i<=n;i++)
        f[gd(m,n)][0]=1LL*i*f[gd(m,n)][0]%mo;
    cout<<f[gd(m,n)][0];
    return 0;
}
