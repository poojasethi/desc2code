#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<vector>
using namespace std;
const int mod=1000000007;
int a[2][320][320];
int n,m,x;
void solve(){
	int i,j,k;
	if (n>m){
		printf("0\n");
		return;
	}
	
	a[0][0][0]=1;	
	int ww,w1;
	for (i=0;i<m;i++){
		for (j=0;j<=n;j++){
			for (k=0;k<=n;k++){
				if (a[0][j][k]==0) continue;
				int tmp=a[0][j][k];
				a[1][j+1][k]+=tmp;
				a[1][j+1][k]%=mod;
				
				a[1][j][k+1]+=tmp;
				a[1][j][k+1]%=mod;
				
				if (i!=x-1){
					if (k>0){
						a[1][j+1][k-1]+=tmp;
						a[1][j+1][k-1]%=mod;
					}
					a[1][j][k]+=tmp;
					a[1][j][k]%=mod;
				}
			}
		}
		for (j=0;j<=n;j++)
		for (k=0;k<=n;k++){
			a[0][j][k]=a[1][j][k];
			a[1][j][k]=0;
		}
	}			
	long long ans=a[0][n][0];
	//printf("%I64d\n",ans);
	for (i=1;i<=n;i++) ans=ans*i%mod;
	printf("%I64d\n",ans);
}
int main(){
	//freopen("a.in","r",stdin);
	scanf("%d%d%d",&n,&m,&x);
	solve();
	//while (1);
	return 0;
}
