#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#define MAXN 100005
#define MAXM 320
#define mod 1000000007
using namespace std ;
typedef long long LL ;

int N, M, X ;
int now[MAXM][MAXM], fr[MAXM][MAXM] ;
LL ans ;

void add(int& a, int b)
{
	a += b ;
	if(a > mod) a -= mod ;
}

int main()
{
	int i, j, t, cnt = 0, tmp ; 
	scanf("%d %d %d", &N, &M, &X) ;
	now[0][0] = 1 ;
	if(N > M) ans = 0 ;
	else 
	{
		for(i = 1; i <= M; i ++)
		{
			cnt ^= 1 ;
			for(j = 0; j <= N; j ++)
				for(t = 0; t <= N; t ++)
					fr[j][t] = now[j][t], now[j][t] = 0 ;
			for(j = 0; j <= N; j ++)
				for(t = j; t <= N; t ++)
					if(fr[j][t])
					{
						tmp = fr[j][t] ;
						if(i != X) add(now[j][t], fr[j][t]) ;
						if(t != N) add(now[j][t+1], fr[j][t]) ;
						if(j && i != X) add(now[j-1][t], fr[j][t]) ;
						if(j < N && t != N) add(now[j+1][t+1], fr[j][t]) ;  
					}
		}
		ans = now[0][N] ;
	}
	for(i = 1; i <= N; i ++)
		ans = ans*i%mod ;
	printf("%I64d\n", ans) ;
	//system("pause") ;
	return 0 ;
}
