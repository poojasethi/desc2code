#include <cstdio>
#include <memory.h>
using namespace std;

#define P 1000000007

int n, m, i, ans, X, j, p, dp[350][350], dp1[350][350];

int main() {
	scanf("%d%d%d", &n, &m, &X);
	if (n > m) {
		printf("%d\n", 0);
		return 0;
	}
	dp1[0][0] = 1;
	for (i = 1; i <= m; i++) {
		for (j = 0; j <= n; j++)
			for (p = 0; p <= j; p++) {
				if (i != X) {
					dp[j][p] = (dp[j][p] + dp1[j][p]) % P;
					if (p)	dp[j][p - 1] = (dp[j][p - 1] + dp1[j][p]) % P;
				}
				dp[j + 1][p] = (dp[j + 1][p] + dp1[j][p]) % P;
				dp[j + 1][p + 1] = (dp[j + 1][p + 1] + dp1[j][p]) % P;
			}
		for (j = 0; j <= n; j++)
			for (p = 0; p <= j; p++)
				dp1[j][p] = dp[j][p], dp[j][p] = 0;
	}
	ans = dp1[n][0];
	for (i = 1; i <= n; i++)	ans = 1LL * ans * i % P;
	printf("%d\n", ans);
}
