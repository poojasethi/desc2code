#define LL long long
using namespace std;
int f[2][320][320],n,m,A,X,P=7+1e9;
void read(int &x)
{
	char ch=getchar();int mark=1;for(;ch!='-'&&(ch<'0'||ch>'9');ch=getchar());if (ch=='-') mark=-1,ch=getchar();
	for(x=0;ch>='0'&&ch<='9';ch=getchar()) x=x*10+ch-48;x*=mark;
}
int main()
{
	read(n);read(m);read(X);
	if (n>m) printf("0\n");
	else
	{
		f[0][0][0]=1;
		for(int i=1;i<=m;i++)
		{
			A^=1;
			for(int j=0;j<=n;j++)
				for(int k=0;k<=n;k++) f[A][j][k]=0;
			for(int j=0;j<=n;j++)
				for(int k=0;k<=n;k++)
					if (f[A^1][j][k])
					{
						(f[A][j+1][k+1]+=f[A^1][j][k])%=P;
						(f[A][j+1][k]+=f[A^1][j][k])%=P;
						if (i!=X)
						{
							(f[A][j][k]+=f[A^1][j][k])%=P;
							if (k) (f[A][j][k-1]+=f[A^1][j][k])%=P;
						}
					}
		}
		int M=1;
		for(int i=1;i<=n;i++) M=(LL)M*i%P;
		printf("%I64d\n",(LL)M*f[A][n][0]%P);
	}
	return 0;
}
