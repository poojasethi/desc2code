#include <cstdio>
const int p=1000000007,N=400;
int n,m,x,i,j,k,ans,f[N][N];
void add(int &x,int y) {x=(0ll+x+y)%p;}
int main() {
    scanf("%d%d%d",&n,&m,&x);
    if (n>m) {printf("0\n");return 0;}
    f[0][0]=1;
    for (i=1;i<=m;i++)
        for (j=n;j>=0;j--)
            for (k=j;k>=0;k--) {
                if (i==x) f[j][k]=0;
                else if (k>0) add(f[j][k],f[j][k-1]);
                if (j>0) add(f[j][k],f[j-1][k]);
                if (j*k>0) add(f[j][k],f[j-1][k-1]);
            }
    ans=f[n][n];
    for (i=2;i<=n;i++) ans=(1ll*ans*i)%p;
    printf("%d\n",ans);
    return 0;
}
