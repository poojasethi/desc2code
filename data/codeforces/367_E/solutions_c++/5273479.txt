#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int MOD = 1000000007;
const int MaxN = 410;

int n,m,x;
int f[2][MaxN][MaxN];

int main()
{
	scanf("%d%d%d",&m,&n,&x);
	if(m>n) {puts("0");return 0;}
	f[0][0][0]=1;
	for(int i=1;i<=n;i++)
		for(int j=0;j<=i && j<=m;j++)
			for(int k=0;k<=j;k++){
				f[i&1][j][k]=0;
				if(j>=1){
					(f[i&1][j][k]+=f[(i+1)&1][j-1][k])%=MOD;
					if(k>=1) (f[i&1][j][k]+=f[(i+1)&1][j-1][k-1])%=MOD;
				}
				if(i!=x){
					(f[i&1][j][k]+=f[(i+1)&1][j][k])%=MOD;
					(f[i&1][j][k]+=f[(i+1)&1][j][k+1])%=MOD;
				}
			}
	long long ans=f[n&1][m][0];
	for(int i=1;i<=m;i++) (ans*=i)%=MOD;
	printf("%I64d\n",ans);
	return 0;
}
