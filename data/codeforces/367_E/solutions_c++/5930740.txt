#include<cstdio>
const int N=320,P=1000000007;
int f[N][N],n,m,x,L,z;
inline void add(int &x,int y){x+=y; if (x>=P) x-=P;}
int main()
{
	scanf("%d%d%d",&n,&m,&x); if (n>m) return puts("0"),0; f[0][0]=1;
	for (int i=1;i<=m;++i){
		L=i<n?i:n;
		for (int j=L;j>=0;--j) for (int k=L-j;k>=0;--k) if (z=f[j][k]){
			add(f[j+1][k],z),add(f[j][k+1],z);
			if (i==x) f[j][k]=0; else if (k) add(f[j+1][k-1],z);
		}
	}
	for (int i=1;i<=n;++i) f[n][0]=1LL*f[n][0]*i%P; printf("%d\n",f[n][0]);
	return 0;
}
