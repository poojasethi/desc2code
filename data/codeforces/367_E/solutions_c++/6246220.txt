#include <iostream>

#define MX 500
#define MOD 1000000007
#define rep(i, n) for (int i = 0; i <= n; i ++)

using namespace std;

typedef long long LL;

LL RLT;
int N, M, X;
int a[MX][MX], b[MX][MX];

int add(int &x, int y) {
	x += y;
	if (x >= MOD) x -= MOD;
}

int find() {
	if (N > M) return 0;
	
	b[0][0] = 1;
	
	int n;
	
	rep(i, M) {
		n = min(i, N);
		rep(k, n) rep(j, k) a[k][j] = 0;
		rep(k, n) rep(j, k) rep(dx, 1) rep(dy, 1) {
			if (i == X && !dx) continue;
			if (k + dx < j + dy) continue;
			if (k + dx > n) continue;
			add(a[k + dx][j + dy], b[k][j]);		 
		}
		rep(k, n) rep(j, k) b[k][j] = a[k][j];
	}
		 	
	RLT = b[N][N];
	
	rep(i, N - 1) {
		RLT *= i + 1;
		RLT %= MOD;
	}
}

int main() {
	
	cin >> N >> M >> X;
	find();
	cout << RLT << endl;
	
	return 0;
}
