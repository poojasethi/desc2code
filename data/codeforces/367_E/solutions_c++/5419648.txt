#include <stdio.h>
#define MAX 200003
#define MOD 1000000007LL

typedef long long ll;

ll dp[2][MAX];

int main()
{
    ll n,m,x,r,i,c,j,k,ans;
    scanf("%I64d%I64d%I64d",&n,&m,&x);
    if (n>m)
    {
        printf("0\n");
        return 0;
    }
    dp[0][0]=1;
    r=0;
    for (i=0;i<m;i++)
    {
        r=1-r;
        c=1-r;
        for (j=0;j<=n*(n+1)+n;j++) dp[r][j]=0;
        for (j=0;j<=i&&j<=n;j++)
        {
            for (k=0;k<=j;k++)
            {
                if (j+1<=n) dp[r][(j+1)*(n+1)+k]=(dp[r][(j+1)*(n+1)+k]+dp[c][j*(n+1)+k])%MOD;
                if (i+1!=x&&k+1<=j) dp[r][j*(n+1)+k+1]=(dp[r][j*(n+1)+k+1]+dp[c][j*(n+1)+k])%MOD;
                if (j+1<=n) dp[r][(j+1)*(n+1)+k+1]=(dp[r][(j+1)*(n+1)+k+1]+dp[c][j*(n+1)+k])%MOD;
                if (i+1!=x) dp[r][j*(n+1)+k]=(dp[r][j*(n+1)+k]+dp[c][j*(n+1)+k])%MOD;
            }
        }
    }
    ans=dp[r][n*(n+1)+n];
    for (i=1;i<=n;i++) ans=(ans*i)%MOD;
    printf("%I64d\n",ans);
    return 0;
}
