#include <cstdio>
#define P 1000000007
#define add(x,y) (x+=y)%=P
int N,M,X,f[2][355][355];
int main()
{
	scanf("%d%d%d",&N,&M,&X),f[0][0][0]=1;
	if (N>M) {puts("0"); return 0;}
	int x=0,y;
	for (int i=1; y=x^1,i<=M; i++,x=y)
		for (int j=0; j<=N; j++)
			for (int k=0,t; k<=j; k++) if (t=f[x][j][k])
			{
				if (i!=X) add(f[y][j][k],t);
				if (k&&i!=X) add(f[y][j][k-1],t);
				add(f[y][j+1][k+1],t),add(f[y][j+1][k],t),f[x][j][k]=0;
			}
	int ans=f[x][N][0];
	for (int i=1; i<=N; i++) ans=1ll*i*ans%P;
	printf("%d\n",ans);
}
