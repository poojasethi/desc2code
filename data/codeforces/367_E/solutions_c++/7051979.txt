#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
const int MOD=1e9+7;
const int MAX=400;
int dp[2][MAX][MAX];
int main(){
	int n,m,x,next;
	scanf("%d%d%d",&n,&m,&x);
	memset(dp,0,sizeof(dp));
	dp[0][0][0]=1;
	next=0;
	if(n>m) printf("0\n");else{
		for(int k=1;k<=m;k++){
			next=next^1;
			for(int i=0;i<=n;i++)
			    for(int j=0;j<=n;j++)
			         dp[next][i][j]=0;
			for(int i=0;i<=n;i++) 
			     for(int j=0;j<=i;j++){
			     				   dp[next][i+1][j]=(dp[next][i+1][j]+dp[next^1][i][j])%MOD;
			     				   dp[next][i+1][j+1]=(dp[next][i+1][j+1]+dp[next^1][i][j])%MOD;
if(k!=x){
	dp[next][i][j+1]=(dp[next][i][j+1]+dp[next^1][i][j])%MOD;
	dp[next][i][j]=(dp[next][i][j]+dp[next^1][i][j])%MOD;
}
}
}
long long ans=dp[next][n][n]%MOD;
for(int i=1;i<=n;i++) ans=(ans*i)%MOD;
printf("%lld\n",ans);
}

return 0;
}
