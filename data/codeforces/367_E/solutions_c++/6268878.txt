#include <iostream>
using namespace std;
const int N = 320, mod = 1e9 + 7;

int dp[2][N][N], n, m, x;

int main() {
	cin >> n >> m >> x;
	if (n > m) {
		cout << 0 << '\n';
		return 0;
	}
	dp[0][0][0] = 1;
	for (int i = 1; i <= m; i++) {
		for (int l = 0; l <= n; l++)
			for (int r = 0; r <= l; r++) {
				int &d = dp[i & 1][l][r] = 0;
				if (l && r) {
					d += dp[i - 1 & 1][l - 1][r - 1];
					if (d >= mod)
						d -= mod;
				}
				if (l) {
					d += dp[i - 1 & 1][l - 1][r];
					if (d >= mod)
						d -= mod;
				}
				if (i != x) {
					if (r) {
						d += dp[i - 1 & 1][l][r - 1];
						if (d >= mod)
							d -= mod;
					}
					d += dp[i - 1 & 1][l][r];
					if (d >= mod)
						d -= mod;
				}
			}
	}
	int ans = dp[m & 1][n][n];
	for (int i = 1; i <= n; i++)
		ans = 1ll * ans * i % mod;
	cout << ans << '\n';
	return 0;
}
