#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
using namespace std;
int dp[2][405][405],sum,i,j,k,n,m,x,A,B;
const int MOD=1000000007;
int main()
{
    dp[0][0][0]=1;
    cin>>n>>m>>x;
    if (n>m) {cout<<0; return 0;}
    A=0; B=1;
    for (i=1; i<=m; i++)
    {
        for (j=0; j<=n; j++)
          for (k=j; k<=n; k++)
          {
              if (i!=x) dp[B][j][k]=dp[A][j][k]; else dp[B][j][k]=0;
              if (i!=x && j) dp[B][j][k]+=dp[A][j-1][k]; dp[B][j][k]%=MOD;
              dp[B][j][k]+=dp[A][j][k-1]; dp[B][j][k]%=MOD;
              if (j) dp[B][j][k]+=dp[A][j-1][k-1]; dp[B][j][k]%=MOD;
          }
        sum=dp[B][n][n];
        swap(A,B);
    }
    for (i=2; i<=n; i++) sum=(long long)sum*i%MOD;
    cout<<sum;
    return 0;
}
