#include<iostream>
#include<cstdlib>
#include<cstdio>
#define mod 1000000007
#include<cstring>
#define LL long long
#define N 100010
using namespace std;
int n,m,x;
int f[2][320][320];

int main()
{
//	freopen("e.in","r",stdin);
	scanf("%d%d%d",&n,&m,&x);
	if (n>m) {printf("0\n");return 0;}
	
	f[0][0][0]=1;
	for (int i=1;i<=m;++i) {
//		memset(f[i%2],0,sizeof(f[i%2]));
		for (int j=0;j<=n;++j) {
			for (int k=0;k<=j;++k) {
				if (i!=x) f[i%2][j][k]+=f[(i-1)%2][j][k],f[i%2][j][k]%=mod;
				if (j) f[i%2][j][k]+=f[(i-1)%2][j-1][k],f[i%2][j][k]%=mod;
				if (i!=x&&k) f[i%2][j][k]+=f[(i-1)%2][j][k-1],f[i%2][j][k]%=mod;
				if (j&&k) f[i%2][j][k]+=f[(i-1)%2][j-1][k-1],f[i%2][j][k]%=mod;
//				printf("%d %d %d   %d\n",i,j,k,f[i][j][k]);
			}
		}
		for (int j=0;j<=n;++j) for (int k=0;k<=j;++k) f[(i-1)%2][j][k]=0;
	}
	for (int i=1;i<=n;++i) f[m%2][n][n]=(LL)f[m%2][n][n]*i%mod;
	printf("%d\n",f[m%2][n][n]);
//	printf("%d\n",(LL)ans*fac[n]%mod);
	return 0;
}
