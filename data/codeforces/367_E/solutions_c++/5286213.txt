#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <map>
#include <vector>
#include <set>
#include <cmath>
#include <cstring>
#include <string>
using namespace std;
typedef long long LL;

const int maxn = 350;
int f[maxn][maxn], g[maxn][maxn];
const int mod = 1000000007;
void safeadd(int &a, int b){
	a = a + b;
	if (a>=mod) a-=mod;
}
int main(){
	int n, m, x;
	cin>>n>>m>>x;
	if (n > m){
		cout<<0<<endl;
		return 0;
	}
	f[0][0] = 1;
	for (int t = 1; t<=m; t++){
		for (int i = 0; i<=n; i++) for (int j = 0; j<=i; j++) g[i][j] = 0;
		for (int i = 0; i<=n; i++){
			for (int j = 0; j<=i; j++){
				for (int dx = (t == x); dx<=1; dx++){
					for (int dy = 0; dy<=1 && i+dx >= j+dy; dy++){
						safeadd(g[i+dx][j+dy], f[i][j]);
					}
				}
			}
		}
		for (int i = 0; i<=n; i++) for (int j = 0; j<=i; j++)
			f[i][j] = g[i][j];
	}
	LL res = f[n][n];
	for (int i = 1; i<=n; i++){
		res = res * i % mod;
	}
	cout<<res<<endl;
	return 0;
}
