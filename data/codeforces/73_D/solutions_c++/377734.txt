#include <stdio.h>
#include <string.h>
#define maxn 1000010
int father[maxn];
int tot[maxn];
int sum[maxn];
int a[maxn],an;
int n,m,p;
int Find(int x){
	int f,next,p;
	for (p=x;father[p]!=p;p=father[p]);
	for (f=p,p=x;father[p]!=f;p=next){
		next=father[p];
		father[p]=f;
	}
	return f;
}
int main(){
	int i,j,k;
	scanf("%d %d %d",&n,&m,&p);
	for (i=1;i<=n;i++) {father[i]=i;sum[i]=1;}
	tot[1]=n;
	for (i=0;i<m;i++){
		int u,v;
		scanf("%d %d",&u,&v);
		u=Find(u);
		v=Find(v);
		if (u!=v){
			tot[sum[u]]--;
			tot[sum[v]]--;
			sum[v]+=sum[u];
			tot[sum[v]]++;
			father[u]=v;
		}
	}
	an=0;
	for (i=1;i<=n;i++)
		for (j=0;j<tot[i];j++)
			if (i>p) a[an++]=p;else a[an++]=i;
	int ans=0;
	if (p==1){
		if (an==1) ans=0;else ans=an-2;
	}else{
		int zerop=0,tot=0;
		for (i=0;i<an;i++)
			if (a[i]==1) zerop++;else tot+=a[i]-2;
		tot+=2;
		while (zerop>tot){
			zerop-=2;
			ans++;
		}
	}
	printf("%d\n",ans);
	return 0;
}
