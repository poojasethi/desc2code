#include <bits/stdc++.h>
using namespace std;

void maximize(long long &a, long long b){
	if(a<b) a=b;
}

int main(){
	//freopen("input.inp", "r", stdin);
	int n;
	scanf("%d", &n);
	int a[n+1];
	long long ans=-1e12;
	for(int i=1; i<=n; i++) scanf("%d", a+i);
	for(int i=1; i<=n; i++)
		for(int j=n; j>i; j--) if(a[i]==a[j]){
			long long t=a[i]+a[j];
			for(int k=i+1; k<j; k++) if(a[k]>0) t+=a[k];
			maximize(ans, t);
			break;
		}
	cout << ans;
	bool tt=true;
	for(int i=1; i<=n&&tt; i++)
		for(int j=n; j>i&&tt; j--) if(a[i]==a[j]){
			long long t=a[i]+a[j];
			int cnt=0;
			for(int k=i+1; k<j; k++) if(a[k]>0) t+=a[k]; else cnt++;
			if(t==ans){
				printf(" %d\n", cnt+i-1+n-j);
				for(int k=1; k<i; k++) printf("%d ", k);
				for(int k=i+1; k<j; k++) if(a[k]<=0) printf("%d ", k);
				for(int k=j+1; k<=n; k++) printf("%d ", k);
				tt=false;
			}
			break;
		} 
}
