#include<stdio.h>
int n,i,j,l,s=0,ir,jr,en=0;
long a[400000],r[400000]={0};
long long k,t;
int main()
{
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%ld",&a[i]);
    for(i=0;i<n;i++)
    {
        for(j=n-1;j>i&&a[i]!=a[j];j--);
        if(i==j) continue;
        for(t=a[i]*2,l=i+1;l<j;l++)
            if(a[l]>0) t=t+a[l];
        if(s==0||k<t)
        {s=1;k=t;ir=i;jr=j;}
    }
    for(l=ir+1;l<jr;l++)
        if(a[l]<=0) r[en++]=l;
    printf("%I64d %d\n",k,en+ir+n-jr-1);
    for(i=0;i<ir;i++)
        if(i==0) printf("%d",i+1);
        else printf(" %d",i+1);
    for(i=0;i<en;i++) if(ir==0&&i==0) printf("%d",r[i]+1);else printf(" %d",r[i]+1);
    for(i=jr+1;i<n;i++) if(en==0&&ir==0&&i==jr+1) printf("%d",i+1);else printf(" %d",i+1);
    printf("\n");

    return 0;
}
