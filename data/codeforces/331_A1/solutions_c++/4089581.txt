#include<iostream>
#include<algorithm>
#include<set>
#include<vector>
using namespace std;
const int Maxn = 101;
vector<int> v;
int a[Maxn];
int main()
{
	long long n;
	cin>>n;
	long long max = -11111111111;
	int tmp1,tmp2;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	for(int i=1;i<n;i++)
	{
		long long k=0;
		for(int j=n;j>i;j--)
		{
			if(a[j]==a[i])
			{
				for(int z=i+1;z<j;z++)
				{
					if(a[z]>0)
					{
						k+=a[z];
					}
				}
				if(k+(2*a[i])>max)
				{
					tmp1=i;
					tmp2=j;
					max=k+(2*a[i]);
					v.resize(0);
					for(int z=i+1;z<j;z++)
					{
						if(a[z]<=0)
						{
							v.push_back(z);
						}

					}
					j=i;
				}
				else
					j=i;
			}
		}
	}
	for(int i=1;i<tmp1;i++)
		v.push_back(i);
	for(int i=tmp2+1;i<=n;i++)
		v.push_back(i);
	cout<<max<<" "<<v.size()<<endl;
	for(int i=0;i<v.size();i++)
		cout<<v[i]<<" ";
}
