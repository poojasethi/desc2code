#include<cstdio>
#include<vector>
#include<fstream>
#define INF -235434
using namespace std;
int main(){
    int n;
    vector<int> v;
    scanf("%d",&n);
    int a[n],k;
    long long ans=INF;
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    for(int i=0;i<n-1;i++)
        for(int j=i+1;j<n;j++)
        if(a[i]==a[j]){
            long long s=a[i];
            if(i!=j) s+=a[i];
            for(int f=i+1;f<j;f++)
                if(a[f]>0) s+=(long long)a[f];
            if(ans==INF || ans<s){
                ans=s;
                v.resize(0);
                for(int f=0;f<i;f++) v.push_back(f+1);
                for(int f=i+1;f<j;f++)
                    if(a[f]<0) v.push_back(f+1);
                for(int f=j+1;f<n;f++) v.push_back(f+1);
            }
        }
    printf("%I64d %d\n",ans,v.size());
    for(int i=0;i<v.size();i++) printf("%d ",v[i]);
    return 0;
}
