#include <iostream>
#include <cstdio>

using namespace std;

int a[200],n;
long long ans,ansi,ansj;

long long cnt(int i,int j)
{
	long long f = 0;
	for (int k = i+1; k < j; k++)
	if (a[k] >= 0)
		f += a[k];
	f = f + a[i]+a[j];
	return f;
}

int main()
{
	scanf("%d",&n);
	for (int i = 0; i < n; i++)
		scanf("%d",&(a[i]));
	ans = -8223372036854775808;
	for (int i = 0; i < n; i++)
		for (int j = i+1; j < n; j++) if (a[i] == a[j])
		{
			long long t = cnt(i,j);
			if (ans < t)
			{
				ans = t;
				ansi = i;
				ansj = j;
			}
		}

	int a2 = 0;
	for (int i = 0; i < ansi; i++)
		a2++;
	for (int i = ansi+1; i < ansj; i++)
		if (a[i] < 0)
			a2++;
	for (int i = ansj+1; i < n; i++)
		a2++;


	cout << ans << " " << a2 << endl;
	for (int i = 0; i < ansi; i++)
		printf("%d ",i+1);
	for (int i = ansi+1; i < ansj; i++)
		if (a[i] < 0)
			printf("%d ",i+1);
	for (int i = ansj+1; i < n; i++)
		printf("%d ",i+1);
	printf("\n"); 
	return 0;
}
