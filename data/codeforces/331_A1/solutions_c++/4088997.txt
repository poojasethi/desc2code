#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>

using namespace std;

int n,a[128];
int main()
{
	cin >> n;
	for (int i=1;i<=n;++i) scanf("%d",&a[i]);
	long long ans(1LL<<60);
	ans = -ans;
	int x,y;
	for (int i=1;i<n;++i)
	{
		for (int j=i+1;j<=n;++j)
		{
			if (a[j] == a[i])
			{
				long long sum(a[i]+a[j]);
				for (int k=i+1;k<j;++k)
					if (a[k]>0) sum+=a[k];	
				if (sum > ans)
				{
					ans = sum;
					x = i;y = j;
				}
			}
		}
	}
	cout << ans << " ";
	int k(0);
	for (int i=x+1;i<y;++i)
		if (a[i]<0) ++k;
	cout << k+x-1+n-y << endl;
	for (int i=1;i<x;++i) printf("%d ",i);
	for (int i=x+1;i<y;++i) 
		if (a[i] < 0)
			printf("%d ",i);
	for (int i=y+1;i<=n;++i) printf("%d ",i);
	cout << endl;	
	return 0;
}
