#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
 __int64 i,j,n,a[102],re=0,rem[102],sumi=0,last[102]={0},maxsum=-100000000000,sum=-100000000000;
 scanf("%I64d",&n);
 for(i=1;i<=n;i++)
 {scanf("%I64d",&a[i]);
 }
 for(i=1;i<=n;i++)
 for(j=n;j>i;j--)
 {
  if(a[i]==a[j])
  {
   last[i]=j;break;                       
  }                  
 }
 for(i=1;i<=n;i++)
 {
  if(last[i]!=0)
  {sum=0;
   for(j=i+1;j<=last[i]-1;j++)
   {if(a[j]>=0) sum+=a[j];}
   sum+=a[i]+a[last[i]];           
  } 
  if(sum>maxsum) {sumi=i;maxsum=sum;}               
 }
 for(i=1;i<=n;i++)
 {
  if(i<sumi) {rem[re++]=i;}                 
 }
 for(i=1;i<=n;i++)
 {
  if(i>last[sumi]) {rem[re++]=i;}                 
 }
 for(i=1;i<=n;i++)
 {
  if(i>sumi&&i<last[sumi]&&a[i]<0) {rem[re++]=i;}                 
 }
 sort(rem,rem+re);
 printf("%I64d %I64d\n",maxsum,re);
 for(i=0;i<re;i++)
 printf("%I64d ",rem[i]);
 return 0;    
}
