#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    __int64 a[105],sum=0;
    for(int i=0;i<n;i++) {scanf("%I64d",&a[i]);sum+=a[i];}
    long long  m=-1e12,s;
    int im=-1,jm=-1;
    for(int i=0;i<=n-1;i++)
    {
        for(int j=n-1;j>=i;j--)
        {
            if(i==j) break;
            if(a[i]==a[j])
            {
                s=sum;
                for(int k=0;k<i;k++) s-=a[k];
                for(int k=n-1;k>j;k--) s-=a[k];
                for(int k=i+1;k<=j-1;k++) if(a[k]<0) s-=a[k];
                if(s>m) {m=s;im=i,jm=j;}
            }
        }
    }
    int del[105],cnt=0;
    s=sum;
    for(int k=0;k<im;k++) {s-=a[k];del[cnt++]=k+1;}
    for(int k=n-1;k>jm;k--) {s-=a[k];del[cnt++]=k+1;}
    for(int k=im+1;k<=jm-1;k++) {if(a[k]<0) {s-=a[k];del[cnt++]=k+1;}}
    printf("%I64d %d\n",s,cnt);
    for(int i=0;i<cnt;i++) printf("%d ",del[i]);
    return 0;
}
