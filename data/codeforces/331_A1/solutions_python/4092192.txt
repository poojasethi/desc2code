#include <iostream>
#include <vector>
#include <cstdio>
#include <cstring>
#include <string>

using namespace std;

int main()
{
	int n;
	vector <int> a;
	vector <int> del;
	int x,y;
	int i,j,k;
	__int64 z,sum;
	z=-1e12;
	cin>>n;
	a.resize(n+1);
	for (i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	for (i=1;i<n;i++)
	{
		for (j=n;j>i;j--)
		{
			if (a[i]!=a[j]) continue;
			sum=0;
			sum+=a[i]; sum+=a[j];
			for (k=i+1;k<j;k++)
			{
				if (a[k]>0)
				{
					sum+=a[k];
				}
			}
			if (sum>z)
			{
				z=sum;
				x=i; y=j;
			}
		}
	}
	for (i=1;i<x;i++)
	{
		del.push_back(i);
	}
	for (i=y+1;i<=n;i++)
	{
		del.push_back(i);
	}
	for (i=x+1;i<y;i++)
	{
		if (a[i]<0)
		{
			del.push_back(i);
		}
	}
	cout<<z<<" "<<del.size()<<endl;
	for (i=0;i<del.size();i++) cout<<del[i]<<" "; cout<<endl;
	return 0;
}
