#include <iostream>
#include <cstring>
#include <string>
#include <algorithm>
#include <vector>

using namespace std;
typedef long long LL;
typedef pair <LL , LL> pii;

LL n , a[100 + 10] , ted , MAX = -1000000000LL * 100LL - 100;
pair <LL , LL> ind;

int main(){
  cin >> n;
  for(LL i = 0 ; i < n ; i++)
    cin >> a[i];
  for(LL i = 0 ; i < n ; i++)
    for(LL j = i + 1 ; j < n ; j++)
      if(a[i] == a[j]){
	LL sum = 0 , t = 0;
	for(LL k = i + 1 ; k < j ; k++)
	  if(a[k] > 0)
	    sum += a[k];
	  else
	    t++;
	sum += a[i] + a[j];
	if(sum > MAX){
	  MAX = max(MAX , sum);
	  ind = pii(i , j);
	  ted = t;
	}
      }
  cout << MAX << ' ' << ted + ind.first + n - ind.second - 1 << endl;
  for(LL i = 0 ; i < ind.first ; i++)
    cout << i + 1 << ' ' ;
  for(LL i = ind.first + 1 ; i < ind.second ; i++)
    if(a[i] <= 0)
      cout << i + 1 << ' ';
  for(LL i = ind.second + 1 ; i < n ; i++)
    cout << i + 1 << ' ';
  cout << endl;
  return 0;
}
