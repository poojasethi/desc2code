int main()
{
    int n;
    cin>>n;
    int a[n],b[n];
    bool c[n];
    ll s=0,k=0,kx,maxim=MN;
    for(int i=0;i<n;i++)
        {cin>>a[i];b[i]=a[i];c[i]=true;}
    for(int i=0;i<n-1;i++)
        for(int j=i+1;j<n;j++)
         if (a[i]==a[j])
    {
        s=2*a[i];k=2;
        for(int p=i+1;p<j;p++)
        if (a[p]>=0) {s+=a[p];k++;}
        if (s>maxim) {maxim =s;kx=k;
        fill(c,c+n,true);
        c[i]=false;
        c[j]=false;
        for(int p=i+1;p<j;p++)
        if (a[p]>=0) {c[p]=false;}}
    }
    cout<<maxim<<' '<<n-kx<<endl;
    for(int i=0;i<n;i++)
        if (c[i]) cout<<i+1<<' ';
    return 0;
}
