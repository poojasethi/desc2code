#include <cstdio>
#include <algorithm>
#define LL long long 
#define maxn 105
LL sum[maxn];
int a[maxn],n;
int main()
{
    scanf("%d",&n);
    sum[0]=0;
    for(int i=1;i<=n;++i)scanf("%d",a+i);
    for(int i=1;i<=n;++i)
        if(a[i]>0)sum[i]=sum[i-1]+a[i];
        else sum[i]=sum[i-1];
    int l,r;
    LL maxx=-0x7fffffffffffLL;
    for(int i=1;i<n;++i)
        for(int j=i+1;j<=n;++j)
            if(a[i]==a[j])
            {
                if(sum[j-1]-sum[i]+a[i]+a[j]>maxx)
                {
                    maxx=sum[j-1]-sum[i]+a[i]+a[j];
                    l=i; r=j;
                }
            }
    int num=0;
    for(int i=l+1;i<r;++i)
        if(a[i]<0)++num;
    printf("%I64d %d\n",maxx,l-1+n-r+num);
    for(int i=1;i<l;++i)printf("%d ",i);
    for(int i=l+1;i<r;++i)
        if(a[i]<0)printf("%d ",i);
    for(int i=r+1;i<=n;++i)
        printf("%d ",i);
    puts("");
    return 0;
}
