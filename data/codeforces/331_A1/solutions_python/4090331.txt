#include<iostream>
#include<algorithm>
#include<string>
#include<set>
#include<vector>
using namespace std;
const int maxn=1000;
int A[maxn],B[maxn];
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>A[i];
	long long maxi=-1e11;
	vector<int> res;
	vector<int> ans;
	int ii,jj;
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
			if(A[i]==A[j])
			{
				long long sum=A[i]+A[j];
				for(int k=i+1;k<j;k++)
				{
					if(A[k]>=0)
						sum+=A[k];
					else
						ans.push_back(k);
				}
				if(sum>maxi)
				{
					maxi=sum;
					res=ans;
					ii=i;
					jj=j;
					//cout<<sum<<" "<<i<<" "<<j<<endl;
				}
				ans.resize(0);
			}
	for(int i=1;i<ii;i++)
		res.push_back(i);
	for(int i=jj+1;i<=n;i++)
		res.push_back(i);
	cout<<maxi<<" "<<res.size()<<endl;
	for(int i=0;i<(int)res.size();i++)
		cout<<res[i]<<" ";
	cout<<endl;
	return 0;
}
