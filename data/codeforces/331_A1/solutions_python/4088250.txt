#include<cstdio>
#include<cstdlib>
#include<cstring>

using namespace std;

__int64 a[205];
int n;
const __int64 inf = 1ll<<50;
int main()
{
    scanf("%d",&n);
    __int64 ans = -inf;
    for (int i= 1; i<=n; i++) scanf("%I64d",&a[i]);
    int aa ,b;
    for (int i = 1; i <= n; i++)
      for (int j = i +1 ; j<= n; j++)
        if (a[i]==a[j])
        {
          __int64 tmp = 0;
          for (int k = i+1; k < j; k++)
            if (a[k] > 0 ) tmp += a[k];
          tmp += a[i] + a[j];
          if (tmp > ans ) ans = tmp,aa = i, b= j;
        }
    int N =  aa-1 + n-b ;
    for (int i = aa+1; i< b; i++)
      if (a[i] < 0) N ++;
    printf("%I64d %d\n",ans,N);
    for (int i = 1; i < aa; i++) printf("%d ",i);
    for (int i = aa+1; i < b; i++) 
      if (a[i] < 0 )printf("%d ",i);
    for (int i = b+1; i <= n; i++)
     printf("%d ",i);
    printf("\n");
   // system("pause");
    return 0;
}
