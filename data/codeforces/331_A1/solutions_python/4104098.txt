#include<iostream>
#include<vector>
using namespace std;
typedef long long ll;
const ll INF=10000001000;
ll a[300010];
ll ans=INF,ansnum=-INF;
int ansi,ansj;
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			if (a[i]==a[j])
			{
				ll tans=n-j+i-1,tansnum=a[i]+a[j];
//				cout<<tans<<" "<<tansnum<<endl;
				for(int k=i+1;k<j;k++)
				{
					if (a[k]<0) tans++;
					else tansnum+=a[k];
				}
//				cout<<tans<<" "<<tansnum<<endl;
				if (tansnum>ansnum)
				{
					ans=tans;
					ansi=i;ansj=j;
					ansnum=tansnum;
				}
			}
		}
	}
	cout<<ansnum<<" "<<ans<<endl;
	vector<ll> v;
	for(int i=0;i<n;i++)
	{
		if (ansi<i && ansj>i && a[i]<0) v.push_back(i);
		else if (i<ansi || i>ansj) v.push_back(i);
	}
	if (v.size()==0) return 0;
	cout<<v[0]+1;
	for(int i=1;i<v.size();i++)
	{
		cout<<" "<<v[i]+1;
	}
	cout<<endl;
}
