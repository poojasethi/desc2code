#include <cstdio>
long long int S[1010];
long long int sum(int a, int b)
{
    long long int  s = S[a]+S[b], i;
    for( i = a+1 ; i < b ; i++ )
        if( 0 < S[i] ) s += S[i];
    return s;
}
int max(int a, int b){ return a>b?a:b; }
int main()
{
    int n, i, j, I=0, J=0, cnt = 0, s = 0;
    long long int ans = 0x8000000000000000LL;
    scanf("%d", &n);
    for( i = 0 ; i < n ;i ++ )
        scanf("%lld", &S[i]);
    for( i = 0 ; i < n-1 ; i++ )
        for( j = n-1 ; j > i ; j-- )
            if( S[i] == S[j] )
            {
                if( ans < sum(i,j) )
                    ans = sum(i,j), I = i, J = j;
            }
    cnt += I, cnt += n-1-J;
    for( i = I+1 ; i < J ; i++ ) if( 0 >= S[i] ) cnt++;
    printf("%lld %d\n", ans, cnt);
    for( i = 0 ; i < I ; i++ ) printf("%d ", i+1);
    for( i = I+1 ; i < J ; i++ ) if( 0 >= S[i] ) printf("%d ", i+1);
    for( i = J+1 ; i < n ; i++ ) printf("%d ", i+1);
}
