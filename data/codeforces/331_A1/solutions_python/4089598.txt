#include <cstdlib>
#include <iostream>

using namespace std;

int n;
int dsize,d[1000];
int bdsize,bd[1000],flag;
long long num,bnum,a[1000];
int main(int argc, char *argv[])
{
    cin>>n;
    for(int i=1;i<=n;i++)cin>>a[i];
    for(int i=1;i<=n;i++)
      for(int j=i+1;j<=n;j++)
        if(a[i]==a[j])
        {
          dsize=0;num=a[i]*2;
          for(int k=1;k<i;k++)d[dsize++]=k;
          for(int k=i+1;k<j;k++)
          {
            if(a[k]<0)d[dsize++]=k;
            else num=num+a[k];
          }
          for(int k=j+1;k<=n;k++)d[dsize++]=k;
          if(num>=bnum || flag==0)
          {
            flag=1;
            for(int k=0;k<dsize;k++)bd[k]=d[k];
            bnum=num;bdsize=dsize;
          }
        }
    cout<<bnum<<" "<<bdsize<<endl;
    for(int i=0;i<bdsize;i++)cout<<bd[i]<<" ";
    cout<<endl;
    return 0;
}
