#include <iostream>
#include <algorithm>
#include <cmath>
#include <vector>
#include <string>
using namespace std;

#define Size(x) ((int)x.size())
#define Pour(s, a, b) for(int s = a; s<b; s++)

const int N = 1001;
int a[N];

int main(void)
{
	int n;
	cin>>n;
	Pour(i, 0, n)
		cin>>a[i];
        
	long long maxi = -1000000000000ll;
	int imaxi;
	int jmaxi;
	long long sum;

	Pour(i, 0, n)
	{
		sum = (long long)a[i];
		Pour(j, i+1, n)
		{
			sum += (long long)a[j];
			if(a[j] == a[i] && sum > maxi)
			{
				imaxi = i;
				jmaxi = j;
				maxi = sum;
			}
			if(a[j]<0)
				sum-=(long long)a[j];
		}
	}


	int nc=0;
	Pour(i, imaxi+1, jmaxi)
	{
		if(a[i]<0)
			nc ++;
	}
	
	nc += imaxi;
	nc += n-jmaxi-1;

	//int nc2 = 0;

	cout<<maxi<<" "<<nc<<endl;
	Pour(i, 0, imaxi)
	{
		cout<<i+1<<" ";
		//nc2++;
	}
	Pour(i, imaxi+1, jmaxi)
		if(a[i]<0)
			{cout<<i+1<<" ";}
	Pour(i, jmaxi+1, n)
		{cout<<i+1<<" ";}
	cout<<endl;

	return 0;
}
