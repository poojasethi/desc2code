#include<iostream>
using namespace std;
long long  a[100],count=0;
long long f(int i,int j){
long long sum=a[i]+a[j];
for(int k=i+1;k<j;k++)
	if(a[k]>=0)
		sum+=a[k];
return sum;
}
int main(){
			int n;
			cin>>n;
			
			for(int i=0;i<n;i++)
				cin>>a[i];
			int index=0;
			long long b[1000][3];
			for(int i=0;i<n-1;i++){
				for(int j=n-1;j>i;j--){
					if(a[j]==a[i]){
						b[index][0]=i;
						b[index][1]=j;
						b[index][2]=f(i,j);
						index++;
						break;
						}
						}
						}
			int max=0;
			for(int i=0;i<index;i++)
				if(b[max][2]<b[i][2])
					max=i;
			cout<<b[max][2]<<' ';
			int l=b[max][0]+n-1-b[max][1];
			for(int i=b[max][0]+1;i<b[max][1];i++)
				if(a[i]<0)
					l++;
			cout<<l<<endl;
			for(int i=0;i<b[max][0];i++)
				cout<<i+1<<' ';
				for(int i=b[max][0]+1;i<b[max][1];i++)
				if(a[i]<0)
					cout<<i+1<<' ';
			for(int i=b[max][1]+1;i<n;i++)
				cout<<i+1<<' ';
			
			return 0;
		}
