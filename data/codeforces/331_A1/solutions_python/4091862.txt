#include <iostream>
using namespace std;
typedef long long int LLI;

int a, b, n;
LLI m = -(1LL << 62), s, A[101], S[101];

int main() {
	ios_base::sync_with_stdio(0);
	cin >> n;
	for (int i = 1; i <= n; i++) {
		cin >> A[i];
		if (A[i] < 0) S[i] = S[i - 1];
		else S[i] = S[i - 1] + A[i];
	}
	for (int i = 1; i <= n; i++) {
		for (int j = i + 1; j <= n; j++) {
			if (A[i] != A[j]) continue;
			s = A[i] + A[i] + S[j - 1] - S[i];
			if (s > m) {
				m = s;
				a = i;
				b = j;
			}
		}
	}
	s = a - 1 + n - b;
	for (int i = a + 1; i < b; i++) {
		if (A[i] < 0) s++;
	}
	cout << m << " " << s << "\n";
	for (int i = 1; i < a; i++) cout << i << " ";
	for (int i = a + 1; i < b; i++) {
		if (A[i] < 0) cout << i << " ";
	}
	for (int i = b + 1; i <= n; i++) cout << i << " ";
	cout << "\n";
	return 0;
}
