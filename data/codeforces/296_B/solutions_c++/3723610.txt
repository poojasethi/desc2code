#include <string>
#include <iostream>
#include <algorithm>
using namespace std;
const long p=1000000007;
long long f[110000][4]={};
long n,i,a,b,c;
string w,s;
int main() {
	ios::sync_with_stdio(false);
	cin >> n >> s >> w;
	s=" "+s;w=" "+w;
	f[0][0]=1;f[0][1]=f[0][2]=0;
	for (i=1;i<=n;i++) {
		if ((s[i]!='?')&&(w[i]!='?')) {
			a=long(s[i]>w[i]);
			b=long(s[i]<w[i]);
			c=long(s[i]==w[i]);
		} else if ((s[i]=='?')&&(w[i]=='?')) {
			a=b=45;c=10;
		} else if (s[i]=='?') {
			b=w[i]-'0';a=9-b;c=1;
		} else {
			a=s[i]-'0';b=9-a;c=1;
		}
		f[i][0]=c*f[i-1][0]%p;
		f[i][1]=((a+c)*f[i-1][1]+a*f[i-1][0])%p;
		f[i][2]=((b+c)*f[i-1][2]+b*f[i-1][0])%p;
		f[i][3]=((a+b+c)*f[i-1][3]+b*f[i-1][1]+a*f[i-1][2])%p;
	}
	cout << f[n][3] << endl;
	return 0;
}
