#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long ans=1,x=1,y=1,z=1;
int main(){
    int i,n;
    string a,b;
    cin>>n>>a>>b;
    for(i=0;i<n;i++)
        if(a[i]=='?'&&b[i]=='?'){
            ans=ans*100%mod;
            x=x*55%mod;
            y=y*55%mod;
            z=z*10%mod;
        }
        else if(a[i]=='?'&&b[i]!='?'){
            ans=ans*10%mod;
            x=x*(10-b[i]+'0')%mod;
            y=y*(b[i]-'0'+1)%mod;
        }
        else if(a[i]!='?'&&b[i]=='?'){
            ans=ans*10%mod;
            x=x*(a[i]-'0'+1)%mod;
            y=y*(10-a[i]+'0')%mod;
        }
        else{
            if(a[i]>b[i]){y=0;z=0;}
            if(b[i]>a[i]){x=0;z=0;}
        }
    printf("%I64d\n",(ans+mod-x-y+z)%mod);
}
