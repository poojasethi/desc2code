#include<iostream>
#include<cstdio>
using namespace std;
#define MOD 1000000007
#define MAX 100005
char c[MAX];
char d[MAX];
int main(){
long long int ans=1,x=1,y=1,n,i;
long long int z=1;
char p;
	cin>>n;
	scanf("%c",&p);
	scanf("%s",c);
	scanf("%c",&p);
	scanf("%s",d);
//	printf("%s",c);printf("%s",d);
	for(i=0;i<n;i++){
		if(c[i]=='?' && d[i]=='?'){
			ans=(ans*100)%MOD;
			x=(x*55)%MOD;
			y=(y*55)%MOD;
			z=(z*10)%MOD;
		}
		else if(c[i]=='?' && d[i]!='?'){
			ans=(ans*10)%MOD;
			x=(x*(10-d[i]+48))%MOD;//greater than or equal to
			y=(y*(d[i]-47))%MOD;//less than or equal to
		}
		else if(c[i]!='?' && d[i]=='?'){
			ans =(ans*10)%MOD;
			y=(y*(10-c[i]+48))%MOD;
			x=(x*(c[i]-47))%MOD;
		}
		else if(c[i]!='?' && d[i]!='?'){
			if(c[i]>d[i]){y=0;z=0;}
			if(d[i]>c[i]){x=0;z=0;}
		}
	}
 ans=(ans-x-y+z)%MOD;
    if(ans<0)ans+=MOD;
    cout<<ans<<endl;
	return 0;
}
