#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
#define MOD 1000000007
#define inc(a,b) {a+=b;a%=MOD;}
int n;
long long sx,sy,sd;
char a[100005],b[100005];
long long dp[100005][5];
int main()
{
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	scanf("%d\n",&n);
	for (int i=1;i<=n;++i) scanf("%c",&a[i]);
	scanf("\n");
	for (int i=1;i<=n;++i) scanf("%c",&b[i]);
	dp[0][0]=1;
	for (int i=1;i<=n;++i)
	{
		sx=sy=sd=0;
		if (a[i]!='?'&&b[i]!='?')
		{
			if (a[i]>b[i]) sx=1;
			if (a[i]<b[i]) sy=1;
			if (a[i]==b[i]) sd=1;
		}
		if (a[i]=='?'&&b[i]!='?')
		{
			sx='9'-b[i];sy=b[i]-'0';sd=1;
		}
		if (a[i]!='?'&&b[i]=='?')
		{
			sx=a[i]-'0';sy='9'-a[i];sd=1;
		}
		if (a[i]=='?'&&b[i]=='?')
		{
			sx=sy=45;
			sd=10;
		}
		for (int j=0;j<=3;++j) inc(dp[i][j],dp[i-1][j]*sd);
		inc(dp[i][1],(dp[i-1][0]+dp[i-1][1])*sx);inc(dp[i][3],(dp[i-1][2]+dp[i-1][3])*sx);
		inc(dp[i][2],(dp[i-1][0]+dp[i-1][2])*sy);inc(dp[i][3],(dp[i-1][1]+dp[i-1][3])*sy);
	}
	cout<<dp[n][3];
	return 0;
}
