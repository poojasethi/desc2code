#include <iostream>
#include <string>

const int M = 1e+9 + 7;

int n;
std::string s1, s2;
int dp[100500][2][2];

int main(){
    std::cin >> n >> s1 >> s2;
    dp[0][0][0] = 1;
    for (int i = 0; i < n; i++)
        for (int f1 = 0; f1 < 2; f1++)
            for (int f2 = 0; f2 < 2; f2++)
                for (int j = 0; j < 10; j++)
                    for (int k = 0; k < 10; k++){
                        if ((s1[i] != '?' && s1[i] != '0' + j) || (s2[i] != '?' && s2[i] != '0' + k))
                            continue;
                        #define madd(a, b) ((a) = ((a) + (b)) % M)
                        madd(dp[i + 1][f1 || j < k][f2 || j > k], dp[i][f1][f2]);
                    }
    std::cout << dp[n][1][1] << "\n";
    return 0;
}
