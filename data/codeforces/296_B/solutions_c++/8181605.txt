#include<bits/stdc++.h>
#define mod 1000000007
#define ll long long int
using namespace std;
int main(){
ll n;
cin>>n;
string s,t;
cin>>s>>t;
ll g=1,l=1,e=1;
ll tot=1;
for(int i=0;i<n;i++){
if(s[i]=='?'&&t[i]=='?'){
e*=10;
g*=55;
l*=55;
tot*=100;
}
else if(s[i]=='?'){
int k=t[i]-'0';
g=g*(9-k+1);
l*=(k+1);
tot*=10;
}
else if(t[i]=='?'){
int k=s[i]-'0';
g=g*(k+1);
l=l*(9-k+1);
tot*=10;
}
else{
int k1=s[i]-'0';
int k2=t[i]-'0';
if(k1>k2){
l=0;
e=0;
}
else if(k1<k2){
g=0;
e=0;
}
}
tot%=mod;
l%=mod;
g%=mod;
e%=mod;
}
tot=tot-l+mod;tot%=mod;
tot=tot-g+mod;tot%=mod;
tot=tot+e;tot%=mod;
tot%=mod;
cout<<tot;
}
