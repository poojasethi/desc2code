#include <stdio.h>
#include <iostream>

using namespace std;

#define mod 1000000007

int n;

long long mult (long long a, long long b) {
	return (a*b)%mod;
}

long long calc (char *a, char* b) {
	int i, j;
	long long ans = 1;
	for (i = 0; i < n; i++) {
		if (a[i] == '?' && b[i] == '?') {
			ans = mult(ans, 55);
		} else if (a[i] == '?') {
			ans = mult(ans, b[i]-'0'+1);
		} else if (b[i] == '?') {
			ans = mult(ans, 10-(a[i]-'0'));
		} else {
			ans = mult(ans, (a[i] <= b[i])?1:0);
		}
	}
	return ans;
}

long long count (char *a) {
	long long ans = 1;
	int i;
	for (i = 0; i < n; i++) {
		if (a[i] == '?')	ans = mult (ans, 10);
	}
	return ans;
}

long long same (char *a, char *b) {
	long long ans = 1;
	int i;
	for (i = 0; i < n; i++) {
		if (a[i] == '?' && b[i] == '?')	ans = mult(ans, 10);
		else if (a[i] != '?' && b[i] != '?' && a[i] != b[i])	ans = 0;
	}
	return ans;
}

int main (void) {
	char a[100001], b[100001];
	cin >> n;
	cin >> a >> b;
	long long ans = mult(count(a),count(b))-(calc(a, b)+calc(b, a))+same(a, b);
	ans %= mod;
	ans += mod;
	ans %= mod;
	cout << ans << endl;
	return 0;
}
