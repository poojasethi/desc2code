#include <iostream>
#include <string>

using namespace std;

typedef long long int ll;

ll modulo=1e9+7;

ll c[4];
ll nextc[4];

int main()
{
  ios::sync_with_stdio(false);
  string s,w;
  int n;
  cin>>n>>s>>w;
  c[0]=1;
  for (int i=0;i<n;i++) {
    if (s[i]=='?') {
      if (w[i]=='?') {
	nextc[0]=c[0]*10;
	nextc[1]=c[1]*(10+45)+c[0]*45;
	nextc[2]=c[2]*(10+45)+c[0]*45;
	nextc[3]=c[3]*100+c[1]*45+c[2]*45;
      } else {
	nextc[0]=c[0];
	nextc[1]=c[1]*('9'-w[i]+1)+c[0]*('9'-w[i]);
	nextc[2]=c[2]*(w[i]-'0'+1)+c[0]*(w[i]-'0');
	nextc[3]=c[3]*10+c[2]*('9'-w[i])+c[1]*(w[i]-'0');
      }
    } else if (w[i]=='?') {
      nextc[0]=c[0];
      nextc[1]=c[1]*(s[i]-'0'+1)+c[0]*(s[i]-'0');
      nextc[2]=c[2]*('9'-s[i]+1)+c[0]*('9'-s[i]);
      nextc[3]=c[3]*10+c[1]*('9'-s[i])+c[2]*(s[i]-'0');
    } else {
      nextc[0]=c[0]*(s[i]==w[i]);
      nextc[1]=c[1]*(s[i]>=w[i])+c[0]*(s[i]>w[i]);
      nextc[2]=c[2]*(s[i]<=w[i])+c[0]*(s[i]<w[i]);
      nextc[3]=c[3]+c[1]*(s[i]<w[i])+c[2]*(s[i]>w[i]);
    }
    for (int j=0;j<4;j++)
      c[j]=nextc[j]%modulo;
  }
  cout<<c[3]<<endl;
}
