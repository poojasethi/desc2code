#include <iostream>
#include <string>
using namespace std;

typedef long long LL;
LL mod = 1000000007;
LL size = 100010;
string s , w;
LL n;
LL maxSum , minSum , equSum , allSum , ans;

void solve( )
{
	allSum = maxSum = minSum = equSum = 1;
	for( int i = 0 ; i<n ; i++ )
	{
		if( s[i] == '?' )
		{
			allSum = allSum * 10 % mod;
		}
		if( w[i] == '?' )
		{
			allSum = allSum * 10 % mod;
		}
		if( s[i] == '?' && w[i] == '?' )
		{
			maxSum = maxSum * 55 % mod;
			minSum = minSum * 55 % mod;
			equSum = equSum * 10 % mod;
		}
		else if( s[i] == '?' )
		{
			maxSum = maxSum * ( '9' - w[i] + 1 ) % mod;
			minSum = minSum * ( w[i] - '0' + 1 ) % mod;
		}
		else if( w[i] == '?' )
		{
			maxSum = maxSum * ( s[i] - '0' + 1 ) % mod;
			minSum = minSum * ( '9' - s[i] + 1 ) % mod;
		}
		else
		{
			maxSum *= s[i] >= w[i];
			minSum *= s[i] <= w[i];
			equSum *= s[i] == w[i];
		}
	}
	ans = ( allSum - minSum - maxSum + equSum + mod ) % mod;
}

int main()
{
	cin >> n >> s >> w;
	solve( );
	cout << ans << endl;
	return 0;
}
