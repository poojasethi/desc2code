#include<iostream>
#include<cstdio>

using namespace std;

typedef long long ll;
const ll MOD=1000000007;
const ll C=55;
int n;
string s1,s2;

ll sum()
{
	int i,j,k;
	ll res=1;
	for(i=0;i<n;i++)
	{
		if(s1[i]=='?')
			res=res*10%MOD;
		if(s2[i]=='?')
			res=res*10%MOD;	
	}
	return res;
}

ll sum1()
{
	int i,j,k;
	ll res1=1,res2=1;
	for(i=0;i<n;i++)
	{
		if(s1[i]=='?'&&s2[i]=='?')
		{
			res1=res1*C%MOD;
			res2=res2*C%MOD;
		}
		else if(s1[i]=='?')
		{
			res1=res1*((s2[i]-'0')+1)%MOD;
			res2=res2*(('9'-s2[i])+1)%MOD;
		}
		else if(s2[i]=='?')
		{
			res1=res1*(('9'-s1[i])+1)%MOD;
			res2=res2*((s1[i]-'0')+1)%MOD;
		}
		else
		{
			if(s1[i]>s2[i])
				res1=0;
			if(s1[i]<s2[i])
				res2=0;
		}
	}
	return (res1+res2)%MOD;
}

ll sum2()
{
	int i,j,k;
	ll res=1;
	for(i=0;i<n;i++)
	{
		if(s1[i]=='?'&&s2[i]=='?')
		{
			res=res*10%MOD;
		}
		if(s1[i]!='?'&&s2[i]!='?'&&s1[i]!=s2[i])
		{
			return 0;			
		}
	}
	return res;
}

int main()
{	
	cin>>n;
	cin>>s1;
	cin>>s2;
	cout<<((sum()-sum1()+sum2()+MOD)%MOD)<<endl;
	return 0;
}
