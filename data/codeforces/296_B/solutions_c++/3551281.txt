#include<stdio.h>
#include<string.h>
#define M 1000000007
char s1[100005],s2[100005];
__int64 dp[2][4];
int main()
{
	int i,j,k,n;
	scanf("%d%s%s",&n,s1,s2);
	memset(dp,0,sizeof(dp));
	dp[0][0]=1;
	for(i=0;i<n;i++)
	{
		for(j='0';j<='9';j++)
			if(s1[i]==j||s1[i]=='?')
				for(k='0';k<='9';k++)
					if(s2[i]==k||s2[i]=='?')
					{
                          if(j==k)
						  {
							  dp[(i+1)%2][0]+=dp[i%2][0];
							  dp[(i+1)%2][1]+=dp[i%2][1];
                              dp[(i+1)%2][2]+=dp[i%2][2];
							  dp[(i+1)%2][3]+=dp[i%2][3];
						  }
						   if(j<k)
						  {
                              dp[(i+1)%2][2]+=dp[i%2][2]+dp[i%2][0];
							  dp[(i+1)%2][3]+=dp[i%2][3]+dp[i%2][1];
						  }
						    if(j>k)
						  {
							  dp[(i+1)%2][1]+=dp[i%2][1]+dp[i%2][0];
							  dp[(i+1)%2][3]+=dp[i%2][3]+dp[i%2][2];
						  }
							  dp[(i+1)%2][0]%=M;
							  dp[(i+1)%2][1]%=M;
                              dp[(i+1)%2][2]%=M;
							  dp[(i+1)%2][3]%=M;
					}
			memset(dp[i%2],0,sizeof(dp[i%2]));
	}
	printf("%I64d\n",dp[n%2][3]);
	return 0;
}
