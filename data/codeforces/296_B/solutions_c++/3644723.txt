#include<cstdio>
#define mod 1000000007
using namespace std;
char a[110000],b[110000];
int n;
long long ans;

long long gao(char *a,char *b) {
    long long tmp=1;
    int i;
    for (i=0;i<n;i++) {
        if (a[i]!='?' && b[i]!='?' && a[i]>b[i]) tmp=0;
        if (a[i]=='?' && b[i]!='?') tmp=tmp*(b[i]-'0'+1)%mod;
        if (a[i]!='?' && b[i]=='?') tmp=tmp*('9'-a[i]+1)%mod;
        if (a[i]=='?' && b[i]=='?') tmp=tmp*55%mod;
    }
    return tmp;
}

int main() {
    int i;
    long long tmp;
    scanf("%d%s%s",&n,&a,&b);
    ans=1;tmp=1;
    for (i=0;i<n;i++) {
        if (a[i]=='?') ans=ans*10%mod;
        if (b[i]=='?') ans=ans*10%mod;
        if (a[i]=='?' && b[i]=='?') tmp=tmp*10%mod;
        if (a[i]!='?' && b[i]!='?' && a[i]!=b[i]) tmp=0;
    }
    ans=(ans+tmp)%mod;
    ans=(ans-gao(a,b))%mod;
    ans=(ans-gao(b,a))%mod;
    if (ans<0) ans+=mod;
    printf("%I64d\n",ans);
}
