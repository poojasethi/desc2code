#include <iostream>
#include <sstream>
#include <cstring>
#include <climits>
#include <cmath>
#include <vector>
#include <map>
#include <algorithm>
using namespace std;
const int M=1000000007;
int main(void)
{
  int n; cin >> n;
  string s, w; 
  cin >> s >> w;
  long long big(1),small(1),eq(1),num(1);
  for (int i=0; i<n; ++i) 
  {
    char cs = s[i]; char cw = w[i];
    if (cs=='?' && cw=='?') 
    {
      big = (big*55)%M;
      small = (small*55)%M;
      eq = (eq*10)%M;
      num = (num*100)%M;
    } 
    else if (cs=='?') 
    {
      int nw = cw-'0';
      big = (big*(10-nw))%M;
      small = (small*(nw+ 1))%M;
      num = (num*10)%M;
    } 
    else if (cw=='?') 
    {
      int ns = cs-'0';
      big = (big*(ns+ 1))%M;
      small = (small*(10-ns))%M;
      num = (num*10)%M;
    } 
    else 
    {
      if (cs>cw) {small = 0; eq=0;}
      if (cs<cw) {big = 0; eq=0;}
    }
  }
  cout << (num-big-small+eq+2*M)%M << endl;
//  system("pause");
  return 0;
}
