#include <stdio.h>
#include <string.h>
#define mod 1000000007
#define mmax 100010

int n;
__int64 dp[mmax][4];
char s1[mmax],s2[mmax];

int main()
{
	int i,j,k;
	scanf("%d",&n);
	scanf("%s%s",s1+1,s2+1);
	memset(dp,0,sizeof(dp));
	dp[0][0]=1;
	for(i=1;i<=n;i++)
	{
		for(j='0';j<='9';j++)
		{
			if(s1[i]==j||s1[i]=='?')
			{
				for(k='0';k<='9';k++)
				{
					if(s2[i]==k||s2[i]=='?')
					{
						if(j==k)
						{
							dp[i][0]+=dp[i-1][0];
							dp[i][1]+=dp[i-1][1];
							dp[i][2]+=dp[i-1][2];
							dp[i][3]+=dp[i-1][3];
						}
						else if(j<k)
						{
							dp[i][1]+=dp[i-1][0]+dp[i-1][1];
							dp[i][3]+=dp[i-1][3]+dp[i-1][2];
						}
						else if(j>k)
						{
							dp[i][2]+=dp[i-1][0]+dp[i-1][2];
							dp[i][3]+=dp[i-1][3]+dp[i-1][1];
						}
					}
				}
			}
		}
		for(j=0;j<=3;j++) dp[i][j]%=mod;
	}
	printf("%I64d\n",dp[n][3]);
	return 0;
}//当时怎么就想去模拟，然后写了200多行呢…
