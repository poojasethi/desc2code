#include <iostream>
#include <string>
#define msk(a, b) (a > b ? 2 : a < b)
#define base 1000000007
#define update(a, b) f[i + 1][j | a] = (f[i + 1][j | a] + (b) * f[i][j]) % base
using namespace std;

string s1, s2;
int n;
__int64 f[111111][4];

int main(){
	cin >> n >> s1 >> s2;
	f[0][0] = 1;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < 4; j++)
			if (s1[i] != '?' && s2[i] != '?')
				update(msk(s1[i], s2[i]), 1);
			else if (s1[i] == '?' && s2[i] == '?'){
				update(msk(0, 1), 45);
				update(msk(1, 0), 45);
				update(0, 10);
			}
			else{
				update(msk(0, 1), s1[i] == '?' ? s2[i] - '0' : '9' - s1[i]);
				update(msk(1, 0), s1[i] == '?' ? '9' - s2[i] : s1[i] - '0');
				update(0, 1);
			}
	cout << f[n][3] << endl;
}
