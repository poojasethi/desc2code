#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define mo 1000000007
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int f[N][4],a[3];
char p[N],s[N];
int main() {
	scanf("%d%s%s",&n,p+1,s+1);
	f[0][0]=1;
	For(i,1,n) {
		memset(a,0,sizeof(a));
		if (p[i]!='?'&&s[i]!='?') {
			if (p[i]>s[i]) a[1]=1;
			else if (p[i]<s[i]) a[2]=1;
			else a[0]=1;
		} else if (p[i]=='?'&&s[i]=='?') {
			a[0]=10,a[1]=a[2]=45;
		} else if (p[i]=='?') {
			int A=s[i]-'0';
			a[0]=1,a[1]=9-A,a[2]=A;
		} else {
			int A=p[i]-'0';
			a[0]=1,a[1]=A,a[2]=9-A;
		}
		For(j,0,3)For(k,0,2) f[i][j|k]=(f[i][j|k]+(int64)f[i-1][j]*a[k]%mo)%mo;
	}
	printf("%d\n",f[n][3]);
	return 0;
}
