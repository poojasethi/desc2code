#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007  

int main(){
    ios_base::sync_with_stdio(false);
    int n; cin >> n;
    string s, t; cin >> s >> t;
    long long res1 = 1, res2 = 1, res3 = 1, res4 = 1;
    for(int i=0; i<n; i++){
        if(s[i]=='?' && t[i]=='?'){
            res1 = (55*res1)%MOD; res2 = (55*res2)%MOD;
            res3 = (100*res3)%MOD; res4 = (10*res4)%MOD;            
        }
        else if(t[i]=='?'){
            res1 = ((10-(s[i]-'0'))*res1)%MOD;
            res2 = ((s[i]-'0'+1)*res2)%MOD;
            res3 = (10*res3)%MOD;
        }
        else if(s[i]=='?'){
            res1 = ((t[i]-'0'+1)*res1)%MOD;
            res2 = ((10-(t[i]-'0'))*res2)%MOD;
            res3 = (10*res3)%MOD;           
        }
        else{
            if(t[i]>s[i]) res2 = res4 = 0;
            else if(s[i]>t[i]) res1 = res4 = 0;         
        }
    }
    long long res = (res4+res3-res2-res1+MOD+MOD)%MOD;
    cout << res <<'\n';
    return 0;
}
