#include <iostream>
#include <string>

using namespace std;

const int M = 1e+9 + 7;

int n;
string s1, s2;
int dp[100500][2][2];

int main(){
    cin >> n >> s1 >> s2;
    dp[0][0][0] = 1;
    for (int i = 0; i < n; i++)
        for (int f1 = 0; f1 < 2; f1++)
            for (int f2 = 0; f2 < 2; f2++)
                for (int j = 0; j < 10; j++)
                    for (int k = 0; k < 10; k++){
                        if ((s1[i] != '?' && s1[i] != '0' + j) || (s2[i] != '?' && s2[i] != '0' + k))
                            continue;
                        dp[i + 1][f1 || j < k][f2 || j > k] = (dp[i + 1][f1 || j < k][f2 || j > k] + dp[i][f1][f2]) % M;
                    }
    cout << dp[n][1][1] << endl;
    return 0;
}
