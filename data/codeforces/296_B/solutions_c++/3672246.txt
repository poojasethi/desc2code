#include<cstdio>
#include<cstring>
typedef __int64 LL;
const int LMT = 100003;
const int mod =  1000000007;
LL dp[4][LMT];
char a[LMT], b[LMT];
int main(void)
{
    int n, i, j, ad, bd, sta, ea , stb, eb, tag, prh;
    dp[0][0] = 1;
    a[0] = 3; b[0] = 4;
    scanf("%d", &n);
    scanf("%s", a + 1); scanf("%s", b + 1);
    for(i = 1; i <= n; ++i)
    {
        if(a[i] == '?')
        {
            sta = 0; ea = 9;
        }
        else sta = ea = a[i] - '0';
        if(b[i] == '?')
        {
            stb = 0; eb = 9;
        }
        else stb = eb = b[i] - '0';
        for(ad = sta; ad <= ea; ++ad)
            for(bd = stb; bd <= eb; ++bd)
            {
                if(ad > bd) tag = 2;
                else if(ad < bd) tag = 1;
                else tag = 0;
                for(prh = 0; prh < 4; ++prh)
                {
                    dp[prh | tag][i] += dp[prh][i - 1];
                    dp[prh | tag][i] %= mod;
                }
            }
    }
    printf("%I64d\n", dp[3][n]);
    return 0;
}
