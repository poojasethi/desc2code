#include <iostream>
using namespace std;

typedef long long ll;
const ll mod = 1e9 + 7;

ll all = 1;

int n;
char s[2][100000 + 4];

ll cal(int, int k = 0);
ll eq();

int main() {
	cin >> n >> s[0] >> s[1];
	bool d[2] = {};
	for (int i = 0; i < 2; i++)
		for (int j = 0; j < n; j++)
			if (s[i][j] == '?')
				all = all * (ll) 10 % mod;
	cout << (all - cal(0) - cal(1) + eq() + 2 * mod) % mod << '\n';
	return 0;
}
ll cal(int c, int k) {
	if (k == n)
		return (ll) 1;
	if (s[c][k] != '?' && s[1 - c][k] != '?')
		if (s[c][k] >= s[1 - c][k])
			return cal(c, k + 1);
		else
			return (ll) 0;
	if (s[c][k] == '?' && s[1 - c][k] == '?')
		return (ll) 55 * cal(c, k + 1) % mod;
	if (s[c][k] == '?')
		return (ll) (10 - (s[1 - c][k] - '0')) * cal(c, k + 1) % mod;
	return (ll) (s[c][k] - '0' + 1) * cal(c, k + 1) % mod;
}
ll eq() {
	ll res = 1;
	for (int i = 0; i < n; i++)
		if (s[0][i] != '?' && s[1][i] != '?' && s[0][i] != s[1][i])
			res = 0;
		else if (s[0][i] == '?' && s[1][i] == '?')
				res = res * 10 % mod;
	return res;
}
