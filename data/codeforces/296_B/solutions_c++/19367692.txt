#include <bits/stdc++.h>
using namespace std;

const int maxN = 1e5+100;
const int base = 1e9+7;

int n, f[maxN][2][2];
char s[maxN], t[maxN];

int main() {
	scanf("%i\n%s\n%s\n", &n,s,t);
	f[0][0][0]=1;
	for(int i=0; i<n; i++) 
		for(int j=0; j<2; j++) 
			for(int k=0; k<2; k++) 
				for(char ch1='0'; ch1<='9'; ch1++) if (s[i]=='?' || s[i]==ch1) 	
					for(char ch2='0'; ch2<='9'; ch2++) if (t[i]=='?' || t[i]==ch2) {
						int tj = (ch1 > ch2) | j;
						int tk = (ch1 < ch2) | k;
						f[i+1][tj][tk] = (f[i+1][tj][tk] + f[i][j][k]) %base;
					}
	cout << f[n][1][1];
	return 0;
}
