#include <cstdio>
#include <algorithm>
using namespace std;

typedef long long LL;

const int MX = 100006;
const int MOD = 1000000007;
int n;
char s1[MX];
char s2[MX];

int main() {
  scanf("%d ", &n);
  scanf("%s ", s1);
  scanf("%s ", s2);
  LL res = 1;
  for(int i=0; i<n; ++i) {
    if(s1[i] == '?') res*=10;
    if(s2[i] == '?') res*=10;
    if(res >MOD) res%=MOD;
  }
  LL t1 = 1;
  LL t2 = 1;
  LL t3 = 1;
  for(int i=0; i<n; ++i) {
    if(s1[i] == '?') {
      if(s2[i] == '?') {
        t1*=55;
        t2*=55;
        t3*=10;
      } else {
        t1*= s2[i] - '0' + 1;
        t2*= '9' - s2[i] + 1;
      }
    } else {
      if(s2[i] == '?') {
        t1 *= '9' - s1[i] + 1;
        t2 *= s1[i] -'0' +1;
      } else {
        if(s1[i] < s2[i]) {
          t2*=0;
          t3*=0;
        } else if(s1[i] > s2[i])  {
          t1*=0;
          t3*=0;
        }
      }
    }
    if(t1 > MOD) t1%=MOD;
    if(t2 > MOD) t2%=MOD;
  }
  //printf("%lld %lld %lld %lld\n", res, t1, t2, t3); 
  res = res + 2*MOD - t1 - t2+t3;
  res %= MOD;
  printf("%d\n", (int)res);
}
