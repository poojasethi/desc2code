#include<iostream>
#include<cstring>
#include<cstdio>
#include<cstdlib>
using namespace std;
#define LL long long
const int N=100010,mod=1000000007;
int n;
char a[N],b[N];
LL ans=1,x=1,y=1,z=1;
int main() {
	scanf("%d%s%s",&n,a,b);
	for(int i=0;i<n;i++)
		if(a[i]=='?'&&b[i]=='?') {
			ans=ans*100%mod;
			x=x*55%mod;
			y=y*55%mod;
			z=z*10%mod;
		}
		else if(a[i]=='?'&&b[i]!='?') {
			ans=ans*10%mod;
			x=x*(10-(b[i]-'0'))%mod;
			y=y*(b[i]-'0'+1)%mod;
			z=z*1%mod;
		}
		else if(a[i]!='?'&&b[i]=='?') {
			ans=ans*10%mod;
			x=x*(a[i]-'0'+1)%mod;
			y=y*(10-(a[i]-'0'))%mod;
			z=z*1%mod;
		}
		else {
			if(a[i]>b[i]) {y=0;z=0;}
			if(b[i]>a[i]) {x=0;z=0;}
		}
	ans=ans-x-y+z;
	printf("%I64d\n",(ans+mod)%mod);
	return 0;
}
