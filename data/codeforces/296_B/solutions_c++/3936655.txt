#include <iostream>
#include <string>
using namespace std;
const int MAX_N = 1e5+7;
const int mod = 1e9+7;
int dp[MAX_N][4];
int main()
{
  int n;
  cin>>n;
  string s, t;
  cin>>s>>t;
  dp[0][0] = 1;
  for (int i = 0; i < n; i++)
    for (int j = 0; j < 4; j++)
      for (int a = 0; a < 10; a++) if (s[i] == '?' || s[i] == '0' + a)
        for (int b = 0; b < 10; b++) if (t[i] == '?' || t[i] == '0' + b) {
          int jj = j;
          if (a > b) jj |= 1;
          if (b > a) jj |= 2;
          dp[i+1][jj] = (dp[i+1][jj] + dp[i][j])%mod;
        }
  cout<<dp[n][3]<<endl;
  return 0;
}
