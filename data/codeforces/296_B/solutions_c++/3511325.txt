#include<stdio.h>
#include<stdlib.h>
#define MOD 1000000007
#define LL long long
char a[100010],b[100010];
int main(){
	int n,i;
	scanf("%d",&n);
	scanf("%s%s",a,b);
	LL res_big=1,res_small=1,res_equal=1;		//big~~a[i]>b[i]
	LL numa=1,numb=1;
	for(i=0;i<n;i++){
		if(a[i]=='?'&&b[i]=='?'){
			res_big*=55;res_big%=MOD;
			res_small*=55;res_small%=MOD;
			res_equal*=10;res_equal%=MOD;
			numa*=10;numa%=MOD;
			numb*=10;numb%=MOD;
			}
		else if(a[i]=='?'){
			int t=b[i]-'0';
			res_big*=10-t;res_big%=MOD;
			res_small*=t+1;res_small%=MOD;	
			numa*=10;numa%=MOD;
			}
		else if(b[i]=='?'){
			int t=a[i]-'0'; 
			res_big*=t+1;res_big%=MOD;
			res_small*=10-t;res_small%=MOD;	
			numb*=10;numb%=MOD;	
			}
		else {
			if(a[i]>b[i])res_small=0;
			if(a[i]<b[i])res_big=0;
			if(a[i]!=b[i])res_equal=0;
			}
		}
//	printf("%d %d %d\n",res_small,res_big,res_equal);
	LL res=res_small+res_big-res_equal;res%=MOD;
	LL temp=numa*numb;
	temp%=MOD;
	res=temp-res;
	while(res<0)res+=MOD;
	res%=MOD;
	printf("%I64d\n",res); 
//	return main();
	return 0;
	}
