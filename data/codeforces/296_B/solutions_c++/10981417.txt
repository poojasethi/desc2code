#include <iostream>
#include <cstdio>
using namespace std;
typedef long long LL;
LL MOD = (LL)1e9 + 7;
LL dp[100010][2][2];
char s[100010], t[100010];
int n;

int main() {
	//freopen( "F.in", "r", stdin );

	scanf( "%d", &n );
	scanf( "%s%s", s + 1, t + 1 );

	dp[0][0][0] = 1;
	for ( int i = 1; i <= n; i ++ ) {
		for ( int j = 0; j < 2; j ++ ) {
			for ( int k = 0; k < 2; k ++ ) {
				LL val = dp[i - 1][j][k];
				if ( !val ) continue;

				for ( int a = 0; a <= 9; a ++ ) {
					int numA = ( s[i] == '?' ) ? a : (int)s[i] - '0';
					for ( int b = 0; b <= 9; b ++ ) {
						int numB = ( t[i] == '?' ) ? b : (int)t[i] - '0';
						if ( numA > numB ) dp[i][1][k] = ( dp[i][1][k] + val ) % MOD;
						if ( numB > numA ) dp[i][j][1] = ( dp[i][j][1] + val ) % MOD;
						if ( numA == numB ) dp[i][j][k] = ( dp[i][j][k] + val ) % MOD;
						if ( t[i] != '?' ) break;
					}
					if ( s[i] != '?' ) break;
				}
			}
		}
	}

	cout <<dp[n][1][1]<<endl;

	return 0;
}
