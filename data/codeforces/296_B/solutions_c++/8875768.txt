#include<stdio.h>
#include<algorithm>
#define LL long long
const int N=1e5+10,Z=1e9+7;
int n,m,i;
LL tot,big,small,equ,wrong,ans;
char a[N],b[N];
void end(int x)
{
	printf("%d\n",x);
	exit(0);
}
int main()
{
	scanf("%d%s%s",&n,a,b);
    tot=big=small=equ=1;
    for(i=0;i<n;i++)
    {
        if(a[i]=='?'&&b[i]=='?')
        {
            tot=tot*100%Z;
            big=big*55%Z;
            small=small*55%Z;
            equ=equ*10%Z;
        }
        else if(a[i]=='?')
        {
            tot=tot*10%Z;
            big=big*(58-b[i])%Z;
            small=small*(b[i]-47)%Z;
        }
        else if(b[i]=='?')
        {
            tot=tot*10%Z;
            big=big*(a[i]-47)%Z;
            small=small*(58-a[i])%Z;
        }
        else
        {
            if(a[i]>b[i])small=0,equ=0;
            if(a[i]<b[i])big=0,equ=0;
        }
    }
    wrong=big+small;
    if(big>0&&small>0)wrong-=equ;
    ans=(tot+Z+Z-wrong)%Z;
    printf("%I64d\n",ans);
}
