#include<stdio.h>
#include<string.h>
#include<algorithm>
#define mod 1000000007
using namespace std;
char s1[100005],s2[100005];
__int64 dp[100005][4];
int main()
{
	int i,j,cas,n;
	while(scanf("%d",&n)!=EOF)
	{
		scanf("%s%s",s1+1,s2+1);
		memset(dp,0,sizeof(dp));
		dp[0][0]=1;
		for(i=1;i<=n;i++)
		{
			for(char t1='0';t1<='9';t1++)
			{
				if(s1[i]==t1||s1[i]=='?')
				    for(char t2='0';t2<='9';t2++)
				    {
    					if(s2[i]==t2||s2[i]=='?')
    					{
					    	if(t1==t2)
					    	{
	    						dp[i][0]+=dp[i-1][0];dp[i][0]%=mod;
	    						dp[i][1]+=dp[i-1][1];dp[i][1]%=mod;
	    						dp[i][2]+=dp[i-1][2];dp[i][2]%=mod;
	    						dp[i][3]+=dp[i-1][3];dp[i][3]%=mod;
	    					}
	    					else if(t1<t2)
	    					{
					    		dp[i][1]+=dp[i-1][0]+dp[i-1][1];dp[i][1]%=mod;
					    		dp[i][3]+=dp[i-1][2]+dp[i-1][3];dp[i][3]%=mod;
					    	}
					    	else
					    	{
	    						dp[i][2]+=dp[i-1][0]+dp[i-1][2];dp[i][2]%=mod;
					    		dp[i][3]+=dp[i-1][1]+dp[i-1][3];dp[i][3]%=mod;
	    					}
					    }
    				}
			}
			
		}
		printf("%I64d\n",dp[n][3]);
	}
	return 0;
}
