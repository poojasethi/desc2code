#include<cstdio>
#define MOD 1000000007
using namespace std;

int main()
{
    int N;
    long long a, b, c, d;
    a = b = c = d = 1;
    char s[111111], w[111111];
    scanf( "%d", &N );
    scanf( "%s %s", &s, &w );
    long long ge, le, eq, all;
    for( int i = 0; i < N; i++ )
    {
         if( s[i]=='?' && w[i]=='?' )
             all = 100, ge = 55, le = 55, eq = 10;
         else if( s[i] == '?' )
              all = 10, ge = '9' - w[i] + 1, le = w[i] - '0' + 1, eq = 1;
         else if( w[i] == '?' )
              all = 10, ge = s[i] - '0' + 1, le = '9' - s[i] + 1, eq = 1;
         else if( s[i] == w[i] )
              all = 1, ge = 1, le = 1, eq = 1;
         else if( s[i] > w[i] )
              all = 1, ge = 1, le = 0, eq = 0;
         else
             all = 1, ge = 0, le = 1, eq = 0;
         a = (a * all)% MOD;
         b = (b * ge)% MOD;
         c = (c * le)% MOD;
         d = (d * eq)% MOD;
    }
    long long ans = (a - b - c + d + MOD)% MOD;
    printf( "%I64d\n",ans );
    return 0;
}
