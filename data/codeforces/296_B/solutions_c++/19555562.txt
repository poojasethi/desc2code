#include <bits/stdc++.h>

using namespace std;
typedef long long LL;
const LL mod = 1e9 + 7;
const int N = 1e5 + 10;

int n;
char s[N], t[N];

LL cal(char *s, char *t)
{
	LL res = 1;
	
	for(int i = 1; i <= n; i++)
	{
		if(s[i] == '?' && t[i] == '?')
		{
			res = res * 55 % mod;
		}
		else if(s[i] == '?')
		{
			res = res * (t[i] - '0' + 1) % mod;
		}
		else if(t[i] == '?')
		{
			res = res * (10 - (s[i] - '0')) % mod;
		}
		else if(s[i] > t[i])
		{
			res = 0;
		}
	}
	
	return res;
}

LL cal2()
{
	LL res = 1;
	
	for(int i = 1; i <= n; i++)
	{
		if(s[i] == '?' && t[i] == '?') res = res * 10 % mod;
		else if(s[i] != '?' && t[i] != '?' && s[i] != t[i]) res = 0;
	}
	
	return res;
}
int main()
{
	cin >> n;
	scanf("%s%s", s + 1, t + 1);
	
	LL ans = 1;
	for(int i = 1; i <= n; i++)
	{
		if(s[i] == '?') ans = ans * 10 % mod;
		if(t[i] == '?') ans = ans * 10 % mod;
	}
	
	ans -= cal(s, t) + cal(t, s);
	ans += cal2();
	ans %= mod;
	
	cout << (ans + mod) % mod <<endl;
	
	return 0;
}
