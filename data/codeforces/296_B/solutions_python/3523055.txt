M=10**9+7
input()
R=lambda:map(lambda x:ord(x)-ord('0'),raw_input())
a,b=R(),R()
p=zip(a,b)
A=lambda v:reduce(lambda x,y:x*(1,10)[y>10]%M,v,1)
def C(p):
  w=1
  for x,y in p:
    w=w*((55,10-y),(x+1,x>=y))[x<10][y<10]%M
  return w
def D(p):
  w=1
  for x,y in p:
    w=w*((10,1),(1,x==y))[x<10][y<10]%M
  return w
s=A(a)*A(b)-C(p)-C(zip(b,a))+D(p)
print (s%M+M)%M
