import sys

MOD = 10 ** 9 + 7

def ReadIn():
    data = sys.stdin.read().split()
    offset = 0
    while offset < len(data):
        n = int(data[offset])
        offset += 1
        yield data[offset : offset + 2]
        offset += 2

def count(s):
    ans = 1
    for a, b in zip(*s):
        if '?' not in (a, b): continue
        if a == '?' and b == '?':
            ans *= 55
            ans %= MOD
        elif a == '?':
            ans *= ord(b) - ord('0') + 1
            ans %= MOD
        else:
            ans *= ord('9') - ord(a) + 1
            ans %= MOD
    return ans

def equal(s):
    ans = 1
    for a, b in zip(*s):
        if a == '?' and b == '?':
            ans *= 10
            ans %= MOD
        elif '?' not in (a, b) and a != b:
            return 0
    return ans

def Solve(s):
    less = sum('?' not in (a, b) and a < b for a, b in zip(*s))
    more = sum('?' not in (a, b) and a > b for a, b in zip(*s))
    ques = sum(x == '?' for x in ''.join(s))
    total = pow(10, ques, MOD)
    if less and more:
        ans = total
    elif less:
        ans = total - count(s)
    elif more:
        ans = total - count(reversed(s))
    else:
        ans = total - count(s) - count(reversed(s)) + equal(s)
    ans %= MOD
    if ans < 0: ans += MOD
    print ans

if __name__ == '__main__':
    for s in ReadIn():
        Solve(s)
