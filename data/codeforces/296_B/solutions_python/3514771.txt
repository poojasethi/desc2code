#!/usr/bin/env python
import sys

MOD = 1000000007

EQUAL   = 0
FIRST   = 1
SECOND  = 2

def comparable(s, w):
    s_big, w_big = False, False
    for i in xrange(len(s)):
        if s[i] != '?' and w[i] != '?':
            if int(s[i]) > int(w[i]):
                s_big = True
            elif int(s[i]) < int(w[i]):
                w_big = True

        if s_big and w_big:
            return (False, None)

    if s_big:
        return (True, FIRST)
    elif w_big:
        return (True, SECOND)
    else:
        return (True, EQUAL)

def sum_to(n):
    return (n * (n + 1)) / 2
            

if __name__ == '__main__':
    n = int(sys.stdin.readline().strip())
    
    s = sys.stdin.readline().strip()
    w = sys.stdin.readline().strip()
    
    can_be_comp, bigger = comparable(s, w)
    if can_be_comp:
        # The string can be comparable (or not)
        count = 1
        total = 1

        s1_big, s2_big, s_equal = 1, 1, 1
        for i in xrange(n):
            if bigger == FIRST:
                if s[i] == '?' and w[i] == '?':
                    count *= sum_to(10)
                    total *= (10 * 10)
                elif s[i] == '?':
                    count *= (10 - int(w[i]))
                    total *= 10
                elif w[i] == '?':
                    count *= (int(s[i]) + 1)
                    total *= 10
            elif bigger == SECOND:
                if s[i] == '?' and w[i] == '?':
                    count *= sum_to(10)
                    total *= 10 * 10
                elif s[i] == '?':
                    count *= (int(w[i]) + 1)
                    total *= 10
                elif w[i] == '?':
                    count *= (10 - int(s[i]))
                    total *= 10
            elif bigger == EQUAL:
                if s[i] == '?' and w[i] == '?':
                    s1_big  *= sum_to(10)
                    s2_big  *= sum_to(10)
                    s_equal *= 10 
                    total *= (10 * 10)
                elif s[i] == '?':
                    s1_big *= (10 - int(w[i]))
                    s2_big *= (int(w[i]) + 1)
                    total *= 10
                elif w[i] == '?':
                    s1_big *= (int(s[i]) + 1)
                    s2_big *= (10 - int(s[i]))
                    total *= 10

            s1_big = s1_big % MOD
            s2_big = s2_big % MOD
            s_equal = s_equal % MOD
            count = count % MOD
            total = total % MOD
        
        if bigger == EQUAL:
            count = (s1_big + s2_big - s_equal) % MOD

        print (total - count) % MOD
    else:
        # The string is imcomparable for sure no matter what we do
        total = 1
        for i in xrange(n):
            if s[i] == '?':
                total *= 10
            if w[i] == '?':
                total *= 10
            total = total % MOD

        print total % MOD
