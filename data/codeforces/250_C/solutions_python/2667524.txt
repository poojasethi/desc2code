#!/usr/bin/python
from sys import stdin
Input = lambda:map(int,stdin.readline().split())

n,k = Input()
lia = Input()
bv = [0]
bn = [1]
bl = 0
bv[0] = lia[0]
for i in lia:
    if(i == bv[bl]):
        bn[bl] += 1
    else:
        bl += 1
        bv.append(i)
        bn.append(1)
bl += 1
k += 1
retv = [0]
for i in xrange(1,k):
    retv += [0]
for pos,v in enumerate(bv):
    if pos == 0 or pos == bl - 1:
        retv[v] += 1
    else:
        if(bv[pos-1] == bv[pos+1]):
            retv[v] += 2
        else:
            retv[v] += 1
retvv = 0
ret = 1
for i in xrange(1,k):
    if(retvv < retv[i]):
        retvv = retv[i]
        ret = i
print ret
