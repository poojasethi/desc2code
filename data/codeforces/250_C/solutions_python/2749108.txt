n,k=map(int,raw_input().split())
a=map(int,raw_input().split())
cnt=[0 for _ in xrange(k+1)]
tot = 0
a=[0]+a+[0]
i,j=1,1
while i <= n:
    while j <= n and a[j]==a[i]: j+=1
    tot+=1
    if a[i-1]==a[j]: cnt[a[i]] += 2
    else: cnt[a[i]] += 1
    i = j
tot -= 1
ans = tot
p = 0
for i in xrange(1,k+1):
    if ans > tot-cnt[i]:
        ans = tot-cnt[i]
        p = i
print p
