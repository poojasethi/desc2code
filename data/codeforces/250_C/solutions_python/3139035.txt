from itertools import *
from operator import *
r = lambda: map(int, raw_input().split())
n, k = r()
a = map(itemgetter(0), groupby(r()))
p = [0] * k
for i in xrange(len(a)):
  if 0 < i < len(a) - 1:
    x = 1 if a[i - 1] != a[i + 1] else 2
  else:
    x = 1
  p[a[i] - 1] += x
print p.index(max(p)) + 1
