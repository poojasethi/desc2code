def read_list(converter=int, splitter=' '):
    return map(converter, raw_input().strip().split(splitter))

def read_stuff(converter=int):
    return converter(raw_input().strip())

n, k = read_list()
ais = read_list()

stress_increase = [0 for _ in xrange(k + 1)]
stress_increase[0] = n + 1
stress_release = [0 for _ in xrange(k + 1)]
stress_release[0] = -(n + 1)

prev = -1
runner = ais[0]
for ai in ais:
    if ai != runner:
        if prev != -1:
            if prev != ai:
                stress_increase[runner] += 1
        stress_release[runner] += 1
        stress_release[ai] += 1
        prev = runner
        runner = ai


min_stress = n + 1
min_val = -1
for i in xrange(k + 1):
    if stress_increase[i] - stress_release[i] < min_stress:
        min_stress = stress_increase[i] - stress_release[i]
        min_val = i

print min_val
