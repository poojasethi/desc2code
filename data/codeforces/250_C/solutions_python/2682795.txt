def row(func):
    return map(func, raw_input().split())

N, M = row(int)

X = row(int)

Y = []

for x in X:
    if Y == [] or Y[-1] != x:
        Y.append(x)

L = len(Y)

V = [0]*M

for i in range(L):
    stress = 0
    if i == 0 or i == L-1:
        stress += 1
    elif i > 0 and i < L-1:
        if Y[i-1] == Y[i+1]:
            stress += 2
        else:
            stress += 1
    V[Y[i]-1] += stress

p = max(V)

for i in range(M):
    if V[i] == p:
        print i+1
        break
