#!/usr/bin/python

n,k = (int(a) for a in raw_input().split())
lia = [int(a) for a in raw_input().split()]
bv = [0]
bn = [1]
bl = 0
bv[0] = lia[0]
for i in lia:
    if(i == bv[bl]):
        bn[bl] += 1
    else:
        bl += 1
        bv.append(i)
        bn.append(1)
bl += 1

mm = {}
for i in xrange(bl):
    try:
        mm[bv[i]].append(i)
    except:
        mm[bv[i]] = []
        mm[bv[i]].append(i)

ret = 1
retv = bl


k += 1
for i in xrange(1,k):
    val = bl-2
    for j in mm[i]:
        if j == 0:
            val -= 1
        elif j == bl-1:
            val -= 1
        else:
            if(bv[j-1] == bv[j+1]):
                val -= 2
            else:
                val -= 1
    if val < retv:
        ret = i
        retv = val
print ret
