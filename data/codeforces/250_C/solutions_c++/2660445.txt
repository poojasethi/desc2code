#include <iostream>
#include <cstdio>

using namespace std;

int a[110000],b[110000],N,K,i,x,y,ans,cnt;

int main()
{
    scanf("%d%d",&N,&K);
    y=-1;
    for (i=1;i<=N;i++){
        scanf("%d",&x);
        if (x!=y) a[++cnt]=x;
        y=x;
    }
    ++b[a[1]]; 
    if (cnt!=1) ++b[a[cnt]];
    for (i=3;i<=cnt;i++)
        if (a[i]==a[i-2]) b[a[i-1]]+=2;
        else ++b[a[i-1]];
    ans=1;
    for (i=2;i<=K;i++)
        if (b[i]>b[ans]) ans=i;
    printf("%d\n",ans);
}
