#include <cstring>
#include <cstdlib>
#include <cstdio>

int n,k,t;
int a[100010],ans[100010];
int i,j,min;

int main()
{
	scanf("%d%d",&n,&k);
	for (i=0;i<n;i++)scanf("%d",&a[i]);
	j=1;
	for (i=1;i<n;i++)if (a[i]!=a[i-1])a[j++]=a[i];
	n=j;
	for (i=1;i<=k;i++)ans[i]=j-1;
	ans[a[0]]--;ans[a[n-1]]--;
	for (i=1;i<n-1;i++)
		if (a[i-1]==a[i+1])ans[a[i]]-=2;
		else ans[a[i]]--;
	min=1;
	for (i=1;i<=k;i++)
		if (ans[min]>ans[i])min=i;
	printf("%d\n",min);
	return 0;
}
