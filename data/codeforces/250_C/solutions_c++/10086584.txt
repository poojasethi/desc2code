#include<iostream>
using namespace std;
const int MAXN=1000*100+10;
int day[MAXN];
int cnt[MAXN];
int main(){
	int n,k;
	cin>>n>>k;
	for(int i=1;i<=n;i++)
		cin>>day[i];
	int r=1;
	day[0]=day[1];	
	for(int i=1;i<=n;i++){
		if(day[i]!=day[i-1]){
			cnt[day[i-1]]--;
			if(day[r]==day[i])
				cnt[day[i-1]]--;
			r=i-1;
		}
	}
	cnt[day[n]]--;
	int ans=10000*1000;
	int l=1;
	for(int i=1;i<=k;i++){
		if(cnt[i]<ans){
			ans=cnt[i];
			l=i;
		}
	}
	cout<<l<<'\n';
}
