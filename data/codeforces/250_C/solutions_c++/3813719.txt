#include <stdio.h>
int v[100000],f[100000];
int main()
{
	int n,k,m=1;
	scanf("%d%d",&n,&k);
	scanf("%d",&v[1]);
	for(int i=2;i<=n;++i){
		scanf("%d",&v[++m]);
		if(v[m]==v[m-1])
			--m;
	}
	f[v[1]]=m!=1;
	f[v[m]]+=m!=1;
	int mx=0,mxi=1;
	for(int i=2;i<m;++i)
		f[v[i]]+=2*(v[i-1]==v[i+1])+(v[i-1]!=v[i+1]);
	for(int i=1;i<=k;++i)
		if(mx<f[i]){
			mx=f[i];
			mxi=i;
		}
	printf("%d",mxi);
	return 0;
}
