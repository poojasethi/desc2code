#include<stdio.h>
int n,k;
int cd[100001]={0};
int val[100001]={0};
main(){
int a,b,d=0;
int i;
int max=0,bj=1;
scanf("%d%d",&n,&k);
for(a=1;a<=n;a++){
scanf("%d",&b);
if(b!=cd[d])cd[++d]=b;
}
n=d;
for(i=2;i<n;i++){
if(cd[i-1]==cd[i+1])val[cd[i]]+=2;
else val[cd[i]]+=1;
}
val[cd[1]]++;
val[cd[n]]++;
for(i=1;i<=k;i++)
if(val[i]>max){max=val[i];bj=i;}
printf("%d\n",bj);
}
