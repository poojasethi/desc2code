#include <iostream>
using namespace std;
int d[100005];
int main( void )
{
  int n,k,a,b,c;
  cin >> n >> k;
  cin >> a >> b;
  d[a]++;
  if( a != b ){
    d[b]++;
  }
  for( int i=2; i<n; i++ ){
    cin >> c;
    if( b != c ){
      d[c]++;
      if( a == c ) d[b]++;
      a = b;
      b = c;
    }
  }
  int f=0,ret=1;
  for( int i=1; i<=k; i++ ){
    if( f < d[i] ){
      f = d[i];
      ret = i;
    }
  }
  cout << ret << endl;
  return 0;
}
