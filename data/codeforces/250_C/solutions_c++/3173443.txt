#include<cstdio>
#include<vector>
#define NMAX 100005
using namespace std;
int A[NMAX];

int main() {
	int n, k;
	vector<int> V;
	scanf("%d%d", &n, &k);

	for(int i = 0; i < n; ++i) {
		int val;
		scanf("%d", &val);
		if(V.empty() || V.back() != val) {
			V.push_back(val);
			++A[val];
		}
	}

	for(int i = 1; i < (int)V.size() - 1; ++i)
		A[V[i]] += V[i - 1] == V[i + 1];

	int ans = 1;
	for(int i = 2; i <= k; ++i)
		if(A[i] > A[ans])
			ans = i;

	printf("%d\n", ans);
	return 0;
}
