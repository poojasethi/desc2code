#include<cstdio>
#include<cstdlib>
#define S(a) scanf("%d",&a)
#define foru(i,a,b) for(int i=a;i<=b;++i) 

using namespace std;
int max,c[210000],x[210000],ans,n,m,k,a; 
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	S(n),S(k);
	foru(i,1,n)
	{
		S(a);
		if(a!=x[m]) x[++m]=a;
	}
	c[x[1]]=1;
	foru(i,2,m) c[x[i]]+=1,c[x[i]]+=x[i-1]==x[i+1];
	max=0; foru(i,1,k) if(c[i]>max) max=c[i],ans=i;
	printf("%d\n",ans); 
	return 0;
}
