using namespace std;
long n,k,i,j=1,t,ans,ans1,s;

main()
{cin>>n>>s; cin>>a[0]; k=n;
 for(i=1;i<n;i++)
 {cin>>a[j];
  if(a[j-1]==a[j]) k--; else j++;}

 x[a[0]]=1;
