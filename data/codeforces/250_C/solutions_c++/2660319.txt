#include<stdio.h>
int a[100001]={0};
int ans[100001]={0};
int main()
{
  int n,k,i,left=0,right,answer=1;
  scanf("%i %i",&n,&k);
  for(i=1;i<=n;i++)scanf("%i",&a[i]);
  for(right=1;right<=n;right++)
  {
     if(a[right]!=a[right-1])
     {
        ans[a[right]]++;
        if(a[left]==a[right])ans[a[right-1]]++;
        left=right-1;
     }
  }
  for(i=1;i<=k;i++)if(ans[answer]<ans[i])answer=i;
  printf("%i",answer);
  return 0;
}
