#include<cstdio>
#include<algorithm>
using namespace std;
const int MAX_N = 1e5 + 5;
int n, k, a[MAX_N];
int cnt[MAX_N];
int main(){
  scanf("%d %d", &n, &k);
  for(int i = 1; i <= n; scanf("%d", a + i++));
  a[n = unique(a, a + n + 1) - a] = 0;
  for(int i = 1; i < n; i++){
    cnt[a[i]]++;
    cnt[a[i]] += a[i - 1] == a[i + 1];
  }
  printf("%d\n", max_element(cnt + 1, cnt + k + 1) - cnt);
}
