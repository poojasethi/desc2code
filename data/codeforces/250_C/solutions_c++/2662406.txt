#include <vector>
#include <map>
#include<iostream>
#include <deque>
#include <queue>
using namespace std;
int n;
vector <int> v;
int vis[100001]={0};
int main()
{
	int i,j,x,k=0,k1=-1;
	cin>>n>>k;
	for(i=0;i<n;i++)
	{cin>>x;
	if(x!=k1)v.push_back(x);
	k1=x;}
	n=v.size();
	for(i=1;i<n-1;i++)
		if(v[i-1]==v[i+1])
		vis[v[i]]+=2;
		else
		vis[v[i]]++;
	vis[v[0]]++;
    vis[v[n-1]]++;
	int p=1;
	for(i=1;i<=k;i++)if(vis[p]<vis[i])p=i;
	cout<<p<<endl;
	return 0;
}
