#include<cstdio>
#include<iostream>
using namespace std;
int sum, d[100010], a[100010], pre, s, n, m, i;
int main(){
	scanf("%d%d", &n, &m);
	for(i=1;i<=n;i++)scanf("%d", &a[i]), sum+=int(a[i]!=a[i-1]);sum--;
	s = a[1];if(sum != 0)d[s]--;
	for(i=1;i<=n;i++){
		if(s != a[i]){
			if(pre == a[i])d[s]-=1;
			d[a[i]]--;
			pre = s;
			s = a[i];
		}
	}
	int ans = 0;d[0]=100000;
	for(i=1;i<=m;i++)if(d[i]<d[ans])ans = i;
	printf("%d\n", ans);
	return 0;
}
