#include <iostream>
#include <fstream>
using namespace std;
int main(){
  int q,w,e,r,t=0,y=-1,a[100000],s[100000];
  cin>>q>>w;
  for(e=0;e<w;e++)
   s[e]=0;
  for(e=0;e<q;e++){
   cin>>r;r--;
   if(r!=y){
    a[t]=r;
    t++;
    y=r;}}
  s[a[0]]++;
  s[a[t-1]]++;
  for(e=1;e<t-1;e++){
   s[a[e]]++;
   if(a[e-1]==a[e+1])
    s[a[e]]++;}
  for(e=1,r=0;e<w;e++)
   if(s[e]>s[r])
    r=e;
  cout<<r+1;
  return 0;}
