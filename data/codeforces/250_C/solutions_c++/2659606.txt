#include <cstdio>
#include <algorithm>
using namespace std;

int n,k,ans,a[100010],b[100010];

int main() {
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;++i) {
		scanf("%d",&a[++a[0]]);
		if ((a[0]>1)&&(a[a[0]]==a[a[0]-1])) --a[0];
	}
	for (int i=2;i<a[0];++i)
		if (a[i-1]==a[i+1]) b[a[i]]+=2;
						else b[a[i]]+=1;
	++b[a[1]];
	++b[a[a[0]]];
	ans=0;
	for (int i=1;i<=k;++i)
		if ((!ans)||(b[i]>b[ans])) ans=i;
	printf("%d\n",ans);
	return 0;
}
