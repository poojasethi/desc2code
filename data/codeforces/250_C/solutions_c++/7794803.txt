#include<bits/stdc++.h>
#define MAX 120000
using namespace std;
int main(){
	int a[MAX], num[MAX];
	int n, k, ans, max = 0, i , j;
	cin>>n>>k;
	
	for(i = 1; i <= n; i++) cin>>a[i];
	
	for(i = 1; i <= n; i = j){
		for(j = i; j <= n && a[j] == a[i]; j++);
		
		if(a[i - 1] == a[j]) num[a[i]] += 2;
		else num[a[i]] += 1;
	}
	
	for(i = 1; i <= k; i++){
		if(num[i] > max){
			max = num[i];
			ans = i;
		}
	}
	
	cout<<ans<<endl;
	return 0;
}
