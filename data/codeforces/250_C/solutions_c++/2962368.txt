#include <cstdio>
#include <algorithm>
using namespace std;
#define N 100005
int n, k, a[N], c[N], A;

int main()
{
    scanf("%d %d", &n, &k);
    for(int i=1;i<=n;++i)
        scanf("%d", &a[i]);
    for(int i=1;i<=n;++i)
    {
        int j = i;
        while(a[j] == a[i]) j++;
        if(a[i - 1] == a[j]) c[a[i]] += 2;
        else c[a[i]]++;
        i = j - 1;
    }
    A = 1;
    for(int i=2;i<=k;++i)
        if(c[i] > c[A]) A = i;
    printf("%d\n", A);
}
