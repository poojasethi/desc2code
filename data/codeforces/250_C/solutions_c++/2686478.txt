#include <iostream>
#include <algorithm>
using namespace std;
#define N 100005
int q[N],c[N],n,k;
int main(){
    int i,j;
    cin>>n>>k;
    for(i=0;i<n;i++) cin>>q[i];
    for(i=0;i<n;i=j)
    {
        for(j=i;j<n&&q[j]==q[i];j++);
        if(i&&j<n) c[q[i]]+=(q[i-1]==q[j])+1;
        else if(i||j<n) c[q[i]]++;

    }
    int a=1,b=c[1];
    for(i=1;i<=k;i++)
    {
        if(c[i]>b) b=c[i],a=i;
    }
    cout<<a<<endl;
}
