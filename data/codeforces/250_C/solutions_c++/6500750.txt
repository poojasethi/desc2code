#include <cstdio>
int arr2[100010],arr[100010];
int main(){
    int n,k,i,d,m,l=0,maxi=0,ans;
    scanf("%d%d%d",&n,&k,&d);
    arr[l++]=d;
    for(i=1;i<n;i++){
       scanf("%d",&m);
        if(m!=d)
            arr[l++]=m;
        d=m;
    }
    for(i=0;i<n;i++){
         if(i>0&&i+1<l&&arr[i-1]==arr[i+1])
            arr2[arr[i]]+=2;
         else arr2[arr[i]]++;
    }
   for(i=1;i<=k;i++)
       if(maxi<arr2[i]) maxi=arr2[i],ans=i;
   printf("%d",ans);
   return 0;
}
