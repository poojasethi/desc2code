#include <stdio.h>
#include <string.h>
int v[100000];
int c[100001];
int main(){
  int n, k, m, x, imx, i;
  scanf("%d %d", &n, &k);
  scanf("%d", &v[0]);
  memset(c, 0, sizeof(c));
  m=1;
  for(i=1; i<n; i++){
    scanf("%d", &x);
    if(v[m-1]!=x){
      v[m++]=x;
    }
  }

  for(i=0; i<m; i++){
    c[v[i]]++;
    if(i>0 && i+1<m && v[i-1]==v[i+1]){
      c[v[i]]++;
    }
  }
  
  imx=1;
  for(i=2; i<=k; i++){
    if(c[i] > c[imx])
      imx = i;
  }
  
  printf("%d\n", imx);
  return 0;
}
