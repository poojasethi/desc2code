//In the name of God
#include <algorithm>
#include <iostream>
using namespace std;

int n, k, a[100 * 1000], ans[100 * 1000 + 1];

int main() {
	cin.sync_with_stdio(false);
	cin >> n >> k;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	n = unique(a, a + n) - a;
	if (n > 1)
		ans[a[0]]++, ans[a[n - 1]]++;
	for (int i = 1; i < n - 1; i++)
		if (a[i - 1] == a[i + 1])
			ans[a[i]] += 2;
		else
			ans[a[i]]++;
	cout << max_element(ans + 1, ans + k + 1) - ans << '\n';
	return 0;
}
