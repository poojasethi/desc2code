#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
typedef vector<int> vI;

int main(){
  for(int n,k; cin>>n>>k;){
    vI a(n),sts(n); 
    for(int i=0;i<n;i++) cin>>a[i];
    int last_a=a[0],ll_a=-1;
    for(int j=1;j<n;j++){
      if(last_a!=a[j]){
	sts[last_a-1]+=(a[j]==ll_a ? 2 : 1);
	ll_a=last_a;
      }
      last_a=a[j];
    }
    sts[last_a-1]++;
    cout << (max_element(sts.begin(),sts.end())-sts.begin())+1 << endl;
  }
}
