#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;
int a[120000],num[120000],ans,maxx,i,j,k,l,n,m;
int main(){
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;++i)scanf("%d",&a[i]);
	for(i=1;i<=n;i=j){
		for(j=i;j<=n && a[j]==a[i];++j);
		if(a[i-1]==a[j])num[a[i]]+=2;else num[a[i]]++;
	}
	for(i=1;i<=k;++i)if(num[i]>maxx)maxx=num[i],ans=i;
	printf("%d\n",ans);
}
