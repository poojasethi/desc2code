#include<stdio.h>
#define X 100010
int a[X],dp[X];
int main(){
	int i,j,n,k;
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++){
		scanf("%d",&a[i]);
		if(a[i]==a[i-1])
			i--,n--;
	}
	for(i=1;i<=n;i++){
		dp[a[i]]++;
		if(a[i+1]==a[i-1])
			dp[a[i]]++;
	}			
	for(i=1,j=0;i<=k;i++)
		if(dp[i]>dp[j])
			j=i;
	printf("%d\n",j);
	return 0;
}
