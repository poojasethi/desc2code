#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int main()
{
   int n,k,x,t=1,s[100010],d[100010];
   memset(d,0,sizeof(d));
   cin>>n>>k;
   s[0]=0;
    for(int i=1;i<=n;i++)
     {
       cin>>x;
       if(x!=s[t-1]) s[t++]=x;
     }
     s[0]=s[t]=0;
   for(int i=1;i<t;i++)
   {
     if(s[i-1]==s[i+1]) d[s[i]]+=2;
     else d[s[i]]+=1;
   }
   int p=1;
   for(int i=2;i<=k;i++)
    if(d[i]>d[p]) p=i;
   cout<<p<<endl;
  return 0;
}
