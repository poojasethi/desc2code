#include <iostream>
using namespace std;
int n,s,k,ans,w;
int a[100001];
int f[100001];
int main()
{
w=1;
cin >> n >> k;
int last;
cin >> last;
a[0]=last;
for (int i=0;i<=k;i++) f[i]=0;
for (int i=1;i<n;i++)
{
cin >> s;
if (s!=last) {a[w]=s;w++;}
last=s;
}
f[a[0]]++;f[a[w-1]]++;
for (int i=1;i<w-1;i++) if (a[i-1]==a[i+1]) f[a[i]]+=2; else f[a[i]]+=1;
int maxz=-1;
for (int i=1;i<=k;i++) if (f[i]>maxz) {ans=i;maxz=f[i];}
cout << ans;
}
