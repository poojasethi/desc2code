#include<cstdio>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n,k; scanf("%d%d",&n,&k);
	static int a[100000];
	rep(i,n){
		scanf("%d",a+i); a[i]--;
		if(i>0 && a[i-1]==a[i]) i--, n--;
	}

	static int gain[100000];
	rep(i,n){
		gain[a[i]]++;
		if(0<i && i<n-1 && a[i-1]==a[i+1]) gain[a[i]]++;
	}

	printf("%d\n",max_element(gain,gain+k)-gain+1);

	return 0;
}
