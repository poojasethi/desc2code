#include<cstdio>
#include<algorithm>
#define N 100010
using namespace std;
int n, k, a[N], cnt[N];
int main(){
	scanf("%d%d", &n, &k);
	for(int i = 0; i < n; i++) scanf("%d", &a[i]);
	int prv = -1;
	for(int i = 1; i < n; i++){
		if(a[i] != a[i-1]){
			cnt[a[i]]++;
			if(prv == -1 || a[prv] == a[i]) cnt[a[i-1]]++;
			prv = i-1;
		}
	}
	int mx = 0;
	for(int i = 1; i <= k; i++) mx = max(mx, cnt[i]);
	for(int i = 1; i <= k; i++) if(mx == cnt[i])
		return printf("%d\n", i), 0;
}
