#include <iostream>
using namespace std ;
#define MAXX 100001

int i, n, k, a[MAXX] = {0}, d[MAXX] = {-1} ;
int main()
{
	cin >> n ;
	cin >> k ;
	for (i = 1 ; i <= n ; i++)
	{
		cin >> a[i] ;
		d[i] = 0 ;
		if (a[i] == a[i-1]) i--, n-- ;
	}

	for (i = 1 ; i <= n ; i++)
	{
		if (a[i-1] == a[i+1]) d[a[i]] += 2 ;
		else d[a[i]]++ ;
	}
	int min = 1 ;
	for (i = 1 ; i <= k ; i++) {
		if (d[min] < d[i]) min = i ;
	}

	cout << min << endl ;
	return 0 ;
}
