#include <cstdio>
#include <iostream>
#include <algorithm>
#define ll long long
#define f first
#define s second
using namespace std;
const ll M=1ll<<35;
const long N=110000;
ll i,n,x,y,l,r,mid,maxv[N],minv[N];
pair<ll,ll> a[N];
bool check(ll m) {
	if (a[n-1].f-a[1].f<=m) return true;
	ll p=1,q=n,x=M,y=-M,tx,ty;
	while (a[p].f+m<a[q].f) p++;
	while (p>0) {
		tx=min(minv[p-1],x);ty=max(maxv[p-1],y);
		if (tx+m>=ty) return true;
		for (p--;a[p].f+m<a[q].f;q--) {
			x=min(x,a[q].s);
			y=max(y,a[q].s);
		}
	}
	return false;
}
int main() {
	scanf("%I64d",&n);
	for (i=1;i<=n;i++) {
		scanf("%I64d%I64d",&x,&y);
		a[i]=make_pair(x+y,x-y);
	}
	sort(a+1,a+n+1);
	maxv[0]=-M;minv[0]=M;
	for (i=1;i<=n;i++) {
		maxv[i]=max(maxv[i-1],a[i].s);
		minv[i]=min(minv[i-1],a[i].s);
	}
	l=0;r=M;
	while (l<r) {
		mid=(l+r)/2;
		if (check(mid)) r=mid;
		else l=mid+1;
	}
	printf("%.15lf\n",(double)l/2);
	return 0;
}
