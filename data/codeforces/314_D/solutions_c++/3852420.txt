#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define N 120000
using namespace std;
struct ppp{
	int x,y;
}a[N];
int i,j,k,l,n,m,r,mid,Min1y[N],Min2y[N],Max1y[N],Max2y[N],x,y;
bool cmp(const ppp&a,const ppp&b){
	return a.x<b.x;
}
bool check(int mid){
	int l=1,i;
	for(i=1;i<=n;++i){
		while(l<=n && a[i].x+mid>=a[l].x)l++;
		if(max(Max1y[i-1],Max2y[l])-min(Min1y[i-1],Min2y[l])<=mid)
			return 1;
	}
	return 0;
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i){
		scanf("%d%d",&x,&y);
		a[i].x=x+y;
		a[i].y=x-y;
	}
	sort(a+1,a+n+1,cmp);
	Max1y[0]=Max2y[n+1]=-2e9;
	Min1y[0]=Min2y[n+1]=2e9;
	for(i=1;i<=n;++i){
		Max1y[i]=max(Max1y[i-1],a[i].y);
		Min1y[i]=min(Min1y[i-1],a[i].y);
	}
	for(i=n;i>=1;--i){
		Max2y[i]=max(Max2y[i+1],a[i].y);
		Min2y[i]=min(Min2y[i+1],a[i].y);
	}
	l=0; r=2e9;
	while(l<=r){
		mid=(0ll+l+r)/2;
		if(check(mid))r=mid-1;
		else l=mid+1;
	}
	printf("%.10lf\n",(r+1)/2.0);
}
