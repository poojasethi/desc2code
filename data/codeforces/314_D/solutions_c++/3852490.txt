#include <cstdio>
#include <algorithm>
using namespace std;
#define MAXN 100010
#define INF 0x3f3f3f3f
#define MAX(a,b) (a>b?a:b)
#define MIN(a,b) (a<b?a:b)
typedef long long LL;
int n,f[2][MAXN];
struct Point
{
    int x,y;
    bool operator<(const Point &a) const{
        return x<a.x;
    }
}p[MAXN];
void Init()
{
    int i,x,y;
    scanf("%d",&n);
    for(i=1;i<=n;++i){
        scanf("%d%d",&x,&y);
        p[i].x=x+y; p[i].y=y-x;
    }
    sort(p+1,p+1+n);
    f[0][n]=-INF; f[1][n]=INF;
    for(i=n-1;i>=1;--i){
        f[0][i]=MAX(f[0][i+1],p[i+1].y);
        f[1][i]=MIN(f[1][i+1],p[i+1].y);
    }
}
bool Check(double len)
{
    int i,j,Max=-INF,Min=INF;
    for(i=j=1;i<=n;++i){
        for(;j<=n&&p[i].x-p[j].x>len;++j){
            Max=MAX(Max,p[j].y);
            Min=MIN(Min,p[j].y);
        }
        if((double)MAX(Max,f[0][i])-(double)MIN(Min,f[1][i])<=len)
            return true;
    }
    return false;
}
void Work()
{
    LL l=0,r=(LL)MAX(p[1].y,f[0][1])-(LL)MIN(p[1].y,f[1][1]),mid;
    for(;mid=(l+r)>>1LL,l<r;Check(mid)?r=mid:l=mid+1);
    printf("%.6lf\n",(double)r/2);
}
int main()
{
    Init();
    Work();
    return 0;
}
