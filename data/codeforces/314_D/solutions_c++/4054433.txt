#include <iostream>
#include <algorithm>
#include <vector>
#include <cstdio>

#define rep(i,n) for(int i=0;i<n;i++)
#define pi pair<long long, long long>
#define ll long long
#define MS 2000000010
#define fst first
#define snd second

using namespace std;

int n;
ll t1,t2,l,r,byl[100010],byr[100010],syl[100010],syr[100010];
vector< pi > v;

int main(){
    cin>>n;
    rep(i,n){
    cin>>t1>>t2;
    v.push_back(pi(t1+t2,t1-t2));
    }
    sort(v.begin(),v.end());

    byl[0] = syl[0] = v[0].snd;
    rep(i,n-1){
    byl[i+1] = max(byl[i],v[i+1].snd);
    syl[i+1] = min(syl[i],v[i+1].snd);
    }

    byr[n-1] = syr[n-1] = v[n-1].snd;

    for(int i=n-2;i>=0;i--){
    byr[i] = max(byr[i+1],v[i].snd);
    syr[i] = min(syr[i+1],v[i].snd);
    }
    r = MS;

    rep(ii,100){
    ll m = (l+r)/2;
    int flag = 0;
    int rx  = 0;

    rep(i,n){
        while(v[rx].fst-v[i].fst<=m && rx<n)rx++;
        if(i==0 && rx==n)flag = 1;
        else if(i==0){if(byr[rx]-syr[rx]<=m)flag = 1;}
        else if(rx==n){if(byl[i-1]-syl[i-1]<=m)flag=1;}
        else {
        ll by = max(byl[i-1],byr[rx]);
        ll sy = min(syl[i-1],syr[rx]);
        if(by-sy<=m)flag = 1;
        }
        if(flag)break;
    }
    if(flag)r=m;
    else l=m;
    }
    printf("%.6lf\n",(double)r/2);

}
