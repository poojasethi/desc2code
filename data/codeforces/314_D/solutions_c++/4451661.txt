#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <cmath>
#define N 100010
#define db long double
#define inf 1e10
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	db x,y;
} a[N];
int i,j,k,n,m,xx,yy;
db LU[N],RU[N],LD[N],RD[N];
inline bool judge(db len) {
	int i,j; j=0;
	For(i,1,n) {
		for (;a[j+1].x+len<a[i].x;j++);
		if (max(LU[j],RU[i+1])-min(LD[j],RD[i+1])<=len) return 1;
	}
	return 0;
}
inline bool cc1(const ww &a,const ww &b) {
	return a.x<b.x;
}
int main() {
	//freopen("314d.in","r",stdin);
	//freopen("314d.out","w",stdout);
	scanf("%d",&n);
	For(i,1,n) {
		scanf("%d%d",&xx,&yy);
		a[i].x=yy-xx,a[i].y=xx+yy;
	}
	sort(a+1,a+n+1,cc1);
	LD[0]=RD[n+1]=inf; LU[0]=RU[n+1]=-inf;
	For(i,1,n) LD[i]=min(LD[i-1],a[i].y),LU[i]=max(LU[i-1],a[i].y);
	for (i=n;i;i--) RD[i]=min(RD[i+1],a[i].y),RU[i]=max(RU[i+1],a[i].y);
	db l=0,r=inf,mid;
	for (;l+1e-7<r;) {
		mid=(l+r)/2;
		if (judge(mid)) r=mid;
		else l=mid;
	}
	double an=r;
	printf("%.10lf\n",an/2);
	//for(;;);
	return 0;
}
