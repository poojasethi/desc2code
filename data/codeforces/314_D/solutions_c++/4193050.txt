#include<stdio.h>
#include<algorithm>
#include<map>
#define INF 1000000000000000LL
using namespace std;
typedef pair<long long,long long> P;
P p[100000];
P myf[100000];
P myb[100000];
long long min(long long a,long long b){return a<b?a:b;}
long long max(long long a,long long b){return a>b?a:b;}

int main(){
    int n,i;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        long long ox,oy;
        scanf("%I64d %I64d",&ox,&oy);
        p[i]=P((ox-oy)*2,(ox+oy)*2);
    }
    sort(p,p+n);
    myf[0]=P(p[0].second,p[0].second);
    for(i=1;i<n;i++)myf[i]=P(min(myf[i-1].first,p[i].second),max(myf[i-1].second,p[i].second));
    myb[n-1]=P(p[n-1].second,p[n-1].second);
    for(i=n-2;i>=0;i--)myb[i]=P(min(myb[i+1].first,p[i].second),max(myb[i+1].second,p[i].second));
    long long l=-1,r=INF;
    while(r-l>1){
        long long mid=(l+r)/2;
        int b=0;
        long long res=INF;
        for(i=0;i<n;i++){
            while(b<n&&p[b].first-p[i].first<=mid*2)b++;
            long long upper=max((i==0?-INF:myf[i-1].second),(b==n?-INF:myb[b].second));
            long long lower=min((i==0?INF:myf[i-1].first),(b==n?INF:myb[b].first));
            res=min(res,upper-lower);
        }
        if(res<=mid*2)r=mid;
        else l=mid;
    }
    printf("%.12lf\n",(double)r/2);
    return 0;
}
