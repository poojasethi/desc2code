#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
#define N 100005
#define rep(i,n) for (int i=0;i<n;i++)
ll l,r,mid,LD[N],LU[N],RD[N],RU[N],x[N],y[N]; int p[N],n,X,Y;
bool cmp(const int i,const int j){return x[i]<x[j];}
bool check()
{
	int j=0;
	rep(i,n){
		while (j<n && (x[p[j]]-x[p[i]])<=mid) j++;
		if (max(RU[j],LU[i])-min(RD[j],LD[i])<=mid) return 1;
		}
	return 0;
}
int main()
{
	scanf("%d",&n),r=1LL<<35,LD[0]=RD[n]=1LL<<50,LU[0]=RU[n]=-1LL<<50;
	rep(i,n) scanf("%d%d",&X,&Y),x[i]=X+Y,y[i]=X-Y,p[i]=i; sort(p,p+n,cmp);
	rep(i,n) LD[i+1]=min(LD[i],y[p[i]]),LU[i+1]=max(LU[i],y[p[i]]);
	for (int i=n;i;i--) RD[i-1]=min(RD[i],y[p[i-1]]),RU[i-1]=max(RU[i],y[p[i-1]]);
	while (l<r) mid=(l+r)/2,check()?r=mid:l=mid+1;
	printf("%.9lf\n",l/2.); return 0;
}
