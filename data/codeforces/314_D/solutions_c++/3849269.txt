using namespace std;

typedef long long ll ;
typedef pair<ll,ll> pl ;

const int MAXN = 100010 ;

int n ;
pl p[MAXN] ;
ll pmx[MAXN],pmi[MAXN] ;

bool check(ll mid){
    ll mx=-2e9,mi=2e9;
    for (int i=0,j=0;i<n;i++){
        while (j<n&&p[j].first<p[i].first-mid){
            mx=max(mx,p[j].second) ;
            mi=min(mi,p[j].second) ;
            j++;
        }
        ll tmx = max(pmx[i+1],mx);
        ll tmi = min(pmi[i+1],mi);
        if (tmx-tmi<=mid) return true;
    }
    return false;
}

int main(){
    cin>>n;
    for (int i=0;i<n;i++){
        int x,y ;
        cin>>x>>y ;
        p[i] = make_pair(x-y,x+y) ;
    }
    sort(p,p+n) ;
    ll l = -4e9 , r = 4e9 , ans ;
    pmx[n] = -2e9 ; pmi[n] = 2e9 ;
    for (int i=n-1;i>=0;i--){
        pmx[i] = max(pmx[i+1],p[i].second) ;
        pmi[i] = min(pmi[i+1],p[i].second) ;
    }
    while (l<=r){
        ll mid = (l+r)>>1;
        if (check(mid)){
            ans=mid; r=mid-1;
        }else
            l=mid+1;
    }
    printf("%.13f\n",double(ans)/2.0);
}
