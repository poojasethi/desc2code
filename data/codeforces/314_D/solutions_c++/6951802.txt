#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
typedef pair<ll, ll> pii;
const long long INF = 2222222222;
const int N = 200005;
pii s[N];
ll Lmin[N], Lmax[N], Rmin[N], Rmax[N];
int n;
bool check(ll mid) {
	for(int i = 0, j = 0; i < n; i ++) {
		while(j < n && s[j].first <= s[i].first + mid) j++;
		
		ll mx = max(Lmax[i], Rmax[j]);
		ll mn = min(Lmin[i], Rmin[j]);
		if(mx - mn <= mid) return true;
	}
	return false;
}
int main() {
	ll x, y;
	scanf("%d", &n);
	for(int i = 0; i < n; i ++) {
		scanf("%I64d%I64d", &x , &y);
		s[i] = make_pair(x+y, x-y);
	}
	sort(s, s+n);
	
	Lmin[0] = INF, Lmax[0] = -INF;
	for(int i = 1 ; i <= n; i ++) {
		Lmin[i] = min(Lmin[i-1], s[i-1].second);
		Lmax[i] = max(Lmax[i-1], s[i-1].second);
	}
	Rmin[n] = INF, Rmax[n] = -INF;
	for(int i = n-1; i >= 0; i --) {
		Rmin[i] = min(Rmin[i+1], s[i].second);
		Rmax[i] = max(Rmax[i+1], s[i].second);
	}
	
	ll l = 0, r = INF;
	while(l + 1 < r) {
		ll mid = (l+r) / 2;
		if(check(mid)) r = mid;
		else l = mid;
	}
	if(check(l)) printf("%.10f\n", l / 2.0);
	else printf("%.10f\n", r / 2.0);
	
	return 0;
}
