#include<iostream>
using namespace std;
int main()
{
int n,m,a[105]={0},b[105]={0},i,x,y,z,k=0,p;
cin>>n>>m;
for(i=1;i<=m;i++)
{
cin>>x>>y;
b[x]=a[x];
a[x]=y;
b[y]=a[y];
a[y]=x;
}
for(i=1;i<=n;i++)
{
x=max(a[i],b[i]);
y=i;
p=1;
while(x>0 && x!=i)
{
p^=1;
z=y;
y=x;
x=a[x]+b[x]-z;
}
if(x==i && p) a[i]=b[i]=0,k++;
}
if((n-k)%2==0) cout<<k;
else cout<<k+1;
return 0;
}
