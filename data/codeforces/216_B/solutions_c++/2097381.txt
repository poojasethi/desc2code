#include<cstdio>
#include<algorithm>
using namespace std;
int fa[102], s[102];
int f(int x){ return fa[x]=(fa[x]==x?x:f(fa[x])); }
int main(){
	int n, m, a, b, ans = 0;
	scanf("%d%d", &n, &m);
	for(int i = 0; i < n; i++) fa[i] = i, s[i] = 1;
	for(int i = 0; i < m; i++){
		scanf("%d%d", &a, &b);
		int x = f(a), y = f(b);
		if(x==y && (s[x]&1)) ans++;
		fa[x] = fa[y], s[y] += s[x];
	}
	printf("%d\n", ans+((n-ans)&1));
}
