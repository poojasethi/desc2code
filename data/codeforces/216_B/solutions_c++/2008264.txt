#include <cstdio>
#define N 105

int n, m, s[N], z[N], A;

int f(int x){
	return x == s[x] ? x : s[x] = f(s[x]);
}

int main(){
	scanf("%d %d", &n, &m);
	for(int i = 1; i <= n; i++) s[i] = i, z[i] = 1;
	
	while(m--){
		int a, b;
		scanf("%d %d", &a, &b);
		
		if(f(a) == f(b))
			A += z[f(a)] & 1;
		else{
			z[f(b)] += z[f(a)];
			s[f(a)] = f(b);
		}
	}
	
	if(n - A & 1) A++;
	printf("%d\n", A);
}
