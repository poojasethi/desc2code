#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,u,v,k,r[1000],s[1000];
int get(int u){
    return (r[u]==u ? u: get(r[u]));
}
int main(){
    //freopen("a.in","r",stdin);
    scanf("%d %d\n",&n,&m);
    for(int i=1;i<=n;++i) {r[i]=i;s[i]=1;}
    while (m--){
        scanf("%d %d\n",&u,&v);
        u=get(u);v=get(v);
        if (u==v && (s[u]&1)) k++;
        r[u]=r[v];s[v]+=s[u];
    }
    printf("%d\n",k + ((n-k)&1));
    return 0;
}
