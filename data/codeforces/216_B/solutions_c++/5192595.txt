#include<stdio.h>
int cnt[110],root[110];
int find(int x){
	return root[x] == x?x:find(root[x]);
}

int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	int i;
	for(i=0;i<=n;i++)
		root[i]=i,cnt[i]=1;
	int a,b,x,y;
	int ans =0;
	while(m--){
		scanf("%d%d",&a,&b);
		x = find(a);
		y = find(b);
		if(x==y && (cnt[y]%2==1))ans++;
		root[x] = root[y];
		cnt[y] += cnt[x];
	}
	if((n-ans)&1)ans++;
	printf("%d\n",ans);
	return 0;
}
