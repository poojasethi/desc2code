#include <bits/stdc++.h>

using namespace std;

vector<size_t>parent;
vector<size_t>rank;

size_t find(size_t x){
return x==parent[x] ? x : parent[x]=find(parent[x]);}

int main()
{int n,m,i,x,y,s=0,px,py;
scanf("%d %d",&n,&m);
parent.reserve(n+1);
rank.reserve(n+1);


for(i=0;i<=n;i++)
{parent[i]=i;
rank[i]=1;
}

while(m--)
{

scanf("%d %d",&x,&y);
px=find(x);
py=find(y);

if(px== py && rank[px]%2)
s++;
parent[py]=parent[px];
rank[px]+=rank[py];

}
if((n-s)%2) s++;
printf("%d",s);
    return 0;
}
