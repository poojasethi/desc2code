#include <cstdio>
int n,m,p[105],s[105],ans;
int f(int x){
    return x==p[x]?x:p[x]=f(p[x]);
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1; i<=n; ++i) p[i]=i,s[i]=1;
    ans=0;
    for(int i=1; i<=m; ++i){
        int u,v,x,y;
        scanf("%d%d",&u,&v);
        x=f(u),y=f(v);
        if (x==y) ans+=(s[x]&1);
        else s[x]+=s[y],p[y]=p[x];
    }
    if ((n-ans)%2==1) ++ans;
    printf("%d\n",ans);
    return 0;
}
