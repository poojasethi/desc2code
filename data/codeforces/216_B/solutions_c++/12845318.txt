#include<bits/stdc++.h>

using namespace std;
int a,b,arr[115],arrs[115];
int sfind(int x)
{
    if(x==arrs[x])return x;
    else sfind(arrs[x]);
}
int main()
{
cin>>a>>b;
int ans=0;
for(int i=0;i<a;i++)arrs[i]=i,arr[i]=1;

for(int i=0;i<b;i++)
{
    int h,j;
    cin>>h>>j;
   int s= sfind(h);
   int y=sfind(j);

    if(s==y&&arr[s]%2==1)ans++;

    arrs[y]=arrs[s];
    arr[s]+=arr[y];
}
if((a-ans)%2==1)ans++;
cout<< ans;

}
