#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
const int nMax = 110;

int f[nMax*2];
int getfa(int p){	return f[p]==p?p:getfa(f[p]);}
int main()
{
	int i,j,k,n,m,tot,u,v;
	while(scanf("%d%d",&n,&m)!=EOF){
		for(i=1;i<=n*2;i++)	f[i]=i;
		tot=0;
		while(m--){
			scanf("%d%d",&u,&v);
			if(getfa(u)==getfa(v))	tot++;
			else{
				f[getfa(u+n)]=f[v];
				f[getfa(v+n)]=f[u];
			}
		}
		if((n-tot)&1)	tot++;
		printf("%d\n",tot);
	}
	return 0;
}
