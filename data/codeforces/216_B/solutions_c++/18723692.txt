#include<bits/stdc++.h>
using namespace std;
int p[105],rnk[105];
int fnd(int x)
{
    if(p[x]==x)
        return x;
    return fnd(p[x]);
}
int main()
{
    int n,m,ans=0;
    cin>>n>>m;
    for(int i=1;i<=n;i++)
    {
        p[i]=i;
        rnk[i]=1;
    }
    for(int i=0;i<m;i++)
    {
        int u,v;
        cin>>u>>v;
        int a=fnd(u);
        int b=fnd(v);
        if(a==b&&rnk[a]%2)
            ans++;
        p[b]=p[a];
        rnk[a]+=rnk[b];
    }
    if((n-ans)%2)
        ans++;
    cout<<ans;
    return 0;
}
