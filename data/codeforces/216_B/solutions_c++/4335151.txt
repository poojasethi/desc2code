#include<iostream>
using namespace std;const int N = 111;int f[N], num[N];int find(int x){return x==f[x] ? x:f[x]=find(f[x]);}int main(){
    int n, m, a, b, x, y, ans;    while(cin>>n>>m)    {
        for(int i = 1;i<=n; i++)f[i]=i, num[i]=1;        ans = 0;        while(m--){
cin >> a >>b;x=find(a);y=find(b);if(x!=y)f[x]=y,num[y]+=num[x];
else if(num[x]&1)ans++;
}n-=ans;cout<<ans+(n&1)<<endl;
    }
}
