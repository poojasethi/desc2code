#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int fa[105],si[105];
int find(int x)
{
    if(x!=fa[x])fa[x]=find(fa[x]);
    return fa[x];
}
int main()
{
    int n,m,ans=0;
    cin>>n>>m;
    for(int i=1;i<=n;++i)fa[i]=i,si[i]=1;
    while(m--)
    {
        int x,y;
        cin>>x>>y;
        x=find(x);
        y=find(y);
        if(x!=y)fa[x]=y,si[y]+=si[x];
        else if(si[x]&1)++ans;
    }
    n-=ans;
    cout<<ans+(n&1)<<endl;
}
