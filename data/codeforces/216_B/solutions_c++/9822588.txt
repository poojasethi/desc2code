#include<iostream>
#include<vector>
using namespace std;
int n,m,ans,res,vis[110];
vector<int> v[110];
void dfs(int b,int l)
{
	if (vis[b]==1)
	{
		if (l%2==0)
		{
			res++;
			vis[b]=2;
		}
		return;
	}
	if (vis[b]!=0)
	{
		return;
	}
	vis[b]=l;
	for (int i=0;i<v[b].size();i++)
	{
		dfs(v[b][i],l+1);
	}
}
int main()
{
	cin>>n>>m;
	int x,y;
	for (int i=0;i<m;i++)
	{
		cin>>x>>y;
		v[x].push_back(y);
		v[y].push_back(x);
	}
	for (int i=1;i<=n;i++)
	{
		dfs(i,1);
	}
	if ((n-res)%2==1)
	{
		res++;
	}
	cout<<res<<endl;
}
