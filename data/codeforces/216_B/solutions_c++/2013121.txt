#include <stdio.h>
#include <vector>
#include <set>

using namespace std;
int a[105][105];
int n,m,rst,cnt;
bool v[105];
void go(int x,int be){
	v[x]=1;
	cnt++;
	for(int i=1;i<=n;i++){
		if(i==x)continue;
		if(i==be)continue;
		if(!a[x][i])continue;
		if(v[i]){
			if(cnt%2==1 &&cnt!=1)rst++;
			continue;
		}
		go(i,x);
	}
	cnt--;
}
int main(){
	int i,j,s,e;
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;i++){
		scanf("%d%d",&s,&e);
		a[s][e]=a[e][s]=1;
	}
	for(i=1;i<=n;i++)if(!v[i])go(i,0);
	n-=rst;
	if(n%2==1)rst++;
	printf("%d",rst);

}
