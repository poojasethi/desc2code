#include<cstdio>
int t[111],s[111];

int find(int p){
	return p==t[p]?p:find(t[p]);
}

int main(){
	int n=0, m=0, a=0, b=0, x=0, y=0, c=0;
	scanf("%d%d",&n,&m);

	for(int i=0;i<=n;i++)
		t[i]=i,s[i]=1;
		
	while(m--){
		scanf("%d%d",&a,&b);
		
		x=find(a);
		y=find(b);
		
		if(x==y&&(s[x]%2))c++;
		
		t[y]=t[x];
		s[x]+=s[y];
	}
	if((n-c)%2)c++;
	printf("%d",c);
	
	return 0;
}
