#include <iostream>
#include <algorithm>
using namespace std;
int f[200],s[200],a[105],b[105],c[105],ans;
int getfather(int x){
	if (f[x]==x){
		return x;
	}
	return getfather(f[x]);
}
int main(){
	int n,m;
	cin>>n>>m;
	for (int i=1;i<=n;i++){
		f[i]=i;
		s[i]=1;
	}
	for (int i=1;i<=m;i++){
		cin>>a[i]>>b[i];
		int fa=getfather(a[i]);
		int fb=getfather(b[i]);
		if(fa==fb && s[fa]%2==1){
			ans++;
		}
		f[fb]=fa;
		s[fa]+=s[fb];
		
	}
	if ((n-ans)%2==1){
		ans++;
	}
	cout<<ans;
	return 0;
}
