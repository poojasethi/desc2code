#include <stdio.h>
#include <iostream>
#include <map>
#include <cstring>
using namespace std;
map <int,int> mp;
int f[120],n,m,a,b,x,y,s=0;
int gf(int p){
	if (p!=mp[p]) return gf(mp[p]); else return p;
}
int main( ){
	cin>>n>>m;
	memset(f,1,sizeof(f));
	for (int i=1;i<=n;i++)
		mp[i]=i;
	for (int k=1;k<=m;k++){
		cin>>a>>b;
		x=gf(a);
		y=gf(b);
		if (x==y && (f[x]%2>0))
			s++;
		mp[y]=mp[x];
		f[x]+=f[y];
	}
	if ((n-s)%2>0) s++;
	cout<<s<<endl;
	return 0;
}
