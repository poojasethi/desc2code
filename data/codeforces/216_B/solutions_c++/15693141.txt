#include <bits/stdc++.h>
using namespace std;
int d[101][101],vis[101],r=0,n,m;
void dfs(int v,int depth)
{
    if(vis[v])
    {
        if(depth-vis[v]>1&&(depth-vis[v])%2)r++;
        return;
    }
    vis[v]=depth;
    for(int i=1;i<=n;i++)
        if(d[v][i])
        dfs(i,depth+1);
}
int main()
{
    int i,a,b;
    cin>>n>>m;
    for(i=0;i<m;i++)
    {
        cin>>a>>b;
        d[a][b]=d[b][a]=1;
    }
    for(i=1;i<=n;i++)
        if(!vis[i])
        dfs(i,1);
    cout<<r+(n-r)%2<<endl;
    return 0;
}
