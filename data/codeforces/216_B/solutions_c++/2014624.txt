#include<cstdio>

int n,m;
int bleg[110],cnt[110];

int find(int x){
    while(x!=bleg[x])
        x=bleg[x];
    return x;
}

int main()
{
    int i,a,b,sum=0;
    scanf("%d%d",&n,&m);
    int tt=n;
    for(i=0;i<=n;i++)
        bleg[i]=i,cnt[i]=1;
    for(i=0;i<m;i++){
        scanf("%d%d",&a,&b);
        int ra=find(a),rb=find(b);
        if(ra==rb && cnt[rb]&1) sum++,tt--;
        else bleg[ra]=rb,cnt[rb]+=cnt[ra];
    }
    if(tt%2) sum++;
    printf("%d\n",sum);
    return 0;
}
