#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
int f[150],s[150];
int gf(int x)
{
	if(f[x]==x)return x; else return gf(f[x]);
}
int main(){
	int n,m,x,y,fx,fy,ans=0;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	{
		f[i]=i;
		s[i]=1;
	}
	for(int i=1;i<=m;i++)
	{
		cin>>x>>y;
		fx=gf(x);fy=gf(y);
		if(fx==fy&&s[fx]%2==1)ans++;
		f[fy]=fx;s[fx]+=s[fy];
	}
	if((n-ans)%2==1)ans++;
	cout<<ans;
}
