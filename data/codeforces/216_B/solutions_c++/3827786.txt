#include <iostream>
using namespace std;
int g[110],r[110];
int n,m,a,b,ans;
int FIND(int x)
{
	return (g[x]==x?x:g[x]=FIND(g[x]));
}
int main()
{
	cin >>n >>m;
	for (int i=1; i<=m; i++)
	{
		cin >>a >>b;
		if (g[a]==0) {g[a]=a; r[a]=1;}
		if (g[b]==0) {g[b]=b; r[b]=1;}
		a=FIND(a);
		b=FIND(b);
		if (a==b && r[a]%2==1) ans++;
		else {g[a]=b; r[b]+=r[a];}
    }
    if ((n&1)!=(ans&1)) ++ans;
    cout <<ans <<endl;
    return 0;
}
