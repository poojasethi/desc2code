#include<stdio.h>

int fa[105];
int find(int x)
{
    if(!fa[x]) return x;
    return fa[x]=find(fa[x]);
}

int main()
{
    int n,m,i,sum;
    scanf("%d%d",&n,&m);

    for(sum=i=0;i<m;i++)
    {
        int x,y;
        scanf("%d%d",&x,&y);
        x=find(x); y=find(y);
        if(x!=y) fa[x]=y;
        else
        {
            int j,s=0;
            for(j=1;j<=n;j++)
             if(find(j)==x) s++;
            if(s&1) sum++;
        }
    }

    if((n-sum)&1) sum++;
    printf("%d\n",sum);
}
