def dfs(x, c):
	if not clr[x] == -1: return clr[x] == c
	clr[x] = c
	ret = True
	for i in xrange(n):
		if g[x][i]:
			ret&=dfs(i, 1-c)
	return ret


n,m = [int(i) for i in raw_input().split()]
g = [[False]*n for i in xrange(n)]

for i in xrange(m):
	a, b = [int(i)-1 for i in raw_input().split()]
	g[a][b] = g[b][a] = True


clr = [-1]*n
ret = 0
for i in xrange(n):
	if clr[i] == -1:
		if not dfs(i, 0):
			ret+=1
if (n - ret)%2 == 1:
	ret+=1
print ret
