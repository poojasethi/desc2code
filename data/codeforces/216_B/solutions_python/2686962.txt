n, m = map(int, raw_input().split())
e = [[] for _ in range(n)]
for _ in range(m):
	x, y = map(int, raw_input().split())
	e[x - 1] += [y - 1]
	e[y - 1] += [x - 1]
t = [0] * n

def f(u, v = -1, c = 0):
	if t[u]:
		return c & 1
	t[u] = 1
	for w in e[u]:
		if w != v:
			return f(w, u, c + 1)
	return 0

r = 0
[f(i) for i in range(n) if len(e[i]) == 1]
for i in range(n):
	if not t[i]:
		r += f(i)
print n - (n - r) / 2 * 2
