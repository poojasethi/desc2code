n,m=map(int,raw_input().split())
adj=[[] for k in range(n)]
for i in range(m):
  u,v=map(int,raw_input().split())
  adj[u-1].append(v-1)
  adj[v-1].append(u-1)
  
state = [0]*n
def dfs(root, value):
  state[root]=value
  for k in adj[root]:
    if not state[k]:
      dfs(k,3-value)

for r in range(n):
  if not state[r]:
    dfs(r,1)

res=0
for u in range(n):
  for v in adj[u]:
    res += u<v and state[u]==state[v]
if (n-res)%2:
  res+=1
print res
