#!/usr/bin/env python
n, m = map(int, raw_input().strip().split())
a = [[0]*(n+1) for i in xrange(n+1)]
for i in xrange(m):
    u,v = map(int, raw_input().strip().split())
    a[u][v] = a[v][u] = 1

d = [-1]*(n+1)

def dfs(i,v):
    d[i] = v;
    for j in xrange(1,n+1):
        if a[i][j] and d[j]==-1:
            dfs(j,1-v)

for i in xrange(1,n+1):
    if d[i]==-1:
        dfs(i,0)
res = 0
for i in xrange(1,n+1):
    for j in xrange(i+1,n+1):
        if a[i][j] and d[i]==d[j]:
            res+=1
if (n-res)%2==1: 
    res+=1
print res
