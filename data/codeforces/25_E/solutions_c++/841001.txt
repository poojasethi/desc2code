const int MAXN=100005;
char cs[MAXN];
string s[3];
int p[3];
int q[MAXN*3];
const int inf=10000000;
string doit(string a,string b)
{
    int n=a.length(),m=b.length();
    string c=a+"#"+b;
    int len=n+m+1;
    q[0]=-1;
    for (int i=1;i<len;i++)
    {
        int t=q[i-1];
        while (t>-1 && c[t+1]!=c[i]) t=q[t];
        if (c[t+1]==c[i])t++;
        q[i]=t;
    }
    for (int i=n+1;i<len;i++)
        if (q[i]==n-1) return b;
    return b.substr(0,m-q[len-1]-1)+a;
}
int main()
{
    for (int i=0;i<3;i++)
    {
        scanf("%s",cs); s[i]=string(cs);
        p[i]=i;
    }
    int ans=inf;
    do
    {
        string tret=doit(s[p[0]],doit(s[p[1]],s[p[2]]));
        ans=min(ans,(int)tret.length());
    }while(next_permutation(p,p+3));
    printf("%d\n",ans);
    return 0;
}
