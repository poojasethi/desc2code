#include<cstdio>
#include<cstring>
#define maxlen 200010
using namespace std;

char s[4][maxlen];

int MIN(int x,int y) { if(x<y) return x; return y; }

int p[maxlen];
int kmp(int a,int b) //how many s[b]'s chars are left 
{
	int i,j,k,m=strlen(s[b]),n=strlen(s[a]);
	p[0]=-1; j=-1;
	for(i=1;i<m;i++)
	{
		while(j>=0 && s[b][i]!=s[b][j+1]) j=p[j];
		if(s[b][i]==s[b][j+1]) j++;
		p[i]=j;
	}
	j=-1;
	for(i=0;i<n;i++)
	{
		if(j==m-1) return 0;
		while(j>=0 && s[a][i]!=s[b][j+1]) j=p[j];
		if(s[a][i]==s[b][j+1]) j++;
	}
	return (m-1-j);
}

int solve(int st,int nd,int rd)
{
	int len=kmp(st,nd);
	s[0][0]='\0';
	strcat(s[0],s[st]);
	strcat(s[0],s[nd]+strlen(s[nd])-len);
	len=kmp(0,rd);
	return strlen(s[0])+len;
}

int main()
{
	int i;
	for(i=1;i<=3;i++)
		gets(s[i]);
	
	int ans=solve(1,2,3);
	ans=MIN(ans,solve(1,3,2));
	ans=MIN(ans,solve(2,1,3));
	ans=MIN(ans,solve(2,3,1));
	ans=MIN(ans,solve(3,1,2));
	ans=MIN(ans,solve(3,2,1));
	
	printf("%d\n",ans);
	
	return 0;
}
