#include <cstdio>
#include <iostream>

#include <string>

using namespace std;

string s[3];
int pi[300010];

inline void kmp(string &x,string &y)
{
    pi[1] = 0;
    int q = 0;

    for(size_t i = 1; i < y.size(); ++i)
    {
        while(q>0 && y[q]!=y[i])
            q = pi[q];
        if(y[q]==y[i])
            ++q;
        pi[i+1] = q;
    }

    q = 0;
    for(size_t i = 0; i < x.size(); ++i)
    {
        while(q>0 && y[q]!=x[i])
            q = pi[q];
        if(y[q]==x[i])
            ++q;
        if(q==y.size())
            return;
    }
    x+=y.substr(q);
}

inline int rezolva(string x, string y, string z)
{
    kmp(x,y);
    kmp(x,z);
    return x.size();
}
