string combine(string a, string b)
{
    string c=a+"$"+b;
    int p[c.size()];
    p[0]=-1;
    for (int i=1; i<c.size(); i++)
    {
        int t=p[i-1];
        while (t>=0 && c[t+1]!=c[i])
        {
            t=p[t];
        }
        if (c[t+1]==c[i]) t++;
        p[i]=t;
    }
    for (int i=a.size()+1; i<c.size(); i++)
    {
        if (p[i]==a.size()-1) return b;
    }
    int len=p[c.size()-1]+1;
    return b.substr(0,b.size()-len)+a;
}
