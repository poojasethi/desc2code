#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
int n[3],x[] = {0,1,2},ans,z[300003],n_s3;
char s[3][100003],s2[300003],s3[300003];
int w(char* s0,char* s1, int a,int b){
  for(int i=1;i<=b;i++) s2[i] = s1[i]; s2[b+1] = '$';
  for(int i=1;i<=a;i++) s2[i+b+1] = s0[i];
  for(int i=2,run;i<=a+b+1;i++){
    run = z[i-1];
    while(run != 0 && s2[run+1] != s2[i] ) run = z[run];
    if(s2[run+1] == s2[i]) run++;
    z[i] = run;
    if(i-z[i] >= b && z[i] == b) return b;
  }
  return min(b,z[a+b+1]);
}
int main(){
  for(int i=0;i<3;i++)
    scanf("%s",s[i]+1), n[i] = strlen(s[i]+1),ans += n[i];
  do{
    int sv;
    sv = w(s[x[0]], s[x[1]],n[x[0]],n[x[1]]);
    for(int i=1;i<=n[x[0]];i++)
      s3[i] = s[x[0]][i];
    n_s3 = n[x[0]];
    for(int i=sv+1;i<=n[x[1]];i++)
      s3[++n_s3] = s[x[1]][i];
    ans = min(n_s3 + n[x[2]] - w(s3,s[x[2]],n_s3,n[x[2]]),ans);
  }while(next_permutation(x,x+3));
  printf("%d\n",ans);
  return 0;
}
