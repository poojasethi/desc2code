#include <cstdio>
#include <string>
#include <iostream>
#define MAXN 100000
using namespace std;
string s1,s2,s3;
int F[MAXN+1];
int res=1000000000;
void preKMP(string str)
{
	int i=0,j=-1;
	int N=str.size();
	for(i=0;i<=N;i++)
		F[i]=0;
	i=0; j=-1;
	F[i]=j;
	while(i<N)
	{
		while(j>=0 && str[i]!=str[j])
			j=F[j];
		i++; j++;
		F[i]=j;
	}
}
string KMP(string str1,string str2)
{
	int N,M;
	M=str1.size(); N=str2.size();
	int i=0,j=0;
	int maxx=-1,maxi=0;
	while(i<M)
	{
		while(j>=0 && str1[i]!=str2[j])
			j=F[j];
		i++; j++;
		if(j==N) return str1;
		if(i==M) return str1+str2.substr(j);
	}
}
void f(string str1,string str2,string str3)
{
	string str;
	preKMP(str2);
	str=KMP(str1,str2);
	preKMP(str3);
	str=KMP(str,str3);
	res=min((int)str.size(),res);
}
int main()
{
	cin >> s1 >> s2 >> s3;
	f(s1,s2,s3);
	f(s1,s3,s2);
	f(s2,s1,s3);
	f(s2,s3,s1);
	f(s3,s1,s2);
	f(s3,s2,s1);
	printf("%d\n",res);
	return 0;
}
