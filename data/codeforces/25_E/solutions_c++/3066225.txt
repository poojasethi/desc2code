string did(const string &a,const string &b){
	string c=a+'&'+b;
	int *p=new int[c.size()];
	p[0]=-1;
	for (int q=1;q<(int)c.size();q++){
		int t=p[q-1];
		while(t>=0 && c[t+1]!=c[q])t=p[t];
		if(c[t+1]==c[q])t++;
		p[q]=t;
	}
	for(int i=a.size()+1;i<(int)c.size();i++)
	{
		if(p[i]==(int)a.size()-1){
			delete[]p;
			return b;
		}
	}
	int len=p[c.size()-1]+1;
	delete[]p;
	return b.substr(0,b.size()-len)+a;
}
