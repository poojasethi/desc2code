string cal(string a,string b){

    int k=n+m+1;
    string c=a+'$'+b;
    next[0]=-1;
    for(i=1;i<k;i++){
        int t=next[i-1];
        while(t!=-1&&c[t+1]!=c[i]) t=next[t];
        if(c[t+1]==c[i]) t++;
        next[i]=t;
    }
    for(i=n+1;i<k;i++){
        if(next[i]==n-1) return b;
    }
    return b.substr(0,m-next[k-1]-1)+a;
}
