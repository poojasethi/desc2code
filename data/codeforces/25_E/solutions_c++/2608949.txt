#include<iostream>
#include<climits>
#include<cstring>
#include<algorithm>
#include<string>
#define NN 100001
using namespace std;

string s[3] ;
int b[3][NN],o[] ={0,1,2};

void prekmp( int x ){
	int i = 0 ,j = -1; 
	b[x][i] = j;
	while( i < s[x].size() ){
		while( j >= 0 && s[x][i] != s[x][j] ) j = b[x][j];
		++i; ++j;
		b[x][i] = j;
	}
}

string join( string t , int y ){
	int i = 0, j = 0 ; 
	while( i < t.size() ){
		while( j >= 0 && t[i] != s[y][j] ) j = b[y][j];
		++i; ++j;
		if( j == s[y].size()  ) return t;
	}
	return t + s[y].substr(j);
}

int main(){
	for( int j= 0 ; j < 3; ++j)
		cin >> s[j] , prekmp( j );
	
	int ans = INT_MAX;
	string t;
	do{
		t = join( s[o[0]] , o[1] );
		t = join( t , o[2] );
		ans = min(ans ,(int) t.size() );
	}while( next_permutation( o , o+3 ) );
	cout<< ans << endl;
	return 0;
}
