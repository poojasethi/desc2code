#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
#define N 300005
int q[3];
int best;
string s[3];
int p[N];
string gao(string s1,string s2){
    int n=s1.size(),m=s2.size();
    string s3=s2+'#'+s1;
    for(int i=1;i<=n+m;i++){
        int t=p[i-1];
        while(t&&s3[t]!=s3[i]) t=p[t-1];
        if(s3[t]==s3[i]) t++;
        p[i]=t;
    }
    for(int i=m+1;i<=n+m;i++){
        if(p[i]==m) return s1;
    }
    return s1+s2.substr(p[n+m],m-p[n+m]);


}

   // cout<<gao("1234","3456")<<" "<<gao("1234","23")<<endl;

    best=N;
    q[0]=0,q[1]=1,q[2]=2;
    do{
        string t=gao(gao(s[q[0]],s[q[1]]),s[q[2]]);
        if(best>t.size()) best=t.size();
    }while(next_permutation(q,q+3));
    cout<<best<<endl;
}
