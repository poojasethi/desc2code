#include <bits/stdc++.h>
using namespace std;

const int MAXN = 3e5 + 100;

int ar[6][3]={{0,1,2} , {0,2,1} , {1,0,2} , {1,2,0} , {2,0,1} , {2,1,0}} , ans = MAXN , fail[MAXN];
string str[3];

string f( string x , string y ){
	
	string a = ' ' + y + ' ' + x;
	int n = a.size()-1;
	
	for( int i=2 , p=0 ; i <= n ; i++ ){
		
		while( p and a[p+1] != a[i] )
			p = fail[p];
		
		if( a[p+1] == a[i] )
			p++;
		
		fail[i] = p;
		
		if( p == (int)y.size() )
			return x;
	}
	
	x.resize( x.size() - fail[n] );
	return x + y;
}

int main(){
	
	cin >> str[0] >> str[1] >> str[2];
	
	for( int i=0 ; i < 6 ; i++ )
		ans = min( ans , (int)f( f(str[ar[i][0]] , str[ar[i][1]]) , str[ar[i][2]] ).size() );
	
	printf("%d\n",ans);
	
	return 0;
}
