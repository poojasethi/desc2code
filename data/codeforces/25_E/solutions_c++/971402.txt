#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
#include<iostream>
using namespace std;
string s,s1,s2,s3;
int minx,len,next[300010];
string unit(string s1,string s2){
    s='#'+s2+'#'+s1;len=s.size()-1;
    next[1]=0;
    for(int i=2,j=0;i<=len;++i){
        while(j&&s[j+1]!=s[i])j=next[j];
        if(s[j+1]==s[i])++j;
        if(j==(int)s2.size())return s1;
        next[i]=j;
    }
    if(next[len]==0)s=s1+s2;
    else s=s1.substr(0,s1.size()-next[len])+s2;
    return s;
}
void solve(string s1,string s2,string s3){
    s1=unit(s1,s2);s1=unit(s1,s3);
    if((int)s1.size()<minx)minx=s1.size();
}
int main(){
    while(cin>>s1>>s2>>s3){
        minx=0x7FFFFFF;
        solve(s1,s2,s3);
        solve(s1,s3,s2);
        solve(s2,s3,s1);
        solve(s2,s1,s3);
        solve(s3,s1,s2);
        solve(s3,s2,s1);
        printf("%d\n",minx);
    }
    return 0;
}
