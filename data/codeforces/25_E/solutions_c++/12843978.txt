
#include<bits/stdc++.h>
using namespace std ;

#define mx 100010

int p[3] ;
string s[3] ;
int B[4*mx] ;

string cal( string a , string b )
{
    string c = a + '#' + b ;
    int n = a.size() ;
    int m = b.size() ;

    int k = n + 1 + m ;

    int i = 0 ;
    int j = -1 ;
    B[0] = -1 ;

    for( i = 1 ; i < k ; i++ )
    {
        j = B[i-1] ;
        while( j >= 0 && c[i] != c[j+1] ) j = B[j] ;

        if( c[i] == c[j+1] ) j++ ;

        B[i] = j ;
    }

    for(  i = n+1 ; i < k ; i++ )
    {
        if( B[i] + 1 == n ) return b ;
    }

    return b.substr(0,m-B[k-1] - 1 ) + a ;
}
