#define INF 10000000

using namespace std;

int pi[300002];

string compute_prefix(string a, string b){
	memset(pi,0, sizeof(pi));
	int m = a.size(), n = b.size(),k = m + n + 1, q;
	string c = a + string("|") + b;
	pi[0] = -1;
	for(int i = 1; i < k; i++){
		q = pi[i - 1];
		while(q >= 0 && c[q + 1] != c[i])q = pi[q];
		if(c[q + 1] ==  c[i])q++;
		pi[i] = q;
	}
	for(int i = m + 1; i < k; i++) if(pi[i] ==  m - 1) return b;
	return b + a.substr(pi[k - 1] + 1);
}

int main(){
	int p[3], ans = INF;
	string s[3];
	for(int i = 0; i < 3; i++){
		p[i] = i; cin >> s[i];
	}
	do{
		ans = min(ans, (int)compute_prefix(compute_prefix(s[p[0]], s[p[1]]), s[p[2]]).size());
	}while(next_permutation(p, p + 3));
	printf("%d\n", ans);
	return 0;
}
