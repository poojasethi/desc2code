#include <iostream>
#include <string>
#include <vector>
#include <cmath>

using namespace std;

int longitud(string s1,string s2)
{
  string s=s2+"#"+s1;
  int n=int(s.size());
  int j=-1;
  vector<int> anterior(n);
  anterior[0]=-1;
  for (int i=1;i<n;i++) {
    while (j!=-1 and s[j+1]!=s[i]) j=anterior[j];
    if (s[j+1]==s[i])
      j++;
    anterior[i]=j;
    if (j==int(s2.size())-1) return int(s1.size());
  }
  return int(s1.size())+int(s2.size())-1-anterior[n-1];
}

int main()
{
  vector<string> v(3);
  for (int i=0;i<3;i++) cin>>v[i];
  vector<vector<int> > len(3,vector<int> (3));
  for (int i=0;i<3;i++)
    for (int j=0;j<3;j++)
      if (i!=j)
	len[i][j]=longitud(v[i],v[j]);
  int minimo=1000000000;
  for (int i=0;i<3;i++)
    for (int j=0;j<3;j++)
      for (int k=0;k<3;k++)
	if (i!=j and i!=k and j!=k)
	  if (len[i][j]==int(v[i].size()))
	    minimo=min(minimo,len[i][k]);
	  else
	    minimo=min(minimo,len[i][j]+len[j][k]-int(v[j].size()));
  cout<<minimo<<endl;
}
