string cal(string x,string y)
{
    string tem=y+'-'+x;
    memset(p,-1,sizeof(p));
    int n=tem.length(),k=-1,ly=y.length();
    for(int i=1;i<n;++i)
    {
        while(~k && tem[k+1]!=tem[i]) k=p[k];
        if(tem[k+1]==tem[i]) ++k;
        p[i]=k;
        if(p[i]==ly-1)return x;
    }
    return x+y.substr(p[n-1]+1,ly-p[n-1]);
}
