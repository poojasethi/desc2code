#include<bits/stdc++.h>
using namespace std;
const int inf = 0x7fffffff;
const int maxn = 100010;
int p[3];
string s[3];
int f[maxn*3];
string xf(string s1,string s2)
{
	string ss=s1+'@'+s2;
	f[0]=f[1]=0;
	int len1=s1.size(),len2=s2.size(),len=ss.size();
	for(int i=1;i<len;i++)
	{
		int j=f[i];
		while(j&&ss[j]!=ss[i]) j=f[j];
		f[i+1]=ss[i]==ss[j]?j+1:0;
		if(i+1>len1&&f[i+1]==len1) return s2;
	}
	return s2+s1.substr(f[len],s1.size()-f[len]);
}
int main()
{
	cin>>s[0]>>s[1]>>s[2];
	for(int i=0;i<3;i++) p[i]=i;
	int ans=inf;
	do
	{
		string s1=s[p[0]],s2=s[p[1]],s3=s[p[2]];
		ans=min(ans,(int)xf(xf(s1,s2),s3).size());
	}
	while(next_permutation(p,p+3));
	printf("%d\n",ans);
	return 0;
}
