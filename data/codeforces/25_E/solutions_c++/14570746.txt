#include<stdio.h>
#include<string>
#include<string.h>
#include<iostream>
#include<algorithm>
#define M 100005
#define oo 100000007
#define ull unsigned long long
using namespace std;
string str[3];
char str1[M];
int max(int a,int b){
	if(a>b)return a;
	return b;
}
int overlap(string a,string b){
	int ans=0,al=a.length(),bl=b.length();
	ull ah=0,bh=0,t=1;
	for(int i=1;i<=min(al,bl);i++){
		ah=ah+a[al-i]*t;
		bh=bh*oo+b[i-1];
		if(ah==bh)ans=i;
		t*=oo;
	}return ans;
}
int solve(string a,string b,string c){
	int res1=overlap(a,b),bl=b.length();
	string A=a;
	for(int i=res1;i<bl;i++){
		A+=b[i];
	}int res2=overlap(A,c);
	res2=max(res2,overlap(c,A));
	return res1+res2;
}
int main(){
	int i,t=0,ans=0,mx=0;
	for(i=0;i<3;i++){
		scanf("%s",str1);
		str[i]=str1;
		ans+=str[i].length();
	}
	for(i=0;i<3;i++)
		for(int j=0;j<3;j++){
			if(j==i)continue;
			for(int k=0;k<3;k++){
				if(k==j||k==i)continue;
				mx=max(solve(str[i],str[j],str[k]),mx);
			}
		}
	printf("%d",ans-mx);
	return 0;
}
