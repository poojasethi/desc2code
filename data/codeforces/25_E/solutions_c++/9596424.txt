#include<bits/stdc++.h>
using namespace std;
#define f(i,a,b) for(int i=a;i<=b;i++)
int c=3000000,*b=new int[c];
string unir(string A,string B)
{
    string C=A+"_"+B;
    int m=C.size();
    int i=0,j=-1;
    b[i]=-1;
    while(i<m)
    {
        while(j>=0&&C[i]!=C[j])
            j=b[j];
        i++,j++;
        b[i]=j;
    }
    f(i,A.size()+1,m)
        if(b[i]==A.size())
            return B;
    return B.substr(0,B.size()-b[m])+A;
}
int main()
{
    string x,y,z;
    cin>>x>>y>>z;
    c=min(c,(int)(unir(unir(x,y),z)).size());
    c=min(c,(int)(unir(unir(x,z),y)).size());
    c=min(c,(int)(unir(unir(y,x),z)).size());
    c=min(c,(int)(unir(unir(y,z),x)).size());
    c=min(c,(int)(unir(unir(z,x),y)).size());
    c=min(c,(int)(unir(unir(z,y),x)).size());
   cout<<c<<endl;
   return 0;
}
