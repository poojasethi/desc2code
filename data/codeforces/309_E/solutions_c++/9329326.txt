#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;
const int MAXN=5000;
int tot[MAXN],limit[MAXN],d[MAXN],n;
int x[MAXN],y[MAXN];
bool F[MAXN];
bool check(int lim)
{
	for(int i=1;i<=n;i++)
		tot[i]=limit[i]=!(F[i]=true);
	for(int i=1;i<=n;i++)
	{
		int t=-1,now=-1,sum=0;
		for(int j=i;j<=n;j++)
		{
			sum+=tot[j];
			if(sum>j-i+1)
				return false;
			if(sum==j-i+1)
			{
				t=j;
				break;
			}
		}
		for(int j=1;j<=n;j++)
		{
			if(F[j]&&limit[j]&&i>limit[j])
				return false;
			if(F[j]&&(t==-1||limit[j]<=t&&limit[j])&&(now==-1||y[now]>y[j]||y[now]==y[j]&&x[now]<x[j]))
				now=j;
		}
		F[d[i]=now]=false;
		tot[limit[now]]--;
		limit[now]=0;
		for(int j=1;j<=n;j++)
			if(F[j]&&!limit[j]&&(x[j]<=y[now]&&x[j]>=x[now]||x[now]<=y[j]&&x[now]>=x[j]))
				limit[j]=i+lim,++tot[limit[j]];
	}
	return true;
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d%d",&x[i],&y[i]);
	int l=0,r=n,mid;
	while(l<r)
	{
		mid=l+r>>1;
		if(check(mid))
			r=mid;
		else
			l=mid+1;
	}
	check(r);
	for(int i=1;i<=n;i++)
		printf("%d ",d[i]);
	printf("\n");
	return 0;
}
