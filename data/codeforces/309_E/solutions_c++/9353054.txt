#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <string>
using namespace std;
const int Maxn=2005;
int n,num[Maxn],last[Maxn],cnt[Maxn];
bool e[Maxn][Maxn],flag[Maxn];

inline int get() {
	char ch;
	while (!isdigit(ch=getchar()));
	int v=ch-48;
	while (isdigit(ch=getchar())) v=v*10+ch-48;
	return v;
}

struct info{
	int l,r,i;
	bool operator <(const info &rhs)const{
		return r<rhs.r;
	}
}a[Maxn];

bool check(int limi) {
	for (int i=1;i<=n;i++) last[i]=n,flag[i]=0;
	for (int i=1,w=1;i<=n;i++) {
		num[i]=w; flag[w]=1;
		for (int j=1;j<=n;j++)
			if (!flag[j]&&e[w][j]&&i+limi<last[j]) last[j]=i+limi;
		for (int j=i-1;j<=n;j++) cnt[j]=0;
		for (int j=1;j<=n;j++)
			if (!flag[j]) ++cnt[last[j]];
		for (int j=i;j<=n;j++)
			if ((cnt[j]+=cnt[j-1])>j-i) return 0;
		for (w=i+1;w<=n;w++) if (cnt[w]==w-i) break;
		for (int j=1;j<=n;j++)
			if (!flag[j]&&last[j]<=w) {w=j; break;}
	}
	return 1;
}

int main() {
	n=get();
	for (int i=1;i<=n;i++)
		a[i].l=get(),a[i].r=get(),a[i].i=i;
	sort(a+1,a+n+1);
	for (int i=1;i<=n;i++)
		for (int j=i+1;j<=n;j++)
			if (a[j].l<=a[i].r) e[j][i]=e[i][j]=1;
	int l=0,r=n,mid;
	while (l<r) {
		mid=(l+r)>>1;
		if (check(mid)) r=mid;
		else l=mid+1;
	}
	check(l);
	for (int i=1;i<=n;i++)
		printf(i==n?"%d\n":"%d ",a[num[i]].i);
	return 0;
}
