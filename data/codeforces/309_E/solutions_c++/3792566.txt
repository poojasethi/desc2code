#include<cstdio>
#include<algorithm>
#define fo(i,a,b) for(int i=a;i<=b;++i)
using namespace std;
const int mn=2010,inf=1<<30;
int L[mn],R[mn],ans[mn];
int h[mn],s[mn];
bool del[mn],u[mn];
bool e[mn][mn];
int n;
bool check(int hi)
{
    int tt=0;
    fill(h+1,h+1+n,n);
    fill(del+1,del+1+n,0);
    fill(u+1,u+1+n,1);
    for(;;)
    {
        int k=0;
        fo(i,1,n)
            if(u[i]&&R[i]<R[k])k=i;
        ans[++tt]=k,del[k]=1;
        if(tt==n)break;
        fo(i,1,n)
            if(!del[i]&&e[k][i])
                h[i]=min(h[i],tt+hi);
        fill(s,s+n+1,0);
        fo(i,1,n)
            if(!del[i])++s[max(h[i]-tt,0)];
        fo(i,1,n-tt)s[i]+=s[i-1];
        fo(i,0,n-tt)
            if(s[i]>i)
                return 0;
        k=n;
        fo(i,1,n-tt)
            if(s[i]==i)
            {
                k=i;
                break;
            }
        fo(i,1,n)
            u[i]=!del[i]&&(h[i]<=tt+k);
    }
    return 1;
}
int main()
{
    scanf("%d",&n);
    fo(i,1,n)
        scanf("%d%d",L+i,R+i);
    fo(i,1,n)
        fo(j,i+1,n)
            if(L[i]<=R[j]&&R[i]>=L[j])
                e[i][j]=e[j][i]=1;
    int lo=0,hi=n,mid;
    R[0]=inf;
    while(lo<hi)
    {
        mid=(lo+hi)/2;
        if(check(mid))hi=mid;
        else lo=mid+1;
    }
    check(lo);
    fo(i,1,n-1)printf("%d ",ans[i]);
    printf("%d\n",ans[n]);
    return 0;
}
