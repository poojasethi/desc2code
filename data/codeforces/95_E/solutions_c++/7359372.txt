#include <cstdio>
#include <cstring>
using namespace std;
int n,m;
#define N 100005
int f[N],s[N],lucky[N],g[N],dp[N];

int getf(int x){
	return (f[x] == x)?x:f[x] = getf(f[x]);
}

int main(){
	lucky[0] = 1;
	for (int i=1;i<N;i++){
		lucky[i] = (i % 10 == 4 || i % 10 == 7)? lucky[i / 10] : 0;
	}
	scanf("%d%d",&n,&m);
	for (int i=0;i<n;i++){
		f[i] = i;
		s[i] = 1;
	}
	int x, y;
	for (int i=0;i<m;i++){
		scanf("%d%d",&x,&y);
		--x; --y;
		x = getf(x); y = getf(y);
		if (x != y){
			f[x] = y;
			s[y] += s[x];
		}
	}
	for (int i=0;i<n;i++){
		if (f[i] != i) continue;
		g[s[i]]++;
	}
	memset(dp,10,sizeof(dp));
	dp[0] = 0;
	for (int i=1;i<=n;i++) if (g[i]){
		int sum = g[i];
		for (int j=1;sum>0;j<<=1){
			if (j > sum) j = sum;
			sum -= j;
			int now = i * j;
			for (int k=n;k>=now;k--){
				if (dp[k] > dp[k-now] + j)
					dp[k] = dp[k-now] + j;
			}
		}
	}
	int ans = n + 1;
	for (int i=1;i<=n;i++)
		if (lucky[i] && ans > dp[i]){
			ans = dp[i];
		}
	if (ans > n) printf("-1"); else printf("%d",ans-1);
	return 0;
}
