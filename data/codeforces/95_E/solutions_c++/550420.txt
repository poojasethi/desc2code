#include<iostream>
#include<stdio.h>
#include<string>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;
#define rep(i,n) for(i=0;i<n;i++)
#define FOR(i,j,k) for(i=j;i<=k;i++)
#define FORD(i,j,k) for(i=j;i>=k;i--)
const long maxint=0x7fffffff,maxn=100000+5;
long n,m,sum[maxn],size[maxn],dp[maxn],f[maxn];
long fa(long v)
{
    if(f[v]==v)return v;
    f[v]=fa(f[v]);
    return f[v];
}

bool ok(long v)
{
    for(;v>0;v/=10)if((v%10!=4) && (v%10!=7))return 0;
    return 1;
}

int main()
{
    scanf("%d%d",&n,&m);
    long i,j,p,q,t1,t2;
    FOR(i,1,n)f[i]=i;
    FOR(i,1,m)
    {
        scanf("%d%d",&t1,&t2);
        f[fa(t1)]=fa(t2);
    }
    FOR(i,1,n)size[fa(i)]++;
    FOR(i,1,n)sum[size[i]]++;
    FOR(i,1,n)dp[i]=0x2fffffff;
    FOR(i,1,n)
    if(sum[i]>0)
    {
        for(p=1;sum[i]>0;p*=2)
        {
            if(p>sum[i])p=sum[i];
            sum[i]-=p;
            q=p*i;
            FORD(j,n,q)dp[j]=min(dp[j],dp[j-q]+p);
        }
    }
    long ans=maxint;
    FOR(i,1,n)
    if(ok(i))ans=min(ans,dp[i]-1);
    if(ans>n)printf("-1\n");else printf("%d\n",ans);
   
}
