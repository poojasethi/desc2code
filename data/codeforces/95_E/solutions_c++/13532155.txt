const int N=1e5+5;
const int mod=1e9+7;

int n,m,i,x,y,num,q,p,j,ans;
int dp[N],ata[N],S[N],C[N];

bool H[N];

int atabul(int x){
	return ata[x]==x?x:(ata[x]=atabul(ata[x]));
}

int islucky(int x){

	while(x){
		if(x%10!=4 and x%10!=7)
			return 0;
		x/=10;
	}

	return 1;
}

main(){

	scanf("%d %d",&n,&m);

	for(i=1 ; i<=n ; i++)
		ata[i]=i;

	for(i=1 ; i<=m ; i++){
		scanf("%d %d",&x,&y);
		ata[atabul(x)]=atabul(y);
	}

	for(i=1 ; i<=n ; i++)
		S[atabul(i)]++;

	for(i=1 ; i<=n ; i++){
		C[S[i]]++;
		dp[i]=INT_MAX/2;
	}

	for(i=1 ; i<=n ; i++){
		num=C[i];
		for(p=1 ; num>0 ; p*=2){
			p=min(p,num);
			num-=p;
			for(j=n-p*i ; j>=0 ; j--)
				dp[j+p*i]=min(dp[j]+p,dp[j+p*i]);
		}
	}

	ans=INT_MAX/2;

	for(i=1 ; i<=n ; i++)
		if(islucky(i))
			ans=min(ans,dp[i]);

	if(ans!=INT_MAX/2)
		cout<<ans-1<<endl;
	else cout<<-1<<endl;

}
