#include<iostream>
#include<string.h>

using namespace std;

int f[100001]={};
int c1[100001]={};
int c2[100001]={};
int dp[100001]={};

int find(int x)
{
	return f[x]==x ? x : f[x] = find(f[x]);
}

bool check(int n)
{
	bool f = true;

	while(n && f)
	{
		if(n%10 != 4 && n % 10 !=7)
		{
			f = false;
		}

		n/=10;
	}

	return f;
}

int main()
{
	int m,n,u,v,s,ans=1<<30;

	cin>>n>>m;

	for(int i = 1 ; i <= n ; ++i )	f[i] = i;

	for(int i = 1 ; i <= m ; ++i )
	{
		cin>>u>>v;

		f[ find(u) ] = find(v);
	}
	for(int i = 1 ; i <= n ; ++i )	c1[ find(i) ]++;
	for(int i = 1 ; i <= n ; ++i )	c2[ c1[i] ] ++;

	//for(int i = 1 ; i <= n ; ++i )	cout<<c2[i]<<' ';cout<<endl;
//	cout<<'e';
	memset(dp,0x7f,(n+1)*4);
	dp[0] = -1;

	for(int i = 1 ; i <= n ; ++i )
	{
		for(int k = 1 ; c2[i] ; k<<=1)
		{
			if( k > c2[i])	k = c2[i];
			c2[i] -= k;
			s = k*i;	
			for(int j = n ; j - s >= 0 ; --j )	dp[j] = min(dp[j],dp[j-s]+k);
		}
		
	//	for(int i = 1 ; i <= n ; ++i )  cout<<dp[i]<<' ';cout<<endl;
	}

	//for(int i = 1 ; i <= n ; ++i )	cout<<dp[i]<<' ';

	for(int i = 1 ; i <= n ; ++i ) if(check(i))	ans = min(dp[i],ans);
	if(ans > n)	cout<<-1<<endl;
	else cout<<ans<<endl;
	

}
