#include <cstdio>
#include <cstring>
using namespace std;
const int maxn=110000,inf=maxn*2;;
int f[maxn],s[maxn],c[maxn],d[maxn];
int find(int u)
{
    if(f[u]==u) return u;
    return f[u]=find(f[u]);
}
bool lucky(int a)
{
    while(a)
    {
        int b=a%10;
        if(b!=4&&b!=7) return false;
        a/=10;
    }
    return true;
}
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++) f[i]=i;
    memset(s,0,sizeof(s));
    for(int i=0;i<m;i++)
    {
        int u,v;
        scanf("%d%d",&u,&v);
        u--;
        v--;
        f[find(u)]=find(v);
    }
    for(int i=0;i<n;i++) s[find(i)]++;
    for(int i=0;i<n;i++) c[s[i]]++;
    for(int i=1;i<=n;i++) d[i]=inf;
    d[0]=0;
    for(int k=1;k<=n;k++)
    {
        if(c[k]==0) continue;
        //printf("%d:%d\n",k,c[k]);
        int z=c[k];
        for(int y=1;z;y+=y)
        {
            if(y>z) y=z;
            z-=y;
            int x=k*y;
            for(int i=n;i>=x;i--)
            {
                int t=d[i-x]+y;
                if(d[i]>t) d[i]=t;
            }
        }
    }
    int ans=inf;
    for(int i=1;i<=n;i++)
        if(ans>d[i]&&lucky(i)) ans=d[i];
    if(ans>=inf) puts("-1"); else printf("%d\n",ans-1);
    return 0;
}
