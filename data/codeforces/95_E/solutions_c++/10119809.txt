#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int fa[100005],f[100005],cnt[100005],c[4000],w[4000],tot;
int dp[100005];
int find(int x) {return fa[x]==x?x:fa[x]=find(fa[x]);}
int check(int x)
{
    while(x>0)
    {
        if(x%10==4||x%10==7) x/=10;
        else return 0;
    }
    return 1;
}
main()
{
    int n,m,s,t,x,y;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++) fa[i]=i;
    for(int i=1;i<=m;i++)
    {
        scanf("%d%d",&s,&t);
        x=find(s);y=find(t);
        if(x!=y) fa[x]=y;
    }
    for(int i=1;i<=n;i++) f[find(i)]++;
    for(int i=1;i<=n;i++) if(f[i]>0) cnt[f[i]]++;
    for(int i=1;i<=n;i++) if(cnt[i]>0&&i<=77777)
    {
        for(int k=1;k<=cnt[i];k<<=1)
        {
            c[++tot]=k;w[tot]=i*k;cnt[i]-=k;
        }
        if(cnt[i]>0) {c[++tot]=cnt[i];w[tot]=i*cnt[i];}
    }
    for(int i=1;i<=n;i++) dp[i]=1000000;dp[0]=0;
    for(int i=1;i<=tot;i++)
    for(int j=n;j>=w[i];j--) dp[j]=min(dp[j],dp[j-w[i]]+c[i]);
    int ans=1000000;
    for(int i=4;i<=min(n,77777);i++) if(check(i)) ans=min(ans,dp[i]);
    if(ans==1000000) printf("-1\n");
    else printf("%d\n",ans-1);

}
