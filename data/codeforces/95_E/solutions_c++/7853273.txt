#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,x,y,t,an;
int f[N],s[N],g[N],w[N];
int get(int x) {
	return f[x]==x?x:f[x]=get(f[x]);
}
inline bool ju(int x) {
	for (;x;x/=10) {
		int A=x%10;
		if (A!=4&&A!=7) return 0;
	}
	return 1;
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n) f[i]=i;
	For(i,1,m) {
		scanf("%d%d",&x,&y);
		f[get(x)]=get(y);
	}
	For(i,1,n) s[get(i)]++;
	memset(w,120,sizeof(w));
	an=w[0],w[0]=0;
	For(i,1,n) if (i==get(i)) {
		if (1ll*s[i]*s[i]<=n) g[s[i]]++;
		else for (j=n;j>=s[i];j--) w[j]=min(w[j],w[j-s[i]]+1);
	}
	For(i,1,n) if (g[i]) {
		For(j,0,i-1) {
			int l=1,r=0,u;
			for (k=j,u=1;k<=n;k+=i,u++) {
				int A=w[k]-u;
				for (;r>=l&&s[r]>=A;r--);
				s[++r]=A,f[r]=u;
				if (f[l]+g[i]<u) l++;
				w[k]=s[l]+u;
			}
		}
	}
	For(i,1,n) if (ju(i)) an=min(an,w[i]);
	printf("%d\n",an>1e9?-1:an-1);
	return 0;
}
