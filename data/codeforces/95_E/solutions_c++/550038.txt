#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=100005;
int fa[N],dp[N],n,m,cnt1[N],cnt2[N];
int find(int n)
{
    if (fa[n]==n) return n;
    return fa[n]=find(fa[n]);
}
bool check(int n)
{
    for(;n;n/=10)if(n%10!=4&&n%10!=7) return 0;
    return 1;
}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)fa[i]=i;
    for(int i=0,a,b;i<m;i++)
    {
        scanf("%d%d",&a,&b);
        fa[find(a)]=find(b);
    }
    memset(dp+1,0x7f,4*n);
    for(int i=1;i<=n;i++)cnt1[find(i)]++;
    for(int i=1;i<=n;i++)cnt2[cnt1[i]]++;
    for(int i=1;i<=n;i++)
    {
        for(int p=1;cnt2[i];p<<=1)
        {
            if (p>cnt2[i]) p=cnt2[i];
            cnt2[i]-=p;
            int pv=p*i;
            for(int j=n-pv;j>=0;j--)dp[j+pv]=min(dp[j+pv],dp[j]+p);
        }
    }
    int ans=1<<30;
    for(int i=1;i<=n;i++)if(check(i))ans=min(dp[i],ans);
    if (ans>n) puts("-1");
    else printf("%d\n",ans-1);
}
