#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cmath>
#define N 100005
using namespace std;

int f[N],q[N],s[N],o[N],dv[N],fa[N];
int find(int i) {return (fa[i]==i)?i:fa[i]=find(fa[i]);}

bool ck(int x) {
  while (x && ((x%10)==4 || (x%10)==7)) x/=10;
  return !x;
};

int main() {
  int n,m;
  scanf("%d%d",&n,&m);
  for (int i=1;i<=n;i++) fa[i]=i,s[i]=1;
  while (m--) {
    int a,b;
    scanf("%d%d",&a,&b);
    a=find(a),b=find(b);
    if (a!=b) {
      fa[a]=b;
      s[b]+=s[a];
    };
  };
  for (int i=1;i<=n;i++)
    if (fa[i]==i) o[s[i]]++;
  memset(f,1,sizeof f);
  f[0]=0;
  for (int i=1;i<=n;i++) if (o[i]) 
    for (int j=0;j<i;j++) {
      int l=1,r=1;
      q[1]=f[j];
      dv[1]=0;
      for (int k=j+i;k<=n;k+=i) {
        int md=dv[r]+1;
        while (l<=r && md-dv[l]>o[i]) l++;
        while (l<=r && f[k]+md<q[r]+dv[r]) r--;
        q[++r]=f[k],dv[r]=md;
        f[k]=q[l]+md-dv[l];
      };
    };
  int ans=1<<30;
  for (int i=1;i<=n;i++) if (ck(i)) ans=min(ans,f[i]);
  if ((--ans)>n) puts("-1"); else printf("%d\n",ans);
  return 0;
}
