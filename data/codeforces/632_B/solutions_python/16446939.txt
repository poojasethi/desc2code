N = int(raw_input())
changes = map(lambda x: int(x[0]) if x[1]=="A" else -1*int(x[0]), zip(str(raw_input()).split(), list(str(raw_input()))))

prefix_sum = 0
suffix_sum = 0
max_change = 0
for I in range(N):
	prefix_sum += changes[I]
	suffix_sum += changes[N-I-1]
	max_change = max(prefix_sum, suffix_sum, max_change)

print sum([-change for change in changes if change<0]) + max_change
