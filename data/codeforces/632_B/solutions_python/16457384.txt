import math


n = int(raw_input())
Ss = raw_input().split(' ')
ABs = raw_input()

leftA = 0
leftB = 0
rightA = 0
rightB = 0


for i in range(n):
	t = int(Ss[i])
	if ABs[i] == 'A':
		rightA += t
	else:
		rightB += t

maxA = rightA
maxB = rightB

for i in range(n):
	t = int(Ss[i])
	if ABs[i] == 'A':
		rightA -= t
		leftB += t
	else:
		rightB -= t
		leftA += t
	maxB = max(maxB, leftB + rightB)
	maxA = max(maxA, leftA + rightA)

print max(maxB,maxA)
