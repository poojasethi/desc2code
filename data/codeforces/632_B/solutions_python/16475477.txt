n = int(raw_input())
p = map(int, raw_input().split())
s = raw_input()

a = 0
b = 0
for i, v in enumerate(s):
    if v == "A":
        a += p[i]
    else:
        b += p[i]

ans = [b]
for i in range(n):
    if s[i] == "A":
        ans.append(ans[-1] + p[i])
    else:
        ans.append(ans[-1] - p[i])

ans.append(b)
for i in range(n-1, -1, -1):
    if s[i] == "A":
        ans.append(ans[-1] + p[i])
    else:
        ans.append(ans[-1] - p[i])

print max(ans)
