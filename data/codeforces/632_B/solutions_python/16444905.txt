n = input()
P = map(int, raw_input().split())
S = raw_input()

B = 0

for i in xrange(n):
    if S[i] != 'A': B += P[i]

o = B

B_ = B
for i in xrange(n):
    if S[i] == 'A': B += P[i]
    else: B -= P[i]
    o = max(o, B)
    
for i in xrange(n - 1, -1, -1):
    if S[i] == 'A': B_ += P[i]
    else: B_ -= P[i]
    o = max(o, B_)

print o
