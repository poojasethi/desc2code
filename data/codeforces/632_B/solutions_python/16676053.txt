n = map(int,raw_input())

store = map(int,raw_input().split())
rev = store[::-1]

team  = raw_input()

init = 0

for index,value in enumerate(team):
    if value == 'B':
        init += store[index]

best_pre = init

pre = init

for index,value in enumerate(team):
    if value == 'A':
        pre += store[index]
        if pre > best_pre:
            best_pre = pre
    if value == 'B':
        pre -= store[index]

best_sux = init
sux = init


for index,value in enumerate(team[::-1]):
    if value == 'A':
        sux += rev [index]
        if sux > best_sux:
            best_sux = sux
    if value == 'B':
        sux -= rev [index]

print max(best_sux,best_pre)
