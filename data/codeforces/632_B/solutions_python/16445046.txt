import sys
import itertools as it


n = int(sys.stdin.readline())
p = map(int, sys.stdin.readline().split())
s = sys.stdin.readline().strip()


t1 = t2 = 0
r = 0
rr = 0

for i in xrange(n):
  if s[i] == 'B':
    rr += p[i] 

  if s[i] == 'A':
    t1 += p[i]
  else:
    t1 -= p[i]

  j = n-i-1
  if s[j] == 'A':
    t2 += p[j]
  else:
    t2 -= p[j]

  r = max(r, t1, t2)

print r + rr
