#include<bits\stdc++.h>
#define ll long long
using namespace std;
ll a[500005];
ll b[500005];
ll k[500005];
int main(){
	int n;
	string s;
	cin >> n;
	for( int i=1 ; i<=n ; i++ ){
		scanf("%I64d", &k[i]);
	}
	cin >> s;
	for( int i=1 ; i<=n ; i++ ){
		if( s[i-1]=='A' ) a[i] = a[i-1]+k[i], b[i] = b[i-1];
		else b[i] = b[i-1]+k[i], a[i] = a[i-1];
	}
	ll ansmax = 0;
	for( int i=1 ; i<=n ; i++ ){
		ansmax = max( ansmax, a[i]+b[n]-b[i]);
		ansmax = max( ansmax, b[i]+a[n]-a[i]);
	}
	cout << ansmax << endl;
	return 0;
}
