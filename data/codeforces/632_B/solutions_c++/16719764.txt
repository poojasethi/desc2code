#include <bits/stdc++.h>

typedef long long ll;
using namespace std;

bool A[500100];
int SD[500100],DS[500100],N,P[500100];
ll B,rs,pas;

int main(){
	cin >> N;
	for(int i = 0;i<N;i++){
		cin >> P[i];
	}
	
	for(int i = 0;i<N;i++){
		char c;
		cin >> c;
		if(c == 'A') A[i] = 1;
		else B+=P[i];
	}
	rs = max(B,rs);
	
	pas = B;
	for(int i = 0;i<N;i++){
		if(A[i]){
			pas+=P[i];
		}else pas-=P[i];
		if(pas > rs) rs = pas;
	}
	
	pas = B;
	for(int i = N;i;i--){
		if(A[i]){
			pas+=P[i];
		}else pas-=P[i];
		if(pas > rs) rs = pas;
	}
	
	cout << rs;
	
	
	
	return 0;
}
