#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    long long p[n],ans=0,sum=0;
    char s[n+1];
    for(int i=0;i<n;i++) cin>>p[i];
    cin>>s;
    for(int i=0;i<n;i++) if(s[i]=='B') sum+=p[i];
    ans=sum;
    long long last=sum;
    for(int i=0;i<n;i++){
        last=(s[i]=='B')?(last-p[i]):(last+p[i]);
        if(last>ans) ans=last;
    }
    last=sum;
    for(int i=n-1;i>=0;i--){
        last=(s[i]=='B')?(last-p[i]):(last+p[i]);
        if(last>ans) ans=last;
    }
    cout<<ans;
    return 0;
}
