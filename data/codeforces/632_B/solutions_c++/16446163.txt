#include<iostream>
#include<cstring>
using namespace std;
long long num[500005];
int main()
{
	ios::sync_with_stdio(0);
	int n;
	long long sum=0;
	long long ans=0;
	string p;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>num[i];
	}
	cin>>p;
	for(int i=1;i<=n;i++)
	{
		if(p[i-1]=='B')
		{
			sum+=num[i];
			num[i]=-num[i];
		}	
	}
	long long mmax=0;
	for(int i=1;i<=n;i++)
	{
		ans+=num[i];
		mmax=max(mmax,ans);
	}
	ans=0;
	for(int i=n;i>=1;i--)
	{
		ans+=num[i];
		mmax=max(mmax,ans);
	}
	if(mmax<0)
	mmax=0;
	cout<<sum+mmax<<endl;
}
