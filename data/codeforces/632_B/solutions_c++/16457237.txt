#include "cstdio"
#define NMAX 500001
using namespace std;
long long sum=0, c[NMAX];
int main()
{
    int n, p[NMAX];
    long long a=0, b=0;
    char x;
    scanf("%d", &n);
    for (int i=1; i<=n; i++)
    {
        scanf("%d", &p[i]);
        sum+=(long long)p[i];
    }
    scanf("%c", &x);
    for (int i=1; i<=n; i++)
    {
        scanf("%c", &x);
        c[i]=c[i-1]+(long long)p[i]*(x=='B'?1:-1);
    }
    for (int i=0; i<=n; i++)
    {
        if (a>c[i]) a=c[i];
        if (b>c[n]-c[i]) b=c[n]-c[i];
    }
    long long k=c[n]-2*(a<b?a:b);
    printf("%lld", (sum+k)/2);
    return 0;
}
