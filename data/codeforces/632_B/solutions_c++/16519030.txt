#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
    long long i,j,k,l=0,n,m;
    int p[600005];
    string s;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>p[i];
    }
    cin>>s;
    for(i=0;i<n;i++)
    {
        if(s[i]=='B') l=l+p[i];
    }k=l,m=l;
    for(i=n-1;i>=0;i--)
    {
     if(s[i]=='A') k=k+p[i];
     else if(s[i]=='B') k=k-p[i];
     l=max(l,k);
    }
    for(i=0;i<n;i++)
    {
    if(s[i]=='A') m=m+p[i];
     else if(s[i]=='B') m=m-p[i];
     l=max(l,m);
    }
    cout<<l;
    return 0;

}
