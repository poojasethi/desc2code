#include<iostream>
#include<string>
#include<cstring>
#include<cstdio>
using namespace std;
typedef long long ll;
const int maxn=5000010;
int a[maxn];
char s[maxn];
int main()
{
	
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	getchar();
	for(int i=1;i<=n;i++) s[i]=getchar();
	ll tot=0,b=0;
	for(int i=1;i<=n;i++) tot+=a[i];
	for(int i=1;i<=n;i++) if(s[i]=='B') b+=a[i];
	ll ans=b,tot1=0,b1=0;
	for(int i=1;i<=n;i++)
	{
		tot1+=a[i];
		if(s[i]=='B') b1+=a[i];
		ans=max(ans,max(tot1-b1+b-b1,b1+tot-b-tot1+b1));
	}
	printf("%I64d\n",ans);
	return 0;
}
