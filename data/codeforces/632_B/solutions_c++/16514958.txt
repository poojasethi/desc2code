#include<iostream>
#include<cstdio>
long long  a[600000];
using namespace std;
long long sum1[666666],sum2[666666];
int main()
{
	string s;
	
	int n;
	while(cin>>n)
	{
	for(int i=0;i<n;i++)
	cin>>a[i];
	cin>>s;
	int len=s.length();
	int l=s.size();
	
	for(int i=0;i<len;i++)
	{
		sum1[i]=sum1[i-1];
		sum2[i]=sum2[i-1];
		if(s[i]=='A')
		sum1[i]+=a[i];
		else
		sum2[i]+=a[i];
		
	}
	
	long long ans;
	 ans=max(sum1[n-1],sum2[n-1]);
	for(int i=0;i<len;i++)
	{
		ans=max(ans,sum2[n-1]-sum2[i]+sum1[i]);
		ans=max(ans,sum1[n-1]-sum1[i]+sum2[i]);
	}
	cout<<ans<<endl;
	}
}
