#include <bits/stdc++.h>
using namespace std;
#define MAXN 500001
int n;
int power[MAXN];
char s[MAXN];
int pref[MAXN];
long long sum = 0;
long long f()
{
	long long wyn = sum;
	long long sumal = sum;
	for(int i = 0;i < n;++i)
	{
		sumal += s[i] == 'A' ? power[i] : -power[i];
		wyn = max(wyn,sumal);
	}
	return wyn;
	
}
int main()
{
	scanf("%d",&n);
	for(int i = 0;i < n;++i) scanf("%d",&power[i]);
	scanf("%s",s);
	for(int i = 0;i < n;++i) s[i] == 'B' ? sum += power[i] : 0;
	long long  wyn = max(sum,f());
	reverse(s,s + n);
	reverse(power,power + n);
	cout << max(wyn,f()) << endl;
	return 0;
}
