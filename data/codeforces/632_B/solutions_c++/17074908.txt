#include<stdio.h>
#include<iostream>
#include<string>
#define N 5*100000+5
using namespace std;
__int64 n,a[N],A,B;
char b[N];
__int64 sum1,sum2;
int main()
{
	scanf("%I64d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%I64d",&a[i]);
	}
	scanf("%s",b);
	__int64 ans=0;
	for(int i=0;i<n;i++)
	{
		if(b[i]=='A')sum1+=a[i];
		if(b[i]=='B')sum2+=a[i];
	}
	ans=max(sum2,sum1);
	if(b[0]=='A')A=a[0];
	else B=a[0];
	ans=max(A+sum2-B,B+sum1-A);
	for(int i=1;i<n;i++)
	{
		if(b[i]=='A')A+=a[i];
		if(b[i]=='B')B+=a[i];
		ans=max(ans,max(A+sum2-B,B+sum1-A));
	}
	printf("%I64d\n",ans);
}
