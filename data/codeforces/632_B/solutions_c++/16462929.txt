
#include <bits/stdc++.h>

using namespace std;
const int N = 500005;

char s[N];
int n,a[N];
long long res,A[N],B[N];

void init(){
	scanf("%d",&n);
	for(int i = 1; i <= n; i++){
		scanf("%d",&a[i]);
	}
	scanf("%s",s+1);
	for(int i = 1; i <= n; i++){
		A[i] = A[i-1];
		B[i] = B[i-1];
		if(s[i]=='A') A[i] += a[i];
		else B[i] += a[i];
	}
	for(int i = 0; i <= n+1; i++){
		res = max(res,A[i] + B[n] - B[i]);
		res = max(res,B[i] + A[n] - A[i]);
	}
	cout << res << endl;
}

int main(){
	init();
	return 0;
}
