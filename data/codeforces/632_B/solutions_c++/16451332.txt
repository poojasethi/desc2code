#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll a[500005],m,n,i,d,j;
string s;
int main()
{
    scanf("%I64d",&n);
    for(i=0;i<n;i++)
      scanf("%I64d",&a[i]);
    cin>>s;
    for(i=0;i<n;i++)
        if(s[i]=='B')
            m+=a[i];
    j=d=m;
    for(i=0;s[i];i++)
    {
        if(s[i]=='A')
            j+=a[i];
        else
            j-=a[i];
        d=max(j,d);
    }
    j=m;
    for(i=n-1;i>=0;i--)
    {
        if(s[i]=='A')
            j+=a[i];
        else
            j-=a[i];
        d=max(j,d);
    }
    cout<<d;
}
