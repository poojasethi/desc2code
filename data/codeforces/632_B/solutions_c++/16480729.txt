#define __USE_MINGW_ANSI_STDIO 0
#include <bits/stdc++.h>

using namespace std;

int main()
{
    const int mn=1e6+10;
    int n,a[mn]; __int64 r,t,s; char c[mn];
    cin >> n;
    for (int i=1; i<=n; i++) scanf("%d",&a[i]);
    s=0;
    for (int i=1; i<=n; i++)
    {
        cin >> c[i];
        if (c[i]=='B') s=s+a[i];
    }
    r=0; t=0;
    for (int i=1; i<=n; i++)
    {
        if (c[i]=='A') t=t+a[i]; else t=t-a[i];
        if (t>r) r=t;
    }
    t=0;
    for (int i=n; i>0; i--)
    {
        if (c[i]=='A') t=t+a[i]; else t=t-a[i];
        if (t>r) r=t;
    }
    cout << r+s;
}
