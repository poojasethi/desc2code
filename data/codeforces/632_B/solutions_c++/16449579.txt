#include <bits/stdc++.h>
using namespace std;
typedef long long int lli;
int main(){
	int n;
	cin>>n;
	int p[n];
	for(int i=0;i<n;i++)cin>>p[i];
	lli score=0,maxscore=0;
	lli a_score=0,b_score=0,init=0;
	char c[n];
	for(int i=0;i<n;i++){
		cin>>c[i];
		if(c[i]=='A')a_score+=p[i];
		else b_score+=p[i];
	}
	//prefix
	init=a_score+b_score;
	maxscore=b_score;
	for(int i=0;i<n;i++){
		if(c[i]=='B')score-=p[i];
		else score+=p[i];
		maxscore=max(maxscore,b_score+score);
	}
	score=0;
	for(int i=n-1;i>=0;i--){
		if(c[i]=='B')score-=p[i];
		else score+=p[i];
		maxscore=max(maxscore,b_score+score);
	}
	cout<<maxscore<<endl;
}
