#include <bits/stdc++.h>

using namespace std;


char str[500000];
long long int a [500000];


int main()
{
	int n;
	cin>>n;

	for (int i=0; i<n; i++){
		long long int aux;
		cin>>aux;
		a[i]=aux;
	}

	string s;
	cin>>s;

	long long int st=0;
	for (int i=0; i<n; i++) if (s[i]=='B') st += a[i];
	long long int max=st;

	long long int suf=0;
	long long int pref=0;
	for (int i=0; i<n; i++){
		if (s[i]=='A') suf += a[i];
		else suf -= a[i];
		
		if (s[n-i-1]=='A') pref+=a[n-i-1];
		else pref -= a[n-i-1];

		if (st+suf>max) max=st+suf;
		if (st+pref>max) max=st+pref;
	}

	cout<<max<<endl;

	return 0;
}
