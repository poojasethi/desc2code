#include <bits/stdc++.h>
using namespace std;
long long n,s[1000000], cntb = 0,res = -10000000000000;
string p;
int main(){
	scanf("%I64d",&n);
	for(int i = 0; i<n; i++) scanf("%I64d",&s[i]);
	cin >> p;
	for(int i = 0; i<p.size();i++){
		if(p[i] == 'B') cntb+=s[i];
	}
	long long k = cntb,l = cntb;
	for(int i = 0; i<p.size();i++){
		if(p[i] == 'A') cntb+=s[i];
		else cntb-=s[i];
		res = max(res,cntb);
	}
	for(int i = p.size()-1; i>=0;i--){
		if(p[i] == 'A') k+=s[i];
		else k-=s[i];
		res = max(res,k);
	}
	res = max(res,l);
	cout << res << endl;


}
