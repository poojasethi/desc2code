#include <bits/stdc++.h>
using namespace std;
#define f(i,a,b) for(int i=a;i<b;++i)
#define max(a,b) ( (a) > (b) ? (a) : (b))
#define min(a,b) ( (a) < (b) ? (a) : (b))
#define ll long long
#define N  500005
ll x[N],y[N],s[N],ans;
int main() 
{
	int n;
	cin >> n;
	f(i,0,n) scanf("%I64d",&s[i]);
	char str[N];
	scanf("%s",str);
	f(i,0,n)
	{
		x[i+1]=x[i];
		y[i+1]=y[i];
		if(str[i]=='A') x[i+1]+=s[i];
		else y[i+1]+=s[i];
	}
	ans=max(x[n],y[n]);
	f(i,1,n+1)
	{
		ans=max(ans,x[i]+y[n]-y[i]);
		ans=max(ans,y[i]+x[n]-x[i]);
	}
	cout << ans << endl;
}
