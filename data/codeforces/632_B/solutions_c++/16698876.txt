#include<iostream>
#include<cstdio>
using namespace std;
const int maxn=5555555;
typedef long long ll;
ll x[maxn];
ll a[maxn],b[maxn];
string s;
int main()
{
	int n;cin>>n;
	for(int i=1;i<=n;i++)
		cin>>x[i];
	cin>>s;
	for(int i=1;i<=n;i++)
	{
		if(s[i-1]=='A'){
			a[i]=a[i-1]+x[i];
			b[i]=b[i-1];
		}
		else{
			b[i]=b[i-1]+x[i];
			a[i]=a[i-1];
		}
	}
	ll ans=0;
	for(int i=1;i<=n;i++){
		ans=max(ans,a[i]+b[n]-b[i]);
		ans=max(ans,b[i]+a[n]-a[i]);
	}
	cout<<ans<<endl;
}
