#include<iostream>
#include<stdio.h>
using namespace std;
typedef long long ll;
const int max_n=500000+10;
char s[max_n];
int num[max_n];
int main()
{
	int n;
	ll now=0;
	scanf("%d",&n);
	for(int i=0;i<n;++i) scanf("%d",&num[i]);
	scanf("%s",s);
	for(int i=0;i<n;++i)
	{
		if(s[i]=='A') num[i]=-num[i];
		else now+=num[i];
	}
	ll get=0;
	ll add=0;
	for(int i=0;i<n;++i)
	{
		get-=num[i];
		add=max(add,get);
	}
	get=0;
	for(int i=n-1;i>=0;--i)
	{
		get-=num[i];
		add=max(add,get);
	}
	printf("%I64d\n",add+now);
	return 0;
}
