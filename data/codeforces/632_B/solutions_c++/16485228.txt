#include <iostream>
using namespace std;
int main()
{
	int n,i;
	cin>>n;
	string st;
	int A[n];
	long long N=0,N1=0,N2=0;
	for(i=0;i<n;++i)
		cin>>A[i];
		cin>>st;
	for(i=0;i<n;++i)
	{
	    if(st[i]=='B')
	        N+=A[i];
	}
	N1=N2=N;
	for(i=0;i<n;++i)
	{
		if(st[i]=='A')
			N1+=A[i];
		else
			N1-=A[i];
		if(N1>N2)
			N2=N1;
	}
	N1=N;
	for(i=n-1;i>0;--i)
	{
		if(st[i]=='A')
			N1+=A[i];
		else
			N1-=A[i];
		if(N1>N2)
			N2=N1;
	}
		cout<<N2;
	
}
