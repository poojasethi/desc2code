#include "cstdio"
#include "algorithm"
#define NMAX 500001
using namespace std;
long long sum, c[NMAX];
int main()
{
    int n, p[NMAX];
    long long a=0, b=0;
    char x;
    scanf("%d", &n);
    for (int i=1; i<=n; i++)
        scanf("%d", &p[i]);
    scanf("%c", &x);
    for (int i=1; i<=n; i++)
    {
        scanf("%c", &x);
        sum+=(long long)p[i];
        c[i]=c[i-1]+(long long)p[i]*(x=='B'?1:-1);
    }
    for (int i=0; i<=n; i++)
    {
        a=min(a, c[i]);
        b=min(b, c[n]-c[i]);
    }
    printf("%I64d", (sum+c[n]-2*(a<b?a:b))/2);
    return 0;
}
