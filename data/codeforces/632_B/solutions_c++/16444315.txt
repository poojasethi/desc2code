#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int n;
long long p[500010];
char str[500010];
int main()
{
	scanf("%d",&n);
	for (int i=0;i<n;i++)
		scanf("%d",&p[i]);
	scanf("%s",str);
	long long maxdet=0,now=0;
	for (int i=0;i<n;i++)
	{
		if (str[i]=='A')
			now+=p[i];
		else
			now-=p[i];
		if (now>maxdet)
			maxdet=now;
	}
	now=0;
	for (int i=n-1;i>=0;i--)
	{
		if (str[i]=='A')
			now+=p[i];
		else
			now-=p[i];
		if (now>maxdet)
			maxdet=now;
	}
	for (int i=0;i<n;i++)
		if (str[i]=='B')
			maxdet+=p[i];
	cout<<maxdet<<endl;
	return 0;
	
}
