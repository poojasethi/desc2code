#include<cstdio>
#include<algorithm>

int n;
int a[555555];
char s[555555];
long long ori, tot, ans;

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) scanf("%d", &a[i]);
	scanf("%s", s);
	ori = 0;
	for(int i=0;i<n;i++)
		if(s[i]=='B') ori+=a[i];
	tot = ans = ori;
	for(int i=0;i<n;i++) {
		if(s[i]=='A') tot+=a[i];
		else tot-=a[i];
		ans = std::max(ans, tot);
	}
	tot = ori;
	for(int i=n-1;i>=0;i--) {
		if(s[i]=='A') tot+=a[i];
		else tot-=a[i];
		ans = std::max(ans, tot);
	}
	printf("%I64d\n", ans);
	
	return 0;
}
