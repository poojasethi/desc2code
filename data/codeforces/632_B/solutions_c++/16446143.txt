#include <cstdio>
#include <cstring>
using namespace std;

const int N=1000000;
int p[N];
long long ans,a[N],b[N];
char s[N];

inline long long max(long long a,long long b)
{
    return a>b?a:b;
}

int main()
{
    int n,i;
    scanf("%d",&n);
    for (i=1;i<=n;++i) scanf("%d",&p[i]);
    scanf("%s",s);
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    for (i=1;i<=n;++i)
    {
        a[i]=a[i-1];
        b[i]=b[i-1];
        if (s[i-1]=='A') a[i]+=p[i]; else b[i]+=p[i];
    }
    ans=b[n];
    for (i=1;i<=n;++i)
        ans=max(ans,max(a[i]+b[n]-b[i],b[i]+a[n]-a[i]));
    printf("%I64d",ans);
    return 0;
}
