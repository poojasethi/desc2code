#include <bits/stdc++.h>

using namespace std;

#define int long long

const int N = 500011;
int a[N];
int prefix[N];
int sum[N];

main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	string s;
	cin >> s;
	for (int i = 0; i < n; i++) {
		sum[i + 1] = sum[i] + a[i];
		prefix[i + 1] = prefix[i] + (s[i] == 'B' ? a[i] : 0);
	}
	int ans = prefix[n - 1];
	for (int i = 0; i < n; i++) {
		int temp = prefix[n] - prefix[i + 1] + sum[i + 1] - prefix[i + 1];
		ans = max(ans, temp);
		ans = max(ans, sum[n] - temp);
	}
	cout << ans << endl;
}
