#include <stdio.h>
int main()
{
int i,n,p[500005];
long long sumn=0,sbt=0,ans=0,d=0,sbn=0;
char c[500005];

scanf("%d",&n);

for(i=0;i<n;i++)
	scanf("%d",&p[i]);

scanf("%s",c);

for(i=0;i<n;i++)
{
	sumn+=p[i];
	if(c[i]=='B')
	{
		sbt+=p[i];
		sbn+=p[i];
	}
	d=sumn-sbn-sbn;
	if(d>ans) ans=d;
}

sumn=0;sbn=0;

for(i=n-1;i>=0;i--)
{
	sumn+=p[i];
	if(c[i]=='B') sbn+=p[i];
	d=sumn-sbn-sbn;
	if(d>ans) ans=d;
}

printf("%lld",sbt+ans);

return 0;
}
