#include<cstdio>
long long a[500005],b[500005];
char s[500005];
int main() {
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%lld",&a[i]);
    scanf("%s",s+1);
    long long ans=0;
    for(int i=1;i<=n;i++) {
        if(s[i]=='A') b[i]=b[i-1]+a[i];
        else {
            b[i]=b[i-1]-a[i];
            ans+=a[i];
        }
    }
    long long max=0;
    for(int i=1;i<=n;i++) {
        if(b[i]>max) max=b[i];
        if(b[n]-b[n-i]>max) max=b[n]-b[n-i];
    }
    printf("%lld\n",ans+max);
    return 0;
}
