#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll A,B,AA=0,BB=0;
long long n,p,ans;
string s;
ll a[500005];
ll b[100005];
main()
{
	cin>>n;
	for(int i=0;i<n;i++)cin>>a[i];
	cin>>s;
	
	for(int i=0;i<n;i++)if(s[i]=='A')A+=a[i];else B+=a[i];
	
	ans=B;
	
	
	
	for(int i=0;i<n;i++)
	{
		if(s[i]=='A')AA+=a[i];else BB+=a[i];
		
		if(B-BB+AA>ans)ans=B-BB+AA;
	}
	AA=BB=0;
	for(int i=n-1;i>=0;i--)
	{
		if(s[i]=='A')AA+=a[i];else BB+=a[i];
		
		if(B-BB+AA>ans)ans=B-BB+AA;
	}
	
	
	cout<<ans<<endl;
	
}
