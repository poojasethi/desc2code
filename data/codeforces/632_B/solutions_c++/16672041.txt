#include <cstdio>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define MAXN 500010

int p[MAXN];
char str[MAXN];

int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",i+p);
	}
	scanf("%s",str);
	long long sa=0,sb=0,pa=0,pb=0;
	for(int i=0;i<n;i++){
		if(str[i]=='A') sa+=p[i];
		else sb+=p[i];
	} 
	long long ans=0;
	for(int i=n-1;i>=0;i--){
		if(str[i]=='A'){
			pa+=p[i];
		}
		else pb+=p[i];
		ans=max(ans,max(sa-pa+pb,sb-pb+pa));
		
	}
	printf("%I64d\n",ans);
	return 0;
}
