#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long

using namespace std;

int n,s[500005],v[500005];
char t[500005];

LL cur,ans,ini;

int main(){
  scanf("%d",&n);
  FI(i,1,n) scanf("%d",&s[i]);
  scanf(" %s",t+1);
  FI(i,1,n){
    if(t[i]=='B') ini+=s[i],v[i]=-s[i];
    else v[i]=s[i];
  }
  //pref
  cur=0;
  FI(i,1,n){
    cur+=v[i];
    if(cur>ans) ans=cur;
  }
  //suff
  cur=0;
  FD(i,n,1){
    cur+=v[i];
    if(cur>ans) ans=cur;
  }
  printf("%lld\n",ans+ini);
  return 0;
}
