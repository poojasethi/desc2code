#include <stdio.h>
#include <algorithm>
using namespace std;
__int64 a[500010],b[500010],c[500010];
int main(void)
{
    __int64 n,i;
	char s[500010];
	__int64 amax=0;
	scanf("%I64d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%I64d",&c[i]);
	}
	scanf("%s",&s);
	for(i=1;i<=n;i++)
	{
		if(s[i-1]=='A')
		{
			a[i]=a[i-1]+c[i];
			b[i]=b[i-1];
		}
		else
		{
			b[i]=b[i-1]+c[i];
			a[i]=a[i-1];
		}
	}
	for(i=1;i<=n;i++)
	{
		amax=max(amax,a[i]+b[n]-b[i]);
		amax=max(amax,b[i]+a[n]-a[i]);
	}
	printf("%I64d",amax);
	
	
	return 0;
}
