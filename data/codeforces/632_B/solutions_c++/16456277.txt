#include <iostream>
#define MAX 500005
using namespace std;
int n, p[MAX];
string s;
long long ans, str, str2;

int main()
{
    cin >> n;
    for(int i = 0; i < n; i++) cin >> p[i];
    cin >> s;
    for(int i = 0; i < n; i++) 
        if(s[i] == 'B') str += p[i];

    ans = str2 = str;
    for(int i = 0; i < n; i++)
    {
        if(s[i] == 'A') str = str + p[i];
        else str -= p[i];
        if(s[n - i - 1] == 'A') str2 += p[n - i - 1];
        else str2 -= p[n - i - 1];
        ans = max(ans, max(str, str2));
    }

    cout << ans;
}
