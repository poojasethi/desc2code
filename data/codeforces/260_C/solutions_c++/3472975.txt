#include <iostream>
#include <vector>

using namespace std;

typedef long long int lint;

int main()
{
  int n,x;
  cin>>n>>x;
  vector<int> v(n);
  for (int i=0;i<n;i++)
    cin>>v[i];
  x--;
  int imin=x;
  for (int i=x-1;i!=x;i=(i-1+n)%n)
    if (v[i]<v[imin])
      imin=i;
  lint total=0;
  while (x!=imin) {
    v[x]--;
    x--;
    total++;
    if (x<0) x=n-1;
  }
  total+=lint(v[x])*n;
  int resta=v[x];
  for (int i=0;i<n;i++) {
    if (i>0) cout<<" ";
    if (i==x) cout<<total;
    else cout<<v[i]-resta;
  }
  cout<<endl;
}
