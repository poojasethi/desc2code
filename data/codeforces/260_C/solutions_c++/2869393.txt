__int64 n,x,w[100001],i,M=1999999999,P,y,a,b,q;

    x--;
    for(i=0;i<n;i++)scanf("%I64d",&w[i]);
    q=0;
    for(i=0;i<n;i++){
        if(M>w[i]){
        M=w[i];
        q=i;
        }
    }
    for(i=x+n;i>x;i--){
        if(w[i%n]==M)break;
    }
    y=i%n;
//  y=q;
    a=(x+n-y)%n;
    b=n*w[y]+a;
    for(i=y+1;i<n+y;i++){
        w[i%n]-=w[y];
        if(a)a--,w[i%n]--;
    }
    w[y]=b;
    for(i=0;i<n;i++)printf("%I64d ",w[i]);
}
