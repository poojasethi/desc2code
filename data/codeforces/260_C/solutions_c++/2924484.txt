#include <stdio.h>

int a[111111];
int n, m;

int main() {
    int i, j = 2147483647;
    long long t;
    scanf("%d%d", &n, &m);
    for (i = 1; i <= n; ++i) {
        scanf("%d", a + i);
        if (j > a[i]) j = a[i];
    }
    t = (long long)j * n;
    for (i = 1; i <= n; ++i) a[i] -= j;

    while (a[m]) {
        --a[m], ++t;
        if (!--m) m = n;
    }

    for (i = 1; i <= n; ++i) printf("%I64d ", i == m ? t : a[i]);
    return 0;
}
