#include <cstdio>
#include <algorithm>
using namespace std;
long long arr[100000];
int main(){
	int n,x;
	scanf("%d%d",&n,&x);
	x--;
	long long mini=(1<<30);
	for(int i=0;i<n;i++){
		scanf("%I64d",&arr[i]);
		mini=min(mini,arr[i]);
	}
	
	int i=x,b;
	long long tot=0;
	while(1){
		if(arr[i]==mini){b=i;break;}
		arr[i]--;tot++;
		if(i==0)i=n-1;
		else i--;
	}
	long long turn=arr[b];
	arr[b]=((long long)n*(long long)turn)+tot;
	
	for(int i=0;i<n;i++){
		if(i!=b)arr[i]-=turn;
		printf("%I64d ",arr[i]);
	}
}
