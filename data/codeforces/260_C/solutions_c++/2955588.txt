#include <iostream>
#include <algorithm>
using namespace std;
long long int a[100000];
int main(){
	long long int n,x,mn=1000000001;
	cin >> n >> x;
	x--;
	for(int i=0;i<n;i++){
		cin >> a[i];
		mn=min(mn,a[i]);
	}
	long long int st;
	for(int i=0;i<n;i++){
		if(a[(x-i+n)%n]==mn){
			st=(x-i+n)%n;
			break;
		}
	}
	for(int i=0;i<n;i++)	a[i]-=mn;
	for(int i=1;i<=(x-st+n)%n;i++) a[(st+i)%n]--;
	a[st]=mn*n+(x-st+n)%n;
	for(int i=0;i<n;i++) cout << a[i] << " ";
	return 0;
}
