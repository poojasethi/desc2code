#include<bits/stdc++.h>
#define ll long long
using namespace std;
const int N=1e5+9;
ll a[N],cnt;
int main()
{
	ios::sync_with_stdio(false);
	ll n,x,i,mn=1e9+5;
	cin>>n>>x;
	for(i=0;i<n;i++)
		cin>>a[i],mn=min(mn,a[i]);
	for(i=0;i<n;i++)
        a[i]-=mn;
    for(i=x-1;a[i];i--)
    	cnt++,a[i]-=1,i=(i==0?n:i);
    a[i]=mn*n+cnt;
    for(i=0;i<n;i++)
    	cout<<a[i]<<" ";
	return 0;
}
