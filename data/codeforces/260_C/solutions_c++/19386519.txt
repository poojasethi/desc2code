#include<iostream>
using namespace std;
int main()
{
	long long n,x,min=999999999999,i,j=0,k,arr[1000009];
	cin>>n>>x;
	for(i=1;i<=n;i++)
	{
		cin>>arr[i];
		if(arr[i]<min)
		{
			min=arr[i];
		}
	}
	for(i=1;i<=n;i++)
	arr[i]-=min;
	
	for(i=x;arr[i]>0;i--)
	{
		arr[i]-=1;
		j++;
		if(i==1) i=n+1;
	}
	arr[i]=min*n+j;
	for(i=1;i<=n;i++)
	cout<<arr[i]<<' ';
	return 0;
}
