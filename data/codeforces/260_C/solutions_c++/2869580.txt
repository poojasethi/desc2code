#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<iostream>
#include<fstream>
#include<algorithm>
using namespace std;
long long a[100006],k;
int main(){
	std::ios::sync_with_stdio(false);
	long i,m,n;
	cin >> n >> m;
	k=1000000006;
	for (i=1;i<=n;++i){
		cin >> a[i];
		k=min(k,a[i]);
	}
	for (i=1;i<=n;++i)
		a[i]-=k;
	a[0]=k*n;
	for (i=m;a[i]!=0;++a[0],i=i==1?n:i-1)
		--a[i];
	a[i]=a[0];
	for (i=1;i<=n;++i)
		cout << a[i] << ' ';
	cout << endl;
	return 0;
}
