#include<cstdio>
#include<algorithm>

#define MAXX 100111

int n,x,i,j;
long long a[MAXX],t,min;

int main()
{
    scanf("%d %d",&n,&x);
    --x;
    for(i=0;i<n;++i)
        scanf("%I64d",a+i);
    min=*std::min_element(a,a+n);
    for(j=x;a[j]!=min;j=(j-1+n)%n)
        --a[j];
    t=a[j];
    a[j]=a[j]*n+(x-j+n)%n;
    a[j]+=t;
    for(i=0;i<n;++i)
        a[i]-=t;
    for(i=0;i<n;++i)
        printf("%I64d ",a[i]);
    puts("");
    return 0;
}
