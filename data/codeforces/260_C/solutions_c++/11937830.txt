#include <bits/stdc++.h>
using namespace std;
long long a[100010],mini = 1e10,n,x;
int main()
{
  scanf("%I64d%I64d",&n,&x);
  for(int i=0;i<n;i++) 
  {
    scanf("%I64d",&a[i]);
    if(a[i]<mini) mini=a[i];
  }
  for(int i=0;i<n;i++) a[i]=a[i]-mini;
  long long last=mini*n;
  int ind=x-1;
  while(a[ind]!=0) 
  {
    a[ind]--;
    last++;
    ind--;
    if(ind==-1) ind=n-1;
  }
  a[ind]=last;
  for(int i=0;i<n;i++) printf("%I64d ",a[i]);
  return 0;
}
