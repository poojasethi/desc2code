#include<cstdio>
#include<iostream>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

typedef long long ll;

int main(){
	int n,x; cin>>n>>x; x--;
	static ll a[100000];
	rep(i,n) cin>>a[i];

	ll a_min=*min_element(a,a+n);
	rep(i,n) a[i]-=a_min;

	ll hand=n*a_min; // temochi
	while(1){
		if(a[x]==0) break;
		a[x]--;
		hand++;
		x=(x-1+n)%n;
	}

	a[x]+=hand;

	rep(i,n) cout<<a[i]<<' ';

	return 0;
}
