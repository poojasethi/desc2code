#include <iostream>

using namespace std;

int main()
{
	int N,x;
	cin>>N>>x;
	long long mn;
	long long * arr=new long long [N];
	cin>>arr[0];
	mn=arr[0];
	for (int i=1;i<N;i++)
		cin>>arr[i],mn=min(mn,arr[i]);
	if (mn)
	for (int i=0;i<N;i++)
		arr[i]-=mn-1;
	int steep=0;
	x--;
	while (arr[x])
	{
		arr[x]--;
		steep++;
		x=(N+x-1)%N;
	}
	if (mn==0)
		mn=1;
	arr[x]=(mn-1)*N+steep;
	for (int i=0;i<N;i++)
		cout<<arr[i]<<" ";
	cout<<endl;
	return 0;
}
