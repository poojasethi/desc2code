#include <cstdio>

long long all;
int n,x;
long long a[100002];
int main() {
	scanf("%d%d",&n,&x);
	x--;
	for(int i=0;i<n;i++) scanf("%I64d",&a[i]);
	long long min=2000000001;
	for(int i=0;i<n;i++) if(min>a[i]) min=a[i];
	for(int i=0;i<n;i++) a[i]-=min;
	all+=min*n;
	while(1) {
		if(a[x]==0) {
			a[x]=all;
			break;
		}
		a[x]--;
		all++;
		x--;
		if(x<0) x=n-1;
	}
	for(int i=0;i<n;i++) {
		printf("%I64d",a[i]);
		if(i<n-1) printf(" ");
	}
	printf("\n");
	return 0;
}
