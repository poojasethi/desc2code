#include<cstdio>
#include<iostream>
using namespace std;
typedef long long ll;
const ll maxn=100005,inf=1LL<<62LL;
ll rem=inf,n,m,s,t,a[maxn];
int main()
{
	cin>>n>>t;
	for(int i=1;i<=n;i++)
	{
		scanf("%I64d",&a[i]);
		if(a[i]<rem)rem=a[i],s=i;
	}
	for(int i=t;;)
	{
		if(a[i]==rem){s=i;break;}
		i--;if(!i)i=n;
	}
	for(int i=s;i!=t;i=i%n+1)a[i]--,a[s]++;
	for(int i=1;i<=n;i++)if(i!=s)a[i]-=rem;
	a[s]+=(n-1)*rem;if(s!=t)a[t]--,a[s]++;
	for(int i=1;i<=n;i++)printf("%I64d ",a[i]);
	return 0;
}
