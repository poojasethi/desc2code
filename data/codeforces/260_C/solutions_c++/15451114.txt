#include<stdio.h>
int n,m,wz;
long long a[100005],minn=1e18,tot;
int main()
{
  scanf("%d%d",&n,&m);
  for(int i=1;i<=n;i++)
    scanf("%I64d",&a[i]);
  for(int i=m;i>0;i--)
    {
	 a[i]--;
	 tot++;
	} 
  for(int i=1;i<=n;i++)
    if(minn>=a[i])
    { 
      minn=a[i];
      wz=i;
    }
  tot+=(minn*n);
  for(int i=1;i<=n;i++)
    a[i]-=minn;
  for(int i=n;i>wz;i--)
    {a[i]--;tot++;} 
  a[wz]=tot;
  for(int i=1;i<=n;i++)
    printf(i==n?"%I64d\n":"%I64d ",a[i]);
  return 0;
}
