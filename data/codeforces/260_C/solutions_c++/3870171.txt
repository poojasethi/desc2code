#include <algorithm>
#include <iostream>
#include <cstdio>
using namespace std;

typedef long long ll;

int n, x;
ll a[100 * 1000 + 2];

int main() {
	scanf("%d%d", &n, &x);
	x--;
	for (int i = 0; i < n; i++)
		scanf("%lld", a + i);
	ll mn = *min_element(a, a + n), tot = mn * n;
	for (int i = 0; i < n; i++)
		a[i] -= mn;
	while (a[x]) {
		tot++, a[x]--;
		x = (x + n - 1) % n;
	}
	a[x] = tot;
	for (int i = 0; i < n; i++)
		printf("%I64d ", a[i]);
	printf("\n");
	return 0;
}
