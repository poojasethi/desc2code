#include<iostream>
using namespace std;
int n;
int pre(int k){
    if(k==0)
        return n-1;
    return k-1;
}
int main(){
    long long x,a[100001],mx=1000000000,k=0;
    cin>>n>>x;
    for(int i=0;i<n;i++)
        cin>>a[i],mx=min(mx,a[i]);
    for(int i=0;i<n;i++)
        a[i]-=mx;
    k+=mx*n;
    x--;          
    while(a[x]>0)
        a[x]--,x=pre(x),k++;
    a[x]=k;
    for(int i=0;i<n;i++)
        cout<<a[i]<<" ";
}
