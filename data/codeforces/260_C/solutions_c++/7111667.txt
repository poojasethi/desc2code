#include<iostream>
using namespace std;
int main()
{
long long int n,x,i,j,min=1000000001;
cin>>n>>x;
long long int *a= new long long int [n];
for(i=0; i<n; i++)
{
cin>>a[i];
if(min>a[i])
min=a[i];
}
x--;
for(i=0; i<n; i++)
a[i]-=min;
/*for(i=0; i<n; i++)
cout<<a[i]<<" ";
cout<<"\n";*/
for(i=x,j=0; ; i--,j++)
{
if(i<0)
i+=n;
if(a[i]==0)
break;
a[i]--;
}
//cout<<j<<" "<<n<<" "<<min<<"\n";
a[i]=j+n*min;
for(i=0; i<n; i++)
cout<<a[i]<<" ";
delete a;
return 0;
}
