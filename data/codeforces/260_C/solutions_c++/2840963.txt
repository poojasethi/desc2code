#include <cstdio>
#include <cstring>

int n,x;
long long f[100010];

int main()
{
    scanf("%d%d",&n,&x);
    int r=x;
    for(int i=1;i<=n;i++)
        scanf("%I64d",&f[i]);
    for(int i=(x==1)?n:(x-1);i!=x;i=(i==1)?n:(i-1))
        if(f[i]<f[r])r=i;
    long long p=0;
    for(int i=(r==n)?1:(r+1);i!=x;i=(i==n)?1:(i+1))
        f[i]--,p++;
    f[x]--,p++;
    long long s=f[r];
    for(int i=1;i<=n;i++)
        f[i]-=s,p+=s;
    f[r]=p;
    for(int i=1;i<=n;i++)
        printf("%I64d ",f[i]);
    return 0;
}
