#include<iostream>
using namespace std;
int main()
{
int n,k,i,j;
long long a[100005];
cin>>n>>k;
for(i=1;i<=n;i++) cin>>a[i];
for(i=k;i>=1;i--)
{
if(a[i]==0)
{
for(j=1;j<i;j++) cout<<a[j]<<" ";
cout<<k-i<<" ";
for(j=i+1;j<=n;j++) cout<<a[j]<<" ";
return 0;
}
a[i]--;
}
j=n;
for(i=1;i<=n;i++) if(a[i]<=a[j]) j=i;
for(i=1;i<j;i++) cout<<a[i]-a[j]<<" ";
cout<<a[j]*(n*1LL)+(n-i+k)*1LL<<" ";
for(i=j+1;i<=n;i++) cout<<a[i]-a[j]-1<<" ";
return 0;
}
