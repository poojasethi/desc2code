#include<stdio.h>

int main(){
    long long i,j,n,m,d[100005],mn=9999999999,top,temp,s;
	scanf("%lld %lld",&n,&m);
	d[0]=1;
	for(i=1 ; i<=n ; i++){
		scanf("%lld",&d[i]);
		if(d[i] < mn)
			mn=d[i];
	}
	top=mn*n;
	for(i=1 ; i<=n ; i++)
		d[i]-=mn;
	temp=m;
	while(d[temp]){

		if(!temp) temp=n;
		if(!d[temp])
			break;
		d[temp]--;
		temp--;
		s++;


	}

	d[temp--]=top+s;
	for(i=1 ; i<=n ; i++)
		printf("%lld ",d[i]);
	return 0;
}
