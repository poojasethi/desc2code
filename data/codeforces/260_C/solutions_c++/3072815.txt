#include<stdio.h>
long long a[1000005];
int main()
{
	int i,n,k;
	long long sum,min;
	min=999999999;
	scanf("%d %d",&n,&k);
	for(i=0;i<n;i++)
	{
		scanf("%I64d",&a[i]);
		if(a[i]<min)
			min=a[i];
	}
	sum=min*n;
	for(i=0;i<n;i++)
	{
		a[i]-=min;
	}
	for(i=k-1;a[i]!=0;i=(i+n-1)%n)
	{
		a[i]--;
		sum++;
	}
	a[i]=sum;
	for(i=0;i<n-1;i++)
	{
		printf("%I64d ",a[i]);
	}
	printf("%I64d\n",a[n-1]);
	return 0;
}
