#include<stdio.h>
long long a[200000];
int main()
{
	long long n,x,i;
	scanf("%I64d%I64d",&n,&x);
	x=(x+n-1)%n;
	int t=x;
	for (i=0;i<n;i++) scanf("%I64d",&a[i]);
	for (i=(x+n-1)%n;i!=x;i=(i+n-1)%n) 
		if (a[i]<a[t]) t=i;
	long long num=a[t]*n;
	
	for (long long t1=(t+1)%n;t1!=(x+1)%n;t1=(t1+1)%n) {
		a[t1]--;
		num++;
	}
	
	long long t1=a[t];
	for (i=0;i<n;i++) a[i]-=t1;
	
	a[t]=num;
	printf("%I64d",a[0]);
	for (i=1;i<n;i++) printf(" %I64d",a[i]);
	printf("\n");
}
