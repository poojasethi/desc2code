#include <iostream>
#include <vector>

using namespace std;

int main(void){

	int n,x;

	cin >> n >> x;
	x = x-1;

	long long a[n];
	long long min = 10000000000;

	int i;
	for (i=0;i<n;i++){
		cin >> a[i];
		if (min > a[i]){
			min=a[i];
		}
	}

	long long count=0;
	for (i=0;i<n;i++){
		a[i]=a[i]-min;
		count += min;
	}

	i=x;
	while (true){
		if (a[i] == 0){
			a[i] = count;
			break;
		} else {
			a[i]--;
			i--;
			i=((i<0)? (n-1):i);
			count++;
		}
	}

	for (i=0;i<n;i++){
		cout << a[i] << " ";
	}

	cout << endl;

	return 0;

}
