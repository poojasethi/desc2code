#include<cstdio>
#include<iostream>
#include<climits>
#include<algorithm>
#define NN 100000
#define LL long long
using namespace std;
LL v[NN];
int main(){
	int n,x;scanf("%d%d",&n,&x);--x;
	LL mn = INT_MAX ;
	for(int i=0;i<n;++i){
		cin >>v[i];
		mn = min( mn,v[i]);
	}
	LL c = mn*n;
	for(int i =0;i<n;++i)v[i] -= mn;

	int j;
	for(j=x;v[j]!=0;j=(j-1+n)%n,++c)
		--v[j];
	v[j]=c;
	for(int i=0;i<n;++i)
		cout << v[i] << " "; 
	cout << endl;
	return 0;
}
