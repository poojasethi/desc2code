#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
int n,x; cin>>n>>x;
long long a[100005];
long long m=1000000009;
for (int i=1;i<=n;i++) {cin>>a[i]; m=min(m,a[i]);}
for (int i=1;i<=n;i++) a[i]-=m;
long long s=1LL*n*m;
while (a[x])
{a[x]--;
 x--;s++;
 if (x==0) x=n;
}
a[x]=s;
for (int i=1;i<=n;i++) {cout<<a[i]<<" ";}
return 0;
}
