#include<iostream>
#include<vector>
using namespace std;
int n,x;
const int N=100*1000+4;
long long a[N];
long long minor;
int main()
{
		cin>>n>>x;
		for(int i=1;i<=n;i++)
				cin>>a[i];
		minor=a[1];
		for(int i=1;i<=n;i++)
				minor=min(minor,a[i]);
		long long  u=0;
		for(int i=1;i<=n;i++)
				a[i]-=minor;
		while(a[x]!=0)
		{
				u++;
				a[x]--;
				x--;
				if(x<1)x=n;
		}
		a[x]+=u+minor*n;
		for(int i=1;i<=n;i++)
				cout<<a[i]<<" ";
		cout<<endl;
		return 0;
}
