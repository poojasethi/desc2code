#include<cstdio>
using namespace std;
__int64 f[111111],n,v,i,j,s,g;
int main(){
// freopen("input.txt","r",stdin);
// freopen("output.txt","w",stdout);
scanf("%d%d",&n,&v);
for(i=1;i<=n;i++)scanf("%d",f+i);
int k = 1111111111 , l ; 
for(i=1;i<=n;i++){
  if(f[v]<k){
   l = v;
   k = f[v];
  }
 --v;
 if(v == 0)v = n;
 }
g = l;
s = k * n;
for(i=1;i<=n;i++)f[i]-=k;
 while(v != l){
  l%=n;
  ++l;
  --f[l];
  ++s;
 }
f[g] = s;
for(i=1;i<=n;i++)printf("%I64d ",f[i]);
return 0;
}
