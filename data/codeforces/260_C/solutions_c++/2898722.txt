#include<iostream>

using namespace std;


int main()
{
    int n,i,x;
    long long a[100000],t,min=999999999999;
    cin>>n>>x;
    
    for(i=1;i<=n;i++){
        cin>>a[i];
        if(a[i]<min)
            min=a[i];
    }
    if (min>50){
    for(i=1;i<=n;i++)
    a[i]-=(min-3);
    
    t+=(min-3)*n;
    }
    while(a[x]>0){
        a[x]-=1;
        t++;
        x--;
        if(x==0)
        x=n;
    }
    
    a[x]=t;

    for(i=1;i<=n;i++)
    cout<<a[i]<<" ";

    return(0);

}
