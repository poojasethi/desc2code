// 有点意思，算个逆推吧
#include<cstdio>
#include<algorithm>
using namespace std;

int n,x;
long long v[111111];
long long mi,ans;

int pre(int x) {
	return x==1?n:x-1;
}

int main() {
	scanf("%d%d", &n, &x);
	mi=1111111111;
	for(int i=1;i<=n;i++){
		scanf("%I64d", &v[i]);
		mi=min(mi,v[i]);
	}
	ans=1LL*mi*n;
	for(int i=1;i<=n;i++)
		v[i]-=mi;
	for(;v[x];x=pre(x)){
		ans++;v[x]--;
	}
	for(int i=1;i<=n;i++)
		printf("%I64d%c",i==x?ans:v[i],i<n?' ':'\n');
	
	return 0;
}
