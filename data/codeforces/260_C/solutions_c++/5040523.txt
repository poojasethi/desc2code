#include <iostream>

using namespace std;

int main() {
  int n, x;
  cin >> n >> x;
  unsigned long long a[n];
  x--;
  for (int i = 0; i < n; i++)
    cin >> a[i];
  unsigned long long M = a[0];
  
  for (int i = 1; i < n; i++)
    M = min(M, a[i]);
    
  for (int i = 0; i < n; i++)
    a[i] -= M;
    
  unsigned long long d = M * n;
  while (a[x]) {
    a[x]--;
    d++;
    if (!x)
      x = n - 1;
    else
      x--;
  }
  a[x] = d;
  for (int i = 0; i < n; i++)
    cout << a[i] << " ";
  return 0;
}
