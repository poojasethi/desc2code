#include <iostream>

using namespace std;

int main()
{
    long long n, x, arr[100000], i, k, mm = 1LL << 50;
    
    cin >> n >> x;
    
    for (i = 0; i < n; i++)
    {
        cin >> arr[i];
        if (arr[i] < mm) mm = arr[i];
    }
    
    for (i = 0; i < n; i++) arr[i] -= mm;
    k = n * mm;

    for (i = x - 1; ; i--)
    {
        if (!arr[i]) { arr[i] = k; break; }
        arr[i]--; k++;
        if (!i) i = n;
    }

    for (i = 0; i < n; i++) cout << arr[i] << " ";
    return 0;
}
