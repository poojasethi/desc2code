#include<stdio.h>
int main()
{
    long long a[100003],n,mmin,te;
    int x,i;
    while(~scanf("%I64d%d",&n,&x))
    {
        for(mmin=1000000001,i=1;i<=n;i++)
            scanf("%I64d",&a[i]),mmin=(a[i]<mmin? a[i]:mmin);
        for(te=n*mmin,i=1;i<=n;i++)
            a[i]-=mmin;
        for(i=x;a[i]>0;a[i]--,i--,te++,i=(i==0? n:i));
        a[i]=te;
        for(i=1;i<n;i++)
            printf("%I64d ",a[i]);
        printf("%I64d\n",a[n]);
    }
    return 0;
}
