#include<cstdio>
#define FOR(a,b) for(i=a;i<=b;++i)
#define FORD(a,b) for(i=a;i>=b;--i)
int n,x,p=1,f,i,t,a[100010],l;long long w;
main(){
  scanf("%d%d",&n,&x);
  FOR(1,n)scanf("%d",&a[i]),a[p]>a[i]?p=i:p=p;
  FORD(x,1)
    if (a[i]==a[p]){p=i,f=1;break;}
  if (!f)
    FORD(n,x+1)
      if (a[i]==a[p]){p=i;break;}
  l=a[p],w=n*1ll*l;
  FOR(1,n) a[i]-=l;
  if (p>x){w+=n-(p-x);FOR(1,x)--a[i];FOR(p+1,n)--a[i];}
  else{w+=x-p;FOR(p+1,x)--a[i];}
  FOR(1,n)if(i==p)printf("%lld ",w);else printf("%d ",a[i]);
}
