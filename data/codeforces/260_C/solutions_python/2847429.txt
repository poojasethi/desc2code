#!/usr/bin/env python
# -*- coding: utf-8 -*-

n, x = [ int(t) for t in raw_input().split() ]
balls = [ int(t) for t in raw_input().split() ]

x -= 1

m = 0
if not 0 in balls:
    i = -1
    for j in range(n):
        if i==-1 or balls[j]<balls[i]:
            i = j
    m = balls[i]
    for j in range(n):
        balls[j] -= m

i = balls.index(0)
if i>x:
    i -= n
sum = 0
for j in range(x, i, -1):
    if balls[j] == 0:
        i = j
        break
    balls[j] -= 1
    sum += 1

balls[i] += sum + m * n

print ' '.join([ str(i) for i in balls ])
