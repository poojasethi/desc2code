n, x = map(int, raw_input().split())
A = map(int, raw_input().split())
minv = min(A)
ans = minv*n
A = map(lambda x: x-minv, A)
for i in xrange(n):
        idx = x-1-i
        if A[idx] == 0:
                A[idx] = ans
                break
        A[idx] -= 1
        ans += 1

print " ".join(map(str, A))
