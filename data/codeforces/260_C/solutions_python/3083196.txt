#!/usr/local/bin/python

n, x = map(int, raw_input().split())
x-=1

a = map(int, raw_input().split())

minim = min(a);
if minim > 0:
	a = map(lambda x: x-minim+1, a)

inc = 0
while True:
	if x < 0: x = n-1
	if a[x] > 0:
		a[x]-=1
		inc+=1
	else: break
	x-=1
a[x]+=inc
if minim > 0:
	a[x]+=n*(minim-1)

print ' '.join(map(str, a))
