ri = lambda: int(raw_input())
rl = lambda: raw_input().split()
rli = lambda: map(int, raw_input().split())

n, x = rli()
data = rli()

m = min(data)
x = x-1
i = x
a = n
while True:
  if data[i] == m:
    break
  i = (i-1)%n
  a -= 1

a = n-a
#print i, a
'''
if x < i:
  a = x + n - i
else:
  a = x-i
'''
ans = [ 0 for k in xrange(n)]
ans[i] = m*n + a
j = (i+1)%n
while j != i:
  #print ans
  ans[j] = data[j] - m
  if a:
    ans[j] -= 1
    a -= 1

  j = (j+1)%n


print ' '.join(map(str, ans))
