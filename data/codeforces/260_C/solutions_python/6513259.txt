import sys

n_boxes, end_index = tuple([int(x) for x in sys.stdin.readline().split()])
vals = [int(x) for x in sys.stdin.readline().split()]


r = list(range(0, end_index)[::-1]) + list(range(end_index, n_boxes)[::-1])
start_index, mini = r[0], vals[r[0]]
for i in r:
    val = vals[i]
    if val < mini:
        mini = val;
        start_index = i

n_balls_in_start = n_boxes * vals[start_index]
if end_index > start_index:
    n_balls_in_start += end_index-start_index-1
else:
    n_balls_in_start += n_boxes-start_index+end_index-1

every_increment = n_balls_in_start / n_boxes
leftover = n_balls_in_start % n_boxes

#print(start_index, n_balls_in_start, every_increment, leftover)
for i in range(n_boxes):
    if i != start_index:
        vals[i] -= every_increment

c_index = (start_index+1) % n_boxes
while leftover != 0:
    vals[c_index] -= 1
    leftover -= 1
    c_index = (c_index+1) % n_boxes

vals[start_index] = n_balls_in_start

for val in vals:
    sys.stdout.write(str(val) + " ")
print("")
