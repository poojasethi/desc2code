#!/usr/bin/python2.7
def nxtIter(i, n):
    if i == 0:
        return n-1
    else:
        return i-1
#    return i + 1 if i != n - 1 else 0


def main():
    n, stop = map(int, raw_input().split())
    data = map(int, raw_input().split())
    
    stop -= 1
    pos_value = min(data)
    pos = None
    i = stop
    while True:
        if data[i] == pos_value:
            pos = i
            break;
        i = nxtIter(i, n)
        
    #i = f(pos, n)
    sum = 0
    for x in range(n):
        data[x] = data[x]-pos_value
        sum += pos_value
    i = stop
    while True:
        if data[i] == 0:
            data[i]=sum
            break
        else:
            data[i] = data[i]-1
            sum += 1
        i = nxtIter(i,n)

    print ' '.join(map(str,data))



if __name__ == '__main__':
    main()
