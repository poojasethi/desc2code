line1 = raw_input()
n, x = map(int, line1.split())
line2 = raw_input()
a = map(int, line2.split())
b = [i for i in a]
minindex = x-1
minvalue = a[minindex]
for i in range(n):
    curindex = (x-1-i) % n
    curvalue = a[curindex]
    if curvalue < minvalue:
        minindex = curindex
        minvalue = curvalue
selected = minindex
numselected = (x-1 - selected) % n + (n * a[selected])
for i in range(n):
    b[i] = b[i] - int(numselected / n)
for i in range((x-1-selected)%n):
    exindex = (selected + i + 1) % n
    b[exindex] = b[exindex] - 1

b[selected] = numselected
