#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define LL long long
const int mod=1000000007;
const int N=600020;
struct SAMnode
{
    int son[28],pre,step;
}h[N];
int v[N],next[N],point[N],bs=0;
int sum1[N],sum2[N],sum3[N],rt[N],last=0,tot=0;
int sta[N],pin[N],pout[N],bj[N];
char s[N],s1[N];

void addchar(int j)
{
    int p=last,np=++tot;h[np].step=h[p].step+1;
    for (;p&&!h[p].son[j];p=h[p].pre) h[p].son[j]=np;
    if (!p&&!h[p].son[j]) h[p].son[j]=np,h[np].pre=p; else
        {
            int q=h[p].son[j];
            if (h[q].step==h[p].step+1) h[np].pre=q; else
                {
                    int nq=++tot;h[nq]=h[q];
                    h[nq].step=h[p].step+1;h[q].pre=h[np].pre=nq;
                    for (;h[p].son[j]==q;p=h[p].pre) h[p].son[j]=nq;
                }
        }
    last=np;
}
void addedge(int x,int y) {bs++;v[bs]=y;next[bs]=point[x];point[x]=bs;}
void dfs()
{
    int top,x,p,tmt=0;sta[top=1]=0;
    while (top)
    {
        x=sta[top],p=point[x];
        if (!p) {pout[x]=tmt;top--;continue;}
        pin[v[p]]=++tmt;sta[++top]=v[p];point[x]=next[p];
        sum1[tmt]=sum1[tmt-1],sum2[tmt]=sum2[tmt-1],sum3[tmt]=sum3[tmt-1];
        if (rt[v[p]]==1) sum1[tmt]++;
         else if (rt[v[p]]==2) sum2[tmt]++;
          else if (rt[v[p]]==3) sum3[tmt]++;
    }
}
void modify(int l,int r,int val)
{
    bj[l]+=val;if (bj[l]>=mod) bj[l]-=mod;
    bj[r+1]-=val;if (bj[r+1]<0) bj[r+1]+=mod;
}
void Tongji()
{
    int i;LL a,b,c;
    for (i=1;i<=tot;i++)
    {
         a=sum1[pout[i]]-sum1[pin[i]-1];
         b=sum2[pout[i]]-sum2[pin[i]-1];
         c=sum3[pout[i]]-sum3[pin[i]-1];
         modify(h[h[i].pre].step+1,h[i].step, ((int)(a*b%mod*c%mod)) );
    }
}
int main()
{
   // freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
    int i,n,n1,n2,n3,m;
    scanf("%s",s+1);n1=strlen(s+1);
    scanf("%s",s1+1);n2=strlen(s1+1);
    s[n1+1]=((char)('z'+1));for (i=1;i<=n2;i++) s[n1+1+i]=s1[i];
    scanf("%s",s1+1);n3=strlen(s1+1);
    s[n1+1+n2+1]=((char)('z'+2));for (i=1;i<=n3;i++) s[n1+1+n2+1+i]=s1[i];
    n=n1+n2+n3+2;m=min(min(n1,n2),n3);for (i=1;i<=n/2;i++) swap(s[i],s[n-i+1]);
    
    for (i=1;i<=n3+1;i++) addchar(s[i]-'a'),rt[last]=3;
    for (i=n3+1+1;i<=n3+1+n2+1;i++) addchar(s[i]-'a'),rt[last]=2;
    for (i=n3+1+n2+1+1;i<=n;i++) addchar(s[i]-'a'),rt[last]=1;
    for (i=1;i<=tot;i++) addedge(h[i].pre,i);
    
    dfs();Tongji();int ans=0;
    for (i=1;i<=m;i++)
    {
        ans+=bj[i];if (ans>=mod) ans-=mod;
        if (i>1) printf(" ");printf("%d",ans);
    }
    printf("\n");
    return 0;
}
