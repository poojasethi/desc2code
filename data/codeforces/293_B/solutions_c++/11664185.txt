#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define MOD 1000000007
using namespace std;
int n,m,k,Log2[1<<20],a[20][20],f[20][20],used[20];
int dfs(int x,int y){
	if(y==m+1){
		y=1;
		x++;
	}
	if(x==n+1){
		return 1;
	}
	int p=0,w=-1,cc=f[x-1][y]|f[x][y-1];
	for(int i=(~cc)&((1<<k)-1);i;i-=(i&(-i))){
		int t=Log2[i&(-i)]+1;
		if(!a[x][y] || a[x][y]==t){			
			f[x][y]=cc|(1<<(t-1));
			used[t]++;
			if(used[t]==1)
			{
				if(w==-1)w=dfs(x,y+1);
				p+=w;
			}
			else if(used[t]){
				p+=dfs(x,y+1);
			}
			if(p>=MOD)p-=MOD;
			used[t]--;
		}
	}
	return p;
}
int main(){
	cin>>n>>m>>k;
	if(n+m-1>k){
		cout<<0<<endl;
		return 0;
	}
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++)
		{
			scanf("%d",&a[i][j]);
			used[a[i][j]]++;
		}
	}
	for(int i=1;i<=k;i++){
		Log2[1<<i]=i;
	}
	cout<<dfs(1,1)<<endl;
	return 0;
}
