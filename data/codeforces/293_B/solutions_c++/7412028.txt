#define MOD 1000000007
using namespace std;
int n,m,k,Log2[1<<12],g[15][15],used[15],f[15][15];

int dfs(int x,int y){
	if(y==m+1){
		x++;
		y=1;
	}
	if(x==n+1){
		return 1;
	}
	int cc=f[x-1][y]|f[x][y-1],p=0,i,w=-1;
	for(i=(~cc)&((1<<k)-1);i;i-=(i&(-i))){
		int t=Log2[i&(-i)]+1;
		if(!g[x][y] || g[x][y]==t){
			f[x][y]=cc|(1<<(t-1));
			used[t]++;
			if(used[t]==1){
				if(w==-1){
					w=dfs(x,y+1);
				}
				p+=w;
			}else if(used[t]){
				p+=dfs(x,y+1);
			}
			if(p>MOD)p-=MOD;
			used[t]--;
		}
	}
	return p;
}

int main(){
	scanf("%d%d%d",&n,&m,&k);
	if(n+m-1>k){
		cout<<"0"<<endl;
		return 0;
	}
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			scanf("%d",&g[i][j]);
			used[g[i][j]]++;
		}
	}
	for(int i=1;i<=k;i++){
		Log2[1<<i]=i;
	}
	printf("%d\n",dfs(1,1));
	return 0;
}
