using namespace std;
const int N=20;
const int MOD=1e9+7;
int n,m,k;
int g[N][N],f[N][N],vis[N],Log2[1<<N];
int dfs(int x,int y)
{
	if(y==m+1)
	{
		x++;
		y=1;
	}
	if(x==n+1) return 1;
	int res=0,i,w=-1;
	int cur=(f[x-1][y]|f[x][y-1]);
	for(i=(~cur)&((1<<k)-1);i;i-=(i&(-i)))
	{
		int t=Log2[i&(-i)]+1;
		if(!g[x][y]||g[x][y]==t)
		{
			f[x][y]=cur|(1<<(t-1));
			vis[t]++;
			if(vis[t]==1){
				if(w==-1)
					w=dfs(x,y+1);
				res+=w;
			}
			else if(vis[t])
				res+=dfs(x,y+1);
			if(res>=MOD) res-=MOD;
			vis[t]--;
		}
	}
	return res;
}
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	if(n+m-1>k)
	{
		printf("0\n");
		return 0;
	}
	for(int i=1;i<=n;i++)
	    for(int j=1;j<=m;j++)
	    {
	   		scanf("%d",&g[i][j]);
	   		vis[g[i][j]]++;
        }
    for(int i=1;i<=k;i++)
    Log2[1<<i]=i;
    printf("%d\n",dfs(1,1));
	return 0;
}
