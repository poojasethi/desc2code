#include <iostream>
#define MOD 1000000007
using namespace std;
int n,m,k;
int g[15][15],a[15][15],f[15][15];
int log[1<<15],used[15];

int dfs(int x,int y){
	if(y>m){
		y=1;x++;
	}
	if(x>n)return 1;
	int cc=(f[x-1][y]|f[x][y-1]);
	int i,w=-1,res=0,t;
	for(int i=(~cc)&((1<<k)-1);i!=0;i-=i&(-i)){
		t=log[i&(-i)]+1;
		if(g[x][y]==0||g[x][y]==t){
			f[x][y]=cc|(1<<(t-1));
			used[t]++;
			if(used[t]==1){
				if(w==-1)w=dfs(x,y+1);
				res+=w;
			}
			else{
				if(used[t])res+=dfs(x,y+1);
			}
			used[t]--;
			res%=MOD;
		}
	}
	return res;
}

int main(){
	ios::sync_with_stdio(false);
	cin>>n>>m>>k;
	if(n+m-1>k){
		cout<<0;
		return 0;
	}
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			cin>>g[i][j];
			used[g[i][j]]++;
		}
	}
	for(int i=1;i<=k;i++)log[1<<i]=i;
	cout<<dfs(1,1);
	return 0;
}
