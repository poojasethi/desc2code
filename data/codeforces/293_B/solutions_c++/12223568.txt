#include<cstdio>
#include<iostream>
const int P=1000000007;
using namespace std;

int n,m,k;
int g[15][15],f[15][15],used[11];
int log[1500];

int dfs(int x,int y){
     if(y==m+1){x++;y=1;}
     if(x==n+1) return 1;
     f[x][y]=f[x-1][y]|f[x][y-1];
     int i=f[x][y],c;
     int s=0,p=-1;
     int t=f[x][y];
     for(i=(~i)&((1<<k)-1);i;i-=i&(-i))
     {   c=log[i&(-i)]+1;
         if(g[x][y]==0 || g[x][y]==c)
         {  ++used[c];
            f[x][y]=t|(i&(-i));
            if(used[c]==1)
            {  if(p<0) p=dfs(x,y+1);
               s+=p;
            }
            else
               s+=dfs(x,y+1);
            if(s>P) s-=P;
            used[c]--;
         }
         
     }
     return s;
}
     
int main(){
    scanf("%d%d%d",&n,&m,&k);
    if(n+m-1>k) {printf("0\n");return 0;}
    int i,j;
    for(i=0;i<=k;i++)
        log[1<<i]=i;

    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++){
            scanf("%d",&g[i][j]);
            used[g[i][j]]++;
    }
    printf("%d\n",dfs(1,1));
    return 0;
}
