#include <bits/stdc++.h>
#define MAX_N 100010 

using namespace std; 

char s[MAX_N]; 

int n,m,k,l,r,c,f; 

int main () 
	{
		cin >> n >> m >> k; 
		
		scanf("%s",s + 1); 
		
		int sum = m + k;
		
		for (int i = 1; i <= sum; i++) 
			{
				cin >> f >> l >> r >> c;
				if (f == 1) memset(s+l,c + '0', r-l+1);
					else cout << ((memcmp(s+l,s+l+c,r-l-c+1)) ? "NO" : "YES") << endl; 
			}
		return 0; 
	}
