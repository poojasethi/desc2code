#include<bits/stdc++.h>
using namespace std;
const int maxn = 1e5+5;
char s[maxn];
int main()
{
    int n,m,k; scanf("%d%d%d\n",&n,&m,&k);
    gets(s+1);
    int ct = m+k;
    while(ct--){
        int op,l,r,x; scanf("%d%d%d%d",&op,&l,&r,&x);
        if(op == 1) memset(s+l,'0'+x,r-l+1);
        else puts(memcmp(s+l,s+l+x,r-l+1-x)?"NO":"YES");
    }
    return 0;
}
