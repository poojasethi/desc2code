#include <bits/stdc++.h>

using namespace std;

int n, m, k;
char s[100005];

int main () {
    cin >> n >> m >> k;
    
    scanf("%s",s+1);
    
    for (int i=0;i<(m+k);i++) {
        int t, l, r, c;
        scanf("%d %d %d %d",&t,&l,&r,&c);
        
        if (t == 1) { memset(s + l, '0' + c, r - l + 1); continue; }
        
        if (memcmp(s + l, s + l + c, r - l + 1 - c)) {
            cout << "NO" << endl;
        } else {
            cout << "YES" << endl;
        }
    }
}
