#include<stdio.h>
#include<string.h>
int a,b,c,d,e,i,g;
char s[100001];
int main(){
    scanf("%d%d%d",&a,&b,&c);
    scanf("%s",s+1);
    b+=c;
    for(i=0;i<b;i++){
        scanf("%d%d%d%d",&c,&d,&e,&g);
        if(c==1) memset(s+d,g+'0',e-d+1);
        else printf("%s\n",memcmp(s+d,s+d+g,e-d-g+1) ? "NO" : "YES");
    }
    return 0;
}
