#include <bits/stdc++.h>
using namespace std;
const int maxn = (int) 1e5 + 5;
char s[maxn];
int n, m, k;
int main() {
    cin >> n >> m >> k;
    scanf("%s", s+1);
    int x, l, r, c;
    for(int i = 1; i <= m + k; ++i)  {
        scanf("%d%d%d%d", &x, &l, &r, &c);
        //r--; l--;
        if (x == 1) {
            memset(s+l, c + '0',r - l + 1);
        }
        else {
                if (!memcmp(s + l, s+l+c, r-l+1-c)) puts("YES");
                else puts("NO");
        }
    }
}
