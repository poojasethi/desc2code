int main() {

    cin >> N >> M >> K;
    cin >> s;
    for (int i = 0; i < N; i++)s[i] -= '0';
    int t, l, r, c;
    for (int i = 0; i < M + K; ++i) {
        cin >> t >> l >> r>> c;
        --l; --r;
        if (t == 1) memset(s + l, c, r - l + 1);
        else {
            bool flag = true;
            if (r - l + 1 - c > 0) {
                flag = memcmp(s + l, s + l + c, r - l + 1 - c) == 0;
            }
            cout << (flag ? "YES\n" : "NO\n");
        }
    }
    return 0;
