#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=100005;
char s[N];int n,m,k;
int main(){
	scanf("%d%d%d%s",&n,&m,&k,s);
	int tp;int l,r;int d;char c;
	for(int i=1;i<=m+k;i++){
		scanf("%d",&tp);
		if(tp==1){
			scanf("%d%d %c",&l,&r,&c);
			//cout<<c<<endl;
			memset(s+l-1,c,r-l+1);
			//cout<<s<<endl;
		}
		else {
			scanf("%d%d%d",&l,&r,&d);
			puts( memcmp(s+l-1,s+l+d-1,r-l+1-d) ? "NO":"YES");
		}
	}
	return 0;
}
