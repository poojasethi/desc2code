#include<iostream>
#include<cmath>
#include<cstdio>
#include<algorithm>
#include<list>
#include<queue>
#include<stack>
#include<string.h>

#define ll long long
#define MIN(a,b) ((a) < (b) ? (a) : (b))
#define MAX(a,b) ((a) > (b) ? (a) : (b))

using namespace std;

int n, m, k, op, l, r ,d;
char a[ 100009 ], tmp;

int main(){

    scanf("%d %d %d\n", &n, &m, &k);
    for ( int i=0;i<n;++i)
        scanf("%c",&a[i]);

    for ( k+=m; k>0; --k ){
        scanf("%d %d %d %d", &op, &l, &r, &d);

        switch( op ){
            case 1:
                tmp = ((char)d) + '0';
                memset( a+l-1, tmp, sizeof( char ) * (r-l+1));
                break;
            case 2:
                if (memcmp( a+l-1, a+l+d-1, (r-l-d+1) ) == 0)
                    printf("YES\n");
                else
                    printf("NO\n");
                break;
        }

    }


}
