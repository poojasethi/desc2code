    # include <iostream>
# include <cstring>
# include <cstdio>
using namespace std;
char s[100100];

char ans[600005];
int ln = -1;

int main() {
    int n, m, k;
    cin >> n >> m >> k;
    scanf("%s", &s);
    for (int i = 0; i < n; ++i) {
        s[i] -= '0';
    }
    char t[2];
    int l, r, c;
    for (int i = 0; i < m + k; ++i) {
        
        scanf("%s%d%d%d", t, &l, &r, &c);
        --l; --r;
        if (t[0] == '1') {
            memset(s + l, c, r - l + 1);
        } else {
            char e = true;
            if (r - l + 1 - c > 0) {
                e = memcmp(s + l, s + l + c, r - l + 1 - c) == 0;
            }
            if (e){
                ans[++ln] = 'Y';ans[++ln] = 'E';ans[++ln] = 'S';ans[++ln] = '\n';
            }else{
                ans[++ln] = 'N';ans[++ln] = 'O';ans[++ln] = '\n';
            
            }
        }
    }
    ans[++ln] = '\0';
    puts(ans);
    return 0;
}
