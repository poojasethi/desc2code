# include <iostream>
# include <cstring>
# include <cstdio>
using namespace std;
char s[100100];

int main() {
    int n, m, k;
    cin >> n >> m >> k;
    scanf("%s", &s);
    for (int i = 0; i < n; ++i) {
        s[i] -= '0';
    }
    for (int i = 0; i < m + k; ++i) {
        char t[2];
        int l, r, c;
        scanf("%s%d%d%d", t, &l, &r, &c);
        --l; --r;
        if (t[0] == '1') {
            memset(s + l, c, r - l + 1);
        } else {
            bool e = true;
            if (r - l + 1 - c > 0) {
                e = memcmp(s + l, s + l + c, r - l + 1 - c) == 0;
            }
            puts(e ? "YES" : "NO");
        }
    }
    return 0;
}
