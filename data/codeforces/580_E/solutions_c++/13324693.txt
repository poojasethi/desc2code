#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

const int N = 1e5 + 7;

char s[N];

int n,m,k,id,foo,bar,x;

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif
	cin>>n>>m>>k;
	k += m;
	scanf("%s",s);
	while(k--)
	{
		scanf("%d%d%d%d",&id,&foo,&bar,&x);
		foo--,bar--;
		if(id == 1) memset(s+foo,x+'0',bar-foo+1);
		else
		{
			int r = 0;
			if(x+1 <= bar-foo+1) r = memcmp(s+foo,s+foo+x,bar-foo+1-x);
			puts(r ? "NO":"YES");
		}
	}
	return 0;
}
