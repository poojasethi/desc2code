#include <bits/stdc++.h>
using namespace std;

#define ll long long int 
#define ten5 100001
#define mp make_pair
#define pb push_back

char s[ten5];

int main(){
	int n,m,k;
	scanf("%d%d%d%s",&n,&m,&k,s);
	for(int i=0;i<m+k;i++){
		int f,l,r,c;
		scanf("%d%d%d%d",&f,&l,&r,&c);
		l--; r--;
		if(f==1) memset(s+l,c+'0',r-l+1);
		else if(memcmp(s+l,s+l+c,r-l-c+1)) puts("NO");
		else puts("YES");
	}
}
