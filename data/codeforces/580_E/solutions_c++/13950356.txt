#include <bits/stdc++.h>

using namespace std;
const int maxn = 100005;
int n,m,k;
char s[maxn];

int main()
{
    ios_base::sync_with_stdio(0); cin.tie(0);
    //freopen("watch.inp","r",stdin);
    //freopen("watch.out","w",stdout);
    cin >> n >> m >> k;
    cin >> s;
    for (int i = 0; i < n; i++)s[i] -= '0';
    int t, l, r, c;
    for (int i = 0; i < m + k; ++i) {
        cin >> t >> l >> r>> c;
        --l; --r;
        if (t == 1) memset(s + l, c, r - l + 1);
        else {
            bool flag = true;
            if (r - l + 1 - c > 0) {
                flag = memcmp(s + l, s + l + c, r - l + 1 - c) == 0;
            }
            if (flag) cout << "YES\n";
            else cout << "NO\n";
        }
    }
    return 0;
}
