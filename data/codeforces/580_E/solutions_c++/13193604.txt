
#include<bits/stdc++.h>
using namespace std ;

int n , m , k , t ;
char s[100010] ;
int op , l ,  r , x ;

int main()
{

    scanf("%d %d %d\n",&n,&m,&k) ;

    gets(s+1) ;
    t = m+k ;
    while(t--)
    {

        scanf("%d %d %d %d", &op , &l , &r , &x) ;
        if( op == 1 ) memset( s+l , '0'+x , r-l+1 ) ;
        else cout << ( memcmp( s+l , s+l+x , r-l+1-x )  ?"NO\n":"YES\n") ;
    }
    return 0 ;
}
