#include <bits/stdc++.h>
#define foru(i, a, b) for(int i = a; i <= b; i++)

using namespace std;
const int maxn = 1e5 + 1;
int n, m, k;
char a[maxn];

int main() {
    cin >> n >> m >> k;
    scanf("%s", a);
    int x, l, r, c;
    foru (i, 1, m + k) {
        scanf("%d %d %d %d", &x, &l, &r, &c);
        l--; r--;
        if (x == 1) memset(a + l, c + '0', r - l + 1);
        if (x == 2) 
            cout << ((memcmp(a + l, a + l + c, r - l - c + 1) == 0) ? "YES" : "NO") << endl;
    }
    return 0;
}
