#include<bits/stdc++.h>

using namespace std;
const int N = 1000000 + 5;
char ch[N];
int main(){
	int n, m, k;
	scanf("%d%d%d", &n, &m, &k);
	for (int i = 1; i <= n; i++) scanf(" %c", &ch[i]);
	for (int i = 1; i <= m + k; i++){
		int d, p, q, z;
		scanf("%d%d%d%d", &d, &p, &q, &z);
		if (d == 1) memset(ch + p, z + 48, q - p + 1);
		else puts(!(memcmp(ch + p, ch + p + z, q - p + 1 - z)) ? "YES" : "NO");
	}
}
