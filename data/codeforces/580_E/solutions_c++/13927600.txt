const int maxn = 100010;
char s[maxn];
int main()
{
    int n, m, k;
    scanf("%d%d%d", &n, &m, &k);
    scanf("%s", s);
    int o = m + k, f, l, r, d;
    while(o--){
        scanf("%d%d%d%d", &f, &l, &r, &d);
        if(f == 1) memset(s+l-1, '0' + d, r - l + 1);
        else printf("%s\n", memcmp(s+l-1, s+l-1+d, r-l+1-d) == 0? "YES":"NO");
    }
    return 0;
}
