#include <cstdio>
#include <cstring>
int main(){
    int n, m, k, ope, l, r, x;
    char s[1000001];
    scanf("%d%d%d", &n, &m, &k);
    scanf("%s",s+1);
    for(int i = 0; i < m + k; i++) {
        scanf("%d%d%d%d", &ope, &l, &r, &x);
        if(ope == 1) memset(s + l, x + '0', r - l + 1);
    else puts(memcmp(s + l, s + l + x, r - l + 1 - x) ? "NO" : "YES");  
    } 
}
