#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <queue>
#include <cmath>
#include <map>
#include <cstring>
using namespace std;
const int M = 100010;
int h[M<<2];
int n,m,k;
char a[M];
int main()
{
    scanf("%d%d%d",&n,&m,&k);
    getchar();
    for(int i=1;i<=n;i++)
        a[i]=getchar();
    for(int i=1;i<=m+k;i++)
    {
        int x,y,z,w;
        scanf("%d%d%d%d",&x,&y,&z,&w);
        if(x==1)
            memset(a+y,'0'+w,z-y+1);
        else
        {
            bool flag=memcmp(a+y,a+y+w,z-w-y+1);
            if(!flag)
                printf("YES\n");
            else
                printf("NO\n");
        }
    }
    return 0;
}
