using namespace std;
#include <iostream>
#include <cstdio>
#include <sstream>
#include <cstring>
#include <string>
#include <algorithm>
#include <vector>

#define MAX 100010

int n, m, k;
char vet[MAX];

int main(){

	scanf("%d %d %d", &n, &m, &k);

	for(int i = 0;i < n;i++){
		scanf(" %c", &vet[i]);
	}

	for(int i = 0;i < m+k;i++){
		int c, s, e, v;

		scanf("%d %d %d %d", &c, &s, &e, &v);
		s--; e--;

		if(c == 1){
			memset(vet+s, '0'+v, e-s+1);
		}
		else{
			if(!(memcmp(vet+s, vet+s+v, e-s-v+1))){
				printf("YES\n");
			}
			else{
				printf("NO\n"); 
			}
		}
	}
	return 0;
}
