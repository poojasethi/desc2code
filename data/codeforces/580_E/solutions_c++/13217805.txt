#include<bits/stdc++.h>
using namespace std;
#define ll long long 

int main()
{
    std::ios::sync_with_stdio(false);
    int n, m, k; cin>>n>>m>>k;
    char str[100005]; cin>>str;
    
    int temp=(m+k);
    while(temp--)
    {
        int c, l, r, d; cin>>c>>l>>r>>d;
        if(c==1)
            memset(str+l-1, d+'0', r-l+1);
        else
            if(memcmp(str+l-1 , str+l+d-1, r-d-l+1)==0)
                cout<<"YES\n";
            else
                cout<<"NO\n";         
    }
    return 0;    
}
