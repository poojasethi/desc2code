# include <iostream>
# include <cstring>
# include <cstdio>
using namespace std;
char s[100100];

int main() {
    ios_base::sync_with_stdio(0);
    int n, m, k;
    cin >> n >> m >> k;
    cin >> s;
    for (int i = 0; i < n; ++i) {
        s[i] -= '0';
    }
    for (int i = 0; i < m + k; ++i) {
        char t;
        int l, r, c;
        cin >> t >> l >> r >> c;
        --l; --r;
        if (t == '1') {
            memset(s + l, c, r - l + 1);
        } else {
            bool e = true;
            if (r - l + 1 - c > 0) {
                e = memcmp(s + l, s + l + c, r - l + 1 - c) == 0;
            }
            puts(e ? "YES" : "NO");
        }
    }
    return 0;
}
