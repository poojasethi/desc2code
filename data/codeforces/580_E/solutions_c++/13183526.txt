const int oo=1e5+5;
using namespace std;

inline int read()
{
	int x=0;
	int f=1,hhh=getchar();
	while(!isdigit(hhh))
	{
		if(hhh=='-')
			f=-1;
		hhh=getchar();
	}
	while(isdigit(hhh))
	{
		x=x*10+hhh-'0';
		hhh=getchar();
	}
	return x*f;
}

char s[oo];

int main()
{
    int n,m,k;
	n=read(),m=read(),k=read();
    scanf("%s",s+1);
    int T=m+k;
    while(T--)
	{
        int op,l,r,x;
		op=read(),l=read(),r=read(),x=read();
        if(op==1)
			memset(s+l,'0'+x,r-l+1);
        else
			puts(memcmp(s+l,s+l+x,r-l+1-x)?"NO":"YES");
    }
    return 0;
}
