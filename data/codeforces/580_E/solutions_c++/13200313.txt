#include <bits/stdc++.h>
using namespace std;  
const int maxn = 100100;

char str[maxn];

bool equal(const int& l, const int& r, const int& L, const int& R) {  
    if (r - l != R - L) {
        return false;
    }
    return !memcmp(str+l, str+L, r-l+1);
}


int main() {  
    int n, m, k;
    scanf("%d%d%d", &n, &m, &k);
    scanf("%s", str);
    for (int i = 0; i < m + k; i++) {
        int kind, l, r, c;
        scanf("%d%d%d%d", &kind, &l, &r, &c);
        l--;
        r--;
        if (kind == 1) {
            memset(str+l, c+'0', r-l+1);
        }
        else {
            if (equal(l+c, r, l, r-c)) {
                puts("YES");
            }
            else {
                puts("NO");
            }
        }
    }
    return 0;
}
