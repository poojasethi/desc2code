#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;

int main()
{
   int l,d,v,g,r;
   cin>>l>>d>>v>>g>>r;
   double t1=d*1.0/v;
   int tmp=floor(t1);
   if((tmp%(g+r))>=g)t1=(tmp/(g+r)+1)*(g+r);
   t1+=(l-d)*1.0/v;
   printf("%.8f\n",t1);
   return 0;
}
