#include<stdio.h>
int main()
{
	int l,d,g,r,v,p,a,b;
	double s,s1;
	while(scanf("%d%d%d%d%d",&l,&d,&v,&g,&r)!=EOF)
	{
	   s=(l+0.0)/v;
	   s1=(d+0.0)/v;
	   r+=g;
	   while(s1>=r)
	     s1-=r;
	   s1-=g;
	   if(s1>=0)
		   s+=(r-g-s1);
	   printf("%.8lf\n",s);
	}
	return 0;
}
