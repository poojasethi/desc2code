#include<iostream>
#include<iomanip>
#include<cmath>
using namespace std;
int main()
{
	double l,d,v,g,r,sum;
	cin>>l>>d>>v>>g>>r;
	if(v*g>d) sum=l/v;
	else
	{
		sum=d/v;
		sum=ceil(sum/(g+r));
		sum=sum*(g+r)+(l-d)/v;
	}

		cout<<setprecision(8)<<sum<<endl;


	return 0;
}
