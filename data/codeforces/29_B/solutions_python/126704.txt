import sys

def sol():
	(l, d, v, g, r) = [int(it) for it in sys.stdin.readline().split()]
	t = (d * 1.0) / v 	# time to reach light
	t1 = int(t) % (g + r)
	add = 0.0
	if t1 >= g:
		times = int(t) / (g + r)
		add = g + r - (t -  times * (g + r))
	ans = (l * 1.0) / v + add
	#print t, t1, add
	return ans
	
	
def main():
	print '%.7f' % sol()

if len (sys.argv) > 1:
	sys.stdin = open (sys.argv[1], 'r')
main()
