#include<stdio.h>
#define ll __int64
ll n,m,k,l,r,mid;
ll judge(ll x){
	ll ans=0;
	for(int j=1;j<=n;j++){
		ans+=(x/j>m?m:x/j);
	}
	return ans;
}
int main(){
	scanf("%I64d%I64d%I64d",&n,&m,&k);
	l=1;r=n*m;
	while(l<=r){
		mid=l+r>>1;
		if(judge(mid)>=k)r=mid-1;
		else l=mid+1;
	}
	printf("%I64d\n",l);
}
