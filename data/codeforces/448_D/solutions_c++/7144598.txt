#include <cstdio>
#include <algorithm>
int main()
{
	long long n,m,k,l=1;
//	scanf("%lld%lld%lld",&n,&m,&k);
	scanf("%I64d%I64d%I64d",&n,&m,&k);
	long long r=n*m;
	while (l<r) {
		long long mid=(l+r)>>1,sum=0;
		for (int i=1; i<=n; ++i) {
			sum+=std::min(mid/i,m);
		}
		sum<k?l=mid+1:r=mid;
	}
//	printf("%lld",l);
	printf("%I64d",l);
	return 0;
}
