#include<cstdio>
#include<iostream>
using namespace std;

int main()
{
	long long n,m,k;
	scanf("%I64d%I64d%I64d",&n,&m,&k);
	long long l=0,r=n*m;
	while(l<r)
	{
		long long mid=l+(r-l)/2;
		long long res=0;
		for(int i=1;i<=n;i++)
			res+=min(m,mid/i);
		if(res>=k)
			r=mid;
		else
			l=mid+1;
	}
	printf("%I64d\n",r);
	return 0;
}
