#include <bits/stdc++.h>
using namespace std;
int main(){
	long long n , m , k;
	cin>>n>>m>>k;
	long long l = 1 , r = n * m;
	while (l < r){
	    long long mid = (l + r) >> 1;
	    long long ans = 0;
	    for (int i = 1; i <= n; i++) ans += min(mid / i , m);
	    if(ans < k) l = mid + 1;
	    else r = mid; 
	}
	cout<<l<<endl; 
}
