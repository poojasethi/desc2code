#include<stdio.h>
#define ll long long

ll fu(int n,int m,ll x){
	int i;
	ll s=0;
	for(i=1;i<=n;i++)
		s+=((x-1)/i<m)?(x-1)/i:m;
	return s;
}
int main(){
int n,m;
ll k;
scanf("%d%d%I64d",&n,&m,&k);
ll l=1,r=(long long)n*m;
while(l<r){
	ll x=(l+r+1)/2;
	if(fu(n,m,x)<k)
		l=x;
	else r=x-1;
}
printf("%I64d",l);
return 0;
}
