#include <iostream>
using namespace std;
typedef long long ll;
ll n,m,k;
ll cnt(ll mid){
	ll ans = 0;
	for(ll i = 1; i <= min(n,mid); ++i){
		ll c = min(m,mid/i);
		ans+=c;
	}
	return ans;
}
int main(){
	cin >> n >> m >> k;
	ll lo = 1,hi=n*m,mid;
	while(lo<hi){
		ll mid = (lo+hi)/2;
		if(cnt(mid)<k)
			lo=mid+1;
		else
			hi=mid;
	}
	cout << lo << '\n';
}
