#include<bits/stdc++.h>
using namespace std;
long long int n,m;
long long k;
int main(){
    cin>>n>>m>>k;
    long long l=0,r=n*m;
    while(l+1<r){
        long long mid=(l+r)/2;
        long long res=0;
        for(int i=1;i<=n;++i)
            res+=min(mid/i,m);
        if(res>=k) r=mid;
        else l=mid;
    }
    cout<<r;
}
