#include<stdio.h>
#define LL long long
LL min(LL a,LL b)
{
	return a<b?a:b;
}
int main()
{
    LL n,m,k;
    scanf("%I64d%I64d%I64d",&n,&m,&k);
	LL l,r,mid;
	l=1;r=n*m;
	while(l<=r)
	{
		LL kk=0;
		mid=(l+r)/2;
		for(int i=1;i<=n;i++)
			kk+=min(m,mid/i);
		if(kk<k) l=mid+1;
		if(kk>=k) r=mid-1;
	}
	printf("%I64d\n",l);
}
