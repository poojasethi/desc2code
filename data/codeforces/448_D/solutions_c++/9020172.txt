#include <iostream>
using namespace std;
int main()
{
  long long n, m, k, s, i;
  cin >> n >> m >> k;
  long long a = 0, b = n * m, c;
  while(b - a > 1)
    {
      c = (a + b) / 2;
      s = 0;
      for(i = 1; i <= n; i++) s += min(m, c / i);
      if(s < k) a = c;
      else b = c;
    }
  cout << b << endl;
  return 0;
}
