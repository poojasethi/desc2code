#include<bits/stdc++.h>
using namespace std;
#define li long long 
li i,n,m,k,b,o,s;
li f(li x){
    li t=0;
    for(li i=1;i<=min(x,m);i++) t+=min(x/i,n);
    return t;
}
int main(){
    cin>>n>>m>>k;
    b=1; s=n*m;
    while(b<s){
        o=(b+s)/2;
        if((f(o))<k) b=o+1;
        else s=o;
    }
    cout<<b;
}
