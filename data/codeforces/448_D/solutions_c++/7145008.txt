#include <cstdio>
#include <algorithm>
using namespace std;

typedef long long ll;

int n,m;
ll k,l,r,now;

int main(){
	scanf("%d%d%I64d",&n,&m,&k);
	l=1,r=(ll)n*m+1;
	while(r-l>1){
		ll mid=l+r>>1;
		now=0;
		for(int i=1;i<=n;i++)
			now+=min((ll)m,(mid-1)/i);
		if(k<=now)
			r=mid;
		else
			l=mid;
	}
	printf("%I64d\n",l);
}
