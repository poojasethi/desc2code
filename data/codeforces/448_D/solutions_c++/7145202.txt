#include <iostream>

using namespace std;

typedef long long ll;

int main() {
	ll n, m, k;
	cin >> n >> m >> k;
	ll s = 1, e = n * m + 1;
	while (e - s > 1) {
		ll mid = (s + e) >> 1;
		ll cnt = 0;
		for (int i = 1; i <= n; i++)
			cnt += min((mid - 1) / i, m);
		if (cnt >= k)
			e = mid;
		else
			s = mid;
	}
	cout << s << endl;
	return 0;
}
