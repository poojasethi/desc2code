#include<bits/stdc++.h>

using namespace std;

#define lli long long int

lli n,m,k,i,j,x,y,z;

lli f(lli bas,lli son){
	lli i,top=0,x=(bas+son)/2;

	if(bas>=son){
		cout << x << endl;
		exit(0);
	}

	for(i=1 ; i<=m ; i++)
		top+=min(n,x/i);

	if(top>=z)
		f(bas,x);
	else 
		f(x+1,son);
}

int main(){

	cin >> n >> m >> z;

	f(1,n*m);
}
