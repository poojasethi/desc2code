#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	long long m,n,k;
	cin>>n>>m>>k;
	long long mid,low = 1 , high = m*n+1 ;
	while(low < high)
	{
		mid = (low+high)/2;
		long long add=0;
		for(long long i=1;i<=n;i++)
		{
			add+=min((mid)/i,m);
		}
		//add++;
		
		if(add<k)
		low = mid+1;
		
		else
		high = mid;
	}
	cout<<low<<endl;
}
