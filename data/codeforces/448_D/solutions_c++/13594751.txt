#include <bits/stdc++.h>
using namespace std;
long long  n,m,k,mid,low,high;
int main()
{
scanf("%I64d%I64d%I64d",&n,&m,&k);
low=1;
high=n*m;
while(low<high)
{
mid=(low+high)/2;
long long  l=0;
for(long long i=1;i<=n;i++)
l+=min(m,(mid)/i);
if(l<k)
low=mid+1;
else
high=mid;
}
printf("%I64d",high);
    return 0;
}
