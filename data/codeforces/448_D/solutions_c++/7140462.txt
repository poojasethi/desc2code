#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
	ll n, m, k;
	cin>>n>>m>>k;
	
	ll left = 1, right = n*m;
	
	while(left < right) {
		ll mid = (left + right)/2;
		ll total = 0;
		
		for(ll i=1; i<=n; i++)
			total += min(m, mid/i);
		
		if(total < k)
			left = mid + 1;
		else
			right = mid;
	}
	
	cout<<right;
	
	return 0;
}
