#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
int main()
{
	ll n,m,k,cnt,i;
	cin>>n>>m>>k;
	ll lo=1;
	ll hi=n*m;
	while(lo<hi)
	{
		ll mid=lo+(hi-lo)/2;
		cnt=0;
		//cntmid=0;
		for(i=1;i<=n;i++)
		{
			cnt+=min(m,mid/i);	
		}
		if(cnt>=k)
			hi=mid;
		else lo=mid+1;
	}
	cout<<lo;
}
