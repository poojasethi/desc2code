#include<bits/stdc++.h>
using namespace std;
#define Int long long
Int getSum(Int x,Int n,Int m){
	Int res = 0;
	x--;
	for(Int i=1;i<=n;i++){
		res += min(x/i,m);
	}
	return res;
}
int main(){

	Int n,m,k,l,r,x;
	cin>>n>>m>>k;
	l =1;r=n*m +1;
	while(l<r){
		x = (l+r)/2;
		if(getSum(x,n,m)<k)
			l = x+1;
		else
			r = x;
	}
	cout<<l-1<<endl;
	return 0;
}
