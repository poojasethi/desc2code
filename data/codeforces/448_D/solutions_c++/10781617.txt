#include<iostream>
using namespace std;
typedef long long int ll;
ll count(ll n,ll m,ll x){
	ll sum=0;
	x--;
	for(ll i=1;i<=n;i++)sum+=min(x/i,m);
	return sum;
}
int main()
{
	ll n,k,m,s,e,mid;
	cin>>n>>m>>k;
	s=1;
	e=n*m+1;
	while(s<e){
		mid=(s+e)/2;
		ll val=count(n,m,mid);
		if(val<k)s=mid+1;
		else e=mid;
	}
	cout<<s-1<<endl;
}
