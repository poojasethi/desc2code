#include <bits/stdc++.h>
#define ll long long int
using namespace std;
int main(){
	ll n,m,lo,hi,mid,k,sum;
	cin>>n>>m>>k;
	lo=0;
	hi=n*m;
	while(lo<hi-1){
		mid=(lo+hi)/2;
		sum=0;
		for(int i=1;i<=n;i++){
			sum+=min((mid)/i,m);
		}
		if(sum>k-1){
			hi=mid;
		}
		else{
			lo=mid;
		}
	}
	cout<<hi<<endl;
}
