#include <iostream>
using namespace std;
typedef long long ll;
ll n,m,k,l,r,s,mi;
int main(){
	cin>>n>>m>>k;
	l=0;
	r=n*m;
	while(l<r-1){
				 mi=(l+r)/2;
				 s=0;
				 for(int i=1;i<=n;i++){
				 		 ll t=mi/i;
				 		 if(t>m)t=m;
				 		 s+=t;
				 }
				 if(s>=k)r=mi;else l=mi;
	}
	cout<<r<<endl;
	return 0;
}
