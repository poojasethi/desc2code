#include<iostream>
using namespace std;
int main()
{
	long long n,m,k;
	cin>>n>>m>>k;
	long long mink=0,maxk=m*n;
	while(maxk-mink>1)
	{
		long long midk=(maxk+mink)/2,s=0;
		for(int i=1;i<=n;i++)s+=min(m,midk/i);
		//cout<<midk<<" "<<s<<" "<<ss<<endl;
		if(k>s)mink=midk;
		else maxk=midk;
	}
	cout<<maxk<<endl;
	return 0;
}
