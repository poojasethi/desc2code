#include <iostream>

using namespace std;

int main() {
    long long n,m,k,l,r,tm,ans;
    cin>>n>>m>>k;
    l=1; r=n*m+1;
    while(l<r) {
        tm=(l+r)/2;
        ans=0;
        for(int i=1;i<n+1;i++) ans+=min((tm-1)/i,m);
        if(ans<k) l=tm+1;
        else r=tm;
    }
    cout<<l-1;
    
    
    
    
    return 0;
}
