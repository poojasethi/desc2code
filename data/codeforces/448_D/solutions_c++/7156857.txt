#include<cstdio>
#include<algorithm>
using namespace std;
long long n,m,k,l,r;
int main(){
  scanf("%I64d%I64d%I64d",&n,&m,&k);
  l = 1, r = n*m;
  while(l <= r){
    long long chk = 0ll, t = (l+r)/2ll;
    for(int i=1;i<=n;i++)
      chk += min(t/i,m);
    if(chk >= k) r = t-1ll;
    else l = t+1ll;
  }
  printf("%I64d",l);
}
