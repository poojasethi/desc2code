#include<bits/stdc++.h>
using namespace std;
int main()
{
  cin.tie(NULL);
  ios_base::sync_with_stdio(0);
  long long n,m,k,l,r,mid,t;
  cin>>n>>m>>k;
  l=1; r=m*n+1;
  while(l<r){
    t=0;
    mid=(l+r)/2;
    for(int i=1;i<=n;i++){
      t+=min((mid-1)/i,m);
    }
    if(t<k) l=mid+1;
    else
      r=mid;
  }
  cout<<l-1<<"\n";
  return 0;
}
