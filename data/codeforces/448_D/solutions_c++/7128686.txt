#include <iostream>

using namespace std;

int main()
{
	long long n, m, k;
	cin >> n >> m >> k;
	long long lo = 0, hi = n * m;
	while (hi - lo > 1)
	{
		long long mid = (hi + lo) / 2;
		long long le = 0;
		for (int i = 1; i <= n; i++)
			le += min(m, mid / i);
		if (le >= k)
			hi = mid;
		else
			lo = mid;
	}
	cout << hi << endl;
}
