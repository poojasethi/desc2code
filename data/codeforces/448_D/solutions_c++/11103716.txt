# include <iostream>
# include <cstdio>
using namespace std;
typedef long long LL;

int main(void)
{
	LL n,m,k;cin>>n>>m>>k;
	LL l=1,r=n*m,t;
	while(l<r)
	{
		t=(l+r)/2;
		LL cnt=0;
		for(int i=1;i<=n;i++)
			cnt+=i*m<t?m:t/i;
		if(cnt>=k) r=t;
		else l=t+1;
	}
	printf("%lld\n",r);
	return 0;
}
