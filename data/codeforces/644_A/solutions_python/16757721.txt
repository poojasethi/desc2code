from sys import stdin

n, a, b = map(int, stdin.readline().split())
if a*b<n:
	print -1
else:
	nn = 1
	m = [[0]*b]*a
	for i in xrange(a):
		if i % 2 == 0:
			for j in xrange(b):
				m[i][j] = 0 if nn > n else nn
				nn += 1
		else:
			for j in xrange(b):
				m[i][b - 1 - j] = 0 if nn > n else nn
				nn += 1
		print " ".join([str(j) for j in m[i]])
