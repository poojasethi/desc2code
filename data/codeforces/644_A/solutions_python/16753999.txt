#!/usr/bin/python

import sys

def Nil(): return map(int, sys.stdin.readline().split())

n, a, b = Nil()

if a * b < n:
    print -1
else:
    R = [[0] * b for i in range(a)]
    i = 0
    j = 0
    dj = 1
    for p in range(1, n+1):
        R[i][j] = p
        j += dj
        if j == b or j == -1:
            i += 1
            j -= dj
            dj *= -1

    for r in R:
        print " ".join(map(str, r))
