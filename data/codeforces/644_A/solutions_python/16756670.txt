#!/usr/bin/env python

from sys import stdin, stderr

def main():
   N, A, B = map(int, stdin.readline().split())
   even_nums = range(2, N+1, 2)
   odd_nums = range(1, N+1, 2)
   T = [ [0] * B for r in xrange(A) ]
   for r in xrange(A):
      for c in xrange(B):
         if 0 == ((r+c) & 1):
            if len(odd_nums) > 0:
               T[r][c] = odd_nums.pop()
         else:
            if len(even_nums) > 0:
               T[r][c] = even_nums.pop()
   if len(odd_nums) + len(even_nums) > 0:
      print -1
   else:
      for row in T:
         print ' '.join(map(str, row))
   return 0

if __name__ == '__main__': main()
