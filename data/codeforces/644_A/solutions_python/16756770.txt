n, a, b = map(int, raw_input().split())
if n > a * b:
    print -1
else:
    seats = [[0] * b for i in xrange(a)]
    x = 0
    for i in xrange(a):
        for j in xrange(b):
            x += 1
            if x <= n:
                seats[i][j] = x
    for i in xrange(1, a, 2):
        seats[i] = seats[i][::-1]
    for row in seats:
        print ' '.join(map(str, row))
