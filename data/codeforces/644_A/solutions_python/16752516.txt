def read_int_list():
    return map(int, raw_input().split())


n, a, b = read_int_list()
if a * b < n:
    print -1
else:
    hall = [[0]*b for i in xrange(a)]
    x, y = 0, 0
    shift = 1
    i = 1
    while i <= n:
        hall[x][y] = i
        if (y == 0 and shift == -1) or (y == b-1 and shift == 1):
            x += 1
            shift *= -1
        else:
            y += shift
        i += 1
    for x in xrange(a):
        print " ".join(map(str, hall[x]))
