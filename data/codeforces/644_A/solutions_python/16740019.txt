import sys

n, a, b = map(int, raw_input().split())

if n > a*b:
  print -1
  sys.exit(0)

z = 1
for r in range(a):
  row = [0] * b
  if b % 2 == 0 and r % 2 == 1:
    # zig zag
    r2 = range(b-1, -1, -1)
  else:
    r2 = range(b)
  for c in r2:
    if z > n:
      row[c] = 0
    else:
      row[c] = z
      z += 1
  print ' '.join(map(str, row))
