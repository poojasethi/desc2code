n,a,b = [int(x) for x in raw_input().split(' ')]
z = 1
if a* b < n:
	print("-1")
else:
	ans = ''
	for i in xrange(a):			
		if i % 2 != 0:
			ans += "".join([(str(j) +' ')if j <=n else "0 "for j in xrange((i+1)*b, i*b, -1)])
		else:
			ans += "".join([(str(j) +' ')if j <=n else "0 " for j in xrange(i*b+1, (i+1)*b + 1, 1)])
		ans = ans[:-1] + '\n'
	ans = ans[:-1]
	print(ans)
