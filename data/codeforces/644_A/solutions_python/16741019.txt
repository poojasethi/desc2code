n, a, b = map(int, raw_input().split())

if a * b < n:
    print -1
    exit()

for i in xrange(a):
    for j in xrange(b):
        if i % 2 == 0:
            k = i * b + j + 1
            print k if k <= n else 0,
        else:
            k = i * b + (b - j)
            print k if k <= n else 0,
    print
