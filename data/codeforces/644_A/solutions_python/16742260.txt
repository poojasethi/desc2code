#!/usr/bin/python2
# -*- coding: utf-8 -*-

import sys
import os

def main():
    n, a, b = map(int, sys.stdin.readline().split())
    ret = [[0] * b for _ in xrange(a)]
    i = 0
    for r in xrange(a):
        it = xrange(b-1, -1, -1) if (r & 1) else xrange(b)
        for c in it:
            i += 1
            if i > n:
                break
            ret[r][c] = i

    if i >= n:
        for r in ret:
            print " ".join(map(str, r))
    else:
        print -1

if __name__ == '__main__':
    main()
