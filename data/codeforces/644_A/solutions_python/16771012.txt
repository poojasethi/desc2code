n, a, b = map(int, raw_input().split())

if  n / a > b or n / a == b and n % a > 0:
    print -1
    exit(0)

for i in range(1, a + 1):
    s = ''
    for j in range(1, b + 1):
        p = i * b - b + j
        if b % 2 == 0 and i % 2 == 0:
            p = i * b - j + 1
        if p > n:
            s = s + ' ' + str(0)
        else:
            s = s + ' ' + str(p)
    print s[1:]
