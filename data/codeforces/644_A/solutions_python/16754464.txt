#!/usr/bin/python
import sys
import re

n, a, b = [int(x) for x in raw_input().split()]

def next_id(n, a, b):
    while n > 0:
        yield n 
        n -= 1
    while True:
        yield 0

f = []
gen = next_id(n, a, b)
for i in xrange(a):
    f += [[]]
    for j in xrange(b):
        f[i] += [gen.next()]
    if b % 2 == 0 and i % 2 == 1:
        f[i] = f[i][::-1] # Reverse the i-th row.

if gen.next() == 0 or min(min(i for i in f)) == 0:
    for i in f:
        print " ".join(str(x) for x in i)
else:
    print -1
