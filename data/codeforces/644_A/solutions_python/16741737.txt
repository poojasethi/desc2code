import sys
n,a,b = map(int,sys.stdin.readline().split(' '))
if n>a*b:
  print -1
else:
  while n:
    i = b
    x = []
    while n and i:
      x += [n]
      i-=1
      n-=1
    while i:
      x += [0]
      i -= 1
    a -= 1
    if b&1==0 and a&1==0:
      x = x[::-1]
    print " ".join(map(str,x))
  while a:
    print " ".join(map(str,[0]*b))
    a -= 1
