n,a,b=map(int,raw_input().split())
if n>a*b:
  print -1
else:
  c=[[0]*105 for _ in range(105)]
  for i in range(a):
    for j in range(b):
      if i%2==1: c[i][b-j-1]=n
      else: c[i][j]=n
      n-=1
  for i in range(a):
    s=""
    for j in range(b):
      s+=str(max(0,c[i][j]))+(" " if (j!=b-1) else "")
    print s
