# Parliament of Berland.py

n,a,b = map(int,raw_input().split())
arr = [[0]*b for i in xrange(a)];
odd = 1;
even = 2;
for i in xrange(a):
	for j in xrange(b):
		if (i+j)%2 == 0 and odd <= n :
			arr[i][j] = odd;
			odd += 2;
		if (i+j)%2 == 1 and even <= n :
			arr[i][j] = even;
			even += 2;
if odd > n and even > n:
	for a in arr :
		print ' '.join(map(str,a));
else:
	print -1;
