import sys

n, a, b = map(int, sys.stdin.readline().split())

if n > a*b:
    print >>sys.stdout, -1
else:
    ind = [1, 2]
    d = 1
    r = 2
    for i in xrange(a):
        s = []
        for j in xrange(b):
            k = (j%2 + i%2)%2
            if ind[k] > n:
                s.append(0)
                continue
            s.append(ind[k])
            ind[k] += 2
                
        print >>sys.stdout, ' '.join(map(str, s))
