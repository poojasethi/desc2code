n,a,b = map(int,raw_input().split())
if n > a*b:
	print -1
else:
	for i in xrange(a):
		if n >= b:
			if i % 2 == 0:
				print " ".join(map(str,xrange(1+i*b,1+(i+1)*b)))
			else:
				print " ".join(map(str,xrange((i+1)*b,i*b,-1)))
			n -= b
		elif n > 0:
			if i % 2 == 0:
				print " ".join(map(str,range(1+i*b,1+i*b+n)+[0]*(b-n)))
			else:
				print " ".join(map(str,[0]*(b-n)+range(i*b+n,i*b,-1)))
			n = 0
		else:
			print " ".join(map(str,[0]*b))
