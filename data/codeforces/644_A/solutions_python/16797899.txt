N, a, b = map(int, raw_input().split())

ans = [ [0]*b for i in xrange(a)]
cur = [1, 2]

for i in xrange(a):
    p = i % 2
    for j in xrange(b):
        if cur[p] <= N:
            ans[i][j] = cur[p]
            cur[p] += 2
        p = 1 - p

if cur[0] <= N or cur[1] <= N:
    print -1
else:
    for x in ans:
        print ' '.join(map(str, x))
