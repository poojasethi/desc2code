n,a,b=list(map(int,raw_input().split()))
if n>a*b:
    print -1
    sys.exit()

#print x

        #print i,j
        #print a*i+j+1
        #print a*i+b-j
        if i%2==0 and b*i+j+1<=n:
            x[i][j]=b*i+j+1

            x[i][j]=b*i+b-j
        #else:
        #    x[i][j]=0
        #print i,j,x[i][j]
for i in range(a):
