s = raw_input()
s = [x for x in s.split(' ') if len(x)]
n = int(s[0])
a = int(s[1])
b = int(s[2])
if n>a*b :
	print -1
else:
	# do reverse or not
	rev = b % 2 == 0
	val = 1
	result = []
	for i in xrange(0,a):
		mid = []
		for j in xrange(0,b):
			num = val if val<=n else 0
			mid.append(num)
			val+=1
		if rev and i%2==0:
			mid = reversed(mid)
		result.append(mid)
	print '\n'.join(' '.join([str(x) for x in row]) for row in result)
