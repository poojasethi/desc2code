import sys


n, a, b = map(int, raw_input().split(' '))


if n > a * b:
    print '-1'
    sys.exit(0)

even = 0
odd = 1

for i in range(a):
    for j in range(b):
        if (i+j) % 2 == 0:
            if(even < n):
                print even+1,
            else:
                print 0,
            even += 2
        else:
            if(odd < n):
                print odd+1,
            else:
                print 0,
            odd += 2
    print;
