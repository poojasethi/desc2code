p, a, b = map(int, raw_input().split())
if a*b < p:
    print -1
else:
    cnt = 1
    room = [['0' for i in range(b)] for j in range(a)]
    for i in range(a):
        for j in range(b):
            if cnt < p + 1:
                if b % 2 != 0 or i % 2 == 0:
                    room[i][j] = str(cnt)
                else:
                    room[i][j + 1 if j % 2 == 0 else j - 1 ] = str(cnt)
                cnt += 1
    
    for line in room:      
        print ' '.join(line)
