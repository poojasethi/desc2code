n, r, c = map(int, raw_input().strip().split())

if n > r * c:
    print -1
    exit()

table = []
nums = range(1, n + 1)
if n < r * c:
    nums += [0 for _ in xrange(r * c - n)]

cr = 0
while cr < r:
    rev = c % 2 == 0 and cr % 2 == 1
    p = cr * c 
    if rev:
        table.append(nums[p:(p + c)][::-1])
    else:
        table.append(nums[p:(p + c)])
    cr += 1

for row in table:
    print " ".join(map(str, row))
