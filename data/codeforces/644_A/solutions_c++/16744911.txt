#include <bits/stdc++.h>

using namespace std;

int a[109][109];
int n , m , k ;
int x=1;

int main()
{
cin>>k>>n>>m;
for(int i=1;i<=n;i++){

 if( i&1 ){
    for(int j=1;j<=m;j++,x++){
        a[i][j]=(( x<=k )?x:0);
    }
 }else{
    for(int j=m;j>=1;j--,x++){
        a[i][j]=(( x<=k )?x:0);
 }
}
}
if( x<=k )cout<<-1<<endl;
else{

  for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
    cout<<a[i][j]<<((j<m)?" ":"\n");
}

    return 0;
}
