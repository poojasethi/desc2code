#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,a,b,k=1,i,j;
	cin>>n>>a>>b;
	int A[105][105]={0};
	if(a*b<n)
	{
		cout<<"-1";return 0;
	}
	for(i=0;i<a&&k<=n;i++)
	{
		if(i%2==0)
		{
		for(j=0;j<b&&k<=n;j++)
		{
			A[i][j]=k++;//k++;
		}
		    
		}
		else if(i%2==1)
		{
		for(j=b-1;j>=0&&k<=n;j--)
		{
			A[i][j]=k++;//k++;
		}
		    
		}
	}
	
	for(i=0;i<a;i++)
	{
		for(j=0;j<b;j++)
		{
			cout<<A[i][j]<<" ";
		}
		cout<<endl;
	}
	
	
	
}
