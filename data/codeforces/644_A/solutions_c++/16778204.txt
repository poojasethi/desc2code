#include <iostream>
#include <cstdio>

using namespace std;

int k, a, b, d[101][101];

int main(){
	cin >> k >> a >> b;
	if (a*b < k) cout << -1;
	else 
	{
		int i = 1;
		while (i <= a and k > 0) {
		    int j = 1; 
		    while (k > 0 and j <= b) d[i][j++] = k--;
		    i++;
		    while (k > 0 and j > 1) d[i][--j] = k--;
		    i++; 
		}
		for (int i=1; i<=a; i++)   {
		for (int j=1; j<=b; j++)                   
			cout << d[i][j] << ' ';  cout << endl;  }

	}		
}
