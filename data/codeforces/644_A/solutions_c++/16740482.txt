#include <iostream>
using namespace std;
int x,y,a,i,j,p[200][200];
int main(){
	cin>>a>>x>>y;
	if(x*y>=a){
        for(i=0;i<x;i++){
			for(j=0;j<y;j++){
				if(i%2==0)p[i][j]=a;
				else p[i][y-j-1]=a;
				if(a>0)a--;
			}
		}
        for(i=0;i<x;i++){
			for(j=0;j<y;j++){
				cout<<p[i][j]<<" ";
			}
			cout<<endl;
		}		
	}
	else cout<<-1;
    return 0;
}
