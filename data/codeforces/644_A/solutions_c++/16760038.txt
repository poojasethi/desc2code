#include<iostream>
using namespace std;
main()
{
      int n,a,b;
      cin>>n>>a>>b;
      
      if(a*b<n){cout<<-1;return 0;}
      
      
      int k;
      
      for(int i=0;i<a;i++){
      for(int j=0;j<b;j++)
      {
              if(i%2==0 || b%2!=0){k=i*b+j+1;if(k>n)k=0;cout<<k<<" ";}else 
                        {k=i*b+b-j-1+1;if(k>n)k=0;cout<<k<<" ";}   
      }
              cout<<endl;    
      }
      cin>>n;
}
