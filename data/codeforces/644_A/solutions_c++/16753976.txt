#include <iostream>
using namespace std;

int main()
{
    int a,b,n;
    cin >> n >> a >> b;

    if (a*b<n)
        cout << "-1\n";
    else
        for (int i=0;i<a;i++)
        {
            for (int j=0;j<b;j++)
            {
                int k=j;
                if (i%2==1)
                    k=b-j-1;
                int n1=i*b+k+1;
                cout << n1*(n1<=n) << " ";
            }

            cout << "\n";
        }

    return 0;
}
