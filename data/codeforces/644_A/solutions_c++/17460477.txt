#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n, a, b, i, k, x = 1, y = 2;
	scanf("%d %d %d", &n, &a, &b);
	if(n > a*b)
	{
		printf("-1");
		return 0;
	}
	for(i = 0; i < a; i++)
	{
		for(k = 0; k < b; k++)
		{
			if(k) printf(" ");
			if(!((i + k)&1) && x <= n)
			{
				printf("%d", x);
				x += 2;
			}
			else
				if(((i + k)&1) && y <= n)
				{
					printf("%d", y);
					y += 2;
				}
				else
					printf("0");				
		}
		printf("\n");
	}
	return 0;
}
