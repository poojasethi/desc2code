#include<cstdio>
int ans[111][111]={};
int main()
{
	int n,a,b;
	scanf("%d%d%d",&n,&a,&b);
	if(n>a*b){printf("-1\n");return 0;}
	int cnt=1;
	for(int i=0;i<a;i++)
	{
		for(int j=1&i;j<b;j=j+2)
			ans[i][j]=cnt,cnt+=2;
	}
	cnt=2;
	for(int i=0;i<a;i++)
		for(int j=!(1&i);j<b;j=j+2)
			ans[i][j]=cnt,cnt+=2;
	for(int i=0;i<a;i++)
	{
		for(int j=0;j<b;j++)
			if(ans[i][j]>n)
				printf("0 ");
			else printf("%d ",ans[i][j]);
		printf("\n");
	}	
}
