#include <iostream>

using namespace std;


int main() {
	int a,b,n,i,j,d[2]={1,2},L=0,K;
	cin>>n>>a>>b; if(n==1)d[1]=0;
	if(a*b<n)cout<<-1;
	else 
	{for(i=0;i<a;i++)
		{K=L;
			cout<<d[K]; if(d[K])d[K]+=2; if(d[K]>n)d[K]=0; K=!K;
			for(j=1;j<b;j++,K=!K)
				{
					cout<<' '<<d[K]; if(d[K])d[K]+=2; if(d[K]>n)d[K]=0;
				}
		 L=!L;
		 cout<<endl;
		}
		
	}
	return 0;
}
