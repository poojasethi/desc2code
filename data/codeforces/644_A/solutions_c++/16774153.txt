#include <bits/stdc++.h>
using namespace std;

int n,a,b;

int main() {
	cin>>n>>a>>b;
	if (a*b<n) {
		cout<<-1;
		return 0;
	}
	int d1=1,d2=1;
	for(int i=0;i<a;i++) {
		for(int j=0;j<b;j++) 
			if ((i+j)%2) 
				if (d2*2<=n) cout<<d2++*2<<" "; else cout<<"0 ";
			else 
				if (d1*2-1<=n) cout<<d1++*2-1<<" "; else cout<<"0 ";
		cout<<endl;
	}
	return 0;
}
