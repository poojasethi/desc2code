#include <bits/stdc++.h>
using namespace std;

int main() {
	int n, m, k;
	scanf("%d%d%d", &k, &n, &m);
	if(k > n*m) {
		printf("-1");
		return 0;
	}
	for(int i = 0; i < n; i++) {
		if(i%2 == 0)
			for(int j = i*m; j < (i+1)*m; j++) {
				if(j < k)
					printf("%d ", j+1);
				else
					printf("0 ");
			}
		else 
			for(int j = (i+1)*m - 1; j >= i*m; j--) {
				if(j < k)
					printf("%d ", j+1);
				else
					printf("0 ");
			}
		puts("");
	}
}
