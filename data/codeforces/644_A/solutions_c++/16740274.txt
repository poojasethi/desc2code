#include <bits/stdc++.h>
using namespace std;
int n,a,b,j,x=1,c[10000];
int main()
{
	scanf("%d%d%d",&n,&a,&b);
	if(a*b<n)
	{
		printf("-1");
		return 0;
	}
	for(int i=0;i<a;i++)
	{
		if(i%2==0)
		{
			for(j=0;j<b;j++)
			{
				if(x<=n)
				printf("%d ",x++);
				else 
				printf("0 ");
			}
		}
		else if(i%2==1)
		{
			for(j=b-1;j>=0;j--)
			{
				if(x<=n)
				c[j]=x++;
				else 
				c[j]=0;
			}
			for(j=0;j<b;j++)
			{
				printf("%d ",c[j]);
			}
		}
		printf("\n");
	}
}
