#include <iostream>
using namespace std;

int main()
{
	int a,b,c,p;
	int f=1;
	cin >> a >> b >> c;
	if(a<=(b*c))
	{
		for(int d=1;d<=b;d++)
		{
			f -= 1;
			if(d%2==0)
			{
			p = -1;
			f = f + c+1;
			}
			else
			p = 1;
			for(int e=1;e<=c;e++)
			{
				f += p;
				if(f>a)
				cout << 0;
				else
				cout << f;
				if(e<c)
				cout << " ";
			}
			if(d<b)
			cout << endl;
			if(d%2==0)
			f = f +c;
			else
			f = f + 1;
		}
	}
	else
	cout << -1;
}
