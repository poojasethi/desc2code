using namespace std;

int n, a, b;

int main(){
	cin >> n >> a >> b;
	if(n > a*b){
		puts("-1");
		return 0;
	}
	int g = 2, k = 1;
	rep(i,a){
		rep(j,b){
			if((i+j)%2==0){
				if(k > n) printf("0");
				else{
					printf("%d", k);
					k += 2;
				}
			} else{
				if(g > n) printf("0");
				else {
					printf("%d", g);
					g += 2;
				}
			}
			if(j != b-1) printf(" ");
		}
		cout << endl;
	}
}
