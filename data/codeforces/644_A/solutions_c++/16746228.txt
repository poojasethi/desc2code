#include<bits/stdc++.h>
using namespace std;
#define M 100
int t[M + 5][M + 5] , even = 2, odd = 1 , n , a , b;
int main(){
	scanf("%d %d %d",&n,&a,&b);
	if(n > a*b){
		printf("-1");
		return 0;
	}
	for(int i = 1 ; i <= a ; i++)
		for(int j = 1 ; j <= b ; j++)
			if((i + j)%2 == 0)	{	t[i][j] = odd ; odd += 2; }
			else {	t[i][j] = even ; even += 2 ; }
	for(int i = 1 ; i <= a ; i++){
		for(int j = 1 ; j <= b ; j++)
			printf("%d ",(t[i][j] <= n ? t[i][j] : 0));
		printf("\n");
	}
	
}
