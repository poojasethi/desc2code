#include<bits/stdc++.h>

int n, a, b, ans[105][105], l;

int main() {
	scanf("%d %d %d", &n, &a, &b);
	for (int i = 0; i < a && l < n; i++) {
		if (b % 2 || i % 2 == 0) for (int j = 0; j < b && l < n; j++) ans[i][j] = ++l;
		else for (int j = b - 1; j >= 0 && l < n; j--) ans[i][j] = ++l;
	}
	if (a * b >= n)  for (int i = 0; i < a; i++) for (int j = 0; j < b; j++) printf("%d%c", ans[i][j], j == b - 1 ? '\n' : ' ');
	else printf("-1\n");
	scanf("\n");
	return 0;
}
