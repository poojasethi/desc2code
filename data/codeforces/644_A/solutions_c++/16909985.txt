#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
    int N,a,b;
    cin>>N>>a>>b;
    if(N>a*b){cout<<"-1";}
    else
    {
    	for(int i=0;i<a;i++)
        {
    		for(int j=0;j<b;j++)
    		{
    			int x=b*i+j+1;
    			if(b%2==0 && i%2){x=b*i+b-j;}
    			if(x<0 || x>N){x=0;}
    			cout<<x<<' ';
    		}
    		cout<<endl;
    	}
    }
    return 0;
}
