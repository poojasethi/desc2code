#include<cstdio>
int n,a,b,t[105][105];
int main()
{
	scanf("%d%d%d",&n,&a,&b);
	if(n>a*b)return puts("-1"),0;
	for(int i=1;i<=n;i++)
	{
		for(int A=1;A<=a;A++)
		for(int B=1;B<=b;B++)
		if(((A+B)&1)!=(i&1)&&!t[A][B])
		{
			t[A][B]=i;
			goto nxb;
		}
		nxb:;
	}
	for(int i=1;i<=a;i++,puts(""))
	for(int j=1;j<=b;j++)
	printf("%d ",t[i][j]);
}
