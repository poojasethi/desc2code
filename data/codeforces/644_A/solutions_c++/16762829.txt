#include <stdio.h>

int main()
{
	int N, A, B;
	scanf("%d%d%d",&N, &A, &B);
	if( N > A * B ){ 
		printf("-1\n");
		goto done;
	}
	for(int i=1;i<=A;i++) {
		for(int j=1;j<=B;j++) {
			if( i % 2 == 1 ) {
				printf("%d ",(((i-1)*B+j) > N ? 0 : (i-1)*B+j));
			} else {
				printf("%d ",(i*B-j+1 > N ? 0 : i*B-j+1));
			}
		}
		printf("\n");
	}
done:;
}
