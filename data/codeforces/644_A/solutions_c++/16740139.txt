#include<bits/stdc++.h>
using namespace std;
int a[120][120];
int main()
{
	int p,n,m;
	scanf("%d%d%d",&p,&n,&m);
	if (p>n*m)
	{
		printf("-1\n");
		return 0;
	}
	int d1=1,d2=2;
	for (int i=1;i<=n;i++)
	for (int j=1;j<=m;j++)
	{
		if ((i+j)%2==0)
		{
			if (d1<=p) {a[i][j]=d1;d1+=2;}
		}
		else
		{
			if (d2<=p) {a[i][j]=d2;d2+=2;}
		}
	}
	for (int i=1;i<=n;i++)
	for (int j=1;j<=m;j++)
	{
		printf("%d ",a[i][j]);
		if (j==m) printf("\n");
	}
}
