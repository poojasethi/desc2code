#include<iostream>

using namespace std;

int main(){
	int m[200][200];
	int n,a,b,c=1,i=0,j=0;
	cin >> n >> a >> b;
	if(n > a*b){
		cout << -1;
		return 0;	
	}
	for(int i = 0;i<200;i++)
		for(int j=0;j<200;j++)
			m[i][j]=0;
	while(c<=n){
		m[i][j] = c++;
		if(i%2==0){
			j++;
			if(j>=b){
				j=b-1;
				i++;
			}
		}
		else{
			j--;
			if(j<0){
				j = 0;
				i++;
			}
		}
	}
	for(int i = 0; i < a ;i++){
		for(int j =0;j<b;j++)
			cout << m[i][j] << " ";
	cout << endl;
	}
			
}
