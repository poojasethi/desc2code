#include <bits/stdc++.h>
using namespace std;
int n, a, b, X = 1, Y = 2;
int ans[2222][2222];
int main(){
  scanf("%d %d %d", &n, &a, &b);
  if (n > a * b) return printf("-1"), 0;
  else {
    for (int i = 1; i <= a; i++){
      for (int j = 1; j <= b; j++)
	if (~(i + j) & 1) printf("%d ", X > n ? 0 : X), X += 2;
	else printf("%d ", Y > n ? 0 : Y), Y += 2;
      printf("\n");
    }
  }
  return 0;
}
