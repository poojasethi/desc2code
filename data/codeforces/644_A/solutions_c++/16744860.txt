#include <algorithm>
#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cmath>
using namespace std;
const int maxn=1e4+10;
int n,r,c,a[maxn]; 
int main()
{
	scanf("%d%d%d",&n,&r,&c);
	if(n>r*c)
	{
		printf("-1\n");
		return 0;
	}
	for(int i=1;i<=n;++i)a[i]=i;
	for(int i=n+1;i<=r*c;++i)a[i]=0;
	for(int i=1;i<=r;++i)
	{
		if(i%2==1 || c%2==1)for(int j=i*c-c+1;j<=i*c;++j)printf("%d ",a[j]);
		else for(int j=i*c;j>=i*c-c+1;--j)printf("%d ",a[j]);
		printf("\n");
	}
	return 0;
}
