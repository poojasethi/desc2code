#include<cstdio>
using namespace std;
int i,j,s,n,m,a=-1,b,f[105][105];
int main()
{
    scanf("%d%d%d",&s,&n,&m);
    if(s>n*m)return puts("-1"),0;
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
            f[i][j]=i+j&1?b+=2:a+=2;
    for(i=1;i<=n;i++,puts(""))
        for(j=1;j<=m;j++)
            printf("%d ",f[i][j]>s?0:f[i][j]);
}
