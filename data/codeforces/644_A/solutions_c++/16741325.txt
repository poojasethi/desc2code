#include<cstdio>
int main(){
	int n,a,b;scanf("%d%d%d",&n,&a,&b);
	if(n>a*b){printf("-1");return 0;}
	int o,e;
	if(n&1)o=n,e=n-1;
	else o=n-1,e=n;
	int i,j;
	for(i=1;i<=a;i++){
		for(j=1;j<=b;j++){
			if(j!=1)printf(" ");
			if((i+j)&1){
				printf("%d",e>0?e:0);
				e-=2;
			}
			else{
				printf("%d",o>0?o:0);
				o-=2;
			}
		}
		printf("\n");
	}
	return 0;
}
