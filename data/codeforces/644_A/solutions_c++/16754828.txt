#include <iostream>

using namespace std;

int main() {
	int n, a, b;
	cin >> n >> a >> b;
	if (n > a * b) {
		cout << -1;
		return 0;
	}
	for (int i = 0; i < a; i++) {
		for (int j = 0; j < b; j++) {
			int s = b * i + j + 1;
			if ((i % 2) == 1) {
				s = b * i + (b - j);
			}
			if (s > n) {
				s = 0;
			}
			cout << s << " ";
		}
		cout << endl;
	}

	return 0;
}
