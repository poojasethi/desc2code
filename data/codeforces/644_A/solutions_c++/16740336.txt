#include <iostream>
using namespace std;

int main() {
  int n, a, b;
  cin >> n >> a >> b;

  if (n > a * b) {
    cout << -1 << endl;
    return 0;
  }

  int p = 1;

  for (int i = 0; i < a; ++i) {
    for (int j = 0; j < b; ++j) {
      int p = i * b + j + 1;
      if (i % 2 == 1 && b % 2 == 0) {
        p = (i + 1) * b - j;
      }
      if (p > n) {
        p = 0;
      }
      cout << p;
      if (j < b) {
        cout << ' ';
      }
    }
    cout << endl;
  }
}
