#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
	int n,a,b; scanf("%d%d%d",&n,&a,&b);
	if(n>a*b){
		puts("-1");
		return 0;
	}
	int x=1,y=2;
	for(int i=1;i<=a;i++){
		for(int j=1;j<=b;j++){
			if((i+j+1)&1)
				if(x<=n)
					printf("%d ",x),x+=2;
				else
					printf("0 ");
			else
				if(y<=n)
					printf("%d ",y),y+=2;
				else
					printf("0 ");
		}
		printf("\n");
	}
}
