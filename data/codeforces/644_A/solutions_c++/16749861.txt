#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n, a, b;
int c[110];

int main()
{
	cin >> n >> a >> b;
	if (a*b < n) puts("-1");
	else
	{
		for (int i = 1; i <= a; i++)
		{
			for (int j = 1; j <= b; j++)
			{
				c[j] = n;
				n = max(n - 1, 0);
			}
			if (!(b & 1) && (i & 1)) reverse(c + 1, c + b + 1);
			for (int j = 1; j < b; j++) printf("%d ", c[j]);
			printf("%d\n", c[b]);
		}
	}
}
