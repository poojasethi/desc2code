#include <cstdio>

int main() {
    int n, a, b;
    scanf("%d%d%d", &n, &a, &b);
    if (n > a * b) {
        printf("-1\n");
        return 0;
    }
    for (int i = 0; i < a; i++) {
        int start = b * i;
        int dir = i % 2;
        for (int j = 0; j < b; j++) {
            int v = dir == 0 ? start + j : start + (b - 1) - j;
            if (v >= n)
                v = -1;
            printf("%d%c", v + 1, (j < (b - 1) ? ' ' : '\n'));
        }
    }
    return 0;
}
