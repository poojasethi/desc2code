#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int n,a,i,j,b;
	cin>>n>>a>>b;
	if((a*b)<n) {cout<<"-1\n"; return 0;}
	for(i=0;i<a;i++)
	{
		for(j=0;j<b;j++)
		{
			if(i%2==0||b%2!=0)
			{ if((b*i+j+1)>n) cout<<"0 "; else cout<<b*i+j+1<<" ";}
			else
			{ if((b*i+b-j)>n) cout<<"0 "; else	cout<<(b*i+b-j)<<" ";}
		}
		cout<<"\n";
	}
	return 0;
}
