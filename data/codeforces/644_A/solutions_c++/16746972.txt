#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;

int n,a,b,m;

int main() {
	scanf("%d%d%d",&n,&a,&b);
	if (n > a*b) {
		printf("-1\n");
		return 0;
	}
	m = n/2;
	n = n-m;
	for (int i=0; i<a; i++) {
		for (int j=0; j<b; j++) {
			if (j>0) printf(" ");
			if ((i+j)%2 == 0 && n>0) { printf("%d",n*2-1); n--; }
			else if ((i+j)%2 == 1 && m>0) { printf("%d",m*2); m--; }
			else printf("0");
		}
		puts("");
	}
    return 0;
}
