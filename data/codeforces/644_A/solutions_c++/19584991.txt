#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int a,b,n;
	cin>>n>>a>>b;
	int n2=n-1;
	if(a*b<n){
		cout<<-1;
		return 0;
	}
	for(int i=0;i<a;i++){
		for(int j=0;j<b;j++){
			if((i+j)%2==1 ){
				cout<<n2<<endl;
				n2=max(n2-2,0);
			}
			else{
				cout<<n<<endl;
				n=max(0,n-2);
			}
		}
	}
}
