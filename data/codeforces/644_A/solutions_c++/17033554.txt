#include <iostream>
using namespace std;

int main() {
	int n, a, b;
	cin >> n >> a >> b;

	if (a*b < n) {
		cout << -1 << endl;
		return 0;
	}

	for (int i = 0; i < a; i++) {
		for (int j = 0; j < b; j++) {
			int m = i*b + j + 1;
			if (b%2==0 && i%2==1) {
				m = (i+1)*b - j;
			}
			if (m > n)
				m = 0;
			
			cout << m << ' ';
		}
		cout << endl;
	}
}
