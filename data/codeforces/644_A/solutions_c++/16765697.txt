#include <iostream>

using namespace std;

int main()
{
	unsigned int n, a, b;
	cin>>n>>a>>b;
	if (a*b<n) cout<<"-1"<<endl;
	else {
		for (int i=0; i<a; i++) {
			for (int j=0; j<b; j++) {
				if (i%2) {
					if (i*b+(b-j-1)+1>n) cout<<"0 ";
					else cout<<i*b+(b-j-1)+1<<" ";
				} else {
					if (i*b+j+1>n) cout<<"0 ";
					else cout<<i*b+j+1<<" ";
				}
			}
			cout<<endl;
		}
	}
	return 0;
}
