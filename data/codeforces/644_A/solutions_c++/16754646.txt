#include <bits/stdc++.h>
using namespace std;

int n,a,b;
int f(int x) {
	return x>n?0:x;
}
int main() {
	cin >> n >> a >> b;
	if (a*b < n) {
		cout << "-1\n";
		return 0;
	}
	for (int i=0; i<a; i++) {
		if (i%2) {
			cout << f((i+1)*b);
			for (int j=(i+1)*b-1; j>i*b; j--)
				cout << ' ' <<f(j);
			cout << '\n';
		} else {
			cout << f(i*b+1);
			for (int j=i*b+2; j<=(i+1)*b; j++)
				cout << ' ' << f(j);
			cout << '\n';
		}
	}
	return 0;
}
