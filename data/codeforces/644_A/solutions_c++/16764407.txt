#include<bits/stdc++.h>
using namespace std;
const int maxn=105;
int ans[maxn][maxn];
int main()
{
	int n,a,b;
	scanf("%d%d%d",&n,&a,&b);
	if(n>a*b)puts("-1");
	else
	{
		int i,j,k=0;
		for(i=0;i<a;++i)
		{
			if(i&1)
				for(j=b-1;j>=0;--j)
					ans[i][j]=(k<n?++k:0);
			else
				for(j=0;j<b;++j)
					ans[i][j]=(k<n?++k:0);
		}
		for(i=0;i<a;++i)
		{
			for(j=0;j<b;++j)
				printf("%d ",ans[i][j]);
			printf("\n");
		}
	}
	return 0;
}
