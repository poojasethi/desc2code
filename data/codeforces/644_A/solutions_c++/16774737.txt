#include<stdio.h>
int a[105][105];
int main()
{
int x,m,n;
scanf("%d%d%d",&x,&n,&m);
if(x > m*n) return printf("-1")*0;
int i,j;
int di = 0;
for(i=1;i<=n;i++)
{
if(di==0)
{
for(j=1;j<=m;j++)
{
a[i][j] = x;
if(x)x--;
}

}
else {
for(j=m;j>=1;j--)
{
a[i][j] = x;
if(x)x--;
}
}
di = 1-di;
}
for(i=1;i<=n;i++)
{
for(j=1;j<=m;j++)
printf("%d ",a[i][j]);
printf("\n");
}
return 0;
}
