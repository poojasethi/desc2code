#include<bits/stdc++.h>
#define SIZE 40000
using namespace std;

int main(){
	int r,c,np,A[SIZE];
	cin>>np>>r>>c;
	
		for(int i =0; i< np; i++){
			A[i] = i+1;
		}
	
	if(!(c&1)){
		for(int i =0; i<np; i+=2)
			if((i/c)%2)swap(A[i],A[i+1]);
	}
	if(np >r*c){cout<<-1<<endl;return 0;}
	for(int i =1;i <= r*c; i++){
		if(i<=np+1)cout<< A[i-1]<<' ';
		else cout<<0<<' ';
		if(i%c==0)cout<<endl;
	}
	return 0;
}
