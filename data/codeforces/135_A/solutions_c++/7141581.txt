#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,m,a[100010];
	cin>>n;
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	m=0;
	for(i=0;i<n;i++)
	{
		if(a[i] > a[m])
			m=i;
	}
	if(a[m]==1)
		a[m]=2;
	else
		a[m]=1;
	sort(a,a+n);
	for(i=0;i<n;i++)
		printf("%d ",a[i]);
	cout<<endl;
	return 0;
}
