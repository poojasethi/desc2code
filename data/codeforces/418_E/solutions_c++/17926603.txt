#include<bits/stdc++.h>
#define MAXN 200009
#define INF 1000000007
#define LLINF 1000000000000000007
#define mp(x,y) make_pair(x,y)
#define wr cout<<"Continue Debugging!!!"<<endl;
#define ff first
#define ss second
#define lg(x) ceil(log2(x))
#define all(x) x.begin(),x.end()
#define pb(x) push_back(x)
#define ppb() pop_back()
#define tr(i, c) for(typeof((c).begin()) i = (c).begin(); i!=(c).end(); i++)
using namespace std;
typedef long long ll;
typedef pair<int,int> PII;
template<class T> bool umin(T& a, T b) { if(a > b){ a = b;return 1;}return 0;}
template<class T> bool umax(T& a, T b) { if(a < b){ a = b;return 1;}return 0;}
//freopen(".in", "r", stdin);
//freopen(".out", "w", stdout);
int arr[MAXN],type[MAXN],v[MAXN],p[MAXN];
int st[MAXN],en[MAXN],sq,ord[MAXN];
int cnt[320][MAXN];
struct BIT{
	int F[MAXN];
	int tap(int x){
		int sum=0;
		for(;x<MAXN;x+=x&(-x))
			sum+=F[x];
		return sum;	
	}
	void upd(int x,int val){
		for(;x>0;x-=x&(-x))
			F[x]+=val;	
	}
}T[320];
map<int,int>pm;
int water[MAXN];
int main(){
	//int x1,x2;
	//cin>>x1>>x2;
	//cout<<__gcd(x1,x2)<<endl;
	int a,c=0,num=1,q;
	scanf("%d",&a);
	for(int i=1;i<=a;i++)
		scanf("%d",arr+i),pm[arr[i]]=1;
	scanf("%d",&q);
	for(int i=1;i<=q;i++){
		scanf("%d%d%d",type+i,v+i,p+i);
		if(type[i]==1)
			pm[v[i]]=1;
	}
	tr(it,pm){
		water[++c]=it->ff;
		it->ss=c;
	}sq=sqrt(a);st[1]=1;
	vector<int>vv;
	for(int i=1;i<=a;i++){
		arr[i]=pm[arr[i]];ord[i]=num;
		vv.pb(arr[i]);
		if(i%sq==0 or i==a){
			en[num]=i;
			if(i<a)
				num++,st[num]=i+1;
		}
	}
	sort(all(vv));
	vv.erase(unique(all(vv)),vv.end());
	for(int i=1;i<=num;i++){
		for(int j=1;j<=en[i];j++)
			cnt[i][arr[j]]++;
		tr(it,vv)
			T[i].upd(cnt[i][*it],1);
	} 
	for(int i=1;i<=q;i++){
		int pl=ord[p[i]];
		if(type[i]==1){
			v[i]=pm[v[i]];
			for(int j=pl;j<=num;j++){
				T[j].upd(cnt[j][arr[p[i]]],-1);
				cnt[j][arr[p[i]]]--;
				T[j].upd(cnt[j][arr[p[i]]],1);
			}
			arr[p[i]]=v[i];
			for(int j=pl;j<=num;j++){
				T[j].upd(cnt[j][v[i]],-1);
				cnt[j][v[i]]++;	
				T[j].upd(cnt[j][v[i]],1);
			}
		}
		else{
			if(v[i]==1){
				printf("%d\n",water[arr[p[i]]]);
				continue;
			}
			int ans=cnt[pl-1][arr[p[i]]];
			for(int j=st[pl];j<=p[i];j++)
				if(arr[j]==arr[p[i]])
					ans++;	
			if(v[i]%2==0){
				printf("%d\n",ans);
				continue;
			}
			for(int j=p[i]+1;j<=en[pl];j++){
				T[pl].upd(cnt[pl][arr[j]],-1);
				cnt[pl][arr[j]]--;
				T[pl].upd(cnt[pl][arr[j]],+1);
			}
			printf("%d\n",T[pl].tap(ans));
			for(int j=p[i]+1;j<=en[pl];j++){
				T[pl].upd(cnt[pl][arr[j]],-1);
				cnt[pl][arr[j]]++;
				T[pl].upd(cnt[pl][arr[j]],+1);
			}
		}
	}
	return 0;
}
