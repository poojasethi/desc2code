n=int(raw_input())

a=list(map(int,raw_input().split()) )

transmissions=[]

for i in xrange(len(a)-1):
	m=min(a[i+1::])
	#print m
	if m<a[i]:
		ind=i+1+(a[i+1::]).index(m)
		transmissions.append((i,ind))
		temp=a[ind]
		a[ind]=a[i]
		a[i]=temp
	#print (a)

print(len(transmissions))

for t in transmissions:
	print (str(t[0])+" "+str(t[1]))
