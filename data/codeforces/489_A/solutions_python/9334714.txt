n=int(raw_input())
a=map(int,raw_input().split())
ans=[]
for i in range(n-1):
    s=a[i]
    ad=i
    for j in range(i+1,n):
        if s>a[j]:
            s=a[j]
            ad=j
    if s!=a[i]:
        a[i],a[ad]=a[ad],a[i]
        ans.append((i,ad))
if len(ans):
    print len(ans)
    for x in ans:
        print x[0],x[1]
else:
    print 0
