n=input()
a=map(int,raw_input().split())
swap=[]
for i in range(n):
    j=i
    for t in range(i+1,n):
        if a[j]>a[t]:
            j=t
    if i!=j:
        swap.append(i)
        swap.append(j)
        a[i],a[j]=a[j],a[i]
    
print len(swap)/2
for t in range(0,len(swap)-1,2):
    print swap[t],swap[t+1]
