n = int(raw_input())
a = map(int,raw_input().split())
cnt = 0
f1=[]
f2=[]
for i in range(0,n):
    tmp = a[i]
    t = i
    for j in range(i+1,n):
        if(a[j] < tmp):
            tmp = a[j]
            t = j
    if t > i:
        f1.append(i)
        f2.append(t)
        tt = a[i]
        a[i] = a[t]
        a[t] = tt
        cnt = cnt + 1

print cnt
for i in range(0,cnt):
    print f1[i],f2[i]
