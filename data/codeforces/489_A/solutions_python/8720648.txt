def solve(n, a):
    res = []
    for i in range(n):
        mina, minj = a[i], i
        for j in range(i+1, n):
            if a[j] < mina:
                mina, minj = a[j], j
        res.append((i, minj))
        a[i], a[minj] = a[minj], a[i]
    return res

n = int(raw_input())
a = map(int, raw_input().split())
res = solve(n, a)
print len(res)
for x, y in res:
    print x, y
