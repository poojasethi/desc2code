n = int(raw_input())
a = map(int, raw_input().split())
b = sorted(a)
ans = []
for i in xrange(n):
    if a[i] != b[i]:
        for j in xrange(i + 1, n):
            if a[j] == b[i]:
                break
        a[i], a[j] = a[j], a[i]
        ans.append((i, j))

print len(ans)
for e in ans:
    print e[0], e[1]
