n = int(raw_input())

thelist=[]

thelist=map(int,raw_input().split())
assert len(thelist) == n

answer=[]
nswap=0
for x in xrange(n-1):
	imin = x
	for y in xrange(x+1,n):
		if thelist[y]<thelist[imin]:
			imin=y
	if(imin!=x):
		answer.append(x)
		answer.append(imin)
		thelist[x],thelist[imin] = thelist[imin],thelist[x]
		nswap+=1

print nswap

x=0
while(x<len(answer)):
	print answer[x],answer[x+1]
	x+=2
