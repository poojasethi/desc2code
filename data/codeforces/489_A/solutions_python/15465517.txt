n=input()
l=[]
li=map(int,raw_input().split())
for i in xrange(n):
    m=li[i]
    p=i
    for j in xrange(i,n):
        if(li[j]<m):
            m=li[j]
            p=j
    if(li[i]!=li[p]):
        li[i],li[p]=li[p],li[i]
        l+=[(i,p)] 
print len(l)           
for i,j in l:
    print i,j
