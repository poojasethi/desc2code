n = int(raw_input())
a = map(int, raw_input().split())
ans = []
for st in xrange(n-1):
    minm = 1100000000
    minmj = st
    for j in xrange(st, n):
        if a[j] < minm:
            minm = a[j]
            minj = j
    if st != minj:
        a[st],a[minj] = a[minj], a[st]
        ans.append((st, minj))
print len(ans)
for e in ans:
    print e[0],e[1]
