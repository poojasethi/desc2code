n = input()

chisla = map(int, raw_input().strip().split())
chislas = sorted(chisla)

if chisla == chislas:
    print 0
    exit()
k = []
for i in range(n):
    if min(chisla[i:]) < chisla[i]:
        neww = min(chisla[i+1:])
        inew = chisla[i:].index(neww)+i
        old = chisla[i]
        chisla[i] = chisla[inew]
        chisla[inew] = old
        k.append([i, inew])

print len(k)
for i in k:
    print ' '.join(str(j) for j in i)
