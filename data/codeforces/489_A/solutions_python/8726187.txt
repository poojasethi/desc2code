
def run():
    n=input()
    nums=map(int,raw_input().split(" "))
    #insert sort
    res=[]
    for i in xrange(n):
        cm=nums[i]
        ci=i
        for j in xrange(i+1,n):
            if nums[j]<cm:
                cm=nums[j]
                ci=j
        res.append((i,ci))
        temp=nums[i]
        nums[i]=nums[ci]
        nums[ci]=temp
    #print nums
    print len(res)
    for each in res:
        print "%d %d"%each


run()
