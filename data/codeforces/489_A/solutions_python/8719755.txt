def readarray(f): return map(f, raw_input().split())
def readint(): return int(raw_input())
def printlist(l): print ' '.join(map(str, l))

n = readint()
A = readarray(int)

B = [(A[i], i) for i in range(n)]
B.sort()
Brev = range(n)
for i in range(n):
  Brev[B[i][1]] = i

C = range(n)
res = []
i = 0
while i < n:
  if C[i] == B[i][1]:
    i += 1
    continue
  j = Brev[C[i]]
  res.append((i, Brev[C[i]]))
  C[i], C[j] = C[j], C[i]

print len(res)
for i, j in res:
  print i, j
