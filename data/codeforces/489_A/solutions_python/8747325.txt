n=int(raw_input())
l=map(int,raw_input().split())
chk=cnt=0
ans=[]
if n==1:
    print 0
    exit()
while 1:
    if l[cnt]>min(l[cnt+1:]):
        p=l[cnt+1:].index(min(l[cnt+1:]))
        l[cnt],l[p+cnt+1]=l[p+cnt+1],l[cnt]
        ans.append((cnt,p+cnt+1))
        chk+=1
    cnt+=1
    if cnt+1>n-1:
        break
print chk
if chk>0:
    for i in ans:
        print i[0],i[1]
