n = int(input())
s = raw_input().split()
a = [int(i) for i in s]
swap = []
for i in range(len(a) - 1):
    mi = a[i]
    im = i
    for j in range(i + 1, len(a)):
        if a[j] < mi:
            mi = a[j]
            im = j
    if im != i:
        a[im] = a[i]
        a[i] = mi
        swap.append((i, im))
print len(swap)
for i in swap:
    print '%s %s'%i
