N=input()
Ans=[]
A=map(int, raw_input().split())
for i in xrange(N):
    minInd=i
    for j in xrange(i,N):
        if A[j]<A[minInd]:
            minInd=j
    if minInd==i:
        continue
    else:
        A[minInd],A[i] = A[i] , A[minInd]
        Ans.append([minInd, i])
print len(Ans)
for i in Ans:
    print i[0],i[1]
