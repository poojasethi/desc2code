def main():
    n = int(raw_input())
    a = map(int, raw_input().split())
    b = [(x, i) for i, x in enumerate(a)]
    c = b[:]
    c.sort()
    p = range(n)
    ans = []
    for i, x in enumerate(c):
        j = x[1]
        if p[j] != i:
            ans.append((i, p[j]))
            p[b[i][1]], b[i], b[p[j]] = p[j], b[p[j]], b[i]
    print len(ans)
    if ans:
        for i, j in ans:
            print i, j
main()
