n = input()
a = [int(i) for i in raw_input().split(' ')]
number = 0;
ansi = []
ansj = []
for i in range(0,n-1):
    k = i
    for j in range(i+1,n):
        if a[j] < a[k]:
            k = j;
    if i != k:
        number = number + 1
        ansi.append(i)
        ansj.append(k)
        t = a[i]
        a[i] = a[k]
        a[k] = t
print number
for i in range(0,number):
    print ansi[i], ansj[i]
