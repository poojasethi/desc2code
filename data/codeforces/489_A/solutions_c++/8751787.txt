#include <iostream>

using namespace std;

int a[3010];

int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    cout<<n<<endl;
    for(int i=0;i<n;i++)
    {
        int m=i;
        for(int j=i+1;j<n;j++)
            if(a[m]>a[j]) m=j;
        cout<<i<<" "<<m<<endl;
        swap(a[m],a[i]);
    }
    return 0;
}
