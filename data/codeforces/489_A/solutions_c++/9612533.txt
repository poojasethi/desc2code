#include<cstdio>
using namespace std;

int A[3001], k;

int  main(){
	scanf("%d", &k);
	for(int i=0; i<k; i++){
		scanf("%d", &A[i]);
	}
	printf("%d\n", k);
	for(int i=0; i<k; i++){
		int min = i;
		for(int j=i; j<k; j++){
			if(A[j]<A[min]){
				min = j;
			}
		}
		printf("%d %d\n", i, min);
		int temp;
		temp = A[i];
		A[i] = A[min];
		A[min] = temp; 
	}
	printf("\n");
	return 0;
}
