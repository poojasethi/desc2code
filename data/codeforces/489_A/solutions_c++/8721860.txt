#include<bits/stdc++.h>
using namespace std;
const int N = 3003;
int a[N],n;
int main()
{
    int i,j,k;
    cin>>n;
    for(i=0;i<n;i++)cin>>a[i];
    cout<<n<<endl;
    for(i=0;i<n;i++)
    {
        k=i;
        for(j=i;j<n;j++)if(a[k]>a[j])k=j;
        swap( a[i],a[k] );
        cout<<i<<" "<<k<<endl;
    }
    return 0;
}
