#include <iostream>
#include <algorithm>

int a[5000], n;

using namespace std;

int min_, pos;

int main(){
	cin >> n;
	for(int i = 0;i < n;++i)cin >> a[i];
	cout << n << "\n";
	for(int i = 0;i < n;++i){
		min_ = 1e9 + 2;
		for(int j = i;j < n;++j){
			if(a[j] < min_){
				min_ = a[j];
				pos = j;
			}
		}
		std::swap(a[i], a[pos]);
		cout << i << " " << pos << "\n";
	}
}
