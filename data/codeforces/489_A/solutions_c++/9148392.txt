#include<stdio.h>
#include<string.h>

int main(){
	int n,i,j,t;
	int a[30001],temp;

	scanf("%d", &n);
	for(i=0;i<n;i++){
		scanf("%d", &a[i]);
	}

	printf("%d\n", n);
	for(i=0;i<n;i++){
		t = i;
		for(j=i;j<n;j++){
			if(a[j] < a[t]){
				t = j;
			}
		}
		printf("%d %d\n", i, t);
		temp = a[i];
		a[i] = a[t];
		a[t] = temp;
		
	}

	return 0;
}
