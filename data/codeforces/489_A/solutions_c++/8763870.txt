#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,a[3005];
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;++i)scanf("%d",&a[i]);
	printf("%d\n",n);
	for(int i=0;i<n;++i){
		int k=i;
		for(int j=i+1;j<n;++j)
			if(a[j]<a[k])k=j;
		printf("%d %d\n",i,k);
		swap(a[i],a[k]);
	}
	return 0;
}
