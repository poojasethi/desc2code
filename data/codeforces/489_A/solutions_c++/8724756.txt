#include<bits/stdc++.h>
using namespace std;
int main()
{
int n;
cin>>n;
int arr[n],sor[n];
for (int i=0;i<n;i++)
{
	cin>>arr[i];
	sor[i]=arr[i];
}
sort(sor,sor+n);
cout<<n<<endl;
for (int i=0;i<n;i++)
{
	for (int j=i;j<n;j++)
		if (sor[i]==arr[j])
		{
			swap(arr[i],arr[j]);
			cout<<i<<" "<<j<<endl;
			break;
		}
}
return 0;
}
