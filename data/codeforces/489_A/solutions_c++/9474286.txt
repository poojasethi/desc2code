#include<cstdio>
int main()
{
	int n,a[3001];
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	printf("%d\n",n);
	for(int i=0;i<n;i++)
	{
		int cur=i;
		for(int j=i;j<n;j++)
		{
			if(a[cur]>a[j])
			{
				cur=j;
			}
		}
		printf("%d %d\n",i,cur);
		int tmp = a[i];
		a[i] = a[cur];
		a[cur] = tmp;
	}
	return 0;
}
