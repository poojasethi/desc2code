#include<bits/stdc++.h>
using namespace std;
int main()
{
int n;
cin>>n;
int a[n],b[n];
for (int i=0;i<n;i++)
{
	cin>>a[i];
	b[i]=a[i];
}
sort(b,b+n);
cout<<n<<endl;
for(int i=0;i<n;i++)
{
	for (int j=i;j<n;j++)
		if (b[i]==a[j])
		{
			swap(a[i],a[j]);
			cout<<i<<" "<<j<<endl;
			break;
		}
}
return 0;
}
