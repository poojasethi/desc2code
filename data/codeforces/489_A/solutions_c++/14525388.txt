#import<bits/stdc++.h>

using namespace std;

int n , a[3010] , cnt;

int main() {
	cin >> n;
	for (int i = 0; i < n; i++) 
		cin >> a[i];
	cout << n << endl;
	for (int i = 0; i < n; i++) {
		cnt = i;
		for (int j = i; j < n; j++)
			if (a[j] < a[cnt])
				cnt = j;
		cout << i << ' ' << cnt << endl;
		swap(a[i] , a[cnt]);
	}
	return 0;
}
