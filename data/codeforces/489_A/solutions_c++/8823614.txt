#include <iostream>
#include <algorithm>

using namespace std;

const int MAXN = 3010;

int a[MAXN];

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	cout << n << endl;
	for (int i = 0; i < n; i++)
	{
		int idx = min_element(a + i, a + n) - a;
		swap(a[idx], a[i]);
		cout << idx << " " << i << endl;
	}
}
