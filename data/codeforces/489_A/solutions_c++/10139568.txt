#include <bits/stdc++.h>

int n, mn, mj, a[3005];
int main() {
	int i, j;
	scanf("%d", &n);
	for (i = 0; i < n; i++) {
		scanf("%d", &a[i]);
	}
	printf("%d\n", n);
	for (i = 0; i < n; i++) {
		mn = 2e9;
		for (j = i; j < n; j++) {
			if (a[j] < mn) {
				mn = a[j];
				mj = j;
			}
		}
		printf("%d %d\n", i, mj);
		std::swap(a[i], a[mj]);
	}
}
