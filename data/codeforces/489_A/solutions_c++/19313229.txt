#include<cstdio>
#include<algorithm>
using namespace std;

int n,a[3030];

int main(){
	scanf("%d",&n);
	for(int i = 1;i <= n;i++)
		scanf("%d",&a[i]);
	printf("%d\n",n-1);
	for(int i = 1;i < n;i++)
	{
		int tmp=i;
		for(int j = i+1;j <= n;j++)
			if(a[j]<a[tmp])tmp=j;
		printf("%d %d\n",i-1,tmp-1);
		swap(a[i],a[tmp]);
	}
	return 0;
}
