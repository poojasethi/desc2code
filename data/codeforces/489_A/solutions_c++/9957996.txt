#include<bits/stdc++.h>
using namespace std;
int main(){
int n,arr[3000];
cin>>n;
for(int i=0;i<n;i++)
cin>>arr[i];
cout<<n<<endl;
for(int i=0;i<n;i++){
	int k=arr[i],l=i;
	for(int j=i+1;j<n;j++){
		if(arr[j]<k){
			k=arr[j];
			l=j;
		}
		
	} 
	arr[i]=arr[i]+arr[l]-(arr[l]=arr[i]);
	cout<<i<<" "<<l<<endl;
	
}

return 0;
}
