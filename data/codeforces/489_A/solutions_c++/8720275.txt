#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,i,j,k,a[3005],b[3005];
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		b[i]=a[i];
	}
	sort(a,a+n);
	cout<<n<<"\n";
	for(i=0;i<n;i++)
		for(j=i;j<n;j++)
			if(a[i]==b[j])
			{
				k=b[j];
				b[j]=b[i];
				b[i]=k;
				cout<<i<<" "<<j<<"\n";
				break;
			}
	return 0;
}
