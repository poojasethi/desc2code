#include<bits/stdc++.h>
using namespace std;
int a[3005],n,i,j,id;
int main()
{
  scanf("%d",&n);
  for (i=0;i<n;i++) {
    cin>>a[i];
  }
  cout<<n<<endl;
  for (i=0;i<n;i++)
  {
    id=i;
    for (j=i+1;j<n;j++) {
      if (a[j]<a[id]) {
       id=j;
      }
  }
       printf("%d %d\n",i,id);
      swap(a[id],a[i]);
  }
}
