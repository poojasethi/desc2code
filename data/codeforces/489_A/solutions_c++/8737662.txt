#include<iostream>
#include<algorithm>

using namespace std;

int nums[3003];

int main(){
  int n;
  cin >> n;
  for(int i = 0; i < n; i++){
    cin >> nums[i];
  }
  cout << n << endl;
  int *mx;
  for(int i = 0; i < n; i++){
    mx = max_element(nums, nums+n-i);
    swap(*mx, *(nums+n-1-i));
    cout << mx-nums << " " << n-i-1 << endl;
  }
}
