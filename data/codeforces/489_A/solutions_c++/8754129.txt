#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	int n;
	cin>>n;
	long long a[n];
		
	for(int i=0;i<n;cin>>a[i++]);
	cout<<n<<endl;
	for(int i=n-1;i>=0;i--)
	{
		int m=0;
		for(int j=0;j<=i;j++)
			if(a[m]<a[j]) m=j;
		cout<<i<<" "<<m<<endl;
		swap(a[i],a[m]);
	}
	return 0;
}
