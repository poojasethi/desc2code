#include <stdio.h>
const int maxn = 3000;
int a[maxn];
int main(){
	int n;
	scanf("%d",&n);
	for(int i = 0; i < n; i ++) scanf("%d",a+i);
	printf("%d\n",n);
	for(int i = 0; i < n; i ++){
		int id = i;
		for(int j = i; j < n; j ++) if(a[j] < a[id]) id = j;
		printf("%d %d\n", i, id);
		int t = a[id];
		a[id] = a[i];
		a[i] = t;
	}
	return 0;
}
