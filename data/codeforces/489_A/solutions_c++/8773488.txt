#include<cstdio>
#include<algorithm>
using namespace std;
int a[3005],n,i,j,id;
int main()
{
  scanf("%d",&n);
  for (i=1;i<=n;i++)
    scanf("%d",&a[i]);
  printf("%d\n",n);
  for (i=1;i<=n;i++)
  {
    id=i;
    for (j=i+1;j<=n;j++)
      if (a[j]<a[id]) id=j;
    printf("%d %d\n",i-1,id-1);
    swap(a[id],a[i]);
  }
}
