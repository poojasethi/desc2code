#include<iostream>
using namespace std;

const int oo = 3010;
int n, a[oo];

int main(){
  cin >> n;
  for(int i = 0; i < n; i++)
    cin >> a[i];
  cout << n << endl;
  for(int i = 0; i < n; i++){
    int m = i;
    for(int j = i; j < n; j++)
      if(a[m] > a[j])
	m = j;
    cout << i << ' ' << m << endl;
    swap(a[i], a[m]);
  }
  return 0;
}
