#include <bits/stdc++.h>
using namespace std;

int a[4000];

int main()
{
    int n;
    cin >> n;
    for (int i=0;i<n;i++) cin >> a[i];

    cout << n << endl;
    for (int i=0;i<n;i++)
    {
        int ind=i;
        for (int j=i;j<n;j++) if (a[j]<a[ind]) ind=j;
        cout << i << " " << ind << endl;
        swap(a[ind],a[i]);
    }
}
