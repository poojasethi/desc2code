#include <bits/stdc++.h>
using namespace std;
int a[5005];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)cin>>a[i];
	int ans=0;
	cout<<n<<endl;
	for(int i=0;i<n;i++)
	{
		int m = a[i];
		int ind = i;
		for(int j=i;j<n;j++)
			if(a[j]<m)m=a[j],ind=j;
		ans++;
		cout<<i<<" "<<ind<<endl;
		swap(a[i],a[ind]);
	}
	return 0;
}
