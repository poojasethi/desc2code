#include <cstdio>
long long n,a[501],f[501][501],mod=1e9+7;
int main(){
	//freopen("Progress Monitoring.in","r",stdin);
	//freopen("Progress Monitoring.out","w",stdout);
	scanf("%I64d",&n);
	for (int i=1;i<=n;i++) scanf("%I64d",&a[i]);
	for (int i=n;i>=1;i--){
		f[i][i]=1;
		for (int j=n;j>=i+1;j--){
			for (int k=i+1;k<=j;k++){
				if ((k==j) || a[i+1]<a[k+1]) f[i][j]=(f[i][j]+f[i+1][k]*f[k][j]%mod)%mod;
			}
		}
	}
	printf("%I64d",f[1][n]);
	//fclose(stdin); fclose(stdout);
}
