#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll mp = 1000000007;
ll t[501][501],st[501][501];
int a[501];
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	for(int i=1;i<=n;i++)
		t[i][i]=st[i][i]=1;
	for(int j=2;j<=n;j++)
	{
		for(int i=1;i+j-1<=n;i++)
		{
			t[i][i+j-1]=st[i+1][i+j-1];
			ll res=t[i][i+j-1];
			for(int k=i+1;k<=i+j-1;k++)
			{
				if(a[k]>a[i])res+=(t[i][k-1]*st[k][i+j-1]);
				res%=mp;
			}
			st[i][i+j-1]=res;
		}
	}
	cout<<t[1][n]<<endl;
}
