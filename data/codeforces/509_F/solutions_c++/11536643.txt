#include<bits/stdc++.h>

using namespace std;

#define ll long long

int A[505],n,DP[505][505];

int mod=1e9+7;

int f(int x,int y){

	if(x>=y)
		return 1;
	
	int &r=DP[x][y];

	if(r!=-1)
		return r;
	
	r=f(x+1,y);

	for(int i=x+1; i<=y ; i++)
		if(A[x]<A[i])
			r=(r+(ll)f(x+1,i-1)*f(i,y)) % mod;

	return r;
}

int main(){

	memset(DP,-1,sizeof DP);

	int i;

	cin>>n;

	for(i=1 ; i<=n ; i++)
		scanf("%d",A+i);
	
	cout<<f(2,n)<<endl;
	
	return 0;
}
