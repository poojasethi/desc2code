#include<cstdio>
const int MOD = 1e9 + 7;
long long n,a[501],d[1001][1001];
int main(){
    scanf("%I64d",&n);
    for(long long i=1;i<=n;i++)scanf("%I64d",&a[i]);
    for(long long i=1;i<=n;i++)d[i][i]=1ll,d[i+1][i]=1ll;
    for(long long l=1;l<n-1;l++){
        for(long long i=n;i>l;i--){
            if(i-l+1<=i)d[i-l][i]=(d[i-l][i]+d[i-l+1][i])%MOD;
            for(long long j=i-l+1;j<=i;j++)if(a[i-l]<a[j])d[i-l][i]=(d[i-l][i]+(d[i-l+1][j-1]*d[j][i])%MOD)%MOD;
        }
    }
    printf("%I64d",d[2][n]);
}
