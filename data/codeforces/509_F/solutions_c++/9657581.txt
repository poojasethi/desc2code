#include <bits/stdc++.h>

#define l(i,n) for(int i=0;i<n;i++)

#define mod 1000000007
#define LL long long
using namespace std;

int dp[520][520];
int a[520];


int rec(int l, int r) {
	if(l>=r-1) return 1;
	if(dp[l][r]!=-1) return dp[l][r];
	int &ret = dp[l][r];
	ret = rec(l + 1, r);
	for (int j = l + 1; j < r; j++) {
		if (a[l]<a[j]) {
			ret = ( (LL)rec(l+1,j)*rec(j,r) % mod +ret)%mod;
		}
	}
	return ret;
}


int main() {
std::ios::sync_with_stdio(false);cin.tie(0);cout.tie(0);

	int n;
	cin>>n;
	l(i,n)  cin>>a[i];

	l(i,n){
		l(j,n+1){
			dp[i][j]=-1;
		}
	}
	cout<<rec(1,n)<<endl;

	return 0;
}
