#include <cstdio>
int dp[510][510];
int a[510];
const int mod = 1000000007;
int go(int l, int r) {
	if (l >= r - 1) return 1;
	if (dp[l][r] != -1) return dp[l][r];
	int &res = dp[l][r];
	res = go(l + 1, r);
	for (int j = l + 1; j < r; j++) {
		if (a[l] < a[j]) {
			res = ((long long)go(l + 1, j) * go(j, r) % mod + res) % mod;
		}
	}
	return res;
}
int main() {
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
	}
	for (int i = 0; i < n; i++) {
		for (int j = 0; j <= n; j++) {
			dp[i][j] = -1;
		}
	}
	printf("%d\n", go(1, n));
	return 0;
}
