#include <iostream>
#include <vector>
#define P7 1000000007
#define N 502
using namespace std;
int a[N],dp[N][N],n,i,j,r,l;
main()
 {cin>>n;
  for (i=0;i<n;i++) cin>>a[i];
  for (i=0;i<=n;i++) dp[i][i]=dp[i][i+1]=1;
  for (j=2;j<=n;j++)
   for (l=0;l+j<=n;l++)
    {r=j+l;
     dp[l][r]=dp[l+1][r];
     for (i=l+1;i<r;i++)
      if (a[l]<a[i])
       dp[l][r]=(dp[l][r]+1LL*dp[l+1][i]*dp[i][r])%P7;
     }
 cout<<dp[1][n]<<endl;
}
