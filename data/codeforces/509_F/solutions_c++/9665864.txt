#include<iostream>
#include<cstdio>
using namespace std;

#define mod 1000000007ll
int a[510];
long long f[510][510];

int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=n;i>=0;i--)
	{
		for(int j=i;j<=n;j++)
		{
			if (j-i<=1)
				f[i][j]=1;
			else
			{
				for(int k=i+1;k<=j;k++)
					if (k==j || a[k]>a[i])
						f[i][j]=(f[i][j]+f[i+1][k]*f[k][j])%mod;
			}
		}
	}
	cout<<f[1][n]<<endl;
	return 0;
}
