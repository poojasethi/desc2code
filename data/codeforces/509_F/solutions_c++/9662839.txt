#include<stdio.h>
#include<string.h>
#define MOD 1000000007;

__int64 dp[505][505];
__int64 b[505]={0};

__int64 doit(int i,int j){
    if (dp[i][j]) return dp[i][j];
    if (i>=j) return 1;
    for (int k=i+1 ;k<=j ;k++){
        if (b[k]<b[i]) continue;
        dp[i][j]+=doit(i+1,k-1)*doit(k,j);
        dp[i][j]%=MOD;
    }
    dp[i][j]+=doit(i+1,j);
    dp[i][j]%=MOD;
    return dp[i][j];
}

int main(){
    int n;
    scanf("%d",&n);
    for (int i=1 ;i<=n ;i++) scanf("%d",&b[i]);
    memset(dp,0,sizeof(dp));    
    for (int i=1 ;i<=n ;i++) dp[i][i]=1;
    
    printf("%I64d\n",doit(2,n));
}
