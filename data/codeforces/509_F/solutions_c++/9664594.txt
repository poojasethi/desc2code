#include <bits/stdc++.h>
using namespace std;
const int MOD = (int)1e9 + 7;
const int maxn = 505;
long long dp[maxn][maxn];
int b[maxn];
int main() {
	int n;
	scanf("%d", &n);
	for(int i = 1; i <= n; i++) scanf("%d", &b[i]);
	for(int i = 1; i <= n + 1; i++) dp[i][i] = dp[i][i - 1] = 1;
	for(int i = n; i > 1; i--) {
		for(int j = i + 1; j <= n; j++) {
			dp[i][j] = dp[i + 1][j];
			for(int k = i + 1; k <= j; k++) {
				if(b[k] > b[i]) {
					dp[i][j] = (dp[i][j] + dp[i + 1][k - 1] * dp[k][j]) % MOD;
				}
			}
		}
	}
	printf("%I64d\n", dp[2][n]);
	return 0;
}
