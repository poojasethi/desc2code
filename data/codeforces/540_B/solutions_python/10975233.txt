n, k, p, x, y = map(int, raw_input().split())

takentests = [int(tok) for tok in raw_input().split()]

onleft = n / 2
onright = onleft + 1

possible = True

for mark in takentests:
    if mark >= y and onright:
        onright -= 1
    elif onleft:
        onleft -= 1
    else:
        possible = False
        break

newmarks = []
if possible:
    newmarks = ([1] * onleft) + ([y] * onright)

    if sum(newmarks) + sum(takentests) > x:
        possible = False

if not possible:
    print '-1'
else:
    print ' '.join(map(str, newmarks))
