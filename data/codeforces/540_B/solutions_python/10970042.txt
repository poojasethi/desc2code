
n, k, p, x, y = map(int, raw_input().split())

marks = map(int, raw_input().split())

count = 0
for mark in marks:
    if mark >= y:
        count+=1

ans = []
m = (n+1) / 2

while len(ans) < (n-k):
    if count < m:
        ans.append(y)
    else:
        ans.append(1)
    count+=1

if sum(marks) + sum(ans) <= x and count >= m:
    print " ".join(map(str, ans))
else: 
    print -1
