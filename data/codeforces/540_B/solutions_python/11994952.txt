n,k,p,x,y=map(int,raw_input().split())
l=map(int,raw_input().split())
total=sum(l)
large=small=0
for i in l:
    if i>=y:
        large+=1
    else:
        small+=1
ans=[]
cnt=n-k
while cnt:
    cnt-=1
    if small>=large:
        large+=1
        tmp=y
    else:
        small+=1
        tmp=1
    ans.append(tmp)
    total+=tmp
print ' '.join(map(str,ans)) if total<=x and large>small else -1
