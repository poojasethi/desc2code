n, k, p, X, y = map(int, raw_input().split())

l = sorted(map(int, raw_input().split()))
cnt0 = len([x for x in l if x < y])
ans = []
while cnt0 < n/2 and len(l) < n:
    l.append(1)
    ans.append(1)
    cnt0 += 1
while len(l) < n:
    l.append(y)
    ans.append(y)

l.sort()
if l[n/2] < y or sum(l) > X: print -1
else: print ' '.join([str(x) for x in ans])
