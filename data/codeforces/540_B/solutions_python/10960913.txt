f = lambda: map(int, raw_input().strip().split())
n, k, p, x, y = f()
a = f()
left, need = x - sum(a), (n - 1) / 2
lcnt = min(n - k, need - sum(i < y for i in a))
rcnt = min(n - k, (need + 1) - sum(i >= y for i in a))
b = []
for i in range(lcnt):
  b.append(1)
  left -= 1
for i in range(rcnt):
  b.append(y)
  left -= y
if left < 0 or sum(i >= y for i in (a + b)) <= n / 2:
  print -1
else:
  print " ".join([str(i) for i in b])
