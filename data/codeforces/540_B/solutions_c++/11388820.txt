#include <iostream>

using namespace std;

int n,k,p,x,y,a[1000],s=0,low=0,high=0;
int main()
{
    cin >> n >> k >> p >> x >> y;
    for (int i=1;i<=k;i++) {cin >> a[i];s+=a[i];}
    for (int i=1;i<=k;i++)
    if (a[i]<y) low++;else high++;
    if (low>(n-1)/2) {cout << -1;return 0;}
    if (high<(n+1)/2) {s=s+((n+1)/2-high)*y;k=k+(n+1)/2-high;}
    if (s+n-k>x) {cout << -1;return 0;}
    for (int i=1;i<=(n+1)/2-high;i++) cout << y << " ";
    for (int i=1;i<=n-k;i++) cout << 1 << " ";
}
