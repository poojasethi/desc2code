#include<cstdio>
#include<cstdlib>
int a[1001],n,K,P,X,Y;
void noans(){printf("-1\n");exit(0);}
int main()
{
	scanf("%d%d%d%d%d",&n,&K,&P,&Y,&X);
	int i,sum=0,num=0,x;
	for (i=1;i<=K;i++)
	{
		scanf("%d",&x);
		num+=(x<X);
		sum+=x;
	}
	if (X>P) noans();
	for (i=1;i<=n-K;i++)
	{
		if (num>=n/2) a[i]=X;
		else a[i]=1,num++;
	}
	if (num>n/2) noans();
	for (i=1;i<=n-K;i++) sum+=a[i];
	if (sum>Y) noans();
	for (i=1;i<=n-K;i++) printf("%d ",a[i]);
	return 0;
}
