#include <bits/stdc++.h>
using namespace std;
int n, k, p, x, y, d = 0, a[1000], s = 0;
int main(void)
{
    scanf("%d%d%d%d%d",&n,&k,&p,&x,&y);
    for (int i=0;i<k;++i) scanf("%d",a+i), d += a[i] >= y, s += a[i];
    d = n-(n+1)/2+1-d;
    for (int i=k;i<n;++i) {
        if (d > 0) a[i] = y, d--;
        else a[i] = 1;
        s += a[i];
    }
    if (d <= 0 && s <= x) for (int i=k;i<n;++i) printf("%d ",a[i]);
    else printf("-1");
    return 0;
}
