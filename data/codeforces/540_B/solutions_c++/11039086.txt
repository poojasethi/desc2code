#include<iostream>
#include<algorithm>
using namespace std;
int n,k,p,x,y,sum,i,diff,marks[1001],nos[1001];
int main()
{
	cin>>n>>k>>p>>x>>y;
	for(i=0;i<k;sum+=marks[i],i++)cin>>marks[i];
	for(i=k,diff=x-sum-n+i+1;i<n;sum+=marks[i],i++,diff=x-sum-n+i+1)
		if(diff>=y) marks[i]=nos[i-k]=y;
		else marks[i]=nos[i-k]=1;
	sort(marks,marks+n);
	if(sum>x||(marks[(n+1)/2-1]&&marks[(n+1)/2-1]<y))cout<<"-1";
	else for(i=0;i<n-k;i++)cout<<nos[i]<<" ";
}
