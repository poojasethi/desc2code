#include <bits/stdc++.h>
using namespace std;

int n, k, p, x, y, s, cnt, k1, ky;
int main() {
    int i, t;
    cin >> n >> k >> p >> x >> y;
    for (i = 0; i < k; i++) {
        scanf("%d", &t);
        s += t;
        if (t < y) cnt++;
    }
    k1 = min(n / 2 - cnt, n - k);
    ky = n - k - k1;
    if (cnt > n / 2 || s + k1 + ky * y > x)
     {
     printf("-1"); return 0;
     }
    while (k1--) 
    printf("1 ");
    while (ky--)
     printf("%d ", y);
}
