#include<bits/stdc++.h>
using namespace std;
long int a[10001];
int main() {
 long int n,k,p,x,y,i,j,s=0,l,r,m,v,rn,ln;
	cin>>n>>k>>p>>x>>y; l=0;r=n-1;
	m=r/2;
	for(i=0;i<k;i++)
	{ cin>>v;s+=v;
	if(v<y) a[l++]=v;
	else a[r--]=v;
		} ln=l;rn=r;
		while(r>=m&&r>=l)
		{ a[r--]=y;
			s+=y;
		}
		while(l<=r)
		{
			a[l++]=1;s+=1;
		}
		if(s>x||a[m]<y) {cout<<"-1"; return 0;}
		while(ln<=rn) cout<<a[ln++]<<" ";
	return 0;
}
