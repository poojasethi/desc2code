#include<bits/stdc++.h>
using namespace std;
int main(){
int i,j,k,l,m,n,a,b=0,p,x,y;
scanf("%d%d%d%d%d",&n,&k,&p,&x,&y);
for(i=0;i<k;i++)
{
    scanf("%d",&a);
    b+=a<y;
    x-=a;
}
if(x>0)
{
    l=n/2-b>=0?n/2-b:-1;
    if(l!=-1){
    j=n-k-l;
    if(y*j+l<=x){
        for(int i=0;i<j+l;i++){
			printf("%d%c",i<j?y:1,i+1<j+l?' ':'\n');
        }
    }
    else
     cout<<"-1"<<endl;
    }
    else
        cout<<"-1"<<endl;
}
else
  cout<<"-1"<<endl;
return 0;
}
