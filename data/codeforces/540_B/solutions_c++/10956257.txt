#include <bits/stdc++.h>
using namespace std;

int a[2000];

int main()
{
	int n,k,p,x,y,i;

	cin>>n>>k>>p>>x>>y;
	
	for(i=0;i<k;i++)cin>>a[i];
	
	int b = x;
	for(i=0;i<k;i++)b-=a[i];
	if(b< 1*(n-k)) {cout<<-1<<"\n";return 0;}
	
	int c=0;
	for(i=0;i<k;i++)if(a[i]>=y)c++;
	c = (n/2 +1)-c;
	if(c<0)c=0;
	
	if(c> n-k){cout<<-1<<"\n";return 0;}
	
	if(c*y + (n-k-c)*1 > b){cout<<-1<<"\n";return 0;}


	for(i=0;i<c;i++)cout<<y<<" ";
	for(i=0;i<n-k-c;i++)cout<<"1 ";
	cout<<"\n";
	

}
