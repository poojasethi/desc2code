#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
int n,k,p,x,y, t, r, tot;
int main() {
   scanf("%d %d %d %d %d", &n,&k,&p,&x,&y);
   for (int i=0; i<k; i++) {
      scanf("%d", &t);
      tot+=t;
      if (t<y) r++;
   }
   if (r>n/2) {
      printf("-1");
      return 0;
   }
   int m = min(n-k, n/2-r);
   if (m+tot+(n-k-m)*y>x) {
      printf("-1");
      return 0;
   }
   for (int i=0; i<m; i++) printf("1 ");
   for (int i=0; i<n-k-m; i++) printf("%d ", y);
   
}
