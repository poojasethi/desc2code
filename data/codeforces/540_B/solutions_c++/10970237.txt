#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n, k, p, x, y, sum = 0, temp, len = 0;
	cin >> n >> k >> p >> x >> y;
	for (int i = 0; i < k; i++){
		cin >> temp;
		sum += temp;
		len += (temp < y);
	}
	int a = min(n / 2 - len, n - k);
	int b = n - k - a;
	if (a < 0 || sum + a + b *y > x){
		cout << -1;
		return 0;
	}
	while (a--) cout << 1 << ' ';
	while (b--) cout << y << ' ';
	return 0;
}
