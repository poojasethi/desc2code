#include<stdio.h>
#include<string.h>
int main(){
	int n,k,p,mx,md;
	while(scanf("%d%d%d%d%d",&n,&k,&p,&mx,&md)==5){
		int sml=n/2,sum=0;
		for(int i=0;i<k;i++){
			int d;
			scanf("%d",&d);
			sum+=d;
			if(d<md)sml--;
		}
		n-=k;
		if(sml>n)sml=n;
		if(sml<0||sml+(n-sml)*md+sum>mx)printf("-1\n");
		else{
			for(int i=0;i<sml;i++)
				printf("%s1",i?" ":"");
			for(int i=0;i<n-sml;i++)
				printf("%s%d",i+sml?" ":"",md);
			printf("\n");
		}
	}

}
