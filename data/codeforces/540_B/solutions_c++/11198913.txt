#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
int n,k,p,x,y,i,o=0,g,chkx=0;
cin>>n>>k>>p>>x>>y;
for(i=1;i<=k;i++){
    int u;
cin>>u;
chkx=chkx+u;
if(u>=y){
o++;
}
}
if(o+n-k<((n+1)/2)){
cout<<"-1";
return 0;
}
if(o>=((n+1)/2)){
g=0;
}
else{g=((n+1)/2)-o;}
if(chkx+(g*y)+(n-k-g)>x){
cout<<"-1";
return 0;
}

int arr[n-k];
for(i=0;i<n-k;i++){
if(o<((n+1)/2)){
arr[i]=y;
o++;
}
else{
arr[i]=1;
}
cout<<arr[i];
if(i==n-k-1){
cout<<"\n";
}
else{
cout<<" ";
}
}

}
