#include <iostream>
#include <cstdio>
using namespace std;

int n, k, p, x, y, s, n1, ny, cnt;
int main()
{
	int i, t;
//	freopen("input.txt", "r", stdin);
	cin >> n >> k >> p >> x >> y;
	for (i = 0; i < k; i++) {
		scanf ("%d", &t);
		if (t < y) cnt++;
		s += t;
	}
	n1 = min(n / 2 - cnt, n - k);
	ny = n - k - n1;
	s += n1 + ny * y;
	if (s > x || cnt >= (n + 1) / 2) {
		cout << -1 << endl;
		return 0;
	}
	for (i = 0; i < n1; i++) {
		cout << 1 << " ";
	}
	for (i = 0; i < ny; i++) {
		cout << y << " ";
	}
    return 0;
}
