#include<iostream>
#include<cstring>
using namespace std;
int main()
{
	int n,k,p,x,y;
	int mid;
	while(cin>>n>>k>>p>>x>>y){
	    int tmp=0,a,sum=0;
	    for(int i=0;i<k;i++){
	    	cin>>a;
	    	tmp+=a;
	    	if(a<y) sum++;
	    }
	    if(sum>n/2) cout<<"-1"<<endl;
		else{
			int t1=min(n-k,n/2-sum);
			int t2=(n-k-t1)*y;
			if(t1+t2+tmp>x) cout<<"-1"<<endl;
			else{
				for(int i=0;i<t1;i++) cout<<"1 ";
				for(int i=0;i<t2/y;i++) cout<<y<<" ";
				cout<<endl;
			}
		} 
	}
}
