#include <bits/stdc++.h>

using namespace std;

int t,m,n,k,p,x,y,s,c,a[1001];

int main()
{ cin>>n>>k>>p>>x>>y;m=n-k;for(int i=0;i<k;i++){cin>>a[i];s+=a[i];if (a[i]>=y)c++;}
  if ((m>0 && s>x)||(m+s>x)) {cout<<-1; return 0;} 
  if (c>n/2) {for(int i=0;i<m;i++) cout<<"1 "; return 0;} 
  t=n/2-c+1;if (t*y+m-t+s>x||k-c>n/2){cout<<-1; return 0;}
  for(int i=0;i<m-t;i++) cout<<"1 ";
  for(int i=0;i<t;i++) cout<<y<<" ";
}
