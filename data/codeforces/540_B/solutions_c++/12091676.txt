#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int a[1010],x,y,n,k,p,num;
int main(){
	scanf("%d%d%d%d%d",&n,&k,&p,&x,&y);
	for (int i=1;i<=k;i++) scanf("%d",&a[i]),x-=a[i];
	for (int i=k+1;i<=n;i++) {a[i] = 1;x-=1;}
	if (x<0) return printf("-1"),0;
	for (int i=k+1;i<=n;i++){int s = min(x,y-1);x-=s;a[i]+=s;}
	num = 0;
	for (int i=1;i<=n;i++) if (a[i]>=y) num++;
	if (num<n/2+1) return printf("-1"),0;
	for (int i=k+1;i<=n;i++) printf("%d ",a[i]);
}
