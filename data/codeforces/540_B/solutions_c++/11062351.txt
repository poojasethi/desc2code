#include<iostream>
using namespace std;
int main()
{
	int n,k,p,x,y,l=0,h=0,lm,hm;
	cin>>n>>k>>p>>x>>y;
	int t,s=0;
	for(int i=1;i<=k;i++)
	{
		cin>>t;
		s+=t;
		if(t<y){l++;lm=max(lm,t);}
		if(t>=y){h++;hm=min(hm,t);}
	}
	if(s+n-k>x){cout<<-1;return 0;}
	 if(l>=(n+1)/2){cout<<-1;return 0;}
	if(h>=(n+1)/2){for(int i=1;i<=n-k;i++)cout<<1<<' ';return 0;}
	if((n/2-l)+(n-k-(n/2-l))*y+s>x){cout<<-1;return 0;}
	for(int i=1;i<=n-k-(n/2-l);i++)cout<<y<<' ';
	for(int i=1;i<=n/2-l;i++)cout<<1<<' ';
}
