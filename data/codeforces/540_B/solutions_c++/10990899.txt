#include <iostream>
#include<string>
#include<cmath>
using namespace std;

int main() {
	// your code goes here
	int n, k, p, x, y;
	int t, cnt = 0, s = 0;
	cin >> n >> k >> p >> x >> y;
	for(int i = 0; i < k; i++){
		cin >> t;
		if(t < y) cnt++;
		s += t;
	}
	int k1 = min(n -k, n/2 - cnt);
	int ky = n - k - k1;
	if(cnt > n/2 || s + k1 + ky*y > x){
		cout << "-1" << endl;
		return 0;
	} 
	while(k1--) cout << "1" << " ";
	while(ky--) cout << y << " ";
		
		
	return 0;
}
