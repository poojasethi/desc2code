#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
    int n,m,p,x,y;
    cin>>n>>m>>p>>x>>y;
    int a[m];
    for(int i=0;i<m;i++)cin>>a[i],x-=a[i];
    sort(a,a+m);
    int w=0;
    while(a[w]<y)w++;
    if(n/2-w+(n/2+1+w-m)*y<=x&&w*2<n&&x>0)
    {
        for(int i=0;i<n/2-w&&i<n-m;i++)cout<<1<<" ";
        for(int i=0;i<n/2+1+w-m&&i<n-m;i++)cout<<y<<" ";
    }
    else cout<<-1<<endl;
    return 0;
}
