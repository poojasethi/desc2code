#include<bits/stdc++.h>

using namespace std;

int n,k,p,x,y,a[1500],z,sum,k1;

int main()
{
    cin>>n>>k>>p>>x>>y;
    for(int i=0;i<k;++i)
    {
        cin>>a[i];
        z+=(a[i]<y);
        sum+=a[i];
    }
    if(z>n/2||sum+min(n-k,n/2-z)+max(0,(n-k-n/2+z))*y>x)return cout<<-1,0;
    for(int i=0;i<min(n-k,n/2-z);++i)
        cout<<"1 ";
    for(int i=0;i<n-k-n/2+z;++i)
        cout<<y<<' ';
}
