#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int n,m,x,y,a=0,b;
    scanf("%d%d%d%d%d",&n,&m,&b,&x,&y);
    for(int i=0;i<m;++i)
    {scanf("%d",&b);x-=b;a+=b<y;}
    if(a>n/2){puts("-1");return 0;}
    a=min(n-m,n/2-a);m=n-m-a;
    if(a+m*y>x){puts("-1");return 0;}
    for(int i=a+m-1;~i;--i)
    {
        printf("%d",i<a?1:y);
        putchar(i?' ':'\n');
    }
}
