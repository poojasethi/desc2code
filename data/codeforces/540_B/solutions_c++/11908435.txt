//In the Name of God

#include<iostream>

using namespace std;

int n,k,p,x,y,t,a[10000],s,d;
  
int main()
{
  cin>>n>>k>>p>>x>>y;
  for(int i=0;i<k;i++)
    {
      cin>>t;
      s+=t;
      if(t<y)
	d++;
    }
  for(int i=0;i<n-k;i++)
    {
      if(d<n/2)
	{
	  d++;
	  a[i]=1;
	}
      else
	a[i]=y;
      s+=a[i];
    }
  if(s>x || d>n/2)
    {
      cout<<-1<<endl;
      return 0;
    }
  for(int i=0;i<n-k;i++)
    cout<<a[i]<<" ";
  cout<<endl;
  //cin>>t;
  return 0;
}
