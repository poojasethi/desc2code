#include <stdio.h>
#include <algorithm>
using namespace std;

int main()
{
    int n, k, p, x, y;

    scanf("%d%d%d%d%d", &n, &k, &p, &x, &y);

    int sum = 0, low = 0;
    for(int i = 0, t; i < k; i++)
    {
        scanf("%d", &t);
        sum += t;
        if(t < y) low++;
    }

    int l = min(n / 2 - low , n - k);
    int r = n - k - l;
    if(l < 0 || sum + l + y * r > x)
        {printf("-1"); return 0;}

    while(l--) printf("1 ");
    while(r--) printf("%d ", y);

    return 0;
}
