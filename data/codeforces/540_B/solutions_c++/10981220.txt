#include<bits/stdc++.h>

using namespace std;

long ar,n,k,p,x,y,ttl,les,mid;

int main()
{
     cin>>n>>k>>p>>x>>y;  mid=(n/2);
     for(int i=1;i<=k;i++) {cin>>ar;ttl+=ar;if(ar<y) les++;}
     if(les>mid || (ttl+min(mid-les,n-k)+((n-k-min(mid-les,n-k))*y))>x)
     {
          cout<<"-1";return 0;
     }
     k++;les++;
     while(les<=mid && k<=n) {cout<<"1 ";k++;les++;}
     while(k<=n) {cout<<y<<" ";k++;}
     return 0;
}
