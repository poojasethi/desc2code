#include <bits/stdc++.h>
using namespace std;

vector<int> res;
int n,k,p,x,y,t,sum,cnt;

int main()
{
	cin>>n>>k>>p>>x>>y;
	for(int i=0;i<k;i++)
	{
		scanf("%d",&t),sum+=t;
		if(t<y) cnt++;
	}
	if(cnt>n/2)
		puts("-1");
	else
	{
		int c=min(n-k,n/2-cnt);
		for(int i=0;i<c;i++)
			res.push_back(1),sum++;
		for(int i=0;i<n-k-c;i++)
			res.push_back(y),sum+=y;
		if(sum>x)
			puts("-1");
		else
		{
			for(int i=0;i<res.size();i++)
				cout<<res[i]<<" ";
			cout<<endl;
		}
	}

	return 0;
}
