#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,k,p,x,y,i,cnt=0,sum=0,a[1009];
	cin>>n>>k>>p>>x>>y;
	for(i=0;i<k;i++)
	{
		cin>>a[i];
		if(a[i]>=y)
			cnt++;
		sum+=a[i];
	}
	for(i=0;i<n-k;i++)
	{
		if(cnt<(n+1)/2)
		{
			a[i+k]=y;
			cnt++;
		}
		else
			a[i+k]=1;
		sum+=a[i+k];
	}
	if(cnt<(n+1)/2)
	{
		cout<<"-1"<<endl;
	}
	else if(sum>x)
	{
		cout<<"-1"<<endl;
	}
	else
	{
		for(i=k;i<n;i++)
			cout<<a[i]<<" ";
		cout<<endl;
	}
	return 0;
}
