#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int n,k,p,x,y,s,cnt,k1,ky;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cin>>n>>k>>p>>x>>y;
    for(int i=0;i<k;i++)
    {
        int t;cin>>t;
        s+=t;
        if(t < y)cnt++;
    }
    //cout<<n/2-cnt<<" "<<n-k<<"\n";
    k1=min(n/2-cnt,n-k);
    ky=n-k-k1;
    if(cnt>n/2 || s+k1+ky*y>x){cout<<"-1\n";return 0;}
    while(k1--)cout<<"1 ";
    while(ky--)cout<<y<<" ";
    return 0;
}
