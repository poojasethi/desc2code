#include <cstdio>

int main(){
	int n,k,p,x,y;
	scanf("%d%d%d%d%d",&n,&k,&p,&x,&y);
	int a,b=0,c=0;
	for(int i=0;i<k;i++){
		scanf("%d",&a);
		b+=a<y;
		x-=a;
	}
	c=n-k<n/2-b?n-k:n/2-b;
	b=n-k-c;
	if(c>=0&&x>=b*y+c){
		for(int i=0;i<b+c;i++){
			printf("%d%c",i<b?y:1,i+1<b+c?' ':'\n');
		}
	}else{
		printf("-1\n");
	}
	return 0;
}
