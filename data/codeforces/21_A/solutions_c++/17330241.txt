#include <stdio.h>
#include <math.h>
#include <string.h>
#include <ctype.h>
#include <queue>
using namespace std;
bool isValid(const char& ch) {
    return isalnum(ch) || ch == '_';
}
int main() {
    char s[101];
    scanf("%s", s);
    int i = 0, cnt = 0;
    bool flag = false;
    while (s[i] && isValid(s[i])) ++cnt, ++i;
    if (cnt < 1 || cnt > 16 || !s[i] || s[i] != '@') {puts("NO");return 0;}
    int tol = 0;
    while (true) {
        if (!s[++i] || !isValid(s[i])) {flag = true; break;}
        cnt = 0;
        while (s[i] && isValid(s[i])) ++cnt, ++i, ++tol;
        if (cnt < 1 || cnt > 16) {flag = true; break;}
        if (!s[i]) break;
        if (s[i] == '.') {
            ++tol;
            continue;
        }
        if (s[i] == '/') break;
        flag = true; break;
    }
    if (flag || tol > 32) {puts("NO");return 0;}
    if (!s[i]) {puts("YES");return 0;}
    cnt = 0;
    while (s[++i] && isValid(s[i])) ++cnt;
    if (cnt < 1 || cnt > 16 || s[i]) puts("NO");
    else puts("YES");
    return 0;
}
