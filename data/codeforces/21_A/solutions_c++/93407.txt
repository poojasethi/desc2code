#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
char inp[200];
bool isit(int i,int j)
{
	if(j-i+1<1 || j-i+1>16) return false;
	for(int t=i;t<=j;t++){
		if(inp[t]=='_' || inp[t]>='0' && inp[t]<='9' || inp[t]>='a' && inp[t]<='z' || inp[t]>='A' && inp[t]<='Z'){}
		else return false;		
	 }
	return true;
}
bool dots(int i,int j)
{
	if(j-i+1>32 || j-i+1<0) return false;
	int pi=i;
	for(int t=i;t<=j;t++){
		if(inp[t]=='.'){
			if(!isit(pi,t-1)) return false;
			pi=t+1;
		}
	}
	if(isit(pi,j)) return true;
	return false;
}
int main()
{
	gets(inp);
	int len=strlen(inp);
	bool ok=false;
	for(int i=0;i<len;i++){
		if(inp[i]!='@')continue;
		//nsider for this only 
		if(i<1 || i>16)continue;
		if(!isit(0,i-1))continue;
		if(dots(i+1,len-1)){ok=true;break;}
		for(int j=i+1;j<len;j++){
			if(inp[j]!='/')continue;
			if(len-1-j<1 || len-1-j>16)continue;
			if(isit(j+1,len-1) && dots(i+1,j-1)) {ok=true;break;}
		}
		if(ok) break;
	}
		
	if(ok) puts("YES");
	else puts("NO");
	return 0;
}
