#include <cctype>
#include <cstdio>

char s[128];
char *p;

bool ok( char c ) {
   if( islower(c) ) return true;
   if( isupper(c) ) return true;
   if( isdigit(c) ) return true;
   return c == '_';
}

bool get_word() {
   int len = 0;
   while( ok( *p ) ) { ++p; ++len; }
   return len >= 1 && len <= 16;
}

bool check() {
   p = s;
   if( !get_word() ) return false;
   if( *p++ != '@' ) return false;
   char *xx = p;
   if( !get_word() ) return false;
   while( *p == '.' ) {
      ++p;
      if( !get_word() ) return false;
   }
   if( p - xx > 32 ) return false;
   if( *p == '/' ) {
      ++p;
      if( !get_word() ) return false;
   }
   return !*p;
}

int main( void ) {
   scanf( "%s", s );
   
   if( check() ) printf( "YES\n" );
   else printf( "NO\n" );

   return 0;
}
