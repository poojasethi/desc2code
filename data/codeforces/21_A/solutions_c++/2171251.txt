#include<stdio.h>
#include<string.h>
#include<string>
#include<math.h>
#include<iostream>
#include<algorithm>
using namespace std;
char c[105];
int judge()
{
	int i,j,k,a,b,n,aa=0,bb=0,s;
	n=strlen(c);b=n;
	for(i=0;i<n;i++)
	{
		if(c[i]=='@')
		{
			aa++;
			a=i;
		}
		else if(c[i]=='/')
		{
			bb++;
			b=i;
		}
		else if(c[i]=='_'||c[i]=='.'||
			('0'<=c[i]&&c[i]<='9')||
			('a'<=c[i]&&c[i]<='z')||
			('A'<=c[i]&&c[i]<='Z')) continue;
		else return 0;
	}	
	if(aa!=1||bb>1) return 0;
	if(a>b) return 0;
	if(a<1||a>16) return 0;
	if(b-a-1<1||b-a-1>32) return 0;
	if(bb==1&&(n-b-1<1||n-b-1>16)) return 0;
	for(i=0;i<a;i++)
	if(c[i]=='.') return 0;
	for(i=b+1;i<n;i++)
	if(c[i]=='.') return 0;
	s=0;
	for(i=a+1;i<b;i++)
	{
		if(c[i]=='.')
		{
			if(s<1||s>16) return 0;
			s=0;
		}
		else s++;
	}
	if(s<1||s>16) return 0;
	return 1;
}
int main()
{
	int i;
	while(~scanf("%s",c))
	{
		if(judge()) printf("YES\n");
		else printf("NO\n");
	} 
}
