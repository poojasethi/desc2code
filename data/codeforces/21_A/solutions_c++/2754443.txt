#include<stdio.h>
#include<ctype.h>
char S[1000],*d;
long i,j;
const char SS[2][5]={"NO","YES"};
inline bool check1(){
    d=S+1;
    for(i=1;i<=16;++i,++d){
        if(*d=='@') break;
        if(!isalpha(*d)&&!isdigit(*d)&&*d!='_') return 0;
        }
    if(*d!='@') return 0;
    return i!=1;
}
inline bool check2(){
    do{
        ++d;
        for(i=1;i<=16;++i,++d){
            if(*d=='.'||*d=='/'||*d=='\0') break;
            if(!isalpha(*d)&&!isdigit(*d)&&*d!='_') return 0;
            }
        if(i==1) return 0;
        j+=i;
        if(33<j) return 0;
        if(*d=='\0') return 1;
        if(*d!='.'&&*d!='/') return 0;
    }while(*d=='.');
    return 1;
}
inline bool check3(){
    if(!(*d)) return 1;
    ++d;
    for(i=1;i<=16;++i,++d){
        if(!(*d)) break;
        if(!isalpha(*d)&&!isdigit(*d)&&*d!='_') return 0;
        }
    if(i==1) return 0;
    return !(*d);
}
int main(){
    gets(S+1);
    puts(SS[check1()&&check2()&&check3()]);
    scanf("%s",S);
    return 0;
}
