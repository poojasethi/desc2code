#include <cstdio>
#include <cstring>
#include <cstdlib>

using namespace std;

char s[101];
int n;
int pos;

bool normal(char c) {
	return c >= 'a' && c <= 'z' || c >= 'A' && c <= 'Z' || c >= '0' && c <= '9' || c == '_';
}

void bad() {
	printf("NO\n");
	exit(0);
}

void checkword(int a, int b) {
	if(b - a < 1 || b - a > 16) bad();
	for(int i = a; i < b; i++) if(!normal(s[i])) bad();
}

void checkhost(int a, int b) {
	if(b - a < 1 || b - a > 32) bad();
	int prev = a;
	for(int i = a; i < b; i++) {
		if(s[i] == '.') {
			checkword(prev, i);
			prev = i + 1;
		}
	}
	checkword(prev, b);
}

int main() {
	gets(s);
	n = strlen(s);
	int i = 0;
	for(; i < n; i++) if(s[i] == '@') break;
	int dogpos = i;
	for(; i < n; i++) if(s[i] == '/') break;
	int slashpos = i;
	if(dogpos == n) bad();
	checkword(0, dogpos);
	checkhost(dogpos + 1, slashpos);
	if(slashpos != n) checkword(slashpos + 1, n);
	printf("YES\n");
	return 0;
}
