#include <iostream>
#include <string>
#include <cctype>
using namespace std;

string str;
int n,m,i,j,k,p,t,state,r,pp;

int main(){
  cin>>str;n=str.size();r=1;

  for(i=0;i<n;i++){
    if(isalnum(str[i])||str[i]=='_')continue;
    else if(str[i]=='@'&&state==0){
      if(i<=15&&i!=0){state=1;p=i;pp=i;continue;}
      else {r=0;break;}
    }
    else if(str[i]=='.'){
      if(i-pp>=34||i==pp+1||i-p>=18||i-p==1||state!=1){r=0;break;}
      p=i;continue;
    }
    else if(str[i]=='/'){
      if(state==1&&i-pp<=33&&i-pp>=2&&i-p>=2&&i-p<=17){p=i;state=2;}
      else {r=0;break;}
    }
    else {r=0;break;}
  }

  if(r&&(state==1||state==2)&&i!=p+1)cout<<"YES"<<endl;
  else cout<<"NO"<<endl;

  return 0;

}
