#include<cstdio>
using namespace std;
int be_end(bool ok){
	if(ok) printf("YES\n"); else printf("NO\n");
	return 0;
}
int main(){
	char c;int len=0;
	while((c=getchar())!='@'){
		len++;
		if(!(c>='A'&&c<='Z')&&!(c>='a'&&c<='z')&&!(c>='0'&&c<='9')&&!(c=='_')||len>16)
			return be_end(false);
	}
	if(len==0) return be_end(false);
	int ewl=0;len=0;
	while(c=getchar()){
		if(c=='/'||c=='\n') break;
		if(c=='.'){
			len++;
			if(ewl==0) return be_end(false);
			ewl=0;
			continue;
		}
		ewl++;len++;
		if(!(c>='A'&&c<='Z')&&!(c>='a'&&c<='z')&&!(c>='0'&&c<='9')&&!(c=='_')||ewl>16||len>32)
			return be_end(false);
	}
	if(ewl==0||len==0) return be_end(false);
	len=0;
	if(c=='\n') return be_end(true);
	while((c=getchar())!='\n'){
			if(c=='/') return be_end(false);
			len++;
			if(!(c>='A'&&c<='Z')&&!(c>='a'&&c<='z')&&!(c>='0'&&c<='9')&&!(c=='_')||len>16)
				return be_end(false);
		}
	if(len==0) return be_end(false);
	return be_end(true);
}
