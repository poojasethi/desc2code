#include <cstdio>
#include <algorithm>
#include <cstring>
#include <iostream>
#include <string>

std::string s;

bool ok(char c) {
	if (c == '_' || isalpha(c) || isdigit(c))
		return true;
	else 
		return false;
}
int main() {
	std::cin >> s;
	
	int i, j;
	for (i = 0; i < s.length() && s[i] != '@'; ++i) 
		if (not ok(s[i])) {
			puts("NO");
			return 0;
		}
	if (i < 1 || i > 16) {
		puts("NO");
		return 0;
	}
	
	for (j = i + 1; j < s.length() && s[j] != '/'; ++j) {
		if (s[j] != '.' && not ok(s[j])) {
			puts("NO");
			return 0;
		}
		if (j - 1 >= i + 1 && s[j] == '.' && s[j - 1] == '.') {
			puts("NO");
			return 0;
		}
	}
	
	if (j - i - 1 > 32 || j - i - 1 < 1) {
		puts("NO");
		return 0;
	}
	
	if (s[i + 1] == '.' || s[j - 1] == '.') {
		puts("NO");
		return 0;
	}
	
	if (j != s.length()) {
		for (i = j + 1; i < s.length(); ++i) 
			if (not ok(s[i])) {
				puts("NO");
				return 0;
			}
			
		if (i - j - 1 > 16 || i - j - 1 < 1) {
			puts("NO");
			return 0;
		}
	}
	
	puts("YES");
	return 0;
}
