using namespace std;

string s;

bool ok(char c){
  return isalpha(c) || isdigit(c) || c=='_';
}

void bad(){
  cout<<"NO"<<endl;
  exit(0);
}

int main() {

  cin>>s;

  int i=0;
  int l=s.length();

  while(s[i]!='@' && i<l){
    if(!ok(s[i]))
      bad();
    i++;
  }
  if(i==l || i>16 || i<1)
    bad();

  i++;
  int q=i;
  int p=i;

  while(s[i]!='/' && i<l){
    int d=i-p;
    if(s[i]=='.'){
      if(d<1 || d>16)
	bad();
      else{
	p=i+1;
      }
    }
    else if(!ok(s[i]))
      bad();
    i++;
  }

  int d=i-p;
  if(d<1 || d>16)
    bad();

  d=i-q;
  if(d<1 || d>32)
    bad();

  if(i==l){
    cout<<"YES"<<endl;
    return 0;
  }

  i++;
  p=i;

  while(i<l){
    if(!ok(s[i])) bad();
    i++;
  }

  d=i-p;
  if(d<1 || d>16) bad();

  cout<<"YES"<<endl;

}
