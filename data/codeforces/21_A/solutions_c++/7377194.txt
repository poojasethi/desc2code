#include<cstdio>
#include<cstring>
char s[110];
inline bool judge(char c){
	if(c>='0'&&c<='9'||c>='A'&&c<='Z'||c>='a'&&c<='z'||c=='_')return true;
	return false;
}

int main()
{
	scanf("%s",s);
	int i=0,n=strlen(s);
	while(judge(s[i]))++i;
	if(s[i]!='@'||!i||i>16){puts("NO");return 0;}
	if(!judge(s[++i])){puts("NO");return 0;}
	int cnt=0,cnt2=0;
	while(judge(s[i])||s[i]=='.'){
		++cnt;
		if(s[i]=='.'){
			if(!cnt2){puts("NO");return 0;}
			else cnt2=0;
		}
		else ++cnt2;
		if(cnt>32||cnt2>16){puts("NO");return 0;}
		++i;
	}
	if(s[i-1]=='.'||i!=n&&s[i]!='/'||i==n-1){puts("NO");return 0;}
	if(i!=n){
		cnt=0;++i;
		while(judge(s[i])){
			++cnt;++i;
			if(cnt>16){puts("NO");return 0;}
		}
		if(i!=n){puts("NO");return 0;}
	}
	puts("YES");
	return 0;
}
