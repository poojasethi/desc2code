#include <iostream>
#include <cstdio>
#include <cstring>
#include <cctype>
#include <algorithm>
using namespace std;
bool ok(char s[]) {
	for (; *s; ++s)
		if (!isalpha(*s) || !isdigit(*s) || *s != '_')
			return false;
	return true;
}
int main() {
	string s;
	int size = 0;
	getline(cin, s);
	char flag = '@';
	for (int i = 0; i < s.size(); ++i) {
		if (!isdigit(s[i]) && !isalpha(s[i]) && s[i] != '_') {
			if (size == 0 || size > 16) {
				printf("NO\n");
				return 0;
			}
			if (s[i] == flag) {
				if (flag == '@')
					flag = '.';
			} else if (flag == '.' && s[i] == '/')
				flag = '\0';
			else {
				printf("NO\n");
				return 0;
			}
			size = 0;
		} else
			
			size++;
	}
	if (size >= 1 && size <= 16 && (flag == '\0' || flag == '.'))
		printf("YES\n");
	else printf("NO\n");
}
