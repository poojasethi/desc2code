#include <stdio.h>
#include <string.h>
char str[500];
int judge(char c){
	if (c>='0'&&c<='9')return 1;
	if (c>='a'&&c<='z')return 1;
	if (c>='A'&&c<='Z')return 1;
	if (c=='_')return 1;
	return 0;
}
int main()
{
	int i;
	while (gets(str)){
		int a=0;
		int len=strlen(str);
		int cou=0,ture=1;
		for (i=0;i<len;i++){
			if (str[i]=='@'||str[i]=='/'||str[i]=='.'){
				if (cou>16||cou==0||i==len-1){ture=0;break;}
				cou=0;
				if (a==0&&str[i]!='@'){ture=0;break;}
				if(a==2&&str[i]!='/'){ture=0;break;}
				if (str[i]=='@'){a=1;}
				if (str[i]=='/'){a=2;}
			}
			else if (judge(str[i])){cou++;}
			else {
				ture=0;
				break;
			}
		}
		if (a&&ture){printf ("YES\n");}
		else printf ("NO\n");
	}
	return 0;
}
