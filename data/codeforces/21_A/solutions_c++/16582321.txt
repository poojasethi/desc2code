#include<iostream>

#define tw ((s[p] >= 'a' && s[p] <= 'z') || (s[p] >= 'A' && s[p] <= 'Z') || (s[p] >= '0' && s[p] <= '9') || s[p] == '_')

std::string s;


int main(int agrc, char*argv[]){
  std::getline(std::cin, s);
  int p = 0;
  bool wrong = false;
  while(tw && p!=s.length())
    p++;
  wrong |= (p == s.length() || p>16 || p==0 || s[p]!='@');
  int hp = p+1;
  while(p!=s.length() && s[p]!='/'){
    int wp = p+1;
    p++;
    while(tw && p!=s.length()) p++;
    wrong |= (p!=s.length() && s[p]!='.' && s[p]!='/') || p==wp || p-wp > 16;
  }
  wrong |= (hp==p || p-hp >32);
  if(p!=s.length() && s[p] == '/'){
    int rw = p+1; p++;
    while(p!=s.length() && tw)
      p++;
    wrong |= (p-rw>16 || p==rw || p!=s.length());
  }
  if(wrong){
    std::cout<<"NO";
  }else{
    std::cout<<"YES";
  }
  return 0;
}
