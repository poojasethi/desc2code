#define ll __int64
int n;
char s[200];
int is(char x)
{
    if(!isdigit(x)&&!isalpha(x)&&x!='_')return 0;
    return 1;
}
int ch()
{
    int x=0;
    while(x<n&&s[x]!='@')
    {
        if(!is(s[x]))return 0;
        ++x;
    }
    if(x>=16)return 0;
    x++;
    int y=x;
    int tlen=0;
    while(x<n&&s[x]!='/')
    {
        if(s[x]=='.')
        {
            if(tlen==0||tlen>16)return 0;
            tlen=0;
            x++;
            continue;
        }
        if(!is(s[x]))return 0;
        tlen++;
        x++;
    }
    if(tlen==0||tlen>16)return 0;
    y=x-y;
    if(y==0||y>32)return 0;
    if(x==n)return 1;
    x++;
    if(n-x>16||n-x==0)return 0;
    for(int i=x;i<n;i++)
    if(!is(s[i]))return 0;
    return 1;
}
int main()
{
    while(~scanf("%s",s))
    {
        n=strlen(s);
        puts(ch()?"YES":"NO");
    }
}
