#include <iostream>
#include <string>

using namespace std;


bool username(string w) {
	if(w.size()<1||w.size()>16) return false;
	for(int i = 0; i < w.size(); i++) {
		if(!isalnum(w[i]) && w[i]!='_') return false;
	}
	return true;
}

bool hostname(string s) {
	if(s.size()<1||s.size()>32||s[s.size()-1]=='.') return false;
	do {
//		cout << s.substr(0,s.find('.')) << endl;
		if(!username(s.substr(0,s.find('.')))) return false;
		if(s.find('.')==string::npos) s = ""; else s = s.substr(s.find('.')+1);
	} while(s.size());
	return true;
}

int main () {
	string s;
	cin >> s;
	
	string hn="",us="",rs="";
	bool cond = true;
	if(s.find('/')!=string::npos) {
		rs = s.substr(s.find('/')+1);
		s = s.substr(0,s.find('/'));
		cond = username(rs);
	}
	if(s.find('@')!=string::npos) {
		us = s.substr(0,s.find('@'));
		hn = s.substr(s.find('@')+1);
	}
//	cout << us << endl << hn << endl << rs << endl;
	if(username(us)&&hostname(hn)&&cond) {
		cout << "YES" << endl;
	} else {
		cout << "NO" << endl;
	}
	return 0;	
}
