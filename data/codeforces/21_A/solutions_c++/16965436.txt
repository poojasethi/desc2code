#include <iostream>
#include <string>
using namespace std;
string str;

bool check(string s,int mxlen){
	if(s.size()==0||s.size()>mxlen)return false;
	for(int i=0;i<s.size();i++){
		if(s[i]>='0'&&s[i]<='9'||s[i]>='A'&&s[i]<='Z'||s[i]>='a'&&s[i]<='z'||s[i]=='_')continue;
		return false;
	}
	return true;
}

bool check1(string s,int mxlen){
	if(s.size()==0||s.size()>mxlen)return false;
	if(s[0]=='.'||s[s.size()-1]=='.')return false;
	int p,q;
	p=0;q=s.find('.');
	while(q!=-1){
		if(!check(s.substr(p,q-p),16))return false;
		p=q+1;q=s.find('.',p);
	}
	q=s.size();
	if(!check(s.substr(p,q-p),16))return false;
	return true;
}

int main(){
	int p,q;
	cin>>str;
	p=str.find('@');
	if(p==-1){
		cout<<"NO"<<endl;
		return 0;
	}
	q=str.find('/');
	if(q==-1)q=str.size();
	if(!check(str.substr(0,p),16)||!check1(str.substr(p+1,q-p-1),32)||
		q!=str.size()&&!check(str.substr(q+1),16)){
		cout<<"NO"<<endl;
		return 0;
	}
	cout<<"YES"<<endl;
	return 0;
}
