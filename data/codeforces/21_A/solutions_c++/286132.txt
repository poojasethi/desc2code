#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <cmath>
#include <string.h>
#define sqr(x) ((x)*(x))
#define y1 y100500
#define pi 3.1415926535898
using namespace std;

char s[1000];
int n,i,j,k,l;

bool us(int k, int n){
    if(n-k<1 || n-k>16) return false;
    int i;
    for(i=k;i<n;++i)
    if((s[i]>='a' && s[i]<='z') || 
       (s[i]>='A' && s[i]<='Z') ||
       (s[i]>='0' && s[i]<='9') ||
       s[i]=='_'); else return false;
    return true;  
}

bool host(int k, int n){
    if(n-k<1 || n-k>32) return false;
    int t=k-1;
    bool r=true; 
    for(i=k;i<=n;++i)
    if(i==n || s[i]=='.'){
        r = r & us( t+1,i );
        t=i;
    }
    return r;       
}

bool isid(int k, int n){
    int i, j;
    i=0;
    while(i<n && s[i]!='@') ++i;
    if(i==n) return false;
    
    j=i+1;
    while(j<n && s[j]!='/') ++j;
    if(j==n) return  us(k,i) && host(i+1,n);
    else   return  us(k,i) && host(i+1,j) && us(j+1,n);         
}

int main(){
    cin>>s;
    n=strlen(s);
    
    cout<<(isid(0,n)==1 ? "YES":"NO");       

    return 0;
}
