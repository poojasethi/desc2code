#include <iostream>
#include <cctype>
using namespace std;

inline bool valid(char c)
{
	return isalpha(c) || isdigit(c) || c == '_';
}

int check(string& str)
{
	int len = str.length();
	int i = 0;
	int namelen = 0;
	while ( i < len && valid(str[i]) )
	{
		++i;
		++namelen;
	}
	if ( namelen<1 || namelen>16 ) return -1;
	if ( str[i++] != '@' ) return -2;

	int hostlen = 0;
	int wordlen = 0;
	while ( i<len && (valid(str[i]) || str[i]=='.') )
	{
		if ( str[i] == '.' )
		{
			if ( wordlen < 1 ) return -9;
			wordlen = 0;
			hostlen++;
		}
		else 
		{
			hostlen++;
			wordlen++;
		}
		if ( wordlen > 16 || hostlen > 32 ) return -3;
		++i;
	}
	if ( wordlen < 1 || wordlen > 16 || hostlen < 1 || hostlen > 32 ) return -4;
	if ( i == len ) return 1;

	if ( str[i++] != '/' ) return -5;

	int reslen = 0;
	while ( i < len && valid(str[i]) )
	{
		++i;
		++reslen;
	}
	if ( i == len && 1<=reslen && reslen<=16 ) return 1;
	return -7;
}

int main()
{
	string line;
	while ( cin >> line )
	{
		cout << (check(line)>0?"YES":"NO") << endl;
	}
	return 0;
}
