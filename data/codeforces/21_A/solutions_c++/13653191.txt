#include <stdio.h>
#define LET(c) ((c)>='a' && (c)<='z' || (c)>='A' && (c)<='Z' || \
                (c)>='0' && (c)<='9' || (c)=='_')

int Check(void);

int main(void){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);

  if (Check()) printf("YES");
  else printf("NO");

  return 0;
}

int Check(void){
  char c,part=0,len=0,lenh=0;
  while (scanf("%c",&c)>0 && c!=10){
    if (LET(c)){
      len++;
      if (part==1) lenh++;
    }
    else if (part==0 && c=='@' && len>0){
      len=0; part=1;
    }
    else if (part==1 && len>0){
      if (c=='.') len=0;
      else if (c=='/' && len>0 && lenh>0){
        len=0; part=2;
      }
      else return 0;
    }
    else return 0;

    if (len>16) return 0;
    if (lenh>32) return 0;
  }
  if (part>0 && len>0) return 1;
  else return 0;
}
