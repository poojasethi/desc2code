#include<iostream>
#include<string.h>
using namespace std;

int lenOfWord=0;
char a[110];
bool ans=true;

bool ok(char c)
{
	return ((c>='A' && c<='Z') || (c>='a' && c<='z') || (c>='0' && c<='9') || c=='_');
}

void refreshLenOfWord()
{
	if (lenOfWord>17 || lenOfWord<2)
		ans=false;
	lenOfWord=0;
}

int main()
{
	cin>>a;
	int step=0,placeOfAt=-1,placeOfBar=-1;
	bool b[2]={0,0};
	for (int i=0;i<strlen(a);i++)
	{
		lenOfWord++;
		if (a[i]=='@')
		{
			if (b[0])
				ans=false;
			b[0]=true;
			placeOfAt=i;
			refreshLenOfWord();
			step=1;
		}
		else if (a[i]=='/')
		{
			if (b[1] || (!b[0]))
				ans=false;
			b[1]=true;
			placeOfBar=i;
			refreshLenOfWord();
			step=2;
		}
		else if (step==1)
		{
			if (a[i]=='.')
				refreshLenOfWord();
			else if (!ok(a[i]))
				ans=false;
		}
		else if (!ok(a[i]))
			ans=false;
	}
	lenOfWord++;
	refreshLenOfWord();
	if (placeOfAt<0 || placeOfAt>16)
		ans=false;
	if (placeOfBar<0)
		placeOfBar=strlen(a);
	if (placeOfBar-placeOfAt>33)
		ans=false;
	if (ans==true)
		cout<<"YES\n";
	else
		cout<<"NO\n";
	return 0;
}
