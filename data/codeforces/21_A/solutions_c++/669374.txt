#include <iostream>
#define alfanum(a) ((a>='A' && a<='Z') || (a>='a' && a<='z') || (a>='0' && a<='9'))
using namespace std;
string host,reso,str,user;
void read()
{
     cin >> str;
     int i=0,l=str.size();
     for(; str[i]!='@' && i<l; i++)
           user+=str[i];
     for(i++; str[i]!='/' && i<l; i++)
     	   host+=str[i];
     for(i++; i<l; i++)
	 	   reso+=str[i];	
}
bool fuser()
{
	int l=user.size();
	if(!l) return 0;
	for(int i=0; i<l; i++)
		if(!(user[i]=='_' || alfanum(user[i])))
			return 0;
	return 1;
}
bool fhost()
{
	int l=host.size();
	if(!l) return 0;
	for(int i=0; i<l; i++)
	{
		if(!(host[i]=='.' || host[i]=='_' || alfanum(host[i])))
			return 0;
		if(host[i]=='.' && host[i+1]=='.') return 0;
	}
	return !(host[l-1]=='.' || host[0]=='.');
}
bool freso()
{
	int l=reso.size();
	for(int i=0; i<l; i++)
		if(!(reso[i]=='.' || reso[i]=='_' || alfanum(reso[i])))
			return 0;
	return str[str.size()-1]!='/';
}
int main()
{
	read();
    (fuser() && fhost() && freso())? cout << "YES" : cout << "NO"; 
    return 0;
}
