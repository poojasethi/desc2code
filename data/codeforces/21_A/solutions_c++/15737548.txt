#include <bits/stdc++.h>
using namespace std;

char in[1234];
char *p;

bool ck_me_plz(char me)
{
    if(isalpha(me) || isdigit(me) || me == '_') return true;
    return false;
}

bool ck_me_again()
{
    int x = 0;
    while(ck_me_plz(*p))
    {
        p++;
        x++;
    }
    if(x<1 || x>16) return false;
    return true;
}

bool ans_to_the_king()
{
    char me;
    if(!ck_me_again()) return false;
    me = *p;
    p++;
    if( me != '@') return false;
    int cc = 0;
    if(!ck_me_again()) return false;
    while( *p == '.')
    {
        p++;
        if(!ck_me_again()) return false;
        cc++;
    }
    if(cc>32) return false;
    if(*p == '/')
    {
        p++;
        if(!ck_me_again()) return false;
    }
    if(*p == 0) return true;
    else return false;
}

int main()
{
    scanf("%s",&in);
    p = in;
    if(ans_to_the_king()) cout << "YES";
    else cout << "NO";
    return 0;
}
