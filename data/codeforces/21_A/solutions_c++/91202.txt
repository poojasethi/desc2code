#include<iostream>
#include<string>

using namespace std;

inline bool checkword(string s)
{
	if(s.size()<1||s.size()>16)
		return false;
	for(int i=0;i<(int)s.size();i++)
		if(!((s[i]<='Z'&&s[i]>='A')||(s[i]<='z'&&s[i]>='a')||(s[i]<='9'&&s[i]>='0')||s[i]=='_'))
			return false;
	return true;
}

int main(void)
{
	string s,s1="",s2="",s3="";;
	cin>>s;
	
	for(int i=0;i<(int)s.size();i++)
		if(s[i]=='@')
		{
			if(!checkword(s1))
			{
				cout<<"NO"<<endl;
				return 0;
			}
			for(int j=i+1;j<(int)s.size();j++)
				if(s[j]=='/')
				{
					for(int k=j+1;k<(int)s.size();k++)
						s3+=s[k];
					if(!checkword(s3))
					{
						cout<<"NO"<<endl;
						return 0;
					}
					else
						break;
				}
				else
					s2+=s[j];
			if(s2.size()>32)
			{
				cout<<"NO"<<endl;
				return 0;
			}
			string t="";
			for(int j=0;j<(int)s2.size();j++)
				if(s2[j]=='.')
				{
					if(!checkword(t))
					{
						cout<<"NO"<<endl;
						return 0;
					}
					t="";
				}
				else
					t+=s2[j];
			if(!checkword(t))
				cout<<"NO"<<endl;
			else
				cout<<"YES"<<endl;
			return 0;
		}
		else
			s1+=s[i];
		
	cout<<"NO"<<endl;
	
	return 0;
}
