import re

def main():
    r = re.compile(r'^\w{1,16}@(\w{1,16}(\.\w{1,16})*)(\/\w{1,16})?$')
    m = r.match(raw_input())
    print 'YES' if m and len(m.group(1)) <= 32 else 'NO'


if __name__ == '__main__':
    main()
