#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
ifs = sys.stdin
ofs = sys.stdout

import re


def solve(s):
    jabber_id = r'^\w{1,16}@\w{1,16}((\.\w{1,16})*)(/\w{1,16})?$'
    r = re.match(jabber_id, s)
    if r:
        #print 'regular expression matched!'
        beg = s.find('@')+1
        end = s.find('/')
        #print 'positions:', beg, end
        if end<0:
            end = len(s)
        #print 'positions:', beg, end, 'e-b=', end-beg
        #print 'fragment:', s[beg:end]
        if end-beg <= 32:
            return 'YES'
    return 'NO'


s = ifs.readline().strip()

a = solve(s)

ofs.write('%s\n' % a)
