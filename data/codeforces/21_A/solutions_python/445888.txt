#!/usr/bin/env python
# -*- coding: utf-8 -*-

import re

def main():
    s = raw_input()
    m = re.match("[a-zA-Z0-9_]{1,16}@[a-zA-Z0-9_.]{1,32}([/][a-zA-Z0-9_]{1,16})?", s)
    if m is not None and m.group() == s:
        l = re.findall("[a-zA-Z0-9_]{1,16}@([a-zA-Z0-9_.]{1,32})([/][a-zA-Z0-9_]{1,16})?", s)[0][0].split('.')
        l = set([1 <= len(e) <= 16 for e in l])
        if len(l) == 1:
            print "YES"
            return 0
    print "NO"
    return 0

if __name__ == '__main__':
    main()
