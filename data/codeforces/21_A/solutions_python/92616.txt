import re
str = raw_input()

if re.match("[a-zA-Z0-9_]{1,16}@([a-zA-Z0-9_]{1,16}\.)*([a-zA-Z0-9_]{1,16})(/[a-zA-Z0-9_]{1,16})?$", str) == None or str.find("/") == len(str)-1:
    print "NO"
else:
    if str.find("/") == -1: 
        sz = len(str) - str.find("@")
    else:
        sz = len(str[str.find("@"):str.find("/")]) - 1
    
    if sz <= 32 and sz>0:
        print "YES"
    else:
        print "NO"
