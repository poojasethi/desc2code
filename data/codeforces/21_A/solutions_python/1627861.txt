import re

def can(s):
    t=re.match(r"[\w_]{1,16}@[\w_]{1,16}(\.[\w_]{1,16})*(\/[\w_]{1,16})?",s)
    return t and t.group(0)==s

def solve(s):
    if not can(s): return False
    a=s.find('@')
    b=s.find('/')
    if b==-1: b=100000000
    s0=s[a+1:b]
    return 1<=len(s0)<=32

print 'YES' if solve(raw_input()) else 'NO'
