import re
import sys

def main():
  s = sys.stdin.readline()
  regexp = r'^[A-Za-z0-9_]{1,16}@([A-Za-z0-9_.]{1,32})(/[A-Za-z0-9_]{1,16})?$'
  matches = re.findall(regexp, s)
  if not matches: print 'NO'
  else:
    hostname = matches[0][0]
    regexp = r'^[A-Za-z0-9_]{1,16}(.[A-Za-z0-9_]{1,16})*$'
    match = re.match(regexp, hostname)
    print 'YES' if match else 'NO'

if __name__ == "__main__":
  main()
