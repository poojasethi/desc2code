#!/usr/bin/env python

import re, sys

s = raw_input()

for ch in s:
    if ord(ch) == 127:
        print "NO"
        sys.exit()


m =re.match('^[a-zA-Z0-9_]{1,16}@([a-zA-Z0-9_]{1,16}(\.[a-zA-Z0-9_]{1,16})*)(\/[a-zA-Z0-9_]{1,16})?$' ,s)
res = "NO"
if m and 1 <= len(m.group(1)[0]) <= 32:
    res = "YES"
print res
