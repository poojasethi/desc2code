#coding: utf-8
import re
def main():
    inp = raw_input()
    g = re.match("[a-zA-Z0-9_]{1,16}@([a-zA-Z0-9_]{1,16}(\.[a-zA-Z0-9_]{1,16})*)(/[a-zA-Z0-9_]{1,16})?$", inp)
    if g == None:
        print "NO"
        return 0
    if len(g.groups()[0]) > 32:
        print "NO"
        return 0
    print "YES"
    return 0
main()
