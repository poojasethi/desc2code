#!/usr/bin/python
import re

def is_jabber_id(s):
	str_l = s.split('@')
	if len(str_l) != 2:
		return False

	username = str_l[0]

	if (None == re.match('^[A-Za-z0-9_]{1,16}$', username)):
		return False

	h_r = str_l[1].split('/')

	if len(h_r) > 2:
		return False

	if len(h_r) == 2:
		resource = h_r[1]
		if (None == re.match('^[A-Za-z0-9_]{1,16}$', resource)):
			return False

	hostname = h_r[0]
	if len(hostname) < 1 or len(hostname) > 32:
		return False
	
	words = hostname.split('.')
	for word in words:
		if (None == re.match('^[A-Za-z0-9_]{1,16}$', word)):
			return False

	return True
	

s = raw_input()

if is_jabber_id(s):
	print "YES"
else:
	print "NO"
