re_str = re.compile(r'^(?P<username>[a-z0-9_]{1,16})@(?P<hostname>[a-z0-9_.]{1,32})(?P<content>(/[a-z0-9_.]{1,16}){0,1})$',re.I)
def main():
    m = re_str.search(raw_input())
    try:
        re_dict = m.groupdict()
    except AttributeError:
            print 'NO'
            return
    for name in re_dict['hostname'].split('.'):
        if not (0 < len(name) < 17):
            print 'NO'
            return
    print 'YES'
if __name__ == "__main__":
	main()
