#!/usr/bin/python
import sys
import re
import string

pattern = re.compile("^[a-zA-Z0-9_]*$")

def checkValidity(s):
	if len(s) < 1 or len(s) > 16 or not pattern.match(s):
		print "NO"
		sys.exit(0)

s = raw_input()
i = string.find(s, '@')
if i == -1:
	print "NO"
	sys.exit(0)

username = s[0:i]
s = s[i+1:]
checkValidity(username)

i = string.find(s, '/')
hasResource = True if i >= 0 else False
if hasResource:
	hostname = s[0:i]
	s = s[i+1:]
else:
	hostname = s[:]
	s = ""
if len(hostname) < 1 or len(hostname) > 32:
	print "NO"
	sys.exit(0)
while len(hostname) != 0:
	i = string.find(hostname, '.')
	if i == len(hostname) - 1:
		print "NO"
		sys.exit(0)
	if i >= 0:
		word = hostname[0:i]
		hostname = hostname[i+1:]
	else:
		word = hostname[:]
		hostname = ""
	checkValidity(word)

if hasResource:
	checkValidity(s)

print "YES"
