
#include<iostream>
#include<algorithm>
#include<string>
#include<cstring>
#include<sstream>
#include<vector>

#define rep(i, n) for (int (i) = 0; (i) < (n); (i) ++)
#define rep1(i, n) for (int (i) = 1; (i) <= (n); (i) ++)

using namespace std;
int mat[1005][1005], N, M, x, y, cnt[1005];
stringstream ss;
int main()
{
	ios_base::sync_with_stdio(0);
	cin >> N;
	rep1(i, N) cin >> x >> y, mat[x][y]=1;

	while (N > 1)
	{
		memset(cnt, 0, sizeof(cnt));

		rep1(i, N) rep1(j, N) cnt[j] += mat[i][j];

		if (cnt[N] >  0) rep1(i, N) if (cnt[i] == 0)
		{
			// swap i-th column and N-th column
			rep1(r, N) swap(mat[r][i], mat[r][N]);
			ss << 2 << " " << i << " " << N << endl;
			M++;
			break;
		}


		memset(cnt, 0, sizeof(cnt));

		rep1(i, N) rep1(j, N) cnt[i] += mat[i][j];

		if (cnt[N] == 0) rep1(i, N) if (cnt[i] > 0)
		{
			swap(mat[i], mat[N]);
			ss << 1 << " " << i << " " << N << endl;
			M++;
			break;
		}
		N--;
	}
	cout << M << endl;
	cout << ss.str();
}
