#include<cstdio>
int hang[1005];
int lie[1005];
int map[1005][1005]; 
int ans[100005][3];
int ansn;
int gai(int &a,int &b){
	int temp=a;
	a=b;
	b=temp;
}
void fun(int n){
	if(n==1)
	return;
	int i;
	for(i=1;i<=n;i++){
		if(lie[i]==0){
			break;
		}
	}
	if(i<n){
		gai(lie[i],lie[n]);
		int j;
		for(j=1;j<=n;j++){
			gai(map[j][i],map[j][n]);
		}
		ans[ansn][0]=2;
		ans[ansn][1]=i;
		ans[ansn++][2]=n;
	}
	for(i=1;i<=n;i++){
		if(hang[i]!=0){
			break;
		}
	}
	if(i<n){
		gai(hang[i],hang[n]);
		int j;
		for(j=1;j<=n;j++){
			gai(map[i][j],map[n][j]);
		}
		ans[ansn][0]=1;
		ans[ansn][1]=i;
		ans[ansn++][2]=n;
	}
	for(i=1;i<=n;i++){
		if(map[n][i]==1){
			lie[i]--;
		}
	}
	fun(n-1);
}
main(){
	int n;
	scanf("%d",&n);
	int i;
	for(i=1;i<=n-1;i++){
		int a,b;
		scanf("%d%d",&a,&b);
		hang[a]++;
		lie[b]++;
		map[a][b]=1;
	}
	fun(n);
	printf("%d\n",ansn);
	for(i=0;i<ansn;i++){
		printf("%d %d %d\n",ans[i][0],ans[i][1],ans[i][2]);
	}
}
