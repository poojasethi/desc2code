#include<iostream>
#include<cstdio>
using namespace std;
struct act
{
	int key;
	int p,q;
}ans[100001];
int g[1001][1001],n,l;
void solve(int k)
{
	bool b;
	for(int i=k;i>=1;i--)
	    {
	    	b=1;
	    	for(int j=1;j<=k;j++)
	    	    if(g[j][i]==1){b=0;break;}
	    	if(b&&i!=k)
	    	{
	    		act t;
	    		t.key=2;
	    		t.p=i;
	    		t.q=k;
	    		ans[++l]=t;
	    		for(int j=1;j<=n;j++)swap(g[j][i],g[j][k]);
	    		break;
	    	}
	    	if(b&&i==k)break;
		}
	for(int i=k;i>=1;i--)
	{
		b=0;
		for(int j=1;j<=k;j++)
		    if(g[i][j]==1){b=1;break;}
		if(b&&i!=k)
		{
			act t;
	    	t.key=1;
	    	t.p=i;
	    	t.q=k;
	    	ans[++l]=t;
	    	for(int j=1;j<=n;j++)swap(g[i][j],g[k][j]);
	    	break;
	    }
	    if(b&&i==k)break;
	}
	return;
}
int main()
{
	cin>>n;
	int x,y;
	for(int i=1;i<=n-1;i++)
	{
		cin>>x>>y;
		g[x][y]=1;
	}
	for(int i=n;i>=1;i--)solve(i);
	cout<<l<<endl;
	for(int i=1;i<=l;i++)cout<<ans[i].key<<' '<<ans[i].p<<' '<<ans[i].q<<endl;
	return 0;
}
