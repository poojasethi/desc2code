



#include <stdio.h>
#include <iostream>
using namespace std;
int da[1001][2], ans[5001][3];
int main()
{
	int n, i, j, k;
	while(scanf("%d", &n) == 1){
		for(i = 1;i <= n - 1;i ++)
			scanf("%d%d", &da[i][0], &da[i][1]);
		for(i = k = 1;i <= n - 1;i ++){
			if(da[i][0] != (i + 1)){
				for(j = i + 1;j <= n - 1;j ++){
					if(da[j][0] == da[i][0])
						da[j][0] = i + 1;
					else if(da[j][0] == i + 1)
						da[j][0] = da[i][0];
				}
				ans[k][0] = 1;
				ans[k][1] = da[i][0];
				ans[k ++][2] = i + 1;
			}
		     
			if(da[i][1] > i){
				for(j = i + 1;j <= n;j ++){
					if(da[j][1] == da[i][1])
						da[j][1] = i;
					else if(da[j][1] == i)
						da[j][1] = da[i][1];
				}
				ans[k][0] = 2;
				ans[k][1] = da[i][1];
				ans[k ++][2] = i;
			}
		}
		printf("%d\n", k - 1);
		for(i = 1;i <= k - 1;i ++)
			printf("%d %d %d\n", ans[i][0], ans[i][1], ans[i][2]);
	}
	return 0;
}
