#include <bits/stdc++.h>

using namespace std;
int x[1010],y[1010];
int a[100010],b[100010],c[100010],m=0;
int main()
{
    int n,i,j;

    cin>>n;

    for(i=1;i<n;i++)
    {
        cin>>x[i]>>y[i];
    }

    for(i=1;i<n;i++)
    {
        if(x[i]!=i+1)
        {
            for(j=i+1;j<n;j++)
            {
                x[j]=x[j]==x[i]?i+1:x[j]==i+1?x[i]:x[j];
            }
            a[m]=1;b[m]=x[i];c[m]=i+1;m++;
        }
        if(y[i]>i)
        {
            for(j=i+1;j<n;j++)
            {
                y[j]=y[j]==y[i]?i:y[j]==i?y[i]:y[j];
            }
            a[m]=2;b[m]=y[i];c[m]=i;m++;
        }
    }
    cout<<m<<endl;

    for(i=0;i<m;i++)
    {
        cout<<a[i]<<" "<<b[i]<<" "<<c[i]<<endl;
    }
}
