#include<cstdio>
#include<algorithm>
const int maxn=1010;
int n,ans,tx[maxn<<1],ty[maxn<<1];
bool t[maxn<<1],a[maxn][maxn],tmp[maxn];
void exchange(int x,int y){
	for(int i=1;i<=y;i++)
		tmp[i]=a[i][x];
	for(int i=1;i<=y;i++)
		a[i][x]=a[i][y];
	for(int i=1;i<=y;i++)
		a[i][y]=tmp[i];
}
void work(int x){
	if(x==1) return;
	int pos=0;
	for(int i=1;i<x;i++){
		bool ok=1;
		for(int j=1;j<=x;j++)
			if(a[j][i]){
				ok=0;
				break;
			}
		if(ok){
			pos=i;
			break;
		}
	}
	if(pos){
		exchange(pos,x);
		tx[++ans]=pos;
		ty[ans]=x;
		pos=0;
	}
	for(int i=1;i<x;i++)
		for(int j=i;j<=x;j++)
			if(a[i][j]){
				pos=i;
				break;
			}
	if(pos){
		std::swap(a[pos],a[x]);
		t[++ans]=1;
		tx[ans]=pos;
		ty[ans]=x;
	}
	work(x-1);
}
int main(){
	scanf("%d",&n);
	for(int x,y,i=1;i<n;i++)
		scanf("%d%d",&x,&y),a[x][y]=1;
	work(n);
	printf("%d\n",ans);
	for(int i=1;i<=ans;i++){
		if(t[i]) printf("1 %d %d\n",tx[i],ty[i]);
		else printf("2 %d %d\n",tx[i],ty[i]);
	}
	return 0;
}
