#include <iostream>
#include <algorithm>
#include <cmath>
#include <vector>

#define rep(i,n) for(int i=0;i<n;i++)
#define pi pair<int,int>

using namespace std;

int n,x,y,a[1010][1010],s[1010],r[1010],cnt;
vector<pair<int,int> > res;

int main(){

  cin>>n;
  rep(i,n-1){
    cin>>x>>y; x--;y--;
    a[x][y] = 1;
    s[y] = 1;
  }
  rep(i,n){
    int my = i;
    for(int j=i;j<n;j++)if(s[j]>s[my])my = j;
    if(my!=i){
      res.push_back(pi(-(my+1),i+1));
      rep(k,n)swap(a[k][i],a[k][my]);
      swap(s[my],s[i]);
    }
  }

  rep(i,n)r[i]=-1;
  rep(i,n)rep(j,n)if(a[i][j]==1)r[i]=j;

  rep(i,n){
    int mx = i;
    for(int j=i;j<n;j++)if(r[j]<r[mx])mx=j;
    if(mx!=i){
      res.push_back(pi(mx+1,i+1));
      swap(r[mx],r[i]);
    }
  }
  cout<<res.size()<<endl;
  rep(i,res.size()){
    if(res[i].first>0)cout<<1<<" "<<res[i].first<<" ";
    else cout<<2<<" "<<-res[i].first<<" ";
    cout<<res[i].second<<endl;
  }

}
