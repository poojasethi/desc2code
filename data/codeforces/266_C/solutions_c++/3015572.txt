#include<stdio.h>
#include<algorithm>
#include<iostream>
#include<vector>
using namespace std;
int n,i,c[1001],col[1001][1001],ans[1001][3],tmp[1001],A,j,mx,p;
int swp(int md,int a,int b){
    if (md==2){
               for (int I=0; I<c[a]; I++) 
               tmp[col[a][I]]=b;}
    
    else {
        swap(tmp[a],tmp[b]);}
        
    if(a==b) return 0;
    ans[A][0]=md;
    ans[A][1]=a;
    ans[A][2]=b;
    A++;
    return 0;
}
main(){

   //  freopen("a.in","r",stdin);
   //  freopen("a.out","w",stdout);
   int a,b;
   scanf("%d",&n);
   for (i=0; i<n; i++){
   scanf("%d%d",&a,&b);
   col[b][c[b]]=a;
   c[b]++;}
   
   for (i=1,j=1; i<=n; i++)
       if (c[i]>0 && i>=j) {swp (2,i,j); j++;}
       
   for(i=1;i<n;i++) {
        for(j=1,mx=0; j<=n-i+1; j++)
            if(tmp[j]>mx) {mx=tmp[j];p=j;}
        swp(1,p,n-i+1);
    }
    
    printf("%ld\n",A);
    for(i=0;i<A;i++) printf("%d %d %d\n",ans[i][0],ans[i][1],ans[i][2]);
}
