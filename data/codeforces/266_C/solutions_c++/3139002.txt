#include<stdio.h>
const int NN=100010;
int x[NN],y[NN];
int a[NN*2],b[NN*2],c[NN*2];
int main()
{
    int n,i,j,tol=0;
    scanf("%d",&n);
    for(i=1;i<n;i++)
        scanf("%d%d",&x[i],&y[i]);
    for(i=1;i<n;i++)
    {
        if(x[i]!=i+1)
        {
            tol++;
            a[tol]=1,b[tol]=x[i],c[tol]=i+1;
            for(j=i+1;j<n;j++)
                x[j]=x[j]==x[i]?i+1:x[j]==i+1?x[i]:x[j];
        }
        if(y[i]>i)
        {
            tol++;
            a[tol]=2,b[tol]=y[i],c[tol]=i;
            for(j=i+1;j<n;j++)
                y[j]=y[j]==y[i]?i:y[j]==i?y[i]:y[j];
        }
    }
    printf("%d\n",tol);
    for(i=1;i<=tol;i++)
        printf("%d %d %d\n",a[i],b[i],c[i]);
    return 0;
}
