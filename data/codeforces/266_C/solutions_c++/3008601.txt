#include<cstdio>
int b[3001][3], i, j, t = 0, a[1001][2], n;
void push_res(int key, int i, int j) {
    b[t][0] = key, b[t][1] = i, b[t++][2] = j;
}
int main()  {
    scanf("%d", &n);
    for(i = 1; i < n; i++)
        scanf("%d%d", &a[i][0], &a[i][1]);
    //第n个数据放在第n+1行n列 
    for(i = 1, t = 1; i < n; i++) {
        if(a[i][0] != i + 1) {
            for(j = i + 1; j < n; j++)
                if(a[j][0] == a[i][0])
                    a[j][0] = i + 1;
                else if(a[j][0] == i + 1)
                    a[j][0] = a[i][0];
            push_res(1, a[i][0], i + 1);
        }
        if(a[i][1] > i) {
            for(j = i + 1; j < n; j++)
                if(a[j][1] == a[i][1])
                    a[j][1] = i;
                else if(a[j][1] == i)
                    a[j][1] = a[i][1];
            push_res(2, a[i][1], i);
        }
    }
    printf("%d\n",t - 1);
    for(i = 1; i < t; i++)
        printf("%d %d %d\n", b[i][0], b[i][1], b[i][2]);
}
