#include <iostream>
#include <algorithm>
using namespace std;

const int maxn = 1005;
const int mm = 100005;
int x[maxn], y[maxn];
int op[mm], a[mm], b[mm];

int main() {
	int n;
	while (cin >> n) {
		for (int i = 1; i < n; i++)
			cin >> x[i] >> y[i];
		int m = 0;
		for (int i = 1; i < n; i++) {
			if (x[i] != i + 1) {
				for (int j = i + 1; j < n; ++j)
					x[j] = x[j] == i + 1 ? x[i] : x[j] == x[i] ? i + 1 : x[j];
				op[m] = 1;
				a[m] = x[i];
				b[m] = i + 1;
				++m;
			}
			if (y[i] > i) {
				for (int j = i + 1; j < n; ++j)
					y[j] = y[j] == i ? y[i] : y[j] == y[i] ? i : y[j];
				op[m] = 2;
				a[m] = y[i];
				b[m] = i;
				++m;
			}
		}
		cout<<m<<endl;
		for(int i=0;i<m;i++)cout<<op[i]<<" "<<a[i]<<" "<<b[i]<<endl;
	}
}
