#include<cmath>
#include<cstdio>
#include<vector>
using namespace std;
#define x first
#define y second
#define pb push_back
#define mp make_pair
const int MAXN=1010;
int n;
pair<int, int >p[MAXN];
vector<pair<int,pair<int,int > > >que;
int main()
{
	int i,j,n;
	scanf("%d",&n);
	for(i=1;i<n;i++)
		scanf("%d%d",&p[i].x,&p[i].y);
	for(i=1;i<n;i++)
	{
		if(p[i].x!=i+1)
		{
			for(j=i+1;j<n;j++)
				p[j].x=p[j].x==p[i].x?i+1:p[j].x==i+1?p[i].x:p[j].x;
			que.pb(mp(1,mp(p[i].x,i+1)));
		}
		if(p[i].y>i)
		{
			for(j=i+1;j<n;j++)
				p[j].y=p[j].y==p[i].y?i:p[j].y==i?p[i].y:p[j].y;
			que.pb(mp(2,mp(p[i].y,i)));
		}
	}
	printf("%d\n",que.size());
	for(i=0;i<que.size();i++)
		printf("%d %d %d\n",que[i].x,que[i].y.x,que[i].y.y);
	return 0;
}
