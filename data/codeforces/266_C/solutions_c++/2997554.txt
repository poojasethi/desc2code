#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	short n;
	cin >> n;

	short r, c, a[n-1][2], b[n-1][2], s = 0, t = 0, x[n+1], rx[n+1], y[n+1], ry[n+1];
	for (short i = 1; i <= n; i++)
		x[i] = rx[i] = y[i] = ry[i] = i;

	for (short i = 1; i < n; i++) {
		cin >> r >> c;
		if (x[c] < y[r] && y[r] <= i) continue;
		if (y[r] != i+1) {
			a[s][0] = i+1, a[s][1] = y[r], ++s;
			ry[y[r]] = ry[i+1], y[ry[i+1]] = y[r];
			y[r] = i+1, ry[i+1] = r;
		}
		if (x[c] > i) {
			b[t][0] = i, b[t][1] = x[c], ++t;
			rx[x[c]] = rx[i], x[rx[i]] = x[c];
			x[c] = i, rx[i] = c;
		}
	}
	cout << s+t << '\n';
	for (short i = 0; i < s; ++i)
		cout << 1 << ' ' << a[i][0] << ' ' << a[i][1] << '\n';
	for (short i = 0; i < t; ++i)
		cout << 2 << ' ' << b[i][0] << ' ' << b[i][1] << '\n';
	return 0;
}
