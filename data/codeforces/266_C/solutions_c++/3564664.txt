#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

struct g
{
	int x,y;
}s[1010];
struct gg
{
	int x,y,t;
}a[101000];
int main()
{
	int i,j,k,n,m,t=0;
	while(scanf("%d",&n)!=EOF)
	{
		t=0;
		for(i=1;i<n;i++)
			scanf("%d%d",&s[i].x,&s[i].y);
		for(i=1;i<n;i++)
		{
			if(s[i].x!=i+1)
			{
				for(j=i+1;j<n;j++)
				{
					if(s[j].x==i+1)
						s[j].x=s[i].x;
					else if(s[j].x==s[i].x)
						s[j].x=i+1;
				}
				a[t].t=1;a[t].x=s[i].x;a[t].y=i+1;t++;
			}
			if(s[i].y>i)
			{
				for(j=i+1;j<n;j++)
				{
					if(s[j].y==i)
						s[j].y=s[i].y;
					else if(s[j].y==s[i].y)
						s[j].y=i;
				}
				a[t].t=2;a[t].x=s[i].y;a[t].y=i;t++;
			}
		}
		printf("%d\n",t);
		for(i=0;i<t;i++)
			printf("%d %d %d\n",a[i].t,a[i].x,a[i].y);
	}
}
