#include <iostream>
using namespace std;
int main(void){
    int n,x[1024],y[1024],a[1<<17],b[1<<17],c[1<<17],m=0;
    cin>>n;
    for(int i=1;i<n;++i)cin>>x[i]>>y[i];
    for(int i=1;i<n;++i){
        if(x[i]!=i+1){
            for(int j=i+1;j<n;++j)
                x[j]=x[j]==i+1?x[i]:x[j]==x[i]?i+1:x[j];
            a[m]=1,b[m]=x[i],c[m]=i+1,++m;
        }
        if(y[i]>i){
            for(int j=i+1;j<n;++j)
                y[j]=y[j]==i?y[i]:y[j]==y[i]?i:y[j];
            a[m]=2,b[m]=y[i],c[m]=i,++m;
        }
    }
    cout<<m<<endl;
    for(int i=0;i<m;++i)
        cout<<a[i]<<" "<<b[i]<<" "<<c[i]<<endl;
}
