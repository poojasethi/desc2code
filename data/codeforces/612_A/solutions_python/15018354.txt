n, p, q = map(int, raw_input().split(" "))
s = raw_input()

flag = False
for i in range(n + 1):
    for j in range(n + 1):
        if flag:
            break
        if (i * p + j * q) == n:
            flag = True
            print i + j
            for k in range(i):
                print s[k * p: (k + 1) * p]
            for k in range(j):
                print s[i * p + k * q: i * p + (k + 1) * q]
            break
if not flag:
    print -1
