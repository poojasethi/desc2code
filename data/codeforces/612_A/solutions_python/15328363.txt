def solve():
    n, p, q = map(int, raw_input().split())
    s = raw_input()

    tmp = ''
    ix = 0
    while ix <= n:
        l = n - ix
        if l % q == 0:
            print ix / p + l / q
            for i in xrange(0, len(tmp), p):
                print s[i:i+p]
            for i in xrange(len(tmp), len(s), q):
                print s[i:i+q]
            return
        tmp += s[ix:ix+p]
        ix += p
    print -1

solve()
