n,p,q = map(int,raw_input().split(' '))
str = raw_input()
Len = len(str)

found = False
for i in range(101):
  if i * p > n :
    break
  if (n - i * p) % q == 0:
    print i + (n - i * p ) / q
    found = True
    for j in range(i):
      print str[j*p:j*p+p]
    for j in range(101):
      if i * p + j * q >= n:
        break
      else:
        print str[i * p + j * q: i * p + j * q + q]
    break
if not found:
  print -1
