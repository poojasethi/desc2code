n , p , q = [int(x) for x in raw_input().split()]
s = raw_input()

x = 0
y = 0

while x * p <= n:
    c = n - x * p
    if c % q == 0:
        y = c / q
        break
    x += 1
    
if x * p > n:
    print -1

else:
    print (x + y)
    for i in xrange(x):
        print s[(i * p) : ((i + 1) * p)]
    for j in xrange(y):
        print s[(x * p + j * q) : (x * p + (j + 1) * q)]
