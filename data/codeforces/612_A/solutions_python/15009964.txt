from sys import stdin
n,p,q = map(int,stdin.readline().strip().split())
a = stdin.readline().strip()
valid = 0
j = 0
for i in xrange(n):
 cur = i*p
 if cur > n:
  break
 rem = n-cur
 if rem%q==0:
  j = rem/q
  valid = 1; break
if valid:
 pr = 0
 print i+j
 for x in xrange(i):
  print a[pr:pr+p]
  pr += p
 for x in xrange(j):
  print a[pr:pr+q]
  pr+=q
else:
 print -1
