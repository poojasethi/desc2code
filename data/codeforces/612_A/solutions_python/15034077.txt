n, p, q = map(int, raw_input().split())
string = raw_input()
f = 0
for i in range(n):
    if i * p > n:
        break
    if (n - i * p) % q == 0:
        # print (i)
        print (i + (n - i * p) / q)
        for j in range(i):
            print (string[j * p : (j + 1) * p])
        for j in range((n - i * p) / q):
            print (string[i * p + j * q : i * p + (j + 1) * q])
        f = 1
        break
if f == 0:
    print (-1)
