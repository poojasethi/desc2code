n,p,q=map(int,raw_input().split())
s=raw_input()
flag=0
count=0
if n<p and n<q:
	print -1
	exit(0)
p,q=min(p,q),max(p,q)
li=[]
while(n>0):
	if(n%q==0):
		count+=n/q
		for i in xrange((n/q)):
			li.append(s[:q])
			s=s[q:]
		flag=1
		break
	n-=p
	count+=1
	li.append(s[:p])
	s=s[p:]
	
if n==0 or flag:
	print count
	for i in li:
		print i
else:
	print -1
