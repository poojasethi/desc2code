line = raw_input().split()
a = int(line[0])
b = int(line[1])
c = int(line[2])
ans1 = 0
ans2 = a/c+1
found = False
while (ans2 != -1):
	if (b*ans1+c*ans2 > a):
		ans2 -= 1
	elif (b*ans1+c*ans2 < a):
		ans1 += 1
	else:
		found = True
		break
if not found:
	print -1
else:
	pointer = 0
	string = str(raw_input())
	print ans1 + ans2
	for i in range(ans1):
		print string[pointer:b+pointer]
		pointer += b
