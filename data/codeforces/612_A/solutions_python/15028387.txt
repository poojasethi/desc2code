n,p,q = map(int,raw_input().split())
s=raw_input()
a = n/p
b = 200
for x in xrange(a,-1,-1):
    if (n-x*p)%q==0:
        a=x
        b=(n-x*p)/q
        break
if b!=200:
    print a+b
    for x in xrange(a):
        print s[x*p:p*(x+1)]
    for y in xrange(b):
        print s[a*p+y*q:q*(y+1)+a*p]
else:
    print -1
