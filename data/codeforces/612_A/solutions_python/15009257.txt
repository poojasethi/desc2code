def f(s, p, q):
    arr = []
    while s != '':
        if len(s) % q == 0:
            arr.append(s[:q])
            s = s[q:]
        elif len(s) >= p:
            arr.append(s[:p])
            s = s[p:]
        else: return None
    return arr

def ff():
    n, p, q = map(int,raw_input().split(' '))
    s = raw_input().strip()
    rs = f(s, min(p, q), max(p, q))
    if rs == None: print -1
    else:
        print len(rs)
        print '\n'.join(rs)
ff()
