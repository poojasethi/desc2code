line = [int(i) for i in raw_input().strip().split()]
n = line[0]
p = line[1]
q = line[2]
word = raw_input()

a = n // p
b = n % p

X = -1
Y = -1

for x in xrange(0 , a + 1) :
	if ((a - x) * p + b) % q == 0 :
		X = x
		Y = ((a - x) * p + b) // q
		break

if (X == -1) :
	print -1
else :
	print (X + Y)
	count = 0
	while count < X*p :
		print word[count:count+p]
		count += p
	while count < n :
		print word[count:count+q]
		count += q
