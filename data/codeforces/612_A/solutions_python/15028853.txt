n, p, q = map(int, raw_input().split())
s = raw_input()

x, y = 0, 0
flag = False
while p * x <= n:
	if (n - p * x) % q == 0:
		flag = True
		break
	x += 1
	
y = (n - p * x) / q if flag == True else -x
print x + y if x + y > 0 else -1

index = 0
if flag:
	for i in xrange(x):
		print s[index:index + p]
		index += p
	for i in xrange(y):
		print s[index:index + q]
		index += q
