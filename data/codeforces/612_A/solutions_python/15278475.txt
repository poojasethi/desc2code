n,p,q=map(int, raw_input().split())

word=raw_input()

def print_result(a,b):
    print a+b
    cur=0
    for _ in xrange(a):
        print word[cur:cur+p]
        cur+=p
        
    for _ in xrange(b):
        print word[cur:cur+q]
        cur+=q
        
for i in xrange((n+p-1)/p+1):  
    if p*i<=n and (n-p*i)%q==0:
        print_result(i,(n-p*i)/q)
        exit(0)
      
print -1
