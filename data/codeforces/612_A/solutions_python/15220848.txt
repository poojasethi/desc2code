def main():
	n,p,q = map(int, raw_input().split())
	s = raw_input()

	#print n,p,q
	#print s

	for i in range(n/p+1):
		#print i
		if (n-i*p) % q == 0:
			print i+(n-p*i)/q
			for j in range(i):
			   print s[j*p:(j+1)*p]
			for j in range((n-p*i)/q):
			   print s[p*i+j*q:p*i+(j+1)*q]
			return

	print -1

main()
