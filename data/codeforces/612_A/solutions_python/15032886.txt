n, p, q = map(int, raw_input().split())
s = raw_input()

for x in xrange(n/p + 1):
    if (n-p*x)%q == 0:
        y = (n-p*x)/q
        print x+y
        for i in xrange(x):
            print s[p*i:p*(i+1)]
        for j in xrange(y):
            print s[p*x+q*j:p*x+q*(j+1)]
        break
else:
    print -1
