if __name__ == "__main__":
  n, p, q = map(int, raw_input().split(" "))
  s = raw_input()
  
  a = 0
  b = -1
  while a <= n/p:
    if (n - a*p) % q == 0:
      b = (n - a*p)/q
      break
    a += 1

  if b is -1:
    print b
  else:
    print a + b
    for i in range(a): print s[i*p:i*p + p]
    for i in range(b): print s[a*p + i*q:a*p + i*q + q]
