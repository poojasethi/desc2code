__author__ = 'Shashi'

n,p,q=[int(x) for x in raw_input().split(' ')]
string=raw_input()

m=min(p,q)
N=n/m
a=[]
for i in range(0,N+2):
    for j in range(0,N+2):

        if (i*p+j*q)==n:
            a=[i,j]
            break

if not a:
    print "-1"
else:
    current=0
    print sum(a)
    while a[0]:
        print(string[current:current+p])
        a[0]-=1
        current+=p
    while a[1]:
        print(string[current:current+q])
        a[1]-=1
        current+=q
