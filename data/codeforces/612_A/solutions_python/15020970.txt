n, p, q = tuple(map(int, raw_input().split()))
s = raw_input()
r = 0

rp = rq = -1

for a in range(101):
	for b in range(101):
		if p*a + q*b == n:
			rp, rq = a, b

if rp < 0 or rq < 0:
	print -1
else:
	print (rp + rq)
	c = 0
	for i in range(rp):
		print s[c:c+p]
		c += p
	for i in range(rq):
		print s[c:c+q]
		c += q
