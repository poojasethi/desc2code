ans=-1
for i in range(100):
	if i*n[1]>n[0] or ans!=-1:
		break
	for j in range(n[0]-i*n[1]+1):
		if i*n[1]+j*n[2]>n[0]:
			break
		if i*n[1]+j*n[2]==n[0]:
			ans=[i,j]
			break
if ans==-1:
	print ans
else:
	print ans[0]+ans[1]
	for i in range (ans[0]):
		print s[n[1]*i : n[1]*(i+1)]
	temp=ans[0]*n[1]
	for i in range(ans[1]):
