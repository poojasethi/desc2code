n,p,q = (int(i) for i in raw_input().split(" "))
s = str(raw_input())

def trysplit(i, j):
    return p * i + q * j == n

def dosplit(i, j):
    print i + j
    pos = 0
    while i > 0:
        print s[pos:pos+p]
        pos += p
        i -= 1
    while j > 0:
        print s[pos:pos+q]
        pos += q
        j -= 1

for i in range(100):
    for j in range(100):
        if trysplit(i, j):
            dosplit(i, j)
            exit(0)

print -1
