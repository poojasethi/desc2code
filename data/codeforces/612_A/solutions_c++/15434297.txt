#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
	int n,p,q;
	string s;
	cin>>n>>p>>q;
	cin>>s;
	int k;
	for (int i=0;i<=n/p;i++)
	{
		k=n-i*p;
		if (k%q==0)
		{
			cout<<i+k/q<<endl;
			for (int j=0;j<i;j++)
			{
				cout<<s.substr(j*p,p)<<endl;
			}
			for (int j=0;j<k/q;j++)
			{
				cout<<s.substr(i*p+j*q,q)<<endl;
			}
			return 0;
		}
	}
	cout<<-1;
	return 0;
}
