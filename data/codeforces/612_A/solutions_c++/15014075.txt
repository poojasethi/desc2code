#include <iostream>
#include <string>
#include <sstream>
using namespace std;

int main() {
  long long n,p,q,a,b,t,i;
  string s;

  cin >> n; cin>> p; cin >>q;
  cin >> s;
  
  a=0; b=0; t=n;
  
  while (t%q) {
    t-=p;
    a++;
    if ( t<0 ) { cout << "-1" << endl; return 0; }
  }
  b=t/q;
  cout << a+b << endl;

  t=n-t; i=0;
  while (i<t) {
    cout << s.substr(i,p) <<endl;
    i+=p;
  }
  while (i<n) {
    cout << s.substr(i,q) <<endl;
    i+=q;
  }

  return 0;
}
