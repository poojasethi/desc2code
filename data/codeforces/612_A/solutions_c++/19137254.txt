#include<bits/stdc++.h>

using namespace std;

int main()
{
  int n,a,b;
  string s;
  cin>>n>>a>>b;
  cin>>s;
  int k,t;
  bool ok=false;
  for(k=0;k<=n/a;k++)
    if((n-k*a)%b==0)
      {
	ok=true;
	t=(n-k*a)/b;
	break;
      }
  if(ok==false){
    cout<<-1<<endl;
    return 0;}
  int now=0;
  cout<<k+t<<endl;
  for(int i=1;i<=k;i++,now+=a)
    cout<<s.substr(now,a)<<endl;
  
  for(int i=1;i<=t;i++,now+=b)
    cout<<s.substr(now,b)<<endl;
  return 0;
}
