#include <bits/stdc++.h>
using namespace std;

int main()
{
string s;
int n,p,q,k;

cin>>n>>p>>q;

cin>>s;

for(int i=0; i<=n/p; i++)
{
 k=(n-(p*i))/q;
 if(((p*i)+(q*k))==n)
 {

  cout<<endl<<i+k<<endl;

  for(int j=0; j<i; j++)
  {
  cout<<s.substr(p*j,p)<<endl;
  }

  for(int j=0; j<k; j++)
  {
  cout<<s.substr(p*i+q*j,q)<<endl;
  }
 return 0;
 }

}
cout<<"-1"<<endl;



}
