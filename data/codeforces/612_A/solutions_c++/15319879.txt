#include<bits/stdc++.h> 
using namespace std; 
string s; 
int main() { 
int n, p, q; 
cin >> n >> p >> q; 
cin >> s; 
for (int i = 0; i < 100; i++) { 
for (int j = 0; j < 100; j++) { 
if (i*p + j*q == s.size()) { 
cout << i + j << endl; 
for (int k = 0; k < i; k++) { 
for (int t = 0; t < p; t++) 
cout << s[k*p+t]; 
cout << endl; 
} 
for (int k = 0; k < j; k++) { 
for (int t = 0; t < q; t++) 
cout << s[i*p+k*q+t]; 
cout << endl; 
} 
return 0;
} } } 
return puts("-1"); }
