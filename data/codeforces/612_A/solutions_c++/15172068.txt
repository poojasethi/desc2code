#include <iostream>
using namespace std;
int main() 
{	int n,p,q,i,j,k,l,flag=1;
	cin>>n>>p>>q;
	string s;
	cin>>s;
	for(i=0;i<=100;i++)
	{	for(j=0;j<=100;j++)
		{	if ((p*i+q*j)==n)
			{	cout<<i+j<<endl;
				int p1 = 0;
				for(k=0;k<i;k++)
				{	cout<<string(s.begin()+p1,s.begin()+p1+p)<<endl;
					p1+=p;
				}
				for(l=0;l<j;l++)
				{	cout<<string(s.begin()+p1,s.begin()+p1+q)<<endl;
					p1+=q;
				}
				return 0;
			}
		}
	}
	cout<<"-1";
	return 0;
}
