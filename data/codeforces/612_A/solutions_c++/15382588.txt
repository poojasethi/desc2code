#include<stdio.h>
int main(){
	int i,j,n,m,x,y,p,q;
	char s[101],*st;
	scanf("%d%d%d",&n,&p,&q);
	scanf("%s",s);
	st=s;
	for(i=0;;++i){
		m=n-i*p;
		if(m<0){
			printf("-1");
			return 0;
		}
		else if(!(m%q)){
			x=i;
			y=m/q;
			break;
		}
	}
	printf("%d\n",x+y);
	for(i=0;i<x;++i){
		for(j=0;j<p;++j){
			printf("%c",*st);
			++st;
		}
		printf("\n");
	}		
	for(i=0;i<y;++i){
		for(j=0;j<q;++j){
			printf("%c",*st);
			++st;
		}
		printf("\n");
	}
	return 0;
}
