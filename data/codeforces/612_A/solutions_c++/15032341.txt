#include <iostream>
#include <cstdio>
using namespace std;

int n, p, q;
string a;
int main()
{
	int i, j;
//	freopen ("input.txt", "r", stdin);
	cin >> n >> p >> q;
	cin >> a;
	for (i = 0; i <= n; i+=p) {
		if ((n - i) % q == 0) {
			cout << i / p + (n - i) / q << endl;
			for (j = 0; j < i; j+=p) {
				cout << a.substr(j, p) << endl;
			}
			for (; j < n; j+=q) {
				cout << a.substr(j, q) << endl;
			}
			return 0;
		}
	}
	cout << -1 << endl;
    return 0;
}
