#include <iostream>
#include <cstring>
using namespace std;


int main(){
  int n, p, q;
  cin>>n>>p>>q;
  string s;
  cin>>s;

  for(int i = 0; i <= n/p; ++i){
    int j = (n - i * p)/q;

    if(i * p + j * q == n){
        cout<<i+j<<endl;
        for(int p_i = 0; p_i < i; p_i++){

          cout<<s.substr(p_i * p, p)<<endl;
        }
        for(int q_j = 0; q_j < j; q_j++){
          cout<<s.substr(i*p + q_j*q, q)<<endl;
        }
        return 0;
    }


  }
  cout<<-1<<endl;
}
