#include <bits/stdc++.h>
using namespace std;

int N, A, B;
string S;

int main() {
//	freopen("in.txt", "r", stdin);
	cin >> N >> A >> B >> S;
	for (int i = 0; i <= N/A; i++) {
		int j = (N - A * i) / B;
		if (A * i + B * j == N) {
			cout << i + j << '\n';
			while (i--) {
				cout << S.substr(0, A) << '\n';
				S = S.substr(A);
			}
			while (j--) {
				cout << S.substr(0, B) << '\n';
				S = S.substr(B);
			}
			return 0;
		}
	}
	cout << -1;
}
