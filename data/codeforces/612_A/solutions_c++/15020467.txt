#include <cstdio>

int main() {
    int N, P, Q, i, k;
    char s[1000];
    scanf("%d %d %d", &N, &P, &Q);
    scanf("%s", s);
    for (k=0; k<N; ) {
        if ((N-k)%P == 0) {
            break;
        }
        k += Q;
    }
    if (k > N) {
        printf("-1\n");
        return 0;
    }
    printf("%d\n", k/Q + (N-k)/P);
    for (i=0; i<k; i += Q) {
        printf("%.*s\n", Q, s+i);
    }
    for (; i<N; i += P) {
        printf("%.*s\n", P, s+i);
    }
}
