#include<bits/stdc++.h>
using namespace std;
int n,p,q,i,j,x;
string s;
int main()
{
	cin>>n>>p>>q;
	cin>>s;
	i=0;
	j=-1;
	while(n>=p*i)
	{
		if((n-p*i)%q==0)// && n-p*i>=0)
		{
			j=(n-p*i)/q;
			break;
		}
		i++;
	}
	x=0;
	if(j==-1)
	{
		cout<<"-1\n";
		return 0;
	}
	cout<<i+j<<endl;
	while(i--)
	{
		cout<<s.substr(x,p)<<endl;
		x+=p;
	}
	while(j--)
	{
		cout<<s.substr(x,q)<<endl;
		x+=q;
	}
	return 0;
}
