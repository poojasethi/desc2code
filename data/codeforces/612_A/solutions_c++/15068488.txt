#include <iostream>
#include <cstdio>
using namespace std;

int n, p, q;
string a;
int main()
{
	int i, k;
//	freopen ("input.txt", "r", stdin);
	cin >> n >> p >> q >> a;
	for (k = 0; k <= n; k += p) {
		if ((n - k) % q == 0) {
			cout << k / p + (n - k) / q << endl;
			for (i = 0; i < k; i += p) {
				cout << a.substr(i, p) << endl;
			}
			for (; i < n; i += q) {
				cout << a.substr(i, q) << endl;
			}
			return 0;
		}
	}
	cout << "-1";
	return 0;
}
