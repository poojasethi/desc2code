#include <iostream>

using namespace std;

int main(){
	int n, p ,q;
	cin >> n >> p >> q;
	for(int i=0; i<=n; i++){
		if(n-i*p >= 0 && (n-i*p) % q == 0){
			cout << i + ((n-i*p) / q) << endl;
			string s;
			cin >> s;
			for(int j=0; j<i; j++){
				cout << s.substr(0,p) << endl;
				s = s.substr(p);
			}
			for(int j=0; j<((n-i*p) / q); j++){
				cout << s.substr(0,q) << endl;
				s = s.substr(q);
			}
			return 0;
		}
	}
	cout << -1 << endl;
}
