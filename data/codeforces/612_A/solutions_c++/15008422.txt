#include<bits/stdc++.h>
using namespace std;
int n,p,q,l=-1;
char s[120];
int main(){
	scanf("%d%d%d",&n,&p,&q);
	scanf("%s",s);
	for (int i=0; i<=100; i++){
		if (n-i*p>=0 && (!((n-i*p)%q))){
		    printf("%d\n",i+(n-i*p)/q);
			for (int j=1; j<=i; j++){
			    for (int k=1; k<=p; k++) printf("%c",s[++l]);
				printf("\n");
			}
			for (int j=1; j<=(n-i*p)/q; j++){
				for (int k=1; k<=q; k++) printf("%c",s[++l]);
				printf("\n");
			}
			return 0;
		}
	}
	printf("-1");
}
