#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,p,q,m;
    cin>>n>>p>>q;
    string s;
    cin>>s;
    for (int i=0;i<=n/p;i++)
	{
		m=n-i*p;
		if (m%q==0)
		{
			cout<<i+(m/q)<<endl;
			for (int j=0;j<i;j++)
				cout<<s.substr(j*p,p)<<endl;
			for (int j=0;j<(m/q);j++)
				cout<<s.substr(i*p+j*q,q)<<endl;
			return 0;
		}
	}
	cout<<-1;
	return 0;
}
