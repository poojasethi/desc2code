#include <iostream>
using namespace std;

int main()
{
    
    string s;
    int k,n,m;  cin>>k>>n>>m;
    cin>>s;
    
    int i=0,j=0;
    bool f=0;
    for(i=0;i<=100;i++) {   for(j=0;j<=100;j++) if(i*n+j*m==k)  {   f=1;    break;  }   if(f)   break;}
    if(!f)  {   cout<<-1;   return 0;   }

    cout<<i+j<<'\n';

    for(int ii=0;ii<i;ii++) cout<<s.substr(ii*n,n)<<'\n';
    for(int jj=0;jj<j;jj++) cout<<s.substr(jj*m+n*i,m)<<'\n';

    return 0;
}
