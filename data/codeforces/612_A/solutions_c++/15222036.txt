#include<cstdio>

int n, a, b;
char s[111];

void prt(int l, int r) {
	for(int i=l;i<r;i++)
		printf("%c", s[i]);
	printf("\n");
}

int main() {
	scanf("%d%d%d", &n, &a, &b);
	scanf("%s", s);
	for(int i=0;i<=n;i+=a) {
		if((n-i)%b) continue;
		printf("%d\n", i/a+(n-i)/b);
		for(int j=0;j<i;j+=a)
			prt(j, j+a);
		for(int j=i;j<n;j+=b)
			prt(j,j+b);
		return 0;
	}
	printf("-1\n");
	
	return 0;
}
