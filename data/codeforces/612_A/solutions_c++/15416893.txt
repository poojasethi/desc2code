#include<stdio.h>
int main()
{
	char s[108]={0};
	int n,p,q;
	scanf("%d%d%d",&n,&p,&q);
	scanf("%s",s);
	int i;
	int m=n/p;
	for(i=0;i<=m;i++)
	{
		if((n-p*i)%q==0)
		{
			printf("%d\n",i+(n-p*i)/q);
			int j;
			for(j=0;j<n;j++)
			{
				printf("%c",s[j]);
				if(j<p*i)
				{
					if((j+1)%p==0)
					printf("\n");
				}
				else
				{
					if((j+1-i*p)%q==0)
					printf("\n");
				}
			}
			return 0;
		}
	}
	printf("-1\n");
	return 0;
}
