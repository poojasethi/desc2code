#include <iostream>
#include <string>
using namespace std;
int n,p,q;
string s;
int main ()
{  
    cin>>n>>p>>q;
	cin>>s;
    for (int i=0;i<=n/p;i++)
    {
        int j=(n-i*p)/q;
        if (i*p+j*q==n)
        {
            cout<<i+j<<endl;
            for (int i1=0;i1<i;i1++)
                cout<<s.substr(i1*p,p)<< endl;
          for (int i2=0;i2<j;i2++)
                cout<<s.substr(i*p+i2*q,q)<<endl;
            return 0;
        }
    }
    cout<<-1;
}
