#include <iostream>
#include <cmath>
#include <string>
using namespace std;

int main()
{
	int n, p, q;
	string a;
	cin >> n >> p >> q >> a;
	for (int i = 0; i * p <= n; i++) {
		if ((n - i*p) % q == 0) {
			int j = (n - i*p) / q, t = 0;
			cout << i + j << '\n';
			for (int z = 0; z < i; z++) cout << a.substr(t, p) << '\n', t += p;
			for (int z = 0; z < j; z++) cout << a.substr(t, q) << '\n', t += q;
			return 0;
		}
	}
	cout << -1;
}
