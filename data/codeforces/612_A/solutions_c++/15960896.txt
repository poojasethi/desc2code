#include <iostream>
using namespace std;
int p , q;
int main(){
    int N;
    cin >> N >> p >> q;
    string s;
    cin >> s;
    int i , j = -1;
for( i = 0 ; i <= N/p ; i++ )
if((N - i * p) % q == 0) { j = (N - i * p) / q; break;}

if(j == -1) cout << "-1";

else {
cout << i+j << endl;
for(int k = 0 ; k < i ; k++)
cout << s.substr(k * p ,p) << endl;

for(int k = 0 ; k < j ; k++)
cout << s.substr(i*p + k*q , q) << endl;

}
    return 0;
}
