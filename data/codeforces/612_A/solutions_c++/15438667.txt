#include <bits/stdc++.h>

using namespace std;

int main() {
	string s;
	
	int l,a,b;
	
	cin>>l>>a>>b;
	cin>>s;
	
	int x=0,y=0;
	for(int i=0; i < l/a + 1 ; i++){
		if((l-a*i)%b == 0){
			x=i;
			y=(l-a*i)/b;
			break;
		}
	}
	
	if(x || y){
		cout<<x+y<<endl;
		int i=0,k=0,j;
		for(;i<x;i++){
			for(j=0; j < a;j++)
			cout<<s[k++];
		cout<<endl;
		}
		
		for(i=0;i<y;i++){
			for(j=0;j<b;j++)
			cout<<s[k++];
		cout<<endl;
		}
		
	}
	else
	cout<<"-1";
}
