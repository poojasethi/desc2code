#include <bits/stdc++.h>
using namespace std;
int main() {
	int i,j,k,n,m,l,p=0,flag=0;
	char c[105];
	cin>>n>>k>>l>>c;
	for(i=0;i<=n/k&&!flag;i++)
		if(!((n-k*i)%l)) {
			flag=1;
			cout<<i+(n-k*i)/l<<endl;
			for(j=0;j<i;j++) {
				m = k;
				while(m--)
					cout<<c[p++];
				cout<<endl;
			}
			for(j=0;j<(n-k*i)/l;j++) {
				m=l;
				while(m--)
					cout<<c[p++];
				cout<<endl;
			}
		}
	if(!flag)
		cout<<"-1\n";
	return 0;
}
