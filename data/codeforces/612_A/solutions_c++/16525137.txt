#include<iostream>
using namespace std;
main()
{
	int n,m ,k;
	cin>>n>>m>>k;
	string s; cin>>s;
	int i=0,j=0;
	for(i=0;i*m<=n;i++){
		j = (n-i*m)/k;	
		if((i*m+j*k)==n){
			break;
		}
	}
	if((i*m+j*k)==n && i>=0 && j>=0){
		cout<<i+j<<endl;
		int l=0;
		for(int p=0;p<i;p++){
			for(int q=0;q<m;q++){
				cout<<s[l++];
			}
			cout<<endl;
		}
		for(int p=0;p<j;p++){
			for(int q=0;q<k;q++){
				cout<<s[l++];
			}
			cout<<endl;
		}
	}
	else cout<<"-1";
}
