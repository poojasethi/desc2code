#include <bits/stdc++.h>

using namespace std;

#define ll long long int

int main()
{
	ll n,p,q,x,y,i,j,k;
	string str;
	cin>>n>>p>>q;
	cin>>str;
	ll m=n/p;
	for(x=0;x<=m;x++)
		if(!((n-x*p)%q)) break;
	if(x==(m+1))
	{
		cout<<-1<<endl;
		return 0;
	}
	y=(n-x*p)/q;
	cout<<(x+y)<<endl;
	k=0;
	for(i=0;i<x;i++)
	{
		j=0;
		while(j<p)
		{
			cout<<str[k];
			j++;
			k++;
		}
		cout<<endl;
	}
	for(i=0;i<y;i++)
	{
		j=0;
		while(j<q)
		{
			cout<<str[k];
			j++;
			k++;
		}
		cout<<endl;
	}
	return 0;
}
