#include <iostream>
#include <cstdio>
using namespace std;

int n,p,q;
string s;
int main(){
    int i,j;
    cin>>n>>p>>q>>s;
    for(i=0; i<=n; i+=p){
        if((n-i)%q==0){
            cout<<i/p+(n-i)/q<<endl;
            for(j=0; j<i; j+=p) cout<<s.substr(j,p)<<endl;
            for(; j<n; j+=q) cout<<s.substr(j,q)<<endl;
            return 0;
        }
    }
    cout<<-1;
    return 0;
}
