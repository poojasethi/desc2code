#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,p,q;
	cin >> n >> p >> q;
	string s;
	cin >> s;
	
	for(int i = 0 ; i <= n ; i += p){
		if((n-i) % q != 0)
        {
            continue;
        }
		cout << i/p + (n-i)/q;
		int j = 0;
		for(j = 0 ; j < i ; j++){
			if(j % p == 0) cout << endl;
			cout << s[j];
		}
		for(j; j < n ; j++){
			if((j-i) % q == 0) cout << endl;
			cout << s[j];
		}
		return 0;
	}
	cout << -1;
	
    return 0;

}
