#include<stdio.h>
void Cut(char s[],int n,int q)
{
	int i;
	for(i=0;i<n;i++)
	{	
		if(i%q==0)
			printf("\n");
		printf("%c",s[i]);
	}
}
int main()
{
	int n,p,q;
	int a,b,i;
	char s[110]={'\0'};
	scanf("%d%d%d",&n,&p,&q);
	scanf("%s",s);
	a=n/q;
	for(i=0;i<=a;i++)	
	{
		b=n-i*q;
		if(b%p==0)
		{
			printf("%d",i+b/p);
			Cut(s,i*q,q);
			Cut(s+i*q,b,p);
			return 0; 
		}
	}
	printf("-1");
	return 0;
}
