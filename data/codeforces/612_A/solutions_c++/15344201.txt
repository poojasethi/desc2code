#include<iostream>
using namespace std;

int main(){
	int n, p, q;
	string s;
	cin >> n >> p >> q >> s;
	int m = n, noq = 0;
	while(m >= 0 && m % p != 0){
		m -= q;
		noq++;
	}
	if(m < 0){
		cout << -1 << endl;
		return 0;
	}
	int nop = m / p;
	cout << nop + noq << endl;
	for(int i = 0; i < nop; i++){
		cout << s.substr(i * p, p) << endl;
	}
	for(int i = 0; i < noq; i++){
		cout << s.substr(nop * p + i * q, q) << endl;
	}
	return 0;
}
