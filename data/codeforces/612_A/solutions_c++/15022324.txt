#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,p,q;
	cin>>n>>p>>q;
	string s;
	cin>>s;
	int i,j,k=s.length();i++;
	for(i=0;i<=k/p;i++)
    {
		for(j=0;j<=k/q;j++)
		{
			if(i*p + j*q ==k)
			{
				cout<<i+j<<endl;
				k=0;
				while(i)
				{
					cout<<s.substr(k,p)<<endl;
					k+=p;
					i--;
				}
				while(j)
				{
					cout<<s.substr(k,q)<<endl;
					j--;
					k+=q;
				}
				return 0;
			}
		}
	}
	cout<<"-1"<<endl;
	return 0;
}
