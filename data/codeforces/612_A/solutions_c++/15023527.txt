#include<bits/stdc++.h>
using namespace std;
int main(){
int m,n;
int a;
cin>>a>>m>>n;
string p;
cin>>p;
int x=0;
while(a%n!=0 and a>0){
    a=a-m;x++;
}
int y=a/n;
if(a%n!=0 or a<0){cout<<-1<<endl;return 0;}
cout<<x+y<<endl;
for(int i=0;i<m*x;i++){
    cout<<p[i];
    if((i+1)%m==0){cout<<endl;}
}
for(int i=0;i<n*y;i++){
    cout<<p[i+m*x];
    if((i+1)%n==0){cout<<endl;}
}
}
