#include<bits/stdc++.h>
using namespace std;

int n,p,q,te,tes,re;
string s;

int main(){
cin>>n>>p>>q>>s;re=-1;
for(int i=0;i*p<=n;i++){
    te=n-i*p;
    tes=te/q;
    if(tes*q==te) {re=i;break;}
}
if(re==-1) cout<<re<<endl;
else{
    cout<<(re+tes)<<endl;
    for(int i=1;i<=re;i++) cout<<s.substr((i-1)*p,p)<<endl;
    for(int i=1;i<=tes;i++) cout<<s.substr(re*p+(i-1)*q,q)<<endl;
}
return 0;
}
