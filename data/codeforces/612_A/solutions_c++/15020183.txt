#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,p,q;
    cin>>n>>p>>q;
    string s;
    cin>>s;
    for(int i=0;i<=n/p;i++){
        int j=(n-p*i)/q;
        if(p*i+q*j==n){
            cout<<i+j<<endl;
            for(int k=0;k<i;k++){
                cout<<s.substr(k*p,p)<<endl;
            }
            for(int k=0;k<j;k++){
                cout<<s.substr(i*p+k*q,q)<<endl;
            }
            return 0;
        }
    }
    cout<<-1;
}
