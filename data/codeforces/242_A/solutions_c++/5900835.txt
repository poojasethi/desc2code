#include <cstdio>
int a[10010],b[10010],x,y,n,m,p;
int main(){
	scanf("%d%d%d%d",&x,&y,&n,&m);
	p = 0;
	for (int i = n;i <= x;i++)
		for (int j = m;j <= y && j < i;j++){
			p++;
			a[p] = i;
			b[p] = j;
		}
	printf("%d\n",p);
	for (int i = 1;i <= p;i++) printf("%d %d\n",a[i],b[i]);
}
