#include<stdio.h>
#include<string.h>

int t[100005][2];

int main()
{
	int ans=0,x,y,a,b,i,j;
	scanf("%d%d%d%d",&x,&y,&a,&b);
	for(i=a;i<=x;i++)
		for(j=b;j<=y;j++)
			if(i>j)
			{
				t[ans][0]=i;
				t[ans++][1]=j;
			}
	printf("%d\n",ans);
	
		for(i=0;i<ans;i++)
			printf("%d %d\n",t[i][0],t[i][1]);
}
