#include <iostream>
#include <algorithm>
using namespace std;
typedef pair<int,int> PII;
int x,n,m,t,same;
PII p[200010];
long long ans=1;
int main(){
    cin>>n;
    for(int i=0;i<2*n;i++){
        cin>>x;
        p[i]=PII(x,i%n);
    }
    cin>>m;
    sort(p,p+2*n);
    int i=0,j=0;
    for(;i<2*n;i=++j){
        while(p[j+1].first==p[j].first)
            if(p[++j].second==p[j-1].second) same++;
        for(int k=2;k<=j-i+1;k++){
            t=k;
            while(!(t&1) && same) t/=2,same--;
            ans=(ans*t)%m;
        }
    }
    cout<<ans;
}
