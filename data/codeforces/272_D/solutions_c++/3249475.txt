#include <iostream>
#include <algorithm>
using namespace std;

const int MAXN=2*1000*100+10;
int a[MAXN];

int main(){
    int n, m;
    cin>>n;
    int eq=0;
    for (int i=0; i<n; i++)
        cin>>a[i];
    for (int i=0; i<n; i++){
        cin>>a[i+n];
        if (a[i]==a[i+n])
            eq++;
    }
    cin>>m;
    sort(a, a+2*n);
    int num=1;
    long long ans=1;
    for (int i=0; i<2*n; i++){
        if (a[i-1]==a[i]){
            num++;
            ans*=num;
            while (eq && !(ans%2)){
                eq--;
                ans/=2;
            }
            ans%=m;
        }
        else
            num=1;
    }
    cout<<ans<<endl;
    return 0;
}
