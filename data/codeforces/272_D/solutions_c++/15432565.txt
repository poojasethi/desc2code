#include<bits/stdc++.h>

using namespace std;

#define LL long long
#define mod 1000000007

int a[100000], b[100000];
map<LL,LL> mp;
map<LL,LL>::iterator it;

int main()
{
    int n,i,c,m,j;

    c = 0;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>a[i];
        mp[a[i]]++;
    }
    for(i=0;i<n;i++)
    {
        cin>>b[i];
        mp[b[i]]++;
        if(a[i]==b[i])
            c++;
    }
    cin>>m;

    LL ans = 1;
    for(it = mp.begin();it!=mp.end();it++)
        for(i=1;i<=it->second;i++)
        {
            j = i;
            while(c && j%2==0)  j/=2,c--;
            ans = (ans*j)%m;
        }
    cout<<ans;
    return 0;
}
