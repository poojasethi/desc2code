#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
using namespace std;
int a[200010];
int main()
{
	int n,m,num=0;
	__int64 ans=1;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a[i+n]);
		if(a[i]==a[i+n]) num++;
	}
	scanf("%d",&m);
	sort(a,a+2*n);
	int c=1;
	for(int i=1;i<2*n;i++)
	{
		if(a[i]==a[i-1]) 
		{
			c++;
			ans=ans*c;
			while(num>0&&ans%2==0)
			{
				ans=ans/2;
				num--;
			}
			ans=ans%m;
		}
		else c=1;
	}
	printf("%I64d\n",ans);
	return 0;
}
