#include <cstdio>
#define N 10001
#define M (N*50)
int n, d, a;
bool go[M][51];

int main()
{
	scanf("%d%d", &n, &d);
	go[0][0] = 1;
	for (int i = 0; i < n; ++i)
	{
		scanf("%d", &a);
		for (int j = a; j < M; ++j)
			go[j][i+1] |= go[j-a][i];
		for (int j = 0; j < M; ++j)
			go[j][i+1] |= go[j][i];
	}
	int now = 0, res = 0, g = 1, p, u;
	while (1)
	{
		u = -1;
		for (p = now+1; p < M && p-now <= d; ++p)
			if (go[p][n]) u = p;
		if (u == -1) break;
		now = u; ++res;
	}
	printf("%d %d\n", now, res);
}
