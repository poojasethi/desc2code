#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
#define N 600000
int n,d,f[N];
int main(){
	scanf("%d%d",&n,&d);
	f[0]=1;int now=0,times=0;
	for(int x,i=1;i<=n;i++){
		scanf("%d",&x);
		for(int j=N-1;j>=x;j--)f[j]|=f[j-x];
	}while(1){
		int old=now;
		for(int j=1;j<=d;j++)
			if(f[old+j])now=j+old;
		if(now==old)break;
		times++;
	}cout<<now<<" "<<times<<endl;
}
