#include<cstdio>
using namespace std;
int n,t,flag,k,sum,a[51],ans;
bool d[500001];
int main()
{
	int i,j;
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++)
	  scanf("%d",&a[i]);
	d[0]=true;
	for(i=1;i<=n;i++)
	{
		sum=sum+a[i];
		for(j=sum;j>=0;j--)
		  if(d[j])
		    d[j+a[i]]=true;
	}
	while(1)
	{
		for(j=t+k;j>t;j--)
		if(d[j])
		{
			ans++;
			flag=j;
			break;
		}
		if(flag>=sum||flag==t)
		  break;
		t=flag;	  
	}
	printf("%d %d\n",flag,ans);
}
