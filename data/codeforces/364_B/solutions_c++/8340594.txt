#include<cstdio>
using namespace std;
int f[550010],n,a,d,sum;
int main(){
	scanf("%d%d",&n,&d);
	f[0]=1;
	for (int i=1;i<=n;i++){
		scanf("%d",&a);
		sum+=a;
		for (int j=sum;j>=a;j--)
			if (f[j-a]) f[j]=1;
	}
	int ans=0,sum=0;
	while (1){
		int i;
		sum+=d;
		for (i=0;i<d;i++)
			if (f[sum-i]) 
				break;
		if (i==d) break;
		sum-=i,ans++;
	}
	printf("%d %d",sum-d,ans);
	return 0;
}
