#include<stdio.h>
int dp[600008];

int main()
{
    int i,n,d,x,cnt=0,sum=0,w=0;
    scanf("%d%d",&n,&d);
    dp[0]=1;
    for(i=1;i<=n;i++)
    {
        scanf("%d",&x);
        for(int j=(sum+=x);j>=x;j--)
        {
            if(dp[j-x]) dp[j]=1;
        }
    }
    while(true)
    {
        int j=w+d;
        while(!dp[j]&&j>w) j--;
        if(j==w) break;
        w=j;cnt++;
    }
    printf("%d %d\n",w,cnt);
    return 0;
}
