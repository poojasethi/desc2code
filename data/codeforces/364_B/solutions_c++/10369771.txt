#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int dp[600005],a[55];
int main()
{
  int sum,i,j,n,d,ans,t1,t2;
  memset(dp,0,sizeof(dp));
  dp[0]=1; sum=0;
  scanf("%d%d",&n,&d);
  for (i=1;i<=n;i++)
  {
	scanf("%d",&a[i]);
	sum+=a[i];
	for (j=sum;j>=a[i];j--)
	  dp[j]=dp[j]||dp[j-a[i]];
  }
  ans=0; t1=0;
  while (1)
  {
	t2=t1;
	for (i=t2;i<=t2+d;i++)
	  if (dp[i]) t1=i;
	if (t1==t2) break;
	else ans++;
  }
  printf("%d %d\n",t1,ans);
  return 0;
}
