#include<iostream>
using namespace std;
bool comb[500001];
int ln[500001], c, d, n, nneed, cur;
int main() {
	comb[0]=true;
	cin >> n >> d;
	for(int i=0;i<n;i++) {
		cin >> c;
		for(int j=500000 - c;j>=0;j--)
			if(comb[j])
				comb[j+c] = true;
	}
	for(int i=0;i<=d;i++)
		if(comb[i])
			ln[0]=i;
	for(int i=1;i<=500000;i++) {
		ln[i] = ln[i-1];
		if(i+d <= 500000 && comb[i+d])
			ln[i] = i+d;
	}
	while(ln[cur] > cur){
		nneed++;
		cur = ln[cur];
	}
	cout << cur << ' ' << nneed << '\n';
	return 0;
}
