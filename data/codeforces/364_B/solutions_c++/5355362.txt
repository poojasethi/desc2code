#include <cstdio>
#define M 550005
int a[55],N,D,g[M];
bool f[M];
int main()
{
	scanf("%d%d",&N,&D),f[0]=1;
	for (int i=1,x; i<=N; i++)
	{
		scanf("%d",&x);
		for (int j=M-1; j>=x; j--) if (f[j-x]) f[j]=1;
	}
	for (int i=1; i<M; i++) g[i]=f[i]?i:g[i-1];
	for (int i=1,x=0;;i++) if (g[x+D]==x) {printf("%d %d\n",x,i-1); break;} else x=g[x+D];
}
