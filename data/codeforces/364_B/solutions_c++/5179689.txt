#include <cstdio>
int n,d,x,max,i,j,ans;
bool f[5100000];
int main() {
	scanf("%d%d",&n,&d);
	f[0]=true;
	for (i=1;i<=n;i++) {
		scanf("%d",&x);
		for (j=(max+=x);j>=x;j--)
			if (f[j-x]) f[j]=true;
	}
	x=0;
	while (1) {
		j=ans+d;
		while (!f[j]&&(j>ans)) j--;
		if (j==ans) break;
		ans=j;x++;
	}
	printf("%d %d\n",ans,x);
	return 0;
}
