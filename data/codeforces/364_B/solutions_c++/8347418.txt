#include<iostream>
using namespace std;
int ans,n,d,sum,a[60],f[5000010];
int main(){
	cin>>n>>d;
	for (int i=1;i<=n;++i) cin>>a[i],sum+=a[i];
	f[0]=1;
	for (int i=1;i<=n;++i)
		for (int j=sum;j>=a[i];--j)
			f[j]=f[j]|f[j-a[i]];
	sum=0;
	while (1){
		int i;
		sum+=d;
		for (i=0;i<d;++i)
			if (f[sum-i]) break;
		if (i==d) break;
		sum-=i;
		ans++;
	}
	cout<<sum-d<<' '<<ans<<endl;
	return 0;
}
