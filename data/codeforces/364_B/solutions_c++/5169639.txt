#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
const int maxm=52*10000;
int x,n,d,g[maxm+10],ans,fans,i,j;
bool f[maxm+10];
int main(){
	scanf("%d%d",&n,&d);
	f[0]=1;
	for(i=1;i<=n;++i){
		scanf("%d",&x);
		for(j=maxm-x;j>=0;--j)if(f[j])
			f[j+x]=1;
	}
	for(i=0;i<=maxm;++i)if(f[i])
		g[i]=i;
	else g[i]=g[i-1];
	ans=fans=0;
	for(;;){
		if(ans==g[ans+d])break;
		ans=g[ans+d];
		fans++;
	}
	printf("%d %d\n",ans,fans);
}
