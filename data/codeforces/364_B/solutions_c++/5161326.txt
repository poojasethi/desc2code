#include <bits/stdc++.h>

using namespace std;

int dp[55 * 10010];
int c[55];
const int MAX_N = 520000;

int main() {
	int n, d;
	while (cin >> n >> d) {
		memset(dp, 0, sizeof(dp));
		for (int i = 0; i < n; i++) {
			scanf("%d", c + i);
			for (int j = MAX_N; j >= c[i]; j--) {
				dp[j] = max(dp[j], dp[j - c[i]] + c[i]);
			}
		}
		int day = 0, mm = 0;
		while (mm <= MAX_N) {
			if (dp[mm + d] > mm) {
				day++;
				mm = dp[mm + d];
			} else {
				break;
			}
		}
		printf("%d %d\n", mm, day);
	}
	return 0;
}
