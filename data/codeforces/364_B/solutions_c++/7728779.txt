#include<stdio.h>
#define M 600000
int c[50];
bool dp[M+1];

int main(){
    int n,d,i,j;
    scanf("%d %d",&n,&d);
    for(i=0;i<n;i++)scanf("%d",&c[i]);
    for(i=0;i<=M;i++)dp[i]=false;
    dp[0]=true;
    for(i=0;i<n;i++){
        for(j=M;j>=c[i];j--)dp[j]|=dp[j-c[i]];
    }
    int s=0;
    int t=0;
    while(1){
        for(i=d;i>0;i--)if(dp[s+i])break;
        if(i==0)break;
        s=s+i;
        t++;
    }
    printf("%d %d\n",s,t);
    return 0;
}
