#include <iostream>
using namespace std;

int n,d;
int c[50];
int posso[500001];

int mas, gio;

int main(){
	cin >> n >> d;
	for(int i=0; i<n; i++) cin >> c[i];
	
	posso[0] = 1;
	
	for(int i=0; i<n; i++){
		for(int x=500000-c[i]; x>=0; x--) if(posso[x]) posso[x+c[i]]=1;
	}
	
	gio = 0;
	mas = 0;
	while(1){
		int i=mas+d; if(i>500000) i=500000;
		while(i>mas && !posso[i]) i--;
		if(i==mas){cout << mas << ' ' << gio; return 0;}
		mas = i;
		gio++;
	}
	
	
	return 0;
}
