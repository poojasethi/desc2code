#include <cstdio>
const	int	maxn = 55;
const	int	maxd = 550000;
int	n, d, sum, ans, res;
int	c[maxn], dp[maxd];
int	main()
{
	scanf("%d%d", &n, &d); sum = 0;
	for (int i=1; i<=n; i++) scanf("%d", &c[i]), sum += c[i];
	for (int i=1; i<=n; i++)
		for (int j=sum-c[i]; j>=0; j--) if (dp[j] == j) dp[j+c[i]] = j+c[i];
	for (int i=1; i<=sum+d; i++) if (!dp[i]) dp[i] = dp[i-1];
	for (ans=res=0; dp[ans+d] != ans; ans = dp[ans + d]) res++;
	printf("%d %d\n", ans, res);
	return 0;
}
