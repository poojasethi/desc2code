#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;

bool dp[51*10001];

int main()
{
    int n,d,x,cnt=0,sum=0,w=0;
    cin>>n>>d;
    dp[0]=1;
    for(int i=0;i<n;i++)
    {
        cin>>x;
        for(int j=(sum+=x);j>=x;j--)
        {
            if(dp[j-x]) dp[j]=true;
        }
    }
    while(true)
    {
        int j=w+d;
        while(!dp[j]&&j>w) j--;
        if(j==w) break;
        w=j;cnt++;
    }
    cout<<w<<" "<<cnt<<endl;
    return 0;
}
