#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int i,j,k,n,m,tmp,f[600001],s,a[61];
int main(){
	scanf("%d%d",&n,&m);memset(f,0,sizeof(f));
	for (i=1,s=0;i<=n;i++) scanf("%d",&a[i]),s+=a[i];
	for (i=1,f[0]=1;i<=n;i++)
	 for (j=s;j>=a[i];j--) f[j]|=f[j-a[i]];
	for (i=k=0;;k++){
		for (tmp=i,j=i+1;j<=s&&j-i<=m;j++)
		 if (f[j]) tmp=j;
		if (tmp==i) break;
		i=tmp;
	}
	printf("%d %d\n",i,k);
	return 0;
}
