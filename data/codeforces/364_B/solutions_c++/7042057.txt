#include <stdio.h>
const int N=550000;
int n,d,a[N];
int main(){
  scanf("%d%d",&n,&d);
  for(int i=0,x;i<n&&1==scanf("%d",&x);++i)
    for(int j=N;j-->x;)if(a[j-x]==j-x)a[j]=j;
  for(int i=1;i<N;++i)if(a[i]<1)a[i]=a[i-1];
  for(int i=0,x=0;i<N;++i)
    if(x==a[x+d]){
      printf("%d %d\n",x,i);
      break;
    }else x=a[x+d];
}
