#include <iostream>  
 

using namespace std;

bool dp[51 * 10001];

int main()
{
	int n, d, temp, day = 0, sum = 0, w = 0;
	cin >> n >> d;
	dp[0] = 1;
	for (int i = 0; i<n; i++)
	{
		cin >> temp;
		for (int j = (sum += temp); j >= temp; j--)
		{
			if (dp[j - temp]) dp[j] = true;
		}
	}
	while (true)
	{
		int j = w + d;
		while (!dp[j] && j>w) j--;
		if (j == w) break;
		w = j; day++;
	}
	cout << w << " " << day << endl;
	return 0;
}
