#include<cstdio>
# define maxn 550000
using namespace std;
bool f[maxn];
int main(){
		int n,d,sum=0;
		scanf("%d%d",&n,&d);
		f[0]=1;
		for (int i=1;i<=n;i++){
				int a;
				scanf("%d",&a);
				sum+=a;
				for (int j=sum;j>=a;j--)	
					if (f[j-a]!=0) f[j]=1;
			}
		int ans=0;sum=0;
		while(1){
				int i;
				sum=sum+d;
				for (i=0;i<d;i++) if (f[sum-i]!=0) break;
				if (i==d) break;
				sum=sum-i; ans++;
			}
		printf("%d %d\n",sum-d,ans);
		return 0;
	}
