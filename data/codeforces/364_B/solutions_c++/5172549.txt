#include <iostream>
using namespace std;
bool price[51][500001];
int main(){
	int n,d,a[50];
	cin >> n >> d;
	for(int i=0;i<n;i++) cin >> a[i];
	price[0][0]=true;
	for(int i=0;i<n;i++){
		for(int j=0;j<=n*10000;j++){
			if(price[i][j]){
				price[i+1][j]=true;
				price[i+1][j+a[i]]=true;
			}
		}
	}
	int now=0,cnt=0;
	for(;;cnt++){
		int nowc=now;
		for(int i=min(now+d,500000);i>now;i--){
			if(price[n][i]) now=i;
		}
		if(nowc==now) break;
	}
	cout << now << " " << cnt << endl;
	return 0;
}
