#include<cstdio>
#include<cstring>
#include<bitset>
#include<algorithm>
using namespace std;
typedef long long ll;
#define rep(i,n) for (int i=0;i<n;++i)
int n,d,s,ans,a[55];
bitset<500005> b;
int main()
{
	scanf("%d%d",&n,&d);
	rep(i,n) scanf("%d",a+i); sort(a,a+n);
	rep(i,n) if (s+d<a[i]){n=i; break;} else s+=a[i];
	b[0]=1; rep(i,n) b|=b<<a[i];
	int x=0,y=0,j=0;
	while (x<s){
		while (j<=x+d) {if (b[j]) y=j; ++j;}
		x=y,++ans;
	}
	printf("%d %d\n",x,ans);
	return 0;
}
