#include<iostream>
using namespace std;

int main()
{
	int n,x=1;
	int a[200001];
	int count[200001];
	
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		count[a[i]]++;
	}
	
	for(int i=0;i<n;i++)
	{
		while(count[x]>0)
		x++;
		
		if( count[a[i]]>1 || a[i]>n )
		{
			count[a[i]]--;
			count[x]++;
			a[i]=x;
		}
	}
	
	for(int i=0;i<n;i++)
	{
		cout<<a[i]<<" ";
	}
	
	return 0;
}
