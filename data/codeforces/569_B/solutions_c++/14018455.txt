#include<stdio.h>   
int n;
int main(){
    while(~scanf("%d",&n)){
     int p[100001]={0},d[100001],k=1;
     for(int i=0;i<n;i++) scanf("%d",&d[i]),p[d[i]]++;
     for(int i=0;i<n;i++){
        if(p[d[i]]==1&&d[i]<=n) printf("%d ",d[i]); 
        else{
            while(p[k]) k++;
            printf("%d ",k),p[k]++;
            p[d[i]]--;
         }
       }
       printf("\n");
    }
}
