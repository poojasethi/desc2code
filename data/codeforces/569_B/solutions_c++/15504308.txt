#include<stdio.h>
#include<string.h>
int n,i,j,cnt=1;
int a[101000];
int b[101000];
int main(){
	memset(b,0,sizeof(b));
	scanf("%d",&n);
	for(i=1;i<n+1;i++){
		scanf("%d",&a[i]);
		b[a[i]]++;
	}
	for(i=1;i<n+1;i++){
		if(b[a[i]]==1&&a[i]<n+1){
			printf("%d ",a[i]);
		}
		else{
			while(b[cnt]){
				cnt++;
			}
				b[a[i]]--;
				printf("%d ",cnt);
				b[cnt]++;
			}
		}
	return 0;
	}
