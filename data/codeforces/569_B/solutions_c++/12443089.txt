#include <iostream>
using namespace std;
long long n,i,v[100010],ap[100010],x[100010],k;
int main(){
    cin>>n;
    for(i=1;i<=n;++i){
            cin>>v[i];
    ap[v[i]]++;}
    for(i=1;i<=n;++i)
        if(!ap[i])x[++k]=i;
    k=1;
    for(i=1;i<=n;++i)
        if(ap[v[i]]>1||v[i]>n)ap[v[i]]--,v[i]=x[k],k++;
        for(i=1;i<=n;++i)
            cout<<v[i]<<" ";
        return 0;}
