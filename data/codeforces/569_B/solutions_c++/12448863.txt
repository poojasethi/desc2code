using namespace std;
int main()
{
	long long int n,i,j=1;;
	cin>>n;
	vector<long long int> a(n,0),b(100001,0);
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		b[a[i]]++;
	}
	for(i=0;i<n;i++)
	{
		if(b[a[i]]>1||a[i]>n)
		{
			while(b[j]!=0)
				j++;
			
			b[a[i]]--;
			a[i]=j;
			b[j]++;
			
		}
		
	}
	for(i=0;i<n;i++)
	cout<<a[i]<<" ";
}
