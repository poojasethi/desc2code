#include <stdio.h>
int x,b[100002],check[100002],kuy[100002];
int main()
{
    int n,c=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++){scanf("%d",&x);
        if(check[x] == 1 || x>n)
    	{
    		kuy[c++]=i;
    	}
    	check[x]=1;
        b[i]=x;
  
	}
    int cc=0;
    for(int i=0;i<n;i++)
    {
    	if(check[i+1] == 0)
           b[kuy[cc++]] = i+1;

    }
  for(int i=0;i<n;i++)printf("%d ",b[i]);
  	return 0;

}
