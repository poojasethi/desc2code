#include<iostream>
using namespace std;
 int a[100009],j,b[100009];
int main()
{
int n,i;
cin>>n;

for(i=0;i<=n;i++)
b[i]=0;
for(i=0;i<n;i++)
{
cin>>a[i];
b[a[i]-1]++;
}
int f=0;
for(i=0;i<n;i++)
{
    if(a[i]<=n && b[a[i]-1]==1)
        cout<<a[i]<<" ";
else
{
while(b[f]>0)
f++;
cout<<f+1<<" ";
f++;
b[a[i]-1]--;
}
}
return 0;
}
