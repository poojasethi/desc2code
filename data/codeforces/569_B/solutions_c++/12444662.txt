#include<stdio.h>
#include<stdlib.h>
int n,a[100500],b[100500];
int main(){
	scanf("%ld",&n);
	for(int i=1;i<=n;++i){
		scanf("%ld",&a[i]);
		if(a[i]<1 || a[i]>n || b[a[i]]){
			a[i]=0;
		}else b[a[i]]=1;
	}
	for(int i=1,j=1;i<=n;++i)
		if(!a[i]){
			while(j<=n && b[j])++j;
			a[i]=j++;
		}
	for(int i=1;i<=n;++i){
		printf("%ld",a[i]);
		printf(i<n?" ":"\n");
	}
	return 0;
}
