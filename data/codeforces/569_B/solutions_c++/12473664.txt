#include <iostream>
#include <vector>
using namespace std;
typedef vector<int> ve;
int main()
{
	int n;
	cin >>n;
	ve v(n);
	ve s(n,false);
	for (int i=0;i<n;i++)
	{
		cin >>v[i];
		--v[i];
		if (v[i]>=n or s[v[i]]) v[i]=-1;
		else s[v[i]]=true;
	}
	int j=0;
	for (int i=0;i<n;i++)
	if (v[i]==-1){
		while (s[j])++j;
		v[i]=j++;
	}
	for (int i=0;i<n;++i)cout <<(i ?" ":"")<<v[i]+1;
}
