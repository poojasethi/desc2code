#include<cstdio>

int n, p, v;
int a[111111], occ[111111], rem[111111];

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%d", &a[i]);
		if (a[i] <= n && !occ[a[i]])
			occ[a[i]] = rem[i] = true;
	}
	p = 1;
	for(int i=0;i<n;i++) {
		if (rem[i]) v = a[i];
		else {
			while (occ[p]) p++;
			v = p;
			occ[p] = true;
		}
		printf("%d%c", v, i<n-1 ? ' ' : '\n');
	}
	
	return 0;
}
