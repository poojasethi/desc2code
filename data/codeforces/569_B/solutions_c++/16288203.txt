#include<stdio.h>
#include<string.h>

int a[100000],vis[100001];

int main()
{
	int n,cur=1,i;
	scanf("%d",&n);
	memset(vis,0,sizeof(vis));
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		vis[a[i]]++;
	}
	for(i=0;i<n;i++)
	{
		if(vis[a[i]]==1 && a[i]<=n)
			printf("%d ",a[i]);
		else{
			while(vis[cur]) cur++;
			vis[a[i]]--;
			printf("%d ",cur);
			vis[cur]++;
		}
	}
	puts("");
	return 0;
}
