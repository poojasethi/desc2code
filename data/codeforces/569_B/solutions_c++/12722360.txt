#include<bits/stdc++.h>
using namespace std;

int main()
{
	int i,st,used[100001],ar[100000],n;
	cin >> n;
	
	for(i=1;i<=n;i++)
		used[i] = 0;
	for(i=0;i<n;i++)
	{
		cin >> ar[i];
		used[ar[i]] += 1;
	}
	st = 1;
	for(i=0;i<n;i++)
	{
		while(used[st] != 0 && st<=n)
			st++;
		if(used[ar[i]] > 1 || ar[i] > n)
		{
			used[ar[i]] --;
			ar[i] = st;
			used[st++] = 1;
		}
		cout << ar[i] << " ";		
	}
	cout << "\n";
	return 0;
}
