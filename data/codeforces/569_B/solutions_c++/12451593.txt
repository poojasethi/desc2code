#include <iostream>
using namespace std;

bool b[100005]{};
int c[100005];

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> c[i];
		if (b[c[i]] || c[i] > n)
			c[i] = 0;
		else b[c[i]] = true;
	}
	int j = 1;
	for (int i = 0; i < n; i++)
	{
		if (c[i] == 0)
		{
			while (b[j]) j++;
			c[i] = j;
			b[j] = true;
		}
		cout << c[i] << ' ';
	}
//  system("pause");
	return 0;
}
