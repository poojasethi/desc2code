#include<stdio.h>
int main(){

	int i,n, b[100001]={0}, a[100001];
	scanf("%d",&n);
	for( i=0; i < n ;i++){
		scanf("%d",&a[i]);
		if( b[a[i]] == 1 || a[i] > n )
			a[i]=0;
		else 
			b[a[i]] =1;
	}
	int k =1;
	for(i=0; i < n;i++){
		if( a[i] != 0 )
			printf("%d ",a[i]);
		else{
			while( b[k] != 0 && k < n)
				k++;
			printf("%d ", k++);
		}
	}
	return 0;
}
