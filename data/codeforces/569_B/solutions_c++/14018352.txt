#include<stdio.h>
int n,a[100001],i;
int d[100001],k=1;
int main(){
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
        d[a[i]]++;
    }
    for(i=0;i<n;i++){
        if(d[a[i]]==1&&a[i]<=n){
            printf("%d ",a[i]);
        }
        else{
            while(d[k])k++;
            d[a[i]]--;
            printf("%d ",k);
            d[k]++;
        }
    }
}
