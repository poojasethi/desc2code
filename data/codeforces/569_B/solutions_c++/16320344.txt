#include <bits/stdc++.h>
using namespace std;
const int N = 1e5+1;
int ar[N],cc[N];
int main(){
	int n,i,j;cin>>n;
	vector<int> per;
	for(i=0;i<n;++i){
		scanf("%d",&ar[i]);
		cc[ar[i]]++;
		if(cc[ar[i]]>=2 || ar[i]>n)
			per.push_back(i);
	}
	for(i=1;i<=n;++i){
		if(cc[i])continue;
		ar[per.back()]=i;
		per.pop_back();
	}
	for(i=0;i<n;++i)
		printf("%d ",ar[i]);
	putchar('\n');
}
