#include <bits/stdc++.h>
using namespace std;
int a[100005],b[100005];
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;++i)
		cin>>a[i],b[a[i]]++;
	int x=1;
	for(int i=1;i<=n;++i){
		while(b[x])
			x++;
		if(a[i]>n||b[a[i]]>=2){
			b[x]++;
			b[a[i]]--;
			a[i]=x;
		}
	}
	for(int i=1;i<=n;++i){
		if(i==n)
			printf("%d\n",a[i]);
		else
			printf("%d ",a[i]);
	}
	return 0;
}
