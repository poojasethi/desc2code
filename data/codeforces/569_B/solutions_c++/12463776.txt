#include <iostream>
using namespace std;
int n,m,r;
int t[100000]={0},a[100000]={0},v[100000]={0};
int main()
{
	cin >> n;
	for (int i=1;i<=n;i++)
	{
		cin >> a[i];
		if ((a[i]<=n) && (v[a[i]]==0))
			v[a[i]]=1;
		else
		{
			m++;
			t[m]=i;
		}
	}
	for (int i=1;i<=n;i++)
		if (v[i]==0)
		{
			r++;
			a[t[r]]=i;
		}
	for (int i=1;i<=n;i++)
		cout << a[i] << " ";
}
