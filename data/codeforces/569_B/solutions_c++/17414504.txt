#include <iostream>
using namespace std;
int a[100005], d[100003], n, k = 1;
int main(){
	cin >> n;
	for (int i = 0; i < n; i ++)
	    cin >> a[i], d[a[i]] ++;
	for (int i = 0; i < n; i ++){
		if(d[a[i]] == 1 && a[i] <= n){
			cout << a[i] << ' ' ;
		} else {
			while (d[k]) k ++;
			d[a[i]] --;
			d[k] ++;
			cout << k << ' ';
		}
	}
}
