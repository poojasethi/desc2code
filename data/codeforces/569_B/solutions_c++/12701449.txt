#include<iostream>
using namespace std;
int main()
{
int i,j,k,l,m,n,o,arr[100001],a[100001]={0},b[100001]={0};
cin>>n;
for(i=1;i<=n;i++)
{
    cin>>arr[i];
    a[arr[i]]++;
}
j=1;
for(i=1;i<=n;i++)
{
    if(b[arr[i]]==0&&arr[i]<=n)
    {
        cout<<arr[i]<<endl;
        b[arr[i]]=1;
    }
    else
    {
        while(a[j]!=0)
            j++;
        cout<<j<<endl;
        j++;
    }

}


}
