#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,a[100005],i,b[100005],j=0,hash[100005]={0},k=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		hash[a[i]]++;
	}
	for(i=1;i<=n;i++)
		if(hash[i]==0)
			b[j++]=i;
	for(i=0;i<n;i++)
	{
		if(hash[a[i]]>1||a[i]>n)
		{
			hash[a[i]]--;
			a[i]=b[k++];
		}
	}
	for(i=0;i<n;i++)
		cout<<a[i]<<" ";
	return 0;
}
