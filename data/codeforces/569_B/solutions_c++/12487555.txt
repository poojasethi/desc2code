#include <bits/stdc++.h>
using namespace std;
int a[100005];
int mark[100005];
int main() {
	int N;
	cin>>N;
	for (int i=1; i<=N; i++){
		cin>>a[i];
		mark[a[i]]++;
	}
	int p = 1;
	for (int i = 1; i <= N; i++){
		if (mark[a[i]] == 1 && a[i] <= N){
			cout<<a[i]<<" ";
		}
		else {
			mark[a[i]]--;
			while (mark[p] != 0){
				p++;
			}
			cout<<p<<" ";
			mark[p]++;
		}
	}
}
