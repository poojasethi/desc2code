#include<iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int a[n], b[100000+1]={};
	for(int i=0;i<n;i++){
		cin>>a[i];
		if(a[i]<=n && b[a[i]]!=1) b[a[i]] = 1;
		else a[i] = 0;
	}
	int i=0, j=1;
	while(i<n){
		while(i<n && a[i]!=0){
			cout<<a[i]<<" ";
			i++;
		}
		while(j<=n && b[j]==1) j++;
		a[i] = j;
		j++;
	}
	return 0;
}
