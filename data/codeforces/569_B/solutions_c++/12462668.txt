#include<stdio.h>
int i,j=1,n,arr[200001],H[200001];
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		scanf("%d",&arr[i]);
		H[arr[i]]++;
	}
	for(i=1;i<=n;i++){
		while(H[j]) j++;
		
		if(arr[i] > n || H[arr[i]] > 1){
			H[arr[i]]--;
			H[j]++;
			arr[i]=j;
		}
	}
	for(i=1;i<=n;i++)
		printf("%d ",arr[i]);
}
