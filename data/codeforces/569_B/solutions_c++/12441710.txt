#include <iostream>
using namespace std;

const int N=100002;
int a[N], b[N], c[N];
int n;
int main()
{
	int i,j,k;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<n;i++)
		b[a[i]]++;
	k=1;
	for(i=0;i<n;i++){
		j=a[i];
		if(c[j]==0 && j<=n)
			c[j]=1;
		else {
			while(b[k])
				k++;
			a[i]=k;
			b[k]=1;
		}
	}
	for(i=0;i<n;i++)
		cout<<a[i]<<' ';
	return 0;
}
