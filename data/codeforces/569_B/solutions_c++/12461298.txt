#include<stdio.h>
#include<string.h>
int count[100100];
int main(){
	memset(count,0,sizeof(count));
	int n;
	scanf("%d",&n);
	int a[n];
	for(int i=0;i<n;i++) {
		scanf("%d",&a[i]);
		count[a[i]]++;
	}
	int j=1;
	for(int i=0;i<n;i++){
		while(count[j]) j++;
		if(count[a[i]]>1||a[i]>n) {
			count[j]++;
			count[a[i]]--;
			a[i]=j;
		}
	}
	for(int i=0;i<n;i++) printf("%d ",a[i]);
}
