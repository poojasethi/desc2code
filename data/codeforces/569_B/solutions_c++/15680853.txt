#include <iostream>
#include <algorithm>

using namespace std;

int A[100100],K[100100],R[100100],r,N;

int main(){
	cin >> N;
	for(int i = 0;i<N;i++){
		cin >> A[i];
		K[A[i]]++;
		if(A[i] > N) K[A[i]]++;
	}
	for(int i = 1;i<=N;i++){
		if(K[i] == 0) R[++r] = i;
	}
	for(int i = 0;i<N;i++){
		if(K[A[i]] > 1) K[A[i]]--,A[i] = R[r--];
		cout << A[i] << ' ';
	}
	
	

	return 0;
}
