#include<iostream>
using namespace std;
int i,j,n,x[100001],b[100001],a[100001];
main()
{
   cin>>n;
   for(i=1; i<=n; i++)
   cin>>x[i];
   for(i=1; i<=n; i++)
     if(a[x[i]]==0 && x[i]<=n)
      a[x[i]]=i;
   int j=1;
   for(i=1; i<=n; i++)
   if(a[x[i]]!=i || x[i]>n)
    {while(a[j]!=0) j++;
     a[j]=i;}
   for(i=1; i<=n; i++)
      b[a[i]]=i;
   for(i=1; i<=n; i++)
    cout<<b[i]<<" ";
}
