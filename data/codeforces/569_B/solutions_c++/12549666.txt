#include<bits/stdc++.h>
using namespace std;

int main ()
{
 int n, i;
 cin >> n;
 int a[100010], d[100010]={0};
 for (i=0; i<n; i++)
 {
  cin >> a[i];
  d[a[i]]++;
 }
 
 int k=1;
 for (i=0; i<n; i++)
 {
  while (d[k])
    k++;
  if (d[a[i]]==1 && a[i]<=n)
     cout << a[i] << " ";
  else
     {
      cout << k << " ";
      d[k]++;
      d[a[i]]--;
     }
 }
 return 0;
}
