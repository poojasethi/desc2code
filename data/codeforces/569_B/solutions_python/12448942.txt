n = int(raw_input())
arr = map(int, raw_input().split())

present = [True] + [False] * n

for a in arr:
	if a <= n:
		present[a] = True

s = set()
p = 1
for i in xrange(len(arr)):
	if arr[i] > n or arr[i] in s:
		while present[p]:
			p += 1
		arr[i] = p
		present[p] = True
	s.add(arr[i])

print " ".join(map(str,arr))
