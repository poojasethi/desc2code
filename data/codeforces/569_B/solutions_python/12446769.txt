from sys import stdin

n = int(stdin.readline())
occurences = {}
unseen = []

nums = map(int, stdin.readline().split())

for num in nums:
    if num in occurences:
        occurences[num] += 1
    else:
        occurences[num] = 1
        
for x in xrange(1,n+1):
    if not x in occurences:
        unseen.append(x)
        
for x in xrange(n):
    num = nums[x]
    if (num in occurences and occurences[num] > 1) or num > n:
        nums[x] = unseen.pop()        
        occurences[num] -= 1
        
print " ".join(str(num) for num in nums)
