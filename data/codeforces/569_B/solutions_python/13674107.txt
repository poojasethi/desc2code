def solve(n, items):
	used = [0 for i in range(n)]
	indicies = []
	for i in range(len(items)):
		if items[i] > n or used[items[i] - 1] >= 1:
			indicies.append(i)
		if items[i] <= n:
			used[items[i] - 1] += 1
	for i in range(len(used)):
		if used[i] == 0:
			items[indicies[-1]] = i + 1
			indicies.pop()
	return items

n = int(raw_input())
items = map(int, raw_input().split())
print " ".join(map(str, solve(n, items)))
