count = int(raw_input())
value = map(int, raw_input().split())
inventory = set(range(1,count+1))
to_replace = []
for i in range(count):
	val = value[i]
	if val in inventory:
		inventory.remove(val)
	else:
		to_replace.append(i)
for i in to_replace:
	value[i] = inventory.pop()

print " ".join(map(str,value))
