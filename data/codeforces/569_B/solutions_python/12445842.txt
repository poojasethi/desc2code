n = int(raw_input())
a = map(int, raw_input().split())

numbers = set(range(1, n+1))
bad_positions = set()

for i, x in enumerate(a):
    if x in numbers:
        numbers.remove(x)
    else:
        bad_positions.add(i)

for i in xrange(n):
    if i in bad_positions:
        a[i] = numbers.pop()

print ' '.join(map(str, a))
