#!/usr/bin/python

def readints(): return map(int, raw_input().split())

n = readints()[0]
a = readints()

reserved = [False] * (n+1)
for aa in a:
    if aa <= n:
        reserved[aa] = True

output = []
nextAvail = 1

used = [False] * (n+1)
for aa in a:
    if aa <= n and reserved[aa]:
        # stays in place
        reserved[aa] = False
    else:
        # find next available slot
        while reserved[nextAvail] or used[nextAvail]:
            nextAvail += 1
        aa = nextAvail

    output.append(aa)
    used[aa] = True

print " ".join(map(str, output))
