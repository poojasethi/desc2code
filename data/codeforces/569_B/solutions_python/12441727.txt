n = int(raw_input())
orig = map(int,raw_input().split(' '))

s=set()
used = set(orig)
unused = set([i for i in range(1,n+1)]).difference(used)
unused = list(unused)
unused.sort()

if len(unused)==0:
    print ' '.join(map(str,orig))
else:
    ans=[];
    j=0;
#    print used
#    print unused
    for o in orig:
	if o not in s and o <= n:
	    ans.append(o)
	    s.add(o)
	else:
	    ans.append(unused[j])
	    j+=1
    print ' '.join(map(str,ans))
