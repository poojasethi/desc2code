n = int(raw_input())
s = raw_input()
ii = map(int, s.split())

N = 102345
a = [0 for i in range(N)]
index_q = []

for index, i in enumerate(ii):
    if a[i-1] == 1 or i > n:
        index_q.append(index)
    else:
        a[i-1] = 1

#print index_q

index_i = 0
for i in range(n):
    if a[i] == 0:
        ii[index_q[index_i]] = i+1
        index_i += 1

print ' '.join(map(str, ii))
