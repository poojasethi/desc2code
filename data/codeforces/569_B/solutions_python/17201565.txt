n=input()
a=map(int,raw_input().split())
used=[0]*(n+1)
used[0]=1
freq=[0]*(10**5+2)
for i in range(n):
	freq[a[i]]+=1
for i in range(n):
	x=a[i]
	if freq[x]>1 or x>n:
		a[i]=-1
		freq[x]-=1
	if x<=n:
		used[x]=1
rem=1
for i in range(n):
	if a[i]==-1:
		while used[rem]!=0:
			rem+=1
		a[i]=rem
		rem+=1
print ' '.join(map(str,a))
