n = int(raw_input())
u = [0] + n * [0]
a = map(lambda x: 1 if int(x) > n else int(x), raw_input().split())
for i in set(a): u[i] = 1
f = sorted(set(range(1, n+1)) - set(a))
fp = 0
for i in range(n):
    if u[a[i]]:
        u[a[i]] = 0
    else:
        a[i] = f[fp]
        fp += 1
