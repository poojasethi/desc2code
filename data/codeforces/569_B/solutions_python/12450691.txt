import sys;
write = sys.stdout.write;

N = int(raw_input());
numbers = [int(i) for i in raw_input().split()];
foo = [0 for i in range(N)];
bar = [0 for i in range(N)];

for i in range(N) :
  if numbers[i] - 1 < N :
    if foo[numbers[i] - 1] == 0 :
       foo[numbers[i]-1] = 1;
       bar[i] = 1;
temp = 0;

for i in range(N) :
  if bar[i] == 1 :
    write(str(numbers[i])+' ');
  else :
    while foo[temp] == 1 :
      temp += 1;
    write(str(temp + 1)+' ');
    temp += 1;
