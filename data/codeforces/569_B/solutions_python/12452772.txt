n = int(raw_input())
values = map(int, raw_input().split())

missing = set(range(1, n + 1))
to_replace = []
for i in range(n):
	val = values[i]
	if val in missing:
		missing.remove(val)
	else:
		to_replace.append(i)

for index in to_replace:
	values[index] = missing.pop()

print " ".join(map(str, values))
