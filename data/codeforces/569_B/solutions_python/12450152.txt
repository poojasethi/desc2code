n = int(raw_input())
a = map(int, raw_input().split())

data = [-1] * n
for i, v in enumerate(a):
    if v > n:
        a[i] = 0
    elif data[v - 1] == -1:
        data[v - 1] = i
    else:
        a[i] = 0

nf = []
for i, d in enumerate(data):
    if d == -1:
        nf.append(i + 1)

p = 0
for i in xrange(n):
    if a[i] > 0:
        print a[i],
    else:
        print nf[p],
        p += 1
