from sys import stdin
n = int(stdin.readline())
a = map(int, stdin.readline().split())
d = [0] * (n+1)
for i in xrange(n):
    if a[i] <= n:
        d[a[i]] += 1
tag = 1
for i in xrange(n):
    while tag <= n and d[tag] > 0:
        tag += 1
    if a[i] > n:
        a[i] = tag
        d[tag] = 1
    elif d[a[i]] > 1:
        d[a[i]] -= 1
        a[i] = tag
        d[tag] = 1
print " ".join(map(str, a))
