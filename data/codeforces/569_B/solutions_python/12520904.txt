n = input()
a = list(map(lambda(x): int(x) - 1, raw_input().split(' ')))
mark = range(n)
ans = [0] * n

for i, v in enumerate(a):
  if v < n:
    if mark[v] != -1:
      mark[v] = -1
      a[i] = -1
      ans[i] = v
cur = 0
mark = list(filter(lambda(x): x != -1, mark))
for i, v in enumerate(a):
  if v != -1:
    ans[i] = mark[cur]
    cur += 1
print ' '.join(map(lambda(x): str(x+1), ans))
