n = int(raw_input())
p = map(int,raw_input().split())

exists = set([])
map(lambda x: exists.add(x),p)

free = []

for i in xrange(1,n+1):
    if not (i in exists):
        free.append(i)


j = 0
exists = set([])
for i in xrange(len(p)):
    if p[i] in exists or p[i] >= n+1:
        p[i] = free[j]
        j +=1
    exists.add(p[i])

print " ".join(map(str,p))
