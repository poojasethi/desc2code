#!/usr/bin/env python

n = int(raw_input())
a = map(int, raw_input().split())

s = set(xrange(1, n+1))

for i in xrange(len(a)):
  if a[i] > 0 and a[i] <= n:
    if a[i] not in s:
      a[i] = None
    else:
      s.remove(a[i])
  else:
    a[i] = None

l = list(s)
for i in xrange(len(a)):
  if a[i] == None:
    a[i] = l.pop()

print ' '.join(map(str, a))
