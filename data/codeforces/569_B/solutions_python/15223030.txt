n = int(raw_input().strip())
nums = [int(a) for a in raw_input().strip().split(" ")]
s = set(xrange(1, n+1))
idxs = []

for i, v in enumerate(nums):
    if v in s:
        s.discard(v)
    else:
        idxs.append(i)

for i, v in zip(idxs, list(s)):
    nums[i] = v

print " ".join(str(a) for a in nums)
