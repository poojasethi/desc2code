n = input()
visited = [False]*(10**5+5)
mark = []
arr = map(int,raw_input().split())

for i in range(n):
    if arr[i] > n or visited[arr[i]]:
        mark.append(i)
    else:
        visited[arr[i]] = True

for i in range(1,n+1):
    if not visited[i]:
        arr[mark[-1]] = i
        del mark[-1]
        visited[i] = True
