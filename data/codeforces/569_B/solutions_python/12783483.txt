n = int(input())
a = [ int(item) for item in raw_input().split(' ')]
b = ["" for i in range(0,n)]
c = []
mark = [0 for i in range(0,n+1)]

for i in range(0,i):
    item = a[i]
    if 1 <= item <= n and mark[item] == 0:
        mark[item] = 1
        b[i] = str(item)
    else:
        c.append(i)
k = 0
for i in range(1,n+1):
    if k >= len(c):
        break
    if mark[i] == 0:
        b[c[k]] = str(i)
        k += 1
print " ".join(b)
