n = input()
perm = map(int, raw_input().split())

counts = [0] * (n+1)

for i in perm:
	if i <= n:
		counts[i] += 1

l = []
for i in range(1, n+1):
	if counts[i] == 0:
		l.append(i)

# print l

for i in range(n):
	if perm[i] > n:
		perm[i] = l.pop()
	elif counts[perm[i]] > 1:
		counts[perm[i]] -= 1
		perm[i] = l.pop()
	

perm = map(str, perm)
print ' '.join(perm)
