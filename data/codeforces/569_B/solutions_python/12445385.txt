n = int(raw_input())
arr = map(int, raw_input().split())
ss = dict()
for i in range(1, n+1): ss[i] = True
ans = [0]*n
for i in range(n):
    if arr[i] <= n and ss[arr[i]]:
        ans[i] = arr[i]
        ss[arr[i]] = False
curr = 0
for i in ss:
    if ss[i]:
        while ans[curr]!=0: curr += 1
        ans[curr] = i
