def solve(idxs):
  n = len(idxs)
  all_idxs = set([x for x in xrange(1, n+1)])
  need_fix = [False]*(n+1)
  for x in xrange(n):
    if idxs[x]>n or idxs[x]<1:
      need_fix[x] = True
    else:
      if idxs[x] in all_idxs:
        all_idxs.remove(idxs[x])
      else:
        need_fix[x] = True

  for x in xrange(n):
    if need_fix[x]:
      idxs[x] = all_idxs.pop()

  print ' '.join(map(lambda num: str(num), idxs))

raw_input()
solve(map(int, raw_input().split()))
