
n = int(raw_input())
nums = [int(x) for x in raw_input().split()]

used = [0]*n
re = []

for i in xrange(n):
	x = nums[i]
	if x-1 >= n:
		re.append(i)
	else:
		used[x-1] += 1
		if used[x-1] > 1:
			re.append(i)
	
for i in xrange(n):
	if used[i] == 0:
		nums[re.pop()] = i+1

print ' '.join(str(i) for i in nums)
