def reorder(n, items):
    seen = set()
    labelled = set(items)
    unlabelled = set(range(1, n + 1)) - labelled
    all = set(range(1, n + 1))
    
    for i, item in enumerate(items):
        if item in seen or item not in all:
            new_label = unlabelled.pop()
            items[i] = new_label
        else:
            seen.add(item)
    return items


n = int(raw_input())
items = map(int, raw_input().split())

print(" ".join(map(str, reorder(n, items))))
