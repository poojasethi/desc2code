#include<cstdio>
#include<iostream>
#include<cstring>
#include<cmath>
#define mod 1000000007
using namespace std;
int fac[1000006],f[1000006],ny[1000006],s[1005][1005];
void pre()
{
    fac[0]=fac[1]=f[0]=f[1]=ny[0]=ny[1]=s[0][0]=1;
    for(int i=2;i<1000006;i++)
    {
        fac[i]=1LL*i*fac[i-1]%mod;
        long long t=mod/i,k=mod%i;
        f[i]=1LL*i*t%mod*t%mod*f[k]%mod*f[k]%mod;
        ny[i]=1LL*f[i]*ny[i-1]%mod;
    }
    for(int i=1;i<1005;i++)
    for(int j=1;j<=i;j++) s[i][j]=(1LL*j*s[i-1][j]+s[i-1][j-1])%mod;
}
int ch(int a,int b)
{
    int sum=1;
    while(b)
    {
        if(b&1) sum=1LL*sum*a%mod;
        a=1LL*a*a%mod;
        b>>=1;
    }
    return sum;
}
int C(int a,int b)
{
    if(a<b) return 0;
    return 1LL*fac[a]*ny[b]%mod*ny[a-b]%mod;
}
int main()
{
    pre();
    int n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    int ans=0;
    if(m==1)
    {
        int ans=ch(k,n);printf("%d\n",ans);
        return 0;
    }
    else
    {
        int ans=0;
        for(int i=1;i<=n&&i<=k;i++)
        for(int j=0;j<=i;j++) if(2*i-j<=k)
        {
            long long te=1LL*C(k,j)*C(k-j,i-j)%mod*C(k-i,i-j)%mod;
            te=te*s[n][i]%mod*s[n][i]%mod*ch(j,n*(m-2))%mod*fac[i]%mod*fac[i]%mod;
            ans=(ans+te)%mod;
        }
        printf("%d\n",ans);
    }
}
