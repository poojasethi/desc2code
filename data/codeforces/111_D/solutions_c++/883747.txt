#include<iostream>
#include<algorithm>
#include<cstring>
#include<numeric>
using namespace std;
const unsigned long long mod=1e9+7;
unsigned long long n,m,k,p[1000001],ans[1001],Ans;

unsigned long long pow(unsigned long long a,unsigned long long b)
{
	unsigned long long tmp=a,ret=1;
	for(;b;tmp*=tmp,tmp%=mod,b>>=1)
		if(b&1)ret*=tmp,ret%=mod;
	return ret;
}

unsigned long long C(int a,int b)
{
	return p[a]*pow(p[a-b],mod-2)%mod*pow(p[b],mod-2)%mod;
}

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("yoshi.in","rb",stdin);
		freopen("yoshi.out","wb",stdout);
	#endif
	cin>>n>>m>>k;
	if(m==1)
	{
		cout<<pow(k,n)<<endl;
		return 0;
	}
	p[0]=1;
	for(int i=1;i<=k;++i)p[i]=p[i-1]*i%mod;
	for(int i=1;i<=n;++i)
	{
		ans[i]=pow(i,n);
		for(int j=1;j<i;++j)
			ans[i]=(ans[i]+mod-C(i,j)*ans[j]%mod)%mod;
	}
	for(int i=0;i<=n&&i<=k;++i)
	{
		unsigned long long tmp=0;
		for(int j=0;j<=n-i;++j)if(i+2*j<=k)
		tmp=(tmp+ans[i+j]*ans[i+j]%mod*C(k,i)%mod*C(k-i,j)%mod*C(k-i-j,j)%mod)%mod;
		Ans=(Ans+tmp*pow(i,n*(m-2))%mod)%mod;
	}	
	cout<<Ans<<endl;
	return 0;
}
