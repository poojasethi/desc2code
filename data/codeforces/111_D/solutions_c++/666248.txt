#include<cstdio>
const long long M=1000000007LL;
long long to[1010];
long long jie[1000010],ny[1000010];
long long mod(long long a,long long b,long long m)
{
    long long ans=1;
    while(b)
    {
        if(b&1)
        ans=ans*a%m;
        a=a*a%m;
        b>>=1;
    }
    return ans;
}
long long c(long long a,long long b)
{
    if(b>a/2)
    b=a-b;
    if(b==0)
    return 1;
    long long ans=jie[a];
    ans=ans*ny[b]%M*ny[a-b]%M;
    return ans;
}
int main()
{
    long long  n,m,k;
    scanf("%I64d%I64d%I64d",&n,&m,&k);
    if(m==1)
    {
        long long ans=mod(k,n,M);
        printf("%I64d",ans);
        return 0;
    }
    long long max,chao;
    if(n>k)
    max=k,chao=n;
    else
    max=n,chao=k;
    jie[0]=ny[0]=1;
    for(long long i=1;i<=chao+5;i++)
    {
        jie[i]=jie[i-1]*i%M;
        ny[i]=mod(jie[i],M-2,M);
    }
    long long ans=0;
    for(long long i=1;i<=max;i++)
    {
        long long temp=mod(i,n,M);
        for(long long j=i-1;j>=1;j--)
        temp=(temp-c(i,j)*to[j])%M;
        to[i]=temp;
        for(long long j=(m==2?0:1);j<=i;j++)
        {
            temp=to[i]*to[i]%M;
            temp=temp*mod(j,n*(m-2),M)%M;
            temp=temp*jie[k]%M*ny[k-2*i+j]%M*ny[j]%M*ny[i-j]%M*ny[i-j]%M;
            ans=(ans+temp)%M;
        }
    }
    printf("%I64d",ans);
}
