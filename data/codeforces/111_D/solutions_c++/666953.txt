#include<cstdio>
typedef unsigned ui;
typedef unsigned long long ull;
const ui H=1000000007, N=1000,K=1000010;
ui f[K];
ui pow(ui a,ui m){
	ull ret =1,w=a;
	for( ;m;m>>=1,w*=w,w%=H)
		if(m&1)ret*=w,ret%=H;
	return ui(ret);
}
ull C(ui a,ui b){
	return ull(f[a])*pow(f[a-b],H-2)%H*pow(f[b],H-2)%H;
}
ui solve(){
	ui n,m,k,i,j,p[N];
	ull a=0,t;
	scanf("%d%d%d",&n,&m,&k);
	if(m==1)return pow(k,n);f[0]=1;
	for(i=1;i<=k;++i)f[i]=ui(ull(f[i-1])*i%H);
	p[0]=0;
	for(i=1;i<=n;++i){
		p[i]=pow(i,n);
		for(j=1;j<i;++j){
			p[i]+=ui(H-p[j]*C(i,j)%H);
			if(p[i]>=H)p[i]-=H;
		}
	}
	for(i=1;i<=n;++i)p[i]=ui(ull(p[i])*p[i] %H);
	for(i=0;i<=n && i<=k;++i){
		t=0;
		for(j=0;j+i<=n && i+j*2<=k;++j)
			t+=p[i+j]*C(k,i)%H*C(k-i,j)%H*C(k-i-j,j)%H;
		a+=t%H*pow(i,n*(m-2))%H;
	}
	return ui(a%H);
}
int main(){
	return printf("%d\n",solve()),0;
}
