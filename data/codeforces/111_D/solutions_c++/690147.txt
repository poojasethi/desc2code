#define MOD 1000000007
ll POW(ll w,ll step){
  ll ret=1;
  for(;step;step>>=1, w=(w*w)%MOD)
   if(step&1)ret=(ret*w)%MOD;
  return ret;
}
ll REV(ll w){return POW(w,MOD-2);}
ll F[1000010],RF[1000010];
ll C(ll el,ll kl){
  return ( (F[el]*RF[kl])%MOD *RF[el-kl] )%MOD;
}
ll dp[2010][2010]; // x cvetami, y kletok
int main(){
  dp[0][0]=1;
  for(int X=1;X<2010;++X)
   for(int Y=X;Y<2010;++Y)
    dp[X][Y]=( X*(dp[X-1][Y-1] + dp[X][Y-1]) )%MOD;
  F[0]=RF[0]=1;
  for(int i=1;i<1000010;++i){F[i]=(i*F[i-1])%MOD; RF[i]=REV(F[i]);}
  ll n,m,k;
  cin>>n>>m>>k;
  if(m==1)cout<<POW(k,n)<<'\n';
  else{
   ll ans=0;
   for(int X=1;X<=n && X<=k;++X)
    for(int Y=X;Y<=2*X && Y<=k;++Y){
     ll cur=( (C(k,X) * C(k-X,Y-X))%MOD * C(X,Y-X) )%MOD;
     cur=( (cur*dp[X][n])%MOD *dp[X][n] )%MOD;
     cur=( cur*POW(X+X-Y,n*(m-2)) )%MOD;
     ans=(ans+cur)%MOD;
                                   }
   cout<<ans<<'\n';
      }
  //system("pause");
  return 0;
}
