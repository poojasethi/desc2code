using namespace std;
const long long MOD=1000000007,N=1228,M=1000228;

long long ans,l,r,t,i,j,n,m,k,fac[M],inv[M],dp[N][N];

long long Pow(long long a,long long b){
	if(!b)return 1;
	if(b%2==0){long long d=Pow(a,b/2);return d*d%MOD;}else return Pow(a,b-1)*a%MOD;	
}

long long C(long long n, long long m){
	return ((fac[n]*inv[n-m])%MOD)*inv[m]%MOD;
}

int main() {
	fac[0]=1,dp[0][0]=1;
	for(i=1;i<=M-1;i++)fac[i]=fac[i-1]*i%MOD;
	inv[M-1]=Pow(fac[M-1],MOD-2);
	for(i=M-1;i>0;i--)inv[i-1]=inv[i]*i%MOD;
	for(i=1;i<N;i++)for(j=1;j<=i;j++)dp[i][j]=(((dp[i-1][j]+dp[i-1][j-1])%MOD)*j)%MOD;
	cin>>n>>m>>k;
	l=1,r=min(n,k);
	if(m==1)for(i=l;i<=r;i++)ans=(ans+(dp[n][i]*C(k,i))%MOD)%MOD;else{
		if(m==2)l=0;
		for(i=l;i<=r;i++)for(j=0;j*2+i<=k && j+i<=n;j++){
			t=dp[n][i+j]*dp[n][i+j]%MOD;
			t*=C(k,i),t%=MOD;
			t*=C(k-i,j),t%=MOD;
			t*=C(k-i-j,j),t%=MOD;
			t=t*Pow(i,n*(m-2))%MOD;
			ans=(ans+t)%MOD;
		}
	}
	cout<<ans;
}
