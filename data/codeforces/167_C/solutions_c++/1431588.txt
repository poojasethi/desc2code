#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <math.h>
using namespace std;

typedef long long ll;

int gao(ll x, ll y) {
  if (x < y) swap(x, y);
  if (!y) return 0;

  if (!gao(y, x % y)) return 1;
  int sg = 0;
  x /= y;
  if (y % 2)
    return (x % 2 == 0);
  else
    return (x % (y + 1) % 2 == 0);
}

int main() {
  int T; scanf("%d", &T);
  while (T--) {
    ll a, b;
    scanf("%I64d %I64d", &a, &b);
    puts(gao(a, b) ? "First" : "Second");
  }
  return 0;
}
