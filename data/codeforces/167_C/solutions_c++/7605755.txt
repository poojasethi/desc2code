#include <cstdio>
#include <iostream>
using namespace std;
typedef long long LL;
LL a,b;
bool solve(LL x,LL y){
  if(!x)return 0; 
  return !solve(y%x,x) ||  !(y/x%(x+1) & 1 );
}
int main(){
  int t;
  cin >> t;
  while(t--){
    cin >> a >> b;
    if (a > b) a ^= b ^= a ^= b;
    puts(solve(a,b) ? "First" : "Second");
  }
}
