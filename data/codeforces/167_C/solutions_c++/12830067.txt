#include <bits/stdc++.h>
using namespace std;
#define ll long long

bool fwin(ll a, ll b)
{
    if(!a || !b) return false;
    if(a < b) swap(a, b);
    if(a % b == 0) return true;
    if(!fwin(b, a % b)) return true;
    ll k = a / b % (b + 1);
    return !(k&1);
}

int main()
{
    int t;
    ll a, b;
    scanf("%d", &t);
    while(t--)
    {
        scanf("%I64d %I64d", &a, &b);
        puts(fwin(a, b) ? "First" : "Second");
    }
    return 0;
}
