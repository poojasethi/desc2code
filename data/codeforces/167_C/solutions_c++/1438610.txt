#include<stdio.h>
#include<iostream>
using namespace std;
#define LL long long
bool dfs(LL a,LL b)
{
    if(a==0||b==0) return 0;
    return !dfs(b,a%b)||(a/b%(b+1)%2==0);
}
int main()
{
    LL a,b;
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%I64d%I64d",&a,&b);
        if(a<b) swap(a,b);
        if(dfs(a,b)) puts("First");
        else puts("Second");
    }
}
