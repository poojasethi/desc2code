#include<iostream>
using namespace std;
typedef long long LL;
int dfs(LL a,LL b)
{
    if(!a)
        return 1;
    if(dfs(b%a,a))
        return 0;
    if(a&1)
        return (b/a)&1;
    return (b/a)%(a+1)&1;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        LL a,b;
        cin>>a>>b;
        if(a>b) swap(a,b);
        if(dfs(a,b))
            cout<<"Second"<<endl;
        else
            cout<<"First"<<endl;
    }
    return 0;
}
