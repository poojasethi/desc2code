#include <iostream>
#include <algorithm>
using namespace std;

typedef long long LL;

int G (LL a, LL b)
{
	if (a < b) swap(a, b);
	if (!b) return 0;
	if (!G(b, a % b)) return 1;
	return ((a / b) % (b + 1)) % 2 == 0;
}

int main ()
{
	int T; for (cin >> T; T--; )
	{
		LL a, b;
		cin >> a >> b;
		cout << (G(a, b) ? "First" : "Second") << endl;
	}
	return 0;
}
