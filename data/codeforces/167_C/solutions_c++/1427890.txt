#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

typedef long long LL;
int go(LL b, LL a) {
	if(a==0) return 0;
	int v = go(a, b%a);
	if(!v) return 1;
	if(a%2==1) return !((b/a)%2);
	return !((b/a)%(a+1)%2);
}

void solve() {
	LL a, b;
	scanf("%I64d%I64d", &a, &b);
	if(a>b) swap(a, b);
	int r = go(b, a);
	puts(r?"First":"Second");
}

int main(void) {
	int T;
	scanf("%d", &T);
	while(T--) solve();
	return 0;
}
