#include <stdio.h>
using namespace std;

typedef long long ll;

#include <algorithm>
bool ok(ll a, ll b) {
	if (a == 0 || b == 0)
		return false;
	if (a == b)
		return true;
	if (a > b)
		swap(a,b);
	if (a == 1)
		return true;
	if (!ok(b%a,a))
		return true;
	ll k = b/a-1;
	return k%(a+1)%2 == 1 || k%(a+1) == a;
}

int main() {
	int T;
	scanf("%d", &T);
	for (int i = 0; i < T; i++) {
		ll a, b;
		scanf("%I64d%I64d", &a, &b);
		printf(ok(a,b)?"First\n":"Second\n");
	}
	return 0;
}
