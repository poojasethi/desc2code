#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;

typedef long long int64;

int64 A, B;

void init(){
	cin >> A >> B;
}

bool rec(int64 A, int64 B){
	if(A == 0 || B == 0){
		return false;
	}
	if(A > B){
		swap(A, B);
	}
	if(!rec(B%A, A)){
		return true;
	}
	return ( (B/A)%(A+1) )%2 == 0;
}

int main(){

	int T;
	cin >> T;
	for(;T--;){
		init();
		puts(rec(A, B) ? "First" : "Second");
	}

	return 0;
}
