#include <cstdio>
#include <algorithm>
using namespace std;

typedef long long LL;

bool calc(LL a, LL b)
{
    if(a == 0) return 0;
    if(!calc(b % a, a)) return 1;
    if(a & 1) return !((b/a) & 1);
    return !((b/a % (a+1)) & 1);
}

int main()
{
    int tc;
    scanf("%d", &tc);
    while(tc--) {
        LL a, b;
        scanf("%lld%lld", &a, &b);
        if(a > b) swap(a, b);
        puts(calc(a, b) ? "First" : "Second");
    }
    return 0;
}
