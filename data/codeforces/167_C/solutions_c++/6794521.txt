#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define rep(i,n) for (int i=0;i<n;++i)
typedef long long LL;
LL T,a,b;
bool work(LL a,LL b)
{
	if (a>b) swap(a,b); if (!a) return 0;
	if (!work(a,b%a)) return 1; LL c=b/a;
	return !(a&1?c&1:c%(a+1)&1);
}
int main()
{
	cin>>T;
	while (T--) cin>>a>>b,puts(work(a,b)?"First":"Second");
	return 0;
}
