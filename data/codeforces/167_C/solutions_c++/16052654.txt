#include <bits/stdc++.h>

using namespace std;

int check(long long a , long long b){
	if(a == 0) return 0;
	if(!check( b % a , a)) return 1;
	long long x = b / a; x = x % ( a + 1 );
	if(x & 1LL) return 0;
	return 1;
}

int main(int argc,char *argv[]){
	int Case;
	scanf("%d",&Case);
	while(Case--){
		long long a , b ;
		scanf("%I64d%I64d",&a,&b);
		if(a > b) swap( a , b );
		check( a , b ) ? printf("First\n") : printf("Second\n");
	}
	return 0;
}
