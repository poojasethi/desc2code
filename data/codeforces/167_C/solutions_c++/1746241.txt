#include <cstdio>
#include <algorithm>
#include <iostream>
using namespace std;
typedef long long LL;
int cal(LL a,LL b)
{
    if(a==0||b==0) return 0;
    if(a>b) swap(a,b);
    return !cal(a,b%a)||(b/a)%(a+1)%2==0;
}
 int main()
 {
     int t;
     cin >> t;
     while(t--){
         LL a,b;
         cin >> a >> b;
         int ans=cal(a,b);
         if(ans) cout << "First" << endl;
         else cout << "Second" << endl;
     }
 }
