#include <bits/stdc++.h>
using namespace std;
bool dfs(__int64 a, __int64 b)
{
	if (a == 0)
		return true;
	if (dfs(b%a, a))
		return false;
	if (a & 1)
		return b / a % 2;
	return b / a % (a + 1) % 2;
}
int main()
{
	int t;
	__int64 a, b;
	scanf("%d", &t);
	while (t--)
	{
		scanf("%I64d%I64d", &a, &b);
		if (a > b)
			swap(a, b);
		if (dfs(a, b))
			printf("Second\n");
		else
			printf("First\n");
	}
	return 0;
}
