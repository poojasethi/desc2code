#include <cstdio>

bool win(long long a, long long b) { // a <= b
  if (a == 0) return false;
  if (!win(b % a, a)) return true;
  long long r = (b/a - 1) % (a + 1) + 1;
  return r == a+1 || r % 2 == 0;
}

int main() {
  int t;
  scanf("%d", &t);
  while (t--) {
    long long a, b;
    scanf("%I64d%I64d", &a, &b);
    if (a > b) a ^= b ^= a ^= b;
    puts(win(a, b) ? "First" : "Second");
  }
  return 0;
}
