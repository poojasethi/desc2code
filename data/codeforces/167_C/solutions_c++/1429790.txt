#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int solve(long long a,long long b){
    if(!b)return 0;
    if(solve(b,a%b)==0)return 1;
    int t=(a/b)%(b+1);
    return t%2==0;
}
int main(){
    long long a,b;
    int test;
    scanf("%d",&test);
    for(int i=1;i<=test;++i){
        scanf("%I64d%I64d",&a,&b);
        if(a<b)swap(a,b);
        printf("%s\n",solve(a,b)?"First":"Second");
    }
    
    return 0;
}
