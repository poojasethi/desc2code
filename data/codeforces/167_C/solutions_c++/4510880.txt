#include <algorithm>
#include <cstring>
#include <cstdio>

using namespace std;


bool check(long long a, long long b){
	if(a<b) swap(a,b);
	
	if(b==0) return 0;
	if(!check(b,a%b)) return 1;
	
	a/=b;
	if((b&1)==1) return (a&1)==0;
	
	a%=(b+1);
	if((a&1)==0) return 1;
	return 0;
}
int main() {
	int T;
	scanf("%d",&T);
	while(T--) {
		long long a,b;
		scanf("%I64d%I64d",&a,&b);
		check(a,b)?puts("First"):puts("Second");
	}
	return 0;
};
