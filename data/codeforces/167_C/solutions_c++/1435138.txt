#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long LL;
bool dfs( LL a, LL b ){
	if( a < b ) swap( a, b );
	if( b == 0 ) return 0;
	if( !dfs( b, a%b ) ) return 1;
	if( b&1 ) return ((a/b)&1) == 0;
	else return (((a/b)%(b+1))&1) == 0;
}
int main(){
	LL t; scanf("%I64d",&t);
	while( t-- ){
		LL a,b;
		scanf("%I64d%I64d",&a,&b);
		if( dfs( a, b ) ) puts("First");
		else puts("Second");
	}
	return 0;
}
