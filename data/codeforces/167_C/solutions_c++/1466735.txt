#include<iostream>
#include <cstdio>
#include <cstring>
using namespace std;
typedef __int64 ll;
bool dfs(ll a,ll b)
{
    if(a<b)swap(a,b);
    if(b==0)return 0;
    if(!dfs(b,a%b))return 1;
    return (a/b%(b+1)+1)%2;
}
int main()
{
    int cas;
    scanf("%d",&cas);
    ll a,b;
    while(cas--)
    {
        scanf("%I64d%I64d",&a,&b);
        if(dfs(a,b))printf("First\n");
        else printf("Second\n");
    }
    return 0;
}
