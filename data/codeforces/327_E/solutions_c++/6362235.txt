//In the name of God
#include <iostream>
using namespace std;
const int N = 25, mod = 1e9 + 7;

int n, k, a[N], b[] = {-1, -1}, way[1 << N], l[1 << N];

int main() {
	ios_base::sync_with_stdio(false);
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i], l[1 << i] = a[i];
	cin >> k;
	for (int i = 0; i < k; i++)
		cin >> b[i];
	way[0] = 1;
	for (int mask = 1; mask < 1 << n; mask++) {
		l[mask] = l[mask & ~(mask & -mask)] + l[mask & -mask];
		if (l[mask] != b[0] && l[mask] != b[1])
			for (int j = mask; j; j -= j & -j)
				if ((way[mask] += way[mask & ~(j & -j)]) >= mod)
					way[mask] -= mod;
	}
	cout << way[(1 << n) - 1] << '\n';
	return 0;
}
