#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#define MOD 1000000007
#define MAXN 24
using namespace std;

typedef long long int ll;
int N,K,dp[1<<MAXN];
ll S[MAXN],A[2],sum[1<<MAXN];

int main() {
  scanf("%d",&N);
  for(int i=0;i<N;++i) cin >> S[i];
  scanf("%d",&K);
  for(int i=0;i<K;++i) cin >> A[i];
  sort(A,A+K);
  dp[0] = 1;
  for(int z=0;z<(1<<N);++z) {
    for(int i=0;i<K;++i)
      if(sum[z] == A[i]) dp[z] = 0;
    for(int i=0;i<N;++i)
      if( (z&(1<<i)) == 0) {
        int j = (z|(1<<i));
        dp[j] += dp[z];
        sum[j] = sum[z] + S[i];
        if(dp[j] >= MOD) dp[j] -= MOD;
      }
  }
  printf("%d\n",dp[(1<<N)-1]);
}
