#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define int33 unsigned int
#define mo 1000000007
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,b[3],a[30],f[1<<24],h[1<<24];
int33 g[1<<24];
int main() {
	scanf("%d",&n); m=(1<<n)-1;
	For(i,0,n-1) scanf("%d",&a[i]);
	scanf("%d",&k); b[1]=b[2]=-1;
	For(i,1,k) scanf("%d",&b[i]);
	For(i,0,n-1) h[1<<i]=i;
	f[0]=1;
	For(i,1,m) {
		g[i]=g[i-(i&-i)]+a[h[i&-i]];
		if (g[i]!=b[1]&&g[i]!=b[2]) {
			for (j=i;j;j-=j&-j) {
				f[i]+=f[i-(j&-j)];
				if (f[i]>=mo) f[i]-=mo;
			}
		}
	}
	printf("%d\n",f[m]);
	//for(;;);
	return 0;
}
