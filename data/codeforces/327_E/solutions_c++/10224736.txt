#include<set>
#include<cstdio>
#include<iostream>
#define MAXN (1<<24)
#define MOD 1000000007
#define lowbit(x) (x&(-x))

using namespace std;
set<int>s;
int dp[MAXN+1];
long long dis[MAXN+1];
int main()
{
	int n,i,j,k;
	long long S=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		cin>>dis[1<<i];
		S+=dis[1<<i];
	}
	for(scanf("%d",&k);k;k--)
		scanf("%d",&i),s.insert(i);
	k=1<<n;
	dp[0]=1;
	for(i=1;i<k;i++)
	{
		dis[i]=dis[i^lowbit(i)]+dis[lowbit(i)];
		if(s.count(dis[i]))
			continue;
		for(j=i;j;j-=lowbit(j))
			dp[i]=(dp[i]+dp[i^lowbit(j)])%MOD;
	}
	printf("%d",dp[k-1]);
    return 0;    
}
