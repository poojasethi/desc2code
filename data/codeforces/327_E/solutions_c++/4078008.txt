#include<cstdio>
#define N 24
#define M 1000000007
#define LO(x) (x&(-x))
int v[N],b[2];
int pd[1<<N];
int mod(int x){
	if(x<M) return x;
	return x-M;
}
int main(){
	int n;scanf("%d",&n);
	for(int i=0;i<n;++i){
		scanf("%d",v+i);
		pd[1<<i]=v[i];
	}
	int k;scanf("%d",&k);
	for(int i=0;i<k;++i)
		scanf("%d",b+i);
	
	pd[0]=0;
	for(int i=1;i<(1<<n);++i)
		pd[i]=mod(pd[i-LO(i)]+pd[LO(i)]);
	
	pd[0]=1;

	for(int i=1,q;i<(1<<n);++i){
		q=pd[i]; pd[i]=0;
		if( q==b[0] || q==b[1] ) continue;
		for(int t=i;t;t-=LO(t))
			pd[i]= mod(pd[i] + pd[i^LO(t)]);
	}

	printf("%d\n",pd[(1<<n)-1]);
	return 0;
}
