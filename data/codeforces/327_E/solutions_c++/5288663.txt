#include <iostream>
#include <cstdio>
using namespace std;

long long SUM[1<<24];
int k, ban[2]={-1, -1}, dp[1<<24]={};
int main()
{
    int n, a[30];
    scanf("%d", &n);
    for (int i=0;i<n;++i)
        scanf("%d", a+i);
    scanf("%d", &k);
    for (int i=0;i<k;++i)
        scanf("%d", ban+i);
    dp[0] = 1;
    for (int i=1;i<(1<<n);++i)
        SUM[i] += SUM[i^(i&(-i))]+a[__builtin_ctz(i)];
    for (int i=1;i<(1<<n);++i)
    {
        long long t = 0;
        if (SUM[i]==ban[0]||SUM[i]==ban[1])
            continue;
        for (int j=i;j;j^=(j&(-j)))
            t += dp[i^(j&(-j))];
        dp[i] = t%1000000007;
    }

    printf("%d\n", dp[(1<<n)-1]);
    return 0;
}
