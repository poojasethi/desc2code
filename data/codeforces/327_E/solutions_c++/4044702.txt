#include <iostream>
#include <cmath>
#define MAXN 24
#define MOD 1000000007

using namespace std;

int n, m, a[1 << MAXN], b[2], c[1 << MAXN];
long long sum[1 << MAXN];

main()
{
  cin >> n;
  for (int i = 0; i < n; i++) cin >> a[1 << i];
  cin >> m;
  for (int i = 0; i < m; i++) cin >> b[i];
  
  c[0] = 1;
  for (int i = 1; i < 1 << n; i++)
  {
    sum[i] = sum[i - (i & -i)] + a[i & -i];
    if (sum[i] == b[0] || sum[i] == b[1]) continue;
    for (int j = i; j ; j -= j & -j)
      c[i] = (c[i] + c[i - (j & -j)]) % MOD;
  }
  
  cout << c[(1 << n) - 1];
  //system("pause");
}
