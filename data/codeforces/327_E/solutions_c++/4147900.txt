# include <iostream>
# include <cstdio>
using namespace std;
const int N = 24, Mod = 1000000007;
long long a[N], b[2], s[1<<N], f[1<<N];
int n, m;

int main(){
    b[0] = b[1] = -1; f[0] = 1;
    cin >> n; for(int i=0; i<n; i++) cin >> a[i];
    cin >> m; for(int i=0; i<m; i++) cin >> b[i];
    for(int i=1; i<(1<<n); i++) {
        int j = __builtin_ctz(i);
        s[i] = s[i^(1<<j)] + a[j];
    }

    for(int i=1; i<(1<<n); i++) {
        if (s[i]==b[0] || s[i]==b[1])
            continue;
        for(int j=0; j<n; j++)
            if( i & (1<<j))
                f[i] += f[i ^ (1<<j)];
        f[i] %= Mod;
    }
    cout << f[(1<<n)-1] << endl;
}
