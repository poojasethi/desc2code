#include<bits/stdc++.h>
#define mod 1000000007
#define M 24
using namespace std;
#define lowbit(x) ((x)&(-x))
int dp[1<<M],a[1<<M];
int n,k,p1,p2;
int main()
{
	cin>>n;
	int i,j,tmp,k,l;
	for(i=0;i<n;i++) scanf("%d",&a[1<<i]);
	p1=p2=-1;
	cin>>k;
	if(k) cin>>p1;
	if(k>1) cin>>p2;
	dp[0]=1;
	k=(1<<n);
	for(i=1;i<k;i++)
	{
		for(j=i,tmp=0,l=lowbit(j);j;tmp+=a[l],j^=l,l=lowbit(j))
			dp[i]+=dp[i^l],dp[i]%=mod;
		if(tmp==p1||tmp==p2) dp[i]=0;
	}
	cout<<dp[k-1]<<endl;
}
