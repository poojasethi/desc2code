#include <cstdio>
#define p 1000000007
typedef long long int ll;
int dp[1<<24];
int a[23];

int main(){
    int n,k;
    scanf("%d", &n);
    for (int i=0; i<n; i++){
        scanf("%d", a+i);
    }
    int x,y;
    x=y=-1;
    dp[0]=1;
    scanf("%d", &k);
    if (k==1) scanf("%d", &x);
    if (k==2) scanf("%d%d", &x,&y);

    for (int i=1; i<(1<<n); i++){
        ll sum=0;
        for (int j=0; j<n; j++){
            if (i&(1<<j)){
                sum+=a[j];
                dp[i]=dp[i]+dp[i^(1<<j)];
                if (dp[i]>=p) dp[i]-=p;
            }
        }
        if (sum==x || sum==y) dp[i]=0;
    }
    printf("%d", dp[(1<<n)-1]);
    return 0;
}
