#include<stdio.h>
#define M 1000000007
int dp[1<<24];
int n,k;
int a[24];
int b[2];

int main(){
    int i,j;
    scanf("%d",&n);
    long long sd=0;
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
        sd+=a[i];
    }
    scanf("%d",&k);
    b[0]=b[1]=-1;
    for(i=0;i<k;i++)scanf("%d",&b[i]);
    dp[0]=(sd==b[0]||sd==b[1])?0:1;
    for(i=1;i<1<<n;i++){
        long long d=0;
        long long res=0;
        for(j=0;j<n;j++){
            if(i>>j&1)res+=dp[i&~(1<<j)];
            else d+=a[j];
        }
        if(d==b[0]||d==b[1])dp[i]=0;
        else dp[i]=res%M;
    }
    printf("%d\n",dp[(1<<n)-1]);
    return 0;
}
