#include <stdio.h>
#include <string.h>

const int M = 1000000007;

int a[33], b[3] = {-1, -1, -1};
int f[1 << 24], g[1 << 24];

int main() {
	int n, m;
	int i, j, k;
	scanf("%d", &n);
	for (i = 0; i < n; ++i) scanf("%d", a + i);
	scanf("%d", &m);
	for (i = 0; i < m; ++i) scanf("%d", b + i);

	int N = 1 << n;
	memset(f, 0, sizeof f);
	f[0] = 1;
	g[0] = 0;
	for (i = 1; i < N; ++i) {
		g[i] = g[i & (i - 1)] + a[__builtin_ctz(i)];
		if (g[i] == b[0] || g[i] == b[1]) continue;
		for (j = i; j; j ^= k) {
			k = 1 << __builtin_ctz(j);
			if ((f[i] += f[i ^ k]) >= M) f[i] -= M;
		}
	}

	printf("%d", f[N - 1]);
	return 0;
}
