#include <iostream>
using namespace std;

const int MOD = 1000000007;

int n,k;
int a[24];
int u[2];
int som[1 << 24];
int ris[1 << 24];

int main(){
	cin >> n;
	for(int i=0; i<n; i++) cin >> a[i];
	
	cin >> k;
	for(int i=0; i<2; i++){
		if(i<k) cin >> u[i];
		else u[i]=-1;
	}
	
	som[0] = 0;
	for(int i=1; i<(1 << n); i++){
		int j = __builtin_ctz(i);
		som[i] = som[i] = (som[i^(1<<j)] + a[j]) % MOD;
	}
	
	ris[0] = 1;
	for(int i=1; i<(1 << n); i++)
		if(som[i]!=u[0] && som[i]!=u[1]){
			int x = i;
			while(x > 0){
				int j = __builtin_ctz(x);
				x = x^(1<<j);
				ris[i] = (ris[i] + ris[i^(1<<j)]) % MOD;
			}
		}
	
	cout << ris[(1<<n)-1];
	
	return 0;
}
