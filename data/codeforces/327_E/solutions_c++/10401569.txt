#include<stdio.h>
#include<string.h>
int n, k;
int a[25];
int b[5];
int dp[1<<25];
const int mod = 1000000007;
int main()
{
	while(scanf("%d", &n)!=EOF)
	{
		memset(dp, 0, sizeof(dp));
		for(int i = 0;i<n;i++)
			scanf("%d", &a[i]);
		scanf("%d", &k);
		for(int i = 0;i<k;i++)
			scanf("%d", &b[i]);
		dp[0] = 1;
		for(int s = 1;s<(1<<n);s++)
		{
			int sum = 0;
			for(int i = 0;i<n;i++)
			{
				if(s&(1<<i))
				{
					dp[s]+=dp[s^(1<<i)];
					sum+=a[i];
					dp[s]%=mod;
				}
			}
			for(int j = 0;j<k;j++)
			{
				if(sum == b[j])
				dp[s] = 0;
			}
		}
		printf("%d\n", dp[(1<<n)-1]);
	}
	return 0;
}
