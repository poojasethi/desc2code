#include <stdio.h>
const int N = 24;
int dp[1<<N], a[N], b[N];
int n, k;
const int mod = 1e9+7;
int main(){
	scanf("%d", &n);
    for(int i = 0; i < n; i++)scanf("%d", &a[i]);
    scanf("%d", &k);
    for(int i = 0; i < k; i++)scanf("%d", &b[i]);
    
    dp[0] = 1;
    for(int i = 1, sum; i < (1<<n); i++){
        sum = 0;
        for(int j = 0; j < n; j++)
            if(i&(1<<j))
            {
                dp[i] += dp[i^(1<<j)];
                if(dp[i]>=mod)dp[i] -= mod;
                sum += a[j];
        }
        for(int j = 0; j < k; j++)if(sum == b[j])dp[i] = 0;
	}       
	printf("%d\n",dp[(1<<n)-1]);
	return 0;
}
