#include <iostream>
using namespace std;

int k, ban[2]={-1, -1}, dp[1<<24]={};
int main()
{
    int n, a[30];
    cin >> n;
    for (int i=0;i<n;++i)
        cin >> a[i];
    cin >> k;
    for (int i=0;i<k;++i)
        cin >> ban[i];
    dp[0] = 1;
    for (int i=1;i<(1<<n);++i)
    {
        int SUM = 0, t = 0;
        for (int j=0;j<n;++j)
            if (i & (1<<j))
            {
                SUM += a[j];
                t += dp[i^(1<<j)];
                if (t >= 1000000007)
                    t -= 1000000007;
            }
        if (SUM==ban[0]||SUM==ban[1])
            continue;
        dp[i] = t;
    }

    cout << dp[(1<<n)-1] << endl;
    return 0;
}
