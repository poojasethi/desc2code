#include <stdio.h>
const int M=1000000007;
int n,k,a[32],b[2]={-1,-1},s[1<<24],w[1<<24];
main(){
  scanf("%d",&n);
  for(int i=0;i<n;++i)scanf("%d",&a[i]),s[1<<i]=a[i];
  scanf("%d",&k);
  for(int i=0;i<k;++i)scanf("%d",&b[i]);
  w[0]=1;
  for(int i=1,j;i<(1<<n);++i){
    j=i&-i;
    if(j!=i&&(s[i]=s[i-j]+s[j])>M)s[i]=M;
    if(s[i]!=b[0]&&s[i]!=b[1])for(j=i;j;j-=j&-j)if((w[i]+=w[i-(j&-j)])>=M)w[i]-=M;
  }
  printf("%d\n",w[(1<<n)-1]);
}
