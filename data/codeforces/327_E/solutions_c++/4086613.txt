#include<stdio.h>
#include<string.h>
int n,k;
int a[25];
int b[2];
long long c[20000000];
int main()
{
    long long dist=0;
    memset(c,-1,sizeof(c));
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    b[0]=b[1]=-1;
    scanf("%d",&k);
    for(int i=0;i<k;i++) scanf("%d",&b[i]);
    c[0]=1;
    for(int i=1;i<(1<<n);i++)
    {
        c[i]=0;
        long long sum=0;
        for(int j=0;j<n;j++)
        {
            if((i&(1<<j))==0) continue;
            c[i]+=c[i&(~(1<<j))];
            sum+=a[j];
        }
        c[i]%=1000000007;
        if(sum==b[0] || sum==b[1]) c[i]=0;
    }
    printf("%I64d\n",c[(1<<n)-1]);
    return 0;
}
