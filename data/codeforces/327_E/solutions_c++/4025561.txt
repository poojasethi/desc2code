#include<iostream>
#include<cstdio>
using namespace std;
const int Mod =1000000007;
int dp[1<<24],a[24];
int sum[1<<24];
int b[2];

int main(){
	int i,j,k,n;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",a+i);
		sum[1<<i]=a[i];
	}
	scanf("%d",&k);
	dp[0]=1;
	for(i=0;i<k;i++)scanf("%d",b+i);
	for(i=1;i<1<<n;i++){
		sum[i]=sum[i-(i&-i)]+sum[i&-i];
		if(sum[i]>1000000007)sum[i]=1000000007;
	
		if(sum[i]==b[0]||sum[i]==b[1])continue;
		for(j=i;j>0;j-=j&-j){
			dp[i]+=dp[i-(j&-j)];
			if(dp[i]>Mod)dp[i]-=Mod;
		}
	}	
	printf("%d\n",dp[(1<<n)-1]);
	return 0;
}
