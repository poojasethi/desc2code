#include<cstdio>
#include<iostream>
using namespace std;
int n,k;
int dp[17000000];
long long v[17000000];
int p2[30];
long long step[17000000];
long long bl[5];
int main()
{
	int i,j;
	scanf("%d",&n);
	for(i=0;i<n;++i)
		cin>>step[1<<i];
	bl[1]=bl[2]=-1;
	scanf("%d",&k);
	for(i=1;i<=k;++i)
		cin>>bl[i];
	p2[1]=1;
	for(i=2;i<=n+1;++i)
		p2[i]=p2[i-1]<<1;
	dp[0]=1;
	for(i=1;i<p2[n+1];++i)
	{
		v[i]=v[i&(i-1)]+step[i&-i];
		if(v[i]==bl[1]||v[i]==bl[2])
			continue;
		for(j=i;j;j=(j-1)&j)
		{
			dp[i]+=dp[i^(j&-j)];
			if(dp[i]>1000000007)
				dp[i]-=1000000007;
		}
	}
	cout<<dp[p2[n+1]-1];
	return 0;
}
