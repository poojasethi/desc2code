#include<cstdio>
#define MAXN 24
const int mod=1000000007;
int A[MAXN+10],n;
int dp[1<<MAXN];
int k,B[MAXN+10];
int main()
{
	int i,j;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&A[i]);
	scanf("%d",&k);
    for(i=1;i<=k;i++)
		scanf("%d",&B[i]);
	int up=(1<<n)-1;
    dp[0]=1;
    int dist=0;
    for(i=1;i<=up;i++)
	{
		dist=0;
		for(j=0;j<n;j++)
			if(i&(1<<j))
			{
				dp[i]=(dp[i]+dp[i^(1<<j)])%mod;
				dist+=A[j];
				if(dist>mod) dist=mod;
			}
		for(j=1;j<=k;j++)
			if(dist==B[j])
				dp[i]=0;
	}
	printf("%d\n",dp[up]);
}
