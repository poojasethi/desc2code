#include<cstdio>
#include<cstring>
#define LL long long
const int p=1000000007;
int f[16777216];
int a[24];
int b[3];
int n,k;
int main()
{
    LL sum;
    int i,j,MX;
    scanf("%d",&n);
    for (i=0;i<=n-1;i++) scanf("%d",&a[i]);
    scanf("%d",&k);
    b[1]=-1; b[2]=-1;
    for (i=1;i<=k;i++) scanf("%d",&b[i]);
    f[0]=1; MX=(1<<n)-1;
    for (i=1;i<=MX;i++)
    {
        sum=0;
        for (j=0;j<=n-1;j++)
            if (i&(1<<j))
            {
               sum+=a[j]; f[i]+=f[i^(1<<j)];
               if (f[i]>=p) f[i]-=p;
            }
        if ((sum==b[1])||(sum==b[2])) f[i]=0;
    }
    printf("%d\n",f[(1<<n)-1]); return 0;
}
