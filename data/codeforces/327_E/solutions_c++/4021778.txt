#include <cstdio>
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#define M 1000000007
using namespace std;
typedef pair<int, int> pii;
typedef long long ll;

int n, k;
int sum[1<<24], dp[1<<24], v[1<<24];

int main() {
	int i, j, a[2];
	cin>>n;
	for(i=0;i<n;i++) cin>>v[1<<i];
	cin>>k;
	for(i=0;i<k;i++) cin>>a[i];
	dp[0]=1;
	for(i=1;i<1<<n;i++) {
		sum[i] = sum[i-(i&-i)] + v[i&-i];
		if(sum[i]>M) sum[i]=M;
		for(j=1;j<1<<n;j<<=1) if(i&j) {
			dp[i] += dp[i-j];
			if(dp[i]>M) dp[i]-=M;
		}
		if(sum[i]==a[0] || sum[i]==a[1]) dp[i]=0;
	}
	cout<<dp[(1<<n)-1]<<endl;
	return 0;
}
