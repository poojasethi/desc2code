#include <iostream>
#include <cmath>

#define rep(i,n) for(int i=0;i<n;i++)
#define MOD 1000000007

using namespace std;

int n,k,t[10],a[1<<24],sum[1<<24],dp[1<<24],b,j;

int main(){
    cin>>n; rep(i,n)cin>>a[1<<i];
    cin>>k; rep(i,k)cin>>t[i];
    dp[0] = 1;
    for(b=1;b<(1<<n);b++){
    sum[b] = sum[b-(b&-b)]+a[b&-b];
    sum[b] = min(sum[b],MOD);
    if(sum[b]==t[0] || sum[b]==t[1])continue;
    for(j=b;j;j-=j&(-j)){
        dp[b] = (dp[b]+dp[b-(j&-j)])%MOD;
    }
    }
    cout<<dp[(1<<n)-1]<<endl;

}
