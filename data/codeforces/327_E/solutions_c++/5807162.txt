#include <iostream>
using namespace std;
const int mod = 1000000007;
int sum[1<<24], dp[1<<24], v[1<<24];
int main() {
    int n, k, i, j, a[2];
    cin>>n;
    for(i=0;i<n;i++) cin>>v[1<<i];
    cin>>k;
    for(i=0;i<k;i++) cin>>a[i];
    dp[0]=1;
    for(i=1;i<1<<n;i++) {
        sum[i] = sum[i-(i&-i)] + v[i&-i];
        if(sum[i]>mod) sum[i]=mod;
        if(sum[i]==a[0] || sum[i]==a[1]) continue;
        for(j=i; j>0; j -= (j&(-j))) {
            dp[i] += dp[i-(j&-j)];
            if(dp[i] > mod) dp[i] -= mod;
        }
    }
    cout<<dp[(1<<n)-1]<<endl;
    return 0;
}
