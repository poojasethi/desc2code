#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int mod=1e9+7;
ll dp[1<<24],n,a[25],pos[3],k,sum[1<<24];
inline void go(ll mask){
	ll tmp=__builtin_ctz(mask);
	sum[mask]=sum[mask^(1ll<<tmp)]+a[tmp];
	for(int j=0 ; j<k ; j++)
		if(sum[mask]==pos[j])
			return ;
	for(int i=mask ; i>0 ; i-=(i & -i)){
		tmp=__builtin_ctz(i);
		dp[mask]=dp[mask]+dp[mask^(1<<tmp)];
		if(dp[mask]>=mod)dp[mask]-=mod;	
	}
}
int main(){
	cin>>n;
	for(int i=0 ; i<n ; i++){
		cin>>a[i];
		sum[1<<i]=a[i];
	}
	cin>>k;for(int i=0 ; i<k ; i++)cin>>pos[i];
	dp[0]=1;
	for(int i=1 ; i<(1<<n) ; i++)go(i);
	cout<<dp[(1<<n)-1]<<endl;
	return 0;
}
