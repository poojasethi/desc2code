#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<string.h>
#define M 1000000007
#define llu long long int 
using namespace std;
int dp[1<<24],a[30],x,y;
int main()
{
	int n,i,j,k;
	x=-1,y=-1;	
	cin>>n;
	for(i=0;i<n;i++) scanf("%d",&a[i]);
	scanf("%d",&k);
	if(k==1) scanf("%d",&x);
	else if(k==2)scanf("%d %d",&x,&y);
	memset(dp,0,sizeof(dp));
	dp[0]=1;
	int max=1<<n;
	for(i=1;i<max;i++)
	{
		llu sum=0; 
		for(j=0;j<n;j++)
		{
			if((1<<j)&i)
			{
				sum+=a[j];
				dp[i]+=dp[i^(1<<j)];
				if(dp[i]>=M) dp[i]-=M;
			}
		}
		//cout<<dp[i]<<endl;
		if(sum==x || sum==y) dp[i]=0;
	}
	printf("%d\n",dp[max-1]);
}
