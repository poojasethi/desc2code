#include <iostream>
using namespace std;
const int mod = 1e9+7;
int sum[1<<24], a[1<<24], b[1<<24];
int main() {
    int n, k, c[2];
    cin>>n;
    for( int i=0;i<n;i++) cin>>b[1<<i];
    cin>>k;
    for( int i=0;i<k;i++) cin>>c[i];
    a[0]=1;
    for(int i=1;i<1<<n;i++) {
        sum[i] = sum[i-(i&-i)] + b[i&-i];
        if(sum[i]>mod) sum[i]=mod;
        if(sum[i]==c[0] || sum[i]==c[1]) continue;
        for(int j=i; j>0; j -= (j&(-j))) {
            a[i] += a[i-(j&-j)];
            if(a[i] > mod) a[i] -= mod;
        }
    }
    cout<<a[(1<<n)-1]<<endl;

}
