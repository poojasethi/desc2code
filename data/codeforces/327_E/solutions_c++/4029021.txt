#include <iostream>

using namespace std;

typedef long long int ll;

ll modulo=1000000007;

const int limite=1<<24;

ll rutas[limite];
ll suma[limite];
int k;
ll bad[2];
int n;
ll a[30];

int main()
{
  cin>>n;
  for (int i=0;i<n;i++)
    cin>>a[i];
  cin>>k;
  for (int i=0;i<k;i++)
    cin>>bad[i];
  rutas[0]=1;
  int tope=1<<n;
  for (int c=1;c<tope;c++) {
    int bb=__builtin_ctz(c);
    suma[c]=suma[c^(1<<bb)]+a[bb];
    bool error=false;
    for (int i=0;i<k;i++)
      if (suma[c]==bad[i])
	error=true;
    if (error) {
      rutas[c]=0;
    } else {
      for (int b=0;b<n;b++)
	if (c&(1<<b))
	  rutas[c]+=rutas[c-(1<<b)];
      rutas[c]%=modulo;
    }
  }
  cout<<rutas[tope-1]<<endl;
}
