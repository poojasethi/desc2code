#include <iostream>
using namespace std;

#define num long long
const int M=24;
const num MOD=1e9+7;

num dp[1<<M]; int a[M+10],bad[3];
int place[1<<M];

int main()
{
  int n;
  cin>>n;
  for(int i=0;i<n;++i) cin>>a[i];
  int k; cin>>k; for(int i=0;i<k;++i) cin>>bad[i];
  for(int i=1;i< 1<<n;++i)
    {
    place[i]=place[i^(1<<__builtin_ctz(i))]+a[__builtin_ctz(i)];
    for(int j=0;j<k;++j)
      if(place[i]==bad[j])
	dp[i]=-1;
    }
  dp[0]=1;
  for(int i=1;i< 1<<n;++i)
    {
      if(dp[i]==-1) {dp[i]=0; continue;}
    for(int j=0;j<n;++j)
      if((1<<j)&i)
	dp[i]+=dp[i^(1<<j)];
    dp[i]%=MOD;
    }
  cout<<fixed<<dp[(1<<n)-1]<<endl;
  return 0;
}
