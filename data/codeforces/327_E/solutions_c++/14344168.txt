#include <bits/stdc++.h>
using namespace std;
int dp[1<<24];int n,m, a[24],k[2],sum[1<<24];
const int mod =1e9+7;
int main(){
	cin>>n;for(int i=0;i<n;++i){cin>>a[i];sum[1<<i]=a[i];}
	cin>>m;for(int i=0;i<m;++i)cin>>k[i];
	int i,j;dp[0]=1;
	for(i=1;i<(1<<n);++i){
		j = i&-i;
		if(j!=i)
			sum[i]= min(mod,sum[i-j]+sum[j]);bool ok=true;
		for(int x=0;x<2;++x)if(sum[i]==k[x]){ok=false;break;}
		if(!ok)continue;
		for(j=i;j;j-=j&-j){
			dp[i]= dp[i]+dp[i-(j&-j)];
			if(dp[i]>=mod)dp[i]-=mod;
		}
	}
	cout<<dp[(1<<n)-1]<<endl;
}
