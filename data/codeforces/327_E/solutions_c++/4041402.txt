#include <stdio.h>
int a[25],b[3];
int div = 1000000007;
int dp[1<<24];
int main()
{
    int n,k;
    scanf("%d",&n);
    int i,j;
    for(i=0;i<n;i++)scanf("%d",a+i);
    scanf("%d",&k);
    for(i=0;i<k;i++) scanf("%d",b+i);
    dp[0]=1;
    for(i=1;i<(1<<n);i++)
    {
        int cnt=0;
        for(j=0;j<n;j++)
        {
            if((i>>j)==0) break;
            if((i>>j)&1)
            {
                cnt+=a[j];
                dp[i] += dp[i&(~(1<<j))];
                dp[i]%=div;
            }
        }
        for(j=0;j<k;j++)
        {
            if(cnt==b[j])
                dp[i]=0;
        }
    }
    printf("%d",dp[(1<<n)-1]);
}
