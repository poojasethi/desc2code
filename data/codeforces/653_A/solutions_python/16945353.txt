n=raw_input()
t=map(int,raw_input().split(' '))
s=set(t)
s=list(s)
s.sort()
m=1
flag=False
if len(s)<3:
    print 'NO'
else:
    for i in range(len(s)-1):
        if m==3:
            break;
        if s[i+1]-s[i]==1:
            m+=1
        else:
            m=1
    if m==3:
        print 'YES'
    else:
        print 'NO'
