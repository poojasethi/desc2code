	high=False
	low2=False
	high2=False
	for j in range(i+1,len(m)):
		if(m[j]==m[i]-1):
			low=True
		elif(m[j]==m[i]+1):
			high=True
		elif(m[j]==m[i]-2):
			low2=True
		elif(m[j]==m[i]+2):
			high2=True
	if((low and high) or (low and low2) or (high and high2)):

		break
