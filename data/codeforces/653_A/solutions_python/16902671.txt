import sys

N = int(raw_input())
T = raw_input()


sizes = [int(a) for a in T.split()]

sizes_sorted = sorted(set(sizes))



def window_check(a,b,c):
    if(abs(a-b)>2 or abs(b-c) > 2 or abs(a-c) > 2):
        return False
    return True

check = False
for i in xrange(len(sizes_sorted)-2):
    check = window_check(sizes_sorted[i],sizes_sorted[i+1],sizes_sorted[i+2]) or check

if check:
    sys.stdout.write("YES")
else:
    sys.stdout.write("NO")
