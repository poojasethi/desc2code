n = input()
size = map(int, raw_input().split())
check = False
for i in range(n):
    for j in range(n):
        for k in range(n):
            if size[i] + 1 == size[j] == size[k]-1 and check == False:
                print 'YES'
                check = True
else:
    if check == False:
        print 'NO'
