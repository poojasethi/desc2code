#!/usr/bin/python
import sys
import re

def input_numbers():
    return [int(n) for n in raw_input().split()]

n = input()
t = set(input_numbers())

def no():
    print "NO"
    exit()

def yes():
    print "YES"
    exit()

if len(t) < 3: no()

for i in t:
    for j in t:
        for k in t:
            if i != j and j != k and k != i and max(abs(i -j), abs(i -k), abs(j - k)) <= 2:
                yes()

no()
