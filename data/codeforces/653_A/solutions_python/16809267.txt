input = raw_input()  
n=int(input)
lst = raw_input().split()
size = list() 
for i in range(0,n):
  if int(lst[i]) not in size:
    size.append(int(lst[i]))  
size.sort()
result=0
for index in range(0,len(size)-2):
  if size[index]+1==size[index+1] and size[index]+2==size[index+2]:
    result=1
if n<3:
  result=0
if result==1:
  print "YES"
else:
  print "NO"
