import sys


def main():
    for i, line in enumerate(sys.stdin):
        if i < 1:
            continue
        balls = sorted(list(set(map(int, line.split()))))
        length = len(balls)
        j = 0
        for i, b in enumerate(balls):
            while j < length and balls[j] - b <= 2:
                j += 1
            j -= 1
            if j - i >= 2:
                print "YES"
                return
    print "NO"


if __name__ == "__main__":
    main()
