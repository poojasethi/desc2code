#include<cstdio>
#include<cstring>

using namespace std;
int a[55],n;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;++i)scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	n=unique(a+1,a+n+1)-a-1;
	for(int i=1;i+2<=n;++i)if(a[i+1]-a[i]==1&&a[i+2]-a[i+1]==1)return puts("YES"),0;
	puts("NO");
}
