#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

int n,a[1010];
int x;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&x);
		a[x]++;
	}
	for(int i=1;i<=998;i++)
	{
		if(a[i] && a[i+1] && a[i+2])
		{
			printf("YES\n");
			exit(0);
		}
	}
	printf("NO");
}
