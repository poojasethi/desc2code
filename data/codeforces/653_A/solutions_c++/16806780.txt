#include<bits/stdc++.h>
using namespace std;
int cnt[1010];
main()
{
	int n,i,t;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&t);cnt[t]++;
	}
	int flag=0;
	for(i=0;i<=998;i++)
	{
		if(cnt[i]>0 && cnt[i+1]>0 && cnt[i+2]>0)
		{flag=1;break;}
	}
	if(flag)
	printf("YES\n");
	else printf("NO\n");
}
