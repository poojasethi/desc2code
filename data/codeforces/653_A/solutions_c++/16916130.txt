#include<bits/stdc++.h>
using namespace std;
int a[100];
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++) cin>>a[i];
	sort(a,a+n);
	int m=unique(a,a+n)-a;
	int f=0;
	for(int i=0;i<m-2;i++){
		if(a[i+2]-a[i+1]==1 && a[i+1]-a[i]==1) f=1;
	}
	if(f) cout<<"Yes"<<endl;
	else cout<<"No"<<endl; 
	return 0;
}
