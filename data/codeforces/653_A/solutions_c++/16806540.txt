#include <iostream>
using namespace std;

int main() {
int n;
cin>>n;
int a[n];
int p=0,q=0;
for(int i=0;i<n;i++)cin>>a[i];
for(int i=0;i<n;i++)
{p=0;q=0;
for(int j=0;j<n;j++)
{if(a[j]==a[i]-1)p=1;
else if(a[j]==a[i]+1)q=1;
if(p==1&&q==1){cout<<"YES"<<endl;return 0;}}}
cout<<"NO"<<endl;
	return 0;
}
