#include<bits/stdc++.h>

using namespace std;

int a[1111]={0};
int main()
{
	int n,x,i;
	scanf("%d",&n);
	while(n--&&scanf("%d",&x)==1)
		a[x]++;
	for(i=1;i<=998;i++)
		if(a[i]>=1&&a[i+1]>=1&&a[i+2]>=1)
		{
			printf("YES");
			return 0;
		}
	printf("NO");
	return 0;
}
