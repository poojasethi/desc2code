#include <cstdio>
const int N=2000;
bool have[N];
int main(){
    int n,a;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&a),have[a]=true;
    for(int i=2;i<N-1;i++){
        if(have[i-1]&&have[i]&&have[i+1]){
            printf("YES\n");
            return 0;
        }
    }
    printf("NO\n");
    return 0;
}
