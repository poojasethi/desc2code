#include<bits/stdc++.h>
using namespace std;

int main()
{
	int i,j,k,r,fl,n;
	cin>>n;
	int a[1001];
	for(i=0;i<1001;i++)
		a[i]=0;
	for(i=0;i<n;i++)
	{
		cin>>k;
		a[k]++;	
	}	
	for(i=0;i<999;i++)
		if(a[i]>0 && a[i+1]>0 && a[i+2]>0)
			break;
	if(i>=999)
		cout<<"NO"<<endl;
	else
		cout<<"YES"<<endl;
return 0;
}
