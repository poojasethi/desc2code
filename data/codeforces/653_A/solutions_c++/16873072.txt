#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
	bool visit[1100]={};
	int n,x;
	cin>>n;
	for(int i = 0; i < n; i++)
	{
		cin>>x;
		visit[x] = 1;
	}
	for(int i = 0; i < 1000; i++)
		if(visit[i] && visit[i+1] && visit[i+2])
		{
			printf("YES\n");return 0;
		}
	printf("NO\n");
	return 0;
}
