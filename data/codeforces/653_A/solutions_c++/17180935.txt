#include <bits/stdc++.h>
using namespace std;

int main() 
{
	int n;
	cin >> n;
	int a[n], i;
	bool b[1005] = {0};
	for(i = 0; i < n; ++i)
	{
		cin >> a[i];
		b[a[i]] = 1;
	}
	for(i = 1; i < 1000; ++i) {
		if(b[i] && b[i + 1] && b[i + 2])
			break;
	}
	if(i < 1000)
		cout << "YES";
	else
		cout << "NO";
	return 0;
}
