#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int has[1005];
int main()
{
	int n;
	int num;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>num;
		has[num]=1;
	}
	for(int i=1;i<=1000;i++)
	{
		if(has[i]&&has[i+1]&&has[i+2])
		{
			cout<<"YES"<<endl;
			return 0;
		}
	}
	cout<<"NO"<<endl;
}
