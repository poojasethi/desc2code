#!/usr/bin/env python


def read():
    raw_input()
    return map(int, raw_input().split())


def work(vList):
    N = len(vList)

    dp = [[0 for j in range(N)] for i in range(2)]

    dp[0][0] = 1
    dp[1][0] = 0
    
    for i in range(1, N):
        if vList[i - 1] < vList[i]:
            dp[0][i] = dp[0][i - 1] + 1
            dp[1][i] = dp[1][i - 1] + 1

        if i >= 2 and vList[i - 2] + 1 < vList[i]:
            dp[0][i] = max(dp[0][i], 1)
            dp[1][i] = max(dp[1][i], dp[0][i - 2] + 2)

        
        dp[0][i] = max(dp[0][i], 1)
        dp[1][i] = max(dp[1][i], 2)

    print max(min(N, max(dp[0]) + 1), max(dp[1]))


if __name__ == "__main__":
    work(read())
