n = input()
a = map(int,raw_input().split())
l,r,ans = [1]*n, [1]*n, 1

for i in xrange(1,n):
    if a[i] > a[i-1]: 
        l[i]+=l[i-1]
for i in xrange(n-2,0,-1):
    if a[i+1] > a[i]: 
        r[i]+=r[i+1]

if n == 1:
    ans = 1
else:
    ans = max(ans,max(l))
    if ans != n:
        ans += 1

for i in xrange(1,n-1):
    if a[i-1]+1 < a[i+1]:
        ans=max(ans, r[i+1] + l[i-1]+1)
print ans
