n, a = int(raw_input()), map(int, raw_input().split())
l, r, ans = [1]*n, [1]*n, 1
for i in xrange(1, n):
	if a[i-1] < a[i]: l[i]+=l[i-1]
	ans = max(ans, l[i])
for i in xrange(n-2, -1, -1):
	if a[i] < a[i+1]: r[i]+=r[i+1]
if ans < n: ans += 1
for i in xrange(1, n-1):
	if a[i-1]+1 < a[i+1]: ans = max(ans, l[i-1]+1+r[i+1])
print ans
