n, a = int(raw_input()), [10**9+1] + map(int, raw_input().split()) + [0]
l, r, ans = [0]+[1]*n+[0], [0]+[1]*n+[0], 1
for i in xrange(1, n+2):
	if a[i-1] < a[i]: l[i]+=l[i-1]
for i in xrange(n, 0, -1):
	if a[i] < a[i+1]: r[i]+=r[i+1]
for i in xrange(1, n+1):
	if a[i-1]+1 < a[i+1]: ans = max(ans, l[i-1]+1+r[i+1])
	else: ans = max(ans, max(l[i-1]+1, 1+r[i+1]))
print ans
