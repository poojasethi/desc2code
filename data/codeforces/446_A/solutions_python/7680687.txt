n=input()
r=map(int,raw_input().split())
R=[1]*n
L=[1]*n
a=1
for i in range(1,n):
    if r[i] > r[i-1]: L[i]+=L[i-1]
    a=max(a,L[i])

for i in range(n-2,0,-1):
    if r[i+1] > r[i]: R[i]+=R[i+1]

if a != n: a+=1

for i in range(1,n-1):
    if r[i-1]+1<r[i+1]: 
        a=max(a, R[i+1] + L[i-1]+1)
print a
