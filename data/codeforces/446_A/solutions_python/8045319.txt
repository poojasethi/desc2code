n = input()
v = list(map(int, raw_input().split()))
inc, dec = [1] * n, [1] * n

for i in range(1, n):
	if v[i-1] < v[i]: inc[i] = inc[i-1] + 1

for i in range(n-2, -1, -1):
	if v[i] < v[i+1]: dec[i] = dec[i+1] + 1

print max(min(n, max(max(inc), max(dec)) + 1), max([0]+[inc[i-1]+1+dec[i+1] for i in range(1, n-1) if v[i-1]+2<=v[i+1]]))
