n = int(raw_input())
a = [2000000000] + map(int, raw_input().split()) + [-2000000000]
l = [0 for i in xrange(n + 2)]
r = [0 for i in xrange(n + 2)]
for i in xrange(1, n + 1):
    l[i] = 1
    if a[i - 1] < a[i]:
        l[i] += l[i - 1]
for i in xrange(n, 0, -1):
    r[i] = 1
    if a[i] < a[i + 1]:
        r[i] += r[i + 1]
ans = min(n, max(l) + 1)
for i in xrange(1, n+1):
    if (a[i - 1] + 1 < a[i + 1]):
        ans = max(ans, l[i - 1] + r[i + 1] + 1)
print ans
