n = input()
v = list(map(int, raw_input().split()))
inc = [1 for i in range(0, n)]
dec = [1 for x in inc]

for i in range(1, len(v)):
	if v[i-1] < v[i]: inc[i] = inc[i-1] + 1

for i in range(len(v)-2, -1, -1):
	if v[i] < v[i+1]: dec[i] = dec[i+1] + 1

sol = max(max(inc), max(dec)) + 1
for i in range(0, len(v)):
	if i > 0 and i < len(v)-1 and v[i-1]+2 <= v[i+1]:
		sol = max(sol, inc[i-1] + 1 + dec[i+1])

print min(sol, n)
