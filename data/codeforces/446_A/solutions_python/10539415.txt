#!/usr/bin/python

n = int(raw_input())
data = [int(item) for item in raw_input().split()]
seq = []
start,end = (0,0)
while start < n:
    end = start + 1
    while end < n and data[end] > data[end-1]:
        end += 1
    seq.append((start,end-1,end-start))
    start = end
rlt = seq[0][1] - seq[0][0] + 1
for i in range(1,len(seq)):
    if (seq[i][1] - seq[i][0]) == 0 or (seq[i-1][1] - seq[i-1][0]) == 0:
        rlt = max(seq[i-1][2] + seq[i][2], rlt)
    else:
        if data[seq[i][0]] - 1 > data[seq[i-1][1]-1]:
            rlt = max(rlt, seq[i][2]+seq[i-1][2])
        else:
            rlt = max(rlt, seq[i][2]+1)
        if data[seq[i-1][1]] + 1 < data[seq[i][0]+1]:
            rlt = max(rlt, seq[i][2]+seq[i-1][2])
        else:
            rlt = max(rlt, seq[i-1][2]+1)
print rlt
