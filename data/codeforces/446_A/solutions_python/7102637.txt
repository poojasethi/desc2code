#!/usr/bin/env python

n = int(raw_input())
v = map(int, raw_input().split(' '))

res = min(2, n)

s = map(lambda x: 1, v)
t = map(lambda x: 1, v)
for i in xrange( +1, +n, +1):
    if v[i-1] < v[i]:
        s[i] += s[i-1]

for i in xrange(n-2, -1, -1):
    if v[i] < v[i+1]:
        t[i] += t[i+1]

for i in xrange(n):
    got = 1
    if i > 0 and i+1<n and v[i-1] + 1 >= v[i+1]:
        got += max(s[i-1], t[i+1])
    else:
        if i > 0: got += s[i-1]
        if i+1 < n: got += t[i+1]
    res = max(res, got)

print res
