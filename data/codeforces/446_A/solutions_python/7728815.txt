n=input()
a=map(int,raw_input().split())

l=[1]*n
r=[1]*n

MAX=1

for i in range(1,n):
    if (a[i]>a[i-1]):
        l[i]+=l[i-1]
    MAX=max(MAX,l[i])
    
if MAX!=n:
    MAX+=1

for i in range(n-2,0,-1):
    if (a[i+1]>a[i]):
        r[i]+=r[i+1]
        
for i in range(1,n-1):
    if (a[i-1]+1<a[i+1]):
        MAX=max(MAX,r[i+1]+l[i-1]+1)
        
print MAX
