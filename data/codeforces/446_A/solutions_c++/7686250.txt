#include <cstdio>
#include <iostream>
#define MAXN 100100
using namespace std;
int N,mx,taken,ar[MAXN],dn[MAXN];

int main(){
	scanf("%d",&N);
	for(int i=1 ; i<=N ; i++){
		scanf("%d",&ar[i]);
		dn[i]=1;
		if(ar[i-1] < ar[i]) dn[i]=dn[i-1]+1;
	}
	
	for(int i=N ; i ; i--){
		
		mx=max(max(mx , dn[i-1]+1) , taken+1);
		
		if(ar[i-1]+1 < ar[i+1])
			mx=max(mx , dn[i-1] + taken + 1);
		
		if(ar[i] < ar[i+1]) taken++;
		else 				taken=1;
	
	}
	printf("%d\n",mx);
	return 0;
}
