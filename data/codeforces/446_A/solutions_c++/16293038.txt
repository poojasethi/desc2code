//http://codeforces.com/problemset/problem/446/A
#include<bits/stdc++.h>
using namespace std;
int arr[100005],f[100005];
int main()
{
	int n;
	int ans=1;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>arr[i];
		f[i]=1;
		if(arr[i]>arr[i-1])
			f[i]=f[i-1]+1;
	}
	for(int i=1;i<=n;i++)
	{
		ans=max(ans,min(1+f[i],n));
		if(arr[i-f[i]+1]>arr[i-f[i]-1]+1)
			ans=max(ans,f[i]+f[i-f[i]]);
		if(arr[i-f[i]+2]>arr[i-f[i]]+1)
			ans=max(ans,f[i]+f[i-f[i]]);
	}
	cout<<ans<<endl;
	return 0;
}
