#include <iostream>
using namespace std;
int n,a[100005],s[100005],e[100005],mx;
int main()
{
  int i;
  cin>>n;
  a[n+1]=2000000069;
  for(i=1;i<=n;++i)
    {
      cin>>a[i];
      e[i]=(a[i-1]<a[i])?(e[i-1]+1):1;
      mx=max(mx,e[i]);
    }
  if(mx<n)
    ++mx;
  for(i=n;i;--i)
    {
      s[i]=(a[i]<a[i+1])?(s[i+1]+1):1;
      if(a[i+1]-a[i-1]>1)
	mx=max(mx,e[i-1]+s[i+1]+1);
    }
  cout<<mx;
  return 0;
}
