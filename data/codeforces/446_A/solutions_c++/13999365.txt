#include <iostream>
using namespace std;

int n,a[100009],f[100009],ans;

int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		f[i]=1;
		if(a[i]>a[i-1]) f[i]=f[i-1]+1;
	}	
	for(int i=1;i<=n;i++)
	{
		ans=max(ans,min(f[i]+1,n));
		if(a[i-f[i]+1]>a[i-f[i]-1]+1) ans=max(ans,f[i]+f[i-f[i]]);
		if(a[i-f[i]+2]>a[i-f[i]]+1)	ans=max(ans,f[i]+f[i-f[i]]);
	}
	cout<<ans;	return 0;
}
