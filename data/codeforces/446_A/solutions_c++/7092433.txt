#include<iostream>
using namespace std;
int i,j,n,a[100005],b[100005],c[100005],ans;
int main()
{
  cin>>n;
  for(i=1;i<=n;++i) cin>>a[i];
    for(i=1;i<=n;++i)
  if(a[i]>a[i-1]) b[i]=b[i-1]+1,ans=max(ans,b[i]); else b[i]=1;
  if(ans!=n) ++ans;
   for(i=n;i>=1;--i)
  {
    if(a[i]<a[i+1]) c[i]=c[i+1]+1; else c[i]=1;             
    if(a[i+1]-a[i-1]>1) ans=max(ans,b[i-1]+c[i+1]+1);                   
  }
  cout<<ans<<'\n';     
 return 0;   
}
