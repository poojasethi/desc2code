#include <iostream>
#include<algorithm>

using namespace std;

int a[100001],b[100001];
int n,x,q,z,f;

int main(){
   cin>>n;
   for(int k=1;k<=n;k++){
   cin>>a[k];  
   if(a[k]>a[k-1]){
      b[k]=b[k-1]+1;    
   }
   else b[k]=1;
   }
   for(int k=n;k>0;k--){
     q=max(q,b[k-1]+1);  
     q=max(q,x+1);
     if(k==n||k==1||a[k-1]+1<a[k+1]){
         q=max(q,b[k-1]+x+1);
     }
     if(a[k]<a[k+1]){
         x++;
     }else{x=1;}  
     
   }
   cout<<q<<endl;
}
