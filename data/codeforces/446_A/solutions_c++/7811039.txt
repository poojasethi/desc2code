#include <iostream>
#include <stdlib.h>
using namespace std;
long n;
long l[100001],cacah,t[100001];
long out;

int main(){
	
	cin>>n;
	
	for (long i=1;i<=n;i++){
		cin>>t[i];
		l[i]= (t[i]>t[i-1]) ? l[i-1]+1 : 1;
		out=max(out,l[i]);
	}
	
	for (long i=n;i>0;i--){
		out=max(out,l[i-1]+1);
		out=max(out,cacah+1);
		if (t[i+1]-t[i-1]>1) out=max(out,l[i-1]+cacah+1);
		if (t[i]<t[i+1]) cacah++;
		else cacah=1;
	}
	
	cout<<out<<endl;
	
	return 0;
}
