#include<iostream>
using namespace std;
int n,i,a[100005],h[100005],hh[100005],j,k,s,t;
main()
{
	cin>>n;
	for(;i<n;i++)
	{
		cin>>a[i];
		h[i]=t;
		if(i&&a[i-1]<a[i])t++;
		else t=1;
		//cout<<h[i]<<' ';
	}
	for(i=n-1,t=0;i>-1;i--)
	{
		hh[i]=t;
		if(i!=n-1&&a[i+1]>a[i])t++;
		else t=1;
		//cout<<hh[i]<<' ';
	}
	for(i=0;i<n;i++)
	{
		if(!i||i==n-1||a[i+1]-a[i-1]>1)t=h[i]+hh[i];
		else t= h[i]>hh[i]?h[i]:hh[i];
		if(s<t)s=t;
	}
	cout<<s+1<<endl;
}
