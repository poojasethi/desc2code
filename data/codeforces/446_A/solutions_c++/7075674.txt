#include<stdio.h>
#include<algorithm>
using namespace std;
int n,a[100001],mr[100001],ml[100001],ans;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1,j=1;i<=n;j=i)
	{
		while(j<n&&a[j+1]>a[j])++j;
		for(int k=i;k<=j;k++)mr[k]=j-k+1;
		for(int k=i;k<=j;k++)ml[k]=k-i+1;
		i=j+1;
	}
	for(int i=1;i<=n;i++)ans=max(ans,mr[i]);
	if(ans<n)++ans;
	for(int i=1;i<n-1;i++)
		if(a[i+2]>a[i]+1)
			ans=max(ans,mr[i+2]+ml[i]+1);
	printf("%d\n",ans);
	return 0;
}
