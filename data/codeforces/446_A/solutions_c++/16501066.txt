#include <iostream>
using namespace std;
int n,a[100005],s[100005],e[100005],mx;
int main()
{
	int i;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
		if (a[i-1]<a[i])
		e[i]=e[i-1]+1;
		else
		e[i]=1;
		mx=max(mx,e[i]);
	}
	if(mx<n)
		mx++;
	for(i=n;i>0;i--)
	{
		if (a[i+1]>a[i])
		s[i]=s[i+1]+1;
		else
		s[i]=1;
		if(a[i+1]-a[i-1]>1)
			mx=max(mx,e[i-1]+s[i+1]+1);
	}
	cout<<mx;
	return 0;
}
