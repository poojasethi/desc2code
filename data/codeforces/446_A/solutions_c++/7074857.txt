#include <cstdio>
int n,a[100011],i,sl[100011],sr[100011],ans;

int main()
{
	scanf("%d",&n);
	a[0]=1000000007;
	for (i=1; i<=n; ++i)
	{
		scanf("%d",&a[i]);
		if (a[i]>a[i-1]) sl[i]=sl[i-1]+1; else sl[i]=1;
		if (sl[i]+1>ans) ans=sl[i]+1;
	}
	a[n+1]=-1000000007;
	for (i=n; i>=1; --i)
	{
		if (a[i]<a[i+1]) sr[i]=sr[i+1]+1; else sr[i]=1;
	}
	for (i=2; i<n; ++i)
	if (a[i-1]+1<a[i+1])
	if (sl[i-1]+1+sr[i+1]>ans) ans=sl[i-1]+1+sr[i+1];
	if (ans>n) ans=n;
	printf("%d\n",ans);
	return 0;
}
