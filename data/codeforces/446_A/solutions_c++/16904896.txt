#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,a[100010]={0},i,m=0,l[100010]={0},r[100010]={0};
	cin>>n;
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(i=1;i<=n;i++)
	{
		if(a[i]<=a[i-1])
			l[i]=1;
		else
		{
			l[i]=l[i-1]+1;
			m=max(m,l[i]);
		}
	}
	if(m!=n)
		m++;
	for(i=n;i>=1;i--)
	{
		if(a[i]>=a[i+1])
			r[i]=1;
		else
			r[i]=r[i+1]+1;
	}
	for(i=1;i<=n;i++)
	{
		if(a[i-1]+1<a[i+1])
			m=max(m,l[i-1]+1+r[i+1]);
	}
	cout<<m<<endl;
	return 0;
}
