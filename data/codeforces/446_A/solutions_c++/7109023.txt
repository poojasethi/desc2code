#include<cstdio>
#include<algorithm>
using namespace std;
int a1[100000],a2[100000],a3[100000];
int main()
{
	int n,a,o=0;
	scanf("%d",&n);
	for(int i=0;i<n;++i)
		scanf("%d",a1+i);
	for(int i=1;i<n;++i)
	if(a1[i]>a1[i-1])
	{
		a2[i]=a2[i-1]+1;
		o=max(o,a2[i]);
	}
	else a2[i]=0;
	for(int i=n-2;~i;--i)
	if(a1[i]<a1[i+1])
	{
		a3[i]=a3[i+1]+1;
		o=max(o,a3[i]);
	}
	else a3[i]=0;
	for(int i=n-1;i;--i)
	if(a1[i-1]+1<a1[i+1])
		o=max(o,a2[i-1]+a3[i+1]+1);
	printf("%d\n",min(n,o+2));
}
