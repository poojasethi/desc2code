#include <iostream>
using namespace std;
int n;
int a[100005];
int l[100005];
int main() {
	cin>>n;
	for(int i=1;i<=n;i++) {
		cin>>a[i];
		if(a[i]>a[i-1]) l[i]=l[i-1]+1;
		else l[i]=1;
	}
	int ans=1;
	int r=0;
	for(int i=n;i;i--) {
		ans=max(ans,l[i-1]+1);
		ans=max(ans,r+1);
		if(i==1||i==n||a[i-1]+1<a[i+1]) ans=max(ans,l[i-1]+r+1);
		if(a[i]<a[i+1]) r++;
		else r=1;
	}
	cout<<ans<<endl;
	return 0;
}
