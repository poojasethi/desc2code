#include<bits/stdc++.h>
#define f(i,a,b) for(i=a;i<b;i++)
#define s int(1e5+2)
using namespace std;
int a[s],b[s],c[s];
int main()
{
    int n,i,j,mx=-1;
    cin>>n;
    a[n+1]=2e9;
    f(i,1,n+1){
        cin>>a[i];
        b[i]=a[i]>a[i-1]?b[i-1]+1:1;
        mx=max(mx,b[i]);
    }
    if(mx<n)mx++;
    for(i=n;i;i--){
        c[i]=a[i]<a[i+1]?c[i+1]+1:1;
        if(a[i+1]-a[i-1]>1)mx=max(mx,b[i-1]+c[i+1]+1);
    }
    cout<<mx;
}
