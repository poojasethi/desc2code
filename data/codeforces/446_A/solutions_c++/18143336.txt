#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll a[100005],l[100005],r[100005],ans=0;
int main(){
    int n;
    cin>>n;
    a[0]=(ll)1e10;a[n+1]=0;
    for(int i=1;i<=n;i++)
     {
        cin>>a[i];
        l[i]=a[i]>a[i-1]?l[i-1]+1:1;
        ans=max(ans,l[i]);
     }
     if(ans!=n)
     ans++;
     for(int i=n;i>=1;i--){
         r[i]=(a[i]<a[i+1])?r[i+1]+1:1;
       if(a[i-1]+1<a[i+1])
       ans=max(ans,1+l[i-1]+r[i+1]);
     }
     cout<<ans;
    return 0;
}
