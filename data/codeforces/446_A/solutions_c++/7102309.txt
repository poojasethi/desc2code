#include <iostream>
#include <algorithm>
using namespace std;
const int N=100100;
int n,a[N],b[N],c[N],t=1;
int main(){
  cin>>n;
  for(int i=0;i<n;++i)cin>>a[i];
  for(int i=0;i<n;++i){
    b[i]=c[i]=1;
    if(i>0){
      c[i]=2;
      if(a[i]>a[i-1])b[i]=max(b[i],b[i-1]+1),c[i]=max(c[i],c[i-1]+1);
    }
    if(i>1){
      if(a[i]>a[i-2]+1)c[i]=max(c[i],b[i-2]+2);
    }
    c[i]=max(c[i],b[i]);
    t=max(t,c[i]);
    if(i>0)t=max(t,b[i-1]+1);
  }
  cout<<t<<endl;
}
