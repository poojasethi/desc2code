#include <cstdio>
#include <algorithm>
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

int n,ans,tim,ansx,ansy,sum;
int a[5050];
int f[5050][5050];
int g[5050][5050];

int main() {
	scanf("%d",&n);
	REP(i,n) scanf("%d",&a[i]);
	REP(i,n) REP(j,n) {
		f[i][j]=f[i][j-1];
		g[i][j]=g[i][j-1];
		if (a[j]>a[i]) ++f[i][j];
		if (a[j]<a[i]) ++g[i][j];
	}
	ansx=ansy=1;
	ans=0;
	REP(j,n) REP(i,j-1) {
		int de=g[i][j-1]-g[i][i]+f[j][j-1]-f[j][i];
		int in=f[i][j-1]-f[i][i]+g[j][j-1]-g[j][i];
		if (a[i]>a[j]) ++de; else ++in;
		if (de-in>ans)
			ans=de-in,ansx=i,ansy=j,tim=1;
		else
			if (de-in==ans) ++tim;
	}
	ans=-ans;
	REP(j,n) REP(i,j-1)
		if (a[i]>a[j]) ++ans;
	printf("%d %d\n",ans,tim);
	return 0;
}
