#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 5010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,an,re,sum;
int s[N][N],a[N];
inline int C(int l,int r,int x) {
	return (r-l-1)-(s[r-1][x]-s[l][x]);
}
inline int D(int l,int r,int x) {
	return s[r-1][x-1]-s[l][x-1];
}
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%d",&a[i]),a[i]++,s[i][a[i]]++;
	For(i,1,n)For(j,1,n) s[i][j]+=s[i][j-1]+s[i-1][j]-s[i-1][j-1];
	For(i,1,n) sum+=i-1-s[i-1][a[i]-1];
	an=N*N;
	For(i,1,n)For(j,i+1,n) {
		int A=sum+C(i,j,a[i])-D(i,j,a[i])+D(i,j,a[j])-C(i,j,a[j]);
		if (a[i]<a[j]) A++; else A--; 
		if (A<an) an=A,re=1;
		else if (A==an) re++;
	}
	printf("%d %d\n",an,re);
	//for(;;);
	return 0;
}
