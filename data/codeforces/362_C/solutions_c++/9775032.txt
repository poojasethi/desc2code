#include<iostream>
#include<fstream>
#include<cstring>

using namespace std;
int aib[5005],aib1[5005],i,n,a[5005],sol=0,nr,inv,j,p1,p2;

void add(int poz) {
	while (poz<=n) { aib[poz]+=1; poz+=( poz&(-poz) ); }
}

int sum(int poz) {
	int r=0;
	while (poz>0) { r+=aib[poz]; poz-=( poz&(-poz) ); }
 return(r);
}

int main(void) {
		// ifstream cin("file.in");
	cin>>n;
	for (i=1; i<=n; ++i) { cin>>a[i]; a[i]+=1; }
	add(a[n]);
	for (i=n-1; i>=1; --i) {
		memset(aib,0,sizeof(aib));
		add(a[i]);
		for (j=i+1; j<=n; ++j) {
			add(a[j]); int ef=sum(a[i]-1)-sum(a[j]-1);
			if (ef>sol) { sol=ef; p1=i; p2=j; nr=1; }
			else if (ef==sol) ++nr;
		}
	}
	swap(a[p1],a[p2]);
	memset(aib,0,sizeof(aib));
	for (i=n; i>=1; --i) {
		add(a[i]);
		inv+=sum(a[i]-1);
	}
	cout<<inv<<" "<<nr;
		//getch();
	return(0);
}
