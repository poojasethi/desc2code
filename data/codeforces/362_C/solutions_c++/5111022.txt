#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int a[6000];
int mx[6000];
int my[6000];
int d[5010][5010];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			d[i][j]=d[i - 1][j]+(a[i] > j);
		//      cout<<d[i][j]<<" ";
        }
       // cout<<endl;
    }
	int minak=0,x=-1,y=-1,b=0;
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
		{
			int miin=d[j][a[i]]-d[i][a[i]]+d[i - 1][a[j]]-d[j][a[j]];
			if(miin<minak)
			{
				minak=miin;
				x=i;
				y=j;
				b=1;
			}
			else if(miin==minak)
				b++;
		}
	swap(a[x],a[y]);
	int minn=0;
	for (int i=0;i<n;i++)
		for (int j=0;j<i;j++)
			if (a[i]<a[j])
				minn++;
	cout << minn << " " << b << endl;
//	system("pause");
}
