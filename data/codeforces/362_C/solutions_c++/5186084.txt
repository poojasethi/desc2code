#include <algorithm>
#include <stdio.h>
#define  maxn      5003
using    namespace std;

int n,ans,sum,cnt;
int ar[maxn];
int dp[maxn][maxn];

int main()
{
  scanf("%d",&n);
  for(int i=1 ; i<=n ; i++)
  {
    scanf("%d",&ar[i]);
    ar[i]++;
    for(int j=1 ; j<=n ; j++)
      dp[i][j]=dp[i-1][j]+(ar[i-1]>j);
    sum+=dp[i][ar[i]];
  }
  ans=sum;
  for(int i=1 ; i<=n ; i++) 
    for(int j=i+1 ; j<=n ; j++)
      if(ar[i]>ar[j])
      {
        int diff=dp[i][ar[i]]+dp[j][ar[j]]-dp[i][ar[j]]-dp[j][ar[i]];
        int tmp=sum-2*diff+1;
        if(tmp<ans)          { ans=tmp; cnt=1; }
        else if(tmp==ans)    cnt++;
      }
  printf("%d %d\n",ans,cnt);
  return 0;
}
