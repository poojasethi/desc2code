#include <cstdio>
using namespace std;
const int maxn=5010,maxr=13000010;
int a[maxn],f[maxr],l[maxn][maxn];
int main(){
    int n,i,j,s=0,x,y,z;
    scanf("%d",&n);
    for (i=1;i<=n;i++) scanf("%d",&a[i]);
    for (i=1;i<=n;i++){
        for (j=0;j<n;j++) l[i][j]=l[i-1][j];
        for (j=0;j<a[i];j++) l[i][j]++;
    }
    for (i=1;i<=n;i++) s+=l[i][a[i]];
    for (i=1;i<n;i++)
        for (j=i+1;j<=n;j++) if (a[i]>a[j]){
            x=l[j][a[j]]-l[i-1][a[j]];
            y=l[j][a[j]]-l[i][a[j]]-l[j][a[i]]+l[i][a[i]];
            z=l[j][a[i]]-l[i][a[i]];
            f[s+z-x-y]++;
        }
    for (i=0;i<maxr;i++) if (f[i]){
        printf("%d %d\n",i,f[i]);
        return 0;
    }
    return 0;
}
