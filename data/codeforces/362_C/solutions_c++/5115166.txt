#include <cstdio>
const int maxn = 5000;
int a[maxn],f[maxn][maxn];
int main()
{
	int n,cnt = 0;
	scanf("%d",&n);
	for(int i = 0; i < n; ++i) scanf("%d",a + i);
	for(int i = 1; i < n; ++i)
		for(int j = 0; j < i; ++j)
			if(a[j] > a[i]) ++cnt;
	for(int j = 0; j < n; ++j) f[0][j] = (a[0] < j);
	for(int i = 1; i < n; ++i)
		for(int j = 0; j < n; ++j)
			f[i][j] = f[i - 1][j] + (a[i] < j);
	int ans = cnt,tmp,t;
	for(int i = 0; i < n - 1; ++i)
		for(int j = i + 1; j < n; ++j){
			tmp = cnt + 2 * (f[j][a[j]] - f[i][a[j]] + f[i][a[i]] - f[j][a[i]]) + 1;
			if(tmp < ans){
				t = 1; ans = tmp;
			}
			else
				if(tmp == ans) ++t;
		}
	printf("%d %d\n",ans,t);
	return 0;
}
