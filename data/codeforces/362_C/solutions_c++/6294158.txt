#include <iostream>
using namespace std;
const int N = 5e3 + 3, oo = 1e9;

int n, sum, a[N], dp[N][N];

int main() {
	ios::sync_with_stdio(false);
	cin >> n;
	for (int i = 1; i <= n; i++) {
		cin >> a[i], a[i]++;
		for (int j = 1; j <= n + 1; j++)
			dp[i][j] = dp[i - 1][j] + (a[i] < j);
		for (int j = 1; j < i; j++)
			if (a[j] > a[i])
				sum++;
	}
	int mn = oo, cnt = 0;
	for (int i = 1; i <= n; i++)
		for (int j = i + 1; j <= n; j++)
			if (a[i] > a[j]) {
				int A = dp[j - 1][a[j]] - dp[i][a[j]], B = dp[j - 1][a[i]] - dp[i][a[i]] - A;
				int res = sum - 2 * B - 1;
				if (res < mn)
					mn = res, cnt = 1;
				else if (res == mn)
					cnt++;
			}
	cout << mn << ' ' << cnt << '\n';
	return 0;
}
