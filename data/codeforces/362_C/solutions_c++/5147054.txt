#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int n,a[6100],res,ans,tot,num,f[5100][5100],i,j;
int calc(int x,int l,int r,int v){
	int res=f[x][r]-f[x][l-1];
	if(!v)res=r-l+1-res;
	return res;
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i)scanf("%d",&a[i]);
	for(i=1;i<=n;++i){
		for(j=1;j<=n;++j)f[i][j]=f[i][j-1]+(a[j]>a[i]);
		tot+=f[i][i];
	}
	ans=1<<30;
	for(i=1;i<=n;++i){
		for(j=i+1;j<=n;++j){
			res=tot-calc(i,i+1,j-1,0)+calc(i,i+1,j-1,1)-
				calc(j,i+1,j-1,1)+calc(j,i+1,j-1,0);
			if(a[i]<a[j])res++;
			else res--;
			if(res<ans)ans=res,num=1;
			else if(res==ans)num++;
		}
	}
	printf("%d %d\n",ans,num);
}
