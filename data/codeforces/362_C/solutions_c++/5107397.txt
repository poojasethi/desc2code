#include<iostream>
#include<cstdio>
#include<cstring>

int a[5001][5001];
int b[5001];
int main(){
	int n;
	scanf("%d",&n);
        for (int i=0; i<n; ++i) scanf("%d",&b[i]);
	for (int i=1; i<n; ++i)
		for (int j=i-1; j>=0; --j)
			if (b[j]>b[i]) a[i][j]=a[i][j+1]+1;
			else a[i][j]=a[i][j+1];
	int max=0,sum=0,sum1=0;
	for (int i=n-2; i>=0; --i){
		int l=0;
		int ll=0;
		for (int j=i+1; j<n; ++j){
			if (b[j]>b[i]) ++l;
			else ++ll;
			if (b[j]<b[i]){
				int x=j-i-a[j][i];
				x=ll-1-x;
				if (a[j][i]+x-l>max) max=a[j][i]+x-l,sum=1;
				else if (a[j][i]+x-l==max) ++sum;
			}
		}
	}
	for (int i=0; i<n; ++i) sum1+=a[i][0];
	printf("%d %d\n",sum1-max,sum);
	return 0;
}
