#include <iostream>
#include <cstring> 
using namespace std;
int n,a[5001],arr[5001];
inline void modify (int t)
{
	for (;t>0;t=t-(t&-t)) arr[t]++;
}
inline int query (int t)
{
	int re=0;
	for (;t<=n;t=t+(t&-t)) re=re+arr[t];
	return re;
}
int main()
{
	int i,j,ans=0,s=0,ways=0;
	cin>>n;
	for (i=1;i<=n;i=i+1)
	{
		cin>>a[i];
		a[i]=a[i]+1;
	}
	for (i=1;i<=n;i=i+1)
	{
		memset(arr,0,sizeof(arr));
		for (j=i+1;j<=n;j=j+1)
		{
			if (a[i]<a[j]) continue;
			s=s+1;
			int tmp=1+2*query(a[j]);
			if (tmp>ans) {ans=tmp;ways=1;}
			else if (tmp==ans) ways++;
			modify(a[j]);
		}
	}
	cout<<s-ans<<" "<<ways<<endl;
	return 0;
}
