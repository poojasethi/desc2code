#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int a[5010];
int n;
struct bit
{
	int s[5010];
	inline void add(int x)
	{
		for (; x; x-=(x&(-x))) s[x]++;
	}
	inline int sum(int x)
	{
		int res=0;
		for (; x<=n; x+=(x&(-x))) res+=s[x];
		return res;
	}
}seq;
int main()
{
	scanf("%d",&n);
	for (int i=1; i<=n; i++) scanf("%d",&a[i]),a[i]++;
	int tot=0,ans=0,ways=0;
	for (int i=1; i<n; i++)
	{
		memset(seq.s,0,sizeof(seq.s));
		for (int j=i+1; j<=n; j++)
		{
			if (a[i]<a[j]) continue;
			tot++;
			int temp=1+2*seq.sum(a[j]);
			if (temp>ans)
			{
				ans=temp;
				ways=1;
			}
			else if (temp==ans) ways++;
			seq.add(a[j]);
		}
	}
	printf("%d %d\n",tot-ans,ways);
	return 0;
}
