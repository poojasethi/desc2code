#include <stdio.h>
#include <stdlib.h>
#define N 5050

int a[N],num[N][N],pos[N];

int main(void)
{
	int i,j,times=0,plan=0,n,tot=0,min;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		pos[a[i]]=i;
	}
	for(i=0;i<n;i++)
		for(j=n-1;j>=0;j--)
			num[i][j]+=(a[j]<i?1:0)+num[i][j+1];
	for(i=0;i<n;i++)tot+=num[i][pos[i]];
	min=tot;
	for(i=0;i<n;i++)
		for(j=0;j<i;j++)
			if(a[j]>a[i])
			{
				times=tot+2*(num[a[i]][j]-num[a[i]][i]-num[a[j]][j]+num[a[j]][i])-1;
				if(times<min)
				{
					min=times;
					plan=1;
				}
				else if(times==min) plan++;
			}
	printf("%d %d\n",min,plan);
	return 0;
}
