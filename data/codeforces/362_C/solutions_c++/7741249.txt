#include <stdio.h>
#include <string.h>
int n, f[5010][5010], a[5010], old, ans, cnt;

int main() {
	int i, j, t;
	scanf("%d", &n);
	for (i = 1; i <= n; i++) scanf("%d", &a[i]);
	for (i = 1; i <= n; i++)
		for (j = 1; j <= n; j++)
			f[i][j] = f[i-1][j] + (a[i] < j);
	for (i = 1; i <= n; i++)
		for (j = i + 1; j <= n; j++)
			old += (a[i] > a[j]);
	ans = old;
	for (i = 1; i <= n; i++) {
		for (j = i + 1; j <= n; j++) {
			if (a[i] < a[j]) continue;
			t = old + f[j][a[j]] - f[i][a[j]] - (j - i - 1 - f[j][a[j]] + f[i][a[j]]);
			t -= f[j-1][a[i]] - f[i][a[i]];
			t += j - i - 1 - f[j-1][a[i]] + f[i][a[i]];
			t--; 
			if (t < ans) {
				ans = t;
				cnt = 1;
			} 
			else if (t == ans) cnt++;
		}
	}
	printf("%d %d\n", ans, cnt);
	return 0;
}
