#include <cstdio>

int a[5005], dp[5005][5005], n, tot, t, way, temp;

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) scanf("%d", a + i);
	for (int i = 1; i <= n; ++i) {
		for (int j = 1; j <= n; ++j)
			dp[i][j] = dp[i - 1][j] + (a[i] > a[j] ? 1 : 0);
		tot += dp[i][i];
	}
	t = tot;
	for (int i = 1; i < n; ++i)
		for (int j = i + 1; j <= n; ++j)
			if (a[i] > a[j]) {
				temp = tot + 2 * (dp[j][i] - dp[i][i] - dp[j][j] + dp[i - 1][j]) + 1;
				if (temp < t) { t = temp; way = 1;}
				else if (temp == t) ++way;
			}
	printf("%d %d\n", t, way);
	return 0;
}
