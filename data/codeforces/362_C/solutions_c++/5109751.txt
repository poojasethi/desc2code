#include<cstdio>
int x[5005],dt[5005][5005];
int main() {
  int n,all=0,mn=1e9,nm,rc;
  scanf("%d",&n);
  for(int i=1;i<=n;i++) {
    scanf("%d",&x[i]);
    x[i]++;
  }
  for(int i=1;i<=n;i++){
    for(int j=i-1;j>0;j--)
      if(x[i]>x[j]) 
    dt[i][j]=dt[i][j+1]+1;
      else
    dt[i][j]=dt[i][j+1];
    for(int j=i+1;j<=n;j++)
      if(x[i]>x[j]) {
    dt[i][j]=dt[i][j-1]+1;
    all++;
      }
      else
    dt[i][j]=dt[i][j-1];
  }
  //printf("%d\n",all);
  for(int i=1;i<=n;i++)
    for(int j=i+1;j<=n;j++) {
      rc=all-2*dt[i][j]+2*dt[j][i]+(x[i]>x[j]?1:-1);
      if(rc<mn) {
    mn=rc;
    nm=1;
      }
      else if(rc==mn) nm++;
    }
  printf("%d %d",mn,nm);
}
