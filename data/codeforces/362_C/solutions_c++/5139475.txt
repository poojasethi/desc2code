#include <iostream>
#include <cstdio>

using namespace std;

int dp[5001][5001];
int arr[5001];

int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;++i)
		cin>>arr[i];
	for(int j=0;j<n;++j)
		if(arr[0]<j) dp[0][j]=1;
	for(int j=0;j<n;++j)
		for(int i=1;i<n;++i)
		{
			if(arr[i]<j) ++dp[i][j];
			dp[i][j]+=dp[i-1][j];
		}
//	for(int i=0;i<n;++i)
//		for(int j=0;j<n;++j)
//			cout<<"0----"<<i<<" <"<<j<<" :"<<dp[i][j]<<endl;
	int sum=0;
	for(int i=0;i<n;++i)
		sum+=i-dp[i][arr[i]];
	int Max=sum;
	int cnt=0;
	for(int i=0;i<n;++i)
		for(int j=i+1;j<n;++j)
		{
			int tmp=1+sum-(dp[j][arr[i]]-dp[i][arr[i]])+(j-i-1-(dp[j][arr[i]]-dp[i][arr[i]]))+(dp[j][arr[j]]-dp[i][arr[j]])-(j-i-1-(dp[j][arr[j]]-dp[i][arr[j]]));
			if(tmp<Max) 
			{
				Max=tmp;
				cnt=1;
			}
			else if(tmp==Max) ++cnt;
		}
	cout<<Max<<" "<<cnt<<endl;
	return 0;
}
