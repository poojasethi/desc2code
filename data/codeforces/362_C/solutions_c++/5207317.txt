#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>

using namespace std;

const int N = 5005;
int C[N], n, a[N];
int lowbit(int x) {return x & (-x); }
void add(int x) {
	for(;x<=n;x+=lowbit(x)) C[x] ++;
}
int sum(int x) {
	int ret = 0;
	for(;x;x-=lowbit(x)) ret += C[x];
	return ret;
}
int main() {
	cin >> n;
	int _max = -1, cnt = 0, tot = 0;
	for(int i=1;i<=n;++i) cin >> a[i], a[i]++;
	for(int i=1;i<=n;++i) {
		memset(C, 0, sizeof C);
		for(int j=i+1;j<=n;++j) {
			if( a[i] > a[j] ) {
				tot ++ ;
				int tmp = sum(a[i]) - sum(a[j]);
				if( tmp == _max ) cnt ++;
				if( tmp > _max ) {
					_max = tmp; cnt = 1;
				}
				add(a[j]);
			}
		}
	}	
	tot -= 2*_max + 1;
	cout << tot << " " << cnt << endl;
	return 0;
}
