#include<cstdio>
int a[5005], s, cret, ret, cnt, dp[5005][5005];
int main()
{
	int n = 0;
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		scanf("%d", &a[i]);
	for (int i = 0; i < n; i++)
	{
		for (int j = i - 1; j >= 0; j--) 
			dp[i][j] = dp[i][j + 1] + (a[i]>a[j]);
		for (int j = i + 1; j < n; j++) 
			dp[i][j] = dp[i][j - 1] + (a[i]>a[j]);
	}
	for (int i = 0; i < n; i++)
	{
		for (int j = i + 1; j<n; j++)
		{
			if (a[i]>a[j])
				s++;
		}
	}
	ret = 1 << 30;
	for (int i = 0; i < n; i++)
	{
		for (int j = i + 1; j < n; j++)
		{
			cret = s - 2 * dp[i][j - 1] + 2 * dp[j][i + 1];
			if (a[i]>a[j])
				cret--;
			else
				cret++;
			if (cret < ret)
			{ ret = cret; cnt = 1; }
			else if (cret == ret)
				cnt++;
		}
	}
	printf("%d %d\n", ret, cnt);
	return 0;
}
