#include <algorithm>
#include <stdio.h>
#define  maxn      5003
#define  inf       1 << 30
using    namespace std;

int n,ans,cnt,sum;
int ar[maxn];
int dp[maxn][maxn];

int main()
{
  scanf("%d",&n);
  for(int i=1 ; i<=n ; i++)
    scanf("%d",&ar[i]);
  for(int i=1 ; i<=n ; i++)
  {
    for(int j=i+1 ; j<=n ; j++)
      dp[i][j]=dp[i][j-1]+(ar[j]<ar[i]);
    for(int j=i-1 ; j>=1 ; j-- )
      dp[i][j]=dp[i][j+1]+(ar[j]<ar[i]);
    sum+=dp[i][n];
  }
  ans=inf;
  for(int i=1 ; i<n ; i++)
    for(int j=i+1 ; j<=n ; j++ )
      if(ar[j]<ar[i])
      {
        int tmp=sum-2*(dp[i][j-1]-dp[j][i+1])-1;
        if(tmp<ans)               ans=tmp,cnt=1;
        else if(tmp==ans)         cnt++;
      }
  printf("%d %d\n",ans,cnt);
  return 0;
}
