#include <cstdio>
#include <algorithm>
#define F(l,r,x) (f[r][x]-f[l-1][x])
using namespace std;
int N,S,a[5005],f[5005][5005],x=1e9,y;
int main()
{
	scanf("%d",&N);
	for (int i=1; i<=N; i++) scanf("%d",&a[i]);
	for (int i=1; i<=N; i++)
	{
		for (int j=0; j<N; j++) f[i][j]=f[i-1][j]+(a[i]>j);
		S+=f[i][a[i]];
	}
	for (int i=1; i<=N; i++)
		for (int j=i+1; j<=N; j++)
		{
			int u=F(i+1,j-1,a[i]),v=F(i+1,j-1,a[j]),t=S-(j-i-1-u)+u-v+(j-i-1-v);
			if (a[i]<a[j]) t++; else t--;
			if (t<x) x=t,y=1; else if (t==x) y++;
		}
	printf("%d %d\n",x,y);
}
