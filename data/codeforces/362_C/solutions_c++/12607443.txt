using namespace std;
const long long A=100000000000000LL,N=5002;

int i,j,k,n,m,b[N][N],a[N],ot[2]={A},o;

int main(){
    cin>>n;
    for(i=0;i<n;i++)scanf("%d",&a[i]);
    for(i=0;i<n;i++){
        k=0;
        for(j=0;j<n;j++)k+=(a[j]>i),b[i][j]=k;
    }
    for(i=0;i<n;i++)o+=b[a[i]][i];
    for(i=0;i<n;i++)for(j=i+1;j<n;j++)if(a[i]>a[j]){
        k=o-2*(b[a[i]][i]+b[a[j]][j]-b[a[j]][i]-b[a[i]][j])-1;
        if(k<ot[0])ot[0]=k,ot[1]=0;
        if(k==ot[0])ot[1]++;
    }
    cout<<ot[0]<<" "<<ot[1];
}
