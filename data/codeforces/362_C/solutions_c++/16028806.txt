#include <iostream>
#include <algorithm>

using namespace std;

int a[5010];
int dp[5010][5010];
int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++)
			dp[i][j] = dp[i - 1][j] + (a[i] > j);
	int best = 0, besti = -1, bestj = -1, ans2 = 0;
	for (int i = 0; i < n; i++)
		for (int j = i + 1; j < n; j++)
		{
			int good = dp[j][a[i]] - dp[i][a[i]] + dp[i - 1][a[j]] - dp[j][a[j]];
			if (good < best)
			{
				best = good;
				besti = i;
				bestj = j;
				ans2 = 1;
			}
			else if (good == best)
				ans2++;
		}
	swap(a[besti], a[bestj]);
	int ans = 0;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < i; j++)
			if (a[i] < a[j])
				ans++;
	cout << ans << " " << ans2 << endl;
}
