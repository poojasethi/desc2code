#include <stdio.h>
const int N=5050;
int n,t,v=-1,w,p[N],c[N],d[N],l[N];
main(){
  scanf("%d",&n);
  for(int i=0;i<n&&1==scanf("%d",&p[i]);++i){
    int s=0;
    for(int j=p[i];j>0;j-=j&-j)s+=c[j];
    l[i]=s,t+=i-s;
    for(int j=p[i]+1;j<=n;j+=j&-j)++c[j];
  }
  for(int i=0;i<n;++i){
    int a=0;
    for(int j=i+1;j<n;++j)if(p[j]<p[i]){
      int b=l[j];
      for(int u=p[j];u>0;u-=u&-u)b-=d[u];
      if(a-b>v)v=a-b,w=1;else if(a-b==v)++w;
      ++a;
    }
    for(int u=p[i] + 1;u<=n;u+=u&-u)++d[u];
  }
  printf("%d %d\n",t-v-v-1,w);
}
