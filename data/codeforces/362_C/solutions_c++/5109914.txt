#include <iostream>
using namespace std;
int a[5000],c[5000][5000],s,cnt,m=20000000;
int main()
{	
	int i,j,k,t,n;
	cin>>n;
	for (i=0;i<n;i++) cin>>a[i];
	for (i=0;i<n;i++)
	{
		for (j=i-1;j>=0;j--)
			if (a[j]>a[i]) c[i][j]=c[i][j+1]+1;
			else c[i][j]=c[i][j+1];
		for (j=i+1;j<n;j++)
			if (a[j]>a[i]) c[i][j]=c[i][j-1]+1;
			else c[i][j]=c[i][j-1];
		s+=c[i][0];
	}
	for (i=0;i<n;i++)
		for (j=i+1;j<n;j++)
		{
			t=s+c[i][j-1]-c[j][i+1]+c[i][j]-c[j][i];
			if (t<m) {m=t;cnt=1;}
			else if (t==m) cnt++;
		}
	cout<<m<<" "<<cnt<<endl;
	return 0;
}
