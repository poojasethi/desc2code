#include <iostream>
#include <algorithm>
using namespace std;
struct node
{
    __int64 k,c;
}t[101];
__int64 f[101];
bool cmp(node a,node b)
{
    return a.c<b.c;
}
int main()
{
    int n,i,m,j;
    __int64 cnt=0,sum=0;
    cin>>n;
    for(i=0;i<n;++i) cin>>t[i].k>>t[i].c;
    sort(t,t+n,cmp);
    cin>>m;
    for(i=0;i<m;++i) cin>>f[i];
    for(i=0,j=0;i<n&&j<m;)
    {
        if(cnt+t[i].k<=f[j])
        {
            cnt+=t[i].k;
            sum+=t[i].k*t[i].c*(j+1);
            ++i;
        }
        else
        {
            sum+=(f[j]-cnt)*t[i].c*(j+1);
            t[i].k=cnt+t[i].k-f[j];
            cnt=f[j++];
        }
    }
    for(;i<n;++i) sum+=t[i].k*t[i].c*(j+1);
    cout<<sum<<endl;
    return 0;
}
