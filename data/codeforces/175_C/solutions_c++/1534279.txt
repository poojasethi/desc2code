#include<iostream>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

typedef long long ll;

const ll INF=1LL<<61;

struct mato{
	ll num,score;
	bool operator<(const mato &M)const{ return score<M.score; }
};

int main(){
	int n; cin>>n;
	mato M[100];
	rep(i,n) cin>>M[i].num>>M[i].score;
	sort(M,M+n);

	int m; cin>>m;
	ll chg[101];
	rep(j,m) cin>>chg[j];
	chg[m]=INF;

	ll ans=0,total=0;
	for(int i=0,j=0;i<n;i++){
		while(M[i].num>0){
			ll delta=min(chg[j]-total,M[i].num);
			M[i].num-=delta;
			total+=delta;
			ans+=delta*(j+1)*M[i].score;
			if(chg[j]<=total) j++;
		}
	}
	cout<<ans<<endl;

	return 0;
}
