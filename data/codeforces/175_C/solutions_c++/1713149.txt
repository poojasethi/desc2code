#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;

typedef long long LL;

const int maxn = 1005;

LL tot[maxn] , data[maxn] , n , t;

int main() {
    cin >> n;
    for ( int i = 0; i < n; i++ ) {
        int c , k;
        cin >> k >> c;
        tot[c] += k;
    }
    cin >> t;
    for ( int i = 0; i < t; i++ ) cin >> data[i];
    LL ans = 0 , p = 0 , i = 0 , j = 1;
    while ( i <= 1000 ) {
        if ( p == data[j-1] ) j++;
        if ( tot[i] == 0 ) i++;
        else {
            LL a;
            if ( tot[i] + p <= data[j-1] || j == t+1 ) a = tot[i];
            else a = data[j-1] - p;
            tot[i] -= a;
            p += a;
            ans += LL(a) * j * i;
        }
    }
    cout << ans << endl;
    return 0;
}
