#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
long long n,m;
long long p[105],ans=0;
struct note{long long k,c;}a[105];
bool cmp(const note i,const note j){return i.c<j.c;}
int main()
{
	cin>>n;
	for (int i=0;i<n;i++) cin>>a[i].k>>a[i].c;
	sort(a,a+n,cmp);
	cin>>m; for (int i=0;i<m;i++) cin>>p[i]; p[m]=1000000000000000ll;
	long long i=0,j=0,x=0;
	while (i<n && j<=m){
		if (p[j]-x<=a[i].k){
			long long z=p[j]-x;
			a[i].k-=z; x=p[j];
			ans+=z*a[i].c*(j+1); j++;
			if (!a[i].k) i++;
			}
		else{
			long long z=a[i].k;
			x+=z; ans+=z*a[i].c*(j+1); i++;
			}
		}
	cout<<ans<<endl; return 0;
}
