#include<iostream>
#include<algorithm>

#define int long long
using namespace std;

main()
{
	int n,t,p[101],u[101];
	pair<int,int> c[101];
	cin>>n;
	for (int i=0;i<n;i++)
		cin>>c[i].second>>c[i].first;
	sort(c,c+n);
	cin>>t;
	for (int i=0;i<t;i++)
		cin>>p[i];
	p[t] = 2000000ll*100000ll*100000ll;
	u[0] = p[0];
	for (int i=1;i<=t;i++)
		u[i] = p[i] - p[i-1];
	
	int ans = 0;
	int cu = 0;
	for (int i=0;i<n;i++)
	{
		int mn = min(u[cu], c[i].second);
		ans += mn * (cu+1) * c[i].first;
		u[cu] -= mn;
		c[i].second -= mn;
		if (c[i].second > 0)
			i--;
		if (u[cu] == 0)
			cu++;
	}
	cout<<ans<<endl;
	return 0;
}
