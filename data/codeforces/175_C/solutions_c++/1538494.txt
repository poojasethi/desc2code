#include<cstdio>
#include<iostream>
#include<algorithm>

using namespace std;

#define LL long long
#define MAXT 100
#define MAXN 100

LL n, t, ans, p[MAXT+5];
pair<LL,LL> a[MAXN+5];

int main()
{
	LL i, k, temp;
	scanf("%d", &n);
	for (i = 0; i < n; i++)
		scanf("%I64d %I64d", &a[i].second, &a[i].first);
	sort(a,a+n);
	scanf("%d", &t);
	for (i = 0; i < t; i++)
		scanf("%I64d", &p[i]);
	p[t] = 10000000000000;
	for (i = t; i > 0; i--) p[i] -= p[i-1];
	
	k = 0; 
	for (i = 0; i < n; i++)
	{
		while (a[i].second)
		{
			temp = min(a[i].second, p[k]);
			ans += (k+1)*a[i].first*temp;
			p[k] -= temp;
			a[i].second -= temp;
			if (!p[k]) ++k;
		}
	}
	printf("%I64d\n", ans);
}
