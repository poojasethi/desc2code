#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

pair<int,int>   s[110];
int n,t,x,y;
long long p[110],ans,now,ak;

int main()
{
    cin>>n;
    for (int i=0;i<n;i++)
    {
        cin>>x>>y;
        s[i]=make_pair(y,x);
    }
    cin>>t;
    for (int i=0;i<t;i++)   cin>>p[i];
    p[t]=1000000000000000ll;
    sort(s,s+n);
    
    int i=0,j=0;
    while (i<n)
    {
        ak=min(p[j],now+s[i].second);
        ans+=(ak-now)*s[i].first*(j+1);
        s[i].second-=ak-now;
        now=ak;
        if (!s[i].second)   i++;
        if (now==p[j])  j++;
    }
    
    cout<<ans<<endl;
}
