#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
typedef pair <ll, ll> pii;
const int N = 110;
const ll INF = 1e13;
int n, k;
ll p[N], ans;
pii a[N];
int main()
{
  cin >> n;
  for(int i = 0; i < n; i++)
    cin >> a[i].second >> a[i].first;
  sort(a, a + n);
  cin >> k;
  for(int i = 0; i < k; i++)
    cin >> p[i];
  p[k++] = INF;
  int t = 0;
  ll last = 1;
  for(int i = 0; i < n; i++)
  {
    ll r = last + a[i].second;
    while(p[t] < r)
    {
      ans += 1ll * (p[t] - last + 1) * (t + 1) * a[i].first;
      last = p[t] + 1;
      t++;
    }
    if (last < r)
    {
      ans += 1ll * (r - last) * (t + 1) * a[i].first;
      last = r;
    }
  }
  cout << ans;
  return 0;
}
