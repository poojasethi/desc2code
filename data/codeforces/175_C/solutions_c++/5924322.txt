#include<bits/stdc++.h>
using namespace std;

int main()
{
    pair<long long,long long> a[105];
    long long p[105], f=1, d=0, points=0; int n, t, pIndex=0;
    cin>>n; for(int i=0;i<n;i++) { cin>>a[i].second>>a[i].first; }
    cin>>t; for(int i=0;i<t;i++) { cin>>p[i]; }
    sort(a,a+n);
    for(int i=0;i<n;i++){
        while((pIndex<t)&&((a[i].second+d)>=p[pIndex])){
            points+=((p[pIndex]-d)*a[i].first*f);
            a[i].second-=(p[pIndex]-d);
            d+=(p[pIndex]-d);
            f++; pIndex++;
        }
        points+=(a[i].second*a[i].first*f);
        d+=a[i].second;
        a[i].second=0;
    }

    cout<<points;
    return 0;
}
