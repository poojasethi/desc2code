#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;
typedef long long LL;
typedef pair<LL,LL> PLL;
PLL a[111];
LL b[111];
LL n,m;
LL ans=0;

bool cmp(PLL a,PLL b)
{
    return a.second<b.second;
}
int main()
{
    cin>>n;
    for(int i=0;i<n;i++) cin>>a[i].first>>a[i].second;
    sort(a,a+n,cmp);
    cin>>m;
    for(int i=0;i<m;i++) cin>>b[i];
    for(int i=m-1;i>=1;i--) b[i]-=b[i-1];
    b[m]=(LL)1e18;
    LL cur=0;
    for(int i=0;i<n;i++)
    {
        while(a[i].first>=b[cur]) a[i].first-=b[cur],ans+=b[cur]*(cur+1)*a[i].second,cur++;
        if(a[i].first!=0) b[cur]-=a[i].first,ans+=(cur+1)*a[i].first*a[i].second;
    }
    cout<<ans<<endl;
}
