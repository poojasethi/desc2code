#include <iostream>
#include <algorithm>
typedef long long ll;
using namespace std;
ll b[110],ans,k,x;
struct rec{int n,c;};
rec a[110];
int n,t,p,q,f;
bool cmp(rec a,rec b){
	return a.c < b.c;
}
int main(){
	cin >> n;
	for (int i = 1;i <= n;i++) cin >> a[i].n >> a[i].c;
	std::sort(a + 1,a + n + 1,cmp);
	cin >> t;
	for (int i = 1;i <= t;i++) cin >> b[i];
	b[t + 1] = 1e13;
	k = 0;
	ans = 0;
	f = 1;
	p = q = 1;
	while (p <= n){
		if (k + a[p].n >= b[q]){
			x = b[q] - k;
			k = b[q];
			a[p].n -= x;
			ans += f * x * a[p].c;
			f++;
			q++;
			continue;
		}
		x = a[p].n;
		ans += f * x * a[p].c;
		k += x;
		p++;
	}
	cout << ans;
}
