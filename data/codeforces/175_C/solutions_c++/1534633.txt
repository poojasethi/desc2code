#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

vector<pair<int, int> > a;
typedef long long LL;
LL p[105];

int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		int c, k;
		scanf("%d%d", &k, &c);
		a.push_back(make_pair(c, k));
	}
	sort(a.begin(), a.end());
	int m;
	scanf("%d", &m);
	for(int i=1;i<=m;i++) scanf("%I64d", &p[i]);
	p[m+1]=1;
	for(int i=0;i<n;i++)p[m+1] += a[i].second;
	int f = 1;
	LL now=0;
	LL ans=0;
	for(int i=0;i<n;i++) {
		while(a[i].second>0) {
			if(p[f]==now) ++f;
			int R=min(p[f] - now, a[i].second*1LL);
			ans += 1LL*a[i].first*f*R;
			a[i].second -= R;
			now += R;
		}
	}
	printf("%I64d\n", ans);
	return 0;
}
