#include <cstdio>
#include <iostream>
using namespace std;
long long k[109],c[109];
long long p[109];
long long f;
int main()
{
	int n,t;
	long long x;
	cin>>n;
	for (int i=0;i<n;i++) cin>>k[i]>>c[i];
	cin>>t;
	for (int i=0;i<t;i++) cin>>p[i];
	for (int i=t-1;i>0;i--) p[i]-=p[i-1];



	for (int i=0;i<n;i++)
		for (int j=i+1;j<n;j++)
			if (c[i]>c[j])
			{
				long long tp;
				tp=c[i];c[i]=c[j];c[j]=tp;
				tp=k[i];k[i]=k[j];k[j]=tp;
			}
	f=1;
	long long ans=0;
	p[t]=10000000000000000LL;

	int i=0,j=0;
	while (i<n)
	{
		if (k[i]>=p[j])
		{
			ans+=p[j]*c[i]*f;
			k[i]-=p[j++];
			f++;
			if (!k[i]) i++;
		}
		else
		{
			ans+=k[i]*c[i]*f;
			p[j]-=k[i];
			i++;
		}
	}
	cout<<ans<<endl;
	return 0;
}
