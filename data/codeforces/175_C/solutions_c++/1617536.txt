#include <stdio.h>
#include <string.h>
#include <vector>
#include <algorithm>
using namespace std;
typedef long long ll;
#define REP(i,n) for(int(i)=0;(i)<(int)(n);(i)++)
vector < pair <int,int> > a;
ll p[110];
int main()
{
	int n;
	scanf("%d",&n);
	REP(i,n) {
		int c,k;
		scanf("%d%d",&k,&c);
		a.push_back(make_pair(c,k));
	}
	sort(a.begin(),a.end());
	int m;
	scanf("%d",&m);
	for (int i=1; i<=m; i++) scanf("%I64d",&p[i]);
	for(int i=0;i<n;i++) p[m+1]+=a[i].second;
	ll f=1,ans=0,cur=0;
	REP(i,n) {
		while (a[i].second>0) {
			if (p[f]==cur) f++;
			int d=min(p[f]-cur,1LL*a[i].second);
			ans+=1LL*a[i].first*f*d;
			a[i].second-=d;
			cur+=d;
		}
	}
	printf("%I64d\n",ans);
	return 0;
}
