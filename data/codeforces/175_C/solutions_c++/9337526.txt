#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define int64 long long
#define N 110
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int k,c;
} a[N];
int i,j,k,n,m;
int64 an,b[N];
inline bool cc1(const ww &a,const ww &b) {
	return a.c<b.c;
}
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%d%d",&a[i].k,&a[i].c);
	sort(a+1,a+n+1,cc1);
	scanf("%d",&m);
	For(i,1,m) scanf("%I64d",&b[i]);
	for (i=m;i;i--) b[i]-=b[i-1];
	b[++m]=(int64)1e15;
	j=1;
	For(i,1,m) {
		int64 k=b[i];
		for (;j<=n&&k;j++) {
			if (a[j].k>=k) {
				an+=k*a[j].c*i;
				a[j].k-=k;
				break;
			} else {
				an+=1ll*a[j].k*a[j].c*i;
				k-=a[j].k;
			}
		}
	}
	printf("%I64d\n",an);
	return 0;
}
