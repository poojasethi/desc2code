#include<stdio.h>
#include<algorithm>
typedef long long int lnt;
typedef struct{lnt k,c;}ele;
int cmp(ele a,ele b){return a.c<b.c;}
ele e[200];
lnt p[200];
int main(){
	int n,i,j,t;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%I64d %I64d",&e[i].k,&e[i].c);
	std::sort(e,e+n,cmp);
	scanf("%d",&t);
	for(i=0;i<t;i++)
		scanf("%I64d",&p[i]);
	p[t]=(1ll<<63)-1;
	lnt total=0,ans=0;
	for(i=j=0;i<n;i++){
		for(;total<=p[j]&&e[i].k;){
			if(p[j]-total>=e[i].k){
				ans+=(j+1)*e[i].k*e[i].c;
				total+=e[i].k;
				e[i].k=0;
				}
			else{
				ans+=(j+1)*(p[j]-total)*e[i].c;
				e[i].k-=p[j]-total;
				total=p[j];
				j++;
				}
			}
		}
	printf("%I64d\n",ans);
	//while(getchar());
	return 0;
	}
