#include <iostream>
using namespace std;
#define LL long long
#define N 1024
LL n,t,k[N],c[N],tk;
int main() {
  cin>>n;
  for (LL i=0;i<n;i++)
    cin>>k[i]>>c[i],
    tk+=k[i];
  cin>>t;
  LL p[t+1];
  for (LL i=0;i<t;i++)
    cin>>p[i];
  p[t]=tk;
  for (LL i=t;i>=1;i--)
    p[i]-=p[i-1];
  LL id[n];
  for (LL i=0;i<n;i++)
    id[i]=i;
  for (LL i=0;i<n;i++)
    for (LL j=i+1;j<n;j++)
      if (c[id[i]]>c[id[j]]) {
        LL swap=id[i];
        id[i]=id[j];
        id[j]=swap;
      }
  LL ans=0;
  for (LL i=0,j=0;i<n;i++)
    while (k[id[i]]) {
      while (!p[j]) j++;
      LL delt=min(k[id[i]],p[j]);
      ans+=delt*c[id[i]]*(j+1);
      k[id[i]]-=delt;
      p[j]-=delt;
    }
  cout<<ans<<endl;
}
