#include <cstdio>
#include <algorithm>

using namespace std;

struct figure
{
  int t,c;
  bool operator<(const figure& x) const
  {
    return c<x.c;
  }
};

figure a[101];
long long p[101];

int main(void)
{
  int n,m;
  scanf("%d",&n);
  for(int i=1;i<=n;++i)
    scanf("%d %d",&a[i].t,&a[i].c);
  scanf("%d",&m);
  for(int i=1;i<=m;++i)
    scanf("%I64d",&p[i]);
  sort(a+1,a+n+1);
  long long ans=0,s=0;
  int f=1;
  for(int i=1;i<=n;++i)
  {
    while((f<=m)&&(a[i].t>0)&&(s+a[i].t>=p[f]))
    {
      ans+=(p[f]-s)*f*a[i].c;
      a[i].t-=int(p[f]-s);
      s=p[f];
      ++f;
    }
    if(a[i].t>0)
    {
      ans+=a[i].t*((long long)f)*a[i].c;
      s+=a[i].t;
    }
  }
  printf("%I64d\n",ans);
  return 0;
}
