#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int n;
    cin >> n;
    pair<long long,long long> f[100];
    for(int i=0;i<n;i++)
        cin >> f[i].second >> f[i].first;
    sort(f,f+n);
    int t;
    cin >> t;
    long long p[101];
    for(int i=0;i<t;i++) cin >> p[i];
    p[t] = 10000000000001LL;
    int factor = 1;
    long long sum = 0;
    long long cnt = 0;
    for(int i=0;i<n;i++) {
        long long delta = min(f[i].second, p[factor-1]-cnt);
//        cout << i << " " << delta << " " << factor << "\n";
        f[i].second -= delta;
        cnt += delta;
        sum += factor * f[i].first * delta;
        if (cnt == p[factor-1]) factor++;
        if (f[i].second > 0) i--;
    }
    cout << sum << endl;
    return 0;
}
