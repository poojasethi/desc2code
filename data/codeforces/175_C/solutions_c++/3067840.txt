#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long LL;
struct fig{ LL k,c; } a[101];
bool comp( fig x, fig y ){ return x.c < y.c; }
int main(){
    LL n, m, p[101];
    scanf( "%I64d", &n );
    for( LL i = 0 ; i < n ; i++ ) scanf( "%I64d%I64d", &a[i].k, &a[i].c );
    scanf( "%I64d", &m );
    for( LL i = 0 ; i < m ; i++ ) scanf( "%I64d", &p[i] );
    sort( a, a+n, comp );
    LL kil = 0, it = 0, pnt = 0;
    for( LL i = 0 ; i < n ; i++ ){
        kil += a[i].k;
        pnt += a[i].k*a[i].c*(it+1);
        while( it < m && kil >= p[it] ) pnt += (kil-p[it++])*a[i].c;
    }
    printf( "%I64d\n", pnt );
    return 0;
}
