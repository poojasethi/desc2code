#include<iostream>
using namespace std;
int main()
{
int n,i,j,tt;
long long a[105],b[105],t[105],r=0,z=0;
cin>>n;
for(i=1;i<=n;i++)
{
cin>>a[i]>>b[i];
}
cin>>tt;
for(i=1;i<=tt;i++)
{
cin>>t[i];
}
for(i=1;i<=n;i++)
{
for(j=i+1;j<=n;j++)
{
if(b[i]>b[j]) swap(a[i],a[j]),swap(b[i],b[j]);
}
}
i=j=1;
while(i<=n)
{
if(a[i]<=t[j])
{
r+=a[i]*b[i]*j;
t[j]-=a[i];
z+=a[i];
i++;
}
else
{
if(j<=tt)
{
r+=t[j]*b[i]*j;
a[i]-=t[j];
z+=t[j];
j++;
t[j]-=z;
}
else r+=a[i]*b[i]*j,i++;
}
}
cout<<r;
return 0;
}
