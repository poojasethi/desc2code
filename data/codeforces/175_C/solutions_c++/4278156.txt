#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;

struct FIGURE{
	int num,cost;
}figure[300];

long long p[300];

int cmp(const FIGURE &a, const FIGURE &b){
	return a.cost<b.cost;
}

int main(){
	int n,t;
	scanf("%d",&n);
	for (int i=0; i<n; ++i) scanf("%d%d",&figure[i].num,&figure[i].cost);
	sort(figure,figure+n,cmp);
	scanf("%d",&t);
	for (int i=1; i<=t; ++i) cin>>p[i];
	int mark=1;
	long long ans=0,tot=0;
	for (int i=0; i<n; ++i){
		while (mark<=t && figure[i].num>=p[mark]-tot) {
			ans+=(long long)mark*figure[i].cost*(p[mark]-tot);
			figure[i].num-=p[mark]-tot;
			tot=p[mark];
			++mark;
		}
		tot+=figure[i].num;
		ans+=(long long)mark*figure[i].cost*figure[i].num;
	}
	cout<<ans;
	return 0;
}
