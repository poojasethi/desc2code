#include <iostream>
#include <algorithm>
#include <stdio.h>
using namespace std;
struct fig
{
 long long k,c;
} a[101];
bool cmp(fig x,fig y)
{
 return x.c<y.c;
}
int main()
{
//freopen("i.txt","r",stdin);
int n,t;
long long p[101];
cin>>n;
for(int i=0;i<n;i++) cin>>a[i].k>>a[i].c;
cin>>t;
for(int i=0;i<t;i++) cin>>p[i];
sort(a,a+n,cmp);

long long s=0;

long long k=0,i1=0,t1=1;
for (int i=0;i<n;i++)
{
 long long j=a[i].k;
 for (;;)
  {
   int k1=min(p[i1]-k,j);
   k+=k1;
   s+=(t1*a[i].c*k1);
   j-=k1;
   if (k==p[i1]) {t1++; i1++;}
   if (j==0) break;
  }
}
cout<<s;
return 0;
}
