#include <iostream>
#include <map>
using namespace std;
long long C[1001];
int main(){
    long long  n,k,c,t,f = 1;
    cin>>n;
    for (int i = 0; i < n; i++)
        cin>>k>>c, C[c]+=k;
    cin>>t;
    long long p, count = 0, points = 0;
    c = 0;
    for (int i = 0; i < t; i++){
        cin>>p;
        while (c<=1000 && count<p)
            if (count+C[c]<p)
                points+=c*C[c++]*f, count+=C[c-1];
            else
                points+=c*(p-count)*f, C[c]-=p-count, count = p;
        //cout<<points<<endl;
        f++;
    }
    while (c<=1000)
        points+=c*C[c++]*f;
    cout<<points<<endl;
    //system("PAUSE");
    return 0;
}
