#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
#include <vector>
#include <map>

using namespace std;

typedef long long llong;
typedef pair<int, llong> pil;

int main()
{
  int n;
  cin >> n;
  vector<pil> c(n);
  for (int i = 0; i < n; ++i)
    cin >> c[i].second >> c[i].first;
  sort(c.begin(), c.end());
  int t;
  cin >> t;
  vector<llong> p(t);
  for (int i = 0; i < t; ++i)
    cin >> p[i];
  llong sum = 0;
  llong died = 0;
  for (int i = 0, j = 0; i < n; ++i) {    
    while (j < t && died + c[i].second > p[j]) {
      sum += c[i].first * (p[j] - died) * (j + 1);
      c[i].second -= p[j] - died;
      died += p[j] - died;
      ++j;
    }
    sum += c[i].first * c[i].second * (j + 1);
    died += c[i].second;
  }
  cout << sum << "\n";
  return 0;
}
