#include <iostream>
#include <algorithm>
#include <stdio.h>

using namespace std;


struct fig
{
 long long k,c;
} a[101];
bool cmp(fig x, fig y)
{
 return x.c<y.c;
}

int main()
{int n;
 int t;
 long long p[101];
 //freopen("i.txt","r",stdin);
 cin>>n;
 for (int i=0; i<n;i++) cin>>a[i].k>>a[i].c;
 sort(a,a+n,cmp);
 cin>>t;
 for (int i=0; i<t;i++) cin>>p[i];

 long long s=0,k=0;
 long long i1=0,t1=1;

 for (int i=0; i<n;i++)
 {
   int j=a[i].k;
  for (; ;)
   {
    int kol=min(p[i1]-k,(long long)j);
    k+=kol;
    j-=kol;
    s+=(long long)(t1*a[i].c*kol);
    if (k==p[i1]) {t1++;i1++;}
    if (j==0) break;
   }
 }

 cout << s << endl;
 return 0;
}
