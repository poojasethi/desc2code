#include <iostream>
#include <utility>
#include <algorithm>

using namespace std;

#define ii pair<int, int>

int main()
{
	int n, t;
	cin >> n;
	ii fig[101];
	for(int i=0;i<n ;i++)
	{
		cin>>fig[i].second >> fig[i].first;
	}
	sort(fig, fig+n);
	
	cin >> t;
	unsigned long long p[101];
	for(int i=0;i<t;i++){
		cin>>p[i];
	}
	for(int i=t-1; i>0; --i)
		p[i]-=p[i-1];

	unsigned long long ans = 0;
	
	int tp = 0;
	unsigned long long int f = 1;	
	for(int i=0; i<n;i++)
	{
		unsigned long long int fr = fig[i].second;
		while(tp != t && fr > 0){
			if(fr >= p[tp]){
				ans+= p[tp]*f*fig[i].first;
				fr -= p[tp];
				f++;
				tp++;
			}else{
				ans+= fr*f*fig[i].first;
				p[tp] -= fr;
				fr = 0;
			}
		}
		ans += fr*f*fig[i].first;
	}
	
	cout << ans <<endl;
	return 0;
}
