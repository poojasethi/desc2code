#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
struct rec{long long x,y;}a[110];
long long p[110],ans;
int n,m,i,t;

bool cmp(rec a,rec b)
{
	return a.y<b.y;
}

int main()
{
	cin>>n;
	for(i=1;i<=n;i++) scanf("%I64d%I64d",&a[i].x,&a[i].y);
	cin>>m;
	for(i=1;i<=m;i++) scanf("%I64d",&p[i]);
	for(i=m;i;i--) p[i]-=p[i-1];
	p[++m]=1e14;
	sort(a+1,a+n+1,cmp);
	for(i=t=1;i<=n;i++)
		while(a[i].x)
		{
			if(p[t]<=a[i].x)
			{
				a[i].x-=p[t];
				ans+=a[i].y*p[t]*t;
				t++;
			}
			else{
				p[t]-=a[i].x;
				ans+=a[i].y*a[i].x*t;
				a[i].x=0;
			}
		}
	cout<<ans<<endl;
	return 0;
}
