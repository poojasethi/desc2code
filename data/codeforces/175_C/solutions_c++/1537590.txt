#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main() {
    int n;
    cin >> n;
    vector<pair<long long, long long> > a(n);
    for (int i = 0; i < n; ++ i) {
        cin >> a[i].second >> a[i].first;
    }
    int t;
    cin >> t;
    vector<long long> p(t+1);
    for (int i = 0; i < t; ++ i) {
        cin >> p[i];
    }
    p[t] = 1LL<<60;
    sort(a.begin(), a.end());
    long long q = 0, r = 0, s = 0;
    for (int i = 0; i < n; ++ i) {
        while (a[i].second > 0) {
            long long m = min(a[i].second, p[s]-q);
            a[i].second -= m;
            q += m;
            r += m * a[i].first * (s+1);
            if (p[s] == q) ++ s;
        }
    }
    cout << r << endl;
}
