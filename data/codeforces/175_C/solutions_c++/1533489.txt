#include<cstdio>
#include<cstring>
#include<algorithm>
#include<string>
#include<map>
#include<iostream>
using namespace std;
struct ss{
    int num,cost;
}p[110];
int n,t;
long long ans,ti[110];
bool cmp(ss a,ss b){return a.cost<b.cost;}
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;++i)
        cin>>p[i].num>>p[i].cost;
    sort(p+1,p+n+1,cmp);
    scanf("%d",&t);
    for(int i=1;i<=t;++i)cin>>ti[i];
    ti[t+1]=100000000000000LL;
    long long now=0,last,h=1;
    for(int i=1,j=1;i<=n;++i){
        last=now;
        now+=p[i].num;
        while(now>ti[j]){
            ans+=(ti[j]-last)*p[i].cost*h;
            last=ti[j];++j;++h;
        }
        ans+=(now-last)*p[i].cost*h;
    }
    cout<<ans<<endl;
    return 0;
}
