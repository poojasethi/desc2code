#include<cstdio>
#include<algorithm>
using namespace std;

struct node{
	long long k;
	int c;
	bool operator<(const node &a)const
	{
		return c<a.c;
	}
}a[105];
long long p[105];

int main()
{
	int n,m,i,j;
	long long c,s;
	for(scanf("%d",&n),i=0;i<n;i++)
		scanf("%I64d%d",&a[i].k,&a[i].c);
	sort(a,a+n);
	for(scanf("%d",&m),i=1;i<=m;i++)
		scanf("%I64d",&p[i]);
	p[i]=1ll<<60;
	for(c=s=0,j=1,i=0;i<n;i++)
	{
		s+=a[i].k*a[i].c*j;
		c+=a[i].k;
		while(c>p[j])
		{
			s+=(c-p[j])*a[i].c;
			j++;
		}
	}
	printf("%I64d\n",s);
	return 0;
}
