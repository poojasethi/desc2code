#include <iostream>
#include <vector>
#include <algorithm>
#include <cassert>

using namespace std;

typedef long long ll;
typedef pair<ll,ll> pii;
pii kc[110];
ll t;
ll p[110];
int main(){
  int n;
  cin>>n;
  for(int i=0;i<n;i++)cin>>kc[i].second>>kc[i].first;
  sort(kc,kc+n);
  cin>>t;
  for(int i=0;i<t;i++)cin>>p[i];
  p[t++]=1000000000000000000LL;
  ll pos=0;
  ll scr=0;
  for(ll i=0;i<t;i++){
    ll sum=p[i];
    if(i!=0)sum-=p[i-1];
    for(;pos<n;pos++){
      ll tmp=min(sum,kc[pos].second);
      sum-=tmp;
      kc[pos].second-=tmp;
      scr+=(i+1)*(tmp*kc[pos].first);
      if(sum==0)break;
      if(kc[pos].second!=0){
    assert(0);
    pos--;
      }
    }
    if(pos>=n)break;
  }
  cout<<scr<<endl;
  
  return 0;
}
