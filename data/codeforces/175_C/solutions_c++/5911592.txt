#include <algorithm>
#include <cstdio>
#include <iostream>
using namespace std;
struct M
{
    long long k;
    long long c;
    __int64 dick;
}a[110];
bool cmp(M x,M y)
{
    return x.c<y.c;
}
int main()
{
    long long  n,i,j,t;
    long long b[110];
    __int64 sum;
    while(cin>>n){
      for(i=0;i<n;i++){
         cin>>a[i].k>>a[i].c;
      }
      sort(a,a+n,cmp);
      cin>>t;
      for(j=0;j<t;j++)
      cin>>b[j];
      //sort(b+1,b+t+1);
      j=0; sum=0;
      long long mm=0;
      for(i=0;i<n;i++)
      {
         sum+=a[i].k*a[i].c*(j+1);
         mm+=a[i].k;
         while(j<t&&mm>=b[j]){
            sum+=(mm-b[j++])*a[i].c;
         }
      }
      printf("%I64d\n",sum);
    }
 return 0;
}
