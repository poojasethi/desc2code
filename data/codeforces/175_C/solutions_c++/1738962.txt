# include <stdio.h>
# include <stdlib.h>

int n,t,k=1;

double t1[150];

double ans,e;

struct bet
{
	double a,b;
};
bet n1[150];

	int compare(void const *x, void const *y)
	{
		bet q = *(bet*)x;
		bet w = *(bet*)y;
		
		return q.b-w.b;
	}

int main()
{
	int u=0,y=0;
	
	scanf("%d",&n);
	for (int i = 0; i < n; i++) scanf("%lf %lf",&n1[i].a,&n1[i].b);
	
	qsort(n1,n,sizeof(bet),compare);
	
	for (int i = 1; i < n; i++) n1[i].a+=n1[i-1].a;
	
	scanf("%d",&t);
	for (int i = 0; i < t; i++) scanf("%lf",&t1[i]);
	
	t1[t]=999999999999999999;
	
	while(u < n)
	{
		if(n1[u].a <= t1[y]) ans+=(n1[u].a-e)*k*n1[u].b,e=n1[u].a,u++;
		if(n1[u].a > t1[y]) ans+=(t1[y]-e)*k*n1[u].b,e=t1[y],k++,y++;
	}
	
	printf("%.0lf\n",ans);
}
