#include<iostream>
#include<algorithm>
using namespace std;

int n, t;
pair<int, int> v[111];
long long p[111], tot, ans;

long long calc(long long rem, int mul) {
	long long det, ret = 0;
	for(int i=0;i<n && rem;i++){
		det = min(rem, (long long)v[i].second);
		rem -= det; v[i].second -= det;
		ret += det * v[i].first * mul;
	}
	return ret;
}

int main() {
	tot=0LL;
	cin>>n;
	for(int i=0;i<n;i++) {
		int k, c;
		cin>>k>>c;
		v[i] = make_pair(c, k);
		tot += k;
	}
	sort(v,v+n);
	
	cin>>t;
	p[0] = 0LL;
	for(int i=1;i<=t;i++) {
		cin>>p[i];
		ans += calc(p[i] - p[i-1], i);
	}
	ans += calc(tot, t+1);
	cout<<ans<<endl;
	
	return 0;
}
