import sys

f = sys.stdin
#f = open('in','r')
n = int(f.readline())
fig = list()
s = 0
for _ in range(n):
	k, cst = map(int, f.readline().split())
	fig.append((k,cst))
	s += k
t = int(f.readline())
p = map(int, f.readline().split())
p.append(s+1)
fig.sort(key = lambda x: x[1])
j,c,r = 0,0,0
for i in range(n):
	k,cst = fig[i]
	while p[j]-c <= k:
		#print j, c, k
		r += cst*(j+1)*(p[j]-c)
		k -= p[j]-c
		c += p[j]-c
		j += 1
	if k > 0:
		r += cst*(j+1)*k
		c += k
print r
f.close()
