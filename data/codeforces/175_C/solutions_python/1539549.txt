n=int(raw_input())
ff=[]
for i in xrange(n):
    k,c=map(int,raw_input().split())
    ff.append([c,k])
ff.sort()
t=int(raw_input())
p=map(int,raw_input().split())
p+=[10**100]
tp=0
res=0
mn=1
cnt=0
while tp<len(ff):
    if ff[tp][1]==0:
        tp+=1
    elif p[mn-1]-cnt==0:
        mn+=1
    else:
        cc=min(p[mn-1]-cnt,ff[tp][1])     
        ff[tp][1]-=cc
        cnt+=cc
        res+=mn*cc*ff[tp][0]
print res
