n = int(raw_input())
a = []
for i in xrange(n):
    x, y = map(int, raw_input().split())
    a.append([y, x])
a.sort()
ans, fac = 0, 1
k = int(raw_input())
p = map(int, raw_input().split())
p.append(10 ** 16)
apos = dn = ppos = 0
while apos < len(a):
    x = min(p[ppos], a[apos][1])
    ans += x * a[apos][0] * fac
    p[ppos] -= x; dn += x; a[apos][1] -= x
    if not p[ppos]:
        ppos += 1
        fac += 1
        p[ppos] -= dn
    if not a[apos][1]:
        apos += 1
print ans
