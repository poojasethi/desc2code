n = int(raw_input())
sl = []
for i in xrange(n):
    k, c = map(int, raw_input().split())
    sl.append([c,k])
sl.sort()
t = int(raw_input())
p = map(int, raw_input().split())
p += [10**100]
cnt = 0
item = 0
pi = 1
res = 0
while item < len(sl):
    if sl[item][1] == 0:
        item += 1
    elif p[pi - 1] - cnt == 0:
        pi += 1
    else:
        x = min(p[pi - 1] - cnt, sl[item][1])
        sl[item][1] -= x
        cnt += x
        res += x*pi*sl[item][0]
print res
