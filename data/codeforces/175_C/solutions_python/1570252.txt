n=int(raw_input())
s=[]
for i in range(n):
    a,b=map(int,raw_input().split())
    s.append([b,a])
s.sort()
otv = 0
t=int(raw_input())
p=map(int,raw_input().split())
p.append(10 ** 12)
spos=ppos=kol=0
while spos<len(s):
        x=min(s[spos][1],p[ppos])
        otv+=s[spos][0]*x*(ppos+1)
        s[spos][1]-=x
        p[ppos]-=x
        kol+=x
        if s[spos][1]==0: spos+=1
        if p[ppos]==0:
            ppos+=1
            p[ppos]-=kol
print otv
