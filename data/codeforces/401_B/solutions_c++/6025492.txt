#include <iostream>
using namespace std;

int arr[4001];
int main() {
	int x,k,a,b,c,mx=0,mn=0;
	cin>>x>>k;
	for(int i=0;i<k;i++){
		cin>>a;
		if(a==1){	cin>>b>>c;
					arr[b]=1;
					arr[c]=1;
		}
		else{	cin>>b;
				arr[b]=1;
		}
	}
	arr[x]=1;
	for(int i=1;i<x;i++)
		if(!arr[i])	mx++;
	for(int i=1;i<x;i++)
		if(!arr[i]){
			mn++;
			if(!arr[i+1])
				i++;
		}
	cout<<mn<<" "<<mx;
	return 0;
}
