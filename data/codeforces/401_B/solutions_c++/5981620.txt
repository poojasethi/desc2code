#include<stdio.h>
int main()
{
	int x,k;
	scanf("%d%d",&x,&k);
	int i,a[4001]={0},b,c,d,mi=0,ma=0;
	while(k--)
	{
		scanf("%d",&b);
		if(b==1)
		{
			scanf("%d%d",&c,&d);
			a[c]=2;
			a[d]=1;
		}
		else
		{
			scanf("%d",&c);
			a[c]=2;
		}
	}
	b=0;
	for(i=1;i<x;i++)
	{
		if(a[i]==0)
		{
			b++;
		}
		else
		{
			ma+=b;
			mi=mi+(b+1)/2;
			b=0;
		}
	}
	ma+=b;
	mi=mi+(b+1)/2;
	printf("%d %d\n",mi,ma);
	return 0;
}
