#include <iostream>
using namespace std;
int x,y,i,a,b,c,d[5000],o1=0,o2=0,o3=0;
int main(){
    cin>>y>>x;
    for(i=0;i<x;i++){
		cin>>a>>b;
		d[b]=1;
		if(a==1)cin>>c,d[c]=1;
	}
	for(i=1;i<y;i++){
		if(i>1 && d[i-1]==0 && d[i]==0){
			d[i]=1;
			o2++;
			o1++;
			o3--;
		}
		if(d[i]!=1)o1++,o3++;
	}
	cout<<o3+o2<<" "<<o1;
    return 0;
}
