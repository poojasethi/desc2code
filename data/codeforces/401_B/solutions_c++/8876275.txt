#include<iostream>
using namespace std;
int main()
{
int xx,k,q,a,b,coun=0,r=0,maxx,i;
cin>>xx>>k;
int x[4010]={0};
for(;k;k--)
{
    cin>>q;
    if(q==2)
        {cin>>b; x[b]++;  coun++;}
        else {cin>>a>>b;   x[a]++; x[b]++; coun+=2;}
}
for(i=2;i<=xx-1;i++)
{
    if(x[i]==0 && x[i-1]==0)
    {r++; x[i]=-100;}
}
maxx=xx-1-coun;
cout<<maxx-r<<" "<<maxx;
}
