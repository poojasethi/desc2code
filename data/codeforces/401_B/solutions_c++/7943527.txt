#include<iostream>
#include<algorithm>

using namespace std;
long long a[4001],b,c,o,n,m,p;
string s;
int main(){
	cin>>n>>m;
	for(int i=1;i<=m;i++){
		cin>>b;
		if(b==2) cin>>c,a[c]=2,a[c+1]=2,o++;
		else cin>>c ,a[c]=2,cin>>c,a[c]=1,o+=2;
	}
	o=n-1-o;
	
	a[n]=2;
	for(int i=2;i<n;i++){
		if(a[i]==0 && a[i-1]!=1)p++,a[i]=1;
	}
	cout<<o-p<<" "<<o;
}
