#include<iostream>
using namespace std;
bool a[4002];
int main()
{
	int x, k, b, c, d = 0, f = 0;
	cin>>x>>k;
	while( k-- ){
		cin>>b>>c;
		a[c] = 1;
		if( 1 == b ){
			cin>>c;
			a[c] = 1;
		}
	}
	c = 0;
	for( int i = 1; i < x; i++ ){
		if( 1 == a[i] ){
			d += (c+1)/2;
			c = 0;
		}
		else{
			++c;
			++f;
		}
	}
	cout<<d+(c+1)/2<<" "<<f<<endl;
	return 0;
}
