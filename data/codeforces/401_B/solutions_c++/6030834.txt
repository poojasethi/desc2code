#include<iostream>
using namespace std;
int main()
{
    int x,y,aa,bb,a1=0,a2=0;cin>>x>>y;bool a[4005]={0};a[x]=1;a[0]=1;
    while(y--)
    {
        int i;cin>>i;
        if(i==2) cin>>aa,a[aa]=1;
        else cin>>aa>>bb,a[aa]=1,a[bb]=1;
    }
    for(int i=1;i<x;++i)
    {
        if(!a[i])
        {
            a2++;a1++;if(!a[i-1])a1--,a[i]=1,a[i-1]=1;
        }
    }
    cout<<a1<<" "<<a2<<endl;
}
