#include <stdio.h>

int div[4005];
int main() {
	int xx,k;
	scanf("%d%d",&xx,&k);
	int x = xx - 1;
	int mmin = 0, mmax = 0;
	int type , a;
	while(k--) {
		scanf("%d%d",&type,&a);
		div[a] = 1;
		if(type == 1) 
			scanf("%d",&a),
			div[a] = 1,
			x -= 2;
		else x -= 1;
	}
	mmax = x ;
	for(int i=2;i<xx;i++) 
		if(!div[i] && !div[i-1])
			x--,
			div[i] = 1;
	mmin = x;
	printf("%d %d\n",mmin,mmax);
	return 0;
}
