#include <cstdio>
#include<algorithm>
int N,K,ans1,ans2;
bool a[4005];

int main()
{
	scanf("%d%d",&N,&K),ans2=N-1;
	for (int i=1,x,y,t; i<=K; i++)
	{
		scanf("%d%d",&t,&x),a[x]=1,ans2--;
		if (t==1) scanf("%d",&y),a[y]=1,ans2--;
	}
	for (int i=1,j; i<N; i++) if (!a[i])
	{
		for (j=i; j<N&&!a[j]; j++);
		ans1+=(j-i+1)/2,i=j-1;
	}
	printf("%d %d\n",ans1,ans2);
}
