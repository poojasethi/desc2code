#include<iostream>
using namespace std;
int main()
{
	int x,k,max,min,o,a,b,c;
	bool r[4010]={0};
	cin>>x>>k;
	for(int i=0;i<k;i++)
	{
		cin>>o;
		if(o==1){
			cin>>a>>b;
			r[a]=r[b]=1;
		}
		else if(o==2){
			cin>>a;
			r[a]=1;
		}
	}
	max=min=c=0;
	for(int i=1;i<x;i++)
	{
		if(r[i]==0)
		{
			max++,c++;
		}
		else{
			min+=((c/2)+c%2);
			c=0;
		}
	}
	min+=((c/2)+c%2);
	cout<<min<<" "<<max<<endl;
}
