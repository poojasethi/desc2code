#include<iostream>
using namespace std;

int a[5000];
int n,m,q,imin,imax;

int main()
{
	cin>>n>>m;
	for(int i=0;i<m;i++)
	{
		cin>>q;
		if(q==1)
		{
			cin>>q;
			a[q]=2;
			cin>>q;
			a[q]=1;
		}	
		else
		{
			cin>>q;
			a[q]=2;
		}
	}
	for(int i=1;i<n;i++)
		if(a[i]==0) 
		{
			imax++;
			if(a[i-1]==3)
				a[i]=1;
			else
			{
				a[i]=3;
				imin++;
			}
		}
		
	cout<<imin<<" "<<imax;
	
	return 0;
}
