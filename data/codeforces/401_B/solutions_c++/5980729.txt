#include<stdio.h>
const int maxn=4050;
int x,k,vis[maxn]={0},ansl=0,ansr=0;
int main()
{
	int i,j,t;
	scanf("%d%d",&x,&k);
	for(i=0;i<k;i++)
	{
		int o;
		scanf("%d%d",&o,&t);
		vis[t]=1;
		if(o==1) scanf("%d",&t),vis[t]=1;
	}
	for(i=1;i<x;i++)
		if(!vis[i])
		{
			for(j=i+1;j<x&&!vis[j];j++);
			ansl+=(j-i-1)/2+1;
			ansr+=j-i;
			i=j-1;
		}
	printf("%d %d\n",ansl,ansr);
	return 0;
}
