#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll x,k,N;
ll dum,ch,yum;
bool g[4001];
int main()
{
	for(ll i=0;i<=4000;i++) g[i]=1;
	cin>>x>>k;
	for(ll i=0;i<k;i++)
	{
		cin>>ch;
		if(ch==1) {cin>>dum>>yum; g[dum]=0; g[yum]=0;}
		else {cin>>dum; g[dum]=0;}
	}
	for(ll i=1;i<x;i++) N+=g[i];
	ll M=0;
	for(ll i=1;i<x;i++)
	{
		if(g[i]==1)
		{
			if(i+1<x&&g[i+1]==1) {M++;i++;}
			else M++;
		}
	}
	cout<<M<<" "<<N<<endl;
	return 0;
}
