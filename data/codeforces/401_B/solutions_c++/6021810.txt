#include<stdio.h>

int a[4002];
int main(){
	int x,k,op,t;
	scanf("%d %d",&x,&k);
	for(int i=0;i<k;i++){
		scanf("%d",&op);
		if(op==1){
			scanf("%d",&t);
			a[t]=1;
		}
		scanf("%d",&t);
			a[t]=1;
	}
	a[x]=1;
	int min=0,max=0,cnt=0;
	for(int i=1;i<=x;i++){
		if(a[i]==0)
			cnt++;
		else{
			min+=(cnt+1)/2;
			max+=cnt;
			cnt=0;
		}
	}
	printf("%d %d\n",min,max);
}
