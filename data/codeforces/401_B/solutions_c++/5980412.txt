#include <cstdio>

#define N 4000

int a[N + 1], n, k, i, t, t1, t2, s1, s2;

int main()
{
	scanf("%d%d", &n, &k);

	for (i = 0; i < k; i++)
	{
		scanf("%d", &t);
		if (t == 1)
		{
			scanf("%d%d", &t1, &t2);
			a[t1] = 1;
			a[t2] = 2;
		}
		else
		{
			scanf("%d", &t2);
			a[t2] = 2;
		}
	}

	k = 0;

	for (i = 1; i < n; i++)
		if (!a[i])
			++k, ++s2;
		else
			s1 += k / 2 + k % 2, k = 0;

	s1 += k / 2 + k % 2;

	printf("%d %d\n", s1, s2);

	return 0;
}
