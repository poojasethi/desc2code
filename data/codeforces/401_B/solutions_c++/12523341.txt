#include<stdio.h>
#define Mix(x,y) x<y?x:
int a[4005];
int main(void){
	int x,k;int t=0;int min=0,max=0;
	scanf("%d%d",&x,&k);
	while(k--){
		int m,n;
		scanf("%d",&m);
		if(m==2){
			scanf("%d",&n);a[n]=1;
		}
		else {
			scanf("%d%d",&n,&m);
			a[m]=a[n]=1;
		}
	}
	for(int i=1;i<x;i++){
		if(a[i]==0) t++;
		if(a[i]!=0||i==x-1){
			min+=(t%2==0)?t/2:t/2+1;
			max+=t;
			t=0;
		}
	}printf("%d %d\n",min,max);
	return 0;
}
