#include<iostream>
using namespace std;
int i,a,b,n,x,k,s[4000],c,q;
int main()
{
    cin>>x>>k;c=x-1;
    for(i=0;i<k;i++)
    {
        cin>>n>>a;s[a]=a;c--;
        if(n==1){cin>>b;s[b]=b;c--;}
    }
    for(i=2;i<x;i++)
    {
        if(s[i]>0) continue;
        if(s[i-1]+s[i]==0)
        {q++;i++;}
    }q=c-q;
    cout<<q<<" "<<c;
}
