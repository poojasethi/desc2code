#include<iostream>
using namespace std;
int a[4001];
int main()
{
int n,i,x,l,count=0,cnt=0,y,z;
cin>>n>>l;
a[n]=2;
for(i=0;i<l;i++)
{
cin>>x;
if(x==2)
{
count+=1;
cin>>y;
a[y]=2;
}
else if(x==1)
 {
 count+=2;
 cin>>y>>z;
 a[y]=2;
 a[z]=1;
 }
}
 for(i=1;i<n;i++)
 {
 if(a[i]==0 && a[i+1]==0)
 cnt+=1,i++;
   else if(a[i]==0 && a[i+1]!=0)
 cnt++;
 }
 cout<<cnt<<" "<<n-1-count;
 return 0;
}
