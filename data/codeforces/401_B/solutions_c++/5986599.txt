#include<iostream>
//#include<algorithm>
using namespace std;
int main()
{
int x,k,i,t,l=0,f=0,n1,n2,mi,ma;
cin>>x>>k;
int a[x-1];
for(i=0;i<x-1;i++)
{
a[i]=0;
}

for(i=0;i<k;i++)
{
cin>>t;
if(t==1)
cin>>n2>>n1;
else
cin>>n2;
a[n2-1]=1;
if(t==1)
a[n1-1]=1;
}
for(i=0;i<x-1;i++)
{
if(a[i]==0)
f++;
if(a[i]==0&&a[i+1]==0&&i!=x-2)
{
f++;
a[i+1]=1;
l++;
}
}

mi=(f-(2*l))+l;
ma=f;
cout<<mi<<" "<<ma;
return 0;

}
