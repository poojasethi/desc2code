#include <iostream>
#include <math.h>

using namespace std;
int main(){
int a[5000];
for(int i=1;i < 4001;i++)
a[i]=0;
int n, x, y,c, p=0, k, max1=0,min1=0;
cin >> n >> k;a[n]=1;
for(int i=0;i<k;i++)
{ cin >> c;
if (c==1)
 {cin >>x>>y;a[x]=1;a[y]=1;}
  else
  {cin >>x;a[x]=1;}
}
for(int i=1;i<n+1;i++)
if (a[i]==1)
{min1+=(p+1)/2;p=0;
}else {
p++;
max1++;
}
cout <<min1 << " "<<max1;
return 0;
}
