#include<cstdio>
int x,k;
int a[4050];
int main(){
 scanf("%d%d",&x,&k);
 int u,v;
 for(int i=1;i<=k;i++)
   {
       int _;scanf("%d",&_);
	   if(_==1)	{scanf("%d%d",&u,&v);a[u]=1;a[v]=1;}
	   else {scanf("%d",&u);a[u]=1;}
   }
 a[x]=1;a[0]=1;
 int ans1=0,ans2=0;
 for(int i=0;i<=x;){
   int j=i+1;		
   for(;j<=x;j++)
     if(a[j]) break;
   int l=j-i-1;
   ans1+=l;ans2+=(l+1)/2;
   i=j;	  
 }
 printf("%d %d\n",ans2,ans1);   
}
