#include<iostream>
using namespace std;
bool id[4001];
int main() {
	int x,k,a,b,s1=0,s2=0;
	cin>>x>>k;
	id[x]=true;
	while(k--) {
		cin>>a;
		a=3-a;
		while(a--) {
			cin>>b;
			id[b]=true;
			s2++;
		}
	}
	for(int i=1;i<x;i++) {
		if(id[i]==false) {
			if(id[i+1]==false)i++;
			s1++;
		}
	}
	s2=x-s2-1;
	cout<<s1<<" "<<s2;
	return 0;
}
