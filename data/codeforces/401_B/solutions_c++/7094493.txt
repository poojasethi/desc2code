#include<bits/stdc++.h>
using namespace std;
int main()
{
	int x,k,i,t,a,b,c=0,m=0;
	bool v[4010]={0};
	cin>>x>>k;
	for(i=0;i<k;i++)
	{
		cin>>t;
		if(t==1)
		{
			cin>>a>>b;
			v[a]=v[b]=1;
		}
		else if(t==2)
		{
			cin>>a;
			v[a]=1;
		}
	}
	for(i=1;i<x;i++)
	{
		if(!v[i])
			c++;
	}
	for(i=2;i<x;i++)
	{
		if(!v[i] && !v[i-1])
		{
			m++;
			i++;
		}
	}
	cout<<c-m<<' '<<c<<endl;
	return 0;
}
