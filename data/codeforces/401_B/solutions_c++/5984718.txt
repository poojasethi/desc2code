#include<iostream>
using namespace std;
int main()
{
	int i,j,s=0,h=0,ind[5000]={0};
	int x,k,a,b,c,d;
	cin>>x>>k;
	for(i=0;i<k;i++)
	{
		cin>>a;
		if(a==1)
		{
			cin>>b;
			cin>>c;
			ind[b]=1;
			ind[c]=1;
		}
		else
		{
			cin>>d;
			ind[d]=1;
		}
	}
	ind[x]=1;
	for(i=1;i<x;i++)
	{
		if(ind[i]!=1)
		{
			s++;
			h++;
			if (ind[i+1]!=1)
			{
				s++;
				ind[i+1]=1;
			}
			
		}
	}
	cout<<h<<" "<<s;
	return 0;
}
