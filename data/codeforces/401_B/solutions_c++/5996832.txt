#include<bits/stdc++.h>
using namespace std;
int main()
{
	int a,b;
	cin>>a>>b;
	int arr[a+1];
        memset(arr,0,sizeof(arr));	
        arr[a]=1;
	
	for(int i=0;i<b;++i)
	 {
		int id,val1,val2;
		cin>>id;
		if(id==1) {cin>>val1>>val2;arr[val1]=1;arr[val2]=1;}
		else { cin>>val1;arr[val1]=1;}
	 }  
	int i=1,t=0,count=0;
	while(i<a+1)
	{
		if(arr[i]==0 and arr[i+1]==0) {count++;i++;t+=2;}
		else if(arr[i]==0) t++;
		 ++i;
	}
	printf("%d %d",t-count,t);
}
