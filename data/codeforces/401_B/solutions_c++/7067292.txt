#include<cstdio>
using namespace std;int n,m,i,x,opt,Min,Max;
int f[50005];
int main()
{
  scanf("%d%d",&n,&m);
  for (i=1;i<=m;i++)
  {
    scanf("%d",&opt);
    scanf("%d",&x);f[x]=2;
    if (opt==1) scanf("%d",&x),f[x]=1;
  }
  Min=0;Max=0;
  for (i=1;i<n;i++)
    if (!f[i]) Max++;
  for (i=1;i<n;i++)
    if (!f[i]&&!f[i+1]) f[i+1]=1,Min++;
    else if (!f[i]&&f[i+1]) Min++;
  printf("%d %d",Min,Max);
  return 0;
}
