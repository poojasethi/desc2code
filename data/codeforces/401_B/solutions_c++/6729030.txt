#include<cstdio>
using namespace std;
bool b1[4001];
int main()
{
	int n,k,p,a,o1=0,o2=0;
	scanf("%d%d",&k,&n);
	while(n--)
	{
		scanf("%d%d",&p,&a);
		b1[a]=1;
		if(p==1)
		{scanf("%d",&a);b1[a]=1;}
	}
	a=0;
	for(int i=1;i<k;++i)
	if(b1[i])
	{o1+=(a+1)/2;a=0;}
	else{++a;++o2;}
	printf("%d %d\n",o1+(a+1)/2,o2);
}
