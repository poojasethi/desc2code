#include <iostream>

using namespace std;

int mark[5000];
int d[5000];

int main()
{
	int x,k;
	cin>>x>>k;
	int r,p,q;
	mark[x]=1;
	for (int i=1;i<=k;i++)
	{
		cin>>r>>p;
		mark[p]=1;
		if (r==1){
			cin>>q;
			mark[q]=1;
		}
	}	
	int tmax=0,tmin=0;
	for (int i=1;i<=x;i++)
	{
		d[i]=d[i-1];
		if (!mark[i])
			d[i]++;
		if (mark[i]){
			tmax+=d[i-1];
			tmin+=(d[i-1]+1)/2;
			d[i]=0;
		}
	}
	cout<<tmin<<" "<<tmax;


}
