#include<iostream>
using namespace std;
int main(){
	int x,k;
	int F[10001]={0};
	cin>>x>>k;
	for(int i=1;i<=k;++i){
		int a,b,c;
		cin>>a>>b;
		F[b]=1;
		if(a==1){
			cin>>c;
			F[c]=1;
		}
	}
	int min=0,max=0;
	for(int i=1;i<x;++i){
		int sum=0;
		while(F[i]==0 && i<x){
			sum++;
			i++;
		}
		min+=sum/2+sum%2;
		max+=sum;
	}
	cout<<min<<" "<<max;
	//system("pause");
	return 0;
}
