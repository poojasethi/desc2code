#include<iostream>
#include<cmath>
using namespace std;
int main()
{
int n,k,i,j,x,y,z,h[4009]={0};
float c;
cin>>n>>k;
h[n]=1;
for(i=0;i<k;i++)
{
cin>>x>>y;
//c++;
h[y]++;
if(x==1)
{cin>>z;h[z]++;}
}
j=0;

float m=0,M=0;
for(i=1;i<=n;i++)
{
if(h[i]==1)
{

}
else

}

}
