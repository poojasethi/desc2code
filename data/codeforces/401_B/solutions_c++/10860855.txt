#include <iostream>
using namespace std;
int x,k,n,m,t, a[4005];
int main(){
    cin>>x>>k;
    for (int i=0; i<k; i++){
        cin>>t;
        if (t%2) {cin>>n>>m; a[n]=1, a[m]=1;}
        else {cin>>n; a[n]=1;}
    }
    t=0, n=0, m=0;
    for (int i=1; i<x; i++){
        if (!a[i])  t++;
        else{m+=t; n+=t/2; t=0;}
    }
    if (t>0) {m+=t; n+=t/2;}
    cout<<m-n<<" "<<m;
    return 0;
}
