#include<stdio.h>

int x,y,z,i,j;
int k,n,m,s=0,s1=0;
int arr[4005]={0};

int main(){

	scanf("%d %d",&n,&m);

	for(i=0 ; i<m ; i++){
		scanf("%d",&x);

		if(x==2){
			scanf("%d",&y);
			arr[y]++;
		}
		if(x==1){
			scanf("%d %d",&y,&z);
			arr[y]++;
			arr[z]++;
		}
	}

	arr[n]=1;

	for(i=1 ; i<n ; i++){
		if(!arr[i] || arr[i]==2)
			s1++;
		if(!arr[i] && !arr[i+1]){
			s++;
			arr[i+1]=2;
		}
	}

	printf("%d %d",s1-s,s1);

	return 0;

}
