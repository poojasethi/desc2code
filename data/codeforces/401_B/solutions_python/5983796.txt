x,k=map(int,raw_input().split())
b=[0]*x
for i in range(k):
    l=map(int,raw_input().split())
    if l[0]==1:
        b[l[1]]=1
        b[l[2]]=1
    else:
        b[l[1]]=1
mx_ans=0
mn_ans=0
for i in range(1,x):
    if b[i]==0:
        mx_ans+=1

#print mx_ans
for i in range(1,x-1):
    if b[i]==0 and b[i+1]==0:
        b[i+1]=1

for i in range(1,x):
    if b[i]==0:
        mn_ans+=1
print mn_ans,mx_ans
