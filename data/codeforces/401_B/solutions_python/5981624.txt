import sys

if __name__ == '__main__':
	x,n = map(int,sys.stdin.readline().split())
	rounds = [0]*(x-1)
	for i in range(0,n):
		values = map(int,sys.stdin.readline().split())
		if values[0] == 1:
			rounds[values[1]-1], rounds[values[2]-1] = 1,1
		else:
			rounds[values[1]-1] = 1

	min_v, max_v = 0,rounds.count(0)
	g = False
	for i in range(0,len(rounds)):
		if rounds[i] == 0:
			if not g:
				min_v += 1
				g = True
			else:
				g = False
		else:
			g = False
			

	print min_v,max_v
