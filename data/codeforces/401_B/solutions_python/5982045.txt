def main():
    x, k = [int(c) for c in raw_input().split()]

    prev = [0]

    for _ in xrange(k):
        temp = [int(c) for c in raw_input().split()]
        for i in temp[1:]:
            prev.append(i)
    prev = sorted(prev) + [x]
    answer = [0, x - len(prev) + 1]
    for i in range(len(prev)-1):
        gap =  (prev[i+1]-prev[i]-1)
        answer[0] += gap//2 + (1 if gap%2 else 0)
    return ' '.join([str(i) for i in answer])

if __name__=='__main__':
    print main()
