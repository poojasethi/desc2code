x,k = map(int,raw_input().split())
a = [0] * x
for i in range(k):
  v = map(int,raw_input().split())
  if v[0]==2:
    a[v[1]] = 2
  if v[0]==1:
    a[v[1]] = 2
    a[v[2]] = 1


m,M = 0,0
i = 1
while i<x:
  j=0
  while i+j < x and a[i+j] == 0:
    j += 1
  m += j//2 + j%2
  M += j
  i += j+1
print m,M
