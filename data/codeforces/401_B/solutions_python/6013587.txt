r=lambda:map(int,raw_input().split())
x,k=r()
a=[0]*x+[1]
for i in range(k):
    s=r()
    if s[0] == 1:
        a[s[1]]=a[s[2]]=1
    else:
        a[s[1]]=1
i=1
m=0
M=0
while True:
    try:
        r = a.index(1,i)
        d=r-i
        m+=d/2+d%2
        M+=d
        i=r+1
    except:
        break
print m,M
