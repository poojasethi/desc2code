#!/usr/bin/python
def main():
    x, k = map(int, raw_input().split())
    list = []
    for line in xrange(k): # merge all lines in one list
        list.append(map(int, raw_input().split()))
    #print list
    f_list = []
    for t in list: # remove 1st item from inner lists
        f_list.append(t[1:])
    #print f_list
    flat_list = [] # merge all inner lists in one
    flat_list = [item for sub in f_list for item in sub]
    #print flat_list
    missed_list = []
    for n in xrange(1, x):
        if n not in flat_list: missed_list.append(n)
    #print missed_list
    if len(missed_list) == 0: print 0, 0
    else:
        min_missed_rounds = 0
        max_missed_rounds = len(missed_list)
        i = 0
        while i < max_missed_rounds:
            if missed_list[i] + 1 == missed_list[min(i+1, max_missed_rounds - 1)]:
                i += 2
            else: i += 1
            min_missed_rounds += 1
        print min_missed_rounds, max_missed_rounds

if __name__ == '__main__':
  main()
