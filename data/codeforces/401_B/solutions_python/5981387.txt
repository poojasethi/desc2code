def main():
    n, m= map(int, raw_input().split())
    rem = [0]
    for i in range(m):
        l = map(int, raw_input().split())
        rem.extend(l[1:])
    rem.append(n)
    rem.sort()
    maxn, minn = 0, 0
    for i, s in enumerate(rem):
        if i > 0:
            p = rem[i] - rem[i - 1] - 1
            maxn += p
            minn += (p+1) / 2
    print minn, maxn

if __name__ == '__main__':
    main()
