x, k = map(int, raw_input().split())
iterator = k
myarray = []
while iterator:
    myarray.append(map(int, raw_input().split()))
    iterator -= 1

doneEvents = []

for event in myarray:
    doneEvents.append(event[1])
    if event[0] == 1:
        doneEvents.append(event[2])

notDoneEvents = [i for i in xrange(1, x)]

notDoneEvents = list(set(notDoneEvents).difference(set(doneEvents)))

notDoneEvents.sort()

maxEvents = len(notDoneEvents)
minEvents = 0
flag = 0
i = 0
while i < (len(notDoneEvents) - 1):
    if notDoneEvents[i+1] - notDoneEvents[i] == 1:
        notDoneEvents.remove(notDoneEvents[i+1])
        notDoneEvents.remove(notDoneEvents[i])
        minEvents += 1
    else:
        i += 1

minEvents += len(notDoneEvents)

print minEvents, maxEvents
