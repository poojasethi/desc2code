I=lambda:map(int, raw_input().split())
n, k = I()
a = [0] * n 
for _ in xrange(k):
    b = I()
    if b[0] == 1:
        a[b[1]], a[b[2]] = 1, 1
    else:
        a[b[1]] = 1
c = [i for i in xrange(1, n) if a[i] == 0]
i = 0
ans, n = 0, len(c)
while i < n:
    ans += 1
    if i+1 < n and c[i] + 1 == c[i+1]: i += 2
    else: i += 1
print ans, n
