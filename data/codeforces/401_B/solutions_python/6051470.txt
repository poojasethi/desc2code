[x, k] = map(int, raw_input().split())
t = [0]*(x+1)
t[0] = 2
t[x] = 1
for i in xrange(k):
    a = map(int, raw_input().split())
    t[a[1]] = 2
    if a[0] == 1:
        t[a[2]] = 1

M = sum([1 for i in t if i == 0])
for i in xrange(x):
    if t[i] != 0: continue
    t[i] = -2
    if t[i+1] == 0: t[i+1] = -1

m = sum([1 for i in t if i == -2])
print "%d %d" % (m, M)
