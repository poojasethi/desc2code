x, k = map(int, raw_input().split())
A = [0] * x
for i in range(k):
    a = map(int, raw_input().split())
    if (a[0] == 1):
        A[a[1]] = 1
        A[a[2]] = -1
    else:
        A[a[1]] = 1
ans1 = 0
ans2 = 0
for i in range(1, x):
    if A[i] == 0:
        ans2 += 1
i = 1
while i < x - 1:
    if A[i] == 0 and A[i + 1] == 0:
        ans1 += 1
        A[i] = 1
        A[i + 1] = 1
        i += 1
    i += 1
print ans1 + A[1:].count(0), ans2
