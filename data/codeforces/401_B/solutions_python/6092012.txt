x, k = map(int, raw_input().split())

contests = [(0,)]
for i in range(k):
	t = map(int, raw_input().split())
	contests.append(tuple(t[1:]))
contests.append((x,))
contests = sorted(contests)
minimum = 0
maximum = 0
for i in range(1, len(contests)):
	now = contests[i]
	before = contests[i-1]
	diff = now[0] - before[-1] - 1
	if diff == 0:
		continue
	elif diff % 2:
		minimum += diff / 2 + 1
		maximum += diff
	else:
		minimum += diff / 2
		maximum += diff
print minimum, maximum
