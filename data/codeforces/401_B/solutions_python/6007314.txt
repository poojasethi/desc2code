def run():
    n,x=map(int,raw_input().split())
    s=set()
    for i in range(1,n):
        s=s|set([i])

    for i in range(x):
        m=map(int,raw_input().split())
        if m[0]==1:
            s=s-set([m[1],m[2]])
        else:
            s=s-set([m[1]])

    mi=len(s)
    mx=len(s)
    ii=0;
    s=list(s)
    while ii<len(s)-1:
        if s[ii]+1==s[ii+1]:
            ii=ii+1
            mi=mi-1
        ii=ii+1
    print mi,mx

if __name__=='__main__':
    run()
