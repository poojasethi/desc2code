x,k=map(int,raw_input().split())
f=[0 for i in xrange(x)]
for i in xrange(k):
    arr=map(int,raw_input().split())
    arr=arr[1:]
    for i in arr:
        f[i]+=1
arr=[]
for i in xrange(1,x):
    if f[i]==0:
        arr.append(i)
l,s=len(arr),len(arr)
c=0
i=0
while i<len(arr)-1:
    if arr[i+1]-arr[i]==1:
        c+=1
        i+=2
    else:
        i+=1
print s-c,l
