[k,n] = [int(s) for s in raw_input().split()]
arr = [0]*k + [1]
for i in xrange(n):
        a = [int(s) for s in raw_input().split()]
        arr[a[1]] = 1
        if(a[0] == 1):
                arr[a[2]] = 1
maximum = k - sum(arr)
minimum = 0
l = 0
for i in xrange(1,k+1):
        if(arr[i] == 0):
                l += 1
        else:
                minimum += (l+1)/2
                l = 0
print "{} {}".format(minimum,maximum)
