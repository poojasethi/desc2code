#!/usr/bin/env python

import sys
import logging
import traceback

logging.basicConfig(level = logging.INFO)

x, k = map(int, raw_input().split())

nums = []
for _ in range(k) : 
	nums += map(int, raw_input().split())[1:]

maxCnt = x - 1 - len(nums)
minCnt = maxCnt

s = set(nums)
pre = -1
for i in range(1, x) : 
	if i not in s : 
		if pre == 1 : 
			minCnt -= 1
			pre = -1
		else : 
			pre = 1
	else : 
		pre = -1
		continue

print minCnt, maxCnt
