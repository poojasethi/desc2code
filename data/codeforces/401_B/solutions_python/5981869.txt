R=lambda:map(int, raw_input().split(' ')) 
x,k=R()

flag=[0]*(x)

for i in xrange(k):
    v = R()
    if v[0] == 1:
        flag[v[1]] = 1
        flag[v[2]] = 1
    else:
        flag[v[1]] = 1

def cnt_min(num):

    v = [x for x in num[1:] if x <= 0]
