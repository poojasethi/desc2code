# CF R235 Div2-B
x, k = map(int, raw_input().strip().split())
oc = [False] * x
oc[x-1] = True
for _ in xrange(k):
    line = map(int, raw_input().strip().split())
    if line[0] == 2:
        oc[line[1]-1] = True
    else: # line[0] == 1
        oc[line[1]-1] = True
        oc[line[2]-1] = True

max_miss = oc.count(False)
min_miss = 0

i = 0
while i < x:
    if oc[i]:
        i += 1
    else:
        min_miss += 1
        i += 2
    
print min_miss, max_miss
