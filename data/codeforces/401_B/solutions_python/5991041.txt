x, k = map(int, raw_input().split())
rs = [0] * x
for _ in range(k):
	a = map(int, raw_input().split())
	for i in range(3 - a[0]):
		rs[a[i + 1]] = 1
rmin, rmax, f = 0, 0, 2
for i in range(1, x):
	if rs[i] == 0:
		rmax += 1
		if f % 2 == 0:
			rmin += 1
		f = (f + 1) % 2
	else:
		f = 2
print rmin, rmax
