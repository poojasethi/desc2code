x,k = map(int,raw_input().split())
s = set()
for i in range(1,x):
    s = s|set([i])
for i in range(k):
    y = map(int,raw_input().split())
    if len(y) == 3:
        s = s - set([y[1],y[2]])
    else:
        s = s - set([y[1]])
mi = len(s)
ma = len(s)
s = list(s)
i = 0
while i < len(s)-1:
    if s[i]+1 == s[i+1]:
        mi -= 1
        i += 1
    i += 1
print mi,ma
