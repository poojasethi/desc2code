import math
current,rounds=[int(i) for i in raw_input().split()]
missed=current-rounds-1
mrounds=range(1,current)
for i in range(rounds):
    day=[int(i) for i in raw_input().split()]
    mrounds.remove(day[1])
    if day[0] == 1:
        missed-=1
        mrounds.remove(day[2])
pairs=0
i=1
while 1 <= i <current:
    if i in mrounds and i+1 in mrounds:
        pairs+=1
        i+=1
    i+=1
print missed-pairs,missed
