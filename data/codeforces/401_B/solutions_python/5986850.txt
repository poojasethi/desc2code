n, x = map(int, raw_input().split())

nums = [0] * 4010
for i in range(x):
    temp = map(int, raw_input().split())
    if temp[0] == 1:
        nums[temp[1]], nums[temp[2]] = 1, 1
    else:
        nums[temp[1]] = 1

ma, mi = 0, 0

for i in range(1, n):
    if not nums[i]:
        ma += 1

i = 1
while i < n:
    if nums[i] != 1:
        t = 0
        while i < n and not nums[i]:
            i += 1
            t += 1
        mi += (t + 1) / 2
    else:
        i += 1

print mi, ma
