def ar():
    return map(int,raw_input().split())

x,k=ar()
b=range(1,x)
for i in range(k):
    a=ar()
    if len(a)==3:
        b.remove(a[1])
        b.remove(a[2])
    else:
        b.remove(a[1])
ma=len(b)
mi=ma
i=0
while i< len(b)-1:
    if b[i+1]-b[i]==1:
        mi-=1
        i+=1
    i+=1
print mi,ma
