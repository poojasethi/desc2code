#include <stdio.h>

int main()
{
    int n;
    char s[100100], t[100100];

    scanf("%d%s%s", &n, s, t);

    int l = 0, r = n - 1;
    while(s[l] == t[l]) l++;
    while(s[r] == t[r]) r--;

    int ans = 2;
    for(int i = l; i != r; i++)
        if(s[i] != t[i + 1])
            {ans--; break;}

    for(int i = l; i != r; i++)
        if(t[i] != s[i + 1])
            {ans--; break;}

    printf("%d", ans);

    return 0;
}
