#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main() {
	int n;
	cin >> n;
	string s, t;
	cin >> s >> t;
	int first = -1, last = -1;
	for (int i = 0; i < n; i++) {
		if (s[i] != t[i]) {
			if (first == -1) {
				first = i;
			}
			last = i;
		}
	}
	bool one = true, two = true;
	for (int i = first; i < last; i++) {
		if (s[i] != t[i+1]) {
			one = false;
		}
		if (t[i] != s[i+1]) {
			two = false;
		}
	}
	int ans = 0;
	if (one) ans++;
	if (two) ans++;
	cout << ans;
	return 0;
}
