#include <bits/stdc++.h>
#define L first
#define R second
#define MP make_pair
#define PB push_back

using namespace std;

main() {
	int n;
	string s, t;
	cin >> n;
	cin >> s >> t;
	int p1 = 0;
	while(s[p1] == t[p1]) p1 ++;
	int p2 = s.size() - 1;
	while(s[p2] == t[p2]) p2 --;
	bool correct = true;
	for(int i = p1;i < p2;i ++)
		if(s[i] != t[i + 1]) {
			correct = false;
			break;
		}
	int ans = 0;
	ans += correct;
	correct = true;
	for(int i = p2;i > p1;i --)
		if(s[i] != t[i - 1]) {
			correct = false;
			break;
		}
	ans += correct;
	cout << ans << endl;
	return 0;
}
