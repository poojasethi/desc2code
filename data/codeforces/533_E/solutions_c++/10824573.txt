#include <bits/stdc++.h>
using namespace std;
string a, b;
int main() {
    int n;
    cin >> n >> a >> b;
    int p = 0, q = n - 1;
    while(p < n && a[p] == b[p]) p++;
    while(q >= 0 && a[q] == b[q]) q--;
    if(p == q) {
        cout << "2" << endl;
    } else {
        int flag1 = 0, flag2 = 0;
        for(int i = p + 1; i <= q; i++) {
            if(a[i] != b[i - 1]) flag1 = 1;
        }
        for(int i = p; i <= q - 1; i++) {
            if(a[i] != b[i + 1]) flag2 = 1;
        }
        cout << !flag1 + !flag2 << endl;
    }
    return 0;
}
