#include<iostream>
#include<string>

using namespace std;

int f(string s1,int i,string s2,int j)
{
  while(s1[i]==s2[j]){
    i++;j++;
  }

  if(i<j)
    i++;
  else
    j++;

  while(i<s1.size()){
    if(s1[i]!=s2[j])
      return 0;
    i++;j++;

  }

  return 1;
}

int main(){

  int n;

  string s1,s2;

  cin>>n>>s1>>s2;

  for(int i=0;i<n;i++)
    {
      if(s1[i]==s2[i])
	continue;

      cout<<f(s1,i,s2,i+1) + f(s1,i+1,s2,i);

      return 0;
    }
}
