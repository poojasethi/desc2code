#include<bits/stdc++.h>
using namespace std;

string s, t;

bool same(int i, int j, int ii){
	while(i<ii){
		if(s[i] != t[j])
			return false;
		i++, j++;
	}
	return true;
}

int main(){
	int n;
	cin >> n >> s >> t;
	for(int i=0;i<n;i++){
		if(s[i] != t[i]){
			if(same(i+1, i+1, n)){
				cout << 2 << endl;
				return 0;
			}
			for(int j=n-1;j>=0;j--){
				if(s[j]!=t[j]){
					int x=0;
					x=same(i, i+1, j) + same(i+1, i, j+1);
					cout << x << endl;
					return 0;
				}
			}
			break;
		}
	}
	cout << 0 << endl;
	return 0;
}
