#include<bits/stdc++.h>
using namespace std;
char a[100005],b[100005];
int main(){
    int i,j,la,ra,lb,rb,n;
    scanf("%d%s%s",&n,a,b);
    la=lb=0;
    ra=rb=n-1;
    int cnt=0;
    while(la<=ra && lb<=rb && a[la]==b[lb]) la++,lb++;
    while(la<=ra && lb<=rb && a[ra]==b[rb]) ra--,rb--;
    if(la==ra){
        puts("2");
        return 0;
    }
    int ans=0,flag=1;
    for(i=la+1;i<=ra;i++){
        if(a[i]!=b[i-la-1+lb])
            flag=0;
    }
    if(flag) ans++;
    flag=1;
    for(i=lb+1;i<=rb;i++){
        if(b[i]!=a[i-lb-1+la])
            flag=0;
    }
    if(flag) ans++;
    printf("%d\n",ans);
}
