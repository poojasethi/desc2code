#include<stdio.h>
int i,l,d,a,b,p,n;
char s[1000000],t[1000000];
int main(){
    l=-1;
    scanf("%d\n", &n);
    for(i=0; i<=n; i++) scanf("%c", &s[i]);
    for(i=0; i<=n; i++) {
            scanf("%c", &t[i]);
        if(t[i]==s[i]){
               if(d<0) d=i;
               if(p==0) l=i;
           }
           else{
               d=-1;
               p=1;
           }
    }
    if (l>=d){
        printf("%d", (n+1)*26);
        return 0;
    }

    a=1;
    b=1;
    for(i=l+2; i<d; i++){
        if (s[i]!=t[i-1]) a=0;
        if (s[i-1]!=t[i]) b=0;
    }
    printf("%d", a+b);
    return 0;
}
