#include <bits/stdc++.h>
using namespace std;

int main() {
	int n,i,j,c=0;
	string s,t;
	scanf("%d",&n);
	cin>>s;
	cin>>t;
	for(i=0;i<n;i++)
	if(s[i]!=t[i])
	break;
	for(j=n-1;j>=0;j--)
	if(s[j]!=t[j])
	break;
	if(i==j)
	{
		printf("2");
		return 0;
	}

	if(s.substr(i,(j-i))==t.substr((i+1),(j-i)))
	c++;
	if(s.substr(i+1,(j-i))==t.substr(i,(j-i)))
	c++;
	printf("%d",c);
	return 0;
}
