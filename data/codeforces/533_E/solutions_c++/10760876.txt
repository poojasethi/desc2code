#include <stdio.h>
#include <string.h>

char S[100009];
char T[100009];

int n;

int main()
{
    scanf("%d\n%s\n%s", &n, S, T);

    int pref = 0;
    while (pref < n && S[pref] == T[pref]) pref++;
    int suff = n-1;
    while (suff >= 0 && S[suff] == T[suff]) suff--;

    //printf("%d %d", pref, suff);

    int ans = 0;

    int i;
    for (i = pref; i < suff; i++) {
        if (S[i] != T[i+1]) break;
    }
    if (i == suff) {
        ans++;
    }

    int j;
    for (j = suff; j > pref; j--) {
        if (S[j] != T[j-1]) break;
    }
    if (j == pref) {
        ans++;
    }

    printf("%d", ans);
}
