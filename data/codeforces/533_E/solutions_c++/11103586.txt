#include<iostream>
using namespace std;

int main(){
	ios::sync_with_stdio(0);
	string s1, s2;
	int n;
	cin >> n >> s1 >> s2;
	if (s1 == s2){
		cout << (n+1) * 26 << endl;
		return 0;
	}
	int I = 0;
	while(s1[I] == s2[I])
		I++;
	int J = 0;
	while(s1[n-1-J] == s2[n-1-J])
		J++;
	if (I+J+1 == n){
		cout << 2 << endl;
		return 0;
	}
	int tx = n - I - J - 1;
	int ans = 0;
	if (s1.substr(I, tx) == s2.substr(I+1, tx))
		ans++;
	if (s1.substr(I+1, tx) == s2.substr(I, tx))
		ans++;
	cout << ans << endl;	
}
