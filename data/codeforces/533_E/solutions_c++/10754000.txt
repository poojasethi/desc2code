#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
using namespace std;
char s1[100100],s2[100100];
int main(){
    int n,x,y;
    scanf("%d",&n);
    scanf("%s%s",s1,s2);
    for(x=0;x<n;x++) if(s1[x]!=s2[x])break;
    for(y=n-1;y>=0;y--) if(s1[y]!=s2[y]) break;
        int flag=1,ans=0;
        for(int i=x;i<y;i++) if(s1[i]!=s2[i+1]) flag=0;
        if(flag) ans++;
        flag=1;
        for(int i=x;i<y;i++) if(s1[i+1]!=s2[i]) flag=0;
        if(flag) ans++;
        printf("%d\n",ans);
}
