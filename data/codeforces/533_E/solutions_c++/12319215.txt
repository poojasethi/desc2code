#include <stdio.h>

int main()
{
    int n;
    char s[100100], t[100100];

    scanf("%d%s%s", &n, s, t);

    int pos_l, pos_r;
    for(int i = 0; i < n; i++)
        if(s[i] != t[i])
            {pos_l = i; break;}
    for(int i = n - 1; i >= 0; i--)
        if(s[i] != t[i])
            {pos_r = i; break;}

    if(pos_l == pos_r)
        {printf("2"); return 0;}

    int ans = 0, i;
    for(i = pos_l; i != pos_r; i++)
        if(s[i] != t[i + 1]) break;
    if(i == pos_r) ans++;

    for(i = pos_l; i != pos_r; i++)
        if(t[i] != s[i + 1]) break;
    if(i == pos_r) ans++;

    printf("%d", ans);

    return 0;
}
