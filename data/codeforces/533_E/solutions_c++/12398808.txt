#include<cstdio>

int len, x, y, ans;
char a[111111], b[111111];

bool check(int l1, int r1, int l2, int r2) {
	for(int i=l1;i<=r1;i++)
		if(a[i] != b[i-l1+l2]) return false;
	return true;
}

int main() {
	scanf("%d%s%s", &len, a, b);
	x = 0, y=len-1;
	while(a[x] == b[x]) x++;
	while(a[y] == b[y]) y--;
	ans=0;
	if(check(x+1, y, x, y-1)) ans++;
	if(check(x, y-1, x+1, y)) ans++;
	printf("%d\n", ans);
	
	return 0;
}
