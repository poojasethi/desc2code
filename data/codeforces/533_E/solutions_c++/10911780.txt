#include<bits/stdc++.h>
using namespace std;

int n , i , j , ans ;
string a , b ;

bool HandleSpl() {
  if(i==n) cout << 26*(n+1) ;
  if(i==j) cout << 2 ;
  return (i==n || i==j) ;
}

int main() {
  cin >> n >> a >> b ;
  for( i=0 ; i<n ; i++) if(a[i]!=b[i]) break ; 
  for( j=n ; j>0 ; j--) if(a[j-1]!=b[j-1]) break ;
  j-- ;

  if(HandleSpl()) return 0 ;

  if(a.substr(i,j-i)==b.substr(i+1,j-i)) ans++ ;
  if(b.substr(i,j-i)==a.substr(i+1,j-i)) ans++ ;   
  cout << ans ;
  
  return 0 ;
}
