#include <bits/stdc++.h>
using namespace std;
int main(){
	string s,t;int n;cin>>n>>s>>t;
	int i=0,j=n-1,ret=0;
	while(i<n && s[i]==t[i])
		++i;
	while(j>=0 && s[j]==t[j])
		--j;
	if(s==t){ret= 26*n;	return 0;}
	int x= i,y= j;
	while(x<j && s[x]==t[x+1])
		++x;
	if(x==j)++ret;
	while(i<y && s[y]==t[y-1])
		--y;
	if(i==y)++ret;
	cout<<ret<<endl;
}
