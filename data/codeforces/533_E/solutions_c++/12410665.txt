#define N (1<<10)
#include <bits/stdc++.h>

using namespace std;

string s,t;
int n,cnt,Left,Right;

bool judge(int x,int y,int z)
{
	for(int i=x,j=z;i<=y;i++,j++)
		if(s[i]!=t[j]) return 0;
	return 1;
}

int main()
{
	cin>>n>>s>>t;
	for(int i=0;i<n;i++)
		if(s[i]!=t[i])
		{
			Left=i;
			break;
		}

	for(int i=n-1;i>=1;i--)
		if(s[i]!=t[i])
		{
			Right=i;
			break;
		}
	cout<<judge(Left+1,Right,Left)+judge(Left,Right-1,Left+1)<<endl;
}
