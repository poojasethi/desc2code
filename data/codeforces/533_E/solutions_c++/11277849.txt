#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
char s[100010], t[100010];
int n;

int main() {
	//freopen( "A.in", "r", stdin );

	scanf( "%d", &n );
	scanf( "%s%s", s, t );

	int l = 0, r = n - 1;
	while ( s[l] == t[l] ) l ++;
	while ( s[r] == t[r] ) r --;

	int ret = 0, flag = 1;
	for ( int i = l; i < r; i ++ ) {
		if ( s[i + 1] != t[i] )	{
			flag = 0;
			break;	
		}
	}
	ret = flag;
	flag = 1;

	for ( int i = l; i < r; i ++ ) {
		if ( s[i] != t[i + 1] ) {
			flag = 0;
			break;
		}
	}
	ret += flag;

	cout <<ret<<endl;

	return 0;	
}
