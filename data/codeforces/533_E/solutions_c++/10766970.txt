#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define pb push_back

int main(){
	int n; cin >> n;
	string s,t;
	cin >> s >> t;
	int cnt=0;
	for(int i=0;i<s.size();i++){
		cnt += (s[i] != t[i]);
	}
	if(cnt >= 2){
		vector<int>vec;
		for(int i=0;i<s.size();i++){
			if(s[i] != t[i]) vec.pb(i);
		}
		int a = vec[0],b = vec.back();
		int res=0;
		for(int i=a;i<b;i++){
			if(s[i] != t[i+1]) goto nxt;
		}
		res++;
		nxt:;
		for(int i=a;i<b;i++){
			if(t[i] != s[i+1]) goto nxt2;
		}
		res++;
		nxt2:;
		cout << res << endl;
	}
	else if(cnt == 1){
		puts("2");
	}
}
