#include <string.h>
#include <stdio.h>
char s[100111],t[100111];
int n;

int dp[100111][2][2];
int ans;
void dfs(int i,int j,int sc,int xc)
{
    for(;i<=n&&j<=n;i++,j++)
    {
        if(s[i]!=t[j])
        {
            if(sc==0)
            dfs(i+1,j,sc+1,xc);
            if(xc==0)
            dfs(i,j+1,sc,xc+1);
            return ;
        }
    }
    if(i>n||j>n){
        ans++;
    }
}
int main()
{
    scanf("%d%*c",&n);
    scanf("%s",s+1);
    scanf("%s",t+1);
    memset(dp,0,sizeof(dp));
    dp[0][0][0]=1;
    ans=0;
    dfs(1,1,0,0);
    int aaa=0;
    aaa++;
    printf("%d\n",ans);
    
}
