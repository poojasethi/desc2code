#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
const int maxn = 101000;
int ans = 0, n;
char s[maxn], t[maxn];
void dfs(int i, int j, int ii, int jj) {
    if(i == n && j == n) {
        ans ++;
        return;
    }
    if(s[i] == t[j]) {
        dfs(i+1, j+1, ii, jj);
        return;
    }
    if(ii == 0) {
        dfs(i+1, j, 1, jj);
    }
    if(jj == 0) {
        dfs(i, j+1, ii, 1);
    }
    return;
}
int main() {
    scanf("%d", &n);
    scanf("%s%s", s, t);
    dfs(0, 0, 0, 0);
    printf("%d\n", ans);
    return 0;
}
