#include<stdio.h>
#include<string.h>
char s[100005],t[100005],str[100005];
int n;
bool cmp(char*ss,char*tt){
	int i=0,j=0;
	while(i<n&&ss[i]==tt[i])i++;
	while(j<n&&ss[n-j]==tt[n-j-1])j++;
	return i+j>=n;
}
int main(){
	while(scanf("%d%s%s",&n,s,t)!=EOF){
		int ps=0;
		int ans=0;
		while(s[ps]==t[ps])ps++;
		memcpy(str,s,sizeof(char)*ps);
		str[ps]=t[ps];
		memcpy(str+ps+1,s+ps,sizeof(char)*(n-ps));
		if(cmp(str,t))ans++;
		memcpy(str,t,sizeof(char)*ps);
		str[ps]=s[ps];
		memcpy(str+ps+1,t+ps,sizeof(char)*(n-ps));
		if(cmp(str,s))ans++;
		printf("%d\n",ans);
	}
}
