#include<cmath>
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int N=1e5+5;
char a[N],b[N];
int n;
int main()
{
	cin>>n;
	scanf("%s%s",&a[1],&b[1]);
	int i=1,j=n;
	while(a[i]==b[i])i++;
	while(a[j]==b[j])j--;
	if(i==j){cout<<2;return 0;}
	int f=2;
	for(int k=i+1;k<=j;k++)
		if(a[k]!=b[k-1]){f--;break;}
	for(int k=i+1;k<=j;k++)
		if(a[k-1]!=b[k]){f--;break;}
	if(f>0){cout<<f;return 0;}
	cout<<0;
}
