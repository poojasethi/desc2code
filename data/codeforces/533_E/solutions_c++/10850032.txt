#include<bits/stdc++.h>

using namespace std;

int n,i,j,k,l;
string S,T;

int main(){

	cin>>n>>S>>T;

	for(i=0 ; i<n ; i++)
		if(S[i]!=T[i])
			break;

	for(j=n-1 ; j>=0 ; j--)
		if(S[j]!=T[j])
			break;
	
	for(k=i+1 ; k<=j ; k++)
		if(S[k]!=T[k-1])
			break;

	for(l=i+1 ; l<=j ; l++)
		if(T[l]!=S[l-1])
			break;

	cout<<(k==j+1)+(l==j+1);

	return 0;
}
