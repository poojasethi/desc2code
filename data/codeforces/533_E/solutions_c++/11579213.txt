#include <cstdio>
#include <cstring>

#define N 100100

char a[N], b[N];

bool can(int x, int y, int xx, int yy) {
	for (int i = 0; i <= y - x; i ++) if (a[x + i] != b[xx + i]) return 0;
	return 1;
}

main() {
//	freopen("E.in", "r", stdin);
	int n;
	scanf("%d", &n);
	scanf("%s", a);
	scanf("%s", b);
	int S, E;
	for (S = 0; S < n; S ++) if (a[S] != b[S]) break;
	for (E = n - 1; E >= 0; E --) if (a[E] != b[E]) break;
	if (S == E) puts("2");
	else printf("%d\n", can(S + 1, E, S, E - 1) + can(S, E - 1, S + 1, E));
}
