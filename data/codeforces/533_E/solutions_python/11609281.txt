n = int(raw_input())
s = raw_input().split()[0]
t = raw_input().split()[0]

i=0
while i<n:
    if s[i] is t[i]:
        i += 1
    else:
        break
j=n-1
while j>0:
    if s[j] is t[j]:
        j -= 1
    else:
        break
        
l = i
m = j
count = 0
if s[l:m] == t[l+1:m+1] :
    count += 1
if s[l+1:m+1] == t[l:m]:
    count += 1
print count
