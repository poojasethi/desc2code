#include <iostream>

using namespace std;

int main()
{
    string x;
    getline(cin,x);
    for(int i=0;i<x.size();i++)
    {
         if(x[i]!=' ' || x[i+1] >= 'a')
    {
      cout << x[i];
      if(x[i]>' ' && x[i] < 'a')
        {
            x[i]=' ';
            i--;
        }
    }}
    return 0;
}
