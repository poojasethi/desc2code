#include<iostream>
#include<string>
#include<ctype.h>
using namespace std; 
int main()
{
    string word ; 
    getline(cin , word);
for(int i = 0 ; i < word.size()-1 ; i++)
{
    if ((isspace(word[i])) && (ispunct(word[i+1])))
            word.erase(i,1),i--;
    if ((isspace(word[i])) && (isspace(word[i+1])))
            word.erase(i,1),i--;
    if (ispunct(word[i]) && !isspace(word[i+1]))
            word.insert(i+1," ");
}
cout<<word<<endl;
}
