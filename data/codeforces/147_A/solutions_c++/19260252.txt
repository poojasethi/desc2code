#include <iostream>

using namespace std;

string s;

int main() {
    getline(cin, s);
    for(int i = 0; i < s.size(); ++i) {
        if(s[i] != ' ' || s[i + 1] >= 'a') {
            cout << s[i];
            if(s[i] > ' ' && s[i] < 'a') {
                s[i] = ' ';
                --i;
            }
        }
    }
    return 0;
}
