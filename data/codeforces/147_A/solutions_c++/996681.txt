#include <stdio.h>
#include <string>

using namespace std;
int main()
{
	int s = 0;
	char x;
	do
	{
		scanf("%c",&x);
		if(x != ' ' && x != '.' && x != ',' && x != '!' && x != '?')
		{
			if(s != 0){printf(" ");s = 0;}
			printf("%c",x);
		}
		else if(x != ' '){printf("%c",x);s = 1;}
		else s = 1;
	}while(x != 10);
}
