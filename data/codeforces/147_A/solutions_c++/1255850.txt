#include <iostream>
using namespace std;

int main()
{
	string str;
	while (getline(cin, str) )
	{
		string res = "";
		for ( int i=0 ; i<str.length() ; ++i )
		{
			if ( isalpha(str[i]) )
			{
				res += str[i];
			}
			else if ( ispunct(str[i]) )
			{
				res += str[i];
				res += ' ';
			}
			else if ( str[i] == ' ' && res[res.length()-1] != ' ' && isalpha(str[i+1]) )
			{
				res += ' ';
			}
		}
		cout << res << endl;
	}

	return 0;
}
