#include <iostream>
#include <cstring>
#include <cstdio>

using namespace std;

int main()
{
    char s[10010],rez[10010];
    gets(s);
    int j=0,length=strlen(s);
    for (int i=0; i<=length; i++)
    {
        if (not(s[i]==' '&&(s[i+1]==' '|s[i+1]=='\0'|s[i+1]=='.'
            |s[i+1]==','|s[i+1]=='!'|s[i+1]=='?'))) {rez[j]=s[i]; j++;}
        if ((s[i]=='.'|s[i]==','|s[i]=='!'|s[i]=='?')&&s[i+1]!=' ') {rez[j]=' '; j++;}
    }
    rez[j]='\0';
    cout<<rez<<endl;
}
