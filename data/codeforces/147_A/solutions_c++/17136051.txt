#include<bits/stdc++.h>
using namespace std;

int is_punc(char c)
{
    if(c=='.' || c==',' || c=='!' || c=='?') return 1;
    return 0;
}

int main()
{
	char s[10050];
	gets(s);
	for(int i=0;i<strlen(s);i++){
		if(is_punc(s[i])) cout<<s[i];
		if(s[i+1]>='a' && s[i+1]<='z' && (s[i]==' '||s[i]=='.'||s[i]==','||s[i]=='?'||s[i]=='!') && i+1<strlen(s)){
			cout<<" ";
		}
		if(s[i]>='a'&&s[i]<='z'){
			cout<<s[i];
		}
	}
	return 0;
}
