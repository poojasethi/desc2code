#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
string st;
int main(){
	getline(cin,st);
	int i;
	for(i=0;i<st.size();i++){
		if(st[i]==' ' && (st[i+1]==' ' || st[i+1]=='.' || st[i+1]==',' || st[i+1]=='!' || st[i+1]=='?'))
			continue;
		cout<<st[i];
		if((st[i]=='.' || st[i]==',' || st[i]=='!' || st[i]=='?') && st[i+1]!=' ')
			cout<<" ";
	}
	return 0;
}
