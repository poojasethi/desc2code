#include<cstdio>

using namespace std; 
char c,last ;

void ico(){
	last='a';
	while(1){
		c=getchar();
		if(c=='\n')
			break;
		if(c=='.' || c=='!' || c=='?' || c==',')
			putchar(c);
		else if(c!=' ' &&( last>'z' || last<'a')){
			putchar(' ');
			putchar(c);
		}
		else if(c!=' ')
			putchar(c);
		last=c;
	}
}

int main(){
	ico();
	return 0;
}
