#include <cstdio>
#include <cstring>

int main()
{
	bool espacar = false;
	char st[10001];
	gets(st);
	int tam = strlen(st);
	for(int i = 0; i < tam; i++)
	{
		if(st[i] == ' ')
		{
			espacar = true;
		}
		else if(st[i] == ',' || st[i] == '!' || st[i] == '.' || st[i] == '?')
		{
			printf("%c", st[i]);
			espacar = true;
		}
		else
		{
			if(espacar)
			{
				espacar = false;
				printf(" ");
			}
			printf("%c", st[i]);
		}
	
	}
}
