#include<string.h>
#include<stdio.h>
int main()
{
   int l,i,j=0;
   char s[10009],ans[10009];
   gets(s);
   l=strlen(s);
   for(i=0;i<l;i++)
   {
      if(s[i]==' ' && (s[i+1]==' ' || s[i+1]=='.' || s[i+1]==',' || s[i+1]=='!' || s[i+1]=='?' ))
         continue;
      if((s[i-1]=='.' || s[i-1]== ',' || s[i-1]=='!' || s[i-1]== '?') && s[i]!=' ')
         ans[j++]=' ';
      ans[j++]=s[i];
   }
   ans[j]='\0';
   printf("%s\n",ans);
   return 0;
}
