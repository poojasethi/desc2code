#include <iostream>
#include <cstring>

using namespace std;

char s[10002];
int i, lg;

int main() {
    cin.getline(s, 10002);
    
    lg = strlen(s);
    
    for(i = 0;i < lg;i++) {
       if(s[i] != ' ' || s[i + 1] >= 'a') {
           cout << s[i];
       }
       if(s[i] > ' ' && s[i] < 'a') {
           s[i] = ' ';
           i--;
       }
    }

    return 0;
}
