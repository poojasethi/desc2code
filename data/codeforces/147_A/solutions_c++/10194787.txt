#include<bits/stdc++.h>
using namespace std;
int main()
{
    char st[100001];
    gets(st);
    int i,l=strlen(st);
    for(i=0;i<l;i++){
        if(st[i]==' ' && (st[i+1]==' ' || st[i+1]=='.' || st[i+1]==',' || st[i+1]=='!' || st[i+1]=='?'))
            continue;
        cout<<st[i];
        if((st[i]=='.' || st[i]==',' || st[i]=='!' || st[i]=='?') && st[i+1]!=' ')
            cout<<" ";
    }
    return 0;
}
