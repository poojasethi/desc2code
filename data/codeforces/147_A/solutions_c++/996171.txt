#include<iostream>
using namespace std;
int main()
{
	bool tre=0;
	string a;
	getline(cin,a);
	a=a+' ';
	for(int i=0 ; i < a.size() ;i++)
	{
		if(a[i]!=' ' && a[i]!='.' && a[i]!=','&& a[i]!='!'&& a[i]!='?') 
		{
			int j=i;
			i++;
			while(a[i]!=' ' && a[i]!='.' && a[i]!=','&& a[i]!='!'&& a[i]!='?')
				i++;
			if(tre)
				cout<<" "<<a.substr(j,i-j);
			else
			{
				cout<<a.substr(j,i-j);	
				tre=1;
			}
			i--;
			continue;
		}
		if(a[i]!=' ')
			cout<<a[i];
	}
	cout<<endl;
}
