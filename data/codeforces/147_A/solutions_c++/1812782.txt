#include<stdio.h>
#include<ctype.h>
int main()
{
    char c;
    int flag=0;
    while(1)
    {
    while(isalpha(c=getchar())) 
    {if(flag) putchar(' ');flag=0;putchar(c);}
    flag=1;
    
    while(c!=','&&c!='?'&&c!='.'&&c!='!'&&!isalpha(c)&&c!='\n'&&c!=EOF) c=getchar();
    if(c=='\n'||c==EOF) return 0;
    if(isalpha(c)) {printf(" %c",c);flag=0;}
    else putchar(c);
    }
    return 0;
}
