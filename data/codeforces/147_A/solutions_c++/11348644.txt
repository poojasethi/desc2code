#include<iostream>
#include<stdio.h>

using namespace std;

int main(){
    char c,last='a';
	while(c!='\n'){
		c=getchar();
		if(c=='.' || c=='!' || c=='?' || c==',')
			putchar(c);
		else if(c!=' ' &&( last>'z' || last<'a')){
			putchar(' ');
			putchar(c);
		}
		else if(c!=' ')
			putchar(c);
		last=c;
	}
	return 0;
}
