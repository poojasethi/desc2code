#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
string s;
int ind, p;
int main()
{
while(cin>>s){
for(int i=0;i<s.length();i++){
if(s[i]!='!' && s[i]!='?' && s[i]!='.' && s[i]!=','){
    if(i==0 && ind>0 && p==0)
    cout<<" ";
cout<<s[i];
p=0;
}
else{
cout<<s[i]<<" ";
p=1;
}
}
ind++;
}
}
