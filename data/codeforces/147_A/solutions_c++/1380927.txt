#include<stdlib.h>

	char c,dizi[10005],sa=0;

	while(scanf("%c",&c)!=EOF){
		if(c=='\n')break;
		
		dizi[say++]=c;
	}
	dizi[say]='\0';
	for(i=0;i<say;i++)
	{	bek=0;
		if(dizi[i]==' ')
			for(; dizi[i]==' ' && i<say ;i++)bek=1;
		if(dizi[i]=='.' || dizi[i]==',' || dizi[i]=='!' || dizi[i]=='?'){
			printf("%c",dizi[i]);

				printf(" ");
			continue;
		}

		printf("%c",dizi[i]);
	}

}
