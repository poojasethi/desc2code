#include<iostream>

using namespace std;

bool l;
string s,t;
int ind;

int main()
{
  getline(cin,s);
  s+='.';
  for(int i=0;i<s.size()-1;i++)
    if(s[i]<='z' && s[i]>='a')
      {
        cout<<s[i];
	l=false;
      }
    else
      {
        if(s[i]==' ')
	  {
	    if(s[i+1]<='z' && s[i+1]>='a' && !l)
	      {
		cout<<s[i];
		l=true;
	      }
	  }
	else
	  {
	    cout<<s[i]<<" ";
	    l=true;
	  }
      }
  return 0;
}
