#include<iostream>
#include <string>

using namespace std;

int main()
{
	string s;
	int flag = 0;

	getline(cin,s);

	for(int i=0; i < s.size(); i++)
	{
		if(s[i] == ' ' )
		{	
			flag = 1;	
		}

		else if((s[i] >= 'a' && s[i] <= 'z') || (s[i] >= 'A' && s[i] <= 'Z'))
		{
			if(flag ==1)
			{
				cout << " ";
			}
		       cout << s[i];
		       flag =0;
		}

		else {
			cout << s[i];
			flag =1;
		}
	}

}
