#include <iostream>
#include <string>
using namespace std;
#define sz(x) int((x).size())
string s;
int main()
{
	getline(cin,s);
	for (int i=0;i<sz(s);i++)
		if (isalpha(s[i]))
			cout<<s[i];
		else if (s[i]==' ')
		{
			if (i<sz(s)-1 && isalpha(s[i+1]))
				cout<<s[i];
		}
		else
		{
			cout<<s[i];
			if (i<sz(s)-1 && isalpha(s[i+1]))
					cout<<' ';
		}
	cout<<endl;
	return 0;
}
