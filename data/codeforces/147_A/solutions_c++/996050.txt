#include<iostream>
#include<string>
using namespace std;
int main()
{
	string str,res;
	getline( cin , str);
	bool s = false , f =false;
	for(int i = 0; i < str.length(); ++i )
	{
		if( str[i] == ' ' ) {  s =true; }
		else if( str[i] >= 'a' && str[i] <= 'z' )
		{
			
			if( s == true )
			{
				res+=' ';
				s=false;
			}
			res+=str[i];
		}
		else
		{
			res+=str[i];
		
			s = true;
		}
	}
	cout << res << "\n";
	return 0;
}
