#include <iostream>
#include <cstdio>

using namespace std;

char p [] = ".,!?";
char s [10009];

int main () {
	int i;
	bool ws = true;
	gets (s);
	for (i = 0; s [i]; ++ i) {
		if ('a' <= s [i] && s [i] <= 'z') {
			if (!ws) {
				ws = true;
				cout << " ";
			}
			cout << s [i];
		}
		else {
			if (ws) {
				ws = false;
			}
			if (s [i] != ' ') {
				cout << s [i];
			}
		}
	}
	cout << '\n';
	return 0;
}
