#include<stdio.h>
#include<string>
#include<iostream>
using namespace std;
int main()
{
int n,i,j;
string a;
getline(cin,a);
n=a.length();
for (i=1;i<n;i++){
if(a[i-1]==' ' && a[i]==' ') {
for (j=i;j<n;j++)
a[j]=a[j+1];
n--;
i--;
}
if(a[i]>32 && a[i]<65 && a[i+1]!=' ') {
for (j=n;j>i+1;j--)
a[j]=a[j-1];
a[i+1]=' ';
n++;
}
if(a[i]>32 && a[i]<65 && a[i-1]==' '){
a[i-1]=a[i];
a[i]=' ';
i--;
}
}
for (i=0;i<n;i++)
printf("%c",a[i]);
return 0;
}
