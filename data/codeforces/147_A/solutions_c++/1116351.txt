#include<cstdio>
#include<cctype>
char s[10240];
int main(){
	for(;gets(s);puts("")){
		bool k=true;char *p=s;
		for(;*p==' ';++p);
		for(*p;*p;){
			if(*p==' '){
				for(;*p==' '&&*p;++p);
				if(!k && !ispunct(*p) && *p){putchar(' ');k=false;}
			}
			if(ispunct(*p)){putchar(*p);putchar(' ');k=true;}
			else {putchar(*p);k=false;}
			++p;
		}
	}
	return 0;
}
