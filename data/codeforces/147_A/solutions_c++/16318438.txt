#include <iostream>
#include <string>
using namespace std;
int i,o;
string s;
bool pr(char a){
	if(a=='!' || a=='?' || a=='.' || a==',')return 0;
	else return 1;
}
int main(){
    getline(cin,s);
	for(i=0;i<s.length();i++){
		if(s[i]!=' '){
			cout<<s[i];
		    if(pr(s[i])==0 && s[i+1]!=' ')cout<<" ";
		}
		else{
			while(s[i]==' ')i++;
			i--;
			if(pr(s[i+1])==1)cout<<" ";
		}
	}
    return 0;
}
