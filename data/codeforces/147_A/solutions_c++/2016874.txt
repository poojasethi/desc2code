#include <iostream>

using namespace std;

bool f(char a) {
  if (a == '.' || a == ',' || a == '!' || a == '?')
    return false;
  else
    return true;
}

int main() {
  string s;
  getline(cin, s);
  for (int i = 0, n = s.size(); i < n; i++)
    if (s[i] != ' ') {
      if (i && f(s[i]) && (s[i - 1] == ' ' || !(f(s[i - 1]))))
        cout << " ";
      cout << s[i];
    }
  
  return 0;
}
