s = raw_input()
s = s.strip()
s1 = ''
punkt = [',', '.', '!', '?']
for i in range(len(s)):
    if s[i] == ' ' and s[i + 1] == ' ':
        s1 += ''
    else:
        if s[i] == ' ' and s[i + 1] in punkt:
            s1 += ''
        else:
            if s[i] in punkt and s[i + 1] != ' ':
                s1 += s[i] + ' '
            else:
                s1 += s[i]  
print(s1)
