#!/usr/bin/python

import re

pr1 = re.compile(r" *[,]")
ps1 = ", "

pr2 = re.compile(r" *[!]")
ps2 = "! "

pr3 = re.compile(r" *[.]")
ps3 = ". "

pr4 = re.compile(r" *[?]")
ps4 = "? "

m = re.compile(r"[ ]+")
l = re.compile(r"[ ]$")

s = raw_input()

s = re.sub(pr1, ps1, s)
s = re.sub(pr2, ps2, s)
s = re.sub(pr3, ps3, s)
s = re.sub(pr4, ps4, s)

s = re.sub(m,  " ", s)
s = re.sub(l,   "", s)

print s
