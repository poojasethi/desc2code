import sys

s = sys.stdin.readline().strip()

new_s = ''
l = ['.', ',', '!', '?']
i = 0
while i < len(s):
    if s[i].isalpha():
        new_s += s[i]
        i += 1
    elif s[i] == ' ' and new_s[-1]==' ':
        i += 1
    elif s[i] == ' ' and s[i+1].isalpha() and new_s[-1]!=' ':
        new_s += s[i]
        i += 1
    elif s[i] == ' ' and not s[i+1].isalpha() and new_s[-1]!=' ':
        i += 1
    elif s[i] in l:
        new_s += s[i] + ' '
        i += 1
print new_s
