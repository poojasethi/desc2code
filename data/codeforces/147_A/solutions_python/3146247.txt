def is_p(c):
    return c!=' ' and not (c>='a' and c<='z')
    
s=raw_input()
i,j=0,0
n=len(s)
ans=[]
while j<n:
    if s[j]==' ':
        while i<j and s[i]==' ': i+=1
        if i<=j-1:ans.append(s[i:j])
        i=j
    elif is_p(s[j]):
        while i<j and s[i]==' ':i+=1
        if i<=j-1: ans.append(s[i:j])
        ans.append(s[j:j+1])
        i=j+1
    j+=1
while i<j and s[i]==' ':i+=1
if i<j: ans.append(s[i:j])
z=''
for x in ans:
    if is_p(x[0]): z+=x
    else:z+=' '+x
print z[1:]
