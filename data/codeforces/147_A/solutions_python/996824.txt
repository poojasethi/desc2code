import sys

lst = sys.stdin.readline().strip().split()
res = ""
puncs = [',', '.', '!', '?']
for w in lst:
    for p in puncs:
        w = w.replace(p, p+' ')
    res += w + ' '   
lst = res.split()
res = ""
for w in lst:
    if w in puncs:
        res = res.strip()
    res += w + " "

print res.strip()
