import re
s = raw_input()

s = re.sub(r"\ *,", ",", s)
s = re.sub(r",", ", ", s)
s = re.sub(r",\ +,", ", ", s)

s = re.sub(r"\ *\.", ".", s)
s = re.sub(r"\.", ". ", s)
s = re.sub(r",\ +\.", ". ", s)

s = re.sub(r"\ *!", "!", s)
s = re.sub(r"!", "! ", s)
s = re.sub(r"!\ +,", "! ", s)

s = re.sub(r"\ *\?", "?", s)
s = re.sub(r"\?", "? ", s)
s = re.sub(r"\?\ +,", "? ", s)

s = re.sub(r"\ +", " ", s)

print s
