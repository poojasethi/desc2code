s = raw_input().strip()
a = [""]
for i in s:
    if i.isalpha():
        a[-1] += i
    else:
        a += [i, ""]
na = []
for i in range(len(a)):
    a[i] = a[i].strip()
    if a[i] != "":
        na += [a[i]]
res = ' '
for i in na:
    if i.isalpha():
        res += ' ' + i
    else:
        res += i
print res.strip()
