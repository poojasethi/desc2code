# -*- coding:utf-8 -*-
import sys

def solve(text):
    text = text.replace(',', ' , ').replace('.', ' . ')\
            .replace('!', ' ! ').replace('?', ' ? ').split()
    ret = ''
    for word in text:
        if word is ',' or word is '.' or word is '!' or word is '?':
            ret += word
        else:
            ret += ' ' + word

    ret = ret[1:]
    print ret

def main():
    text = raw_input()
    solve(text)

if __name__ == '__main__':
    main()
