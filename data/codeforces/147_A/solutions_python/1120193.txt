import sys
import re
def punc(s):
	if re.search(r'[?!,.]', s):
		return 1
	else:
		return 0
def printf(fmt, *args):
	sys.stdout.write(fmt % args)
line = raw_input()
parts = line.split(' ')
while parts.count('') > 0:
	parts.remove('')
tmp = ' '.join(parts)
lent = len(tmp)
tmp += '#'
i = 0
while i < lent:
	if tmp[i] == ' ' and punc(tmp[i+1]):
		i += 1
	elif punc(tmp[i]) > 0 and tmp[i+1] != ' ':
		printf("%s ", tmp[i:i+1])
		i += 1
	else:
		printf("%s", tmp[i:i+1])
		i += 1
