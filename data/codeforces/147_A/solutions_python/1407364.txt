needSpace = False
needSpaceIf = False
s = []
for c in raw_input():
    if c == ' ':
        if not needSpace and not needSpaceIf:
            needSpaceIf = True
        pass
    elif c == ',' or c == '.' or c == '!' or c == '?':
        if needSpace: s.append(' ')
        s.append(c)
        needSpace = True
        needSpaceIf = False
    else:
        if needSpaceIf or needSpace: s.append(' ')
        s.append(c)
        needSpace = False
        needSpaceIf = False

print ''.join(s)
