#include<iostream>
using namespace std;
int f(int n,int m){int r=0;while((n)&&(n%m==0)){++r;n/=m;}return r;}
int main(){
 int a,b,w;cin>>a>>b;w=0;
 while(f(a,2)>f(b,2)){++w;a/=2;}while(f(b,2)>f(a,2)){++w;b/=2;}
 while(f(a,3)>f(b,3)){++w;a/=3;}while(f(b,3)>f(a,3)){++w;b/=3;}
 while(f(a,5)>f(b,5)){++w;a/=5;}while(f(b,5)>f(a,5)){++w;b/=5;}
 if(a==b)cout<<w<<endl;else cout<<-1<<endl;
 return 0;
}
