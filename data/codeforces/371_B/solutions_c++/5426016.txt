#include <cstdio>
#include <algorithm>
using namespace std;
int main(){
    int a,b,k,s=0;
    scanf("%d %d",&a,&b);
    k=__gcd(a,b);
    a/=k;b/=k;
    while (!(a%2)) a/=2,s++;
    while (!(a%3)) a/=3,s++;
    while (!(a%5)) a/=5,s++;
    while (!(b%2)) b/=2,s++;
    while (!(b%3)) b/=3,s++;
    while (!(b%5)) b/=5,s++;
    if (a>1||b>1) printf("-1\n"); else
    printf("%d\n",s);
    return 0;
}
