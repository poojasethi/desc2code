#include <iostream>
#include <cmath>
#include<cstdlib>
using namespace std;
int a,b,ret=0;
void func(int i)
{
    int c1=0,c2=0;
    while (a%i==0)a/=i,c1++;
    while (b%i==0)b/=i,c2++;
    ret+=abs(c1-c2);
}
int main()
{
    cin >> a >> b;
    func(2);
    func(3);
    func(5);
    if (a==b)cout << ret;
    else cout << -1;
    return 0;
}
