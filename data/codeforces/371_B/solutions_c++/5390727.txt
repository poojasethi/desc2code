#include <iostream>
#include <cmath>

using namespace std;

int a,b;
int  ans = 0;

void check(int div) {
	int ansa,ansb;
	ansa = 0;
	ansb = 0;
	while(a % div == 0 ) { a /= div; ansa ++ ; }
	while(b % div == 0 ) { b /= div; ansb ++; }
	ans += abs(ansa - ansb);
}
int main() {

	
	cin>>a>>b;
	check(2);
	check(3);
	check(5);
	if(a == b )
	cout << ans << endl; 
	else cout << "-1"<< endl;


	return 0;	
}
