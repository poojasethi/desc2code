#include <iostream>
#include <cmath>

using namespace std;

int A , B , ans;

int Count(int k) {
    int u = 0 , v = 0;
    while (A % k == 0) {A /= k; u++;}
    while (B % k == 0) {B /= k; v++;}
    return abs(u - v);
}

int main(){
    cin >> A >> B;

    ans += Count(2) + Count(3) + Count(5);
    cout << (A == B ? ans : -1) << endl;
}
