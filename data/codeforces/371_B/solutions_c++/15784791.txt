#include<cstdio>
#include<cstdlib>
int main()
{
	int a,b,c=0,d=0,e=0,f=0,g=0,h=0;
	scanf("%d%d",&a,&b);
	while(a%2==0){
		a/=2;c++;
	}
	while(a%3==0){
		a/=3;d++;
	}
	while(a%5==0){
		a/=5;e++;
	}
	while(b%2==0){
		b/=2;f++;
	}
	while(b%3==0){
		b/=3;g++;
	}
	while(b%5==0){
		b/=5;h++;
	}
	if(a!=b) printf("-1");
	else printf("%d",abs(c-f)+abs(d-g)+abs(e-h));
}
