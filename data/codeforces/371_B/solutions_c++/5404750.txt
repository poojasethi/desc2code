#include <cstdio>
#include <algorithm>
using namespace std;
int a,b,ans(0);
int f(int *x,int y){
	int k(0);
	while(*x){
		if(*x%y)break;
		*x/=y;
		k++;
	}
	return k;
}
int abs(int x){return x>0?x:-x;}
int main(){
	scanf("%d%d",&a,&b);
	ans=abs(f(&a,2)-f(&b,2))+abs(f(&a,3)-f(&b,3))+abs(f(&a,5)-f(&b,5));
	if(a==b)printf("%d\n",ans);else printf("-1\n");
	return 0;
}
