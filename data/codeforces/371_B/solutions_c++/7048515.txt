#include<iostream>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
#include<cmath>
#include<vector>
using namespace std;
int find(int *n,int d){
	int c=0;
	while(*n%d==0){
		c++;
		*n/=d;
	}
	return c;
}
int main(){
	int a,b;
	cin>>a>>b;
	int c=abs(find(&a,2)-find(&b,2))+abs(find(&a,3)-find(&b,3))+abs(find(&a,5)-find(&b,5));
	if(a==b)
		cout<<c;
	else
		cout<<"-1";
	return 0;
}
