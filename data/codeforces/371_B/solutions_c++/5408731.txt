#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
    int a, b, ans=0;
    scanf("%d %d", &a, &b);
    int c = __gcd(a, b);
    a/=c, b/=c;
    while(a%2==0) a/=2, ++ans;
    while(a%3==0) a/=3, ++ans;
    while(a%5==0) a/=5, ++ans;
    while(b%2==0) b/=2, ++ans;
    while(b%3==0) b/=3, ++ans;
    while(b%5==0) b/=5, ++ans;
    if(a!=1||b!=1) ans = -1;
    printf("%d\n", ans);
    return 0;
}
