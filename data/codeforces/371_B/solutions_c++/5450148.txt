#include<stdio.h>
__int64 gcd(__int64 a,__int64 b)
{
	if (a==0) return b;
	return gcd(b%a,a);
}
int main()
{
	__int64 A,B;
	while (scanf("%I64d%I64d",&A,&B)!=EOF)
	{
		__int64 x=gcd(A,B);
		__int64 n=A/x*B/x;
		int ops=0;
		while (n%2==0) {n/=2;ops++;}
		while (n%3==0) {n/=3;ops++;}
		while (n%5==0) {n/=5;ops++;}
		if (n>1) printf("-1\n");
		else printf("%d\n",ops);
	}
	return 0;
}
