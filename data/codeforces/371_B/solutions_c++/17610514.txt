#include <cstdio>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
		return a;
	return gcd(b,a%b) ;
}
int tagh(int &a,int k)
{
	int u=0 ;
	while(a%k==0)
	{
		u++ ;
		a/=k ;
	}
	return u;
}
int main()
{
	int a,b ;
	scanf("%d %d",&a,&b) ;
	int gc=gcd(a,b) ;
	a/=gc ;
	b/=gc ;
	int sum=tagh(a,2)+tagh(a,3)+tagh(a,5)+tagh(b,2)+tagh(b,3)+tagh(b,5) ;
	if(a==1 && b==1)
		printf("%d",sum) ;
	else
		printf("-1") ;
}
