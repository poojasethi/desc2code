#include <iostream>
using namespace std;
int n, m;
int sum;
int abs(int a)
{
	if (a < 0) return -a;
	return a;
}
void isDivisible(int a)
{
	int x = 0, y = 0;
	while (n%a == 0)
	{
		n /= a;
		x++;
	}
	while (m%a == 0)
	{
		m /= a;
		y++;
	}
	sum += abs(x-y);
}
int main()
{
	cin >> n >> m;
	isDivisible(2);
	isDivisible(3);
	isDivisible(5);
	if (n == m)  cout << sum;
	else cout << "-1";
}
