#include<stdio.h>
#include<math.h>

int main()
{
    int a, b, ans = 0,kount,n;
    scanf("%d %d",&a,&b);
    for (n = 2; n <= 5; n++)
    {
        kount =0;
        while (a % n == 0) a /= n, ++kount;
        while (b % n == 0) b /= n, --kount;
        ans +=fabs(kount);
    }
    if(a==b) printf("%d",ans);
    else  printf("-1");
    return 0;
}
