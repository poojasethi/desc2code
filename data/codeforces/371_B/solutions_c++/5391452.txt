#include <cstdio>
#include <algorithm>
using namespace std;
//我不知道为什么
int a, b;
int ans;

void judge(int p)
{
	int u = 0, v = 0;	
	while (b % p == 0) { b /= p; v++; }
	while (a % p == 0) { a /= p; u++; }
	ans += abs(u - v);
}

int main()
{
	scanf("%d %d", &a, &b);
	judge(5); judge(2); judge(3);
	if (a == b) printf("%d\n", ans);
	else printf("-1\n");
	return 0;
}
