#include <stdio.h>
#include <iostream>
using namespace std;
int p=0,a,b;
void make(int v){
	int s=0,t=0;
	while (a%v==0) a/=v,s++;
	while (b%v==0) b/=v,t++;
	if (s-t>0) p+=s-t; else p+=t-s;
}
int main( ){
	cin>>a>>b;
	make(2);
	make(3);
	make(5);
	if (a==b) cout<<p<<endl; else cout<<-1<<endl;
	return 0;
}
