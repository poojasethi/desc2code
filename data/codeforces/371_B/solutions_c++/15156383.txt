#include<stdio.h>
int main(void)
{
	int a, b, sum, i, k;
	scanf("%d%d", &a, &b);
	sum = 0;
	for(i=2;i<=5;i++)
	{
		k = 0;
		while(a%i==0)
		{
			a /= i;
			k += 1;
		}
		while(b%i==0)
		{
			b /= i;
			k -= 1;
		}
		if(k>=0)
		    sum += k;
		else
			sum += -k;
	}
	if(a==b)
	    printf("%d\n", sum);
	else
		printf("-1\n");
	return 0;
}
