#include <iostream>
#include <algorithm>

using namespace std;

int main(){
  int n;
  int m;
  cin >> n >> m;
  int g=__gcd(n,m); 
  n=n/g;
  m=m/g;
  int rpta=0; 
  for (int i=2;i<6;i++){
    if(n%i==0){
      n=n/i;
      i--;
      rpta+=1;
    }
  }
  for (int i=2;i<6;i++){
    if(m%i==0){
      m=m/i;
      i--;
      rpta+=1;
    }
  }
  if (n==1 and m==1) cout << rpta << endl;
  else cout << -1 << endl;
  return 0;
}
