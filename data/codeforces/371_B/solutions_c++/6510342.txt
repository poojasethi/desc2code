#include <iostream>
#include <cstdio>
#include <cstring>
int a,b,ans,cnt,i;
void solve(int flag){
	cnt = 0;
	while(a % flag == 0){
		a /= flag;
		cnt ++;
	}
	while(b % flag == 0){
		b /= flag;
		cnt --;
	}
	if(cnt < 0)
		cnt = -cnt;
	ans += cnt;
}
int main(){
	scanf("%d%d",&a,&b);
	ans = 0;
	solve(2);
	solve(3);
	solve(5);
	if(a != b)
		ans = -1;
	printf("%d\n",ans);
    return 0;
}
