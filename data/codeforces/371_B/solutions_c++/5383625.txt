#include <iostream>
using namespace std;
int gcd(int x, int y)
{
	if (x%y==0) return y;
	else return gcd(y, x%y); 
} 
int main()
{
	int a, b, k, x;
	cin>>a>>b;
	x=gcd(a,b);
	a/=x; b/=x;
	k=0;
	while (a!=b) {
		if (a<b) {
			x=a; a=b; b=x;
		}
		if (a%5==0) a/=5;
		else if (a%3==0) a/=3;
		else if (a%2==0) a/=2;
		else {
			k=-1; break;
		}
		k++;
	}
	cout<<k<<endl;
	return 0;
}
