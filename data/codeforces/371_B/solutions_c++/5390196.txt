#include <cstdio>
#include <algorithm>
using namespace std;

int a, b;
int res;

void solve(int x){
    int u = 0, v = 0;
    while (a % x == 0){
        a /= x; u++;
    }
    while (b % x == 0){
        b /= x; v++;
    }
    res += abs(u - v);
}

int main()
{
	scanf("%d %d", &a, &b);
	solve(2);
	solve(3);
	solve(5);
	if (a == b) printf("%d\n", res);
	else printf("-1\n");
	return 0;
}
