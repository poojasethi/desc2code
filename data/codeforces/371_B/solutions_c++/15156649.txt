#include<stdio.h>
//#include<stdlib.h>
//#include<math.h>
int main()
{
	int a,b,n,k,i,f;
	while(scanf("%d%d",&a,&b)!=EOF)
	{
		f=0;
		for(i=2;i<=5;i++)
		{
			n=k=0;
			while(a%i==0)
			{
				a=a/i;
				n++;
			}
			while(b%i==0)
			{
				b=b/i;
				k++;
			}
			if(n>k)
				f+=n-k;
			if(n<k)
				f+=k-n;
		}	
		if(a==b)	
			printf("%d\n",f);
		else
			printf("-1\n");
	}
	return 0;
}
