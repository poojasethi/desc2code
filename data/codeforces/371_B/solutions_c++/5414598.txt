#include <iostream>
#include <stdio.h>
using namespace std;
int a,b;
int kol(int d)
{int k1=0,k2=0;
 while (a%d==0) {a/=d;k1++;}
 while (b%d==0) {b/=d;k2++;}
 if (k1>k2) return k1-k2; else return k2-k1;
}
int main()
{cin>>a>>b;
int ans=kol(2)+kol(3)+kol(5);
if (a==b) cout<<ans; else cout<<-1;
return 0;
}
