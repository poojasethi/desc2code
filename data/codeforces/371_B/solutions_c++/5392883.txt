# include <iostream>
using namespace std;
int main ()
{
    int a, o, ans=0, i=2;
    cin>>a>>o;
    while(i<6)
    {
        while(a%i==0&&o%i==0)
        {
            a=a/i;
            o=o/i;
        }
        while(a%i==0||o%i==0)
        {
            ans++;
            if(a%i==0)a=a/i;
            if(o%i==0)o=o/i;
        }
        i++;
    }
    if(a==o)cout<<ans;
    else cout<<-1;
}
