#include<stdio.h>
#include<algorithm>

using namespace std;

int a,b,s=0;

void Check(int p)
{
   int x,y;
   x=y=0;
   while(a%p==0) { a=a/p; x++;}
   while(b%p==0) { b=b/p; y++;}
   s=s+abs(x-y);
}

main()
{
//  freopen("0.in","r",stdin);
//  freopen("0.out","w",stdout);
  
  scanf("%d %d",&a,&b);
  Check(2);
  Check(3);
  Check(5);
  if(a==b) printf("%d\n",s);
    else printf("-1\n");
}
