#include<iostream>
using namespace std;

int a,b,ans;

void Wrap(int p){
	while(a%p==0 && b%p==0){
		a/=p;
		b/=p;
	}
	while(a%p==0){
		a/=p;
		ans++;
	}
	while(b%p==0){
		b/=p;
		ans++;
	}
}

int main(){
	cin>>a>>b;
	Wrap(2);
	Wrap(3);
	Wrap(5);
	if((a!=1 || b!=1) && (a!=b)) cout<<-1<<endl;
	else cout<<ans<<endl;
	return 0;	
}
