#include <iostream>
#include <cmath>
using namespace std;
int main()
{
    int n[2],x[3]= {2,3,5},ans;
    while(ans=0,cin>>n[0]>>n[1])
    {
        for(int i=0,a[2]={0}; i<3;ans+=fabs(a[1]-a[0]),a[0]=a[1]=0,i++)
            for(int j=0; j<2; j++)while(n[j]%x[i]==0)n[j]/=x[i],a[j]++;
        n[1]!=n[0]?cout<<-1<<endl:cout<<ans<<endl;
    }
    return 0;
}
