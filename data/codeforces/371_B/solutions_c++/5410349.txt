# include <iostream>
using namespace std;

int main()
{
  int a=0,b=0,k=0,i=2;
  cin>>a>>b;
  while(i<6)
   {
     while(a%i==0 && b%i==0)
      {
        a=a/i;
        b=b/i;
      }
     while(a%i==0||b%i==0)
      {
        k++;
        if(a%i==0)a=a/i;
        if(b%i==0)b=b/i;
      }
     i++;
    }
  if(a==b) cout<<k<<"\n";
  else cout<<-1<<"\n";
}
