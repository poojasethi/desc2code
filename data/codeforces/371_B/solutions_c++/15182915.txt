#include <bits/stdc++.h>
using namespace std;

int a_mult[3], b_mult[3];

int main() {
	int a, b, tab[]={2,3,5};
	ios_base::sync_with_stdio(0);
	
	cin >> a >> b;
	
	for(int i=0; i<3; i++) {
		while(a%tab[i]==0) { a_mult[i]++; a /= tab[i]; }
		while(b%tab[i]==0) { b_mult[i]++; b /= tab[i]; }	
	}
	
	if(a!=b) cout << "-1\n";
	else cout << abs(a_mult[0]-b_mult[0])+abs(a_mult[1]-b_mult[1])+abs(a_mult[2]-b_mult[2]);
	
	return 0;
}
