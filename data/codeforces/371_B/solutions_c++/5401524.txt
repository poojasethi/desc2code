#include<stdio.h>
#include<algorithm>
using namespace std;
int ans=0;
void eq(int *a,int *b,int num)
{
	if(*a==*b)
	{
		return;
	}
	int op1=0,op2=0;
	while(*a%num==0){*a=*a/num; op1++;}
	while(*b%num==0){*b=*b/num; op2++;}
	ans+=abs(op1-op2);
}
int main()
{
	int a,b;
	scanf("%d %d",&a,&b);
	eq(&a,&b,5);
	eq(&a,&b,3);
	eq(&a,&b,2);
	if(a==b)
	{
		printf("%d\n",ans);
	}
	else
		printf("-1\n");
	return 0;
}
