#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int cnt1[10],cnt2[10];
int main(){
	ios_base::sync_with_stdio(false);
	int a,b;
	cin >> a >> b;
	for(int i = 2; i <= 5; ++i){
		while(a%i==0){
			cnt1[i]++;
			a/=i;
		}
		while(b%i==0){
			cnt2[i]++;
			b/=i;
		}
	}
	if(a!=b){
		cout << "-1\n";
		return 0;
	}
	cout << abs(cnt1[2]-cnt2[2])+abs(cnt1[3]-cnt2[3])+abs(cnt1[5]-cnt2[5]) << '\n';
}
