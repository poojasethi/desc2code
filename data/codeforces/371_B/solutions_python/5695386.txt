a, b = map(int, raw_input().split())
a1, a2, a3 = 0, 0, 0
b1, b2, b3 = 0, 0, 0
while a & 1 == 0:
    a /= 2
    a1 += 1
while a % 3 == 0:
    a /= 3
    a2 += 1
while a % 5 == 0:
    a /= 5
    a3 += 1
while b & 1 == 0:
    b /= 2
    b1 += 1
while b % 3 == 0:
    b /= 3
    b2 += 1
while b % 5 == 0:
    b /= 5
    b3 += 1
if a != b:
    print -1
else:
    print abs(a1 - b1) + abs(a2 - b2) + abs(a3 - b3)
