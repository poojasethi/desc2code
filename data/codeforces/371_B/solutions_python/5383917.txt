#!/usr/bin/env python
from fractions import gcd

import sys
l = map(int, raw_input().split())
a = l[0];b=l[1]

g = gcd(a, b)

g1 = a/g
g2 = b/g


c1 = 0;c2=0
p = g1
while g1 != 1:
	p = g1
	# print g1
	if g1 % 2 == 0:
		g1 = g1/2
		c1 +=1
	elif g1 % 3 == 0:
		g1 /= 3
		c1+=1
	elif g1 % 5 ==0:
		g1 /= 5
		c1+=1
	if p == g1:
		print "-1"
		sys.exit(0)

while g2 != 1:
	p = g2
	# print g2
	if g2 % 2 == 0:
		g2 = g2/2
		c2 +=1
	elif g2 % 3 == 0:
		g2 /= 3
		c2+=1
	elif g2 % 5 ==0:
		g2 /= 5
		c2+=1
	if p == g2:
		print "-1"
		sys.exit(0)

print c1+c2
