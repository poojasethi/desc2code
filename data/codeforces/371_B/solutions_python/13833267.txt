x , y = map(int,raw_input().split())

x_2,x_3,x_5,y_2,y_3,y_5 = 0,0,0,0,0,0

while x % 2 == 0:
	x /= 2
	x_2 += 1

while x % 3 == 0:
	x /= 3
	x_3 += 1

while x % 5 == 0:
	x /= 5
	x_5 += 1

while y % 2 == 0:
	y /= 2
	y_2 += 1

while y % 3 == 0:
	y /= 3
	y_3 += 1

while y % 5 == 0:
	y /= 5
	y_5 += 1


if x != y:
	print -1

else:
	print abs(x_2 - y_2) + abs(x_5 - y_5) + abs(x_3 - y_3)
