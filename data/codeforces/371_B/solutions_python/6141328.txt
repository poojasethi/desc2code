
import sys

a,b=map(int,raw_input().split())

t1,d1,v1=0,0,0
while a%2==0:
    t1+=1
    a/=2
while a%3==0:
    d1+=1
    a/=3
while a%5==0:
    v1+=1
    a/=5
t2,d2,v2=0,0,0
while b%2==0:
    t2+=1
    b/=2
while b%3==0:
    d2+=1
    b/=3
while b%5==0:
    v2+=1
    b/=5
if a==b:
    print abs(t1-t2)+abs(d1-d2)+abs(v1-v2)
else:
    print -1
