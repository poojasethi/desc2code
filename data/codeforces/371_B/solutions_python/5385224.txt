import fractions
def count(x):
    res = 0
    div = [2,3,5]
    for i in div:
        while x%i == 0:
            x /= i
            res += 1
    if x == 1:
        return res
    else:
        return -1
a, b = map(int, raw_input().split())
g = fractions.gcd(a, b)
tmp_a = a/g
tmp_b = b/g
cnt_a = count(tmp_a)
cnt_b = count(tmp_b)
if cnt_a < 0 or cnt_b < 0:
    print -1
else:
    print cnt_a + cnt_b
