from fractions import gcd
a,b=map(int,raw_input().split())
c=gcd(a,b)
a1=a/c
b1=b/c
c1=0
c2=0
while(a1%2==0):
    a1/=2
    c1+=1
while(a1%3==0):
    a1/=3
    c1+=1
while(a1%5==0):
    a1/=5
    c1+=1
while(b1%2==0):
    b1/=2
    c2+=1
while(b1%3==0):
    b1/=3
    c2+=1
while(b1%5==0):
    b1/=5
    c2+=1
if a1==1 and b1==1:
    print c1+c2
else:
    print -1
