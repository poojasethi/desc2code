n, k = map(int, raw_input().split())

def spl(i):
	a, b, c = (0, 0, 0)
	while not i % 2:
		a += 1
		i /= 2
	while not i % 3:
		b += 1
		i /= 3
	while not i % 5:
		c += 1
		i /= 5
	return (i, a, b, c)

ns = spl(n)
ks = spl(k)

if ns[0] != ks[0]:
	print -1
else:
	print abs(ns[1] - ks[1]) + abs(ns[2] - ks[2]) + abs(ns[3] - ks[3])
