a, b = map(int, raw_input().split())

ad2 = 0
bd2 = 0
ad3 = 0
bd3 = 0
ad5 = 0
bd5 = 0

while a % 2 == 0:
    a /= 2
    ad2 += 1
while a % 3 == 0:
    a /= 3
    ad3 += 1
while a % 5 == 0:
    a /= 5
    ad5 += 1
while b % 2 == 0:
    b /= 2
    bd2 += 1
while b % 3 == 0:
    b /= 3
    bd3 += 1
while b % 5 == 0:
    b /= 5
    bd5 += 1
    
if a != b:
    print -1
else:
    s = abs(ad2-bd2)+abs(ad3-bd3)+abs(ad5-bd5)
    print s
