import collections
piece_one,piece_two = map(int,raw_input().split())


def primes(n):
    primfac = []
    d = 2
    while d*d <= n:
        while (n % d) == 0:
            primfac.append(d)  # supposing you want multiple factors repeated
            n /= d
        d += 1
    if n > 1:
       primfac.append(n)
    return primfac

def remove_num(a,b):
	p_a = primes(a)
	p_b = primes(b)
	delta = list(collections.Counter(p_a)-collections.Counter(p_b))
	for x in delta:
		if x != 2 and x != 3 and x != 5:
			return -1
	if 2 in delta:
		return a/2
	if 3 in delta:
		return a/3
	if 5 in delta:
		return a/5

def start(x,y):
	iterations = 0
	while x != y:
		if x > y:
			x = remove_num(x,y)
			if x == -1:
				return -1
			iterations += 1
		else:
			y = remove_num(y,x)
			if y == -1:
				return -1
			iterations +=1
	return iterations

print start(piece_one,piece_two)
