dp = dict()

def rec(a, b):
    if a == b:
        return 0
    if (a, b) in dp:
        return dp[(a, b)]
    res = 1000
    for k in [2, 3, 5]:
        if a % k == 0:
            res = min(res, rec(a / k, b) + 1)
        if b % k == 0:
            res = min(res, rec(a, b / k) + 1)
    dp[(a, b)] = res
    return res

a, b = map(int, raw_input().split())

if rec(a, b) == 1000:
    print -1
else:
    print rec(a, b)
