r = lambda: map(int, raw_input().split())

import sys

a, b = r()
def gcd(a, b):
    if b == 0: return a
    return gcd(b, a % b)
    
tem = gcd(a, b)
a, b = a / tem, b / tem

def disp(n):
    ret = 0
    for s in [2, 3, 5]:
        while n and n % s == 0:
            ret += 1
            n /= s
    if n > 1: 
        print -1
        sys.exit(0)
    return ret
    
print disp(a) + disp(b)
