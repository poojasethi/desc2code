def mdc(a,b):
    if(b==0):
        return a
    else:
        return mdc(b, a%b)

m1, m2 = map(int, raw_input().split())

if m1 == m2:
    print 0
    exit()

num = mdc(max(m1,m2),min(m1,m2))
num1 = m1/num
num2 = m2/num
c = 0
for i in [2,3,5]:
    while num1 % i == 0:
        c += 1
        num1 /= i
    while num2 % i == 0:
        c += 1
        num2 /= i

if num1 != 1 or num2 != 1:
    print -1
else:
    print c
