def max_power(n, x):
    r = 0
    while (n % x == 0):
        r += 1
        n /= x
    return r

a,b = raw_input().split()
a = int(a)
b = int(b)

a2 = max_power(a, 2)
a /= (2 ** a2)
a3 = max_power(a, 3)
a /= (3 ** a3)
a5 = max_power(a, 5)
a /= (5 ** a5)

b2 = max_power(b, 2)
b /= (2 ** b2)
b3 = max_power(b, 3)
b /= (3 ** b3)
b5 = max_power(b, 5)
b /= (5 ** b5)

if (a != b):
    print "-1"
else:
    res = abs(a2 - b2) + abs(a3 - b3) + abs(a5 - b5)
    print str(res)
