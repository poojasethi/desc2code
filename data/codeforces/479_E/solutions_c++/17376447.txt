#include<bits/stdc++.h>
using namespace std;
long long dp[5123][5123];
int main()
{
	long long n,a,b,k,mod=1e9+7;
	cin>>n>>a>>b>>k;
	if(a<b)
	{
		a=n-a+1;
		b=n-b+1;
	}
	for(int i=a;i<=n;i++)
		dp[0][i]=1;

	for(int i=1;i<=k;i++){
		for(int j=b+1;j<=n;j++){
			dp[i][j]+=dp[i][j-1];
			dp[i][j]+=dp[i-1][n]-dp[i-1][j];
			dp[i][j]+=dp[i-1][j-1]-dp[i-1][(j+b)/2];
			while(dp[i][j]<0) dp[i][j]+=mod;
			dp[i][j]%=mod;
		}
	}
	cout<<dp[k][n]<<endl;
	return 0;
}
