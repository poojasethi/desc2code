#include<bits/stdc++.h>
const int  mod =  1e9+7;
using namespace std;
int  dp[5001][5001];
int n,x,b,k;
int main ()
{
	cin>>n>>x>>b>>k;
	if(x>b) x=n-x+1,b=n-b+1;
	for(int i=0; i<=k; i++)
		for(int j=1; j<=b; j++)
			dp[i][j] = (((i ?  (((dp[i-1][b-1]%mod) - (dp[i-1][j]%mod)+mod)%mod) +(((dp[i-1][j-1]%mod) - ((dp[i-1][max(2*j-b,0)])%mod)+mod)%mod) : 1)%mod) + 
					(dp[i][j-1]%mod)) % mod;

	cout<<((dp[k][x]%mod)-(dp[k][x-1]%mod)+mod)%mod;
}
