#include <iostream>
#include <algorithm>
using namespace std;
#define MOD 1000000007
long long n,a,b,k,dpa[5005],pref[5005];
int main(){
    int low,high;
    cin>>n>>a>>b>>k;
    for(int i=1;i<=n;i++)dpa[i]=1;
    for(int i=1;i<=k;i++){
        pref[0]=0;
        for(int j=0;j<=n;j++)pref[j]=(pref[j-1]+dpa[j])%MOD;
        for(int j=1;j<=n;j++){
            low=max(j-(abs(b-j)-1),1LL);
            high=min(j+(abs(b-j)-1),n);
            dpa[j]=pref[high]-pref[low-1]-dpa[j];
        }
    }
    dpa[a]%=MOD;
    if(dpa[a]<0)dpa[a]+=MOD;
    cout<<dpa[a];
    return 0;
}
