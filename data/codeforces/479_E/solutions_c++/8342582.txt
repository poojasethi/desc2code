#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;
const int m=1000000007;
int n,a,b,k;
long long maxn;
long long sum[5001];
int f[5001][5001];
int main(){
	scanf("%d%d%d%d",&n,&a,&b,&k);
	f[0][a]=1;
	for(int i=1;i<=k;i++){
		for(int j=1;j<=n;j++) sum[j]=sum[j-1]+f[i-1][j];
		for(int j=1;j<=n;j++){
			if(j>b){
				f[i][j]=(sum[n]-sum[j-(j-b-1)/2-1]-f[i-1][j])%m;
			}else{
				f[i][j]=(sum[j+(b-j-1)/2]-f[i-1][j])%m;
			}
			f[i][b]=0;
		}
	}
	for(int i=1;i<=n;i++) maxn=(maxn+f[k][i])%m;
	cout<<maxn; 
	return 0;
}
