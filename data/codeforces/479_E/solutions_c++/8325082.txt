#include<bits/stdc++.h>
#define maxx 5005
#define ll long long
#define mod 1000000007
using namespace std;
ll dp[maxx][maxx];
int main()
{
	int i,j,n,k,a,b;
	cin>>n>>a>>b>>k;
	for(i=a;i<=n;i++) dp[0][i]=1;
	for(i=1;i<=k;i++){
		for(j=1;j<=n;j++){
			if(j<b)
				dp[i][j]=(dp[i-1][j+b-1>>1]-dp[i-1][0]-dp[i-1][j]+dp[i-1][j-1])%mod;
			else if(j>b)
				dp[i][j]=(dp[i-1][n]-dp[i-1][j+b>>1]-dp[i-1][j]+dp[i-1][j-1])%mod;
		}
		for(j=1;j<=n;j++)
			dp[i][j]+=dp[i][j-1];
	}
	cout<<(dp[k][n]+mod)%mod;
}
