#include <stdio.h>
#include <string.h>
const int M=1000000007;
int n,a,b,k,v[2][5432],*p=v[0],*q=v[1];
int main(){
  scanf("%d%d%d%d",&n,&a,&b,&k);
  if(a>b)a=n-a+1,b=n-b+1;
  p[a]=1;
  for(int i=0;i<k;++i){
    for(int j=1;j<b;++j)q[j]=M-p[j],(p[j]+=p[j-1])%=M;
    for(int j=1;j<b;++j)(q[j]+=p[(j+b-1)/2])%=M;
    int* w=p;p=q,q=w;
  }
  int s=0;
  for(int i=1;i<b;++i)(s+=p[i])%=M;
  printf("%d\n",s);
}
