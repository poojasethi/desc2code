#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int n,a,b,k,f[50010],p[50010],i,j;
int main()
{
    scanf("%d%d%d%d",&n,&a,&b,&k);f[a]=1;
    for(i=1;i<=k;i++)
    {
        for(j=1;j<=n;j++)p[j]=(p[j-1]+f[j])%mod;
        for(j=1;j<=n;j++)
        {
            if(j<b)f[j]=(p[(j+b-1)/2]-f[j]+mod)%mod;
            if(j>b)f[j]=((p[n]-p[(j+b)/2]+mod)%mod-f[j]+mod)%mod;
        }
    }for(j=1;j<=n;j++)p[j]=(p[j-1]+f[j])%mod;
    printf("%d",p[n]);
}
