#include <bits/stdc++.h>
#define rep(i,j,k)	for (int i = j; i <= k; i++)
#define dep(i,j,k)	for (int i = j; i >= k; i--)
using namespace std;
typedef long long ll;
const int N = 5010, mod = 1e9+7;
int n, k, a, b, ans, f[N], sum[N];
int main()
{
	cin >> n >> a >> b >> k;
	f[a] = 1;
	rep(i,1,k)
	{
		rep(j,1,n)	sum[j] = (sum[j-1]+f[j])%mod;
		rep(j,1,n)
		{
			if (j < b)	f[j] = (sum[(j+b-1)>>1]-f[j]+mod)%mod;
			if (j > b)	f[j] = ((sum[n]-sum[(j+b)>>1]+mod)%mod-f[j]+mod)%mod;
		}
	}
	rep(i,1,n)	ans = (ans+f[i])%mod;
	cout << ans << endl;
}
