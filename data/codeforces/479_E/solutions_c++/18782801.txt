#include <iostream>
#include <cstdio>
const int mo=1e9+7;
int f[5010][5010],i,j,n,a,b,k;
int main()
{
	scanf("%d%d%d%d\n",&n,&a,&b,&k);
	for (i=a;i<=n;++i) f[0][i]=1;
	for (i=1;i<=k;++i)
	{
		for (j=1;j<b;++j)
			f[i][j]=(f[i-1][j-1]+f[i-1][(j+b-1)/2]-f[i-1][j])%mo;
		for (j=n;j>b;--j)
			f[i][j]=(f[i-1][n]-f[i-1][j]+f[i-1][j-1]-f[i-1][(j+b)/2])%mo;
		for (j=2;j<=n;++j) (f[i][j]+=f[i][j-1])%=mo,f[i][j]=(f[i][j]+mo)%mo;
	}
	printf("%d\n",(f[k][n]+mo)%mo);
}
