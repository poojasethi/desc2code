#include<stdio.h>
#include<math.h>
#define max(x,y) x>y?x:y
#define module 1000000007 
int dp[5005][5005];
int sum[5005][5005];
int main(void){
	int n,a,b,k;
	scanf("%d%d%d%d",&n,&a,&b,&k);
	if(a>b){
		a=n+1-a;
		b=n+1-b;
	}
	for(int i=1;i<b;i++){
		dp[0][i]=1;
		sum[0][i]=i;
	}
	for(int i=1;i<=k;i++){
		for(int j=1;j<b;j++){
			int t=max(0,(2*j-b));
			dp[i][j]=((sum[i-1][b-1]-sum[i-1][t])%module-dp[i-1][j])%module;
			sum[i][j]=(sum[i][j-1]+(dp[i][j]%module))%module;		
		}
	}
	if(dp[k][a]<0) dp[k][a]+=module;
	printf("%d\n",dp[k][a]);
	return 0;
}
