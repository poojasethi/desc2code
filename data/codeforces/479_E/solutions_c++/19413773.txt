#include<iostream>
#include<cmath>
using namespace std;
long long dp[5010],s[5010];
const long long mod = 1000000007;
int abs(int x){
	return x>0?x:-x;
}
int min(int a,int b){
	return a<b?a:b;
}
int max(int a,int b){
	return a>b?a:b;
}
int main(){
	int n,a,b,k;
	cin>>n>>a>>b>>k;
	
	for(int i=0;i<=n;i++)
	dp[i]=i; 
	for(int j=1;j<=k;j++)
	{
		for(int i=1;i<=n;i++)
		s[i]=((dp[i-1]-dp[max(1,i-(abs(b-i)-1))-1])%mod+(dp[min(n,i+abs(b-i)-1)]-dp[i])%mod)%mod;
		for(int i=1;i<=n;i++) dp[i]=dp[i-1]+s[i];
	}
	cout<<s[a]<<endl; 
}
