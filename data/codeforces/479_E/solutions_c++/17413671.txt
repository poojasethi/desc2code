#include <cstdio>

const int p=1000000007;
int f[5250][5250],s[5250][5250];
int i,j,m,n,x,y;

int main()
{
	scanf("%d%d%d%d",&m,&x,&y,&n);
	f[0][x]=1;
	for (j=x;j<=m;j++)
		s[0][j]=1;
	for (i=1;i<=n;i++)
		for (j=1;j<=m;j++)
		{
			if (j<y)
				f[i][j]=(s[i-1][(j+y-1)/2]+(p-f[i-1][j])%p)%p;
			if (j>y)
				f[i][j]=(s[i-1][m]+((p-s[i-1][(j+y)/2])%p+(p-f[i-1][j])%p)%p)%p;
			s[i][j]=(s[i][j-1]+f[i][j])%p;
		}
	printf("%d",s[n][m]);
	return 0;
}
