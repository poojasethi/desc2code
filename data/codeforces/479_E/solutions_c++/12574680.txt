#include <cstdio>
#include <algorithm>
using namespace std;

const int maxn = 5010;
const long long mod = 1e9+7;
long long s[maxn], d[maxn];

int main() {
	int n, a, b, k, lb, ub;
	scanf("%d%d%d%d", &n, &a, &b, &k);
	for(int j=1; j <= n; j++) d[j]=1, s[j]=j;
	for(int i=1; i <= k; i++) {
		for(int j=1; j <= n; j++) {
			if(j>b) lb=b, ub=min(n, 2*j-b-1);
			if(j<b) lb=max(0, 2*j-b), ub=b-1;
			d[j]=(s[ub]-s[lb]-d[j]+2*mod)%mod;
		}
		s[1]=d[1];
		for(int j=2; j <= n; j++) s[j]=(s[j-1]+d[j])%mod;	 
	}
	printf("%I64d\n", d[a]);
	return 0;
}
