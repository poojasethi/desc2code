#include <iostream>
#define f(i,a,b) for(int i=a;i<=b;i++)
long long N,A,B,K,DP[5005][5005],r,S[5005][2],M=1000000007,X;
main()
{
    std::cin>>N>>A>>B>>K;
    if(A>B)A=1+N-A,B=N-B+1;
    DP[A][0]=1;
    f(i,A,N)S[i][0]=1;
    f(j,1,K)f(i,1,B-1)
    {
        DP[i][j]=(S[(i+B-1)/2][1-j%2]-DP[i][j-1]+M)%M;
        S[i][j%2]=(S[i-1][j%2]+DP[i][j])%M;
    }
    f(i,1,N)X=(X+DP[i][K])%M;
    std::cout<<X;
}
