#include<bits/stdc++.h>

using namespace std;

#define lint long long 

#define MOD 1000000007

lint dp[5005][5005];
lint n,a,b,k,i,j,x,y;
lint top[5005];

int main(){

	cin >> n >> a >> b >> k;

	for(i=1 ; i<=n ; i++)
		top[i] = i;

	for(i=1 ; i<=n ; i++)
		dp[i][0] = 1;

	for(j=1 ; j<=k ; j++){

		for(i=1 ; i<=n ; i++){
			y = abs(i-b)-1;
			dp[i][j] = top[min(i+y,n)] - top[max(i-y,1LL)-1] - dp[i][j-1];
			dp[i][j] %= MOD;
		}
		
		memset(top,0,sizeof top);

		for(i=1 ; i<=n ; i++){
			top[i] = top[i-1] + dp[i][j];
			top[i] %= MOD;
		}
	}

	cout << (dp[a][k]+MOD)%MOD << endl;
}
