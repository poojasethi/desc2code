#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
vector<vector<int> > u(5050,vector<int>(5050,0));
vector<int> v(5040,0);
int n, a, b, k;
void update(int i)
{
	for(int x = 1; x < b; ++x) v[x] =( v[x-1] + u[i][x] ) %mod;
}
int main()
{
	cin >> n >> a >> b >> k;
	if(a>b)
	{
		 a = n - a + 1;
		 b = n - b + 1;
	}
	u[0][a] = 1;
	update(0);
	for(int x = 1; x <= k; ++x)
	{
		for(int y = 1; y < b; ++y)
		{
			u[x][y] = (v[(b-1-y)/2+y] - u[x-1][y] + mod) %mod;
		} update(x);
	}
	cout << v[b-1];
	return 0;
}
