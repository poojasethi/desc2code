#include<iostream>
#include<cstdio>
#include<cstring>
#define mod 1000000007
#define N 5500
using namespace std;
int s[N][N],f[N][N]; 
int n,a,b,K;
int main()
{
	scanf("%d%d%d%d",&n,&a,&b,&K);
	if(a>b){a=n-a+1;b=n-b+1;}
	f[a][0]=1;
	for(int i=a;i<=n;i++)s[i][0]=1;
	for(int j=1;j<=K;j++)
	{
		for(int i=1;i<=n;i++)
		{
			if(i<b)f[i][j]=(f[i][j]+(s[(i+b-1)/2][j-1]-f[i][j-1]+mod)%mod)%mod;
			if(i>b)f[i][j]=(f[i][j]+((s[n][j-1]-s[(i+b)/2][j-1])%mod)-f[i][j-1]+mod)%mod;
			s[i][j]=(s[i-1][j]+f[i][j])%mod;
		}
	}
	printf("%d",s[n][K]);
	return 0;
}
