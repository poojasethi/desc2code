from sys import stdin

a, b, c = [int(x) for x in stdin.readline().split()]
time = 0
while a > 0 or b > 0 or c > 0:
    if a > 2 or b > 2 or c > 2:
        time += 3
    elif c > 0:
        time += 3
    elif b > 0:
        time += 2
    elif a > 0:
        time += 1
    a -= 2; b -= 2; c -= 2

print time + 29
