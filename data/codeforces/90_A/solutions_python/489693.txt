#! /usr/bin/env python

from sys import stdin

from collections import deque

def solve(start_time,Q):
	ret=0
	while Q>0:
		#print start_time,ret,Q
		ret=max(start_time+30,ret)
		Q=max(0,Q-2)
		start_time+=3
	#print "ret=",ret
	return ret

if __name__=='__main__':
	r,g,b=map(int,stdin.readline().split())
	print max([solve(x,y) for (x,y) in ((0,r),(1,g),(2,b))])
