import sys
import itertools
import math

def is_empty(nums):
    for i in nums:
        if i > 0:
            return False
    return True
        
lines = [i.rstrip() for i in sys.stdin.readlines()]
nums = [int(i) for i in lines[0].split(" ")]
c = 0

while (not is_empty(nums)):
    for i in range(len(nums)):
        if (is_empty(nums)):
            break
        nums[i] = nums[i] - 2
        c += 1
print c - 1 + 30
