s=raw_input();
L=s.split(" ");
a=int(L[0]);
b=int(L[1]);
c=int(L[2]);
ans=29;
t=0;
while ((a>0) or (b>0) or (c>0)) :
    if (t==0) :
        a-=2;
        ans+=1;
    elif (t==1) :
        b-=2;
        ans+=1;
    else :
        c-=2;
        ans+=1;
    t=(t+1);
    if (t==3) :
        t=0;
print ans,"\n";
