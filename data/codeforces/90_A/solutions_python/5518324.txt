r, g, b = map(int, raw_input().strip().split())

rmin = 0
gmin = 1
bmin = 2

if r > 2 and r%2 == 0:
    rmin = (r/2-1)*3
elif r > 2 and r%2 != 0:
    rmin = r/2*3
if g > 2 and g%2 == 0:
    gmin = 1 + (g/2-1)*3
elif g > 2 and g%2 != 0:
    gmin = 1 + g/2*3
elif g == 0:
    gmin = 0
if b > 2 and b%2 == 0:
    bmin = 2 + (b/2-1)*3
elif b > 2 and b%2 != 0:
    bmin = 2 + b/2*3
elif b == 0:
    bmin = 0
#print rmin,gmin,bmin
print max(rmin,gmin,bmin)+30
