#include <iostream>
#include <cstdio>
#include <map>
using namespace std;

int n, k, a[400010];
long long ans = 0;
map<int , int> pp;

int main()
{
    scanf("%d%d",&n,&k);
    for(int i = 0 ; i < n ; i++)
        scanf("%d",&a[i]);
    int l = 0 , r = 1;
    for(; r <= n ; r++)
    {
        pp[a[r-1]]++;
        for(; l < r && k <= pp[a[r-1]] ; l++)
        {
            pp[a[l]]--;
            ans += n-r+1;
        }
    }
    printf("%I64d\n",ans);
    return 0;
}
