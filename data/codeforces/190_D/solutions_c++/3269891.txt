#include <iostream>
#include <map>

using namespace std;

int arr[400001]={};
map<int, int> cnt;
int main()
{
	int n, k, front = 0 ;

	long long ans = 0;

	cin>> n>> k;

	for(int i = 0 ; i < n ; ++i)
	{
		cin>> arr[i];
		cnt[arr[i]]++;
		while(cnt[arr[i]] >= k )
		{
			cnt[arr[front]]--;
			front++;

			ans += n-i;
		}
	}

	cout<<ans<<endl;
}
