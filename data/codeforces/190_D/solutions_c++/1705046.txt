#include<iostream>
#include<cstdio>
#include<map>
#define maxn 400010
#define rep(i,n) for(int i=0;i<n;++i)
using namespace std;

map<int,int> m;
int data[maxn],n,k;
long long ans=0;

int main()
{
	scanf("%d%d",&n,&k);
	rep(i,n) scanf("%d",&data[i]);
	int tail=0;
	rep(head,n)
	{
		int num=++m[data[head]];
		if(num<k) continue;
		while(1)
		{
			ans+=n-head;
			--m[data[tail]];
			if(data[tail++]==data[head]) break;
		}
	}
	cout<<ans<<endl;
	return 0;
}
