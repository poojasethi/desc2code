#include <bits/stdc++.h>
using namespace std;
int main() {
	int n,k;
	cin >> n >> k;
	int a[n];
	map <int,int> m1;
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}
	long long int ans = 0;
	int l = 0;
	for (int i = 0; i < n; i++) {
		m1[a[i]]++;
		if (m1[a[i]] >= k) {
			ans += (n-i);
			m1[a[l]]--;
			l++;
		}
		while (m1[a[i]] >= k && l <= i) {
			ans += (n-i);
			m1[a[l]]--;
			l++;
		}
	}
	cout << ans << endl;
	return 0;
}
