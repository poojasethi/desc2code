#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAXN=400000;
int a[MAXN+5];
map<int,int>vis;
int main(){
    int n,k;
    cin>>n>>k;
    for(int i=0;i<n;i++)cin>>a[i];


    int s=0;
    ll res=0;
    for(int i=0;i<n;i++){
        vis[a[i]]++;
        while(vis[a[i]]==k){
            res+=(ll)n-i;
            vis[a[s]]--;
            s++;
        }
    }
    cout<<res<<"\n";
}
