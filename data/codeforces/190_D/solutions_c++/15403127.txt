#include <bits/stdc++.h>

using namespace std;
int a[500001];
int main()
{
    map <int,int>m;
    int n,k;
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++)
    {
     scanf("%d",&a[i]);
     m[a[i]]=0;
    }int i=0,j=0;
  long long int sum=0;
    while(i<n&&j<n)
    {

        m[a[j]]+=1;
        while(m[a[j]]>=k)
        {
            sum+=(n-j);
            m[a[i]]-=1;
            ++i;
        }
        ++j;

    }
   printf("%I64d\n",sum);
    return 0;
}
