#include<iostream>
#include<cstdio>
#include<string.h>
#include<string>
#include<math.h>
#include<algorithm>
#include<map>
using namespace std;

map<int,int> Map;
int a[500000];
int n,k;

int main(){
	cin >> n >> k;
	int l = 0,r = 0;
	long long ans = 0;
	for(int i=0;i<n;i++){
		cin >> a[i];
		Map[a[i]]++;
		for(;l<=i&&Map[a[i]]>=k;l++){
			ans += n-i;
			Map[a[l]]--;
		}
	}
	cout << ans << endl;
	return 0;
}
