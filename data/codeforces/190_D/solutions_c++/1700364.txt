#include<cstdio>
#include<algorithm>
#include<iostream>
#include <map>
using namespace std;
int n, m;
map<int, int> cnt;
int a[1000000];
long long res;
int main()
{
    ios::sync_with_stdio(false);
    cin >> n >> m;
    for(int i  = 0, j = 0; i < n; ++i)
    {
        cin >> a[i];
        if(++cnt[a[i]] < m) continue;
        //--cnt[a[i]];
        do{
            res += n - i;
            --cnt[a[j]];
        }while(a[j++] != a[i]);
    }
    cout << res << endl;
    return 0;
}
