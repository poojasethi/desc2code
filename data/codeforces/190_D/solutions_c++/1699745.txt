#include <iostream>
#include <stdio.h>
#include <string.h>
#include <map>
#define ll __int64

using namespace std;

int a[400005];
map<int, int> mp;

int main() {
    int n, k, i, j;
    scanf("%d%d", &n, &k);
    for (i = 0; i < n; i++) scanf("%d", &a[i]);
    ll res = 0;
    j = 0;
    for (i = 0; i < n; i++) {
        mp[a[i]]++;
        for (; j <= i && k <= mp[a[i]]; j++)
            mp[a[j]]--, res += n - i;
    }
    printf ("%I64d\n", res);
    return 0;
}
