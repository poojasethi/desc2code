def work():
        mp = {}
        ret = 0
        tail = 0
        for i in xrange(n):
                if a[i] not in mp:
                        mp[a[i]] = 1
                else:
                        mp[a[i]] += 1
                while tail <= i and mp[a[i]] >= k:
                        ret += n - i
                        mp[a[tail]] -= 1
                        tail += 1
        return ret

n, k = map(int, raw_input(' ').split())
a = map(int, raw_input(' ').split())
print work()
