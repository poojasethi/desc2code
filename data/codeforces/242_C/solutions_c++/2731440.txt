#include <cstdio>
#include <vector>
#include <map>
#include <queue>
#include <iostream>
using namespace std;
#define pii pair<int,int>
#define x first
#define y second
#define mp make_pair

int n,i,j, r, a, b;
map<pii, int> m;
pii p0,p1;
int dx[8] = {1,1,1,0,0,-1,-1,-1};
int dy[8] = {-1,0,1,-1,1,-1,0,1};

int main()
{
	#ifndef ONLINE_JUDGE
	freopen("a.in","r",stdin);
	#endif

	cin>>p0.x>>p0.y>>p1.x>>p1.y>>n;
	for(i=0; i<n; i++)
	{
		cin>>r>>a>>b;
		for(j=a; j<=b; j++)
			m[mp(r,j)] = -1;
	}
	m[p0] = 0;
	queue<pii> q;
	q.push(p0);
	while(!q.empty())
	{
		pii now = q.front();
		q.pop();
		for(i=0;i<8;i++)
		{
			pii j = mp(now.x + dx[i], now.y + dy[i]);
			if (m[j] == -1)
			{
				m[j] = m[now] + 1; q.push(j);
			}
		}
	}
	cout<<m[p1];


}
