#include<bits/stdc++.h>
using namespace std;
long long x,y,fx,fy,n,a,b,c;
int adx[8]={0,1,1,1,0,-1,-1,-1},ady[8]={1,1,0,-1,-1,-1,0,1};
map<pair<long long,long long> ,int> arr;
queue<pair<long long,long long> >q;
pair<long long,long long> s,d;
int main(){
    cin>>x>>y>>fx>>fy>>n;
    for(int i=0;i<n;i++){
        cin>>c>>a>>b;
        for(int j=a;j<=b;j++)arr[make_pair(c,j)]=1;
    }
    q.push(make_pair(x,y));
    arr[make_pair(x,y)]=2;
    while(q.empty()==0){
        s=q.front();
        q.pop();
        for(int i=0;i<8;i++){
            d=make_pair(adx[i]+s.first,ady[i]+s.second);
            if(arr[d]==1){
                arr[d]=arr[s]+1;
                q.push(d);
            }
        }
    }
    cout<<arr[make_pair(fx,fy)]-2<<endl;
}
