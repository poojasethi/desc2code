#/usr/bin/env python2

w,m,k = tuple(int(x) for x in raw_input().split(" "))

w = int(w/k)

sm = len(str(m))

sm1=1
for i in range(sm):
	sm1*=10
sm1 = sm1 -1

r = 0

while True:
	count = sm1 - m + 1
	if w >= count * sm:
		# print (m,sm,sm1,r,w)
		w -= count * sm
		m = sm1 + 1
		sm1 = m*10 - 1
		sm+=1
		r+=count
	else:
		count = int(w / sm)
		r += count
		break

print r
