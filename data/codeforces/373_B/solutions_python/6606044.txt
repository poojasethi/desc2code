w, m, k = map(int, raw_input().split())
count = 0
level = len(str(m))


while w > 0:
    upper = 10 ** level
    remain = upper - m
    cost = remain * level * k
    if w >= cost:
        w -= cost
        m = upper
        level += 1
        count += remain
    else:
        count += w // (k * level)
        break

print count
