import math
a = raw_input()
input = []
input = a.split()
w = int(input[0])
k = int(input[2])
mylen = len(input[1])
m = int(input[1])
count = int(0)
while True:
    m1 = int(math.pow(10,mylen))
    if ((m1-m)*mylen*k <= w) :
        w -= (m1-m)*mylen*k
        count = count+m1-m
        m = m1
    else:
        t = int(w/(k*mylen))
        count = count + t
        break
    mylen = mylen+1
print count
