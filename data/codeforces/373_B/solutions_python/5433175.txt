a = map(int, raw_input().split())
m = a[0]
n = a[1]
k = a[2]
i=n;
len=0;
while(i>0):
    len=len+1;
    i/=10;
n=n-1;
j=1;
i=1;
while(i<=len):
    j=j*10;
    i=i+1;
j=j-1;
ans=0;
cost=k*len;
while(m>0):
    if(j-n<m/cost):
        ans=ans+j-n;
    else:
        ans=ans+m/cost;
    m=m-(j-n)*cost;
    n=j;
    j=((j+1)*10)-1;
    cost=cost+k;
print(ans);
