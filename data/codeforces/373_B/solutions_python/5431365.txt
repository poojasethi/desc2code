w, m, k = map(int, raw_input().split())

l = 0
cur_l = len(str(m))
next = (10 ** (cur_l))

cur = next - m
cost = cur_l * k
while w > 0:
    if w - (cur*cost) > 0:
        w -= (cur*cost)
        l += cur
        cur = 9 * (10**cur_l)
        cur_l += 1
        cost = cur_l * k
    else:
        l += w/cost
        break

print l
