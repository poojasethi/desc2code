#!/usr/bin/env python
# encoding: utf-8

line = raw_input();
ll = line.split();
w = long(ll[0])
m = long(ll[1])
k = long(ll[2])

#print "%d %d %d" % (w,m,k)
kk = w / k;

temp = 1L
cnt = 0L
ans = 0L

while(temp <= m):
    temp *= 10
    cnt += 1

cost = (temp - m) * cnt;
#print '%d %d' % (kk, cost)

while(kk >= cost):
    kk -= cost
    ans += (temp -m)
#    print "temp value %d" % ans
    m = temp
    temp *= 10
    cnt += 1
    cost = (temp - m) * cnt

if(cnt != 0):
    ans += kk / cnt

print ans
