import sys

cost, offset, coef = map(int, sys.stdin.readline().split())

cost /= coef

length = 0

while True:
    n = 1
    numDigits = 0
    while n <= offset:
        n *= 10
        numDigits += 1
    if (n - offset) * numDigits <= cost:
        length += n - offset
        cost -= (n - offset) * numDigits
        offset = n
    else:
        length += cost / numDigits
        print length
        exit()
