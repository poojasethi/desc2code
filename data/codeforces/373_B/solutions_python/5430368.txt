import fileinput

lines = []
for line in fileinput.input():
	lines.append(line.strip())

# lines = ['1 1 2']

w, m, k = map(lambda x: int(x), lines[0].split())

def get_bucket_size(n):
	tmp_n = n
	ds = 0
	while tmp_n > 0:
		ds += 1
		tmp_n /= 10

	return 10**ds - n, ds

max_len = 0
while w > 0:
	n, d = get_bucket_size(m)
	if w < n*d*k:
		max_len += w/(d*k)
		break
	else:
		max_len += n
		w -= n*d*k
		m = 10**d

print max_len
