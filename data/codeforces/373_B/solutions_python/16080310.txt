n, m, k = map(int, raw_input().split())

n /= k

base = 9

s = []
for i in xrange(17):
    s.append((i + 1) * base)
    base *= 10

base = 10

for i in xrange(17):
    if m >= base:
        n += s[i]
    else:
        n += (m - base / 10) * (i + 1)
        break
    base *= 10

base = 9
a = 0

for i in xrange(17):
    if n >= s[i]:
        a += base
        n -= s[i]
    else:
        a += n / (i+1)
        break
    base *= 10

print a-m+1
