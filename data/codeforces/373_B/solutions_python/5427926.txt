def val(n,k) :
   pow10 = 1
   cnt = 0
   ans = 0
   len = 1
   while(pow10 * 10 <= n) :
      cnt = pow10 * 9
      ans += len * cnt * k;
      pow10 *= 10
      len += 1

   m = n - pow10 + 1
   ans += m * k * len
   return ans;


w,m,k = map(int,raw_input().split())

lo = 0
hi = 10**18
mcnt = val(m-1,k)

while (hi - lo > 1):
   mid = (lo + hi)/2
   if(val(mid,k) <= mcnt + w):
      lo = mid
   else:
      hi = mid
 
if(lo < m): print 0
else: print (lo - m + 1)
