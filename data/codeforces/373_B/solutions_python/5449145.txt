w,m,k = map(int, raw_input().split())

answer = 0
while w > 0:
    length = len(str(m))

    attempts = (10**length - m)
    tmp_weight = attempts*k*length

    if tmp_weight <= w:
        w -= tmp_weight
        answer += attempts
        m = 10**length
    else:
        answer += w/(length*k)
        w = 0


print answer
