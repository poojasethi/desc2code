import math
digits = lambda x : int(math.log10(x)) + 1
w, m, k = map(int, raw_input().split())
w /= k
ans = 0
while w >= digits(m):
  d = digits(m)
  next_pow = 10 ** d
  if w >= (next_pow - m) * d:
    w -= (next_pow - m) * d
    ans += next_pow - m
    m = next_pow
  else:
    left = w / d
    w -= left * d
    m += left
    ans += left
print ans
