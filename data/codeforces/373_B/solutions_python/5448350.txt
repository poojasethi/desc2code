def dcount(n):
    cnt=0;
    while n>0:
        n=int(n/10)
        cnt+=1
    return cnt


w,m,k=raw_input().split()
w=int(w)
m=int(m)
k=int(k)
d=dcount(m)
ten=1
for i in range(d):
    ten*=10

cnt=0
while w>(ten-m)*d*k:
    cnt+=ten-m
    w-=(ten-m)*d*k
    d+=1
    m=ten
    ten*=10

cnt+=w/(d*k)
print cnt
