from sys import stdin

def f(x):
    i = ret = 0
    while 10 ** (i + 1) <= x:
        ret += (i + 1) * (10 ** (i + 1) - 10 ** i)
        i += 1
    return ret + (i + 1) * (x - 10 ** i + 1) + 1

w, m, k = [int(x) for x in stdin.readlines()[0].split()]
w //= k

low = m - 1
up = 2 * 10 ** 16
while low < up:
    mid = (low + up) // 2 + 1
    if f(mid) - f(m - 1) <= w:
        low = mid
    else:
        up = mid - 1
print low - (m - 1)
