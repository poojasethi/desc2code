def main():
    w, m, k = map(int, raw_input().split())
    count = 0
    while w > 0:
        bits = len(str(m))
        number = ((10**bits)-1-m+1)
        if number * bits * k <= w:
            w -= number * bits * k
            count += number
            m = 10**bits
        else:
            count += w / (len(str(m))*k)
            break
    print count

if __name__ == '__main__':
    main()
