s=raw_input().split()
w = long(s[0])
m=long(s[1])
k=long(s[2])

x=m
d=0
while x!=0:
        d+=1
        x/=10
c=0
diff = 10**d - m
ans=0
while c<=w:
        if w>= c+diff*d*k:
                ans+=diff
                c+=diff*d*k
        else:
                t=(w-c)/(d*k)
                ans+=t
                c+=t*d*k
                break
        d+=1
        diff = 10**d - 10**(d-1)

print ans
