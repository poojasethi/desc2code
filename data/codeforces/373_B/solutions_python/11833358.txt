from sys import exit
line = raw_input()
#line = '10000000000000000 9999999999999999 1'
w, m, k = map(int, line.split())
l = len(str(m))
p = k * l
b = 10**(l-1)
n = 10**l-m
ans = 0

tmp = n*p
if w < tmp:
	tmp = w/p
	print tmp
	exit(0)
	
ans += n
w -= tmp
l += 1
p += k
b *= 10

while True:
	n = b * 9
	tmp = n * p
	if w < tmp:
		ans += w / p
		break
	ans += n
	w -= tmp
	l += 1
	p += k
	b *= 10
	
print ans
