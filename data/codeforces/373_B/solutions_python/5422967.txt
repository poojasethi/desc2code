w, m, k = map(int, raw_input().split())
ans = 0
while w > 0:
    n = len(str(m))
    t = 0
    for i in xrange(n):
        t = t * 10 + 9

    t1 = t - m + 1
    if t1 * k * n > w:
        ans += w / (k * n)
        break
    else:
        w -= t1 * k * n
        ans += t1
    m = t + 1
print ans
