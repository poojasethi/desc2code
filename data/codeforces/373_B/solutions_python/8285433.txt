w,m,k=map(int,raw_input().split())
ans=0
chk=m
while 1:
    dig=len(str(chk))
    digb=int('9'*dig)
    digs=max(int('1'+'0'*(dig-1)),chk)
    nk=digb-digs+1
    abl=w-ans
    if abl>(dig*k)*nk:
        ans+=(dig*k)*nk
        chk=digb+1
    elif abl==(dig*k)*nk:
        chk=digb+1
        break
    else:
        digs+=abl/(dig*k)
        chk=digs
        break
print chk-m
