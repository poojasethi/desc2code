import math
w, m, k = map(int, raw_input().split())

ten = [10]
for i in xrange(2,18):
    ten.append(ten[-1]*10)
    
ans = 0
for i in xrange(0,len(ten)):
    if ten[i]>m:
        if (i+1)*k*(ten[i]-m)<=w:
            w-= ((i+1)*k*(ten[i]-m))
            ans+=(ten[i]-m)
            m=ten[i]
            #print ans, w
        else:
            ans += math.floor(w*1.0/((i+1)*k))
            break
print int(ans)
