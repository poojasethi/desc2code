def dig(n):
    z = 0;
    while n > 0:
        z += 1
        n = n / 10
    return z
line = raw_input().split()
w = int(line[0])
m = int(line[1])
k = int(line[2])
d = dig(m)
val = (10 ** d - m) * d * k
while w >= val:
    w = w - val
    d += 1
    val = 9 * 10 ** (d - 1) * d * k
num = max(m - 1,10 ** (d - 1) - 1) + w / (d * k)
ans = num - m + 1
print ans
