I=lambda:map(int, raw_input().split())
w, m, k = I()
i = 0
while m >= 10 ** i: i += 1
cost, cnt = 0, 0
while cost <= w:
    if w - cost > (10**i - m) * i * k: 
        m, i, cnt, cost = 10**i, i+1, cnt + 10**i-m, (10**i - m) * i * k + cost
    else: 
        cnt += (w - cost) // (i * k); 
        break
print cnt
