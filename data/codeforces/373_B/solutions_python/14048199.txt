w,m,k = map(int,raw_input().split())
def solve(w,m,k):
    w /= k
    d = len(str(m))
    if d > w:
        return 0
    
    a,b=m,10**d
    n=0
    while w>0:
        if w >= d*(b-a):
            w -= d*(b-a)
            n += b-a
            d += 1
            a,b = b,10**d
        else:
            n += w/d
            w = 0
    return (n)

print solve(w,m,k)
