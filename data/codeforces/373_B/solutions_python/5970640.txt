w, m, k = map(int, raw_input().split())

# find how many more n s.t. len(n) = len(m)
i, d, l = 1, 10, 0
while d < m:
	d *= 10
	i += 1

# add as many as can of that length
c = min(w / (i * k), d - m)
l, w, i = l + c, w - (c * i * k), i + 1

#increment length and try again
while w >= i * k:
	c = min(w / (i * k), 10 * d - d)
	l, w = l + c, w - (c * i * k)
	d, i = 10 * d, i + 1
print l
