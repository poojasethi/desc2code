#!/usr/bin/env python

l = map(int, raw_input().split())

w = l[0];m = l[1];k = l[2]

c = 0
while w>0:
	ln = len(str(m))
	mx = '9'*ln
	d = int(mx) -  int(m) + 1
	d = d*k*ln
	# print d,w
	if d <= w:
		c += int(mx) - int(m) + 1
		m = int(mx) + 1
		w -= d
	else:
		c += w/(len(str(m))*k)
		break
	# print c,w,m

print c
