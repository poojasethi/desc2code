import math
def fun( n):

    i =0
    while ( n>0):
    
        n/=10;
        i+=1;
    
    return i;

p =0;
(w, m, k) = map(int, raw_input().split())
dig  = fun(m)

b = j =math.pow(10, dig);
j = j - m;
while (j*k*dig<=w ):

    p+=j;
    w -=dig*j*k;
    j = b*9;
    b*=10;
    dig+=1;

p+= w//(k*dig);

print int(p)
