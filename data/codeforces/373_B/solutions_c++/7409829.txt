#include<cstdio>
#define LL long long
LL a[20]={1};

int main()
{
	LL w,m,k;
	scanf("%I64d%I64d%I64d",&w,&m,&k);
	int i;
	for(i=1;i<=17;++i)a[i]=a[i-1]*10;
	for(i=1;i<=17;++i)if(m<a[i])break;
	LL n=w/k,ans=0;
	while(i<=17){
		if(n>=i*(a[i]-m)){
			n-=i*(a[i]-m);
			ans+=a[i]-m;
			m=a[i++];
		}
		else {ans+=n/i;break;}
	}
	printf("%I64d\n",ans);
	return 0;
}
