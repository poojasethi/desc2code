#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    unsigned long long w,m,k,i,ans=0,x=1,y=9,z=0;
    cin>>w>>m>>k;
    while(m>y){y=y*10+9;x++;}
    while(1)
    {
        z=y-m+1;
        if(z*x*k<=w){ans+=z;w-=(z*x*k);}
        else {ans+=(w/(x*k));break;}
        m=y+1;
        y=y*10+9;
        x++;
    }
    cout<<ans;
    return 0;
}
