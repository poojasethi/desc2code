#include<stdio.h>
__int64 wei(__int64 x){
	__int64 ans=0;
	while(x>0){
		ans++;
		x/=10;
	}
	return ans;
}
__int64 w,m,k,t,tt,i,tot,tmp;
int main(){
	scanf("%I64d%I64d%I64d",&w,&m,&k);
	t=wei(m);
	tt=1;
	for(i=1;i<=t;i++)tt=tt*10;
	tot=0;
	while(w>0){
		tmp=(tt-m)*t*k;
		if((tt-m)<=w/t/k){
			w=w-tmp;
			tot+=(tt-m);
			t++;
			m=tt;
			tt*=10;
		}
		else {
			tot+=(w/t/k);
			w=0;
		}
	}
	printf("%I64d\n",tot);
}
