#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    long long int w,m,k,len=0,t=1,y=0;
    cin >> w >> m >> k;
    w=w/k;
    while(m>t){
        len++;
        t=t*10;
    }
    while(w>=(t-m)*len){
        y+=(t-m);
        w-=(t-m)*len;
        m=t;
        t=t*10;
        len++;
    }
    y+=w/len;
    cout << y << "\n";
    return 0;
}
