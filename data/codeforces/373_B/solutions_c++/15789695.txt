#include<iostream>
using namespace std;

typedef long long ll;
ll w, m, k, ans, x, ten[19];

ll s(ll x){
  ll y = 0;
  while(x)
    y++, x /= 10;
  return y;
}

int main(){
  ten[0] = 1;
  for(int i = 1; i < 19; i++)
    ten[i] = ten[i - 1] * 10;
  cin >> w >> m >> k;
  ll i = s(m);
  while(w >= i * k)
    x = min(w / (i * k), ten[i] - m), w -= x * k * i, m = ten[i], i++, ans += x;
  cout << ans;
  return 0;
}
