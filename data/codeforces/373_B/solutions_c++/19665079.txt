#include <stdio.h>
int main()
{
    long long w,m,k,ans=0,x=1,y=0;
    scanf("%I64d%I64d%I64d",&w,&m,&k);
    w/=k;
    while(x<m)
    {
        x*=10;
        y++;
    }
    while((x-m)*y<=w)
    {
        ans+=x-m;
        w-=(x-m)*y;
        m=x;
        x*=10;
        y++;
    }
    ans+=w/y;
    printf("%I64d\n",ans);
    return 0;
}
