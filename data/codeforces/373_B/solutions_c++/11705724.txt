#include <bits/stdc++.h>

using namespace std;
long long w, m, k;
int main() {
	cin >> w >> m >> k;
	long long first, dec, s;
	first = m;
	dec = 10;
	s = 1;
	while(dec < m) {
		dec *= 10;
		s++;
	}
	w /= k;
	while(w) {
		if(w >= (dec-m)*s) {
			w -= (dec-m)*s;
			m = dec;
			dec *= 10;
			s++;
		}
		else {
			m += w/s;
			w = 0;
		}
	}
	cout << m - first << endl;
	return 0;
}
