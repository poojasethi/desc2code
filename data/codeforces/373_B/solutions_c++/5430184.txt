#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	long long int w,m,k,t1,t2,t3,ans=0;
	cin>>w>>m>>k;
	int len=0;
	t1=m;
	while(t1>0)
	{
		t1=t1/10;
		len++;
	}
	long long int lim=pow(10,len);
	int flag=1;
	while(flag)
	{
		t2=lim-m;
		t3=(w)/(len*k);
		if(t2<=t3)
		{
			ans+=t2;
			m=lim;
			lim=m*10;
			w = w-(t2*len*k);
			len++;
		}
		else
		{
			ans+=t3;
			break;
		}
	}
	cout<<ans<<"\n";
	return 0;
}
