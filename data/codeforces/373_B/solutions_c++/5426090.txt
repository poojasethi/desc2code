#include<algorithm>
#include<cmath>

using namespace std;

unsigned long long pow10(int x) {
	unsigned long long j=1;
	for(int i=1;i<=x;i++) j*=10;
	return j;
}

int main() {
	unsigned long long w,k,m;
	cin>>w>>m>>k;

	unsigned long long q=m,h=0;
	int t=log10(m);
	while((pow10(t+1)-m)*(t+1)<=w) {
		w-=(pow10(t+1)-m)*(t+1);
		m=pow10(t+1);
		t++;
	}
	m+=w/(t+1);
	cout<<m-q;
	

}
