#include<iostream>
using namespace std;
int main(){
    long long n,m,k,sum=1,r=0,res=0;
    cin>>n>>m>>k;
    n/=k;
    while(sum<m)sum*=10,r++;
    while((sum-m)*r<=n){
        res+=sum-m;
        n-=(sum-m)*r;
        m=sum;
        sum*=10;
        r++;
    }
    res+=n/r;
    cout<<res<<endl;
    return 0;
}
