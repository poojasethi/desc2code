#include<stdio.h>
int main()
{
    __int64 w,m,k,ans=0,i,j;
    scanf("%I64d%I64d%I64d",&w,&m,&k);
    w/=k;
    for(i=10,j=1;i<=100000000000000000;i*=10,j++)
    {
        if(w&&m<i)
        {
            if((j*(i-m))<=w)
                ans+=(i-m),w-=(j*(i-m)),m=i;
            else
                ans+=(w/j),w=0;
        }
    }
    printf("%I64d\n",ans);
    return 0;
}
