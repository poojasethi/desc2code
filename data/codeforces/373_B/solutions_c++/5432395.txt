# include <stdio.h>


long long f(long long w, long long m, long long k)
{
    long long mm = m, cc = 0, dc = 1;
    while (mm)
        dc *= 10, cc++, mm /= 10;
    if ((dc - m) * cc * k < w)
        return dc - m + f(w - (dc - m) * cc * k, dc, k);
    return w / (cc * k);
}


int main ()
{
    long long w, m, k;
    while (~scanf ("%I64d%I64d%I64d", &w, &m, &k))
        printf ("%I64d\n", f(w/k, m, 1));
    return 0;
}
