#include<iostream>
using namespace std;

int main()
{
    long long w,m,k,t=1,y=0,len=0;
    cin>>w>>m>>k;
    w/=k;
    while (m>t)
    {
          len++;
          t*=10;
    }
    while (w>=(t-m)*len)
    {
          y+=t-m;
          w-=(t-m)*len;
          m=t;
          t*=10;
          len++;
    }
    y+=w/len;
    cout<<y<<endl;
    
   //stem("pause");
    return 0;
}
