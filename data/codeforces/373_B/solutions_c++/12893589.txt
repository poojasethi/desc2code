//In the Name of God

#include<iostream>
#define int long long

using namespace std;

int ans,w,a,M,m,dm,k;

main()
{
  cin>>w>>m>>k;
  w/=k;
  M=m;
  a=1;
  while(M)
    M/=10,dm++,a*=10;
  //cout<<dm<<" "<<a<<endl;
  while(w>=dm)
    {
      ans+=min(w/dm,a-m);
      w-=min(w/dm,a-m)*dm;
      m=a;
      a*=10;
      dm++;
    }
  cout<<ans<<endl;
  return 0;
}
