#include<cstdio>
#include<cmath>
#include<algorithm>
#include<map>
#include<cstring>
#include<iostream>
#include<vector>
using namespace std;
typedef long long ll;
int main(){
	ll w,m,k,t=1,y=0;
	ll digit=0;
	cin>>w>>m>>k;
	w/=k;
	while(m>t){
		digit++;
		t*=10;
	}
	while(w>=(t-m)*digit){
		y+=t-m;
		w=w-(t-m)*digit;
		m=t;
		t*=10;
		digit++;
	}
	y+=w/digit;
	cout<<y<<endl;

}
