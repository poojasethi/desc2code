#include <iostream>

using namespace std;

int S(unsigned long long x)
{
	int num=0;

	while(x)
	{
		++num;
		x/=10;
	}

	return num;
}

int main()
{
	unsigned long long w,m,k;
	cin>>w>>m>>k;

	unsigned long long lim=1;
	while(m>=lim) lim*=10;

	unsigned long long len=0;

	while(w>=(lim-m)*S(m)*k)
	{
		w-=(lim-m)*S(m)*k;
		len+=lim-m;

		m=lim;
		lim*=10;
	}

	unsigned long long x=w/S(m)/k+m;
	len+=x-m;

	cout<<len;

	return 0;
}
