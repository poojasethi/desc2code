#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>

using namespace std;


int main()
{
    long long w,m,k,t=1,ans=0;
    cin>>w>>m>>k;
    w/=k;
    int len=0;
    while(t<m)len++,t*=10;
    while(w>=(t-m)*len)w-=(t-m)*len,ans+=t-m,m=t,t*=10,len++;
    ans+=w/len;
    cout<<ans<<endl;

}
