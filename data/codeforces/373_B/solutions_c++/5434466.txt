#include <iostream>

int main(void)
{
    long long w,m,k;
    std::cin>>w>>m>>k;
    long long res = 0;
    w/=k;
    long long tmp = 1;
    int Sm = 0;
    while(w > 0)
    {
        while(tmp <= m)
        {
            tmp*=10;
            Sm++;
        }
        long long price = std::min(w,(tmp-m)*Sm);
        w -= price;
        res += w==0? price/Sm : tmp-m;
        m = tmp;
    }
    std::cout<<res;
    return 0;
}
