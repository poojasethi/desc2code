#include <iostream>
using namespace std;

typedef long long ll;
ll w, m, k;

ll sum(ll a){
	ll pow = 1, res = 0;
	for(int i = 1; i<=17; i++){
		if(pow>a) break;
		res+=i*(min(a+1, 10*pow)-pow);
		pow*=10;
	}
	return res;
}

int main(){
	cin >> w >> m >> k;
	ll lo = m-1, hi = m+w/k+1, mid = (lo+hi)/2;
	while(lo<mid){
		if(sum(mid)-sum(m-1)>w/k) hi = mid;
		else lo = mid;
		mid = (lo+hi)/2;
	}
	cout << lo-m+1;
}
