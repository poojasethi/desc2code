#include<stdio.h>
#include<math.h>
#define ll long long
int main(){
	ll w,m,k;
	scanf("%I64d %I64d %I64d",&w,&m,&k);
	ll sn=0,x,count;
	ll temp=m;
	while(temp>0){
		sn++;
		temp/=10;
	}
	count=0;
	if(sn*k>w){
		count=0;
	}
	else{
		x=w/k;
		temp=pow(10,sn);
		while((temp-m)*sn<x){
			x-=(temp-m)*sn;
			count+=(temp-m);
			m=temp;
			temp*=10;
			sn++;
		}
		count+=x/sn;
			
	}
	printf("%I64d\n",count);
	return 0;
}
