#include <cstdio>
typedef long long LL;
LL w,m,k,ans(0);
LL f(LL x){
	LL t(0);
	while(x){t++;x/=10;}
	return t;
}
LL h(int x){
	LL t(1);
	for(int i=1;i<=x;++i)t*=10;
	return t;
}
int main(){
	scanf("%I64d%I64d%I64d",&w,&m,&k);
	w/=k;
    for(int i=f(m);;++i){
		if(w>=(LL)i*(h(i)-m)){
			ans+=h(i)-m;
			w-=(LL)i*(h(i)-m);
			m=h(i);
		}else{
			ans+=w/(LL)i;
			break;
		}
	}
	printf("%I64d\n",ans);
	return 0;
}
