#include <stdio.h>
__int64 func(__int64 R, __int64 B){
 if(B == (__int64)1e18) return R / 19;
 __int64 s = 1,r = 0,d;
 while(s <= B) s *= 10, r ++;
 d = s - B;
 if(d * r > R)
  return R / r;
 return d + func(R - d * r,s);
}
int main(){
 __int64 P,B,N,R,res;
 scanf("%I64d %I64d %I64d",&P,&B,&N);
 R = P / N;
 printf("%I64d",func(R,B));
 return 0;
}
