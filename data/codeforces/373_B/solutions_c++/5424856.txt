#include <stdio.h>


__int64 w, m, k, s, e[20], answer, i;



int main(){
    
    scanf("%I64d%I64d%I64d",&w,&m,&k);
    
    s = w / k;

    e[1]=1;
    for(i=2;i<20;i++) e[i]=e[i-1]*10;
    
    for(i=1;i<20;i++){
        if(e[i]<=m && m<e[i+1] && s){
            if( (e[i+1]-m)*i <= s) answer+=e[i+1]-m, s-=(e[i+1]-m)*i, m=e[i+1];
            else answer+=s/i;
        }
    }
    printf("%I64d",answer);
    
}
