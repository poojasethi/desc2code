#include <cstdio>

long long w,m,k,base,count;
int digit;

int main(){
    scanf("%I64d%I64d%I64d",&w,&m,&k);
    w /= k,base = 1,count = digit =0;
    
    while(w>0){
        //m's digit
        while(m>=base)
            base *= 10,digit++;
        
        if(w>=digit*(base-m)) w -= (digit*(base-m)),count += (base-m),m = base;
        else count += (w/digit),w = 0;
    }
    printf("%I64d\n",count);
    return 0;
}
