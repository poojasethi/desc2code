#include <bits/stdc++.h>
using namespace std;
int main(){
  long long int t = 1, y = 0;
  long long int l = 0,len=0,w,m,k;
  cin >> w >> m >> k;
  w/=k;
  while(m>t)
  {
    len++;
    t*=10;
  }
  while(w>=(t-m)*len)
  {
    y+=t-m;
    w-=(t-m)*len;
    m = t;
    t*=10;
    len++;
  }
  y+=w/len;
  cout<<y<<endl;
  return 0;
}
