//In the name of God
#include <iostream>
using namespace std;

long long n, m, w, k;
int S(long long x) { return x? 1 + S(x / 10): 0; }
long long ten(int x) { return x? 10ll * ten(x - 1): 1; }
int main() {
	cin >> w >> m >> k;
	w /= k, n = m;
	while (S(n) * (ten(S(n)) - n) <= w)
		w -= S(n) * (ten(S(n)) - n), n = ten(S(n));
	cout << n + w / S(n) - m << '\n';
	return 0;
}
