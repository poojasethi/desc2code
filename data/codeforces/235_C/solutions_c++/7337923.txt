#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;

const int MaxN = 1000010;

struct Node{
	int len,cnt,mark;
	Node *ch[26],*fa;
}pool[MaxN*2],*tail=pool,*op=pool,*ed=pool;
vector<Node *> list[MaxN];
int n,d[MaxN];
char s[MaxN];

void extend(int c,int len){
	Node *p=ed,*np=ed=++tail;
	np->len=len;np->cnt=1;
	for(;p && !p->ch[c];p=p->fa) p->ch[c]=np;
	if(!p) np->fa=op;
	else{
		if(p->ch[c]->len==p->len+1) np->fa=p->ch[c];
		else{
			Node *q=p->ch[c],*r=++tail;*r=*q;
			r->cnt=0;r->len=p->len+1;q->fa=np->fa=r;
			for(;p && p->ch[c]==q;p=p->fa) p->ch[c]=r;
		}
	}
}
int main()
{
	scanf("%s%d",s,&n);
	for(int i=0;s[i];i++) extend(s[i]-'a',i+1);
	for(Node *p=op;p<=tail;p++) list[p->len].push_back(p);
	for(int i=strlen(s);i;i--)
		for(int j=0;j<list[i].size();j++)
			list[i][j]->fa->cnt+=list[i][j]->cnt;
	for(int i=1;i<=n;i++){
		int ans=0,len=0,tmp=0;Node *p=op;
		scanf("%s",s);len=strlen(s);
		for(int j=0;j<len*2-1;j++){
			int c=s[j%len]-'a';
			for(;p && !p->ch[c];p=p->fa,tmp=p ? p->len : 0);
			p ? (tmp++,p=p->ch[c]) : (tmp=0,p=op);
			for(;p->fa && p->fa->len>=len;p=p->fa,tmp=p->len);
			if(tmp>=len && p->mark!=i) ans+=p->cnt,p->mark=i;
		}
		printf("%d\n",ans);
	}
	return 0;
}
