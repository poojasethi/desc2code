#include<cstdlib>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<iostream>
using namespace std;
struct suffix_auto{
	int l,son[27],fa,cnt,co;
} a[2100000];
int len,root,last,i,j,k,l,n,m,x,d[2100000],rank[2100000],o,p,ans,nl;
char c[2100000];
void build(){
	int np,p,q,r;
	root=last=++len;
	for(i=1;i<=l;++i){
		x=c[i]-96;
		np=++len;
		a[np].l=i; a[np].cnt=1;
		for(p=last;p && !a[p].son[x];p=a[p].fa)a[p].son[x]=np;
		last=np;
		if(p==0)a[np].fa=root;
		else{
			q=a[p].son[x];
			if(a[p].l+1==a[q].l)a[np].fa=q;
			else{
				r=++len;
				a[r].l=a[p].l+1;
				a[r].fa=a[q].fa;
				for(j=1;j<=26;++j)a[r].son[j]=a[q].son[j];
				a[np].fa=a[q].fa=r;
				for(;p&&a[p].son[x]==q;p=a[p].fa)a[p].son[x]=r;
			}
		}
	}
}
void dp(){
	for(i=1;i<=len;++i)d[a[i].l]++;
	for(i=1;i<=len;++i)d[i]+=d[i-1];
	for(i=1;i<=len;++i)rank[d[a[i].l]--]=i;
	for(i=len;i>=1;--i){
		x=rank[i];
		a[a[x].fa].cnt+=a[x].cnt;
	}
}
int main(){
	scanf("%s",c+1);
	l=strlen(c+1);
	build();
	dp();
/*	for(i=1;i<=len;++i)
		printf("%d %d\n",i,a[i].fa);
		//for(j=1;j<=26;++j)if(a[i].son[j])printf("%d %c->%d\n",i,j+96,a[i].son[j]);
	for(;;);*/
	for(scanf("%d",&n),o=1;o<=n;++o){
		scanf("%s",c+1);
		l=strlen(c+1);
		for(i=1;i<=l;++i)c[i+l]=c[i];
		p=root; ans=0;
		for(i=1;i<=l*2;++i){
			x=c[i]-96;
			for(; p && !a[p].son[x];p=a[p].fa,nl=a[p].l);
			if(a[p].son[x])p=a[p].son[x],nl++;
			if(i>=l){
				for(;p && a[a[p].fa].l>=l;p=a[p].fa,nl=a[p].l);
				if(nl>=l && a[p].co!=o){
					a[p].co=o;
					ans+=a[p].cnt;
				}
			}
		}
		printf("%d\n",ans);
	}
}
