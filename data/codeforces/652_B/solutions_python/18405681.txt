def zsort(lists):
    lists.sort()
    newlist=[]
    lens=len(lists)
    newlist=[0 for k in range(lens)]
    k=0
    p=0
    while k<lens:
        newlist[k]=lists[p]
        k+=2
        p+=1
    k=1
    while k<lens:
        newlist[k]=lists[p]
        p+=1
        k+=2
    return newlist
