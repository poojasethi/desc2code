from sys import stdin

n = int( stdin.readline() )
elements = sorted( map( int, stdin.readline().split() ) )

answer = [elements.pop(0)]

while len(elements) > 0:
    answer.append( elements.pop() )
    if len(elements) > 0:
        answer.append( elements.pop(0) )

isImposilble = False
for i in xrange(1, n ):
    if i % 2 == 1:
        if not (answer[i] >= answer[i-1]):
            isImposilble = True
            break
    else:
        if not (answer[i] <= answer[i-1]):
            isImposilble = True
            break

if isImposilble:
    print "Impossible"
