n = int(raw_input())
num = map(int,raw_input().split())
sort = []
sort.append(min(num))
num.remove(sort[0])
for i in xrange(1,len(num)+1):
	if i%2!=0:
		sort.append(max(num))
		num.remove(max(num))
	else:
		sort.append(min(num))
		num.remove(min(num))
find = True
for i in xrange(1,len(sort)):
	if ((i+1)%2==0 and sort[i]<sort[i-1]) or ((i+1)%2!=0 and sort[i]>sort[i-1]):
		find = False
if find:
	print ' '.join(str(x) for x in sort)
else:
	print "Impossible"
