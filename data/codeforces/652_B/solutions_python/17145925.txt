def getArr(f):
    return [f(x) for x in raw_input().strip().split(' ')]

def gi():
    return int(raw_input().strip())

n = gi()
A = getArr(int)
A.sort()

A_ = A[:]
l = 0; r = n-1
ind = 0

while (r >= l):
    A_[ind] = A[l]
    l += 1; ind+=1
    if (r >= l):
        A_[ind] = A[r]
        r -= 1;ind+=1


print ' '.join([str(x) for x in A_])
