n = int(raw_input())
s = raw_input().split()
l = []
for i in range(n):
    l.append(int(s[i]))
#print l

for i in range(n):
    for j in range(n - 1):
        if l[j] > l[j + 1]:
            k = l[j]
            l[j] = l[j + 1]
            l[j + 1] = k

for i in range(n - 1):
    if i % 2 == 1:
        k = l[i]
        l[i] = l[i + 1]
        l[i + 1] = k
s = ""
for i in range(n):
    s = s +  str(l[i]) + ' '
