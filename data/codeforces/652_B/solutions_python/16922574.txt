n = int(raw_input())
store = sorted(map(int,raw_input().split()))

k = n/2

index = 0
backindex = n-1

ans = []

for i in range(1,n+1):
    if i % 2 == 1:
        ans.append(store[index])
        index += 1
    else:
        ans.append(store[backindex])
        backindex -= 1

print ' '.join(str(i) for i in ans)
