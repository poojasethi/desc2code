def z_sorted(a):
    for i in xrange(1,len(a)):
        if i%2==0 and a[i]>a[i-1]:
            return False
        if i%2!=0 and a[i]<a[i-1]:
            return False
    return True


n = int(raw_input())
a = map(int, raw_input().split())
a.sort()
for x in xrange(n-1,0,-1):
    if x%2==0:
        a[x],a[x-1] = a[x-1],a[x]
if z_sorted(a):
