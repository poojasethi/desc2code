n=int(raw_input())
a=map(int,raw_input().split())
b=sorted(a)
ans=[b[0]]
l=1
r=n-1
while l<=r:
    if len(ans)%2==0:
        ans.append(b[l])
        l+=1
    else:
        ans.append(b[r])
        r-=1
poss=True
for i in xrange(1,len(ans)):
    if i%2 and ans[i]<ans[i-1] or i%2==0 and ans[i]>ans[i-1]:
        poss=False
if poss:print ' '.join(map(str,ans))
else:print 'Impossible'
