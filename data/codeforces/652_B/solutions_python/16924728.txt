from copy import deepcopy
def input_list():
    i = raw_input()
    return map(eval, i.split(' '))

def main():
    [length] = input_list()
    l = sorted(input_list())
    rl = list(reversed(l))
    i = 0
    n = 0
    ans = ""
    while True:
        ans += str(l[i]) + " "
        n += 1
        if n >= length:
            break
        ans += str(rl[i]) + " "
        n += 1
        if n >= length:
            break
        i += 1
    print ans

main()
