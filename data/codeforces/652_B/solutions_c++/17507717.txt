#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin >> n;
	int A[n+1];
	int i;
	for(i=0;i<n;i++)
		cin >> A[i];
	sort(A,A+n);
		for(i=0;i<n/2;i++)
		{
			if(i)
				cout << " ";
			cout << A[i];
			cout << " " << A[n/2+i + n%2];
		}
	if(n%2)
		cout << " " << A[n/2];
	cout << endl;
	return 0;
}
