#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<bitset>
using namespace std;
using std::bitset;
#define rep(i,n) for(i=0;i<n;++i)
#define FOR(i,j,k) for(i=j;i<=k;++i)
#define FORD(i,j,k) for(i=j;i>=k;--i)
#define met(i,j) memset(i,j,sizeof(i))
#define sz size()
const int oo=2147483647;
#define M 600005
#define N 505
struct question
{
    int x1,x2,y1,y2,bh,ans;
}ask[M];
int n,m,num;
char map[N][N];
bitset<512> f[N][N],g[N][N],tq;

void init()
{
    int i,j;
    scanf("%d%d\n",&n,&m);
    FOR(i,1,n)
    {
        FOR(j,1,m)
        {
            map[i][j]=getchar();
        }
        getchar();
    }
    scanf("%d",&num);
    FOR(i,1,num)
    {
        scanf("%d%d%d%d",&ask[i].x1,&ask[i].y1,&ask[i].x2,&ask[i].y2);
        ask[i].bh=i;
    }
}

void calc(int l,int r)
{
    if(l>r)return;
    int i,t1=oo,t2=0,j,t3,temp;
    FOR(i,l,r)
    {
        t1=min(t1,ask[i].y1);
        t2=max(t2,ask[i].y2);
    }
    t3=(t1+t2)/2;
    FORD(i,n,1)
    {
        FORD(j,t3,t1)
        {
            f[i][j].reset();
            if(i!=m)f[i][j]|=f[i+1][j];
            if(j!=t3)f[i][j]|=f[i][j+1];
            if(map[i][j]=='#')f[i][j].reset();else {if(j==t3)f[i][j].set(i);}
        }
    }
    FOR(i,1,n)
    {
        FOR(j,t3,t2)
        {
            g[i][j].reset();
            if(i!=1)g[i][j]|=g[i-1][j];
            if(j!=t3)g[i][j]|=g[i][j-1];
            if(map[i][j]=='#')g[i][j].reset();else {if(j==t3)g[i][j].set(i);}
        }
    }
    FOR(i,l,r)
    {
        if(ask[i].y1<=t3&&ask[i].y2>=t3)
        {
            tq=f[ask[i].x1][ask[i].y1]&g[ask[i].x2][ask[i].y2];
            if(tq.any())ask[i].ans=1;
        }
    }
    for(i=l;i<=r;++i)
    {
        while(ask[i].y1<=t3&&ask[i].y2>=t3&&i<=r)
        {
            swap(ask[i],ask[r]);
            --r;
        }
    }
    temp=r;
    for(i=l;i<=temp;++i)
    {
        while(ask[i].y1>t3&&i<=temp)
        {
            swap(ask[i],ask[temp]);
            --temp;
        }
    }
    calc(l,temp);
    calc(temp+1,r);
}

bool comp(question i,question j)
{
    return i.bh<j.bh;
}

void work()
{
    calc(1,num);
    sort(ask+1,ask+1+num,comp);
    int i;
    FOR(i,1,num)
    {
        if(ask[i].ans)puts("Yes");else puts("No");
    }
}

int main()
{
    init();
    work();
}
