#include<cstdio>
#include<cstdlib>
#include<vector>
#include<algorithm>
using namespace std;
vector<int> q[120000];
struct ppp{
	int x1,x2,y1,y2;
} a[710000];
int i,j,k,l,n,m,f[510][510][20],r,mid,p,qn,mm,ms,mp,x,ans[710000],g[510][510][20];
char c[510][510];
void work(int p,int l,int r){
	if(l>r)return;
	int mid=(l+r)/2,i,j;
	for(j=mid;j>=l;--j)
		for(i=n;i>=1;--i)if(c[i][j]=='.'){
			for(k=1;k<=mm;++k)f[i][j][k]=f[i+1][j][k];
			if(j==mid){
				x=(i-1)/32+1;
				f[i][j][x]|=1<<(i-(x-1)*32-1);
			}
			if(j!=mid)for(k=1;k<=mm;++k)f[i][j][k]|=f[i][j+1][k];
		}
	for(j=mid;j<=r;++j)
		for(i=1;i<=n;++i)if(c[i][j]=='.'){
			for(k=1;k<=mm;++k)g[i][j][k]=g[i-1][j][k];
			if(j==mid){
				x=(i-1)/32+1;
				g[i][j][x]|=1<<(i-(x-1)*32-1);
			}
			if(j!=mid)for(k=1;k<=mm;++k)g[i][j][k]|=g[i][j-1][k];
		}
	for(i=0;i<q[p].size();++i){
		x=q[p][i];
		for(j=1;j<=mm;++j){
		//	printf("%d %d\n",f[a[x].x1][a[x].y1][j],f[a[x].x2][a[x].y2][j]);
			if(f[a[x].x1][a[x].y1][j]&g[a[x].x2][a[x].y2][j])break;
		}
		if(j==mm+1)ans[x]=0;else ans[x]=1;
	}
	work(p*2,l,mid-1);
	work(p*2+1,mid+1,r);
}
int main(){
	scanf("%d%d",&n,&m);
	mm=(n-1)/32+1;
	for(i=1;i<=n;++i)scanf("%s",c[i]+1);
	for(scanf("%d",&qn),i=1;i<=qn;++i){
		scanf("%d%d%d%d",&a[i].x1,&a[i].y1,&a[i].x2,&a[i].y2);
		l=1; r=m; p=1;
		while(1){
			mid=(l+r)/2;
			if(a[i].y1<=mid && a[i].y2>=mid)break;
			if(a[i].y2<mid)p=p*2,r=mid-1;
			else p=p*2+1,l=mid+1;
		}
	//	printf("%d\n",p);
		q[p].push_back(i);
	}
	work(1,1,m);
	for(i=1;i<=qn;++i)printf("%s\n",ans[i]==1?"Yes":"No");
//	for(;;);
}
