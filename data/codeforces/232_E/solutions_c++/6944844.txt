#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <bitset>
#include <vector>
#define N 510
#define M 600010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int x,y,xx,yy,id;
};
int i,j,k,n,m,Q;
int an[M];
char p[N][N];
vector<ww> b;
bitset<N> bi[N][N],s,Bi[N][N];
void work(int x,int xx,int y,int yy,vector<ww> a) {
	if (!a.size()) return;
	int i,j,mid,lx=xx-x+1,ly=yy-y+1;
	if (lx<ly) {
		mid=ly/2+y;
		For(i,x,xx) {
			For(j,y,mid) bi[i][j].reset();
			For(j,mid,yy) Bi[i][j].reset();
		}
		for (i=xx;i>=x;i--) if (p[i][mid]=='.') {
			bi[i][mid][i]=Bi[i][mid][i]=1;
			if (i<xx) bi[i][mid]|=bi[i+1][mid];
		}
		for (j=mid-1;j>=y;j--) for (i=xx;i>=x;i--) if (p[i][j]=='.') {
			bi[i][j]=bi[i][j+1];
			if (i<xx) bi[i][j]|=bi[i+1][j];
		}
		for (j=mid+1;j<=yy;j++) for (i=x;i<=xx;i++) if (p[i][j]=='.') {
			Bi[i][j]=Bi[i][j-1];
			if (i>x) Bi[i][j]|=Bi[i-1][j];
		}
		vector<ww> b,c;
		for (i=0;i<a.size();i++) {
			ww A=a[i];
			if (A.y<=mid&&A.yy>=mid) {
				s=bi[A.x][A.y]&Bi[A.xx][A.yy];
				an[A.id]=s.any();
			}
			if (A.yy<mid) b.push_back(A);
			if (A.y>mid) c.push_back(A);
		}
		work(x,xx,y,mid-1,b);
		work(x,xx,mid+1,yy,c);
	} else {
		mid=lx/2+x;
		For(i,x,mid)For(j,y,yy) bi[i][j].reset();
		For(i,mid,xx)For(j,y,yy) Bi[i][j].reset();
		for (i=yy;i>=y;i--) if (p[mid][i]=='.') {
			bi[mid][i][i]=Bi[mid][i][i]=1;
			if (i<yy) bi[mid][i]|=bi[mid][i+1];
		}
		for (i=mid-1;i>=x;i--) for (j=yy;j>=y;j--) if (p[i][j]=='.') {
			bi[i][j]=bi[i+1][j];
			if (j<yy) bi[i][j]|=bi[i][j+1];
		}
		for (i=mid+1;i<=xx;i++) for (j=y;j<=yy;j++) if (p[i][j]=='.') {
			Bi[i][j]=Bi[i-1][j];
			if (j>y) Bi[i][j]|=Bi[i][j-1];
		}
		vector<ww> b,c;
		for (i=0;i<a.size();i++) {
			ww A=a[i];
			if (A.x<=mid&&A.xx>=mid) {
				s=bi[A.x][A.y]&Bi[A.xx][A.yy];
				an[A.id]=s.any();
			}
			if (A.xx<mid) b.push_back(A);
			if (A.x>mid) c.push_back(A);
		}
		work(x,mid-1,y,yy,b);
		work(mid+1,xx,y,yy,c);
	}
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n) scanf("%s",p[i]+1);
	scanf("%d",&Q);
	For(i,1,Q) {
		ww A;
		scanf("%d%d%d%d",&A.x,&A.y,&A.xx,&A.yy),A.id=i;
		b.push_back(A);
	}
	work(1,n,1,m,b);
	For(i,1,Q) printf("%s\n",an[i]?"Yes":"No");
	//for(;;);
	return 0;
}
