#include<cstdio>
#include<cctype>
#include<iostream>
#include<sstream>
#include<cstring>
#include<vector>
#include<map>
#include<set>
#include<queue>
#include<bitset>
#include<string>
#include<cmath>
#include<algorithm>
#define ll long long
#define N 509
#define inf 1000000009
#define mod 1000000007
using namespace std;
int n,m,q;
char s[N][N];
bool ans[600009];
bitset<N> f[N][N],g[N][N];
struct node{int x1,y1,x2,y2,p;}Q;
void go(int l,int r,vector<node> v){
    if(v.empty()) return;
    int mid=(l+r)>>1;
    for(int i=mid;i>=l;i--)
        for(int j=m;j;j--){
            f[i][j].reset();
            if(s[i][j]=='#') continue;
            if(i==mid) f[i][j].set(j),f[i][j]|=f[i][j+1];
            else f[i][j]=f[i+1][j]|f[i][j+1];
        }
    for(int i=mid;i<=r;i++)
        for(int j=1;j<=m;j++){
            g[i][j].reset();
            if(s[i][j]=='#') continue;
            if(i==mid) g[i][j].set(j),g[i][j]|=g[i][j-1];
            else g[i][j]=g[i-1][j]|g[i][j-1];
        }
    vector<node> left,right;
    for(int i=0;i<v.size();i++){
        if(v[i].x2<mid) left.push_back(v[i]);
        else if(v[i].x1>mid) right.push_back(v[i]);
        else ans[v[i].p]=(f[v[i].x1][v[i].y1]&g[v[i].x2][v[i].y2]).any();
    }
    go(l,mid,left),go(mid+1,r,right);
}
int main(){
    cin>>n>>m;
    for(int i=1;i<=n;i++) cin>>s[i]+1;
    cin>>q;
    vector<node> v;
    for(int i=0;i<q;i++) scanf("%d%d%d%d",&Q.x1,&Q.y1,&Q.x2,&Q.y2),Q.p=i,v.push_back(Q);
    go(1,n,v);
    for(int i=0;i<q;i++) ans[i]?puts("Yes"):puts("No");
}
