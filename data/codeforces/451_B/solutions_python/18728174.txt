n=input()
a=map(int,raw_input().split())
i=j=0
for k in range(n-1):
    if a[k]>a[k+1]:
        i=k
        break
for k in range(n-1,0,-1):
    if a[k]<a[k-1]:
        j=k
        break
if any(a[k]<a[k+1] for k in range(i,j)) or (j<n-1 and a[i]>a[j+1]) or (i>0 and a[i-1]>a[j]):
    print 'no'
    exit(0)
print 'yes'
print i+1,j+1
