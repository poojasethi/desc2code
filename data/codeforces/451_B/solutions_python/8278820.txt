def solve(n, a):
  d = [0] + a + [2**32]
  b, e = 0, 0
  for i in range(1, n+2):
    if d[i] < d[i-1]:
      if e: return False
      if not b: b = i-1
    elif b and not e:
      e = i-1
  if not b and not e: return 1, 1
  if not (d[b-1] < d[e] < d[b] < d[e+1]): return False
  return b, e

n = int(raw_input())
a = map(int, raw_input().split())
res = solve(n, a)
if not res:
  print 'no'
else:
  print 'yes'
  print res[0], res[1]
