n = input()
m = map(int, raw_input().strip().split())

sm = sorted(m)

begin = 0
end = 0

for i in xrange(n):
    if m[i] != sm[i]:
        begin = i
        break
for i in range(n)[::-1]:
    if m[i] != sm[i]:
        end = i
        break
out = m[:begin]+m[begin:end+1][::-1]+m[end+1:]
if sm == out:
    print 'yes'
    print begin+1, end+1
else:
    print 'no'
