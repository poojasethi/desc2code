n = input()
a = map(int, raw_input().split())

startFlag = False
start = 0
end = 0

for i in range(len(a) - 1):
    if a[i] >= a[i+1] and not startFlag:
        start = i
        startFlag = True
    if a[i] >= a[i+1] and startFlag:
        end = i + 1
    if a[i] < a[i+1] and startFlag:
        break

res = a[:start] + sorted(a[start:end + 1]) + a[end + 1:]
for i in range(len(res) - 1):
    if res[i] > res[i + 1] - 1:
        print 'no'
        quit()

print 'yes'
print start + 1, end + 1
