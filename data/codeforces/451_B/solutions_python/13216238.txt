n = int(raw_input())
a = map(int, raw_input().split())

left = right = n
for i in range(0, n - 1):
    if a[i] > a[i + 1]:
        left = i
        break

if left == n:
    print('yes\n1 1')
else:
    for i in range(left, n - 1):
        if a[i] < a[i + 1]:
            right = i + 1
            break

    if a[:left] + a[left:right][::-1] + a[right:] == sorted(a):
        print('yes\n{0} {1}'.format(left + 1, right))
    else:
        print('no')
