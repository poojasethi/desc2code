n = int(raw_input())
r = list(enumerate([int(c) for c in raw_input().split()],1))
r.sort(key=lambda x:x[1])
r = map(lambda x:x[0],r)
x = [i for i in xrange(n) if i+1 != r[i]]
if x:
	a,b = min(x),max(x)
	if range(1,n+1) == r[:a] + list(reversed(r[a:b+1])) + r[b+1:]:
		print 'yes'
		print a+1,b+1
	else:
		print 'no'
else:
	print 'yes'
	print 1,1
