n = int(raw_input())
k = map(int, raw_input().split())
k = [0] + k
l = k[:]
l = sorted(l)
flag, starting, ending = 1, -1, -1
for i in xrange(1, n+1):
	if (k[i]!=l[i]):
		ending = i
		if (starting==-1):
			starting = i
for i in xrange(starting, ending+1, 1):
	if starting==-1:
		break
	else:
		if (k[i]!=l[ending-i+starting]):
			flag = 0
if (starting == -1):
	starting = ending = 1
if (flag):
	print "yes"
	print starting, ending
else:
	print "no"
