import sys

n = int(sys.stdin.readline())

l = list(map(int,sys.stdin.readline().split()))

s=f=-1

for i in xrange(n-1):
    if l[i+1] < l[i]:
        s = i
        break

for i in xrange(n-1,0,-1):
    if l[i-1]>l[i]:
        f = i
        break

if s==-1 and f==-1:
    print "yes"
    print "1 1"
else:
    #print s,f
    l1 = sorted(l)
    l2 = l[:s]+l[s:f+1][::-1]+l[f+1:]
    #print l1,l2
    if l1==l2:
        print "yes"
        print "%d %d"%(s+1,f+1)
    else:
        print "no"
