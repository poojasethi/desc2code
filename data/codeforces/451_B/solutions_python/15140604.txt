n = int(raw_input())
a = map(int, raw_input().split())
l = n
for i in range(n-1):
    if a[i+1] < a[i]:
        l = i
        break
if l == n:
    print "yes"
    print "1 1"
else:
    for i in range(n-1, 0, -1):
        if a[i-1] > a[i]:
            r = i
            break
    a[l:r+1] = a[l:r+1][::-1]
    if a == sorted(a):
        print "yes"
        print l+1, r+1
    else:
        print "no"
