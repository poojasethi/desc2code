n, a = int(raw_input()), map(int, raw_input().split())
o, m = sorted(a), {}
for i in xrange(n):
	m[a[i]] = i
for i in xrange(n):
	a[m[o[i]]] = i
l, r = -1, -1
for i in xrange(n):
	if a[i] != i:
		l = i; break
for i in xrange(n-1, -1, -1):
	if a[i] != i:
		r = i; break
if l < 0 or r < 0: print "yes\n1 1"
else:
	a = a[:l] + a[l:r+1][::-1] + a[r+1:]
	o = sorted(a)
	if a == o: print "yes\n", l+1, r+1
	else: print "no"
