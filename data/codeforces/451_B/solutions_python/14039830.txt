n = int(raw_input())
a = [0]
a = a + map(int,raw_input().split()) + [10**9 + 1]
# print a
flag = 0
for i in range(1,n+2):
	if flag!=1 and a[i]<=a[i-1]:
		flag+=1
		if flag==1:
			left=i-1
	if flag==1 and a[i]<=a[i-1]:
		right=i
	elif flag==1:
		flag+=1
		if a[i]<a[left] or a[left-1]>a[i-1]:
			flag+=1
# print flag
if flag==2:
	print 'yes'
	print left,right
elif flag==0:
	print 'yes'
	print '1 1'
else:
	print 'no'
