R = lambda:raw_input()
a = int(R())
k = [int(x) for x in R().split()]
y = []
for x in xrange(a-1):
    if k[x] > k[x+1]:
        y.append(x)

if y == []:
    print "yes"
    print 1,1

else:
    start = y[0]
    end = y[-1]+1
    
    if k[0:start] + k[start:end+1][::-1] + k[end+1:a] == sorted(k):
        print "yes"
        print start+1,end+1

    else:
        print "no"
