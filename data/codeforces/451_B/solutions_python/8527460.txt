n = input()
a = map(int, raw_input().split())

d = []
for i in range(1, n):
    if a[i] < a[i-1] and (len(d)+1)%2: d += [i]
    if a[i] > a[i-1] and len(d)%2: d += [i]

if len(d)==1: d=[d[0], n]
if d==[]: d=[1, 1]

x = a[d[0]-1:d[1]]
x.reverse()
a[d[0]-1:d[1]] = x

print 'yes\n'+' '.join(map(str, d)) if len(d)==2 and a==sorted(a) else 'no'
