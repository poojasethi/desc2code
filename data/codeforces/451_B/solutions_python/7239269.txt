n = int (input ())
a = map (int , raw_input ().split ())
l , r = 0 , n - 1
while l + 1 < n :
    if a[l] > a[l + 1] :
        break
    l += 1
while r > 0 :
    if a[r] < a[r - 1] :
        break;
    r = r - 1
if l == n - 1 :
    print 'yes\n1 1'
    exit ()
x , y = l , r
while x < y :
    a[x] , a[y] = a[y] , a[x]
    x += 1
    y -= 1
for i in range (n - 1) :
    if a[i] > a[i + 1] :
        print 'no'
        exit ()
print 'yes' 
print l + 1 , r + 1
