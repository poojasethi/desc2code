n=int(raw_input())
y=map(int,raw_input().split())
g=y[:]
st=0
en=1
x=0
flag=0
if (n==1):
    flag=2
while(x<n):
    st=x
    while (x<n-1) and (y[x]>y[x+1]):
        x=x+1
        flag=1
    en=x
    if (flag==1):
        break
    x=x+1
g[st:en+1] = reversed(g[st:en+1])
y.sort()
if (g==y):
    print "yes"
    if flag==0:
        print st,en
    else:
        print st+1,en+1
else:
    print "no"
