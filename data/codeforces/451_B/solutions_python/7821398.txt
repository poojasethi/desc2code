n = input()
a = map(int,raw_input().split())
s = sorted(a)
i,j = 0,0
for x in xrange(n):
    if a[x] != s[x]:
        i = x
        break
for x in xrange(n-1,-1,-1):
    if a[x] != s[x]:
        j = x
        break
if s == a[:i] + a[i:j+1][::-1] + a[j+1:]:
    print "yes"
    print i+1,j+1
else:
    print "no"
