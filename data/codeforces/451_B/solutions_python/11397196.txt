n = input()
a = map(int, raw_input().split())
b = sorted(a)
l, r = 0, 0

for i in xrange(n):
    if a[i] != b[i]:
        l = i
        break

for i in xrange(n - 1, 0, -1):
    if a[i] != b[i]:
        r = i
        break
# print a[:l]
# print a[l:r+1][::-1]
# print a[r+1:]
c = a[:l] + a[l:r+1][::-1] + a[r+1:]
# print l, r
# print c
if c == b:
    print 'yes'
    print l + 1, r + 1
else:
    print 'no'
