n = int(raw_input())
a = map(int, raw_input().split())
b = sorted(a)
x = 0
y = 0
for i in xrange(n):
    if a[i] != b[i]:
        x = i
        break
for i in reversed(xrange(n)):
    if a[i] != b[i]:
        y = i
        break
if a[x:y+1] == b[x:y+1][::-1]:
    print 'yes'
    print x+1, y+1
else:
    print 'no'
