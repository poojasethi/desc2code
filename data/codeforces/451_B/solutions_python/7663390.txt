n=input()
m=map(int, raw_input().split())
s=-1
e=-1
for i in range(n-1):
    if s==-1 and m[i] >= m[i+1]:
        s = i
    if s != -1 and e==-1 and m[i] < m[i+1]:
        e = i
if e == -1: e = n-1

if s > -1 and e > -1 and e > s:
    m[s:e+1]=m[s:e+1][::-1]

if sorted(m) == m:
    print 'yes'
    if s == -1:s=e
    print s+1,e+1
else:
    print 'no'
