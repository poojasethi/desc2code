n = input()
a = map(int, raw_input().split())
b = sorted(a)
seg_begin, seg_end = -1, -1

for i in xrange(len(a)):
	if a[i] != b[i]:
		if seg_begin == -1: seg_begin = i
		seg_end = i

ans_b, ans_e = seg_begin + 1, seg_end + 1

while seg_begin < seg_end:
	b[seg_begin] ^= b[seg_end]
	b[seg_end] ^= b[seg_begin]
	b[seg_begin] ^= b[seg_end]
	seg_begin += 1
	seg_end -= 1

if seg_begin == -1:
	print 'yes\n1 1'
elif a == b:
	print 'yes'
	print ans_b, ans_e
else:
	print 'no'
