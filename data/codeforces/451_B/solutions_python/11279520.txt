n = int(raw_input())
s = map ( int , raw_input().split() )

l = -1
r = n-1
for i in range(0,n - 1):
    if l == -1 and  s[i] > s[i+1]:
        l = i

    elif l != -1 and s[i] < s[i+1]:
        r=i
        break

new = s[l:r+1]
new.reverse()
for i in range(0 , len(new)):
    s[l+i] = new[i]

for i in range (0 , len (s) -1 ):
    if s[i] > s[i+1]:
        print "no"
        exit()
print "yes"
if l == -1 :
    print 1, 1
else:
    print l + 1, r +1
