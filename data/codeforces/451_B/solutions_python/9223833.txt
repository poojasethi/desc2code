n=int(raw_input())
arr=[int(i) for i in raw_input().split()]
b=sorted(arr)
x,y=0,0
for i in xrange(n):
    if arr[i]!=b[i]:
        x=i
        break
for i in range(n)[::-1]:
    if arr[i]!=b[i]:
        y=i
        break
c=arr[:x]+arr[x:y+1][::-1]+arr[y+1:]
if c!=b:
    print "no"
else:
    print "yes"
    print x+1,y+1
