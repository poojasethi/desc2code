def is_sorted(l):
  return all(l[i]<=l[i+1] for i in xrange(len(l)-1))


def solve(l,n):
  a = min(i for i in xrange(n) if l[i]!=i)
  b = max(i for i in xrange(n) if l[i]!=i)
  l[a:b+1] = l[a:b+1][::-1]
  if is_sorted(l):
    print "yes"
    print a+1,b+1
  else:
    print "no"

n = int(raw_input())
A = map(int, raw_input().split())
if is_sorted(A):
  print "yes"
  print "1 1"
else:
  B = sorted(xrange(n),key=lambda k: A[k])
  solve(B,n)
