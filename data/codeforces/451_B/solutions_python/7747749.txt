n = int(raw_input())
a = map(int, raw_input().split())

left, right = 0, n-1

while left < n-1 and a[left] < a[left+1]:
    left += 1

while right >= 1 and a[right] > a[right-1]:
    right -= 1

if left == n-1:
    print "yes"
    print "1 1"

else:
    b = []
    b.extend(a[:left])

    x = a[left:right+1]
    x.reverse()
    b.extend(x)
    b.extend(a[right+1:])

    if sorted(b) == b:
        print "yes"
        print left+1, right+1
    else:
        print "no"
