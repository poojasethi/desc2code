a = [map(int, raw_input().split()) for i in [0]*2][1]
res = (-1, -1)
i = 0
while i < len(a) - 1:
	if a[i] > a[i + 1]: break
	i += 1
j = i

while j < len(a) - 1:
	if a[j] < a[j + 1]: break
	j += 1
	
b = a[:i]
b.extend(list(reversed(a[i:j+1])))
b.extend(a[j+1:])
if b == sorted(a): res = (i + 1, j + 1)
if a == sorted(a): print "yes\n", 1, 1
elif res == (-1, -1): print "no"
else: print "yes\n", res[0], res[1]
