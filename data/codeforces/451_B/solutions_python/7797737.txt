def main():
    n=int(input())
    a=map(int,raw_input().strip().split())
    i=0
    while i<n-1 and a[i]<=a[i+1]:i+=1

    if i==n-1:print "yes";print n,n;return

    j=i+1
    while j<n-1 and a[j]>=a[j+1]:j+=1

    if i>0 and a[j]<a[i-1]:print "no";return
    if j<n-1 and a[i]>a[j+1]:print "no";return
    #check that the rest is ordered
    for k in range(j+1,n-1):
        if a[k]>a[k+1]:print "no";return
    print "yes"
    print i+1,j+1


main()
