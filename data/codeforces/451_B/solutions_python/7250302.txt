n, a = input(), map(int, raw_input().split())
a.append(1000000001)
flag = 1
n = m = len(a)
for i in range(1, n):
	if a[i] < a[i - 1]:
		m = i
		for j in range(i + 1, n):
			if a[j] > a[j - 1]: 
				k = j
				if (i >= 2 and a[i - 2] > a[j - 1]) or (a[j] < a[i - 1]) or any(a[k - 1] > a[k] for k in range(j + 1, n)):
					flag = 0
				break
		break
if flag:
	print 'yes'
	print '1 1' if m == n else '%d %d' % (m, k)
else:
	print 'no'
