n = int(raw_input())
a = map(int, raw_input().split())
i = 0
while i < n - 1 and a[i] < a[i + 1]:
    i += 1
start = i
while i < n - 1 and a[i] > a[i + 1]:
    i += 1
end = i
while i < n - 1 and a[i] < a[i + 1]:
    i += 1
if i == n - 1 and start <= end and (end == n - 1 or a[end + 1] > a[start]) and (start == 0 or a[start - 1] < a[end]):
    print 'yes'
    print start + 1, end + 1
else:
    print 'no'
