#include <iostream>
#include <algorithm>
using namespace std;
int a[100005];
bool check = true;
int n,k=0;
int main() {
	cin>>n;
	for (int i=0;i<n;i++)cin>>a[i];
	int i=0,j=n-1;	
	while (i<n-1 && a[i]<a[i+1])i++;	
	while (j>i && a[j-1]<a[j])j--;
	
	reverse(a+i,a+j+1);
	while(k<n-1 &&check){
		if (a[k]<a[k+1]) check=true;
		else check=false;	
		k++;
	}
	if (check) {
		cout<<"yes"<<endl;
		i++;j++;	
		cout<<i<<" "<<j;
	}
	else cout<<"no";
}
