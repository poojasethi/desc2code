#include<bits/stdc++.h>

using namespace std;

int ii=1,i,jj,n,arr[230000];

int main(){
	cin >> n;
	
	for(i=1;i<=n;i++)
		cin >> arr[i];

	for(i=1;i<n;i++)
		if(arr[i] > arr[i+1]){
			ii=i;
			break;
		}
	
	for(i=ii ;i<n;i++)
		if(arr[i] < arr[i+1]){
			jj = i;
			break;
		}
	if(i == n)
		jj = n;

	reverse(arr + ii , arr + jj + 1);

	for(i=1;i<n;i++)
		if(arr[i] > arr[i+1]){
		cout << "no" << endl;
		return 0;
		}
	cout << "yes" << endl << ii << " "<< jj << endl;
}
