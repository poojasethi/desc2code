#include<bits/stdc++.h>
using namespace std;

int A[100005],B[100005],n,i,x=1,y=1,s;

int main(){

	cin >> n;

	for(i = 0; i < n ; i++){
		cin >> A[i];
		B[i] = A[i];
	}
   if(n == 1) { puts("yes\n1 1"); return 0; }
	sort(B , B + n);

	for(i = 0;i < n;i++)
		if(A[i] != B[i]){
			s++; 
			if(s == 1) x = i;
			else y = i;
		}
	reverse(A + x, A + y + 1);
	for(i = x; i <= y; i++)
		if(A[i] != B[i])
		{cout << "no"; return 0;}
	cout << "yes\n" << x + 1 << " " << y + 1;
	return 0;
}
