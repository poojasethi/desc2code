#include<stdio.h>
#include<algorithm>

using namespace std;

#define MAX 100007

int a[MAX], b[MAX];

int main()
{
	int n, i, j, l, r;
	scanf("%d", &n);
	for(i=0; i^n; i++)
	{
		scanf("%d", a+i);
		b[i]=a[i];
	}
	sort(a, a+n);
	i=0;
	while(i<n && a[i]==b[i])i++;
	l=i;
	r=j=n-1;
	if(i<n)
	{
		while(a[j]==b[j])j--;
		r=j;
		while(i<=r && b[i]==a[j])i++, j--;
	}
	else l--;
	printf(i<=r?"no\n":"yes\n%d %d\n", l+1, r+1);
	return 0;
}
