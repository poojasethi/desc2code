#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n+1],l=0,r=1,b[n+1];
	bool w=false;
	for(int i=1;i<=n;i++)cin>>a[i],b[i]=a[i];
	sort(b+1,b+n+1);
	for(int i=1;i<=n;i++)
	{
		if(a[i]!=b[i])
		{
			r=i;
			if(l==0)l=i; 
		} 
	}
	if(l==0)l=1;
	if(r==0)r=1;
	for(int i=l,j=r;i<=r;i++,j--)if(a[i]!=b[j])
	{
		cout<<"no"<<endl;
		return 0;
	}
	cout<<"yes"<<endl;
	cout<<l<<" "<<r<<endl;
	return 0;
}
