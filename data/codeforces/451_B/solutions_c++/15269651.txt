#include<bits/stdc++.h>
using namespace std;
int n,ar[1000000],l=1,r=1,cnt;
int main(){
	scanf("%d",&n);
	for(int i=1 ; i<=n ; i++){
		scanf("%d",&ar[i]);
	}
	for(int i=2 ; i<=n ; i++){
		if(ar[i-1]>ar[i]){
			l=i-1;
			while(i+1<=n && ar[i]>ar[i+1]){
				i++;
			}
			r=i;
			reverse(ar+l,ar+r+1);
			break;
		}
	}
	for(int i=1 ; i<n ; i++){
		if(ar[i]>ar[i+1]){
			cnt=1;
		}
	}
	if(!cnt){
		printf("yes\n%d %d\n",l,r);
	}
	else
		printf("no\n");
	return 0;
}
