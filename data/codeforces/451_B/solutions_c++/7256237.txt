#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n,s,a[100001],b[100001],ans[100001];
int main()
{
	scanf("%d",&n);for(int i=1;i<=n;i++)scanf("%d",&a[i]),b[i]=a[i];sort(b+1,b+n+1);
	for(int i=1;i<=n;i++)if(a[i]!=b[i])ans[++s]=i;
	for(int i=ans[1];i<=ans[s];i++)if(a[ans[1]+ans[s]-i]!=b[i]){printf("no\n");return 0;}
	if(!s)ans[0]=ans[1]=1;printf("yes\n%d %d\n",ans[1],ans[s]);return 0;
}
