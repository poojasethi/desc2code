#include <stdio.h>
#include <algorithm>
#include <vector>

using namespace std;

int main(){
	int i,j,n,ai=0;
	scanf("%d",&n);
	vector<int> a(n);
	for (i=0;i<n;i++){
		scanf("%d",&a[i]);
	}

	for(i=0;ai<a[i]&&i<n;i++){
		ai=a[i];
	}

	for(j=i;a[j]<ai&&j<n;j++){
	}
	reverse(a.begin()+i-1,a.begin()+j);
	for(int ii=1;ii<n;ii++){
		if(a[ii-1]>a[ii]){
			printf("no");
			return 0;
		}
	}
	printf("yes\n%d %d",i,j);
	return 0;
}
