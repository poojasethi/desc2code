#include <cstdio>
#include <algorithm>
using namespace std;
int main(void){
	int n,i,p=1,q=1;
	int a[200000];
	scanf("%d",&n);
	for (i=1; i<=n; i++)
		scanf("%d",&a[i]);
	for (i=1; i<n; i++)
		if (a[i]>a[i+1])
		{
			p=i;
			break;
		}
	for (i=2; i<=n; i++)
		if (a[i-1]>a[i]){
			q=i;
		}
    reverse(a + p, a + q + 1);
		for (i=1; i<n; i++)
			if (a[i]>a[i+1]){
				printf("no\n");
				return 0;
			}

	printf("yes\n");		
	printf("%d %d\n",p,q);	
	return 0;
}
