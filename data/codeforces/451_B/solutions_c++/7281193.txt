#include <cstdio>
#include <algorithm>
using namespace std;
int a[111111];
int main()
{
    int n,l,r;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(l=0;l<n-1&&a[l]<a[l+1];l++);
    for(r=l;r<n-1&&a[r]>a[r+1];r++);
    reverse(a+l,a+r+1);
    for(int i=0;i<n-1;i++){
        if(a[i]>a[i+1]){
            puts("no");
            return 0;
        }
    }
    printf("yes\n%d %d",l+1,r+1);
    return 0;
}
