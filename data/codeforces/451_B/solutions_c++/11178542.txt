#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,in,fin;
	cin>>n;
	int a[100002];
	for(i=1;i<=n;i++)
		scanf("%d",a+i);
	a[0]=0;
	a[n+1]=2e9;
	for(i=1;a[i]>a[i-1] && i<=n;i++);
	in=i-1;
	for(;a[i]<a[i-1];i++);
	fin=i-1;
	for(;i<=n;i++)
		if(a[i]<a[i-1])
		{
			puts("no");
			return 0;
			}
	if(a[in]<a[fin+1] && a[fin]>a[in-1])
	{
		puts("yes");
		printf("%d %d",in,fin);
		return 0;
		}
		puts("no");
}
