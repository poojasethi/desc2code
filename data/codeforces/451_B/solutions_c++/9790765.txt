#include <cstdio>
#include <algorithm>
int main() {
	int n, ar[100005], f = -1,  j;
	scanf("%d", &n);
	for(int i = 0; i < n; ++i) scanf("%d", &ar[i]);
	for(int i = 0; i < n - 1; ++i)
		if(ar[i] > ar[i + 1]) {
			for(f = i, j = i + 1; ar[j] < ar[j - 1] && j < n; ++j);
			std::reverse(ar + f, ar + j);
			break;
		}
	for(int i = 0; i < n - 1; ++i) if(ar[i] > ar[i + 1]) { puts("no"); return 0; }
	if(f == -1) printf("yes\n%d %d", 1, 1);
	else printf("yes\n%d %d", f + 1, j);
}
