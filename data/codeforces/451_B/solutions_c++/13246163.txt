#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,c=-3,s=0,i,j;
	cin>> n;
	int a[n+2],b[n+2];
	for(i=0; i< n ;i++)
	{
		cin>> a[i];
		b[i] = a[i];
	}
	sort(a,a+n);
	for(i=0 ;i<n ;i++)
	{
		if(a[i]!=b[i])
		{
			if(c==-3)
			c=i;

			s=i;
		}
	}
	if(c==-3)
	c=0;
	if(s==0)
	s=0;
	i=c;
	j=s;
	while(i<=s)
	{
		if(a[i]!=b[j])
		{
			cout<< "no"<<endl;
			return 0;
		}
		i++;
		j--;
	}
	cout<<"yes" << endl<< c+1 << " "<< s+1;
	return 0;
}
