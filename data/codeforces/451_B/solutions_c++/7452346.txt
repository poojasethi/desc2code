#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;
long a[100001];
int main()
{
long n,i,x,y;
x=1;
scanf("%ld",&n);
for(i=1;i<=n;i++)
{
scanf("%ld",&a[i]);
}
for(i=1;i<n;i++)
{if(a[i]>a[i+1])
  {x=i;
   break;
   }
}
for(i=x;i<n;i++)
{
if(a[i]<a[i+1])
{
y=i;
break;
}
}
if(i==n)
{y=n;
}
reverse(a+x,a+y+1);
for(i=1;i<n;i++)
{ if(a[i]>a[i+1])
   {break;}
}
if(i!=n)
{printf("no\n");}
else
{
printf("yes\n");
printf("%ld %ld\n",x,y);
}
}
