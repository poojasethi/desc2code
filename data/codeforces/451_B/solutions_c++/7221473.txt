#include <iostream>
#include <algorithm>
using namespace std;
int a[100005], b[100005], n, i, j, L, R;
int main()
{
	cin>>n;
	for (i=1; i<=n; i++) {
		cin>>a[i]; b[i]=a[i];
	}
	sort(a+1, a+n+1);
	for (i=1; i<=n; i++) if (a[i]!=b[i]) {
		R = i;
		if (L==0) L=i;
	}
	if (L==0) L=1;
	if (R==0) R=1;
	for (i=L,j=R; i<=R; i++,j--) if (a[i]!=b[j]) {
		cout<<"no"<<endl; return 0;
	}
	cout<<"yes"<<endl<<L<<" "<<R<<endl;
	return 0;
}
