#include <cstdio>
int x[100005],n;
bool no()
{
    int i,j,t;
    for(i=1;i<n&&x[i+1]>x[i];++i);
    if(i==n){printf("yes\n1 1\n");return 0;}
    for(j=n;j>1&&x[j-1]<x[j];--j);
    if(j<n&&x[i]>x[j+1]||i>1&&x[i-1]>x[j])return 1;
    for(t=i;t<j;++t)if(x[t]<=x[t+1])return 1;
    puts("yes");
    printf("%d %d\n",i,j);
    return 0;
}
int main()
{
    int i,j;
    scanf("%d",&n);
    for(i=1;i<=n;++i)scanf("%d",x+i);
    if(no())puts("no");
    return 0;
}
