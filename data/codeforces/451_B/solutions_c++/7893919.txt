#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=100050;
int n,a[maxn],anss=1;
char ans[2][5]={"no","yes"};
int a0[maxn];
int main()
{
	int i,l,r;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]),a0[i]=a[i];
	sort(a+1,a+1+n);
	for(l=1;l<=n&&a[l]==a0[l];l++);
	for(r=n;r&&a[r]==a0[r];r--);
	for(i=0;i<=r-l&&anss;i++)
		if(a[l+i]!=a0[r-i]) anss=0;
	printf("%s\n",ans[anss]);
	if(l>n) l=r=1;
	if(anss) printf("%d %d\n",l,r);
	return 0;
}
