#include <iostream>

using namespace std;

int t[100001],n,i,j,k;

void rev(int b,int c){
	
	int i=b,j=c,temp;
	
	while (i<j){
		temp=t[i];
		t[i]=t[j];
		t[j]=temp;
		i++;j--;
	}
	
}

int main(){
	
	cin>>n;
	for(i=1;i<=n;i++) cin>>t[i];

	for (i=1;i<n;i++) if (t[i]>t[i+1]) break;
	for (j=i;j<n;j++) if (t[j]<t[j+1]) break;
	
	
	rev(i,j);
	
	for (k=1;k<n;k++) if (t[k]>t[k+1]) break;
	
	if (k!=n) cout<<"no"<<endl;
	else {
		cout<<"yes"<<endl;
		cout<<i<<' '<<j<<endl;
	}
	

return 0;}
