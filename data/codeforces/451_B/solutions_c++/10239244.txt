#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int n,a[110000]={0};
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	scanf("%d",&a[i]);
	
	int k=1,j=1;
	for(int i=1;i<n;i++)
	if(a[i]>a[i+1])
	{k=i;
	break;}

	for(int i=n;i>=1;i--)
	if(a[i]<a[i-1])
	{j=i;
	break;}

	int biao=0;
	reverse(a+k,a+j+1);
	for(int i=1;i<n;i++)
	if(a[i]>a[i+1])
	{biao=1;
	break;}
	
	if(biao==0)
	{printf("yes\n");
	printf("%d %d",k,j);}
	else
	printf("no");
	return 0;
}
