#include<bits/stdc++.h>
using namespace std;
int i,j,n,b,s,t,k,a[100005],a1[100005];
int main()
{
	cin>>n;
	for(i=0;i<n;i++){cin>>a[i];a1[i]=a[i];}
	sort(a1,a1+n);
	for(i=0;i<n;i++)
		if(a[i]!=a1[i])
			break;
	if(i==n){printf("yes\n1 1\n");return 0;}
	b=i;
	for(i=n-1;i>=0;i--)if(a[i]!=a1[i])break;
	s=i;
	for(i=s-1;i>=b;i--)
	{
		if(a[i+1]>a[i]){cout<<"no"<<endl;return 0;}
	}
	cout<<"yes"<<endl<<b+1<<' '<<s+1<<endl;
	return 0;
}
