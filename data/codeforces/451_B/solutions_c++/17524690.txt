#include <bits/stdc++.h>
using namespace std;
int main()
{
    long int n,s[100002]={0};
    cin>>n;
    for(int i=1;i<=n;i++)cin>>s[i];
    int a1=1,a2=1;
    for(int i=1;i<=n;i++)if(s[i]<s[i-1]){a1=i-1;break;}
    for(int i=a1+1;i<=n;i++)if(s[i]<s[i-1])a2=i;else break;
    for(int i=a2+2;i<=n;i++)if(s[i]<s[i-1]){cout<<"no";return 0;}
    if((s[a1-1]&&s[a2]<s[a1-1])||(s[a2+1]&&s[a1]>s[a2+1])){cout<<"no";return 0;}
    cout<<"yes"<<endl<<a1<<" "<<a2<<endl;
    return 0;
}
