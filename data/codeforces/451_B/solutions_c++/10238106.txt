#include <bits/stdc++.h>

using namespace std;

int n,v[100009];
int r=1,l=1;

int main()
{
	
	cin >> n;

	for (int i = 1; i <= n; i++)
		cin	>> v[i];
		 
	for (int i = 1; i < n; i++)
		if (v[i]>v[i+1]){
			l=i;break;}
		
	for (int i = 2; i <= n; i++)
		if ( v[i] < v[i-1] )
			r=i;
		
	reverse(v+l,v+r+1);
	
	for (int i = 1; i < n; i++)
	{
		if ( v[i] > v[i+1] )
		{
			cout << "no";
			return 0;
		}
	}
	
	
	cout << "yes\n";
	cout << l << " " << r;
	
	return 0;
}
