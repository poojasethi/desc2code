#include <cstdio>
#define ll long long

int N,a[55];
ll K,f[55],g[55],h[55];

bool check(int l,int r,int t)
{
	for (int i=l; i<=r; i++) if (a[i]!=(a[l+r-i]^t)) return a[i]<=(a[l+r-i]^t);
	return 1;
}

int main()
{
	scanf("%d%I64d",&N,&K),K++,f[1]=2,g[1]=f[0]=g[0]=1;
	for (int i=2; i<=N; i++) f[i]=f[i-2]*2+(1ll<<i-2),g[i]=g[i-2]*2+(1ll<<i-2);
	a[1]=0;
	for (int i=2,k; i<=N; i++)
	{
		ll s; h[i]=h[i-1]*2;
		if (i==N)
		{
			if (K) a[N]=0,K-=check(1,N,0)&&check(1,N,1);
			if (K) a[N]=1,K-=check(1,N,0)&&check(1,N,1);
			break;
		}
		if (i+i<=N) s=((1ll<<i-1)-h[i]-1)*(1ll<<N-i-i+1)+f[N-i-i]+g[N-i-i];
		else s=((1ll<<N-i-1)-h[N-i]-1)*2+check(N-i+1,i,0)+check(N-i+1,i,1);
		if (s>=K) a[i]=0; else K-=s,a[i]=1,h[i]++;
	}
	if (K) puts("-1"); else for (int i=1; i<=N; i++) putchar(a[i]+48);
}
