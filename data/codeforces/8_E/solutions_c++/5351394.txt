#include<cstdio>
#include<cstring>
using namespace std;
typedef long long LL;
LL f[55][2][2],K; int a[55],n;
#define cl memset(f,-1,sizeof f)
#define F (f[l][rev][inv])
LL calc(int l, int r, int rev, int inv){
   if (l>r)return 1;if (~F)return F;F=0;
   for (int i=0;i<2;++i)for(int j=0;j<2;++j)
       if (a[l]!=(!i)&&a[r]!=(!j)&&(l<r||i==j)&&(rev||i<=j)&&(inv||i<=(!j)))F+=calc(l+1,r-1,rev||(i<j),inv||(i<!j));
   return F;
}
int main(){
    scanf("%d%I64d",&n,&K);++K;
    memset(a,-1,sizeof a);a[1]=0;cl;
    if (calc(1,n,0,0)<K){puts("-1");return 0;}
    for (int i=2;i<=n;++i){a[i]=0,cl;LL d=calc(1,n,0,0);K-=(a[i]=(d<K))*d;}
    for (int i=1;i<=n;++i)printf("%d",a[i]);puts("");
}
