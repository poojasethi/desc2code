#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
__int64 f[50][50][2][2],m,temp;
bool v[50][50][2][2];
int a[50],n,i;

__int64 calc(int l,int r,bool rev,bool inv)
{
    if(l>r) return 1;
    if(v[l][r][rev][inv]) return f[l][r][rev][inv];
    v[l][r][rev][inv]=1,f[l][r][rev][inv]=0;
    for(int i=0;i<2;i++) if(a[l]==-1||a[l]==i)
        for(int j=0;j<2;j++) if(a[r]==-1||a[r]==j)
            if((l<r||i==j)&&(rev||i<=j)&&(inv||i<=1-j))
                f[l][r][rev][inv]+=calc(l+1,r-1,rev||i<j,inv||i<1-j);
    return f[l][r][rev][inv];
}

int main() {
    cin>>n>>m,m++;
    memset(a,-1,sizeof(a));
    if(calc(0,n-1,0,0)<m) return puts("-1"),0;
    for(a[i]=0,i=1;i<n;i++) {
        memset(v,0,sizeof(v));
        a[i]=0;
        if((temp=calc(0,n-1,0,0))<m) m-=temp,a[i]=1;
    }
    for(i=0;i<n;i++) printf("%d",a[i]);
    puts("");
    return 0;
}
