#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define LL long long
using namespace std;
int s[51];
LL f[26][2][2];
bool vis[26][2][2];
int n;
LL m;
LL dfs(int p,int x,int y)
{
	if(p*2-1>n)
		return 1;
	if(f[p][x][y]!=-1)
		return f[p][x][y];
	f[p][x][y]=0;
	for(int i=0;i<2;i++)
		if(s[p]<0||s[p]==i)
			for(int j=0;j<2;j++)
				if((s[n-p+1]<0||s[n-p+1]==j)&&(p*2-1!=n||i==j)&&(x||i<=j)&&(y||i+j<=1))
					f[p][x][y]+=dfs(p+1,x||i<j,y||i+j==0);
	return f[p][x][y];
}
int main()
{
	scanf("%d%I64d",&n,&m);
	memset(s,-1,sizeof(s));
	memset(f,-1,sizeof(f));
	if(dfs(1,0,0)<=m)
	{
		printf("-1\n");
		return 0;
	}
	s[1]=0;
	for(int i=2;i<=n;i++)
	{
		memset(f,-1,sizeof(f));
		s[i]=0;
		LL tmp=dfs(1,0,0);
		if(tmp<=m)
			s[i]=1,m-=tmp;
	}
	for(int i=1;i<=n;i++)
		printf("%d",s[i]);
	printf("\n");
	return 0;
}
