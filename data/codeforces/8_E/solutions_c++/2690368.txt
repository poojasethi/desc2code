#include<cstdio>
#include<cstring>

typedef long long ll;

const int fin=0,maxn=62;
int n;
ll k;
int a[maxn];
ll f[maxn][2][2];

ll sol(int L,bool x,bool y){
  int R=n-L+1,i,j;
  ll &F=f[L][x][y];
  if(L>R)return 1;
  if(F!=-1)return F;
  for(F=0,i=0;i<2;++i)if(a[L]==i||a[L]==-1)
    for(j=0;j<2;++j)if(a[R]==j||a[R]==-1)
      if((L<R||i==j)&&(x||i<=j)&&(y||i<=1-j))
        F+=sol(L+1,x||i<j,y||i<1-j);
  return F;
}

int main(){
  if(fin){
    freopen("t.in","r",stdin);
    freopen("t.out","w",stdout);
  }
  int i;
  ll t;
  scanf("%d%I64d",&n,&k),++k;
  for(i=2;i<=n;++i)a[i]=-1;
  if(memset(f,-1,sizeof(f)),sol(1,0,0)<k)
    return puts("-1"),0;
  for(i=2;i<=n;++i){
    memset(f,-1,sizeof(f));
    a[i]=0;
    t=sol(1,0,0);
    if(t<k)a[i]=1,k-=t;
  }
  for(i=1;i<=n;++i)printf("%d",a[i]);
  return 0;
}
