#include <iostream>
#include <cstring>
using namespace std;
long long k,m,f[51][51][2][2];
bool v[51][51][2][2];
int n,a[51],i;
long long calc(int l,int r,bool rev,bool inv)
{
		 			if (l>r) return 1;
		 			if (v[l][r][rev][inv]) return f[l][r][rev][inv];
		 			v[l][r][rev][inv]=true;
		 			f[l][r][rev][inv]=0;
		 			int i,j;
		 			for (i=0;i<2;i++) 
					if (a[l]==-1 || a[l]==i)
		 			for (j=0;j<2;j++)
		 			if (a[r]==-1 || a[r]==j)
		 			if ((l<r || i==j)&&(rev || i<=j)&&(inv || i<=1-j))
		 				 f[l][r][rev][inv]+=calc(l+1,r-1,rev || i<j,inv || i<1-j);
 				  return f[l][r][rev][inv];
}
int main()
{
		cin >> n >> m,m++;
		memset(a,-1,sizeof(a));
		a[0]=0;
		if (calc(0,n-1,0,0)<m)
		{
			 cout << -1;
			 return 0;
		}
		for (i=1;i<n;i++)
		{
				memset(v,0,sizeof(v));
				a[i]=0;
				k=calc(0,n-1,0,0);
				if (k<m)
				{
					 m-=k;
					 a[i]=1;
				}
		}
		for (i=0;i<n;i++) cout << a[i];
		return 0;
}
