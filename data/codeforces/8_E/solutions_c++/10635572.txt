#include<cstdio>
#include<cstring>
int N,i;
long long K,tmp;
int A[52];
long long f[52][2][2];
long long Dp(const int &pos,const bool &f1,const bool &f2){
    int i,j;
    if(pos>N-pos+1) return 1;
    if(f[pos][f1][f2]!=-1) return f[pos][f1][f2];
    f[pos][f1][f2]=0;
    for(i=0;i<2;++i) if(A[pos]==i||A[pos]==-1)
        for(j=0;j<2;++j) if(A[N-pos+1]==j||A[N-pos+1]==-1)
            if((pos!=N-pos+1||i==j)&&(f1||i<=j)&&(f2||i<=1-j))
                f[pos][f1][f2]+=Dp(pos+1,f1||i!=j,f2||i!=1-j); 
    return f[pos][f1][f2];
}
int main(){
    scanf("%d%I64d",&N,&K);
    memset(A,-1,sizeof(A));memset(f,-1,sizeof(f));
    tmp=Dp(1,0,0);K++;
    if(tmp<K){printf("-1\n");return 0;}
    for(i=1;i<=N;++i){
        memset(f,-1,sizeof(f));
        A[i]=0;tmp=Dp(1,0,0);
        if(tmp<K)K-=tmp,A[i]=1;
    }
    for(i=1;i<=N;++i)printf("%d",A[i]);
    printf("\n");
    return 0;
}
