#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
long long int f[100][2][2],n,k; int s[100];
long long int calc(long long int x,long long int a,long long int b)
{
	if(f[x][a][b] > -1) return f[x][a][b];
	f[x][a][b] = 0;
	int y = n-x+1;
	if(x>y) return f[x][a][b] = 1;
	int i,j;
	for(i=0;i<2;i++)
	if(s[x]==-1 || s[x]==i)
	{
		for(j=0;j<2;j++)
			if(s[y]==-1 || s[y]==j)
			{		
				if(i==j || x<y)
				if(a || i<=j)
				if(b || i+j<=1)
					f[x][a][b] += calc(x+1,a||(i<j),b||((i+j)<=0));
			}
	}
	return f[x][a][b];
}
int main()
{
	//freopen("input.txt","r",stdin);
	int i;
	cin>>n>>k; k++;
	memset(f,-1,sizeof(f));
	memset(s,-1,sizeof(s));
	if(calc(1,0,0) < k)
	{
		printf("-1");
		return 0;
	}
	for(i=1;i<=n;i++)
	{
		memset(f,-1,sizeof(f));
		s[i] = 0; 
		if(calc(1,0,0) < k)
		{
			k -= calc(1,0,0);
			s[i] = 1;
		}
	}
	for(i=1;i<=n;i++)
		printf("%d",s[i]);
}
