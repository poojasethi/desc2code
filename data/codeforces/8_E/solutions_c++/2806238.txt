#include<cstdio>
#include<cstring>
#define fr(i,n) for(int i=0;i<n;i++)
#define ms(i,n) memset(i,n,sizeof i)
typedef long long ll;
ll n,k,f[60][2][2],s[60];
ll F(ll x,ll a,ll b)
{
    ll&g=f[x][a][b];
    if(x+x>=n||g!=-1) return g=(g==-1?1:g);g=0;
    fr(i,2)if(s[x]==-1||i==s[x])fr(j,2)if((x+x+1!=n||i==j)&&(s[n-x-1]==-1||j==s[n-x-1]))if((a||i<=j)&&(b||i+j<2))g+=F(x+1,a||i<j,b||i+j<1);
    return g;
}
int main()
{
    scanf("%I64d%I64d",&n,&k),k++,ms(f,-1),ms(s,-1);
    if(F(0,0,0)<k)return puts("-1"),0;
    fr(i,n)ms(f,-1),s[i]=0,s[i]=F(0,0,0)>=k?0:(k-=F(0,0,0),1);
    fr(i,n)putchar(s[i]+48);
}
