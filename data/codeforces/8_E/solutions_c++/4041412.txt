#include <cstdio>
long long k;
int n, ans[50], ansl;
int main() {
    scanf("%d%I64d", &n, &k);
    if (k >= (1LL << n-2) + (1LL << n/2-1)) {
        puts("-1");
        return 0;
    }
    int cur = 0;
    for (int i = 1; i <= n>>1; i++) {
        cur *= 2;
        long long now = ((1<<i)-1-2*cur) * (1LL << n-2*i) + (1LL << n/2-i);
        if (k >= now) cur++, ans[ansl] = 1, k -= now;
        ansl++;
    }
    int l = cur, r = (1<<n/2)-1-cur;
    if (n&1) {
        long long now = (1<<n/2)-2*cur;
        if (k >= now) ans[ansl] = 1, k -= now, r--;
        ansl++;
    }
    for (int i = 0; i < n>>1; i++) {
        int now = (r-l+1+!(r&1))/2;
        if (k >= now) {
            ans[ansl] = 1, k -= now;
            l = l/2;
            r = r/2 - !(r&1);
        } else {
            l = l/2 + (l&1);
            r = r/2;
        }
        ansl++;
    }
    for (int i = 0; i < n; i++) putchar(ans[i]+48);
    puts("");
    return 0;
}
