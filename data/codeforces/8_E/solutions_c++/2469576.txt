#include<cstdio>
#include<cstdlib>
#define ll long long

int i,j,BL,_BL,n;

long long Cnt,Rank;

char Str[60];

int Max(int a,int b){return a>b?a:b;}

int main()
{	scanf("%d%I64d",&n,&Rank);
	BL=_BL=1;
	for(i=0;i<n;i++)
	{	Str[i]='0';
		Cnt=0;
		for(j=n-2;j>i&&j>n-j-1;j--)
			if(!Str[n-j-1])
				Cnt+=2ll<<(n-2*(n-j)+(n-j-i-2));
			else if(Str[n-j-1]=='0')
				Cnt+=2ll<<(n-(i+1)-(n-j));	
		if(BL&&!(i>=n-i-1&&Str[i]=='0'&&Str[n-i-1]=='1'))Cnt+=1ll<<Max((n+1)/2-i-1,0);
		if(i!=n-1&&(_BL||(i>=n-i-1&&Str[i]=='0'&&Str[n-i-1]=='0')))Cnt+=1ll<<Max(n/2-i-1,0);
		if(!Cnt||Cnt<=Rank)
		{	Rank-=Cnt;
			Str[i]='1';
		}
		if(i>=n-i-1&&Str[i]=='0'&&Str[n-i-1]=='1')BL=0;
		if(i>=n-i-1&&Str[i]=='1'&&Str[n-i-1]=='1')_BL=0;
		if(i>=n-i-1&&Str[i]=='1'&&Str[n-i-1]=='0')BL=1;
		if(i>=n-i-1&&Str[i]=='0'&&Str[n-i-1]=='0')_BL=1;
	}
	if((BL||_BL)&&(Str[0]!='1'))printf("%s\n",Str);
		else printf("-1\n");
	return 0;
}
