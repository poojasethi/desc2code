#include<stdio.h>
#include<string.h>
typedef long long ll;
ll K;
int n;
int ans[100];
ll f[100][2][2];
ll calc(int L,bool x,bool y)
{
	ll &F=f[L][x][y];
	if(F!=-1)return F;
	int R=n-L+1;
	if(L>R)return 1;
	F=0;
	for(int i=0;i<2;i++)if(ans[L]==-1||ans[L]==i)
		for(int j=0;j<2;j++)if(ans[R]==-1||ans[R]==j)
			if((L<R||i==j)&&(x||i<=j)&&(y||i<=1-j))
				F+=calc(L+1,x||i<j,y||i<1-j);
	return F;
}
int main()
{
	scanf("%d%I64d",&n,&K);
	K++;
	memset(ans,-1,sizeof(ans));
	memset(f,-1,sizeof(f));
	ans[1]=0;
	ll tot=calc(1,0,0);
	if(tot<K)return puts("-1"),0;
	for(int i=1;i<=n;i++)
	{
		memset(f,-1,sizeof(f));
		ans[i]=0;
		ll x=calc(1,0,0);
		if(K>x)ans[i]=1,K-=x;
	}
	for(int i=1;i<=n;i++)printf("%d",ans[i]);
	putchar('\n');
	return 0;
}
