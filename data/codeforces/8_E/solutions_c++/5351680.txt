#include<iostream>
#include<cstring>
using namespace std;
typedef long long LL;
LL f[55][2][2],K;int a[55],n;
#define c(f) memset(f,-1,sizeof f)
#define rep(i,s,t) for(int i=s;i<=t;++i)
#define R return
LL C(int l,int r,int x,int y){
   if (l>r)R 1;LL&F=f[l][x][y];if(~F)R F;F=0;
   rep(i,0,1)rep(j,0,1)if(a[l]!=(!i)&&a[r]!=(!j)&&(l<r||i==j)&&(x||i<=j)&&(y||i<=(!j)))F+=C(l+1,r-1,x||(i<j),y||(i<!j));R F;
}
int main(){
    cin>>n>>K;c(a);c(f);
    if (C(1,n,a[1]=0,0)<++K){cout<<"-1";R 0;}
    rep(i,2,n){a[i]=0,c(f);LL d=C(1,n,0,0);K-=(a[i]=(d<K))*d;}
    rep(i,1,n)cout<<a[i];
}
