#include<cstdio>
#include<map>
#include<algorithm>
using namespace std;
const int maxn=100005;


map<int,int> ma[maxn];

int f[maxn],pre[maxn];
int num[maxn];
int n,v,c,l,r;
int ans=0,sum=0;
int main(){

#ifndef ONLINE_JUDGE
	freopen("1.in","r",stdin);
	freopen("1.out","w",stdout);
#endif

	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d%d%d%d",&v,&c,&l,&r);
		if(!l || ma[l].count(c+r)){
			int x=ma[l][c+r];
			f[i]=f[x]+v;
			pre[i]=x;
			if(f[ma[l+c][r]]<f[i]) 
				ma[l+c][r]=i;
			if(!r && f[ans]<f[i]) 
				ans=i;
		}
	}
	
	while (ans!=0){
		num[++sum]=ans;
		ans=pre[ans];
	}
	
	printf("%d\n",sum);
	for(int i=sum;i>=1;i--) 
		printf("%d ",num[i]);
}
