#include <cstdio>
#include <map>
using namespace std;
map<int,pair<int,int> > M[100001];
int ans,pre[100001],out[100001];
int main()
{
	//freopen("input.txt","r",stdin);
	int i,n,v,c,l,r,rec=0,tail=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d%d%d%d",&v,&c,&l,&r);
		if(l+r+c > 100000) continue;
		if(l==0 || M[l+r+c].find(l)!=M[l+r+c].end())
		{
			pair<int,int> box = M[l+r+c][l];
			pre[i] = box.second;
			if(box.first+v > M[l+r+c][l+c].first)
			{
				M[l+r+c][l+c].first = box.first+v;
				M[l+r+c][l+c].second = i;
				if(M[l+r+c][l+c].first > ans && r == 0)
				{
					ans = M[l+r+c][l+c].first;
					rec = i;
				}
			}
		}
	}
	for(;rec;rec=pre[rec]) out[++tail] = rec;
	printf("%d\n",tail);
	while(tail--)
		printf("%d ",out[tail+1]);
}
