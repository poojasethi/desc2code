#include<cstdio>
#include<cstring>
#include<map>
#include<vector>
#define maxn 100050
#define mp(x,y) make_pair((x),(y))
#define X first
#define Y second
using namespace std;

int     from[maxn],n;
map<pair<int,int>,int> mm,id;
map<pair<int,int>,int>::iterator iter;
vector<int> ans;

bool    updata( pair<int,int> p ,int v ,int i  ){
    if ( mm.find(p) == mm.end() || mm[p] < v ){
        mm[p] = v;
        id[p] = i; 
        return true;
    }
    return false;
}

int     main(){
    scanf("%d", &n );
    for (int i = 0; i<n; i++ ) {
        int v,c,x,y;
        scanf("%d%d%d%d", &v, &c , &x, &y );
        if ( x == 0 && updata( mp(x+c,y) , v , i ) ) from[i] = -1;
        if ( mm.find(mp(x,c+y)) != mm.end() && updata( mp(x+c,y), mm[mp(x,c+y)]+v,i) ) from[i] = id[mp(x,c+y)];
    }
    int ed = -1, key = -1;
    for ( iter = mm.begin(); iter != mm.end(); iter++)
        if ( iter->X.Y == 0 && iter->Y > key  ) {
            key = iter->Y;
            ed = id[iter->X];
        }
    while ( ed != -1 ) {
        ans.push_back(ed);
        ed = from[ed];
    }
    printf("%d\n" , (int)ans.size() );
    for (int i = ans.size()-1; i>=0; i--) printf("%d%c" , ans[i]+1 , i == 0?'\n':' ' );
    return 0;
}
