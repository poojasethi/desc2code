#include<cstdio>
#include<utility>
#include<map>
#define mp make_pair
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=101000;
int n,m,i,j,l,r,v,c;
int pre[mn],f[mn];
map<pair<int,int>,int>Q;
void upd(int&i,int&j){if(f[i]<f[j])i=j;}
int main()
{
    scanf("%d",&n);
    pair<int,int> T;
    fo(i,1,n)
    {
        scanf("%d%d%d%d",&v,&c,&l,&r);
        T=mp(l+r+c,l);
        if(l==0)f[i]=v;
        else if(Q.count(T))pre[i]=Q[T];
        else continue;
        f[i]=f[pre[i]]+v;
        if(!r&&f[i]>f[m])m=i;
        upd(Q[mp(l+r+c,l+c)],i);
    }
    for(l=0;m;m=pre[m])f[++l]=m;
    printf("%d\n",l);
    if(l)
    {
        printf("%d",f[l]);
        while(--l)printf(" %d",f[l]);
        printf("\n");
    }
    return 0;
}
