#include<cstdio>
#include<map>
using namespace std;
int n,v,c,l,r;
map<int ,int> ma[100010];
int f[100010],pre[100010];
int num[100010],ans=0,sum=0;
int main(){
    scanf("%d",&n);
    for (int i=1;i<=n;i++){
        scanf("%d%d%d%d",&v,&c,&l,&r);
        if (!l||ma[l].count(c+r)){//can match limits for prefix and postfix
            int x=ma[l][c+r];//ma[][] largest value can get from following which truck
            f[i]=f[x]+v;
            pre[i]=x;
            if (f[ma[l+c][r]]<f[i]) ma[l+c][r]=i;//note how the prefix current postfix change
            if (!r&&f[ans]<f[i]) ans=i;//end condition is that a truck no needs any thing following it ,and the whole set can gen largest value
        }
    }
    while (ans!=0){
        num[++sum]=ans;//roll back to form queue
        ans=pre[ans];
    }
    printf("%d\n",sum);
    for (int i=sum;i>1;i--) printf("%d ",num[i]);
    if (sum>=1) printf("%d\n",num[1]);
}
