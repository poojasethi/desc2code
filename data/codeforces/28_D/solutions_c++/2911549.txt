#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <map>
using namespace std;

const int maxn = 100005;
int f[maxn], pre[maxn], res[maxn];
map<int, int> tab[maxn];
int n, tot, ans;

int main() {
    int i, x, v, c, l, r;
    scanf("%d", &n);
    for (i = 1; i <= n; ++i) {
        scanf("%d%d%d%d", &v, &c, &l, &r);
        if (r + c < maxn && (!l || tab[l].count(r + c))) {
            x = tab[l][r + c]; f[i] = f[x] + v; pre[i] = x;
            if (r == 0 && f[i] > f[ans]) ans = i;
            if (f[i] > f[tab[l + c][r]]) tab[l + c][r] = i;
        }
    }
    for (; ans; ans = pre[ans]) res[tot++] = ans;
    printf("%d\n", tot);
    for (i = tot - 1; i >= 0; --i)
        printf("%d%c", res[i], i ? ' ' : '\n');
    return 0;
}
