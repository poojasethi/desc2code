#include<iostream>

using namespace std;

int main()
{	
	int n, m, k;

	cin >> n >> m >> k;

	if(n & 1)
	{
		int a, mi = 1 << 30;
		for(int i = 1; i <= n; ++i)
		{
			cin >> a;
			if(i & 1)
				mi = min(mi, a);
		}

		long long z = (long long)(m/((n + 1)/2))*(long long)k;
		cout << min((long long)(mi), z);
	}
	else
		cout << 0;

	return 0;
}
