#include<cstdio>
#define ll __int64
const int N=100000;
ll n,m,k,a[N];
ll min(ll a,ll b){return a<b?a:b;}
void solve()
{
	if(n&1)
	{
		ll ans=k*(m/(n/2+1));
		for(int i=0;i<n;i+=2)
			if(ans>a[i])ans=a[i];
			printf("%I64d\n",ans);
			return;
	}
	printf("0\n");
}
int main()
{
	scanf("%I64d%I64d%I64d",&n,&m,&k);
	for(int i=0;i<n;i++)
		scanf("%I64d",&a[i]);
	solve();
	return 0;
}
