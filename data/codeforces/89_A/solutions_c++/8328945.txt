#include <cstdio>
#include <algorithm>
using namespace std;
const int N=1e4+5;
int n,m,k,x,a[N];
int main(){
	scanf("%d%d%d",&n,&m,&k);
	if(n%2==0) return puts("0"),0;
	x=1e9+5;
	for(int i=1;i<=n;i++){
		scanf("%d",a+i);
		if(i&1) x=min(x,a[i]);
	}
	m/=n/2+1;
	printf("%d\n",(int)min(1LL*x,1LL*m*k));
}
