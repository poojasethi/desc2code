#include <iostream>
using namespace std;

long long n, m, k;
long long v[10010];

int main()
{
	cin >> n >> m >> k;
	for (int i = 0; i < n; ++i) cin >> v[i];
	
	if (n%2 == 0)
	{
		cout << 0 << endl;
		return 0;
	}
	
	long long p = (n+1)/2;
	long long q = m / p;
	
	long long ret = q*k;
	for (int i = 0; i < n; i += 2)
		ret = min(ret, v[i]);
	cout << ret << endl;
}
