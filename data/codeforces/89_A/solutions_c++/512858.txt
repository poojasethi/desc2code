#include<iostream>

using namespace std;

int n,m;
long long a[10010], ans, k;

int main(){
    cin>>n>>m>>k;
    for (int i=0; i<n; i++) cin>>a[i];

    if (n%2==0) cout<<0<<endl;
    else{
        ans = k*(m/((n+1)/2));
        for (int i=0; i<n; i+=2)
            ans = min(ans, a[i]);
        cout<<ans<<endl;
    }
}
