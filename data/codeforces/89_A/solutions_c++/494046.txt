#include <cstdio>
#include <algorithm>
#define LL long long int
using namespace std;

int n,m,k,a[10005];
LL M,K,AC=1LL<<60;



int main(){
	scanf("%d%d%d",&n,&m,&k);
	M=m; K=k;
	for (int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		if (i&1) AC=min((LL)a[i],AC);
	}
	if (n%2==0 || (n%2==1 && m<(n+1)/2)) printf("0\n");
	else printf("%d\n",min(AC,(M/((n+1)/2))*K));
	scanf("\n");
	return 0;
}
