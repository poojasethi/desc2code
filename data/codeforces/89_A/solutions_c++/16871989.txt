#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
	long long n,m,k;
	long long a[10005];
	long long c=1000000000;
	cin>>n>>m>>k;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		if(i%2==0)
		c=min(c,a[i]);
	}
	if (n%2==0) cout<<("0\n");
  else cout<<min((m/((n+1)/2))*k, c);
  return 0;
}
