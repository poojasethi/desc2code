#include<iostream>
#include<cstdio>

using namespace std;
long long a[10005];

int main()
{
    long long n,m,k,i,s;
    scanf("%I64d%I64d%I64d",&n,&m,&k);
    for (i=1; i<=n; i++) scanf("%I64d",&a[i]);
    if (n%2==0)
    {
        printf("0\n");
        return 0;
    }
    s=(m*2)/(n+1)*k;
    for (i=1; i<=n; i+=2)
     s=min(s,a[i]);
    printf("%I64d\n",s);
}
