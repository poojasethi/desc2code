#include <iostream>
#include <cstdio>
#include <string>
#include <vector>
using namespace std;
long long a[10010], n, m, k, p = -1;
int main()
{
  cin >> n >> m >> k;
  for(int i = 0; i < n; i++)
  {
    cin >> a[i];
    if (i % 2 == 0 && (p == -1 || p > a[i]))
      p = a[i];
  }
  if (n % 2 == 0)
    cout << 0;
  else
    cout << min(m / ((n + 1) / 2) * k, p);
  return 0;
}
