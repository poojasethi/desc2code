#include <iostream>
#define inf 1000000000
using namespace std;
typedef long long L;
int A[10001];

int main(){

   int n,m,k;
   cin>>n>>m>>k;
   for(int i=1;i<=n;i++)cin>>A[i];

   if(n%2==0){cout<<0<<endl;return 0;}



   int mini=inf;
   for(int j=1;j<=n;j+=2)mini=min(mini,A[j]);

   L res=(L)m/(L)(n/2+1)*(L)k;
   res=min(res,(L)mini);
   cout<<res<<endl;


   return 0;

}
