#include<bits/stdc++.h>
#define ll long long int
using namespace std;
ll a[1000000];
int main()
{
    ll n,m,k,i;
    cin>>n>>m>>k;
    ll mini=1e12;
    for(i=0;i<n;i++)
        cin>>a[i];
    if(n%2==0)
    {
        cout<<0<<endl;
        return 0;
    }
    for(i=0;i<n;i+=2)
        mini=min(mini,a[i]);
    cout<<min(((2*m)/(n+1))*k,mini);
    return 0;
}
