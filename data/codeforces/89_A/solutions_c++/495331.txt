#include<cstdio>
#include<iostream>
using namespace std;
int main() {
	int t;
	int n, m;
	long long int k;
	cin >> n >> m >> k;
	if(n%2 == 0) {
		printf("0\n");
	} else {
		k *= m / (n / 2 + 1);
		int sum;
		scanf("%d", &sum);
		for(int i = 1; i < n; i += 2) {
			int num;
			scanf("%*d%d", &num);
			sum = min(num, sum);
		}
		cout<<min((long long int)sum, k)<<endl;
	}
	return 0;
}
