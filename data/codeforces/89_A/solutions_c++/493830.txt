#include<cstdio>
inline long long min(long long A, long long B){return A < B? A: B;}
long long N, M, K, arr[10001];
int main(){
	scanf("%I64d%I64d%I64d", &N, &M, &K);
	for(int i = 0; i < N; i++)
		scanf("%I64d", &arr[i]);
	long long ans = 0, tmp = 9999999;
	for(int i = 0; i < N; i += 2)
		tmp = min(tmp, arr[i]);
	if(N % 2)ans = min(K * (M / (N / 2 + 1) ) , tmp);
	printf("%I64d\n", ans);
}
