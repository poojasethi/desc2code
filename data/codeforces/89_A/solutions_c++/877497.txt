#include <iostream>
#include <cmath>

using namespace std;

typedef long long int lint;

int main()
{
  lint n,m,k;
  cin>>n>>m>>k;
  if (n%2==0) {
    cout<<0<<endl;
  } else {
    lint minimo=1000000000;
    for (int i=0;i<n;i++) {
      lint x;
      cin>>x;
      if (i%2==0) minimo=min(minimo,x);
    }
    lint oper1=(n+1)/2;
    lint d1=m/oper1;
    lint d=d1*k;
    lint sol=min(d,minimo);
    cout<<sol<<endl;
  }
}
