#include<algorithm>
#include<iostream>
using namespace std;
long long d[10005];
int main(){
    long long n,m,k,c=0,m1=100000;
    cin>>n>>m>>k;
    while(c<n){
        cin>>d[c];
        if(c%2==0&&d[c]<m1)m1=d[c];c++;
    }c=0;
    if(n%2==0||m<n/2+1)c=0;
    else{
        if(n==1)c=min(d[0],m*k);
        else{
            c=min(m1,m/(n/2+1)*k);
        }
    }
    cout<<c;
    return 0;
}
