#include <iostream>
using namespace std;
long long n,m,k,a,ans;
int i;
int main()
{
    //ifstream cin("input.txt");
    //ofstream cout("output.txt");
	cin >> n >> m >> k;
	if (n%2==0) {cout << 0 << endl; return 0;}
	ans=m/(n/2+1)*k;
	for (i=1;i<=n;i++)
	{
		cin >> a;
		if (i%2==1 && ans>a) ans=a;
	}
	cout << ans << endl;
    return 0;
}
