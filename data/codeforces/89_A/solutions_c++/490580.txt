#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define int long long
int n,m,k;
int list[1000000];
main()
{
	ios::sync_with_stdio(false);
	cin>>n>>m>>k;
	for (int i=0;i<n;i++)
		cin>>list[i];
	if (n%2==0)
	{
		cout<<0<<endl;
		return 0;
	}
	int temp=m/((int)(n/2)+1);
	temp*=k;
	for (int i=0;i<n;i+=2)
		temp=min(temp,list[i]);
	cout<<temp<<endl;
	return 0;
}
