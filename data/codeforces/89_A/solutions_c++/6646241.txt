#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long n,m,k; cin>>n>>m>>k;
   long long f,mini=9999999;
   for(int i=0;i<n;i++)
   {
       cin>>f;
       if(i%2==0&&f<mini)mini=f;
   }
    if(n%2==0)
    {
        cout<<"0"<<endl;
        return 0;
    }

cout<<min(mini,k*(m/((n+1)/2)))<<endl;
return 0;}
