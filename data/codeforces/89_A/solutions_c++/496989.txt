#include<cstdio>
#include<algorithm>
using namespace std;
int x,n,m,k,ans;
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	int now=1000000;
	for (int i=0;i<n;i++)
	{
		scanf("%d",&x);
		if (i%2==0 && x<now) now=x;
	}
	ans=min((long long)n%2*now,(long long)(m/((n+1)/2))*k);
	printf("%d\n",ans);
	return 0;
}
