#include<bits/stdc++.h>
#define LL long long
using namespace std;
LL n,m,k;
LL a[10005];
int main()
{
    while(scanf("%I64d%I64d%I64d",&n,&m,&k)==3)
    {
        for(int i=1;i<=n;i++)scanf("%d",&a[i]);
        if(n%2==0){printf("0\n");continue;}
        LL Min=2*1e5;
        for(int i=1;i<=n;i+=2)Min=min(Min,a[i]);
        cout<<min(Min,m/(1+n/2)*k)<<endl;
    }
    return 0;
}
