#include <stdio.h>
#include <iostream>
using namespace std;
int v[10005];
int main() {
  int n,m,k;
  scanf("%d%d%d",&n,&m,&k);
  m/=(n/2+1);
  long long res = (long long)m*k;
  for(int i=0; i<n; ++i) {
    scanf("%d",v+i);
    if(i%2==0 && res > v[i]) res = v[i];
  }
  
  if(n%2==0) res=0;
  cout << res;
  return 0;
}
