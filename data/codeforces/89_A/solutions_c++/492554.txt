#include<cstdio>
#include<algorithm>
using namespace std;

int n,m,k;
int a[10010];

int main()
{
	scanf("%d%d%d",&n,&m,&k);
	if (n%2==0 || m<(n+1)/2)
	{
		printf("0\n");
		return 0;
	}
	for (int i=0;i<n;i++) 
	{
		scanf("%d",&a[i]);
		if (i%2==0) a[0]=min(a[0],a[i]);
	}
	//if (a[n-1]<a[0]) a[0]=a[n-1];
	int t=m/((n+1)/2);
	printf("%d\n",min((long long)a[0],(long long)k*t));
}
