#include<cstdio>
#define NN 100000
int v[NN],sol[NN];
int main(){
	int n;scanf("%d",&n);
	for(int i=0;i<n;++i)scanf("%d",v+i);
	sol[n-1]=1;
	int q=v[n-1];
	for(int i=n-2;i>=0;--i){
		if(q>=0){
			q-=v[i]; sol[i]=0;
		}else{
			q+=v[i]; sol[i]=1;
		}
	}
	if(q>0){
		for(int i=0;i<n;++i) 
			if(sol[i]) printf("+");
			else printf("-");
	}else{
		for(int i=0;i<n;++i)
			if(sol[i]) printf("-");
			else printf("+");
	}
	puts("");
	return 0;
}
