#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
const int N(1e5+200);
int n,a[N];
bool doubi[N];
int main()
{
    cin>>n;
    for(int i(1);i<=n;i++)
        cin>>a[i];
    long long now(0);
    for(int i(n);i>=1;i--)
    {
        if(now>0)doubi[i]=1,now-=a[i];
        else now+=a[i];
    }
    for(int i(1);i<=n;i++)
        putchar(now>0?(doubi[i]?'-':'+'):(doubi[i]?'+':'-'));
    return 0;
}
