#include <iostream>
#include <cstdio>
using namespace std;
int n;
int a[101000];
int b[101000];
long long sum=0;
int main()
{
    int flag=0;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    for(int i=n;i>=1;i--)
        if(a[i]>sum) b[i]=1,sum=a[i]-sum;
        else sum=sum-a[i];
    for(int i=1;i<=n;i++)
    {
        flag^=b[i];
        if(flag) printf("+");
        else printf("-");
    }   
    return 0;
}
