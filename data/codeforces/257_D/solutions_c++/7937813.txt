#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define N 100000
ll a[N],dp[N];
int main()
{
	ll n,i,c;
	cin>>n;
	for(i=0;i<n;++i)cin>>a[i];
	ll diff = 0;
	i = n -1 ;
	while(i>=0)
	{
		if(diff<a[i])
		{
			++c;
			dp[i]=1;
		}
		else
		{
			dp[i]=0;
		}
		diff = abs(diff-a[i]);
		--i;
	}
	char sign[2] = {'+','-'};
	int j = 1;
	for(i=0;i<n;++i)
	{
		j = (j+dp[i])%2;
		cout<<sign[j];
	}
	return 0;

}
