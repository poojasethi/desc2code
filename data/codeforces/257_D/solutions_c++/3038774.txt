#include <cstdio>

int n;
int a[100001];
int minus[100001];
int main() {
	scanf("%d",&n);
	for(int i=0;i<n;i++) scanf("%d",&a[i]);
	int sum=a[n-1];
	for(int i=n-2;i>=0;i--) {
		if(a[i]<sum) {
			minus[i]++;
			minus[i+1]++;
			sum=sum-a[i];
		} else {
			minus[i+1]++;
			sum=a[i]-sum;
		}
	}
	int pa=0;
	for(int i=0;i<n;i++) {
		pa+=minus[i];
		if(pa%2) {
			printf("-");
		} else {
			printf("+");
		}
	}
	printf("\n");
	return 0;
}
