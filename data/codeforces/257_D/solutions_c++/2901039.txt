#include <cstdio>

const int MaxN = 100005;
int n, a[MaxN], r[MaxN], s;

int main(void) {
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) scanf("%d", a+i);
    for (int i = n-1; i >= 0; --i)
        if (s > 0) s -= a[i];
        else r[i] = true, s += a[i];
    for (int i = 0; i < n; ++i)
        printf("%c", (r[i] ^ (s<0)) ? '+' : '-');
    
    return 0;
}
