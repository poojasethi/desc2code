#include<cstdio>
#define NMAX 100005
using namespace std;
int A[NMAX], Sol[NMAX], Sw[NMAX];

int main() {
	int n;
	scanf("%d", &n);
	for(int i = 1; i <= n; ++i)
		scanf("%d", A + i);
	
	int sum = A[n];
	for(int i = n - 1; i > 0; --i)
		if(sum >= A[i]) {
			Sol[i] = 1;
			sum -= A[i];
		} else {
			Sw[i] = 1;
			sum = A[i] - sum;
		}
	
	int sw = 0;
	for(int i = 1; i <= n; ++i) {
		if(Sol[i] ^ sw)
			printf("-");
		else
			printf("+");
		sw ^= Sw[i];
	}

	return 0;
}
