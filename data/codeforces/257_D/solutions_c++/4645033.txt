#include<iostream>
#include<cmath>
using namespace std;
main()
{
    int i,j,k,l,m;
    cin>>m;
    int a[m],b[m];
    for(i=0;i<m;i++)cin>>a[i];
    l=0;
    for(i=m-1;i>=0;i--)
    {
        k=l;
        j=l;
        k-=a[i]; j+=a[i];
        if(abs(k)>abs(j)){b[i]=1; l+=a[i];} else {b[i]=0; l-=a[i];}
    }
    if(l<0) k=1;
    else k=0;
    for(i=0;i<m;i++) if((b[i]+k)%2) cout<<'+'; else cout<<'-';
}
