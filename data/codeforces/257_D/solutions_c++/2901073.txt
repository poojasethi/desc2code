# include <stdio.h>
# include <iostream>

using namespace std;

int n,m;
int t[100009];
bool v[100009];
int k[100009];

int main()
{
	scanf("%d",&n);
	
	for(int h=0; h<n; h++)
		scanf("%d",&t[h]);
	
	m=t[n-1];
	v[n-1]=1;
	for(int h=n-2; h>=0; h--)
	{
		if(m-t[h]>=0)
			v[h]=0,m-=t[h];
		else
			k[h]=1,v[h]=1,m=t[h]-m;
	}
	
	m=0;
	
	for(int h=0; h<n; h++)
	{
		if((v[h]==1 && m%2==0) || (v[h]==0 && m%2==1))
			printf("+");
			
		else
			printf("-");
		m+=k[h];
	}
}
