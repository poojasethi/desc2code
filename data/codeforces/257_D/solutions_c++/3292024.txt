#include <iostream>

using namespace std;

const int max_n = 1e5 + 15;

int n, s, niz[max_n], sgn[max_n];

int main()
{
  cin >> n;
  for (int i = 0; i < n; i++) cin >> niz[i];
  for (int i = n - 1; i >= 0; i--)
    if (s > 0) s -= niz[i], sgn[i] = 1;
    else s += niz[i];
  for (int i = 0; i < n; i++)
    cout << (sgn[i] ^ (s > 0) ? '+' : '-');
  return 0;
}
