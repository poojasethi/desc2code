#include<stdio.h>
int ar[100000];
bool ans[100000];
int main(){
    int n,i;
    scanf("%d",&n);
    for(i=0;i<n;i++)scanf("%d",&ar[i]);
    int res=0;
    for(i=n-1;i>=0;i--){
        if(res>=0){
            ans[i]=false;
            res-=ar[i];
        }
        else{
            ans[i]=true;
            res+=ar[i];
        }
    }
    bool rev=res<0;
    for(i=0;i<n;i++)printf("%c",ans[i]^rev?'+':'-');
    return 0;
}
