#include<cstdio>
using namespace std;
#define MAXN 100010
int n,a[MAXN],v[MAXN]={};
long long s=0;
main(){
    scanf("%d",&n);
    for(int i=0;i<n;++i)
        scanf("%d",a+i);
    for(int i=n;i--;)
        if (a[i]>s)
            v[i]=1,s=a[i]-s;
        else
            s-=a[i];
    int t=0;
    for(int i=0;i<n;++i)
        printf((t^=v[i])?"+":"-");
}
