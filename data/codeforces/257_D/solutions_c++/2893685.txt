#include<cstdio>
using namespace std;
const int NMAX = 100005;
int n,a[NMAX],i,sum;
char str[NMAX];
int main()
{
    scanf("%d",&n);
    for(i=1;i<=n;i++) scanf("%d",&a[i]);
    for(i=n;i>=1;i--)
        if(sum<0) {sum+=a[i];str[i]='+';}
        else {sum-=a[i];str[i]='-';}
    if(sum<0)
        for(i=1;i<=n;i++)
            if(str[i]=='+') str[i]='-';
            else str[i]='+';
    printf("%s\n",str+1);
    return 0;
}
