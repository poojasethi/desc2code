#include<cstdio>
#include<vector>
using namespace std;
int n,a[100005],i,sum;
char str[100005];
int main()
{
    scanf("%d",&n);
    for(i=1;i<=n;i++) scanf("%d",&a[i]);
    for(i=n;i>=1;i--)
    {
        if(sum<0) {sum+=a[i];str[i]='+';}
        else {sum-=a[i];str[i]='-';}
    }
    if(sum<0)
        for(i=1;i<=n;i++)
            if(str[i]=='+') str[i]='-';
            else str[i]='+';
    printf("%s\n",str+1);
    return 0;
}
