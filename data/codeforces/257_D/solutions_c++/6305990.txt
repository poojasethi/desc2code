#include<cstdio>
using namespace std;
int a1[100000];
bool b1[100000];
int main()
{
	int n,a;
	scanf("%d",&n);
	for(int i=0;i<n;++i)
		scanf("%d",a1+i);
	a=a1[n-1];b1[n-1]=1;
	for(int i=n-2;~i;--i)
	if(a1[i]>a)
	{a=a1[i]-a;b1[i]=1;}
	else
	{a-=a1[i];b1[i]=0;}
	a=1;
	for(int i=0;i<n;++i)
	{
		a^=b1[i];
		printf(a?"-":"+");
	}
	puts("");
}
