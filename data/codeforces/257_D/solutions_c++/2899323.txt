#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
#define N 100005
string ans;
int n,q[N];
int main(){
    cin>>n;
    for(int i=0;i<n;i++) cin>>q[i];
    int k=0;
    for(int i=n-1;i>=0;i--){
        if(k<0) ans+='+',k+=q[i];
        else ans+='-',k-=q[i];
    }
    reverse(ans.begin(),ans.end());
    if(k<0) for(int i=0;i<n;i++) ans[i]=(ans[i]=='+'?'-':'+');
    cout<<ans<<endl;
}
