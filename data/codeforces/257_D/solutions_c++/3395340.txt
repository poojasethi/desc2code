#include <iostream>
#include<string.h>
using namespace std;

int n,a[100010],s,r[100010];

int main(){
  cin>>n;
  s=0;
  memset(r,0,sizeof(r));
  for(int i=1;i<=n;i++)cin>>a[i];
  for(int i=n;i>=1;i--){
    if(s>0)s-=a[i];
    else {r[i] = 1;s+=a[i];}
  }
  for(int i=1;i<=n;i++)cout<<((((s>0)+r[i])%2)?'-':'+'); cout<<endl;

}
