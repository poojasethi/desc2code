#include <cstdio>
typedef __int64 ll;
const int N = 100005;
int f[N];
ll a[N];
char x[2]={'+','-'};

int main() {
	int n, i;
	scanf("%d", &n);
	for(i=1; i<=n; i++)
		scanf("%I64d", &a[i]);
	
	ll s = a[n];
	f[n] = 0;

	for(i=n-1; i>=1; i--) {
		if(s >= 0) {
			s -= a[i];
			f[i] = 1;
		}
		else {
			s += a[i];
			f[i] = 0;
		}
	}
	int flag= s>0?0:1;

	for(i=1; i<=n; i++) 
		printf("%c", x[(flag+f[i])%2]);
	puts("");			
	return 0;
}
