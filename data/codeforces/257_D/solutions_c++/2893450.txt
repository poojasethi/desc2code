#include<iostream>
using namespace std;

int a[100005],b[100005];

int main()
{
    int n,i;
    cin>>n;
    for(i=1;i<=n;i++) cin>>a[i];
    b[n]=1;
    int s=a[n];
    for(i=n-1;i>0;i--)
    {
        if(s<a[i]) b[i]=1,s=a[i]-s;
        else b[i]=0,s=s-a[i];
    }
    int c=0;
    for(i=1;i<=n;i++)
    {
        if((c+b[i])%2) cout<<"+";
        else cout<<"-";
        c+=b[i];
    }
    cout<<endl;
    return 0;
}
