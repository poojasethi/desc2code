#include <iostream>

using namespace std;

int n,s;

int main(){
  cin>>n;
int*a=new int[n*100];
int*r=new int[n*100];
  for(int i=1;i<=n;i++)cin>>a[i];
  for(int i=n;i>=1;i--){
    if(s>0)s-=a[i];
    else {r[i] = 1;s+=a[i];}
  }
  for(int i=1;i<=n;i++)cout<<((((s>0)+r[i])%2)?'-':'+'); cout<<endl;

}
