#include <cstdio>
#define FOR(i,s,e) for (int i=(s); i<(e); i++)
#define FOD(i,s,e) for (int i=(s)-1; i>=(e); i--)
#define N 100005

int n, s, a[N];
char c[N];

int main(){
	scanf("%d", &n);
	FOR(i,0,n) scanf("%d", &a[i]);
	FOD(i,n,0){
		if (s + a[i] <= a[i]) s += a[i], c[i] = '+';
		else s -= a[i], c[i] = '-';
	}
	if (s < 0) FOR(i,0,n) c[i] = (c[i] == '+' ? '-' : '+');
	printf("%s\n", c);
	return 0;
}
