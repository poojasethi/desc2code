#include <cstdio>
const int N=100100;
int A[N];
bool ans[N];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d",&A[i]);
    int now=0;
    for(int i=n;i>=1;i--)
        if(now<0){now+=A[i],ans[i]=1;}
        else {now-=A[i],ans[i]=0;}
    if(now<0)for(int i=1;i<=n;i++)ans[i]=!ans[i];
    for(int i=1;i<=n;i++)
        putchar(ans[i]?'+':'-');
    puts("");
    scanf(" ");
}
