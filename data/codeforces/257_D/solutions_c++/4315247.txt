#include<cstdio>
#include<cstdlib>
#define MAXN 100010
int a[MAXN],s,opt[MAXN],n;
int main()
{
    scanf("%d",&n);
    for (int i=1;i<=n;i++)scanf("%d",&a[i]);
    for (int i=n;i>0;i--)
        if (s>0)s-=a[i];
        else    s+=a[i],opt[i]=1;
    if (s<0)for (int i=1;i<=n;i++)opt[i]^=1;
    for (int i=1;i<=n;i++)
        opt[i]==1?printf("+"):printf("-");
    printf("\n");
//    system("pause");
    return 0;
}
