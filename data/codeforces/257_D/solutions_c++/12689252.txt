#include <iostream>

using namespace std;

int n,s;

int main(){
  cin>>n;
int*g=new int[n*11];
int*r=new int[n*11];
  for(int i=1;i<=n;i++)cin>>g[i];
  for(int i=n;i>=1;i--){
    if(s>0)s-=g[i];
    else {r[i] = 1;s+=g[i];}
  }
  for(int i=1;i<=n;i++)cout<<((((s>0)+r[i])%2)?'-':'+'); cout<<endl;

}
