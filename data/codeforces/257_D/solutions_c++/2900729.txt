#include <iostream>

using namespace std;

long long mas[100000]; int d[100000];

int main(int argc, char *argv[])
{int n,i,L; long long D;
 cin>>n;
 for(i=0;i<n;i++)
 cin>>mas[i];
 for(i=n-2,D=mas[n-1],d[n-1]=1;i>=0;i--)
 if(D>0)D-=mas[i],d[i]=0; else D+=mas[i],d[i]=1;
 L=D<0;
 for(i=0;i<n;i++)
 if(L ^ d[i])cout<<'+'; else cout<<'-';
    
    return EXIT_SUCCESS;
}
