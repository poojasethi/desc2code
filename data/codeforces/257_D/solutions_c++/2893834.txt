#include <stdio.h>
const int maxn = 100010;
int n;
int a[maxn];
int v[maxn];
int main()
{
    int n,i,j;
    scanf("%d",&n);
    for(i = 0; i < n; i++) scanf("%d",&a[i]);
    int s = 0;
    for(i = n - 1; i >=0; i--)
    {
        if(a[i] >= s)  v[i] = 1,s = a[i] - s;
        else s -= a[i];
    }
    int f = 0;
    for(i = 0; i < n; i++)
    {
        f ^= v[i];
        printf("%c",f?'+':'-');
    }
    return 0;
}
