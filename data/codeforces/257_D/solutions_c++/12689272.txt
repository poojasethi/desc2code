#include <iostream>

using namespace std;

int n,s=0;

int main(){
  cin>>n;
int*t=new int[n*10];
int*w=new int[n*10];
  for(int i=1;i<=n;i++)cin>>t[i];
  for(int i=n;i>=1;i--){
    if(s>0)s-=t[i];
    else {w[i] = 1;s+=t[i];}
  }
  for(int i=1;i<=n;i++)cout<<((((s>0)+w[i])%2)?'-':'+'); cout<<endl;

}
