#!/usr/bin/python

n = int(raw_input())
a = map(int, raw_input().split())

sign, s =[' ']*n, 0

for i in xrange(n-1,-1,-1):
	if s > a[i]:
		s-=a[i];
		sign[i] = '-'
	else:
		s = a[i] - s
		sign[i] = '+'

ans = ''
negate = False
for i in xrange(n):
	if negate:
		if sign[i] == '-':
			ans+='+'
		else:
			ans+='-'
	else:
		ans+=sign[i]

	if sign[i] == '+':
		negate = not negate

print ans
