n = input()
num = map(int, raw_input().split())
Sum = 0; ans=[]
for i in xrange(n-1, -1, -1):
    if Sum<=0:
        Sum += num[i]
        ans.append('+') 
    else :
        Sum -= num[i]
        ans.append('-')
if Sum >= 0:
    print ''.join(reversed(ans))
else :
    tmp=[]
    for i in reversed(ans):
        if i =='-':  tmp.append('+')
        else :  tmp.append('-')
    print ''.join(tmp)
