raw_input()
A = map(int, raw_input().split())
i = len(A) - 1
Sum = A[i];
i -= 1
res = "+"
while i >= 0:
    if Sum < 0:
        Sum += A[i]
        res += '+'
    else:
        Sum -= A[i]
        res += '-'
    i -= 1
if Sum < 0:
    ret = ""
    for i in res:
        if i == '+':
            ret += '-'
        else:
            ret += '+'
    res = ret
print res[::-1]
