n = int(raw_input())
a = sorted(list(enumerate(map(int, raw_input().split(" ")))), key = lambda tup: -tup[1])

set1 = set([])
set2 = set([])
sum1 = 0
sum2 = 0

for i in a:
    if sum1 < sum2:
        set1.add(i[0])
        sum1 += i[1]
    else:
        set2.add(i[0])
        sum2 += i[1]
        
char1, char2 = ('+', '-') if sum1 > sum2 else ('-', '+')

def getc(i):
    if i in set1: return char1
    else: return char2
    
ans = [getc(i) for i in range(n)]
print ''.join(ans)
