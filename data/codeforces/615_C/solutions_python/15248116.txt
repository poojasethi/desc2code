s = raw_input()
pat = raw_input()
r_s = s[::-1]
pat=pat+"#"
ans=[]
temp=pat[0]
n=len(s)
for i in xrange(1,len(pat)):
	cur = temp+pat[i]
	if cur in s or cur in r_s:
		temp=cur
	else:
		if temp in s:
			start = s.find(temp)
			ans.append((start+1,start+len(temp)))
		elif temp in r_s:
			start = r_s.find(temp)
			ans.append((n-start,n-start-len(temp)+1))
		else :
			print -1
			exit(0)
		temp=pat[i]
print len(ans)
for i in xrange(len(ans)):
	print ans[i][0],ans[i][1]
