a = raw_input()
b = raw_input()

ra = a[::-1]
ans=[]

b=b+"."
aux=b[0]

for i in xrange(1,len(b)):
	act = aux+b[i]
	if act in a or act in ra:
		aux=act
	else:
		if aux in a:
			ans.append((a.find(aux)+1,a.find(aux)+len(aux)))
		elif aux in ra:
			ans.append((len(a)-ra.find(aux),len(a)-ra.find(aux)-len(aux)+1))
		else :
			print -1
			exit(0)
		aux=b[i]

print len(ans)

for i in xrange(len(ans)):
	print ans[i][0],ans[i][1]
