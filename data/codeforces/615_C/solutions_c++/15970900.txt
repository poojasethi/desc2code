#include<iostream>
#include<cstring>
#include<cstdlib>
#include<bits/stdc++.h>
using namespace std;
char a[22016],b[22016],ss[22016];
int ans[2216][2];
int main() {
	int l1,l2;
	while(cin>>a>>b) {
		strcpy(ss,a);
		strrev(ss);
		l1=strlen(a);
		l2=strlen(b);
		int num=0;
		for(int i=0; i<l2;) {
			int np=-1;
			int p1,p2;
			for(int j=0; j<l1; j++) {
				int k=j;
				int next=i;
				while(next<l2&&a[k]==b[next]) next++,k++;
				if(next>np) {
					np=next,p1=j+1,p2=k;
				}
			}
			for(int j=0; j<l1; j++) {
				int k=j, next=i;
				while(next<l2&&ss[k]==b[next]) next++,k++;
				if(next>np) {
					np=next,p1=l1-j,p2=l1-k+1;
				}
			}
			if(i==np) {
				cout<<"-1"<<endl;
				return 0;
			}
			i=np,ans[num][0]=p1,ans[num++][1]=p2;
		}
		cout<<num<<endl;
		for(int i=0; i<num; i++)
			cout<<ans[i][0]<<" "<<ans[i][1]<<endl;

	}
	return 0;
}
