#include <bits/stdc++.h>

using namespace std;

char s[2110], t[2110];
vector <int> a[26];
int main()
{
	scanf("%s%s",&s,&t);
	int n=strlen(s), m=strlen(t);
	for(int i=0;i<n;i++)
	{
		a[s[i]-'a'].push_back(i);
	}
	int ind=0, match=0, c, temp, ans[2110][2], index=0, tempM=0, flag=0;
	while(ind<m)
	{
		match=0;
		temp=ind;
		c=t[ind]-'a';
		for(int i=0;i<a[c].size();i++)
		{
			int j=a[c][i];
			temp=ind;
			tempM=0;
			while(j<n && temp<m && s[j] == t[temp])
			{
				tempM++;
				temp++;
				j++;
			}
			if(match<tempM)
			{
				match = tempM;
				ans[index][0] = a[c][i];
				ans[index][1] = j-1;
			}
			match = max(match, tempM);
			j = a[c][i];
			temp = ind;
			tempM=0;
			while(j>=0 && temp<m && s[j] == t[temp])
			{
				tempM++;
				temp++;
				j--;
			}
			if(match<tempM)
			{
				match = tempM;
				ans[index][0] = a[c][i];
				ans[index][1] = j+1;
			}
		}
		if(match<=0){
			flag=1;
			break;
		}
		ind = ind + match;
		index++;
	}
	if(flag==1){
		printf("-1");
	} else {
		printf("%d\n", index);
		for(int i=0;i<index;i++)
		{
			printf("%d %d\n", ans[i][0]+1, ans[i][1]+1);
		}
	}
	return 0;
}
