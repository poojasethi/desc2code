#include <cstdio>
#include <cstring>
#include <string>
#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
#include <queue>
using namespace std;
typedef pair<int,int> P;

string s,t;

int lcp[2200][2200],lcprev[2200][2200];

int main(void){
	cin >> s;
	cin >> t;
	int n=s.size();
	int m=t.size();
	for(int i=m-1;i>=0;i--){
		for(int j=n-1;j>=0;j--){
			if(t[i]==s[j])lcp[i][j]=lcp[i+1][j+1]+1;
		}
	}
	reverse(s.begin(),s.end());
	for(int i=m-1;i>=0;i--){
		for(int j=n;j>=0;j--){
			if(t[i]==s[j])lcprev[i][j]=lcprev[i+1][j+1]+1;
		}
	}
	int cnt=0;
	vector<P> res;
	int f=0;
	while(f<m){
		int len=0;
		P p=P(0,0);
		for(int i=0;i<n;i++){
			if(lcp[f][i]>len){
				len=lcp[f][i];
				p=P(i+1,i+len);
			}
			if(lcprev[f][i]>len){
				len=lcprev[f][i];
				p=P(n+1-(i+1),n+1-(i+len));
			}
		}
		if(len==0){
			printf("-1\n");
			return 0;
		}
		f+=len;
		cnt++;
		res.push_back(p);
	}
	printf("%d\n",cnt);
	for(int i=0;i<cnt;i++){
		printf("%d %d\n",res[i].first,res[i].second);
	}
	return 0;
}
