#include<cstdlib>
#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
const int maxn = 2105;
char a[maxn],b[maxn];
int ans[maxn][2],n;
int main(){
	scanf("%s%s",a,b);
	int lenA,lenB;
	n=0;
	lenA= strlen(a);
	lenB= strlen(b);
	int mat = 0;
	while(mat<lenB){
		int matchs = 0;
		for(int i=0;i<lenA;i++){
			int j;
			for(j=0;j+mat<lenB&&i+j<lenA;j++){
				if(b[j+mat]!=a[i+j]){
					break;
				}
			}
			if(j>matchs){
				matchs=j;
				ans[n][0]=i+1;
				ans[n][1]=i+j;
			}
		}
		for(int i=lenA-1;i>=0;i--){
			int j;
			for(j=0;j+mat<lenB&&i-j>=0;j++){
				if(b[j+mat]!=a[i-j]){
					break;
				}
			}
			if(j>matchs){
					matchs = j;
					ans[n][0]=i+1;
					ans[n][1]=i+2-j;
			}
		}
		// printf("---%d\n",matchs);
		// printf("%d %d\n",ans[n][0],ans[n][1]);
		if(matchs==0){
			break;
		}
		mat += matchs;
		n++;
	}
	if(mat<lenB){
		puts("-1");
	}else {
		printf("%d\n",n);
		for(int i=0;i<n;i++){
			printf("%d %d\n",ans[i][0],ans[i][1]);
		}
	}
	return 0;

}
