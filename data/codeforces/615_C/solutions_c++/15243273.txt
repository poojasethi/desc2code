#include <bits/stdc++.h>
using namespace std;
int cnt=0;
typedef pair<int,int> ii;
vector <ii> v;
string str1,str2,str3;
char str[3005];
int main(){
	gets(str);
	str1=string(str);
	int n=str1.length();
	str2=str1;
	reverse(str2.begin(),str2.end());
	gets(str);
	str3="";
	int l=strlen(str);
	v.push_back(ii(0,0));
	for(int x=0;x<l;x++){
		str3+=str[x];
		size_t i=str1.find(str3);
		if(i!=string::npos){
			v[cnt]=ii(int(i)+1,int(i)+str3.length());
			continue;
		}
		i=str2.find(str3);
		if(i!=string::npos){
			v[cnt]=ii(n-int(i),n-int(i)-str3.length()+1);
			continue;
		}
		if(str3.length()==1){
			printf("-1\n");
			return 0;
		}
		cnt++;
		v.push_back(ii(0,0));
		str3="";
		x--;
	}
	printf("%d\n",cnt+1);
	for(int x=0;x<=cnt;x++){
		printf("%d %d\n",v[x].first,v[x].second);
	}
	return 0;
}
