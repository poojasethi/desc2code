#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

vector <pair <int, int> > vec;

int main()
{
	string s, t;
	cin>> s >> t;

	for(int i = 0; i < t.size(); i ++)
	{
		int u = -1, v = -1, x = 0;
		for(int j = 0; j < s.size(); j ++)
		{
			int k = i, h = j;
			if(s[h] == t[k])
			{
				while(h < s.size() && k < t.size() && s[h] == t[k])
				{
					h ++;
					k ++;
				}

				h --;
				if(abs(h - j) + 1 > x)
				{
					u = j;
					v = h;
					x = abs(h - j) + 1;
				}

				k = i;
				h = j;
				while(h >= 0 && k < t.size() && s[h] == t[k])
				{
					h --;
					k ++;
				}

				h ++;
				if(abs(h - j) + 1 > x)
				{
					u = j;
					v = h;
					x = abs(h - j) + 1;
				}
			}

		}

		if(u == -1)
		{
			cout<< "-1\n";
			return 0;
		}
		vec.push_back(make_pair(u, v));
		i += abs(v - u);
	}	

	cout<< vec.size() << endl;
	for(int i = 0; i < vec.size(); i ++)
		cout<< vec[i].first + 1 << " " << vec[i].second + 1 << endl;


}
