#include<iostream>
#include<stdio.h>
#include<vector>
#include<utility>
#include<string>
#include<bits/stdc++.h>
using namespace std;
string s1,s2,t,a;
vector<pair<int,int> > ans;
int main()
{
	cin>>s1>>s2;
	t=s1;
	int n=s1.size();
	reverse(t.begin(),t.end());
	int x,y;
	for(int i=0;i<s2.size();i++)
	{
		a+=s2[i];
		int f=s1.find(a);
		if(f<0)
		 f=t.find(a);
		else
		{
			x=f+1,y=x+a.size()-1;
			continue;
		}
		if(f<0)
		{
			if(a.size()==1) {cout<<-1<<endl; return 0;}
			else
			{
				ans.push_back(make_pair(x,y));
				i--,a.clear();
			}
		}
		else
		{
			x=f+1,y=x+a.size()-1,x=n+1-x,y=n+1-y;
		}
	}
	ans.push_back(make_pair(x,y));
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++)
	{
		printf("%d %d\n",ans[i].first,ans[i].second);
	}
}
