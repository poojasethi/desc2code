#include<cstdio>
#include<cstring>
#include<cmath>
#include<ctime>
#include<cstdlib>
typedef long long ll;
typedef double lf;
const int INF=1<<30;
inline int read(){
	int x=0;bool f=false;char ch=getchar();
	while(ch<'0'||ch>'9'){if(ch=='-')f=true;ch=getchar();}
	while(ch>='0'&&ch<='9'){x=(x<<1)+(x<<3)+ch-'0';ch=getchar();}
	if(f)return -x;return x;
}
const int MAXN=3000,D=50;
int n,m;
char ch[MAXN+D],str[MAXN+D];
int st[MAXN+D],ed[MAXN+D],tot;
int main(){
	//freopen("in","r",stdin);
	scanf("%s%s",ch+1,str+1);
	n=strlen(ch+1),m=strlen(str+1);
	for(int now=1;now<=m;now++){
		int a,b,len=-1;bool flag=false;
		for(int i=1;i<=n;i++)
			if(ch[i]==str[now]){
				int l;flag=true;
				for(l=1;i+l<=n&&now+l<=m;l++)
					if(ch[i+l]!=str[now+l])
						break;
				l--;
				if(l>len){len=l,a=i,b=i+l;}
				for(l=1;i-l>0&&now+l<=m;l++)
					if(ch[i-l]!=str[now+l])
						break;
				l--;
				if(l>len){len=l,a=i,b=i-l;}
			}
		if(flag==false){printf("-1");return 0;}
		st[++tot]=a,ed[tot]=b;
		now+=len;
	}
	printf("%d\n",tot);
	for(int i=1;i<=tot;i++)
		printf("%d %d\n",st[i],ed[i]);
	return 0;
}
