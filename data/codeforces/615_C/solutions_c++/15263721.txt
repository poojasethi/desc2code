#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int len,L,N,Pre[5805];
char s[5805],t[5805];
int ans,d[5985][3];
int main()
{	

	scanf("%s",s+1);
	len=strlen(s+1);	L=len;
	s[len+1]='#';
	for (int i=1;i<=len;++i) s[i+len+1]=s[len-i+1];
	len=len*2+1;
	scanf("%s",t+1);
	N=strlen(t+1);
	int st=0,X,Y;
	for (int i=1,j=0;i<=N;) {
		if (i-st>1) {
			//j=max(j,st);
			while (j>st&&t[j+1]!=t[i]) j=Pre[j];
			if (t[j+1]==t[i]) ++j;
			Pre[i]=j;
		}
		bool boo=0;
		int k,l,last;
		for (k=1,l=st;k<=len;++k) {
			//l=max(l,st);
			while (l>st&&t[l+1]!=s[k]) l=Pre[l];
			if (t[l+1]==s[k]) ++l;
			if (l==i) {
				last=k-i+st+1;
				boo=1; break;
			}
		}
		if (!boo) {
			if (i-st==1) { ans=-1; break; }
			d[++ans][0]=X; d[ans][1]=Y;
			st=j=i-1;	Pre[i]=st;
		} else {
			++i;			
			if (last>L) {
				last=len-last+1; k=len-k+1;
			}
			X=last; Y=k;
		}
	}
	if (ans>=0) d[++ans][0]=X; d[ans][1]=Y;
	printf("%d\n",ans);
	for (int i=1;i<=ans;++i) {
		printf("%d %d\n",d[i][0],d[i][1]);
	}
}
