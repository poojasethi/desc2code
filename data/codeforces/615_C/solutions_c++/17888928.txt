#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,tot,ans[2200][2];char a[2200],b[2200],c[2200];
int main(){
	scanf("%s%s",a,c);n=strlen(a);m=strlen(c);
	for(int i=0;i<n;++i)b[i]=a[i];reverse(b,b+n);
	for(int i=0;i<m;){
		int nxt=-1,l,r;
		for(int j=0;j<n;++j){
			int k=j,tmp=i;
			while(tmp<m&&a[k]==c[tmp])++k,++tmp;
			if(tmp>nxt)nxt=tmp,l=j+1,r=k;
		}
		for(int j=0;j<n;++j){
			int k=j,tmp=i;
			while(tmp<m&&b[k]==c[tmp])++k,++tmp;
			if(tmp>nxt)nxt=tmp,l=n-j,r=n-k+1;
		}
		if(i==nxt)return puts("-1"),0;
		i=nxt,ans[tot][0]=l,ans[tot++][1]=r;
	}
	printf("%d\n",tot);for(int i=0;i<tot;++i)printf("%d %d\n",ans[i][0],ans[i][1]);
}
