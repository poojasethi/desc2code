#include<cstdio>
#include<cstring>
const int N=3010;
int n,m,i,j,k,l,L,R,flag,ans,tmp[N][2],f[N],g[N];
char a[N],b[N];
int main(){
  scanf("%s%s",a+1,b+1);
  n=strlen(a+1),m=strlen(b+1);
  for(i=1;i<=m;i=j){
    for(k=1;k<=n;k++)f[k]=g[k]=1;
    for(l=0,j=i;j<=m;j++){
      l++;flag=0;
      for(k=1;k<=n;k++){
        if(f[k]){
          if(k+l-1>n)f[k]=0;
          else if(a[k+l-1]!=b[j])f[k]=0;
          else{
            flag=1;
            L=k;
            R=k+l-1;
          }
        }
        if(g[k]){
          if(k-l+1<1)g[k]=0;
          else if(a[k-l+1]!=b[j])g[k]=0;
          else{
            flag=1;
            L=k;
            R=k-l+1;
          }
        }
      }
      if(!flag)break;
    }
    if(i==j)return puts("-1"),0;
    ans++;
    tmp[ans][0]=L,tmp[ans][1]=R;
  }
  printf("%d\n",ans);
  for(i=1;i<=ans;i++)printf("%d %d\n",tmp[i][0],tmp[i][1]);
  return 0;
}
