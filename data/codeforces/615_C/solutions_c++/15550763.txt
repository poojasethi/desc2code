#include <bits/stdc++.h>
using namespace std;
string s,t,sr,comp;
int x,y;
std::vector<pair<int,int> >ans;
int main()
{
	cin>>s>>t;
	sr=s;
	reverse(sr.begin(),sr.end());
	for(int i=0;i<t.length();i++)
	{
		comp=comp+t[i];
		int l=s.find(comp);
		if(l>=0)
		{
			x=l+1;
			y=x+comp.length()-1;
			continue;
		}
		else
		{
			l=sr.find(comp);
		}
		if(l<0)
		{
			if(comp.length()==1)
			{
				printf("-1\n");
				return 0;
			}
			ans.push_back(make_pair(x,y));
			i--;
			comp.clear();
		}
		else
		{
			x=l+1;
			y=x+comp.length()-1;
			x=s.length()+1-x;
			y=s.length()+1-y;
		}
	}
	ans.push_back(make_pair(x,y));
	printf("%d\n",(int)ans.size());
	for(int i=0;i<(int)ans.size();i++)
	{
		printf("%d %d\n",ans[i].first,ans[i].second);
	}
	return 0;
}
