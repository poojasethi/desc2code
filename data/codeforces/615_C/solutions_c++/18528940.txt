#include<bits/stdc++.h>
#define MX 100000
#define pb push_back
#define mp make_pair
#define fs first
#define sec second
#define sc scanf
#define pr printf
using namespace std;
int main(){
	//freopen("input.in","r",stdin);
	string s,s1,s2,s3;
	int n,m,i,j,k,x,y,z;
	vector<pair<int ,int> > v;
	cin>>s>>s1;
    n=s.length();
    m=s1.length();
    while(m!=0){
    	x=0;
    	for(i=0;i<n;++i){
    		j=0;
    		while(i+j<n && j<m && s[i+j]==s1[j])
    			++j;
               if(j>x){
               	x=j;
               	y=i+1;
               	z=i+j;
               }
    	}
    	for(i=n-1;i>=0;--i){
    		j=0;
    		while(i-j>=0 && j<m && s[i-j]==s1[j])
    			++j;
               if(j>x){
               	x=j;
               	y=i+1;
               	z=i+2-x;
               }
    	}
    	s2=s1.substr(x,m-x);
    	s1=s2;
    	m=s1.length();
    	if(x!=0)
    	v.pb(make_pair(y,z));
    else
    	break;
    }
    if(m!=0)
    	cout<<"-1";
    else{
    cout<<v.size()<<endl;
    for(i=0;i<v.size();++i)
    	cout<<v[i].first<<" "<<v[i].second<<endl;
}
	return 0;
}
