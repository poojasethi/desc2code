#include <bits/stdc++.h>
using namespace std;
int main() 
{
    string s1,s2,x="";
    cin>>s1>>s2;
    int f;
    string s3=s1;
    reverse(s3.begin(),s3.end());
    vector < pair<int,int> > res;
    pair<int,int> p;
    int l=s1.size();
    int l2=s2.size();
    for(int i=0;i<l2;i++)
    {
        x+=s2[i];
        f=s1.find(x);
        if(f==-1)
            f=s3.find(x);
        else
        {
            p.first=f+1;
            p.second=f+x.size();
            continue;
        }
        if(f==-1)
        {
            if(x.size()==1)
            {
                cout<<-1;
                return 0;
            }
            res.push_back(p);
            x="";
            i--;
        }
        else
        {
            p.first=l-f;
            p.second=l-f-x.size()+1;
        }
    }
    res.push_back(p);
    l=res.size();
    cout<<l<<endl;
    for(int i=0;i<l;i++)
    {
        cout<<res[i].first<<" "<<res[i].second<<endl;
    }
	return 0;
}
