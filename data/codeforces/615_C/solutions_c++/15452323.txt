#include <bits/stdc++.h>
#define pi pair<int,int>
#define pb push_back
#define fi first
#define se second
#define  L 2500
#define mp make_pair
using namespace std ;
int lcp[L][L],lcpr[L][L] ;
string t,s ;
int main()
{
    cin>>s>>t ;
    int szt=t.size() ;
    int szs=s.size() ;
    t="$" +t,s="$" +s ;
    vector< pi > ans ;
    for(int i=szt;i>=1;--i)
    {
        for(int j=szs;j>=0;--j)     if(t[i]==s[j]) lcp[i][j]=lcp[i+1][j+1] +1 ;
        for(int j=1;j<=szs;++j)    if(t[i]==s[j]) lcpr[i][j]=lcpr[i+1][j-1] +1 ;
    }
    for(int i=1;i<=szt ; )
    {
        int i1 =max_element(lcp[i]+1,lcp[i]+szs+1) -lcp[i] ;
        int i2 =max_element(lcpr[i]+1,lcpr[i]+szs+1) -lcpr[i] ;
        int len =max(lcp[i][i1],lcpr[i][i2]) ;
        if(len==0) { printf("-1\n") ; return 0 ;}
        else if(len==lcp[i][i1]) ans.pb(mp(i1,i1+len-1) ) ;
        else  ans.pb(mp(i2,i2-len+1)) ;
        i+=len ;
    }
    printf("%d\n",(int)ans.size()) ;
    vector < pi >::iterator it ;
    for(it=ans.begin();it!=ans.end();++it) printf("%d %d\n",(*it).fi,(*it).se) ;
    return 0 ;
}
