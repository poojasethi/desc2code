#include<cstdio>
#include<cstring>
char s1[2200],s2[2200];
int now,p,i,j,k,l1,l2,ma;
int ansx[2200],ansy[2200];
void find1(int i,int j)
{
    for (k=j;k<=l1;k++)
        if (s1[k]!=s2[i+k-j]) break;
    if (k-j>ma) ma=k-j,ansx[now]=j,ansy[now]=k-1;
}
void find2(int i,int j)
{
    for (k=j;k>=1;k--)
        if (s1[k]!=s2[i+j-k]) break;
    if (j-k>ma) ma=j-k,ansx[now]=j,ansy[now]=k+1;
}
bool check()
{
    now=0;
    for (int i=1;i<=l2;i+=ma)
    {
        now++;
        ma=0;
        for (int j=1;j<=l1;j++) find1(i,j);
        for (int j=l1;j>=1;j--) find2(i,j);
        if (ma==0) return 0;
    }
    return 1;
}
int main()
{
    scanf("%s",s1+1);
    scanf("%s",s2+1);
    l1=strlen(s1+1); l2=strlen(s2+1);
    if (!check()) printf("-1\n");
    else
    {
        printf("%d\n",now);
        for (int i=1;i<=now;i++) printf("%d %d\n",ansx[i],ansy[i]);
    }
}
