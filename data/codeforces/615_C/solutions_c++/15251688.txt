#include<bits/stdc++.h>
using namespace std;
int ind,subs[2200][2],lolnumb=234234;
int main()
{
    for(int i=0;i<2200;i++)
        subs[i][0] = subs[i][1] = lolnumb;
    char h[2200],j[2200]; scanf("%s%s",h,j); string x = string(h); string y = string(j);
    string z="";
    for(int i=0;i<y.size();i++)
    {
        z+=y[i];
        if(x.find(z)!= string::npos){
            subs[ind][0] = x.find(z); subs[ind][1] = subs[ind][0]+z.size()-1;}
        else
        {
            reverse(z.begin(),z.end());
            if(x.find(z)!=string::npos){
                subs[ind][1] = x.find(z); subs[ind][0] = subs[ind][1]+z.size()-1;}
            else
            {
                if(subs[ind][0]==lolnumb){
                    cout << "-1" << endl; return 0;}
                z="";   ind++; i--;
            }
            reverse(z.begin(),z.end());
        }
    }
    cout << ind+1 << endl;
    for(int i=0;i<=ind;i++)
        cout << subs[i][0]+1 << " " << subs[i][1]+1 << endl;
    return 0;

}
