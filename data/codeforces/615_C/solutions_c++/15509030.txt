#include<bits/stdc++.h>
using namespace std;
#define N 2111
char a[N],b[N];
int ans1[N],ans2[N],lena,lenb,cnt;
int f(int l)
{
    int A,B,ans=0;
    for(int i=0;i<lena;i++){
        int t=i,k=0,tt=l;
        while(a[t]==b[tt]&&t<lena&&tt<lenb){
            t++,k++,tt++;
        }
        if(k>ans){
            ans=k;
            A=i+1;
            B=k+i;
        }
    }
    for(int i=lena-1;i>=0;i--){
        int t=i,k=0,tt=l;
        while(a[t]==b[tt]&&t>=0&&tt<lenb){
            t--,k++,tt++;
        }
        if(k>ans){
            ans=k;
            A=i+1;
            B=i+2-k;
        }
    }
    ans1[cnt]=A,ans2[cnt++]=B;
    return ans;
}
int main()
{
    scanf("%s%s",a,b);
    cnt=0;
    lena=strlen(a),lenb=strlen(b);
    int st=0;
    while(1){
        if(st==lenb) break;
        int t=f(st);
        if(t==0){
            printf("-1\n");
            return 0;
        }
        st+=t;
    }
    printf("%d\n",cnt);
    for(int i=0;i<cnt;i++){
        printf("%d %d\n",ans1[i],ans2[i]);
    }
    return 0;
}
