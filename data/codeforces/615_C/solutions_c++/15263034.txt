#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int dpf[3200][3200],dps[3200][3200],ans[3200][2],markl,markr,flag;
int main()
{
	char str[3200],text[3200];
	scanf("%s%s",str,text);
	int sl,tl,i,j,maxx,cunt=0;
	sl=strlen(str);
	tl=strlen(text);
	for(i=1;i<=tl;i++)
	{
		for(j=1;j<=sl;j++)
		{
			if(str[j-1]==text[i-1])
			{
				dpf[i][j]=dpf[i-1][j-1]+1;
			}
		}
	}
	for(i=1;i<=tl;i++)
	{
		for(j=sl;j>=1;j--)
		{
			if(str[j-1]==text[i-1])
			{
				dps[i][j]=dps[i-1][j+1]+1;
			}
		}
	}
	for(i=tl;i>=1;i--)
	{
		maxx=0;
		for(j=1;j<=sl;j++)
		{
	//		printf("%d %d	%d\n",i,j,maxx);
			if(dpf[i][j]>maxx)
			{
				maxx=dpf[i][j];
				markl=j-dpf[i][j]+1;
				markr=j;
			}
			if(dps[i][j]>maxx)
			{
				maxx=dps[i][j];
				markl=j+dps[i][j]-1;
				markr=j;
			}
		}
	//	printf("\n");
		if(!maxx)
		{
			flag=1;break;
		}
		ans[cunt][0]=markl;
		ans[cunt++][1]=markr;
		i-=(maxx-1);
	}
	if(flag)
	{
		printf("-1"); return 0;
	}
	printf("%d\n",cunt);
	for(i=cunt-1;i>=0;i--)
	{
		printf("%d %d\n",ans[i][0],ans[i][1]);
	}
	return 0;
}
