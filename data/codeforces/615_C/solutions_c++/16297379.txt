#include<iostream>
#include<cstdio>
#include<cstring>

using namespace std;

const int maxn=10000;
string s,t,r,c;
int n,p,x[maxn+10],y[maxn+10],sl,tl,l;

int main()
{
	
	
	cin>>s>>t;
	 sl=s.length();
	 tl=t.length();

	for(int i=0;i<sl;i++)
	{
		r=s[i]+r;
	}

	p=s.find(t[0]);
	if(p==-1)
	{
		printf("-1\n");
		return 0;
	}

	else
	{
		c=t[0];
		n=1;
		x[1]=y[1]=p+1;
		l=0;
	}

	for(int i=1;i<tl;i++)
	{
		l++;
		c= c+t[i];

		p=s.find(c);
		if(p!=-1)
		{
			x[n]=p+1;
			y[n]=p+l+1;
			continue;
		}

		p=r.find(c);
		if(p!=-1)
		{
			x[n]= sl-p;
			y[n]=sl-p-l;
			continue;
		}

		p=s.find(t[i]);
		if(p==-1)
		{
			printf("-1\n");
			return 0;
		}

		c=t[i];
		++n;
		x[n]=y[n]=p+1;
		l=0;
	}

	printf("%d\n",n);
	for (int i=1;i<=n;i++)
		printf("%d %d\n",x[i],y[i]);
	return 0;
}
