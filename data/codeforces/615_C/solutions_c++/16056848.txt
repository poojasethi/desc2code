#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long
#define mod 1000000007

#define PII pair<int,int>
#define mp make_pair
#define fi first
#define se second

using namespace std;

char s[2222],t[2222];
int ls,lt,p;

vector<PII> V;

int main(){
  scanf("%s %s",s,t);
  ls=strlen(s);
  lt=strlen(t);
  p=0;
  while(p<lt){
    int m=p;
    int ml,mr,cur;
    FI(i,0,ls-1){
      //right
      cur=p;
      FI(j,i,ls-1){
        if(s[j]==t[cur]) cur++;
        else break;
        if(cur==lt) break;
      }
      if(cur>m){
        m=cur;
        ml=i;
        mr=i+cur-1-p;
      }
      //left
      cur=p;
      FD(j,i,0){
        if(s[j]==t[cur]) cur++;
        else break;
        if(cur==lt) break;
      }
      if(cur>m){
        m=cur;
        ml=i;
        mr=i-(cur-1-p);
      }
    }
    if(m==p){
      printf("-1\n");
      return 0;
    }
    p=m;
    V.push_back(mp(ml+1,mr+1));
  }
  int sz=V.size();
  printf("%d\n",sz);
  FI(i,0,sz-1) printf("%d %d\n",V[i].fi,V[i].se);
  return 0;
}
