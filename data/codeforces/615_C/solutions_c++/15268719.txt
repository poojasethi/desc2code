#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
using namespace std;
const int Maxn = 3000;

char A[Maxn],B[Maxn];
int lena,lenb;
int Outl[Maxn],Outr[Maxn];

void solve()
{
	scanf("%s%s",A+1,B+1);
	lena = strlen(A+1); lenb = strlen(B+1);
	int pos;int l,r;
	pos=1;int ans=0;
	while(pos<=lenb){
		ans++;
		int maxl=0,maxlen=0,maxr=0;
		for(int i=1; i <= lena; i++){
			int len=0;
			while(i+len<=lena && pos+len<=lenb && A[i+len]==B[pos+len])
                len++;
			if(len>maxlen){
                maxlen=len;
                maxl=i;
                maxr=i+len-1;
            }
		}
		for(int i=lena; i>=1; i--){
			int len=0;
			while(i>=len+1  && pos+len<=lenb && A[i-len]==B[pos+len])
                len++;
			if(len>maxlen){
			    maxlen=len;
                maxl=i;
                maxr=i-len+1;
            }
		}
		if(!maxlen){printf("-1\n");return;}
		Outl[ans]=maxl; Outr[ans]=maxr;
		pos+=maxlen;
	}
	printf("%d\n",ans);
	for(int i=1;i<=ans;i++) 
        printf("%d %d\n",Outl[i],Outr[i]);
}

int main()
{
	solve();
	return 0;
}
