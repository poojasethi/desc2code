#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
char s1[6000],s2[3000];
int len1,len2,belong[6000],m,r[3000],l[3000];
void solve()
{
	int st=1,left,right,ans,ansl,tot;
	while (st<=len2)
	{
		left=ans=tot=ansl=0;
		for (int i=1;i<=len1;++i)
		 if (s2[st]==s1[i])
		 {
		 	left=i;ans=1;
		 	for (;s2[st+ans]==s1[left+ans] && st+ans<=len2 && left+ans<=len1;++ans);
		 	if (ans>tot)
		 	{
		 		ansl=left;
		 		tot=ans;
			}
		 }
		if (!ans)
		{
			m=0;
			return;
		}
		l[++m]=belong[ansl];
		r[m]=belong[ansl+tot-1];
		st+=tot;
	}	
}
int main()
{
//	freopen("c.in","r",stdin);
//	freopen("c.out","w",stdout); 	
	scanf("%s",s1+1);
	scanf("%s",s2+1);
	len1=strlen(s1+1);
	len2=strlen(s2+1);
	s1[len1+1]='$';
	for (int i=1;i<=len1;++i)
	 s1[len1*2-i+2]=s1[i],belong[len1*2-i+2]=belong[i]=i;
	len1=len1*2+1;
	solve();
	if (m==0)printf("-1\n");
	else 
	{
		printf("%d\n",m);
		for (int i=1;i<=m;++i)
		 printf("%d %d\n",l[i],r[i]);		
	}
	fclose(stdin);
	fclose(stdout);
	return 0;
}
