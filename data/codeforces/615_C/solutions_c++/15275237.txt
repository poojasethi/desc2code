#include<bits/stdc++.h>
#define fi first
#define se second
using namespace std;

int n,m,dp[2105],p[2105],d[2105][2105],c[2105][2105],te;
string s,t;
int f(int i,int j){
if(c[i][j]!=-1) return c[i][j];
else if(i==n||j==m||!(s[j]==t[i])) return c[i][j]=0;
else return c[i][j]=1+f(i+1,j+1);
}
int g(int i,int j){
if(i==n||j<0||!(s[j]==t[i])) return 0;
else if(d[i][j]!=-1) return d[i][j];
else if(s[j]==t[i])return d[i][j]=1+g(i+1,j-1);
}
void P(int x){
te=p[x];
if(te>0){
    te--;
    cout<<te+1<<" "<<te+f(x,te)<<endl;
    if(x+f(x,te)!=n) P(x+f(x,te));
}else{
    te=-te;te--;
    cout<<te+1<<" "<<te-g(x,te)+2<<endl;
    if(x+g(x,te)!=n) P(x+g(x,te));
}
}
int main(){
cin>>s>>t;m=s.length();n=t.length();
memset(d,-1,sizeof(d));memset(c,-1,sizeof(c));
dp[n]=0;
for(int i=n-1;i>=0;i--){
    dp[i]=1e9;
    for(int j=0;j<m;j++){
        te=f(i,j);
        if(te>0) if(dp[i+te]+1<dp[i]) dp[i]=dp[i+te]+1,p[i]=j+1;
        te=g(i,j);
        if(te>0) if(dp[i+te]+1<dp[i]) dp[i]=dp[i+te]+1,p[i]=-j-1;
    }
}
if(dp[0]!=1e9){
    cout<<dp[0]<<endl;
    P(0);
}
else cout<<-1<<endl;
return 0;
}
