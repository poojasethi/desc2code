#include<iostream>
#include<string>
using namespace std;
string s1,s2;
string t;
const int maxn=2222;
int dp[maxn][2];
int main()
{
	cin>>s1>>t;
	s2.assign(s1.rbegin(),s1.rend());
	int lent=t.length(),lens=s1.length();
	int num=0;
	for(int i=0;i<lent;)
	{
		int p1,p2;
		int mmaxn=0;
		for(int j=0;j<lens;j++)
		{
			int len=0,nexts=j,nextt=i;
			while(s1[nexts]==t[nextt]&&nexts<lens&&nextt<lent){
				nexts++,nextt++;len++;
			}
			if(mmaxn<len)
			{
				mmaxn=len;p1=j+1,p2=j+mmaxn;
			}
		}
		for(int j=0;j<lens;j++)
		{
			int len=0,nexts=j,nextt=i;
			while(s2[nexts]==t[nextt]&&nexts<lens&&nextt<lent){
				nexts++;nextt++;len++;
			}
			if(mmaxn<len)
			{
				mmaxn=len;p1=lens-j;p2=lens-j-mmaxn+1;
			}
		}
		if(mmaxn==0)
		{
			cout<<"-1"<<endl;
			return 0;
		}
		dp[num][0]=p1,dp[num++][1]=p2;
		i+=mmaxn;
	}
	cout<<num<<endl;
	for(int i=0;i<num;i++)
		cout<<dp[i][0]<<" "<<dp[i][1]<<endl;
}
