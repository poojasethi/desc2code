#include <bits/stdc++.h>
using namespace std;
string s,rs,w,a;
int len,cnt,p,l,r,ls;
pair<int,int> ans[3000];
int main(){
    cin>>s>>w;
    rs=s;
    reverse(rs.begin(),rs.end());
    int i,j;
    i=0;
    len=w.size();ls=s.size();
    while (i<len){
        j=i;a.clear();
        while (j<len){
            a+=w[j];
            p=s.find(a);
            if (p==-1){
                p=rs.find(a);
                if (p==-1) break;
                l=ls-p;
                r=l-(j-i);
            } else{
                l=p+1;
                r=l+(j-i);
            }
            j++;
        }
        if (j==i){
            printf("-1\n");
            return 0;
        }
        i=j;
        ans[++cnt]=make_pair(l,r);
    }
    printf("%d\n",cnt);
    for (int i=1; i<=cnt; i++) printf("%d %d\n",ans[i].first,ans[i].second);
}
