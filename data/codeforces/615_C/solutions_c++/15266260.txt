#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define maxn 3010
char s[maxn],t[maxn];
int n,m;
int best1[maxn][maxn],best2[maxn][maxn];
int ansl[maxn],ansr[maxn];
int main()
{
	scanf("%s%s",s+1,t+1);
	n=strlen(s+1),m=strlen(t+1);
	for(int i=n; i; i--)
		for(int j=m; j; j--)
		{
			if(s[i]==t[j]) best1[i][j]=best1[i+1][j+1]+1;
			else best1[i][j]=0;
		}
	for(int i=1; i<=(n>>1); i++) swap(s[i],s[n-i+1]);
	for(int i=n; i; i--)
		for(int j=m; j; j--)
		{
			if(s[i]==t[j]) best2[i][j]=best2[i+1][j+1]+1;
			else best2[i][j]=0;
		}
	int duan=0;
	for(int j=1; j<=m; )
	{
		duan++;
		int bt=0;
		for(int i=1; i<=n; i++)
		{
			if(best1[i][j]>bt)
			{
				bt=best1[i][j];
				ansl[duan]=i; ansr[duan]=i+bt-1;
			}
			if(best2[i][j]>bt)
			{
				bt=best2[i][j];
				ansl[duan]=n-i+1; ansr[duan]=n-(i+bt-1)+1;
			}
		}
		if(bt==0)
		{
			puts("-1");
			return 0;
		}
		j+=bt;
	}
	printf("%d\n",duan);
	for(int i=1; i<=duan; i++)
		printf("%d %d\n",ansl[i],ansr[i]);
	return 0;
}
