#include <bits/stdc++.h>
using namespace std;

int n, m;
string a, b, c;
int main() {
    cin >> a >> b;
    n = a.size(), m = b.size();
    c = a;
    reverse(c.begin(), c.end());
    int i = 0;
    vector<pair<int,int> > ans;
    while (i < m) {
        string x = "";
        int from = -1, to = -1, last = i;
        for(int j=i; j<m; j++) {
            x += b[j];
            int val1 = a.find(x);
            int val2 = c.find(x);
            if (val1 == string::npos && val2 == string::npos) break;
            if (val1 != string::npos)
                from = val1+1, to = from + (j-i);
            else
                from = n - val2, to = from - (j-i);
            last = j;
        }
        if (from == -1) { cout << "-1\n"; return 0; }
        ans.push_back(make_pair(from, to));
        i = last+1;
    }
    cout << ans.size() << "\n";
    for(int i=0; i<(int)ans.size(); i++) {
        cout << ans[i].first << " " << ans[i].second << "\n";
    }


    return 0; 
}
