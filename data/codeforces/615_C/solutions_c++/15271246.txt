#include <bits/stdc++.h>
using namespace std;
int main(){
	string s,t,q,a;
	cin>>s>>q;
	t=s;
	reverse(t.begin(),t.end());
	int i,ff,x,y,n = t.size();
	vector<pair<int,int> >ret;
	for(i=0;i<q.size();++i){
		a+= q[i];
		ff = s.find(a);
		if(ff==-1)
			ff = t.find(a);
		else{
			x= ff+1,y = x+a.size()-1;
			continue;
		}
		if(ff==-1){
			if(a.size()==1){
				cout<<-1<<endl;return 0;
			}
			ret.push_back(make_pair(x,y));--i;
			a.clear();
		}
		else
			x= ff+1,y= x+a.size()-1,x=n+1-x,y=n+1-y;
	}
	ret.push_back(make_pair(x,y));
	cout<<ret.size()<<endl;
	for(i=0;i<ret.size();++i)
		cout<<ret[i].first<<" "<<ret[i].second<<endl;
}
