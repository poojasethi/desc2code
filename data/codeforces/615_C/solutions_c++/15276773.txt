#include<cstdio>
#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
string s, t, sr;
int ok;
int st[2110], ed[2110], p, fnd, ts, top, i;
int main()
{
//	freopen("input.txt", "r", stdin);
	cin >> s >> t;
	sr = s;
	reverse(sr.begin(), sr.end());
//	cout << s << endl << sr << endl;
	while(p < t.size()){
		ok = 0;
		for(i = 1; i <= t.size() - p; i++){
			fnd = s.find(t.substr(p, i));
			if(fnd != s.npos){
				ts = fnd;
				ok = 1;
				continue;
			}
			fnd = sr.find(t.substr(p, i));
			if(fnd != sr.npos){
				ts = s.size() - fnd;
				ok = -1;
				continue;
			}
			break;
		}
		i--;
		if(ok == 1){
			st[top] = ts + 1;
			ed[top++] = ts + i;
		}
		else if(ok == -1){
			st[top] = ts;
			ed[top++] = ts - i + 1;
		}
		if(ok == 0)
			break;
		p = p + i;
	}
	if(ok == 0 ){
		cout << -1 << endl;
		return 0;
	}
	cout << top << endl;
	for(i = 0; i < top; i++)
		cout << st[i] << " " << ed[i] << endl;
	return 0;
}
