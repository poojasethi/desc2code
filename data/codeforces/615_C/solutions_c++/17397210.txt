#include<bits/stdc++.h>
using namespace std;
const int maxn = 2100;
char a[maxn],b[maxn],c[maxn];
int x[maxn],y[maxn];
int main()
{
     scanf("%s%s",a,c);
	 int len1 = strlen(a);
	 for (int i = 0;i<len1;i++)
		 b[i]=a[len1-i-1];
	 int len2 = strlen(c);
	 int ans = 0;
	 for (int i = 0;i<len2;i++)
	 {
		 int flag=0;
		 int xx,yy;
		 for (int j = 0;j<len1;j++)
		 {
			 if (c[i]==a[j])
			 {
				 int k = i;
				 int h = j;
				 while (c[k]==a[h]&&k<len2&&h<len1)
				 {
					 k++;
					 h++;
				 }
				 int l = k-i;
				 if (flag<l)
				 {
					 flag=l;
					 xx=j+1;
					 yy=j+l;
				 }
			 }
		 }
		 for (int j = 0;j<len1;j++)
		 {
			 if (c[i]==b[j])
			 {
				 int k = i;
				 int h = j;
				 while (c[k]==b[h]&&k<len2&&h<len1)
				 {
					 k++;
					 h++;
				 }
				 int l = k-i;
				 if (flag<l)
				 {
					 flag=l;
					 xx=len1-j;
					 yy=len1-(j+l-1);
				 }
			 }
		 }
		 if (flag)
		 {
			 x[ans]=xx;
			 y[ans]=yy;
			 ans++;
			 i+=(flag-1);
		 }
		 else
		 {
			 ans=0;
			 break;
		 }
	 }
	 if (ans==0)
	 {
		 printf("-1\n");
		 return 0;
	 }
	 printf("%d\n",ans);
	 for (int i = 0;i<ans;i++)
		 printf("%d %d\n",x[i],y[i]);
}
