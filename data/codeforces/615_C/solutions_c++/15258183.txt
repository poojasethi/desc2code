#include <iostream>
#include <cmath>
#include <vector>

using namespace std;

vector <pair <int, int> > ans;

int main()
{
	string a, s;
	cin >> a >> s;
	int i = 0;
	while (i < s.size())
	{
		int x = -1, y = -1, maxi = 0;
		for (int j = 0; j < a.size(); j++)
			if (a[j] == s[i])
			{
				int cnt = 1;
				while (j + cnt < a.size() && i + cnt < s.size() && a[j + cnt] == s[i + cnt])
					cnt++;
				if (cnt > maxi)
				{
					maxi = cnt;
					x = j;
					y = j + cnt - 1;
				}
				cnt = 1;
				while (j - cnt >= 0 && i + cnt < s.size() && a[j - cnt] == s[i + cnt])
					cnt++;
				if (cnt > maxi)
				{
					maxi = cnt;
					x = j;
					y = j - cnt + 1;
				}
			}
		if (maxi == 0)
		{
			cout << "-1\n";
			return 0;
		}
		ans.push_back(make_pair(x + 1, y + 1));
		i += abs(x - y) + 1;
	}
	cout << ans.size() << endl;
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i].first << " " << ans[i].second << endl;
	return 0;
}
