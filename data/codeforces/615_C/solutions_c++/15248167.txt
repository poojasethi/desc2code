#include <bits/stdc++.h>

using namespace std;

const int maxn = 2200;

int n, m;
char s[maxn], rs[maxn], t[maxn];
char tmp[maxn];
vector <pair<int, int> > res;

bool check(int st, int ed, int &a, int &b) {
	char *p;
	strncpy(tmp, t+st, ed-st+1);
	tmp[ed-st+1] = '\0';
	p = strstr(s, tmp);
	if(p != NULL) {
		a = p-s+1;
		b = a+ed-st;
		return 1;
	}
	p = strstr(rs, tmp);
	if(p != NULL) {
		a = n-(p-rs);
		b = a-ed+st;
		return 1;
	}
	return 0;
}

int main() {
	scanf("%s%s", s, t);
	n = strlen(s);
	m = strlen(t);
	for(int i = 0; i < n; i ++) rs[i] = s[n-i-1];
	bool ok = 1;
	for(int i = 0; i < m && ok; i ++) {
		int L = i+1, R = m-1, M, a, b;
		while(L <= R) {
			M = (L+R)/2;
			if(check(i, M, a, b)) L = M+1;
			else R = M-1;
		}
		L --;
		if(check(i, L, a, b)) {
			res.push_back(make_pair(a, b));
			i = L;
		} else {
			ok = 0;
		}
	}
	if(ok) {
		printf("%d\n", (int)res.size());
		for(int i = 0; i < res.size(); i ++) {
			printf("%d %d\n", res[i].first, res[i].second);
		}
	} else {
		printf("-1\n");
	}
	return 0;
}
