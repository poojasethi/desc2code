#include <stdio.h>
#include <string.h>
char s[2105],t[2105];
int si[26]={},ti[26]={};
int ans[2105][2]={},num=0;

int strl()
{
	int i,l;
	for(i=0;t[i];i++)
		ti[t[i]-'a']++;
	l=i;
	for(i=0;s[i];i++)
		si[s[i]-'a']++;
	for(i=0;i<26;i++)
		if(ti[i]!=0&&si[i]==0)
			return -1;
	return l;
}

int cut(int p)
{
	int i,j,a,b,len=0;
	for(i=0;s[i];i++)
		if(s[i]==t[p])
		{
			for(j=0;s[i+j]==t[p+j]&&t[p+j];j++);
			if(j>len)
				len=j,a=i+1,b=i+j;
		}
	for(i--;i>=0;i--)
		if(s[i]==t[p])
		{
			for(j=0;s[i-j]==t[p+j]&&t[p+j]&&i-j>=0;j++);
			if(j>len)
				len=j,a=i+1,b=i-j+2;
		}
	ans[num][0]=a,ans[num][1]=b;
	return len;
}

int main()
{
	int i,l;
	scanf("%s%s",s,t);
	l=strl();
	if(l==-1)
	{
		printf("-1\n");
		return 0;
	}
	while(i<l)
	{
		i+=cut(i);
		num++;
	}
	printf("%d\n",num);
	for(i=0;i<num;i++)
		printf("%d %d\n",ans[i][0],ans[i][1]);
}
