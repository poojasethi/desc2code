#include <bits/stdc++.h>
using namespace std;
int main(){
	int n,temp,i,j;cin>>n;
	int num[n];
	for(i=0;i<n;i++){
		cin>>temp;
		num[i]=temp;
	}
	int min=INT_MAX;
	int total=0;
	for(i=0;i<n;i++){
		total=0;
		for(j=0;j<num[i];j++){cin>>temp;total+=temp;}
		if(min>total*5+num[i]*15)min=total*5+num[i]*15;
	}
	cout<<min<<endl;
}
