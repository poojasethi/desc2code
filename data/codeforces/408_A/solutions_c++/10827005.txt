#include<stdio.h>
int i,j,arr[10000],n,max=123213,top,a;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&arr[i]);

	for(i=1;i<=n;i++){
		for(j=1;j<=arr[i];j++){
			scanf("%d",&a);
			top+=a;
		}
		if(top*5 + arr[i]*15 < max)
			max = top*5 + arr[i]*15;
		top=0;
	}
	printf("%d",max);
	return 0;
}
