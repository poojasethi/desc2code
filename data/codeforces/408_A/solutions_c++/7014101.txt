#include<stdio.h>
int main()
{
	int n,k[110],m,t,min=100000;
	int i,j;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&k[i]);
	for(i=0;i<n;i++)
	{
		for(j=0,t=0;j<k[i];j++)
		{
			scanf("%d",&m);
			if(m)
				t+=m*5+15;
		}
		min=t<min?t:min;
	}
	printf("%d\n",min);
	return 0;
}
