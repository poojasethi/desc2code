#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,i,j;
cin>>n;
int k[101],ans=INT_MAX,temp,mini;
for(i=1;i<=n;i++)
cin>>k[i];

for(i=1;i<=n;i++)
{
mini=0;
for(j=1;j<=k[i];j++)
{
//scanf("%d",&temp);
cin>>temp;
mini=mini+temp*5;

}
mini=mini+k[i]*15;

if(mini<ans)
ans=mini;
}
cout<<ans;
return 0;
}
