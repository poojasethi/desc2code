#include <iostream>
using namespace std;

int main(){
	int n,k[100],mint=1<<31-1;
	cin>>n;
	for(int i=0;i<n;++i)cin>>k[i];
	
	for(int i=0;i<n;++i){
		int tmpm=0;
		for(int j=0;j<k[i];++j){
			int tmp;
			cin>>tmp;
			tmpm+=tmp;
		}
		mint=min(mint,tmpm*5+k[i]*15);
	}
	cout<<mint<<endl;
	
	return 0;
}
