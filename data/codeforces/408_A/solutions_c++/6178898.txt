#include<stdio.h>
int main()
{ int k[110];
  int i,j,a,sum,n,min=9999999;
  scanf("%d",&n);
  for(i=1;i<=n;i++)
  scanf("%d",&k[i]);
  for(i=1;i<=n;i++)
  { sum=0;
    for(j=1;j<=k[i];j++)
    {scanf("%d",&a);
     sum+=a*5+15;
    }
    if(sum<min)min=sum;
  }     
 printf("%d\n",min);
 return 0;  
}
