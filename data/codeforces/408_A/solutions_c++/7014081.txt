#include<stdio.h>

main()
{
	int i,j,k,l,m,n,min=1e10,sum;
	int a[110];
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	for(i=0;i<n;i++)
	{
		sum=0;
		for(j=0;j<a[i];j++)
		{
			scanf("%d",&l);
			sum=sum+l*5+15;
		}
		min=min<sum?min:sum;
	}
	printf("%d\n",min);
}
