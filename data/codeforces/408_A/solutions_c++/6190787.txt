#include <iostream>
#include <climits>
using namespace std;

int main() {
	int n,i,k,j,mi=INT_MAX;
	cin>>n;
	int a[n+1];
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<n;i++)
	{
		int ans=0;
		for(j=0;j<a[i];j++)
		{
			cin>>k;
			ans+=(k*5);
		}
		ans+=(a[i]*15);
		mi=min(ans,mi);
	}
	cout<<mi<<endl;
	return 0;
}
