#include <iostream>

using namespace std;

int main()
{
  int ar[109],n,mq=10000009,sum,r;
  cin>>n;
  for(int x=0; x<n; x++) cin>>ar[x];
  for(int x=0; x<n; x++){
    sum=15*ar[x];
    while(ar[x]--){
            cin>>r;
        sum+=5*r;
    }
    if(sum<mq) mq=sum;
  }
  cout<<mq<<endl;

    return 0;
}
