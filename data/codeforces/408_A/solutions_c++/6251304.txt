#include<bits/stdc++.h>

using namespace std;

int main()
{
	int sum=0,time,res=1e9;
	
	int n,a;
	cin>>n;

	int k[n];
	for(int i=0;i<n;i++)
		cin>>k[i];
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<k[i];j++)
		{
			cin>>a;
			sum+=a*5;
		}
		sum+=k[i]*15;
		res=min(res,sum);
		sum=0;
	}
	cout<<res<<endl;
	return 0;
}
