#include<stdio.h>

int k[100005],a,i,n,min=100000,top,j;

int main(){

	scanf("%d",&n);
	for(i = 1; i <= n; i++)
		scanf("%d",&k[i]);
	for(i = 1; i <= n; i++){
		top = 15*k[i];
		for(j = 1; j <= k[i]; j++){
			scanf("%d",&a);
			top += 5*a;
		}
		if(top < min)
			min = top;
	}
	printf("%d\n",min);

	return 0;
}
