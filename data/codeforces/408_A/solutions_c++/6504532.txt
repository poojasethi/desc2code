#include<stdio.h>
int i,j,k,n,A[105][2],mn=999999;

int main(){
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&A[i][0]);
	for(i=0;i<n;i++){
		for(j=0;j<A[i][0];j++){
			scanf("%d",&k);
			A[i][1]+=k*5;
		}
		A[i][1]+=A[i][0]*15;
	}
	for(i=0;i<n;i++)
		if(mn>A[i][1])
			mn=A[i][1];
	printf("%d",mn);
	return 0;
}
