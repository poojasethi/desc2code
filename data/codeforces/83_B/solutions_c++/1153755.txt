#include <iostream>
using namespace std;

const int MAX = 100010, INFI = 1000000001;
int que[MAX], n, ans;
long long k;

long long check(int m)
{
	long long sum = 0;
	for (int i = 1; i <= n; ++ i)
		sum += min(que[i], m);
	return sum;
}

int main()
{
	cin >> n >> k;
	for (int i = 1; i <= n; ++ i)
		cin >> que[i];
	if (check(INFI) < k)
		cout << -1 << endl;
	else
	{
		int le = 0, ri = INFI, mid;
		while (le <= ri)
		{
			mid = (le + ri) / 2;
			if (check(mid) <= k)
			{
				ans = mid;
				le = mid + 1;
			}
			else
				ri = mid - 1;
		}
		k -= check(ans);
		for (int i = 1; i <= n; ++ i)
			que[i] -= min(que[i], ans);
		int t;
		for (t = 1; k; ++ t)
			if (que[t])
			{
				-- que[t];
				-- k;
			}
		for (int i = t; i <= n; ++ i)
			if (que[i])
				cout << i << ' ';
		for (int i = 1; i < t; ++ i)
			if (que[i])
				cout << i << ' ';
	}
	return 0;
}
