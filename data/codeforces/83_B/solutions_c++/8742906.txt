#include<iostream>
#include<algorithm>
#include<cstdio>
#define M 100100
using namespace std;
typedef long long ll;
ll n,k,s,sum,a[M],tot,ans;
int main()
{
    cin>>n>>k;
    for(int i=1;i<=n;i++)scanf("%d",&a[i]),s+=a[i];
    if(s<k)    { puts("-1");return 0;}
    if(s==k){puts("");return 0;}
    ll l=0,r=k;
    while(l<=r)
    {
        ll mid=(r+l)>>1;
        sum=0;
        for(int i=1;i<=n;i++)
            sum+=min(a[i],mid);
        if(sum<=k)  l=mid+1,ans=mid;
        else        r=mid-1;
    }
    for(int i=1;i<=n;i++) k-=min(a[i],ans);
    int b[100100];
    int flag=0;
    for(int i=1;i<=n;i++)if(a[i]>ans)b[++tot]=i;
    for(int i=k+1;i<=tot;i++)
        printf("%s%d",flag++==0?"":" ",b[i]);
    for(int i=1;i<=k;i++)
        if(a[b[i]]>ans+1)
            printf("%s%d",flag++==0?"":" ",b[i]);
    return 0;
}
