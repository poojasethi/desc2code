#include <iostream>
#include <cstdio>
#include <map>
#include <deque>
using namespace std;

int N;
long long int K,tot,tmp,hi,lo,mid,cutoff,S[100005];
deque <int> D;

int main(){
	cin >> N >> K;
	for(int i=1;i<=N;++i){
		cin >> S[i];
		tot += S[i];
		hi = max(hi,S[i]);
	}
	if(tot < K){
		printf("-1\n");
		return 0;
	}
	while(hi >= lo){
		mid = (hi+lo)/2ll;
		tmp = 0;
		for(int i=1;i<=N;++i) tmp += (S[i] < mid)?S[i]:mid;
		if(tmp <= K){
			cutoff = mid;
			lo = mid+1;
		}
		else hi = mid-1;
	}
	for(int i=1;i<=N;++i){
		K -= (S[i] < cutoff)?S[i]:cutoff;
		S[i] -= cutoff;
		if(S[i] > 0) D.push_back(i);
	}
	for(unsigned int i=0;i<D.size();++i){
		if(K == 0){
			for(unsigned int j=i;j<D.size();++j) printf("%d ",D[j]);
			printf("\n");
			break;
		}
		--K;
		if(S[D[i]] > 1) D.push_back(D[i]);
	}
}
