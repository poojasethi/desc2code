#include<iostream>
#include<queue>
using namespace std;
long long int a[100004];long long int n,k;
long long int b(long long i){long long int s=0;
for(long j=1;j<=n;j++){
if(a[j]<=i)s=s+a[j];else s=s+i;
}
    return s;}
int main(){
ios::sync_with_stdio(false);
        	cin.tie(0);
        	cout.tie(0);
cin>>n>>k;
for(long i=1;i<=n;i++)cin>>a[i];
long long int l,r;
queue<long >q;
l=0;r=10000000000;
if(b(r)<k){cout<<"-1"<<endl;}
while(r-l>1){
    long long int mid=(l+r)/2;
    if(b(mid)>k)r=mid;
    else l=mid;
}
//cout<<l<<endl;
long long int p=k-b(l);long o=0;
//cout<<p<<endl;
for(long i=1;i<=n;i++){
    if(a[i]>l){o++;
   // cout<<a[i]<<" "<<o<<endl;
        if(o>p)cout<<i<<" ";
        else{
            if(a[i]>l+1)q.push(i);
        }
    }
}
while(!q.empty()){cout<<q.front()<<" ";q.pop();
}
cout<<endl;}
