#include<stdio.h>
#include<string.h>
typedef long long int64;
int a[100000],n;
int main()
{
	int i,j,l,r,c,v;
	int64 s,k;
	scanf("%d %I64d",&n,&k);
	s=0;
	r=0;
	for (i=0;i<n;i++)
	{
		scanf("%d",a+i);
		s+=a[i];
		if (a[i]>r)
			r=a[i];
	}
	if (s<k)
		printf("-1\n");
	else
	{
		v=0;
		l=0;
		while (l<=r)
		{
			c=(l+r)/2;
			s=0;
			for (i=0;i<n;i++)
			{
				if (a[i]>c)
					s+=c;
				else
					s+=a[i];
			}
			if (s<=k)
			{
				l=c+1;
				v=c;
			}
			else
				r=c-1;
		}
		for (i=0;i<n;i++)
		{
			if (a[i]>v)
			{
				a[i]-=v;
				k-=v;
			}
			else
			{
				k-=a[i];
				a[i]=0;
			}
		}
		i=0;
		while (k>0)
		{
			if (a[i]>0)
			{
				a[i]--;
				k--;
			}
			i++;
		}
		j=i;
		c=0;
		for (i=j;i<n;i++)
		{
			if (a[i]>0)
			{
				if (c==0)
					c=1;
				else
					printf(" ");
				printf("%d",i+1);
			}
		}
		for (i=0;i<j;i++)
		{
			if (a[i]>0)
			{
				if (c==0)
					c=1;
				else
					printf(" ");
				printf("%d",i+1);
			}
		}
		printf("\n");
	}
	return 0;
}
