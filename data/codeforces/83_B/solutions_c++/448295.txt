#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <queue>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define ll long long
#define lim 100001

ll n,m,h,i,r,s,mx;
ll A[lim],R[lim];

ll moves(ll h)
{
  ll i,s=0;
  
  for(i=0;i<n;i++)
    s+=min(A[i],h);
  return s;
}
ll bs(ll i,ll j)
{
  ll k=(i+j+1)/2;
  
  if(i==j)
    return i;
  if(moves(k)<=m)
    return bs(k,j);
  return bs(i,k-1);
}
int main()
{
  scanf("%I64d%I64d",&n,&m);
  for(i=0;i<n;i++)
  {
    scanf("%I64d",&A[i]);
    s+=A[i];
    mx=max(mx,A[i]);
  }
  if(s<m)
  {
    printf("-1");
    return 0;
  }
  h=bs(0,mx);
  m-=moves(h);
  i=0;
  while(m>0)
  {
    while(A[i]<=h)
      i++;
    i++;
    if(A[i-1]>h+1)
      R[r++]=i;
    m--;
  }
  for(;i<n;i++)
    if(A[i]>h)
      printf("%d ",i+1);
  for(i=0;i<r;i++)
    printf("%d ",R[i]);
  return 0;
}
