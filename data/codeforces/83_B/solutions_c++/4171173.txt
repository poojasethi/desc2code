#include<iostream>
using namespace std;
int main()
{
    long long a[100002],n,k,r,l,i,mid,sum=0;
    cin>>n>>k;
    for(i=0;i<n;i++) {cin>>a[i];sum+=a[i];}
    if(sum<k) {cout<<"-1\n";return 0;}
    if(sum==k) {cout<<endl;return 0;}
    l=0;r=k-1;
    while(l<r)
    {
        mid=(r+l+1)/2;
        sum=0;
        for(i=0;i<n;i++) sum+=min(a[i],mid);
        if(sum>k) r=mid-1;
        else l=mid;
    }
    for(i=0;i<n;i++) k-=min(a[i],l);
    int b[100002],kk=0;
    for(i=0;i<n;i++) if(a[i]>l) b[kk++]=i;
    for(i=k;i<kk;i++) cout<<b[i]+1<<' ';
    for(i=0;i<k;i++) if(a[b[i]]-l-1>0) cout<<b[i]+1<<' ';
    cout<<endl;
    return 0;
}
