#include <iostream>
#include <algorithm>
using namespace std;
int main(void) {
  long long n,k,a[100100],m=0,s=0;
  cin>>n>>k;
  for(int i=0;i<n;++i)cin>>a[i],s+=a[i];
  if(s<k){
    cout<<-1<<endl;
    return 0;
  }
  long long lo=0,hi=k;
  while(lo<hi){
    long long mid=(lo+hi+1)>>1;
    s=0;
    for(int i=0;i<n;++i)s+=min(a[i],mid);
    if(s>k)hi=mid-1;else lo=mid;
  }
  for(int i=0;i<n;++i)k-=min(a[i],lo);
  int b[100100],u=0;
  for(int i=0;i<n;++i)if(a[i]>lo)b[m++]=i;
  for(int i=k;i<m;++i)cout<<(u++>0?" ":"")<<b[i]+1;
  for(int i=0;i<k;++i)if(a[b[i]]>lo+1)cout<<(u++>0?" ":"")<<b[i]+1;
  cout<<endl;
  return 0;
}
