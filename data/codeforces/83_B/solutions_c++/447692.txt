#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int a[100005];

int main(void) {
	int n, i, j;
	long long k;
	cin.sync_with_stdio(false);
	cin >> n >> k;
	int R = 0;
	for(i=0;i<n;i++) {
		cin >> a[i];
		if (R < a[i])
			R = a[i];
	}
	int L = 0, M, ans = -1;
	while (L <= R) {
		M = (L+R)/2;
		long long tmp = 0;
		for (i=0;i<n;i++)
			tmp += (long long)min(a[i], M);
		if (tmp >= k) {
			ans = M;
			R = M-1;
		} else
			L = M+1;
	}
	vector<int> ret;
	if (ans == -1)
		cout << "-1" << endl;
	else if (k == 0) {
		for(i=0;i<n;i++)
			ret.push_back(i+1);
	} else {
		--ans;
		for(i=0;i<n;i++) {
			k -= min(a[i], ans);
			a[i] -= min(a[i], ans);
		}
		for(i=0;i<n && k>0;i++) {
			if(a[i]>0) {--a[i];--k;}
		}
		for(j=i;j<n;j++)
			if(a[j]>0)
				ret.push_back(j+1);
		for(j=0;j<i;j++)
			if(a[j]>0)
				ret.push_back(j+1);
	}
	for(i=0;i<ret.size();i++)
		cout << ret[i] << (i==ret.size()-1?'\n':' ');
	return 0;
}
