#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int a,id;
} a[N];
int i,j,k,n,m,ge,cen;
int f[N],b[N];
int64 K;
inline bool cc1(const ww &a,const ww &b) {
	return a.a<b.a;
}
inline void Do() {
	int i,j;
	For(i,1,n) if (!f[i]) {
		int A=cen+1;
		K--;
		if (A==b[i]) f[i]=1;
		if (!K) break;
	}
	for (j=i+1;j<=n;j++) if (!f[j]) printf("%d ",j);
	For(j,1,i) if (!f[j]) printf("%d ",j);
	printf("\n");
	exit(0);
}
int main() {
	scanf("%d%I64d",&n,&K);
	For(i,1,n) scanf("%d",&a[i].a),a[i].id=i,b[i]=a[i].a;
	sort(a+1,a+n+1,cc1);
	ge=n;
	for (i=1;i<=n;i=j) {
		int64 A=a[i].a-a[i-1].a;
		if (A*ge>=K) {
			cen=a[i-1].a+K/ge;
			K%=ge;
			if (!K) K=ge,cen--;
			Do();
		}
		K-=A*ge;
		for (j=i;j<=n&&a[j].a==a[i].a;j++) f[a[j].id]=1,ge--;
	}
	printf("-1\n");
	return 0;
}
