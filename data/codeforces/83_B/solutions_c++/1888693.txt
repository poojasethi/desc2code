#include <cstdio>
#include <algorithm>

using namespace std;

int a[100001],b[100001];

int main(void)
{
  int n;
  long long s;
  scanf("%d %I64d",&n,&s);
  for(int i=1;i<=n;++i)
  {
    scanf("%d",&a[i]);
    b[i]=a[i];
  }
  sort(b+1,b+n+1);
  int m=n,p=0;
  while(m>0)
  {
    int i=n-m+1;
    if(1ll*m*(b[i]-p)>=s)
      break;
    s-=1ll*(m--)*(b[i]-p);
    p=b[i];
  }
  if(m==0)
  {
    printf("-1\n");
    return 0;
  }
  if(m==1)
  {
    for(int i=1;i<=n;++i)
      if(a[i]-p>s)
        printf("%d",i);
    printf("\n");
    return 0;
  }
  int d=b[n-m+1],c=0,t=1;
  while(c<s%m)
  {
    if(a[t]>=d)
      ++c;
    ++t;
  }
  for(int i=t;i<=n;++i)
    if(a[i]>p+s/m)
      printf("%d ",i);
  for(int i=1;i<t;++i)
    if(a[i]>p+s/m+1)
      printf("%d ",i);
  printf("\n");
  return 0;
}
