#include <iostream>
#include <cstdio>
#include <map>   
#include <vector>
using namespace std;
int n,i,a[100001],u;
long long k,sum; 
int main()
{
cin>>n>>k;
for (i=0;i<n;i++)
{cin>>a[i]; sum+=a[i];}
if (sum<k){
cout<<-1;
return 0;}
int l=0,r=1000000000;
while (l+1<r)
{
	int m=(l+r)/2;
	long long cur=0;
	for (i=0;i<n;i++)
		cur+=min(a[i],m);
	if (cur>=k) r=m;else l=m;
}
r--;
long long cur=0;
for (i=0;i<n;i++)
	cur+=min(r,a[i]),a[i]-=r;
cur=k-cur;
if (cur==0)  {
	for (i=0;i<n;i++)
		if (a[i]>0)	
			cout<<i+1<<" ";
			//cout<<"e"; 
			}
else {
	for (i=0;i<n;i++)
		if (a[i]>0)
		{	--cur;
				if (cur==0) 
			{	cur=i;break;}
				  }
	for (i=cur+1;i<n;i++)
		if (a[i]>0)
			cout<<i+1<<" ";
	for (i=0;i<=cur;i++)
		if (a[i]-1>0)
			cout<<i+1<<" ";
}


return 0;
}
