#include <iostream>

#define int long long

using namespace std;

int n, k, a[1000 * 1000];
int f[1000 * 1000];

main()
{
	cin >> n >> k;
	for (int i = 0; i < n; i++)
		cin >> a[i];

	int nn = n;
	while (k >= n && n > 0)
	{
		int v = k / n;
		for (int i = 0; i < nn; i++)
			if (!f[i])
			{
				if (a[i] > v)
					a[i] -= v,
					k -= v;
				else
					k -= a[i],
					a[i] -= a[i],
					f[i] = 1,
					n--;
			}

		//for (int i = 0; i < nn; i++)
		//	cerr << a[i] << ' ';
		//cerr << endl;

		//cerr << n << ' ' << k << endl;
	}

	//for (int i = 0; i < nn; i++)
	//	cerr << f[i] << ' ';
	//cerr << endl;

	if (n == 0 && k)
		cout << -1 << endl;
	else if (n)
	{
		int i, j;
		for (i = 0, j = 0; j < k; i++) if (!f[i])
		{
			a[i]--;
			if (a[i] == 0)
				f[i] = 1,
				n--;
			j++;
		}

		for (int j = 0; j < nn; j++)
			if (!f[(j + i) % nn])
				cout << (j + i) % nn + 1 << ' ';
		cout << endl;
	}

	return 0;
}
