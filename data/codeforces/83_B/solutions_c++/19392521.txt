#include<iostream>
#include<cstdio>
using namespace std;
typedef long long ll;
int a[100005];
int n;
ll k,sum=0;
ll min(ll q,ll p) {
	if(q>p) return p;
	return q;
}
bool check(ll m) {
	ll t=0;
	for(int i=1;i<=n;i++) {
		t+=min(m,a[i]);
		if(t>=k) return false;
	} 
	//cout<<"m"<<m<<"t"<<t<<endl;
	return true;
} 
int main() {
	scanf("%d%I64d",&n,&k);
	for(int i=1;i<=n;i++) {
		scanf("%d",&a[i]);
		sum+=a[i];
	}
	if(sum<k) printf("-1\n");
	else if(sum==k|| sum==0) printf("\n");
	else {
		ll l=1,r=1e9,mid=0;
		while(l<r) {
			mid=l+(r-l)/2;
			if(check(mid)) l=mid+1;
			else r=mid;	
		}
		for(int i=1;i<=n;i++) {
			int temp=min(a[i],l-1);
			a[i]-=temp;
			k-=temp;
		}
		int pos=0;
		for(int i=1;i<=n;i++) {
			if(a[i]>=1) {
				if(k>0) {
					k--;a[i]--;
				}	
			}
			if(k==0) {
				pos=i;break;
			}
		}
		for(int j=pos+1;j<=n;j++) {
			if(a[j]>=1) printf("%d ",j);
		}
		for(int j=1;j<=pos;j++) {
			if(a[j]>=1) printf("%d ",j);
		}
		printf("\n");
	}
	return 0;
}
