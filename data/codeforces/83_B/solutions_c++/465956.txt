#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;

const int N = 1e5 + 3;
int a[N];
__int64 n, k;

int main()
{
    __int64 sum = 0;
    cin>>n>>k;
    int i, j;
    for(i = 0; i < n; i ++)
    {
        scanf("%I64d", &a[i]);
        sum += a[i];
    }
    if( sum < k )
    {
        printf("-1\n");
        return 0;
    }
    int nn = n;
    __int64 sb = 0, ssb;
    sum = k; i = 0;
    while( n > 0 && sum > 0)
    {
        sb = max( sum / n, 1LL );
        for(i = 0; i < nn; i ++)
        {
            if( ! n || !sum ) break;
            if( a[i] )
            {
                ssb = min( sb, (__int64)a[i] );
                sum -= ssb; a[i] -= ssb;
                if( !a[i] ) n --;
            }
        }
    }
    i %= nn;
    for(j = 0; j < nn; j ++)
    {
        if( a[i] ) printf("%d ",i + 1);
        i = (i + 1) % nn;
    }
    cout<<endl;

    //while(1);
    return 0;
}
