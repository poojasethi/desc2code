#include <cstdio>
#include <cstdlib>
#include <iostream>

using namespace std;
int n, niz[ 100005 ];
long long k, S = 0, ss = 0;

int main( void ) {
	cin >> n >> k;
	for( int i = 0; i < n; ++i ) {
		cin >> niz[ i ];
		ss += niz[ i ];
	}
	
	int a = 0, b = 1000000001;
	while( a < b ) {
		int mid = ( a + b + 1 ) / 2;
		
		long long suma = 0;
		for( int i = 0; i < n; ++i )
			suma += min( niz[ i ], mid );
		
		if( suma > k ) b = mid - 1; else { a = mid; S = suma; }
	}
	
	int pom = k - S, ind = 0;
	while( 1 ) {
		while( ind < n && niz[ ind ] <= a ) ++ind;
		if( --pom < 0 || ind == n ) break; else ++ind;
	}
	
	for( int i = 0; i < n; ++i ) {
		int x = ( i + ind ) % n;
		if( niz[ x ] > a + ( x < ind ) )
			printf( "%d ", x + 1 );
	}
	
	if( ss != k && a == 1000000001 )
		printf( "-1" );
	printf( "\n" );
	
	return( 0 );
}
