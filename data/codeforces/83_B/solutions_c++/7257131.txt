#include <iostream>
#include <algorithm>
using namespace std;

int a[100001], b[100001], c[100001], d[100001], n, i, j;
long long k, x, y, z;
int main()
{
	cin>>n>>k;
	for (i=1; i<=n; i++) cin>>a[i], b[i]=a[i], z+=a[i];
	if (z<k) {
		cout<<-1; return 0;
	}
	sort(b, b+n+1);
	for (j=1,i=0; i<n; i++) if (b[i]<b[i+1]) {
		c[j]=n-i; d[j++]=b[i+1]-b[i];
	}
	for (y=0,i=1; i<j; i++) {
		x=c[i]; x*=d[i];
		if (k>=x) {
			k-=x; y+=d[i];
		}
		else {
			z=k/c[i];
			k-=z*c[i];
			y+=z;
			break;
		}
	}
	for (i=1; k>0; i++) if (a[i]>y) k--;
	for (j=i; j<=n; j++) if (a[j]>y) cout<<j<<" ";
	for (j=0; j<i; j++) if (a[j]>y+1) cout<<j<<" ";
	cout<<endl;
	return 0;
}
