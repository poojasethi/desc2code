#include <iostream>
#include <algorithm>
#include <cmath>
#include <vector>

#define rep(i,n) for(int i=0;i<n;i++)
#define ll       long long

using namespace std;

int n,a[100010];
vector<int> res,m;
ll l,r,k,s;

ll count(ll p){
  ll sum = 0;
  rep(i,n)sum+=(a[i]<=p?a[i]:p);
  return sum;
}

int main(){
  cin>>n>>k;rep(i,n)cin>>a[i],s+=a[i];
  if(s<=k){
    if(s<k)cout<<-1<<endl;return 0;
  }

  r=s;
  while(r-l>1){
    ll mid = (l+r)/2;
    if(count(mid)<=k)l = mid;
    else r = mid;
  }
  k-=count(l);

  rep(i,n){
    if(a[i]<=l)continue;
    if(k==0)res.push_back(i+1);
    else {
      k--;
      if(a[i]>l+1)m.push_back(i+1);
    }
  }
  rep(i,m.size())res.push_back(m[i]);
  rep(i,res.size())cout<<res[i]<<" ";cout<<endl;

}
