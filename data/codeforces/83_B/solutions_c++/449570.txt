#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

typedef long long ll;

int n;
ll k;
int a[110000], b[110000];

int main() {
	cin >> n >> k;
	
	ll sum = 0;
	for (int i=0; i<n; ++i) {
		cin >> a[i];
		b[i] = a[i];
		sum += a[i];
	}
	
	if (sum < k) {
		printf("-1\n");
		return 0;
	}
	if (sum == k) {
		return 0;
	}
	
	sort(b, b+n);
	
	ll cnt = n, ps = 0;
	for (int i=0; i<n; ) {		
		k -= cnt * (b[i] - ps);
		if (k < 0) {
			k += cnt * (b[i] - ps);
						
			bool eq = true;
			if (k > cnt * (b[i] - 1 - ps)) eq = false;
			
			k = (k % cnt) + 1;
			
			int r = 0;
			for (int j=0; j<n; ++j) if (a[j] >= b[i]) {
				++r;
				if (r == k) {
					for (int t=j; t<n; ++t) if (a[t] >= b[i]) printf("%d ", t + 1);
					for (int t=0; t<j; ++t) {
						if ((a[t] > b[i]) || (a[t] == b[i] && eq))
							printf("%d ", t+1);
					}
					printf("\n");
					return 0;
				}
			}
		}
		
		int j = i;
		while ((j<n) && (b[j] == b[i])) ++j;
		cnt -= (j - i);
		ps = b[i];
		i = j;
	}
	
	
	return 0;
}
