#include<iostream>
#include<cstdio>
#include<algorithm>
#define LL long long
using namespace std;
const int INFI=1000000001;

LL a[100100];
LL n,k;
LL check(LL m){
	LL sum=0;
	for(int i=1;i<=n;i++) sum+=min(a[i],m);
	return sum;
}

int main(){
	
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);	
	
	while(~scanf("%I64d%I64d",&n,&k)){
		for(int i=1;i<=n;i++){
			scanf("%I64d",&a[i]);
		}
		if(check(INFI)<k) printf("-1\n");
		else {
			LL ans,mid,l=0,r=INFI;
			while(l<=r){
				mid=(l+r)>>1;
				if(check(mid)<=k){
					ans=mid;
					l=mid+1;
				}
				else r=mid-1;
			}
			k-=check(ans);
			int t=0;
			for(int i=1;i<=n,k;i++){
				if(a[i]>ans){
					--a[i];--k;
					t=i;
				}
			}
			for(int i=t+1;i<=n;i++){
				if(a[i]>ans) printf("%d ",i);
			}
			for(int i=1;i<=t;i++){
				if(a[i]>ans) printf("%d ",i);
			}
			printf("\n");
		}
				
	}
	
	return 0;
}
