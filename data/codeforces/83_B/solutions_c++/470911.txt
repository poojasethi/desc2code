#include<cstdio>
#include<iostream>
using namespace std;

const int maxn=100001;
long long kth;
int n,a[maxn],ret[maxn];

void init()
{
	cin>>n>>kth;
	for (int i=0; i<n; i++) scanf("%d",&a[i]);
}

long long check(int mid)
{
	long long ret=0;
	for (int i=0; i<n; i++) ret+=min(a[i],mid);
	return ret;
}

void solve()
{
	int le=0,ri=1000000001,mid,ans;
	while (le<=ri)
	{
		mid=(le+ri)/2;
		if (check(mid)<=kth)
		{
			ans=mid;
			le=mid+1;
		} else ri=mid-1;
	}
	if (check(1000000001)<kth)
	{
		cout<<-1<<endl;
		return;
	}
	kth-=check(ans);
	le=ri=0;
	for (int i=0; i<n; i++)
		if (a[i]-ans>0) ret[ri++]=i;
	for (int i=0; i<kth; i++)
		if (a[ret[i]]-ans-1>0) ret[ri++]=ret[i];
	for (int i=kth; i<ri; i++) printf("%d ",ret[i]+1);
}

int main()
{
	init();
	solve();
	return 0;
}
