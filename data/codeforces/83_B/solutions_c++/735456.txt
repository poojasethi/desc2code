#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;
const int N=100005;
long long a[N],b[N];

int main()
{
    long long i,l,x,y,k,m,n,sum=0;
    cin>>n>>m;
    for (i=1; i<=n; i++)
    {
        scanf("%lld",&a[i]);
        b[i]=a[i]; sum=sum+a[i];
    }
    if (sum<m) {printf("-1\n"); return 0;}
    if (sum==m) {return 0;}
    sort(b+1,b+n+1);
    l=1; x=n; y=0;
    for (i=2; i<=n+1; i++)
     if (i==n+1 || b[i]!=b[i-1])
     {
         if (x*(b[i-1]-y)<=m)
         {
             m-=x*(b[i-1]-y);
             x-=l; y=b[i-1];
             l=1;
         }
         else break;
     }
     else l++;
    y+=m/x; m=m%x;
    for (i=1; i<=n; i++) if (a[i]>y) a[i]-=y; else a[i]=0;
    if (m>0)
    for (i=1; i<=n; i++)
     if (a[i]>0) {a[i]--; m--; if (m==0) break;}
    k=i;
    for (i=k+1; i<=n; i++)
     if (a[i]>0) printf("%lld ",i);
    for (i=1; i<=k; i++)
     if (a[i]>0) printf("%lld ",i);
}
