#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
using namespace std;
typedef long long ll;
const ll inf = 1e14;
const int N = 1000005;
int n;
ll a[N];
queue<int> q;
ll check(ll times) {
	ll ret = 0;
	for(int i = 0; i < n; i ++) {
		ret += min(times, a[i]);
	}
	return ret;
}
int main() {
	ll m, s = 0;
	scanf("%d%I64d", &n, &m);
	for(int i = 0; i < n ; i ++) {
		scanf("%I64d", &a[i]);
		s += a[i];
	}
	if(s <= m) {
		if(s != m) puts("-1");
		return 0;
	}
	ll l = 0, r = inf;
	while(l + 1 < r) {
		ll mid = (l + r + 1) >> 1;
		if(check(mid) > m) r = mid;
		else l = mid;
	}
//	printf("%d\n", l);
	for(int i = 0; i < n; i ++) {
		if(a[i] > l) {
			q.push(i);
			a[i] -= l;
			m -= l;
		} else {
			m -= a[i];
		}
	}
	while(m > 0) {
		int pos = q.front(); q.pop();
		if(a[pos] > 1) {
			a[pos] --;
			q.push(pos);
		}
		m--;
	}
	while(!q.empty()) {
		printf("%d ", q.front() + 1);
		q.pop();
	} puts("");
	return 0;
}
