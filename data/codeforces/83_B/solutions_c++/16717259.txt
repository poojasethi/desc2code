#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
#define ll __int64
const int N=100001;
int n;
ll k;
int a[N],b[N];
queue<int>q;
int main()
{
	int i,m,min;
	ll d,sum;
	scanf("%d%I64d",&n,&k);
	for(sum=0,i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		sum+=a[i];
		b[i]=a[i];
	}
	if(sum<k)
		printf("-1\n");
	else if(sum==k)
		return 0;
	else
	{
		sort(b+1,b+n+1);
		i=1,d=0;
		while(1)
		{
			min=b[i];
			m=n-i+1;
			while(b[i]==min)i++;
			min-=d;
			if(k/min>m)
				k-=(ll)m*min,d+=min;
			else
			{
				d+=k/m,k-=k/m*m;
				for(i=1;i<=n;i++)
				{
					if(a[i]>d)
					{
						if(!k)printf("%d ",i);
						else 
						{
							k--;
							if(a[i]>d+1)q.push(i);
						}
					}
				}
				while(!q.empty())printf("%d ",q.front()),q.pop();
				printf("\n");
				return 0;
			}
		}
	}
	return 0;
}
