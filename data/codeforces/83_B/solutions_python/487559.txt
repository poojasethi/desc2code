#!/usr/bin/env python
# -*- coding: utf-8 -*-
import itertools

fc = 0
n, k = map(int, raw_input().split())
sa = list()
for i, a in enumerate(itertools.imap(int, raw_input().split())):
    sa.append((a, i+1))
sa = sorted(sa)[::-1]
while (len(sa) <= k and len(sa) != 0):
    fc += k / len(sa)
    k -= len(sa) * (k / len(sa))
    try:
        while (sa[-1][0] <= fc):
            k += fc - sa.pop(-1)[0]
    except IndexError:
        pass
if (len(sa) != 0):
    ans = [(x2, x1) for x1, x2 in sa]
    ans.sort()
    for i in xrange(k, len(sa)):
        print ans[i][0],
    fc += 1
    for i in xrange(0, k):
        if fc < ans[i][1]:
            print ans[i][0],
    print ''
else:
    if (k > 0):
        print '-1'
