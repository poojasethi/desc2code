#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
const int N = 1005;
int n;
int ji[N],ou[N],nj,no;
int nopri[2000006];
int main()
{
	for(int i=2;i<=2000000;++i)
	{
		if(nopri[i]) continue;
		for(int j=i+i;j<=2000000;j+=i)
			nopri[j]=1;
	}
	int n1=0;
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
	{
		int t;
		scanf("%d",&t);
		if(t==1) ++n1;
		if(t%2) ji[++nj]=t;
		else ou[++no]=t;
	}
	
	if(n1<=1)
	{
		for(int i=1;i<=nj;++i)
			for(int j=1;j<=no;++j)
			{
				if(nopri[ji[i]+ou[j]]==0)
				{
					printf("%d\n%d %d\n",2,ji[i],ou[j]);
					return 0;
				}
			}
		if(nj)
			printf("1\n%d\n",ji[1]);
		else printf("1\n%d\n",ou[1]);
	}
	else
	{
		int iso=0;
		
		for(int i=1;i<=no;++i)
			if(nopri[1+ou[i]]==0)
			{
				iso=ou[i];
				break;
			}
		if(!iso)
		{
			printf("%d\n1",n1);
			for(int i=2;i<=n1;++i) printf(" 1");
		}
		else
		{
			printf("%d\n1",n1+1);
			for(int i=2;i<=n1;++i) printf(" 1");
			printf(" %d",iso);
		}
		
				
		
	}
	
}
