#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;
int prime[2010000],p[2010000],tot;
int ans[101000],cnt,n,temp;
int a[1010];
int main()
{
	for (int i = 2;i <= 2000000;i++)
	{
		if (p[i] == 0) p[i] = i,prime[++tot]=i;
		for (int j = 1;j <= tot && prime[j]*i <= 2000000;j++)
		{
			p[prime[j]*i] = prime[j];
			if (prime[j] >= p[i]) break;
		}
	}

	cin >> n;
	for (int i = 1;i <= n;i++) scanf("%d",&a[i]);
	int can = 0;
	for (int i = 1;i <= n;i++)
	{
		if (p[a[i]+1] == a[i] + 1 && a[i] > 1 && can == 0) can = 1,ans[++cnt] = a[i];
		if (a[i] == 1) ans[++cnt] = 1;
	}
	if (cnt >= 2) {cout << cnt << endl;
		for (int i = 1;i <= cnt;i++) printf("%d ", ans[i]); return 0;}
	for (int i = 1;i <= n;i++)
		for (int j = i + 1;j <= n;j++)
			if (p[a[i] + a[j]] == a[i] + a[j])
			{
				cout << 2 << endl;
				cout << a[i] <<" " << a[j] << endl;
				return 0;
			}
	cout << 1 << endl;
	cout << a[1] << endl;
}
