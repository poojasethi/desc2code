#include<cstdio>
#include<algorithm>
using namespace std;
#define MAXN 1000
#define MAXA 2000000
int a[MAXN+10],n,p[MAXA+10],pcnt;
bool f[MAXA+10];
void prepare(){
	int i,j;
	for(i=2;i<=MAXA;i++){
		if(!f[i])
			p[++pcnt]=i;
		for(j=1;i*p[j]<=MAXA;j++){
			f[i*p[j]]=1;
			if(i%p[j]==0)
				break;
		}
	}
}

void Read(int &x){
	char c;
	while(c=getchar(),c!=EOF)
		if(c>='0'&&c<='9'){
			x=c-'0';
			while(c=getchar(),c>='0'&&c<='9')
				x=x*10+c-'0';
			ungetc(c,stdin);
			return;
		}
}
void read(){
	Read(n);
	int i;
	for(i=1;i<=n;i++)
		Read(a[i]);
	sort(a+1,a+n+1);
}
void solve(){
	int i,j;
	j=upper_bound(a+1,a+n+1,1)-a-1;
	if(j){
		for(i=j+1;i<=n;i++)
			if(!f[a[i]+1]){
				printf("%d\n",j+1);
				while(j--)
					printf("1 ");
				printf("%d\n",a[i]);
				exit(0);
			}
			if(j>=2){
				printf("%d\n",j);
				while(--j)
					printf("1 ");
				puts("1");
				exit(0);
			}
		}
	for(i=1;i<=n;i++)
		for(j=i+1;j<=n;j++)
			if(!f[a[i]+a[j]]){
				printf("2\n%d %d\n",a[i],a[j]);
				exit(0);
			}
	printf("1\n%d\n",a[1]);
}
int main()
{
	prepare();
	read();
	solve();
}
