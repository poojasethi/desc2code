#include <bits/stdc++.h>
using namespace std;
const int maxn=1233,maxm=1e7+5;
int arr[maxn],n,m,a1,a2,isp[maxm];
int main()
{
    isp[1]=1;
    for(int i=2; i<maxm; ++i)
        if(!isp[i])
            for(int j=i*2; j<maxm; j+=i)
                isp[j]=1;
    cin>>n;
    for(int  i=0; i<n; ++i) {
        cin>>arr[i];
        if(arr[i]==1)
            ++m;
    }
    for(int i=0; i<n; ++i) {
        if(!isp[arr[i]+1]&&m&&arr[i]!=1) {
            cout<<m+1<<endl;
            for(int i=0; i<m; ++i)cout<<1<<' ';
            cout<<arr[i]<<endl;
            return 0;
        }
        for(int j=i+1; j<n; ++j)
            if(!isp[arr[j]+arr[i]])
                a1=arr[i],a2=arr[j];
    }
    if(m>2) {
        cout<<m<<endl;
        for(int i=0; i<m; ++i)cout<<1<<' ';
        return 0;
    }
    if(a1) {
        cout<<2<<endl<<a1<<' '<<a2<<endl;
    } else
        cout<<1<<endl<<arr[0]<<endl;
    return 0;
}
