#include <cstdio>
#include <cstdlib>
#include <algorithm>
#define MAXN 1005
using namespace std;
int n, m, p[2000005], cnt;
bool v[2000005];
void Shaker() {
    for(int i = 2; i <= 2000000; ++ i) {
        if(!v[i]) p[++ cnt] = i;
        for(int j = 1; j <= cnt && p[j] * i <= 2000000; ++ j) {
            v[i*p[j]] = 1;
            if(!(i%p[j])) break;
        }
    }
}
int a[MAXN], b[MAXN], c[MAXN];
int main() {
    Shaker();
    scanf("%d", &n);
    int ans = 0, cnt = 0; srand(100086);
    for(int i = 1; i <= n; ++ i) scanf("%d", a + i);
    for(int T = 1; T <= 300; ++ T) {
        cnt = 0;
        for(int i = 1; i <= n; ++ i) {
            bool f = 1;
            for(int j = 1; j <= cnt; ++ j)
                if(v[a[i] + b[j]]) { f = 0; break; }
            if(f) b[++ cnt] = a[i];
        }
        if(ans < cnt) {
            ans = cnt;
            for(int i = 1; i <= cnt; ++ i) c[i] = b[i];
        } random_shuffle(a + 1, a + n + 1);
    }
    printf("%d\n", ans);
    for(int i = 1; i <= ans; ++ i)
        printf("%d%c", c[i], i == ans ? '\n' : ' ');
    return 0;
}
