#include<bits/stdc++.h>
using namespace std;
int a,s,d[2000003],f[10002],g[1000002],h,j,k,l,i,n,m;
string x;
void prnt(){
    for(i=2;i<=1000000;i++){
        if(d[i+1]==0 && g[i]>0) {h=i;break;}
    }
    cout<<g[1]+(h>0)<<endl;
    for(i=0;i<g[1];i++)
        cout<<1<<" ";
    if(h) cout<<h;
}
main(){
    cin>>n;
    for(i=2;i<2000000;i++){
        for(a=2*i;a<=2000000;a+=i) d[a]=1;
    }
    for(i=0;i<n;i++){
        cin>>f[i];
        g[f[i]]++;
    }
    if(g[1]>=2) {prnt();return 0;}

    for(i=0;i<n;i++){
        for(a=i+1;a<n;a++){
            if(d[f[i]+f[a]]==0) {
                cout<<2<<endl<<f[i]<<" "<<f[a]<<endl;
                return 0;
            }
        }
    }

    cout<<1<<endl<<f[0];
}
