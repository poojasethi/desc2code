#include<bits/stdc++.h>
using namespace std;
int p[2000005],a[1005];
int main()
{
	for(int i=2;i<=2000000;i++)
	{
		if(p[i]==0)
		{
			for(long long j=1ll*i*i;j<=2000000;j+=i)
			p[j]=1;
		}
	}
	p[1]=1;
	int n;
	cin>>n;
	int c=0;
	int first=0,second=0;
	for(int i=1;i<=n;i++)
	{
		int x;
		cin>>a[i];
		if(a[i]==1)
		c++;
		else
		{
			for(int j=1;j<i;j++)
			{
				if(a[j]!=1&&p[a[j]+a[i]]==0)
				{
					first=a[j];
					second=a[i];
				}
			}
			if(first==0&&second==0&&p[a[i]]==0)
			first=a[i];
		}
		
	}
	int f=0;
	for(int i=1;i<=n;i++)
	{
		if(a[i]!=1&&p[1+a[i]]==0)
		f=a[i];
	}
	if(c==0)
	f=0;
	if(first&&second)
	{
		if(c+(f?1:0)<2)
		{
			cout<<2<<endl<<first<<" "<<second<<endl;
		}
		else
		{
			cout<<c+(f?1:0)<<endl;
			for(int i=1;i<=c;i++)
			cout<<1<<" ";
			if(f)
			cout<<f<<endl;
		}
	}
	else 
	{
		if(c+(f?1:0)<1)
		{
			cout<<1<<endl<<a[1]<<endl;
		}
		else
		{
			cout<<c+(f?1:0)<<endl;
			for(int i=1;i<=c;i++)
			cout<<1<<" ";
			if(f)
			cout<<f<<endl;
		}
	}
}
