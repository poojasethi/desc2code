#include <iostream>
using namespace std;
int fix[2000001];
int a[1001],x[1001],nx;
int i,j,n,ne;
main(){
  
  for (i=2;i*i<=2000000;i++)
   if (fix[i]==0)
    for (j=i*i;j<=2000000;j=j+i)
     fix[j]=1;     
       
  cin>>n;
  for (i=1;i<=n;i++){
   cin>>a[i];
   if (a[i]==1) nx++;
  }      
  
  for (i=1;i<=nx;i++) x[i]=1;
  
  if (nx>0) 
   for (i=1;i<=n;i++)
    if (a[i]!=1 && fix[a[i]+1]==0) {
     x[++nx]=a[i]; break;
  }
  
  if (nx>=2) {
   cout<<nx<<endl;
   for (i=1;i<=nx;i++)    
    cout<<x[i]<<" ";
    return 0;
  }
  
  for (i=1;i<n;i++)
   for (j=i+1;j<=n;j++)
    if (fix[a[i]+a[j]]==0) {     
     cout<<2<<endl;      
     cout<<a[i]<<" "<<a[j]<<endl;
     return 0;          
    }
    
  
    cout<<1<<endl;
    cout<<a[1]<<endl;
       
//   system("pause");    
       
}
