#include <iostream>
#include <vector>
#include <cstdlib>

using namespace std;

void morir(vector<int> sol)
{
  cout<<int(sol.size())<<endl;
  for (int i=0;i<int(sol.size());i++) {
    if (i>0) cout<<" ";
    cout<<sol[i];
  }
  cout<<endl;
  exit(0);
}

const int limite=3000000;

int n;
int esprimo[limite];
int a[limite];

int main()
{
  for (int i=2;i<limite;i++)
    esprimo[i]=1;
  for (int i=2;i<limite;i++)
    if (esprimo[i])
      for (int j=i+i;j<limite;j+=i)
	esprimo[j]=0;
  cin>>n;
  int num1=0;
  for (int i=1;i<=n;i++) {
    cin>>a[i];
    num1+=a[i]==1;
  }
  vector<int> sol;
  if (num1>=2) {
    bool trobat=false;
    for (int i=1;i<=n;i++) {
      if (a[i]==1) sol.push_back(1);
      else if (not trobat and esprimo[a[i]+1]) {
	sol.push_back(a[i]);
	trobat=true;
      }
    }
    morir(sol);
  }
  for (int i=1;i<=n;i++) {
    for (int j=i+1;j<=n;j++) {
      if (esprimo[a[i]+a[j]]) {
	sol.push_back(a[i]);
	sol.push_back(a[j]);
	morir(sol);
      }
    }
  }
  sol.push_back(a[1]);
  morir(sol);
}
