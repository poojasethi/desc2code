#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#include <deque>
#include <set>
#include <map>
using namespace std;

int cb[2000020];
void sp ()
{
	cb[1] = 1;
	for (int i = 2; i*i <= 2000010; i++)
	{
		if (cb[i]) continue;
		for (int j = i+i; j <= 2000010; j+=i) cb[j] = 1;
	}
}
int main()
{
	sp();
	int n;
	cin >> n;
	vector <int> arr (n);
	int o = 0;
	int cp = 0;
	for (int i = 0; i < n; i++)
	{
		scanf("%d",&arr[i]);
		if (arr[i] == 1) o++;
		else if (!cb[arr[i]+1]) cp = arr[i];
	}
	if (o >= 2)
	{
		if (cp != 0) printf("%d\n%d ",o+1,cp);
		else printf("%d\n",o);
		for (int i = 0; i < o; i++)
		{
			if (i != 0) printf(" ");
			printf("%d",1);
		}
	}
	else
	{
		for (int i = 0; i < n; i++)
		{
			for (int j = i+1; j < n; j++)
			{
				if (!cb[arr[i]+arr[j]])
				{
					printf("2\n%d %d",arr[i],arr[j]);
					return 0;
				}
			}
		}
		printf("1\n%d",arr[0]);
	}
	
}
