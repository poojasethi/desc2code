#define problem "665C"
#include <cstdio>
#include <iostream>
#include <memory.h>
using namespace std;
int prime[2000010];

void sieve(const int N){
    memset(prime, 1, sizeof(prime));
    for(int i = 3; i * i <= N; i += 2) if(prime[i])
        for(int j = i * i; j <= N; j += 2 * i) prime[j] = 0;
    for(int i = 4; i <= N; i += 2) prime[i] = 0;
}

int main(){
    sieve(2e6);
    int N, A[2000], C(0);
   
    cin >> N;
    for(int i = 0; i < N; i++) cin >> A[i], C += A[i] == 1;
    if(C > 1){
        for(int i = 0; i < N; i++) if(A[i] > 1 && prime[A[i] + 1]){
            cout << C + 1 << '\n';
            for(int j = 0; j < C; j++) cout << 1 << ' ';
            cout << A[i];
            return 0;
        }
        cout << C << '\n';
        for(int i = 0; i < C; i++) cout << 1 << ' ';
        return 0;
    }else{
        for(int i = 0; i < N - 1; i++) for(int j = i + 1; j < N; j++) if(prime[A[i] + A[j]]){
            cout << 2 << '\n' << A[i] << ' ' << A[j];
            return 0;
        }
    }
    cout << 1 << '\n' << A[0];
}
