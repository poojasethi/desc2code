#include <bits/stdc++.h>

using namespace std;

int main(int argc, char **argv)
{                   
	
	long long n;
	cin >> n;
	vector<long long> A (n);
	
	for (long long i = 0; i < n; i++)
		cin >> A[i];
		//A[i] = i + 1;
	sort(A.begin(), A.end());
	
	
	vector<long long> RES;
	vector<bool> B;
	B.assign( A[n -1]*2 + 1, true);
	
	

	for (long long i = 2; i*i <= A[n -1]*2; i++)
		if (B[i])
			for (long long j = i*i; j <= A[n -1]*2; j += i)
				B[j] = false;

	//if >=  2 1
	if (count(A.begin(), A.end(), 1) >= 2)
	{
		for (int i = 0; i < n; i++)
		{
			if (A[i] == 1)
				RES.push_back(A[i]);
			else 
			{
				if (B[A[i] + 1])
				{
					RES.push_back(A[i]);
					break;
				}
			}
		}
		
		cout << RES.size() << "\n";
		for (int i = 0; i < RES.size(); i++)
			cout << RES[i] << " ";
		exit(0);
	}
	
	for (int i = 0; i < n - 1; i++)
		for (int j = i + 1; j < n; j++)
		{
			if (B[A[i] + A[j]])
			{
				cout << "2\n" << A[i] << " " << A[j];//system("PAUSE");
				exit(0);
			}
		}
	
	cout << "1\n" << A[0];

	
    return 0;
}
