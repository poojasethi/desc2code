#include <iostream>
#include <math.h>
#include <stdio.h>
using namespace std;
int n,a[5000005],f[5000005],d[5000005],k;
int go(int x)
{
	if(x<2) return 0;
	if(x==2) return 1;
	
	if(f[x])return d[x];
	f[x]=1;
	
	for(int i=2; i<=sqrt(x); i++)
		if(x%i==0) return 0;
	
	d[x]=1;
	return 1;
}
main ()
{
	cin>>n;
	
	
	for(int i=0; i<n; i++)
		scanf("%d",&a[i]);
	
	
	for(int i=0; i<n; i++)
		if(a[i]==1) k++;
	
	if(k>=2)
	for(int i=0; i<n; i++)
	{
		if(a[i]!=1)
		{
			if(go(1+a[i])==0) continue;
			
			cout<<k+1<<endl;
			
			for(int i=0; i<k; i++)
				cout<<1<<" ";
			
			cout<<a[i]<<endl;
			
			return 0;
		}
	}
	
	if(k>=2)
	{
		cout<<k<<endl;
			for(int i=0; i<k; i++)
				cout<<1<<" ";
			return 0;		
	}
	
	
		for(int i=0; i<n; i++)
		for(int j=i+1; j<n; j++)
		{
			if(go(a[i]+a[j]))
			{
				cout<<2<<endl<<a[i]<<" "<<a[j]<<endl;
				return 0;
			}
		}
		
		
	cout<<1<<endl<<a[0]<<endl;
	
	

	cin>>n;
}
