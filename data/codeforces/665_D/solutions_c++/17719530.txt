#include<cstdio>
#include<algorithm>
using namespace std;
int n,ans,now,i,j,a[1010];
bool prime[2000010];
int main()
{
	for (i=2;i<=1500;i++)
	{
	    if (prime[i]) continue;
		for (j=i*i;j<=2000000;j+=i)
		  prime[j]=1;
    }
	scanf("%d",&n);
	for (i=1;i<=n;i++) scanf("%d",&a[i]);
    sort(a+1,a+n+1);
    //ans=2
    //ans=3   1,1,x
    if (a[1]==1&&a[2]==1)
    {
    	ans=2;
    	now=3;
    	while (a[now]==1) {now++;ans++;}
    	if (now>n)
    	{
    		printf("%d\n",n);
    	    for (j=1;j<=n;j++) printf("1 ");
    	    printf("\n");
		}
		else
		{
    	for (i=now;i<=n;i++)
    	  if (!prime[a[i]+1])
    	  {
    	  	ans++;
    	  	break;
		  }
    	printf("%d\n",ans);
    	for (j=1;j<=now-1;j++) printf("1 ");
    	if (now==ans) printf("%d\n",a[i]);
        }
		return 0;  
	}
	for (i=1;i<=n;i++)
	  for (j=i+1;j<=n;j++)
	    if (!prime[a[i]+a[j]])
	    {
	    	printf("2\n");
	    	printf("%d %d\n",a[i],a[j]);
	    	return 0;
		}
	printf("1\n%d\n",a[1]);
	return 0;
}
