#include<bits/stdc++.h>
#define N 2000001
using namespace std;int i,j,m,one,x,cnt,n;
int prime[N],flag[N],a[1005],get[1005],ans[1005];
void compare(){
	if ((*get)<=(*ans)) return;
	memcpy(ans,get,sizeof(get));
}
int main(){
	for (flag[1]=1,i=2;i<N;i++){
		if (!flag[i]) prime[++cnt]=i;
		for (j=1;j<=cnt&&prime[j]*i<N;j++)
			flag[prime[j]*i]=1;
	}scanf("%d",&m);
	for (i=1;i<=m;i++){
		scanf("%d",&x);
		one+=x==1;if (x>1) a[++n]=x;
  }
  ans[*ans=1]=x;
  //include sum two
  if (one>1){
  	*get=one;
  	for (i=1;i<=*get;i++) get[i]=1;
  	for (i=1;i<=n;i++)
  		if (!flag[a[i]+1]){
  			get[++*get]=a[i];break;
  	  }
  	compare();
  }
  while (one--) a[++n]=1;
  for (i=1;i<n;i++)
  	for (j=i+1;j<=n;j++)
  		if (!flag[a[i]+a[j]])
  			*get=2,get[1]=a[i],get[2]=a[j],compare();
  printf("%d\n",*ans);
  for (i=1;i<=*ans;i++)
  	printf("%d ",ans[i]);
}
