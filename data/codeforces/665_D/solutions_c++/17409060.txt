
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;
const int N = 2000005;

LL memo[N];

void sieve()
{
LL i,j;

memo[1]=0;
memo[2]=1;

for(i=2;i<N;i++)
	if(memo[i]==1)
		for(j=i+i;j<N;j+=i)
			memo[j]=0;

}





int main()
{LL n,i,j,q=1;

	cin>>n;

vector <LL> v(n);

map <LL,LL> mp;

for(i=0;i<N;i++)
	memo[i]=1;




//for(i=1;i<100;i++)
//	cout<<i<<' '<<memo[i]<<endl;

sieve();

for(i=0;i<n;i++)
	{cin>>v[i];
		mp[v[i]]++;
		if(v[i]!=1 && memo[v[i]+1])
			q=v[i];
	}



if(mp[1]>1)
{
if(q!=1)
	cout<<mp[1]+1<<endl;
else
	cout<<mp[1]<<endl;

for(i=0;i<mp[1];i++)
	cout<<1<<' ';

if(q!=1)
	cout<<q<<endl;
else
	cout<<endl;

return 0;

}
else
{
for(i=0;i<n;i++)
	for(j=i+1;j<n;j++)
		if(memo[v[i]+v[j]])
		{
			cout<<2<<endl<<v[i]<<' '<<v[j]<<endl;
			return 0;
		}

}	

cout<<1<<endl<<v[0]<<endl;


}
