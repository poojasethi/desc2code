#include <iostream>
#include <vector>
#include <math.h>
using namespace std;
bool p[2000001]={},isc[2000001]={};
int main(){
	ios::sync_with_stdio(0);
	int n,ans=0,v1=0,a[1001]={},pf=0,pp=0;
	cin>>n;
	const int c=2000010;
	bool prime[c];
	for (int i=0;i<c;i++) prime[i] = true;
	prime[1] = false;
	for (int i=2;i*i<c;i++) if (prime[i])
		for (int j=i;i*j<c;j++) prime[i*j] = false;
	vector <int> os;
	for (int i=1;i<=n;i++){
		cin>>a[i];
		if (a[i]==1)
			v1++;
		else{
			if (pp==0){
				if (prime[a[i]])
					pp=a[i];					
			}
			if (pf==0){
				if (prime[a[i]+1])
					pf=a[i];	
			}
		}
	}
	if ((pf+v1)>1&&v1>0){
		ans=v1;
		if (pf>0)
			ans++;
		cout<<ans<<endl;
		for (int i=1;i<=v1;i++)
			cout<<1<<" ";
		if (pf>0)
			cout <<pf;
	}
	else{
		bool fl(0);
		for (int i=1;i<=n;i++){
			if (fl)
				break;
			for (int j=i+1;j<=n;j++){
				if (prime[a[i]+a[j]]){
					cout<<2<<endl<<a[i]<<" "<<a[j];
					fl++;
					ans=1;
					break;
				}
			}	
		}
	}
	if (ans==0){
		if (pp==0)
			cout<<1<<endl<<a[1];
		else 
			cout<<1<<endl<<pp;
	}
}
