#include<bits/stdc++.h>
using namespace std;

bool prime[(int)2e6+5];
int n, a[1003], one = 0;
int main() {
    memset(prime, 1, sizeof prime);
    prime[1] = 0;

    for(int i = 2; i <= (int)sqrt((double)2e6); ++i) if(prime[i])
        for(int j = i*i; j <= 2e6; j+=i) prime[j] = 0;
    scanf("%d",&n);
    for(int i = 0; i < n; ++i) {
        scanf("%d",&a[i]);
        if(a[i] == 1) ++one;
    }

    if(one >= 2) {
        int theOther = 0;
        for(int i = 0; i < n; ++i) if(a[i] > 1 && prime[a[i]+1]) theOther = a[i];
        printf("%d\n",theOther == 0? one: one+1);
        for(int i = 0; i < one; ++i) printf("1 ");
        if(theOther) printf("%d\n",theOther);
        return 0;
    }

    for(int i = 0; i < n; ++i)
        for(int j = 0; j < n; ++j) if(j != i && prime[a[i]+a[j]]) {
            printf("2\n%d %d\n", a[i],a[j]);
            return 0;
        }
    printf("1\n%d\n",a[0]);
}
