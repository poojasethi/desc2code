#include <bits/stdc++.h>
using namespace std;
#define mp make_pair
#define pb push_back
#define ff first
#define ss second
#define sd(x) scanf("%d",&x)
bool prime[2000005];
void seive(){
	prime[0]=prime[1]=false;
	for(int i=2;i<=2000000;i++){
		if(prime[i]){
			int k=2*i;
			for(int j=k;j<=2000000;j+=i){
				prime[j]=false;		
			}
		}
	}
}
int main(){
	memset(prime,true,sizeof(prime));
	seive();
	int n;
	sd(n);
	int a[n+1];
	int cnt=0;
	for(int i=0;i<n;i++){
		sd(a[i]);
		if(a[i]==1){
			cnt++;
		}
	}
	if(cnt>=2){
		vector<int>ans;
		for(int i=0;i<n;i++){
			if((a[i]!=1)and(prime[a[i]+1])){
				ans.pb(a[i]);
				break;
			}
		}
		for(int i=1;i<=cnt;i++){
			ans.pb(1);
		}
		cout<<ans.size()<<endl;
		for(int i=0;i<ans.size();i++){
			cout<<ans[i]<<" ";
		}
		return 0;
	}
	for(int i=0;i<n;i++){
			for(int j=i+1;j<n;j++){
				if(prime[a[i]+a[j]]){
					cout<<"2"<<endl;
					cout<<a[i]<<" "<<a[j];
					return 0;
				}
			}
		}
	cout<<"1"<<endl;
	cout<<a[0];
	return 0;
}
