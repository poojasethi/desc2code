#include<iostream>
using namespace std;
int m[2000001],fix[2000001],i,j,n,k,x,t;
main (){
  cin>>n;
  for (i=2; i*i<=2000000; i++)
    if (fix[i]==0)
      for (j=2*i; j<=2000000; j=j+i)
        fix[j]=1;
  for (i=1; i<=n; i++){
    cin>>m[i];
    if (m[i]==1) k++;
  }
    if (k>0){
    for (i=1; i<=n; i++)
      if (fix[m[i]+1]==0 && m[i]>1) {
        x=m[i];
        t=1;
        break;
      }  
    }
    if (k+t>=2){
      if (x!=0) cout<<k+1<<endl;
        else cout<<k<<endl;
      for (i=1; i<=k; i++)
        cout<<1<<" ";
      if (x!=0) cout<<x<<endl;
      return 0;
    }
    for (i=1; i<n; i++)
      for (j=i+1; j<=n; j++)
        if (fix[m[i]+m[j]]==0) {
          cout<<2<<endl;
          cout<<m[i]<<" "<<m[j];
          return 0;
        }
    cout<<1<<endl;
    cout<<m[1]<<endl;
}
