#include <bits/stdc++.h>
using namespace std;

const int N=2e6+5;
int is[N];
int a[1005];

void init()
{
	memset(is,1,sizeof(is));
	is[1]=is[0]=0;
	for(int i=2;i<N;i++)
	{
		if(is[i]==0)
			continue;
		for(int j=2*i;j<N;j+=i)
			is[j]=0;
	}
}
vector<int> v;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	init();

	int n;

	cin>>n;
	int ans=0;
	int cnt1=0;
	int other=0,oans=0;
	for(int i=1;i<=n;i++)
	{   cin>>a[i];
		if(a[i]==1)
			cnt1++;
		if(is[a[i]+1]&&a[i]!=1)
			{other++;oans=a[i];}
		if(true)
		{
			v.clear();
			v.push_back(a[i]);
		}

		}

	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
		if(is[a[i]+a[j]])
		{
		v.clear();
		v.push_back(a[i]);
		v.push_back(a[j]);	

		}
		if(cnt1>0)
	{
		if(cnt1==1&&other==0)
			;
		else{
		ans=cnt1+((other>=1)?1:0);
		v.clear();
		for(int i=1;i<=cnt1;i++)	
			v.push_back(1);
		if(other>=1)
			v.push_back(oans);
			}
	}	

	cout<<v.size()<<endl;
	for(int i=0;i<v.size();i++)
		cout<<v[i]<<" ";
	cout<<endl;

	return 0;
}
