#include<iostream>
using namespace std;
int fix[2000001],a[2000001],n,n1;
main()
{
for (int i=2;i*i<=2000000;i++)
if (fix[i]==0)
for (int j=i*i;j<=2000000;j=j+i)
fix[j]=1; 

cin>>n;
for(int i=1;n>=i;i++)
cin>>a[i];

for(int i=1;n>=i;i++)
if(a[i] == 1)
n1++;

if(n1 > 0)
for(int i=1;n>=i;i++)
if(fix[a[i]+1] == 0 && a[i] != 1)
{
cout<<n1+1<<endl;
cout<<a[i]<<" ";
for(int j=1;n1>=j;j++)
cout<<"1"<<" ";
return 0;
}

if(n1 > 2)
{
cout<<n1<<endl;
for(int i=1;i<=n1;i++)
cout<<"1"<<" ";
return 0;
}

for (int i=1;i<n;i++)
for (int j=i+1;j<=n;j++)
if (fix[a[i]+a[j]]==0)
{     
cout<<2<<endl;      
cout<<a[i]<<" "<<a[j]<<endl;
return 0;          
}

cout<<1<<endl<<a[1]<<endl;
}
