#include <iostream>
#include <algorithm>

using namespace std;

int a[2100000], pi[1100000], x[1000];

void sieve(int n)
{
	a[0] = a[1] = 0;
	for (int i = 2; i <= n; i++)
		a[i] = 1;
	for (int i = 2; i * i <= n; i++)
		if (a[i])
			for (int j = i * i; j <= n; j += i)
				a[j] = 0;
	return;
}

int main()
{
	sieve(2000000);
	int n;
	cin >> n;
	int ones = 0;
	for (int i = 0; i < n; i++)
	{
		cin >> x[i];
		if (x[i] == 1)
			ones++;
	}
	if (ones < 2)
	{
		for (int i = 0; i < n - 1; i++)
			for (int j = i + 1; j < n; j++)
				if (a[x[i] + x[j]])
				{
					cout << 2 << endl << x[i] << ' ' << x[j];
					return 0;
				}
		cout << 1 << endl << x[0];
	}
	else
	{
		int found = -1;
		for (int i = 0; i < n; i++)
			if (x[i] != 1 && a[x[i] + 1])
			{
				found = i;
				break;
			}
		if (found == -1)
		{
			cout << ones << endl;
			for (int i = 0; i < ones; i++)
				cout << 1 << ' ';
		}
		else
		{
			cout << ones + 1 << endl;
			for (int i = 0; i < ones; i++)
				cout << 1 << ' ';
			cout << x[found];
		}
	}
	return 0;
}
