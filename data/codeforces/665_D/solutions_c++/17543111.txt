// 一个偶数加一排1，或者2个加起来为素数。实在没有就随便。
#include<cstdio>
#include<algorithm>
using namespace std;

#define ran 2222222

int n, s1, s2, x1, x2;
bool f[ran];
int v[1111];

int main() {
	for(int i=2;i*i<=ran;i++)
		if(!f[i]) {
			for(int j=i*i;j<=ran;j+=i)
				f[j] = true;
		}
	
	scanf("%d", &n);
	s1=s2=0;
	for(int i=0;i<n;i++) {
		scanf("%d", &v[i]);
		if(v[i]==1) s1++;
		if(v[i]>1&&!f[v[i]+1]) s2=v[i];
	}
	x1=x2=0;
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
			if(!f[v[i]+v[j]]) {
				x1=v[i];x2=v[j];
			}
	if(s1+min(1,s2)>=2) {
		printf("%d\n", s1+min(1,s2));
		if(s2) printf("%d ", s2);
		for(int i=0;i<s1;i++)
			printf("1%c", i<s1-1 ? ' ' : '\n');
	}else if(x1) {
		printf("2\n%d %d\n", x1, x2);
	}else printf("1\n%d\n", v[0]);
	
	return 0;
}
