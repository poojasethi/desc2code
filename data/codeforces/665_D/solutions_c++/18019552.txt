#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
int a[100000];int n;
bool isp(int a){
	for(int i=2;i*i<=a;i++)
		if((a%i)==0)return 0;
	return 1;
}
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	sort(a+1,a+1+n);int sol;
	if(n==1){
		printf("1\n%d",a[1]);
	}
	else{ 
		int num=0;int t=0;
		while((num<n)and(a[num+1]==1))num++;
		if(num>=2){
			t=0; 
			for(int i=num+1;i<=n;i++){
				if(isp(1+a[i])){
					t=i;
					break;
				}
			}
			printf("%d\n",num+min(1,t));
			for(int i=1;i<=num;i++)
				printf("1 ");
			if(t>0)printf("%d",a[t]);
		}
		else{
			int ta,tb;ta=tb=0;
			for(int i=1;i<n;i++){
				for(int j=i+1;j<=n;j++){
					if(isp(a[i]+a[j])){
						ta=i;tb=j;
						break;
					}
				}
				if(ta!=0)break;
			} 
			if(ta>0){
				printf("2\n%d %d",a[ta],a[tb]);
			}
			else{ 
				printf("1\n%d",a[1]);
			}
		}
	}	
	return 0;
}
