#include <bits/stdc++.h>
using namespace std;

const int maxn = 1010 , maxa = 2e6 + 5;

int n,a[maxn];
bool pr[maxa];

int main(){
	
	scanf("%d",&n);
	
	for( int i = 2 ; i < maxa ; i++ )
		if( !pr[i] )
			for( int j = i + i ; j < maxa ; j += i )
				pr[j] = 1;
	
	for( int i = 2 ; i < maxa ; i++ ) pr[i] = !pr[i];
	
	int cnt = 0 , tmp = -1;
	
	for( int i = 1 ; i <= n ; i++ ){
		scanf("%d",&a[i]);
		if( a[i] == 1 ) cnt++;
		if( a[i] > 1 and pr[a[i] + 1] ) tmp = a[i];
	}
	
	if( cnt >= 1 and tmp != -1 ){
		printf("%d\n",cnt + 1);
		for( int i = 1 ; i <= cnt ; i++ )
			printf("1 ");
		printf("%d",tmp);
		return 0;
	}
	if( cnt >= 2 ){
		printf("%d\n",cnt);
		for( int i = 1 ; i <= cnt ; i++ ){
			printf("1");
			if( i != cnt ) printf(" ");
		}
		return 0;
	}
	
	for( int i = 1 ; i < n ; i++ )
		for( int j = i + 1 ; j <= n ; j++ )
			if( pr[a[i] + a[j]] ){
				printf("2\n");
				printf("%d %d",a[i],a[j]);
				return 0;
			}
	
	printf("1\n");
	printf("%d",a[1]);
	
	return 0;
}
