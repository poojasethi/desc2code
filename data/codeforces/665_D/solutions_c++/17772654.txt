using namespace std;
bool f[2000002];
bool prime(int x){
	return f[x];
}
int main(){
	memset(f, 1, sizeof(f));
	f[0]=0;f[1]=0;
	for (int i = 2; i <= 2000001; i++)
		for (int j = 2; i * j <= 2000001; j++)
			f[i*j]=0;
	int n, a[10001];
	cin >> n;
	bool flag =0;
	for (int i= 1;i<=n;i++) cin>>a[i];
	int tmp = 0,j=0;
	for (int i = 1;i<=n;i++) if (a[i]==1) tmp++;
	for (int i=1;i<=n;i++) if ((tmp>0)&&(a[i]!=1)&&(prime(1+a[i]))){
		flag = 1;
		j=i;
		break;
	}
	if (flag){
		cout<<tmp+1<<endl;
		for (int i=1;i<=tmp;i++) cout<<1<<" ";
		cout<<a[j]<<endl;
	}
	else if (tmp>=2){
				cout<<tmp<<endl;
		for (int i=1;i<=tmp;i++) cout<<1<<" ";
		cout<<endl;
	}
	else{
	for (int i=1;i<=n;i++){
		if (flag) break;
		for (int j=1;j<i;j++){
			if (flag) break;
			if (prime(a[i]+a[j])){
				cout<<2<<endl<<a[i]<<" "<<a[j]<<endl;
				flag=1;
			}
		}
	}
	if (!flag) cout<<1<<endl<<a[1]<<endl;
}
	return 0;
}
