
#include <stdio.h>
#include <bits/stdc++.h>
using namespace std;


#define eps 1e-9
#define N (2000000 + 10)
#define M (400000 + 10)
#define inf 0x3f3f3f3f

bool isnot[N];
int a[N], ans[N];

void init() {
	isnot[0] = isnot[1] = 1;
	for(int i = 2; i < N; ++i) {
		if(isnot[i]) continue;
		for(int j = i + i; j < N; j += i)
			isnot[j] = 1;
	}
}

int main() {
	int n;
	init();
	scanf("%d", &n);
	int cnt = 0;
	for(int i = 0; i < n; ++i) {
		scanf("%d", &a[i]);
		if(a[i] == 1) ans[cnt++] = 1;
	}
	if(cnt == 0 || cnt == 1) {
		for(int i = 0; i < n; ++i)
			for(int j = i+1; j < n; ++j)
				if(!isnot[a[i]+a[j]]) {
					printf("2\n%d %d\n", a[i], a[j]);
					return 0;
				}
	}
	if(cnt != 0) {
		int b = -1;
		for(int i = 0; i < n; ++i)
			if(!isnot[a[i]+1] && a[i] != 1) {
				b = a[i];
				break;
			}

		if(b != -1) ans[cnt++] = b;
		printf("%d\n", cnt);
		for(int i = 0; i < cnt; ++i)
			printf("%d%c", ans[i], i == cnt-1 ? '\n' : ' ');
		return 0;
	}
	else {
		printf("1\n%d\n", a[0]);
	}
}
