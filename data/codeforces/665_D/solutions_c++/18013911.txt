#include<cstdio>
#include<algorithm>
using namespace std;

int n,a[1010];

bool flag(int x)
{
	if(x == 1)return false;
	for(int i = 2;i*i <= x;i++)
		if(x%i == 0)return false;
	return true; 
}

int main(){
	scanf("%d",&n);
	for(int i = 1;i <= n;i++)
		scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	
	int ans = 1,q=-1,s=0,tmp1,tmp2;
	for(int i = n;a[i] > 1;i--)
		for(int j = i-1;a[j]>1;j--)
			if(flag(a[i]+a[j]))
			{
				ans = 2;
				q = 0;
				tmp1 = a[i];
				tmp2 = a[j];
			}
	
	for(int i = 1;i <= n;i++)
		if(a[i]==1)s++;
		else break;
	for(int i = n;a[i]>1;i--)
		if(flag(a[i]+1) && ans<s+1)
		{
			ans = s+1;
			q = s;
			tmp1 = a[i];
		}
	if(ans<s)
	{
		ans = s;
		q = s-1;
		tmp1 = 1;
	}
	
	if(q == -1)
	{
		printf("1\n%d\n",a[1]);
		return 0;
	}
	if(q == 0)
	{
		printf("%d\n%d %d\n",ans,tmp1,tmp2);
		return 0;
	}
	if(q>0)
	{
		printf("%d\n",ans);
		for(int i = 1;i <= q;i++)
			printf("1 ");
		printf("%d\n",tmp1);
		return 0;
	}
	return 0;
}
