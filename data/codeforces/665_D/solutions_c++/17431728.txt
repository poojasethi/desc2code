#include<bits/stdc++.h>
#define LL long long
using namespace std;
inline void splay(int &v){
	v=0;char c=0;int p=1;
	while(c<'0' || c>'9'){if(c=='-')p=-1;c=getchar();}
	while(c>='0' && c<='9'){v=(v<<3)+(v<<1)+c-'0';c=getchar();}
	v*=p;
}
bool is[2000010];
int a[100010],n;
int s[100010],s1;
int t[100010],t1;
int main(){
	memset(is,1,sizeof is);
	is[0]=is[1]=0;
	for(int i=2;i<=2000000;i++){
		if(is[i]){
			for(int j=i+i;j<=2000000;j+=i){
				is[j]=false;
			}
		}
	}
	splay(n);
	for(int i=1;i<=n;i++){
		splay(a[i]);
	}
	for(int i=1;i<=n;i++){
		for(int j=i+1;j<=n;j++){
			if(is[a[i]+a[j]]){
				s[++s1]=a[i];
				s[++s1]=a[j];
				goto end;
			}
		}
	}
	if(!s1){
		s[++s1]=a[1];
	}
	end:;bool fir=true;
	for(int i=1;i<=n;i++){
		if(a[i]==1)t[++t1]=1;
		else if(fir&&is[a[i]+1])t[++t1]=a[i],fir^=1;
	}
	if(s1>t1){
		cout<<s1<<endl;
		for(int i=1;i<=s1;i++)printf("%d ",s[i]);
	}
	else{
		cout<<t1<<endl;
		for(int i=1;i<=t1;i++)printf("%d ",t[i]);
	}
}
