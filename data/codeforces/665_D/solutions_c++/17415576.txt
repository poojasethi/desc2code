#include <bits/stdc++.h>
using namespace std;
bool a[(int)2e6+3];
int aa[1003];
void sieve(){
   a[2]=true;
   for(int i=3; i<2e6+3; i+=2) a[i]=true;
   for(int i=3; i<2004; i++)
      for(int j=2; i*j<2e6+3; j++) a[i*j]=false;
}
vector <int> ans;
int main()
{
   sieve();
   int n;
   cin>>n;
   for(int i=0; i<n; i++) cin>>aa[i];
   sort(aa, aa+n);
   for(int i=0; i<n; i++){
      if(aa[i]==1) ans.push_back(1);
   }
   if(ans.size()){
      for(int i=ans.size(); i<n; i++)
         if(a[1+aa[i]]){
            ans.push_back(aa[i]); break;
         }
   }
   if(ans.size()>1){
      cout<<ans.size()<<endl;
      for(int i=0; i<ans.size(); i++) cout<<ans[i]<<' '; return 0;
   }
   if(ans.size()) ans.pop_back();
   for(int i=0; i<n-1; i++){
      for(int j=i+1; j<n; j++){
         if(a[aa[i]+aa[j]]){
            cout<<2<<endl;
            cout<<aa[i]<<' '<<aa[j]<<endl; return 0;
         }
      }
   }
   cout<<1<<endl<<aa[0]<<endl;
   return 0;
}
