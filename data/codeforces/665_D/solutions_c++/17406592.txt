#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;
int use[2100000],num[1100],n,i,all,j;
int main()
{
    //freopen("c.in","r",stdin);
    //freopen("c.out","w",stdout);
   for (i=2;i<=1000000;i++)
   if (use[i]==0)
   {
       for (j=i+i;j<=2000000;j+=i) use[j]=1;
   }
    scanf("%d",&n);
    for (i=1;i<=n;i++)
    {
        scanf("%d",&num[i]);
        if (num[i]==1) all++;
    }
    if (all!=0)
    {
        for (i=1;i<=n;i++)
        if (num[i]!=1 && use[num[i]+1]==0)
        {
            printf("%d\n",all+1);
            for (j=1;j<=all;j++) printf("1 ");
            printf("%d\n",num[i]);
            return 0;
        }
        if (all>=2)
        {
            printf("%d\n",all);
            for (i=1;i<=all;i++) printf("1 ");
            return 0;
        }
    }
    for (i=1;i<=n;i++)
    {
        for (j=i+1;j<=n;j++)
        if ((num[j]+num[i])%2==1 && use[num[i]+num[j]]==0)
        {
            printf("2\n%d %d\n",num[i],num[j]);
            return 0;
        }
    }
    printf("1\n%d\n",num[1]);
}
