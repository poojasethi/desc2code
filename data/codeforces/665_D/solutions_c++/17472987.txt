#include <iostream>
using namespace std;
int p[2000001];
int a[1001],x[1001],nn;
int i,j,n,b;
main()
{  
  for (i=2;i*i<=2000000;i++)
   if (p[i]==0)
    for (j=i*i;j<=2000000;j=j+i)
     p[j]=1;          
  cin>>n;
  for (i=1;i<=n;i++)
  {
   cin>>a[i];
   if (a[i]==1) 
   nn++;
  }      
  
  for (i=1;i<=nn;i++) 
  x[i]=1;
  
  if (nn>0) 
   for (i=1;i<=n;i++)
    if (a[i]!=1 && p[a[i]+1]==0)
	 {
     x[++nn]=a[i];
	  break;
  }
  
  if (nn>=2) 
  {
   cout<<nn<<endl;
   for (i=1;i<=nn;i++)    
    cout<<x[i]<<" ";
    return 0;
  }
  
  for (i=1;i<n;i++)
   for (j=i+1;j<=n;j++)
    if (p[a[i]+a[j]]==0)
	 {     
     cout<<2<<endl;      
     cout<<a[i]<<" "<<a[j]<<endl;
     return 0;          
    }
    cout<<1<<endl;
    cout<<a[1]<<endl;
}
