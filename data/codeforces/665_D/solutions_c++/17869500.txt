#include<cstdio>
#include<cstring>
#include<algorithm>
#include<ctime>
using namespace std;
int a[1010],n,ans,pr[200010],cnt,out[1010],tp[1010];const int TRY=300;bool isp[2000010];
int main(){
	srand(time(0));
	for(int i=2;i<=2000000;++i){
		if(!isp[i])pr[cnt++]=i;
		for(int j=0;j<cnt&&pr[j]*i<=2000000;++j){
			isp[pr[j]*i]=1;
			if(i%pr[j]==0)break;
		}
	}
	scanf("%d",&n);for(int i=1;i<=n;++i)scanf("%d",&a[i]);
	for(int t=1;t<=TRY;++t){
		int tmp=0;
		for(int i=1;i<=n;++i){
			bool f=1;
			for(int j=1;j<=tmp;++j)if(isp[a[i]+tp[j]]){f=0;break;}
			if(f)tp[++tmp]=a[i];
		}
		if(tmp>ans){ans=tmp;for(int i=1;i<=tmp;++i)out[i]=tp[i];}
		random_shuffle(a+1,a+n+1);
	}
	printf("%d\n",ans);for(int i=1;i<=ans;++i)printf("%d%c",out[i],i==ans?'\n':' ');
}
