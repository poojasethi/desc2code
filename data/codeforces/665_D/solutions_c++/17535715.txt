#include<cstdio>
#include<cstring>
using namespace std;
const int maxn=2000005;
bool vis[maxn];
typedef __int64 LL;
void init()
{
    memset(vis,false,sizeof(vis));
    for(int i=2;i<=2000000;i++)
        if(!vis[i])
        for(LL j=(LL)i*i;j<=2000000;j+=i) vis[j]=true;
}
int N,A[1005];
int ans[1005];
int solve()
{
    int ret=1;
    ans[0]=A[1];
    for(int i=1;i<=N;i++)
        for(int j=i+1;j<=N;j++)
        if(!vis[A[i]+A[j]]){ ret=2; ans[0]=A[i]; ans[1]=A[j]; }
    int cnt=0;
    for(int i=1;i<=N;i++) if(A[i]==1) cnt++;
    if(cnt>=2)
    {
        ret=cnt;
        for(int i=0;i<cnt;i++) ans[i]=1;
        for(int i=1;i<=N;i++)
            if(A[i]!=1&&!vis[A[i]+1]){ ans[ret++]=A[i]; break; }

    }
    return ret;
}
int main()
{
    init();
    while(scanf("%d",&N)!=EOF)
    {
        for(int i=1;i<=N;i++) scanf("%d",&A[i]);
        int k=solve();
        printf("%d\n",k);
        for(int i=0;i<k;i++) printf("%d%c",ans[i],i==k-1?'\n':' ');
    }
    return 0;
}
