#include <cstdio>
#include <algorithm>
#define REP(i, a, b) for (int i = a; i <= b; ++i)
#define rd(a) scanf("%d", &a)
using namespace std;

int n, a[1005], ans = 0, cnt = 0;
bool nop[2000005];

void Gen(){
	REP(i, 2, 2000000)
		if (!nop[i])
			for (int j = 2 * i; j <= 2000000; j += i)
				nop[j] = true;
}

int main(){
	rd(n); Gen();
	REP(i, 1, n) rd(a[i]), cnt += a[i] == 1;
	int l = -1, r = -1;
	REP(i, 1, n) REP(j, i + 1, n)
		if (!nop[a[i] + a[j]]){
			l = i, r = j; break;
		}
	if (l != -1) ans = 2;
	int x = 0;
	REP(i, 1, n) if (a[i] != 1 && !nop[a[i] + 1]){
		x = i; break;
	}
	if (cnt + (x > 0) > ans){
		//printf("cnt = %d\n", cnt);
		printf("%d\n", cnt + (x > 0));
		REP(i, 1, cnt) printf("1 ");
		if (x > 0) printf("%d\n", a[x]);
		return 0;
	}
	if (ans) printf("2\n%d %d\n", a[l], a[r]);
	else printf("1\n%d\n", a[1]);
	return 0;
}
