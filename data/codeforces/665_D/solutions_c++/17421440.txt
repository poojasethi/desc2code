#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>

using namespace std;
const int N = 2000005;

int is[N], a[1005], one, n;

int main() {
	fill(is + 2, is + N, true);
	for (int i = 2; i * i < N; ++i)
		if (is[i])
			for (int j = i * i; j < N; j += i)
				is[j] = false;
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &a[i]);
		if (a[i] == 1)
			one++;
	}
	vector<int> ans;
	// case 1: 1 1 ... even
	if (one > 1) {
		ans.assign(one, 1);
		for (int i = 0; i < n; ++i)
			if (a[i] % 2 == 0 && is[a[i] + 1]) {
				ans.push_back(a[i]);
				break;
			}
	} else {
		// pick an odd and an even
		for (int i = 0; i < n; ++i)
			for (int j = i + 1; j < n; ++j)
				if (is[a[i] + a[j]]) {
					ans.push_back(a[i]);
					ans.push_back(a[j]);
					goto done;
				}
		// pick a single number
		ans.push_back(a[0]);
		done:;
	}
	printf("%d\n", ans.size());
	for (int i = 0; i < ans.size(); ++i) {
		if (i) cout << ' ';
		cout << ans[i];
	}
	cout << endl;
}
