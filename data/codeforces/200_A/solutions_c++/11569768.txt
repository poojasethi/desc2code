#include<bits/stdc++.h>
using namespace std;
int a,b,n,m,k,x,y,i,d[2003][2003];
bool v[2003][2003];
#define rep(i,a,b) for (int i=a;i<b;i++)
bool solve(int x,int y,int k)
{
    int l=max(1,x-k),r=min(x+k,n),t;
    rep(i,l,r+1)
    {
        t=k-abs(x-i);
        if (y-t>=1&&!v[i][y-t]) return a=i,b=y-t,1;
        if (y+t<=m&&!v[i][y+t]) return a=i,b=y+t,1;
    }
    return 0;
}
main()
{
    scanf("%d%d%d",&n,&m,&k);
    rep(i,0,k)
    {
        scanf("%d%d",&x,&y);
        rep(i,-2,3)
            rep(j,-2,3)
                if (x+i<1||x+i>n||y+j<1||y+j>m) continue;
                else d[x][y]=max(d[x][y],d[x+i][y+j]-abs(i)-abs(j));
        while (!solve(x,y,d[x][y])) d[x][y]++;
        printf("%d %d\n",a,b);
        v[a][b]=1;
    }
}
