#include<iostream>
#include<cstdio>
#include<cmath>
#include <algorithm>
using namespace std;
int a[2020][2010],n,m,i,j,k,l,o,p,q,x,y;
bool b[2020][2020];
int pd(int x,int y,int o)
{
    int i,l,r;
    l=max(1,x-o);r=min(x+o,n);
    for(i=l;i<=r;i++)
    {
        j=o-abs(x-i);
        if(y-j>0 && !b[i][y-j])
        {
            a[x][y]=o;
            b[i][y-j]=true;
            printf("%d %d\n",i,y-j);
            return 0;
        }
        if(y+j<=m && !b[i][y+j])
        {
            a[x][y]=o;
            b[i][y+j]=true;
            printf("%d %d\n",i,y+j);
            return 0;
        }
    }
    return 1;
}
int main( )
{
    cin >> n >> m >> q;
    for(;q;q--)
    {
        scanf("%d%d",&x,&y);
        o=0;
        for(i=-2;i<=2;i++)
            for(j=-2;j<=2;j++)
            {
                int X=x-i,Y=y-j;
                if(X>0 && Y>0 && X<=n && Y<=m) 
                    o=max(o,a[X][Y]-abs(i)-abs(j));
            }
        while(pd(x,y,o)) o++;
    }
    cin >> n;
    return 0;
}
