#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int u=2012;
int dx[8]={0,0,-1,1,-1,-1,1,1},dy[8]={-1,1,0,0,-1,1,-1,1};
int a[u][u],v[u][u],n,m,k,x,y;

void debug(){
    for (int i = 1;i <= n;i ++){
        for (int j = 1;j <= m;j ++){
            printf ("a[%d][%d] = %d\n",i,j,a[i][j]);
        }
    }
}

bool check(int x,int y,int d){
//    printf ("k = %d\n",k);
	for(int i=max(1,x-d);i<=min(x+d,n);i++){
		int t=d-abs(i-x),j=y-t;
       // printf ("t = %d i = %d j = %d d = %d\n",t,i,j,d);
		if(j>0&&j<=m&&!v[i][j]){
			v[i][j]=1;
			printf("%d %d\n",i,j);
			a[x][y]=d;
			return 0;
		}
		j=y+t;
        //printf ("j = %d\n",j);
		if(j>0&&j<=m&&!v[i][j])
		{
			v[i][j]=1;
			printf("%d %d\n",i,j);
			a[x][y]=d;
			return 0;
		}
	}
	return 1;
}

int main(){
	cin>>n>>m>>k;
	while(k--){
		scanf("%d%d",&x,&y);
		int d=a[x][y];
		for(int i=0;i<8;i++){
			int nx=x+dx[i],ny=y+dy[i];
			d=max(d,a[nx][ny]-1-(i>3));
		}
		while(check(x,y,d)) d++;
        //debug();
	}
	return 0;
}
