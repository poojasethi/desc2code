#define speed ios::sync_with_stdio(false)
#define N 2005

int dis[N][N];
bool vis[N][N];
int n,m,k,a,b;
bool find(int x,int y,int d)
{
    int l=max(1,x-d);
    int r=min(n,x+d);
    int dy;
    for(int i=l;i<=r;i++)
    {
        dy=d-abs(x-i);
        if(y-dy>0&&!vis[i][y-dy])
        {
            a=i;
            b=y-dy;
            return true;
        }
          if(y+dy<=m&&!vis[i][y+dy])
          {
               a=i;
              b=y+dy;
            return true;
          }
    }
    return false;
}
int main()
{   speed;
    scanf("%d%d%d",&n,&m,&k);
    int x,y;
    for(int i=0;i<k;i++)
    {
        scanf("%d%d",&x,&y);
        for(int i=-1;i<=1;i++)
            for(int j=-1;j<=1;j++)
          if(x+i>0&&x+i<=n&&y+j>0&&y+j<=m)
              dis[x][y]=max(dis[x][y],dis[x+i][y+j]-abs(i)-abs(j));
       while(!find(x,y,dis[x][y]))dis[x][y]++;
       printf("%d %d\n",a,b);
       vis[a][b]=true;
    }
    return 0;
}
