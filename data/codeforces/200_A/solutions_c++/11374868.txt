typedef long long ll;

using namespace std;
int dx[8] = {0, 0, -1, 1, -1, -1, 1, 1};
int dy[8] = { -1, 1, 0, 0, -1, 1, -1, 1};
int a[2002][2002], n, m;
bool ok[2002][2002];
bool check(int x, int y, int d){
	int j, t;
    for (int i = max(1, x - d); i <= min(x + d, n); i++){
        t = d - abs(i - x);
        j = y - t;
        for(int k=0; k<2; k++){
        	if (j > 0 && j <= m && !ok[i][j]){
        	    ok[i][j] = 1;
        	    printf("%d %d\n", i, j);
        	    a[x][y] = d;
        	    return 0;
        	}
        	j+=2*t;
        }
    }		
	return 1;
}
int main()
{
	int x,y,d,k;
	scanf("%d %d %d\n", &n, &m, &k);
	while(k--){
		scanf("%d %d\n", &x, &y);
		d=a[x][y];
		for(int i=0; i<8; i++){
			d=max(d,a[x+dx[i]][y+dy[i]]-1-(i>3));
		}
		while(check(x,y,d))
			d++;
	}

	
	return 0;
}
