#include <cstdio>
#include <algorithm>
using namespace std;
#define rep(i,n) for (int i=0;i<n;i++)
int n,m,k,d,mx,my,x,y,a[2010][2010],l[2010][2010],r[2010][2010];
void upd(int dx,int dy) {
	int e=abs(dx-x)+abs(dy-y);
	if (d>e||(d==e&&dx<mx)||(d==e&&dx==mx&&dy<my)) mx=dx,my=dy,d=e;
}
void Upd(int x,int y) {
	a[x][y]=1;
	if (!y) l[x][y]=100000;else l[x][y]=l[x][y-1];
	if (y==m-1) r[x][y]=100000;else r[x][y]=r[x][y+1];
	for (int i=y;a[x][i]&&i<m;i++) l[x][i]=l[x][y];
	for (int i=y;a[x][i]&&i>=0;i--) r[x][i]=r[x][y];
}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	rep(i,n) rep(j,m) l[i][j]=j,r[i][j]=j;
	rep(_,k) {
		scanf("%d%d",&x,&y);x--;y--;
		mx=my=d=100000;
		rep(j,d+1) {
			if (x>=j) upd(x-j,l[x-j][y]),upd(x-j,r[x-j][y]);
			if (x+j<n) upd(x+j,l[x+j][y]),upd(x+j,r[x+j][y]);
		}
		printf("%d %d\n",mx+1,my+1);
		Upd(mx,my);
	}
}
