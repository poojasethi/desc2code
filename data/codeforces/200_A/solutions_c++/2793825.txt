#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;

int d[2048][2048],n,m,k,x,y;
int h[2048][2048];

bool check(int x,int y,int D)
{
    int L=x-D,R=x+D;
    if (L<1) L=1;
    if (R>n) R=n;
    for (int i=L,j; i<=R; i++)
    {
        j=D-abs(i-x);
        if (y-j>0 && !h[i][y-j])
        {
            h[i][y-j]=true;
            printf("%d %d\n",i,y-j);
            return true;
        }
        if (y+j<=m && !h[i][y+j])
        {
            h[i][y+j]=true;
            printf("%d %d\n",i,y+j);
            return true;
        }
    }
    return false;
}
int main()
{
    scanf("%d%d%d",&n,&m,&k);
    while (k--)
    {
        scanf("%d%d",&x,&y);
        for (int i=-2; i<=2; i++)
            for (int j=-2; j<=2; j++)
                if (1<=x+i && x+i<=n && 1<=y+j && y+j<=m)
                    d[x][y]=max(d[x][y],d[x+i][y+j]-abs(i)-abs(j));
        while (!check(x,y,d[x][y]))
            d[x][y]++;
    }
    return 0;
}
