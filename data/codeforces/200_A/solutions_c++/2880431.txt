#include<cstdio>
#include<cstdlib>
#define MAXN 2010

int n,m,K,d[MAXN][MAXN],b[MAXN][MAXN],Ansx,Ansy;

bool Get(int X,int Y,int D)
{
	int i,j,L,R;
	L=X-D;
	if(L<1)L=1;
	R=X+D;
	if(R>n)R=n;
	for(i=L;i<=R;i++)
	{
		j=D-abs(i-X);
		if(Y-j>=1&&!b[i][Y-j])
		{
			Ansx=i;
			Ansy=Y-j;
			return 0;
		}
		if(Y+j<=m&&!b[i][Y+j])
		{
			Ansx=i;
			Ansy=Y+j;
			return 0;
		}
	}
	return 1;
}

int main()
{
	int i,j,x,y;
	scanf("%d%d%d",&n,&m,&K);
	for(;K--;)
	{
		scanf("%d%d",&x,&y);
		for(i=-5;i<=5;i++)
			for(j=-5;j<=5;j++)
				if(	x+i>=1&&x+i<=n&&
					y+j>=1&&y+j<=m&&
					d[x+i][y+j]-abs(i)-abs(j)>d[x][y])
				d[x][y]=d[x+i][y+j]-abs(i)-abs(j);
		for(;Get(x,y,d[x][y]);d[x][y]++);
		b[Ansx][Ansy]=1;
		printf("%d %d\n",Ansx,Ansy);
	}
	return 0;
}
