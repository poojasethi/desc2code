def is_pali(s):
    if s == s[::-1]:
        return s
    else:
        return None

s = raw_input()
n = len(s)
for i in range(n+1):
    pre, suf = s[:i], s[i:]
    if i > n / 2:
        mid = n - i
    else:
        mid = n - i - 1
    ss = pre + s[mid] + suf
    if is_pali(ss) is not None:
        print ss
        break
else:
    print "NA"
