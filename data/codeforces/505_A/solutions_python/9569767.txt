from sys import stdin
import string

s = stdin.readline()[:-1]
for i in xrange(len(s) + 1):
    t1, t2 = "", ""
    for ch in string.ascii_lowercase:
        t1 = s[:i] + str(ch) + s[i:]
        t2 = t1[::-1]
        if t1 == t2:
            break
    if t1 == t2:
        print t1
        break
else:
    print "NA"
