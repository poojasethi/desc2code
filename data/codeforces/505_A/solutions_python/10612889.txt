def isPalindrome(a):
	for i in xrange(0,len(a)/2):
		if a[i] != a[-i-1]:
			return False
	return True

s = raw_input()
closed = ''

for i in s:
	if i in closed:
		continue
	for j in xrange(0,len(s)+1):
		new_s = s[:j] + i + s[j:]
		if isPalindrome(new_s):
			print new_s
			exit(0)
	closed += i
print 'NA'
