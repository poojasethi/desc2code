# -*- coding: utf-8 -*-

import itertools

if __name__ == '__main__':
    s = raw_input()

    result = 'NA'
    alphabet = 'abcdefghijklmnopqrstuvyxwz'
    for i, a in itertools.product(range(len(s)+1), alphabet):
        candidate = s[:i] + a + s[i:]
        if candidate == candidate[::-1]:
            result = candidate
            break
    print(str(result))
