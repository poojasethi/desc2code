s = raw_input()

def f(s):
	n = len(s)
	for i in range(n/2+1):
		if s[i] != s[n-1-i]: return False
	return True

n = len(s)
flag = False
for i in range(n):
	t = s[0:i] + '1' + s[i:]
	t = t.replace('1',t[n-i])
	if f(t):
		flag = True
		print t
		break

t = s + s[0]
if not flag:
	if f(t):
		print t
	else:
		print 'NA'
