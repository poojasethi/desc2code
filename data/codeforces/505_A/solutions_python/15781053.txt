from string import ascii_lowercase

def is_a_palindrome(string):
    return string == string[::-1]

string = raw_input()
for i in range(len(string) + 1):
    for character in list(ascii_lowercase):
        newString = string[:i] + character + string[i:]
        if(is_a_palindrome(newString)):
            print(newString)
            exit(0)
print("NA")
