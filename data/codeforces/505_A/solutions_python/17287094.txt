def isPalindrome(s):
	n = len(s)
	for i in xrange(n / 2):
		if s[i] != s[n - 1 - i]: return False
	return True

def main():
	s = map(str, raw_input())
	for i in xrange(len(s) + 1):
		for j in xrange(26):
			sNew = s[: i] + [chr(ord('a') + j)] + s[i:]
			if isPalindrome(sNew):
				print "".join(sNew)
				return
	print "NA"
	 
main()
