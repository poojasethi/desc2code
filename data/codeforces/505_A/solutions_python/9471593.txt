from sys import stdin
def ch(s):
 t = ''+s
 t = t[::-1]
 return t==s
x = stdin.readline().strip()
l = len(x)
ans = 'NA'
y = 'abcdefghijklmnopqrstuvwxyz'
for i in xrange(l+1):
 fir = x[0:i] 
 sec = x[i:l]
 for j in y:
  nn = fir + j + sec
  #print "checking for ",nn
  if ch(nn):
   ans = nn
print ans
