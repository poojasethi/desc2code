def fun(s):
    alphabet=['a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z']
    l=len(s)
    for i in range(0,l+1):
        for j in alphabet:
            ans=s[:i]+j+s[i:]
            if(ans==ans[::-1]):
                return ans
    return "NA"
s=raw_input()
print fun(s)
