#include<bits/stdc++.h>
using namespace std;
int check_palin(string &a)
{
    int i,l=a.size();
    for(i=0;i<l;i++)
     if(a[i]!=a[l-i-1])
        return 0;
    return 1;
}
int main()
{
string s,s1;
cin>>s;
int l=s.size(),i;
for(i=0;i<=l;i++)
{
    s1=s;
    s1.insert(i,"0");
    s1[i]=s1[l-i];
    if(check_palin(s1))
    {
        cout<<s1<<endl;return 0;
    }
}
cout<<"NA"<<endl;
}
