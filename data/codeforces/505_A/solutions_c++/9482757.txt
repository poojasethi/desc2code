#include <stdio.h>
#include <string.h>

int main()
{
	char s[12],t[12];
	int i,j,l,k;
	scanf("%s",s);
	l = strlen(s);
	for(i=0;i<=l;i++)
	{
		t[i]='a';
		for(k=0,j=0;j<l;j++,k++)
		{
			if(k==i) k++;
			t[k]=s[j];
		}
		t[i]=t[l-i];
		t[l+1]=0;
		for(k=l,j=0;k>j;j++,k--)
		{
			if(t[k]!=t[j]) break;
		}		
		if(k<=j) break;
	}
	if(i<=l) printf("%s\n",t);
	else printf("NA\n");
	return 0;
}
