#include <bits/stdc++.h>
using namespace std;
int main()
{
    string str;
    cin>>str;
    for(int i=0;i<=str.size();i++)
    {
        for(char j='a'; j <= 'z'; j++)
        {
            string str1 = str;
            str1.insert(i,1,j);
            string str2=str1;
            reverse(str1.begin(),str1.end());
            if(str1==str2){cout<<str1;return 0;}
        }
    }
    cout<<"NA";
    return 0;
}
