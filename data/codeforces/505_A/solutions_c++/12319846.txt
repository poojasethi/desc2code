#include <stdio.h>
#include <string.h>

char p[11],t[11];
int i,j,k,l;

int main()
{
	gets(t);
	l=strlen(t);
	for (i=0;i<=l;i++)
	{
		for (j=0;j<l-i;j++)
		    p[j]=t[j];
		p[l-i]=p[i];
		for (j=l-i+1;j<=l;j++)
		    p[j]=t[j-1];
		k=0;
		while ((k<l) && (p[k]==p[l-k]))
		    k++;
		if (k==l)
		{
			puts(p);
			return 0;
		}
	}
	printf("NA");
	return 0;
}
