#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s1,s2;
    cin >> s1;
    for(int i=0; i<=s1.length(); i++ ){
        for(char ch='a'; ch<='z'; ch++){
            s2=s1;
            char a=ch;
            s2.insert(i,1,a);
            if(s2 == string (s2.rbegin(), s2.rend())){
                cout << s2;
                return 0;
            }
        }
    }
    cout<< "NA";
    return 0;
}
