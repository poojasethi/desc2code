#include<iostream>
using namespace std;
int main()
{
	string s;
	cin>>s;
	int len=s.length();
	for(int i=0;i<=len;i++)
	{
		string p="a";
		string a;
		for(;p[0]<='z';p[0]++)
		{
			a=s;
			a.insert(i,p);
			int state=0;
			for(int i=0,j=len;i<=len;i++,j--)
				if(a[i]!=a[j])
				{
					state=1;
					break;
				}
			if(state==0)
			{
				cout<<a;
				return 0;
			}
		}
	}
	cout<<"NA";
	return 0;
}
