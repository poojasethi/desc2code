#include<bits/stdc++.h>
using namespace std;
typedef long long int lld;
int main()
{
	string a,b,s;
	char ch;
	cin>>a;
	lld i,j,k;
	for(i=0;i<=a.length();i++)
	{
		for(ch='a';ch<='z';ch++)
		{
			b=a;
			s=ch;
			b.insert(i,s);
			if(b==string(b.rbegin(),b.rend()))
			{
				cout<<b<<endl;
				return 0;
			}
		}
	}
	cout<<"NA"<<endl;
	return 0;
}
