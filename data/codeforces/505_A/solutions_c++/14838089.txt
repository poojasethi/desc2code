#include<iostream>
#include<string>
using namespace std;

string s, t;

bool chk(string s) {
	for(int i=0, j=s.length()-1; i<j; i++,j--)
		if (s[i] != s[j]) return false;
	return true;
}

int main() {
	cin>>s;
	for(int i=0;i<=s.length();i++)
		for(char j='a';j<='z';j++) {
			t = s.substr(0, i) + j + s.substr(i,s.length()-i);
			if(chk(t)) {
				cout<<t<<endl;
		 		return 0;   
			}
		}
	cout<<"NA"<<endl;
	
	return 0;
}
