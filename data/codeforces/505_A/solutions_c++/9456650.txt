#include <bits/stdc++.h>
using namespace std;
bool ispalin(string s)
{
	for(unsigned int i=0;i<=s.length()/2;i++)
		if(s[i]!=s[s.length()-1-i])
			return false;
	return true;
}
int main()
{
	string s;
	cin>>s;
	for(unsigned int i=0;i<=s.length();i++)
	for(char j='a';j<='z';j++)
	{
		string temp=s;
		temp.insert(i,1,j);
		if(ispalin(temp))
		{
			cout<<temp<<endl;
			return 0;
		}
	}
	cout<<"NA"<<endl;
	return 0;
}
