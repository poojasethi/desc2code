#include<iostream>
#include<string>

using namespace std;

int main()
{
	string s;
	cin >> s;
	for(int i = 0; i <= s.size(); i++)
	{
		string tmp = s;
		tmp.insert(i,"0");
		tmp[i]=tmp[tmp.size()-1-i];
		bool check = 1;
		for (int j = 0; j < tmp.size()/2; j++)
		{
			if(tmp[j] != tmp[tmp.size()-1-j])
			{
				check = 0;
				break;
			}
		}
		if(check == 1)
		{
			cout << tmp << endl;
			return 0;
		}
	}
	cout << "NA" << endl;
	return 0;
}
