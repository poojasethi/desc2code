#include<bits/stdc++.h>

using namespace std;

int i,j,k;

int main(void)
{
	string s,ss,cc;
	cin>>s;
	for(i=0;i<=s.length();i++)
	{
		for(char ch='a';ch<='z';ch++)
		{
			ss=s;
			cc=ch;
			ss.insert(i,cc);
			if(ss==string(ss.rbegin(),ss.rend()))
			{
				cout<<ss<<endl;
				return 0;
			}
		}
	}
	cout<<"NA";
	return 0;
}
