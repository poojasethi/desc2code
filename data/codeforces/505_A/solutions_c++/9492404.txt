#include<stdio.h>
#include<string.h>
int main()
{
	int n,i,j;
	char s1[12],s2[12],c;
	scanf("%s",s1);
	n=strlen(s1);
	for(i=n;i>=0;i--)
	{
		strcpy(s2,s1);
		for(j=n+1;j>i;j--)
			s2[j]=s2[j-1];
		for(c='a';c<='z';c++)
		{
			s2[i]=c;
			for(j=0;j<(n+1)/2;j++)
				if(s2[j]!=s2[n-j])
					break;
			if(j>=(n+1)/2)
			{
				printf("%s\n",s2);
				return 0;
			}
		}
	}
	printf("NA\n");
	return 0;
}
