#include<iostream>
#include<string.h>
#include<algorithm>
#define rep(i,a,n) for(i=a;i<n;i++)
#define sz(a) a.length()
using namespace std;
int main(){
	string a;
	cin>>a;
	int i,j;
	rep(i,0,sz(a)+1){
		rep(j,'a','z'+1){
			string b=a.substr(0,i)+string(1,j)+a.substr(i);
			string k=b;
			reverse(k.begin(),k.end());
			if(k==b){
				cout<<"\n"<<b;
				return 0;
			}
		}
	}
	cout<<"\nNA";
	return 0;
}
