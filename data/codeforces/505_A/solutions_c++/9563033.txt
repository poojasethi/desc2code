#include <bits/stdc++.h>
using namespace std;

int main() {
  ios_base::sync_with_stdio(false); cin.tie(0);
  string st;
  cin >> st;
  for(int i=0; i<=st.size(); i++) {
    for(char ch = 'a'; ch<='z'; ch++) {
      string s = st.substr(0, i) + string(1, ch) + st.substr(i);
      string t = s;
      reverse(t.begin(), t.end());
      if(s==t) { cout << s; return 0;}
    }
  }
  cout << "NA";
  return 0;
}
