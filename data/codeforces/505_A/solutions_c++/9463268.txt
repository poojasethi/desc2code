# include <bits/stdc++.h>

using namespace std;

string a,b;

bool pm(string x)
{
	for(int i=0 ; i<x.size() ; i++)
		if(x[i]!=x[x.size()-1-i])
			return 0;
	return 1;
}

int main()
{
	cin >> a;
	b = a;

	for(int i=0 ; i<=a.size() ; i++)
		for(char j='a' ; j<='z' ; j++){
			b.insert(b.begin()+i,j);
			if(pm(b)==1){
				cout << b <<endl;
				return 0;
			}
			b = a;
		}

	cout << "NA" <<endl;

	return 0;
}
