#include<bits/stdc++.h>

using namespace std;

int main()
{
    string s, ss, in;
    cin >> s;

    for (int i=0; i<=s.size();i++)
        for (char c='a';c <='z';c++) {
            ss=s;
            in=c;
            ss.insert(i, in);
            if (ss==string(ss.rbegin(),ss.rend())) {
                cout<<ss<<endl;
                return 0;
            }
        }

    cout<<"NA"<<endl;
    return 0;
}
