#include<iostream>
#include<algorithm>
#include<string>
using namespace std;

string s,t;

bool work(int i,char c) {
	t=s;
	t.insert(i,string("")+c);
	if(equal(t.begin(),t.begin()+t.length()/2,t.rbegin())) {
		s=t;
		return 1;
	}
	return 0;
}

int main() {
	cin>>s;
	for(int i=0;i<=s.size();i++)
		for(char c='a';c<='z';c++)
			if(work(i,c)) {
				cout<<s<<endl;
				return 0;
			}
	cout<<"NA"<<endl;
	return 0;
}
