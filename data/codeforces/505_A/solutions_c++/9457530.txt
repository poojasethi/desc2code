#include<stdio.h>
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main()
{
	int i,j,n,k;
	string ch;
	string s,p;
	cin >> s;
	n=s.size();
	for(i=0;i<=n;i++)
	{
		for(j=97;j<=122;j++) {
			p=s;
			ch=(char)j;
			p.insert(i,ch);	
			for(k=0;k<=n;k++) {
				if(p[k]!=p[n-k])	break;
			}
			if(k==(n+1)){
				cout << p << endl;
				return 0;
			}
		}
	}
	cout << "NA" << endl;
	return 0;
}
