#include <iostream>
using namespace std;

bool check(std::string n){
  for (int i=0; i<=n.size()/2; ++i) if (n[i]!=n[n.size()-i-1]) return false;
  return true;
}


int main(){
  std::string k;
  std::cin >>k;
  for (int i=0; i<=k.size(); ++i){
    for (char c='a'; c<='z'; ++c){
      string t=k.substr(0, i)+c+k.substr(i, k.size()-i);
      if (check(t)){
	std::cout << t;
	return 0;
      }
    }
  }
  std::cout << "NA";
  return 0;
  
}
