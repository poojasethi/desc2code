#include <bits/stdc++.h>
using namespace std;
int main()
{
	string s;
	cin >> s;
	int uz=s.length();
	for(int i=0;i<=uz;i++)
	{
		string p="a";
		string a;
		for(;p[0]<='z';p[0]++)
		{
			a=s;
			a.insert(i,p);
			int flag=0;
			for(int i=0,j=uz;i<=uz;i++,j--)
			{
				if(a[i]!=a[j])
				{
					flag=1;
					break;
				}
			}
			if(flag==0)
			{
				cout << a;
				return 0;
			}
		}
	}
	printf("NA");
	return 0;
}
