#include <bits/stdc++.h>
using namespace std;
main()
{
    string s ,ch;
    cin >>s ;

    for(int i=0;i<=s.size();i++)
    {
        for(char cc = 'a' ; cc<='z';cc++)
        {
              string ss;
              ss=s;
              ch=cc;
              ss.insert(i,ch);
              if(ss== string(ss.rbegin(),ss.rend())){cout << ss<<endl;return 0;}
        }
    }
    cout << "NA\n";
}
