#include <bits/stdc++.h>

using namespace std;

int main()
{
    string str;
    cin>>str;
    for(int i=0;i<=str.size();i++){

        for(int j=(int)'a';j<=(int)'z';j++){
            string tmp = str;
            tmp.insert(i,1,(char)j);
            string tmp2=tmp;
            reverse(tmp.begin(),tmp.end());
            if(tmp==tmp2){cout<<tmp;return 0;}
        }
    }
    cout<<"NA";
    return 0;
}
