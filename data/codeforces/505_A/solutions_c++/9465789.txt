#include <bits/stdc++.h>
#define z size()
using namespace std;
int main()
{
    string s,a,b="abcdefghijklmnopqrstuvwxyz";
    cin>>s;
    for(int i=0;i<=s.z;i++)
    {
        for(int j=0;j<b.z;j++)
        {
            int h=1;
            a=s;
            a.insert(a.begin()+i,b[j]);
            for(int k=0;k<a.z/2;k++) if ((a[k]!=a[a.z-k-1])) h=0;
            if (h) {cout<<a;return 0;}

        }
    }
    cout<<"NA";

}
