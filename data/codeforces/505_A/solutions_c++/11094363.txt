#include<iostream>
#include<algorithm>
using namespace std;

int main(){

string s,tmp,atmp;

char i;
int j,f=0;
cin>>s;


for(i='a';i<='z' && !f ;++i){
tmp=s;

for(j=0;j<=s.size();++j){

tmp.insert(tmp.begin()+j,i);

atmp=tmp;

reverse(atmp.begin(),atmp.end());

if(tmp==atmp){f=1;break;}

tmp=s;
}

}

if(f)cout<<tmp<<endl;
else cout<<"NA"<<endl;


}
