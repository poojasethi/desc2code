#include <bits/stdc++.h>

using namespace std;

int main(){

    int i,j;
	string s;
	cin >> s;

	for(i=0;i<=s.size();i++){
        for(char ch='a';ch<='z';ch++){
		string t = s.substr(0, i) + string(1, ch) + s.substr(i);
		string u = t;
		reverse(u.begin(),u.end());
		if(t == u){
			cout << t << endl;
			return 0;
		}
        }
	}
	cout << "NA" << endl;
}
