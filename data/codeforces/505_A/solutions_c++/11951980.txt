#include<iostream>
#include<string>
using namespace std;
int main()
{
    string s , t , c;
    cin >> s;
    for(int i = 0 ; i <= s.size() ; i++)
        for(char j = 'a' ; j <= 'z' ; j++)
        {
            t = s;
            c = j;
            t.insert(i,c);
            if(t == string(t.rbegin(),t.rend()))
            {cout << t ; return 0;}
        }
    cout <<"NA";
}
