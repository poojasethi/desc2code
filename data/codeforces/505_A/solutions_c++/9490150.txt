#include<stdio.h>
#include<string.h>
char a[15],b[15];
int len;
int fan()
{
	for(int i=0;i<(len+1)/2;i++)
	if(b[i]!=b[len-i]) return 0;
	return 1;
}
int main()
{
	int i,j,t,m;
	scanf("%s",a);
	len=strlen(a);
	for(i=0;i<=len;i++)
	for(j='a';j<='z';j++)
	{
		for(t=0;t<i;t++)
		b[t]=a[t];
		b[t]=j;
		for(m=i;m<len;m++)
		b[++t]=a[m];
		if(fan())
		{
			printf("%s\n",b);
			return 0;
		}
	}
	printf("NA\n");
	return 0;
}
