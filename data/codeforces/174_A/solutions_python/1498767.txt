#!/usr/bin/python2

if __name__ == '__main__':
	n, b = map(int, raw_input().split())
	a = map(int, raw_input().split())
	res = [0.0] * len(a)
	m = max(a)
	for i, val in enumerate(a):
		res[i] += m - val
		b -= res[i]
	res = map(lambda x: x + b / n, res)
	if b < 0:
		print -1
		exit()
	for v in res:
		print v
