# -*- coding: utf-8 -*-

import sys

n, b = [int(e) for e in raw_input().split(' ')]
an = [int(e) for e in raw_input().split(' ')]
cn = []

maxa = max(an)
for a in an:
    b -= maxa - a
    cn.append(maxa - a)

d = b*1.0 / n
cn = [e + d for e in cn]

if min(cn) < 0:
    print -1
    sys.exit()
for c in cn:
    print str(c) + '00000'
