debug = False

import sys

def out():
    FIN.close()
    FOUT.close()
    sys.exit()



if debug:
    FIN = open('input.txt', 'r')
    FOUT = open('output.txt', 'w')
else:
    FIN = sys.stdin
    FOUT = sys.stdout
n, b = map(int, FIN.readline().split())
a = list(map(float, FIN.readline().split()))
s = (sum(a) + b) / n
ans = [s - a[i] for i in range(n)]
for i in range(n):
    if ans[i] < 0:
        FOUT.write('-1')
        out()
for i in range(n):
    FOUT.write(str(ans[i]) + '\n')

out()
