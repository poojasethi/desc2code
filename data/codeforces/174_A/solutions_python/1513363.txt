import sys

(n, b) = sys.stdin.readline().split()
n = int(n)
b = int(b)
a = sys.stdin.readline().split()
na = len(a)
for i in range(na):
    a[i] = int(a[i])
mx = max(a)
r = []
for i in range(na):
    r.append(mx - a[i])
    b = b - (mx - a[i])

if b < 0:
    print -1
    exit()

b = float(b) / na
for i in range(na):
    print "%.6f" % (r[i] + b)
