from __future__ import division

n, b = [int(x) for x in raw_input().split()]
nums = [int(x) for x in raw_input().split()]
max_ = max(nums)
left = [max_ - x for x in nums]
least = sum(left)
if b < least:
    print -1
else:
    temp = (b - least) / n
    for i in xrange(n):
        print "%.6f" % (temp + left[i])
