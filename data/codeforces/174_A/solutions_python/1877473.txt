import sys

file = sys.stdin
#file = open("test", "r")
n, b = map(int, file.readline().rstrip().split())
p = map(int, file.readline().rstrip().split())
d = (b + sum(p))/float(n)
if n*d != b + sum(p): 
    print "-1"
else:
    if d < max(p):
        print "-1"
    else:
        for i in range(n):
            print("%.6f" % (d - p[i]))
