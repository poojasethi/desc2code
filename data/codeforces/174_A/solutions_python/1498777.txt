def sol():
    n, b = map(int, raw_input().split())
    a = map(int, raw_input().split())
    summa = sum(a) + b
    to_each = float(summa) / n
    i = 0
    while i < n:
        out = to_each - a[i]
        if out < 0:
            print '-1'
            return
        else:
            a[i] = out
        i += 1
    i = 0
    while i < n:
        print "%.6f" % a[i]
        i += 1

sol()
