def solve():
    n,b = map(int, raw_input().strip().split())
    a = map(int, str(raw_input()).split())
    avg = round((b+sum(a))/float(n), 7)
    x = []
    for i in a:
        if i>avg:
            print '-1'
            return
        x+=[avg-i]
    for i in x:
        print "%.6f" % i

solve()
