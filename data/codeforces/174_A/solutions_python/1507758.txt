n, b = map(int, raw_input().strip().split())
c = map(int, raw_input().strip().split())

max_c = max(c)
up_to_max = map(lambda i: max_c - i, c)
total = sum(up_to_max)
if total > b:
    print "-1"
else:
    additional = float(b-total)/float(n)
    for i,v in enumerate(c):
        print "%06f" % (up_to_max[i]+additional)
