debug = False
if debug: f = open('temp.in', 'r')

def inp():
    if debug: return f.readline().strip()
    else: return raw_input().strip()

n, b = map(int, inp().split())
a = map(int, inp().split())
total = sum(a) + b
avg = total/(1.0 * len(a))

if any(ai > avg for ai in a):
    print -1
else:
    for ai in a: print ("%.6f")%(avg-ai)
