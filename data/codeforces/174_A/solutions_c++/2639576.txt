# include <stdio.h>
int main()
{
	float a[100],s=0.0,sum,t,x=0;
	int i,b,n;
	while(scanf("%d %d",&n,&b)!=EOF)
	{
		for(i=0;i<n;i++)
		{
			scanf("%f",&a[i]);
		}
		for(i=0;i<n;i++)
		{s=s+a[i];}
		sum=s+b;
		t=sum/n;
		for(i=0;i<n;i++)
		{
			a[i]=t-a[i];
			if(a[i]>=0)
				x=x+1;
		}
		if(x==n)
		{	for(i=0;i<n;i++)
			{printf("%f\n",a[i]);}
		}
		else
			printf("-1\n");
	}
}
