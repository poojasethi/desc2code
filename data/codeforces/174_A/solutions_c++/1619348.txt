#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
	double n,b,a[105];
	while(~scanf("%lf%lf",&n,&b))
	{
		double max=0,sum=b;
		for(int i=0;i<n;i++)
		{
			scanf("%lf",&a[i]);
			sum+=a[i];
			max=max>a[i]?max:a[i];
		}
		if(max>sum/n) printf("-1\n");
		else
			for(int i=0;i<n;i++)
				printf("%.6lf\n",sum/n-a[i]);
	}
	return 0;
}
