#include <cstdio>
#include <cstring>
const int maxn = 103;

int main() {
	int n, i;
	double b, MM, d;
	double a[maxn];
	scanf("%d%lf",&n,&b);
	MM = d = 0;
	for (i=0; i < n; i++) {
		scanf("%lf",&a[i]);
		if ( a[i] > MM )	MM = a[i];
		d += a[i];
	}
	d = n*MM - d;
	if ( d > b )  
		printf("-1");
	else {
		for (i=0; i < n; i++) {
			printf("%.6lf\n",MM-a[i]+(b-d)/n);
		}
	}
	return 0;
}
