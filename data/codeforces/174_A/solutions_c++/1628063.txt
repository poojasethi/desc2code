#include<iostream>
#include<cstdio>
using namespace std;

int main(){
	int n;
	double b,m[105],m2[105];
	cin>>n>>b;
	double sum = b;
	for(int i=0;i<n;i++)
		cin>>m[i], sum+=m[i];
	double ser = sum/n;
	for(int i=0;i<n;i++)
	{
		if((ser-m[i])<0) {cout<<"-1\n";return 0;}
		m2[i]=(ser-m[i]);
	}
	cout.precision(7);
	for(int i=0;i<n;i++)
		printf("%.7lf\n",m2[i]);
return 0;
}
