#include <cstdio>
#include <algorithm>
using namespace std;
int n,b,a[100005];
int main(){
    scanf("%d%d",&n,&b);
    for(int i=1; i<=n; ++i) scanf("%d",&a[i]);
    int M=0,sum=0;
    for(int i=1; i<=n; ++i) M=max(M,a[i]);
    for(int i=1; i<=n; ++i) sum+=M-a[i];
    if (sum>b) printf("-1\n");
    else
        for(int i=1; i<=n; ++i) printf("%.6lf\n",M-a[i]+(b-sum)*1.0/n);
    return 0;
}
