#include <iostream>
#include <cstdio>

using namespace std;

int N,B,A[110],total;

void solve()
{
	double amount=double(total+B)/N;
	for (int i=0;i<N;i++)
		if (amount<A[i])
		{
			cout << -1 << endl;
			return;
		}
	for (int i=0;i<N;i++)
		printf("%.6lf\n",amount-A[i]);
}

int main()
{
	cin >> N >> B;
	for (int i=0;i<N;i++)
	{
		cin >> A[i];
		total+=A[i];
	}
	solve();
	return 0;
}
