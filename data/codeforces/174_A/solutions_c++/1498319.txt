#include<cstdio>

int n,b;
double maxa;
double a[101];
double sum;

int main(){
 scanf("%d%d",&n,&b);
 int i;
 maxa = 0;
 sum = 0;
 for(i=0;i<n;i++){
  scanf("%lf",a+i);
  sum+=a[i];
  if(a[i]>maxa) maxa=a[i];
 }
 if(sum+b<maxa*n){
  printf("-1\n");
  return 0;
 }
 maxa = (sum+b)/(double)n;
 for(i=0;i<n;i++) printf("%lf\n",maxa-a[i]);

 return 0;
}
