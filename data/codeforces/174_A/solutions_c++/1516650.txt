#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
int a[1000],n,b;
float c[1000];
cin>>n>>b;
int i,max=0,sum=0;
for(i=0;i<n;i++){cin>>a[i];sum+=a[i];if(a[i]>max)max=a[i];}
//if(sum>b){cout<<-1;return 0;}
for(i=0;i<n;i++)b=b-(max-a[i]);
if(b<0){cout<<-1;return 0;}
if(b>0)for(i=0;i<n;i++)c[i]=(float)b*1.0/n;

for(i=0;i<n;i++){c[i]+=max-a[i];printf("%.6f\n",c[i]);}
return 0;
}
