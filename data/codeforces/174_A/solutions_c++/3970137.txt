#include<stdio.h>
#define maxi(a,b) ((a>b)? a:b)
int n,b,sum=0;
double x;
int A[105];
int main()
{
      int i,max=-1;
      scanf("%d %d",&n,&b);

      for(i=0;i<n;i++) {scanf("%d",A+i); sum+=A[i]; max=maxi(max,A[i]);}

      x=(b+sum)/(double)n;

      if(x<max) {puts("-1"); return 0;}

      for(i=0;i<n;i++) printf("%lf\n",x-A[i]);

      return 0;
}
