#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;


float c[102],b,n,a[102],m=-1;
int main()
{
	
	cin>>n>>b;
	for(int i=0; i<n; i++)
	{
		cin>>a[i];
		m=max(m,a[i]);
	}
	for(int i=0; i<n; i++)
	{
		c[i]+=m-a[i];
		b-=c[i];
	}
	if(b<0)
		cout<<-1<<endl;
	else
	{
		float pe=b/n;
		for(int i=0; i<n; i++)
			printf("%.6f\n",pe+c[i]);
		
	}
}
