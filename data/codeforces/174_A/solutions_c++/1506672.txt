#include <cstdio>
double t[1000];
int main()
{
	int n;
	double all,sum=0.0;
	scanf("%d%lf",&n,&all);
	sum=all;
	for(int i=0;i<n;i++)
		scanf("%lf",&t[i]),sum+=t[i];
	double avg=sum/n;
	bool ans=true;
	for(int i=0;i<n;i++)
		if(t[i]>avg)
			ans=false;
	if(!ans)puts("-1");
	else
		for(int i=0;i<n;i++)
			printf("%.6lf\n",avg-t[i]);
	scanf(" ");
}
