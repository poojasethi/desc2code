#include <cstdio>
using namespace std;
int tab[101];
int n,b;
int sum;
int main()
{
	scanf("%d%d",&n,&b);
	sum = b;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&tab[i]);
		sum+=tab[i];	
	}

	for(int i=0;i<n;i++)	
		if(tab[i] > double(sum)/double(n)) {printf("-1"); return 0;}	
	for(int i=0;i<n;i++)
		printf("%.6lf\n",double(sum)/double(n) - tab[i]);

	return 0;
}
