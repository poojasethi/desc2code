#include<stdio.h>
int main()
{
	int n,b,a[100],i,f=0;double c=0;
	scanf("%d %d",&n,&b);
	c+=b;
	for(i=0;i<n;++i)
	{
		scanf("%d",a+i);
		c+=a[i];
	}
	c/=n;
	for(i=0;i<n;++i)
	{
		if(c-a[i]<0)
		{
			f=1;
			break;
		}
	}
	if(f==0)
	{
		for(i=0;i<n;++i)printf("%f ",c-a[i]);
	}
	else printf("-1");
	return 0;
}
