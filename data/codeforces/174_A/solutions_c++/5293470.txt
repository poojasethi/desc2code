#include <stdio.h>
int main()
{
	int n,m,i,a[200],s,max=0;
	scanf("%d%d",&n,&m);
	s=m;
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		if(a[i]>max)
			max=a[i];
		s+=a[i];
	}
	if(s>=n*max)
		for(i=0;i<n;i++)
			printf("%.6lf\n",1.0*s/n-a[i]);
	else
		printf("-1\n");
	return 0;
}
