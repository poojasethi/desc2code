#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	int n,b;
	double s=0,ave,a[100];
	int i,c=0;
	cin >>n>>b;
	for(i=0;i<n;i++)
	{
		cin >>a[i];
		s=s+a[i];
	}
	s=s+b;
	ave=s/n;
	for(i=0;i<n;i++)
	{
		if(a[i]>ave)
		{
			
			cout <<"-1"<<endl;
			c=1;
			break;
		}
		a[i]=ave-a[i];
	}
	if(c==0)
	{
		for(i=0;i<n;i++)
		{
			printf("%.6lf\n",a[i]);
		}
	}
}
