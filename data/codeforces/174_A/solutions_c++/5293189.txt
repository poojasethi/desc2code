#include<stdio.h>
int main()
{
	int n,b,j,i;
	__int64 a[200],sum;
	double ave;
	scanf("%d%d",&n,&b);
	sum=b;
	for(i=0;i<n;i++)
	{
		scanf("%I64d",&a[i]);
		sum+=a[i];
	}
	ave=1.0*sum/n;
	for(i=0;i<n;i++)
	{
		if(ave<a[i])
		{
			printf("-1\n");
			break;
		}
	}
	if(i==n)
	{
		for(j=0;j<n;j++)
		printf("%.6lf\n",ave-a[j]);
	}
	return 0;
}
