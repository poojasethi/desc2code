#include <iostream>
#include <cstdio>

using namespace std;

int main() {
  int n, b;
  cin >> n >> b;
  int a[n];
  int sum = 0, M = 0;
  for (int i = 0; i < n; i++) {
    cin >> a[i];
    M = max(M, a[i]);
    sum += a[i];
  }
  double A = (double)(sum + b) / n;
  if (A < M)
    cout << -1;
  else
    for (int i = 0; i < n; i++)
      printf("%.7lf\n", A - a[i]);
}
