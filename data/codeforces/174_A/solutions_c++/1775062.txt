#include <cstdio>

int main()
{
    int n,b,i;
    int a[100];
    float c[100];
    float f,l;
    scanf("%d%d",&n,&b);
    for (f=b,i=0;i<n;i++)
	{
		scanf("%d",a+i);
		f+=a[i];
	}
	l=f;f=f/n;
	for (i=0;i<n;i++)
	{
		c[i]=f-a[i];
		if (c[i]<0)
		{
			printf("%d",-1);
			return 0;
		}
		l-=c[i];
	}

	for (i=0;i<n;i++)
		printf("%6f\n",c[i]);



    return 0;
}
