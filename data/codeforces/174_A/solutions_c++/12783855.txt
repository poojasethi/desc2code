#include<iostream>
#include<iomanip>
using namespace std;

int n,a[100],b,to=0;

int main(){
	cin >> n >> b;
	for(int i=0; i<n; i++){
		cin >> a[i];
		if(to<a[i])to=a[i];
	}
	for(int i=0; i<n; i++)b-=to-a[i];
	if(b<0){cout << -1 << endl; return 0;}
	for(int i=0; i<n; i++){
		cout << setprecision(6) << fixed << (double)(to-a[i]+b/(double)n) << " ";
	}
	return 0;
}
