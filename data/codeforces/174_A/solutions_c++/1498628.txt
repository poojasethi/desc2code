#include <cstdio>
#include <algorithm>

double a[1000];

int main()
{
  int i;
  double n,b,max=0,tot=0;
  scanf("%lf %lf",&n,&b);
  for (i=0;i<n;i++)
  {
    scanf("%lf",&a[i]);
    if (a[i]>max)max=a[i];
  }
  for (i=0;i<n;i++)
    tot+=max-a[i];
  if (b<tot)
    puts("-1");
  else
  {
    for (i=0;i<n;i++)
      printf("%lf\n",(b-tot)/n+max-a[i]);
  }
}
