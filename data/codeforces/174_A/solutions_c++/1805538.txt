#include <cstdio>
double coup[105];
int main(){
	int i,n,v;
	double soma,vf;
	scanf ("%d%d",&n,&v);
	soma = 0;
	for (i = 0; i < n; i++){ scanf("%lf",&coup[i]); soma += coup[i];}
	vf = (soma+v) / n;

	for (i = 0; i < n; i++) if (coup[i] > vf) {printf("-1\n"); return 0;}

	for (i = 0; i < n; i++){ 
		printf ("%.6lf\n",vf-coup[i]);
	}

	return 0;
}
