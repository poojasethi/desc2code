#include<stdio.h>
int main()
{
	 int i;
	 double n,b,a[1000],t,s;
	 scanf("%lf%lf",&n,&b);
	 s=b;
	 for(i=0;i<n;i++)
	 {
		 scanf("%lf",&a[i]);
		 s+=a[i];
	 }
	 t=s/n;
	 for(i=0;i<n;i++)
	 {
		 a[i]=t-a[i];
		 if(a[i]<0)
		 {
			 printf("-1\n");
			 return 0;
		 }
	 }
	 for(i=0;i<n;i++)
		 printf("%.6lf\n",a[i]);
	 return 0;
}
