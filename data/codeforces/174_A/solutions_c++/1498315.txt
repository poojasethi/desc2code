#include <iostream>
#include <cstdio>
using namespace std;
int n, b, s, a[1000];
int main()
{
  cin >> n >> b;
  s = b;
  for(int i = 0; i < n; i++)
    cin >> a[i], s += a[i];
  double x = s * 1.0 / n;
  for(int i = 0; i < n; i++)
    if (a[i] > x)
      return cout << -1, 0;
  for(int i = 0; i < n; i++)
    printf("%.7lf\n", x - a[i]);
  return 0;
}
