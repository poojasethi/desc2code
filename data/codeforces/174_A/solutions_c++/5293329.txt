#include <stdio.h>
#include <math.h>
int main()
{
int n,i,b;
double a[100];
double  sum,ave;
int temp=0;
scanf("%d%d",&n,&b);
sum=b;
for (i=0;i<n;i++)
{
	scanf("%lf",&a[i]);
	sum=sum+a[i];
}
ave=sum/n;
for (i=0;i<n;i++)
{if (a[i]>ave){temp=1;break;}}
if (temp==1) printf("-1\n");
else
for (i=0;i<n;i++)
{printf("%.6lf\n",ave-a[i]);
}

return 0;}
