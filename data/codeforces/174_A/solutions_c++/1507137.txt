#include<cstdio>
#include<iostream>
using namespace std;
int n,s,m=-1,a[100001];
int main(){
  scanf("%d%d",&n,&s);
  for(int i=1;i<=n;++i){
    scanf("%d",&a[i]);
    s+=a[i];
    if(a[i]>m) m=a[i];
  }
  if(s/n<m)
    printf("-1");
  else
    for(int i=1;i<=n;++i)
      printf("%.8lf\n",(double)s/n-(double)a[i]);
}
