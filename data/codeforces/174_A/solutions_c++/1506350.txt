#include<cstdio>
int n,b;
int q[105];
int main()
{
    scanf("%d%d",&n,&b);
    double s=0;
    int mx=0;
    for(int i=0;i<n;i++)
    {
        scanf("%d",q+i);
        s+=q[i];
        if(q[i]>mx) mx=q[i];
    }
    double x=(s+b)/n;
    if(mx>x) puts("-1");
    else
    {
        for(int i=0;i<n;i++)
        {
            printf("%.6lf\n",x-q[i]);
        }
    }

}
