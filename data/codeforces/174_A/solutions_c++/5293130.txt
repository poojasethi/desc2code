#include<stdio.h>

int main()
{
	int n,b;
	__int64 a[100];
	int i,x=0;
	double v;
	scanf("%d%d",&n,&b);
	__int64 s=0;
	for(i=0;i<n;i++)
	{scanf("%I64d",&a[i]);
	s+=a[i];
	}
	s+=b;
	v=1.0*s/n;
	for(i=0;i<n;i++)
	{if(a[i]>v)
	{x=1;break;}
	}
	if(x==1)
		printf("-1\n");
	else
	{
		for(i=0;i<n;i++)
			printf("%.6f\n",v-a[i]);
	}
	return 0;
}
