#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<set>
#include<stack>
#include<queue>
#include<algorithm>
#define LiangJiaJun main
using namespace std;
struct dance{
    int l,r,v;
    dance(int a, int b, int c){l=a,r=b,v=c;}
    bool operator < (const dance&a) const{
        return a.v<v || a.v==v && a.l<l;
    }
};
priority_queue<dance>q;
struct ANS{
    int l,r;
}ans[200004];
int num=0,n;
char s[200004],vis[200004];
int pre[200004],next[200004],w[200004];
int LiangJiaJun(){
    memset(vis,0,sizeof(vis));
    scanf("%d%s",&n,s+1);
    for(int i=1;i<=n;i++)scanf("%d",&w[i]), pre[i]=i-1,next[i]=i+1;
    for(int i=1;i<n;i++)
        if(s[i]!=s[i+1])q.push(dance(i,i+1,abs(w[i]-w[i+1])));
    while(!q.empty()){
        dance u=q.top(); q.pop();
        if(vis[u.l]||vis[u.r]) continue;
        vis[u.l]=vis[u.r]=1;
        ans[++num].l=u.l;
        ans[num].r=u.r;
        int L= pre[u.l],R=next[u.r];
        next[L]=R,pre[R]=L;
        if(L>=1 && R<=n && s[L]!=s[R])
            q.push(dance(L,R,abs(w[L]-w[R])));
    }
    printf("%d\n",num);
    for(int i=1;i<=num;i++)printf("%d %d\n",ans[i].l,ans[i].r);
    return 0;
}
