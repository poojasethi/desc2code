#include <algorithm>
#include <iostream>
#include<stdio.h>
#include<string.h>
#include<queue>
#include<cmath>
#include<list>
#include<vector>
#define N 200005
#define inf 10000000
#define lst p<<1,l,mid
#define rst p<<1|1,mid+1,r
#define ll long long
#define Y second
#define X first
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define piii pair<long long,pii>
using namespace std;
priority_queue<piii>g;
pair<int,int>ans[N];
int v[N],pre[N],next[N];
ll a[N];
int main()
{
  //freopen("in.txt","r",stdin);
  int n,i;
  char s[N];
  while(~scanf("%d",&n))
  {
   getchar();
   for(i=1;i<=n;i++)
   scanf("%c",&s[i]);
   for(i=1;i<=n;i++)
   {
    scanf("%I64d",&a[i]);
    pre[i]=i-1;
    next[i]=i+1;
   }
   next[0]=1;pre[n+1]=n;
   for(i=2;i<=n;i++)
   {
    if(s[i]!=s[i-1])
    {
     ll j=-abs(a[i]-a[i-1]);
     g.push(mp(j*200000-i,mp(i-1,i)));
    }
   }
   memset(v,0,sizeof(v));
   int k=0;
   while(!g.empty())
   {
     piii pp=g.top();g.pop();
     int x=pp.Y.X,y=pp.Y.Y;
     if(v[x]||v[y])continue;
    v[x]=1;v[y]=1;
    int p=pre[x],q=next[y];
    next[p]=q;pre[q]=p;
    if(p>=1&&q<=n&&s[p]!=s[q])
    g.push(mp(-abs(a[p]-a[q])*200000-q,mp(p,q)));
    ans[k++]=mp(x,y);
   }
   printf("%d\n",k);
   for(i=0;i<k;i++)
   printf("%d %d\n",ans[i].X,ans[i].Y);
  }
  return 0;
}
