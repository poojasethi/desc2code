#include<cstdio>
#include<queue>
#include<cmath>
#include<cstdlib>
#include<vector>
using namespace std;
const int maxn = 200000+100;
typedef pair<int ,int > p1;
typedef pair<int ,p1  > p2;
int a[maxn],pre[maxn],next[maxn];
bool used[maxn];
char str[maxn];
vector<p1> ans;
int main(){
    priority_queue<p2,vector<p2>,greater<p2> > p;
    int n; scanf("%d",&n);
    scanf("%s",str+1);
    for(int i=1;i<=n;i++) scanf("%d",&a[i]);
    for(int i=1;i<=n;i++){
       pre[i]=i-1;
       next[i]=i+1;
       used[i]=true;
       if(i>1&&str[i]!=str[i-1]){
          p.push( make_pair( abs(a[i]-a[i-1]), make_pair(i-1,i) ) );
       }
    }
    while(!p.empty()){
        int i=p.top().second.first; int j=p.top().second.second; p.pop();
        if(used[i]&&used[j]){
            used[i]=used[j]=false;
            ans.push_back( make_pair(i,j) );
            i=pre[i]; j=next[j];
            next[i]=j;
            pre[j]=i;
            if(i>=1&&used[i]&&used[j]&&str[i]!=str[j]){
             p.push( make_pair( abs(a[j]-a[i]), make_pair(i,j) ) );
            }
        }
    }
    printf("%d\n", (int)ans.size());
    for (int i = 0; i < (int)ans.size(); ++i) {
        printf("%d %d\n", ans[i].first, ans[i].second);
    }

    return 0;
}
