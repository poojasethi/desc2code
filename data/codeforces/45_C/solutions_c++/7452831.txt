#include<cstdio>
#include<algorithm>
#include<vector>
#include<set>
#define pii pair<int,int> 
#define A first
#define B second
using namespace std;
set < pair<int,pii> > S;
set < pair<int,pii> >::iterator it;
pii l[200001];
vector <pii> ans;
int a[200001],tag[200001],n;
char s[200001];
int main(){
    scanf("%d",&n);
    scanf("%s",s);
    for(int i=0;i<n;i++) s[i] = (s[i] == 'B' ? 0 : 1);
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    for(int i=0;i<n;i++){
        l[i] = pii(i-1,i+1);
        if(i != n-1 && s[i]^s[i+1] == 1)
            S.insert(make_pair(abs(a[i]-a[i+1]),pii(i,i+1)));   
    }
    while(!S.empty()){
            it = S.begin();
            pair<int,pii> SV = *it;
            if(tag[SV.B.A] != 1 && tag[SV.B.B] != 1){
                tag[SV.B.A] = tag[SV.B.B] = 1;   
                ans.push_back(pii(SV.B.A,SV.B.B));
                int x1 = l[SV.B.A].A, x2 = l[(*it).B.B].B;
                if(x1 != -1) l[x1].B = x2;
                if(x2 != n) l[x2].A = x1;
                if(x1 != -1 && x2 != n && (s[x1]^s[x2] == 1))
                    S.insert(make_pair(abs(a[x1]-a[x2]),pii(x1,x2)));
            }
            S.erase(it);
    }
    printf("%d\n",ans.size());
    for(int i=0;i<ans.size();i++)
        printf("%d %d\n",ans[i].A+1,ans[i].B+1);
    return 0;
}
