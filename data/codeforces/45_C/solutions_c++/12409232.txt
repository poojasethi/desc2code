//In the Name of God

#include<iostream>
#include<set>
#include<vector>
#include<cmath>
#define mp make_pair
#define pb push_back

using namespace std;

int n,next[200200],prev[200200],a[200200],f,l;
string s;
vector<pair<int,int> > ans;
set<pair<int,pair<int,int> > > v;

int main()
{
  cin>>n;
  cin>>s;
  s="A"+s+"A";
  next[0]=1;prev[0]=0;
  next[n+1]=n+1,prev[n+1]=n;
  for(int i=1;i<=n;i++)
    next[i]=i+1,prev[i]=i-1;
  for(int i=1;i<=n;i++)
    cin>>a[i];
  for(int i=1;i<n;i++)
    if(s[i]!=s[i+1])
      v.insert(mp(abs(a[i]-a[i+1]),mp(i,i+1)));
  while(v.size())
    {
      f=v.begin()->second.first;
      l=v.begin()->second.second;
      ans.pb(mp(f,l));
      v.erase(v.begin());
      if(s[next[l]]!='A' && s[l]!=s[next[l]])
	v.erase(mp(abs(a[l]-a[next[l]]),mp(l,next[l])));
      if(s[prev[f]]!='A' && s[f]!=s[prev[f]])
	v.erase(mp(abs(a[f]-a[prev[f]]),mp(prev[f],f)));
      if(s[next[l]]!='A' && s[prev[f]]!='A' && s[next[l]]!=s[prev[f]])
	v.insert(mp(abs(a[next[l]]-a[prev[f]]),mp(prev[f],next[l])));
      next[prev[f]]=next[l];
      prev[next[l]]=prev[f];
    }
  cout<<ans.size()<<endl;
  for(int i=0;i<ans.size();i++)
    cout<<ans[i].first<<" "<<ans[i].second<<endl;
  return 0;
}
