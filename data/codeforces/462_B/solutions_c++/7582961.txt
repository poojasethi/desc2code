#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	long long no[26]={0},N,K,a,b,c,res=0,l;
	string s;
	cin>>N>>K;
	cin>>s;
	l=s.length();
	for(int i=0;i<l;i++)
		no[s[i]-'A']++;
	sort(no,no+26);
	int i=25;
	while(i>=0 && K>0)
	{
		a=min(K,no[i]);
		res+=(a*a);
		K-=a;
		i--;
	}
	cout<<res<<"\n";
	
	return 0;
}
