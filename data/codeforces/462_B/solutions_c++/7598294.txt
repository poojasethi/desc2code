#include<iostream>
#include<algorithm>
#include<string>
using namespace std;

long long n, k, ans, a[26], mi;
string s;

int main(){
  cin >> n >> k >> s;
  for(int i = 0; i < n; i++)
    a[s[i] - 'A']++;
  sort(a, a + 26);
  int t = 25;
  while(k)
    mi = min(k, a[t]), ans += (mi * mi), k -= mi, t--;
  cout << ans;
  return 0;
}
