#include<bits/stdc++.h>
#define swap(a,b) (((a)^=(b)),((b)^=(a)),((a)^=(b)))
using namespace std;
int n,a[26],k;
long long p,s;
char c;

int main(void)
{
	cin>>n>>k;
	while(cin>>c)a[c-'A']++;
	n=25;
	sort(a,a+26);
	while(n>=0&&k>0)
	{
		s=min(a[n],k);
		p+= s*s;
		k-=s;
		n--;
	}
	cout<<p;

	return 0;
}
