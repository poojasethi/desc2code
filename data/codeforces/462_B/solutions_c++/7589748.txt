#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,k,i,ans=0,f[30]={0};
	string s;
	cin>>n>>k>>s;
	for(i=0;i<s.size();i++)
		f[s[i]-'A']++;
	sort(f,f+26);
	for(i=25;i>=0 && k>=0;i--)
	{
		if(k-f[i]>=0)
		{
			k-=f[i];
			ans+=f[i]*f[i];
		}
		else
		{
			ans+=k*k;
			break;
		}
	}
	cout<<ans<<endl;
	return 0;
}
