#include <iostream>
#include <algorithm>
using namespace std;
int n, k, i;
long a[26];
long long res, x;
string s;
int main()
{
	cin>>n>>k>>s;
	for (i=0; i<n; i++) a[s[i]-'A']++;
	sort(a, a+26);
	for (i=25; k>0; i--) {
		x=a[i];
		if (x>k) x=k;
		k -= x;
		res += x*x;
	}
	cout<<res<<endl;
	return 0;
}
