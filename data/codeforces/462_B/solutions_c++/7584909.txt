#include <iostream>
#include <algorithm>
using namespace std;

long long n,k,a[100],ts;
char s[100001];

int main()
{
	int i,j;
	cin>>n>>k;
	cin>>s;
	for (i=0;i<n;i++)
		a[s[i]-'A'+1]++;
	sort(a+1,a+27);
	for (i=26;i;i--)
		if (a[i]<k)
		{
			ts+=a[i]*a[i];
			k-=a[i];
		}
		else
		{
			ts+=k*k;
			break;
		}
	cout<<ts;
}
