#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;
long long ans,c[256],n,k;
char s[100005];

int main()
{
	cin >> n >> k;
	scanf("%s",s);
	while(n--)
		c[s[n]]++;
	sort(c+'A',c+'Z'+1);
	ans = 0; n = 'Z';
	while(k>0)
	{
		ans += min(k*k,c[n]*c[n]);
		k -= c[n];
		n--;
	}
	cout << ans << '\n';
	return 0;
}
