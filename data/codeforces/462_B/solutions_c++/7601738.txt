#include<bits/stdc++.h>
using namespace std;
int n, k, i, a[26];
long long ans, x;
string s;
int main()
{
    cin>>n>>k>>s;
    for (i=0; i<s.size(); i++) a[s[i]-'A']++;
    sort(a, a+26);
    for (i=25; k>0; i--) {
        x = min(k,a[i]);
        k -= x;
        ans += x*x;
    }
    cout<<ans<<endl;
    return 0;
}
