#include <iostream>
#include <algorithm>
using namespace std;

long long n,k,i,sum=0,c=0,a[26];
string s;

int main()
{
    cin>>n>>k>>s;
    for(i=0;i<n;i++)
        a[s[i]-'A']++;
    sort(a,a+26);
    for(i=25;k>0;i--)
    {
        c=min(a[i],k);
        sum+=c*c;
        k-=c;
    }
    cout<<sum<<endl;
    return 0;
}
