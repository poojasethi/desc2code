#include <bits/stdc++.h>
using namespace std;
long long n,m,r[29],x,idx;
string a;
int main(){
    cin>>n>>m>>a;
    for(int i=0;i<a.size();i++){
        r[a[i]-'A']++;
    }
    sort(r,r+26);
    for (int i=25;m>0;i--){
        x = min(m,r[i]);
        idx+=x*x;
        m-=x;
    }
    cout<<idx<<endl;
    return 0;
}
