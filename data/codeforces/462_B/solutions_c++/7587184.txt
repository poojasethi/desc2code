#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	long long n,k,i,h[26]={0},sum=0;
	cin>>n>>k;
	char a[n];
	cin>>a;
	for(i=0;a[i];i++)
	h[a[i]-'A']++;
	sort(h,h+26,greater<long long>());
	for(i=0;i<26;i++)
	{
		if(k<=h[i])
		{
			sum+=k*k;
			break;
		}
		else
		sum+=h[i]*h[i];
		k-=h[i];
	}
	cout<<sum<<endl;
	return 0;
}
