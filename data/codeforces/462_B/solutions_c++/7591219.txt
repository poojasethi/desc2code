#include <bits/stdc++.h>

using namespace std;

long long i,j,t,k,n,m,a[1000],s;

char c;

int main(){
	
	scanf("%lld %lld",&n,&k);
	for(i=0;i<n;i++){
		scanf(" %c",&c);
		a[c-'A']++;
	}
	
	sort(a,a+26,greater<int>());
	
	for(i=0;i<26;i++)
	{
		long long t=min(a[i],k);
		s+=t*t;
		k-=t;
	}

	cout << s;	
	
	return 0;
}
