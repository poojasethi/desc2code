#include <bits/stdc++.h>
using namespace std;


string s;
long long n, k, freq[128], res,i;

int main(){
cin >> n >> k >> s;

for(i=0;i<n;i++)
freq[s[i]]++;

sort(freq,freq+128);
reverse(freq,freq+128);

for(i=0;i<128;i++)
{
    long long j=min(k,freq[i]);
    k-=j;
    res+=j*j;
}

cout<<res;
    
    return 0;
}
