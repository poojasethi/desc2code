#include<cstdio>
#include<algorithm>
using namespace std;
int a1[26];
char s[100001];
int main()
{
    int n,k;long long a,o=0;
    scanf("%d%d%s",&n,&k,s);
    for(int i=0;i<n;++i)
        ++a1[s[i]-65];
    sort(a1,a1+26);
    for(int i=25;k&&~i;--i)
    {
        a=min(a1[i],k);
        k-=a;o+=a*a;
    }
    printf("%I64d\n",o);
}
