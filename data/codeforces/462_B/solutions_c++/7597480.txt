#include <bits/stdc++.h>
using namespace std;
long long k,n;
string t;
long long a[26];
int main(){
	cin >> n >> k;
	cin >> t;
	for(int i=0;i<t.size();i++)
		a[t[i]-'A']++;
	sort(a,a+26,greater<long long>());
	long long ans=0;
	for(int i=0;i<26&&k;i++){
		ans+=min(k,a[i])*min(k,a[i]);
		k-=min(a[i],k);
	}
	cout << ans << endl;
	return 0;
}
