#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
int a[33];
char c;
ll ans;

int main(){
	int n,m;
	cin>>n>>m;
	c=getchar();
	while (n--){
		c=getchar();
		a[c-'A']++;
	}
	sort(a,a+26,greater<int>());
	for (int i=0;i<26;i++){
		ll t=min(m,a[i]);
		m-=t;
		ans+=t*t;
	}
	cout<<ans<<endl;
}
