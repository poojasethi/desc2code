#include<iostream>
#include<set>
#include <algorithm>
#include <map>
using namespace std;
long long n,a[100],f,o,k;
string s;
int main(){
	cin>>n>>k>>s;
	for(int i=1;i<=n;i++){
		a[int(s[i-1])-64]++;
	}
	sort(a+1,a+1+26);
	for(int i=1;i<=26;i++){
		if(a[27-i]<=k)o+=a[27-i]*a[27-i],k=k-a[27-i];
		else o+=k*k,k=0;
	}
	cout<<o;

	
}
