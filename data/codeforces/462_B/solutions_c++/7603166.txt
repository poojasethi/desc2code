#include<iostream>
#include<string>
#include<algorithm>
#include<cstdio>
using namespace std;
long long a[100],b,c,n,k,i,res;
string s;

int main()
{
cin>>n>>k>>s;
for(i=0;i<n;i++)
a[s[i]-65]++;
sort(a,a+26);
for(i=25;i>=0;i--){
	if(k>=a[i]){
		k-=a[i];
		res+=a[i]*a[i];
	}
	else{
		res+=k*k;
		k=0;
	}
	if(k==0)
	break;
}
cout<<res;
}
