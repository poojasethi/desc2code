#include<bits/stdc++.h>
using namespace std;
long long a[26],i,j,n,k,x,ans=0;
string s;
int main()
{
  scanf("%lld %lld",&n,&k);
  cin>>s;
  for(i=0;i<n;i++)
        a[s[i]-'A']++;
        sort(a,a+26);
  for(i=25;k>0;i--)
  {
     x=min(a[i],k);
      ans=ans+(x*x);
      k=k-x;
  }
  printf("%lld",ans);
  return 0;

}
