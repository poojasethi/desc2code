#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
long long  n,k,o,sum=0,i;
cin>>n>>k;
char s[100010];
cin>>s;
long long a[30];
for(i=0;s[i]!='\0';i++)
{  a[s[i]-64]++;
}
sort(a+1,a+27);
o=26;
while(k>=a[o])
{
    sum+=a[o]*a[o];
    k-=a[o];
    o--;
}
if(k!=0)
    sum+=k*k;
 cout<<sum;


}
