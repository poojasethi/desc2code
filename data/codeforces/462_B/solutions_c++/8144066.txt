#include <iostream>
#include <algorithm>

using namespace std;

string s;

long long ans, v[30], k, n;

int main()
{
	cin >>  n >> k >> s;
	
	for (int i = 0; i < s.length(); i++)
		v[s[i]-65]++;
	
	sort(v,v+26);
	
	for (int i = 25; i >= 0; i--)
	{
		if (k >= v[i])
			ans += v[i]*v[i], k-= v[i];
		else
			ans += k*k, k = 0;
	}
	cout << ans;
}
