from collections import Counter
n,k=map(int, raw_input().split())
cnt=Counter(raw_input())
x=[]
for i in cnt:
    x.append(cnt[i])
x.sort(reverse=True)
ans,cards,i=0,0,0
while i<len(x):
    if x[i]+cards >= k:
        ans+= (k-cards)**2
        break
    cards+=x[i]
    ans+= x[i]**2
    i+=1
print ans
