from collections import Counter
import operator

#br = open('b.in')
n, k = map(int, raw_input().strip().split())
s = raw_input().strip()
vk = sorted(Counter(s).iteritems(), key=operator.itemgetter(1), reverse=True)
r = 0

for u, v in vk:
  if k - v < 0:
    r += k * k
    break
  r += v * v
  k -= v

print r
