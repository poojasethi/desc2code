n,k=map(int,raw_input().split())
s=raw_input()
s=list(s)
hash={}
for i in s:
    if i in hash:
        hash[i]+=1
    else:
        hash[i]=1
li=sorted(hash.values())
li=li[::-1]
ans=0
for i in li:
    if(i<=k):
        ans+=i*i
        k-=i
    else:
        ans+=k*k
        break
print ans
