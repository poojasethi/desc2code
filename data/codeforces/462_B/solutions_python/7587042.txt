import operator
N,K=raw_input().split(' ')
N=int(N)
K=int(K)
a=raw_input()
ans={}
for i in a:
	try:
		ans[i]+=1
	except KeyError:
		ans[i]=1
ans= sorted(ans.iteritems(), key=operator.itemgetter(1),reverse=True)
req=0
for i in ans:
	if(i[1]<K):
		req+=i[1]*i[1]
		K-=i[1]
	else:
	 	req+=K*K
	 	break
print req
