nk = raw_input().split()
n = int(nk[0])
k = int(nk[1])
count = [0]*26
l = raw_input()
for i,c in enumerate(l):
    count[ord(c)-ord('A')]+=1
count.sort(reverse=True)
ind = 0;ans = 0
while(k>0 and ind<26):
    if count[ind]>k:
        ans+=(k**2)
    else:
        ans+=(count[ind]**2)
    k-=count[ind]
    ind+=1
print ans
