import sys
from collections import Counter
[n,k] = [int(x) for x in sys.stdin.readline().split()]
c = Counter(sys.stdin.readline().strip())
li = sorted(c.values(), reverse=True)
ctr = 0
for x in range(len(li)):
    if k - li[x] >= 0:
        k -= li[x]
        ctr += li[x]**2
    else:
        ctr += k**2
        break
print ctr
