n, k = map(int, raw_input().split())
s = raw_input()
d = {}
for c in s:
  if not c in d:
    d[c] = 0
  d[c] += 1
d = list(sorted(d.items(), key=lambda x: x[1], reverse=True))
r = []
for x, y in d:
  t = min(y, k)
  y, k = y-t, k-t
  r.append((x, t))
  if k == 0: break
res = 0
for x, y in r:
  res += y*y
print res
