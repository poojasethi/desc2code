
n, k = map(int, raw_input().split()) 

cards = raw_input()

cnt = [0 for i in range(26)] 

for c in cards: 
    cnt[ord(c)-ord('A')] += 1 

cnt.sort(reverse=True) 

scores = 0
for i in range(26): 
    if k <= 0: 
        break 
    num = min(k, cnt[i]) 
    scores += num*num
    k -= num 

print scores
