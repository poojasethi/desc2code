n,k=map(int,raw_input().split())
w=raw_input()
l=[0]*26
for i in w:
    l[ord(i)-65]+=1
ans=0
chk=k
while 1:#while 1:,while True ha mugen roop
    ma=max(l)
    if chk>=ma:
        chk-=ma
        ans+=ma**2
        l[l.index(ma)]=0
    elif chk<ma:
        ans+=chk**2
        chk=0
    if chk==0:
        break
print ans
