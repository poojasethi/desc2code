#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
const double pi=acos(-1);
struct point{
       double x,y;
       }a;
int n;
double x,y;
double len(point a){
       return sqrt(a.x*a.x+a.y*a.y);
       }
point turn(point a,double ang){
      point b;
      b.x=a.x*cos(ang)-sin(ang)*a.y;
      b.y=a.x*sin(ang)+cos(ang)*a.y;
      return b;
      }
point make(point a,double l){
      double k=l/len(a);
      a.x*=k;
      a.y*=k;
      return a;
}
double cal(point a,point b){
       return acos((a.x*b.x+a.y*b.y)/len(a)/len(b));
       }
int main(){
    cin>>n;
    if(n<=4){
      printf("No solution\n");
      return 0;
      }
    x=y=0;
    double l=400;
    for(int i=0;i<n-1;i++){
      double ang=pi*2*i/n;
      x+=l*cos(ang);y+=l*sin(ang);
      l+=0.015;
      printf("%.3lf %.3lf\n",x,y);
    }
    double ang=pi*2/n;
    printf("%.3lf 0\n",x+y/sin(ang)*cos(ang));
    return 0;
}
