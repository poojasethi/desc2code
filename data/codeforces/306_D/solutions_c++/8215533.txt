#include <cmath>
#include <cstdio>
#include <algorithm>

using namespace std;

const double pi = acos(-1);

int main() {
	int n;
	scanf("%d", &n);
	if (n <= 4) {
		printf("No solution\n");
		return 0;
	}
	double A = pi * 2 / n;

	double x = 0, y = 0, step = 200;
	for (int i = 1; i < n; ++i) {
		if (i == (n + 1) / 2 + 1) step += 100;
		x += step * cos(A * (i - 1));
		y += step * sin(A * (i - 1));
		if (i == (n + 1) / 2 + 1) step -= 100;
		printf("%lf %lf\n", x, y);
		step -= 0.01;
	}
	double t = -y / sin(A * (n - 1));
	x += t * cos(A * (n - 1));
	y += t * sin(A * (n - 1));
	printf("%lf %lf\n", x, y);
}
