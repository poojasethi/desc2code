#include<cstdio>
#include<cmath>
const double pi = 4.0*atan(1.0);
int n;
int main()
{
	int i;
	scanf("%d",&n);
	if(n<5)
	{
		printf("No solution\n");
		return 0;
	}
	double x=0,y=0;
	printf("%.3f %.3f\n",x,y);
	double len=300;
	double jiao=0;
	double every=pi*2.0/n;
	for(i=1;i<=n-2;i++)
	{
		jiao+=every;
		x+=cos(jiao)*len;
		y+=sin(jiao)*len;
		printf("%.3f %.3f\n",x,y);
		len+=0.01;
	}
	jiao+=every;
	x-=y/tan(jiao);
	y=0;
	printf("%.3f %.3f\n",x,y);
	return 0;
}
