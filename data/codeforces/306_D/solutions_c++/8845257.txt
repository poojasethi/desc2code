#include <cstdio>
#include <cmath>
const double PI = acos(-1);
int n;
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("306D.in","r",stdin);
	freopen("306D.out","w",stdout);
	#endif
	scanf("%d",&n);
	if (n <= 4) return puts("No solution");
	double x = 0,y = 0,px,py,l = 100;
	for (int i = 1; i < n; ++i) {
		printf("%.5f %.5f\n",x,y);
		px = x; py = y;
		x += l * cos(2 * PI / n * i); y += l * sin(2 * PI / n * i);
		l += 0.002;
	}
	printf("%.5f %.5f\n",(x - px) / (y - py) * (-py) + px,0.0);
}
