#include <cstdio>
#include <cstring>
#include <cmath>
#include <iostream>

using namespace std;

int main()
{
	int n;
	scanf("%d", &n);
	if (n < 5) return puts("No solution"), 0;
	long double len = 604, x = 0, y = 0, T, t, pi = 3.1415926535897932384626433832795;
	T = t = pi * 2 / n;
	for (int i = 1; i < n; i++) {
		printf("%.8f %.8f\n", double(x), double(y));
		if (i == n - 1) break;
		x += cos(t) * len; 
		y += sin(t) * len;
		len += 0.01;
		t += T;
	}
	printf("%.8f 0.00000000\n", double(x - y / tan(t)), 0);

	return 0;	
}
