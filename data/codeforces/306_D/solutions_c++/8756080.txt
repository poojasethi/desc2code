#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
#define pi M_PI
using namespace std;

int n,i;

double a[11111];

double x[11111];
double y[11111];

double px,py,px1,py1;
double jiao,ag,cg;

int main()
{/*
	freopen("306D.in","r",stdin);
	freopen("306D.out","w",stdout);
*/	
	scanf("%d",&n);
	
	if (n<5)
	{
		printf("No solution\n");
		return 0;
	}
	
	for (i=1;i<=n;i++)
		a[i]=500;
	
	jiao=pi*2.0/n;
	
	ag=2;
	cg=1/cos(jiao);
	
		a[1]+=1*ag,a[n  ]-=1*cg,a[2  ]-=1*cg;
	
	for (i=2;i<n;i++)
		a[i]+=i*ag,a[i-1]-=i*cg,a[i+1]-=i*cg;
	
		a[n]+=n*ag,a[n-1]-=n*cg,a[1  ]-=n*cg;
	
	x[0]=y[0]=500000;
	
	px=1, py=0;
	
	ag=cos(jiao);
	cg=sin(jiao);
	
	for (i=1;i<=n;i++)
	{
		x[i]=x[i-1]+a[i]*px;
		y[i]=y[i-1]+a[i]*py;
		
		px1=px*ag-py*cg;
		py1=py*ag+px*cg;
		
		px=px1; py=py1;
	}
	
	for (i=1;i<=n;i++)
		printf("%.3lf %.3lf\n",x[i],y[i]);
	
	return 0;
}
