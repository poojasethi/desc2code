#include <algorithm>
#include <cstring>

#include <cstdio>
#include <cmath>
#define db double
#define pi 3.1415926535898
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
    db x,y;
} O;
int i,j,k,n,m;
db angel,l,x,y,s;
inline db cross(ww A,ww B,ww C) {
    return (B.x-A.x)*(C.y-A.y)-(B.y-A.y)*(C.x-A.x);
}
inline void work(ww A,ww B,ww C,ww D) {
    db s1=cross(A,B,C),s2=cross(A,B,D);
    printf("%lf %lf\n",C.x+(D.x-C.x)/(s1-s2)*s1,C.y+(D.y-C.y)/(s1-s2)*s1);
}
int main() {
    scanf("%d",&n); angel=pi-pi*(n-2)/n;
    if (n<=4) {
        printf("No solution"); return 0;
    } 
    printf("-100 0\n");
    For(i,1,n-2) {
        if (i==n/2) l=1000;
        else l=1.*i/900+1;
        x=O.x+cos(s)*l; y=O.y+sin(s)*l;
        printf("%lf %lf\n",x,y);
        O.x=x,O.y=y;
        s+=angel;
    }
    ww A={-100,0},B={-100+cos(pi-angel),sin(pi-angel)},C=O,D={O.x+cos(s),O.y+sin(s)};
    work(A,B,C,D); return 0;
}
