#include<bits/stdc++.h>
using namespace std;
double cita,X[1005],Y[1005];int n;
struct node{
    double x,y,len;
    void Rotate(){
        double _x=x,_y=y;
        x=_x*cos(cita)-_y*sin(cita);
        y=_y*cos(cita)+_x*sin(cita);
        len=sqrt(x*x+y*y);
  }
};
int work(){
    double x=0,y=0,len=500;
    node v={1,0};
    for (int i=2;i<n;i++){
        v.Rotate();
        x+=len*v.x/v.len;
        y+=len*v.y/v.len;
        X[i]=x;Y[i]=y;len+=0.005;
  }if (x>0||y<0) return 0;
  v.Rotate();x-=y*v.x/v.y;
  if (x>=0) return 0;
  X[n]=x;Y[n]=0;
  for (int i=1;i<=n;i++)
    printf("%.10f %.10f\n",X[i],Y[i]);
  return 1;
}
int main(){
    scanf("%d",&n);srand(233);
    if (n<=4) return puts("No solution"),0;
    cita=2*acos(-1.)/n;
    work();
}
