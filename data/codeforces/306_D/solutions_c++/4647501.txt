#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <cmath>
#define db double
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
const db pi=acos(-1);
struct ww {
	db x,y;
} O;
int i,j,k,n,m;
db ang,jiao;
inline db cross(ww a,ww b,ww c) {
	return (b.x-a.x)*(c.y-a.y)-(b.y-a.y)*(c.x-a.x);
}
inline void work(ww a,ww b,ww c,ww d) {
	db A=cross(a,b,c),B=cross(a,b,d);
	printf("%.12lf %.12lf\n",c.x+(d.x-c.x)/(A-B)*A,c.y+(d.y-c.y)/(A-B)*A);
}
int main() {
	scanf("%d",&n);
	if (n<=4) printf("No solution\n"),exit(0);
	ang=pi-(n-2)*pi/n;
	printf("-100 0\n");
	For(i,1,n-2) {
		db l=i==n/2?1000:1.*i/900+1;
		db x=O.x+cos(jiao)*l,y=O.y+sin(jiao)*l;
		printf("%.12lf %.12lf\n",x,y);
		O.x=x,O.y=y;
		jiao+=ang;
	}
	ww A={-100,0},B={-100+cos(pi-ang),sin(pi-ang)},C=O,D={O.x+cos(jiao),O.y+sin(jiao)};
	work(A,B,C,D);
	//for(;;);
	return 0;
}
