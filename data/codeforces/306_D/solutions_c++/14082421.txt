#include<cmath>
#include<cstdio>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	if(n<5)
		return puts("No solution"),0;
	double arg=0,t=M_PI*2/n,L=100,X=-L,Y=0;
	for(int i=1;i<n;i++)
	{
		X+=cos(arg)*L,Y+=sin(arg)*L;
		arg+=t;
		printf("%lf %lf\n",X,Y);
		L+=0.002;
	}
	printf("%lf 0\n",X-Y/tan(arg));
}
