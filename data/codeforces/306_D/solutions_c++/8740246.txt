#include <cstdio>
#include <cmath>
using namespace std;
int n;
int main()
{
	int i; scanf("%d",&n);
	if (n<=4) printf("No solution\n");
	else
	{
		double a=2.0*M_PI/n,x=0,y=0;
		printf("0.00000 0.00000\n");
		for (i=1;i<=n-2;i++)
		{
			double l=600.0+0.01*i;
			x+=cos(a*i)*l; y+=sin(a*i)*l;
			printf("%.5lf %.5lf\n",x,y);
		}
		x-=y/sin(a*(n-1))*cos(a*(n-1)); y=0;
		printf("%.5lf %.5lf\n",x,y);
	}
	return 0;
}
