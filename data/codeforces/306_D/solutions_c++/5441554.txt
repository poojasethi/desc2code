#include<math.h>
#include<iostream>
#include<stdio.h>
using namespace std;
const long double PI=3.14159265;
const long double md=500;
long double ml;
int n;
long double a[300];
long double V[300][2];
void vectors()
{
     if (n>4)
     {
        float N;
        N=n;
        ml=(999-md)/N;
        int i;
        long double ii;
        long double j;
        j=0;
        a[0]=0;
        j=0;
        for (i=1;i<n;i++)
        {
            ii=i;
            a[i]=2*PI/N*ii;
            V[i][0]=V[i-1][0]+cos(a[i-1])*(md+(N+ii)/N/2*ml*j*((i%2)*2-1));
            V[i][1]=V[i-1][1]+sin(a[i-1])*(md+(N+ii)/N/2*ml*j*((i%2)*2-1));
            j++;
        }
        V[0][0]=V[n-1][0]-V[n-1][1]/tan(a[n-1]);
        V[0][1]=0;
        for(i=0;i<n;i++)
            cout<<V[i][0]<<" "<<V[i][1]<<endl;
     }
     else cout<<"No solution"<<endl;
}
main()
{
      cin>>n;
      vectors();
}
