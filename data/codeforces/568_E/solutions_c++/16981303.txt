#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>

#define N 100010

using namespace std;

int n,m,a[N],b[N],val[N],pre[N],fr[N],mc[N],pos[N],cnt[N],next[N],get[N],ans[N],ot[N],kt,top;
int main()
{
	scanf("%d",&n);
	int tt=0;
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		fr[i]=tt;
		if (a[i]==-1) 
			tt=i;
	}
	fr[n+1]=tt;
	a[0]=0,a[n+1]=1<<30;
	scanf("%d",&m);
	for (int i=1;i<=m;i++)
		scanf("%d",&b[i]);
	sort(b+1,b+m+1);
	val[++top]=-1;
	for (int i=1;i<=n+1;i++)
	{
		if (a[i]>-1)
		{
			int l=1,r=top;
			while (l<r)
			{
				int mid=(l+r+1)>>1;
				if (a[i]>val[mid])
					l=mid;
				else
					r=mid-1;
			}
			pre[i]=pos[l],mc[i]=cnt[l];
			val[l+1]=a[i],pos[l+1]=i,cnt[l+1]=0;
			if (l==top) top++;
		}
		else
		{
			int k=top;
			for (int j=m;j>=1;j--)
			{
				while (k && b[j]<=val[k]) k--;
				val[k+1]=b[j],pos[k+1]=pos[k],cnt[k+1]=cnt[k]+1;
				if (k==top) top++;
			}
		}
	}
	int nt=n+1,t=m;
	while (nt)
	{
		if (mc[nt])
		{
			while (b[t]>=a[nt]) t--;
			int ct=0;
			for (int i=t,k=fr[nt];ct<mc[nt];i--,k=fr[k])
			{
				while (b[i]==b[i+1]) i--;
				ct++;
				get[i]=1,a[k]=b[i];
			}
			t=t-mc[nt];
		}
		nt=pre[nt];
	}
	for (int i=1;i<=m;i++)
		if (!get[i])
			ans[++kt]=b[i];
	int he=1;
	for (int i=1;i<=n;i++)
		printf("%d ",(a[i]==-1)?ans[he++]:a[i]);
	return 0;
}
