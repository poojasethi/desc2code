#include<iostream>
#include<map>
#include<cstdio>
#include<set>
#include<algorithm>
using namespace std;
const int maxn=200000+10;
#define mp(x,y) make_pair((x),(y))
#define pb push(back)
#define fr first
#define se second
typedef long long ll;
multimap <int, pair<int,bool> > seg;
multimap <int, pair<int,bool> >::iterator it;
multiset <int> T;
int sz=1;
int L[maxn],R[maxn],sl[maxn],sr[maxn];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<m;i++)
	{
		int L,R,t;
		cin>>L>>R>>t;
		seg.insert(mp(L,mp(t,true)));
		seg.insert(mp(R,mp(t,false)));
	}
	int pre=-1;
	for(it =seg.begin();it!=seg.end();it++)
	{
		int cc=it->first;
		if(pre!=-1 && pre!=cc && T.size())
		{
			int t=*T.begin();
			L[sz]=t-pre;
			R[sz++]=t-cc;
		}
		pre=cc;
		//左边 
		if(it->se.se==true)T.insert(it->se.first);
		//右边 
		else	T.erase(T.lower_bound(it->se.first));
	}
	sort(L+1,L+sz);
	sort(R+1,R+sz);
	for(int i=1;i<sz;i++)
	{
		sl[i]=sl[i-1]+L[i];
		sr[i]=sr[i-1]+R[i];
	}
	int q;
	while(n--)
	{
		int ql,qr;
		cin>>q;
		ql=lower_bound(L+1,L+sz,q)-L-1;
		qr=lower_bound(R+1,R+sz,q)-R-1;
		int tot=(ll)(qr-ql)*q-(ll)sr[qr]+sl[ql];
		cout<<tot<<endl;
	}
	//system("pause");
}
