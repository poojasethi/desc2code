#include "bits/stdc++.h"
#define MAXN 600009
#define INF 1000000007
#define mp(x,y) make_pair(x,y)
#define all(v) v.begin(),v.end()
#define pb(x) push_back(x)
#define wr cout<<"----------------"<<endl;
#define ppb() pop_back()
#define tr(ii,c) for(typeof((c).begin()) ii=(c).begin();ii!=(c).end();ii++)
#define ff first
#define ss second
#define left cep
#define right sag

using namespace std;

typedef long long ll;
typedef pair<ll,int> PII;
template<class T>bool umin(T& a,T b){if(a>b){a=b;return 1;}return 0;}
template<class T>bool umax(T& a,T b){if(a<b){a=b;return 1;}return 0;}
//freopen("file.in","r",stdin);
//freopen("file.out","w",stdout);
class BigInteger{
	public:
	long double val;
	BigInteger(){val=0;}
}L[MAXN],R[MAXN],k;
queue<PII>q;
vector<ll>v,lef,righ;
ll mn=1e18;
int ok(int x){
	for(int i=0;i+x<=int(v.size());i++){
		ll bir=v[i],iki=v[i+x-1];
		if(iki-bir+1>mn)
			continue;
		int uc=lower_bound(all(lef),bir)-lef.begin();
		int dort=(upper_bound(all(righ),iki)-righ.begin())-1;
		BigInteger bas,alty;
		bas.val=(lef.size()-uc);alty.val=dort+1;
		bas.val*=bir;alty.val*=iki;
		if((L[uc].val-bas.val)+(alty.val-(dort>=0?R[dort].val:0))<=k.val)
			return 1;
	}
	return 0;
}
int main(){
	q.push(mp(4LL,1));q.push(mp(7LL,1));
	while(!q.empty()){
		ll node=q.front().ff;
		ll cost=q.front().ss;
		q.pop();v.pb(node);
		if(cost>=18)
			continue;
		q.push(mp(node*10+4,cost+1));	
		q.push(mp(node*10+7,cost+1));	
	}
	//printf("%d\n",int(v.size()));
	int n;
	scanf("%d",&n);
	cin>>k.val;
	for(int i=1;i<=n;i++){
		ll l,r;
		scanf("%lld%lld",&l,&r);
		umin(mn,r-l+1);
		lef.pb(l);righ.pb(r);
	}
	sort(all(lef));sort(all(righ));
	for(int i=int(lef.size()-1);i>=0;i--)
		L[i].val=L[i+1].val+lef[i];
	R[0].val=righ[0];	
	for(int i=1;i<int(righ.size());i++)
		R[i].val=R[i-1].val+righ[i];
	int st=1,en=(1<<19)-2;
	while(st+1<en){
		int mid=(st+en)>>1;
		if(ok(mid))
			st=mid;
		else
			en=mid;
	}		
	for(int i=en;i>=st;i--)
		if(ok(i))
			return printf("%d\n",i),0;
	printf("0\n");		
	return 0;
}
