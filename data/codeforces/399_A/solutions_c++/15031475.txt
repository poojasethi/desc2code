#include <iostream>
using namespace std;
int main()
{
	int n,p,k;
	cin>>n>>p>>k;
	if(p-k>1)cout<<"<<"<<" ";
	int be,en;
	be=p-k>1?p-k:1;
	en=p+k<n?p+k:n;
	for(int i=be;i<=en;i++)
	{
		if(i!=p) cout<<i<<" ";
		else cout<<"("<<p<<")"<<" ";
	}
	if(p+k<n)cout<<">>";
	return 0;
}
