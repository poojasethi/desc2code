#include<iostream>
using namespace std;
int main ()
{
  int p,c,r;  cin>>p>>c>>r;
  if((c-r)>1)cout<<"<< ";
   for(int i=(c-r);i<=(c+r);i++)
   {
     if(i<1)continue;
     if(i>p)break;
     if(i==c)cout<<"("<<i<<")";
     if(i!=c)cout<<i;
     if(i<(c+r))cout<<" ";
   }
  if((c+r)<p)cout<<" >>";
 return 0;
}
