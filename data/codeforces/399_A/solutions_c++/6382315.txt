#include <iostream>

using namespace std;

int n,p,k,a,b;

int main()
{
	cin >> n >> p >> k;
	
	a = max(p-k,1);
	b = min(k+p,n);
	
	if (a != 1) cout << "<< ";
	
	for (int i = a; i <= b; i++)
	{
		if (i == p)	
			cout << "(" << i << ") ";
		else
			cout << i << " ";
	}
	
	if (b != n)
		cout << ">>";
}
