#include <iostream>
using namespace std;
int main(){
	int n,p,k;
	cin>>n>>p>>k;
	if(p-k>1)
	cout<<"<< ";
	for(int i=max(p-k,1);i<=p+k && i<=n ;i++)
	{
		if(i==p)
		cout<<"("<<i<<") ";
		else
		cout<<i<<" ";

	}
	if(p+k<n)
	cout<<">> ";
	return 0;
}
