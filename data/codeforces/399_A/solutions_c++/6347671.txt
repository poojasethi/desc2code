#include <iostream>
using namespace std;

int n, p, k;

int main ()
{
	cin >> n >> p >> k;
	if(p - k > 1) cout << "<< ";
	for (int i = p-k; i<p; i++)
		if(i > 0) cout << i << " ";
	cout << "(" << p << ") ";
	for (int i = p+1; i<=n && i <= p+k; i++)
		cout << i << " ";
	if(p + k < n) cout << ">>\n";
return 0;
}
