#include <iostream>
using namespace std;

void print(int f, int t) {
	for (int i = f; i <= t; i++)
		cout << i << " ";
}

int main() {
	int n, p, k;
	cin >> n >> p >> k;
	if (p > k+1)
		cout << "<< ";
	print(max(1, p-k), p-1);
	cout << "(" << p << ") ";
	print(p+1, min(n, p+k));
	if (p+k < n)
		cout << ">>";
}
