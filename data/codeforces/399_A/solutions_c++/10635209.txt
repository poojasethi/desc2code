#include <bits/stdc++.h>
using namespace std;

int n,p,k,i;
int main()
{
    cin>>n>>p>>k;
    if(p-k>1) cout<<"<< ";
    for(i=p-k;i<p;i++)
    {
        if(i>=1) cout<<i<<" ";
    }
    cout<<"("<<p<<") ";
    for(i=p+1;i<=p+k&&i<=n;i++)
    {
        cout<<i<<" ";
    }
    if(p+k<n) cout<<">>"<<endl;
}
