#include<stdio.h>

int i,j,n,k,p;

int main(){

	scanf("%d %d %d",&n,&p,&k);

	if(p-k > 1)
		printf("<< ");

	for(i = p-k; i <= p-1; i++)
		if(i >= 1)
			printf("%d ",i);
	printf("(%d) ",p);
   for(i = p+1; i <= k+p; i++)
		if(i > n)
			break;
		else
			printf("%d ",i);

	if(k+p < n)
		printf(">>");

	return 0;
}
