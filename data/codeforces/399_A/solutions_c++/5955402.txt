#include<stdio.h>
#include<algorithm>
using namespace std;
int n,p,k;
int main()
{
	int i,j,l,r;
	scanf("%d%d%d",&n,&p,&k);
	l=max(1,p-k),r=min(n,p+k);
	if(l>1) printf("<< ");
	for(i=l;i<p;i++)
		printf("%d ",i);
	printf("(%d)",p);
	for(i=p+1;i<=r;i++)
		printf(" %d",i);
	if(r<n) printf(" >>");
	printf("\n");
	return 0;
}
