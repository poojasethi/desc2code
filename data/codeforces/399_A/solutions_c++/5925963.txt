#include<stdio.h>
#include<algorithm>

using namespace std;

int main() {
  int n,p,k;
  scanf("%d%d%d", &n,&p,&k);
  if (p-k>1) { printf("<< "); }
  for (int i=max(1,p-k); i<= min(n,p+k); i++) {
    if (i==p) { printf("(%d) ", p); }
    else { printf("%d ", i); }
  }
  if (p+k<n) { printf(">>"); }
}
