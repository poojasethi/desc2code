#include <cstdio>

using namespace std;

int main() {
	int n, p, k;
	scanf("%d%d%d", &n, &p, &k);
	if(p - k > 1) printf("<< ");
	for(int i = p - k; i <= p + k; i++) {
		if(i >= 1 && i <= n) {
			if(i == p) printf("(%d) ", i);
			else printf("%d ", i);
		}
	}
	if(p + k < n) printf(">>");
	puts("");
    return 0;
}
