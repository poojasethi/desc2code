#include <bits/stdc++.h>
using namespace std;
int n, p, k;
int main(){
	cin >> n >> p >> k;
	if(p - k > 1) printf("<<");
	for(int i = p - k; i <= p + k; i++){
		if(i < 1 || i > n) continue;
		if(i == p) printf(" (%d)", i);
		else printf(" %d",i);
	}
	if(p + k < n) printf(" >>");
	cout << endl;
	return 0;
}
