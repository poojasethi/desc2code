#include<iostream>
using namespace std;

int n, p, k;

int main(){
  cin >> n >> p >> k;
  if(p - k > 1)
    cout << "<< ";
  for(int i = max(1, p - k); i <= min(p + k, n); i++){
    if(i == p)
      cout << "(" << p << ") ";
    else 
      cout << i << ' ';
  }
  if(p + k < n)
    cout << ">>";
  return 0;
}
