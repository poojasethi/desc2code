#include <iostream>

using namespace std;

int n, m, k;

int main(){
	cin >> n >> m >> k;
	if(m - k > 1)
		cout << "<< ";
	for(int i = m - k; i <= m + k; i++){
		if(i == m)
			cout << "(" << m << ") "; 
		else if(i > 0 && i <= n)
			cout << i << " ";
	}
	if(m + k < n)
		cout << ">>";
	cout << endl;
	return 0;
}
