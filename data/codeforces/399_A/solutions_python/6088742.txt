[n,p,k] = [int(x) for x in raw_input().split()]

pages = range(1,n+1)
fwd,rev = True, True
if p+k >= n:
    fwd = False
if p-k <= 1:
    rev = False
stri = ""
if rev == True:
    stri = stri+"<< "
for i in range(k):
    if p-(k-i)<1:
        continue
    else:
        stri = stri+str(p-(k-i))+" "
stri = stri+"("+str(p)+")"
for i in range(k):
    if p+i+1 >n:
        continue
    else:
        stri = stri+" "+str(p+i+1)
if fwd == True:
    stri=stri+" >>"
print stri
