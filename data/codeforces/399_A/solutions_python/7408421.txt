n, p, k = map(int, raw_input().split())
b = False
t = False
a = ""
if p-k <= 1:
    b = True
if p+k >= n:
    t = True
if b:
    for i in range(1, p):
        a += str(i) + " "
    a += "(" + str(p) + ") "
else:
    a += "<< "
    for i in range(p-k, p):
        a += str(i) + " "
    a += "(" + str(p) + ") "
if t:
    for i in range(p+1, n+1):
        a += str(i) + " "
else:
    for i in range(p+1, p+k+1):
        a += str(i) + " "
    a += ">>"
print a.strip()
