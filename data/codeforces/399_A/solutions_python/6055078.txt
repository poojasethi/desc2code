n,m,k = raw_input().split(' ')
n=int(n);m=int(m);k=int(k)
ans=[]
ans.append(m)

for x in xrange(1,k+1):
    if(m+x<=n):
        ans.append(m+x)
    if(m-x>=1):
        ans.insert(0,m-x)
string=""
if ans[0]!=1:
    string+="<< ";

for x in ans:
    if(x==m):
        string+="("+str(x)+") "
    else:
        string +=str(x)+" ";

if(ans[-1]!=n):
    string +=">>"

print string
