n,k,p=map(int,raw_input().split())
l=k-p
r=k+p
if l<=0:
    l=1
if n<r:
    r=n
ans=[]
for i in range(l,r+1):
    if i==l and i!=1:
        ans.append('<<')
        ans.append(str(i))
    elif i==r and i!=n:
        ans.append(str(i))
        ans.append('>>')
    elif i==k:
        ans.append('('+str(i)+')')
    else:
        ans.append(str(i))
print ' '.join(ans)
