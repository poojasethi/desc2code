lis=raw_input().split()
for i in range(len(lis)):
    lis[i]=int(lis[i])
n=lis[0]
p=lis[1]
k=lis[2]
st=''
if p-k>1:
    st=st+'<< '
c=k   
while k>=0:
    if k==0:
        st=st+'(%d) '%p
        break
    if p-k>0:
        st=st+'%d '%(p-k)        
    k=k-1
f=p+1
while 1>0:
    if f<=n:
        if (f)<=p+c:
            st=st+'%d '%(f)
            f=f+1
        else:
            st=st+'>>'
            break
    else:
        break
print st
