inp = raw_input().split(" ")
n = int(inp[0])
p = int(inp[1])
k = int(inp[2])

beg = p-k
end = p+k
beg_brac = True
end_brac = True

if beg <= 1:
    beg = 1
    beg_brac = False

if end >= n:
    end = n
    end_brac = False

l = range(beg,end+1)

if beg_brac:
    print "<<",

for i in l:
    if i == p:
        print "(" + str(i) + ")",
    else:
        print str(i),

if end_brac:
    print ">>"
