#!/usr/bin/env python

import sys
import logging
import traceback

logging.basicConfig(level = logging.INFO)

n, p, k = map(int, raw_input().split())

l = max(1, p-k)
r = min(n, p+k)

formatStr = ''
if l > 1 : 
	formatStr += '<< '
for i in range(l, r+1) : 
	if i != l :
		formatStr += ' '
	if i == p : 
		formatStr += '(%d)' %(i)
	else : 
		formatStr += '%d' % i

if r < n : 
	formatStr += ' >>'

print formatStr
