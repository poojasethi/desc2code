def main():
    n,p,k = map(int, raw_input().split())
    
    if p-k > 1:
        print "<<",
        
    i = max(1,p-k)
    while i <= n and i <= p+k:
        if i != p:
            print i,
        else:
            print "("+str(i)+")",
        i+=1
        
    if p+k < n:
        print ">>"
        
if __name__ == "__main__":
    main()
