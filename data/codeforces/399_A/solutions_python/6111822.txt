import sys

def main():
  npk = [int(x) for x in sys.stdin.readline().split()]
  n = npk[0]
  p = npk[1]
  k = npk[2]

  s = []
  if p-k > 1:
    s.append("<<")
  for x in xrange(p-k,p+k+1):
    if x == p:
      y = "("+str(x)+")"
      s.append(y)
    elif x >= 1 and x <= n:
      s.append(str(x))
  if p+k < n:
    s.append(">>")

  print " ".join(s)

main()
