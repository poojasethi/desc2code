a,b,c = map(int,raw_input().split(' '))

l = "("+str(b)+")"
if b+c>a:
    y = a+1
else:
    y = b+c+1
for i in xrange(b+1,y):
    l += " " + str(i)
if b-c-1<1:
    x = 1
else:
    x = b-c
for i in reversed(xrange(x,b)):
    v = str(i)+" "
    l = v+l
if x!=1:
    l = "<< "+l
if y-1!=a:
    l += " >>"
print l
