#define f first

typedef pair <int, int> pi;
typedef vector <int> vi;
typedef long long ll;

int n, m;
int f[33333];
pi a[33333];

int main() {
  scanf("%d%d", &n, &m);
  for (int i = 0; i < n; i++) {
    scanf("%d", &a[i].f);
    a[i].s = i;
  }
  sort(a, a + n);
  f[0] = f[1] = 1 % m;
  for (int i = 2; i <= n; i++) {
    f[i] = f[i - 1] + f[i - 2];
    if (f[i] >= m) {
      f[i] -= m;
    }
  }

  int l, r, q;
  scanf("%d", &q);
  while (q--) {
    scanf("%d%d", &l, &r);
    --l, --r;
    ll sum = 0, color = -1, i = 0;
    for (int j = 0; j < n; j++) {
      if (color == a[j].f) continue;
      if (l <= a[j].s && a[j].s <= r) {
        color = a[j].f;
        sum += ((ll)f[i++]) * a[j].f;
      }
    }
    printf("%d\n", int(sum % m));
  } 
  return 0;
}
