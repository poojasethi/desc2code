#include <iostream>
#include <cmath>

using namespace std;

typedef long long int ll;

ll pasos(ll i,ll mi,ll n)
{
  if (mi==0) return 1000000000000LL;
  if (mi>0)
    return (n-i)/mi;
  return (i-1)/(-mi);
}

int main()
{
  ll n,m;
  cin>>n>>m;
  ll i,j;
  cin>>i>>j;
  ll k;
  cin>>k;
  ll amount=0;
  for (int t=0;t<k;t++) {
    ll mi,mj;
    cin>>mi>>mj;
    ll pi=pasos(i,mi,n);
    ll pj=pasos(j,mj,m);
    ll p=min(pi,pj);
    amount+=p;
    i+=mi*p;
    j+=mj*p;
  }
  cout<<amount<<endl;
}
