#include <iostream>
#include <algorithm>
using namespace std;
long long step(long long n, long long x, long long dx){
    if(dx>0) return (n-x)/dx;
    if(dx<0) return (1-x)/dx;
    return (1<<30);
}
int main(){
    long long n,m,x,y,k;
    cin>>n>>m>>x>>y>>k;
    long long ans=0;
    while(k--){
        long long dx,dy,move;
        cin>>dx>>dy;
        move=min(step(n,x,dx),step(m,y,dy));
        x+=move*dx;
        y+=move*dy;
        ans+=move;
    }
    cout<<ans;
}
