#include <bits/stdc++.h>
using namespace std;
int lx=1,ly=1,rx,ry,x,y,k,vx,vy;
long long ans;
int main() 
{
	cin>>rx>>ry>>x>>y;
	cin>>k;
	while(k--)
	{
		cin>>vx>>vy;
		int yy=INT_MAX,xx=INT_MAX;
		if(vy>0)yy=(ry-y)/vy;else if(vy<0) yy=(ly-y)/vy;
		if(vx>0)xx=(rx-x)/vx;else if(vx<0) xx=(lx-x)/vx;
		yy=min(yy,xx);
		ans+=yy;
		x+=vx*yy;
		y+=vy*yy;
	}
	cout<<ans<<endl;
	return 0;
}
