#include<stdio.h>

main(){
	int i,j,k;
	int n,m;
	int x,y;
	int nx,ny;
	int dx,dy;
	int a,b,c;
	long long cnt;
	while(scanf("%d%d",&x,&y)==2){
		scanf("%d%d",&nx,&ny);
		scanf("%d",&n);
		cnt=0;
		for(i=0;i<n;i++){
			scanf("%d%d",&dx,&dy);
			if(dx>0)a=(x-nx)/dx;else if(dx<0)a=(nx-1)/-dx; else a=1000000001;
			if(dy>0)b=(y-ny)/dy;else if(dy<0)b=(ny-1)/-dy; else b=1000000001;
			c=a>b?b:a;
			cnt+=c;
			nx+=dx*c;
			ny+=dy*c;
			//printf("[%d %d]\n",a,b);
		}
		printf("%I64d\n",cnt);
	}
		
}
