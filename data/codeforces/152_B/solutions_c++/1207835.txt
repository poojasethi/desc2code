#include <stdio.h>
#include <algorithm>
#define INF 2147483647

int main()
{
  int n,m,k,i,x,y,dx,dy,step;
  long long int ans=0;
  scanf("%d %d %d %d %d",&n,&m,&x,&y,&k);
  while(k--)
  {
    scanf("%d %d",&dx,&dy);
    step=INF;
    if (dx>0)step=(n-x)/dx;
    else if (dx<0)step=(x-1)/abs(dx);
    if (dy>0)step=std::min((m-y)/dy,step);
    else if (dy<0)step=std::min((y-1)/abs(dy),step);
    ans+=step;
    x+=dx*step;
    y+=dy*step;
  }
  printf("%I64d\n",ans);
}
