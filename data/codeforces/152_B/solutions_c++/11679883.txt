#include<bits/stdc++.h>
using namespace std;
int n,m,k,x,y,a,b,p;
long long v=0;
main()
{
    scanf("%d%d%d%d%d",&n,&m,&x,&y,&k);
    while (k--&&x>=1&&x<=n&&y>=1&&y<=m)
    {
        scanf("%d%d",&a,&b);
        if (!a) p=(b>0)?(m-y)/b:(y-1)/(-b);
        else if (!b) p=(a>0)?(n-x)/a:(x-1)/(-a);
        else p=min(((b>0)?(m-y)/b:(y-1)/(-b)),((a>0)?(n-x)/a:(x-1)/(-a)));
        x+=p*a,y+=p*b;
        v+=p;
    }
    printf("%I64d",v);
}
