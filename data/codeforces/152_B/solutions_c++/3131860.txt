#include<cstdio>
#include<algorithm>
int a,b,x,y,k,h,t,p,n,m;
long long s;
main(){
	scanf("%d%d%d%d%d",&n,&m,&x,&y,&k);
	for(;k--;){
		scanf("%d%d",&a,&b);
		if(!a)p=(b>0)?(m-y)/b:(y-1)/-b;
		else if(!b)p=(a>0)?(n-x)/a:(x-1)/-a;
			 else p=std::min(((b>0)?(m-y)/b:(y-1)/-b),((a>0)?(n-x)/a:(x-1)/-a));
		x+=p*a;y+=p*b;s+=p;
	}
	printf("%I64d",s);
}
