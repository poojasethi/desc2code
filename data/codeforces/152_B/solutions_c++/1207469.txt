#include <iostream>
#include <string>

using namespace std;

long long xc,yc,guf,n,m,L,R,C,vx,vy,res=0;

int main()
{
 cin>>n>>m;
 cin>>xc>>yc;
 cin>>guf;
 for(int i=0;i<guf;++i)
 {
   cin>>vx>>vy;
   L=0;
   R=2000000000;
   while (L+1<R)
    {
          C=(L+R)/2;
          if (vx*C+xc<=n && vx*C+xc>0 && vy*C+yc<=m && vy*C+yc>0) L=C;
           else R=C;
    }
   res=res+L;
   xc=xc+vx*L;
   yc=yc+vy*L;
 }   
 cout<<res<<endl;
}
