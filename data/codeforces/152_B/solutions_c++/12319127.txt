#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    long long n,m,k,xc,yc,a,b,ans=0;
    cin>>n>>m>>xc>>yc>>k;
    for(int i=0;i<k;i++){
        cin>>a>>b;
        int nx,ny,nt;
        if(a<0) nx=(1-xc)/a;
        else if(a>0) nx=(n-xc)/a;
        if(b<0) ny=(1-yc)/b;
        else if(b>0) ny=(m-yc)/b;
        if(a==0) nt=ny;
        else if(b==0)   nt=nx;
        else nt=min(nx,ny);
        ans+=nt;
        xc+=nt*a; yc+=nt*b;
    }
    cout<<ans;
    return 0;
}
