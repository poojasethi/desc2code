#include<cstdio>
#define ll __int64
const ll INF=10000000000000;
ll min(ll a,ll b){return a<b?a:b;}
int main(){
	ll n,m,xc,yc,k,a,b,s1,s2,s;
	while(~scanf("%I64d%I64d",&n,&m)){
		scanf("%I64d%I64d%I64d",&xc,&yc,&k);
		ll ans=0;
		while(k--){
			scanf("%I64d%I64d",&a,&b);
			if(a<0)s1=(xc-1)/(-a);
			else if(a>0)s1=(n-xc)/a;
			else s1=INF;
			if(b<0)s2=(yc-1)/(-b);
			else if(b>0)s2=(m-yc)/b;
			else s2=INF;
			s=min(s1,s2);
			xc+=a*s,yc+=b*s;
			ans+=s;
		}
		printf("%I64d\n",ans);
	}
	return 0;
}
