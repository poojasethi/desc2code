#include <iostream>
using namespace std;

int main()
{
	int N,M;
	int x,y;
	cin>>N>>M;
	cin>>x>>y;
	int k;
	cin>>k;
	long long ans=0;
	while(k--)
	{
		int vx,vy;
		cin>>vx>>vy;
		int stx=1<<30,sty=1<<30;
		if(vx>0)
			stx=(N-x)/vx;
		else if(vx<0)
			stx=(x-1)/(-1*vx);

		if(vy>0)
			sty=(M-y)/vy;
		else if(vy<0)
			sty=(y-1)/(-1*vy);
		int st=stx<sty?stx:sty;
		x+=vx*st;
		y+=vy*st;
		ans+=st;
	}
	cout<<ans<<endl;
	return 0;
}
