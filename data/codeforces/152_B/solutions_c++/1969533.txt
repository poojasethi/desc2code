#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>

using namespace std;

int main()
{
    long long m,n,x,y,o,p,q,k,l,res=0;
    cin>>m>>n>>x>>y>>o;
    while (o--)
    {
        cin>>p>>q;
        if (p>0) k=(m-x)/p; else if (p<0) k=(1-x)/p; else k=n;
        if (q>0) l=(n-y)/q; else if (q<0) l=(1-y)/q; else l=m;
        k=min(k,l); res+=k;
        x=x+k*p; y=y+k*q;
    }
    cout<<res<<endl;
}
