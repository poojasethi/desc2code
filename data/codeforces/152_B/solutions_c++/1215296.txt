#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std ;
typedef long long LL ;
const int N = 10010, INF = 2000000000 ;
LL n, m, k ;
int main (){
    LL res=0,a,b;
    cin>>n>>m>>a>>b>>k ;
    while(k--){
        LL x,y,s=INF ;
        cin>>x>>y ;
        if(x>0)s=min(s,(n-a)/x) ;
        if(x<0)s=min(s,(a-1)/-x) ;
        if(y>0)s=min(s,(m-b)/y) ;
        if(y<0)s=min(s,(b-1)/-y) ;
        res+=s,a+=x*s,b+=y*s ;
    }
    cout << res << endl ;
    return 0 ;
}
