#include<cstdio>
#define ll __int64
int main()
{
	int i,j,k,n,m,s,e,x,y,a,b;
	ll ans=0;
	scanf("%d%d",&n,&m);
	scanf("%d%d",&s,&e);
	scanf("%d",&k);
	for(i=0;i<k;i++)
	{
		scanf("%d%d",&x,&y);
		a=b=0x7f7f7f7f;
		if(x>0)
			a=(n-s)/x;
		else if(x<0)
			a=(s-1)/(-x);
		if(y>0)
			b=(m-e)/y;
		else if(y<0)
			b=(e-1)/(-y);
		if(a<b)
			ans+=a,s+=a*x,e+=a*y;
		else
			ans+=b,s+=b*x,e+=b*y;
	}
	printf("%I64d\n",ans);
	return 0;
}
