#include <iostream>
#include <algorithm>
using namespace std;
int main(){
	int n,m,a,b;int px,py;
	cin>>n>>m>>px>>py;
	int k;cin>>k;
	long long int ret= 0;
	for(int i=0;i<k;++i){
		cin>>a>>b;
		int step=1e9;
		if(a!=0)
			if(a>0){
				step = min(step,(n-px)/a);
			}
			else{
				step =min(step, (1-px)/a);
			}
		if(b!=0)
			if(b>0)
				step = min(step,(m-py)/b);
			else
				step = min(step,(1-py)/b);
		ret+=step;
		px += step*a,py += step*b;
	}
	cout<<ret<<endl;
}
