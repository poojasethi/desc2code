#include <math.h>
#include <stdio.h>
#include <stdlib.h>

__int64 n,m,k,s,cx,cy;

int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  scanf("%I64d%I64d%I64d%I64d%I64d",&n,&m,&cx,&cy,&k);
  for(int i=0; i<k; i++){
		int dx,dy,k1=-1,k2=-1;
		scanf("%i%i",&dx,&dy);
		if(dx<0)k1=(1-cx)/dx; else
		if(dx>0)k1=(n-cx)/dx;
		if(dy<0)k2=(1-cy)/dy; else
		if(dy>0)k2=(m-cy)/dy;
		if((k2<k1&&k2!=-1)||k1==-1) k1=k2;
		s+=k1;
		cx+=k1*dx;
		cy+=k1*dy;
	} 
	printf("%I64d",s);
}
