#include<stdio.h>
#include<string.h>
int n,m;
int nx,my;
int main(){
	scanf("%d%d",&n,&m);
	scanf("%d%d",&nx,&my);
	int N;
	scanf("%d",&N);
	__int64 num=0;
	for(int K=1;K<=N;K++){
		int x,y;
		scanf("%d%d",&x,&y);
		int p,q;
		if(x<0){
			p=(nx-1)/(-x);
		}
		else if(x>0){
			p=(n-nx)/x;
		}
		else p=1000000001;
		if(y<0){
			q=(my-1)/(-y);
		}
		else if(y>0){
			q=(m-my)/y;
		}
		else q=1000000001;
		int mi=q<p?q:p;
		num+=mi;
		nx+=x*mi;
		my+=y*mi;
	}
	printf("%I64d\n",num);
}
