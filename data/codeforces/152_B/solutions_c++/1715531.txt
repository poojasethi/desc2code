#include<iostream>
#include<cmath>
#define inf 1000000000
using namespace std;
int main()
{
int k,i;
long long n,m,x,y,dx,dy,o=0,z1,z2,z;
cin>>n>>m>>x>>y>>k;
for(i=1;i<=k;i++)
{
cin>>dx>>dy;
if(dx<0) z1=(1-x)/dx;
else if(dx>0) z1=(n-x)/dx;
else z1=inf;
if(dy<0) z2=(1-y)/dy;
else if(dy>0) z2=(m-y)/dy;
else z2=inf;
z=min(z1,z2);
o+=z;
x=x+z*dx;
y=y+z*dy;
}
cout<<o;
return 0;
}
