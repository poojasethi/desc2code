#include <iostream>
#include <algorithm>

using namespace std;

int N,M,Xc,Yc,K;
long long result;

int main()
{
	cin >> N >> M >> Xc >> Yc >> K;
	for (int i=0;i<K;i++)
	{
		int dx,dy,step_x=1000000000,step_y=1000000000;
		cin >> dx >> dy;
		if (dx>0)
			step_x=(N-Xc)/dx;
		else if (dx<0)
			step_x=-(Xc-1)/dx;
		if (dy>0)
			step_y=(M-Yc)/dy;
		else if (dy<0)
			step_y=-(Yc-1)/dy;
		result+=min(step_x,step_y);
		Xc+=min(step_x,step_y)*dx;
		Yc+=min(step_x,step_y)*dy;
	}
	cout << result << endl;
	return 0;
}
