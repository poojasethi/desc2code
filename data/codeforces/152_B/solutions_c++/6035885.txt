#include<iostream>
using namespace std;
int n,m,xc,yc;
int main()
{
	cin>>n>>m;
	cin>>xc>>yc;
	int k;
	cin>>k;
	long long i=0;
	while(k--)
	{
		int dx,dy;
		cin>>dx>>dy;
		int tx,ty;
		tx=1<<30;
		ty=1<<30;
		if(dx>0)
		tx=(n-xc)/dx;
		else 
		if(dx<0)
		tx=(xc-1)/(-1*dx);
		
		if(dy>0)
		ty=(m-yc)/dy;
		else 
		if(dy<0)
		ty=(yc-1)/(-1*dy);
		
		int j=tx<ty?tx:ty;
		xc+=dx*j;
		yc+=dy*j;
		i+=j;
	}
	cout<<i<<endl;
	return 0;
}
