# include<iostream>
# include<cstdio>
# include<algorithm>
using namespace std;
int main()
{
   long long p,n,m,x,y,k,a,b,sum=0,k1,k2;
   cin>>n>>m>>x>>y>>k;
   while(k--)
   {
      cin>>a>>b;
      if(a>0)
	 k1=(n-x)/a;
      else if(a<0)
	 k1=(1-x)/a;
      else
	 k1=m;
      if(b>0)
	 k2=(m-y)/b;
      else if(b<0)
	 k2=(1-y)/b;
      else
	 k2=n;
      k1=min(k1,k2);
      sum+=k1;
      x+=k1*a;
      y+=k1*b;
   }
   cout<<sum<<endl;
}
