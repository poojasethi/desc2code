#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int n, m, x, y, k,a,b,p;
    long long s=0;
    cin >> n >> m >> x >> y >> k;
    while (k--){
        cin >> a >> b;
        if (!a)p = (b>0) ? (m - y) / b : (y - 1) / -b;
        else if (!b)p = (a>0) ? (n - x) / a : (x - 1) / -a;
        else p = std::min(((b>0) ? (m - y) / b : (y - 1) / -b), ((a>0) ? (n - x) / a : (x - 1) / -a));
        x += p*a; y += p*b; s += p;
    }
    cout << s;
}
