#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    long long n,m,i,j,t=0,x,y,k;
    long long ans=0;
    cin>>n>>m>>x>>y>>k;
    for ( i=0;i<k;i++) {
        long long  xx,yy,l,r;
        cin>>xx>>yy;
        l=0;r=max(n,m);
        while (l+1!=r) {
              int w=(l+r)/2;
              if (x+w*xx>=1 && x+w*xx<=n &&
                 y+w*yy>=1 && y+w*yy<=m) l=w; else r=w;
              }
        x+=l*xx;
        y+=l*yy;
        ans+=l;
        }
   cout<<ans;
  //system("pause");
}
