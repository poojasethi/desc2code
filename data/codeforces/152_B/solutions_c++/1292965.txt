#include <algorithm>
#include <iostream>

using namespace std;

int main() {
	int n,m,xc,yc,k;
	cin >> n >> m >> xc >> yc >> k;
	long long res=0;
	for(int i=0;i<k;i++) {
		int dx,dy;
		cin >> dx >> dy;
		int kx = dx == 0 ? 1000000100 : dx<0 ? (1-xc)/dx : (n-xc)/dx;
		int ky = dy == 0 ? 1000000100 : dy<0 ? (1-yc)/dy : (m-yc)/dy;
		int t=min(kx,ky);
		xc+=t*dx;
		yc+=t*dy;
		res+=t;
	}
	cout << res << endl;
	return 0;
}
