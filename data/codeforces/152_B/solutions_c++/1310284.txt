#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
#define INF 2000000000
int main()
{
    long long  res=0, a, b, n, m, k;
    cin >> n >> m >> a >> b >> k;
    while (k--)
    {
        long long x, y, s=INF;
        cin >> x >> y ;
        if (x>0) s=min(s, (n - a)/x) ;
        if (x<0) s=min(s, (a - 1)/ -x) ;
        if (y>0) s=min(s, (m - b)/y) ;
        if (y<0) s=min(s, (b - 1)/ -y) ;
        res+=s; a+=x*s; b+=y*s;
    }
    cout << res << endl ;
    return 0;
}
