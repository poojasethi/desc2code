#include <bits/stdc++.h>
using namespace std;

#define ll long long
const int INF=1e9+5;
int main(){
	ll x,y,n,m;
	scanf("%I64d%I64d",&n,&m);
	scanf("%I64d%I64d",&x,&y);
	ll k,ans=0;
	scanf("%I64d",&k);
		while (k--){
			ll dx,dy,sx,zy;
			scanf("%I64d%I64d",&dx,&dy);
			if (dx==0) sx=INF;
			else if (dx>0) sx=(n-x)/dx;
			else sx=(1-x)/dx;
			if (dy==0) zy=INF;
			else if (dy>0) zy=(m-y)/dy;
			else zy=(1-y)/dy;
			ans+=min(zy,sx);
			x+=min(zy,sx)*dx;
			y+=min(zy,sx)*dy;
		}
	printf("%I64d\n",ans);
}
