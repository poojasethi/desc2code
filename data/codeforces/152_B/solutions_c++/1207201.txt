#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int R, C, x, y, n, stp, dx, dy;
long long ret;

int f(int x, int d, int b){
	if (d == 0) return (1<<30);
	if (d < 0) return f(1, -d, x);
	return (b - x) / d;
}

int main(){
	scanf("%d%d%d%d%d", &R, &C, &x, &y, &n);
	ret = 0;
	for (int i=0; i<n; i++){
		scanf("%d%d", &dx, &dy);
		stp = min(f(x, dx, R), f(y, dy, C));
		x += dx * stp;
		y += dy * stp;
		ret += stp;
	}
	
	printf("%I64d\n", ret);
	return 0;
}
