#include <iostream>
using namespace std;
typedef long long ll;
ll n,m,x,y;
bool valid(ll cx, ll cy){
	return (1<=cx&&1<=cy&&n>=cx&&m>=cy);
}
ll solve(ll dx, ll dy){
	ll lo = 0, hi = 1e9, mid;
	while(lo<hi){
		mid = (lo+hi+1)/2;
		if(valid(x+mid*dx,y+mid*dy))
			lo=mid;
		else
			hi=mid-1;
	}
	x+=lo*dx;
	y+=lo*dy;
	return lo;
}
int main(){
	cin >> n >> m >> x >> y;
	ll k;
	cin >> k;
	ll ans = 0;
	for(ll i = 0; i < k; ++i){
		ll dx,dy;
		cin >> dx >> dy;
		ans+=solve(dx,dy);
	}
	cout << ans << '\n';
}
