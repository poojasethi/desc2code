#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

typedef long long LL;
int main(void) {
	LL n, m, xc, yc;
	scanf("%I64d%I64d", &n, &m);
	scanf("%I64d%I64d", &xc, &yc);
	int k;
	scanf("%d", &k);
	LL ans=0;
	while(k--) {
		LL x, y;
		scanf("%I64d%I64d", &x, &y);
		LL r=min(x>0?(n-xc)/x: x==0?(1LL<<60): (xc-1)/-x,
				 y>0?(m-yc)/y: y==0?(1LL<<60): (yc-1)/-y);
		ans += r;
		xc += r*x;
		yc += r*y;
	}
	printf("%I64d\n", ans);
	return 0;
}
