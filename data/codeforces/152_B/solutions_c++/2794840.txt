#include<iostream>
#define MAX_INT 2000000000
using namespace std;
int main(){
	int n,m;
	cin>>n>>m;
	int xc,yc;
	cin>>xc>>yc;
	int k;
	cin>>k;
	long long steps=0;
	for(int i=0;i<k;i++){
		int dx,dy;
		cin>>dx>>dy;
		int mx,my;
		if(dx<0)
			mx=(xc-1)/(-dx);
		else if(dx>0)
			mx=(n-xc)/dx;
		else
			mx=MAX_INT;

		if(dy<0)
			my=(yc-1)/(-dy);
		else if(dy>0)
			my=(m-yc)/dy;
		else
			my=MAX_INT;
		int s=mx>my?my:mx;
		xc=xc+s*dx;
		yc=yc+s*dy;
		steps+=s;
	}
	cout<<steps;

	return 0;
}
