#include<iostream>
using namespace std;
int main()
{
  __int64 dx,dy,x,y,n,m,k,i,j,s=0,k1,k2,t;
  cin>>n>>m>>x>>y;
  cin>>k;
  for(i=0;i<k;i++)
  {
    cin>>dx>>dy;
    if (dx<0) k1=x-1; else k1=n-x;
    if (dy<0) k2=y-1; else k2=m-y;

    if (dx!=0)k1/=dx; 
    if (dy!=0)k2/=dy; 
    
    if (k1<0)k1=-k1; if (k2<0)k2=-k2;
    t=min(k1,k2);  
    if (dx==0) t=k2; else if (dy==0) t=k1;
    s+=t; 
    x+=t*dx; y+=t*dy;
  }
  cout<<s<<endl;
//  system("pause");
  return 0;
}
