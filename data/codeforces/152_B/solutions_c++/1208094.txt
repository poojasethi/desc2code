#include<cstdio>
long long x,y,vx,vy,ac=0,n,m,k,s,t,ss;
int main(){
	scanf("%I64d%I64d",&n,&m);
	scanf("%I64d%I64d",&x,&y);
	scanf("%I64d",&k);
	while (k--){
		  scanf("%I64d%I64d",&vx,&vy);
		  s=1<<30;
		  if (vx>0) s=(n-x)/vx;
		  else if (vx<0) s=-((x-1)/vx);
		  t=1<<30;
		  if (vy>0) t=(m-y)/vy;
		  else if (vy<0) t=-((y-1)/vy);
		  //printf("OK\n"); scanf("\n");
		  ss=s<t?s:t;
		  ac+=ss;
		  x=x+vx*ss;
		  y=y+vy*ss;
	}
	printf("%I64d\n",ac);
	return 0;
}
