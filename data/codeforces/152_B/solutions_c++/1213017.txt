#include<cstdio>
#include<iostream>
using namespace std;
long long n,m,x,y,dx,dy;

bool pred(int mi){
	long long x1=x+mi*dx, y1=y+mi*dy;
	if(x1>=1 && x1<=n && y1>=1 && y1<=m)return 0;
	return 1;
}

int binarySearch(){
	int lo=0 ,hi=1000000000;
	while(hi-lo>1){
		int mi=(lo+hi)/2;
		if(pred(mi)){
			hi=mi;
		}
		else lo=mi;
	}
	return lo;
}

int main(){
	int k;
	cin>>n>>m;
	cin>>x>>y;
	cin>>k;
	long long ct=0;
	for(int i=0;i<k;++i){
		cin>>dx>>dy;
		int paso=binarySearch();
		ct+=paso;
		x=x+paso*dx,y=y+paso*dy;
	}
	cout<<ct<<endl;
}
