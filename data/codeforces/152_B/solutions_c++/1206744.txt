#include <iostream>
using namespace std;

int main() {
	int n, m;
	int xc, yc;
	int k;
	int dx, dy;
	int tx, ty, t;
	long long int r = 0;

	cin >> n >> m;
	cin >> xc >> yc;
	cin >> k;
	--xc;
	--yc;
	for (int i = 0; i < k; ++i) {
		cin >> dx >> dy;
		if (dx < 0)
			tx = xc/(-dx);
		else if (dx > 0)
			tx = (n - 1 - xc)/dx;
		else
			tx = 1000000000;
		if (dy < 0)
			ty = yc/(-dy);
		else if (dy > 0)
			ty = (m - 1 - yc)/dy;
		else
			ty = 1000000000;
		t = min(tx, ty);
		xc += t*dx;
		yc += t*dy;
		r += t;
	}
	cout << r;
	return 0;
}
