#include<cmath>
#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
long long int n ,m ,k,x_i,y_i,x,y ,steps=0 ; cin>>n>>m;
cin>>x_i>>y_i>>k;
while(k>0)
{
cin>>x>>y;
for(long long int i = 10000000000;i>0;i/=2)
{while((x_i+i*x )>0 && (x_i+i*x) <=n&&(y_i+i*y )>0 &&(y_i+i*y<=m))
{x_i+=i*x; y_i+=i*y; steps+=i;}
}
k--;
}
cout<<steps;
}
