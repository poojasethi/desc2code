#include <bits/stdc++.h>

using namespace std;

const int inf = 1000000005;

int n, m, x, y, k, vx, vy;
long long ans;

int main() {
  cin >> n >> m >> x >> y >> k;
  while (k--) {
    cin >> vx >> vy;
    int dx, dy;
    if (vx == 0) dx = inf; else if (vx < 0) dx = (1 - x) / vx; else dx = (n - x) / vx;
    if (vy == 0) dy = inf; else if (vy < 0) dy = (1 - y) / vy; else dy = (m - y) / vy;
    int d = max(0, min(dx, dy));
    ans += d;
    x += d * vx;
    y += d * vy;
  }
  cout << ans;
}
