I=lambda:map(int, raw_input().split())
n,m = I()
x,y = I()
ans = 0
for _ in xrange(input()):
    dx, dy = I()
    a, b = 999999999999,999999999999
    if dx > 0: a = (n - x) / dx;
    elif dx < 0: a = (x - 1) / (-dx)
    if dy > 0: b = (m - y) / dy;
    elif dy < 0: b = (y - 1) / (-dy)
    c = min(a,b)
    x, y = x + c * dx, y + c * dy
    ans += c
print ans
