n, m = map(int, raw_input().split())
xc, yc = map(int, raw_input().split())
k = input()

s = 0
for i in xrange(k):
    dx, dy = map(int, raw_input().split())
    
    l, r = 0, 1000000000
    while r - l > 1:
        mid = (l + r) / 2
        if 1 <= xc + mid * dx <= n and 1 <= yc + mid * dy <= m:
            l = mid
        else:
            r = mid
    xc = xc + l * dx 
    yc = yc + l * dy
    s += l
    
print s
