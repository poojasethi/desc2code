#!/usr/bin/env python

def mov(N, M, X, Y, Dx, Dy):
    Qx = Qy = 1000000000
    Lx = Ly = 1
    if Dx > 0: 
        Qx =(N - X) / Dx
        Lx = 1 
    else: 
        if (Dx != 0):
            Qx = (X - 1) / (0 - Dx)
            Lx = -1
    if Dy > 0: 
        Qy =(M - Y) / Dy
        Ly = 1 
    else: 
        if (Dy != 0):
            Qy = (Y - 1) / (0 - Dy)
            Ly = -1
    Q = min(Qx , Qy)
    return X + Lx * abs(Dx) * Q, Y + Ly * abs(Dy) * Q,Q 
    

if __name__ == "__main__":
    N, M = map (int, raw_input().split()) 
    X, Y = map (int, raw_input().split())
    K = int(raw_input())
    Count = 0
    for i in range(K):
        Dx, Dy =map (int, raw_input().split())
        X, Y, Q = mov(N, M, X, Y, Dx, Dy)
        Count += Q
    print Count
