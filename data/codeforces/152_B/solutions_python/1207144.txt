n, m = [int(x) for x in raw_input().split()]
x, y = [int(x) for x in raw_input().split()]
k = input()
ans = 0
for i in [[int(p) for p in raw_input().split()] for q in range(k)]:
    l, r = 0, 10**10
    while r - l > 1:
        z = (l+r)/2
        if 1 <= x + i[0]*z <= n and 1 <= y + i[1]*z <= m: l = z
        else: r = z
    ans += l
    x, y = x + i[0]*l, y + i[1]*l
print ans
