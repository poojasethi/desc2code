import sys;

#sys.stdin = file("in.txt");

def ri():
    return map(int, raw_input().split());

def cal(x, len, dx):
    if (dx == 0):
        return 10000000000;
    if (dx > 0):
        return (len - x)/dx;
    else:
        return (x - 1)/(-dx);

n, m = ri();
x, y = ri();
k = ri()[0];

ans = 0;
for i in xrange(k):
    dx, dy = ri();
    step = min(cal(x, n, dx), cal(y, m, dy));
    ans += step;
    x += step * dx;
    y += step * dy;
   
print ans;
