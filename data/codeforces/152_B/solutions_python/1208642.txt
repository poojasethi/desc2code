import sys
n, m = map(int, sys.stdin.readline().split())
x, y = map(int, sys.stdin.readline().split())
k = int(sys.stdin.readline())
cnt = 0
for i in xrange(k):
    dx, dy = map(int, sys.stdin.readline().split())
    tmp = n+m
    if dx > 0:
        tmp = min(tmp, (n - x) / dx)
    if dx < 0:
        tmp = min(tmp, (x - 1) / -dx)
    if dy > 0:
        tmp = min(tmp, (m - y) / dy)
    if dy < 0:
        tmp = min(tmp, (y - 1) / -dy)
    cnt += tmp
    x += dx * tmp
    y += dy * tmp
print cnt
