n,m = map(int, str(raw_input()).split())
x,y = map(int, str(raw_input()).split())
size = input()
total = 0
while size:
    size -= 1
    dx, dy = map(int, str(raw_input()).split())
    if dx<0: cx = x/abs(dx) if (x%abs(dx)) else (x-1)/abs(dx)
    elif dx>0: cx = (n-x)/dx
    else: cx = 1000000000
    if dy<0: cy = y/abs(dy) if (y%abs(dy)) else (y-1)/abs(dy)
    elif dy>0: cy = (m-y)/dy
    else: cy = 1000000000
    step = min(cx,cy)
    x,y = x+step*dx, y+step*dy
    total += step
print total
