def readAB():
    s = raw_input()
    i = s.index(' ')
    return int(s[:i]), int(s[i+1:])

def move(n,m,x,y,dx,dy):
    t1 = t(n, x, dx)
    t2 = t(m, y, dy)
    c = min(t1, t2)
    return x + dx * c, y + dy * c, c

def t(n,x,dx):
    if dx == 0: return 10000000000
    elif dx > 0: return (n-x) / dx
    else: return (x - 1) / (-dx)

n,m = readAB()
x,y = readAB()
c = 0
for i in xrange(int(raw_input())):
    dx,dy = readAB()
    x,y,z = move(n,m,x,y,dx,dy)
    c += z
print c
