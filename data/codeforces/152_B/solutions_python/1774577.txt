n, m = map(int, raw_input().split())
x, y = map(int, raw_input().split())
k = input()
ans = 0
for i in range(k):
    dx, dy = map(int, raw_input().split())
    move = -1
    if dx != 0:
        movex = (n - x) / dx if dx > 0 else (1 - x) / dx
        move = movex
    if dy != 0:
        movey = (m - y) / dy if dy > 0 else (1 - y) / dy
        move = min(movey, movex) if move != -1 else movey
    ans += move
    x += move * dx
    y += move * dy
print ans
