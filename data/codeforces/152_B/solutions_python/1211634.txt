#!/usr/bin/python2
from sys import stdin
from math import floor


n, m = map(int,stdin.readline().split()) #width, height
x, y = map(int,stdin.readline().split())
i    = int(stdin.readline())
#print n,m,x,y
res = 0
for xx in range(0, i):
  st = 0
  vx, vy = map(int,stdin.readline().split())
  if vx == 0:
    stx = -1
  elif vx > 0:
    #print "nxvx",n,x,vx
    stx = (n-x)/vx
  else:
    stx = -(x-1)/vx

  if vy == 0:
    sty = -1
  elif vy > 0:
    sty = (m-y)/vy
  else:
    sty = -(y-1)/vy

  if(stx == -1): st = sty
  elif(sty == -1): st = stx
  else:
    st = min(stx,sty)
  #print st,stx,sty
  if st > 0:
    res += st
    x += st * vx
    y += st * vy
  #print ""

print res
