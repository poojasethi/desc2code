n, m = map(int, raw_input().split())
x, y = map(int, raw_input().split())
k = input()
d = [map(int, raw_input().split()) for _ in xrange(k)]
res = 0
for dx, dy in d:
    t = (1 << 31) - 1
    if dy > 0: t = min(t, (m - y) // dy)
    elif dy < 0: t = min(t, (1 - y) // dy)
    if dx > 0: t = min(t, (n - x) // dx)
    elif dx < 0: t = min(t, (1 - x) // dx)
    res += t
    x += t * dx
    y += t * dy
print res
