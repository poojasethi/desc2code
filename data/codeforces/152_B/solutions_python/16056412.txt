n, m = map(int, raw_input().split())
xc, yc = map(int, raw_input().split())
k = input()
ans = 0
for i in range(k):
	dx, dy = map(int, raw_input().split())
	sx = 10 ** 9
	if dx > 0: sx = (n - xc) / dx
	elif dx < 0: sx = (xc - 1) / (-dx)
	sy = 10 ** 9
	if dy > 0: sy = (m - yc) / dy
	elif dy < 0: sy = (yc - 1) / (-dy)
	s = min(sx, sy)
	xc += s * dx
	yc += s * dy
	ans += s
print ans
