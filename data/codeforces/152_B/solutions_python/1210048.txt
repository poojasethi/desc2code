n,m=[int(zz) for zz in raw_input().split(' ')]
x,y=[int(zz) for zz in raw_input().split(' ')]
x-=1
y-=1
k=int(raw_input())
res=0
for step in range(0,k):
    dx,dy=[int(zz) for zz in raw_input().split(' ')]
    MAX=1000000001
    if(dx<0):
        MAX=min(MAX,x//(-dx))
    elif(dx>0):
        MAX=min(MAX,(n-1-x)//dx)
    if(dy<0):
        MAX=min(MAX,y//(-dy))
    elif(dy>0):
        MAX=min(MAX,(m-1-y)//dy)
    x+=MAX*dx
    y+=MAX*dy
    res+=MAX
print res
