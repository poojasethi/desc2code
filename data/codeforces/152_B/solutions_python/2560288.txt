n, m = map(int, raw_input().split())
x, y = map(int, raw_input().split())
r, o = 0, max(n, m)
for _ in range(input()):
	a, b = map(int, raw_input().split())
	s = min((x - 1) / -a if a < 0 else (n - x) / a if a > 0 else o, (y - 1) / -b if b < 0 else (m - y) / b if b > 0 else o)
	r += s
	x += a * s
	y += b * s
print r
