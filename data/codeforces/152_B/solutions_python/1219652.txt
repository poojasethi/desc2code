def ri():
	return [int(i) for i in raw_input().split()]

def calc(p,n,step):
	if step == 0:
		return 1000000000000
	elif step > 0:
		return int((n-p)/step)
	else:
		return int((p-1)/-step)

n,m = ri()
cx,cy = ri()

k = ri()[0]

cnt = 0

while k>0:
	k=k-1
	dx,dy = ri()
	c = min(calc(cx,n,dx),calc(cy,m,dy))
	cx += dx*c
	cy += dy*c
	cnt += c

print cnt
