n, m = map(int, raw_input().split())
x, y = map(int, raw_input().split())
res, INF = 0, max(n, m)
for _ in range(input()):
	dx, dy = map(int, raw_input().split())
	step = min((x - 1) / -dx if dx < 0 else (n - x) / dx if dx > 0 else INF, (y - 1) / -dy if dy < 0 else (m - y) / dy if dy > 0 else INF)
	res += step
	x += dx * step
	y += dy * step
print res
