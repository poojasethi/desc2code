def in_range(x, y, n, m):
    return x >= 1 and x <= n and y >= 1 and y <= m

n, m = map(int, raw_input().split())
x, y = map(int, raw_input().split())
k = input()
ts = 0
for i in xrange(k):
    vx, vy = map(int, raw_input().split())
    s = 10 ** 10
    if vx > 0:
	s = min(s, (n - x) / vx)
    elif vx < 0:
	s = min(s, (1 - x) / vx)
    if vy > 0:
	s = min(s, (m - y) / vy)
    elif vy < 0:
	s = min(s, (1 - y) / vy)
    x += s * vx
    y += s * vy
    ts += s
print ts
