n,m=map(int,raw_input().split())
xc,yc=map(int,raw_input().split())
k=input()
A=[map(int,raw_input().split()) for i in range(k)]
c=0
for x in A:
    dx=n-xc if x[0]>=0 else 1-xc
    if x[0]!=0:
        dx/=x[0]
    else:
        dx=10**10
    dy=m-yc if x[1]>=0 else 1-yc
    if x[1]!=0:
        dy/=x[1]
    else:
        dy=10**10
    t=min(dx,dy)
    xc+=t*x[0]
    yc+=t*x[1]
    c+=t
print c
