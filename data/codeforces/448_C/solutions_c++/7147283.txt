 #include <iostream>
 
 using namespace std;
 
 int main() {
 int n; cin >> n;
 int a[n+1];
 a[0] = 0;
 for (int i=1; i<=n; i++) cin >> a[i];
 int dp[n+1][n+1];
 for (int j=0; j<=n; j++) dp[n][j] = 0;
 for (int i=n-1; i>=0; i--) for (int j=0; j<=i; j++) {
 if (a[j] >= a[i+1]) dp[i][j] = dp[i+1][i+1]; //Already painted
 else {
 dp[i][j] = min( 1 + dp[i+1][j], a[i+1]-a[j] + dp[i+1][i+1]);
 }
 }
 
 cout << dp[0][0] <<endl;
 
 }
