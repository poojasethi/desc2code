int solve(int a[],int l,int r,int h)
{
    if(l>r)
            return 0;
    int p=min_element(a+l,a+r+1)-a;
    int mini=a[p];
    return min(r-l+1,solve(a,l,p-1,mini)+solve(a,p+1,r,mini)+mini-h);
}
