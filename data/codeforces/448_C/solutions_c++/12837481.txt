#include <bits/stdc++.h>

using namespace std;

const int N = 5010;

int a[N], n;

int dfs(int l, int r, int h) {
  if (l > r) return 0;
  int m = min_element(a + l, a + r + 1) - a;
  return min(r - l + 1, dfs(l, m - 1, a[m]) + dfs(m + 1, r, a[m]) + a[m] - h);
}

int main() {
  scanf("%d", &n);
  for (int i = 1; i <= n; ++i) {
    scanf("%d", a + i);
  }
  printf("%d\n", dfs(1, n, 0));
  return 0;
}
