#include <cstdio>
#include <algorithm>
using namespace std;
int const MAX = 5005;
int a[MAX], dp[MAX][MAX];
int main()
{
    int n; 
    scanf("%d", &n);
    for(int i = 1; i <= n; i++)
        scanf("%d", &a[i]);
    for(int i = n; i >= 1; i--)
        for(int j = 0; j < i; j++)
            dp[i - 1][j] = (a[j] >= a[i]) ? dp[i][i] : min(dp[i][j] + 1, dp[i][i] + a[i] - a[j]);
    printf("%d\n", dp[0][0]);
}
