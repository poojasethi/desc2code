#include<iostream>
#include<cstdio>
using namespace std;

#define maxn 5010

int a[maxn];

int cnt(int l,int r,int w)
{
	if (l>r) return 0;
	int mi=l;
	for(int i=l+1;i<=r;i++)
	{
		if (a[mi]>a[i]) mi=i;
	}
	return min(r-l+1,cnt(l,mi-1,a[mi])+cnt(mi+1,r,a[mi])-w+a[mi]);
}

int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	cout<<cnt(1,n,0)<<endl;
	return 0;
}
