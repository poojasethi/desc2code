#include <iostream>

using namespace std;

int wall[100000];
int n;

int paint(int l, int r, int h) 
{
  if (l >= r) 
  	return 0;
  int m = l;
  for (int i = l; i < r; i++) 
  	if (wall[i] < wall[m]) m = i;
  //h or v
  return min(r-l, wall[m]-h + paint(l, m, wall[m]) + paint(m+1, r, wall[m]));
}

int main() 
{
  cin >> n;
  for (int i = 0; i < n; i++) 
  	cin >> wall[i];
  cout << paint(0, n+1, 0) << endl;
  return 0;
}
