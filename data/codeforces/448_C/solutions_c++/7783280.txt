#include <iostream>
using namespace std;
int n,a[100005];
int dfs(int l,int r,int k)
{
	int i,j;
	if (l>r) return 0;
	int mi=l;
	for (i=l;i<=r;i=i+1)
	{
		if (a[mi]>a[i]) mi=i;
	}
	return min(r-l+1,dfs(l,mi-1,a[mi])+dfs(mi+1,r,a[mi])+a[mi]-k);
}
int main()
{
	int i,j;
	cin>>n;
	for (i=1;i<=n;i=i+1)
	{
		cin>>a[i];
	}
	cout<<dfs(1,n,0)<<endl;
	return 0;
}
