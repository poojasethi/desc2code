#include <cstdio>
#include <algorithm>
using namespace std;
int h[5000+10];
int strokes(int from,int to,int hi)
{
    if(from > to) return 0;
    int minh=min_element(h+from,h+to+1)-h;
    return min(to-from+1,strokes(from,minh-1,h[minh])+strokes(minh+1,to,h[minh])+h[minh]-hi);
}
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1; i<=n; i++)
        scanf("%d",&h[i]);
    printf("%d",strokes(1,n,0));
    return 0;
}
