#include<bits/stdc++.h>
using namespace std;
int n,a[5001];

int f(int l,int u,int val)
{
    if(u<l)
        return 0;
    int m=min_element(a+l,a+u+1)-a;
    return min(u-l+1,f(l,m-1,a[m])+f(m+1,u,a[m])+a[m]-val);
}
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>a[i];

    cout<<f(1,n,0)<<endl;
}
