#include <cstdio>
#include <algorithm>
using namespace std;
int a[5010];
int dp(int l,int r,int k){
    if(l > r)   return 0;
    int t = l;
    for(int i = l;i <= r;i++){
        if(a[i] < a[t]) t = i;
    }
    return min(r-l+1,dp(l,t-1,a[t])+a[t]-k+dp(t+1,r,a[t]));
}

int main(){
    int n;
    scanf("%d",&n);
    for(int i = 0;i < n;i++)    scanf("%d",&a[i]);
    printf("%d\n",dp(0,n-1,0));
    return 0;
}
