#include <cstdio>
#include <algorithm>
using namespace std;

const int N=5005;

int n,ans=N,a[N],f[N];

int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d",a+i);
		f[i]=N;
		int h=a[i];
		for(int j=i-1;j>=0;j--){
			h=min(h,a[j]);
			f[i]=min(f[i],f[j]+(i-j-1)+(a[i]-h));
		}
	}
	for(int i=0;i<=n;i++) ans=min(ans,f[i]+(n-i));
	printf("%d\n",ans);
}
