#include<bits/stdc++.h>
using namespace std;
#define maxn 5005
int a[maxn];
int gao(int l, int r, int h) {
	if(l > r) return 0;
	int p, m = 0x7fffffff;
	for(int i = l; i <= r; i++) {
		if(a[i] < m) {
			m = a[i];
			p = i;
		}
	}
	return min(r - l + 1, gao(l, p - 1, m) + gao(p + 1, r, m) + m - h);
}
int main() {
	int n;
	scanf("%d", &n);
	for(int i = 1; i <= n; i++) scanf("%d", &a[i]);
	printf("%d\n", gao(1, n, 0));
	return 0;
}
