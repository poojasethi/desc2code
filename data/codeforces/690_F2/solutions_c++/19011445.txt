ll HH(ll x){
	return x * x * x * x * 3 + 425689 * x - 23156667;
}

ll go(int par, int u){
	ll res = 32131;

	Rep(i, sz(KK[u])){
		int v = KK[u][i];
		if(v == par) continue;
		res += HH(go(u, v));
	}

	return res;
}

ll hash_tree(vector<II> V){
	int d[105];
	ms(d, 0);
	For(i, 1, n) if(sz(KK[i])) KK[i].clear();
	Rep(i, sz(V)){
		int u = V[i].fi, v = V[i].se;
		KK[u].pb(v);
		KK[v].pb(u);
		d[u]++;
		d[v]++;
	}

	int number = 0;
	int que[105], size = 0, que1[105], size1 = 0;
	For(i, 1, n) {
		if(d[i] >= 1) number++;
		if(d[i] == 1) que[size++] = i;
	}


	while(true){
		if(size <= 2) break;
		size1 = 0;
		Rep(t, size){
			que1[size1++] = que[t];
		}
		size = 0;
		Rep(t, size1){
			int u = que1[t];
			Rep(j, sz(KK[u])){
				int v = KK[u][j];
				d[v]--;
				if(d[v] == 1){
					que[size++] = v;
				}
			}
		}
	}

	ll res = 325843;
	vector<ll> Vll;

	Rep(i, size){
		Vll.pb(go(0, que[i]));
	}
	sort(all(Vll));
	Rep(i, size){
		res = res * mod + Vll[i];
	}
	return res;
}

ll hash_forest(vector<II> V){
	bool flag[105];
	vector<int> VV[105];
	ms(flag, 0);

	Rep(i, sz(V)){
		int u = V[i].fi, v = V[i].se;
		VV[u].pb(v);
		VV[v].pb(u);
	}

	vector<ll> Vll;
	for(int i = 1; i <= n; i++) if(!flag[i]){
		flag[i] = 1;
		int que[105], size = 0;
		que[size++] = i;
		vector<II> temp;
		Rep(t, size){
			int u = que[t];
			Rep(j, sz(VV[u])){
				int v = VV[u][j];
				if(!flag[v]){
					flag[v] = 1;
					que[size++] = v;
					temp.pb(mp(u, v));
				}
			}
		}
		Vll.pb(hash_tree(temp));
	}
	sort(all(Vll));
	ll res = 132435;
	for(int i = 0; i < sz(Vll); i++) res = res * mod + Vll[i];
	return res;
}

ll hash_alot(vector<vector<II> > V){
	vector<ll> Vll;
	for(int i = 0; i < sz(V); i++){
		Vll.pb(hash_forest(V[i]));
	}
	sort(all(Vll));

	ll res = 132435;
	for(int i = 0; i < sz(Vll); i++) res = res * mod + Vll[i];
	return res;
}
