n = int(raw_input())
a = map(int, raw_input().split())

d = {}
for i in a:
    if d.get(i):
        d[i] += 1
    else:
        d[i] = 1
a.sort()
a[0] = 1
ans = a[0] + 1
for i in xrange(n - 1):
    if a[i + 1] > a[i] + 1:
        a[i + 1] = a[i] + 1
    if a[i + 1] + 1 > ans:
        ans = a[i + 1] + 1
print ans
