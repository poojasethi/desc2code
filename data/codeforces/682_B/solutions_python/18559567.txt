def findMaxMex():
    n = int(raw_input())
    s = raw_input()
    tokens = s.split()
    values = [0]*n
    for i in range(0,n):
        values[i] = int(tokens[i])

    values.sort()
    currNum = 1
    for i in range(0, n):
        if values[i] >= currNum:
            currNum+=1
    print(str(currNum))




findMaxMex()
