
t = str(raw_input())
a = int(raw_input())

t = t.split(':')

hours = int(t[0])
mins = int(t[1])

time = hours*60 + mins + a

nh = (time / 60) % 24

nm = time % 60

if nh < 10:
    nh = '0'+str(nh)
else:
    nh = str(nh)

if nm < 10:
    nm = '0'+str(nm)
else:
    nm = str(nm)

print nh + ':' + nm
