#!/usr/bin/python

def ir():
    return int(raw_input())

def read_time():
    s = raw_input()
    s = s.split(":")
    return map(int, s)

def next_min(h, m):
    m += 1
    if m==60: m=0; h+=1
    if h==24: h=0
    return h, m

h, m = read_time()
n = ir()

for i in range(n):
    h, m = next_min(h, m)

print "%02d:%02d" % (h, m)
