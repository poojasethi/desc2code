import math

time = raw_input()
timehr = int(time.split(':')[0])
timemin = int(time.split(':')[1])
epochtime = (timehr*60)+timemin
nexttime = epochtime + int(raw_input())
outhr = int(math.floor(nexttime/60))
outmin = int(nexttime - (60*outhr))
print str(outhr%24).zfill(2) + ":" + str(outmin).zfill(2)
