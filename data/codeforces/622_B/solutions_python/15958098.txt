
time = raw_input()

hour = int(time[:2])
minute = int(time[3:])

a = input()

if a % 60 == 0:

	temp = a / 60
	hour = hour + temp
else:

	addh = a / 60
	addm = a % 60
	hour += addh
	minute += addm

if minute >= 60:
	minute = minute - 60
	hour += 1
if hour >= 24:
	hour = hour % 24

if hour < 10:

	hour = "0" + str(hour)
if minute < 10:

	minute = "0" + str(minute)

print str(hour) + ":" + str(minute)
