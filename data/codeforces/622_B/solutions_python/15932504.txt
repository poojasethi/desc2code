inp = raw_input()
a = input()
minute = inp.split(":")[1]
hour = inp.split(":")[0]
minute = int(minute)
hour = int(hour)
fin_min = str((minute+a)%60)
if len(fin_min)<2:
	fin_min="0"+fin_min
extra_hours = (minute+a)/60
fin_hour = str((hour+extra_hours)%24)
if len(fin_hour)<2:
	fin_hour="0"+fin_hour
print fin_hour+":"+fin_min
