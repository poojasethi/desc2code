hour, minute = [int(tmp) for tmp in raw_input().split(':')]
after = int(raw_input())
minute += after
h = 0
if(minute >= 60):
	h = minute//60
	minute = minute%60

hour += h
if(hour >= 24):
	hour = hour % 24

ans = ""
if(hour < 10):
	ans = "0" + str(hour) + ":"
else:
	ans = str(hour) + ":"

if(minute < 10):
	ans = ans + "0" + str(minute)
else:
	ans = ans + str(minute)

print(ans)
