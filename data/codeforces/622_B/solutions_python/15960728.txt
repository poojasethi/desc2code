ar = []

a = 0
b = 0

for i in xrange(24*60):
    hours = str(a)
    mins = str(b)

    if len(hours)==1:
        hours = "0" + hours
    if len(mins)==1:
        mins = "0" + mins

    tim = hours + ":" + mins

    ar.append(tim)

    b += 1
    a += b/60
    b = b%60

tim = raw_input()
ind = ar.index(tim)
lapse = input()
ind += lapse
ind %= (24*60)
print ar[ind]
