r = raw_input()
dt = int(input())
r = r.strip().split(':')
r = [int(i) for i in r ]

hr = (dt - dt%60)/60
mi = dt%60

r[0] = r[0]+hr
r[1] = r[1]+mi

if r[1] >= 60:
  r[0] += 1
  r[1]=r[1]%60

if r[0] >= 24:
  r[0] = r[0]%24

if r[0]<10:
  r[0]='0'+str(r[0])
else:
  r[0]=str(r[0])

if r[1]<10:
  r[1]='0'+str(r[1])
else:
  r[1] = str(r[1])
  
print r[0]+':'+str(r[1])
