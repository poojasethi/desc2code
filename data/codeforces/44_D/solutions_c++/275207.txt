#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
#include<cmath>
#include<iomanip>

using namespace std;

int n;
int pos[5001][3];

double dist(int i,int j)
{
    return sqrt((pos[i][0]-pos[j][0])*(pos[i][0]-pos[j][0])+(pos[i][1]-pos[j][1])*(pos[i][1]-pos[j][1])+(pos[i][2]-pos[j][2])*(pos[i][2]-pos[j][2]));
}

int main()
{
    cin>>n;
    for (int i=0;i<n;i++)
        cin>>pos[i][0]>>pos[i][1]>>pos[i][2];
    double ans2 = (1<<30);
    for (int i=1;i<n;i++)
        for (int j=i+1;j<n;j++)
            ans2 = min(ans2,dist(0,i) + dist(0,j) + dist(i,j));
    cout<<setprecision(10)<<ans2/2<<endl;
    
    return 0;
}
