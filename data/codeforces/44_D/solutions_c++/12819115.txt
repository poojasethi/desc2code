#include<bits/stdc++.h>
using namespace std;
double dist(double a,double b,double c){
  return sqrt(a*a+b*b+c*c);
}
int n;
double x[5005],y[5005],z[5005];
int main(){
  scanf("%d",&n);
  for(int i=1;i<=n;i++) scanf("%lf %lf %lf",&x[i],&y[i],&z[i]);
  double ans=1E18;
  for(int i=2;i<=n;i++){
    for(int j=i+1;j<=n;j++){
      double d=dist(x[1]-x[i],y[1]-y[i],z[1]-z[i])
              +dist(x[i]-x[j],y[i]-y[j],z[i]-z[j])
              +dist(x[j]-x[1],y[j]-y[1],z[j]-z[1]);
      ans=min(ans,d/2);
    }
  }
  printf("%.10lf\n",ans);
  return 0;
}
