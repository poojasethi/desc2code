#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;

const double finf = 1e18;

int n;
double x[5005],y[5005],z[5005],out;

double dis(int i, int j) {
    return sqrt((x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j])+(z[i]-z[j])*(z[i]-z[j]));
}

int main() {
    scanf("%d",&n);
    for (int i=0; i<n; i++) 
        scanf("%lf%lf%lf",&x[i],&y[i],&z[i]);
    out=finf;
    for (int i=1; i<n; i++) 
        for (int j=i+1; j<n; j++) 
            out=min(out,(dis(0,i)+dis(0,j)+dis(i,j))/2);
    printf("%f\n",out);
    return 0;
}
