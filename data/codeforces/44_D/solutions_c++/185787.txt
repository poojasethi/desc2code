#include <iostream>
#include <cmath>
using namespace std;
struct p{
	double x,y,z;
} a[5100];
inline double sqr(double x){
	return x * x;
}
inline double dist(p a, p b){
	return sqrt(sqr(a.x - b.x) + sqr(a.y - b.y) + sqr(a.z - b.z));
}
int n;
double s[5100];
int main(){
	cin >> n;
	for(int i = 0; i < n; ++i)
		cin >> a[i].x >> a[i].y >> a[i].z;
	double ans = 1e9;
	for(int i = 1; i < n; ++i)
		s[i] = dist(a[0], a[i]);
	for(int i = 1; i < n; ++i)
		if(s[i] < ans)
			for(int j = i + 1; j < n; ++j)
				ans = min(ans, s[i] + s[j] + dist(a[i], a[j]) );
	cout.precision(15);
	cout << fixed << ans / 2 << endl;
	return 0;
}
