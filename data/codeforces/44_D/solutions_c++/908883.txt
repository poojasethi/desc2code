#include <cstdio>
#include <cstring>
#include <vector>
#include <cmath>
#include <algorithm>
using namespace std;

int x[5005], y[5005], z[5005];

double sq(double a) {
	return a*a;
}
double dist(int i, int j) {
	return sqrt(sq(x[i]-x[j])+sq(y[i]-y[j])+sq(z[i]-z[j]));
}
double st[5005];

int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++)
		scanf("%d%d%d", &x[i], &y[i], &z[i]);
	double ans=1e100;
	for(int i=1;i<n;i++)
		st[i] = dist(i, 0);
	for(int i=1;i<n;i++)
		for(int j=1;j<n;j++) {
			if(i==j) continue;
			ans = min(ans, st[i] + (st[j]-st[i] + dist(i, j))/2);
		}
	printf("%.9lf\n", ans);
	return 0;
}
