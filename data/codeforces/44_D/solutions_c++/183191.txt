#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

const int MAXn = 5000 + 5;

int n;
double x[MAXn], y[MAXn], z[MAXn];
double ans;

double dist (int i, int j)
{
	return sqrt ((x[i] - x[j]) * (x[i] - x[j]) + (y[i] - y[j]) * (y[i] - y[j]) + (z[i] - z[j]) * (z[i] - z[j]));
}

int main()
{
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> x[i] >> y[i] >> z[i];
	ans = 1e15;
	for (int i = 1; i < n; i++)
		for (int j = i+1; j < n; j++)
			ans = min (ans, (dist (0, i) + dist (0, j) + dist (i, j)) / 2.0);
	cout << fixed << setprecision(7) << ans << endl;
}
