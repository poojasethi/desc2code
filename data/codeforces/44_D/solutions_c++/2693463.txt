#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#define sqr(x) ((x)*(x))
struct  Point{
    int x,y,z;
    void init(){
        scanf("%d%d%d", &x, &y , &z );
    }
}data[5005];
using namespace std;
double  dis( Point a, Point b){
    return sqrt(sqr(a.x-b.x)+sqr(a.y-b.y)+sqr(a.z-b.z));
}

int     n;
int     main(){
    scanf("%d", &n );
    for (int i = 0; i<n; i++) data[i].init();
    double ans = 1e100;
    for (int i = 1; i<n; i++)
        for (int j = i+1; j<n; j++)
            ans = min( ans , dis(data[0],data[i])+dis(data[0],data[j])+dis(data[i],data[j]) );
    printf("%.8f\n" , ans/2 );
    return 0;
}
