#include <iostream>
#include <vector>
#include <cmath>
#include <climits>
#include <cstdio>

using namespace std;

int x[5001],y[5001],z[5001];

double abs(int i ,int j){
  return sqrt( (x[i] -x[j]) * (x[i] -x[j]) + 
	       (y[i] -y[j]) * (y[i] -y[j]) +
	       (z[i] -z[j]) * (z[i] -z[j]) );
}

double calc(int i ,int j){
  return abs(i,0) + abs(i,j) + abs(j,0);
}

int main(){
  int n;cin >> n;
  for(int i = 0;i< n;i++){
    cin >> x[i] >> y[i] >> z[i];
  }
  double mn = INT_MAX;
  for(int i = 1;i<n;i++){
    for(int j = i+1;j<n;j++){
      mn = min (calc(i,j), mn);
    }
  }
  printf("%.7f",mn/2);
  return 0;
}
