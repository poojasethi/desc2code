#include<math.h>
using namespace std;

double x[5100],y[5100],z[5100];

double dis(int i,int j)
{
    return sqrt((x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j])+(z[i]-z[j])*(z[i]-z[j]));
}
int main()
{
    int n,i,j,k;

    cin>>n;
    for(i=1;i<=n;i++)
	cin>>x[i]>>y[i]>>z[i];
    double ans=1e50;

    for(i=2;i<=n;i++)
	for(j=i+1;j<=n;j++)
	    ans=min(ans,dis(1,i)+dis(1,j)+dis(i,j));
    printf("%.12lf\n",ans/2);
}
