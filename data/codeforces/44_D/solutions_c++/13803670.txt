#include <iostream>
#include <vector>
#include <cmath>
#include <cstdio>

using namespace std;

struct Planet {
    int x, y, z;
};

double dis(Planet& a, Planet& b) {
    return sqrt((a.x - b.x) * (a.x - b.x) + (a.y - b.y) * (a.y - b.y) + (a.z - b.z) * (a.z - b.z));
}

int main() {
    int n;
    cin >> n;
    vector<Planet> p(n);
    for (int i = 0; i < n; ++i) {
        cin >> p[i].x >> p[i].y >> p[i].z;
    }
    double ans = 1e20;
    for (int i = 1; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            ans = min(ans, dis(p[0], p[i]) + dis(p[0], p[j]) + dis(p[i], p[j]));
        }
    }
    printf("%.7f\n", ans / 2); 
}
