#include<iostream>
#include<cmath>
#include<cstdio>

using namespace std;

int n;
int x[5000], y[5000], z[5000];
double sol = 1e20;

inline double dist(int i, int j)
{
    int q = x[i] - x[j];
    int w = y[i] - y[j];
    int e = z[i] - z[j];
    return (double)sqrt(double(q*q+w*w+e*e));
}

int main()
{
    scanf("%d", &n);
    for(int i = 0; i < n; ++i)
        scanf("%d %d %d", x+i, y+i, z+i);
    for(int i = 1; i < n - 1; ++i)
        for(int j = i + 1; j < n; ++j)
            sol = min(sol, dist(0, i) + dist(0, j) + dist(i, j));
    printf("%.10lf\n", sol / 2.);
    return 0;
}
