#include <stdio.h>
#include <math.h>
double x[5010],y[5010],z[5010];
double dis(int id1,int id2)
{
    return sqrt((x[id1]-x[id2])*(x[id1]-x[id2])+(y[id1]-y[id2])*(y[id1]-y[id2])+
                (z[id1]-z[id2])*(z[id1]-z[id2]));
}
int main()
{
    int n;
    while (scanf("%d",&n)==1)
    {
        for (int i=0;i<n;i++)
            scanf("%lf%lf%lf",&x[i],&y[i],&z[i]);
        double ans=-1;
        for (int i=1;i<n;i++)
            for (int j=i+1;j<n;j++)
                if (dis(0,i)+dis(i,j)+dis(0,j)<ans||ans<0)
                    ans=dis(0,i)+dis(i,j)+dis(0,j);
        printf("%.10f\n",ans/2);
    }
    return 0;
}
