#include<cstdio>
#include<iostream>
#include<cmath>
using namespace std;

int x[5010],y[5010],z[5010];

double dis(int u,int v)
{
    return sqrt((x[u]-x[v])*(x[u]-x[v])+(y[u]-y[v])*(y[u]-y[v])+(z[u]-z[v])*(z[u]-z[v]));
}

int main()
{
    int n;
    scanf("%d",&n);
    for (int i=0;i<n;i++)
        scanf("%d%d%d",&x[i],&y[i],&z[i]);
    double ans=1e18;
    for (int i=1;i<n;i++)
        for (int j=1;j<i;j++)
            if (dis(0,i)+dis(i,j)+dis(j,0)<2*ans) ans=(dis(0,i)+dis(i,j)+dis(j,0))/2;
    printf("%.9lf\n",ans);
    return 0;
}
