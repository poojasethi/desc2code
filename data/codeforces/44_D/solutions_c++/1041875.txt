#include <iostream>
#include <cmath>
#include <cstdio>
using namespace std;
#define REP(i,a,b) for(i=a;i<b;i++)

int n,i,j;
double x[6000],y[6000],z[6000],ans;

double sqr(double x){return x*x;}

double dist(int i,int j){return sqrt(sqr(x[i]-x[j])+sqr(y[i]-y[j])+sqr(z[i]-z[j]));}

int main(){
        cin>>n;
        REP(i,0,n) cin>>x[i]>>y[i]>>z[i];
        ans=1e20;
        REP(i,1,n)REP(j,i+1,n) ans=min(ans,(dist(0,i)+dist(0,j)+dist(i,j))*0.5);
        printf("%.9lf",ans);
}
