#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int n,i,j;
double ans,x[5001],y[5001],z[5001];

double min(double x,double y)
{
       if (x<y) return x; else return y;
}

double d(int i,int j)
{
       return sqrt(pow(x[i]-x[j],2)+pow(y[i]-y[j],2)+pow(z[i]-z[j],2));
}

int main()
{
    cin>>n;
    ans=1e100;
    for (int i=1;i<=n;++i) cin>>x[i]>>y[i]>>z[i];
    for (int i=2;i<=n-1;++i)
      for (int j=i+1;j<=n;++j)
        ans=min(ans,d(1,i)+d(1,j)+d(i,j));
    cout.setf(ios::fixed); 
    cout.precision(10);
    cout<<ans*0.5;
    return 0;
}
