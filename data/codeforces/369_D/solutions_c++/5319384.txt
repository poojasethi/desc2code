#include<iostream>
#include<cstring>
#define SIZE 3010
using namespace std;
int f[SIZE][SIZE];
int p[SIZE];
int s[SIZE];
int n;
int sol;
void F(int a,int b,int k)
{
    if(f[a][b])return;
    f[a][b] = 1;
    sol++;
    if(b>=n || k ==0)return;
    if(p[a] && s[b]) F(b+1,b+2,k-1);
    if(p[a] && s[b]!=100)F(a,b+1,k-1);
    if(p[a]!=100 && s[b])F(b,b+1,k-1);
}
int main()
{
    int k;
    while(cin>>n>>k)
    {
        sol=0;
        memset(f,0,sizeof(f));
        for(int i=0;i<n;i++)
        cin>>p[i];
        for(int i =n-1;i>=0;i--)
        {
            if(i==n-1)s[i] = p[i];
            else s[i] = max(p[i],s[i+1]);
        }
        F(0,1,k);
        cout<<sol<<endl;
    }
    return 0;
}
