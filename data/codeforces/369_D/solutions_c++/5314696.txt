#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
const int L=3010;
int a[L],s[L],f[L][L];
int ans,n,k;
void init(void)
{
   // freopen("cf369d.in","r",stdin);
    //freopen("cf369d.out","w",stdout);
    cin>>n>>k;
    int i;
    for(i=1;i<=n;i++)   scanf("%d",&a[i]);
    for(i=n;i>=1;i--)   s[i]=max(s[i+1],a[i]);
}
void dfs(int k,int x,int y)
{
    if(f[x][y]) return;
//  cout<<"DFS "<<x<<' '<<y<<' '<<a[x]<<' '<<s[y]<<endl;
    f[x][y]=1;ans++;
    if(!k||y>n) return;
    if(a[x])
    {
        if(s[y])    dfs(k-1,y+1,y+2);
        if(s[y]!=100)   dfs(k-1,x,y+1);
    }
    if(a[x]!=100&&s[y]) dfs(k-1,y,y+1);
}
int main(void)
{
    init();
    dfs(k,1,2);
    printf("%d\n",ans);
    return 0;
}
