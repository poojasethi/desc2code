#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;

const int N=3005;

bool vis[N][N];
int Maxp[N],p[N],n,res;

void dfs(int a,int b,int k){
    if(vis[a][b]) return;
    vis[a][b]=1;
    res++;
    if(!k || b>n) return;
    if(Maxp[b] && p[a]) dfs(b+1,b+2,k-1);
    if(Maxp[b] && p[a]!=100) dfs(b,b+1,k-1);
    if(Maxp[b]!=100 && p[a]) dfs(a,b+1,k-1);
}

int main(){
    int k;
    while(cin>>n>>k){
        for(int i=1;i<=n;i++) scanf("%d",&p[i]);
        Maxp[n]=p[n];
        for(int i=n-1;i;i--) Maxp[i]=max(Maxp[i+1],p[i]);
        res=0;
        memset(vis,0,sizeof(vis));
        dfs(1,2,k);
        printf("%d\n",res);
    }
    return 0;
}
