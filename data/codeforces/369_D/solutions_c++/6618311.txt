#include<iostream>
#include<string.h>
using namespace std;
int dp[5000][5000];
int p[5000];
int s[5000];
int n,k;
int count=0;
void dfs(int x,int y,int c)
{
	if(c>k)return;
	if(x>=n) x=n;
	if(y>=n) y=n;
	if(dp[x][y])return;
	dp[x][y]=1;
	count++;
	if(p[x])
	{
		if(s[y])dfs(y+1,y+2,c+1);
		if(s[y]<100) dfs(x,y+1,c+1);
	}
	if(p[x]<100)
		if(s[y])
			dfs(y,y+1,c+1);
}
int main()
{
	cin>>n>>k;
	memset(dp,0,sizeof(dp));
	memset(p,0,sizeof(p));
	memset(s,0,sizeof(s));
	for(int i=0;i<n;i++)
		cin>>p[i];
	for(int i=n-1;i>=0;i--)
		s[i]=max(s[i+1],p[i]);
	dfs(0,1,0);
	cout<<count<<endl;
}
