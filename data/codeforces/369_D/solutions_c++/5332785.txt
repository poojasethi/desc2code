#include <algorithm>
#include <cstring>
#include <cstdio>
using namespace std;
const int N=3100;
int n,k,i,j,t,m,ans,a[N],g[N],f[N][N];
int main() {
    scanf("%d%d",&n,&k);
    for (i=1;i<=n;i++) scanf("%d",a+i);
    for (i=n;i>0;i--) g[i]=max(g[i+1],a[i]);
    memset(f,0x3f,sizeof(f));
    f[1][n]=0;
    for (j=n;j>1;j--)
        for (i=1;i<=n;i++) {
            t=f[i][j];
            if (t>N) continue;
            if ((g[n-j+2]>0)&&(a[i]<100))
                f[n-j+2][j-1]=min(f[n-j+2][j-1],t+1);
            if ((g[n-j+2]<100)&&(a[i]>0))
                f[i][j-1]=min(f[i][j-1],t+1);
            if ((g[n-j+2]>0)&&(a[i]>0))
                f[n-j+3][j-2]=min(f[n-j+3][j-2],t+1);
    }
    for (i=1;i<=n+1;i++)
        for (j=0;j<=n;j++)
            if (f[i][j]<=k) ans++;
    printf("%d\n",ans);
    return 0;
}
