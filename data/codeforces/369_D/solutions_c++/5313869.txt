#include<iostream>
using namespace std;
  int n,k,i,nextmax[100001],viz[3003][3003],ans,p[10000];
 void dfs(int A,int B,int round){
      if (round>k) return;
      viz[A][B]=1;
      ans++;
      if (A>n || B>n) return;
      if (p[A]>0 && nextmax[B]>0 && viz[B+1][B+2]==0) dfs(B+1,B+2,round+1);
      if (nextmax[B]>0 && p[A]!=100 && viz[B][B+1]==0) dfs(B,B+1,round+1);
      if (p[A]>0 && nextmax[B]<100 && viz[A][B+1]==0) dfs(A,B+1,round+1);
 };
int main(){
 #ifndef ONLINE_JUDGE
   freopen("input.txt","r",stdin);
   freopen("output.txt","w",stdout);
 #endif
    cin>>n>>k;
    for (i=1;i<=n;i++)
        cin>>p[i];
    for (i=n;i>=0;i--)
        nextmax[i]=max(nextmax[i+1],p[i]);
    dfs(1,2,0);
    cout<<ans;
 return 0;
}
