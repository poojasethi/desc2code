#define pb push_back
#define X first
#define Y second

int n,k,ans;
int a[3005],b[3005];
bool vis[3005][3005];

void dfs(int p,int x,int y)
{
    if(vis[x][y])
        return;
    vis[x][y]=1;
    ans++;
    if(y>n||!p)
        return;
    if(a[x]&&b[y])
        dfs(p-1,y+1,y+2);
    if(a[x]&&b[y]!=100)
        dfs(p-1,x,y+1);
    if(b[y]&&a[x]!=100)
        dfs(p-1,y,y+1);
}

int main()
{
    cin>>n>>k;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    b[n]=a[n];
    for(int i=n-1;i>0;i--)
        b[i]=max(b[i+1],a[i]);
    dfs(k,1,2);
    cout<<ans;
}
