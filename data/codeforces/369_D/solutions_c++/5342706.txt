#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define N 3005
typedef long long LL;
int n,p[N];
int tmp[N];
LL ans;
int dp[N][N];
void go ( int a, int b, int k ) {
	if ( dp[a][b] ) return ;
	dp[a][b]=1;
	ans++;
	if ( b>=n || k==0 ) return ;
	if ( p[a] && tmp[b] ) go(b+1,b+2,k-1);
	if ( p[a] && tmp[b]!=100 ) go(a,b+1,k-1);
	if ( p[a]!=100 && tmp[b] ) go(b,b+1,k-1);
}
int main ()
{
//  freopen(".in","r",stdin);
//  freopen(".out","w",stdout);
	int k; scanf("%d%d",&n,&k);
	for ( int i=0; i<n; i++ ) scanf("%d",&p[i]);
	tmp[n-1]=p[n-1]; for ( int i=n-2; i>=0; i-- ) tmp[i]=max(tmp[i+1],p[i]);
	go(0,1,k);
	printf("%I64d\n",ans);
//  fclose(stdin);
//  fclose(stdout);
//  system("pause");
    return 0;
}
