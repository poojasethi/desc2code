#include<cstdio>
#define N 3100
int n,k,i,ans;
int p[N],v[N];
int f[N][N];
int max(int a,int b)
{
    if (a>b) return a;return b;
}
void dfs(int a,int b,int c)
{
    if (f[a][b]) return;
    f[a][b]=1;
    ans++;
    if (b>n||c==k) return;
    

    if ((p[a])&&(v[b])) dfs(b+1,b+2,c+1);
        if ((p[a])&&(v[b]!=100)) dfs(a,b+1,c+1);
    if ((p[a]!=100)&&(v[b])) dfs(b,b+1,c+1);
}

int main()
{
    scanf("%d%d",&n,&k);
    for (i=1;i<=n;i++)
       scanf("%d",&p[i]);
    for (i=n;i>=1;i--)
     v[i]=max(v[i+1],p[i]);
     
     dfs(1,2,0);
     printf("%d",ans);
}
