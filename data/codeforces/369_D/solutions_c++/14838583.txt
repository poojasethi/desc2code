#include <bits/stdc++.h>

#define ll long long
#define ld long double

using namespace std;

const int MAXN=100*10*3+10;
int p[MAXN];
int mx[MAXN];
bool mark[MAXN][MAXN];
int n,k,ans;

void calc(int f,int t, int m){
	//cerr<<m<<'#'<<k<<endl;
	if(m>k)
		return;
	//cerr<<f<<' '<<t<<endl;
	ans++;
	mark[f][t]=true;
	if(p[f-1]>0 and mx[t]>0 and mark[t+1][t+2]==false)
		calc(t+1,t+2,m+1);
	if(p[f-1]>0 and mx[t]!=100 and mark[f][t+1]==false and t+1<n+2)
		calc(f,t+1,m+1);
	if(p[f-1]!=100 and mx[t]>0 and mark[t][t+1]==false)
		calc(t,t+1,m+1);
}

int main(){
	cin>>n>>k;
	for(int i=0;i<n;i++)
		cin>>p[i];
	for(int i=0;i<MAXN;i++)
		for(int j=0;j<MAXN;j++)
			mark[i][j]=false;
	for(int i=n;i>0;i--)
		if(i==n)
			mx[i]=p[i-1];
		else
			mx[i]=max(p[i-1],mx[i+1]);
	calc(1,2,0);
	cout<<ans<<endl;
	return 0;
}
