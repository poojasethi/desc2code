#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
#define max(x,y) (x)>(y)?(x):(y);

const int N = 3001;

int n, k;
int p[N];
int MAX_P[N];
bool vis[N][N];

void dfs(int a, int b, int gun, int &cnt) {
	if (vis[a][b]) return ;
	vis[a][b] = true;
	++cnt;
	if (gun == k || b >= n)
		return ;

	if (p[a] > 0) { // 2 die 
		if (MAX_P[b] > 0) //  1 die 
			dfs(b + 1, b + 2, gun + 1, cnt);
		if (MAX_P[b] != 100)  // 1 alive
			dfs(a, b + 1, gun + 1, cnt);
	}
	if (MAX_P[b] != 0 && p[a] != 100) // 2 alive
		dfs(b, b + 1, gun + 1, cnt);
}

int main() {
	scanf("%d%d", &n, &k);
	int i;
	for (i = 0; i < n; ++i) 
		scanf("%d", &p[i]);
	for (i = n - 1; i >= 0; --i)
		MAX_P[i] = max(MAX_P[i+1], p[i]); //后面最大命中率
	int cnt=0;
	dfs(0, 1, 0, cnt);
	printf("%d\n", cnt);
	return 0;
}
