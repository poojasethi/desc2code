#include<stdio.h>
#include<string.h>
#include<algorithm>

using namespace std;

const int MAX=3010;

int dp[MAX][MAX];
typedef long long LL;
LL ans=0;

int tmp[MAX];
int p[MAX];

int n;

void dfs(int a,int b,int k)
{
    if(dp[a][b]) return;
    dp[a][b]=1;
    ans++;
    if(b>=n||k==0) return ;
    if( p[a] && tmp[b] ) dfs(b+1,b+2,k-1);
    if( tmp[b]!=100 && p[a]) dfs(a,b+1,k-1);
    if( p[a]!=100 && tmp[b] ) dfs(b,b+1,k-1);
}

int main()
{
    int k;
    scanf("%d%d", &n ,&k);
    for(int i=0;i<n;i++)
        scanf("%d",&p[i]);
    tmp[n-1]=p[n-1];
    for(int i=n-2;i>=0;i--)
        tmp[i]=max(p[i],tmp[i+1]);
    if(n>1)
        dfs(0,1,k);
    else
        ans=1;
    printf("%I64d\n",ans);
    return 0;
}
