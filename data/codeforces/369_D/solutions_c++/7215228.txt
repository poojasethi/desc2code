#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
#define maxn 4000
int visit[maxn][maxn];
int n,k,p[maxn],sp[maxn],ans=0;
void dfs(int A,int B,int t){    
    if(visit[A][B]) return ;    
    visit[A][B]=1;
    ans++;
	if(t==k||B>n) return;     
	if(p[A]){        
	   if(sp[B]) dfs(B+1,B+2,t+1);        
	   if(sp[B]!=100) dfs(A,B+1,t+1);   
    }
	if(p[A]!=100&&sp[B]) dfs(B,B+1,t+1);
}
int MM(int x,int y){
	return x>y?x:y;
}
int main(){  
   int i;  
   scanf("%d%d",&n,&k); 
   memset(visit,0,sizeof(visit));  
   for(i=1;i<=n;i++) scanf("%d",&p[i]);  
   for(i=n;i>=1;i--) sp[i]=MM(sp[i+1],p[i]);   
   dfs(1,2,0);
   printf("%d\n",ans);    
   return 0;
}
