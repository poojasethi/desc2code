#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
#define N 3010
int n, k;
int a[N];
int b[N];
bool f[N][N];
int ans;

void Dfs(int k, int x, int y)
{
  if (f[x][y]) return;
  f[x][y] = true;
  ++ans;
  if (y > n || k <= 0) return;
  if (b[y] > 0 && a[x] > 0) Dfs(k-1, y+1, y+2);
  if (b[y] > 0 && a[x] != 100) Dfs(k-1, y, y+1);
  if (b[y] != 100 && a[x]) Dfs(k-1, x, y+1);
}

int main()
{
  int i, j;
  memset(a, 0, sizeof(a));
  memset(b, 0, sizeof(b));
  memset(f, 0, sizeof(f));
  ans = 0;
  scanf("%d%d", &n, &k);
  for (i = 1; i <= n; ++i) scanf("%d", &a[i]);
  for (i = n; i; i--) b[i] = max(b[i+1], a[i]);
    //b[i]=b[i+1]<a[i]?a[i]:b[i+1];
  Dfs(k, 1, 2);
  printf("%d\n", ans);
  return 0;
}
