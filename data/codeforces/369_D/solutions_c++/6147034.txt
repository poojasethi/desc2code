#include<stdio.h>
#include<algorithm>
using namespace std;
int k,n;
int f[3011];
int s[3011];
int visit[3011][3011];
int cont=0;
int dfs(int first,int second,int all)
{
	if(all>k)return 0;
	if(visit[first][second])return 0;
    visit[first][second]=1;	
    cont++;	
	if(first>n||second>n)return 0;
	if(f[first])
	{
		if(s[second])dfs(second+1,second+2,all+1);
		if(s[second]<100)dfs(first,second+1,all+1);
	}
	if(f[first]<100)
	{
		if(s[second])dfs(second,second+1,all+1);
	}
return 0;
}
int main()
{
	scanf("%d%d",&n,&k);
	int i,j;
	for(i=1;i<=n;i++)
	scanf("%d",&f[i]),s[i]=0;
	s[n+1]=0;
	for(i=n;i>=1;i--)
	s[i]=max(s[i+1],f[i]);
	for(i=0;i<=n+3;i++)
	for(j=0;j<=n+3;j++)
	visit[i][j]=0;
	dfs(1,2,0);
	printf("%d\n",cont);
	
	
		
}
