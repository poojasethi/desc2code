#include <stdio.h>
#include <string.h>
#define MAX 3010
#define max(a,b) (a<b?b:a)
int vist[MAX][MAX];
int p[MAX],ma[MAX],n,k,count;
void fact (int a,int b,int c)
{
    if(vist[a][b])
        return ;
    count++;
    vist[a][b]=1;
    if(c>=k||b>=n)
        return ;
    if(p[a]!=0&&ma[b]!=0)
        fact(b+1, b+2, c+1);
    if(p[a]!=0&&ma[b]!=100)
        fact(a, b+1, c+1);    
    if(p[a]!=100&&ma[b]!=0)
        fact(b, b+1, c+1);
}
int main (void)
{
    int i;
    count = 0;
    memset(vist, 0, sizeof(vist));
    memset(ma, 0, sizeof(ma));
    scanf("%d%d",&n,&k);
    for (i=0; i<n; i++) {
        scanf("%d",&p[i]);
    }
    for (i=n-1; i>=0; i--) {
        ma[i]=max(ma[i+1],p[i]);
    }
    fact(0,1,0);
    printf("%d\n",count);
    return 0;
}
