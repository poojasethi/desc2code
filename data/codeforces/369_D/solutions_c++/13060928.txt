#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int n,k,ans;
int a[3005],b[3005],vis[3005][3005];
void init()
{
    scanf("%d%d",&n,&k);
    for(int i=1;i<=n;++i)scanf("%d",&a[i]);
    for(int i=n;i>=1;--i)b[i]=max(b[i+1],a[i]);
}
void dfs(int x,int y,int t)
{
    if(t>k)return;
    x=min(n+1,x);
    y=min(n+1,y);
    if(vis[x][y])return;
    vis[x][y]=1;++ans;
    if(a[x])
    {
        if(b[y])dfs(y+1,y+2,t+1);
        if(b[y]<100)dfs(x,y+1,t+1);
    }
    if(a[x]<100&&b[y])dfs(y,y+1,t+1);
}
void solve()
{
    if(n>1)dfs(1,2,0);
    else ans=1;
    printf("%d\n",ans);
}
int main()
{
    init();
    solve();
    return 0;
}
