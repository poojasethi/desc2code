#include <iostream>
#include <cstring>
#include <vector>
using namespace std ;
const int maxn = 30 * 1000 ;

#define FOR(i,j,k) for (int i = (j) ; i < (k) ; ++i)
#define pb push_back

int n , a[maxn] ;
vector <int> v ;
bool is[maxn] ;

bool check() {
    v.clear() ;
    FOR (i,0,n) if (!is[i]) v.pb (a[i]) ;
    if (v.size() == 0) v.pb(a[n-1]) , is[n-1] = false ;
    FOR (i,2,(int)v.size()) if (v[i]-v[i-1] != v[1]-v[0]) return false ;
    FOR (i,0,n) if (is[i] ) cout << a[i] << " " ; cout << endl ;
    FOR (i,0,n) if (!is[i]) cout << a[i] << " " ; cout << endl ;
    return true ;
}

bool make (int f , int s) {
    memset(is,false,sizeof(is)) ;
    int d = a[s]-a[f] , l = -1 ;
    int nxt = a[f] ;
    FOR (i,0,n) if (a[i] == nxt) is[i] = true , nxt+=d , l = i ;
    if (check()) return true ;
    is[l] = false ;
    if (check()) return true ;
    return false ;
}

int main() {
    ios_base::sync_with_stdio(false) ;
    cin >> n ;
    for (int i = 0 ; i < n ; ++i) cin >> a[i] ;
    if (!make(0,1) && !make(0,2) && !make(1,2))
        cout << "No solution" << endl ;
    return 0 ;
}
