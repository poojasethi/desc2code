void check(int k1, int k2){
	if(k1<0 || k2<0) return;
	
	fill(c,0);
	
	int i,j,k,l,x;
	bool ok=true;
	
	c[k1]=1;
	x=p[k1]-p[k2];
	
	i=k2;
	while(i>=0){
		c[i]=1;
		j=i-1;
		while(j>=0 && p[i]-p[j]!=x) --j;
		i=j;
	}
	
	for(i=n-1;i>=0;--i){
		if(!c[i]) break;
	}
	
	if(i<0) c[0]=0; else{
		for(j=i-1;j>=0;--j){
			if(!c[j]) break;
		}
		if(j>=0){
			x = p[i]-p[j];
			i=j;
			for(j=i-1;j>=0;--j) if(!c[j]){
				if(p[i]-p[j]!=x) ok=false;
				i=j;
			}
		}
	}
	
	if(ok){
		for(i=0;i<n;++i) if(c[i]) cout<<p[i]<<' '; cout<<endl;
		for(i=0;i<n;++i) if(!c[i]) cout<<p[i]<<' '; cout<<endl;
		exit(0);
	}
}
