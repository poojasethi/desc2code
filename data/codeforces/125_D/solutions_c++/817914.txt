#include <iostream>
#include <math.h>
#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <string>
using namespace std;
int n, a[300000], l = 0, b[300000];
bool u[300000];
void answer ()
{
	for (int i = 0; i < n; i++)
		if (u[i]) printf ("%d ", a[i]);
	printf ("\n");
	for (int i = 0; i < n; i++)
		if (! u[i]) printf ("%d ", a[i]);	
	exit (0);
}
bool scheck ()
{
	l = 0;
	for (int i = 0; i < n; i++)
		if (! u[i]) b[l++] = a[i];
	int r = b[1] - b[0], nxt = b[1];
	if (l == 0) return 0;
	if (l < 3) return 1;
	for (int i = 0; i < l; i++)
	{
		if (nxt != b[i] + r) return 0; 
		else nxt += r;
	}	
	return 1;
}
void check (int x, int y)
{
	int r = y - x, nxt = x;
	for (int i = 0; i < n; i++)
		if (nxt == a[i]) {u[i] = 1; nxt += r;}
	if (scheck ()) answer ();
	nxt = n - 1;
    	while(u[nxt] == 0) --nxt;
    	u[nxt] = 0;
    	if(scheck ()) answer ();
	for (int i = 0; i < n; i++)
		u[i] = 0;
}
int main () {

	#endif
	scanf ("%d", &n);
	for (int i = 0; i < n; i++)
		scanf ("%d", &a[i]);
	check (a[0], a[1]);
//	cout << "ok\n";
	check (a[0], a[2]);
//	cout << "ok\n";	
	check (a[1], a[2]);	
	printf ("No solution");
return 0;
}
