#include<iostream>
#include<cstring>
using namespace std;

int n,s,b,a[30000],last,s1,s2;
bool check=false,f[30000];

bool search(int x,int y)
{
  int k,l;
  bool out;
  memset(f,false,sizeof(f));
  f[x]=true;
  f[y]=true;
  if (!(y==n-1||y==n-2))
  {
    for (int i=y+1;i<n;i++)
      if (a[i]-last==s2-s1)
      {
	f[i]=true;
	last=a[i];
      }
  }
  else 
  {
    for (int i=x-1;i>=0;i--)
	if (a[i]-last==s1-s2)
	{
	  f[i]=true;
	  last=a[i];
	}
  }
  k=0,l=0;
  out=true;
  for (int j=0;j<n;j++)
    if (!f[j])
    {
      if (k==0)
      {
	k++;
	s=a[j];
      }
      else if (k==1)
      {
	k++;
	b=a[j];
	l=b;
      }
      else 
	if(a[j]-l==b-s) l=a[j];
	else
	{
	  out=false;
	  break;
	}
    }
  if (out)
  {
    for (int j=0;j<n;j++)
    check=false;
    for (int j=0;j<n;j++)
      if (!f[j])
      {
	check=true;
	break;
      }
    if (check)
    {
      for (int j=0;j<n;j++)
	if (f[j])
	  cout<<a[j]<<' ';
      cout<<endl;
      for (int j=0;j<n;j++)
	if (!f[j])
	  cout<<a[j]<<' ';
    }
    else
    {
      for (int j=0;j<n-1;j++)
	cout<<a[j]<<' ';
      cout<<endl<<a[n-1];
    }
    return true;
  }
  return false;
}

int main()
{
  cin>>n;
  for (int i=0;i<n;i++)
    cin>>a[i];
  s1=a[0];
  last=s2=a[1];
  if (search(0,1)) return 0;
  s1=a[0];
  last=s2=a[2];
  if (search(0,2)) return 0;
  s1=a[1];
  last=s2=a[2];
  if (search(1,2)) return 0;
  last=s1=a[n-2];
  s2=a[n-1];
  if (search(n-2,n-1)) return 0;
  last=s1=a[n-3];
  s2=a[n-1];
  if (search(n-3,n-1)) return 0;
  last=s1=a[n-3];
  s1=a[n-2];
  if (search(n-3,n-2)) return 0;
  cout<<"No solution"<<endl;
  return 0;
}
