#include<iostream> 
#include<stdlib.h> 
using namespace std; 
int n,a[30000],b[30000],c[30000]; 
 
void output() 
{ 
    int k=0,i; 
    for(i=0;i<n;i++) 
    if(c[i])cout<<a[i]<<" "; 
    cout<<endl; 
    for(i=0;i<n;i++) 
    if(!c[i])cout<<b[k++]<<" "; 
    cout<<endl; 
    exit(0); 
} 
bool iscontent() 
{ 
    int k=0,i; 
    for(i=0;i<n;i++) 
    if(!c[i])b[k++]=a[i]; 
    if(k==0)return false; 
    if(k<=2)return true; 
    int temp=b[1]-b[0]; 
    for(i=2;i<k;i++) 
    if(b[i-1]+temp!=b[i])return false; 
    return true; 
} 
void check(int x,int y) 
{ 
    int temp=y-x,i; 
    fill(c,c+30000,0); 
    for(i=0;i<n;i++) 
    { 
        if(x==a[i]) 
        { 
            c[i]=1; 
            x+=temp; 
        } 
    } 
    if(iscontent())output(); 
    x=n-1; 
    while(!c[x])--x; 
    c[x]=0; 
    if(iscontent())output(); 
} 
int main() 
{ 
    int i; 
    cin>>n; 
    for(i=0;i<n;i++) 
    cin>>a[i]; 
    check(a[0],a[1]); 
    check(a[0],a[2]); 
    check(a[1],a[2]); 
    cout<<"No solution"<<endl; 
    return 0; 
}
