#include <bits/stdc++.h>

using namespace std;

#define int  long long int

const int maxn = 1e5+10;

int a[maxn];
bool mark[maxn];
map <int , int> mp;
int n;

void Print(){
	for(int i = 1;i <= n;i++) if(mark[i] != 0) cout << a[i] << " ";
	cout << endl;
	for(int i = 1;i <= n;i++) if(mark[i] == 0) cout << a[i] << " ";
}

void check(int x , int d){
	memset(mark , false,sizeof mark);
	mark[mp[x]] = true;
	vector <int> g;
	g.push_back(x);
	while(mp[x + d] > mp[x]){
		x += d;
		mark[mp[x]] = true;
		g.push_back(mp[x]);
	}
	int k = 10000000 / n;
	while(g.size() > 1 and k){
		vector<int> v;
		for(int i = 1;i <= n;i++) if(mark[i] == false) v.push_back(a[i]);
		if(v.size() == 0){
			mark[1] = false;
			v.push_back(a[1]);
		}
		if(v.size() == 1){
			Print();
			exit(0);
		}
		int f = v[1] - v[0];
		bool Mark = false;
		for(int i = 1;i < v.size();i++) if(v[i] - v[i-1] != f) Mark = true;
		if(Mark == false){
			Print();
			exit(0);
		} 
		mark[g[g.size() - 1]] = false;
		g.pop_back();
		k--;
	}
	return ; 
}

main(){
 	cin >> n;
	for(int i = 1;i <= n;i++){
		cin >> a[i];
		mp[a[i]] = i;
	}
	if(n == 2){
		cout << a[1] << endl;
		cout << a[2] << endl;
		return 0;
	}
	if(n == 3){
		cout << a[1] << " " << a[2] << endl;
		cout << a[3] << endl;
		return 0;
	}
	if(n >= 2){
		int d = a[2] - a[1] , x = a[1]; 
		check(x , d);
	}
	if(n >= 3){
		int d = a[3] - a[1] , x = a[1]; 
		check(x , d);
		d = a[3] - a[2] , x = a[2];
		check(x , d);
	}
	cout << "No solution" << endl;
}
