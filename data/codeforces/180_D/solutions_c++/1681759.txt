#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
#include<vector>
#include<queue>
#include<map>
#include<set>
using namespace std;

int cnt[30];
char str1[5010],str2[5010];
int dfs(int k,int L,int n)
{
	if(k>=L)
		return 0;
	if(k>=n)
		return 1;
	if(cnt[str2[k]-'a']){
		cnt[str2[k]-'a']--;
		str1[k]=str2[k];
		if(dfs(k+1,L,n))
			return 1;
		cnt[str2[k]-'a']++;
	}
	for(int i=str2[k]-'a'+1;i<26;i++)
		if(cnt[i]){
			str1[k]='a'+i;
			cnt[i]--;
			return 1;
		}
	return 0;
}
int main()
{
	scanf("%s",str1);
	scanf("%s",str2);
	int len1=strlen(str1);
	int len2=strlen(str2);
	for(int i=0;i<len1;i++)
		cnt[str1[i]-'a']++;
	int ans=dfs(0,len1,len2);
	if(ans){
		int k=len1-1;
		for(int i=25;i>=0;i--){
			while(cnt[i]){
				cnt[i]--;
				str1[k--]=i+'a';
			}
		}
		printf("%s\n",str1);
	}
	else
		printf("-1\n");

}
