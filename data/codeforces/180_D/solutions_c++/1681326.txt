#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;

string s,t,ans,ans1 = "";
int d[400], i;

char bolwe(char c){
	for ( i = c + 1; i <= 'z'; i++) 
	if ( d[i] > 0 )	
	return i;
	return '*';         	
}
void rec(int v){

	if ( d[t[v]] != 0 )
	{
		ans1 = ans;
		d[t[v]]--;
		ans += t[v];
		rec(v+1);
		ans = ans.substr(0,v);	
		d[t[v]]++;
	}
	char ch = bolwe(t[v]);
	if ( ch == '*' ) return;
	d[ch]--;
	ans += ch;

	for(i = 'a'; i <= 'z'; i++)
	while ( d[i] > 0 ) 
	{
		d[i]--;
		ans += i;
	}
	cout << ans;
	exit(0);
}

int main(){
	cin >> s >> t;
	for (i = 0; i < s.size(); i++)
	d[s[i]]++;
	rec(0);
	cout << -1;

 	return 0;
}
