#include <bits/stdc++.h>
using namespace std ;
int cnt[33];
string s, t;
int n, m;
int main() {
	cin>>s ;
	n = s.length(); 
	for(int i = 0; i < n;i++){
		cnt[s[i] - 'a']++;
	}
	cin>>t ;
	m = t.size() ;
	for(int i = 0; i < n; i++) {
		bool f = 0;
		for(int let = 0; let < 26; let++) {
			if (!cnt[let]) continue;
			s[i] = 'a' + let;
			cnt[let]--;
			
			int pos = i + 1;
			for(int j = 25; j >= 0; --j) {
				for(int k = 0; k < cnt[j]; k++)
					s[pos++] = 'a' + j;
			}
			s[n] = 0;
			if (s>t) { 
				f = 1; 
				break; 
			}
			cnt[let]++;
		}
		if (!f) {
			cout << "-1\n";
			return 0;
		}
	}
	
	cout<<s<<"\n" ;
	return 0;
}
