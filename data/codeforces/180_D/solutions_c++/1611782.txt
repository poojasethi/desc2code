#include <string>

using namespace std;
int p[100010],a[26],m,n;
string t;
int sub(int q)
{
	int i,j,k,l;
	if (q+1>m) {return 1;}
	if (m==n&&(q+1)==n)
	{j=t[q]-'a'+1;}
	else {j=t[q]-'a';}
	l=0;
	for (i=j;i<26;i++) if (a[i]>0)
	{
		if (i>t[q]-'a') {p[q]=i; m=q+1; return 1;}
		a[i]--;
		p[q]=i;
		k=sub(q+1);
		a[i]++;
		if (k==1) {return 1;}
	}
	return 0;
}

{

	for (i=0;i<n;i++) {a[s[i]-'a']++;}
	if (m>n) {m=n;}
	i=sub(0);
	if (i==1)
	{
		for (j=0;j<m;j++) {cout<<(char)(p[j]+'a'); a[p[j]]--;}
		for (j=0;j<26;j++) for (k=1;k<=a[j];k++) {cout<<(char)(j+'a');}
		cout<<endl;

	}
}
