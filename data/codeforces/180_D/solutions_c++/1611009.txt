#include <cstdio>
#include <iostream>
#include <cstring>
#include <cmath>

using namespace std;

const int N=5012;
char s1[N],s2[N];
int n,m,a[27],b[27];
char ch;

bool check(int t)
  {
    for (int i=0;i<26;++i) b[i]=a[i];
    for (int i=0;i<t;++i)
      if (b[s2[i]-'a']==0) return false;else --b[s2[i]-'a'];
    for (int i=s2[t]-'a'+1;i<26;++i)
      if (b[i]>0) {--b[i],ch=i+'a';return true;}
    return false;
  }
int main()
  {
    scanf("%s",s1);
    scanf("%s",s2);
    n=strlen(s1);
    m=strlen(s2);
    for (int i=0;i<n;++i) ++a[s1[i]-'a'];s2[m]='a'-1;
    for (int i=m;i>=0;--i) 
      if (check(i))
        {
          for (int j=0;j<i;++j) printf("%c",s2[j]);printf("%c",ch);
          for (int j=0;j<26;++j)
            for (int k=1;k<=b[j];++k) printf("%c",'a'+j);
          puts("");
          return 0;
        }
    puts("-1");
  }
