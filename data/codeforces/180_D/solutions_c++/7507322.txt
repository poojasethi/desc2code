const int nm=5002;

char s[nm],t[nm];int ns,nt;
int dem[30],dem2[30];
char s2[nm];

bool kt(int d) {
    int i;
    for(i=0;i<26;++i) dem2[i]=dem[i];
    for(i=0;i<d;++i) {
        if (dem2[t[i]-97]==0) return 0;
        s2[i]=t[i];dem2[t[i]-97]--;
    }
    for(i=0;i<26;++i) if (i+97>t[d] && dem2[i]) break;
    if (i>=26) return 0;
    s2[d]=i+97;dem2[i]--;
    for(i=0;i<26;++i) {
        while (dem2[i]--) s2[++d]=i+97;
    }
    return 1;
}

    for(i=min(ns,nt);i>=0;--i) if (kt(i)) break;
    if (i>=0) printf("%s",s2);else printf("-1");
}
