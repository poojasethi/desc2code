#include<cstdio>
#include<cstring>
using namespace std;

char ps[5020],ts[5020];
int d0[30],d1[30];

int main(){
    int i,j,l,ml,kp;
    gets(ps);
    gets(ts);
    for(i=0;i<30;i++){
        d0[i]=0;
        d1[i]=0;
    }
    ml=-1;
    for(i=0;i<strlen(ps);i++){
        d0[ps[i]-97]++;
        d1[ps[i]-97]++;
        if(ps[i]-97>ml)ml=ps[i]-97;
    }
    kp=-1;
    for(i=0;i<strlen(ts);i++){
        if(ml>ts[i]-97)kp=i;
        if(d0[ts[i]-97]==0)break;
        d0[ts[i]-97]--;
        while(ml>=0&&d0[ml]==0)ml--;
    }
    if(i==strlen(ts)&&ml>=0)kp=i;
    if(kp==-1)printf("-1\n");
    else{
        for(i=0;i<kp;i++){
            printf("%c",ts[i]);
            d1[ts[i]-97]--;
        }
        if(kp<strlen(ts)){
            ts[kp]++;
            while(d1[ts[kp]-97]==0)ts[kp]++;
            printf("%c",ts[kp]);
            d1[ts[kp]-97]--;
        }
        for(i=0;i<30;i++)
            for(j=0;j<d1[i];j++)printf("%c",i+97);
        printf("\n");
    }
    return 0;
}
