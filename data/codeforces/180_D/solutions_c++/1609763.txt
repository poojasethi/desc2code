#include<cstdio>
#include<cstring>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int n;
char s[5001],t[5001];

bool greedy(int i,const int *_freq){
	static int freq[128];
	for(int c='a';c<='z';c++) freq[c]=_freq[c];

	int c='z';
	for(;i<n;i++){
		while(freq[c]==0) c--;
		s[i]=c;
		freq[c]--;
	}
	return strcmp(s,t)>0;
}

int main(){
	scanf("%s%s",s,t);
	n=strlen(s);

	int freq[128]={};
	rep(i,n) freq[s[i]]++;

	if(!greedy(0,freq)){
		puts("-1");
		return 0;
	}

	rep(i,n){
		for(int c='a';c<='z';c++) if(freq[c]>0) {
			freq[c]--;
			s[i]=c;
			if(greedy(i+1,freq)) break;
			freq[c]++;
		}
	}
	puts(s);

	return 0;
}
