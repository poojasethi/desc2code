#include <cstdio>
#include <cstring>
#include <algorithm>
#define rep(i,s,t) for (int i = s; i <= t; ++i)
#define per(i,s,t) for (int i = s; i >= t; --i)
using namespace std;
int n, m, a[5010], b[5010];
int Cnt[300];
char s[5010];
void Init(){
	scanf("%s", s+1); n = strlen(s+1);
	rep(i,1,n) a[i] = s[i];
	scanf("%s", s+1); m = strlen(s+1);
	rep(i,1,m) b[i] = s[i]; b[m+1] = -1;
}
void Solve(){
	int ans = -1, x = min(n, m)+1;
	rep(i,1,n) ++Cnt[a[i]];
	rep(i,1,x){
		rep(j,0,255) if (j > b[i] && Cnt[j]) ans = i-1;
		if (i > m || !Cnt[b[i]]) break;
		--Cnt[b[i]];
	}
	if (ans == -1){ printf("-1\n"); return; }
	memset(Cnt, 0, sizeof(Cnt));
	rep(i,1,n) ++Cnt[a[i]];
	rep(i,1,ans) { printf("%c", b[i]); --Cnt[b[i]]; }
	rep(i,0,255) if (i > b[ans+1] && Cnt[i]) { printf("%c", i); --Cnt[i]; break; }
	rep(i,0,255) rep(j,1,Cnt[i]) printf("%c", i);
	printf("\n");
}
int main(){
	Init();
	Solve();
	return 0;
}
