#include <iostream>
#include <cstring>
#include <stdlib.h>

using namespace std;

string s,t,c="";
int n,d[300];

char more(char q)
{
 for(char i=q+1;i <='z';i++)
 if (d[i] > 0) return i;
 return '#';

}

void v(int r)
{
 if (d[t[r]] > 0) 
 {
  c+=t[r];
  d[t[r]]--;
  v(r+1);
  d[t[r]]++;
  c = c.substr(0,r);
 }
 
 char x = more(t[r]);
 if (x == '#') return;
 
 c+=x;
 d[x]--;
 
 for(char i='a';i<='z';i++)
 {
  while (d[i] > 0)
  {
   c+=i;
   d[i]--;
  }
 } 
  cout << c;
  
  //system("PAUSE >> void");
  
  exit(0);
 
 
 
 

}

int main()
{
 cin >> s >> t;
 n = s.length();
 
 for(int i=0;i<n;i++) d[s[i]]++;
 
 v(0);
 
 cout << "-1";
 
 //system("PAUSE >> void");

}
