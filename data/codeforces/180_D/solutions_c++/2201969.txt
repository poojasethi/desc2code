#include <cstdio>
#include <iostream>
#include <cstring>
#define MAXN 5010
using namespace std;

int N;
int M;
int Har[305];
int tHar[305];
char Per[MAXN];
char Nar[MAXN];
char Res[MAXN];

int main(){
	scanf(" %s %s",Per+1,Nar+1);
	N=strlen(Per+1);
	M=strlen(Nar+1);
	
	for(int i=1;i<=N;i++)
		Har[Per[i]]++;
	
	for(int i=M+1;i<=N;i++)
		Nar[i]='a'-1;
		
	for(int i=min(N-1,M),flag=0;i>=0;i--){
		for(int j=0;j<300;j++)
			tHar[j]=Har[j];
		
		flag=1;	
		for(int j=1;j<=i;j++){
			tHar[Nar[j]]--;
			if(tHar[Nar[j]]<0){
				flag=0;
				break;
			}
			Res[j]=Nar[j];
		}
		
		if(flag){
			for(int j=Nar[i+1]+1;j<='z';j++)
				if(tHar[j]){
					tHar[j]--;
					Res[i+1]=j;
					
					for(int k='a',pus=i+1;k<='z';k++)
						for(int l=1;l<=tHar[k];l++)
							Res[++pus]=k;
					
					printf("%s\n",Res+1);
					return 0;
				}
		}
	}
	
	printf("-1\n");
	return 0;
}
