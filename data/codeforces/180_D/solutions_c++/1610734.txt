const int MAXN = 10000;

char s[MAXN], t[MAXN], ans[MAXN];
int cnt[60];

bool dfs(int u, int gr)
{
	if (!t[u]) return gr;

	for (int i = 0; i < 26; i++)
		if (cnt[i])
		if (gr || i >= int(s[u])-'a')
		{
		 	ans[u] = i+'a';
		 	cnt[i]--;
		 	if (dfs(u+1, gr||(i > s[u]-'a'))) return 1;
		 	cnt[i]++;
		}

	return 0;
}

int main()
{
	#ifndef ONLINE_JUDGE

	#endif

	scanf("%s%s", t, s);
	for (int i = 0; t[i]; i++)
		cnt[t[i]-'a']++;

	if (!dfs(0, 0))
		puts("-1");
	else
		printf("%s\n", ans);

	return 0;
}
