#include <iostream>
#include <fstream>
#include <cstring>
#include <stdlib.h>
#include <algorithm>
#include <cmath>

using namespace std;

string s,t,c="";
int n,d[300];

char bolwe(char q)
{
 for(int i=q+1;i<='z';i++)
 if (d[i] > 0) return i;
 
 return '#';
}

void v(int r)
{
 if (d[t[r]] > 0)
 {
  c+=t[r];
  d[t[r]]--;
  v(r+1);
  d[t[r]]++;
  c=c.substr(0,r);
 }
 
 char q = bolwe(t[r]);
 if (q == '#') return;
 
 c+=q;
 d[q]--;
 
 for(q = 'a';q<='z';q++) 
 {
  while (d[q] > 0)
  {
   c+=q;
   d[q]--;
  }
 } 

 cout << c;
 
 //system("PAUSE >> void");
 
 exit(0);
}

int main()
{
 cin >>  s >> t;
 n = s.length();
 
 for(int i=0;i<n;i++) d[s[i]]++;
 
 v(0);
 
  cout << "-1";
 
 //system("PAUSE >> void");
}
