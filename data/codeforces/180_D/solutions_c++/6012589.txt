#include <set>

#include <cstdio>
#include <vector>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
#define rep(i,a,b) for (int i=a;i<=b;++i)
#define dep(i,a,b) for (int i=a;i>=b;--i)

int ff,la,lb,c[30],g[5100];
char A[5100],B[5100];

void dfs(int k)
{
	if (ff<0) return;
	if (k&&!ff&&g[k-1]+'a'<B[k-1])return;
	if (k&&g[k-1]+'a'>B[k-1]) ff=1;
	if (ff&&k==la)
	{
		rep(i,0,la-1) printf("%c",g[i]+'a');
		puts("");
		ff=-1;
		return;
	}
	rep(i,0,25)
	if (c[i])
	{
		g[k]=i;
		c[i]--;
		dfs(k+1);
		if (ff<0) return;
		c[i]++;
	}
}
int main()
{
	scanf("%s%s",A,B);
	la=strlen(A);lb=strlen(B);
	rep(i,0,la-1) c[A[i]-'a']++;
	dfs(0);
	if (ff>=0) puts("-1");
	return 0;
}
