#include <iostream>
#include <algorithm>
using namespace std;
string s,t;
int cnt[26];
string solve(int pos){
	int cur[26]={};
	string ans=t.substr(0,pos);
	for(int i = 0; i < pos; ++i)
		cur[t[i]-'a']++;
	for(int i = 0; i < 26; ++i)
		if(cur[i]>cnt[i])
			return "";
	for(int i = t[pos]-'a'+1; i < 26; ++i)
		if(cnt[i]-cur[i]>0){
			cur[i]++;
			ans.push_back(i+'a');
			break;
		}
	for(int i = 0; i < 26; ++i)
		while(cur[i]<cnt[i]){
			ans.push_back(i+'a');
			cur[i]++;
		}
	return ans;
}
int main(){
	cin >> s >> t;
	sort(s.begin(),s.end());
	if(s>t){
		cout << s << '\n';
		return 0;
	}
	sort(s.rbegin(),s.rend());
	if(s<=t){
		cout << "-1\n";
		return 0;
	}
	for(int i = 0; i < s.size(); ++i)
		cnt[s[i]-'a']++;
	t.push_back('a'-1);
	for(int i = t.size()-1; i >= 0; --i){
		string cur = solve(i);
		if(cur>t){
			cout << cur << '\n';
			return 0;
		}
	}
}
