#include <iostream>
#include <cstdio>
#include <cstdlib>

using namespace std;


string s,t, ans;
int d[4000];

char get(char c)
{
 	for (int i = c + 1; i <= 'z'; i++)
 	if ( d[i] > 0 )
 	return i;
 	return '*';
}
void rec(int v)
{
//	cerr << "1" << v ;
	if (d[t[v]] > 0)
	{
		
	 	ans += t[v];
	 	d[t[v]]--;
	 	rec(v+1);
	 	ans = ans.substr(0, v);
	 	d[t[v]]++;
    }
    char ch = get(t[v]);
	if ( ch == '*' ) return;
	ans += ch;
	d[ch]--;
 	for (int i = 'a'; i <= 'z'; i++)
 	while ( d[i] > 0 ) ans += i, d[i]--;
 	cout << ans;
 	exit(0);
}
int main()
{
	cin >> s >> t;

	for (int i = 0; i < s.size(); i++)
	d[s[i]]++;

	rec(0);
	cout << -1;

 	return 0;
}
