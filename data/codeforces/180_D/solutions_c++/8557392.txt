#include <cstdio>
#include <cstring>


int n, c[26] = {0};
char str1[5005], str2[5001];

bool solve(int i) {
  int c1[26];
  for(int h = 0; h < 26; h++)
    c1[h] = c[h];
  int h = 25;
  for(;i < n; i++) {
    while(c1[h] == 0) h--;
    str1[i] = h + 'a';
    c1[h]--;
  }
  return strcmp(str1,str2) > 0;
}

int main() {
  scanf("%s%s", str1,str2);
  n = strlen(str1);
  for(int i = 0; i < n; i++)
    c[str1[i] - 'a']++;
  if(!solve(0)) printf("-1\n");
  else {
    for(int i = 0; i < n; i++) {
      for(int h = 0; h < 26; h++) 
    if(c[h] > 0) {
      c[h]--;
      str1[i] = h + 'a';
      if(solve(i + 1)) break;
      c[h]++;
    }
    }
    printf("%s\n", str1);
  }
  return 0;
}
