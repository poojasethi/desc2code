#include<cstdio>
#include<cstring>
#include<iostream>
#include<string>
using namespace std;
#define MAXN 26

string a,b,r;
int cnta[MAXN],cntb[MAXN],cntr[MAXN];

void init()
{
	cin>>a>>b;
	for(int i=0;i<a.size();i++) cnta[a[i]-'a']++;
	for(int i=0;i<b.size();i++) cntb[b[i]-'a']++;
}
void dfs(int k)
{
	for(int i=0;i<MAXN;i++) if(cnta[i]>0&&'a'+i>b[k])
	{
		r.assign(b,0,k);
		r.push_back('a'+i);
		cnta[i]--;
		memcpy(cntr,cnta,sizeof(cnta));
		cnta[i]++;
		break;
	}
	if(cnta[b[k]-'a']>0)
	{
		cnta[b[k]-'a']--;
		dfs(k+1);
	}
}
int main()
{
	init();
	dfs(0);
	if(r.size())
	{
		cout<<r;
		for(int i=0;i<MAXN;i++) while(cntr[i]-->0) putchar('a'+i);
		cout<<endl;
	}
	else cout<<-1<<endl;
	return 0;
}
