#include<cstdio>
#include<cstdlib>
#include<cstring>
int a[200], Max, i;
char s[5010], ss[5010];
int main()
{
	scanf("%s", ss);
	for (int j = 0; ss[j]; ++j) ++a[ss[j]];
	scanf("%s", s);
	for (int j = strlen(s); ss[j]; ++j) s[j] = 1;
	for (Max = 125; a[Max] == 0; --Max);
	s[strlen(s)] = 127;
	for (i = 0; s[i]; ++i)
	{
		if (s[i+1] > Max) break;
		if (a[s[i]])
		{
			--a[s[i]];
			while (a[Max] == 0 && Max > 95) --Max;
		}
		else break;
	}
	while (i >= 0)
	{
		int t = s[i]+1;
		if (t <= Max)
		{
			for (; a[t] == 0; ++t);
			--a[t];
			while (a[Max] == 0 && Max > 96) --Max;
			for (int j = 0; j < i; ++j) printf("%c", s[j]);
			printf("%c", t);
			for (int j = 97; j <= 122; ++j) for (int k = 1; k <= a[j]; ++k) printf("%c", j);
			return 0;
		}
		--i;
		++a[s[i]];
		if (s[i] > Max) Max = s[i];
	}
	printf("-1\n");
	return 0;
}
