#define MAXN 5005
char str1[MAXN],str2[MAXN];
int len,cnt[128];
int isok(int n){
  int i,j;
  int _cnt[128];
  memcpy(_cnt,cnt,sizeof(cnt));
  for(i=n,j='z';i<len;i++){
    while(_cnt[j]==0) j--;
    str1[i]=j;
    _cnt[str1[i]]--;
  }
  if(strcmp(str1,str2)>0) return 1;
  else return 0;
}
