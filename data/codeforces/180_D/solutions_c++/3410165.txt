#include<stdio.h>
#include<string.h>
#include<stdlib.h>
long l,i;
char s[5010],ans[5010];
long a[30];
void add(long x){
     for (long i=1;i<=26;i++)
         for (long j=1;j<=a[i];j++)
             ans[x++]=i+96;
}
bool dfs(long x){
     if (x>=l){add(x);return 1;}
     if ((--a[s[x]-96])>=0 && dfs(x+1)){
         ans[x]=s[x];return 1;
         }
     a[s[x]-96]++;
     long i;
     for (i=s[x]-96+1;i<=26;i++)
         if (a[i]) break;
     if (i<=26){
         ans[x]=i+96;
         a[i]--;
         add(x+1);
         return 1;
         }
     return 0;
}
int main(){
    gets(s);
    for (l=strlen(s),i=0;i<l;i++)
        a[s[i]-96]++;
    gets(s);s[strlen(s)]=96;
    l=strlen(s);
    if (!dfs(0)){printf("-1\n");return 0;}
    puts(ans);
    return 0;
}
