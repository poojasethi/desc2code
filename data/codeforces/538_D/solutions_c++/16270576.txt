#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
const int maxn=105;

char s[maxn][maxn];
bool a[maxn][maxn],b[maxn][maxn];
int n,m;

int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%s",s[i]+1);
		
	m=2*n-1;
	for(int i=1;i<=m;i++)
		for(int j=1;j<=m;j++) a[i][j]=1;
		
	for(int x=1;x<=n;x++)
		for(int y=1;y<=n;y++)
			if(s[x][y]=='o'){
				for(int i=1;i<=n;i++)
					for(int j=1;j<=n;j++)
						if(s[i][j]=='.')
							a[i-x+n][j-y+n]=0;
			}
	
	for(int x=1;x<=n;x++)
		for(int y=1;y<=n;y++)
			if(s[x][y]=='o'){
				for(int i=1;x+i<=2*n;i++)
					for(int j=1;y+j<=2*n;j++)
						if(a[i][j]&&x+i-n>0&&y+j-n>0)
							b[x+i-n][y+j-n]=1;
			}
	
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
			if(s[i][j]=='x'&&!b[i][j]){
				printf("NO\n");
				return 0;
			}
	
	printf("YES\n");
	for(int i=1;i<=m;i++){
		for(int j=1;j<=m;j++){
			if(i==n&&j==n) printf("o");
			else if(a[i][j]) printf("x");
			else printf(".");
		}
		printf("\n");
	}
	
	return 0;
}
