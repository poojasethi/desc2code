#include<bits/stdc++.h>

using namespace std;

int n,i,j,q,w,b,k,l;
char A[105][105],Ans[105][105],H[205][205];

int main(){

	cin>>n;

	for(i=1 ; i<=n ; i++)
		for(j=1 ; j<=n ; j++)
			scanf(" %c",&A[i][j]);

	for(i=1 ; i<=2*n-1 ; i++)
		for(j=1 ; j<=2*n-1 ; j++){
			
			if(i==n and j==n){
				Ans[i][j]='o';
				continue;
			}

			q=i-n;
			w=j-n;

			b=1;
			
			for(k=1 ; k<=n ; k++)
				for(l=1 ; l<=n ; l++){
					if(A[k][l]=='o' and (k+q<1 or k+q>n or w+l<1 or w+l>n or A[k+q][w+l]=='x' or A[k+q][w+l]=='o'));
					else if(A[k][l]=='o') b=0;
				}

			if(b)
				Ans[i][j]='x';
			else Ans[i][j]='.';
		}

	for(i=1 ; i<=2*n-1 ; i++)
		for(j=1 ; j<=2*n-1 ; j++)
			if(Ans[i][j]=='x'){
				
/// asdasfds

				q=i-n;
				w=j-n;


///a sdadfsfsd

				for(k=1 ; k<=n ; k++)
					for(l=1 ; l<=n ; l++){
						if(A[k][l]=='o')
							H[max(0,k+q)][max(0,l+w)]=1;
					}
			}

	for(i=1 ; i<=n ; i++)
		for(j=1 ; j<=n ; j++)
				if(A[i][j]=='x' and !H[i][j])
					return cout<<"NO"<<endl,0;

	cout<<"YES"<<endl;

	for(i=1 ; i<=2*n-1 ; i++,puts(""))
		for(j=1 ; j<=2*n-1 ; j++)
			cout<<Ans[i][j];
	
	return 0;
}
