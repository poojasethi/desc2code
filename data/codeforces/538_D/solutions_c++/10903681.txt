#include<cstdio>
#include<cstring>
#include<algorithm>
#define MAXN 110
using namespace std;
int n;
char s[MAXN][MAXN],ans[MAXN][MAXN];
bool a[MAXN][MAXN],b[MAXN][MAXN];
int main()
{
	int i,j,k,l;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%s",s[i]+1);
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(s[i][j]=='o')
				for(k=1;k<=n;k++)
					for(l=1;l<=n;l++)
						if(s[k][l]=='.')
							a[k-i+n][l-j+n]=1;
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(s[i][j]=='o')
				for(k=1;k<=n;k++)
					for(l=1;l<=n;l++)
						if(!a[k-i+n][l-j+n])
							b[k][l]=1;
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(s[i][j]=='x'&&!b[i][j])
			{
				puts("NO");
				return 0;
			}
	for(i=1;i<=2*n-1;i++)
		for(j=1;j<=2*n-1;j++)
			ans[i][j]=(a[i][j]?'.':'x');
	ans[n][n]='o';
	puts("YES");
	for(i=1;i<=2*n-1;i++)
		puts(ans[i]+1);
	return 0;
}
