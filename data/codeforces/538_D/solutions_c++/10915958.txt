#include <bits/stdc++.h>
using namespace std;

int n, tot, cnt;
char in[55][55], h[55][55], out[200][200];

bool rip(int x, int y){ return x < 0 || y < 0 || x >= n || y >= n; }

int main(){
	scanf("%d\n", &n);
	for(int r = 0; r < n; r++) scanf("%s\n", in[r]);
	for(int r = 0; r < n; r++) for(int c = 0; c < n; c++) tot += (in[r][c] == 'o');
	for(int dr = -n + 1; dr < n; dr++) for(int dc = -n + 1; dc < n; dc++) {
		cnt = 0;
		for(int r = 0; r < n; r++) for(int c = 0; c < n; c++){
			if(in[r][c] != 'o') continue;
			if(rip(r + dr, c + dc) || in[r+dr][c+dc] != '.') cnt++;
		}
		if( cnt == tot ) {
			out[dr + 55][dc + 55] = 'x';
			for(int r = 0; r < n; r++) for(int c = 0; c < n; c++)
				if(in[r][c] == 'o' && !rip(r + dr, c + dc)) h[r+dr][c+dc] = 1;
		}
	}
	for(int r = 0; r < n; r++) for(int c = 0; c < n; c++)
		if(in[r][c] == 'x' && !h[r][c]) { printf("NO"); return 0; }
	printf("YES\n");
	for(int dr = -n + 1; dr < n; dr++) {
		for(int dc = -n + 1; dc < n; dc++) {
			if(dr == 0 && dc == 0) out[dr+55][dc+55] = 'o';
			if(!out[dr+55][dc+55]) out[dr+55][dc+55] = '.';
			printf("%c", out[dr+55][dc+55]);
		}
		printf("\n");
	}
	return 0;
}
