#include<iostream>
#include<math.h>
#include<algorithm>
#include<stdio.h>
#include<map>
#include<vector>
#include<set>
#define F first
#define S second
#define P system("PAUSE");
#define H return 0;
#define pb push_back
using namespace std;
const long long A=100000000000000LL,N=10000;
                     
char a[228][228];
vector<long long> k[2],p[2];
long long i,j,ok,n,m,x,y,o[228][228],K,d[228][228];

int main(){
	cin>>n;
	K=n-1;
	for(i=0;i<n;i++)for(j=0;j<n;j++)cin>>a[i][j];	
	for(i=0;i<n;i++)for(j=0;j<n;j++)if(a[i][j]=='o')p[0].pb(i),p[1].pb(j);
	for(i=0;i<n;i++)for(j=0;j<n;j++)if(a[i][j]=='x')k[0].pb(i),k[1].pb(j);
	m=p[0].size();
	for(i=0;i<m;i++)for(x=0;x<n;x++)for(y=0;y<n;y++)if(a[x][y]=='.')d[x-p[0][i]+K][y-p[1][i]+K]=1;
	for(i=0;i<k[0].size();i++){
		ok=0;
		for(j=0;j<m;j++)if(!d[k[0][i]-p[0][j]+K][k[1][i]-p[1][j]+K]){
			o[(k[0][i]-p[0][j])+K][(k[1][i]-p[1][j])+K]=ok=1;
			break;
		}
		if(!ok){cout<<"NO";H;}
	}	
	cout<<"YES\n";
	for(i=0;i<n*2-1;i++,cout<<"\n")for(j=0;j<n*2-1;j++)if(i==n-1 && j==n-1)cout<<"o";else if(o[i][j])cout<<"x";else cout<<".";
		
}
