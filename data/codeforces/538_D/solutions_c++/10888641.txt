#include <cstdio>
#include <vector>
using namespace std;
const int N = 105;
struct node {
	int x, y;
	node(int x, int y): x(x), y(y) {}
};
vector<node> a, b, c;
char s[N][N], ans[N][N];
int n;
int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%s", s[i]);
		for (int j = 0; j < n; j++) {
			if (s[i][j] == '.') a.push_back(node(i, j));
			if (s[i][j] == 'o') b.push_back(node(i, j));
			if (s[i][j] == 'x') c.push_back(node(i, j));
		}
	}
	for (int i = 1; i < 2*n; i++)
		for (int j = 1; j < 2*n; j++) ans[i][j] = 'x';
	ans[n][n] = 'o';
	for (int i = 0; i < a.size(); i++)
		for (int j = 0; j < b.size(); j++)
			ans[n+a[i].x-b[j].x][n+a[i].y-b[j].y] = '.';
	for (int i = 0; i < c.size(); i++) {
		bool ok = false;
		for (int j = 0; j < b.size(); j++)
			if (ans[n+c[i].x-b[j].x][n+c[i].y-b[j].y] == 'x')
				ok = true;
		if (!ok) { puts("NO"); return 0; }
	}
	puts("YES");
	for (int i = 1; i < 2*n; i++) printf("%s\n", ans[i]+1);
	return 0;
}
