#include <bits/stdc++.h>
using namespace std;


typedef long long LL;
 
char brd[100][100];



char ans[200][200];
  
 
int main()
{ 
	int n,i,j,x,y,y1,x1;
	scanf("%d",&n);
	
	for(i=0;i<n;i++)scanf("%s",brd[i]); 
	 
	for(i=0;i<2*n-1;i++)
		for(j=0;j<2*n-1;j++)
		   ans[i][j]='x';

	ans[n-1][n-1]='o';
	 
	for(x=0;x<n;x++)
	  for(y=0;y<n;y++)
		if(brd[x][y]=='o')
		{
			for(x1=0;x1<n;x1++)
			  for(y1=0;y1<n;y1++)
			    if(brd[x1][y1]=='.')
			      ans[n-1+x1-x][n-1+y1-y]='.';
		}
	  
	 char OK =1;
	  
	 for(x=0;x<n;x++)
	  for(y=0;y<n;y++)
		if(brd[x][y]=='x')
		{
			char fired=0;
			
			for(x1=0;x1<n;x1++)
			  for(y1=0;y1<n;y1++)
			    if(brd[x1][y1]=='o')
			      if(ans[n-1+x-x1][n-1+y-y1]=='x')fired=1;
			      		
			if(fired==0)OK=0;
		}
	  
	 if(OK==0)
	 {
	    cout<<"NO\n";
	    return 0;	 
     } 

	 cout<<"YES\n";
	  
	 for(i=0;i<2*n-1;i++)
	 {
	   ans[i][2*n-1]=0;
	   printf("%s\n",ans[i]);	 
     } 
	
	
	
	return 0;
}
