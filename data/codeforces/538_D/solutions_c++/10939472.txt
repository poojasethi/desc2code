#include <iostream>
using namespace std;
struct node
{
	int x,y;
}o[2500];
int b[99][99];
int main()
{
	int n,i,j,ocnt=0,k,dx,dy,yes;
	string s[50];
	cin>>n;
	b[n-1][n-1]=2;
	for (i=0;i<n;i++)
	{
		cin>>s[i];
		for (j=0;j<n;j++)
		{
			if (s[i][j]=='o')
			{
				o[ocnt].x=i;
				o[ocnt].y=j;
				ocnt++;
			}
		}
	}
	for (i=0;i<2*n-1;i++)
		for (j=0;j<2*n-1;j++)
		{
			if (b[i][j]==2) continue;
			dx=i-n+1;
			dy=j-n+1;
			for (k=0;k<ocnt;k++)
			{
				if (o[k].x+dx<0||o[k].x+dx>=n||o[k].y+dy<0||o[k].y+dy>=n) continue;
				if (s[o[k].x+dx][o[k].y+dy]=='.') break;
			}
			if (k==ocnt) b[i][j]=1;
		}
	for (i=0;i<n;i++)
		for (j=0;j<n;j++)
		{
			if (s[i][j]!='x') continue;
			for (k=0;k<ocnt;k++)
			{
				dx=i-o[k].x;
				dy=j-o[k].y;
				if (n-1+dx<0||n-1+dx>=2*n-1||n-1+dy<0||n-1+dy>=2*n-1) continue;
				if (b[n-1+dx][n-1+dy]==1) break;
			}
			if (k==ocnt) {cout<<"NO"<<endl;return 0;}
		}
	cout<<"YES"<<endl;
	for (i=0;i<2*n-1;i++)
	{
		for (j=0;j<2*n-1;j++)
			if (b[i][j]==0) cout<<'.';
			else if (b[i][j]==1) cout<<'x';
			else cout<<'o';
		cout<<endl;
	}
	return 0;
}
