#include<bits/stdc++.h>
using namespace std;

int n ;
char a[101][101] ;
int f[101][101] , sig ;
#define pii pair<int,int> 
vector<pii> S[101][101] ;

int fix(int x ,int y) {
  for(int i=0 ; i<n ; i++) for(int j=0 ; j<n ; j++) {
    if(a[i][j]=='.') f[i+x][j+y]=0 ; 
    if(a[i][j]=='x') S[i][j].push_back(pii(i+x,j+y)) ; 
  }  
  
  for(int i=0 ; i<n ; i++) for(int j=0 ; j<n ; j++) if(a[i][j]=='x') {
    int k;
    for(k=0 ; k<S[i][j].size() ; k++) {
       int u=S[i][j][k].first , v=S[i][j][k].second ; 
       if(!f[u][v]) continue ;
       f[u][v]=1 ; break ;
    }
    if(k==S[i][j].size()) return 1 ; 
  }

  return 0 ;    
}

int main() {
  cin >> n ; 

  memset(f,-1,sizeof(f)) ; 
  for(int i=0 ; i<n ; i++) cin >> a[i] ;
  for(int i=0 ; i<n ; i++) for(int j=0 ; j<n ; j++) 
    if(a[i][j]=='o') sig=fix(n-i-1,n-j-1) ;        
  
  if(sig) { cout << "NO" ; return 0 ; }
  cout << "YES\n" ;
  for(int i=0 ; i<2*n-1 ; i++,puts("")) for(int j=0 ; j<2*n-1 ; j++) {
    if(i==n-1 && j==n-1) cout << "o" ;
    else if(f[i][j]<=0) cout << "." ;
    else cout << "x" ; 
  }
  
  return 0;
}
