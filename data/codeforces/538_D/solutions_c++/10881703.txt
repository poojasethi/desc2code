#include <cstdio>
#include <algorithm>
#define N 55
#define fi(a, b, c) for(int a = (b); a < (c); a++)
#define fd(a, b, c) for(int a = (b); a > (c); a--)
#define FI(a, b, c) for(int a = (b); a <= (c); a++)
#define FD(a, b, c) for(int a = (b); a >= (c); a--)
#define fe(a, b, c) for(int a = (b); a; a = c[a])
using namespace std;

int n, x[N * N], y[N * N], m;
char s[N][N], v[N + N][N + N];
bool pos;

int main(){
	scanf("%d", &n);
	fi(i, 0, n){
		scanf("%s", s[i]);
		fi(j, 0, n) if(s[i][j] == 'o'){
			y[m] = i;
			x[m++] = j;
		}
	}
	fi(i, 0, n + n - 1) fi(j, 0, n + n - 1){
		bool ok = 1;
		fi(k, 0, m){
			int ny = y[k] + i - n + 1, nx = x[k] + j - n + 1;
			if(ny < 0 || ny >= n || nx < 0 || nx >= n) continue;
			if(s[ny][nx] == '.') ok = 0;
		}
		v[i][j] = ok ? 'x' : '.';
	}
	
	pos = 1;
	fi(i, 0, n) fi(j, 0, n) if(s[i][j] == 'x'){
		bool ok = 0;
		fi(k, 0, m) if(v[i - y[k] + n - 1][j - x[k] + n - 1] == 'x') ok = 1;
		if(!ok) pos = 0;
	}
	v[n - 1][n - 1] = 'o';
	
	if(pos){
		puts("YES");
		fi(i, 0, n + n - 1) puts(v[i]);
	}else puts("NO");
	scanf("\n");
}
