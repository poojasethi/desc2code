#include <cstdio>

using namespace std;

char board[51][51], res[102][102];;

int main()
{
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) scanf("%s", board[i]);
	for (int i = -n + 1; i < n; ++i)
	{
		for (int j = -n + 1; j < n; ++j)
		{
			bool move = true;
			for (int k = 0; k < n && move; ++k)
			{
				for (int l = 0; l < n && move; ++l)
				{
					if (k + i < 0 || k + i >= n || l + j < 0 || l + j >= n) continue;
					if (board[k][l] != 'o') continue;
					if (board[k + i][l + j] == '.') move = false;
				}
			}
			res[i + n - 1][j + n - 1] = (move ? 'x' : '.');
			if (move)
			{
				for (int k = 0; k < n; ++k)
				{
					for (int l = 0; l < n; ++l)
					{
						if (k + i < 0 || k + i >= n || l + j < 0 || l + j >= n) continue;
						if (board[k][l] != 'o') continue;
						if (board[k + i][l + j] == 'x') board[k + i][l + j] = 'm';
					}
				}
			}
		}
		res[i + n - 1][2 * n - 1] = 0;
	}
	res[n - 1][n - 1] = 'o';
	for (int i = 0; i < n; ++i) for (int j = 0; j < n; ++j) if (board[i][j] == 'x')
	{
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	for (int i = 0; i < 2 * n - 1; ++i) printf("%s\n", res[i]);
	return 0;
}
