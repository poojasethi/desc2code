#include <bits/stdc++.h>

using namespace std;
const int N = 105;

int n;
char c[N][N], a[N][N], s[N][N];

int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);
	cin >> n;
	for (int i = 0; i < n; ++i) cin >> c[i];
	memset(a, 'x', sizeof a);
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < n; ++j)
			if (c[i][j] == 'o')
				for (int k = 0; k < n; ++k)
					for (int l = 0; l < n; ++l)
						if (c[k][l] == '.')
							a[n-1+k-i][n-1+l-j] = '.';
	a[n - 1][n - 1] = 'o';
	// check if overlay is correct
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < n; ++j)
			if (c[i][j] == 'o')
				for (int k = 0; k < n; ++k)
					for (int l = 0; l < n; ++l)
						if (a[n-1+k-i][n-1+l-j] == 'x')
							s[k][l] = 'x';
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < n; ++j) {
			if (s[i][j] == 0) {
				if (c[i][j] != 'x')
					s[i][j] = c[i][j];
				else
					s[i][j] = '.';
			}
			if (c[i][j] != 'o' && (s[i][j] == 'x') != (c[i][j] == 'x')) {
				cout << "NO" << endl;
				return 0;
			}
		}
	}
	cout << "YES" << '\n';
	for (int i = 0; i < 2*n - 1; ++i) {
		a[i][2*n - 1] = '\0';
		cout << a[i] << '\n';
	}
	cout << flush;
}
