#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
const int N = 500;
const int T = 250;

int n, a[N][N], b[N][N];
char mp[N][N];
int i, j, k, l;

inline bool in(int x, int y) {
	return 1 <= x && x <= n && 1 <= y && y <= n;
}

int main() {
	scanf("%d\n", &n);
	for (i = 1; i <= n; ++i) gets(mp[i] + 1);
	for (k = -n; k <= n; ++k)
		for (l = -n; l <= n; ++l)
			a[k + T][l + T] = 1;
	for (i = 1; i <= n; ++i)
		for (j = 1; j <= n; ++j) if (mp[i][j] == 'o') {
			for (k = -n; k <= n; ++k)
				for (l = -n; l <= n; ++l)
					if (in(i + k, j + l))
						if (mp[i + k][j + l] == '.') a[k + T][l + T] = 0;
		}
	for (i = 1; i <= n; ++i)
		for (j = 1; j <= n; ++j) if (mp[i][j] == 'o') {
			for (k = -n; k <= n; ++k)
				for (l = -n; l <= n; ++l)
					if (in(i + k, j + l) && a[k + T][l + T]) b[i + k][j + l] = 1;
		}
	for (i = 1; i <= n; ++i)
		for (j = 1; j <= n; ++j)
			if (!b[i][j] && mp[i][j] == 'x') {
				puts("NO");
				return 0;
			}
	puts("YES");
	for (i = -n + 1; i < n; ++i) {
		for (j = - n + 1; j < n; ++j) {
			if (i == 0 && j == 0) putchar('o'); else
			if (a[i + T][j + T] != 0) putchar('x'); else
			putchar('.');
		}
		puts("");
	}
	return 0;
}
