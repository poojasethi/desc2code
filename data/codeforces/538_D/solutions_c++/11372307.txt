#include <cstdio>
#include <cstring>

#define N 105

char s[N][N], ans[N][N];
int n, flag[N][N];
int x[N * N], y[N * N], m;

int main() {
	scanf("%d", &n);
	for(int i = 0; i < n; i ++) scanf("%s",s[i]);
	for(int i = 0; i < n; i ++) for(int j = 0; j < n; j ++) if(s[i][j] == 'o') x[m] = i, y[m ++] = j;
	for(int i = 0; i < n; i ++) for(int j = 0; j < n; j ++) if(s[i][j] == '.') {
		for(int k = 0; k < m; k ++) {
			int dx = i - x[k], dy = j - y[k];
			flag[dx + n][dy + n] = 1;
		}
	}
	for(int i = 0; i < n; i ++) for(int j = 0; j < n; j ++) if(s[i][j] == 'x') {
		int k;
		for(k = 0; k < m; k ++) {
			int dx = i - x[k], dy = j - y[k];
			if(!flag[dx + n][dy + n]) break;
		}
		if(k == m) { puts("NO"); return 0;}
	}
	puts("YES");
	for(int i = 0; i < 2 * n - 1; i ++) {
		for(int j = 0; j < 2 * n - 1; j ++) {
			ans[i][j] = '.';
			int dx = i - n + 1, dy = j - n + 1;
			if(!flag[dx + n][dy + n]) ans[i][j] = 'x';
		}
		ans[i][2 * n - 1] = 0;
	}
	ans[n-1][n-1] = 'o';
	for(int i = 0; i < 2 * n - 1; i ++) puts(ans[i]);
	return 0;
}
