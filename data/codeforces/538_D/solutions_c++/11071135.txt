//Brute force
#include<stdio.h>
char ans [100][100];
int n;
char data [52][52];


int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%s",data[i]);
    for(int i=0;i<2*n-1;i++)    for(int j=0;j<2*n-1;j++)    ans[i][j] = 'x';
    ans [n-1][n-1] = 'o';
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            if(data[i][j]=='o')
                for(int k=0;k<n;k++)
                    for(int l=0;l<n;l++)
                        if(data[k][l]=='.')
                            ans[n-1+k-i][n-1+l-j] = '.';
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            if(data[i][j]=='x')
    {
        bool good = false;
        for(int k=0;k<n;k++)
            for(int l=0;l<n;l++)
                if(data[k][l]=='o' && ans[n-1+i-k][n-1+j-l]=='x')   good = true;
        if(!good)   {printf("NO"); return 0;}
    }
    printf("YES\n");
    for(int i=0;i<2*n-1;i++)
        printf("%s\n",ans[i]);
    return 0;
}
