#include<iostream>
#include<cstdio>
using namespace std;
int n;
char mp1[55][55], ans[105][105];
bool vis[55][55];
bool ok(int d1,int d2)
{
	bool you = false;
	for(int i = 1; i <= n; ++i)
		for(int j = 1; j <= n; ++j)
			if(mp1[i][j] == 'o')
			{
				if(i + d1 < 1 || i + d1 > n || j + d2 < 1 || j + d2 > n ) ;
				else if(mp1[i + d1][j + d2] != '.') you = true;
				else return false;
			}
	for(int i = 1; i <= n; ++i)
		for(int j = 1; j <= n; ++j)
			if(mp1[i][j] == 'o')
			{
				if(i + d1 < 1 || i + d1 > n || j + d2 < 1 || j + d2 > n);
				else vis[i + d1][j + d2] = 1;
			}
	return you;
}
int main()
{
	scanf("%d",&n);
	for(int i = 1; i <= n; ++i)
		scanf("%s", mp1[i] + 1);
	ans[n][n] = 'o';
	int ier = n * 2 - 1;
	for(int i = 1; i <= ier; ++i)
		for(int j = 1; j <= ier; ++j)
			if(i == n && j == n);
			else if(ok(i - n, j - n)) ans[i][j] = 'x';
			else ans[i][j] = '.';
	for(int i = 1; i <= n; ++i)
		for(int j = 1; j <= n; ++j)
			if(mp1[i][j] == 'x' && !vis[i][j])
			{
				printf("NO\n");
				return 0;
			}
	printf("YES\n");
	for(int i = 1; i <= ier; ++i) printf("%s\n", ans[i] + 1);
	return 0;
}
