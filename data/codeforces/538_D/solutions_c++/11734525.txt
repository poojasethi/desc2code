#define N (1<<7)
#include <bits/stdc++.h>

using namespace std;

int n;
char s[N][N];
bool f[N][N],g[N][N];

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>s[i];

	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
		{
			if(s[i][j]!='o') continue;
			for(int k=0;k<n;k++)
				for(int r=0;r<n;r++)
					if(s[k][r]=='.')
						f[k-i+n][r-j+n]=1;
		}

	bool flag=0;
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
		{
			if(s[i][j]!='o') continue;
			for(int k=0;k<n;k++)
				for(int r=0;r<n;r++)
					if(!f[k-i+n][r-j+n]) g[k][r]=1;
		}

	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			if(!g[i][j] && s[i][j]=='x') flag=1;

	if(flag)
	{
		puts("NO");
		return 0;
	}

	puts("YES");
	for(int i=1;i<n+n;i++)
	{
		for(int j=1;j<n+n;j++)
		{
			if(i==n && j==n) putchar('o');
			else putchar(f[i][j]?'.':'x');
		}
		cout<<endl;
	}
}
