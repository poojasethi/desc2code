#define wez(n) int (n); scanf("%d",&(n));

using namespace std;

const int N=110;
const ll P=1e9+7, INF=2*1e9;

int n;
string board[N],res[N];

int main(){
    cin>>n;
    for(int i=0;i<n;i++) cin>>board[i];
    for(int i=0;i<2*n-1;i++) res[0]+='x';
    for(int i=1;i<2*n-1;i++) res[i]=res[0];
    res[n-1][n-1]='o';
    for(int i=0;i<n;i++) for(int j=0;j<n;j++) if(board[i][j]=='o'){
        for(int ii=0;ii<n;ii++) for(int jj=0;jj<n;jj++) if(board[ii][jj]=='.'){
            res[n-1+ii-i][n-1+jj-j]='.';
        }
    }
    for(int ii=0;ii<n;ii++) for(int jj=0;jj<n;jj++) if(board[ii][jj]=='x'){
        int ok=0;
        for(int i=0;i<n;i++) for(int j=0;j<n;j++) if(board[i][j]=='o' && res[n-1+ii-i][n-1+jj-j]=='x'){
            ok=1;
        }
        if(!ok){
            cout<<"NO";
            return 0;
        }
    }
    cout<<"YES\n";
    for(int i=0;i<2*n-1;i++) cout<<res[i]<<"\n";
}
