#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	char a[55][55];
	char ans[150][150];
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<2*n;i++)
		for(int j=0;j<2*n;j++)
			ans[i][j]='x';
	ans[n][n]='o';
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			if(a[i][j]=='o')
				for(int p=0;p<n;p++)
					for(int q=0;q<n;q++)
						if(a[p][q]=='.')
							ans[n+p-i][n+q-j]='.';
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			if(a[i][j]=='x')
			{
				int state=0;
				for(int p=0;p<n;p++)
					for(int q=0;q<n;q++)
						if(a[p][q]=='o' && ans[n+i-p][n+j-q]=='x')
							state=1;
				if(state==0)
				{
					cout<<"NO\n";
					return 0;
				}
			}
	cout<<"YES\n";
	for(int i=1;i<2*n;i++)
	{
		for(int j=1;j<2*n;j++)
			cout<<ans[i][j];
		cout<<endl;
	}
	return 0;
}
