#include <bits/stdc++.h>

using namespace std;

int n, m;

char s[55][55];
bool b[110][110];

typedef pair<int, int> ii;
vector< ii > p;

int main() {
	scanf("%d", &n);
	for(int i = 0; i<n; i++) {
		scanf("%s", s[i]);
		for(int j = 0; j<n; j++)
			if(s[i][j] == 'o')
				p.push_back(ii(n-1-i, n-1-j));
	}
	memset(b, true, sizeof b);
	for(int i = 0; i<n; i++)
		for(int j = 0; j<n; j++) {
			if(s[i][j] == '.') {
				for(int k = 0; k<p.size(); k++)
					b[i+p[k].first][j+p[k].second] = false;
			}
		}

	for(int i = 0; i<n; i++)
		for(int j = 0; j<n; j++) {
			if(s[i][j] == 'x') {
				bool found = false;
				for(int k = 0; k<p.size(); k++)
					if(b[i+p[k].first][j+p[k].second]) {
						found = true;
						break;
					}
				if(!found) {
					puts("NO");
					return 0;
				}
			}
		}
	puts("YES");
	for(int i = 0; i<2*n-1; i++) {
		for(int j = 0; j<2*n-1; j++) {
			if(i == n-1 && j == n-1)
				printf("o");
			else
				printf("%c", b[i][j] ? 'x' : '.');
		}
		puts("");
	}
	return 0;
}
