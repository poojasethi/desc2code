#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int N=55;

int n;

char a[N][N];
char b[N*2][N*2];

int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%s",a[i]+1);
	for(int i=0;i<=n*2;i++) for(int j=0;j<=n*2;j++) b[i][j]='x';
	b[n][n]='o';
	for(int i=1;i<=n;i++) for(int j=1;j<=n;j++) if(a[i][j]=='o')
		for(int x=1;x<=n;x++) for(int y=1;y<=n;y++) if(a[x][y]=='.')
			b[n+x-i][n+y-j]='.';
	for(int i=1;i<=n;i++) for(int j=1;j<=n;j++) if(a[i][j]=='o')
		for(int x=1;x<=n;x++) for(int y=1;y<=n;y++) if(b[n+x-i][n+y-j]=='x')
			if(a[x][y]=='x') a[x][y]='.';
	for(int i=1;i<=n;i++) for(int j=1;j<=n;j++) if(a[i][j]=='x'){
		puts("NO"); return 0;
	}
	puts("YES");
	for(int i=1;i<=n*2-1;i++,puts(""))
		for(int j=1;j<=n*2-1;putchar(b[i][j++]));
	scanf("\n");
}
