#include<stdio.h>

int main(void){
	int n;
	char B[100][100];
	int i,j,k,l,ans;
	char A[200][200],C[100][100];

	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%s",B[i]);
	}
	
	for(i=0;i<2*n-1;i++){
		for(j=0;j<2*n-1;j++){
			A[i][j] = 'x';
		}
	}
	
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			if(B[i][j]=='o'){
				for(k=0;k<n;k++){
					for(l=0;l<n;l++){
						if(B[k][l]=='.'){
							A[n-1+k-i][n-1+l-j] = '.';
						}
					}
				}
			}
		}
	}

	A[n-1][n-1] = 'o';


	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			if(B[i][j] == 'o'){
				C[i][j] = 'o';
			}else{
				C[i][j] = '.';
			}
		}
	}

	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			if(C[i][j] == 'o'){
				for(k=0;k<n;k++){
					for(l=0;l<n;l++){
						if(C[k][l] == '.' && A[n-1+k-i][n-1+l-j] == 'x'){
							C[k][l] = 'x';
						}
					}
				}
			}
		}
	}
	ans = 1;
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			if(B[i][j] != C[i][j]){
				ans = 0;
			}
		}
	}

	if(ans == 0){
		printf("NO\n");
	}else{
		printf("YES\n");
		for(i=0;i<2*n-1;i++){
			printf("%s\n",A[i]);
		}
	}

}
