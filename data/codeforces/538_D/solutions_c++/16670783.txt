#include<stdio.h>
#include<string.h>
#include<algorithm>
#define N 110
using namespace std;
typedef long long ll;
char str[N][N];
char ans[N][N];
bool vis[N][N];
int X[N*N],Y[N*N],tot;
int main()
{
	int n,m,i,j,k;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%s",str[i]+1);
		for(j=1;j<=n;j++)
		{
			if(str[i][j]=='o')
			{
				tot++;
				X[tot]=i;
				Y[tot]=j;
			}
		}
	}
	memset(ans,'.',sizeof(ans));
	ans[n][n]='o';
	for(i=1;i<=2*n-1;i++)
	{
		for(j=1;j<=2*n-1;j++)
		{
			if(i==n&&j==n)	continue;
			bool flag=1;
			for(k=1;k<=tot;k++)
			{
				if(1<=X[k]+i-n&&X[k]+i-n<=n&&1<=Y[k]+j-n&&Y[k]+j-n<=n)
				{
					if(str[X[k]+i-n][Y[k]+j-n]=='.')
						flag=0;
				}
			}
			if(flag==1)
			{
				ans[i][j]='x';
				for(k=1;k<=tot;k++)
				{
					if(1<=X[k]+i-n&&X[k]+i-n<=n&&1<=Y[k]+j-n&&Y[k]+j-n<=n)
					{
						vis[X[k]+i-n][Y[k]+j-n]=1;
					}
				}
			}
		}
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			if(str[i][j]=='x'&&!vis[i][j])
			{
				puts("NO");
				return 0;
			}
		}
	}
	puts("YES");
	for(i=1;i<=2*n-1;i++)
	{
		ans[i][2*n]='\0';
		printf("%s\n",ans[i]+1);
	}
	return 0;
}
