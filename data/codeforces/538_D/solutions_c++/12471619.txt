#include <iostream>
using namespace std;
#define FOR(i, to, from) for (int i=to; i<from; i++)

string board[50];
string att[50];
bool moves[100][100]={};
int n;

void mov(int x, int y) {
	FOR(i, 0, n)
	FOR(j, 0, n)
	if (board[i][j]=='.') {
		moves[n+i-x-1][n+j-y-1]=true;
	}
}

bool esAlcanzada (int i, int j) {
	FOR(x, 0, n)
	FOR(y, 0, n)
	if (board[x][y]=='o' && !moves[n+i-x-1][n+j-y-1]) return true;
	return false;
}

int main () {
	cin >> n;
	FOR(i, 0, n)
		cin >> board[i];
	
	FOR(i, 0, n)
	FOR(j, 0, n)
		if (board[i][j]=='o') mov(i, j);
	
	FOR(i, 0, n)
	FOR(j, 0, n)
		if (board[i][j]=='x' && !esAlcanzada(i, j)) {
			cout << "NO" << endl;
			return 0;
		}
	cout << "YES" << endl;
	FOR(i, 0, 2*n-1) {
		FOR(j, 0, 2*n-1) {
			if (i==n-1 && j==n-1)
				cout << 'o';
			else if (moves[i][j])
				cout << '.';
			else
				cout << 'x';
		}
		cout << endl;
	}
}
