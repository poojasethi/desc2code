#include<iostream>
#include<vector>
#include<set>
#include<queue>
#include<algorithm>
#include<map>
#include<string>

using namespace std;
#define REP(i,n) for(int i=0;i<(int)n;++i)

int main() {
  int n;cin>>n;
  vector<string> s(n);REP(i,n) cin>>s[i];
  vector<pair<int, int> > forbiden;
  REP(i,n) REP(j,n) REP(y,n) REP(x,n) if(i!=y || j!=x) {
    if(s[i][j]=='o' && s[y][x]=='.') forbiden.push_back(make_pair(y-i, x-j)); 
  }
  sort(forbiden.begin(), forbiden.end());
  forbiden.erase(unique(forbiden.begin(), forbiden.end()),forbiden.end());
  int ok=1;
  vector<string> o(2*n-1, string(2*n-1,'.'));
  o[n-1][n-1]='o';
  REP(i,n) if(ok)REP(j,n) if(s[i][j]=='x') {
    int w=1;
    REP(y,n)if(w) REP(x,n) if(s[y][x]=='o' && w) {
      pair<int,int> v(i-y,j-x);
      int z=lower_bound(forbiden.begin(), forbiden.end(), v)-forbiden.begin();
      if(z==(int)forbiden.size()|| forbiden[z]!=v) {
        w=0;
        o[n-1+v.first][n-1+v.second]='x';
        break;
      }
    }
    if(w==1) {ok=0;break;}

  }

  if(!ok){
    cout<<"NO"<<endl;
  } else {
    cout<<"YES"<<endl;
    REP(i,o.size()) cout<<o[i]<<endl;
  }
}
