#include <iostream>
#include <vector>
using namespace std;
vector< pair<int, int> > p;
int main() {
	int n; cin >> n;
	char s[n][n], S[2 * n - 1][2 * n - 1];
	for(int i = 0; i < 2 * n - 1; ++i)
		for(int j = 0; j < 2 * n - 1; ++j) S[i][j] = '.';
	S[n - 1][n - 1] = 'o';
	for(int i = 0; i < n; ++i)
		for(int j = 0; j < n; ++j) {
			cin >> s[i][j];
			if(s[i][j] == 'o') p.push_back(make_pair(i, j));
		}
	for(int i = 0; i < n; ++i)
		for(int j = 0; j < n; ++j)
			if(s[i][j] == 'x')
				for(int k = 0; k < p.size(); ++k) {
					int dx = i - p[k].first, dy = j - p[k].second, ok = 1;
					for(int o = 0; o < p.size(); ++o) {
						int x = p[o].first + dx, y = p[o].second + dy;
						if(x >= 0 && y >= 0 && y < n && x < n && s[x][y] == '.') ok = 0;
					}
					if(ok) {
						S[n - 1 + dx][n - 1 + dy] = 'x';
						s[i][j] = 'd';
						break;
					}
				}
	for(int i = 0; i < n; ++i)
		for(int j = 0; j < n; ++j)
			if(s[i][j] == 'x') { cout << "NO"; return 0; }
	cout << "YES" << endl;
	for(int i = 0; i < 2 * n - 1; ++i) {
		for(int j = 0; j < 2 * n - 1; ++j) cout << S[i][j];
			cout << endl;
	}
}
