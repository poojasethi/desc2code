#include <iostream>
using namespace std;
#define REP(i,n) for(int i = 0; i < (n); ++i)

int main()
{
  int n; cin>>n;
  char t[n][n];
  char s[2*n][2*n];
  REP(i,2*n-1) REP(j,2*n-1) s[i][j] = 'x';
  s[n-1][n-1] = 'o';
  for(int i=0; i<n; i++) for(int j=0; j<n; j++) cin>>t[i][j];
  for(int i=0; i<n; i++) for(int j=0; j<n; j++) if(t[i][j] == 'o')
  {
    REP(it,n) REP(jt,n) if(t[it][jt] == '.') s[n-1+it-i][n-1+jt-j] = '.';
  }
  REP(i,n) REP(j,n) if(t[i][j] == 'x')
  {
    bool any = false;
    REP(it,n) REP(jt,n) if(t[it][jt] == 'o' && s[n-1+i-it][n-1+j-jt] == 'x') any = true;
    if(any == false) {cout<<"NO\n"; return 0;}
  }
  cout<<"YES\n";
  REP(i,2*n-1){REP(j,2*n-1) cout<<s[i][j]; cout<<"\n";}
  return 0;
}
