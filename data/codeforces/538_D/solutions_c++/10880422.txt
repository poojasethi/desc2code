#include <bits/stdc++.h>
using namespace std;
#define fo(i,a,b) for (int i = (a); i < (b); i++)
#define N 123

int n;
char gr[N][N], ans[N][N], cov[N][N];
bool ingr(int y, int x) {
	return y >= 0 && y < n && x >= 0 && x < n;
}
int main() {
	scanf("%d", &n);
	fo(i,0,n) scanf("%s", gr[i]);
	fo(i,0,2*n-1) fo(j,0,2*n-1) ans[i][j] = '.';
	ans[n-1][n-1] = 'o';
	fo(dy,-n,n) fo(dx,-n,n) if (dy || dx) {
		fo(i,0,n) fo(j,0,n) if (gr[i][j]=='o') {
			if (ingr(i + dy, j + dx) && gr[i + dy][j + dx] == '.') goto bad;
		}
		ans[n-1 + dy][n-1 + dx] = 'x';
		fo(i,0,n) fo(j,0,n) if (gr[i][j] == 'o' && ingr(i + dy, j + dx)) cov[i+dy][j+dx] = 1;
		bad:;
	}
	fo(i,0,n) fo(j,0,n) if (gr[i][j] == 'x' && !cov[i][j]) {
		puts("NO"); return 0;
	}
	puts("YES");
	fo(i,0,2*n-1) puts(ans[i]);

	return 0;
}
