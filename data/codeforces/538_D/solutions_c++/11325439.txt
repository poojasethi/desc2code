#include <iostream>
#include <vector>

using namespace std;

int n;
char m[200][200];
char ans[200][200];
vector<int> oi;
vector<int> oj;

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
		{
			cin>>m[i][j];
			if(m[i][j]=='o')
			{
				oi.push_back(i);
				oj.push_back(j);
			}
		}

	for(int i=0;i<2*n-1;i++)
	{
		for(int j=0;j<2*n-1;j++)
		{
			int ii=n-1-i;
			int jj=n-1-j;
			if(i==n-1&&j==n-1)
			{
				ans[i][j]='o';
				continue;
			}

			for(int k=0;k<oi.size();k++)
			{
				if(oi[k]-ii>=0&&oj[k]-jj>=0&&m[oi[k]-ii][oj[k]-jj]=='.')
				{
					ans[n-1-ii][n-1-jj]='.';
					break;
				}
			}

			if (ans[n-1-ii][n-1-jj]!='.')
			{
				ans[n-1-ii][n-1-jj]='x';
				for(int k=0;k<oi.size();k++)
				{
					if(oi[k]-ii>=0&&oj[k]-jj>=0&&m[oi[k]-ii][oj[k]-jj]!='!')
					{
						m[oi[k]-ii][oj[k]-jj]='!';
					}
				}
			}
		}
	}
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
		{
			if(m[i][j]=='x')
			{
				cout<<"NO"<<endl;
				return 0;
			}
		}
	cout<<"YES"<<endl;
	for(int i=0;i<2*n-1;i++, cout<<endl)
	{
		for(int j=0;j<2*n-1;j++)
			cout<<ans[i][j];
	}
}
