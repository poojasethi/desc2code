#include <stdio.h>
#include <string.h>
const int N=64;
int main(){
  int n;
  char g[N][N],a[N*2][N*2];
  scanf("%d",&n);
  for(int i=0;i<n;++i)scanf("%s",g[i]);
  memset(a,'x',sizeof(a));
  a[n][n]='o';
  for(int i=0;i<n;++i)for(int j=0;j<n;++j)if('o'==g[i][j])for(int u=0;u<n;++u)for(int v=0;v<n;++v)
    if('.'==g[u][v])a[n+u-i][n+v-j]='.';
  for(int i=0;i<n;++i)for(int j=0;j<n;++j)if('x'==g[i][j]){
    bool any=false;
    for(int u=0;u<n;++u)for(int v=0;v<n;++v)if('o'==g[u][v]&&'x'==a[n+i-u][n+j-v])any=true;
    if(!any){puts("NO");return 0;}
  }
  puts("YES");
  for(int i=1;i<n*2;++i)printf("%.*s\n",n*2-1,a[i]+1);
}
