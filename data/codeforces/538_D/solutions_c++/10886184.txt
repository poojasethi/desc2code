#include<iostream>
using namespace std;
#include<string.h>
int n,chk[110][110];
char a[60][60];
int main()
{
	int i,j,x,y,flag;
	cin>>n;
	for(i=1;i<=n;++i)cin>>a[i]+1;
	for(i=1;i<2*n;++i)for(j=1;j<2*n;++j)chk[i][j]=1;
	for(i=1;i<=n;++i)for(j=1;j<=n;++j)if(a[i][j]=='o')for(x=1;x<=n;++x)for(y=1;y<=n;++y)if(a[x][y]=='.')chk[x-i+n][y-j+n]=0;
	for(i=1;i<=n;++i)
	{
		for(j=1;j<=n;++j)
		{
			if(a[i][j]=='x')
			{
				flag=0;
				for(x=1;x<=n;++x)for(y=1;y<=n;++y)if(a[x][y]=='o'&&chk[i-x+n][j-y+n])flag=1;
				if(!flag)
				{
					cout<<"NO";
					return 0;
				}
			}
		}
	}
	cout<<"YES"<<endl;
	for(i=1;i<2*n;++i)
	{
		for(j=1;j<2*n;++j)	
		{
			if(i==n&&j==n)cout<<"o";
			else if(chk[i][j])cout<<"x";
			else cout<<".";
		}
		cout<<endl;
	}
	return 0;
}
