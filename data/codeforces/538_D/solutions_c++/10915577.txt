#include <bits/stdc++.h>
using namespace std;
int n;
char s[111][111],r[111][111];
int main() 
{
  cin>>n;
  for(int i=0;i<n;i++) 
  {
  	for(int j=0;j<n;j++)
  	cin>>s[i][j];
  }
  for(int i=0;i<2*n-1;i++) 
  	for(int j=0; j<2*n-1; j++) 
	  r[i][j]='x';
  r[n-1][n-1]='o';
  for (int i=0; i<n; i++)
   for (int j=0; j<n; j++)
    if (s[i][j]=='o')
   	 for (int x=0; x<n; x++)
	  for (int y=0; y<n; y++) 
	   if (s[x][y]=='.')
      	r[n-1+x-i][n-1+y-j]='.';
  for (int x=0; x<n; x++) 
  for (int y=0; y<n; y++)
   if (s[x][y]=='x') 
   {
    bool ok=false;
    for (int i=0; i<n; i++) 
	for (int j=0; j<n; j++)
	 if(s[i][j]=='o' && r[n-1+x-i][n-1+y-j]=='x')
	 ok=true;
    if (!ok)
	{
	 	puts("NO"); 
		 return 0;
	}
  }
  puts("YES");
  for (int i=0; i<2*n-1; i++)
  {
   for(int j=0;j<2*n-1;j++)
   cout<<r[i][j];
   cout<<endl;
  }
  return 0;
}
