#include <iostream>
#include <cstdio>
#include <string>
#include <vector>
using namespace std;

int n;
string inp[110];
string ans[110];

int main() {
    while (cin >> n) {
        for (int i = 0; i < n; ++i) cin >> inp[i];
        for (int i = 0; i < n*2-1; ++i) {ans[i] = ""; for (int j = 0; j < n*2-1; ++j) ans[i] += 'x';}
        ans[n-1][n-1] = 'o';
        for (int i = 0; i < n; ++i) for (int j = 0; j < n; ++j) 
            if (inp[i][j] == 'o')
                for (int k = 0; k < n; ++k) for (int l = 0; l < n; ++l)
                    if (inp[k][l] == '.')
                        ans[n-1 + k-i][n-1 + l-j] = '.';
        bool ok = true;
        for (int i = 0; i < n; ++i) for (int j = 0; j < n; ++j) 
            if (inp[i][j] == 'x') {
                bool ok2 = false;
                for (int k = 0; k < n; ++k) for (int l = 0; l < n; ++l)
                    if (inp[k][l] == 'o' && ans[n-1+i-k][n-1+j-l] == 'x') ok2 = true;
                if (!ok2) ok = false; 
            }
        if (ok) {
            puts("YES");
            for (int i = 0; i < n*2-1; ++i) cout << ans[i] << endl;
        }
        else puts("NO");
    }
}
