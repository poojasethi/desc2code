#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
char a[50][51],s[99][99];
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t,c,v;
  scanf("%d",&q);
  for(w=0;w<q*2-1;w++)
   for(e=0;e<q*2-1;e++)
    s[w][e]='x';
  s[q-1][q-1]='o';
  for(w=0;w<q;w++)
   cin>>a[w];
  for(e=0;e<q;e++)
   for(r=0;r<q;r++)
    if(a[e][r]=='o')
     for(c=0;c<q;c++)
      for(v=0;v<q;v++)
       if(a[c][v]=='.')
        s[q-1+c-e][q-1+v-r]='.';
  bool b;
  for(e=0;e<q;e++)
   for(r=0;r<q;r++)
    if(a[e][r]=='x'){
     b=0;
     for(c=0;c<q;c++)
      for(v=0;v<q;v++)
       if((a[c][v]=='o')&&(s[q-1+e-c][q-1+r-v]=='x'))
        b=1;
     if(!b){
      cout<<"NO";
      return 0;}}
  cout<<"YES";
  for(e=0;e<q*2-1;e++){
   cout<<"\n";
   for(r=0;r<q*2-1;r++)
    cout<<s[e][r];}
  return 0;}
