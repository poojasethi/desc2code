#include <bits/stdc++.h>
using namespace std;

int main() {
    int n,a[101],min=1000,g=0;
    cin >> n; for (int i=0; i<n; i++) cin >> a[i]; a[n]=a[0];
    for (int i=0; i<n; i++)
        if (abs(a[i]-a[i+1])<min) { min=abs(a[i]-a[i+1]); g=i; }
    cout << (g%n)+1 << " " << ((g+1)%n)+1 << endl;
    return 0;
}
