#include <cstdlib>
#include <iostream>
using namespace std;
int main(int argc, char *argv[])
{   int a[103],n,i,m,k=1,l=2;
    cin>>n;
    for(i=1;i<=n;i++)cin>>a[i];
    a[n+1]=a[1];m=abs(a[1]-a[2]);
    for(i=2;i<=n;i++)if(abs(a[i]-a[i+1])<m){k=i;l=i+1;m=abs(a[i]-a[i+1]);}
    if(l>n)l-=n;
    cout<<k<<" "<<l<<endl;
    return 0;
}
