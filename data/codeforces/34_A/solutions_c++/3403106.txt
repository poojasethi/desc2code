#include<cstdio>
int abs(int x){return x<0?-x:x;}
int a[102];
int main(){
	int n,m,i,min,x,y,t;
	while(~scanf("%d",&n)){
		for(i=1;i<=n;i++)scanf("%d",a+i);
		x=n,y=1,min=abs(a[x]-a[y]);
		for(i=2;i<=n;i++){
			t=abs(a[i]-a[i-1]);
			if(t<min){min=t;x=i-1;y=i;}
		}
		printf("%d %d\n",x,y);
	}
	return 0;
}
