#include<cstdio>
#include<cstdlib>
#include<cmath>
int a[110];
int main(){
	int n,min = 1e9,c,b;
	scanf("%d", &n);
	for(int i=0;i<n;i++){
		scanf("%d", &a[i]);
	}
	for(int i=1;i<n;i++){
		if(abs(a[i]-a[i-1])< min) {min = abs(a[i]-a[i-1]);c = i;b=i+1;}
	}
	if(abs(a[n-1]-a[0]) < min) printf("1 %d\n",n);
	else printf("%d %d\n", c,b);
	return 0;
}
