#include<bits/stdc++.h>
using namespace std;
int main(){
int x;
cin>>x;
int num[100];
for(int i=0;i<x;i++){
    cin>>num[i];
}
int mini=abs(num[x-1]-num[0]),a=x,b=1;
for(int i=0;i<x-1;i++){
    if(abs(num[i]-num[i+1])<mini){
        mini=abs(num[i]-num[i+1]);
        a=i+1;
        b=i+2;
    }
}
cout<<a<<" "<<b<<endl;
}
