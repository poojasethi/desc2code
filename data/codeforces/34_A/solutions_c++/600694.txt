#include <cstdio>
inline int abs(int A){return A>0?A:-A;}
int arr[100000], N;
int main(){
scanf("%d",&N);
int min = 2147483647,pos;
for(int i = 0; i < N; i++)
scanf("%d",&arr[i]);
for(int i = 0; i < N; i++)
    if(abs(arr[i]- arr[(i+1)%N])<min)min=abs(arr[i]- arr[(i+1)%N]), pos = i;
printf("%d %d\n", pos + 1, (pos + 1) % N + 1);
}
