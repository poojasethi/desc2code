#include <stdio.h>
int bj(int x)
{
	if (x>0) return x; else return -x;
}

int main()
{
	int a[1111];
	int i,n;
	scanf("%d",&n);
	for (i=1; i<=n; i++) scanf("%d",&a[i]);
	int l=n,r=1;
	for (i=1; i<n; i++)
	 if (bj(a[i]-a[i+1])<bj(a[l]-a[r]))
	  l=i,r=i+1;
	 printf("%d %d\n",l,r);
}
