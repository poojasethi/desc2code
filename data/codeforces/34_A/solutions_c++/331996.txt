#include <iostream>
#include <cstdlib>
using namespace std;

int main() {
	int n;
	cin >> n;

	int a[1001];
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}
	a[n] = a[0];

	int x = 0;
	for (int i = 0; i < n; i++) {
		if (abs(a[i]-a[i+1]) < abs(a[x]-a[x+1])) {
			x = i;
		}
	}

	cout << x+1 << " " << (x+1)%n + 1 << endl;
	return 0;
}
