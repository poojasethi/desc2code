#include<stdio.h>
int main()
{
int a[50005];
int n;
scanf("%d",&n);
int x,y;
x=0;
y=1;
int ma = 10000000;
int i;
for(i=0;i<n;i++)
scanf("%d",&a[i]);
for(i=0;i<n;i++)
{
int u = i;
int v = (i+1)%n;
int di = a[u]-a[v];
if(di <0)di = -di;
if(di < ma)
{
ma = di;
x = u;
y = v;
}
}

printf("%d %d\n",x+1,y+1);
return 0;
}
