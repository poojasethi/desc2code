#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<math.h>
int main()
{ int n,a[1001],i,min=1001,t,k;
scanf("%d",&n);
for(i=1;i<=n;i++) scanf("%d",&a[i]);
if(abs(a[n]-a[1])<min) {min=abs(a[n]-a[1]);t=n; k=1;}
for(i=2;i<=n;i++) if(abs(a[i]-a[i-1])<min) {min=abs(a[i]-a[i-1]);t=i-1; k=i;}
printf("%d %d",t,k);
}
