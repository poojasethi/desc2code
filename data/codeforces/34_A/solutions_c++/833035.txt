#include<iostream>
#include<cmath>
#include<string>
using namespace std;
int i,a[101],res=(int)1e9,n,num,i1,i2;
int main()
{
	cin>>n;
	for(i=1;i<=n;i++)
		cin>>a[i];
	a[0]=a[n];
	for(i=1;i<=n;i++)
		if(abs(a[i-1]-a[i])<res){res=abs(a[i-1]-a[i]); num=i;}
	i1=num-1;
	i2=num;
	if(!i1) i1=n;
	cout<<i1<<" "<<i2<<endl;
}
