#include <cstdio>
#include <cstdlib>

int main(){
  int arr[100];
  int n;scanf("%d",&n);
  for(int i=0;i<n;++i) scanf("%d",arr+i);

  int min=1000-1;
  int s,t;
  for(int i=0;i<n;++i){
    int a=i,b=(i+1)%n;
    if(abs(arr[a]-arr[b])<min){
      min=abs(arr[a]-arr[b]);
      s=a;t=b;
    }
  }
  printf("%d %d\n",s+1,t+1);
  return 0;
}
