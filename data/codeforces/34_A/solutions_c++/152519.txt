#include <iostream>
#include <cstdlib>

using namespace std;

int main() {
   long a[101],n,i,vt1,vt2;
   cin>>n;
   for (i=1;i<=n;i++) cin>>a[i];
   vt1=1;
   vt2=n;
   for (i=1;i<=n-1;i++) 
      if (abs(a[vt1]-a[vt2])>abs(a[i]-a[i+1])) {
         vt1=i;
         vt2=i+1;
         }
   cout<<vt1<<" "<<vt2;
}
