#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,a[110],i,j,m=INT_MAX;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	a[n]=a[0];
	for(i=0;i<n;i++)
	{
		if(abs(a[i]-a[i+1]) < m)
		{
			j=i+1;
			m=abs(a[i]-a[i+1]);
		}
	}
	if(j+1>n)
		cout<<j<<' '<<1<<endl;
	else
		cout<<j<<' '<<j+1<<endl;
	return 0;
}
