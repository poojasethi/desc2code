
#include<iostream>
using namespace std;
int main(){
    int l,a,x,b,i,x1,x2,n,min=100000;
    cin>>n;
    cin>>a;
    l=a;
    for(i=1;i<n;i++)
    {cin>>b;x=b-a;
    if(x<0)x=0-x;
    if(x<min){x1=i;x2=i+1;min=x;}
    a=b;
    }
    x=l-b;if(x<0)x=0-x;
    if(x<min)cout<<n<<" 1";
    else cout<<x1<<" "<<x2;return 0;}
