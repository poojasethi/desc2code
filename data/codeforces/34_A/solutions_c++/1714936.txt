#include <iostream>
#include <cmath>

using namespace std;

int N,A[110],P1,P2,min_diff=1000;

int main()
{
	cin >> N;
	for (int i=0;i<N;i++)
		cin >> A[i];
	A[N]=A[0];
	for (int i=0;i<N;i++)
		if (min_diff>abs(A[i]-A[i+1]))
		{
			min_diff=abs(A[i]-A[i+1]);
			P1=i;
			P2=(i+1)%N;
		}
	cout << P1+1 << " " << P2+1 << endl;
	return 0;
}
