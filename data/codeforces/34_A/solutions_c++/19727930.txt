#include <cstdio>

#define abs(x) (((x)>=0)?(x):(-(x)))

int main()
{
	int n, s = 0, h[102];
	scanf( "%d", &n );
	for ( int i=0; i<n; i++ )
		scanf( "%d", h+i );
	for ( int i=0; i<n; i++ )
		if ( abs(h[i]-h[(i+1)%n]) < abs(h[s]-h[(s+1)%n]) )
			s = i;
	printf( "%d %d\n", s+1, (s+1)%n+1 );
	return 0;
}
