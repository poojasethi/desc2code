#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,m=1001,c,d;
    cin>>n;
    int a[n],b[n];
    cin>>a[0];
    for(int i=1;i<n;i++) {cin>>a[i];b[i]=abs(a[i]-a[i-1]);}
    for(int i=1;i<n;i++) { if (m>b[i]) { m=b[i];c=i;d=i+1; } }
    if (m>abs(a[n-1]-a[0])) { c=1;d=n; }
    cout<<c<<" "<<d;

    return 0;
}
