#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int n, h[1001];
	cin>>n;
	for(int i = 0; i < n; i++) cin>>h[i];
	h[n] = h[0];
	int m = 1000;
	int mi, mj;
	for(int i = 1; i <= n; i++){
		if(abs(h[i]-h[i-1]) < m){
			m = abs(h[i]-h[i-1]);
			mi = i;
			mj = i-1;
		}
	}
	cout<<mj+1<<" "<<(mi == n ? 1: mi+1);
	return 0;
}
