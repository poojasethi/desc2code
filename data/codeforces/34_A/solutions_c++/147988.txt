#include<iostream>
#include<math.h>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
int n,a[105],i,j,mn,x,y;
cin>>n;
for(i=1;i<=n;i++)
{
cin>>a[i];
}
mn=abs(a[n]-a[1]);
x=1;
y=n;
for(i=1;i<n;i++)
{
if(abs(a[i]-a[i+1])<mn)
{
mn=abs(a[i]-a[i+1]);
x=i;
y=i+1;
}
}
cout<<x<<" "<<y;
return 0;
}
