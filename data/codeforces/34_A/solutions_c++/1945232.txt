#include<stdio.h>
#include<stdlib.h>
int main()
{
	int i,a,c=0,d,min,e;
	scanf("%d",&a);
	int b[a];
	for(i=0;i<a;i++)
	{
		scanf("%d",&b[i]);
	}
	d=a-1;
	min=abs(b[0]-b[a-1]);
	for(i=0;i<a-1;i++)
	{
		e=abs(b[i]-b[i+1]);
		if(e<min)
		{
			min=e;
			c=i;
			d=i+1;
		}
	}
	printf("%d %d\n",c+1,d+1);
	return 0;
}
