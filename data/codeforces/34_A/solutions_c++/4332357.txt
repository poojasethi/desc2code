#include <iostream>
#define dif(i) x[i]>x[i-1]?x[i]-x[i-1]:x[i-1]-x[i];
using namespace std;
int n,i,a,b,c,d,x[101];
int main()
{
    cin>>n;
    for(i=1;i<=n;i++)cin>>x[i];x[0]=x[n];
    d=dif(1);a=n;b=1;
    for(i=2;i<=n;i++)
    {
        c=dif(i);if(c<d){d=c;a=i-1;b=i;}
    }
    cout<<a<<' '<<b;
    return 0;
}
