#include <iostream>
#include <cmath>
#include <algorithm>
#include <stdio.h>
#include <memory.h>
using namespace std;
int n,a[1003],i,j=0;
int main(){
    cin>>n;
    for(i=0;i<n;++i) cin>>a[i];
    a[n] = a[0];
    for(i=0;i<n;++i)
        if (abs(1.0*(a[i]-a[i+1]))<abs(1.0*(a[j]-a[j+1]))) j=i;
    cout<<j+1<<' '<<(j+1)%n+1;
    return 0;
}
