#include<stdio.h>
#include<string.h>
#include<set>
#define maxn 100050
using namespace std;
typedef __int64 ll;
ll is_prime(ll k)
{
	for (ll i=2;i*i<=k;i++)
	{
		if (k%i==0) return 0;
	}
	return k!=1;
}
ll dfs(ll x,ll k)
{
	if (x<k||!is_prime(k))
	{
		return 0;
	}
	ll ans=x/=k;
	for (ll i=2;i<k&&i<=x;i++)
	{
		ans-=dfs(x,i);
	}
	return ans;
}
int main()
{
	ll a,b,k;
	scanf("%I64d%I64d%I64d",&a,&b,&k);
	ll ans=dfs(b,k)-dfs(a-1,k);
	printf("%I64d",ans);
	return 0;
}
