#include<bits/stdc++.h>

using namespace std;

int pan(int x)
{
    if (x==1) return 0;
    for (int i=2;i<=sqrt(x);i++)
        if (x%i==0) return 0;
    return 1;
}


int f(int a,int b)
{
    if (!pan(b)) return 0;
    int tmp=a/b;
    for (int i=2;i<b&&i*b<=a;i++)
        tmp-=f(a/b,i);
    return tmp;
}

int main()
{
    int l,r,k;
    cin>>l>>r>>k;
    cout<<f(r,k)-f(l-1,k);
    return 0;
}
