#include<cstdio>

int f[50000];
int A,B,C;
int res,tot,m;
int p[50000];

void dfs(int u,int now,int s)
{
	if (u==m || (long long)now*p[u]>tot) return;
	dfs(u+1,now,s);
	now*=p[u];
	res+=(tot/now)*-s;
	dfs(u+1,now,-s);
}

int calc(int n,int k)
{
	if (n<k) return 0;
	if ((long long)k*k>n) return 1;
	m=0;
	for (int i=2;i<k;i++)
	if (!f[i]) p[m++]=i;
	tot=n/k;res=tot;
	dfs(0,1,1);
	return res;
}

int main()
{
	for (int i=2;i<50000;i++)
	if (!f[i])
		for (int j=i*2;j<50000;j+=i)
			f[j]=i;
	
	scanf("%d%d%d",&A,&B,&C);
	for (int i=2;i*i<=C;i++)
	if (C%i==0)
	{
		printf("0\n");return 0;
	}

	printf("%d\n",calc(B,C)-calc(A-1,C));
}
