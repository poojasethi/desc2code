#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
int a,b,k,ans=0;
bool prime(int k){
    int t=sqrt(k);
    for(int i=2;i<=t;i++) if(k%i==0){
        return 0;
    }
    return 1;
}
int find(int n,int k){
    if(n<k||!prime(k))return 0;
    int t=n/k,ans=t;
    if(k==2)return n/2;
    for(int i=2;i<k&&i<=t;i++)ans-=find(t,i);
    return ans;
}
int main(){
    //freopen("in.txt","r",stdin);
    cin>>a>>b>>k;
    ans=find(b,k)-find(a-1,k);
    cout<<ans<<endl;
    return 0;
}
