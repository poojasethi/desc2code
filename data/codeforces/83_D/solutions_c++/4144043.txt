#include <stdio.h>
#include <cmath>
int prime(int k)
{
	for (int i=2;i<=sqrt(k)+0.001;i++)
		if (k%i==0) return 1;
	return 0;
}
int f(int x,int k)
{
	if (x<k||prime(k)) return 0;
	x/=k;
	int a=x;
	for (int i=2;i<k&&i<=x;i++)
		a-=f(x,i);
	return a;
}
int main()
{
	int a,b,k;
	scanf("%d %d %d",&a,&b,&k);
	int ans=0;
//	printf("%d\n",prime(4));
	ans=f(b,k)-f(a-1,k);
	printf("%d\n",ans);
	return 0;
}
