#include<cstdio>
int Prime(int x)
{
	for(int i=2;i*i<=x;i++)	if(x%i==0)	return 1;
	return 0;
}
int F(int x,int k)
{
	if(x<k||Prime(k)) return 0;
        x/=k;
        int re=x;
	for(int i=2;i<k&&i<=x;i++)	re-=F(x,i);
	return re;
}
int main()
{
	int l,r,k;
	scanf("%d %d %d",&l,&r,&k);
	printf("%d",F(r,k)-F(l-1,k));
	return 0;
}
