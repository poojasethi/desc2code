#include <cstdio>
bool is(int x) {
    for (int i = 2; i * i <= x; i++)
        if (x % i == 0) return false;
    return true;
}
int f(int a, int b) {
    if (!is(b) || a < b) return 0;
    if (a < (__int64)(b) * b) return 1;
    int res = a / b;
    for (int i = 2; i < b; i++) res -= f(a/b, i);
    return res;
}
int main() {
    int a, b, c; while (~scanf("%d%d%d", &a, &b, &c)) printf("%d", f(b, c) - f(a-1, c));
}
