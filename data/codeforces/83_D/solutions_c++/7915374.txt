#include <iostream>
#include <cmath>
#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

bool isPrime(long long x)
{
    if(x == 2)
        return true;

    if(!(x & 1))
        return false;

    for(int i = 3;i * i <= x;i += 2)
        if(!(x % i))
            return false;
    return true;
}

int rec(int num , int div)
{
    if(num < div || (!isPrime(div)))
        return 0;

    int ans = num / div;
    for(int i = 2;i < div && i <= (num / div);++i)
        ans -= rec(num / div , i);
    return ans;
}

int main()
{
    int a , b , k;
    scanf("%d%d%d" , &a , &b , &k);

    printf("%d" , rec(b , k) - rec(a - 1 , k));

    return 0;
}
