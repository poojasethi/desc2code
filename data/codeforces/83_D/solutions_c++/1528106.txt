#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
#define LL long long
bool isprime(int p)
{
    if(p<2)return false;
    for(int i=2;i*i<=p;i++)
    {
        if(p%i==0)return false;
    }
    return true;
}
int cal(int n,int k);
int find(int m,int k)
{
    int sum=m;
    for(int i=2;i<k;i++)sum-=cal(m,i);
    return sum;
}
int cal(int n,int k)
{
    if(!isprime(k)||n<k)return 0;
    if(n<(LL)k*k)return 1;
    return find(n/k,k);
}
int main()
{
    int a,b,k;
    scanf("%d%d%d",&a,&b,&k);
    printf("%d\n",cal(b,k)-cal(a-1,k));
    return 0;
}
