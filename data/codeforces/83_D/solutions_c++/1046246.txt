#include<iostream>
using namespace std;

bool isPrime(int a) {
	for (int i = 2; i * i <= a; ++i) {
		if (a % i == 0) {
			return false;
		}
	}
	return true;
}
int rec(int a, int b) {
	if (b > a || !isPrime(b)) {
		return 0;
	}
	int s = a / b;
	for (int i = 2; i < b && i <= a / b; ++i) {
		s -= rec(a / b, i);
	}
	return s;
}
int main() {
	int A, B, K;
	cin >> A >> B >> K;
	cout << rec(B, K) - rec(A - 1, K);
}
