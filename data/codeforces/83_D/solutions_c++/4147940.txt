#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
#define S(a) scanf("%d",&a)
#define foru(i,a,b) for(int i=a;i<=b;++i)

using namespace std;
int ans,a,b,k;
bool notp(int a)
{
	for(int i=2;i*i<=a;++i)
	  if(a%i==0) return 1;
	return 0;
}
int f(int a,int b)
{
	if(a<b||notp(b)) return 0;
	if(a<(long long)b*b) return 1;
	int res=a/=b; 
	foru(i,2,b-1) res-=f(a,i);
	return res;
}
int main()
{
	S(a),S(b),S(k);
	ans=f(b,k)-f(a-1,k);
	printf("%d\n",ans);
	return 0;
}
