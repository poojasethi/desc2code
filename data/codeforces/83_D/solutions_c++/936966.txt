#include<cstdio>
#include<cmath>
bool check(int x){
    int l=(int)sqrt(x);
    for(int i=2;i<=l;++i)
        if(x%i==0)return 0;
    return 1;
}
int solve(int x,int k){
    if(x<k||!check(k))return 0;
    int n=x/k,res;if(n<k)return 1;
    res=n;
    for(int i=2;i<k;++i)
        res-=solve(n,i);
    return res;
}
int main(){
    int l,r,k;
    scanf("%d%d%d",&l,&r,&k);
    printf("%d\n",solve(r,k)-solve(l-1,k));
    return 0;
}
