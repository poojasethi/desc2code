#include <cstdio>

using namespace std;

bool p(int x)
{
  for(int i=2;i*i<=x;++i)
    if(x%i==0)
      return false;
  return true;
}

int s(int x,int y)
{
  if((!p(y))||(x<y))
    return 0;
  if(x<1ll*y*y)
    return 1;
  int t=x/y;
  for(int z=2;z<y;++z)
    t-=s(x/y,z);
  return t;
}

int main(void)
{
  int a,b,k;
  scanf("%d %d %d",&a,&b,&k);
  printf("%d\n",s(b,k)-s(a-1,k));
  return 0;
}
