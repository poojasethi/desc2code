#include<stdio.h>
#include<math.h>
#include<algorithm>
#include<string.h>
using namespace std;

bool isprim(int x) {
    int n = sqrt(x + .1);
    for (int i = 2; i <= n; i++) {
        if (x % i == 0)
            return false;
    }
    return true;
}
int solve(int n, int k) {
    if (!isprim(k))
        return 0;
    if (k > n)
        return 0;
    if (k * k > n)
        return 1;
    int c = n / k;
    for (int i = 2; i < k && i <= n / k; i++) {
        c = c - solve(n / k, i);
    }
    //printf("(%d,%d):%d\n", n, k, c);
    return c;
}
int main() {
    setvbuf(stdout,NULL,_IONBF,0);
    int n,m,k;
    while(scanf(" %d %d %d",&n,&m,&k)==3) {
        printf("%d\n",solve(m,k)-solve(n-1,k));
    }
    return 0;
}
