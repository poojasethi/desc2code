#include <iostream>
#include <cstdio>
#include <cstring>
#include <utility>
#include <set>
#include <algorithm>
#include <ctime>
#include <queue>
using namespace std;

#define clr(x) memset(x,0,sizeof(x))
#define eps 1e-8
#define LL long long

LL n,m,k;

bool judge(LL x)
{
    LL i;
    for(i=2;i*i<=x;i++)
    {
        if(x%i==0) return false;
    }
    return true;
}

LL f(LL n,LL k)
{
    if(n<k||judge(k)==0) return 0;
    if(n<k*k) return 1;
    LL x=n/k,t,i;t=x;
    for(i=2;i<=x&&i<k;i++)
    {
        t-=f(x,i);
    }
    return t;
}

int main()
{
    scanf("%I64d%I64d%I64d",&n,&m,&k);
    printf("%I64d\n",f(m,k)-f(n-1,k));
    return 0;
}
