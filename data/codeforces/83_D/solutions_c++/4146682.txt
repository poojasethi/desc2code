#include<iostream>
#include<cstring>
#include<string>
#include<cmath>
using namespace std;
bool prime(int x)
{
    for (int i = 2; i <= floor(sqrt(x)); i++)
    {
        if (x % i == 0) return 0;
    }
    return 1;
}
long long f(long long x,long long y)
{
    if (x < y || !prime(y)) return 0;
    long long tmp = x / y;
    x = tmp;
    for (int i = 2; i < y && i <= x; i++) tmp = tmp - f(x,i);
    return tmp;
}
int main()
{
    int a,b,k;
    cin>>a>>b>>k;
    if (!prime(k))
    {
        cout<<0<<endl;
        return 0;
    }
    long long ans = f(b,k) - f(a - 1,k);
    cout<<ans<<endl;
}
