#include <cstdio>
using namespace std;

bool P(int k)
{
    for (int i = 2; i * i <= k; i++)
        if ((k % i) == 0) return 0;
    return 1;
}

int calc(int n, int k)
{
    if (k > n || !P(k)) return 0;
    int res = n / k;
    
    for (int i = 2; i < k && i <= n/k; i++)
        res -= calc(n/k, i);
    return res;
}
int main()
{
   // freopen("nimo.in", "r", stdin);
    int l, r, k;
    scanf("%d%d%d", &l, &r, &k);
    printf("%d\n", calc(r, k) - calc(l - 1, k));
    return 0;
}
