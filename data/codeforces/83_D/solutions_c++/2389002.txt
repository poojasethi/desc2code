#include <iostream>
#include <cmath>
using namespace std;

bool isPrime(int x) {
    for (int i = 2; i * i <= x; i++) {
        if (x % i == 0) return false;
    }
    return true;
}

int f(int x, int k) {
    if (x < k || isPrime(k) == 0) return 0;
    int ans = x / k;
    for (int i = 2; i < k && i <= x / k; i++) {
        ans -= f(x/k, i);
    }
    return ans;
}

int main() {
    int a, b, k;
    cin >> a >> b >> k;
    cout << f(b, k) - f(a-1, k) <<endl;
    //system("pause");
    return 0;
}
