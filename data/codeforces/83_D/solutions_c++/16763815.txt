#include <iostream>
#include <cstdio>
using namespace std;
bool is_p(int x){
	for(int i=2;i*i<=x;i++){
		if(x%i==0)	return false;
	}
	return true;
}
int F(int x,int k){
	if(x<k || !is_p(k))	return 0;
	int ans=x/=k;
	for(int i=2;i<k&&i<=x;i++){
		ans-=F(x,i);
	}
	return ans;
}
int main(){
	int l,r,k;
	cin>>l>>r>>k;
	printf("%d",F(r,k)-F(l-1,k));
	

	return 0;
}
