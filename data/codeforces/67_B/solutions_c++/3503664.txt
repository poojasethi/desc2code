#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int main()
{
    int n,b[1005],k,t,j,i;
    while(~scanf("%d%d",&n,&k))
    {
        for(i=1;i<=n;i++)
        scanf("%d",&b[i]);
        t = n;
        while(t--)
        {
            for(i=1;i<=n;i++)
            if(b[i]==0)
            break;
            cout << i << " ";
            b[i] = -1;
            for(j=1;j+k<=i;j++)
            b[j]--;
        }
        cout << endl;
    }
    return 0;
}
