#include <cstdio>
using namespace std;
int n,a[1003],b[1003],k;
int main()
{
    scanf("%d%d",&n,&k);
    for (int i=1;i<=n;i++) scanf("%d",b+i);
    for (int i=1;i<=n;i++)
    {
        int Cur=1;
        while (b[Cur]!=0) Cur++;
        a[i]=Cur;
        b[Cur]--;
        for (int j=1;j+k<=Cur;j++) b[j]--;
    }
    for (int i=1;i<=n;i++) printf("%d ",a[i]);
    return 0;
}
