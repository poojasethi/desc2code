#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
#define fo(i,a,b) for (int i=a; i<=b; ++i)
#define fd(i,a,b) for (int i=a; i>=b; --i)

const int mn=1010;

int a[mn],Q[mn];
int n,k,r;

int main(){
	scanf("%d%d",&n,&k);
	fo (i,1,n) scanf("%d",&a[i]);
	fd (i,n,1){
		int S=a[i],j=0;
		while (S) S-=Q[j++]>=i+k;
		memcpy(Q+j+1,Q+j,sizeof(Q[1])*(r-j));
		Q[++r,j]=i;
	}
	fo (i,0,n-2) printf("%d ",Q[i]);
	printf("%d\n",Q[n-1]);
}
