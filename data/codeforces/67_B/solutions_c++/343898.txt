#include <cstdio>
#define REP(i,n) for (int i=0; i<(int)(n); ++i)

int vi[1020];

int main() {
    int N, K;
    while (scanf("%d%d", &N, &K) != EOF) {
        REP(j, N) scanf("%d", vi+j);
            // greedy...?
        REP(j, N) {
            REP(k, N) if (vi[k] == 0) {
                REP(l, k-K+1) vi[l]--;
                vi[k] = -666;
                if (j) printf(" ");
                printf("%d", k+1);
                break;
            }
        }
        puts("");
    }
}
