#include <iostream>
#include <vector>
using namespace std;
int N,K,B[1<<20];
int main()
{
	cin>>N>>K;
	for(int i=0; i<N; ++i) cin>>B[i];
	vector<int> res;
	for(int i=N-1; i>=0; --i) {
		int x = B[i];
		int a = K+i+1;
		int p,s;
		for(p=0,s=0; s<x; s+=res[p++]>=a);
//		cout<<"asd "<<i<<' '<<x<<' '<<p<<' '<<a<<'\n';
		res.insert(res.begin()+p, i+1);
	}
	for(int i=0; i<N; ++i) {
		if (i>0) cout<<' ';
		cout<<res[i];
	}
	cout<<'\n';
}
