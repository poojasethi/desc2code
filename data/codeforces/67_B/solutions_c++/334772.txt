#include<iostream>
using namespace std;
int used[1001],a[1000],b[1000];
int main()
{
	int n,kk;
	cin>>n>>kk;
	for(int i = 0; i < n; i++)
		cin>>a[i];
	for(int i = 0; i < n; i++)
		for(int j = 1; j <= n; j++)
			if (!used[j])
			{
				int ok = 0;
				for(int k = 0; k < i; k++)
					if (j + kk <= b[k]) ok++;
				if (ok == a[j - 1])
				{
					cout<<j<<" ";
					b[i] = j;
					used[j] = 1;
					break;
				}
			}
	return 0;
}
