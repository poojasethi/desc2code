#include <iostream>
#include <cstdio>
#include <string>
#include <vector>
#include <queue>
#include <algorithm>
#include <list>

using namespace std;

int n,k;
int a[1011];
int main()
{
	cin>>n>>k;
	for(int i=1;i<=n;i++) scanf("%d",a+i);
	list<int> res;
	list<int>::iterator lt;
	for(int i=n;i>0;i--)
	{
		int t=i+k;
		int cnt=0;
		for(lt=res.begin();lt!=res.end()&&cnt<a[i];lt++) if(*lt>=t)
			cnt++;
		res.insert(lt,i);
	}
	for(lt=res.begin();lt!=res.end();lt++)
		printf("%d ",*lt);
	cout<<endl;
	return 0;
}
