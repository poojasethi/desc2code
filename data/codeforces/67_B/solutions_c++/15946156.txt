#include <iostream>
#include <string>
using namespace std;

int B[1010];
int n,k;
int A[1010];
int main(){
	cin>>n>>k;
	for(int i=0;i<n;i++){
		cin>>B[i];
	}
	int s=0;
	for(int i=n-1;i>=0;i--){
		int nleft=max(n-(i+k),0);
		A[s++]=i;
		int cur=s-1;
		while(cur>0){
			if(A[cur-1]>=A[cur]+k){
				if(nleft==B[i])break;
				nleft--;
			}
			swap(A[cur-1],A[cur]);
			cur--;
		}
	}
	for(int i=0;i<n;i++){
		cout<<A[i]+1<<" ";
	}
}
