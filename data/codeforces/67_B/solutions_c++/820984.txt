#include <iostream>

using namespace std;

const int N_MAX = 2000;

int N, K;
int counts[N_MAX];
int output[N_MAX];

int main() {
  cin >> N >> K;
  for (int i = 0; i < N; i++)
    cin >> counts[i];

  for (int i = 0; i < N; i++) {
    int x = 0;
    while (counts[x] != 0)
      x++;
    output[i] = x;
    counts[x] = -1;
    for (int j = 0; j + K <= x; j++)
      counts[j]--;
  }

  for (int i = 0; i < N; i++)
    cout << output[i] + 1 << (i == N - 1 ? '\n' : ' ');
}
