#include<cstdio>

int p[1010];
int n,m;

int main()
{
    scanf("%d%d",&n,&m);
    for (int i=1;i<=n;i++) scanf("%d",&p[i]);
    for (int i=1;i<=n;i++)
    {
        int k;
        for (k=1;k<=n;k++)
        if (!p[k]) break;
        printf("%d%c",k,i==n?'\n':' ');
        for (int j=1;j<=k-m;j++) p[j]--;
        p[k]=-1;
    }
}
