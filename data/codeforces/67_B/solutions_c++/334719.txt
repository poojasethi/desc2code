#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;


int a[1010];
bool used[1010];

int main() {
	int n, k;
	cin >> n >> k;
	for (int i = 0; i < n; ++i)
		cin >> a[i];
	for (int u = 0; u < n; ++u) {
		for (int i = 0; i < n; ++i) if (!used[i] && a[i] == 0) {
			used[i] = true;
			printf("%d ", i + 1);
			for (int j = 0; j <= i - k; ++j) --a[j];
			break;
		}
	}
	printf("\n");
	return 0;
}
