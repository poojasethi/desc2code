#include <iostream>
using namespace std;
int n,k,i,j,m,b[1001];
int main()
{
    //ifstream cin("input.txt");
    //ofstream cout("output.txt");
	cin >> n >> k;
	for (i=1;i<=n;i++)
		cin >> b[i];
    for (i=1;i<=n;i++)
	{
		for (j=1;j<=n;j++)
			if (!b[j]) break;
		cout << j << " ";
        b[j]=-1;
        for (m=1;m<=j-k;m++)
			b[m]--;
    }
	cout << endl;
    return 0;
}
