#include <string>
#include <stdio.h>
#include <iostream>
#include <vector>
using namespace std;

int main() {
	vector<int> A(1001), B(1001);
	int N, K;
	scanf("%d %d", &N, &K);
	for ( int i = 0; i < N; i++ ) {
		scanf("%d", &B[i]);
		A[i] = i+1;
	}
	
	for ( int i = N-1; i >= 0; i-- ) {
		int m = K+i+1;
		int n = B[i];
		int j;
		for ( j = i+1; j < N && n > 0; j++ )
			if ( A[j] >= m )
				n--;
		
		j--;
		A.erase(A.begin()+i);
		A.insert(A.begin()+j,i+1);
	}

	for ( int i = 0; i < N; i++ )
		printf("%d ", A[i]);

	printf("\n");	

}
