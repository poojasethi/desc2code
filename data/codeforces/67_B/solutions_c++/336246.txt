#include <stdio.h>

int n, A[1024], k;

int main()
{
  int i, j;
  scanf("%d %d", &n, &k);
  for(i = 0; i < n; ++i)
    scanf("%d", &A[i]);
  for(i = 0; i < n; ++i)
  {
    for(j = 0; j < n; ++j)
      if(A[j] == 0)
        break;
    printf("%d ", j + 1);
    A[j] = -1;
    for(j -= k; j >= 0; --j)
      A[j]--;
  }
  printf("\n");
  return 0;
}
