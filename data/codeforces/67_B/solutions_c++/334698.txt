#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 1001

int n,k,i,j,p;
int A[lim];
bool B[lim];

int main()
{
  scanf("%d%d",&n,&k);
  for(i=0;i<n;i++)
    scanf("%d",&A[i]);
  for(i=0;i<n;i++)
  {
    for(j=0;A[j] || B[j];j++);
    printf("%d ",j+1);
    for(p=j-k;p>=0;p--)
      A[p]--;
    B[j]=1;
  }
  return 0;
}
