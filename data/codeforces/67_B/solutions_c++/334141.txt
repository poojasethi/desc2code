#include<cstdio>
int N,K,a[1005];

int main(){
    scanf("%d%d",&N,&K);
    for (int i=1;i<=N;++i){
        scanf("%d",&a[i]);
    }	

    for (int z=1;z<=N;++z){
        int i = 1;
        while (a[i]!=0) ++i;
        printf("%d",i);
        if (z==N) puts("");
        else printf(" ");
        
        for (int j=i-K;j>=1;--j){
            if (a[j]>0) --a[j];
        }    
        a[i] = -1;
    }
	return 0;
}
