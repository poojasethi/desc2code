# include <cstdio>
# include <cstring>
# include <cstdlib>
# include <iostream>
using namespace std;
# define N 1050
int a[N];
int main (void)
{
    int n,k;
    cin>>n>>k;
    for (int i=1;i<=n;i++)
        cin>>a[i];
    for (int i=1;i<=n;i++)
    {
        int j=1;
        for (;j<=n;j++)
            if (!a[j])
                break;
        printf("%d%c",j,i==n?'\n':' ');
        for (int p=1;p<=j-k;p++)
            a[p]--;
        a[j]=-1;
    }
    return 0;
}
