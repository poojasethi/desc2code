#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

int n,m,a[1005],b[1005],v[1005];

int main(){

	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++) scanf("%d",&b[i]);
	
	for (int i=1;i<=n;i++){
		for (int j=1;j<=n;j++)
		    if (!v[j] && b[j]==0){
				a[i]=j;
				v[j]=1;
				break;
			}
		for (int j=a[i]-m;j>=1;j--) b[j]--;
	}
	
	for (int i=1;i<=n;i++) printf("%d%c", a[i], i<n?' ':'\n');

	return 0;
}
