#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<cstring>
#include<cmath>
using namespace std;

int n,k,a[1005];

int main() {
	scanf("%d%d",&n,&k);
	for (int i=0; i<n; i++) {
		scanf("%d",&a[i]);
	}
	for (int i=0; i<n; i++) {
		int r=0;
		for (int j=0; j<n; j++)
			if (a[j]==0) {
				if (i!=0) printf(" ");
				printf("%d",j+1);
				a[j]--;
				r=j;
				break;
			}
		for (int j=0; j<n; j++)
			if (a[j]>0 && j+k<=r) a[j]--;
	}
	printf("\n");	
}
