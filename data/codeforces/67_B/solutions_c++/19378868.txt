#include<stdio.h>
int in[1010];
int n,k;
int i,j,q;
int main(){
    scanf("%d%d",&n,&k);
    for(i=1;i<=n;i++)
		scanf("%d",&in[i]);
    for(i=1;i<=n;i++){
		for(j=1;j<=n;j++)
			if(!in[j])
				break;
                if(i==n)
                      printf("%d\n",j);
		else  
                      printf("%d ",j);
		for(q=1;q<=j-k;q++)in[q]--;
		in[j]--;
    }
}
