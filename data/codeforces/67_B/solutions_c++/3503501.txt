#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int main()
{
    int n,b[1005],k,t,j,i;
    while(~scanf("%d%d",&n,&k))
    {
        for(i=0;i<n;i++)
        scanf("%d",&b[i]);
        t = n;
        while(t--)
        {
            for(i=0;i<n;i++)
            if(b[i]==0)
            break;
            cout << i+1 << " ";
            b[i] = -1;
            for(j=1;j+k<=i+1;j++)
            b[j-1]--;
        }
        cout << endl;
    }
    return 0;
}
