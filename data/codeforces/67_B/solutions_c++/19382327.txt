#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int n,k;
int b[1100];
int big[2200];
int vis[1100];
int main(){
	cin>>n>>k;
	for (int i=1 ; i<=n ; i++) cin>>b[i];
	memset(big,0,sizeof big);
	memset(vis,0,sizeof vis);
	for (int i=1 ; i<=n ; i++)
		for (int j=1 ; j<=n ; j++)
		if (vis[j]==0 && b[j]==big[j+k]){
			for (int k=0 ; k<=j ; k++) big[k]++;
			vis[j] = 1;
			cout<<j;
			if (i != n) cout<<" ";
			break;
		}
	return 0;
}
