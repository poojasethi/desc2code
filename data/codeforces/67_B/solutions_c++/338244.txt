#include <stdio.h>

const int kMaxN = 1000;

bool used[kMaxN];
int b[kMaxN], cnt[kMaxN];
int n, k;

int main() {
  scanf("%d%d", &n, &k);
  for (int i = 0; i < n; ++i) scanf("%d", &b[i]);
  for (int i = 0; i < n; ++i) {
    if (i != 0) printf(" ");
    for (int j = 0; j < n; ++j) {
      if (used[j]) continue;
      if (b[j] == cnt[j]) {
        printf("%d", j + 1);
        used[j] = true;
        for (int l = 0; l <= j - k; ++l) ++cnt[l];
        break;
      }
    }
  }
  puts("");
  return 0;
}
