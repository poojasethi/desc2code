#include <iostream>
#include <vector>
using namespace std;
int main() {
	vector<int> arr, ord;
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; ++i) {
		int t;
		cin >> t;
		arr.push_back(t);
	}
	for (int i = n - 1; i >= 0; --i) {
		int need = arr[i], cnt = 0;
		for (int j = 0; j <= ord.size(); ++j) {
			if (need == cnt) {
				ord.insert(ord.begin() + j, i);
				break;
			}
			cnt += (ord[j] >= i + m);
		}
	}
	for (int i = 0; i < ord.size(); ++i) {
		cout << ord[i] + 1 << ' ';
	}
}
