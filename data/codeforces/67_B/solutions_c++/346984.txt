#include <iostream>
using namespace std;

int main()
{
    int n, k;
    int b[1005];
    cin>>n>>k;
    for(int i = 1; i <= n ; i++)
        cin>>b[i];
    for(int i = 1; i <= n; i++)
    {
        int j ;
        for( j = 1; j <= n; j++)
            if(b[j] == 0) break;
        cout<<j<<" ";
        b[j] = -1;
        for(int p = 1; p <= j - k; p ++)  --b[p];
    }
    return 0;
}
