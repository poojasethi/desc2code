#include <cstdio>
#include <iostream>
#include <vector>

using namespace std;

const int maxn = 1000 + 5;

vector <int> A;
int B[maxn];

int main()
{
  int n, k;
  cin >> n >> k;
  for (int i = 0; i < n; ++i) {
    cin >> B[i];
  }
  
  for (int i = n - 1; i >= 0; --i) {
    int pos = 0;
    while (B[i]) {
      ++pos;
      if (A[pos - 1] > i + k) {
        --B[i];
      }
    }
    
    A.insert(A.begin() + pos, i + 1);
  }
  
  for (int i = 0; i < n; ++i) {
    cout << A[i] << ' ';
  }
  
  return 0;
}
