#include <stdio.h>
#include <vector>
using namespace std;

int main(void) {
	int i, n, k;
	scanf("%d%d", &n, &k);
	int b[1005];
	for(i=0;i<n;i++)
		scanf("%d", &b[i]);
	vector<int> a;
	for(i=n-1;i>=0;--i) {
		if(b[i]==0)
			a.insert(a.begin(), i);
		for(vector<int>::iterator it = a.begin(); it != a.end(); it++)
			if(*it>=i+k && --b[i]==0) {
				a.insert(++it, i);
				break;
			}
	}
	for(i=0;i<n;i++)
		printf("%d ", a[i]+1);
	printf("\n");
	return 0;
}
