#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

int n, k;
const int N=10000;
int b[N], a[N], p[N];
int was[N];

int main(){
	scanf("%d%d\n", &n, &k);
	for (int i=0; i<n; ++i)
		scanf("%d", &b[i]);
	for (int i=0; i<n; ++i){
		for (int j=0; j<n; ++j){
			if ((j+k>n || p[n]-p[j+k]==b[j]) && was[j]==0){
				a[i]=j;
				was[j]=1;
				break;
			}
		}
		for (int j=a[i]+1; j<=n; ++j)
			p[j]++;
	}
	for (int i=0; i<n; ++i)
		printf("%d ", a[i]+1);
}
