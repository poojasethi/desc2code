#include <cstdio>
#include <vector>

using namespace std;

int main() {
	int n, k;
	scanf("%d%d", &n, &k);

	vector<int> b(n);

	for (int i = 0; i < n; ++i) {
		scanf("%d", &b[i]);
	}

	vector<int> ans(0);
	for (int i = n - 1; i >= 0; --i) {
		int cnt = 0;
		vector<int>::iterator j;
		for (j = ans.begin(); j != ans.end(); ++j) {
			if (cnt == b[i]) {
				break;
			} else if (*j >= i + 1 + k) {
				cnt++;
			}
		}
		ans.insert(j, i + 1);
	}

	for (int i = 0; i < n; ++i) {
		printf("%d ", ans[i]);
	}
	return 0;
}
