import sys

lines = sys.stdin.readlines()

n,k = map(lambda x: int(x), lines[0].split())
inv_table = map(lambda x: int(x), lines[1].split())

result = [n-1]

i = n-2
while i >= 0:
    j = 0
    inv = inv_table[i]
    while inv > 0:
        if result[j] >= i+k:
            inv = inv - 1
        j = j+1
    result = result[:j] + [i] + result[j:]
    i = i-1

for i in result:
    print i+1,
