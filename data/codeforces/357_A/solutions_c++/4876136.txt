#include <bits/stdc++.h>

using namespace std;

int main ()
{
	int i,m,arr[105],x,y, a=0, sum=0;
	cin >> m;
	for(i=1; i<=m; sum+=arr[i], i++)
		cin >> arr[i];
	cin >> x >> y;
	for(i=1; i<=m; i++)
	{
		a+=arr[i];
		if(a>=x and a<=y and sum-a>=x and sum-a<=y) break;
	}
	cout << ((i>m) ? 0 : i+1) << endl;
}
