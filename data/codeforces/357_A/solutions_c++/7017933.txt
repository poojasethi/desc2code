#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,c[110],i,x,y,tot=0,sum=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>c[i];
		tot+=c[i];
	}
	cin>>x>>y;
	for(i=0;i<n;i++)
	{
		if(sum>=x && sum<=y && (tot-sum)>=x && (tot-sum)<=y)
		{
			cout<<i+1<<endl;
			return 0;
		}
		sum+=c[i];
	}
	cout<<"0\n";
	return 0;
}
