#include <iostream>
using namespace std;
int m,n,v,x,y,c[100],k;
int main()
{
	cin >> m;
	for (int i=0;i<m;i++)
	{
		cin >> c[i];
		v+=c[i];
	}
	cin >> x >> y;
	k=m; n=0;
	while ((n<x||v>y)&&k>0)
	{
		k--;
		n+=c[k];
		v-=c[k];
	}
	if (n<x||n>y||v<x||v>y)
		cout << 0 << endl;
	else
		cout << k+1 << endl;
	return 0;
}
