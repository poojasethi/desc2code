#include <iostream>
using namespace std;
int n,s[112],t,x,y;
int main(){
  cin>>n>>s[0];
  for(int i=1;i<n;i++){
    cin>>t;
    s[i]=t+s[i-1];
  }
  cin>>x>>y;
  for(int i=0;i<n;i++)
    if(x<=s[i] && s[i]<=y && x<=s[n-1]-s[i] && s[n-1]-s[i]<=y){
      cout<<i+2<<endl;
      return 0;
    }
  cout<<0<<endl;
  return 0;
}
