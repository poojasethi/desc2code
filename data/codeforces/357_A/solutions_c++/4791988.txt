#include<iostream>
using namespace std;

int main()
{
	int m;
	cin >> m;
	int a[m+1];
	a[0]=0;
	for(int i=1;i<=m;i++){
		cin >> a[i];
		a[i]+=a[i-1];
	}
	int x,y;
	cin >> x >> y;
	for(int k=1;k<=m;k++){
		if((a[k]-a[0]>=x && a[k]-a[0]<=y) && (a[m]-a[k]>=x && a[m]-a[k]<=y)){
			cout << k+1 << endl;
			return 0;
		}
	}
	cout << "0\n";
	return 0;
}
