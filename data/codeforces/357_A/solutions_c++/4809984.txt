#include<iostream>
#include<cmath>
using namespace std;
main()
{
	int m,s=0,x,y,ball=0,a[1000];
	cin>>m;
	for(int i=0;i<m;i++)
	{cin>>a[i];
	s+=a[i];}
	cin>>x>>y;
	
	for(int i=0;i<m;i++)
	{
		if(ball>=x && ball<=y && x<=s-ball && y>=s-ball){cout<<i+1;return 0;}
		ball+=a[i];
	}
		
	cout<<0;
	return 0;
}
