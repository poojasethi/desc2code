#include<iostream>

using namespace std;

int main()
{

	int m,c[101],s1=0,s2=0,x,y;
	cin>>m;
		
	for(int i=1;i<=m;i++)
	{
		cin>>c[i];
		s1+=c[i];
	}	
	cin>>x>>y;
	int i=1;
	while(i<=m && !(s2<=y && s2>=x && s1-s2>=x && s1-s2<=y ))
	{
		s2+=c[i];
		i++;
	}	
	if(i==m+1) {cout<<0<<endl;return 0;}

	else cout<<i<<endl;

return 0;
}
