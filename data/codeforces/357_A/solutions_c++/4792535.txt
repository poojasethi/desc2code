#include<stdio.h>
int main()
{
	int m,i,sum,x,y;
	scanf("%d",&m);
	int a[m+1],total=0;
	for(i=1;i<=m;i++)
	{
		scanf("%d",&a[i]);
		total+=a[i];
	}
	scanf("%d %d",&x,&y);
	sum=0;
	for(i=1;i<=m;i++)
	{
		sum+=a[i];
		if(sum>=x && sum<=y && total-sum>=x && total-sum<=y)
		{
			printf("%d\n",i+1);
			return 0;
		}
	}
	printf("0\n");
	return 0;
}
