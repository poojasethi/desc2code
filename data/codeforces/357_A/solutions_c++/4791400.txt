#include <iostream>
using namespace std;
int main() {
	int n, i, c[102] = {0}, sum, x, y;
	cin >> n;
	for (i = 1; i <= n; i++) {cin >> x; c[i] = c[i-1]+x;};
	cin >> x >> y;
	for (i = 1; i <= n; i++) 
		if ((x <= c[i-1]) && (c[i-1] <= y) && (x <= c[n]-c[i-1]) && (c[n]-c[i-1] <= y)) {cout << i; return 0;};
	cout << 0;
	return 0;
}
