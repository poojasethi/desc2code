#include "bits/stdc++.h"
using namespace std;

int m, x, y, gp, gd;
int c[108];
int main(){
	cin>>m;
	for(int i=1; i<=m; i++){
		cin>>c[i];
		gp+=c[i];
	}
	cin>>x>>y;
	for(int i=1; i<=m; i++){
		if(gp>=x and gd>=x and gp<=y and gd<=y){
			cout<<i;
			return 0;
		}
		gp-=c[i];
		gd+=c[i];
	}
	cout<<"0";
	return 0;
}
