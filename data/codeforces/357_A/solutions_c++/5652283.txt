#include<stdio.h>
int main()
{
	int t,sum=0,s[200],tot=0,a,b,i;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		scanf("%d",&s[i]);
		sum+=s[i];
	}
	scanf("%d%d",&a,&b);
	for(i=0;i<t;i++)
	{
		tot+=s[i];
		if(tot>=a&&tot<=b&&sum-tot>=a&&sum-tot<=b)
		{
			printf("%d\n",i+2);
			return 0;
		}
	}
	printf("0\n");
	return 0;
}
