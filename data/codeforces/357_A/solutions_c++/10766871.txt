#include<stdio.h>
int i,j,n,top,fark,arr[10000],x,y;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&arr[i]),top+=arr[i];

	scanf("%d %d",&x,&y);

	for(i=1;i<=n;i++){
		top-=arr[i];
		fark+=arr[i];

		if(top>=x && top<=y && fark>=x && fark<=y){
			printf("%d\n",i+1);
			return 0;
		}
	}
	puts("0");
	return 0;
}
