#include<stdio.h>
main()
{
	int n,sum=0,x,y,i,ans=0;
	scanf("%d",&n);
	int arr[n+1];
	for(i=1;i<=n;i++)
	{
		scanf("%d",&arr[i]);
		sum+=arr[i];
	}
	scanf("%d%d",&x,&y);
	for(i=1;i<=n;i++)
	{
		ans+=arr[i];
		if(ans>=x && ans<=y && (sum-ans)>=x && (sum-ans)<=y){
			printf("%d\n",i+1);
			return 0;
		}
	}
	printf("0\n");
}
