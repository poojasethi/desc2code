#include<iostream>

using namespace std;

int main()
{
	int n,x,y,c=0,d=0;
	cin>>n;
	int*a=new int [n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		c+=a[i];
	}
	cin>>x>>y;
	for(int i=0;i<n;i++)
	{
		d+=a[i];
		if(d>=x&&d<=y&&c-d>=x&&c-d<=y)
		{
			cout<<i+2<<endl;
			return 0;
		}
	}
	cout<<"0"<<endl;
	
	return 0;
}
