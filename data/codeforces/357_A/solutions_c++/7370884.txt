#include<iostream>
#include<algorithm>
#include<math.h>
using namespace std;
long long a[100001],x,y,n,o,q,m;
int main(){
cin>>n;
for(int i=1;i<=n;i++){
	cin>>a[i];
	m+=a[i];
}
cin>>x>>y;
for(int i=1;i<=n;i++){
	o+=a[i];
	m-=a[i];
	if(o>=x&&o<=y&&m>=x&&m<=y)cout<<i+1,i=n+1,q=1;
}
if(q==0)cout<<"0";
}
