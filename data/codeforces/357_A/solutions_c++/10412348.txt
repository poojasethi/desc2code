#include<iostream>
using namespace std;
int main(){
int a[250],i,sum=0,sum1=0,m,x,y;
cin>>m;
for(i=0;i<m;i++){
    cin>>a[i];
    sum+=a[i];
}
cin>>x>>y;
sum1=0;
for(i=0;i<m;i++){
    if(sum1>=x && sum1<=y && (sum-sum1)>=x && (sum-sum1)<=y){
        cout<<i+1;
        return 0;
    }
    sum1+=a[i];
}
cout<<"0";
return 0;
}
