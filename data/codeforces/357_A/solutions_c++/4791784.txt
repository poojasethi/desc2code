#include <iostream>
using namespace std;
int main()
{
	int m, x, y, sum=0, k=0, i, S=0;
	cin >> m;
	int c[m];
	for(i=0; i<m; i++) 
	{
		cin >> c[i];
		S+=c[i];
	}
	cin >> x >> y;
	for(i=0; i<m; i++)
	{
		sum+=c[i];
		if(sum>=x && sum<=y && S-sum>=x && S-sum<=y)
		{
			k=i+2;
			break;
		}
	}
	cout << k << "\n";
}
