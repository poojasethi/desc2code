#include <bits/stdc++.h>
using namespace std;
int ar[105];
int main(){
	int n,i,j,k,x,y,temp,sum=0;cin>>n;
	for(i=1;i<=n;++i)cin>>ar[i],sum+=ar[i];
	cin>>x>>y;
	temp=0;
	for(i=1;i<=n;++i){
		if(x<= temp && temp<= y && x<= (sum-temp)&& (sum-temp)<= y){
			cout<<i<<endl;return 0;
		}
		temp+=ar[i];
	}
	cout<<0<<endl;
}
