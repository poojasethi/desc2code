#include <iostream>
using namespace std;
int c[110],n,i,m,k,d[110],x,y;
int main()
{
	cin>>m;
	for(i=1;i<=m;i++)
	{
		cin>>c[i];
		d[i]=d[i-1]+c[i];
	}
	cin>>x>>y;
	for(i=1;i<=m;i++)
	{
		if((d[i-1]<=y && d[i-1]>=x) && (d[m]-d[i-1]<=y && d[m]-d[i-1]>=x))
		{
			cout<<i<<endl;
			return 0;
		}
	}
	cout<<"0\n";	
	return 0;
}
