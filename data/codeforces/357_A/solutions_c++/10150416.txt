#include<stdio.h>
main()
{
	int m,k,i,j=0,x,y,s=0;
	int c[110];
	scanf("%d",&m);
	for(i=0;i<m;i++){scanf("%d",&c[i]);s+=c[i];}
	scanf("%d%d",&x,&y);
	for(i=0;i<m;i++){
		j+=c[i];
		if(j>=x){
			if(j>y)k=0;
			else if((s-j)<x||(s-j)>y)k=0;
			else {k=i+2;break;}
		}k=0;
	}
	printf("%d",k);
}
