#include<stdio.h>
	
int main(){
	int i,m,n,s,x,y,a[101];
	scanf("%d",&m);
	for(i=1,n=s=0;i<=m;i++) scanf("%d",a+i),n+=a[i];
	scanf("%d%d",&x,&y);
	for(i=1;i<=m;i++){
		s+=a[i];
		if(s>=x&&s<=y&&n-s>=x&&n-s<=y) break;
	}
	if(i<=m) printf("%d\n",i+1);
	else printf("0\n");
	return 0;
}
