#include <stdio.h>
#include <stdlib.h>

int main() {
  int n, q=0, t=0; scanf("%d", &n);
  int s[n], x, y; for (int i = 0; i < n; ++i) {scanf("%d", &s[i]); t+=s[i];} scanf("%d%d", &x, &y);
  for (int i=0; i<n; ++i) if (q<=y&&q>=x&&t-q<=y&&t-q>=x) { printf("%d\n", i+1); exit(0); } else q+=s[i];
  puts("0");
}
