#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int n,a[200],x,y,s=0,sa=0,i;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		sa+=a[i];
	}
	cin>>x>>y;
	for(i=0;i<n;i++)
	{
		s+=a[i];
		if(s>=x && sa-s<=y && sa-s>=x && s<=y)
		{
			cout<<i+2<<endl;
			return 0;
		}
	}
	cout<<0<<endl;
}
