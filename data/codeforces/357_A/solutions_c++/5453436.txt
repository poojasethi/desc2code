#include<cstdio>
int n,s,mi,ma;
int a[111];

int main() {
	scanf("%d",&n);
	s=0;
	for(int i=0;i<n;i++) {
		scanf("%d", &a[i]);
		s+=a[i];
	}
	scanf("%d%d",&mi,&ma);
	int l=0;
	for(int i=0;i<n;i++){
		l+=a[i];
		if(l>=mi&&l<=ma&&s-l>=mi&&s-l<=ma){
			printf("%d\n",i+2);return 0;
		}
	}
	printf("0\n");

	return 0;
}
