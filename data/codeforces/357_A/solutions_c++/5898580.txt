#include <iostream>

using namespace std;

long n,i,a[1000],g1 = 0,g2,x,y, ans = 0;

int main()
{
	cin >> n;
	for (i=1;i<=n;i++) {
		cin >> a[i];
		g2 += a[i];
	}
	cin >> x >> y;
	
	for (i=1;i<=n;i++){
		g1 += a[i-1];
		g2 -= a[i-1];
		if (g1 >= x && g1 <=y && g2>=x && g2<=y) ans = i;
	}
	cout << ans;
	return 0;
}
