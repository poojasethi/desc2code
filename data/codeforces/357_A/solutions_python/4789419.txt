import sys

inp = [map(int, i.split(' ')) for i in sys.stdin.read().strip().splitlines()]

si = inp[1]
x, y = inp[2]

res = 0
for ct in xrange(0, len(si)):
    s1 = sum(si[ct:])
    s2 = sum(si[:ct])
    if x <= s1 <= y and x <= s2 <= y:
        res = ct + 1
        break

sys.stdout.write(str(res))
