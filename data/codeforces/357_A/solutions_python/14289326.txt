n = int(raw_input())
c = map(int, raw_input().split())
sc = sum(c)
x, y = map(int, raw_input().split())
r = range(x, y+1)
sm = 0
ans = 0
for i in range(n):
    sm += c[i]
    if sm < x: continue
    if sm in r:
        if sc - sm in r:
            ans = i + 2
            break
        else:
            continue
    else:
        break
print ans
