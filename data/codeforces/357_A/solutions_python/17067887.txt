#!/usr/bin/python

m=int(input())
r = map(int,raw_input().split())
x,y = map(int,raw_input().split())

def solve():
	a=0
	b=0
	for i in xrange(1,m+1):
		a=0
		b=0
		for j in xrange(len(r)):
			if j+1>=i:
				a+=r[j]
			
			else:
				b+=r[j]

		if a>=x and a<=y and b>=x and b<=y:
			return i
	return 0

print solve()
