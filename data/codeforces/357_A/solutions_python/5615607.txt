m = input()
stu = map(int, raw_input().split())
x, y = map(int, raw_input().split())
s, count, h = sum(stu), 0, 0
sol = False
for i in stu:
    h += i
    s -= i
    count += 1
    if (h >= x and s >= x) and (h <= y and s <= y):
        print count + 1
        sol = True
        break

if not sol:
    print 0
