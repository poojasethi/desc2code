m = int(raw_input())
c = map(lambda x: int(x), raw_input().split(' '))
x, y = map(lambda x: int(x), raw_input().split(' '))

s_r = sum(c)
s_l = 0

for i in xrange(len(c)):
	if x <= s_r and s_r <= y and x <= s_l and s_l <= y:
		print i+1
		break
	else:
		s_l += c[i]
		s_r -= c[i]

if not ( x <= s_r and s_r <= y and x <= s_l and s_l <= y ):
	print 0
