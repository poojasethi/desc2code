n = int(input())
a = [int(t) for t in raw_input().split()]
x, y = map(int, raw_input().split())
cnt = 0
s = sum(a)
ans = 0
for index, z in enumerate(a):
    if cnt > y:
        break
    if ((x <= cnt) and (cnt <= y)) and ((x <= s - cnt) and (s - cnt <= y)):
        ans = index + 1
        break
    cnt += z

print ans
