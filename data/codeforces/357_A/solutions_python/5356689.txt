import sys,math

n=int(sys.stdin.readline())

c=[int(x) for x in sys.stdin.readline().split()]
hulp=[int(x) for x in sys.stdin.readline().split()]

x,y=hulp[0],hulp[1]
klas=sum(c)
som=0
grens=0

for i in range(n):
    som+=c[i]
    if som>=x and som<=y:
        if klas-som>=x and klas-som<=y:
            grens=i+2

print grens
