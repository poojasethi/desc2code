m = int(raw_input())
c = map(int,raw_input().split())
x, y = map(int,raw_input().split())
cnt = sum(c)
s = 0
r = 0
for i in range(m):
    s += c[i]
    if s >= x and s <= y and (cnt - s) >= x and (cnt - s) <= y:
        r = i + 2
        break
print 0 if r == -1 else r
