n=int(input())
ar=map(int,raw_input().split())
x,y=map(int,raw_input().split())
ar1=[]
ar1+=ar
for i in range(1,n):
    ar1[i]+=ar1[i-1]
ans=0   
for i in range(0,n-1):
    a1=ar1[i]
    a2=ar1[n-1]-ar1[i]
    if a1>=x and a1<=y and a2<=y and a2>=x:
        ans=i+1
if ans!=0:
    print ans+1
else:
    print ans
