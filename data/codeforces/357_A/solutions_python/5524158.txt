n = input()
ls = map(int,raw_input().split())
ts = sum(ls)
x,y = map(int,raw_input().split())

sum = 0
check = False
for i in range(len(ls)):
    sum += ls[i]
    k = ts-sum
    if sum>=x and sum<=y and k>=x and k<=y:
        print(i+2)
        check = True
        break

if check==False:
    print(0)
