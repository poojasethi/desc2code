m=int(input())
c=list(map(int,raw_input().split()))
x,y=map(int,raw_input().split())
s=[0]*(m+1)
for i in range(1,m+1):
    s[i]=s[i-1]+c[i-1]
#print(s)
result=0
for i in range(1,m+1):
    if s[i-1]<x:
        continue
    if s[i-1]>y:
        break
    t=s[m]-s[i-1]
    if t>=x and t<=y:
        result=i
        break
print(result)
