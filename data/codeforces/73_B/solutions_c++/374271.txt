int b[maxn];

string x;
char s[1<<20];
pair<int,string> a[maxn];
pair<int,string> aa[maxn];
int id;

int main()
{
  scanf("%d", &n); gets(s);
  forn (i, n)
  {
    gets(s);
    istringstream sin(s);
    string name; int x;
    sin >> name >> x;
    a[i] = mp(-x, name);    
  }
  scanf("%d", &m);
  forn (i, m) scanf("%d", b+i);
  gets(s);
  gets(s); x = s;
  
  sort(a, a+n);
  while (m < n) b[m++] = 0;
  int nn = 0;
  forn (i, n) if (x == a[i].second)
  {
     id = i;
     aa[nn++] = a[i];
     break;
  }

  forn (i, n) if (i != id) aa[nn++] = a[i];
  sort(b, b+n);// reverse(b, b+n);

  int place1 = 0;
  aa[0].first -= b[n-1];
  int l = 0, r = n-2;
  for (int i=1; i<n; ++i)
  {
    pair<int,string> cur = aa[i];
    cur.first -= b[l];
    if (aa[0] < cur)
    {
      ++l;
    }
    else
    {
      place1++;
      --r;
    }
  }

  sort(a, a+n); reverse(a, a+n);
  forn (i, n) if (x == a[i].second ) id = i;

//  printf("sc = %d\n", aa[n-1].first);
  for (int i=0; i<n-1; ++i)
  {
    pair<int,string> cur = aa[i];
    cur.first -= b[l];
    if (cur < aa[n-1])
    {
      ++l;
    }
    else
    {
      --place2;
      --r;
    }
  }
  printf("%d %d\n", place1+1, place2+1);  
  
  
    
  return 0;
}
