using namespace std;
int n,m,s[100500],strr;
string str;
struct node
{
       int sc;
       string nam;
       node(const string &ss="",int aa=0)
       {sc=aa;nam=ss;}
       bool operator <(const node &r)const {
            if(sc!=r.sc)return sc<r.sc;
            return nam>r.nam;
       }
}a[100500];
int hst()
{
    int lr=1,ans=1;
    for (int i=n-1;i>0;i--)
    {
       //
        if(node(a[lr].nam,a[lr].sc+s[i])<node(str,strr+s[n]))lr++;
        else ans++;
    }return ans;
}
int lst()
{
    int rr=n-1,ans=n;
    for (int i=2;i<=n;i++)
    { //cout<<a[rr].nam<<' '<<a[rr].sc+s[i]<<endl;
        if(node(str,strr+s[1])<node(a[rr].nam,a[rr].sc+s[i]))rr--;
        else ans--;
    }return ans;
}
int main()
{
    ios::sync_with_stdio(0);
    cin>>n;
    for (int i=1;i<=n;i++)
    {
        cin>>a[i].nam>>a[i].sc;
    }
    cin>>m;
    for (int i=1;i<=m;i++)
    {
        cin>>s[i];
    }
    cin>>str;
    for (int i=1;i<=n;i++)
    {
        if(a[i].nam==str){strr=a[i].sc;a[i].sc=-10;}
    }a[0].sc=100000000;
    a[0].nam="zzzzz";
    sort(a,a+n+1);
    sort(s+1,s+n+1);
    cout<<hst()<<' '<<lst()<<endl;
    //system("pause");
    return 0;
}
