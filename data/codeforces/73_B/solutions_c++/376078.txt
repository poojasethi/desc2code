#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n,o[100005],m,b[100005];
double d,a[100005],A;
char s[100005][25],S[25];

bool f1(int i, int j)
{
    return strcmp(s[i],s[j])>0;
}

int hi(double A)
{
    for (int i=0,j=n-2; i<n-1; i++)
    {
        while ((j>=0)&&(a[i]+b[j]>A))
        {
//printf("%lf %lf\n",a[i]+b[j],A);
            j--;
        }
        if (j<0)
        {
            return n-i;
        }
        j--;
    }
    return 1;
}

int lo(double A)
{
    for (int i=n-2,j=1; i>=0; i--)
    {
        while ((j<=n-1)&&(a[i]+b[j]<A))
        {
            j++;
        }
        if (j>n-1)
        {
            return (n-i-2)+1;
        }
        j++;
    }
    return n;
}

int main()
{
    scanf("%d",&n);
    d=1./n;
    for (int i=0; i<n; i++)
    {
        scanf("%s%lf",s[i],&a[i]);
        o[i]=i;
    }
    sort(o,o+n,f1);
    for (int i=0; i<n; i++)
    {
        a[o[i]]+=i*d;
    }
    scanf("%d",&m);
    for (int i=0; i<m; i++)
    {
        scanf("%d",&b[i]);
    }
    for (int i=m; i<n; i++)
    {
        b[i]=0;
    }
    sort(b,b+n);
    scanf("%s",S);
    for (int i=0; i<n; i++)
    {
        if (strcmp(s[i],S)==0)
        {
            A=a[i];
            for (int j=i+1; j<n; j++)
            {
                a[j-1]=a[j];
            }
            break;
        }
    }
    sort(a,a+n-1);
/*for (int i=0; i<n-1; i++)
{
    printf("%lf ",a[i]);
}printf("\n");
for (int i=0; i<n; i++)
{
    printf("%d ",b[i]);
}printf("\n");*/
    printf("%d %d\n",hi(A+b[n-1]),lo(A+b[0]));
    return 0;
}
