#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

const int maxn=100000+10;
string myteam;
string name[maxn];
int a[maxn];
int b[maxn];
int c[maxn];
int n,m,ma;

int main()
{
   // freopen("input.txt","r",stdin);

    cin >> n;
    for (int i=1;i<=n;i++) cin >> name[i] >> a[i];
    cin >> m;
    for (int i=1;i<=m;i++) cin >> b[i];
    cin >> myteam;
    int nn=n;n=0;
    for (int i=1;i<=nn;i++)
    if (name[i]==myteam) ma=a[i];
    else {name[++n]=name[i];a[n]=a[i];}
    sort(b+1,b+1+m);
    
    int mm=m,mma=ma;
    ma+=b[m];mm--;
    for (int i=1;i<=n;i++)
    if (name[i]<myteam) c[i]=ma-a[i];
    else c[i]=ma+1-a[i];
    sort(c+1,c+1+n);
    int res=0,now;
    for (now=1;now<=n;now++)
    if (c[now]<=0) {mm--;res++;}
    else break;
    for (int i=n;i>=now;i--)
    if (mm>0) res+=(c[i]<=b[mm--]);
    else res+=(c[i]<=0);
    printf("%d ",res+1);

    res=0;ma=mma;
    
    if (m==n+1)
    {
        ma+=b[1];
        for (int i=2;i<=m;i++) b[i-1]=b[i];
        m--;
    }
    for (int i=1;i<=n;i++)
    if (name[i]<myteam) c[i]=ma-a[i];
    else c[i]=ma+1-a[i];
    sort(c+1,c+1+n);
    for (int i=1,j=1;i<=n;i++)
    if (c[i]<=0) res++;
    else
    {
        while (j<=m && b[j]<c[i]) j++;
        if (j<=m) {res++;j++;}
    }
    printf("%d\n",res+1);
}
