#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
const int mm=111111;
char name[mm][22],word[22];
int p[mm],a[mm],b[mm],c[mm],g[mm];
int i,j,k,n,m,id;
bool cmp(int u, int v)
{
    return a[u]>a[v]||(a[u]==a[v]&&strcmp(name[v],name[u])>0);
}
bool cmp2(int u, int v)
{
    return c[u]>c[v]||(c[u]==c[v]&&strcmp(name[v],name[u])>0);
}
int main()
{
    while(~scanf("%d",&n))
    {
        for(i=0;i<n;++i)
        {
            scanf("%s%d",name[i],&a[i]);
            p[i]=i;
        }
        sort(p,p+n,cmp);
        scanf("%d",&m);
        for(i=0;i<m;++i)
            scanf("%d",&b[i]);
        while(m<n)b[m++]=0;
        scanf("%s",word);
        for(i=0;i<n;++i)
            if(strcmp(name[i],word)==0)break;
        id=i;
        sort(b,b+m);


        for(i=0;i<n;++i)c[i]=a[i],g[i]=p[i];
        c[id]+=b[m-1];
        for(j=m-2,k=0,i=0;i<n;++i)
            if(p[i]!=id)
            {
                c[p[i]]+=b[k];
                if(cmp2(p[i],id))
                    c[p[i]]+=b[j--]-b[k];
                else ++k;
            }
        sort(g,g+n,cmp2);
        //for(i=0;i<n;++i)printf("%s %d\n",name[g[i]],c[g[i]]);
        for(i=0;i<n;++i)
            if(g[i]==id)break;
        printf("%d ",i+1);

        for(i=0;i<n;++i)c[i]=a[i],g[i]=p[i];
        c[id]+=b[0];
        for(i=n-1,k=m-1,j=1;i>=0;--i)
            if(p[i]!=id)
            {
                c[p[i]]+=b[k];
                if(cmp2(id,p[i]))
                    c[p[i]]+=b[j++]-b[k];
                else --k;
            }
        sort(g,g+n,cmp2);
        //for(i=0;i<n;++i)printf("%s %d\n",name[g[i]],c[g[i]]);
        for(i=0;i<n;++i)
            if(g[i]==id)break;
        printf("%d\n",i+1);
    }
    return 0;
}
