def readint(): return int(raw_input())
def readfloat(): return float(raw_input())
def readarray(N, foo):
        res = []
        for _ in xrange(N):
                res.append(foo())
        return res
def readlinearray(foo): return map(foo, raw_input().split())

MAX_N=51
POINTS = {}
p = [0, 25, 18, 15, 12, 10, 8, 6, 4, 2, 1]
for i in xrange(len(p)):
    POINTS[i] = p[i]

def winner_rule1(drivers):
    ranking = []
    for d, r in drivers.iteritems():
        ranking.append((r, d))
    ranking.sort()
    return ranking[-1][1]
    
def winner_rule2(drivers):
    for result in drivers.values():
        result[0], result[1] = result[1], result[0] 
    return winner_rule1(drivers)

def calc_points(drivers):
    for result in drivers.values():
        for i in xrange(1, len(result)):
            result[0] += result[i] * POINTS.get(i, 0)

if __name__ == '__main__':
    T = input() #the number of races
    drivers = {}
    for t in xrange(T):
        N = input() #the number of clasified drivers in the given race
        for n in xrange(1, N+1):
            driver = raw_input()
            if not driver in drivers:
                drivers[driver] = MAX_N * [0]
            result = drivers[driver]
            result[n] += 1
    calc_points(drivers)
    print winner_rule1(drivers)
    print winner_rule2(drivers)
