#!/usr/bin/python
import sys
MaxN = 50
NA = -1
score = [25, 18, 15, 12, 10, 8, 6, 4, 2, 1] + [0] * 40
n = int (sys.stdin.readline ())
s = {}
places = [{} for i in range (50)]
for i in range (n):
	k = int (sys.stdin.readline ())
	for j in range (k):
		t = sys.stdin.readline ().strip ()
		if not t in s:
			s[t] = 0
		s[t] += score[j]
		if not t in places[j]:
			places[j][t] = 0
		places[j][t] += 1
for t in s:
	for j in range (MaxN):
		if not t in places[j]:
			places[j][t] = 0

def cmp1 (x, y):
	if s[x] > s[y]:
		return 1
	elif s[x] < s[y]:
		return -1
	for j in range (MaxN):
		if places[j][x] > places[j][y]:
			return 1
		elif places[j][x] < places[j][y]:
			return -1
	return 0

def cmp2 (x, y):
	j = 0
	if places[j][x] > places[j][y]:
		return 1
	elif places[j][x] < places[j][y]:
		return -1
	return cmp1 (x, y)

t = s.keys ()
t.sort (cmp1)
print t[-1]
t.sort (cmp2)
print t[-1]
