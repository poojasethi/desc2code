#include<stdio.h>
#include<cmath>
#include<cstring>
int main()
{
	int d[101][101],n,i,j,m,t,k,ans1,ans2,a[101]={0, 25, 18, 15, 12, 10, 8, 6, 4, 2, 1},b[101]={0};
	char s[101],c[101][101];
	bool tt;
	scanf("%d",&n);
	t=0; tt=true;
	for (i=1;i<=n;i++)
	{
		scanf("%d",&m);
		for (j=1;j<=m;j++)
		{
			scanf("%s",s); tt=false;
			for (k=1;k<=t;k++) if (strcmp(s,c[k])==0) {b[k]=b[k]+a[j]; d[k][j]++; tt=true;}
			if (!tt) 
			{
				t++;
				strcpy(c[t],s);
				b[k]=b[k]+a[j]; 
				d[k][j]++;
			}
		}
	}
	m=0; 
	for (i=1;i<=t;i++) if (b[i]>m) {m=b[i]; ans1=i;}
	else if (b[i]==m)
	{
		for (j=1;j<=50;j++) 
			if (d[i][j]>d[ans1][j])
			{
				ans1=i; break;
			}
			else if (d[i][j]<d[ans1][j]) break;
	}
	m=0; k=0;
	for (i=1;i<=t;i++) if (d[i][1]>k) {k=d[i][1]; m=b[i]; ans2=i;}
	else if (d[i][1]==k)
	if (b[i]>m) {m=b[i]; ans2=i;}
	else if (b[i]==m)
	{
		for (j=1;j<=50;j++) 
			if (d[i][j]>d[ans2][j])
			{
				ans2=i; break;
			}
			else if (d[i][j]<d[ans1][j]) break;
	}
	printf("%s\n%s\n",c[ans1],c[ans2]);
}
