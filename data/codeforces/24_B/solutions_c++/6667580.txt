#include <iostream>
#include <map>
using namespace std;
const int score[10]={25,18,15,12,10,8,6,4,2,1};
struct runner{
	int p,w[50];
};
map<string,runner> x;
bool betterx(const runner&h1,const runner&h2){
	if (h1.p!=h2.p) return h1.p>h2.p;
	for(int i=0;i<50;i++){
		if (h1.w[i]!=h2.w[i]) return h1.w[i]>h2.w[i];
	}
	return false;
}
bool bettery(const runner&p1,const runner&p2){
	if (p1.w[0]!=p2.w[0]) return p1.w[0]>p2.w[0];
	return betterx(p1,p2);
}
int main(){
	int t,n;
	cin>>t;
	for(int i=1;i<=t;i++){
		cin>>n;
		for(int j=0;j<n;j++){
			string s;
			cin>>s;
			if (j<10)
				x[s].p+=score[j];
			x[s].w[j]++;
		}
	}
	string w1=x.begin()->first,w2=x.begin()->first;
	for(map<string,runner>::iterator it=x.begin();it!=x.end();it++){
		if (betterx(it->second,x[w1])) w1=it->first;
		if (bettery(it->second,x[w2])) w2=it->first; 
	}
	cout<<w1<<endl;
	cout<<w2<<endl;
	return 0;
}
