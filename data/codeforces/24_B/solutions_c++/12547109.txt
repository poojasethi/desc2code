#include <iostream>
#include <algorithm>
#include <map>
#include <string>
using namespace std;
int t,n,m,d[55],sc[55],p[55][55];
int c[55]={25,18,15,12,10,8,6,4,2,1};
map<string,int> id;
string names[55];

bool cmp(int a,int b){
	if(sc[a]>sc[b])return true;
	else if(sc[a]<sc[b])return false;
	else{
		for(int i=0;i<50;i++){
			if(p[a][i]>p[b][i])return true;
			if(p[a][i]<p[b][i])return false;
		}
	}
	return true;
}

bool cmp2(int a,int b){
	if(p[a][0]>p[b][0])return true;
	if(p[a][0]<p[b][0])return false;
	else{
		if(sc[a]>sc[b])return true;
		else if(sc[a]<sc[b])return false;
		else{
			for(int i=1;i<50;i++){
				if(p[a][i]>p[b][i])return true;
				if(p[a][i]<p[b][i])return false;
			}
		}
	}
	return true;
}

int main(){
	int x;
	string s;
	cin>>t;
	for(int i=0;i<t;i++){
		cin>>m;
		for(int j=0;j<m;j++){
			cin>>s;
			if(id.find(s)==id.end()){
				d[n]=n;
				names[n]=s;
				id[s]=n++;
			}
			x=id[s];
			sc[x]+=c[j];
			p[x][j]++;
		}
	}
	sort(d,d+n,cmp);
	cout<<names[d[0]]<<endl;
	sort(d,d+n,cmp2);
	cout<<names[d[0]]<<endl;
	return 0;
}
