#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
int a[1005][55];
int ans[1005];
map<string,int> m;
string ms[1005];
int point,n;
int b[]={25,18,15,12,10,8,6,4,2,1};
int fun(int x,int y)
{
	if(ans[x]!=ans[y])
	{
		return ans[x]>ans[y];
	}
	else
	{
		for(int i=0;i<55;i++)
		{
			if(a[x][i]!=a[y][i])
			{
				return a[x][i]>a[y][i];
			}
		}
		return 0;
	}
}
int fun2(int x,int y)
{
	if(a[x][0]!=a[y][0])
	{
		return a[x][0]>a[y][0];
	}
	else
	{
		return fun(x,y);
	}
}
int main()
{
	int t;
	cin>>t;
	for(int ii=0;ii<t;ii++)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		{
			string s;
			cin>>s;
			if(m.find(s)==m.end())
			{
				m[s]=point++;
				ms[point-1]=s;
			}
			a[m[s]][i]++;
			if(i<=9)
			ans[m[s]]+=b[i];
		}
	}
	int res[1005];
	for(int i=0;i<m.size();i++)
	{
		res[i]=i;
	}
	sort(res,res+m.size(),fun);
	cout<<ms[res[0]]<<endl;
	sort(res,res+m.size(),fun2);
	cout<<ms[res[0]]<<endl;
}
