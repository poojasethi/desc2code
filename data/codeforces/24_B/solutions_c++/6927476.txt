#include<stdio.h>
#include<string.h>
#include<string>
#include<map>
#include<algorithm>
using namespace std;
struct Race
{
	char name[55];
	int score,rac[55];
}r[1000];
int cmp1(Race x,Race y)
{
	int i;
	if(x.score==y.score)
		for(i=0;i<50;i++)
			if(x.rac[i]!=y.rac[i])
				return x.rac[i]>y.rac[i];
	return x.score>y.score;
}
int cmp2(Race x,Race y)
{
	int i;
	if(x.rac[0]==y.rac[0])
	{
		if(x.score==y.score)
		{
			for(i=1;i<50;i++)
				if(x.rac[i]!=y.rac[i])
					return x.rac[i]>y.rac[i];
		}
		else
			return x.score>y.score;
	}
	return x.rac[0]>y.rac[0];
}
map<string,int> m;
int main()
{
	char str[55];
	int t,n,i,j,k,num,p[11]={25,18,15,12,10,8,6,4,2,1};
	while(scanf("%d",&t) != EOF)
	{
		num=1;
		memset(r,0,sizeof(r));
		for(i=0;i<t;i++)
		{
			scanf("%d",&n);
			for(j=0;j<n;j++)
			{
				scanf("%s",str);
				if(m[str]==0)
					m[str]=num++;
				strcpy(r[m[str]].name,str);
				if(j<10)
					r[m[str]].score+=p[j];
				r[m[str]].rac[j]++;
			}
		}
		sort(r,r+num,cmp1);
		printf("%s\n",r[0].name);
		sort(r,r+num,cmp2);
		printf("%s\n",r[0].name);
	}
	return 0;
}
