#include <stdlib.h>
#include <vector>
#include <string>
#include <map>
#include <stdio.h>
using namespace std;
int pt[]={25, 18, 15, 12, 10, 8, 6, 4, 2, 1};

map<string,int> mp;
string name[111];
vector<int> sc[100];
int sz;
char buf[110];
int main()
{
  int t;
  scanf("%d",&t);
  while (t--)
  {
    int n;
    scanf("%d",&n);

    string s;
    for (int i=0; i < n; i++)
    {
      scanf("%s",buf);
      s=buf;
      if(mp.find(s)==mp.end())
      {
        name[sz]=s;
        mp[s]=sz;
        sc[sz]=vector<int> (100,0);
        sz++;
      }
      int j=mp[s];
      if(i<10)sc[j][0]+=pt[i];
      sc[j][i+1]++;
    }
  }

  int a0=0;
  vector<int> mx=sc[a0];

  for (int i=0; i < sz; i++)
  {
    if(sc[i]>mx)
    {
      mx=sc[i];
      a0=i;
    }
  }

  for (int i=0; i < sz; i++)
  {
    swap(sc[i][0],sc[i][1]);
  }


  int a1=0;
  mx=sc[a1];

  for (int i=0; i < sz; i++)
  {
    if(sc[i]>mx)
    {
      mx=sc[i];
      a1=i;
    }
  }

  printf("%s\n%s\n",name[a0].c_str(),name[a1].c_str());

  return (0);
}
