    #include <iostream>
    #include <string>
    #include <set>  
    #include <algorithm>
    #include <vector>
    #include <cmath>
    #include <queue>
    #include <cstring>
    #include <cstdio>
    #include <map>
    #include <deque>
using namespace std;
map <string,int> M;
pair<vector<int>,string>s1[1111];
pair<vector<int>,string>s2[1111];
int sc[55]={25,18,15,12,10,8,6,4,2,1};
int main()
{
	for(int i=0;i<1111;i++)
		for(int j=0;j<55;j++)
			{
				s1[i].first.push_back(0);
				s2[i].first.push_back(0);
			}
	int t,n,cnt,k;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		{
			string s;
			cin>>s;
			
			if(M.count(s)==0)M[s]=cnt++;
			k=M[s];
			s1[k].second=s;
			s2[k].second=s;
			s1[k].first[0]-=sc[i];
			s2[k].first[1]-=sc[i];
			if(i==0){
				s1[k].first[1]--;
				s2[k].first[0]--;
			}
			else {
				s1[k].first[i+1]--;
				s2[k].first[i+1]--;
			}
		}
		
	}
		sort(s1,s1+cnt);
		sort(s2,s2+cnt);
		cout<<s1[0].second<<endl;
		cout<<s2[0].second<<endl;
		return 0;
}
