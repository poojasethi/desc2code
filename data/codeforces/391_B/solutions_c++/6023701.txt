#include<stdio.h>
#include<algorithm>
#include<string.h>
#include<vector>
using namespace std;

vector<int>g[26];
int num[26]={0};

int main()
{
    char s[1005];
    scanf("%s",s);
    int n;
    n=strlen(s);
    for(int i=0;i<26;i++)
        num[i]=1;
    for(int i=0;i<n;i++)
    {
        g[s[i]-'A'].push_back(i);
    }
    for(int i=0;i<26;i++)
    {
        int k=g[i].size();
        for(int j=1;j<k;j++)
        {
            if((g[i][j]-g[i][0])%2==1&num[i]%2==1)
            {
               num[i]++;
            }
            else if((g[i][j]-g[i][0])%2==0&num[i]%2==0)
                num[i]++;

        }

    }
    sort(num,num+26);
    printf("%d",num[25]);

}
