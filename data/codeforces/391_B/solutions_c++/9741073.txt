#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
char s[1005];
int len,ans=1;
int main(){
    cin.getline(s,1005);
    len=strlen(s);
    for(int i=0;i<len;i++){
        int cur=1;
        for(int j=i+1;j<len;j+=2){
            if(s[j]==s[i]){
                cur++;
                ans=max(ans,cur);
                j--;
            }
        }
    }
    printf("%d\n",ans);
    return 0;
}
