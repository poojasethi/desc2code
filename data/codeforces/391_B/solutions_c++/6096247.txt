#include<iostream>
using namespace std;
int main()
{
    string s;
    cin>>s;
    int res=0,a,b,c,j,i;
    for(char ch='A';ch<='Z';ch++)
    {
        i=0;
        while(i<s.length() && s[i]!=ch)
            i++;
        a=1;
        for(j=i+1;j<s.length();j++)
        {
            if(s[j]==ch)
            {
                if((j-i)%2)
                {
                    a++;
                    i=j;
                }
            }
        }
        if(a>res)
            res=a;
    }
    cout<<res;
    return 0;
}
