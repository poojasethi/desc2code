#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
char str[1000+100];
int main()
{
    int len,maxx;
    int i,j,k;
    while(scanf("%s",str)!=EOF)
    {
        maxx=1;
        len=strlen(str);
        for(i=0;i<len-1;i++)
        {
            int temp=1;
            int mark=i;
            for(j=i+1;j<len;j++)
            {
                if(str[i]==str[j]&&(j-mark)%2==1){ temp++; mark=j; }
            }
            if(temp>maxx) maxx=temp;
        }
        printf("%d\n",maxx);
    }
    return 0;
}
