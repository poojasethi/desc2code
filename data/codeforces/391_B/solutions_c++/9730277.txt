#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main(){ 
  ios_base::sync_with_stdio(0);
  string s;
  cin>>s;
  int n = s.size();
  int res = 0;
  for(int i=0; i<26; i++) {
    char c = 'A'+i;
    for(int o=0; o<2; o++) {
      int d = o;
      int cur = 0;
      int e = 0;
      while(e<n) {
        while(e<n && (e%2!=d || s[e]!=c)) e++;
        if(e<n) cur++;
        e++;
        d = 1-d;
      }
      res = max(res,cur);
    }
  }
  cout<<res<<endl;
  
  return 0;
}
