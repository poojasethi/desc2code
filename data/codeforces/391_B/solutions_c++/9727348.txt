#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define N 1000
char s[N + 5];
int dp[N + 5];
int ans[26][2];
int main()
{
    int mx=0,i,n;
    int cnt = 1;
    scanf("%s",s+1);
    n = strlen(s+1);
    for(i = n ; i>0 ; i--)
    {
        dp[i] = 1;
        for(int j = i+1 ; j<=n ; j++)
        {
            if(s[j] == s[i] && (j-i-1)%2 == 0)
            {
                dp[i] = max(dp[i],1+dp[j]);
            }
        }
        mx = max(mx,dp[i]);
    }
    printf("%d\n",mx);
    return 0;
}
