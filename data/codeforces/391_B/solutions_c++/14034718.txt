#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,an;
char p[N];
int main() {
	scanf("%s",p+1); n=strlen(p+1);
	For(i,1,n) {
		k=1;
		for (j=i+1;j<=n;j++) if (p[i]==p[j]) {
			if (!((k&1)^((j-i)&1))) k++;
		}
		an=max(an,k);
	}
	printf("%d\n",an);
	return 0;
}
