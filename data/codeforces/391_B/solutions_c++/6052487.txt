#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int occ[26][2];

int main()
{
	string s;
	cin >> s;

	for (int i = 0; i < s.length(); i++)
	{
		int num = s[i] - 'A';
		if (occ[num][0] == 0)
		{
			occ[num][0] = 1;
			occ[num][1] = i % 2;
		}
		else if (occ[num][1] != i % 2)
		{
			occ[num][0]++;
			occ[num][1] = i % 2;
		}
	}

	int maxlen = 0;
	for (int i = 0; i < 26; i++)
		maxlen = max(maxlen, occ[i][0]);

	cout << maxlen << "\n";
	return 0;
}
