#include <iostream>
using namespace std;
const int N = 1000;
int dp[N];
int main()
{
    string str = "";
    cin >> str;
    int n = str.length();
    int pile_size = 0;
    for ( int i = n - 1 ; i >= 0 ; i -- )
    {
        dp[i] = 1;
        for ( int j = i + 1 ; j < n ; j += 2 )
        {
            if ( str[i] == str[j] )
            {
                dp[i] = dp[j] + 1;
                break;
            }
        }
        pile_size = max(pile_size, dp[i]);
    }
    cout << pile_size << endl;
    return 0;
}
