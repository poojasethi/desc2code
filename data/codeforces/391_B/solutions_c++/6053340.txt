
#include <iostream>
#include <string>

using namespace std;

int fr[1001];

int main()
{
	
	int last = 0, ind = 0, maxi = 1, cl = 0, sum = 0;
	char c;
	string s;
	cin>>s;

	for (int i = 0; i < 26; i++) 
	{

		c = 'A' + i;
		ind = 0;

		for (int j = 0; j < s.length(); j++)
		{

			if (s[j] == c)
			{

				fr[ind] = last;
				last = 1;
				++ind;

			}
			else ++last;

		}

		for (int j = 1; j < ind; j++)
		{

			sum = 0;
			cl = 1;
			
			for (int h = j; h < ind; h++) { sum += fr[h]; if (sum % 2 == 1) { sum = 0; ++cl; } }

			if (cl > maxi) maxi = cl;

		}

	}

	cout<<maxi;

	return 0;

}
