#include <iostream>
#include <string>
#include <cmath>
using namespace std;

#define forn(i, n, k) for(int i = n; i < k; i++)

int mx = 0;
int dp[1001];
void operate(string input, int i, char obj, int acarr);

void operate(string input, int i, char obj, int acarr) {
	if (dp[i] == 0) {
		for (int j = i + 1; j < input.size(); j += 2) {
			if (input[j] == obj) {
				dp[i] = max(dp[i], acarr + 1);
				operate(input, j, obj, acarr + 1);
			}
		}
		if (mx < acarr) mx = acarr;
	} else  if (mx < dp[i]) mx = dp[i];
}

int main() {
	string input;
	cin >> input;
	
	int res = 1;
	
	forn(i, 0, 1001) dp[i] = 0;
	
	forn(i, 0, input.size()) {
		char curr = input[i];
		operate(input, i, curr, 1);
		if (mx > res) res = mx;
		mx = 0;
	}
	
	cout << res << endl;
	
	return 0;
}
