#include<cstdio>
#include<iostream>
#include<algorithm>
#include<string.h>
#include<math.h>
using namespace std;
char s[1005];
int main()
{
    cin>>s;
    int len=strlen(s);
    int maxx=0;
    for(int i=0;i<len;i++)
    {
        int tot=1;
        int now=i;
        int opp=now+1;
        while(opp<len)
        {
            if(s[opp]==s[now])
            {
                now=opp;opp=now+1;tot++;
                //cout<<now<<" "<<opp<<endl;
            }
            else opp+=2;
        }
        if(tot>maxx)maxx=tot;
    }
    cout<<maxx<<endl;
}
