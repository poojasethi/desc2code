#include<stdio.h>
#include<algorithm>

char s[1100];

int main() {
  int n;
  scanf("%s%n", s, &n);
  int mm=0;
  for (char c='A'; c<='Z'; c++) {
    int i=0;
    int m=0;
    while (i<n && s[i]!=c) { i++; }
    if (i>=n) { continue; }
    m++;
    i++;
    while (i<n) {
      while (i<n && s[i]!=c) { i+=2; }
      if (i>=n) { break; }
      m++;
      i++;
    }
    mm = std::max(mm, m);
  }

  printf("%d\n", mm);
  return 0;
}
