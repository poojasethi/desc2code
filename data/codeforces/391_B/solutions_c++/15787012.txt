#include <bits/stdc++.h>
using namespace std;

#define int long long

string s;
int n;

int f(int x, char c){

    int cnt = 0, ans = 0;
    for(int i = x; i < n; i++){
        if(s[i] == c && cnt % 2 == 0){
            cnt = 0;
            ans++;
        }
        else
            cnt++;
    }

    return ans;
}

main(){
    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> s;
    n = s.size();

    int ans = 1;
    for(int i = 0; i < 26; i++)
        for(int j = 0; j < n; j++){
            //cout << 'A' + i << ' ';
            if(s[j] == 'A' + i)
                ans = max(ans, f(j, 'A' + i));
            //cout << ans << '\n';
        }
    cout << ans;
    return 0;
}
