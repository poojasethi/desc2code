#include <stdio.h>
#include <string>
#include <string.h>
#include <cmath>
#include <set>
#include <queue>
#include <stack>
#include <vector>
#include <algorithm>
using namespace std;

bool vis[30];
char s[1024];

int main() {
	scanf("%s", s);

	int len = strlen(s);
	int cnt, ans = 0;

	for(int i = 0; i < len; i++) {
		int last = i;
		cnt = 1;

		if(vis[s[i] - 'A']) continue;
		vis[s[i] - 'A'] = true;

		for(int j = i + 1; j < len; j++) {
			if(((j - last) % 2 == 1) && (s[j] == s[last])) {
				last = j;
				cnt++;
			}
		}

		ans = max(ans, cnt);
	}

	printf("%d\n", ans);
	return 0;
}
