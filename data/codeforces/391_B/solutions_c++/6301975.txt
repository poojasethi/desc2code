#include <algorithm>
#include <iostream>
#include <string>
using namespace std;

int main() {
	ios_base::sync_with_stdio(false);
	string s; cin >> s;

	int result = 1;

	for (char c = 'A'; c <= 'Z'; c++) {
		int b = -1;
		int i = 0;
		for (; i < s.size(); i++) {
			if (s[i] == c) { b = i; break; }
		}
		if (b == -1) { continue; }
		int res = 1;
		for (i = b+1; i < s.size(); i++) {
			if (s[i] == c) {
				if ((i - b) & 1) {
					res += 1;
					b = i;
				}
			}
		}
		result = max(res, result);
	}

	cout << result << endl;

	return 0;
}
