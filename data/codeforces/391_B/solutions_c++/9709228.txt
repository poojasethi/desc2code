#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdio>
#include <queue>
#include <cmath>
using namespace std;
const int N = 111111;

char s[5111];
int vis[1111];

int main()
{
    cin>>s;
    int n = strlen(s);
    int ans = 0;
    for(int i = 0; i < n; i++) {
        int sum = 1, pos = i;
        for(int j = i+1; j < n; j++) {
                if(s[j] == s[i]) {
                  //  printf("!j = %d, pos = %d, sum = %d\n", j, pos, sum);
                    if((j-pos) % 2) {
                        sum ++;
                   //     printf("i = %d, pos = %d, sum = %d\n", i, pos, sum);
                        pos = j;
                    }
                }
        }
        ans = max(ans, sum);
    }

    cout << ans << endl;
    return 0;
}
