#include <bits/stdc++.h>
using namespace std;

int dp[1006];

int main()
{
    int n,i,j,d;
    string s;
    cin >> s;
    n=s.size();
    int max1=0;
    for(i=0;i<=n;i++)
        dp[i]=1;
    for(i=2;i<=n;i++)
    {
        for(j=ceil((double)(i+2)/2);j<=i;j++)
        {
            if(s[i-1]==s[2*j-2-i])
                dp[i]=max(dp[i],dp[2*j-1-i]+1);
        }
    }
    for(i=1;i<=n;i++)
        max1=max(max1,dp[i]);
    printf("%d\n",max1);
        
    return 0;
}
