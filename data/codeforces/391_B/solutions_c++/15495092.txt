#include<bits/stdc++.h>
#define ll long long

using namespace std;

const int MAXN=1e3+10;
int dp[MAXN];
vector<int>v[30];

int main(){
	string s;
	cin>>s;
	int mx=1;
	for(int i=0;i<s.size();i++)
		v[s[i]-'A'].push_back(i);
	for(int i=0;i<27;i++){
		for(int j=0;j<s.size();j++)
			dp[j]=1;
		for(int j=1;j<v[i].size();j++)
			for(int t=j-1;t>-1;t--)
				if((v[i][j]-v[i][t])%2==1)
					dp[j]=max(dp[j],dp[t]+1);		
		for(int j=0;j<v[i].size();j++)
			mx=max(mx,dp[j]);
	}
	cout<<mx<<endl;
	return 0;
}
