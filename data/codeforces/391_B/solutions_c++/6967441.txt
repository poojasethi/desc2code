#include <iostream>
using namespace std;

int main() {
    // your code goes here
    string s;
    int i,j,k,n,r=1,a[1002];
    cin>>s;
    n=s.length();
    a[n]=0;
    for(i=n-1;i>=0;i--)
    {
        for(j=i+1,k=0;j<n ;j++)
          if(s[i]==s[j] && (j-i)%2==1)
            k=max(k,a[j]);
        a[i]=k+1;
    }
    for(i=0;i<n;i++)
    {
        r=max(r,a[i]);
        //cout<<a[i]<<endl;
    }
    cout<<r<<endl;
    return 0;
}
