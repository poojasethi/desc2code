#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int MaxN = 1000 + 10;
int n, f[MaxN];
char s[MaxN];
int main()
{
	scanf("%s", s + 1);
	n = strlen(s + 1);
	for(int i = 1; i <= n; ++i)
		for(int j = i - 1; j >= 1; j -= 2)
			if(s[i] == s[j]) f[i] = max(f[i], f[j] + 1);
	int ans = 0;
	for(int i = 1; i <= n; ++i) if(f[i] > ans) ans = f[i];
	cout << ans + 1 << endl;
	return 0;
}
