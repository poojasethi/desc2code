#include <bits/stdc++.h>
#define mp make_pair
#define ll long long

using namespace std;

int main()
{
    string s;
    int ans= INT_MIN;
    cin >> s;
    for(int i = 0 ; i < s.length() ; i++)
    {
        int x = 1;
        int c = i;
        for(int j = i+1 ; j < s.length() ; j++)
        {
            if(s[c]==s[j] && (j-c-1)%2==0)
            {
                x++;
                c=j;
            }
        }
        ans = max(x , ans);
    }
    cout << ans << endl;
}
