#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int N=1e3+10;
char str[N];
int main()
{
    //freopen("cin.txt","r",stdin);
    while(~scanf("%s",str)){
        int len=strlen(str);
        int ans=0;
        for(int i=0;i<len;i++){
            int j=0,h=1,t=i,dex=t;
            while(str[dex]){
                if(dex!=i && str[dex]==str[i]){
                    h++;
                    t=dex;
                    j=0;
                }
                dex=2*j+1+t;
                j++;
            }
            ans=max(ans,h);
        }
        printf("%d\n",ans);
        memset(str,0,sizeof(str));
    }
    return 0;
}
