#include<iostream>
#include<vector>
#include<stdio.h>
using namespace std;
typedef struct Alpha{
 int cnt=1;
 vector<int> v;
}Alpha;
int main()
{
  Alpha A[26];
  int m=0;
  char ch,i=0,j;
  while((ch=getchar())!='\n')
      {
          A[ch-'A'].v.push_back(i);
          i++;
      }
    for(i=0;i<26;i++)
    {
        if(A[i].v.size()!=0)
        {

            while(A[i].v.size()!=1)
            {
                if(!((A[i].v.at(1)-A[i].v.at(0))%2==0))
                    A[i].cnt++;
                    A[i].v.erase(A[i].v.begin());
            }
            if(m<A[i].cnt)
                m=A[i].cnt;
        }
    }
    cout<<m;
    return 0;
}
