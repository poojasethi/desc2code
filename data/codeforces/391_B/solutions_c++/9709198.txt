#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int N=1000+5;
char s[N];
int dp[N],maxs;
int main(){
while(~scanf("%s",s)){
    maxs=1;
    int len=strlen(s);
    for(int i=0;i<len;i++){
        dp[i]=1;
        for(int j=1;i-j>=0;j+=2){
            if(s[i]==s[i-j]){
                dp[i]=max(dp[i-j]+1,dp[i]);
            }
            maxs=max(dp[i],maxs);
        }
    }
    printf("%d\n",maxs);
}

return 0;
}
