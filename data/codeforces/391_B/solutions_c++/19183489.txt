#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;

char a[1001];
int dp[1001];

int main()
{
    int ans=1;
    while(cin>>a)
    {
        for(int i=0;i<strlen(a);i++)
        {
            dp[i]=1;
            for(int j=1;i-j>=0;j+=2)
            {
                if(a[i]==a[i-j])
                {
                    dp[i]=max(dp[i],dp[i-j]+1);
                }
            }
            if(ans<dp[i])
                ans=dp[i];
        }
        cout<<ans<<endl;
    }
    return 0;
}
