
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char s[1100];
int dp[1100];
int main()
{
    while(~scanf("%s",s+1))
    {
        int len=strlen(s+1);
        memset(dp,0,sizeof(dp));
        int mm=0;
        for(int i=2;i<=len;i++)
        {
            for(int j=1;j<i;j++)
                if(s[i]==s[j]&&(i-j)%2)
                    dp[i]=max(dp[i],dp[j]+1);
        }
        for(int i=1;i<=len;i++)
            if(dp[i]>mm)
                mm=dp[i];
        printf("%d\n",mm+1);
        memset(s,0,sizeof(s));
    }
}
