#include <bits/stdC++.h>

using namespace std;
const int N=(1e+3)+10;

int dp[N]={0};

int main(){
    ios::sync_with_stdio(false);cin.tie(0);
    string s;

    cin>>s;
    int maxi=0;
    for(int i=s.length()-1;i>=0;i--){
        dp[i]=1;
        for(int j=i+1;j<s.length();j+=2){
            if(s[i]==s[j]){
                dp[i]=max(dp[i],dp[j]+1);
            }
        }
        //cout<<i+1<<": "<<dp[i]<<"\n";
        maxi=max(maxi,dp[i]);
    }
    cout<<maxi;
    return 0;
}
