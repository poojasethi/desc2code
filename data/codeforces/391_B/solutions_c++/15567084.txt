#include <iostream>
using namespace std;
main()
{
    string s;
    cin >> s;
    int max1=1;
    for(int i=0; i<26; i++){
        char a = 'A' + i;
        int x1=-1, t=1;
        for(int j=0; j<s.size(); j++){
            if(s[j]==a){
                if(x1==-1)x1=j;
                else {
                    if((j-x1) % 2==1){
                        t++;
                        x1=j;
                    }
                }
            }
        }
        max1=max(t, max1);
    }
    cout << max1;
}
