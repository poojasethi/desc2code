#include<bits/stdc++.h>
using namespace std;
pair<int,int> arr[10000];
int main(){
int n,a,b,i;
scanf("%d",&n);
for(i=0;i<n;i++)
	scanf("%d%d",&a,&b),arr[i]=make_pair(a,b);
sort(arr,arr+n);
int la=arr[0].second;
for(i=1;i<n;i++)
	if(arr[i].second<la)
		la=arr[i].first;
	else la=arr[i].second;
printf("%d",la);
return 0;
}
