#include <iostream>
#include <algorithm>
using namespace std;
int main() {
	int n, i, ans = 0;
	cin >> n; pair <int, int> ex[n];
	for(i=0; i<n; ++i) cin >> ex[i].first >> ex[i].second;
	sort(ex, ex+n);
	for(i=0; i<n; ++i){
		if(ex[i].second >= ans) ans = ex[i].second;
		else ans = ex[i].first;
	}
	cout << ans << endl;
}
