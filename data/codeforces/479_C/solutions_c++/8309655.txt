#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,i,ans;
	cin>>n;
	pair<int,int>a[15005];
	for(i=0;i<n;i++)
	{
		cin>>a[i].first;
		cin>>a[i].second;
	}
	sort(a,a+n);
	
	ans=a[0].second;
	for(i=1;i<n;i++)
	{
		if(a[i].second>=ans)
		ans=a[i].second;
		else
		ans=a[i].first;
	}
	cout<<ans<<"\n";
	return 0;
}
