#include <bits/stdc++.h>

using namespace std;

int main()
{
	pair<int, int> a[5000];
	int n, j=0;
	cin >> n;
	for(int i = 0; i < n; i++) cin >> a[i].first >> a[i].second;
	sort(a, a+n);
	for(int i = 0; i < n; i++)
		if(j <= a[i].second)
			j = a[i].second;
		else
			j = a[i].first;
	cout << j << endl;
}
