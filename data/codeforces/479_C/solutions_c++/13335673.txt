#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	pair < int ,int > arr[5001];
	int i;
	cin >> n;
	for(i=0;i<n;i++)
		cin >> arr[i].first >> arr[i].second;
	sort(arr,arr+n);
	int exam=0;
	for(i=0;i<n;i++)
	{
		if(arr[i].second>=exam)
			exam=arr[i].second;
		else
			exam=arr[i].first;

	}
	cout << exam << endl;
	return 0;
}
