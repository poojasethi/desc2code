#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,Min=0;scanf("%d",&n);
    pair<int,int>p[n];
    for(int i=0; i<n; i++)
    {scanf("%d%d",&p[i].first,&p[i].second);}
    sort(p,p+n);
    for(int i=0;i<n;i++)
    {if(Min>p[i].second){Min=p[i].first;}
    else{Min=p[i].second;}}
printf("%d",Min);
    return 0;}
