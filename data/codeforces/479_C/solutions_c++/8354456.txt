#include <iostream>
#include <algorithm>
#include <utility>

using namespace std;

int main()
{
	pair<int,int> arr[100005];
	int n; cin>>n;
	for(int i=0;i<n;i++) cin>>arr[i].first>>arr[i].second;
	sort(arr,arr+n);
	int ans=-1;
	for(int i=0;i<n;i++)
		if(arr[i].second>=ans) ans=arr[i].second;
		else ans=arr[i].first;
	cout<<ans<<"\n";
	return 0;
}
