#include<bits/stdc++.h>
using namespace std;
const int M=1e4+5;
int n,day;
pair<int,int>a[M];
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i].first>>a[i].second;
	sort(a,a+n);
	for(int i=0;i<n;i++)
		if(a[i].second>=day)
			day=a[i].second;
		else
			day=a[i].first;
	cout<<day<<endl;
	return 0;
}
