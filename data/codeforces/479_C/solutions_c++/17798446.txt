#include<bits/stdc++.h>
using namespace std;
int main(){
	int i,n,max,x,y;vector< pair<int,int> > a;
	cin>>n;max=0;
	for(i=1;i<=n;i++){
		cin>>x>>y;
		a.push_back(make_pair(x,y));
	}
	sort(a.begin(),a.end());
	max = a[0].second;
	for(i=1;i<n;i++){
		if(a[i].second<max)max = a[i].first;
		else max = a[i].second;
	}
	cout<<max<<endl;
	return 0;
}
