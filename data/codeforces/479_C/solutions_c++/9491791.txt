#include <bits/stdc++.h>
using namespace std;
const int N = 5003;
pair<int,int> tab[N];

int main()
{
	int n;
	scanf("%d",&n);
	for (int i=0;i<n;i++)
		scanf("%d%d",&tab[i].first,&tab[i].second);
	sort( tab, tab+n );
	int x = 0;
	for (int i=0;i<n;i++)
	{
		if ( tab[i].second >= x ) x = tab[i].second;
		else x = tab[i].first;
	}
	printf("%d\n",x);
}
