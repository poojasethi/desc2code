#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,res=0;
    cin>>n;
    pair<int,int> a[n];
    for(int i=0;i<n;i++)cin>>a[i].first>>a[i].second;
    sort(a,a+n);
    for(int i=0;i<n;i++)
        if(res>a[i].second)res=a[i].first;
        else res=a[i].second;
    cout<<res<<endl;
}
