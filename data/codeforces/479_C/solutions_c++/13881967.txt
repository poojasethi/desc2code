#include <bits/stdc++.h>
using namespace std;

int main()
{
pair<int,int> A[5000];
int n,x = 0;
cin >> n;
for(int i = 0;i < n;i++)
{
    cin >> A[i].first >> A[i].second;
}
sort(A,A+n);
for(int i = 0;i < n;i++)
{
    if(x <= A[i].second){x = A[i].second;}
    else{x = A[i].first;}
}

cout << x;
return 0;
}
