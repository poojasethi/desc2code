#include <bits/stdc++.h>

using namespace std;

pair < int , int > p[5555];
int n;

int main()
{
cin>>n;
for(int i=0;i<n;i++)
cin>>p[i].first>>p[i].second;

sort(p,p+n);
int day=-1;

for(int i=0;i<n;i++){

  if( day <= p[i].second ){
    day=p[i].second;

  }else
  day=p[i].first;

}
cout<<day<<endl;

    return 0;
}
