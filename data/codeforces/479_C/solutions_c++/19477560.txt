#include <bits/stdc++.h>

using namespace std;

int main(){
	int n;
	scanf("%d",&n);
	vector<pair<int, int> > v(n);
	for(int i=0; i<n; i++)
		scanf("%d %d",&v[i].first, &v[i].second);
	sort(v.begin(), v.end());
	int date=0;
	for(int i=0; i<n; i++)
		date = date<=v[i].second?v[i].second:v[i].first;
	printf("%d\n",date);
	return 0;
}
