#include <bits/stdc++.h>
using namespace std;
pair <int, int> a[5005];
int main()
{
    int n,k=0;
    cin >> n;
    for (int i=1;i<=n;i++)
        cin >> a[i].first >> a[i].second;
    sort(a+1,a+n+1);
    for (int i=1;i<=n;i++)
    {
      if (a[i].second>=k) k=a[i].second; else k=a[i].first;
    }
    cout << k;
}
