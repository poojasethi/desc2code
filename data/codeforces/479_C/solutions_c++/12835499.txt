#include <iostream>
#include <algorithm>
using namespace std;

pair <int,int> de [5000];
int main()
{
    int n,l;
    cin >> n;
    l = 0;
    for (int i = 0; i < n; i++) cin >> de[i].first >> de[i].second;
    sort(de,de+n);
    for (int i = 0; i < n; i++) l = de[i].second < l ? de[i].first : de[i].second;
    cout << l << endl;
}
