#include<bits/stdc++.h>
using namespace std;
pair<int,int> a[5100], b[5100];
int main(){
    int n; cin>>n;
    for(int i=1; i<=n; i++) cin>>a[i].first>>a[i].second;
    sort(a+1,a+n+1);
    int ans=0;
    for(int i=1; i<=n; i++){
        if(a[i].second>=ans) ans=a[i].second;
        else ans=a[i].first;
    }
    cout<<ans<<endl;
}
