#include <iostream>
#include <algorithm>
using namespace std;
int n,res;
pair<int,int>a[5012];
int main(){
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i].first>>a[i].second;
	sort(a,a+n);
	res=-12;
	for(int i=0;i<n;i++){
		if(res<=a[i].second)
			res=a[i].second;
		else
			res=a[i].first;
	}
	cout<<res<<endl;
	return 0;
}
