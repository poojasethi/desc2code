#include<iostream>
#include<algorithm>
using namespace std;
pair<int,int>a[5000];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i].first;
		cin>>a[i].second;
	}
	sort(a, a + n);
	int best = -1;
	for(int i = 0; i < n; i++) {
		if (best <= a[i].second) 
			best = a[i].second;
		 else 
		 	best = a[i].first;
	}
	cout<<best;	
}
