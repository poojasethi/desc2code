#include<bits/stdc++.h>

using namespace std;

pair <int, int> p[5000];
int main()
{
	int n, ans = -1;
	cin >> n;
	for(int i = 0; i < n; i++)
	cin >> p[i].first >> p[i].second;
	sort(p, p+n);
	for(int i = 0; i < n; i++)
	{
		if(p[i].second >= ans) 
		 	ans = p[i].second; 
		else
		 	ans = p[i].first;
	}
	cout << ans;
	return 0;
}
