#include<bits/stdc++.h>
using namespace std;

int x,n;
pair < int , int > p[5050];

int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d%d",&p[i].first,&p[i].second);
	sort(p,p+n);
	x = p[0].second;
	for(int i=1;i<n;i++){
		if(p[i].second >= x) x = p[i].second;
		else x = p[i].first;
	}
	cout<<x<<endl;
}
