#include<bits/stdc++.h>

using namespace std;

pair<int,int> a[100000];
int n,k,i,ans;

int main()
{
    cin>>n;
    for(i=0;i<n;i++)
        cin>>a[i].first>>a[i].second;
    sort(a,a+n);
    for(i=0;i<n;i++)
        if(a[i].second>=ans)ans=a[i].second;
                else ans=a[i].first;
    cout<<ans;
}
