#include <bits/stdc++.h>
using namespace std;
pair<int,int> arr[10000];

int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		scanf("%d%d",&arr[i].first,&arr[i].second);
	sort(arr,arr+n);
	int end=arr[0].second;
	for(int i=1;i<n;i++)
	{
		if(arr[i].second>=end)
			end=arr[i].second;
		else
			end=arr[i].first;

	}
	cout<<end;
	return 0;
}
