#include<bits/stdc++.h>

using namespace std;

int n,i,mn;
pair<int,int> arr[5005];

int main(){

	cin >> n;

	for(i=1 ; i<=n; i++)
		cin >> arr[i].first >> arr[i].second;

	sort(arr+1,arr+n+1);

	for(i=1 ; i<=n ; i++){
		if(arr[i].second >= mn)
			mn = arr[i].second;
		else
			mn = arr[i].first;
	}

	cout << mn << endl;

}
