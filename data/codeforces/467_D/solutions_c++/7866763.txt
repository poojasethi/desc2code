//http://codeforces.com/problemset/problem/467/D
#include <iostream>
#include <cstdio>
#include <string>
#include <map>
#include <utility>
#include <vector>
#include <algorithm>
using namespace std;
string str,str1,str2;
map <string,int> mp;
pair < pair<int,int>,int > res[300005];
pair <int,int> ans[300005];
vector <int> v[300005];
int num,n,m,a[300005],t1,t2;
long long ans1,ans2;
bool vis[300005];
int tn(string &str)
{
    int i;
    for (i=0;i<str.size();i++)
    {
        if (str[i]>='A' && str[i]<='Z') str[i]-='A'-'a';
    }
    if (mp[str])
    {
        return mp[str];
    }
    num++;
    for (i=0;i<str.size();i++)
    {
        res[num].first.first+=(str[i]=='r');
    }
    res[num].first.second=str.size();
    res[num].second=num;
    mp[str]=num;
    return num;
}
void dfs(int x)
{
    vis[x]=1;
    for (int i=0;i<v[x].size();i++)
    {
        if (!vis[v[x][i]])
        {
            ans[v[x][i]]=ans[x];
            dfs(v[x][i]);
        }
    }
    return;
}
int main()
{
    scanf("%d",&n);
    int i;
    for (i=1;i<=n;i++)
    {
        cin>>str;
        a[i]=tn(str);
    }
    scanf("%d",&m);
    for (i=1;i<=m;i++)
    {
        cin>>str1>>str2;
        t1=tn(str1);
        t2=tn(str2);
        v[t2].push_back(t1);
    }
    sort(res+1,res+num+1);
    for (i=1;i<=num;i++)
    {
        if (!vis[res[i].second])
        {
            ans[res[i].second]=res[i].first;
            dfs(res[i].second);
        }
    }
    for (i=1;i<=n;i++)
    {
        ans1+=ans[a[i]].first;
        ans2+=ans[a[i]].second;
    }
    cout<<ans1<<" "<<ans2<<endl;
    return 0;
}
