#define FEN(i,n) for(int i=1;i<=(int)(n); ++i)

void chcase(string &str)
{
  int size=str.size();  FN(i,size)str[i]=tolower(str[i]);
}

int ctr(string a){int sz=a.size(),ct=0;FN(i,sz)if(a[i]=='r')ct++; return ct;}
bool cmp(string a, string b){ int ar=ctr(a), br=ctr(b); if(ar==br){return a.size() < b.size();} else return ar<br; }

map< string, vector<string> > graph;
map<string, pair<int,int> >final;

void dfs(string cur, pair<int,int> min)
{
  // cout<<cur<<" - "<<min<<endl;
  if(final.find(cur)!=final.end())return;
  final[cur]=min;
  FA(i,graph[cur]) dfs(*i,min);
}
