#include <cstdio>
#include <iostream>
#include <queue>
#include <vector>
#include <map>
#include <string>
#include <cstring>
using namespace std;

map<string,int> m;
priority_queue<pair<pair<int,int>,int> > q;
int n,qq,tot=0;
vector<int> article,v[500001];
pair<int,int> ans[500001];

int doit(string s)
{
	int i,cnt=0;
	for (i=0;i<=s.size();i++)
	{
		if ('A'<=s[i] && s[i]<='Z') s[i]+=32;
		if (s[i]=='r') cnt++;
	}
	if (m[s]) return m[s];
	m[s]=++tot;
	ans[m[s]]=make_pair(-cnt,-s.size());q.push(make_pair(make_pair(-cnt,-s.size()),m[s]));
	return m[s];
}

int main ()
{
	int i;
	string s,s1,s2;
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	{
		cin>>s;
		int ind=doit(s);
		article.push_back(ind);
	}
	scanf("%d",&qq);
	for (i=1;i<=qq;i++)
	{
		cin>>s1>>s2;
		int x=doit(s1),y=doit(s2);
		v[y].push_back(x);
	}
	while (!q.empty())
	{
		pair<int,int> x=q.top().first;int y=q.top().second;
		q.pop();
		for (i=0;i<v[y].size();i++)
		{
			if (x>ans[v[y][i]])
			{
				ans[v[y][i]]=x;
				q.push(make_pair(ans[v[y][i]],v[y][i]));
			}
		}
	}
	long long ans1=0,ans2=0;
	for (i=0;i<article.size();i++)
	{
		ans1-=ans[article[i]].first;
		ans2-=ans[article[i]].second;
	}
	printf("%I64d %I64d\n",ans1,ans2);
	return 0;
}
