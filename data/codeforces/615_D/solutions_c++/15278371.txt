#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
int cnt[200010]={0},p[200010];
ll calmod(ll a,ll b)
{
	if(b==0)
		return 1;
	if(b==1)
		return a;
	if(b%2==0)
		return (calmod(a,b/2)%mod * calmod(a,b/2)%mod)%mod;
	else 
		return (calmod(a,b-1)%mod * a)%mod;
}
int main()
{
	int m,i;
	cin>>m;
	for(i=0;i<m;i++)
	{
		cin>>p[i];
		cnt[p[i]]++;
	}
	ll dn=1;//no of divisors
	for(i=0;i<=200000;i++)
	{
		if(cnt[i]>0)
			dn=(dn * (cnt[i]+1) )%(2*(mod-1));
	}
	int f=1;
	for(i=0;i<=200000;i++)
	{
		if(cnt[i]%2==0)
			f=1;
		else
		{
			f=0;
			break;
		}
	}

	//cout<<dn<<endl;
	ll ans=1;
	for(i=0;i<m;i++)
		ans=(ans*calmod(p[i],dn/2))%mod;
	if(f==1)
	{
		for(i=0;i<=200000;i++)
			ans=(ans*calmod(i,cnt[i]/2))%mod;	//To multiply sqrt term
	}
	cout<<ans<<endl;
}
