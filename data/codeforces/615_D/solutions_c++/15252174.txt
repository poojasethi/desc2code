#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
#include<vector>
#include<cmath>
#include<map>
using namespace std;
typedef long long ll;
const int N=2e5+10;
const ll p=1e9+7;
int a[N];
int main()
{
    int n,i,j,f=0;
    ll ans=1;memset(a,0,sizeof a);
    cin>>n;
    for(i=0;i<n;i++)
    {
        ll x;
        scanf("%I64d",&x);
        ans=(ans*x)%p;
        a[x]++;
    }
    for(i=2;i<N;i++)
    if(a[i]%2==1){a[i]/=2;f=1;break;}
    if(f==0)
    {
        ans=1;
        for(i=2;i<N;i++)
            for(j=0;j<(a[i]+1)/2;j++)ans=(ans*i)%p;

    }
    for(i=2;i<N;i++)
    {
        ll x=ans;
        for(j=0;j<a[i];j++)ans=(ans*x)%p;
    }
    printf("%I64d",ans);
    return 0;
}
