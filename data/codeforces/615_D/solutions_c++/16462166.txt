#include <cstdio>
#include <cstring>
#include <string>
#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
#define MOD 1000000007LL
#define MOD2 1000000006LL
using namespace std;
typedef long long ll;

int m;
ll cnt[200001];
ll mod_pow(ll x,ll n,ll mo){
	ll res=1;
	while(n>0){
		if(n&1)res=res*x%mo;
		x=x*x%mo;
		n>>=1;
	}
	return res;
}

int main(void){
	scanf("%d",&m);
	for(int i=0;i<m;i++){
		int a;
		scanf("%d",&a);
		cnt[a]++;
	}
	ll res=1;
	ll d=1;
	for(int i=2;i<=200000;i++){
		ll fp=mod_pow(i,(ll)(cnt[i]+1)*cnt[i]/2,MOD);
		res=mod_pow(res,(cnt[i]+1),MOD)*mod_pow(fp,d,MOD)%MOD;
		d=d*(cnt[i]+1)%MOD2;
	}
	printf("%lld\n",res);
	return 0;
}
