#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <iostream>
#include <cstring>
using namespace std;
int a,b;
int n;
int s[120000];
int maxx;
int main()
{
	cin>>n;
	for (int i = 1; i<= n;i++) scanf("%d",&s[i]);
	sort(s+1,s+1+n);
	n=unique(s+1,s+1+n)-s-1;
	cin>>a>>b;
	int ans=0;
	while (a!=b)
	{
		maxx=0;
		for (int i = 1; i<= n;i++)
			{
			while (a-b<(a%s[i])&&i<=n)
			{
				s[i]=s[n--];
			}
			if (a-b>=(a%s[i]))
			maxx=max(maxx,a%s[i]);
			
			}
		if (maxx==0) ans++,a--;
		else a-=maxx,ans++;
	}
	cout<<ans<<endl;
}
