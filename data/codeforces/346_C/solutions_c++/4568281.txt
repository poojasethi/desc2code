#include <iostream>
#include <algorithm>
#define MAXN 100007

using namespace std;

int a,b,t,res,n,m,x[MAXN];

int main()
{
    //freopen("i.i","r",stdin);freopen("o.o","w",stdout);
    cin>>n;
    for (int i=1;i<=n;i++) cin>>x[i];
    cin>>a>>b;
    sort(x+1,x+n+1);
    m=1;
    for (int i=2;i<=n;i++)
    if (x[i]!=x[i-1]) x[++m]=x[i];
    for (;a!=b;)
    {
        res++;
        t=a-1;
        for (int i=1;i<=m;)
            if (a-a%x[i]<b) {x[i]=x[m];m--;}
            else {t=min(t,a-a%x[i]);i++;}
        a=t;
    }
    cout<<res;
}
