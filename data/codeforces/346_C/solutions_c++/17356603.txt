#include <algorithm>
#include <cstdio>
using namespace std;
int X[100005],A,B,D,N,m;
main()
{
    scanf("%d",&N);
    for(int i=0;i<N;i++)scanf("%d",&X[i]);
    scanf("%d%d",&A,&B);
    sort(X,X+N);
    N=unique(X,X+N)-X;
    while(A>B)
    {
        m=A-1;
        for(int i=0;i<N;i++)
            if(A-A%X[i]>=B)
            m=min(m,A-(A%X[i]));
        A=m,D++;
        while(N>0&&A-(A%X[N-1])<B)
        N--;
    }
    printf("%d",D);
}
