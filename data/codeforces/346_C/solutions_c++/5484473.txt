#include <bits/stdc++.h>
using namespace std;

int n, x[100005], a, b, id, ans;

int main(){

  scanf("%d", &n);
  for(int i = 0; i < n; i ++)
    scanf("%d", &x[i]);

  scanf("%d %d", &a, &b);

  sort(x, x + n);
  id = 1;
  for(int i = 1; i < n; i ++)
    if(x[i] != x[i - 1])
      x[id ++] = x[i];
  n = id;

  ans = 0;
  while(a != b){
    int prox = a - 1;
    while(n && (a - (a % x[n - 1])) < b)
      n --;
    for(int i = n - 1; i >= 0; i --)
      if((a - (a % x[i])) >= b)
	prox = min(prox, a - (a % x[i]));
    a = prox;
    ans ++;
  }

  printf("%d\n", ans);

  return 0;
}
