#include <algorithm>
#include <cstring>
#include <iostream>
using namespace std;
int x[100005];
int main() {
  int n;
  cin >> n;
  for (int i = 0; i < n; ++i) cin >> x[i];
  sort(x, x + n);
  n = unique(x, x + n) - x;
  int a, b;
  cin >> a >> b;
  int ans = 0, f = 1, t;
  while (f) {
    t = 1;
    f = 0;
    for (int i = 0; i < n; ++i) {
      int r = a % x[i];
      if (a - r >= b) {
         t = max(t, r);
         f = 1;
      } else {
        x[i] = x[n - 1];
        n--;
        i--;
      }
    }
    a -= t;
    ans++;
  }
  cout << ans + a - b;
}
