#include <stdio.h>
#include <algorithm>
using namespace std;
#define max(a,b) (a>b?a:b)
#define N 1000009
#define M 32009
int n,x[N],a,b,ans;
int step[N];
main() {
	//freopen("C.in","r",stdin);
	scanf("%d",&n);
	for (int i=0;i<n;i++) scanf("%d",x+i);
	scanf("%d%d",&a,&b);
	sort(x,x+n);n=unique(x,x+n)-x;
	for (int B=b;B<=a;B++) step[B-b]=1;
	for (int i=0;i<n;i++) for (int B=(b+x[i]-1)/x[i]*x[i];B<=a;B+=x[i]) step[B-b]=x[i]-1;
	for (int B=b+1;B<=a;B++) step[B-b]=max(step[B-b-1]-1,step[B-b]);
	for (int B=b;B<a;B+=step[B-b],ans++);
	printf("%d\n",ans);
}
