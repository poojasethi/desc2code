#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int i,j,k,l,r,a[150001],b[150001],n,ans;
int main(){
	scanf("%d",&n);
	for (i=1;i<=n;i++) scanf("%d",&a[i]);
	scanf("%d%d",&r,&l);
	sort(a+1,a+1+n);
	for (k=n,n=0,i=1;i<=k;i++)
	 if (i==1||a[i]!=a[i-1]) b[++n]=a[i];
	for (ans=0;l<r;ans++){
		for (i=j=1;i<=n;i++){
			k=r%b[i];
			if (r-k<l){
				swap(b[n],b[i]);n--;i--;continue;
			}else j=max(j,k);
		}
		r-=j;
	}
	printf("%d\n",ans);
	return 0;
}
