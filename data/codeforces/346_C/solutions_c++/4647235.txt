#include<iostream>
#include<algorithm>
#define For(i,x,y) for(int i=x;i<=y;i++) 
using namespace std;
const int Maxn=100001;
int x[Maxn],a,b,n,ans;
int main()
{
    cin>>n;
    For(i,1,n)cin>>x[i];
    cin>>b>>a;
    sort(x+1,x+n+1);
    n=unique(x+1,x+n+1)-x-1;
    while(a<b){
        int Mx=1;
        For(i,1,n) if (b-b%x[i]>=a) Mx=max(Mx,b%x[i]);
        b-=Mx;
        while(n&&b-b%x[n]<a) n--;
        ans++;
    }
    cout<<ans<<endl;
    return 0;
}
