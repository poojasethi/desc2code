#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;
int x[100005];
int main()
{
	int n, A, B;
	scanf("%d", &n);
	for (int i = 0; i < n; ++ i) scanf("%d", &x[i]);
	sort(x, x + n);
	n = unique(x, x + n) - x;
	scanf("%d%d", &A, &B);
	int ans = 0;
	while (A > B)
	{
		int res = 1;
		for (int i = 0; i < n; ++ i)
		{
			if (A % x[i] <= A - B)
			{
				res = max(res, A % x[i]);
			}
			else
			{
				swap(x[i], x[n - 1]);
				-- n;
				-- i;
			}
		}
		A -= res;
		ans ++;
	}
	printf("%d\n", ans);
}
