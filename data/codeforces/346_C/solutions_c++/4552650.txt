#include<iostream>
#include<algorithm>
#define F(i,y) for(int i=1;i<=y;i++) 
using namespace std;
int x[100001],a,b,n,ans;
int main(){
	cin>>n;
	F(i,n)cin>>x[i];
	cin>>b>>a;
	sort(x+1,x+n+1);
	n=unique(x+1,x+n+1)-x-1;
	while(a<b){
		int Mx=1;
		F(i,n)if (b-b%x[i]>=a)Mx=max(Mx,b%x[i]);
		b-=Mx;
		while(n&&b-b%x[n]<a)n--;
		ans++;
	}
	cout<<ans<<endl;
 	return 0;
}
