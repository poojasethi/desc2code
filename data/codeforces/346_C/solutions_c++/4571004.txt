#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
int N,M,A,B,ans;
int a[100010];
int main(){
    scanf("%d",&N);
    for (int i=1;i<=N;i++)scanf("%d",&a[i]);
    scanf("%d%d",&A,&B);
    sort(a+1,a+N+1);
    M=1;
    for (int i=2;i<=N;i++)if (a[i]!=a[M])a[++M]=a[i];
    N=M;
    for (int x=A;x>B;){
        int r=1;
        for (int i=1;i<=N;i++)
        if (x-x%a[i]>=B)r=max(r,x%a[i]);
        else{
            a[i]=a[N];
            N--;i--;
        }
        x-=r;
        ans++;
    }
    printf("%d",ans);
}
