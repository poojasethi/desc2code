#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int n,s,t,ans,down;
int a[100010];
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++) scanf("%d",a+i);
	scanf("%d%d",&s,&t);
	sort(a,a+n);
	n=unique(a,a+n)-a;
	ans=0;
	while(s>t){
		down=1;
		for(int i=n-1;i>=0;i--){
			if(down>=a[i])break;
			int r=s%a[i];
			if(r>down && r<=s-t) down=r;
		}
		s-=down;
		ans++;// cout<<"s "<<s<<endl;
		while(n&& s-s%a[n-1]<t)n--;
	}
	ans+=(s-t);
	printf("%d\n",ans);
}
