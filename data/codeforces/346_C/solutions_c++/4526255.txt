#include <cstdio>
#include <algorithm>
using namespace std;

const int N = 1e5 +5;

int n;
int p, q;
int a[N];

int main() {
  scanf("%d", &n);
  for(int i = 0; i < n; i++) {
    scanf("%d", a+i);
  }  
  sort(a, a+n);
  n = unique(a, a+n) - a;
  
  scanf("%d %d", &q, &p);

  int ans = 0;
  for(int x = q; x > p; ) {
    int mx = 1;
    for(int i = 0; i < n; i++) {
      int r = x%a[i];
      if(x-r >= p) {
	mx = max(mx, r);
      } else {
	a[i] = a[n-1];
        n--, i--;
      }
    }
    x -= mx;
    ans++;
  }
  
  printf("%d\n", ans);
}
