#include <cstdio>
#include <algorithm>
#define scanf(arg...) (scanf(arg)?:0)
using namespace std;

const int N = 1e5;

int n, x[N], a, b;


int main()
{
	scanf("%d", &n);
	for (int i=0; i<n; ++i) scanf("%d", &x[i]);
	scanf("%d%d", &a, &b);
	
	sort(x, x+n);
	n = unique(x, x+n) - x;
	
	int res = 0;
	while (a > b) {
		int mov = 1;
		for (int i=0; i<n; ++i) {
			int now = a % x[i];
			if (a-now < b) x[i--] = x[--n];
			else mov = max(mov, now);
		}
		a -= mov;
		++res;
	}
	
	printf("%d\n", res);
	return 0;
}
