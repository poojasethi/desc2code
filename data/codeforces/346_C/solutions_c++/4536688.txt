#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int c[100001];
int n;
int z;
int ans;
int s[100001];
int a, b;
int main()
{
	scanf("%d", &n);
	for(int i = 1; i <= n; i++)
		scanf("%d", &c[i]);
	scanf("%d%d", &b, &a);
	sort(c+1, c+n+1);
	for(int i = 1; i <= n; i++)
		if(c[i] != c[i-1])
			s[++z] = c[i];
	int t;
	while(b > a)
	{
		ans ++;
		while(z > 0 && b - b%s[z] < a)
			z--;
		t = 1;
		for(int i = 1; i <= z; i++)
			if(t < b%s[i] && b - b%s[i] >= a)
				t = b%s[i];
			
		b -= t;
	}
	printf("%d\n", ans);
	return 0;
}
