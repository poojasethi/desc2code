#include<stdio.h>
#include<algorithm>
using namespace std;
#define M 100010
#define Min(a,b) ((a)<(b))?(a):(b)
int x[M],A,B,n,ans;

void process(){
	int i,j,k,rr,aa,a;
	sort(x,x+n);
	rr=unique(x,x+n)-x;
	while(A!=B){
		k=A-1;
		for(i=0;i<rr;i++){
			aa=A-A%x[i];
			if(aa>=B)k=Min(k,aa);
		}

		ans++;
		A=k;
		while(rr>=1&&A-A%x[rr-1]<B)rr--;
	}
	return;
}
int main(){
	int i;
	//freopen("in.txt","r",stdin);
	scanf("%d",&n);
	for(i=0;i<n;i++)scanf("%d",x+i);
	scanf("%d %d",&A,&B);
	process();
	printf("%d\n",ans);
	return 0;
}
