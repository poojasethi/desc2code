#include <cstdio>
#include <algorithm>
using namespace std;
#define N 100000

int vet[N], n, a, b, r;

int main()
{
	scanf("%d", &n);
	for (int i = 0; i < n; ++i)
		scanf("%d", vet+i);
	scanf("%d%d", &a, &b);
	sort(vet, vet+n);
	n = unique(vet, vet+n)-vet;
	while (n && a != b)
	{
		int x = 1;
		for (int i = 0; i < n; ++i)
		{
			int r = a%vet[i];
			if (a-r >= b) x = max(x, r);
			else
			{
				vet[i] = vet[n-1];
				--n; --i;
			}
		}
		a -= x;
		++r;
	}
	printf("%d\n", r+a-b);
}
