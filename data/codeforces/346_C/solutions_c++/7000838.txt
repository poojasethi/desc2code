#include <stdio.h>
#include <algorithm>
using namespace std;
#define N 1000009
#define max(a,b) (a>b?a:b)
int n,x[N],a,b,ans;
int cnt[N],step[N];
main() {
	//freopen("1.in","r",stdin);
	scanf("%d",&n);
	for (int i=0;i<n;i++) scanf("%d",x+i);
	scanf("%d%d",&a,&b);
	sort(x,x+n);
	n=unique(x,x+n)-x;
	for (int i=0;i<=a-b;i++) step[i]=1;
	for (int i=0;i<n;i++) for (int X=(b+x[i]-1)/x[i]*x[i];X<=a;X+=x[i]) step[X-b]=x[i]-1;
	for (int i=1;i<=a-b;i++) step[i]=max(step[i],step[i-1]-1);
	for (int i=0;i<a-b;i+=step[i]) ans++;
	printf("%d\n",ans);
}
