#include<cstdio>
#include<cstring>
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
const int maxn = 1000100;
int x[maxn];
int main(){
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&x[i]);
	}
	sort(x,x+n);
	n=unique(x,x+n)-x;
	int a,b;
	scanf("%d%d",&a,&b);
	int ans=0;
	while(b<a){
		int mx=1;
		for(int i=0;i<n;i++){
			if(a-a%x[i]>=b){
				mx=max(mx,a%x[i]);
			}
		}
		while(n&&x[n-1]>a-b)n--;
		a-=mx;
		ans++;
	}
	printf("%d\n",ans);
}
