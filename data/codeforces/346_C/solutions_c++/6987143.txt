#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxn=100010;
int n;
int x[maxn],a,b;
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&x[i]);
	scanf("%d%d",&b,&a);
	sort(x+1,x+n+1);
	n=unique(x+1,x+n+1)-x-1;
	int ans=0,MAX;
	int r=n;
	while(b>a){
		while(r>1&&b-b%x[r]<a) r--;
		MAX=1;
		for(int i=r;i>=1;i--){
			if(b-a>=b%x[i]) MAX=max(MAX,b%x[i]);
			if(MAX>=x[i]-1) break;
		}
		b-=MAX;ans++;
	}
	printf("%d\n",ans);
	return 0;
}
