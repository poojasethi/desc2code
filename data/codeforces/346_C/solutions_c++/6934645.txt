#include <cstdio>
#include <algorithm>
using namespace std;
const int N = 100005;
int s[N];
int main() {
	int n, a, b;
	scanf("%d", &n);
	for(int i = 0; i < n; i ++) {
		scanf("%d", &s[i]);
	}
	scanf("%d%d", &a, &b);

	sort(s, s+n);
	n = unique(s, s+n) - s;
	int ans = 0;
	while(a != b) {
		int dt = 1;
		for(int i = 0; i < n; i ++) {
//			if(s[i] == -1) continue;
			int t = a % s[i];
			if(a - t >= b) {
				dt = max(dt, t);
			} else {
//				s[i] = -1;
				s[i] = s[n-1];
				n--;
				i--;
			}
		}
		a -= dt;
		ans ++;
	}
	printf("%d\n", ans);
	return 0;
}
