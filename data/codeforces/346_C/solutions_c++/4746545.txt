#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;
const int MaxN=100005;
int a[MaxN];
int main() {
	//freopen("C.in","r",stdin);
	int n,A,B;
	scanf("%d",&n);
	for(int i = 0; i < n; i++)
		scanf("%d",a+i);
	sort(a,a+n),n = unique(a,a+n)-a;
	scanf("%d%d",&A,&B);
	int ans = 0;
	while (A > B) {
		int t = 1;
		for(int i = 0; i < n; i++)
			if(A - A % a[i] >= B)
				t = max(t, A % a[i]);
			else
				a[i--] = a[--n];
		A -= t;
		ans++;
	}
	printf("%d\n",ans);
	return 0;
}
