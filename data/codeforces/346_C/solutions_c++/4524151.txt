#include <stdio.h>
#include <algorithm>
#include <iostream>

using namespace std;

int main (void) {
	int n, i, cnt, m, waw;
	int v[100000];
	int a, b;
	scanf ("%d", &n);
	for (i = 0; i < n; i++)	scanf ("%d", &v[i]);
	scanf ("%d%d", &a, &b);
	sort (v, v+n, greater<int>());
	n = unique(v, v+n)-v;
	//printf ("%d\n", n);
	cnt = 0;
	waw = 0;
	while (a > b && waw < n) {
		m = 0;
		for (i = waw; i < n && v[i] > m; i++) {
			if (a-a%v[i] < b)	waw = i+1;
			else	m = max (m, a%v[i]);
		}
		if (m == 0)	a--;
		else	a -= m;
		cnt++;
	}
	cnt += a-b;
	printf ("%d\n", cnt);
	return 0;
}
