n, m = map(int, raw_input().split())
price = map(int, raw_input().split())
aucnum = map(int, raw_input().split())
auc = []
res = 0
for idx, x in enumerate(price):
    if idx + 1 in aucnum:
        auc.append(x)
    else:
        res += x
auc.sort(reverse=True)
for x in auc:
    if res > x:
        res += res
    else:
        res += x
print res
