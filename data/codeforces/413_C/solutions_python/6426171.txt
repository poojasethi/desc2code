N,M = map(int,raw_input().split())
A = map(int,raw_input().split())
B = map(int,raw_input().split())
#print N,M,A,B
AuctionQ = [A[i-1] for i in B]
#print AuctionQ
for _ in AuctionQ :
	A.remove(_)
#print N,M,A,B,AuctionQ
score = sum(A)
rem = len(AuctionQ)
if rem > 0 :
	score = max(score,max(AuctionQ))+score
	#print score
	rem-=1
if rem > 0 :
	score = score<<(len(AuctionQ)-1)
	#print score
print score
