#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

typedef long long int ll;

int main()
{
  int n,m;
  cin>>n>>m;
  vector<pair<int,int> > v(n,pair<int,int> (1,0));
  for (int i=0;i<n;i++)
    cin>>v[i].second;
  for (int i=0;i<m;i++) {
    int ind;
    cin>>ind;
    ind--;
    v[ind].first=0;
  }
  sort(v.begin(),v.end());
  ll sol=0;
  for (int i=n-1;i>=0;i--) {
    if (v[i].first==1)
      sol+=v[i].second;
    else
      sol+=max(sol,ll(v[i].second));
  }
  cout<<sol<<endl;
}
