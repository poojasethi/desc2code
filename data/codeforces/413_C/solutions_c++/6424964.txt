#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,m,i,a[109],b[109];
    long long int t=0;
    cin>>n;
    cin>>m;
    for(i=1;i<=n;i++)
        {
            cin>>a[i];
            t+=a[i];
        }
    for(i=0;i<m;i++)
    {
        cin>>b[i];
        t-=a[b[i]];
        b[i]=a[b[i]];
    }
    sort(b,b+m);
    if(b[m-1]>t)
    {
        t+=b[m-1];
        m--;
    }
    for(i=1;i<=m;i++)
        t=t*2;
    cout<<t<<endl;
    return 0;
}
