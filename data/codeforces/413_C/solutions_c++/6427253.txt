#include <iostream>
#include <algorithm>
using namespace std;

int main() 
{
    int n,m,i,j,a[100],b,l=0,d[30];
    bool c[100];
    long long kq=0;
    cin>>n>>m;
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
        kq+=a[i];
    }
    for(i=1;i<=m;i++)
    {
        cin>>b;
        kq-=a[b];
        l++;
        d[l]=a[b];
    }
    sort(d,d+l+1);
    for(i=l;i>0;i--)
    if(kq>d[i]) kq*=2;
    else kq+=d[i];
    cout<<kq;
}
