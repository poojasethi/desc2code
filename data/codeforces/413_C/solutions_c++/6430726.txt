#include <iostream>
using namespace std;

int main() {
	
	int n, m;
	cin >> n >> m;
	
	long long a[n];
	
	for (int i=0; i<n; i++) 
		cin >> a[i];
	
	int x;
	long long max=0;
	
	for (int i=0; i<m; i++){
		
		cin >> x;
		if (a[x-1]>max) max=a[x-1];
		a[x-1]=0;
		
	}
		
	long long sum=0;
	for (int i=0; i<n; i++)
		if (a[i]>0) sum+=a[i];
	
	int g=0;
	if (sum > max) {max=sum;}
	else { max+=sum; g++; }
	
	for (int i=g; i<m; i++)
		max*=2;
	
	cout << max;
	
	return 0;
}
