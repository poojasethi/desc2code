#include <iostream>
#include <algorithm>
using namespace std;
int a[100], b[30];
int main()
{
	int n, m, i, x;
	long long s;
	
	cin>>n>>m;
	s = 0;
	for (i=0; i<n; i++) {
		cin>>a[i];
		s += a[i];
	}
	for (i=0; i<m; i++) {
		cin>>x;
		x--;
		b[i]=a[x];
		s -= b[i];
	}
	sort(b, b+m);
	for (i=m-1; i>=0; i--) {
		if (b[i]>s) s+=b[i];
		else s+=s;
	}
	cout<<s<<endl;
	return 0;
}
