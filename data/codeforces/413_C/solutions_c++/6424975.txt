#include <iostream>
#include <algorithm>
using namespace std;
long long i,j,z,n,m,a[102],b[102],c,mark[102],sum;
int main()
{
    cin>>n>>m;
    for(i=1;i<=n;i++)
        cin>>a[i];
    for(j=1;j<=m;j++)
    {
        cin>>c;
        mark[c]=1;
        b[++z]=a[c];
    }
    for(i=1;i<=n;i++)
        if(mark[i]==0)
            sum+=a[i];
    sort(b+1,b+z+1);
    for(i=z;i>=1;i--)
    {
        if(sum>b[i])
            sum*=2;
        else
            sum+=b[i];
    }
    cout<<sum<<endl;
    return 0;
}
