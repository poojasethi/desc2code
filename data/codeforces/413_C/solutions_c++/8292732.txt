#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int main() {
	long long int n, m, a[110], b[110], it = 0, sum = 0, y;
	cin >> n >> m;
	for (int i = 0; i < n; i++){
		cin >> a[i];
		sum += a[i];
	}
	for (int i = 0; i < m; i++){
		cin >> y;
		y--;
		sum -= a[y];
		b[it++] = a[y];
	}
	sort(b, b + it);
	for (int i = it - 1; i >= 0; i--){
		if (sum < b[i])
			sum += b[i];
		else 
			sum *= 2;
	}
	cout << sum << endl;
}
