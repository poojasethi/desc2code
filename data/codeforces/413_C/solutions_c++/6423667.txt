#include<cstdio>
#include<iostream>
#include<algorithm>
#define inf 0x7fffffff
int n,m,tot;
long long ans;
int a[101],c[101];
bool b[101];
using namespace std;
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1;i<=m;i++){int x;scanf("%d",&x);b[x]=1;}
	for(int i=1;i<=n;i++)
	   if(!b[i])ans+=a[i];
	   else c[++tot]=a[i];
	sort(c+1,c+tot+1);
	for(int i=tot;i>0;i--)
	   if(ans<c[i])ans+=c[i];
	   else ans<<=1;
	printf("%I64d",ans);
	return 0;
}
