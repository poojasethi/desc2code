# include <iostream>
# include <algorithm>
# define fi cin
# define fo cout
# define nmax 105
using namespace std;
long long x[nmax],y[nmax];
int main(void)
{
    long long n,m;
    fi>>n>>m;
    for (int i=1;i<=n;++i) fi>>x[i];
    long long p;
    for (int i=1;i<=m;++i) fi>>p,y[i]=x[p],x[p]=0;
    long long S=0;
    for (int i=1;i<=n;++i) S+=x[i];
    sort(y+1,y+1+m);
    for (int i=m;i;--i) S+=(S>y[i] ? S:y[i]);
    fo<<S<<'\n';
    return 0;
}
