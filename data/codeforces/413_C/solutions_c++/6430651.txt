#include<iostream>
#include<algorithm>
using namespace std;

const int oo = 110;
long long n, m, a[oo], ans, b[oo];

int main(){
  cin >> n >> m;
  for(int i = 1; i <= n; i++) cin >> a[i], ans += a[i];
  for(int i = 1; i <= m; i++) cin >> b[i], b[i] = a[b[i]], ans -= b[i];
  sort(b + 1, b + m + 1);
  for(int i = m; i; i--)
    if(ans > b[i])
      break;
    else
      ans += b[i], m--;
  cout << ans * (1 << m);
  return 0;
}
