#include<cstdio>
#include<algorithm>

int a[105],b[105];
int sta[105],top;
bool cmp(int x,int y){return x>y;}

#define max(x,y) (x>y?x:y) 

int main(){
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++) scanf("%d",&a[i]);
    for(int i=1;i<=m;i++){int x;scanf("%d",&x);b[x]=1;}
    long long ans=0;
    for(int i=1;i<=n;i++)
      if(!b[i]) ans+=a[i];
      else sta[++top]=a[i];
    std::sort(sta+1,sta+top+1,cmp);
    for(int i=1;i<=top;i++)
      ans+=max(ans,sta[i]);
    printf("%I64d\n",ans);
}
