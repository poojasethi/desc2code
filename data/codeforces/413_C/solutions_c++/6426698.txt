
#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;
ll a[110],b[40];
bool cmp(int a,int b)
{
	return a>b;
}
int main()
{
	int n,m;
	ll sum=0;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		sum+=a[i];
	}
	for(int i=1;i<=m;i++)
	{
		int id;
		cin>>id;
		b[i]=a[id];
		sum-=a[id];
	}
	sort(b+1,b+m+1,cmp);
	for(int i=1;i<=m;i++)
		if(b[i]>=sum)
			sum+=b[i];
		else
			sum+=sum;
	cout<<sum<<endl;
	return 0;
}
