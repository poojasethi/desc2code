#include <iostream>
#include <algorithm>

using namespace std;

long long ans;

long long n,m,a,v[109];

bool vis[109];

int main()
{
	cin >> n >> m;
	
	for (int i = 0; i < n; i++)
		cin >> v[i];
		
	for (int i = 0; i < m; i++)
	{
		cin >> a;
		
		vis[a-1] = 1;
	}
	
	for (int i = 0; i < n; i++)
		if (vis[i] == 0) ans += v[i],v[i] = 0;
	
	sort(v,v+n);
	
	for (int i = n-1; i >= 0; i--)
		if (v[i]!=0) ans += max(v[i],ans);
	
	cout << ans;
}
