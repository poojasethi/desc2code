#include <iostream>
#include <algorithm>

using namespace std;

int main ()
{
  int n, m, a[131], b[31], t;
  long long int och = 0;
  cin >> n >> m;
  for (int i=0; i < n; ++i)
    cin >> a[i];
  for (int i=0; i < m; ++i)
    {
      cin >> t;
      b[i] = a[t-1];
      a[t-1] = 0;
    }
  for (int i=0; i < n; ++i)
    och += a[i];
  sort(b, b + m);
  for (int i=m-1; i >= 0; --i)
    if (b[i] > och)
      och += b[i];
    else
      och *= 2;
  cout << och << '\n';
  return 0;
}
