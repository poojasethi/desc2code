#include<iostream>
#include<vector>
#include<stdio.h>
#include<algorithm>
using namespace std;
long long n,m,i,j,a[1002],h;
bool mark[1002];
vector <long long> g;
long long answ;
int main()
{
	cin>>n>>m;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
		answ+=a[i];
	}
	for(i=1;i<=m;i++)
	{
		cin>>h;
		g.push_back(a[h]);
		answ-=a[h];
	}
	sort(g.begin(),g.end());
	for(i=g.size()-1;i>=0;i--)
	{
		if(g[i]<answ)
			answ*=2;
		else
			answ+=g[i];
	}
	cout<<answ<<endl;
	return 0;
}
