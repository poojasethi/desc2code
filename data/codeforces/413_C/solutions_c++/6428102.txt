#include <bits/stdc++.h>
using namespace std;
int a[105],b[105],i,n,m;
long long ans=0;

int main() {
    cin>>n>>m;
    for(i=0;i<n;i++) {
        scanf("%d",&a[i]);
        ans+=a[i];
    }
    for(i=0;i<m;i++){
        scanf("%d",&b[i]);
        b[i]=a[b[i]-1];
        ans-=b[i];
    }
    sort(b,b+m);
    for(i=m-1;i>=0;i--)
        ans+=max(ans,(long long)b[i]);
    cout<<ans;
    return 0;
}
