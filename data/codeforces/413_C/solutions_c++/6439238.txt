#include<bits/stdc++.h>

using namespace std;
long long int sayac,n,m,k,a[105],j,t,x,i,arr[105][2];
int main(){
	scanf("%lld %lld",&n,&m);
	for(i=1;i<=n;i++)
		scanf("%lld",&arr[i][0]);
	for(i=1;i<=m;i++){
		scanf("%lld",&x);
		arr[x][1]=1;

	}
	for(i=1;i<=n;i++){
		if(!arr[i][1])
			t+=arr[i][0];
		else 
			a[++sayac]=arr[i][0];
	}
	sort(a+1,a+sayac+1);
	for(i=sayac;i>=1;i--){
		if(a[i]<t)
			t*=2;
		else
			t+=a[i];
	}
	printf("%lld",t);
	
	return 0;
}
