#include <iostream>
#include <algorithm>
using namespace std;
long long a[150],b[150],c[150],n,i,m,x,ans,l;
int main()
{
	cin>>n>>m;
	for (i=0; i<n; i++)
	    cin>>a[i];
	for (i=0; i<m; i++)
	{
		cin>>x;
		b[x-1]=1;
	}
	for (i=0; i<n; i++)
	    if (b[i]==0) ans+=a[i];
	for (i=0; i<n; i++)
		if (b[i]==1)
		{
			c[l]=a[i];
			l++;
		}
	sort(c,c+l);
	for (i=l-1; i>=0; i--)
	    if (ans>c[i]) ans*=2;
	    else ans+=c[i];
	cout<<ans<<endl;
	return 0;
}
