#include<iostream>
#include<algorithm>
using namespace std;
long long n,m,i,j,s,a[123],c[123];
bool b[123];
int main(){
cin>>n>>m;
for(i=0;i<n;i++)cin>>a[i];
for(i=0;i<m;i++){cin>>j;b[j-1]=1;}
for(j=i=0;i<n;i++){
    if(b[i]) c[j++]=a[i];
    else s+=a[i];
}
sort(c,c+j);
while(--j>=0)s+=c[j]<s?s:c[j];
cout<<s;
return 0;
}
