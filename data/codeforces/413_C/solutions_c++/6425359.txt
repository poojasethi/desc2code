#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int a[10000001];
int main(){
	int n, m;

	cin >> n >> m;
	long long na = 0;
	vector<int> v;
	for(int i = 1; i <= n; i++){
		cin >> a[i];
		na += a[i];
	}
	for(int i = 1; i <= m; i++){
		int x;
		cin >> x;
		v.push_back(a[x]);
		na -= a[x];
	}
	sort(v.begin(), v.end());
	long long p = na;
	
	for(vector<int>::iterator it = v.end()-1; it >= v.begin(); it--){
		if(p > *it)
			p += p;
		else p+= *it;
	}
	cout << p << endl;
	return 0;
}
