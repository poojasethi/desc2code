#include<cstdio>
#include<algorithm>
#define N 1005
using namespace std;
int a[N],b[N],x,i,n,m;
long long now;
bool f[N];
bool cmp(int a,int b){return a>b;}
int main()
{
  scanf("%d%d",&n,&m);
  for (i=1;i<=n;i++) scanf("%d",&a[i]);
  for (i=1;i<=m;i++) 
  {
    scanf("%d",&x);
    f[x]=true;b[i]=a[x];
  }
  for (i=1;i<=n;i++) if (!f[i]) now+=a[i];
  sort(b+1,b+m+1,cmp);
  for (i=1;i<=m;i++)
    if (b[i]<now) now*=2;else now=now+(long long)(b[i]);
  printf("%I64d",now);
  return 0;
}
