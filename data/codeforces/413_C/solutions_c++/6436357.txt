#include <iostream>
#include <algorithm>

using namespace std;

const int MAXN = 200;
int a[MAXN], b[MAXN], sum;

int main() {
	int n, m, x;
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		cin >> a[i], sum += a[i];
	for (int i = 0; i < m; i++)
		cin >> b[i], b[i] = a[b[i] - 1], sum -= b[i];
	sort(b, b + m);
	reverse(b, b + m);
	if (sum < b[0]) sum += b[0], m--;
	cout << (long long)sum * (1 << m) << endl;
	return 0;
}
