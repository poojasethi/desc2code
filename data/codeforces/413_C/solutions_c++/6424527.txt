#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
typedef long double ld;

const int MAXN = 107;

ll ans = 0;
ll nor[MAXN], auc[MAXN];
int n, m;

int main(){
	cin >> n >> m;
	for (int i = 0; i < n; ++i)
		cin >> nor[i];
	for (int j = 0; j < m; ++j){
		int ind;
		cin >> ind;
		auc[j] = nor[--ind];
		nor[ind] = 0;
	}
	sort(auc, auc + m);
	for (int i = 0; i < n; ++i){
		ans += nor[i];
	}
	for (int j = m - 1; j >= 0; --j){
		if (ans > auc[j])
			ans *= 2;
		else 
			ans += auc[j];
	}
	cout << ans << endl;
	return 0;
}
