#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int n,m,k;
long long a[110],b[110];
long long ans;
long long aus[110];
int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	for(int i=1;i<=m;i++)
		cin>>b[i];
	for(int i=1;i<=m;i++) aus[i]=a[b[i]],a[b[i]]=-1;
	for(int i=1;i<=n;i++) if(a[i]>0) ans+=a[i];
	sort(aus+1,aus+m+1);
	for(int i=m;i>0;i--)
	{
		if(ans>aus[i]) ans=ans*2ll;
		else ans=ans+aus[i];
	}
	cout<<ans<<endl;
	return 0;
}
