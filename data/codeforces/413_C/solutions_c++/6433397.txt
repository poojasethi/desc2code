# include<iostream>
# include<algorithm>

using namespace std;

int a[30000],b[200000],c[30000],p;

int main()
{
	
	unsigned long long int n,m,x=0;
	cin>>n>>m;
	
	for(int i=0;i<n;i++)
	    cin>>a[i];
	    
	for(int i=0;i<m;i++)
	{
		int q;
		cin>>q;
		b[p]=a[q-1]; 
		a[q-1]=0;
		p++;
	}
	
	for(int i=0;i<n;i++)
		x+=a[i];
	
	sort(b,b+m);
	
	for(int i=p-1;i>=0;i--)
	{
			if(x>=b[i])
			   x*=2;
			else
		       x+=b[i];	
	}
	
	cout<<x<<endl;
	return 0;
}
