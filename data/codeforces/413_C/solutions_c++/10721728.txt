#include<iostream>
using namespace std;
long long poi[100];
bool auc[100];
int main() {
	int n,m,x;
	long long sum=0ll,max=0ll;
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>poi[i];
	for(int i=0;i<m;i++) {
		cin>>x;
		auc[x-1]=true;
	}
	for(int i=0;i<n;i++) {
		if(auc[i]) {
			if(poi[i]>max)
				max=poi[i];
		}
		else {
			sum+=poi[i];
		}
	}
	if(sum<max) {
		sum+=max;
		m--;
	}
	for(int i=0;i<m;i++)
		sum<<=1;
	cout<<sum;
	return 0;
}
