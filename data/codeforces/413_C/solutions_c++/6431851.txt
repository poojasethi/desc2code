#include<iostream>
#include<algorithm>
#include<cstdio>
#include<set>
using namespace std;
long long arr[1000],x,sum=0,n,m,mxy=0;
int main(){
    cin>>n>>m;
    for(long long j=1;j<=n;j++){
        cin>>arr[j];
        sum+=arr[j];
        mxy=max(mxy,arr[j]);
    }
    for(long long j=1;j<=m;j++){
        cin>>x;
        sum-=arr[x];
    }
    if(sum<mxy){
        sum+=mxy;
        m--;
    }
    for(int j=0;j<m;j++)
        sum*=2;
    cout<<sum<<endl;

}
