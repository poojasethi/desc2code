#include<iostream>
#include<algorithm>
using namespace std;
#define int long long
int n,m;
pair<int,int> a[200];
main()
{
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>a[i].second;
	for(int i=0;i<m;i++)
		{
			int u;
			cin>>u;
			a[u-1].first=1;
			a[u-1].second*=-1;
		}
	sort(a,a+n);
	int sum=0;
	for(int i=0;i<n;i++)
	{
		if(a[i].first==0)
		sum+=a[i].second;
		else
		{
			if(-1*a[i].second>sum)
			sum-=a[i].second;
			else
			sum*=2;
		}
	}
	cout<<sum<<endl;
}
