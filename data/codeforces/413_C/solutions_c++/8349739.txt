#include<cstdio>
#include<algorithm>
using namespace std;
struct node{
	bool bo;
	long long v;
}a[200];
int m,n,k;
long long sum;
bool cmp(node a,node b){
	return a.bo==b.bo?a.v>b.v:a.bo<b.bo;
}

int main(){
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++)
		scanf("%I64d",&a[i].v);
	for (int i=1;i<=m;i++){
		scanf("%d",&k);
		a[k].bo=1;
	}
	sort(a+1,a+1+n,cmp);
	for (int i=1;i<=n;i++){
		if (!a[i].bo) sum+=a[i].v;
		else sum+=max(sum,a[i].v);
	}
	printf("%I64d\n",sum);
	return 0;
}
