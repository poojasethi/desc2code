#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int n,m;
long long ans;
int a[101],b[101];
inline bool cmp(int a,int b){return a>b;}
int main(){
    scanf("%d%d",&n,&m);
    for (int i=1;i<=n;++i) scanf("%d",&a[i]),ans+=a[i];
    for (int i=1;i<=m;++i) scanf("%d",&b[i]),b[i]=a[b[i]],ans-=b[i];
    sort(b+1,b+m+1,cmp);
    for (int i=1;i<=m;++i)
        ans+=b[i]<ans?ans:b[i];
    printf("%I64d\n",ans);
    return 0;
}
