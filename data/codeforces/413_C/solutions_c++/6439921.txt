#include<bits/stdc++.h>

using namespace std;

long long int m,n,k,i,j,h,A[105][2],B[105];

int main(){
	scanf("%lld %lld",&n,&m);
	for(i=0;i<n;i++)scanf("%lld",&A[i][0]);
	for(i=0;i<m;i++){scanf("%lld",&k);A[k-1][1]=1;}
	m=0;
	for(i=0;i<n;i++){if(!A[i][1])m+=A[i][0]; else B[h++]=A[i][0];}
	sort(B,B+h);
//	for(i=0;i<h;i++)printf("%lld ",B[i]);
	for(i=h-1;i>=0;i--){if(m>B[i])m*=2; else m+=B[i];}
	printf("%lld",m);
	return 0;
}
