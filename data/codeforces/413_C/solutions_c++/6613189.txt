#include <bits/stdc++.h>
using namespace std;
#define LL long long
LL A[110];
int F[110];
int main()
{
	int N,M,v;
	cin>>N>>M;
	std::vector<LL> V;
	for(int i = 1;i<=N;i++)
	{
		cin>>A[i];
	}
	for(int i = 1;i<=M;i++){
		cin>>v;
		F[v] = 1;
	}
	LL Sum = 0;
	for(int i = 1;i<=N;i++)
	{
		if(F[i])
			V.push_back(A[i]);
		else
			Sum+=A[i];
	}
	sort(V.begin(),V.end(),greater<LL>());
	for(int i = 0;i<V.size();i++)
	{
		if(Sum>V[i])
			V[i] = Sum;
		Sum+=V[i];
	}
	cout<<Sum<<endl;

}
