#include<algorithm>
#include<iostream>
#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<cmath>
using namespace std;
typedef long long int64;
const int maxn=110,maxm=40;
int v[maxn],w[maxm];
int n,m; int64 ans;

int main(){
    cin>>n>>m; ans=0;
    for(int i=1;i<=n;++i) cin>>v[i],ans+=v[i];
    for(int i=1,now=0,x;i<=m;++i) cin>>x,w[++now]=v[x],ans-=v[x];
    sort(w+1,w+m+1);
    if(ans>w[m]) cout<<(ans<<m)<<endl; else cout<<((ans+w[m])<<(m-1))<<endl;
    return 0;
}
