#include<iostream>
#include<math.h>
#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
 int n,m;
 cin>>n>>m;
 long long int a[n],b[m],sum1=0,sum2=0;
 for(int i=0;i<n;++i)
 {cin>>a[i];sum1+=a[i];}
 for(int j=0;j<m;++j)
 {
   int x;
   cin>>x;
   b[j]=a[x-1];
   sum1-=a[x-1];       
 }
 sort(b,b+m);
 for(int i=m-1;i>=0;--i)
 {
  if(b[i]<sum1)
  sum1+=sum1;
  else
  sum1+=b[i];        
 }
 cout<<sum1<<endl;
// system("pause");
 return 0;   
}
