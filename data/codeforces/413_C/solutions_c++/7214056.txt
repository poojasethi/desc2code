#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;
int n,m; long long a[105],b[105],ans;
int main()
{
    scanf("%d%d",&n,&m); int x;
    for(int i=0;i<n;++i) scanf("%lld",&a[i]);
    for(int i=0;i<m;++i) {scanf("%d",&x); b[i]=a[x-1]; a[x-1]=0;}
    for(int i=0;i<n;++i) ans+=a[i];
    sort(b,b+m);
    for(int i=m-1;i>=0;--i) ans+=max(ans,b[i]);
    printf("%lld\n",ans);
}
