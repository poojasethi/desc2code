# include <bits/stdc++.h>
int h[105];
int a[105];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);

	int i,j ,maxI;
	for(i=0;i<m;i++)
	{
		maxI=0 ;
		for(j=0;j<n;j++)
		{
			scanf("%d",&a[j]);
			if(a[j]>a[maxI])
				maxI=j;
		}
		h[maxI]++;
	}
	maxI=0;
	for( i=0;i<n;i++)
	{
		if(h[i]>h[maxI])
			maxI=i;
	}

	printf("%d\n",maxI+1 );
}
