#include <stdio.h>

int main()
{	
	int n,m,a[103]={0},i,j,max,pos,x;
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;i++)
	{
		max=-1;
		for(j=1;j<=n;j++)
		{
			scanf("%d",&x);
			if(max<x)
			{
				max=x;
				pos=j;
			}
		}
		a[pos]++;
	}
	pos=0;
	max=-1;
	for(i=1;i<=n;i++)
	{
		if(max<a[i])
		{
			max=a[i];
			pos=i;
		}	
	}
	printf("%d",pos);
	return 0;
}
