#include <bits/stdc++.h>
using namespace std;
int main(){
	int n,m;
	cin>>n>>m;
	int wins[n]={0};
	for(int i=0;i<m;i++){
		int max=-1,ind=-1,tmp;
		for(int j=0;j<n;j++){
			cin>>tmp;
			if(tmp>max){max=tmp;ind=j;}
		}
		wins[ind]++;
	}
	int max=-1,ind=-1;
	for(int i=0;i<n;i++){
		if(wins[i]>max){max=wins[i];ind=i;}
	}
	cout<<ind+1<<endl;
}
