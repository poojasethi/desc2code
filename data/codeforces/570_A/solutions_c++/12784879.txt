#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,m;
	cin>>n>>m;
	int B[n];
	memset(B,0,sizeof B);
	for(int i=0;i<m;i++){
		int mx=-1,idx=-1;
		for(int j=0;j<n;j++){
			int a;
			cin>>a;
			if(mx<a){ mx=a; idx=j;}
		}
		B[idx]++;
	}
	int mx=-1,idx=-1;
	for(int i=0;i<n;i++) if(mx<B[i]) { mx=B[i];idx=i;}
	cout<<idx+1<<endl;	
return 0;
}
