#include<bits/stdC++.h>
using namespace std;
int cnt[100001];
int main()
{

int n,m,i,j;
cin>>n>>m;
for(i=0;i<m;i++)
{
int x=0,mx=INT_MIN,wi=-1;
for(j=0;j<n;j++)
{
cin>>x;
if(x>mx)
{
mx=x;
wi=j;

}


}
cnt[wi]++;

}

int ans=0,mx=0;
for(j=0;j<n;j++)
if(cnt[j]>mx)
{
mx=cnt[j];
ans=j+1;
}


cout<<ans<<endl;

}
