#include <bits/stdc++.h>
using namespace std;
const int N = 101;
int pp[N],rr[N];
int main(){
	int n,m,j,i,r=0,x;
	cin>>n>>m;
	pp[0]=-1;rr[0]=-1;
	for(i=0;i<m;++i){
		for(j=1;j<=n;++j)
			cin>>rr[j];
		
		r=0;
		for(j=1;j<=n;++j)
			if(rr[r]<rr[j])r=j;
		pp[r]++;
	}
	r=0;
	for(j=1;j<=n;++j)
		if(pp[r]<pp[j])r=j;
	cout<<r<<endl;
}
