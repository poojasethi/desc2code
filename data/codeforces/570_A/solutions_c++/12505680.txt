#include <bits/stdc++.h>
using namespace std;

int main(){
	int n,m;
	cin>>n>>m;
	int b[n];
	for(int i=0;i<n;i++){
		b[i]=0;
	}
	int y=0;
	for(int i=0;i<m;i++){
		int a=0;
		int t=0,x;
		for(int j=0;j<n;j++){
			cin>>x;
			if(x>t){
				a=j;
				t=x;
			}
		}
		b[a]++;
		if(b[a]>b[y])y=a;
	}
	y=0;
	for(int i=0;i<n;i++){
		if(b[i]>b[y])y=i;
	}
cout<<y+1<<endl;

}
