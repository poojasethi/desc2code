#include<iostream>
using namespace std;

int d[111],n,m,f[111];

int main(){

cin>>n>>m;

int mx=-99999,x;

for(int i=1; i<=m; i++){

mx=-999999;
int k=0;

for(int j=1; j<=n; j++){
cin>>x;
if(x>mx){mx=x; k=j;}
}

d[k]++;

}

mx=-99999;

int cnt;

for(int i=1; i<=n; i++){

}

}
