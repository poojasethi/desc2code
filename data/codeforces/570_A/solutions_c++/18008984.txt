#include <bits/stdc++.h>
using namespace std;
int b[1001];
int main(){
 int n,m,a[1001][1001],maxx,ind,i,j;
 cin>>n>>m;
 for(i=1;i<=m;i++){
  maxx=-1;ind=0;
  for(j=1;j<=n;j++){
   cin>>a[i][j];
   if(maxx<a[i][j]){
    maxx=a[i][j];
    ind=j;
   }
  }
  b[ind]++;
 }
 ind=0;
 for(i=1;i<=n;i++){
  if(b[ind]<b[i]){
   ind=i;
  }
 }
 cout<<ind;
}
