#include<stdio.h>
long long n,m,i,a[101][101],max,b[101],k,j;
int main(){
	scanf("%d%d",&n,&m);
	for(i=0;i<m;i++){
		max=0,k=0;
		for(j=0;j<n;j++){
			scanf("%d",&a[i][j]);
			if(max<a[i][j]){max=a[i][j];k=j;}
		}
		b[k]++;
	}
	max=0;k=0;
	for(i=0;i<n;i++)
		if(max<b[i]){max=b[i];k=i;}
	printf("%d",k+1);
}
