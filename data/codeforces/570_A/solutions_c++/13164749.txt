#include<iostream>
#include<algorithm>
using namespace std;
long long n, a[100], index, i, j, k, b, m;
main(){
  cin>>k>>n;
  for(i=0;i<n;i++){
    index=0;m=-1;
    for(j=0;j<k;j++){
      cin>>b;
      if(b>m){m=b;index=j;}
    }
    a[index]++;
  }
  index=0;m=-1;
  for(i=0;i<100;i++)if(a[i]>m){m=a[i];index=i;}
  cout<<index+1;
}
