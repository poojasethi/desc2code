#include<bits/stdc++.h>
using namespace std;


int main()
{

	int n,m;
	cin>>n>>m;
	vector <int> count(n+1,0);
	for(int i=1;i<=m;i++)
	{
		int j=1,k=INT_MIN;
		
		for(int i=1;i<=n;i++)
		{
			int x;
			cin>>x;
			if(x>k)
			{
				j=i;
				k=x;
			}
		}
		
		count[j]++;
	}
	
	cout<<max_element(count.begin(),count.end())-count.begin();
}
