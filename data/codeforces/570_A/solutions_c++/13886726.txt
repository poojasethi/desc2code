#include <iostream>
using namespace std;
int a[100][100],b,c,d,e,m[100];
int main(){
	cin>>b>>c;
	for(int i=0;i<c;i++){
		for(int j=0;j<b;j++){
			cin>>a[i][j];
			if(a[i][j]>d){
				d=a[i][j];
				e=j;
			}
		}
		m[e]++;
		e=0;
		d=0;
	}
	for(int i=0;i<b;i++){
		if(m[i]>d){
			d=m[i];
			e=i;
		}
	}
	cout<<e+1;
}
