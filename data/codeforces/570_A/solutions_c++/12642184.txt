#include <bits/stdc++.h>
using namespace std;
#define REP(i,j) for(int i=0;i<j;i++)

int n,m,k;
int T[101];

int main(){
	scanf("%d%d",&n,&m);
	REP(i,m){
		int it = 0, mx = 0;
		REP(i,n){
			int a;scanf("%d",&a);
			if(a > mx){mx = a;it = i;}
		}
		
		T[it]++;
		if(T[k] == T[it] && it < k)k = it;
		if(T[k] < T[it])k = it;
	}
	
	printf("%d\n",k+1);
}
