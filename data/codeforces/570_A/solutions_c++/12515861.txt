#include<iostream>
#include<algorithm>

using namespace std;

int a[105], sum[105];

int main()
{
	int n, m;
	cin >> n >> m;
	for(int i = 0; i < m; i++)
	{
		for(int j = 0; j < n; j++)
			cin >> a[j];
		sum[max_element(a, a + n) - a]++;
	}
	cout << max_element(sum, sum + n) - sum + 1 << endl;
	return 0;
}
