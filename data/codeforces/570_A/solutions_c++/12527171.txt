#define N (1<<7)
#include <bits/stdc++.h>

using namespace std;

int n,m,x,f[N],id,res=-1;

int main()
{
	cin>>n>>m;
	for(int i=1;i<=m;i++)
	{
		int mx=-1,id;
		for(int j=1;j<=n;j++)
		{
			scanf("%d",&x);
			if(x>mx) mx=x,id=j;
		}
		f[id]++;
	}
	
	for(int i=1;i<=n;i++)
		if(f[i]>res) id=i,res=f[i];
	cout<<id<<endl;
}
