int main()
{int n,m,i,j,maxi=-1,max=-1,r,k,t;

{for(j=0;j<n;j++)
{cin>>A[i][j];}}

{for(j=0;j<n;j++)
{if(A[i][j]>max)
{max=A[i][j];
t=j+1;}
}max=-1;

}

{maxi=B[k];
