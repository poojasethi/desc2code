#include <stdio.h>

int main(){
	int n,m,i,j,k,s,x;

	scanf("%d%d", &n, &m);

	int a[n];

	for(i=0;i<n;i++)
		a[i]=0;

	while(m--){
		for(i=0,s=0,k=0;i<n;i++){
			scanf("%d", &x);

			if(x>s){
				s=x;
				k=i;
			}
		}

		a[k]++;
	}

	for(i=0,k=0;i<n;i++)
		if(a[i]>a[k])
			k=i;

	printf("%d\n", k+1);

	return 0;
}
