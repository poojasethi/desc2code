#include<iostream>
#include<algorithm>
using namespace std;
long long n, a[100],r, i, j, k, q, mx;
main(){
  cin>>k>>n;
  for(i=0;i<n;i++)
   {
    r=0; mx=-1;
    for(j=0;j<k;j++)
     {
      cin>>q;
      if(q>mx){ mx=q; r=j;}
     }
    a[r]++;
   }
  r=0; mx=-1;
  for(i=0;i<100;i++) 
   if(a[i]>mx){ mx=a[i]; r=i;}
  cout<<r+1<<endl;
}
