#include <iostream>

using namespace std;

int x[100],n,m,d,e,a,b,c;

int main(){
  cin>>n>>m;
  for(int i=0;i<m;i++){
    b=-1;
    for(int j=0;j<n;j++){
      cin>>a;
      if (a>b){
	b=a;
	c=j;
      }
    }
    x[c]++;
  }
  for(int i=0;i<101;i++)
    if(x[i]>d){
      d=x[i];
      e=i+1;
    }
  cout<<e;
}
