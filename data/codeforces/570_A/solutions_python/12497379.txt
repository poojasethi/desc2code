n, m = map(int,raw_input().split())
flg = [0]*113
for i in xrange(m):
    lst = map(int, raw_input().split())
    mx = 0
    for j in xrange(1,n):
        if(lst[j] > lst[mx]):
            mx = j
    flg[mx]+=1
ans = 0
for k in xrange(1,n):
    if(flg[k] > flg[ans]):
        ans = k
ans = ans+1
print ans
