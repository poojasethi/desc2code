R = lambda: map(int, raw_input().split())
n, m = R()

points = [0] * n

for _ in range(m):

    sub = R()
    winner = 0
    winnerI = 0
    for i in range(n):
        if sub[i] > winner:
            winner = sub[i]
            winnerI = i

    points[winnerI] += 1

print points.index(max(points)) + 1
