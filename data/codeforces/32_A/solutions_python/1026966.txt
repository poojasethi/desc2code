ysl = raw_input('')

d = ysl.find(' ')
n = int(ysl[:d])
d = int(ysl[d+1:])

strDl = raw_input('')

spDl = []
spDl = strDl.split(' ')
spDl = [int(x) for x in spDl]

dl = len(spDl)
kol = 0

while dl > 1:
    pCh = spDl[0] - d
    vtCh = pCh + 2*d

    for x in range(1,dl):
        if pCh <= spDl[x] <= vtCh:
            kol = kol + 2

    del spDl[0]

    dl = dl - 1

print kol
