ysl = raw_input('')

d = ysl.find(' ')
n = int(ysl[:d])
d = int(ysl[d+1:])

strDl = raw_input('')

spDl = []
spDl = strDl.split(' ')

dl = len(spDl)
kol = 0

while dl > 1:
    pCh = int(spDl[0]) - d
    vtCh = pCh + 2*d
    
    for x in range(1,dl):
        if pCh <= int(spDl[x]) <= vtCh:
            kol = kol + 2

    spDl.remove(spDl[0])

    dl = len(spDl)

print kol
