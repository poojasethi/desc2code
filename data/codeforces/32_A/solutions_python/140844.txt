from sys import stdin

def line():
    return stdin.readline()

def answer(n, d, heights):
    heights.sort()
    count = 0
    i = j = 0
    while j < n - 1:
        j += 1
        while heights[j] - heights[i] > d:
            i += 1
        count += j - i
    return 2 * count

n, d = map(int, line().split())
heights = list(map(int, line().split()))
print(answer(n, d, heights))
