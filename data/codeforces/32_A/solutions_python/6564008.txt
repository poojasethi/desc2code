__author__ = 'firago'
from sys import stdin, stdout
from bisect import bisect_right


n,d = map(int, stdin.readline().split())
l = map(int, stdin.read().split())
a = sorted(l)
s = 0
for i in xrange(n):
    D = d + a[i]
    j = bisect_right(a,D)
    c = j - i - 1
    s += 2 * c

print s
