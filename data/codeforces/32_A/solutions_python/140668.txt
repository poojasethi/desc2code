import sys

if __name__ == "__main__":
    lines = sys.stdin.readlines()
    dif = int(lines[0].split(" ")[-1])
    n = [int(i) for i in lines[1].split(" ")]
    ret = 0

    for i in range(len(n)):
        for j in range(len(n)):
            if i == j:
                continue
            if abs(n[i] - n[j]) <= dif:
                ret += 1        
    
    sys.stdout.write(str(ret))
