import math
def main():

    inp = map(int, raw_input().split())

    ls = [sorted([inp[i],inp[j]]) for i in xrange(sz-1) for j in xrange(i+1, sz) if math.fabs(inp[i]-inp[j]) <= d]

    print len(ls)*2
