ysl = raw_input('')

d = ysl.find(' ')
n = int(ysl[:d])
d = int(ysl[d+1:])

strDl = raw_input('')

spDl = []
spDl = strDl.split(' ')

dl = len(spDl)
kol = 0

for x in range(dl-1):
    pCh = int(spDl[x]) - d
    vtCh = pCh + 2*d

    for y in range(x+1,dl):
        if pCh <= int(spDl[y]) <= vtCh:
            kol = kol + 2

print kol
