import sys
from itertools import product

file = sys.stdin
#file = open("test", "r")

n, d = map(int, file.readline().rstrip().split())
p = map(int, file.readline().rstrip().split())

cnt = 0
for i, j in [(i, j) for i in range(n) for j in range(n) if i != j]:
    if abs(p[i] - p[j]) <= d: cnt += 1

print cnt
