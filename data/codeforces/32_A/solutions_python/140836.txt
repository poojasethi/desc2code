

def split_int():
  s=raw_input()
  return map(int, s.split(" "))


def compute(m,n,d):
  s=0
  for i in xrange(n):
    for j in xrange(n):
      if abs(m[i]-m[j])<=d:s+=1
  return s-n

def solve():
  n,d=split_int()
  print compute( split_int() , n,d )
