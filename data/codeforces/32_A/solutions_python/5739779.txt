#!/usr/bin/env python
# -*- coding: UTF-8 -*-
import time
import sys, io
import re, math
start = time.clock()
ans=0
(n,d)=map(int, raw_input().split())
l=[int(x) for x in raw_input().split()]
l.sort()
for i in range(n):
    j=i
    while 1:
        j+=1
        if j>n-1: break
        elif l[j]-l[i]>d: break
        elif l[j]-l[i]<=d: ans+=2
print ans
