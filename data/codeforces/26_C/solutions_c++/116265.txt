#include <cstdlib>
#include <cstdio>

int n, m, a, b, c;
char p[100][100];
char l[3] = "uv";

void impossible()
{
    puts("IMPOSSIBLE");
    exit(0);
}

int main()
{
    scanf("%d %d %d %d %d", &n, &m, &a, &b, &c);
    int N = n, M = m;

    if (m*n % 2 == 1) impossible();
    if (m & 1) {
        b -= n / 2;
        if (b < 0) impossible();
        m--;
        for(int i = 0; i < n; i += 2)
            p[i][m] = p[i + 1][m] = l[(i / 2) & 1];

    } else if (n & 1) {
        a -= m / 2;
        if (a < 0) impossible();
        n--;
        for(int i = 0; i < m; i += 2)
            p[n][i] = p[n][i + 1] = l[(i / 2) & 1];
    }

    a = a & ~1;
    b = b & ~1;

    if (m*n > 2*(a + b + 2*c)) impossible();

    for(int i = 0; i < n; i += 2) {
        for(int j = 0; j < m; j += 2) {
            int k = ((i / 2) % 2) + 2 * ((j / 2) % 2);
            if (c) {
                p[i][j] = p[i][j + 1] = p[i+1][j] = p[i + 1][j + 1] = k + 'a';
                --c;
            } else if (a) {
                char plank = 2 * k + 'e';
                p[i][j] = p[i][j + 1] = plank;
                p[i+1][j] = p[i+1][j+1] = plank + 1;
                a -= 2;
            } else {
                char plank = 2 * k + 'k';
                p[i][j] = p[i+1][j] = plank;
                p[i][j+1] = p[i+1][j+1] = plank + 1;
                //b -= 2;
            }
        }
    }

    for(int i = 0; i < N; i++) {
        for(int j = 0; j < M; j++)
            putchar(p[i][j]);
        putchar('\n');
    }
}
