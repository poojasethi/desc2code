#define pb(x) push_back(x)

using namespace std;

const int maxn=110;

char d[maxn][maxn];
int n,m,a,b,c;

void f(int x,int y){
	for(int i=x;i<n;i+=2)
		for(int j=y;j<m;j+=2)
			if(c){
				d[i][j]=d[i+1][j]=d[i][j+1]=d[i+1][j+1]='a'+(i+j)%4;
				c--;
			}else if(a > 1){
				a-=2;
				d[i][j]=d[i][j+1]='a'+(i+j)%4;
				d[i+1][j]=d[i+1][j+1]='a'+(i+j+1)%4;
			}else{
				b-=2;
				d[i][j]=d[i+1][j]='a'+(i+j)%4;
				d[i][j+1]=d[i+1][j+1]='a'+(i+j+1)%4;
			}

}

int main(){
	cin >> n >> m >> a >> b >> c;
	if(n%2 == 0 && m%2 == 0){
		if(a/2+b/2+c >= n*m/4){
			f(0,0);
			rep(i,n){
				rep(j,m)
					cout << d[i][j];
				cout << endl;
			}
		}else
			cout << "IMPOSSIBLE" << endl;
	}else if(n%2 == 0){
		if(b < n/2 || a/2+(b-n/2)/2+c < n*(m-1)/4)
			cout << "IMPOSSIBLE" << endl;
		else{
			b-=n/2;
			for(int i=0;i<n;i+=2)
				d[i][0]=d[i+1][0]='y'+(i%4)/2;
			f(0,1);
			rep(i,n){
				rep(j,m)
					cout << d[i][j];
				cout << endl;
			}
		}
	}else if(m%2 == 0){
		if(a < m/2 || (a-m/2)/2+b/2+c < (n-1)*m/4)
			cout << "IMPOSSIBLE" << endl;
		else{
			a-=m/2;
			for(int i=0;i<m;i+=2)
				d[0][i]=d[0][i+1]='y'+(i%4)/2;
			f(1,0);
			rep(i,n){
				rep(j,m)
					cout << d[i][j];
				cout << endl;
			}
		}
	}else
		cout << "IMPOSSIBLE" << endl;
	return 0;
}
