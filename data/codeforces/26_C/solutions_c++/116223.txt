#include <iostream>
using namespace std;

char room[100][100];
int n,m,a,b,c;
char sample[] = "abcdefghijklmnopqrstuvwxyz";

void pfillroom(int x,int y,int &ct,int &flag)
{
    if (flag) return;
    if (c > 0)
    {
        --c;
        room[x*2][y*2]=room[x*2+1][y*2]=room[x*2][y*2+1]=room[x*2+1][y*2+1]=sample[(x%2)*11+(y%2)*6];
        return;
    }
    if (a > 1) 
    {
        a-=2;
        room[x*2][y*2]=room[x*2][y*2+1]=sample[(x%2)*11+(y%2)*6+1];
        room[x*2+1][y*2]=room[x*2+1][y*2+1]=sample[(x%2)*11+(y%2)*6+2];
        return;
    }
    if (b > 1) 
    {
        b-=2;
        room[x*2][y*2]=room[x*2+1][y*2]=sample[(x%2)*11+(y%2)*6+3];
        room[x*2][y*2+1]=room[x*2+1][y*2+1]=sample[(x%2)*11+(y%2)*6+4];
        return;
    }
    flag=1;
}
int main()
{
#ifdef ONLINE_JUDGE
#else
    freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
#endif
    int i,j,nt,mt,ct = 0,flag = 0;
    cin >> n >> m >> a >> b >> c;
    nt=n / 2;mt=m / 2;
    if ((n * m) % 2) cout << "IMPOSSIBLE";
    else
    {
        if (n % 2)
            for (i = 0;i < mt;++i) {room[n - 1][i * 2]=room[n - 1][i * 2 + 1]=sample[22+(i%2)];--a;}
        else if (m % 2)
            for (i = 0;i < nt;++i) {room[i * 2][m - 1]=room[i * 2 + 1][m - 1]=sample[24+(i%2)];--b;}
        if (a < 0 || b < 0) flag=1;
        for (i = 0;i<nt;++i)
            for (j = 0;j<mt;++j)
                pfillroom(i,j,ct,flag);
        if (flag) cout << "IMPOSSIBLE";
        else
        {
            for (i = 0;i < n;++i)
            {
                for (j = 0;j < m;++j) cout << room[i][j];
                cout << endl;
            }
        }
    }
    cout << endl;
    return 0;
}
