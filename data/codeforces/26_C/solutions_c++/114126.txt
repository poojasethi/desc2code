#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
#include <map>
#include <cstdio>
#include <cmath>
#include <map>

using namespace std;

int n,m,a,b,c,N,M;

char ans[110][110];
char ch[10];

int get(int x,int y){
	return ((x%4)-1)*2+(y%4)-1;
}

int main(){
	for(int i=0;i<=8;i++)
		ch[i]='a'+i;
	cin >> n >> m >> a >> b >> c;
	N=n,M=m;
	if(n%2==1 && m%2==1){
		cout << "IMPOSSIBLE" << endl;
		return 0;
	}
	if(n%2==1){
		for(int i=1;i<=m;i+=2){
			ans[n][i]=ans[n][i+1]=ch[get(n,i)];;
			a--;
			if(a<0){
				cout << "IMPOSSIBLE" << endl;
				return 0;
			}
		}
		n--;
	}
	if(m%2==1){
		for(int i=1;i<=n;i+=2){
			ans[i][m]=ans[i+1][m]=ch[get(i,m)];
			b--;
			if(b<0){
				cout << "IMPOSSIBLE" << endl;
				return 0;
			}
		}
		m--;
	}
	for(int i=1;i<=n;i+=2)
		for(int j=1;j<=m;j+=2){
			if(c>0){
				ans[i][j]=ans[i+1][j]=ans[i][j+1]=ans[i+1][j+1]=ch[get(i,j)];
				c--;
				continue;
			}
			if(a>1){
				int g=ch[get(i,j)];
				ans[i][j]=ans[i][j+1]=g;
				ans[i+1][j]=ans[i+1][j+1]=g+1;
				a-=2;
				continue;
			}
			if(b>1){
				int g=ch[get(i,j)];
				ans[i][j]=ans[i+1][j]=g;
				ans[i][j+1]=ans[i+1][j+1]=g+1;
				b-=2;
				continue;
			}
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}
	for(int i=1;i<=N;i++){
		for(int j=1;j<=M;j++)
			cout << ans[i][j];
		cout << endl;
	}
}
