#include<iostream>
#include<algorithm>
using namespace std;

char r[110][110];
int a, b, c, n, m, nn, mm;


int main(){
	cin >> n >> m >> a >> b >> c;
	nn = n;
	mm = m;
	if((n*m) % 2 == 1){
		cout << "IMPOSSIBLE" << endl;
		return 0;
	}
	if(n %2 == 1){
		if(a < m/2){
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}
		a -= m/2;
		for(int i = 0; i < m; i += 4){
			r[n-1][i] = r[n-1][i+1] = 'x';
			r[n-1][i+2] = r[n-1][i+3] = 'z';
		}
		n--;
	}
	if(m % 2 == 1){
		if(b < n/2){
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}
		b -= n/2;
		for(int i = 0; i < n; i += 4){
			r[i][m-1] = r[i+1][m-1] = 'x';
			r[i+2][m-1] = r[i+3][m-1] = 'z';
		}
		m--;
	}
	if(m*n > ((a/2)*4) +((b/2)*4) + c*4){
	//cerr << ((a/2)*4) +((b/2)*4) + c*4 << "  " ;
		cout << "IMPOSSIBLE" << endl;
		return 0;
	}
	
	a = (a/2);
	b = (b/2);
	
	//cerr << "N " << n << " " << m << endl;
	for(int i = 0; i < n; i += 2)
		for(int j = 0; j < m; j += 2){
			if(c > 0){
				c--;
				char ch = 'a' + ((i+j)%4)/2;
				r[i][j] = r[i+1][j] = r[i][j+1] = r[i+1][j+1] = ch;
			}
			else if(b > 0){
				b--;
				char ch = 'c' +  ((i+j)%4);
				r[i][j] = r[i+1][j] = ch;
				r[i][j+1] = r[i+1][j+1] = ch+1;
			}
			else if(a > 0){
				a--;
				char ch = 'g' +  ((i+j)%4);
				r[i][j] = r[i][j+1] = ch;
				r[i+1][j+1] = r[i+1][j] = ch+1;
			}
			else {
				//cerr << " ROOM " << i << " " << j << endl;
				cout << "IMPOSSIBLE" << endl;
				return 0;
			}
		}
	//cerr << nn << " " << mm << endl;
	for(int i = 0; i < nn; i++){
		for(int j = 0; j < mm; j++)
			cout << r[i][j];
		cout << endl;
	}
	return 0;
}
