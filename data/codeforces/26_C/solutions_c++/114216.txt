#include <cstdio>
#include <cstring>

char s[111][111];


int main (void)
{
  int i, j, n, m, a, b, c, k, n0, m0;
  bool f = true;
  scanf ("%d %d %d %d %d", &n, &m, &a, &b, &c);
  n0 = n, m0 = m;
  if ((n * m) & 1)
    f = false;
  else if (n & 1)
  {
    if (2 * a < m)
      f = false;
    else
      for (i = 0; i < m; i++)
        s[n - 1][i] = 'a' + ((i % 4) > 1);
    n--, a -= m/2;
  }
  else if (m & 1)
  {
    if (2 * b < n)
      f = false;
    else
      for (i = 0; i < n; i++)
        s[i][m - 1] = 'a' + ((i % 4) > 1);
    m--, b -= n/2;
  }
  if (a & 1)
    a--;
  if (b & 1)
    b--;
  if (f)
  {
    for (i = 0; i < n; i+=2)
      for (j = 0; j < m; j+=2)
      {
        k = (i/2 + j/2) & 1;
        if (c > 0)
        {
          s[i][j] = 'c' + k;
          s[i + 1][j] = 'c' + k;
          s[i][j + 1] = 'c' + k;
          s[i + 1][j + 1] = 'c' + k;
          c--;
        }
        else if (a > 0)
        {
          s[i][j] = 'e' + 2*k;
          s[i + 1][j] = 'f' + 2*k;
          s[i][j + 1] = 'e' + 2*k;
          s[i + 1][j + 1] = 'f' + 2*k;
          a -= 2;
        }
        else if (b > 0)
        {
          s[i][j] = 'i' + 2*k;
          s[i + 1][j] = 'i' + 2*k;
          s[i][j + 1] = 'j' + 2*k;
          s[i + 1][j + 1] = 'j' + 2*k;
          b -= 2;
        }

        else
        {
          f = false;
          break;
        }
      }  
  }

   
  if (f)
    for (i = 0; i < n0; i++)
    {
      for (j = 0; j < m0; j++)
        putchar (s[i][j]);
      putchar('\n');
    }
  else
    printf ("IMPOSSIBLE\n");
  return 0;
}
