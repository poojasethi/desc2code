#define FOR(i,m,n) for(int i=m; i<n; i++)
#define ms(obj, val) memset(obj, val, sizeof(obj))
#define mp make_pair
#define pb push_back
#define ri(a) scanf("%d",&a)
#define rii(a,b) scanf("%d%d",&a,&b)

typedef long long ll;
typedef vector<int> vi;

const int MAXN=105;
char sol[MAXN][MAXN];

int main(){
	int n, m, a, b, c, M, N;
	rii(n, m); ri(a); ri(b); ri(c);
	M=m; N=n;
	if(n%2 && m%2) {
		printf("IMPOSSIBLE\n");
		return 0;
	}
	if(n%2){
		if(a<m/2) {
			printf("IMPOSSIBLE\n");
			return 0;
		}
		FOR(j, 0, m) sol[n-1][j]=(j%4<2 ? 'x' : 'y');
		a-=m/2;
	}
	if(m%2){
		if(b<n/2){
			printf("IMPOSSIBLE\n");
			return 0;
		}
		FOR(i, 0, n) sol[i][m-1]=((i%4)<2 ? 'x' : 'y');
		b-=n/2;
	}
	m/=2; n/=2; a/=2; b/=2;
	if(a+b+c<n*m){
		printf("IMPOSSIBLE\n");
		return 0;
	}
	FOR(i, 0, n) FOR(j, 0, m){
		if(c){
			if(i%2==j%2) FOR(k1, 0, 2) FOR(k2, 0, 2) sol[2*i+k1][2*j+k2]='a';
			else         FOR(k1, 0, 2) FOR(k2, 0, 2) sol[2*i+k1][2*j+k2]='b';
			c--;
		}
		else if(a){
			if(j%2){
				sol[2*i][2*j]=sol[2*i][2*j+1]='c';
				sol[2*i+1][2*j]=sol[2*i+1][2*j+1]='d';
			}
			else{
				sol[2*i][2*j]=sol[2*i][2*j+1]='d';
				sol[2*i+1][2*j]=sol[2*i+1][2*j+1]='c';
			}
			a--;
		}
		else{
			if(i%2){
				sol[2*i][2*j]=sol[2*i+1][2*j]='m';
				sol[2*i][2*j+1]=sol[2*i+1][2*j+1]='n';
			}
			else{
				sol[2*i][2*j]=sol[2*i+1][2*j]='n';
				sol[2*i][2*j+1]=sol[2*i+1][2*j+1]='m';
			}
			b--;
		}
	}
	FOR(i, 0, N){
		FOR(j, 0, M) printf("%c", sol[i][j]);
		printf("\n");
	}
}
