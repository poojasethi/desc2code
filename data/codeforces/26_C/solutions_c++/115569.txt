#include <iostream>
#include <vector>
#include <cstring>
#include <algorithm>
using namespace std;

typedef long long ll;
typedef pair<int, int> PII;
#define F first
#define S second

int nn,mm,a,b,c;
char p[200][200];

void bad(){
	cout<<"IMPOSSIBLE"<<endl;
	exit(0);
}
int main(){
	cin >> nn >> mm >> a >> b >> c;
	int n=nn, m=mm;
	if(n%2 && m%2)	bad();
	if(n%2){
		if(a < m/2)	bad();
		for(int i=0; i<m/2; i++){
			a--;
			if(i%2==0)	p[n-1][i*2]=p[n-1][i*2+1]='a';
			else		p[n-1][i*2]=p[n-1][i*2+1]='b';
		}
		n--;
	}
	if(m%2){
		if(b < n/2)	bad();
		for(int i=0; i<n/2; i++){
			b--;
			if(i%2==0)	p[i*2][m-1]=p[i*2+1][m-1]='a';
			else		p[i*2][m-1]=p[i*2+1][m-1]='b';
		}
		m--;
	}

	for(int i=0; i<n-1; i+=2)
		for(int j=0; j<m-1; j+=2){
			if(c>0){
				c--;
				p[i][j]=p[i+1][j]=p[i][j+1]=p[i+1][j+1]=char('c'+(i/2)%2+(j/2)%2);

			}else if(a > 1){
				a-=2;
				p[i][j]=p[i][j+1]=char('g'+(j/2)%2);
				p[i+1][j]=p[i+1][j+1]=char('i'+(j/2)%2);
			}else if(b>1){
				b-=2;
				p[i][j]=p[i+1][j]='k'+(i/2)%2;
				p[i][j+1]=p[i+1][j+1]='m'+(i/2)%2;
			}else
				bad();
		}
	for(int i=0; i<nn; i++,cout<<endl)
		for(int j=0; j<mm; j++)
			cout<<p[i][j];
}
