#include<iostream>
#include<cstring>
#include<cstdio>
#include<cctype>

using namespace std;

int n , m , a , b , c;
char str[100][100];

bool solve() {
	if ( a < 0 || b < 0 ) return false;
	
	for ( int i = 0 ; i + 1 < n ; i += 2 )
		for ( int j = 0 ; j + 1 < m ; j += 2 ) {
			int p = ( i / 2 + j / 2 ) & 1;
			if ( c >= 1 ) {
				str[i][j] = str[i][j + 1] = str[i + 1][j] = str[i + 1][j + 1] = 'a' + p;
				c--;
			} else
			if ( a >= 2 ) {
				str[i][j] = str[i][j + 1] = 'c' + p;
				str[i + 1][j] = str[i + 1][j + 1] = 'e' + p;
				a -= 2;
			} else
			if ( b >= 2 ) {
				str[i][j] = str[i + 1][j] = 'c' + p;
				str[i][j + 1] = str[i + 1][j + 1] = 'e' + p;
				b -= 2;
			} else
				return false;
		}
	
	return true;
}

int main() {
	scanf( "%d%d%d%d%d" , &n , &m , &a , &b , &c );
	
	if ( n % 2 == 1 && m % 2 == 1 ) {
		printf( "IMPOSSIBLE\n" );
		return 0;
	}
	if ( n % 2 == 1 ) {
		for ( int i = 0 ; i < m ; i += 2 ) {
			a--;
			str[n - 1][i] = str[n - 1][i + 1] = 'x' + ( i / 2 % 2 );
		}
	}
	if ( m % 2 == 1 ) {
		for ( int i = 0 ; i < n ; i += 2 ) {
			b--;
			str[i][m - 1] = str[i + 1][m - 1] = 'x' + ( i / 2 % 2 );
		}
	}
	if ( solve() ) {
		for ( int i = 0 ; i < n ; i++ ) {
			for ( int j = 0 ; j < m ; j++ )
				printf( "%c" , str[i][j] );
			printf( "\n" );
		}
	} else
		printf( "IMPOSSIBLE\n" );
	
	return 0;
}
