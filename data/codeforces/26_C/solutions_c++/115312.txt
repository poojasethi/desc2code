#include <iostream>
#include <cstdio>
#define imp "IMPOSSIBLE"
using namespace std;

int mas[205][205],m,n,a,b,c;
int amount;

int main(){
	//freopen("c.in","r",stdin);
	scanf("%d%d%d%d%d",&n,&m,&a,&b,&c);
	if ((m*n)%2==1) {printf("%s",imp); return 0;}
	if (m*n>a*2+b*2+c*4) {printf("%s",imp); return 0;}
	if (m%2==1){
		amount=n/2;
		if (amount>b) {printf("%s",imp); return 0;} else b-=amount;
		for (int i=1; i<=n; i+=2)
			if ((i/2)%2==0) {
				mas[i][m]='a'; mas[i+1][m]='a';
			} else {
				mas[i][m]='b'; mas[i+1][m]='b';
			}
	}
	if (n%2==1){
		amount=m/2;
		if (amount>a) {printf("%s",imp); return 0;} else a-=amount;
		for (int i=1; i<=m; i+=2)
			if ((i/2)%2==0) {
				mas[n][i]='c'; mas[n][i+1]='c';
			} else {
				mas[n][i]='d'; mas[n][i+1]='d';
			}
	}
	char c1,c2;
	for (int i=1; i<=(n/2); ++i)
		for (int j=1; j<=(m/2); ++j){
			if ((i&1)==(j&1)) 
				{if (i&1) {c1='e'; c2='f';} else {c1='g'; c2='h';}}
			else
				{if (i&1) {c1='i'; c2='j';} else {c1='k'; c2='l';}}
			if (c>0) {
				mas[i*2-1][j*2-1]=c1;
				mas[i*2][j*2]=c1;
				mas[i*2][j*2-1]=c1;
				mas[i*2-1][j*2]=c1;
				--c;
			} else {
			if (a>1) {
				mas[i*2-1][j*2-1]=c1;
				mas[i*2-1][j*2]=c1;
				mas[i*2][j*2]=c2;
				mas[i*2][j*2-1]=c2;
				a-=2;
			} else {
			if (b>1) {
				mas[i*2-1][j*2-1]=c1;
				mas[i*2][j*2-1]=c1;
				mas[i*2-1][j*2]=c2;
				mas[i*2][j*2]=c2;
				b-=2;
			} else {
				printf("%s",imp); return 0;
			}}}
		}
	for (int i=1; i<=n; ++i){
		for (int j=1; j<=m; ++j) printf("%c",mas[i][j]);
		printf("\n");
	}
	return 0;
}
