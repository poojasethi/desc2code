#include <iostream>
#include <string>
#include <vector>

using namespace std;

void pintacuadrat(vector<string> &v,int n,int m,int a,int b,int c)
{
  int nn=n;
  int mm=m;
  n=n-(n%2);
  m=m-(m%2);
  int cc=c+(a/2)+(b/2);
  a=a-(a%2);
  b=b-(b%2);
  if (4*cc<n*m) {
    cout<<"IMPOSSIBLE"<<endl;
  } else {
    for (int i=0;i<n/2;i++) {
      for (int j=0;j<m/2;j++) {
	char lletra='a'+2*((i+j)%2);
	if (c>0) {
	  v[2*i][2*j]=v[2*i+1][2*j]=v[2*i][2*j+1]=v[2*i+1][2*j+1]=lletra;
	  c--;
	} else if (a>=2) {
	  v[2*i][2*j]=v[2*i][2*j+1]=lletra;
	  v[2*i+1][2*j]=v[2*i+1][2*j+1]=lletra+1;
	  a--;a--;
	} else {
	  v[2*i][2*j]=v[2*i+1][2*j]=lletra;
	  v[2*i][2*j+1]=v[2*i+1][2*j+1]=lletra+1;
	  b--;b--;
	}
      }
    }
    for (int i=0;i<nn;i++) {
      for (int j=0;j<mm;j++) {
	cout<<v[i][j];
      }
      cout<<endl;
    }
  }
}

int main()
{
  int n,m,a,b,c;
  cin>>n>>m>>a>>b>>c;

  vector<string> v(n,string(m,' '));
  if (n%2==1 and m%2==1) {
    cout<<"IMPOSSIBLE"<<endl;
  } else if (n%2==1) {
    if (m/2>a) {
      cout<<"IMPOSSIBLE"<<endl;
    } else {
      for (int j=0;j<m/2;j++) {
	char lletra='e'+(j%2);
	v[n-1][2*j]=v[n-1][2*j+1]=lletra;
	a--;
      }
      pintacuadrat(v,n,m,a,b,c);
    }
  } else if (m%2==1) {
    if (n/2>b) {
      cout<<"IMPOSSIBLE"<<endl;
    } else {
      for (int i=0;i<n/2;i++) {
	char lletra='e'+(i%2);
	v[2*i][m-1]=v[2*i+1][m-1]=lletra;
	b--;
      }
      pintacuadrat(v,n,m,a,b,c);
    }
  } else {
    pintacuadrat(v,n,m,a,b,c);
  }
}
