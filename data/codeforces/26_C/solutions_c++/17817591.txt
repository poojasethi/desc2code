#include<stdio.h>

char p[110][110];
int m, n, a, b, c, mm, nn;

int main(){
    scanf("%d %d %d %d %d", &n, &m, &a, &b, &c);
    if(m%2 and n %2){ puts("IMPOSSIBLE"); return 0; }
    for(int i = 0; i <= n / 2; i ++) for(int j = 0; j <= m / 2; j ++)
     p[2*i][2*j] = p[2*i+1][2*j] = p[2*i][2*j+1] = p[2*i+1][2*j+1] = (i+j) % 2;
    mm = m / 2;   nn = n / 2;
    if(m%2) b -= nn;
    if(n%2) a -= mm;
    if(a < 0 || b < 0) { puts("IMPOSSIBLE"); return 0; }
    a = a / 2; b = b / 2;
    if(a + b + c < mm * nn) { puts("IMPOSSIBLE"); return 0; }
    for(int i = 0; i < nn; i ++) for(int j = 0; j < mm; j ++){
        if(c > 0){ c--; continue; }
        if(a > 0){
            a --;
            p[2*i][2*j] = p[2*i][2*j+1] = p[2*i][2*j] + 2;
            p[2*i+1][2*j] = p[2*i+1][2*j+1] = p[2*i+1][2*j] + 4;
        }
        else if(b > 0){
            b --;
            p[2*i][2*j] = p[2*i+1][2*j] = p[2*i][2*j] + 6;
            p[2*i][2*j+1] = p[2*i+1][2*j+1] = p[2*i][2*j+1] + 8;
        }
    }
    for(int i = 0; i < n; i ++){
        for(int j = 0; j < m; j ++) printf("%c", p[i][j]+97); puts("");
    }
}
