#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
using namespace std;
#define maxn 120
int n,m,a,b,c;
char st[maxn][maxn];

int main()
{
	scanf("%d%d%d%d%d",&n,&m,&a,&b,&c);
	if (n&1) {
		for (int j=0;j<m-1;j+=2) {
			if (!a) break;
			--a;
			int k=(j/2)&1;
			st[n-1][j]='c'+k;
			st[n-1][j+1]='c'+k;
		}
	}
	if (m&1) {
		for (int i=0;i<n-1;i+=2) {
			if (!b) break;
			--b;
			int k=(i/2)&1;
			st[i][m-1]='e'+k;
			st[i+1][m-1]='e'+k;
		}
	}
	for (int i=0;i<n-1;i+=2) {
		for (int j=0;j<m-1;j+=2) {
			int k=(i/2+j/2)&1;
			if (c>0) {
				c--;
				for (int x=0;x<2;x++)
				for (int y=0;y<2;y++)
					st[i+x][j+y]='a'+k;
			} else if (a>1) {
				a-=2;
				for (int x=0;x<2;x++)
				for (int y=0;y<2;y++)
					st[i+x][j+y]='g'+k*2+x;
			} else if (b>1) {
				b-=2;
				for (int x=0;x<2;x++)
				for (int y=0;y<2;y++)
					st[i+x][j+y]='k'+k*2+y;
			}
		}
	}
	bool Flag=true;
	for (int i=0;i<n;i++)
	for (int j=0;j<m;j++)
	if (!st[i][j]) Flag=false;
	if (Flag) {
		for (int i=0;i<n;i++)
			printf("%s\n",st[i]);
	} else {
		printf("IMPOSSIBLE\n");
	}
	return 0;
}
