#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <cstring>
#define rep(i,n) for(int i=0;i<n;i++)
#define pr(i) cout<<i<<endl
using namespace std;

int r[120][120];
int n,m,a,b,c;
int dx[4]={0,1,0,-1},dy[4]={1,0,-1,0};
bool used[26];

void j(int x,int y){

  rep(i,4){
    int nx=x+dx[i],ny=y+dy[i];
    if(nx>=0&&nx<=n-1&&ny>=0&&ny<=m-1&&r[nx][ny]>=0)used[r[nx][ny]]=true;
  }

}

int f(int x,int y,int type){

  fill(used,used+26,false);
  int cx[4]={x+1,x,x,x+1},cy[4]={y,y,y+1,y+1};

  if(type==1){
    for(int i=1;i<=2;i++)j(cx[i],cy[i]);
    rep(i,26)if(!used[i]){
      if(a>0){r[x][y]=i;r[x][y+1]=i;a--;return 1;}
      else return 0;
    }
  }
  else if(type==2){
    for(int i=0;i<=1;i++)j(cx[i],cy[i]);
    rep(i,26)if(!used[i]){
      if(b>0){r[x][y]=i;r[x+1][y]=i;b--;return 1;}
      else return 0;
    }
  }
  else {
    if(c>0){
      rep(i,4)j(cx[i],cy[i]);
      rep(i,26)if(!used[i]){
    rep(j,4)r[cx[j]][cy[j]]=i;
    c--;
    return 1;
      }
    }
    else if(a>=2) return f(x,y,1)&&f(x+1,y,1);
    else if(b>=2) return f(x,y,2)&&f(x,y+1,2);
    else return 0;
  }
  
}

int solve(){

  if(n%2&&m%2)return 0;

  int nn=n,mm=m;

  if(nn%2){
    rep(i,mm/2)if(f(n-1,i*2,1))continue;
    else return 0;
    nn--;
  }
  if(mm%2){
    rep(i,n/2)if(f(i*2,m-1,2))continue;
    else return 0;
    mm--;
  }

  rep(i,nn/2)rep(j,mm/2){
    if(f(i*2,j*2,3))continue;
    else return 0;
  }
  return 1;
  
}


int main(){

  cin>>n>>m>>a>>b>>c;
  rep(i,120)rep(j,120)r[i][j]=-1;

  if(solve())rep(i,n){rep(j,m)cout<<(char)(r[i][j]+'a');cout<<endl;}
  else pr("IMPOSSIBLE");

  return 0;
}
