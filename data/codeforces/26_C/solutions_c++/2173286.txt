#include <iostream>
#include <list>
#include <string>
#include <vector>
#include <iterator>
#include <algorithm>
#include <stdio.h>
#include <string.h>
#include <math.h>

using namespace std;

#define FOR(i,n) for(int i = 0;i < n;++i)
#define MOD 1000000007ll

char ar[101][101];

int main(int argc, char** argv)
{
	int n,m,A,B,C;
	cin >> n >> m >> A >> B >> C;
	int n0 = n,m0 = m;
	if((n*m)%2)
	{
		cout << "IMPOSSIBLE" << endl;
		return 0;
	}
	if(n%2)
	{
		if(A<m/2)
		{
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}
		FOR(i,m/2)
		{
			ar[n-1][2*i] = ar[n-1][2*i+1] = 'a' + i%2;
			--A;
		}
		--n;
	}
	if(m%2)
	{
		if(B<n/2)
		{
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}
		FOR(i,n/2)
		{
			ar[2*i][m-1] = ar[2*i+1][m-1] = 'a'+i%2;
			--B;
		}
		--m;
	}
	FOR(i,n/2)
		FOR(j,m/2)
		{
			if(C)
			{
				--C;
				ar[2*i][2*j] = ar[2*i+1][2*j] = ar[2*i][2*j+1] = ar[2*i+1][2*j+1] = 'c'+(i%2)*2 + j%2;
				continue;
			}
			if(A>=2)
			{
				ar[2*i][2*j] = ar[2*i][2*j+1] = 'h' + (j%2);
				ar[2*i+1][2*j] = ar[2*i+1][2*j+1] = 'j' + (j%2);
				A-=2;
				continue;
			}
			if(B>=2)
			{
				ar[2*i][2*j] = ar[2*i+1][2*j] = 'l' + (i%2);
				ar[2*i][2*j+1] = ar[2*i+1][2*j+1] = 'n' + (i%2);
				B-=2;
				continue;
			}
			cout << "IMPOSSIBLE" << endl;
			return 0;
		}
	FOR(i,n0)
	{
		FOR(j,m0)
			cout << ar[i][j];
		cout << endl;
	}
	return 0;	
}
