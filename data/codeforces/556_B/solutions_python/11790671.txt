n = input()
a = map(int, raw_input().split())

d = range(n)
x = a[0]-d[0]

flag=1
for i in range(1,n):
    if i%2==1 and (d[i]-x)%n != a[i]:
        flag=0
        break
    if i%2==0 and (d[i]+x)%n != a[i]:
        flag=0
        break

if flag==1:
    print 'Yes'
else:
    print 'No'
