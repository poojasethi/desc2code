t=input()
n=map(int,raw_input().split())
flag=0
d1=t-n[0]
for i in xrange(1,t):
    if(i%2==1):
        if((n[i]+t-d1)%t!=i):
            print "NO"
            flag=1
            break
    else:
        if((n[i]+d1)%t!=i):
            print "NO"
            flag=1
            break
if(flag==0):
    print "YES"
