n = int(input())
x = map(int, raw_input().split())

def solve():
    d = (n - x[0]) % n
    for i in xrange(1, n):
        if (i & 1):
            if (x[i] - d) % n != i:
                print "No"
                return
        else:
            if (x[i] + d) % n != i:
                print "No"
                return
    print "Yes"

solve()
