n = int(raw_input())

a = [int(r) for r in raw_input().split()]

while a[0] != 0:
    clock = True
    for i in range(n):
        if clock:
            a[i] = (a[i] + 1) % n
        else:
            a[i] = (a[i] - 1) % n
        clock = not clock
for i in range(n):
    if a[i] != i:
        print 'No'
        break
else:
    print 'Yes'
