from sys import exit

N = input()
V = map(int, raw_input().split())

Target = [i for i in xrange(N)]

for i in xrange(N + 1):

	for j in xrange(N):
		if j % 2 == 0:
			V[j] += 1;
		else:
			V[j] -= 1;

		if V[j] < 0:
			V[j] = N - 1;
		if V[j] >= N:
			V[j] = 0;

	if V == Target:
		print "Yes"
		exit(0)

print "No"
