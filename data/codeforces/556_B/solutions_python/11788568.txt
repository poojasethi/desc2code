n = input()
act = [int(i) for i in raw_input().split()]
req = range(n)
xy=0
for i in xrange(n+1000):
    if(act==req):
        print "Yes"
        xy=1
        break
    for j in xrange(n):
        if(j%2==0):
            act[j]=(act[j]+1)%n
        else:
            act[j]=(act[j]+n-1)%n
if(xy==0):
    print "No"
