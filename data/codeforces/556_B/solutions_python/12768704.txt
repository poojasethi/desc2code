n = int(raw_input().strip())
settings = map(int, raw_input().strip().split(' '))
diff = settings[0]
for i in xrange(len(settings)):
    if i % 2 == 0:
        if (settings[i] - diff) % n != i:
            print 'No'
            exit()
    else:
        if (settings[i] + diff) % n != i:
            print 'No'
            exit()

print 'Yes'
