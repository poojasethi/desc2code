n = int(raw_input())
a = map(int, raw_input().split())
while a[0]:
    for i in xrange(n):
        if i % 2 == 0:
            a[i] += 1
        else:
            a[i] -= 1
        a[i] %= n

ok = True

for i in xrange(n-1):
    if a[i] + 1 != a[i+1]:
        ok = False
        break

print 'Yes' if ok else 'No'
