n = int(raw_input().strip())

a = map(int, raw_input().strip().split(' '))

first = a[0]

turns = n - a[0]

i = 0
flag = 0
while(i < n):
  if(i%2==0):
    a[i] = (a[i] + turns) % n
  else:
    a[i] = (a[i] - turns) % n
  
  if(a[i] != i):
    print 'No'
    flag = 1
    break
  
  i += 1

if(flag != 1):
  print 'Yes'
