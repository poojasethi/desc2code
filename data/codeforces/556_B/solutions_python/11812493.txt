n=input()
m = raw_input()
a = map(int, m.strip().split(' '))
#print a
for i in range(0, n, 2):
    a[i] = (n + a[i] - i) % n
for i in range(1, n, 2):
    a[i] = (n - a[i] + i) % n

s = False
for i in range(1, n):
    if a[i] != a[0]:
        s = True
if s:
    print 'No'
else:
    print 'Yes'
