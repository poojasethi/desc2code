ans=0
b=0

n=int(raw_input())
ar=map(int, raw_input().split())

for i in range(n):
    
    b=1

    for j in range(n):
        if(ar[j]!=j):
            b=0
        if(j%2):
            ar[j]=(ar[j]+1)%n
        else:
            ar[j]=(ar[j]+n-1)%n
    ans=ans|b

if ans:
    print 'Yes'
else:
    print 'No'
