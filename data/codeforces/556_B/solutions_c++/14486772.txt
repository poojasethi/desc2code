#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n, c=1;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int i=0;i<n;i++)
	{
		int j=( n+a[i]+c*(n-a[0]) );
	    if(j % n != i)
		{
            cout<<"NO";
            return 0; 
        }
        c*=-1;
    }
    cout<<"YES";
    return 0;
}
