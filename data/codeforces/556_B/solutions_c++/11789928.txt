#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	vector<int> ar(n);
	int i;
	for(i=0;i<n;i++)
		cin>>ar[i];
	int diff=ar[0];
	for(i=0;i<n;i++)
	{
		if(i%2==0)
			ar[i]=(ar[i]-diff+n)%n;
		else
			ar[i]=(ar[i]+diff+n)%n;
		if(ar[i]!=i)
			break;
	}
	if(i==n)
		cout<<"Yes";
	else
		cout<<"No";
	return 0;
}
