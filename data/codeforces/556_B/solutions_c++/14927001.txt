#include<iostream>
using namespace std;
int main()
{
	int j,i,n,a[1005],f;
	while(cin>>n)
	{
		for(i=0;i<n;i++)
			cin>>a[i];
		for(i=0;i<n;i++)
		{
			for(f=1,j=0;j<n;j++)
			{
				j%2?(a[j]>0?a[j]--:a[j]=n-1):(a[j]<n-1?a[j]++:a[j]=0);
				if(a[j]!=j)
					f=0;
			}
			if(f)
				break;
		}
		f?cout<<"Yes\n":cout<<"No\n";
	}
}
