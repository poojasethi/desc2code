#include <iostream>
using namespace std;
int main()
{
	ios::sync_with_stdio(0);
	int n;
	cin>>n;
	int k;
	cin>>k;
	int add=n-k;
	int i=1;
	for(i=1;i<n;i++)
	{
		cin>>k;
		if(i&1)
		{
			if((k-add+n)%n!=i)
				break;
		}
		else
		{
			if((k+add)%n!=i)
				break;
		}
	}
	if(i!=n)
		cout<<"NO\n";
	else cout<<"YES\n";
}
