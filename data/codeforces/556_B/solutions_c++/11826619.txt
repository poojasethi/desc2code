#include<iostream>
#include<vector>
using namespace std;
int main()
{
	int n,i,j,c,diff;
	cin>>n;
	vector<int> arr;
	cin>>c;
	diff = (n-c)%n;
	bool flag = true;
	for(i=1;i<n;i++)
	{
		cin>>c;
		if(i%2)
			c = (c-diff+n)%n;
		else
			c = (c+diff)%n;
		if(c!=i)
			flag=false;
	}
	if(!flag)
		cout<<"No"<<endl;
	else
		cout<<"Yes"<<endl;
	return 0;
}
