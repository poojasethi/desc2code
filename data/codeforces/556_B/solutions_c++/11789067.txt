#include <iostream>

using namespace std;

int main() {
  int n;
  cin >> n;
  int flag = 1;
  int num = 0;
  for (int i=0; i<n; ++i) {
    int val;
    cin >> val;
    if (i == 0) {
      num = (n-val);
    } else {
      if (((n + val + num) % n) != i) flag = 0;
    }
    num = -num;
  }
  cout << (flag ? "Yes" : "No") << endl;
  return 0;
}
