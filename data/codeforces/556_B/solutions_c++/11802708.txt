#include<iostream>
using namespace std;
int main()
{
    int n,x,cha,d=1;
    bool yes=1;
    cin>>n>>x;
    cha=n-x;
    for(int i=1;i<n;++i)
    {
        d=d*-1;
        cin>>x;
        //cout<<x<<"   "<<cha<<"   "<<d<<endl;
        if((i-x-d*cha)%n!=0){yes=0;}
    }
    if(yes)cout<<"Yes";
    else cout<<"No";

}
