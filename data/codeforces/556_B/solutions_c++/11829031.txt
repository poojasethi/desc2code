#include<iostream>


int main () {
    int n, a, x;
    std::cin >> n >> a;
    x=n-a;
    for (int i=1; i<n; i++) {
        std::cin >> a;
        if (i%2 == 0 && (a+x)%n != i || i%2 == 1 && (a-x+n)%n != i) {
            std::cout << "no";
            return 0;
        }
    }
    std::cout << "yes";
    return 0;
}
