#include<iostream>
using namespace std;
int main()
{
int x,n,count=1,i;
cin>>n;
int a[n];
for(i=0;i<n;i++)
cin>>a[i];
x=n-a[0];
for(i=1;i<n;i+=2)
if((a[i]-x==i) || (n+a[i]-x==i))
count++;
for(i=2;i<n;i+=2)
if((a[i]+x==i) || (a[i]+x-n==i))
count++;
if(count==n)
cout<<"Yes";
else
cout<<"No";
return 0;
}
