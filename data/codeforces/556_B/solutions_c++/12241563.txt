#include <bits/stdc++.h>

using namespace std;

int main(){
    int n, flag = 1, delta, x, tmp;
    scanf("%d", &n);
    scanf("%d", &delta);
    for (int i = 1; i < n; ++i){
        scanf("%d", &x);
        flag &= (((i & 1 ? 1 : -1) * delta + x + n) % n == i);
    }
    puts(flag ? "Yes" : "No");
}
