#include <cstdio>
int a,i,j;
bool f;
int s[2000];
int main(){
	scanf("%d\n",&j);
	for(;i<j;++i) scanf("%d",s+i);
	a=0-s[0];
	s[0]=0;
	for(i=1;i<j;++i) s[i]+=(a=-a);
	for(i=1;i<j;++i) s[i]=(s[i]%j+j)%j;
	f=1;
	for(i=1;i<j;++i) if(s[i]!=s[i-1]+1){
		f=0;
		break;
	}
	if(f) printf("Yes\n"); else printf("No");
	return 0;
}
