#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,d,x;
    ios::sync_with_stdio();
    cin.tie(0);
    cin>>n>>d;
    for(int i=1; i<n; i++)
    {
        cin>>x;
        if((x+d*(i&1?1:-1)+n)%n!=i)
        {
            cout<<"NO"<<endl;
            return 0;
        }
    }
    cout<<"Yes"<<endl;
    return 0;
}
