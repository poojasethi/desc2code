#include<stdio.h>
int n,i,a[2000];
int main(){
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	int k=n-a[0];
	for(i=0;i<n;i++){
		if(i%2==0){
			a[i]+=k;
			a[i]%=n;
		}
		else{
			a[i]-=k;
			a[i]%=n;
			a[i]+=n;
			a[i]%=n;
		}
	}
	for(i=1;i<n;i++)
		if(a[i]!=i){printf("NO");return 0;}
	printf("YES");
	return 0;
}
