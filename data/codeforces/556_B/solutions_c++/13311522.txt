#include <cstdio>
int n;
int a, b;
int main (){
    scanf("%d", &n);
    scanf("%d", &a);
    for (int i = 1; i < n; i++){
        scanf("%d", &b);
        (i % 2) ? (b = (b + a) % n) : (b = (b + n - a) % n);
        if (b != i){
            printf("No\n");
            return 0;
        }
    }
    printf("Yes\n");
    return 0;
}
