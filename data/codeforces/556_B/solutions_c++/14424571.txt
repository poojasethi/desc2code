#include<bits/stdc++.h>
using namespace std;
int main() {
int n, k, a, i;
    scanf("%d %d", &n, &a);
    k = (n - a) % n;
    for (i = 1; i < n; i++) {
        scanf("%d", &a);
        i % 2 ? a -= k : a += k;
        a = (a + n) % n;
        if (a != i) {cout<<"No"<<endl; return 0;}
    }
    cout<<"Yes"<<endl;
}
