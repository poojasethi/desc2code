#include<iostream>
using namespace std;
int main()
{
	int n,t,tmp,step,i=0;
	bool flag=false;
	cin>>n;cin>>t;
	tmp=(t-i+n)%n;
	for(int i=1;i<n;i++)
	{
		cin>>t;
		if(flag) step=(t-i+n)%n;
		else step=(i-t+n)%n;
		if(step!=tmp) {cout<<"No"<<endl;return 0;}
		flag=!flag;
	}	
	cout<<"Yes"<<endl;
	return 0;
}
