#include <cstdio>
using namespace std;

int N, j, t, r;

int main(void){
	scanf("%d%d", &N, &j);
	j = (N - j) % N;
	for (int i = 1; i < N; i++){
		scanf("%d", &t);
		if (i % 2 == 0) r = (N + i - t) % N;
		else r = (N + t - i) % N;
		if (r != j) {
			printf("No");
			return 0;
		}
    }
    printf("Yes");
	return 0;
}
