#include <cstdio>

int main(){
  int N, arr[1000], diff = -1, temp;
  scanf("%d", &N);
  for(int i=0; i<N; i++){
    scanf("%d", &arr[i]);
    temp = (i&1)? (N-i+arr[i])%N: (i+N-arr[i])%N;
    if(diff==-1) diff = temp;
    else if(diff!=temp){
      printf("No\n"); return 0;
    }
  }
  printf("Yes\n");
  return 0;
}
