#include<cstdio>

int N;
int A[1005];

inline int f(int k){
	return k&1?1:-1;
}
inline int turn(int v,int i,int s){
	return (v+f(i)*s+N)%N;
}
int main(){
	scanf("%d",&N);
	for(int i=1;i<=N;i++)
		scanf("%d",A+i);

	for(int i=1;i<=N;i++){
		if(turn(A[i],i,N-A[1])!=i-1){
			printf("No");
			return 0;
		}
	}
	printf("Yes");
	return 0;
}
