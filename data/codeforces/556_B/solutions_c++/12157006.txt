#include <iostream>
#include <cmath>
#include <iomanip>
#include <vector>

using namespace std;

int main(){
	int n;
	int x;
	cin >> n;
	cin >> x;
	int k = x%n;
	for(int i = 1; i < n; i++){
		cin >> x;
		if(i%2)	x += k;
		else x -= k;
		if(x != i && x != i-n && x != i+n){
			cout << "No" << endl;
			return 0;	
		} 
	}
	cout << "Yes" << endl;
}
