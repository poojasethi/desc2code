#include<iostream>
#include<cstdio>
#include<algorithm>

using namespace std;
int n,a[1005];
int main(){
	cin>>n;
	for(int i=0;i<n;++i) cin>>a[i];
	bool fail=0;
	for(int i=1;i<n;++i){
		a[i]+=((2*(i&1)-1)*a[0]+n);
		if(a[i]%n!=i){
			fail=1;
			break;
		}	
	}
	if(fail) cout<<"No"<<endl;
	else cout<<"Yes"<<endl;

}
