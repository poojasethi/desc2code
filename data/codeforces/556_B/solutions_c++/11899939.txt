#include <cstdio>
using namespace std;

int n, tmp, cur;

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &tmp);
		tmp = (i & 1) ? (tmp + n - i) % n : (i + n - tmp) % n;
		if (i == 0) cur = tmp;
		else {
			if (cur == tmp) continue;
			puts("No");
			return 0;
		}
	}
	puts("Yes");
	return 0;
}
