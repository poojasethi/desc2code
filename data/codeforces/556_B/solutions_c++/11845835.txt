#include<iostream>
using namespace std;
int main()
{
    int n,a;
    int i,j,k,l,c=0;
    cin>>n>>a;
    k=(n-a)%n;
    for(i=1;i<n;i++)
    {
        cin>>a;
        (i%2)?a-=k:a+=k;
        a=(a+n)%n;
        if(a!=i)
        {
            cout<<"NO"<<endl;
            return 0;
        }
    }
    cout<<"YES"<<endl;
}
