#include <bits/stdc++.h>

using namespace std;

int a[1000 + 10];

int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		if(i%2==0)
		{
			a[i]=i-a[i];
			while(a[i]<0)
				a[i]+=n;
		}
		else
		{
			if(a[i]>=i)
				a[i]-=i;
			else
				a[i]+=n-i;
		}
		if(i&&a[i]!=a[i-1])
		{
			cout<<"No"<<endl;
			return 0;
		}
	}
	cout<<"Yes"<<endl;
}
