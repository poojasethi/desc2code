#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,f,a,x;
    cin >> n >> f;
    a=n-f;
    for(int i=1;i<n;i++)
        {   cin >> x;                 
            if(i%2==0 && (x+a)%n!=i or i%2==1 && (x-a+n)%n!=i)
            {return cout << "No",0;}}
       cout << "Yes";
       return 0;
}
