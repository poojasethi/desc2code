#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	cin >> n;

	map<string, string> m;
	map<string, string> old;

	for (int i = 0; i < n; i++)
	{
		string a, b;
		cin >> a >> b;

		if (old.count(a))
		{
			m[old[a]] = b;
			old[b] = old[a];
		}
		else
		{
			m[a] = b;
			old[b] = a;
		}
	}

	cout << m.size()<<endl;

	for (map<string, string>::iterator it = m.begin(); it != m.end(); it++)
	{
		cout << it->first << " " << it->second << endl;
	}
}
