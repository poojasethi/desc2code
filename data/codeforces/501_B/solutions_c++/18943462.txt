#include<iostream>
#include<string>
#include<map>
using namespace std;

int main()
{
	map<string, string>mp;
	int n;
	cin >> n;
	while (n--)
	{
		string n1, n2;
		cin >> n1 >> n2;
		if (mp.find(n1) != mp.end())
		{
			mp[n2] = mp[n1];
			mp.erase(n1);
		}
		else mp[n2] = n1;
	}
	cout << mp.size() << endl;
	map<string, string>::iterator it;
	for (it = mp.begin(); it != mp.end(); it++)
		cout << it->second << " " << it->first << endl;
}
