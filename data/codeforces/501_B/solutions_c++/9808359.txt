#include<iostream>
#include<string.h>
using namespace std;

typedef char str[25];
str a[1000][2],from,to;
int n,j,k=0;

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>from>>to;
		for(j=0;j<k;j++)
			if(strcmp(from,a[j][1])==0) break;
		if(j==k)
		{
			strcpy(a[k][0],from);
			strcpy(a[k][1],to);
			k++;
		}
		else strcpy(a[j][1],to);
	}
	cout<<k<<endl;
	for(int i=0;i<k;i++) cout<<a[i][0]<<" "<<a[i][1]<<endl;
	return 0;
}
