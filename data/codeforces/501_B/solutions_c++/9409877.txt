#include<bits/stdc++.h>
using namespace std;

int main(){
map<string,int> mp;
map<int,string> ma;
vector<string> mx;
int t,cnt=0;
cin>>t;
string s,d;
while(t--){
cin>>s>>d;
if(mp.count(s)==0){
mp[s]=cnt++;
mx.push_back(s);
}
mp[d]=mp[s];
ma[mp[s]]=d;
}
cout<<cnt<<endl;
for(int i=0;i<cnt;i++)
cout<<mx[i]<<" "<<ma[i]<<endl;

return 0;
}
