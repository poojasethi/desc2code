using namespace std;

int main(){

map<string,string>mp;

int q,i;

cin>>q;

string nw,old;
map<string,string>::iterator it;

{

cin>>old>>nw;

it=mp.find(old);

if(it==mp.end()){

mp[nw]=old;

}
else {

mp[nw]=it->second;

}

}

int ans=mp.size();
