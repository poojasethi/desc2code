#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
	int q,f;
	string a,b;
	vector< pair<string, string> > v;
	cin>>q;
	while(q--)
	{
		cin>>a>>b;
		f=0;
		for(int i=0;i<v.size();i++)
			if(v[i].second.compare(a)==0)
			{
				f=1;
				v[i].second=b;
				break;
			}
		if(!f)
			v.push_back(make_pair(a,b));
	}
	cout<<v.size()<<endl;
	for(int i=0;i<v.size();i++)
		cout<<v[i].first<<" "<<v[i].second<<endl;
	return 0;
}
