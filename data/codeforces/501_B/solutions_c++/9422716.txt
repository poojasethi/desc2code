#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int main()
{
    int q,n,i,j;  string a[1000], b[1000];  cin>>q; n=q;
    for(i=0;i<q;i++) {
        cin>>a[i]>>b[i];}
    for(i=0;i<q;i++){
        for(j=0;j<q;j++){
            if(b[i]==a[j] && a[j]!=""){b[i]=b[j];n--;a[j]=b[j]= "";}}}
    cout<<n<<"\n";
    for(i=0;i<q;i++){
        if(a[i]!= ""){cout<<a[i]<<" "<<b[i]<<"\n";}}
    return 0;
}
