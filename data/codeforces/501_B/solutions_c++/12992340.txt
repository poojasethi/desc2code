#include<bits/stdc++.h>
using namespace std;
map<string,string> m,ans;
int n;
string u,v;
int main() {
	cin >> n;
	for(int i = 1; i <= n; i++) {
		cin >> u >> v;
		if (m.count(u)) {
			m[v] = m[u];
			ans[m[u]] = v;
		} else {
			m[u] = m[v] = u;
			ans[u] = v;
		}
	}
	printf("%d\n",ans.size());
	for(map<string,string>::iterator i = ans.begin(); i != ans.end(); ++i)
		printf("%s %s\n",(i->first).c_str(),(i->second).c_str());
}
