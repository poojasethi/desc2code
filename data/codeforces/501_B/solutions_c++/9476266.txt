#include <bits/stdc++.h>
using namespace std;

int main() {
	int q;
	cin >> q;
	
	string a, b, v;
	
	map<string, string> tr;
	
	while(q--) {
		cin >> a >> b;
		
		if(tr.find(a) == tr.end())
			v = a;
		else
			v = tr[a];
		
		tr.erase(a);
		
		tr[b] = v;
	}
	
	cout << tr.size() << endl;
	map<string, string>::iterator i;
	for(i = tr.begin(); i != tr.end(); i++)
		cout << i->second << " " << i->first << endl;
}
