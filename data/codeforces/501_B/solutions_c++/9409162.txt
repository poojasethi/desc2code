#include<bits/stdc++.h>

using namespace std;


int main()
{
int n;
cin>>n;
string s[n][2];
int flag,pos=0;
for(int i=0;i<n;i++)
{
	string a,b;
	cin>>a>>b;
	flag=0;
	for(int j=0;j<pos;j++)
	{
		if(s[j][1]==a)
			{
				flag=1;
				s[j][1]=b;
				break;
			}
	}
	if(!flag){
		s[pos][0]=a;s[pos][1]=b;
		pos++;
		}
}
cout<<pos<<endl;
for(int i=0;i<pos;i++)
	cout<<s[i][0]<<" "<<s[i][1]<<endl;
return 0;
}
