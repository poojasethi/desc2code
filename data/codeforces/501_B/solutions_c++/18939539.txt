
#include <iostream>
#include <map>
using namespace std;
int main()
{
	int n;
	string a,b;
	map<string,string>mp;
	while(cin>>n)
	{
		for(int i=0;i<n;i++)
		{
			cin>>a>>b;
			if(mp.count(a)==0)
			{
				mp[a]=a;
			}
				mp[b]=mp[a];
				mp.erase(a);
			
		}
		map<string,string>::iterator it;
		cout<<mp.size()<<endl;
		for(it=mp.begin();it!=mp.end();it++)
		{
			cout<<it->second<<" "<<it->first<<endl;
		}
	}
}
