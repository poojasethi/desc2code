#include<bits/stdc++.h>
using namespace std;
string a[1005];
string b[1005];
int main()
{
	int n=0,x=0;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		string s,t;
		cin>>s>>t;
		bool f=false;
		for(int j=0;j<x;j++)
        {
			if(b[j] == s)
			{
				b[j]=t;
				f=true;
				break;
			}
        }
		if(!f)
		{
			a[x]=s;
		   	b[x]=t;
			x++;
		}
	}
	cout<<x<<endl;
	for(int i=0;i<x;i++)
		cout<<a[i]<<" "<<b[i]<<endl;
	return 0;
}
