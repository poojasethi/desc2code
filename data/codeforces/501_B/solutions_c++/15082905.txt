#include <bits/stdc++.h>
using namespace std;
map<string,string> hash;
int main(){
	int n,i;string a,b,c;cin>>n;
	for(i=0;i<n;++i){
		cin>>a>>b;
		if(hash.count(a)){
			c= hash[a];
			hash[b]=c;
			hash.erase(a);
		}
		else
			hash[b]=a;
	}
	map<string,string>::iterator it;
	cout<<hash.size()<<endl;
	for(it=hash.begin();it!=hash.end();++it)
		cout<<it->second<<" "<<it->first<<endl;
}
