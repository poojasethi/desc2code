#include <iostream>
#include <map>
#include <string>
#include <iterator>

using namespace std;
int main()
{
	map<string, string> m;
	string a="ahoh";
	string b;
	int n;
	cin>>n;
	for(int i=0;i<n;i++) {
		cin>>a;cin>>b;
		if(m.count(a)>0) {
			string t=m[a];
			m.erase(a);
			m[b]=t;

		}else
		m[b]=a;
	}

	map<string, string>::iterator it=m.begin();

	cout<<m.size()<<endl;
	for(it=m.begin(); it!=m.end(); it++) {
		cout<<it->second<<" "<<it->first<<endl;
	}

}
