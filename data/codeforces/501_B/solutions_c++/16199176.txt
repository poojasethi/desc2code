#include<iostream>
#include<map>
using namespace std;
int main()
 {
 string ne,old;
 map<string,string>q;
 int n;
 cin>>n;
 while(n--)
  {
   cin>>old>>ne;
   if(q.count(old))
   {
    q[ne]=q[old];
    q.erase(old);
   }
   else
   q[ne]=old;
  }
   cout<<q.size()<<"\n";
   map<string,string>:: iterator t;
  for(t=q.begin();t!=q.end();t++)
   cout<<(*t).second<<" "<<(*t).first<<"\n";
  return 0;
  }
