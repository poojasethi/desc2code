#include<stdio.h>
#include<cstring>
#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
int n;
map<string,string> q;
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	string a,b;
	scanf("%d",&n);
	while(n--)
	{
		cin>>a>>b;
		if(q.count(a)) q[b]=q[a],q.erase(a);
		else q[b]=a;
	}
	cout<<q.size()<<endl;
	map<string,string>::iterator t;
	for(t=q.begin();t!=q.end();t++)
		cout<<(*t).second<<' '<<(*t).first<<endl;
	return 0;
}
