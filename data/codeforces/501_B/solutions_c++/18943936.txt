#include<iostream>
#include<map>
using namespace std;
int main(){
	string a,b;
	int n;
	while(cin>>n){
		map<string,string>mp;
		for(int i=0;i<n;i++){
			cin>>a>>b; 
			if(!mp.count(a)){
				mp[b]=a;
			}
			else {
				mp[b]=mp[a];
				mp.erase(a);
			}
		}
		cout<<mp.size()<<endl;
		for(map<string,string>::iterator it=mp.begin();it!=mp.end();++it){
		cout << it->second <<" "<< it->first <<endl;  
		}
	}
	return 0;
}
