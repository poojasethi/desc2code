#include<iostream>
using namespace std;
string ans[1000][2];
int ansf=0;
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		string a,b;
		cin>>a>>b;
		for(int j=0;j<ansf;j++)
		{
			if(ans[j][1]==a)
			{
				ans[j][1]=b;
				goto hell;
			}
		}
		ans[ansf][0]=a;
		ans[ansf][1]=b;
		ansf++;
		hell:;
	}
	cout<<ansf<<endl;
	for(int i=0;i<ansf;i++)
	{
		cout<<ans[i][0]<<" "<<ans[i][1]<<endl;
	}
}
