#include <bits/stdc++.h>
using namespace std;
map <string,string> cha;
map<string,string>::iterator it;
int main()
{
	ios_base::sync_with_stdio(0);
	int n;
	cin>>n;
	string a,b;
	for (int i=0;i<n;i++)
	{
		cin >> a >> b;
		it = cha.find(a);
		if (it==cha.end()) cha[b] = a;
		else
		{
			cha[b] = it->second;
			cha.erase(it);
		}
	}
	cout<<cha.size()<<"\n";
	for (it = cha.begin();it!=cha.end();it++) cout << it->second <<" "<< it->first <<"\n";
}
