#include <bits/stdc++.h>
using namespace std;
int main()
{
	int q;
	cin >> q;
	string s1, s2;
	map<string, string> m1, m2;
	while (q--){
		cin >> s1 >> s2;
		if (!m2.count(s1)){
			m1[s1] = s2;
			m2[s2] = s1;
		}else{
			m1[m2[s1]] = s2;
			m2[s2] = m2[s1];
		}
	}
	cout << m1.size() << endl;
	for (map<string, string>::iterator it = m1.begin();it != m1.end();it++)
		cout << it->first << " " << it->second << endl;
	return 0;
}
