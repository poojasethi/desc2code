#include<bits/stdc++.h>
using namespace std ;
#define f first
#define s second
map<string,string> m;

int main(){
	int n,count=0;
	cin>>n;
	for(int i=0;i<n;i++){
		string s1,s2;
		cin>>s1>>s2;
		if(m.find(s1)!=m.end()){
			m[s2]=m[s1];
			m.erase(m.find(s1));
		}
		else m[s2]=s1;
	}
	map<string,string>::iterator it;
	it=m.begin();
	cout<<m.size()<<endl;
	while(it!=m.end()){
		cout<<it->s<<" "<<it->f<<endl;
		it++;
	}
	return 0;
}
