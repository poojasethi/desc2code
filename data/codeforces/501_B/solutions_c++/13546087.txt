#include<bits/stdc++.h>
using namespace std;

int main(){
	ios::sync_with_stdio(0);
	string a,b;
	int n;
	map<string,string>NAMA;
	map<string,string>::iterator it;
	cin>>n;
	while(n--){
		cin>>a>>b;
		it=NAMA.find(a);
		if(it!=NAMA.end()){
			NAMA[b]=it->second;
			NAMA.erase(it);
		}
		else{
			NAMA[b]=a;
		}
	}
	cout<<NAMA.size()<<endl;
	for(it=NAMA.begin();it!=NAMA.end();++it){
		cout<<it->second<<' '<<it->first<<endl;
	}
}
