#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll n,ans;
string s,t,res[1111];
map<string,string>v,vv;
int main()
{
    cin>>n;
    for(ll i=1;i<=n;i++)
    {
        cin>>s>>t;
        if(!v.count(s))
            v[s]=s,res[++ans]=s;
        v[t]=v[s];
        vv[v[t]]=t;
    }
    cout<<ans<<endl;
    for(ll i=1;i<=ans;i++)
        cout<<res[i]<<" "<<vv[res[i]]<<endl;
}
