#include<iostream>
using namespace std;
int main()
{
	int n,i,j,count=0;
	string s1[1001],s2[1001];
	cin>>n;
	for(i=0;i<n;i++) cin>>s1[i]>>s2[i];
	for(i=0;i<n;i++)
		for(j=i+1;j<n;j++)
			if(s2[i]==s1[j])
			{
				s2[i]=s2[j];
				s1[j].clear();
				s2[j].clear();
			}
	for(i=0;i<n;i++)
		if(s1[i]!="")
			count++;
	cout<<count<<endl;
	for(i=0;i<n;i++)
	{
		if(s1[i]!="")
		cout<<s1[i]<<" "<<s2[i]<<endl;
	}
	return 0;
}
