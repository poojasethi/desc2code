#include<bits/stdc++.h>
using namespace std;

map<string,int> M1;
map<int,string> M2;
vector<string> V;
int n,cont;
string s1,s2;

int main(){
	cin>>n;
	cont = 0;
	
	for(int i = 0; i < n; i++){
		cin>>s1>>s2;
		
		if(M1.find(s1) == M1.end()){
			M1[s1] = cont++;
			V.push_back(s1);
		}
		
		M1[s2]=M1[s1];
		M2[M1[s1]]=s2;
	}
	cout<<cont<<endl;
	for(int i = 0; i < cont; i++) cout<<V[i]<<" "<<M2[i]<<endl;

	return 0;
}
