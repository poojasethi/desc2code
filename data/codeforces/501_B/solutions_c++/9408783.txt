#include <iostream>
#include <string>
#include <map>

using namespace std;

int q, n;
string a, b, v[1000];
map<string, int> m;

int main ()
{
	cin >> q;
	for (int i = 0; i < q; ++i) {
		cin >> a >> b;
		if (m.count(a)) m[b] = m[a], m.erase(a);
		else v[n] = a, m[b] = n++;
	}
	cout << n << '\n';
	for (map<string, int>::iterator i = m.begin(); i != m.end(); ++i) cout << v[i->second] << ' ' << i->first << '\n';
	return 0;
}
