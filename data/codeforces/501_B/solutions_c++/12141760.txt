#include<iostream>
#include<string>
#include<map>
using namespace std;
map<string,string> mp;
map<string,string>::iterator it;
int main(){
	int q;
	cin>>q;
	while(q--){
		string o,n;
		cin>>o>>n;
		if(mp.find(o)==mp.end()) mp[n]=o;
		else{
			mp[n]=mp[o];
			mp.erase(o);
		}
	}
	cout<<mp.size()<<endl;
	for(it=mp.begin();it!=mp.end();++it)
		cout<<it->second<<' '<<it->first<<endl;
	return 0;
}
