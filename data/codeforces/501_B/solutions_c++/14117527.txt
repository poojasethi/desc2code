#include<iostream>
#include<map>
#include<cstring>
using namespace std;
map<string,string>mp;
int main()
{
	int n;
	cin >> n;
	string s1,s2;
	for(int i=0;i<n;i++)
	{
		cin >> s1 >> s2;
		if(!mp.count(s1))
		{
			mp[s1] = s1;
		}
		mp[s2] = mp[s1];
		mp.erase(s1);
	}
	cout<<mp.size()<<endl;
	map<string,string>::iterator ite;
	for(ite = mp.begin();ite!=mp.end();ite++)
	{
		cout<<(*ite).second<<" "<<(*ite).first<<endl;
	}
	return 0;
}
