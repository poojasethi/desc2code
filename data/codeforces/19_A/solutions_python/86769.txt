#!/usr/bin/python
import re, sys
n = int (sys.stdin.readline ())
c = [sys.stdin.readline ().strip () for i in range (n)]
m = (n * (n - 1)) >> 1
r = dict (zip (c, [0] * n))
s = dict (zip (c, [0] * n))
t = dict (zip (c, [0] * n))

for i in range (m):
	q = sys.stdin.readline ().strip ()
	a, b, x, y = re.split ('[-\ :]', q)
	x, y = int (x), int (y)
	if x > y:
		r[a] += 3
	elif x < y:
		r[b] += 3
	else:
		r[a] += 1
		r[b] += 1
	s[a] += x - y
	s[b] += y - x
	t[a] += x
	t[b] += y

for i in range (n - 1):
	for j in range (i + 1, n):
		if r[c[i]] < r[c[j]] or (r[c[i]] == r[c[j]] and \
		 (s[c[i]] < s[c[j]] or (s[c[i]] == s[c[j]] and \
		 t[c[i]] < t[c[j]]))):
			c[i], c[j] = c[j], c[i]
k = n >> 1
b = c[:k]
b.sort ()
for i in range (k):
	print b[i]
