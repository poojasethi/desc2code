def readint(): return int(raw_input())
def readfloat(): return float(raw_input())
def readarray(N, foo):
        res = []
        for _ in xrange(N):
                res.append(foo())
        return res
def readlinearray(foo): return map(foo, raw_input().split())

from collections import defaultdict

if __name__ == '__main__':
    n = readint()
    teams = []
    for _ in xrange(n):
        teams.append(raw_input())
        
    m = defaultdict(lambda: (0, 0, 0))
    d = n * (n - 1) / 2
    for _ in xrange(d):
        line = raw_input()
        teams, goals = line.split()
        team1, team2 = teams.split('-')
        num1, num2 = map(int, goals.split(':'))
        
        p1 = p2 = 0
        if num1 > num2:
            p1 = 3
        elif num2 > num1:
            p2 = 3
        else:
            p1 = p2 = 1
        
        s1 = num1
        s2 = num2
        
        s_m1 = s1 - s2
        s_m2 = s2 - s1
        
        _p1, _s_m1, _s1 = m[team1]
        _p2, _s_m2, _s2 = m[team2]
        
        t1 = (p1 + _p1, s_m1 + _s_m1, s1 + _s1)
        t2 = (p2 + _p2, s_m2 + _s_m2, s2 + _s2)
        
        m[team1] = t1
        m[team2] = t2
    
    teams = [t + (team,) for team, t in m.iteritems()]
    teams = sorted(teams, reverse=True)[:n/2]
    teams = sorted([t for (_, __, ___, t) in teams])
    print '\n'.join(teams)
