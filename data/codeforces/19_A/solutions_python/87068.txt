#!/usr/bin/python2.5

import sys, os, string, re

in_file = sys.stdin
out_file = sys.stdout

n = int(in_file.readline())
team_names = []
team_data = {}
for i in range(n):
	name = in_file.readline().strip()
	team_names.append(name)
	team_data[name] = {'points': 0, 'goals_scored': 0, 'goals_against': 0}

for i in range(n*(n-1)/2):
	[name_str, score_str] = in_file.readline().split()
	names = name_str.split('-')
	goals = score_str.split(':')
	goals = [int(goals[0]), int(goals[1])]
	if goals[0] == goals[1]:
		for name in names:
			team_data[name]['points'] += 1
	elif goals[0] > goals[1]:
		team_data[names[0]]['points'] += 3
	else:
		team_data[names[1]]['points'] += 3
	team_data[names[0]]['goals_scored'] += goals[0]
	team_data[names[0]]['goals_against'] += goals[1]
	team_data[names[1]]['goals_scored'] += goals[1]
	team_data[names[1]]['goals_against'] += goals[0]

team_keys = []
for name in team_names:
	data = team_data[name]
	team_keys.append([data['points'],
					 data['goals_scored']-data['goals_against'],
					 data['goals_scored'],
					 name])
team_keys.sort()
team_keys.reverse()

advancers = []
for i in range(n/2):
	advancers.append(team_keys[i][-1])
advancers.sort()
for item in advancers:
	out_file.write(item+'\n')
