n=int(raw_input())

m={}
for i in range(n):
	m[raw_input()]=[0,0,0]
for i in range(n*(n-1)/2):
	match,score=raw_input().split()
	a,b=match.split("-")
	c,d=map(int,score.split(":"))
	if c==d:
		m[a][0]+=1
		m[b][0]+=1
	elif c>d:
		m[a][0]+=3
	else:
		m[b][0]+=3
	m[a][1]+=c-d
	m[b][1]+=d-c
	m[a][2]+=c
	m[b][2]+=d
m=[[m[i],i] for i in m]
m.sort()
m.reverse()
m=m[:n/2]
m=[i[1] for i in m]
m.sort()
for i in m:
	print i
