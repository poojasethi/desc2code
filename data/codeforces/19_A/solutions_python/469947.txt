#!/usr/bin/env python
# -*- coding: utf-8 -*-

import sys

def main():
    n = int(raw_input())
    c = dict()
    for _ in xrange(n):
        c[raw_input()] = [0, 0, 0]
    for l in sys.stdin.readlines():
        l = l.split()
        l = l[0].split('-') + map(int, l[1].split(':'))
        if l[2] > l[3]:
            c[l[0]][0] += 3
        elif l[2] < l[3]:
            c[l[1]][0] += 3
        else:
            c[l[0]][0] += 1
            c[l[1]][0] += 1
        c[l[0]][1] += l[2] - l[3]
        c[l[1]][1] += l[3] - l[2]
        c[l[0]][2] += l[2]
        c[l[1]][2] += l[3]
    print "\n".join(sorted([key for _, key in sorted([value, key] for key, value in c.items())[::-1][:(n+1)/2]]))
    return 0

if __name__ == '__main__':
    main()
