n=input()

N=dict([(raw_input(),[0,0,0]) for i in range(n)])

for i in range(n*(n-1)/2):
    a,b = raw_input().split()
    x,y=a.split('-')
    r,p=map(int,b.split(':'))
    N[x][0] += (r>p)*3+(r==p)*1
    N[y][0] += (p>r)*3+(r==p)*1    
    N[x][1] += r
    N[y][1] += p
    N[x][2] += p       
    N[y][2] += r

print '\n'.join(sorted(map(lambda (k,v): k,sorted(N.items(), key=lambda (k,v): (v[0], v[1]-v[2],v[1]))[n/2:])))
