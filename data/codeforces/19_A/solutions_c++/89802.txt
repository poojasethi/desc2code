#include <iostream>
#include <string>
#include <vector>
#include <map>
#include <algorithm>
#include <cstdio>
using namespace std;

int N;
string name[60];

int main()
{
	while (1 == scanf("%d ", &N)) {
		for (int i = 0; i < N; ++i) cin >> name[i];

		map<string, int> ptr, dif, goa;

		for (int i = 0; i < N * (N - 1) / 2; ++i) {
			char ta[256], tb[256];
			int ga, gb;
			scanf(" %[^-]-%s %d:%d ", ta, tb, &ga, &gb);

			ptr[string(ta)] += ga > gb ? 3 : ga == gb ? 1 : 0;
			ptr[string(tb)] += gb > ga ? 3 : gb == ga ? 1 : 0;
			dif[string(ta)] += ga - gb;
			dif[string(tb)] += gb - ga;
			goa[string(ta)] += ga;
			goa[string(tb)] += gb;
		}

		vector<pair<pair<int, pair<int, int> >, string> > v;
		for (int i = 0; i < N; ++i) {
			const string &t = name[i];
			v.push_back(
				make_pair(make_pair(-ptr[t], make_pair(-dif[t], -goa[t])), t)
			);
		}
		sort(v.begin(), v.end());

		vector<string> res;
		for (int i = 0; i < N / 2; ++i) res.push_back(v[i].second);
		sort(res.begin(), res.end());
		for (int i = 0; i < res.size(); ++i) cout << res[i] << endl;
	}
	return 0;
}
