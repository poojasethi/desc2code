#include<iostream>
#include<cstdio>
#include<string>
#include<algorithm>
using namespace std;
const int V=55;
struct team{
	string nam;
	int pnt,sco,mis;
};
int n;
team t[V];
void add(const string &s,int a,int b)
{
	for(int i=0;i<n;i++)
	if(t[i].nam==s){
		if(a>b)
		t[i].pnt+=3;
		else if(a==b)
		t[i].pnt+=1;
		t[i].sco+=a;
		t[i].mis+=b;
		break;
	}
}
bool less1(const team &a,const team &b)
{
	if(a.pnt!=b.pnt)
	return a.pnt>b.pnt;
	else if(a.sco-a.mis!=b.sco-b.mis)
	return a.sco-a.mis>b.sco-b.mis;
	else return a.sco>b.sco;
}
bool less2(const team &a,const team &b)
{
	return a.nam<b.nam;
}
int main()
{
	string a,b;
	int c,d;
	cin>>n;
	for(int i=0;i<n;i++)
	cin>>t[i].nam;
	cin.ignore();
	for(int i=0;i<n*(n-1)/2;i++){
		getline(cin,a,'-');
		getline(cin,b,' ');
		scanf("%d:%d",&c,&d);
		cin.ignore();
		add(a,c,d);
		add(b,d,c);
	}
	sort(t,t+n,less1);
	sort(t,t+n/2,less2);
	for(int i=0;i<n/2;i++)
	cout<<t[i].nam<<endl;
	return 0;
}
