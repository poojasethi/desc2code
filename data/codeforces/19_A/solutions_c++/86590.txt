map <string, pair<int,pair <int,int> > > res;


bool myless(string a,string b)
{
	return res[a]<res[b];
}

int main()
{
	int n;
	cin >> n;
	string name[50];
	for (int i=0;i<n;i++)cin >> name[i];
	for (int i=0;i<n*(n-1)/2;i++){
		string s;
		cin >> s;
		int x,y;
		scanf("%d:%d",&x,&y);
		string t1="",t2="";
		bool q=false;
		for(int i=0;i<(int)s.size();i++)if (s[i]!='-'){
			if (q)t2+=s[i];else t1+=s[i];
		} else q=true;
		if (x>y)res[t1].first+=3;else
		if (x<y)res[t2].first+=3;else{
			res[t1].first++;
			res[t2].first++;
		}
		res[t1].second.first+=x-y;
		res[t1].second.second+=x;
		res[t2].second.first+=y-x;
		res[t2].second.second+=y;
	}
	sort(name,name+n,myless);
	reverse(name,name+n);
	sort(name,name+n/2);
	for (int i=0;i<n/2;i++)cout << name[i] << endl;
	return 0;
}
