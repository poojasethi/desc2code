#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
const int MN=300;
string s[MN];
map<string ,int> m;
int pt[MN],taf[MN],gol[MN],a[MN];
int f(int a,int b){
	if(pt[a]!=pt[b]){
		return pt[a]>pt[b];
	}else{
		if(taf[a]!=taf[b]){
			return taf[a]>taf[b];
		}else{
			return gol[a]>gol[b];
		}
	}
}
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>s[i];
		m[s[i]]=i;
	}
	for(int i=0;i<(n*(n-1))/2;i++){
		string st;
		cin>>st;
		string nm[2];
		int v=0;
		for(int i=0;i<st.size();i++){
			if(st[i]=='-'){
				v=1;
				continue;
			}else{
				nm[v]+=st[i];
			}
		}
		int x,y;
		char c;
		cin>>x>>c>>y;
		if(x==y){
			pt[m[nm[0]]]++;
			pt[m[nm[1]]]++;
		}else{
			if(x>y){
				pt[m[nm[0]]]+=3;
			}else{
				pt[m[nm[1]]]+=3;
			}
		}
		taf[m[nm[0]]]+=x-y;
		taf[m[nm[1]]]+=y-x;
		gol[m[nm[0]]]+=x;
		gol[m[nm[1]]]+=y;
	}
	for(int i=0;i<n;i++){
		a[i]=i;
	}
	sort(a,a+n,f);
	string s2[MN];
	for(int i=0;i<n/2;i++){
		s2[i]=s[a[i]];
	}
	sort(s2,s2+n/2);
	for(int i=0;i<n/2;i++){
		cout<<s2[i]<<endl;
	}
	return 0;
}
