#include<algorithm>
#include<iostream>
#include<cstdio>
#include<map>
#define rep(i,j) for(int i=0;i<j;i++)
using namespace std;
struct node
{
	int p;
	int goal;
	int miss;
	string name;
}b[51];
map<string,node>a;
int n,u,v,tm,mid;
string s[51],x,y,tmp;
bool cmp(node a,node b)
{
	if(a.p<b.p)return false;if(a.p==b.p&&a.goal+b.miss<b.goal+a.miss)return false;
	if(a.p==b.p&&a.goal+b.miss==a.miss+b.goal&&a.goal<b.goal)return false;
	return true;
}
int main()
{
	cin>>n;
	rep(i,n)cin>>s[i];
	rep(i,n*(n-1)/2)
	{
		cin>>tmp;
		tm=tmp.find('-');
		x=tmp.substr(0,tm);
		y=tmp.substr(tm+1,tmp.size()-tm-1);
		scanf("%d:%d",&u,&v);
		if(u>v)a[x].p+=3;
		if(u<v)a[y].p+=3;
		if(u==v)a[x].p++,a[y].p++;
		a[x].goal+=u;a[y].goal+=v;
		a[x].miss+=v;a[y].miss+=u;
	}
	rep(i,n)
	{
		b[i].name=s[i];
		b[i].p=a[s[i]].p;
		b[i].goal=a[s[i]].goal;
		b[i].miss=a[s[i]].miss;
	}
	sort(b,b+n,cmp);
	rep(i,n/2)s[i]=b[i].name;
	sort(s,s+n/2);
	rep(i,n/2)cout<<s[i]<<endl;
	return 0;
}
