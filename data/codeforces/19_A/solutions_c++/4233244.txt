#include<cstdio>
#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<string>
using namespace std;

typedef struct team{
	char nm[35];
	int ap;
	int sc;
	int th;
}team;

team t[55];

int ap(int a,int b){
	if(a>b)return 3;
	else if(a==b) return 1;
	return 0;
}
int cmp(team a ,team b){
	if(a.ap!=b.ap){
		return a.ap>b.ap;
	}else if(a.th!=b.th){
		return a.th>b.th;
	}else return a.sc>b.sc;
}

int main(){
	int n;
	string tp,tp1;

	cin>>n;
	int i,j,k;

	for(i=0;i<n;i++){
		cin>>t[i].nm;
		t[i].sc=t[i].th=t[i].ap=0;
	}
	for(i=0;i<n*(n-1)/2;i++){
		int a,b,t1,t2;
		char ch;
		cin>>tp>>t1>>ch>>t2;

		j=0;
		tp1=tp.substr(0, tp.find('-'));
		tp=tp.substr(tp.find('-')+1);
		for(k=0;k<n;k++){
			if(tp1==t[k].nm){
				a=k;break;
			}
		}
		for(k=0;k<n;k++){
			if(tp==t[k].nm){
				b=k;break;
			}
		}
		t[a].ap+=ap(t1,t2);
		t[a].sc+=t1;
		t[a].th+=t1-t2;

		t[b].ap+=ap(t2,t1);
		t[b].sc+=t2;
		t[b].th+=t2-t1;
	}
	sort(t,n+t,cmp);
	string ans[31];
	for(i=0;i<n/2;i++)
	ans[i]=t[i].nm;
	sort(ans,ans+n/2);
	for(i=0;i<n/2;i++)
		cout<<ans[i]<<endl;
	return 0;
}
