#include <iostream>
#include <sstream>
#include <vector>
#include <map>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
  int n; cin>>n;
  vector<string> names(n);
  for (int i=0; i<n; i++)
    cin>>names[i];

  map<string, int> kt;
  map<string, int> ts;
  map<string, int> gl;

  for (int i=0; i<n*(n-1)/2; i++){
    string s, t; cin>>s>>t;
    int ix=s.find('-');
    string a=s.substr(0, ix);
    string b=s.substr(ix+1);
    //cerr<<a<<", "<<b<<endl;
    istringstream iss(t);
    int as, bs;
    char dmy;
    iss>>as>>dmy>>bs;

    if (as>bs) kt[a]+=3;
    else if (as<bs) kt[b]+=3;
    else kt[a]++, kt[b]++;

    ts[a]+=as-bs;
    ts[b]+=bs-as;

    gl[a]+=as;
    gl[b]+=bs;
  }

  vector<pair<pair<int, pair<int,int> >, string> > ord;
  for (int i=0; i<n; i++)
    ord.push_back(make_pair(make_pair(-kt[names[i]], make_pair(-ts[names[i]], -gl[names[i]])), names[i]));
  sort(ord.begin(), ord.end());

  vector<string> ans;
  for (int i=0; i<n/2; i++)
    ans.push_back(ord[i].second);
  sort(ans.begin(), ans.end());
  for (int i=0; i<n/2; i++)
    cout<<ans[i]<<endl;

  return 0;
}
