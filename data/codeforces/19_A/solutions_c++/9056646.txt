#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
using namespace std;
string nam[54];
int a[54],n,wp[54],dg[54],sg[54];
bool cmp(int a,int b){
	if(wp[a]!=wp[b])return wp[a]>wp[b];
	else if(dg[a]!=dg[b])return dg[a]>dg[b];
	else return sg[a]>sg[b];
}
int main(){
	cin>>n;
	for(int i=1;i<=n;i++)cin>>nam[i];
	for(int i=1;i<=n*(n-1)/2;i++){
		string tmp,a,b;
		int p,q;
		char c;
		cin>>tmp>>p>>c>>q;
		a=tmp.substr(0,tmp.find('-'));
		b=tmp.substr(tmp.find('-')+1);
		int ai=find(nam,nam+n,a)-nam,bi=find(nam,nam+n,b)-nam;
		if(p>q)wp[ai]+=3;
		else if(p<q)wp[bi]+=3;
		else wp[ai]++,wp[bi]++;
		dg[ai]+=p-q,sg[ai]+=p,dg[bi]+=q-p,sg[bi]+=q;
	}
	for(int i=1;i<=n;i++)a[i]=i;
	sort(a+1,a+n+1,cmp);
	string names[54];
	for(int i=1;i<=n;i++)names[i]=nam[a[i]];
	sort(names+1,names+n/2+1);
	for(int i=1;i<=n/2;i++)cout<<names[i]<<endl;
	return 0;
}
