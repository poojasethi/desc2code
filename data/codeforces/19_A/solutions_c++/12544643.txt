#include <stdio.h>
#include <algorithm>
#include <map>
#include <string>
using namespace std;
int n,pts[55],g[55],mg[55],t[55];
map<string,int> id;
string names[55],kt[55];
char s1[35],s2[35];

bool cmp(int a,int b){
	if(pts[a]<pts[b])return false;
	else if(pts[a]>pts[b])return true;
	else{
		if(g[a]-mg[a]<g[b]-mg[b])return false;
		else if(g[a]-mg[a]>g[b]-mg[b])return true;
		else{
			if(g[a]<g[b])return false;
			else if(g[a]>g[b])return true;
		}
	}
	return true;
}

int main(){
	int c1,c2,t1,t2;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%s",s1);
		id[s1]=t[i]=i;
		names[i]=s1;
	}
	for(int i=0;i<n*(n-1)/2;i++){
		scanf("\n%[^-]-%[^ ]%d:%d",s1,s2,&c1,&c2);
		t1=id[s1];t2=id[s2];
		if(c1>c2)pts[t1]+=3;
		else if(c1<c2)pts[t2]+=3;
		else pts[t1]++,pts[t2]++;
		g[t1]+=c1;mg[t2]+=c1;
		g[t2]+=c2;mg[t1]+=c2;
	}
	sort(t,t+n,cmp);
	for(int i=0;i<n/2;i++)kt[i]=names[t[i]];
	sort(kt,kt+n/2);
	for(int i=0;i<n/2;i++)printf("%s\n",kt[i].c_str()); 
	return 0;
}
