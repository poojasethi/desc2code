using namespace std;

const int maxn=100;

int n;
string name[maxn];
map<string,int> g,l,a;

inline bool f(string s1,string s2){
	if(a[s1] != a[s2])
		return a[s1] > a[s2];
	if(g[s1]-l[s1] != g[s2]-l[s2])
		return (g[s1]-l[s1]) > (g[s2]-l[s2]);
	return g[s1] > g[s2];
}

int main(){
	cin >> n;
	rep(i,n){
		cin >> name[i];
	}
	rep(i,n*(n-1)/2){
		int n1,n2;
		string s1,s2;
		char c;
		cin >> s1>> n1>> c >> n2;
		rep(j,s1.length())
			if(s1[j] == '-'){
				s2=s1.substr(j+1);
				s1=s1.substr(0,j);
				break;
			}
		g[s1]+=n1;
		l[s2]+=n1;
		g[s2]+=n2;
		l[s1]+=n2;
		if(n1 > n2)
			a[s1]+=3;
		else if(n2 > n1)
			a[s2]+=3;
		else{
			a[s1]++;
			a[s2]++;
		}
	}
	sort(name,name+n,f);
	sort(name,name+n/2);
	rep(i,n/2)
		cout << name[i] << endl;
	return 0;
}
