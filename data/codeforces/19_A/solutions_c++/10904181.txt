#include<iostream>
#include<algorithm>
#include<string>
#include<cstring>
#include<fstream>
#include<cstdio>
#include<map>
using namespace std;
int goal1,goal2,n,j;
string s[52],tmp,s1,s2;
char c;
map<string,int> sc,g,m;
bool cmp1(string ss1,string ss2)
{
	if(sc[ss1]>sc[ss2])return true;
	if(sc[ss1]<sc[ss2])return false;
	if(g[ss1]-m[ss1]>g[ss2]-m[ss2])return true;
	if(g[ss1]-m[ss1]<g[ss2]-m[ss2])return false;
	return g[ss1]>g[ss2];
}
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>s[i];
	}
	for(int i=0;i<n*(n-1)/2;i++)
	{
		cin>>tmp>>goal1>>c>>goal2;
		j=0;
		while(tmp.substr(j,1)!="-")j++;
		s1=tmp.substr(0,j);
		s2=tmp.substr(j+1,35);
		//cout<<s1<<" "<<s2<<" "<<goal1<<" "<<goal2;
		if(goal1>goal2)sc[s1]+=3;
		if(goal1<goal2)sc[s2]+=3;
		if(goal1==goal2){sc[s1]++;sc[s2]++;}
		g[s1]+=goal1;g[s2]+=goal2;
		m[s1]+=goal2;m[s2]+=goal1;
    }
    sort(s,s+n,cmp1);
    sort(s,s+(n/2));
    for(int i=0;i<n/2;i++)cout<<s[i]<<"\n";
	return 0;
}
