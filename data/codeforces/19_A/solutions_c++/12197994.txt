#include <string>
#include <iostream>
#include <algorithm>
using namespace std;
string names[55];
int n,jf[55],js[55],jq[55];
bool cmp(int a,int b)
{
	if (jf[a] != jf[b])
		return jf[a] > jf[b];
	else if (js[a] != js[b])
		return js[a] > js[b];
	else
		return jq[a] > jq[b];
}
int main ()
{
	cin >> n;
	for (int i = 1;i <= n;++i)
		cin >> names[i];
	
	for (int i = 1;i <= n*(n-1)/2;++i)
	{
		string tmp;int p,q;char ch;
		cin >> tmp >> p >> ch >> q;
		
		string a = tmp.substr(0,tmp.find('-'));
		string b = tmp.substr(tmp.find('-')+1);
		
		int ai = find(names+1,names+1+n,a) - names;
		int bi = find(names+1,names+1+n,b) - names;
		
		if (p > q) jf[ai] += 3;
		else if (q > p) jf[bi] += 3;
		else {jf[ai] ++;jf[bi] ++;}
		
		js[ai] += p-q;jq[ai] += p;
		js[bi] += q-p;jq[bi] += q;
	}
	
	int a[55];
	for (int i = 1;i <= n;++i)
		a[i] = i;
	sort(a+1,a+n+1,cmp);
	
	string s[55];
	for (int i = 1;i <= n/2;++i)
		s[i] = names[a[i]];
	sort(s+1,s+1+(n/2));
	
	for (int i = 1;i <= n/2;++i)
		cout << s[i] << "\n";
	
	return 0;
}
