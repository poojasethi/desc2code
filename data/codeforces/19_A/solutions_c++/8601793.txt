#include <iostream>
#include <string>
#include <algorithm>
#include <cstdio>
using namespace std;
struct team{
	int fs;
	int jq;
	int dq;
	string b;
}a[110];
int cmp(const team u,const team v){
	if (u.fs!=v.fs) return u.fs>v.fs;
	if (u.jq-u.dq!=v.jq-v.dq) return u.jq-u.dq>v.jq-v.dq;
	return u.jq>v.jq;
}
string x,y,s,p[110];
char ch;
int l,k,i,j,t1,t2;
int main(){
	int n;
	cin>>n;
	for(i=1;i<=n;i++){
		cin>>a[i].b;
		a[i].fs=0;
		a[i].jq=0;
		a[i].dq=0;
	}
	for(i=1;i<=n*(n-1)/2;i++){
		cin>>s>>l>>ch>>k;
		x=s.substr(0,s.find("-"));
		y=s.substr(s.find("-")+1,s.find(" ")-1-s.find("-"));
		for(j=1;j<=n;j++){
			if (x==a[j].b) t1=j;
			if (y==a[j].b) t2=j;
		}
		a[t1].jq+=l;
		a[t1].dq+=k;
		a[t2].jq+=k;
		a[t2].dq+=l;
		if (l>k) a[t1].fs+=3;
		if (l==k){a[t1].fs++;a[t2].fs++;}
		if (l<k)a[t2].fs+=3;
	}
	/*for(int i=1;i<=n;i++){
		cout<<a[i].b<<' '<<a[i].fs<<' '<<a[i].jq-a[i].dq<<endl;
	}*/
	sort(a+1,a+n+1,cmp);
	for(int i=1;i<=n/2;i++)
		p[i]=a[i].b;
	sort(p+1,p+n/2+1);
	for(int i=1;i<=n/2;i++)
		cout<<p[i]<<endl;
	return 0;
}
