#include <iostream>
#include <algorithm>
#include <string>
#include <cstdio>
using namespace std;
int n;
struct tea{
       string na;
       int num,in,mis;
};
tea t[50];
void abb(const string & s,int a,int b){
     for(int i=0;i<n;i++)
     if(s==t[i].na){
     if(a>b) t[i].num+=3;else if(a==b) t[i].num+=1;
     t[i].in+=a;
     t[i].mis+=b;
     }
}
bool le(const tea &a,const tea & b){
     if(a.num!=b.num) return a.num>=b.num;
     else 
     if((a.in-a.mis)!=(b.in-b.mis)) 
     return a.in-a.mis>b.in-b.mis; else return a.in>b.in;
}
bool name(const tea&a,const tea&b){
     return a.na<b.na;
}          

int main()
{
    string a,b;
    int c,d;
    char ss;
    cin>>n;
    for(int i=0;i<n;i++)
    cin>>t[i].na;
    cin.ignore();
    for(int i=0;i<n*(n-1)/2;i++){
    getline(cin,a,'-');
    getline(cin,b,' ');
    cin>>c>>ss>>d;
    cin.ignore();
    abb(a,c,d);
    abb(b,d,c);
    }
    sort(t,t+n,le);
    sort(t,t+(n/2),name);
    for(int i=0;i<(n/2);i++)
    cout<<t[i].na<<endl;
        return 0;
}
