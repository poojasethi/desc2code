#include <iostream>
#include <vector>
#include <map>
#include <string>
#include <cassert>
#include <algorithm>
using namespace std;

map<string,int> win, dif, sco;
bool left_win(const string& n1, const string& n2)
{
	if( win[n1] != win[n2] ) return win[n1] > win[n2];
	if( dif[n1] != dif[n2] ) return dif[n1] > dif[n2];
	if( sco[n1] != sco[n2] ) return sco[n1] > sco[n2];
	return false;
}

int main()
{
	int n;
	cin >> n;

	vector<string> name(n);
	for(int i=0; i<n; ++i)
		cin >> name[i];

	for(int i=0; i<n*(n-1)/2; ++i)
	{
		string match;
		int s1, s2; char colon;
		cin >> match >> s1 >> colon >> s2;
		int k = match.find('-');
		string n1 = match.substr(0,k);
		string n2 = match.substr(k+1);

		dif[n1] += s1 - s2;
		dif[n2] += s2 - s1;
		sco[n1] += s1;
		sco[n2] += s2;

		if( s1 < s2 ) {
			win[n1] += 0;
			win[n2] += 3;
		} else if( s1 == s2 ) {
			win[n1] += 1;
			win[n2] += 1;
		} else {
			win[n1] += 3;
			win[n2] += 0;
		}
	}

	sort(name.begin(), name.end(), &left_win);
	sort(name.begin(), name.begin()+n/2);
	for(int i=0; i<n/2; ++i)
		cout << name[i] << endl;
}
