#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <map>

using namespace std;

int n , L , R , l , r;
string a[50 + 2] , s , x , y , v;
char t;
map < string , int > point , goal , num;

inline bool cmp( const string & a , const string & b )
{
	if( point[a] != point[b] ) return point[a] > point[b];
	if( goal[a] != goal[b] ) return goal[a] > goal[b];
	return num[a] > num[b];
}

int main()
{
	cin >> n;
	for( register int i = 1 ; i <= n ; i++ ) cin >> a[i];
	for( register int i = 1 ; i <= n * ( n - 1 ) / 2 ; i++ )
	{
		cin >> s >> l >> t >> r;
		x = s.substr( 0 , s.find( "-" ) );
		y = s.substr( s.find( "-" ) + 1 );
		if( l > r ) L = 3 , R = 0;
		else if( l < r ) L = 0 , R = 3;
		else L = R = 1;
		point[x] += L , point[y] += R;
		goal[x] += l - r , goal[y] += r - l;
		num[x] += l , num[y] += r;
	}
	sort( a + 1 , a + n + 1 , cmp );
	sort( a + 1 , a + n / 2 + 1 );
	for( register int i = 1 ; i <= n / 2 ; i++ )
		cout << a[i] << endl;
	return 0;
}
