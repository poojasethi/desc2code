#include <iostream>
#include <string>
#include <vector>
#include <map>
#include <set>
#include <algorithm>

using namespace std;

int main () {
	
	int n;
	cin >> n;
	vector<string> names(n);
	for(int i=0 ; i<n ; i++) {
		cin >> names[i];
	}

	vector<vector<int> > tab(n, vector<int> (4,0));
	for(int i=0 ; i<n ; i++) tab[i][3] = i;

	int a,b;
	char c;
	string s,na,nb;
	for(int i=0 ; i<n*(n-1)/2 ; i++) {
		cin >> s >> a >> c >> b;
		na = s.substr(0,s.find('-'));
		nb = s.substr(s.find('-')+1);

		int ia, ib;
		ia = find(names.begin(), names.end(), na)-names.begin();
		ib = find(names.begin(), names.end(), nb)-names.begin();

		if(a > b) tab[ia][0] += 3;
		if(a == b) {
			tab[ia][0]++;tab[ib][0]++;
		}
		if(b > a) tab[ib][0] += 3;

		tab[ia][1] += (a-b);
		tab[ia][2] += a;
		tab[ib][1] += (b-a);
		tab[ib][2] += b;
	}

	sort(tab.rbegin(), tab.rend());
	vector<string> nam;
	for(int i=0 ; i<n/2 ; i++)
		nam.push_back(names[tab[i][3]]);
	sort(nam.begin(),nam.end());
	for(int i = 0; i < nam.size(); i++)
		cout << nam[i] << endl;

	return 0;
}
