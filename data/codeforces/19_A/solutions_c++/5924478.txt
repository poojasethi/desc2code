#include<cstdlib>
#include<cstdio>
#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
using namespace std;
int n;
struct team
{
    string nam;
    int pnt,sc,mis;
};
team t[55];

void add(string s,int a,int b)
{
     for(int i=0;i<n;i++)
     if(t[i].nam==s)
     {
         if(a>b)t[i].pnt+=3;
         else if(a==b)t[i].pnt+=1;
         t[i].sc+=a;t[i].mis+=b;
         break;
     }
}
bool cmp1(team a,team b)
{
    if(a.pnt!=b.pnt)return a.pnt>b.pnt;
    else if(a.sc-a.mis!=b.sc-b.mis)return (a.sc-a.mis)>(b.sc-b.mis);
    else return a.sc>b.sc;
}
bool cmp2(team a,team b)
{return a.nam<b.nam;}
int main()
{
    string x,y;
    char z;
    int a,b,i;
    cin>>n;
    for(i=0;i<n;i++)cin>>t[i].nam;
    cin.ignore();
    for(i=0;i<n*(n-1)/2;i++)                                
    {
        getline(cin,x,'-');getline(cin,y,' ');
        scanf("%d:%d",&a,&b);
        add(x,a,b);add(y,b,a);
        cin.ignore();
    }
    sort(t,t+n,cmp1);sort(t,t+n/2,cmp2);
    for(i=0;i<n/2;i++)cout<<t[i].nam<<endl;
    return 0;
}
