#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
string name[100];
int bf[100],jsq[100],jq[100];
bool comp(int a,int b)
{
	if (bf[a]!=bf[b]) return bf[a]>bf[b];
	else if (jsq[a]!=jsq[b]) return jsq[a]>jsq[b];
	else return jq[a]>jq[b];
}
int main()
{
	int n,a[100];
	cin>>n;
	string a1,b1;
	int a2,b2;
	for (int i=0;i<n;i++)
	cin>>name[i];
	for (int i=0;i<n*(n-1)/2;i++)
	{
		string mz;
		int p,q;
		char ch;
		cin>>mz>>p>>ch>>q;
		a1=mz.substr(0,mz.find('-'));
		b1=mz.substr(mz.find('-')+1);
		a2=find(name,name+n,a1)-name;
		b2=find(name,name+n,b1)-name;
		if (p>q) bf[a2]+=3;
		else if (p<q) bf[b2]+=3;
		else 
		{
		    bf[a2]++;
			bf[b2]++;
		}
		jsq[a2]+=p-q;
		jsq[b2]+=q-p;
		jq[a2]+=p;
		jq[b2]+=q;
	}
	for (int i=0;i<n;i++)
	a[i]=i;
	sort(a,a+n,comp);
	string s[100];
	for (int i=0;i<n/2;i++)
	s[i]=name[a[i]];
	sort(s,s+n/2);
	for (int i=0;i<n/2;i++)
	if (i==n/2-1) cout<<s[i];
	else cout<<s[i]<<endl;
	return 0;
}
