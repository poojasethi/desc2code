#include <stdio.h>
#include <iostream>
#include <string>
#include <algorithm>

#define rep(i, n) for(int i=0; i<(int)(n); i++)
using namespace std;

int n;
string names[50];
int wp[50], dg[50], sg[50];

bool comp(int a, int b) {
    if(wp[a]!=wp[b]) return wp[a]>wp[b];
    else if(dg[a]!=dg[b]) return dg[a]>dg[b];
    else return sg[a]>sg[b];
}

int main() {
    cin >> n;
    rep(i, n) cin >> names[i];
    rep(i, n) wp[i] = dg[i] = sg[i] = 0;
    rep(i, n*(n-1)/2) {
        string tmp, a, b;
        int p, q;
        char ch;
        cin >> tmp >> p >> ch >> q;
        a = tmp.substr(0, tmp.find('-'));
        b = tmp.substr(tmp.find('-')+1);
        int ai=find(names, names+n, a)-names, bi=find(names, names+n, b)-names;
        if(p>q) wp[ai] += 3;
        else if(p<q) wp[bi] += 3;
        else { wp[ai]++; wp[bi]++; }
        dg[ai] += p-q; sg[ai] += p;
        dg[bi] += q-p; sg[bi] += q;
    }
    int a[50];
    rep(i, n) a[i] = i;
    sort(a, a+n, comp);
    string s[50];
    rep(i, n/2) s[i] = names[a[i]];
    sort(s, s+n/2);
    rep(i, n/2) printf("%s\n", s[i].c_str());
    return 0;
}
