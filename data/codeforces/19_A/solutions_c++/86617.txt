#include <iostream>
#include <map>
#include <string>
#include <algorithm>
using namespace std;
map<string,int> score_1;
map<string,int> score_2;
map<string,int> score_3;
bool comp(string s1,string s2){
	if (score_1[s1]!=score_1[s2])
		return score_1[s1]>score_1[s2];
	if (score_2[s1]!=score_2[s2])
		return score_2[s1]>score_2[s2];
	return score_3[s1]>score_3[s2];
}
string teams[50];
int main(){
	int n;
	cin>>n;
	for (int i=0;i<n;++i){
		cin>>teams[i];
	}
	for (int i=0;i<n;++i)
		for (int j=i+1;j<n;++j){
			string name,score;
			cin>>name>>score;
			int ind1=name.find('-');
			int ind2=score.find(':');
			string n1=name.substr(0,ind1);
			string n2=name.substr(ind1+1);
			int g1=atoi(score.substr(0,ind2).c_str());
			int g2=atoi(score.substr(ind2+1).c_str());
			if (g1<g2){
				swap(n1,n2);
				swap(g1,g2);
			}
			score_1[n1]+=(g1>g2?3:1);
			score_1[n2]+=(g1==g2?1:0);
			score_2[n1]+=g1-g2;
			score_2[n2]+=g2-g1;
			score_3[n1]+=g1;
			score_3[n2]+=g2;
		}
	sort(teams,teams+n,comp);
	sort(teams,teams+n/2);
	for (int i=0;i<n/2;++i)
		cout<<teams[i]<<endl;
	return 0;
}
