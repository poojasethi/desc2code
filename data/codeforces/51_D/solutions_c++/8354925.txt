#include <cstdio>
#include <algorithm>

using namespace std;

#define maxn 100010
#define eps 1e-8

int n;
double a[maxn];

inline int sign(double x)
{
	if (x > eps)
		return 1;
	if (x < -eps)
		return -1;
	return 0;
}

inline int check(double x, double y)
{
	double q = sign(x) ? y / x : 0;
	int s = 0;
	for (int i = 1; i <= n; i ++)
		if (!sign(x - a[i]))
			x *= q;
		else
			s ++;
	return s;
}

int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; i ++)
		scanf("%lf", &a[i]);
	printf("%d\n", min(min(min(check(a[1], a[3]), check(a[2], a[3])), check(a[1], a[2])), 2));
	return 0;
}
