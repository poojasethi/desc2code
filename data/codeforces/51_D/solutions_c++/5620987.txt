#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int a[120000],ans,i,n;
void check(double x,double y){
	int i,cnt=0;
	for(i=1;i<=n;++i){
		if(abs(x-a[i])>1e-8)cnt++;
		else x*=y;
	}
	ans=min(ans,cnt);
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i)scanf("%d",&a[i]);
	if(n==1)return puts("0"),0;
	if(n==2){
		if(a[1]==0 && a[2]!=0)puts("1");
		else puts("0");
		return 0;
	}
	ans=2;
	check(a[2],!a[2]?0:1.0*a[3]/a[2]); //delete a1
	check(a[1],!a[1]?0:1.0*a[3]/a[1]); //delete a2
	check(a[1],!a[1]?0:1.0*a[2]/a[1]); //use a1 & a2
	printf("%d\n",ans);
}
