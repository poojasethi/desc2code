#include <iostream>
#include <cstdio>
#include <vector>
#include <string>
#include <cstring>
#include <utility>
#include <cmath>
#include <queue>
#include <map>
#include <set>
#include <algorithm>
using namespace std;

int a[111111];

int main(){
    int ans=2,n,p;
    long double d;
    while(cin >> n){
        for(int i=0;i<n;i++) scanf("%d",&a[i]);
        if(n==1){
            cout << 0 << endl;
            continue;
        }else if(n==2){
            if(a[0]==0 && a[1]) cout << 1 << endl;
                           else cout << 0 << endl;
            continue;
        }
        for(int i1=0;i1<2;i1++) for(int i2=i1+1;i2<3;i2++){
                if(a[i1]==0 && a[i2]!=0) continue;
                if(a[i1]==0) d=0; else d=a[i2]/(long double)a[i1];
                p=i2;
                int s=0;
                if (i1!=0||i2!=1)s++;
                for (int i=i2+1;i<n;i++)if (fabs(a[p]*d-a[i])<1e-9)p=i;else s++;
                ans=min(ans,s);
        }
        cout << ans << endl;
    }
    return 0;
}
