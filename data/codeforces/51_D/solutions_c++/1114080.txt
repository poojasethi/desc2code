using namespace std;

int a[111111];
int n;

int main() {
    cin >> n;
    for (int i = 0; i < n; ++i) cin >> a[i];
    if (n == 1) {
        puts("0");
        return 0;
    }
    if (n == 2) {
        if (a[0] == 0 && a[1]) puts("1");
        else puts("0");
        return 0;
    }
    int ans = 2;
    for (int k1 = 0; k1 < 3; ++k1) {
        for (int k2 = k1 + 1; k2 < 3; ++k2) {
            double q;
            if (a[k1] == 0 && a[k2] != 0) continue;
            if (a[k1] == 0) q = 0;
            else q = a[k2] / (a[k1] + 0.0);
            int p = k2;
            int cnt = 0;
            if (k1 != 0 || k2 != 1) cnt++;
            for (int i = k2 + 1; i < n; ++i) {
                if (fabs(a[p] * q - a[i]) < 1E-9) {
                    p = i;
                } else {
                    cnt++;
                }
            }
            ans = min(ans, cnt);
        }
    }
    cout << ans << endl;
}
