#include <iostream>
#include <cmath>
#include <algorithm>
#include <string>
using namespace std;

int nums[100000];

int isGeometric(int s, int n) {
	if (s == n-1)
		return 0;
	int c = nums[s];
	if (c == 0) {
		int ch = 0;
		for (int i = s; i < n; i++)
			if (nums[i] != 0)
				ch++;
		return ch;
	}
	double b = nums[s+1]/(1.0*c);
	int ch = 0;
	c *= b;
	for (int i = s+2; i < n; i++) {
		if (nums[i] != c*b) {
			ch++;
		} else {
			c *= b;
		}
	}
	return ch;
}
int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> nums[i];
	if (isGeometric(0, n) == 0)
		cout << "0" << endl;
	else if(isGeometric(0, n) == 1)
		cout << "1" << endl;
	else if (isGeometric(1, n) == 0){
		cout << "1" << endl;
	} else {
		swap(nums[0], nums[1]);
		if (isGeometric(1, n) == 0)
			cout << "1" << endl;
		else
			cout << "2" << endl;
	}
	
	return 0;
}
