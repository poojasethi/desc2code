#include<stdio.h>
double a[100010];
int n;
int gao(int x,int y)
{
	double c,m;
	int i;
	if(a[x]==0&&a[y]!=0)
	{
		return 2;
	}
	if(a[x]==0&&a[y]==0)
	{
		c=0;
	}
	else
	{
		c=a[y]/a[x];
	}
	int sum=0;
	x=a[y];
	for(i=y+1;i<=n;i++)
	{
		if(a[i]==x*c)
		{
			x=x*c;
			continue;
		}
		else
		{
			sum++;
		}
	}
	return sum;
}
int main(void)
{
	int i;
	int ans;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%lf",&a[i]);
	}
	int flag=2;
	if(n==1) printf("%d\n",0);
	else if(n==2)
	{
		if(a[1]==0&&a[2]!=0)
		{
			printf("%d\n",1);
		}
		else
		{
			printf("%d\n",0);
		}
	}
	else
	{
		ans=gao(1,2);
		if(flag>ans)
		{
			flag=ans;
		}
		ans=gao(1,3);
		if(flag>ans)
		{
			flag=ans+1;
		}
		ans=gao(2,3);
		if(flag>ans)
		{
			flag=ans+1;
		}
		printf("%d\n",flag);
	}
	return 0;
}
