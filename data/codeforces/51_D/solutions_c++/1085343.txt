#define ll __int64
int n,u[100005];
double a[100005];
int ch()
{
    if(n==1)return 1;
    if(n==2&&a[0]!=0)return 1;
    if(a[0]==0)
    {
        for(int i=0;i<n;i++)
        if(a[i]!=0)return 0;
        return 1;
    }
    double b=a[1]/a[0];
    double now=a[1];
    for(int i=2;i<n;i++)
    {
        now*=b;
        if(a[i]!=now)return 0;
    }
    return 1;
}
int ch2(int st,double c,double cb,int del)
{
    if(n==2)return 1;
    if(n==3&&c!=0)return 1;
    if(c==0)
    {
        if(cb!=0)return 0;
        for(int i=3;i<n;i++)
        if(a[i]!=0)
        {
            if(del)return 0;
            del=1;
        }
        return 1;
    }
    double b=cb/c;
    double pre=cb;
    for(int i=st;i<n;i++)
    {
        if(a[i]!=pre*b)
        {
            if(del)return 0;
            del=1;
        }
        else pre*=b;
    }
    return 1;
}
int main()
{
    while(~scanf("%d",&n))
    {
        for(int i=0;i<n;i++)
        {
            scanf("%lf",&a[i]);
            u[i]=1;
        }
        if(ch())puts("0");
        else if(ch2(3,a[1],a[2],1)||ch2(3,a[0],a[2],1)||ch2(2,a[0],a[1],0))puts("1");
        else puts("2");
    }
}
