#define REP(i, n) for (int i = 0, _n = (n); i < _n; ++i)
#define FOR(i, a, b) for (int i = (a), _n = (b); i <= _n; ++i)
#define FORD(i, a, b) for (int i = (a), _n = (b); i >= _n; --i)
#define FORE(it, c) for(__typeof((c).begin()) it = (c).begin(); it != (c).end(); ++it)
#define ALL(c) (c).begin(), (c).end()
#define PB push_back
#define MP make_pair
#define FI first
#define SE second

typedef long long LL;
typedef pair<int,int> PII;
typedef vector<int> VI;

const int INF = 1000000000;

const int NMAX = 100000;

int N, T[NMAX];

int check(int ifir, int isec) {
	if (T[ifir] == 0) {
		int cnt = 0;
		FOR(ind, isec, N-1)
			if (T[ind] != 0) ++cnt;
		return min(cnt, 2);
	}
	
	int cnt = 0, ilast = isec, inext = ilast+1;
	while (inext < N) {
		if (T[inext] * T[ifir] == T[ilast] * T[isec]) {
			ilast = inext++;
		} else {
			++inext;
			if (++cnt == 2) return cnt;
		}
	}
	return cnt;
}

int main() {
	scanf("%d", &N);
	REP(i, N) scanf("%d", T+i);
	if (N <= 2) {
		printf("%d\n", N==2 && T[0] == 0 && T[1] != 0  ? 1 :0);
		return 0;
	}
	
	int res = check(0, 1);
	if (res <= 1) printf("%d\n", res);
	else printf("%d\n", min(2, 1+min(check(0, 2), check(1, 2))));
}
