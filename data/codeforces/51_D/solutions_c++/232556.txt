#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
// check 0
int n;
vector<int> V;
vector<int> TMP;
bool is_geom(){
  int i,j,ch,zm;
  if(TMP.size()<=1)return true;
  if(TMP[0]==0){
   for(j=0;j<TMP.size();++j)if(TMP[j]!=0)return false;
   return true;
               }
  ch=TMP[1];zm=TMP[0];
  int cur=TMP[0],neo;
  for(j=1;j<TMP.size();++j){
   neo=(ch*cur)/zm;
   if(neo*zm!=ch*cur)return false;
   if(neo!=TMP[j])return false;
   cur=neo;
                           }
  return true;
}
int solve(){
  int i,j,ch,zm,br;
  TMP=V;if(is_geom())return 0;
  TMP.clear();
  for(i=0;i<n;++i)if(i!=0)TMP.push_back(V[i]);
  if(is_geom())return 1;
  TMP.clear();
  for(i=0;i<n;++i)if(i!=1)TMP.push_back(V[i]);
  if(is_geom())return 1;
  // LAST
  br=0;
  if(V[0]==0){
   for(j=0;j<n;++j)if(V[j]!=0)++br;
   if(br>1)return 2;
   return 1;
             }
  ch=V[1];zm=V[0];
  int cur=V[0],neo;
  for(j=1;j<V.size();++j){
   neo=(ch*cur)/zm;
   if(neo*zm!=ch*cur){++br;continue;}
   if(neo!=V[j]){++br;continue;}
   cur=neo;
                           }
  if(br<2)return 1;
  return 2;
}
int main(){
  int i,j,k,l;
  //system("pause");
  scanf("%d",&n);
  for(i=0;i<n;++i){scanf("%d",&l);V.push_back(l);}
  TMP=V;
  printf("%d\n",solve());
  //system("pause");
  return 0;
}
