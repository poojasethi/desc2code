#include<iostream>
#include<cstdio>
#include<cmath>

using namespace std;

const int maxn = 100010;
const double Eps = 1e-6;

int n , m;
double data[maxn] , num[maxn];

void input() {
	scanf( "%d" , &n );
	for ( int i = 1 ; i <= n ; i++ )
		scanf( "%lf" , &data[i] );
}

int caln( double a , double q ) {
	int cnt = 0;
	for ( int i = 1 ; i <= n ; i++ )
		if ( fabs( data[i] - a ) < Eps ) {
			cnt++;
			a = data[i] * q;
		}
	if ( cnt == n ) return 0;
	if ( cnt == n - 1 ) return 1;
	return 2;
}

void solve() {
	for ( int i = 1 ; i <= n ; i++ )
		if ( fabs( data[i] ) > Eps ) num[++m] = data[i];
	
	int ans = min( caln( 0 , 0 ) , caln( num[1] , 0 ) );
	if ( m >= 1 ) ans = min( ans , caln( num[1] , 1 ) );
	if ( m >= 2 ) ans = min( ans , caln( num[1] , num[2] / num[1] ) );
	if ( m >= 3 ) {
		ans = min( ans , caln( num[1] , num[3] / num[1] ) );
		ans = min( ans , caln( num[2] , num[3] / num[2] ) );
	}
	printf( "%d\n" , ans );
}

int main() {
	input();
	solve();
	
	return 0;
}
