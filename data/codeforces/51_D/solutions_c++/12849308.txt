#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
int a[100005],b,c,d,e,f,g,h,i,j,k,l,m,n;
void check(double x,double y){
	for(i=1,g=0;i<=n;i++){
		if(abs(x-a[i])<1e-8)x=x*y;
		else g++;
	}
	h=min(g,h);
}

int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	if(n==1){printf("0");}
	else if(n==2){
		if(a[1]==0&&a[2]!=0)printf("1\n");
		else printf("0\n");
	}
	else if(n>=3){
		h=2;
		if(a[2])check(a[2],1.0*a[3]/a[2]);else check(0,0);
		if(a[1])check(a[1],1.0*a[3]/a[1]);else check(0,0);
		if(a[1])check(a[1],1.0*a[2]/a[1]);
		printf("%d\n",h);
	}
	return 0;
}
