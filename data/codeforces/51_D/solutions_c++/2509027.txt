#include<cstdio>
#include<cstring>
#include<cmath>
#define maxn 100010
const double Eps=1e-6;
int n,ans;
double data[maxn];
int min(int a,int b){return a<b?a:b;}
int check(double a,double q){
   int t=0,i;
   for(i=1;i<=n;i++)
       if(fabs(data[i]-a)<Eps){
           a=data[i]*q;
           t++;
       }
   return min(n-t,2);
}
int main(){
    int i;
    scanf("%d",&n);
    for(i=1;i<=n;i++)scanf("%lf",&data[i]);
    ans=min(check(0,0),check(data[1],0));
    if(n>=1)ans=min(ans,check(data[1],1));
    if(n>=2)ans=min(ans,check(data[1],data[2]/data[1]));
    if(n>=3)ans=min(ans,min(check(data[1],data[3]/data[1]),check(data[2],data[3]/data[2])));
    printf("%d\n",ans);
}
