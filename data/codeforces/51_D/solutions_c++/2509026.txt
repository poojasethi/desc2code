#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#include<iostream>
using namespace std;
int minn(int x1,int x2)
{
    return x1<x2?x1:x2;
}
double a[100100],num[100100];
int n,m;
int check(double c,double b)
{
    double now=c;
    int sum=0,i;
    for (i=1; i<=n; i++)
        if (fabs(a[i]-now)<=0.000001) {
            sum++;
            now=a[i]*b;
        }
    if (sum==n) return 0;
    if (sum==n-1) return 1;
    return 2;
}
int main()
{
    int i;
    scanf("%d",&n); m=0;
    for (i=1; i<=n; i++) {
        scanf("%lf",&a[i]);
        if (!(fabs(a[i])<0.000001)) num[++m]=a[i];
    }
    int ans=minn(check(0,0),check(num[1],0));
    if (n>=1) ans=minn(ans,check(num[1],1));
    if (n>=2) ans=minn(ans,check(num[1],num[2]/num[1]));
    if (n>=3) {
        ans=minn(ans,check(num[1],num[3]/num[1]));
        ans=minn(ans,check(num[2],num[3]/num[2]));
    }
    printf("%d\n",ans);
    return 0;
}
