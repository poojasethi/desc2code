#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<stdlib.h>
#include<math.h>
using namespace std;
bool cmp(int a,int b)
{
	return abs(a)<abs(b);
}
int F=-100000;
int a[100010];
bool check(int p,double c,int n)
{
	bool ans=0;
	int bf=p;
	for(int i=p+1;i<n;i++)
	{
		if(a[i]==F)
			continue;
		double x=a[bf]*c-a[i];
		if(fabs(a[bf]*c-a[i])>1e-4)
		{
			ans=1;
			break;
		}
		bf=i;
	}
	return ans;
}
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	if(n==1)
	{
		printf("0\n");
		return 0;
	}
	if(a[0]==0)
	{
		int count=0;
		for(int i=1;i<n;i++)
		{
			if(a[i]!=0)
				count++;
		}
		if(count==n-1)
		{
			if(n==2)
				printf("1\n");
			else if(check(1,1.0*a[2]/a[1],n))
				printf("2\n");
			else
				printf("1\n");
		}
		else
			printf("%d\n",min(count,2));
		return 0;
	}
	int sb=0;
	double c=1.0*a[1]/a[0];
	if(check(0,c,n))
	{
		sb=1;
		int x=a[0];
		a[0]=F;
		if(a[1]==0||check(1,1.0*a[2]/a[1],n))
		{
			a[0]=x;
			x=a[1];
			a[1]=F;
			if(check(0,1.0*a[2]/a[0],n))
			{
				a[1]=x;
				x=a[2];
				a[2]=F;
				if(check(0,c,n))
				{
					a[2]=x;
					for(int i=1;i<n;i++)
					{
						if(fabs(a[i-1]*c-a[i])>1e-4)
						{
							sb=1;
							a[i]=F;
							if(check(0,c,n))
								sb=2;
							break;
						}
					}
				}
			}
		}
	}
	printf("%d\n",sb);
	return 0;
}
