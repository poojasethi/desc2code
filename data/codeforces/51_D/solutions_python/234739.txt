#!/usr/bin/env python
import sys

if __name__ == '__main__':
    n = int(sys.stdin.readline())
    a = map(int, sys.stdin.readline().split())
    if n == 1:
        print 0
        sys.exit()
    elif n == 2 and a[0] != 0:
        print 0
        sys.exit()
    elif n == 2 and a[0] == 0 and a[1] == 0:
        print 0
        sys.exit()
    elif n == 2:
        print 1
        sys.exit()
    c = a[0]
    if c == 0:
        b = 0
    else:
        b = a[1] / float(c)
    p = a[0] * b
    q = 0
    for x in a[1:]:
        if x != p:
            q += 1
        else:
            p *= b
    if q == 0:
        print 0
        sys.exit()
    elif q == 1:
        print 1
        sys.exit()

    c = a[1]

    else:
        b = a[2] / float(c)

            q += 1

            p *= b
