#include <bits/stdc++.h>
using namespace std;
const int maxn = 505;
int a[maxn], b[maxn];
int dp[maxn], par[maxn];
int n, m;

void print(int pos)
{
    if (!pos)
        return ;
    print(par[pos]);
    printf("%d ", b[pos]);
}
int main()
{
    int i, j;
    scanf("%d", &n); for (i = 1; i <= n; i++) scanf("%d", &a[i]);
    scanf("%d", &m); for (i = 1; i <= m; i++) scanf("%d", &b[i]);
    int maxi = 0;
    for (i = 1; i <= n; i++)
    {
        maxi = 0;
        for (j = 1; j <= m; j++)
        if (a[i] == b[j]){
            dp[j] = dp[maxi] + 1;
            par[j] = maxi;
        } else if (b[j] < a[i] && dp[maxi] < dp[j])
            maxi = j;
    }maxi = 0;
    for (i = 1; i <= m; i++)
    if (dp[maxi] < dp[i])
        maxi =i;
    printf("%d\n", dp[maxi]);
    print(maxi);
    printf("\n");

}
