#include<cstdio>
#include<iostream>
#include <algorithm>
using namespace std;
int n,m,t[5000],p[5000],a[5000],b[5000],f[5000],i,w,j,max1;
void outp(int t)
{
     if (t!= 0) outp(p[t]),printf("%d ",b[t]);
}
int main()
{
    scanf("%d",&n);
    for (i=1; i<=n; i++) scanf("%d",&a[i]);
    scanf("%d",&m);
    for (i=1; i<=m; i++) scanf("%d",&b[i]);
    for (i=1; i<=n; i++)
    {
        max1=0;w=0;
        for (j=1; j<=m; j++)
        {
            if (a[i]>b[j]&&max1<f[j])  max1=f[j], w=j;
            if (a[i]==b[j])  f[j]=max1+1,p[j]=w;
        }
    }
    max1=-1;
    for (i=1; i<=m; i++) 
    if (f[i]>max1) 
        {
             max1=f[i];
             w=i;
        }
    printf("%d\n",max1);
    if (max1!=0) outp(w);
    return 0;
}
