#include "bits/stdc++.h"
#define MAXN 505
using namespace std;
int N,M;
int A[MAXN];
int B[MAXN];
int DP[MAXN][MAXN];
int Yol[MAXN][MAXN];
int F(int x,int y)
{
	if(x == N+1 || y == M)	return 0;
	int &ret = DP[x][y];
	if(ret != -1)	return ret;
	ret = F(x+1,y);
	Yol[x][y] = y;
	int i,t;
	if(A[x] > B[y])
		for(i=y+1; i<=M; i++)
			if(A[x] == B[i])
			{
				t = 1 + F(x+1,i);
				if(t > ret)
				{
					ret = t;
					Yol[x][y] = i;
				}
				break;
			}
	return ret;
}
void Print(int x,int y)
{
	if(x == N+2)	return;
	if(A[x-1] == B[y])
		printf("%d ",B[y]);
	Print(x+1,Yol[x][y]);
}
int main()
{
	int i;
	scanf("%d",&N);
	for(i=1; i<=N; i++)
		scanf("%d",&A[i]);
	scanf("%d",&M);
	for(i=1; i<=M; i++)
		scanf("%d",&B[i]);
	memset(DP,-1,sizeof DP);
	B[0] = -1;
	printf("%d\n",F(1,0));
	Print(1,0);
	printf("\n");
	return 0;
}
