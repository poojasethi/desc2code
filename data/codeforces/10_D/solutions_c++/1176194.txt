#include <iostream>
using namespace std;
typedef long long ll;
int n, m, f[505], a[505], b[505], p[505][505], ans[505], cnt;
void out(int x, int l)
{
  if (l == 0) return;
  out(p[x][l], l - 1);
  ans[cnt++] = b[x];
}
int main()
{
  cin >> n;
  for (int i = 0; i < n; i++)
    cin >> a[i];
  cin >> m;
  for (int i = 0; i < m; i++)
    cin >> b[i];
  a[n] = b[m] = 1e9 + 5;
  for (int i = 0; i <= n; i++)
  {
    int mx = 0, ind = -1;
    for (int j = 0; j <= m; j++)
    {
      if (b[j] == a[i] && mx + 1 > f[j])
      {
        f[j] = mx + 1;
        p[j][f[j]] = ind;
      }
      if (b[j] < a[i] && f[j] > mx)
      {
        ind = j;
        mx = f[j];
      }
    }
  }
  //for (int i = 0; i < m; i++) cout << f[i] << " ";cout << endl;
  out(m, f[m]);
  cout << cnt - 1 << endl;
  for (int i = 0; i < cnt - 1; i++)
    cout << ans[i] << " ";
  return 0;
}
