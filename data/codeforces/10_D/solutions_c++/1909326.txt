#include <iostream>

using namespace std;

const int MAXN = 500+10;

int n, m, a[MAXN], b[MAXN], dp[MAXN], par[MAXN];

void solve()
{
	for (int i = 1 ; i <= n ; i++)
	{
		int x = a[i], indmax = 0;
		for (int j = 1 ; j <= m ; j++)
		{
			int y = b[j];
			if (x == y)
			{
				dp[j] = dp[indmax] + 1;
				par[j] = indmax;
			}
			else if (x > y && dp[indmax] < dp[j])
				indmax = j;
		}
	}
}

void print(int ind)
{
	if (!ind)
		return;
	print(par[ind]);
	cout << b[ind] << " ";
}

int main()
{
	cin >> n;
	for (int i = 1 ; i <= n ; i++)
		cin >> a[i];
	cin >> m;
	for (int i = 1 ; i <= m ; i++)
		cin >> b[i];
	solve();
	int ind = 0;
	for (int i = 1 ; i <= m ; i++)
		if (dp[ind] < dp[i])
			ind = i;
	cout << dp[ind] << endl;
	print(ind);
	cout << endl;
	return 0;
}
