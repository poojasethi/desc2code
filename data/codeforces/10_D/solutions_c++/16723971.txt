#include<iostream>
#include<cstring>
using namespace std;
int A1[510],A2[510],ans[510],c[510];
int v[510];
int main(){
	int a1,a2,i,j,flag,maxj,k;
    cin>>a1;
	for(i=1;i<=a1;i++)cin>>A1[i];
    cin>>a2;
	for(i=1;i<=a2;i++)cin>>A2[i];	
		memset(v,0,sizeof(v));
	for(i=1;i<=a1;i++){
		maxj=0;flag=-1;
		for(j=1;j<=a2;j++){
			if(A1[i]>A2[j]&&maxj<v[j]){maxj=v[j];flag=j;}
			if(A1[i]==A2[j]){v[j]=maxj+1;ans[j]=flag;}
		}
	}	
	maxj=0;	i=0;
    for(j=1;j<=a2;j++)if(v[j]>maxj){maxj=v[j];i=j;}
	cout<<maxj<<endl;
	if (maxj>0){
    j=i;	k=0;
	while(j!=-1){
		k++;c[k]=j;j=ans[j];
		}
	for(i=k;i>=1;i--)cout<<A2[c[i]]<<" ";
//	cout<<endl;
//	for(j=1;j<=a1;j++)cout<<ans[j]<<" ";
	}
	return 0;
}
