#include<bits/stdc++.h>
#define M 1000000007
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
int a[510],b[510],dp[510],p[510];
void print(int i)
{
	if(!i)
		return;
	print(p[i]);
	cout<<b[i]<<' ';
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int n,m,i,j,k;
	cin>>n;
	for(i=1;i<=n;i++)
		cin>>a[i];
	cin>>m;
	for(i=1;i<=m;i++)
		cin>>b[i];
	for(i=1;i<=n;i++,k=0)
		for(j=1;j<=m;j++)
		{
			if(a[i]==b[j])
			{
				dp[j]=dp[k]+1;
				p[j]=k;
			}
			else if(a[i] > b[j] && dp[j] > dp[k])
				k=j;
		}
	j=0;
	for(i=1;i<=m;i++)
		if(dp[i]>dp[j])
			j=i;
	cout<<dp[j]<<'\n';
	print(j);
	cout<<'\n';
	return 0;
}
