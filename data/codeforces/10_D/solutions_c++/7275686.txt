#include <bits/stdc++.h>
using namespace std;
const int maxn=500+5;
int a[maxn],b[maxn];
int dp[maxn],par[maxn];
int n,m;
void print(int pos){
	if(!pos)
		return ;
	print(par[pos]);
	cout<<b[pos]<<' ';
}
int main(){
	cin>>n;for(int i=1 ; i<=n ; i++)cin>>a[i];
	cin>>m;for(int j=1 ; j<=m ; j++)cin>>b[j];
	int maxi;
	for(int i=1 ; i<=n ; i++){
		maxi=0;
		for(int j=1 ; j<=m ; j++){
			if(a[i]==b[j]){
				dp[j]=dp[maxi]+1;
				par[j]=maxi;
			}
			else if(b[j]<a[i] && dp[maxi]<dp[j])
				maxi=j;
		}
	}maxi=0;
	for(int i=1 ; i<=m ; i++){
		if(dp[maxi]<dp[i])
			maxi=i;
	}cout<<endl;
	cout<<dp[maxi]<<endl;
	print(maxi);cout<<endl;
	return 0;
}
