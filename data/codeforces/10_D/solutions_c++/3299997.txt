#include <cstdio>
#include <algorithm>
#define N 505
#define FI(a, b, c) for(int a = (b); a <= (c); a++)
using namespace std;

int n, a[N], m, b[N], d[N], e[N], l;

void trace(int x, bool m){
	if(!x) return;
	trace(e[x], 1);
	printf("%d", b[x]);
	putchar(m ? ' ' : '\n');
}

int main(){
	scanf("%d", &n);
	FI(i, 1, n) scanf("%d", &a[i]);
	scanf("%d", &m);
	FI(i, 1, m) scanf("%d", &b[i]);
	
	FI(i, 1, n){
		int c = 0, p = 0;
		FI(j, 1, m){
			if(a[i] == b[j] && d[j] < c + 1)
				d[j] = c + 1, e[j] = p;
			if(a[i] > b[j] && c < d[j])
				c = d[j], p = j;
		}
	}
	
	FI(i, 1, m) l = max(l, d[i]);
	printf("%d\n", l);
	if(l) FI(i, 1, m) if(d[i] == l){
		trace(i, 0);
		break;
	}
	scanf("\n");
}
