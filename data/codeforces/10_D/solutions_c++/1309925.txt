#include <iostream>
#include <fstream>
#include <vector>

#define MAX 505
using namespace std;

int N,M;
int A[MAX],B[MAX];
int D[MAX];
int Y[MAX];

int main()
{
	int i,x,mx,j,in;
	cin >> N;
	for(i=1;i<=N;i++) cin >> A[i];
	cin >> M;
	for(i=1;i<=M;i++) cin >> B[i];
	for(i=1;i<=N;i++)
	{
		in=mx=0;
		for(j=1;j<=M;j++)
		{
			if(A[i] == B[j] && mx+1 > D[j])
			{
				D[j]=mx+1;
				Y[j]=in;
			}
			else if(A[i] > B[j] && D[j] > mx)
			{
				mx=D[j];
				in=j;
			}
		}
	}
	in=mx=0;
	for(i=1;i<=M;i++) if(D[i] > mx) mx=D[i],in=i;
	vector<int> vec;
	while(in)
	{
		vec.push_back(B[in]);
		in=Y[in];
	}
	cout << vec.size()<< endl;
	for(i=vec.size()-1;i>=0;i--) cout << vec[i] << " ";
	cout << endl;
	return 0;
}
