# include <iostream>
# include <vector>
# define PB push_back
using namespace std;
const int MAX_N=500+10;
int A[MAX_N],B[MAX_N],Ans[MAX_N],Par[MAX_N];
int N,M;
int PRV;
int main()
{
    ios::sync_with_stdio(false);
    cin >> N ;
    for(int i=1;i<=N;i++)cin >> A[i];
    cin >> M;
    for(int i=1;i<=M;i++)cin >> B[i];
    
    for(int i=1;i<=N;i++,PRV=0) 
        for(int j=1;j<=M;j++)
            if(A[i]==B[j])
            {
                Ans[j]=Ans[PRV]+1;
                Par[j]=PRV;
            }
            else if(B[j]<A[i] && Ans[j]>Ans[PRV])
                PRV=j;
    
    int MX=-1;
    for(int i=1;i<=M;i++)
            if(MX==-1 || Ans[MX]<Ans[i])
                    MX=i;
    cout<<Ans[MX]<<endl;
    if(Ans[MX]==0)return 0;
    vector<int> V;
    while(MX){V.PB(B[MX]);MX=Par[MX];}
    for(int i=(int)V.size()-1;i>=0;i--) cout<<V[i]<<" ";
    cout<<endl;
    
    return 0;
}
