#include <bits/stdc++.h>
using namespace std;
	
int m,n,res=0;
int a[502],b[502],f[502],t[502]={0};

int main()
{
	int i,j,vt,c,prev;
	
	cin >> m;
	for (i=1;i<=m;i++) cin >> a[i];
	cin >> n;
	for (i=1;i<=n;i++) cin >> b[i];
		
	for (i=1;i<=m;i++)
	{
		c = 0; prev = 0;
		for (j=1;j<=n;j++)
		{
			if (a[i]==b[j] && c+1>f[j])
			{
				f[j] = c+1;
				t[j] = prev;
				prev = j;
			}
			if (b[j]<a[i] && c<f[j]) {c=f[j];prev=j;}
		}
	}

	res=0;
	for (i=1;i<=n;i++) if (f[i] > res)
	{
		res = f[i];
		vt = i;
	}

	cout << res << "\n";
	if (res==0) return 0;
	int r[502];j=0;
	while (vt)
	{
		j++;r[j] = b[vt];
		vt = t[vt];
	}	
	for (i=j;i>=1;i--) cout << r[i] << " ";
	return 0;
}
