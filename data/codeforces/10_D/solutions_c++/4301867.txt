#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define  MAXN 505
int a[MAXN],b[MAXN];
int dp[MAXN],pre[MAXN],m,n;
void dfs(int p)
{
	  if(p==0) return;
	  dfs(pre[p]);
	  printf("%d ",b[p]);
}
int main()
{   
	while(scanf("%d",&n)!=EOF)
	{			  
		for(int i=1;i<=n;i++)
			scanf("%d",&a[i]);
		scanf("%d",&m);
		for(int i=1;i<=m;i++)
			scanf("%d",&b[i]);
		memset(dp,0,sizeof(dp));
		a[0]=b[0]=-1;
		for(int i=0;i<=n;i++)
		{
			int p=0;
			for(int j=0;j<=m;j++)
			{  
				if(a[i]>b[j]&&dp[j]>dp[p]) p=j;
				if(a[i]==b[j])
					dp[j]=dp[p]+1,pre[j]=p;
			}
		}
		int ans=0,index;
		for(int i=0;i<=m;i++)
			if(dp[i]>ans)
				ans=dp[i],index=i;
		printf("%d\n",ans-1);
		dfs(index);
	}
	return 0;
}
