#include <iostream>
#include <cstdio>
#include <string>
#include <cstring>
#include <algorithm>
using namespace std;

int dp[505],i[505],j[505],l[505],ans[505];

int main(){
	
	int n,m,a,b,c,d,e;
	scanf("%d",&n); for(a=0;a<n;a++) scanf("%d",&i[a]);
	scanf("%d",&m); for(a=0;a<m;a++) scanf("%d",&j[a]);
	for(a=0;a<m;a++){
		c=j[a];
		d=0;
		e=a;
		for(b=0;b<n;b++){
			if(i[b]<c && d<dp[b]){ d=dp[b]; e=b; } 
			if(i[b]==c && d+1>dp[b]){ dp[b]=d+1; l[b]=e; }
		}
	}
	c=0; 
	for(a=0;a<n;a++) if(dp[a]>c){ c=dp[a]; d=a; }
	printf("%d\n",c); b=c;
	ans[b--]=i[d];
	for(a=0;;a++){
		if(b<=0) break; 
		d=l[d];
		ans[b--]=i[d];
	}
	for(a=0;a<c;a++){
		if(a) printf(" ");
		printf("%d",ans[a+1]);
	}
	printf("\n");
	return 0;
}
