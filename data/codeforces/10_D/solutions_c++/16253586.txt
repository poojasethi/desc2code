#include<bits/stdc++.h>
using namespace std;
int n,m,i,j,k,p,ans,a[501],b[501],f[501],t[501],s[501];
int main()
{
    scanf("%d",&n);
    for(i=1;i<=n;i++)
        scanf("%d",&a[i]);
    scanf("%d",&m);
    for(i=1;i<=m;i++)
        scanf("%d",&b[i]);
    for(i=1;i<=n;i++)
    {
        k=0;
        for(j=1;j<=m;j++)
        {
            if(a[i]==b[j]&&f[k]+1>f[j])
            {
                f[j]=f[k]+1;
                t[j]=k;
            }
            if(a[i]>b[j]&&f[k]<f[j])
                k=j;
        }
    }
    ans=0;
    for(i=1;i<=m;i++)
        if(f[i]>ans)
    {
        ans=f[i];
        p=i;
    }
    printf("%d\n",ans);
    if(ans==0) return 0;
    k=0;
    while(p!=0)
    {
        k++;
        s[k]=b[p];
        p=t[p];
    }
    for(i=k;i>1;i--)
        printf("%d ",s[i]);
    printf("%d",s[1]);
    return 0;
}
