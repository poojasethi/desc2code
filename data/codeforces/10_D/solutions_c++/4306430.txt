#include <iostream>
#include <vector>
using namespace std;

	int a[500], b[500];
	int dp[500], prev[500];
	int out[500];

int main() {
	int A, B;
	cin >> A;
	for (int i=0; i<A; ++i) cin >> a[i];
	cin >> B;
	for (int i=0; i<B; ++i) cin >> b[i];

	for (int i=0; i<A; ++i) {
		int n = 0, p = -1;
		for (int j=0; j<B; ++j) {
			if (b[j] == a[i])
				dp[j] = n + 1, prev[j] = p;

			if (b[j] < a[i] && n < dp[j])
				n = dp[j], p = j;
		}
	}

	int n = 0, index = -1;
	for (int i=0; i<B; ++i)
		if (dp[i] > n)
			n = dp[index = i];

	for (int i=n-1; i>=0; --i) {
		out[i] = b[index];
		index = prev[index];
	}

	cout << n << '\n';
	for (int i=0; i<n; ++i) {
		if (i) cout << ' ';
		cout << out[i];
	}
	cout << '\n';

	return 0;
}
