#include <iostream>
#include <cstdio>
#include <string>
using namespace std;
int i,j,n,m,a[510],b[510],dp[510],where,p[510],ans[510],wh;
int main()
{
//freopen("gcis.in","r",stdin);
//freopen("gcis.out","w",stdout);
cin>>n;for (i=1;i<=n;i++) cin>>a[i];cin>>m;for (i=1;i<=m;i++) cin>>b[i];
for (i=1;i<=n;i++) for (j=1;j<=m;j++) {if (j==1) where=0; if (a[i]==b[j] && dp[where]+1>dp[j]) dp[j]=dp[where]+1,p[j]=where; else if (dp[where]<dp[j] && b[j]<a[i]) where=j;}
where=0;for (i=1;i<=m;i++) if (dp[i]>dp[where]) where=i;
for (i=where;i>0;i=p[i]) {ans[wh++]=b[i];}
cout<<wh<<endl;
for (i=wh-1;i>=0;i--) cout<<ans[i]<<" ";

return 0;
}
