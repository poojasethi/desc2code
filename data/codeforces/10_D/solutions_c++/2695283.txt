#include <iostream>

using namespace std;

int d [510], a [510], b [510], par [510];

void print (int k) {
	if (! k)
		return;
	print (par [k]);
	cout << b [k] << " ";
}
int main () {
	ios::sync_with_stdio (0);
	int n, m, k, ans = 0;
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> a [i];
	cin >> m;
	for (int i = 1; i <= m; i++)
		cin >> b [i];
	for (int i = 1; i <= n; i++, k = 0)
		for (int j = 1; j <= m; j++) 
			if (a [i] == b [j]) 
				d [j] = d [k] + 1, par [j] = k;
			else if (a [i] > b [j] && d [j] > d [k])
				k = j;
	k = 0;
	for (int i = 1; i <= m; i++)
		if (d [i] > ans)
			ans = d [i], k = i;
	cout << ans << endl;
	print (k);
	return 0;
}
