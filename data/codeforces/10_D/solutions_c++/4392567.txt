#include<stdio.h>
int a,b,A[505],B[505],res[505],C[505],prev[505],index,p=1,cur=0,last=-1,mx=-1;
int main()
{
    scanf("%d",&a);
    for(int i=1; i<=a; i++)scanf("%d",&A[i]);
    scanf("%d",&b);
    for(int i=1; i<=b; i++)scanf("%d",&B[i]);
    for(int i=1; i<=a; i++)
    {
        cur=0,last=-1;
        for(int j=1; j<=b; j++)
        {
            if(A[i]==B[j]&&cur+1>C[j])C[j]=cur+1,prev[j]=last;
            if(B[j]<A[i]&&C[j]>cur)cur=C[j],last=j;
        }
    }
    for(int i=1; i<=b; i++)if(mx<C[i])mx=C[i],index=i;
    if(mx>0)
    {
        printf("%d\n",mx);
        for(;index!=-1;index=prev[index])res[p++]=B[index];
        for(int i=p-1;i>=1; i--)printf("%d%s",res[i],i==1?"\n":" ");
    }
    else puts("0");
}
