#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int n,m;

	cin>>n;
	vector<int> a(n+1,-1);
	for(int i=1;i<=n;i++)
		cin>>a[i];

	cin>>m;
	vector<int> b(m+1,-1);
	for(int i=1;i<=m;i++)
		cin>>b[i];

	vector<int> r(n+1,0),p(n+1,-1);
	for(int i=1;i<m+1;i++)
	{
		int k=0;

		for(int j=1;j<n+1;j++)
		{
			if(b[i]==a[j])
			{
				r[j]=r[k]+1;
				p[j]=k;
			}

			if(a[j]<b[i] && r[k]<r[j])
				k=j;
		}
	}

	int res=0;
	for(int i=1;i<=n;i++)
		if(r[res]<r[i]) res=i;

	cout<<r[res]<<endl;

	vector<int> v;
	while(a[res]!=-1)
	{
		v.push_back(a[res]);
		res=p[res];
	}

	for(int i=v.size()-1;i>=0;i--)
		cout<<v[i]<<" ";

	return 0;
}
