#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int f[510][510],a[510],b[510];
int path[510][510];
int main()
{
	int n,m,i,j,k;
	memset(f,0,sizeof(f));
	memset(path,-1,sizeof(path));
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	scanf("%d",&m);
	for(i=1;i<=m;i++)
		scanf("%d",&b[i]);
	int max=0,maxi=0,maxj=0;
	for(i=1;i<=n;i++){
		k=0;
		for(j=1;j<=m;j++){
			f[i][j]=f[i-1][j];
			if(a[i]>b[j]&&f[i][j]>f[i][k])
				k=j;
			if(a[i]==b[j]&&f[i][j]<f[i][k]+1){
				f[i][j]=f[i][k]+1;
				path[i][j]=k;
				if(f[i][j]>max){
					max=f[i][j];
					maxi=i;
					maxj=j;
				}
			}
		}
	}
	printf("%d\n",max);
	int re[510];
	k=max;
	while(k){
		if(path[maxi][maxj]>-1){
			re[k--]=b[maxj];
			maxj=path[maxi][maxj];
		}
		maxi--;
	}
	for(i=1;i<=max;i++){
		printf("%d",re[i]);
		if(i!=max) printf(" ");
	}
	printf("\n");
	return 0;
}
