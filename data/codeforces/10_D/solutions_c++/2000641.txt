#include<stdio.h>
#include<string.h>
int a[510],b[510];
int dp[510],pre[510];
void print(int d)
{
	if(!d) return ;
	print(pre[d]);
	if(pre[d]) printf(" ");
	printf("%d",b[d]);
}
int main()
{
	int t,m,n,i,j;

	scanf("%d",&m);
	for(i=1;i<=m;i++)
		scanf("%d",&a[i]);
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&b[i]);
	memset(dp,0,sizeof(dp));
	int ans=0,k;
	dp[0]=0;
	memset(pre,0,sizeof(pre));
	for(i=1;i<=m;i++)
	{
		k=0;
		for(j=1;j<=n;j++)
		{
			if(b[j]<a[i]&&dp[j]>dp[k])
				k=j;
			if(b[j]==a[i])
			{
				dp[j]=dp[k]+1;
				pre[j]=k;
			}
		}
	}
	int id=-1;
	for(i=0;i<=n;i++)
	{
		if(ans<dp[i])
		{
			ans=dp[i];
			id=i;
		}
	}
	printf("%d\n",ans);
	if(ans)
	{
		print(id);
		printf("\n");
	}

	return 0;
}
