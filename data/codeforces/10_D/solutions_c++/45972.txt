#include <iostream>
#include <cstring>
using namespace std;

int a[512];
int b[512];

int d[512];
int f[512];

int main()
{
	int n, m;
	
	cin >> n;
	for(int i = 0; i < n; ++i)
	{
		cin >> a[i];
	}
	
	cin >> m;
	for(int i = 0; i < m; ++i)
	{
		cin >> b[i];
	}
	
	memset(d, 0, sizeof(d));
	
	for(int i = n - 1; i >= 0; --i)
	{
		int ma = 0;
		int maj = -1;
		for(int j = m - 1; j >= 0; --j)
		{
			if (b[j] > a[i] && d[j] > ma)
			{
				ma = d[j];
				maj = j;
			}
			else if (b[j] == a[i] && d[j] < ma + 1)
			{
				d[j] = ma + 1;
				f[j] = maj;
			}
		}
	}
	
	int res = 0;
	int v = -1;
	for(int i = 0; i < m; ++i)
	{
		if (d[i] > res)
		{
			res = d[i];
			v = i;
		}
	}
	
	cout << res << endl;
	for(int i = 0; i < res; ++i)
	{
		cout << b[v] << ' ';
		v = f[v];
	}

	return 0;
}
