#include <iostream>
#include <cstdio>
#include <algorithm>
#define MAXN 500

using namespace std;

int N,M,A[MAXN],B[MAXN],dp[MAXN],yer[MAXN];
int mx,last,res;

int main()
{
	int i,j;

	scanf("%d",&N);
	for(i=1;i<=N;i++)
		scanf("%d",&A[i]);

	scanf("%d",&M);
	for(i=1;i<=M;i++)
		scanf("%d",&B[i]);

	for(i=1;i<=N;i++)
	{
		mx=last=0;
		for(j=1;j<=M;j++)
			if(A[i] == B[j] && dp[j] < mx + 1) 
			{
				yer[j] = last;
				dp[j] = mx + 1;
			}
			else if(A[i] > B[j] && mx < dp[j])
			{
				mx = dp[j];
				last = j;
			}
	}

	mx=0;
	for(i=1;i<=M;i++) 
		if(dp[i] > mx)
		{
			mx = dp[i];
			last = i;
		}
	res=mx;

	cout << mx << endl;
	while(last)
	{
		A[res--] = B[last];
		last = yer[last];
	}
	for(i=1;i<=mx;i++)
		cout << A[i] << " ";
	cout << endl;

	return 0;
}
