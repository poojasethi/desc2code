def getints(): return map(int, raw_input().split())

n = input()
a = getints()
input()
f = [0] * (n + 1)
p = [-1] * n
for b in getints():
	k = -1
	for i in xrange(n):
		if a[i] == b: f[i], p[i] = f[k] + 1, k
		if a[i] < b and f[k] < f[i]: k = i

n = max(f)
i, f = f.index(n), []
print n
for n in xrange(n):
	f.append(`a[i]`)
	i = p[i]
print ' '.join(f[::-1])
