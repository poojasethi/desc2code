#include <cstdio>
#include <cstring>

int main()
{
	int n;
	scanf("%d", &n);
	
	char s[200];
	gets(s);
	
	for (int i = 0; i < n; ++i)
	{
		gets(s);
		
		int len = strlen(s);
		int res = 0;
		
		if (len >= 5 && strncmp(s, "miao.", 5) == 0)
			res |= 1;
		if (len >= 5 && strncmp(s + len - 5, "lala.", 5) == 0)
			res |= 2;
		
		if (res == 1)
			printf("Rainbow's\n");
		else if (res == 2)
			printf("Freda's\n");
		else
			printf("OMG>.< I don't know!\n");
	}
	
	return 0;
}
