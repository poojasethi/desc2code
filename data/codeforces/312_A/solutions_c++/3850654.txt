#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

int n,f1,f2,i;
string s;

int main (int argc, char * const argv[]) {
	cin>>n;
	getline(cin,s);
	for(i=1;i<=n;i++){
		getline(cin,s);
		if(s.length()<5)cout<<"OMG>.< I don't know!"<<endl;else{
			f1=s.substr(0,5)=="miao.";
			f2=s.substr(s.length()-5,5)=="lala.";
			if((f1^f2)==0)cout<<"OMG>.< I don't know!"<<endl;else
				if(f1)cout<<"Rainbow's"<<endl;else cout<<"Freda's"<<endl;
		}
	}
    return 0;
}
