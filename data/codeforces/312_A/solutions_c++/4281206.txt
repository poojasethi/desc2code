#include<iostream>
#include<string>
using namespace std;

	int n,i,be,en,si;
	bool a=0,b=0;

	cin >> n; getline(cin,s);
	for(i=1;i<=n;i++)
	{
		s="",
		a=0,
		b=0;
		getline(cin,s);
		si=s.size();
		if(si>4 && s.substr(0,5)=="miao.") a=1;

	}

}
