#include<cstdio>
#include<cstring>

int n;
char s[101];
int main(){
	scanf("%d",&n),gets(s);
	for(int i=1,l,f,r;i<=n;++i){
		gets(s),l=strlen(s),f=0,r=0;
		if ((l>4)&&(s[l-1]=='.')&&(s[l-2]=='a')&&(s[l-3]=='l')&&(s[l-4]=='a')&&(s[l-5]=='l')) f=1;
		if ((l>4)&&(s[0]=='m')&&(s[1]=='i')&&(s[2]=='a')&&(s[3]=='o')&&(s[4]=='.')) r=1;
		puts(f^r?(f?"Freda's":"Rainbow's"):"OMG>.< I don't know!");
	}
	return 0;
}
