#include<iostream>
#include<string>
#include<cstdio>
using namespace std;
int main()
{
	int n; cin>>n;
	string freda=".alal";
	string rain="miao.";
	cin.ignore();
	while(n--) 
	{
		string in;
		getline(cin,in);
		bool f=true,r=true;
		for(int i = 0; i < 5; ++i)
		{
			if(in[i]!=rain[i]) r=false;
			if(in[in.size()-1-i]!=freda[i]) f=false;
		}

		if( (!f&&!r) || (f&&r) ) cout<<"OMG>.< I don't know!"<<endl;
		else if(f) cout<<"Freda's"<<endl;
		else cout<<"Rainbow's"<<endl;
	}
  return 0;
}
