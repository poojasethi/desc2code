#include <cassert>
#include <cstdio>
#include <cstring>

char S[101];
int c, d, n, w;

int main() {
	scanf("%d%c", &n, S);
	while (n--) {
		gets(S);
		d = strlen(S);
		if (d < 5) {
			puts("OMG>.< I don't know!");
			continue;
		}
		w = 0;
		c = S[5];
		S[5] = 0;
		if (strcmp(S, "miao.") == 0) w |= 1;
		S[5] = c;
		if (strcmp(S + d - 5, "lala.") == 0) w |= 2;
		if (w == 1) puts("Rainbow's");
		else if (w == 2) puts("Freda's");
		else puts("OMG>.< I don't know!");
	}
	return 0;
}
