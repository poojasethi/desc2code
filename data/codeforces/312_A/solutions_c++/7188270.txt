#include <iostream>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	cin.ignore();
	while (n--)
	{
		string s;
		getline(cin, s);
		int a = 0;
		if (s.substr(0, 5) == "miao.")
			a ^= 1;
		if (s.substr(max(0, (int)s.length() - 5), 5) == "lala.")
			a ^= 2;
		if (a == 1)
			cout << "Rainbow's" << endl;
		else if (a == 2)
			cout << "Freda's" << endl;
		else
			cout << "OMG>.< I don't know!" << endl;
	}
	return 0;
}
