#include      <iostream>
#include      <iomanip>
using namespace std;
int main(int argc, char **argv) {
	int n;
	cin >> n;
	string s;
	cin.ignore(100, '\n');
	while (n--) {
		getline(cin, s);
		bool f = s.substr(0, 5) == "miao.";
		bool r = s.substr(max(0, (int)s.length() - 5)) == "lala.";
		if (f && !r) cout << "Rainbow's" << endl;
		else if (!f && r) cout << "Freda's" << endl;
		else cout << "OMG>.< I don't know!" << endl;
	}
	return 0;
}
