#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

int main()
{
    int t;
    char s[101];
    bool a,b;
    cin>>t;
    getchar();
    while(t--)
    {
        gets(s);
        a = !strncmp(s,"miao.",5);
        b = !strncmp(s+strlen(s)-5,"lala.",5);
        if((a && b) || (!a && !b))
            cout<<"OMG>.< I don't know!"<<endl;
        else if(a || b)
            cout<<(a ? "Rainbow's" : "Freda's")<<endl;
    }
    return 0;
}
