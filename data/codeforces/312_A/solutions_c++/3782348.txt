#include<iostream>
#include<string>
using namespace std;

int main()
{
	string sent;
	int n;
	cin >> n;
	cin.clear();
	cin.ignore(1000, '\n');
	
	for(int i=0; i<n; i++)
	{
		getline(cin, sent);
		if(sent.find("miao.") == 0 && sent.find("lala.", sent.length()-5) == -1)
			cout << "Rainbow's" << endl;
		else if(sent.find("miao.") != 0 && sent.find("lala.", sent.length()-5) != -1)
			cout << "Freda's" << endl;
		else cout << "OMG>.< I don't know!" << endl;
	}
	return 0;
}
