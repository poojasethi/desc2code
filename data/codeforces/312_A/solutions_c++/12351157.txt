int main()
{
	int n;
	bool r,f;
	string s;
	cin>>n;
	getchar();
	while(n--)
	{
		getline(cin,s);
		r=f=0;
		if(s.rfind("lala.")==s.size()-5)
			f=1;
		if(s.find("miao.")==0)
			r=1;
		if((f && r) || (!f && !r))
			cout<<"OMG>.< I don't know!\n";
		else if(f)
			cout<<"Freda's\n";
		else if(r)
			cout<<"Rainbow's\n";
	}
	return 0;
