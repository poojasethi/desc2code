#include <iostream>
#include <string>

using namespace std;

int main(int argc, char **argv)
{
	int n;
	cin >> n;
	char s[105];
	cin.getline(s, 105);
	for (int i = 0; i < n; ++i)
	{
		cin.getline(s, 105);
		string str(s);
		bool miao = (str.find("miao.") == 0);
		bool lala = (str.rfind("lala.") == str.length() - 5);
		if (miao && !lala)
			cout << "Rainbow's\n";
		else if (lala && !miao)
			cout << "Freda's\n";
		else
			cout << "OMG>.< I don't know!\n";
	}
	return 0;
}
