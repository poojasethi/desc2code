#include <iostream>
#include <string>

using namespace std;

int main(){
    int n; string s;
    cin >> n; getline(cin, s);
    while (n--){
        getline(cin, s);
        bool p1 = s.size() > 4 && s.substr(0, 5) == "miao.",
             p2 = s.size() > 4 && s.substr(s.size() - 5, 5) == "lala.";
        if (p1 == p2) cout << "OMG>.< I don't know!";
        else if (p1) cout << "Rainbow's";
        else cout << "Freda's";
        cout << endl;
    }
}
