#define FI first
#define SE second
using namespace std;
typedef long long LL;
const int maxn=1e4+5;
int n,a[maxn],b[maxn];
pair <int,int> opa[maxn*20],opb[maxn*20];

void Swap(int *a,int x,int y,int &cnt,pair <int,int> *op)
{
    if(x==y) return;
    op[cnt++]=make_pair(x,y);
    op[cnt++]=make_pair(y,x);
    op[cnt++]=make_pair(x,y);
    swap(a[x],a[y]);
}

void Gauss(int *a,int &cnt,pair <int,int> *op)
{
    int r=0;
    for(int i=30;i>=0;--i)
    {
        for(int j=r;j<n;++j) if(a[j]&(1<<i))
        {
             Swap(a,r,j,cnt,op);
             for(int k=0;k<n;++k) if(k!=r&&(a[k]&(1<<i)))
             {
                 a[k]^=a[r];
                 op[cnt++]=make_pair(k,r);
             }
             ++r;
             break;
        }
    }
}

bool check(int x)
{
    int res=0;
    for(int i=0;i<n;++i)
    {
        if(!a[i]) break;
        int p;
        for(p=30;p>=0;--p) if(a[i]&(1<<p)) break;
        if(x&(1<<p)) res^=a[i];
    }
    return res==x;
}

int ca=0,cb=0;
void solve(int x)
{
    if(a[x]==b[x]) return;
    int v=a[x]^b[x];
    for(int i=0;i<n;++i)
    {
        if(!a[i]) break;
        int p;
        for(p=30;p>=0;--p) if(a[i]&(1<<p)) break;
        if(v&(1<<p)) opa[ca++]=make_pair(x,i);
    }
}

int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;++i) scanf("%d",a+i);
    for(int i=0;i<n;++i) scanf("%d",b+i);
    Gauss(a,ca,opa);
    Gauss(b,cb,opb);
    for(int i=0;i<n;++i)
    {
        if(!b[i]) break;
        if(!check(b[i])) return puts("-1");
    }
    for(int i=0;i<n;++i) solve(i);
    printf("%d\n",ca+cb);
    for(int i=0;i<ca;++i) printf("%d %d\n",opa[i].FI+1,opa[i].SE+1);
    for(int i=cb-1;i>=0;--i) printf("%d %d\n",opb[i].FI+1,opb[i].SE+1);
    return 0;
}
