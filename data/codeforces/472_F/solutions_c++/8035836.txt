#include <cstdio>
#include <cstdlib>
#include <vector>
#include <cassert>
using namespace std;
const int N=10010;
typedef pair<int ,int> PII;
vector<PII> opA,opB;
int a[N];
int b[N];
int n;
#define pb(x) push_back(x)
#define mp(x,y) make_pair(x,y)
void sw(int now,int idx,int a[],vector<PII>& op){
	//printf("sw %d %d\n",now,idx);
	swap(a[now],a[idx]);
	op.pb(mp(now,idx));
	op.pb(mp(idx,now));
	op.pb(mp(now,idx));
	return ;
}
int shift(int x,int s){return (x>>s)&1;}
void guass(int a[],vector<PII>& op){
	int now=0;
	for(int i=0;i<31;i++){
		int idx=-1;
		for(int j=now;j<n;j++){
			if(shift(a[j],i)==1){idx=j;break;}
		}
		if(idx==-1){continue;}
		else{
			if(now!=idx)sw(now,idx,a,op);
			for(int j=0;j<n;j++){
				if(j!=now && shift(a[j],i)==1){
					a[j]^=a[now];
					op.pb(mp(j,now));
				}
			}
			now++;
		}
	}
	return;
}
int next_bit(int x){
	assert(x!=0);
	int idx=0;
	while(true){
		if(x&1)return idx;
		x>>=1;idx++;
	}
	return idx;
}
void oper(vector<int> p ,vector<PII>& op,int top){
	//for(int i=0;i<p.size();i++)printf("p:%d ",p[i]);puts("");
	int now=p[0];
	int del=0;
	for(int i=1;i<p.size();i++){
		int idx=p[i];
		a[now]^=a[idx];op.pb(mp(now,idx));
	}
	for(int i=now;i>=top+1;i--)sw(i,i-1,a,op);
	return;
}
int main(){
	while(~scanf("%d",&n)){
		for(int i=0;i<n;i++){scanf("%d",&a[i]);}
		guass(a,opA);
		for(int i=0;i<n;i++){scanf("%d",&b[i]);}
		guass(b,opB);
		vector<PII> op;
		int b_m=-1;
		for(int i=0;i<n;i++){
			if(b[i]==0)break;
			else{
				b_m=i;
				vector<int> p;
				//printf("%d---\n",i);
				for(int j=i;j<n;j++){
					if(b[i]==0)break;
					if(a[j]==0)break;
					int nb=next_bit(b[i]),na=next_bit(a[j]);
					if(nb==na){
						p.pb(j);b[i]^=a[j];//printf("%d\n",b[i]);
					}
					else if(na>nb){puts("-1");return 0;}
				}
				if(b[i]!=0){puts("-1");return 0;}
				else oper(p,op,i);
			}
		}
		for(int i=b_m+1;i<n;i++){
			if(a[i]==0)break;
			a[i]^=a[i];op.pb(mp(i,i));
		}
		printf("%d\n",(int)opA.size()+(int)op.size()+(int)opB.size());
		for(int i=0;i<opA.size();i++)printf("%d %d\n",opA[i].first+1,opA[i].second+1);
		for(int i=0;i<op.size();i++)printf("%d %d\n",op[i].first+1,op[i].second+1);
		for(int i=opB.size()-1;i>=0;i--)printf("%d %d\n",opB[i].first+1,opB[i].second+1);
	}
}
