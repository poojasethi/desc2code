#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

vector<pair<int,int> > work1,work2;
int n,a[10005],b[10005];
inline void gauss1() {
	int now=1;
	for (int bit=30;bit>=0;bit--) {
		for (int i=now;i<=n;i++) if ((b[i]>>bit)&1) {
			if (i!=now) {
				work1.push_back(make_pair(now,i));
				b[now]^=b[i];
			}
			break;
		}
		if (!((b[now]>>bit)&1)) continue;
		for (int i=1;i<=n;i++) if (i!=now&&((b[i]>>bit)&1)) {
			work1.push_back(make_pair(i,now));
			b[i]^=b[now];
		}
		now++;
	}
}
inline void gauss2() {
	int now=1;
	for (int bit=30;bit>=0;bit--) {
		for (int i=now;i<=n;i++) if ((a[i]>>bit)&1) {
			if (i!=now) {
				work2.push_back(make_pair(now,i));
				a[now]^=a[i];
			}
			break;
		}
		if (!((a[now]>>bit)&1)) continue;
		for (int i=1;i<=n;i++) if (i!=now&&(((a[i]>>bit)&1)!=((b[i]>>bit)&1))) {
			work2.push_back(make_pair(i,now));
			a[i]^=a[now];
		}
		if (!((b[now]>>bit)&1)) {
			work2.push_back(make_pair(now,now));
			a[now]=0;
			continue;
		}
		now++;
	}
}
int main() {
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	for (int i=1;i<=n;i++) scanf("%d",&b[i]);
	gauss1();
	gauss2();
	for (int i=1;i<=n;i++) if (a[i]!=b[i]) {
		puts("-1"); return 0;
	}
	printf("%d\n",work1.size()+work2.size());
	for (int i=0;i<work2.size();i++) printf("%d %d\n",work2[i].first,work2[i].second);
	for (int i=work1.size()-1;i>=0;i--) printf("%d %d\n",work1[i].first,work1[i].second);
	return 0;
}
