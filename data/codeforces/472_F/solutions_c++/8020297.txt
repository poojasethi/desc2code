#include <cstdio>
#include <vector>
#include <utility>
#include <algorithm>
using namespace std;

int N;
int x[12345], y[12345];

vector<pair<int, int> > ans;
vector<pair<int, int> > ans2;

void as(int i, int j, vector<pair<int, int> >& ans)
{
    ans.push_back(make_pair(i + 1, j + 1));
}


void f(int x[], vector<pair<int, int> >& ans)
{
    for(int j = 1 << 30, k = 0; j && k < N; j >>= 1)
    {
        for(int i = k; i < N; i++)
            if(x[i] & j)
            {
                if(i != k)
                {
                    swap(x[i], x[k]);
                    as(i, k, ans);
                    as(k, i, ans);
                    as(i, k, ans);
                }
                break;
            }

        if(x[k] & j)
        {
            for(int i = 0; i < N; i++)
                if(i != k && (x[i] & j))
                {
                    x[i] ^= x[k];
                    as(i, k, ans);
                }
            k++;
        }
    }
}

int main()
{ 
    scanf("%d", &N);
    for(int i = 0; i < N; i++)
        scanf("%d", x + i);
    for(int i = 0; i < N; i++)
        scanf("%d", y + i);

    f(x, ans);
    f(y, ans2);

    bool ok = true;
    for(int i = 0; ok && i < N; i++)
        if(y[i])
        {
            int idx = -1;
            for(int j = 0; j < N; j++)
                if(((y[i] & x[j]) << 1) > y[i] && ((y[i] & x[j]) << 1) > x[j])
                    idx = j;

            for(int j = 0; j < N; j++)
                if(((y[i] & x[j]) << 1) > x[j])
                {
                    if(idx != j)
                    {
                        x[idx] ^= x[j];
                        as(idx, j, ans);
                        //x[j] ^= x[j];
                        //as(j, j, ans);
                    }
                }
            if(idx == -1 || x[idx] != y[i])
                ok = false;
            else if(idx != i)
            {
                swap(x[i], x[idx]);
                as(i, idx, ans);
                as(idx, i, ans);
                as(i, idx, ans);
            }
        }

    if(!ok)
        puts("-1");
    else
    {
        for(int i = 0; i < N; i++)
            if(x[i] != y[i])
                as(i, i, ans);

        printf("%d\n", (int)(ans.size() + ans2.size()));
        for(int i = 0; i < ans.size(); i++)
            printf("%d %d\n", ans[i].first, ans[i].second);
        for(int i = ans2.size() - 1; i >= 0; i--)
            printf("%d %d\n", ans2[i].first, ans2[i].second);
    }

    return 0;
}
