void swap(int i,int j,vector<long> &v)
{
    if(i==j)
        return;
    ops.pb(mp(i,j));
    ops.pb(mp(j,i));
    ops.pb(mp(i,j));
    swap(v[i],v[j]);
}
void xorr(int i,int j,vector<long> &v)
{
    v[i]^=v[j];
    ops.pb(mp(i,j));
}
void gauss(vector<long> &v)
{
    int row=0;
    ROF(j,30,0)
    {
        FOR(i,row,n-1)
        {
            if(v[i]&(1<<j))
            {
                swap(i,row,v);
                FOR(k,0,n-1)
                if(k!=row)
                    if(v[k]&(1<<j))
                        xorr(k,row,v);
                row++;
                break;
            }
        }
    }
}
