#include <iostream>
#include <string>
using namespace std;
int main ()
{
	int n, m, t;
	cin >> n;
	cin >> m;
	cin >> t;
	int c[m];
	int i, j;
	for(i=0;i<m;i++){
		c[i] = 0;
	}
	for(i=0;i<n;i++){
		string d;
		cin >> d;
		for(j=0;j<m;j++){
			if(d[j] == 'Y'){
				c[j]++;
			}
		}
	}
	int a = 0;
	for(i=0;i<m;i++){
		if(c[i] >= t){
			a++;
		}
	}
	//kitten
	cout << a;
	return 0;
}
