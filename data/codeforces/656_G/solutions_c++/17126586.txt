#include <stdio.h> //kitten
char a[20][20];
int temp[20];

int main()
{
    int n,m,t;
    scanf("%d %d %d",&n,&m,&t);
    for(int i=0;i<n;i++){
        scanf("%s",a[i]);
        for(int j = 0 ; j < m ; j++)
            if(a[i][j]=='Y')
                temp[j]++;
    }
    int ans=0;
    for(int i=0;i<m;i++)
        if(temp[i]>=t)
            ans++;
    printf("%d\n",ans);
    return 0;
}
