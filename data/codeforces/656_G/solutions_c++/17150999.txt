

#include <bits/stdc++.h>
#include <stdio.h>
using namespace std;
typedef long long ll;
ll POWER(ll e)
{
    ll a = 2;
    if(e == 0)return 1;
    for(int i = 1;i < e;i++)
    {
        a*=2;
        a %= 10000007;
    }
    return a;
}

int main()
{
    ll x,y,z,ans = 0;
    cin >> x >> y >> z;
    vector<string> a(x);
    for(int i = 0;i < x;i++)
    {
        cin >> a[i];
    }
    for(int i = 0;i < a[0].size();i++)
    {
        int c = 0;
        for(int j = 0;j < a.size();j++)
        {
            if(a[j][i] == 'Y')c++;
        }
        if(c >= z)ans++;
        //kitten
    }
    cout << ans <<endl;
    return 0;
}
