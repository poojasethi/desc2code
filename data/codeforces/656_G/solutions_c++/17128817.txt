#include <cstdio>
#define N 110
char s[N][N];
int main()
{
    int i,j,ans=0,tmp=0,n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    for(i=0;i<n;i++)
        scanf("%s",s[i]);
    for(j=0;j<m;j++)
    {
        tmp=0;
        for(i=0;i<n;i++)
            if(s[i][j]=='Y')
                tmp++;
        if(tmp>=k)
            ans++;
    }//kitten
    printf("%d\n",ans);
    return 0;
}
