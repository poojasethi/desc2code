#include<iostream>
#include<string>
#include<algorithm>
using namespace std;

int N,M;
int his[100], me[100];
int hisatk[100];
int hisdef[100];

int main(){
	cin>>N>>M;
	string s;
	int ind = 0, ind1 = 0;
	for (int i=0;i<N;i++){
		cin>>s>>his[i];
		his[i]*=2;
		if (s[0]=='D'){
			his[i]+=1;
			hisdef[ind1]=his[i];
			ind1++;
		}
		else{
			hisatk[ind] = his[i];
			ind++;
		}
	}
	for (int i=0;i<M;i++){
		cin>>me[i];
		me[i]*=2;
	}
	sort(his,his+N);
	sort(me,me+M);
	sort(hisatk,hisatk+ind);
	sort(hisdef,hisdef+ind1);
	bool can_beat_all = true;
	if (M<N) can_beat_all = false;
	for(int i=0;i<N;i++){
		if (can_beat_all == false) break;
		if (his[N-1-i]>me[M-1-i]){
			can_beat_all = false;
		}
	}
	int ans1=0;
	for (int i=0;i<M && i<ind && me[M-1-i]>hisatk[i];i++){
		ans1+=me[M-1-i]-hisatk[i];
	}
	int ans2=0;
	if (can_beat_all){
		bool used[M]; for (int i=0;i<M;i++) used[i] = false;
		for (int i=0;i<ind1;i++){
			int j;
			for (j=0;me[j]<=hisdef[i] || used[j];j++);
			used[j] = true;
		}
		for (int i=0;i<M;i++) if (!used[i]) ans2+=me[i];
		for (int i=0;i<ind;i++) ans2-=hisatk[i];
	}
	cout<<((ans1>ans2)?ans1/2:ans2/2);
		
}
