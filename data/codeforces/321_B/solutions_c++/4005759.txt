#include<stdio.h>
#include<algorithm>
#include<memory.h>
using namespace std;
	
int main(){
	char str[100][5];
	int cl[101],at[101],de[101],mem[101];
	int i,j,k,n,m,t,c1,c2,ans,tmp;
	while(scanf("%d%d",&n,&m)!=EOF){
		memset(mem,0,sizeof(mem));
		for(c1=c2=i=0;i<n;i++){
			scanf("%s",&str[i]);
			if(str[i][0]=='A') scanf("%d",&at[c1]),c1++;
			else scanf("%d",&de[c2]),c2++;
		}
		for(i=0;i<m;i++) scanf("%d",&cl[i]);
		sort(cl,cl+m);
		sort(at,at+c1);
		sort(de,de+c2);
		for(j=i=0;i<c2&&j<m;){
			for(;j<m;j++) if(cl[j]>de[i]){
				mem[j++]=1;i++;
				break;
			}
		}
		ans=0;
		if(i==c2){
			for(j=i=0;i<c1&&j<m;){
				for(;j<m;j++) if(!mem[j]&&cl[j]>=at[i]){
					ans+=cl[j]-at[i++];
					mem[j++]=2;
					break;
				}
			}
			if(i==c1) for(i=0;i<m;i++){
				if(!mem[i]) ans+=cl[i];
			}
		}
		for(j=0,i=m-1,tmp=0;i>=0&&j<c1;i--,j++){
			if(cl[i]>at[j]) tmp+=cl[i]-at[j];
			else break;
		}
		printf("%d\n",ans>tmp?ans:tmp);
	}
	return 0;
}
