#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdio>
#include <vector>
#define INF 1005

using namespace std;
char str[INF];
int A[INF], D[INF], N[INF], NN[INF], vis[INF];
bool cmp(int a, int b)
{
	return a>b;
}
int main()
{
	//freopen("1.txt","r",stdin);
	int i, j, n, m, x;
	int cnt1=0, cnt2=0, sum=0;
	scanf("%d%d", &n, &m);
	
	for (i=1; i<=n; i++)
	{
		scanf("%s%d", str, &x);
		if (str[0]=='A') A[++cnt1]=x;
		else D[++cnt2]=x;
	}
	for (i=1; i<=m; i++)
		scanf("%d", &N[i]), sum+=N[i];
		
	int ans=0 , sa=0, sn=0 ;
	
	sort(N+1, N+m+1, cmp);
	sort(D+1, D+cnt2+1, cmp);
	sort(A+1, A+cnt1+1);
	
	for (i=1; i<=min(m,cnt1); i++)
		sa+=A[i], sn+=N[i], ans=max(ans, sn-sa);
		
	sort(A+1, A+cnt1+1, cmp);
	
	for (i=m, j=cnt2; i && j; i--)
		if (N[i]>D[j])
		{
			j--; sum-=N[i]; vis[i]=1;
		}
	
	if (j) { cout << ans << endl; return 0; }
	
	for (i=m, j=cnt1; i && j; i--)
		if (vis[i]==0 && N[i]>=A[j])
		{
			sum-=A[j]; j--;
		}
		
	if (j) { cout << ans << endl; return 0; }
	ans=max(ans, sum);
	cout << ans<< endl;
	return 0;
}
