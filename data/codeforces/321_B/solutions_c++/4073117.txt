#include <stdio.h>
#include <algorithm>
#include <iostream>
using namespace std;
int min(int a,int b){return a<b?a:b;}
int max(int a,int b){return a>b?a:b;}
int main(){
	int n,m;
	int x;
	int u[105],a[105],d[105],an,dn;
	char s[100];
	scanf("%d%d",&n,&m);
	an=dn=0;
	for(int i=1;i<=n;i++){
		scanf("%s%d",&s,&x);
		if(s[0]=='A')a[an++]=x;
		else d[dn++]=x;
	}
	for(int i=0;i<m;i++)scanf("%d",&u[i]);
	sort(a,a+an);
	sort(d,d+dn);
	sort(u,u+m);
	//for(int i=0;i<m;i++)printf("**%d ",u[i]);
	
	
	int ans0=0;
	for(int i=0;i<min(m,an);i++){
		//printf("%d *\n",u[m-1-i]-a[i]);
		if(u[m-1-i]-a[i]>0)ans0+=u[m-1-i]-a[i];
		else break;
	}
	int ans=0;
	if(m>n){
		for(int i=dn-1;i>=0;i--){
			int ss=1;
			for(int j=0;j<m;j++){
				if(u[j]>d[i]){u[j]=0;ss=0;break;}
			}
			if(ss){printf("%d\n",ans0);return 0;}
		}
		for(int i=an-1;i>=0;i--){
			int ss=1;
			for(int j=m-1;j>=0;j--){
				if(u[j]==0)continue;
				if(u[j]>=a[i]){ans+=u[j]-a[i];u[j]=0;ss=0;break;}
			}
			if(ss){printf("%d\n",ans0);return 0;}
		}
		for(int i=0;i<m;i++)ans+=u[i];
	}
	printf("%d\n",max(ans0,ans));
	return 0;
}
