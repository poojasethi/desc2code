#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n,m,a[105],c[105],d[105],topa,topd;
bool f[105];
char s[5];
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
	{
		scanf("%s",s);
		if(s[0]=='A')
			scanf("%d",&a[++topa]);
		else
			scanf("%d",&d[++topd]);
	}
	for(int i=1;i<=m;i++)
		scanf("%d",&c[i]);
	sort(a+1,a+1+topa);
	sort(d+1,d+1+topd);
	sort(c+1,c+1+m);
	int ans1=0,ans2=0;
	for(int j=1;j<=topa;j++)
		if(a[j]<c[m-j+1])
			ans1+=c[m-j+1]-a[j];
		else
			break;
	int j=1;
	for(int i=1;i<=topd;i++)
	{
		while(j<=m&&d[i]>=c[j])j++;
		if(j>m)
		{
			printf("%d\n",ans1);
			return 0;
		}
		f[j++]=true;
	}
	j=1;
	for(int i=1;i<=topa;i++)
	{
		while(j<=m&&(a[i]>c[j]||f[j]))j++;
		if(j>m)
		{
			printf("%d\n",ans1);
			return 0;
		}
		ans2+=c[j]-a[i];
		f[j++]=true;
	}
	for(int i=1;i<=m;i++)
		if(!f[i])
			ans2+=c[i];
	printf("%d\n",max(ans1,ans2));
	return 0;
}
