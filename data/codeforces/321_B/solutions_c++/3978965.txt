#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int n, m;
	cin >> n >> m;
	vector<int> a, b, c;
	string s;
	int tmp;
	for (int i = 0; i < n; ++i) {
		cin >> s >> tmp;
		if (s == "ATK") a.push_back(tmp);
		else b.push_back(tmp);
	}
	for (int i = 0; i < m; ++i) {
		cin >> tmp;
		c.push_back(tmp);
	}
	sort (a.begin(), a.end());
	sort (b.begin(), b.end());
	sort (c.begin(), c.end());

	int ans = 0;
	int sz = min(a.size(), c.size());
	for (int i = 0, j = c.size()-1; i < sz; ++i, --j)
		if (a[i] <= c[j]) ans += c[j]-a[i];

	sz = c.size();
	for (int i = 0, j = 0; i < int(b.size()); ++i) {
		while (j < sz && c[j] <= b[i]) ++j;
		if (j == sz) {
			cout << ans;
			return 0;
		}
		else c[j] = 0;
	}

	int ans1 = 0;
	sort (c.begin(), c.end());
	for (int i = 0, j = 0; i < int(a.size()); ++i) {
		while(j < sz && c[j] < a[i]) ++j;
		if (j == sz) {
			cout << ans;
			return 0;
		}
		else c[j] -= a[i];
	}

	for (int i = 0; i < sz; ++i)
		ans1 += c[i];
	cout << max(ans,ans1);
	return 0;
}
