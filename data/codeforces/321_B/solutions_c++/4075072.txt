#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int c[110],jd[110],ja[110];

int dp[110][110],vis[110];

int main(){
	int n,m,i,j,ss,jdnum=0,janum=0;
	char kk[10];
	scanf("%d%d",&n,&m);
	memset(dp,0,sizeof(dp));
	for(i=1;i<=n;i++){
		scanf("%s%d",kk,&ss);
		if(kk[0]=='A'){
			janum++;
			ja[janum]=ss;
		} 
		else{
			jdnum++;
			jd[jdnum]=ss;
		}
	}
	for(i=1;i<=m;i++){
		scanf("%d",&c[i]);
	}
	sort(c+1,c+m+1);
	sort(jd+1,jd+1+jdnum);
	sort(ja+1,ja+1+janum);
	
	int ans1=0,ans2=0;
	int jaa,cc,jdd;
	jaa=1;cc=m;
	while(jaa<=janum&&cc>=1){
		if(c[cc]>ja[jaa]){
			ans1+=c[cc]-ja[jaa];
			jaa++;cc--;
		}
		else break;
	}
	int okd=1,ddd=0,oka=1,aaa=0;
	
	if(m>n){
		//破防 
		for(i=jdnum;i>0;i--){
			for(j=1;j<=m;j++){
				if(c[j]>jd[i]){
					c[j]=0;
					ddd++;
					break;
				}
			}
		}
		if(ddd!=jdnum) okd=0;
		
		if(okd){
			for(i=janum;i>0;i--){
				for(j=m;j>0;j--){
					if(c[j]&&c[j]>=ja[i]){
						ans2+=c[j]-ja[i];
						c[j]=0;
						aaa++;
						break;
					}
				}
			}
			if(aaa!=janum) oka=0;
			if(oka){
				for(i=1;i<=m;i++) ans2+=c[i];
			}
		}
	}
	printf("%d\n",max(ans1,ans2));
	
	return 0;
}
