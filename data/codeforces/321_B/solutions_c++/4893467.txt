#include <algorithm>
#include <stdio.h>
#define  maxn      100
using    namespace std;

int n,m;
int cnt1,cnt2;
int sum1,sum2;
int ar[maxn];
int atk[maxn];
int def[maxn];
bool mark[maxn];

int main()
{
  scanf("%d%d",&n,&m);
  for(int i=0 ; i<n ; i++)
  {
    char s[8];
    int x;
    scanf("%s%d",s,&x);
    if(s[0]=='A')  atk[cnt1++]=x,sum1+=x;
    else           def[cnt2++]=x;
  }
  for(int i=0 ; i<m ; i++)
    scanf("%d",&ar[i]);
  sort(ar,ar+m);
  sort(atk,atk+cnt1);
  sort(def,def+cnt2);
  int ans1=0;
  for(int i=0,j=m-1 ; j>=0 && i<cnt1 && ar[j]>=atk[i] ; i++,j--)
    ans1+=ar[j]-atk[i];
  int ans2=0,dst=0;
  for(int i=0,j=0 ; j<m && i<cnt2 ; i++,j++)
  {
    while(j<m && ar[j]<=def[i])                j++;
    if(j<m)                                    mark[j]=true,dst++;
  }
  for(int i=0,j=0 ; j<m && i<cnt1 ; i++,j++)
  {
    while(j<m && (ar[j]<atk[i] || mark[j]))    j++;
    if(j<m)                                    dst++;
  }
  if(dst==n)
  {
    for(int i=0 ; i<m ; i++)
      if(!mark[i])
        sum2+=ar[i];
    ans2=sum2-sum1;
  }
  printf("%d\n",max(ans1,ans2));
  return 0;
}
