#include <cstdio>
#include <cstring>
#include <algorithm>
#include <string>
#include <iostream>
using namespace std;

int n,m,totda,totdd,x,y;
int a[110],da[110],dd[110],vis[110];

int main()
{
	int i,j,tmp;
	char pos[10];
	scanf("%d%d",&n,&m);
	for(i=0,totda=0,totdd=0;i<n;i++)
	{
		scanf("%s%d",pos,&tmp);
		if(strcmp(pos,"DEF")==0) dd[totdd++]=tmp;
		else da[totda++]=tmp;
	}
	for(i=0;i<m;i++) scanf("%d",&a[i]);
	x=0;
	sort(a,a+m);
	sort(da,da+totda);
	sort(dd,dd+totdd);
	for(i=0,j=m-1;j>=0&&i<totda;i++,j--)
	{
		if(a[j]>=da[i]) x+=a[j]-da[i];
		else break;
	}
	memset(vis,0,sizeof(vis));
	y=0;
	for(i=0;i<totdd;i++)
	{
		for(j=0;j<m;j++)
		{
			if(!vis[j]&&a[j]>dd[i])
			{
				vis[j]=1;
				break;
			}
		}
		if(j==m) break;
	}
	if(i==totdd)
	{
		for(i=totda-1;i>=0;i--)
		{
			for(j=m-1;j>=0;j--)
			{
				if(!vis[j]&&a[j]>=da[i])
				{
					vis[j]=1;
					y+=a[j]-da[i];
					break;
				}
			}
			if(j<0) break;
		}
		if(i<0)
		{
			for(i=0;i<m;i++)
				if(!vis[i]) y+=a[i];
		}
	}
	printf("%d\n",(x>y)?x:y);
	return 0;
}
