#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
#include<string.h>
using namespace std;
int attack[105],def[105],a[105];
int main()
{
	int n,m,i,j,anum=0,dnum=0;
	char str[10];
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
	{
		int a;
		scanf("%s%d",str,&a);
		if(strcmp(str,"ATK")==0)attack[++anum]=a;
		else def[++dnum]=a;
	}
	for(i=1;i<=m;i++)
	scanf("%d",&a[i]);
	sort(attack+1,attack+1+anum);
	sort(def+1,def+1+dnum);
	sort(a+1,a+1+m);
	int sum1=0;
	for(i=1;i<=anum&&i<=m;i++)
	if(a[m-i+1]>attack[i])sum1+=a[m-i+1]-attack[i];
	int sum=0;
	for(i=1;i<=dnum;i++)
	{
		for(j=1;j<=m;j++)if(a[j]>def[i])break;
		if(j>m){sum=-1;break;}
		a[j]=-1;
	}
	//printf("%d %d\n",anum,dnum);
	if(sum==0)
	{
		int t=0;
		for(i=anum;i>=1;i--)
		{
			for(j=m;j>=1;j--)
			if(a[j]>=attack[i])
			{
			     sum+=a[j]-attack[i];
			     t++;
			     a[j]=-1;
			     break;
			}
		}
		//for(i=1;i<=m;i++)
		//printf("%d\n",a[i]);
		if(t==anum)
		{
			for(j=1;j<=m;j++)
			if(a[j]!=-1)sum+=a[j];
		}
	}
	sum=sum>sum1?sum:sum1;
	printf("%d\n",sum);
}
