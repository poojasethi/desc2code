#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int atk[105],def[105],ciel[105];
bool vis[105];
int n,m,m1,m2;
int ffind(int x){
	for(int i=0;i<n;i++)
	if(vis[i]&&ciel[i]>=x){
		vis[i]=false;
		return i;	
	}
	return -1;
}
int a(){
	if(m2>n)return 0;
	for(int i=0;i<m2;i++)
	if(ffind(def[i]+1)==-1)
	return 0;
	int res=0;
	for(int i=0;i<m1;i++){
		int k=ffind(atk[i]);
		if(k==-1)return 0;
		res+=ciel[k]-atk[i];
	}
	for(int i=0;i<n;i++)
	if(vis[i])res+=ciel[i];
	return res;
}
int p(){
	int res=0;
	for(int i=0;i<m1&&i<n;i++)
	if(ciel[n-i-1]>atk[i])
	res+=ciel[n-i-1]-atk[i];
	return res;
}
int main(){
	scanf("%d%d",&m,&n);
	char s[3];
	for(int i=0;i<m;i++){
		scanf("%s",s);
		if(s[0]=='A')scanf("%d",&atk[m1++]);
		else scanf("%d",&def[m2++]);
	}
	for(int i=0;i<n;i++){
		scanf("%d",ciel+i);
		vis[i]=true;
	}
	sort(atk,atk+m1);
	sort(ciel,ciel+n);
	sort(def,def+m2);
	printf("%d\n",max(a(),p()));
}
