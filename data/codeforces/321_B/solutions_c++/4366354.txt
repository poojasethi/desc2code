#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,s1,s2,an;
int a1[110],a2[110],a[110],f[110],ff[110];
char p[5];
inline void work1() {
	int i,s=0,ma=0,k=min(s1,m);
	For(i,1,k) s+=a[m-i+1]-a1[i],ma=max(ma,s);
	an=ma;
}
inline void work2() {
	int i,j=1,s=0;
	For(i,1,s2) {
		for (;j<=m&&a[j]<=a2[i];j++);
		if (j>m) break;
		ff[j]=1; j++;
	}
	if (j>m) return;
	j=1;
	For(i,1,s1) {
		for (;j<=m&&(a[j]<a1[i]||ff[j]);j++);
		if (j>m) break;
		s+=a[j]-a1[i]; ff[j]=1; j++;
	}
	if (i>s1) For(i,1,m) if (!ff[i]) s+=a[i];
	an=max(an,s);
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n) {
		scanf("%s",p);
		if (p[0]=='A') scanf("%d",&a1[++s1]);
		else scanf("%d",&a2[++s2]);
	}
	For(i,1,m) scanf("%d",&a[i]);
	sort(a+1,a+m+1);
	sort(a1+1,a1+s1+1);
	sort(a2+1,a2+s2+1);
	work1();
	work2();
	printf("%d\n",an);
	//for(;;);
	return 0;
}
