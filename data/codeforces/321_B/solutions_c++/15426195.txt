#include<bits/stdc++.h>

using namespace std;

#define LL long long

int p[100],a[100],d[100];
bool v[100];

int main()
{
    int x,n,m,i,an,dn,j;
    string str;

    cin>>n>>m;
    an=dn=0;
    for(i=0;i<n;i++)
    {
        cin>>str>>x;
        if(str[0]=='A') a[an++] = x;
        else    d[dn++] = x;
    }
    for(i=0;i<m;i++)
        cin>>p[i];

    sort(p,p+m);
    sort(a,a+an);
    sort(d,d+dn);

    i = m-1, j = 0;
    int l = 0;
    while(i>-1 && j<an)
    {
        if(p[i]<a[j])
            break;
        l+=p[i];
        l-=a[j];
        i--;
        j++;
    }

    int r = 0;
    i = j = 0;

    while(i<m && j<dn)
    {
        if(p[i]>d[j])
        {
            v[i] = 1;
            j++;
        }
        i++;
    }

    if(j==dn)
    {
        i = j = 0;
        while(i<m)
        {
            if(!v[i])
            {
                if(j<an && p[i]>=a[j])
                {
                    r-=a[j];
                    j++;
                }
                r+=p[i];
            }
            i++;
        }
        if(j!=an)
            r = 0;
    }
    cout<<max(l,r);
    return 0;
}
