#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define N 1005
#define rep(i,n) for(int i=1;i<=n;i++)
using namespace std;
int n,m;
int stk1[N],stk2[N],g[N],tot1,tot2,use[N];
char ch[N];
int main(){
	scanf("%d%d",&n,&m);
	rep(i,n){
		scanf("%s",ch);
		if(ch[0]=='A') scanf("%d",&stk1[++tot1]);
		else scanf("%d",&stk2[++tot2]);
	}
	rep(i,m) scanf("%d",&g[i]);
	sort(stk1+1,stk1+1+tot1);
	sort(stk2+1,stk2+1+tot2);
	sort(g+1,g+1+m);
	int now=1;
	int ans=0;
	rep(i,m) ans+=g[i];
	rep(i,tot2){
		while(g[now]<=stk2[i]&&now<=m) now++;
		if(now>m){ans=0; break;}
		ans-=g[now]; use[now]=1; now++;
	}
	if(ans)
	rep(i,tot1){
		while((g[now]<stk1[i]||use[now])&&now<=m) now++;
		if(now>m){ans=0; break;}
		ans-=stk1[i];
	}
	ans=max(ans,0);
	
	
	for(int i=1;i<=tot1&&i<=m;i++){
		int ret=0;
		int k=1;
		for(int j=m-i+1;j<=m;j++){
			if(g[j]<stk1[k]){ret=0; break;}
			else ret+=g[j]-stk1[k];
			k++;
		}
		ans=max(ans,ret);
	}
	printf("%d\n",ans);
}
