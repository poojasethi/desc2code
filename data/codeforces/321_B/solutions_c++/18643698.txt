#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>

using namespace std;

const int N=105;

int na,nd,n,m,ans0,ans1;
int a1[N],d1[N],a[N];
bool use[N];

int main(){
	
	scanf("%d%d",&m,&n);
	na=nd=0;
	for (int i=1; i<=m; i++){
		char str[10]; int val;
		scanf("%s%d",str,&val);
		if (str[0]=='A') a1[++na]=val; else d1[++nd]=val;
	}
	for (int i=1; i<=n; i++) scanf("%d",&a[i]);
	if (na) sort(a1+1,a1+na+1);
	if (nd) sort(d1+1,d1+nd+1);
	if (n) sort(a+1,a+n+1);
	
	//just atk
	ans0=0;
	for (int i=n, j=1; (i>=1) && (j<=na) && (a[i]>=a1[j]); i--,j++)
		ans0+=a[i]-a1[j];
	//all kill
	ans1=0;
	int i,j;
	memset(use,0,sizeof(use));
	for (i=1, j=1; (i<=nd) && (j<=n); i++){
		while ( (j<=n) && ((a[j]<=d1[i]) || (use[j])) ) j++;
		if (j>n) break;
		use[j]=1; 
	}
	if (j<=n) { //all atk have been killed
		for (i=1, j=1; (i<=na) && (j<=n); i++){
			while ( (j<=n) && ((use[j]) || (a[j]<a1[i])) ) j++;
			if (j>n) break;
			use[j]=1; ans1+=a[j]-a1[i];
		}
		if (j<=n){
			for (i=1; i<=n; i++)
				if (!use[i]) ans1+=a[i];
		} else ans1=0;
	}
	///
	printf("%d\n",max(ans0,ans1));
	
	return 0;
}
