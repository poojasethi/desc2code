#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int main(){
	int n, m;
	cin >> n >> m;
	string c;
	int p, a[101], d[101], s = 0, v = 0;
	for (int i = 0; i < n; i++) {
		cin >> c >> p;
		if (c == "ATK")
			a[s++] = p;
		else
			d[v++] = p;
	}
	int b[101];
	for (int i = 0; i < m; i++) 
		cin >> b[i];
	sort(a, a+s); sort(b, b+m); sort(d, d+v);
	int r = 0;
	for (int i = 0; i < s && i < m; i++) {
		if (a[i] <= b[m-1-i])
			r += b[m-1-i] - a[i];
		else break;
	}
	int q = 0;
	bool u[101] = {0};
	for (int i = 0, j = 0; i < v; i++) {
		while (j < m && b[j] <= d[i]) j++;
		if (j == m) {
			cout << r;
			return 0;
		}
		u[j] = true;
		j++;
	}
	for (int i = 0, j = 0; i < s; i++) {
		while (j < m && (b[j] < a[i] || u[j]) ) j++;
		if (j == m) {
			cout << r;
			return 0;
		}
		q += b[j] - a[i];
		u[j] = true;
	}	
	for (int i = 0; i < m; i++)
		if (!u[i])
			q += b[i];
	cout << max(r, q);
	return 0;
}
