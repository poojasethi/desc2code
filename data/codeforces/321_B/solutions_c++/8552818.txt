#include <bits/stdc++.h>
#define	rep(i,j,k)	for (int i = j; i <= k; i++)
using namespace std;
typedef long long ll;
const int N = 1010;
bool u[N];
char s[10];
int n1, n2, m, ans, a[N], b[N], c[N];
int work()
{
	int j = 1, tmp = 0;
	rep(i,1,n2)
	{
		while ((c[j]<=b[i]||u[j]) && j<=m)	j++;
		if (j > m)	return 0;
		u[j] = 1;
	}
	j = 1;
	rep(i,1,n1)
	{
		while ((u[j]||c[j]<a[i]) && j<=m)	j++;
		if (j > m)	return 0;
		u[j] = 1, tmp += c[j]-a[i];
	}
	rep(i,1,m)	if (!u[i])	tmp += c[i];
	return tmp;
}
int main()
{
	cin >> ans >> m;
	rep(i,1,ans)
	{
		scanf("%s", s);
		if (s[0] == 'A')	scanf("%d", &a[++n1]);
		else	scanf("%d", &b[++n2]);
	}
	rep(i,1,m)	scanf("%d", &c[i]);
	sort(c+1,c+1+m);
	sort(a+1,a+1+n1);
	sort(b+1,b+1+n2);
	int k = min(n1,m);	ans = 0;
	rep(i,1,k)	ans += max(0,c[m+1-i]-a[i]);
	cout << max(ans,work()) << endl;
}
