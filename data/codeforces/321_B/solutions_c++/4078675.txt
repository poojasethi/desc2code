#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int max(int a,int b)
{
	return a>b?a:b;
}
int a[150],d[150],f[150],F[150],pan[150];
int main()
{
	int n,m,x;
	char s[10];
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		int an=0,dn=0;
		for(int i=1;i<=n;i++)
		{
			scanf("%s%d",s,&x);
			if(s[0]=='A')a[an++]=x;
			else d[dn++]=x;
		}
		for(int i=0;i<m;i++)
			scanf("%d",&f[i]);
		sort(a,a+an);
		sort(d,d+dn);
		sort(f,f+m);
		int ans=0;
		for(int i=0;i<(m<an?m:an);i++)
			if(f[m-1-i]-a[i]>0)ans+=f[m-1-i]-a[i];
			else break;
		int Ans=0;
		if(m>n)
		{
	 		for(int i=dn-1;i>=0;i--)
	 		{
		 		int fg=1;
		 		for(int j=0;j<m;j++)
			 		if(f[j]>d[i]){f[j]=0;fg=0;break;}
		 		if(fg){printf("%d\n",ans);return 0;}
		 	}
		 	for(int i=an-1;i>=0;i--)
		 	{
	 			int fg=1;
	 			for(int j=m-1;j>=0;j--)
	 			{
			 		if(f[j]==0)continue;
			 		if(f[j]>=a[i]){Ans+=f[j]-a[i];f[j]=0;fg=0;break;}
			 	}
			 	if(fg){printf("%d\n",ans);return 0;}
	 		}
	 		for(int i=0;i<m;i++)Ans+=f[i];
		}
		printf("%d\n",max(Ans,ans));
	}
}
