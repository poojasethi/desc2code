#include<iostream>
#include<cstdio>
#include<cstring>
#include<set>
#include<algorithm>
using namespace std;
int n,m;
char tmp[10];
int s1,s2;
int a1[12000],a2[12000];
int cd[12000];
multiset<int> s;
int main(){
	scanf("%d%d",&n,&m);
	for (int i=0;i<n;i++){
		scanf(" ");
		scanf("%s",tmp);
		if (tmp[0]=='A')
			scanf("%d",&a1[s1++]);
		else
			scanf("%d",&a2[s2++]);
	}
	for (int i=0;i<m;i++)
		scanf("%d",&cd[i]);
	sort(a1,a1+s1);
	sort(cd,cd+m);
	int ans=0;
	for (int i=m-1,j=0;i>=0 && j<s1;i--,j++)
		if (a1[j]<cd[i]){
			ans+=cd[i]-a1[j];
		}
		else break;
	for (int i=0;i<m;i++)
		s.insert(cd[i]);
	for (int i=0;i<s2;i++){
		multiset<int>::iterator it=s.upper_bound(a2[i]);
		if (it==s.end()) {printf("%d\n",ans);return 0;}
		else s.erase(it);
	}
	int ans2=0;
	for (multiset<int>::iterator it=s.begin();it!=s.end();it++){
		ans2+=*it;
	}
	for (int i=0;i<s1;i++){
		multiset<int>::iterator it=s.lower_bound(a1[i]);
		if (it==s.end()) {printf("%d\n",ans);return 0;}
		else s.erase(it);
		ans2-=a1[i];
	}
	ans=max(ans,ans2);
	printf("%d\n",ans);
}
