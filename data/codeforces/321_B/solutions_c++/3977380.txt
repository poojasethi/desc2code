#include <cstdio>
#include <cstring>
#include <iostream>
#include <cstdlib>
#include <algorithm>
using namespace std;

typedef long long LL;

int n,m;
int bb[105],blen,cc[105],clen,aa[105],dd[105];
char ch[5];
int Calc1()
{
	int i,j,res=0;
	for (i=1,j=m;i<=blen && j;++i,--j)
	{
		if (aa[j] <= bb[i]) break;
		res+=aa[j]-bb[i];
	}
	return res;
}
int Calc2()
{
	int i,j;
	memset(dd,0,sizeof(dd));
	for (i=1;i<=clen;++i)
	{
		for (j=1;j<=m;++j)
			if (dd[j] == 0 && aa[j] > cc[i])
			{
				dd[j]=1;
				break;
			}
		if (j > m) return 0;
	}
	int res=0;
	for (i=1;i<=m;++i)
		if (dd[i] == 0) res+=aa[i];
	for (i=1;i<=blen;++i)
	{
		for (j=1;j<=m;++j)
			if (dd[j] == 0 && aa[j] >= bb[i])
			{
				dd[j]=1;
				break;
			}
		res-=bb[i];
		if (j > m) return 0;
	}
	return res;
}
int main()
{
	cin>>n>>m;
	int i,k;
	blen=clen=0;
	for (i=1;i<=n;++i)
	{
		scanf("%s %d",ch,&k);
		if (ch[0] == 'A') bb[++blen]=k; else cc[++clen]=k;
	}
	for (i=1;i<=m;++i)
		scanf("%d",&aa[i]);
	sort(aa+1,aa+m+1);
	sort(bb+1,bb+blen+1);
	sort(cc+1,cc+clen+1);
	int Ans=Calc1(),Tmp=Calc2();
	if (Tmp > Ans) Ans=Tmp;
	cout<<Ans<<endl;
	return 0;
}
