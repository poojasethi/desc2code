#include<string>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;

const int N=110;
int n,m;
int ciel[N];
bool sign[N];
int att[N],def[N],cntatt,cntdef;
char s[10];
int main(){

	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%s",s);
		if (s[0]=='A') scanf("%d",&att[++cntatt]);
		else scanf("%d",&def[++cntdef]);
	}
	for(int i=1;i<=m;i++) scanf("%d",&ciel[i]);
	sort(ciel+1,ciel+m+1);
	sort(att+1,att+cntatt+1);
	sort(def+1,def+cntdef+1);
	int ans=0;
	for(int i=1;i<=min(cntatt,m);i++){
		if (att[i]>=ciel[m-i+1]) break;
		ans+=ciel[m-i+1]-att[i];
	}
	int pos=1;
	for(int i=1;i<=cntdef;i++){
		while (ciel[pos]<=def[i] && pos<=m) pos++;
		if (pos>m) break;
		sign[pos++]=1;
	}
	if (pos>m){
		printf("%d\n",ans);
		return 0;
	}
	pos=m;
	bool yes=1;
	for(int i=cntatt;i;i--){
		while((sign[pos] || ciel[pos]<att[i]) && pos) pos--;
		if (!pos) yes=0;
		pos--;
	}
	if (yes){
		int sum1=0,sum2=0;
		for(int i=1;i<=cntatt;i++) sum1+=att[i];
		for(int i=1;i<=m;i++) if (!sign[i]) sum2+=ciel[i];
		ans=max(ans,sum2-sum1);
	}
	printf("%d\n",ans);
	return 0;
}
