R=lambda:raw_input().split()
n,m=map(int,R())
a=[]
d=[]
for _ in range(n):
  t,x=R()
  [d,a][t<'B']+=[int(x)]
a.sort()
c=sorted(input() for _ in range(m))[::-1]
w=sum(max(0,x-y) for x,y in zip(c,a))
try:
  for x in d:c.remove(min(y for y in c if y>x))
  t=sum(c)-sum(a)
  for x in a[::-1]:c.remove(min(y for y in c if y>=x))
  w=max(w,t)
except:pass
print w
