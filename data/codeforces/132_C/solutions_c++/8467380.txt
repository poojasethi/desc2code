#include<bits/stdc++.h>
using namespace std;
string arr;
int arrsize, k;
int dp[102][202][2][52];
int toop=0;
void rc(int s, int top, int yon, int n){
    if(s>arrsize or n>k) return;
    if(s==arrsize and n==k)
        toop=max(toop,abs(top));
    if(dp[s][top][yon][n]) return;
    dp[s][top][yon][n]=1;
    rc(s+1,top,!yon,n+(arr[s]!='T'));
    rc(s+1,top+((yon==1)?1:-1),yon,n+(arr[s]!='F'));
    rc(s,top,yon,n+2);
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);memset(dp,0,sizeof dp);
    cin >> arr >> k;arrsize=arr.size();
    rc(0,0,1,0); cout << toop << '\n';
}
