#include <cstdio>
#include <cstring>
#include <iostream>

#define abs(a) ( (a)<0 ? ((a)*-1) : (a) )

using namespace std;

const int MAXK=55;
const int MAXN=111;

int N,K,res;

char s[MAXN];

bool used[MAXN][MAXK][2][MAXN*2];

inline void f( int pl , int cur , int dir , int dist ){
	
	if( pl>N+1 || cur>K )
		return;
	
	if( pl==N+1 && cur==K )
		res=max(res,abs(dist));
	
	if( used[pl][cur][dir][dist] )
		return;
	used[pl][cur][dir][dist]=1;
	
	f(pl+1,cur+(s[pl]=='T'),dir,(dir==1) ? dist+1 : dist-1);
	f(pl+1,cur+(s[pl]=='F'),( (dir==1) ? 0 : 1 ),dist);
	f(pl,cur+2,dir,dist);
}

int main(){
	
	scanf(" %s %d",s+1,&K);
	N=strlen(s+1);
	
	f(1,0,1,0);
	printf("%d\n",res);
	return 0;
}
