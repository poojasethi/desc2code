#include<bits/stdc++.h>
#include<cstring>
using namespace std;
char line[105];
int len,n,dp[2][105<<1][105][52];
int dfs(bool goog,int pos,int i,int nd)
{
    if(nd<0)return 0;
    if(!line[i])return nd>0?0:abs(pos);
    int &re=dp[goog][pos+100][i][nd];
    if(re!=-1)return re;
    int mv=goog?1:-1;
    return re=max(dfs(goog,pos+mv,i+1,nd-(line[i]!='F')),dfs(!goog,pos,i+1,nd-(line[i]!='T')));
}
main()
{
    int ans=0;
    scanf("%s%d",line,&n);
    memset(dp,-1,sizeof(dp));
    for(;n>=0;n-=2)ans=max(ans,dfs(1,0,0,n));
    printf("%d\n",ans);
    return 0;
}
