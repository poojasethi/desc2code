#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int m, n, add[] = {+1, -1}, f[111][55][2];
char c[111];

int main() {
    scanf("%s%d",c+1,&n); m = strlen(c+1);
    memset(f, 0x80, sizeof f); f[0][0][0] = f[0][0][1] = 0;
    for(int i = 1; i <= m; ++i)
        for(int j = 0; j <= n; ++j)
            for(int k = 0; k < 2; ++k)
                for(int t = 0; t <= j; ++t)
                    if((t % 2 == 0) != (c[i] == 'F'))
                        f[i][j][k] = max(f[i][j][k], f[i-1][j-t][!k]);
                    else f[i][j][k] = max(f[i][j][k], f[i-1][j-t][k] + add[k]);
    printf("%d\n", max(f[m][n][0], f[m][n][1]));
    return 0;
}
