#include<cstdio>
#include<cstring>
#define rep(i,n) for (int i=0;i<n;i++)
#define inf 0x3f3f3f3f
#define upd(x,y) if (y>x) x=y
char s[105]; int n,m,ans,f[105][55][2];//dp to command number|changes used|facing direction
int main()
{
    scanf("%s%d",s,&m),n=strlen(s); memset(f,-inf,sizeof(f)),f[0][0][0]=f[0][0][1]=0;
    rep(i,n) rep(j,m+1) rep(k,2){//j=use 0 to m change
        upd(f[i+1][j+(s[i]!='F')][k],f[i][j][k]+(k?1:-1));
        upd(f[i+1][j+(s[i]!='T')][!k],f[i][j][k]);//用了轉了方向，最遠距離一樣, | 不用，多走一step:
    }
    for (int i=m;i>=0;i-=2){upd(ans,f[n][i][0]); upd(ans,f[n][i][1]);}
    printf("%d\n",ans); return 0;
}
