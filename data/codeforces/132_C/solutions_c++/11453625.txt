#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cstdlib>
#include <cmath>
#define For(i,x,y) for(int i=x;i<=y;i++)
using namespace std;

bool f[101][51][201][2];
char s[100005];
int n,k;
int a[2]={-1,1};

int main(){
	scanf("%s",s+1);
	n=strlen(s+1);
	scanf("%d",&k);
	int c=min(n,k);
	f[0][0][100][1]=1;
	For(i,0,n-1)
	For(j,0,c)
	For(u,0,200)
	For(l,0,1)
	if(f[i][j][u][l]){
		char z=s[i+1];
		if(z=='T'){
			f[i+1][j][u][l^1]=1;
			if(j+1<=c) f[i+1][j+1][u+a[l]][l]=1;
		}else{
			f[i+1][j][u+a[l]][l]=1;
			if(j+1<=c) f[i+1][j+1][u][l^1]=1;
		}
	}
	int ans=0;
	For(j,0,c)
	For(u,0,200)
	For(l,0,1)
	if(f[n][j][u][l] && ((k-j)%2==0)){
		ans=max(ans,abs(u-100));
	}
	printf("%d\n",ans);
	return 0;
}
