#include<bits/stdc++.h>

using namespace std;

int M[102][202][51][2],n,N=102;

char s[102];

int f(int i, int t, int n, int h){

	if(n < 0)
		return 0;

	if(i == strlen(s+1)+1){
		if(n&1)
			return 0;
		else
			return abs(t);
	}

	int &r = M[i][t+N][n][h];
	
	if(r != -1)
		return r;
	
	if(s[i] == 'F')
		r = max(f(i+1,t+h,n,h) , f(i+1,t,n-1,-h));
	
	else
		r = max(f(i+1,t,n,-h) , f(i+1,t+h,n-1,h));
	
	return r;
}

int main(){

	cin >> s + 1 >> n;

	memset(M , -1 , sizeof M);
	cout << f(1,0,n,1);

	return 0;
}
