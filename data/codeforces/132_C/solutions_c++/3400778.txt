#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>

using namespace std;

const int MAXN = 105;

int N,K,res;
char ar[MAXN];
int dp[MAXN][MAXN][2][MAXN];
bool used[MAXN][MAXN/2][2][MAXN*2];

inline void f( int pos, int cur, int dir, int dist )
{
	if( used[pos][cur][dir][dist+100] )
		return;
		
	used[pos][cur][dir][dist+100]=1;
	
	if( pos > N+1 || cur > K )
		return;
	
	if( pos == N+1 && cur == K ){
		res = max( res, abs(dist) );
		return;
	}
	
	f( pos+1, cur+(ar[pos]=='T'), dir, (dir==1) ? dist+1 : dist-1 );
	f( pos+1, cur+(ar[pos]=='F'), !dir, dist);
	f( pos, cur+2, dir, dist );
}

int main()
{
	scanf(" %s",ar+1);
	N=strlen(ar+1);
	
	cin >> K;
	
	f( 1, 0, 1, 0 );
	
	cout << res << endl;
	
	return 0;
}
