#include <bits/stdc++.h>

using namespace std;

string cad;
int mem[111][333][55][3];
const int inicio = 150;

int solve ( int i, int x, int pend, int dir ) {
	if ( i == cad.size() ) return pend ? 0 : abs ( x - inicio );

	int &r = mem[i][x][pend][dir + 1];

	if ( r != -1 ) return r;

	int mov = cad[i] == 'F';
	r = 0;

	for ( int j  = 0; j <= pend; j ++ ) {
		if ( mov )
			r = max ( r, solve ( i + 1, x + dir, pend - j, dir ) );
		else
			r = max ( r, solve ( i + 1, x, pend - j, dir*-1 ) );
		mov = 1 - mov;
	}

	return r;
}

int main() {
   
	int n;	

	cin >> cad >> n;

	memset ( mem, -1, sizeof ( mem ) );

	cout << solve ( 0, inicio, n, 1 ) << '\n';

	return 0;
}
