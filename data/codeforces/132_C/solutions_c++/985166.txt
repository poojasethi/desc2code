#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

char s[1005];
int dp[105][52][205][2];

int main(void) {
	int n;
	scanf("%s%d", s+1, &n);
	memset(dp, 0, sizeof(dp));
	dp[0][0][100][0] = 1;
	for(int i=1;s[i];i++) {
		for(int j=0;j<=n;j++)
			for(int k=1;k<200;k++)
				for(int l=0;l<2;l++) {
					int v=0;
					if((v=dp[i-1][j][k][l]) > 0) {
						dp[i][j+(s[i]!='F')][k+(l==0)-(l==1)][l] = 1;
						dp[i][j+(s[i]!='T')][k][l^1] = 1;
					}
				}
	}
	int L = strlen(s+1);
	int ans=0;
	for(int j=n%2;j<=n;j+=2)
		for(int k=0;k<=200;k++)
			for(int l=0;l<2;l++)
				if(dp[L][j][k][l])
					ans = max(ans, abs(k-100));
	printf("%d\n", ans);
	return 0;
}
