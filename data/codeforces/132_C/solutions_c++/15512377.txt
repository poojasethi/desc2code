#include <bits/stdc++.h>
#define MAX_N 106 
#define MAX_M 55 
#define oo -1e9 
#define FOR(i,a,b) for (int i = a; i <= b; i++) 

using namespace std; 

string s; 
int m,n; 
int dp[MAX_N][MAX_M][2]; 

int main () 
	{
		cin >> s; 
		cin >> m;
		n = s.size(); 
		
		memset(dp,200,sizeof(dp)); 
		dp[0][0][0] = dp[0][0][1] = 0; 
		FOR(i,0,n-1) 
			FOR(j,0,m)
				FOR(k,0,1)
					{
						dp[i+1][j+(s[i] != 'F')][k] = max(dp[i+1][j+(s[i] != 'F')][k],dp[i][j][k] + (k?1:-1));
						dp[i+1][j+(s[i] != 'T')][!k] = max(dp[i+1][j+(s[i] != 'T')][!k],dp[i][j][k]);
					}
					
		int res = oo; 
		for (int i = m; i >= 0; i-= 2) res = max(res,max(dp[n][i][0],dp[n][i][1]));
		cout << res; 

		return 0; 
	}
