#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int i,j,k,l,n,m,x,y,ans,f[110][51][110][2];
char c[120];
int main(){
	scanf("%s%d",c+1,&m);
	n=strlen(c+1);
	f[1][0][0][1]=1;
	for(i=1;i<=n;++i)
		for(j=0;j<=m;++j)
			for(k=0;k<=n;++k)
				for(l=0;l<=1;++l)if(f[i][j][k][l]){
					int d;
					if(l==1)d=1;
					else d=-1;
					if(c[i]=='T'){
						f[i+1][j][k][1-l]=1;
						x=k+d; y=l;
						if(x<0)x=-x,y=1-y;
						f[i+1][j+1][x][y]=1;
					}else{
						x=k+d; y=l;
						if(x<0)x=-x,y=1-y;
						f[i+1][j][x][y]=1;
						f[i+1][j+1][k][1-l]=1;
					}
				}
	for(i=m;i>=0;i-=2)
		for(j=1;j<=n;++j)if(f[n+1][i][j][0] || f[n+1][i][j][1])
			ans=max(ans,j);
	printf("%d\n",ans);
}
