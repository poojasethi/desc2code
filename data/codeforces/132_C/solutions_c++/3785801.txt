#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 210
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,i1,an;
int f[N][N][N][2];
char p[1010];
int main() {
    scanf("%s%d",p+1,&m); n=strlen(p+1);
    f[0][100][0][1]=1;
    For(i1,0,n)For(i,0,N-1)For(j,0,m)For(k,0,1) if (f[i1][i][j][k]) {
        p[i1+1]=='F'?f[i1+1][i+(k?1:-1)][j][k]=1:f[i1+1][i][j][k^1]=1;
        p[i1+1]=='T'?f[i1+1][i+(k?1:-1)][j+1][k]=1:f[i1+1][i][j+1][k^1]=1;
        f[i1][i][j+2][k]=1;
        if (i1==n&&j==m) an=max(an,abs(i-100));
    } printf("%d",an); return 0;
}
