#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 101

int n;
int M[lim][2*lim][51][3];
char str[lim];

int solve(int i,int j,int k,int p)
{
  int &res=M[i][j][k][p];
  
  if(res==-1)
    if(i==strlen(str))
	  if(k&1)
	    res=0;
	  else
	    res=abs(j-100);
	else
	  if(str[i]=='F')
	  {
	    res=solve(i+1,j+2*p-1,k,p);
		if(k)
	      res=max(res,solve(i+1,j,k-1,1-p));
	  }
	  else
	  {
	    res=solve(i+1,j,k,1-p);
		if(k)
	      res=max(res,solve(i+1,j+2*p-1,k-1,p));
	  }
  return res;
}
int main()
{
  scanf("%s%d",str,&n);
  memset(M,-1,sizeof M);
  printf("%d",solve(0,100,n,1));  
  return 0;
}
