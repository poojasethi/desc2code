#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>

using namespace std;

bool hash[106][55][205][2]={0};
char str[205]={0};
long m,n;
long ans=0;

void ff(long l,long s,long p,long flag)
{
 if(hash[l][s][p][flag])
  return;
 hash[l][s][p][flag]=true;
 if(l<=n)
  {
   for(long i=s;i<=m;i+=2)
     ff(l+1,i,p+(2*flag-1)*(str[l]=='F'),flag^(str[l]=='T'));
   for(long i=s+1;i<=m;i+=2)
     ff(l+1,i,p+(2*flag-1)*(str[l]=='T'),flag^(str[l]=='F'));
  }
 else if(abs(p-100)>ans&&s==m)
  ans=abs(p-100);
}

int main()
{
 scanf("%s",str+1);
 n=strlen(str+1);
 scanf("%ld",&m);
 ff(1,0,100,1);
 cout<<ans<<endl;
 return 0;
}
