#include <bits/stdc++.h>
using namespace std;
int dp[110][55][2];
int main()
{
    char str[110];
    int n,i,j,k,len,val,step;
    scanf("%s",str);
    scanf("%d",&n);
    len=strlen(str);
    memset(dp,500,sizeof(dp));
    dp[0][0][0]=dp[0][0][1]=0;
    for(i=1;i<=len;i++)
        for(j=0;j<=n;j++)
            for(k=0;k<2;k++)
                for(step=0;step<=j;step++)
                {
                    if(k==1) val=-1;
                    else val=1;
                    if((str[i-1]=='F'&&step%2==1)||(str[i-1]=='T'&&step%2==0)) dp[i][j][k]=max(dp[i][j][k],dp[i-1][j-step][1-k]);
                    else dp[i][j][k]=max(dp[i][j][k],dp[i-1][j-step][k]+val);
                }
    printf("%d\n",max(dp[len][n][0],dp[len][n][1]));
    return 0;
}
