#include <cstdio>
#include <algorithm>
#include <cstring>

using namespace std;

int m, n, a[] = {-1, 1}, f[111][55][2];
char s[105];

int main()
{
    scanf("%s%d", s + 1, &n);
    m = strlen(s + 1);
    memset(f, 200, sizeof(f));
    f[0][0][0] = f[0][0][1] = 0;
    for(int i = 1; i <= m; i++)
        for (int j = 0; j <= n; j++)
            for (int k = 0; k < 2; k++)
                for (int t = 0; t <= j; t++)
                    if ((t % 2 == 0) == (s[i] == 'F'))
                        f[i][j][k] = max(f[i][j][k], f[i - 1][j - t][k] + a[k]);
                    else f[i][j][k] = max(f[i][j][k], f[i - 1][j - t][!k]);
    printf("%d\n", max(f[m][n][0], f[m][n][1]));
    return 0;
}
