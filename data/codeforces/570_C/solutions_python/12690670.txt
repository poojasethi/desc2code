#! /usr/bin/python
#coding: utf-8

def get_original_ans(x):
    a = 0
    xl = len(x)
    i = 0
    while i < xl:
        if x[i] == '.':
            i += 1
            while i < xl:
                if x[i] == '.':
                    a += 1
                    i += 1
                else:
                    break
        i += 1
    return a

m, n = map(int, raw_input().split())
s = list(raw_input())

ans = get_original_ans(s)

for i in xrange(n):
    temp = raw_input().split()
    pos = int(temp[0]) - 1
    if temp[1] == '.':
        if s[pos] == '.':
            print ans
        else:
            if pos > 0 and s[pos - 1] == '.':
                ans += 1
            if pos + 1 < len(s) and s[pos + 1] == '.':
                ans += 1
            print ans
    else:
        if s[pos] != '.':
            print ans
        else:
            if pos > 0 and s[pos - 1] == '.':
                ans -= 1
            if pos + 1 < len(s) and s[pos + 1] == '.':
                ans -= 1
            print ans
    s[pos] = temp[1]
