n,k=map(int,raw_input().split())
v=['?']+list(raw_input())+['?']
t=0
for i in range(n):
  if '.'==v[i] and '.'==v[i+1]:
    t+=1
for _ in range(k):
  x,c=raw_input().split()
  x=int(x)
  if '.'==c and v[x]!='.':
    if '.'==v[x-1]:
      t+=1
    if '.'==v[x+1]:
      t+=1
  elif '.'!=c and v[x]=='.':
    if '.'==v[x-1]:
      t-=1
    if '.'==v[x+1]:
      t-=1
  v[x]=c
  print t
