#!/usr/bin/env python

import re

def count(s):
	return len(re.findall(r'\.(?=\.)', s))

n, m = map(int, raw_input().split())
s = [c for c in raw_input()]
queries = [map(str, raw_input().split()) for _ in range(m)]

countl = count("".join(s))

for q in queries:
	x = int(q[0]) - 1
	c = q[1]
	if s[x] == '.' and c != '.':
		if x > 0 and s[x-1] == '.':
			countl -=1
		if x < n-1 and s[x+1] == '.':
			countl -=1
	elif s[x] != '.' and c == '.':
		if x > 0 and s[x-1] == '.':
			countl +=1
		if x < n-1 and s[x+1] == '.':
			countl +=1

	s[x] = c
	print countl
