import os

n, m = map(int, raw_input().split())
sum = 0
str = ['a'] + list(raw_input()) + ['a']
for i in range(1, n+1) :
    if str[i] == '.' and str[i-1] == '.':
        sum += 1
# print 'sum ', sum
for i in range(m) :
    x, y = raw_input().split()
    x = int(x)
    if y == str[x] or (y != '.' and str[x] != '.') : 
        print sum
        continue
    if y == '.' :
        if y == str[x-1] : 
            sum += 1
        if y == str[x+1] : 
            sum += 1
    else : 
        if str[x-1] == '.' : 
            sum -= 1
        if str[x+1] == '.' : 
            sum -= 1
    str[x] = y
    print sum
