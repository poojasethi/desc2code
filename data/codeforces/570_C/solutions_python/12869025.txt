
n,m = map(int,raw_input().split())
str = list(raw_input())
ans = 0
for i in xrange(n-1):
    if str[i] == str[i+1] and str[i]=='.':
        ans += 1

for i in xrange(m):
    x,c = raw_input().split()
    x = int(x)
    if str[x-1]=='.' and c != '.':
        ans = ans - int(x-2 >= 0 and str[x-2] == '.')-int(x < n and str[x] == '.')
    elif str[x-1] != '.' and c == '.':
        ans = ans + int(x-2 >= 0 and str[x-2] == '.')+int(x < n and str[x] == '.')
    str[x-1] = c
    print ans
