n,m = map(int, raw_input().split(" "))
s = list("*" + raw_input() + "*")
count = 0
for i in range(n):
    if s[i] == s[i + 1] == ".":
        count += 1
for i in range(m):
    x, c =raw_input().split(" ")
    x = int(x)
    count -= (s[x] == s[x + 1] == ".")
    count -= (s[x] == s[x - 1] == ".")
    count += (c == s[x + 1] == ".")
    count += (c == s[x - 1] == ".")
    s[x] = c
    print count
