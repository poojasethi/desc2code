#include <iostream>

using namespace std;

int main()
{
    string s;
    int a,i,x,n,m;
    char c;
    cin >> n >> m >> s;
    s= ' '+s+' ';
    a=0;
    for(i=1; i<=n; i++){
      if(s[i]=='.'&&s[i-1]=='.'){
        a++;
      }
    }
    for(i=0;i<m;i++){
      cin >> x >> c;
      if (s[x]!='.'&&c=='.'){
        a+=(s[x-1]=='.')+(s[x+1]=='.');
      }
      if (s[x]=='.'&&c!='.'){
        a-=(s[x-1]=='.')+(s[x+1]=='.');
      }
      s[x]=c;
      cout << a << endl;
    }
    return 0;
}
