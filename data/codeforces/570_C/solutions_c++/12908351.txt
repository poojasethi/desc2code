#include<iostream>
using namespace std;
int main(){
    int na,nb,count=0;
    string s;
    cin>>na>>nb>>s;
    for(int i=0;i<na;i++){
	if(s[i-1]=='.'&&s[i]=='.'){
	    count++;
	}
    }
    for(int i=0;i<nb;i++){
	int nq;
	char ch;
	cin>>nq>>ch;
	if(s[nq-1]!='.'&&ch=='.'){
	    if(s[nq-2]=='.') count++;
	    if(s[nq]=='.') count++;
	}
	if(s[nq-1]=='.'&&ch!='.'){
	    if(s[nq-2]=='.') count--;
	    if(s[nq]=='.') count--;
	}
	s[nq-1]=ch;
	cout<<count<<endl;
    }
}
