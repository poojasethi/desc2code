#include<iostream>
using namespace std;
int main()
{
	int i,j,k,n,m,ans=0,x;

	cin>>n>>m;

	for(i=0;i<n;i++)
	if(a[i]=='.'&& a[i+1]=='.')
	ans++;
	char c;
	while(m--)
	{
		cin>>x>>c;
		x-=1;
		if(x>0 && a[x-1]=='.' && a[x]=='.')
		ans--;
		if(x<n-1 && a[x+1]=='.' && a[x]=='.')
		ans--;
		a[x]=c;
		if(x>0 && a[x-1]=='.' && a[x]=='.')
		ans++;
		if(x<n-1 && a[x+1]=='.' && a[x]=='.')
		ans++;
			cout<<ans<<endl;
	}

}
