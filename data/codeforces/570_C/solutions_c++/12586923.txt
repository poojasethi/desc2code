#include <bits/stdc++.h>
using namespace std;


	int t,i, n,m,a;
	char c;
	string s;
int main() {
	a=0;
	cin>>n>>m;
	cin>>s;
	for(i=1;i<n;i++)
	{
		if(s[i]=='.' && s[i-1]=='.')
		{
			a++;
		}
	}
	
	while(m--)
	{
	cin>>t>>c;
	t--;
	if(c=='.' && s[t]!='.' &&  s[t-1]=='.' ) { a++;}
	if(c=='.' && s[t]!='.' &&  s[t+1]=='.' ) { a++;}
	if(c!='.' && s[t]=='.' &&  s[t+1]=='.' ){ a--;}
	if (c!='.' && s[t]=='.' && s[t-1]=='.'){ a--;}
	s[t] = c;
	std::cout<<a<<endl;
	}
	return 0;
}
