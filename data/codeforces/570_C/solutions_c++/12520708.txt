#include <iostream>
using namespace std;
int main(){
int n,m,i,a=0,b;cin>>n>>m;string s;cin>>s;s=' '+s+' ';char c;
for(i=1;i<=n;i++)if(s[i]==s[i-1]&&s[i]=='.')a++;
for(i=0;i<m;i++){cin>>b>>c;
if(c=='.'&&s[b-1]=='.'&&s[b]!='.')a++;
if(c=='.'&&s[b+1]=='.'&&s[b]!='.')a++;
if(s[b]=='.'&&s[b+1]=='.'&&c!='.')a--;
if(s[b]=='.'&&s[b-1]=='.'&&c!='.')a--;
s[b]=c;
cout<<a<<endl;}
}
