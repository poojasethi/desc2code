#include <bits/stdc++.h>
using namespace std;
int main(){
	int n,m;
	cin>>n>>m;
	char s[300005];
	scanf("%s",s+1);
	long long ans=0;
	for (int i=1;i<=n;++i){
		if (s[i]=='.'&&s[i+1]=='.') ans++;
	}
	for (int i=0;i<m;++i){
		int x;
		char c;
		getchar();
		cin>>x>>c;
		if (s[x]=='.'&&s[x+1]=='.') ans--;
		if (s[x]=='.'&&s[x-1]=='.') ans--;
		s[x]=c;
		if (s[x]=='.'&&s[x+1]=='.') ans++;
		if (s[x]=='.'&&s[x-1]=='.') ans++;
		cout<<ans<<endl;
	}
}
