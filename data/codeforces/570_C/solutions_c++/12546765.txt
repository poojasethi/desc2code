#include<bits/stdc++.h>
using namespace std;

int main()
{
	int l,t,i,j,k,n,m,x,ans;
	char s[300005];char ch;
	scanf("%d %d",&n,&m);
	scanf("%s",s+1);
	for(i=1;i<n;i++)
		if(s[i]=='.' && s[i+1]=='.')
			ans++;
	for(i=0;i<m;i++)
	{
		scanf("%d %c",&x,&ch);
		if(s[x]=='.' && s[x-1]=='.')
			ans--;
		if(s[x]=='.' && s[x+1]=='.')
			ans--;
		s[x] = ch;
		if(s[x]=='.' && s[x+1]=='.')
			ans++;
		if(s[x]=='.' && s[x-1]=='.')
			ans++;
		printf("%d\n",ans);
	}
	return 0;
}
