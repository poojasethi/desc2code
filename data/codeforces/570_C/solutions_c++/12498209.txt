#include <iostream>
#include <fstream>
#include <algorithm>
using namespace std;
int n,m,f=0,ii,a;
char s[300010]={},b;
int main(){
	cin >> n >> m >> s;
	for(ii=0; ii<n; ii++){
		if(s[ii]=='.' && s[ii+1]=='.') f++;
	}
	for(int i=0; i<m; i++){
		cin >> a >> b;
		if(s[a-1]=='.'){
			if(s[a]=='.') f--;
			if(a!=1 && s[a-2]=='.') f--;
		}
		s[a-1]=b;
		if(b=='.'){
			if(s[a]=='.') f++;
			if(a!=1 && s[a-2]=='.') f++;
		}
		cout << f << '\n';
	}
}
