#include <bits/stdc++.h>
using namespace std;

string s; char ch;
int n,m,p,a,i;
int main()
{
    //freopen("inp.txt", "r", stdin);
    cin>>n>>m>>s;
    s=' '+s+' ';
    for(i=1;i<=n;i++)a+=(s[i]=='.'&&s[i-1]=='.');
    for(i=0;i<m;i++)
    {
        cin>>p>>ch;
        if (s[p]!='.' && ch=='.')a+=(s[p-1]=='.')+(s[p+1]=='.');
        if (s[p]=='.' && ch!='.')a-=(s[p-1]=='.')+(s[p+1]=='.');
        s[p]=ch;
        cout<<a<<endl;
    }
}
