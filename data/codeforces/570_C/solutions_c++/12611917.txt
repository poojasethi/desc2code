#include<iostream>
using namespace std;
int main()
{
int c,n,m,pos;
char ch;
cin>>n>>m;
char a[n+1];
cin>>a;
a[n]='\0';
c=0;
for(int i=0;i<n;i++)
{
if(a[i]=='.'&&a[i+1]=='.')
c++;
}
for(int i=0;i<m;i++)
{
cin>>pos>>ch;
pos--;
if(ch=='.')
{
if(pos-1>=0&&a[pos-1]=='.'&&a[pos]!='.')
c++;
if(pos+1<n&&a[pos+1]=='.'&&a[pos]!='.')
c++;

}
else
{
if(pos-1>=0&&a[pos-1]=='.'&&a[pos]=='.')
c--;
if(pos+1<n&&a[pos+1]=='.'&&a[pos]=='.')
c--;
}
a[pos]=ch;
cout<<c<<endl;
}
return 0;
}
