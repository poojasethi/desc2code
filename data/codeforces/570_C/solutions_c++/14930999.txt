#include <bits/stdc++.h>
using namespace std;
int main()
{
std::ios_base::sync_with_stdio(false);
int n,m,c=0,x;
char y;
string s;
cin>>n>>m>>s;
for (int i=0;i<n;i++)
if (s[i]=='.'&&s[i+1]=='.') c++;
while(m--)
{
cin>>x>>y;
--x;
if (s[x]=='.')
{
if(x+1<n)
if(s[x+1]=='.'&&y!='.') c--;
if(x-1>=0)
if(s[x-1]=='.'&&y!='.') c--;
}
else if (y=='.')
{
if(x-1>=0)
if(s[x-1]=='.') c++;
if(x+1<n)
if(s[x+1]=='.') c++;
}
s[x]=y;
cout<<c<<endl;
}
return 0;
}
