#include<cstdio>
using namespace std;
char s[300001],op[2];
int main()
{
    int n,m,a,o=0;
    scanf("%d%d%s",&n,&m,s);
    for(int i=1;i<n;++i)
        o+=s[i]=='.'&&s[i-1]=='.';
    while(m--)
    {
        scanf("%d%s",&a,op);--a;
        if(s[a]=='.'&&op[0]!='.')
            o-=(a&&s[a-1]=='.')+(a+1<n&&s[a+1]=='.');
        if(s[a]!='.'&&op[0]=='.')
            o+=(a&&s[a-1]=='.')+(a+1<n&&s[a+1]=='.');
        s[a]=op[0];
        printf("%d\n",o);
    }
}
