#include<cstdio>
int n,m;
char s[300001];
int sum;
char ch;
int x;
int main(){
	scanf("%d%d\n",&n,&m);
	gets(s);
	sum=0;
	for(int i=1;i<n;i++){
		if(s[i]=='.'&&s[i-1]=='.')sum++;
	}
	
	for(int i=1;i<=m;i++){
		scanf("%d %c",&x,&ch);
		x--;
		if(s[x]=='.'){
			if(x>0&&s[x-1]=='.')sum--;
			if(x<n-1&&s[x+1]=='.')sum--;
			s[x]=ch;
		}
		if(ch=='.'){
			if(x>0&&s[x-1]=='.')sum++;
			if(x<n-1&&s[x+1]=='.')sum++;
			s[x]='.';
		}
		printf("%d\n",sum);
	}
	return 0;
}
