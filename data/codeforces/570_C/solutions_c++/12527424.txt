#define N (1<<19)
#include <bits/stdc++.h>

using namespace std;

int n,m,cnt,x;
char s[N],t[N];

int main()
{
	cin>>n>>m>>(s+1);
	for(int i=1;i<=n;i++)
		if(s[i]=='.' && s[i-1]=='.') cnt++;
	for(int i=0;i<m;i++)
	{
		scanf("%d%s",&x,t);
		if(s[x]=='.' && t[0]!='.')
		{
			if(s[x-1]=='.') cnt--;
			if(s[x+1]=='.') cnt--;
		}
		else if(s[x]!='.' && t[0]=='.')
		{
			if(s[x-1]=='.') cnt++;
			if(s[x+1]=='.') cnt++;
		}
		s[x]=t[0];
		printf("%d\n",cnt);
	}
	return 0;
}
