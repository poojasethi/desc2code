#include<stdio.h>
int main()
{
	int n,m,sum=0,k,i;
	char s[300010],p;
	scanf("%d%d",&n,&m);
	scanf("%s",s);
	for(i=1;i<n;i++){
		if(s[i]=='.'&&s[i-1]=='.')sum++;
	}
	while(m--){
		scanf("%d %c",&k,&p);
		if(p!='.'&&s[k-1]=='.'){
			if(s[k-2]=='.')sum--;
			if(s[k]=='.')sum--;
		}
		else if(p=='.'&&s[k-1]!='.'){
			if(s[k-2]=='.')sum++;
			if(s[k]=='.')sum++;
		}
		s[k-1]=p;
		printf("%d\n",sum);
	}
	return 0;
}
