#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	string s;
	char q;
	cin>>s;
	s[-1]='?';
	int i,k=0;
	for (i=0; i<n; i++)
	if (s[i]=='.' && s[i+1]=='.') k++;	
	for (i=1; i<=m; i++)
	{
		cin>>n>>q;
		if (s[n-1]=='.' && s[n-2]=='.') k--;
		if (s[n-1]=='.' && s[n]=='.') k--;
		s[n-1]=q;
		if (s[n-1]=='.' && s[n-2]=='.') k++;
		if (s[n-1]=='.' && s[n]=='.') k++;
		cout<<k<<endl;	
	}
return 0;
}
