#include<iostream>
#include<string>
using namespace std;
int main()
{
	int n,m;
	string s;
	cin>>n>>m;
	cin>>s;
	int ans=0;
	for(int i=1;i<s.length();i++)
	{
		if(s[i-1]=='.'&&s[i]=='.')
			ans++;
	}
	for(int i=0;i<m;i++)
	{
		int p;char a;
		cin>>p>>a; p-=1;
		if(a=='.'&&s[p]!='.')
		{
			if(s[p-1]=='.')
				ans++;	
			if(s[p+1]=='.')
				ans++;
		}
		else if(a!='.'&&s[p]=='.')
		{
			if(s[p-1]=='.')
				ans--;	
			if(s[p+1]=='.')
				ans--;
		}
		cout<<ans<<endl;
		s[p]=a;
	}
}
