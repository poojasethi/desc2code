#include<bits/stdc++.h>
using namespace std;
char s[1000001];
int main()
{
	int n,m,i,j,ans=0,p,flag=0,q=0,l=0;
	char c;
	cin>>n>>m;
	scanf("%s\n",s);
	for(i=0;i<n;i++)
		if(s[i]=='.'&&s[i+1]=='.')
			ans++;
	for(i=0;i<m;i++)
	{
		cin>>p>>c;
		if(s[p-1]!='.'&&c=='.')
		{
			if(s[p]=='.')
				ans++;
			if(p>1&&s[p-2]=='.')
				ans++;
		}
		if(s[p-1]=='.'&&c!='.')
		{
			if(s[p]=='.')
				ans--;
			if(p>1&&s[p-2]=='.')
				ans--;
		}
		s[p-1]=c;
		cout<<ans<<endl;
	}
	return 0;
}
