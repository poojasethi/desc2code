#include<stdio.h>
char a[300006],c;
int n,m,i,j,k,s,t,h[300006],p;
int main()
{
	scanf("%d%d",&n,&m)	;
	for(i=1;i<=n;i++)
	{
		scanf(" %c",&a[i]);
		if(i!=1 && a[i-1]=='.' && a[i]=='.')s++;
	}
	for(i=0;i<m;i++)
	{
		scanf("%d %c",&k,&c);
		if(k!=1 && a[k]=='.' && c!='.' && a[k-1]=='.')s--;
		if(k!=1 && a[k]!='.' && c=='.' && a[k-1]=='.')s++;
		if(k!=n && a[k]=='.' && c!='.' && a[k+1]=='.')s--;
		if(k!=n && a[k]!='.' && c=='.' && a[k+1]=='.')s++;
		a[k]=c;
		printf("%d\n",s);
	}
	return 0;
}
