#include <bits/stdc++.h>
using namespace std;

int N, M, x, F;
char S[300010], ch;

main()
{
    scanf("%d %d\n%s", &N, &M, S + 1);
    for (int i = 1; i < N; i++)
        F += (S[i] == '.' && S[i + 1] == '.');
    while (M--)
    {
        scanf("\n%d %c", &x, &ch);
        if (S[x] == '.' && ch != '.') F -= (S[x - 1] == '.') + (S[x + 1] == '.');
        if (S[x] != '.' && ch == '.') F += (S[x - 1] == '.') + (S[x + 1] == '.');
        S[x] = ch;
        printf("%d\n", F);
    }
}
