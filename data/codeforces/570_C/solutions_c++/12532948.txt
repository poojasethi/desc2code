#include<stdio.h>
char str[310000];
int main(){
	int n,m,i,x,ans,j;
	char s;
	scanf("%d%d",&n,&m);
	scanf("%s",str);
	ans=0;
	for(j=0;j<n;j++){
		if(str[j]=='.'&&str[j+1]=='.'){
			ans++;
		}
	}
	for(i=0;i<m;i++){
		scanf("%d %c",&x,&s);
		if(s=='.'&&str[x-1]!='.'){
			if(str[x-2]=='.') ans++; 
			if(str[x]=='.') ans++;
		}
		else if(s!='.'&&str[x-1]=='.'){
			if(str[x-2]=='.') ans--;
			if(str[x]=='.') ans--;
		}
	str[x-1]=s;
	printf("%d\n",ans);
	}
	return 0;
}
