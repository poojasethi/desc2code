#include<bits/stdc++.h>
using namespace std;
int n, m, t, x;
string s;
char c;
int main()
{	
	cin>>n>>m>>s, t=0;
	for(int i=0;i<n-1;i++)
		t+=(s[i]==s[i+1] and s[i]=='.');
	while(m--)
	{
		cin>>x>>c;
		x--;
		t-=(s[x]==s[x+1] and s[x]=='.');
		t-=(s[x]==s[x-1] and s[x]=='.');
		s[x]=c;
		t+=(s[x]==s[x+1] and s[x]=='.');
		t+=(s[x]==s[x-1] and s[x]=='.');
		cout<<t<<'\n';
	}
	return 0;
}
