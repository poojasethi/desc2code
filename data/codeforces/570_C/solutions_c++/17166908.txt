#include <stdio.h>

char s[300005];
int main()
{
	int n, m;
	scanf("%d%d", &n, &m);
	scanf("%s", s + 1);
	int answer = 0;
	for (int i = 1; i <= n; i++)
		if (s[i] == '.' && s[i - 1] == '.')
			answer++;
	while (m--)
	{
		int x; char c;
		scanf("%d %c", &x, &c);
		int sig = (c == '.' ? 1 : -1);
		int old = (s[x] == '.' ? 1 : -1);
		if (old != sig)
		{
			s[x] = c;
			answer += ((s[x - 1] == '.') + (s[x + 1] == '.')) * sig;
			
		}
		printf("%d\n", answer);
	}
	return 0;
}
