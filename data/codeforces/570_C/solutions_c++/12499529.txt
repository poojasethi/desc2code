#include <iostream>
using namespace std;
int n,m,ans;
char str[300005];
int main() {
    cin>>n>>m;
    for(int i=1;i<=n;i++){
        cin>>str[i];
    }
    for(int i=0;i<n;i++){
        if(str[i]=='.'&&str[i+1]=='.')ans++;
    }
    for(int j=0;j<m;j++){
        int i;cin>>i;
        ans-=(str[i]=='.'&&str[i+1]=='.')+(str[i]=='.'&&str[i-1]=='.');
        cin>>str[i];
        ans+=(str[i]=='.'&&str[i+1]=='.')+(str[i]=='.'&&str[i-1]=='.');
        cout<<ans<<"\n";
    }
}
