#include<cstring>
#include<iostream>
#include<cstdio>
#define N 100005
#define LL long long
using namespace std;
const LL mod=1e9+7;
LL exgcd(LL a0,LL a1,LL v0,LL v1){
	if(v0==0) return a1;
	LL t=v1/v0;
	return exgcd(a1-t*a0,a0,v1-v0*t,v0);
}
inline LL rv(LL k){
	return ( exgcd(1,0,k,mod)%mod + mod )%mod;
}
LL fac[N];
LL rvfac[N];
inline LL C(LL r,LL n){
	return (LL)fac[n]*rvfac[r]%mod*rvfac[n-r]%mod;
}
void pre1(){
	fac[0]=1;
	for(int i=1;i<N;i++){
		fac[i]=(LL)fac[i-1]*i%mod;
	}
	for(int i=0;i<N;i++){
		rvfac[i]=rv(fac[i]);
	}
}
//-----------------------------------------
LL cf26[N],cf25[N];
char s[N];
int len,m;
int ans[500][N];
LL les[500];int lcnt;
int now;
inline LL f(LL len,LL j){
	return (LL)C(len-1,j-1)*cf25[j-len]%mod;
}
void build(){
	les[++lcnt]=len;
	for(int i=len;i<N;i++){
		ans[lcnt][i]=(f(len,i)+(LL)ans[lcnt][i-1]*26)%mod;
	}
	now=lcnt;
}
bool check(){
	for(int i=1;i<=lcnt;i++) if(les[i]==len) {now=i;return true;}
	return false;
}
void init(){
	cf25[0]=cf26[0]=1;
	for(int i=1;i<N;i++) cf26[i]=(LL)cf26[i-1]*26%mod;
	for(int i=1;i<N;i++) cf25[i]=(LL)cf25[i-1]*25%mod;
	scanf("%d%s",&m,s+1);
	len=strlen(s+1);
	if(!check()) build();
}
//-------------------------------------------
int main(){
	
	pre1();init();	
	for(int i=1,ins,x;i<=m;i++){
		scanf("%d",&ins);
		if(ins==1){
			scanf("%s",s+1);
			len=strlen(s+1);
			if(!check()) build(); 
		}else if(ins==2){
			scanf("%d",&x);
			printf("%d\n",ans[now][x]);
		}else puts("He");
	}
	return 0;
}
