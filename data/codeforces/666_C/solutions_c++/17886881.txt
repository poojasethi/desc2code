#define mod 1000000007
vector<int>ar[100005];
ll f[1000005],r[1000005];
ll modpow(ll x,ll n)
{
	ll res=1;
	while(n>0)
	{
		if(n&1) res=res*x%mod;
		x=x*x%mod;
		n>>=1;
	}
	return res;
}
int main(){
	int m; string s;
	cin >> m >> s;
	f[0] = 1,r[0] = 1;
	for(int i=1;i<=100005;i++){
		f[i] = f[i-1]*i%mod;
		r[i] = modpow(f[i],mod-2);
	}
	int x = s.size();
	ll M = 1;
	for(int a=x;a<=100000;a++){
		ll pre = 0;
		if(a != x){
			pre = ar[x][a-x-1];
		}
		pre *= 26;
		ll ad = M;
		ad = ad*f[a-1]%mod;
		ad = ad*r[a-x]%mod;
		ad = ad*r[x-1]%mod;
		pre = pre+ad;
		pre %= mod;
		ar[x].pb((int)pre);
		M = M*25%mod;
	}
	for(int i=0;i<m;i++){
		int t; scanf("%d",&t);
		if(t == 2){
			int n; scanf("%d",&n);
			if(n < s.size()){
				puts("0");continue;
			}
			printf("%d\n",ar[s.size()][n-s.size()]);
		}
		else{
			cin >> s;
			if(ar[s.size()].empty()){
				x = s.size(); M = 1;
				for(int a=x;a<=100000;a++){
					ll pre = 0;
					if(a != x){
						pre = ar[x][a-x-1];
					}
					pre *= 26;
					ll ad = M;
					ad = ad*f[a-1]%mod;
					ad = ad*r[a-x]%mod;
					ad = ad*r[x-1]%mod;
					pre = pre+ad;
					pre %= mod;
					ar[x].pb((int)pre);
					M = M*25%mod;
				}
			}
		}
	}
}
