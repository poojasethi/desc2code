#include <cstdio>
#include <cstring>
#include <algorithm>
#define maxn 101000
#define intl long long
using namespace std;

const int N = 100001;
const intl Mo = 1000000007LL;

int f[510][maxn];
char s[maxn];
int n,l,t,m,k,mark[maxn];
intl fac[maxn],Ni[maxn];

intl fpm(intl a,intl b) {
	if (!a) return 0LL;
	intl ans = 1LL;
	for (; b; b >>= 1) {
		if (b & 1LL) ans *= a, ans %= Mo;
		a *= a; a %= Mo;
	}
	return ans;
}

intl C(int a,int b) {
	return (fac[a] * Ni[b] % Mo * Ni[a - b] % Mo);
}

void prepare(int l,int t) {
	intl co = 1LL;
	for (int i = l; i <= N; i++) {
		f[t][i] = C(i - 1,l - 1) * co % Mo;
		co *= 25LL; co %= Mo;
	}
	for (int i = l; i <= N; i++)
		f[t][i] = ((intl)f[t][i] + (intl) f[t][i - 1] * 26LL) % Mo;
	return;
}

int main() {
	fac[0] = 1LL;
	for (int i = 1; i <= N; i++) fac[i] = fac[i - 1] * (intl) i % Mo;
	for (int i = 0; i <= N; i++) Ni[i] = fpm(fac[i],Mo - 2LL);
	scanf("%d",&m);
	scanf("%s",s + 1); l = strlen(s + 1);
	mark[l] = ++t;
	prepare(l,t);
	for (; m; m--) {
		scanf("%d",&k);
		if (k == 1) {
			scanf("%s",s + 1); l = strlen(s + 1);
			if (!mark[l]) { mark[l] = ++t; prepare(l,t); }	
		} else {
			scanf("%d",&n);
			printf("%d\n",f[mark[l]][n]);
		}
	}
	return 0;
}
