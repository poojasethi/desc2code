#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int p,ch;
inline int read(){
    p=0;ch=getchar();
    while (ch<'0'||ch>'9') ch=getchar();
    while (ch>='0'&&ch<='9') p=p*10+ch-48, ch=getchar();
    return p;
}
const int MOD=1e9+7,N=1e5;
int T,n,m,o,f[100001],t[100001],l[100001][501],M=0,map[100001],mi25[100001];
char c[100001];
inline int mi(int a,int b){
	int mmh=1;
	while (b){
		if (b&1) mmh=1LL*mmh*a%MOD;
		b>>=1;
		a=1LL*a*a%MOD;
	}
	return mmh;
}
inline void MMH(){
	map[m]=++M;
	l[m][M]=1;
	for (register int i=m+1;i<=N;i++) l[i][M]=(26LL*l[i-1][M]+1LL*mi25[i-m]*f[i-1]%MOD*t[m-1]%MOD*t[i-m])%MOD;
}
int main(){
	/*freopen("a.in","r",stdin);
	freopen("a.out","w",stdout);*/
	register int i,j;
	mi25[0]=f[0]=t[0]=1;
	for (i=1;i<=N;i++) f[i]=1LL*f[i-1]*i%MOD,t[i]=mi(f[i],MOD-2),mi25[i]=25LL*mi25[i-1]%MOD;
	T=read();scanf("%s",c);m=strlen(c);MMH();
	while (T--){
		o=read();
		if (o==1){
			scanf("%s",c);m=strlen(c);
			if (!map[m]) MMH();
		}else{
			n=read();
			printf("%d\n",n>=m?l[n][map[m]]:0);
		}
	}
}
