#include <iostream>
#include <cstdio>
#include <cstring>
#define mod 1000000007
#define MAXN 100005
using namespace std;

char w[MAXN];
int len, n, m, ans[455][MAXN], vis[MAXN], cnt;
int fac[MAXN], ni[MAXN], p[MAXN];

int C(int n,int m)
{
	if(n<m)return 0;
	return 1ll*fac[n]*ni[m]%mod*ni[n-m]%mod;
}

void cal(int n)
{
	if(vis[n])return;
	vis[n]=++cnt;
	ans[cnt][n]=1;
	for(int i=n+1;i<MAXN;++i)
		ans[cnt][i]=(1ll*26*ans[cnt][i-1]+1ll*C(i-1,n-1)*p[i-n])%mod;
}

int main()
{
	p[0]=1;
	for(int i=1;i<MAXN;++i)p[i]=1ll*25*p[i-1]%mod;
	fac[0]=fac[1]=ni[0]=ni[1]=1;
	for(int i=2;i<MAXN;++i)
		fac[i]=1ll*fac[i-1]*i%mod, ni[i]=1ll*(mod-mod/i)*ni[mod%i]%mod;
	for(int i=2;i<MAXN;++i)ni[i]=1ll*ni[i]*ni[i-1]%mod;
	
	int opt;
	while(~scanf("%d",&m))
	{
		scanf("%s",w);
		len=strlen(w);
		cal(len);
		while(m--)
		{
			scanf("%d",&opt);
			if(opt==1)scanf("%s",w), cal(len=strlen(w));
			else
			{
				scanf("%d",&n);
				printf("%d\n",ans[vis[len]][n]);
			}
		}
	}
	return 0;
}
