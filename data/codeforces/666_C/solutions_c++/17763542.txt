using namespace std;



typedef long long LL;



int n;

char s[100005];

int m,k,z;

vector<pair<int,int> >v[100005];

LL now,cur,c,sum,dq;

LL ans[100005];
LL f[100005];
LL jc1[100005],jc2[100005];
LL inv[100005];
LL p25,p26;	

const LL mod=1e9+7;



LL powmod(LL a,LL b,LL c)

{

	LL ans=1;

	while (b)

	{

		if (b%2==1) ans=ans*a%c;

		a=a*a%c;	

		b/=2;

	}

	return ans;

}

int main()

{

	scanf("%d",&n);

	scanf("%s",s);

	m=strlen(s);int num=0;
	for (int i=1;i<=100000;++i) inv[i]=powmod(i,mod-2,mod);
	jc1[0]=jc2[0]=1;
	for (int i=1;i<=100000;++i) jc1[i]=jc1[i-1]*i%mod;
	for (int i=1;i<=100000;++i) jc2[i]=jc2[i-1]*inv[i]%mod;

	for (int i=1;i<=n;++i)

	{

		scanf("%d",&z);

		if (z==2)

		{

			scanf("%d",&k);

			++num;

			v[m].push_back(make_pair(k,num));

		}	

		else

		{

			scanf("%s",s);

			m=strlen(s);

		}

	}

	for (int i=1;i<=100000;++i)

	{
		if (v[i].size()==0) continue;
		memset(f,0,sizeof(f));
		p25=1;
		p26=1;
		for (int j=1;j<=i;++j) p26=p26*inv[26]%mod;
		for (int j=i;j<=100000;++j)
		{
			dq=jc1[j-1]*jc2[i-1]%mod*jc2[j-i]%mod;
			dq=dq*p25%mod*p26%mod;
			p25=p25*25%mod;
			p26=p26*inv[26]%mod;
			f[j]=f[j-1]+dq;
			f[j]%=mod;
		}
		p26=1;
		for (int j=1;j<=100000;++j)
		{
			p26=p26*26%mod;
			f[j]=f[j]*p26%mod;
		}
		for (int j=0;j<v[i].size();++j)
			ans[v[i][j].second]=f[v[i][j].first];

	}

	for (int i=1;i<=num;++i) printf("%I64d\n",ans[i]);

	return 0;

}
