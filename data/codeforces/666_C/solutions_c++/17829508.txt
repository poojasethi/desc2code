#include <cstdio>
#include <cstring>
#include <vector>
using namespace std;
#define MAXN 100005
const int MOD = 1e9+7;
int jc[MAXN], jci[MAXN], pow[MAXN], l[MAXN], stp[MAXN], f[500][100005], n;
void init() {
    pow[0] = jc[0] = jci[0] = jc[1] = jci[1] = 1; pow[1] = 25;
    for(int i = 2; i <= 100000; ++ i) {
        jc[i] = 1LL * jc[i-1]*i%MOD;
        pow[i] = 25LL* pow[i-1] % MOD;
        jci[i] = 1LL * (MOD-MOD/i) * jci[MOD%i] % MOD;
    }
    for(int i = 2; i <= 100000; ++ i)
        jci[i] = 1LL * jci[i-1] * jci[i] % MOD;
}
int Calc(int x, int ed) {
    if(!stp[x]) stp[x] = x;
    f[l[x]][x] = 1;
    for(int i = stp[x] + 1; i <= ed; ++ i)
        f[l[x]][i] = (26LL * f[l[x]][i-1] % MOD + 1LL * pow[i-x] * jc[i-1] % MOD * jci[x-1] % MOD * jci[i-x] % MOD) % MOD;
    stp[x] = max(ed, stp[x]);
    return f[l[x]][ed];
}
char s[MAXN];
int main() {
    scanf("%d%s", &n, s);
    int len = strlen(s), op, p, cnt = 0;
    l[len] = ++ cnt;
    init();
    for(int i = 1; i <= n; ++ i) {
        scanf("%d", &op);
        if(op == 1) {
            scanf("%s", s);
            len = strlen(s);
            if(!l[len]) l[len] = ++ cnt;
        } else if(op == 2) {
            scanf("%d", &p);
            printf("%d\n", Calc(len, p));
        }
    }
    return 0;
}
