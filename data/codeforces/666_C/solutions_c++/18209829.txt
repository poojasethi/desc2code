#include <bits/stdc++.h>
#define Max 100005
#define ll long long
#define Mod 1000000007
using namespace std;

vector <pair<int , vector <int> > > input;
vector <int> v;
char str[Max];
int i26;
int sum[Max],p26[Max],inv[Max];

int mul(int x,int y){   return ((x*1LL)*y)%Mod;}
int mul(int x,int y,int z){ return mul(mul(x,y),z);}
int bigmod(int p,int e){
    int ret = 1;
    for(; e > 0; e >>= 1){
        if(e & 1) ret = mul(ret , p);
        p = mul(p,p);
    }
    return ret;
}

void precalc(int l,int n)
{
    if(n<l) return;
    int i,a,b,c,cn,cr;

    a=c=1;
    b=bigmod(i26,l);
    cn=l;
    cr=l-1;
    sum[0]=b;

    for(i=1;i<=n-l;i++,cn++)
    {
        a=mul(a,25);
        b=mul(b,i26);
        c=mul(c,cn);
        c=mul(c,inv[cn-cr]);
        sum[i] = (sum[i-1] + mul(a,b,c))%Mod;
    }
}

void solve(int l,int n)
{
   printf("%d\n", (n<l)?0: mul( p26[n], sum[n-l]) );
}

int main()
{
    int i,j,k,cas,T,l,n,t,mxn;

    for(inv[1]=1,i=2;i<Max;i++) inv[i]=bigmod(i,Mod-2);
    for(p26[0]=i=1;i<Max;i++) p26[i]=mul(p26[i-1],26);
    i26 = inv[26];
    //freopen("in.txt","r",stdin);

    scanf("%d %s",&T,&str);
    l=strlen(str);

    for(cas=1;cas<=T;cas++)
    {
        scanf("%d",&t);

        if(t==2)
        {
            scanf("%d",&k);
            v.push_back(k);
        }
        else
        {
            input.push_back(make_pair(l,v));
            v.clear();
            scanf("%s",str);
            l=strlen(str);
        }
    }

    if(v.size())
    {
        input.push_back(make_pair(l,v));
    }

    for(i=0;i<input.size();i++)
    {
        l=input[i].first;

        v=input[i].second;
        if(!v.size()) continue;

        mxn=v[0];

        for(j=1;j<v.size();j++) mxn=max(mxn,v[j]);

        precalc(l,mxn);

        for(j=0;j<v.size();j++) solve(l,v[j]);
    }

    return 0;





}
