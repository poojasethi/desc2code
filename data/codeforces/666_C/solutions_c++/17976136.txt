#include<cstdio>
#include<algorithm>
#include<cstring>
#define ll long long
using namespace std;
const int maxn=1e5+5,maxm=450,mod=1e9+7;

char s[maxn];
int n=1e5,m;

int done[maxn];
int f[maxm][maxn];
int now,cnt;

ll ine[maxn];
ll pow(ll x,int k){
	ll ret=1;
	for(int i=k;i;i>>=1,x=x*x%mod)
		if(i&1) ret=ret*x%mod;
	return ret;
}
void prepare(){
	for(int i=1;i<=n;i++) ine[i]=pow(i,mod-2);
}

void Calculate(int x){
	if(done[x]){now=done[x];return;}
	else now=done[x]=++cnt;
	
	ll C=1,P=1;
	f[now][x]=1;
	for(int i=x+1;i<=n;i++){
		C=C*(i-1)%mod*ine[i-x]%mod,P=P*25%mod;
		f[now][i]=(26ll*f[now][i-1]%mod+C*P%mod)%mod;
	}
}

int main(){
	prepare();
	scanf("%d%s",&m,s);
	Calculate(strlen(s));
	
	while(m--){
		int op,len;
		scanf("%d",&op);
		if(op==1){
			scanf("%s",s);
			Calculate(strlen(s));
		}
		else{
			scanf("%d",&len);
			printf("%d\n",f[now][len]);
		}
	}
	
	return 0;
}
