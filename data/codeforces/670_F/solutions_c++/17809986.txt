#include<bits/stdc++.h>
using namespace std;
string S,P;
int cs[10],cp[10],cn[10];
int main()
{
//    freopen("input.txt","r",stdin);
    while(cin>>S>>P)
    {
        memset(cs,0,sizeof(cs));
        memset(cp,0,sizeof(cp));
        for(int i=0;i<S.length();i++)cs[S[i]-'0']++;
        for(int i=0;i<P.length();i++)cp[P[i]-'0']++;
        if(S.length()==2&&cs[0]==cs[1]&&cs[0]==1)
        {
            printf("0\n");continue;
        }
        for(int d=7;d>=1;d--)
        {
            if(d>=S.length())continue;
            int N=S.length()-d;
            if(P.length()>N)continue;
            memset(cn,0,sizeof(cn));

            int digits=0;
            while(N)
            {
                ++cn[N%10];
                N/=10;
                ++digits;
            }
            if(d!=digits)continue;

            bool y=true,any=P[0]!='0';
            for(int i=0;i<10;i++)
            {
                y=y&&(cn[i]+cp[i]<=cs[i]);
                any=any||(i>0&&cn[i]+cp[i]<cs[i]);
            }
            if(y&&any)break;
        }

        for(int i=0;i<10;i++)cs[i]-=(cn[i]+cp[i]);


        string ft=P,nft="";
        for(int i=0;i<10;i++)if(cs[i])ft=ft+string(cs[i],char('0'+i));

        int pd=P[0]-'0',pd2=-1;
        for(int i=1;i<P.length();i++)
            if(P[i]!=P[0])
            {
                pd2=P[i]-'0';
                break;
            }

        int d=1;
        while(!cs[d])d++;
        nft=string(1,char(d+'0'));
        --cs[d];

        for(d=0;d<10;d++)
        {
            if(d==pd&&pd2<pd)
                nft=nft+P;
            if(cs[d]>0)nft=nft+string(cs[d],char(d+'0'));
            if(d==pd&&pd2>=pd)nft=nft+P;
        }

        string &ans=ft;
        if(ft[0]=='0'||nft<ft)ans=nft;
        cout<<ans<<endl;
    }
    return 0;
}
