#include <iostream>
#include <cstdio>
using namespace std;

int n, m, num, cnt[99];
char c, k;
string a, b, x, y;

int f(int p) {
    int s = 0;
    while(p) {
        s++;
        p /= 10;
    }
    return s;
}

void g() {
	int i;
	y = b;
	for(c = '0'; c <= '9'; c++) y.append(cnt[c], c);
    for(c = '1'; c <= '9' && cnt[c] == 0; c++);
	if(c > '9') {
		x = b;
		x.append(cnt['0'], '0');
		return;
	}
	cnt[c]--;
	x = c;
    k = b[0];
    for(i = 0; i < m && b[i] == b[0]; i++);
    if(b[i] < b[0]) k--;
	for(c = '0'; c <= k; c++) x.append(cnt[c], c);
	x.append(b);
	for(c = k + 1; c <= '9'; c++) x.append(cnt[c], c);
}

int main() {
    int i, t;
    cin >> a >> b;
    n = a.size();
    m = b.size();
    for(i = 0; i < n; i++) cnt[a[i]]++;
    for(i = 0; i < m; i++) cnt[b[i]]--;
    t = f(n);
    num = n - t;
    if (t != f(num)) num++;
	while(num) {
        cnt[(num % 10) + '0']--;
        num /= 10;
    }
    g();
    cout << (x<y || y[0]=='0' ? x : y);
    return 0;
}
