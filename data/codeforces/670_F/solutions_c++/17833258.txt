#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

int n, m, cnt[99];
string a, b, s;
vector<string> v;

int f(int p) {
    int c;
    for (c = 0; p > 0; c++, p /= 10);
    return c;
}

void g(int p) {
    while(p) {
        cnt[p % 10 + '0']--;
        p /= 10;
    }
}

void h(int p) {
    int i;
    string t1, t2;
    t1 = t2 = s;
    for (i = p; i < n && s[i] < b[0]; i++);
    t1.insert(i, b);
    v.push_back(t1);
    for (; i < n && s[i] == b[0]; i++);
    t2.insert(i, b);
    v.push_back(t2);
}

int main()
{
    int i, j, t, k;
//    freopen("input.txt", "r", stdin);
    cin >> a >> b;
    n = a.size();
    m = b.size();
    for(i = 0; i < n; i++) cnt[a[i]]++;
    for(i = 0; i < m; i++) cnt[b[i]]--;
    t = f(n);
    k = n - t;
    if (t != f(k)) k++;
    g(k);
    for(i = '0'; i <= '9'; i++) {
        for(j = 0; j < cnt[i]; j++) {
            s += i;
        }
    }
    n = s.size();
    v.push_back(b + s);
    h(0);
    for (i = 0; i < n && s[i] == '0'; i++);
    if (i < n) {
        swap(s[0], s[i]);
        h(1);
    }
    sort(v.begin(), v.end());
    for (i = 0; i < v.size(); i++) {
        if (v[i][0] != '0' || v[i] == "0") break;
    }
    cout << v[i];
    return 0;
}
