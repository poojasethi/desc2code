#include<bits/stdc++.h>
using namespace std;
#define FOR(ii,aa,bb) for(int ii=aa;ii<=bb;ii++)
#define N (int)(1e6+10)
int n,len,temp,H[12],len2,m,pt=-1,bas,son;
string str3,mn;
char str[N],str2[N];
int main(){
	scanf("%s %s",str+1,str2+1);
	len = strlen(str+1);
	len2 = strlen(str2+1);
	if(len == 2) return puts(str2+1),0;
	if(len <= 10) n = len-1;
	else if(len <= 101) n = len-2;
	else if(len <= 1002) n = len-3;
	else if(len <= 10003) n = len-4;
	else if(len <= 100004) n = len-5;
	else n = len-6;
	if(len2 == n or len == 2) return puts(str2+1),0;
	FOR(i,1,len) H[str[i]-'0']++;
	temp = n;
	while(temp){
		H[temp%10]--;
		temp/=10;
	}
	FOR(i,1,len2) H[str2[i]-'0']--;
	FOR(i,0,9)
		while(H[i]){
			str3+=i+'0';
			H[i]--;
		}
	m = str3.size();
	string str4 = str2+1;
	int m2 = str4.size();
	FOR(i,0,m-1)
		if(str3[i] != '0'){
			pt = i;
			break;
		}
	if(str4[0] != '0')	mn = str4+str3;
	char c = str4[0];
	if(str3[0] == '0'){
		FOR(i,0,m-1)
			if(str3[i] != '0') {pt = i;break;}
		if(pt == -1)
			return cout << str4+str3 << endl,0;
		swap(str3[0],str3[pt]);
	}
	FOR(i,0,m-1){
		if(str3[i] < c) bas = i;
		if(str3[i] <= c) son = i;
	}
	if(mn.empty())mn =  str3.substr(0,bas+1) + str4 + str3.substr(bas+1,m-bas-1);
	mn = min(mn , str3.substr(0,bas+1) + str4 + str3.substr(bas+1,m-bas-1));
	mn = min(mn , str3.substr(0,son+1) + str4 + str3.substr(son+1,m-son-1));
	cout << mn << endl;
}
