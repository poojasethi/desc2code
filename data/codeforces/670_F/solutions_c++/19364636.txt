#include<iostream>
#include<string>
#include<cstdio>
#include<cstring>
using namespace std;
string n,s;string ans1,ans2;
int a[10],len,flag=0;
int main(){
	cin>>n>>s;
	for(int i=0;i<n.size();i++)a[n[i]-'0']++;
	for(int i=0,l=1;n.size()>i+l;l*=10,i++)len=n.size()-i-1;
	for(int k=len;k;k/=10)a[k%10]--;
	for(int i=0;i<s.size();i++)a[s[i]-'0']--,len--;
	for(int i=1;i<s.size();i++)if(s[i]!=s[i-1]){if(s[i]>s[i-1])flag=1;break;}
	if(len==0){cout<<s<<endl;return 0;}
	ans2=s;for(int i=0;i<10;i++)for(int j=0;j<a[i];j++)ans2+=i+'0';	
	for(int i=1;i<10;i++)if(a[i]){a[i]--;ans1+='0'+i;break;}
	for(int i=0;i<10;i++){
		if(i==s[0]-'0'&&!flag){for(int j=0;j<s.size();j++)ans1+=s[j];}
		for(int j=0;j<a[i];j++)ans1+=i+'0';
		if(i==s[0]-'0'&&flag){for(int j=0;j<s.size();j++)ans1+=s[j];}
	}
	if(ans1[0]=='0')ans1[0]='9'+1;
	if(ans2[0]=='0')ans2[0]='9'+1;
	cout<<min(ans1,ans2)<<endl;
	return 0;
}
