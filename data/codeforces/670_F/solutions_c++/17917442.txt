int dig[11],vis[11],n,m,k;
string s,t,ans1,ans;
void yaz(int type){
	if(!type){
		for(int i=1;i<=m;i++)
			ans1+=t[i];
	}
	else{
		for(int i=1;i<=m;i++)
			ans+=t[i];	
	}
}
inline void go_launcher(){
	yaz(1);
	for(int j=0;j<10;j++){			
		for(int h=dig[j];h>=1;h--)
			ans+=char(j+48);
	}
}
string cmp(string x,string y){
	if(x[0]=='0')
		return y;
	if(y[0]=='0')
		return x;
	return min(x,y);		
}
int main(){
	cin>>s>>t;
	n=s.size(),m=t.size(),k;
	s="#"+s;t="#"+t;
	for(int i=1;i<=n;i++)
		dig[s[i]-'0']++;
	for(int i=1;i<=m;i++)
		dig[t[i]-'0']--;		
	for(int i=1;i<=n;i++){
		int done=1,digit=0;
		for(int j=0;j<10;j++)
			vis[j]=dig[j];
		for(int j=i;j>=1;j/=10){
			vis[j%10]--;
			if(vis[j%10]<0)
				done=0;
			digit++;	
		}
		if(n-digit==i){
			k=i;
			break;
		}
	}	
	for(int j=k;j>=1;j/=10)
		dig[j%10]--;
	int st=-1;	
	for(int j=1;j<10;j++)
		if(dig[j]){
			st=j;
			break;
		}
	int bas=t[1]-'0';
	int en=1;
	for(int j=1;t[j]-'0'==bas;j++)	
		en=j;		
	if(en+1<=m)
		en=t[en+1]-'0';
	else
		en=bas;	
	go_launcher();
	if(st==-1){
		cout<<ans<<endl;
		return !1;
	}
	ans1=char(st+48),dig[st]--;
	for(int j=0;j<10;j++){
		if(j==bas and j>=en)
			yaz(0);				
		for(int h=dig[j];h>=1;h--)
			ans1+=char(j+48);
		if(j==bas and j<en)
			yaz(0);		
	}
	cout<<cmp(ans,ans1)<<endl;
	return 0;
}
