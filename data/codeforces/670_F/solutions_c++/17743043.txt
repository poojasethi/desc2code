#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int a[10],k,ko[10],m;
string s,b,c[30],dau;


bool check(int kk)
{
    k=((s.length())-kk);
    int kt=k;
    int len=0;
    for (int i=0;i<=9;++i) ko[i]=0;
    while (k)
    {
        ++len;
        ++ko[k%10];
        k/=10;
    }
    if (len!=kk) return false;
    bool ye=(b[0]!='0');
    for (int i=0;i<=9;++i)
        if (a[i]<ko[i]) return false;
    for (int i=1;i<=9;++i)
        if (a[i]-ko[i]>0) ye=true;
    k=kt;
    return ye;
}


string xau(int v)
{
    string s="";
    for (int i=v;i<=9;++i)
    {
        for (int j=1;j<=a[i];++j) s+=char(i+'0');
    }
    return s;
}

int main()
{
    cin>>s;
    cin>>b;
    if (s=="01" || s=="10")
    {
        cout<<0;
        return 0;
    }
    for (int i=0;i<s.length();++i)
    {
        ++a[s[i]-'0'];
    }
    for (int i=0;i<b.length();++i)
    {
        --a[b[i]-'0'];
    }
    for (int i=1;i<=s.length();++i)
    {
        if (check(i))
        {
            for (int j=0;j<=9;++j) a[j]-=ko[j];
            break;
        }
    }
    c[++m]=b;
    for (int i=0;i<=9;++i)
    {
        for (int j=1;j<=a[i];++j) c[m]+=char(i+'0');
    }
    for (int i=1;i<=9;++i)
        if (a[i])
        {
            for (int j=2;j<30;++j) c[j]=char(i+'0');
            --a[i];
            break;
        }
    for (int i=0;i<=9;++i)
    {
        ++m;
        c[m]+=dau+b+xau(i);
        for (int j=1;j<=a[i];++j) dau+=char(i+'0');
        ++m;
        c[m]+=dau+b+xau(i+1);
    }
    sort(c+1,c+m+1);
    m=1;
    while (c[m][0]=='0') ++m;
    cout<<c[m];
    return 0;
}
