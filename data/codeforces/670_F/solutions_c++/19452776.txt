#include <iostream>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#define MAXN 1000010

using namespace std;

int lena , lenb , l , r , mid , digit;
char a[ MAXN ] , b[ MAXN ];
string x , y;
int cnt[ 255 ];

inline int calc( int x )
{
	int cnt = 0;
	while( x ) x /= 10 , cnt++;
	return cnt;
}

inline void get_ans()
{
	register int i;
	x = b;
	for( i = '0' ; i <= '9' ; i++ ) x.append( cnt[i] , i );
	bool flag = 1;
	for( i = '1' ; i <= '9' ; i++ )
		if( cnt[i] )
			flag = 0;
	if( flag )
	{
		cout << x << endl;
		exit( 0 );
	}
	for( i = '1' ; i <= '9' ; i++ )
		if( cnt[i] ) break;
	cnt[i]-- , y = i;
	string temp;
	temp.append( lenb , b[0] );
	if( temp < b )
	{
		for( i = '0' ; i <= b[0] ; i++ ) y.append( cnt[i] , i );
		y.append( b );
		for( i = b[0] + 1 ; i <= '9' ; i++ ) y.append( cnt[i] , i );
	}
	else
	{
		for( i = '0' ; i < b[0] ; i++ ) y.append( cnt[i] , i );
		y.append( b );
		for( i = b[0] ; i <= '9' ; i++ ) y.append( cnt[i] , i );
	}
}

int main()
{
	scanf( "%s" , a ) , scanf( "%s" , b );
	for( register int i = 0 ; a[i] ; i++ ) cnt[ a[i] ]++ , lena++;
	for( register int i = 0 ; b[i] ; i++ ) cnt[ b[i] ]-- , lenb++;
	l = 0 , r = lena;
	while( l <= r )
	{
		mid = ( l + r ) >> 1;
		if( mid + calc( mid ) <= lena ) l = mid + 1 , digit = mid;
		else r = mid - 1;
	}
	while( digit )
	{
		cnt[ digit % 10 + '0' ]--;
		digit /= 10;
	}
	get_ans();
	if( x[0] == '0' ) cout << y << endl;
	else if( y[0] == '0' ) cout << x << endl;
	else if( x < y ) cout << x << endl;
	else cout << y << endl;
	return 0;
}
