#include <iostream>
#include <cstdio>
#define N 1000005
using namespace std;

int n, m, sz, a[N], b[N], cnt[10], x1[N], x2[N];

int f(int p) {
    int c = 0;
    while(p) {
        c++;
        p /= 10;
    }
    return c;
}

void g(int p) {
    while(p) {
        cnt[p % 10]--;
        p /= 10;
    }
}

void k1() {
	int i, j, z, p = 0;
    for(i = 1; i < 10 && cnt[i] == 0; i++);
    if(i == 10) {
		for(i = 0; i < m; i++) {
			x1[p++] = b[i];
		}
		for(i = 0; i < cnt[0]; i++) {
			x1[p++] = 0;
		}
		return;
    }
	cnt[i]--;
	x1[p++] = i;
    z = b[0];
    for(i = 0; i < m && b[i] == b[0]; i++);
    if(b[i] < b[0]) z = b[0] - 1;
	for(i = 0; i <= z; i++) {
		for(j = 0; j < cnt[i]; j++) {
			x1[p++] = i;
		}
	}
	for(i = 0; i < m; i++) {
		x1[p++] = b[i];
	}
	for(i = z + 1; i <= 9; i++) {
		for(j = 0; j < cnt[i]; j++) {
			x1[p++] = i;
		}
	}
}

void k2() {
	int i, j, p = 0;
	for(i = 0; i < m; i++) {
		x2[p++] = b[i];
	}
	for(i = 0; i <= 9; i++) {
		for(j = 0; j < cnt[i]; j++) {
			x2[p++] = i;
		}
	}
}

int main()
{
    int i, j;
    string t;
//    freopen("input.txt", "r", stdin);
    cin >> t;
    n = t.size();
    for(i = 0; i < n; i++) {
		a[i] = t[i] - '0';
		cnt[a[i]]++;
    }
    cin >> t;
    m = t.size();
    for(i = 0; i < m; i++) {
		b[i] = t[i] - '0';
    }
    for(i = 1; ; i++) {
        if(f(n - i) == i) {
			sz = n - i;
            g(sz);
            break;
        }
    }
    for(i = 0; i < m; i++) {
        cnt[b[i]]--;
    }
    k2();
    k1();
    for (i = 0; i < sz && x1[i] == x2[i]; i++);
    if(x1[i] < x2[i] || b[0] == 0) {
		for (i = 0; i < sz; i++) {
			printf("%d", x1[i]);
		}
    } else {
		for (i = 0; i < sz; i++) {
			printf("%d", x2[i]);
		}
    }
    return 0;
}
