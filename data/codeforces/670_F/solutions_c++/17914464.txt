#include<string>
#include<algorithm>
#include<iostream>
#include<cstdio>
using namespace std;
string a,b,ans,s;
int n,m,p,cnt[10];
int len(int x){
	int c=0;
	while (x){
		x/=10;
		c++;
	}
	return c;
}
void solve1(){
	p=len(n);
	if (len(n-p)!=p) p--;
	for (int i=n-p;i;i/=10)
	cnt[i%10]--;
}
void solve2(){
	int i;
	for  (i=0;i<10;i++) s.append(cnt[i],i+'0');
	if (b[0]!='0') ans=min(ans,b+s);
	for (i=0;s[i]=='0' && i<n-p-m;i++);
	if (i==n-p-m) return;
	swap(s[0],s[i]);
	for (i=1;s[i]<b[0] && i<n-p-m;i++);
	string t=s;
	ans=min(ans,t.insert(i,b));
	for (;s[i]==b[0] && i<n-p-m;i++);
	 ans=min(ans,s.insert(i,b));
}
int main(){
	cin>>a>>b;
	n=a.size();
	m=b.size();
	if (a=="01" || a=="10") { puts("0");return 0;} 
	for (int i=0;i<n;i++) cnt[a[i]-'0']++;
	for (int i=0;i<m;i++) cnt[b[i]-'0']--;
	ans="A";
	solve1();
	solve2();
	cout<<ans; 
	return 0;
}
