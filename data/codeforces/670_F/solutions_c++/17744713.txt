#include <bits/stdc++.h>

using namespace std;

int all[10];
string s,q;
int len,i,j,ii,tmp,lst;

int main()
 {
  cin>>s;
  cin>>q;

  if (s == "01" || s == "10")
   {
    cout<<0<<endl;
    return 0;
   }

  for (len=1;len<=1000000;len++)
   {
    int k=len,dig=0;
    while (k) dig++,k/=10;
    if (dig+len == s.size()) break;
   }

  while (len)
    all[len%10]--,len/=10;

  for (i=0;i<s.size();i++)
    all[s[i]-'0']++;

  for (i=0;i<q.size();i++)
   all[q[i]-'0']--;

  for (j=1;j<=9;j++)
   if (all[j]) break;

  if (j == 10) goto printrev;
  if (q[0] == '0') goto printnorm;

  while (lst+1 < q.size() && q[lst] == q[0]) lst++;

  len=all[0]+1;
  for (int c=0;c<q.size();c+=len)
   {
    if (q[c]-'0' > j) goto printnorm;
    if (q[c]-'0' < j) goto printrev;

    for (int k=1;k<len && c+k<q.size();k++)
     if (q[c+k] != '0') goto printnorm;
   }

  goto printrev;

  printnorm:
   {
    putchar(j+'0'); all[j]--;
    if (q[lst] < q[0]) tmp=q[0]-'0'-1; else tmp=q[0]-'0';
    for (ii=0;ii<=tmp;ii++)
     while (all[ii]--) putchar(ii+'0');
    cout<<q;
    for (ii=tmp+1;ii<=9;ii++)
     while (all[ii]--) putchar(ii+'0');
    return 0;
   }

  printrev:
   {
    cout<<q;
    for (ii=0;ii<=9;ii++)
     while (all[ii]--) putchar(ii+'0');

   }
 }
