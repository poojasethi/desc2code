#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<iostream>
#define ll long long
using namespace std;
char c1[1111111],c2[1111111];
int len,num[111],ans[1111111];
void findnum(int s)
{
	int llen=len-s;
	int flag=0,num1=0;
	while(llen>0)
	{
		int k=llen%10;
		llen/=10;
		if(!num[k]||num1+1>s)flag=1;
		num[k]--;
	    num1++;
	}
	if(flag||num1!=s){
	   llen=len-s;
       while(llen>0){
       	  int k=llen%10;
		  llen/=10;
		  num[k]++;
       }
       findnum(s-1);
	}
}
int main()
{
	int i,j;
	scanf("%s",c1);
	scanf("%s",c2);
	memset(num,0,sizeof(num));
	len=strlen(c1);
	//printf("%d.........\n",len);
	for(i=0;i<len;i++)
	num[c1[i]-'0']++;
	len=strlen(c2);
	for(i=0;i<len;i++)
	num[c2[i]-'0']--;
	len=strlen(c1);
	findnum(7);
	int tot=0;
	for(i=1;i<=9;i++)
	{
		if(num[i]){
			num[i]--;
			ans[tot++]=i;
			break;
		}
	}
	for(i=0;i<=9;i++)
	{
		while(num[i]--)
		{
			ans[tot++]=i;
		}
	}
	int len1=strlen(c2);
	int flag=0;
	if(ans[0]==0){
		printf("%s",c2);
		for(i=0;i<tot;i++)
		printf("0");
		printf("\n");
		return 0;
	}
	int kk=-1;
	if(c2[0]-'0'==0)printf("%d",ans[0]);
	else{
		
		flag=1;
		for(i=0;i<len1&&i<tot;i++)
		{
		    if(c2[i]-'0'==ans[i])continue;
			else{
				if(c2[i]-'0'>ans[i])flag=0;
				break;
			}	
	    } 
    	if(flag){printf("%s",c2);kk=ans[0];}
		else printf("%d",ans[0]);
	}
	for(i=1;i<tot;i++)
	{
		if(flag){
			printf("%d",ans[i]);
			if((kk!=-1&&ans[i]>=kk)){
				printf("%d",kk);
				kk=-1;
			}
			continue;
		}
		for(j=0;j<=len1;j++)
		{
			int x1,x2;
			if(j==len1)x1=ans[i];
			else x1=c2[j]-'0';
			if(j==0)x2=ans[i];
			else x2=c2[j-1]-'0';
			if(x1==x2)continue;
			else{
				if(x1<x2)flag=1;
				break;
			}	
		}
		if(flag)printf("%s",c2);
		printf("%d",ans[i]);
	}
	if(kk!=-1)printf("%d",kk);
	if(!flag)printf("%s",c2);
	printf("\n");
}
