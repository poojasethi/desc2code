#include <iostream>
#include <vector>
#include <cstdio>
#include <cstring>

using namespace std;

const int maxN = 1e6+10;

vector <int> a(11);
vector <int> b(11);
int m,n,vtb;
char s[maxN],t[maxN];
bool chXb;

void ghiB(int v)
{
    for (int i=v; i<m; i++)
        printf("%c",t[i]);
    if (v>1)
        for (int i=1; i<v; i++)
            a[t[i] - '0']++;
    chXb=false;
}
void ghi(int k)
{
    bool prk=true;
    a[k]--;
    if (chXb)
    {
        if ((t[vtb]-'0')==k)
        {
            vtb++;
            if (vtb==m)
            {
                chXb=false;
                for (int i=0; i<10; i++)
                    a[i]+=b[i];
            }
        }
        else
        {
            if (vtb>0)
            {
                if ((t[vtb]-'0')>k) vtb=0;
                else
                {
                    ghiB(vtb);
                    a[k]++;
                    prk=false;
                }
            }
        }
        if (chXb)
            if (k==(t[0]-'0') && a[k]==b[k])
            {
                prk=false;
                ghiB(0);
            }
    }
    if (prk)
        printf("%d",k);
}
int dcs(int n)
{
    int tmp;
    tmp=0;
    while (n!=0)
    {
        tmp++;
        n/=10;
    }
    return tmp;
}
int main()
{
    gets(s);
    gets(t);
    if (strcmp(s,"10")==0 || strcmp(s,"01")==0)
    {
        puts("0");
        return(0);
    }
    n=strlen(s);
    m=strlen(t);
    for (int i=0; i<n; i++)
        a[s[i] - '0']++;
    for (int i=0; i<m; i++)
        b[t[i] - '0']++;
    b[t[0] - '0']--;
    for (int j=1; j<8; j++)
        if (dcs(n-j)==j)
        {
            int c,p;
            p=n-j; c=dcs(p);
            while (c>0)
            {
                a[p%10]--;
                p/=10; c--;
            }
            chXb=true; vtb=0;
            int k=1;
            while (a[k]-b[k]==0) k++;
            ghi(k);
            a[10]=0; b[10]=0;
            k=0;
            while (k<10)
            {
                k=0;
                while ((k<10) && (a[k]-b[k]==0))
                    k++;
                if (k<10)
                ghi(k);
            }
        }
}
