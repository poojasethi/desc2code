#include<bits/stdc++.h>
using namespace std;
char a[1000005],b[1000005];
int tkrar[10]={0};
int len(int x){
    int ret=0;
    while(x){
        ++ret;
        x/=10;
    }
    return ret;
}
int main()
{
    scanf("%s%s",a,b);
    int n=strlen(a),m=strlen(b);
    for(int i=0;i<n;++i)++tkrar[a[i]-'0'];
    for(int i=0;i<m;++i)--tkrar[b[i]-'0'];
    int l=0;
    for(int i=1;i<1e7;++i){
        if(i==len(n-i)){l=i;break;}
    }
    int x=n-l;
    while(x){
        --tkrar[x%10];
        x/=10;
    }
    bool st=0;
    string s;
    string ans,ans1=b;
    for(int i=0;i<10;++i){
        for(int j=0;j<tkrar[i];++j)ans1+=('0'+i);
    }
    for(int i=1;i<10;++i){
        if(tkrar[i]==0)continue;
        s=('0'+i);
        --tkrar[i];
        st=1;
        break;
    }
    if(!st){
        s=b;
        m=0;
    }
    for(int i=0;i<10;++i){
        for(int j=0;j<tkrar[i];++j)s+=('0'+i);
    }
    bool cmp=0;
    for(int i=0;i<s.size();++i){
        if(m==0){
            ans+=s[i];
            continue;
        }
        if(s[i]>b[0]&&(i||(b[0]>'0'))){
            ans+=b;
            ans+=s[i];
            m=0;
            continue;
        }
        if(s[i]<b[0]||cmp){
            ans+=s[i];
            continue;
        }
        if(i==0){
            ans+=s[i];
            continue;
        }
        string s2=b,s1;s1+=s[i];
        for(int j=i;j<s.size();++j)s2+=s[j];
        for(int j=i+1;j<s.size();++j){
            if(s[j]!=b[0]&&s[j-1]==b[0])s1+=b;
            s1+=s[j];
        }
        for(int j=0;j<s1.size();++j){
            cmp=1;
            if(s1[j]<s2[j])break;
            if(s2[j]<s1[j]){
                ans+=b;
                m=0;
                break;
            }
        }
        ans+=s[i];
    }
    if(m)ans+=b;
    if(ans.size()!=ans1.size())return 0;
    if(ans<ans1||ans1[0]=='0'){
        for(int i=0;i<ans.size();++i)printf("%c",ans[i]);
    }
    else
        for(int i=0;i<ans1.size();++i)printf("%c",ans1[i]);
}
