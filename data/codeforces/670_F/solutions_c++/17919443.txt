#include <iostream>
#include <sstream>

#define pb push_back
#define pf push_front
#define mk make_pair
#define ft first
#define sd second
#define rfile freopen("input.txt", "r", stdin);
#define wfile freopen("output.txt", "w", stdout);

typedef long long ll;
typedef unsigned long long ull;

using namespace std;

int a[10];

int main()
{
	//rfile;
	string s, z, pp, s1, s2, s3;
	stringstream ss;
	int ln;
	cin >> s >> z;
	for (int i = 1; i <= 1e6; i++)
	{
		string p;
		ss.clear();
		ss << i;
		ss >> p;
		if (s.length() - p.length() == i)
			{
				ln = i;
				pp = p;
				break;
			}
	}
	for (int i = 0; i < s.length(); i++)
		a[s[i] - '0']++;
	for (int i = 0; i < pp.length(); i++)
		a[pp[i] - '0']--;
	for (int i = 0; i < z.length(); i++)
		a[z[i] - '0']--;
	s1 = s2 = s3 = "z";
	s1 = z;
	for (int i = 0; i <= 9; i++)
		for (int j = 0; j < a[i]; j++)
			s1 += i + '0';
	if (s1[0] == '0')
        s1 = "z";
	int i = 1;
	/*for (int i = 0; i <= 9; i++)
        cout << a[i];
    cout << '\n';*/
	while (i <= 9 && !a[i])
		i++;
	if (i <= 9)
    {
        pp = i + '0';
        a[i]--;
    } else
    {
        cout << z;
        for (int i = 0; i <= 9; i++)
			for (int j = 0; j < a[i]; j++)
				cout << i;
        return 0;
    }
	for (i = 0; i <= 9; i++)
		for (int j = 0; j < a[i]; j++)
			pp += i + '0';
    string ans;
	for (int i = 1; i < pp.length(); i++)
	{
		if (pp[i] >= z[0] && (!(i - 1) || pp[i - 1] < z[0]))
		{
			s2 = pp;
			s2.insert(i, z);
		}
		if (pp[i] <= z[0] && (i + 1 == pp.length() || pp[i + 1] > z[0]))
		{
			s3 = pp;
			s3.insert(i + 1, z);
			break;
		}
	}
	if (pp.length() == 1)
		s3 = pp + z;
	ans = min(s1, s2);
	ans = min(ans, s3);
	if (ans == "z")
		ans = z;
	cout << ans;
	//cout << s1 << '\n' << s2 << '\n' << s3 << '\n';

	return 0;
}
