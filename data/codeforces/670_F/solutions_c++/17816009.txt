#include <iostream>
#include <cstdio>
using namespace std;

int n, m, sz, cnt[99];
string a, b, s, x, ans = "A";

int len(int p) {
    int i;
    for(i = 0; p > 0; i++, p /= 10);
    return i;
}

void f() {
	int t, k;
    t = len(n);
    k = n - t;
    if (t != len(k)) k++;
    sz = k - m;
	while(k) {
        cnt[(k % 10) + '0']--;
        k /= 10;
    }
}

void g() {
	int i;
	char c;
	for(c = '0'; c <= '9'; c++) s.append(cnt[c], c);
	if (b[0] != '0') {
		ans = min(ans, b + s);
	}
    for(i = 0; i < sz && s[i] == '0'; i++);
    if(i == sz) return;
	swap(s[0], s[i]);
    for(i = 1; i < sz && s[i] < b[0]; i++);
    x = s;
	ans = min(ans, x.insert(i, b));
    for(; i < sz && s[i] == b[0]; i++);
	ans = min(ans, s.insert(i, b));
}

int main() {
    int i;
    cin >> a >> b;
    if(a == "01" || a == "10") {puts("0"); return 0;}
    n = a.size();
    m = b.size();
    for(i = 0; i < n; i++) cnt[a[i]]++;
    for(i = 0; i < m; i++) cnt[b[i]]--;
    f();
    g();
	cout << ans;
    return 0;
}
