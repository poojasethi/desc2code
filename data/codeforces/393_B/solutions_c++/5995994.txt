#include<cstdio>
#define N 1000
int i,j,n;
int w[N][N];
double a[N][N],b[N][N];
int main()
{
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	for (j=1;j<=n;j++)
	{
		scanf("%d",&w[i][j]);
	}
	
	for (i=1;i<=n;i++)
	a[i][i]=w[i][i];
	for (i=1;i<=n;i++)
	for (j=1;j<=n;j++)
	if (i!=j)
	{
		a[i][j]=(w[i][j]+w[j][i])/2.0;
		b[i][j]=w[i][j]-a[i][j];
	}
	
		for (i=1;i<=n;i++)
		{
	for (j=1;j<=n;j++)
	printf("%.8f ",a[i][j]);printf("\n");
}
			for (i=1;i<=n;i++)
		{
	for (j=1;j<=n;j++)
	printf("%.8f ",b[i][j]);printf("\n");
}
}
