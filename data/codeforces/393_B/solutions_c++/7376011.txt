#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
	int n;
	cin >> n;
	vector<vector<int> > mat(n);
	int i, j;
	int read;
	for(i=0; i<n; ++i){
		for(j=0; j<n; ++j){
			cin >> read;
			mat[i].push_back(read);
		}
	}
	for(i=0; i<n; ++i){
		for(j=0; j<n; ++j){
			cout << ((double) mat[i][j] + (double) mat[j][i])*0.5;
			cout << " ";
		}
		cout << endl;
	}	
	for(i=0; i<n; ++i){
		for(j=0; j<n; ++j){
			cout << ((double) mat[i][j] - (double) mat[j][i])*0.5;
			cout << " ";
		}
		cout << endl;
	}	

	return 0;
}
