
#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

const int MAXN = 176;

double w[MAXN][MAXN];

int main(){
	ios::sync_with_stdio( false );
	
	int n; cin >> n;

	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++)
			cin >> w[i][j];

	for (int i = 0; i < n; i++, cout << "\n")
		for (int j = 0; j < n; j++)
			cout << (w[i][j] + w[j][i]) /2 << " ";

	for (int i = 0; i < n; i++, cout << "\n")
		for (int j = 0; j < n; j++)
			cout << (w[i][j] - w[j][i]) /2 << " ";

}
