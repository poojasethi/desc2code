#include "stdio.h"
float w[170][170],a[170][170],b[170][170];
int main(){
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++) scanf("%f",&w[i][j]);
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            a[i][j]=(w[i][j]+w[j][i])/2;
            b[i][j]=(w[i][j]-w[j][i])/2;
        }
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++) printf("%f ",a[i][j]);
        printf("\n");
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++) printf("%f ",b[i][j]);
        printf("\n");
    }
}
