#include <iostream>
#include <cstdio>
using namespace std;
int m[200][200];

int main(){
	int n; cin >> n;
	for(int i = 0; i < n; i++)
		for(int j = 0; j < n; j++)
			cin >> m[i][j];

	for(int i = 0; i < n; i++){
		for(int j = 0; j < n; j++){
			if(i == j) cout << m[i][j] << " ";
			else printf("%.4f ", (m[i][j]+m[j][i])/2.0);
		}
		cout << endl;
	}
	for(int i = 0; i < n; i++){
		for(int j = 0; j < n; j++){
			if(i == j) cout << "0 ";
			else printf("%.4f ", (m[i][j] - (m[i][j]+m[j][i])/2.0));
		}
		cout << endl;
	}
	return 0;
}
