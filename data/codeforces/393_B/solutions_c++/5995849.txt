#include<stdio.h>
int main()
{
	int n, i, j, w[200][200];
	double a[200][200], b[200][200];
	scanf("%d",&n);
	for (i=0;i<n;++i)
	{
		for (j=0;j<n;++j)
		{
			scanf("%d",&w[i][j]);
		}
	}
	for (i=0;i<n;++i)
	{
		for (j=i;j>=0;--j)
		{
			a[i][j]=(w[i][j]+w[j][i])/2.0;
			a[j][i]=a[i][j];
			b[i][j]=(w[i][j]-w[j][i])/2.0;
			b[j][i]=- b[i][j];
		}
	}
	for (i=0;i<n;++i)
	{
		for (j=0;j<n;++j)
		{
			printf("%f ",a[i][j]);
		}
		printf("\n");
	}
	for (i=0;i<n;++i)
	{
		for (j=0;j<n;++j)
		{
			printf("%f ",b[i][j]);
		}
		printf("\n");
	}
	return 0;
}
