#include<cstdio>
#include<cstring>
const int N = 200;
double w[N][N],a[N][N],b[N][N];
int main()
{
	int i,j,n;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)scanf("%lf",&w[i][j]);
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			a[i][j]=(w[i][j]+w[j][i])/2;
			b[i][j]=(w[i][j]-w[j][i])/2;
		}
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)printf("%.6lf ",a[i][j]);puts("");
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)printf("%.6lf ",b[i][j]);puts("");
	}
	return 0;
}
