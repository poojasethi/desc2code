#include<stdio.h>

int main()
{
	double w[170][170],a[170][170],b[170][170];
	int i,j,n;

	while(scanf("%d",&n)!=EOF)
	{	
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
				scanf("%lf",&w[i][j]);
		}
		
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				a[i][j] = (w[i][j] + w[j][i]) / 2;
				b[i][j] = (w[i][j] - w[j][i]) / 2;
			}
		}

		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
				printf("%.8lf%c",a[i][j],j+1==n?'\n':' ');
		}
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
				printf("%.8lf%c",b[i][j],j+1==n?'\n':' ');
		}
	}
	
	return 0;
}
