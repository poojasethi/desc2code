#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main() {
   int w[175][175];
   int n;
   cin>>n;
   for(int i=0;i<n;i++)
   {
   for(int j=0;j<n;j++)
   cin>>w[i][j];
   }
   for(int i=0;i<n;i++)
   {
   for(int j=0;j<n;j++)
   printf("%.8f\t",(float)(w[i][j]+w[j][i])/2);
   cout<<"\n";
   }
   for(int i=0;i<n;i++)
   {
   for(int j=0;j<n;j++)
   printf("%.8f\t",(float)(w[i][j]-w[j][i])/2);
   cout<<"\n";
   }
    return 0;
}
