#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int n,ar[180][180];
    cin>>n;
    for(int x=0; x<n; x++)
        for(int z=0; z<n; z++)
           cin>>ar[x][z];

   for(int x=0; x<n; x++){
        for(int z=0; z<n; z++){
           cout<<fixed<<setprecision(8)<<(ar[x][z]+ar[z][x])/2.0<<" ";}
           cout<<endl;}

   for(int x=0; x<n; x++){
        for(int z=0; z<n; z++){
           cout<<fixed<<setprecision(8)<<ar[x][z]-(ar[x][z]+ar[z][x])/2.0<<" ";}
           cout<<endl;}

    return 0;
}
