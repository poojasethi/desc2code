#include<cstdio>
using namespace std;
int n,i,j,map[175][175];
double a[175][175],b[175][175];
int main()
{
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	for (j=1;j<=n;j++)
	scanf("%d",&map[i][j]);
	for (i=1;i<=n;i++)
	for (j=i+1;j<=n;j++)
	{
		a[i][j]=a[j][i]=double(map[i][j]+map[j][i])/2;
		b[i][j]=b[j][i]=double(map[i][j])-a[i][j];
		b[j][i]*=-1;
	}
	for (i=1;i<=n;i++)
	a[i][i]=double(map[i][i]);
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=n;j++)
		printf("%.8lf ",a[i][j]);
		printf("\n");
	}
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=n;j++)
		printf("%.8lf ",b[i][j]);
		printf("\n");
	}
}
