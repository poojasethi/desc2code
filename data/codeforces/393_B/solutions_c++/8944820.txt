#include<cstdio>
#include<iostream>
#include<cstring>

int main(){
    int n;
    scanf("%d",&n);
    int a[n][n],b[n][n],c[n][n];
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            scanf("%d",&a[i][j]);
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++)
            printf("%.8f ",(a[i][j]+a[j][i])/2.0);

    }

}
