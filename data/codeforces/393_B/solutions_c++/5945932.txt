#include<iostream>
#include<vector>

using namespace std;

int main()
{
	int n,i,j;
	cin >> n;
	double w[n][n];
	double a[n][n];
	double b[n][n];
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			cin >> w[i][j];
		}
	}
	for(i=0;i<n;i++)
	{
		for(j=i;j<n;j++)
		{
			a[i][j]=(w[i][j]+w[j][i])/2;
			a[j][i]=a[i][j];
			b[i][j]=(w[i][j]-w[j][i])/2;
			if(i!=j)	b[j][i]=-b[i][j];
		}
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			cout << a[i][j] << " ";
		}
		cout << endl;
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			cout << b[i][j] << " ";
		}
		cout << endl;
	}
	return 0;
}
