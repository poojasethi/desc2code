#include <cstdio>
#include <cstring>
#include <algorithm>
#define rep(i,l,r) for (int i=l;i<=r;++i)
double w[200][200];
int n;
int main(){
	scanf("%d",&n);
	rep(i,1,n) rep(j,1,n)
		scanf("%lf",&w[i][j]);
	rep(i,1,n) rep(j,1,n)
		printf("%f%c",(w[i][j]+w[j][i])/2,j==n?'\n':' ');
	rep(i,1,n) rep(j,1,n)
		printf("%f%c",(w[i][j]-w[j][i])/2,j==n?'\n':' ');
}
