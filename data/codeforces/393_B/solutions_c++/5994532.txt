#include <bits/stdc++.h>

using namespace std;

#define rep(i,n) for(int i = 0 ; i < n ; i++)

int main() {

    double mat[200][200];
    int n;
    cin >> n;
    rep(i,n)
        rep(j,n)
            cin >> mat[i][j];
    rep(i,n)
     rep(j,n)
       printf("%lf%c" , (mat[i][j] + mat[j][i])/2 , " \n"[j == n-1]);

    rep(i,n)
     rep(j,n)
       printf("%lf%c" , (mat[i][j] - mat[j][i])/2 , " \n"[j == n-1]);
}
