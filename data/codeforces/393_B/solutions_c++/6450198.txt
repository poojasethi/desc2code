#include<stdio.h>
main()
{
    float a;
    int w[200][200],n,i,j;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            scanf("%d",&w[i][j]);
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            a=(w[i][j]+w[j][i])/2.0;
            printf("%.6f ",a);
        }
        printf("\n");
    }
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            a=(w[i][j]-w[j][i])/2.0;
            printf("%.6f ",a);
        }
        printf("\n");
    }
}
