#include<stdio.h>
int a[555][555];
double b[555][555],c[555][555];
int n;
int main()
{
int i,j;
scanf("%d",&n);
for(i=0;i<n;i++)
for(j=0;j<n;j++)
scanf("%d",&a[i][j]);
for(i=0;i<n;i++)
{
for(j=0;j<n;j++)
{
b[i][j] = (a[i][j] + a[j][i])/2.0;
c[i][j] = a[i][j] -b[i][j];
}
}

for(i=0;i<n;i++)
{
for(j=0;j<n;j++)
printf("%.8lf ",b[i][j]);
printf("\n");
}


for(i=0;i<n;i++)
{
for(j=0;j<n;j++)
printf("%.8lf ",c[i][j]);
printf("\n");
}

return 0;
}
