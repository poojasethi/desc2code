#include <iostream>
#include <string>

using namespace std;
int main(){
	int q;
	cin>>q;
	double a[q][q],b[q][q],w[q][q];
	for(int i=0;i<q;i++){
		for(int j=0;j<q;j++){
			cin>>w[i][j];
		}
	}
	for(int i=0;i<q;i++){
		for(int j=0;j<q;j++){
			a[i][j]=(w[i][j]+w[j][i])/2;
			b[i][j]=w[i][j]-a[i][j];
		}
	}
	cout<<fixed;
	cout.precision(8);
	for(int i=0;i<q;i++){
		for(int j=0;j<q;j++){
			cout<<a[i][j]<<" ";
		}
		cout<<endl;
	}
	for(int i=0;i<q;i++){
		for(int j=0;j<q;j++){
			cout<<b[i][j]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
