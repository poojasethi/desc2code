#include<stdio.h>
#include<algorithm>
using namespace std;

int main(){
    int i,j;
    double n,w[171][171];
    scanf("%lf",&n);
    for(i=1;i<=n;i++){
        for(j=1;j<=n;j++){
            scanf("%lf",&w[i][j]);
        }
    }
    for(i=1;i<=n;i++){
        for(j=1;j<=n;j++){
            printf("%lf ",(w[i][j]+w[j][i])/2);
        }
        printf("\n");
    }
    for(i=1;i<=n;i++){
        for(j=1;j<=n;j++){
            printf("%lf ",(w[i][j]-w[j][i])/2);
        }
        printf("\n");
    }
    return 0;
}
