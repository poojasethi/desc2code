#include <stdio.h>

double W[170][170], A[170][170], B[170][170];

int main (void) {
	int n, i, j;
	scanf ("%d", &n);
	for (i = 0; i < n; i++) 
		for (j = 0; j < n; j++)	scanf("%lf", &W[i][j]);
	
	for (i = 0; i < n; i++) {
		for (j = 0; j < n; j++) {
			A[i][j] = (W[i][j]+W[j][i])/2;
			B[i][j] = (W[i][j]-W[j][i])/2;
		}
	}
	for (i = 0; i < n; i++) {
		for (j = 0; j < n; j++)	printf ("%lf ", A[i][j]);
		printf ("\n");
	}

	for (i = 0; i < n; i++) {
		for (j = 0; j < n; j++)	printf ("%lf ", B[i][j]);
		printf ("\n");
	}
}
