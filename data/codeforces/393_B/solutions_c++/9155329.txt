#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    int n,i,j;
    cin >> n;
    double a[n][n];
    for (i = 0; i < n; i++) for (j = 0; j < n; j++) cin >> a[i][j];
    
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++)
            printf("%.8lf ",(a[i][j]+a[j][i])/2.0);
        cout << endl;        
    }
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++)
            printf("%.8lf ",(a[i][j]-a[j][i])/2.0);
        cout << endl;        
    }
    //system("pause");
    return 0;
}
