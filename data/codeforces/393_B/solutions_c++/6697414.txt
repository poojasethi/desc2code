#include <iostream>
#include <string.h>
using namespace std;
int w[200][200];
double a[200][200],b[200][200];
int main()
{
	int n,i,j,k;
	cin>>n;
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=n;j++)
		{
			cin>>w[i][j];
		}
	}
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=n;j++)
		{
			a[i][j]=0.5*(w[i][j]+w[j][i]);
			b[i][j]=0.5*(w[i][j]-w[j][i]);
		}
	}
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=n;j++)
		{
			cout<<a[i][j]<<" ";
		}
		cout<<endl;
	}
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=n;j++)
		{
			cout<<b[i][j]<<" ";
		}
		cout<<endl;
	}
}
