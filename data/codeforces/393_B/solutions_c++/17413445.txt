#include <stdio.h>
main()
{
    double mat[200][200];
    int len;
    scanf("%d",&len);
    for(int i=0;i<len;i++)
    {
        for(int m=0;m<len;m++)
            scanf("%lf",&mat[i][m]);
    }
    for(int i=0;i<len;i++)
    {
        for(int m=0;m<len;m++)
        {
            printf("%f ",(mat[i][m]+mat[m][i])/2.0);
        }
        printf("\n");
    }
    for(int i=0;i<len;i++)
    {
        for(int m=0;m<len;m++)
        {
            printf("%f ",mat[i][m]-(mat[i][m]+mat[m][i])/2);
        }
        printf("\n");
    }
}
