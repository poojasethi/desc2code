#include <cstdio>
#include <iostream>
#define long long ll
#define rep(i,a,b) for (int i=a;i<=b;++i)
#define d(x) cout<<x<<endl
using namespace std;

int n;
int w[1000][1000];
double a[1000][1000],b[1000][1000];

int main() {
	//freopen("1.in","r",stdin);
	scanf("%d", &n);
	rep(i,1,n) rep(j,1,n) scanf("%d", &w[i][j]);
	rep(i,1,n) rep(j,1,n) b[i][j] = (w[i][j] - w[j][i]) / 2.0;
	rep(i,1,n) rep(j,1,n) a[i][j] = w[i][j] - b[i][j];
	rep(i,1,n) {
		rep(j,1,n) printf("%.9f ",a[i][j]);
		puts("");
	}
	rep(i,1,n) {
		rep(j,1,n) printf("%.9f ",b[i][j]);
		puts("");
	}	
	return 0;
}
