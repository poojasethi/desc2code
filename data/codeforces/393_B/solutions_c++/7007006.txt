#include<bits/stdc++.h>
#define MAX 200
using namespace std;
double a[MAX][MAX],b[MAX][MAX],c[MAX][MAX];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			cin>>c[i][j];
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
		{
			a[i][j]=(c[i][j] + c[j][i])/2.0;
			b[i][j]=(c[i][j] - c[j][i])/2.0;
		}
	for( int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			cout<<a[i][j];
			if(j+1<n)
				cout<<" "; 
		}
		cout<<endl;
	}
	for( int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			cout<<b[i][j];
			if(j+1<n)
				cout<<" "; 
		}
		cout<<endl;
	}		
	
}
