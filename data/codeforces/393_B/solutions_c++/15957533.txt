#include<cstdio>
int main()
{
	int i,j,n;
	scanf("%d",&n);
	double w[n][n],a[n][n],b[n][n];
	for(i=0;i<n;i++)
		for(j=0;j<n;j++)
			scanf("%lf",&w[i][j]);
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			if(i!=j){
				a[i][j]=(w[i][j]+w[j][i])/2;
			}else
				a[i][j]=w[i][j];
		}
	}
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			if(i!=j){
				b[i][j]=w[i][j]-a[i][j];
			}else
				b[i][j]=0;
		}
	}
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			printf("%.8lf ",a[i][j]);
		}
		printf("\n");
	}
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			printf("%.8lf ",b[i][j]);
		}
		printf("\n");
	}
}
