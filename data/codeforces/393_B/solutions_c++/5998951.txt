#include<iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int w[n][n];
    for(int i=0; i<n; i++)
    for (int j=0; j<n; j++)
    cin>>w[i][j];
    for(int i=0; i<n; i++)
    {
       for(int j=0; j<n; j++)
       cout<<(w[i][j]+w[j][i])*1.0/2.0<<" ";
       cout<<endl;
    }
    for(int i=0; i<n; i++)
    {
        for(int j=0; j<n; j++)
        cout<<(w[i][j]-w[j][i])*1.0/2.0<<" ";
        cout<<endl;
    }
    //system("pause");
    return 0;
}
