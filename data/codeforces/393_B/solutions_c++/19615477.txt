#include <iostream>
#include<stdio.h>
#include<vector>
#include<cmath>
using namespace std;
#define mp make_pair
int n ;
double a[175][175];
int main() {
	//freopen("input.txt","r",stdin);
	scanf("%d",&n);
	for(int i=1;i<=n;++i) for(int j=1;j<=n;++j) scanf("%lf",&a[i][j]);

	for(int i=1;i<=n;++i){
		for(int j=1;j<=n;++j)
			printf("%.8lf ",(a[i][j]+a[j][i])/2);
	printf("\n");
	}

	for(int i=1;i<=n;++i){
			for(int j=1;j<=n;++j)
				printf("%.8lf ",(a[i][j]-a[j][i])/2);
		printf("\n");
		}

	return 0;
}
