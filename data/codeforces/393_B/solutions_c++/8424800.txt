#include<iostream>
using namespace std;

int main()
{
int n;
cin>>n;
double w[n][n],a[n][n],b[n][n];
for(int i=0;i<n;i++)
	for(int j=0;j<n;j++)
		cin>>w[i][j];
for(int i=0;i<n;i++)
{
a[i][i]=w[i][i];
b[i][i]=0.0000000;
}
for(int i=0;i<n;i++)
{
	for(int j=0;j<=i;j++)
	{
		if(i!=j)
		{
			a[i][j] = (w[i][j]+w[j][i])/2.0;
			b[i][j] = (w[i][j]-w[j][i])/2.0;
			a[j][i] = a[i][j];
			b[j][i] = -b[i][j]; 
		}
	}
}
for(int i=0;i<n;i++){
	for(int j=0;j<n;j++)
		cout<<a[i][j]<<" ";
	cout<<endl;
	}
for(int i=0;i<n;i++){
	for(int j=0;j<n;j++)
		cout<<b[i][j]<<" ";
	cout<<endl;
	}
return 0;
}
