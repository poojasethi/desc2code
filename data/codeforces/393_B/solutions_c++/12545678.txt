#include<bits/stdc++.h>
using namespace std;

int N, arr[172][172];

int main()
{
	scanf("%d", &N);
	for(int i=0; i<N; i++)
		for(int j=0; j<N; j++)
			scanf("%d", &arr[i][j]);
	for(int i=0; i<N; i++)
	{
		for(int j=0; j<N; j++)
		{
			if(j)printf(" ");
			printf("%.8lf", (double)(arr[i][j] + arr[j][i]) / 2.0);
		}
		printf("\n");
	}
	for(int i=0; i<N; i++)
	{
		for(int j=0; j<N; j++)
		{
			if(j)printf(" ");
			printf("%.8lf", (double)(arr[i][j] - arr[j][i]) / 2.0);
		}
		printf("\n");
	}
}
