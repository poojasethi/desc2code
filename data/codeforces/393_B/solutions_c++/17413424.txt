#include<stdio.h>
int main()
{
    int n;
    scanf("%d",&n);
    float a[171][171];
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            scanf("%f",&a[i][j]);
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            printf("%.8f ",(a[i][j]+a[j][i])/2);
        }
        printf("\n");
    }
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            printf("%.8f ",(a[i][j]-a[j][i])/2);
        }
        printf("\n");
    }
}
