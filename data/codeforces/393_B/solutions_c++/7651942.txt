#include <iostream>
using namespace std;
#define N 170
#define forall for(int i = 0; i < n; i++) for (int j = 0; j < n; j++)
double w[N][N], a[N][N], b[N][N];
int main() {
	int n;
	cin >> n;
	forall cin >> w[i][j];
	forall {
		a[i][j] = (w[i][j] + w[j][i])/2;
		b[i][j] = (w[i][j] - w[j][i])/2;
	}
	forall {
		cout << a[i][j] << " ";
		if (j == n-1)
			cout << endl;
	}
	forall {
		cout << b[i][j] << " ";
		if (j == n-1)
			cout << endl;
	}
}
