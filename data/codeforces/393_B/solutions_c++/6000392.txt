#include<iostream>
using namespace std;
#define N 171
double a[N][N],b[N][N],w[N][N];
int main()
{
	int n,i,j;
	while(cin>>n)
	{
		for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		cin>>w[i][j];
		for(i=0;i<n;i++)
		for(j=0;j<=i;j++)
		{
			b[i][j]=(w[i][j]-w[j][i])/2;
			b[j][i]=-b[i][j];
			a[i][j]=w[i][j]-b[i][j];
			a[j][i]=w[j][i]-b[j][i];
		}
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			cout<<a[i][j]<<' ';
			cout<<endl;
		}
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			cout<<b[i][j]<<' ';
			cout<<endl;
		}
	}
}
