#include<stdio.h>

int main(){

	int x,y,z,i,j;
	int arr[175][175];
	double A[2][175][175];
	int n;

	scanf("%d",&n);

	for(i=0 ; i<n ; i++)
		for(j=0 ; j<n ; j++)
			scanf("%d",&arr[i][j]);

	for(i=0 ; i<n ; i++){
		A[0][i][i]=arr[i][i];
		A[1][i][i]=0;
	}

	for(i=0 ; i<n ; i++)
		for(j=0 ; j<n ; j++){
			if(i!=j){

				A[0][i][j]=(arr[i][j]+arr[j][i])/2.0;
				
				A[1][i][j]=arr[i][j]-A[0][i][j];
			
			}
		}

	for(x=0 ; x<2 ; x++)
		for(i=0 ; i<n ; i++,puts(""))
			for(j=0 ; j<n ; j++)
				printf("%lf ",A[x][i][j]);

	return 0;

}
