#include<iostream>
#include<cstdio>
#define REP(a,b,c) for(short int a=b;a<c;a++)
using namespace std;

int main(void )
{
   short int n;
   cin>>n;
   short int W[n][n];
   REP(i,0,n)
      REP(j,0,n)
         cin>>W[i][j];
   REP(i,0,n)
      REP(j,0,n)
         printf("%f%c",(W[i][j]+W[j][i])/2.0,(j==n-1)?'\n':' ');
   REP(i,0,n)
      REP(j,0,n)
         printf("%f%c",(W[i][j]-W[j][i])/2.0,(j==n-1)?'\n':' ');
   return 0;
}
