#include <iostream>

int main() {
	int n; std::cin >> n;
	int matrix[n][n];
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < n; j++) {
			std::cin >> matrix[i][j];
		}
	}
	// W_i,j == A_i,j + B_i,j
	// A_i,j == A_j,i
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < n; j++) {
			if(j > 0) std::cout << ' ';
			std::cout << (matrix[i][j] + matrix[j][i]) / 2.0;
		}
		std::cout << std::endl;
	}
	// B_i,j == -B_j,i
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < n; j++) {
			if(j > 0) std::cout << ' ';
			std::cout << (matrix[i][j] - matrix[j][i]) / 2.0;
		}
		std::cout << std::endl;
	}
	return 0;
}
