#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
double a[200][200]= {0},b[200][200]= {0},s;
int main()
{
    int  n,i,j,k;
    cin>>n;
    for(i=0; i<n; i++)
        for(j=0; j<n; j++)
            scanf("%lf",&a[i][j]);
    for(i=0; i<n; i++)
    {
        for(j=0; j<n; j++)
        {
            b[i][j]=(a[i][j]+a[j][i])/2;
            printf("%.8f ",b[i][j]);
        }
        cout<<endl;
    }
    for(i=0; i<n; i++)
    {
        for(j=0; j<n; j++)
            printf("%.8f ",a[i][j]-b[i][j]);
        cout<<endl;
    }
}
