#!/usr/bin/env python


def read():
    n = int(raw_input())
    ret = []
    for i in range(n):
        ret.append(map(int, raw_input().split()))
    return ret


def work(mat):
    n = len(mat)
    
    A = [[0 for j in range(n)] for i in range(n)]
    B = [[0 for j in range(n)] for i in range(n)]

    for i in range(n):
        for j in range(n):
            A[i][j] = (mat[i][j] + mat[j][i]) / 2.0
            B[i][j] = (mat[i][j] - mat[j][i]) / 2.0

    for i in range(n):
        print " ".join(map(str, A[i]))

    for i in range(n):
        print " ".join(map(str, B[i]))


if __name__ == "__main__":
    work(read())
