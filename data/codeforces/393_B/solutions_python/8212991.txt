#!/usr/bin/env python

if __name__ == '__main__':
    n = int(raw_input())
    W = []
    for i in range(n):
        W.append(map(float, raw_input().split()))
    
    A = [[0.0 for i in range(n)] for j in range(n)]
    B = [[0.0 for i in range(n)] for j in range(n)]

    for i in range(n):
        for j in range(n):
            A[i][j] = A[j][i] = (W[i][j] + W[j][i]) / 2.0
            B[i][j] = (W[i][j] - W[j][i]) / 2.0
            if B[i][j]:
                B[j][i] = -B[i][j]
    
    for i in range(n):
        print ' '.join(map(str, A[i]))

    for i in range(n):
        print ' '.join(map(str, B[i]))
