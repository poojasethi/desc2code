n = int(raw_input())
W = []
for i in xrange(n):
    W.append(map(float, raw_input().split()))

for i in xrange(n):
    s = []
    for j in xrange(n):
        s.append(str( (W[i][j] + W[j][i])*.5))
    print ' '.join(s)

for i in xrange(n):
    s = []
    for j in xrange(n):
        s.append(str( (W[i][j] - W[j][i])*.5))
    print ' '.join(s)
