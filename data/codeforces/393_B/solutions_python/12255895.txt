n = int(raw_input())
W = []
for _ in xrange(n):
  W.append(map(float, raw_input().split()))

A = [[0]*n for _ in xrange(n)]
B = [[0]*n for _ in xrange(n)]
for i in xrange(n):
  for j in xrange(n):
    A[i][j] = (W[i][j]+W[j][i])/2
    B[i][j] = (W[i][j]-W[j][i])/2


for i in xrange(n):
  print ' '.join(map(str, A[i]))
for i in xrange(n):
  print ' '.join(map(str, B[i]))
