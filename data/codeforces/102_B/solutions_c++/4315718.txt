#include<cstdio>
#define myitoa(n,out) (sprintf((out),"%d",(n)))
int main()
{
	char ch,temp[100002];
	int sum=0,counter=0,i;
	scanf("%s",temp);
	while(1)
	{
		if (temp[1]=='\0')
			break;
		sum=0;
		for (i=0;temp[i];i++)
			sum+=temp[i]-'0';
		myitoa(sum,temp);
		counter++;
	}
	printf("%d",counter);
	return 0;
}
