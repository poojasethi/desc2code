#include <iostream>
#include <sstream>
#include <string>

using namespace std;

int main()
{
	string num;
	cin >> num;
	int cnt=0;
	while(num.size()>1)
	{
		int sum=0;
		for(int i=0; i<num.size(); i++)
			sum+=num[i]-'0';
		stringstream ss;
		ss << sum;
		ss >> num;
		cnt++;
	}
	cout << cnt << endl;
	return 0;
}
