#include<bits/stdc++.h>
using namespace std;
int main()
{
	int count=0,sum=0,t,k,i=0;
	char s[100000];
	gets(s);
	k=strlen(s);
	if(s[1])
	{
		for(i=0;i<k;i++)
		sum+=s[i]-'0';
		count=1;
	}
			
	while(sum>9)
	{
		t=0;
		while(sum)
		{
			t+=sum%10;
			sum=sum/10;
		}
		count++;
		sum=t;
	}
	cout<<count;
	return 0;
}
