#include <iostream>
#include <string>
#include <sstream>
using namespace std;
int main(){
	string s;
	cin>>s;
	int c=0;
	while(s.size()!=1){
		++c;
		int sum =0;
		for(int i=0;i<s.size();++i)
			sum+= s[i]-'0';
		string ret;
		while(sum!=0){
			ret= char('0'+sum%10)+ret;
			sum/=10;
		}
		s = ret;
	}
	cout<<c<<endl;
}
