#include <cstdio>
#include <cstring>

int main(){
	char c[1000000];
	scanf("%s",c);
	int x,lar,j=0;
	lar = strlen(c);
	if(lar==1){printf("0");return 0;}
	do{
		j++;
		x=0;
		lar=strlen(c);
		for(int i=0;i<lar;i++){
			x=x+c[i]-'0';	
		}	
		sprintf(c,"%d",x);
	}while(x>9);
	printf("%d",j);
	return 0;

}
