#include <iostream>
#include <cstdio>
using namespace std;
char s[100001];
int main(){
	int i, r,cnt=0;
	scanf("%s", s);
	while (s[1])
	{
		cnt++;
		r = 0;
		for (i = 0; s[i]; i++){
			r += s[i] - '0';
		}
		sprintf(s, "%d", r);
	}
	printf("%d\n", cnt);
	return 0;

}
