#include <cstdio>
#include <set>
#include <vector>
#include <algorithm>
using namespace std;

struct event
{
    int l,r,id,t;
};
inline bool operator <(const event &a,const event &b)
{
    return(a.r>b.r || a.r==b.r && a.id>b.id);
}
vector <event> s;
vector <int> X;
int ans[100010];
set <pair <int,int> > a[200010];
void add(int x,const pair <int,int> &p)
{
    for (int i=x;i<=X.size();i+=i&-i)
        a[i].insert(p);
}
int get(int x,int y)
{
    pair <int,int> ret=make_pair(1<<30,-1);
    for (int i=x;i;i-=i&-i)
    {
        set <pair <int,int> >::iterator p=a[i].lower_bound(make_pair(y,0));
        if (p!=a[i].end() && *p<ret)
            ret=*p;
    }
    return(ret.second);
}
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for (int i=1;i<=n;i++)
    {
        event tmp;
        scanf("%d%d%d",&tmp.l,&tmp.r,&tmp.t);
        tmp.id=i;
        s.push_back(tmp);
        X.push_back(tmp.l);
    }
    for (int i=1;i<=m;i++)
    {
        event tmp;
        scanf("%d%d%d",&tmp.l,&tmp.r,&tmp.t);
        tmp.id=-i;
        s.push_back(tmp);
        X.push_back(tmp.l);
    }
    sort(s.begin(),s.end());
    sort(X.begin(),X.end());
    X.erase(unique(X.begin(),X.end()),X.end());
    for (int i=0;i<s.size();i++)
    {
        int l=lower_bound(X.begin(),X.end(),s[i].l)-X.begin()+1;
        if (s[i].id>0)
            add(l,make_pair(s[i].t,s[i].id));
        else
            ans[-s[i].id]=get(l,s[i].t);
    }
    for (int i=1;i<=m;i++)
        printf("%d%c",ans[i],i==m?'\n':' ');
    return(0);
}
