int n,m,ans[111111],szt;

void add(int nd,int l,int r,int x,int v,int d){
	if(f[nd]<v){
		f[nd]=v;
		id[nd]=d;
	}
	if(r-l==1)return ;
	int mid=(l+r)/2;
	if(x>=mid)add(nd*2+1,mid,r,x,v,d);
	else add(nd*2,l,mid,x,v,d);
}
int get(int nd,int l,int r,int xl,int xr,int v){
	if(f[nd]<v)return -1;
	if(r-l==1)return id[nd];
	int ret=-1,mid=(l+r)/2;
	if(mid<=xl)return get(nd*2+1,mid,r,xl,xr,v);
	else if(mid>=xr)return get(nd*2,l,mid,xl,xr,v);
	else{
		ret=get(nd*2,l,mid,xl,mid,v);
		if(ret>0)return ret;
		return get(nd*2+1,mid,r,mid,xr,v);
	}
}
