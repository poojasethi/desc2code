#include<iostream>

using namespace std;

int main()
{
	int a[4];
	int b[4];
	int i,j,w,x,y,z,win1,win2;
	for(i=0;i<4;i++)
	{
		cin >> a[i] >> b[i];
	}
	
	w=a[0];
	x=b[1];
	if((b[2]>w && a[3]>x) || (b[3]>w && a[2]>x))	win1=2;
	else if(b[2]<w && a[3]<x && b[3]<w && a[2]<x)	win1=1;
	else	win1=0;

	w=a[1];
	x=b[0];
	if((b[2]>w && a[3]>x) || (b[3]>w && a[2]>x))	win2=2;
	else if(b[2]<w && a[3]<x && b[3]<w && a[2]<x)	win2=1;
	else	win2=0;

	if(win1==1 || win2==1)	cout << "Team 1\n";
	else if(win1==2 && win2==2)	cout << "Team 2\n";
	else cout << "Draw\n";
	return 0;
}
