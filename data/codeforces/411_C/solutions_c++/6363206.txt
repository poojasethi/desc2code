#include<cstdio>
#include<algorithm>
using namespace std;

const char s[3][10]={"Team 1","Draw","Team 2"};
int a[4],b[4];

inline int cal(int a1,int a2,int b1,int b2)
{
	if(a1>b2&&b1>a2)
		return 0;
	if(a2>b1&&b2>a1)
		return 2;
	return 1;
}

int main()
{
	int i,j,s2,s1;
	for(i=0;i<4;i++)
		scanf("%d%d",&a[i],&b[i]);
	for(s1=2,i=0;i<=1;i++)
	{
		for(s2=0,j=2;j<=3;j++)
		{
			s2=max(s2,cal(a[i],a[j],b[1-i],b[5-j]));
		}
		s1=min(s1,s2);
	}
	puts(s[s1]);
	return 0;
}
