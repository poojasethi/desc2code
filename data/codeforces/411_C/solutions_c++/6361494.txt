#include<stdio.h>


int a[5],b[5];

int check(int i,int j)
{
	if( a[3] > b[i] && b[2] > a[j] )return -1;
	if( a[2] > b[i] && b[3] > a[j] )return -1;
	if( a[3] >= b[i] || b[2] >= a[j] )return 0;
	if( a[2] >= b[i] || b[3] >= a[j] )return 0;
	return 1;
}

void solve()
{
	int c1,c2;
	c1 = check(0,1);
	c2 = check(1,0);
	if( c1 == 1 || c2 == 1 )puts("Team 1");
	else if( c1 == 0 || c2 == 0 )puts("Draw");
	else	puts("Team 2"); 
}

int main()
{
	for( int i = 0; i < 4; i++ )
		scanf(" %d %d",&a[i],&b[i]);

	solve();

	return 0;
}
