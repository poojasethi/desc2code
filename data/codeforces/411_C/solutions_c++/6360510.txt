#include <cstdio>

int a[5], b[5];

bool win(int x, int y) {
    return (x > b[3]) && (y > a[4]) && (x > b[4]) && (y > a[3]);
}

bool lose(int x, int y) {
    return ((x < b[3]) && (y < a[4])) || ((x < b[4]) && (y < a[3]));
}

int main()
{
    for (int i = 1; i <= 4; ++i)
        scanf("%d %d", &a[i], &b[i]);
    if (win(a[1], b[2]) || win(a[2], b[1]))
        printf("Team 1\n");
    else if (lose(a[1], b[2]) && lose(a[2], b[1]))
        printf("Team 2\n");
    else
        printf("Draw\n");
    return 0;
}
