#include <cstdio>
#include <algorithm>
using namespace std;

int a , d , c , x[4] , y[4] ;

int main() 
{
   for( int i=0 ; i<4 ; i++) scanf("%d%d",&x[i],&y[i]);
   for( int i=0 ; i<2 ; i++) 
   {
      if(!i) a=y[0] , d=x[1] ; 
      else   a=y[1] , d=x[0] ;

      if(a>x[2] && d>y[3] && a>x[3] && d>y[2])  { c=2 ; break ; } 
      else if(!((a<x[2] && d<y[3])||(a<x[3] && d<y[2]))) c=1 ; 
   }
   if(c==2) printf("Team 1") ; 
   else if(!c) printf("Team 2") ; 
   else printf("Draw") ;

 return 0;
}
