#include<iostream>
#include<string>
using namespace std;
typedef long long ll;
ll a[5],b[5];
ll f(ll fi, ll se, ll th, ll fo)
{
    return ((a[fi]>b[fo])&&(b[se]>a[th]));
}
int main()
{
    for(ll i=1;i<=4;i++)
        cin>>a[i]>>b[i];
    if((f(1,2,3,4)&&f(1,2,4,3))||(f(2,1,3,4)&&f(2,1,4,3)))
        cout<<"Team 1";
    else if((f(3,4,1,2)||f(4,3,1,2))&&(f(3,4,2,1)||f(4,3,2,1)))
        cout<<"Team 2";
    else
        cout<<"Draw";
}
