#include <iostream>
using namespace std;

struct team{int a,d,a1,d1;} A,B;

void solve(int a,int b){
	if(a>B.d && a>B.d1 && b>B.a1 && b>B.a) { cout<<"Team 1"; return;}
	if((B.a>b && B.d1>a)||(B.a1>b && B.d>a)) {cout<<"Team 2"; return;}
	cout<<"Draw";
}


int main(){

	cin>>A.a>>A.d>>A.a1>>A.d1;
	cin>>B.a>>B.d>>B.a1>>B.d1;
	
	if(A.a+A.d1>A.a1+A.d)
		 solve(A.a,A.d1);
        else	
		 solve(A.a1,A.d);
}
