#include <iostream>

using namespace std;

int a[4],b[4];
int g[2];

int main()
{
  for (int i=0;i<4;i++)
    cin>>a[i]>>b[i];
  for (int i1=0;i1<2;i1++) {
    int i2=1-i1;
    int j1=2,j2=3;
    if ((a[i1]<b[j1] and b[i2]<a[j2]) or (a[i1]<b[j2] and b[i2]<a[j1]))
      g[i1]=1;
    else if ((a[i1]>b[j1] and b[i2]>a[j2]) and (a[i1]>b[j2] and b[i2]>a[j1]))
      g[i1]=0;
    else
      g[i1]=-1;
  }
  if (g[0]==0 or g[1]==0)
    cout<<"Team 1"<<endl;
  else if (g[0]==1 and g[1]==1)
    cout<<"Team 2"<<endl;
  else
    cout<<"Draw"<<endl;
}
