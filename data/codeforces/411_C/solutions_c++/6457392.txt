#include <iostream>

using namespace std;

int main()
{
	int a[4];
	int b[4];

	for (int i = 0; i < 4; ++i)
		cin >> a[i] >> b[i];

	if ((a[0] > b[3] && b[1] > a[2] && a[0] > b[2] && b[1] > a[3]) ||
	    (a[1] > b[3] && b[0] > a[2] && a[1] > b[2] && b[0] > a[3]))
	{
		cout << "Team 1" << endl;
		return 0;
	}	

	if (((a[2] > b[1] && b[3] > a[0]) || (a[3] > b[1] && b[2] > a[0])) &&
            ((a[2] > b[0] && b[3] > a[1]) || (a[3] > b[0] && b[2] > a[1])))
	{
		cout << "Team 2" << endl;
		return 0;
	}

	cout << "Draw" << endl;
	return 0;
}
