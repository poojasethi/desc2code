#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int a,b,c,d;
    int aa,bb,cc,dd;
    while(cin>>aa>>a>>bb>>b)
    {
        cin>>cc>>c>>dd>>d;
        if(a>cc&&a>dd&&bb>c&&bb>d)
            cout<<"Team 1"<<endl;
        else if(b>cc&&b>dd&&aa>c&&aa>d)
            cout<<"Team 1"<<endl;
        else if((a<cc&&d>bb||a<dd&&c>bb)&&(b<cc&&d>aa||b<dd&&c>aa))
            cout<<"Team 2"<<endl;
        else cout<<"Draw"<<endl;
    }
}
