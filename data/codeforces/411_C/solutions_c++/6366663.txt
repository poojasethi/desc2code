#include<iostream>
#include<cstdio>
using namespace std;
int a[5], b[5];
int t1, t2;
int f(int t, int d)
{
	if(a[3] > d && b[4] > t)
		return 1;
	if(a[4] > d && b[3] > t)
		return 1;
	if(a[3] < d && b[4] < t && b[3] < t && a[4] < d)
		return -1;
	return 0;
}
int main()
{
	for(int i = 1; i <= 4; i++)
		scanf("%d%d", &a[i], &b[i]);
	t1 = f(a[1], b[2]);
	t2 = f(a[2], b[1]);
	if(t1 == 1 && t2 == 1)
		printf("Team 2\n");
	else if(t1 == -1 || t2 == -1)
		printf("Team 1\n");
	else
		printf("Draw\n");
	return 0;
}
