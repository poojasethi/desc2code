#include <iostream>

using namespace std;

bool can(int a1, int b1, int a2, int b2) {
	return (a1 > b2 && b1 > a2);
}

int main() {
	int a[4], b[4];
	for (int i = 0; i < 4; i++)
		cin >> a[i] >> b[i];
	if ((can(a[0], b[1], a[2], b[3]) && can(a[0], b[1] , a[3], b[2])) || (can(a[1], b[0], a[2], b[3]) && can(a[1], b[0] , a[3], b[2])))
		cout << "Team 1" << endl;
	else if ((can(a[2], b[3], a[0], b[1]) || can(a[3], b[2] , a[0], b[1])) && (can(a[2], b[3], a[1], b[0]) || can(a[3], b[2] , a[1], b[0])))
		cout << "Team 2" << endl;
	else
		cout << "Draw" << endl;
	return 0;
}
