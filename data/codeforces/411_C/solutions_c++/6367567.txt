#include <cstdio>

int a[4], b[4];

int win(int u, int v, int w, int x);

int main()
{
	for(int i=0; i<4; i++) scanf("%d%d", &a[i], &b[i]);

	if(win(0, 1, 2, 3) && win(0, 1, 3, 2)) {
		printf("Team 1\n");
	}
	else if(win(1, 0, 2, 3) && win(1, 0, 3, 2)) {
		printf("Team 1\n");
	}
	else if((win(2, 3, 0, 1) || win(3, 2, 0, 1)) && (win(2, 3, 1, 0) || win(3, 2, 1, 0))) {
		printf("Team 2\n");
	}
	else {
		printf("Draw\n");
	}

	return 0;
}

int win(int u, int v, int w, int x)
{
	if(a[u]>b[x] && b[v]>a[w]) return 1;
	return 0;
}
