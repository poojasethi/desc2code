#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
int a[5],b[5],t1,t2,t;
int solved(int k1,int k2,int k3,int k4){
	if(a[k1]>b[k4] && b[k2]>a[k3])return 1;
	if(a[k3]>b[k2] && b[k4]>a[k1])return -1;
	return 0;
}
int main(){
	for(int i=1;i<=4;i++)scanf("%d%d",&a[i],&b[i]);
	t1=min(solved(1,2,3,4),solved(1,2,4,3));
	t2=min(solved(2,1,3,4),solved(2,1,4,3));
	t=max(t1,t2);
	if(t>0)printf("Team 1\n");
	if(t<0)printf("Team 2\n");
	if(t==0)printf("Draw\n");
	return 0;
}
