#include <bits/stdc++.h>

typedef long long ll;

using namespace std;

const int N = 4;

int a[N], b[N];

int main () {
	for (int i = 0; i < N; i++)
		scanf ("%d%d", &a[i], &b[i]);
	if ((a[0] > max(b[2], b[3]) && b[1] > max(a[2], a[3])) || (a[1] > max(b[2], b[3]) && b[0] > max(a[2], a[3])))
		printf ("Team 1");
	else {
		bool first = (b[2] > a[0] && a[3] > b[1]) || (b[3] > a[0] && a[2] > b[1]);
		bool second = (b[2] > a[1] && a[3] > b[0]) || (b[3] > a[1] && a[2] > b[0]);
		if (first && second)
			printf ("Team 2");
		else
			printf ("Draw");
	}
}
