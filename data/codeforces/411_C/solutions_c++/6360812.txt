#include <cstdio>
#include <algorithm>
using namespace std;

int a[4],b[4],cnt[2][2];

int main(){
 for(int i=0;i<4;i++)
  scanf("%d %d",&a[i],&b[i]);
 for(int i=0;i<2;i++){
 for(int j=0;j<2;j++){
  if(a[0]>b[3] && b[1]>a[2])
   cnt[i][0]++;
  else if(a[2]>b[1] && b[3]>a[0])
   cnt[i][1]++;
  swap(a[2],a[3]);
  swap(b[2],b[3]);
 }
  swap(a[0],a[1]);
  swap(b[0],b[1]);
 }
 if(cnt[0][0]==2 || cnt[1][0]==2)
  puts("Team 1");
 else if(cnt[0][1]>0 && cnt[1][1]>0)
  puts("Team 2");
 else
  puts("Draw");
 return 0;
}
