#include <cstdio>

const int N = 4;
char ans[][16] = {"Draw", "Team 1", "Team 2"};
int a[N], b[N];

int get_ans() {
  if((a[0] > b[2] && b[1] > a[3] && a[0] > b[3] && b[1] > a[2]) ||
     (a[1] > b[2] && b[0] > a[3] && a[1] > b[3] && b[0] > a[2]))
    return 1;
  else if(((a[2] > b[0] && b[3] > a[1]) || (a[3] > b[0] && b[2] > a[1])) &&
          ((a[2] > b[1] && b[3] > a[0]) || (a[3] > b[1] && b[2] > a[0])))
    return 2;
  return 0;
}

int main() {
  for(int i = 0; i < N; ++i)
    scanf("%d%d", &a[i], &b[i]);
  puts(ans[get_ans()]);
}
