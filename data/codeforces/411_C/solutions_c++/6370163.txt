#include <iostream>
using namespace std;

int a[4],d[4];

int check(int a,int d,int A,int D){
    if (a>D && d>A) return 100;
    if (a<D && d<A) return -300;
    return 0;
}

int main (){
    int P1,P2,P3,P4,ans;
    for (int i=0;i<4;++i) cin>>d[i]>>a[i];
    P1=check(a[0],d[1],a[2],d[3]);
    P2=check(a[0],d[1],a[3],d[2]);
    P3=check(a[1],d[0],a[2],d[3]);
    P4=check(a[1],d[0],a[3],d[2]);
    ans=max(P1+P2,P3+P4);
    if (ans==200) cout<<"Team 1"<<endl;
    else if (ans>=0) cout<<"Draw"<<endl;
    else cout<<"Team 2"<<endl;
    return 0;
}
