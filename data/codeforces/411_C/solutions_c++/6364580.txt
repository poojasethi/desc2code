#include <iostream>
#include <cmath>
using namespace std;
int a[5],b[5];
int main()
{
	for(int i=1;i<=4;i++)
		cin>>a[i]>>b[i];
	if((a[1]>max(b[3],b[4]) && b[2]>max(a[3],a[4])) || (a[2]>max(b[3],b[4]) && b[1]>max(a[3],a[4])))
	{
		cout<<"Team 1"<<endl;
		return 0;
	}
	int judge=0;
	if((b[3]>a[1] && a[4]>b[2]) || (b[4]>a[1] && a[3]>b[2]))
		judge++;
	if((b[3]>a[2] && a[4]>b[1]) || (b[4]>a[2] && a[3]>b[1]))
		judge++;
	if(judge==2)
	{
		cout<<"Team 2"<<endl;
		return  0;
	}
	cout<<"Draw"<<endl;
	return 0;
}
