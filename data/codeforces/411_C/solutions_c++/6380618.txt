#include <iostream>
using namespace std;

int a[4],d[4];

int c(int a,int d,int A,int D){
    int k; (a>D && d>A) ? k=1 : ((a<D && d<A) ?  k=-3 : k=0);
    return k;
}

int main (){
    for (int i=0;i<4;++i) cin>>d[i]>>a[i];
    int ans=max(c(a[0],d[1],a[2],d[3])+c(a[0],d[1],a[3],d[2]),c(a[1],d[0],a[2],d[3])+c(a[1],d[0],a[3],d[2]));
    (ans==2) ? cout<<"Team 1" : (ans>=0) ? cout<<"Draw" : cout<<"Team 2";
}
