a = []
b = []
for i in range(4):
  u,v = map(int, raw_input().split())
  a.append(u)
  b.append(v)

res = 'Team 2'
for i in range(2):
  ii = 1 - i
  ok = True
  for j in range(2,4):
    jj = 5 - j
    ok &= not (a[i] < b[j] and b[ii] < a[jj])
  if ok:
    res = 'Draw'

for i in range(2):
  ii = 1 - i
  ok = True
  for j in range(2,4):
    jj = 5 - j
    ok &= a[i] > b[j] and b[ii] > a[jj]
  if ok:
    res = 'Team 1'
