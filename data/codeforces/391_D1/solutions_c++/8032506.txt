    int n,m;
    int a,b,c;
    int p,q,r;
    int res=0;
    cin>>n>>m;
    VVI v(n, VI(3)), h(m, VI(3));
    FL(i,0,n)   cin>>v[i][0]>>v[i][1]>>v[i][2];
    FL(i,0,m)   cin>>h[i][0]>>h[i][1]>>h[i][2];

    FL(i,0,n)
        FL(j,0,m)
        {
            p=v[i][0], q=v[i][1], r=v[i][2];
            a=h[j][0], b=h[j][1], c=h[j][2];
            if(a<p and a+c>p and q<b and q+r>b)
            {
                res=max(res,min(p-a, min(a+c-p, min(b-q, q+r-b))));
            }
        }
    cout << res;
}
