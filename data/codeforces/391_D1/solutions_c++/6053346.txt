
#include <iostream>
#include <utility>
#include <vector>

using namespace std;

pair<int, int> horiz[1001];
pair<int, int> ver[1001];
int lenh[1001];
int lenv[1001];

int main()
{

	int n, m, maxi = 0;
	cin>>n>>m;
	for (int i = 0; i < n; i++) cin>>ver[i].first>>ver[i].second>>lenv[i];
	for (int i = 0; i < m; i++) cin>>horiz[i].first>>horiz[i].second>>lenh[i];

	for (int i = 0; i < n; i++) 
	{

		for (int j = 0; j < m; j++) if (horiz[j].first < ver[i].first && horiz[j].first + lenh[j] > ver[i].first && ver[i].second < horiz[j].second && ver[i].second + lenv[i] > horiz[j].second) maxi = max(maxi, min(min(lenv[i] + ver[i].second - horiz[j].second, horiz[j].second - ver[i].second), min(lenh[j] + horiz[j].first - ver[i].first, ver[i].first - horiz[j].first)));

	}

	cout<<maxi;

	return 0;

}
