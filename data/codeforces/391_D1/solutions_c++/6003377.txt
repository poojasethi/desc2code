int main()
{
   int n, m;
   ll ans= 0;
   vector < pair <pair<int,int> , int > >v, w;
   cin >> n >> m;
   for (int i=0; i<n; i++)
   {
      int x, y, z;
      cin >> x >> y >> z;
      w.push_back(make_pair(make_pair(x,y),z));
   }
   for (int i=0; i<m; i++)
   {
      int x, y, z;
      cin >> x >> y >> z;
      v.push_back(make_pair(make_pair(x,y),z));
   }
   for (int i=0; i<m; i++)
   {
      for (int j=0; j<n; j++)
      {
         int t=1000000000;
         int x1=v[i].first.first;
         int y1=v[i].first.second;
         int z1=v[i].second;
         int x2=w[j].first.first;
         int y2=w[j].first.second;
         int z2=w[j].second;
         if ((x1<x2 && x1+z1<x2) || (x1>x2))
            continue;
         if ((y2<y1 && y2+z2<y1) || (y2>y1))
            continue;
         t=min(t,abs(x2-x1));
         t=min(t,abs(x1+z1-x2));
         t=min(t,abs(y1-y2));
         t=min(t,abs(y2+z2-y1));
         if (t>ans)
            ans=t;
      }
   }
   cout << ans;
}
