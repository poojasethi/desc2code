#include <iostream>
#include <algorithm>
using namespace std;
#define x first
#define y second

int main()
{
	int n, m;
	cin >> n >> m;

	pair<int, int> v[1005][2], h[1005][2];
	for (int i = 0; i < n; i++)
	{
		int xi, yi, len;
		cin >> xi >> yi >> len;
		v[i][0] = make_pair(xi, yi);
		v[i][1] = make_pair(xi, yi + len);
	}
	for (int i = 0; i < m; i++)
	{
		int xi, yi, len;
		cin >> xi >> yi >> len;
		h[i][0] = make_pair(xi, yi);
		h[i][1] = make_pair(xi + len, yi);
	}

	int maxsz = 0;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			if (h[j][0].x < v[i][0].x && v[i][0].x < h[j][1].x && v[i][0].y < h[j][0].y && h[j][0].y < v[i][1].y)
			{
				int cx = v[i][0].x, cy = h[j][0].y;
				int sz = cx - h[j][0].x;
				sz = min(sz, h[j][1].x - cx);
				sz = min(sz, cy - v[i][0].y);
				sz = min(sz, v[i][1].y - cy);
				maxsz = max(maxsz, sz);
			}

	cout << maxsz << "\n";

	return 0;
}
