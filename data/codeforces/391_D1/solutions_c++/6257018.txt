int x[2][MAXN],y[2][MAXN],l[2][MAXN];

int main(){

    int n,m;
    S("%d%d",&n,&m);
    REP(i,n)
    {
        S("%d%d%d",&x[0][i],&y[0][i],&l[0][i]);
    }
    REP(i,m)
    {
        S("%d%d%d",&x[1][i],&y[1][i],&l[1][i]);
    }

    int ans = 0;
    REP(i,n)
        REP(j,m)
        {
            int xx = x[0][i];
            int yy = y[1][j];

            if(x[1][j] <= xx && xx <=x[1][j]+l[1][j] &&
               y[0][i] <= yy && yy <=y[0][i]+l[0][i])
            {
                int dis = min(yy-y[0][i],y[0][i]+l[0][i]-yy);
                dis = min(dis, min(xx-x[1][j],x[1][j]+l[1][j]-xx));
                ans = max(ans, dis);
            }
        }

    P("%d",ans);
}
