#include<stdio.h>
#include<algorithm>

struct _pt {
  int x,y;
};

_pt N[1100][2];
_pt M[1100][2];

int main() {
  int n,m;
  int x,y,l;
  int sol=0;
  scanf("%d%d", &n,&m);
  for (int i=0; i<n; i++) {
    scanf("%d%d%d", &x,&y,&l);
    N[i][0].x = x;
    N[i][0].y = y;
    N[i][1].x = x;
    N[i][1].y = y+l;
  }
  for (int i=0; i<m; i++) {
    scanf("%d%d%d", &x,&y,&l);
    M[i][0].x = x;
    M[i][0].y = y;
    M[i][1].x = x+l;
    M[i][1].y = y;
  }
  for (int i=0; i<n; i++) {
    for (int j=0; j<m; j++) {
      if (N[i][0].y < M[j][0].y && N[i][1].y > M[j][1].y && M[j][0].x < N[i][0].x && M[j][1].x > N[i][1].x) {
        sol = std::max(sol, std::min(std::min(std::min(M[j][0].y-N[i][0].y, N[i][1].y-M[j][1].y), N[i][0].x-M[j][0].x), M[j][1].x-N[i][1].x));
      }
    }
  }
  printf("%d", sol);
}
