#include <iostream>

using namespace std;

#define F first
#define S second

int t[1005];
pair <int, int> p[1005];

int main()
{
    int n, m;
    cin >> n >> m;
    for (int i = 0; i < n; i++)
    {
        cin >> t[i] >> p[i].F >> p[i].S;
        p[i].S += p[i].F;
    }
    int ans = 0;
    for (int i = 0; i < m; i++)
    {
        int x1, x2, y;
        cin >> x1 >> y >> x2;
        x2 += x1;
        for (int j = 0; j < n; j++)
            ans = max(ans, min(min(t[j] - x1, x2 - t[j]), min(y - p[j].F, p[j].S - y)));
    }
    cout << ans << endl;
    return 0;
}
