#include<stdio.h>
int sort(int x,int y,int a[]){
	int i=x,j=y,mid=a[(i+j)/2],t;
	while(i<j){
		while(a[i]<mid) i++;
		while(a[j]>mid) j--;
		if(i<=j){
			t=a[i];
			a[i]=a[j];
			a[j]=t;
			i++;
			j--;
		}
	}
	if(i<y) sort(i,y,a);
	if(x<j) sort(x,j,a);
}
int main(){
	int n,m,x[2002],y[2002],l[2002],c_x,c_y,max=-1,a[4],i,j,min;
	scanf("%d%d",&n,&m);
	for(i=0;i<n+m;i++){
		scanf("%d%d%d",&x[i],&y[i],&l[i]);
	}
	for(i=0;i<n;i++){
		for(j=n;j<n+m;j++){
			c_x=x[i];
			c_y=y[j];
			a[0]=c_x-x[j];
			a[1]=x[j]+l[j]-c_x;
			a[2]=y[i]+l[i]-c_y;
			a[3]=c_y-y[i];
			sort(0,3,a);
			min=a[0];
			max=max>min?max:min;
		}
	}
	if(max>0)	printf("%d",max);
	else printf("0");
}
