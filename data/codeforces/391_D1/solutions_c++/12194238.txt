#include<iostream>
#define min(a,b) a<b?a:b
#define max(a,b) a>b?a:b
using namespace std;
int n,m;
struct node{
	int x,y,l;
};
node a[50005],b[50005];
int mi,mx;
int main()
{
	cin>>n>>m;
	mx=0;
	for(int i=0;i<n;i++) cin>>a[i].x>>a[i].y>>a[i].l;
	for(int i=0;i<m;i++) cin>>b[i].x>>b[i].y>>b[i].l;
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			//cout<<b[j].y<<" "<<a[i].x<<endl;
			if(b[j].y>=a[i].y&&b[j].y<=a[i].y+a[i].l&&a[i].x>=b[j].x&&a[i].x<=b[j].x+b[j].l){
				mi=a[i].x-b[j].x;
				mi=min(mi,b[j].x+b[j].l-a[i].x);
				mi=min(mi,a[i].y+a[i].l-b[j].y);
				mi=min(mi,b[j].y-a[i].y);
			//	cout<<mi<<endl;
			}
			mx=max(mx,mi);
		}
	}
	cout<<mx<<endl;
}
