#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <cstring>
#define N 1005

using namespace std;
const int INF = (int)2e9+7;
typedef pair<int, int> ii;
typedef pair<ii, int> iii;

int n, m, ans=-INF;

iii line1[N], line2[N];

void read() {
    cin>>n>>m;
    for (int i=0;i<n;i++)
        cin>>line1[i].first.first>>line1[i].first.second>>line1[i].second;
    for (int i=0;i<m;i++)
        cin>>line2[i].first.first>>line2[i].first.second>>line2[i].second;
}

int intersect(iii a, iii b){
    if (b.first.first > a.first.first || b.first.first + b.second < a.first.first) return 0;

    if (a.first.second > b.first.second || a.first.second + a.second < b.first.second) return 0;

    return min(min(b.first.second - a.first.second, a.first.second + a.second - b.first.second),min(a.first.first - b.first.first,b.first.first + b.second - a.first.first));
}

void solve() {
    for (int i=0;i<n;i++)
        for (int j=0;j<m;j++)
            ans = max(ans, intersect(line1[i],line2[j]));
}

void result() {
    cout << ans;
}

int main() {
    read();
    solve();
    result();
}
