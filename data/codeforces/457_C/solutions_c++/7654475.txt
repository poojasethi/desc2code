#include<bits/stdc++.h>
using namespace std;
#define pii pair<int,int>
vector<int>V[100005];
vector<pii>U;
int n;
int query(int x)
{
    int i;
    int ans=0,cnt=0;
    for(i=0;i<n;i++)
        if(U[i].second>=x)
            ans+=U[i].first,
            cnt++;
    for(i=0;cnt<x;i++)
        if(U[i].second<x)
            ans+=U[i].first,
            cnt++;
    return ans;
}
int main()
{
    //freopen("a.txt","r",stdin);
    int i,j;
    while(~scanf("%d",&n)){
        U.clear();
        for(i=0;i<=n;i++)  V[i].clear();
        for(i=1;i<=n;i++){
            int u,v;
            scanf("%d%d",&u,&v);
            V[u].push_back(v);
        }
        for(i=0;i<=1e5;i++){
            sort(V[i].begin(),V[i].end(),greater<int>());
            for(j=0;j<V[i].size();j++)
                U.push_back(pii(V[i][j],j+1));
        }
        sort(U.begin(),U.end());
        int l=1,r=n;
        while(r-l>2){
            int ll=(l*2+r)/3;
            int rr=(l+r*2)/3;
            int vl=query(ll);
            int vr=query(rr);
            if(vl<=vr)  r=rr;
            else l=ll;
        }
        int ans=1e9;
        for(i=l;i<=r;i++)
            ans=min(ans,query(i));
        printf("%d\n",ans);
    }
    return 0;
}
