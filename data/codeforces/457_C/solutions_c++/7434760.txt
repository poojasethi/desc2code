#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
#define N 110000
int w[N],num,n,ans,sum,za,zb,s;
bool p[N];
struct rec{
	int x,y,k;
}a[N],b[N];
bool cmp(rec a,rec b){
	return a.x<b.x || a.x==b.x && a.y>b.y;
}
bool cmp2(rec A,rec B){
	return A.y<B.y;
}
bool cmp3(rec a,rec b){
	return a.k<b.k;
}
int main(){
	scanf("%d",&num);
	for(int i=1;i<=num;i++){
	  int x,y;
	  scanf("%d%d",&x,&y);
	  if(!x)s++;
	  else n++,a[n].x=x,a[n].y=y;
	}
	sort(a+1,a+n+1,cmp);
	a[1].k=1;
	for(int i=2;i<=n;i++)
	  if(a[i].x!=a[i-1].x)a[i].k=1;
	  else a[i].k=a[i-1].k+1;
	sort(a+1,a+n+1,cmp3);
	for(int i=1;i<=n;i++)b[i].x=i,b[i].y=a[i].y;
	sort(b+1,b+n+1,cmp2);
	for(int i=1;i<=n;i++)w[b[i].x]=i;
	ans=0;
	for(int i=1;i<=n;i++)ans+=a[i].y;
	sum=ans;
	
	za=n;
	zb=n;
	s+=n;
	for(int i=num;i;i--){
		while(za && a[za].k>=i){
			if(w[za]>zb)sum+=a[za].y,s++;
			p[w[za]]=1;
			za--;
		}
		while(s>i && zb>0){
		  if(!p[zb]){
			sum-=b[zb].y;
			s--;
			}
			zb--;
		}
		ans=min(ans,sum);
	}
	cout<<ans<<endl;
	return 0;
}
