#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;
int n,k;
long long int to=0;
int arr[3002];
int main()
{
	cin>>n>>k;
	for(int i=0;i<n;i++)
	{
		int a,b;
		cin>>a>>b;
		arr[a]+=b;
	}
	for (int i = 1; i < 3002; i++)
	{
		to += min(arr[i],k);
		arr[i] -= min(k,arr[i]);	
		arr[i+1] += min(arr[i],k);
	}
	
	cout << to ;
	
	return 0;
}
