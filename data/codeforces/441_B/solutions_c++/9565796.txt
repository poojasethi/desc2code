#include<vector>
#include<iostream>
using namespace std;

const int mx=3050;

int main() {
	ios_base::sync_with_stdio(false); cin.tie(NULL);
	int n,v,t=0,a,b, ret=0,p,q; cin>>n>>v;
	vector<int> x(mx,0), y(mx,0);
	for (int i=0; i<n; i++)
		cin>>a>>b, x[a]+=b, t=max(t,a);
	for (int i=0; i<t+10; i++)
		p=min(v,y[i]), q=min(v-p,x[i]),
		y[i+1]+=(x[i]-q), ret+=(p+q);
	cout << ret << endl;
	return 0;
}
