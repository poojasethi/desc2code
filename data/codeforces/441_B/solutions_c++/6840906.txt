#include <iostream>
#include <algorithm>
using namespace std;
int c[3005];
int main( void )
{
  int n,v,a,b;
  cin >> n >> v;
  for( int i=0; i<n; i++ ){
    cin >> a >> b;
    c[a] += b;
  }
  int ret=0;
  for( int i=1; i<3005; i++ ){
    int d = min( v, c[i-1] );
    if( d < v ){
      int e = min( c[i], v-d );
      c[i] -= e;
      d += e;
    }
    ret += d;
  }
  cout << ret << endl;
  return 0;
}
