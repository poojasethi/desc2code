#include <iostream>
using namespace std;
int n,v,d,f,a[3005],res=0;
int main(){
    cin>>n>>v;
    for (int i=0; i<n; i++) {cin>>d>>f, a[d]+=f;}
    for (int i=1; i<3002; i++) {
        if (a[i-1]>=v) res+=v;
        else{
            if (a[i]>v-a[i-1]) {res+=v; a[i]-=v-a[i-1];}
            else {res+=a[i]+a[i-1]; a[i]=0;}
        }
    }
    cout<<res;
    return 0;
}
