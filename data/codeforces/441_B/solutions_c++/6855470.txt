#include <bits/stdc++.h>
using namespace std;

int n, v, res, arr[3030];

int main()
{
	cin >> n >> v;
	for(int a, b, i=0; i<n; i++)
		cin >> a >> b, arr[a]+=b;

	for(int i=1; i<=3001; i++) // :/
	{
		int vv = v;
		res += min(vv, arr[i-1]);
		vv -= min(vv, arr[i-1]);
		res += min(arr[i], vv);
		arr[i] -= min(vv, arr[i]);
	}

	cout << res << endl;

	return 0;
}
