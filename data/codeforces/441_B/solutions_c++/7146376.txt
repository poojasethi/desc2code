#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,v,a[3010]={0},i,j,k,ans=0;
	cin>>n>>v;
	for(i=0;i<n;i++)
	{
		cin>>j>>k;
		a[j]+=k;
	}
	for(i=1;i<=3001;i++)
	{
		j=v;
		while(a[i-1] && j>0)
		{
			a[i-1]--;
			j--;
		}
		while(a[i] && j>0)
		{
			a[i]--;
			j--;
		}
		ans+=v-j;
	}
	cout<<ans<<endl;
	return 0;
}
