#include <iostream>

using namespace std;

int n, k, mx, day[3001], ans, a, b;

int main()
{
	cin >> n >> k;
	
	for (int i = 0; i< n; i++)
	{
		cin >> a >> b;
		
		day[a] += b;
	}
	
	for (int i = 1; i < 3002; i++)
	{
		ans += min(day[i],k);
		
		day[i] -= min(k,day[i]);
		
		day[i+1] += min(day[i],k);
	}
	
	cout << ans;
}
