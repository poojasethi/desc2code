#include<stdio.h>
#include<cstdlib>
#include<algorithm>
using namespace std;
int v[3012];
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
			int x,y;
			scanf("%d%d",&y,&x);
			v[y]+=x;
	}
	int ans=0;
	for(int i=0;i<=3000;i++){
			int a,b;
			a=min(v[i],m);
			ans+=a;
			b=min(v[i+1],m-a);
			ans+=b;
			v[i+1]-=b;
	}
 	printf("%d\n",ans);
   scanf("%d",&n);
}
