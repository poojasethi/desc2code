#include<bits/stdc++.h>
using namespace std;
int f[3333];
int main(){
	int n,v;
	scanf("%d%d",&n,&v);
	int a,b;
	for(int i = 0;i<n;i++){
		scanf("%d%d",&a,&b);
		f[a] += b;
	}
	int ans = 0;
	for(int i = 1;i<=3001;i++){
		if(f[i-1] < v){
			int t = min(v,f[i-1]+f[i]);
			ans += t;
			f[i] -= t-f[i-1];
		}else{
			ans += v;
		}
	}
	printf("%d",ans);
	return 0;
}
