#include<cmath>
#include<iostream>
using namespace std;
int main() {
  int n, v,a,b,ans=0; cin>>n>>v; int x[30003];
  for (int i=0;i<n;i++) {cin>>a>>b; x[a] += b;}

  for (int i=1; i<3002;i++) {
    int m= min(v,x[i-1]);
    ans+=m;
    if (v>m) {ans+= min(v-m,x[i]); x[i]-= min(v-m,x[i]);}}
cout<<ans;
 }
