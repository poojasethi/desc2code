#include <stdio.h>
int main()
{
	int n,v,s,s1,a[3005]={0},i,sum=0;
	scanf("%d  %d",&n,&v);
	while(n--)
	{
		scanf("%d  %d",&s,&s1);
		a[s]+=s1;
	}
	for(i=1;i<3005;i++)
	{
		if(a[i]>v)
		{
			sum+=v;
			a[i]-=v;
			if(a[i]>v)
				a[i+1]+=v;
			else
				a[i+1]+=a[i];
		}
		else
			sum+=a[i];
	}
	printf("%d\n",sum);
	return 0;
}
