#include<iostream>
using namespace std;
int main(){
	int arr[3002]={0};
	int a,b,n,v;
	cin>>n>>v;
	for(int i=0;i<n;i++){
		cin>>a>>b;
		arr[a]+=b;
	}
	int cur=0,now=0;
	int ans=0;
	for(int i=1;i<=3001;i++){
		now=arr[i];
		if(now+cur<=v){
			ans+=now+cur;
			cur=0;
		}
		else{
			ans+=v;
			cur=now-max(v-cur,0);
		}
	}
	cout<<ans;
	return 0;
}
