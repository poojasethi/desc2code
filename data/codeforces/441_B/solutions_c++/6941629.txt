#include <iostream>

using namespace std;

int x[3010], ans, one, two, n, v, a, b;

int main ()
{
    cin >> n >> v;
    for (int i=0;i<n;i++)
    {
        cin >> a>> b;
        x[a]+= b;
    }
    for (int i=0;i<3002;i++)
    {
        one= min (x[i], v);
        ans+= one;
        two= min(x[i+1],v-one);
        ans+= two;
        x[i+1]-= two;
    }

    cout <<ans;
}
