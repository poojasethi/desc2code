#include <bits/stdc++.h>
using namespace std;
const int N = 3001;
int d[N];
int main(){
	int n,v,i,a,b,ret=0;
	cin>>n>>v;
	for(i=0;i<n;++i){
		cin>>a>>b;
		d[a]+=b;
	}
	int u,x;
	for(i=1;i<=N;++i){
		u=0;
		u= min(d[i-1],v);
		d[i-1]-=u;
		ret+=u;x=0;
		x= min(v-u,d[i]);d[i]-=x;
		ret+=x;
	}
	cout<<ret<<endl;
}
