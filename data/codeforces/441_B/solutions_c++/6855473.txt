#include <bits/stdc++.h>
using namespace std;

int n, v, res, arr[3030];

int main()
{
	cin >> n >> v;
	for(int a, b, i=0; i<n; i++)
		cin >> a >> b, arr[a]+=b;

	for(int i=1; i<=3001; i++)
	{
		res += min(v, arr[i-1]);
		res += max(0, min(v-arr[i-1], arr[i]));
		arr[i] -= max(0, min(v-arr[i-1], arr[i]));
	}

	cout << res << endl;

	return 0;
}
