#include <iostream>
#include <cstdio>
using namespace std;

int A[3010];

int main()
{
	int n, v, a, b, i, x, y, ans;
	cin>>n>>v;
	for (i=0; i<n; i++) {
		cin>>a>>b;
		A[a]+=b;
	}
	ans=min(v,A[0]);
	A[0]-=ans;
	for (i=1; i<3002; i++) {
		x = min(v, A[i-1]);	A[i-1] -= x; 
		y = min(v-x, A[i]); A[i] -= y; 
		ans += x+y;
	}
	cout<<ans<<endl;
	return 0;
}
