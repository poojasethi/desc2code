#include <iostream>
#include <map>
using namespace std;

int main() {
    int n, v;
    cin >> n >> v;
    map<int, int> A;
    for (int i = 0; i < n; ++i) {
	int a, b;
	cin >> a >> b;
	A[a] += b;
    }
    int ret = 0, rem = 0;
    for (int i = 0; i < 4000; ++i) {
	int grem = min(rem, v);
	int gcur = min(v-grem, A[i]);
	ret += (grem + gcur);
	rem = A[i] - gcur;
    }
    cout << ret << endl;
}
