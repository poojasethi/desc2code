#include<bits/stdc++.h>
using namespace std;
int main()
{int n1,v,x[3009]={0},y[3009]={0},i,j=0,sum=0,m,n;
cin>>n1>>v;
for(i=0;i<n1;i++)
{cin>>n>>m;
x[n]+=m;}
for(i=1;i<3002;i++)
{if(y[i-1]<=v)
{
    sum+=y[i-1];
    j=v-y[i-1];
    if(x[i]>j)
        {y[i]=x[i]-j;sum+=j;}
    else
        sum+=x[i];
}
else
{    sum+=v;y[i]=x[i];}}
cout<<sum;
return 0;}
