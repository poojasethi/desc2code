#include<stdio.h>

int main()
{
	int n,v,i,d,p,s=0,a[3002]={},b[3002]={};
	scanf("%d %d",&n,&v);
	for (i=1;i<=n;i++)
	{
		scanf("%d %d",&d,&p);
		a[d]+=p;
	}
	for (i=1;i<=3001;i++)
	{
		d=v;
		if (d>=b[i]){d=d-b[i];s+=b[i];}
		else {s+=d;d=0;}
		if (d>=a[i]) {s+=a[i];a[i]=0;}
		else {a[i]=a[i]-d;s+=d;}
		b[i+1]=a[i];
	}
	printf("%d",s);
}
