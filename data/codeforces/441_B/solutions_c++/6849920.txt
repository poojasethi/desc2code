#include<bits/stdc++.h>

using namespace std;
int main()
{
    int n,a,b, p,ans=0;
    int arr[10000]= {0};
    int arr2[10000]= {0};
    cin>>n>>p;
    for (int i = 0; i < n; i++)
	{
		cin >> a >> b;
		arr[a] += b;
	}

	for(int i=0;i<10000;i++)
	{
	    int x= min(p,arr[i]+arr2[i]);
	    arr2[i+1]= max(arr[i]-max(x-arr2[i],0),0);
	    ans+=x;
	}
    cout<<ans<<endl;
}
