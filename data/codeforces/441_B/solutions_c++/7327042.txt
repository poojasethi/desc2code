#include <cstdio>
#include <algorithm>

using namespace std;

int d[3002];

int main() {
	int n, v, sol = 0;
	scanf("%d%d", &n, &v);

	for(int i = 0; i < n; i++) {
		int a, b;
		scanf("%d%d", &a, &b);
		d[a] += b;
	}

	for(int i = 1; i < 3002; i++) {
		int posso = v;
		int ont = min(posso, d[i-1]);
		posso -= ont;
		int hj = min(posso, d[i]);
		sol += hj + ont;
		d[i] -= hj;
	}
	printf("%d\n", sol);
}
