#include<iostream>
using namespace std;
int main()
{
	int n, v, a, b, c[3002] = {0};
	int sum = 0;
	cin>>n>>v;
	for( int i = 0; i < n; i++ ){
		cin>>a>>b;
		c[a] += b;
	}
	for( int i = 1; i < 3002; i++ ){
		int x = min( c[i-1], v );
		sum += x;
		if( x < v ){
			sum += min( c[i], v-x );
			c[i] -= min( c[i], v-x );
		}
	}
	cout<<sum<<endl;
	return 0;
}
