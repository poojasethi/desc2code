#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false); cin.tie(0);
using namespace std;

int n, v, b, c, a[3002], s;

int main() { _
  cin >> n >> v;
  for (int i = 0, d; i < n; i++) {
    cin >> d >> b;
    a[d-1] += b;
  }
  for (int i = 0; i < 3001; i++) {
    s += a[i];
    int x = min(v, s);
    c += x;
    s = max(0, min(s-x, v));
  }
  cout << c << "\n";
  return 0;
}
