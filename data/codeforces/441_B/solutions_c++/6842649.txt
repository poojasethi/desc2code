#include <iostream>
#include <algorithm>
#include <set>
#include <map>
using namespace std;

int n, v, ans;
int f[111111];
int p[111111];

int main() {
	int aa, bb;
	cin >> n >> v;
	for (int i = 0; i < n; i++)
	{
		cin >> aa >> bb;
		f[aa] += bb;
	}
	for (int i = 0; i < 9999; i++)
	{
		int x = min(v, f[i] + p[i]);
		p[i+1] = max(f[i] - max(x-p[i],0), 0);
		ans += x;
	}
	cout << ans;
	return 0;
}
