
 #include<cstdio>
#include<algorithm>
using namespace std;
int a1[3001];
int main()
{
    int n,k,a,b,o=0;
    scanf("%d%d",&n,&k);
    while(n--)
    {
        scanf("%d%d",&a,&b);
        a1[a]+=b;
    }
    a=k;
    for(int i=0;i<=3000;++i)
    if(a1[i]>a)
    {
        b=min(a1[i]-a,k);
        o+=a+b,a=k-b;
    }
    else o+=a1[i],a=k;
    printf("%d\n",o);
}
