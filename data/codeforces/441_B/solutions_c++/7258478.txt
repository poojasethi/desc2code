#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
#define LL  long long
LL j,k,n,v,a[3001],ans;
int main()
{
	scanf("%d%d",&n,&v);for(int i=1;i<=n;i++)scanf("%d%d",&j,&k),a[j]+=k;k=v;
	for(int i=0;i<=3000;i++)if(a[i]>k)j=min(a[i]-k,v),ans+=j+k,k=v-j;else ans+=a[i],k=v;cout<<ans<<endl;return 0;
}
