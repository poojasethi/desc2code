#include <iostream>
#include <algorithm>
using namespace std;
int f[3010]={0};
int main()
{
	int n,v,day=0,a,b,r=0,t;
	cin>>n>>v;
	for(int i=1;i<=n;i++){
		cin>>a>>b;
		f[a]+=b;
	}
	for(day=0;day<=3005;day++){
		if(f[day]<=v) {r+=f[day];continue;}
		t=f[day]-v;
		if(t>v) f[day+1]+=v;
		else f[day+1]+=t;
		r+=v;
	}
	cout<<r<<endl;
	return 0;
}
