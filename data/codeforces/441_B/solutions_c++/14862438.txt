#include <cstdio>
#include <algorithm>
using namespace std;
int a[3001];
int main()
{
	int n,k,x,y,ans=0;
	scanf("%d%d",&n,&k);
	while(n--)
	{
		scanf("%d%d",&x,&y);
		a[x]+=y;
	}
	x=k;
	for(int i = 0; i<= 3000; i++)
		if(a[i] > x)
		{
			y = min(a[i]-x ,k);
		 	ans+=x+y;
		 	x=k-y;
	 	}
		else 
		{
			ans+=a[i];
			x=k;
		}
	printf("%d\n",ans);
}
