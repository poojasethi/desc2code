#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main(){

 ll n,v,x,y;
 cin>>n>>v;
 ll int fr[3005];
 memset(fr,0,sizeof(fr));
 for (int i=0;i<n;i++){
   cin>>x>>y;
   fr[x]+=y;
 }
 ll int ans=0;
 for (int i=1;i<3002;i++){
  ans+= min(fr[i],v);
  fr[i]-=min(v,fr[i]);
  fr[i+1]+=min(fr[i],v);
 }

 cout<<ans;
}
