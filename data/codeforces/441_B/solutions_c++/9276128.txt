#include <iostream>

using namespace std;
int a[100000];
int res;
int main()
{
 int n;
 cin >> n;
 int v;
 cin >> v;
for(int i = 1; i <= n; i++)
{ int x, y;
 cin >> x >> y;
 a[x] += y;
}
 for(int i = 1; i <= 3001; i++)
{
int can = v;
int t = min (a[i - 1], can);
a[i - 1] -= t;
can -= t;
res += t;
t = min (a[i], can);
a[i] -= t;
res += t;
}
cout << res;
}
