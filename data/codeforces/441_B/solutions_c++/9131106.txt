#include <stdio.h>
main()
{
	int n,v,b,c,i;
	int a[3010]={0},s=0;
	scanf("%d%d",&n,&v);
	while(n--)
	{
		scanf("%d%d",&b,&c);
		a[b]+=c;
	}
	for(i=1;i<3010;i++)
	{
		if(a[i]>v)
		{
			a[i]-=v;
			s+=v;
			if(a[i]>v)
				a[i+1]+=v;
			else
				a[i+1]+=a[i];
		}
		else s+=a[i];
	}
	printf("%d\n",s);
}
