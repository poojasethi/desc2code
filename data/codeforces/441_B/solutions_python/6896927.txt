from sys import stdin

first_line = stdin.readline()
n = int(first_line.split()[0])
daily_capacity = int(first_line.split()[1])
trees = []
capacity = {}

for i in range(0, n):
	line = stdin.readline()
	trees.append(  ( int(line.split()[0]), int(line.split()[1]) )  )
	if not int(line.split()[0]) in capacity:
		capacity[int(line.split()[0])] = daily_capacity
	if not int(line.split()[0]) + 1 in capacity:
		capacity[int(line.split()[0]) + 1] = daily_capacity

trees.sort(key=lambda tup: tup[0])
total_harvested = 0

for tree in trees:
	harvested_today = min(capacity[ tree[0] ], tree[1])
	total_harvested += harvested_today
	capacity[ tree[0] ] = capacity[ tree[0] ] - harvested_today
	harvested_tomorrow = min(capacity[tree[0] + 1], tree[1] - harvested_today)
	total_harvested += harvested_tomorrow
	capacity[ tree[0] + 1] = capacity[ tree[0] + 1] - harvested_tomorrow

print total_harvested
