nv = [int(a) for a in raw_input().split()]
v = nv[1]; n=nv[0]
tree = [0]*3010
harvest = 0
leftover = 0

for i in range(nv[0]):
    ab = [int(a) for a in raw_input().split()]
    tree[ab[0]]+= ab[1]

for day in range(1,3010):
    harvest += min(v, leftover)
    space = max(0, v - leftover)
    leftover = 0
    
    harvest += min(space, tree[day])
    leftover = tree[day] - min(space, tree[day])

print harvest
