n, v = map(int, raw_input().split())
tree = {}
for _ in range(n):
    a, b = map(int, raw_input().split())
    tree[a] = tree.get(a, 0) + b
for a, b in sorted(tree.items(), reverse=True):
    if tree.get(a+1, 0) < v:
        t = min(b, v-tree.get(a+1, 0))
        tree[a] -= t
        tree[a+1] = tree.get(a+1, 0) + t
ans = sum(min(v, b) for a, b in tree.items())
print ans
