n,v=map(int,raw_input().split())
ans=chk=0
l=[0]*3002
for i in range(n):
    a,b=map(int,raw_input().split())
    l[a]+=b

for i in range(3002):
    h=0
    if chk<=v:
        h=chk
    else:
        h=v
    chk=l[i]
    if h<v:
        if v>=chk+h:
            h+=chk
            chk=0
        else:
            chk-=(v-h)
            h=v
    ans+=h
print ans
