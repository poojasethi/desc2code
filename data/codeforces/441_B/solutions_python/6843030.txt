if __name__=="__main__":
    n, v = map(int, raw_input().split())

    maxN = 3005
    b1 = [0]*maxN
    b2 = [0]*maxN

    c = 0
    while c < n:
        a,b = map(int, raw_input().split())
        b1[a] += b
        b2[a+1] += b
        c+=1

    s = 0
    c = 0
    leftover = 0
    while c < maxN:
        s += min(b1[c]+leftover, v)
        leftover = max(min(b1[c]+leftover -v,b1[c]), 0)
        c += 1

    print s
