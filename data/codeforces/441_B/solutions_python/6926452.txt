n, k = [int(c) for c in raw_input().strip().split()]
ar = [[int(c) for c in raw_input().strip().split()] for i in xrange(n)]
ar.sort()
ans = 0
fruits = [0] * (max(ar)[0] + 4)
for d, f in ar:
	fruits[d] += f
for day in xrange(1,max(ar)[0] + 2):
	limit = k
	for i in xrange(day-1, day+1):
		offset = min(limit, fruits[i])
		ans, limit, fruits[i] = ans + offset, limit - offset, fruits[i] - offset
print ans
