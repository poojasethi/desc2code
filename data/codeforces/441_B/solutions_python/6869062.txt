import sys 
n, v = [int(x) for x in sys.stdin.readline().split()]

b=[0]*3001;
for i in range(n): # read rest of lines
    line=sys.stdin.readline()
    a=int(line.split()[0])
    b[a-1]+=int(line.split()[1])
sum = 0
remains=0
for i in range (3001):
   if i == 0:
       take=min(v,b[i])
       remains=b[i]-take
   else:
       take=min(v,b[i]+remains)
       remains=min(b[i], b[i]+remains-take)
   sum+=take
print sum
