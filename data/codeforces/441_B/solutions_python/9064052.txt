nv = [ int(x) for x in raw_input().split() ]
n = nv[0]; v = nv[1]

tree = [0] * 3010

for i in range(n):
    ab = [ int(x) for x in raw_input().split() ]
    tree[ab[0]] += ab[1]

harvest = 0
leftover = 0

for i in range(3010):
    space = v

    harvest += min(leftover, v)
    space -= min(leftover, v)

    if tree[i] <= space:
        space -= tree[i]
        harvest += tree[i]
        leftover = 0
    else:
        harvest += space
        leftover = tree[i] - space

print harvest
