[n,v] = [int(s) for s in raw_input().split()]
fruits = [0] * 3002
total = 0
for i in xrange(n):
   t = [int(s) for s in raw_input().split()]
   fruits[t[0]] += t[1]

leftover = 0
for i in xrange(1,3002):
   m = leftover + fruits[i]
   collected = min(m,v)
   total += collected
   leftover = min(m-collected,fruits[i])
print total
