n, v = map(int, raw_input().split(' '))

total = 0
first = [0 for _ in xrange(3003)]
second = [0 for _ in xrange(3003)]
for i in xrange(n):
    ai, bi = map(int, raw_input().split(' '))
    first[ai] += bi
    second[ai + 1] += bi

for i in xrange(3002):
    cv = v
    from_second = min(second[i], v)
    cv -= from_second
    total += from_second
    from_first = min(first[i], cv)
    second[i + 1] -= from_first
    total += from_first

print total
