n,v=map(int,raw_input().split())
a,b=[],[]
for i in range(n):
    x,y=map(int,raw_input().split())
    a.append(x)
    b.append(y)
f=[0 for i in range(max(a)+2)]
for i in range(n):
    f[a[i]]+=b[i]
c=0
left=0
for i in range(1,len(f)):
    ct=v
    if left>v:
        c+=v
        ct=0
    else:
        c+=left
        ct=v-left
    if ct>f[i]:
        c+=f[i]
        left=0
    else:
        c+=ct
        left=f[i]-ct
print c
