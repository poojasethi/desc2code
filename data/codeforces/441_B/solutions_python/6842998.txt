[n, v] = map(int, raw_input().split())
trees = dict()
dayMax = 0
for i in xrange(n):
  [a, b] = map(int, raw_input().split())
  if a not in trees.keys():
    trees[a] = b
  else:
    trees[a] += b
  dayMax = max(dayMax, a+2)

total = 0
for d in xrange(1, dayMax):
  canPick = v

  picked = 0
  if canPick > 0 and d-1 in trees.keys():
    picked = min(canPick, trees[d-1])
    trees[d-1] -= picked
    canPick -= picked
  total += picked
  
  picked = 0
  if canPick > 0 and d in trees.keys():
    picked = min(canPick, trees[d])
    trees[d] -= picked
    canPick -= picked
  total += picked
  
print total
