from sys import stdin
readline = stdin.readline

n, v = map(int, readline().split())
Fruits = [[0, 0] for i in xrange(3001)]
for i in xrange(n):
	a, b = map(int, readline().split())
	Fruits[a-1][0] += b

ans = 0
for i in xrange(3000):
	if v >= (Fruits[i][0] + Fruits[i][1]):
		ans += (Fruits[i][0] + Fruits[i][1])
		continue
	ans += v
	Fruits[i+1][1] += min(Fruits[i][0], (Fruits[i][0] + Fruits[i][1] - v))

ans += min(v, Fruits[3000][1])
print ans
