n,v=(map(int,raw_input().split()))

b=[0 for i in range (3001)]
for i in range(n):
    a=(map(int,raw_input().split()))
    b[a[0]]+=a[1]

pluk,over,nu=0,0,0

for i in range(3001):
    nu=b[i]
    if nu+over<=v:
        pluk+=nu+over
        over=0
    else:
        pluk+=v
        over=nu-max(v-over,0)
if over>=v:
    pluk+=v
else:
    pluk+=over
print pluk
