def solve(fruits, v):
    last_day = max(fruits.keys()) + 1
    result = 0
    for day in range(1, last_day + 1 + 1):
        dv = v
        yesterday = min(dv, fruits.get(day - 1, 0))
        dv -= yesterday
        result += yesterday
        if day - 1 in fruits:
            fruits[day - 1] -= yesterday
        today = min(dv, fruits.get(day, 0))
        result += today
        if day in fruits:
            fruits[day] -= today
    return result
        
n, v = map(int, raw_input().split())
fruits = {}
for i in range(n):
    a, b = map(int, raw_input().split())
    if a in fruits:
        fruits[a] += b
    else:
        fruits[a] = b
print solve(fruits, v)
