# Codeforces 441B
n, v = map(int, raw_input().split())
f = [ 0 ] * (3000+3)
for i in xrange(n):
	d, t = map(int, raw_input().split())
	f[d] += t
total = 0
for i in xrange(1, 3000+2):
	remove_old = min(f[i-1], v)
	remove_new = min(f[i], v-remove_old)
	f[i] -= remove_new
	total += remove_old + remove_new
print total
