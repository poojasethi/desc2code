MAXN = 3000

def solve(n, v, t):
  h = [0]*(MAXN+2)
  for a, b in t:
    h[a] += b
  res = 0
  for i in range(1, MAXN+2):
    r = v
    x = min(r, h[i-1])
    res, r, h[i-1] = res+x, r-x, h[i-1]-x
    x = min(r, h[i])
    res, r, h[i] = res+x, r-x, h[i]-x
  return res

n, v = map(int, raw_input().split())
t = [map(int, raw_input().split()) for i in range(n)]
print solve(n, v, t)
