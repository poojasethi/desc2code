line = map(int, raw_input().split())
n = line[0]
v = line[1]
prod = {}
for i in range(n):
	line = map(int, raw_input().split())
	if line[0] in prod:
		prod[line[0]] += line[1]
	else:
		prod[line[0]] = line[1]
max_day = max(prod)
for i in range(1, max_day + 2):
	if i in prod:
		if prod[i] > v:
			if prod[i] > 2 * v: prod[i] = 2 * v
			if (i + 1) in prod:
				prod[i + 1] += prod[i] - v
			else:
				prod[i + 1] = prod[i] - v
			prod[i] = v
print sum(prod.values()[:max_day+1])
