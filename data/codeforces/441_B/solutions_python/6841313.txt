K = 1
if K == 0:
	inp = open("input.txt", "r")
	R=lambda:map(int, inp.readline().strip().split(' '))
else:
	R=lambda:map(int, raw_input().strip().split(' '))

n, v= R()
arr = [0] * n
for i in xrange(n):
	arr[i] = R()

arr.sort()
cnt = [0] * 3002
for item in arr:
	d_a = item[0]
	cnt[d_a] = cnt[d_a] + item[1]
	if cnt[d_a] > v:
		cnt[d_a + 1] = cnt[d_a + 1] + cnt[d_a] - v
	if cnt[d_a] > v:
		cnt[d_a] = v
	if cnt[d_a + 1] > v:
		cnt[d_a + 1] = v
print sum(cnt)
