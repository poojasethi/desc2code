n,v = map(int, raw_input().split())
ans,a,b,c,d = [],[],[],[],[0 for i in range(3003)]
for i in range(n):
    aa,bb = map(int, raw_input().split())
    a.append(aa)
    b.append(bb)
    c.append((aa,bb))
    d[aa] += bb
c.sort()
cnt = 0
t = 0
rem = 0
for day in range(min(a), max(a)+2):
    t = 0
    t += min(rem, v)
    if t<v:
        x = min(v-t, d[day])
        rem = max(0,d[day]-x)
        t+=x
    else:
        rem = d[day]
    cnt += t
print cnt
