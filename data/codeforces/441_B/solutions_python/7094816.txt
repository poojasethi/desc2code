from sys import stdin

def read_int():
	return int(stdin.readline().split()[0])
def read_ints():
	return [int(x) for x in stdin.readline().split()]

def solve(v, trees):
	days = max(day for day, fruit in trees)
	days += 2
	fruits_by_day = [0] * days
	for day, fruit in trees:
		fruits_by_day[day] += fruit

	ripe, ripe2 = 0, 0
	collected = 0
	for day in range(days):
		ripe2 = ripe
		ripe = fruits_by_day[day]

		collect2 = min(v, ripe2)
		collect = min(v - collect2, ripe)
		ripe -= collect
		collected += collect + collect2

	return collected

n, v = read_ints()
trees = []
for _ in range(n):
	trees.append(read_ints())

print solve(v, trees)
