day1 = [0] * 4000
day2 = [0] * 4001

n,v = map(int,raw_input().split())
for _ in xrange(n):
    a,b = map(int,raw_input().split())
    day1[a] += b
    day2[a+1] += b
tot = 0
for i in xrange(3019):
    v1 = v 
    if day2[i] > 0:
        num = min(v1, day2[i])
        v1 -= num 
    if day1[i] > 0:
        num = min(v1, day1[i])
        day2[i+1] -= num
        v1 -= num 
    tot += v - v1
print tot
