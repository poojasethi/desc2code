line = raw_input()

ans = ""
i = 0
while i < len(line):
    if line[i] == "-":
        i+=1
        if line[i] == "-":
            ans += "2"
            i+=1
        elif line[i] == ".":
            ans += "1"
            i+=1
    elif line[i] == ".":
        ans += "0"
        i+=1
print ans
