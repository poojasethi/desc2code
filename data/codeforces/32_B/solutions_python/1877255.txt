import sys

file = sys.stdin
#file = open("test", "r")
line = file.readline().rstrip()
d = {".":"0", "-.":"1", "--":"2"}
offset = 0
out = ""
while offset < len(line):
    if line[offset] in d:
        out += "0"
        offset += 1
    else:
        out += d[line[offset:offset + 2]]
        offset += 2

print out
