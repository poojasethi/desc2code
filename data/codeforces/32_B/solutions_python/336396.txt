import sys

line = raw_input()
size = len(line)
zero = "."
one = "-."
two = "--"

ans = ""
idx = 0

while 1:
    s = line[idx:idx+2]
    if s == one:
        ans += "1"
        idx += 2
    elif s == two:
        ans += "2"
        idx += 2
    else:
        ans += "0"
        idx += 1

    if idx >= size:
        break
print ans
