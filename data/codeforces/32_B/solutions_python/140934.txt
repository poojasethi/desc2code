
s = raw_input()
l = len(s)

def check(num, char):
    return (num > 0 and num < l) and s[num] == char

result = ''
last = -1
for i in xrange(len(s)):
    if last >= i:
        continue
    if s[i] == '-':
        if check(i + 1, '.'):
            result += '1'
        else:
            result += '2'
        last = i + 1
    else:
        result += '0'
        last = i

print result
