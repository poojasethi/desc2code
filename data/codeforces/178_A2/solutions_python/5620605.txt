#include<cstdio>
#include<iostream>
using namespace std;
int main(){
    int N;
    scanf("%d",&N);
    int a[100000];
    int i,j,c=0;
    for(i=0;i<N;i++)
        cin>>a[i];
    for(i=0;i+1<N;i++){
        c+=a[i];
        printf("%d\n",c);
        for(j=1;i+j<N;j*=2){};
        j/=2;
        a[i+j]+=a[i];
    }
    return 0;
}
