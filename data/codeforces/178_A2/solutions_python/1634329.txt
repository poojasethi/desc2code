#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int a[100005];
int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=1;i<=n;i++) scanf("%d", &a[i]);
	long long ans=0;
	for(int i=1;i<n;i++) {
		int j=0;
		for(j=0;(1<<j)+i<=n;j++);
		--j;
		ans += a[i];
		a[(1<<j)+i]+=a[i];
		printf("%I64d\n", ans);
	}
	return 0;
}
