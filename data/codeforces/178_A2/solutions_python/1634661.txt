#include <cstdio>

using namespace std;

long long a[100100],b[100100],n,ans;

int main()
{
	scanf("%I64d",&n);
	for (int i=1;i<=n;++i) scanf("%I64d",&a[i]);
	long long c=1LL;
	while (c<=n) c=c<<1LL;
	for (int i=1;i<=n-1;++i)
	{
		long long cc=c;
		while (i+cc>n) cc=cc>>1LL;
		b[i]=cc;
	}
	for (int i=1;i<=n-1;++i)
	{
		ans+=a[i];
		a[i+b[i]]+=a[i];
		printf("%I64d\n",ans);
	}
	return 0;
}
