#include <cstdio>
#include <cstring>
const int maxn = 1000003;
typedef __int64 lld;

lld a[maxn];

int main() {
	int n, i, j;
	scanf("%d",&n);
	for (i=1; i <= n; i++) {
		scanf("%I64d",&a[i]);
	}
	lld tag = 0;
	for (i=1; i < n; i++) {
		if (a[i] > 0) {
			for (j=1; i+j <= n; j<<=1);	
			j>>=1;
			a[i+j] += a[i];
			tag += a[i];
		}
		printf("%I64d\n",tag);
	}
	return 0;
}
