#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

int n;
vector<int> seq;

int main(){
	scanf("%d",&n);
	seq.resize(n);
	for (int i=0; i<n; i++)
		scanf("%d",&seq[i]);
	int nturns=0;
	for (int i=0; i<n-1; i++){
		nturns+=seq[i];
		int pow2=2;
		while (i+pow2<n)
			pow2<<=1;
		seq[i+(pow2>>1)]+=seq[i];
		seq[i]=0;
		printf("%d\n",nturns);
	}
	return 0;
}
