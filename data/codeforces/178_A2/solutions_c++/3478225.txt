#include <cstdio>
#include <cstring>

int n, v[1000000];

int main() {

    scanf("%d",&n); for (int i=0;i<n;i++) scanf("%d",v+i);
    int tot =0;
    for (int i=0;i<n-1;i++) {
        if (v[i]==0) { printf("%d\n",tot); continue; }
        int t;
        for (t=0;i+(1<<t)<n;t++); t--;
        tot += v[i];
        v[ i+(1<<t) ] += v[i];
        v[i] = 0;
        printf("%d\n",tot);
    }

    return 0;
}
