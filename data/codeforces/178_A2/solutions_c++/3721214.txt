#include<iostream>
#define int long long
using namespace std;

int arr[100001];

main()
{
    int n;
    cin>>n;
    for (int i=0;i<n;i++)
        cin>>arr[i];
    int s = 0;
    int t = 0;
    for (int i=0;i<n-1;i++)
    {
        s += arr[i];
        while(i + (1<<t) < n)
            t++;
        t--;
        arr[i + (1<<t)] += arr[i];
        cout<<s<<endl;
    }
    return 0;
}
