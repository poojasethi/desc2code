#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int a[100010];
int main()
{
	int n,i,j,k;
	scanf("%d",&n);
	for(int i = 1; i <= n; i ++)
		scanf("%d",&a[i]);
	int ans = 0;
	for(int i = 1; i < n; i ++) {
		int log;
		for(log = 17; log >= 0; log --) {
			if(i + (1<<log) <= n) break;
		}
		ans += a[i];
		a[i+(1<<log)] += a[i];
		printf("%d\n",ans);
	}
	return 0;
}
