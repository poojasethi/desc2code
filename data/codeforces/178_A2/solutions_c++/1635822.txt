#include <iostream>
#include <fstream>

using namespace std;

const int MAX=100005;
int N,A[MAX],S[MAX],V[MAX];

int main(){
    cin >> N;
    //S[0]=0;
    for(int i=1; i<=N; i++){
	cin >> A[i];
	//S[i]=A[i]+S[i-1];
	V[i]+=A[i];
	if(i==N) break;
	for(int k=20; ; k--){
	    if(i+(1<<k)<=N){
		V[i+(1<<k)]+=V[i];
		break;
	    }
	}
    }
    int s=0;
    for(int i=1; i<=N-1; i++){
	s+=V[i];
	cout << s << endl;
    }

}
