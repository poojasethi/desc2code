#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>

using namespace std;

int n, a[100001];
int total = 0;

int main() {
	cin >> n;
	for(int i = 0; i < n; ++ i) 
		cin >> a[i];
	for(int k = 0; k < (n-1); ++ k) {
		int j;
		for(j = 1; k+j+j < n; j += j);
		a[k+j] += a[k];
		total += a[k];
		cout << total << '\n';	
	}
	return 0;
}
