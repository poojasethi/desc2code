#include <iostream>

using namespace std;

int main()
{
    long n, i = 0, mas[100000], t, ans = 0;
    cin >> n;
    for(i = 0; i < n; ++i)
        cin >> mas[i];
    
    for(i = 0; i < n-1; ++i)
    {
        t = 1;
        while(i+t*2 < n)
            t *= 2;
        ans += mas[i];
        cout << ans << endl;
        mas[i+t] += mas[i];
    }
    
    return 0;
}
