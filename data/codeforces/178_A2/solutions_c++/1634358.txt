#include <cstdio>
using namespace std;

int main()
{
	int st[21];
	st[0]=1;
	for (int i=1; i<21; i++)
		st[i]=st[i-1]*2;

	int n;
	int a[100001];
	scanf("%d",&n);
	for (int i=0; i<n; i++)
		scanf("%d",&a[i]);

	int res=0;
	for (int i=0; i<n-1; i++) {
		int k=20;
		while (i+st[k]>=n) k--;
		res+=a[i];
		a[i+st[k]]+=a[i];
		printf("%d\n",res);
	}

	return 0;
}
