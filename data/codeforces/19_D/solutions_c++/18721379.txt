#include <bits/stdc++.h>
using namespace std;

set<pair<int,int> > sets[200005];int x[200005],y[200005],z[200005],p[200005],n;
pair<int,int> c[200005],pr[200005];char s[10];

int main()
{
    char s[10];
    scanf("%d",&n);
    int cnt=0;
    int maxi=0;

    for(int i=0;i<n;i++)
    {
        scanf("%s ",s);
        scanf("%d %d",&x[i],&y[i]);
        if(s[0]=='a') p[i]=1;
        else if(s[0]=='r') p[i]=2;
        else p[i]=3;

        if(p[i]==3) x[i]++,y[i]++;
        pr[i]=make_pair(-y[i],i);
    }

    sort(pr,pr+n);
    for(int i=0,r=0;i<n;z[pr[i].second]=cnt,i++){
        if(i==0 || pr[i].first!=pr[i-1].first) cnt++;
    }

    for(int i=0;i<n;i++)
        if(p[i]==3){
                pair<int,int> X=make_pair(x[i],y[i]);
                pair<int,int> itx=make_pair(INT_MAX,INT_MAX);
                set<pair<int,int> >::iterator it;
                for(int zz=z[i];zz>0;zz=(zz&(zz+1))-1)
                    if(sets[zz].size()>0){
                        it=sets[zz].lower_bound(X);
                        if(it!=sets[zz].end()  && *it<itx)  itx=*it;
                }
                if(itx.first==INT_MAX)  printf("-1\n");
                else printf("%d %d\n",itx.first,itx.second);

        }
        else if(p[i]==1){
            pair<int,int> X=make_pair(x[i],y[i]);
            for(int zz=z[i];zz<200005;zz|=zz+1)  sets[zz].insert(X);
        }
        else if(p[i]==2){
            pair<int,int> X=make_pair(x[i],y[i]);
            for(int zz=z[i];zz<200005;zz|=zz+1)  sets[zz].erase(X);
        }
    return 0;
}
