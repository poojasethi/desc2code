#include <iostream>
#include <cstdio>
#include <map>
#include <set>
using namespace std;
#define REP(i,n) for(i=0;i<n;i++)

const int MAXN=200000+10;
typedef pair< int , int > PII;
int n,m;
int x[MAXN],y[MAXN],f[MAXN];
set< PII > t[MAXN];
map< int , int > num;

void add(int x,int y){for(int i=num[y];i;i-=i&(-i))t[i].insert(make_pair(x,y));}
void del(int x,int y){for(int i=num[y];i;i-=i&(-i))t[i].erase(make_pair(x,y));}
void query(int x,int y){
        PII ret=make_pair(-1,-1),A=make_pair(x,y);
        for(int i=num[y];i<=m;i+=i&(-i))if(!t[i].empty()){
                typeof(t[i].begin()) itr=t[i].lower_bound(A);
                if (itr!=t[i].end()&&(ret.first==-1||*itr<ret)) ret=*itr;
        }
        if (ret.first==-1) printf("-1\n");else printf("%d %d\n",ret.first,ret.second);
}

int main(){
        scanf("%d",&n);
        int i;
        char s[10];
        REP(i,n){
                scanf("%s%d%d",s,&x[i],&y[i]);
                if (s[0]=='a') f[i]=1;else if (s[0]=='r') f[i]=2;else f[i]=3;
                if (f[i]==3) x[i]++,y[i]++;
                num[y[i]]=0;
        }
        i=0;
        for(typeof(num.begin()) itr=num.begin();itr!=num.end();itr++) itr->second=++i;
        m=i;
        REP(i,n){
                if (f[i]==1)add(x[i],y[i]);
                else if (f[i]==2) del(x[i],y[i]);
                else query(x[i],y[i]);
        }
}
