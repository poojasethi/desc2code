#include <cstdio>
#include <cstring>
char a[300500],h[300500],n[300500];
int main(){
	int n,k,i,y=0;	
	
	scanf(" %d %d\n",&n,&k);	
	
	for(i=1;(i<=n)and(y<k);i++){
		scanf("%c",&a[i]);
		if(a[i]=='.'){
			y=0;
		}
		else{
			y=y+1;
		}		
	}
	if(y==k){
		printf("NO");
	}
	else{
		printf("YES");
	}
	
	return 0;
}
