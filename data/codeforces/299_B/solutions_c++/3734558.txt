#include <cstdio>
int main()
	{
	int n,k,i,temp=0,max=-1;
	char arr[300000];
	scanf("%d%d",&n,&k);
	scanf("%s",arr);
	for(i=0;i<n;i++)
	{
	if(arr[i]=='#')
	temp++;
	if(arr[i]=='.' || i==n-1)
		{
		if(temp>max)
		max=temp;
		temp=0;
		}
	}
	if(max<k)
	printf("YES");
	else
	printf("NO");
	
	}
