#include<stdio.h>
int main()
{
int n,k,p=0,m;
scanf(" %d %d",&n,&k);
m=n+1;
int i=1,j=1;
while(m--)
{
char a;
scanf("%c",&a);
if(a=='.')
{   
	i=j;
	j=n-m;
	//printf("%d&%d  ",j,i);
	if(j-i>k)
   {
   p=1;
   printf("NO");
   break;
   }
}
}
if(p==0)
printf("YES");
return 0;
}
