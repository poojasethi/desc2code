#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{int n ,k,cnt=0; cin>>n>>k ; string x; cin>>x;
for(int i = 0 ;i <x.size();i++)
    {if(x[i]=='#')cnt++;
if(cnt>=k){cout<<"NO" ;return 0 ;}
if(x[i]=='.'){cnt=0;}
    }if(cnt>k){cout<<"NO"; return 0 ;}cout<<"YES";}
