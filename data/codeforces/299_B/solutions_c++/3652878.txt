#include<stdio.h>
#include<string.h>
char shi[400000];
int main ()
{
	int i,j,n,m,p;
	scanf("%d%d%*c",&n,&m);
	scanf("%s",shi);
	p=0;
	for(i=0;i<n;i++)
	{
		if(shi[i]=='.')p=0;
		if(shi[i]=='#')p++;
		if(p>=m)break;
	}
	if(i==n) printf("YES\n");
	else
	printf("NO\n");
}
