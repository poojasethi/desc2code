# encoding: utf-8

if __name__ == '__main__':
    n, k = [int(x) for x in raw_input().rstrip().split()]
    roads = raw_input().rstrip()
    run = 0
    max_run = 0

    for sector in roads:
        if sector == '.':
            max_run = max(max_run, run)
            run = 0
        else:
            run += 1
    max_run = max(max_run, run)

    if max_run < k:
        print 'YES'
    else:
        print 'NO'
