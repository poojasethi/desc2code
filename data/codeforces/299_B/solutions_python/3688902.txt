n,k=map(int,raw_input().split())
s=raw_input()
c=[0]*(n+1)
for i in xrange(1,n+1):
    if s[i-1]=='#':
        c[i]=c[i-1]+1
start=1
valid=False
while start<n:
    end=start+k
    if end>=n:
        valid=True
        break
    if s[end-1]=='#':
        end-=c[end]
    if end<=start:
        valid=False
        break
    start=end
if valid:
    print('YES')
else:
    print('NO')
