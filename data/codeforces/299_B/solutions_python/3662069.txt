n, k = map(int, raw_input().split())
s = raw_input()
flg = True
i = 0
while i < n:
    while i < n and s[i] == '.': i += 1
    if i == n: break;
    i -= 1
    f = False
    for j in xrange(1, k + 1):
        if i + j < n and s[i + j] == '.':
            i = i + j
            f = True
            break
    if not f:
        flg = False
        break
if flg:
    print "YES"
else : print "NO"
