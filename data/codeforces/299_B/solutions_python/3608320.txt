n, k = raw_input().split()
n = int(n)
k = int(k)
road = raw_input()

position = 0
fail = False
while True:
    step = k
    if position + step >= n:
        step = n - position - 1
    while road[position + step] == '#':
        step -= 1
        if step == 0:
            fail = True
            break

    if fail:
        break

    position = position + step
    if position == n - 1:
        break

if fail:
    print 'NO'
else:
    print 'YES'
