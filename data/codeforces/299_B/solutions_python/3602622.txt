n, k = raw_input().split()
n = int(n)
k = int(k)

sectors = raw_input()

def reach(s, k):
    if len(s) == 0:
        return True
    for i in range(k):
        if s[i] == '#':
            continue
        elif reach(s[i+1:], k):
            return True
        else:
            continue
    return False

if sectors.find("#"*k) != -1:
    print "NO"
else:
    print "YES"
