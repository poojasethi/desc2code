n, k = map(int, raw_input().split())
if n <= k + 1:
    print "YES"
else:
    road = raw_input()
    ava = True
    count = 0
    for idx, v in enumerate(road):
        if road[idx] == '#':
            count += 1
            if count >= k:
                ava = False
                break
        else:
            count = 0
    if ava:
        print "YES"
    else:
         print "NO"
