IN = lambda : map(int, raw_input().split() )
if __name__ == '__main__':
    n, m = IN()
    ss = raw_input()
    last = n-1
    if ss[-1] == '#':
        print 'NO'; 
    else :
        for i in xrange(n-1, -1, -1):
            if last-i>m:
                print 'NO'; break
            if ss[i]=='.': last = i
        else : print 'YES'
