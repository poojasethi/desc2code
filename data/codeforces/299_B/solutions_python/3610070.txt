num = raw_input().strip().split()

k = int(num[1])

a = raw_input().strip()

cnt = [0 for i in a]

kol, i = 0, 0
while i < len(cnt):
	if a[i] == '#':
		kol += 1
	else:
		kol = 0
	cnt[i] = kol
	i += 1

if max(cnt) >= k:
	print 'NO'
else:
	print 'YES'
