import sys
n,k=map(int,raw_input().split())
lis=raw_input().strip()
i=0
while i<n-1:
    nex=lis[i+1]
    x=lis[i]
    if x=="." and nex=="#":
        ind=lis.find(".",i+1)
        if ind!=-1:
            if ind-i>k:
                print "NO"
                sys.exit(0)
            else:
                i=ind
        else:
            print "NO"
            sys.exit(0)
    elif x=="." and nex==".":
        i+=1

print "YES"
