#include <iostream>

using namespace std;
int x[100][100];
int n , m ;

void check(int n){
if(n%2!=0){cout<<n<<endl;return;}

for(int i=0;i<n/2;i++)
    for(int j=0;j<m;j++)
    if(x[i][j] != x[n-1-i][j])
{  cout<<n<<endl;  return ;   }

check(n/2);
}

int main()
{
    cin>>n>>m;
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
        cin>>x[i][j];
check(n);
    return 0;
}
