#include<bits/stdc++.h>

using namespace std;

int a[101][101],n,m;

int recurse(int r)
{
	int res=0;
	if(r%2!=0)
		return r;
	for(int i=r/2-1;i>=0;i--)
	{
		for(int j=0;j<m;j++)
			if(a[i][j]!=a[r-i-1][j])
				return r;
	}
	
	return recurse(r/2);
}

int main()
{
	//int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			cin>>a[i][j];

	
	int ans = recurse(n);
	cout<<ans<<endl;
	return 0;
}
