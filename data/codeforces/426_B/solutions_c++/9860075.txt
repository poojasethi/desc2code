#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,i,j,k,a[200][200];
	cin>>n>>m;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			cin>>a[i][j];
	while(1)
	{
		if(n%2!=0)
		{
			cout<<n<<endl;
			return 0;
		}
		else
		{
			for(i=0,j=n-1;i<j;i++,j--)
			{
				for(k=0;k<m;k++)
				{
					if(a[i][k]!=a[j][k])
					{
						cout<<n<<endl;
						return 0;
					}
				}
			}
			n=n/2;
		}
	}
	return 0;
}
