#include<iostream>

using namespace std;

int n, m, a[111][111];

int rec(int n)
{
	if(n % 2 == 1)
		return n;
	for(int i = 0; i < n / 2; i++)
		for(int j = 0; j < m; j++)
			if(a[i][j] != a[n - i - 1][j])
				return n;
	return rec(n / 2);
}

int main()
{
	cin >> n >> m;
	for(int i = 0; i < n; i++)
		for(int j = 0; j < m; j++)
			cin >> a[i][j];
	cout << rec(n) << endl;
	return 0;
}
