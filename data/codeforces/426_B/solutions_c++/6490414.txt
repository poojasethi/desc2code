#include <bits/stdc++.h>
using namespace std;

int main() {
  int n,m;
  cin>>n>>m;
  vector <string> X(n);
  for(int i = 0; i < n; i++) {
    for(int j = 0; j < m; j++) {
      string tmp;
      cin>>tmp;
      X[i]+=tmp;
    }
  }

  while(n%2 == 0) {
    n = n/2;
    for(int i = 0; i < n; i++) {
      if(X[i] != X[(n*2)-i-1]) {
	cout<<n*2<<endl;
	return 0;
      }
    }
  }
  cout<<n<<endl;
  return 0;
}
