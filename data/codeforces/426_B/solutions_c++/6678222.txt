#include<bits/stdc++.h>
using namespace std;

int n,jawab,a[110][110],m;
void solved(int x){
	if(x%2) return;
	for(int i=0,j=x-1;i<x/2;i++,j--)
		for(int k=0;k<m;k++){
			if(a[i][k]!=a[j][k]) return;
		}
	jawab/=2;
	solved(x/2);
}

int main(){
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++) scanf("%d",&a[i][j]);
	jawab=n;
	solved(n);
	printf("%d\n",jawab);
}
