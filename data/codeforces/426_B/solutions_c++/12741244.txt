#include<bits/stdc++.h>
using namespace std;
int main (){
    int n, m;
    cin >> n >> m;
    string a[n];
    cin.ignore();
    for (int i=0 ; i<n ; i++)
        getline (cin , a[i]);
    while (n%2==0){
        for (int i=0 ; i<n/2 ; i++)
            if (a[n-1-i]!=a[i]){
                cout << n;
                return 0;
            }
        n/=2;
    }
    cout << n;
    return 0;
}
