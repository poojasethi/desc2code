#include <iostream>
#include <algorithm>
using namespace std;

int main(){
	int n,m;
	int a[100][100];
	cin >>n>>m;
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			cin >>a[i][j];
		}
	}
	
	bool flag=true;
	
	while(n%2==0){
		for(int i=0;i<(n/2);i++){
			for(int j=0;j<m;j++){
				if(a[i][j]!=a[n-1-i][j]){
					flag=false;
					break;
				}
			}
			
		}
		if(flag) n/=2;
		else break;
	}
	cout << n<<endl;
}
