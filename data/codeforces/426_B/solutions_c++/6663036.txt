#include<iostream>
using namespace std;
int main()
{
    int x,y;
    cin>>x>>y;
    int a[x][y];
    for(int i=0;i<x;i++)for(int j=0;j<y;j++)cin>>a[i][j];
    while(x%2==0)
    {
        bool w=true;
        for(int i=0;i<x/2;i++)for(int j=0;j<y;j++)if(a[i][j]!=a[x-i-1][j])
            w=false;
        if(w)x/=2;
        else break;
    }
    cout<<x<<endl;
    return 0;
}
