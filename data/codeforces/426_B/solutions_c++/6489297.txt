#include <cstdio>

int main(){
	int n, m, a[105][105], ans=0;
	scanf("%d %d", &n, &m);
	for(int i=0; i<n; i++){
		for(int j=0; j<m; j++){
			scanf("%d", &a[i][j]);
		}
	}
	for( ans = n; ans % 2 == 0; ans/=2){
		for(int i=0; i<ans/2; i++){
			for(int j=0; j<m; j++){
				if( a[i][j] != a[ans-i-1][j] ){
					printf("%d\n", ans);
					return 0;
				}
			}
		}
	}
	printf("%d\n", ans);
	return 0;
}
