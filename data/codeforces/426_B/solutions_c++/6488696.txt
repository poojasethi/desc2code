#include<cstdio>
#include<cstdlib>
using namespace std;
int n,m;
int a[200][200];
int dfs(int x)
{
	if(x%2==1) return x;
	int i,j,mid=x/2;
	for(i=1;i<=mid;i++) for(j=1;j<=m;j++) if(a[i][j]!=a[x-i+1][j]) return x;
	return dfs(mid);
}
int main()
{
	int i,j;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++) for(j=1;j<=m;j++) scanf("%d",&a[i][j]);
	printf("%d",dfs(n));
	return 0;
}
