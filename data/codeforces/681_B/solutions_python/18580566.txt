# coding: utf-8

s = int(raw_input())
ans = False
for i in xrange(1000):
    if 1234567 * i > s: break
    for j in xrange(10000):
        v = 1234567 * i + 123456 * j
        if v > s: break
        if (s - v) % 1234 == 0:
            ans = True
            break
    if ans: break
print "YES" if ans else "NO"
