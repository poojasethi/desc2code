
import sys
#sys.stdin=open("data.txt")
input=sys.stdin.readline

n=int(input())

if n%2:
    n=(n-1234567)//2
else:
    n=n//2

if n<0: print("NO")
else:
    # now check with 1234567, 61728, 617
    good=0
    for i in range(n,-1,-1234567):
        for j in range(i,-1,-61728):
            if j%617==0:
                good=1
                break
        if good: break
    if good: print("YES")
    else: print("NO")
