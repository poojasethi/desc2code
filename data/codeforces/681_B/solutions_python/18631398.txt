n = int(input())

can = False

##print(n / 1234567)
for i in range(n / 1234567 + 1):
	if (can):
		break
	n1 = n - i * 1234567
	##print("#", n1 / 123456)
	for i1 in range(n1 / 123456 + 1):
		if (can):
			break
		n2 = n1 - i1 * 123456
		##print(n2 % 1234)
		if (n2 % 1234 == 0):
				can = True

if (can):
	print("YES")
else:
	print("NO")
