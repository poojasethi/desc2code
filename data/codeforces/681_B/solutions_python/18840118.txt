n = int(input())
l = [1234567,123456,1234]
t = 0
c = 0
k = 1
while l[0]*t <= n and k:
    while l[1] * c <= n and k:
        m = l[0]*t + l[1]*c
        if m == n:
            print "YES"
            k = 0
            break
        if (n - m)%l[2] == 0 and n>m:
            print "YES"
            k = 0
            break
        c += 1
    c = 0
    t += 1
if k:
    print "NO"
