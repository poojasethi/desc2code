#!/usr/bin/env python

from __future__ import print_function
import sys

DEBUG = '-d' in sys.argv

def debug(*args, **kwargs):
    if DEBUG:
        print(*args, file=sys.stderr, **kwargs)

    return None

N = int(raw_input())

yes = False

for j in xrange(N / 1234567 + 1):
    M = N - j * 1234567
    for i in xrange(M / 123456 + 1):
        if (M - i * 123456) % 1234 == 0:
            yes = True

print("YES" if yes else "NO")
