n = int(raw_input())

A = 1234567
B = 123456
C = 1234
flag = False

for i in range(0,1001):
  if(A*i>n):
    break
  for j in range(0,10001):
    sum = A*i + B*j
    rem = n - sum
    if(rem<0):
      break
    if(rem%C == 0):
      flag = True
      break
  if(flag):
    break

if flag == True:
  print 'YES'
else:
  print 'NO'
