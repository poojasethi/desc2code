#!/usr/bin/env python
# -*- coding: utf-8 -*-

import sys

n = int(raw_input())
coins = (1234567, 123456, 1234)

for i in xrange(0, n / coins[0] + 1):
    total_i = i * coins[0]
    for j in xrange(0, (n - total_i) / coins[1] + 1):
        total_j = n - total_i - coins[1] * j
        if total_j % coins[2] == 0:
            print "YES"
            sys.exit(0)
print "NO"
