a = []
x = 1234567
y = 0
while y <= 10**9:
	a.append(y)
	y += x

b = []
x = 123456
y = 0
while y <= 10**9:
	b.append(y)
	y += x

n = int(raw_input())
ok = False
for i in xrange(len(a)):
	for j in xrange(len(b)):
		u = a[i]
		if u > n:
			break
		v = b[j]
		if v > n:
			break

		rem = n - u - v

		if rem >= 0:
			if rem % 1234 == 0:
				ok = True
				break

	if ok:
		break

if ok:
	print 'YES'
else:
	print 'NO'
