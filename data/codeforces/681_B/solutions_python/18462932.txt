from fractions import gcd
n = input()
x = 1234567
y = 123456
z = 1234
yz = gcd(y,z)
ans ="NO"
ini = 0
while((n-ini)>=0 and ans=="NO"):
    rem = n -ini
    if(rem%yz!=0):
        ini+=x
    else:
        temp = 0
        while((rem-temp)>=0):
            if((rem-temp)%z==0):
                ans = "YES"
                break
            temp+=y
        ini+=x
print ans
