n = int(raw_input())
f = 0
if n % 1234567 == 0:
    f = 1
else:    
    x = n / 1234567
    for i in range(x + 1):
        n1 = n - i * 1234567
        if n1 % 123456 == 0:
            f = 1
            break
        y = n1 / 123456
        for j in range(y + 1):
            n2 = n1 - j * 123456
            if n2 % 1234 == 0:
                f = 1
                break
        if f == 1:
            break
if f == 1:
    print "YES"
else:
    print "NO"
