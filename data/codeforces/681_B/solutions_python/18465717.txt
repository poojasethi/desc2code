import sys
from math import ceil as C

x , y , z = 1234567.0 , 123456.0 , 1234.0
n = input()
flag = False
for i in range(int(C(n/x))+1) :
    for j in range(int(C(n/y))+1) :
        c = n-i*x-j*y
        if  c%z==0 and c>=0 :
            flag = True
            break
    if flag :
        break
print "YES"if flag else "NO"
