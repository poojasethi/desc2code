n=input()
f=1
t=0
while f==1 :
    a=n-t*1234567
    b=a
    c=a
    fl=0
    while a>=1234 or b>=123456 :
        if a%123456==0 or b%1234==0:
            fl=1
            break
        if a-1234>0 :
            a=a-1234
        if b-123456>0 :
            b=b-123456
    if a==0 :
        fl=1
    if fl==1 :
        f=0
        print "YES"
    t=t+1
    if c<0 :
        f=0
        print "NO"
