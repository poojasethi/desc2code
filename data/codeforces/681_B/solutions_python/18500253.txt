#!/usr/bin/env python

def main():
    h_pri, c_pri, co_pri = 1234567, 123456, 1234
    n = int(raw_input())
    ma = n / h_pri
    for i in range(ma + 1):
        mb = (n - i * h_pri) / c_pri
        for j in range(mb + 1):
            if (n - i * h_pri - j * c_pri) % co_pri == 0:
                print 'YES'
                return
    print 'NO'

if __name__ == '__main__':
    main()
