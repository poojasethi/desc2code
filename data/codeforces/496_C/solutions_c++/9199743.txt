#include <iostream>
#include <algorithm>

using namespace std;

string a[111], b[111];
int n, m, ans = 0;

int main()
{
	cin >> n >> m;
	for (int i = 0; i < n; i++)
	{
		cin >> a[i];
		b[i] = a[i];
	}
	for (int j = 0; j < m; j++)
	{
		for (int i = 0; i < n; i++)
		{
			sort(b, b + n);
			if (a[i][j] != b[i][j])
			{
				ans++;
				for (int k = 0; k < n; k++)
				{
					a[k][j] = '!';
					b[k][j] = '!';
				}
				break;
			}
		}
	}
	cout << ans;
}
