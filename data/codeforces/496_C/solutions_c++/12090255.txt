#include<cstdio>
int s[105]={0},n,m;
char word[105][105]={0};
int main()
{
	int i,l=0,j,flag;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)scanf("%s",word[i]);
	for(j=0;j<m;j++)
	{
		for(i=0;i<n-1;i++)
		{
			flag=0;
			if(word[i][j]>word[i+1][j]&&!s[i]){flag=1;l++;break;}
		}
		if(!flag)for(i=0;i<n-1;i++)if(word[i][j]<word[i+1][j]){s[i]=1;}
	}
	printf("%d\n",l);
	return 0;
}
