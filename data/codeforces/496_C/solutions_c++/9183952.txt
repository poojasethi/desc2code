#include<iostream>
#include<algorithm>
#include<numeric>
using namespace std;
string s[100];
int a[100], b[100];
int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		cin >> s[i];
	for (int j = 0; j < m; j++) {
		for (int i = 0; i < n-1; i++)
			if (!a[i] && s[i][j] > s[i+1][j])
				b[j] = 1;
		if (b[j]) continue;
		for (int i = 0; i < n-1; i++)
			if (s[i][j] < s[i+1][j])
				a[i] = 1;
	}
	cout << accumulate(b, b+m, 0);
}
