#include<bits/stdc++.h>

using namespace std;

int r,c;
string ar[105];

int main(){
	int cnt=0;
	scanf("%d%d",&r,&c);
	for(int x=0;x<r;x++) cin>>ar[x];
	int pos=0;
	while(pos<r-1){
		if(ar[pos]>ar[pos+1]){
			for(int x=0;x<c;x++) if(ar[pos][x]>ar[pos+1][x]){
				//cout<<x<<endl;
				for(int y=0;y<r;y++) ar[y].erase(ar[y].begin()+x);
				//for(int y=0;y<r;y++) cout<<ar[y]<<endl;
				cnt++;
				pos=0;
				break;
			}
		 } else pos++;
	}
	printf("%d\n",cnt);
}
