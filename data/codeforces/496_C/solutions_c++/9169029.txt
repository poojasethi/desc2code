#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	vector<string> brd(n);
	for(int i=0;i<n;++i)cin>>brd[i];
	int ans=0;
	vector<bool> ok(n,0);
	for(int i=0;i<m;++i)
	{
		bool ta=0;
		for(int j=1;j<n;++j)
		{
			if(!ok[j] && brd[j][i] < brd[j-1][i])ta=1;
		}
		if(ta)ans++;
		else
		{
			for(int j=1;j<n;++j)
			{
				if(brd[j][i]!=brd[j-1][i])ok[j]=1;
			}
		}

	}
	cout<<ans;
	return 0;
}
