#include<stdio.h>
int n,m,r[101],ans,t,i,j;
char a[101][101];
int main(){
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
		scanf("%s",a[i]);
	for(i=1;i<n;i++)
		r[i]=1;
	for(i=0;i<m;i++){
		t=0;
		for(j=1;j<n;j++)
			if(r[j]==1)
				if(a[j][i]<a[j-1][i]){ans++;t=1;break;}
		if(t==1)continue;
		for(j=1;j<n;j++)
			if(a[j][i]!=a[j-1][i])r[j]=0;
	}
	printf("%d",ans);
}
