#include <iostream>
using namespace std;
int main() {
	int n, m, i, j, ans = 0; 
	cin >> n >> m; char c[n][m];
	for(i = 0; i < n; ++i) for(j = 0; j < m; ++j) cin >> c[i][j];
	for(i = 0; i < n - 1; ++i) {
		for(j = 0; j < m && ans != m; ++j) {
			if(c[i][j] < c[i + 1][j]) break;
			else if(c[i][j] > c[i + 1][j]) { 
				++ans;  for(int k = 0; k < n; ++k) c[k][j] = 1; i = -1; break;
			}
		}
	}
	cout << ans;
}
