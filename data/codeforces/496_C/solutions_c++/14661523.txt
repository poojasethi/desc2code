# include <stdio.h>
# include <string.h>

int n,m,sm;
char a[101][101];
int v[101];

int main()
{
	scanf("%d %d",&n,&m);
	
	for(int i=0;i<n;i++)
	scanf("%s",a[i]);
	
	for(int i=0;i<m;i++)
	{
		int t=1;
		for(int j=1;j<n;j++)
		{
			if(a[j][i]<a[j-1][i] && v[j]==0){
			sm++; t=0; break;}
		}
		if(t==1)
		{
			for(int j=1;j<n;j++)
			if(a[j][i]>a[j-1][i])
			v[j]=1;
		}
	}
	
	printf("%d",sm);
	return 0;
}
