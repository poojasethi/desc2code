#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,c,j;
	char a[101][101];
	scanf("%d %d",&n,&m);
	for(int i=0;i<n;i++)
		scanf("%s",a[i]);
	c=0;
	for(int i=1;i<n;)
	{
		if(strcmp(a[i],a[i-1])<0)
		{
			for(j=0;j<m;j++)
			{
				if(a[i][j]<a[i-1][j])
					break;
			}
			for(int k=0;k<n;k++)
				a[k][j]='b';
			c++;
			i=1;
		}
		else
			i++;
	}
	printf("%d",c);
	return 0;
}
