#include <iostream>
#include <iomanip>
#include <math.h>
#include <algorithm>
using namespace std;



long long m,n,o,a,b,c,mark[1001];
string s[1001];
int main(){
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		cin>>s[i];	
	}
	c=1;
	while(o!=c){
		c=o;
	for(int i=1;i<n;i++){
		
		for(int j=0;j<m;j++){
			if(mark[j]==1)continue;
			if(s[i][j]<s[i+1][j])break;
			if(s[i][j]>s[i+1][j])o++,mark[j]=1;
		}
	}
    }
	cout<<o;
}
