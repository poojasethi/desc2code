#include<bits/stdc++.h>
using namespace std;
int vis[101];
int main(){
	int n,m;
	cin>>n>>m;
	string arr[n];
	for(int i=0;i<n;i++){
		cin>>arr[i];
	}
	int ans=0;
	int j;
	for(int i=0;i<m;i++){
		for(j=1;j<n;j++){
			
			if(!vis[j]&& arr[j-1][i]>arr[j][i]){
				ans+=1;
				break;
			}
		}																		
		if(j==n){
			for(int k=1;k<n;k++){
				if(arr[k-1][i]<arr[k][i])
				vis[k]=1;	
			}
		}
	}
	cout<<ans;
}
