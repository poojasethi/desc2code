#include<cstdio>
using namespace std;
char s[100][101];
bool b1[100];
int main()
{
	int n,m,o=0;bool bl;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;++i)
		scanf("%s",s[i]);
	for(int i=0;i<m;++i)
	{
		bl=0;
		for(int j=1;j<n;++j)
		if(!b1[j]&&s[j][i]<s[j-1][i])
			bl=1;
		if(bl)++o;
		else for(int j=1;j<n;++j)
		if(s[j][i]>s[j-1][i])
			b1[j]=1;
	}
	printf("%d\n",o);
}
