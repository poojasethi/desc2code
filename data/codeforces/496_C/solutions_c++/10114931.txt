#include<iostream>
using namespace std;

int n,m,ans;
char a[110][110];
bool del,l[100];

int main()
{
	ios_base::sync_with_stdio(0);
	
	cin>>n>>m;
	for(int i=0;i<n;i++) cin>>a[i];
	
	for(int i=0;i<m;i++)
	{
		del=false;
		for(int j=1;j<n&&!del;j++)
			if(!l[j]&&a[j][i]<a[j-1][i])
				del=true;
			
		if(del)
			ans++;
		else
			for(int j=1;j<n;j++)
				if(a[j][i]>a[j-1][i])
					l[j]=true;
	}

	cout<<ans;
	
	return 0;
}
