// In the name of god :|:|:|

#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n , m , i , j , k , ans = 0 ;
	cin >> n >> m ; 
	string s[n] ;
	for ( i = 0 ; i < n ; i++ )
		cin >> s[i] ;
	for ( i = 0 ; i < n-1 ; i++ )
	{
		if ( s[i] > s[i+1] )
		{
			for ( j = 0 ; j < m ; j++ ) 
				if ( s[i][j] > s[i+1][j] ) 
					break;
			for ( k = 0 ; k < n ; k ++ ) 
				s[k].erase(j,1);
			ans++ ;
			m-- ;
			i=-1 ;
		}
	}
	cout << ans << endl ;
	return 0;
}
