#include<iostream>
#include<string>
using namespace std;
int a[105][105],safe[105];
int n,m,ans=0;
int main()
{
	int i,j;
	char c;
	cin>>n>>m;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			cin>>c;
			a[i][j]=int(c-'a');
		}
		safe[i]=i;
	}
	for(i=0;i<m;i++)
	{
		for(j=1;j<n;j++)
		{
			if(a[j][i]<a[j-1][i]&&safe[j])
			{ans++;break;}
		}
		if(j==n)for(j=0;j<n;j++)
		{if(a[j][i]>a[j-1][i])safe[j]=0;}
	}
		cout<<ans<<endl;;
	return 0;
}
