#include<stdio.h>
char a[110][110];
int vis[110];
int main(){
	int flag = 0;
	int n,m,i,j,num = 0;
	scanf("%d %d",&n,&m);
	for(i = 0; i<n; i++)
	scanf("%s",a[i]);
	for(j = 0; j<m; j++){
		int flag = 0;
		for( i = 1; i<n; i++){
			if(!vis[i]&&a[i][j]<a[i-1][j]){
				num++;
				flag = 1;
				break;
			}
		}
		if(!flag){
			for(i = 1; i<n; i++){
				if(a[i][j]>a[i-1][j]) vis[i] = 1;
			}
		}
	}
	printf("%d\n",num);
	return 0;
}
