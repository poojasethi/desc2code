#include<bits/stdc++.h>
using namespace std;
main()
{
	int b[101], c[101];
	memset(b,0,sizeof b);
	memset(c,0,sizeof c);
	string a[101];
	int m, n; cin >> n >> m;
	for(int x=0; x<n; ++x)
	cin >> a[x];
	for(int x=0; x<m; ++x)
	{
		for(int y=0; y<n-1; ++y)
		{
		if(a[y][x] > a[y+1][x] && !c[y])
		{
			b[x]=1;
		}
	}
		if(b[x]) continue;
		for(int y=0; y<n-1; ++y)
		{
		if(a[y][x] < a[y+1][x]) c[y] = 1; }
	}
	cout << accumulate(b, b+m, 0);
}
