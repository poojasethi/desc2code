#include<bits/stdc++.h>
#define f(i,a,b) for(i=a;i<b;i++)
typedef long long ll;
using namespace std;
int main()
{
    int n,m,i,j,ans=0;
    cin>>n>>m;
    string s[n];
    f(i,0,n)cin>>s[i];
    if(n==1){cout<<0;return 0;}
    f(i,0,m){
        bool a = false;
        f(j,1,n)if(s[j][i]<s[j-1][i]&&s[j]<s[j-1]){a=true;ans++;break;}
        if(a)f(j,0,n)s[j][i]='a';
    }
    cout<<ans;
}
