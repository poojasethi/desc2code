# include <stdio.h>
# include <algorithm>

using namespace std;

int a,c,z,q;
int v[109];

char b[109][109];

int main()
{
	scanf("%d %d",&a,&c);
	
	for(int i=0;i<a;i++)
	scanf("%s",b[i]);
	
	for(int i=0;i<c;i++)
	for(int j=1;j<a;j++)
	if(b[j][i]-b[j-1][i]<0)
	{
		q=0;
		
		for(int k=0;k<i;k++)
		if(v[k]==0 && b[j][k]-b[j-1][k]>0)
		{q=1;break;}
		
		if(q==0)
		{v[i]=1;z++;break;}
	}
	
	printf("%d",z);
}
