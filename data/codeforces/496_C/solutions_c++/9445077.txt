#include<fstream>
#include<string>
#include<vector>
#include<cmath>
using namespace std;
int lowbit(int x)
{
    return x&(-x);
}






int main()
{
int m,n,k=0;
cin>>m>>n;
string s[m+1];
for(int i=1;i<=m;i++)
cin>>s[i];


for(int i=1;i<m;i++)
{if(s[i]>s[i+1])
{
    
int j=0;
for(j=0;j<n-1;j++)
if(s[i][j]>s[i+1][j])
break;
for(int f=1;f<=m;f++) 
s[f].erase(j,1);
    
k++;
n--;
i=0;


}


}

cout<<k;
}
