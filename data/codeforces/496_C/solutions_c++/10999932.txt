#include <iostream>
using namespace std;
main(){
    int n,k,j=0,beko=0;
    cin>>n>>k;
    string A[n+1];
    for (int i=1;i<=n;i++)
    cin>>A[i];
    for(int i=1;i<n;i++){
        if (A[i] > A[i+1]){
            for(j=0;j<k-1;j++)
                if (A[i][j] > A[i+1][j])
                    break;
            for(int f=1;f<=n;f++)
                A[f].erase(j,1);
           beko=beko+1;
           k=k-1;
            i=-1;}}
    cout<<beko;}
