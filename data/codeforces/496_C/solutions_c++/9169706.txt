#include<bits/stdc++.h>
using namespace std;
const int maxn=1010;
char a[maxn][maxn];
int n,m,f[maxn];
bool illegal(int x)
{
    for(int i=1;i<n;i++) if(!f[i] && a[i][x]<a[i-1][x]) return 1;
    for(int i=1;i<n;i++) if(a[i][x]>a[i-1][x]) f[i]=1;
    return 0;
}
int main()
{
    cin>>n>>m;
    for(int i=0;i<n;i++) cin>>a[i];
    int ans=0;
    memset(f,0,sizeof f);
    for(int i=0;i<m;i++) if(illegal(i)) ans++;
    cout<<ans<<endl;
    return 0;
}
