#include <bits/stdc++.h>
using namespace std;
int j,n,m,ans;
string s[120];
int main()
{
	scanf("%d %d",&n,&m);
	for(int i=0;i<n;i++)
	{
		cin>>s[i];
	}
	for(int i=0;i<n-1;i++)
	{
		if(s[i]>s[i+1])
		{
			for(j=0;j<m;j++)
			{ 
				if(s[i][j]>s[i+1][j]) 
				{
					break;
				}
			}
			for(int k=0;k<n;k++) 
			{
				s[k].erase(j,1);
			}
			ans++;
			m--;
			i=-1;
		}
	}
	printf("%d\n",ans);
	return 0;
}
