#include <bits/stdc++.h>
using namespace std;
char s[105][105];
int a[105];
int main(){
	int n,m,i,j,k;bool flag;
	cin>>n>>m;
	for(i=1;i<=n;++i)
		scanf("%s",s[i]+1);
	int ret=0;
	for(i=1;i<=m;++i){
		flag=1;
		for(j=1;j<n;++j){
			if(!a[j] && s[j][i]>s[j+1][i]){
				flag=0;
			}
		}
		if(flag){
			for(j=1;j<n;++j){
				if(s[j][i]<s[j+1][i])a[j]=1;
			}
		}
		else
			++ret;
	}
	cout<<ret<<endl;
}
