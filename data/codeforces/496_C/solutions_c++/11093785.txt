#include<cstdio>
#include<cstring>
using namespace std;
char s[105][105],ss[105][105];
int main(){
	int n,m,ans=0,f=0,k=0;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)scanf("%s",s[i]);
	for(int i=0;i<m;i++){
		ss[0][k]=s[0][i];f=0;
		for(int j=1;j<n;j++){
			ss[j][k]=s[j][i];
			if(strcmp(ss[j-1],ss[j])>0){
				f=1;ans++;break;
			}
		}if(f==0)k++;
	}printf("%d\n",ans);
	return 0;
}
