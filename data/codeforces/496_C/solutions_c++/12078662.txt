#include<cstdio>
int n,m,i,j,ans,tmp,flag[1000];
char s[1000][1000];
int main()
{
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++)
	scanf("%s",&s[i]);
	for (j=0;j<m;j++)
	{
	tmp=0;
	for (i=1;i<=n-1;i++)
	if ((s[i][j]>s[i+1][j])&&(flag[i]==0))
	tmp=1;
	ans=ans+tmp;
	if (tmp==0)
	for (i=1;i<=n-1;i++)
	if ((s[i][j]<s[i+1][j])&&(flag[i]==0))
	flag[i]=1;
	}
	printf("%d",ans);
}
