#include <iostream>
#include <string>
using namespace std;
char s[110][110];
int n,m,a[110],flag=0,ans;
int main(){
	cin>>n>>m;
	for(int i=1;i<=n;i++)cin>>s[i]+1;
	for(int i=1;i<=m;i++){
		for(int j=2;j<=n;j++){
			if(s[j][i]<s[j-1][i] && !a[j])flag=1;
		}
		if(flag)ans++;
		else {
			for(int j=2;j<=n;j++){
				if(s[j][i]>s[j-1][i])a[j]=1;
			}
		}
		flag=0;
	}
	cout<<ans;

	return 0;
}
