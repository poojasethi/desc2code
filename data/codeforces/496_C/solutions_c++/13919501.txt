#include<cstdio>
using namespace std;
int n,m,i,ans,j,a[110],flag;
char s1[110][110];
int main()
{
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)scanf("%s",s1[i]+1);
	for(i=1;i<=m;i++)
	{
		for(j=2;j<=n;j++)if(s1[j][i]<s1[j-1][i]&&!a[j])flag=1;
		if(flag)ans++;else for(j=2;j<=n;j++)if(s1[j][i]>s1[j-1][i])a[j]=1;flag=0;
	}printf("%d\n",ans);
}
