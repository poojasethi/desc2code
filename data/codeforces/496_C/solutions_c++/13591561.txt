#include<stdio.h>
#include<string.h>
int n,m;
char s[111][111],a[111][111];

int main(){
	int i,j,k;
	while(~scanf("%d %d",&n,&m)){
		for(i=0;i<n;i++){
			//for(j=0;j<m;j++){
				scanf("%s",&s[i]);
		//	}
		}
		k=0;
		for(i=0;i<m;i++){
			for(j=0;j<n;j++){
				a[j][k]=s[j][i];
			}
			for(j=0;j<n-1;j++){
				if(strcmp(a[j],a[j+1])>0)break;
			}
			if(j==n-1)k++;
		}
		
		
		printf("%d\n",m-k);
	}
	return 0;
}
