def solve(n, m, grid):
    res = ["" for i in range(n)]
    for j in range(m):
        vc = True
        for i in range(1, n):
            if res[i-1]+grid[i-1][j] > res[i]+grid[i][j]:
                vc = False
                break
        if vc:
            for i in range(n):
                res[i] += grid[i][j]
    return m-len(res[0])

n, m = map(int, raw_input().split())
grid = []
for i in range(n):
    grid.append([x for x in raw_input()])
print solve(n, m, grid)
