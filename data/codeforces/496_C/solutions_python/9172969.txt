n, m = map(int, raw_input().split())
table = []
for _ in xrange(n):
    table.append(raw_input())

count = 0
i = 0
ignoreList = []
while i < n - 1:
    for j in xrange(len(table[i])):
        if j in ignoreList:
            continue
        if table[i][j] < table[i + 1][j]:
            i += 1
            break
        if table[i][j] > table[i + 1][j]:
            count += 1
            ignoreList.append(j)
            i = 0
            break
    else:
        i += 1

print count
