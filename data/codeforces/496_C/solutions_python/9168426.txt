from copy import deepcopy
n, m = map(int, raw_input().split())
grid = []
for _ in range(n):
    grid.append(raw_input())

do_check = [True] * n

def good_col(grid, col):
    if grid == []:
        return True

    check = deepcopy(do_check)

    cur = grid[0][col]
    for i in range(1, len(grid)):
        if check[i] and grid[i][col] < cur:
            return None
        if check[i] and grid[i][col] != cur:
            check[i] = False
        cur = grid[i][col]
    return check

badcols = 0
for i in range(m):
    ch = good_col(grid, i)
    if ch:
        do_check = ch
    else:
        badcols += 1

print badcols
