import pdb
R = lambda: map(int, raw_input().split())

n, m = R()

# delete column c in Map
def d(Map, c):
    for item in Map:
        del item[c]

def find_index(a, b):
    for i in xrange(len(a)):
        if a[i] > b[i]:
            return i

Map = []
for i in xrange(n):
    s = raw_input()
    Map.append(list(s))

ans = 0
flag = True
while flag:
    flag = False
    for i in xrange(n - 1):
        if Map[i] > Map[i + 1]:
            index = find_index(Map[i], Map[i + 1])
            d(Map, index)
            flag = True
            ans += 1
            break

print ans
