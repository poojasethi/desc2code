#!/usr/bin/env python
# coding=utf-8

import sys

n,m = sys.stdin.readline().split()

n = int(n)
m = int(m)

table = []

for i in xrange(n):
    table.append(sys.stdin.readline()[:-1])

nowRow = 0
ans = 0

isbiggerLS = [False] * n

for j in xrange(m):
    needDel = False
    for i in xrange(1,n):
        if (table[i][nowRow] < table[i-1][nowRow] and not isbiggerLS[i]):
            needDel = True
    if (needDel):
        ans += 1
        for i in xrange(0,n):
            table[i] = table[i][:nowRow] + table[i][nowRow+1:]
    else:
        for i in xrange(1,n):
            if (table[i][nowRow] > table[i-1][nowRow] and not isbiggerLS[i]):
                isbiggerLS[i] = True

        nowRow += 1

print ans
