
if __name__ == '__main__':
	n,m = map(int, raw_input().split())
	a = []
	for i in range(n):
		a.append(list(raw_input()))
	count = 0
	isLess = [False]*(n-1)
	for j in range(m):
		flag = 0
		for i in range(n-1):
			if a[i][j] > a[i+1][j]:
				if isLess[i]:
					continue
				else:
					flag = 1
					break
		count+=flag
		if flag == 0:
			for i in range(n-1):
				if a[i][j] < a[i+1][j]:
					isLess[i] = True
	print count
