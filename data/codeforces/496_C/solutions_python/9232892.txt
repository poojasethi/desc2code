n, m = map(int, raw_input().split())
s = []
for i in xrange(n):
    s.append(raw_input())
flag = False
ans = 0
while not flag:
    flag = True
    for i in xrange(1, n):
	if s[i-1] > s[i]:
	    flag = False
	    break
    if not flag:
	f = False
	for i in xrange(len(s[0])):
	    for j in xrange(1, n):
		if s[j-1][:i+1] > s[j][:i+1]:
		    f = True
		    break
	    if f:
		s = map(lambda c: c[:i] + c[i+1:], s)
		ans += 1
		break
print ans
