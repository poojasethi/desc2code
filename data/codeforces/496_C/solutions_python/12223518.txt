def solve(A):
  nr, nc = len(A),len(A[0])
  possible_ties = set(range(nr-1))
  for col in xrange(nc):
    if all(A[pt][col]<=A[pt+1][col] for pt in possible_ties):
      for pt in list(possible_ties):
        if A[pt][col]<A[pt+1][col]:
          possible_ties.remove(pt)
    else:
      yield 1

n,m = map(int, raw_input().split())
A = []
for _ in xrange(n):
  A.append(raw_input())

print sum(solve(A))
