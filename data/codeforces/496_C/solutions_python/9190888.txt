n, m = map(int, raw_input().strip().split())
s = [list(raw_input()) for i in xrange(n)]
ans = 0

while True:
    i = 0
    while i < n-1 and s[i] <= s[i+1]:
        i += 1
    if i >= n-1:
        break
    j = 0
    l = len(s[i])
    while j < l and s[i][j] == s[i+1][j]:
        j += 1
    ans += 1
    for k in xrange(n):
        s[k].pop(j)

print ans
