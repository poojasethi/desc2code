n,m = (int(x) for x in raw_input().split())
table = [raw_input() for i in xrange(n)]
needComp = [True] * (n-1)
count = 0
for col in xrange(m):
    curNeedComp = needComp[:]
    for row in xrange(n-1):
        if needComp[row] and table[row][col] > table[row+1][col]:
            count += 1
            break
        elif table[row][col] < table[row+1][col]:
            curNeedComp[row] = False
    else:
        needComp = curNeedComp
print count
