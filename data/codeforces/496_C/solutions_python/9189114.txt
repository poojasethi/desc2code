n, m = map(int, raw_input().split())
t = []
tb = []
skip = set()
ans = 0

for i in xrange(0, n):
    t.append(raw_input())

rm = True
while rm:
    rm = False
    for i in xrange(0, n - 1):
        for j in xrange(0, m):
            if j in skip:
                continue
            if t[i][j] > t[i + 1][j]:
                ans += 1
                skip.add(j)
                rm = True
            elif t[i][j] < t[i + 1][j]:
                break

print ans
