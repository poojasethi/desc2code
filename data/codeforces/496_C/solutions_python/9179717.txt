n, m = map(int, raw_input().split())
words = [raw_input() for i in range(n)]

rem = 0
os = [0]*n

for j in range(m):
    ns = os[:]
    for i in range(1, n):
        p = words[i-1][j]
        c = words[i][j]
        if ord(p) < ord(c):
            ns[i] = 1
        elif ord(p) > ord(c) and os[i] == 0:
            rem += 1
            break
    else:
        os = ns
print rem
