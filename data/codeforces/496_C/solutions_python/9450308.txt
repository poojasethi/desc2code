#!/usr/bin/env python
n, m = map(int, raw_input().split())
a = []
for i in range(n):
    a.append(raw_input())
ans = 0
f = ['']*n
for i in range(m):
    t = [f[j] + a[j][i] for j in range(n)]
    if t == sorted(t):
        for j in range(n):
            f[j] += a[j][i]
    else:
        ans += 1
print ans
