n, m = map(int, raw_input().split())
s = [raw_input() for i in xrange(n)]
f = [''] * n
ans = 0
for i in xrange(m):
    t = [f[j] + s[j][i] for j in xrange(n)]
    #maak nieuwe strings met de gesorteerde vorige kolommem em volgende kolom
    #niet gesorteerde worden overgeslagen=geschrapt
    if t == sorted(t): 
        f = t
    else: 
        ans += 1
print ans
