n, m = map(int, raw_input().split())
w = [raw_input() for i in range(n)]
o = [False]*n
o[0] = True
count = 0
for i in range(m):
    lex = True
    toOrder = []
    for j in range(1, n):
        if ord(w[j][i]) < ord(w[j - 1][i]):
            if not o[j]:
                lex = False
                break
        elif ord(w[j][i]) > ord(w[j - 1][i]):
            toOrder.append(j)
    if not lex:
        count += 1
    else:
        for row in toOrder:
            o[row] = True
print count
