#include <stdio.h>
#include <algorithm>
using namespace std;
int main()
{
	long long n,a,b,aa=0x7fffffff,bb=0x7fffffff,b1,i;
	bool flag=false;
	scanf("%I64d%I64d%I64d",&n,&a,&b);
	n*=6;
	if(a*b>=n)printf("%I64d\n%I64d %I64d",a*b,a,b);
    else 
      {
	   if (a>b) {swap(a,b);flag=true;}
	   for (i=a;i*i<=n;++i)
	    {
	    b1=(n-1)/i+1;
		if (b1<b) b1=b;
		if (i*b1<aa*bb||aa*bb<n)aa=i,bb=b1;// i*b1>=n
	    }   
	   if (flag) swap(aa,bb);
	   printf("%I64d\n%I64d %I64d",aa*bb,aa,bb); 
      }
	return 0;  
}
