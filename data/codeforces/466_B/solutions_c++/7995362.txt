#include <cmath>
#include <cstdio>
#include <algorithm>

using namespace std;

bool f;
long long n,a,b,ans=0x7f7f7f7f7f7f7f7fLL,ansa,ansb;

int main(){
    scanf("%I64d%I64d%I64d",&n,&a,&b);
    n*=6;
    if (n<=a*b){printf("%I64d\n%I64d %I64d\n",a*b,a,b);return 0;}
    if (a>b) swap(a,b),f=1;
    for (int i=a;i<=sqrt(n);++i){
        long long t=max(b,n%i?n/i+1:n/i);
        if (t*i<ans) ans=t*i,ansa=i,ansb=t;
    }
    if (f) swap(ansa,ansb);
    printf("%I64d\n%I64d %I64d\n",ans,ansa,ansb);
    return 0;
}
