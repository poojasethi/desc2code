#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long k,n,a,b,r1=0,r2,d,d1,d2,sig=1;
    cin>>n>>a>>b;
    d1=min(a,b);d2=max(a,b);n=max(6*n,a*b);
    while(sig)
    {
        for(k=d1;k<=sqrt(n);k++)
        {
            if(n%k==0&&n/k>=d2)
                r1=k,r2=n/k,k=n,sig=0;
        }
        n=n+1;
    }
    if(a<=b) cout<<n-1<<endl<<r1<<" "<<r2<<endl;
    else     cout<<n-1<<endl<<r2<<" "<<r1<<endl;
	return 0;
}
