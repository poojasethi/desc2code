#include<cstdio>
typedef long long LL;
int main()
{
	LL n,a,b,x1,y1,t=0;
	scanf("%I64d%I64d%I64d",&n,&a,&b);
	n=n*6;
	LL ans=1000000000000000;
	if (a*b>=n) ans=a*b, x1=a, y1=b;
	if (a>b) x1=a,a=b,b=x1,t=1;
	for (LL i = 1, x, y; i*i<=n;i++)
	{
		x=i, y=(n+x-1)/x;
		if (b>y) y=b;
		if (a>x) x=a;
 		if (x*y<ans) ans=x*y, x1=x, y1=y;
	}
	if (t) n=x1,x1=y1,y1=n;
	printf("%I64d\n%I64d %I64d",ans,x1,y1);
}
