#include<iostream>
using namespace std;
int main()
{long long n,a,b,ax,ay;
	cin>>n>>a>>b;
	n*=6;
	long long max=a>b?a:b,flag=a>b?1:0;
	long long min=max==a?b:a;
	long long ans=(long long)9e18;
        if(a*b>=n)
        {cout<<a*b<<"\n"<<a<<" "<<b;
          return 0;}
	for(long long x=min;x<=n;x++)
	{long long y=((n-1)/x)+1;
		if(x>y)
			break;
		if(x*y<ans&&y>=max)
		{ans=x*y;
			ax=x;ay=y;}
	}cout<<ans<<"\n";
         if(!flag) cout<<ax<<" "<<ay;
         else cout<<ay<<" "<<ax; 
	return 0;
}
