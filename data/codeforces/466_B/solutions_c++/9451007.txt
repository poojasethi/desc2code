#include <cstdio>
#include <algorithm>
int main() {
	long long n, a , b, n1, n2, l = 10000000000007; bool o = 0;
	scanf("%I64d %I64d %I64d", &n, &a, &b); n *= 6;
	if(a * b >= n) { printf("%I64d\n%I64d %I64d",a * b, a, b); return 0; }
	if(a > b) o = 1;
	for(long long m = std::min(a, b); m * m <= n; ++m) {
		long long m2 = n / m;
		if(m2 * m < n) ++m2;
		if(m2 >= std::max(a, b) && m2 * m < l) {
			l = m2 * m; n1 = m; n2 = m2;
		}
	}
	if(o) std::swap(n1, n2);
	printf("%I64d\n%I64d %I64d",n1 * n2, n1, n2);
}
