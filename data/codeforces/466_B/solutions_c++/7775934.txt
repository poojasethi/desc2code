# include <iostream>
# define fi cin
# define fo cout
# include <algorithm>
using namespace std;
int main(void)
{
	long long n,a,b;
	fi>>n>>a>>b;
	n*=6;
	bool ok=bool(a>b);
	if (ok) swap(a,b);
	long long besta=a,bestb=b;
	for (long long i=a;i*i<=n;++i)
	{
		long long y=(n+i-1)/i;
		if (y<b) y=b;
		if (i*y<besta*bestb || besta*bestb<n) besta=i,bestb=y;
	}
	a=besta;b=bestb;
	if (ok) swap(a,b);
	fo<<b*a<<'\n'<<a<<' '<<b<<'\n';
	return 0;  
}
