#include <bits/stdc++.h>
using namespace std;
long long n,a,b,minn=98776543987654321,x,y;
bool swapp=false;
int main(){
	cin>>n>>a>>b;

	if(a>b){
		swap(a,b);
		swapp=true;
	}
	if(a*b>=(n*6)){
		cout<<a*b<<"\n"<<a<<" "<<b<<endl;
		return 0;
	}
	for(int q=a;q<=sqrt(n*6);q++){	
		long long bb=0;
		bb=(n*6)/q;
		if((6*n)%q!=0)bb++;
		//cout<<q<<" * "<<bb<<" = "<<q*bb<<endl;
		if(q*bb<minn && bb>=b){
			minn=q*bb;
			x=q;
			y=bb;
		}

	}
	if(swapp){
		swap(x,y);
	}
	cout<<minn<<"\n"<<x<<" "<<y<<endl;

	return 0;
}
