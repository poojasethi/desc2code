#include<stdio.h>
#include<math.h>
#include<algorithm>
using namespace std;
long long n,a,b,ansn=7e9,ansa,ansb;
int main()
{
	long long i,a0,b0;
	scanf("%I64d%I64d%I64d",&n,&a,&b);
	n*=6;
	if(a*b<n)
		for(i=sqrt(n);i;i--)
		{
			a0=max(a,i),b0=max(b,(n-1)/a0+1);
			if(a0*b0<ansn)
				ansn=a0*b0,ansa=a0,ansb=b0;
			b0=max(b,i),a0=max(a,(n-1)/b0+1);
			if(a0*b0<ansn)
				ansn=a0*b0,ansa=a0,ansb=b0;
		}
	else ansn=a*b,ansa=a,ansb=b;
	printf("%I64d\n%I64d %I64d\n",ansn,ansa,ansb);
	return 0;
}
