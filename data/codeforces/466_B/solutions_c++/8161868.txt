#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll n,a,b;
int main()
{
	int flag=0;
	cin>>n>>a>>b;
	if(a*b>=6*n){
		cout<<a*b<<endl<<a<<" "<<b;
		return 0;
	}
	n*=6;
	if(a>b){
		 swap(a,b);
		 flag=1;
	}
	ll ans=1e18,a1,b1,tb;
	for(ll i=a;i*i<=n;i++){
		tb=n/i;
		if(tb*i<n) tb++;
		if(tb<b) continue;
		if(tb*i<ans){
			ans=tb*i;
			a1=i;
			b1=tb;
		}
	}
	if(flag) swap(a1,b1);
	cout<<ans<<endl<<a1<<" "<<b1; 
}
