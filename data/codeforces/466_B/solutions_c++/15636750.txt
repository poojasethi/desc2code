#include <iostream>
#include <cmath>
using namespace std;
typedef unsigned long long ll;
int main()
{
ll n,j,a,b,x,y;
cin>>n>>a>>b; x=a; y=b;
n*=6;
if (n<a*b) {cout<<a*b<<endl<<a<<" "<<b;}
else{
bool t=false;
while (!t)
{
    if (x<y) {swap(x,y);}
    for (ll i=y;i<sqrt(n);i++)
    {
        if (n%i==0 && n/i>=x) {x=i; y=n/i; i=sqrt(n)+1; t=true;}
    }
    n++;
}
    if (a<=b) {cout<<x*y<<endl<<x<<" "<<y;}
        else  {cout<<x*y<<endl<<y<<" "<<x;}
}
}
