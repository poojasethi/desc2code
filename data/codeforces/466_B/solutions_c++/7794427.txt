#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
	long long int n,a,b;
	scanf("%lld%lld%lld",&n,&a,&b);
	n*=6;
	if (a*b >= n)
	{
		printf("%lld\n%lld %lld\n",a*b,a,b);
		return 0;
	}
	n--;
	while(1)
	{
		n++;
		long long int x;
		for (x=min(a,b);x*x <= n && n%x ;x++);
		if (x*x > n) continue;
		if ( n/x < max(a,b) ) continue;
		printf("%lld\n",n);
		if ( a < b )
			printf("%lld %lld\n",x,n/x);
		else printf("%lld %lld\n",n/x,x);
		return 0;
	}
}
