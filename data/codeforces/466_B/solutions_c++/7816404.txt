#include <iostream>
#include <cstdlib>
#include <cstdio>
using namespace std;

int main(){
	long long n,a,b,aa,bb,ans;
	cin>>n>>a>>b;
	n*=6;
	if (a*b>=n) cout<<a*b<<endl<<a<<' '<<b<<endl;
	else{
		int flag=0,ok=0;
		if (a>b){flag=1; swap(a,b);}
		for (long long s=n;;s++){
			for (long long x=a;(x*x<=s && x*b<=s && s/x>=b);x++)if (s%x==0){
				long long y=s/x;
				if (flag) swap(x,y);
				cout<<s<<endl<<x<<' '<<y<<endl;
				ok=1;
				break;
			}
			if (ok) break;
 		}
	}
}
