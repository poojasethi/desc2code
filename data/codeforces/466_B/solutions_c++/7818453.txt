#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
    long long int n,a,b,x,y,j,min=1e18;
    bool a7a=false;
    cin>>n>>a>>b;
    if(6*n<=a*b)
        {cout<<a*b<<endl<<a<<" "<<b;return 0; }
    if(a>b)
        {int temp=a;a=b;b=temp;a7a=true; }
    n*=6;
    for(long long int i=a;(i*i)<=n;i++)
    {
        j=(n/i)+(n%i!=0);
        if(min>i*j&&j>=b){min=i*j;x=i;y=j; }
    }
    if(a7a)
        {int temp=x;x=y;y=temp; }
    cout<<min<<endl<<x<<" "<<y;


    return 0;
}
