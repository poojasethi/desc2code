#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long n,a,b;
	cin>>n>>a>>b;
	n*=6;
	if(a*b>=n){
		cout<<a*b<<endl<<a<<" "<<b;
		return 0;
	}
	long long i,g,h,p=0;
	g=min(a,b);h=max(a,b);
	while(1){
		for(i=g;i*i<=n;i++)
		{
			//if(i>=h && n/i<g)
				//break;
			if(n%i==0)
			{
				if(n/i>=h){
				if(a<=i && b<=n/i){
				a=i;
				b=n/i;
				p=1;
				}
				else{
					a=n/i,b=i,p=1;
				}
				}
				break;
			}
		}
		if(p)
			break;
		n++;
	}
	cout<<a*b<<endl<<a<<" "<<b;
	return 0;
}
