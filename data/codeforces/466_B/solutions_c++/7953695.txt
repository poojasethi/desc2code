#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
int main()
{
    ll n,a,b,c,d,o,oa,ob;bool bl=0;
    scanf("%I64d%I64d%I64d",&n,&a,&b);
    n*=6;
    if((o=a*b)>n)
    {
        printf("%I64d\n%I64d %I64d\n",o,a,b);
        return 0;
    }
    if(a>b)swap(a,b),bl=1;
    o=6000000006;
    for(int i=a;i<=n;++i)
    {
        c=(n-1)/i+1;
        if(c<b||i>c)break;
        d=i*c;
        if(d<o){o=d;oa=i;ob=c;}
    }
    if(bl)swap(oa,ob);
    printf("%I64d\n%I64d %I64d\n",o,oa,ob);
}
