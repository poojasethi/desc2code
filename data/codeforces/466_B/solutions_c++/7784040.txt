#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cerr<<#x<<":"<<x<<"\n"
long long n,a,b,f,x,y,mn,i,j;
int main()
{
  scanf("%I64d%I64d%I64d",&n,&a,&b);
  if(a>b)
  {
    f=1;
    swap(a,b);
  }
  n*=6;
  mn=1000000000000000001ll;
  for(i=a;i*i<=n || i==a;i++)
  {
    j=max((n+i-1)/i,b);
    if(i*j<mn)
    {
      mn=i*j;
      x=i;
      y=j;
    }
  }
  if(f)
    swap(x,y);
  printf("%I64d\n%I64d %I64d\n",mn,x,y);
	return 0;
}
