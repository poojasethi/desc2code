#include <iostream>
using namespace std;
int main()
{
	long long n, a, b, x, y, s, ans=2e18, sx, sy;
	cin>>n>>a>>b;
	s=6*n; x=a; do {
		y = max(b,(s-1)/x+1);
		if (x*y<ans) {
			ans=x*y;
			sx=x; sy=y;
		}
		x++;
	} while (x<a+100000);
	y=b; do {
		x = max(a,(s-1)/y+1);
		if (x*y<ans) {
			ans=x*y;
			sx=x; sy=y;
		}
		y++;
	} while (y<b+100000);
	cout<<ans<<endl<<sx<<" "<<sy<<endl;
	return 0;
}
