#include<iostream>
#include<algorithm>

using namespace std;
typedef  long long int64;

int64 MAX=1e5;

int main(){
	int64 n,a,b;
	cin>>n>>a>>b;

	int64 s=0x7FFFFFFFFFFFFFFF,a1,b1;

	if(a*b>=6*n){
		cout<<a*b<<endl<<a<<' '<<b<<endl;
		return 0;
	}

	int flag=0;
	if(b<a){swap(a,b);flag=1;}

	for(int64 x=a;x<=MAX;x++){
		int64 y=6*n/x;
		if((6*n)%x){y++;}
		if(y<b){y=b;}
		if(x*y<s){s=x*y;a1=x;b1=y;}
	}

	if(flag){swap(a1,b1);}
	cout<<s<<endl<<a1<<' '<<b1<<endl;

	return 0;
}
