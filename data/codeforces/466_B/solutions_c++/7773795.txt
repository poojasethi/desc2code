#include <bits/stdc++.h>
using namespace std;

int main() {
    long long n, a, b;

    cin >> n >> a >> b;
    long long x = 0x3f3f3f3f, y = 0x3f3f3f3f;
    long long xx, yy;
    for (long long i = 1; i <= 100000; i++) {
        xx = max(a, i);
        yy = max(b, (6*n+i-1)/i);
        if (xx*yy < x*y)
            x = xx, y = yy;
        xx = max(a, (6*n+i-1)/i);
        yy = max(b, i);
        if (xx*yy < x*y)
            x = xx, y = yy;
    }
    cout << x*y << endl;
    cout << x << " " << y << endl;
}
