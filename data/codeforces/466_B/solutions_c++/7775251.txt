#include<iostream>
#include<algorithm>
using namespace std;

int main(){
    long long n,a,b;
    cin>>n>>a>>b;n*=6;
    bool swaped=(a>b);
    if (swaped) swap(a,b);
    long long aa=a,bb=b;
    for (long long i=a;i*i<=n;i++){
        long long t=(n+i-1)/i;
        if (t<b) t=b;
        if (aa*bb<n || i*t<aa*bb){
            aa=i;
            bb=t;
        }
    }
    if (swaped) swap(aa,bb);
    cout<<aa*bb<<endl<<aa<<' '<<bb<<endl;
    return 0;
}
