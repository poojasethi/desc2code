//In the Name of God

#include<iostream>
#define int long long

using namespace std;

int n,a,b,a1,b1,S,nn;

main()
{
  cin>>n>>a>>b;
  nn=6*n;
  S=1e18+1e18;
  a1=a;
  b1=b;
  for(int i=a;i*i<=nn;i++)
    if((nn+i-1)/i*i<S && (nn+i-1)/i>=b)
      {
	S=(nn+i-1)/i*i;
	a1=i;
	b1=(nn+i-1)/i;
      }
  for(int i=b;i*i<=nn;i++)
    if((nn+i-1)/i*i<S && (nn+i-1)/i>=a)
      {
	S=(nn+i-1)/i*i;
	b1=i;
	a1=(nn+i-1)/i;
      }
  S=a1*b1;
  cout<<S<<endl;
  cout<<a1<<" "<<b1<<endl;
  return 0;
}
