#include <iostream>

using namespace std;

typedef long long int ll;

void calcula(ll n,ll &a,ll &b)
{
  ll sola=a;
  ll solb=max(b,n/a+(n%a!=0));
  ll sol=sola*solb;
  for (;;) {
    ll bb=n/a+(n%a!=0);
    ll bbb=max(b,bb);
    if (a*bbb<sol) {
      sol=a*bbb;
      sola=a;
      solb=bbb;
    }
    if (bb<=a) break;
    a++;
  }
  a=sola;
  b=solb;
}

int main()
{
  ll n,a,b;
  cin>>n>>a>>b;
  n*=6;
  if (a<=b) calcula(n,a,b);
  else calcula(n,b,a);
  cout<<a*b<<endl;
  cout<<a<<" "<<b<<endl;
}
