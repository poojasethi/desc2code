n,a,b = map(int,raw_input().split())

m = 6*n

a = max(1,a)
b = max(1,b)

swap = False
if a > b:
	a,b = b,a
	swap = True


# print "a"
a1 = a
b1 = max(((m-1)/a1+1), b)

a2 = a1
b2 = b1
A = a2 * b2

limit = int(m**0.5)+1

while a1 < limit and A > m:
	a1 += 1
	b1 = max(((m-1)/a1+1), b)
	# print A, a1,b1,a1*b1,m
	if A > a1 * b1:
		a2 = a1
		b2 = b1
		A = a2 * b2


if swap:
	a2,b2=b2,a2

print a2*b2
print a2,b2
