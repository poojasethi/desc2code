#!/usr/bin/python

from math import sqrt, ceil

n, a, b = [int(n) for n in raw_input().split()]
if 6*n <= a*b:
	print a*b, a, b
else:
	a1 = -1
	b1 = -1
	s = 10**11
	for i in range(min(a, b), int(sqrt(6*n)) + 2):
		j = 6*n/i + 1 * (6*n % i != 0)
		if j >= max(a, b):
			area = i * j
			if area < s:
				s = area
				a1 = i
				b1 = j
	if b < a:
		a1, b1 = b1, a1
	print int(s), int(a1), int(b1)
