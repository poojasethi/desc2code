(n, a, b) = [int(x) for x in raw_input().split(" ")]
r = -1
ra = 0
rb = 0
for i in xrange(0, 100000):
    ta = a + i
    tb = (6 * n + ta - 1) / ta
    tb = max(tb, b)
    if r == -1 or ta * tb < r:
        r = ta * tb
        ra = ta
        rb = tb
    tb = b + i
    ta = (6 * n + tb - 1) / tb
    ta = max(ta, a)
    if r == -1 or ta * tb < r :
        r = ta * tb
        ra = ta
        rb = tb
print(r)
print(str(ra) + " " + str(rb))
