n,a,b=map(int,raw_input().split())
swap=False
if a>b:
    a,b=b,a
    swap=True
if a*b>=6*n:
    print a*b,a,b
else:
    minarea=10**18
    for newa in range(a,int(ceil((6*n)**0.5))+1):
        newb=max(b,int(ceil(6*n*1.0/newa*1.0)))
      #  print newa*newb
        if newa*newb<minarea:
            na,nb=newa,newb
            minarea=newa*newb
    print minarea
