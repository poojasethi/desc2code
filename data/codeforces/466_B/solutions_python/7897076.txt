n,a,b=map(int,raw_input().split())
n*=6
if (a*b>=n):
    print ('%d\n%d %d' %(a*b,a,b))
else:
    swp=False
    if (a>b):
        swp=True
        a,b=b,a
    x=a
    A,B=0,0
    while (x*x<=n):
        y=(n+x-1)//x
        if (y>=b and (A==0 or A*B>x*y)):
            A,B=x,y
        x+=1
    if(swp):
        A,B=B,A
    print ('%d\n%d %d' %(A*B,A,B))
