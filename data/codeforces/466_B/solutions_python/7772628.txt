#!/usr/bin/env python


def read():
    return map(int, raw_input().split())


def calc(n, a, b):
    minV = 1 << 60
    A = -1
    B = -1
    
    while 1:

        if minV > a * max((n + a - 1) / a, b):
            minV = a * max((n + a - 1) / a, b)
            A = a
            B = max((n + a - 1) / a, b)

        if a > (n + a - 1) / a:
            break
        
        a += 1

    return A, B


def work((n, a, b)):
    A1, B1 = calc(n * 6, a, b)
    B2, A2 = calc(n * 6, b, a)

    if A1 * B1 > A2 * B2:
        A1 = A2
        B1 = B2

    print A1 * B1
    print A1, B1


if __name__ == "__main__":
    work(read())
