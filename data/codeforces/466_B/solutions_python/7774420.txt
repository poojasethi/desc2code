n, a, b = map(int, raw_input().split())
r = 10 ** 20
n *= 6
ca, cb = a, b
while ca * cb <= n:
    if (ca - n % ca) % ca < r:
        r = (ca - n % ca) % ca
        rb = n / ca
        if r != 0:
            rb += 1
        ra = ca
    if (cb - n % cb) % cb < r:
        r = (cb - n % cb) % cb
        ra = n / cb
        if r != 0:
            ra += 1
        rb = cb
    ca += 1
    cb += 1
if ca * cb - n < r:
    ra = ca
    rb = cb
print ra * rb
print ra, rb
