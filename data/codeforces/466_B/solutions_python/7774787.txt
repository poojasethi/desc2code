import math

(n, a, b) = map(int, raw_input().split())
n *= 6

if a * b >= n:
	print a * b
	print a, b
	exit(0)

r = False
if a > b:
	(a, b) = (b, a)
	r = True

m = 9999999999
mA = 0
mB = 0

for i in range(a, int(math.floor(math.sqrt(n))) + 1):
	ni = int(math.ceil(float(n) / i))
	if ni >= b and i * ni < m:
		m = i * ni
		mA = i
		mB = ni

print m

if r:
	(mA, mB) = (mB, mA)
print mA, mB
