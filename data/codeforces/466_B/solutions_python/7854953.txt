import math
n,a,b = [int(x) for x in raw_input().split()]
flip = False
if a > b:
    flip = True
    a,b = b,a
n *= 6
if a*b >= n:
    print a*b
    print a,b
else:
    ar = 8*(10**9)
    ma,mb = ar,1
    s = int(n**0.5)
    for i in range(a,s+1):
        j = int(math.ceil(float(n)/i))
        if i >= a and j >= b:
            if i*j < ar:
                ar = i*j
                ma,mb = i,j

    print ar
    if flip:    print mb,ma
    else:   print ma,mb
