X = [-1] * 2;
A = [-1] * 2;
N = -1; 
ans = -1;
def func(cur, x) :
	global ans
	P = 6 * N / (A[cur] + x);
	rem = 6 * N % (A[cur] + x) > 0;
	P = max(P + rem, A[cur ^ 1]);
	res = (A[cur] + x) * P;
	if ans == -1 : ans = res;
	if (res <= ans) :
		ans = res;
		X[cur] = A[cur] + x; X[cur ^ 1] = P;
read = lambda : map(int, raw_input().split())
N, A[0], A[1] = read();
for x in range(100000) :
	func(0, x);
	func(1, x);
print ans
print X[0], X[1]
