#!/usr/bin/python2
# -*- coding: utf-8 -*-

import sys
import os

def main():
    n, a, b = map(int, sys.stdin.readline().split())
    target = 6 * n
    if a * b >= target:
        print a * b
        print a, b
        return
    if a > b:
        a, b = b, a
        swapped = True
    else:
        swapped = False

    best = 10 ** 18
    ans = (-1, -1)
    for ka in xrange(a, 10 ** 5):
        kb = (target + ka - 1) / ka
        if kb < b:
            break
        area = ka * kb
        if area < best:
            best = area
            ans = (ka, kb)

    if swapped:
        ans = (ans[1], ans[0])
    print best
    print ans[0], ans[1]



if __name__ == '__main__':
    main()
