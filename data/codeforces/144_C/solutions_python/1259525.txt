import string
s=raw_input()
p=raw_input()
d=dict()
k=dict()
for i in string.ascii_lowercase:
    d[i]=0
    k[i]=0
for i in p:
    d[i]=d[i]+1
l=0
z=0
ss=0
ln=len(p)
for r in range(len(s)):
    if s[r]!='?':
       if (k[s[r]]==d[s[r]]):
            z+=1
       k[s[r]]=k[s[r]]+1
    if r-l+1>ln:
        if s[l]!='?':
            k[s[l]]-=1
            if (k[s[l]]==d[s[l]]):
                z-=1
        l+=1
    if (r-l+1==ln)and(z==0):
        ss+=1
print ss
