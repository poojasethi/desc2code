import sys
s=sys.stdin.readline()
p=sys.stdin.readline()
cnt1=[0]*30
cnt2=[0]*30
p=p[0:len(p)-1]
for i in range(len(p)):
	cnt1[ord(p[i])-ord('a')]+=1
res=0
for i in range(len(s)-1):
	t=i-len(p)
	if (t>=0 and s[t]!='?'):
		cnt2[ord(s[t])-ord('a')]-=1
	if (s[i]!='?'):
		cnt2[ord(s[i])-ord('a')]+=1
	av=1
	if (i>=len(p)-1):
		for j in range(30):
			if (cnt1[j]<cnt2[j]):
				av=0
				break
		if (av==1):
			res+=1
print res
