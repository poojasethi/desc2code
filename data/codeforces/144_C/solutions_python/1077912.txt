a = raw_input()
b = raw_input()
l = len(b)
model = [0]*26
for i in range(l):
    model[ord(b[i])-ord('a')] = model[ord(b[i])-ord('a')]+1
cnt = [0]*26
total = 0
for i in range(len(a)):
    if a[i] != '?': cnt[ord(a[i])-ord('a')] = cnt[ord(a[i])-ord('a')]+1
    if i >= l and a[i-l] != '?': cnt[ord(a[i-l])-ord('a')] = cnt[ord(a[i-l])-ord('a')]-1
    for j in range(26):
        if cnt[j] > model[j]:
            break
    else:
        if i >= l-1: total = total + 1
print(total)
