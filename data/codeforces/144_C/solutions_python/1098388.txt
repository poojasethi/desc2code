s = raw_input ()
p = raw_input ()
n = len (s)
m = len (p)

countInP = [0] * 26
for i in p:
	countInP[ord (i) - ord ('a')] += 1

ans = 0

for i in xrange (n):
	if s[i] != '?': countInP[ord (s[i]) - ord ('a')] -= 1
	if i < m - 1 : continue
	if i - m >= 0 and s[i - m] != '?': countInP[ord (s[i - m]) - ord ('a')] += 1
	if all (k >= 0 for k in countInP): ans += 1
print ans
