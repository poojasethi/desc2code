#-*- coding: utf-8 -*-

s = raw_input()
p = raw_input()
count = 0
dict = {}

for c in p:
    dict.setdefault(c, 0)
    dict[c] += 1

for i in range(len(s)):
    if s[i] != '?':
        dict.setdefault(s[i], 0)
        dict[s[i]] -= 1
    
    if i < len(p)-1:
        continue

    if i-len(p) >= 0 and s[i-len(p)] != '?':
        dict.setdefault(s[i-len(p)], 0)
        dict[s[i-len(p)]] += 1
   
    if all(v >= 0 for v in dict.values()):
        count += 1

print count
