#!/usr/bin/python2
def test(a, b):
  for i in b:
    if (i != '?'):
      if (a[i] < b[i]):
        return False
#  print a
#  print b
  return True

s = raw_input()
p = raw_input()
if (len(p) > len(s)):
  ans = 0
else:
  a = {}
  b = {}
  for c in p:
    a[c] = 0
    b[c] = 0
  for c in s:
    a[c] = 0
    b[c] = 0
  for c in p:
    a[c] += 1
  for i in xrange(len(p)):
    b[s[i]] += 1
  ans = 0
  if (test(a, b)):
    ans += 1
  for i in xrange(0, len(s) - len(p)):
    b[s[i]] -= 1
    b[s[i + len(p)]] += 1
    if (test(a, b)):
      ans += 1
#      print i
print ans
