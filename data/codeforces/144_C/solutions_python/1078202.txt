s=raw_input();
p=raw_input();
sa=26*[0]
pa=26*[0]
for c in p:
	pa[ord(c)-ord('a')]+=1
c=0
qc=0
for i in xrange(len(s)):
	if s[i] != '?':
		sa[ord(s[i])-ord('a')]+=1
	else:
		qc+=1
	if i >= len(p):
		if s[i-len(p)] != '?':
			sa[ord(s[i-len(p)])-ord('a')]-=1
		else:
			qc-=1
	if i >= len(p)-1:
		sum=0
		flag=True
		for i in xrange(26):
			if(sa[i] > pa[i]):
				flag=False
			else:
				sum+=pa[i]-sa[i]
		if flag and sum == qc:
			c+=1
print c
