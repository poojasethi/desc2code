f = lambda x : [ord(x) - ord('a'), 26][x == '?']
g = lambda x : all(c <= 0 for c in x[ : 26])
s, p = map(f, raw_input()), map(f, raw_input())
n, m = len(s), len(p)
c = [s[ : m].count(i) - p.count(i) for i in range(27)]
r = [0, 1][g(c) and n >= m]
for i in range(m, n):
	c[s[i - m]] -= 1
	c[s[i]] += 1
	r += g(c)
print r
