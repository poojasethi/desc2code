def f(c):
    return ord(c)-ord('a')
    
a,b=raw_input(),raw_input()
c,d=[0]*26,[0]*26

for x in b:
    d[f(x)]+=1
ans=0
lb=len(b)
for i in xrange(len(a)):
    if a[i]!='?':c[f(a[i])]+=1
    if i-len(b)>=0 and a[i-len(b)]!='?': c[f(a[i-lb])]-=1
    if i>=len(b)-1 and all(c[j]<=d[j] for j in xrange(26)):
        ans+=1
print ans
