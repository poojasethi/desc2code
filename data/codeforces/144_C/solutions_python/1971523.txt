def main():
    a = raw_input()
    b = raw_input()

    d = [0] * 128
    for x in b:
        d[ord(x)] += 1

    ans = 0
    for i in xrange(len(a)):
        j = i - len(b)
        if j >= 0 and a[j] != '?':
            d[ord(a[j])] += 1
        if a[i] != '?':
            d[ord(a[i])] -= 1
        if i >= len(b)-1 and all(x >= 0 for x in d[ord('a'): ord('z')+1]):
            ans += 1
    print ans

if __name__ == '__main__':
    main()
