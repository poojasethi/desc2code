s = raw_input().strip()
p = raw_input().strip()
ls = len(s)
lp = len(p)
dicp = [0] * 26
for c in p:
    dicp[ord(c) - ord('a')] += 1
i = 0
ans = 0
for i in xrange(ls):
    if s[i] != '?':
        dicp[ord(s[i]) - ord('a')] -= 1
    if i < lp - 1: continue
    if all([k >= 0 for k in dicp]):
        ans += 1
    if s[i + 1 - lp] != '?':
        dicp[ord(s[i + 1 - lp]) - ord('a')] += 1
print ans
