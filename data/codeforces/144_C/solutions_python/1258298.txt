
latin = "abcdefghijklmnopqrstuvwxyz"
characters =  "abcdefghijklmnopqrstuvwxyz?"

def satisfies(curr,goal):
	if any(curr[c] > goal[c] for c in latin):
		return False
	else:
		s = sum(goal[c]-curr[c] for c in latin)
		return s == curr['?']
s = raw_input().strip()
p = raw_input().strip()
goalCts = dict([(c, p.count(c)) for c in characters])
if len(p) > len(s):
	print 0
else:
	num = 0
	#initial dictionary
	letCts = dict([(c,0) for c in characters])
	for i in xrange(0,len(p)):
		c = s[i+0]
		letCts[c] += 1
	if satisfies(letCts,goalCts):
		num += 1
	for offset in xrange(1,len(s)-len(p)+1):
		#remove the previous character
		c = s[offset-1]
		letCts[c] -= 1

		#add the new character
		c = s[offset+len(p)-1]
		letCts[c] += 1
		if satisfies(letCts,goalCts):
			num += 1
	print num
