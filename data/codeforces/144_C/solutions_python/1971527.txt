def main():
    a = raw_input()
    b = raw_input()

    d = [0] * 26
    diff = ord('a')
    for x in b:
        d[ord(x)-diff] += 1

    ans = 0
    t = len(b) - 1
    for i in xrange(len(a)):
        j = i - len(b)
        if j >= 0 and a[j] != '?':
            d[ord(a[j])-diff] += 1
        if a[i] != '?':
            d[ord(a[i])-diff] -= 1
        if i >= t and all(x >= 0 for x in d):
            ans += 1
    print ans

if __name__ == '__main__':
    main()
