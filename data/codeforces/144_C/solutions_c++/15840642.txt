#include<bits/stdc++.h>
using namespace std;
#define ll long long 
ll cnts[130],cntp[130];

int main()
{
string s,p;
ll slen,plen,i,j,k;
cin>>s>>p;

slen=s.length();
plen=p.length();

if(plen>slen)
{
cout<<"0\n";
return 0;
}

for(i=0;i<plen;i++)
    cntp[p[i]]++;

for(i=0;i<plen;i++)
    cnts[s[i]]++;

i=0,j=plen-1;
ll ans=0;

while(j<slen)
{
    for(k=97;k<=122;k++)
    {
        if(cnts[k]>cntp[k])
            break;
    }
    
    if(k==123)
        ans++;
    cnts[s[i]]--;
    j++;
    i++;
    if(j==slen)
        break;
    cnts[s[j]]++;
}

cout<<ans<<"\n";
return 0;
}
