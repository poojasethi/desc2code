#include <bits/stdc++.h>
using namespace std;

#define SIZE 100005

int main() {
	char S[SIZE] , P[SIZE];
	scanf("%s%s" , S , P);

	int sz1 = strlen(P);
	int sz2 = strlen(S);

	int A[26] = {} , B[26] = {};

	for(int i = 0 ; i < sz1 ; ++i)
		++A[P[i] - 'a'];

	int res = 0;
	for(int i = 0 ; i < sz2 ; ++i){
		if(S[i] != '?')++B[S[i] - 'a'];
		if(i >= sz1 && S[i - sz1] != '?')--B[S[i - sz1] - 'a'];
		if(i >= sz1 - 1){
			bool find = 1;

			for(int j = 0 ; j < 26 ; ++j)
				if(B[j] > A[j])find = 0;

			res += find;
		}
	}
	printf("%d\n" , res);
	return 0;
}
