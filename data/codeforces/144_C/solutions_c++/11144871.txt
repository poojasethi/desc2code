#include<bits/stdc++.h>
using namespace std;

int occ[200];
int cur[200];
int main(){
    string s,p;
    cin>>s>>p;


    int psz=(int)p.size();
    int ssz=(int)s.size();


    for(int i=0;i<psz;i++)occ[p[i]]++;
    for(int i=0;i<psz && i<ssz;i++)cur[s[i]]++;


    int res=0;
    for(int i=0;i+psz<=ssz;i++){

        bool ok=1; for(char j='a';j<='z';j++)if(cur[j]>occ[j])ok=0;
        if(ok)res++;

        if(i+psz<ssz){
            cur[s[i]]--;
            cur[s[i+psz]]++;
        }

    }
    cout<<res<<"\n";



}
