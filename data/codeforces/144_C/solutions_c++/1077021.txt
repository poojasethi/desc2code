#include<cstdio>
#define maxn 100005
char s[maxn],p[maxn];
int cnt[1<<8],c[1<<8];
int main()
{
    scanf("%s%s",s,p);
    int i,j,len,ans=0;
    for(i=0;p[i];i++) cnt[p[i]]++;
    len=i;
    for(i=0;i<len-1;i++) c[s[i]]++;
    for(;s[i];i++)
    {
        c[s[i]]++;
        if(i>=len) c[s[i-len]]--;
        for(j='a';j<='z';j++)
        {
            if(c[j]>cnt[j]) {ans--;break;}
        }
        ans++;
    }
    printf("%d\n",ans);
    return 0;
}
