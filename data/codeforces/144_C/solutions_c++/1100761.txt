#include <cstdio>
#include <string>
#include <cstring>
#define INF 1000000000

int i,j,k,n,m,res;
char s1[111111],s2[111111];
int c[256],d[256];
bool l;

int main()
{
//	freopen("c.in","r",stdin);
//	freopen("c.out","w",stdout);
	gets(s1);
	gets(s2);
	n = strlen(s1);
	m = strlen(s2);
	if(n<m)
	{
		printf("0\n");
		return 0;
	}
	for(i=0;i<m;i++)
	{
		c[s1[i]]++;
		d[s2[i]]++;
	}
	for(i=0;i<=n-m;i++)
	{
		l=true;
		for(j='a';j<='z';j++)
			if(c[j]>d[j])
			{
				l=false;
				break;
			}
		if(l)
			res++;
		c[s1[i]]--;
		c[s1[i+m]]++;
	}
	printf("%d\n",res);
	return 0;
}
