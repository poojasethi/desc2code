#include <algorithm>
#include <stdio.h>

#define MAX 100010

using namespace std;

int m;
char strCit[MAX], strMic[MAX];
int ap[512], ac[512];

int main()
{
	fgets(strCit + 1, MAX, stdin);

	fgets(strMic + 1, MAX, stdin);

	for (m = 1; strMic[m] != '\n'; m++)
		ap[strMic[m]]++;
	m--;

	int sol = 0;
	for (int i = 1; strCit[i] != '\n'; i++)
	{
		ac[strCit[i]]++;
		if (i >= m)
		{
			int ok = 1;
			for (int c = 'a'; c <= 'z'; c++)
				if (ap[c] < ac[c])
					ok = 0;

			sol += ok;
			ac[strCit[i - m + 1]]--;
		}
	}

	printf("%d\n", sol);

	return 0;
}
