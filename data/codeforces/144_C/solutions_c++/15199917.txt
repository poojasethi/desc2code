#include <iostream>
#include <string>

using namespace std;
int a[100005][30] = {0}, b[30] = {0};
int main()
{
 string s,p;
 int i, j, n, m, ans = 0;
 cin>>s>>p;
 n = s.size();
 m = p.size();
 for(i = 0; i < n; i++)
 {  
 	for(j = 0; j <= 26; j++) a[i + 1][j] = a[i][j];
 	if(s[i] != '?') a[i + 1][s[i]-96]++;
 	else a[i + 1][0]++;
 }
  for(i = 0; i < m; i++) b[p[i]-96]++;
 for(i = 1; i <= n - m + 1; i++)
     { 
     int k = i + m - 1;
        for(j = 1; j <= 26; j++) if(a[k][j] - a[i - 1][j] > b[j]) break;
		if(j > 26) ans++;
	 }
	 cout<<ans;
 return 0;
}
