#include <iostream>
#include <string>

using namespace std;

string S,P;
int A[26],B[26],cnt;

bool isAnagram()
{
	for (int i=0;i<26;i++)
		if (B[i]>A[i])
			return false;
	return true;
}

int main()
{
	cin >> S >> P;
	if (S.size()>=P.size())
	{
		for (int i=0;i<P.size();i++)
		{	
			A[P[i]-'a']++;
			if (S[i]!='?')
				B[S[i]-'a']++;
		}
		int start=0,end=P.size()-1;
		while (1)
		{
			if (isAnagram())
				cnt++;
			if (S[start]!='?')
				B[S[start]-'a']--;
			if (end==S.size()-1)
				break;
			start++;
			end++;
			if (S[end]!='?')
				B[S[end]-'a']++;
		}
	}
	cout << cnt << endl;
	return 0;
}
