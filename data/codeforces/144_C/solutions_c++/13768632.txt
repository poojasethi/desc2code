#include <iostream>
using namespace std;

string s, t;
int a[256];

int main() {
	cin >> s >> t;
	if (s.length() < t.length()) {
		cout << 0 << endl;
		return 0;
	}
	a['?'] = 0xDEADBEE;
	for (int i = 0; i < t.length(); ++i) {
		a[t[i]]++;
	}
	int sobra = 0, c = 0;
	for (int i = 0; i < t.length(); ++i) {
		if (a[s[i]]-- == 0)
			++sobra;
	}
	if (sobra == 0)
		++c;
	for (int i = t.length(); i < s.length(); ++i) {
		if (++a[s[i - t.length()]] == 0)
			--sobra;
		if (a[s[i]]-- == 0)
			++sobra;
		if (sobra == 0)
			++c;
	}
	cout << c << endl;
}
