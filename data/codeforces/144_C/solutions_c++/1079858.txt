#include<iostream>
#include<cstring>
using namespace std;

long i,j,k,x,y,z,n,ans;
long f[27],a[27];
char s[100001],p[100001];

int main(){
    cin>>s;cin>>p;
    if(strlen(p)>strlen(s)){cout<<0<<endl;return 0;}
    x=strlen(p);y=strlen(s);
    for(i=0;i<x;i++)a[(long)(p[i])-97]++;
    for(i=0;i<x;i++)if(s[i]!='?')f[(long)(s[i])-97]++;
    for(i=0;i<=y-x;i++){
      for(j=0;j<27;j++)if(f[j]>a[j])break;
      if(j==27)ans++;
      if(s[i]!='?')f[(long)(s[i])-97]--;
      if(i<y-x)
        if(s[i+x]!='?')f[(long)(s[i+x])-97]++;
    }
    cout<<ans<<endl;
    //system("pause");
    return 0;
}
