#include <cstring>
#include <iostream>

using namespace std;

char A[100010], B[100010];
int len1, len2;
int now[30], real[30];
int result;

int main()
{
	cin >> A >> B;
	len1 = strlen(A);
	len2 = strlen(B);
	
	for (int i = 0; i < len2; ++i)
		++real[B[i] - 'a'];
	for (int i = 0; i < len1; ++i)
	{
		if (A[i] != '?') ++now[A[i] - 'a'];
		if (i + 1 >= len2)
		{
			if (i + 1 > len2)
				if (A[i - len2] != '?')
					--now[A[i - len2] - 'a'];
			
			bool ok = true;
			for (int j = 0; j < 26; ++j)
				if (now[j] > real[j])
					ok = false;
			result += ok;
		}
	}
	
	cout << result;
}
