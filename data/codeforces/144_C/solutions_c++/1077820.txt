#include <cstdio>
#define N 100005

char a[N], b[N];
int l, o, c[26], C[27], A;

int main(){
	scanf("%s %s", a, b);
	for(o = 0; b[o]; o++) c[b[o] - 'a']++;
	for(l = 0; a[l]; l++){
		if(a[l] != '?') C[a[l] - 'a']++;
		else C[26]++;
		if(l >= o - 1){
			bool F = 1;
			int s = 0;
			for(int j = 0; j < 26 && F; j++){
				if(C[j] > c[j]) F = 0;
				else s += c[j] - C[j];
			}
			F = F && C[26] == s;
			A += F;
			if(a[l - o + 1] != '?') C[a[l - o + 1] - 'a']--;
			else C[26]--;
		}
	}
	printf("%d\n", A);
}
