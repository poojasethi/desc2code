#include <cstdio>
#include <iostream>
#include <string>
#include <string.h>
using namespace std;
int a[200];
bool check()
{
	for (int i='a';i<='z';i++)
		if (a[i]<0) return 0;
	return 1;
}

int main()
{
	string s,p;
	memset(a,0,sizeof(a));
	cin>>s>>p;
	if (s.size()<p.size()) {cout<<0<<endl;return 0;}
	for (int i=0;i<p.size();i++) a[p[i]]++;
	for (int i=0;i<p.size();i++) a[s[i]]--;
	int ans=0;
	for (int i=0;i<=s.size()-p.size();i++)
	{
		if (check()) ans++;
		a[s[i+p.size()]]--;
		a[s[i]]++;
	}
	cout<<ans<<endl;

	return 0;
}
