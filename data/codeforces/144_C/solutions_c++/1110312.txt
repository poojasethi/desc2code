#include <cstdio>#include <cstring>#include <cctype>char s[100001];char p[100001];int main() {	int sn, pn;	int sc[26] = {0}, pc[26] = {0};	gets(s); gets(p);	sn = strlen(s); pn = strlen(p);	for (int i=0; i!=pn; ++i) {		if (!s[i]) {			puts("0");			return 0;		}		if (islower(s[i])) ++sc[s[i]-'a'];		++pc[p[i]-'a'];	}	int r = 0;	for (int i=0; i<=sn-pn; ++i) {		bool b = true;		for (int j=0; j!=26; ++j) {			if (sc[j] > pc[j]) {				b = false;				break;			}		}		if (b) ++r;		if (islower(s[i])) --sc[s[i]-'a'];		if (islower(s[i+pn])) ++sc[s[i+pn]-'a'];	}	printf("%d\n", r);	return 0;}
