//In the Name of God

#include<iostream>

using namespace std;

string s,t;
int T[100],S[100],ans;

void check()
{
  for(int i=0;i<26;i++)
    if(S[i]>T[i])
      return;
  ans++;
}

int main()
{
  cin>>s>>t;
  if(t.size()>s.size())
    {
      cout<<0<<endl;
      return 0;
    }
  for(int i=0;i<t.size();i++)
    {
      T[t[i]-'a']++;
      if(s[i]!='?')
	S[s[i]-'a']++;
    }
  check();
  for(int i=t.size();i<s.size();i++)
    {
      if(s[i-t.size()]!='?')
	S[s[i-t.size()]-'a']--;
      if(s[i]!='?')
	S[s[i]-'a']++;
      check();
    }
  cout<<ans<<endl;
  return 0;
}
