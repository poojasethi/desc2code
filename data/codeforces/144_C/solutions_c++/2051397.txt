#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
#define N 100005
char s[N],t[N];
int hash[256], aim[256];
int main(){
	while (scanf("%s%s",s,t) != EOF){
		int i, len, ans = 0;
		for (i=0; t[i]; i++) hash[t[i]]++;
		len = i;
		for (i=0; i<len-1; i++) aim[s[i]]++;
		for (;s[i];i++){
			aim[s[i]]++;
			if (i>=len) aim[s[i-len]]--;
			for (int j='a';j<='z';j++)
				if (aim[j]>hash[j]){
					ans--;
					break;
				}
			
			ans++;
		}
		printf("%d\n",ans);
	}
	return 0;
}
