#include <stdio.h>
#include <string.h>

char s1[1000000],s2[1000000];
int cnt[26]={0};

bool check(void)
{
	for (int i=0;i<26;i++)
		if (cnt[i]<0)
			return 0;
	return 1;
}

int main()
{
	int l1,l2,i,ans=0;
	scanf("%s %s",s1,s2);
	l1=strlen(s1);
	l2=strlen(s2);
	if (l2>l1){puts("0");return 0;}
	for (i=0;i<l2;i++)
	{
		cnt[s2[i]-'a']++;
		if (s1[i]!='?')cnt[s1[i]-'a']--;
	}
	if (check())ans++;
	for (i=l2;i<l1;i++)
	{
		if (s1[i]!='?')cnt[s1[i]-'a']--;
		if (s1[i-l2]!='?')cnt[s1[i-l2]-'a']++;
		if (check())ans++;
	}
	printf("%d\n",ans);
}
