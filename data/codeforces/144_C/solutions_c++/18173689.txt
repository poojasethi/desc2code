#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	char c[100001],d[100001];
	cin>>c>>d;
	int n=strlen(c),count=0,m=strlen(d);
	int a[26]={0},b[26]={0};
	for(int i=0;i<m;i++){a[d[i]-97]++;if(c[i]!='?')b[c[i]-97]++;}
	for(int j=0;j<=n-m;j++)
	{
		int f=0;
		for(int i=0;i<26;i++)if(b[i]>a[i])f=1;
		if(!f)count++;
		if(c[j]!='?')b[c[j]-97]--;
		if(c[j+m]!='?')b[c[j+m]-97]++;
	}
	cout<<count;
}
