#include<bits/stdc++.h>
#define MX 100005
using namespace std;

int cnt1[500];
int cnt2[500];

int main(){
     string str1, str2;
     int i, j, k;
     int l1, l2, ans=0, c;
     cin >> str1 >> str2;
     l1=str1.length(); l2=str2.length();
     if(l1<l2) return printf("0\n")*0;
     for(i=0; i<l2; ++i){
          cnt1[str1[i]]++;
          cnt2[str2[i]]++;
     }
     for(i=0; i<=l1-l2; ++i){
          for(j='a', c=0; j<='z'; ++j) c+=abs(cnt1[j]-cnt2[j]);
          if(c==cnt1['?']) ans++;
          cnt1[str1[i]]--;
          if(i+l2<l1) cnt1[str1[i+l2]]++;
     }
     cout << ans << endl;
}
