#include<stdio.h>
#include<string.h>
int numnow[27],ans,nump[27],lens,lenp;
char s[1000001],p[1000001];
int main()
{
	scanf("%s%s",s,p);lens=strlen(s);lenp=strlen(p);
	for(int i=0;i<lenp;i++)nump[p[i]-'a']++;
	if(lenp>lens)return printf("0\n"),0;
	for(int i=0;i<lenp;i++)if(s[i]!='?')numnow[s[i]-'a']++;
	for(int i=0;i<=lens-lenp;i++)
	{
		for(int j=0;j<26;j++)
		if(numnow[j]>nump[j])break;
		else if(j==25)ans++;
		if(s[i]!='?')numnow[s[i]-'a']--;
		if(s[i+lenp]!='?')numnow[s[i+lenp]-'a']++;
	}
	printf("%d\n",ans);
	return 0;
}
