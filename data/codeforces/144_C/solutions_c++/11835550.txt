#include <bits/stdc++.h>
using namespace std;
char s[100010],p[100010];
int cnt1[27],cnt2[27];
int main()
{
	int ans,len1,len2,flag;
	scanf("%s%s",s,p);
	len1=strlen(s);
	len2=strlen(p);
	ans=0;
	for(int i=0;i<len2;i++) cnt2[p[i]-'a']++;
	for(int i=0;i<len1;i++)
	{
		flag=0;
		cnt1[s[i]-'a']++;
		if (i>=len2-1)
		{
			for(int j=0;j<=25;j++)
			{
				if(cnt2[j]<cnt1[j])
				{
					flag=1;
			 		break;
				}
			}
			if(!flag) ans++;
			cnt1[s[i-(len2-1)]-'a']--; 
		}
	}
	printf("%d\n",ans);
    return 0;
}
