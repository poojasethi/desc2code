#include <cstdio>
#include <cstring>
const int N=100100;
char s[N],p[N];
int t[26]={},n,m,now[27]={};
bool ok()
{
	for(int i=0;i<26;i++)
		if(now[i]>t[i])return false;
	return true;
}
int main()
{
	while(~scanf("%s%s",s,p))
	{
		if(strlen(s)<strlen(p)){puts("0");continue;}
		n=strlen(p);
		for(int i=0;i<n;i++)
			t[p[i]-'a']++;
		m=strlen(s);
		int ans=0;
		for(int i=0;i<n;i++)
			if(s[i]!='?')
				now[s[i]-'a']++;
		if(ok())ans++;
		for(int i=n;i<m;i++)
		{
			if(s[i-n]!='?')
				now[s[i-n]-'a']--;
			if(s[i]!='?')
				now[s[i]-'a']++;
			if(ok())ans++;
		}
		printf("%d\n",ans);
	}
}
