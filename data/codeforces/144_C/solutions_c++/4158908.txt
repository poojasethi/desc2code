#include<iostream>
#include<string>
long s3[200],p3[200],ans,x;
using namespace std;
int main()
{
    string s,p;
    long s1,p1,i,j;
    cin>>s>>p;
    s1=s.size();
    p1=p.size();
    if(p1>s1)
    {
        cout<<0<<endl;
        return 0;
    }
    for(i=0;i<p1;i++)
    p3[p[i]]++;
    for(i=0;i<p1-1;i++)
    s3[s[i]]++;
    for(i=p1-1;i<s1;i++)
    {
        x=0;
        s3[s[i]]++;
        for(j='a';j<='z';j++)
        {
            if(s3[j]<p3[j])
            x+=p3[j]-s3[j];
        }
        if(x<=s3['?'])
        ans++;
        s3[s[i-p1+1]]--;
    }
    cout<<ans<<endl;
}
