#include<iostream>
using namespace std;
int main()
{
string s,t;
int a[205]={0},b[205]={0},k=0,p,z=0,i,j;
cin>>s>>t;
if(t.length()>s.length())
{
cout<<0;
return 0;
}
for(i=0;i<t.length();i++)
{
b[t[i]]++;
if(s[i]!='?') a[s[i]]++;
else k++;
}
for(i=t.length();i<=s.length();i++)
{
p=k;
for(j='a';j<='z';j++)
{
p-=max(0,b[j]-a[j]);
}
if(p>=0)
{
z++;
}
if(s[i-t.length()]=='?') k--;
else a[s[i-t.length()]]--;
if(s[i]=='?') k++;
else a[s[i]]++;
}
cout<<z;
return 0;
}
