#include<stdio.h>
#include<stdlib.h>
int main(){
    long n;
    scanf("%d",&n);
    if (n<4){printf("NO");}
    else{
        printf(n&1?"YES\n5 * 4 = 20\n20 + 2 = 22\n22 + 3 = 25\n25 - 1 = 24\n":"YES\n1 * 2 = 2\n2 * 3 = 6\n6 * 4 = 24\n");
        int i;
        if (n&1)i=6; else i=5;
        while (i<n){
            printf("%d - %d = %d\n24 * 1 = 24\n",i+1,i,1);
            i+=2;
        }
    }
    return 0;
}
