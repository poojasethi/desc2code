#include <bits/stdc++.h>

using namespace std;

int main() {
int n;
scanf("%d", &n);
if(n < 4) {
puts("NO");
return 0;
}
puts("YES");
if(n&1) {
puts("5 * 4 = 20");
puts("3 - 1 = 2");
puts("2 * 2 = 4");
puts("20 + 4 = 24");
} else {
puts("1 * 2 = 2");
puts("2 * 3 = 6");
puts("6 * 4 = 24");
}
for(int i = (n&1) ? 7 : 6; i <= n; i += 2) {
printf("%d - %d = 1\n", i, i-1);
puts("24 * 1 = 24");
}
return 0;
}
