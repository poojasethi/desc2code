#include<cstdio>
int main()
{
	int a,i;
	scanf("%d",&a);
	if(a<4) printf("NO");
	else if(a%2==0){
		printf("YES\n");
		printf("1 * 2 = 2\n3 * 4 = 12\n2 * 12 = 24\n");
		for(i=5;i<=a;i+=2)
			printf("%d - %d = 1\n24 * 1 = 24\n",i+1,i);
	}else{
		printf("YES\n");
		printf("2 + 3 = 5\n4 * 5 = 20\n20 + 5 = 25\n25 - 1 = 24\n");
		for(i=6;i<=a;i+=2)
			printf("%d - %d = 1\n24 * 1 = 24\n",i+1,i);
	}
}
