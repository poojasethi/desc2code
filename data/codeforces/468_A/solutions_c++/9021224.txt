#include <cstdio>
void solve(int n) {
  if (n==4) printf("1 + 2 = 3\n3 + 3 = 6\n4 * 6 = 24\n");
  else if (n==5) printf("1 * 5 = 5\n5 - 2 = 3\n3 + 3 = 6\n4 * 6 = 24\n");
  else printf("%d - %d = 1\n1 * 1 = 1\n", n, n-1), solve(n-2);
}
int main(){ 
  int n;
  scanf("%d", &n);
  if (n<4) printf("NO\n");
  else printf("YES\n"), solve(n);
}
