#include<iostream>
using namespace std;
void fx( int n ){
	if( n==4 )	cout << "1 * 2 = 2\n2 * 3 = 6\n6 * 4  = 24";
	if( n==5 )	cout << "2 * 4 = 8\n3 * 5 = 15\n1 + 8 = 9\n9 + 15 = 24";
	if( n>5 ){	
		cout << n << " - " << n-1 << " = 1"<< endl << "1 * 1 = 1" << endl;
		fx( n-2 );
	}
}
int main(){
	int a;
	cin >> a; 
	if( a < 4 )	cout << "NO";
	else{
		cout << "YES" << endl;
		fx( a );
	}
	return 0;
}
