#include <cstdio>
using namespace std;

int n;

int main() {
	scanf("%d",&n);
	if (n<=3) {puts("NO");return 0;}
	puts("YES");
	if (n&1) {
		puts("3 + 4 = 7");
		puts("7 + 5 = 12");
		puts("12 * 2 = 24");
		puts("24 * 1 = 24");
	} else {
		puts("1 * 2 = 2");
		puts("2 * 3 = 6");
		puts("6 * 4 = 24");
	}
	for (int i=n;i>5;i-=2) {
		printf("%d - %d = 1\n",i,i-1);
		puts("24 * 1 = 24");
	}
	return 0;
}
