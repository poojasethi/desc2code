#include <iostream>

int main(){
  int n;std::cin>>n;
  if(n<4)return std::cout<<"NO\n",0;
  std::cout<<"YES\n";
  for(;n>5;n-=2)std::cout<<n<<" - "<<n-1<<" = 1\n1 * 1 = 1\n";
  if(n==4)std::cout<<"1 * 4 = 4\n2 * 3 = 6\n6 * 4 = 24\n";
  else std::cout<<"3 - 1 = 2\n2 + 2 = 4\n4 * 5 = 20\n20 + 4 = 24\n";
}
