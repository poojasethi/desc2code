#include<stdio.h>
int n,i;
int main()
{
	scanf("%i",&n);
	if( n<4 )
	{
		printf("NO\n");
	}
	else
	{
		printf( n&1 ? "YES\n5 * 4 = 20\n20 + 2 = 22\n22 + 3 = 25\n25 - 1 = 24\n" : "YES\n1 * 2 = 2\n2 * 3 = 6\n6 * 4 = 24\n");

		i = 5 + n%2;
		while( i<n )
		{
			printf("%i - %i = 1\n24 * 1 = 24\n",i+1,i);
			i += 2;
		}
	}
	return 0;
}
