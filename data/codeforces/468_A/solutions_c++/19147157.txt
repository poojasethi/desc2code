#include<cstdio>
int main(){
	int n;scanf("%d",&n);
	if(n<=3){printf("NO");return 0;}
	puts("YES");
	int cnt=0;
	while(n>5){
		printf("%d - %d = 1\n",n,n-1);
		n-=2;
		cnt++;
	}
	if(n==4){
		puts("1 * 2 = 2");
		puts("2 * 3 = 6");
		puts("6 * 4 = 24");
	}
	else{
		puts("5 + 4 = 9");
		puts("1 + 2 = 3");
		puts("9 * 3 = 27");
		puts("27 - 3 = 24");	
	}
	while(cnt--)puts("24 * 1 = 24");
	return 0;
}
