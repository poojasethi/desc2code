#include<cstdio>
int main()
{
	int n;scanf("%d",&n);
	if(n<4){printf("NO");return 0;}
	printf("YES\n");
	if(n&1)
	{
		printf("5 - 1 = 4\n");
		printf("4 - 2 = 2\n");
	}
	else printf("1 * 2 = 2\n");
	for(int i=n;i>5;i-=2)
	{
		printf("%d - %d = 1\n",i,i-1);
		printf("1 * 2 = 2\n");
	}
	printf("2 * 3 = 6\n");
	printf("6 * 4 = 24\n");
}
