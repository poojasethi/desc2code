void print(int n)
{
   if(n==4)
   cout<<"1 + 2 = 3\n3 + 3 = 6\n4 * 6 = 24\n";
   else if(n==5)
   cout<<"1 * 5 = 5\n5 - 2 = 3\n3 + 3 = 6\n4 * 6 = 24\n";
   else
   {
     cout<<n<<" - "<<(n-1)<<" = 1\n1 * 1 = 1\n";
     print(n-2);
   }
}
