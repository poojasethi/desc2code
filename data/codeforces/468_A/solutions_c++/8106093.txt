#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    if(n<4)printf("NO");
    else
    {
        if(n&1)printf("YES\n4 * 5 = 20\n20 + 2 = 22\n22 + 3 = 25\n25 - 1 = 24\n");
        else printf("YES\n1 * 2 = 2\n2 * 3 = 6\n6 * 4 = 24\n");
        for(int i=5+n%2;i<n;i+=2)
         printf("%d - %d = 1\n24 * 1 = 24\n",i+1,i);
    }
    return 0;
}
