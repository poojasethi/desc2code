#include <cstdio>

int n, i;

int main()
{
    scanf("%d", &n);

    if (n < 4)
        puts("NO");
    else
    {
        puts(n & 1 ? "YES\n4 * 5 = 20\n20 + 2 = 22\n22 + 3 = 25\n25 - 1 = 24" : "YES\n1 * 2 = 2\n2 * 3 = 6\n6 * 4 = 24");
        for (i = 5 + n % 2; i < n; i += 2)
            printf("%d - %d = 1\n24 * 1 = 24\n", i + 1, i);
    }

    return 0;
}
