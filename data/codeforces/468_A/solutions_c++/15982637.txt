#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    int n;
    cin>>n;
    if (n < 4)cout<<"NO"<<endl;
    else
    {
        printf("YES\n");
        if (n%2) printf("3 + 4 = 7\n7 + 5 = 12\n12 * 2 = 24\n24 * 1 = 24\n");
        else printf("1 * 2 = 2\n2 * 3 = 6\n6 * 4 = 24\n");
        for (int i = n; i > 5; i -= 2) printf("%d - %d = 1\n24 * 1 = 24\n",i,i-1);
    }
}
