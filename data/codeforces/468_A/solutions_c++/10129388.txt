#include<stdio.h>
int main()
{
	int n;
	scanf("%d",&n);
	if(n<4)
	printf("NO");
	else
	{
		printf("YES\n");
		if(n%2==0)
		{
			printf("1 * 2 = 2\n2 * 3 = 6\n6 * 4 = 24\n");
			for(;n>4;n-=2)
			printf("%d - %d = 1\n24 * 1 = 24\n",n,n-1);
		}
		else
		{
			printf("4 - 2 = 2\n2 + 1 = 3\n5 + 3 = 8\n3 * 8 = 24\n");
			for(;n>5;n-=2)
			printf("%d - %d = 1\n24 * 1 = 24\n",n,n-1);
		}
	}
}
