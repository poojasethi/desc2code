#include <bits/stdc++.h>

using namespace std;
int n;

int main(){
	cin >> n;
	if (n < 4){
		return cout << "NO" , 0;
	}
	cout << "YES\n";
	for (; n - 2 >= 4 ; n -= 2){
		cout << n << " - " << n-1 << " = 1\n1 * 1 = 1\n";
	}
	if (n == 4){
		cout << "1 * 2 = 2\n3 * 4 = 12\n12 * 2 = 24";
	}
	else {
		cout << "3 - 1 = 2\n2 * 2 = 4\n4 * 5 = 20\n 20 + 4 = 24";
	}
}
