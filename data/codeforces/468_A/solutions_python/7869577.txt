n = int(raw_input())

if n < 4:
	print "NO"
	exit(0)

print "YES"

mulToOne = 0
if n % 2 == 0:
	while n != 4:
		print n, "-", (n - 1), "= 1"
		n -= 2
		mulToOne += 1

	print "4 * 3 = 12"
	print "2 * 12 = 24"
	print "1 * 24 = 24"
else:
	while n != 5:
		print n, "-", (n - 1), "= 1"
		n -= 2
		mulToOne += 1

	print "4 * 5 = 20"
	print "3 - 1 = 2"
	print "2 * 2 = 4"
	print "20 + 4 = 24"

for i in xrange(0, mulToOne):
	print "24 * 1 = 24"
