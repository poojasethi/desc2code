def solve(n):
    if n < 4:
        print 'NO'
        return
    print 'YES'
    if n % 2 == 0:
        print '1 * 2 = 2'
        print '2 * 3 = 6'
        print '6 * 4 = 24'
    else:
        print '3 - 1 = 2'
        print '2 * 2 = 4'
        print '4 * 5 = 20'
        print '4 + 20 = 24'
    j = 5 if n % 2 == 0 else 6
    for i in range(j, n+1, 2):
        print '%d - %d = 1' % (i+1, i)
        print '24 * 1 = 24'

solve(int(raw_input()))
