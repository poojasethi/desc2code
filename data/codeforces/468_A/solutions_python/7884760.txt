n=int(raw_input())
if(n<4):
    print "NO"
    
elif(n==4):
    print "YES"
    print "4 * 3 = 12"
    print "12 * 1 = 12"
    print "12 * 2 = 24"
    
elif(n==5):
    print "YES"
    print "3 - 2 = 1"
    print "5 + 1 = 6"
    print "6 * 4 = 24"
    print "24 * 1 = 24"
    
else:
    print "YES"
    print str(n)+" - "+str(n-1)+" = "+"1"
    print "1 - 1 = 0"
    for i in range(5,n-1):
        print str(i)+" * 0 = "+"0"
    print "2 * 3 = 6"
    print "6 * 4 = 24"
    print "24 + 0 = 24"
