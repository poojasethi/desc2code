n = input()
if n < 4:
    print "NO"
else:
    print "YES"
    i = n
    while i >= 6:
        print '%s - %s = 1'%(i, i-1)
        print '%s * 1 = %s'%(i-2, i-2)
        i -= 2
    if i == 4:
        print '4 * 3 = 12'
        print '12 * 2 = 24'
        print '24 * 1 = 24'
    else:
        print '5 * 4 = 20'
        print '20 + 3 = 23'
        print '23 + 2 = 25'
        print '25 - 1 = 24'
