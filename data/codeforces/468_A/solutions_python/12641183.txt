n = int(raw_input())

if n < 4:
	print "NO"
	exit()

print "YES"

while n > 5:
	print n, "-", n-1, "= 1"
	print n-2, "* 1 =", n-2
	n -= 2

if n == 5:
	print "4 * 5 = 20"
	print "3 - 1 = 2"
	print "2 + 2 = 4"
	print "20 + 4 = 24"
elif n == 4:
	print "4 * 3 = 12"
	print "12 * 2 = 24"
	print "24 * 1 = 24"
