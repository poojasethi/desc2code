n = input()
if n < 4:
    print "NO"
else:
    print "YES"
    if n % 2 == 0:
        print "4 * 3 = 12\n12 * 2 = 24\n24 * 1 = 24" 
    elif n % 2 == 1:
        print "2 * 4 = 8\n3 * 5 = 15\n1 + 8 = 9\n9 + 15 = 24" 
    for x in xrange(n,4+(n%2),-2):
        print '%d - %d = 1' % (x,x-1)
        print '24 * 1 = 24'
