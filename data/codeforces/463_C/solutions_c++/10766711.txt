#include<bits/stdc++.h>

using namespace std;

#define y1 sadasf
#define int long long 

int n,i,j,mx,mx2,x1,y1,x2,y2,t;

int A[2005][2005];

int C1[4005],C2[4005];

main(){

	cin>>n;

	for(i=1 ; i<=n ; i++)
		for(j=1 ; j<=n ; j++){
			scanf("%d",&A[i][j]);
			C1[i+j]+=A[i][j];
			C2[i-j+n]+=A[i][j];
		}

	for(i=1 ; i<=n ; i++)
		for(j=1 ; j<=n ; j++){
			t=C1[i+j]+C2[i-j+n]-A[i][j];
			if((i+j)%2){
				if(mx<=t){
					mx=t;
					x1=i;
					y1=j;
				}
			}
			else 
				if(mx2<=t){
					mx2=t;
					x2=i;
					y2=j;
				}
		}

	cout<<mx+mx2<<endl<<x1<<' '<<y1<<' '<<x2<<' '<<y2<<endl;
	
}
