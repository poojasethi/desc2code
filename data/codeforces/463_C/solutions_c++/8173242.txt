#include<cstdio>

int n;
int a[2222][2222];
long long s1[5555],s2[5555];
long long ans1,ans2;
int ans1x,ans1y,ans2x,ans2y;

int main() {
	scanf("%d", &n);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++) {
			scanf("%d", &a[i][j]);
			s1[i+j] += a[i][j];
			s2[i-j +n] += a[i][j];
		}
		
	ans1=ans2=-1LL;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++){
			long long x = s1[i+j] + s2[i-j +n] - a[i][j];
			if(((i+j)&1) && ans1<x) {
				ans1=x;
				ans1x=i;
				ans1y=j;
			}
			if(!((i+j)&1) && ans2<x) {
				ans2=x;
				ans2x=i;
				ans2y=j;
			}
		}
		
	printf("%I64d\n", ans1+ans2);
	printf("%d %d %d %d\n", ans1x, ans1y, ans2x, ans2y);

	return 0;
}
