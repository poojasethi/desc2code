#include <cstdio>

using namespace std;

int n,x1,x2,y1,y2;
long long ans1=-1,ans2=-1,map[2001][2001],sum1[100001],sum2[100001];

int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
		for (int j=1;j<=n;j++) scanf("%I64d",&map[i][j]),sum1[i+j]+=map[i][j],sum2[i-j+n]+=map[i][j];
	for (int i=1;i<=n;i++)
		for (int j=1;j<=n;j++)
			if ((i+j)&1){
				if (sum1[i+j]+sum2[i-j+n]-map[i][j]>ans1) ans1=sum1[i+j]+sum2[i-j+n]-map[i][j],x1=i,y1=j;
			}
			else{
				if (sum1[i+j]+sum2[i-j+n]-map[i][j]>ans2) ans2=sum1[i+j]+sum2[i-j+n]-map[i][j],x2=i,y2=j;
			}
	printf("%I64d\n",ans1+ans2);
	printf("%d %d %d %d\n",x1,y1,x2,y2);
	return 0;
}
