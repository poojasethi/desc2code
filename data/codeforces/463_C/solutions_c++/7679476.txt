#include<iostream>
#include<cstdio>
using namespace std;
long long d[4000],e[4000];
int a[2000][2000];
main(){
  int n,i,j;
  scanf("%d",&n);
  for(i=0;i<n;i++)
    for(j=0;j<n;j++)
      scanf("%d",&a[i][j]),d[i+j]+=a[i][j],e[i-j+n]+=a[i][j];
  long long v,y=-1,z=-1;
  int p,q,r,s;
  for(i=0;i<n;i++)
    for(j=0;j<n;j++) {
      v=d[i+j]+e[i-j+n]-a[i][j];
      if ((i+j)&1) {
        if (v>y)
          y=v,p=i,q=j;
      } else {
        if (v>z)
          z=v,r=i,s=j;
      }
    }
  cout<<y+z<<endl<<p+1<<" "<<q+1<<" "<<r+1<<" "<<s+1<<endl;
}
