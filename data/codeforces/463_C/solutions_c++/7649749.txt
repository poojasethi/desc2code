#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2005;
long long N, A[MAXN][MAXN], up[2 * MAXN], dow[2 * MAXN], x[2], y[2], V[MAXN][MAXN];

int main(){
ios::sync_with_stdio(0);
cin.tie(0);
cin >> N;
for(int i = 0; i < N; i++) for(int j = 0; j < N; j++){
cin >> A[i][j];
up[i + j] += A[i][j];
dow[N + i - j] += A[i][j];
}
y[1] = 1;
for(int i = 0; i < N; i++) for(int j = 0; j < N; j++){
int p = (i + j) % 2;
V[i][j] = up[i + j] + dow[N + i - j] - A[i][j];
if(V[i][j] > V[x[p]][y[p]]){
x[p] = i, y[p] = j;
}
}
cout << V[x[0]][y[0]] + V[x[1]][y[1]] << '\n';
cout << x[0] + 1 << ' ' << y[0] + 1 << ' ' << x[1] + 1 << ' ' << y[1] + 1 << '\n';
}
