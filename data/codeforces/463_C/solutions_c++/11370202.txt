#include <iostream>
#include <cstdio>
using namespace std;

const int N=2002;

int a[N][N];
long long b[N+N], c[N+N];
int n;

int main()
{
	int i,j;
	long long m1=0, m2=0;
	int mx1=1, my1=1, mx2=1, my2=2;
	long long k;
	scanf("%d", &n);
	for(i=0;i<n;i++)
		for(j=0;j<n;j++){
			scanf("%d", &a[i][j]);
			b[i+j]=b[i+j]+a[i][j];
			c[i-j+n-1]=c[i-j+n-1]+a[i][j];
		}
	for(i=0;i<n;i++)
		for(j=0;j<n;j++){
			k = b[i+j]+c[i-j+n-1]-a[i][j];
			if((i+j)%2==0 && k>m1){
				m1=k;
				mx1=i+1;
				my1=j+1;
			}
			if((i+j)%2==1 && k>m2){
				m2=k;
				mx2=i+1;
				my2=j+1;
			}
		}
	cout<<m1+m2<<endl;
	cout<<mx1<<' '<<my1<<' '<<mx2<<' '<<my2<<endl;
	return 0;
}
