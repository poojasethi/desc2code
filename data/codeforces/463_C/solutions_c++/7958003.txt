#include<cstdio>
using namespace std;
#define ll long long
int a1[2000][2000];
ll a2[4000],a3[4000],a,o[2];
int main()
{
    int n,ox[2],oy[2];
    scanf("%d",&n);
    for(int i=0;i<n;++i)
    for(int j=0;j<n;++j)
    {
        scanf("%d",a1[i]+j);
        a2[i+j]+=a1[i][j];
        a3[i-j+n-1]+=a1[i][j];
    }
    for(int ii=0;ii<2;++ii)
    for(int i=0;i<n;++i)
    for(int j=ii^(i&1);j<n;j+=2)
    {
        a=a2[i+j]+a3[i-j+n-1]-a1[i][j];
        if(a>=o[ii]){o[ii]=a;ox[ii]=i;oy[ii]=j;}
    }
    printf("%I64d\n%d %d %d %d\n",o[0]+o[1],ox[0]+1,oy[0]+1,ox[1]+1,oy[1]+1);
}
