#include<iostream>
#include<cstdio>
#include<stdio.h>
using namespace std;
int a[2001][2001];
int main()
{
long long  t,n,i,j,k,qk,slope1[4001]={},slope2[4001]={},max[2]={-1},x[2],y[2];
cin >> n;
for(i=0;i<n;i++)
for(j=0;j<n;j++)
scanf("%d",&a[i][j]),slope1[i+j]+=a[i][j],slope2[n-1-j+i]+=a[i][j];
for(i=0;i<n;i++)
{
for(j=0;j<n;j++)
{

qk=slope1[i+j]+slope2[n-1-j+i]-a[i][j];
if(max[(i+j)%2]<=qk)
max[(i+j)%2]=qk,x[(i+j)%2]=i,y[(i+j)%2]=j;
}
}

cout<<max[0]+max[1]<<"\n";
cout<<x[0]+1<<" ";
cout<<y[0]+1<<" ";
cout<<x[1]+1<<" ";
cout<<y[1]+1<<"\n";
return 0;
}
