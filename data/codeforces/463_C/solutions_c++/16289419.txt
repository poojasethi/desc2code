#include <bits/stdc++.h>
#define ll long long int
using namespace std;
ll a[2016][2016],n,x1,x2,yy,y2,s[4032],d[4032],mx1=-1,mx2=-1;
int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			scanf("%I64d",&a[i][j]);
			s[i+j]+=a[i][j];
			d[i-j+n]+=a[i][j];
		}
	}
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			if((i+j)%2){
				if(s[i+j]+d[i-j+n]-a[i][j]>mx1){
					mx1=s[i+j]+d[i-j+n]-a[i][j];
					x1=i;
					yy=j;
				}
			}
			else{
				if(s[i+j]+d[i-j+n]-a[i][j]>mx2){
					mx2=s[i+j]+d[i-j+n]-a[i][j];
					x2=i;
					y2=j;
				}
			}
		}
	}
	cout<<mx1+mx2<<endl<<++x1<<" "<<++yy<<" "<<++x2<<" "<<++y2<<endl;
}
