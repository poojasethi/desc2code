#include<iostream>
#include<cstdio>
using namespace std;
long long n,i,j,l1,l2,r1,r2;
long long a[2005][2005],s1[5000],s2[5000],mx1,mx2;
int main()
{
    scanf("%d",&n);
    for (i=1;i<=n;i++)
     for (j=1;j<=n;j++)
    {
        scanf("%d",&a[i][j]);
        s1[i+j]+=a[i][j];
        s2[n-j+i]+=a[i][j];
    }
    l1=1;r1=1;l2=1;r2=2;
    for (i=1;i<=n;i++)
     for (j=1;j<=n;j++)
       if ((i+j)% 2)
    {
        if (s1[i+j]+s2[n-j+i]-a[i][j]>mx1){mx1=s1[i+j]+s2[n-j+i]-a[i][j];l1=i;r1=j;}
    } else
    {
        if (s1[i+j]+s2[n-j+i]-a[i][j]>mx2){mx2=s1[i+j]+s2[n-j+i]-a[i][j];l2=i;r2=j;}
    }
    cout<<mx1+mx2<<endl<<l1<<" "<<r1<<" "<<l2<<" "<<r2;
}
