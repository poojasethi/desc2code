#include <iostream>
#include <cstdio>
using namespace std;
long long x[4001], y[4001], m0=-1, m1=-1;
int n, a[2001][2001], i, j, x0, y0, x1, y1;
int main()
{
	scanf("%d", &n);
	for (i=1; i<=n; i++) for (j=1; j<=n; j++) {
		scanf("%d", &a[i][j]);
		x[i+j]+=a[i][j];
		y[1+n-i+j]+=a[i][j];
	}
	for (i=1; i<=n; i++) for (j=1; j<=n; j++) {
		if ((i+j)%2==0) {
			if (m0<x[i+j]+y[1+n-i+j]-a[i][j]) {
				x0=i; y0=j;
				m0=x[i+j]+y[1+n-i+j]-a[i][j];
			}
		}
		else {
			if (m1<x[i+j]+y[1+n-i+j]-a[i][j]) {
				x1=i; y1=j;
				m1=x[i+j]+y[1+n-i+j]-a[i][j];
			}
		}
	}
	printf("%I64d\n%d %d %d %d\n", m0+m1, x0, y0, x1, y1);
	return 0;
}
