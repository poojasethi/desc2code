#include <cstdio>
#include <algorithm>
using namespace std;
long long b,c,d,e,f[5005],g[5005],h[5],i,j,k,l,m,n;
long long a[2005][2005],x1,y1,x2,y2;
int main(){
	scanf("%I64d",&n);
	for(i=1;i<=n;i++)
	for(j=1;j<=n;j++){
		scanf("%I64d",&a[i][j]);
		f[i+j]+=a[i][j];
		g[j-i+n]+=a[i][j];
	}
	h[0]=h[1]=-1; 
	for(i=1;i<=n;i++)
	for(j=1;j<=n;j++){
		l=f[i+j]+g[j-i+n]-a[i][j];
		if ((i+j)&1){
			if (h[1]<l){h[1]=l;x1=i;y1=j;}
		}
		else{
			if (h[0]<l){h[0]=l;x2=i;y2=j;}
		}
	}
	printf("%I64d\n",h[0]+h[1]);
	printf("%I64d %I64d %I64d %I64d",x1,y1,x2,y2);
	return 0;
}
