#include <iostream>
#define f(i,a,b) for(int i = a; i < b; i++)
typedef long long ll;

using namespace std;

int n,A[2000][2000],x[2],y[2];
ll a[4000],b[4000],ans[2]={-1,-1},s;

int main(){
    ios_base::sync_with_stdio(0);
    cin >> n;
    f(i,0,n)f(j,0,n){cin >> A[i][j]; a[i-j+n] += A[i][j];b[i+j] += A[i][j];}
    f(i,0,n)f(j,0,n){
        s = a[i-j+n]+b[i+j]-A[i][j];
        if(s>ans[(i+j)%2]){
            x[(i+j)%2] = i+1; y[(i+j)%2] = j+1;
            ans[(i+j)%2] = s;
        }
    }
    cout << ans[0] + ans[1] << "\n";
    cout << x[0] << " " << y[0] << " " << x[1] << " " << y[1] << "\n";
}
