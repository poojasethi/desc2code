typedef long long ll;
int a[2009][2015];
ll d1[4009],d2[4009];
int x=1,xx=1,y=1,yy=2;
int main()
{
    ll mx1=0,mx2=0;
    int n;
    cin>>n;
    f(i,n) f(z,n){
         scanf("%d",&a[i][z]);
    d1[i+z]+=a[i][z];
    d2[i-z+n]+=a[i][z];
    }

    ll d;
     f(i,n) f(z,n){
       if((i+z)%2){
        d=d1[i+z]+d2[i-z+n]-a[i][z];
        if(d>mx1){mx1=d; x=i; y=z;}
       }

       }

    }

}
