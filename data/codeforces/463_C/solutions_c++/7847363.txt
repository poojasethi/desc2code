#include <cstdio>

int t[4000000], x, y, i, n, f = 1, g = 0;
long long a[4000] = {0}, b[4000] = {0}, s, u = 0, v = 0;

int main() {
    scanf("%d", &n);
    for(i = y = 0; y < n; y++) for(x = 0; x < n; x++, i++) {
        scanf("%d", &t[i]);
        a[n + x - y] += t[i];
        b[x + y] += t[i];
    }
    for(i = y = 0; y < n; y++) for(x = 0; x < n; x++, i++) {
        s = a[n + x - y] + b[x + y] - t[i];
        if((x + y) & 1) { if(s > u) u = s, f = i; }
        else if(s > v) v = s, g = i;
    }
    printf("%I64d\n", u + v);
    printf("%d %d %d %d\n", f / n + 1, f % n + 1, g / n + 1, g % n + 1);
}
