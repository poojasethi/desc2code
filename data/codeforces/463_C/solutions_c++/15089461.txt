#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

ll d1[4222] , d2[4222] , s[2] , x , a[2222][2222];
pair< int , int > p[2];
int n  ;

int main()
{
scanf("%d",&n);
for(int i=1;i<=n;i++)
for(int j=1;j<=n;j++){
    scanf("%l64d",&a[i][j]);
    d1[i+j]+=a[i][j];
    d2[i-j+n]+=a[i][j];
}
s[0]=-1LL;
s[1]=-1LL;

for(int i=1;i<=n;i++)
for(int j=1;j<=n;j++){

if( d1[i+j]+d2[i-j+n]-a[i][j] > s[(i+j)&1] ){
    s[(i+j)&1]=d1[i+j]+d2[i-j+n]-a[i][j];
    p[(i+j)&1].first=i;
    p[(i+j)&1].second=j;

}


}

cout<<s[0]+s[1]<<endl;
cout<<p[0].first<<" "<<p[0].second<<" "<<p[1].first<<" "<<p[1].second<<endl;

    return 0;
}
