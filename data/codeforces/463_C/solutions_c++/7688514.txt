#include<cstdio>
#define nn 2010
using namespace std;
typedef long long ll;
struct ok{ll s;int x,y;}ans[2];
int a[nn][nn],i,j,n;
ll z,l[nn*2],r[nn*2];
int main()
{
 scanf("%d",&n);
 for (i=1;i<=n;i++)
 for (j=1;j<=n;j++)
       {scanf("%d",&a[i][j]);
        l[i-j+n]+=a[i][j];
        r[i+j]+=a[i][j]; 
       }
 ans[0].s=ans[1].s=-1;
 for (i=1;i<=n;i++)
  for (j=1;j<=n;j++){
  z=l[i-j+n]+r[i+j]-a[i][j];
  if (ans[(i+j)&1].s<z)
    {ans[(i+j)&1].s=z;
     ans[(i+j)&1].x=i;
     ans[(i+j)&1].y=j;
    }
                    }
  printf("%I64d\n",ans[0].s+ans[1].s);
  printf("%d %d %d %d\n",ans[0].x,ans[0].y,ans[1].x,
  ans[1].y);
  return 0;
}
