#include <stdio.h>
#define nmax 2010
#define inf 0x3f3f3f3f
using namespace std;
int i,j,t[nmax][nmax],n,x1,y1,x2,y2;
long long p[nmax*2],s[nmax*2],val,maxx,maxy;
int main() {
scanf("%d",&n);
for (int i=1;i<=n;i++)
    for (int j=1;j<=n;j++) {
        scanf("%d",&t[i][j]);
        p[i+j]=p[i+j]+t[i][j];
        s[i-j+n]=s[i-j+n]+t[i][j];
}
maxx=-inf; maxy=-inf;
for (int i=1;i<=n;i++)
    for (int j=1;j<=n;j++) {
        val=p[i+j]+s[i-j+n]-t[i][j];
        if ((i+j)%2==0 && val>maxx) { maxx=val; x1=i; y1=j; } else
           if ((i+j)%2==1 && val>maxy) { maxy=val; x2=i; y2=j; }
}
printf("%I64d\n%d %d %d %d",maxx+maxy,x1,y1,x2,y2);
return 0;
}
