#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i,a,b) for (int i=a;i<b;i++)
int i,j,n,c[2004][2004],x[2],y[2];
ll ret[2],a[4003],b[4004];
main()
{
    scanf("%d",&n);
    rep(i,0,n) rep(j,0,n)
    {
        scanf("%d",&c[i][j]);
        a[i+j]+=c[i][j];
        b[i-j+n]+=c[i][j];
    }
    rep(i,0,n) rep(j,0,n)
    if (a[i+j]+b[i-j+n]-c[i][j]>=ret[(i+j)&1])
    {
        int v=(i+j)&1;
        ret[v]=a[i+j]+b[i-j+n]-c[i][j];
        x[v]=i+1,y[v]=j+1;
    }
    printf("%I64d\n",ret[0]+ret[1]);
    printf("%d %d %d %d\n",x[0],y[0],x[1],y[1]);
}
