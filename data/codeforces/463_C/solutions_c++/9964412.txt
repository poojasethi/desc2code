#include <iostream>
#define f(i,a,b) for(int i = a; i < b; i++)
typedef long long ll;

using namespace std;

int n,A[2000][2000],x[2],y[2],parity;
ll a[4000],b[4000],ans[2]={-1,-1},s;

int main(){
    ios_base::sync_with_stdio(0);
    cin >> n;
    f(i,0,n)f(j,0,n){cin >> A[i][j]; a[i-j+n] += A[i][j];b[i+j] += A[i][j];}
    f(i,0,n)f(j,0,n){
        int parity = (i+j)%2;
        s = a[i-j+n]+b[i+j]-A[i][j];
        if (s > ans[parity]){ans[parity] = s; x[parity] = i+1; y[parity]=j+1;}
    }
    cout << ans[0] + ans[1] << "\n";
    cout << x[0] << " " << y[0] << " " << x[1] << " " << y[1] << "\n";
}
