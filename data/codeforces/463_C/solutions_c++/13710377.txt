#include<bits/stdc++.h>
using namespace std;
typedef long long ll ;
long long b[4010],c[4010];
int a[2005][2005],n;
int main(void) {
  int i,j;
	long long m1=0, m2=0;
	int mx1=1, my1=1, mx2=1, my2=2;
	long long k;
	scanf("%d", &n);
	for(i=0;i<n;i++)
		for(j=0;j<n;j++){
			scanf("%d", &a[i][j]);
			b[i+j]=b[i+j]+a[i][j];
			c[i-j+n]=c[i-j+n]+a[i][j];
		}
	for(i=0;i<n;i++)
		for(j=0;j<n;j++){
			k = b[i+j]+c[i-j+n]-a[i][j];
			if((i+j)%2==0 && k>m1){
				m1=k;
				mx1=i+1;
				my1=j+1;
			}
			if((i+j)%2==1 && k>m2){
				m2=k;
				mx2=i+1;
				my2=j+1;
			}
		}
	cout<<m1+m2<<endl;
	cout<<mx1<<" " <<my1<<" " <<mx2<<" " <<my2<<endl;
  return 0;
}
