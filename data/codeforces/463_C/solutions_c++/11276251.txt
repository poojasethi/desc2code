#include<stdio.h>
long long n,i,j,a,arr[3000][3000],G[2005],H[10005],max1,max2,max1x=1,max1y=1,max2x=1,max2y=2,temp;
int main(){
	scanf("%lld",&n);
	for(i=1;i<=n;i++)
		
		for(j=1;j<=n;j++){
			scanf("%lld",&a);
			H[i-j+3000]+=a;
			G[i+j]+=a;
			arr[i][j]=a;
		}
	
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++){
			temp = H[i-j+3000] + G[i+j] - arr[i][j];

			if((i+j)%2==0){
				if(temp > max1)
					max1=temp,max1x=i,max1y=j;
			}
			else{
				if(temp > max2)
					max2=temp,max2x=i,max2y=j;
			}
		}
	printf("%lld\n%lld %lld %lld %lld",max1+max2, max1x,max1y,max2x,max2y);

return 0;
}
