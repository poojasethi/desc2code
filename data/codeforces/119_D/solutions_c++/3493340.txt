#include<cstdio>
#include<cstring>
#include<algorithm>
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=2010000;
int n,m;
int pre[mn],z[mn];
char a[mn],b[mn],s1[mn],s2[mn];
int main()
{
    gets(a+1);
    gets(b+1);
    int i,j;
    n=strlen(a+1),m=strlen(b+1);
    if(n!=m)
    {
        puts("-1 -1");
        return 0;
    }
    fo(i,1,n)s1[i]=a[n-i+1];
    s1[n+1]=0;
    fo(i,1,n)s1[n+i+1]=b[i];
    fo(i,1,n)s2[i]=b[i];
    s2[n+1]=0;
    fo(i,1,n)s2[n+i+1]=a[i];
    pre[1]=j=0;
    fo(i,2,n+n+1)
    {
        for(;j&&s1[j+1]!=s1[i];j=pre[j]);
        if(s1[j+1]==s1[i])++j;
        pre[i]=j;
    }
    int l=0,r=0;
    z[1]=0;
    fo(i,2,n+n+1)
    {
        z[i]=0;
        if(i<=r)z[i]=min(z[i-l+1],r-i+1);
        while(i+z[i]<=n+n+1&&s2[z[i]+1]==s2[i+z[i]])++z[i];
        if(i+z[i]-1>r)
            l=i,r=i+z[i]-1;
    }
    int ans_i=-1,ans_j=-1;
    fo(i,1,n-1)
    {
        if(a[i]!=b[n-i+1])break;
        int len=pre[n+n+1-i];
        if(z[n+i+2]>=n-i-len)
            ans_i=i,ans_j=n-len;
    }
    if(ans_i==-1)puts("-1 -1");
    else printf("%d %d\n",ans_i-1,ans_j);
    return 0;
}
