#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 2000001
#define mod 1000000007
#define ll long long

ll n,i,j,k,i2,j2;
char x[lim],y[lim],s[lim];
ll P[lim],H1[lim],H2[lim];

int main()
{
  gets(x);
  gets(y);
  n=strlen(x);
  if(strlen(y)!=n)
  {
    printf("-1 -1");
    return 0;
  }
  for(i=0;i<n;i++)
  {
    s[i]=x[n-1-i];
	s[n+1+i]=y[i];
  }
  k=P[0]=-1;
  for(i=1;i<2*n+1;i++)
  {
    while(k>=0 && s[k+1]!=s[i])
	  k=P[k];
	if(s[k+1]==s[i])
	  k++;
    P[i]=k;
  }
  i2=j2=-1;
  for(i=0,P[0]=1;i<n;i++)
  {
    H1[i]=(H1[i-1]+P[i]*(x[i]-31))%mod;
    H2[i]=(H2[i-1]+P[i]*(y[i]-31))%mod;
	P[i+1]=P[i]*(128-32)%mod;
  }
  for(i=0;i+1<n && x[i]==y[n-1-i];i++)
  {
    j=n-(P[2*n-1-i]+1);
    if(j==n)
	  continue;
	if(H2[j-i-2]*P[i+1]%mod==(H1[j-1]-H1[i]+mod)%mod)
	{
	  i2=i;
	  j2=j;
	}
  }
  printf("%I64d %I64d",i2,j2);
  return 0;
}
