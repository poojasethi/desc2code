#include<iostream>
#include<cstring>
#include<cstdio>
#define MOD 9876321
using namespace std;
int n,rr[1000010],ll[1000010];
int ha[1000010],hb[1000010],hc[1000010],pp[1000010];
char a[1000010],b[1000010];
int main()
{
	//freopen("input.txt","rb",stdin);
	//freopen("output.txt","wb",stdout);
	gets(a);
	gets(b);
	n=strlen(a);
	if(n!=strlen(b))
	{
		printf("-1 -1\n");
		return 0;
	}
	if(a[n-1]=='\r')n--;
	ha[0]=0;
	hc[0]=0;
	hb[0]=0;
	pp[0]=1;
	for(int i=0;i<n;++i)
	{
		pp[i+1]=pp[i]*37;
		ha[i+1]=ha[i]*37+a[i];
		hb[i+1]=hb[i]*37+b[i];
		hc[i+1]=hc[i]*37+a[n-i-1];
	}
	for(int i=0;i<n-1;++i)
	{
		int l=0,r=n-i,m=(l+r)>>1;
		for(;l<=r;m=(l+r)>>1)
		if(hb[i+m]-hb[i]*pp[m]==hc[m])l=m+1;else r=m-1;
		rr[i]=r;
	}
	//for(int i=0;i<n;++i)cout<<rr[i]<<endl;
	for(int i=0,j=0;i<n;++i)for(;j<i+rr[i];++j)ll[j]=i;
	int ap=-1,aq=-1;
	for(int i=0;i<n-1;++i)
		if(a[i]!=b[n-i-1])break;else
		if(ll[n-i-2]+1+i>n)continue;else
		if(hb[ll[n-i-2]]==ha[i+ll[n-i-2]+1]-ha[i+1]*pp[ll[n-i-2]])ap=i,aq=i+ll[n-i-2]+1;
	printf("%d %d\n",ap,aq);
	return 0;
}
