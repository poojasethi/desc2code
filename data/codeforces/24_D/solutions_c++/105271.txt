#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
const double eps=1e-6;
int n,m;
int sx,sy;
double e[1010][1010];
int main()
{
	//freopen("D.in","r",stdin);
	scanf("%d%d",&n,&m);
	scanf("%d%d",&sx,&sy);
	memset(e,0,sizeof(e));
	double sum;
	int div;
	bool flag;
	for(int i=n-1;i>=sx;--i)
	{
		while(true)
		{						
			flag=true;
			for(int j=1;j<=m;++j)
			{
				sum=e[i][j]+e[i+1][j]+2;
				div=2;
				if(j!=1) sum+=e[i][j-1]+1,div++;
				if(j!=m) sum+=e[i][j+1]+1,div++;
				if(abs(sum/div-e[i][j])>eps) flag=false;
				e[i][j]=sum/div;
			}
			if(flag) break;
		}		
	}
	printf("%.10lf\n",e[sx][sy]);
	return 0;
}
