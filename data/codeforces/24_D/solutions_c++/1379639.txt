#include<cstdio>
#include<cstring>
#define t 1.0/3.0
const int N = 1005;
double e[N][N];
double A[N],B[N];
int main()
{
	int n,m,i,j,x,y;
	scanf("%d%d%d%d",&m,&n,&x,&y);
	if(n==1)
	{
		printf("%d\n",2*(m-x));
		return 0;
	}
	for(i=m-1;i>=x;i--)
	{
		A[1]=1.0+0.5*(e[i+1][1]+1);
		B[1]=0.5;
		for(j=2;j<n;j++)
		{
			A[j]=0.25*(e[i+1][j]+A[j-1]+4)/(1.0-0.25*B[j-1]-0.25);
			B[j]=1.0/(4.0-B[j-1]-1);
		}
		e[i][n]=(A[n-1]*t+1.0+t*e[i+1][n])/(1.0-t*B[n-1]-t);
		for(j=n-1;j>0;j--)
		{
			e[i][j]=B[j]*e[i][j+1]+A[j];
		}
	}
	printf("%.6lf\n",e[x][y]);
	return 0;
}
