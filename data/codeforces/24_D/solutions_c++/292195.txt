#include <cstdio>
#include <cstring>

const int maxn = 1005;
const double eps = 1e-8;

double ans[3][maxn];
int n , m , x , y;

double abs( double a ) {  return a>0?a:-a;  }

int main() {
    scanf( "%d%d%d%d" , &n , &m , &x , &y );
    for ( int i = x; i < n; i++ ) {
        for ( int t = 0; t < 100; t++ ) {
            for ( int j = 2; j < m; j++ ) {
                ans[2][j] = ( ans[1][j-1] + ans[1][j+1] + ans[1][j] + ans[0][j] ) / 4.0 + 1.0;
            }
            ans[2][1] = ( ans[1][2] + ans[0][1] + ans[1][1] ) / 3.0 + 1.0;
            ans[2][m] = ( ans[1][m-1] + ans[0][m] + ans[1][m] ) / 3.0 + 1.0;
            memcpy( ans[1] , ans[2] , sizeof(ans[2]) );
        }
        memcpy( ans[0] , ans[2] , sizeof(ans[2]) );
    }
    if ( m == 1 ) printf( "%.10lf\n" , 2.0*(n-x) );
    else printf( "%.10lf\n" , ans[2][y] );
    return 0;
}
