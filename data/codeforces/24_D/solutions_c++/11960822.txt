#include<bits/stdc++.h>
using namespace std;
const int maxn = 1010;
double A[maxn],B[maxn],C[maxn];
int n,m,x,y;
int main()
{
	scanf("%d%d%d%d",&n,&m,&x,&y);
	if(m==1)
	{
		printf("%.10f\n",(n-x)*2.0);
		return 0;
	}
	memset(A,0,sizeof(A));
	for(int i=x;i<n;i++)
	{
		memset(B,0,sizeof(B));
		for(int T=0;T<100;T++)
		{
			for(int j=1;j<=m;j++)
			{
				if(j==1)
					C[j]=(B[j]+B[j+1]+A[j])/3+1;
				if(j==m)
					C[j]=(B[j]+B[j-1]+A[j])/3+1;
				if(1<j&&j<m)
					C[j]=(B[j-1]+B[j]+B[j+1]+A[j])/4+1;
			}
			memcpy(B,C,sizeof(C));
		}
		memcpy(A,B,sizeof(B));
	}
	printf("%.10f\n",A[y]);
	return 0;
}
