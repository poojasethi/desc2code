typedef vector<int> vi;

const int iter=100;
int n,m,y,x;
double e[1000][1000];

void run(){
  cin>>n>>m>>y>>x;
  y--; x--;
  if(y==n-1){
    cout<<0<<endl;
    return;
  }
  for(int i=n-2;i>=y;i--){
    rep(_,iter)rep(j,m){
      int cnt=2;
      double s=e[i+1][j]+e[i][j];
      if(j>0)s+=e[i][j-1], cnt++;
      if(j<m-1)s+=e[i][j+1], cnt++;
      e[i][j]=1+s/cnt;
      //cerr<<e[i][j]<<(j==m-1?"\n":" ");
    }
  }
  printf("%.9f\n",e[y][x]);
}
int main(){
#ifndef ONLINE_JUDGE

#endif
    run();
  return 0;
}
