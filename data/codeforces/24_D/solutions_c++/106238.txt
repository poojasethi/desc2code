#include <iostream>
#include <cstdio>

using namespace std;

double d[1005][1005],a[1005],b[1005];
int n,m,x,y;

int main()
{
    cin >> n >> m >> x >> y;
//    n = 10;m = 10; x= 10; y = 4;
    if (m == 1) printf("%.10f\n", 2.0 * (n - x)); else {
    for (int i = n-1; i >= x; i--)
    {
        //tinh cac he so tuyen tinh
        a[0] = b[0] = 0;
        for (int j = 1; j <= m; j++)
        {
            double p, k;
            if (j == 1 || j == m) p = 1.0 / 3; else p = 1.0 / 4;
            k = 1 - p * (a[j - 1] + 1);
            a[j] = p / k;
            b[j] = (1 + p * (b[j-1] + d[i+1][j])) / k;
        }
        d[i][m + 1] = 0;
        for (int j = m; j > 0; j--) d[i][j] = d[i][j + 1] * a[j] + b[j];
    }
        printf("%0.10f\n",d[x][y]);
}
    return 0;
}
