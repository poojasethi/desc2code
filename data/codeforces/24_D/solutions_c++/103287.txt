#include <iostream>
#include <iomanip>
#include <vector>
#include <map>
#include <set>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
  int n, m; cin>>n>>m;
  int pi, pj; cin>>pi>>pj;
  pi--, pj--;

  vector<vector<double> > es(n, vector<double>(m, 0));

  for (int i=n-2; i>=pi; i--){
    for (int turn=0; ; turn++){
      bool cont=false;
      for (int j=0; j<m; j++){
	const static int vect[]={0,1,1,0,0,-1};

	double posn=0;
	double sum=0;

	for (int v=0; v<3; v++){
	  int ni=i+vect[v*2+0];
	  int nj=j+vect[v*2+1];
	  if (nj<0) continue;
	  if (nj>=m) continue;
	  posn++;
	  sum+=es[ni][nj];
	}
	double nes=(posn+1+sum)/posn;
	if (abs(es[i][j]-nes)>1e-6) cont=true;
	es[i][j]=nes;
      }
      if (!cont){
	//cerr<<"* "<<turn<<endl;
	break;
      }
    }
  }

  cout<<setiosflags(ios::fixed)<<setprecision(12)<<es[pi][pj]<<endl;

  return 0;
}
