#include <stdio.h>

#define MAX 1000

int n, m, x, y, i, j;
double p[MAX][MAX], a[MAX];

int main()
{
	scanf( "%d%d%d%d", &n, &m, &x, &y );
	x--, y--;

	for( i = n-2; i >= 0; i-- )
	{
		for( j = 0; j < m; j++ ) a[j] = (j>0?1:0) + (j<m-1?1:0) + 1, p[i][j] = 2 + p[i+1][j] + (j>0?1:0) + (j<m-1?1:0);
		for( j = 0; j < m-1; j++ ) a[j+1] += -1/a[j], p[i][j+1] += p[i][j]/a[j];
		for( j = m-1; j > 0; j-- ) p[i][j-1] += p[i][j]/a[j];
		for( j = 0; j < m; j++ ) p[i][j] /= a[j];
	}

	printf( "%0.10f", p[x][y] );
	return 0;
}
