#include <bits/stdc++.h>
#define maxn 1009
using namespace std;
long double E[maxn][maxn],a[maxn],b[maxn],c[maxn],d[maxn],cc[maxn],dd[maxn];
int n,m,x,y;
int main()
{
	cin>>n>>m>>x>>y;
	if(m==1)
	{
		double ans=0;
		for(int i=0;i<n-x;i++)
		{
			ans=(0.5*ans+1)*2;
		}
		printf("%.12f\n",ans);
		return 0;
	}
	for(int i=n-1;i>=1;i--)
	{
		a[1]=0;b[1]=2;c[1]=-1;d[1]=E[i+1][1]+3;
		a[m]=-1;b[m]=2;c[m]=0;d[m]=E[i+1][m]+3;
		for(int j=2;j<m;j++)
		{
			a[j]=-1;b[j]=3;c[j]=-1;d[j]=E[i+1][j]+4;
		}
		cc[1]=c[1]/b[1];
		for(int j=2;j<m;j++)
			cc[j]=c[j]/(b[j]-a[j]*cc[j-1]);
		dd[1]=d[1]/b[1];
		for(int j=2;j<=m;j++)
			dd[j]=(d[j]-a[j]*dd[j-1])/(b[j]-a[j]*cc[j-1]);
		E[i][m]=dd[m];
		for(int j=m-1;j>=1;j--)
			E[i][j]=dd[j]-cc[j]*E[i][j+1];
	}
	printf("%.12f\n",(double)E[x][y]);
	//system("pause");
	return 0;
}
