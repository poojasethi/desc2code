#include <cstdio>
using namespace std;

double p[1010][1010];

int main(){
        int n,m,x,y,i,j,k;
        scanf("%d%d%d%d",&n,&m,&x,&y);
        if (m==1){
                printf("%.6lf",2.0*(n-x));
                return 0;
        }
        for(i=n-1;i>=x;i--){
                for(k=100;k;k--)
                for(j=1;j<=m;j++)
                if (j==1) p[i][j]=(p[i][j]+p[i][j+1]+p[i+1][j])/3+1;
                else if(j==m) p[i][j]=(p[i][j]+p[i][j-1]+p[i+1][j])/3+1;
                else p[i][j]=(p[i][j]+p[i][j+1]+p[i][j-1]+p[i+1][j])/4+1;
        }
        printf("%.6lf",p[x][y]);
}
