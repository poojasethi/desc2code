#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
const int N=1005;
int m,n;
int sx,sy;
double f[N][N];

int main(){
    cin >> m >> n;
    cin >> sx >> sy;
    memset(f,0,sizeof(f));
    for (int i = m - 1; i >= sx; i--){
        for (int k = 0; k < 60; k++){
            for (int j = 1; j <= n; j++){
                int cnt = 4;
                if (j == 1){
                    cnt--;
                } else {
                    f[i][j] += f[i][j - 1] + 1;
                }
                if (j == n){
                    cnt--;
                } else {
                    f[i][j] += f[i][j + 1] + 1;
                }
                f[i][j] += f[i + 1][j] + 2;
                f[i][j] /= cnt;
            }
        }
    }
    printf("%.5lf\n",f[sx][sy]);
}
