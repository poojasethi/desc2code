#include <bits/stdc++.h>
using namespace std;
#define N 1010

int n, m, a, b, d[N];
double f[N][N];

int main() {
//	freopen("d.in", "r", stdin);
	scanf("%d %d %d %d", &n, &m, &a, &b);
	for (int i = 1; i <= m; i ++) d[i] = 4;
	d[1] = d[m] = 3; if (m == 1) d[1] = 2;
	for (int i = n - 1; i >= a; i --) {
		for (int step = 0; step < 50; step ++) {
			for (int j = 1; j <= m; j ++) {
				f[i][j] = (f[i][j] + f[i][j-1] + f[i][j+1] + f[i+1][j]) / d[j] + 1;
			}
		}
	}
	printf("%lf\n", f[a][b]);
	return 0;
}
