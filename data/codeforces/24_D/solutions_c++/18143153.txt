#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#include<string>
using namespace std;
int n,m,a,b;  
double dp[1005][1005],d[1005];  
int main()  
{  
    while(~scanf("%d%d%d%d",&n,&m,&a,&b))  
    {  
        for(int i=1; i<=m; i++)  
        {  
            d[i]=2;  
            if(i>1)d[i]++;  
            if(i<m)d[i]++;  
        }  
        memset(dp,0,sizeof(dp));  
        for(int i=n-1; i>=a; i--)  
        {  
            for(int t=0; t<50; t++)  
            {  
                for(int j=1; j<=m; j++)  
                {  
					dp[i][j]=(1+(dp[i][j-1]+dp[i+1][j]+dp[i][j+1])/d[j])/(1-1/d[j]);                  
                }  		
            }  
        }  
        printf("%.10f\n",dp[a][b]);  
    }  
    return 0;  
}
