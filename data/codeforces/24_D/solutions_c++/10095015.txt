using namespace std;
double dp[1111][1111];

int main(){
    memset(dp,0,sizeof dp);
    int n,m,x,y;
    cin>>n>>m>>x>>y;
    for(int i=n-1;i>=1;i--){
        for(int k=1;k<=50;k++){
            for(int j=1;j<=m;j++){
                int cnt=4;
                if(j==1) cnt--;
                if(j==m) cnt--;
                dp[i][j]=1.0+(dp[i][j]+dp[i+1][j]+dp[i][j-1]+dp[i][j+1])/(double)cnt;
            }
        }
    }
    printf("%.10lf\n",dp[x][y]);
    return 0;
}
