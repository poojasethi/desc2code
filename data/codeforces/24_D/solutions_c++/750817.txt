#include <cstdio>
double f[1010][1010];
int main() {
	int n, m; scanf("%d%d", &n, &m);
	int x, y; scanf("%d%d", &x, &y);
	for (int i = 1; i <= m; ++i) f[n][i] = 0;
	for (int i = n - 1; i >= x; --i) 
		for (int k = 0; k < 40; ++k) 
			for (int j = 1; j <= m; ++j){
				if (m == 1) f[i][j] = (f[i + 1][j] + f[i][j]) / 2 + 1; else
				if (j == 1) f[i][j] = (f[i + 1][j] + f[i][j + 1] + f[i][j]) / 3 + 1; else
				if (j == m) f[i][j] = (f[i][j - 1] + f[i + 1][j] + f[i][j]) / 3 + 1; else
				f[i][j] = (f[i + 1][j] + f[i][j + 1] + f[i][j - 1] + f[i][j]) / 4 + 1;
			}
	printf("%.10f\n", f[x][y]);
	return 0;
}
