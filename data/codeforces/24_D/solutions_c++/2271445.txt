#include <iostream>
#include <iomanip>
using namespace std;
int n,m,x,y,p[1001];
double d[2][1001][121];
int main(){
	cin>>n>>m>>x>>y;n-=x;
	for(int i=1;i<=m;i++) p[i]=2+(i>1)+(i<m);
	for(int i=1,c=1;c<= n;c++,i^=1)
		for (int k=120;k>0;k--)
			for (int j=1;j<=m;j++)
				d[i][j][k-1]=(d[i][j][k]+d[i][j-1][k]+d[i][j+1][k]+d[1-i][j][0])/p[j]+1;
	cout<<fixed<<setprecision(20)<<d[n % 2][y][0]<<endl;
	return 0;
}
