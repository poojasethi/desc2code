#include <iostream>
#include <cstdio>
using namespace std;

struct dd{ double a,b; } g[1010];
int n,m,x,y;
double f[1010][1010];

int main()
{
	scanf("%d%d",&n,&m);
	scanf("%d%d",&x,&y);
	for (int i=n-1;i>=x;i--)
	{
		if (m==1)
		{
			f[i][1]=(0.5*f[i+1][1]+1)/0.5;
		}else
		{
		g[1].a=0.5;g[1].b=f[i+1][1]/2+1.5;
		for (int j=2;j<m;j++)
		{
			g[j].a=0.25/(0.75-0.25*g[j-1].a);
			g[j].b=(0.25*(f[i+1][j]+g[j-1].b)+1)/(0.75-0.25*g[j-1].a);
		}
		f[i][m]=((g[m-1].b+f[i+1][m])/3.0+1)/((2.0-g[m-1].a)/3.0);
		for (int j=m-1;j>=1;j--) f[i][j]=f[i][j+1]*g[j].a+g[j].b;
		}
	}
	printf("%.8f\n",f[x][y]);
	
	return 0;
}
