#include <iostream>
#include <iomanip>
using namespace std;

const int MAXn = 1000+10;
double C[MAXn], Z[MAXn];
double ans[MAXn][MAXn];
int n, m;

int main()
{
	ios::sync_with_stdio(false);
	cin >> n >> m;
	for(int i = n-2; i >= 0; i--)
	{
		double *a = ans[i];
		double *b = ans[i+1];
		Z[0] = .5;
		C[0] = (b[0]+3)/2.0;
		for(int j = 0; j < m-2; j++)
		{
			Z[j+1] = 1.0/(3-Z[j]);
			C[j+1] = (C[j]+b[j+1]+4)/(3.0-Z[j]);
		}
		a[m-1] = (C[m-2]+b[m-1]+3)/(2.0-Z[m-2]);
		for(int j = m-2; j >= 0; j--)
			a[j] = Z[j]*a[j+1]+C[j];
	}
	int r, c;
	cin >> r >> c;
	r--; c--;
	if(m == 1)
		cout << 2*(n-r-1) << endl;
	else
		cout << setprecision(10) << fixed << ans[r][c] << endl;
}
