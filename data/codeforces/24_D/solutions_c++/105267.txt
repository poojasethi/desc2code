#include <cstdio>
#include <cstring>

using namespace std;

#define MAX 1005
#define ITER 500

double A[MAX], B[MAX], C[MAX];

int main(){
    //freopen("test.txt", "r", stdin);
    int N, M, n, m;
    scanf("%d %d\n", &N, &M);
    scanf("%d %d\n", &n, &m);

    for(int r = n; r < N; r++){
        memset(B, 0, sizeof(B));
        for(int i = 0; i < ITER; i++){
            for(int c = 1; c < M; c++)
                C[c] = (B[c - 1] + B[c] +  B[c + 1] + A[c]) / 4.0 + 1.0;
            C[0] = C[M - 1] = (B[0] + B[1] + A[0]) / 3.0 + 1.0;
            memcpy(B, C, sizeof(C));
        }
        memcpy(A, B, sizeof(B));
    }
    if(M == 1){
        printf("%.131g\n", (double)2.0 * (N - n));
    }else{
        printf("%.131g\n", A[m - 1]);
    }

    return 0;
}
