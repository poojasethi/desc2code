#include <stdio.h>
#include <algorithm>
#include <iostream>
using namespace std;
double dp[1001][1001];
int dd[1001];
int main(){
	int n,m,a,b,x,i,j,d;
	scanf("%d %d %d %d",&n,&m,&a,&b);
	
	for(i=1;i<=m;++i){
		x= 2; if(i>1)++x;if(i<m)++x;
		dd[i]=x;
	}

	for(i=n-1;i>=a;i--)
		for(d=1;d<=100;d++)
			for(j=1;j<=m;j++)
				dp[i][j]= (dp[i][j]+dp[i][j-1]+dp[i][j+1]+dp[i+1][j] )/dd[j] + 1;

	printf("%0.6f\n",dp[a][b]);
}
