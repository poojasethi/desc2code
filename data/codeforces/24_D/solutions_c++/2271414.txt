#include <iostream>
#include <iomanip>
using namespace std;
int n, m, x, y;
double d[2][1001][121];
int main()
{
	cin >> n >> m >> x >> y; n -= x;
	cout << fixed << setprecision (20);
	if (m == 1) {
		cout << 2. * n << endl;
		return 0;
	}
	for (int i = 1, c = 1; c <= n; c++, i ^= 1) {
		for (int k = 120; k > 0; k--) {
			d[i][1][k-1] = (d[i][1][k] + d[i][2][k] + d[1-i][1][0]) / 3 + 1;
			for (int j = 2; j < m; j++)
				d[i][j][k-1] = (d[i][j][k] + d[i][j-1][k] + d[i][j+1][k] + d[1-i][j][0]) / 4 + 1;
			d[i][m][k-1] = (d[i][m][k] + d[i][m-1][k] + d[1-i][m][0]) / 3 + 1;
		}
	}
	cout << d[n % 2][y][0] << endl;
	return 0;
}
