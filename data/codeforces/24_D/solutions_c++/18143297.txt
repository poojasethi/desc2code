#include<stdio.h>
#include<math.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxn=1000+10;
double dp[maxn][maxn],d[maxn];
int n,m;
int main()
{
	int i,j;
	scanf("%d%d%d%d",&n,&m,&i,&j);
	//赋初值
	for(int t=1;t<=n;t++)
		for(int p=1;p<=m;p++)
			dp[t][p]=0.0;
	for(int a=1;a<=m;a++)
	{
		d[a]=2;
		if(a>1)d[a]++;
		if(a<m)d[a]++;
	}
	for(int a=n-1;a>=i;a--)
	{
		for(int t=0;t<50;t++)
		{
			for(int b=1;b<=m;b++)
			{
				dp[a][b]=(1.0+(dp[a][b+1]+dp[a][b-1]+dp[a+1][b])/d[b])*d[b]/(d[b]-1);
			}

		}
	}
	printf("%.10lf\n",dp[i][j]);
	return 0;
}
