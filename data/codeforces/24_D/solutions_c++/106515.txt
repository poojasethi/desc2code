
#include <cstdio>
using namespace std;

int n, m, R, C;
double re[1005];
double ex[1005][3], f[1005];

void solve() {
	ex[0][0]=0; ex[0][1]=2/3.0; ex[0][2]=-1/3.0; f[0]=1+re[0]/3;
	for (int c=1; c<m-1; c++) {
		ex[c][0]=-1/4.0;
		ex[c][1]=3/4.0;
		ex[c][2]=-1/4.0;
		f[c]=1+re[c]/4;
	}
	ex[m-1][0]=-1/3.0; ex[m-1][1]=2/3.0; ex[m-1][2]=0; f[m-1]=1+re[m-1]/3;
	
	for (int c=1; c<m; c++) {
		ex[c][1] -= ex[c-1][2]*ex[c][0]/ex[c-1][1];
		f[c] -= f[c-1]*ex[c][0]/ex[c-1][1];
	}
	
	re[m-1]=f[m-1]/ex[m-1][1];
	for (int c=m-2; c>=0; c--) {
		f[c] -= ex[c][2]*re[c+1];
		re[c] = f[c]/ex[c][1];
	}
	
	//for (int c=0; c<m; c++) printf (" %.3f", re[c]);
	//printf("\n");
}

main() {
	scanf("%d%d%d%d", &n, &m, &R, &C);
	R=n-R;
	if (m>1) for (int r=0; r<R; r++) solve();
	else re[C-1]=R*2;
	printf("%.4f", re[C-1]);
}
