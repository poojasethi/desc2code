#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <cstring>
#include <string>
#include <cctype>
#include <algorithm>
#define maxn 1005

using namespace std;

int n,m,x,y;
double f[maxn][maxn];


void init()
 {
  scanf("%d%d",&n,&m);
  scanf("%d%d",&x,&y);
 }
 
void work()
 {
  int i,j,k,cnt;
   for (i=n-1;i>=x;i--)
    for (k=1;k<=100;k++)
    for (j=1;j<=m;j++)
     {
      if (j==1 && j==m)
       {
        f[i][j]=(f[i][j]+f[i+1][j])*0.5+1;
       }
      else if (j==1)f[i][j]=(f[i][j]+f[i+1][j]+f[i][j+1])/3+1;
       else if (j==m)f[i][j]=(f[i][j]+f[i+1][j]+f[i][j-1])/3+1;
        else f[i][j]=(f[i][j]+f[i+1][j]+f[i][j-1]+f[i][j+1])/4+1;
     }
  printf("%.10lf\n",f[x][y]);
 }
 
int main()
 {
  init();
  work();
  return 0;
 }
