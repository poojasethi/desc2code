import sys

n = int(sys.stdin.readline())
x = [int(i) for i in sys.stdin.readline().split()]
s = [0]*100001
d = [0]*100001
s[1] = 1
for i in range(2, len(s)):
    if s[i] == 0:
        for j in range(i, len(s), i):
            s[j] = i

r = 0
for i in range(0, n):
    j = x[i]
    z = 1
    while j != 1: 
        z = max(z, d[s[j]]+1)
        j /= s[j]
    j = x[i]
    while j != 1:
        d[s[j]] = z
        j /= s[j]
    r = max(r, z)

print r
