N = 100001
m = [0] * N
for i in range(2, N):
	if not m[i]:
		m[i] = i
		for j in range(i * 2, N, i):
			if not m[j]:
				m[j] = i
input()
d = [0] * N
r = 1
for v in map(int, raw_input().split()):
	t = 1
	u = v
	while u > 1:
		t = max(t, d[m[u]] + 1)
		u /= m[u]
	r = max(r, t)
	u = v
	while u > 1:
		d[m[u]] = max(d[m[u]], t)
		u /= m[u]
print r
