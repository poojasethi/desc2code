N = 100001
m = [0] * N
for i in range(2, N):
    if not m[i]:
        m[i] = i
        for j in range(i * 2, N, i):
            if not m[j]:
                m[j] = i
input()
d=[0]*N
r=1
for a in map(int,raw_input().split()):
    t = 1
    v = a
    while v > 1:
        t = max(d[m[v]]+1,t)
        v /= m[v]
    r = max(r,t)
    while a > 1:
        d[m[a]] = max(d[m[a]], t)
        a /= m[a]
print r
