if __name__ == "__main__":
    n = int(sys.stdin.readline())
    a = map(int, sys.stdin.readline().split())

    ds = [[] for i in xrange(0, 100005)]
    for i in xrange(2, 100005) :
        j = i
        while j < 100005 :
            ds[j].append(i)
            j += i

    dp = [0] * 100005
    for i in xrange(0, n) :
        me = a[i]
        x = 1
        #print(len(ds[me]))
        for j in ds[me] :
            #print(j)
            x = max(x, dp[j] + 1)
        #print(me)
        #print(x)
        #here()
        for j in ds[me] :
            dp[j] = max(dp[j], x)

    ans = 1
    for i in xrange(0, n) :
        ans = max(ans, dp[a[i]])

    print(ans)
