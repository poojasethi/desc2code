#include<stdio.h>
#include<string.h>
int max(int a,int b){
	if(a>=b)
	return a;
	else return b;
}
int dp[100010];
int main(){
	int i,j,k,m,n,x;
	scanf("%d",&n);
	memset(dp,0,sizeof(dp));
	
	while(n--){
		scanf("%d",&x);m=0;
		for(j=2;j*j<=x;j++){
			if(x%j==0){
				m=max(dp[j],m);
				m=max(dp[x/j],m);
			}
		}
		for(j=1;j*j<=x;j++){
			if(x%j==0){
				dp[j]=max(dp[j],m+1);
				dp[x/j]=max(dp[x/j],m+1);
			}
		}
		
	}
		printf("%d\n",dp[1]);
}
