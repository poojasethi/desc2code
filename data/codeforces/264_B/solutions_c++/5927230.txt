#include <iostream>
#include <cstdio>
using namespace std;

int ans[100001],last[100001];
int main() {
	int n,a,L=0;	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&a);
		ans[a]=1;
		for(int j=2;j*j<=a;j++){
			if(! (a%j)){
				ans[a]=max(ans[a],ans[last[j]]+1);
				ans[a]=max(ans[a],ans[last[a/j]]+1);
				last[j]=a;
				last[a/j]=a;
			}
		}
		last[a]=a;
		L=max(L,ans[a]);
	}
	printf("%d",L);
	return 0;
}
