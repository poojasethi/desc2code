#include<cstdio>
#include<iostream>
#include<cstring>
#define maxn 100009
using namespace std;
int a[maxn],idx[maxn],dp[maxn];
int main()
{
	int n,ans=0;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		for(int j=1;j*j<=a[i];j++)if(a[i]%j==0)
		{
			if(j!=1)dp[i]=max(dp[i],dp[idx[j]]+1);
			dp[i]=max(dp[i],dp[idx[a[i]/j]]+1);
			ans = max(ans,dp[i]);
			idx[j]=idx[a[i]/j]=i;
		}
	}
	printf("%d\n",ans);
	return 0;
}
