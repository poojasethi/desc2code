#include<iostream>
using namespace std;
int main()
{
int a[100005],i,j,n,b[100005]={0},c[100005]={0},p=0,temp;
cin>>n;
for(i=1;i<=n;i++)
{
cin>>a[i];
c[i]=c[b[a[i]]]+1;
if(c[i]>p)
p=c[i];
b[1]=b[a[i]]=i;
for(j=2;j*j<=a[i];j++)
{
if(a[i]%j==0)
{
c[i]=max(c[i],max(c[b[a[i]/j]]+1,c[b[j]]+1));
p=max(p,c[i]);
b[j]=b[a[i]/j]=i;
}
}
}
cout<<p;
}
