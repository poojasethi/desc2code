#include<bits/stdc++.h>

using namespace std;

int M[100005],A[100005],n,i,j,k,mx;

int main(){

	cin >> n;
	
	for(i=1;i<=n;i++)
		cin >> A[i];

	for(i=1;i<=n;i++){

		for(k=0,j=2; j*j <= A[i];j++)
			if(A[i] % j == 0)
				k = max(k,max(M[j],M[A[i] / j]));

		M[ A[i] ] = k+1;
	
		for(j=2;j*j<=A[i];j++)
			if(A[i] % j == 0)
				M[j] = M[ A[i] / j] = k+1;

	}
    for(i=1;i<=n;i++)
	    mx = max(mx , M[ A[i] ]);

	cout << mx;
	return 0;
}
