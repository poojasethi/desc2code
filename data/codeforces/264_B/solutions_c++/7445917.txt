#include<iostream>
#include <cstdio>
using namespace std;
const int N=100001;
int n,m,a,ans;
int s[N],p[N];
int main()
{
    ans=1;
    for(int i=2;i<N;i++)
        if(!s[i])
            for(int j=0;j<N;j+=i) s[j]=i;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&a),m=0;
        for(int j=a;j>1;j/=s[j])m=max(m,p[s[j]]+1);
        for(int j=a;j>1;j/=s[j])p[s[j]]=max(p[s[j]],m);
        ans=max(ans,m);
    }
    cout<<ans;
    return 0;
}
