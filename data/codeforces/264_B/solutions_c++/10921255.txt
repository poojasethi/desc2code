#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
#define MAX 100005
vector <int>d[MAX];
int a[MAX]={0};
int dp[MAX]={0};
int main()
{
int i,n,t=0,j,lim=0;
cin>>n;
for(i=0;i<n;i++){cin>>a[i];lim=max(lim,a[i]);}
for(i=2,d[1].push_back(1);i<=lim;i++) if(d[i].size()==0) for(j=i;j<=lim;j+=i)d[j].push_back(i);
for(i=0;i<n;i++){
	for(j=0,t=0;j<d[a[i]].size();j++)t=max(t,dp[d[a[i]][j]]);
	for(j=0;j<d[a[i]].size();j++)dp[d[a[i]][j]]=t+1;
}
for(i=1,t=0;i<=lim;i++)t=max(t,dp[i]);
cout<<t<<endl;
}
