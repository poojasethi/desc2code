#include <stdio.h>
#define maxn 100005
int dp[maxn],yz[maxn];
int main()
{
	int n,i,j,x,k,ma,maa;
	scanf("%d",&n);
	maa=1;
	for(i=0;i<n;i++){
		scanf("%d",&x);
		k=0;
		ma=0;
		for(j=2;j*j<=x;j++){
			if(x%j==0){
				yz[k++]=j;
				if(dp[j]+1>ma)
					ma=dp[j]+1;
				while(x%j==0)
					x/=j;
			}
		}
		if(x>1)
		{
			yz[k++]=x;
			if(dp[x]+1>ma)
				ma=dp[x]+1;
		}
		for(j=0;j<k;j++)
			dp[yz[j]]=ma;
		if(ma>maa)
			maa=ma;
	}
	printf("%d\n",maa);
	return 0;
}
