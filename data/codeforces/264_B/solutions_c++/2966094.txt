#include<cstdio>
#include<cstring>
int f[100005];
int main(){
	int n,i,j,mxn,ans,a;
	scanf("%d",&n);
	memset(f,0,sizeof f);
	ans=0;
	for (i=0;i<n;i++){
		scanf("%d",&a);
		mxn=0;
		for (j=1;j*j<=a;j++) if (a%j==0 && j!=1 && f[j]>mxn) mxn=f[j];
		for (j--;j;j--) if (a%j==0 && a/j!=1 && f[a/j]>mxn) mxn=f[a/j];
		mxn++;
		for (j=1;j*j<=a;j++) if (a%j==0 && j!=1 && mxn>f[j]) f[j]=mxn;
		for (j--;j;j--) if (a%j==0 && a/j!=1 && mxn>f[a/j]) f[a/j]=mxn;
		if (mxn>ans) ans=mxn;
	}
	printf("%d\n",ans);
	return 0;
}
