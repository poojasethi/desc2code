#include<cstdio>
#include<cstring>
#include<map>
#include<iostream>
using namespace std;
const int N = 100005;
map<int,int>lzs;
int dp[N+N],t[N];
int main()
{
	int n,i,j;
	scanf("%d",&n);
	int ret=1;
	for(i=1;i<=n;i++)
	{
		int v,mx=0,k=0;
		scanf("%d",&v);
		for(j=2;j*j<=v;j++)if(v%j==0)
		{
			if(mx<dp[j]+1)mx=dp[j]+1;
			while(v%j==0)v/=j;
			t[k++]=j;
		}
		if(v>1)
		{
			if(mx<dp[v]+1)mx=dp[v]+1;
			t[k++]=v;
		}
		for(j=0;j<k;j++)if(dp[t[j]]<mx)dp[t[j]]=mx;
		if(ret<mx)ret=mx;
	}
	printf("%d\n",ret);
	return 0;
}
