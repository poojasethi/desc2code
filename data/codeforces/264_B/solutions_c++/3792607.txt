#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
using namespace std;
int i,j,k,l,n,m,x,ans,fans,g[120000];
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i){
		scanf("%d",&x);
		ans=0;
		for(j=1;j*j<=x;++j)if(x%j==0){
			if(j!=1)ans=max(ans,g[j]);
			if(x/j!=1)ans=max(ans,g[x/j]);
		}
		ans++;
		fans=max(fans,ans);
		for(j=1;j*j<=x;++j)if(x%j==0){
			if(j!=1)g[j]=max(g[j],ans);
			if(x/j!=1)g[x/j]=max(g[x/j],ans);
		}
	}
	printf("%d\n",fans);
}
