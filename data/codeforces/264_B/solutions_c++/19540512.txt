#include <bits/stdc++.h>

using namespace std;

int dp[100005];

int main()
{
	int n,x,mx=1;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>x;	
		dp[x]=1;
		int sq=sqrt(x);
		for(int i=2;i<=sq;i++)
		{
			if(x%i==0)
			{
				dp[x]=max(dp[x],dp[i]+1);
				dp[x]=max(dp[x],dp[x/i]+1);
				mx=max(mx,dp[x]);
			}	
		}
		for(int i=2;i<=sq;i++)
		{
			if(x%i==0)
			{
				dp[i]=dp[x];dp[x/i]=dp[x];
			}
		}
	}
	for(int i=0;i<100005;i++)
	{
		mx=max(mx,dp[i]);
	}
	cout<<mx;
//	cout<<dp[x]<<"\n";*
	return 0;
}
