#include <cstdio>

int n, d[100005], A, t, f[20], p;

int main(){
	scanf("%d", &n);
	while(n--){
		scanf("%d", &t);
		p = 0;
		if(t == 1) f[p++] = 1;
		
		for(int i = 2; i * i <= t; i++) if(t % i == 0){
			f[p++] = i;
			while(t % i == 0) t /= i;
		}
		if(t > 1) f[p++] = t;
		
		t = 0;
		for(int i = 0; i < p; i++) if(d[f[i]] + 1 > t)
			t = d[f[i]] + 1;
		
		for(int i = 0; i < p; i++) if(d[f[i]] < t)
			d[f[i]] = t;
			
		if(t > A) A = t;
	}
	printf("%d\n", A);
}
