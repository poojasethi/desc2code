#include<bits/stdc++.h>
using namespace std;
int mx=1,f[100005],b[100005],n,i,j;
main()
{
    scanf("%d",&n);
    for (i=0;i<n;i++) scanf("%d",&b[i]);
    for (i=0;i<n;i++)
    {   int  k=0;
        for (j=2;j*j<=b[i];j++)
            if (b[i]%j==0)
                k=max(k,max(f[j],f[b[i]/j]));
        for (j=1;j*j<=b[i];j++)
            if (b[i]%j==0)
                f[j]=k+1,f[b[i]/j]=k+1;
    }
    for (i=0;i<n;i++)
        mx=max(mx,f[b[i]]);
    printf("%d",mx);
}
