# include <stdio.h>
# include <algorithm>
using namespace std;
int a,k,mx;
int b[100009];
int f[100009];
int main()
{
	scanf("%d",&a);
	for(int i=0;i<a;i++)scanf("%d",&b[i]);
	for(int j=0;j<a;j++)
	{
		k=0;
		for(int i=2;i*i<=b[j];i++)
		if(b[j]%i==0)k=max(k,max(f[i],f[b[j]/i]));
		f[b[j]]=k+1;
		for(int i=2;i*i<=b[j];i++)
		if(b[j]%i==0)
		{f[i]=k+1;f[b[j]/i]=k+1;}
	}
	for(int i=1;i<=1e5;i++)mx=max(mx,f[i]);
	printf("%d",mx);
}
