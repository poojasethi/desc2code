#include<stdio.h>
#define X 100010
int s[X],p[X];
void max(int& a,int b){if(a<b)a=b;}
int main(){
    int i,j,n,m,as,a;
    for(i=2;i<X;i++)if(!s[i])
    for(j=i;j<X;j+=i)s[j]=i;
    scanf("%d",&n);as=1;
    for(i=0;i<n;i++){
        scanf("%d",&a);m=0;
        for(j=a;j!=1;j/=s[j])max(m,p[s[j]]+1);
        for(j=a;j!=1;j/=s[j])max(p[s[j]],m);
        max(as,m);
    }
    printf("%d\n",as);
    return 0;
}
