#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll a[100005],b[100005],c[100005];
int main()
{
ll n,i,j,k;
cin>>n;
for(i=1;i<=n;i++)
	cin>>a[i];
for(i=1;i<=n;i++)
{
	ll ma=0;
	for(j=1;j*j<=a[i];j++)
	{
		if(a[i]%j==0)
		{
			if(j!=1)
			ma=max(max(ma,c[b[j]]),c[b[a[i]/j]]);
			b[j]=i;b[a[i]/j]=i;
		}
	}
		c[i]=1+ma;
}
ll ans=0;
for(i=1;i<=n;i++)
ans=max(ans,c[i]);
cout<<ans<<endl;
}
