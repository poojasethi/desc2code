#include<cstdio>
#include<algorithm>
using namespace std;
int d[120000];
int dp[120000],ans;
int n,a[120000];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
	}
	for (int i=1;i<=n;i++)
	{
		for (int j=1;j*j<=a[i];j++)
		{
			if(a[i]%j==0)
			{
				dp[i]=max(dp[i],dp[d[a[i]/j]]+1);
				if(j!=1)dp[i]=max(dp[i],dp[d[j]]+1);
				d[j]=d[a[i]/j]=i;
			}
		}
		ans=max(ans,dp[i]);
	}
	printf("%d\n",ans);
	return 0;
}
