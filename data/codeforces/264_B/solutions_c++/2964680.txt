#include<cstdio>
#include<cstring>
#include<algorithm>
int n,x,m,ans,a[100005],f[100005];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++){
		scanf("%d",&x),m=0;
		for (int j=1;j*j<=x;j++) if (x%j==0){
			a[++m]=j; if (j*j!=x) a[++m]=x/j;
			}
		int res=0;
		for (int j=2;j<=m;j++) if (f[a[j]]>res) res=f[a[j]];
		res++; if (res>ans) ans=res;
		for (int j=2;j<=m;j++) if (res>f[a[j]]) f[a[j]]=res;		
		}
	printf("%d\n",ans);
	return 0;
}
