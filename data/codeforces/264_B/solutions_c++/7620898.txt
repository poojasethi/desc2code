#include<cstdio>
#include<algorithm>
#define N 100005
using namespace std;
int a[N],b[N],f[N],i,n,j,s,m,ans;
inline int work(int k){b[++s]=k;m=max(m,f[k]);}
int main()
{
  scanf("%d",&n);
  for (i=1;i<=n;i++)
    scanf("%d",&a[i]);
  sort(a+1,a+n+1);
  for (i=1;i<=n;i++)
  {
    for (j=2,s=0,m=0;j*j<=a[i];j++)
      if (a[i]%j==0) {work(j);if (j*j<a[i]) work(a[i]/j);}
    work(a[i]);ans=max(ans,++m);
    for (j=1;j<=s;j++) 
      f[b[j]]=max(m,f[b[j]]);
  }
  printf("%d",ans);return 0;
}
