#include<bits/stdc++.h>
using namespace std;
int n,a[100005],b[100005],c[100005],p;
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
    cin>>a[i];
    for(int i=1;i<=n;i++)
    {
    	for(int j=1;j*j<=a[i];j++)
    	{
    		if(a[i]%j==0)
    		{
    			b[i]=max(b[i],b[c[a[i]/j]]+1);
                if(j>1)b[i]=max(b[i],b[c[j]]+1);
                p=max(p,b[i]);
                c[j]=c[a[i]/j]=i;
    		}
    	}
    }	
    cout<<p;
	return 0;
}
