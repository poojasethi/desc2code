from math import factorial as fact


def comb(n, r):
    return fact(n) / (fact(r) * fact(n - r))

n, k = map(int, raw_input().split())
if n == 1 or comb(n, 2) < k * n:
    print -1
else:
    print k * n
    for i in range(n):
        for j in range(1, k + 1):
            print i + 1, (i + j) % n + 1
