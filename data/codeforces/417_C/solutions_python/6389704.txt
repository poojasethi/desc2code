nk=raw_input().split()
n=int(nk[0])
k=int(nk[1])

matches=[]
if n-1-k<k:
    print -1
else:
    for i in range(1,n+1):
        for j in range(1,k+1):
            beat=i+j
            if i+j>n:
                beat=beat%n
            matches.append([i, beat])
            
    print len(matches)
    for m in matches:
        print m[0],m[1]
