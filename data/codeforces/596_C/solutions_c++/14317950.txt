#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(0);cin.tie(0);
#define ll long long
using namespace std;
map<int,set<pair<int,int> > > m;
vector<pair<int,int> > ans;
int w[100005];
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        int x,y;
        cin>>x>>y;
        m[y-x].insert(make_pair(x,y));
    }
    for(int i=0;i<n;i++)
        cin>>w[i];
    map<int,int> check;
    for(int i=0;i<n;i++)
    {
        int v=w[i];
        if(m[v].empty())
        {
            cout<<"NO";
            return 0;
        }
        pair<int,int> p=*m[v].begin();
        if(check.count(p.second)&&check[p.second]>p.first)
        {
            cout<<"NO";
            return 0;
        }
        check[p.second]=p.first;
        ans.push_back(p);
        m[v].erase(m[v].begin());
    }
    cout<<"YES\n";
    for(int i=0;i<n;i++)
    {
        cout<<ans[i].first<<" "<<ans[i].second<<"\n";
    }
    return 0;
}
