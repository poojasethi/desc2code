#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    map<int,set<pair<int,int> >  >mapp;
    int n;
    cin>>n;
    int x,y;
    for(int i=0;i<n;i++)
    {
        cin>>x>>y;
        mapp[y-x].insert(make_pair(x,y));
    }
    vector< pair<int,int> >p(2000010);
    int diff;
    for(int i=0;i<n;i++)
    {
        cin>>diff;
        if(mapp[diff].empty())
        {
            cout<<"NO"<<endl;
            return 0;
        }
        p[i]= *(mapp[diff].begin());
        mapp[diff].erase(mapp[diff].begin());
    }
    for(int i=0;i<n-1;i++)
    {
        if(p[i+1].first<p[i].first&&p[i+1].second<=p[i].second)
        {
            cout<<"NO"<<endl;
            return 0;
        }
    }
    puts("YES");
    for (int i = 0; i < n; i++)
    {
        int x = p[i].first;
        int y = p[i].second;
        printf("%d %d\n", x, y);
    }
    return 0;
}
