#include<iostream>
#include<algorithm>
using namespace std;

int n, m, k, p, o;
int a[1010][1010];

int main(){
	cin >> n >> m >> k;
	o = n + m + k -1;
	if( (n+m)%2 ){
		cout <<"0" << endl;
		return 0;
	}
	for(int i = 0; i < k; i++){
		int x, y, w;
		cin >> x >> y;
		x--;y--;
		cin >> a[x][y];
	}
	for(int i = 0; i < n; i++){
		int ok = 1;
		for(int j = 0; j < m; j++)
			ok *= a[i][j];
		if(ok == 1){
			cout << "0" << endl;
			return 0;
		}
		if(ok == -1)
			o--;
	}

	for(int i  = 0; i < m; i++){
		int ok = 1;
		for(int j = 0; j < n; j++){
			ok *= a[j][i];
		}
		if(ok == 1){
			cout << "0" << endl;
			return 0;
		}
		if(ok == -1)
			o--;
	}
			
	cin >> p;
	long long ans = 1;
	for(int i = 0; i <( n*m-o); i++)
		ans = (ans*2)%p;
	cout << ans << endl;
	return 0;
}
