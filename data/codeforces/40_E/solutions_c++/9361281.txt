#include <bits/stdc++.h>
#define maxn 1009
using namespace std;
int a[maxn],cnt[maxn],n,m,MOD,k;
int main()
{
	int flag=0;
	scanf("%d%d%d",&n,&m,&k);
	if(n<m)
	{
		flag=1;
		swap(n,m);
	}
	for(int i=1;i<=n;i++)a[i]=1;
	for(int i=1;i<=k;i++)
	{
		int x,y,z;
		scanf("%d%d%d",&x,&y,&z);
		if(flag)
			swap(x,y);
		a[x]*=z;
		cnt[x]++;
	}
	scanf("%d",&MOD);
	flag=0;
	long long ans=1;
	for(int i=1;i<=n;i++)
	{
		if(!flag&&!cnt[i])
			flag=1;
		else if(cnt[i]==m&&a[i]!=-1)
			ans=0;
		else
			for(int j=0;j<m-cnt[i]-1;j++)
				ans=(ans+ans)%MOD;
	}
	if((n+m)%2==1)
		ans=0;
	printf("%I64d\n",ans);
	//system("pause");
	return 0;
}
