#include<cstdio>
#include<cstring>

int n,m,K,a[1010][1010],Cnt[1010],Last,Now,Ans,Peo,P;

int main()
{
	int i,j,k,l,_;
	memset(a,-1,sizeof a);
	scanf("%d%d%d",&n,&m,&K);
	if((n^m)&1)
	{
		printf("0\n");
		return 0;
	}
	if(n<m)
	{
		for(i=0;i<K;i++)
		{
			scanf("%d%d%d",&j,&k,&l);
			j--;k--;
			a[k][j]=(l==-1);
			Cnt[k]++;
		}
		i=n;n=m;m=i;
	}
	else
	{
		for(i=0;i<K;i++)
		{
			scanf("%d%d%d",&j,&k,&l);
			j--;k--;
			a[j][k]=(l==-1);
			Cnt[j]++;
		}
	}
	scanf("%d",&P);
	for(Last=0;Last<n&&Cnt[Last];Last++);
	for(i=0;i<n;i++)if(i!=Last)
	{
		Peo=Now=0;
		for(j=0;j<m;j++)
			if(a[i][j]!=-1)
			{
				Now^=a[i][j];
				Peo++;
			}
		if(Peo==m&&Now==0)
		{
			printf("0\n");
			return 0;
		}
		if(Peo!=m)Ans+=m-Peo-1;
	}
	_=1;
	for(i=0;i<Ans;i++)_=(_*2)%P;
	printf("%d\n",_);
	return 0;
}
