#include <bits/stdc++.h>

struct Substring {
	char *begin, *end;
	Substring(char *p): begin(p), end(p + 1) {
	}
	bool operator<(const Substring &o)const{
		return std::lexicographical_compare(o.begin, o.end, begin, end);
	}
};

#define MAX_N 100010
char buf[MAX_N];

int main() {
	int K;
	std::priority_queue<Substring> q;
	scanf("%s%d", buf, &K);
	for(char *p = buf; *p; p ++)
		q.push(Substring(p));
	while(!q.empty() && -- K) {
		Substring t = q.top();
		q.pop();
		if(*t.end) {
			t.end ++;
			q.push(t);
		}
	}
	if(K) {
		puts("No such line.");
	} else {
		*q.top().end = 0;
		puts(q.top().begin);
	}
	return 0;
}
