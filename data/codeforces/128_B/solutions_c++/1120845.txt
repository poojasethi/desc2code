#include<cstdio>
#define go(n) for(i=0;i<n;++i)
int x[26][102400],l,p[26],k,t,i,z;
long long y[26];
char s[102400];
int main(){
	gets(s);
	for(;s[l];)s[l++]-='a';
	go(l){
		y[s[i]]+=l-i;
		x[s[i]][p[s[i]]++]=i;
	}
	scanf("%d",&k);
	for(;k>0;){
		go(26)
			if(k>y[i])k-=y[i];
				else{z=i;break;}
		if(i<26)putchar(z+'a');else break;
		if(k){
			int n=p[z];
			go(26)p[i]=y[i]=0;
			go(n){
				t=1+x[z][i];
				if(t<l){
					y[s[t]]+=l-t;
					x[s[t]][p[s[t]]++]=t;
				}
				--k;
			}
		}
	}
	if(k>0)puts("No such line.");
	return 0;
}
