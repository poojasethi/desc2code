#include<cstdio>
#include<cstring>
#include<string>
#include<iostream>
#include<algorithm>
#include<string>
#include<queue>
#include<stack>
#include<vector>
using namespace std;
const int maxn = 200005;
struct Str{
	string s;
	int pos;
	bool operator <(const Str &x) const{
		return x.s < s;
	}
}fr,tmp;
char str[maxn];
int k;
int main()
{
	int len;
	scanf("%s",str);
	scanf("%d",&k);
	len = strlen(str);
	priority_queue<Str> q;
	for(int i = 0; str[i];i++)
	{
		tmp.s = str[i];
		tmp.pos = i;
		q.push(tmp);
	}
	while(!q.empty()&&k)
	{
		k--;
		fr = q.top();q.pop();
		//cout<<fr.s<<endl;
		if(k == 0)
		{
			cout<<fr.s<<endl;
			break;
		}
		if(fr.pos < len - 1)
		{
			fr.s+=str[++fr.pos];
			q.push(fr);
		}
	}
	if(k) cout<<"No such line.\n"<<endl;
}
