#include<bits/stdc++.h>
#define maxx 111111
using namespace std;
struct node{
	int pos;
	string s;
	bool operator<(const node &p)const{
		return s>p.s;
	}
};
priority_queue<node> q;
int main()
{
	string s;
	int i,k,len;
	cin>>s>>k;
	len=s.size();
	if(k>(long long)len*(len+1)/2){
        puts("No such line.");
        return 0;
    }
	for(i=0;i<len;i++){
		node u;
		u.pos=i;
		u.s=s[i];
		q.push(u);
	}
	while(k--){
		node u=q.top();q.pop();
		if(k==0){
			cout<<u.s;
			return 0;
		}
		u.pos++;
		if(u.pos<len){
			u.s+=s[u.pos];
			q.push(u);
		}
	}
}
