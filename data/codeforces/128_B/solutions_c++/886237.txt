#include <iostream>
#include <string>
#include <cstdio>
#include <queue>

using namespace std;

string s;
long long k, len;
pair<string, int> unit;
priority_queue<pair<string, int>, vector<pair<string, int> >, greater<pair<string, int> > > pq;

int main() {
	cin >> s >> k;
	len = s.length();

	if(len * (len + 1) / 2 < k) {
		puts("No such line.");
		return 0;
	}
	for(int i = 0; i < len; ++i) {
		pq.push(make_pair(string(1, s[i]), i));
	}

	while(k--) {
		unit = pq.top();

		if(k == 0) {
			puts(unit.first.c_str());
			return 0;
		}
		pq.pop();
		if(unit.second + 1 < len) {
			unit.first += s[++unit.second];
			pq.push(unit);
		}
	}
	return 0;
}
