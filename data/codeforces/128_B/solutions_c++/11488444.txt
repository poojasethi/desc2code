#include <iostream>
#include <queue>
using namespace std;
struct sn{
	string str;
	int id;
	bool operator<(const sn &s)const{
		return str>s.str;
	}
};
int k,n;
priority_queue<sn> q;
sn s,cur;

int main()
{
    cin>>s.str>>k;
    n=s.str.size();
    if(k>((long long)n*(n+1)/2)){
    	cout<<"No such line.";
    	return 0;
	}
	for(int i=0;i<n;i++){
		cur.str=s.str[i];
		cur.id=i+1;
		q.push(cur);
	}
	while(--k){
		cur=q.top();q.pop();
	    if(cur.id<n){
	    	cur.str+=s.str[cur.id];
	    	cur.id++;
	    	q.push(cur);
		}
	}
	cout<<q.top().str;
	return 0;
}
