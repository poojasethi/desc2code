#include<cstdio>
#include<cstring>
#include<queue>
#include<cstring>
#include<string>
#include<iostream>
#define mp make_pair
#define pb push
#define st pair<string,int>
using namespace std; 
long long len,k;
string s;
priority_queue<st,vector<st>,greater<st> >heap;
int main(){
    cin>>s>>k;len=s.size();
    if(len*(len+1)/2<k){
        printf("No such line.\n");
        return 0;
    }
    for(int i=0;i<len;++i)
        heap.pb(mp(string(1,s[i]),i));
    --k;
    while(k){
        st a=heap.top();heap.pop();
        if(a.second<len-1)
            heap.pb(mp(a.first+s[a.second+1],a.second+1));
        --k;
    }
    cout<<heap.top().first<<endl;
    return 0;
}
