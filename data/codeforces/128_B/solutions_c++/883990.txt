#include<iostream>
#include<queue>
using namespace std;
struct st{
  string a;
  int p ;
  bool operator<(const st & b)const{
    return a > b.a;
  }
};

main(){
  string a;
  int k;
  while(cin>>a>>k){
    priority_queue<st> Q;
    for(int i=0;i<a.size();i++){
      Q.push((st){string(1,a[i]),i});
    }
    while(!Q.empty()){
      st now = Q.top();Q.pop();
      k--;
      if (k == 0){cout <<now.a<< endl;goto end;}
      if (now.p+1 == a.size());
      else Q.push((st){now.a+string(1,a[now.p+1]),now.p+1});
    }
    cout <<"No such line." << endl;
  end:;
  }
}
