#include<queue>
#include<string>
#include<iostream>
using namespace std;
struct node
{
    string s;
    int idx;
    bool operator <(const node& t)const
    {
        return t.s<s;
    }
};
priority_queue<node>q;
string s;
int main()
{
    int k;
    cin>>s;
    cin>>k;
    int l=s.length();
    if((long long)k>(long long)(l+1)*l/2)cout<<"No such line.\n";else
    {
        for(int i=0;i<l;i++)
        {
            node t;
            t.s=s[i];
            t.idx=i;
            q.push(t);
        }

        for(int i=1;i<k;i++)
        {
            node t=q.top();
            q.pop();
            t.idx++;
            if(t.idx<l)t.s+=s[t.idx];
            if(t.idx<l)q.push(t);
        }

        cout<<q.top().s<<"\n";
    }
    return 0;
}
