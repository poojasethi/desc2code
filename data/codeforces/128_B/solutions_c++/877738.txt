#include<iostream>
#include<string>
#include<cstdio>
#include<queue>
using namespace std;
char ch[100002];
struct S
{
    string s;
    int id;
    S():s("0"){}
    bool operator<(const S& t)const
    {
        return s > t.s;
    }
};
priority_queue<S> q;
int main()
{
    int k;
    scanf("%s",ch);
    cin>>k;
    S s;
    for(int i = 0;ch[i]; ++i)
    {
        s.s[0] = ch[i];
        s.id = i;
        q.push(s);
    }
    while(--k)
    {
        if(q.empty()) break;
        S str = q.top();
        q.pop();
        if(ch[str.id + 1])
        {
            str.s += ch[++str.id];
            q.push(str);
        }
    }
    puts(k ? "No such line." : q.top().s.c_str());
    return 0;
}
