#include <iostream>
#include <queue>
#include <string>
using namespace std;

struct my{
    string s;
    int position;
};

priority_queue<my> q;

bool operator<(my ob1,my ob2)
{
    return ob1.s>ob2.s;
}

int main()
{
    string str;
    cin>>str;
    long long int i,len,k;
    cin>>k;
    len=str.size();
    if((len*(len+1))/2<k){
        cout<<"No such line."<<endl;
        return 0;
    }
    for(i=0;i<len;i++){
        my ob;
        ob.s=str[i];
        ob.position=i;
        q.push(ob);
    }
    for(i=0;i<k-1;i++){
        my ob=q.top();
        q.pop();
        ob.position++;
        if(ob.position<len){
            ob.s+=str[ob.position];
            q.push(ob);
        }
    }
    cout<<q.top().s<<endl;
    return 0;
}
