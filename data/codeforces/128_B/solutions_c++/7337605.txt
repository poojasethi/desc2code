#include <cstdio>
#include <queue>
#include <string>
#include <cstring>
#include <iostream>
using namespace std;

struct node{
    string x;
    int d;
    bool operator<(const node a)const{
        return x > a.x;
    }
}c;

priority_queue<node> q;
char a[100005];

int main(){
    int n, k;
    scanf("%s",a);
    n = strlen(a);
    scanf("%d",&k);
    for (int i=0;i<n;i++){
        c.x = a[i]; c.d = i;
        q.push(c);
    }
    if (k > (long long) (n + 1) * n / 2){
        puts("No such line.");
        return 0;
    }
    for (int i=0;i<k-1;i++){
        c = q.top();
        q.pop();
        if (c.d + 1 != n){
            c.x += a[++c.d];
            q.push(c);
        }
    }
    cout<<q.top().x;
    return 0;
}
