#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>
#include <vector>
#include <iostream>
using namespace std;
typedef long long llong;
struct nd
{
	string s;
	int pos;
};
struct cmp{bool operator()(nd& a,nd& b){return a.s > b.s;}};
priority_queue<nd,vector<nd>,cmp>q;
string str;
int k;
int main()
{
	int len,i;
	nd tmp;
//	freopen("test","r",stdin);
	cin>>str>>k;
	len = str.length();
	if (k>(llong)len*(len+1)/2)
	{
		cout<<"No such line."<<endl;
		return 0;
	}
	for ( i=0 ; i<len ; i++ )
	{
		tmp.s = str[i];
		tmp.pos = i;
		q.push(tmp);
	}
	while (k--)
	{
		tmp = q.top();q.pop();
		if (k==0)
		{
			cout<<tmp.s<<endl;
			return 0;
		}
		tmp.pos++;
		if (tmp.pos<len)
		{
			tmp.s+=str[tmp.pos];
			q.push(tmp);
		}
	}
	return 0;
}
