import heapq
import sys

s = raw_input()
i = int(raw_input())
L = len(s)

A = [(ss,j) for j,ss in enumerate(list(s))]

heapq.heapify(A)

for ii in xrange(i):
  if not A:
    break

  min_item, index = heapq.heappop(A)
  if index + 1 < L:
    heapq.heappush(A, (min_item + s[index+1], index+1))
else:
  print min_item
  sys.exit()

print 'No such line.'
