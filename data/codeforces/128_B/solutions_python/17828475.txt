import heapq
str = raw_input()
n = int(raw_input())
length = len(str)
string = [(str[i], i) for i in range(length)]
heapq.heapify(string)

if n > (length+1)*length/2:
 print('No such line.')

else:
 while n:
  n -= 1
  trans = heapq.heappop(string)
  if trans[1] < (length - 1):
   heapq.heappush(string, (trans[0] + str[trans[1]+1], trans[1]+1))
 print (trans[0])
