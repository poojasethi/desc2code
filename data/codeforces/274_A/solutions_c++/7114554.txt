#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,n,k,a[100010];
	set<int> s;
	cin>>n>>k;
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		if(a[i]%k==0)
		{
			set<int>::iterator it=s.find(a[i]/k);
			if(it==s.end())
				s.insert(a[i]);
		}
		else
			s.insert(a[i]);
	}
	cout<<s.size()<<endl;
	return 0;	
}
