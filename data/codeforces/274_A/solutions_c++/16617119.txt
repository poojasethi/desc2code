//Editorial
#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll a[200000];

set <ll> s;
int main()
{
ll n,k,i;
cin>>n>>k;

for(i=0;i<n;i++)
cin>>a[i];

sort(a,a+n);

for(i=0;i<n;i++)
{
if(a[i]%k!=0)
s.insert(a[i]);

else if(s.find(a[i]/k)==s.end())
s.insert(a[i]);
}

cout<<s.size();
return 0;
}
