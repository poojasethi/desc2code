#include <iostream>
#include <algorithm>
#include <set>
using namespace std;
int n,k,a[1<<19];
set<int>s;
int main(){
    cin>>n>>k;
    for(int i=0;i<n;i++)cin>>a[i];
    sort(a,a+n);
    for(int i=0;i<n;i++)
        if((a[i]%k!=0)|(s.find(a[i]/k)==s.end()))
            s.insert(a[i]);
    cout<<s.size();
}
