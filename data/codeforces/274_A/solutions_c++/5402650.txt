#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;
map<int,int>s;
int main(){
  int n,k,i;
  int r=0;
  int *a;
  scanf("%d %d",&n,&k);
  a=new int[n];
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  sort(a,a+n);
  for(i=0;i<n;i++)
    if (a[i]%k | !s[a[i]/k]) s[a[i]]=++r;
  printf("%d",r);
  return 0;
}
