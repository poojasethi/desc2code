#include <cstdio>
#include <set>
#include <algorithm>
using namespace std;
set<int> c; int n,k,a[100000];
int main (){
	scanf("%d%d", &n, &k);
	for(int i = 0; i < n; i++)
		scanf("%d", &a[i]);
	sort(a, a+n);
	for(int i = 0; i < n; i++){
		if(a[i]%k == 0 && c.count(a[i]/k)) continue;
		c.insert(a[i]);
	}
	printf("%d\n", c.size());
}
