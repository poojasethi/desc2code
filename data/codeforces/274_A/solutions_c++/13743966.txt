#include<bits/stdc++.h>
using namespace std;

int main() {
    int n,k,i;
    set<int>set;
    cin>>n>>k;
    int a[n];
    for(i=1;i<=n;i++) {
        cin>>a[i];
    }
    sort(a+1,a+n+1);
    for(i=1;i<=n;i++) {
        if(a[i]%k || !set.count(a[i]/k)) set.insert(a[i]);
    }
    cout<<set.size()<<endl;
    return 0;
}
