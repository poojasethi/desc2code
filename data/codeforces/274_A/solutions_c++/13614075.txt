#include <bits/stdc++.h>
using namespace std;

int main(){
        map<int,bool> m;
	int n;
	long long k;
        cin >> n >> k;
	int a[n];
        for(int i =0;i<n;i++) cin >> a[i];
	sort(a,a+n);
	int total = 0;
	for(int i = 0;i<n;i++){
		if (!m[a[i]]){
			if(a[i]*k<=1000000000) m[a[i]*k]=true;
			total++;
		}
	}
	cout << total;
}
