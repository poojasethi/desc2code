#include <iostream>
#include <algorithm>
#include <set>
using namespace std;
int a[100005] = {0};
set<int> s;
int main()
{
	int n,i,k;
	cin>>n>>k;
	for(i=0;i<n;i++)	cin>>a[i];
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		if(a[i]%k||!s.count(a[i]/k))
			s.insert(a[i]);
	}
	cout<<s.size()<<endl;
	return 0;
}
