#include <map>
#include <cstdio>
#include <algorithm>
using namespace std;
int a[100100];
map<int,int> s;
int main()
{
	int n,k,r=0;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++) scanf("%d",&a[i]);
	sort(a,a+n);
	for(int i=0;i<n;i++) if(a[i]%k|!s[a[i]/k]) s[a[i]]=++r;
	printf("%d\n",r);
	return 0;
}
