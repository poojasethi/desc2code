#include <set>
#include <cstdio>
#include <algorithm>

int n, k, a[100000];
int main()
{
  scanf("%d%d", &n, &k);
  for (int i = 0; i < n; ++i)
    scanf("%d", a+i);
  std::sort(a, a+n);

  std::set<int> set;
  for (int i = 0; i < n; ++i)
    if (a[i] % k || !set.count(a[i]/k))
      set.insert(a[i]);
  printf("%d\n", static_cast<int>(set.size()));
}
