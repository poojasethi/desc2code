#include<iostream>
#include<cstdio>
#include<set>
#include<algorithm>

using namespace std;
typedef long long ll;
int main()
{
	set<int> s;
	int n;
	ll k;
	cin>>n>>k;
	ll a[n];
	for(int i=0;i<n;i++)
		cin>>a[i];
	sort(a,a+n);
	for(int i=0;i<n;i++)
	{
		if(a[i]%k!= 0 || s.count(a[i]/k)==0)
			s.insert(a[i]);
	}
	cout<<s.size();
	return 0;
}
