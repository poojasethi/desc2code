#include <iostream>
#include <algorithm>
#include <set>

using namespace std;

int n,k,a[100010];
multiset<int> s;

int main() {
	cin >> n >> k;
	for (int i = 0; i < n; i++) cin >> a[i];
	sort(a,a+n);
	for (int i = 0; i < n; i++)
		if (a[i]%k || s.find(a[i]/k)==s.end()) s.insert(a[i]);
	cout << s.size();
}
