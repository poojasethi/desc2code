#include<iostream>
#include<set>
#include<algorithm>
using namespace std;
set<int>c;
int n,k,a[100005];
int main(){
    cin>>n>>k;
    for(int i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    for(int i=0;i<n;i++){
        if(a[i]%k||!c.count(a[i]/k))
            c.insert(a[i]);
    }
    cout<<c.size()<<endl;
    return 0;
}
