#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
const int N=100050;
map <int,int> flag;
int n,k,a[N],ans,i;
int main()
{
	cin>>n>>k;
	for(i=0;i<n;i++) cin>>a[i];
	ans=0;
	sort(a,a+n);
	for(i=0;i<n;i++)
		if(a[i]%k||!flag[a[i]/k]) 
			flag[a[i]]=++ans;
	cout<<ans<<endl;
	return 0;
}
