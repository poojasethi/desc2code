#!/usr/bin/env python2.7
# -*- coding: utf-8 -*-

def main():
    n, k = map(int, raw_input().split())
    a = map(int, raw_input().split())
    a.sort()
    s = set()
    cnt = 0
    for ai in a:
        if ai in s:
            continue

        s.add(k * ai)
        cnt += 1

    print cnt

if __name__ == '__main__':
    main()
