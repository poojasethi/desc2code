def main():
    n, k = map(int, raw_input().split())
    data = sorted(map(int, raw_input().split()))
    cnt = 0
    forbidden = set()
    for x in data:
        if x not in forbidden:
            cnt += 1
            forbidden.add(k * x)
    print cnt

if __name__ == '__main__':
    # import sys; sys.stdin = open('input')
    main()
