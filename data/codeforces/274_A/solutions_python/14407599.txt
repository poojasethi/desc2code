def main():
    n, k = map(int, raw_input().split())
    l = sorted(map(int, raw_input().split()))
    out = set()
    for element in l:
        if (element % k != 0):
            out.add(element)
        else:
            if (element/k not in out):
                out.add(element)
    return len(out)


print main()
