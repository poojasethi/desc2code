from sys import stdin

n , k = [ int( x ) for x in stdin.readline().strip().split() ]
a = sorted( [ int( x ) for x in stdin.readline().strip().split() ] )
had = {}
res = 0 
for el in a :
    d = int( el / k )
    if d * k == el :
        if d not in had :
            res += 1
            had[ el ] = 1
    else :
        had[ el ] = 1
        res += 1
print res
