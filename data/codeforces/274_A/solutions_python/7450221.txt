#!/usr/bin/env python2.7
# -*- coding: utf-8 -*-

def main():
    n, k = map(int, raw_input().split())
    a = map(int, raw_input().split())
    a.sort()
    s = set()
    cnt = 0
    for ai in a:
        if ai in s:
            cnt += 1
            continue

        s.add(k * ai)

    print n - cnt

if __name__ == '__main__':
    main()
