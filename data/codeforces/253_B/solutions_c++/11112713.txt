#include<iostream>
#include<cmath>
#include<algorithm>
#include<cstdio>
using namespace std;
int a[100001];
int n,l,r,ans;
int main()
{
	freopen("input.txt","rt",stdin);
	freopen("output.txt","wt",stdout);
	cin>>n;
	int i;
	for(i=1;i<=n;i++)
		cin>>a[i];
	sort(a+1,a+n+1);
	l=1,r=1;
	for(;l<=n;l++)
	{
		while(r<=n&&a[l]*2>=a[r])	r++;
			ans=max(ans,r-l);
	}
	cout<<n-ans;
	return 0;
}
