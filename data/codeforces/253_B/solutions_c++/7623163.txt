#include<bits/stdc++.h>

using namespace std;

int main() {

	int n,i,m,t,l,r;
	
	ifstream ins("input.txt");
	//cin >> n;
	ins >> n;
	int a[n];
	
	for(i=0;i<n;i++) {
		ins >> a[i];
	}
	
	sort(a, a+n);
	
	int j = 0, ans = 0;
	for(i=0;i<n;i++) {
		while(j<n && a[j]<=2*a[i]) j++;
		ans = max(ans, j-i);
	}
	
	ofstream out("output.txt");
	
	out<<n-ans<<endl;
	//cout<<n-ans;
		
	return 0;
}
