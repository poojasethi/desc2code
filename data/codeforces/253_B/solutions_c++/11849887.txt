#include <iostream>
#include <fstream>

using namespace std;

int main(){
	int n,i,j,k,x,m,a[5001];

	for(i=0;i<5001;i++)
		a[i]=0;

	ifstream pfin("input.txt");
	ofstream pfout("output.txt");

	pfin >> n;

	for(i=0;i<n;i++){
		pfin >> x;
		a[x]++;
	}

	for(i=1,m=n;i<5001;i++){
		a[i]+=a[i-1];

		j=(i-1)/2;

		x=n-a[i]+a[j];

		if(x<m)
			m=x;
	}

	pfout << m <<'\n';

	return 0;
}
