#include<iostream>
using namespace std;
#include<bits/stdc++.h>
int arr[100001];
int main(){
    ifstream in("input.txt");
    ofstream out("output.txt");
    int n;
    in>>n;
for(int i=0;i<n;i++)
{
    in>>arr[i];
}
sort(arr,arr+n);
int ans=1,p=n-1,q=n-2;
while(q>=0)
{
   if(2*arr[q]>=arr[p])
   {
       if(ans<p-q+1)
        ans=p-q+1;
       q--;
   }
   else
    p--;
}
out<<n-ans;
}
