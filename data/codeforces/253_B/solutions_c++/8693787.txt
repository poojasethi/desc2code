#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int main(){
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int n,a[100001];
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];        
    sort(a,a+n);
    int k=0,j=0;
    for(int i=0;i<n;i++){
        while(a[j]*2<a[i])
            j++;     
        k=max(k,i-j+1);
    }
    cout<<n-k;
}
