#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,i,j,m,arr[100005];
	ofstream f1("output.txt");ifstream f2("input.txt");
	f2>>n;
	for(i=0;i<n;i++){
		f2>>arr[i];
	}
	sort(arr,arr+n);
	for(i=0,j=0,m=0;j<n;){
		while(arr[j]<=2*arr[i]&&j<n){
			//cout<<"yo "<<i<<" "<<j<<endl;
			j++;
		}
		if(m<(j-i))m = j-i;
		i++;
	}
	f1<<n-m<<endl;
	f1.close();f2.close();
	return 0;
}
