#include<iostream>
#include<algorithm>
#include<fstream>
#include<stdio.h>
using namespace std;
int i, j=1, n, a[999999], ans=999999;
main(){
  /**/freopen("input.txt","r",stdin);
  freopen("output.txt","w",stdout);/**/
  cin>>n;
  for(i=1;i<=n;i++)cin>>a[i];
  sort(a+1,a+n+1);
  for(i=1;i<=n;i++){
    while(j<n && a[j+1]<=2*a[i]) j++;
    ans=min(ans,i+n-j-1);
  }
  cout<<ans<<endl;
}
