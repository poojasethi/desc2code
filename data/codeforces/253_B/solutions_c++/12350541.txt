#include<bits/stdc++.h>
using namespace std;

int main() {
	ifstream fin("input.txt");
	ofstream fout("output.txt");

	int n,c, mp[5001]; fin>>n;

	for (int i=0; i<5001; i++)
		mp[i]=0;
	for (int i=0; i<n; i++)
		fin>>c, mp[c]++;

	int ret=1e9;
	for (int i=1; i<=5000; i++) {
		int cur=0;
		for (int j=1; j<=5000; j++)
			if(j<i||j>2*i) cur+=mp[j];
		ret=min(ret,cur);
	}
	fout << ret << endl;
	return 0;
}
