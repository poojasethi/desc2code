#include<iostream>
#include<cstdio>
using namespace std;
int n,i,x,ans,S[10001];

int main()
{
freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	{
		scanf("%d",&x);
		S[x]++;
	}
	for (i=2;i<=5000;i++)	S[i]+=S[i-1];
	for (ans=n,i=1;i<=5000;i++)	ans=min(ans,n-(S[i*2]-S[i-1]));
	printf("%d\n",ans);
	fclose(stdin);
	fclose(stdout);
}
