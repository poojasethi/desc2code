#include<iostream>
#include<fstream>
#include<algorithm>
using namespace std;
int n,a[100005],i,j=1,k=100001; 
main()
{
 ifstream fin("input.txt");
 ofstream fout("output.txt");
 fin>>n;
 for(i=1;i<=n;i++)
    fin>>a[i];
 sort(a+1,a+n+1);
 for(i=1;i<=n;i++)
 {
    while(j<n && a[j+1]<=2*a[i]) 
         j++;
    k=min(k,i+n-j-1);
 }
 fout<<k<<endl;
 //system("pause");
}
