#include<iostream>
#include<fstream>
#include<algorithm>
using namespace std;
int c[100010];
int n;
int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
cin>>n;
int k=0,ans=n;
for(int i=1;i<=n;i++)
cin>>c[i];
sort(c+1,c+n+1);
int j=1;
for(int i=1;i<=n;i++)
{

    for(;c[j]<=2*c[i]&&j<=n;j++);
    k=n-j+i;
    if(ans>k)ans=k;
}
cout<<ans;


}
