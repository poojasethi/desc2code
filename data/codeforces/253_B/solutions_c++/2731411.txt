#include<iostream>
#include<fstream>
#include<algorithm>
using namespace std;
int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
int n,a[100005],i,j=1,k=100001;
cin>>n;
for(i=1;i<=n;i++) cin>>a[i];
sort(a+1,a+n+1);
for(i=1;i<=n;i++)
{
while(j<n && a[j+1]<=2*a[i]) j++;
k=min(k,i+n-j-1);
}
cout<<k;
return 0;
}
