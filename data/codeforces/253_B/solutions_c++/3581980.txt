#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
int main(){
	freopen("input.txt","r",stdin); freopen("output.txt","w",stdout);
	int a[100010],n,i,s=0,k=0;
	cin>>n;
	for (i=0; i<n; i++)
		cin>>a[i];
	sort(a,a+n);
	for (i=0; i<n; i++){
		while (s<n && a[s]<=2*a[i])s++;
		k=max(k,s-i);
	}
	cout<<n-k;
	return 0;
}
