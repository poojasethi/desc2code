#include<stdio.h>
#include<algorithm>
using namespace std;
int a[200005],ans,n,mn,i,j;
int main()
{

    freopen("input.txt","r",stdin); freopen("output.txt","w",stdout);
    scanf("%d",&n);
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    sort(a,a+n);
    for(i=0,j=0;i<n;i++)
    {
       while(j<n && a[j]<=a[i]*2) j++;
       ans=max(ans,j-i);
    }
    printf("%d",n-ans);
}
