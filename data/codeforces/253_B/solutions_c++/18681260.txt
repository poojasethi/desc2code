#include <bits/stdc++.h>

using namespace std;

#define maxn 100000+10

int n,c[maxn];

int main(void)
{
	freopen("input.txt","rb",stdin);
    freopen("output.txt","wb",stdout);
	cin>>n;

	for(int i=1;i<=n;i++)
	{
		cin>>c[i];
	}
	sort(c+1,c+n+1);
	int j=1,t=100000000;
	for(int i=1;i<=n;i++)
	{
		while(j<n && c[j+1]<=2*c[i]) j++;
		t = min(t,n-j+i-1);
	}
	cout<<t<<endl;
	return 0;
}
