#include <iostream>
#include <algorithm>
#include <stdio.h>
using namespace std;
int main()
{
int n,a[100005];
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
cin>>n;
for (int i=0;i<n;i++) cin>>a[i];
sort(a,a+n);
int m=100005;
for (int i=0,j=0;i<n;i++)
 {
  if (2*a[i]>=a[n-1]) {m=min(i,m); break;}
  while (2*a[i]>=a[j]&&j<n) j++;
   m=min(m,i+n-j);
 }
cout<<m;
return 0;
}
