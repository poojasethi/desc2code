#include<iostream>
#include<fstream>
#include<algorithm>
#include<climits>
using namespace std;
int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
int n,i,j=1,k=INT_MAX;
cin>>n;
int arr[n+1];
for(i=1;i<=n;i++) cin>>arr[i];
sort(arr+1,arr+n+1);
for(i=1;i<=n;i++)
{
while(j<n && arr[j+1]<=2*arr[i]) j++;
k=min(k,i+n-j-1);
}
cout<<k;
return 0;
}
