#include<iostream>
#include<fstream>
#include<algorithm>
using namespace std;
int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
int n;
int a[100005];
int i;
int j=1;
int q=100001;
cin>>n;

for(int w2=1;w2<=n;w2++){
 cin>>a[w2];
}
sort(a+1,a+n+1);
for(int w2=1;w2<=n;w2++)
{
while(j<n && a[j+1]<=2*a[w2]){
j++;
}
q=min(q,w2+n-j-1);
}
cout<<q;
return 0;
}
