#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int a[500009];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	sort(a+1,a+1+n);
	int ans=1e9;
	for(int i=1;i<=n-1;i++)
	{
		int p=upper_bound(a+1,a+1+n,2*a[i])-a;
        ans=min(ans,i-1+n+1-p);
	}
	cout << ans <<endl;
	return 0;
}
