#include<bits/stdc++.h>
using namespace std;

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int N;
	cin >> N;
	int A[N];
	int x = 0, y= 0, M= 0;
	for(int i = 0 ; i < N; i++)
		cin >> A[i];
	sort(A,A+N);

	while(y < N && x < N)
	{
		if(A[y] <= A[x]*2)
		{
			y++;
			M = max(M,y-x);
		}
		else
			x++;
	}

	cout << N-M << endl;
	return 0;
}
