#include <cstdio>
int cnt[5555];
int n, i, j, k, l, m;
int main(){
  freopen("input.txt", "r", stdin); freopen("output.txt", "w", stdout);
  scanf("%d\n", &n);
  for (i = 1; i <= n; i++) scanf("%d", &j), cnt[j]++;
  for (i = 1; i <= 5000; i++) cnt[i] += cnt[i - 1];
  k = 0;
  for (i = 1; i <= 2500; i++) if (cnt[i + i] - cnt[i - 1] > k) k = cnt[i + i] - cnt[i - 1];
  printf("%d", n - k);
  return 0;
}
