#include<cstdio>
#include<algorithm>
using namespace std;
int n, a[100010], ans = 100010;
int main(){
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	scanf("%d", &n);
	for(int i = 0; i < n; i++)
		scanf("%d", &a[i]);
	sort(a, a+n);
	for(int i = 0, j = 0; i < n; i++){
		while(j < n && a[i]*2 >= a[j]) j++;
		ans = min(n-j+i, ans);
	}
	printf("%d\n", ans);
}
