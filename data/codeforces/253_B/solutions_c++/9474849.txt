#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;
int n,a[100001],ans = 100001,i,j;
int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	cin >> n ;
	for (i = 0; i < n; ++i)
	{
		cin >> a[i];
	}
	sort(a, a + n);
	for (i=0,j=0;i<n; i++){
		while(j<n && a[i]*2 >= a[j]) j++;
		ans = min(ans, n-j+i);
	}
	cout << ans;

}
