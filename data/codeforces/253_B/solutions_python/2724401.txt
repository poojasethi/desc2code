import sys
from bisect import bisect_right

sys.stdin = open( 'input.txt', 'r' )
sys.stdout = open( 'output.txt', 'w' )

n = int( raw_input() )
c = map( int, raw_input().split(' ') )
c.sort()
r = n - 1
#print c
for i in range(n-1):
	j = bisect_right(c, 2*c[i])
	#print i, j - 1
	r = min( r, n - (j - i) )
print r
