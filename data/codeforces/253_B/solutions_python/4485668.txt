infile = open("input.txt","r")
outfile = open("output.txt","w")

n = int(infile.readline())
c = map(int, infile.readline().split())
c.sort()

min_diff = n
end = 0

for start in range(n):
  target = 2*c[start]
  if target > c[-1]:
    min_diff = min(min_diff, start)
    break

  while end < n-1 and not (c[end] <= target < c[end+1]):
    end += 1
  min_diff = min(min_diff, start + n - 1 - end)

outfile.write('%d'%min_diff)
infile.close()
outfile.close()
