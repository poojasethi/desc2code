import sys
from bisect import bisect_right

sys.stdin=open("input.txt", "r")
sys.stdout=open("output.txt", "w")

n = int(raw_input())
cs = map(int, raw_input().split())
cs.sort()

minn = int(10e9)
for i in range(len(cs)):
    u = cs[i]
    v = 2*u
    j = bisect_right(cs, v)
    total = i + (len(cs)-j)
    if total < minn:
        minn = total

print minn
