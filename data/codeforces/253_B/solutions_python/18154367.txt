import sys

def main():
	sys.stdin = open("input.txt", "r")
	sys.stdout = open("output.txt", "w")
	n = input()
	a = map(int, raw_input().split())
	s = [0] * 5001
	for x in a: s[x] += 1
	for i in xrange(1, 5001): s[i] += s[i - 1]
	ans = n
	for i in xrange(1, 5001):
		if 2 * i >= 5000: ans = min(ans, s[i - 1])
		else: ans = min(ans, s[i - 1] + n - s[2 * i])
	print ans
	
if __name__ == "__main__": main()
