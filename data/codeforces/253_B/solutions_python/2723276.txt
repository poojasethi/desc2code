import sys
sys.stdin = open('input.txt','r')
sys.stdout=open('output.txt','w')
from bisect import bisect_right
n = int(input())
c = map(int,raw_input().split())
c.sort()
if c[-1]<=c[0]*2:
    print(0)
    exit()
ans =10**19
for a in range(n):
    b = bisect_right(c,2*c[a])
    ans=min(ans,n-(b-a))
print(ans)
