import bisect
import sys

def main():
    f = open('input.txt')
    sys.stdout = open('output.txt','w')
    n = int(f.readline())
    c = sorted((map(int, f.readline().strip('\n').split())))

    if c[-1] <= c[0]*2:
        print(0)
        exit()
    res = 100001
    for i in range(n):
#        if c[i] > (c[-1]/2): break
        res = min(res,n - bisect.bisect(c,c[i]*2) + i)
    print(res)

main()
