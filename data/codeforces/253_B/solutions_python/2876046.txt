with open('input.txt') as d:
  d.readline()
  numbers = map(int, d.readline().split())

numbers.sort()
start = 0
end = 0
res = 0

while True:
  while end < len(numbers):
    if numbers[end] <= numbers[start] * 2:
      end += 1
    else:
      break

  res = max(res, end - start)
  start += 1
  if end == len(numbers) - 1 or start == len(numbers) - 1:
    break


with open('output.txt', 'w') as d:
  d.write(str(min(len(numbers) - 1, (len(numbers) - res))))
