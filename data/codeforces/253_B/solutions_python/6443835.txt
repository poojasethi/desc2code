inp = open('input.txt', 'r')
out = open('output.txt', 'w')
n = int(inp.readline())
l = [0]*5001
cum = [0]*5001
mi = 0
for x in inp.readline().split():
    x = int(x)
    l[x] += 1
    if x > mi: mi = x
for i in range(1,5001):
    cum[i] = cum[i-1] + l[i]

minx = min(cum[x/2-1+x%2] + cum[mi] - cum[x] for x in range(mi,0,-1))
out.write("%d" %(minx))
inp.close()
out.close()
