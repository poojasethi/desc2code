import sys
sys.stdin = open("input.txt","r")
sys.stdout = open("output.txt","w")
n = int(raw_input())
c = map(int, raw_input().split())
c.sort()
c_double = []
for i in xrange(n):
    c_double.append(c[i] * 2)
i = 0
j = 0
max_ = 0
while i < n and j < n:
    while j < n and c[j] <= c_double[i]:
        j += 1
    max_ = max(max_, j - i)
    i += 1
print n - max_
