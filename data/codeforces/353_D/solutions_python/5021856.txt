queue = raw_input()

cntF = 0
cntM = 0
preCost = 0

for i in range(len(queue)) :
    
    if queue[i] == 'M' :
        continue
    else :
        dist = i - cntF
        cntF += 1
        if preCost == 0 :
            preCost = dist
        else :
            preCost = max(dist, preCost + 1)

    
print preCost
