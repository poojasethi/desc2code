#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int main()
{
    int ans = 0, m = 0;
    char x;
    while(x = getchar(), (x == 'M' || x == 'F'))
    {
        if(x == 'M') m++;
        else if(m != 0) ans = max(ans + 1, m);
    }
    printf("%d\n", ans);
    return 0;
}
