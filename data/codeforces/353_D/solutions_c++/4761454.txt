#include<cstdio>
#include<algorithm>
#define N 1000000
using namespace std;
char b[N+1];
int main()
{
	scanf("%s",b);
	int c=0;

	int i=0;
	while( b[i]=='F' )
	{
		++c,++i;
	}
	int lt = 0;

	for(;b[i];++i)
	{
		if( b[i]=='F' )
		{
			lt = max(lt+1,i-c);
			++c;
		}
	}
	printf("%d\n",lt);
	return 0;
}
