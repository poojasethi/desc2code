#include <iostream> 
#include <cstring>
 using namespace std; int main() { string str=""; cin >> str; int tmp=0, res=0, M=0, F=0; while(str[tmp]=='F' && tmp<str.length()) tmp++; for(int i=tmp; i<str.length(); i++) if(str[i]=='F') { F++; res = max(res, M+F-1); } else { M++; F = max(0, F-1); } cout << res; }
