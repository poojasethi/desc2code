#include<iostream>
#include<string>
#include<algorithm>

using namespace std;

int main() {
	string s;
	cin >> s;
	int d = 0, o = 0, t = 0;
	for (int i = s.size() - 1; i >= 0; i--) {
		if (s[i] == 'M') {
			t = max(t, d + o);
			if (d != 0) o++;
		} else {
			o--;
			o = max(0, o);
			d++;
		}
	}
	cout << t << endl;
	return 0;
}
