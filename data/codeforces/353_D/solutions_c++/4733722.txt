#include <stdio.h>
#include <string.h>

int n, count;
char a[1230997];

main(){
    int i, x=0, answer=0;
    scanf("%s", a+1);
    n = strlen(a+1);
    for (i=1; i<=n; i++){
        if (a[i]=='M') x--;
        else { count++; answer=i-count+x; if (i!=count) x++; }
        if (x<0) x=0;
    }
    printf("%d\n", answer);
}
