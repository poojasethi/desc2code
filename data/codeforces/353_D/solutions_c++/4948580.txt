#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;

const int N=1000010;

char s[N];
int cnt,ans;

int main()
{
    scanf("%s",s);
    int len=strlen(s);
    for(int i=0;i<len;++i)
        if(s[i]=='M') ++cnt;
        else if(cnt) ans=max(ans+1,cnt);
    printf("%d\n",ans);
}
