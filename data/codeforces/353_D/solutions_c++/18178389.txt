#include <bits/stdc++.h>

using namespace std;

int main()
{
    string str;
    int i=0,j,k,f=0,ans=0;

    cin>>str;
    int l=str.length();
    while(i<l && str[i]=='F')
        i++;

    f=i;

    for(;i<l;i++)
    {
        if(str[i]=='F')
        {ans=max(ans+1,i-f);f++;};
    }
    cout<<ans<<endl;

}
