#include<cstdio>
using namespace std;
char s;
int main()
{
    int r=0,f=1,l=0;
    while(getchar()=='F');
    while((s=getchar())=='F'||s=='M')
    {
        if(s=='F')
        {
            if(f-l>r)r=f-l;
            else r++;
            l++;
        }
        f++;
    }
    printf("%d\n",r);
    return 0;
}
