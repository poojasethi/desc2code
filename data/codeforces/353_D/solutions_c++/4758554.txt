#include <iostream>
#include <string>
using namespace std;

int main()
{
	string line;
	cin >> line;
	int ans = 0;
	int nb = 0;
	for (int i = 0; i < int(line.size()); i++) {
		if (line[i] == 'M') {
			nb += 1;
			continue;
		}
		if (nb > 0) {
			ans = max(ans+1, nb);
		}
	}
	cout << ans << '\n';
}
