#include<cstdlib>
#include<cstdio>
using namespace std;

int n,m,p,f;
char ch;
bool b[1000001];

int main()
{
	while ((ch=getchar())!='\n') if (ch=='M') b[++n]=true;else p=++n;
	for (int i=1;i<=p;i++)
	if (b[i]) {if (--f<0) f=-1;}
	else
	{
		if (++m==i) f=0;
		else f++;
	}
	printf("%d",p-m+f);
	return 0;
}
