#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char s[1000010];

int main()
{
	scanf("%s",s);
	int ans = 0,tmp = 0;
	int tot = 0;
	for(int i = 0; s[i]; i ++) {
		if(s[i] == 'M') tot ++;
		else if(tot){
			tmp = max(tmp + 1,tot);
			ans = max(ans,tmp);
		}
	}
	cout << ans << endl;
}
