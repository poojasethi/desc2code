#include <bits/stdc++.h>

using namespace std;
const int maxn = 1e6+5;
string s;
int main(){
	while (cin >> s){
		int ans = 0, mm = 0;
		for (int i = 0; i < s.size(); ++i){
			if (s[i] == 'M'){
				mm++;
			}
			else{
				if (mm) ans = max(ans + 1, mm);
			}
		}
		cout << ans << endl;
	}
	return 0;
}
