#include<cstdio>
#include<cstring>
#define Max(a,b) ((a)>(b)?(a):(b))

const int Maxn=1e6+5;

int sum[Maxn],l,cnt=0,ans=0;
char s[Maxn];

int main()
{
	scanf("%s",s);
	l=strlen(s);
	for(int i=0;i<l;i++)
	  if(s[i]=='M')
		cnt++;
	  else
		if(cnt>0)
		 ans=Max(ans+1,cnt);
	printf("%d\n",ans);
	return 0;
}
