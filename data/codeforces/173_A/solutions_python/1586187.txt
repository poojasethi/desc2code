def cal(a,b):
    if a == b:
        return 0
    if a == "R" and b == "S" or a == "S" and b == "P" or a == "P" and b == "R":
        return 1
    return 2
n = int(raw_input())
a = raw_input().strip()
b = raw_input().strip()
la = len(a)
lb = len(b)
ans = [0 for i in range(0,3)]
for i in range(0,la*lb):
    ans[cal(a[i%la],b[i%lb])] += 1
for i in range(0,3):
	ans[i] *= n / (la * lb)
for i in range(0,n % (la * lb)):
    ans[cal(a[i%la],b[i%lb])] += 1
print ans[2],ans[1]
