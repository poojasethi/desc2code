n=input()
a,b=raw_input(),raw_input()
p,q=len(a),len(b)
k=p*q
u,v=[0]*(k+1),[0]*(k+1)
for i in range(k):
  c1,c2=a[i%p],b[i%q]
  if c1!=c2:
    if ('R'==c1 and 'S'==c2) or ('S'==c1 and 'P'==c2) or ('P'==c1 and 'R'==c2):
      v[i+1]=1
    else:
      u[i+1]=1
for i in range(k):
  u[i+1]+=u[i]
  v[i+1]+=v[i]
print u[k]*(n/k)+u[n%k],v[k]*(n/k)+v[n%k]
