#include <cstdio>
#include <cstring>

int pa[1000005], pb[1000005];

int win(char a, char b) {
	if (a == b) return 0;
	return a=='R'&&b=='S'||a=='S'&&b=='P'||a=='P'&&b=='R' ? 1 : -1;
}

int main() {
    int n, m, la, lb;
    char a[1005], b[1005];
    scanf("%d%s%s", &n, a, b);
    la = strlen(a);
    lb = strlen(b);
    m = la * lb;
    for (int i=1, j=0, k=0; i <= m; ++i, j=(j+1)%la, k=(k+1)%lb) {
        int f = win(a[j], b[k]);
        pa[i] = (f == -1) + pa[i-1];
        pb[i] = (f ==  1) + pb[i-1];
    }
    int j = n / m;
    int k = n % m;
    int suma = j*pa[m]+pa[k];
    int sumb = j*pb[m]+pb[k];
    printf("%d %d\n", suma, sumb);
}
