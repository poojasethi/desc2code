#include<cstdio>
#include<cstring>

int x[1100000],y[1100000],n,L1,L2,i;

char P[1100],Q[1100];

int main()
{	//freopen("a.in","rb",stdin);
	//freopen("a.out","wb",stdout);
	scanf("%d",&n);
	scanf("%s%s",P,Q);
	L1=strlen(P);
	L2=strlen(Q);
	for(i=0;i<L1*L2;i++)
	{	x[i+1]=x[i];
		y[i+1]=y[i];
		if(P[i%L1]!=Q[i%L2])
		{	if(P[i%L1]=='R'&&Q[i%L2]=='S'||
               P[i%L1]=='S'&&Q[i%L2]=='P'||
               P[i%L1]=='P'&&Q[i%L2]=='R')y[i+1]++;
				else x[i+1]++;
		}
	}
	printf("%d %d\n",x[n%(L1*L2)]+x[L1*L2]*(n/(L1*L2)),y[n%(L1*L2)]+y[L1*L2]*(n/(L1*L2)));
}
