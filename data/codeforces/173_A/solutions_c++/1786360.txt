#include <cstdio>
#include <string>
#include <iostream>
using namespace std;

int main()
{
	int n,x,y,i,ans[2]={0};
	string a,b;
	cin>>n>>a>>b;
	x=a.size();
	y=b.size();
	for (i=0; i<n && i<x*y; i++){
		char p=a[i%x],r=b[i%y];
		if (p!=r){
			int d=(n-i+x*y-1)/(x*y);
			if ((p=='R'&&r=='S') || (p=='S'&&r=='P') || (p=='P'&&r=='R'))
				ans[1]+=d;
			else ans[0]+=d;
		}
	}
	cout<<ans[0]<<" "<<ans[1]<<endl;
	return 0;
}
