#include <cstdio>
#include <cstring>

int n, l1, l2;
char s1[1001], s2[1001];
int r1[1000001], r2[1000001];

int main() {
	scanf("%d%s%s", &n, s1, s2);
	l1 = strlen(s1);
	l2 = strlen(s2);
	r1[0] = r2[0];
	for (int i = 0; i < l1 * l2; ++i) {
		r1[i + 1] = r1[i];
		r2[i + 1] = r2[i];
		char c1 = s1[i % l1];
		char c2 = s2[i % l2];
		if (c1 != c2) {
			if (c1 == 'R' && c2 == 'P' || c1 == 'P' && c2 == 'S' || c1 == 'S' && c2 == 'R') {
				++r1[i + 1];
			} else {
				++r2[i + 1];
			}
		}
	}
	int t = l1 * l2;
	printf("%d %d\n", n / t * r1[t] + r1[n % t], n / t * r2[t] + r2[n % t]);
	return 0;
}
