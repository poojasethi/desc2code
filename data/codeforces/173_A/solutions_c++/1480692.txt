#include <cstdlib>
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <utility>

using namespace std;

#define long long long

long n, m, k, z, t1, t2, l1, l2;
string a, b;

bool loses(char x, char y)
{
	return (x=='P' && y=='S' || x=='S' && y=='R' || x=='R' && y=='P');
}

int main()
{
	cin >> n >> a >> b;
	m = a.size();
	k = b.size();
	z = m*k;
	for (long i=0; i<z; i++)
	{
		if (i==n%z)
		{
			t1=l1;
			t2=l2;
		}
		if (loses(a[i%m], b[i%k])) l1++;
		if (loses(b[i%k], a[i%m])) l2++;
	}
	t1 += n/z*l1;
	t2 += n/z*l2;
	cout << t1 << ' ' << t2 << endl;
	return 0;
}
