#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
#define num 1005
int c[num];
bool f(char g,char h){
	if (g=='R')
		return h=='S'?1:0;
	if (g=='P')
		return h=='R'?1:0;
	if (g=='S')
		return h=='P'?1:0;
}
int main ()
{
//	freopen("test.txt","r",stdin);
	int n,x=0,y=0,ansx=0,ansy=0;
	string a,b,temp;
	cin>>n;
	cin>>a>>b;
	x=a.size(),y=b.size();
	for (int i=0,d;i<n && i<x*y;i++){
		if (a[i%x]==b[i%y])continue;
		d = (n-i+x*y-1)/(x*y);
		if (f(a[i%x],b[i%y]))
			ansx+=d;
		else
			ansy+=d;
	}
	printf("%d %d",ansy,ansx);
    return 0;
}
