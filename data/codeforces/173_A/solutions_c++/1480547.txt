#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
using namespace std;

int sx[1000001]={0};
int sy[1000001]={0};
int main(void){
	int n;
	char c1[1200], c2[1200];
	scanf("%d%s%s",&n,c1,c2);
	int a=strlen(c1);
	int b=strlen(c2);
	
	int x=0,y=0;
	for(int i=0;i<n && i<a*b;i++){
		char m = c1[i%a];
		char k = c2[i%b];
		if(m == 'R') k=='R'?0:k=='P'?x++:y++;
		if(m == 'P') k=='P'?0:k=='S'?x++:y++;
		if(m == 'S') k=='S'?0:k=='R'?x++:y++;
		sx[i]=x;sy[i]=y;
	}
	if(n>a*b){
		int d=n/(a*b);
		x*=d;y*=d;
		n-=d*a*b;
		if(n>0){
			x+=sx[n-1];y+=sy[n-1];
		}
	}
	printf("%d %d\n",x,y);
	return 0;
}
