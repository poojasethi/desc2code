#include<iostream>
#include<cstdio>
#include<string>
#include<cstdlib>
#include<cmath>
#include<algorithm>

using namespace std;

string str,str2;

long long int x2,y2,i,j,u,u2,n,yac,yac2,yac3;

int ekok(int n,int m){
	return n*m/__gcd(n,m);
}

int main()
{
	cin>>n;
	cin>>str>>str2;
	u=str.size(); u2=str2.size();
	int t=ekok(u,u2);
	for(i=0,j=0,yac3=1; yac3<=t ; yac3++)
	{
		if(str[i]=='S' and str2[j]=='P') 
			yac2++;
		else if(str[i]=='R' and str2[j]=='S')
			yac2++;
		else if(str[i]=='P' and str2[j]=='R')
			yac2++;
		else if(str[i]!=str2[j]) yac++;

		if(yac3==n%t) x2=yac,y2=yac2;
		i++; j++;
		i%=u;
		j%=u2;
	}
	cout<<yac*(n/t)+x2<<' '<<yac2*(n/t)+y2;
	return 0;
}
