#include <cstdio>
#include <iostream>
#include <cstring>
using namespace std;
char a[1005],b[1005];
int s[3],r[3];
int play (char x,char y) {
	if (x=='R' && y=='S') return 1;
	if (x=='S' && y=='P') return 1;
	if (x=='P' && y=='R') return 1;
	if (x==y) return 2;
	return 0;
}
int main ( ) {
	int n,i,la,lb;
	scanf("%d%s%s",&n,a,b);
	la=strlen(a);lb=strlen(b);
	for (i=0;i<la*lb;i++) r[play(a[i%la],b[i%lb])]++;
	for (i=0;i<n%(la*lb);i++) s[play(a[i%la],b[i%lb])]++;
	for (i=0;i<2;i++) s[i]+=r[i]*(n/(la*lb));
	printf("%d %d\n",s[0],s[1]);
	return 0;
}
