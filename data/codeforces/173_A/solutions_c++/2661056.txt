#include<iostream>
#include<string>
using namespace std;
int gcd(int a, int b)
{
    if(b == 0)
        return a;
    else
        return gcd(b,a % b);
}


int main()
{string a,b;int x=0,y=0,a1,b1,b2,a2,i,n;
cin>>n;
cin>>a>>b;
a1=a.length ();
b1=b.length ();
int gb=(a1 * b1) / gcd(a1,b1);
	int times=n/gb;
	int rest=n%gb;
	int t1,t2=0;
	

for(i=0;i<gb;i++)
{a2=a[i%a1];b2=b[i%b1];
if(i==rest){t1=x;t2=y;}
if(a2==b2);
else if((a2=='R'&&b2=='S')||(a2=='S'&&b2=='P')||(a2=='P'&&b2=='R'))y++;
else x++;
}
x=x*times+t1;y=y*times+t2;
cout<<x<<" "<<y<<endl;
}
