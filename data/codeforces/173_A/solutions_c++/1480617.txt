#include<cstdio>
#include<cstring>
char a[1024],b[1024];
int F[1<<20],G[1<<20],i,j,k,u,n,m,h;
int main(){
	scanf("%d ",&u);
	n=strlen(gets(a));
	m=strlen(gets(b));
	for(i=j=k=0,h=n*m;k<h;++k){
		F[k+1]=F[k];
		G[k+1]=G[k];
		if(a[i]!=b[j]){
		if(a[i]=='S')
			(b[j]=='R')?++G[k+1]:++F[k+1];
		if(a[i]=='R')
			(b[j]=='P')?++G[k+1]:++F[k+1];
		if(a[i]=='P')
			(b[j]=='S')?++G[k+1]:++F[k+1];
		}
		if(++i==n)i=0;
		if(++j==m)j=0;
	}
	printf("%d %d\n",G[h]*(u/h)+G[u%h],F[h]*(u/h)+F[u%h]);
	return 0;
}
