#include <iostream>
#include <string>
using namespace std;
int p1,p2,n;
string s1,s2;
int main(){
    cin>>n>>s1>>s2;
    int x=s1.size();
    int y=s2.size();
    for(int i=0;i<n and i<x*y;i++){
        char f=s1[i%x], s=s2[i%y];
        if(f!=s){
            int d=(n-i+x*y-1)/(x*y);
            if((f=='R' and s=='S') or (f=='S' and s=='P') or (f=='P' and s=='R'))
                p2+=d;
            else 
                p1+=d;
        }
    }
    cout<<p1<<' '<<p2;
}
