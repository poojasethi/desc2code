#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int c[1000010],d[1000010],t,n,m,i,j,k,ans;
char a[1010],b[1010];
int main()
{
	cin>>t; t--;
	scanf("%s%s",a,b);
	n=strlen(a),m=strlen(b);
	for(i=0;i<n*m;i++)
	{
		j=i%n,k=i%m;
		if(a[j]!=b[k]) d[i]=1;
		if(a[j]=='R'&&b[k]=='S'||a[j]=='S'&&b[k]=='P'||a[j]=='P'&&b[k]=='R') c[i]=1;
		if(i) c[i]+=c[i-1];
		if(i) d[i]+=d[i-1];
	}
	ans=c[n*m-1]*(t/(n*m))+c[t%(n*m)];
	t=d[n*m-1]*(t/(n*m))+d[t%(n*m)];
	cout<<t-ans<<' '<<ans<<endl;
	return 0;
}
