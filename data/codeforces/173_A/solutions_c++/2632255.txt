#include<iostream>
#include<string>
using namespace std;
int play(char a,char b)
{
	if(a==b) return 0;
	if( (a=='R'&&b=='S')||(a=='S'&&b=='P')||(a=='P'&&b=='R'))return 1;
	else return -1;
}
int main()
{
	int n;
	string s1,s2;
	while(cin>>n>>s1>>s2)
	{
		int m=s1.length()*s2.length();
		int k=n%m;
		int i;
		int l1=s1.length(),l2=s2.length();
		int a=0,b=0,a0=0,b0=0;
		int tmp;
		for(i=0;i<m;i++)
		{
			tmp=play(s1[i%l1],s2[i%l2]);
			if(tmp>0)a++;
			else if(tmp<0)b++;
			if(i==k-1){a0=a;b0=b;}
		}
		a0+=n/m*a;
		b0+=n/m*b;
		cout<<b0<<" "<<a0<<endl;
	}
	return 0;
}
