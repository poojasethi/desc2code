#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int n,a,b,yoo1,yoo2;

int re[200][200];
string S,T;
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("A.in","rb",stdin);
	freopen("A.out","wb",stdout);
	#endif
	cin>>n;
	cin>>S>>T;
	int a=S.length(),b=T.length();
	re['R']['P']=1;
	re['P']['S']=1;
	re['S']['R']=1;
	if(n>=a*b)
	for(int i=0;i<a*b;++i)
	{
		yoo1+=re[S[i%a]][T[i%b]];
		yoo2+=re[T[i%b]][S[i%a]];
	}
	yoo1*=n/(a*b);
	yoo2*=n/(a*b);
	for(int i=0;i<n%(a*b);++i)
	{
		yoo1+=re[S[i%a]][T[i%b]];
		yoo2+=re[T[i%b]][S[i%a]];
	}
	cout<<yoo1<<' '<<yoo2<<endl;
	return 0;
}
