#include <cstdio>
#include <cstring>
#include <algorithm>
char s[1001],s1[1001],c,c1;
int main(){
    int n,sl,s1l,ni=0,p=0,d,k;
    scanf("%d",&n);
    scanf(" %s",s);
    scanf(" %s",s1);
    sl=strlen(s);
    s1l=strlen(s1);
    k=sl*s1l;
    k=std::min(k,n);
    for(int i=0;i<k;i++){
        c=s[i%sl],c1=s1[i%s1l];
        d=(n-i+sl*s1l-1)/(sl*s1l);
        if((c=='R'&&c1=='S')||(c=='P'&&c1=='R')||(c=='S'&&c1=='P'))
            ni+=d;
        else if ((c=='R'&&c1=='P')||(c=='P'&&c1=='S')||(c=='S'&&c1=='R'))
            p+=d;
    }
    printf("%d %d",p,ni);
   return 0;
}
