#include <stdio.h>
#include <iostream>
#include <string>
using namespace std;
int p,q=0,w=0,h=0,l=0;
string x="",y="",n,m;
int main( ){
	cin>>p>>x>>y;
	for (int i=1;i<=y.length();i++)
		n+=x;
	for (int i=1;i<=x.length();i++)
		m+=y;
	for (int i=0;i<n.length();i++)
		if (n[i]!=m[i]){
			if ((n[i]=='R'&&m[i]=='S') || (n[i]=='S'&&m[i]=='P') || (n[i]=='P'&&m[i]=='R'))
				q++;
			else w++;
		}
	for (int i=0;i<p%n.length();i++)
		if (n[i]!=m[i]){
			if ((n[i]=='R'&&m[i]=='S') || (n[i]=='S'&&m[i]=='P') || (n[i]=='P'&&m[i]=='R'))
				h++;
			else l++;
		}
	cout<<l+(int)(p/n.length())*w<<' '<<h+(int)(p/n.length())*q<<endl;
	return 0;
}
