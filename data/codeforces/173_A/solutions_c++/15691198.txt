#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
#include <cstdio>
#include <cstring>
#include <map>
using namespace std;
int ans1,ans2;
bool lose(char a,char b)
{
	return(a=='R'&&b=='P'||a=='P'&&b=='S'||a=='S'&&b=='R');
} 
int main()
{
	int n;
	cin>>n;
	string a,b;
	cin>>a>>b;
	int len=a.length()*b.length();
	for(int i=0;i<len&&i<n;i++)
	{
	int t=1+(n-i-1)/len;
	ans1+=lose(a[i%a.length()],b[i%b.length()])*t;
	ans2+=lose(b[i%b.length()],a[i%a.length()])*t;
	}
	cout<<ans1<<" "<<ans2;
	return 0;
}
