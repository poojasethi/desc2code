#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
	int n,c1=0,c2=0,i,m,k,d;
    char a,b;
    string s,t;
    cin>>n>>s>>t;
    m=s.size();
    k=t.size();
    for(i=0;i<n&&i<m*k;++i)
    {
		a=s[i%m];
        b=t[i%k];
        if(a!=b)
        {
			d=(n-i+m*k-1)/(m*k);
            if(a=='P'&&b=='S'||a=='R'&&b=='P'||a=='S'&&b=='R')
				c1+=d;
			else
				c2+=d;
      }      
   }
   cout<<c1<<" "<<c2<<endl;
   return 0;
}
