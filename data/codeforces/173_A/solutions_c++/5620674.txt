#include<cstdio>
#include<string>
#include<iostream>
using namespace std;
int n,x,y,i,S[2],d;
main(){
    string a,b;
    cin>>n>>a>>b;
    x=a.size();
    y=b.size();
    for(i=0;i<n&&i<x*y;i++){
        char p=a[i%x],r=b[i%y];
        if(p!=r){
            d=(n-i+x*y-1)/(x*y);
            if((p=='R'&&r=='S')||(p=='S'&&r=='P')||(p=='P'&&r=='R'))S[1]+=d;
            else S[0]+=d;
        }
    }
    cout<<S[0]<<" "<<S[1];
}
