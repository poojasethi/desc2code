#include <vector>
#include <cstdio>
#include <iostream>
#include <cstring>
using namespace std;
#define REP(i,n) for(int i=0;i<(n);i++)

char a[1001],b[1001];
bool win[255][255];
int resa[1001*1001];
int resb[1001*1001];

int main()
{
	int n;
	int m,k;
	scanf("%d%s%s",&n,a,b);
	m=strlen(a);
	k=strlen(b);
	win['R']['S']=1;
	win['S']['P']=1;
	win['P']['R']=1;
	REP(i,m*k)
	{
		resb[i+1]=resb[i]+win[a[i%m]][b[i%k]];
		resa[i+1]=resa[i]+win[b[i%k]][a[i%m]];
	}
	printf("%d %d\n",n/(m*k)*resa[m*k]+resa[n%(m*k)],n/(m*k)*resb[m*k]+resb[n%(m*k)]);
	return 0;
}
