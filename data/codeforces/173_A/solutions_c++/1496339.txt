#include <iostream>
#include <string>

using namespace std;

int N,ni[1000010],po[1000010];
string A,B;

void solve()
{
	int i=0,j=0,cnt=1,M=A.size(),K=B.size();
	for (;;cnt++)
	{
		ni[cnt]=ni[cnt-1];
		po[cnt]=po[cnt-1];
		if ((A[i]=='R'&&B[j]=='S')||(A[i]=='S'&&B[j]=='P')||(A[i]=='P'&&B[j]=='R'))
			po[cnt]++;
		else if ((B[j]=='R'&&A[i]=='S')||(B[j]=='S'&&A[i]=='P')||(B[j]=='P'&&A[i]=='R'))
			ni[cnt]++;
		if (cnt==N)
		{
			cout << ni[cnt] << " " << po[cnt] << endl;
			return;
		}
		i=(++i)%M;
		j=(++j)%K;
		if (!i&&!j)
			break;
	}
	cout << N/cnt*ni[cnt]+ni[N%cnt] << " " << N/cnt*po[cnt]+po[N%cnt] << endl;
}

int main()
{
	cin >> N >> A >> B;
	solve();
	return 0;
}
