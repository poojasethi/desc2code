#include <stdio.h>
#include <string.h>

int a,b,n,i,j,a2,b2;
char a1[1000009];
char b1[1000009];

int bet(char a3,char b3)
{
	if((a3=='R' && b3=='P') || (a3=='P' && b3=='S') || (a3=='S' && b3=='R'))	return a2++;
	
	if((b3=='R' && a3=='P') || (b3=='P' && a3=='S') || (b3=='S' && a3=='R'))	return b2++;
	
	return 0;
}

int main()
{
	scanf("%d\n%s\n%s",&n,a1,b1);
	
	a=strlen(a1);
	b=strlen(b1);
	
	for(i=a;i<b*a;i++)
	a1[i]=a1[i%a];
	
	for(i=b;i<a*b;i++)
	b1[i]=b1[i%b];
	
	for(i=0;i<a*b;i++)
	bet(a1[i],b1[i]);
	
	a2=a2*(n/(a*b));	b2=b2*(n/(a*b));
	
	for(i=0;i<n%(a*b);i++)
	bet(a1[i],b1[i]);
	
	printf("%d %d\n",a2,b2);
}
