#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

char A[1005], B[1005];
int n;
int win(char x, char y) {
	if(x==y) return 2;
	if(x=='R' && y=='S') return 0;
	if(x=='S' && y=='P') return 0;
	if(x=='P' && y=='R') return 0;
	return 1;
}
int main(void) {
	scanf("%d%s%s", &n, A, B);
	int a=strlen(A), b=strlen(B);
	int s[3]={}, p[3]={};
	for(int i=0;i<a*b;i++) s[win(A[i%a], B[i%b])]++;
	for(int i=0;i<n%(a*b);i++) p[win(A[i%a], B[i%b])]++;
	for(int i=0;i<2;i++) p[i]+=s[i]*(n/(a*b));
	printf("%d %d\n", p[1], p[0]);
	return 0;
}
