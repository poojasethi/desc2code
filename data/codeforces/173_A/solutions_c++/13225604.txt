#include <cstdio>
#include <cstring>
bool c(char x1, char x2){return(x1=='R'&&x2=='S')||(x1=='S'&&x2=='P')||(x1=='P'&&x2=='R');}
char i1[1001],i2[1001];
int main(){
size_t n=0,n1=0,n2=0;scanf("%lu%s%s",&n,i1,i2);
size_t l1=strlen(i1),l2=strlen(i2),z=l1*l2,r=z<n?z:n;
for(size_t i=0;i<r;++i){char a=i1[i%l1],b=i2[i%l2];c(a,b)?++n2:a!=b?++n1:0;}
size_t m=n/r;n1*=m;n2*=m;
for(size_t i=0,e=n%r;i<e;++i){char a=i1[i%l1],b=i2[i%l2];c(a,b)?++n2:a!=b?++n1:0;}
printf("%lu %lu",n1,n2);}
