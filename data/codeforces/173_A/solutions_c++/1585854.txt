#include <iostream>
//#include <conio.h>

using namespace std;

int main()
{
   long long n,c1=0,c2=0,i,m,k,d;
   cin>>n;
   char a,b;
   string s,t;
   cin>>s>>t;
   m=s.length();
   k=t.length();
   for (i=0;(i<n)&&(i<m*k);i++)
   {
      a=s[i%m];
      b=t[i%k];
      if (a!=b)
      {
         d=(n-i+m*k-1)/(m*k);
         if (((a=='P')&&(b=='S'))||((a=='R')&&(b=='P'))||((a=='S')&&(b=='R')))
            c1+=d;
         else
            c2+=d;
      }
      //cout<<a<<" "<<b<<" : "<<c1<<" "<<c2<<endl;         
   }
   cout<<c1<<" "<<c2<<endl;
   //getch();
   return 0;
}
