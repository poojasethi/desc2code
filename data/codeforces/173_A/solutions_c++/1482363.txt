#include <cstdio>
#include <cstring>

int gcd(int a,int b)
{
	while(b)
	{
		int t=a%b;
		a=b;
		b=t;
	}
	return a;
}

int judge(char a,char b)
{
	if(a==b) return 0;
	if(a=='R' && b=='S' || a=='S' && b=='P' || a=='P' && b=='R') return 1;
	return 2;
}

int main()
{ 
	int n,m,k;
	char a[1010],b[1010];
	scanf("%d",&n);
	scanf("%s%s",a,b);
	m=strlen(a);k=strlen(b);
	int l=m*k/gcd(m,k),ans[3]={0};
	for(int i=0;i<l;i++)
		ans[judge(a[i%m],b[i%k])]++;
	for(int i=0;i<3;i++) ans[i]*=n/l;
	n%=l;
	for(int i=0;i<n;i++)
		ans[judge(a[i%m],b[i%k])]++;
	printf("%d %d\n",ans[2],ans[1]);
	return 0;  
}
