#include <iostream>

using namespace std;

int isNWins(char n,char p){
    if(n==p)
        return 0;
    if( (n=='R' && p=='S') || (n=='S' && p=='P') || (n=='P' && p=='R') )
        return -1;
    return 1;
}

int main(){
    int n,rn=0,rp=0,rn0,rp0,p,t;
    string sn,sp;
    cin>>n>>sn>>sp;
    p=sn.length()*sp.length();
    for(int i=0;i<p;i++){
        if(i==n%p) rn0=rn, rp0=rp;
        t=isNWins(sn[i%sn.length()], sp[i%sp.length()]);
        if(t==1) rn++;
        else if(t==-1) rp++;
    }
    cout<<rn*(n/p)+rn0<<" "<<rp*(n/p)+rp0<<endl;
    return 0;
}
