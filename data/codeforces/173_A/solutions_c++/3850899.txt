#include<iostream>
#include<string>
using namespace std;
int main(){
	int n,m,k,sa=0,sb=0;
	string a,b;
	cin>>n;
	cin>>a>>b;
	m=a.length();
	k=b.length();
	for(int i=0;i<m*k;i++){
		if(a[i%m]=='R' && b[i%k]=='S') sa++;
		else if(a[i%m]=='S' && b[i%k]=='P') sa++;
		else if(a[i%m]=='P' && b[i%k]=='R') sa++;
		else if(a[i%m]!=b[i%k]) sb++;
	}
	sa*=n/(m*k);
	sb*=n/(m*k);
	for(int i=0;i<n%(m*k);i++){
		if(a[i%m]=='R' && b[i%k]=='S') sa++;
		else if(a[i%m]=='S' && b[i%k]=='P') sa++;
		else if(a[i%m]=='P' && b[i%k]=='R') sa++;
		else if(a[i%m]!=b[i%k]) sb++;
	}
	cout<<sb<<' '<<sa;
	return 0;
}
