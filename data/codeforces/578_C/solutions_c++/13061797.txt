#include<iostream>
#include<stdio.h>
using namespace std;
int a[200001],n;
double check(double x)
{
    double summax=-1e9,summin=1e9,s1=0,s2=0,q=0;
    for(int i=1;i<=n;i++)
    {
        s1+=(a[i]-x);
        s2-=(a[i]-x);
        s1=max(s1,0.0);
        s2=max(s2,0.0);
        q=max(q,max(s1,s2));
    }
    return q;
}
main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i];
    }
    double l=-1e4,r=1e4;
    for(int i=1;i<=100;i++)
    {
        double m1=l+(r-l)/3;
        double m2=r-(r-l)/3;
        if(check(m1)<check(m2))
            r=m2;
        else l=m1;
    }
    printf("%.6f",check(l));
}
