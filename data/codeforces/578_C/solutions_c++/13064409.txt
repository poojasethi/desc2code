#include <bits/stdc++.h>

using namespace std;
int n;
double a[200010],l,r,ml,mr;

void nhap()
{
    cin>>n;
    for (int i=1;i<=n;++i) cin>>a[i];
}

double func(double x)
{
    double ma=0,mi=0,sum=0;
    for (int i=1;i<=n;++i)
    {
        sum+=a[i]-x;
        ma=max(ma,sum);
        mi=min(mi,sum);
    }
    return abs(ma-mi);
}

int main()
{
    nhap();
    l=-20000,r=20000;
    for (int i=1;i<=100;++i)
    {
        ml=(l+l+r)/3;
        mr=(r+r+l)/3;
        if (func(ml)>func(mr)) l=ml;
        else r=mr;
    }
    printf("%.10f",func(l));
    return 0;
}
