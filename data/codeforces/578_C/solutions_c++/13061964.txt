#include <bits/stdc++.h>
using namespace std;
int main() {
	double u,v,mi,l,m,maxi,mini,st=-10000,en=10000,ans=100000000000,a[200005];
	int i,j,k,n;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	while(en-st>=1e-12) {
		mi = (en+st)/2;
		for(i=1,maxi=mini=u=v=a[0]-mi;i<n;i++) {
			u = max(0.0,u) + a[i]-mi;
			v = min(0.0,v) + a[i]-mi;
			maxi = max(maxi,u);
			mini = min(mini,v);
		}
		ans = min(ans,max(maxi,-mini));
		if(maxi+mini>0)
			st = mi;
		else
			en = mi;
	}
	printf("%0.12lf\n",ans);
	return 0;
}
