#include <algorithm>
#include <cstdio>
using namespace std;
const int N = 200005;
int n, a[N];

double calc(double x) {
	double t_min = 0, t_max = 0, s = 0;
	for (int i = 0; i < n; i++) {
		s += a[i]-x;
		t_max = max(t_max, s);
		t_min = min(t_min, s);
	}
	return t_max-t_min;
}
int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) scanf("%d", a+i);
	double l = -1e5, r = 1e5;
	for (int k = 0; k < 200; k++) {
		double mid = l+(r-l)/3;
		double mid2 = r-(r-l)/3;
		if (calc(mid) > calc(mid2)) l = mid;
		else r = mid2;
	}
	printf("%.12lf\n", calc(l));
	return 0;
}
