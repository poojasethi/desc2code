#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
const int N = 2e5 + 5;
int a[N], n;
double calc(double x)
{
	double sum = 0, min = 0, max = 0;
	for (int i = 1; i <= n; ++ i)
	{
		sum = sum + a[i] - x;
		min = std::min(min, sum);
		max = std::max(max, sum);
	}
	return max - min;
}
int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; ++ i) scanf("%d", &a[i]);
	double l = -1e4, r = 1e4;
	for (int step = 0; step < 150; ++ step)
	{
		double mid = (l + r) / 2;
		double mid2 = (mid + r) / 2;
		if (calc(mid) > calc(mid2))
		{
			l = mid;
		}
		else
		{
			r = mid2;
		}
	}
	printf("%.12f\n", calc(l));
}
