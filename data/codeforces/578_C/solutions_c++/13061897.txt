#include <bits/stdc++.h>

using namespace std;

const int MAXN = 2e5+123;

int N,ar[MAXN];
double er[MAXN],ans;

bool f(double o){
	for(int i=1;i<=N;i++)
		er[i]=ar[i]-o;
	double mt=0.0,pt=0.0,rm=0.0,rp=0.0;
	for(int i=1;i<=N;i++){
		if(mt>0.0) mt=0.0;
		if(pt<0.0) pt=0.0;
		mt+=er[i];
		pt+=er[i];
		rm=max(rm,-mt);
		rp=max(rp,pt);
	}
	ans=rp;
	return rp>rm;
}

int main(){
	
	int mn=1e9,mx=-1e9;
	scanf(" %d",&N);
	for(int i=1;i<=N;i++){
		scanf(" %d",&ar[i]);
		mn=min(mn,ar[i]);
		mx=max(mx,ar[i]);
	}
	
	double b=-1e5,e=1e5,o,cnt=0;
	while(cnt<=70){
		o=(b+e)/2.00;
		if(f(o))
			b=o;
		else
			e=o;
		cnt++;
	}
	
	printf("%.12lf\n",ans);
	
	return 0;
}
