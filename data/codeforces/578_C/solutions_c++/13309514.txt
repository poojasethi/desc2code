#include<bits/stdc++.h>
#define INF 1e60
#define EPS 1e-8
#define rep(i,a,b) for(int i=(a);i<=(b);++i)
using namespace std;
const int maxn=200100;
int n;
double a[maxn];
double f(double x){
	double ma=0,mi=0,sum=0;
	rep(i,1,n){
		sum+=a[i]-x;
		ma=max(ma,sum);
		mi=min(mi,sum);
	}

	return abs(ma-mi);
}
int main()
{
	cin>>n;
	rep(i,1,n)scanf("%lf",&a[i]);
	double l=-20000,r=20000;
	rep(i,1,100){
		double ml=(l+l+r)/3,mr=(l+r+r)/3;
		if(f(ml)>f(mr))l=ml;
		else r=mr;
	}
	printf("%.9lf\n",f(l));
	
	return 0;
}
