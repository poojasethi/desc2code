#include<cstdio>
using namespace std;
int a1[200000];
int main()
{
    int n;double l=-10000,r=10000,md,a,b,c,d;
    scanf("%d",&n);
    for(int i=0;i<n;++i)scanf("%d",a1+i);
    while(r-l>1e-12)
    {
        md=(l+r)/2;a=-10001;b=10001;c=d=0;
        for(int i=0;i<n;++i)
        {
            c+=a1[i]-md;if(c<0)c=0;if(c>a)a=c;
            d+=a1[i]-md;if(d>0)d=0;if(d<b)b=d;
        }
        if(a+b>0)l=md;else r=md;
    }
    printf("%.9f\n",a);
}
