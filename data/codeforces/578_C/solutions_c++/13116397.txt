#include<iostream>
#include<cstdio>

using namespace std;

long n,i,buffer[200000];

double search(double left,double right){
	double mean=(left+right)/2;
	double maxx=-200000,combomax=0;
	double minn=200000,combomin=0;
	for(i=0;i<n;i++){
		combomax+=buffer[i]-mean;
		if(combomax>maxx) maxx=combomax;
		if(combomax<0) combomax=0;
		combomin+=buffer[i]-mean;
		if(combomin<minn) minn=combomin;
		if(combomin>0) combomin=0;
	}
	if(right-left<0.000000000001) return maxx;
	if(maxx+minn>0) return search(mean,right);
	else return search(left,mean);
}

int main()
{
	cin >> n;
	for(i=0;i<n;i++) cin >> buffer[i];
	printf("%.7lf",search(-10000,10000));
	return 0;
}
