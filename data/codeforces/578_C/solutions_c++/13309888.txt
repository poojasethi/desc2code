#include<bits/stdc++.h>
#define INF 1e60
#define EPS 1e-8
#define FOR(i,a,b) for(int i=(a);i<=(b);++i)

using namespace std;

int n;
double a[200005];

double f(double x){
	double ma=0,mi=0,sum=0;
	FOR(i, 1, n){
		sum += a[i] - x;
		ma = max(ma,sum);
		mi = min(mi,sum);
	}
	return abs(ma-mi);
}

int main(){
	scanf("%d", &n);
	FOR(i, 1, n) scanf("%lf", &a[i]);
	double l = -20000, r = 20000, ml, mr;
	FOR(i, 1, 200){
		ml = (l + l + r) / 3;mr = (l + r + r) / 3;
		if(f(ml) > f(mr)) l = ml; else r = mr;
	}
	printf("%0.9lf", f(l));
}
