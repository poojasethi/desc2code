#include <bits/stdc++.h>
#define LL long long
using namespace std;
int n;
double s[200002];
bool g(double cur,int mode){
    double sp=0,cp=0,sm=0,cm=0;
    for(int i=1;i<=n;i++){
        cp+=s[i]-cur;
        cm+=s[i]-cur;
        if(cp>sp) sp=cp;
        if(cp<0) cp=0;
        if(cm<sm) sm=cm;
        if(cm>0) cm=0;
    }
    if(mode==1) printf("%.10lf\n",max(-sm,sp));
    return sm+sp>=0;
}
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%lf",&s[i]);
    double l=-10005,r=10005;
    int it=60;
    while(it--){
        double m=(l+r)/2;
        if(g(m,0)) l=m;
        else r=m;
    }
    g(l,1);
    return 0;
}
