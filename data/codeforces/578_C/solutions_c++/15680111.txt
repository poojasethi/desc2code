#include <stdio.h>
#include <algorithm>

#define lf double
#define maxn 200007

using namespace std;

int n;

lf v[maxn];

lf f(lf x)
{
	lf a = 0, b = 0;
	lf mn = 0, mx = 0;
	
	for (int i = 1; i <= n; i++) {
		a += v[i] - x;
		b += v[i] - x;
		
		mn = min(mn, a);
		mx = max(mx, b);

		if (a > 0) a = 0;
		if (b < 0) b = 0;
	}
	
	return max(mx, -mn);
}

int main()
{
	scanf("%d", &n);
	
	for (int i = 1; i <= n; i++)
		scanf("%lf", &v[i]);
	
	lf l = -10000, r = 10000;
	
	for (int i = 0; i < 100; i++)
	{
		lf m1 = l + (r-l)/3;
		lf m2 = r - (r-l)/3;
		
		if (f(m1) >= f(m2))
			l = m1;
		else
			r = m2;
	}
	
	printf("%.7lf\n", f(l));
}
