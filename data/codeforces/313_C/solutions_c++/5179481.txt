#include <iostream>
#include <algorithm>
using namespace std;

long long ANS=0, a[2000010]={}, n;
int main()
{
    cin >> n;
    for (int i=1;i<=n;++i)
        cin >> a[i];
    sort(a+1, a+n+1);
    for (int i=1;i<=n;++i)
        a[i] += a[i-1];
    for (int i=1;i<=n;i<<=2)
        ANS += a[n]-a[n-i];
    cout << ANS << endl;
    return 0;
}
