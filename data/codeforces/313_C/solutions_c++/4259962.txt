#include <iostream>
#include <algorithm>
using namespace std;

long long a[4000010];
long long sum[4000010];

int main()
{
	int n,i;
	long long cur=0;
	cin >> n;
	for(i=1;i<=n;i++)
		cin >> a[i];
	sort(a+1,a+1+n);
	sum[n]=a[n];
	for(i=n-1;i>=1;i--)
		sum[i]=sum[i+1]+a[i];
	i=1;
	while(i<=n){
		cur+=sum[n-i+1];
		i*=4;
	}
	cout << cur;
	return 0;
}
