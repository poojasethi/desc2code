#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int a[3000000];
int main()
{
int n;
scanf("%d",&n);
for(int i=1;i<=n;i++)
scanf("%d",&a[i]);
sort(a+1,a+1+n);
long long ans=0;
int pow=1;
for(;pow<=n;pow*=4)
{
for(int i=n+1-pow;i<=n;i++)
ans+=a[i];
}
printf("%I64d\n",ans);
return(0);
}
