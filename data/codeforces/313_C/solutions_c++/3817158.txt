#include<stdio.h>
#include<algorithm>
using namespace std;

int a[2000001];

int main(){
	int n, i, j;
	scanf("%d", &n);
	for(i=0;i<n;i++) scanf("%d", &a[i]);
	
	sort(a, a+n);
	
	long long sum = 0;
	for(i=1;i<=n;i*=4){
		for(j=1;j<=i;j++){
			sum+=a[n-j];
		}
	}
	printf("%I64d\n", sum);
	return 0;
}
