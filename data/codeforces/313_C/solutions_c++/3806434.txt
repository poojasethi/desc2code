#include <cstdio>
#include <algorithm>

using namespace std;

int main(){
	int *a, n;
	long long int total=0;

	scanf("%d", &n);
	a = (int *) calloc(n+10, sizeof(int));

	for(int i=0; i<n; i++) scanf("%d",a+i);
	sort(a, a+n);

	for(int i=1; i<=n;i+=3*i)
		for(int j=n-i; j<n; j++)
			total+=a[j];

	printf("%I64d", total);
	return 0;
}
