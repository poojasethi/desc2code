#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
#define ll long long
int a1[2000000];
int main()
{
	int n,a;ll o=0;
	scanf("%d",&n);
	a=log(n+0.1)/log(4);
	for(int i=0;i<n;++i)
		scanf("%d",a1+i);
	sort(a1,a1+n);
	for(int i=0;i<n;++i)
		o+=(ll)a1[i]*(a+1-(int)(log(n-i-0.1)/log(4)+1));
	printf("%I64d\n",o);
}
