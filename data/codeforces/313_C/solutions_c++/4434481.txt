#include<stdio.h>
#include<algorithm>
#define maxn 2000005
using namespace std;
int a[maxn];
bool cmp(int x,int y)
{
	return (x>y);
}
int main()
{
	int N,i;
	long long ans=0;
	scanf("%d",&N);
	for(i=0;i<N;i++) scanf("%d",&a[i]);
	sort(a,a+N,cmp);
	while(N)
	{
		for(i=0;i<N;i++) ans+=a[i];
		N/=4;
	}
	printf("%I64d\n",ans);
}
