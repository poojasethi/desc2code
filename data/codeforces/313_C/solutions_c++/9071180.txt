#include <iostream>
#include <algorithm>
using namespace std;
long long n,a[2000001],c=1,kil=1,sum,k=1;
int main()
{cin>>n;
while (c<n) {c*=4;kil++;}
for (int i=1;i<=n;i++) cin>>a[i];
sort(a+1,a+1+n);
for (int i=n;i>0;i--) {if ((n-i)%k==0) {kil--;k*=4;}sum+=a[i]*kil;}
sum+=a[n];
cout<<sum<<endl;
    return 0;
}
