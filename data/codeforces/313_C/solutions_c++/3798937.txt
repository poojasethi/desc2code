#include<iostream>
#include<algorithm>
using namespace std;

int n,data[3000001];

int cmp(int x,int y) {return x>y;}

int main() {
	cin >> n;
	for (int i=1;i<=n;i++) cin >> data[i];
	sort(data+1,data+1+n,cmp);
	long long ans=0;
	while (n) {
		for (int i=1;i<=n;i++) ans+=data[i];
		n/=4;
	}
	cout << ans<<endl;
	return 0;
}
