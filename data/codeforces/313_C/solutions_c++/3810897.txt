#include <algorithm>
#include <stdio.h>
typedef long long ll;
using namespace std;
ll A[2000002],S[2000002],N,i,ans;
int main()
{
    scanf("%d",&N);
    for(i=0;i<N;i++)scanf("%d",&A[i]);
    sort(A,A+N);
    for(i=N-1;i>=0;i--)S[i]=S[i+1]+A[i];
    for(i=N;i>0;i/=4)ans+=S[N-i];
    printf("%I64d",ans);
}
