#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
long long a[2000002];
int main()
{
long long k,an=0,cn=0,p,n,i,j;
cin>>n;
p=n;
while(p!=1)
{
cn++;
p=p/4;
}
for(i=0;i<n;i++)
{cin>>a[i];
}
sort(a,a+n);
for(i=n-2;i>=0;i--)
a[i]=a[i]+a[i+1];
p=cn+1;
k=1;
while(p--)
{
an+=a[n-k];
k=k*4;
}
cout<<an<<endl;
return(0);
}
