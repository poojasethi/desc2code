#include <cstdio>
#include <algorithm>
using namespace std;

long long sum[1<<22];

int main(){
int n; scanf("%d",&n);
for(int i=0;i<n;i++) scanf("%I64d",&sum[i]);
sort(sum,sum+n);
reverse(sum,sum+n);
for(int i=1;i<n;i++) sum[i]+=sum[i-1];
long long ans=0;
while (n) {
ans+=sum[n-1];
n/=4;
}
printf("%I64d",ans);
return 0;
}
