#include<iostream>
#include<algorithm>
using namespace std;
typedef long long l;
int main()
{
	l n;
	cin>>n;
	l sum=0;
	l a[n];
	for(int i = 0;i<n;i++)
		cin>>a[i];
	sort(a,a+n);
	reverse(a,a+n);
	int j;
	for(int i=1;i<=n;i=i*4)
		for(j=0;j<i;j++)
			sum = sum + a[j];
	cout<<sum<<endl;
	return 0;

}
