#include<bits/stdc++.h>

#define	LLI long long int 

using namespace std;

LLI top=0,n,t,arr[2000005],i,j;
int main(){
	scanf("%lld",&n);
	for(i=1;i<=n;i++)
		scanf("%lld",&arr[i]);
	sort(arr+1,arr+n+1);
	reverse(arr+1,arr+n+1);
	
	t=1;
	while(t<=n){
		for(i=1;i<=t;i++)
			top+=arr[i];
		t*=4;	
	}
	printf("%lld",top);
	
	return 0; 
}
