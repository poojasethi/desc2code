#include <iostream>
#include <algorithm>
using namespace std;

long long a[2000005];

int main(){
    int n, temp;
    cin>>n;
    for(int i=1; i<=n; i++) cin>>a[i];
    sort(a+1, a+1+n);
    long long sum=0;
    temp=n;
    while(temp!=0){
        for(int i=n; i>=n-temp+1; i--) sum+=a[i];
        temp/=4;
    }
    cout<<sum;
    return 0;
}
