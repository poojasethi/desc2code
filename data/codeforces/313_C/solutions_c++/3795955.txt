#include <iostream>
#include <algorithm>
#include <functional>
using namespace std;
int a[2000000];
int main(){
	int n;
	cin >> n;
	for(int i=0;i<n;i++) cin >> a[i];
	sort(a,a+n,greater<int>());
	long long ans=0;
	for(int hoge=1;hoge<=n;hoge*=4){
		for(int j=0;j<hoge;j++){
			ans+=a[j];
		}
	}
	cout << ans;
	return 0;
}
