#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const int N = 2222222;
LL a[N];
int main() {
	int n;
	while (cin >> n) {
		for (int i = 0; i < n; i++) cin >> a[i];
		sort(a, a + n);
		reverse(a, a + n);
		int sz = 1;
		LL ans = 0;
		while (sz <= n) {
			ans += accumulate(a, a + sz, 0ll);
			sz <<= 2;
		}
		cout << ans << endl;
	}
	return 0;
}
