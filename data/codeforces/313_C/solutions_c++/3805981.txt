#include <iostream>
#include <algorithm>
using namespace std;
int main ()
{
	long long int n, j, i, sum = 0;
	cin >> n;
	long long int a[n+1];
	for ( i = 1; i <= n; i ++) {
		cin >> a[i];
	}
	sort (a+1, a+n+1);
	for ( i = 1; i <= n; i = i * 4) {
		for ( j = 1; j <= i; j++)
			sum= sum + a[n-j + 1];
	}
	cout << sum << endl;
	return 0;
}
