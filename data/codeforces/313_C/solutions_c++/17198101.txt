#include <bits/stdc++.h>
using namespace std;
vector<int> p;
int N;
int main(){
	cin >> N;
	for(int i = 0,a; i<N; i++){
		scanf("%d",&a);
		p.push_back(a);
	}
	sort(p.begin(),p.end(),greater<int>());
	long long sum = 0;
	for(int i = 1; i <=N; i*=4){
		for(int j = 0; j<i; j++) sum+=p[j];
	}
	cout << sum << endl;
}
