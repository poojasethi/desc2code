#include <bits/stdc++.h>
#define MAXN 2000000
#define ri(n) scanf("%d",&n)
#define FOR(i,l,r) for(int i=l;i<r;i++)
using namespace std;

int N, M, A[MAXN], i=0;
long long sum=0, cnt=0;

int main() {
	ri(N); FOR(j,0,N) ri(A[j]); sort(A,A+N); M=N;
	while (M>0) {while(i<N-M) sum+=A[i++]*cnt; M/=4; cnt++;} sum+=A[N-1]*cnt;
	printf("%I64d\n",sum);
}
