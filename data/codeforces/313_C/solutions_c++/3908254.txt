#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <string.h>

using namespace std;
int n,a[1111111];

int main()
{
    cin >> n;
    for(int i=0; i<n; i++) cin >> a[i];
    sort(a,a+n); reverse(a,a+n);
    long long ans=0;
	while (n)
	{
		for(int i=0; i<n; i++) ans+=a[i];
		n/=4;
	}
	cout << ans;
	cin >> n;
	return 0;
}
