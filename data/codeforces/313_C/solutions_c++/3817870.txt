#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	int n;
	cin >> n;
	int a[n];
	for (int i = 0; i < n; ++i)
		cin >> a[i];

	long long ans = 0;
	for (; n > 0; n /= 4) {
		for (int i = 0; i < n; ++i)
			ans += a[i];
		nth_element(a, a+n/4, a+n, greater<int>());
	}
	cout << ans;
	return 0;
}
