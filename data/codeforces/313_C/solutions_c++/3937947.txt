#include<iostream>
#include<algorithm>
#define ll long long
#include<vector>
using namespace std;
int main(){
int n;cin>>n;
vector<ll>v;ll ans=0;
for(int i=0;i<n;i++) {ll val;cin>>val;v.push_back(val);}
sort(v.rbegin(),v.rend());
//ans+=v[0];
for(int i=0,j=1;i<n;i++){
if(i<j)  ans+=v[i];
else {i%=j;j*=4;ans+=v[i];}
}
 
cout<<ans;
 
}
