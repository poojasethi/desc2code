#include<iostream>
#include<algorithm>
using namespace std;
long long n,i,a[2000005],sum,p;
int main()
{
  cin>>n;

  for(i=1;i<=n;i++)
    cin>>a[i];

  sort(a+1,a+n+1);

  p = 1;
  while(p <= n)
  {
    for(i=n;i>=n-p+1;i--)
      sum += a[i];
    p *= 4;
//    cout<<sum<<endl;
  }

  cout<<sum<<endl;
}
