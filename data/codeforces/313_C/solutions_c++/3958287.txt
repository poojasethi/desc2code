#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int a[1<<21];
int main()
{
    int n;
    cin>>n;
    for(int i=0 ; i<n ; ++i)
    cin>>a[i];
    sort(a,a+n);
    reverse(a,a+n);
    long long sum=0;
    for(int i=1 ; i<=n ; i*=4)
    for(int j=0 ; j<i ; ++j)
        sum+=a[j];
    cout<<sum<<endl;
}
