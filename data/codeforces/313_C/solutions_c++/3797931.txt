#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
typedef long long ll;
int n,a[2000000];
ll sum;
inline bool comp(int x, int y) {
 	return (x>y);
}
int main(){
 	cin>>n;
 	for(int i=0;i<n;i++) cin>>a[i];
 	sort(a,a+n,comp);
 	while(n) {
 		for(int i=0;i<n;i++) sum+=a[i];
 		n/=4;
 	}
 	cout<<sum;
 	return 0;
}
