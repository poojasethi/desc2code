#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int a[2000009];
int main()
{
	int n,m,i,j;
	long long ans;
	while(cin>>n)
	{
		ans=0;
		for(i=0;i<n;i++) cin>>a[i];
		sort(a,a+n);
		m=n;
		while(n)
		{
			for(i=m-1,j=0;j<n;i--,j++) ans+=a[i];
			n/=4;
		}
		cout<<ans<<endl;
	}
	return 0;
}
