#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
	int k;
	cin>>k;
	int a[k];
	for(int i = 0; i < k; i++) cin>>a[i];

	sort(a, a+k);
	
	int r = log2(k) / 2;
	long long s = 0;

	for(int i = 0; i < r+1; i++){
		int l = pow(4, i);
		for(int j = k - l; j < k; j++){
			s += a[j];
		}
	}
	cout<<s;

}
