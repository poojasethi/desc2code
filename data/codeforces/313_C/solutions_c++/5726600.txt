#include <iostream>
#include <algorithm>
using namespace std;

#define MAX int(2e6)

int n, a[MAX];

int main() {
	cin >> n;
	for(int i = 0; i < n; i++)
		cin >> a[i];
	sort(a, a+n);
	long long int ans = 0;
	for(int i = n; i > 0; i /= 4)
		for(int j = n-i; j < n; j++)
			ans += a[j];

	cout << ans << endl;
}
