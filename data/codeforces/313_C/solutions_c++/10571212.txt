#include <bits/stdc++.h>
using namespace std;

bool cmp(int a , int b){
	return a>b;
}
long long a[2000005] , ans=0ll;
int main() {
	int n ; 

	scanf("%d" , &n);
	for(int i=0 ; i<n ; i++){
		scanf("%I64d" , &a[i]);
	}
	sort(a , a+n , cmp);
	for(int i=1 ; i<n ; i++) a[i]+=a[i-1];
	while(n){
		ans+=a[n-1];
		n/=4;
	}
	printf("%I64d\n" , ans);
	return 0;
}
