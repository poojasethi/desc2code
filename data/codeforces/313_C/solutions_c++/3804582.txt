#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n; cin>>n;
	int d[n];
	for(int i = 0; i < n; ++i)
	{
		cin>>d[i];
	}
	sort(d,d+n);
	long long ans=0;
	for(int i = 1; i <= n; i*=4)
	{
		for(int j = 1; j <= i; ++j)
		{
			ans+=d[n-j];
		}
	}
	cout<<ans<<endl;
  return 0;
}
