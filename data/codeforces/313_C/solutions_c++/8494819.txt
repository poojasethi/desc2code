#include <bits/stdc++.h>
using namespace std;
long long n, arr[2000010];
int main(){
    cin >> n; long long sum=0;
    for(int i=1; i<=n; i++) cin >> arr[i];
    sort(arr+1,arr+1+n); reverse(arr+1,arr+1+n);
    for(int i=1; i<=n; i++) arr[i]+=arr[i-1];
    while(n>0) {sum+=arr[n]; n/=4;} cout<<sum;
}
