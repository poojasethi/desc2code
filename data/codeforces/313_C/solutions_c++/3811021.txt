#include <iostream>
#include <algorithm>
using namespace std;

long long a[2000000], ans;
int x, y;

int main() {
  cin >> x;
  for (int i(0); i < x; ++i)
    cin >> a[i];
  sort(a, a + x);
  for (int i(x); i; i >>= 2, ++y);
  for (int i(0), j(1); j <= x; j <<= 2, --y)
    for (; i < j; ++i)
      ans += a[x - i - 1] * y;
  cout << ans << endl;
}
