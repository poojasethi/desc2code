#include<bits/stdc++.h>
using namespace std;
const int N=100010;
int n,m,fa[N]={},last[N]={},next[N]={},end[N]={},total_edge=0;
int a[N]={},b[N]={},tot=0,c[N]={};
bool to[N]={};
void add_edge(int u,int v)
{
	next[++total_edge]=last[u];
	last[u]=total_edge;
	end[total_edge]=v;
}
void init()
{
	scanf("%d%d",&n,&m);
	for(int i=1,p,q;i<=m;++i)
	{
		scanf("%d%d",&p,&q);
		fa[q]=p;
		add_edge(p,q);
	}
	for(int i=1;i<=n;++i)
		scanf("%d",&a[i]);
	for(int i=1;i<=n;++i)
		to[a[i]]=true;
}
void dfs(int s,int C)
{
	if(to[s])
		C=s;
	c[s]=C;
	for(int i=last[s];i;i=next[i])
		dfs(end[i],C);
	if(to[s])
		b[++tot]=s;
}
void work()
{
	for(int i=1;i<=n;++i)
		if(!fa[i])
			dfs(i,i);
	for(int i=1;i<=n;++i)
		if(a[i]!=c[i])
		{
			puts("-1");
			return;
		}
	printf("%d\n",tot);
	for(int i=1;i<=tot;++i)
		printf("%d\n",b[i]);
}
int main()
{
	init();
	work();
}
