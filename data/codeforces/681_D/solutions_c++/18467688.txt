#include <cstdio>
#include <iostream>
#include <cstring>
#include <string>
#include <algorithm>
#include <queue>
#include <map>
#include <set>
#include <stack>
#include <vector>
#define MAXN 102000
#define pb push_back
using namespace std;
typedef long long lint;
int ar[MAXN];
int in[MAXN];
int giv[MAXN];
int res[MAXN];
int ord[MAXN];
int ust[MAXN];
int K;
vector <int> G[MAXN];
void dfs(int x,int k){
	if(giv[x]) k = x;
	ust[x] = k;
	for(int i = 0 ; i < G[x].size(); i++){
		dfs(G[x][i],ust[x]);
	}
	if(giv[x]){
		K++;
		res[K] = x;
		ord[x] = K;
	}
}
int main(){
	int n,m;
	scanf(" %d %d",&n,&m);
	for(int i = 0 ; i < m ; i++){
		int a,b;
		scanf(" %d %d",&a,&b);
		G[a].pb(b);
		in[b]++;
	}
	for(int i = 1 ; i <= n ; i++){
		scanf(" %d",&ar[i]);
		giv[ar[i]] = 1;
	}
	for(int i = 1 ; i <= n ; i++){
		if(in[i] == 0){
			dfs(i,-1);
		}
	}
	for(int i = 1 ; i <= n ; i++){
		if(ust[i] != ar[i]){
			printf("-1\n");
			return 0;
		}
	}
	printf("%d\n",K);
	for(int i = 1 ; i <= K; i++) printf("%d\n", res[i]);
	return 0;
}
