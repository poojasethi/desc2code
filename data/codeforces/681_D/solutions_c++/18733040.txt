#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const int MAXN = 1e5 + 5;

struct
{
    int v, next;
}e[2*MAXN];

int head[MAXN],tt = 0, cnt = 0, f = 0;
int n, m, w[MAXN], ans[MAXN], q[MAXN];

void Addeage(int a, int b)
{
    e[++tt].v = b;
    e[tt].next = head[a];
    head[a] = tt;
}

void dfs(int u, int p)
{
    for(int k = head[u]; k; k = e[k].next)
    {
        int v = e[k].v;
        if(v == p) continue;
        if(w[v] != w[u] && w[v] != v) f = 1;
        dfs(v, u);
    }
    if(w[u] == u) ans[++cnt] = u;
}

int main()
{
    scanf("%d%d", &n, &m);
    int a, b;
    while(m --)
    {
        scanf("%d%d", &a, &b);
        Addeage(a, b);
        Addeage(b, a);
        q[b]++;
    }
    for(int i = 1;i <= n;i ++) scanf("%d", &w[i]);
    for(int i = 1;i <= n;i ++) if(!q[i]) dfs(i , 0);
    if(f) printf("-1\n");
    else
    {
        printf("%d\n", cnt);
        for(int i = 1;i <= cnt;i ++)
            printf("%d\n", ans[i]);
    }
    return 0;
}
