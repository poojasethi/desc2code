#include<bits/stdc++.h>
using namespace std;
const int maxn = 101010;
vector<int>e[maxn];
vector<int>ans;
bool flag;
int n  , m , k , a[maxn] , fa[maxn] , vis[maxn];
void add(int u , int v){e[u].push_back(v); fa[v] = u;}
void dfs(int num){
    for(int i = 0 ; i < e[num].size();i++){
        dfs(e[num][i]);
    }
    if(vis[num])
    ans.push_back(num);
    if(a[num] != num && a[num] != a[fa[num]]){
        flag = 1;
    }
}
int main()
{
    flag = 0;
    memset(fa,0,sizeof(fa));
    memset(vis,0,sizeof(vis));
    scanf("%d%d",&n,&m);
    while(m--){
        int u , v;
        scanf("%d%d",&u,&v);
        add(u , v);
    }
    for(int i = 1; i <= n ;i++){
        scanf("%d",&a[i]);
        vis[a[i]] = 1;
    }
    for(int i = 1; i <= n;i++){
        if(!fa[i]){
            dfs(i);
        }
    }
    if(!flag ){
            printf("%d\n",ans.size());
    for(int i = 0 ; i < ans.size();i++){
        printf("%d\n",ans[i]);
    }
    }
    else{
        printf("-1\n");
    }
}
