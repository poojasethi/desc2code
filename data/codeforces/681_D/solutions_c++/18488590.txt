#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
struct point
{
	int f,t;
	vector<int>s;
}x[100010];
vector<int>ans;
int ok=1;
void dfs(int t,int n)
{
	if (ok==0) return ;
	if (x[n].t==n) t=n;
	if (x[n].t!=t)
	{
		ok=0;
		return ;
	}
	int l=x[n].s.size();
	for (int i=0;i<l;i++) dfs(t,x[n].s[i]);
	if (x[n].t==n) ans.push_back(t);
	return ;
}

int main()
{
	int n,m,p,q;
	
	scanf("%d %d",&n,&m);
	for (int i=1;i<=n;i++) x[i].f=i;
	for (int i=0;i<m;i++)
	{
		scanf("%d %d",&p,&q);
		x[q].f=p;
		x[p].s.push_back(q);
	}
	for (int i=1;i<=n;i++) scanf("%d",&x[i].t);
	for (int i=1;i<=n;i++)
	{
		if (x[i].f==i) dfs(0,i);
	}
	if (ok)
	{
		int l=ans.size();
		printf("%d\n",l);
		for (int i=0;i<l;i++) printf("%d\n",ans[i]);
	}
	else
	{
		printf("-1");
	}
	return 0;
}
