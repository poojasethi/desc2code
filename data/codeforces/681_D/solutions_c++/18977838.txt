#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
const int maxn = 100000 + 10;
vector<int>g[maxn],ans;
int d[maxn],a[maxn];
bool ok = true;
void dfs(int k,int fa){
    if (!ok)return ;
    int len = g[k].size();
    for (int i = 0; i < len; ++i){
        int u = g[k][i];
        if (u == fa)continue;
        if (a[u] != a[k] && a[u] != u)ok=false;
        dfs(u,k);
    }
    if (a[k] == k)ans.push_back(k);
}
int main(){
    int n,m;
    scanf("%d %d",&n,&m);
    for (int i = 0; i < m; ++i){
        int a,b;
        scanf("%d %d",&a,&b);
        g[a].push_back(b);
        g[b].push_back(a);
        d[b]++;
    }
    for (int i = 1; i <= n; ++i)scanf("%d",&a[i]);
    for (int i = 1; i <= n; ++i){
        if (!d[i])dfs(i,-1);
    }
    if (!ok){
        printf("-1\n");
        return 0;
    }
    int len = ans.size();
    printf("%d\n",len);
    for (int i = 0; i < len; ++i){
        printf("%d\n",ans[i]);
    }
    return 0;
}
