#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=100010;
bool ans=true;
int n,h[N],a[N];
int m,deg[N];
vector<int> e[N];
vector<int>res;
bool ok[N];
void dfs(int u, int par){
   if(!ans)return;
   if(u==a[u]||a[u]==a[par]){
      if(!ok[a[u]])
      res.push_back(a[u]),ok[a[u]]=1;
   }else ans=0;
   for(int i=0;i<e[u].size();++i){
      int v=e[u][i];
      dfs(v,u);
   }
}
int main(){
//   freopen("input.txt","r",stdin);
   ios_base::sync_with_stdio(0);
   cin>>n>>m;
   for(int i=0;i<m;++i){
      int u,v;
      cin>>u>>v;
      e[u].push_back(v);
      ++deg[v];
   }
   for(int i=1;i<=n;++i)
      cin>>a[i];
   for(int i=1;i<=n;++i){
      if(deg[i]!=0)continue;
      dfs(i,a[i]);
   }
   if(ans==0){
      puts("-1");
      return 0;
   }
   cout<<res.size()<<endl;
   for(int i=res.size()-1;i>=0;--i)cout<<res[i]<<endl;
   return 0;
}
