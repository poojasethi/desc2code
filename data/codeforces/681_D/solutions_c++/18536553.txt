#include <bits/stdc++.h>
using namespace std;
#define pb push_back
#define sd(x) scanf("%d", &x)

const int N = 100005;
vector <int> v[N];
vector <int> ans;
int n, m, a[N], x, y, parent[N];
bool flag = true;

void dfs(int u)
{
	if(a[u]==u)	ans.pb(u);
	for(int i = 0; i<v[u].size(); i++)
	{
		x = v[u][i];
		if(a[x]!=x && a[u]!=a[x])	flag = false;
		dfs(x);
	}
}

int main()
{
	sd(n); sd(m);
	for(int i = 1; i<=n; i++)	parent[i] = i;
	while(m--)
	{
		sd(x); sd(y);
		v[x].pb(y);
		parent[y] = x;
	}
	for(int i = 1; i<=n; i++)	sd(a[i]);
	for(int i = 1; i<=n; i++)	if(parent[i]==i)	dfs(i);
	if(flag)
	{
		cout << ans.size() << "\n";
		for(int i = ans.size()-1; i>=0; i--)	cout << ans[i] << "\n";
	}
	else	cout << "-1";
}
