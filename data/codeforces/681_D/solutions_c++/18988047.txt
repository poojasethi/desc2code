#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;

const int N = 100010;
int to[N<<1], nxt[N<<1], head[N], cnt;
void add(int x, int y){
	to[++ cnt] = y; nxt[cnt] = head[x]; head[x] = cnt;
}

int n, m, dep[N], size[N], fa[N], want[N];
int ans[N], tot, du[N];
bool yes = 1;
void dfs(int x){
	for(int i = head[x]; i; i = nxt[i])
		dfs(to[i]);
	if (want[x] != x && want[x] != want[fa[x]]) yes = 0;
	if (want[x] == x) ans[++ tot] = x;
}

int main(){
	scanf("%d%d",&n,&m);
	int x, y;
	for(int i = 1; i <= m; i ++){
		scanf("%d%d", &x, &y);
		fa[y] = x;
		add(x, y);
		du[y] ++;
	}
	for(int i = 1; i <= n; i ++)
		scanf("%d", &want[i]);
	for(int i = 1; i <= n; i ++)
		if (!du[i]) dfs(i);
	if (!yes){
		puts("-1");
		return 0;
	}
	printf("%d\n",tot);
	for(int i = 1; i <= tot; i ++)
		printf("%d\n",ans[i]);
	return 0;
}
