#include <iostream>
#include <vector>
#include <cstdlib>

using namespace std;

const int limite=1000000;

int n,m;
vector<int> g[limite];
int profundidad[limite];
vector<int> sol;
int a[limite];
int aridad[limite];

void genera(int u,int prof,int lastprof)
{
  profundidad[u]=prof;
  if (a[u]==u) {
    lastprof=prof;
  } else if (profundidad[a[u]]!=lastprof) {
    cout<<-1<<endl;
    exit(0);
  }
  vector<int> &ar=g[u];
  for (int i=0;i<int(ar.size());i++) {
    int v=ar[i];
    genera(v,prof+1,lastprof);
  }
  if (lastprof==prof) sol.push_back(u);
}

int main()
{
  ios::sync_with_stdio(false);
  cin>>n>>m;
  for (int i=0;i<m;i++) {
    int u,v;
    cin>>u>>v;
    g[u].push_back(v);
    aridad[v]++;
  }
  for (int i=1;i<=n;i++)
    cin>>a[i];
  for (int u=1;u<=n;u++)
    if (aridad[u]==0)
      genera(u,0,-1);
  cout<<int(sol.size())<<endl;
  for (int i=0;i<int(sol.size());i++)
    cout<<sol[i]<<endl;
}
