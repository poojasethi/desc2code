using namespace std;
const int N=202020;
int head[N],nxt[N],lst[N],fa[N];
int num[N],mark[N],c[N],F[N],deep[N];
int n,m,tot,cnt,t[N];
void insert(int x,int y)
{
	lst[++tot]=y; 
	nxt[tot]=head[x];
	head[x]=tot;
}		
bool dfs(int x)
{
	for(int i=head[x];i;i=nxt[i])
	{
		if(num[lst[i]]!=lst[i]&&num[lst[i]]!=num[x]) return 0;
		if(!dfs(lst[i])) return 0;
	}
	if(num[x]==x) c[++cnt]=x;
	return 1;
}
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		fa[y]=x;
		insert(x,y);
	}
	for(int i=1;i<=n;i++) scanf("%d",&num[i]);
	for(int i=1;i<=n;i++)
	    if(!fa[i])
		{
			if(!dfs(i))
			{
				puts("-1");
				return 0;
			}
		}
	cout<<cnt<<endl;
	for(int i=1;i<=cnt;i++)
	    printf("%d\n",c[i]);
	return 0;
}
