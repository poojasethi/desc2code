#include<bits/stdc++.h>
using namespace std;
vector<int>v[100010];
bool fa[100010];
int aim[100010];
int ans[100010];
bool ok;
int num;
void dfs(int current,int fin)
{
    for(int i=0;i<v[current].size();i++)
    {
        int t=v[current][i];
        if(t==aim[t])
        {
            ans[num++]=t;
            dfs(t,t);
        }
        else if(aim[t]==fin)
            dfs(t,fin);
        else
        {
            ok=true;
            return;
        }
    }
}
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    int s,t;
    while(m--)
    {
        scanf("%d%d",&s,&t);
        fa[t]=true;
        v[s].push_back(t);
    }
    for(int i=1; i<=n; i++)
        scanf("%d",&aim[i]);
    for(int i=1; i<=n; i++)
    {
        if(!fa[i])
        {
            ans[num++]=i;
            dfs(i,i);
        }
    }
    if(ok)
        cout<<-1<<endl;
    else
    {
        printf("%d\n",num);
        for(int i=num-1; i>=0; i--)
            printf("%d\n",ans[i]);
    }
    return 0;
}
