#include <cstdio>
#include <cmath>
#include <cctype>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstdlib>
#include <vector>
#define N 100101
using namespace std;
int tot, n, m, dep[N], c[N], out[N], vis[N], a[N];
int len;
int in[N];
vector<int> g[N];

void dfs(int u, int k){
	dep[u] = k;
	int i, j;
	for (i = 0; i < g[u].size(); ++ i){
		int v = g[u][i];
		dfs(v, k + 1);
		if (a[v] != v){
			if (a[u] != a[v]) {
				printf("-1\n");
				exit(0);
			}
		}
	}
	if (vis[u])  out[++ tot] = u;
}

int  main(){
	int i, j, x, y;
	scanf("%d%d", &n, &m);
	//printf("%d %d\n", n, m);
	for (i = 1; i <= m; ++ i){
		scanf("%d%d", &x, &y);
		g[x].push_back(y);
		in[y]++;
	}
	for (i = 1; i <= n; ++ i){
		scanf("%d", & a[i]);
		vis[a[i]] = 1;
	}
	for (i = 1; i <= n; ++ i){
		if (in[i] == 0) dfs(i, 0);
	}
	printf("%d\n", tot);
	for (i = 1; i <= tot; ++ i)  printf("%d\n", out[i]);
	return 0;
}
