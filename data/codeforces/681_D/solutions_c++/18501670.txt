#include <bits/stdc++.h>

using namespace std;
const int maxn=100010;
int n,m,c[maxn],d[maxn],h[maxn],l[maxn],k,mi[maxn];
vector <int> a[maxn];
bool cq[maxn];


void dfs(int u)
{
    mi[u]=h[c[u]];
    for (int i=0;i<(int)a[u].size();++i)
    {
        int v=a[u][i];
        h[v]=h[u]+1;
        dfs(v);
        if (c[v]!=v)
        {
            if (c[v]!=c[u])
            {
                cout<<-1;
                exit(0);
            }
        }
    }
    if (cq[u]) l[++k]=u;
}

int main()
{
    cin>>n>>m;
    for (int i=1;i<=m;++i)
    {
        int u,v;
        cin>>u>>v;
        a[u].push_back(v);
        ++d[v];
    }
    for (int i=1;i<=n;++i)
    {
        cin>>c[i];
        cq[c[i]]=true;
    }
    for (int i=1;i<=n;++i)
        if (d[i]==0)
        {
            h[i]=1;
            dfs(i);
        }
    cout<<k<<"\n";
    for (int i=1;i<=k;++i) cout<<l[i]<<"\n";
    return 0;
}
