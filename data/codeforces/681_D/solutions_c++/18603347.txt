#include<bits/stdc++.h>
using namespace std;

std::vector<int> out[100001];
bool done[100001];
int a[100001], in[100001];
std::vector<int> v;

int dfs(int q) {
	std::vector<int> re;
	int x;
	done[q]=1;
	for(int i = 0; i < out[q].size(); i++) {
		x = dfs(out[q][i]);
		if(x!=out[q][i]) re.push_back(x);
		else v.push_back(x);
	}
	for (int i = 0; i < re.size(); ++i)
	{
		if(re[i]!=a[q]) return -1;
	}
	return a[q];
}

int main(int argc, char const *argv[])
{
	int n,m,p,q;
	cin>>n>>m;
	for (int i = 0; i < m; ++i)
	{
		cin>>p>>q;
		p--;
		q--;
		out[p].push_back(q);
		in[q]++;
	}
	for (int i = 0; i < n; ++i)
	{
		cin>>a[i];
		a[i]--;
	}
	for (int i = 0; i < n; ++i)
	{
		if(!done[i] && in[i]==0) { p = dfs(i); v.push_back(p); }
	}
	for (int i = 0; i < v.size(); ++i)
	{
		if(v[i]==-1) {
			cout<<v[i]<<endl;
			return 0;
		}
	}
	cout<<v.size()<<endl;
	for (int i = 0; i < v.size(); ++i)
	{
		cout<<v[i]+1<<endl;
	}
	return 0;
}
