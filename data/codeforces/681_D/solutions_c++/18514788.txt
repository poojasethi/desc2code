#include <iostream>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<iostream>
#include<algorithm>
#include<cstdio>
#include<vector>

#define Nmax 100000

using namespace std;

vector <int> ad[Nmax],ans;
int pa[Nmax],A[Nmax],visit[Nmax];

void DFS(int u) {
    if (visit[u]==1) return;
    visit[u]=1;
    for(int i=0;i<ad[u].size();i++) DFS(ad[u][i]);
    if (A[u]==u) ans.push_back(u);
    else
        if (A[u]!=A[pa[u]]) {
            cout << -1;
            exit(0);
        }
}

int main()
{
    //freopen("test.txt","r",stdin);
    int n,m,q,p;
    cin >> n >> m;
    for(int i=0;i<m;i++) {
        cin >> p >> q;
        pa[--q]=--p;
        ad[p].push_back(q);
    }
    for(int i=0;i<n;i++) {
        cin >> A[i];
        A[i]--;
    }
    for(int i=0;i<n;i++)
        if (pa[i]==0) DFS(i);
    cout << ans.size() << endl;
    for(int i=0;i<ans.size();i++) {
        cout << ans[i]+1 << endl;
    }
    return 0;
}
