using namespace std;
int e[110000],last[110000],pre[110000];
int v[110000];
int way[110000];
int ans;
int flag;
int num;
int n,m;
int now;
int fa[110000];
void dfs(int x)
{
	for (int i=last[x];i;i=pre[i])
	{
		dfs(e[i]);
		if (v[e[i]]!=-1 && v[x]!=v[e[i]]) flag=1;
	}
	if (v[x]==x)
	{
		way[++ans]=x;
		v[x]=-1;
	}

}
int main()
{
	scanf("%d %d",&n,&m);
	for (int i=1;i<=m;i++)
	{
		int x,y;
		scanf("%d %d",&x,&y);
		e[i]=y;
		pre[i]=last[x];
		fa[y]=1;
		last[x]=i;
	}
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&v[i]);
	}
	for (int i=1;i<=n;i++)
	{
		if (!fa[i]) 
		{
			dfs(i);
		}
	}
	if (flag)
	{
		printf("-1\n");
	}
	else
	{
		printf("%d\n",ans);
		for (int i=1;i<=ans;i++)
			printf("%d\n",way[i]);
	}
	return 0;
}
