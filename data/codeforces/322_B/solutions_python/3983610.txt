

if __name__=='__main__':
    inp = raw_input()
    arr = inp.split(" ")
    r = int(arr[0])
    g = int(arr[1])
    b = int(arr[2])
    rg = min(r,g,b)
    if rg > 1:
        rg = 3
    else:
        rg=rg+1
    ans = 0
    for i in range(rg):
        ans = max(ans,((r-i)//3)+((g-i)//3)+((b-i)//3)+i)
    print ans
