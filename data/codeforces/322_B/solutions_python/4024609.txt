import sys

num = map(int,raw_input().split(" "))

ans = -1

def Work():
	global ans
	for i in range(0,4):
		m = min(num)
		if m - i >0:
			m = m-i
		r = num[0]
		g = num[1]
		b = num[2]
		
		r = r -m;
		g = g -m;
		b = b- m;

		tmp = m
		tmp = tmp + r//3
		tmp = tmp + g//3
		tmp = tmp + b//3

		ans = max(ans,tmp)
		


if __name__=="__main__":
	Work()
	print ans
