rgb = map(int, raw_input().split())

total = 0
rest = [0, 0, 0]
for i in [0, 1, 2]:
    simple = (rgb[i] // 3) - 1
    if simple >= 0:
        total += simple
        rest[i] = rgb[i] - simple * 3

m = min(rest)
l_total = 0
for i in [0, 1, 2]:
    if rest[i] > 3 or (rest[i] == 3 and m < 2):
        l_total += 1
        rest[i] -= 3    
l_total += min(rest)
        
print total + max(m, l_total)
