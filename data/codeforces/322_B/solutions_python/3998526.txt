import sys

def solve(r, g, b):
    res = 0
    res += r / 3
    res += g / 3
    res += b / 3
    if 0 in [r, g, b]:
        return res

    ostat = [r%3, g%3, b%3]
    ostat.sort(reverse = True)
    res += min(ostat)

    if ostat[0] == 2 and ostat[1] == 2 and ostat[2] == 0:
        res += 1

    return res

r, g, b = map(int, sys.stdin.readline().split())

print solve(r, g, b)
