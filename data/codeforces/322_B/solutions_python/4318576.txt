R = lambda : map(int ,raw_input().split())
r,g,b = R()
total = 0
a,c = divmod(r,3)
if c == 0 and a>0:
    a-=1
total+=a
r-=a*3

a,c = divmod(g,3)
if c == 0 and a>0:
    a-=1
total+=a
g-=a*3

a,c = divmod(b,3)
if c == 0 and a>0:
    a-=1
total+=a
b-=a*3
temp = 0
temp+=r/3
temp+=g/3
temp+=b/3
if temp<2:
    total +=min(r,g,b)
else:
    total+=temp
print total
