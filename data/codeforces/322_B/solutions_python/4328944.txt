from collections import Counter
r, g, b = map(int, raw_input().split())
base = min(r, g, b)
r -= base
g -= base
b -= base
tail = r / 3 + g / 3 + b / 3
r %= 3
g %= 3
b %= 3
c = Counter([r, g, b])
if c[2] >= 2 and base >= 1:
  base -= 1
  tail += c[2]
elif c[1] >= 2 and c[0] == 0 and base >= 2:
  base -= 2
  tail += 3
print base + tail
