flower=map(int,raw_input().split())
mix_bouq=0
maxim=0


for i in range(3):
    if(maxim < flower[0]/3 + flower[1]/3 + flower[2]/3 + mix_bouq):
        maxim = flower[0]/3 + flower[1]/3 + flower[2]/3 + mix_bouq

    if(flower[0]>0 and flower[1]>0 and flower[2]>0):
        mix_bouq+=1
        flower[0]-=1
        flower[1]-=1
        flower[2]-=1


print maxim
