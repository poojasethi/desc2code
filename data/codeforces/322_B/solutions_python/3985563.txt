
def main(data=None):



    if data is None:

        data  = [int(s) for s in (raw_input()).split()]




    mixed = min(data)

    if mixed==0:

        return sum([d//3 for d in data])

    count2 = [mixed -k +sum([(d-mixed + k) //3  for d in data]) for k in range(3)]

    return max(count2)

if __name__ == "__main__":
    print main()
