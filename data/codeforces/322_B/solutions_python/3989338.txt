# B-div2
def solve():
    r, g, b = map(int, raw_input().strip().split())
    if r+g+b < 3: return 0
    ans = -1
    for mix in xrange(3):
        if r < mix or g < mix or b < mix: 
            tmp = 0
        else:
            tmp = mix + (r-mix)/3 + (g-mix)/3 + (b-mix)/3
        ans = max(ans, tmp)
    return ans

if __name__ == '__main__':
    print solve()
