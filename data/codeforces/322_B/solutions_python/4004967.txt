def solve(r,g,b):
  x = r/3
  x+= g/3
  x+= b/3
  return x

R,G,B = map(int,raw_input().split())
sol = min(R,G,B)

if sol == 0:
  sol += solve(R,G,B)
elif sol == 1:
  sol += max(solve(R-sol,G-sol,B-sol),solve(R,G,B)-1)
else:
  sol += max(solve(R-sol,G-sol,B-sol),solve(R-sol+1,G-sol+1,B-sol+1)-1, solve(R-sol+2,G-sol+2,B-sol+2)-2)
print sol
