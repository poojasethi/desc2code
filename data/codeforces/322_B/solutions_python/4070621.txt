#!/usr/bin/python
R = lambda:raw_input().split();
a, b, c = map(int, R())
ret = a/3 + b/3 + c/3 + min(a%3, b%3, c%3)
if a/3 > 0:
	ret = max(ret, a/3-1 + b/3 + c/3 + min(b%3, c%3))
if b/3 > 0:
	ret = max(ret, a/3 + b/3-1 + c/3 + min(a%3, c%3))
if c/3 > 0:
	ret = max(ret, a/3 + b/3 + c/3-1 + min(a%3, b%3))
m = min(a, b, c)
a, b, c = map(lambda x: x-m, [a, b, c])
ret = max(ret, m + a/3 + b/3 + c/3)
print ret
