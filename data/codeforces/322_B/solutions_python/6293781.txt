k=raw_input()
t=k.split(" ",2)
p=[0]*3
sum=0
r=int(t[0])
g=int(t[1])
b=int(t[2])

if r>2 and g>2 and b>2:
    p[2]=2+int((r-2)/3)+int((g-2)/3)+int((b-2)/3)
if r>1 and g>1 and b>1:
    p[1]=1+int((r-1)/3)+int((g-1)/3)+int((b-1)/3)

p[0]=int(r/3)+int(g/3)+int(b/3)
for i in p:
    if sum<i:
        sum=i
print sum
