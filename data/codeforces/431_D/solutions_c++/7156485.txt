#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <stdio.h>
using namespace std;
long long w;
long long l(long long q){
  long long a[64],s[64],e,r,c=0;
  q++;
  for(e=0;e<64;e++){
   s[e]=0;
   a[e]=q%2;
   q/=2;}
  for(e=63;e>=0;e--){
   for(r=62;r>=0;r--)
    s[r+1]+=s[r];
   if(a[e]){
    s[c]++;
    c+=a[e];}}
  return s[w];
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  long long q,e,r,t;
  cin>>q>>w;
  e=((long long)1)<<60;
  for(r=((long long)1)<<59;r;r>>=1){
   t=e-r;
   if(l(t*2)-l(t)>=q)
    e=t;}
  cout<<e;
  return 0;}
