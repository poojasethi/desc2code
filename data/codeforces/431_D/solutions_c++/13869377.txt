#include<bits/stdc++.h>

using namespace std;

#define mp make_pair
#define pb push_back
#define st first
#define nd second
#define MAX 1e18+6

typedef long long int lli;
typedef pair < int,int > pii;

lli m,k;
lli com[100][100];

void DP()
{	
	
	com[0][0] = 1;
	for(int i=1; i<=64; i++)
	{
		com[i][0]=1;
		com[i][1]=i;
		for(int j=1; j<=i; j++)
			com[i][j] =  com[i-1][j] + com[i-1][j-1];
	}
}
lli f(lli n)
{
	lli res=0;
	int t=0,i=62;
	 while(i>=0)
	 {
		 if((n & ((lli)1<<i)) && k>=t)
			 res+=com[i][k-t],t++;
		 i--;
	 }
	return res+(t==k);
}

int main ()
{
	DP();
	cin>>m>>k;
	lli bas=1;
	lli son=MAX;
	

	while(1)
	{
		lli orta = (bas + son) / 2;
		lli res=f(orta<<1) - f(orta);
		if(res==m)
		{
			cout<<orta;
			break;
		}
		else 
		{
			if(res>m) 
				son=orta-1; 
			else
				bas=orta+1;	
		}
	}
	
}
