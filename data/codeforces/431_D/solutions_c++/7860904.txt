#include<cstdio>
#include<cstring>
#define LL unsigned long long
#define N 99
LL m,k,C[N][N];
void init(){
	for(int i=0;i<N;++i){
		C[i][0]=1;
		for(int j=1;j<=i;++j){
			C[i][j]=C[i-1][j-1]+C[i-1][j];
		}
	}
	//printf("%I64d %I64d ...\n",C[5][2],C[5][3]);
}
LL getct(LL x){
	LL r=0;
	int cur=0;
	for(int i=63;i>=0;--i){
		if(x&(1LL<<i)){
			if(k>=cur)r+=C[i][k-cur];
			++cur;
		}
	}
	if(cur==k)++r;
	return r;
}
LL ck(LL x){
	LL a=getct(2LL*x);
	LL b=getct(x);
	return a-b;
}
int main(){
	init();
	scanf("%I64d %I64d",&m,&k);
//	printf("%I64d %I64d",m,k);
	LL l=1LL,r=1000000000000000000,md,ans=-1;
	while(l<=r){
		md=(l+r)>>1LL;
		LL ct=ck(md);
		if(ct>=m)r=md-1;
		else l=md+1;
		if(ct==m){
			ans=md;
			break;
		}
	}
	printf("%I64d\n",ans);
	return 0;
}
