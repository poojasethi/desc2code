#include<stdio.h>
#define LIM 1000000000000000000LL

long long Comb[62][62];

void init(int n){
    for(int i=0;i<=n;i++){
        Comb[i][0]=1;Comb[i][i]=1;
        for(int j=1;j<i;j++)Comb[i][j]=Comb[i-1][j]+Comb[i-1][j-1];
    }
}

long long C(int n,int k){
    if(k<0||k>n)return 0;
    return Comb[n][k];
}

long long solve(long long x,int k){
    long long ans=0;
    for(int i=61;i>=0;i--)if(x>>i&1){
        ans+=C(i,k);
        k--;
    }
    if(k==0)ans++;
    return ans;
}

int main(){
    long long m;
    int k;
    scanf("%I64d %d",&m,&k);
    init(61);
    long long l=0,r=LIM;
    while(r-l>1){
        long long mid=(l+r)/2;
        if(solve(mid*2,k)-solve(mid,k)>=m)r=mid;
        else l=mid;
    }
    printf("%I64d\n",r);
    return 0;
}
