#include <cstdio>

typedef unsigned usd;
typedef unsigned long long ull;

usd k;
ull m,l,r,n,t,c[64][64];

void ini()
{
	c[0][0]=1;
	for (usd i=1;i<64;++i)
	{
		c[i][0]=1;
		for (usd j=1;j<=i;++j) c[i][j]=c[i-1][j-1]+c[i-1][j];	
	}
}
ull com(usd n,usd k)
{
	if (k<=n) return c[n][k];
	else return 0;
}
ull cnt(ull x)
{
	usd s=0;
	ull cnt=0;
	for (usd i=63;i!=-1;--i) if (x&(1ull<<i)) cnt+=com(i,k-s++);
	cnt+=com(0,k-s);
	return cnt;
}
int main()
{
	scanf("%I64u%u",&m,&k);
	ini();
	l=1;
	r=0XFFFFFFFFFFFFFFF;
	while (l<r)
	{
		n=l+r>>1;
		t=cnt(n<<1)-cnt(n);
		if (t<m) l=n+1;
		else if (t>m) r=n;
		else break;
	}
	printf("%I64u",n);
	return 0;
}
