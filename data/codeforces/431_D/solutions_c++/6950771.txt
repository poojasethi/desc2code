#include<stdio.h>
#include<algorithm>
#include<string.h>
#include<stdlib.h>
#define LL long long
using namespace std;
LL c[65][65];
void pre()
{	
	c[0][0] = 1;
	for(int i=1;i<65;i++){
		c[i][0] = 1;
		for(int j=1;j<=i;j++){
			c[i][j] = c[i-1][j]+c[i-1][j-1];	
		}
		
	}
}
LL cal(LL x,int k)
{
	LL re =0 ;
	int cnt = 0;
	for(int i=63;i>=0;i--)
	{
		if(x&((LL)1<<i)){
			if(k>=cnt)
				re+=c[i][k-cnt]; 
			cnt++;
		}
	}
	return re+(cnt==k);
}
int main ()
{
	LL m;
	LL l,r,mi;
	int k;
	pre();
	scanf("%I64d %d",&m,&k);
	l =1 ;
	r = 1e18;
	while(l<r)
	{
		mi = (l+r)>>1;
		if(cal(2*mi,k)-cal(mi,k)>=m){
			r = mi;
		}
		else
			l = mi+1;
	}
	printf("%I64d",l);
	return 0;
}
