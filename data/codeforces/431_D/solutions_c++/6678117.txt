#include<cstdio>
#include<algorithm>
#include<cstdlib>
#include<cstring>
#include<cmath>
using namespace std;
typedef __int64 LL;
LL m;
int k;
LL c[65][65];
bool check(LL p,int pos) {
	while(pos) {
		p>>=1;
		pos--;
	}
	return p&1ll;
}
LL f(LL p) {
	int ct=0;
	LL ans=0ll;
	for(int pos=63;pos>=0;pos--) {
		if(check(p,pos)) {
			if(k>=ct) ans+=c[pos][k-ct];
			ct++;
		}
	}
	if(ct==k) ans++;
	return ans;
}
int main() {
	c[0][0]=c[1][0]=c[1][1]=1ll;
	for(int i=2;i<=64;i++) {
		c[i][0]=c[i][i]=1ll;
		for(int j=1;j<i;j++) c[i][j]=c[i-1][j-1]+c[i-1][j];
	}
	scanf("%I64d%d",&m,&k);
	LL l=1ll,r=1000000000000000000ll,ans=-1ll;
	while(l<=r) {
		LL mid=(l+r)>>1;
		if(f(2*mid)-f(mid)>=m) {
			ans=mid;
			r=mid-1ll;
		}
		else l=mid+1ll;
	}
	printf("%I64d\n",ans);
	return 0;
}
