#include<bits/stdc++.h>
using namespace std;

long long int C[100][100];
int k;

int bcount(long long x)
{
	int ans=0;
	for(;x>0;x&=(x-1))ans++;
	return ans;
}

long long f(long long x)
{
	int y=0;
	long long ans=0;
	for(int i=63; i>=0 && y<=k; i--)
	{
		if((x>>i)&1)
		{
			ans += C[i][k-y];
			y++;
		}
	}
	if(bcount(x)==k)ans++;
	return ans;
}

int main()
{
	for(int i=0; i<65; i++)
	{
		C[i][0] = 1;
		for(int j=1; j<=i; j++)
			C[i][j] = C[i-1][j] + C[i-1][j-1];
	}
	long long m,l,r,x,ans;
	cin>>m>>k;
	l = 1; r = 1e18;
	while(l<r)
	{
		x = l + (r-l)/2;
		ans = f(2*x)-f(x);
		//cout<<x<<" "<<f(2*x)<<" "<<f(x)<<endl;
		if(ans<m)l = x+1;
		else if(ans>m) r = x;
		else break;
	}
	cout<<x<<endl;
	return 0;
}
