#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxb=63;
const long long INF=1e18;
long long c[maxb][maxb]={0};
int K;
long long m,ans=0;
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j;
	c[0][0]=1;
	for(i=1;i<maxb;i++)
	{
		c[i][0]=1;
		for(j=1;j<=i;j++)
			c[i][j]=c[i-1][j]+c[i-1][j-1];
	}
	scanf("%I64d%d",&m,&K);
	m--;
	//if(m==-1) { printf("1\n");return 0; }
	for(i=1;i<maxb;i++)
	{
		ans<<=1;
		if(c[maxb-i][K]-c[maxb-i-1][K]<=m)
		{
			ans++;
			m-=c[maxb-i][K]-c[maxb-i-1][K];
			K--;
		}
	}
	printf("%I64d\n",min(INF,ans+1));
	return 0;
}
