#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const ll INF = 1LL << 60;
ll M, K;

ll binom(ll p, int k){
  ll res = 1;
  for(int i = 1; i <= k; i++){
    res *= (p - i + 1);
    res /= i;
  }
  return res;
}

ll get(ll x, int k){
  if(k == 0) return 1;
  if(x == 0) return 0;
  int p = 63 - __builtin_clzll(x);
  return get(x - (1LL << p), k - 1) + binom(p, k);
}

bool check(ll m){
  return get(2 * m, K) - get(m, K) < M;
}

int main(){
  cin >> M >> K;
  ll lo = 0, hi = INF, mid = (lo + hi) / 2;
  while(lo < mid){
    (check(mid) ? lo : hi) = mid;
    mid = (lo + hi) / 2;
  }
  cout << hi << '\n';
}
