#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long

using namespace std;

LL m,dp[66][66];
int k;

LL f(LL lim){
  LL res=0;
  int kk=k;
  FD(i,60,0){
    if(lim&(1LL<<i)){
      res+=dp[i][kk];
      kk--;
      if(kk<0) break;
    }
  }
  //printf("%lld: %lld\n",lim,res);
  return res;
}
int main(){
  dp[0][0]=1;
  FI(i,1,60) FI(j,0,i){
    if(j==0 || j==i) dp[i][j]=1;
    else dp[i][j]=dp[i-1][j]+dp[i-1][j-1];
  }
  
  scanf("%lld %d",&m,&k);
  LL lt=0,rt=(1LL<<60);
  while(rt-lt>1){
    LL md=(lt+rt)>>1;
    if(f(2*md)-f(md)>=m) rt=md;
    else lt=md;
  }
  printf("%lld\n",rt);
  return 0;
}
