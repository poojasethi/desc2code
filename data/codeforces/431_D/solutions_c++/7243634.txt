#include<bits/stdc++.h>

using namespace std;

int k ;
long long c[65][65];

long long C (int x , int y)
{
		if(c[x][y] != -1)return c[x][y];
		if(y == x || y == 0)return 1;
		if(y > x)return 0;
		return c[x][y] = C(x-1,y-1) + C(x-1,y);
} 
long long F(long long x)
{
	int cnt = 0 ; long long ans = 0;
	for(int i = 63 ; i >= 0 && k - cnt >= 0  ;i--)
	{if(((x>>i)&1))ans += C(i,k-cnt),cnt++;}
	return ans;
}
int main() {
    
	long long m;
	cin >> m >> k;

	memset(c,-1,sizeof c);

	long long lo = 1; long long hi = 1e18;
	long long out = -1;
	while(lo <= hi)
	{
		long long mid = lo + ((hi - lo)/2);		
		long long q = F(2*mid) - F(mid);
		if(q  >= m)hi= mid - 1;
		else if(q < m)lo = mid + 1 ;
		if(q == m) {out = mid ;}
	}
	cout << out << endl;
}
