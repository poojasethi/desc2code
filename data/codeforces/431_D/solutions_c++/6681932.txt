#include<stdio.h>
long long  n,k;
long long  su[80][80];
long long  cont( long long  x)
{
	long long  i;
	long long  re=0;
	long long  mu=0;
	for(i=62;i>=1;i--)
	if((1LL<<i)&x)
	{
		if((k-mu)>=0)re+=su[i][k-mu];
		mu++;
	}
	return re;

}

int main()
{
int i,j;
for(i=0;i<=64;i++)
su[i][0]=1;
for(i=1;i<=64;i++)
for(j=1;j<=i;j++)
su[i][j]=su[i-1][j]+su[i-1][j-1];

scanf("%lld%lld",&n,&k);
long long l,r;
l=1;
r=1LL<<62;
long long  mid;
long long ans;
   while(l<=r)
   {
    mid=(l+r)/2;
    long long  u=cont(2*mid)-cont(mid);
    if(n<=u)ans=mid,r=mid-1;
	else l=mid+1;
    }
    printf("%lld\n",ans);

}
