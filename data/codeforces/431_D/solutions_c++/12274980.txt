#include<bits/stdc++.h>
using namespace std;
typedef long long i64;
#define M 63
i64 ncr[65][65];
i64 m,k;
void nc()
{
  for(int i=0;i<=63;i++)
    ncr[i][i]=ncr[i][0]=1;
  for(int i=1;i<=63;i++)
    for(int j=1;j<i;j++)
       ncr[i][j]=(ncr[i-1][j]+ncr[i-1][j-1]);
}
i64 f(i64 x)
{
   i64 ans=0;
   i64 cnt=0;
   int c=k;
   for(int i=63;i>=0 && c;i--)
   {
     if(x&(1LL<<i))ans+=ncr[i][c--],cnt++;
   }
   if(cnt==k)ans++;
   return ans;
}
int main()
{
   cin>>m>>k;
    i64 lo,hi;
    lo=1;
    nc();
    hi=1000000000000000000LL;
    while(lo<hi)
    {
        i64 mid=(lo+hi)/2;
        // cout<<lo<<" "<<hi<<endl;
        if((f(mid*2)-f(mid))<m)lo=mid+1;
        else hi=mid;
    }
    cout<<lo<<endl;
   return 0;
}
