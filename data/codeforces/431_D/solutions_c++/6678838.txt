#include <iostream>
using namespace std;
typedef unsigned long long ull;
ull m;int k;
ull c[110][110];
ull cal(ull x)
{
    ull res=0;
    int gs=0,i;
    for (i=63;i>=0;i--)
    {
        if (!((x>>i)&1)) continue;
        res+=c[i][k-gs];
        gs++;if (gs>k) break;
    }
    return res;
}
int main()
{
    c[0][0]=1;
    for (int i=1;i<=100;i++)
    {
        c[i][0]=1;
        for (int j=1;j<=i;j++)
            c[i][j]=c[i-1][j-1]+c[i-1][j];
    }
    cin>>m>>k;
    if (m==0||k==1)
    {
        cout<<"1\n";
        return 0;
    }
    ull l=1,r=1ll<<60;
    while (l+1!=r)
    {
        ull mid=(l+r)/2;
        ull ans=cal(mid*2+1)-cal(mid+1);
        if (ans<=m) l=mid;else r=mid;
        if (ans==m)
        {cout<<mid;return 0;}
    }
    return 0;
}
