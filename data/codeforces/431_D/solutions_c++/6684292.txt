#include <iostream>
using namespace std;

long long c[70][70],l,r,mid,x,m,b[70];
int k,i,j;

long long F(long long x)
{
	//cout<<"F("<<x<<")"<<endl;
	long long ans=0; int i,cnt=0;
	for (i=63;i>=0&&cnt<=k;--i)
		if (x&b[i]) ans+=c[i][k-cnt],++cnt;
	if (cnt==k) ++ans; //cout<<"ans="<<ans<<endl;
	return ans;
}

int main()
{
	cin>>m>>k;
	for (i=0;i<=63;++i) c[i][0]=1;
	for (i=1;i<=63;++i)
		for (j=1;j<=i;++j)
			c[i][j]=c[i-1][j]+c[i-1][j-1];
	b[0]=1; for (i=1;i<=63;++i) b[i]=b[i-1]*(long long)2;
	l=1; r=1e18+1;
	while (l<r)
	{
		mid=(l+r)>>1;
		x=F(mid<<1)-F(mid); //cout<<"x="<<x<<endl;
		if (x<m) l=mid+1; else r=mid;
	}
	cout<<l<<endl;
	return 0;
}
