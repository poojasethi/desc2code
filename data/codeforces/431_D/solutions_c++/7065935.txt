#include <stdio.h>

typedef long long ll;

ll D[70][70]={0};

ll f(ll n, ll k)
{
	ll ret=0,p=0;
	while ((1LL<<p) < n) { p++; }
	while (n>0)
	{
		if (n>=(1LL<<(p-1))) {
			ret += D[p][k--];
			n-=(1LL<<(p-1));
		}
		p--;
	}
	return ret;
}

int main()
{
	ll m,k;
	scanf("%I64d%I64d",&m,&k);
	
	
	D[0][0] = 1;
	for (int i=1; i<=64; i++) {
		for (int j=1; j<=i; j++) {
			D[i][j] = D[i-1][j] + D[i-1][j-1];
		}
	}
	ll l=1,h=1e18L, ret;

	while (l<=h)
	{
		ll n = (l+h)/2;
		ll cnt = f(n,k);
		if (cnt == m) {
			ret = n; break;
		}
		if (cnt > m) {
			h = n-1;
		}
		else {
			l = n+1;
		}
		
	}

	printf("%I64d\n", ret);
}
