#include <iostream>
using namespace std;

long long binom[61][62];

long long cnt(int k, long long n)
{
    long long res = 0, nbits = 0;
    for (int i = 60; i >= 0; i--)
        if (((n >> i) & 1) == 1)
            res += binom[i][k==nbits?0:k-nbits++];
    if (nbits == k) res++;
    return res;
}

int main()
{
    long long m, low = 1, high = 1e18;
    int k;
    cin >> m >> k;

    binom[0][0] = 1;
    for (int i = 1; i < 61; i++) {
        binom[i][0] = binom[i][i] = 1;
        for (int j = 1; j < i; j++)
            binom[i][j] = binom[i-1][j-1] + binom[i-1][j];
    }
    
    while(low < high) {
        long long mid = (low + high) / 2;
        if (cnt(k, 2*mid) - cnt(k, mid) < m) low = mid+1;
        else high = mid;
    }
    cout << low << endl;
    return 0;
}
