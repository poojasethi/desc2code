#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;

long long m;int k;
long long C[70][70];

long long c(int x,int y){
  if(y<0||y>x) return 0;
  else return C[x][y];
}

int x[80],sz;
long long ask(long long v){   
  sz=0;
  for(;v;v/=2)x[++sz]=v&1;
  long long ans=0;
  int j=0;
  for(int i=sz;i>=1;i--){
    if(x[i]==1)ans+=c(i-1,k-j);
    if(x[i]==1) j++;
  }
  return ans;
}

int main(){
  for(int i=0;i<=65;i++){
    C[i][0]=1;
    for(int j=1;j<=i;j++)
      C[i][j]=C[i-1][j]+C[i-1][j-1];
  }  
  long long l=1,r=1000000000;
  r=r*r;
  scanf("%I64d%d",&m,&k);
  while(l<r-1){
    long long mid=(l+r)>>1;
    if(ask(2*mid)-ask(mid)<=m)l=mid;
    else r=mid;
  }
  printf("%I64d\n",l);
}
