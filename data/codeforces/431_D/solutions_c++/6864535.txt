#include <iostream>
#include <cstdio>
using namespace std;
typedef unsigned long long ll;
ll m,k,n,l = 1, r=1000000000000000100L, dp[65][65];
ll f(ll a)
{
    ll cnt = 0, ans = 0;
    for(int i = 63; i>=0; i--)
        if((a >> i) & 1LL)
        {
           if(k >= cnt)
              ans+=dp[i][k-cnt];
           cnt++;
        }
    if(cnt==k) ans++;
    return ans;
}
int main()
{
    cin >> m >> k;
    for(int i = 0; i <= 64; i++) dp[i][0]=1;
    for(int i = 1; i <= 64; i++)
        for(int j = 1; j <= 64; j++)
            dp[i][j]=dp[i-1][j-1]+dp[i-1][j];
    while(l<r)
    {
        ll mid = l + (r - l)/2;
        if(f(mid << 1) - f(mid) < m)
            l = mid+1;
        else
            r = mid;
    }
    cout << l << endl;
}
