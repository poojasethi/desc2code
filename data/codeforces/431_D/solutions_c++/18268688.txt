//Gonna be one hit
#include <bits/stdc++.h>

using namespace std;
typedef unsigned long long ull;
ull C[65][65];
int main()
{
    ull m;
    int k;
    cin >> m >> k;
    if(m == 0)
    {
        cout << 1;
        return 0;
    }
    C[0][0] = 1;
    for(int i = 1 ; i <= 64 ; i++)
    {
        C[i][0] = 1;
        for(int j = 1 ; j <= i ; j++)   C[i][j] = C[i-1][j]+C[i-1][j-1];
    }
    ull ans = 0;
    for(int i = 64 ; i > 0 ; i--)
    {
        ans *= 2;
        if((m > C[i][k]-C[i-1][k]) || (i == k))
        {
            if(m == 0)  continue;
            ans++;
            m -= C[i][k]-C[i-1][k];
            k--;
        }
    }
    cout << ans;
}
