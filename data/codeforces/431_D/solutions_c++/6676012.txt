#include <iostream>

using namespace std;

#define MAXL 63
#define MAXN 1000000000000000000ull
unsigned long long c[MAXL + 2][MAXL + 2];
unsigned long long ans = 0;
unsigned long long m;
void bt(int x, int r, unsigned long long p, unsigned long long o)
{
	if (x < r || r < 0 || x == 0)
	{
		if (o == m)
			ans = p;
		return;
	}
	if (m - o >= c[x][r] && MAXN - p >= (1ull << x))
		bt(x - 1, r - 1, p + (1ull << x), o + c[x][r]);
	if (ans == 0)
		bt(x - 1, r, p, o);
}
int main()
{
	c[0][0] = 1;
	for (int i = 1; i <= MAXL; i++)
	{
		c[i][i] = c[i][0] = 1;
		for (int j = 1; j < i; j++)
			c[i][j] = c[i - 1][j] + c[i - 1][j - 1];
	}

	int k;
	cin >> m >> k;
	bt(MAXL, k - 1, 0, 0);
	if (ans == 0)
		cout << 1 << endl;
	else
		cout << ans << endl;
}
