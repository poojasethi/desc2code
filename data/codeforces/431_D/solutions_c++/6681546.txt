#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

typedef unsigned long long ULL;
int k;
ULL f[65][65],m;

ULL find(int n,int m){
	if (m<0) return 0;
	if (n<m) return 0;
	return f[n][m];
}

ULL find1(ULL x){
	ULL ans=0;
	int K=0;
	for (int i=63; i>=0; --i){
		if ((x>>i) & 1)
			ans=ans+find(i,k-K),++K;
	}
	return ans;
}

ULL check(ULL x){
	return find1(2*x)-find1(x);
}

int main(){
	for (int i=0; i<65; ++i)
		for (int j=0; j<=i; ++j)
			f[i][j]=(j==0)?1:(f[i-1][j]+f[i-1][j-1]);
	cin>>m>>k;
	ULL head=0,tail=(ULL)1<<63;
	while (head<tail-1){
		ULL mid=(head+tail)>>1;
		if (check(mid)>=m) tail=mid;
		else head=mid;
	}
	cout<<tail<<endl;
}
