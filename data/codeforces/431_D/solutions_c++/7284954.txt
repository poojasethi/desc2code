#include<iostream>
#include<set>
#include<algorithm>
#define pb push_back
#define forc(i,j,n) for(int(i)=(j);(i)<(n);++(i))
using namespace std;
long long m , k;
long long dp[70][70];
void dpBesaz(){
	dp[0][0] = 1;
	forc(i,0,70)
		dp[i][i] = dp[i][0] =1LL;
	forc(i,1,70)
		forc(j,1,i)
			dp[i][j] = dp[i-1][j-1]+dp[i-1][j];
}

long long count(long long a){
	long long ret = 0;
	while(a>0){
		int bit = __builtin_ctzll(a);
		a-=(1LL<<bit);
		int num = k - __builtin_popcountll(a);
		if(num >= 0)
			ret+=dp[bit][num];
	}
	return ret;
}


int main()
{
	cin >> m >>  k ;
	dpBesaz();
	long long lo = 0 , hi = (long long)(1e18)+100;
	while(hi - lo > 1){
		long long mid = (hi+lo)/2;
		if(count(2*mid+1) - count(mid+1) < m)
			lo = mid; 
		else
			hi = mid;

	}
	cout << hi << endl;
}
