#include <cstdio>
typedef unsigned long long ll;
ll l,r,mid,m;
int k;
ll C[64][64];

void init(){
    for(int i=0;i<64;i++){
        C[i][0]=1;
        for(int j=1;j<=i&&j<=k;j++)
            C[i][j]=C[i-1][j-1]+C[i-1][j];
    }
}

inline ll calc(ll x,int k){
    ll ans=0;
    x++;
    for(int i=63;i>=0;i--){
        if(x>>i&1){
            ans+=C[i][k],k--;
        }
        if(k<0) break;
    }
    return ans;
}

int main(){
    scanf("%I64u%d",&m,&k);
    init();
    l=0;
    r=ll(1e18)+5;
    while(r-l>1){
        mid=l+r>>1;
        ll t=calc(mid<<1,k)-calc(mid,k);
        if(m<=t)
            r=mid;
        else
            l=mid;
    }
    printf("%I64u\n",r);
}
