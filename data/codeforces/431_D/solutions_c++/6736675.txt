#include<cstdio>
using namespace std;
const int maxn=67;
typedef unsigned long long uLL;
uLL ile[maxn][maxn];
uLL solve(uLL n,int k)
{
	uLL w=0;
	for(int i=63;i>=0 && k>=0;i--)
		if((1ll<<i)&n)
		{
			w+=ile[i][k];
			k--;
		}
	return w;
}
int main()
{
	ile[0][0]=1;
	for(int i=1;i<maxn;i++)
		for(int j=0;j<maxn;j++)
			ile[i][j]=ile[i-1][j]+((j)?ile[i-1][j-1]:0);
	uLL m,k;
	scanf("%I64u%I64u",&m,&k);
	uLL p=1,e=1000000000000000007ll;
	while(p<e)
	{
		uLL mid=(p+e)/2;
		if(solve(2*mid,k)-solve(mid,k)<m)
			p=mid+1;
		else
			e=mid;
	}
	printf("%I64u\n",p);
}
