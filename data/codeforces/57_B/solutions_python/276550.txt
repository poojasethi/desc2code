n, m, k = map(int, raw_input().split())
A = []
for i in xrange(m):
    a, b, c = map(int, raw_input().split())
    A.append((a, -a + c, 1))
    A.append((b + 1, a - c, -1))
A.sort()
A.append((1000000000, 0, 0))
B = map(int, raw_input().split())
B.sort()
add = cnt = 0
sum = j = 0
for n in B:
    while A[j][0] <= n:
        add += A[j][1]
        cnt += A[j][2]
        j += 1
    sum += add + cnt * n
print sum
