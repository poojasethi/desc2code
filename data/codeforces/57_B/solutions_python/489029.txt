#!/usr/bin/env python
# -*- coding: utf-8 -*-
from collections import defaultdict
from itertools import imap

n, m, k = map(int, raw_input().split())
ga, gb, gc = defaultdict(list), defaultdict(list), list()
for i in xrange(m):
    a, b, c = map(int, raw_input().split())
    ga[a].append(i)
    gb[b].append(i)
    gc.append((a, b, c))
fld = list()
p = 0
t = 0
for i in xrange(1, n+1):
    for li in gb[i-1]:
        p -= gc[li][2] + gc[li][1] - gc[li][0]
        t -= 1
    p += t
    for li in ga[i]:
        p += gc[li][2]
        t += 1
    fld.append(p)
s = 0
for i in imap(int, raw_input().split()):
    s += fld[i-1]
print s
