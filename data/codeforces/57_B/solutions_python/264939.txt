x = [0 for i in range(0 , 100003)]
n , m , q = eval("[" + raw_input().replace(' ' , ',') + "]")
ans = 0
for i in range(0 , m):
    a , b , c = eval("[" + raw_input().replace(' ' , ',') + "]")
    x[a] += c
    x[a+1] += 1-c
    x[b+1] += a-b-c-1
    x[b+2] += b+c-a
for i in range(0 , 2):
    for j in range(1 , n+1):
        x[j] += x[j-1]
where = eval("[" + raw_input().replace(' ' , ',') + "]")
for loc in where:
    ans += x[loc]
print(ans)
