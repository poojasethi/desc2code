#!/usr/bin/python
n, m, k = map(int, raw_input().split())
A = [0] * (n + 2)
B = [0] * (n + 2)
C = [False] * (n + 2)
for i in xrange(m):
    a, b, c = map(int, raw_input().split())
    A[a] += -a + c
    A[b + 1] += a - c
    B[a] += 1
    B[b + 1] += -1
idx = map(int, raw_input().split())
for i in xrange(k):
    C[idx[i]] = True
result = 0
for i in xrange(1, n + 1):
    A[i] += A[i - 1]
    B[i] += B[i - 1]
    if C[i]:
        result += B[i] * i + A[i]
print(result)
