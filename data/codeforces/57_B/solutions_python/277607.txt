triple = lambda : [int(s) for s in raw_input().split()]
n,m,k = triple()
begin,count,end,grow = [[0]*(n+1) for i in xrange(4)]
for i in xrange(m):
    a,b,c = triple()
    begin[a]+= c
    end[b]+= c + b - a
    grow[a]+= 1
    grow[b]-= 1
add = 0
bricks = 0
for i in xrange(1,n+1):
    bricks += begin[i]
    bricks += add
    count[i] = bricks
    bricks -= end[i]
    add+= grow[i]
print sum(count[int(s)] for s in raw_input().split())
