#include<stdio.h>
#include<iostream>
#define MAXM 100001
using namespace std;
int A[MAXM];
int B[MAXM];
int C[MAXM];
int main()
{
	int i,N,M,K;
	scanf("%d %d %d",&N,&M,&K);
	for(i=0;i<M;i++)
	scanf("%d %d %d",&A[i],&B[i],&C[i]);
	long long int sum = 0;
	while(K--)
	{
		int x;
		scanf("%d",&x);
		//check each
		for(i=0;i<M;i++)
		{
			if(x>=A[i] && x<=B[i])
			sum += (x- A[i] + C[i]);
		}
	}
	cout << sum << endl;
	return 0;
}
