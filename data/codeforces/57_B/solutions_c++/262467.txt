#include <stdio.h>
#define nmax 100010
int n, m, k, i, j, x;
int a[nmax], b[nmax], c[nmax];
long long ans;
int main()
{
	 scanf("%d%d%d", &n, &m, &k);
	 for (i = 1; i <= m; i++) scanf("%d%d%d", &a[i], &b[i], &c[i]);
	 ans = 0;
	 for (i = 1; i <= k; i++)
	 {
		 scanf("%d", &x);
		 for (j = 1; j <= m; j++)
		 {
			 if ((a[j] <= x) && (x <= b[j])) ans += (c[j] + x - a[j]);
		 }
	 }
	 printf("%I64d", ans);
	 //scanf("%d", &n);
	 return 0;
}
