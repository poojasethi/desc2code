#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int x,y,z;
} a[N];
int i,j,k,n,m,K;
int64 an;
int main() {
	scanf("%d%d%d",&n,&m,&K);
	For(i,1,m) scanf("%d%d%d",&a[i].x,&a[i].y,&a[i].z);
	For(i,1,K) {
		scanf("%d",&k);
		For(j,1,m) if (a[j].x<=k&&a[j].y>=k) an+=a[j].z+k-a[j].x;
	}
	printf("%I64d\n",an);
	return 0;
}
