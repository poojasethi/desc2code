#include <iostream>

using namespace std;

typedef long long int lli;

int main()
{
	lli n, m, k;
	lli list[100020][3];
	lli i, x, j, l;
	lli sol;

	cin >> n >> m >> k;
	
	for ( i = 0; i < m; i++ ) {
		cin >> list[i][0] >> list[i][1] >> list[i][2];
	}	

	sol = 0;
	
	for ( i = 0; i < k; i++ ) {
		cin >> x;

		for ( j = 0; j < m; j++ ) {
			if ( x >= list[j][0] && x <= list[j][1] ) {
				sol += (x - list[j][0] + list[j][2]);
			}
		}
	}
	
	cout << sol << endl;
	
	return 0;
}
