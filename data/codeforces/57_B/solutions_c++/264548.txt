#include <iostream>

using namespace std;

int main() {
  int n, m, k;
  cin >> n >> m >> k;
  int v[m][3];
  long long res = 0;
  int a, b, c;
  for (int i = 0; i < m; i++)
    cin >> v[i][0] >> v[i][1] >> v[i][2];
  for (int i = 0; i < k; i++) {
    cin >> b;
    for (int j = 0; j < m; j++)
      if (b <= v[j][1] && b >= v[j][0])
      res += v[j][2] + b - v[j][0];
  }
  cout << res;
  return 0;
}
