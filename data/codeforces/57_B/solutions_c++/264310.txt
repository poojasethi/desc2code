#include <iostream>
#include <string>
#include <vector>
#include <list>
#include <set>
#include <map>
#include <algorithm>

using namespace std;

int main(){
  long long j, n, m, k, i, x, s = 0, a[101000][3];
  cin >> n >> m >> k;
  for(i = 0; i < m; i++){
    cin >> a[i][0] >> a[i][1] >> a[i][2];
  }
  for(i = 0; i < k; i++){
    cin >> x;
    for(j = 0; j < m; j++){
      if((x >= a[j][0]) && (x <= a[j][1])){
	s += (a[j][2] + x - a[j][0]);
      }
    }
  }
  cout << s << endl;
}
