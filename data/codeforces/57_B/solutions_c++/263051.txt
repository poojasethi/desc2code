#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;

int n, m, k, x;
int a[100005], b[100005], c[100005];

int main() {
	scanf("%d%d%d", &n, &m, &k);
	for (int i = 0; i < m; i++) {
		scanf("%d%d%d", a + i, b + i, c + i);
	}
	long long res = 0;
	for (int i = 0; i < k; i++) {
		scanf("%d", &x);
		for (int j = 0; j < m; j++) {
			if (a[j] <= x && x <= b[j]) {
				res += c[j] + x - a[j];
			}
		}
	}
	cout << res << endl;
	return 0;
}
