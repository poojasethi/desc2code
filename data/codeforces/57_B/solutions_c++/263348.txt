#include <iostream>
#include <cstdio>
using namespace std;
struct info
{
	int l,r,x;
};
long long sum=0;
info a[100005];
int main()
{
	int n,m,k,b;
	cin>>n>>m>>k;
	for(int i=1;i<=m;i++)scanf("%d%d%d",&a[i].l,&a[i].r,&a[i].x);
	for(int i=1;i<=k;i++)
	{
		scanf("%d",&b);
		for(int j=1;j<=m;j++)
			if (a[j].l<=b && a[j].r>=b)sum=sum+a[j].x+b-a[j].l;
	}
	cout<<sum;
	return 0;
}
