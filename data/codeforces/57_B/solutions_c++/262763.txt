#include<iostream>
#include<vector>
using namespace std;


long long L[1000000];
long long R[1000000];
long long S[1000000];

int main()
{
	long long N,M,Q;
	cin>>N>>M>>Q;
	for(int i = 0; i < M; ++i)
		cin>>L[i]>>R[i]>>S[i];
	long long s = 0;
	for(int i = 0; i < Q; ++i)
	{
		int k;
		cin>>k;
		for(int j = 0; j < M; ++j)
		{
			if(k>=L[j] && k<=R[j])
				s += (k-L[j]+1)+(S[j]-1);
		}
	}
	cout<<s<<endl;
}
