#include<stdio.h>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<iostream>
using namespace std;

int main(){
	int n,i,j,k,m,l;
	int a[100001], b[100001], c[100001];
	long long total;
	while(scanf("%d%d%d",&n,&m,&k)!=EOF){
		total = 0;
		for(i=1;i<=m;i++)
			scanf("%d%d%d",a+i,b+i,c+i);
		for(l=1;l<=k;l++){
			scanf("%d",&j);
			for(i=1;i<=m;i++)
				if(j>=a[i] && j<=b[i])
					total = total + c[i] + j - a[i];
		}
		cout<<total<<endl;
	}
}
