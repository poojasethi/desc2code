#include <iostream>

using namespace std;

const int MAXN = 1000 * 100 + 10;
int l[MAXN], r[MAXN], x[MAXN];
long long ans = 0;

int main()
{
	int n, m, k;
	cin >> n >> m >> k;
	for (int i = 0; i < m; i++)
		cin >> l[i] >> r[i] >> x[i];
	for (int i = 0; i < k; i++)
	{
		int cur;
		cin >> cur;
		for (int j = 0; j < m; j++)
		{
			if (cur >= l[j] && cur <= r[j])
				ans += x[j] + cur - l[j];
		}
	}
	cout << ans << endl;
	return 0;
}
