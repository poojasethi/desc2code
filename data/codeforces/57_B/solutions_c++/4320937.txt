#include "cstdio"
#include "iostream"

using namespace std;

typedef long long ll;

int n, m, k;
ll ret;
int a[100010], b[100010], c[100010];

int main() {
	scanf("%d%d%d", &n, &m, &k);
	for(int i =	1; i <= m; i++) 
		scanf("%d%d%d", a+i, b+i, c+i);
	ret = 0;
	for(int i = 1; i <= k; i++) {
		int x;
		scanf("%d", &x);
		for(int j = 1; j <= m; j++) 
			if (x >= a[j] && x <= b[j]) ret += c[j] + x-a[j];
	}
	printf("%I64d\n", ret);

	return 0;
}
