#include<iostream>
using namespace std;
int main()
{
    int a[100010],b[100010],c[100010],x,n,m,k;
    long long ans=0;
    cin>>n>>m>>k;
    for(int i=0;i<m;i++)
        cin>>a[i]>>b[i]>>c[i];
    for(int i=0;i<k;i++)
    {
        cin>>x;
        for(int j=0;j<m;j++)
        {
            if(x>=a[j] && x<=b[j])
                ans+=c[j]+(x-a[j]);
        }
    }
    cout<<ans<<endl;
    return 0;
}
