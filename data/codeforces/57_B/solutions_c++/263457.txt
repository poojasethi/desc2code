#include <iostream>

using namespace std;

typedef long long LL;
LL s[1000005];
LL A[100005];
LL B[100005];
LL C[100005];

int main() {
  LL n, m, k;
  cin >> n >> m >> k;
  for (int i = 0; i < m; ++i) {
    LL a, b, c;
    cin >> A[i] >> B[i] >> C[i];
  }
  LL total = 0;
  while (k--) {
    LL x; cin >> x;
    for (int i = 0; i < m; ++i) if (x >= A[i] && x <= B[i])
      total += C[i] + x - A[i];
  }
  cout << total << '\n';
  return 0;
}
