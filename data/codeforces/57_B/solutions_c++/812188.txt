#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int v[100010][3];
int main(void)
{
	int n,m,k;
	long long ans=0;
	scanf("%d%d%d", &n,&m,&k);
	for ( int i = 0; i<m; i++ ) scanf("%d%d%d",&v[i][0],&v[i][1],&v[i][2]);
	for ( int i = 0; i < k; i++ ){
		int b;
		scanf("%d", &b);
		for ( int j = 0; j < m; j++ )
			if ( b >= v[j][0] && b <= v[j][1] ) ans += v[j][2] + b - v[j][0];
	}
	printf("%I64d\n", ans);	
	return 0;
}
