
#include <math.h>
#include <string.h>
#include <iostream>
using namespace std;
int a[100002][3];
int main()
{
	int k, n, m, i, j, t, p;
	unsigned long long result = 0;
	cin >> n >> m >> k;
	for (j = 0; j < m; j++)	
		cin  >> a[j][0] >> a[j][1] >> a[j][2];
	for (i = 0; i < k; i++){
		cin >> t;
		for (j = 0; j < m; j++){
			if (t >= a[j][0] && a[j][1] >= t)
				result += t - a[j][0] + a[j][2];
		}
	}
	cout << result;
	return 0;
}
