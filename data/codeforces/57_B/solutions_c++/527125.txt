#include<cstdio>
#include<algorithm>
using namespace std;

int n,m,q,a[100005],b[100005],c[100005];

int main() {
	scanf("%d%d%d",&n,&m,&q);
	for (int i=0; i<m; i++) {
		scanf("%d%d%d",&a[i],&b[i],&c[i]);
	}
	long long ans=0;
	for (int i=0; i<q; i++) {
		int at;
		scanf("%d",&at);
		for (int j=0; j<m; j++)
			if (a[j]<=at && at<=b[j]) ans+=at-a[j]+c[j];
	}
	printf("%I64d\n",ans);
}
