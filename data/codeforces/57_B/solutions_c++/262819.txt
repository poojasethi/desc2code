#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;

typedef long long int64;

int n,m,k;
int a[100000],b[100000],x[100000];

int main() {

	scanf("%d %d %d",&n,&m,&k);
	for (int i=0;i<m;i++) scanf("%d %d %d",&a[i],&b[i],&x[i]);
	int64 r=0;
	for (int i=0;i<k;i++) {
		int p;
		scanf("%d",&p);
		for (int j=0;j<m;j++) if (a[j]<=p && p<=b[j]) {
			r+=x[j]+p-a[j];
		}
	}
	cout << r << endl;
    return 0;
}
