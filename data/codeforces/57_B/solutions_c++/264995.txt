#include <iostream>

using namespace std;

int a[100000];
int b[100000];
long long c[100000];
int m, k;

int main () {
  int _;
  cin >> _ >> m >> k;

  for (int i=0; i < m; i++)
    cin >> a[i] >> b[i] >> c[i];

    long long res = 0;
  for (int i=0; i < k; i++) {
    int q;
    cin >> q;

    for (int j=0; j < m; j++)
      if (a[j] <= q && b[j] >= q)
        res += q - a[j] + c[j];

  }
    cout << res << endl;
}
