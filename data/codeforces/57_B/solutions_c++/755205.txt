#include <stdio.h>

typedef long long LL;
LL s[100005];
LL ns[100005];

int main(void) {
	int n, m, k, a, b, c;
	scanf("%d%d%d", &n, &m, &k);
	for(int i=0;i<m;i++) {
		scanf("%d%d%d", &a, &b, &c);
		ns[a]++;
		ns[b+1]--;
		s[a] += c-1;
		s[b+1] -= (c+b-a);
	}
	LL sum = 0;
	for(int i=1;i<=n;i++) {
		ns[i] += ns[i-1];
		s[i] += s[i-1] + ns[i];
	}
	for(int i=0;i<k;i++) {
		scanf("%d", &c);
		sum += s[c];
	}
	printf("%I64d\n", sum);
	return 0;
}
