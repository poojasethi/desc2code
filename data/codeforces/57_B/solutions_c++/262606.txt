#include <iostream>
#include <cstdio>
using namespace std;


int n, m, k;
long long a[100000], b[100000], c[100000], res;


int main()
{
    cin >> n >> m >> k;
    for(int i = 0; i < m; i++)  cin >> a[i] >> b[i] >> c[i];
    for(int i = 0; i < k; i++)
    {
        cin >> n;
        for(int j = 0; j < m; j++)
        {
            if(n >= a[j]  &&  n <= b[j])
                res += c[j] + n - a[j];
        }
    }
    cout << res;
}
