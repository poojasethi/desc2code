#include <stdio.h>
int l[100005], r[100005], x[100005];
int main() {
  int n, m, k;
  scanf("%d%d%d", &n, &m, &k);
  for(int i = 1; i <= m; ++i)
    scanf("%d%d%d", l+i, r+i, x+i);
  long long res = 0;
  for(int i = 0; i != k; ++i) {
    int t;
    scanf("%d", &t);
    for(int j = 1; j <= m; ++j)
      if(t >= l[j] && t <= r[j]) res += x[j] + t - l[j];
  }
  printf("%I64d ", res);
  return 0;
}
