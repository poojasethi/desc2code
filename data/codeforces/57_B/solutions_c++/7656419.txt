#include <stdio.h>


int main( void ) {
	int n, m, k;
	scanf("%i %i %i", &n, &m, &k);
	
	int l[m], r[m], h[m];
	
	for( int i = 0; i < m; i++ )
		scanf("%i %i %i", &l[i], &r[i], &h[i]);
	
	long long ans = 0;
	
	for( int i = 0; i < k; i++ ) {
		int cell;
		scanf("%i", &cell);
		
		for ( int j = 0; j < m; j++ )
			if( l[j] <= cell && cell <= r[j] )
				ans += cell - l[j] + h[j];
	}
	
	printf("%lld\n", ans);
	
	return 0;
}
