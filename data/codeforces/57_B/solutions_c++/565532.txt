#include <iostream>
using namespace std;

#define REP(i,b,n) for(int i=b;i<n;i++)
#define rep(i,n) REP(i,0,n)

int n,m,k,a[100000],b[100000],c[100000],q;

long long solve() {
  cin>>n>>m>>k;
  rep(i,m) cin>>a[i]>>b[i]>>c[i];
  
  long long res = 0;
  rep(i,k) {
    cin>>q;
    rep(j,m) if (a[j] <= q && q <= b[j]) {
      res += q - a[j] + c[j];
    }
  }
  return res;
}

int main(int argc, char *argv[]) {
  cout<<solve()<<endl;
  return 0;
}
