#include <iostream>

using namespace std;

int main()
{
	int i, j, t;
	int n, m, k;
	int a[100000] = {0};
	int b[100000] = {0};
	int c[100000] = {0};
	long long sum = 0;
	
	cin >> n >> m >> k;
	
	for (i = 0; i < m; i++)
		cin >> a[i] >> b[i] >> c[i];

	for (i = 0; i < k; i++)
	{
		cin >> t;
		for (j = 0; j < m; j++)
			if (a[j] <= t && t <= b[j])
				sum += c[j] + t - a[j];
	}
	
	cout << sum << endl;
	
	return 0;
}
