#include <iostream>

#define NN 112345

using namespace std;

long long n, m, k;
long long a[NN], b[NN], c[NN];
long long ans;

int main(void) {
	ans = 0LL;
	cin >> n >> m >> k;
	for(int i = 0; i < m; i++)
		cin >> a[i] >> b[i] >> c[i];
	for(int i = 0; i < k; i++) {
		int q;
		cin >> q;
		for(int i = 0; i < m; i++)
			if(a[i] <= q && q <= b[i])
				ans += c[i] + (q - a[i]);
	}
	cout << ans << endl;

	return 0;
}
