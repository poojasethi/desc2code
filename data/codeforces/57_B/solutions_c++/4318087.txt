#include<iostream>
#include<vector>
#include<algorithm>
#define SZ(S) S.size()
using namespace std ;
int main ()
{

    int x , t , t2 , n ;
    cin >> x >> t >> t2 ;

    long long c = 0 ;
    vector<int>B(t) , F(t) , I(t) ;
    for(int i=0 ; i<t ; i++)
        cin >> B[i] >> F[i] >> I[i] ;

    while(t2--)
    {
        cin >> n ;
        for(int i=0 ; i<t ; i++)
            if( n >= B[i] && n <= F[i] )
               c += I[i]-B[i]+n ;
    }
    cout << c ;
}
