#include <iostream>
#include <cstring>
using namespace std;

main()
{
	int n,m,k;
	cin >> n >> m >> k;
	
	int a[m];
	int b[m];
	int a0[m];
	
	for (int i = 0; i < m; i++) cin >> a[i] >> b[i] >> a0[i];
	
	long long sum = 0;
	int c;
	for (int i = 0; i < k; i++)
	{
		cin >> c;
		for (int j = 0; j < m; j++)
		{
			if (c < a[j] || c > b[j]) continue;
			int x = c-a[j];
			sum += a0[j]+x;	
		}	
	}
	cout << sum << "\n";
	//system("pause");
}
