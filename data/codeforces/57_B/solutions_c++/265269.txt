#include<cstdio>
int n,m,k,q[101],l[100001],r[100001],x[100001];
int main()
//   Last Change:  2011-01-26 13:21:45
{
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=m;++i)
		scanf("%d%d%d",l+i,r+i,x+i);
	long long ans=0;
	for(int i=1;i<=k;++i)
	{
		long long t=0;
		int s;
		scanf("%d",&s);
		for(int j=1;j<=m;++j)
			if(s>=l[j]&&s<=r[j])
				t+=s-l[j]+x[j];
		ans+=t;
	}
	printf("%I64d\n",ans);
	return 0;
}
