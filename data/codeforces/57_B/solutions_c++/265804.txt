#include <iostream>

using namespace std;

long long n, m, k, nom, sum, a[100000],b[100000],c[100000];

int main()
{
    cin >> n >> m >> k;
    for ( int i = 0; i < m; ++i )
        cin >> a[i] >> b[i] >> c[i];
    for ( int i = 0; i < k; ++i )
    {
        cin >> nom;
        for ( int i = 0; i < m; ++i )
            if ( nom >= a[i] && nom <= b[i] )
                sum += (nom - a[i]) + c[i];
    }
    cout << sum;
    return 0;
}
