#include <iostream>
using namespace std;
long long a[100000],b[100000],x[100000];
int main()
{
	long long n,m,k,t;
	long long res=0;
	cin>>n>>m>>k;
	for (int i=0;i<m;++i)
		cin>>a[i]>>b[i]>>x[i];
	for (int i=0;i<k;++i){
		cin>>t;
		for (int j=0;j<m;++j){
			if (t<=b[j] && t>=a[j]){
				res+=x[j]+t-a[j];
				//cout<<"res "<<res<<endl;
			}
		}
	}
	cout<<res<<endl;
	return 0;
}
