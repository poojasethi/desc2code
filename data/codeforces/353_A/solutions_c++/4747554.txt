#include <iostream>
using namespace std;

int main(){
    int a = 0, b = 0, n, s = 0;
    cin >> n;
    for(int i = 0; i<n; i++){
        int x, y;
        cin >> x >> y;
        if(x%2!=y%2) s = 1;
        a+=x; b+=y;
    }
    if((a+b)%2==1) cout << -1;
    else if(a%2) cout << 2*s-1;
    else cout << 0;
}
