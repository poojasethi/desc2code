#include <iostream> 
using namespace std;

int N;

int main(){
	cin >> N; int x,y,sx = 0, sy = 0, inv = -1;
	for (int i = 0; i < N; i++){
		cin >> x >> y;
		if (x%2 + y%2 == 1) inv = 1;
		sx += x; sy += y;
	}
	switch(sx%2 + sy%2){
		case 0: cout << 0; break;
		case 1: cout << -1; break;
		case 2: cout << inv; break;
	}
}
