#include<stdio.h>
int main(){
	int a[100],b[100],a1=0,b1=0,t,i,ev=0;
	scanf("%d",&t);
	for(i=0;i<t;i++){
		scanf("%d",&a[i]);
		a1+=a[i];
		scanf("%d",&b[i]);
		b1+=b[i];
		if(a[i]%2!=b[i]%2)ev++;
	}
	if(a1%2==0&&b1%2==0)printf("0");
	else if(a1%2==1&&b1%2==1&&ev!=0)printf("1");
	else printf("-1");
}
