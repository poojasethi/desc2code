#include <cstdio>
#include <cstring>
int a, b, n, x, y, c;
int main() {
	int i;
	scanf("%d", &n);
	for(i = 0; i < n; i++) {
		scanf("%d%d", &x, &y);
		a += x;
		b += y;
		c+= (y-x)&1;
	}
	if((a+b)&1) puts("-1");
	else if(a&1^1 && b&1^1) puts("0");
	else if(a&1 && b&1 && c)puts("1");
	else puts("-1");
}
