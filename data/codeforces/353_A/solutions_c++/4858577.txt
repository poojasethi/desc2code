#include<iostream>
using namespace std;
int main(){
	int r=0,n,m,q,total1=0,total2=0;
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>m>>q;
		total1+=m;
		total2+=q;
		if(m%2!=q%2)r++;
	}
	if(total1%2==0 && total2%2==0)cout<<0;
	else if(total1%2==1 && total2%2==1 && r>1)cout<<1;
	else cout<<-1;
	return 0;
}
