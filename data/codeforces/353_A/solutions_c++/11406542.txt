#include<iostream>
#include<cstring>
using namespace std;
int main()
{
int n,a,b,s1,s2;
s1=s2=0;
bool f=false;
cin>>n;
for(int i=0;i<n;i++)
{
	cin>>a>>b;
	s1+=a;
	s2+=b;
	if((a%2 + b%2) ==1)
		f=true;
}
if(s1%2==0 && s2%2==0)
	cout<<0;
else if(s1%2+s2%2 ==1)
	cout<<-1;
else if(f)
	cout<<1;
else
	cout<<-1;
	

	return 0;
}
