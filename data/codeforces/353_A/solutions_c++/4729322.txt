#include <iostream>
using namespace std;
int n,a,b,c,d,l;
int main()
{
	cin>>n;
	for (int i=0;i<n;i++)
	{
		cin>>c>>d;
		a+=c%2;
		b+=d%2;
		if (c%2	!= d%2)
		{
			l++;
		}
	}
	if (a%2 == b%2)
	{
		if (a%2 == 1)
		{
			if(l>0)
				cout<<1<<endl;
			else cout<<-1<<endl;
		}
		else cout<<0<<endl;
	}
	else cout<<-1<<endl;
}
