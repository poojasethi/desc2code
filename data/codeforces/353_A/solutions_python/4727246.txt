#!/usr/bin/env python

n = int(raw_input())
is_rev_exist = False
x = y = 0
for i in xrange(n):
    xi, yi = map(int, raw_input().split())
    x += xi
    y += yi
    if (xi + yi) % 2 == 1:
        is_rev_exist = True
if x % 2 == 0 and y % 2 == 0:
    print 0
elif x % 2 == 1 and y % 2 == 1 and is_rev_exist:
    print 1
else:
    print -1
