d = [map(int,raw_input().split()) for d in xrange(input())]
s1,s2 = sum([x[0] for x in d]), sum([x[1] for x in d])
if s1 % 2 == s2 % 2 == 0:
    print 0
elif s1 % 2 == s2 % 2 == 1:
    for x in d:
        if x[0] % 2 != x[1] % 2:
            print 1
            exit(0)
    print -1
else:
    print -1
