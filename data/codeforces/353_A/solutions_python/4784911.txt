eo,oe = 0,0
L, R = 0,0
n = int(raw_input())
for i in range(n):
	s = raw_input().split(" ")
	l = int(s[0])
	r = int(s[1])
	L += l
	R += r
	if (l%2)^(r%2) == 1:
		if l%2 == 1:
			oe += 1
		else:
			eo += 1

if (L%2)^(R%2) == 1:
	print "-1"
elif L%2 == 0:
	print "0"
elif eo+oe > 0:
	print "1"
else:
	print "-1"
