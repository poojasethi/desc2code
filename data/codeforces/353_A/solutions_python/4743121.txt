n = int(raw_input())
nums = []
for i in range(n) :
	nums.append(raw_input().split())
upper = 0
lower = 0
for num in nums :
	upper += int(num[0])
	lower += int(num[1])
if (upper + lower) % 2 == 1 :
	print -1
else :
	if upper % 2 == 0 :
		print 0
	else :
		ans = -1
		for num in nums :
			if (int(num[0]) + int(num[1])) % 2 == 1 :
				ans = 1
				break
		print ans
