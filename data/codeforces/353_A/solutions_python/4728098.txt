n = input()

sum_x = 0
sum_y = 0

flag = False

for i in xrange(n):
    x, y = map(int, raw_input().split())
    sum_x = (sum_x + x) % 2
    sum_y = (sum_y + y) % 2

    if (x + y) % 2 != 0:
        flag = True

x = sum_x == 0
y = sum_y == 0

if x and y:
    print 0
elif (not x) and (not y) and flag and n > 1:
    print 1
else:
    print -1
