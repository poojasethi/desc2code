n = input()
tiles = []
for i in xrange(n):
	tiles.append(map(int, raw_input().split()))
xs, ys = zip(*tiles)
if sum(xs) % 2 == 0 == sum(ys) % 2:
	print '0'
elif sum(xs) % 2 != sum(ys) % 2:
	print '-1'
else:
	for tile in tiles:
		if tile[0] % 2 != tile[1] % 2:
			print '1'
			break
	else:
		print '-1'
