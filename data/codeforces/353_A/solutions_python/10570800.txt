T=input()
L=[]
M=[]
for _ in xrange(T):
	K=map(int,raw_input().split())
	L.append(K[0])
	M.append(K[1])
l=sum(L)
m=sum(M)
if(l%2==0 and m%2==0):
	print 0
else:
	if(l%2!=m%2):
		print "-1"
	else:
		flag=False
		for x in xrange(T):
			if(L[x]%2 != M[x]%2):
				flag=True
				break
		if(flag):
			print "1"
		else:
			print "-1"
