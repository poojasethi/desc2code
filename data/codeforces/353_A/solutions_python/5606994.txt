n=input()
a=b=c=0
for i in range(n):
    x,y=map(int,raw_input().split())
    if x%2==1:
        a+=1
    if y%2==1:
        b+=1
    if x%2==1 and y%2==1:
        c+=1
if a==b==c!=0 and c%2==1:
    print -1
elif (a+b)%2==1:
    print -1
else:
    if (a%2)+(b%2)==0:
        print 0
    else:
        print 1
