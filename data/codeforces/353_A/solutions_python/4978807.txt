def main():
  R = lambda:map(int,raw_input().split())

  n=R()[0]
  x,y = [0]*n,[0]*n
  
  for i in range(n):
    x[i],y[i] = R()
  
  if sum(x+y)%2==1:
    print -1
    return

  if sum(x)%2==0 and sum(x)%2==0: 
    print 0
    return

  for i in range(n):
    if (x[i]+y[i])%2==1:
      print 1
      return
  print -1 
  
main()
