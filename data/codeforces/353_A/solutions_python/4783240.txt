n = int(input());

sumX = 0;
sumY = 0;
flag = False;

for i in range(n) : 
	x, y = map(int, raw_input().split());
	sumX += x;
	sumY += y;
	if (x + y) % 2 != 0 :
		flag = True;

if sumX % 2 == 0 and sumY % 2 == 0 : 
	print 0;
elif sumX % 2 != 0 and sumY % 2 != 0 : 
	if flag : 
		print 1;
	else :
		print -1;
else :
	print -1;
