#!/usr/bin/env python
# -*- coding: utf-8 -*-

n = int(input())
x = []
y = []
exis=False
for i in range(n):
	s = raw_input()
	s = s.split()
	a = int(s[0])
	b = int(s[1])
	if (a+b)%2 == 1:
		exis = True
	x.append(a)
	y.append(b)
sx = sum(x)
sy = sum(y)
if (sx+sy)%2 == 1:
	print '-1'	
elif (sx%2==0) and (sy%2==0):
	print '0'
elif exis:
	print '1'
else:
	print '-1'
