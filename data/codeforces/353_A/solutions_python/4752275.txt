n = int(raw_input())
f = False
s1 = s2 = s3 = 0
for i in xrange(n):
    a,b = map(int,raw_input().split(' '))
    if (a & 1) ^ (b & 1) == 1:
        f = True
    s1 += a+b
    s2 += a
    s3 += b

if (s2 & 1) == 0 and (s3 & 1) == 0:
    print 0
elif (s1 & 1) == 0 and f:
    print 1
else:
    print -1
