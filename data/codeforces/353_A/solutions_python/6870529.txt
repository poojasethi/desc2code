n = input();

v=  [[0]*2 for i in range(n)]
sum1,sum2=0,0;
yes=0;
for i in range(n):
	v[i] = map(int, raw_input().split());
	sum1 += v[i][0]; sum2 += v[i][1];
	sum1 %= 2; sum2 %= 2;
	if(v[i][0]%2 != v[i][1]%2):yes=1;

if(sum1 == 0 and sum2 == 0): print 0;
elif (sum1 != sum2): print -1
elif (yes==0): print -1
else: print 1;
