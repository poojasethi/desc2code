import sys, os

n = int(raw_input())

ta = 0
tb = 0
mark = False

for i in range(0, n):
	s = raw_input().split()
	a = int(s[0])
	b = int(s[1])
	ta += a
	tb += b
	if a % 2 == b % 2:
		continue
	mark = True

if ta % 2 == 0 and tb % 2 == 0:
	print 0
else:
	if ta % 2 != tb % 2:
		print -1
	else:
		if mark:
			print 1
		else:
			print -1
