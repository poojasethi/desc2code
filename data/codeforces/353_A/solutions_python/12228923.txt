n = int(raw_input())
up,down = [0]*n,[0]*n
for i in xrange(n):
  up[i],down[i] = map(int, raw_input().split())

up_parity, down_parity = sum(up)%2,sum(down)%2

if sum(up+down)%2==1:
  print -1
else:
  if up_parity==0 and down_parity==0:
    print 0
  else:
    print 1 if any((up[i]+down[i])%2==1 for i in xrange(n)) else -1
