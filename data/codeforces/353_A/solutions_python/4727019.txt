#!/usr/bin python

n = int(raw_input())
flag = 0
sumx = 0
sumy = 0
for i in range(n) :
    x,y = [int(x) for x in raw_input().split()]
    if (x+y) % 2 == 1 :
        flag = 1
    sumx += x
    sumy += y

if sumx %2 ==0 and sumy %2 ==0:
    print 0
elif (sumx + sumy)%2 == 1:
    print -1
else:
    if flag == 1 :
        print 1
    else:
        print -1
