#include<cstdio>
#include<cstring>
const int MOD = 1000000007;
int dp[110][2610];
char str[110];
int main(){
	int i,j,k;
	for(i=0;i<26;i++) dp[1][i]=1;
	for(i=1;i<100;i++){
		for(j=0;j<2610;j++){
			int v=dp[i][j];
			if(v==0) continue;
			for(k=0;k<26;k++) dp[i+1][j+k]=(dp[i+1][j+k]+v)%MOD;
		}
	}
	int T,len,cnt;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%s",str);
		len=strlen(str);
		cnt=0;
		for(i=0;i<len;i++) cnt+=str[i]-'a';
		int r=dp[len][cnt]-1;
		if(r<0) r+=MOD;
		printf("%d\n",r);
	}
	return 0;
}
