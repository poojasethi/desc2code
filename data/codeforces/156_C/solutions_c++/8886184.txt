#include<cstdio>
#include<cstring>
using namespace std;
int n,T,dp[120][3000];
char s[120];
int summ;
int main()
{
	scanf("%d",&T);
	dp[0][0]=1;int top=0;
	for (int i=0;i<100;i++)
	{
		for (int j=0;j<=top;j++)
		{
			if(dp[i][j]!=0)
			{top=j+26;
			for (int k=1;k<=26;k++)
			{
				dp[i+1][j+k]+=dp[i][j];
				dp[i+1][j+k]%=1000000007;
			}
			}
		}
	}
	while (T--)
	{
		summ=0;
		scanf("%s",s+1);
		n=strlen(s+1);
		for (int i=1;i<=n;i++)
		{
			summ+=(s[i]-'a'+1);
		}printf("%d\n",dp[n][summ]-1);
	}
	return 0;
}
