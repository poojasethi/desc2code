#include<cstdio>
#include<cstring>
const int MOD = 1000000007;
int dp[101][2601];
char str[101];
int main()
{
	int i,j,k;
	for(i=0;i<26;i++)dp[1][i]=1;
	for(i=1;i<100;i++)
	{
		for(j=0;j<2501;j++)
		{
			int v=dp[i][j];
			if(!v)continue;
			for(k=0;k<26;k++)
			{
				dp[i+1][j+k]=(dp[i+1][j+k]+v)%MOD;
			}
		}
	}
	int t,n,m;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",str);
		n=strlen(str);
		m=0;
		for(i=0;i<n;i++)m+=str[i]-'a';
		int r=dp[n][m]-1;
		if(r<0)r+=MOD;
		printf("%d\n",r);
	}
	return 0;
}
