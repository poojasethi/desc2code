#include<iostream>
using namespace std;
int dp[2600][110];
int main(){
    string s;
    int t,i,j,k,res,sum;
    cin>>t;
    for(i=0;i<=100;i++) dp[0][i]=1;
    for(i=1;i<=2500;i++){ 
       for(j=1;j<=100;j++){
          res=0;  
          for(k=0;k<=25&&k<=i;k++) res+=dp[i-k][j-1],res%=1000000007;
          dp[i][j]=res;  
       } 
    }
    for(i=0;i<t;i++){
       cin>>s;
       sum=0;
       for(j=s.size()-1;j>=0;j--) sum+=s[j]-'a';
       cout<<dp[sum][s.size()]-1<<endl; 
    }
    return 0;
}
