#include <iostream>
#include <cstring>
using namespace std;
typedef long long ll;
const int MAXN=111,MAXM=26*100+100,MOD=1000*1000*1000+7;
ll dp[MAXN][MAXM],tests;
int main()
{
	dp[0][0]=1;
	for(int i=1;i<=100;i++)
		for(int j=0;j<=25*100;j++)
			for(int k=0;k<26;k++)
				dp[i][j+k]=(dp[i][j+k]+dp[i-1][j])%MOD;
	cin >> tests;
	int i,sum;
	string s;
	while(tests--)
	{
		cin >> s;
		sum=0;
		for(int i=0;i<s.size();i++)
			sum+=int(s[i]-'a');
		cout << (dp[s.size()][sum]+MOD-1)%MOD << endl;
	}
	return 0;
}
