#include<cstdio>
#include<cstring>
#define M 1000000007
using namespace std;
int dp[110][2510];
char s[110];
int main(){
	dp[0][0]=1;
	for(int i=1;i<=100;i++)
		for(int j=0;j<26;j++)
			for(int k=2500;k>=j;k--)
				dp[i][k]=(dp[i][k]+dp[i-1][k-j])%M;
	int t;scanf("%d",&t);
	gets(s);
	while(t--){
		gets(s);
		int n=strlen(s);
		int ans=0;
		for(int i=0;i<n;i++) ans+=s[i]-'a';
		printf("%d\n",dp[n][ans]-1);
	}
	return 0;
}
