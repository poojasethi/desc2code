#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
int dp[110][2700];
int main()
{
	int t,i,n,m;
	ios_base::sync_with_stdio(false);
	dp[0][0] = 1;	
	for(int i = 0 ; i <= 100 ; i++)
		for(int j = 1 ; j <= 26 * i ; j++)
			for(int k = 1 ; k <= min(26, j) ; k++)
				dp[i][j] = (dp[i][j] + dp[i-1][j-k])%mod; 
	cin>>t;
	string s;
	while(t--)
	{
		cin>>s;
		n=s.length();
		m=0;
		for(i=0;i<n;i++)
			m+=(s[i]-'a'+1);
		cout<<dp[n][m]-1<<endl;
	}	
	return 0;
}
