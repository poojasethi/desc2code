#include <stdio.h>
int dp[1251][101];
void init()
{
	dp[0][0]=1;
	int i,j,k;
	for(i=1;i<101;i++){
	for(j=0;j<26;j++)
		for(k=j;k<1251;k++){
		dp[k][i]+=dp[k-j][i-1];
		dp[k][i]%=1000000007;
		}
	}
}

int main(){
	int t;
	init();
	scanf("%d",&t);
	for(;t>0;t--){
		char a[101];
		scanf("%s",a);
		int ans1=0,ans2=0,i;
		for(i=0;a[i]!=0;i++){
		ans1+=a[i]-'a';
		ans2+='z'-a[i];
		}
		if(ans1>ans2)ans1=ans2;
		printf("%d\n",(dp[ans1][i]+1000000006)%1000000007);		
	}
	return 0;
}
