#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 110
#define mo 1000000007
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,u,T;
int f[N][2610];
char p[N];
int main() {
	f[0][0]=1;
	For(i,1,100)For(k,0,2600) if (f[i-1][k]) For(j,1,26) {
		f[i][j+k]+=f[i-1][k];
		if (f[i][j+k]>=mo) f[i][j+k]-=mo;
	}
	scanf("%d",&T);
	for (;T;T--) {
		scanf("%s",p+1);
		n=strlen(p+1);
		int s=0;
		For(i,1,n) s+=p[i]-'a'+1;
		printf("%d\n",(f[n][s]-1+mo)%mo);
	}
	return 0;
}
