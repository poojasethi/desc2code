#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
int T;
char s[115];
int n;
const int MOD=1000000007;
int dp[115][3005];
int main()
{
	scanf("%d",&T);
	dp[0][0]=1;
    for(int i=1;i<=105;i++)
	{
		for(int j=0;j<=3000;j++)
		{
			for(int k=0;k<26;k++)
			{
				if(j>=k)
				{
					dp[i][j]=(dp[i][j]+dp[i-1][j-k])%MOD;
				}
			}
		}
    }
	while(T--)
	{
		scanf("%s",s+1);
		n=strlen(s+1);
		int sum=0;
		for(int i=1;i<=n;i++)sum+=s[i]-'a';
		printf("%d\n",dp[n][sum]-1);
	}
	return 0;
}
