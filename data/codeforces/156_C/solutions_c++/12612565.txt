#include <cstdio>
#include <cstring>

const int m = 1e9+7;
int n, len, sum, d[105][3000]={0};
char s[105];

int main() {
	d[0][0]=1;
	for(int i=0; i <= 100; i++) for(int j=0; j <= 2600; j++) for(int k=0; k<26; k++) d[i+1][j+k]=(d[i+1][j+k]+d[i][j])%m;
	scanf("%d", &n);
	for(int i=0; i<n; i++) {
		sum=0;
		scanf("%s", s);
		len=strlen(s);
		for(int i=0; i<len; i++) sum += s[i]-'a';
		printf("%d\n", d[len][sum]-1);
	}
	return 0;
}
