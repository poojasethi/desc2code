#include<cstdio>
#include<cstring>
using namespace std;
typedef long long LL;
const LL P=1000000007ll;
int f[105][2605],i,sum,k,L,T;
char s[105];
int main()
{
  f[0][0]=1;
  for (i=0;i<100;i++)
    for (sum=0;sum<=2500;sum++)
      if (f[i][sum])
        for (k=0;k<26;k++)
          (f[i+1][sum+k]+=f[i][sum])%=P;
  scanf("%d",&T);
  while (T--)
  {
    scanf("%s",s+1);L=strlen(s+1);sum=0;
    for (i=1;i<=L;i++)
      sum+=s[i]-'a';
    printf("%d\n",(f[L][sum]-1+P)%P);
  }
  return 0;
}
