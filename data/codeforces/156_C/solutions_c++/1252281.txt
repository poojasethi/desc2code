#include<cstdio>
#include<cstring>
int f[105][2605],oo=1000000007;
char s[1005];
int main()
{
	f[0][0]=1;
	for (int i=1;i<=100;i++)
		for (int j=0;j<=25;j++)
			for (int k=2500;k>=j;k--){
				f[i][k]+=f[i-1][k-j];
				if (f[i][k]>=oo) f[i][k]-=oo;
				}
	int n,l,x; scanf("%d",&n); gets(s);
	for (int i=1;i<=n;i++){
		gets(s); l=strlen(s); x=0;
		for (int j=0;j<l;j++) x+=s[j]-97;
		printf("%d\n",f[l][x]-1);
		}
	return 0;
}
