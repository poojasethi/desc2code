#include <cstdio>
#define FOR(i,s,n) for(int i = s; i < n; ++i)

const int MOD = 1000000007;
char word[128];
int dp[128][2550], len, sum, t;

int main(void){
	FOR(i, 0, 101) FOR(j, 0, 2501) dp[i][j] = 0;
	dp[0][0] = 1;
	FOR(i, 1, 101) FOR(j, 0, 2501) FOR(k, 0, 26) 
		dp[i][j + k] = (dp[i - 1][j] + dp[i][j + k]) % MOD;
	for(scanf("%d", &t); t--; ){
		scanf("%s", word);
		for(len = sum = 0; word[len]; ++len) sum += word[len] - 'a';
		printf("%d\n", dp[len][sum] - 1);
	}
	return 0;
}
