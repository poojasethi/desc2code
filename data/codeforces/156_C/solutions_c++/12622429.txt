#include<stdio.h>
#include<string.h>
#define module 1000000007
char s[105];
int dp[105][2550];
int main(void){
	int n;
	scanf("%d",&n);
	for(int i=0;i<26;i++){
		dp[0][i]=1;
	}
	for(int i=1;i<105;i++){
		for(int j=0;j<2500;j++){
			for(int k=0;k<26;k++)
			dp[i][j+k]=(dp[i][j+k]+dp[i-1][j])%module;	
		}	
	}	
	while(n--){
		int tot=0;
		scanf("%s",s);
		int slen=strlen(s);
		for(int i=0;i<slen;i++){
			tot+=s[i]-'a';
		}
		printf("%d\n",(dp[slen-1][tot]+module-1)%module);		
	}
	return 0;
}
