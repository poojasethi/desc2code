#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;

int dp[105][2650];

int main() {
    int t;
    char s[105];
	dp[0][0]=1;
	for(int i=0; i<100; i++) for(int j=0; j<2600; j++) {
        if(!dp[i][j]) continue;
        for(int k=0; k<26; k++) dp[i+1][j+k]=(dp[i+1][j+k]+dp[i][j])%MOD;
	}
	scanf("%d", &t);
	while(t--) {
		scanf("%s", s);
		int sum=0, i;
		for(i=0; s[i]; i++) sum+=s[i]-'a';
		printf("%d\n", (dp[i][sum]-1+MOD)%MOD);
	}
	return 0;
}
