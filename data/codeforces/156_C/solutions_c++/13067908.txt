#include <iostream>

using namespace std;

const int MAXN = 101, MAXW = 25 * 100 + 1, MOD = 1e9 + 7;
int dp[MAXN][MAXW];

int main() {
	dp[0][0] = 1;
	for (int i = 1; i < MAXN; i++)
		for (int j = 0; j < MAXW; j++)
			for (int k = 0; k <= j && k < 26; k++)
				dp[i][j] += dp[i - 1][j - k], dp[i][j] %= MOD;
	int t;
	cin >> t;
	while (t--) {
		int sum = 0;
		string s;
		cin >> s;
		int n = s.size();
		for (int i = 0; i < n; i++)
			sum += s[i] - 'a';
		cout << (dp[n][sum] - 1 + MOD) % MOD << "\n";
	}
	return 0;
}
