#include <iostream>
#include <string>

#define MOD 1000000007

using namespace std;

int T,dp[110][2610];
string S;

int tonum(char c)
{
	return c-'a'+1;
}

void init()
{
	dp[0][0]=1;
	for (int i=1;i<=100;i++)
		for (int j=0;j<=2600;j++)
			for (int k=1;k<=j&&k<=26;k++)
				dp[i][j]=(dp[i][j]+dp[i-1][j-k])%MOD;
}

int main()
{
	init();
	cin >> T;
	for (int i=0;i<T;i++)
	{
		int sum=0;
		cin >> S;
		for (int j=0;j<S.size();j++)
			sum+=tonum(S[j]);
		cout << dp[S.size()][sum]-1 << endl;
	}
	return 0;
}
