#include <cstdio>
#include <cstring>

using namespace std;

long long f[101][2700];
char str[105];

int main() {
	int t, i, j, k, s, l;
	
	f[0][0] = 1ll;
	for (i = 1; i <= 100; ++i)
		for (j = i; j <= 26 * i; ++j)
			for (k = j - 26 > 0 ? j - 26 : 0; k < j; ++k)
				f[i][j] = (f[i][j] + f[i - 1][k]) % 1000000007;
	
	scanf("%d", &t);
	for (i = 0; i < t; ++i) {
		scanf("%s", str);
		s = 0;
		l = strlen(str);
		for (j = 0; j < l; ++j) s += (str[j] - 'a') + 1;
		printf("%d\n", f[l][s] - 1);
	}
	
	return 0;
}
