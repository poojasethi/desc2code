#include<iostream>
using namespace std;
const int mod = 1e9 + 7;
int dp[101][2627];
char str[101];
int main()
{
  int tests;
  cin>>tests;
  dp[0][0] = 1;
  for (int i = 1; i <= 100; i++)
    for (int s = 0; s <= 2600; s++)
      for (int k = 1; k <= 26; k++)
	dp[i][s + k] = (dp[i-1][s] + dp[i][s + k])%mod;
  for (int i = 1; i <= tests; i++) {
    cin>>str;
    int sum = 0;
    int n;
    for (n = 0; str[n]; n++) sum += str[n] - 'a' + 1;
    cout<<dp[n][sum] - 1<<endl;
  }
  return 0;
}
