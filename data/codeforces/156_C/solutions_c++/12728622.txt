#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
int dp[101][2700];char s[120];
int main(){
	memset(dp,0,sizeof(dp));
	dp[0][0]=1;
	int Mod= 1000000007;
	for(int i=0;i<100;i++)
		for(int k=0;k<=2600;k++)
			for(int j=1;j<=26;j++){
				dp[i+1][k+j]+=dp[i][k];
				dp[i+1][k+j]%=Mod;
			}
	int T;
	scanf("%d",&T);
	while(T--){
		scanf("%s",s);
		int l=strlen(s);
		int sum=0;
		for(int i=0;i<l;i++) sum+=s[i]-'a'+1;
		cout << (dp[l][sum]+Mod-1)%Mod << endl;
	}
}
