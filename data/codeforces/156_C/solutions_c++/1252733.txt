#include<cstdio>
#define maxn 101
#define maxm 2701
int dp[maxn][maxm],mod=1000000007;
char s[maxn];
int main()
{
    int i,j,k,n,re,ri,sum;
    dp[0][0]=1;
    for(i=1;i<=100;i++)
    {
        for(j=0;j<26;j++)
        {
            for(k=(i-1)*25;k>=0;k--)
                dp[i][j+k]=(dp[i][j+k]+dp[i-1][k])%mod;
        }
    }
    scanf("%d",&re);
    for(ri=1;ri<=re;ri++)
    {
        scanf("%s",s);
        for(i=sum=0;s[i];i++) sum+=s[i]-'a';
        printf("%d\n",(dp[i][sum]+mod-1)%mod);
    }
}
