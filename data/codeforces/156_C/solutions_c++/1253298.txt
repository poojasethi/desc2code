#include <cstdio>
using namespace std;

#define MOD 1000000007
int dp[105][2650];

int main() {
	dp[0][0] = 1;

	for (int i = 0; i < 100; i++) {
		for (int j = 0; j < 2600; j++) {
			if (!dp[i][j]) continue;
			for (int k = 0; k < 26; k++) {
				dp[i + 1][j + k] = (dp[i + 1][j + k] + dp[i][j]) % MOD;
			}
		}
	}
	int T;
	scanf("%d", &T);
	char s[105];
	while (T--) {
		scanf("%s", s);
		int ret = 0, i;
		for (i = 0; s[i]; i++) {
			ret += s[i] - 'a';
		}
		printf("%d\n", (dp[i][ret] - 1 + MOD) % MOD);
	}

	return 0;
}
