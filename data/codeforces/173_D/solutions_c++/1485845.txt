#include<cstdio>
#include<cstring>
#define MAXN 110000

int M,n,s[MAXN],b[MAXN],x[2*MAXN],y[2*MAXN],f[MAXN],ff[MAXN],c[MAXN],
    T(2),N,XN,YN,o[MAXN],bl[MAXN],X[MAXN],Y[MAXN],Now;

void IN(int a,int b){x[T]=f[a];y[T]=b;f[a]=T++;}

bool DO3()
{	int i;
	for(i=0;i<XN;i++)ff[X[i]]=Now+i/3;
	for(i=0;i<YN;i++)ff[Y[i]]=Now+(XN+i)/3;
}

void Miss(int i)
{    for(int k=f[i];k;k=x[k])o[y[k]]--;
}

bool DO1(int Q)
{	int i,j,k;
	for(k=0;k<XN;k++)
		if(o[X[k]]+2==YN)
			break;
	if(k==XN)
    {   for(k=0;k<XN;k++)
		if(o[X[k]]+2<=YN)
		break;  
        if(k==XN)return 0;
    }
	memset(bl,0,sizeof bl);
	for(i=f[X[k]];i;i=x[i])bl[y[i]]=1;
	for(i=j=0;i<YN&&j<2;i++)if(!bl[Y[i]]){ff[Y[i]]=Now;j++;bl[Y[i]]=2;Miss(Y[i]);}
	ff[X[k]]=Now++;
	Miss(X[k]);
	for(i=j=0;i<XN;i++)if(i!=k)X[j++]=X[i];
	XN=j;
	for(i=j=0;i<YN;i++)if(bl[Y[i]]!=2)Y[j++]=Y[i];
	YN=j;
	if(!Q)DO3();
	return 1;
}

bool DO2(int Q)
{	int i,j,k;
	for(k=0;k<YN;k++)
		if(o[Y[k]]+2==XN)
			break;
	if(k==YN)
    {   for(k=0;k<YN;k++)
		if(o[Y[k]]+2<=XN)
		break;  
        if(k==YN)return 0;
    }
	memset(bl,0,sizeof bl);
	for(i=f[Y[k]];i;i=x[i])bl[y[i]]=1;
	for(i=j=0;i<XN&&j<2;i++)if(!bl[X[i]]){ff[X[i]]=Now;j++;bl[X[i]]=2;Miss(X[i]);}
	ff[Y[k]]=Now++;
	Miss(Y[k]);
	for(i=j=0;i<YN;i++)if(i!=k)Y[j++]=Y[i];
	YN=j;
	for(i=j=0;i<XN;i++)if(bl[X[i]]!=2)X[j++]=X[i];
	XN=j;
	if(!Q)DO3();
	return 1;
}

int main()
{	int i,j,k,h,t;
	//freopen("d.in","rb",stdin);
	//freopen("d.out","wb",stdout);
	scanf("%d%d",&n,&M);
	for(i=0;i<M;i++)
	{	scanf("%d%d",&j,&k);
		IN(j,k);
		IN(k,j);
		o[j]++;
		o[k]++;
	}
	for(i=1;i<=n;i++)
	if(!b[i])
	{	b[i]=++N;
		h=0;
		t=1;
		s[0]=i;
		for(;h<t;h++)
			for(k=f[s[h]];k;k=x[k])
				if(!b[y[k]])
				{	b[s[t++]=y[k]]=b[i];
					c[y[k]]=c[s[h]]^1;
				}
				else if(c[y[k]]!=(c[s[h]]^1))
				{	printf("NO\n");
					return 0;
				}
		for(h=0;h<t;h++)
			if(c[s[h]])X[XN++]=s[h];
				else Y[YN++]=s[h];
	}
	if(!(XN%3))DO3();
	else if((XN%3)==1)
	{	if(!DO1(0)&&!(DO2(1)&&DO2(0)))
		{
			printf("NO\n");
			return 0;
        }
	}
	else
	{	if(!DO2(0)&&!(DO1(1)&&DO1(0)))
		{
			printf("NO\n");
			return 0;
        }
	}
	printf("YES\n");
	for(i=1;i<=n;i++)printf("%d ",ff[i]+1);
	printf("\n");
	return 0;
}
