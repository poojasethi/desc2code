#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,tot=2,vt,s[2],d[100010],ans[100010],col[100010],vis[100010],v[200010],next[200010],head[100010];
void add(int u1,int v1){
    v[++vt]=v1;next[vt]=head[u1];head[u1]=vt;
}
void dfs(int x){
    vis[x]=1;++s[col[x]];
    for(int i=head[x];i;i=next[i])
        if(!vis[v[i]]){
            col[v[i]]=col[x]^1;
            dfs(v[i]);
        }
}
void print(){
    for(int i=1;i<=n;++i)
        if(col[i]==0&&!ans[i])
            ans[i]=++tot;
    for(int i=1;i<=n;++i)
        if(col[i]==1&&!ans[i])
            ans[i]=++tot;
    printf("YES\n");
    for(int i=1;i<=n;++i)printf("%d ",ans[i]/3);
}
void paint(int x){
    ans[x]=++tot;
    memset(vis,0,sizeof(vis));
    for(int i=head[x];i;i=next[i])
        vis[v[i]]=1;
    int t=0;
    for(int i=1;i<=n;++i)
        if(!ans[i]&&!vis[i]&&col[i]!=col[x]){
            ans[i]=++tot;++t;if(t==2)break;
        }
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;++i){
        int u1,v1;
        scanf("%d%d",&u1,&v1);
        add(u1,v1);add(v1,u1);
        ++d[u1];++d[v1];
    }
    for(int i=1;i<=n;++i)
        if(!vis[i]){
            col[i]=0;
            dfs(i);
        }
    if(s[0]%3){
        int x=0,y=1,t=0;
        if(s[0]%3==2)swap(x,y);
        for(int i=1;i<=n;++i)
            if(col[i]==x&&d[i]+2<=s[y]){
                paint(i);print();return 0;
            }
        for(int i=1;i<=n;++i)
            if(col[i]==y&&d[i]+2<=s[x]){
                paint(i);++t;
                if(t==2){print();return 0;}
            }
        printf("NO");
    }else print();
    return 0;
}
