#include <iostream>
#include <vector>
#include <algorithm>

#define rep(i,n) for(int i=0;i<n;i++)
#define rp(i,n) for(i=0;i<n;i++)
#define pr(i) cout<<i<<endl

using namespace std;

int n,m,k=1,i,j,a,b,t;
vector<int> g[100010],s[2];
bool vis[100010];
int d[100010],f[100010];


void visit(int p,int d){

  if(vis[p])return ;  

  vis[p]=true;
  s[d].push_back(p);
  
  rep(i,g[p].size()){
    visit(g[p][i],1-d);
  }

}

bool solve(){

  rp(i,s[0].size())if(g[s[0][i]].size()+2<=s[1].size()&&d[s[0][i]]<=0)break;
  if(i==s[0].size())return false;
  
  i=s[0][i];d[i]=k;t=0;
  
  rep(j,n)f[j]=false;

  rep(j,g[i].size())f[g[i][j]]=true;
  rep(j,s[1].size()){
    if(t>=2)break;
    if(f[s[1][j]]==false){d[s[1][j]]=k;t++;}
  }
  
  k++;

}

int main(){
  
  cin>>n>>m;
  rep(i,m){
    cin>>a>>b;
    g[--a].push_back(--b);
    g[b].push_back(a);
  }

  rep(i,n)if(!vis[i])visit(i,0);

  if(s[0].size()%3==2)swap(s[0],s[1]);

  if(s[0].size()%3==1){
    if(!solve()){
      swap(s[0],s[1]);
      if(!(solve()&&solve())){pr("NO");return 0;}
    }
  }

  t=0;

  rep(ii,2)rep(i,s[ii].size()){
    if(t==3){k++;t=0;}
    if(!d[s[ii][i]]){
      d[s[ii][i]]=k;
      t++;
    }
  }
  
  pr("YES");
  rep(i,n)cout<<d[i]<<" ";cout<<endl;
}
