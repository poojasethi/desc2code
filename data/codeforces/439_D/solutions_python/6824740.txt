inp = raw_input().split()

len_A ,len_B =int(inp[0]),int(inp[1])

A = raw_input().split()
B = raw_input().split()

for i in range (len_A):
    A[i]=int(A[i])
for i in range (len_B):
    B[i]=int(B[i])
A.sort()
B.sort(reverse=True)
total_cost = 0
i=0
while(i < min(len_A,len_B) and A[i]<B[i]):
    total_cost+=(B[i]-A[i])
    i+=1
print total_cost
