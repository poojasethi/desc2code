n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
sa, sb = map(sum, [a, b])
#print sa, sb
x = sorted([(t, 0) for t in a] + [(t, 1) for t in b])
ans = 9e15
cla = 0
sla = 0
clb = 0
slb = 0
for i in range(len(x)):
    if i > 0:
        t, f = x[i-1]
        if f:
            slb += t
            clb += 1
        else:
            sla += t
            cla += 1
    t = x[i][0]
    ans = min(ans, cla*t-sla + (sb-slb)-(len(b)-clb)*t )
print ans
