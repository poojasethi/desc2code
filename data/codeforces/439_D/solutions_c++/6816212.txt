#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,m;
    long long ans=0;
    cin>>n>>m;
    int a[n],b[m];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    for(int i=0;i<m;i++)
    {
        cin>>b[i];
    }
    sort(a,a+n);
    sort(b,b+m,greater<int>());
    for(int i=0;i<min(n,m) && b[i]>a[i];i++)
    {
        ans+=(int)abs(b[i]-a[i]);
    }
    cout<<ans<<endl;

}
