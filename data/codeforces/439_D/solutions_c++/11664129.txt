#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll a[100005],b[100005];
int main()
{
    ll n,m,ans=0;
    cin>>n>>m;
    for(ll i=0;i<n;i++)
    cin>>a[i];
    for(ll i=0;i<m;i++)
    cin>>b[i];
    sort(a,a+n);
    sort(b,b+m,greater<ll>());
    for(ll i=0;i<min(n,m);i++)
    {
        if(a[i]>b[i])
        break;
        ans+=abs(b[i]-a[i]);
    }
    cout<<ans<<"\n";
    return 0;
    }
