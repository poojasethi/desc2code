#include <bits/stdc++.h>
using namespace std;

long long a[101000],b[101000],ans=0,n,m;

int main()
{
    cin>>n>>m;
    for(int i=0;i<n;i++) cin>>a[i];
	for(int i=0;i<m;i++) cin>>b[i];
    sort(a,a+n);sort(b,b+m);reverse(b,b+m);
    for(int i=0;i<min(n,m);i++)if(b[i]>a[i]) ans+=b[i]-a[i];
    cout<<ans<<endl;
}
