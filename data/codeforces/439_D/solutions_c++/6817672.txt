#include<bits/stdc++.h>
using namespace std;
#define LL long long
bool wolta(LL x,LL y)
{return x>y;}
int main()
{
    LL n,m;
    cin>>n>>m;
    LL a[n],b[m];
    for(int i=0;i<n;++i) cin>>a[i];
    for(int i=0;i<m;++i) cin>>b[i];
    sort(a,a+n);sort(b,b+m,wolta);
    LL k = min(n,m),res = 0;
    for(LL i=0;i<k and b[i] > a[i];++i)res += b[i]-a[i];
    cout<<res<<endl;return 0;
}
