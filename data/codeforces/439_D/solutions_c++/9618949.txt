#include <bits/stdc++.h>
using namespace std ;
long long a[100000]={0};
    long long b[100000]={0};
int main(){

    int n , m ;
    cin >>n>>m;
    
    for(int i=0;i<n;i++)cin >> a[i];
    for(int i=0;i<m;i++)cin >> b[i];
    sort (a,a+n);
    sort (b,b+m);
    reverse(b,b+m);
    long long sumA = 0 ;
    long long sumB = 0 ;
     for(int j=0;j<min(n,m);j++){if(b[j]>a[j]){sumB+=b[j];sumA+=a[j];}}
    cout << sumB-sumA ;
return 0 ;
}
