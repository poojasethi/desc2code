#include<stdio.h>
#include<algorithm>
#include<math.h>
#include<string.h>
using namespace std;

int n,m,x[100001],y[100001];

int main()
{
	int i,j;
	long long k=0;
	scanf("%d %d",&n,&m);
	for(i=1; i<=n; i++) scanf("%d",&x[i]);
	for(i=1; i<=m; i++) scanf("%d",&y[i]);
	
	sort(x+1,x+n+1); sort(y+1,y+m+1);

	j=m;
	for(i=1; i<=n; i++)
{
	if(j==0) break;
	if(x[i]<y[j]) k+=y[j]-x[i];
	j--;
	
}
printf("%I64d",k);
return 0;
}
