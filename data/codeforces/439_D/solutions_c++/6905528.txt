#include<cstdio>
#include<algorithm>
#define LL long long
using namespace std;
int a[100100],b[100100];

bool cmp(const int a1,const int a2){
	return a1>a2;
}

int main()
{
	int n,m,i,j;
	scanf("%d%d",&n,&m);
	int d=0;
	for(i=0;i<n;++i)scanf("%d",a+i);
	for(i=0;i<m;++i)scanf("%d",b+i);
	sort(a,a+n);
	sort(b,b+m,cmp);
	LL ans=0;
	for(i=0;i<n&&i<m&&b[i]>a[i];++i)ans+=b[i]-a[i];
	printf("%I64d\n",ans);
	return 0;
}
