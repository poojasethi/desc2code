#include <bits/stdc++.h>
using namespace std;

#define rep(i,n) for(int i = 0;i < n;++i)

const int maxn = 1e5+100;

long long  a[maxn] , b[maxn] , ans;

bool cmp(long long a ,long long b){
	return a > b;
}

main(){
	int n , m ;cin >> n >> m;
	rep(i,n) cin >> a[i]; rep(i , m) cin >> b[i];
	sort(a,a+n) ; sort(b,b+m,cmp);
	for(int i = 0;i < min(n,m) and a[i] < b[i];++i) ans += (b[i] - a[i]);
	return(cout << ans , 0);
}
