#include<iostream>
#include<algorithm>
using namespace std;
long b[100000];
long a[200000];
long n, m;
int main() 
{
  cin >> n >> m;
  for (int k = 1; k <= n; k++) cin >> a[k];
  for (int k = 1; k <= m; k++) cin >> b[k];
  sort(a + 1, a + n + 1);
  sort(b + 1, b + m + 1);
  long long j = m, s = 0;
  for (int k = 1; k <= n; ++k) 
  {
    if (a[k] < b[j]) s += a[k] - b[j];
    --j;
  }
  cout << -s;
  return 0;
}
