#include<bits/stdc++.h>
using namespace std;
long long a1[100010],a2[100010],ans=0ll;
int n,m;

int main(){
    int i;
    cin>>n>>m;
    for(i=0;i<n;i++) cin>>a1[i];
    for(i=0;i<m;i++) cin>>a2[i];
    sort(a1,a1+n);sort(a2,a2+m);
    reverse(a2,a2+m);
    for(i=0;i<min(m,n);++i){
        ans+=max(0ll,(a2[i]-a1[i]));
    }
    cout<<ans;
    
}
