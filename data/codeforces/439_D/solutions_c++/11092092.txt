#include<bits/stdc++.h>
#define lli long long int
using namespace std;
int main()
{
    lli n,m,ans=0;
    cin>>n>>m;
    lli a[n],b[m];
    for(lli i=0;i<n;i++)
        cin>>a[i];
    for(lli i=0;i<m;i++)
        cin>>b[i];
    sort(a,a+n);
    sort(b,b+m,greater<lli>());
    for(lli i=0;i<min(n,m);i++)
    {
        if(a[i]<b[i])
            ans+=(b[i]-a[i]);
    }
    cout<<ans<<endl;
    return 0;
}
