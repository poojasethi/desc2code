#include<iostream>
#include<algorithm>
using namespace std;
int main()
{int dev[100005],mena[100005];
int n,m; cin>>n>>m;
for (int i=0;i<n;i++) cin>>dev[i];
sort(dev,dev+n);
for (int i=0;i<m;i++) cin>>mena[i];
sort(mena,mena+m); reverse(mena,mena+m);
long long ans=0;
for (int i=0;i<min(n,m);i++)
 if (dev[i]<mena[i]) ans+=(mena[i]-dev[i]);
 else
    break;
 cout<<ans;
return 0;
}
