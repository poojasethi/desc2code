#include <iostream>
#include <stdio.h>
#include <algorithm>

using namespace std;
typedef long long lld;
const int maxn = 1e5+9;

int n, m;
lld sol;
lld a[maxn];
lld b[maxn];

int main()
{
	cin >> n >> m;
	
	for (int i=0; i<n; i++)
		cin >> a[i];
	for (int i=0; i<m; i++)
		cin >> b[i];
	
	sort(a, a+n);
	sort(b, b+m);
	reverse(b, b+m);
	
	for (int i=0; i<min(n,m) && a[i]<b[i]; i++)
		sol += b[i] - a[i];
	
	cout << sol;
	
	return 0;
}
