#include <iostream>
#include <algorithm>
using namespace std;
int  a[100001];
int  b[100001];
long long n,m,q;
main(){
	cin >> n >> m ;
	for (int k=1;k<=n;k++){
		cin>>a[k] ;
	}
	for (int k=1;k<=m;k++){
		cin>>b[k];
	}

	sort(a+1,a+n+1);
        sort(b+1,b+m+1);
	for(int k=1;k<=min(m,n)&&a[k]<b[m-k+1];k++){
		q+=(b[m-k+1]-a[k]);
	}
	cout<<q<<endl;
}
