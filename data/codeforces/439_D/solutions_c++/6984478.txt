#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,l;
	cin>>i>>j;
	int A[i],B[j];
	for(k=0;k<i;k++)
		cin>>A[k];
	for(k=0;k<j;k++)
		cin>>B[k];
	sort(A,A+i);
	sort(B,B+j,greater<int>());
	long long ans=0,sum=0;
	k=0;
	while(A[k]<B[k])
	{
		k++;
		if(k==min(i,j))
			break;
	}
	for(l=0;l<k;l++)
	{
		ans+=A[l];
		sum+=B[l];
	}
	cout<<sum-ans;
	return 0;
}
