#include<bits/stdc++.h>
using namespace std;
int a[100005], b[100005];
int main()
{
    int m,n,k;
    cin>>m>>n;
    for(int i = 0; i<m; i++)
        cin>>a[i];
    for(int i = 0; i<n; i++)
        cin>>b[i];
    sort(a, a+m);
    sort(b, b+n);reverse(b, b+n);
    long long ans = 0;
    for(int i = 0; i<min(m, n); i++) {
        if(a[i]<b[i])
            ans+=b[i]-a[i];
    }
    cout<<ans<<endl;
}
