#include<bits/stdc++.h>
using namespace std;
const int mxmx=101000;
int n,m;
int a[mxmx],b[mxmx];
int main(){
	ios::sync_with_stdio(0);cin.tie(0);
	cin>>n>>m;
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<m;i++)cin>>b[i];
	sort(a,a+n);
	sort(b,b+m);
	int I=-1;
	for(int i=0;i<min(n,m);i++){
		if(a[i]<b[m-1-i]){
			I=i;
		}
		else break;
	}
	long long int ANS=0;
	for(int i=0;i<=I;i++){
		ANS+=b[m-i-1]-a[i];
	}
	cout<<ANS;
}
