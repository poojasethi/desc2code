#include<bits/stdc++.h>

using namespace std;

long long s1[100010],s2[100010],ans=0ll;

int main(){
    int n,m;
    cin>>n>>m;
    for (int i=0;i<n;++i)
        cin>>s1[i];
    for (int i=0;i<m;++i){
        cin>>s2[i];
    }
    sort (s1,s1+n);   sort(s2,s2+m);
    reverse(s2,s2+m);
    for (int i=0;i<min(n,m);++i){
        ans+=max(0LL, (s2[i]-s1[i]));
    }
    cout<<ans<<endl;
}
