#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int a[100003],b[100003];
int cmp(int a,int b)
{
	return a>b;
}
int main()
{
    int n,m;
    cin>>n>>m;
    int i;
    for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(i=0;i<m;i++)
		scanf("%d",&b[i]);
	sort(a,a+n);
	sort(b,b+m,cmp);
	int s=n<m?n:m;
	long long sum=0;
	for(i=0;i<s;i++)
		if(a[i]<b[i])
		sum+=b[i]-a[i];
	printf("%lld",sum);
    return 0;
}
