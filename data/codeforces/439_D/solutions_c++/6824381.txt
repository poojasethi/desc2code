#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

long long n,m,ans;

int main()
{
	cin >> n >> m;
	
	vector <int> a(n),b(m);
	
	for (int i = 0; i < n; i++)
		cin >> a[i];
		
	for (int i = 0; i < m; i++)
		cin >> b[i];
		
	sort(a.begin(),a.end());
	sort(b.rbegin(),b.rend());
	
	for (int i = 0; i < min(n,m) && b[i] > a[i]; i++)
		ans+=b[i]-a[i];
		
	cout << ans;
}
