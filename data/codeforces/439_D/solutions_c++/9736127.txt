#include <bits/stdc++.h>

using namespace std;

int main ()
{
    int n,m;
    cin>>n>>m;
    int a[n],b[m];
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int i=0;i<m;i++)
        cin>>b[i];
    sort(a,a+n);
    sort(b,b+m,greater<int>());
    int l;
    l=min(n,m);
    long long c=0;
    for(int i=0;i<l;i++)
    {
        if(b[i]>a[i])
            c=c+(b[i]-a[i]);
    }
    cout<<c<<'\n';
    return 0;
}
