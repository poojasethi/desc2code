#include <iostream>
#include <algorithm>
using namespace std;
const int MAXN=100100;
long long a[MAXN],b[MAXN];

int main()
{
  int n,m;
  cin>>n>>m;
  for(int i=1;i<=n;i++)
    cin>>a[i];
  for(int i=1;i<=m;i++)
    cin>>b[i];
  sort(a+1,a+1+n);
  sort(b+1,b+1+m);
  reverse(b+1,b+1+m);
  long long ans=0;
  for(int i=1;i<=min(n,m);i++)
    if(a[i]<b[i])
      ans+=b[i]-a[i];
  cout<<ans<<endl;
  return 0;
}
