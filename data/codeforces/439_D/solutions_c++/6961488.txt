#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int n,m,i,min;
	__int64 sum;
	int a[100001],b[100001];
	while(scanf("%d%d",&n,&m)!=EOF){
		sum=0;
		for(i=1;i<=n;i++)
			scanf("%d",&a[i]);
		for(i=1;i<=m;i++)
			scanf("%d",&b[i]);
		sort(a+1,a+1+n);
		sort(b+1,b+1+m);
		min=n<m?n:m;
		for(i=1;i<=min;i++)
			if(a[i]<b[m+1-i])
				sum+=b[m+1-i]-a[i];
			else
				break;
		printf("%I64d\n",sum);
	}
}
