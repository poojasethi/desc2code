#include <bits/stdc++.h>
using namespace std;

int a[100010], b[100010];

int main()
{
	int n, m;
	scanf("%d %d\n", &n, &m);
	for(int i = 0; i < n; ++i) scanf("%d", &a[i]);
	for(int i = 0; i < m; ++i) scanf("%d", &b[i]);
	sort(a, a + n);
	sort(b, b + m);
	reverse(b, b + m);
	long long result = 0;
	for(int i = 0; i < min(m, n) && a[i] < b[i]; ++i) result += (long long)b[i] - a[i];
	printf("%I64d\n", result);
	return 0;
}
