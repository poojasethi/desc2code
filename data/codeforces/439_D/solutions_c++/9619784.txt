#include <iostream>
#include <algorithm>
using namespace std;

int main ()
{
    int n, m;
    cin >> n >> m;
    int a[n], b[m];
    for(int i=0; i<n; i++)
        cin >> a[i];
    for(int i=0; i<m; i++)
        cin >> b[i];
    sort(a, a+n); sort(b, b+m); reverse(b,b+m);
    long long counter = 0;
    for(int i=0; i<min(n,m); i++)
        if(b[i]>a[i]) 
            counter += b[i]-a[i];
    cout << counter << endl;

    return 0;
}
