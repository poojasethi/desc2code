#include<bits/stdc++.h>
using namespace std;

bool cmp ( int x, int y){
	return y < x;
}

int n,m,a[100010],b[100010];
long long hasil;

int main(){
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++) scanf("%d",&a[i]);
	for(int i=0;i<m;i++) scanf("%d",&b[i]);
	sort(a,a+n); sort(b,b+m,cmp);
	for(int i=0;i<n;i++){
		if(a[i] < b[i])	hasil += b[i]-a[i];
		else break;
	}
	cout<<hasil<<endl;
}
