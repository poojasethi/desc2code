#include<bits/stdc++.h>
using namespace std;
int main(){
long long n,m,sum=0;
cin >> n >> m;
long long a[n],b[m];
for(int i = 0 ; i <n;++i)
cin >> a[i];
for(int j = 0 ; j < m;++j)
cin >> b[j];
sort(a,a+n);
sort(b,b+m);
reverse(b,b+m);

n=min(n,m);
    for(int i = 0 ; i < n ;++i)
        if(b[i] > a[i])sum+=b[i]-a[i];
        cout << sum;

}
