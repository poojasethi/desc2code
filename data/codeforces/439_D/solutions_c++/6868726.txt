#include<iostream>
#include<algorithm>
using namespace std;
int main()
{

 int n,d;
 cin>>n>>d;
 int a[100002],b[100002];
 long long s=0,k=0;
 for(int i=0;i<n;i++)
  cin>>a[i];
   
 for(int i=0;i<d;i++)
  cin>>b[i];
  sort(a,a+n);
  sort(b,b+d);
  for(int i=0,j=d-1;i<n&&j>=0;++i,--j)
    {
        if(b[j]<=a[i])break;
        s += b[j]-a[i];
    }
    cout<<s<<endl;
   
      //system("pause");
      return 0;
             }
