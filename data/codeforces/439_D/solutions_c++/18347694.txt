 #include <bits/stdc++.h>
    using namespace std;
long long a[100005],b[100005];
int main()
{	
    long long n,m;
   
    cin>>n>>m;
   
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    
	}
	for(int i=0;i<m;i++)
	{
		cin>>b[i];
	}
	sort(a,a+n);
	sort(b,b+m);
	
	long long ans=0;
	for(int i=0;a[i]<b[m-1-i]&&i<n;i++)
	{
		ans+=-a[i]+b[m-1-i];
	}
	cout<<ans;

}
