#include<iostream>
#include<algorithm>

using namespace std ;

const int maxn=1e5+5 ;
long long a[maxn] , b[maxn] ;
int n , m ;
long long ans=0 ;


int main()
{
	cin >> n >> m ;
	for(int i=0 ; i<n ; i++) 
		cin >> a[i] ;
	for(int i=0 ; i<m ; i++)
		cin >> b[i] ;
	sort(a,a+n) ;
	sort(b,b+m) ;
	reverse(b,b+m) ;
	for(int i=0 ; i<min(n,m) ; i++)
		if(a[i]<b[i])
			ans+=b[i]-a[i] ;
	cout << ans << endl ;
	return 0 ;
}
