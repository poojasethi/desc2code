#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
	int N, M;
	cin >> N >> M;

	vector<int> A(N), B(M);
	for (int i = 0; i < N; i++)
		cin >> A[i];
	for (int i = 0; i < M; i++)
		cin >> B[i];

	sort(A.begin(), A.end());
	sort(B.rbegin(), B.rend());

	long long total = 0;
	for (int i = 0; i < N && i < M && A[i] < B[i]; i++)
		total += B[i] - A[i];

	cout << total << "\n";
	
	return 0;
}
