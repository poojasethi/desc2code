#include<iostream>
#include<cstdio>
#include<vector>
#include<cmath>
#include<map>
#define pb push_back
using namespace std;
const int N = 111111;
const int M = 20 * N;
vector<int> edge[N];
int val[N];
int v, n, que;
int tail[M];
int f[N];
int p, q;
int gcd(int p, int q){
    if (q == 0) return p;
    return gcd(q,p % q);
}
void dfs(int u, int fa){
    f[u] = fa;
    for (int i = 0; i < edge[u].size(); i++)
    {
        v = edge[u][i];
        if (v == fa) continue;
        dfs(v, u);
    }
}
int tt(int i){
    int ans = 0;
    for (int j = f[i]; j != 0; j = f[j])
        if (gcd(val[i], val[j]) > 1)
        {
            ans = j;
            break;
        }
    if (ans == 0) ans = -1;
    return ans;
}
int main(){
    scanf("%d%d", &n, &que);
    for(int i = 1; i <= n; i++)
        scanf("%d", &val[i]);
    for (int i = 1; i < n;i++)
    {
        scanf("%d%d", &p, &q);
        edge[p].pb(q);
        edge[q].pb(p);
    }
    dfs(1, 0);
    for (int i = 1; i <= que; i++)
    {
        scanf("%d%d", &p, &q);
        if (p == 1)
            printf("%d\n", tt(q));
        else
        {
            scanf("%d", &p);
            val[q] = p;
        }
    }
}
