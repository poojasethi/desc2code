#include <cstring>
#include <vector>
#include <cstdio>
#include <cmath>
using namespace std;
const int lim=2000000;
const int N=100005;
int a[N], f[N];
vector<int> eg[N];
int gcd(int x,int y)
{
    int tmp;
    while(y){
        tmp=y;
        y=x%y;
        x=tmp;
    }
    return x;
}
void dfs(int cur,int fa)
{
    int to;
    int i;
    f[cur]=fa;
    for(i=0;i<eg[cur].size();i++){
        to=eg[cur][i];
        if(to==fa) continue;
        dfs(to,cur);
    }
}
int main()
{
    int i, j, n, q, x, y;
    scanf("%d%d",&n,&q);
    for(i=1;i<=n;i++) scanf("%d",&a[i]);
    for(i=1;i<n;i++){
        scanf("%d%d",&x,&y);
        eg[x].push_back(y);
        eg[y].push_back(x);
    }
    int op, ret, v, w;
    dfs(1,0);
    while(q--){
        scanf("%d",&op);
        if(op==1){
            scanf("%d",&v);
            for(i=f[v];i;i=f[i]){
                if(gcd(a[i],a[v])>1) break;
            }
            if(i==0) puts("-1");
            else printf("%d\n",i);
        }else {
            scanf("%d%d",&v,&w);
            a[v]=w;
        }
    }
    return 0;
}
