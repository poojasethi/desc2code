#include <cstdio>
#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;
int gcd(int a,int b)
{
	return b?gcd(b,a%b):a;
}
int a[100005],next[100005];
vector<int> V[100005];
void dfs(int x,int f)
{
	for(int i=0;i<V[x].size();i++)
	{
		int u=V[x][i];
		if(u==f)
			continue;
		else
			next[u]=x;
		dfs(u,x);
	}
}
int main()
{
	int n,q;
	scanf("%d%d",&n,&q);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
		for(int i=1;i<n;i++){
			int x,y;
			scanf("%d%d",&x,&y);
			V[x].push_back(y);
			V[y].push_back(x);
		}
	dfs(1,0);
	for(int kk=0;kk<q;kk++)
	{
		int o,v,w;
		scanf("%d",&o);
		if(o==1)
		{
			scanf("%d",&v);
			bool out=false;
			for(int i=next[v];i!=0;i=next[i])
			{
				if(gcd(a[v],a[i])!=1)
				{
					out=true;
					printf("%d\n",i);
					break;
				}
			}
			if(!out)
				cout<<-1<<endl;

		}
		else
		{
			scanf("%d%d",&v,&w);
			a[v]=w;
		}
	}
	return 0;
}
