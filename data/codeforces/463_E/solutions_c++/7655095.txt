#include <bits/stdc++.h>
#define fi freopen("inp.txt", "r", stdin)
#define fo freopen("out.txt", "w", stdout)
#define MAX
#define N 100010

using namespace std;

int n, q;
int par[N], val[N], res[N];
vector <int> a[N];

void DFS(int u, int f) {
    par[u] = f;
    for(int i = 0; i < (int)a[u].size(); i++) {
        int v = a[u][i];
        if(v != f) DFS(v, u);
    }
}

int Ans(int v) {
    if(res[v]) return res[v];
    res[v] = -1;
    int u = par[v];
    while(u) {
        if(__gcd(val[u], val[v]) > 1)
        return (res[v] = u);
        u = par[u];
    }
    return res[v];
}

void Update(int u) {
    res[u] = 0;
    for(int i = 0; i < (int)a[u].size(); i++) {
        int v = a[u][i];
        if(v != par[u]) Update(v);
    }
}

int main() {
    //fi, fo;
    scanf("%d %d", &n, &q);
    for(int i = 1; i <= n; i++) scanf("%d", val + i);
    for(int i = 1; i < n; i++) {
        int u ,v;
        scanf("%d %d", &u, &v);
        a[u].push_back(v);
        a[v].push_back(u);
    }
    DFS(1, 0);
    while(q--) {
        int type, v;
        scanf("%d %d", &type, &v);
        if(type == 1) printf("%d\n", Ans(v));
        else {
            scanf("%d", val + v);
            Update(v);
        }
    }
    return 0;
}
