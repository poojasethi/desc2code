#include <bits/stdc++.h>
#define rep(i,a,b) for(int i=a;i<=b;i++)
#define For(i,a,b) for(int i=a;i< b;i++)
#define pb push_back
#define vi vector<int>

using namespace std;
const int maxn=100005;

int n,q,w[maxn],f[maxn];
vi a[maxn];

void enter()
{
    //freopen("in.tt","r",stdin);
    cin >> n >> q;
    rep(i,1,n) cin >> w[i];
    rep(i,1,n-1)
    {
        int x,y; cin >> x >> y;
        a[x].pb(y);
        a[y].pb(x);
    }
    f[1]=-1;
}

int gcd(int a,int b)
{
    if (b==0) return a;
    return gcd(b,a%b);
}

void dfs(int u)
{
    int v;
    For(i,0,a[u].size())
    {
        v=a[u][i];
        if (!f[v])
        {
            f[v]=u;
            dfs(v);
        }
    }
}

int find_res(int v)
{
    int u=f[v];
    while (u!=-1)
    {
        if (gcd(w[u],w[v])>1) return u;
        u=f[u];
    }
    return -1;
}

void process()
{
    int t;
    dfs(1);
    while (q--)
    {
        cin >> t;
        if (t==1)
        {
            int v;
            cin >> v;
            cout << find_res(v) << endl;
        }
        else
        {
            int x,y; cin >> x >> y;
            w[x]=y;
        }
    }
}

int main()
{
    ios::sync_with_stdio(0); cin.tie(0);    enter();
    process();
}
