#include <bits/stdc++.h>
#define ll long long
using namespace std;
vector<ll>adj[100001];
ll parent[100001],val[100001],mark[100001];
void setval(ll a,ll b)
{
    parent[a]=b;
    mark[a]=1;
   // cout<<a<<b<<endl;
    ll i;
    for(i=0;i<adj[a].size();i++)
    {
    if(!mark[adj[a][i]])
       { setval(adj[a][i],a);}
    }
}
ll gcd(ll a,ll b)
{
    if(b==0)
    {
        return a;
    }
    else
    {
        return gcd(b,a%b);
    }
}
ll getval(ll a, ll b)
{
    if(a==-1)
    {
        return -1;
    }
    if(gcd(val[a],val[b])>1)
    {
        return a+1;
    }
    return getval(parent[a],b);
}
int main()
{
    ll n,q,i,k,g,p,x,y;
    cin>>n>>q;
    for(i=0;i<n;i++)
    {
        cin>>val[i];
        mark[i]=0;
    }
    for(i=0;i<n-1;i++)
    {
        cin>>x>>y;
        adj[x-1].push_back(y-1);
        adj[y-1].push_back(x-1);
    }
    setval(0,-1);
  //  cout<<parent[2]<<" "<<-5<<endl;
    for(i=0;i<q;i++)
    {
        cin>>k;
        if(k==2)
        {
            cin>>g>>p;
            val[g-1]=p;
        }
        else
        {
            cin>>g;
            cout<<getval(parent[g-1],g-1)<<endl;
        }
    }
    return 0;
}
