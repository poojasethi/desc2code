#include <stdio.h>

int main()
{
    int n,m;
    int s,f,t;
    int i, x;

    scanf("%d %d", &n, &m);
    m = m*2-2;

    for(i=0; i<n; i++)
    {
        scanf("%d %d %d", &s, &f, &t);
        if(s == f)
            x = t;
        else
            x = (t+m-(s>f?m+2-s:s))/m*m+(s>f?m+2-f:f)-1;
        printf("%d\n", x);
    }

    return 0;
}
