#include <iostream>
using namespace std;
int main() {
    int n,m,p,s,f,t,x;
    cin>>n>>m; p=(m-1)*2;
    while (n--) {
        cin>>s>>f>>t;
        if (s==f) cout<<t<<endl;
        else {
            x = t/p*p+(s<f?s-1:2*m-1-s);
            cout<<(x<t?x+p:x)+(f>s?f-s:s-f)<<endl;
        }
    }
}
