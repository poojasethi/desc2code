#include <iostream>
#include <stdio.h>
using namespace std;

int main(){
    int n,m,s,f,t;
    cin>>n>>m;
    m=m*2-2;
    while(n!=0){
        cin>>s>>f>>t;
        if (s!=f)
            cout<<(t+m-(s>f?m+2-s:s))/m*m+(s>f?m+1-f:f-1)<<endl;
        else
            cout<<t<<endl;
        n--;
    }
    return 0;
}
