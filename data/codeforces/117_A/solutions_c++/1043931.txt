#include<cstdio>
inline int div(int n, int d) {
  return (n+d-1)/d;
}
main() {
  int i, t, x, y, n, m;
  scanf("%d %d", &n, &m);
  for (i = 0; i < n; ++i) {
    scanf("%d %d %d", &x, &y, &t);
    if (x == y) { } 
    else if (x < y) t = div(t-x+1, 2*m-2)*(2*m-2) + x - 1 + (y-x);
    else t = (m-1) + div(t-2*m+x+1, 2*m-2)*(2*m-2) + (m-x) + (x-y);
    printf("%d\n", t);
  }
}
