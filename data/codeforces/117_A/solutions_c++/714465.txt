#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int m,n,a,b,c,t;
	cin>>n>>m;
	for(int i=0;i<n;i++)
	{
		cin>>a>>b>>c;
		if(b==a)
		{
			t=c;
			cout<<t<<endl;
		}
		else
		{
			
			if(b>a)
				t=a-1;
			else
				t=(m-1)+(m-a);
		
			while(1)
			{
				if(t>=c)
				{
					t=t+abs(b-a);
					cout<<t<<endl;
					break;
				}
				else
					t=t+2*(m-1);
		
			}
		}
	}
	return 0;
}
