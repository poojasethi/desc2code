#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;
int main(void){
    int n,m,s,f,t,x,y;
    scanf("%d%d", &n,&m);
    while( n-- ){
        scanf("%d%d%d", &s,&f,&t);
        if ( s == f ){ printf("%d\n", t); continue;}
        if ( s<f ) x = s-1;
        else x = m-1+m-s;
        while( 1 ){
            if ( x >= t ){
                x += abs(f-s);
                printf("%d\n", x);
                break;  
            }else x += 2*m-2;
        }
    }
}
