#include<iostream>
using namespace std;
int s,f,t;
int main(){
	int n,m;
	while(cin>>n>>m){
		for(int i=0;i<n;++i){
			int ti=0;
			cin>>s>>f>>t;
			if(s>f){
				while(t>(((s-1+(m-s)*2)%(2*m-2))+ti)){
					ti+=(2*m-2);
				}
				if(f==1)ti+=((m-1)*2);
				cout<<((((m-f)*2+f-1)%(2*m-2))+ti)<<endl;
			}
			else if(s<f){
				while(t>(((s-1)%(2*m-2))+ti)){
					ti+=(2*m-2);
				}
				cout<<(((s-1)%(2*m-2))+ti)+(f-s)<<endl;
			}
			else cout<<t<<endl;
		}
	}
	return 0;
}
