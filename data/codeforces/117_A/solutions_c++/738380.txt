#include<stdio.h>
int main(){
	int n,m,ac,a,b,t,i;
	scanf("%d%d",&n,&m);
	ac=2*(m-1);
	for(i=0;i<n;i++){
		scanf("%d%d%d",&a,&b,&t);
		if(a<b){
			if(t%ac<=(a-1)){
				printf("%d\n",(t/ac)*ac+b-1);
			}
			else{
				printf("%d\n",(t/ac+1)*ac+b-1);
			}
		}
		if(a>b){
			if(t<ac/2){
				printf("%d\n",ac/2+m-b);
			}
			else if((t-ac/2)%ac<=m-a){
				printf("%d\n",((t-ac/2)/ac)*ac+m-b+ac/2);
			}
			else{
				printf("%d\n",((t-ac/2)/ac+1)*ac+m-b+ac/2);
			}
		}
		if(a==b) printf("%d\n",t);
	}
}
