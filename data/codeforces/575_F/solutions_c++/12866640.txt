#include<cstdio>
#include<algorithm>
using namespace std;

const int MN=(int)1e6+6;
int n, m, p, k;
long long wyn;

int main(){
    int i, j, a, b, vp, vk; 
    scanf("%d", &n);
    scanf("%d", &p);
    k=p;
    for(i=0; i<n; i++){
        scanf("%d %d", &vp, &vk);
        if(k<vp){   //nowy za
            wyn+=vp-k;
            p=k;
            k=vp;
            continue;   
        }
        if(p>vk){   //nowy przed
            wyn+=p-vk;
            k=p;
            p=vk; 
            continue;  
        }
        //maja przeciecie
        
        p=max(p, vp);
        k=min(k, vk);
        
    }
    
    printf("%I64d\n", wyn);
    scanf("%d", &n);
    return 0;
}
