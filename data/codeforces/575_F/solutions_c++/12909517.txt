typedef long long ll;

const int size = 10111;

int a[size][2],n,x;
ll dp[2][size][2];

int main(){
    cin>>n>>x;
    int l=x,r=x;
    ll ans=0;
    for(int i=1;i<=n;i++){
        int a,b;
        scanf("%d%d",&a,&b);
        if(r<=a){
            ans+=a-r;
            l=r;r=a;
        }
        else if(l>=b){
            ans+=l-b;
            r=l;l=b;
        }
        else {
            l=max(l,a);r=min(r,b);
        }
    }
    cout<<ans<<endl;
}
