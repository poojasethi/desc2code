#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
typedef long long i64;

int n, m;

int main() {

	scanf("%d%d", &n, &m);
	int l = m, r = m;
	i64 ans = 0;
	for (int i = 1; i <= n; ++ i) {
		int p, q;
		scanf("%d%d", &p, &q);
		if (p <= l && r <= q) continue;
		if (q < l) {
			ans += l - q;
			r = l;
			l = q;
		}
		else if (r < p) {
			ans += p - r;
			l = r;
			r = p;
		}
		else {
			l = max(l, p);
			r = min(r, q);
		}
	}
	printf("%I64d\n", ans);

	return 0;
}
