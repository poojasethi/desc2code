#include<bits/stdc++.h>
#define ll long long
using namespace std;
int n,x;
pair<int,int> p[5555];
int main()
{
    while(~scanf("%d%d",&n,&x)){
        for(int i=0;i<n;i++){
            scanf("%d%d",&p[i].first,&p[i].second);
        }
        int l=x,r=x;
        ll ans=0;
        for(int i=0;i<n;i++){
            if(p[i].second<l){
                ans+=l-p[i].second;
                r=l,l=p[i].second;
            }else if(p[i].first>r){
                ans+=p[i].first-r;
                l=r,r=p[i].first;
            }else{
                l=max(l,p[i].first);
                r=min(r,p[i].second);
            }
        }
        printf("%I64d\n",ans);
    }
    return 0;
}
