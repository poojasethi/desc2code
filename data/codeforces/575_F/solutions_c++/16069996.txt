#include<iostream>
using namespace std;

int n,x;
int a,b,c,d;
long long ans=0;

int main(){
	cin >> n >> x;
	c = d = x;
	for(int i = 0; i < n; i++)
    {
		cin >> a >> b;
		if(a <= c&&d <= b)
            continue;

		if(c > b)
        {
			ans += c - b;
			d = c;
			c = b;
		}
		else if(d < a){
			ans += a - d;
			c = d;
			d = a;
		}else{
			if(c < a)
                c = a;
			if(d > b)
                d = b;
		}
	}
	cout << ans << endl;
	return 0;
}
