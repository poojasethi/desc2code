#include <cmath>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#define rint register int
#define ll long long
#define ld long double
#define For(i, a, b) for (int i = a, __end = b; i <= __end; ++i)
#define Ford(i, a, b) for (int i = a, __end = b; i >= __end; --i)
using namespace std;

const int MaxN = 5010;
int N, l, r;
ll ans;

int main()
{
	scanf("%d%d", &N, &l);
	r = l;
	For(i, 1, N) {
		int u, v;
		scanf("%d%d", &u, &v);
		if (max(l, u) <= min(r, v)) {
			l = max(l, u);
			r = min(r, v);
		} else if (v < l) {
			ans += l - v;
			r = l;
			l = v;
		} else if (r < u) {
			ans += u - r;
			l = r;
			r = u;
		}
	}
	cout << ans << endl;
	return 0;
}
