#include <bits/stdc++.h>

#define ll long long

using namespace std;

int n, x;
ll res;

int main() {
#ifdef LOCAL
    freopen("input.txt", "r", stdin);
#endif
    scanf("%d%d", &n, &x);
    int l = x, r = x;
    for (int i = 1; i <= n; ++i) {
        int u, v;
        scanf("%d%d", &u, &v);
        if (u > r) {
            res += (ll) (u - r);
            l = r;
            r = u;
        } else if (v < l) {
            res += (ll) (l - v);
            r = l;
            l = v;
        } else {
            l = max(l, u);
            r = min(r, v);
        }
    }
    printf("%I64d\n", res);
#ifdef LOCAL
    printf("\n%.3lf s\n", (double) clock() / (double) CLOCKS_PER_SEC);
#endif
}
