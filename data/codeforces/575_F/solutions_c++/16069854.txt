#include <iostream>

using namespace std;

int main()
{
    long long int N,start;
    cin >> N >> start;
    long long int le,ri;
    le = ri = start;
    long long int res = 0;
    for(int i = 0; i < N;i++){
        int LE,RI;
        cin >> LE >> RI;
        if (le >= LE && ri<= RI){
            continue;
        }
        else if(ri < LE){
            res += LE - ri;
            le = ri;
            ri = LE;
        }
        else if (le > RI){
            res += le -RI;
            ri = le;
            le =RI;
        }
        else{
            if(le < LE)
                le = LE;
            if(ri > RI)
                ri = RI;
        }
    }
    cout << res << endl;
    return 0;
}
