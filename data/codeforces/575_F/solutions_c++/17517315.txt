#include<iostream>
#include<cstdio>
#include<cstring>
#include<map>
#include<list>
#include<algorithm>
#include<vector>
#include<string>
#include<fstream>
//#pragma comment(linker,"/STACK:1024000000,1024000000")
using namespace std;
const int N=1e5+5;
int n,x;
int main()
{
    while(~scanf("%d",&n))
    {
        scanf("%d",&x);
        long long ans=0;
        int ll,rr;
        ll=rr=x;
        for(int i=1;i<=n;i++)
        {
            int l,r;
            scanf("%d %d",&l,&r);
            if(rr>=l&&ll<=r) rr=min(rr,r),ll=max(ll,l);
            else if(rr<l) ans+=l-rr,ll=rr,rr=l;
            else if(ll>r) ans+=ll-r,rr=ll,ll=r;
        }
        printf("%I64d\n",ans);
    }
   return 0;
}
