#include <cstdio>

int GetInt()
{
	register int aa = 0, bb = 1;
	register char ch = getchar();
	while ((ch < '0' || ch > '9') && ch != '-')
		ch = getchar();
	ch == '-' ? aa = 0, bb = -1 : aa = ch - '0';
	while (ch = getchar(), ch >= '0' && ch <= '9')
		aa = (aa << 1) + (aa << 3) + ch - '0';
	return bb == 1 ? aa : -aa;
}

int n, x;
long long ans;

int main()
{
	register int i, L, R, l, r;
	n = GetInt();
	l = r = GetInt();
	for (i = 1; i <= n; i++)
	{
		L = GetInt();
		R = GetInt();
		if (R < l)
		{
			ans += l - R;
			r = l;
			l = R;
		} else if (L > r)
		{
			ans += L - r;
			l = r;
			r = L;
		} else
		{
			if (L > l) l = L;
			if (R < r) r = R;
		}
	}
	printf("%lld", ans);
}
