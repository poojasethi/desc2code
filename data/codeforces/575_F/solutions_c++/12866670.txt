#include<iostream>
#define ll long long
using namespace std;
int main()
{
	ll ans=0;
	int n;
	ll l[5004],r[5004];
	cin>>n;
	ll x;
	cin>>x;
	for(int i=0;i<n;++i)
	{
		cin>>l[i]>>r[i];
	}
	for(int i=0;i<n;++i)
	{
		if(x<l[i])
		{
			ans+=l[i]-x;
			ll a=x,b=l[i];
			ll q=-1;
			for(int j=i+1;j<n;++j)
			{
				if(r[j]<=a){q=a;break;}
				if(l[j]>=b){q=b;break;}
				b=min(r[j],b);
				a=max(a,l[j]);
			}
			if(q==-1)q=a;
			x=q;
		}
		else if(x>r[i])
		{
			ans+=x-r[i];
			ll a=r[i],b=x;
			ll q=-1;
			for(int j=i+1;j<n;++j)
			{
				if(r[j]<=a){q=a;break;}
				if(l[j]>=b){q=b;break;}
				b=min(r[j],b);
				a=max(a,l[j]);
			}
			if(q==-1)q=a;
			x=q;
		}
	}
	cout<<ans<<endl;
	return 0;
}
