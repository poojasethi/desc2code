#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;

int main()
{
    long long ans=0;
    int n,x,y,l,r;
    scanf("%d%d",&n,&x);
    y=x;
    for(int i=0;i<n;i++)
    {
       scanf("%d%d",&l,&r);
       if(l<=x&&r>=y)
       continue;
       else if(x>r)
       {
           ans=ans+(x-r);
           y=x,x=r;
       }
       else if(l>y)
       {
           ans=ans+(l-y);
           x=y,y=l;
       }
       else
       {
           x=max(x,l);
           y=min(r,y);
       }
      // cout<<i<<" "<<ans<<endl;
    }
    cout<<ans<<endl;
    return 0;
}
