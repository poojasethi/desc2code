# include <stdio.h>
# include <algorithm>
# include <iostream>
using namespace std;
int n;
long long ans=0;
struct range {
	int l,r;
}pos,input;
int main() {
	//freopen("bulbo.in","r",stdin);
	//freopen("bulbo.out","w",stdout);
	scanf("%d%d",&n,&pos.r);
	pos.l=pos.r;
	for (int i=1;i<=n;++i) {
		scanf("%d%d",&input.l,&input.r);
		if (input.l>pos.r) {	
			pos.l=pos.r;
			ans+=input.l-pos.r;
			pos.r=input.l;
		} else if(input.r<pos.l) {
			pos.r=pos.l;
			ans+=pos.l-input.r;
			pos.l=input.r;
		} else {
			pos.l=max(pos.l,input.l);
			pos.r=min(pos.r,input.r);
		}
	}
	cout<<ans<<endl;
	return 0;
}
