#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    int n,x;
    cin >> n >>x;
    int y = x;
    long long res = 0;
    for(int i=0;i<n;i++)
    {
        int l,r;
        scanf("%d%d",&l,&r);
        if(max(x,l)<=min(y,r))
        {
            x = max(x,l);
            y = min(y,r);
        }else
        if(y<l)
        {
            res+=l-y;
            x=y;
            y=l;
        }else
        {
            res+=x-r;
            y=x;
            x=r;
        }
    }
    cout <<res<<endl;



    for(int i=0;i<n;i++)
    return 0;
}
