#include<cstdlib>
#include<cstdio>
#include<algorithm>

using namespace std;

int n,m,i,j,k,l,r,u,v;
long long ans;

int main(){
	scanf("%d%d",&n,&k);
	l=r=k;
	for(i=1;i<=n;i++){
		scanf("%d%d",&u,&v);
		if(max(l,u)<=min(r,v)){
			l=max(l,u);
			r=min(r,v);
		}else if(r<=u){
			ans+=u-r;
			l=r;
			r=u;
		}else{
			ans+=l-v;
			r=l;
			l=v;
		}
	}
	printf("%I64d\n",ans);
	return 0;
}
