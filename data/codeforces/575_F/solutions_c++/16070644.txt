#include<cstdio>
using namespace std;
int n,l,r,x,a,b;
long long ans=0;
int main()
{
    scanf("%d %d",&n,&x);
    l=x;r=x;
    for(int i=0;i<n;i++)
    {
        scanf("%d %d",&a,&b);
        if(l>=a&&r<=b) {continue;}
        else if(l>b)
        {
            ans+=l-b;
            r=l;l=b;
        }
        else if(r<a)
        {
            ans+=a-r;
            l=r;r=a;
        }
        else if(a<=l)
        {
            r=b;
        }
        else if(r<=b)
        {
            l=a;
        }
        else{l=a;r=b;}

    }
    printf("%I64d\n",ans);

}
