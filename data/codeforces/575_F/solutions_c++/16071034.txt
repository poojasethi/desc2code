#include<iostream>
using namespace std;

long long result=0;

int main()
{
    int a,b,total,l,r,position;
	cin >> total;
	cin >> position;
	l=r=position;
	int i;
	for(i=0;i<total;i++)
    {
		cin >> a;
		cin >> b;
		if(a<=l&&r<=b) continue;
		if(r<a)
        {
			result+=a-r;
			l=r;
			r=a;
		}
		else if(l>b)
        {
			result+=l-b;
			r=l;
			l=b;
		}
		else
		{
		    if(r>b)r=b;
			if(l<a)l=a;
		}
	}
	cout << result << endl;
	return 0;
}
