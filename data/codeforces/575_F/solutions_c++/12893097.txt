#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>
#include <iomanip>
#include <cstring>

using namespace std;

const int inf=1e9+7;
typedef long long ll;
int main()
{
    ll ans=0;int i,n,xx,l[5005],r[5005],j,lx,rx;
    cin>>n>>xx;lx=rx=xx;
    for(i=1;i<=n;i++) {
        scanf("%d%d",&l[i],&r[i]);
    }
    for(i=1;i<=n;i++) {
        if(r[i]<=lx) {
            ans+=lx-r[i];rx=lx;lx=r[i];
        } else if(r[i]<=rx) {
            rx=r[i];lx=max(lx,l[i]);
        } else if(l[i]<=rx) {
            lx=max(lx,l[i]);
        }
        else {
            ans+=l[i]-rx;lx=rx;rx=l[i];
        }
    }
    cout<<ans<<endl;
}
