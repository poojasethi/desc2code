#include<cstdio>
using namespace std;

int n,l,r,x,a,b;
long long ans=0;

int main()
{
	scanf("%d %d",&n,&x);
	l=r=x;
	for(int i=0; i<n; i++){
		scanf("%d %d",&a,&b);
		if(a<=l&&r<=b)continue;
		if(l>b){
			ans+=l-b;
			r=l;
			l=b;
		}
		else if(r<a){
			ans+=a-r;
			l=r;
			r=a;
		}else{
			if(l<a)l=a;
			if(r>b)r=b;
		}
	}
	printf("%I64d\n",ans);
	return 0;
}
