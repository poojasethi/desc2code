#include <bits/stdc++.h>
using namespace std;
int ar[300000];
long long dp[5001][5001];
void first(){
	int n,k,i;cin>>n>>k;
	for(i=0;i<n;++i)scanf("%d",&ar[i]);
	sort(ar,ar+n);
	int large= n%k,part= n/k;
	for(int i=0;i<=k;++i)
		for(int j=0;j<=large;++j)
			dp[i][j]=1e18;
	dp[0][0]=0;
	for(int i=1;i<=k;++i)
		for(int j=0;j<=large;++j){
			dp[i][j]= dp[i-1][j]+ar[i*part+j-1]-ar[i*part+j-part];
			if(j)
				dp[i][j]= min(dp[i][j],dp[i-1][j-1]+ar[i*part+j-1]-ar[i*part+j-part-1]);
		}
	cout<<dp[k][large]<<endl;
}
void second(){
	int n,x,y,l,r;cin>>n>>x;y=x;
	long long ret= 0;
	for(int i=0;i<n;++i){
		cin>>l>>r;
		if(l<=x && y<=r) x= min(x,r),y= max(y,l);
		else if(x<l) y=x,ret+= l-x,x=l;
		else if(r<y) x=y,ret+= y-r,y=r;
	}
	cout<<ret<<endl;
}
int main(){
	second();
}
