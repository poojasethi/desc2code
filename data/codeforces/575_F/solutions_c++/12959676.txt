#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
int main()
{
	int n, x, a, b;
	cin >> n >> x;
    a = b = x;
    long long ans = 0;
    for (int i = 1; i <= n; ++i)
    {
        int l, r;
        cin >> l >> r;
        if (l <= a && r >= b) continue;
        if (r < a) ans += a - r, b = a, a = r;
        else if (l > b) ans += l - b, a = b, b = l;
        else a = max(a, l), b = min(b, r);
    }
    cout << ans << endl;
	return 0;
}
