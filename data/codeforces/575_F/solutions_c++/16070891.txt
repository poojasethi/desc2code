#include <iostream>

using namespace std;

int main()
{
    int turn,position,p,q;
    long long int total=0;
    cin>>turn>>position;
    p=q=position;
    for(int i=0;i<turn;i++)
    {
        int s,l;
        cin>>s>>l;

            if((q>=s&&p<=l))continue;
            if(l<q)
            {
                total+=(q-l);
                p=q;
                q=l;
            }
            else if(p<s)
            {
                total+=(s-p);
                q=p;
                p=s;
            }
            else
            {

                if(q<=s)q=s;
                if(p>=l)p=l;
            }
        }


    cout<<total<<endl;
    return 0;
}
