#include <iostream>

using namespace std;

int main()
{
    int i,x;
    int a,b;
    long long int cost=0;
    cin>>i>>x;
    int m,n;
    m=n=x;
    while(i--)
    {
        cin>>a>>b;
        if(a<=m&&n<=b)
            continue;
        if(m>b)
        {
            cost=cost+m-b;
            n=m;
            m=b;
        }
        else if(n<a)
        {
            cost=cost+a-n;
            m=n;
            n=a;
        }
        else
        {
            if(m<a)
                m=a;
            if(n>b)
                n=b;
        }
    }
    cout<<cost<<endl;
    return 0;
}
