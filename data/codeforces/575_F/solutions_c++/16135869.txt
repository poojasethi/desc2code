#include <iostream>
#include <cstdio>
using namespace std;

long long  sum = 0;
int main()
{
    int n,x;
    cin>>n>>x;
    int a,b,l,r;
    l=r=x;
    for(int i=0;i<n;i++)
    {
        cin >> a >> b;
        if( a<=l && r<=b )
        {
            continue;
        }
        if(l>b)
        {
           sum+= (l-b);
           r = l;
           l =b;
        }
        else if(r<a)
        {
            sum+=(a-r);
            l = r;
            r = a;
        }
        else
        {
			if(l<a) l=a;
			if(r>b) r=b;
        }
    }
    printf("%lld\n",sum);
    return  0;
}
