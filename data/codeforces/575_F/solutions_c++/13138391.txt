#include<cstdio>
#include<cstring>

using namespace std;
typedef long long LL;

LL ans;

int main(){
    int l,r,n,a,b;
    scanf("%d%d",&n,&l);r=l;
    while(n--){
        scanf("%d%d",&a,&b);
        if(l > b){
            ans += l-b;
            r = l;
            l = b;
        }else if(r < a){
            ans += a-r;
            l = r;
            r = a;
        }else{
            if(l < a) l = a;
            if(r > b) r = b;
        }
    }
    printf("%I64d\n",ans);
    return 0;
}
