#include<bits/stdc++.h>
using namespace std;
#define FI "bulbo.i1"
#define FO "bulbo.o1"

int N, x;
long long ans;

int main(){
    ios_base::sync_with_stdio(0), cin.tie(0);
    #ifndef ONLINE_JUDGE
    ifstream cin(FI);
    ofstream cout(FO);
    #endif // ONLINE_JUDGE
    cin >> N >> x;
    int l = x, r = x;
    ans = 0;
    for(int a, b, i = 0; i < N; i++){
        cin >> a >> b;
        if(a <= l && r <= b) continue;
        if(l > b){
            ans += l - b;
            r = l;
            l = b;
        }
        else if(r < a){
            ans += a - r;
            l = r;
            r = a;
        }else{
            if( l < a ) l = a;
            if(r > b ) r = b;
        }
    }
    cout << ans;
    return 0;
}
