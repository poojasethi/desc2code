#include <stdio.h>
#include <stdlib.h>
#include <math.h>
long long x, n, d1, d2, now;
int main()
{
    scanf("%lld %lld", &n, &x);
    d1 = x;
    d2 = x;
    long long l, r, ans = 0l;
    for (int i=1;i<=n;i++)
    {
        scanf("%lld %lld", &l, &r);
        if (l <= d1 && r >= d2)
            continue;
        if (r < d1)
        {
            ans += (d1 - r);
            d2 = d1;
            d1 = r;
        }
        else if (l > d2)
        {
            ans += (l - d2);
            d1 = d2;
            d2 = l;
        }
        else
        {
            d1 = (d1 < l)?l:d1;
            d2 = (d2 > r)?r:d2;
        }
#ifdef TEST
        printf("d1:%lld d2:%lld\n", d1, d2);
#endif
    }

    printf("%lld\n", ans);
    return 0;
}
