#include  <bits/stdc++.h>
#define  i64  long long
#define  PB push_back
#define  MP make_pair
using  namespace  std;

long long  x,l,r,ql,qr,ans;
int  n;

int  main(){
	scanf("%d%I64d",&n,&x);
	l = r = x;
	for  (int  i = 1;i <= n; i++){
		scanf("%I64d%I64d",&ql,&qr);
		if  (qr < l)  ans += (l-qr),r = l,l = qr;
		else  if  (ql > r)  ans += (ql-r),l = r,r = ql;
		else  l = max(ql,l),r = min(qr,r);
	}
	printf("%I64d\n",ans);
    return 0;
}
