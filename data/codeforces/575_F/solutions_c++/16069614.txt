#include <iostream>
#include <algorithm>

using namespace std;

const int maxn = 5010;
long long l[maxn], r[maxn];

int main() {
  int n, x;
  cin >> n >> x;
  for (int i = 1; i <= n; i ++) {
    cin >> l[i] >> r[i];
  }
  long long ans = 0;
  long long curl = x, curr = x;
  for (int i = 1; i <= n; i ++) {
    if (r[i] < curl) {
      curr = curl;
      curl = r[i];
      ans += curr - curl;
    } else if (l[i] > curr) {
      curl = curr;
      curr = l[i];
      ans += curr - curl;
    } else {
      curl = max(curl, l[i]);
      curr = min(curr, r[i]);
    }
  }
  cout << ans << endl;
  return 0;
}
