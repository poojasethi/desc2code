#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,t[20010],i,j,k,ans=INT_MIN,sum;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>t[i];
	for(i=1;i<=n/3;i++)
		if(n%i==0)
			for(j=0;j<i;j++)
			{
				sum=0;
				for(k=j;k<n;k+=i)
					sum+=t[k];
				ans=max(ans,sum);
			}
	cout<<ans<<endl;
	return 0;
}
