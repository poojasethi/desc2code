#include <cstdio>

int a[20005],n;

int main(){
	scanf("%d",&n);
	for(int i=0;i<n;++i)
		scanf("%d",&a[i]);
	int ans=-200000001;
	for(int i=3;i<=n;++i)
		if(n%i==0){
			for(int j=0;j<n/i;++j){
				int now=0,k;
				for(k=j;k<n;k=k+(n/i))now+=a[k];
				k=k%n;
				for(;k<j;k=k+(n/i))now+=a[k];
				if(now>ans)ans=now;
			}
		}
	printf("%d\n",ans);
}
