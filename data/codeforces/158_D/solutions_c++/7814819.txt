#include <iostream>
using namespace std;

int main()
{
	int arr[20001],n,i,j,k,sum=0,s1;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
		sum+=arr[i];
	}
	for(j=2;n/j >=3;j++)
	{
		if(! (n%j))
		for(k=0;k<j;k++)
		{
			s1=0;
			for(int h=k;h<n;h+=j)
				s1+=arr[h];
			if(s1>sum)
				sum=s1;
		}
	}
	cout<<sum<<endl;
}
