#include<cstdio>
int a[200005];
int main()
{
	int ans=-100000000,n; scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	for (int i=1;i<=n/3;i++) if (n%i==0){
		for (int j=1;j<=i;j++){
			int sum=0;
			for (int k=j;k<=n;k+=i)
				sum+=a[k];
			if (sum>ans) ans=sum;
			}
		}
	printf("%d\n",ans);	return 0;
}
