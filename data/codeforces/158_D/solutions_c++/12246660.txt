#include <iostream>
#include <vector>

using namespace std;

int main(){
	int n;
	cin>>n;
	vector<int> v(n);
	for(int i=0;i<n;i++)cin>>v[i];
	int res=-1000000000;
	for(int k=3;k<=n;k++)if(n%k==0){
		int d=n/k;
		for(int r=0;r<d;r++){
			int sum=0;
			for(int i=r;i<n;i+=d)sum+=v[i];
			res=max(res,sum);
		}
	}
	cout<<res<<endl;

	return 0;
}
