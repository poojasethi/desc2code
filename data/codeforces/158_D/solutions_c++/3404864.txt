#include<stdio.h>
#include<stdlib.h>
#include<iostream>
using namespace std;
int main(){int N,t[20000],sum=-0x3fffffff;cin>>N;for(int i=0;i<N;i++){cin>>t[i];}for(int i=1;i*3<=N;i++){if(N%i) continue;for(int j=0;j<i;j++){int tsum=0;for(int k=0;k<N/i;k++) tsum+=t[j+k*i];if(tsum>sum) sum=tsum;}}cout<<sum;return 0;}
