#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int main(){
int n;
scanf("%d",&n);
int a[n],sum=0;
for(int i=0;i<n;i++){
scanf("%d",&a[i]);
sum+=a[i];
}
for(int i=2;i<=n/3;i++){
if(n%i!=0)continue;
for(int j=0;j!=i;j++){
int sum1=0;
for(int k=j;k<n;k+=i){
sum1+=a[k];
}
if(sum1>sum)sum=sum1;
}
}
printf("%d\n",sum);
return 0;
}
