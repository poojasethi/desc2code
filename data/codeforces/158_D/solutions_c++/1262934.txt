#include <cstdio>

int a[20200];

int main() {
	//freopen("input.txt","rt",stdin);
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++) scanf("%d",a+i);
	int res=-1000000000;
	for(int i=1;i<=n/3;i++) {
		if (n%i) continue;
		for(int j=0;j<i;j++) {
			int s=0;
			for(int k=j;k<n;k+=i) s+=a[k];
			if (s>res) res=s;
		}
	}
	printf("%d\n",res);
	return 0;
}
