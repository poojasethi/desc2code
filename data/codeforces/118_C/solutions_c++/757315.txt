#include<iostream>
#include<algorithm>
#define inf (1<<31)-1
using namespace std;
int main(){
    int i,j,n,k,min,x,sum;
    min = inf;
    string s,t,res;
    cin >> n >> k >> s;
    for(i='0';i<='9';i++){
        t=s; x=k; sum=0;
        for(int diff =0;diff<10;diff++){
            for(j=0;j<t.length()&&x;j++){
                if(t[j]-i==diff){
                    if(diff)
                        x--;
                    t[j]=i;
                    sum+=diff;
                }
            }
            for(j=t.length();j>=0&&x;j--){
                if(i-t[j]==diff){
                    x--; t[j]=i;
                    sum+=diff;
                }
            }
            if(!x)
                break;
        }
        if((sum<min || (sum==min && t<res))){
            min=sum;
            res=t;
        }
    }
    cout << min << endl << res << endl;
    return 0;
}
