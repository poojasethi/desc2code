#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int main()
{
  ios_base::sync_with_stdio(0);
  string a;
  string b;
  string opt;
  int cost = 100000;
  int n, kaa;
  cin >> n >> kaa;
  cin >> a;
  for(int i = 0; i <= 9; i++)
  {
    int cost_cur = 0;
    int ile_i = 0;
    for(int j = 0; j < n; j++)
      if(a[j] == ('0' + i)) ile_i++;
    b = a;
    for(int j = 1; j <= 9; j++)
    {
      char up = '0' + i + j;
      if(up <= '9')
      for(int k = 0; k < n; k++) if(b[k] == up && ile_i < kaa)
      {
    b[k] = '0' + i;
    ile_i++;
    cost_cur += j;
      }
      char down = '0' + i - j;
      if(down >= '0')
      for  (int k = n - 1; k >= 0; k--) if(b[k] == down && ile_i < kaa)
      {
    b[k] = '0' + i;
    ile_i++;
    cost_cur += j;
      }
    }
    if(cost_cur < cost || (cost_cur == cost && b < opt) )
    {
      opt = b;
      cost = cost_cur;
    }
  }
  cout << cost << "\n";
  cout << opt;
  return 0;
}
