#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstdlib>
#include<queue>
using namespace std;
typedef pair<int,int> PII;
typedef pair<int,string> PIS;
char arr[10005];
int N;

PIS tryN(int num,int K)
{
    string ret = arr;
    priority_queue<PII,vector<PII>,greater<PII> > PQ;
    for(int i = 0; i < N; ++i)
        PQ.push(PII( abs(num-(arr[i]-'0')), (arr[i]-'0'>=num ? -1 : 1)*(N-i) ));
    int cost = 0;
    for(int i = 0; i < K; ++i)
    {
        PII c = PQ.top(); PQ.pop();
        ret[N-abs(c.second)] = num+'0';
        cost += c.first;
    }
    return PIS( cost , ret );
}

int main()
{
    int K;
    cin>>N>>K>>arr;
    PIS arr[10];
    for(int i = 0; i <= 9; ++i)
    {
        arr[i] = tryN(i,K);
    }
    sort(arr,arr+10);
    cout<<arr[0].first<<endl;
    cout<<arr[0].second<<endl;

}
