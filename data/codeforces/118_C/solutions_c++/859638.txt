#include <iostream>
#include <vector>
#include <algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)
#define ALL(A) A.begin(), A.end()

using namespace std;

const int INF = 1 << 30;

typedef pair<int, string> P;


int main()
{
	int n, k;
	cin >> n >> k;
	string s,x;
	cin >> s;
	P res (INF, "ANSWER" );

	rep (i, 10 ){
		x = s;
		int sum = 0;
		int cnt = (int)count (ALL(x), i + '0' );
		for (int j = 1; j < 10; j++ ){
			if (i + j < 10 ){
				for (int m = 0; m < n && cnt < k; m++ ){
					if (x[m] == i + j + '0' ){
						cnt++;
						sum += j;
						x[m] = i + '0';
					} // end if
				} // end for
			} // end if
			if (i - j >= 0 ){
				for (int m = n - 1; m >= 0 && cnt < k; m-- ){
					if (x[m] == i - j + '0' ){
						cnt++;
						sum += j; 				
						x[m] = i + '0';
					} // end if
				} // end for
			} // end if
		} // end for

		res = min (res, P (sum, x ) );
	} // end rep

	cout << res.first << endl;
	cout << res.second << endl;

	return 0;
}
