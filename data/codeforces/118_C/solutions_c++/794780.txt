#include <iostream>
#include <cmath>
#include <vector>

using namespace std;

int main() {
  int n, k;
  cin >> n >> k;

  string s;
  cin >> s;

  int got, mincost, cost;
  char minlet;
  string cur, best;

  mincost = n * 10;
  for (char i = '0'; i <= '9'; i++) {
    got = 0;
    cost = 0;
    cur = s;

    for (int d = 0; d <= 9; d++) {
      for (int j = 0; j < n; j++) {
        if (got < k && s[j] - i == d) {
          got++;
          cost += d;
          cur[j] = i;
        }
      }

      if (d != 0) {
        for (int j = n - 1; j >= 0; j--) {
          if (got < k && i - s[j] == d) {
            got++;
            cost += d;
            cur[j] = i;
          }
        }
      }
    }

    if (cost < mincost) {
      mincost = cost;
      best = cur;
    } else if (cost == mincost && cur < best) {
      best = cur;
    }
  }

  cout << mincost << endl;
  cout << best << endl;
}
