#include <bits/stdc++.h>
using namespace std;

struct trip{
	int ab, pval, weight;
	trip() {}
	trip(int x, int y, int z) {
		ab = x;
		pval = y;
		weight = z;
	}
};

bool operator<(trip a, trip b) {
	if (a.ab != b.ab)
		return a.ab < b.ab;
	if (a.pval != b.pval) 
		return a.pval > b.pval;
	return a.weight < b.weight;
}

int main() {
	ios_base::sync_with_stdio(false);
	int n, k;
	string s;
	pair<int, string> ans[10];
	cin >> n >> k >> s;

	for(char i = '0'; i <= '9'; i++) {
		trip cand[n];
		ans[i-'0'].second = s;
		ans[i-'0'].first = 0;
		for(int j = 0; j < n; j++) {
			if (s[j] >= i) 
				cand[j] = trip(abs(s[j]-i), s[j], j);
			else
				cand[j] = trip(abs(s[j]-i), s[j], -j);
		}
		sort(cand, cand+n);
		for(int j = 0; j < k; j++) {
			ans[i-'0'].second[abs(cand[j].weight)] = i;
			ans[i-'0'].first += cand[j].ab;
		}
	}
	sort(ans, ans+10);
	// for(int i = 0; i < 10; i++) 
	// 	cout << ans[i].first << endl;
	
	cout << ans[0].first << endl << ans[0].second << endl;
}
