#include<iostream>
#include<cstring>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
  int n,m,hs[10],mi=111111111;
  vector<string> in;
  memset(hs,0,sizeof(hs));
  string s,ans;
  cin>>n>>m>>s;
  for(int i=0;i<s.size();i++)
    hs[s[i]-'0']++;
  for(int i=0;i<10;i++){
    int cnt=0,cost=0,now=hs[i];
    string go=s;
    for(int k=1;k<10&&now<m;k++){
      for(int j=0;j<s.size()&&now<m;j++)
    if((s[j]-'0'-i)==k)now++,s[j]='0'+i,cost+=k;
      for(int j=s.size()-1;j>=0&&now<m;j--)
    if((i-s[j]+'0')==k)now++,s[j]='0'+i,cost+=k;
    }
    
    if(cost<mi&&(now>=m)){
      in.clear();
      in.push_back(s);
      mi=cost,ans=s;
    }else if(cost==mi&&(now>=m)){
      in.push_back(s);
      mi=cost,ans=s;
    }
    s=go;
  }
  sort(in.begin(),in.end());
  //cout<<in[0]<<endl;
  cout<<mi<<endl;
  cout<<in[0]<<endl;
  return 0;
}
