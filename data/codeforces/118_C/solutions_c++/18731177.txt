#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <map>
#include <math.h>
#include <limits.h>
#define inf 0x3f3f3f;
using namespace std;
int arr[10];

int main()
{
	int n,k;cin>>n>>k;
	string s,ans;cin>>s;
	for(int i = 0; i < n; i++){
        ++arr[s[i]-'0'];
	}

	int mn=INT_MAX;
	for(char c='0';c<='9';++c){

		char a=c-1,b=c+1;
		int cnt=k-arr[c-'0'],cost=0;
		string tmp=s;
		while((a>='0' || b<='9') && cnt>0){

			if(b<='9')for(int i=0;i<n && cnt>0;++i){
				if(s[i]==b)tmp[i]=c,--cnt,cost+=abs(b-c);
			}

			if(a>='0')for(int i=n-1;i>=0 && cnt>0;--i){
				if(s[i]==a)tmp[i]=c,--cnt,cost+=abs(a-c);
			}
			++b;
			--a;
		}
		if(cost<mn)mn=cost,ans=tmp;
		else if(cost==mn && tmp<ans)ans=tmp;
	}
	cout<<mn<<"\n"<<ans<<"\n";
	return 0;
}
