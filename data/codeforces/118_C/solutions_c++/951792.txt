#include <iostream>
#include <algorithm>

using namespace std;

const int INF = 1 << 30;

int main(int argc, char *argv[])
{
	int n, k;
	cin >> n >> k;
	
	string s;
	cin >> s;
	
	int m = INF;
	string h;
	
	for (int i = 0; i <= 9; ++i)
	{
		string p = s;		
		int r = 0;
		int l = count(p.begin(), p.end(), '0' + i);
		
		for (int d = 1; d <= 9; ++d)
		{
			if (i + d <= 9)
			{
				for (int j = 0; (j < n) && (l < k); ++j)
				{
					if (p[j] == '0' + (i + d))
					{
						p[j] = '0' + i;
						r += d;
						l++;
					}
				}
			}
			
			if (i - d >= 0)
			{
				for (int j = n; (j >= 0) && (l < k); --j)
				{
					if (p[j] == '0' + (i - d))
					{
						p[j] = '0' + i;
						r += d;
						l++;
					}
				}
			}
		}
				
		if (r < m)
		{
			m = r;
			h = p;
		}
		else if ((r == m) && (p < h))
			h = p;			
	}
	
	cout << m << endl;
	cout << h << endl;
			
	return 0;
}
