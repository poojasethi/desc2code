#include <cstdlib>
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
	int n, k;
	string s;
	cin >> n >> k >> s;
	int min_cost = 1234567890;
	string res = s;
	for (int t = 0; t <= 9; ++t) {
		int cnt = 0;
		int cost = 0;
		string rs = s;
		for (int d = 0; d <= 9; ++d) {
			int r0 = t + d;
			if (r0 <= 9) {
				for (int i = 0; i < s.size(); ++i) {
					if (cnt < k && s[i] == r0 + '0') {
						rs[i] = t + '0';
						cost += d;
						cnt += 1;
					}
				}
			}
			int r1 = t - d;
			if (r1 >= 0 && d) {
				for (int i = s.size() - 1; i >= 0; --i) {
					if (cnt < k && s[i] == r1 + '0') {
						rs[i] = t + '0';
						cost += d;
						cnt += 1;
					}
				}
			}
		}
		if (cost < min_cost || cost == min_cost && rs < res) {
			min_cost = cost;
			res = rs;
		}
	}
	cout << min_cost << endl;
	cout << res << endl;
	return 0;
}
