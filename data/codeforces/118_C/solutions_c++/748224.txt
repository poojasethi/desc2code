#include<stdio.h>
#include<stdlib.h>
#include<cstring>
#include<queue>
#define pq priority_queue
using namespace std;
int n,k,maxx=2000000000;
char a[10003],ANS[10003],an[10003],b[10003];
int main()
{
    scanf("%d%d",&n,&k);
    scanf("%s",a);
    for(char s='0';;s++)
    {
        int sum=0;
        pq<int, vector<int>, greater<int> > ans;
        for(int i=0;i<n;i++)
        {
            if(s>a[i])
            ans.push((abs(s-a[i])*100000) + 99999 - i);
            else
            ans.push((abs(s-a[i])*100000) + i);
        }
        strcpy(b,a);
        for(int i=0;i<k;i++)
        {
            sum += ans.top()/100000;
            if(ans.top()%100000>20000)
               b[99999-ans.top()%100000] = s;
            else
               b[ans.top()%100000] = s;
            ans.pop();
        }
        if(sum < maxx || sum==maxx && strcmp(an,b)>0)
        {
            maxx=sum;
            strcpy(an,b);
        }
        if(s=='9')break;
    }
    printf("%d\n",maxx);
    printf("%s\n",an);
}
