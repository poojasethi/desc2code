#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int inf=0x3fffffff;
char str[111111],ding[111111],las[111111];
int vis[111];
int main()
{
	int n,m,i,j,len,ans,k,res,rs;
	memset(vis,0,sizeof(vis));
	ans=inf;
	scanf("%d%d",&n,&m);
	scanf("%s",ding);
	for(i=0;i<n;i++)
	vis[ding[i]-'0']++;
	for(i=0;i<10;i++)
	{
		strcpy(str,ding);
		res=0,rs=vis[i];
		if(vis[i]>=m)
		{
			ans=0,strcpy(las,ding);
		}
		else if(vis[i]<m)
		{
			for(k=1;k<10;k++)
			{
				if(i+k<=9)
				{
					for(j=0;j<n;j++)
					{
						if(rs==m) break;
						if(str[j]==i+k+'0') rs++,res+=k,str[j]=i+'0';
					}
				}
				if(i-k>=0)
				{
					for(j=n-1;j>=0;j--)
					{
						if(rs==m) break;
						if(str[j]==i-k+'0') rs++,res+=k,str[j]=i+'0';
					}
				}
				if(rs==m) break;
			}
			if(rs!=m) continue;
			if(res<ans||(res==ans&&strcmp(las,str)>0)) ans=res,strcpy(las,str);
		}
	}
	printf("%d\n%s\n",ans,las);
}
