#include <bits/stdc++.h>
using namespace std;


int main(){
    int n,k;
    string SS;
    cin>>n>>k>>SS;


    string str[10]; int cost[10];
    for(int d=0;d<=9;d++){


        int rem=k,tot=0;
        string S=SS;
        for(int i=0;i<n && rem;i++)if(S[i]-'0' == d)rem--;
        for(int delta=1;delta<=9 && rem;delta++){
            for(int i=0;i<n && rem;i++)if( (S[i]-'0')-delta==d )tot+=delta,rem--,S[i]=d+'0';
            for(int i=n-1;i>=0 && rem;i--)if( (S[i]-'0')+delta==d )tot+=delta,rem--,S[i]=d+'0';

        }
        str[d]=S;
        cost[d]=tot;
    }

    int mn=1e5; for(int d=0;d<=9;d++)mn=min(mn,cost[d]);
    cout<<mn<<"\n";

    set<string>res;
    for(int d=0;d<=9;d++)if(cost[d]==mn)res.insert(str[d]);
    cout<<*(res.begin())<<endl;
}
