#include <bits/stdc++.h>

using namespace std;

const int inf = 1e9 + 9;
const int N = 1e5 + 9;

int n, k, hash[N], res, mn;

string str, answer, ans;

int main() {

	cin >> n >> k;

	cin >> str;

	for (int i = 0; i < str.size(); i++)
		hash[str[i] - '0']++;

	mn = inf;

	for (int i = 0; i < 10; i++)
		if (hash[i] >= k) {
			cout << 0 << endl << str << endl;
			return 0;
		}

	for (int i = 0; i < 10; i++) {
		int pa = i;
		int temp = k;
		temp -= hash[pa];
		ans = str; res = 0;
		for (int j = 1; j < 10 and temp > 0; j++) {
			if (pa + j < 10) {
				for (int l = 0; temp and l < n; l++) {
					if (str[l] == (pa + j) + '0') {
						temp--;
						res += j;
						ans[l] = pa + '0';
					}
				}
			}
			if (pa - j >= 0) {
				for (int l = n-1; temp and l >= 0; l--) {
					if (str[l] == (pa - j) + '0') {
						temp--;
						res += j;
						ans[l] = pa + '0';
					}
				}
			}
		}
		if (res < mn) {
			answer = ans;
			mn = res;
		}
		else if (res == mn)
			if (answer > ans)
				answer = ans;
	}

	cout << mn << endl << answer << endl;
	

	return 0;
}
