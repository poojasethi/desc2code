#include<iostream>
using namespace std;
int main()
{
int n,m,i,j,l,k,t,p;
string s,r,z;
cin>>n>>m>>s;
t=n*10;
for(l=0;l<s.length();l++)
{
s[l]-='0';
}
for(i=0;i<=9;i++)
{
r=s;
k=p=0;
for(j=0;j<=9;j++)
{
for(l=0;l<s.length();l++)
{
if(s[l]-j==i) k+=j,p++,r[l]=i;
if(p==m)
{
if(t>k || t==k && z>r) t=k,z=r;
break;
}
}
if(j>0 && p<m)
{
for(l=s.length()-1;l>=0;l--)
{
if(s[l]+j==i) k+=j,p++,r[l]=i;
if(p==m)
{
if(t>k || t==k && z>r) t=k,z=r;
break;
}
}
}
}
}
for(i=0;i<z.length();i++) z[i]+='0';
cout<<t<<endl<<z;
return 0;
}
