#include <iostream>
#include <vector>
#include <string>
#include <map>
#include <set>
#include <algorithm>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <queue>

using namespace std;

string s;

bool orden(pair <int, int> p1, pair <int, int> p2)
{
	int n1 = s[p1.first] - p1.second;
	int n2 = s[p2.first] - p2.second;
	
	if(abs(n1) != abs(n2)) return abs(n1) < abs(n2);
	
	if(n1 != n2) return n1 > n2;
	
	if(n1 < 0) return p1.first > p2.first;
	else return p1.first < p2.first;
}

int main ()
{
	int n, k;
	while(cin>>n>>k)
	{
		cin>>s;
		
		int minC = 1<<30;
		string minS = "";
		
		for(char d='0'; d<='9'; d++)
		{
			vector < pair <int, int> > v;
			
			for(int i=0; i<s.size(); i++)
				v.push_back(make_pair(i, d));
			
			sort(v.begin(), v.end(), orden);
			
			string x = s;
			int cost = 0;
			
			for(int i=0; i<k; i++)
			{
				cost += abs(s[v[i].first] - d);
				x[v[i].first] = d;
			}
			
			if(cost < minC || (cost == minC && x < minS))
			{
				minC = cost;
				minS = x;
			}
		}
		cout<<minC<<endl;
		cout<<minS<<endl;
	}
	
	return 0;
}
