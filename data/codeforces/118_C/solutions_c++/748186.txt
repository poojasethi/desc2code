#include<iostream>
#include<cstdlib>
#include<string>
#include<algorithm>
using namespace std;
struct node
{   int p,v;
}a[10001];
int n,k,ans;
char ch;
string s,s1,anss;
bool cmp(node t1,node t2)
{
    if(t1.v!=t2.v) return t1.v<t2.v;
    if(s[t1.p]==s[t2.p])
    {   if(s[t1.p]>=ch) return t1.p<t2.p;
        else return t1.p>t2.p;
    }
    return s[t1.p]>s[t2.p];
}
int main()
{   ans=-1,anss.clear();
    cin>>n>>k;
    cin>>s;
    for(int i=0;i<10;i++)
    {   ch=i+'0';
        s1=s;
        int temp=0;
        for(int j=0;j<n;j++)
            a[j].p=j,a[j].v=abs(s[j]-ch);
        sort(a,a+n,cmp);
        for(int j=0;j<k;j++)
            temp+=a[j].v,s1[a[j].p]=ch;
        if(ans==-1 || ans>temp) ans=temp,anss=s1;
        else if(ans==temp && anss>s1) anss=s1;
    }
    cout<<ans<<endl<<anss<<endl;
}
