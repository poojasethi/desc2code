using namespace std;
#define REP(i,n) for((i)=0;(i)<(int)(n);(i)++)
#define foreach(c,itr) for(__typeof((c).begin()) itr=(c).begin();itr!=(c).end();itr++)
typedef long long ll;
#define INF (1<<29)

int main(){
	int n, k, i, j, c, m=200000;
	string t, s, ans=":";
	vector<pair<int, int> > v;
	cin >> n >> k >> s;
	for(i='0';i<='9';i++){
		v.clear();
		REP(j,n){
			if(s[j]>i)v.push_back(make_pair(s[j]-i,j));
			else v.push_back(make_pair(i-s[j],2*n-j));
		}
		sort(v.begin(),v.end());
		t=s;
		c=0;
		REP(j,k){
			c+=v[j].first;
			if(v[j].second<n)t[v[j].second]=i;
			else t[2*n-v[j].second]=i;
		}
		if(m>c||(m==c&&t<ans)){
			m=c;
			ans=t;
		}
	}
	cout << m << endl << ans << endl; 
	return 0;
}
