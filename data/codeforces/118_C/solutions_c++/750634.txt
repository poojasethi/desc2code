#include <stdio.h>
#include <string>
#include <string.h>
#include <vector>
#include <queue>
#include <set>
#include <stack>
#include <algorithm>
#define FOR(A, B) for(A=0;A<B;A++)
#define MAXN 10001
using namespace std;

char str[MAXN];
char str2[MAXN];
char strans[MAXN];
int counts[10];

int x;

int main()
{
	int i, j, k, l, n, m, y;
	scanf("%d %d", &n, &k);
	scanf("%s", str);
	FOR(i, 10) counts[i] = 0;
	FOR(i, n) counts[str[i]-'0']++;
	int ans = -1;
	strans[n] = '\0';
	FOR(i, 10) {
		int cost = 0;
		int done = counts[i];
		strcpy(str2, str);
		for(j = 1; j <= 9 && done < k; j++) {
			FOR(l, n) if(str2[l]-'0'-j == i) {
				str2[l] = i+'0';
				cost += j;
				done++;
				if(done >= k) break;
			}
			for(l = n-1; l >= 0 && done < k; l--) {
				if(str2[l]-'0'+j == i) {
					str2[l] = i+'0';
					cost += j;
					done++;
				}
			}	
		}

		if(cost < ans || ans == -1) {
			strcpy(strans, str2);
			ans = cost;
		}
		if(cost == ans && strcmp(str2, strans) < 0) {
			ans = cost;
			strcpy(strans, str2);
		}
	}
	printf("%d\n%s", ans, strans);
	return 0;
}
