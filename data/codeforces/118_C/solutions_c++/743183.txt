#include <cstdio>
#include <string>
#include <iostream>
#include <algorithm>
using namespace std;
pair<int, string> get(string str, char c, int n, int k) {
  pair<int, string> ret;
  for (int dif = 0; dif <= 9 && k > 0; dif++) {
    for (int i = 0; i < n && k > 0; i++) {
      if (str[i] == c + dif) {
        str[i] = c;
        ret.first += dif;
        k--;
      }
    }
    if (dif != 0) {
      for (int i = n - 1; i >= 0 && k > 0; i--) {
        if (str[i] == c - dif) {
          str[i] = c;
          ret.first += dif;
          k--;
        }
      }
    }
  }
  ret.second = str;
  return ret;
}
int main() {
  int n, k;
  char i;
  string str;
  pair<int, string> results[10];
  scanf("%d %d\n", &n, &k);
  cin >> str;
  for (i = '0'; i <= '9'; i++) {
    results[i - '0'] = get(str, i, n, k);
  }
  sort(results, results + 10);
  printf("%d\n%s\n", results[0].first, results[0].second.c_str());
  return 0;
}
