#include <cstdio>
#include <cstring>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int n, m;
char buf[10001];

inline int sgn(int n) {
	return n < 0 ? -1 : n == 0 ? 0 : 1;
}

int gao(string& s, int digit) {
	vector <pair <int, pair <int, int> > > v;
	for (int i = 0; i < n; ++i) {
		v.push_back(make_pair(abs(s[i] - '0' - digit), make_pair(-s[i], i * (s[i] < digit + '0' ? -1 : 1))));
	}
	sort(v.begin(), v.end());
	int ret = 0;
	for (int i = 0; i < m; ++i) {
		ret += v[i].first;
		s[abs(v[i].second.second)] = digit + '0';
	}
	return ret;
}

int main() {
	scanf("%d%d", &n, &m);
	scanf("%s", buf);
	string res = "";
	int cost = 0x3f3f3f3f;
	for (int i = 0; i <= 9; ++i) {
		string s = string(buf);
		int cur = gao(s, i);
		if (cur < cost || (cur == cost && s < res)) {
			res = s;
			cost = cur;
		}
	}
	printf("%d\n", cost);
	printf("%s\n", res.c_str());
	return 0;
}
