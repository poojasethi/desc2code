#include <iostream>
#include <stdio.h>
using namespace std;

int N,M;

int main()
{
  int k,t,p;
  string s,r,z;
  cin >> N >> M >> s;
  t=N*10;
  for(int i=0 ; i<N  ; i++)     s[i]-='0';
  for(int i=0 ; i<10 ; i++)
  {
    r=s;
    k=p=0;
    for(int j=0 ; j<=10 ; j++)
    {
      for(int l=0 ; l<N ; l++)
      {
        if(s[l]-j==i) { k+=j;  p++;  r[l]=i; }
        if(p==M)
        {
          if(t>k || t==k && z>r) { t=k;  z=r; }
          break;
        }
      }
      if(j>0 && p<M)
        for(int l=N-1 ; l>=0 ; l--)
        {
          if(s[l]+j==i) { k+=j;  p++;  r[l]=i; }
          if(p==M)
          {
            if(t>k || t==k && z>r) { t=k;  z=r; }
            break;
          }
        }
    }
  }
  for(int i=0 ; i<N ; i++) z[i]+='0';
  cout << t << endl << z;
  return 0;
}
