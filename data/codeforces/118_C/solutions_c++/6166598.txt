#include <cstdio>
#include <iostream>
#include <algorithm>
#include <string>
#include <cstring>
#include <cmath>
#define INF 0x3f3f3f3f
#define MAXN 10005
using namespace std;

int main(){
	int c[10], n, k;
	while(cin >> n >> k){
		string s, ans="";
		cin >> s;
		memset(c, 0, sizeof(c));
		for(int i=0; i<n; i++)
			c[s[i]-'0']++;	
		int m=INF;
		for(int num=0; num<10; num++){
			string t=s;
			int sum=0, tmp=k, last;
			tmp-=c[num];
			if(tmp < 0){
				m = 0, ans = s;
				break;
			}
			for(int d=1; (num-d>=0 || num+d<10) && tmp>0; d++){
				if(num+d<10){					//先减小
					for(int i=0; i<n && tmp>0; i++)
						if(t[i]-'0' == num+d)
							t[i] = num+'0', tmp--, sum+=d;
				}
				if(num-d>=0){
					for(int i=n-1; i>=0 && tmp>0; i--)
						if(t[i]-'0' == num-d)
							t[i] = num+'0', tmp--, sum+=d;
				}
			}
			if(sum < m){
				m = sum;
				ans = t;
			}else if(sum == m && ans > t)
				ans = t;
			//printf("%d. %d %d %d\n", num, sum, m, ans);
		}
		cout << m << endl << ans << endl;
	}

}
