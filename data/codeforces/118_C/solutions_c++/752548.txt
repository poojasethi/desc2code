#include<stdio.h>
#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main(){
	string a,go;
	int n,m,gn,i,j,k,co;
	pair<int,string> ans=make_pair(1000000,"");
	cin>>n>>m>>a;
	for(i=0;i<10;i++){
		go=a;
		gn=0;
		co=count(a.begin(),a.end(),i+'0');
		for(j=1;j<10;j++){
			if(i+j<10){
				for(k=0;k<n;k++){
					if(co>=m) break;
					if(i+'0'==go[k]-j){
						go[k]=i+'0';
						gn+=j;
						co++;
					}
				}
			}	
			if(i-j>=0){
				for(k=n-1;k>=0;k--){
					if(co>=m) break;
					if(i+'0'==go[k]+j){
						go[k]=i+'0';
						gn+=j;
						co++;
					}
				}
			}
		}
		//cout<<go<<endl;
		ans=min(ans,make_pair(gn,go));
	}
	cout<<ans.first<<endl<<ans.second<<endl;
}
