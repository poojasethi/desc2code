void prov (int z)
{
 int p=k,i,j;
 for (i=0;i<n;i++)
  if (c[i]-'0'==z) p--;
 if (p<=0) return;
// cout<< c<<endl;
 for (i=1;i<10;i++)
 {
  for (j=0;j<n;j++)
   if (z+i<10 && c[j]-'0'==z+i) {tp+=i;c[j]-=i;p--;if (p==0) return;}
  for (j=n-1;j>=0;j--)
   if (z-i>=0 && c[j]-'0'==z-i) {tp+=i;c[j]+=i;p--;if (p==0) return;}

 }

}

int main() {

t=n*10;
for (i=0;i<10;i++)
 { tp=0;c=st;
   prov(i);
   if (tp<t) {t=tp;ss=c;} else
    if (tp==t && ss>c) ss=c;
 }
cout<<t<<endl;
