#include <iostream>
#include <algorithm>
using namespace std;


int main()
{
   int n,k,an=1000000;
   string s,t,ans="fghjjknf";

   cin>>n>>k>>s;

   for(int i=0; i<10; i++)
   {
       t=s;
       int ko=count(s.begin(), s.end(),i+'0'),now=0;
       for (int d=1; d<10; d++)
       {
 if(i+d<10)
           for(int j=0; j<t.size()&&ko<k; j++)
            if(t[j]==i+d+'0')
               {
                   ko++;
                   now+=d;
                   t[j]=i+'0';
               }
if(i-d>=0)
            for(int j=t.size()-1;j>=0&&ko<k; j--)
            if(t[j]==i-d+'0')
               {
                   ko++;
                   t[j]=i+'0';
                   now+=d;
               }
       }
       if(now<an){an=now; ans=t;}
       if(now==an && ans>t) {an=now; ans=t;}
   }
   cout<<an<<endl<<ans;



    return 0;
}
