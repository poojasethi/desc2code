#include <iostream>
#include <string>
#include <utility>
#include <algorithm>

using namespace std;

#define F first
#define S second

typedef pair<int,int> ii;

ii a[ 10005 ];

bool cmp(ii A, ii B){
	if ( abs(A.S) != abs(B.S) ) return abs(A.S) < abs(B.S);
	if ( A.S<0 ^ B.S<0 ) return A.S < B.S;
	if ( A.S>0 && B.S>0 ) return A.F > B.F;
	return A.F < B.F;
}

int main(){
	int n, k;
	string s, t;
	int best = 2000000000;
	string ans;
	
	cin >> n >> k;
	cin >> s;
	
	for (int j=0; j<10; j++){
		for (int i=0; i<n; i++){
			a[ i ].F = i;
			a[ i ].S = j - (s[ i ] - '0');
		}
		sort( a, a+n, cmp );
		
		int cost = 0;
		for (int i=0; i<k; i++) cost += abs( a[i].S );
		if ( cost <= best ){
			t = s;
			for (int i=0; i<k; i++) t[ a[i].F ] = j + '0';
			if ( cost < best || (cost==best && t < ans) ){
				best = cost;
				ans = t;
				//cout << t << endl;
			}
		}
	}
	
	cout << best << endl;
	cout << ans << endl;

	return 0;
}
