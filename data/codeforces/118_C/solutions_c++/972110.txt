#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

int n,k;
string str;
int c[10];

void solve(int t,int& sum,string& ans)
{
	ans = str;
	sum = 0;
	int s=c[t];

	for(int i=1;s<k&&i<=9;i++)
	{
		if(t+i<=9)
			for(int j=0;s<k&&j<(int)ans.size();j++)
				if(ans[j]==t+i+'0') { ans[j] = t+'0'; s++; sum+=i;}
		if(t-i>=0)
			for(int j=(int)ans.size()-1;s<k&&j>=0;j--)
				if(ans[j]==t-i+'0') { ans[j] = t+'0'; s++; sum+=i; }
	}
}

int main()
{
	cin>>n>>k;
	cin>>str;
	for(int i=0;i<=9;i++)
		c[i] = count(str.begin(),str.end(),'0'+i);

	int minsum = 100000000;
	string minans;
	int sum;
	string ans;
	for(int i=0;i<=9;i++)
	{
		solve(i,sum,ans);
		if( sum < minsum || sum==minsum && ans < minans )
		{
			minsum = sum;
			minans = ans;
		}
	}

	cout << minsum << '\n' << minans << '\n';

	
	return 0;
}
