#include <iostream>
#include <cstring>
#include <string>
#include <algorithm>
#include <cstdio>

using namespace std;


int main()
{
	//freopen("data.in" , "r" ,stdin);
	int n , k;
	string s;
	cin >> n >> k;
	cin >> s;
	
	int res = 1000000;
	string ss;
	for(int p = 0 ; p <= 9 ; p++)
	{
		int m = 0 , cost = 0;
		string str = s;
		for(int i = 0 ; i < 10; i++)
		{
			if(m  == k)
			    break;
			    
			for(int j = 0 ; j < n; j++)
			{
				if(m == k)  break;
				int c = s[j] - '0';
				if(c - p == i)
				{	
					str[j] = char(p + '0');
					cost += i;
					m++;
				
				}
			}
			for(int j = n - 1 ; j >= 0 ; j--)
			{	
				if(m == k)   break;
				int c  = s[j] - '0';
				if(p - c == i && i != 0)
				{
					str[j] = char(p + '0');
					cost += i;
					m++;
				}
			}
		
		}
	
		if(res > cost)
		{
			res = cost;
			ss = str;
		}
		if(res == cost)
		{
			ss = min(ss ,str);
		}
	}
	cout << res << endl;
	cout << ss << endl;
//	while(1);
	return 0;
	
}
