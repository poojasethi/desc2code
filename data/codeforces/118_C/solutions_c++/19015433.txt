#include <bits/stdc++.h>
#define pii pair<int,int>
#define pll pair<ll,ll>
#define fr first
#define sc second
#define ll  long long
#define mp make_pair
#define mod 1000000007

using namespace std;

int main(){
    ios::sync_with_stdio(0);
    int n,k;
    cin>>n>>k;
    string s;
    cin>>s;
    set<pair<int,string> >st1;
    for(int j=0 ; j<10 ; j++){
        set<pair<pii,int> >st;
        set<pair<pii,int> >::iterator it;
        for(int i=0 ;i <n ; i++){
            if(j >= s[i] - '0')
                st.insert(mp(mp(abs(s[i] - '0' - j),2),-i));
            else
                st.insert(mp(mp(abs(s[i] - '0' - j),0),i));
        }
        ll r = 0;
        int k1 = k;
        string t = s;
        for(it = st.begin() ; it != st.end() ; it++){
            if(!k1)break;
            k1--;
            r += it->fr.fr;
            t[abs(it->sc)] = j+'0';
        }
        st1.insert(mp(r,t));
    }

    cout<<st1.begin()->fr<<endl;
    cout<<st1.begin()->sc<<endl;

    return 0;
}
