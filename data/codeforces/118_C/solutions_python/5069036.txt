n,k=map(int,raw_input().split())
s=map(int,raw_input())
e=list(enumerate(s))
c=9*n
a=[9]*n
for j in range(10):
    r=s[:]
    C=0
    for i,u in sorted(e,key=lambda (i,u): (abs(u-j), -u, i*(-1)**(j>u) ))[:k]:
        r[i]=j
        C+=abs(u-j)
    if C < c:
        c=C
        a=r
    if C== c:
        a=min(a,r)
print c
print ''.join(map(str,a))
