//#include "stdafx.h"
#include <iostream>
#include <algorithm>
#include <string>
#include <map>

using namespace std;
map<string,int> M;

int main(void)
{
	int n,m,v,w;
	long double p;
	cin>>n>>m>>p;
	for (int i=0;i<n;i++) {
		string s;
		cin>>s>>v;
		w=v*p+0.00000000001;
		if (w>=100)
			M[s]=w;
	}
	for (int i=0;i<m;i++) {
		string s;
		cin>>s;
		if (M[s]==0)
			M[s]=0;
	}
	cout<<M.size()<<endl;
	for (map<string,int>::iterator it=M.begin();it!=M.end();it++) {
		cout<<it->first<<" "<<it->second<<endl;
	}
	return 0;
}
