#include <cstdio>
#include <map>
#include <iostream>
using namespace std;
map<string, int> ma;
map<string, int> :: iterator it;
int n, m, v, k;
char s[111];
int main() {
    scanf("%d%d 0.%d", &n, &m, &k);
    while (n--) {
        scanf("%s%d", s, &v);
        if (v * k >= 10000) ma[s] = v * k / 100;
    }
    while (m--) scanf("%s", s), ma[s] = max(ma[s], 0);
    printf("%d\n", ma.size());
    for (it = ma.begin();it != ma.end();it++)
        printf("%s %d\n", it->first.c_str(), it->second);
}
