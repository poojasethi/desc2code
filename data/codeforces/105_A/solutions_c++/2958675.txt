#include<iostream>
#include<string>
#include<map>
using namespace std;
int main()
{
	int m,n;
	double k,eps=1e-7;
	while(cin>>m>>n>>k)
	{
		map<string,int>::iterator it;
		string s;
		int ex,i,count=0;
		map<string,int>ma;
		while(m--)
		{
			cin>>s>>ex;
			ex=int(ex*k+eps);
			if(ex>=100){ma[s]=ex;count++;}
		}
		while(n--)
		{
			cin>>s;
			it=ma.find(s);
			if(it==ma.end()){ma[s]=0;count++;}
		}
		cout<<count<<endl;
		for(it=ma.begin();it!=ma.end();it++)
		{
			cout<<it->first<<' '<<it->second<<endl;
		}
	}
	return 0;
}
