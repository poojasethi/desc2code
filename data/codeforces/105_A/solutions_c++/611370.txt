#include<iostream>
#include<string>
#include<map>

using namespace std;

int main()
{
	
	map<string,int> mx;

	string s;
	int n,m,k,i,val;
	double kk;

	cin>>n>>m>>kk;
	
	kk*=1000;
	

	for(i=0;i<n;i++)
	{
		cin>>s>>val;	
		if(val*kk<100000)continue;
		mx[s]=(val*kk)/1000;
	}
	for(i=0;i<m;i++)
	{
		cin>>s;
		if(mx[s])continue;
		mx[s]=-1;
	}
	
	cout<<mx.size()<<endl;
	for(map<string,int>::iterator it=mx.begin();it!=mx.end();++it)
	{
		cout<<(*it).first<<" ";
		if((*it).second==-1)
			cout<<"0"<<endl;
		else
			cout<<(*it).second<<endl;
	}

	return 0;
}
