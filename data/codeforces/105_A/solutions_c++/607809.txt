#include<cstdio>
#include<map>
#include<iostream>
using namespace std;
char s[21];
int main()
{
    int n,m,k;
    scanf("%d%d 0.%d",&n,&m,&k);
    map<string,int> mp;
    for (int i=0;i<n;i++)
    {
        int x;
        scanf("%s%d",s,&x);
        if ((int)(x*k)>=10000)
            mp[s]=x*k/100;
    }
    for (int i=0;i<m;i++)
    {
        scanf("%s",s);
        if (mp.find(s)==mp.end())
            mp[s]=0;
    }
    printf("%d\n",mp.size());
    for (map<string,int> :: iterator it=mp.begin();it!=mp.end();it++)
        printf("%s %d\n",it->first.c_str(),it->second);

    return 0;
}
