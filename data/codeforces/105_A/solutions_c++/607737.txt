#include <iostream>
#include <map>
#include <cmath>
using namespace std;

#define EPS .0000001

int main()
{
	int n, m;
	double k;
	cin >> n >> m >> k;
	map<string, int> mp;
	
	string s; int a;
	for (int i = 0; i < n; ++i)
	{
		cin >> s >> a;
		if (trunc(a*k+EPS) >= 100) mp[s] = trunc(a*k+EPS);
	}
	
	for (int i = 0; i < m; ++i)
	{
		cin >> s;
		if (mp.find(s) == mp.end()) mp[s] = 0;
	}
	
	cout << mp.size() << endl;
	for (map<string,int>::iterator it = mp.begin(); it != mp.end(); ++it)
		cout << it->first << " " << it->second << endl;
}
