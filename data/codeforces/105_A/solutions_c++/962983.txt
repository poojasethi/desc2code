#include <iostream>
#include <map>
#include <string>

using namespace std;

int main()
{
  int n,m;
  char c1,c2;
  int k;
  cin>>n>>m>>c1>>c2>>k;
  map<string,int> level;
  for (int i=0;i<n;i++) {
    string s;
    int val;
    cin>>s>>val;
    val=int((k*val)/100);
    if (val>=100)
      level[s]=val;
  }
  for (int i=0;i<m;i++) {
    string s;
    cin>>s;
    level[s];
  }
  cout<<int(level.size())<<endl;
  for (map<string,int>::iterator it=level.begin();it!=level.end();it++)
    cout<<it->first<<" "<<it->second<<endl;
}
