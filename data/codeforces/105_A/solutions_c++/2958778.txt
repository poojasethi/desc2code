#include<iostream>
#include<map>
#include<algorithm>
#include<string>
#include<cmath>
using namespace std;
double eps=1e-7;
int main()
{
	int a,b,i;
	double c;
	string s;
	cin>>a>>b>>c;
	map<string,int>m;
	map<string,int>::iterator it;

		for(i=1;i<=a;i++)
	{
		int y;

		cin>>s>>y;
		y=(int)(y*c+eps);
		if(y>=100)m[s]=y;
	}	
	for(i=1;i<=b;i++)
	{
		cin>>s;
		if(m[s]<100)m[s]=0;
	}

	cout<<m.size()<<"\n";
	for(it=m.begin();it!=m.end();it++)
		cout<<(*it).first<<" "<<(*it).second<<"\n";
	return 0;
}
