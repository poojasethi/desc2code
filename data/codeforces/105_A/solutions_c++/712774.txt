#include <stdio.h>
#include <string.h>
#include <string>
#include <algorithm>
#include <map>
#include <math.h>
using namespace std;

int main(void) {
	int a, b;
	double k;
	scanf("%d%d%lf", &a,&b, &k);
	map<string, int> r;
	for(int i=0;i<a;i++){
		char x[1000];
		int v;
		scanf("%s%d", x, &v);
		if(k*v+1e-6<100) continue;
		r[x] = (int)floor(k*v+1e-6);
	}
	for(int i=0;i<b;i++) {
		char x[1000];
		scanf("%s", x);
		if(r.count(x)==0) r[x]=0;
	}
	printf("%d\n", r.size());
	for(map<string, int>::iterator it=r.begin();it!=r.end();it++)
		printf("%s %d\n", it->first.c_str(), it->second);
	return 0;
}
