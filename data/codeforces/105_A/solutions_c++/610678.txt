#include <iostream>
#include <cmath>
#include <string>
#include <map>
using namespace std;
const double eps = 1e-10;

int main() {
	int n, m, val;
	double k;
	string name;
	map <string, int> res;
	cin >> n >> m >> k;
	for (int i = 0; i < n; ++i) {
		cin >> name >> val;
		val = (int) floor(val * k + eps);
		if (val >= 100)
			res[name] = val;
	}
	for (int i = 0; i < m; ++i) {
		cin >> name;
		if (!res.count(name))
			res[name] = 0;
	}
	cout << res.size() << endl;
	for (map <string, int> :: iterator it = res.begin(); it != res.end(); ++it)
		cout << it->first << " " << it->second << endl;
	return 0;
}
