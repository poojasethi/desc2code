#include <iostream>
#include <string>
#include <map>
#define EPS 1E-9
using namespace std;
map<string,int> U;
int n,m; double k;
int main(){
  cin>>n>>m>>k;
  for(int i=0;i<n;++i){
   string T; cin>>T; cin>>U[T];
   U[T]=(int)(U[T]*k+EPS);
   if(U[T]<100)U.erase(T);
                      }
  for(int i=0;i<m;++i){
   string T; cin>>T;
   U[T]+=0;
                      }
  cout<<U.size()<<'\n';
  for(map<string,int>::iterator it=U.begin();it!=U.end();++it)
   cout<<it->first<<' '<<it->second<<'\n';
  return 0;
}
