#include<iostream>
#include<map>
#include<string>
#include<cstdlib>
#include<vector>
#include<algorithm>
#include <cmath>
#define eps 1e-6
using namespace std;
int main() {
	int n,m,e;
	double k;
	string s;
	map<string,int>mp;
	cin>>n>>m>>k;
	for(int i=0;i<n;++i){
		cin>>s>>e;
		if((int)(e*k+eps)>=100)
			mp[s]=(int)(e*k+eps);
	}
	for(int i=0;i<m;++i){
		cin>>s;
		if(mp.find(s)==mp.end())
			mp[s]=0;
	}
	cout<<mp.size()<<endl;
	map<string,int>::iterator it;
	for(it=mp.begin();it!=mp.end();it++)
		cout<<it->first<<" "<<it->second<<endl;
	return 0;
}
