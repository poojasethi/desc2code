#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <map>
#define db double
#define For(i,x,y) for(i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
db K;
string s;
map<string,int> Map;
typedef map<string,int>::iterator w1;
int main() {
	scanf("%d%d%lf",&n,&m,&K);
	For(i,1,n) {
		cin>>s;
		scanf("%d",&k);
		k=int(k*K+1e-9);
		if (k>=100) Map[s]=k;
	}
	For(i,1,m) {
		cin>>s;
		if (!Map.count(s)) Map[s]=0;
	}
	printf("%d\n",Map.size());
	w1 A=Map.begin();
	for (;A!=Map.end();A++) {
		cout<<A->first;
		printf(" %d\n",A->second);
	}
	return 0;
}
