#include<map>
#include<math.h>

	cin>>n>>m>>k;

	{
		cin>>a>>s;
	if(s*k<100) ;

	
	else sk[a]=floor((s*k+0.00001));
	
	}
	for(int i=0;i<m;i++){
		cin>>a;
		if(sk[a]==0) sk[a]=0;
	}
//	cout<<0.94*8700;
	map<string,double>::iterator i;
	cout<<sk.size()<<endl;

		cout<< i->first<<" "<< i->second<<"\n";
	}
	return 0;
	
}
