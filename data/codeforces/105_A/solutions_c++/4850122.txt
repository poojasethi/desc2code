map<string, int> SS;

	while (n--){

		if ((int)((double)v * k + 0.000001) >= 100){
			SS[S] = (int)((double)v * k + 0.000001) ;
		}
	}
	while (m--){
		string S;cin >> S;
		if (SS.find(S) == SS.end())
		SS[S] = 0;
	}
