#include <iostream>
#include <map>
using namespace std;

int n, m, k;
map<string, int> sm;

int main()
{
	char tmp;
	cin >> n >> m >> tmp >> tmp >> k;
	for(int i = 0; i < n; i++)
	{
		string name;
		int exp;
		cin >> name >> exp;
		int ne = (exp * k) / 100;
		if(ne >= 100)
		{
			sm[name] = ne;
		}
	}
	for(int i = 0; i < m; i++)
	{
		string name;
		cin >> name;
		if(sm.find(name) == sm.end())
		{
			sm[name] = 0;
		}
	}
	cout << sm.size() << endl;
	for(map<string, int>::iterator i = sm.begin(); i != sm.end(); i++)
	{
		cout << i->first << ' ' << i->second << endl;
	}
	return 0;
}
