#include <stdio.h>
#include <iostream>
#include <map>
#include <string.h>
using namespace std;
map <string,int> mp;
int n,m,v,k;
char w[200];
main( ){
	scanf("%d %d 0.%d",&n,&m,&k);
	for (int i=1;i<=n;i++){
		scanf("%s %d",w,&v);
		if (v*k>=10000) mp[w]=v*k/100;
    }
	for (int i=1;i<=m;i++){
		scanf("%s",w);
		if (mp[w]<0) mp[w]=0;
	}
	cout<<mp.size()<<endl;
	for (map<string,int>::iterator it=mp.begin();it!=mp.end();it++)
		cout<<it->first.c_str()<<' '<<it->second<<endl;
    return 0;
}
