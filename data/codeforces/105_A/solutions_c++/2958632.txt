#include<iostream>
using namespace std;
#include<map>
#include<string>
#include<algorithm>
int main()

{
	map<string,int>ma;
	int m,n,i,t,h;
	double k;
	string s,s1;
	const double eps=1e-7;
	cin>>n>>m>>k;
	for(i=1;i<=n;i++)
	{
		cin>>s;
		cin>>t;
		if(t*k-100>=0)
		ma[s]=t;
	}
	map<string,int>::iterator it;
	for(it=ma.begin();it!=ma.end();) {ma[(*it).first]=int((*it).second*k+eps); it++;}
	for(i=1;i<=m;i++)
	{
		cin>>s1;
		it=ma.find(s1);
		if(it==ma.end()) ma[s1]=0;
	}
	cout<<ma.size()<<endl;
	for(it=ma.begin();it!=ma.end();it++)
		cout<<(*it).first<<" "<<(*it).second<<endl;
}
