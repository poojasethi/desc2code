#define EPS 1e-9
#include <stdio.h>
#include <iostream>
#include <map>
#include <algorithm>
using namespace std;
int n,m;
double k;
string ss[50];
int kk=0;
int main()
{
	cin>>n>>m>>k;
	
	map<string,int> M;
	for(int i=0;i<n;i++)
	{
		string ts;
		int ti;
		cin>>ts>>ti;
		if(int(ti*k+EPS)>=100)
		{
			M[ts]=int(ti*k+EPS);
			ss[kk++]=ts;	
		}
	}
	for(int i=0;i<m;i++)
	{
		string ts;
		cin>>ts;
		if(M[ts]==0) {M[ts]=0;ss[kk++]=ts;}
	}
	sort(ss,ss+kk);
	cout<<kk<<endl;
	for(int i=0;i<kk;i++)
	{
		cout<<ss[i]<<" "<<M[ss[i]]<<endl;
	}
	
}
