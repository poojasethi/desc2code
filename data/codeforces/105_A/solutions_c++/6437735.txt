#include <iostream>
#include <cstdio>
#include <map>
#include <cstring>

using namespace std;

map<string, int> ans;
int N, M, K;

int main() {
  scanf("%d %d 0.%d", &N, &M, &K);
  for (int i = 0; i < N; ++i) {
    string s;
    int v;
    cin >> s;
    cin >> v;
    int res = v * K;
    res /= 100;
    if (res < 100) continue;
    ans[s] = res;
  }
  for (int i = 0; i < M; ++i) {
    string s;
    cin >> s;
    if (ans[s] != 0) continue;
    ans[s] = 0;
  }
  cout << ans.size() << endl;
  for (map<string, int>::iterator it = ans.begin(); it != ans.end(); ++it) {
    cout << it->first << " " << it->second << endl;
  }
}
