#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <map>
using namespace std;

int n,m;
double k;
map <string,int> Map;

int main(){
	Map.clear();
	cin >> n >> m >> k;
	for (int i=0;i<n;i++){
		string s;
		int level;
		cin >> s >> level;
		int ret=level*(int)(k*100+1e-6)/100;
		if (ret>=100) Map[s]=ret;
	}
	for (int i=0;i<m;i++){
		string s;
		cin >> s;
		if (!Map.count(s)) Map[s]=0;
	}
	cout << Map.size() << endl;
	for (map<string,int>::iterator it=Map.begin();it!=Map.end();it++)
		cout << it->first << " " << it->second << endl;
}
