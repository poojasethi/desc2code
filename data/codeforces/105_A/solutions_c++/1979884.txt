#include <iostream>
#include <map>

using namespace std;

int main() {
  int n, m, k;
  char F;
  cin >> n >> m >> F >> F >> k;
  
  map <string, int> skills;
  for (int i = 0; i < n; i++) {
    string s;
    int x;
    cin >> s >> x;
    if (x * k / 100 < 100)
      continue;
    skills[s] = x * k / 100;
  }
  
  for (int i = 0; i < m; i++) {
    string s;
    cin >> s;
    skills[s] += 0;
  }
  
  map <string, int>::iterator cur;
  cout << skills.size() << endl;
  for (cur = skills.begin(); cur != skills.end(); cur++)
    cout << (*cur).first << " " << (*cur).second << endl;
  
  return 0;
}
