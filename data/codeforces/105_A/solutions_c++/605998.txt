#include <iostream>
#include <string>
#include <map>
#include <cmath>

using namespace std;

int main(int argc, char* argv[])
{
	int n,m;
	double k;
	cin >> n >> m >> k;
	map<string,int> res;
	string name;
	int level;
	for(int i=0; i<n; i++)
	{
		cin >> name >> level;
		int r = (int)(floor(level*k+1e-6)+1e-6);
		if(r>=100)
			res[name] = r;
	}
	for(int i=0; i<m; i++)
	{
		cin >> name;
		if(res.find(name)==res.end())
			res[name] = 0;
	}
	cout << res.size() << endl;
	for(map<string,int>::iterator it = res.begin(); it != res.end(); it++)
		cout << it->first << " " << it->second << endl;
	return 0;
}
