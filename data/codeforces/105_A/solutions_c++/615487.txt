#include <iostream>
#include <cstdio>
#include <map>
#include <string>
#include <cstring>

using namespace std;

map <string, int> sk;

int main()
{
    int n,l,m,k=0;
    scanf("%d %d 0.%d",&n,&m,&k);
    string t;
    for(int i=0; i<n; i++)
    {
    	cin >> t>>l;
    	if (k*l>=10000)
    		sk[t]=(k*l)/100;
    } 
    for(int i=0;i<m;i++)
    {
    	cin >> t;
    	sk[t]+=0;
    }
    cout << sk.size()<<endl;
    for(map<string,int>::iterator it=sk.begin(); it!=sk.end(); it++)
    	cout << it->first<<" "<<it->second<<endl; 
	return 0;
}
