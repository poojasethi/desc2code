

def main():
  x={}
  n,m,k=map(eval, raw_input().split())
  k=int(k*100+0.5)
  for _ in xrange(n):
    ss=raw_input()
    s=ss.split()[0]
    v=int(ss.split()[1])*k/100
    if v<100:continue
    x[s]=v
  for _ in xrange(m):
    s=raw_input()
    if s not in x:x[s]=0
  print len(x)
  for s,v in sorted(x.iteritems()):
    print s,v

main()
