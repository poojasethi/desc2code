ss = raw_input().split()
n,m,k = int(ss[0]), int(ss[1]), int(ss[2][2:])
dic = {}
res = []
for i in xrange(n):
    ss = raw_input().split()
    v = int((int(ss[1]) * k) / 100)
    dic[ss[0]] = v
    if v >= 100:
        res.append((ss[0], v))
for i in xrange(m):
    s = raw_input()
    if not s in dic or dic[s] < 100:
        res.append((s,0))
        
res = sorted(res)
print len(res)
for t in res:
    print t[0], t[1]
