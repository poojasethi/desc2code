import time
import sys, io
import re, math
#start = time.clock()
n,m,k=map(str, raw_input().split())
n,m=int(n),int(m)
k=int(k[-2]+k[-1])
#l=[int(x) for x in raw_input().split()]
a1,a2,b,answ,ansj=[],[],[],[],[]
for i in range(n):
    sk,ju=map(str, raw_input().split())
    a1.append(sk)
    a2.append(int(ju))
for j in range(m):
    b.append(raw_input())

for jk in range(n):
    chk=int(a2[jk]*k)
    if chk>=10000:
        answ.append(a1[jk])
        ansj.append(chk/100)
for jd in range(m):
    if b[jd] not in answ:
        answ.append(b[jd])
        ansj.append(0)

to=[]
for i in range(len(answ)):
    to.append([answ[i],ansj[i]])
to.sort()
print len(to)
for j in range(len(to)):
    print to[j][0],to[j][1]
