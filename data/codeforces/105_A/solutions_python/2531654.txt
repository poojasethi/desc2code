import string

n, m, k = raw_input().rstrip('\n').split()
n = int(n)
m = int(m)
k = int(k.split('.')[1])

i = 0
d = dict()
while i < n:
    
    t = raw_input().rstrip('\n').split()
    v = int((int(t[1]) * k) / 100)
    if v >= 100:
        d[t[0]] = v
        
    i += 1

i = 0
while i < m:
        
    t = raw_input().rstrip('\n')
    
    if not t in d:
        d[t] = 0
            
    i += 1

print len(d)    
sk = sorted(d.keys())
for k in sk:
    print k + ' ' + str(d[k])
