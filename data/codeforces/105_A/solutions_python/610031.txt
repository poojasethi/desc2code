#!/usr/bin/env python
import math

n,m,k=raw_input().split()
n = int(n)
m = int(m)
k = int(k[2:])
d = {}

for i in range(n):
    nm,val = raw_input().split()
    wk = int(val)*k
    if wk >= 10000:
        d[nm] = wk/100
    
for i in range(m):
    nm = raw_input()
    if d.has_key(nm):
        continue
    d[nm] = 0
    
print len(d)
for k, v in sorted(d.items()):
    print k, v
