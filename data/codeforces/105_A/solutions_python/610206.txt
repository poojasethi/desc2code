#!/usr/bin/env python
# -*- coding: utf-8 -*-
import math

n, m, k = raw_input().split()
n, m, k = int(n), int(m), int(k[2:])
d = {}
for _ in xrange(n):
    name, val = raw_input().split()
    val = int(val)*k
    if val >= 10000:
        d[name] = val/100
for _ in xrange(m):
    name = raw_input()
    if not name in d:
        d[name] = 0
print len(d)
for name, val in sorted(d.items()):
    print name, val
