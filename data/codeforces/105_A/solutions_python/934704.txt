n,m,k=map(float,raw_input().split())
n=int(n)
m=int(m)
M={}
for i in range(n):
    s=raw_input().split()
    t=int(round(int(s[1])*k,6))
    if t>=100:
        M[s[0]]=t
for i in range(m):
    s=raw_input()
    if s not in M:
        M[s]=0
print len(M)
for x in sorted(M.iterkeys()):
    print x,M[x]
