n, m, k = raw_input().split()
n = int(n)
m = int(m)
k = int(k[2:])
s = {}
for i in range(n):
    n, v = raw_input().split()
    v = int(v)
    if v * k >= 100 * 100:
        s[n] = v * k / 100
for i in range(m):
    n = raw_input()
    if n not in s:
        s[n] = 0
print len(s)
for n in sorted(s.keys()):
    print n, s[n]
