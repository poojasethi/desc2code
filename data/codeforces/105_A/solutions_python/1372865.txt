n, m, k = raw_input().split()
n = int(n)
m = int(m)
k = float(k)
skill = set()
res = []
for _ in xrange(n):
    s, v = raw_input().split()
    v = int(k * float(v) + 0.001)
    if v >= 100:
        skill.add(s)
        res.append((s, v))
for _ in xrange(m):
    s = raw_input()
    if s not in skill: res.append((s, 0))
res.sort()
print len(res)
for s, v in res: print s, v
