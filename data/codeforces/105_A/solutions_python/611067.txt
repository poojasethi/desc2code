import sys
n, m, k = sys.stdin.readline().split()
n = int(n)
m = int(m)
k = int(k[2:])
d = {}
for i in range(n):
    a, b = sys.stdin.readline().split()
    b = int(b)
    if b*k >= 100*100:
        d[a] = int(b*k/100)

for i in range(m):
    a = sys.stdin.readline().strip()
    if a not in d:
        d[a] = 0

v = [(a, b) for a, b in d.iteritems()]
v.sort()
print len(v)
for a, b in v:
    print a, b
