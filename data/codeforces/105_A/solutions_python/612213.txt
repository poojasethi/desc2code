import sys
n, m, k = sys.stdin.readline().strip().split()
n, m, k = int(n), int(m), int(k[2:])
a = {}
for _ in xrange(n):
    i, j = sys.stdin.readline().strip().split()
    j = int(j) * k
    if j >= 10000:
        a[i] = j / 100
for _ in xrange(m):
    q = sys.stdin.readline().strip()
    if q not in a:
        a[q] = 0
print len(a)
for k in sorted(a.iterkeys()):
    print k, a[k]
