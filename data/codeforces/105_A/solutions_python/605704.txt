n, m, k = raw_input().split()
n, m, k = int(n), int(m), int(k[2:])
ans = {}
for i in range(n):
	t = raw_input().split()
	skill = k * int(t[1])
	if skill >= 10000:
		ans[t[0]] = skill // 100
for i in range(m):
	t = raw_input()
	if t not in ans:
		ans[t] = 0
print(len(ans))
for (name, skill) in sorted(ans.items()):
	print name + ' ' + str(skill)
