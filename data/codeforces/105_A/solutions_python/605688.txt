#!/usr/bin/python
import itertools, math, os, random, re, sys
random.seed (1234)

while True:
	s = sys.stdin.readline ().strip ()
	if s == '':
		s = sys.stdin.readline ().strip ()
		if s == '':
			break
	n, m, k = s.split ()
	n = int (n)
	m = int (m)
	k = int (float (k) * 100 + 0.1)
	d = {}
	for i in range (n):
		name, exp = sys.stdin.readline ().split ()
		exp = (int (exp) * k) / 100
		if exp >= 100:
			d[name] = exp
	for j in range (m):
		name = sys.stdin.readline ().strip ()
		if name not in d:
			d[name] = 0
	a = [(k, v) for k, v in d.items ()]
	a.sort ()
	print len (a)
	for k, v in a:
		print k, v
