n, m, k = raw_input().split()
n, m, k = int(n), int(m), float(k)

a = {}
for i in range(n):
    name, exp = raw_input().split()
    exp = float(exp)
    t = int(exp * k + 0.0001) 
    if  t >= 100:
        a[name] = t

for i in range(m):
    name = raw_input()
    if not(name in a.keys()):
        a[name] = 0

b = []
for key in a.keys():
    b.append((key,a[key]))

b.sort()

print len(b)
