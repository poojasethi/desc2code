S = raw_input().split()
n = int(S[0])
m = int(S[1])
k = int(S[2][2:])
abil = {}
for i in xrange(n):
    skill, level = raw_input().split()
    level = int(int(level)*k/100)
    if level > 99:
        abil[skill] = level
        
for i in xrange(m):
    skill = raw_input()
    if skill not in abil.keys():
        abil[skill] = 0
print len(abil.keys())
for i in sorted(abil.keys()):
    print i, abil[i]
