n,m,k = map(float, str(raw_input()).split())
new_s = {}
for i in xrange(int(n)):
    line = raw_input().strip().split()
    level = int(round(k*int(line[1]),2))
    if level>=100:
        new_s[line[0]] = level
for i in range(int(m)):
    skill = raw_input().strip()
    try:
        new_s[skill]+=0
    except KeyError:
        new_s[skill] = 0
print len(new_s)
for i in sorted(new_s):
    print i,new_s[i]
