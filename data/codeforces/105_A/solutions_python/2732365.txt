n, m, k = raw_input().split()
n, m, k = int(n), int(m), int(k[2 : ])
d = dict((i, int(v) * k / 100) for i, v in [raw_input().split() for _ in range(n)] if int(v) * k >= 10000)
for _ in range(m):
	s = raw_input()
	if s not in d:
		d[s] = 0
r = sorted(d.items())
print len(r)
for p in r:
	print p[0], p[1]
