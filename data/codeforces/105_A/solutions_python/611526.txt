import math
ln = raw_input("").split(" ")
n, m, k = int(ln[0]), int(ln[1]), float(ln[2])
EPS = 0.00001

ant = {}
ant2 = {}
resp = {}

for i in xrange(n):
	ln = raw_input("").split(" ")
	ant[ln[0].strip()] = int(ln[1])
	
for key in ant.keys():
	if int(ant[key]*k+EPS) >= 100:
		resp[key] = int(ant[key]*k+EPS)	

for i in xrange(m):
	ln = raw_input("").strip()
	if not resp.has_key(ln):
		resp[ln] = 0
		

keys = resp.keys()
