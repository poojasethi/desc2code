n,m,k = raw_input().split();
k = float(k)
mp = {}
for i in range(int(n)):
    name,ex = raw_input().split()
    ex = int(ex)
    ex*=k
    ex+=1e-9
    if ex>=100:
        mp[name] = ex
for i in range(int(m)):
    name = raw_input()
    if mp.get(name)==None:
        mp[name] = 0
print(len(mp))
for key in sorted(mp.keys()):
    print key,"%d"%mp[key]
