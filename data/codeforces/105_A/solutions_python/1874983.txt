import sys

file = sys.stdin
#file = open("test", "r")
n, m, k = map(float, file.readline().rstrip().split())
d = {}
for _ in range(int(n)):
    p = file.readline().rstrip().split()
    name, level = p[0], int(p[1]) 
    if level*k >= 100:
        d[name] = int(level*k + 0.00000001)

for _ in range(int(m)):
    name = file.readline().rstrip()
    if name not in d:
        d[name] = 0

print len(d)
keys = list(d.iterkeys())
keys.sort()
for n in keys:
    print n, d[n]
