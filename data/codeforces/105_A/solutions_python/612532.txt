n,m,k = map(float, raw_input().split())
d = {}
e = {}
for i in range(int(n)):
	x,y = raw_input().split()
	d[x] = int(y)
for i in range(int(m)):
	e[raw_input()] = 0
for x in d:
	if d[x]*k >= 100:
		e[x] = d[x]*k
d = []
for x in e:
	d.append((x, int(e[x]+10**-6)))
d.sort()
print len(d)
for x in d:
	print x[0], x[1]
