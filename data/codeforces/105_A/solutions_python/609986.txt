n, m, k = raw_input().split()
n = int(n)
m = int(m)
k = int(k[2]) * 10 + int(k[3])
skills = dict()
while n > 0:
    name, exp = raw_input().split()
    exp = int(exp)
    if k * exp >= 10000:
        skills[name] = exp * k / 100
    n = n - 1
while m > 0:
    name = raw_input()
    if name not in skills:
        skills[name] = 0
    m = m - 1
print len(skills)
for skill in sorted(skills.keys()):
    print skill, skills[skill]
