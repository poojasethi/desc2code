from decimal import *

n, m, k = raw_input().split()

n = int(n)
m = int(m)
k = int(Decimal(k) * 100)

skills = {}
for i in range(n):
    s = raw_input().split()
    p = (int(s[1]) * k) / 100
    if p >= 100:
        skills[s[0]] = p

for i in range(m):
    s = raw_input().strip()
    if skills.get(s) is None:
        skills[s] = 0

print len(skills)
for k, v in sorted(skills.items()):
    print k, v
