n,m,k=map(float,raw_input().split())
e=dict([(a,`int(float(b) * k+1./10**12)`) for a,b in [raw_input().split() for _ in int(n)*[0]] if float(b) * k >= 100])
for i in [raw_input() for _ in int(m)*[0]]:
    if i not in e: e[i]='0'
print len(e.items())
print '\n'.join(map(lambda x: ' '.join(x), sorted(e.items())))
