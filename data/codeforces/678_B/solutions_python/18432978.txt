#!/usr/bin/python
import sys
import re

def input_numbers():
    return [int(n) for n in raw_input().split()]

y = input_numbers()[0]

def is_leap(y):
    return y % 400 == 0 or y % 4 == 0 and y % 100 != 0

def days(y):
    return 365 + is_leap(y)

leap_init = is_leap(y)
dow_delta = 0
while True:
    y += 1
    dow_delta = (dow_delta + days(y-1)) % 7;
    if (dow_delta) % 7 == 0 and is_leap(y)  == leap_init:
        break

print y
