y = int(raw_input())
if y % 400 == 0:
    vis = True
    ans = 366 % 7
    cnt = 366
elif y % 4 == 0 and y % 100 != 0:
    vis = True
    ans = 366 % 7
    cnt = 366
else:
    vis = False
    ans = 365 % 7
    cnt = 365
y += 1
while True:
    if y % 400 == 0:
        cnt += 366
        cur = True
    elif y % 100 != 0 and y % 4 == 0:
        cnt += 366
        cur = True
    else:
        cnt += 365
        cur = False
    if cnt % 7 == ans and vis == cur:
        break
    y += 1

print y
