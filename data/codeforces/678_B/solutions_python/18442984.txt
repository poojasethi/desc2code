import sys;

def getDays(y):
	if (0 == y % 400):
		return 366;
	if (0 == y % 4 and 0 != y % 100):
		return 366;
	return 365;

y = int(sys.stdin.readline())
day = getDays(y)

offset = getDays(y) % 7;
day2 = getDays(y);

while((0 != offset % 7) or day != day2):
	y += 1;
	day2 = getDays(y);
	offset += day2 % 7;
