def is_visokos(year):
    return year % 400 == 0 or (year % 4 == 0 and year % 100 != 0)

y = int(raw_input())
yn = y
y_visokos = is_visokos(y)
days = 0
while True:
    yn += 1
    yn_visokos = is_visokos(yn)
    days += 366 if yn_visokos else 365
    days %= 7
    if days == 0 and yn_visokos == y_visokos:
        print yn
        break
