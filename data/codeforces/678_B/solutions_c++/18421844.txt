#include<bits/stdc++.h>
using namespace std;
bool leap(long long y)
{
	return y%400==0||(y%4==0&&y%100!=0);
}
int main()
{
	long long y,y1;
	cin>>y;
	y1=y;
	int diff=0;
	while(1)
	{
		if(leap(y))
			diff+=2;
		else
			diff+=1;
		y++;
		if(diff%7==0&&leap(y)==leap(y1))
		{
			cout<<y;
			break;
		}

	}
}
