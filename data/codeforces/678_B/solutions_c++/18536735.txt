#include <iostream>
using namespace std;

bool leap(int a){
	if((a%4==0 && !(a%100==0)) || a%400==0)
		return true;
	else
		return false;
}

int main(){
	int y,y1,rem=0;
	cin >> y;
	y1 = y;

	do{
		y1++;
		if(leap(y1))
			rem += 2;
		else
			rem++;
	}
	while(!(rem%7==0) || (leap(y)^leap(y1)));

	cout << y1;
	return 0;
}
