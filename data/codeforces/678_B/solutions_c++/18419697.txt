#include <cstdio>
using namespace std;

int N, x, No;

int leap(int N){
	return (N % 100 != 0 && N % 4 == 0) || (N % 100 == 0 && N % 400 == 0);
}

int main(void){
	scanf("%d", &N);
	No = N;
	x = 0;
	while (1){
		x++;
		N++;
		if (leap(N)) x++;
		if (x % 7 == 0 && leap(No) == leap(N)) break;
	}
	printf("%d", N);
	return 0;
}
