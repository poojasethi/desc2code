#include <cstdio>
#include <iostream>
using namespace std;

bool leap(const int n){
    return n % 400 == 0 || (n % 4 == 0 && n % 100 != 0);
}

int main(){
    int N, a(0);
    cin >> N;
    int tmp(N);
    do{
        a = (a + 1 + leap(tmp)) % 7;
        tmp++;
    }while(a || (leap(tmp) != leap(N)));
    cout << tmp;
    
}
