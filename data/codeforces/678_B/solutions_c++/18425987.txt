#include <bits/stdc++.h>
using namespace std;

int leap(int n) {
	if (n%4 != 0) return 1;
	if (n%100 == 0 && n%400 != 0) return 1;
	return 2;
}

int main() {
	int n, days = 0;
	cin >> n;

	int x = leap(n);
	while (days == 0 || days %7 != 0 || leap(n) != x) {
		days += leap(n++);
	}
	
	cout << n << endl;
}
