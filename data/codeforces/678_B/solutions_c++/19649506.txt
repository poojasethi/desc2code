#include <stdio.h>
#include <stdlib.h>

bool is_ves(int y)
{
    return ((y % 400 == 0) || ((y % 4 == 0) && (y % 100 != 0))); 
}    

int main()
{
    int y, d = 0;;
    scanf("%d", &y);
    bool ves = is_ves(y);
    do {
        d++;
	if (is_ves(y))
	    d++;
	y++;
	d %= 7;
    } while (d || (ves != is_ves(y)));
    printf("%d\n", y);
    return 0;
}
