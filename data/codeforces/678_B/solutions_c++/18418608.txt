#include <cstdio>
#include <cstring>
int judge (int x) {
	if ((x % 4 == 0 && x % 100 != 0) || x % 400 == 0) return 366; else return 365;
}
int main() {
	int n;
	scanf("%d", &n);
	int n1 = n + 1, temp = judge(n1) % 7;
	while (!(judge(n1) == judge(n) && !temp)) n1++, temp = (temp + judge(n1)) % 7;
	printf("%d\n", n1);
	return 0;
}
