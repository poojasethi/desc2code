#include <bits/stdc++.h>
int cnt(int y) {
    return (y%4==0 && (y%100!=0 || y%400==0)) ? 366 : 365;
}
int main() {
    int y;
    scanf("%d",&y);
    int c0 = cnt(y);
    int f = 0;
    while (1) {
        f = (f+cnt(y))%7;
        ++y;
        if (cnt(y)==c0 && f==0)
            break;
    }
    printf("%d\n", y);
    return 0;
}
