#include<bits/stdc++.h>
using namespace std;
bool isLeap(int year)
{
	if(year%400==0 || year%4==0 && year%100!=0)
		return true;
	return false;
}
int main()
{
	int y;
	cin>>y;
	bool is_leap = isLeap(y);
	int d = 0;
	do 
	{
		d++;
		if(isLeap(y))
			d++;
		y++;
		d %= 7;
	} while(d || isLeap(y) != is_leap);
	cout<<y<<endl;
	return 0;
}
