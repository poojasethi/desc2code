#include<stdio.h>


int days(int year){
if(year%400==0||(year%4==0&&year%100!=0))return 366;
else return 365;
}


int main(){
	int year,sum,i;
	scanf("%d",&year);
	sum=days(year);
	for(i=1;;i++){
	sum+=days(year+i);
	if((sum%7==0)&&days(year+i+1)==days(year)){
	printf("%d",year+i+1);
	break;
	}
	}



return 0;
}
