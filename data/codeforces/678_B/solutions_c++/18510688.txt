#include<iostream>
using namespace std;

bool bi(int n){ return n%400==0 || (n%4==0) && !(n%100==0); }

int main(){
    int n,k=0;
    cin >> n;
    int y=n;
    while(k ==0 || (k%7!=0 || bi(n)!=bi(y) ) ){
        if(bi(n)) k++;
        k++;
        n++;
    }
    cout << n << endl; 
return 0;    
}
