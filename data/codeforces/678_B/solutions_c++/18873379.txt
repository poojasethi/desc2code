#include <iostream>
using namespace std;

bool cal(int y){
	return (( y % 4 == 0 && y % 100 != 0 ) || y % 400 == 0);
}

int main(){
	int y, s = 0 , ys ;
    cin >> y;
	ys = y;
	do{
		if(cal(y))
			s += 2;
		else 
			s += 1;
		y++;
	} while( s % 7 != 0 || cal(ys) != cal(y));
	cout << y;
    return 0;
}
