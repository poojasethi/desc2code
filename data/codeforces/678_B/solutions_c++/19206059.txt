#include <bits/stdc++.h>

using namespace std;

typedef __int64 ll;

int n, k,x,d;

main()
{
	cin >> x;
	if (x%400==0 || x%4==0 && x%100!=0)
		k=1;
	while (1)
	{
		int j=0;
		x++;
		if (x%400==0 || x%4==0 && x%100!=0)
		{
			d+=2;
			j=1;
		}
		else
			d++;
		d%=7;
		if (d==0 && j==k)
			break;
	}
	cout << x;
}
