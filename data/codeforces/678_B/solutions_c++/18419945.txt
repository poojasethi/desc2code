#include <iostream>
#include <cstdio>
using namespace std;
int day(int x)
{
	if((x%4==0&&x%100!=0)||(x%400==0)) return 366;
	return 365;
}
int main(int argc, char *argv[])
{
	int n,i;
	long long tot=0;
	scanf("%d",&n);
	for(i=n+1;;i++)
	{
		tot+=day(i);
		if(tot%7==0&&day(i)==day(n)) break;
	}
	printf("%d\n",i);
	return 0;
}
