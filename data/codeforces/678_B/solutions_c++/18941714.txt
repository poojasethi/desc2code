#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
int isLeap(int n){
	if(n%400==0||(n%4==0&&n%100!=0))
		return 1;
	return 0;
}
int main()
{
	int y,i,d;
	cin>>y;
	d=0;
	for(i=y;;i++){
		if(isLeap(i))
			d+=366;
		else d+=365;
		if(d%7==0&&isLeap(y)==isLeap(i+1))
			break;
	} 
	cout<<i+1<<endl;
	return 0;
}
