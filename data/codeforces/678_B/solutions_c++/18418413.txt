#include<bits/stdc++.h>
using namespace std;
int date(int x){
	if(x%400==0||(x%4==0&&x%100!=0))return 366;
	return 365;
}
int a[200010];
int main(){
	for(int i=1000;i<=200000;i++){
		a[i]=(a[i-1]+date(i))%7;
	}
	int x;
	cin>>x;
	for(int i=x+1;;i++){
		if(a[x]==a[i]&&date(x)==date(i)){cout<<i<<endl;break;}
	}
	return 0;
}
