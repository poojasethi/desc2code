#include<cstdio>
#include<cstring>
using namespace std;
int jud(int y)
{
    return y%400==0||(y%100!=0&&y%4==0);
}
int main()
{
    int y;
    scanf("%d",&y);
    int cnt=0,i;
    for(i=y+1;;i++)
    {
        if(jud(i))cnt+=2;
        else cnt+=1;
        if(cnt%7==0&&jud(y)==jud(i))break;
    }
    printf("%d\n",i);
    return 0;
}
