#include <cstdio>
int main()
{
	int y;
	scanf("%d",&y);
	int t=0,p,q;
	if ( (y%4==0 && y%100!=0) || y%400==0)
	{
		p=1;
		q=1;
	}
	else
	{
		p=0;
		q=0;
	}
	do
	{
		if (q) t+=366;
		else t+=365;
		y++;
		if ( (y%4==0 && y%100!=0) || y%400==0) q=1;
		else q=0;
	}while (t%7!=0 || p!=q);
	printf("%d",y);
}
