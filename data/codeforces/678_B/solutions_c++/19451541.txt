#include<stdio.h>

int i,j,n;

bool check(int n){
	
	if (n%400 == 0||(n%4==0&&n%100!=0)) return true;
	else return false;
	
}


int main()
{
	scanf("%d",&n);
	int tot = 0;
	for (int i = n+1;;i++){
		
		if (check(i)) tot++;
		tot++;
		
		if (tot%7==0&&check(i)==check(n)){
			
			printf("%d",i);
			break;
		}
		
	}
	
	return 0;
 }
