#include<stdio.h>
#include<iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
#define ll long long
int main(void)
{
	ll n,i,j,k,p;
	cin>>n;
	p=1;if(n%400==0||n%4==0&&n%100)p++;
	k=p;
	for(i=n+1;;i++)
	{
		j=1;
		if(i%400==0||(i%4==0&&i%100))j++;
		if(k==0&&j==p)break;
		k=(k+j)%7;
	}
	cout<<i;
	return 0;
}
