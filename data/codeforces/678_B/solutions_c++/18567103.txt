#include<bits/stdc++.h>
using namespace std;
bool is(int x){
    if(x%400==0)return 1;
    if(x%4==0&&!(x%100==0))return 1;
    return 0;
}
int main(){
    int y,mod=0;
    cin>>y;
    for(int i=y;;++i){
        mod+=1+is(i);
        mod%=7;
        if(mod==0&&(is(y)==is(i+1))){cout<<i+1<<endl;break;}
    }

}
