#include<stdio.h>
using namespace std;
int N;
int isLeap(int n){
	if(n%400==0||(n%4==0&&n%100!=0))return 366;
	return 365;
}
int main(void){
	scanf("%d",&N);
	int day=0,t=0;
	do{
		t++;
		day+=isLeap(N+t);
	}while(day%7!=0||isLeap(N)!=isLeap(N+t));
	printf("%d\n",N+t);
	return 0;
}
