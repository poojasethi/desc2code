#include<cmath>
#include<cstdio>
int leap(int x)
{
	if(x%400==0)	return 1;
	if(x%4==0&&x%100!=0)	return 1;
	return 0;
}
int main()
{
	int year;
	scanf("%d",&year);
	int n=year;
	int ans;
	year++;
	ans=(365+leap(year))%7;
	while(ans!=0||leap(n)!=leap(year))
		ans=(ans+365+leap(++year))%7;
	printf("%d\n",year);
	return 0;
}
