#include <iostream>
using namespace std;
bool Kiem_Tra(int y) {
	return y % 400 == 0 || (y % 4 == 0 && y % 100 != 0);
}
int main(){
    int y;
    cin>>y;
    int d = 0;
    bool CoPhai = Kiem_Tra(y);
	do {
		d++;
		if (Kiem_Tra(y)) d++;
		y++;
		d %= 7;
	} while (d || Kiem_Tra(y) != CoPhai);
	cout << y << endl;
    return 0;
}
