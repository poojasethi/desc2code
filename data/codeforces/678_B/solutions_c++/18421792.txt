#include<bits/stdc++.h>
using namespace std;
bool isleap(int y){
    return y%4==0&&y%100>0||y%400==0;
}
int main(){
    int y,cnt=0;
    bool f;
    cin>>y;
    f=isleap(y);
    for(int i=y+1;i;i++){
       if(isleap(i))cnt+=2;
       else cnt++;
       if(cnt%7==0&&isleap(i)==f){
        cout<<i<<endl;
        break;
       }
    }
}
