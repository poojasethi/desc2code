#include <bits/stdc++.h>

using namespace std;

int get(int x) {
  return 365 + (x % 400 == 0 || x % 4 == 0 && x % 100 != 0);
}

int main() {
  int y;
  cin >> y;
  int add = get(y) % 7, i;
  for (i = 1; add || get(y) != get(y + i); i ++) {
    add = (add + get(y + i)) % 7;
  }
  cout << y + i << endl;
}
