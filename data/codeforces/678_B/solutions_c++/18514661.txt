#include<bits/stdc++.h>
using namespace std;

bool check(int p)
{
    if(p%400==0)return 1;
    if(p%4==0&&p%100!=0)return 1;
    return 0;
}
int main()
{
    int y;
    scanf("%d",&y);
    int ans = y+1;
    int p=(365+check(ans))%7;
    while(p!=0||check(y)!=check(ans))
        p=(p+365+check(++ans))%7;
    cout<<ans<<endl;
}
