#include <iostream>
using namespace std;
typedef long long ll;
int y,st,dl;

int is_leap(int a)
{
	if(a%400 == 0 || (a%4 == 0 && a%100))
		return 1;
	else
		return 0;
}

int main()
{
	cin>>y;
	st = y;
	dl = 0;

	while(true){
		++y;
		dl = (dl + 365 + is_leap(y) ) % 7;

		if(dl == 0 && is_leap(st) == is_leap(y)){
			cout<<y<<endl;
			return(0);
		}
	}

}
