#include<bits/stdc++.h>
using namespace std;
int leap(int x){if(x%400==0 || (x%4==0 && x%100!=0)){return 1;}return 0;}
int main()
{
    int sum=0,n;
    cin>>n;
    int i=n;
    while(1)
    {
        sum+=leap(i)+365;
        sum%=7;
        if(sum==0 && leap(i+1)==leap(n)){cout<<i+1<<endl;break;}
        else {i++;}
    }
    return 0;
}
