#include<stdio.h>
int main()
{
	int a,b;
	int flag;
	scanf("%d",&a);
	b=0;
	flag=0;
	if(!(a%400)||!(a%4)&&a%100)
	{
		b++;
		flag=1;
	}
	for(a++;a<=101000;a++)
	{
		b=(b+1)%7;
		if(!b&&(!(a%400)||!(a%4)&&a%100)==flag)
			break;
		if(!(a%400)||!(a%4)&&a%100)
			b=(b+1)%7;
	}
	printf("%d\n",a);
	return 0;
}
