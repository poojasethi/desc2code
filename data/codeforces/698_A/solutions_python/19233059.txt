def q(curr,last):
    if curr==n:return 0
    if dp[curr][last]!=-1:return dp[curr][last]
    ans=1e9
    for i in xrange(4):
        if (i&1) and (last&1):continue
        if (i&2) and (last&2):continue
        if (i&1) and a[curr]<2:continue
        if (i&2) and a[curr]%2==0:continue
        ans=min(ans,(i==0)+q(curr+1,i))
    dp[curr][last]=ans
    return ans
n=int(raw_input())
a=map(int,raw_input().split())
dp=[[-1 for i in xrange(4)]for j in xrange(n)]
print q(0,0)
