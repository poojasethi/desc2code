n = int(raw_input())
a = map(int, raw_input().split())
dp = [0, 10000, 10000]
for x in a:
    ndp = [10000] * 3
    for i in xrange(3):
        if i & x != i:
            continue
        for j in xrange(3):
            if i & j:
                continue
            if ndp[i] > dp[j] + (not i):
                ndp[i] = dp[j] + (not i)
    dp = ndp
print min(dp)
