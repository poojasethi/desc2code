#include<bits/stdc++.h>
using namespace std;
int sit[105];
int dp[105][4];
int n;
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&sit[i]);
	for(int i=1;i<=n;i++){
		for(int j=0;j<=1;j++){
			if(sit[i]&(1<<j)) dp[i][j]=max(dp[i-1][j^1],dp[i-1][3])+1;
		}
		dp[i][3]=max(max(dp[i-1][0],dp[i-1][1]),dp[i-1][3]);
	}
	cout<<n-max(max(dp[n][0],dp[n][1]),dp[n][3])<<endl;
	return 0;
}
