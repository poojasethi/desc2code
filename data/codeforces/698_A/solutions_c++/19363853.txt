#include<stdio.h>
int a[110]; 
int main()
{
	int n;
	while(~scanf("%d",&n))
	{
		int t=0;
		for(int i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			if(a[i]==0)
			t++;
			else if(i!=0&&a[i]==3)
			{
			if(a[i-1]==1)a[i]=2;
			else if(a[i-1]==2)a[i]=1;
			}
			else if(a[i]==a[i-1]&&a[i-1]!=3)
			{
			a[i]=0;
			t++;	
			}
		}
		printf("%d\n",t);
	}
	
	return 0;
}
