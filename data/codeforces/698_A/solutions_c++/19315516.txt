#include <iostream>
using namespace std;

int days[105];
int main() {
	int n, ans=0;
	cin>>n;
	for (int i=0; i<n; i++) {
		cin>>days[i];
		if (days[i]==0)
			ans++;
		else if (days[i]==3) {
			if (days[i-1]==1)
				days[i]=2;
			else if (days[i-1]==2)
				days[i]=1;
			else
				days[i]=0;
		}
		else if (days[i]==days[i-1]) {
			days[i]=0;
			ans++;
		}
	}
	cout<<ans<<'\n';
	return 0;
}
