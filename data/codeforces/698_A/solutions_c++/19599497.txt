#include <bits/stdc++.h>
using namespace std ;
const int L = 105 ;
int dp[L][3] ;
int main()
{
		int N ;
		cin>>N ;
		for(int i=1;i<=N;++i)
		{
				int val ; cin>>val ;
				if(val&1) dp[i][0] = max(dp[i-1][1],dp[i-1][2]) + 1 ;
				if(val&2) dp[i][1] = max(dp[i-1][0],dp[i-1][2]) + 1 ;
				dp[i][2] = max(dp[i-1][2],max(dp[i-1][0],dp[i-1][1])) ;
		}
		int mx = max(dp[N][2],max(dp[N][0],dp[N][1])) ;
		cout<<N-mx<<endl ;
		return 0 ;
}
