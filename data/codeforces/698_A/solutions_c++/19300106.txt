#include <bits/stdc++.h>
using namespace std;
long long a[105],n,i,ans;
int main()
{
    cin >> n;
    for(i=1;i<=n;i++)
    {
        cin >> a[i];
    }
    for(i=1;i<=n;i++)
    {
        if(!a[i])
            ans++;
        else if(a[i]==a[i-1] && a[i]<3)
        {
            a[i]=0;
            ans++;
        }
        else if(a[i]==3)
        {
            a[i]=3-a[i-1];
        }
    }
    cout << ans;
	return 0;
}
