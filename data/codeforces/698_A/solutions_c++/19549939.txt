
#include<cstdio>

const int MAX=105;
int n,a[MAX];

int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
	}
	int l=1;
	while(a[l]==3){
		l++;
	}
	for(int i=l+1;i<=n;i++){
		if(a[i]==3){
			if(a[i-1]!=3) a[i]=3-a[i-1];
		}
		else{
			if(a[i]==a[i-1]) a[i]=0;
		}
	}
	int ans=0;
	for(int i=1;i<=n;i++){
		if(!a[i]) ans++;
	}
	printf("%d\n",ans);
	return 0;
}
