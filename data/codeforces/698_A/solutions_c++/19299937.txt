#include<bits/stdc++.h>
using namespace std;

int main(void)
{
	int n;
	cin >> n;
	int a[n],i;
	for(i=0;i<n;i++)	cin >> a[i];
	for(i=0;i<n-1;i++)
	{
		if(a[i]==1)
		{
			if(a[i+1]==1)	a[i+1]=0;
			else if(a[i+1]==3)	a[i+1]=2;
		}
		else if(a[i]==2)
		{
			if(a[i+1]==2)	a[i+1]=0;
			else if(a[i+1]==3)	a[i+1]=1;
		}
	}
	int sum=0;
	for(i=0;i<n;i++)
		if(a[i]==0)	sum++;
	cout << sum << endl;
}
