#include<iostream>
#include<cstdio>
using namespace std;
int main(){
	int n,a,l=0,ans=0;cin>>n;
	for(int i=0;i<n;i++){
		scanf("%d",&a);
		if(a==0)ans++,l=0;
		else if(a==1){
			if(l==2)ans++,l=0;
			else l=2;
		}
		else if(a==2){
			if(l==1)ans++,l=0;
			else l=1;
		}
		else {
			if(l==1)l=2;
			else if(l==2)l=1;
		}
	}
	cout<<ans;
	return 0;
}
