#include<stdio.h>
int a[102];
int main() {
	int n;
	while(scanf("%d",&n)!=EOF) {
		int sum=0;
		for(int l=0; l<n; l++) {
			scanf("%d",&a[l]);
			if(a[l]==0) {
				sum++;
			} else if(a[l]==3&&l) {
				if(a[l-1]==1)
					a[l]=2;
				else if(a[l-1]==2)
					a[l]=1;
			} else if(a[l]==a[l-1]) {
				a[l]=0;
				sum++;
			}
		}
		printf("%d\n",sum);
	}
	return 0;
}
