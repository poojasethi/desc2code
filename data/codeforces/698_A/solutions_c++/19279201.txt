#define N (1<<7)
#include <bits/stdc++.h>
using namespace std;

int n,a[N],dp[N][3];

int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
		scanf("%d",a+i);

	for(int i=1;i<=n;i++)
	{
		for(int j=0;j<2;j++)
		{
			dp[i][j]=N;
			if(a[i]&(1<<j)) dp[i][j]=min(dp[i][j],min(dp[i-1][1-j],dp[i-1][2]));
		}
		dp[i][2]=min(dp[i-1][0],min(dp[i-1][1],dp[i-1][2]))+1;
	}
	cout<<min(dp[n][0],min(dp[n][1],dp[n][2]))<<endl;
	return 0;
}
