#include <stdio.h>
#include <stdlib.h>
#include <algorithm>

using namespace std;

int n,m,ans=0,now=3;

int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d",&m);
		if((now&m)==0){
			now=3;
			ans++;
		}
		else{
			now&=m;
			if(now==1)
				now=2;
			else if(now==2)
				now=1;
		}
	}
	printf("%d\n",ans);
	return 0;
}
