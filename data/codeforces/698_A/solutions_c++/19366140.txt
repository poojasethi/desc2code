#include<stdio.h>
int main()
{
	int n,m,k,i,j;
	int a[100+2],sum;
	while(~scanf("%d",&n))
	{
		a[0]=0;sum=0;
		for(i=1;i<=n;i++)
			{
			scanf("%d",&a[i]);
			if(a[i]==1||a[i]==2)
			{
				if(a[i]==a[i-1])
				{
					sum++;a[i]=0;
				}
			}
			else if(a[i]==0)
			{
				sum++;
			}
			else if(a[i]==3)
			{
				if(a[i-1]==1)a[i]=2;
				if(a[i-1]==2)a[i]=1;
			}
			}
			printf("%d\n",sum);
		}
		
		
	return 0;
}
