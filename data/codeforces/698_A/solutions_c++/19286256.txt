#include <iostream>
using namespace std;
int n,ans=0,a[101];
int main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	for(int i=1;i<=n;i++){
		if(a[i]==0){
			ans++;
		}
		else if(a[i]==3){
			if(a[i-1]==1){
				a[i]=2;
			}
			else if(a[i-1]==2){
				a[i]=1;
			}
			else{
				a[i]=0;
			}
			
		}
		else if(a[i]==a[i-1]){
			ans++;
			a[i]=0;
		}
		
	}
	cout<<ans;
}
