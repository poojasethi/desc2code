#include<bits/stdc++.h>
using namespace std;
int y,s,sum,k;
int main ()
{
	cin>>s;
	for(int i=0;i<s;i++)
	{
		cin>>y;
		if(y==0)
		sum++,k=0;
		else if(y==1 && k==1)
		sum++,k=0;
		else if(y==2 && k==2)
		sum++,k=0;
		else if(y==2 && k!=y)
		k=2;
		else if(y==1 && k!=y)
		k=1;
		else if(y==3)
		{
			if(k==1)
			k=2;
			else if(k==2)
			k=1;
			else k=0;
		}
		
	}
	cout<<sum;
	return 0;
}
