#!/usr/bin/python
from sys import stdin, stdout
from collections import deque

def B():
    neib = [(1,0,0), (0,-1, 0), (-1, 0, 0), (0, 1, 0), (0, 0, 1), (0, 0, -1)]
    k, n, m = map(int, stdin.readline().split())
    layers = []
    stdin.readline()
    for i in xrange(k):
        layer = [list(stdin.readline()) for j in xrange(n)]
        layers.append(layer)
        stdin.readline()
    x, y = map(int, stdin.readline().split())
    q = deque([(x-1, y-1, 0)])
    c = 1
    while q:
        x, y, l = q.popleft()
        for nei in neib:
            x1, y1, l1 = nei
            x1, y1, l1 = x+x1, y+y1, l+l1
            if 0 <= x1 < n and 0 <= y1 < m and 0 <= l1 < k and layers[l1][x1][y1] == '.':
                if (x1,y1,l1) not in q:
                    q.append((x1,y1,l1))
                    c+=1
        layers[l][x][y] = '#'
    return c

print B()
