B = map(int, raw_input().split())
b = []
for i in range(B[0]):
	raw_input()
	b.append([list(raw_input()) for j in range(B[1])])
raw_input()
x, y = map(int, raw_input().split())
q = [[0, x - 1, y - 1]]
b[0][x - 1][y - 1] = '#'
for p in q:
	for x, y, z in [[p[j] + [0, d][i == j] for j in range(3)] for i in range(3) for d in [-1, 1] if 0 <= p[i] + d < B[i]]:
		if b[x][y][z] == '.':
			b[x][y][z] = '#'
			q.append([x, y, z])
print len(q)
