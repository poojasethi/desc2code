#!/usr/bin/python
k, n, m = map(int, raw_input().split())
a = []
for i in xrange(k):
    raw_input()
    a.append([])
    for j in xrange(n):
        a[i].append([])
        tmp = raw_input()
        for l in xrange(m):
            a[i][j].append(tmp[l])
move = [[0, 0, -1], [0, 0, 1], [0, -1, 0], [0, 1, 0], [-1, 0, 0], [1, 0, 0]]
def calc(x, y, z):
    if x < 0 or x >= k:
        return 0
    if y < 0 or y >= n:
        return 0
    if z < 0 or z >= m:
        return 0
    if a[x][y][z] == '#':
        return 0
    a[x][y][z] = '#'
    ret = 1
    for i in xrange(6):
        ret += calc(x + move[i][0], y + move[i][1], z + move[i][2])
    return ret
raw_input()
x, y = map(int, raw_input().split())
print(calc(0, x - 1, y - 1))
