N = 13
d = [-N,N,-1,1,-N*N,N*N]
k,n,m = map(int,raw_input().split())
raw_input()
a=set([])
for z in xrange(k):
    a.update(((z*N+y)*N+x for x in range(n) for y,c in enumerate(raw_input()) if c=='.'))
    raw_input()
x,y = map(int,raw_input().split())
s = x-1+y*N-N
q=[s]
a.remove(s)
for x in q:    
    for e in d:
        if e+x in a:
            a.remove(e+x)
            q.append(e+x)
            
print len(q)
