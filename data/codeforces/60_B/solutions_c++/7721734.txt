#include<cstdio>
using namespace std;

char v[10][10][11];
int k,n,m,x,y,i,j,R;

void r(int x,int y,int z){
	if(v[x][y][z]=='#'||x<0||x>=k||y<0||y>=n||z<0||z>=m)
		return;
	++R;
	v[x][y][z]='#';
	r(x+1,y,z);
	r(x-1,y,z);
	r(x,y+1,z);
	r(x,y-1,z);
	r(x,y,z+1);
	r(x,y,z-1);
	return;
}

int main()
{
	scanf("%d%d%d",&k,&n,&m);
	for(;i<k;++i)
	{
		for(j=0;j<n;++j)
		{
			scanf("%s",v[i][j]);
		}
	scanf("%d%d",&x,&y);
	}
	r(0,x-1,y-1);
	printf("%d",R);
	
	return 0;
	
}
