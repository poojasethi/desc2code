#include <bits/stdc++.h>
using namespace std;
const int MAXN=101;
vector<int>a[MAXN]={};
int mark[11][11][11]={};
string s[11][11];
int n,m,k,ans=0;
void dfs(int z,int x,int y)
{
    if(x<0||y<0||z<0)return ;
    if(s[z][x][y]!='.'||mark[z][x][y])return ;
    ans++;
    mark[z][x][y]=1;
    dfs(z+1,x,y);
    dfs(z-1,x,y);
    dfs(z,x-1,y);
    dfs(z,x+1,y);
    dfs(z,x,y+1);
    dfs(z,x,y-1);
}
main()
{
    cin>>n>>m>>k;
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
            cin>>s[i][j];
    int x,y;
    cin>>x>>y;
    dfs(0,x-1,y-1);
    cout<<ans;
}
