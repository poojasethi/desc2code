#include<stdio.h>
char a[15][15][15];
int num,k,n,m;
void dfs(int z,int x,int y)
{
	if(a[z][x][y]=='#' || z==0 || z==k+1 || x==0 || x==n+1 ||y==0 || y==m+1)
		return ;
	a[z][x][y]='#';
	num++;//printf("%d %d %d %d\n",z,x,y,num);
	dfs(z,x-1,y);
	dfs(z,x,y-1);
	dfs(z,x+1,y);
	dfs(z,x,y+1);
	dfs(z+1,x,y);
	dfs(z-1,x,y);
}
int main()
{
	int i,j,l,x,y;
	scanf("%d%d%d",&k,&n,&m);
	for(i=1;i<=k;i++)
	{	
		getchar();
		for(j=1;j<=n;j++)
		{
			getchar();
			for(l=1;l<=m;l++)
			{
					scanf("%c",&a[i][j][l]);
			}
		}
	}
	
	scanf("%d%d",&x,&y);
	dfs(1,x,y);
	printf("%d\n",num);
	return 0;
}
