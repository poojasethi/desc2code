#include <iostream>
#include <set>
#include <cstring>
#include <string>
#include <cmath>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
char v[10][10][10];
int k,n,m,x,y,i,j,d,r;
void c(int x,int y,int z)
{
	if (v[x][y][z]=='#'||x<0||x>=k||y<0||y>=n||z<0||z>=m)
	return;
	r=r+1;
	v[x][y][z]='#';
	c(x+1,y,z);
	c(x-1,y,z);
	c(x,y+1,z);
	c(x,y-1,z);
	c(x,y,z+1);
	c(x,y,z-1);
	return;
}
int main()
{
	cin>>k>>n>>m;
	for(i=0;i<k;i++)
	for(j=0;j<n;j++)
	for(d=0;d<m;d++)
	cin>>v[i][j][d];
	cin>>x>>y;
	c(0,x-1,y-1);
	cout<<r;
	return 0;
}
