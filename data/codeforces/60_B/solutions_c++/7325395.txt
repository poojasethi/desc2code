#include <iostream>
using namespace std;
int res;
int k,n,m;
char a[15][15][15];
void dfs(int x,int y,int z){
	if (a[x][y][z]=='#' || x<=0 || x>k || y<=0 || y>n || z<=0 || z>m){
		return;
	}
	res++;
	a[x][y][z]='#';
	dfs(x+1,y,z);
	dfs(x-1,y,z);
	dfs(x,y+1,z);
	dfs(x,y-1,z);
	dfs(x,y,z+1);
	dfs(x,y,z-1);
	return;
}
int main(){
	
	cin>>k>>n>>m;
	for (int i=1;i<=k;i++){
		for (int j=1;j<=n;j++){
			for (int l=1;l<=m;l++){
			    cin>>a[i][j][l];
			}
		}
	}
	int x,y;
	cin>>x>>y;
	dfs(1,x,y);
	cout<<res;
	return 0;
}
