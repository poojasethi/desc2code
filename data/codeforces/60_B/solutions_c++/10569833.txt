#include<bits/stdc++.h>
using namespace std;
int n,k,m,x,y,visited[20][20][20];
char str[20][20][20];
int dfs(int x,int y,int z){
	if(x<0||y<0||z<0||x>=k||y>=n||z>=m) return 0;
	if(visited[x][y][z]==1 || str[x][y][z]!='.') return 0;
	visited[x][y][z]=1;
	return dfs(x-1,y,z)+dfs(x+1,y,z)+dfs(x,y-1,z)+dfs(x,y+1,z)+dfs(x,y,z-1)+dfs(x,y,z+1)+1;
}
int main(){
	 cin>>k>>n>>m;
	 for(int i=0;i<k;i++){
	 	for(int j=0;j<n;j++){
	 		scanf("%s",str[i][j]);
	 	    //gets(str[i][j]);
		 }
		 //getchar();
	 }
	 cin>>x>>y;
	 //memset(visited,0,sizeof visited);
	 cout<<dfs(0,x-1,y-1)<<endl;
}
