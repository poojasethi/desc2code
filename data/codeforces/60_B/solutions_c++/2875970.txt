//#include "stdafx.h"
#include <stdlib.h>
#include <stdio.h>

int k,n,m, xs,ys,kol=0;
char map[10][10][10];

void dfs(int t,int i, int j)
{
	if(k<0 || i<0 || j<0 || t>=k || i>=n || j>=m) return;
	if(map[t][i][j]!='.') return;
	kol++;
	map[t][i][j]='+';
	dfs(t,i,j+1);
	dfs(t,i+1,j);
	dfs(t+1,i,j);
	dfs(t,i,j-1);
	dfs(t,i-1,j);
	dfs(t-1,i,j);
}

int main()
{	
	scanf("%d %d %d",&k,&n,&m);
	for(int i=0;i<k;i++)
		for(int j=0;j<n;j++)
			scanf("%s",map[i][j]);
	scanf("%d %d",&xs,&ys);
	
	dfs(0,xs-1,ys-1);

	printf("%d", kol);
	return 0;
}
