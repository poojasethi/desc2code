#include<iostream>
using namespace std;
int k, n, m, x, y, sol;
char a[10][10][11];
void DFS(int i,int j,int l)
{
     if(i < 0 || i >= k || j < 0 || j >= n || l < 0 || l >= m || a[i][j][l]=='#') 
		return;
     sol++;
     a[i][j][l]='#';
     DFS(i+1,j,l);
     DFS(i-1,j,l);
     DFS(i,j+1,l);
     DFS(i,j-1,l);
     DFS(i,j,l+1);
     DFS(i,j,l-1);
     return;
}
int main()
{
    cin >> k >> n >> m;
    for(int i = 0; i < k; i++)
            for(int j = 0; j < n; j++)
                    cin >> a[i][j];
    cin >>x >>y;
    DFS(0, x - 1, y - 1);
    cout << sol;
    return 0;
}
