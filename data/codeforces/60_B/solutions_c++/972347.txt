#include <iostream>
#include <algorithm>
using namespace std;
int k, n, m;
char g[32][32][32];
int fill(int u,int v,int w) {
  g[u][v][w]='x';
  int s=1;
  for(int u1=u-1;u1<=u+1;++u1)
    for(int v1=v-1;v1<=v+1;++v1)
      for(int w1=w-1;w1<=w+1;++w1)
        if(u1>=0&&u1<k&&v1>=0&&v1<n&&w1>=0&&w1<m&&('.'==g[u1][v1][w1])&&(1==abs(u1-u)+abs(v1-v)+abs(w1-w)))
          s+=fill(u1,v1,w1);
  return s;
}
int main(void) {
  cin>>k>>n>>m;
  for(int i=0;i<k;++i)for(int j=0;j<n;++j)cin>>g[i][j];
  int r,c;
  cin>>r>>c;
  cout<<fill(0,r-1,c-1)<<endl;
  return 0;
}
