#include<cstdio>
int R,C,L;
char grid[10][10][11];
bool vis[10][10][10];
int dr[6]={0,0,-1,1,0,0};
int dc[6]={-1,1,0,0,0,0};
int dl[6]={0,0,0,0 ,1,-1};
int cnt=0;
void flood(int l,int r,int c){
	if(r<0 || r>=R ||c<0 || c>=C || l<0 || l>=L || vis[l][r][c] || grid[l][r][c]=='#')
		return ;
	vis[l][r][c]=1; 
	cnt++;
	
	for(int i=0;i<6;i++)
		flood(l+dl[i],r+dr[i],c+dc[i]);
}
int main(){
	int x,y;
	scanf("%d%d%d",&L,&R,&C);
	for(int k=0;k<L;k++)
		for(int i=0;i<R;i++)
			scanf("%s",grid[k][i]);
	scanf("%d%d",&x,&y);
	flood(0,x-1,y-1);
	printf("%d\n",cnt);
}
