#include<iostream>

using namespace std ;

int n , m , k , ans=0 , x , y ;
char a[11][11][11] ;

void dfs(int x , int y , int z)
{
	if(a[x][y][z]=='#' || x<0 || y<0 || z<0 || x==k || y==n || z==m)
		return ;
	a[x][y][z]='#' ;
	ans++ ;
	dfs(x+1,y,z) ;
	dfs(x-1,y,z) ;
	dfs(x,y+1,z) ;
	dfs(x,y-1,z) ;
	dfs(x,y,z+1) ;
	dfs(x,y,z-1) ;
	return ;
}

int main()
{
	cin >> k >> n >> m ;
	for(int i=0 ; i<k ; i++)
		for(int j=0 ; j<n ; j++)
			for(int l=0 ; l<m ; l++)
				cin >> a[i][j][l] ;
	cin >> x >> y ;
	dfs(0,x-1,y-1) ;
	cout << ans << endl ;
	return 0 ;
}
