#include<iostream>
using namespace std;
int ans=0,k,n,m;
 char a[11][11][11];


int main()
{
    int x,y;

    cin>>k>>n>>m;

    void dfs(int z,int x,int y);
    for(int i=0;i<k;i++)
      for(int j=0;j<n;j++)
        for(int x=0;x<m;x++)
      cin>>a[i][j][x];
      cin>>x>>y;
      dfs(0,x-1,y-1);
      cout<<ans<<endl;
      return 0;
}
void dfs(int z,int x,int y)
{
    if(z<0||z>=k||x<0||x>=n||y<0||y>=m||a[z][x][y]=='#') return;
    ans++;
    a[z][x][y]='#';
    dfs(z,x,y+1);
    dfs(z,x,y-1);
    dfs(z,x+1,y);
    dfs(z,x-1,y);
    dfs(z+1,x,y);
    dfs(z-1,x,y);
}
