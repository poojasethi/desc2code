#include<iostream>
#include<cstdio>
using namespace std;

char v[10][10][11];
int k,n,m,x,y,i,j,R;

void find(int x,int y,int z)
{
     if(x<0||x>=k||y<0||y>=n||z<0||z>=m||v[x][y][z]=='#')
     return;
     ++R;
     v[x][y][z]='#';
     find(x+1,y,z);
     find(x-1,y,z);
     find(x,y+1,z);
     find(x,y-1,z);
     find(x,y,z+1);
     find(x,y,z-1);
     return;
}

int main()
{
    cin>>k>>n>>m;
    for(int i=0;i!=k;i++)
    {
            for(int j=0;j!=n;j++)
            cin>>v[i][j];
    }
    cin>>x>>y;
    R=0;
    find(0,x-1,y-1);
    cout<<R<<endl;
    return 0;
}
