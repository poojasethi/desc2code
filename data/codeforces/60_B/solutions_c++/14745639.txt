#include<iostream>
using namespace std;

int n,m,k,a[15][15][15],vst[15][15][15]={0},count1=0;

void dfs(int r,int s,int t)
{
if(a[r][s][t]==0||vst[r][s][t]==1) return;
if(r<=k&&r>0&&s<=n&&s>0&&t<=m&&t>0)
{
vst[r][s][t]=1;
dfs(r+1,s,t);
dfs(r,s+1,t);
dfs(r,s,t+1);
dfs(r,s-1,t);
dfs(r,s,t-1);
dfs(r-1,s,t);
count1++;
}
return;
}

int main()
{

int x,y,z,p,q;
char c;
cin>>k>>n>>m;

for(x=1;x<=k;x++){
for(y=1;y<=n;y++){
for(z=1;z<=m;z++)
{
cin>>c;
if((c-'.')==0) a[x][y][z]=1;
else a[x][y][z]=0;

}}}

cin>>p>>q;
dfs(1,p,q);
cout<<count1;
}
