
#include <iostream>
using namespace std;
char v[10][10][11];
int sumi;
int k,n,m;
void dfs(int x,int y,int z)
{
	if (x<0 || y<0 || z<0 || x>=k || y>=n|| z>=m || v[x][y][z]=='#')
	return;
	v[x][y][z]='#';
	sumi++;
	//cout <<x<<' '<<y<<' '<<z <<endl;
	dfs(x-1,y,z);
	dfs(x+1,y,z);
	dfs(x,y-1,z);
	dfs(x,y+1,z);
	dfs(x,y,z-1);
	dfs(x,y,z+1);
}
int main()
{
	cin >>k>>n>>m;
	for (int i=0;i<k;i++)
	{
		for (int j=0;j<n;j++)
		{
			for (int q=0;q<m;q++)
			{
				cin >>v[i][j][q];
			}
		}
	}
	int num1,num2;
	cin >>num1>>num2;
	dfs(0,num1-1,num2-1);
	cout <<sumi; 
}
