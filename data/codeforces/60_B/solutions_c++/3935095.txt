#include <iostream>
#include <queue>
using namespace std;

int K,N,M,X,Y;
char g[10][10][11];
int ret=0;

void flood(int x, int y, int z){
	if (x>=M||y>=N||z>=K||x<0||y<0||z<0) return;
	if (g[z][y][x] == '#') return;
	g[z][y][x] = '#';
	ret++;
	flood(x+1,y,z);
	flood(x,y+1,z);
	flood(x-1,y,z);
	flood(x,y-1,z);
	flood(x,y,z+1);
	flood(x,y,z-1);
}

int main(){
	cin >> K >> N >> M;
	for (int i = 0; i < K; i++){
		for (int j = 0; j < N; j++){
			cin >> g[i][j];
		}
	}
	cin >> Y >> X;
	flood(X-1, Y-1, 0);
	cout << ret << endl;
	return 0;
}
