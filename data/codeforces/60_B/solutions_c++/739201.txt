#include<cstdio>
int ans=0,n,m,k;
char g[15][15][15];
void dfs(int a,int b,int c)
{
    if(a<0||a>=k||b<0||b>=n||c<0||c>=m||g[a][b][c]=='#')return;
    ans++;
    g[a][b][c]='#';
    dfs(a-1,b,c);
    dfs(a+1,b,c);
    dfs(a,b-1,c);
    dfs(a,b+1,c);
    dfs(a,b,c-1);
    dfs(a,b,c+1);
}
int main()
{
    scanf("%d%d%d",&k,&n,&m);
    int i,j,w;
    for(i=0;i<k;i++)
    {
        for(j=0;j<n;j++)scanf("%s",g[i][j]);
    }

    int x,y;
    scanf("%d%d",&x,&y);
    x--,y--;
    dfs(0,x,y);
    printf("%d\n",ans);
    return 0;
}
