#include <cstdio>
#include<iostream>
using namespace std;
char gragh[10][10][11];
int k,n,m,x,y,ans;
void bfs(int x, int y, int z)
{
   if(gragh[x][y][z]=='#'||x<0||x>=k||y<0||y>=n||z<0||z>=m)
      return;
   ++ans;
   gragh[x][y][z]='#';
   bfs(x+1,y,z);
   bfs(x-1,y,z);
   bfs(x,y+1,z);
   bfs(x,y-1,z);
   bfs(x,y,z+1);
   bfs(x,y,z-1);
   return;
}
int main(){
   int i,j;
   scanf("%d%d%d",&k,&n,&m);
   for(i=0;i<k;++i)
      for(j=0;j<n;++j)
         scanf("%s",gragh[i][j]);
   scanf("%d%d",&x,&y);
   bfs(0,x-1,y-1);
   printf("%d",ans);
   return 0;
}
