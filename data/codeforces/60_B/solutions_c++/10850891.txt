#include<bits/stdc++.h>
using namespace std;

int k,n,m;
char v[10][10][11];
int R;

void solve(int x,int y,int z) {
  if(v[x][y][z]=='#'||x<0||y<0||z<0||x>=k||y>=n||z>=m) {
    return;
  }
  v[x][y][z]='#';
  R++;
  solve(x-1,y,z);
  solve(x+1,y,z);
  solve(x,y-1,z);
  solve(x,y+1,z);
  solve(x,y,z-1);
  solve(x,y,z+1);
  return;
}
int main() {
  //freopen("a.cpp","r",stdin);
  scanf("%d%d%d",&k,&n,&m);
  for(int i=0;i<k;i++) {
    for(int j=0;j<n;j++) {
      scanf("%s",v[i][j]);
    }
  }
  int x,y;
  scanf("%d%d",&x,&y);
  solve(0,x-1,y-1);
  printf("%d\012",R);
  return 0;
}
