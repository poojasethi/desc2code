#include <iostream>

using namespace std;

const int MAXN = 100;
int K, N, M, X, Y;
string a[MAXN][MAXN];

int dfs(int x, int y, int z)
{
	if (x < 0 || x >= K || y < 0 || y >= N || z < 0 || z >= M || a[x][y][z] == '#')
		return 0;
	a[x][y][z] = '#';
	return dfs(x-1,y,z) + dfs(x+1,y,z) + dfs(x,y-1,z) + dfs(x,y+1,z) + dfs(x,y,z-1) + dfs(x,y,z+1) + 1;
}

int main()
{
	cin >> K >> N >> M;
	for (int i = 0 ; i < K ; i++)
		for (int j = 0 ; j < N ; j++)
			cin >> a[i][j];
	cin >> X >> Y;
	cout << dfs(0, X-1, Y-1) << endl;
	return 0;
}
