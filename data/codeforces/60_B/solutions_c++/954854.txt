#include<iostream>
using namespace std;
const int maxn=20;
string arr[maxn][maxn];
int k,n,m;
int dfs(int x,int y,int z){
//	cout<<x<<' '<<y<<' '<<z<<endl;
	if(x<0||x>=k||y<0||y>=n||z<0||z>=m||arr[x][y][z]=='#')
		return 0;
	arr[x][y][z]='#';
	return dfs(x+1,y,z)+dfs(x-1,y,z)+dfs(x,y+1,z)+dfs(x,y-1,z)+dfs(x,y,z+1)+dfs(x,y,z-1)+1;
}	
int main(){
//	cout<<"silam"<<endl;
	cin>>k>>n>>m;
	for(int i=0;i<k;i++)
		for(int j=0;j<n;j++)
				cin>>arr[i][j];
	int x,y;
	cin>>x>>y;
	cout<<dfs(0,x-1,y-1)<<endl;
}
