#include<iostream>
using namespace std;
int k,n,m;
char s[15][15][15];
bool mark[15][15][15];
int dfs(int x,int y,int z)
{
    if(mark[x][y][z]==true || s[x][y][z]=='#' || x<0 || x>=k || y<0 || y>=n || z<0 || z>=m)
        return 0;
    mark[x][y][z]=true;
    return 1+dfs(x-1,y,z)+dfs(x+1,y,z)+dfs(x,y-1,z)+dfs(x,y+1,z)+dfs(x,y,z-1)+dfs(x,y,z+1);
}
int main()
{
    cin>>k>>n>>m;
    for(int x=0; x<k; x++)
        for(int y=0; y<n; y++)
            for(int z=0; z<m; z++)
                cin>>s[x][y][z];
    int y,z;
    cin>>y>>z;
    cout<<dfs(0,y-1,z-1)<<'\n';
    return 0;
}
