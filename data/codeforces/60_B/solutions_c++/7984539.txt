#include <iostream>
using namespace std;
char v[11][11][11];
int a,b,r,n,m,k;
void dfs(int x,int y,int z){
	if (v[x][y][z]=='#'||x<0||y<0||z<0||x==n||y==m||z==k) return;
	v[x][y][z]='#';
	r++;
	dfs(x+1,y,z);
	dfs(x-1,y,z);
	dfs(x,y+1,z);
	dfs(x,y-1,z);
	dfs(x,y,z+1);
	dfs(x,y,z-1);
	return;
}
int main(){
	cin>>n>>m>>k;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			for(int u=0;u<k;u++)
				cin>>v[i][j][u];
	cin>>a>>b;
	dfs(0,a-1,b-1);
	cout<<r;
	return 0;
}
