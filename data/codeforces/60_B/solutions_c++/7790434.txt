#include <iostream>
#include <cstdio>
using namespace std;
char v[10][10][11];
int n,m,k,x,y,i,j,p,R;
void r(int x,int y,int z)
{
if (v[x][y][z]=='#'||x<0||x>=k||y<0||y>=n||z<0||z>=m)
return;
R++;
v[x][y][z]='#';
r(x-1,y,z);
r(x+1,y,z);
r(x,y-1,z);
r(x,y+1,z);
r(x,y,z-1);
r(x,y,z+1);
return;
}
main()
{
cin>>k>>n>>m;
for (i=0;i<k;i++)
    for (j=0;j<n;j++)
        for (p=0;p<m;p++)
            cin>>v[i][j][p];
cin>>x>>y;
r(0,x-1,y-1);
cout<<R;
return 0;
}
