#include <iostream>
using namespace std;

bool plate[12][12][12];
int cnt=0;

void backtrack(int k,int n,int m){
    if(!plate[k][n][m]) return;
    cnt++;
    plate[k][n][m]=false;
    backtrack(k,n-1,m);
    backtrack(k,n,m-1);
    backtrack(k,n,m+1);
    backtrack(k,n+1,m);
    backtrack(k-1,n,m);
    backtrack(k+1,n,m);
}

main(){
    int k,n,m;
    cin>>k>>n>>m;
    for(int i=1;i<=k;i++) for(int j=1;j<=n;j++) for(int l=1;l<=m;l++){
        char c;
        cin>>c;
        if(c=='.') plate[i][j][l]=true;
    }
    int x,y;
    cin>>x>>y;
    backtrack(1,x,y);
    cout<<cnt;
}
