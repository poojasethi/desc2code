#include<iostream>
using namespace std;
char v[10][10][11];
int k,n,m,x,y,i,j,l,r;
void dfs(int x,int y,int z)
{
     if(v[x][y][z]=='#'||x<0||x>=k||y<0||y>=n||z<0||z>=m)
           return;
     ++r;
     v[x][y][z]='#';
     dfs(x+1,y,z);
     dfs(x-1,y,z);
     dfs(x,y+1,z);
     dfs(x,y-1,z);
     dfs(x,y,z+1);
     dfs(x,y,z-1);
}
int main()
{
    cin>>k>>n>>m;
    for(i=0;i<k;i++){
       for(j=0;j<n;j++){
          for(l=0;l<m;l++){
          cin>>v[i][j][l];}}}
    cin>>x>>y;
    dfs(0,x-1,y-1);
    cout<<r<<endl;
    return 0;
}
