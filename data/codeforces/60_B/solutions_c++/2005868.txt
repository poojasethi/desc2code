#include<cstdio>
using namespace std;
char c[10][10][11];
int k,n,m,x,y,i,j,R;
void find(int x,int y,int z)
{
	if(x<0||x>=k||y<0||y>=n||z<0||z>=m||c[x][y][z]=='#')return;
	++R;
	c[x][y][z]='#';
	find(x+1,y,z);
	find(x-1,y,z);
	find(x,y+1,z);
	find(x,y-1,z);
	find(x,y,z+1);
	find(x,y,z-1);
	return;
}
int main()
{
	scanf("%d%d%d",&k,&n,&m);
	for(int i=0;i!=k;i++)
	{
		for(int j=0;j!=n;j++)
		{
			scanf("%s",c[i][j]);
		}
	}
	scanf("%d%d",&x,&y);
	R=0;
	find(0,x-1,y-1);
	printf("%d\n",R);
}
