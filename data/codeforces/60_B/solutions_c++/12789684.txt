#include<iostream>
using namespace std;
char s[10][10][11];
int k,n,m,a,b,c,sum;
void dfs(int a, int b, int c)
{
    if(a<0 || a>=k || b<0 || b>=n || c<0 || c>=m || s[a][b][c]=='#')
    return;
    sum++;
    s[a][b][c]='#';
    dfs(a+1,b,c);
    dfs(a-1,b,c);
    dfs(a,b+1,c);
    dfs(a,b-1,c);
    dfs(a,b,c+1);
    dfs(a,b,c-1);
    return;
}
int main()
{
	cin>>k>>n>>m;
    for(int i=0;i<k;i++)
    {
        for(int j=0;j<n;j++)
        cin>>s[i][j]; 
	}
    cin>>a>>b;
    dfs(0,a-1,b-1);
    cout<<sum;
}
