#include <cstdio>

char s[11][11][11];
bool mk[11][11][11]={false};
int key=0,k,n,m;

void dfs(int t,int x,int y)
{
    if(s[t][x][y]!='.' || mk[t][x][y]) return;
    mk[t][x][y]=true;
    ++key;
    if(t)
        dfs(t-1,x,y);
    dfs(t,x-1,y);
    dfs(t,x+1,y);
    dfs(t,x,y-1);
    dfs(t,x,y+1);
    if(t<k-1)
        dfs(t+1,x,y);
}
int main()
{
    int x,y,i,j,t;
    scanf("%d%d%d",&k,&n,&m);
    for(i=0;i<k;++i)
        for(j=0;j<n;++j)
            scanf("%s",s[i][j]);
    scanf("%d%d",&x,&y);
    dfs(0,x-1,y-1);
    printf("%d\n",key);
    return 0;
}
