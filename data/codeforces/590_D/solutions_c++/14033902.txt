#include <bits/stdc++.h>
using namespace std;
const int N=151;
int n,m,s,a[N];
int d[2][N][N*N/2];

int main() {
    scanf("%d%d%d",&n,&m,&s);
    for (int i=1;i<=n;i++) scanf("%d",&a[i]);
    int I=0,J=1;
    s=min(s,n*(n-1)/2);

    for (int j=1;j<=n;j++)
    for (int k=0;k<=s;k++)
    d[0][j][k]=d[1][j][k]=1000000000;

    for (int i=1;i<=n;i++)
    {
        for (int j=0;j<=i;j++)
        for (int k=0;k<=s;k++)
        if (j>0 && k-i+j>=0)
        d[I][j][k]=min(d[J][j][k],d[J][j-1][k-i+j]+a[i]);
        else
        d[I][j][k]=d[J][j][k];
        swap(I,J);
    }
    int ans=d[J][m][s];
    for (int i=0;i<s;i++)
    ans=min(ans,d[J][m][s]);
    cout<<ans<<endl;
    return 0;
}
