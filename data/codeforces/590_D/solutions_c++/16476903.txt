#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
#define maxn 160
long long f[2][maxn][maxn*maxn];
int a[maxn],n,k,s;
int main()
{
	scanf("%d%d%d",&n,&k,&s);
	for(int i=1; i<=n; i++) scanf("%d",&a[i]);
	s=min(s,(n-k+n)*k/2);
	memset(f,0x3f,sizeof(f));
	f[0][0][0]=0;
	for(int i=1; i<=n; i++)
	{
		for(int j=0; j<=k; j++)
			for(int l=0; l<=s; l++)
			{
				//not choose
				f[i&1][j][l]=min(f[i&1][j][l],f[(i&1)^1][j][l]);
				//choose
				if(j && l-i+j>=0)
					f[i&1][j][l]=min(f[i&1][j][l],f[(i&1)^1][j-1][l-i+j]+a[i]);
			}
		memset(f[(i&1)^1],0x3f,sizeof(f[0]));
	}
	long long ans=1ll<<60;
	for(int i=0; i<=s; i++) ans=min(ans,f[n&1][k][i]);
	printf("%I64d\n",ans);
	return 0;
}
