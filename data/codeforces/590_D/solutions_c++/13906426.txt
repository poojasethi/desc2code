#include <iostream>
int n,k,s,a[300],d[300][2][300*300];
void min(int&a,int b){if(a>b)a=b;}
int main(){
    std::cin>>n>>k>>s;
    for (int i=0;i<n;i++)std::cin>>a[i];
    min(s,n*(n-1)/2);
    for (int l=0;l<=k;l++)for(int c=0;c<=s;c++)d[l][0][c]=1e9;
    d[0][0][0]=0;
    int x=0;
    for (int i=0;i<n;i++,x^=1){
        for (int l=0;l<=k;l++)for(int c=0;c<=s;c++)d[l][x^1][c]=1e9;
        for (int l=0;l<=k;l++){
            for (int c=0;c<=s;c++){
                min(d[l+1][x^1][c+(i-l)],d[l][x][c]+a[i]);
                min(d[l][x^1][c],d[l][x][c]);
            }
        }
    }
    int result=1e9;
    for (int c=0;c<=s;c++)min(result,d[k][x][c]);
    std::cout<<result;
    return 0;
}
