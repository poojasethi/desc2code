#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;
int a[155],dp[155][22505];
int ans=999999999;

int main()
{
	memset(dp,80,sizeof(dp));
    dp[0][0]=0;

	int n,k,s;
    cin>>n>>k>>s;
    for (int i=1;i<=n;i++)
    cin>>a[i];
    
    for(int k=1;k<=n;k++)
    for(int i=k;i>0;i--)
    for(int j=k-i;j<=k*i;j++)
 	dp[i][j]=min(dp[i][j],dp[i-1][j-k+i]+a[k]);

    s=min(s,n*(n-1));
    for (int j=0;j<=s;j++)
    ans=min(ans,dp[k][j]);
    cout<<ans;
}
