#include <bits/stdc++.h>
#define INF 0x3f3f3f3f
using namespace std;
long long dp[207][30007],a[207],ans,maxm,n,k,s;
int main()
{

    while(scanf("%I64d%I64d%I64d",&n,&k,&s)!=EOF){
        memset(dp,INF,sizeof(dp));
        for(int i=1;i<=n;i++){
            scanf("%I64d",&a[i]);
        }
        dp[0][0]=0;
        maxm=n*n;
        for(int i=1;i<=n;i++){
            for(int j=i-1;j>=0;j--){
                for(int q=0;q<=maxm;q++){
                    dp[j+1][q+(i-j-1)]=min(dp[j+1][q+(i-j-1)],dp[j][q]+a[i]);
                }
            }
        }
        ans=INF;
        for(long long i=0;i<=min(s,maxm);i++){
            //cout<<dp[k][i]<<' ';
            ans=min(ans,dp[k][i]);
        }
        printf("%I64d\n",ans);
    }
}
