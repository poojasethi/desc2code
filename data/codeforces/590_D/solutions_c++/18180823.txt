#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

#define ll long long
#define inf 1e18
#define eps 1e-8
#define md
#define N 160
using namespace std;
ll f[N][N*N];
int a[N];
int main()
{
	memset(f,-1,sizeof(f));
	int n,K,s;
	scanf("%d%d%d",&n,&K,&s); s=min(s,n*(n-1)/2);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	for (int i=0;i<=n;i++)
	  for (int j=0;j<=s;j++)
	    f[i][j]=inf;
	f[0][0]=0;
	for (int i=1;i<=n;i++)
	  for (int j=min(K,i);j;j--)
	    for (int k=s,t=k-(i-j);(t=k-(i-j))>=0;k--)
	      f[j][k]=min(f[j][k],f[j-1][t]+a[i]);
	ll ans=1e18; for (int i=0;i<=s;i++)  ans=min(ans,f[K][i]);
	printf("%lld\n",ans);
	return 0;
}
