#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
int dp[2][155][155*155],n,m,now,pre,k,s,i,j,d[155],sum,l,o,res;
int main(){
	scanf("%d %d %d",&n,&k,&s);
	for(i=1;i<=n;i++){
	scanf("%d",&d[i]);
	}
	if(s>=n*(n-1)/2){
	   sort(d+1,d+n+1);
	   for(i=1;i<=k;i++)
	   sum+=d[i];
	   printf("%d\n",sum);	
	   return 0;
	}
	now=1,pre=0;
	l=n*(n-1)/2;
	for(i=1;i<=n;i++)
	for(j=0;j<=s;j++)
	dp[0][i][j]=1e9;
	for(j=0;j<=s;j++)
	dp[0][k+1][j]=0;
	for(i=n;i;i--){
		for(j=1;j<=n;j++)
		for(o=0;o<=s;o++){
			res=2*1e9;
		    if(i-j<=o)
		    res=min(res,dp[pre][j+1][o-i+j]+d[i]);
		    res=min(res,dp[pre][j][o]);
		    dp[now][j][o]=res;
		}
		swap(now,pre);
	}
	printf("%d\n",dp[pre][1][s]);
	return 0;
}
