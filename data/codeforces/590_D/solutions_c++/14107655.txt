#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int f[2][160][161 * 160];
int o,n,m,k,s;
int A[160],ans;

int main() {
	scanf("%d %d %d",&n,&k,&s);
	for (int i = 1; i <= n; i++) scanf("%d",&A[i]);
	for (int i = k; i >= 1; i--) {
		memset(f[o],0,sizeof(f[o]));
		for (int j = k + 1; j <= n; j++) {
			for (int l = 0; l <= n * (n - 1) / 2; l++) {
				f[o][j][l] = min(f[o][j][l],f[o ^ 1][j][l]);
				f[o][j][l] = min(f[o][j][l],f[o][j - 1][l]);
				if (l + j - i <= n * (n - 1) / 2)
					f[o][j][l + j - i] = min(f[o][j][l + j - i],f[o ^ 1][j - 1][l] + A[j] - A[i]);
			}
		}
		o ^= 1;
	}
	ans = 0;
	for (int i = 1; i <= k; i++) ans += A[i];
	ans += f[o ^ 1][n][min(s,n * (n - 1) / 2)];
	printf("%d\n",ans);
	return 0;
}
