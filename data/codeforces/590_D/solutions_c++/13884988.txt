#include <bits/stdc++.h>

using namespace std;

const int N=152;

int f[N][N*N];
int ans,n,K,s;


void cc(int &x,const int &y)
{
    if (x==-1 || x>y) x=y;
}

int main()
{
    int i,j,k,x;
    scanf("%d%d%d",&n,&K,&s);
    memset(f,-1,sizeof(f));
    for (i=0;i<N*N;++i) f[0][i]=0;
    for (i=1;i<=n;++i)
    {
        scanf("%d",&x);
        f[i][0]=f[i-1][0]+x;
        for (j=i;j>=1;--j)
          for (k=1;k<=n*n;++k)
          {
              if (k>0 && f[j][k-1]!=-1)
                  cc(f[j][k],f[j][k-1]);
              if (k>=i-j && f[j-1][k-i+j]!=-1)
                  cc(f[j][k],f[j-1][k-i+j]+x);
          }
    }
    s=min(s,n*n);
    cout <<f[K][s] <<endl;
    return 0;
}
