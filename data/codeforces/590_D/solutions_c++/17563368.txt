#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
const int Max=1000000000;
int n,m,v,i,j,k,s,a[200],f[151][40000],ans;
int main()
{
    scanf("%d%d%d",&n,&m,&v);
    for(i=1;i<=n;i++)
        scanf("%d",&a[i]);
    s=min(v,(n-m)*m);
    for(i=1;i<=n;i++)
     for(j=0;j<=s;j++)
      f[i][j]=Max;
    f[0][0]=0;
    for(i=1;i<=n;i++)
     for(j=min(i,m);j>=1;j--)
      for(k=i-j;k<=min(s,(i-j)*j);k++)
      f[j][k]=min(f[j][k],f[j-1][k-i+j]+a[i]);
    ans=Max;
    for(i=0;i<=s;i++)
        ans=min(ans,f[m][i]);
    printf("%d",ans);
    return 0;
}
