#include<cstdio>
#include<cstring>
#include<algorithm>
#define inf 1000000000
using namespace std;
int n,m,s,f[2][155][152*152],a[155];
int main(){
	scanf("%d%d%d",&n,&m,&s);s=min(s,n*(n+1)/2);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	for (int i=0;i<=1;i++) for (int j=0;j<=m;j++) for (int k=0;k<=s;k++) f[i][j][k]=inf;
	f[0][0][s]=0;
	for (int i=0;i<n;i++){
		for (int j=0;j<=m;j++) for (int k=0;k<=s;k++) f[(i+1)&1][j][k]=inf;
		for (int j=0;j<=m;j++){
			for (int k=0;k<=s;k++){
				if (k-(i-j)>=0)
					f[(i+1)&1][j+1][k-(i-j)]=min(f[(i+1)&1][j+1][k-(i-j)],f[i&1][j][k]+a[i+1]);
				f[(i+1)&1][j][k]=min(f[(i+1)&1][j][k],f[i&1][j][k]);
			}
		}
	}
	int ans=inf;
	for (int i=0;i<=s;i++) ans=min(ans,f[n&1][m][i]);
	printf("%d",ans);
}
