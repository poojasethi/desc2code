const int N=1e5+5;
const int mod=1e9+7;
const int inf=1e9+7;

int n,k,s,i,x,w,rem;
int A[155];
int dp[2][155][23001];

main(){

	scanf("%d %d %d",&n,&k,&s);

	for(i=1 ; i<=n ; i++)
		scanf("%d",A+i);

	s=min(s,n*n);

	for(x=n+1 ; x>=1 ; x--)
		for(w=k+1 ; w>=1 ; w--)
			for(rem=0 ; rem<=s ; rem++){
				int &r=dp[x&1][w][rem];
				if(w==k+1){
					r=0;
					continue;
				}
				if(x==n+1){
					r=inf;
					continue;
				}
				r=dp[!(x&1)][w][rem];
				if(rem-(x-w)>=0)
					r=min(r,A[x]+dp[!(x&1)][w+1][rem-(x-w)]);
			}

	printf("%d\n",dp[1][1][s]);
}
