#include<bits/stdc++.h>
#define N 155
using namespace std;
int f[2][N][N*N],a[N];
int n,L,s,INF,i,j,k,ans,p;
int main(){
    scanf("%d%d%d",&n,&L,&s);
    for (i=1;i<=n;i++) scanf("%d",&a[i]);
    if (s>=n*(n-1)/2){
        sort(a+1,a+n+1);
        for (i=1;i<=L;i++) ans+=a[i];
        return printf("%d\n",ans),0;
  }s+=L*(L+1)/2;
  memset(f[0],60,sizeof(f[0]));
  INF=f[0][0][0];f[0][0][0]=0;p=0;
  for (i=0;i<n;i++,p^=1){
    memset(f[p^1],60,sizeof(f[p^1]));
    for (j=0;j<=L;j++)
        for (k=0;k<=s;k++)
            if (f[p][j][k]<INF){
                f[p^1][j][k]=min(f[p^1][j][k],f[p][j][k]);
                f[p^1][j+1][k+i+1]=min(f[p^1][j+1][k+i+1],f[p][j][k]+a[i+1]);
          }
  }for (ans=INF,k=0;k<=s;k++)
        ans=min(ans,f[p][L][k]);
    printf("%d\n",ans);
}
