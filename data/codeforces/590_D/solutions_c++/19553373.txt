#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
const int maxn = 155;
int n, k, s;
int a[maxn];
int sum[maxn];
int dp[maxn][maxn*maxn];

int main() {
    scanf("%d%d%d", &n, &k, &s);
    sum[0] = 0;
    for(int i=1; i<=n; i++) {
        scanf("%d", &a[i]);
        sum[i] = a[i] + sum[i-1];
    }
    memset(dp, 0x3f, sizeof(dp));
    memset(dp[0], 0, sizeof(dp[0]));
    for(int j=0; j<=n; j++) dp[j][0] = sum[j];
    int res = 0x3f3f3f3f;
    for(int i=1; i<=n; i++)
    for(int j=i; j>=1; j--)
    for(int w=0; w<=n*n; w++) {
        if(w-i+j < 0) continue;
        dp[j][w] = min(dp[j][w], dp[j-1][w-i+j] + a[i]);
        if(j==k && w <= s) res = min(res, dp[j][w]);
    }
    printf("%d\n", res);
    return 0;
}
