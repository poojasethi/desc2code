#include <bits/stdc++.h>

using namespace std;

const int N=152;

int ans,n,K,s;
int f[N][N*N];

inline void cc(int &x,const int &y)
{
    if (x==-1 || x>y) x=y;
}

int main()
{
    int i,j,k,x;
    scanf("%d%d%d",&n,&K,&s);
    memset(f,-1,sizeof(f));
    memset(f[0],0,N*N*4);
    for (i=1;i<=n;++i)
    {
        scanf("%d",&x);
        for (j=i;j>=1;--j)
          for (k=i-j;k<=i*j;++k)
            if (f[j-1][k-i+j]!=-1)
                cc(f[j][k],f[j-1][k-i+j]+x);
    }
    s=min(s,n*n);
    ans=1e9;
    for (i=1;i<=s;++i)
      if (f[K][i]!=-1)
          ans=min(ans,f[K][i]);
    cout <<ans <<endl;
    return 0;
}
