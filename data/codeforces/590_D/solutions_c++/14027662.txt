#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#define inf 0x3f3f3f3f
#define N 155
using namespace std;
int f[N][N*N],n,k,s,a[N];
int main(){
	scanf("%d%d%d",&n,&k,&s);
	memset(f,0x3f,sizeof f);
	f[0][0]=0;
	int x;
	for (int i=1;i<=n;i++){
		scanf("%d",&x);
		for (int j=k-1;j>=0;j--)
			for (int k=0;k<=j*i;k++){
				if (f[j][k]==inf) continue;
				f[j+1][k+i-j-1]=min(f[j+1][k+i-j-1],f[j][k]+x);
			}
	}
	int ans=inf;
	for (int i=0;i<=min(s,n*n);i++)
		ans=min(ans,f[k][i]);
	printf("%d",ans);
}
