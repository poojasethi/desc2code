#include <stdio.h>
#include <string.h>
#include <algorithm>
#define N (150+10)
#define INF 0x3f3f3f3f
using namespace std;
int f[N][N*N],num[N],Hive,Bosh,T;
int main()
{
	memset(f,0x3f,sizeof(f));
	f[0][0]=0;
	scanf("%d%d%d",&Hive,&Bosh,&T);
	for(int i=1;i<=Hive;i++)
	{
		scanf("%d",&num[i]);
	}
	for(int k=1;k<=Hive;k++)
	{
		for(int i=k-1;i>=0;i--)
		{
			for(int j=0;j<=Hive*Hive;j++)
			{
				f[i+1][j+(k-i-1)]=min(f[i+1][j+(k-i-1)],f[i][j]+num[k]);
			}
		}
	}
	int ans=INF;
	for(int i=0;i<=min(Hive*Hive,T);i++)
	{
		ans=min(ans,f[Bosh][i]);
	}
	printf("%d\n",ans);
	return 0;
}
