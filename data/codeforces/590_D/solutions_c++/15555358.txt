#include<bits/stdc++.h>
using namespace std;
int dp[2][155][12000],inp[200];
int main()
{
    int n,k,s;cin >> n>>k>>s;
    s= min(s,n*(n+1)/2);
    dp[0][0][0] = 0;
    for(int i = 1;i<=n;i++)
        cin >> inp[i];
    int majd = 1;
    for(int i = 1;i<=k;i++,majd^=1)
        for(int j = i;j<=n;j++)
            for(int ss = 0;ss<=s;ss++)
    {
        dp[majd][j][ss] = INT_MAX;
        if(ss)dp[majd][j][ss] = dp[majd][j][ss-1];
        if(j>i)dp[majd][j][ss] = min(dp[majd][j][ss],dp[majd][j-1][ss]);
        if(ss>=(j-i))dp[majd][j][ss] = min(dp[majd][j][ss],dp[majd^1][j-1][(ss-(j-i))]+inp[j]);
    }
    cout << dp[majd^1][n][s];
}
