#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int n,k,s;
int a[200];
int f[200][200*200];
int main()
{
	scanf("%d %d %d",&n,&k,&s);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
	}
	memset(f,0x3f,sizeof(f));
	f[0][0]=0;
	for(int i=1;i<=n;i++)
	{
		for(int j=i-1;j>=0;j--)
		{
			for(int k=0;k<=i*j;k++)
			{
				if(f[j][k]!=0x3f3f3f3f)
				{
					f[j+1][k+i-j-1]=min(f[j+1][k+i-j-1],f[j][k]+a[i]);
				}
			}
		}
	}
	int ans=0x3f3f3f3f;
	for(int i=0;i<=min(s,n*n);i++)
	{
		ans=min(ans,f[k][i]);
	}
	printf("%d\n",ans);
}
