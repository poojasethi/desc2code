#include<iostream>
#include<cstdlib>
#include<cstring>
#include<cstdio>
using namespace std;
int  a[200];
int  f[200*200][200];
int n, k, s;
int main()
{
    scanf ("%d%d%d", &n, &k, &s);
    for (int i = 1; i <= n; i++)
    {
        scanf ("%d", &a[i]);
    }
    memset(f, 0x3f3f3f3f, sizeof(f));
    f[0][0] = 0;
    for (int k = 1; k <= n; k++)
    for (int j = k; j > 0; j--)
    {
         for (int i = 0; i <= n*n; i++)
            if (i >= k- j)f[i][j] = min (f[i][j], f[i - (k-j)][j-1] + a[k]);
    }
    int re = 0x3f3f3f3f;
    for (int i = 0; i <= s && i <= n*n; i++)
        re = min(re, f[i][k]);
    printf("%d\n", re);

    return 0;
}
