#include <cstdio>

#define N 155
#define M 12000

int dp[N][M];
bool chk[N][M];

void chkmin(int &a, int b) { if (a > b) a = b; }
void chkmax(int &a, int b) { if (a < b) a = b; }

main() {
	int n, m, s, cur = 0;
	scanf("%d %d %d", &n, &m, &s);
	s += (n - m) * (n - m - 1) / 2;
	chkmin(s, n * (n - 1) / 2);

	chk[0][0] = 1;
	for (int i = 1; i <= n; i ++) {
		int a, b = n - i;
		scanf("%d", &a);
		cur += a;
		for (int cnt = n - m - 1; cnt >= 0; cnt --) {
			for (int j = s - b; j >= 0; j --) if (chk[cnt][j]) {
				chk[cnt + 1][j + b] = 1;
				chkmax(dp[cnt + 1][j + b], dp[cnt][j] + a);
			}
		}
	}

	int mx = 0;
	for (int i = 0; i <= s; i ++) chkmax(mx, dp[n - m][i]);
	printf("%d\n", cur - mx);
}
