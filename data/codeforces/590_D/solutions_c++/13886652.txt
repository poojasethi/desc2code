#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
typedef long long ll;
const ll inf=(ll)1<<62;
ll f[2][201][31001],ans;
int i,j,k,n,m,now,pre,a[201],lim,tmp;
int main(){
	scanf("%d%d%d",&n,&m,&lim);lim=min(lim,m*(n-m));
	for (i=1;i<=n;i++) scanf("%d",&a[i]);
	for (now=0,pre=1,i=0;i<=m;i++)
	 for (j=0;j<=lim;j++) f[now][i][j]=inf;
	f[now][0][0]=0;
	for (i=1;i<=n;i++){
		now^=1;pre^=1;
		for (j=0;j<=m;j++)
		 for (k=0;k<=lim;k++){
		 	f[now][j][k]=f[pre][j][k];tmp=i-j;
		 	if (i>=j&&j&&k>=tmp) f[now][j][k]=min(f[now][j][k],f[pre][j-1][k-tmp]+a[i]);
		 }
	}
	for (ans=inf,i=0;i<=lim;i++) ans=min(ans,f[now][m][i]);
	cout<<ans<<endl;
	return 0;
}
