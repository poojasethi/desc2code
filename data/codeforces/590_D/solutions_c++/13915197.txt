#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>

using namespace std;

const int N=150;
int a[N+3],f[N+3][N*N+3],n,m,S,ans;

int main()
{
	scanf("%d%d%d",&n,&m,&S);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	S=min(S,m*n);
	memset(f,120,sizeof(f));
	f[0][0]=0;
	for (int i=1;i<=n;i++)
	{
		int jj=min(i,m);
		for (int j=jj-1;j>=0;j--)
			for (int k=0;k+i-j-1<=S;k++)
				if (f[j][k]<2e9) f[j+1][k+i-j-1]=min(f[j+1][k+i-j-1],f[j][k]+a[i]);
	}
	ans=2e9;
	for (int i=0;i<=S;i++) ans=min(ans,f[m][i]);
	printf("%d",ans);
}
