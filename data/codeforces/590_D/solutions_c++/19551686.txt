#include <cstdio>
#include <iostream>
#include <cstring>

using namespace std;

typedef long long ll;

int n,k,s;

const int N = 251;

int q[N];

int dp[N][N * N];

int main(){
	cin >> n >> k >> s;
	for (int i = 1;i <= n;i++)
		scanf("%d",&q[i]);
	int inf = 150000001;
	for (int i = 0;i <= n;i++)
		for (int j = 0;j <= n * n;j++)
			dp[i][j] = inf;
    dp[0][0] = 0;
	for (int r = 1;r <= n;r++)
		for (int i = r;i;i--)
			for (int j = r - i;j <= n * n;j++){
				dp[i][j] = min(dp[i][j],dp[i - 1][j - (r - i)] + q[r]);
			}
    int ans = inf;
	for (int i = 0;i <= min(n*n,s);i++)
		ans = min(ans,dp[k][i]);
	cout << ans << endl;
}
