#include<cstdio>
#include<cstdlib>
#include<cstring>

#include<iostream>
#include<algorithm>
using namespace std;

const int _N=160;

int n,kth,d;
int f[_N][_N*_N];

int main()
{
    scanf("%d%d%d",&n,&kth,&d);
    static int s;
    for(int i=0;i<=n;i++)
        for(int j=0;j<=n*n;j++)
            f[i][j]=-1;
    f[0][0]=0;
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&s);
        for(int j=i-1;j>=0;j--)
            for(int k=0;k<=i*j;k++)
                if(f[j][k]>=0)
                    f[j+1][k+i-j-1]=f[j+1][k+i-j-1]<0?f[j][k]+s:min(f[j+1][k+i-j-1],f[j][k]+s);
    }
    static int ans=-1;
    for(int i=0;i<=min(d,n*n);i++)
        if(f[kth][i]>=0)
            ans=ans<0?f[kth][i]:min(f[kth][i],ans);
    printf("%d\n",ans);
    return 0;
}
