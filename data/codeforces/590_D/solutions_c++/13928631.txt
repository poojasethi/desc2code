#include<iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>
#define INF 0x3f3f3f3f3f
using namespace std;
int ans=INF,n,x,s,K;
int f[200][40000];
int main()
{
	scanf("%d%d%d",&n,&K,&s);
	memset(f,INF,sizeof(f));
	f[0][0]=0;
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&x);
		for(int j=i-1;j>=0;j--)
		{
			for(int k=0;k<=i*j;k++)
			{
				if(f[j][k]!=INF)
				f[j+1][k+i-j-1]=min(f[j][k]+x,f[j+1][k+i-j-1]);
			}
		}
	}
	for(int i=0;i<=min(s,n*n);i++)
	ans=min(f[K][i],ans);
	printf("%d",ans);
	return 0;
}
