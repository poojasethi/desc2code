#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<iostream>
#include<algorithm>
#define inf 1000000000

using namespace std;

int  dp[210][210*210];
int n,K,s;

void add(int &x,int y)
{
	if (x>y) x=y;
}

int main()
{
	scanf("%d%d%d",&n,&K,&s);
	for (int i=0;i<=n;i++)
	  for (int j=0;j<=n*n;j++)
	    dp[i][j]=inf;
	dp[0][0]=0;
	for (int i=1;i<=n;i++)
	{
		int x;
		scanf("%d",&x);
		for (int j=i-1;j>=0;j--)
		  for (int k=0;k<=i*j;k++)
		    if (dp[j][k]!=inf) add(dp[j+1][k+i-j-1],dp[j][k]+x);
	}
	int ans=inf;
	for (int i=0;i<=min(s,n*n);i++) ans=min(ans,dp[K][i]);
	printf("%d\n",ans);
	return 0;
}
