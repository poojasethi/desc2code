#include<stdio.h>
#include<algorithm>
using namespace std;
int dp[151][22501];
int num[151];
int main()
{
    int n,k,s;
    scanf("%d%d%d",&n,&k,&s);
    s=s+k*(k+1)/2;
    if(s>(2*n-k+1)*k/2)s=(2*n-k+1)*k/2;
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&num[i]);
    }
    for(int j=0;j<=k;j++)
    {
        for(int l=0;l<=s;l++)
        {
            dp[j][l]=1000000007;
        }
    }
    dp[0][0]=0;
    for(int i=1;i<=n;i++)
    {
        for(int j=min(i,k);j>=1;j--)
        {
            for(int l=i;l<=s;l++)
            {
                dp[j][l]=min(dp[j][l],dp[j-1][l-i]+num[i]);
            }
        }
    }
    int ans=1000000007;
    for(int l=0;l<=s;l++)
    {
        ans=min(ans,dp[k][l]);
    }
    printf("%d\n",ans);
    return 0;
}
