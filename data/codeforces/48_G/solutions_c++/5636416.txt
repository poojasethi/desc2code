#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
#define N 210000
using namespace std;
int head[N],next[N*2],e[N*2],c[N*2],size[N*2],sum[N],i,j,loop[N*2],S,o,bo[N],len,
	x,y,n,fa[N],d[N],tot,k;
long long f[N],g[N],tmp;
void findcircle(int x){
	bo[x]=1;
	for(int i=head[x];i;i=next[i])if(e[i]!=fa[x]){
		if(bo[e[i]]){
			if(tot)return;
			for(int j=x;j!=e[i];j=fa[j])loop[++tot]=j,sum[tot+1]=d[j];
			loop[++tot]=e[i]; sum[tot+1]=c[i];
		}else fa[e[i]]=x,d[e[i]]=c[i],findcircle(e[i]);
	}
}
void dfs(int x){
	bo[x]=1;
	size[x]=1;
	for(int i=head[x];i;i=next[i])if(!bo[e[i]]){
		dfs(e[i]);
		size[x]+=size[e[i]];
		f[x]+=f[e[i]]+1ll*c[i]*size[e[i]];
	}
}
void dfs2(int x){
	bo[x]=1;
	for(int i=head[x];i;i=next[i])if(!bo[e[i]]){
		g[e[i]]=g[x]+f[x]-f[e[i]]+1ll*(n-2*size[e[i]])*c[i];
		dfs2(e[i]);
	}
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i){
		scanf("%d%d%d",&x,&y,&k);
		next[++len]=head[x]; head[x]=len; e[len]=y; c[len]=k;
		next[++len]=head[y]; head[y]=len; e[len]=x; c[len]=k;
	}
	memset(bo,0,sizeof(bo));
	findcircle(1);
	memset(bo,0,sizeof(bo));
	for(i=1;i<=tot;++i)bo[loop[i]]=1;
	for(i=1;i<=tot;++i)dfs(loop[i]);
	for(i=1;i<=tot;++i)loop[i+tot]=loop[i],sum[i+tot+1]=sum[i+1];
	for(i=1;i<=tot*2;++i)sum[i]+=sum[i-1];
	S=sum[tot+1];	j=1;	tmp=0;	o=0;
	for(i=2;i<=tot;++i)tmp+=f[loop[i]]+1ll*size[loop[i]]*(S-sum[i]);
	for(i=1;i<=tot;++i){
		while((sum[j+1]-sum[i])*2<S){
			++j;
			tmp=tmp+1ll*size[loop[j]]*(2*(sum[j]-sum[i])-S);
			o+=size[loop[j]];
		}
		g[loop[i]]=tmp;
		tmp+=1ll*(sum[i+1]-sum[i])*(n-o-o)+f[loop[i]]-f[loop[i+1]];
		o-=size[loop[i+1]];
	}
	memset(bo,0,sizeof(bo));
	for(i=1;i<=tot;++i)bo[loop[i]]=1;
	for(i=1;i<=tot;++i)dfs2(loop[i]);
	for(i=1;i<=n;++i)printf("%I64d ",f[i]+g[i]);
}
