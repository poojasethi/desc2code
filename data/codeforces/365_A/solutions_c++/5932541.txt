#include <iostream>
#include <set>
using namespace std;
int main()
{
int x,a,y,z=0;
cin>>x>>y;
set<int>ss;
for(int i=0;i<x;i++)
{ss.clear();
    cin>>a;
    while(a>0)
    {if(a%10<y+1)
        ss.insert(a%10);
        a=a/10;}
            if(ss.size()==y+1)
                z++;
        }
cout<<z;
}
