#include<cstdio>
int n,i,x,b[10],y,ans=0,k,ii,cont=0;
int main()
{
	scanf("%d%d",&n,&k);
	for(ii=1;ii<=n;ii++)
	{
		for(x=0;x<=9;x++)b[x]=0;
		scanf("%d",&x);
		for(;x;x/=10)
		{
			b[x%10]=1;
			++cont;
		}
		y=1;
		for(i=0;i<=k;i++)
            y&=b[i];
		if(y)ans++;
	}
	printf("%d\n",ans);
	return 0;
}
