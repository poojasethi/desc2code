#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main(){
	long long n,a,b,i,j,t,s=0;
	string x;
	cin>>n>>a;
	for(i=0;i<n;++i){
		cin>>x;		
		t=0;
		sort(x.begin(),x.end());
		for(j=0;j<x.length();++j){
			if(x[j]-'0'==t)
				t++;}
		if(t>=a+1)
			s++;}
	cout<<s;
	return 0;
}
