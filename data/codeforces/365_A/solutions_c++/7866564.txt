#include<iostream>
using namespace std;
int main()
{
	int n,k; cin>>n>>k;
	int  ans=0;
	while(n--)
	{
		bool flag=true;
		long int n; cin>>n;
		int a[10]={0};
		for(;n;n=n/10)
			a[n%10]=1;
		for(int i=0;i<=k;i++)
			if(a[i]==0)
			{ flag=false; break; }
		if(flag) ans++;
	}
	cout<<ans<<endl;
	return 0;
}
