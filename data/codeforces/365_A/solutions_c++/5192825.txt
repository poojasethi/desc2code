#include<stdio.h>
int n,k,r=0,i;
char a[100];
int main()
{
    scanf("%d %d",&n,&k);
    while(n--)
    {
        scanf("%s",a);
        int re[10]={0},t=0;
        for(i=0;a[i];i++) if(a[i]-'0'<=k&&re[a[i]-'0']==0) {re[a[i]-'0']=1;t++;}
        if(t==k+1) r++;
    }
    printf("%d\n",r);
    return 0;
}
