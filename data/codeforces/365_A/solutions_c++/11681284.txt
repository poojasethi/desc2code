#include <iostream>
using namespace std;
int n,k,sum;
string s;
bool is(){
  bool b[212]={0};
  for(int i=0;i<s.length();i++)
    b[s[i]]=1;
  for(int i='0';i<='0'+k;i++)
    if(!b[i])
      return 0;
  return 1;
}
int main(){
  cin>>n>>k;
  while(n--){
    cin>>s;
    sum+=is();
  }
  cout<<sum<<endl;
  return 0;
}
