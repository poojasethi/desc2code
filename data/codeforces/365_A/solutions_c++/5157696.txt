#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
string s="0123456789",p;
int main() {
	int n,k,t,ans=0,j;
	cin>>n>>k;
	if(k>9)k=9;
	for(int i=0;i<n;i++){
		cin>>p;
		t=1;
		for(j=0;j<=k;j++)
		  if(p.find(s[j])==-1)t=0;
		if(t==1)ans++;
	}
	cout<<ans<<endl;
	return 0;
}
