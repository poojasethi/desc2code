#include<iostream>

using namespace std;

int main(){
  int n,k,ans=0;
  cin>>n>>k;
  while(n--){
    bool s[15];
    for(int i=0;i<=k;i++)
      s[i]=0;
    int a;
    cin>>a;
    ans++;
    while(a){
      s[a%10]=1;
      a/=10;
    }
    for(int i=0;i<=k;i++)
      if(!s[i]){
	ans--;
	break;
      }
  }
  cout<<ans<<endl;
  return 0;
}
