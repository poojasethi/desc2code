#include<stdio.h>
#include<string.h>
int main (void)
{
	char s[99];
	int j,n,k,l,i,c=0;
	scanf("%d %d%*c",&n,&k);
	for(i=0;i<n;i++){
		gets(s);
		for(l=0;l<=k;l++){
			for(j=0;j<strlen(s);j++)
				if(s[j]==l+48)
					break;
			if(!s[j])
				break;
		}
		if(l==k+1)
			c++;
	}
	printf("%d",c);
    return 0;
}
