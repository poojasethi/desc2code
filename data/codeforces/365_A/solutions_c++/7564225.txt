#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int main()
{
    int n,cnt=0 ;
    char c ; cin >> n >> c ;
    string s;
    cnt=n;
    while (n--){
        cin >> s ;
        for (char i='0'; i<=c; i++) if (s.find(i)==-1){ cnt--; break;}
    }
    cout << cnt ;
    return 0;
}
