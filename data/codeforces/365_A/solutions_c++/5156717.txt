# include <iostream>

using namespace std;

int n,a,k,ans;
int B[20];

int func ( int a )
{
	int h=0;
	while (a>0)
	{
		B[a%10]=1;
		a/=10;
	}
	
	for ( int i=0; i<=k; i++ )
		h+=B[i],B[i]=0;
	
	return ( h == k+1 ? 1 : 0 );
}

int main()
{
	cin >> n >> k;
	for ( int i=0; i<n; i++ )
		cin >> a,
			ans+=func(a);
	cout << ans;
	return 0;
}
