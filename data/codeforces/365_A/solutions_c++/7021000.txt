#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,n,k,ans=0;
	cin>>n>>k;
	string s;
	while(n--)
	{
		cin>>s;
		bool a[10]={0},flag=1;
		for(i=0;i<s.length();i++)
			a[s[i]-'0']=1;
		for(i=0;i<=k;i++)
			if(!a[i])
			{
				flag=0;
				break;
			}
		if(flag)
			ans++;
	}
	cout<<ans<<endl;
	return 0;
}
