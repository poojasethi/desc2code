#include<iostream>

using namespace std;

int main()
{
int n,k,a;int count=0;;
cin>>n>>k;
for(int i=0;i<n;i++)
{
cin>>a;
bool hash[11]={false};
while(a!=0)
{
int x=a%10;
if(x<=k)
hash[x]=true;
a=a/10;
}
int j;
for(j=0;j<=k;j++)
{
if(hash[j]==false)
{break;}
}
if(j==k+1)
count++;
}
cout<<count<<"\n";
}
