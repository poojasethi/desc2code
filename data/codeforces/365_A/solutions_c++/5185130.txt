#include <iostream>
#include <cmath>
using namespace std;
string m;
int k,n,z,s=0;
int lg[10];
int main()
{
cin >> n >> k;
for (int i=0;i<=k;i++) lg[i]=-1;
for (int i=0;i<n;i++)
{
cin >> m;
for (int j=0;j<m.length();j++) lg[m[j]-'0']=i;
for (z=0;z<=k;z++) if (lg[z]!=i) {break;}
if (z-1==k) s++;
}
cout << s;
}
