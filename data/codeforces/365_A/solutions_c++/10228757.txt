#include <bits/stdc++.h>

using namespace std;

int n, k;

int ans;

int main (){
	cin >> n >> k;
	for ( int i = 0 ; i < n ; i++ ){
		string s;
		cin >> s;

		sort ( s.begin (), s.end () );
		int ind = 0, x = 0;
		while ( ind < s.size () ){
			x += (s [ ind ] == x + '0');
			ind++;
		}

		if ( x > k )
			ans++;
	}

	cout << ans << endl;
}
