#include<bits/stdc++.h>
using namespace std;
int n, k;
char s[15];
int main()
{
	cin>>n>>k;
	int ans = 0;
	int i;
	while(n--)
	{
		cin>>s;
		for(i = 0;i<=k;i++)
		{
			char str[10];
			str[0] = i+'0';
			str[1] = '\0';
			if(!strstr(s, str))
			break;
		}
		if(i > k)	ans++;
	}
	cout<<ans<<endl;
	return 0;
}
