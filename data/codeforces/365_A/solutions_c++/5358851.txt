#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int l,s,a[15],k,i,n,m;
string x;
main(){
cin>>n>>k;
s=n;
for(l=0; l<n; l++){
cin>>x;

for(i=0; i<x.size(); i++){
a[x[i]-48]++;
}

for(i=0; i<=k; i++){
if(a[i]==0) {s-=m;m=0;}

}}

cout<<s<<endl;

}
