#include<iostream> 
using namespace std;
int main(){
	int n, k, a, c[10], i, x=0;
	cin >> n >> k;
	while(n>0){
		n--;
		cin >> a;
		for(i=0; i<=k; i++){
			c[i]=0;
		}
		while(a!=0){
			c[a%10]=1;
			a/=10;
		}
		for(i=0;i<=k&&c[i]==1;i++);
		if(i==k+1){
			x++;
		}
	}
	cout << x << endl;
	return 0;
}
