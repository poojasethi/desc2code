#include<iostream>
#include<cstdio>
#include<algorithm>
#include<string>
#include<cstring>
using namespace std;
#define MAXN 300010
#define LL long long
int num[MAXN],p[MAXN],cnt[MAXN];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		scanf("%d",&num[i]);
		p[num[i]]=i;
	}
	int a,b;
	memset(cnt,0,sizeof(cnt));
	while(m--){
		scanf("%d%d",&a,&b);
		int l=p[a],r=p[b];
		if(l>r) swap(l,r);
		cnt[r]=max(cnt[r],l);
	}
	LL ans=0;
	cnt[0]=0;
	for(int i=1;i<=n;i++){
		cnt[i]=max(cnt[i],cnt[i-1]);
		ans+=i-cnt[i];
	}
	cout<<ans<<endl;
	return 0;
}
