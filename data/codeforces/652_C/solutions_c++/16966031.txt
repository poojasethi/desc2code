#include <iostream>
#include <fstream>
#include <algorithm>
using namespace std;

long long n,i,j,k,ans,m,b,c,pos[300001],a[300001];

main(){
cin>>n>>m;
for(i=1;i<=n;i++)
{scanf("%d",&j);
 pos[j]=i;}
       
for(i=1;i<=m;i++)       
{scanf("%d%d",&b,&c) ;    
if(pos[b]>pos[c])
a[pos[b]]=max(a[pos[b]],pos[c]);
else a[pos[c]]=max(a[pos[c]],pos[b]);
}       
       
for(i=1;i<=n;i++)
{
k=max(k,a[i]);
ans+=i-k;
}       
       
cout<<ans<<endl;

}
