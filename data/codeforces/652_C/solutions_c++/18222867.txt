#include <bits/stdc++.h>
using namespace std;

const int n=300300,m=300300;

int j,i,arr[n+1],x[m+1],y[m+1],pos[n+1];
vector<int> z[n+1];

int main() {
	
	ios_base::sync_with_stdio(false);cin.tie(0);
	
	int N,M;
	cin>>N>>M;
	
	
	for(i=1;i<=N;i++)
	cin>>arr[i],pos[arr[i]]=i;
	
	for(i=1;i<=M;i++)
		cin>>x[i]>>y[i],z[min(pos[x[i]],pos[y[i]])].push_back(max(pos[x[i]],pos[y[i]]));
	
	long long int ans=0;
	int rg=N+1;
	
	for(i=N;i>=1;i--)
	{
		for(j=0;j<z[i].size();j++)
			rg=min(rg,z[i][j]);
			
		ans+=(rg-i);
	}
	
	cout<<ans;
	
	return 0;
}
