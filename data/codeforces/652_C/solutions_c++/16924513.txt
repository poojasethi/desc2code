
#include <bits/stdc++.h>
using namespace std;

int t[1000000];
int cl[1000000];
int d[1000000];

int main() {
	int n, m;
	cin >> n >> m;
	int a, b;
	for(int i = 1; i <= n; i++) {
		scanf("%d", &a);
		t[a] = i;
	}
	
	for(int i = 0; i < m; i++) {
		scanf("%d%d", &a, &b);
		a = t[a];
		b = t[b];
		if(a > b) swap(a, b);
		cl[b] = max(cl[b], a);
	}
	
	long long w = 0;
	
	
	for(int i = 1; i <= n; i++) {
		if(cl[i] == 0)
			d[i] = d[i-1]+1;
		else
			d[i] = min(d[i-1]+1, i-cl[i]);
		w += d[i];
	}
	cout << w;
}
