#include<stdio.h>
#include<iostream>
using namespace std;
const int maxn = 3e5+7;

int n,m,p[maxn],t[maxn];
int c[maxn];
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&p[i]);
        t[p[i]]=i;
    }
    for(int i=1;i<=m;i++)
    {
        int x,y;scanf("%d%d",&x,&y);
        int a = 0,b = 0;
        a=max(t[x],t[y]),b=min(t[x],t[y]);
        c[a]=max(c[a],b);
    }
    long long ans = 0;
    int l = 0;
    for(int i=1;i<=n;i++)
    {
        l=max(c[i],l);
        ans+=i-l;
    }
    cout<<ans<<endl;
}
