#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;

int main() {
	int n, m;
	scanf("%d%d", &n, &m);
	vector<int> invperm(n);
	for (int i = 0; i < n; i++) {
		int x;
		scanf("%d", &x);
		invperm[x-1] = i;
	}
	vector<int> foes(n, -1);
	for (int i = 0; i < m; i++) {
		int x, y;
		scanf("%d%d", &x, &y);
		x = invperm[x-1];
		y = invperm[y-1];
		if (x > y) {
			swap(x, y);
		}
		foes[y] = max(foes[y], x);
	}
	long long ans = 0;
	int mx = -1;
	for (int i = 0; i < n; i++) {
		mx = max(mx, foes[i]);
		ans += i-mx;
	}
	printf("%lld\n", ans);
	return 0;
}
