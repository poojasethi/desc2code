#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define fo(i,a,b) for(int i=a;i<=b;i++)
#define fd(i,a,b) for(int i=a;i>=b;i--)
#define maxn 300005
#define ll long long
using namespace std;

ll wz[maxn];

int n,m;

ll f[maxn];

int main(){
	scanf("%d%d",&n,&m);
	fo(i,1,n) {
		int x;
		scanf("%d",&x);
		wz[x]=i;
	}
	fo(i,1,m) {
		int x,y;
		scanf("%d%d",&x,&y);
		if (wz[x]<wz[y]) swap(x,y);
		f[wz[x]]=max(f[wz[x]],wz[y]+1);
	}
	ll ans=0;
	ll tmp=1;
	fo(i,1,n) {
		tmp=max(tmp,f[i]);
		ans+=(i-tmp+1);
	}
	cout<<ans;
	return 0;
}
