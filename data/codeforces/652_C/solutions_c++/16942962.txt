#include <bits/stdc++.h>
using namespace std;

const int kMaxN=400000;

vector<int> foe[kMaxN];
int p[kMaxN];
int pos[kMaxN];

int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%d",p+i);
		pos[p[i]]=i;
	}
	for(int i=1;i<=m;i++){
		int a,b;
		scanf("%d%d",&a,&b);
		foe[a].push_back(b);
		foe[b].push_back(a);
	}
	long long ans=0;
	for(int l=1,r=1;r<=n;r++){
		for(int i=0;i<foe[p[r]].size();i++){
			const int &f=foe[p[r]][i];
			if(pos[f]<r){
				l=max(l,pos[f]+1);
			}
		}
		ans+=r-l+1;
	}
	cout<<ans;
	//printf("%I64d",ans);
	return 0;
}
