#include<bits/stdc++.h>
using namespace std;
static int ans[1000001],ind[1000001],a[1000001];
int main(){
	int n,m;
	scanf("%d %d",&n,&m);
	for(int i=0;i<=n;i++)
		ans[i]=0;
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
		ind[a[i]]=i+1;
	}
	for(int i=0;i<m;i++){
		int x,y;
		scanf("%d %d",&x,&y);
		if(ind[y]<ind[x])
		   ans[x]=max(ans[x],ind[y]);
		   else
		   ans[y]=max(ans[y],ind[x]);	
	}
	long long ans1=0;
	int pr=0;
	for(int i=1;i<=n;i++){
	//	printf("i=%d %d\n",ind[i],ans[i]);
		pr=max(pr,ans[a[i-1]]);
		ans1+=ind[a[i-1]]-pr;
	}
	printf("%lld\n",ans1);
	return 0;
}
