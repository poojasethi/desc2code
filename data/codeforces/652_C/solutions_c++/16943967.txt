#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
int xu[311111],to[311111];
int main(){
	int n,m,temp,l,r;
	long long ans=0;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)	{
		scanf("%d",&temp);
		xu[temp]=i;
	}
	for(int i=1;i<=n;i++)	to[i]=n;
	for(int i=0;i<m;i++){
		scanf("%d%d",&l,&r);
		if(xu[l]>xu[r]){
			temp=l;
			l=r;
			r=temp;
		}
		to[xu[l]]=min(xu[r]-1,to[xu[l]]);
	}
	for(int i=n-1;i>=0;i--)	to[i]=min(to[i],to[i+1]);
	for(int i=1;i<=n;i++)	ans+=to[i]-i+1;
	printf("%I64d\n",ans);
	return 0;
}
