#include<cstdio>
#include<algorithm>
using namespace std;
struct seg{int l,r;}x[300010];
int n,m,i,j,now,a,b,c,p[300010];
long long ans;
bool cmp(seg a,seg b){return a.r<b.r;}
int main()
{
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++){scanf("%d",&a);p[a]=i;}
	for (i=1;i<=m;i++)
	{
		scanf("%d%d",&a,&b);
		a=p[a];b=p[b];
		if (a>b) {c=a;a=b;b=c;}
		x[i].l=a;x[i].r=b; 
	}
	sort(x+1,x+m+1,cmp);
	now=0;j=1;
	for (i=1;i<=n;i++)//ends with i
	{
		while (j<=m&&x[j].r==i)
		{
			if (x[j].l>now) now=x[j].l;
			j++;
		}
		ans+=i-now;
	}
	printf("%I64d\n",ans);
	return 0;
}
