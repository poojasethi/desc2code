#include <bits/stdc++.h>

using namespace std;

const int MAX = 3E5 + 2;

int dp[MAX], p[MAX], N, M;
long long ans;

int main(int argc, char **argv) {
   // freopen(argv[1], "r", stdin);
    
    scanf("%d %d", &N, &M);
    for (int i = 1, x; i <= N; ++i) {
        scanf("%d", &x);
        p[x] = i;
    }

    for (int i = 0, x, y; i < M; ++i) {
        scanf("%d %d", &x, &y);
        x = p[x]; y = p[y];
        if (x > y) swap(x, y);
        dp[y] = max(dp[y], x);
    }

    ans = 0;
    for (int i = 1; i <= N; ++i) {
        dp[i] = max(dp[i], dp[i - 1]);
        ans += i - dp[i];
    }

    cout << ans;
    return 0;
}
