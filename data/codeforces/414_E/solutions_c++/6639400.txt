#include <cmath>
#include <vector>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
#define maxn 200005
using namespace std;

namespace Splay
{
	struct node
	{
		int d,mi,ma,p,x;
		node*l,*r,*f;
		node(int _p=0,int _d=-1) {p=_p,d=_d,x=0,ma=-1e9,mi=1e9;}
		void renew(int t) {if (p!=-1) x+=t,d+=t,mi+=t,ma+=t;} 
		void down() {if (x) l->renew(x),r->renew(x),x=0;}
		void update()
		{
			ma=max(d,max(l->ma,r->ma));
			mi=min(d,min(l->mi,r->mi));
		}
	}*null,*root,t[maxn];
	void Clear() {null=new node,null->l=null->r=null->f=root=null,null->p=-1;}
	void zig(node*p)
	{
		node*x=p->f,*y=x->f;
		x->l=p->r,p->r->f=x,p->r=x,x->f=p,p->f=y,x->update();
		if (y!=null) if (x==y->l) y->l=p; else y->r=p;
	}
	void zag(node*p)
	{
		node*x=p->f,*y=x->f;
		x->r=p->l,p->l->f=x,p->l=x,x->f=p,p->f=y,x->update();
		if (y!=null) if (x==y->l) y->l=p; else y->r=p;
	}
	void splay(node*p,node*r=null)
	{
		static node*q[maxn]; int t=0;
		for (node*x=p; q[++t]=x,x->f!=r; x=x->f);
		for (;t;) q[t--]->down();
		for (node*x,*y; x=p->f,y=x->f,p->f!=r;)
		{
			if (y==r) if (p==x->l) zig(p); else zag(p);
			else if (x==y->l)
				if (p==x->l) zig(x),zig(p); else zag(p),zig(p);
			else
				if (p==x->r) zag(x),zag(p); else zig(p),zag(p);
		}
		p->update();
		if (r==null) root=p;
	}
	node*build(int L,int R,node*f)
	{
		if (L>R) return null;
		int mi=(L+R)>>1; node*p=&t[mi];
		p->l=build(L,mi-1,p),p->r=build(mi+1,R,p),p->f=f,p->update();
		return p;
	}
	node*Left(node*p) {for (;p->l!=null; p=p->l); return p;}
	node*Right(node*p) {for (;p->r!=null; p=p->r); return p;}
	node*Get(node*p,node*q)
	{
		splay(p),p=Right(p->l);
		splay(q),q=Left(q->r);
		splay(p),splay(q,p);
		return q->l;
	}
	int Dist(node*p,node*q)
	{
		if (p==q) return 0;
		splay(p),splay(q,p);
		int d1=p->d,d2=q->d,w;
		if (p->r==q) w=q->l->mi; else w=q->r->mi;
		return d1+d2-2*min(min(d1,d2),w-1);
	}
	int Query(node*p,int k)
	{
		p->down();
		if (p->l->mi<=k&&k<=p->l->ma) return Query(p->l,k);
		if (p->d==k) return p->p;
		return Query(p->r,k);
	}
};
using namespace Splay;
int S,N,M,L[maxn],R[maxn],head[maxn],lk[maxn],next[maxn],d[maxn];

void add(int u,int v,int t) {lk[t]=v,next[t]=head[u],head[u]=t;}

void read(int&x)
{
	char c;
	for (;!isdigit(c=getchar());) (x*=10)+=c-48;
	for (x=0; isdigit(c); c=getchar()) (x*=10)+=c-48;
}

void init()
{
	read(N),read(M);
	for (int i=1,x,s,t=0; i<=N; i++)
	{
		read(s);
		for (;s; s--) read(x),add(i,x,++t);
	}
}

void dfs(int u,int f)
{
	t[L[u]=++S]=node(u,d[u]);
	for (int i=head[u],v; v=lk[i],i; i=next[i]) if (v!=f) d[v]=d[u]+1,dfs(v,u);
	t[R[u]=++S]=node(u,d[u]);
}

void move(int x,int y)
{
	node*p,*q,*r;
	splay(&t[R[x]]),q=&t[L[Query(root->r,t[R[x]].d-y)]];
	p=Get(&t[L[x]],&t[R[x]]),p->renew(-y+1);
	p->f->l=null,p->f->update(),root->update();
	splay(q),splay(r=Left(q->r),q);
	r->l=p,p->f=r,r->update(),q->update();
}

void doit()
{
	Clear(),dfs(1,0),root=build(0,S+1,null);
	for (int i=1,t,x,y; i<=M; i++)
	{
		read(t),read(x);
		if (t==1) read(y),printf("%d\n",Dist(&::t[L[x]],&::t[L[y]]));
		else if (t==2) read(y),move(x,y);
		else printf("%d\n",Query(root,x));
	}
}

int main()
{
	init();
	doit();
	return 0;
}
