#define Ls Z[x][0]

using namespace std;
I N,M,R,num[X],Z[X][2],fa[X];
I D[X],U[X],T[X],V[X],S[X][2],W,P[X];
vector<int>dn[X];
void Ud(I x)
{
	T[x]=T[Ls]+T[Rs]+V[x];
	D[x]=min(T[Ls]+V[x],min(D[Ls],D[Rs]+T[Ls]+V[x]));
	U[x]=max(T[Ls]+V[x],max(U[Ls],U[Rs]+T[Ls]+V[x]));
}
void Ro(I x,I d)
{
	I t=fa[x];
	Z[t][d^1]=Z[x][d],fa[Z[x][d]]=t;
	if(fa[t]) Z[fa[t]][Z[fa[t]][1]==t]=x;
	fa[x]=fa[t],fa[t]=x,Z[x][d]=t,Ud(t),Ud(x);
}
void Sp(I x,I to)
{
	I t,d1,d2;
	while((t=fa[x])-to)
		if(fa[t]==to) Ro(x,Z[t][0]==x);
		E if((d1=(Z[fa[t]][0]==t))^(d2=(Z[t][0]==x)))
			Ro(x,d2),Ro(x,d1);
		E Ro(t,d1),Ro(x,d2);
	R=to?R:x;
}
I Last(I x){Sp(x,0);for(x=Ls;Rs;x=Rs);O x;}
I Next(I x){Sp(x,0);for(x=Rs;Ls;x=Ls);O x;}
I De(I l,I r)
{

	tl=Last(l),tr=Next(r);
	if(!(tl+tr)) {O r;}
	E if(!tl||!tr)

		Z[tr][f]=fa[Z[tr][f]]=0,Ud(tr);
		O t;
	}
	Sp(tl,0),Sp(tr,tl),t=Z[tr][0];
	Z[tr][0]=fa[Z[tr][0]]=0;
	Ud(tr),Ud(tl);
	O t;
} 
void In(I x,I y)
{
	if(!x) {R=y;O;}
	I t=Next(x);
	if(!t) Rs=y,fa[y]=x,Ud(x);
	E Sp(t,x),Z[t][0]=y,fa[y]=t,Ud(t),Ud(x);
	Sp(y,0);
}
I Ask(I x,I k)
{
	for(k++;;)if(D[Rs]+T[Ls]+V[x]<=k&&U[Rs]+T[Ls]+V[x]>=k)
		k-=T[Ls]+V[x],x=Rs;
	E if(T[Ls]+V[x]==k)
		{Sp(x,0);O x==S[num[x]][0]?num[x]:P[num[x]];}
	E x=Ls;
}
I Dis(I x,I y)
{
	x=S[x][0],y=S[y][0];
	if(x==y)O 0;
	Sp(y,0),Sp(x,y);
	if(x==Z[y][1])swap(x,y);
	Sp(y,0),Sp(x,y);
	I t=min(T[Ls]+V[x],min(D[Rs]+T[Ls]+V[x],T[x]+V[y]));
	O T[Ls]+T[x]+V[x]+V[y]-t*2;
}
void Tr(I x,I h)
{
	I t=x,par;
	x=S[x][0],Sp(x,0),par=Ask(Ls,T[Ls]+V[x]-h-1);
	P[t]=par,t=De(S[t][0],S[t][1]);
	In(Last(S[par][1]),t);
}
void Dfs(I x)
{
	S[x][0]=++W,num[W]=x;
	V[W]=T[W]=D[W]=U[W]=1,In(R,W);
	for(I i=0;i<dn[x].size();i++)
		P[dn[x][i]]=x,Dfs(dn[x][i]);
	S[x][1]=++W,num[W]=x;
	V[W]=T[W]=D[W]=U[W]=-1,In(R,W);
}
