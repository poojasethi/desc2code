#include <cstdio>

int f[100005];
int arr[100005],n;
int lc[10005];

int add(int x,int a)
{
    for(;x<=n;x+=(x&-x)) f[x] += a;
    return 0;
}

int sum(int x)
{
    int s;
    for(s=0;x>0;x-=(x&-x)) s += f[x];
    return s;
}

int gen(int x)
{
    if(x>=10000) return 0;
    lc[x] = 1;
    gen(x*10+4);
    gen(x*10+7);
    return 0;
}

int main()
{
    int i;
    int m;
    char cmd[100];
    int l,r,d;
    scanf("%d %d",&n,&m);
    for(i=0;i<=10000;i++) lc[i] = 0;
    gen(4);
    gen(7);
    for(i=0;i<=n;i++) f[i] = 0;
    for(i=1;i<=n;i++) {
        scanf("%d",arr+i);
        add(i,lc[arr[i]]);
    }
    while(m--) {
        scanf("%s %d %d",cmd,&l,&r);
        if(cmd[0]=='a') {
            scanf("%d",&d);
            for(;l<=r;l++) {
                if(lc[arr[l]]) add(l,-1);
                arr[l] += d;
                if(lc[arr[l]]) add(l,1);
            }
        } else printf("%d\n",sum(r)-sum(l-1));
    }
    return 0;
}
