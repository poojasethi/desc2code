#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std ;

const int MAXN = 100001 ;


int c[MAXN] , n , a[MAXN];
bool v[MAXN] ;
char str[MAXN] ;

bool calc(int x){
    while(x){
        int t=x%10;
        if (t!=4&&t!=7) return false;
        x/=10 ;
    }
    return true ;
}

void add(int x,int v){
    while (x<=n){
        c[x]+=v;
        x+=(x & (-x));
    }
}

int query(int x){
    int ret=0;
    while (x>0){
        ret+=c[x];
        x-=(x & (-x));
    }
    return ret ;
}

int main(){
    int q ;
    scanf("%d%d",&n,&q);
    for (int i=1;i<=10000;i++)
    {
        v[i]=0;
        if (calc(i))
            v[i]=1;
    }
    for (int i=1;i<=n;i++){
        scanf("%d",&a[i]) ;
        if (v[a[i]])
            add(i,1) ;
    }
    while (q--){
        scanf("%s",str);
        if (str[0]=='a'){
            int l , r , vv ;
            scanf("%d%d%d",&l,&r,&vv);
            for (int i=l;i<=r;i++)
            {
                if (v[a[i]])
                    add(i,-1) ;
                a[i]+=vv;
                if (v[a[i]])
                    add(i,1);
            }
        }else{
            int l , r ;
            scanf("%d%d",&l,&r);
            printf("%d\n",query(r)-query(l-1));
        }
    }
    return 0 ;
}
