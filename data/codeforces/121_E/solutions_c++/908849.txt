#include <iostream>
#include <cstdio>
#include <string>
#include <sstream>
using namespace std;

int t[101000],a[101000],b[101000];
bool f[101000];
int n,m;

bool judge(int x)
{
    while (x) { if (x%10!=7 && x%10!=4) return false;x=x/10; }
    return true;
}

int query(int x)
{
    int ans=0;
    while (x)
    {
        ans+=t[x];
		x-=x&(-x);
    }
    return ans;
}

void insert(int x,int y)
{
    while (x<=n)
    {
        t[x]+=y;x+=x&(-x);
    }
}

int main()
{
  //  freopen("in.txt","r",stdin);
    for (int i=1;i<=10000;i++) f[i]=judge(i);
    cin>>n>>m;
    string ss;stringstream s;int x,y,z;
    for (int i=1;i<=n;i++) { cin>>a[i];if (f[a[i]]){b[i]=1;insert(i,1); } }
    getline(cin,ss);
    for (m=m;m;m--)
    {
        s.clear();s.str("");
        getline(cin,ss);s<<ss;
        s>>ss;
        s>>x>>y;
        if (ss[0]=='c')
        {
            cout<<query(y)-query(x-1)<<endl;
        }else
        {
            s>>z;
            for (int i=x;i<=y;i++)
            {
                if (b[i]==1) { insert(i,-1);b[i]=0; }
                a[i]+=z;
                if (f[a[i]]) { insert(i,1);b[i]=1; }
            }
        }
    }
    return 0;
}
