#include<cstdio>

#define FOR(i, a, b) for(int i(a); i < b; i++)
#define REP(i, n) FOR(i, 0, n)

#define MAX_N 100005

int n, q;
bool lucky[MAX_N];
int a[MAX_N];
int f[MAX_N];

inline void add(int i, int x)
{
    for(; i <= n; i += (i & -i))
        f[i] += x;
}

inline int sum(int i, int j)
{
    int ret = 0;
    for(; j > 0; j -= (j & -j))
        ret += f[j];
    for(i--; i > 0; i -= (i & -i))
        ret -= f[i];
    return ret;
}

int main()
{
    lucky[0] = true;
    FOR(i, 1, 10000)
        lucky[i] = (i % 10 == 4 or i % 10 == 7) and lucky[i / 10];
    scanf("%d %d", &n, &q);
    FOR(i, 1, n + 1)
    {
        scanf("%d", &a[i]);
        if(lucky[a[i]])
            add(i, 1);
    }
    while(q--)
    {
        int l, r, d;
        char c[10];
        scanf("%s %d %d", c, &l, &r);
        if(c[0] == 'c')
            printf("%d\n", sum(l, r));
        else
        {
            scanf("%d", &d);
            for(; l <= r; l++)
            {
                if(lucky[a[l]] and not lucky[a[l] + d])
                    add(l, -1);
                else if(not lucky[a[l]] and lucky[a[l] + d])
                    add(l, 1);
                a[l] += d;
            }
        }
    }
}
