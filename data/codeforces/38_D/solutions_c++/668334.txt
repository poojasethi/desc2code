#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
    int n,x1[100],y1[100],x2[100],y2[100],w[100];
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d%d%d%d",x1+i,y1+i,x2+i,y2+i);
        if(x1[i]>x2[i]) swap(x1[i],x2[i]);
        if(y1[i]>y2[i]) swap(y1[i],y2[i]);
        w[i]=(x2[i]-x1[i])*(x2[i]-x1[i])*(x2[i]-x1[i]);
        for(int j=0;j<i;j++){
            long long X1=0,X2=0,Y1=0,Y2=0;
            for(int k=j+1;k<=i;k++){
                X1+=((x1[k]+x2[k])-x1[j]*2)*w[k];
                X2+=(x2[j]*2-(x1[k]+x2[k]))*w[k];
                Y1+=((y1[k]+y2[k])-y1[j]*2)*w[k];
                Y2+=(y2[j]*2-(y1[k]+y2[k]))*w[k];
                if(X1<0 || X2<0 || Y1<0 || Y2<0)
                    return !printf("%d\n",i);
            }
        }
    }
    printf("%d\n",n);
}
