#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;

#define ll long long

const double eps=0.000000001;

double x1[150],x2[150],y1[150],y2[150];

int main()
{
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    int n;
    cin>>n;
    double mx,my,m,m1,mx1,my1;
    for (int i=1;i<=n;i++) {
        cin>>x1[i]>>y1[i]>>x2[i]>>y2[i];
        if (x1[i]-x2[i]>eps) swap(x1[i],x2[i]);
        if (y1[i]-y2[i]>eps) swap(y1[i],y2[i]);
    }
    for (int i=2;i<=n;i++) {
        mx=(x1[i]+x2[i])/2;
        my=(y1[i]+y2[i])/2;
        m=(x2[i]-x1[i])*(x2[i]-x1[i])*(x2[i]-x1[i]);
        for (int j=i-1;j>0;j--) {
            if ((mx-x2[j]>eps)||(my-y2[j]>eps)||(x1[j]-mx>eps)||(y1[j]-my>eps)) {
                cout<<i-1;
                return 0;
            }
            m1=(x2[j]-x1[j])*(x2[j]-x1[j])*(x2[j]-x1[j]);
            mx1=(x1[j]+x2[j])/2;
            my1=(y1[j]+y2[j])/2;
            mx+=(m1/(m+m1))*(mx1-mx);
            my+=(m1/(m+m1))*(my1-my);
            m+=m1;
        }
    }
    cout<<n;
    return 0;
}
