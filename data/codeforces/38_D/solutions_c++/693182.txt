using namespace std;
#define li       	long long int
#define rep(i,to)	for(li i=0;i<((li)(to));i++)
#define pb       	push_back
#define sz(v)    	((li)v.size())
#define bit(n)   	(1ll<<(li)(n))

struct point{
	li x0,y0,x1,y1;
};

int main(){
	li n;
	cin>>n;
	vector<point> vec(n);
	rep(i,n) cin>>vec[i].x0>>vec[i].y0>>vec[i].x1>>vec[i].y1;
	
	li res=n;
	rep(i,n){
		li x=0,w=0;
		for(li j=i+1;j<n;j++){
			li t=abs(vec[j].x0-vec[j].x1);
			x+=t*t*t*(vec[j].x0+vec[j].x1);
			w+=t*t*t;
			if(max(vec[i].x0,vec[i].x1)*2*w<x) res=min(res,j);
			if(x<min(vec[i].x0,vec[i].x1)*2*w) res=min(res,j);
		}
	}
	rep(i,n){
		li y=0,w=0;
		for(li j=i+1;j<n;j++){
			li t=abs(vec[j].y0-vec[j].y1);
			y+=t*t*t*(vec[j].y0+vec[j].y1);
			w+=t*t*t;
			if(max(vec[i].y0,vec[i].y1)*2*w<y) res=min(res,j);
			if(y<min(vec[i].y0,vec[i].y1)*2*w) res=min(res,j);
		}
	}
	cout<<res<<endl;
}
