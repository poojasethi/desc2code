#include <iostream>

using namespace std;

const int Gmax = 110;

int n;
int x1[Gmax];
int x2[Gmax];
int y1[Gmax];
int y2[Gmax];

void INPUT();

const double eps = 1e-6;

int sign(double k){
	if (k < -eps)
		return -1;
	if (k > eps)
		return 1;
	return 0;
}

bool bad(int i, double X, double Y, int w){
	//	if not in;	bye!
	if (sign(X-x1[i]) < 0)
		return 1;
	if (sign(X-x2[i]) > 0)
		return 1;
	if (sign(Y-y1[i]) < 0)
		return 1;
	if (sign(Y-y2[i]) > 0)
		return 1;

	if(i <= 1)
		return 0;

	int a = (x2[i]-x1[i]);
	a *= a*a;
	X = X*w + (double(x1[i]+x2[i]))/2*a;
	X /= (w+a);

	Y = Y*w + (double(y1[i]+y2[i]))/2*a;
	Y /= (w+a);

	w += a;

	return bad(i-1, X, Y, w);
}

int main(){
	INPUT();

	int ans = 1;
	for (int i = 2; i <= n; i++){
		if (bad(i, x1[i]+1, y1[i]+1, 0))
			break;
		ans++;
	}

	cout << ans;

	return 0;
}

void INPUT(){
	cin >> n;
	for (int i = 1; i <= n; i++){
		cin >> x1[i] >> y1[i] >> x2[i] >> y2[i];
		if (x1[i] > x2[i])
			swap(x1[i], x2[i]);
		if (y1[i] > y2[i])
			swap(y1[i], y2[i]);
	}
}
