#include <stdio.h>
#include <string>
#include <string.h>

#include <queue>
#include <set>
#include <stack>
#include <algorithm>
#define FOR(A, B) for(A=0;A<B;A++)
#define MAXN 111

using namespace std;

int main()
{
	int i, j, k, n, m;
	scanf("%d", &n);
	int X1[101], Y1[101], X2[101], Y2[101];
	FOR(i, n) {
		scanf("%d %d %d %d", X1+i, Y1+i, X2+i, Y2+i);
		if(X1[i] > X2[i]) swap(X1[i], X2[i]);
		if(Y1[i] > Y2[i]) swap(Y1[i], Y2[i]);
	}
	int weight[101];
	double center_x[101], center_y[101];
	for(i = 0; i < n; i++) {
		weight[i] = (X2[i]-X1[i])*(X2[i]-X1[i])*(X2[i]-X1[i]);
		center_x[i] = (X1[i]+X2[i])/2.0;
		center_y[i] = (Y1[i]+Y2[i])/2.0;
		double X = 0, Y = 0, sum = 0;
		for(j = i; j > 0; j--) {
			X += weight[j]*center_x[j];
			Y += weight[j]*center_y[j];
			sum += weight[j];
			double x = X/sum, y = Y/sum;
			if(x < X1[j-1] || x > X2[j-1]) {
				printf("%d\n", i);
				return 0;
			}
			if(y < Y1[j-1] || y > Y2[j-1]) {
				printf("%d\n", i);
				return 0;
			}
		}
	}
	printf("%d\n", n);
	return 0;
}
