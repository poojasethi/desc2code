#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <cassert>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 101
#define eps 1e-10

int n,i,j;
double x,y;
int X1[lim],Y1[lim],X2[lim],Y2[lim],A[lim];

void center(int i,int j,double &x,double &y)
{
  int s=0;

  x=y=0;
	for(;i<=j;i++)
	{
		x+=A[i]*A[i]*A[i]*(X1[i]+A[i]/2.);
		y+=A[i]*A[i]*A[i]*(Y1[i]+A[i]/2.);
		s+=A[i]*A[i]*A[i];
	}
	x/=s;
	y/=s;
}
int main()
{
  scanf("%d",&n);
  for(i=0;i<n;i++)
	{
	  scanf("%d%d%d%d",&X1[i],&Y1[i],&X2[i],&Y2[i]);
	  if(X2[i]<X1[i])
		  swap(X2[i],X1[i]);
	  if(Y2[i]<Y1[i])
		  swap(Y2[i],Y1[i]);
		A[i]=X2[i]-X1[i];
	}
	for(i=0;i<n;i++)
	  assert(X2[i]-X1[i]==Y2[i]-Y1[i]);
	for(i=1;i<n;i++)
	{
	  for(j=1;j<=i;j++)
		{
		  center(j,i,x,y);
			if(x+eps<X1[j-1] || x-eps>X2[j-1] || y+eps<Y1[j-1] || y-eps>Y2[j-1])
			  break;
    }
	  if(j<=i)
		  break;
	}
	printf("%d",i);
	return 0;
}
