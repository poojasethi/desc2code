#include<iostream>
#include<algorithm>
using namespace std;

double v[100],ix[100],iy[100];
int ax[100],ay[100],bx[100],by[100];
int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++) {
		cin>>ax[i]>>ay[i]>>bx[i]>>by[i];
		if(bx[i]<ax[i]) swap(ax[i],bx[i]);
		if(by[i]<ay[i]) swap(ay[i],by[i]);
		int a=bx[i]-ax[i];
		v[i]=a*a*a;
		ix[i]=v[i]*(ax[i]+bx[i])/2.;
		iy[i]=v[i]*(ay[i]+by[i])/2.;
		for(int j=0;j<i;j++) {
			double sx=0,sy=0,s=0;
			for(int k=j+1;k<=i;k++) {
				sx+=ix[k];
				sy+=iy[k];
				s+=v[k];
			}
			sx/=s;
			sy/=s;
			if(sx<ax[j]-1e-9 || sx>bx[j]+1e-9 || sy<ay[j]-1e-9 || sy>by[j]+1e-9) {
				cout<<i;
				return 0;
			}
		}
	}
	cout<<n;
}
