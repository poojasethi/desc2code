#include <iostream>
#include <vector>
#include <map>
#include <utility>
#include <algorithm>
#include <string>
#include <set>
#include <stdio.h>
#include <sstream>
#include <math.h>
#include <list>

using namespace std;

int main() {
    cout << fixed;
    cout.precision(16);

    int n;
    cin >> n;
    vector<long long> x(n), y(n), a(n);
    for (int i = 0; i < n; i++) {
        long long x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        x[i] = min(x1, x2);
        y[i] = min(y1, y2);
        a[i] = max(x1, x2) - x[i];
    }
    
    int k = 0;
    for (; k < n; k++) {
        long long sumx = 0, sumy=0, summ=0;
        for (int i = k; i > 0; --i) {
            sumx += (2*x[i] + a[i])  * a[i]*a[i]*a[i];
            sumy += (2*y[i]+a[i]) * a[i]*a[i]*a[i];
            summ += a[i]*a[i]*a[i];
//            cerr << sumx << " " << sumy << " " << summ << endl;
            if (sumx < 2 *summ * x[i-1] || sumx > 2*summ * (x[i-1] + a[i-1]) ||
                sumy < 2*summ * y[i-1] || sumy > 2*summ * (y[i-1] + a[i-1])
                    ) {
                cout << k << endl;
                return 0;
            }
        }
    }
    
    cout << k << endl;
    
    return 0;
}
