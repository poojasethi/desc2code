#include <stdio.h>

int n;

const double eps=1e-6;
bool out(double x, double y, double xa, double ya, double xb, double yb) {
	return !(x >= xa-eps && x<=xb+eps && y >= ya-eps && y <= yb+eps);
}

void go() {
	double xa[105], ya[105], xb[105], yb[105], w[105];
	for(int i=1;i<=n;i++) {
		scanf("%lf%lf%lf%lf", &xa[i], &ya[i], &xb[i], &yb[i]);
		if(xa[i] > xb[i]) { double t=xa[i];xa[i]=xb[i];xb[i]=t;}
		if(ya[i] > yb[i]) { double t=ya[i];ya[i]=yb[i];yb[i]=t;}
		w[i] = (xa[i]-xb[i])*(xa[i]-xb[i])*(xa[i]-xb[i]);
		if(w[i]<0) w[i]=-w[i];
		double wsum=0, xsum=0, ysum=0;
		for(int j=i;j>1;j--) {
			xsum += w[j] * (xa[j]+xb[j])/2.0;
			ysum += w[j] * (ya[j]+yb[j])/2.0;
			wsum += w[j];
			if (out(xsum/wsum, ysum/wsum, xa[j-1], ya[j-1], xb[j-1], yb[j-1]))
			{
				printf("%d\n", i-1);
				return;
			}
		}
	}
	printf("%d\n", n);
}

int main(void) {
	scanf("%d", &n);
	go();
	return 0;
}
