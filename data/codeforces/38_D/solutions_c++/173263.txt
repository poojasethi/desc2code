#include<iostream>
#include<algorithm>
using namespace std;

const int maxn = 110;
double x1[maxn], x2[maxn], y1[maxn], y2[maxn], mx[maxn], my[maxn], r[maxn];
int n;

bool stable(int q){
	double ax = 0, ay = 0, w = 0;
	for(int k = q; k > 0 ; k--){
		ax = ( ax * w + mx[k] * r[k] ) / (r[k]+w);
		ay = ( ay * w + my[k] * r[k] ) / (r[k]+w);	
		w += r[k];
		if(ax < x1[k-1] || ax > x2[k-1] || ay < y1[k-1] || ay > y2[k-1])
			return false;
	}
	return true;
}

int main(){
	cin >> n;
	for(int i = 0; i < n; i++){
		cin >> x1[i] >> y1[i] >> x2[i] >> y2[i];	
		if(x1[i] > x2[i])
			swap(x1[i], x2[i]);
		if(y1[i] > y2[i])
			swap(y1[i], y2[i]);
		mx[i] = (x1[i] + x2[i])/2.0;
		my[i] = (y1[i] + y2[i])/2.0;
		r[i] = (x2[i] - x1[i])*(x2[i] - x1[i])*(x2[i] - x1[i]);
	}
	
	for(int i = 1; i < n; i++)
		if(stable(i) == false){
			cout << i << endl;
			return 0;
		}
	cout << n << endl;
	return 0;
}
