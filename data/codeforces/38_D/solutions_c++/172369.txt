#include<algorithm>
#include<iostream>
#include<iomanip>
#include<cstring>
#include<string>
using namespace std;

int n;
double weight[101];
double x1[101] , x2[101] , y1[101] , y2[101];
double center_x[101] , center_y[101];

int solveit()
{
	int ret = n;
	for(int i = 1 ; i <= n ; i++)
	{
		cin>>x1[i]>>y1[i]>>x2[i]>>y2[i];
		if(x1[i] > x2[i])
			swap(x1[i] , x2[i]);
		if(y1[i] > y2[i])
			swap(y1[i] , y2[i]);
		weight[i] = (x1[i] - x2[i]) * (x1[i] - x2[i]) * (x1[i] - x2[i]);
		center_x[i] = (x1[i] + x2[i]) / 2.0;
		center_y[i] = (y1[i] + y2[i]) / 2.0;
		double X = 0 , Y = 0 , S = 0;
		for(int j = i ; j >= 2 ; j--)
		{
			S += weight[j];
			X += weight[j] * center_x[j];
			Y += weight[j] * center_y[j];
			double x = X / S;
			double y = Y / S;
			if(x < x1[j - 1] || x > x2[j - 1])
				ret = min(ret , i - 1);
			if(y < y1[j - 1] || y > y2[j - 1])
				ret = min(ret , i - 1);
		}
	}
	return ret;
}

int main()
{
	ios::sync_with_stdio(false);
	while(cin>>n)
	{
		cout<<solveit()<<endl;		
	}
	return 0;
}
