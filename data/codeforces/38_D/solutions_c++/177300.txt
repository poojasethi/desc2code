#include <algorithm>
#include <cstring>
#include <cstdio>
#define cube(x) ((x) * (x) * (x))

using namespace std;

double x1[101], y1[101], x2[101], y2[101];

bool Check(int l, int r)
  {
    double g_x = 0, g_y = 0, sum = 0, tmp;
    for (int i = l + 1; i <= r; i ++)
      {
        tmp = cube(x2[i] - x1[i]), sum += tmp;
        g_x += (x1[i] + x2[i]) / 2 * tmp;
        g_y += (y1[i] + y2[i]) / 2 * tmp;
      }
    g_x /= sum, g_y /= sum;
    return x1[l] <= g_x && g_x <= x2[l] && y1[l] <= g_y && g_y <= y2[l];
  }

int main()
  {
    int n, i, j;
    
    //freopen("input.txt", "r", stdin);
    //freopen("output.txt", "w", stdout);
    
    scanf("%d", &n);
    for (i = 1; i <= n; i ++)
      {
        scanf("%lf%lf", &x1[i], &y1[i]);
        scanf("%lf%lf", &x2[i], &y2[i]);
        if (x1[i] > x2[i]) swap(x1[i], x2[i]);
        if (y1[i] > y2[i]) swap(y1[i], y2[i]);
        for (j = 1; j < i; j ++)
          if (! Check(j, i)) break;
        if (j < i) break;
      }
    
    printf("%d\n", i - 1);
    
    return 0;
  }
