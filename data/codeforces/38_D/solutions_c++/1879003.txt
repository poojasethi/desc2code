#include <iostream>
#include <cstdio>
#include <limits.h>
#include <cmath>
#include <cstdlib>
#include <cstring>
using namespace std;
double x[105], y[105], w[105], r[105];
int main()
{
    int n, x1, y1, x2 ,y2, i, j;
    bool f, all;
    double w0, x0, y0;
    cin>>n;
    for(i = 1; i <= n; i++)
    {
        cin>>x1>>y1>>x2>>y2;
        r[i] = abs(x2 - x1) / 2.0;
        w[i] = abs(x2 - x1) * abs(x2 - x1) * abs(x2 - x1);
        x[i] = (x1 + x2) / 2;
        y[i] = (y1 + y2) / 2;
    }
    all = true;
    for(i = 1; i <= n; i++)
    {
        w0 = x0 = y0 = 0.0;
        f = true;
        for(j = i; j >= 2; j--)
        {
            x0 += (w[j] / (w[j] + w0)) * (x[j] - x0);
            y0 += (w[j] / (w[j] + w0)) * (y[j] - y0);
            w0 += w[j];
            if( fabs(x0 - x[j-1]) > r[j-1] || fabs(y0 - y[j-1]) > r[j-1])
            {
                f = false;
                break;
            }
        }
        if(!f)
        {
            cout<< i - 1<<endl;
            all = false;
            break;
        }
    }
    if(all)
        cout<<n<<endl;
    return 0;
}
