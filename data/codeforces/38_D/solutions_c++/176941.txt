#include<iostream>
using namespace std;
const int MN=110;
const double eps=1e-7;
int a[MN][4];
int r[MN],n;
double gm[2];
bool in(int pt){
	if( a[pt][0]<=gm[0]+eps && gm[0]<=a[pt][2]+eps && a[pt][1]<=gm[1]+eps && gm[1]<=a[pt][3]+eps){
		return 1;
	}else{
		return 0;
	}
}
void f(int p,int q){
	gm[0]=0;
	gm[1]=0;
	double hj=0;
	for(int i=p;i<q;i++){
		double nm[2];
		nm[0]=a[i][0]+a[i][2];
		nm[0]/=2;
		nm[1]=a[i][1]+a[i][3];
		nm[1]/=2;
		double Hj=max(a[i][0]-a[i][2],a[i][2]-a[i][0]);
		Hj=Hj*Hj*Hj;
		gm[0]=gm[0]*hj+nm[0]*Hj;
		gm[0]/=Hj+hj;
		gm[1]=gm[1]*hj+nm[1]*Hj;
		gm[1]/=Hj+hj;
		hj+=Hj;
	}
}
int main(){
	cin>>n;
	double ms[2];
	ms[0]=0;
	ms[1]=0;
	double hz=0;
	for(int i=0;i<n;i++){
		cin>>a[i][0]>>a[i][1]>>a[i][2]>>a[i][3];
		if(a[i][0]>a[i][2]){
			swap(a[i][0],a[i][2]);
		}if(a[i][1]>a[i][3]){
			swap(a[i][1],a[i][3]);
		}
	}
	int Max=1;
	for(int i=2;i<=n;i++){
		bool t=1;
		for(int j=1;j<i;j++){
			f(j,i);
			if(!in(j-1)){
				t=0;
				break;
			}
		}
		if(!t){
			cout<<i-1<<endl;
			return 0;
		}
	}
	cout<<n<<endl;
	return 0;
}
