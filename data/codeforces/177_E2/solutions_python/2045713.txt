#include <stdio.h>

typedef long long ll;

const int MAXN = 1e4+10 ; 
const ll inf = 1e18 ; 

struct city{
	
	ll a,b;

	void inp(){
		
		scanf(" %I64d %I64d",&a,&b);
	
	}
	
}ar[MAXN];

int N;
ll C;

void read(){
	
	int i;
	
	scanf(" %d %I64d",&N,&C);
	
	for( i=1 ; i<=N ; i++ ) 
		ar[i].inp();
		
}

ll calc(ll x){
	
	int i;	
	ll res=N;
	
	for( i=1 ; i<=N ; i++ ){	
	
		if((ar[i].a*x)/ar[i].b < 0)	return C+1LL;
	
		res+=(ar[i].a*x)/ar[i].b;
		
	}	
	return res;	
		
}

ll searchleft(){
	
	ll l=0,r=inf,m;
	
	while( l+1 < r ){
		
		m=(l+r)/2;
		
		if(calc(m) >= C)	r=m;	
		else l=m;
	
	}
	
	return l;
	
}

ll searchright(){
	
	ll l=0,r=inf,m;
	
	while( l+1 < r ){
		
		m=(l+r)/2;
		
		if(calc(m) <= C)	l=m;
		else r=m;
	
	} 
	
	return r;
	
}

int main(){

#ifndef ONLINE_JUDGE

	freopen("f.out","w",stdout);
#endif
	
	read();

	ll min,max;
	
	max=searchright();
	min=searchleft();	
	
	//printf("%lld %lld\n",max,min);
	
	if(calc(1)==C && calc(inf)==C)	puts("-1");
	else 	printf("%I64d\n",max-min-1);
		
	return 0;

}
