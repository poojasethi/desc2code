#include<cstdio>
#include<iostream>
#include<cmath>
using namespace std;
#define SUP 0xfffffffffffffffll
#define MAXN 10010
#define ll long long

int N,C;
int a[MAXN],b[MAXN];
bool zf=0;

int check(ll k)
{
    ll t=0;
    for(int i=1;i<=N;i++)
    {
        t+=k*a[i]/b[i]+1;
        if(t<0) return 1;
        if(t>C) return 1;
    }
    if(t<C) return -1;
    return 0;
}
int main()
{
    cin>>N>>C;
    for(int i=1;i<=N;i++)
    {
        scanf("%d%d",a+i,b+i);
        if(a[i]) zf=1;
    }
    ll l=1,r=SUP,mid,inf=-1,sup=-1;
    while(l<=r)
    {
        mid=(l+r)>>1;
        if(check(mid)>0) r=mid-1;
        else if(check(mid)<0) l=mid+1;
        else r=mid-1,inf=mid;
    }
    l=1,r=SUP;
    while(l<=r)
    {
        mid=(l+r)>>1;
        if(check(mid)>0) r=mid-1;
        else if(check(mid)<0) l=mid+1;
        else l=mid+1,sup=mid;
    }
    if(sup==-1&&inf==-1) cout<<0<<endl;
    else 
    {
        if(zf==0) cout<<-1<<endl;
        else
        {
            if(inf==0) inf++;
            cout<<sup-inf+1<<endl;
        }
    }
    return 0;
}
