#include <iostream>
#include <cstdio>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
typedef pair <int, int> pii;
typedef vector <int> vi;
#define ll long long
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
const int N = 1e4 + 10;
const ll INF = 2e18;
int n;
ll a[N], b[N], c;
bool flag = true;
int len(ll x)
{
  int res = 0;
  while(x > 1)
  {
    x /= 10;
    res++;
  }
  return res;
}
int check(ll x)
{
  ll res = 0;
  for(int i = 0; i < n; i++)
  {
    if (len(a[i]) + len(x) >= 19)
      return 1;
    res += a[i] * x / b[i];
    if (res > c - n)
      return 1;
  }
  if (res < c - n)
    return -1;
  return 0;
}
int main()
{
  cin >> n >> c;
  for(int i = 0; i < n; i++)
  {
    cin >> a[i] >> b[i];
    flag &= a[i] == 0;
  }
  if (c < n)
    return cout << 0, 0;
  if (flag && c == n)
    return cout << -1, 0;
  ll l = 1, r = INF;
  while(l < r)
  {
    ll m = (l + r) / 2;
    if (check(m) == -1)
      l = m + 1;
    else
      r = m;
  }
  ll lpos = l;
  l = 1, r = INF;
  while(l < r)
  {
    ll m = (l + r) / 2;
    if (check(m) != 1)
      l = m + 1;
    else
      r = m;
  }
  cout << l - lpos;
  return 0;
}
