#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define N 220000
#define int64 long long
using namespace std;
int64 l,r,mid,ans;
int n,c,a[N],b[N],i;
int64 check(int64 mid){
    int64 res=0;
    for(int i=1;i<=n;++i){
        if(a[i]!=0 && mid/b[i]>c/a[i])return 1e10;
        res+=mid*a[i]/b[i];
    }
    return res;
}
int main(){
    scanf("%d%d",&n,&c);
    c-=n;
    for(i=1;i<=n;++i)scanf("%d%d",&a[i],&b[i]);
    l=1; r=1e18;
    while(l<=r){
        mid=l+r>>1;
        if(check(mid)>=c)r=mid-1;
        else l=mid+1;
    }
    ans=-(r+1);
    l=1; r=1e18;
    while(l<=r){
        mid=l+r>>1;
        if(check(mid)<=c)l=mid+1;
        else r=mid-1;
    }
    ans+=(l-1);
    printf("%I64d\n",ans>=(int64)1e18-1?-1:ans+1);
}
