#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <queue>
#define int64 long long
#define ld long double 
using namespace std;

const int64 Maxn = ((int64)1 << 60);
int n;
int64 a[100100],b[100100],whe1,C;

int64 judge(int64 mid)
{
    int64 tot = 0;
    for (int i = 1;i <= n;i++)  {
        tot += (int64)((ld)a[i] * ((ld)1.0*mid/b[i]));
        if (tot > C)  return tot;
    }
    if (tot < 0)   return Maxn;
    return tot;
}

int main()
{
    cin >> n >> C;  C -= n;
    for (int i = 1;i <= n;i++)   scanf("%I64d %I64d",&a[i],&b[i]);
    int64 l = 0,r = ((int64)1 << 60);
    while (l < r)
        {
            int64 mid = (l + r + 1) >> 1;
            if (judge(mid) <= C)   l = mid;
            else r = mid - 1;
        }
    whe1 = l;
    l = 0,r = ((int64)1 << 60);
    while (l < r)
        {
            int64 mid = (l + r + 1) >> 1;
            if (judge(mid) >= C)   r = mid - 1;
            else l = mid;
        }
    if (whe1 - l == ((int64)1 << 60))  printf("-1");
    else 
        cout << whe1 - l;
}
