#include<iostream>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

typedef long long ll;

const ll INF=1LL<<61;

int n;
ll tar,a[10000],b[10000];

ll f(ll x){
	ll res=0;
	rep(i,n){
		if((long double)a[i]*x>1e18 || res>tar) return INF; // to avoid overflow
		res+=1+a[i]*x/b[i];
	}
	return res;
}

int main(){
	cin>>n>>tar;
	rep(i,n) cin>>a[i]>>b[i];

	if(count(a,a+n,0)==n){
		cout<<(n==tar?-1:0)<<endl;
		return 0;
	}

	ll lo,hi;
	for(lo=1,hi=INF;lo<hi;){
		ll mi=(lo+hi)/2;
		ll spend=f(mi);
		if(spend<tar) lo=mi+1; else hi=mi;
	}
	ll lb=lo;

	for(lo=1,hi=INF;lo<hi;){
		ll mi=(lo+hi+1)/2;
		ll spend=f(mi);
		if(spend>tar) hi=mi-1; else lo=mi;
	}
	ll ub=lo;

	cout<<(f(lo)==tar?ub-lb+1:0)<<endl;

	return 0;
}
