#include<iostream>

using namespace std;

#define int long long
const int MAXN = 10004;
int a[MAXN], b[MAXN];
int n, c;

int check(int x)
{
	int cnt = 0;
	for(int i = 0; i < n; i++)
	{
		double t = double(x) * double(a[i]) / double(b[i]) + 1;
		if(t > 1e10)
			return 1e10;
		cnt += 1ll * t;
	}
	return cnt;
}

main()
{
	cin >> n >> c;
	for(int i = 0; i < n; i++)
		cin >> a[i] >> b[i];
	int lo = 0, hi = 2e18;
	while(lo < hi - 1)
	{
		int mid = (lo + hi) / 2;
		if(check(mid) < c)
			lo = mid;
		else
			hi = mid;
	}
	int ans = lo;
	lo = 0, hi = 2e18;
	while(lo < hi - 1)
	{
		int mid = (lo + hi) / 2;
		if(check(mid) <= c)
			lo = mid;
		else
			hi = mid;
	}
	if(lo - ans > 1e15)
		cout << -1 << endl;
	else
		cout << lo - ans << endl;
	return 0;
}
