#include<stdio.h>
#define N 10005
using namespace std;
typedef long long LL;
const double EPS=1e-8;
int a[N],b[N],n,c,i,flag;
LL two[N],L,R;
int check(LL x){
	double now=0;LL ret=0;
	for (int i=1;i<=n;i++){
		now+=1.*a[i]*x/b[i];
		if (now>c+1) return c+1;
		ret+=(LL)(1.*a[i]*x/b[i]+EPS);now=ret;
  }return ret;
}
int main(){
	scanf("%d%d",&n,&c);
	if ((c-=n)<0) return puts("0"),0;
	for (flag=i=1;i<=n;i++)
		scanf("%d%d",&a[i],&b[i]),flag&=a[i]==0;
	if (flag) return printf("%d\n",c==0?-1:0),0;
	for (two[0]=1,i=1;i<=60;i++) 
		two[i]=two[i-1]<<1ll;
	L=0;for (i=60;i>=0;i--)
		if (check(L+two[i])<c) L+=two[i];L++;
	R=two[60];
	for (i=60;i>=0;i--)
		if (check(R-two[i])>c) R-=two[i];R--;
	if (L>R) return puts("0"),0;
	printf("%I64d\n",R-L+1);
}
