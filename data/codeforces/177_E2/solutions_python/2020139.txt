#include <iostream>
#include <climits>
#include <cstdio>
#include <cstring>
#define FOR(i, n) for (int i = 0; i < (n); ++i)
#define MAX 10000
using namespace std;

typedef unsigned long long ull;
int n, c, a[MAX], b[MAX];

inline ull muldiv(ull a, ull x, ull b) {
	return (a / b) * (x / b) * b +
		   (a / b) * (x % b) +
		   (a % b) * (x / b) +
		   ((a % b) * (x % b)) / b;
}

inline ull query(ull x) {
	ull res = n;
	FOR (i, n) {
		res += muldiv(a[i], x, b[i]);
		if (res > c) return res;
	}
	return res;
}

int main() {
	bool notzero = false;
	scanf("%d%d", &n, &c);
	FOR (i, n) scanf("%d%d", &a[i], &b[i]), notzero |= a[i];

	if (!notzero) printf("%d\n", (c == n) ? -1 : 0);
	else {
		long long l, r, s, e;
		for (s = 1, e = 2000000000000000000LL; s < e; ) {
			long long m = (s + e) / 2;
			if (query(m) > c) e = m;
			else s = m + 1;
		}
		r = s;
		for (s = 1, e = 2000000000000000000LL; s < e; ) {
			long long m = (s + e) / 2;
			if (query(m) < c) s = m + 1;
			else e = m;
		}
		l = s;
		printf("%I64d\n", r - l);
	}
	return 0;
}
