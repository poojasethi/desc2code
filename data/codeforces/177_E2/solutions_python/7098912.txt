#include <iostream>
#include <cstdio>

using namespace std;

typedef long long ll;


ll N, C;
ll A[20000], B[20000];

bool all_zero = true;
ll mxa = 0, mnb = 2000000000;

ll process(ll P)
{
    ll re = 0;

    for(int Ni = 0; Ni < N; Ni++)
    {
        if( (long double)A[Ni]*P > (long double)C*B[Ni] ) return 1e10;
        re += A[Ni]*P/B[Ni] + 1;
    }


    return re;
}

int main()
{
    scanf("%I64d %I64d", &N, &C);

    for(int Ni = 0; Ni < N; Ni++)
    {
        scanf("%I64d %I64d", &A[Ni], &B[Ni]);

        if( A[Ni] )
        {
            all_zero = false;
            mxa = max(mxa, A[Ni]);
            mnb = min(mnb, B[Ni]);
        }
    }

    if( all_zero && N == C ){ puts("-1"); return 0; }

    ll p1, p2;

    ll l = 0, r = 1e18, mid;

    while( l+1 != r )
    {
        mid = (l+r)/2;

        if( process(mid) >= C+1 ) r = mid;
        else l = mid;
    }

    p1 = r;

    l = 0, r = 1e18;

    while( l+1 != r )
    {
        mid = (l+r)/2;

        if( process(mid) < C ) l = mid;
        else r = mid;
    }

    p2 = r;

    printf("%I64d\n", p1-p2);
}
