#define sqr(x) (x)*(x)
typedef unsigned long long Int;
const int inf=1000000000;
const int MOD=1000000007;

Int a[10005],b[10007];
Int n,c;

Int ok(Int x)
{
    Int cnt=n;
    FOR(i,1,n)
    {
        if(a[i] && x/b[i] + cnt > c)return c+1;
        if(x*a[i]/b[i]+cnt>c)return c+1;

        Int cur=(a[i]*x)/b[i];
        cnt+=cur;

        if(cnt>c)break;
    }
    return cnt;
}

int main()
{
    //freopen("input.txt","r",stdin);
    cin>>n>>c;
    Int ss=0;
    FOR(i,1,n){cin>>a[i]>>b[i];ss+=a[i];}
    if(ss==0 && c==n){cout<<"-1"<<endl;return 0;}

    Int l=1,r=inf*(Int)inf*2LL;
    while(l<r)
    {
        Int mid=(l+r)/2;
        if(ok(mid)<c)l=mid+1;else r=mid;
    }
    Int mn=r;
    
    l=1,r=inf*(Int)inf*2LL;
    while(l<r)
    {
        Int mid=1+(l+r)/2;
        if(ok(mid)>c)r=mid-1;else l=mid;
    }
    Int mx=r;

    //cout<<mn<<" "<<mx<<endl;

    if(ok(mn)!=c)cout<<"0"<<endl;else
        cout<<mx-mn+1<<endl;
}
