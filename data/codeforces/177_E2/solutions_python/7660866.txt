#include<cstdio>
#include<cstring>
using namespace std;
const int N=16384;
int n,i,j;
unsigned long long days,l,r,mid,c,a[N],b[N],max,min;
int main(){
	//freopen("input.txt","r",stdin);
	scanf("%lld%lld",&n,&c);
	for(i=1;i<=n;++i)scanf("%d%d",&a[i],&b[i]);
	l=1,r=1e18,min=r;
        for(j=0;j<256;++j){
        	mid=(l+r)/2;
        	days=n;
        	for(i=1;i<=n;++i)days+=(a[i]*mid)/b[i];
        	if(days<c)l=mid+1;else
        		if(days>c)r=mid;else {
        			if(mid<min)min=mid;
        			r=mid;
        		}
        }
        l=1,r=1e18,max=l;
        for(j=0;j<256;++j){
        	mid=(l+r)/2;
        	days=n;
        	for(i=1;i<=n;++i)days+=(a[i]*mid)/b[i];
         	if(days<c)l=mid+1;else
         		if(days>c)r=mid;else {
         			if(mid>max)max=mid;
         			l=mid+1;
         		}
        }
        unsigned long long ans;
        //printf("%lld %lld\n",min,max);
        if(max>=min)ans=max-min+1;else ans=0;
        //if (ans==2) printf("%lld %lld\n",min,max);
        if(min==1 && max==1e18)ans=-1;
        printf("%lld\n",ans);
        return 0;
}
