#include <iostream>
#include <cstdio>
using namespace std;
typedef long long ll;
const int N=10010;
const ll INF=ll(1e18);
ll a[N],b[N],c;
int n;
int cnt(ll x)
{
    int ans=0;
    while (x>1)
    {
        ans++;
        x/=10;
    }
    return ans;
}
ll sum(ll x)
{
    ll ans=0;
    for(int i=0;i<n;i++)
    {
        if (cnt(a[i])+cnt(x)>=19)
            return INF;
        ans+=(a[i]*x/b[i]);
        if (ans>c)
            return ans;
    }
    return ans;
}
int main()
{
    cin>>n>>c;
    c-=n;
    bool f=true;
    for(int i=0;i<n;i++)
    {
        cin>>a[i]>>b[i];
        if (a[i])
            f=false;
    }
    if (c<0)
    {
        cout<<0;
        return 0;
    }
    if (!c && f)
    {
        cout<<-1;
        return 0;
    }
    ll l1=1,r1=2*INF,l2=1,r2=2*INF;
    while (l1<r1)
    {
        ll m1=(l1+r1)/2;
        ll s=sum(m1);
        if (s<c)
            l1=m1+1;
        else
            r1=m1;
    }
    while (l2<r2)
    {
        ll m2=(l2+r2)/2;
        ll s=sum(m2);
        if (s<=c)
            l2=m2+1;
        else
            r2=m2;
    }
    cout<<l2-l1;
    return 0;
}
