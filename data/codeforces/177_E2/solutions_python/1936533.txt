#include <cassert>

#define INF 1000000000000000000LL

using namespace std;

int n, a[10005], b[10005];

long long f(long long x)
{
	long long ans = 0;
	for(int i=0; i<n; i++)
	{
		if(a[i] >= 1 + (INF - 1) / x) return INF;
		else
		{
			ans += 1 + a[i] * x / b[i];
			ans = min(ans, INF);
		}
	}
	return ans;
}

int main()
{
	int c;
	while(scanf("%d %d", &n, &c)  == 2)
	{
		for(int i=0; i<n; i++)
			scanf("%d %d", &a[i], &b[i]);
		
		if(c < f(1)) cout<<0<<endl;
		else if(f(INF) < c) cout<<0<<endl;
		else if(f(INF) == c) cout<<-1<<endl;
		else
		{
			long long lo1 = 1, hi1 = INF;
			if(f(1) == c) hi1 = 1;
			else
			{
				while(hi1 - lo1 > 1)
				{
					long long mid = lo1 + (hi1 - lo1) / 2;
					
					if(f(mid) >= c) hi1 = mid;
					else lo1 = mid;
				}
			}
			
			long long lo2 = 1, hi2 = INF;
			while(hi2 - lo2 > 1)
			{
				long long mid = lo2 + (hi2 - lo2) / 2;
				
				if(f(mid) > c) hi2 = mid;
				else lo2 = mid;
			}
			
			if(f(hi1) == c && f(lo2) == c) cout<<lo2-hi1+1<<endl;
			else cout<<0<<endl;
		}
	}
	
	return 0;
}
