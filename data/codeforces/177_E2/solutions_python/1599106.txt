#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
#define ll long long
const ll oo=1000000000000000000ll;
int n,c;
ll a[10005],b[10005];
int check(ll x)
{
	ll sum=0;
	for (int i=1;i<=n;i++){
		if (a[i]>oo/x) return 1;
		ll day=1+a[i]*x/b[i];
		if (day>c || sum>c-day) return 1;
		sum+=day;
		}
//	printf("x=%I64d %I64d\n",x,sum);
	if (sum>c) return 1;
	if (sum==c) return 0;
	if (sum<c) return -1;
}
int main()
{
	scanf("%d%d",&n,&c);
	for (int i=1;i<=n;i++) scanf("%I64d%I64d",&a[i],&b[i]);
	ll flag=0;
	for (int i=1;i<=n;i++)
		if (a[i]) flag=-1000000; else flag++;
	if (flag==c) {printf("-1\n"); return 0;}	
	ll l=0,r=oo,mid,L,R;
	while (l<r){
		mid=(l+r)/2; if (l==mid) mid++; int z=check(mid);
		if (z<0) l=mid; else r=mid-1; 
		}
	L=l;
	l=0; r=oo;
	while (l<r){
		mid=(l+r)/2; if (l==mid) mid++; int z=check(mid);
//		printf("l=%I64d %I64d %I64d %d\n",l,r,mid,z);
		if (z<=0) l=mid; else r=mid-1;
		}
	R=r;
	printf("%I64d\n",R-L); //for (;;);
	return 0;
}
