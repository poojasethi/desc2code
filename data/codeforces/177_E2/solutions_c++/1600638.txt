#include<cstdio>
#include<cstring>
using namespace std;
const long long inf=1000000000000000000ll;
int n,c;
long long a[10005],b[10005],ansl,ansr;
long long check(long long x)
{
	long long res=0;
	for (int i=1; i<=n; i++)
	{
		if (a[i]>inf/x) return inf;
		long long day=1+a[i]*x/b[i];
		if (day>c||res>c-day) return inf;
		res+=day;
	}
	return res;
}
int main()
{
	scanf("%d%d",&n,&c);
	long long sum=0;
	for (int i=1;i<=n;i++) 
		scanf("%I64d%I64d",&a[i],&b[i]),sum+=a[i];
	
	if (sum==0&&c==n)
	{
		printf("-1\n"); return 0;
	}
	
	long long left=0,right=inf,mid;
	while (left<right)
	{
		mid=(left+right)>>1; 
		if (left==mid) mid++; 
		long long res=check(mid);
		if (res<c) left=mid; else right=mid-1; 
	}
	ansl=left;
	
	left=0; right=inf;
	while (left<right)
	{
		mid=(left+right)>>1; 
		if (left==mid) mid++;
		long long res=check(mid);
		if (res<=c) left=mid; else right=mid-1;
	}
	ansr=right;
	printf("%I64d\n",ansr-ansl);
	return 0;
}
