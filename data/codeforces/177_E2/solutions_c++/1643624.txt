#include <cstdio>
#include <iostream>

using namespace std;

typedef unsigned long long ll;

inline int len(ll x)
{
  int res = 0;
  while (x) {
    ++res;
    x /= 10;
  }
  return res;
}

const int maxn = 10 * 1000 + 5;

ll a[maxn], b[maxn];

int n;


ll func(ll x, ll c)
{
  ll res = 0;
  for (int i = 0; i < n; ++i) {
    
    if (!a[i]) {
      res += 1LL;
      continue;
    }
    
    if ((len(x) + len(a[i]) - 1 > 18 && !(x * a[i] == 1000 * 1000 * 1000 * 1LL * 1000 * 1000 * 1000)) || (a[i] && x > (1LL << 60) / a[i] + 3) || a[i] * x > (1LL << 60)) {
      return 1LL << 60;
    }
    
    if (x * a[i] / b[i] + 1 > c) {
      return 1LL << 60;
    }
    
    res += 1LL + x * a[i] / b[i];
    
    if (res > c) {
      return 1LL << 60;
    }
  }
  return res;
}

ll bin_search(ll c, ll mx)
{
  ll l = 0, r = 1LL << 60;//mx + 1;
  while (l + 1 < r) {
    ll x = (l + r) >> 1;
    if (func(x, c) < c) {
      l = x;
    }
    else {
      r = x;
    }
  }

  return l;
}

int main()
{
  ll c;
  cin >> n >> c;
  
  ll ma = 0;
  for (int i = 0; i < n; ++i) {
    cin >> a[i] >> b[i];
    ma = max(ma, a[i]);
  }
  ll mx = 111;
  
  if (ma == 0) {
    cout << (n == c ? -1 : 0);
    return 0;
  }

  ll res = bin_search(c + 1, mx) - bin_search(c, mx);
  cout << res << '\n';
  
  return 0;
}
