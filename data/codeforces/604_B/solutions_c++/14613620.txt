#include<bits/stdc++.h>
using namespace std ;
int tall[(int)1e5] ;
int main()
{
	int n,k,size ;scanf("%d %d",&n,&k) ;
	for(int i=0;i<n;++i) scanf("%d",&tall[i]) ;
	size=tall[n-1] ;
	for(int i=0,j=2*n-2*k-1;i<j;++i,--j)
	if(tall[i]+tall[j]>size) size=tall[i]+tall[j] ;
	printf("%d\n",size) ;
	return 0;
}
