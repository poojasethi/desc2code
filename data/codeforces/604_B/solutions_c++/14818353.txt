#include<bits/stdc++.h>
using namespace std;
int size[400100],ans=0,n,k;
int main()
{
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++)
		scanf("%d",&size[i]);
	for (int i=1;i<=n-k;i++)
		ans=max(size[2*(n-k)-i+1]+size[i],ans);
	printf("%d",max(ans,size[n]));
	return 0;
}
