#include <iostream>
#include <cmath>
using namespace std;

int s[100000];

int main()
{
    int n,k,m;
    cin >> n >> k;
    for(int i=0;i<n;i++) cin >> s[i];
    m = s[n-1];
    for(int i=0;n-k-1-i >= 0;i++)
        if(s[n-k-1-i]+s[n-k+i] > m) m = s[n-k-1-i]+s[n-k+i];
    cout << m;
    return 0;
}
