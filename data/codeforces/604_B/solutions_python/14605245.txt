n,k = map(int,raw_input().split())
ar = map(int,raw_input().split())

if k==n/2:
    ar2 = [ar[i] + ar[-i-1] for i in xrange(n/2)]
    print max(ar2)
else:
    rq = n
    ex = 2*k-rq
    ar2 = ar[-ex:]
    ar = ar[:-ex]
    for i in xrange(len(ar)):
        ar2 += [ar[i]+ar[-i-1]]
    print max(ar2)
