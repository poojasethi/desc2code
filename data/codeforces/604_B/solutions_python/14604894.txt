
def main():
    n, k = tuple(map(int, raw_input().split()))
    S = map(int, raw_input().split())

    s = S[-1]
    if n<=k:
        print s
    else:
        box = S[n-k:]
        for i, v in enumerate(S[n-k-1::-1]):
            box[i] += v

        print max(box)
        
if __name__ == "__main__":
    main()
