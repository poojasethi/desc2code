import sys

def solve(bells, n, k):
    max_s = bells[-1]

    remain = n - (2*k - n)
    # print remain
    if remain <= 0:
        return max_s

    for i in range(remain):
        if bells[i] + bells[remain-i-1] > max_s:
            max_s = bells[i] + bells[remain-i-1]

    return max_s

if __name__=='__main__':
    lines = sys.stdin.readlines()
    
    n, k = [int(v) for v in lines[0].strip().split()]

    bells = [int(v) for v in lines[1].strip().split()]

    print solve(bells, n, k)
