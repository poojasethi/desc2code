from sys import stdin, stdout
n,k= [int(x) for x in stdin.readline().rstrip().split()]
cowbells = [int(x) for x in stdin.readline().rstrip().split()]
end = cowbells[-1]
if k >= n:
    stdout.write(str(end))
else:
    overflow = cowbells[0] + cowbells[2*(n-k)-1]
    for i in xrange((2*(n-k)-1)+1):
        overflow = max(cowbells[i] + cowbells[2*(n-k)-1-i],overflow)
    out = max(end,overflow)
    stdout.write(str(out))
