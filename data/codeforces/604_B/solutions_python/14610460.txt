
k,n = map(int,raw_input().split())
arr = map(int,raw_input().split())

i,k1,n1 = 0,0,0
maxm = -10**10
newarr = []
while (k-k1) > n-n1:
    newarr.append(arr[i])
    newarr.append(arr[i+1])
    i += 2
    k1 += 2
    n1 += 1
for i in range(len(newarr)):
    maxm = max(maxm,newarr[i] + newarr[-i-1])
for z in range(i,len(arr)):
    maxm = max(maxm,arr[z])
print maxm
