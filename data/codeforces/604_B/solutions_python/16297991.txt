import sys
x=sys.stdin.readline().rstrip().split()
s=sys.stdin.readline().rstrip().split()
#x=[4,3]
#s=[2,3,5,9]
n=int(x[0])
k=int(x[1])

singles=k-(n-k)
high=int(s[-1])
b=0
e=len(s)-1-singles
while(b<e):
    if high<int(s[b])+int(s[e]):
        high=int(s[b])+int(s[e])
    b=b+1
    e=e-1

print high
