n, k = map(int, raw_input().split())
S = map(int, raw_input().split())

if n <= k:
    print max(S)
else:


    cases = [[] for _ in xrange(k)]

    for i in xrange(k):
        cases[i].append(S.pop())

    cases.reverse()

    for i in xrange(len(S)):
        cases[i].append(S.pop())

    cases = map(lambda x: sum(x), cases)

    print max(cases)
