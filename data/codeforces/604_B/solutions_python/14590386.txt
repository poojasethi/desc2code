n , k = map(int ,raw_input().split())
s = map(int , raw_input().split())

spare =( k * 2 ) - len(s)

maxx = 0
if spare < k * 2:
    maxx = s[-1]

limit =  n - spare
for i in range(0 , limit / 2):
    if i != limit - i:
        maxx = max(maxx , s[i] + s[limit -i-1] )

if limit % 2 !=0:
    maxx = max(maxx , s[limit / 2])

print maxx
