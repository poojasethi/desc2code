n, m = (int(i) for i in raw_input().split(' '))
S = [int(i) for i in raw_input().split(' ')]
s = S[-1]

p = n-m
S_table_max = 0

if p>0:
    S_table = [ (S[1+i-1] + S[2*p-i-1]) for i in xrange(p) ]

if p>0:
    S_table_max = max(S_table)

if S_table_max < s:
    print s
else:
    print S_table_max
