[n, k] = map(int, raw_input().split())
s = map(int, raw_input().split())


r = 2 * s[n - 1]; l = s[n - 1]
while l < r:
    mid = (l + r) / 2
    lend, rend, b = -1, n, 0
    while lend < rend - 1:
        b += 1; rend -= 1; rem_cap = mid - s[rend]
        if s[lend + 1] <= rem_cap:
            lend += 1
        if b == k:
            break
    if rend > lend + 1:
        if l == mid:
            break
        l = mid
    else:
        r = mid
print r
