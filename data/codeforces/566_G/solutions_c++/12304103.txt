using namespace std;
typedef long long ll;
typedef double db;
const int N=100005;

inline int IN(){
	char ch;CH; int f=0,x=0;
	for(;pend(ch);CH); if(ch=='-')f=1,CH;
	for(;!pend(ch);CH) x=x*10+ch-'0';
	return (f)?(-x):(x);
}

int n,m,tot,top;
struct po{
	db x,y;int ty;
	void read(int _ty){
		scanf("%lf%lf",&x,&y);
		ty=_ty;
	}
	inline int operator <(const po &a)const{
		if(x==a.x && y==a.y) return ty<a.ty;
		if(x==a.x) return y>a.y;
		return x<a.x;
	}
	po operator -(const po &a)const{
		return (po){x-a.x,y-a.y};
	}
	db operator *(const po &a)const{
		return x*a.y-y*a.x;
	}
}a[N*2],stk[N*2];

void Return(string t){
	cout<<t<<endl;
	exit(0);
}

int main(){
	scanf("%d%d",&n,&m);
	scanf("%*d%*d");
	For(i,1,n) a[++tot].read(0);
	For(i,1,m) a[++tot].read(1);
	sort(a+1,a+tot+1);
	
	For(i,1,tot){
		if(i>1&&a[i].x==a[i-1].x&&a[i].y==a[i-1].y)continue;
		while(top&&a[i].y>stk[top].y) top--;
		while(top>=2&&(stk[top-1]-stk[top])*(a[i]-stk[top])<0)top--;
		stk[++top]=a[i];
	}
	For(i,1,top) if(stk[i].ty==0) Return("Max");
	Return("Min");
	return 0;
}
