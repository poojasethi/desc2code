#define N 100010

pii a[N<<1];
bool cmp(pii a,pii b){
    return a.x<b.x||(a.x==b.x&&a.y>b.y);
}
int q[N<<1];
ll det(int i,int j,int k){
    ll x1=a[j].x-a[i].x,y1=a[j].y-a[i].y;
    ll x2=a[k].x-a[i].x,y2=a[k].y-a[i].y;
    return x1*y2-x2*y1;
}
int main()
{
    //freopen("2.txt","w",stdout);
    int Ta,i,j,k,ca=0,m,n;
    int x,y;
    scanf("%d%d%d%d",&n,&m,&x,&y);
    set<pii>s;
    for(i=0;i<n;i++)scanf("%d%d",&a[i].x,&a[i].y),s.insert(a[i]);
    for(i=0;i<m;i++)scanf("%d%d",&a[i+n].x,&a[i+n].y);
    n+=m;
    sort(a,a+n,cmp);
    n=unique(a,a+n)-a;
    int top=0;
    for(i=0;i<n;i++){
        while(top&&a[i].y>a[q[top-1]].y)top--;
        while(top>=2&&det(q[top-2],q[top-1],i)>0)top--;
        q[top++]=i;
    }
    string ans="Min";
    for(i=0;i<top;i++){
        if(s.find(a[q[i]])!=s.end()){ans="Max";break;}
    }
    cout<<ans<<"\n";
    return 0;
}
