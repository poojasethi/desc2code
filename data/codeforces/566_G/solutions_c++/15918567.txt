using namespace std;
typedef long long ll;
const int n=200003;
ll pov(ll q,ll w,ll e,ll r,ll t,ll y){
  return (e-q)*(y-w)-(t-q)*(r-w);
}
int a1[n],a2[n],a3[n];
int cmp(ll e,ll e1,ll r,ll r1){
  if(e1*r<r1*e)
   return -1;
  if(e1*r>r1*e)
   return 1;
  if(e+e1>r+r1)
   return -1;
  if(e+e1<r+r1)
   return 1;
  return 0;
}
void qs(int q,int w){
  int e=q,r=w;
  ll y=q+(rand()%(w-q+1)),t1=a1[y],t2=a2[y];
  do{
   while(cmp(a1[e],a2[e],t1,t2)>0)e++;
   while(cmp(a1[r],a2[r],t1,t2)<0)r--;
   if(e<=r){
    y=a1[e];a1[e]=a1[r];a1[r]=y;
    y=a2[e];a2[e]=a2[r];a2[r]=y;
    y=a3[e];a3[e]=a3[r];a3[r]=y;
    e++;r--;
   }
  }while(e<=r);
  if(q<r)qs(q,r);
  if(e<w)qs(e,w);
}
int s[n];

   maxx=max(maxx,a1[t]);

   a3[t]=1;

  }
  a1[t]=0;
  a2[t]=maxy;

  t++;
  a1[t]=maxx;

  t++;
  qs(1,t-1);
  int t1=2;
  for(w=2;w<t;w++)
   if((a1[w]==a1[t1-1])&&(a2[w]==a2[t1-1]))
    a3[t1-1]=min(a3[t1-1],a3[w]);

    t1--;
