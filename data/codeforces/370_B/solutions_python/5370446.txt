def test_cards(cards):
    out = []
    for number, card in enumerate(cards):
        flag = 1
        for another_number, another_card in enumerate(cards):
            if not number == another_number:
                if another_card.issubset(card):
                    flag = 0
                    break
        if flag == 0:
            out.append("NO")
        else:
            out.append("YES")
    return out

number_of_cards = input()
cards = []
for i in xrange(number_of_cards):
    card = raw_input().split()
    del card[0]
    cards.append(set(card))
for answer in test_cards(cards):
    print answer
