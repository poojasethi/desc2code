players = int(raw_input())

lst = []
results = []


for i in xrange(0, players):
    cards = raw_input().split()
    cards.pop(0)
    lst.append(cards)
    results.append('YES')



for i,v in enumerate(lst):
    ident = 0
    for j, w in enumerate(lst):
        ident = 0
        if j == i:
            continue
        for item in v:
            if item in w:
                continue
            else:
                ident = -1
                break
        if(ident != -1):
            results[j] = 'NO'
    

for i in results:
    print i
