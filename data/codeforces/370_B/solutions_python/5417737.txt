#!/usr/bin/python
## Berland Bingo/ Mejorado ##
import sys

no = int(input())
sets = []
for _ in range(no):
    lista = [int(a) for a in raw_input().split()]
    sets.append(set((lista[1:])))    

for i in range(no):
    win = True
    for j in range(no):
        if j!=i and sets[j] <= sets[i]:
            win = False

    print "YES" if win else "NO"
