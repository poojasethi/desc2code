def main():
    n = int(raw_input())
    r = []
    for i in range(n):
        c = map(int,raw_input().split())
        r.append(set(c[1:]))
    for i in range(n):
        for j in range(n):
            if j == i:
                continue
            if r[j].issubset(r[i]):
                print "NO"
                break
        else:
            print "YES"

if __name__ == '__main__':
    main()
