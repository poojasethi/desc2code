from sets import Set
players = int(raw_input())
can_win = [1] * players
cards = []

for i in xrange(players):

    card = map(int, raw_input().split(' '))
    card.remove(card[0])
    cards.append(card)

for i in xrange(players):
    card = Set(cards[i])
    for j in xrange(players):
        if i != j:
            card_2 = Set(cards[j])
            if card.issubset(card_2):
                can_win[j] = 0

for w in xrange(players):
    if can_win[w] == 1:
        print "YES"
    else:
        print "NO"
