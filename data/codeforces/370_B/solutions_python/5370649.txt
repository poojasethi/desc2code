n=int(raw_input())
cards=[]
for i in range(0,n):
	card=map(int,raw_input().split())
	del(card[0])
	#print card
	card=set(card)
	#print card
	cards.append(card)
for i in range(0,n):
	flag=0
	for j in range(0,n):
		if i!=j:
			if cards[j].issubset(cards[i]):
				#print cards[i],cards[j]
				flag=1
				break
	if flag:
		print "NO"
	else:
		print "YES"
