ar=[]
n=int(input())
for i in xrange(n):
    A=map(int,raw_input().split())
    del(A[0])
    ar.append(A)
for i in xrange(n):
    flg=True
    for j in xrange(n):
        if i!=j:
            if set(ar[i])&set(ar[j])==set(ar[j]):
                flg=False
    if flg==False:
        print "NO"
    else:
        print "YES"
