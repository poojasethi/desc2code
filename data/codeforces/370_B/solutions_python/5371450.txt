from sets import Set
j=0
n=input()
size = n
a=[]
while n!=0:
    st=raw_input().split()
    m=int(st[0])
    a.append(Set([]))
    for i in range(0,m):
        a[j].add(int(st[i+1]))
    j+=1
    n-=1

flag=size*[0]

for i in range (0,size):
    for j in range (0,size):
        if i!=j:
            if a[i].issubset(a[j]):
                flag[j]=1

for i in range (0,size):
    if flag[i] == 0:
        print "YES"
    else:
        print "NO"
