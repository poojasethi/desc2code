n=int(raw_input())

P=[]

for i in range(n):
    P.append([int(e) for e in raw_input().split()[1:]])

for i in range(n):
    bad=False
    for j in range(n):
        if i!=j:
            if all(e in P[i] for e in P[j]) and len(P[j])<=len(P[i]):
                bad=True
                break

    print "YES" if not bad else "NO"
