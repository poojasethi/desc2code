lstCards,lstOut=[],[]
for i in xrange(int(raw_input())):
    tmp = raw_input().split()
    tmp.pop(0)
    lstCards.append(set([int(i) for i in tmp]))

for i, item in enumerate(lstCards):
    yes=True
    for j, others in enumerate(lstCards):
        if (i!=j and item>=others):
            yes=False
            break
    lstOut.append(yes)

for i in lstOut:
    if (i):
        print "YES"
    else:
        print "NO"
