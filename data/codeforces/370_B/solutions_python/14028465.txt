p = int(raw_input())
asses = []

for i in xrange(p):
    ass = map(int, raw_input().split()[1:])
    asses.append(ass)
    
for i in xrange(len(asses)):
    pg = True
    for j in xrange(len(asses)):
        if i != j:
            cont = True
            for k in asses[j]:
                if k not in asses[i]:
                    cont = False
                    break
            if cont:
                pg = False
                break
    if pg:
        print "YES"
    else:
        print "NO"
