def main():
    #from sys import stdin, stdout
    #start here
    n=raw_input()
    n=int(n)
    d=[]
    for i in xrange(n):
        k=map(int,raw_input().split())
        d.append(set(k[1:]))
    for i in xrange(n) :
        f=1
        for j in xrange(n):
            if(i!=j and d[j].issubset(d[i])):
                f=0
                break
        if(f):
            print "YES"
        else:
            print "NO"
if __name__ == "__main__":
    main()
