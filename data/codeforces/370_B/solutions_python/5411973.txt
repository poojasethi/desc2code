n = int(raw_input())

sets = [0] * n

for i in xrange(n):
    x = map(int, raw_input().split())
    m = x[0]
    x = x[1:]
    for j in xrange(m):
        sets[i] |= 1 << x[j]

for i in xrange(n):
    ok = True
    for j in xrange(n):
        if i <> j and (sets[i] & sets[j]) == sets[j]:
            ok = False
            break
    print "YES" if ok else "NO"
