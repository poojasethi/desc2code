n=input()
chk=[1]*n
l=[]
for i in range(n):
    m=map(int,raw_input().split())
    m.insert(1,i)
    l.append(m)
l.sort()
for i in xrange(n):
    for j in xrange(n):
        if i==j:
            pass
        elif l[i][0]<l[j][0]:
            break
        elif len(set(l[i][2:]+l[j][2:]))==l[i][0]:
            chk[l[i][1]]=0
            break
for i in chk:
    print 'YES' if i else 'NO'
