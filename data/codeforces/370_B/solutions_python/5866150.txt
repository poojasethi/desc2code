n = int(raw_input())
card = []

for i in xrange(n):
    card.append (([int(s) for s in raw_input().split()])[1:])

for i in xrange(n):
    for j in xrange(n):
        if j != i:
            #found = True
            for k in card[j]:
                if k not in card[i]:
                    break
            else:
                print "NO"
                break
    else:
        print "YES"
