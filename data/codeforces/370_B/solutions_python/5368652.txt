from bisect import *

def index(a, x):
    'Locate the leftmost value exactly equal to x'
    i = bisect_left(a, x)
    if i != len(a) and a[i] == x:
        return True
	return False	

def check_subset(x, y):
	flag = False
	if len(y) > len(x):
		return False
	else:
		for i in range(len(y)):
			if not index(x, y[i]):
				flag = True
	if flag:
		return False
	else:
		return True

n = int(raw_input())
a = [[] for i in range(n)]
for i in range(n):
	a[i] = [int(j) for j in raw_input().split()]
	a[i] = a[i][1:]
	a[i].sort()
for i in range(n):
	flag = True
	for j in range(n):
		if(i != j and check_subset(a[i], a[j])):
			flag = False
	if flag:
		print "YES"
	else:
		print "NO"
