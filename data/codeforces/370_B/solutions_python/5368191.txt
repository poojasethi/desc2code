#coding=utf-8



n = int(input())

a = [[] for i in range(n)]

a = [map(int, raw_input().split())[1:] for i in range(n)]


def contain(s1, l2):
  for i in l2:
    if i not in s1:
      return False
  return True

for l1 in a:
  s1 = set(l1)
  cnt = 0
  for l2 in a:
    if contain(s1, l2):
      cnt += 1
  if cnt >= 2:
    print "NO"
  else:
    print "YES"
