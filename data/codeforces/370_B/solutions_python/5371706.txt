from operator import or_
N  = int(raw_input())
M = [0]*N

for _ in xrange(N) :
	l = map(int,raw_input().split())
	M[_] = set(l[1:])
#print M
if N ==1 :
	print "YES"
else :
	for i in xrange(N) :
		b = reduce(or_,[M[j].issubset(M[i]) for j in xrange(N) if j!=i])
		print "YES"  if b == False else "NO"
