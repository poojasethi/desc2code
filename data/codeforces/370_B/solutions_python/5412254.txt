def WinnerOrNot(lista, num, pos):
    for p in range(len(lista)):
        counter = 0
        if int(lista[p][0]) <= int(num[0]) and pos != p:
            for i in lista[p][1]:
                for j in num[1]:
                    if i == j:
                        counter += 1
            if counter == len(lista[p][1]):
                return 'NO' 
        
    return 'YES'
