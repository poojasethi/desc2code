#!/usr/bin/env python

n = input()
cards = {}
for player in range(n):
	card = map(int, raw_input().split())
	cards[player] = sorted(card[1:])

def isWinner(player):
	for p in range(n):
		if p == player: continue

		def isSubseq(c1, c2):
			x, y = 0, 0
			while x < len(c1) and y < len(c2):
				if c1[x] == c2[y]:
					x, y = x+1, y+1
				else:
					y = y+1
			if x == len(c1):
				return True
			else:
				return False

		if isSubseq(cards[p], cards[player]):
			return False

	return True

for player in range(n):
	if isWinner(player):
		print 'YES'
	else:
		print 'NO'
