#include<bitset>
#include<iostream>
using namespace std;
const int MAXN=120;
bitset<100>bs[MAXN];

int main(){
    ios_base::sync_with_stdio(false);
    int n,m,a;
    cin>>n;
    for (int i=0;i<n;i++){
        cin>>m;
        for (int j=0;j<m;j++){
            cin>>a;
            bs[i].set(a-1);
        }
    }
    for (int i=0;i<n;i++){
        bool ok=true;
        for (int j=0;j<n&&ok;j++){
            if (i==j) continue;
            if ((bs[i]&bs[j])==bs[j]){
                ok=false;
            }
        }
        cout<<(ok?"YES":"NO")<<endl;
    }
    return 0;
}
