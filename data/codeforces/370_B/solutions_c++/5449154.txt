#include<bits/stdc++.h>
using namespace std;

bool a[105][105];
int n;

bool check(int i,int j)
{
	for(int k=1;k<=100;k++)
	  if(a[i][k]==true&&a[j][k]==false)
		return false;
	return true;
}
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		int t,x;
		cin>>t;
		for(int j=0;j<t;j++)
		{
			cin>>x;
			a[i][x]=true;
		}
	}

	for(int i=1;i<=n;i++)
	{
		bool f=true;
		for(int j=1;j<=n;j++)
		  if(i!=j&&check(j,i))
		  {
			 f=false;
			 break;
		  }
		if(f) puts("YES");
		else puts("NO");
	}
	return 0;
}
