#include <cstdio>
#include <cstring>
using namespace std;
bool number[101];
int player[101][101],m[101];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&m[i]);
		for(int j=0;j<m[i];j++)scanf("%d",&player[i][j]);
	}
	int j=0;
	for(int i=0;i<n;i++){
		memset(number,0,sizeof number);
		for(j=0;j<m[i];j++)number[player[i][j]]=true;
		for(j=0;j<n;j++){int k=0;
			if(i==j)continue;
			for(;k<m[j];k++){if(!number[player[j][k]])break;}
			if(k==m[j])break;
		}
		if(j==n)printf("YES\n");
		else printf("NO\n");
	}
	return 0;
}
