#include <cstdio>
#include <cstdlib>
#include <bitset>
#define N 105
using namespace std;
bitset<N> a[N];

int main()
{
	int n,ans[N]={0};
	scanf("%d",&n);
	for(int i=1,m;i<=n;++i)
	{
		scanf("%d",&m);
		for(int j=1,x;j<=m;++j)
			scanf("%d",&x),a[i][x]=1;
	}
	for(int i=1;i<n;++i)
		for(int j=i+1;j<=n;++j)
		{
			if((a[i]&a[j])==a[i]) ans[j]=1;
			if((a[i]&a[j])==a[j]) ans[i]=1;
		}
	for(int i=1;i<=n;++i)
		puts(ans[i]?"NO":"YES");
	return 0;
}
