#include<cstdio>
int main () {
	int n,num,i,j,k;
	int len[101]={0};
	int tmp[101][101]={0};
	int ans[101]={0};
	scanf("%d",&n);
	for(i=0;i<n;i++) {
		scanf("%d",&len[i]);
		for(j=0;j<len[i];j++) {
			scanf("%d",&num);
			tmp[i][num]++;
		}
	}
	for(i=0;i<n;i++) {
		for(j=0;j<n;j++) {
			if(j!=i) {
				if(len[j]>=len[i]) {
					for(k=1;k<=100;k++) 
						if(tmp[i][k]==1&&tmp[j][k]==0) break;
					if(k==101) ans[j]=1;
				}
			}
		}
	}
	for(i=0;i<n;i++) if(ans[i]==0) printf("YES\n"); else printf("NO\n");
	return 0;
}
