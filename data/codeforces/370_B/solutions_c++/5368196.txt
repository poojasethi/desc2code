#include<iostream>
using namespace std;

int a[101][101],n;
int check(int x,int y){
	for (int i=1;i<=a[x][0];i++){
		int t=0;
		for (int j=1;j<=a[y][0];j++)
			if (a[x][i] == a[y][j]){
				t=1;
				break;
			}
		if (!t) return 0;
	}
	return 1;
}	
int main(){
	cin>>n;
	for (int i=0;i<n;i++){
		cin>>a[i][0];
		for (int j=1;j<=a[i][0];j++) cin>>a[i][j];
	}
	for (int i=0;i<n;i++){
		int k=1;
		for (int j=0;j<n;j++)
			if ((i!=j) && (check(j,i))){
					k=0;
					break;
			}
		if (k) cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
	return 0;
}
