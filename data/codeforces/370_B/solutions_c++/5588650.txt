#include <cstdio>
#include <iostream>
using namespace std;

const int MAX = 105;
int cards[MAX][MAX];

int main() {
	int n; scanf("%d", &n);
	for(int i = 0;i < n;++i) {
		scanf("%d", &cards[i][0]);
		for(int j = 1;j <= cards[i][0];++j) {
			int k; scanf("%d", &k);
			cards[i][k] = 1;
		}
	}
	
	for(int i = 0;i < n;++i) {
		bool bisa = 1;
		for(int j = 0;j < n && bisa;++j) {
			if(i == j) continue;
			bisa = 0;
			for(int k = 1;k <= 100 && !bisa;++k)
				if(cards[j][k] && !cards[i][k]) bisa = 1;
		}
		puts(bisa ? "YES" : "NO");
	}
}
