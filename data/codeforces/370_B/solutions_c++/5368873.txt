#include <cstdio>
bool has[120][120];
bool meixi[120];
int n, m, i, j, k, l;
int main(){
    scanf("%d\n", &n);
    for (i = 1; i <= n; i++){
        scanf("%d", &k);
        while (k--){
            scanf("%d", &l);
            has[i][l] = true;
        }
    }
    for (i = 1; i <= n; i++) for (j = 1; j <= n; j++) if (i != j){
        for (k = 1; k <= 100; k++) if (has[i][k] && !has[j][k]) break;
        if (k > 100) meixi[j] = true;
    }
    for (i = 1; i <= n; i++) if (meixi[i]) printf("NO\n"); else printf("YES\n");
    return 0;
}
