#include<cstdio>
#include<algorithm>
#include<set>

using namespace std;


int n;
int m[100];
set<int> a[100];

int main()
{
	scanf("%d", &n);
	for( int i = 0; i < n; ++i ) {
		scanf("%d", &m[i]);
		for( int j = 0; j < m[i]; ++j ) {
			int v;
			scanf("%d", &v);
			a[i].insert(v);
		}
	}

	for( int i = 0; i < n; ++i ) {
		bool ans = false;
		for( int j = 0; j < n; ++j ) {
			if( i == j ) continue;

			bool flag = true;
			for( set<int>::iterator ite = a[j].begin(); ite != a[j].end(); ++ite ) {
				flag &= a[i].find(*ite) != a[i].end();
			}

			ans |= flag;
		}
		printf("%s\n", ans ? "NO" : "YES");
	}
	
	return 0;
}
