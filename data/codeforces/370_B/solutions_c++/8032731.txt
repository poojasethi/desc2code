#include<cstdio>
#include<algorithm>
#include<cstring>
#include<climits>
using namespace std;
int a[110][110];
int main(){
	int n;scanf("%d",&n);
	for(int i=1;i<=n;i++){
		int k;scanf("%d",&k);
		for(int j=1;j<=k;j++){
			int p;scanf("%d",&p);a[i][p]++;
		}
	}
	for(int i=1;i<=n;i++){ 
		int flag=1;
        for(int j=1;j<=n;j++)
    	if(i!=j){
			flag=0;
      		for(int k=1;k<=100;k++)
       		if((a[j][k]>0&&a[i][k]==0)){   
			   flag=1;
           		break;
       		}
       		if(!flag)break;
    }if(flag)printf("YES\n");
    else printf("NO\n");
	}
}
