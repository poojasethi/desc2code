#include<cstdio>
#include<iostream>
using namespace std;
int a[101][101];
int main()
{
	int n;
	int m;
	scanf("%d",&n);
	int i,j,k;
	int val;
	int ava[n];
	for(i=0;i<n;i++)
	{
		scanf("%d",&ava[i]);
		for(j=0;j<ava[i];j++)
		{
			scanf("%d",&val);
			a[i][val]=1;
		}
	}
	for(i=0;i<n;i++)
	{
		int flag=0;
		for(j=0;j<n;j++)
		{
			if(j!=i)
			{
				int count=0;
				for(k=0;k<101;k++)
					if(a[i][k]==1 && a[j][k]==1)
						count++;
				if(count==ava[j])
					flag=1;
			}
			if(flag==1)
				break;
		}
		if(flag==1)
			printf("NO\n");
		else
			printf("YES\n");
	}
	return 0;
}
