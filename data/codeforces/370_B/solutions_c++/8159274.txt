#include<iostream>
#include<algorithm>
using namespace std;
int n,i,j,q,a[105][105],k,ans,x,m;
int main()
{
 cin>>n;
 for(i=1;i<=n;i++)
 {
  cin>>m;
  for(j=1;j<=m;j++)
  {
   cin>>x;
   a[i][x]++;
  }
 }
 for(i=1;i<=n;i++)
 {
  for(j=1;j<=n;j++)
  {
   for(q=1;q<=100;q++)
   if(a[j][q]==1 && a[i][q]==0)k++;
   if(k>0)ans++;
   k=0;
  }
  if(ans==n-1)cout<<"YES"<<endl;
             else cout<<"NO"<<endl;
  ans=0;
 }

    return 0;
}
