#include <iostream>
using namespace std;

int main() {
	int n, i, j, m[101], a[101][101], win=0, k;
	static int has[101][101];
	cin>>n;
	for (i=0;i<n;i++) {
		cin>>m[i];
		for (j=0;j<m[i];j++) {
			cin>>a[i][j];
			has[i][a[i][j]] = 1;
		}
	}
	
	for (i=0;i<n;i++) {
		win = 1;
		for (j=0; j<n && win; j++) {
			if (i==j || m[i]<m[j])
				continue;
			win = 0;
			for (k=0; k < m[j];k++) {
				if (!has[i][a[j][k]]) {
					win = 1;
					break;
				}
			}
		}
		
		if (win) 
			cout<<"YES\n";
		else
			cout<<"NO\n";
	}
	return 0;
}
