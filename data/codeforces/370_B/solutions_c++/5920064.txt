#include <iostream>
#include <cstdio>
#include <vector>
#include <bitset>

using namespace std;

int N, M;
vector <bitset <101> > V;

int main() {
	ios_base::sync_with_stdio(false);
#ifndef ONLINE_JUDGE
	freopen("input.txt", "r", stdin);
#endif
	
	int i, x, j;
	cin >> N;
	V.resize(N);
	for (i = 0; i < N; ++i) {
		cin >> M;
		for (j = 0; j < M; ++j) {
			cin >> x;
			V[i][x] = 1;
		}
	}
	
	for (i = 0; i < N; ++i) {
		for (j = 0; j < N; ++j) {
			if (i != j) {
				if ((V[i] & V[j]) == V[j]) {
					cout << "NO\n";
					break;
				}
			}
		}
		if (j == N) {
			cout << "YES\n";
		}
	}
	
	return 0;
}
