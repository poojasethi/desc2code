#include <cstdio>
int n,a[110][110],t[110],c,swi1,swi2;

int main()
{
	int i,j,k;
	//freopen("B.in","r",stdin);
	//freopen("B.out","w",stdout);
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&t[i]);
		for(j=1;j<=t[i];j++)
		{
			scanf("%d",&c);
			a[i][c]=1;
		}
	}
	for(i=1;i<=n;i++)
	{
		swi1=1;
		for(j=1;j<=n;j++)
		if(j!=i)
		{
			swi2=0;
			for(k=1;k<=100;k++)
			{
				if(a[j][k])
					if(!a[i][k]) swi2=1;
			}
			if(swi2==0) swi1=0;
		}
		if(swi1==0) printf("NO\n");
		else printf("YES\n");
	}
	
	
	return 0;
}
