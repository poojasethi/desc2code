#include<algorithm>
#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int n,m[101][101],ans[101];
int main(){
	scanf("%d",&n);
	for(int i=1,t;i<=n;i++)
	{
		scanf("%d",&t);
		for(int j=1,k;j<=t;j++)
			scanf("%d",&k),m[i][k]=1;
	}for(int i=1;i<=n;i++)
	for(int j=1;j<=n;j++)
		if(j!=i){
			int flag=0;
			for(int k=1;k<=100;k++)
				if(m[j][k]==1&&m[i][k]==0)
					flag=1;
			ans[i]|=!flag;
		}
	for(int i=1;i<=n;i++)
		puts(ans[i]==1?"NO":"YES");
}
