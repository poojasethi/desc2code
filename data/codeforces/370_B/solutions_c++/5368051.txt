#include <iostream>
#include <cstdio>
using namespace std;
int m[110],vis[110][110];
int main()
{
	int n;
	scanf("%d",&n);
	for (int i=1; i<=n; i++)
	{
		scanf("%d",&m[i]);
		for (int j=1; j<=m[i]; j++)
		{
			int a;
			scanf("%d",&a);
			vis[i][a]=1;
		}
	}
	for (int i=1; i<=n; i++)
	{
		int num=0;
		for (int j=1; j<=n; j++)
		{
			if (i==j) continue;
			int c=0;
			for (int k=1; k<=100; k++)
			{
				if (vis[i][k]==1 && vis[j][k]==1) c++;
			}
			if (m[j]>c) num++;
		}
		if (num==n-1) printf("YES\n"); else printf("NO\n");
	}
	return 0;
}
