#include<iostream>
using namespace std;
int a[110][110], b[110], n, k, x;
int main()
{
	cin >> n;
	for (int i=0; i<n; i++)
	{
		cin >> b[i];
		for (int j=0; j<b[i]; j++)
		{
			cin >> x;
			a[i][x]++;
		}
	}
	for (int i=0; i<n; i++)
	{
		bool flag = true;
		for (int j=0; j<n && flag; j++)
		{
			if (i == j) continue;
			flag = false;
			for (int k=1; k<=100 && !flag; k++) if (a[j][k] && !a[i][k]) flag = true;
		}
		if (flag) cout << "YES" << "\n";
		else cout << "NO" << "\n";
	}
}
