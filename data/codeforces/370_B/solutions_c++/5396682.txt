#include<iostream>
using namespace std;

int cnt[500][2242];

bool cont(int a,int b) //a contains b	b in a	b is a de ziji
{
	for(int i=0;i<111;i++)
	{
		if(cnt[a][i]<cnt[b][i])
			return 0;
	}
	return 1;
}
int n;
bool ok(int i)
{
	for(int j=0;j<n;j++)
		if(i!=j&&cont(i,j))
			return 0;
	return 1;
}
int main()
{
	
	cin>>n;
	for(int i=0;i<n;i++)
	{
		int m;
		cin>>m;
		while(m--)
		{
			int x;cin>>x;
			cnt[i][x]++;
		}
	}

	if(ok(i))cout<<("YES\n");
	else cout<<("NO\n");
}
