#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

using namespace std;

int n,s[105][105],res[105];

int main(){
  scanf("%d",&n);
  FI(i,1,n) res[i]=true;
  FI(i,1,n){
    int x,y;
    scanf("%d",&x);
    while(x--){
      scanf("%d",&y);
      s[i][y]++;
    }
  }
  FI(i,1,n) FI(j,1,n){
    if(i==j) continue;
    bool ok=false;
    FI(k,1,100) if(s[j][k] && !s[i][k]) ok=true;
    res[i]&=ok;
  }
  FI(i,1,n) puts(res[i]?"YES":"NO");
  return 0;
}
