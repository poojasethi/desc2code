#include<iostream>
using namespace std;
int a[101][101]={};
int main()
{
	int n,i,j,x,y,k;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>x;
		while(x--)
		{
			cin>>y;
			a[i][y]=1;
		}
	}
	for(i=0;i<n;i++)
	{
		int canwin=1;
		for(j=0;j<n;j++)
		{
			if(i==j)continue;
			int issub=1;
			for(k=1;k<=100;k++)
			{
				if(a[j][k] && !a[i][k])
				{
					issub=0;
					break;
				}
			}
			if(issub)
			{
				canwin=0;
				break;
			}
		}
		if(canwin)cout<<"YES\n";
		else cout<<"NO\n";
	}
	return 0;
}
