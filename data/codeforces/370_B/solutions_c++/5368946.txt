#include<cstdio>
#include<cstring>
int n, m, i, j, k, a, f[101][101];
bool mk, flag;
int main(){
	while(~scanf("%d", &n)){
		memset(f, 0, sizeof(f));
		for(i=0; i<n; i++){
			scanf("%d", &m);
			for(j=0; j<m; j++){
				scanf("%d", &a);
				f[i][a]++;
			}
		}
		for(i=0; i<n; i++){
			flag=1;
			for(j=0; j<n; j++){
				if(i!=j){
					mk=0;
					for(k=0; k<=100; k++){
						if(f[j][k]>f[i][k])	mk=1;
					}
					if(!mk){
						flag=0;
						break;
					}
				}
			}
			if(!flag)	puts("NO");
			else	puts("YES");
		}
	}
	return 0;
}
