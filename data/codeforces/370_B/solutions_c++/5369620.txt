#include<iostream>
using namespace std;
int main()
{

int n;
int cards[100][100];
int m;
for(int i=0;i<100;i++) for(int j=0;j<100;j++) cards[i][j]=0;

cin>>n;
for(int i=0;i<n;i++)
{
	cin>>m;
	for(int j=0;j<m;j++)
	{
		int no;
		cin>>no;
		no-=1;
		cards[i][no]=1;
	}
}

for(int i=0;i<n;i++)
{
	int ans=1;
	for(int j=0;j<n;j++)
	{
		if(j!=i)
		{
			int tempans=0;
			for(int k=0;k<100;k++)
			{
				if( cards[i][k]!=1 && cards[j][k]==1 )
				{
					tempans=1;
					break;
				}
			}
			if(tempans==0)
			{
				ans=0;
				break;
			}
		}
		
	}
	if(ans==1) cout<<"YES\n";
	else cout<<"NO\n";
}

return 0;
}
