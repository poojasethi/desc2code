#include <cstdio>
#include <bitset>

#define N 100

using namespace std;

int n, k, c, i, j;
bitset <N + 3> b[N];

int main()
{
	scanf("%d", &n);

	for (i = 0; i < n; i ++)
	{
		scanf("%d", &k);
		for (j = 0; j < k; j ++)
		{
			scanf("%d", &c);
			b[i].set(c - 1);
		}
	}

	for (i = 0; i < n; i ++)
	{
		c = 1;
		for (j = 0; j < n; j ++)
			if (i != j && (b[i] & b[j]) == b[j])
			{
				c = 0;
				break;
			}
		puts(c ? "YES" : "NO");
	}

	return 0;
}
