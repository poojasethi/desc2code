#include <algorithm>
#include <iostream>
using namespace std;

int n;
bool a[110][110];

bool check(int d) {
	for (int i = 0; i < n; i++)
		if (i != d) {
			bool flag = false;
			for (int j = 0; j < 100; j++)
				if (!a[d][j] && a[i][j])
					flag = true;
			if (!flag)
				return false;
		}
	return true;
}
int main() {
	cin >> n;
	for (int i = 0; i < n; i++) {
		int m; cin >> m;
		for (int j = 0; j < m; j++) {
			int t; cin >> t;
			a[i][t - 1] = true;
		}
	}
	for (int i = 0; i < n; i++)
		cout << (check(i)? "YES": "NO") << '\n';
	return 0;
}
