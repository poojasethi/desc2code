#include <cstdio>
int n,t;
int m[101];
int nums[101][101];
int temp[101];

int main(void)
{
	scanf("%d",&n);
	for(int i=0;i<n;++i) {
		scanf("%d",&m[i]);
		for(int j=0;j<m[i];++j) {
			scanf("%d",&t);
			nums[i][t]++;
		}
	}
	
	for(int i=0;i<n;++i) {
		bool fin=true;
		for(int j=0;j<n;++j) temp[j]=m[j];
		for(int j=0;j<=100;++j)
			if(nums[i][j]) {
				for(int k=0;k<n;++k)
					if(k==i) continue;
					else if(nums[k][j]) {
						temp[k]--;
						if(temp[k]==0) fin=false;
					}
			}
		printf((fin)?"YES":"NO");
		if(i<n-1) printf("\n");
	}
	return 0;
}
