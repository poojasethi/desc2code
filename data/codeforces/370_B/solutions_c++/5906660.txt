#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
int n, sz, q, has[200][200];
int fl, ok;
int main(){

cin>>n;
for (int i = 1;i <= n; ++i)
{
 cin>>sz;
 for (int j = 1; j <= sz; ++j)
 {
   cin>>q;
   has[i][q]=1;
 }
}

for (int i = 1; i <= n; ++i)
{
    ok = 1;
    for ( int j = 1; j <= n; ++j)
    {
     if (j == i)continue;
     fl = 0;
     for (int q = 1;q <= 100; ++q)
     if ( has[j][q] == 1 && has[i][q] == 0) fl = 1;
     if (fl == 0) ok = 0;
    }
    if (ok)cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
}
return 0;
}
