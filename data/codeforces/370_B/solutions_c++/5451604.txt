#include <iostream>

using namespace std;

int a[101][101], n, k, x, c, arr[101];

int main()
{
	cin >> n;
	
	for (int i = 1; i <= n; i++)
	{
		cin >> k;
		for (int j = 1; j <= k; j++)
			cin >> x, a[i][x] = 1;
	}
	
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= n; j++)
		{
			if (i != j)
			{
				for (int k = 1; k <= 100; k++)
					if (a[j][k] == 1 && a[i][k] == 0)
						c = 1;
				if (c == 0)
					arr[i] = 1;
				c = 0;
			}
		}
	}
	for (int i = 1; i <= n; i++)
	{
		if (arr[i] == 1)
			cout << "NO" << endl;
		else
			cout << "YES" << endl;
	}
	
	return 0;
}
