#include <cstdio>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n) ;
	if(n<3)
		printf("-1") ;
	else if(n==3)
		printf("210") ;
	else
	{
		int k=1 ;
		for(int i=0;i<n-2;i++)
			k=(k*10)%21 ;
		int d=21-k ;
		printf("1");
		if(d<10)
			printf("0");
		for(int i=0;i<n-4;i++)
			printf("0") ;
		printf("%d0",d) ;
	}
}
