
#include <iostream>

using namespace std;

string a[6]={"50", "80", "170", "20", "200", "110"};

int main()
{
	int n;
	cin>>n;
	if(n<3) { cout<<-1<<endl; return 0; }
	if(n==3) { cout<<210<<endl; return 0; }
	int x=(n-4)%6;
	cout<<1;
	for(int i=n-a[x].size()-1; i>0; i--) cout<<0;
	cout<<a[x]<<endl;
	return 0;
}
