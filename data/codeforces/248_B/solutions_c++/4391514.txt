#include <iostream>
using namespace std;
int main() {
    int n,i,x=1;
    cin>>n;
    if (n<3) { cout<<"-1"; return 0; }
    if (n==3) { cout<<"210"; return 0; }
    for (i=1;i<n;i++) 
     x=(x*10) % 210;
    cout<<1; for (i=1;i<n-3;i++) cout<<0;
    if (210-x<100) cout<<0<<210-x; else cout<<210-x; 
}
