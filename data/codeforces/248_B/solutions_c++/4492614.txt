#include<cstdio>
const char a[6][4] = {"050", "080", "170", "020", "200", "110"};

int main()
{
	int n;
	scanf("%d", &n);
	if (n < 3)
		puts("-1");
	else if (n == 3)
		puts("210");
	else
	{
		putchar('1');
		n -= 4;
		for (int i = 0; i < n; ++i)
			putchar('0');
		puts(a[n % 6]);
	}
	return 0;
}
