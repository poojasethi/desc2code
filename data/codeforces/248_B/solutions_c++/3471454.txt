#include<stdio.h>
int main()
{
	int ans[8]={20,200,110,50,80,170};
	int n;
	while(scanf("%d",&n)!=EOF)
	{
		if(n<3) printf("-1\n");
		else if(n==3) printf("210\n");
		else
		{
			if(n>3);
				printf("1");
			for(int j=n-4;j>0;j--) printf("0");
			printf("%03d\n",ans[(n-1)%6]);
		}
	
	}
	return 0;
}
