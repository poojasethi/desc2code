#include <iostream>
using namespace std;
long long n, m, s[1000+100], t[1000+100],ans;
char c[1000+100][1000+100];
int main(){
std::cin >> n >> m;
for(int i=0;i<n;i++)
for(int j=0;j<m;j++)
{std::cin>>c[i][j];
if(c[i][j]=='*')
{
s[i]++;
t[j]++;
}
}
for(int i=0;i<n;i++)
for(int j=0;j<m;j++)
if(c[i][j]=='*')
{
ans+=(s[i]-1)*(t[j]-1);
//cout<<ans<<endl;
}
std::cout<<ans;
return 0;
}
