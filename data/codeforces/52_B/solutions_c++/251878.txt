#include <cstdio>
#include <cmath>
#include <iostream>
using namespace std;


char c[1005][1005];

int main()
{
	int n,m,i,a[1005]={0},b[1005]={0},j;
	__int64 res=0;

	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			cin>>c[i][j];
			//scanf("%c",&q);
			if(c[i][j] == '*')
			{
				a[i]++;
				b[j]++;
			}
		}
	}

	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
			if(a[i] > 0 && b[j] > 0 && c[i][j] == '*')
			res+=(a[i]-1)*(b[j]-1);
	}
	printf("%I64d",res);
	return 0;
}
