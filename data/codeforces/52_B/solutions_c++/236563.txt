#include <iostream>
using namespace std;

int n, m;
char t[1024][1024];
long long ans, r[1024], c[1024];

int main(void) {
	cin >> n >> m;
	for(int i = 0; i < 1024; i++) r[i] = c[i] = 0;
	for(int i = 0; i < n; i++) for(int j = 0; j < m; j++) {
		cin >> t[i][j];
		if(t[i][j] == '*') r[i]++, c[j]++;
	}
	ans = 0;
	for(int i = 0; i < n; i++)
		for(int j = 0; j < m; j++)
			ans += (r[i]-1)*(c[j]-1)*(t[i][j] == '*');
	cout << ans << endl;

	return 0;
}
