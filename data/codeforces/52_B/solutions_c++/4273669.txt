#include <iostream>
#include <cstdio>
#include <string>
using namespace std;
int n,m,i,j;
char a[1001][1001];
int h[1001],v[1001];
long long res;
int main()
{
cin>>n>>m;
for (i=1;i<=n;i++)
{
	for (j=1;j<=m;j++)
	{
	cin>>a[i][j];
	if (a[i][j]=='*')
		h[i]++;
	}
}
for (i=1;i<=n;i++)
{
for (j=1;j<=m;j++)
	{
	if (a[i][j]=='*')
		v[j]+=h[i]-1;
	}
}
for (i=1;i<=n;i++)
for (j=1;j<=m;j++)
	if (a[i][j]=='*')
		res+=v[j]-h[i]+1;
cout<<res;




return 0;
}
