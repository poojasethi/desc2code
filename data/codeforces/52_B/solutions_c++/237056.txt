#include<iostream>

using namespace std;

char arr[1001][1001];

int main()
{
	int row[1001] = {0};
	int col[1001] = {0};
	int n,m;
	cin>>n>>m;
	for (int i=0;i<n;i++)
		for (int j=0;j<m;j++)
		{
			cin>>arr[i][j];
			if (arr[i][j] == '*')
			{
				row[i]++;
				col[j]++;
			}
		}
	
	long long int ans = 0;
	for (int i=0;i<n;i++)
		for (int j=0;j<m;j++)
			if (arr[i][j] == '*')
				ans += (row[i]-1)*(col[j]-1);
	cout<<ans<<endl;
	return 0;
}
