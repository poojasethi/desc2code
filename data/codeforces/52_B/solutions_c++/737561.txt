  #include <iostream>
using namespace std;
    long long n, m, ans, sa[1010], so[1010];
    char c[1010][1010];
    int main()
    { cin>>n>>m;
    for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			cin>>c[i][j];
    for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			if(c[i][j]=='*')
			{	sa[i]++;
				so[j]++;
			}
    for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			if(c[i][j]=='*')
				ans+=(sa[i]-1)*(so[j]-1);


    cout<<ans;
	return 0;
}
