#include <iostream>
#include<cstdio>

using namespace std;
int main(){
  int m, k;
  cin>> m>>k;
  char a [1001][1001];
  int s[1001], t[1001];
  long long tri = 0;
  for(int i = 0; i<m; i++){
    for(int j = 0; j<k; j++){
      cin >>a[i][j];
      if(a[i][j] == '*'){
	s[i]++;
	t[j]++;
      }
    }
  }
for(int i = 0; i<m; i++){
    for(int j = 0; j<k; j++){
      if(a[i][j] == '*') tri+=(s[i]-1)*(t[j]-1);
    }
 }
 printf("%I64d\n", tri);
  return 0;
}
