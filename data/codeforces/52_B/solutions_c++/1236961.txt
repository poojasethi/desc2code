#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

char a[1005][1005];
int r[1005], c[1005];

int main(void) {
	int n, m;
	scanf("%d%d", &n, &m);
	for(int i=0;i<n;i++) {
		scanf("%s", a[i]);
		for(int j=0;j<m;j++)
			if(a[i][j]=='*') ++r[i], ++c[j];
	}
	long long ans=0;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			if(a[i][j]=='*') {
				ans += (r[i]-1)*(c[j]-1);
			}
	printf("%I64d\n", ans);
	return 0;
}
