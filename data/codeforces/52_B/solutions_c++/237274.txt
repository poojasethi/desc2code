#include<iostream>
using namespace std;
int v[1001][1001],a[1001],b[1001],m,n,i,j;
long long x;
char c;
int main()
{
	cin>>m>>n;
	for(i=1;i<=m;i++)
		for(j=1;j<=n;j++){
			cin>>c;
			if(c=='*'){
				v[i][j]=1;
				a[i]++;
				b[j]++;
			}
		}
	for(i=1;i<=m;i++)
		for(j=1;j<=n;j++)
			if(v[i][j]==1)x+=(a[i]-1)*(b[j]-1);
	cout<<x;
	return 0;
}
