#include<iostream>
using namespace std;
const int maxn=1000;
char arr[maxn][maxn];
int a[maxn],b[maxn];
int main(){
	long long m,n,c=0;
	cin>>m>>n;
	for(int i=0;i<m;i++)
		for(int j=0;j<n;j++){
			cin>>arr[i][j];
			if(arr[i][j]=='*'){
				a[i]++;
				b[j]++;
			}
		}
	for(int i=0;i<m;i++)
		for(int j=0;j<n;j++)
			if(arr[i][j]=='*')
					c+=(a[i]-1)*(b[j]-1);
	cout<<c<<endl;
}
