#include<iostream>
#include<stdio.h>
#define FOR(i,x,n) for(lli i=x;i<n;i++)
#define lli long long int
using namespace std;
char c[1001][1001];
lli sR[1001],sC[1001],ans;
int main()
{
  lli n,m; scanf("%lld %lld",&n,&m);
  FOR(i,0,n) 
    FOR(j,0,m) 
    { cin>>c[i][j]; 
      if(c[i][j]=='*') sR[i]++ , sC[j]++;
    }
  FOR(i,0,n) 
    FOR(j,0,m) 
      if(c[i][j]=='*') ans+=(sR[i]-1)*(sC[j]-1); 
  printf("%lld",ans);
}
