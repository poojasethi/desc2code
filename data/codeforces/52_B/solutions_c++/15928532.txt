#include <iostream>

using namespace std;

#define FOR(i, a, b) for(int i = a; i < b; ++i)

long long res = 0;
int n, m;
int r[1010] = {0}, c[1010] = {0};
char field[1010][1010];

int main() {
	ios::sync_with_stdio(false); cin.tie(0);
	cin >> n >> m;
	FOR(i, 0, n) FOR(j, 0, m){ cin >> field[i][j]; if(field[i][j] == '*') {r[i]++; c[j]++;} }
	FOR(i, 0, n) FOR(j, 0, m){ if(field[i][j] == '*') res +=(r[i] - 1) * (c[j] - 1); }
	cout << res;
	return 0;
}
