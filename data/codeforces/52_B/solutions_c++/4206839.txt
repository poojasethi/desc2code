#include<iostream>
using namespace std;
long long ans;
long long a[1050],b[1050];
char str[1050][1050];
long long n,m;
int main(){
    cin>>n>>m;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            {cin>>str[i][j];
            if(str[i][j] == '*'){a[i]++; b[j]++;}
            }
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            if(str[i][j] == '*') ans += (a[i] - 1)*(b[j] - 1);
    cout<<ans<<endl;
    return 0;
}
