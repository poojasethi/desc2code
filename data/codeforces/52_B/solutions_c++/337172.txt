#include <iostream>
#include <cstdio>

using namespace std;

const int maxN= 1000+5;

char mat[maxN][maxN];
int n,m;
int col[maxN], row[maxN];

int main(){
	cin >> n >> m;
	for (int i=1;i<=n;i++)
		for (int j=1;j<=m;j++){
			cin >> mat[i][j];
			row[i]+= (mat[i][j]=='*');
			col[j]+= (mat[i][j]=='*');
		}


	long long res=0;

	for (int i=1;i<=n;i++)
		for (int j=1;j<=m;j++) if(mat[i][j]=='*'){
			res+= (row[i]-1) * (col[j]-1);
		}
	
	cout << res << endl;

	return 0;
}
