#include <iostream>
using namespace std;

const int MAXN = 1000 + 10;
char arr[MAXN][MAXN];
int r[MAXN];
int c[MAXN];

int main()
{
	int n, m;
	cin>> n >> m;

	for(int i = 0; i < n; i ++)
		for(int j = 0; j < m; j ++)
		{
			cin>> arr[i][j];
			if(arr[i][j] == '*')
			{
				r[i] ++;
				c[j] ++;
			}
		}	


	long long ans = 0;
	for(int i = 0; i < n; i ++)
		for(int j = 0; j < m; j ++)
			if(arr[i][j] == '*')
				ans += (r[i] - 1) * (c[j] - 1);

	cout<< ans << endl;


}
