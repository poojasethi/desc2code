#include<iostream>
using namespace std;
int a[1001][1001],m,n,h[1001],c[1001];

main(){
    int i,j,k;
    long long rs=0;
    char ch;

    cin>>m>>n;
    for (i=0;i<m;i++){
        for (j=0;j<n;j++){
            cin>>ch;
            k = (ch=='.')?0:1;
            h[i] += k;
            c[j] += k;
            a[i][j] = k;
        }
    }

    for (i=0;i<m;i++){
        for (j=0;j<n;j++){
            if (a[i][j])
            rs += (h[i]-1)*(c[j]-1);
        }
    }
    cout<<rs;
}
