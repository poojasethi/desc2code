#include <iostream>
#include<string>
using namespace std;

int main()
{
int n,m;cin>>n>>m;string a[1000];
long long r[1000],c[1000],s=0;//cout<<n;
for(int i=0;i<n;i++)
cin>>a[i];


for(int i=0;i<n;i++)
{
    for(int j=0;j<m;j++)
    if(a[i][j]=='*')
    r[i]++;
}
for(int j=0;j<m;j++)
{
    for(int i=0;i<n;i++)
    if(a[i][j]=='*')
    c[j]++;
}
for(int i=0;i<n;i++)
{
    for(int j=0;j<m;j++)
    {
        if(a[i][j]=='*')
        s+=(r[i]-1)*(c[j]-1);
    }
}
cout<<s;
  return 0;
}
