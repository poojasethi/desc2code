#include <iostream>
#define N 1024
typedef long long LL;
using namespace std;
char tbl[N][N];
LL x[N],y[N];
int n,m;
LL ans;
int main(){
  cin >> n >> m;
  for(int i=0;i<n;++i) for(int j=0;j<m;++j) cin >> tbl[i][j];
  for(int i=0;i<n;++i) for(int j=0;j<m;++j) if(tbl[i][j]=='*') x[j]++,y[i]++;
  for(int i=0;i<n;++i) for(int j=0;j<m;++j) if(tbl[i][j]=='*') ans+=(x[j]-1)*(y[i]-1);
  cout << ans << endl;
}
