#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

int n,m;
long long int col[1005],S,ret;
char s[1005][1005];

int main(){

	scanf("%d%d",&n,&m);
	for (int i=0;i<n;i++){
		scanf("%s",s[i]);
		for (int j=0;j<m;j++)
		    if (s[i][j]=='*') col[j]++;
	}
	
	ret=0;
	for (int i=0;i<n;i++){
		S=0;
		for (int j=0;j<m;j++){
			if (s[i][j]=='*') S++;
		}
		for (int j=0;j<m;j++){
			if (s[i][j]=='*') ret += (S-1)*(col[j]-1);
		}
	}
	printf("%I64d\n",ret);
	
	return 0;
}
