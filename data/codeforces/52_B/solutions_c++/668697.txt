#include <iostream>

using namespace std;
const int MAXN = 1000 + 10;
char c[MAXN][MAXN];
int dc[MAXN], dr[MAXN];

int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < m; ++j) {
			cin >> c[i][j];
			if (c[i][j] == '*') {
				dr[i]++;
				dc[j]++;
			}
		}
	long long count = 0;
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < m; ++j)
			if (c[i][j] == '*')	
				count += (dr[i] - 1) * (dc[j] - 1);
	cout << count << endl;
	return 0;
}
