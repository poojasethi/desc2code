#include <stdio.h>

char t[2000][2000];

int h[2000], v[2000];

typedef unsigned long long LL;

int main() {
	int n, m;
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; i++) {
		scanf("%s", t[i]);
	}
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			if (t[i][j] == '*') {
				h[i]++;
				v[j]++;
			}
		}
	}
	
	LL ans = 0;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			if (t[i][j] == '*') {
				ans += (h[i] - 1) * (v[j] - 1);
			}
		}
	}
	
	printf("%I64d\n", ans);
	return 0;
}
