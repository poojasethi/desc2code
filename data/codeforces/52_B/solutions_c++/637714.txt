#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
char c[1010][1010];
long long i, j, res = 0, x[1010],y[1010], n, m;

int main(){
  cin>>m>>n;
  for(i=1;i<=m;i++) for(j = 1; j<=n;j++){
    cin>>c[i][j];
    if(c[i][j] == '*') {x[i]++;y[j]++;}
  }
  for(i = 1; i <= m; i++)for(j=1;j<=n;j++) if (c[i][j] == '*') res += (x[i] - 1)*(y[j] - 1);
  cout<<res;
}
