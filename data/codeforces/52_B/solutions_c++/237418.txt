#include <iostream>
#include <cstdio>
using namespace std;
const int SZ=1005;
int str[SZ],stol[SZ];
char a[SZ][SZ];
int main()
{
  int n,m;
  cin>>n>>m;
  for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
    {
      cin>>a[i][j];
      str[i]+=a[i][j]=='*';
      stol[j]+=a[i][j]=='*';
    }
  long long sum=0;
  for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
      if (a[i][j]=='*')sum+=(str[i]-1)*(stol[j]-1);
  cout<<sum;
  return 0;
}
