#include<cstdio>
#include<cstring>
#define maxn 1010
int n,m;
int hang[maxn],lie[maxn];
long long ans;
char map[maxn][maxn];
int main(){
    int i,j;
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;i++){
        getchar();
        for(j=1;j<=m;j++){
            map[i][j]=getchar();
            if(map[i][j]=='*')hang[i]++,lie[j]++;
        }
    }
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
            if(map[i][j]=='*')ans+=(hang[i]-1)*(lie[j]-1);
    printf("%I64d\n",ans);
}
