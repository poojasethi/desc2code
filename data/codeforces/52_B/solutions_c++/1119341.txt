#include <iostream>
#include <string>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;

int t[1010],y[1010],n,m;
char s[1010][1010];
long long r;


int main(){
  cin>>n>>m;
  rep(i,n){
    cin>>s[i];
    rep(j,m)if(s[i][j]=='*')y[i]++,t[j]++;
  }

  rep(i,n)rep(j,m)if(s[i][j]=='*')r+=(y[i]-1)*(t[j]-1);
  cout<<r<<endl;
  

  return 0;
}
