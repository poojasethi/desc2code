#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char map[1100][1100];
long long x[1100],y[1100],ans;
int n,m;
int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++)
		scanf("%s",map[i]+1);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			if(map[i][j]=='*')
			{
				x[i]++;
				y[j]++;		
			}
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			if(map[i][j]=='*')
				ans+=(x[i]-1)*(y[j]-1);
	printf("%I64d\n",ans);
	return 0;
}
