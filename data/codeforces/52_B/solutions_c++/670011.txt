#include <iostream>
long long n, m, ans, sa[1000+100], so[1000+100];
char c[1000+100][1000+100];
int main(){std::cin >> n >> m;
for(int i=0;i<n;i++)for(int j=0;j<m;j++)std::cin>>c[i][j];
for(int i=0;i<n;i++)for(int j=0;j<m;j++)if(c[i][j]=='*'){sa[i]++;so[j]++;}
for(int i=0;i<n;i++)for(int j=0;j<m;j++)if(c[i][j]=='*')ans+=(sa[i]-1)*(so[j]-1);
std::cout<<ans;}
