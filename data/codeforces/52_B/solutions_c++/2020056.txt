#include <cstdio>
#define LL long long

int n,m,a[1010][2];
char c[1010][1010];
LL ans;

int main() {
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i) {
		getchar();
		for (int j=1;j<=m;++j) {
			c[i][j]=getchar();
			if (c[i][j]=='*') {
				++a[i][0];
				++a[j][1];
			}
		}
	}
	for (int i=1;i<=n;++i)
		for (int j=1;j<=m;++j)
			if (c[i][j]=='*')
				ans=ans+((LL)(a[i][0]-1))*((LL)(a[j][1]-1));
	printf("%I64d\n",ans);
	return 0;
}
