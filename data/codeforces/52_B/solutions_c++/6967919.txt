#include <iostream>
using namespace std;
long long sum[1005];
char in[1005][1005];
int main (){
	int N,M;
	long long ans,a,b;
	cin >> N >> M ;
	for(int i=0;i<N;i++)cin >> in[i];
	for(int i=0;i<N;i++){
		for(int j=0;j<M;j++){
			if(in[i][j] == '*')sum[i]++;
		}
	}
	ans = 0;
	for(int j=0;j<M;j++){
		a = 0;
		b = 0;
		for(int i=0;i<N;i++){
			if(in[i][j] == '*'){
				a+=sum[i]-1;
				b++;
			}
		}
		ans += a*(b-1);
	}
	cout << ans << "\n";
	return 0;
}
