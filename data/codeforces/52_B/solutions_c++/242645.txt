#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>

using namespace std;

const int MAX=1200;
long long row[MAX]={0},col[MAX]={0},ret=0;
char s[MAX][MAX];

int main(){
	int n,m;
	
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++){
		scanf("%s",s[i]);
		for(int j=0;j<m;j++){
			if(s[i][j]=='*'){
				row[i]++;
				col[j]++;
			}
		}
	}
	
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			if(s[i][j]=='*') ret+=(row[i]-1)*(col[j]-1);
		}
	}
	printf("%I64d\n",ret);
	
	return 0;
}
