#include<stdio.h>

__int64 s;
__int64 b[1001]={0},c[1001]={0};
char str[1001][1001];

main()
{
  int n,m,i,j;
  scanf("%d%d",&n,&m);
  for(i=0;i<n;i++)
  {
    scanf("%s",&str[i]);
    for(j=0;j<m;j++)
    if(str[i][j]=='*')
    {
      b[i]++;
      c[j]++;
    }
  }
  s=0;
  for(i=0;i<n;i++)
    for(j=0;j<m;j++)
      if(str[i][j]=='*') s+=(b[i]-1)*(c[j]-1);
  printf("%I64d\n",s); 
}
