#include <iostream>
long long n, m, ans, s[1000+100], t[1000+100];

if(c[i][j]=='*'){s[i]++;t[j]++;}}
for(int i=0;i<n;i++)for(int j=0;j<m;j++)if(c[i][j]=='*')ans+=(s[i]-1)*(t[j]-1);
std::cout<<ans;}
