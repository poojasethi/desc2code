#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
using namespace std;
int i,n,m,col[1200],row[1200],j;
long long ans;
char c[1200][1200];
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i){
		scanf("%s",c[i]+1);
		for(j=1;j<=m;++j){
			row[i]=row[i]+(c[i][j]=='*');
			col[j]=col[j]+(c[i][j]=='*');
		}
	}
	for(i=1;i<=n;++i)
		for(j=1;j<=m;++j)if(c[i][j]=='*')
			ans+=1ll*(row[i]-1)*(col[j]-1);
	printf("%I64d\n",ans);
}
