#!/usr/bin/env python
# coding = utf-8
import string

x, y = values = map(int, string.split(raw_input()))

field = []
a = [0] * x
b = [0] * y
for i in range(x):
	string = raw_input()

	field.append([0] * y)
	for j in range(y):
		if string[j] == '*':
			field[i][j] = 1
			a[i] += 1
			b[j] += 1

result = 0
for i in range(x):
	for j in range(y):
		if field[i][j] == 1:
			result += (a[i] - 1) * (b[j] - 1)		

print result
