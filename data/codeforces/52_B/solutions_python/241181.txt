n, m = map(int, raw_input().split())

pole = []
r = [0] * n
c = [0] * m
for i in xrange(n):
    nstr = raw_input().rstrip()
    pole.append(nstr)
    for j in xrange(m):
        if nstr[j] == '*':
            r[i] += 1
            c[j] += 1

res = 0
for i in xrange(n):
    for j in xrange(m):
        if pole[i][j] == '*':
            res += (r[i] - 1) * (c[j] - 1)
            
print res
