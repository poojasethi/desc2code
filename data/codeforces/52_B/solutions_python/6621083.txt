import time
import sys, io
import re, math
import itertools
#start = time.clock()
(h,w)=map(int, raw_input().split())
f,memo=[],[]
ans=0

for i in range(h):
    f.append(raw_input())
#縦にいくつ*があるか、いくつ頂点に出来るかメモ
for jk in range(w):
    tmp=0
    for jd in range(h):
        if f[jd][jk]=='*': tmp+=1
    memo.append(tmp)

for jc in range(h):
    an=[]
    cnt=f[jc].count('*')
#横に2つ以上*があれば個数-1通りの底辺が作れるのと、
#縦に2つ以上*があるかメモから見ていくつ三角を作れるか掛け算
    if cnt>1:
        for jy in range(w):
            if f[jc][jy]=='*':
                ans+=((memo[jy]-1)*(cnt-1))

print ans
