n,m=map(int,raw_input().split())
l,chk=[],[]
ans=0
for i in range(n):
    l.append(raw_input())
    
for i in range(m):
    cnt=0
    for j in range(n):
        if l[j][i]=='*':
            cnt+=1
    chk.append(cnt)
for i in range(n):
    c=l[i].count('*')
    if c>1:
        for j in range(m):
            if l[i][j]=='*' and chk[j]>1:
                ans+=((chk[j]-1)*(c-1))
print ans
