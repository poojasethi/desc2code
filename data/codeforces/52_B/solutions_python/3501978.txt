

def main():
	n,m = [int(i) for i in raw_input().split()]
	h = [0]*n
	v = [0]*m
	strs = []
	for i in xrange(n):
		strs.append(raw_input())
	for i in xrange(n):
		for j in xrange(m):
			if strs[i][j]=='*':
				h[i]+=1
				v[j]+=1
	res=0
	for i in xrange(n):
		for j in xrange(m):
			if strs[i][j]=='*':
				res+= (h[i]-1)*(v[j]-1)
	print res
	


main()
