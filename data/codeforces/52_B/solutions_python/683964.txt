K = map(int, raw_input().strip().split())
n = K[0]
m = K[1]

cnt1 = [0 for i in range(n)]
cnt2 = [0 for i in range(m)]

L = [ ]

for i in range(n):
    L.append(raw_input().strip())
    for j in range(m):
        if L[i][j] == '*':
            cnt1[i] = cnt1[i] + 1
            cnt2[j] = cnt2[j] + 1

answer = 0

for i in range(n):
    for j in range(m):
        if L[i][j] == '*':
            answer = answer + (cnt1[i] - 1) * (cnt2[j] - 1)

print answer
