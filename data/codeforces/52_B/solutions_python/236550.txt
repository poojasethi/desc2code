#!/usr/bin/python
import sys
n, m = [int (x) for x in sys.stdin.readline ().split ()]
a = []
for i in range (n):
	a += [sys.stdin.readline ().strip ()]

d = [0] * n
e = [0] * m
for i in range (n):
	for j in range (m):
		if a[i][j] == '*':
			d[i] += 1
			e[j] += 1

res = 0
for i in range (n):
	for j in range (m):
		if a[i][j] == '*':
			res += (d[i] - 1) * (e[j] - 1)
print res
