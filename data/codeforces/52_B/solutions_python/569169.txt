s = raw_input().split()
n = int(s[0])
m = int(s[1])
p = [0]*m
q = [0]*m
r = [0]*m
ans = 0

for i in range(n):
	s = raw_input()
	j = x = y = 0
	for z in s:
		if z == '*':
			ans += x*p[j]+q[j]
			q[j] += x
			x += 1
			#y += p[j]
			p[j] += 1
		j += 1
	j = m-1
	x = 0
	for z in s[::-1]:
		if z == '*':
			ans += r[j]+x*(p[j]-1)
			r[j] += x
			x += 1
		j -= 1

print ans
