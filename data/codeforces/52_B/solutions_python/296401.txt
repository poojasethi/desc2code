mesh = []
row = [0] * 1001
col = [0] * 1001
sol = 0

n, m = [int(i) for i in raw_input().split()]
for i in range(n):
    mesh.append([])
    for j in range(m):
        mesh.append(m + n)

for i in range(n):
    mesh[i] = raw_input()

for i in range(n):
    for j in range(m):
        if mesh[i][j] == "*":
            row[i] += 1
            col[j] += 1

for i in range(n):
    for j in range(m):
        if mesh[i][j] == "*":
            sol += (row[i] - 1)*(col[j] - 1)

print sol
