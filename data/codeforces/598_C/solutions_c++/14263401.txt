#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>

#define PI acos(-1.0)

using namespace std;

struct vec {
	int id;
	long double angle;
	bool operator < (const vec& other) const {
	    return angle < other.angle;
	}
	
};
vec v[100005];
int n;

int main() {
	int i, xc, yc;
	scanf("%d", &n);
	for (i = 0; i < n; i++) {
		scanf("%d%d", &xc, &yc);
		v[i].id = i+1;
		v[i].angle = atan2(yc, xc);
	}
	sort(v, v+n);
	long double ans = 1000000;
	long double tt;
	v[n] = v[0];
	int a, b;
	for (i = 1; i <= n; i++) {
		tt = v[i].angle - v[i-1].angle;
		if (tt < 0) {
			tt = -tt;
		}
		tt = min(tt, 2 * PI - tt);
		if (tt < ans) {
			ans = tt;
			a = min(v[i].id, v[i-1].id);
			b = max(v[i].id, v[i-1].id);
		}
	}
	printf("%d %d\n", a, b);
	return 0;
}
