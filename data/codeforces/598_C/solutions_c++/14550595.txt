#include<cstdio>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
using namespace std;
struct vec
{
    int x,y;
    long double k;
    int id;
}a[100010];
bool cmp(vec a,vec b)
{
    return a.k<b.k;
}
int main()
{
//freopen("1.txt","r",stdin);
    int n;
    cin>>n;
    for (int i=1;i<=n;i++)
    {
        cin>>a[i].x>>a[i].y;
        long double angle=acosl(a[i].y/sqrt(a[i].x*a[i].x+a[i].y*a[i].y))/acosl(-1)*180;
        if (a[i].x<0) a[i].k=360-angle;
            else a[i].k=angle;
        a[i].id=i;
    }
    sort(a+1,a+n+1,cmp);
    long double mina=100000;
    int ans=1;
    for (int i=1;i<=n;i++)
    {
        if (fabs(a[i+1].k-a[i].k)<mina)
        {
            mina=fabs(a[i+1].k-a[i].k);
            ans=i;
        }
    }
    if (a[1].k+360-a[n].k<mina) ans=n;
    cout<<a[ans].id<<' '<<a[ans%n+1].id;

}
