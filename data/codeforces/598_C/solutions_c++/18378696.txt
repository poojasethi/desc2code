#include<bits/stdc++.h>

using namespace std;

#define st first
#define nd second
#define openfile ({freopen("circles.inp", "r", stdin); freopen("circles.out", "w", stdout);})

typedef pair<int,int> II;
typedef long long LL;

const long double pi=2*acos(0);
const int N=111111;
const int MOD=1e9+7;

int n;
int x,y;
pair<long double,int> a[N];

main()
{
  long double best=181;
  cin >> n;
  for(int i=0;i<n;++i)
  { cin >> x >> y;
    a[i]=make_pair(atan2(x,y)*180/pi,i+1);
    //cout << a[i].st << endl << endl;
  }
  sort(a,a+n);
  int id1=a[0].nd,id2=a[1].nd;
  for(int i=1;i<n;++i)
   if(best>a[i].st-a[i-1].st)
   { best=a[i].st-a[i-1].st;
     id1=a[i].nd;
     id2=a[i-1].nd;
   }
  if(best>(360-a[n-1].st+a[0].st)) return cout << a[0].nd << " " << a[n-1].nd,0;
  cout << id1 << " " << id2;
}
