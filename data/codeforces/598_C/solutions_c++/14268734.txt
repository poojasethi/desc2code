#include<bits/stdc++.h>
#define ll long long int
#define PI (long double)(acosl(-1.0))
#define EPSILON pow(2.00,-52.00)
#define angle(y,x) atan2l(y,x)

using namespace std;


int main(){
	
	int n,i,f,s,u,v;
	cin>>n;

	pair<long double,int> a[n];

    long double x[n],y[n],p,q,m;

	for(i=0;i<n;i++){
		cin>>u>>v;
		
		x[i]=u;
		y[i]=v;
		q=angle(y[i],x[i]);
		a[i].first=q; 

		a[i].second=i+1;
	}

	sort(a,a+n);

	
    m=1e10;

	for(i=0;i<n-1;i++){
       
       if(a[i+1].first-a[i].first<m){
       	  f=a[i+1].second;
       	  s=a[i].second;
       	  m=a[i+1].first-a[i].first;
       }
	}

	if((long double)(2*PI)+a[0].first-a[n-1].first<m){
		f=a[0].second;
		s=a[n-1].second;
	}

	cout<<f<<" "<<s<<endl;

   return 0;
}
