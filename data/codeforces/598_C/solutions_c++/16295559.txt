#include<iostream>
#include<stdio.h>
#include<vector>
#include<cstring>
#include<stdlib.h>
#include<algorithm>
#include<cmath>
using namespace std;
const int maxn=100005;

int id[maxn];
long double ang[maxn];

bool cmp(int i, int j)
{
	return ang[i]<ang[j];
}

int main()
{
	int n;
	cin>>n;
	int u=0;
	for (int i=0;i<n;i++){
		int x,y;
		id[i]=i;
		cin>>x>>y;
		ang[i]=atan2(y,x);
	}
	sort(id,id+n,cmp);
	long double min=1e11;
	int x,y;
	for (int i=0;i<n;i++){
		long double tmp=ang[id[(i+1)%n]]-ang[id[i]];
		if (tmp<0) tmp+=2*acos(-1);
		if (min>tmp){
			min=tmp;
			x=id[(i+1)%n];
			y=id[i];
		}
	}
	cout<<x+1<<" "<<y+1;
	return 0;
}
