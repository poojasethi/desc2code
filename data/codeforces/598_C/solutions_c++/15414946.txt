#include <bits/stdc++.h>
using namespace std;
typedef long double ld;
const int maxn = 1e5+10;
const long double PI = acos((ld) -1.0);
pair<ld, int >V[maxn];
int main(){
    int T;
    cin >> T;
    for(int i = 0 ; i < T ; i ++){
        int x,y;
        scanf("%d %d", &x,&y);
        ld tmp = atan2((ld)x,(ld)y);
        if(tmp < 0.0)
            tmp += 2*PI;
        V[i] = make_pair(tmp,i);
    }
    sort(V,V+T);
    ld mi = 2*PI + 1;
    int ans = 0;
    for(int i = 0 ; i < T ; i ++){
        ld tmp = fabs(V[(i+1)%T].first - V[i].first);
        tmp = min(tmp,2*PI - tmp);
        if(tmp < mi){
            ans = i;
            mi = tmp;
        }
    }
    cout << V[(ans)].second+1 << " " << V[(ans+1)%T].second+1;
}
