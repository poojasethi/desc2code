#include<iostream>
#include<math.h>
#include<algorithm>

struct point
{
	long double angle;
	int pos;
};

bool cmp(point a,point b)
{
	return a.angle<b.angle;
}

point p[100000];

int main()
{
	int n;
	std::cin >> n;
	for(int i=0; i<n; i++)
	{
		int x,y;
		std::cin >> x >> y;
		p[i].angle = atan2(1.0*x, 1.0*y);
		p[i].pos = i+1;
	}
	std::sort(p,p+n,cmp);
	int x,y;
	long double min=10*acos(-1);
	for(int i=0;i<n;i++)
	{
		long double tmp=p[(i+1)%n].angle-p[i].angle;
		if(tmp<0) tmp+=2*acos(-1);
		if(tmp<min)
		{
			min=tmp;
			x=p[i].pos;
			y=p[(i+1)%n].pos;
		}
	}
	std::cout << x << " " << y << std::endl;
	return 0;
}
