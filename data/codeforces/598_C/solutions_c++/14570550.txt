#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#include<cstdio>
#include<iostream>
#include<queue>
#define LiangJiaJun main
using namespace std;
int n;
struct mult{
    int sit;
    long double k;
} a[100004];
inline bool dex(mult A,mult B){return A.k<B.k;}
long double mint ;
int s1,s2;
int LiangJiaJun(){
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        int x ,y;
        scanf("%d%d",&x,&y);
        a[i].sit = i;
        a[i].k = atan2(y,x);
    }
    sort(a+1,a+n+1,dex);
    s1=a[1].sit;s2=a[n].sit;
    mint = a[1].k + 2*acos(-1) - a[n].k;
    for(int i=2;i<=n;i++){
        long double now = a[i].k-a[i-1].k;
        if(now < mint){
            mint = now ;
            s1 = a[i].sit;
            s2 = a[i-1].sit;
        }
    }
    printf("%d %d\n",s1,s2);
    return 0;
}
