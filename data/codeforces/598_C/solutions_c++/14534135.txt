#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>

using namespace std;

int n,i,p,q;
long double x,y,mn,t;
const long double PI=acos((long double) (-1));

struct Node {
    long double d;
    int id;
    bool operator <(const Node& a) const {
        return d<a.d;
    }
}v[100005];

int main() {
	while(cin>>n) {
        for(i=1;i<=n;++i) {
            cin>>x>>y;
            v[i].id=i;
            v[i].d=atan2(y,x);
        }
        sort(v+1,v+n+1);
        mn=2*PI-v[n].d+v[1].d;
        p=v[1].id;
        q=v[n].id;
        for(i=1;i<n;++i) {
            t=v[i+1].d-v[i].d;
            t=min(t,2*PI-t);
            if(t<mn) {
                mn=t;
                p=v[i].id;
                q=v[i+1].id;
            }
        }
        printf("%d %d\n",p,q);
	}
	return 0;
}
