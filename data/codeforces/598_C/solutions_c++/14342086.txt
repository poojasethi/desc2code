#include <bits/stdc++.h>

using namespace std;

typedef long double ldbl;
typedef pair<int, int> pii;
typedef pair<ldbl, int> pdi;
#define mp make_pair
#define fi first
#define id second
#define pb push_back

const ldbl PI = acos((ldbl)-1);

int main() {
  int n; cin >> n;
  vector<pdi> V;
  for (int i = 1; i <= n; ++i) {
    ldbl x, y; cin >> x >> y;
    V.pb(mp(atan2(y, x), i));
  }
  sort(V.begin(), V.end());
  V.pb(V[0]); V[n].fi += 2*PI;

  ldbl mindfi = 2*PI;
  pii ans = mp(-1,-1);
  for (int i = 0; i < n; ++i) {
    ldbl dfi = V[i+1].fi - V[i].fi;
    if (ans == mp(-1,-1) || dfi < mindfi) {
      mindfi = dfi;
      ans = mp(V[i].id, V[i+1].id);
    }
  }
  cout << ans.first << " " << ans.second << endl;

  return 0;
}
