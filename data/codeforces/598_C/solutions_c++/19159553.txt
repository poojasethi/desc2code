#include <iostream>
#include <cmath>
#include <vector>
#include <algorithm>
#include <utility>

using namespace std;

#define PI  3.1415926535897932384626433

int main ()
{
	int n;
	cin >> n;
	vector <pair<long double, int> > A(n);

	long double x, y;
	for (int i = 0; i < n; i++) {
		cin >> x >> y;
		A[i].first = atan2(y, x);
		A[i].second = i + 1;
	}

	sort(A.begin(), A.end());

	int value = 0;
	long double min = 1000;
	long double temp;
	for (int i = 0; i < n; i++) {
			temp = A[(i+1)%n].first - A[i].first;
			if (temp < 0)
				temp += (2*PI); 
			if (min > temp) {
				min = temp;
				value = i;
			}
	}

	cout << A[value].second << " " << A[(value + 1)%n].second;
	return 0;	
}
