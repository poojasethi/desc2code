#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <vector>
#include <queue>
using namespace std;
typedef long double dou;
const int N=1E5+5;
const dou PI=acos(-1);
struct map{
    int x,y,p;
    dou k;
}a[N];
int n,p;
dou ang,mini=2*PI;

int comp(const map&a,const map&b)
{
    //return a.y*b.x<a.x*b.y;
    return a.k<b.k;
}

int main()
{
    cin>>n;
    for (int i=1;i<=n;i++) 
    {
        scanf("%d%d",&a[i].x,&a[i].y);
        a[i].p=i; a[i].k=atan2(a[i].y,a[i].x);
    }
    sort(a+1,a+n+1,comp);
    a[0]=a[n]; 
    for (int i=0;i<n;i++)
    {
        ang=abs(a[i].k-a[i+1].k);
        if (ang>PI) ang=2*PI-ang;
        if (mini>ang)
        {
            mini=ang;
            p=i;
        }
    }
    cout<<a[p].p<<" "<<a[p+1].p;
}
