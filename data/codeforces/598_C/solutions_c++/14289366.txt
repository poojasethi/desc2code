#include <bits/stdc++.h>
using namespace std;
#define f first
#define s second
#define pb push_back
#define mp make_pair
#define PI 3.1415926535897932384626433832795028
int main() {
	int i,m,n,q;
	long double x,y,o,p,c,mi=10000000.0;
	vector<pair<long double,int> > vec;
	cin>>n;
	for(i=0;i<n;i++) {
		cin>>x>>y;
		//	if(x==0) 
		//		x=0.0000000001;
		o=fabs(x);
		p=fabs(y);
		if(x>=0&&y>=0) c=atan(p/o);
		else if(x<=0&&y>=0)  c=PI-atan(p/o);
		else if(x<=0&&y<=0)  c=PI+atan(p/o);
		else  c=2*PI-atan(p/o);
		vec.pb(mp(c,i+1));
//		cout<<c<<endl;
	}
	sort(vec.begin(),vec.end());
	for(i=0;i<n-1;i++)
		if(vec[i+1].f-vec[i].f<mi) {
			mi = vec[i+1].f-vec[i].f;
			m =vec[i].s;
			q = vec[i+1].s;
		}
	if(2*PI-vec[n-1].f+vec[0].f<mi)
	{
		mi = vec[n-1].f-vec[0].f;
		m =vec[n-1].s;
		q = vec[0].s;
	}
	cout<<m<<" "<<q<<endl;
	return 0;
}
