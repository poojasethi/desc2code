#include<bits/stdc++.h>
using namespace std;

#define pi 2*acos(0.0)
#define eps 1e-9
typedef long long ll;

pair< long double , int > inp[100005];

int a,b;

int main(){
    int n; cin >> n;
    for(int i=0;i<n;i++){
        double x,y; cin >> x >> y;
        inp[i] = make_pair(atan2(y,x), i+1);
    }
    sort(inp,inp+n);
    inp[n] = inp[0];
    long double theta = 2 * pi - fabs(inp[0].first - inp[n-1].first);
    a = inp[0].second; b = inp[n-1].second;
    for(int i=0;i<n-1;i++){
        long double tmp = fabs(inp[i+1].first - inp[i].first);
        //tmp = min(tmp , 2 * pi - tmp);
        if(tmp < theta){
            a = inp[i].second; b = inp[i + 1].second;
            theta = tmp;
        }
    }
    cout << a << " " << b << '\n';
    return 0;
}
