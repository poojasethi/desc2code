#include<iostream>
#include<algorithm>
#include<cmath>

#define mp make_pair

using namespace std;

const int maxn = 100000+10;
pair<long double,int> vecs[maxn];

int main(){

	int n,x,y;
	cin >> n;
	for(int i=0;i<n;i++){
		cin >> x >> y;
		vecs[i] = mp(atan2((long double)y, x),i+1);
	}
	sort(vecs,vecs+n);
	long double temp;
	long double temp2;
	long double ans = 1111111111111;
	pair<int,int> anz;
	for(int i=0;i<n;i++){
		temp = abs(vecs[i].first-vecs[(i+1)%n].first);
		temp2 = min(temp,2*M_PI-temp);
		if(temp2 < ans){
			ans = temp2;
			anz = mp(vecs[i].second,vecs[(i+1)%n].second);
		}
	}
	cout << anz.first << " " << anz.second;
	return 0;
}
