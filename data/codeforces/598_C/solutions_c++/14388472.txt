#include <iostream>
#include <cmath>
#include <algorithm>
#include <utility>

using namespace std;

int main() {
    int n;
    cin >> n;

    pair<long double, int> angles[100000];

    for(int i = 0; i < n; i++) {
        long double x, y;
        cin >> x;
        cin >> y;

        angles[i].first = atan2(y, x);
        angles[i].second = i+1;
    }

    sort(angles, angles + n);    

    long double min_dif = abs(angles[n-1].first - angles[0].first - 8*atan(1));
    int mini1 = angles[0].second;
    int mini2 = angles[n-1].second;

    for(int i = 1; i < n; i++) {
        long double dif = angles[i].first - angles[i-1].first;
        if(dif < min_dif) {
            min_dif = dif;
            mini1 = angles[i-1].second;
            mini2 = angles[i].second;
        }
    }

    cout << mini1 << " " << mini2 << endl;

    return 0;
}
