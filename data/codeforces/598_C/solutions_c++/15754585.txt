#include <iostream>
#include <algorithm>
#include <cmath>
#include <iomanip>
using namespace std;
int n;
long double M__PI = 3.141592653589793238462;
pair < long double, int > arr[100024];
int main () {
    cin >> n;
    for ( int i = 0 ; i < n ; i ++ ) {
        int x, y;
        cin >> x >> y;
        arr[i] = make_pair( atan2(y,x), i+1 );
        if(arr[i].first < 0 ) { arr[i].first += 2*M__PI;}
    }
    sort( arr, arr+n);

    arr[n] = arr[0];
    arr[n].first += 2*M__PI;
    long double cAng = 10;
    int minInd;
    for ( int i = 0 ; i < n ; i ++ ) {
        if ( cAng >= arr[i+1].first-arr[i].first ) {
            cAng = arr[i+1].first-arr[i].first;
            minInd = i;
        }
    }
    cout << arr[minInd].second << " " << arr[minInd+1].second << "\n";
    return 0;
}
