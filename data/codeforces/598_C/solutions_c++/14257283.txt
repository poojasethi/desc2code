#include<iostream>
#include<cmath>
#include<cstring>
#include<cstdio>
#include<algorithm>


using namespace std; 

typedef long double ld; 
	
const ld PI=(ld)acosl( ld(-1) ); 
	
struct node{ 
	int id; 
	ld angle; 
}P[100003]; 

bool cmp(node ta,node tb){ 
	return ta.angle<tb.angle; 
}

int main(){ 
	int n; 
	scanf("%d",&n);
	for(int i=0;i<n;i++){
	    int x,y; 
		scanf("%d %d",&x,&y); 
		P[i].id=i; 
		P[i].angle=atan2l(y,x); 
	}
	sort(P,P+n,cmp); 
    P[n]=P[0]; 
	P[n].angle+=(ld)2* PI; 
	int x,y; 
	ld ans=1e10; 
	for(int i=0;i<n;i++) { 
		ld t=P[i+1].angle-P[i].angle; 
		if (t<ans) {
			ans=t; 
			x=P[i].id+1; 
			y=P[i+1].id+1; 
		}
	}
	printf("%d %d\n",x,y); 
	return 0; 
}
