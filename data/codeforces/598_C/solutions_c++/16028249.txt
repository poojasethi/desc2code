#include<bits/stdc++.h>
using namespace std;
vector< pair<long double, int> > v ;
int main()
{
    int i, j, k, n, p, q, id=-1;
    long double   m, a, b, angle, mn=100000.0;
    double x, y;
    cin>>n;
    for(i=1;i<=n;i++){
        scanf("%lf %lf", &x, &y);
        angle = atan2(y,x)*180/M_PI;
        if(angle<0) angle+=360;
        v.push_back(pair<long double,int> (angle,i));
    }
    sort(v.begin(),v.end());
    v.push_back(pair<long double,int> (v[0].first+360,v[0].second));

    for(i=1;i<n+1;i++){
        a = v[i].first - v[i-1].first;
        if(a<mn){
            mn = a;
            id=i;
        }
    }
    printf("%d %d\n", v[id].second, v[id-1].second);




    return 0;
}
