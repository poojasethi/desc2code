#include <bits/stdc++.h>
using namespace std;
long double PI = M_PI;
pair<long double, int> a[200010];  
int main(){
	int n;
	cin>>n;
	long double x, y;
	for (int i = 0; i < n; i++) {
		cin>>x>>y;
		long double tmp = atan2(x, y) * 180.0 / PI;
		if(tmp < 0) tmp += 360.0;
		a[i] = make_pair(tmp, i + 1);
	}
	sort(a, a + n);
	int ind1, ind2;
	long double mini = 2e18;
	for (int i = 0; i < n; i++) {
		long double diff = a[(i + 1) % n].first - a[i].first;
		if(diff < 0) diff += 360.0;
		if(diff <= mini) {
			mini = diff;
			ind1 = a[i].second;
			ind2 = a[(i + 1) % n].second;
		}
	}
	cout<<ind1<<" "<<ind2<<endl;
}
