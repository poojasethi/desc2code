#include <cmath>
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main(){
	int n,x,y;
	long double angle,da,minVal=100000;
	cin>>n;
	vector< pair<long double,int> > cases;
	for(int i=0;i<n;i++){
		cin>>x>>y;
		angle = atan2(y,x)*180/M_PI;
		if(angle<0) angle+=360;
		cases.push_back(pair<long double,int>(angle,i));
	}
	sort(cases.begin(),cases.end());
	cases.push_back(pair<long double,int>(cases[0].first+360,cases[0].second));
	int minInd=-1;
	for(int i=1;i<n+1;i++){
		da = abs(cases[i].first - cases[i-1].first);
		if(da<minVal){
			minVal = da;
			minInd = i;
		}
	}
	cout << cases[minInd].second + 1<<" "<<cases[minInd-1].second + 1<<endl;
	return 0;
}
