#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
#include <cmath>

using namespace std;

const int maxn = 1e5+10;
pair<long double, int> save[maxn];
int main() {
    int n;
    cin>>n;
    for(int i = 1; i <= n; i++) {
        int x,y;
        cin>>x>>y;
        save[i].first = atan2(y,x);
        save[i].second = i;
    }
    sort(save+1, save+1+n);
    save[n+1] = make_pair(save[1].first + 2 * acos(-1.0), save[1].second);
    long double ans = 2 * acos(-1.0);
    int id1,id2;
    for(int i = 1; i <= n; i++) {
        if(save[i+1].first - save[i].first < ans) {
            ans = save[i+1].first - save[i].first;
            id1 = save[i].second;
            id2 = save[i+1].second;
        }
    }
    cout<<id1<<" "<<id2<<endl;
    return 0;
}
