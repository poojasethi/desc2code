#include <math.h>
#include <stdio.h>
#include <algorithm>
using namespace std;
#define double long double
const double PI = acos(-1);
struct point {
    int id;
    double ang;
    bool operator <(const point& other)const {
        return ang < other.ang;
    }
} p[100005];
int main()
{
    int n,x,y;
    scanf("%d",&n);
    for (int i=1; i<=n; i++) {
        scanf("%d%d",&x,&y);
        p[i].ang = atan2(y, x);
        p[i].id = i;
    }
    sort(p+1, p+n+1);
    int ans1 = p[1].id, ans2 = p[n].id;
    double mi = p[1].ang + 2 * PI - p[n].ang;
    for (int i=1; i<n; i++) {
        double tmp = p[i+1].ang - p[i].ang;
        if (tmp < mi) {
            mi = tmp;
            ans1 = p[i].id, ans2 = p[i+1].id;
        }
    }
    printf("%d %d\n",ans1,ans2);
    return 0;
}
