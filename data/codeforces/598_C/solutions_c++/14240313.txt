#include <bits/stdc++.h>
using namespace std;
typedef long double LD;
#define N 100002
LD angle(LD l1, LD l2) {
	return min(l2 - l1, 2 * M_PI + l1 - l2);
}
int main() {
	int n, i;
	pair<LD, int> a[N];
	LD x, y, d, c;
	cin >> n;
	for(i = 0; i < n; i++) {
		cin >> x >> y;
		d = atan2(y, x);
		if(d < 0.0) d+= 2 * M_PI;
		a[i] = make_pair(d, i+1);
	}
	sort(a, a + n);
	int s = 0, e = n - 1;
	d = angle(a[0].first, a[n-1].first);
	for(i = 1; i < n; i++) {
		c = angle(a[i-1].first, a[i].first);
		if(c < d) {
			d = c;
			s = i-1;
			e = i;
		}
	}
	cout << a[s].second << " " << a[e].second << endl;
	return 0;
}
