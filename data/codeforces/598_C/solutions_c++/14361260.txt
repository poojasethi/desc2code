#include <cmath>
#include <cstdio>
#include <algorithm>

using namespace std;

#define N 100005

typedef long double LD;

const LD EPS = 1e-11;

int id[N];
LD ang[N];

bool cmp(int i, int j) { return ang[i] < ang[j]; }

main() {
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i ++) {
		id[i] = i;
		int x, y;
		scanf("%d %d", &x, &y);
		ang[i] = atan2(1.0 * y, 1.0 * x);
	}
	sort(id, id + n, cmp);
	LD rlt = 1e10;
	int x, y;
	for (int i = 0; i < n; i ++) {
		LD tmp = ang[id[(i + 1) % n]] - ang[id[i]];
		if (tmp < 0) tmp += 2 * acos(-1);
		if (rlt > tmp) rlt = tmp, x = id[(i + 1) % n], y = id[i];
	}
	printf("%d %d\n", x + 1, y + 1);
}
