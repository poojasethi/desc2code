#include <iostream>
#include <algorithm>
#include <cmath>
#include <vector>
using namespace std;

#define PI 3.14159265358979L

bool cmp(pair<long double,int> a, pair<long double,int> b){
	return a.first < b.first;
}
int main(){
	int n; cin >> n;
	vector<pair<long double, int> > v(0);
	int x,y;
	for (int i=1;i<=n;i++){
		cin >> x >> y;
		v.push_back(pair<long double,int>{atan2(y,x)*100,i}); 
	}
	sort(v.begin(),v.end(),cmp);
	int ans1, ans2; long double diff = 200*PI;
	for (int i=0;i<n-1;i++){
		if (v[i+1].first - v[i].first < diff){
			ans1 = v[i+1].second;
			ans2 = v[i].second;
			diff = v[i+1].first - v[i].first;
		}
	}
	if (200*PI + v[0].first - v[n-1].first < diff){
		ans1 = v[0].second;
		ans2 = v[n-1].second;
	}
	cout << ans1 << " " << ans2 << endl;
}
