#include <bits/stdc++.h>

using namespace std;


int main()
{

int n;
long double x , y , full_round = 2.0 * acos(-1.0) ;
vector< pair<long double,int> > v ;

cin>>n;
for(int i=1;i<=n;i++){

cin>>x>>y;
long double angle = atan2(x,y);

// cout<<angle<<endl;
 v.push_back(make_pair(angle,i));
}

sort(v.begin(),v.end());

long double ans=100.0;
int a , b ;

for(int i=1;i<n;i++){
    long double f= v[i].first - v[i-1].first;
   if(  f < ans   ){

   ans=f;
   a=v[i-1].second;
   b=v[i].second;

   }

}

    double f=  (v[0].first+full_round) - v[n-1].first;
   if(  f < ans   ){

   ans=f;
   b=v[n-1].second;
   a=v[0].second;

   }
cout<<a<<" "<<b<<endl;

    return 0;
}
