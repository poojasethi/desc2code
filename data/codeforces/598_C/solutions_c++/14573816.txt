#include<bits/stdc++.h>

using namespace std;
typedef long double ld;

int n, x, y;
pair<ld, int> p[100010];
ld Pi = acos((ld)-1.0);

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d", &n);
	for(int i=0 ; i<n ; ++i) {
		scanf("%d %d", &x, &y);
		p[i] = make_pair(atan2((ld)y, (ld)x), i);
	}
	sort(p, p+n);
	ld ans = 1e9;
	for(int i=0 ; i+1<n ; ++i) {
		if(fabs(p[i].first - p[i+1].first) < ans ){
			ans = fabs(p[i].first - p[i+1].first);
			x = p[i].second, y = p[i+1].second;
		}
	}
	if(p[0].first < 0) p[0].first += 2*Pi;
	if(p[n-1].first < 0) p[n-1].first += 2*Pi;
	if(fabs(p[0].first - p[n-1].first) < ans)
		x = p[0].second, y = p[n-1].second;
	printf("%d %d\n", x+1, y+1);
	return 0;
}
