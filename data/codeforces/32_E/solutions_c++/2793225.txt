#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<algorithm>
using namespace std;
int i,j,k,l,n,m,s,an,sum;
double pp=0.001;
double eps=1e-7;
class arr
{
    public:
        double x1,y1,x2,y2,k,b;
}a[5];
double XX,YY;
void zhuan(double &x,double &y,double pp)
{
    double   xx=x*cos(pp)-y*sin(pp);
    double   yy=x*sin(pp)+y*cos(pp);
    x=xx;y=yy;
}
bool jiao(arr a,arr b)
{
    if (fabs(a.k-b.k)<eps) return false;
    double X=(a.b-b.b)/(b.k-a.k);
    double Y=X*a.k+a.b;
    XX=X;
    YY=Y;
    if (((X>min(a.x1,a.x2)-eps)&&(X<max(a.x1,a.x2)+eps))
    &&((X>min(b.x1,b.x2)-eps)&&(X<max(b.x1,b.x2)+eps))
    &&((Y>min(a.y1,a.y2)-eps)&&(Y<max(a.y1,a.y2)+eps))
    &&((Y>min(b.y1,b.y2)-eps)&&(Y<max(b.y1,b.y2)+eps)))
    return true;
    else return false;
}
bool jiao2(arr a,arr b)
{
    if ((fabs(a.k-b.k)<eps)&&(fabs(a.b-b.b)<eps)) 
    {
        if (min(b.x1,b.x2)-eps<max(a.x1,a.x2)) return true;
        else return false;
    }
    return false;
}
int main()
{
    for (int i=1;i<=3;i++)
    {
        scanf("%lf%lf%lf%lf",&a[i].x1,&a[i].y1,&a[i].x2,&a[i].y2);
        zhuan(a[i].x1,a[i].y1,pp);
        zhuan(a[i].x2,a[i].y2,pp);
        a[i].k=(a[i].y1-a[i].y2)/(a[i].x1-a[i].x2);
        a[i].b=a[i].y1-a[i].x1*a[i].k;
    }
    if (!jiao2(a[1],a[2])&&(!jiao(a[1],a[2])&&!jiao(a[1],a[3]))) {printf("YES\n");return 0;}
    arr c,d;
    c.k=(-1.0/(a[3].k));c.b=a[1].y1-a[1].x1*c.k;
    d.k=a[3].k;d.b=a[3].k*a[1].x1+a[3].b*2-a[1].y1;
    double X=(c.b-d.b)/(d.k-c.k);
    double Y=X*c.k+c.b;
    arr t;
    t.x1=X;t.y1=Y;t.x2=a[1].x2;t.y2=a[1].y2;
    t.k=(Y-a[1].y2)/(X-a[1].x2);
    t.b=Y-X*t.k;
    if (!jiao(t,a[3])) {printf("NO\n");return 0;}
    t.x1=XX;t.y1=YY;
    t.x2=a[1].x1;t.y2=a[1].y1;
    t.k=(YY-a[1].y1)/(XX-a[1].x1);
    t.b=YY-XX*t.k;
    if (jiao(t,a[2])) {printf("NO\n");return 0;}
    t.x2=a[1].x2;t.y2=a[1].y2;
    t.k=(t.y1-a[1].y2)/(t.x1-a[1].x2);
    t.b=t.y1-t.x1*t.k;
    if (jiao(t,a[2])) {printf("NO\n");return 0;}
    printf("YES\n");
    return 0;
}
