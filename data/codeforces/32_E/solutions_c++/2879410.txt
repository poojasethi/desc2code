#include <cstdio>
#include <cmath>
#include <algorithm>
#define eps 1e-8
using namespace std;
struct Point
	{double x,y;}a,b,w1,w2,m1,m2;
int cross(Point u,Point v,Point o)
{
	double c=(u.x-o.x)*(v.y-o.y)-(u.y-o.y)*(v.x-o.x);
	if(abs(c)<=eps) return 0;
	return c>0?1:-1;
}
Point calc(Point u)
{
	double e,f,g,d;
	e = m2.y-m1.y; f = m1.x-m2.x;
	g = m2.x*m1.y-m1.x*m2.y;
	d = e*e+f*f;
	Point ans;
	ans.x = f*f*u.x-e*e*u.x-2*e*f*u.y-2*e*g;
	ans.x /= d;
	ans.y = e*e*u.y-f*f*u.y-2*e*f*u.x-2*f*g;
	ans.y /= d;
	return ans;
}
bool fail(Point a,Point b,Point p,Point q)
{
	bool ok=true;
	ok = ok&&(min(a.x,b.x)<=max(p.x,q.x)+eps);
	ok = ok&&(max(a.x,b.x)>=min(p.x,q.x)-eps);
	ok = ok&&(min(a.y,b.y)<=max(p.y,q.y)+eps);
	ok = ok&&(max(a.y,b.y)>=min(p.y,q.y)-eps);
	ok = ok&&((cross(b,p,a)*cross(b,q,a))<=0);
	ok = ok&&((cross(q,a,p)*cross(q,b,p))<=0);
	return ok;
}
void work()
{
	bool flag=true;
	if(fail(a,b,w1,w2) || fail(a,b,m1,m2))
	{
		flag = false;
		Point b1 = calc(b); 
		Point w11 = calc(w1); Point w22 = calc(w2);
		if(fail(a,b1,m1,m2) && !fail(a,b1,w1,w2) && !fail(a,b1,w11,w22))
			flag = true;
	}
	if(flag) printf("YES");
	else printf("NO");
}
int main()
{
	//freopen("input.txt","r",stdin);
	scanf("%lf%lf%lf%lf",&a.x,&a.y,&b.x,&b.y);
	scanf("%lf%lf%lf%lf",&w1.x,&w1.y,&w2.x,&w2.y);
	scanf("%lf%lf%lf%lf",&m1.x,&m1.y,&m2.x,&m2.y);
	if(cross(m1,m2,a) * cross(m1,m2,b) == 0)
	{
		if(fail(a,b,w1,w2)) printf("NO");
		else printf("YES");
	}
	else
		work();
}
